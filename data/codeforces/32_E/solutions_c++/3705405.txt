struct point 
{
    double x,y;
    point() {}
    point(double x,double y) : x(x),y(y) {}
};
point a,b,c,x1,y1,x2,y2;
double cross(point a,point b) {return a.x*b.y-a.y*b.x;}
point operator - (point a,point b) {return point(a.x-b.x,a.y-b.y);}
point operator + (point a,point b) {return point(a.x+b.x,a.y+b.y);}
point operator * (point a,double b) {return point(a.x*b,a.y*b);}
bool check1() 
{
    if((a.y-b.y)*(x2.x-y2.x)!=(a.x-b.x)*(x2.y-y2.y)) 
        if(cross(x2-b,a-b)*cross(y2-b,a-b)<=0&&
        cross(a-y2,x2-y2)*cross(b-y2,x2-y2)<=0) return false;
    if(cross(x1-b,a-b)*cross(y1-b,a-b)<=0&&
    cross(a-y1,x1-y1)*cross(b-y1,x1-y1)<=0) return false;
    return true;
}
bool check2()
{
    if(cross(a-y2,x2-y2)*cross(b-y2,x2-y2)<=0) return false;
    double k1,k2,b1,b2;
    int flag=0;
    if(x2.x==y2.x)
        flag=1,k2=0.0,b2=a.y;
    else if(x2.y==y2.y)
        flag=2,k1=0.0,b1=x2.y;
    else
    {
        k1=(y2.y-x2.y)/(y2.x-x2.x);
        k2=-(y2.x-x2.x)/(y2.y-x2.y);
        b1=x2.y-x2.x*k1;
        b2=a.y-a.x*k2;
    }
    if(flag==1)
        c=point(2*x2.x-a.x,a.y);
    else if(flag==2)
        c=point(a.x,2*x2.y-a.y);
    else
    {
        double aa,bb;
        aa=(b2-b1)/(k1-k2);
        bb=aa*k1+b1;
        c=point(2*aa-a.x,2*bb-a.y);
    }
    flag=0;
    point u=b-x2,j;
    double t=cross(y2-x2,u)/cross(c-b,y2-x2);
    j=b+(c-b)*t;
    if(j.x>max(x2.x,y2.x)||j.x<min(x2.x,y2.x)) return false;
    if(cross(x1-j,a-j)*cross(y1-j,a-j)<0&&
    cross(a-y1,x1-y1)*cross(j-y1,x1-y1)<0) return false;
    if(cross(x1-b,j-b)*cross(y1-b,j-b)<0&&
    cross(j-y1,x1-y1)*cross(b-y1,x1-y1)<0) return false;
    return true;
}
