#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

struct Point
{
  double X,Y;
  void Read()
    {
      cin>>X>>Y;
    }
  Point Reverse(Point V0,Point V1)
    {
      double A=V0.Y-V1.Y,B=V1.X-V0.X,C=V0.X*V1.Y-V1.X*V0.Y;
      Point O;
      O.X=-(A*A*X-B*B*X+2*A*B*Y+2*A*C)/(A*A+B*B);
      O.Y=-(B*B*Y-A*A*Y+2*A*B*X+2*B*C)/(A*A+B*B);
      return O;
    }
};

Point F(Point A,Point B,Point C,Point D)
{
  double A0=A.Y-B.Y,B0=B.X-A.X,C0=A.X*B.Y-B.X*A.Y;
  double A1=C.Y-D.Y,B1=D.X-C.X,C1=C.X*D.Y-D.X*C.Y;
  Point O;
  O.X=(B0*C1-B1*C0)/(A0*B1-A1*B0);
  O.Y=-(A0*C1-A1*C0)/(A0*B1-A1*B0);
  return O;
}

double G(Point A,Point B,Point C)
{
  return (A.X*B.Y+B.X*C.Y+C.X*A.Y)-(A.Y*B.X+B.Y*C.X+C.Y*A.X);
}

bool Cross(Point A,Point B,Point C,Point D)
{
  double A0=A.Y-B.Y,B0=B.X-A.X,C0=A.X*B.Y-B.X*A.Y;
  double A1=C.Y-D.Y,B1=D.X-C.X,C1=C.X*D.Y-D.X*C.Y;
  if(A0*B1==A1*B0)
    {
      if(B0*C1!=B1*C0)return false;
      double a,b,c,d;
      a=A.X;b=B.X;c=C.X;d=D.X;
      if(a>b)swap(a,b);if(c>d)swap(c,d);
      if(max(a,c)>min(b,d))return false;
      a=A.Y;b=B.Y;c=C.Y;d=D.Y;
      if(a>b)swap(a,b);if(c>d)swap(c,d);
      if(max(a,c)>min(b,d))return false;
      return true;
    }
  Point O;
  O.X=(B0*C1-B1*C0)/(A0*B1-A1*B0);
  O.Y=-(A0*C1-A1*C0)/(A0*B1-A1*B0);
  if((A.X-O.X)*(B.X-O.X)>0)return false;
  if((A.Y-O.Y)*(B.Y-O.Y)>0)return false;
  if((C.X-O.X)*(D.X-O.X)>0)return false;
  if((C.Y-O.Y)*(D.Y-O.Y)>0)return false;
  return true;
}

int main()
{
  Point A,B,W0,W1,M0,M1;
  A.Read();B.Read();
  W0.Read();W1.Read();
  M0.Read();M1.Read();
  if(!Cross(A,B,W0,W1)&&!Cross(A,B,M0,M1))
    {printf("YES\n");return 0;}
  if(!Cross(A,B,W0,W1)&&(A.X-B.X)*(M0.Y-M1.Y)==(A.Y-B.Y)*(M0.X-M1.X))
    {printf("YES\n");return 0;}
  if(G(M0,M1,A)*G(M0,M1,B)<0)
    {printf("NO\n");return 0;}
  Point _B=B.Reverse(M0,M1);
  if(!Cross(A,_B,M0,M1)){printf("NO\n");return 0;}
  Point O=F(A,_B,M0,M1);
  if(Cross(A,O,W0,W1)||Cross(B,O,W0,W1))
    printf("NO\n");
  else
    printf("YES\n");
  return 0;
}
