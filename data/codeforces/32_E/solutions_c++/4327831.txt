#include <iostream>
#include <cstdio>
#include <complex>

using namespace std;
typedef complex<double> C;
C p,v,m1,m2,w1,w2;

double dot(C a,C b)
{
	return real(a*conj(b));
}

double det(C a,C b)
{
	return imag(a*conj(b));
}
C sym(C x,C a,C b)
{
	return x+2*dot(a,b)/dot(b,b)*b-a;
}
bool in(C c,C a,C b)
{
	return det(a-c,b-c)==0 && dot(a-c,b-c)<=0;
}

bool ins(C a,C b,C c,C d)
{
	double t1=det(c-a,b-a)*det(b-a,d-a),t2=det(a-c,d-c)*det(d-c,b-c);
	if (t1<0 || t2<0)
		return 0;
	if (t1==0 && t2==0)
		return in(a,c,d) || in(b,c,d) || in(c,a,b);
	return 1;
}

bool check()
{
	if (ins(p,v,m1,m2))
		return det(p-v,m1-m2)==0 && !ins(p,v,w1,w2);
	if (ins(p,v,w1,w2))
	{
		C P=sym(m1,p-m1,m2-m1),V=sym(m1,v-m1,m2-m1);
		return ins(P,v,m1,m2) && !ins(P,v,w1,w2) && !ins(p,V,w1,w2);
	}
	return 1;
}

void scan(C &p)
{
	int x,y;
	scanf("%d%d",&x,&y);
	p=C(x,y);
}
int main()
{
//	freopen("1.txt","r",stdin);
	scan(p);
	scan(v);
	scan(w1);scan(w2);
	scan(m1);scan(m2);
	printf("%s\n",check()?"YES":"NO");
}
