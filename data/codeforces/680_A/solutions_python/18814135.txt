from collections import Counter

a = map(int, raw_input().split())
a_count = Counter(a)

max_count = 0

for n in a_count:
    if a_count[n] >= 2:
        n_max = 0
        if a_count[n] == 2:
            n_max = 2 * n
        else:
            n_max = 3 * n

        max_count = max(max_count, n_max)

print sum(a) - max_count
