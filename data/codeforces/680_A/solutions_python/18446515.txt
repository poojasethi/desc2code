def f(l,i):
	k = list(range(4))
	jj = 0
	for j in range(5):
		if j!=i:
			k[jj] = l[j]
			jj += 1
	return k
l = list(map(int,raw_input().split(' ' )))
m = sum(l)
for i in range(5):
	k = f(l,i)
	if l[i] in k:
		t=2
		while t and l[i] in k:
			t -=1
			k.pop(k.index(l[i]))
		if m>sum(k):
			m = sum(k)
print(m)
