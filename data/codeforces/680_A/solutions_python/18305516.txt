from sys import stdout
li = map(int, raw_input().split())
dic={}
for i in li:
    if i in dic:
        dic[i]+=1
    else:
        dic[i] = 1
ans = sum(li)
summ = sum(li)
for i in li:
    if(dic[i]>=3):
        ans = min(ans, summ-(i*3))
    elif(dic[i]==2):
        ans = min(ans, (summ - (i*2)))
print ans
