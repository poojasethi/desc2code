import sys

mark = [0]*101
lists = map(int,sys.stdin.readline().split()) 
ans = sum(lists)
for i in lists :
    mark[i] += 1
ma = 0
for i in range(100,0,-1) :
    if mark[i] and mark[i]>1 :
        if mark[i] < 4 :
            ma = max(ma,mark[i]*i)
        else :
            ma = max(ma,3*i)
print ans - ma
