a = list(map(int, raw_input().split()))
res = sum(a)
for i in range(5):
    for j in xrange(i + 1, 5):
        if a[i] == a[j]:
            res = min(res, sum(a) - 2 * a[i])
        for k in xrange(j + 1, 5):
            if (a[i] == a[j] == a[k]):
                res = min(res, sum(a) - 3 * a[i])
print res
