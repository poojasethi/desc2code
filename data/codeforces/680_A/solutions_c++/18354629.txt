#include <iostream>
using namespace std;
int a[5],s,c[101],ans;
int main()
{
	for(int i = 0;i<5;++i){
		cin>>a[i];
		s+=a[i];
		++c[a[i]];
	}

	ans = s;

	for(int i = 0;i<101;++i){
		if(c[i]>=3)
			ans = min(ans,s-3*i);
		else if(c[i]>=2)
			ans = min(ans,s-2*i);
		else
			continue;
	}

	cout<<ans<<endl;
}
