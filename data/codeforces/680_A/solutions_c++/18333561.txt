#include <bits/stdc++.h>
using namespace std;
int main()
{
int a[100],sum=0;
for(int i=0;i<5;i++)
{
cin>>a[i];
sum+=a[i];
}
sort(a,a+5);

int r = 1000;
for(int i=0;i<=4;i++)
{
if(a[i]==a[i+1])
{
if(i<3 && a[i]==a[i+2])
r = min(r,sum-3*a[i]);
else
{r = min(r,sum-2*a[i]);}
}
}
r=min(r,sum);
cout<<r<<endl;
}
