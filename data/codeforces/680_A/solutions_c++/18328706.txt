#include<bits/stdc++.h>
using namespace std;
int arr[100010];
int main()
{
    int i,k,mx=0,s=0;;
    for(i=0;i<5;i++)
    {
        cin>>k;
        arr[k]++;
        s+=k;
        if(arr[k]>=3){mx=max(mx,k*3);}
        else if(arr[k]==2){mx=max(mx,k*2);}
    }
    cout<<s-mx<<endl;
    return 0;
}
