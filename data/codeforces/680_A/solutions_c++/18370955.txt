#include <bits/stdc++.h>
using namespace std;

int sum=0,a[105];
int main()
{
    int x;
    for(int i=1;i<=5;i++)
    {
        cin>>x;
        a[x]++;
        sum+=x;
    }
    int res=sum;
    for(int i=1;i<=100;i++)
        if(a[i]>1)
            res=min(res,sum-min(a[i],3)*i);
    cout<<res;
}
