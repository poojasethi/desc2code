#include<cstdio>
#include<algorithm>
using namespace std; 
const int N=100;
int a[N+5],sum,ans,t;
int main(){
	for (int i=0;i<5;i++) scanf("%d",&t),a[t]++,sum+=t;
	for (int i=1;i<=100;i++){
		if (a[i]>=3) ans=max(ans,i*3);
		else if (a[i]>=2) ans=max(ans,i*2); 
	} 
	printf("%d",sum-ans); 
	return 0;
}
