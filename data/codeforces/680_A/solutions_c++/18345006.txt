#include <bits/stdc++.h>

using namespace std;
int cnt[105];
int main()
{
    int  sum=0, x , mx=-1;

    for (int i=0; i<5; i++)
    {
        cin>>x;
        sum+=x , cnt[x]++;

        if((cnt[x]*x > mx )&&(cnt[x]>1)&&(cnt[x]<4))
            mx=cnt[x]*x;

    }



    cout<<sum- ((mx>0)?mx:0);
}
