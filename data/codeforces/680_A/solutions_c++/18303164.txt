#include <iostream>
using namespace std;
int v[120],a;
int main()
{
    int S=0,smax=0;
    for(int i=1;i<=5;++i)
        cin>>a, v[a]++, S+=a;
    for(int i=1;i<=100;++i){
        if(v[i]>=2) smax=max(smax,2*i);
        if(v[i]>=3) smax=max(smax,3*i);
    }
    cout<<S-smax<<'\n';
    return 0;

}
