#include<cstdio>
#include<cstring>
using namespace std;
int max,num[150],tmp,sum;
int main()
{
	memset(num,0,sizeof(num));
	for(int i=0;i<5;i++){
		scanf("%d",&tmp);
		if(num[tmp]<tmp*3)num[tmp]+=tmp;
		sum+=tmp;
		if(num[tmp]>tmp&&num[tmp]>=max)
			max=num[tmp];
	}
	printf("%d",sum-max);
	return 0;
}
