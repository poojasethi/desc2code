#include <iostream>
#include <algorithm>
using namespace std;
int n[105];
int main(){
    int sum = 0, maxi = 0,a;
    for (int i = 0;i<5;i++){
        cin >> a;
        n[a]++;
        sum+=a;
        if (n[a]>=2){
            maxi = max(maxi, min(n[a],3)*a);
        }
    }
    cout << sum - maxi;
}
