#include <bits/stdc++.h>
using namespace std;

int l[105],sum;
int main(){
    int m=0;
    for(int i=0; i<5; ++i){
        int a;
        cin>>a;
        ++l[a];
        sum+=a;
    }
    for(int i=0; i<105; ++i){
        if(l[i]>1){
            m=max(m,min(3,l[i])*i);
        }
    }
    cout<<sum-m;
}
