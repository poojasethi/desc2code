#include<bits/stdc++.h>
using namespace std;
int main()
{
	int k,i,j,t,f=0;
	string a;
	cin>>a>>k;
	for(i=0;i<a.size();i++)
	{
		t=i;
		for(j=i+1;j<a.size() && j<=i+k;j++)
		{
			if(a[j] > a[t])
				t=j;
		}
		for(;t!=i;t--)
		{
			swap(a[t],a[t-1]);
			k--;
			if(!k)
			{
				f=1;
				break;
			}
		}
		if(f)
			break;
	}
	cout<<a<<endl;
	return 0;
}
