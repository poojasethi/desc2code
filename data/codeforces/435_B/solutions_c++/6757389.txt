#include <iostream>
using namespace std;

int main()
{
	string s;
	int k, i, j, n, d;
	char t;
	cin>>s>>k;
	n = s.size();
	for (i=0; i<n && k>0; i++) {
		for (d=i,t=s[i],j=i+1; j<=i+k && j<n; j++) if (s[j]>t) {
			d=j; t=s[j];
		}
		for (j=d; j>i; j--) {
			t=s[j]; s[j]=s[j-1]; s[j-1]=t; k--;
		}
	}
	cout<<s<<endl;
	return 0;
}
