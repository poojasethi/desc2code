#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	int k;
	cin>>s;
	cin>>k;
	int i;
	int j;
	for(i=0;i+1<s.length();i++)
	{
		int t=i;
		for(j=i+1;j<s.length()&&j-i<=k;j++)
		{
			if(s[j]>s[t])
			t=j;
		}
		k-=t-i;
		while(t!=i)
		{
			swap(s[t],s[t-1]);
			t--;
		}
	}
	cout<<s<<endl;
	return 0;
}
