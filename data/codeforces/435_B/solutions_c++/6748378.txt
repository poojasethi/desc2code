#include <bits/stdc++.h>
using namespace std;
int main()
{
	int k;
	string s;
	cin>> s>> k;
	int j = 0, n = s.length();
	while (k > 0 && j < n)
	{
		int ind = 0;
		for (int i = 1; i <= min(k, n - j - 1); i++)
			if (s[i + j] > s[ind + j])
				ind = i;
		if (ind)
		{
			for (int i = ind; i > 0; i--)
				swap(s[i + j], s[i + j - 1]);
			k -= ind;
		}
		j++;
	}
	cout<<s<<endl;
}
