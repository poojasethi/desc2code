#include <bits/stdc++.h>
using namespace std;
string s;
int k;
int main()
{
	cin>>s>>k;
	int SIZ = (int) s.size();
	for(int i=0;i<SIZ && k>0;i++)
	{
		int cur = i;
		for(int j=i+1;j<min(SIZ,i+k+1);j++)
		{
			if(s[j]>s[cur]) cur = j;
		}
		for(int j=cur;j>i;j--)
		{
			k--;
			swap(s[j],s[j-1]);
		}
	}
	puts(s.c_str());
}
