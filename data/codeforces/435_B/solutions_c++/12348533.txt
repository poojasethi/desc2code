#include<bits/stdc++.h>
using namespace std;

int main() {
	string s; cin>>s;
	int k,n=s.size(); cin>>k;
	for (int i=0; i<n; i++) {
		if(k==0) break;
		int x=i;
		for (int j=i+1; j<n&&j-i<=k; j++)
			if(s[j]>s[x]) x=j;

		for (int j=x; j-1>=i; j--)
			swap(s[j], s[j-1]);
		k-=x-i;
	}
	cout << s << endl;
	return 0;
}
