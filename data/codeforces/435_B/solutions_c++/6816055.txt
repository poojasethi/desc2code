#include<iostream>
#include<string>
using namespace std;
int main()
{
    string s;
    int k;
    cin>>s>>k;
    for(int i=0;i<s.size();i++)
    {
        int t=i;
        for(int j=i+1;j<s.size()&&(j-i)<=k;j++)
            if(s[j]>s[t]) t=j;
        k-=(t-i);
        while(t!=i)
            swap(s[t],s[t-1]),t--;
    }
    cout<<s<<endl;
    return 0;
}
