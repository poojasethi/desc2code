#include<stdio.h>
char str[20];
int k;
int main(void){
	scanf("%s%d",str,&k);
	for(int i=0;str[i]&&k;i++){
		char t=str[i];
		int temp=i;
		for(int j=1;str[i+j]&&j<=k;j++)
			if(str[i+j]>t){
				temp=i+j;
				t=str[temp];
			}
		for(int j=temp;j>i;j--){
			char tt=str[j];
			str[j]=str[j-1];
			str[j-1]=tt;
		}
		k-=temp-i;
	}
	printf("%s\n",str);
	
	return 0;
}
