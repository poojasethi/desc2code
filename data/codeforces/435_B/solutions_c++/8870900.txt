#include<bits/stdc++.h>
using namespace std;
long long n,k,ar[100005],i,j,id;
string s;
int main()
{
   cin>>s>>k;
   n=s.size();
   for(i=0;i<n;i++)
   {
    id=i;
       for(j=i+1;j<n&&j-i<=k;j++)
       {
           if(s[j]>s[id])
           id=j;
       }
       for(j=id;j>i;j--)
       swap(s[j],s[j-1]);
       k-=(id-i);
   }
   cout<<s<<endl;
}
