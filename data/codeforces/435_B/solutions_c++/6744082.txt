#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char a[100];
int k, m;
int main()
{
	scanf("%s%d", a+1, &m);
	int l = strlen(a+1);
	for(int i = 1; i <= l; i++)
	{
		k = i;
		for(int j = i+1; j <= l && j <= i+m; j++)
			if(a[j] > a[k])
				k = j;
		while(k > i)
		{
			swap(a[k], a[k-1]);
			m--;
			k--;
		}
	}
	printf("%s\n", a+1);
	return 0;
}
