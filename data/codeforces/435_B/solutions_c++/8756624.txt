#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	char a[20];int k,pos,max;
	scanf("%s %d",a,&k);
//cout<<k<<a[0]<<endl;
max=a[0];
	for(int i=0;a[i]!='\0' && k!=0;i++){max=a[i];pos=i;
		for(int j=i+1;j-i<=k && a[j]!='\0';j++)
			if(max<a[j])
			{max=a[j];pos=j;}
		
for(int j=pos;j>i;j--)
		a[j]=a[j-1];
a[i]=max;
		k-=pos-i;}
	cout<<a<<"\n";
}
