#include<bits/stdc++.h>
using namespace std;
char s[20];
int k,len,mak,x;
int main() {
	scanf("%s %d",&s,&k);
	len = strlen(s);
	for(int i = 0; i < len - 1; i++) {
		x = i;
		for(int j = i + 1; j < len; j++)
			if (s[j] > s[x] && (j - i) <= k) x = j;
		k -= (x - i);
		for(int j = x; j > i; j--) swap(s[j], s[j - 1]);
	}
	printf("%s\n",s);
}
