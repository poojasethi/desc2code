#include <bits/stdc++.h>
using namespace std;
string a;
long long k,i,j,t;
int main()
{
	cin >> a >> k;
	for(i=0;i+1<a.size();i++)
    {
        t=i;
        for(j=i+1;j<a.size()&&j-i<=k;j++)
        {
            if(a[j]>a[t]) t=j;
        }
        k-=t-i;
		while(t!=i)
        {
			swap(a[t],a[t-1]);
			t--;
		}
	}
	cout << a;
}
