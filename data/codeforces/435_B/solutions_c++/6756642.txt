#include <cstdio>
#include <algorithm>
#include <string.h>
char s[20];
int main(){
    int l,in,k;
   scanf("%s %d",s,&k);
   l=strlen(s);
   for(int i=0;i<l&&k>0;i++){
        in=i;
     for(int j=i;j<l;j++)
         if(s[in]<s[j]&&j-i<=k)
             in=j;
     for(int j=in;j>i;j--)
        std::swap(s[j],s[j-1]);
     k-=in-i;
   }
   printf("%s",s);
   return 0;
}
