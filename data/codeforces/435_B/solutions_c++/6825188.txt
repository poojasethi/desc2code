#include <iostream>
using namespace std;

string num;

int main() {
	int m;
	cin >> num >> m;
	for(int i = 0;i < num.size();i ++) {
		int maxf = i;
		for(int j = i + 1;j <= min(i + m,(int)num.size() - 1);j ++) {
			if(num[j] > num[maxf]) {
				maxf = j;
				
			}
		}
		for(int j = maxf;j >= i + 1;j --) {
			swap(num[j - 1],num[j]);
			m --;
		}
	}
	cout << num << endl;
}
