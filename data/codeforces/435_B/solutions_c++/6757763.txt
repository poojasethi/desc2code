#include <iostream>

using namespace std;

int main()
{
   string s;
   int k;
   cin>> s >> k;

   for(int i=0;i<s.size();i++) {
    int n = i;
    for(int j=i;j<s.size();j++)
        if(s[j]>s[n]&&j-i<=k)
            n = j;

    for(int h = n; h > i; h--)
            swap(s[h],s[h-1]);

   k -= n-i;
   }
   cout<<s;

}
