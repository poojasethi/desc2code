#include <bits/stdc++.h>
#define sz(a) ((int) (a).size())
using namespace std;

int main(){
	//freopen("input.inp","r",stdin);
	string a;
	int k;
	cin >> a >> k;
	for(int i=0;i+1<sz(a);i++){
		int t=i;
		for(int j=i+1;j<sz(a)&&j-i<=k;j++){
			if(a[j]>a[t]) t=j;
		}
		k-=t-i;
		while(t!=i){
			swap(a[t],a[t-1]);
			t--;
		}
	}
	cout << a;
}
