# include <iostream>
using namespace std;

int main()
{
int k;
string a;
cin>>a>>k;
for(int i=0 ; i<a.size();i++)
{
    int temp = i;

    for( int j = i+1 ;j<a.size()&& j-i<=k ;j++)
        if(a[j] > a[temp] )
             temp = j;

for(int y = temp ; y>i ;y--)
    swap(a[y] , a[y-1]);

k -=temp-i;

}
cout<<a<<endl;
return 0;
}
