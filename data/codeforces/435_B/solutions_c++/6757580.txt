#include<iostream>
using namespace std;

int main(){
    string s;
    int k;
    cin>>s>>k;
    for(int i=0;i<s.size();i++){
        int oni=i;
        for(int j=i+1;j<s.size()&&j-i<=k;j++)
            if(s[j]>s[oni])oni=j;
        k-=(oni-i);
        for(;oni!=i;oni--)
            swap(s[oni],s[oni-1]);

    }
    cout<<s<<endl;
    return 0;
}
