#include <iostream>
using namespace std;
int main()
{string j;
cin>>j;
int n=j.length(),d[n],i,k,h,m,b;
for(i=0;i<n;i++)d[i]=j[i]-'0';
cin>>k;
i=0;
while(k&&i<n-1)
{m=d[i];
for(h=i+1;h<=i+k&&h<n;h++)
{
if(d[h]>m){m=d[h];b=h-i;}
}
if(m>d[i])
{
k-=b;
for(h=i+b;h>i;h--)swap(d[h],d[h-1]);
}
i++;
}
for(i=0;i<n;i++)cout<<d[i];
}
