#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
  string s;
  cin>>s;
  int k;
  cin>>k;
  for(int i=0;i<s.length();i++)
    {
      int m=i;
      for(int j=i;j<s.length();j++)
	if(s[j]>s[m] && j-i<=k)
	  {
	    m=j;
	  }
      for(int j=m;j>i;j--)
	swap(s[j],s[j-1]);
      k-=m-i;
    }
  cout<<s<<endl;
  return 0;
}
