#include <iostream>

using namespace std;

int main()
{string s; int k; cin>>s>>k;
int n=s.size ();
for (int i=0; i<n; i++)
  {if (k==0) break ;
  int i0=i;
  for (int j=i+1; j<n&& j-i<=k; j++)
    if (s[j]>s[i0]) i0=j;
   if (i0==i) continue ;
   for ( int j=i0; j>i; j--) swap (s[j], s[j-1]);
    k-= (i0-i);
}
cout<<s;
return 0;
}
