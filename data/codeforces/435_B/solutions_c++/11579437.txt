#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
string n;
int k,c=0;
cin>>n>>k;

for(int x=0; x<n.size()-1; x++){
    int mi=x;
    for(int z=x+1; z<n.size() && z-x<=k; z++)
        if(n[z]>n[mi]) mi=z;

    if(n[mi]>n[x]) {k-=(mi-x);
    for(int i=mi; i>x; i--)
        swap(n[i],n[i-1]);}

}

   cout<<n<<endl;
    return 0;
}
