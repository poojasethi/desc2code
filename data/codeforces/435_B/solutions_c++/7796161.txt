#include<iostream>

using namespace std ;

string s ;
int n , k , t ;

int main()
{
	cin >> s >> n ;
	for(int i=0 ; i+1<s.size() ; i++)
	{
		t=i ;
		for(int j=i+1 ;j<s.size() && j-i<=n ; j++)
		{
			if(s[j]>s[t])
				t=j ;
		}
		n-=t-i ;
		while(t!=i)
		{
			swap(s[t],s[t-1]) ;
			t-- ;
		}
	}
	cout << s << endl ;
	return 0 ;
}
