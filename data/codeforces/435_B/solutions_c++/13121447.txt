#include<bits/stdc++.h>
using namespace std;

int main(){
	int i,j,k,a,t,sw,l;
	string s;
	cin>>s>>k;
	l=s.length();
	for(i=0;i<l&&k>0;i++){
		a=s[i];
		t=i;
		for(j=i+1;j<=min(i+k,l-1);j++){
			if(s[j]>a){
				a=s[j];
				t=j;
			}
		}
		if(i!=t){
			sw=t-i;
			k-=sw;
			while(sw--){
				swap(s[t],s[t-1]);
				t--;
			}
		}
	}
	cout<<s<<endl;
}
