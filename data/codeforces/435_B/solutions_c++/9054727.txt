#include<stdio.h>
#include<string.h>
int main()
{
	char a[1000],max;
	int max1,i,t,j,len,n;
	scanf("%s%d",a,&n);
	len=strlen(a);
	for(i=0;n>0&&i<len;i++)
	{
		for(max1=i,max=a[i],t=i+1;t<len&&t-i<=n;t++)
		if(a[t]>max)
		{
			max=a[t];max1=t;
		}
		for(j=max1;j>i;j--)
		a[j]=a[j-1];
		
		a[i]=max;
		n-=max1-i;
	}
	printf("%s\n",a);
}
