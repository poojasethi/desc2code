#include<iostream>

#include<string>


using namespace std;
int k,p;
string a;

int main()
{
	cin>>a>>k;

	for(int i=0;i<a.length();i++)
	{
		p=i;
		for(int j=i+1;j<a.length();j++)
			if(a[p]<a[j] && j-i<=k)
				p=j;
		for(int b=p;b>i;b--)
				swap(a[b],a[b-1]);
		k-=p-i;
	}
	cout<<a;
	//getch();
	return 0;
    
}
