#include<bits/stdc++.h>
using namespace std;

int main()
{
   long long n,k,i,j,id;
   string ch;
   cin>>ch>>k;
   n=ch.size();
   for(i=0;i<n;i++)
   {
    id=i;
       for(j=i+1;j<n&&j-i<=k;j++)
       {
           if(ch[j]>ch[id])
           id=j;
       }
       for(j=id;j>i;j--)
       swap(ch[j],ch[j-1]);
       k-=(id-i);
   }
   cout<<ch<<endl;
}
