#include<bits/stdc++.h>
using namespace std;
int main()
{
    string a;
    int k;
    cin>>a>>k;
    for(int i=0;i<a.size();i++)
    {
        int n=i;
        for(int j=i+1;j<a.size()&&j<=i+k;j++)
        if(a[j]>a[n])
        n=j;
        for(int j=n;j>i;j--)
        swap(a[j],a[j-1]),k--;
    }
    cout<<a<<endl;
    return 0;
}
