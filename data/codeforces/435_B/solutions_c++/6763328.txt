#include<iostream>
using namespace std;
string s;
int k;
int main()
{
    cin>>s>>k;
    for (int i=0;i<s.length();i++) {
        int n=i;
        for (int j=i+1;j<s.length()&&j-i<=k;j++) if (s[j]>s[n]) n=j;
        for (int h=n;h>i;h--) swap(s[h],s[h-1]);
        k-=n-i;
    }
    cout<<s;
    return 0;
}
