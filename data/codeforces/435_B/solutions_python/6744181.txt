if __name__ == '__main__':
    a, k = map(int, raw_input().split())
    a = map(int, str(a))
    for i in xrange(len(a)):
        mx = a[i]
        mx_idx = i
        for t in xrange(1, min(k, len(a) - i) + 1):
            j = i + t
            if j >= len(a):
                continue
            if mx < a[j]:
                mx = a[j]
                mx_idx = j
        for j in xrange(mx_idx, i, -1):
            a[j], a[j - 1] = a[j - 1], a[j]
            k -= 1
    print ''.join(map(str, a))
