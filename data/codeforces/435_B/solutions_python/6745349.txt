

def new_number(s, k):
	slen = len(s)
	if (slen < 2) or (k <= 0):
		return s
	else:
		m = int(s[0])
		indm = 0
		b = min(len(s), k + 1)
		for i in xrange(1, b):
			if int(s[i]) > m:
				m = int(s[i])
				indm = i
		tmps = s[: indm] + s[indm + 1:]
		tmpk = k - indm
		return s[indm] + new_number(tmps, tmpk)
		
a, k = raw_input().split()
k = int(k)
print new_number(a, k)
