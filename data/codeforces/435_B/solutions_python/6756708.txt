a, k = raw_input().strip().split()
k = int(k)

for i in xrange(len(a)):
    c = a[i]
    if k == 0: break
    try:
        if max(a[i+1:i+1+k]) > c:
            j = a[i:i+1+k].index(max(a[i+1:i+1+k]))
            k -= j
            a = a[:i] + a[i+j] + a[i:i+j] + a[i+j+1:]
    except: break

print a
