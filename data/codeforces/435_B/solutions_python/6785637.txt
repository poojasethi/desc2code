s,t=raw_input().split()
s=list(s)
t=int(t)
iLen = len(s)
index=0
while index<iLen and t>0:
    if s[index]=='9':
        index+=1
        continue
    iMaxIndex=index
    for i in xrange(index,index+t+1):
        if i >= iLen:
            break
        if int(s[iMaxIndex]) < int(s[i]):
            iMaxIndex = i
    tmp = s[iMaxIndex]
    for i in xrange(iMaxIndex,index,-1):
        s[i]=s[i-1]
    s[index] = tmp
    t-=(iMaxIndex - index)
    index+=1
print ''.join(s)
