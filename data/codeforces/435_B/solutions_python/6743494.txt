s, k = raw_input().split()
S = [s[i] for i in range(len(s))]
k = int(k)
z = 0
res = ''
while z < len(S) and k > 0:
    i = z
    r = S[z]
    x = z
    while i < len(S) and i - z <= k:
        if ord(S[i]) > ord(r):
            r = S[i]
            x = i
        i += 1
    k -= x - z
    for i in range(x - 1, z - 1, -1):
        S[i + 1] = S[i]
    S[z] = r
    z += 1
for i in S:
    res += i
print res
