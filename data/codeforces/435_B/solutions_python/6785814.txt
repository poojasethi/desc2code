import operator

a,k = raw_input().split()

k = int(k)

i = 0
while k>=1 and i<len(a):
	sublist = a[i:min(i+k+1,len(a))]
	index, max_value = max(enumerate(sublist), key=operator.itemgetter(1))
	max_index = index + i
	k -= max_index - i
	a = a[:i] + max_value + a[i:max_index] + a[max_index+1:]
	i += 1

print ''.join(a)
