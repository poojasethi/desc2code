from sys import stdin

inp = stdin.readlines()

(x,k) = tuple([int(a) for a in inp[0].split()])

d = []
while x:
	d.append(x%10)
	x /=10

d.reverse()

#print (d,k)


	

digit = 0
#print (d,k)
while k :
	ts = sorted([(-v,(p)) for (p,v) in enumerate(d[digit:digit+1+k])])[0]
	#print ts
	del d[digit+ts[1]]
	d.insert(digit,-ts[0])
	k -= ts[1]
	digit +=1
	#print (d,k)
	if digit >= len(d):
		break

print "".join([str(x) for x in d])
