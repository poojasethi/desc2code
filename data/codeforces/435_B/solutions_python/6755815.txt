a=raw_input()
a=a.split()
n=int(a[1])
s=a[0]
k=0
maxy=0
while(n!=0 and k<len(s)):
    maxy=int(s[k])  
    pos=k
    
    for i in range(k,k+n+1):
        if(i<len(s)):
            if(int(s[i])>maxy):
                maxy=int(s[i])
                pos=i
    if pos>k:
        s=s[0:k]+s[pos]+s[k:pos]+s[pos+1:]
    n=n-pos+k
    k=k+1
print int(s)
