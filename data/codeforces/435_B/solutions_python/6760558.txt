s,m = raw_input().split()
s = list(s)
m = int(m)
#print s , m
length = len(s)
for st in range(length):
	if not m: break
	ed = min(st+m , length-1)
	index = st
	Max = ord(s[st])
	for i in range(st,ed+1):
		if ord(s[i]) > Max:
			Max = ord(s[i])
			index = i
	m -= index-st
	while index != st:
		s[index-1] , s[index] = s[index] , s[index-1]
		index -= 1
ans = s[0]
for i  in range(1,length):
	ans += s[i]
print ans
