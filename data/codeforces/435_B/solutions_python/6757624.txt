a, k = raw_input().split()
k = int(k)
a = list(a)
for index, value in enumerate(a):
    pos = index
    maxVal = value
    for j in xrange(index+1, index + k + 1):
        if len(a) > j:
            val = a[j]
            if maxVal < val:
                maxVal = val
                pos = j
    if pos != index:
        for i in xrange(pos, index, -1):
            a[i], a[i-1] = a[i-1], a[i]
        k = k - (pos - index)
        if k <= 0:
            break
print "".join(a)
