[a,k] = [x for x in str(raw_input()).split(" ")]
k = int(k)
l = len(a)
soln=[]
while k>0 and len(a)>0:
	s = max(a[0 : min(k,l-1)+1])
	i = a.index(s)
	soln.append(s)
	k -= i
	a = a[0:i] + a[i+1:]
