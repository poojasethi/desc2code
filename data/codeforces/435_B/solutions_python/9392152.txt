s,k=raw_input().split()
k=int(k)
arr=map(int,list(s))
for i in xrange(len(arr)):
    pos=0
    m=arr[i]
    for j in xrange(i+1,min(len(arr),i+k+1)):
        if arr[j]>m:
            pos=j
            m=arr[j]
    if m==arr[i]:
        continue
    arr=arr[:i]+[arr[pos]]+arr[i:pos]+arr[pos+1:]
    k-=pos-i
    if k==0:
        break
temp=""
for x in arr:
    temp+=str(x)
print temp
