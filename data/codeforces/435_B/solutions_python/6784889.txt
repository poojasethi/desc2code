a, b = raw_input().split()
a = list(a)
b = int(b)
n = len(a)
cur = 0
while b and cur < n:
    p = (-1, -1)
    for i in xrange(cur, n):
        if i - cur <= b:
            p = max(p, (a[i], 100 - i))
    del a[100 - p[1]]
    a = a[:cur] + [p[0]] + a[cur:]
    b -= (100 - p[1]) - cur
    cur += 1
print ''.join(a)
