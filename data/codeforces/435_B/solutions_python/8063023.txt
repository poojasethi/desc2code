import sys

data = next(sys.stdin).split(" ")
number = data[0]
swaps = int(data[1])

#5 lines like I said!
def solve(number, swaps):
    if not number:
        return ""
    max_index = number.find(max(number[:swaps+1]))
    return number[max_index] + solve(number[:max_index] + number[max_index+1:], swaps - max_index)

print(solve(number, swaps))
