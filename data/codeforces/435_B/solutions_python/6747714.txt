def best(a, k):
    a = list(str(a))
    n = len(a)
    for i in range(n):
        ma = a[i]
        mj = 0
        for j in range(1, k + 1):
            if i + j >= n:
                break
            if a[i + j] > ma:
                ma = a[i + j]
                mj = j
        a = a[:i] + [a[i+mj]] + a[i:i+mj] + a[i+mj+1:]
        k -= mj
    return ''.join(a)
            
a, k = map(int, raw_input().split())
print best(a, k)
