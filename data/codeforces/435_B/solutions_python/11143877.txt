a,k = raw_input().split()
k=int(k)
a=map(int,list(a))
# p=[]s
for i in range(len(a)):
    index=-1
    m=a[i]
    for j in range(i+1,min(i+1+k,len(a))):
        if a[j]>m:
            m=a[j]
            index=j
    # print i,m
    if index==-1 or m==a[i]:
        # print i
        continue
    # k-=1
    for j in range(index,i,-1):
        a[j]=a[j-1]
        k-=1
    a[i]=m
    if k==0:
        break
    # print "k is",k
    # print i,a,m,index
a=map(str,a)
print ''.join(a)
