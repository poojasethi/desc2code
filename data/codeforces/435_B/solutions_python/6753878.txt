import math
import itertools
#10**9+7
mod=1000000007
#start = time.clock()
n1,k=map(str,raw_input().split())
k=int(k)
ans=chk=0
n=[int(x) for x in n1]
l=len(n)


#先頭から交換可能範囲により大きい数がないかチェック
for i in range(l-1):
    if k==chk:
        break
    en=k-chk
    h=min(l+1,i+en+1)
    s=max(n[i+1:h])
    if n[i]<s:
        d=n[i+1:h].index(max(n[i+1:h]))
        chk+=(d+1)
        n.pop(d+i+1)
        n.insert(i,s)
        
#        print chk,p,n
print ''.join(map(str,n))
