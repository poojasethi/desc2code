a, k = raw_input().strip().split()
k = int(k)

def findMax(s, fro, mx):
    to = fro + min(mx+1, len(s)-fro)
    M, idx = s[fro], fro
    for i in xrange(fro+1, to):
        if s[i] > M:
            M, idx = s[i], i
    return idx

i = 0
while k > 0 and i < len(a):
    j = findMax(a, i, k)
    k -= (j-i)
    a = a[:i] + a[j] + a[i:j]+ a[j+1:]
    i += 1
print a
