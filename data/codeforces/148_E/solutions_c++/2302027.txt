#include<stdio.h>
#include<string.h>
const int maxn=10005;
int num[120],sum[120],dp[maxn],l,n,m;
int Max(int a,int b){
    return a>b?a:b; }
int main()
{
    int i,j,k;
    scanf("%d%d",&n,&m);
    memset(dp,0,sizeof(dp));
    for(i=0;i<n;i++)
    {
        scanf("%d",&l);
        for(j=1;j<=l;j++)
        {
            scanf("%d",&num[j]);
            num[j]+=num[j-1];
        }
        memset(sum,0,sizeof(sum));
        for(j=0;j<l;j++)
            for(k=j;k<=l;k++)
                sum[l-k+j]=Max(sum[l-k+j],num[l]-num[k]+num[j]);
        for(j=m;j>0;j--)
            for(k=j<l?j:l;k>0;k--)
                dp[j]=Max(dp[j],dp[j-k]+sum[k]);
    }
    printf("%d\n",dp[m]);
    return 0;
}
