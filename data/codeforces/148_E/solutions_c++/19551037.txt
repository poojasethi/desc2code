#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<cstring>
#include<algorithm>
using namespace std;

int n,m,i,j,k;
int a[105],b[105][105],p[105][105],q[105][105],d[105][105],dd[10005];

int main(){
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++){
	scanf("%d",&a[i]);
    for(j=1;j<=a[i];j++)
    scanf("%d",&b[i][j]);
}
    for(i=0;i<n;i++){
    for(j=1;j<=a[i];j++)p[i][j]=p[i][j-1]+b[i][j];
    for(j=a[i];j>0;j--)q[i][j]=q[i][j+1]+b[i][j];
    for(j=1;j<=a[i];j++)
    for(k=0;k<=j;k++)d[i][j]=max(d[i][j],p[i][k]+q[i][a[i]-j+k+1]);
    for(j=m;j>=1;j--)
    for(k=1;k<=a[i];k++)if(j>=k)dd[j]=max(dd[j],dd[j-k]+d[i][k]);
}
    printf("%d\n",dd[m]);
	return 0;
}
