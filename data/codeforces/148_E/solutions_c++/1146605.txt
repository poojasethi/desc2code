#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int shelf[101][101],sum[101][101];
int best[101][101];
int dp[101][10001];
int num[101];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",num+i);
		for(int j=1;j<=num[i];j++)
		{
			scanf("%d",&shelf[i][j]);
			sum[i][j]=sum[i][j-1]+shelf[i][j];
		}
	}
	for (int i=1;i<=n;i++)
		for(int j=1;j<=num[i];j++)
			for(int k=0;k<=j;k++)
			{
				best[i][j]=max(best[i][j],sum[i][k]+sum[i][num[i]]-sum[i][num[i]-j+k]);
			}
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			for(int k=0;k<=num[i]&&k<=j;k++)
			{
				dp[i][j]=max(dp[i][j],dp[i-1][j-k]+best[i][k]);
			}
	printf("%d\n",dp[n][m]);
	return 0;
}
