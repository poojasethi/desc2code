#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
#define REP(i,n) for(int(i)=0;(i)<(int)(n);(i)++)
const int MAX=(int)1e4+5;
int n,m,a[MAX],dp[MAX],val[MAX];
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1; i<=n; i++) {
		int c;
		scanf("%d",&c);
		for (int j=1; j<=c; j++) scanf("%d",a+j);
		for (int j=1; j<=c; j++) a[j]+=a[j-1];
		for (int j=1; j<=c; j++)
			for (int k=val[j]=0; k<=j; k++)
				val[j]=max(val[j],a[k]+a[c]-a[c-j+k]);
		for (int j=m; j>=0; j--)
			for (int k=1; k<=c&&k<=j; k++)
				dp[j]=max(dp[j],dp[j-k]+val[k]);
	}
	printf("%d\n",dp[m]);
	return 0;
}
