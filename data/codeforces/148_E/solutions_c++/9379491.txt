#include <iostream>
#include <algorithm>

using namespace std;
int best[10005],dp[105],i,j,k,a[105],n,m,nr;
int main()
{
    cin>>n>>m;
    for(i=1;i<=n;i++)
    {
        cin>>nr;
        for(j=1;j<=nr;j++)
        {
            cin>>a[j];
            dp[j]=0;
            a[j]+=a[j-1];
        }
        for(j=1;j<=nr;j++)
        {
            for(k=0;k<=j;k++)
            {
                dp[j]=max(dp[j],a[k]+a[nr]-a[nr-(j-k)]);
            }
        }
        for(j=m;j>=1;j--)
        {
            for(k=1;k<=min(j,nr);k++)
            {
                best[j]=max(best[j],dp[k]+best[j-k]);
            }
        }
    }
    cout<<best[m];
    return 0;
}
