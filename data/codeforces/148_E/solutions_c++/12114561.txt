#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

int s[101];
int f[101],dp[2][10001];
int main(){
    int n,k,m,i,j,l,v;
    scanf("%d%d",&n,&k);
    for(i=1;i<=n;i++)
    {   scanf("%d",&m);
        memset(s,0,sizeof(s));
        for(j=1;j<=m;j++)
        {   scanf("%d",&v);
            s[j]=s[j-1]+v;
        }
        memset(f,0,sizeof(f));
        for(j=1;j<=m;j++)
            for(l=0;l<=j;l++)
                f[j]=max(f[j],s[l]+s[m]-s[m-(j-l)]);        
        for(j=1;j<=k;j++)
            for(l=0;l<=m && l<=k && l<=j;l++)
                dp[i&1][j]=max(dp[i&1][j],dp[(i-1)&1][j-l]+f[l]);
    } 
    printf("%d\n",dp[n&1][k]);
    //system("pause");
    return 0;
}
