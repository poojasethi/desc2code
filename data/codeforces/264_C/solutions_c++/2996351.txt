#include <stdio.h>
#include <iostream>
#include <string.h>
using namespace std;
const int N=100010;
typedef long long LL;
int n,q;
int v[N],c[N];
LL dp[N];
LL query(LL A,LL B){
	memset(dp,0x80,sizeof(dp));
	int p1=0,p2=0;
	dp[0]=0;
	LL ret=0;
	for(int i=0;i<n;i++){
		int p=c[i]!=p1?p1:p2;
		dp[c[i]]=max(dp[c[i]],dp[c[i]]+A*v[i]);
		dp[c[i]]=max(dp[c[i]],dp[p]+B*v[i]);
		if(c[i]!=p1 && c[i]!=p2){
			if(dp[c[i]]>=dp[p1]) p2=p1,p1=c[i];
			else if(dp[c[i]]>=dp[p2]) p2=c[i];
		}else {
			if(dp[p1]<dp[p2]) swap(p1,p2);
		}
		ret=max(ret,dp[c[i]]);
	}
	return ret;
}
int main() {
	scanf("%d%d",&n,&q);
	for(int i=0;i<n;i++) scanf("%d",&v[i]);
	for(int i=0;i<n;i++) scanf("%d",&c[i]);
	while(q--){
		LL a,b;
		cin>>a>>b;
		cout<<query(a,b)<<endl;
	}
}
