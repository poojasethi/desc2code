#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

typedef long long ll;
const int MAXN = 100002;
const ll INF = 1ll<<60;

ll dp[MAXN];
int c[MAXN],v[MAXN];

int main()
{
    int n,q;

    scanf("%d %d",&n,&q);
    for(int i=1;i<=n;i++) scanf("%d",&v[i]);
    for(int i=1;i<=n;i++) scanf("%d",&c[i]);
    ll a,b,u;
    for(int k=1;k<=q;k++)
    {
        scanf("%I64d %I64d",&a,&b);
        ll ans=0;
        int ca=0,cb=0;
        for(int i=1;i<=n;i++) dp[i]=-INF;
        for(int i=1;i<=n;i++)
        {
			u=max(dp[c[i]]+v[i]*a,dp[cb]+v[i]*b);
			if(ca!=c[i])
			{
			    u=max(u,dp[ca]+v[i]*b);
				if(dp[ca]<u)cb=ca,ca=c[i];
				else if(dp[cb]<u) cb=c[i];
			}
			dp[c[i]]=max(dp[c[i]],u);
			ans=max(ans,dp[c[i]]);
        }
        printf("%I64d\n",ans);
    }
    return 0;
}
