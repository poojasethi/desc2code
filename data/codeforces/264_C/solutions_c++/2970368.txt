#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 100005
#define ll long long
#define mk make_pair
#define x first
#define y second
int n,m,c[N],v[N];
ll f[N],g[N],ans,A,B;
pair<ll,int> m1,m2,res;
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) scanf("%d",&v[i]);
	for (int i=1;i<=n;i++) scanf("%d",&c[i]);
	while (m--){
		scanf("%I64d%I64d",&A,&B),ans=0;
		m1=m2=mk(-1LL<<55,0); memset(g,200,sizeof(g));
		for (int i=1;i<=n;i++){
			f[i]=max(B*v[i],g[c[i]]+A*v[i]);
			f[i]=max(f[i],(m1.y==c[i]?m2.x:m1.x)+B*v[i]);
			if (f[i]>g[c[i]]){
				g[c[i]]=f[i]; res=mk(f[i],c[i]);
				if (m1.y==c[i]) m1=res;
				else{
					if (m2.y==c[i]) m2=res;
					if (res>m1) m2=m1,m1=res;
						else if (res>m2) m2=res;
					}
				}
			ans=max(ans,f[i]);
			}
		printf("%I64d\n",ans);
		}
	return 0;
}
