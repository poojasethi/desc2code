#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<iostream>
#include<iomanip>
using namespace std;

#define INF (int) 1e8

int main()
{
  int n, q;
  scanf("%d %d",&n,&q);
  int c[n+2], v[n+2], m1 = 0;
  for(int i=1;i<=n;++i)
    scanf("%d",&v[i]);
  for(int i=1;i<=n;++i) {
    scanf("%d",&c[i]);
    m1 = max(m1, c[i]);
  }
  while(q--) {
    int a, b;
    scanf("%d %d",&a,&b);
    long long dp[100002];
    fill(dp+1, dp+n+1, -(long long) 1e15);
    //    memset(dp, ((long long) -1e15), sizeof dp);
    int k=0, u=0;
    for(int i=1;i<=n;++i) {
      long long as = max(dp[c[i]] + (v[i] * 1LL * a) , dp[k] + (v[i] * 1LL * b));
      if(u != c[i])
	as = max(as, dp[u] + (1LL * v[i] * b));
      if(u != c[i]) {
	if(dp[u] < as) {
	  k = u;
	  u = c[i];
	}
        else if(dp[k] < as)
	  k = c[i];
      }
      dp[c[i]] = max(dp[c[i]], as);
    }
    cout << dp[u] << endl;
  }
}
