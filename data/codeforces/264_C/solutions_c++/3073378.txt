#include <cstdio>
long long maxn[100005];
bool used[100005];
int c[100005];
int v[100005];
template <class T>
T max(const T& a, const T& b) {return a>b?a:b;}
const int INF=1<<29;
int main() {
	int n, q;
	scanf("%d %d", &n, &q);
	for(int i=0; i<n; ++i) scanf("%d", &v[i]);
	for(int i=0; i<n; ++i) scanf("%d", &c[i]);
	while( q-- ) {
		for(int i=1; i<=n; ++i) maxn[i]=-INF, used[i]=false;
		long long a, b;
		scanf("%I64d %I64d", &a, &b);
		int max1=0, max2=0;
		long long tmpa, tmpb;
		for(int i=0; i<n; ++i) {
			tmpa=used[c[i]] ? maxn[c[i]]+max(0ll, a*v[i]) : b*v[i];
			tmpb=maxn[ (c[i]^max1)?max1:max2 ]+b*v[i];
			maxn[c[i]]=max(tmpa, tmpb);
			if( c[i]!=max1 ) {
				if( maxn[c[i]]>maxn[max1] ) max2=max1, max1=c[i];
				else if( maxn[c[i]]>maxn[max2] ) max2=c[i];
			}
			used[c[i]]=true;
		}
		printf("%I64d\n", maxn[max1]);
	}
}
