#include <cstdio>
typedef long long ll;
#define MN 100005
#define max(a, b) ((a)>(b)?(a):(b))
const ll inf = 99999999999999999LL;
int n, Q, c[MN];
ll v[MN], a, b, d[MN], D;
ll othermax(int i, int j, int k) {
	if (i == k) return d[j];
	return d[i];
}
int main() {
	scanf("%d%d", &n, &Q);
	for (int i = 1; i <= n; i++) scanf("%I64d", &v[i]);
	for (int i = 1; i <= n; i++) scanf("%d", &c[i]);
	for (int k = 1; k <= Q; k++) {
		scanf("%I64d%I64d", &a, &b);
		for (int i = 1; i <= n; i++) d[i] = -inf;
		int big1 = 0, big2 = 0;
		for (int i = 1; i <= n; i++) {
			d[c[i]] = max(d[c[i]], max(d[c[i]] + a*v[i], othermax(big1, big2, c[i]) + b*v[i]));
			if (d[big1] < d[c[i]] && big1 != c[i]) big2 = big1, big1 = c[i];
			if (d[big2] < d[c[i]] && big1 != c[i]) big2 = c[i];
		}
		ll res = 0;
		for (int i = 1; i <= n; i++) if (res < d[i]) res = d[i];
		printf("%I64d\n", res);
	}
}
