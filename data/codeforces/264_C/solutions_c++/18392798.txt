#include<bits/stdc++.h>

using namespace std;

#define f for(i=1;i<=n;i++)
#define N 100005

long long A[N],C[N],Mx[N],n,w,i,a,b,c,x,y,asd;

int main(){

    cin >> n >> w;
    f cin >> A[i];
    f cin >> C[i];

    while(w--){

        cin >> a >> b;

        memset(Mx , -9 , sizeof Mx);

        Mx[0] = asd = x = 0;

       f{  c = C[i];

            asd = max( asd , Mx[c] = max(Mx[c] , max( Mx[c] + A[i] * a , Mx[c == x ? y : x] + A[i] * b )));

            if(Mx[c] > Mx[y] && c != x)
                y = c;

            if(Mx[x] < Mx[y])
                swap(x , y);
        }

        cout << asd << endl;
    }
}
