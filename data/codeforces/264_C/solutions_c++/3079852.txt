#include <cstdio>
#include <vector>
#include <algorithm>
#define N 100010
#define INF ((1ll)<<62)
using namespace std;
typedef long long LL;

LL v[N], c[N], dp[N];
int main (){
	LL n, q; scanf("%I64d%I64d", &n, &q);
	for(LL i = 0; i < n; i++) scanf("%I64d", &v[i]);
	for(LL i = 0; i < n; i++) scanf("%I64d", &c[i]), c[i]--;
	while(q--){
		LL a, b; scanf("%I64d%I64d", &a, &b);
		fill(dp, dp + n, -INF);
		LL ca = n, cb = n; dp[n] = 0;
		for(LL i = 0; i < n; i++){
			LL x = max(dp[cb]+v[i]*b, dp[c[i]]+v[i]*a);
			if(ca != c[i]) x = max(x, dp[ca]+v[i]*b);
			dp[c[i]] = max(dp[c[i]], x);
			if(c[i] != ca){
				if(dp[c[i]] > dp[ca]) cb = ca, ca = c[i];
				else if(dp[c[i]] > dp[cb]) cb = c[i];
			}
		}
		printf("%I64d\n", dp[ca]);
	}
}
