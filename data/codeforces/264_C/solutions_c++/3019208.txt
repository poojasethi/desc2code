#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
LL I=1LL<<40;
int n,q,v[1<<17],c[1<<17];
LL w[1<<17];
int main(void) {
  cin>>n>>q;
  for(int i=0;i<n;++i)cin>>v[i];
  for(int i=0;i<n;++i)cin>>c[i];
  for(LL a,b;q-->0&&cin>>a>>b;){
    for(int i=0;i<=n;++i)w[i]=-I;
    LL t1=-I,t2=-I;
    int c1=0,c2=0;
    for(int i=0;i<n;++i){
      int c3=c[i];
      LL t3=max(b*v[i],a*v[i]+w[c3]);
      t3=max(t3,(c1==c3?a:b)*v[i]+t1);
      t3=max(t3,(c2==c3?a:b)*v[i]+t2);
      w[c3]=max(w[c3],t3);
      if(c1==c3)t1=max(t1,t3);
      else if(c2==c3)t2=max(t2,t3);
      else {
        if(t2>t1)swap(t1,t2),swap(c1,c2);
        if(t3>t1)swap(t1,t3),swap(c1,c3);
        if(t3>t2)swap(t2,t3),swap(c2,c3);
      }
    }
    cout<<max(0LL,max(t1,t2))<<endl;
  }
}
