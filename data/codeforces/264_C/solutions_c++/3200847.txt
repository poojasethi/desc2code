#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

const int N=100005;
const long long INF=(1ll)<<60;

int n, q, c[N];
long long f[N], v[N], a, b;
bool vs[N];

int main()
{
	scanf("%d%d", &n, &q);
	for(int i=0; i<n; i++) scanf("%I64d", &v[i]);
	for(int i=0; i<n; i++) scanf("%d", &c[i]);
	while(q--)
	{
		scanf("%I64d%I64d", &a, &b);
		for(int i=1; i<=n; i++) f[i]=-1, vs[i]=false;
		f[0]=-INF;
		int id1=0, id2=0;

		for(int i=0; i<n; i++)
		{
			int id=c[i];
			long long max1=f[id], max2=(id==id1)?f[id2]:f[id1];

			if(!vs[id]) f[id]=max(0ll, max2)+b*v[i], vs[id]=true;
			else f[id]=max(f[id], max(max1+a*v[i], max(0ll, max2)+b*v[i]));

			if(f[id]>=f[id1])
			{
				if(id!=id1) id2=id1;
				id1=id;
			}
			else if(f[id]>=f[id2]) id2=id;
		}
		printf("%I64d\n", max(0ll, f[id1]));
	}
	return 0;
}
