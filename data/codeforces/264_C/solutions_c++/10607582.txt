#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N=1e5+10;
const ll INF=1e18;
int v[N], c[N];
ll dp[N];

int main() {
    int n, q;
    scanf("%d %d", &n, &q);
    for(int i=0; i<n; i++) scanf("%d", &v[i]);
    for(int i=0; i<n; i++) scanf("%d", &c[i]);
    while(q--) {
        int a, b, c1=0, c2=0;
        ll ans;
        scanf("%d %d", &a, &b);
        for(int i=1; i<=n; i++) dp[i]=-INF;
        for(int i=0; i<n; i++) {
            ans=max(dp[c[i]]+1ll*v[i]*a, dp[c2]+1ll*v[i]*b);
            if(c1!=c[i]) {
                ans=max(ans, dp[c1]+1ll*v[i]*b);
                if(dp[c1]<ans) c2=c1, c1=c[i];
                else if(dp[c2]<ans) c2=c[i];
            }
            dp[c[i]]=max(dp[c[i]], ans);
        }
        printf("%I64d\n", dp[c1]);
    }
    return 0;
}
