#include<cstdio>
#include<cstring>
#include<algorithm>
#define INF 9187201950435737472ll
#define N 100005
using namespace std;
typedef long long LL;
LL big1,big2,f[N],a[N],now,ans,A,B;
int Q,n,i,b[N],num;
inline void update(LL k)
{
  ans=max(ans,k);
  f[b[i]]=max(f[b[i]],k);
  if (k>big1)
  {
    if (b[i]==num) big1=k;
    else big2=big1,num=b[i],big1=k;
  }
  else if (k>big2&&b[i]!=num) big2=k;
}
int main()
{
  scanf("%d%d",&n,&Q);
  for (i=1;i<=n;i++) scanf("%I64d",&a[i]);
  for (i=1;i<=n;i++) scanf("%d",&b[i]);
  while (Q--)
  {
    scanf("%I64d%I64d",&A,&B);
    memset(f,128,sizeof(f));
    ans=0;big1=big2=0;num=0;
    for (i=1;i<=n;i++)
    {
      now=(num==b[i])?big2:big1;
      if (now<0) now=0;now+=a[i]*B;
      if (f[b[i]]>-INF) now=max(now,f[b[i]]+a[i]*A);
      update(now);
    }
    printf("%I64d\n",ans);
  }
  return 0;
}
