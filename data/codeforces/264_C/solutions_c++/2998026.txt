#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
LL c[100010],v[100010],f[100010];
	int
main ( int argc, char *argv[] )
{
	int n,m;
	cin >> n >> m;
	for(int i=1;i<=n;++i) cin >> v[i];
	for(int i=1;i<=n;++i) cin >> c[i];
	LL a,b,tmp;
	int fst,sec;
	while(m --) {
		cin >> a >> b;
		for(int i=0;i<=n;++i) f[i] = (LL)-1e15;
		fst = sec = 0;
		f[0] = 0;
		for(int i=1;i<=n;++i) {
			tmp = max(f[c[i]]+v[i]*a,f[sec]+v[i]*b);
			if(fst != c[i]) {
				tmp = max(tmp,f[fst]+b*v[i]);
				if(tmp > f[fst]){
					sec=fst ; fst = c[i];
				} else if(tmp > f[sec]) sec = c[i];
					
			}
			f[c[i]] = max(f[c[i]],tmp);
		}
		
		cout << f[fst];
		if (m) cout << endl;
	}
	return EXIT_SUCCESS;
}				/* ----------  end of function main  ---------- */
