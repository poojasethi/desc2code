#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<queue>
#include<set>
#include<map>
#include<cmath>
using namespace std;
#define N  100010
#define ll long long
ll v[N],c[N],d[N];

int main()
{
	int n,q,i,j,k,ma,mb;
	ll a,b,m[2],tmp;
	scanf("%d%d",&n,&q);
	for(i=1; i<=n; i++)	scanf("%I64d",v+i);
	for(i=1; i<=n; i++)	scanf("%I64d",c+i);
	while(q--){
		scanf("%I64d%I64d",&a,&b);
		ma=mb=0;
		for(i=1; i<=n; i++)	d[i]=(ll)-1e15;	d[0]=0;
		for(i=1; i<=n; i++){

			tmp=max(d[c[i]]+a*v[i],d[mb]+b*v[i]);
			if(c[i]!=ma)	tmp=max(tmp,d[ma]+b*v[i]);

			if(c[i]!=ma){
				if(tmp>d[ma])	mb=ma,ma=c[i];
				else if(tmp>d[mb])	mb=c[i];
			}
			d[c[i]]=max(d[c[i]],tmp);
		}
		printf("%I64d\n",d[ma]);
	}
	return 0;
}
