#include<cstdio>
#include<iomanip>
#include<iostream>
using namespace std;
int main(){
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    int n,q,a,b,ca,cb,*v=new int[100001],*c=new int[100001];
    long long dp[100001],as;
    cin>>n>>q;
    for(int i=1;i<=n;i++)
        cin>>v[i];
    for(int i=1;i<=n;i++)
        cin>>c[i];
    for(int j=0;j<q;j++){
        scanf("%d%d",&a,&b);
        for(int i=1;i<=n;i++)dp[i]=(long long)-1e15;
        ca=cb=0;
        for(int i=1;i<=n;i++){
            as=max(dp[c[i]]+v[i]*1LL*a,dp[cb]+v[i]*1LL*b);
            if(ca!=c[i]) as=max(as,dp[ca]+v[i]*1LL*b);
            if(ca!=c[i]){
                if(dp[ca]<as)cb=ca,ca=c[i];
                else if(dp[cb]<as) cb=c[i];
            }
            dp[c[i]]=max(dp[c[i]],as);
        }
        printf("%I64d\n",dp[ca]);
    }    
    //system("pause");
}
