#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;

int n,v[100010],c[100010],q;
long long b1[100010],ba,bac,b2;

long long dp()
{
	long long a,b,ans,best;
	while (q--)
	{
		ba=0,bac=-1,b2=0;
		scanf("%I64d%I64d",&a,&b);
		memset(b1,-63,sizeof b1);
		for (int i=0;i<n;i++)
		{
			ans=-1000000000000000000ll;
			ans=max(ans,b1[c[i]]+a*v[i]);
			if (bac==c[i])	best=b2;else best=ba;
			ans=max(ans,best+b*v[i]);
			b1[c[i]]=max(b1[c[i]],ans);
			if (bac==c[i])	ba=max(ba,ans);
			else	
			{
				if (ans>ba)	bac=c[i],b2=ba,ba=ans;
				else b2=max(b2,ans);
			}
		}
		printf("%I64d\n",ba);
	}
}

int main()
{
	scanf("%d%d",&n,&q);
	for (int i=0;i<n;i++)	scanf("%d",v+i);
	for (int i=0;i<n;i++)	scanf("%d",c+i);
	dp();
	return 0;
}
