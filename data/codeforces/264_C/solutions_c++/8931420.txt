#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 233333
using namespace std;
int n,m;
long long v[N],c[N],id[N],ans[4],f[N],ret,a,b;
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%I64d",&v[i]);
	for(int i=1;i<=n;i++) scanf("%I64d",&c[i]);
	
	while(m--){
		scanf("%I64d%I64d",&a,&b);
		
		ans[0]=ans[1]=-23333333333333LL; id[0]=id[1]=0;
		for(int i=1;i<=n;i++) f[i]=-23333333333333LL;
		
		
		for(int i=1;i<=n;i++){
			
			if(id[0]==c[i]) ret=ans[1];
			else ret=ans[0];
			
			ret=max(ret+v[i]*b,f[c[i]]+v[i]*a);
			ret=max(ret,v[i]*b);
			
			f[c[i]]=max(f[c[i]],ret);
			if(id[0]==c[i]){ans[0]=max(ans[0],ret); continue;}
			
			if(ret>ans[0]){
				ans[1]=ans[0]; id[1]=id[0];
				ans[0]=ret,id[0]=c[i];
			}
			else if(ret>ans[1]){
				ans[1]=ret; id[1]=c[i];
			}
		}
		
		printf("%I64d\n",max(0LL,ans[0]));
		
	}
}
