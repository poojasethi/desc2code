#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll,ll> pll;
#define X first
#define Y second
const int u=100010;
const ll inf=1ll<<60;
int c[u],n,q;
ll f[u],v[u],a,b;

void solve(ll a,ll b)
{
	for(int i=0;i<=n;i++) f[i]=-inf;
	f[0]=0;
	pll x(0,0),y(-inf,1);
	for(int i=1;i<=n;i++)
	{
		f[c[i]]=max(f[c[i]],v[i]*a+f[c[i]]);
		if(x.Y!=c[i]) f[c[i]]=max(f[c[i]],v[i]*b+x.X);
		else f[c[i]]=max(f[c[i]],v[i]*b+y.X);
		pll p(f[c[i]],c[i]);
		if(p>x)
		{
			if(p.Y!=x.Y) y=x;
			x=p;
		}
		else if(p>y&&p.Y!=x.Y) y=p;
	}
	ll ans=0;
	for(int i=0;i<=n;i++) ans=max(ans,f[i]);
	printf("%I64d\n",ans);
}

int main()
{
	cin>>n>>q;
	for(int i=1;i<=n;i++) scanf("%I64d",&v[i]);
	for(int i=1;i<=n;i++) scanf("%d",&c[i]);
	while(q--)
	{
		scanf("%I64d%I64d",&a,&b);
		solve(a,b);
	}
	return 0;
}
