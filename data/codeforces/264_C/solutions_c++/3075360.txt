#include<iostream>
using namespace std;
#define R(i,n) for(int i=0;i<n;++i)
#define N 100010
#define I 1ll<<50
int n,q,c,C[N],V[N],L,H;
long long F[N],A,B,a,b,s;
main(){
	cin>>n>>q;
	R(i,n)cin>>V[i];
	R(i,n)cin>>C[i];
	R(j,q){
		R(i,N)F[i]=-I;		
		cin>>a>>b,L=H=F[0]=0;
		R(i,n){
			c=C[i],A=V[i]*a,B=V[i]*b,s=-I;
			s=max(s,max(H!=c?F[H]:F[L],0ll)+B);
			if (F[c]!=-I)s=max(s,F[c]+A);
			if (s>F[c]){
				F[c]=s;
				if (c==L||c==H){
					if (F[L]>F[H])L^=H^=L^=H;
				}else{
					if (F[c]>F[H])L=H,H=c;
					else if (F[c]>F[L])L=c;		
				}
			}
		}
		cout<<F[H]<<endl;
	}
}
