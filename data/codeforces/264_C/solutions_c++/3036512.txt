#include<stdio.h>
#include<algorithm>
#include<string.h>
#define N 100010
#define ll long long

int n,c[N],v[N];
ll dp[N];

ll solve(ll a,ll b)
{
	using namespace std;
	ll tmp;
	int fst=0,sec=0;
	memset(dp+1,128,sizeof(dp)-8);                  //set -INF
	for(int i=1;i<=n;i++){
		tmp=max(dp[c[i]]+a*v[i],dp[sec]+b*v[i]);
		if(fst!=c[i]){
			tmp=max(tmp,dp[fst]+b*v[i]);
			if(dp[fst]<tmp){
				sec=fst;
				fst=c[i];
			}else if(dp[sec]<tmp)
				sec=c[i];
		}
		dp[c[i]]=max(dp[c[i]],tmp);
	}
	return dp[fst];
}
int main()
{
	int q;
	scanf("%d %d",&n,&q);
	for(int i=1;i<=n;i++)
		scanf("%d",&v[i]);
	for(int i=1;i<=n;i++)
		scanf("%d",&c[i]);
	while(q--){
		ll a,b;
		scanf("%I64d %I64d",&a,&b);
		printf("%I64d\n",solve(a,b));
	}
	return 0;
}
