#include <cstdio>
#include <cstring>
#include <algorithm>
#define INF 210000000000000000ll
using namespace std;

int n,m,a[100005],c[100005];
long long A,B;
long long f[100005];

void solve()
{
	scanf("%I64d%I64d",&A,&B);
	for(int i=1;i<=n+1;i++)f[i]=-INF;
	long long maxv=-INF,tmaxv=-INF,ans=0;
	for(int i=1,maxc=0;i<=n;i++)
	{
		long long nowf=1ll*B*a[i];
		if(c[i]!=maxc) nowf=max(nowf,maxv+1ll*B*a[i]);
		else nowf=max(nowf,tmaxv+1ll*B*a[i]);
		nowf=max(nowf,f[c[i]]+1ll*A*a[i]);
		f[c[i]]=max(f[c[i]],nowf);
		if(nowf>maxv)
		{
			if(maxc^c[i])tmaxv=maxv;
			maxv=nowf;
			maxc=c[i];
		}else
		{
			if(c[i]!=maxc && nowf>tmaxv)	tmaxv=nowf;
		}
		ans=max(ans,nowf);
	}
	printf("%I64d\n",ans);
}

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)scanf("%d",&c[i]);
	while(m--)
			solve();
	return 0;
}
