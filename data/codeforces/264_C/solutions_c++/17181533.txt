#include<bits/stdc++.h>
using namespace std;
long long n,q,col[100005],val[100005],vis[505],a,b,max1,max2,max3,ind1,ind2,ind3,dp[100005];
int main()
{
	cin>>n>>q;
	col[0]=0;
	
	for(int i=1;i<=n;i++)
		cin>>val[i-1];
	for(int i=1;i<=n;i++)
		cin>>col[i-1];
	
	while(q--)
	{
		for(int i=1;i<100002;i++)
			dp[i]=-1000000000000ll;
		cin>>a>>b;
		max1=0;max2=1;
		
		for(int i=0;i<n;i++)
		{
			dp[col[i]]=max(dp[col[i]],dp[col[i]]+a*val[i]);
		
			if(max1==col[i])
			{
				dp[col[i]]=max(dp[col[i]],dp[max2]+b*val[i]);
				continue;	
			}
			else
				dp[col[i]]=max(dp[col[i]],dp[max1]+b*val[i]);
			
			if(dp[col[i]]>dp[max1])
			{
				max2=max1;
				max1=col[i];
			}
			else if(dp[col[i]]>=dp[max2])
				max2=col[i];
		}
		cout<<dp[max1]<<endl;
	}
	return 0;
}
