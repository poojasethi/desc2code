#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 100000;
const long long INF = 1000000000000000;
int c[N+1];
long long v[N+1], cw[N+1];

int main()
{
	int n, q;
	scanf("%d%d", &n, &q);
	for(int i = 1; i<=n; ++i) scanf("%lld", &v[i]);
	for(int i = 1; i<=n; ++i) scanf("%d", &c[i]);
	for(int i = 1; i<=q; ++i) {
		long long a, b;
		scanf("%lld%lld", &a, &b);

		long long w1 = 0, w1c = 0, w2 = 0;
		for(int i = 1; i<=n; ++i) cw[i] = -INF;
		for(int i = 1; i<=n; ++i) {
			cw[c[i]] = max(cw[c[i]], max(cw[c[i]]+a*v[i], (w1c==c[i]?w2:w1)+b*v[i]));
			if(w1c==c[i]) w1 = cw[c[i]];
			else if(cw[c[i]]>w1) {
					w2 = w1;
					w1 = cw[c[i]];
					w1c = c[i];
			}
			else w2 = max(w2, cw[c[i]]);
		}

		printf("%lld\n", w1);
	}
}
