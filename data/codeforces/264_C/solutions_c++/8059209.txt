#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int64 a,b;
} ma1,ma2;
int i,j,k,n,m;
int v[N],c[N];
int64 A,B,a[N],an;
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) scanf("%d",&v[i]);
	For(i,1,n) scanf("%d",&c[i]);
	For(j,1,m) {
		scanf("%I64d%I64d",&A,&B);
		ma1.a=ma2.a=ma1.b=ma2.b=0;
		memset(a,130,sizeof(a));
		an=0;
		For(i,1,n) {
			int64 C;
			if (ma1.b==c[i]) C=ma2.a;
			else C=ma1.a;
			C=max(a[c[i]]+A*v[i],C+B*v[i]);
			if (ma1.a<C) {
				if (ma1.b!=c[i]) ma2=ma1;
				ma1=(ww){C,c[i]};
			} else if (ma1.b!=c[i]&&ma2.a<C) ma2=(ww){C,c[i]};
			a[c[i]]=max(a[c[i]],C);
			an=max(an,C);
		}
		printf("%I64d\n",an);
	}
	return 0;
}
