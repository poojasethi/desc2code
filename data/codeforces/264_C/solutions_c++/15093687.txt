#include<bits/stdc++.h>
using namespace std ;
#define ll long long 
ll N,Q ,u,ans ;
#define INF 1ll<<60
ll col[100009],val[100009] ,dp[100009] ;
int main()
{
	scanf("%I64d%I64d",&N,&Q) ;
	for(int i = 1;i<=N;i++)
	  scanf("%I64d",&val[i]) ;
	for(int i =1;i<=N;i++)
	  scanf("%I64d",&col[i]) ;
	ll a,b,ca ,cb;
	for(int j = 1;j<=Q;j++)
	{
		 ans = 0;
		 ca = cb = 0;
		for(int i = 1;i<=100008;i++)
		  dp[i] = -INF ;
		scanf("%I64d%I64d",&a,&b) ;
		for(int i = 1;i<=N;i++)
		  {
//		  	cout<<a<<" "<<b<<endl;
		  	  u = max(dp[col[i]]+val[i]*a,dp[cb]+val[i]*b) ;
		  	  if(col[i]!=ca)
		  	  {
		  	  	u  = max(dp[ca]+val[i]*b,u) ;
					if(u>dp[ca]) cb= ca,ca = col[i] ;
					else if(dp[cb]<u) cb = col[i] ;
				}
			dp[col[i]]  = max(u,dp[col[i]]) ;
			ans = max(ans,dp[col[i]]) ;
//			cout<<ans<<" ";
		  }
		cout<<ans<<endl;
	}
}
