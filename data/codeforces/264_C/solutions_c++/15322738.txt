#include<bits/stdc++.h>
using namespace std;
long long c[100005];
long long  v[100005];
const long long ooll=100000000000;
int main()
{
	int n,q,a,b;
	cin>>n>>q;
	for(int i=0;i<n;i++)
		cin>>v[i];
	for(int i=0;i<n;i++)
		cin>>c[i],--c[i];
	while(q--)
	{
		long long ans=0;
		cin>>a>>b;
		vector<long long> dp(n,-ooll/2);
			int max0=0,max1=1;
			for(int i=0;i<n;i++)
			{
				long long get=v[i]*b;
				get=max(get,dp[c[i]]+v[i]*a);
				if (max0!=c[i]) get=max(get,dp[max0]+v[i]*b);
				else get=max(get,dp[max1]+v[i]*b);
				dp[c[i]]=max(dp[c[i]],get);
				if (max0!=c[i])
				{
					if (dp[c[i]]>dp[max0]) max1=max0,max0=c[i];
					else if (dp[c[i]]>dp[max1]) max1=c[i];
				}
				ans=max(ans,get);
			}
		cout<<ans<<"\n";
	}
	return 0;
}
