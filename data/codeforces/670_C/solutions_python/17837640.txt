co={}
for i in map(int,raw_input().split()):
    if i not in co:co[i]=1
    else:co[i]+=1
a,b=0,0
ind=0
n=int(raw_input())
b2=map(int,raw_input().split())
c=map(int,raw_input().split())
for i in xrange(n):
    a1,b1=0,0
    if b2[i] in co:a1=co[b2[i]]
    if c[i] in co:b1=co[c[i]]
    if a1>a or a1>=a and b1>b:
        a=a1
        b=b1
        ind=i
print ind+1
