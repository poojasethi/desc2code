import os

n = int(raw_input())
guset_list = map(int,raw_input().split())
n_film = int(raw_input())
audio_list = map(int,raw_input().split())
title_list = map(int,raw_input().split())

guset_d = dict()
for x in guset_list:
	guset_d[x] = guset_d[x]+1 if x in guset_d else 1

ans = 0
max_m = 0
max_mm = 0

for i in xrange(len(audio_list)):
	ids = audio_list[i]
	ids2 = title_list[i]

	if ids in guset_d:
		curm = guset_d[ids]
	else:
		curm = 0
	if ids2 in guset_d:
		curmm = guset_d[ids2]
	else:
		curmm = 0

	if curm>max_m:
		ans = i
		max_m = curm
		max_mm = curmm

	elif curm == max_m:
		if curmm > max_mm:
			ans = i
			max_mm = curmm

print ans+1
