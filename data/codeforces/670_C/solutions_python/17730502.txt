n = int(raw_input())
a = map(int, raw_input().split())
m = int(raw_input())
b = map(int, raw_input().split())
c = map(int, raw_input().split())

count = {}
for num in a:
    count[num] = count.get(num, 0) + 1

ans = 0
for i in xrange(m):
    if count.get(b[i], 0) > count.get(b[ans], 0):
        ans = i
    elif count.get(b[i], 0) == count.get(b[ans], 0) and count.get(c[i], 0) > count.get(c[ans], 0):
        ans = i

print ans + 1
