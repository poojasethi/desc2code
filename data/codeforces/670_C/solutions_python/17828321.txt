n=input()
li=map(int,raw_input().split())
dic={}
for i in li:
	if i in dic:
		dic[i]+=1
	else:
		dic[i]=1
m=input()
a=map(int,raw_input().split())
b=map(int,raw_input().split())
l=[]
for i in xrange(m):
	if a[i] in dic:
		x=dic[a[i]]
	else:
		x=0
	if b[i] in dic:
		y=dic[b[i]]
	else:
		y=0
	l+=[[x,y]]
m1=0
m2=0
p=0
for i in xrange(m):
	if(l[i][0]>m1):
		m1=l[i][0]
		m2=l[i][1]
		p=i
	elif(l[i][0]==m1):
		if(l[i][1]>m2):
			m1=l[i][0]
			m2=l[i][1]
			p=i
print p+1
