from collections import Counter

n = int(raw_input())
aa = map(int, raw_input().split())
aa = Counter(aa)
m = int(raw_input())
bb = map(int, raw_input().split())
cc = map(int, raw_input().split())
best = 0, 0
ans = 0
for i in range(m):
    b, c = bb[i], cc[i]
    if best < (b in aa and aa[b], c in aa and aa[c]):
        best = (b in aa and aa[b], c in aa and aa[c])
        ans = i
print ans + 1
