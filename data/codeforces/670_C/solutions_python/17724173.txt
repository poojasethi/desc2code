from sys import stdin
from collections import defaultdict
def main():
    n = int(stdin.readline())
    a = stdin.readline().split()
    d = defaultdict(int)
    for x in a:
        d[x] += 1
    m = int(stdin.readline())
    b = stdin.readline().split()
    c = stdin.readline().split()
    t = [(d[b[i]], d[c[i]], i + 1) for i in xrange(m)]
    print max(t)[2]
main()
