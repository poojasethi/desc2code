#!/usr/bin/python

import operator

n = input()
n = int(n)

lang = raw_input()
langs = lang.split()

d = {}
for l in langs:
	d[l]=0

for l in langs:
	d[l]+=1

m = input()
m = int(m)

aud = raw_input()
auds = aud.split()

sub = raw_input()
subs = sub.split()

count = {}
count_subs = {}

for mov in auds:
	count[mov] = d.get(mov,0)

movie_max = max(count.iteritems(), key=operator.itemgetter(1))[0]

max_val = d.get(movie_max,0)

i = 0
mov_lis = []
for mov in auds:
	if count[mov] == max_val:
		mov_lis.append(i);
	i+=1

for movie in mov_lis:
	l = subs[movie]
	count_subs[l] = d.get(l,0)

movie_max = max(count_subs.iteritems(), key=operator.itemgetter(1))[0]
max_val = d.get(movie_max,0)

for movie in mov_lis:
	l = subs[movie]
	if count_subs[l] == max_val:
		print movie + 1;
		break;
