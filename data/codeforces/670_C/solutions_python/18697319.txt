from collections import Counter

n = int(raw_input())
language_counter = Counter(map(int, raw_input().split()))
m = int(raw_input())
sound_languages = map(int, raw_input().split())
subtitle_languages = map(int, raw_input().split())

best = (0, 0, 1)
for i in xrange(m):
    new_item = (language_counter[sound_languages[i]], language_counter[subtitle_languages[i]], i + 1)
    if new_item > best:
        best = new_item

print best[-1]
