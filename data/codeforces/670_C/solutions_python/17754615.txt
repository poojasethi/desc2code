r=lambda:map(int,raw_input().split())
n=int(raw_input())
ints=r()

speakers = {}
for integer in ints:
	if integer not in speakers:
		speakers[integer] = 1
	else: speakers[integer] += 1

m=int(raw_input())
movieAudio = {}
ints=r()
i=1
for integer in ints:
	movieAudio[i] = integer
	#print [i,integer]
	i+=1

subTitle = {}
ints=r()
i=1
for integer in ints:
	subTitle[i] = integer
	#print [i,integer]
	i+=1

optimalSolution = 1
satisfaction = [ 0, 0 ] 

for j in range(1,i):
	if movieAudio[j] in speakers and speakers[movieAudio[j]] > satisfaction[0]:
		optimalSolution=j
		satisfaction[0] = speakers[movieAudio[j]] 
		if subTitle[j] in speakers:
			satisfaction[1] = speakers[subTitle[j]]
		else:
			satisfaction[1] = 0
	elif movieAudio[j] in speakers and speakers[movieAudio[j]] == satisfaction[0] and subTitle[j] in speakers and speakers[subTitle[j]] > satisfaction[1] :
		optimalSolution=j
		satisfaction[1] = speakers[subTitle[j]]
	elif not satisfaction[0] and subTitle[j] in speakers and speakers[subTitle[j]] > satisfaction[1] :
		optimalSolution=j
		satisfaction[1] = speakers[subTitle[j]]

print (optimalSolution)
