b = map( int, raw_input().split())
c = map( int, raw_input().split())


d = dict()
for x in a:
    if x not in d:
        d [x] = 0
    d [x] += 1

idx,mb,mc = -1,-1,-1

for i in xrange(len(b)):
    ct1 = d.get(b[i], 0)
    ct2 = d.get(c[i], 0)
    if (mb,mc) <= (ct1,ct2):
        idx = i + 1
        mb = ct1
        mc = ct2
