n = int(raw_input())
langlist = map(int,raw_input().split())
m = int(raw_input())
audiolist = map(int,raw_input().split())
sublist = map(int,raw_input().split())

langcount = {}
for lang in langlist:
    try:
        langcount[lang] += 1
    except KeyError:
        langcount[lang] = 1

try:
    audiomax = langcount[audiolist[0]]
except KeyError:
    audiomax = 0
try:
    submax = langcount[sublist[0]]
except KeyError:
    submax = 0
choose = 0

for i in range(1,m):
    try:
        aud = langcount[audiolist[i]]
    except KeyError:
        aud = 0
    try:
        sub = langcount[sublist[i]]
    except KeyError:
        sub = 0
    if (aud > audiomax):
        audiomax = aud
        submax = sub
        choose = i
    elif (aud == audiomax) and (sub > submax):
        submax = sub
        choose = i

print choose + 1
