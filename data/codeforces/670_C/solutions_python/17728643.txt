#!/usr/bin/python
import sys
n = int(sys.stdin.readline())
know = {}
a = map(int, sys.stdin.readline().split())
for x in a:
    if x in know:
        know[x] += 1
    else:
        know[x] = 1
m = int(sys.stdin.readline())
b = map(int, sys.stdin.readline().split())
c = map(int, sys.stdin.readline().split())

best_sat = 0
best_almost = 0
best_index = 0
for i in range(m):
    sat = know.get(b[i], 0)
    almost = know.get(c[i], 0)
    if sat > best_sat or (sat == best_sat and almost > best_almost):
        best_sat = sat
        best_almost = almost
        best_index = i
print best_index + 1
