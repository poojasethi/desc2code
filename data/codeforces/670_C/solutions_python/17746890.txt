from sys import stdin, stdout
import math

nScientists = int(stdin.readline().rstrip())
lSientists = [int(x) for x in stdin.readline().rstrip().split(" ")]
nMovies = int(stdin.readline().rstrip())
audios = [int(x) for x in stdin.readline().rstrip().split(" ")]
subs = [int(x) for x in stdin.readline().rstrip().split(" ")]

langRate = {}
for s in lSientists:
    if s in langRate:
        langRate[s] += 1
    else:
        langRate[s] = 1

maxAudioRate = 0
for i, a in enumerate(audios):
    if (a in langRate) and (langRate[a] > maxAudioRate):
        maxAudioRate = langRate[a]

bestMovies = []
for i, a in enumerate(audios):
    if (a in langRate) and (langRate[a] == maxAudioRate):
        bestMovies.append(i)

if len(bestMovies) == 0:
    bestMovies = range(0, nMovies)

preferred = bestMovies[0]
if len(bestMovies) > 1:
    max = 0
    for i in bestMovies:
        subLang = subs[i]
        if (subLang in langRate) and (langRate[subLang] > max):
            max = langRate[subLang]
            preferred = i

print preferred + 1
