from collections import defaultdict
def solve():
    maxnum = 0
    for avail in audio.keys():
	if pref[avail] > maxnum:
	    maxnum = pref[avail]

    maxlang = []

    for language, number in pref.iteritems():
	if number == maxnum:
	    maxlang.append(language)

#    print 'maxnum', maxnum
    contenders = []
    for lang in maxlang:
	contenders.extend(audio[lang])

    contenders = list(set(contenders))

    if len(contenders) == 0:
	return 1

    maxwin = 0
    idxwin = contenders[0]
    for idx in contenders:
#	print 'idx:', idx, ', ti[idx]', ti[idx], 'pref[ti[idx]]', pref[ti[idx]], 'maxwin', maxwin, 'idxwin', idxwin
	if pref[ti[idx]] > maxwin:
	    maxwin = pref[ti[idx]]
	    idxwin = idx

    return idxwin + 1

n = int(raw_input())
N = map(int, raw_input().split(' '))

m = int(raw_input())
au = map(int, raw_input().split(' '))
ti = map(int, raw_input().split(' '))

pref = defaultdict(int)
audio = defaultdict(list)
sub = defaultdict(list)

for p in N:
    pref[p] += 1

for i, a in enumerate(au):
    audio[a].append(i)

for i, t in enumerate(ti):
    sub[t].append(i)

print solve()
