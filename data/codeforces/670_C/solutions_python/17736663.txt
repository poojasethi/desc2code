from operator import itemgetter
r=lambda: map(int, raw_input().split())
n =input()
a = r()
m = input()
b= r()
c= r()

languagedict=dict()
for i in a:
    languagedict[i]=languagedict.get(i,0)+1
happiness=[languagedict.get(b[i],0) for i in xrange(m)]
satisfy=[languagedict.get(c[i],0) for i in xrange(m)]
maxhappy=max(happiness)
happyindex=[i for i in xrange(m) if happiness[i]==maxhappy]
if len(happyindex)==1:
    print happyindex[0]+1
else:
    bestj=happyindex[0]
    for j in happyindex:
        if satisfy[j]>satisfy[bestj]:
            bestj=j
    print bestj+1
