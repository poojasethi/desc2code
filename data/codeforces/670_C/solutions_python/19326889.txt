n = int(raw_input())
a = map(int,raw_input().split(' '))
m = int(raw_input())
b = map(int,raw_input().split(' '))
c = map(int,raw_input().split(' '))

d = {}
for ai in a:
    if not ai in d: d[ai] = 0
    d[ai] += 1

max_v = -1
max_ind = -1
for i in range(m):
    if b[i] in d:
        vb = d[b[i]]
    else:
        vb = 0
    if c[i] in d:
        vc = d[c[i]]
    else:
        vc = 0
    if max_v < vb*1000000 + vc:
        max_v = vb*1000000+vc
        max_ind = i+1

print max_ind
