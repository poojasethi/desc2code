#!/usr/bin/python

from collections import defaultdict

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
aa = ia()

m = ir()
bb = ia()
cc = ia()

d = defaultdict(int)

for a in aa:
    d[a] = d[a] + 1

m1, m2 = -1, -1
mi     = 1

for i, (b, c) in enumerate(zip(bb, cc)):
    i = i + 1
    if b==c:
        c1, c2 = d[b], 0
    else:
        c1, c2 = d[b], d[c]
    if c1>m1:
        m1, m2, mi = c1, c2, i
    elif c1==m1 and c2>m2:
        m1, m2, mi = c1, c2, i

print mi
