#include <cstdio>
#include <map>
using namespace std;
const int N=200005;
int date[2][N];
int main()
{
	map<int,int>p;
	map<int,int>g;
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		int x;
		scanf("%d",&x);
		p[x]++;
		g[x]++;
	}
	int m;
	scanf("%d",&m);
	for(int i=0;i<2;i++){
		for(int j=0;j<m;j++){
			scanf("%d",&date[i][j]);
		}
	}
	int m1=-1,m2=-1,ans=0;
	for(int i=0;i<m;i++){
		if(p[date[0][i]]>m1||p[date[0][i]]==m1&&g[date[1][i]]>m2){
			ans=i+1;
			m1=p[date[0][i]];
			m2=g[date[1][i]];
		}
	}
	printf("%d\n",ans);
}
