#include<iostream>
#include<algorithm>
#include<utility>
#include<map>
using namespace std;
int a[222222];
map<int,int> c;
pair<int,int> p[222222];
int main()
{
    int m1,m2,r;
    int i,n,m;
    cin>>n;
    for(i=1;i<=n;i++) cin>>a[i],c[a[i]]++;
    cin>>m;
    for(i=1;i<=m;i++) cin>>p[i].first;
    for(i=1;i<=m;i++) cin>>p[i].second;
    m1=m2=-1;
    for(i=1;i<=m;i++)
    {
        if(c[p[i].first]>m1) m1=c[p[i].first],m2=c[p[i].second],r=i;
        if(c[p[i].first]==m1&&c[p[i].second]>m2) m2=c[p[i].second],r=i;
    }
    cout<<r;
}
