#include<iostream>
#include<vector>
#include<string>
#include<map>
#define MX 300000

using namespace std;

int n, m, a[MX], b[MX], c[MX];
map<int,int> mm;

int main() {
  cin>>n;
  for (int i=0;i<n;i++) {
    cin>>a[i];
    mm[a[i]]++;
  }
  cin>>m;
  for (int i=0;i<m;i++)
    cin>>b[i];
  for (int i=0;i<m;i++)
    cin>>c[i];
  
  int s1 = -1, s2 = -1, best = -1, y;
  for (int i=0;i<m;i++)
    if (mm[b[i]] > s1 || (mm[b[i]] == s1 && mm[c[i]] > s2)) {
      best = i + 1;
      s1 = mm[b[i]];
      s2 = mm[c[i]];
    }
    
  cout<<best<<endl;
  return 0;
}
