#include <bits/stdc++.h>
using namespace std;
const int maxn=200010;

map <int,int> M;

int b[maxn],c[maxn];
int ans,ansb,ansc;
int n,m;

int main()
{
	scanf("%d",&n);
	for (int i=1,x;i<=n;i++) scanf("%d",&x),M[x]++;
	scanf("%d",&m);
	for (int i=1;i<=m;i++) scanf("%d",&b[i]);
	for (int i=1;i<=m;i++) scanf("%d",&c[i]);
	ans=1;
	for (int i=1;i<=m;i++)
	{
		int mb=M[b[i]],mc=M[c[i]];
		if (mb>ansb) ans=i,ansb=mb,ansc=mc;
		else if (mb==ansb && mc>ansc) ans=i,ansb=mb,ansc=mc;
	}
	printf("%d\n",ans);
	return 0;
}
