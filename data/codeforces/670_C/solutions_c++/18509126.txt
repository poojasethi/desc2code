#include <map>
#include <iostream>

#define ll long long
#define MAXN 200001
using namespace std;
int main(){
	int n,m,b[MAXN],c[MAXN],p=0,s=0,x,tp=0,ts=0;
	map<ll,int>a;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>x;
		a[x]++;
	}
	cin>>m;
	for(int i=0;i<m;i++)cin>>b[i];
	for(int i=0;i<m;i++)cin>>c[i];
	/*
	1 = 2
	3 = 2
	6 = 1
	7 = 1
	*/
	x=1;
	for(int i=0;i<m;i++){
		tp = a[b[i]];
		ts = a[c[i]];
		if(tp>p || (tp==p && ts>s)){
			p = tp;
			s = ts;
			x = i+1;
		}
		//cout<<tp<<" "<<ts<<endl;
	}
	cout<<x<<endl;
}
