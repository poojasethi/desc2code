#include "cstdio"
#include "map"
#define NMAX 200000
using namespace std;
std::map<int, int> lang;
int n, m, b[NMAX], c[NMAX];
int main()
{
    int pos=0;
    scanf("%d", &n);
    while (n-->0)
    {
        int x;
        scanf("%d", &x);
        lang[x]++;
    }
    scanf("%d", &m);
    for (int i=0; i<m; i++)
        scanf("%d", &b[i]);
    for (int i=0; i<m; i++)
    {
        scanf("%d", &c[i]);
        if ((lang[b[i]]>lang[b[pos]])||((lang[b[i]]==lang[b[pos]])&&(lang[c[i]]>lang[c[pos]])))
            pos=i;
    }
    printf("%d", pos+1);
    return 0;
}
