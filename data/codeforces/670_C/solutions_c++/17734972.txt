#include <bits/stdc++.h>
using namespace std;

const int N = 200005;

int x,b[N],c[N],n,m,ans;
map<int,int> mp;

int main()
{
	int i;
	scanf("%d",&n);
	for(i = 1; i <= n; ++ i)
	{
		scanf("%d",&x);
		mp[x] ++;
	}
	scanf("%d",&m);
	for(i = 0; i < m; ++ i)
		scanf("%d",&b[i]);
	for(i = 0; i < m; ++ i)
		scanf("%d",&c[i]);
	ans = 0;
	for(i = 0; i < m; ++ i)
	{
		if(mp[b[i]] > mp[b[ans]]) ans = i;
		if(mp[b[i]] == mp[b[ans]] && mp[c[i]] > mp[c[ans]]) ans = i;
	}
	printf("%d",ans + 1);
	return 0;
}
