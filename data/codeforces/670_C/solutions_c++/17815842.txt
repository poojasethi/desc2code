#include<bits/stdc++.h>
#define ll long long
using namespace std;
struct box{
    ll a, b, c;
 }st[200020];

bool cmp(box y, box z){
    if(y.a==z.a) return y.b>z.b;
    else return y.a>z.a;

}
int main()
{
    ll n, x, i, m;
    map<ll, ll>mp;
    cin>>n;
    for(i=0; i<n; i++){cin>>x; mp[x]++;}
    cin>>m;
    for(i=0; i<m; i++){cin>>x; st[i].a=mp[x]; st[i].c=i;}
    for(i=0; i<m; i++){cin>>x; st[i].b=mp[x];}
    sort(st, st+m, cmp);
    cout<<st[0].c+1<<endl;
    return 0;
}
