#include<bits/stdc++.h>
using namespace std;
int main()
{
    map<int,int> m,x,y;
    int i,idx,n,ma=-1,mm,xx,pre;
    cin>>n;
    for(i=0;i<n;i++){
        cin>>xx;
        m[xx]++;
    }
    cin>>mm;
    for(i=0;i<mm;i++)cin>>x[i];
    for(i=0;i<mm;i++)cin>>y[i];
    for(i=0;i<mm;i++){
        if(ma<m[x[i]]) {ma=m[x[i]];idx=i;pre=m[y[i]];}
        else if(ma==m[x[i]]){
            if(m[y[i]]>pre) {pre=m[y[i]];idx=i;}
        }
    }
    cout<<idx+1;
    return 0;
}
