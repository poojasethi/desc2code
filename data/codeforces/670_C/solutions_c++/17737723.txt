#include <stdio.h>
#include <map>
using namespace std;
map<int,int>mp;
int b[1010100];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		int d;
		scanf("%d",&d);
		mp[d]++;
	}
	int m;
	scanf("%d",&m);
	int ans=0,bb=-1,cc=-1;
	for(int i=1;i<=m;i++)
	{

		scanf("%d",&b[i]);
	}
	for(int i=1;i<=m;i++)
	{
		int c;
		scanf("%d",&c);
		if(mp[b[i]]>bb)
		{
			ans=i;
			bb=mp[b[i]];
			cc=mp[c];
		}
		else if(mp[b[i]]==bb&&mp[c]>cc)
		{
			ans=i;
			cc=mp[c];
		}
	}
	printf("%d\n",ans);
}
