#include <bits/stdc++.h>
using namespace std;

const int s = 1e6;

int n, m, a[s], b[s], c[s], y, l = -1, r = -1, ind;
map <int,int> x;

int main(){
	scanf("%d", &n);
	for (int i = 0; i < n; ++i){
		scanf("%d", &y);
		x[y]++;
	}
	scanf("%d", &m);
	for (int i = 0; i < m; ++i)
		scanf("%d", b + i);
	for (int i = 0; i < m; ++i)
		scanf("%d", c + i);
	for (int i = 0; i < m; ++i){
		if (x[b[i]] > l){
			l = x[b[i]];
			r = x[c[i]];
			ind = i;
		} 
		else if (x[b[i]] == l && r < x[c[i]]){
			r = x[c[i]];
			ind = i;
		}
	}
	cout << ind + 1;
	return 0;
}
