#include <bits/stdc++.h>

using namespace std;

const int N=200100;
map <int,int> d;
int a[N],b[N],c[N],n,m,ans;
int main()
{
    cin >> n;
    for (int i=1; i<=n; i++)
    {
        scanf("%d",&a[i]);
        d[a[i]]++;
    }
    cin >> m;
    for (int i=1; i<=m; i++) scanf("%d",&b[i]);
    for (int i=1; i<=m; i++) scanf("%d",&c[i]);
    ans=1;
    for (int i=1; i<=m; i++)
        if ((d[b[ans]]<d[b[i]]) || ((d[b[ans]]==d[b[i]]) && (d[c[ans]]<d[c[i]]))) ans=i;
    cout << ans;
}
