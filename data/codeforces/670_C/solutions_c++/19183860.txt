#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

#define MAXN 200001

map<int,int>m1;
int a[MAXN],b[MAXN],c[MAXN];

int main(){
	int n;scanf("%d",&n);
	for(int i=1;i<=n;++i) scanf("%d",&a[i]),++m1[a[i]];
	int m;scanf("%d",&m);
	for(int i=1;i<=m;++i) scanf("%d",&b[i]);
	for(int i=1;i<=m;++i) scanf("%d",&c[i]);
	int tmp1=-1,tmp2=-1,p=0;
	for(int i=1;i<=m;++i)
		if(m1[b[i]]>tmp1||(m1[b[i]]==tmp1&&m1[c[i]]>tmp2))
			tmp1=m1[b[i]],tmp2=m1[c[i]],p=i;
	printf("%d\n",p);
}
