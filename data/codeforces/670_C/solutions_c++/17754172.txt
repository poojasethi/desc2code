#include<iostream>
#include<algorithm>
#include<map>
using namespace std;

struct node{
	int p,s;
	int cas;
} a[200001];

bool cmp(node x,node y){
	if (x.p != y.p) 
		return x.p > y.p;
	else 
		return x.s > y.s;
}
int n,m;
map<int,int> mp;
int main(){
	cin>>n;
	int x;
	mp.clear();
	for (int i=0 ; i<n ; i++){
		cin>>x;
		mp[x]++;
	}
	cin>>m;
	for (int i=0 ; i<m ; i++){
		cin>>x;
		a[i].p = mp[x];
		a[i].cas = i;
	}
	for (int i=0 ; i<m ; i++){
		cin>>x;
		a[i].s = mp[x];
	}
	sort(a,a+m,cmp);
	cout<<a[0].cas+1<<endl;
}
