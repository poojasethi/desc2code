#include<bits/stdc++.h>
using namespace std;
long long a[200500],b[200500],c[200500];
int n;
int main(){
	map<long long,long long>mp;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
		mp[a[i]]++;
	}
	int k,bb=0,cc=0,res=0;
	cin>>k;
	for(int i=0;i<k;i++)cin>>b[i];
	for(int i=0;i<k;i++)cin>>c[i];
	for(int i=0;i<k;i++){
		if(mp[b[i]]>bb||(mp[b[i]]==bb&&mp[c[i]]>cc)){
			bb=mp[b[i]];
			cc=mp[c[i]];
			res=i;
		}
	}
	cout<<res+1<<endl;
}
