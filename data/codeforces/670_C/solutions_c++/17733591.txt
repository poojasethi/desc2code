#include<iostream>
#include<map>
using namespace std;
main()
{
	ios_base::sync_with_stdio(false);
	map<int,int> mapi;
	int n,m,i,num,bb,bc,bi;
	int b[200001],c[200001];
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>num;
		mapi[num]++;
	}
	cin>>m;
	for(i=0;i<m;i++)
		cin>>b[i];
	for(i=0;i<m;i++)
		cin>>c[i];
	bb=bc=bi=-1;
	for(i=0;i<m;i++)
	{
		if(mapi[b[i]]>bb)
		{
			bb=mapi[b[i]];
			bc=mapi[c[i]];
			bi=i+1;
		}
		else if(mapi[b[i]]==bb&&mapi[c[i]]>bc)
		{
			bb=mapi[b[i]];
			bc=mapi[c[i]];
			bi=i+1;
		}
	}
	cout<<bi<<endl;
}
