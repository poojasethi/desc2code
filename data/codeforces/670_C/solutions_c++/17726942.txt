
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,t;
cin>>n;

map <LL,LL> mp;

for(i=0;i<n;i++)
	{cin>>t;
	mp[t]++;}

LL m;
cin>>m;

vector <LL> v(m),u(m);

for(i=0;i<m;i++)
	cin>>v[i];

for(i=0;i<m;i++)
	cin>>u[i];


LL m1=0,m2=0,id=0;

for(i=0;i<m;i++)
{
if(mp[v[i]]>m1 || (mp[v[i]]==m1 && mp[u[i]]>m2))
{
m1=mp[v[i]];
m2=mp[u[i]];
id=i;
}

}
cout<<id+1<<endl;
}
