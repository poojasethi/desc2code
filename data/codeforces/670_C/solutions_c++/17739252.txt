#include<bits/stdc++.h>
using namespace std;
map<int,int> m;
//map<pair<int,int>,int> m1;
int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		int x;
		scanf("%d",&x);
		m[x]++;
	}
	int q;
	int ans=0,ind=0;
	scanf("%d",&q);
	int a[q+10],b[q+10];
	for(int i=0;i<q;i++)
		scanf("%d",&a[i]);
	for(int i=0;i<q;i++)
		scanf("%d",&b[i]);
	int ans1=0;
	for(int i=0;i<q;i++){
		int x,y;
		x=a[i],y=b[i];
		if(ans<m[x]||(ans==m[x]&&ans1<m[y])){
			ans=m[x];
			ans1=m[y];
			ind=i;
		}
	}
	printf("%d\n",ind+1);
	return 0;
}
