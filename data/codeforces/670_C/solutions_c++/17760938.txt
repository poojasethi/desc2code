#include <bits/stdc++.h>
using namespace std;

int main(){
	int n,k;
	cin>>n; int temp,i,max,ans=0,mov; 
	std::map<int, int> map;
	for(i=0;i<n;i++){
		cin>>temp;
		map[temp]++;
	}
	cin>>k; int aud[k];
	for(i=0;i<k;i++){
		cin>>aud[i];
		if(map[aud[i]]>ans){
			ans=map[aud[i]];
			mov=i;
		}
	}//cout<<mov;
	n=-1;
	for(i=0;i<k;i++){
		cin>>temp;
		if(ans==map[aud[i]]){
			if(n<map[temp]){
				n=map[temp];
				mov=i;
			}
		}
	}
	cout<<mov+1;
	return 0;
}
