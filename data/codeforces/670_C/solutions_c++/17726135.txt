#include <cstdio>
#include <map>

using namespace std;

const int MAXN = 200000, MAXM = 200000;
map<int, int> sp;
int a[MAXN], b[MAXM], c[MAXM], n, m, u, v, p, q, sol;

int main() {
	p = q = sol = -1;
	scanf("%d", &n);
	for(int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
		sp[a[i]]++;
	}
	scanf("%d", &m);
	for(int i = 0; i < m; i++)
		scanf("%d", &b[i]);
	for(int i = 0; i < m; i++) {
		scanf("%d", &c[i]);
		u = sp[b[i]];
		v = sp[c[i]];
		if(p < u || (p == u && q < v)) {
			p = u;
			q = v;
			sol = i;
		}
	}
	printf("%d\n", sol + 1);
	return 0;
}
