#include <iostream>
#include <map>
using namespace std;
map<int,int> mymap;
int n,m,i,p,q,r,x;
long long b[200000],c[200000];
main(){
 cin>>n;
 for(i=0;i<n;i++){
  
  cin>>x;
 if(mymap[x]==0) mymap[x]=1; 
  else mymap[x]++;
} 
 cin>>m;
 for(i=0;i<m;i++)
  cin>>b[i];
 for(i=0;i<m;i++)
  cin>>c[i];        
 for(i=0;i<m;i++)
  if(mymap[b[i]]>p ||( mymap[b[i]]==p && mymap[c[i]]>=q)){
        p=mymap[b[i]]; 
        q=mymap[c[i]]; 
        r=i+1;
        }

cout<<r<<endl;
//system("pause");
}
