#include <bits/stdc++.h>
using namespace std;

int n;
map <int,int> m;
int b[212345],c[212345];

int init(){
	cin>>n;
	for (int i=0;i<n;i++){
		int x;
		cin>>x;
		m[x]++;
	}
	int mm;
	int mx = 0;
	cin>>mm;
	for (int i=0;i<mm;i++){
		cin>>b[i];
	}
	for (int i=0;i<mm;i++){
		cin>>c[i];
	}
	for (int i=0;i<mm;i++){
		if (m[b[i]]>m[b[mx]]) mx = i;
		else if (m[b[i]]==m[b[mx]]){
			if (m[c[i]]>m[c[mx]]) mx = i;
		}
	}
	cout << mx+1;
	return 0;	
}

int main() {
	init();
	return 0;
}
