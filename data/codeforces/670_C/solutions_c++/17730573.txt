//in the name of God

#include <iostream>
#include <map>
using namespace std;

int n, m, b[200200], c[200200];
map<int, int> l;

int main()
{
	cin >> n;
	for(int i=0; i<n; i++)
	{
		int x;
		cin >> x;
		l[x]++;
	}

	cin >> m;
	for(int i=0; i<m; i++)
		cin >> b[i];
	for(int i=0; i<m; i++)
		cin >> c[i];

	int best = 0;
	for(int i=1; i<m; i++)
	{
		if(l[b[i]] > l[b[best]])
			best = i;
		else if(l[b[i]] == l[b[best]] && l[c[i]] > l[c[best]])
			best = i;
	}

	cout << best+1 << endl;	

	return 0;
}
