#include <iostream>
#include <map>

using namespace std;

map <int, int> count;

int n,a,b[300000],c[300000],m;

int main()
{	
	cin>>n;
	for (int i = 0; i < n; i++)
	{
		cin>>a;
		count[a]++;
	}
	cin>>m;
	for (int i = 0; i < m; i++)
		cin>>b[i];
	for (int i = 0; i < m; i++)
		cin>>c[i];
	int maxi = 0;
	for (int i = 1; i < m; i++)
	{
		if (count[b[i]] > count[b[maxi]])
			maxi = i;
		else
			if (count[b[i]] == count[b[maxi]] && count[c[i]] > count[c[maxi]])
				maxi = i;
	}
	cout<<maxi+1;
	return 0;
}
