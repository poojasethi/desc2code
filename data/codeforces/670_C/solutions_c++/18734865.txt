#include <iostream>
#include <map>
using namespace std;
int n, m, i, b[200001], c[200001], ai, m1, m2, m3;
map <int, int> a;

int main(){
	for(cin >> n; n; n--) {
		cin >> ai;
		a[ai]++;		
	}
	cin >> m;
	for(i = 0; i < m; i++) cin >> b[i];
	for(i = 0; i < m; i++) cin >> c[i];
	for(i = 0; i < m; i++) {
		if(a[b[i]] > m1 || a[b[i]] == m1 && a[c[i]] > m2) {
			m1 = a[b[i]];
			m2 = a[c[i]];
			m3 = i;
		}
	}	
	cout << m3 + 1;
}
