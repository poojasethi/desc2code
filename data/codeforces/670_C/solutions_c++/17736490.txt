# include <bits/stdc++.h>

using namespace std;

map<int,int>mp;

int n,a[200001],m,b[200001],c[200001],imx=0;
int main ()
{
scanf("%d",&n);
for(int i=0;i<n;i++)
{
scanf("%d",&a[i]);
mp[a[i]]++;
}
scanf("%d",&m);
for(int i=0;i<m;i++)
scanf("%d",&b[i]);
for(int i=0;i<m;i++)
scanf("%d",&c[i]);
for(int i=0;i<m;i++)
{
if(mp[b[imx]]<mp[b[i]] || mp[b[imx]]==mp[b[i]] && mp[c[imx]]<mp[c[i]])
imx=i;
}
printf("%d",imx+1);
return 0;
}
