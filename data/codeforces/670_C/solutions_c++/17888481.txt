#include<map>
#include<iostream>
#include<cstdio>
using namespace std;

map<int, int> mp;
int n, a;
pair<int, int> b[200005];

int main(){
	scanf("%d",&n);
	for(int i = 0; i < n; i++)
    {
		scanf("%d",&a);
		mp[a]++;
	}
	scanf("%d",&n);
	int res = 0;
	for(int i = 0; i < n; i++)
	{
		scanf("%d",&b[i].first);
		b[i].first = mp[b[i].first];
	}
	for(int i = 0; i < n; i++)
	{
		scanf("%d",&b[i].second);
		b[i].second = mp[b[i].second];
	}
	for(int i = 0; i < n; i++)
		if (b[res] < b[i]) res = i;
	printf("%d\n",res+1);
	return 0;
}
