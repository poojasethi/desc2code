#include <bits/stdc++.h>
using namespace std;
map<int,int> lang;
int L[300010], S[300010];
int main(){
	int n;
	scanf("%d", &n);
	for(int i=0; i<n; i++){
		int l;
		scanf("%d", &l);
		lang[l]++;
	}
	int m;
	scanf("%d", &m);
	for(int i=1; i<=m; i++) scanf("%d", &L[i]);
	for(int i=1; i<=m; i++) scanf("%d", &S[i]);
	int ans = 0, talk = -1, sub = -1;
	for(int i=1; i<=m; i++){
		if(lang[L[i]] > talk || (lang[L[i]] == talk && lang[S[i]] > sub)){
			talk = lang[L[i]];
			sub = lang[S[i]];
			ans = i;
		}
	}
	printf("%d\n", ans);
	return 0;
}
