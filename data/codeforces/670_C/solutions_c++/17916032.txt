#include<bits/stdc++.h>


using namespace std;

int main(){
map<int, int > nos;

int n,m;
cin>>n;

for(int i=0;i<n;i++){
int temp;
cin>>temp;
nos[temp]++;
}

cin>>m;
int b[m],c[m];

for(int i=0;i<m;i++){
cin>>b[i];
}

for(int i=0;i<m;i++){
cin>>c[i];
}

int bmax=-1,cmax=-1,index=-1;

for(int i=0;i<m;i++){
int bi=nos[b[i]],ci=nos[c[i]];
if(bi>bmax || (bi==bmax && ci>cmax))bmax=bi,cmax=ci,index=i+1;
}

printf("%d\n",index);



return 0;
}
