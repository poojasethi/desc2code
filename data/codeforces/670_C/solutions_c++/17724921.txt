#include <bits/stdc++.h>
using namespace std;

int N, M;
map<int, int> Mp;
int A[200001], B[200001];

pair<int, int> p(int x) { return pair<int, int>(Mp[A[x]], Mp[B[x]]); }

int main() {
	ios_base::sync_with_stdio(false); cin.tie(0);
	cin >> N; 
	for (int i = 0; i < N; ++i) {
		int in; cin >> in; ++Mp[in];
	}
	cin >> M;
	int ch = 1;
	for (int i = 1; i <= M; ++i) cin >> A[i];
	for (int i = 1; i <= M; ++i) cin >> B[i];
	for (int i = 1; i <= M; ++i) {
		if (p(ch) < p(i)) ch = i;
	}
	cout << ch << endl;
}
