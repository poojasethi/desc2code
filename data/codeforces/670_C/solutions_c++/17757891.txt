#include<bits/stdc++.h>

using namespace std;

map<long long ,long long> a;

int n,k,m,k1,k2,b[200001];

int main(){
    cin>>n;
    for(int i=1;i<=n;++i){
        int x;
        cin>>x;
        a[x]++;
    }
    cin>>m;
    for(int i=1;i<=m;++i) cin>>b[i];
    int m1=0,m2=0,t=1;
    for(int i=1;i<=m;++i)
    {
        int x,k1=0,k2=0;
        cin>>x;
        k1=a[b[i]];
        k2=a[x];
        if(k1>m1) m1=k1,m2=k2,t=i;
        else if(k1==m1&&k2>m2) m1=k1,m2=k2,t=i;
    }
    cout<<t;
}
