#include<bits/stdc++.h>
#define ll long long
using namespace std;
map<ll,ll> sc;
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	ll n;
	cin>>n;
	for(ll i = 0;i<n;i++){
		int x;
		cin>>x;
		sc[x]++;
	}
	ll m;
	cin>>m;
	ll b[m];
	ll c[m];
	pair<ll , pair<ll,ll> > a[m];
	for(int i = 0;i<m;i++) cin>>b[i];
	for(int i = 0;i<m;i++) cin>>c[i];
	for(int i = 0;i<m;i++){
		a[i] = make_pair(sc[b[i]],make_pair(sc[c[i]],i+1));
	}
	sort(a,a+m);
	cout<<(a[m-1].second).second;
}
