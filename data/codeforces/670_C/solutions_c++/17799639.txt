int n,m,i,j,satisfied,pleased,audio,subtitles,a;
pair <int,pair<int,int> > sci[200005];
map <int,int> x;

int main(){
	cin>>n;
	for(i=1; i<=n; i++){
		cin>>a;
		x[a]++;
	}
	cin>>m;
	for(i=0; i<m; i++){

		cin>>a;
		sci[i].second.first = x[a];
	}

}
