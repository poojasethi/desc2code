#include<bits/stdc++.h>
using namespace std;
map<long,long>s;
long a[201000],b[201000],n,i,a1,ma,ma1,ans;
int main()
{
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a1;
		s[a1]++;
	}
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	for(i=1;i<=n;i++)
	{
		cin>>b[i];
	}
	ma=s[a[1]];
	ma1=s[b[1]];
	ans=1;
	for(i=1;i<=n;i++)
	{
		if((s[a[i]]>ma)||((s[a[i]]==ma)&&(s[b[i]]>ma1)))
		{
			ma=s[a[i]];ma1=s[b[i]];ans=i;
		}
	}
	cout<<ans;
}
