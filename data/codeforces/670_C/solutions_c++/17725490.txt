#include <bits/stdc++.h>
#define maxn 200005
using namespace std;
map < int ,int > mp;
int n,m,i,j,k,d[maxn],e[maxn],f[maxn],id,temp1,temp2,a,b,c;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d",&d[i]);
		mp[d[i]]+=1;
	}
	scanf("%d",&m);
	for(i=1;i<=m;i++)
	scanf("%d",&e[i]);
	for(i=1;i<=m;i++)
	scanf("%d",&f[i]);
	temp1=temp2=-1;
	for(i=1;i<=m;i++){
		a=e[i];
		b=f[i];
		if(mp[a]>temp1){
			temp1=mp[a];
			temp2=mp[b];
			id=i;
		}
		else if(mp[a]==temp1 and mp[b] > temp2){
			id=i;
			temp2=mp[b];
		}
	}
	printf("%d\n",id);
	return 0;
}
