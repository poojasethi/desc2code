#include <iostream>
#include <map>

using namespace std;

map<int, int> a;
int ll1[200010], ll2[200010];

int main() {
	int n;
	cin >> n;
	while (n--) {
		int tmp; cin >> tmp;
		a[tmp]++;
	}
	cin >> n;
	int b1 = 0, b2 = 0, ans = 0;
	for (int i = 1; i <= n; ++i) cin >> ll1[i];
	for (int i = 1; i <= n; ++i) cin >> ll2[i];
	for (int i = 1; i <= n; ++i) {
		int l1 = ll1[i], l2 = ll2[i];
		if (a[l1] > b1 || (a[l1] == b1 && a[l2] >= b2)) {
			ans = i;
			b1 = a[l1];
			b2 = a[l2];
		}
	}
	cout << ans << endl;
}
