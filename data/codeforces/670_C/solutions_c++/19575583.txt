#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
	int N,i,A[200000],B[200000],C[200000],M;
	cin>>N;
	map<int,int> freq;
	for(i=0;i<N;i++)
	{
	    cin>>A[i];
	    freq[A[i]]++;
	}
	cin>>M;
	for(i=0;i<M;i++)
	    cin>>B[i];
	for(i=0;i<M;i++)
	    cin>>C[i];
	
	pair<int,int> score[200000];
	int maxS=0;
	for(i=0;i<M;i++)
	{
	    score[i].first=freq[B[i]];
	    score[i].second=freq[C[i]];
	    if(score[maxS]<score[i])
	        maxS=i;
	}
	cout<<maxS+1;
	    
	return 0;
}
