#include<bits/stdc++.h>
#define ll long long
using namespace std;
map<ll,ll>mp;
int main()
{
	ll n,i,no,m,maxi=0,maxii=0,index,b[200001],c[200001];
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>no;
		mp[no]++;
	}
	cin>>m;
	for(i=1;i<=m;i++)
	cin>>b[i];
	for(i=1;i<=m;i++)
	cin>>c[i];
	for(i=1;i<=m;i++)
	{
		if(mp[b[i]]>maxi || (mp[b[i]]==maxi && mp[c[i]]>maxii))
		{
			maxi=mp[b[i]];
			maxii=mp[c[i]];
			index=i;
		}
	}
	if(index==0)
	index=1;
	cout<<index;
}
