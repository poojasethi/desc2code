#include<stdio.h>
#include<map>

using namespace std;

int main()
{
	map<int, int> lang;
	int n,m, i, maxi=1, k=0, l=0;
	scanf("%d", &n);
	for(i=0; i<n; i++)
	{
		int a;
		scanf("%d", &a);
		lang[a]++;
	}
	scanf("%d", &m);
	int a[200001], s[200001];
	for(i=1; i<=m; i++)
		scanf("%d", &a[i]);
	for(i=1; i<=m; i++)
		scanf("%d", &s[i]);
	for(i=1; i<=m; i++)
	{
		if(lang[a[i]]>k || lang[a[i]]==k && lang[s[i]]>l)
		{
			maxi=i;
			k=lang[a[i]];
			l=lang[s[i]];	
		}	
	}
	printf("%d\n", maxi);	
}
