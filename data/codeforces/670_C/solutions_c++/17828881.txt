#include<bits/stdc++.h>
using namespace std;
map<int, int > m;
int  main() {
	int n;
	scanf("%d", &n);
	int t;
	for(int  i = 0; i < n; i++){ scanf("%d", &t); m[t]++;}
	int  ma;
	scanf("%d", &ma);
	int a[ma], b[ma];
	for(int i = 1; i <= ma; i++) scanf("%d", &a[i]);
	for(int i = 1; i <= ma; i++) scanf("%d", &b[i]);
	int  mak = 0, pl = 0, ind = 1;
	for(int  i = 1; i <= ma; i++) {
		int x = a[i], y = b[i];
		if(m[x]>mak or (m[x]==mak and m[y]>pl))	{ 
			ind = i;
			mak = m[x]; 
			pl = m[y];
		}
	}
	printf("%d\n", ind);
}
