n = input()
ar = map(int,raw_input().split())
ind = []

armin = min(ar)
for i in xrange(n):
    ar[i] -= armin
    if ar[i] == 0:
        ind.append(i)

cnt = n*armin
dif = 0
for i in xrange(1, len(ind)):
    dif = max(dif, ind[i] - ind[i-1] - 1)

tmp = n-ind[-1]-1
tmp += ind[0]

dif = max(dif, tmp)
cnt += dif
print cnt
