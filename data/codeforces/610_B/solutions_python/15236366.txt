n = input()
s = raw_input()
l = [int(x) for x in s.split(' ')]
index = []
mini = 2*(10**9)
for i in range(n):
	if l[i] < mini:
		mini = l[i]
		index = [i]
	elif l[i] == mini:
		index.append(i)
		
maxi = 0
for i in range(1,len(index)):
	if index[i] - index[i - 1] > maxi:
		maxi = index[i] - index[i - 1] - 1
		
temp = n - 1 - index[-1] + index[0]
if temp > maxi:
	maxi = temp
if len(index) == n:
	maxi = 0
print n * mini + maxi
