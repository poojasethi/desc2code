n = int(raw_input())
minW, mini = [], 10**9+7
num = map(int, raw_input().split())
for i in xrange(len(num)):
	mini = min(mini,num[i])
for i in xrange(n):
	if num[i] == mini:
		minW.append(i)
cnt = 0
for i in xrange(len(minW)):
	if i == len(minW)-1:
		cnt = max(cnt, (n-minW[i]-1+minW[0]))
	else:
		cnt = max(cnt, minW[i+1]-minW[i]-1)
ans = mini * n + cnt
print ans
