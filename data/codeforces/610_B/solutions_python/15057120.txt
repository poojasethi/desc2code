aaa = input()
ccc = map(int,raw_input().split(" "))
min = 100000000000
num = []
num.append(-1)
num.append(-1)
count = 0
for i in range(aaa):
	if ccc[i] <= min:
		if ccc[i] != min:
			num[0] = i
			num[1] = i
			count = 1
		else:
			num[1] = i
			count = count + 1
		min = ccc[i]

ppp = num[0]+(aaa - (num[1]+1))
qqq = 0
for i in range(aaa):
	if ccc[i] == min:
		ppp = max(ppp,qqq)
		qqq = 0
	else:
		qqq = qqq + 1
ppp = max(ppp,qqq)
print aaa*min + ppp
