n = int(raw_input())
paintList = map(int, raw_input().split())

minPaint = min(paintList)

mySum = 0
maxSum = 0
newPaintList = map(lambda x: x - minPaint, paintList)
zeroIndex = newPaintList.index(0)
if (zeroIndex + 1 == n):
	loopIndex = 0
else:
	loopIndex = zeroIndex + 1
while loopIndex != zeroIndex:
	val = newPaintList[loopIndex]
	if val == 0:
		if mySum > maxSum:
			maxSum = mySum
		mySum = 0
	else:
		mySum += 1
	loopIndex += 1
	if loopIndex == n:
		loopIndex = 0

if mySum > maxSum:
	maxSum = mySum

print minPaint * n + maxSum
