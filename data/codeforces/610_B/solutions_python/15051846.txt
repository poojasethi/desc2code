n = int(raw_input())

l = [int(_) for _ in raw_input().split()]
m = min(l)

tmp_len = 0
M = 0
s = 1
st = 0
for item in l:
    if item > m:
        tmp_len += 1
    else:
        if s:
            st = tmp_len
        M = max(M, tmp_len)
        tmp_len = 0
        s = 0

tmp_len = st

l.reverse()
for item in l:
    if item > m:
        tmp_len += 1
    else:
        break
M = max(M, tmp_len)

print m*len(l)+M
