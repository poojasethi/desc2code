#!/usr/bin/python
# -*- coding: UTF-8 -*-
while True:
	try:
		n=input()
	except:
		break
	a=map(int,raw_input().split())
	mi =a[0]
	flag=0;
	for i in range(0,n):
		if a[i]<mi:
			mi=a[i]
			flag=i
	lmx=0
	last=flag
	ans=0
	for i in range(1,n+1):
		x=flag+i
		if(x>=n):
			x-=n
		if(a[x]==mi):
			ll=x-last-1
			if(ll<0):
				ll+=n
			ans=max(ans,ll)
			last=x
	mi=long(mi)*long(n)
	ans=long(ans)+mi
	print ans
