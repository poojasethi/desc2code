from math import *
from Queue import *
from sys import *
from random import *




n = int(raw_input())
a = map(int, raw_input().split())
m = min(a)
index = []
for i in range(n):
    if a[i] == m:
        index.append(i)
if len(index) == 1:
    print(m*n+n-1)
else:
    d = 1
    for i in range(1,len(index)):
        d = max(d, index[i] - index[i-1])
    d = max(d, n+index[0]-index[-1])
    print(m*n+d-1)
