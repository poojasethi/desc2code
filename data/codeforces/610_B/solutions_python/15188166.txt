n=input()
li=map(int,raw_input().split())
m=min(li)
ind=[]
for i in xrange(n):
    if(li[i]==m):
        ind+=[i]
maximum=0
l=len(ind)
for i in xrange(1,l):
    if(ind[i]-ind[i-1]>maximum):
        maximum=ind[i]-ind[i-1]      
if(n-ind[l-1]+ind[0]>maximum):
    maximum=n-ind[l-1]+ind[0]
print n*m+maximum-1
