p = int(input())
minval = 10**9+1
minfirst = 0
minlast = 0
biggap = 0
for i,e in enumerate(map(int,raw_input().split())):
    if e < minval:
        minval = e
        minfirst = i
        minlast = i
    if e == minval:
        biggap = max(biggap, i - minlast - 1)
        minlast = i
biggap = max(biggap, p - minlast + minfirst - 1)

print(minval * p + biggap)
