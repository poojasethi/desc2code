n = int(raw_input())
a = []
b = [0]
c = [int(i) for i in raw_input().split()]
for i in xrange(n):
    a.append(c[i])
    if i != 0:
        if a[i] < a[b[0]]:
            b = [i]
        elif a[i] == a[b[0]]:
            b.append(i)
m = n-1-b[-1]+b[0]
l = b[-1]
for i in xrange(len(b)-1):
    k = b[i+1]-1-b[i]
    if k > m:
        m = k
        l = b[i]
print a[l]*n+m
