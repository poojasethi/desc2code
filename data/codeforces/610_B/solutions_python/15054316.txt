n=int(raw_input())
a=map(int,raw_input().split())
x=min(a)
z=n
if a.count(x)==n:
    print n*x
else:
    a=a+a
    n*=2
    i=c=s=0
    while(i<n-1):
        c=0
        k=i
        while(i<n-1 and a[i]!=x and a[(i+1)]!=x):
            c+=1
            i+=1
            
        if i!=k:
            i-=1
        if(c>s):
            s=c
        i+=1
        
    
    print (z*x)+s+1
