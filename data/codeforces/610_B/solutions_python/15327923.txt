n = input()
v = map(int, raw_input().split())
min_value = min(v)
start = -1
longest = 0;
for i in range(0, len(v)):
	if v[i] > min_value:
		if start == -1:
			start = i
	else:
		if start != -1:
			t = i - start
			if t > longest:
				longest = t
		start = -1

if start > 0:
	i = 0
	while v[i] > min_value:
		i = i + 1
	t = len(v) - start + i
	if t > longest:
		longest = t

ans = min_value * len(v) + longest
print ans
