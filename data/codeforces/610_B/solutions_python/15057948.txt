n = long(raw_input())
col = raw_input().split()
for i in range (n):
    col[i] = int (col[i])
m = min (col)
ml = [];
for i in range (n):
    if (col[i] == m):
        ml.append(i)
maxd = 0;
for i in range (1, len (ml)):
    d = ml[i] - ml [i - 1] - 1
    if (d > maxd):
        maxd = d
d = ml[0] - ml [-1] + n - 1
if (d > maxd):
    maxd = d
res = n * m + maxd
print res
