n=input()
a=map(int,raw_input().split())
m = 100100100100
for i in a:
    if i<m:
        m=i
if m==a[0]:
    f=0
    l=0
else:
    f=-1
    l=-1
r=0
for i in xrange(0,n):
     if f == -1 and a[i] == m:
        f=i
        l=i
     if a[i]==m:
        r = max(r,i-l-1)
        l=i
r=max(r,n-(l-f+1))
print n*m+r
