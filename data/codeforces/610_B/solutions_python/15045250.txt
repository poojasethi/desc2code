#!/usr/bin/env python
# encoding: utf-8

n = int(raw_input())
a = map(int, raw_input().split())

m = min(a)
ans = m * n
b = []
for i in xrange(n):
    if a[i] == m:
        b.append(i)

t = len(b)
tmp = 0
for i in xrange(1, t):
    tmp = max(b[i]-b[i-1]-1, tmp)
if t > 0:
    tmp = max(tmp, b[0]+(n-1-b[t-1]))
print ans + tmp
