import sys

N = int(sys.stdin.readline())
As = map(int, sys.stdin.readline().split())

m = min(As)
res = m * N
max_len = 0
for i, _ in enumerate(As):
    As[i] -= m
    if As[i] == 0:
        start = i

idx = (start + 1) % N
cnt = 0
while idx != start:
    if As[idx] == 0:
        cnt = 0
    else:
        cnt += 1
        max_len = max(max_len, cnt)

    idx += 1
    if idx >= N:
        idx = 0

res += max_len
print res
