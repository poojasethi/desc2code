t=input()
a=map(long,raw_input().split())
m=min(a)

if t==1:
    print a[0]
    exit(0)
    
j=0
dist=[0]*t

for i in range(len(a)):
    if a[i]==m:
        j=i
        break

for i in range(len(a)-1,-1,-1):
    if a[i]==m:
        dist[i]=len(a)-i+j

for i in range(j+1,len(a)):
    if a[i]==m:
        dist[j]=i-j
        j=i

l=max(dist)
i=dist.index(l)


print (a[i]*len(a)) + l -1
