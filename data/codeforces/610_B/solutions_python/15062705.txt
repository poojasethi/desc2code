i = map(int, raw_input().split())
m = min(i)
l = [0]*len(i)
if i[0] > m:
    l[0] = 1
for j in xrange(1, len(i)):
    if i[j] > m:
        l[j] = l[j-1]+1
    else:
        l[j] = 0
if i[0] > m:
    l[0] = l[-1] + 1
for j in xrange(1, len(i)):
    if i[j] > m:
        l[j] = l[j-1]+1
    else:
        l[j] = 0
print len(i)*m + max(l)
