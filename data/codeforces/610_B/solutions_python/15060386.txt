n = int(raw_input())
A = [int(x) for x in raw_input().split()]

min_A = min(A)

ans = n * min_A

A = A + A

i = A.index(min_A)
right = i

gap = 0

while i < 2 * n:
    left = right
    while i < 2 * n:
        i += 1
        if i < 2 * n :
            if A[i] == min_A:
                break
    right = i
    if i < 2 * n:
        gap = max((right - left - 1) , gap)

print ans + gap
