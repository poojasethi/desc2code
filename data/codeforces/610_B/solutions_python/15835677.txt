n = int(raw_input())
a = [int(x) for x in raw_input().split()]
minx = a[0]
for i in range(n):
	minx = min(a[i],minx)
for i in range(n):
	a[i] = a[i] - minx
ans = minx * n
tmax = 0
now = 0
a = a + a
for i in range(len(a)):
	if (a[i]):
		now += 1
	else: now = 0
	tmax = max(tmax,now)

print ans + tmax
