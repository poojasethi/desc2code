n = int(raw_input())
a = map(int, raw_input().split(' '))

mina = min(a)
sqmax = n*mina
nw = a[a.index(mina)+1:] + a[:a.index(mina)]

gap = 0
start = False
l = r = 0
for i in xrange(len(nw)):
	if nw[i] > mina:
		if start == False:
			l = i
			start = True
		r = i
	if (nw[i] == mina or i == len(nw) - 1) and start == True:
		gap = max(r - l+1, gap)
		start = False

print sqmax + gap
