n = int(input())
A = map(int, raw_input().split())
r_min = 2E9
rin = 0
for i, a in enumerate(A):
    if a <= r_min:
        r_min = a
        rin = i
max_count = 0
count = 0
for i in range(n):
    j = i + rin + 1
    if j >= n:
        j -= n
    if A[j] == r_min:
        if count > max_count:
            max_count = count
        count = 0
    else:
        count += 1
    
print len(A)*r_min + max_count
