n = input()
a = map(int, raw_input().split())
s = 0
minimum = min(a)
a = map(lambda x: x-minimum, a)
s += minimum*n
#print a, s
c = 0
max_c = 0
for j in xrange(2):
    for i in xrange(n):
        if a[i] > 0:
            c += 1
            #print "(%d, %d), %d, %d" % (i, j, a[i], c)
        else:
            if c > max_c:
                max_c = c
                #print "(%d, %d), a=%d, max_c=%d" % (i, j, a[i], max_c)
            c = 0
s += max_c
print s
