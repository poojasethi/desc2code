def vikaAndSquares(colors):
	minLitre = min(colors)
	return minLitre * len(colors) + findMaxSequenceLagerthanN(colors, minLitre)

def findMaxSequenceLagerthanN(nums, minimal):
	tmp, ans, length = 0, 0, len(nums)
	for i in range(2 * length):
		if(nums[i % length] > minimal):
			tmp += 1
		else:
			if(tmp > ans):
				ans = tmp
			if(i >= length):
				break
			tmp = 0
	return ans
