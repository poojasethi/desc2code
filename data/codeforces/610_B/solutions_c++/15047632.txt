#include <iostream>
#include <cmath>

using namespace std;

typedef long long int ll;

ll n;
int v[1000000];

int main()
{
  ios::sync_with_stdio(false);
  cin>>n;
  int minimo=1000000000;
  for (int i=0;i<n;i++) {
    cin>>v[i];
    v[i+n]=v[i];
    minimo=min(minimo,v[i]);
  }
  int last=2*n-1;
  int maxi=0;
  for (int i=2*n-1;i>=0;i--) {
    if (v[i]==minimo) last=i;
    if (i<n) maxi=max(maxi,last-i);
  }
  cout<<n*minimo+maxi<<endl;
}
