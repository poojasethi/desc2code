#include <stdio.h>
long long l,min;
int main()
{
int n,a[200005],i,t1,tp,max=1;
scanf("%d",&n);
scanf("%d",&a[1]);
min=a[1];
for(i=2;i<=n;i++)
{
	scanf("%d",&a[i]);
	if(a[i]<min)
		min=a[i];
}
for(i=1;i<=n;i++)
{
	if(a[i]==min)
	{
		t1=i;break;
	}
}
tp=t1;
for(i=t1+1;i<=n;i++)
{
	if(a[i]==min)
	{
		if(i-tp>max)
			max=i-tp;
		tp=i;
	}
}
if(t1+n-tp>max)
	max=t1+n-tp;
l=min*n+max-1;
printf("%lld\n",l);
return 0;
}
