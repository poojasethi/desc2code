#include<bits/stdc++.h>
using namespace std;
long long int n, a[511111], mn = 1000000007, mx;
int main()
	{
	cin >> n;
	for(int i = 1; i <= n; i++)
		{
		cin >> a[i];
		a[i + n] = a[i];
		mn = min(a[i], mn);
		}
	int x = 0;
	for(int i = 1; i <= 2 * n; i++)
		if(a[i] == mn)
			{
			if(x != 0)
				mx = max((i - x * 1ll), mx);
			x = i;
			}
	cout << n * 1ll * mn + mx - 1;
	}
