#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[400010],n,mv=1000000001,len,tmp;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
		scanf("%d",&a[i]),mv=min(mv,a[i]),a[i+n]=a[i];
	for(int i=1;i<=2*n;++i)
	{
		a[i]-=mv;
		if(a[i])++tmp,tmp=min(tmp,n);
		else tmp=0;
		len=max(len,tmp);
	}
	printf("%I64d\n",1ll*mv*n+len);
}
