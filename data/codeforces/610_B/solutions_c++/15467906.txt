#include <bits/stdc++.h>
using namespace std;
long long n, a[500001], x, mx;
int main()
{
	cin >> n;
	long long mn = 1e9 + 9;
	for (int i = 0; i < n; ++ i) {
		cin >> a[i];
		mn = min(mn, a[i]);
	}
	for (int i = 0; i < (2 * n); ++ i) {
		if (a[i % n] != mn) {
			++ x;
		}
		else {
			mx = max (mx, x);
			x = 0;
		}
	}
	cout << mn * n + mx;
	return 0;
}
