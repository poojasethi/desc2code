#include<bits/stdc++.h>
using namespace std;
long long a[401000],n,ans,z,j,k,p;
long i;
int main()
{
	cin>>n;k=1000001000;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		if(k>a[i]) k=a[i];
	}
	ans=k*n;
	for(i=1;i<=n;i++)
	{
		a[i]-=k;
		a[i+n]=a[i];
	}
	j=1;z=0;
	for(j=1;j<=n;j++)
	{
		z=0;
		if(a[j-1]!=0) continue;
		while(a[j]!=0)
		{
			j++;z++;
		}
		if(z>p) p=z;
	}
	ans+=p;
	cout<<ans;
}
