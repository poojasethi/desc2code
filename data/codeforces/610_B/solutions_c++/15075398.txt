#include <iostream>
using namespace std;
typedef __int64 ull;

int main() {
	ull n, i, m, a[200000], k;
	cin >> n;
	for(i = 0; i < n; i++) cin >> a[i];
	m = a[0];
	for(i = 1; i < n; i++) m = min(m, a[i]);
	k = n * m;
	int j = 2, x = 0, p = 0;
	while(j) {
		for(i = 0; i < n; i++) {
			if(a[i] == m) {
				x = max(p, x);
				p = 0;
			}
			else p++;
		}
		j--;
	}
	k += x;
	cout << k;
}
