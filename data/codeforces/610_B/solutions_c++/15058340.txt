#include<cstdio>
int n,a[400001];
int main()
{
    int i,j,k;
    scanf("%d",&n);
    for(i=1;i<=n;i++){scanf("%d",&a[i]);a[i+n]=a[i];}
    int mn=1<<30;
    for(i=1;i<=n;i++)if(a[i]<mn)mn=a[i];
    int mx=0;
    for(i=1,j=0;i<=n+n;i++)if(a[i]==mn)
    {
        if(i-j-1>mx)mx=i-j-1;
        j=i;
    }
    printf("%I64d\n",1ll*mn*n+mx);
    return 0;
}
