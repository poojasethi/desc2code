#include<iostream>
using namespace std;
int main()
{long long int n;
cin>>n;long long int a[n];
long long int p1,p2,count=0,t,min=2000000000,max=-1;
for(long long int i=0;i<n;i++)
{cin>>a[i];
if(a[i]<min)
{min=a[i];p1=p2=i;
}
else if(a[i]==min)p2=i;
}
max=p1+n-p2-1;
for(long long int i=p1+1;i<=p2;i++)
{if(a[i]==min)
{if(count>max)max=count;count=0;
}
else count++;
}
t=n*min+max;
cout<<t;
}
