#include<bits/stdc++.h>
using namespace std;

#define l long long
#define pb push_back
#define ii pair<int,int>

int main()
{
    int n;cin>>n;
    l ar[n];
    for(int i=0;i<n;i++)cin>>ar[i];

    l min_val=*min_element(ar,ar+n),m=0,c=0;

    for(int i=0;i<2*n;i++)
    {
        if(ar[i%n]>min_val)
        {
            c++;
        }
        else c=0;
        m=max(m,c);
    }
    cout<<min_val*n+m;
}
