#include <iostream>
#include <cstdio>
using namespace std;

long long n,s,cnt,mn=9e9,a[200005];
int main(){
    int i;
    cin>>n;
    for(i=0; i<n; i++) {
        scanf("%d",&a[i]);
        mn = min(mn,a[i]);
    }
    for(i=0; i<2*n; i++) {
        if(a[i%n]==mn) {
            s = max(s,cnt);
            cnt = 0;
        } else cnt++;
    }
    cout<<s+mn*n;
    return 0;
}
