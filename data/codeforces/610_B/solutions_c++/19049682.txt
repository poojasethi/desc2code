#include <cstdio>

using namespace std;

#define NMAX 200000

int m[NMAX*2];

int main(){
	int n;
	int mi=2e9,mx=-1;
	int i,j;

	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",m+i);
		m[i+n]=m[i];
		if(m[i]<=mi)mi=m[i];
	}

	for(i=0;i<n;i++){
		if(m[i]==mi){
			for(j=i+1;m[j]!=mi;j++);
			if(j-i-1>mx)mx=j-i-1;
			i=j-1;
		}
	}
	printf("%I64d",mi*(long long)n+mx);
	return 0;
}
