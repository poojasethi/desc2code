#include <iostream> 
using namespace std ;
long long x=0,ans=0,mn=1000000010,n,a[400010];

int main () 
{
cin >> n ;


for (int i=1;i<=n;i++) 
{
cin >> a[i];
a[i+n]=a[i];
if (a[i]<mn)
mn=a[i];
} 

for (int i=1;i<=n+n;i++) 
{
if (a[i]==mn) 
x=0;
else
x++; 
if (x>ans)
ans=x;
} 
cout << ans+mn*n;






return 0;
}
