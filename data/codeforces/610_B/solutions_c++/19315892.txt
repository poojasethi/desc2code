#include <iostream>
#include <stdio.h>
#define ll long long
using namespace std;
int main(){
	ll n,MIN=0x7f7f7f7f,cnt=0,num=0;
	int a[200005],pos;
	cin>>n;
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		if(MIN>a[i]){
			MIN=a[i];
			pos=i;
		}
	}
	for(int i=pos;i<=pos+n;i++){
		if(a[i%n]>MIN) cnt++;
		if(a[i%n]==MIN) num=max(num,cnt),cnt=0;
	}
	cout<<MIN*n+num<<endl;
}
