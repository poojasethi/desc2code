#include <stdio.h>
using namespace std;

int main() {

	int N, j, pos, min = 1000000001, maxDist = -1;
	scanf("%d\n", &N);

	for (int X, i = 0; i < N; i++) {

		scanf("%d", &X);
	
		if (X < min) {
			j = i;
			min = X;
			pos = i;
			maxDist = -1;
		} else if (X == min) {
			if (i - j > maxDist)
				maxDist = i - j;
			j = i;
		}
	}

	if (N - j + pos > maxDist)
		maxDist = N - j + pos;

	printf("%I64d\n", (long long)N*min + maxDist - 1);
}
