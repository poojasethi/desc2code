#include<stdio.h>
long long a[400005];
int main()
{
	long long n,rs,p;
	scanf("%lld",&n);
	rs=1000000001;
	for(int i=1;i<=n;++i)
	{
		scanf("%lld",&a[i]);
		if(a[i]<rs)rs=a[i];
	}p=rs;
	for(int i=1;i<=n;++i)
	a[i]-=rs,a[i+n]=a[i];
	rs=0;
	int u=0;
	for(int i=1;i<=2*n;++i)
	if(a[i]==0)
	{
		if(u>rs)rs=u;
		u=0;
	}
	else u++;
	if(rs>n)rs=n;
	rs+=(n*p);
	printf("%lld\n",rs);
}
