#include<cstdio>
#include<algorithm>
using namespace std;

int n, mi, ma, cur;
int v[444444];

int main() {
	scanf("%d", &n);
	mi = 1111111111;
	for(int i=0;i<n;i++) {
		scanf("%d", &v[i]);
		v[i+n] = v[i];
		mi = min(mi, v[i]);
	}
	ma = cur = 0;
	for(int i=0;i<n+n;i++) {
		if(v[i]>mi)cur++;
		else cur=0;
		ma = max(ma, cur);
	}
	printf("%I64d\n", 1LL*mi*n+ma);
	
	return 0;
}
