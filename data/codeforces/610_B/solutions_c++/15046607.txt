#include<cstdio>
#define maxn 200010
int n,a[maxn];

int main()
{
	scanf("%d",&n);
	for(int i=1; i<=n; i++) scanf("%d",&a[i]);
	int mi=a[1],pla0=1,pla1=1;
	long long ans=0;
	for(int i=2; i<=n; i++)
	{
		if(a[i]<mi) {ans=0; pla0=pla1=i; mi=a[i];}
		else if(a[i]==mi)
		{
			if(i-pla1-1>ans) ans=i-pla1-1;
			pla1=i;
		}
	}
	if(pla0-1+(n-pla1)>ans) ans=pla0-1+n-pla1;
	ans=ans+1ll*mi*n;
	printf("%I64d\n",ans);
	return 0;
}
