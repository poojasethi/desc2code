#include <cstdio>

using namespace std;

#define NMAX 200000

long long m[NMAX*2];

int main(){
	int n;
	long long mi=2e9,mx=-1;
	long long ans;
	int i,j;

	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%I64d",m+i);
		m[i+n]=m[i];
		if(m[i]<=mi){
			mi=m[i];
		}
	}

	for(i=0;i<n;i++){
		if(m[i]==mi){
			for(j=i+1;m[j]!=mi;j++);
			if(j-i-1>mx)mx=j-i-1;
			i=j-1;
		}
	}
	printf("%I64d",mi*n+mx);
	return 0;
}
