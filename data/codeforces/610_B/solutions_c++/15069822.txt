#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n, x, a[200005];
	cin>>n;
	for(int i=0 ; i<n ; i++)
	{
		cin>>a[i];
	}
	x = (long long int)*min_element(a,a+n);
	int cnt=0, maxcnt=-1;
	for(int j=0,i=0 ; j<2*n ; j++,i++,i=i%n)
	{
		if(a[i]!=x)
			cnt++;
		else
			cnt=0;
		maxcnt = max(cnt,maxcnt);
	}
	cout<<x*n+maxcnt<<endl;
}
