#include <bits/stdc++.h>

using namespace std;
typedef int64_t ll;
int main()
{
    long long n,x,mi=2e9,vmin,vmax;
    scanf("%I64d",&n);
    ll res=0;
    for(int i=1;i<=n;i++)
    {
        scanf("%I64d",&x);
        if(mi>x) {mi=x;vmin=i;res=0;vmax=i;}
        if(mi==x) {if(res<i-vmax-1) res=i-vmax-1;vmax=i;}
    }
    if(res<n-vmax+vmin-1) res=n-vmax+vmin-1;
    res+=mi*n;
    printf("%I64d",res);
}
