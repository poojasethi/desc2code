#include<bits/stdc++.h>
using namespace std;
long long x=0,ans=0,mn=1000000010,n,a[400010];
int main(){
	scanf("%I64d",&n);
	for (int i=1; i<=n; i++){
	    scanf("%I64d",&a[i]);
	    a[i+n]=a[i];
	    if (a[i]<mn) mn=a[i];
	}
	for (int i=1; i<=2*n; i++){
		if (a[i]==mn) x=0;
	    else x++;
	    if (x>ans) ans=x;
	}
	ans=ans+mn*n;
	printf("%I64d",ans);
}
