#include <iostream>

using namespace std;
typedef long long int ll;
int main()
{
ll n;
cin>>n;
ll t[n];
cin>>t[0];
ll m=t[0]; ll d1=0,d=0,p=0;
for (ll i=1;i<n;i++)  {
    cin>>t[i];
    if (t[i]<=m) {m=t[i]; p=i;}
 }
 ll l;
 for (ll i=p; i<n+p; i++)
 {
     if (i>=n) {l=i-n;} else {l=i;}
     if (t[l]>m)  {d1++;}
 else if (t[l]=m) { if (d<d1) { d=d1;} d1=0; }
 }
 if (d<d1) { d=d1;}
cout<<m*n+d ;
    return 0;
}
