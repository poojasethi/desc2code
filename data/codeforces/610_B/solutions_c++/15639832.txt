#include <stdio.h>
long long int a[400010];
int main()
{
	long long int n,n1,min,max=0,sum=0;
	scanf("%lld",&n);
	for(n1=0;n1<n;n1++)
	{
		scanf("%lld",&a[n1]);
		a[n+n1]=a[n1];
		if(n1==0) min=a[n1];
		else if(a[n1]<min) min=a[n1];
	}
	for(n1=0;n1<2*n;n1++)
	{ 
		if(a[n1]!=min) max++;
		else if(a[n1]==min)
		{
			if(max>sum) sum=max;
			max=0;
		}
	}
	printf("%lld\n",min*n+sum);
	return 0;
}
