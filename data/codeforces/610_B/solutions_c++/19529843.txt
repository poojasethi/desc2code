#include<bits/stdc++.h>
#define Rep(i,n) for(int i = 1;i <= n;i ++)
using namespace std;
const int N = 200005;
int a[N],mn = (int)1e9,pos[N],n;
int main ()
{
	scanf("%d",&n);
	int cnt = 0;
	Rep(i,n)scanf("%d",&a[i]),mn = min(a[i],mn);
	Rep(i,n)if(a[i] == mn)pos[++ cnt] = i;
	pos[cnt + 1] = n + pos[1];
	int tmp = 0;
	Rep(i,cnt)tmp = max(tmp,pos[i + 1] - pos[i] - 1);
	printf("%lld\n",1ll * n * mn + tmp);
	return 0;
}
