#include <iostream>
#include <algorithm>
#include <map>
#include <vector>
#define N 200005

using namespace std;
map<long long,long long> mp;

long long n,X,m,p,mx=10000000000000,k,ans,A[N];
int main(){
cin>>n;
for(int i=0;i<n;i++)
	cin>>A[i],mx=min(mx,A[i]);
	
for(int i=0;i<n;i++)
if(A[i]==mx)mp[X]=i,X++;
//X=n-X;
for(int i=0;i<X-1;i++)
	m=max(m,mp[i+1]-mp[i]-1);
	
cout<<mx*n + max(m,n-mp[X-1]+mp[0]-1);
}
