#include <iostream>
using namespace std;
int main()
{
	int n,t,minid,fst;
	long long mmin=(long long)1e10,mmx=0,ans=0;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>t;
		if(t<mmin)
		{
			mmin=t;
			fst=minid=i;
			mmx=0;
		}
		else if(t==mmin)
		{
			mmx=max(mmx,(long long)i-minid-1);
			minid=i;
		}
	}
	mmx=max(mmx,(long long)fst-minid+n-1);
	ans=n*mmin+mmx;
	cout<<ans<<endl;
	return 0;
}
