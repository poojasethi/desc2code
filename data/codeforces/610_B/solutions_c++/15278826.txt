#include<iostream>

using namespace std;

int main(){
int i,n,d=0,l,ll;
cin>>n;
long long int x=1000000000,a[n],ans;

for(i=0;i<n;i++){
    cin>>a[i];
    x=min(x,a[i]);
}

for(i=0;i<n;i++){
    if(a[i]==x){
        l=i;
        break;
    }
}
ll=l;

for(i=0;i<n;i++){
    if(a[i]==x){
        d=max(d,i-l-1);
        l=i;
    }
}
d=max(d,n-1-l+ll);
ans=x*n;
ans+=d;
cout<<ans;
return 0;
}
