#include <bits/stdc++.h>
using namespace std;
int main() {
	long long int ne,ma=0,rem,i,j,k,mi=2e9,n,f=0,a[200005];
	cin>>n;
	for(i=0;i<n;i++) {
		cin>>j;
		a[i] = j;
		if(j<mi)
			mi = j;
	}
	for(i=0;i<n;i++)
		if(mi==a[i])
			if(!f) {
				f=1;
				rem = i;
				ne = i;
				k = i;
			}
			else {
				ma = max(ma,i-rem-1);
				rem = i;
				k = rem;
			}
	cout<<mi*n+max(n-k-1+ne,ma)<<endl;
	return 0;
}
