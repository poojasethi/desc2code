#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

long long a[200000];

int main()
{
	long long n, i, min, cnt = 0, mr = 0;
	cin >> n >> a[0];
	min = a[0];
	for (i = 1; i < n; i++)
	{
		cin >> a[i];
		if (a[i] < min)
			min = a[i];
	}
	for (i = 0; i < 2 * n; i++)
	{
		if (a[i % n] == min)
		{
			cnt = 0;
		}
		else
		{
			cnt++;
			if (cnt > mr)
				mr = cnt;
		}
	}
	cout << min * n + mr << endl;
	return 0;
}
