#include <bits/stdc++.h>

using namespace std;

#define ll long long int

int main()
{
	ios::sync_with_stdio(false);
	ll n,i,j,k,ans=0;
	cin>>n;
	ll A[n];
	ll Min=10000000000;
	for(i=0;i<n;i++){
		cin>>A[i];
		Min=min(Min,A[i]);
	}
	i=0;
	while(i<n && A[i]!=Min) i++;
	j=i;
	k=j;
	for(;i<n;i++)
	{
		if(A[i]==Min) 
		{
			ans=max(ans,i-k-1);
			k=i;
		}
	}
	ans=max(ans,j+(n-1-k));
	cout<<(Min*n+ans)<<endl;
	return 0;
}
