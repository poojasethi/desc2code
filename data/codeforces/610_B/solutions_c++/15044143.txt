#include<cstdio>
using namespace std;
int n,max,s,ans;
int a[400005];
int main(){
    scanf("%d",&n);
    ans=1000000000;
    for(int i=1;i<=n;i++){
        scanf("%d",&a[i]);
        a[i+n]=a[i];
        if(a[i]<ans)ans=a[i];
    }
    for(int i=1;i<=2*n;i++){
        if(a[i]>ans){
            s++;
            if(max<s)max=s;
        }else s=0;
    }
    printf("%I64d",1LL*ans*n+max);
}
