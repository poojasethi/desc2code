#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{ll n,t;ll m=INT_MAX; ll i,r,l,maxx=0;
cin>>n;
ll a[n+1];
for(i=1;i<=n;i++)
{cin>>a[i];
	m=min(m,a[i]);
}
for(i=1;i<=n;i++)
if(m==a[i])
{t=l=i; break;}
for(i=i+1;i<=n;i++)
if(m==a[i])
{maxx=max(maxx,i-t-1);
t=i;
}
maxx=max(maxx,n-t+l-1);
cout<<m*n+maxx;
return 0;
}
