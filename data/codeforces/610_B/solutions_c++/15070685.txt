#include <bits/stdc++.h>
using namespace std;
int a[400010];

int main() {
	int n, min = (1<<31) - 1, maxdis = 0, x = 0;
	scanf ("%d", &n);
	for (int i = 0; i < n; i++){
		scanf ("%d", &a[i]);
		a[n + i] = a[i];
		if (a[i] < min) min = a[i];
	}
	for (int i = 0; i < 2 * n; i++){
		if (a[i] == min) x = 0;
		else x++;
		maxdis = max(x, maxdis);
	}
	printf ("%lld\n", 1LL * n * min + 1LL * maxdis);
	return 0;
}
