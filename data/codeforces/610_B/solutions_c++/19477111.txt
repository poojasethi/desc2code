#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
int n,a[400001];
long long ans=1000000000,maxi=-1;
int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
		a[i+n]=a[i];
		if(a[i]<ans){
			ans=a[i];
		}
	}
	long long cnt=0;
	for(int i=1;i<=2*n;i++){	
		if(a[i]>ans){
			cnt++;	
		}
		else{
			maxi=max(maxi,cnt);
			cnt=0;
		}
	}
	cout<<(ans*n)+maxi;
}
