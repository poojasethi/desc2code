#include<iostream>
using namespace std;

bool col[1010],row[1010];

int main()
{
	int n,m,x,y,ans=0;
	cin>>n>>m;
	for(int i=1;i<=m;i++){
	   cin>>x>>y;
	   col[x]=true;
	   row[y]=true;
	}
	for(int i=2;i<n;i++)
		ans+=2-(col[i]+row[i]);
	ans-=(n&1)&(!col[n/2+1])&(!row[n/2+1]);
	cout<<ans;
	return 0;
}
