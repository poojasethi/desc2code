#include <bits/stdc++.h>
using namespace std;

int main ( )
{
	int n, m, cnt=0, i, x, y;
	cin >> n >> m;
	vector<int> r(n+1,1), c(n+1,1);
	for ( i = 0; i < m; ++i ) {
		scanf ( "%d%d", &x, &y );
		r[x]=c[y]=0;
	}

	for ( i = 2; i < n; ++i )
		cnt += r[i]+c[i];

	if ( r[n/2+1] && c[n/2+1] && n%2 == 1 ) cnt--;
	cout << cnt << endl;
	return 0;
}
