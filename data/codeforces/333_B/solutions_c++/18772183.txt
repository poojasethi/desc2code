#include<bits/stdc++.h>
using namespace std;
int x[1010],y[1010],cnt;
int main(){
    int n,m,xx,yy;
    cin>>n>>m;
    for(int i=1;i<=n;i++)x[i]=y[i]=1;
    for(int i=1;i<=m;i++){
        scanf("%d%d",&xx,&yy);
        x[xx]=0;
        y[yy]=0;
    }
    for(int i=2;i<=n/2;i++){
        cnt+=x[i]+x[n-i+1]+y[n-i+1]+y[i];
    }
    if(n%2==1)cnt+=x[n/2+1]||y[n/2+1];
    cout<<cnt<<endl;

}
