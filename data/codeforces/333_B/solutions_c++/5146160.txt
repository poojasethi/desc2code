#include<stdio.h>
#include<string.h>
int n,m;
int r[2010],c[2010];
int ans;
int main(){
	int i,x,y;
	scanf("%d%d",&n,&m);
	for (i=0;i<m;i++){
		scanf("%d%d",&x,&y);
		r[x]=r[x+n]=1;
		c[y]=c[y+n]=1;
	}
	ans=0;
	for (i=2;i<n;i++){
		if (r[i]==0) ans++;
		if (c[i]==0) ans++;
	}
	if (n%2==1&&r[n/2+1]==0&&c[n/2+1]==0) ans--;
	printf("%d\n",ans);
	return 0;
}
