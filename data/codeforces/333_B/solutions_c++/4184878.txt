#include<cstdio>
int n,m;
int r[1010],c[1010];

int main()
{
    int i,j,x,y,ans=0;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i)
    {
        scanf("%d%d",&x,&y);
        r[x]=c[y]=1;
    }
    for(i=2;i<=n/2;++i)
        ans+=4-r[i]-c[i]-r[n-i+1]-c[n-i+1];
    if(n%2&&(r[n/2+1]==0||c[n/2+1]==0))
        ++ans;
    printf("%d\n",ans);
    return 0;
}
