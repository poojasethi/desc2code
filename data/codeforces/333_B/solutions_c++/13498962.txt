#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,a[1001],b[1001],u,v,ans;
int main(){
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;i++){
        scanf("%d%d",&u,&v);
        a[u] = 1;b[v] = 1;
    }
    for(int i=2;i<=n/2;i++)
        ans += 4-(a[i] + a[n+1-i] + b[i] + b[n+1-i]);
    if(n%2)
        ans += min(1,2-(a[(n+1)/2]+b[(n+1)/2]));
    printf("%d\n",ans);
    return 0;
}
