#include <iostream>
#include <cstdio>
using namespace std;
#define MAXN 1111
int N,M,X[MAXN],Y[MAXN],x,y,Ans;
int main()
{
    scanf("%d%d",&N,&M);
    while(M--)
    {
        scanf("%d%d",&x,&y);
        X[x]=1;
        Y[y]=1;
    }
    for(int n=2;2*n<=N;++n)
        Ans+=4-(X[n]+Y[n]+X[N+1-n]+Y[N+1-n]);
    if(N&1)
        Ans+=(!X[N/2+1]||!Y[N/2+1]);
    printf("%d\n",Ans);
    return 0;
}
