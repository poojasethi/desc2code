#include <iostream>
using namespace std;
const int N=1010;
int r[N],c[N],n,m;
int main()
{
    cin >>n >>m;
    r[0]=r[n-1]=c[0]=c[n-1]=1;
    for (int i=0; i<m; i++)
    {
        int x,y; 
        cin >>x >>y;
        r[x-1]=1; c[y-1]=1;
    }
    int res=2*n;
    for (int i=0; i<n; i++) res-=r[i]+c[i];
    if (n%2==1 && r[n/2]==0 && c[n/2]==0) --res;
    cout <<res <<endl;
    return 0;
}
