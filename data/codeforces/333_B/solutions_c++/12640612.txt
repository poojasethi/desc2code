#include<stdio.h>
int row[1005],col[1005];

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	int i,j,x,y;
	for(i=1;i<=n;i++)row[i] = col[i] = 1;
	for(i=1;i<=m;i++)
	{
		scanf("%d%d",&x,&y);
		row[x] = col[y] = 0;
	}
	int ans = 0;
	for(i=2;i<n;i++)ans=ans+row[i] + col[i];
		if(n%2 && col[(n+1)/2]==1 && row[(n+1)/2]==1)ans--;
	printf("%d\n",ans);
}
