#include <cstdio>

int a[1005];
int b[1005];

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	while(m--)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		a[x]=1;
		b[y]=1;
	}
	int wyn=0;
	for(int i=2;i<n;i++)
	{
		if(!a[i]) wyn++;
		if(!b[i]) wyn++;
	}
	if((n&1)&&(!(a[(n+1)>>1]+b[(n+1)>>1]))) wyn--;
	printf("%d\n",wyn);
	return 0;
}
