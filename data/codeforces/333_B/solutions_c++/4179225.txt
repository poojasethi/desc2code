#include <cstdio>

int n,m;
int rows[1001];
int cols[1001];
int main() {
	scanf("%d%d",&n,&m);
	int vr,vc;
	for(int i=0;i<m;i++) {
		scanf("%d%d",&vr,&vc);
		rows[vr]=1;
		cols[vc]=1;
	}
	int sol=0;
	for(int i=2;i<n;i++) {
		if(rows[i]==0) sol++;
		if(cols[i]==0) sol++;
	}
	if(n%2) {
		if(rows[n/2+1]==0&&cols[n/2+1]==0) sol--;
	}
	printf("%d\n",sol);
	return 0;
}
