#include<cstdio>
#include<cstring>

int r[1111], c[1111];
int n,m, x, y,ans;

int main() {
	memset(r,0,sizeof r);
	memset(c,0,sizeof c);
	scanf("%d%d", &n, &m);
	for(int i=0;i<m;i++){
		scanf("%d%d", &x, &y);
		r[x]=1;c[y]=1;
	}
	ans=0;
	for(int i=2;i<n;i++){
		ans+=(1-r[i]);
		ans+=(1-c[i]);
	}
	if(n&1 && !r[n/2+1] && !c[n/2+1]) ans--;
	printf("%d\n", ans);

	return 0;
}
