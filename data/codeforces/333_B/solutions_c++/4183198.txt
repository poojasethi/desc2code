#include<stdio.h>
#include<stdlib.h>
long n,m,i,x,y,ans;
long b[100010];
long a[100010];
int main(){
    scanf("%ld%ld",&n,&m);
    for (i=1;i<=m;i++){
        scanf("%ld%ld",&x,&y);
        a[x]=1;
        b[y]=1;
    }
    for (i=2;i<n;i++)
        ans+=!b[i]+!a[i];
    if (n%2) ans-=(!b[n/2+1] && !a[n/2+1]);
    printf("%ld\n",ans);
    return 0;
}
