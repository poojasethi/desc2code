#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,x,y,i,ans;
bool a[1010],b[1010];
int main()
{
	cin>>n>>m;
	for(i=1;i<=m;i++)
	{
		scanf("%d%d",&x,&y);
		a[x]=b[y]=1;
	}
	for(i=2;i<n;i++)
		if(!a[i]&&!b[i]&&i*2-1!=n) ans+=2;
		else if(!a[i]||!b[i]) ans++;
	cout<<ans<<endl;
}
