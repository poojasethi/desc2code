#include<iostream>
using namespace std;
int n,m,x,y,i,ans;
bool a[1011],b[1011];
int main()
{
    
    cin>>n>>m;ans=0;
    for(i=0;i<m;i++)
    {
        cin>>x>>y;
        a[x]=b[y]=1;
    }
    for(i=2;i<n;i++)
        if(!a[i]&&!b[i]&&i*2-1!=n)ans+=2;
    else
        if(!a[i]||!b[i])ans++;
    cout<<ans;
    return 0;
}
