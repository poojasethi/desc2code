#include <iostream>
#include <vector>
using namespace std;

int main(){
  int n,m;
  cin>>n>>m;
  int x,y;
  vector<int> r(n+1,1),c(n+1,1);
  r[1]=0; r[n]=0; c[1]=0; c[n]=0;
  for(int i=0;i<m;i++){
    cin>>x>>y;
    r[x]=0;
    c[y]=0;
  }
  int ans=0;
  for(int i=1;i<=n;i++){
    if(r[i]==1)ans++;
    if(c[i]==1)ans++;
  }
  if(n%2==1&&r[n/2+1]==1&&c[n/2+1]==1)ans--;
  cout<<ans<<endl;
  return 0;
}
