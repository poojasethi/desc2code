#include<cstdio>
#include<cstdlib>
#define rep(i,n) for(int i=1;i<=n;i++)
#define N 1005
int n,m,dx[N],dy[N],ans;
int main(){
	scanf("%d%d",&n,&m);
	rep(i,m){
		int x,y; scanf("%d%d",&x,&y);
		dx[x]=1; dy[y]=1;
	}
	rep(i,n)
	if(i!=n/2+1||(n%2==0))
	if(i!=1&&i!=n){
		if(!dx[i]) ans++;
		if(!dy[i]) ans++;
	}
	if(n%2){
		if(!dx[n/2+1]||!dy[n/2+1]) ans++;
	}
	printf("%d\n",ans);
}
