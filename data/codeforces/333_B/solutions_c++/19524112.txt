#include<bits/stdc++.h>
using namespace std;
int r[1005],c[1005];
int main(){

	int n,m,x,y,i,ans=0;
	cin>>n>>m;
	for(i=1;i<=m;i++){
		cin>>x>>y;
		r[x]=-1;c[y]=-1;
	}
	for(i=2;i<n;i++){
		if(r[i]!=-1){
			r[i]++;
			ans++;
		}
	}
	for(i=2;i<n;i++){
		if(c[i]!=-1){
			if(r[i]==1){
				if(n+1 != 2*i){
					c[i]++;
					ans++;
				}
			}else{
				c[i]++;
				ans++;
			}
		}
	}
	cout<<ans<<endl;
	return 0;
}
