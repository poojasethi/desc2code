#include <cstdio>
#include <cstring>
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

int n,m,x,y,ans;
int bx[1010];
int by[1010];

int main() {
	scanf("%d%d",&n,&m);
	REP(i,m) {
		scanf("%d%d",&x,&y);
		bx[x]=by[y]=1;
	}
	for (x=2,y=n-1;x<=y;++x,--y)
		if (x==y) {
			if ((!bx[x])||(!by[x])) ++ans;
		} else
			ans+=4-bx[x]-bx[y]-by[x]-by[y];
	printf("%d\n",ans);
	return 0;
}
