#include <iostream>
using namespace std;
int main(){
	int n,m;
	bool tate[2000]={},yoko[2000]={};
	cin >> n >> m;
	for(int i=0;i<m;i++){
		int a,b;
		cin >> a >> b;
		a--;
		b--;
		tate[a]=true;
		yoko[b]=true;
	}
	int ans=0;
	for(int i=1;i<n-1;i++){
		if(2*i+1==n){
			if(!tate[i] || !yoko[i]) ans++;
			continue;
		}
		if(!tate[i]) ans++;
		if(!yoko[i]) ans++;
	}
	cout << ans;
	return 0;
}
