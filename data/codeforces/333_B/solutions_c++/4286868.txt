#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
int i,n,m,x,y,ans,h[2200],l[2200];
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;++i){
		scanf("%d%d",&x,&y);
		h[x]=l[y]=1;
	}
	for(i=2;i<n;++i){
		if(!h[i])ans++;
		if(!l[i])ans++;
	}
	if(n&1 && !h[(n+1)/2] && !l[(n+1)/2])ans--;
	printf("%d\n",ans);
}
