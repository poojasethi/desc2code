#include<cstdio>
#include<cstring>
int n,m,ans=0,x,y;
bool a[1010],b[1010];
int main(){
	scanf("%d%d",&n,&m);
	memset(a,1,sizeof(a));
	memset(b,1,sizeof(b));
	a[1]=b[1]=a[n]=b[n]=0;
	for (int i=1;i<=m;i++){
		scanf("%d%d",&x,&y);
		a[x]=b[y]=0;
	}
	for (int i=1;i<=n;i++){
		if (a[i]) ans++;
		if (b[i]) ans++;
	}
	if (n&1 && a[(n+1)/2] && b[(n+1)/2]) ans--;
	printf("%d\n",ans);
}
