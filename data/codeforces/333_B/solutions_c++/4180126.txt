#include<cstdio>

bool a[1001],b[1001],f;

int main()
{
	int n,m,x,y,i,s=0;

	scanf("%d%d",&n,&m);
	while(m--)
	{
		scanf("%d%d",&x,&y);
		a[x]=b[y]=1;
	}

	for(i=2;i<n;i++)
	{
		if(!a[i])
		{
			s++;
			if(i==n/2+1)
				f=1;
		}
	}
	for(i=2;i<n;i++)
	{
		if(!b[i])
		{
			if(n%2&&i==n/2+1&&f)
				continue;
			s++;
		}
	}
	printf("%d\n",s);

	return 0;
}
