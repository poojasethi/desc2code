rd = lambda : map(int, raw_input().split())
n, m = rd()
rows, cols = set(), set()
for i in range(m):
    xi, yi = rd()
    rows.add(xi)
    cols.add(yi)
res = 0
for i in range(2, n):
    if not i in rows: res += 1
    if not i in cols: res += 1
if (n & 1 and 
    not n / 2 + 1 in rows and 
    not n / 2 + 1 in cols): res -= 1
print res
