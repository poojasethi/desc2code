n,m=map(int,raw_input().split())
sumRow=[0 for i in range(n+10) ]
sumCol=[0 for i in range(n+10) ]
for i in range(m):
    x,y=map(int,raw_input().split())
    sumRow[x]=1
    sumCol[y]=1
ans=0
for i in range(2,n/2+1):
    ans=ans+4-sumRow[i]-sumCol[i]-sumRow[n-i+1]-sumCol[n-i+1]
if n%2==1 and (sumRow[n/2+1]==0 or sumCol[n/2+1]==0):
    ans+=1
print ans
