#define rep(i,n) for(int i=0;i<n;i++)
#define mp make_pair
#define pb push_back
#define X first
#define Y second
using namespace std;
typedef long long LL;
typedef pair<int,int> pii;
const int B=250;
char cmap[100005][10];
pii dp[100005][10];
int n,m,q;
void go(int r1,int r2){
	for(int i=r1;i<=r2;i++){
		rep(j,m)
			dp[i][j]=mp(i,j);
		rep(j,m)
			if(cmap[i][j]=='^')
				dp[i][j]=i==r1?mp(i-1,j):dp[i-1][j];
		rep(j,m)
			if(cmap[i][j]=='<')
				dp[i][j]=j?dp[i][j-1]:mp(i,j-1);
		for(int j=m-1;j>=0;j--)
			if(cmap[i][j]=='>')
				dp[i][j]=j==m-1?mp(i,j+1):dp[i][j+1];
	}
	return;
}
inline int Min(int a,int b){
	return a<b?a:b;
}
void init(){
	scanf("%d%d%d",&n,&m,&q);
	rep(i,n)
		scanf("%s",cmap+i);
	for(int i=0;i<=n;i+=B)
		go(i,Min(n-1,i+B-1));
	return;
}
void slove(){
	while(q--){
		char t;
		int a,b;
		scanf("%s%d%d",&t,&a,&b);
		a--;
		b--;
		if(t=='A'){
			while(a>=0 && b>=0 && b<m){
				int tx,ty;
				tx=dp[a][b].X;
				ty=dp[a][b].Y;
				if(tx==a && ty>=0 && ty<m){
					a=tx;
					b=ty;
					break;
				}
				a=tx;
				b=ty;
			}
			if(a>=0 && b>=0 && b<m)
				a=b=-2;
			printf("%d %d\n",a+1,b+1);
		}
		else{
			getchar();
			cmap[a][b]=getchar();
			int buc=a/B;
			go(buc*B,min(n-1,buc*B+B-1));
		}
	}
}
int main(){
	init();
	slove();
	return 0;
}
