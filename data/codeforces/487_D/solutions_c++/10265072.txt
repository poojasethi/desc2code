#include<cstdio>
#include<algorithm>
#include<utility>
#include<cmath>
using namespace std;

#define maxn 100010
char a[maxn][12];
pair<int,int> dp[maxn][12];
int sz;
int n,m,q;

void build(int r1,int r2)
{
	for(int i=r1;i<=r2;i++)
	{
		for(int j=0;j<m;j++)
			dp[i][j]=make_pair(i,j);
		for(int j=0;j<m;j++)
			if (a[i][j]=='^')
				dp[i][j]=(i==r1)?make_pair(i-1,j):dp[i-1][j];
		for(int j=0;j<m;j++)
			if (a[i][j]=='<')
				dp[i][j]=(j==0)?make_pair(i,j-1):dp[i][j-1];
		for(int j=m-1;j>=0;j--)
			if (a[i][j]=='>')
				dp[i][j]=(j==m-1)?make_pair(i,j+1):dp[i][j+1];
	}
}

int main()
{
	scanf("%d%d%d",&n,&m,&q);
	sz=(int)(pow(n,0.47));
	for(int i=0;i<n;i++)
		scanf("%s",a[i]);
	for(int i=0;i<n;i+=sz)
		build(i,min(i+sz-1,n-1));
	
	int x,y;
	char c;
	while (q--)
	{
		getchar();
		scanf("%c%d%d",&c,&x,&y);
		x--;y--;
		int nx,ny;
		if (c=='A')
		{
			while (x>=0 && y>=0 && y<m)
			{
				nx=dp[x][y].first;
				ny=dp[x][y].second;
				if (x==nx && ny>=0 && ny<m)
				{
					x=nx,y=ny;
					break;
				}
				x=nx,y=ny;
			}
			if (x>=0 && y>=0 && y<m)
				x=y=-2;
			printf("%d %d\n",x+1,y+1);
		}
		else
		{
			getchar();
			a[x][y]=getchar();
			build(x/sz*sz,min(n-1,x/sz*sz+sz-1));
		}
	}
	return 0;
}
