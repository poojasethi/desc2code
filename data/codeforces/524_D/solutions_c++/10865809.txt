#include <cstdio>
#include <queue>
using namespace std;
const int N=20005;
int n,M,T,ans[N];
deque<int> q;
int main(){
	scanf("%d%d%d",&n,&M,&T);
	int ma=0;
	for(int i=1;i<=n;i++){
		int h,m,s; scanf("%d:%d:%d",&h,&m,&s);
		int t=(h*60+m)*60+s;
		while(!q.empty()&&q.front()+T<=t) q.pop_front();
		if(q.size()+1<=M)
			ans[i]=ans[i-1]+1;
		else
			ans[i]=ans[i-1],
			q.pop_back();
		q.push_back(t);
		if(q.size()>ma) ma=q.size();
	}
	if(ma!=M)
		puts("No solution");
	else{
		printf("%d\n",ans[n]);
		for(int i=1;i<=n;i++) printf("%d\n",ans[i]);
	}
}
