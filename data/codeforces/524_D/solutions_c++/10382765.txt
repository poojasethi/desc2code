#include <iostream>
#include <deque>
#include <string>
#include <queue>

using namespace std;

int main()
{
    int n,m,t,f,k=0,ind=0;
    cin>>n>>m>>t;
    queue<int>q;
    deque<int>d;
    string s;
    for(int i=0;i<n;i++)
    {
        cin>>s;
        f=0;
        f=f+s[0]-'0';
        f*=10;
        f=f+s[1]-'0';
        f*=6;
        f=f+s[3]-'0';
        f*=10;
        f=f+s[4]-'0';
        f*=6;
        f=f+s[6]-'0';
        f*=10;
        f=f+s[7]-'0';
        if(!d.empty()&&d.front()+t<=f)
            d.pop_front();
        if(d.size()==m)
        {
            k--;
            d.pop_back();
        }
        k++;
        d.push_back(f);
        q.push(k);
        if(d.size()==m)
            ind=1;
    }
    if(ind==1)
    {
        cout<<k;
        while(!q.empty())
        {
            cout<<endl<<q.front();
            q.pop();
        }
    }
    else
        cout<<"No solution";
    return 0;
}
