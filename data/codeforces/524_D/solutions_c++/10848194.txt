#include<iostream>
#include<cstdio>
#include<cstring>

using namespace std; 

int n,m,t,req[30003],que[30003],ans[30003];

void gn(int &x){
	char c;while((c=getchar())<'0'||c>'9');x=c-'0';
	while((c=getchar())>='0'&&c<='9')x=x*10+c-'0';
}	

int main() {
	scanf("%d%d%d",&n,&m,&t); 
	for ( int i=1;i<=n;i++) {
		int a,b,c;
		gn(a);gn(b);gn(c); 
		req[i]=a*3600+b*60+c; 
	}
	int l=0,r=-1,tot=0,mx=0;
	for (int i=1;i<=n;i++){
		while ( l<=r && req[i]>que[l]) l++; 
		if ( r-l+1<m) {
			ans[i]=++tot; 
			que[++r]=req[i]+t-1; 
		} else { 
		    que[r]=req[i]+t-1; 
			ans[i]=tot; 
		}
		mx=max(mx,r-l+1); 
	}
	if ( mx==m ) { 
		printf("%d\n",tot); 
		for (int i=1;i<=n;i++) 
			printf("%d\n",ans[i]); 
	}else puts("No solution"); 
	return 0; 
}
