#define INF (int)1e9
typedef pair<int,int> P ;
int x[4],y[4] ;

int dx[] = {-1,-1,-1,0,0,0,1,1,1} ;
int dy[] = {-1,0,1,-1,0,1,-1,0,1} ;

bool check(int kx,int ky)
{
 for(int i = 0;i < 9;i++)
 {
  int xx = x[2] + dx[i] ;
  int yy = y[2] + dy[i] ;
  if(xx == kx && yy == ky) return true ;
 }
 for(int i = 0;i < 2;i++)
  if(kx != x[i] || ky != y[i])
  {
   if(kx == x[i])
   {
    int mn = min(ky,y[i]) ;
    int mx = max(ky,y[i]) ;
    if(x[2] != kx || !(y[2] > mn && y[2] < mx)) return true ;
   }
   else if(ky == y[i])
   {
    int mn = min(kx,x[i]) ;
    int mx = max(kx,x[i]) ;
    if(y[2] != ky || !(x[2] > mn && x[2] < mx)) return true ;
   }
  }
 return false ;
}

int main()
{
 int i,j,k ;
 for(i = 0;i < 4;i++)
 {
  string s ;
  cin >> s ;
  x[i] = s[0] - 'a' ;
  y[i] = s[1] - '1' ;
 }
 for(i = 0;i < 9;i++)
 {
  int kx = x[3] + dx[i] ;
  int ky = y[3] + dy[i] ;
  if(kx < 0 || ky < 0 || kx >= 8 || ky >= 8) continue ;
  if(!check(kx,ky)) break ;
 }
 if(i == 9) printf("CHECKMATE\n") ;
 else printf("OTHER\n") ;
 return 0 ;
}
