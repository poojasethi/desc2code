#include <stdio.h>
int abs(int a){
    if (a<0)
        return -a;
    return a;
}
bool check(int x4,int y4,int x1,int y1,int x2,int y2,int x3,int y3){
    if (abs(x4-x3)<=1&&abs(y3-y4)<=1)
        return true;
    if (x4==x1&&y4!=y1&&(x3!=x4||abs(y4-y1)<abs(y4-y3)+abs(y3-y1)))
        return true;
    if (y4==y1&&x4!=x1&&(y3!=y4||abs(x4-x1)<abs(x4-x3)+abs(x3-x1)))
        return true;
    if (x4==x2&&y4!=y2&&(x3!=x4||abs(y4-y2)<abs(y4-y3)+abs(y3-y2)))
        return true;
    if (y4==y2&&x4!=x2&&(y3!=y4||abs(x4-x2)<abs(x4-x3)+abs(x3-x2)))
        return true;
    return false;
    
}
int main(){
    char s1[3];
    char s2[3];
    char s3[3];
    char s4[3];
    scanf("%s%s%s%s",s1,s2,s3,s4);
    int x1=s1[0]-'a';
    int y1=s1[1]-'0'-1;
    int x2=s2[0]-'a';
    int y2=s2[1]-'0'-1;
    int x3=s3[0]-'a';
    int y3=s3[1]-'0'-1;
    int x4=s4[0]-'a';
    int y4=s4[1]-'0'-1;
    for (int i=x4-1;i<=x4+1;i++)
        for (int j=y4-1;j<=y4+1;j++){
            if (i>=0&&i<=7&&j>=0&&j<=7)
                if (!check(i,j,x1,y1,x2,y2,x3,y3)){
                    printf("OTHER");
                    return 0;
                }
        }
    printf("CHECKMATE");
}
