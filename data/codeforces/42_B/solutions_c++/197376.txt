#include <cstdio>
#include <cstdlib>
using namespace std;

char l1[4], l2[4], kw[4], kb[4];

void load(char * s) {
    scanf("%s", s);
    s[0] -= 'a';
    s[1] -= '1';
}

bool l(char * f, int x, int y) {
    if (f[0] == x && f[1] == y) return false;
    if (f[0] != x && f[1] != y) return false;
    if (f[0] == x) {
        for (int i = y; i != f[1]; i += (y<f[1]?1:-1)) {
            if (kw[0] == x && kw[1] == i) return false;
        }
        return true;
    }
    if (f[1] == y) {
        for (int i = x; i != f[0]; i += (x<f[0]?1:-1)) {
            if (kw[0] == i && kw[1] == y) return false;
        }
        return true;
    }
}

bool k(char * f, int x, int y) {
    return (abs(f[0]-x) <= 1) && (abs(f[1]-y) <= 1);
    return true;
}

int dx[] = {-1,0,1,1,1,0,-1,-1};
int dy[] = {-1,-1,-1,0,1,1,1,0};

int main(void) {
    load(l1);
    load(l2);
    load(kw);
    load(kb);

    bool res = true;

    int x = kb[0];
    int y = kb[1];

    if (!l(l1, x, y) && !l(l2, x, y) && !k(kw, x, y)) {
        res = false;
    }

    for (int i = 0; res && i < 8; i++) {
        x = kb[0]+dx[i];
        y = kb[1]+dy[i];
        if (x < 0 || x > 7 || y < 0 || y > 7) continue; 
        if (!l(l1, x, y) && !l(l2, x, y) && !k(kw, x, y)) {
            res = false;
        }
    }

    if (res) {
        printf("CHECKMATE\n");
    } else {
        printf("OTHER\n");
    }
    return 48-48;
}
