#include <iostream>
using namespace std;

int x1,y1,x2,y2,x3,y3;
int mx[] = {1,1,1,-1,-1,-1,0,0,0};
int my[] = {-1,1,0,-1,1,0,-1,1,0};

bool check(int x,int y)
{	
	bool flag = false;
	int i,j,k;
	
	for(i=0;i<8;i++){
		j = x3 + mx[i];
		k = y3 + my[i];
		if(j == x && k == y){
			flag = true;
			break;
		}
	}
	if((x1 != x || y1 != y)){
		if(x1 == x){
			if(x3 == x && y3 > min(y1,y) && y3 < max(y1,y));
			else flag = true;
		}
		if(y1 == y){
			if(y3 == y && x3 > min(x1,x) && x3 < max(x1,x));
			else flag = true;
		}
	}
	if((x2 != x || y2 != y)){
		if(x2 == x){
			if(x3 == x && y3 > min(y2,y) && y3 < max(y2,y));
			else flag = true;
		}
		if(y2 == y){
			if(y3 == y && x3 > min(x2,x) && x3 < max(x2,x));
			else flag = true;
		}
	}
	return flag;
}

int main(){
	int i,j,k;
	int x,y,nx,ny;
	string s1,s2,s3,s4;
	
	cin>>s1>>s2>>s3>>s4;
	x1 = s1[0] - 'a';
	y1 = s1[1] - '1';
		
	x2 = s2[0] - 'a';
	y2 = s2[1] - '1';
	
	x3 = s3[0] - 'a';
	y3 = s3[1] - '1';
	
	x = s4[0] - 'a';
	y = s4[1] - '1';
	
	for(i=0;i<9;i++){
		nx = x + mx[i];
		ny = y + my[i];
		if(nx >= 0 && nx < 8 && ny >= 0 && ny < 8 && !check(nx,ny)){
			cout<<"OTHER"<<endl;
			return 0;
		}
	}
	cout<<"CHECKMATE"<<endl;
	return 0;
}
