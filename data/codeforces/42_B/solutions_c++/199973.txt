char a[2], b[2], c[2], d[2];
int i, j, x, y, a1, b1, c2, d1, c1, d2, b2, a2, e[10][10]
  , o[] = { 1, 0, -1, 0, 0, 1, 0, -1};
bool p;
void f1(int X, int Y) {
    for (i = 0; i < 8; i += 2) {
        x = X; y = Y;
        for (;;) {
            x += o[i]; y += o[i + 1];
            if (e[x][y] > 1 || x < 1 || x > 8 || y < 1 || y > 8) break;
            e[x][y] = 1;
        }
        e[x][y] = 1;
    }
}
void f2(int x, int y) {
    p = false;
    for (i = x - 1; i <= x + 1; i++)
        for (j = y - 1; j <= y + 1; j++) {
            if (e[i][j] == 0)
                p = true;
            e[i][j] = 4;
        }
}
