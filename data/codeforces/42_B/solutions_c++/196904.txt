#include <iostream>
#include <fstream>

using namespace std;

bool used[10][10] = { };

int di[4] = { 0, 0, -1, 1 };
int dj[4] = { -1, 1, 0 ,0 };

int DI[9] = { 0,0,0,1,1,1,-1,-1,-1 };
int DJ[9] = { 0,1,-1,0,-1,1,0,-1,1 };

char l1[10]; int l1i; int l1j;
char l2[10]; int l2i; int l2j;
char kb[10]; int kbi; int kbj;
char kw[10]; int kwi; int kwj;


bool valid(int a)
{
	return (1 <= a && a <= 8);
};

void processL(int i, int j)
{
	for (int k = 0; k < 4; k ++)
	{
		int i1 = i + di[k];
		int j1 = j + dj[k];
		while (true)
		{
			if (!valid(i1) || !valid(j1)) break;
			used[i1][j1] = true;
			if (i1 == kwi && j1 == kwj) break;
			i1 += di[k];
			j1 += dj[k];
		}
	}
}

int coord(char a)
{
	if ('1' <= a && a <= '8')
		return a - '0';
	return a - 'a' + 1;
}

int main()
{
	scanf("%s",l1); l1i = coord(l1[0]); l1j = coord(l1[1]);
	scanf("%s",l2); l2i = coord(l2[0]); l2j = coord(l2[1]);
	scanf("%s",kw); kwi = coord(kw[0]); kwj = coord(kw[1]);
	scanf("%s",kb); kbi = coord(kb[0]); kbj = coord(kb[1]);
	processL(l1i,l1j);
	processL(l2i,l2j);
	bool mat = true;
	for (int i = 0; i < 9; i ++)
	{
		int i1 = kwi + DI[i];
		int j1 = kwj + DJ[i];
		used[i1][j1] = true;
	}
	for (int i = 0; i < 9; i ++)
	{
		int i1 = kbi + DI[i];
		int j1 = kbj + DJ[i];
		if (valid(i1) && valid(j1) && !used[i1][j1])
			mat = false;
	}
	if (mat)
		printf("CHECKMATE\n");
	else
		printf("OTHER\n");
	return 0;
}
