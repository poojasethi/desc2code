#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>

int	fx[] = {-1, 0, +1, 0}, fy[] = {0, +1, 0, -1},
	ex[] = {-1, -1, 0, +1, +1, +1, 0, -1}, ey[] = {0, +1, +1, +1, 0, -1, -1, -1};

using namespace std;

string rk1, rk2, wk, bk;
int stand[20][20], hit[20][20];

void proc (string &s) {stand[s[0] - 'a'][s[1] - '1'] = 1;}
void rook (int x, int y)
{
	for (int i = 0; i < 4; i++)
	{
		int xx = x, yy = y;

		for (;;)
		{
		 	xx += fx[i]; yy += fy[i];

		 	if (xx < 0 || yy < 0 || xx > 7 || yy > 7) break;

		 	hit[xx][yy] = 1;

		 	if (stand[xx][yy]) break;
		}
	}
}

int main ()
{
	cin >> rk1 >> rk2 >> wk >> bk;

	proc (wk);

	rook (rk1[0] - 'a', rk1[1] - '1');
	rook (rk2[0] - 'a', rk2[1] - '1');

	for (int i = 0; i < 8; i++)
	{
	 	int xx = wk[0] - 'a' + ex[i], yy = wk[1] - '1' + ey[i];
	 	if (xx < 0 || yy < 0 || xx > 7 || yy > 7) continue;
	 	hit[xx][yy] = 1;
	}

	bool bad = hit[bk[0] - 'a'][bk[1] - '1'];

	for (int i = 0; i < 8; i++)
	{
	 	int xx = bk[0] - 'a' + ex[i], yy = bk[1] - '1' + ey[i];
	 	if (xx < 0 || yy < 0 || xx > 7 || yy > 7) continue;
	 	bad &= hit[xx][yy];
	}

	if (bad)
		puts ("CHECKMATE");
	else
		puts ("OTHER");

	return 0;
}
