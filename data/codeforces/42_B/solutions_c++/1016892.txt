#include<iostream>
#include<algorithm>
using namespace std;
typedef pair<int,int> ii;

void scan(ii &i) {
	char c;
	cin>>c;
	int t;
	cin>>t;
	i=ii(c-'a',t-1);
}
ii rk1,rk2,white,black;
bool between(int a,int m,int b) {
	return (m>a && m<b) || (m>b && m<a);
}
bool threat(ii &rook,ii &king) {
	return ((rook.first==king.first && !(white.first==king.first && between(rook.second,white.second,king.second) ) )
		|| (rook.second==king.second && !(white.second==king.second && between(rook.first,white.first,king.first) ) ) )
		&& rook!=king;
}
bool threatk(ii &k1,ii &k2) {
	return abs(k1.first-k2.first)<=1 && abs(k1.second-k2.second)<=1;
}
bool in(ii &t) {
	return t.first>=0 && t.first<8 && t.second>=0 && t.second<8;
}
int main() {
	scan(rk1);
	scan(rk2);
	scan(white);
	scan(black);
	for(int i=-1;i<=1;i++) {
		for(int j=-1;j<=1;j++) {
			ii pos=ii(black.first+i,black.second+j);
			if(in(pos) && !threat(rk1,pos) && !threat(rk2,pos) && !threatk(white,pos)) {
				cout<<"OTHER";
				return 0;
			}
		}
	}
	cout<<"CHECKMATE";
}
