#include <iostream>
#include <string>
#include <vector>

using namespace std;

pair<int,int> coord(string s)
{
  return pair<int,int> (s[0]-'a',s[1]-'1');
}

int movi[4]={1,-1,0,0};
int movj[4]={0,0,-1,1};


bool muere(vector<vector<int> > t,int ii,int jj)
{
  for (int k=0;k<4;k++) {
    int i=ii+movi[k];
    int j=jj+movj[k];
    while (i>=0 and i<8 and j>=0 and j<8) {
	if (t[i][j]==3) return true;
	else if (t[i][j]!=0) break;
	i+=movi[k];
	j+=movj[k];
    }
  }
  for (int i=ii-1;i<=ii+1;i++) {
    for (int j=jj-1;j<=jj+1;j++) {
      if (i>=0 and i<8 and j>=0 and j<8 and t[i][j]==2)
	  return true;      
    }
  }  
  return false;
}

bool jaque(vector<vector<int> > t,int ii,int jj)
{
  for (int i=ii-1;i<=ii+1;i++) {
    for (int j=jj-1;j<=jj+1;j++) {
      if (i>=0 and i<8 and j>=0 and j<8) {
	vector<vector<int> > taux=t;
	taux[ii][jj]=0;
	taux[i][j]=1;
	if (not muere(taux,i,j)) return false;
      }
    }
  }
  return true;
}

int main()
{
  vector<vector<int> > t(8,vector<int> (8,0));
  string s;
  pair<int,int> p;
  cin>>s;
  p=coord(s);
  t[p.first][p.second]=3;
  cin>>s;
  p=coord(s);
  t[p.first][p.second]=3;
  cin>>s;
  p=coord(s);
  t[p.first][p.second]=2;
  cin>>s;
  p=coord(s);
  t[p.first][p.second]=1;
  int ii=p.first;
  int jj=p.second;
  if (jaque(t,ii,jj)) {
    cout<<"CHECKMATE"<<endl;
  } else {
    cout<<"OTHER"<<endl;
  }
}
