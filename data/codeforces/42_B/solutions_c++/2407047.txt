#include<algorithm>
#include<iostream>
#include<cstdio>
#include<string>

using namespace std;

string str;

int x[5] , y[5];

int dis( int u , int v ) {
	return abs( x[u] - x[v] ) + abs( y[u] - y[v] );
}

bool check() {
	if ( x[4] < 0 || x[4] >= 8 || y[4] < 0 || y[4] >= 8 ) return true;
	if ( max( abs( x[2] - x[4] ) , abs( y[2] - y[4] ) ) == 1 ) return true;
	if ( ( x[0] == x[4] || y[0] == y[4] ) && dis( 0 , 4 ) != 0 && 
		( dis( 4 , 1 ) == 0 || dis( 0 , 4 ) != dis( 0 , 1 ) )+ dis( 1 , 4 ) && dis( 0 , 4 ) != dis( 0 , 2 ) + dis( 2 , 4 ) ) return true;
	if ( ( x[1] == x[4] || y[1] == y[4] ) && dis( 1 , 4 ) != 0 && 
		( dis( 4 , 0 ) == 0 || dis( 1 , 4 ) != dis( 1 , 0 ) ) + dis( 0 , 4 ) && dis( 1 , 4 ) != dis( 1 , 2 ) + dis( 2 , 4 ) ) return true;
	return false;
}

int main() {
	for ( int i = 0 ; i < 4 ; i++ ) {
		cin >> str;
		x[i] = str[0] - 'a';
		y[i] = str[1] - '1';
	}
	
	for ( x[4] = x[3] - 1 ; x[4] <= x[3] + 1 ; x[4]++ )
		for ( y[4] = y[3] - 1 ; y[4] <= y[3] + 1 ; y[4]++ )
			if ( !check() ) {
				printf( "OTHER\n" );
				return 0;
			}
	printf( "CHECKMATE\n" );
			
	return 0;
}
