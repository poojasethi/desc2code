#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

char c[20];
int x[4], y[4];

bool check(int X, int Y)
{
    if( X < 0 || X >= 8 || Y < 0 || Y >= 8 ) return true;
    if( abs(X-x[2]) <= 1 && abs(Y-y[2]) <= 1 ) return true;

    for(int i = 0; i <= 1; i++)
    {
        if( x[i] == X && y[i] == Y );
        else if( x[i] == X )
        {
            if( x[2] != X || abs(y[2]-Y) > abs(y[i]-Y) ||  min(y[i], y[2]) < Y && Y < max(y[i], y[2])  ) return true;
        }
        else if( y[i] == Y )
        {
            if( y[2] != Y || abs(x[2]-X) > abs(x[i]-X) ||  min(x[i], x[2]) < X && X < max(x[i], x[2])   ) return true;
        }
    }

    return false;
}

int main()
{
    gets(c);

    for(int i = 0; i < 4; i++)
        x[i] = c[ 3*i]-'a', y[i] = c[ 3*i+1]-'1';

    for(int i = -1; i <= 1; i++)
        for(int j = -1; j <= 1; j++)
            if( !check(x[3]+i, y[3]+j) )
            {
                puts("OTHER"); return 0;
            }

    puts("CHECKMATE"); return 0;
}
