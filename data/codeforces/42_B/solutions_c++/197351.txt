#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <string>
#include <vector>
using namespace std;

typedef pair<int, int> pii;
#define R first
#define C second

int mrk[10][10];

pii read()
{
	char c;
	int r;
	cin >> c >> r;
	return pii(r, c-'a'+1);
}

void trt(int i, int j)
{
	if(i >= 1 && i <= 8 && j >= 1 && j <= 8)
		mrk[i][j] = true;
}

int main()
{
	pii r[2], k1, bk;
	r[0] = read();
	r[1] = read();
	k1 = read();
	bk = read();

	for(int k=0; k<2; k++)
	{
		for(int i=r[k].C-1; i>=1; i--)
			if(r[k].R == k1.R && i == k1.C)
				break;
			else
				trt(r[k].R, i);
		for(int i=r[k].C+1; i<=8; i++)
			if(r[k].R == k1.R && i == k1.C)
				break;
			else
				trt(r[k].R, i);
		for(int i=r[k].R-1; i>=1; i--)
			if(r[k].C == k1.C && i == k1.R)
				break;
			else
				trt(i, r[k].C);
		for(int i=r[k].R+1; i<=8; i++)
			if(r[k].C == k1.C && i == k1.R)
				break;
			else
				trt(i, r[k].C);

	}

	for(int i=-1; i<=1; i++)
		for(int j=-1; j<=1; j++)
			trt(k1.R+i, k1.C+j);
/**
	for(int i=1; i<=8; i++, cout << endl)
		for(int j=1; j<=8; j++)
			if(mrk[i][j])
				cout << ".";
			else
				cout << " ";
/**/

	bool can = false;
	for(int i=-1; i<=1; i++)
		for(int j=-1; j<=1; j++)
			if(!mrk[bk.R+i][bk.C+j] && bk.R+i>=1 && bk.R+i<=8 && bk.C+j>=1 && bk.C+j<=8)
				can = true;
	if(can)
		cout << "OTHER" << endl;
	else
		cout << "CHECKMATE" << endl;
	return 0;
}
