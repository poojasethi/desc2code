#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

bool hash[8][8];
int dx[] = { -1, 0, 0, 1, -1, -1, 1, 1 };
int dy[] = { 0, -1, 1, 0, -1, 1, -1, 1 };

bool yes(int x, int y) {
	if (!hash[x][y])
		return false;
	for (int i = 0; i < 8; i++) {
		int xx = x + dx[i];
		int yy = y + dy[i];
		if (xx < 0 || xx > 7 || yy < 0 || yy > 7)
			continue;
		if (!hash[xx][yy])
			return false;
	}
	return true;
}
int main() {
	string a, b, c, d;
	cin >> a >> b >> c >> d;
	int kx = c[0] - 'a';
	int ky = c[1] - '1';
	for (int i = 0; i < 8; i++) {
		int xx = kx + dx[i];
		int yy = ky + dy[i];
		if (xx < 0 || xx > 7 || yy < 0 || yy > 7)
			continue;
		hash[xx][yy] = true;
	}
	int x = a[0] - 'a';
	int y = a[1] - '1';
	for (int i = 0; i < 4; i++) {
		int xx = x + dx[i];
		int yy = y + dy[i];
		while (xx >= 0 && xx <= 7 && yy >= 0 && yy <= 7) {
			if (xx == kx && yy == ky)
				break;
			hash[xx][yy] = true;
			xx += dx[i];
			yy += dy[i];
		}
	}
	x = b[0] - 'a';
	y = b[1] - '1';
	for (int i = 0; i < 4; i++) {
		int xx = x + dx[i];
		int yy = y + dy[i];

			if (xx == kx && yy == ky)
				break;
			hash[xx][yy] = true;
			xx += dx[i];
			yy += dy[i];
		}
	}

	if (yes(d[0] - 'a', d[1] - '1'))
