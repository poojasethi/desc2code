#include <iostream>

using namespace std;
int abs(int x) {return x<0 ? -x : x;}
int in(int a, int b, int c) {return b<=a && a<=c ||  b>=a && a>=c;}
int main()
{
    char x,y,c;

    cin >> y >> x;
    int L1y = y-'a';
    int L1x = x-'1';

    cin >> y >> x;
    int L2y = y-'a';
    int L2x = x-'1';

    cin >> y >> x;
    int K1y = y-'a';
    int K1x = x-'1';

    cin >> y >> x;
    int K2y = y-'a';
    int K2x = x-'1';

    bool b = true;
    for(int X=K2x-1; X<=K2x+1; ++X)
		for(int Y=K2y-1; Y<=K2y+1; ++Y) {
			if(X<0 || Y<0 || X>7 || Y>7)
				continue;
			if(!(L1x==X && L1y!=Y && !(K1x==X && in(K1y, L1y, Y))) &&
			!(L1x!=X && L1y==Y && !(K1y==Y && in(K1x, L1x, X))) &&
			!(L2x==X && L2y!=Y && !(K1x==X && in(K1y, L2y, Y))) &&
				!(L2x!=X && L2y==Y && !(K1y==Y && in(K1x, L2x, X))) &&
				!(max(abs(K1x-X), abs(K1y-Y)) == 1)) {
					b = false;
					break;
				}
		}


    cout << (b ? "CHECKMATE" : "OTHER");
    return 0;
}
