#include<iostream>

using namespace std;

int main(){
    int n,t[15][15],a[]={-1,-1,0,1,1,1,0,-1,0,0},b[]={0,1,1,1,0,-1,-1,-1,0,0};
    int x,y,xk,yk;
    bool cz=true;
    string m[4];
    for(int i=0; i<4; ++i) cin>>m[i];
    for(int i=0; i<10; ++i)
    for(int j=0; j<10; ++j) t[i][j]=0;
    
    xk = int(m[2][0]-'a'+1);
    yk = int(m[2][1]-'0');
    for(int i=0; i<10; ++i) { t[0][i]=1; t[9][i]=1; t[i][0]=1; t[i][9]=1; }
    for(int i=0; i<2; ++i) { 
            x = int(m[i][0]-'a'+1);
            y = int(m[i][1]-'0');
            for(int j=x; j<10; ++j) { if(j==xk && y==yk) break; t[(j!=x)?j:0][y] = 1; }
            for(int j=x; j>=0; --j) { if(j==xk && y==yk) break; t[(j!=x)?j:0][y] = 1; }
            for(int j=y; j<10; ++j) { t[x][(j!=y)?j:0] = 1; if(x==xk && j==yk) break; }
            for(int j=y; j>=0; --j) { t[x][(j!=y)?j:0] = 1; if(x==xk && j==yk) break; }
    }
    for(int i=0; i<8; ++i) {
            xk = int(m[2][0]-'a'+1);
            yk = int(m[2][1]-'0');
            t[ xk+a[i] ][ yk+b[i] ] = 1;
    }
    for(int i=0; i<8; ++i) {
            x = int(m[3][0]-'a'+1);
            y = int(m[3][1]-'0');
            if( t[ x+a[i] ][ y+b[i] ] == 0 ) cz=false;
            if( t[x][y] == 0 ) cz=false;
    }
    //cout.precision(4);
    if(cz) cout<<"CHECKMATE"<<endl;
    else cout<<"OTHER"<<endl;
    //system("pause");
    return 0;
}
