#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#include <cstdio>
#include <cstring>
using namespace std;
char board[10][10],have[10][10];
int xx,yy;
int dx[]={1,0,-1,1,0,-1,1,0,-1};
int dy[]={0,-1,1,1,1,0,-1,0,-1};
void put(string s,char k)
{
	int a=s[0]-'a';
	int b=s[1]-'1';
	board[a][b]=k;
	if(k=='K')
	xx=a,yy=b;
}
bool SAFE(int x,int y)
{
	for(int i=0;i<8;i++)
	for(int j=0;j<8;j++)
	{
	if(have[i][j]=='k'&&abs(i-x)<=1&&abs(j-y)<=1)
	{
	return false;
	}
	if(have[i][j]=='r'&&(i==x||j==y))
	{
	bool dangerous=true;
	for(int k=min(i,x);k<=max(i,x);k++)
	for(int l=min(j,y);l<=max(j,y);l++)
	if(have[k][l]=='k')
	dangerous=false;
	if(dangerous)
	{
	return false;
	}
	}
	}
	//cout<<"OK"<<endl;
	return true;
}
int main()
{
	string a,b,c,d;
	cin>>a>>b>>c>>d;
	put(a,'r');
	put(b,'r');
	put(c,'k');
	put(d,'K');
	bool mate=true;
	//cout<<xx<<" "<<yy<<endl<<endl;
	for(int i=0;i<9;i++)
	{
	int x=xx+dx[i],y=yy+dy[i];
	//cout<<x<<" "<<y<<endl;
	if(x>=8||x<0||y>=8||y<0)
	continue;
	for(int j=0;j<8;j++)
	for(int k=0;k<8;k++)
	have[j][k]=board[j][k];
	have[xx][yy]='-';
	have[x][y]='K';
	if(SAFE(x,y))
	mate=false;
	}
	if(mate)
	cout<<"CHECKMATE";
	else cout<<"OTHER";
	return 0;
}
