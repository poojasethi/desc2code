#define _abs(x)((x)>0?(x):-(x))
using namespace std;
char board[10][10],board2[10][10];
int dx[]={-1,-1,-1,0,0,0,1,1,1},dy[]={-1,0,1,-1,0,1,-1,0,1};
void put(string s,char c){
	board[s[0]-'a'][s[1]-'1']=c;
}
bool func(void){
	int x,y,i,j,k,l;
	REP(x,8)REP(y,8)if(board2[x][y]=='K'){
		REP(i,8)REP(j,8){
			if(board2[i][j]=='k'&&_abs(i-x)<=1&&_abs(j-y)<=1)return true;
			if(board2[i][j]=='r'&&(i==x||j==y)){
				bool king =false;
				for(k=min(i,x);k<=max(i,x);k++)for(l=min(j,y);l<=max(j,y);l++)if(board2[k][l]=='k')king =true;
				if(!king)return true;
			}
		}
	}
	return false;
}
int main(){
	int i,j,x,y,k;
	REP(i,8)REP(j,8)board[i][j]='-';
	string a,b,c,d;
	cin>>a>>b>>c>>d;
	put(a,'r');put(b,'r');put(c,'k');put(d,'K');
	bool mate=true;
	REP(x,8)REP(y,8)if(board[x][y]=='K')REP(i,9){
		int x2=x+dx[i],y2=y+dy[i];
		if(x2<0||x2>=8||y2<0||y2>=8)continue;
		REP(j,8)REP(k,8)board2[j][k]=board[j][k];
		board2[x][y]='-';board2[x2][y2]='K';
		if(!func())mate=false;
	} 
	cout<<(mate?"CHECKMATE":"OTHER")<<endl;
	return 0;
}
