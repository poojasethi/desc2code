{

{

}
else if(y[1]==y[4] && x[1]!=x[4] && ((y[3]!=y[1]) || (y[3]==y[1] && (x[3]<min(x[1],x[4]) || x[3]>max(x[1],x[4])))))

return 1;
}
else if((abs(y[3]-y[4]))<=1 && (abs(x[3]-x[4])<=1))
{

}
else
{
return 0;
}
}
