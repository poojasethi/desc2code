#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<iostream>
#include<vector>
#include<string>
using namespace std;
typedef long long LL;
int AB(int x){if(x<0)return -x;return x;}
char ch[120];
int dx[]={-1,-1,-1,0,0,1,1,1,0};
int dy[]={-1,0,1,-1,1,-1,0,1,0};
int x[120],y[120],i,j,tx,ty;
int main()
{
	while(~scanf("%s",ch))
	{
		x[0]=ch[0]-'a';y[0]=ch[1]-'1';
		for(i=1;i<4;i++)
		{
			scanf("%s",ch);
			x[i]=ch[0]-'a';y[i]=ch[1]-'1';
		}
		bool win=true;
		for(i=0;i<9;i++)
		{
			tx=x[3]+dx[i];
			ty=y[3]+dy[i];
			if(tx<0||tx>=8)continue;
			if(ty<0||ty>=8)continue;
			bool eat=false;
			int d=max(AB(tx-x[2]),AB(ty-y[2]));
			if(d==1)eat=true;
			for(j=0;j<2;j++)
			{
				if(x[j]==tx&&y[j]==ty)continue;
				if(x[j]==tx&&(x[2]!=tx||(ty-y[2])*(y[j]-y[2])>0 ))eat=true;
				if(y[j]==ty&&(y[2]!=ty||(tx-x[2])*(x[j]-x[2])>0 ))eat=true;
			}
		//	printf("%d %d\n",tx,ty);
		//	if(eat)puts("Yes");
			if(!eat)
			{win=false;break;} 
		}
		if(win)puts("CHECKMATE");
		else puts("OTHER");
	}
}
