#include <stdio.h>

char ra,rb,ka,kb;
int r1,r2,k1,k2,flg;

int sp1(char x,int y)
    {
        if ((ra==x)&&(ka==x)&&((k1-y)*(k1-r1)<0))
            return 1;
        if ((r1==y)&&(k1==y)&&((ka-x)*(ka-ra)<0))
            return 1;
        return 0;
    }

int sp2(char x,int y)
    {
        if ((rb==x)&&(ka==x)&&((k1-y)*(k1-r2)<0))
            return 1;
        if ((r2==y)&&(k1==y)&&((ka-x)*(ka-rb)<0))
            return 1;
        return 0;
    }


void chk(char x,int y)
    {
        if ((x<'a')||(x>'h')||(y<1)||(y>8))
            return;
        if ((x-ka<2)&&(ka-x<2)&&(y-k1<2)&&(k1-y<2))
            return;
        if ((((x==ra)^(y!=r1))||(sp1(x,y)))&&(((x==rb)^(y!=r2))||(sp2(x,y))))
            flg=1;
    }

int main()
    {
        scanf(" %c%d",&ra,&r1);
        scanf(" %c%d",&rb,&r2);
        scanf(" %c%d",&ka,&k1);
        scanf(" %c%d",&kb,&k2);
        flg=0;
        chk(kb-1,k2-1);
        chk(kb-1,k2);
        chk(kb-1,k2+1);
        chk(kb,k2-1);
        chk(kb,k2);
        chk(kb,k2+1);
        chk(kb+1,k2-1);
        chk(kb+1,k2);
        chk(kb+1,k2+1);
        if (flg)
            printf("OTHER\n");
        else
            printf("CHECKMATE\n");
        return 0;
    }
