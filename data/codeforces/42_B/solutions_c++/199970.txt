#include <cstdio>
char a[2], b[2], c[2], d[2];
int i, j, a1, b1, c2, d1, c1, d2, b2, a2, e[10][10];
bool p = false;
void f1(int x, int y) {
    for (i = x + 1; e[i][y] < 2 && i < 9; i++) e[i][y] = 1; e[i][y] = 1;
    for (i = x - 1; e[i][y] < 2 && i > 0; i--) e[i][y] = 1; e[i][y] = 1;
    for (i = y + 1; e[x][i] < 2 && i < 9; i++) e[x][i] = 1; e[x][i] = 1;
    for (i = y - 1; e[x][i] < 2 && i > 0; i--) e[x][i] = 1; e[x][i] = 1;
}
void f2(int x, int y) {
    for (i = x - 1; i <= x + 1; i++)
        for (j = y - 1; j <= y + 1; j++)
            e[i][j] = 4;
}
void f3(int x, int y) {
    for (i = x - 1; i <= x + 1; i++)
        for (j = y - 1; j <= y + 1; j++)
            if (e[i][j] == false)
                p = true;
}
int main ()
{
    for (i = 0; i < 10; i++)
        e[i][0] = e[0][i] = e[i][9] = e[9][i] = 4;
    scanf("%s %s %s %s", a, b, c, d);
    a1 = a[0] - 96; a2 = a[1] - 48;
    b1 = b[0] - 96; b2 = b[1] - 48;
    c1 = c[0] - 96; c2 = c[1] - 48;
    d1 = d[0] - 96; d2 = d[1] - 48;
    e[c1][c2] = 3;
    f1(a1, a2);
    f1(b1, b2);
    f2(c1, c2);
    f3(d1, d2);
    if (p) printf("OTHER");
    else printf("CHECKMATE");
}
