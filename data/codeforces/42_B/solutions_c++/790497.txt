#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int MaxN = 9;
const int dx[4] = { -1, 1, 0, 0 };
const int dy[4] = { 0, 0, -1, 1 };
int mp[MaxN][MaxN];
int x[MaxN], y[MaxN];
char a[5], b[5], c[5], d[5];

bool in(int x, int y)
{
	return x >= 0 && x < 8 && y >= 0 && y < 8;
}

bool doit()
{
	int nx, ny;

	for (int k = 0; k < 2; k++)
	{
		for (int i = 0; i < 4; i++)
		{
			nx = x[k] + dx[i];
			ny = y[k] + dy[i];
			while (in(nx, ny) && (nx != x[2] || ny != y[2]))
			{
				mp[nx][ny] = 1;
				nx += dx[i];
				ny += dy[i];
			}
		}
	}

	for (int i = -1; i <= 1; i++)
		for (int j = -1; j <= 1; j++)
		{
			nx = x[2] + i;
			ny = y[2] + j;
			if (in(nx, ny)) mp[nx][ny] = 1;
		}

	for (int i = -1; i <= 1; i++)
		for (int j = -1; j <= 1; j++)
		{
			nx = x[3] + i;
			ny = y[3] + j;
			if (in(nx, ny) && !mp[nx][ny]) return false;
		}

	return true;
}

int main()
{
#ifdef __FISH__
	freopen("in.txt", "r", stdin);
#endif
	while (scanf("%s%s%s%s", a, b, c, d) == 4)
	{
		x[0] = a[0] - 'a';
		y[0] = a[1] - '1';
		x[1] = b[0] - 'a';
		y[1] = b[1] - '1';
		x[2] = c[0] - 'a';
		y[2] = c[1] - '1';
		x[3] = d[0] - 'a';
		y[3] = d[1] - '1';
		memset(mp, 0, sizeof(mp));
		if (doit()) puts("CHECKMATE");
		else puts("OTHER");
	}

	return 0;
}
