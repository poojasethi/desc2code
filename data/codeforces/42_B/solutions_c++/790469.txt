#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<iostream>
#include<vector>
#include<string>
#define pb push_back
#define MAXN 100
using namespace std;
bool g[MAXN][MAXN];
int dx[]={-1,-1,-1,0,0,1,1,1},dy[]={-1,0,1,-1,1,-1,0,1};
bool f[MAXN][MAXN];
char ss[4][10];
bool in(int x,int y)
{
	return x>=0&&x<8&&y>=0&&y<8;
}
void read(int &x,int &y,char *str)
{
	x=str[0]-'a';
	y=str[1]-'1';
}
int main()
{
	int x,y,xx,yy;
	bool flag;
	memset(f,false,sizeof(f));
	for(int i=0;i<4;i++){
		scanf("%s",ss[i]);
		read(x,y,ss[i]);
		if(i<3)
			f[x][y]=true;
	}
	memset(g,false,sizeof(g));
	for(int i=0;i<2;i++){
		read(x,y,ss[i]);
		for(int i=y-1;i>=0;i--){
			g[x][i]=true;
			if(f[x][i])
				break;
		}
		for(int i=y+1;i<8;i++){
			g[x][i]=true;
			if(f[x][i])
				break;
		}
		for(int i=x-1;i>=0;i--){
			g[i][y]=true;
			if(f[i][y])
				break;
		}
		for(int i=x+1;i<8;i++){
			g[i][y]=true;
			if(f[i][y])
				break;
		}
	}
	read(x,y,ss[2]);
	for(int i=0;i<8;i++){
		xx=x+dx[i];
		yy=y+dy[i];
		if(in(xx,yy))
			g[xx][yy]=true;
	}
	flag=true;
	read(x,y,ss[3]);
	if(!g[x][y])
		flag=false;
	for(int i=0;i<8;i++){
		xx=x+dx[i];
		yy=y+dy[i];
		if(in(xx,yy)&&!g[xx][yy])
			flag=false;
	}
	if(flag)
		puts("CHECKMATE");
	else
		puts("OTHER");
	return 0;
}
