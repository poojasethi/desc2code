#include <cstdio>
#include <cstring>

char a[4][3], i, j, k, l, d[][2]={{1,1},{1,0},{1,-1},{-1,1},{-1,0},{-1,-1},{0,1},{0,-1}}, x, y;
bool g[8][8], p[8][8];

int main(){
    scanf("%s%s%s%s", a[0], a[1], a[2], a[3]);
    memset(g,0,sizeof g);
    memset(p,1,sizeof p);
    
//    p[a[0][0]-'a'][a[0][1]-'1']=0;
//    p[a[1][0]-'a'][a[1][1]-'1']=0;
    p[a[2][0]-'a'][a[2][1]-'1']=0;
    
    
    for(l=0;l<2;++l){
        i=a[l][0]-'a';
        j=a[l][1]-'1';
        for(k=i+1;k<8&&p[k][j];++k)g[k][j]=1;
        for(k=i-1;k>=0&&p[k][j];--k)g[k][j]=1;
        
        for(k=j+1;k<8&&p[i][k];++k)g[i][k]=1;
        for(k=j-1;k>=0&&p[i][k];--k)g[i][k]=1;
    }
    
    
    // for(i=0;i<8;++i){for(j=0;j<8;++j)printf("%d ", g[i][j]);printf("\n");}
    
    
    i=a[2][0]-'a';
    j=a[2][1]-'1';
    g[i][j]=1;
    for(k=0;k<8;++k){
        x=i+d[k][0];
        y=j+d[k][1];
        if((0<=x&&x<8)&&(0<=y&&y<8)){
            g[x][y]=1;
        }
    }
    
    bool ok=1;
    
    i=a[3][0]-'a';
    j=a[3][1]-'1';
    if(g[i][j]){
        for(k=0;k<8&&ok;++k){
            x=i+d[k][0];
            y=j+d[k][1];
            if((0<=x&&x<8)&&(0<=y&&y<8)){
                ok=g[x][y];
            }
        }
    }else ok=0;
    
    if(ok)printf("CHECKMATE\n");
    else printf("OTHER\n");
    
    return 0;
}
