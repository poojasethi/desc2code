#include<cstdio>
#include<iostream>
#include<algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;

const int lkx[9]={0,0,0,-1,-1,-1,1,1,1},lky[9]={1,0,-1,-1,0,1,-1,0,1},lx[4]={0,0,1,-1},ly[4]={1,-1,0,0};
int lrx[15],lry[15],dau[10][10];
bool kq;

bool trong(int x,int y)
{
	return x>=0&&x<8&&y>=0&&y<8;
}

bool loang(int x,int y)
{
	kq=true;
	rep(i,9)
	{
		int x1=x+lkx[i],y1=y+lky[i];
		if (trong(x1,y1))
		{
			if (dau[x1][y1]==0)
			{
				kq=false;
				dau[x1][y1]=1;
			}
		}
	}
}

char c1,c2,rx1,rx2,ry1,ry2;

void loangxe(int rx,int ry)
{
	rep(i,4)
	{
		int x=rx,y=ry;
		while (true)
		{
			x=x+lx[i];
			y=y+ly[i];
			if (!trong(x,y)||dau[x][y]==2) break;
			dau[x][y]=1;
		}
	}
}

int main()
{

	rx1-='a';ry1-='1';
	rx2-='a';ry2-='1';
	c1-='a';c2-='1';
	loang(c1,c2);
	dau[c1][c2]=2;
	loangxe(rx1,ry1);
	loangxe(rx2,ry2);
	dau[c1][c2]=1;
	scanf("%c%c ",&c1,&c2);
	loang(c1-'a',c2-'1');

}
