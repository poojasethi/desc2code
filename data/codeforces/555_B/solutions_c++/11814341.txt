#include <bits/stdc++.h>
using namespace std;
const int maxn=200005;
typedef long long LL;
#define fi first
#define se second
#define mp make_pair
#define pb push_back
#define rep(i,a,n) for(int i=(a);i<(n);i++)
set<pair<LL,int> >st;
pair<pair<LL,LL>,int>s[maxn];
LL l[maxn],r[maxn];
int Ans[maxn];
int n,m;
int main()
{
//    freopen("in","r",stdin);
    cin>>n>>m;
    for(int i=0;i<n;i++)scanf("%I64d%I64d",l+i,r+i);
    for(int i=0;i<n-1;i++)s[i]=mp(mp(r[i+1]-l[i],l[i+1]-r[i]),i);
    sort(s,s+n-1);
    for(int i=0;i<m;i++){
        pair<LL,LL>now;
        scanf("%I64d",&now.fi);
        now.se=i;
        st.insert(now);
    }
    for(int i=0;i<n-1;i++){
        set<pair<LL,int> >::iterator it=st.lower_bound(mp(s[i].fi.se,-1));
        if(it==st.end()||it->fi>s[i].fi.fi){
            puts("No");
            return 0;
        }
        Ans[s[i].se]=it->se+1;
        st.erase(it);
    }
    puts("Yes");
    for(int i=0;i<n-1;i++)printf("%d%c",Ans[i],i==n-2?'\n':' ');
    return 0;
}
