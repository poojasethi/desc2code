#include<bits/stdc++.h>
using namespace std ;
#define ll long long
#define f first
#define s second
#define mp make_pair

set<pair<pair<ll,ll>,ll> >island ;
set<pair<ll,ll> > br ;
ll ans[200009] ;
set<pair<pair<ll,ll>, ll > >::iterator itt ;
set<pair<ll,ll> >::iterator it ;
int main()
{
  int n,m;
  scanf("%d%d",&n,&m) ;
  ll x,y,px,py ;
  for(int i =0;i<n;i++)
  {
      cin>>x>>y ;
      if(i)
      {
          ll a = y-px ,b = x-py ;
          island.insert(mp(mp(a,b),i)) ;
      }
      px = x ,py = y;
  }
  for(int i =0;i<m;i++)
    cin>>x, br.insert(mp(x,i)) ;
  for(itt  = island.begin();itt!=island.end();itt++)
  {
      it = br.lower_bound(mp(itt->f.s,-1)) ;
      if(it==br.end()|| it->f > itt->f.f)
      {
          cout<<"No";
          exit(0) ;
      }
      ans[itt->s] = it->s ;
      br.erase(it) ;
  }
  cout<<"Yes \n" ;
  for(int i =1;i<n;i++)
   cout<<ans[i]+1<<" ";
}
