#include<bits/stdc++.h>
#define f(i,x,y) for (int i = x; i < y; i++)
#define fd(i,x,y) for(int i = x; i>= y; i--)
#define all(v) (v).begin(), (v).end()
#define clr(A,x) memset(A, x, sizeof A)
#define pb push_back
#define cua(x) (x)*(x)
#define eps (1e-9)
#define oo (1<<30)
using namespace std;
int n,m;
long long lo,ln,ro,rn;
vector<pair<pair<long long,long long>,int> >v;
set<pair<long long,int> >a;
vector<pair<int,int> >res;
int main(){
	cin >> n >> m;
	cin >> lo >> ro;
	f(i,1,n){
		cin >> ln >> rn;
		v.pb(make_pair(make_pair(rn-lo,ln-ro),i));
		lo=ln; ro=rn;
	}
	f(i,0,m){
		long long aux=0;
		cin >> aux;
		a.insert(make_pair(aux,i+1));
	}
	if(m<n-1) { cout << "No" << endl; return 0; }
	sort(all(v));
	std::set<pair<long long,int> >::iterator it;
	f(i,0,v.size()){
		it=a.lower_bound(make_pair(v[i].first.second,-1));
		if(it==a.end()||(*it).first>v[i].first.first) { 
			cout << "No" << endl; 
			return 0; 
		}
		else{
			res.pb(make_pair(v[i].second,(*it).second));
			a.erase(it);
		}
	}
	sort(all(res));
	cout << "Yes" << endl;
	f(i,0,res.size()) cout<<res[i].second<<" ";
	cout << endl;
}
