#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define pii pair<ll,ll>
#define pb push_back
#define ff first
#define ss second

const ll MN=100*100*100;

ll n,m;
set<pii> point;
pii seg[MN];
pair<pii,ll> inter[MN];
bool mark[MN];
pii ans[MN];

int main(){
	cin>>n>>m;
	for (int i=0;i<n;i++){
		cin>>seg[i].ff>>seg[i].ss;
		
	}
	for (int i=1;i<n;i++){
		inter[i-1].ff.ff=seg[i].ss-seg[i-1].ff;
		inter[i-1].ff.ss=seg[i].ff-seg[i-1].ss;
		inter[i-1].ss=i;
	}
	for (int i=0;i<m;i++){
		ll x;
		cin>>x;
		point.insert(pii(x,i));
	}
	sort(inter,inter+n-1);
	/**for (int i=0;i<n-1;i++){
		cout<<inter[i].ff.ff<<" "<<inter[i].ff.ss<<" "<<inter[i].ss<<"\n";
	}/**/
	for (int i=0;i<n-1;i++){
		set<pii>:: iterator e=point.end(),j=point.lower_bound(pii(inter[i].ff.ss,0)),z=point.begin();
		//cerr<<(*z).ss<<endl;
		if (j==e || (*j).ff>inter[i].ff.ff){
			cout<<"No";
			return 0;
		}
		ans[inter[i].ss]=*j;
		point.erase(*j);
	}
	cout<<"Yes\n";
	for (int i=1;i<n;i++)
		cout<<ans[i].ss+1<<" ";
    return 0;
}
