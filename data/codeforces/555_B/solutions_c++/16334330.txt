#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pp pair<ll,int>
#define x first
#define y second
#define mp make_pair
ll maxi[(int)2e5+5];
int ret[(int)2e5+5];
int main(){
	int n,m,i,j;
	vector<pair<pp,int> >v;
	scanf("%d%d",&n,&m);
	ll L,R,l,r;
	for(i=0;i<n;++i){
		cin>>l>>r;
		if(i){
			v.push_back(mp(mp(l-R,0),i));
		}
		maxi[i]= r-L;
		L=l;R=r;
	}
	for(i=0;i<m;++i){
		cin>>l;
		v.push_back(mp(mp(l,1),i+1));
	}
	sort(v.begin(),v.end());
	set<pp >::iterator it;
	set<pp > s;
	memset(ret,-1,sizeof(ret));
	pair<pp,int> u;
	for(i=0;i<v.size();++i){
		u = v[i];
		if(u.x.y==0)
			s.insert(mp(maxi[u.y],u.y));
		else{
			l = u.x.x;
			if(!s.empty()){
				if( (it = s.begin())->x<l){
					cout<<"No\n";return 0;
				}
				ret[it->y]= u.y;
				s.erase(it);
			}
		}
	}
	for(i=1;i<n;++i)
		if(ret[i]==-1){
			cout<<"No\n";return 0;
		}
	cout<<"Yes\n";
	for(i=1;i<n;++i)
		printf("%d ",ret[i]);
	putchar('\n');
}
