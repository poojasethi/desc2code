#include <bits/stdc++.h>

using namespace std;
#define f(i, n) for (int i = 0; i < (int)(n); i++)
#define mp make_pair
#define pb push_back
#define F first
#define S second
typedef long long ll;
typedef pair<ll,ll> LL;

vector<pair<LL,int> > island;
set<pair<ll,int> > bridge;
set<pair<ll,int> > ::iterator it;
int mark[200005];


int main()
{
    ll m,n,a,b,la,lb;
    cin>>n>>m;
    f(i,n){
    cin>>a>>b;
    if(i)island.pb(mp(mp(b-la,a-lb),i));
    la=a; lb=b;
    }

    f(i,m){
    cin>>a;
    bridge.insert(mp(a,i));
    }

    sort(island.begin(),island.end());

   // int i=n-2,j=m-1;
    for(int i=0; i<n-1; i++){
      it =bridge.lower_bound(mp(island[i].F.S,-1));
      if(it==bridge.end() || it->F>island[i].F.F ){
        cout<<"No\n";
        return 0;
      }
      else {
        mark[island[i].S]=it->S;
        bridge.erase(it);
      }
    }


     cout<<"Yes\n";
     for(int i=1; i<n; i++)cout<<mark[i]+1<<" ";

    return 0;
}
