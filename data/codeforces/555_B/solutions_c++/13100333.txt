#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
pair< pair<ll,ll> , ll > a[200005];
pair< ll, ll > b[200005];
set < pair<ll,ll> > s;
ll x[200005],y[200005];
int ans[200005];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	int i,j;
	for(i=1;i<=n;i++)
	{
		cin>>x[i]>>y[i];
	}
	for(i=1;i<=m;i++)
	{
		cin>> b[i].first;
		b[i].second = i;
		s.insert(b[i]);
	}
	for(i=1;i<n;i++)
	{
		ll ma,mi;
		ma = y[i+1] - x[i];
		mi = x[i+1] - y[i];
		a[i] = make_pair(make_pair(ma,mi),i);
	}
	sort(a+1,a+n);
	int np  = 0;
	for(i=1;i<n;i++)
	{
		set< pair<ll,ll> >::iterator it;
		it = s.lower_bound(make_pair(a[i].first.second,-1));
		if(it==s.end())
		{
			np = 1;
			break;
		}
		if(it ->first > a[i].first.first)
		{
			np = 1;
			break;
		}
		ans[a[i].second] = it->second;
		s.erase(it);
	}
	if(np)
	{
		printf("No\n");
	}
	else {
		printf("Yes\n");
		for(i=1;i<n;i++)
		printf("%d ",ans[i]);
	}
	}
