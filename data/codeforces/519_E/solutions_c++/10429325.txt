#include <cstdio>
#include <iostream>
#include <vector>
#define N 100010
using namespace std;
vector <int> to[N];
int n,m,dep[N],f[N][20],size[N],x0,y0;
void dfs(int x,int fa){
	size[x]=1;
	for(int i=0;i<to[x].size();++i)
	    if(to[x][i]!=fa){
	    	dep[to[x][i]]=dep[x]+1;
	    	f[to[x][i]][0]=x;
	    	dfs(to[x][i],x);
	    	size[x]+=size[to[x][i]];
	    }
}
int up(int x,int goal){
	for(int i=16;i>=0;--i)
	    if(dep[f[x][i]]>=goal)x=f[x][i];
	return x;
}
int lca(int x,int y){
	if(dep[x]>dep[y])x=up(x,dep[y]);
	if(dep[y]>dep[x])y=up(y,dep[x]);
	if(x==y)return x;
	for(int i=16;i>=0;--i)
	    if(f[x][i]!=f[y][i])x=f[x][i],y=f[y][i];
	x0=x,y0=y;
	return f[x][0];
}
int main(){
	scanf("%d",&n);
	for(int i=1,a,b;i<n;++i){
		scanf("%d%d",&a,&b);
		to[a].push_back(b);
		to[b].push_back(a);
	}
	dep[1]=1,dfs(1,0);
	for(int i=1;i<=16;++i)
	    for(int j=1;j<=n;++j)
		    f[j][i]=f[f[j][i-1]][i-1];
	scanf("%d",&m);
	while(m--){
		int x,y;
		scanf("%d%d",&x,&y);
		if(x==y){
			printf("%d\n",n);
			continue;
		}
		if((dep[x]+dep[y])&1){
			printf("0\n");
			continue;
		}
		int fa=lca(x,y);
		int l=(dep[x]+dep[y])/2-dep[fa];
		if(dep[x]==dep[y])printf("%d\n",n-size[x0]-size[y0]);
		else if(dep[x]>dep[y])printf("%d\n",size[up(x,dep[x]-l)]-size[up(x,dep[x]-l+1)]);
		else printf("%d\n",size[up(y,dep[y]-l)]-size[up(y,dep[y]-l+1)]);
	}
}
