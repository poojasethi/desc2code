#include<bits/stdc++.h>

using namespace std;

#define pb push_back

vector<int> V[100005];

int n,l,qw,we;
int size[100005],dep[100005];
int lca[100005][20];

int f(int x,int root,int lvl){

	int i;

	size[x]=1;

	dep[x]=lvl;

	lca[x][0]=root;

	for(i=0 ; i<V[x].size() ; i++)
		if(V[x][i]!=root)
			size[x]+=f(V[x][i],x,lvl+1);

	return size[x];
}

int flca(int x,int y){
	
	if(dep[x]<dep[y])
		swap(x,y);

	int t=dep[x]-dep[y],i;

	for(i=0 ; i<17 ; i++)
		if((t>>i)&1)
			x=lca[x][i];

	if(x==y)
		return x;

	for(i=16 ; i>=0 ; i--)
		if(lca[x][i]!=lca[y][i]){
			x=lca[x][i];
			y=lca[y][i];
		}

	qw=x;
	we=y;

	return lca[qw][0];
}

void init(){
	
	int i,j;

	for(i=1 ; i<17 ; i++)
		for(j=1 ; j<=n ; j++)
			lca[j][i]=lca[lca[j][i-1]][i-1];
}

int dist(int x,int y){
	return dep[x]+dep[y]-dep[l]*2;
}

int anc(int x,int t){
	
	int i;

	for(i=0 ; i<17 ; i++)
		if((t>>i)&1)
			x=lca[x][i];

	return x;
}

int main(){

	int x,y,i,m,ans,a,b,p1,p2,len;

	cin>>n;

	for(i=1 ; i<=n-1 ; i++){
		cin>>x>>y;
		V[x].pb(y);
		V[y].pb(x);
	}

	cin>>m;

	f(1,0,0);

	init();
	
	for(i=1 ; i<=m ; i++){
		
		cin>>a>>b;

		l=flca(a,b);

		if(a==b)
			ans=n;

		else if(dist(a,b)&1)
			ans=0;

		else if(dist(a,l)==dist(b,l))
			ans=n-size[qw]-size[we];

		else {
			
			if(dep[a]<dep[b])
				swap(a,b);

			len=dist(a,b)/2;

//			cout<<len<<endl;
	
			p1=anc(a,len);
			p2=anc(a,len-1);

//			cout<<p1<<' '<<p2<<endl;

			ans=size[p1]-size[p2];

		}

		cout<<ans<<endl;

	}
	
	return 0;
}
