#include<cmath>
#include<cstdio>
#include<string>
#include<vector>
#include<cstring>
#include<algorithm>

using namespace std;

const int maxn=100010;

int data[2*maxn],next[2*maxn],head[maxn],d[maxn],fa[maxn][18],s[maxn];

void push(int k,int x,int y)
{
    data[k]=y;next[k]=head[x];head[x]=k;
}

void dfs(int k)
{
    int p=head[k];s[k]=1;
    while (p)
    {
        if (data[p]!=fa[k][0]) {d[data[p]]=d[k]+1;fa[data[p]][0]=k;dfs(data[p]);s[k]+=s[data[p]];}
        p=next[p];
    }
}

int kthfa(int x,int k)
{
    int i=0;
    while (k)
    {
        if (k&1) x=fa[x][i];k>>=1;i++;
    }
    return x;
}

int lca(int x,int y)
{
    x=kthfa(x,d[x]-d[y]);
    if (x==y) return x;
    int k=16;
    while (k>=0)
    {
        if (fa[x][k]!=fa[y][k]) {x=fa[x][k];y=fa[y][k];}
        k--;
    }
    return fa[x][0];
}

int main()
{
    //freopen("e.in","r",stdin);freopen("e.out","w",stdout);
    int n;scanf("%d",&n);
    for (int i=1;i<n;i++)
    {
        int x,y;scanf("%d%d",&x,&y);
        push(2*i-1,x,y);push(2*i,y,x);
    }
    dfs(1);
    for (int i=1;i<17;i++)
        for (int j=1;j<=n;j++)
            fa[j][i]=fa[fa[j][i-1]][i-1];
    int q;scanf("%d",&q);
    while (q--)
    {
        int x,y;scanf("%d%d",&x,&y);
        if (x==y) {printf("%d\n",n);continue;}
        if (d[x]<d[y]) swap(x,y);
        int z=lca(x,y);
        if ((d[x]+d[y])&1) {printf("0\n");continue;}
        int dis=d[x]+d[y]-2*d[z];
        if (d[x]==d[y]) {printf("%d\n",n-s[kthfa(x,dis/2-1)]-s[kthfa(y,dis/2-1)]);continue;}
        printf("%d\n",s[kthfa(x,dis/2)]-s[kthfa(x,dis/2-1)]);
    }
    return 0;
}
