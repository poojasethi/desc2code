#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cmath>
const int N=1e5+1;
using namespace std;
struct st{int to,next;} e[N<<1];
int n,i,j,x,y,cnt,m,s,t,S,T,l[21],fi[N],siz[N],d[N],fa[N][19];
bool vis[N];
void add(int x,int y)
{
	e[++cnt].to=y; e[cnt].next=fi[x]; fi[x]=cnt;
	e[++cnt].to=x; e[cnt].next=fi[y]; fi[y]=cnt;
}

void build(int x)
{
	vis[x]=siz[x]=1;
	for (int i=fi[x];i;i=e[i].next)
		if (!vis[e[i].to])
		{
			d[e[i].to]=d[x]+1;
			fa[e[i].to][0]=x;
			for (int j=1,k=x;j<=17 && k;k=fa[e[i].to][j++]) fa[e[i].to][j]=fa[k][j-1];
			build(e[i].to);
			siz[x]+=siz[e[i].to];
		}
}

int main()
{
		for (i=l[0]=1;i<=20;i++) l[i]=l[i-1]<<1;
		scanf("%d\n",&n);
		for (i=1;i<n;i++)
		{
			scanf("%d%d\n",&x,&y);
			add(x,y);
		}
		build(d[1]=1);
		scanf("%d\n",&m);
		for (;m;m--)
		{
			scanf("%d%d\n",&x,&y);
			if (x==y)
			{
				printf("%d\n",n);
				continue;
			}
			if (d[y]>d[x])
			{
				t=y;
				T=x;
				swap(x,y);
			}
			else t=x,T=y;
			s=0;
			for (;d[x]>d[y];x=fa[x][i])
			{
				for (i=0;d[fa[x][i]]>=d[y];)
				i++;
				s+=l[--i];
			}
			if (s&1)
			{
				printf("0\n");
				continue;
			}
			for (;x!=y;s+=l[max(0,i-1)]<<1,x=fa[x][max(0,i-1)],y=fa[y][max(0,i-1)])
				for (i=0;fa[x][i]!=fa[y][i];i++);
			s>>=1;
			for (S=s;s>1;s-=l[i-1],t=fa[t][i-1])
				for (i=0;s-l[i]>=1;i++);
			if (fa[t][0]==x)
			{
				for (;S>1;S-=l[i-1],T=fa[T][i-1])
					for (i=0;S-l[i]>=1;i++);
				printf("%d\n",n-siz[t]-siz[T]);
			}
			else printf("%d\n",siz[fa[t][0]]-siz[t]);
		}
}
