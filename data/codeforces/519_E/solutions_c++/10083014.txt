#include <cstdio>
#include <algorithm>
using namespace std;

int a,b,c,d,e,f,g,h,i,j,k,l,m,n,r,A,B,tot;
int fa[200005][25],dl[200005],size[200005],deep[200005];
int way[200005],next[200005],last[200005];

inline void make(int a,int b){
	way[++tot]=b;next[tot]=last[a];last[a]=tot;
}

inline void bfs(){
	dl[l=r=1]=1;deep[1]=1;
	for(;l<=r;l++){
		int s=dl[l];
		for(i=last[s];i;i=next[i])if(way[i]!=fa[s][0]){
			fa[way[i]][0]=s;dl[++r]=way[i];deep[way[i]]=deep[s]+1;
		}
	}
	for(i=1;i<=r;i++)
		for(j=1;j<=20;j++)fa[dl[i]][j]=fa[fa[dl[i]][j-1]][j-1];
	for(i=r;i;i--){
		size[dl[i]]++;
		size[fa[dl[i]][0]]+=size[dl[i]];
	}
}

inline int lca(int a,int b){
	if(a==b)return a;
	for(j=20,k=deep[b]-deep[a];j>=0;j--){
		if(k&(1<<j)){b=fa[b][j];k-=(1<<j);}
	}
	if(a==b)return a;
	for(j=20;j>=0;j--){
		if(fa[a][j]!=fa[b][j]){a=fa[a][j];b=fa[b][j];}
	}
	return fa[a][0];
}

int main(){
	scanf("%d",&n);
	for(i=1;i<n;i++){
		scanf("%d%d",&a,&b);
		make(a,b);make(b,a);
	}
	bfs();
	scanf("%d",&m);
	for(;m;m--){
		scanf("%d%d",&a,&b);
		if(a==b){printf("%d\n",n);continue;}
		if(deep[a]>deep[b])swap(a,b);
		c=lca(a,b);k=deep[b]-deep[c]+abs(deep[a]-deep[c]);
		if(k&1){printf("0\n");continue;}k>>=1;
		for(j=20,g=b;j>=0;j--)if(k&(1<<j))g=fa[g][j];
		for(j=20,A=a;j>=0;j--)if((k-1)&(1<<j))A=fa[A][j];
		for(j=20,B=b;j>=0;j--)if((k-1)&(1<<j))B=fa[B][j];
		h=n;
		if(fa[A][0]==g)h-=size[A];
		else h-=(n-size[g]);
		h-=size[B];
		printf("%d\n",h);
	}
	return 0;
}
