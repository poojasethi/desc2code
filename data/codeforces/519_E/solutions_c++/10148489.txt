#include<bits/stdc++.h>
using namespace std;
int a,s,d[100002],d1[100002][23],f[100002],g[100002],q,w,h,k,l,l1,i,n,m;
vector<int> v[100002];
int prep(int pos,int hgh){
    f[pos]=1;
    d[pos]=hgh;
for(int i1=0;i1<v[pos].size();i1++){
    if(v[pos][i1]!=d1[pos][0]){d1[v[pos][i1]][0]=pos;f[pos]+=prep(v[pos][i1],hgh+1);}
}
return f[pos];
}

int lca(int i1,int i2){
if(d[i1]>d[i2]) swap(i1,i2);
for(int i3=20;i3>=0;i3--){
    if(d[i2]-d[i1]>=(1<<i3)) {i2=d1[i2][i3];}
}//cout<<i1<<" "<<i2;
if(i2==i1) return i1;
for(int i3=20;i3>=0;i3--){
    if(d[i2]>(1<<i3) && d1[i2][i3]!=d1[i1][i3]) {i2=d1[i2][i3];i1=d1[i1][i3];}
}
return d1[i1][0];
}

int mid(int pos){
for(int i3=20;i3>=0;i3--){
    if(k/2>=q+(1<<i3)) {pos=d1[pos][i3];q+=(1<<i3);}
}
return pos;
}
main(){
cin>>n;
for(i=0;i<n-1;i++){

    cin>>a>>s;

    v[a].push_back(s);
    v[s].push_back(a);
}
prep(1,1);
//cout<<"*";
for(a=1;a<=20;a++){
for(i=1;i<=n;i++){//cout<<d1[i][0]<<" ";

        d1[i][a]=d1[d1[i][a-1]][a-1];
    }
}
//cout<<"*";
cin>>m;
for(i=0;i<m;i++){
    cin>>a>>s;
    if(a==s) {cout<<n<<endl;continue;}
q=0;w=0;
    k=lca(a,s);
    k=d[a]+d[s]-2*d[k];

    if(k%2==1) {cout<<0<<endl;continue;}
     k--;
    //cout<<"*ST*"<<endl;
    q=0;
    if(d[a]>=d[s])l=mid(a); else l=mid(s);
    if(d[a]==d[s]) {q=0;l1=mid(s);}//cout<<"&"<<l<<" "<<l1<<"&";//cout<<"*"<<k<<"*"<<endl;
    if(d[a]==d[s])cout<<n-f[l]-f[l1]<<endl; else
    cout<<f[d1[l][0]]-f[l]<<endl;
}
}
