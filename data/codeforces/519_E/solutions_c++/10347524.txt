#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
class sp{
  public:
  int q;
  sp *w;
  sp(int q1,sp *w1){
   q=q1;w=w1;}
};
sp *a[100000];
int i=0,mn,s[100000],s1[100000],d[100000],f[100000][17];
void l(int q,int w,int t){
  s[q]=i;
  i++;
  sp *c;
  d[t]=q;
  for(c=a[q];c;c=c->w)
   if(c->q!=w)
    l(c->q,q,t+1);
  int e,r;
  for(e=16,r=1<<16;e>=0;e--,r>>=1)
   if(t<r)
    f[q][e]=-1;
   else
    f[q][e]=d[t-r];
  s1[q]=i;
}
int m(int c,int v){
  int e,r=0,w;
  for(e=16;e>=0;e--){
   w=f[c][e];
   if((w>=0)&&(!((s[w]<=v)&&(v<s1[w])))){
    c=w;r+=1<<e;}}
  if(!((s[c]<=v)&&(v<s1[c])))
   r+=1;
  return r;
}
int fnd(int c,int v){
  int e;
  for(e=16;e>=0;e--)
   if(v>1<<e){
    c=f[c][e];
    v-=1<<e;}
  mn=f[c][0];
  return s1[c]-s[c];
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,o,c,v;
  scanf("%d",&q);
  for(w=0;w<q;w++)
   a[w]=0;
  for(w=1;w<q;w++){
   scanf("%d%d",&e,&r);
   e--;r--;
   a[e]=new sp(r,a[e]);
   a[r]=new sp(e,a[r]);}
  l(0,0,0);
  for(scanf("%d",&o);o;o--){
   scanf("%d%d",&e,&r);
   if(e==r){
    printf("%d\n",q);
    continue;}
   e--;r--;
   if(s[e]>s[r]){
    w=e;e=r;r=w;}
   c=m(e,s[r]);
   v=m(r,s[e]);
   if(((c+v)&1)==1){
    printf("0\n");
    continue;}
   w=(c+v)>>1;
   if(c>v){
    t=-fnd(e,w);
    t+=s1[mn]-s[mn];}
   if(c<v){
    t=-fnd(r,w);
    t+=s1[mn]-s[mn];}
   if(c==v)
    t=q-fnd(e,w)-fnd(r,w);
   printf("%d\n",t);}
  return 0;}
