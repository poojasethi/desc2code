#include<bits/stdc++.h>

using namespace std;

#define N 100005

vector < int > V[N];

int DW[N],D[N],L[N][22],T[N],n,i,u,v,x,y,t,w,q;

void f(int x, int s){

	int i;

	for(i=0;(1 << i) <= s;i++)
		L[x][i] = T[s - (1 << i)];

	DW[x] = 1;
	D[x] = s;
	T[s] = x;

	for(i=0;i<V[x].size();i++)
		if(!DW[ V[x][i] ]){
			f(V[x][i] , s + 1);
			DW[x] += DW[ V[x][i] ];
		}
}
int main(){

	cin >> n;

	for(i=n;--i;){
		cin >> u >> v;
		V[u].push_back(v);
		V[v].push_back(u);
	}

	f(1,0);

	cin >> w;

	while(w--){
		cin >> u >> v;
		if(u == v)
		{ cout << n << endl; continue; }

		if(D[u] > D[v]) x = u, y = v;
		else            y = u, x = v;
	
		t = D[x] - D[y];

		for(i=0;i<22;i++)
			if(t & (1<<i))
				x = L[x][i];

		if(x != y){
			for(i=21;i>=0;i--)
				if(L[x][i] != L[y][i]){
					x = L[x][i];
					y = L[y][i];
				}

			q = L[x][0];
		}
		else q = x;

		t = D[u] + D[v] - 2 * D[q];

		if(t&1)
		{ cout << 0 << endl; continue; }

		t >>= 1;
		t--;

		if(D[u] > D[v]) x = u, y = v;
		else            y = u, x = v;

		for(i=0;i<22;i++)
			if((t>>i) & 1)
				y = L[y][i];

		for(i=0;i<22;i++)
			if((t>>i) & 1)
				x = L[x][i];
		
		if(L[x][0] == q)
			cout << n - DW[x] - DW[y] << endl; 
		else
		    cout << DW[ L[x][0] ] - DW[x] << endl;
	}

	return 0;
}
