#include<iostream>
#include<cmath>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int sum=0,count=0,m=0x7fffffff;
	for(int i=0;i<2*n-1;i++)
	{
		int x;
		cin>>x;
		if(x<0)
			count++;
		x=abs(x);
		sum+=x;
		if(x<m)
			m=x;
	}
	if(count%2==0||n%2==1)
		cout<<sum<<endl;
	else
		cout<<sum-2*m<<endl;
	return 0;
}
