#include<stdio.h>
int main()
{
	int n,i,j,k,a,sum,min;
  scanf("%d",&n);
	
			sum=0,k=0;	 min=2500000;
	for(i=0;i<2*n-1;i++)
	{
		scanf("%d",&a);
		if(a<0){a=-a;k++;}
		sum=sum+a;
	
		if(min>a)
		{min=a;}
	}
	if(n%2==1)printf("%d\n",sum);
	else
		 {
		 	if(k%2==1)printf("%d\n",sum-2*min);
		 	else printf("%d\n",sum);
		
	}
	

}
