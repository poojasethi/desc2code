#include <bits/stdc++.h>
using namespace std;
int a[300];
int main(){
    int n;
    cin>>n;
    for(int i=0;i<2*n-1;i++)cin>>a[i];
    int neg=0;for(int i=0;i<2*n-1;i++)if(a[i]<0)neg++,a[i]*=-1;
    sort(a,a+2*n-1);

    if(neg%2 && n%2==0)a[0]*=-1;

    int sum=0;for(int i=0;i<2*n-1;i++)sum+=a[i];

    cout<<sum<<"\n";

}
