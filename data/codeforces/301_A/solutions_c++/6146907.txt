#include <stdio.h>
#include <stdlib.h>
int main()
{
	int n,x,sum=0,mi,hehe=0;
	scanf("%d",&n);
	for(int i=0;i<2*n-1;i++){
		scanf("%d",&x);
		if(x>=0)
			hehe++;
		if(abs(x)<mi)
			mi=abs(x);
		sum+=abs(x);
	}
	if(n%2==0&&hehe%2==0)
		sum-=2*mi;
	printf("%d\n",sum);
	return 0;
}
