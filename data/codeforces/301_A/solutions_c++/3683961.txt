#include <iostream>
using namespace std;
long n,minn,s,i,k,a[500]={};
int main() {
	cin >> n;minn=10000000;
	for (i=1;i<2*n;i++) {
		cin >> a[i];
		if (a[i]<0) {
			k++;a[i]=-a[i];
		}
		s+=a[i];minn=min(minn,a[i]);
	}
	if ((n%2==1)||(k%2==0)) cout << s << endl;
	else cout << s-2*minn << endl;
	return 0;
}
