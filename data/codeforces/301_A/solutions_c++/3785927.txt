#include <iostream>
using namespace std;
int n, M = 1000, neg, s = 0, n21, e;
int main() {
	cin >> n;
	n21 = 2 * n - 1;
	for(int i = 0; i < n21; i++) {
		cin >> e;
		if(e < 0) neg++, e = -e;
		if(e < M) M = e;
		s += e;
	}
	if(n % 2 == 1 || neg % 2 == 0) cout << s;
	else cout << (s - 2 * M);
	return 0;
}
