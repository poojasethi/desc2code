#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    int n,i,a,s=0,ne=0,x;
    cin>>n;
    for(i=0;i<2*n-1;i++) {cin>>a;
    if(a<0) {a=-a;ne++;}s+=a;
    x=min(x,a);}
    if(ne%2==1&&n%2==0) s-=2*x;
    cout<<s;
    return 0;

}
