#include<cstdio>
#include<iostream>
using namespace std;
int a[300]={0};
int main()
{
int n,tot=0,min=2345,sum=0;
scanf("%d",&n);
for (int i=1;
i<=n*2-1;
i++)
{
scanf("%d",&a[i]);
if (a[i]<0)
{
tot++;
a[i]=0-a[i];
}
sum+=a[i];
if (a[i]<min) min=a[i];
}
if (tot%2 && (n+1)%2) sum-=min*2;
printf("%d",sum);
}
