#include <iostream>
using namespace std;
int main()
{
	int n;
	cin >> n;
	int neg = 0,sum = 0,min = 1001;
	for(int i = 0;i < 2*n-1;i++)
	{
		int a;
		cin >> a;
		if(a < 0) ++neg,a = -a;
		if(a < min) min = a;
		sum += a;
	}
	cout << (neg%2 == 0 || n%2 == 1? sum : sum-2*min) << endl;
}
