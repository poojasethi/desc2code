#include<stdio.h>
int main()
{
	int n,i,sum=0,s=0,tmp,min=1002;
	scanf("%d",&n);
	for(i=0;i<2*n-1;i++)
	{
		scanf("%d",&tmp);
		if(tmp<0)
		{s++;tmp=-tmp;}
		sum+=tmp;
		if(tmp<min)min=tmp;
	}
	if(n%2==0&&s%2)
		printf("%d\n",sum-2*min);
	else printf("%d\n",sum);
	return 0;
}
