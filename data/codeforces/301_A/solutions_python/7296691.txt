n = int(raw_input())
data = map(int, raw_input().split())
cnt = 0
for i in xrange(len(data)):
    if data[i] < 0:
        data[i] *= -1
        cnt += 1

sums = sum(data)
if n % 2 == 1:
    print sums
else:
    if cnt % 2 == 0:
        print sums
    else:
        data = sorted(data)
        print sums - data[0] * 2
