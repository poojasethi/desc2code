n = int(raw_input().split()[0]) # this is what 'n' is
a = map(int, raw_input().split()) # this is the list of the starting numbers

x = a.count(0) # number of 0's

# calc abs sum of list:
counter = 0
for num in a:
	counter += abs(num)

# find smallest negative number and number of negs
num_neg = 0
best = 10**9
for num in a:
	best = min(best, abs(num))
	if num < 0:
		num_neg += 1
		
def func(n,x,counter, num_neg):	

	if x == 0:
		if n % 2 == 1:
			return counter
		else:
			if num_neg % 2 == 1:
				return counter - 2*best
			else:
				return counter		
	else:
		return counter
			
			
print func(n,x,counter,num_neg)
