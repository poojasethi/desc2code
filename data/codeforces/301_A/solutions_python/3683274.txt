r = lambda: map(int, raw_input().split())
n = input()
a = r()
neg = 0
sum = 0
min_abs = 1000
for i in range(len(a)):
  sum += abs(a[i])
  if abs(a[i]) < min_abs:
    min_abs = abs(a[i])
  if a[i] < 0:
    neg += 1
if n % 2 == 0:
  if neg % 2 == 0:
    print sum
  else:
    print sum - 2 * min_abs
else:
  print sum
