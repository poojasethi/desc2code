n = int(raw_input().strip())
nums = [int(i) for i in raw_input().strip().split()]
nums.sort()

best = sum(nums)
i = 0
if n % 2 == 1:
    while i < len(nums):
        nums[i] *= -1
        if sum(nums) > best:
            best = sum(nums)
            i += 1
        else:
            break
else:
    while (i+1) < len(nums):
        nums[i] *= -1
        nums[i+1] *= -1
        if sum(nums) > best:
            best = sum(nums)
            i += 2
        else:
            break

print best
