import sys

def solve(n, data):
    min_val = abs(data[0])
    cnt = 0

    for ind in xrange(len(data)):
        if data[ind]<0:
            data[ind] *= -1
            cnt += 1
        if data[ind] < min_val:
            min_val = data[ind]
    #print cnt, min_val, (cnt % 2 ==0 )

    if cnt % 2 == 0 or n % 2 != 0:
        return sum(data)

    
    return sum(data)-2*min_val
    
n = int(sys.stdin.readline())
data = map(int, sys.stdin.readline().split())

print solve(n ,data)
