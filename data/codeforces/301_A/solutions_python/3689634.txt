#! /usr/bin/env python
#coding=utf-8


n=input()
a=map(int,raw_input().split())
sum=0
min=10000;
tmp=0;
count=0
for i in range(n*2-1):
    if a[i]<0:
        tmp=a[i]*(-1)
        count=count+1
    else:
        tmp=a[i]
    if tmp<min:
        min=tmp
    sum=sum+tmp
if n%2==0:
    if count%2==0:
        print sum
    else:
        print sum-2*min
else:
    print sum
