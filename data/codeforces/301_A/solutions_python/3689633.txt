nums = map(int,raw_input().split())
if n % 2 == 1:
	for i in range(len(nums)):
		if nums[i] < 0:
			nums[i] = -nums[i]
	print sum(nums)
else:
	t = 0
	nums.sort()
	for i in range(len(nums)):
		if nums[i] < 0:
			t += 1
			nums[i] = -nums[i]
		else:
			break
	nums.sort()
	if t % 2 == 1:
		print sum(nums)-2*nums[0]
