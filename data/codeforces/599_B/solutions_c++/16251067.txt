#include <cstdio>
#include <vector>
using namespace std;
int n,m,f[100005],b[100005],tab[100005],f1,f2,ile[100005];
int main()
{
	scanf("%d%d", &n, &m);
	for(int i=1;i<=n;i++)
	{
		scanf("%d", &f[i]);
		ile[f[i]]++;
		tab[f[i]]=i;
	}
	for(int i=1;i<=m;i++)
	{
		scanf("%d", &b[i]);
		if(tab[b[i]]==0)f2=1;
		if(ile[b[i]]>1)f1=1;
	}
	if(f2){
		printf("Impossible");
		return 0;
	}
	if(f1){
		printf("Ambiguity");
		return 0;
	}
	printf("Possible\n");
	for(int i=1;i<=m;i++)
	{
		printf("%d ", tab[b[i]]);
	}
	return 0;
}
