#include<iostream>
#include<algorithm>
using namespace std;
int n,m,a[100011],b[100011],f[100011], cnt[100011];
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;++i){
		cin>>f[i];
		a[f[i]] = i;
		cnt[f[i]]++;
	}
	for(int i=1;i<=m;++i){
		cin>>b[i];
		
	}
	for(int i = 1; i <= m; ++i){
		if(!a[b[i]]){
			cout << "Impossible";
			return 0;
		}
	}
	
	for(int i=1;i<=m;i++){
		if(cnt[b[i]] > 1){
			cout<<"Ambiguity";
			return 0;
		}
	}
	cout<<"Possible"<<endl;
	for(int i=1;i<=m;++i){
		cout<<a[b[i]]<<" ";
	}
	return 0;
}
