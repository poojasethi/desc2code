#include<bits/stdc++.h>
using namespace std;
int n, m, b[1111111], a[1111111], ans[1111111];
int main()
	{
	cin >> n >> m;
	for(int i = 1; i <= n; i++)
		{
		int x;
		cin >> x;
		a[x] = i;
		b[x]++;
		}
	bool q = 0, am = 0;
	for(int i = 1; i <= m; i++)
		{
		int x;
		cin >> x;
		if(!b[x])
			q = 1;
		else if(b[x] == 1)
			ans[i] = a[x]; 
		else
			am = 1;
		}
	if(q)
		puts("Impossible"), exit(0);
	if(am)
		puts("Ambiguity"), exit(0);
	puts("Possible");
	for(int i = 1; i <= m; i++)
			cout << ans[i] << " ";
	}
