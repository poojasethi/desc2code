#include <bits/stdc++.h>
using namespace std;
long t1[100000],t2[100000],d[100000],s[100000],i,n,m,j,q;
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>n>>m;
	for(i=1;i<=n;++i)
	{
		cin>>t1[i];
		s[t1[i]]++;
		d[t1[i]]=i;
		
		
	}
	for(i=1;i<=m;++i)
	{
		cin>>t2[i];
	}
	for(i=1;i<=m;i++)
	{
		if (s[t2[i]]>1) q=1;
		if (d[t2[i]]==0)
		{
			cout<<"Impossible";
			return 0;
		}
	}
	if (q==1)
	{
		cout<<"Ambiguity";
		return 0;
	}
	cout<<"Possible\n";
	for(i=1;i<=m;i++)
	cout<<d[t2[i]]<<' ';
	
}
