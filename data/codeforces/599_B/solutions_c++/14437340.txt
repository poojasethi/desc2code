#include <iostream>
using namespace std;
	int arr[100002],brr[100002],a[100000],n,m,flag,amb;
int main() {
int i;
	flag=amb=0;
	cin>>n>>m;
	for(i=1;i<=n;i++){
	cin>>arr[i];
	if(a[arr[i]]==0)
	a[arr[i]]=i;
	else
	a[arr[i]]=-1;
	
	}
	for(i=1;i<=m;i++){
	cin>>brr[i];

	}
	
for(i=1;i<=m;i++)
{
if(a[brr[i]]==0)
flag=1;
else if(a[brr[i]]==-1)
amb=1;
else
brr[i]=a[brr[i]];

}
	

	if(flag)
	cout<<"Impossible";
	else if(amb)
	cout<<"Ambiguity";
	else
	{
	cout<<"Possible"<<"\n";
	for(i=1;i<=m;i++)
cout<<brr[i]<<" ";
	}
	
	return 0;
}
