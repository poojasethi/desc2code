#include<cstdio>
const int N=100005;
int f[N],b[N],id[N],a[N];
int i,n,m;
bool amb;
bool mul[N];
void solve()
{
	for (i=1;i<=n;i++)
	{
		if (id[f[i]]) mul[f[i]]=1;
		id[f[i]]=i;
	}
	for (i=1;i<=m;i++)
		if (!id[b[i]])
		{
			printf("Impossible");
			return;
		}
		else
		{
			if (mul[b[i]]) amb=1;
			a[i]=id[b[i]];
		}
	if (amb)
	{
		printf("Ambiguity");
		return;
	}
	printf("Possible\n");
	printf("%d",a[1]);
	for (i=2;i<=m;i++) printf(" %d",a[i]);
}
int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++) scanf("%d",&f[i]);
	for (i=1;i<=m;i++) scanf("%d",&b[i]);
	solve();
}
