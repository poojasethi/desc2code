#include<bits/stdc++.h>
using namespace std;
#define pb push_back
vector <int> v[1000000];
int ans[1000000],n,m,x,a;
int main(){
	scanf("%d %d",&n,&m);
	for(int i=1 ;i<=n ; i++){
		scanf("%d",&x);
		v[x].pb(i);
	}
	int cnt=1;
	for(int i=1 ; i <=m ; i++){
		scanf("%d",&a);
		if(v[a].size()>1 && cnt==1)
			cnt=2;
		else{
			if(v[a].size()==0){
				printf("Impossible\n");
				return 0;
			}
			else
				ans[i]=v[a][0];
				
		}
	}
	if(cnt==2)
		printf("Ambiguity\n");
	else{
		printf("Possible\n");
		for(int i=1 ; i<=m ; i++){
			printf("%d ",ans[i]);
		}
	}
	return 0;
}
