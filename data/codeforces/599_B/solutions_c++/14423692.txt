#include<bits/stdc++.h>
using namespace std;
typedef unsigned long long ll;
#define s(n) scanf("%llu",&n)
#define p(n) printf("%llu ",n)
ll freq[100005],pos[100005];
int main()
{ ll m,n,i;
s(m);s(n);
ll f[m+1],b[n+1];
for(i=1;i<=m;++i)
{cin>>f[i];
freq[f[i]]++;
pos[ f[i] ]=i;
}
for(i=1;i<=n;++i)
cin>>b[i];
for(i=1;i<=n;++i) if( freq[ b[i] ] ==0) {cout<<"Impossible";exit(0);}
for(i=1;i<=n;++i) if( freq[ b[i] ] >1) {cout<<"Ambiguity";exit(0);}
cout<<"Possible\n";
for(i=1;i<=n;++i)
cout<<pos [ b[i] ]<<" ";

}
