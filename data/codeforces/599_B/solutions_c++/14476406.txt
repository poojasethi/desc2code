#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	int f[n],b[m];
	int cnt[100010]={0};
	int org[100010]={0};
	for(int i=0;i<n;i++)
	{
		cin>>f[i];
		cnt[f[i]]++;
		org[f[i]]=i+1;
	}
	for(int i=0;i<m;i++)
		cin>>b[i];
	for(int i=0;i<m;i++)
	{
		if(cnt[b[i]]==0)
		{
			cout<<"Impossible";
			return 0;
		}				
	}
	for(int i=0;i<m;i++)
	{
		if(cnt[b[i]]>1)
		{
			cout<<"Ambiguity";
			return 0;
		}				
	}
	cout<<"Possible"<<endl;
	for(int i=0;i<m;i++)
	{
		cout<<org[b[i]]<<" ";
	}
	return 0;
}
