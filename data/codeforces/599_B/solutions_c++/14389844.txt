#include <cstdio>
#include <set>

using namespace std;

#define N 100000

int n,m,f,b;
int a[N+9];
set<int> g[N+9];
bool imp,amb;

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i)
	{
		scanf("%d",&f);
		g[f].insert(i);
	}
	for (int i=1;i<=m;++i)
	{
		scanf("%d",&b);
		if (g[b].empty()) imp=true;
		else if (g[b].size()>1) amb=true;
		else
		{
			a[i]=*g[b].begin();
		}
	}
	if (imp) puts("Impossible");
	else if (amb) puts("Ambiguity");
	else
	{
		puts("Possible");
		for (int i=1;i<=m;++i) printf("%d ",a[i]);
	}
	return 0;
}
