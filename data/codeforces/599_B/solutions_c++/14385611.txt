#include <cstdio>
using namespace std;
int ape[100000],a[100000],f[100000],b[100000];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) ape[i]=0;
	for (int i=1;i<=n;i++) {scanf("%d",&f[i]);if (ape[f[i]]) ape[f[i]]=-1;else ape[f[i]]=i;}
	for (int i=1;i<=m;i++) {scanf("%d",&b[i]);if (ape[b[i]]==0) {printf("Impossible\n");return 0;}}

	  for (int i=1;i<=m;i++)  {a[i]=ape[b[i]];if (ape[b[i]]==-1) {printf("Ambiguity\n");return 0;}}
		printf("Possible\n");
		for (int i=1;i<=m;i++) printf("%d ",a[i]);
	
	return 0;
}
