#include<stdio.h>
#define maxn 100005
int n,m,f[maxn],b[maxn],Map[maxn],tong[maxn],a[maxn];
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%d",&f[i]);
		Map[f[i]]=i;
		tong[f[i]]++;	
	}
	bool amb=false,imp=false;
	for(int i=1;i<=m;i++){
		scanf("%d",&b[i]);
		if(!tong[b[i]])imp=true;
		if(tong[b[i]]>1)amb=true;
		a[i]=Map[b[i]];
	}
	if(!imp&&!amb){
		printf("Possible\n");
		for(int i=1;i<=m;i++)
			printf("%d ",a[i]);
		printf("\n");
	}
	else if(imp)printf("Impossible\n");
	else printf("Ambiguity\n");
	return 0;
}
