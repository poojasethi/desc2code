#include<stdio.h>
int main()
{
	int n,m,t,i,f[100005][2]={0},b[100005],flag=0;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&t);
		f[t][0]++;
		f[t][1]=i;
	}
	for(i=0;i<m;i++)
	{
		scanf("%d",b+i);
		if(!f[b[i]][0])flag=3;
		if(f[b[i]][0]>1&&flag!=3)flag=2;
		else b[i]=f[b[i]][1];		
	}
	if(flag==2)printf("Ambiguity\n");
	if(flag==3)printf("Impossible\n");
	if(!flag){
		printf("Possible\n");
		for(i=0;i<m;i++)
			printf("%d ",b[i]);
	}
	return 0;
}
