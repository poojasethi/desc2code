#include <cstdio>
const int N = 100005;
int n, m, ans, f[N], g[N], h[N], a[N], b[N];
int main() {
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) {
		scanf("%d", f+i);
		g[f[i]] = i; h[f[i]]++;
	}
	for (int i = 0; i < m; i++) {
		scanf("%d", b+i);
		a[i] = g[b[i]];
		if (h[b[i]] > 1) ans |= 1;
		else if (h[b[i]] == 0) ans |= 2;
	}
	if (ans == 0) {
		puts("Possible");
		for (int i = 0; i < m; i++) printf("%d ", a[i]);
	} else if (ans == 1) puts("Ambiguity");
	else puts("Impossible");
	return 0;
}
