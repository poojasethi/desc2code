#include<bits/stdc++.h>
using namespace std;
int main()
{
long long n,m,i;
cin>>n>>m;
long long f[n],b[m],c[100005]={0},d[100005];
for(i=0;i<n;i++)
{
cin>>f[i];
c[f[i]]+=1;
d[f[i]]=i+1;
}
for(i=0;i<m;i++)cin>>b[i];
for(i=0;i<m;i++)
{
if(c[b[i]]==0){cout<<"Impossible"; return 0;}
}
for(i=0;i<m;i++)
{
if(c[b[i]]>1){cout<<"Ambiguity"; return 0;}
}
cout<<"Possible"<<"\n";
for(i=0;i<m;i++){cout<<d[b[i]]<<" ";}
}
