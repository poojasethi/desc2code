#include <cstdio>
#define M 100005

int n,m,f[M],b[M],c[M],d[M];
bool a,p;

int main () {
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i++) {
		scanf("%d", &f[i]);
		c[f[i]]++;
		d[f[i]] = i+1;
	}
	for (int i = 0; i < m; i++) {
		scanf("%d", &b[i]);
	}
	for (int i = 0; i < m; i++) {
		if (c[b[i]] == 0) p = 1;
		else if (c[b[i]] > 1) a = 1;
	}
	if (p) printf("Impossible\n");
	else if (a) printf("Ambiguity\n");
	else {
		printf("Possible\n");
		for (int i = 0; i < m; i++) {
			printf("%d%c", d[b[i]], i<m-1?' ':'\n');
		}
	}
	
	return 0;
}
