#include<iostream>

using namespace std;

int main(){
	int n,m;
	cin >> n >> m;
	int f[n], b[m], fc[n+1], por[n+1];
	for (int i=0; i<=n; i++) fc[i]=0;
	for (int i=0; i<n; i++) {
		cin >> f[i];
		fc[f[i]]++;
	}
	bool amb=false;
	for (int i=0; i<m; i++) {
		cin >> b[i];
		if (fc[b[i]]==0) {
			cout << "Impossible" << endl;
			return 0;
		}
		if (fc[b[i]]>1) amb=true;
	}
	if (amb) {
		cout << "Ambiguity" << endl;
		return 0;
	}

	cout << "Possible" << endl;
	for (int i=0;i<n;i++) por[f[i]]=i;
	for (int i=0;i<m;i++) cout << por[b[i]]+1 << ' ';
	cout << endl;
}
