#include<bits/stdc++.h>
using namespace std;
int id[200005];
int main(){
ios::sync_with_stdio(0);
int n, m;
cin>>n>>m;
for(int i=0;i<n;i++){
int x;cin>>x;
if(id[x])id[x]=-1;
else id[x]=i+1;
}
vector<int> v(m);
bool aaa = 0;
bool bad=0;
for(int i=0;i<m;i++){
int x; cin>>x;
if(id[x]==-1)aaa=1;
else if(id[x])
v[i]=id[x];
else bad=1;
}
if(bad)
puts("Impossible");
else if(aaa)
puts("Ambiguity");
else
{
cout<<"Possible"<<endl;
for(int i=0;i<m;i++)
cout<<v[i]<<" ";
}

}
