#include <bits/stdc++.h>
using namespace std;
typedef vector<int> vi;
typedef vector<vi> vvi;
int main()
{
	int n,m,i,x,f=0;
	cin>>n>>m;
	vi ans;
	vvi fi(100010);
	for(i=0;i<n;++i) {
       cin>>x;
       fi[x].push_back(i);
	}
	for(i=0;i<m;++i) {
		cin>>x;
		if(fi[x].size()==1) {
           ans.push_back(fi[x][0]);}
		else if(fi[x].size()==0) {
			puts("Impossible");
			return 0;}
		else {
			f=1;}}
	if(f) {
	   puts("Ambiguity");}
	else {
		puts("Possible");
		for(i=0;i<m;++i) {
            cout<<ans[i]+1<<" ";}
        cout<<endl;}

}
