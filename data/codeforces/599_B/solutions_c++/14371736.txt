#include <algorithm>
#include <iostream>

using namespace std;

pair<int,int> a[100000]=make_pair(0,0);

int ans[100000];
int main()
{
int n,m;
cin >> n >> m;
for (int i=0;i<n;i++)
{
int x;
cin >> x;
a[x].first++;
a[x].second=i+1;
}
int av=0;
for (int i=0;i<m;i++)
{
int b;
cin >> b;
if (a[b].first==0)
{
cout << "Impossible";
return 0;
}
if (a[b].first>1)
av=1;
else
ans[i]=a[b].second;

}
if (av==1)
cout  << "Ambiguity";
else
{
cout << "Possible\n";
cout << ans[0];
for (int i=1;i<m;i++)
cout << " " << ans[i];
}
return 0;
}
