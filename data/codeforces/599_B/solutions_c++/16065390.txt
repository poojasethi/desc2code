#include<stdio.h>
#include<string.h>
int f[110000],b[110000],x[110000],y[110000];
int main(){
	int n,m,i;
	scanf("%d%d",&n,&m);
	memset(x,0,sizeof(x));
	for(i=1;i<=n;i++)
	{
		scanf("%d",&f[i]);
		x[f[i]]++;
		y[f[i]]=i;
	}
	int flag=0,haha=0;
	for(i=1;i<=m;i++)
	{
		scanf("%d",&b[i]);
		if(x[b[i]]==0)
			flag=1;
		if(x[b[i]]>1)
			haha=1;
	}
	if(flag==1)
		printf("Impossible\n");
	else if(haha==1)
		printf("Ambiguity\n");
	else
	{
		printf("Possible\n");
		for(i=1;i<=m;i++)
			printf("%d ",y[b[i]]);
		printf("\n");
	}
	return 0;
}
