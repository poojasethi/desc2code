#include <iostream>
using namespace std;

int main() {
    // your code goes here
    int n,m,temp;int n1=0;
    cin>>n>>m;
    int f[n],b[m],a[m],flag[n];
    int i,j;
    for(i=1;i<=n;i++){f[i]=-1;flag[i]=0;}
    for(i=1;i<=n;i++)
    {cin>>temp;if(f[temp]!=-1)flag[temp]=1;f[temp]=i;}
    for(i=1;i<=m;i++)
    {cin>>b[i];if(flag[b[i]]==1)n1=1;if(f[b[i]]==-1){cout<<"Impossible";return 0;}a[i]=f[b[i]];}
    if(n1==1){cout<<"Ambiguity";return 0;}
    cout<<"Possible\n";
    for(i=1;i<=m;i++)cout<<a[i]<<" ";
    
    return 0;
}
