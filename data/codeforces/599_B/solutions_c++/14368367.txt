#include<bits/stdc++.h>

using namespace std;

struct st
{
int a; int b;
};

int n,m,f[100001],b[100001];

st x[100001];

int main()
{
	int i,j,k=0;
	
	scanf("%d %d",&n,&m);
	
	for(i=1; i<=n; i++) {scanf("%d",&f[i]); x[f[i]].a++; x[f[i]].b=i;}
	
	for(i=1; i<=m; i++) scanf("%d",&b[i]);
	
	
	for(i=1; i<=m; i++)
	if(x[b[i]].a==0) {printf("Impossible"); return 0;}

	for(i=1; i<=m; i++)
	if(x[b[i]].a>1){printf("Ambiguity"); return 0;}
	
	printf("Possible\n");
	for(i=1; i<=m; ++i)
	printf("%d ",x[b[i]].b);
return 0;
}
