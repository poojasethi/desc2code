#include <bits/stdc++.h>
using namespace std;
int f,b,x[100010],a[100010],pro[100010],pr[100010];
int n,m;
int main()
{
	int p=0;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		cin>>f;
		x[f]=i;
		pro[f]++;
	}
	for(int i=0;i<m;i++)
	{
		cin>>b;
		a[i]=x[b];
		if(pro[b]>1)p=1;
	}
	for(int i=0;i<m;i++)
	{
		if(a[i]==0)
		{
			cout<<"Impossible";
			return 0;
		}
	}
	if(p==1)
	{
		cout<<"Ambiguity";
		return 0;
	}
	cout<<"Possible"<<endl;
	for(int i=0;i<m;i++)cout<<a[i]<<" ";
	return 0;
}
