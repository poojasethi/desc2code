#include <cstdio>

using namespace std;

int n, m, f, b, t[1111111], c[111111], a[111111];
bool amb;
int main() {
    scanf("%d%d", &n, &m);
    for(int i=0; i<n; ++i) scanf("%d", &f), t[f]++, c[f]=i;
    for(int i=0; i<m; ++i) {
        scanf("%d", &b);
        if (t[b] == 0) {printf("Impossible\n"); return 0;}
        if (t[b] == 1) {a[i] = c[b];}
        if (t[b] > 1) {amb = true;}
    }
    if (amb) {printf("Ambiguity\n"); return 0;}
    printf("Possible\n");
    for(int i=0; i<m; ++i) printf("%d%c", a[i]+1, i<m-1 ? ' ' : '\n');
    return 0;
}
