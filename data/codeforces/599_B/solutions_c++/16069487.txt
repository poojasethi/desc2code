#include<stdio.h>
int main(){
	int a[100005],f[100005]={0},b,input,n,m,i,status=1;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i){
		scanf("%d",&input);
		f[input]=*(f+input)?-1:i;
	}
	for(i=1;i<=m;++i){
		scanf("%d",&b);
		if(*(f+b)==-1){
			status=0;
		}
		else if(!*(f+b)){
			printf("Impossible\n");
			return 0;
		}
		else{
			a[i]=*(f+b);
		}
	}
	if(status){
		printf("Possible\n");
		for(i=1;i<=m;++i){
			printf("%d ",*(a+i));
		}
	}
	else{
		printf("Ambiguity\n");
	}
	return 0;
}
