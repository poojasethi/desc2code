#include<stdio.h>
#include<string.h>
int biao[100005],b[100005],a[100005],f[100005],m,n;
int main()
{
	int i,j,k,l,x=0,y=0;
		
	scanf("%d%d",&n,&m);
	memset(biao,0,sizeof(biao));
	for(i=1;i<=n;i++)
	{
	
	scanf("%d",&f[i]);
	biao[f[i]]++;
	a[f[i]]=i;
	}
	
	for(i=1;i<=m;i++)
	{
	scanf("%d",&b[i]);
	if(biao[b[i]]==1) x++;
	else if(biao[b[i]]>1) y++;
	}
	if(x+y<m) printf("Impossible\n");
	else if(x+y==m&&y>0) printf("Ambiguity\n");
	else 
	{
		printf("Possible\n");
		for(i=1;i<=m;i++)
		printf("%d ",a[b[i]]);
	}
	
	
}
