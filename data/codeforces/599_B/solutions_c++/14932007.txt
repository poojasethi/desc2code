#include<stdio.h>
#define M 100005
int mark[M]={0},res[M],pos[M];
int main(){
	int n,m,i,j,k,f,b;
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;i++){
		scanf("%d",&f);
		mark[f]++;
		pos[f]=i;
	}
	int ans=0;
	for(i=1;i<=m;i++){
		scanf("%d",&b);
		if(ans==-1)continue;
		if(mark[b]==0)ans=-1;
		else if(mark[b]>1)ans=1;
		else res[i]=pos[b];
	}
	if(ans==-1)puts("Impossible");
	else if(ans==1)puts("Ambiguity");
	else{
		puts("Possible");
		for(i=1;i<=m;i++)
			printf("%d ",res[i]);
		puts("");	
	}
	return 0;
}
