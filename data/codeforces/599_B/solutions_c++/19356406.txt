#include <cstdio>

int d[120000],e[120000],x[120000],y[120000];
int i,m,n;

int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
		scanf("%d",&x[i]);
	for (i=1;i<=m;i++)
		scanf("%d",&y[i]);
	for (i=1;i<=n;i++)
		d[x[i]]=i;
	for (i=1;i<=m;i++)
	{
		e[y[i]]=1;
		if (! d[y[i]])
		{
			puts("Impossible");
			return 0;
		}
	}
	for (i=1;i<=n;i++)
		if ((e[x[i]]) && (d[x[i]]!=i))
		{
			puts("Ambiguity");
			return 0;
		}
	puts("Possible");
	for (i=1;i<=m;i++)
		printf("%d ",d[y[i]]);
	return 0;
}
