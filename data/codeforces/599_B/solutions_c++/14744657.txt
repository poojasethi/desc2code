#include<iostream>
using namespace std;

int main()
{
 int i,n,m,b[100000],c[100005],d[100005],f[100000];
 cin>>n>>m;
 for(i=0;i<n;i++)
 {
   cin>>f[i];
   c[f[i]]+=1;//count
   d[f[i]]=i+1;//ref
 }
 for(i=0;i<m;i++)
 {
   cin>>b[i];
 }
 for(i=0;i<m;i++)
   if(c[b[i]]==0)
    {cout<<"Impossible"; return 0;}
 for(i=0;i<m;i++)
   if(c[b[i]]>1)
     {cout<<"Ambiguity"; return 0;}
 cout<<"Possible\n";
 for(i=0;i<m;i++)
  cout<<d[b[i]]<<" " ;
   
 return 0;
}
