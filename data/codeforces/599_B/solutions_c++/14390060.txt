#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    int n, m, f[100001], b[100001];
    map<int,int> M;
    cin >> n >> m;
    for(int i=0; i<n; i++)cin >> f[i];
    for(int i=0; i<m; i++)cin >> b[i];
    for(int i=0; i<n; i++)if(M[f[i]]==0)M[f[i]]=i+1;else M[f[i]]=-1;
    for(int i=0; i<m; i++)if(M[b[i]]==0){cout << "Impossible"; return 0;}
    for(int i=0; i<m; i++)if(M[b[i]]==-1){cout << "Ambiguity"; return 0;}
    cout << "Possible\n";
    for(int i=0; i<m; i++)cout << M[b[i]] << ' ';
}
