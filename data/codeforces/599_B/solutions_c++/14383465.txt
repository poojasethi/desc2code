#include <cstdio>
int q[100001],w[100001],used[100001],yer[100001],cevap[100001];
int main(){
	
	int n,m,i;
	scanf("%d %d",&n,&m);
	
	for(i=1;i<=n;i++){
		scanf("%d",&q[i]);
		used[q[i]]++;
		yer[q[i]]=i;		
	}
	for(i=1;i<=m;i++){
		scanf("%d",&w[i]);
		if(used[w[i]]==0){
			puts("Impossible");
			return 0;
		}		
	}
	
	for(i=1 ; i<=m ; i++){
		if(used[w[i]]>1){
			puts("Ambiguity");
			return 0;
		}
		cevap[i]=yer[w[i]];
	}
	puts("Possible");
	for(i=1 ; i<=m ; i++)
		printf("%d ",cevap[i]);
	
	return 0;
}
