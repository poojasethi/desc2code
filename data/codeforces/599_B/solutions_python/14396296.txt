from sys import stdin

n, m = map(int, stdin.readline().rstrip().split(' '))
f = map(int, stdin.readline().rstrip().split(' '))
b = map(int, stdin.readline().rstrip().split(' '))
a = []


def process():
    pos_dict = {}
    ambiguity_flag = False
    for pos, num in enumerate(f):
        pos_dict.setdefault(num, []).append(pos)
    for num in b:
        indices = pos_dict.get(num, [])
        if len(indices) == 0:
            print "Impossible"
            return
        if len(indices) > 1:
            ambiguity_flag = True
        a.append(indices[0]+1)
    if ambiguity_flag:
        print "Ambiguity"
        return
    print "Possible"
    for num in a:
        print num,

process()
