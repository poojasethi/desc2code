from collections import Counter 

n, m = map(int, raw_input().split())
fs = map(int, raw_input().split())
bs = map(int, raw_input().split())

f_counter = Counter(fs)

if any(f_counter[b] == 0 for b in bs):
	print "Impossible"
elif any(f_counter[b] > 1 for b in bs):
	print "Ambiguity"
else:
	f_inverse ={v:i+1 for i,v in enumerate(fs)}
	print "Possible"
	print " ".join(str(f_inverse[b]) for b in bs)
