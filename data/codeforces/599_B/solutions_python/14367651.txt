from math import acos,asin,atan,sqrt,pi,sin,cos,tan,ceil,floor
import re
from operator import itemgetter as ig

n,m=map(int,raw_input().split())
f=map(int,raw_input().split())
b=map(int,raw_input().split())
dic={}
for i in range(len(f)):
	if f[i] not in dic:
		dic[f[i]]=[i+1,]
	else:
		dic[f[i]]+=[i+1,]
imp=False
amb=False
for i in b:
	if i not in dic:
		imp=True
	elif len(dic[i])>1:
		amb=True
if imp:
	print "Impossible"
elif amb:
	print "Ambiguity"
else:
	print "Possible"
	res=[]
	for i in b:
		res.append(str(dic[i][0]))
	print " ".join(res)
exit(0)
