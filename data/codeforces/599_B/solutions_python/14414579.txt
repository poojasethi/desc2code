n,m = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())

ans = [-1]*100001

for i in range(n):
	if ans[a[i]]!=-1:
		ans[a[i]]=-2
	else:
		ans[a[i]]=i+1
for i in b:
	if ans[i]==-1:
		print("Impossible")
		exit(0)
for i in b:
	if ans[i]==-2:
		print("Ambiguity")
		exit(0)

print("Possible")

print " ".join(str(ans[bb])for bb in b)
