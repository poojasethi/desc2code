s = raw_input()
b = s.split()

f_mapping = {}

for i in range(len(f)):
    if f[i] not in f_mapping:
        f_mapping[f[i]] = set()
    f_mapping[f[i]].add(i + 1)

for x in b:
    if x not in f_mapping:
        print 'Impossible'
        sys.exit(0)

for x in b:
    if len(f_mapping[x]) > 1:
        print 'Ambiguity'
        sys.exit(0)

print 'Possible'
print ' '.join(str(list(f_mapping[x])[0]) for x in b)
