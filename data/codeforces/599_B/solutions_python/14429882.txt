from collections import Counter
inputs = lambda: map(int, raw_input().split())
n, m = inputs()
f, b = inputs(), inputs()
func = Counter(f)
if any(func[i] == 0 for i in b):
    print("Impossible")
elif any(func[i] > 1 for i in b):
    print("Ambiguity")
else:
    temp = {j: i + 1 for i, j in enumerate(f)}
    print("Possible\n{}".format(" ".join(str(temp[i]) for i in b)))
