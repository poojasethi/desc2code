n, m = map(int, raw_input().strip().split(' '))
f = map(int, raw_input().strip().split(' '))
b = map(int, raw_input().strip().split(' '))
ind = [0 for i in range(n+10)]
res = [0 for i in range(n+10)]
i = 1
for x in f:
	ind[x] = i
	res[x] += 1
	i += 1
can = True
amb = False
ans = []
for x in b:
	if res[x] > 1:
		amb = True
	elif res[x] == 0:
		can = False
	else:
		ans.append(ind[x])
if not can:
	print "Impossible"
elif amb:
	print "Ambiguity"
else:
	print "Possible"
	for x in ans:
		print x
