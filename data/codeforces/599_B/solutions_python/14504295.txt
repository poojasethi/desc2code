r = lambda: map(int, raw_input().split())
n, m = r()
cnt = {}
for i, x in enumerate(r()):
    if cnt.get(x, 0):
        cnt[x] = -1
    else:
        cnt[x] = i + 1
ans = [cnt.get(x, 0) for x in r()]
if ans.count(0):
    print "Impossible"
elif ans.count(-1):
    print "Ambiguity"
else:
    print "Possible"
    print " ".join(map(str, ans))
