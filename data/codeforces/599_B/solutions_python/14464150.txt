from collections import defaultdict

n, m = map(int, raw_input().split())
fs = map(int, raw_input().split())
bs = map(int, raw_input().split())

dic = defaultdict(list)
for i in range(1, n + 1):
    dic[fs[i - 1]].append(i)

ls = []
ans = 'Possible'
for b in bs:
    s = len(dic[b])
    if s == 0:
        ans = 'Impossible'
        break
    elif 1 < s:
        ans = 'Ambiguity'
    ls.append(dic[b][0])

print ans
if ans == 'Possible':
    print ' '.join(map(str, ls))
