from collections import Counter
n,m = map(int,raw_input().split())
f = map(int,raw_input().split())
b = map(int,raw_input().split())
counter = Counter(f)

if any(counter[z] == 0 for z in b):
    print "Impossible"
elif any(counter[z] > 1 for z in b):
    print "Ambiguity"
else:
    print "Possible"
    f_inv = {v:i+1 for i,v in enumerate(f)}
    print ' '.join(str(f_inv[v]) for v in b)
