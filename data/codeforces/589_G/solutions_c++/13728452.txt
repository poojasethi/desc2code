#include <iostream>
#include <string.h>
#include <stdio.h>
#include <algorithm>
#define maxn 200010
#define ll long long
using namespace std;
int lson[maxn*22],rson[maxn*22],root[maxn],tot;
ll sum[2][maxn*22];
void insert(int proot,int &now,int l,int r,int po) {
    sum[0][now=++tot]=sum[0][proot]+1;
    sum[1][now]=sum[1][proot]+po;
    if(l==r)
    return;
    int mid=(l+r)>>1;
    if(po>mid){
        lson[now]=lson[proot];
        insert(rson[proot],rson[now],mid+1,r,po);
    }
    else{
        rson[now]=rson[proot];
        insert(lson[proot],lson[now],l,mid,po);
    }
}
ll query(int proot,int l,int r,int d){
      if(l==r){
        return 0;
      }
      int mid=(l+r)/2;
      if(d<=mid){
            ll right=rson[proot];
            ll tmp=sum[1][right]-d*sum[0][right];
            return tmp+query(lson[proot],l,mid,d);
      }
      else{
            return query(rson[proot],mid+1,r,d);
      }
}
int main()
{
    //freopen("dd.txt","r",stdin);
    int n,m;
    scanf("%d%d",&m,&n);
    for(int i=1;i<=n;i++){
        int x;
        scanf("%d",&x);
        insert(root[i-1],root[i],0,1000000,x);
    }
    for(int i=1;i<=m;i++){
        int d,r;
        scanf("%d%d",&d,&r);
        int ans=0,mi=1,ma=n,mid;
        while(mi<=ma){
            mid=(mi+ma)>>1;
            ll tmp=query(root[mid],0,1000000,d);
            if(tmp>=r){
                ans=mid;
                ma=mid-1;
            }
            else
                mi=mid+1;
        }
        printf("%d ",ans);
    }
    printf("\n");
    return 0;
}
