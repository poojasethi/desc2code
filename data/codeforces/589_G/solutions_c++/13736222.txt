#include<iostream>
#include<cstdio>
#include<algorithm>
#include<utility>
using namespace std;
#define LL long long
const int MM=201000;
LL ft[MM],bit[MM],ans[MM];
int n,m;
pair<LL,int> a[MM];
pair<pair<LL,LL>,int> q[MM]; //query<<d,r>,idx>

void update(int x, LL val){
    for(int i=x; i<=m; i+=i&-i)bit[i]+=val,ft[i]++;
}

LL rsq(int x,LL d){
    //rsq 1 to i
    LL ans=0,res=0;
    for(int i=x; i>0; i-=i&-i)ans+=bit[i],res+=ft[i];
    return ans-res*d;
}

int main(){
	scanf("%d %d",&n,&m);
    for(int i=1; i<=m; i++)scanf("%I64d",&a[i].first),a[i].second=i;
	for(int i=1; i<=n; i++)scanf("%I64d %I64d",&q[i].first.first,&q[i].first.second),q[i].second=i;
	sort(q+1,q+n+1); 
	sort(a+1,a+m+1);
	LL last=m;
	for(int i=n; i>0; i--){
		LL hah=last;
		if(q[i].first.first!=a[last].first)while(last>0&&q[i].first.first<a[last].first){
			update(a[last].second,a[last].first);
		//	update(1,1,m,a[last].second,a[last].first,1);
			last--;
		}
		LL d=q[i].first.first,r=q[i].first.second,res=m+1;
		int lo=1,hi=m;
		while(hi>=lo){
			int mid=(lo+hi)/2;
//			cout << lo << " " << mid << " " << hi << endl;
			if(rsq(mid,d)>=r)res=mid,hi=mid-1; else lo=mid+1;
		}
//		cout << "!\n";
		ans[q[i].second]=res==m+1?0:res;
	}
	for(int i=1; i<=n; i++)printf("%I64d ",ans[i]);
	return 0;
}
