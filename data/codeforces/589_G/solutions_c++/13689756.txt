#include <iostream>
#include <algorithm>
#include <math.h>
using namespace std;


int n,m,i,T[200005],M,sul,sad,rao[51][1000005],sum[51][1000005],j,x,y,now;

main()
{


 cin>>n>>m;

   for (i=1;i<=m;i++)
    cin>>T[i];

    //M=(int)sqrt(m);
      M=4000;

    sul=(m-1)/M +1;

     for (i=1;i<=m;i++){

        sad=(i-1)/M + 1;

        rao[sad][T[i]]++;
        sum[sad][T[i]] += T[i];
     }

     for (i=1;i<=sul;i++){

       for (j=1000000; j>=0; j--)
        rao[i][j]+=rao[i][j+1],
        sum[i][j]+=sum[i][j+1];
     }


     while (n--){

         cin>>x>>y;
         now=0;

         for (i=1;i<=sul;i++){
             if (now +  sum[i][x] - x*rao[i][x] >= y) break; else
                 now += sum[i][x] - x*rao[i][x];
          }

          if (i==sul+1) { cout<<0<<" "; continue; }

          for (j=1;j<=M && (i-1)*M + j <= m; j++){

            if (T[(i-1)*M + j]>=x) now+=T[(i-1)*M + j]-x;
            if (now>=y) break;
          }

          cout<<(i-1)*M + j<<" "<<endl;
     }

}
