using namespace std;
int val[30][30],cost[30][30],mark[30][30];
string s[30];
int dp[1<<20];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>s[i];
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin>>cost[i][j];
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		{
			int mx=0;
			for(int k=0;k<n;k++)
				if(s[i][j]==s[k][j])
				{
					val[i][j]+=cost[k][j];
					mx=max(mx,cost[k][j]);
					mark[i][j]|=1<<k;
				}
			val[i][j]-=mx;
		}
	int len=(1<<n)-1;
	memset(dp,63,sizeof(dp));
	dp[0]=0;
	for(int i=0;i<len;i++)
	{
		for(int j=0;;j++)
			if(!((i>>j)&1))
			{
				for(int k=0;k<m;k++)
				{
					dp[i|(1<<j)]=min(dp[i|(1<<j)],dp[i]+cost[j][k]);
					dp[i|mark[j][k]]=min(dp[i|mark[j][k]],dp[i]+val[j][k]);
				}
				break;
			}
	}
	cout<<dp[len];
}
