#include <cstdio>
#include <cstring>
#include <vector>

using namespace std;

const int MAXN = 20;
int n, m;
char s[MAXN][MAXN + 1];
int a[MAXN][MAXN];
int dp[1 << MAXN];

int main() {
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i++) {
		scanf("%s", s[i]);
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			scanf("%d", &a[i][j]);
		}
	}
	for (int i = 1; i < (1 << n); i++) {
		dp[i] = 1000000000;
	}
	for (int j = 0; j < m; j++) {
		int tot[256];
		int mx[256];
		int bm[256];
		memset(tot, 0, sizeof tot);
		memset(mx, 0, sizeof mx);
		memset(bm, 0, sizeof bm);
		for (int i = 0; i < n; i++) {
			tot[s[i][j]] += a[i][j];
			mx[s[i][j]] = max(a[i][j], mx[s[i][j]]);
			bm[s[i][j]] |= 1 << i;
			for (int b = 0; b < (1 << n); b++) {
				dp[b | (1 << i)] = min(dp[b | (1 << i)], dp[b] + a[i][j]);
			}
		}
		for (int i = 'a'; i <= 'z'; i++) {
			if (bm[i]) {
				for (int b = 0; b < (1 << n); b++) {
					dp[b | bm[i]] = min(dp[b | bm[i]], dp[b] + tot[i] - mx[i]);
				}
			}
		}
	}
	printf("%d\n", dp[(1 << n) - 1]);
}
