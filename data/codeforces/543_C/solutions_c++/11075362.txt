#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;
int cmp(int i,int j) {return i<j;}
int v[25][25],V[25][25],f[25][25],dp[1<<20],cnt,a[25],i,n,m,j,k;
char s[25][25];
int main()
{
    scanf("%d%d",&n,&m);
    for (i=0; i<n; i++) scanf("%s",s[i]);
    for (i=0; i<n; i++) for (j=0; j<m; j++) scanf("%d",&v[i][j]);
    for (i=0; i<m; i++)
    {
        for (j=0; j<n; j++)
        {
            cnt=0;
            for (k=0; k<n; k++)
            if (s[k][i]==s[j][i]) {a[++cnt]=v[k][i];f[j][i]|=(1<<k);}
            sort(a+1,a+cnt+1,cmp);
            for (k=1; k<cnt; k++) V[j][i]+=a[k];
        }
    }
    for (j=0; j<(1<<n); j++) dp[j]=453266144;
    dp[0]=0;
    for (i=0; i<m; i++)
      for (j=0; j<(1<<n); j++)
      if (dp[j]<453266144)
        for (k=0; k<n; k++)
          if ((1<<k)&j); else
          {
              dp[j|(1<<k)]=min(dp[j|(1<<k)],dp[j]+v[k][i]);
              dp[j|f[k][i]]=min(dp[j|f[k][i]],dp[j]+V[k][i]);
          }
    cout<<dp[(1<<n)-1];
    return 0;
}
