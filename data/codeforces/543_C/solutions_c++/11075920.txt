#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int NN=23;
int cost[NN][NN],f[(1<<20)+111];
int n,m;
char s[NN][NN];
bool qwe=false;
int main()
{
	scanf("%d%d\n",&n,&m);
	int i,j;
	for (i=1;i<=n;i++) scanf("%s\n",s[i]+1);
	for (i=1;i<=n;i++)
		for (j=1;j<=m;j++) scanf("%d",&cost[i][j]);
	memset(f,60,sizeof(f));
	f[0]=0;
	for (int S=1;S<1<<n;S++)
	{
		for (i=1;i<=n;i++)
			if (S&(1<<(i-1))) break;
		int t=i,lowbit=1<<(t-1),&res=f[S];
		for (i=1;i<=m;i++)
		{
			res=min(res,f[S^lowbit]+cost[t][i]);
			int T=0,tmp=0,mm=0;
			for (j=1;j<=n;j++)
				if (s[t][i]==s[j][i])
				{
					T|=1<<(j-1),tmp+=cost[j][i];
					if (cost[j][i]>mm) mm=cost[j][i];
				}
			res=min(res,f[S&(S^T)]+tmp-mm);
		}
	}
	printf("%d\n",f[(1<<n)-1]);
	return 0;
}
