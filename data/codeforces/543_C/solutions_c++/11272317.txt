#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstring>
using namespace std;
#define maxn 23 
char s[23][23];
int f[1<<21],a[23][23];
int n;
inline int lowbit(int x) {
	for (int i=0;i<n;i++) if (!((1<<i)&x)) return i;
}
int main(){
	int m;
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++) scanf("%s",s[i]+1);
	for (int i=0;i<n;i++) 
		for (int j=1;j<=m;j++) scanf("%d",&a[i][j]);
	memset(f,0x3f,sizeof(f));
	f[0]=0;
	for (int i=0;i<(1<<n)-1;i++) {
		int x=lowbit(i);
		for (int j=1;j<=m;j++) {
			f[i|(1<<x)]=min(f[i|(1<<x)],f[i]+a[x][j]);
			int y=0,sum=0,mx=0;
			for (int k=0;k<n;k++) {
				if (s[x][j]==s[k][j]) {
					y|=1<<k;
					sum+=a[k][j];
						mx=max(a[k][j],mx);
				}
			}
			f[i|y]=min(f[i|y],f[i]+sum-mx);
		}
	}
	printf("%d\n",f[(1<<n)-1]);
	return 0;
}
