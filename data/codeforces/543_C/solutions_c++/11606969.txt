#include<cstdio>
#include<cstring>
#include<algorithm>

#define oo (1<<30)

using namespace std;
int n,m;
int f[(1<<20)+10],mem[100][100],c[100][100],a[100][100];
char s[100][100];
int main()
{
	//freopen("C1.in","r",stdin),freopen("C1.out","w",stdout);
	scanf("%d%d\n",&n,&m);
	for (int i=0;i<n;i++)
		scanf("%s",s[i]);
	for (int i=0;i<n;i++)
		for (int j=0;j<m;j++)
			scanf("%d",&a[i][j]);
	for (int i=0;i<n;i++)
		for (int j=0;j<m;j++)
		{
			int mm=-1;
			for (int k=0;k<n;k++)
				if (s[i][j]==s[k][j])
				{
					mem[i][j]|=1<<k;
					c[i][j]+=a[k][j];
					mm=max(mm,a[k][j]);
				}
			c[i][j]-=mm;
		}
	for (int cir=1;cir<1<<n;cir++)
	{
		f[cir]=oo;
		int lowbit=0;
		for (int i=0;i<n;i++)
		{
			if (cir&(1<<i))
			{
				lowbit=i;
				break;
			}
		}
		for (int i=0;i<m;i++)
		{
			f[cir]=min(f[cir],f[cir&(cir^mem[lowbit][i])]+c[lowbit][i]);
			f[cir]=min(f[cir],f[cir^(1<<lowbit)]+a[lowbit][i]);
		}
	}
	printf("%d",f[(1<<n)-1]);
	return 0;
}
