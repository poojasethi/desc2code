#include<bits/stdc++.h>
#define LL long long
using namespace std;
const LL mod=1000000007ll;
int n,k;
LL a[211];
LL dp[2][211][1011]={};
int main()
{
	cin>>n>>k;
	for (int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	sort(a+1,a+n+1);
	a[0]=a[1];a[n+1]=a[n];
	dp[1][0][0]=1;
	for (int i=1;i<=n;i++)
	{
		for (int j=0;j<=n+1;j++)
		{
			for (int w=0;w<=k;w++)
			{
				dp[0][j][w]=dp[1][j][w];
				dp[1][j][w]=0ll;
			}
		}
		for (int j=0;j<=i;j++)
		{
			for (int w=0;w<=k;w++)
			{
				if (dp[0][j][w]==0) continue;
				int nw;
				nw=w+(a[i+1]-a[i])*j;
				if (nw<=k)
				{
					dp[1][j][nw]+=dp[0][j][w]*(j+1);
					dp[1][j][nw]%=mod;
				}
				nw=w+(a[i+1]-a[i])*(j-1);
				if (j>0&&nw<=k)
				{
					dp[1][j-1][nw]+=dp[0][j][w]*j;
					dp[1][j-1][nw]%=mod;
				}
				nw=w+(a[i+1]-a[i])*(j+1);
				if (nw<=k)
				{
					dp[1][j+1][nw]+=dp[0][j][w];
					dp[1][j+1][nw]%=mod;
				}
			}
		}
	}
	LL ans=0ll;
	for (int i=0;i<=k;i++)
	{
		ans+=dp[1][0][i];
		ans%=mod;
	}
	cout<<ans;
	
}
