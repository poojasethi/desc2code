#define MP make_pair
#define ls (i<<1)
#define rs (i<<1|1)
#define md (ll+rr>>1)
#define N (200 + 10)
#define M (2000000 + 10)
#define inf 0x3f3f3f3f
#define ULL unsigned long long 
#define LL long long
#define mod 1000000007


int dp[N][N][1100];
int a[N];
void add(int &u,int v){ 
	u +=v;
	if(u >= mod) u -= mod;
	}


int main() {
	int n;
	int m;
	scanf("%d", &n);
	scanf("%d", &m);
	for(int i = 1; i<= n; ++i)
		scanf("%d", &a[i]);
	sort(a+1, a+n+1);
	dp[0][0][0] = 1;
	for(int i = 0; i < n; ++i) {
		for(int j = 0; j <= i; ++j) {
			for(int k = 0; k <= m; ++k) {
				if(!dp[i][j][k]) continue;
				int nxt = k + (a[i+1]-a[i]) * j;
				if(nxt <= m) { 
					add(dp[i+1][j][nxt], (LL)dp[i][j][k]*(j+1)%mod);
					
					add(dp[i+1][j+1][nxt], dp[i][j][k]);

					if(j) 
					add(dp[i+1][j-1][nxt], (LL)dp[i][j][k]*j%mod);
				}
			}
		}
	}
	int ans = 0;
	for(int i = 0; i <= m; ++i)
		add(ans, dp[n][0][i]);
	printf("%d\n", ans);
}
