#include <cstdio>
#include <algorithm>

const int p = 1000000007;

using namespace std;

int a[202];
int d[2][111][2222];

int main()
{
	int i, j, k, n, m, r = 0;
	scanf("%d%d", &n, &m);
	for (i = 0; i < n; i++) scanf("%d", a + i);
	sort(a, a + n);
	d[0][0][0] = 1;
	for (i = 0; i < n; i++) 
	{
		for (j = 0; j <= i + 1 && i + j < n; j++) for (k = 0; k <= m; k++) d[!(i & 1)][j][k] = 0;
		for (j = 0; j <= i && i + j <= n; j++) for (k = 0; k <= m; k++)
		{
			if (k + j*(a[i] - a[i - 1]) <= m)
			{
				d[!(i & 1)][j][k + j*(a[i] - a[i - 1])] = (d[!(i & 1)][j][k + j*(a[i] - a[i - 1])] + (j + 1LL)*d[i & 1][j][k]) % p;
				d[!(i & 1)][j + 1][k + j*(a[i] - a[i - 1])] = (d[!(i & 1)][j + 1][k + j*(a[i] - a[i - 1])] + d[i & 1][j][k]) % p;
				if(j) d[!(i & 1)][j - 1][k + j*(a[i] - a[i - 1])] = (d[!(i & 1)][j - 1][k + j*(a[i] - a[i - 1])] + (long long)j*d[i & 1][j][k]) % p;
			}
		}
	}
	for (i = 0; i <= m; i++) (r += d[n & 1][0][i]) %= p;
	printf("%d", r);
}
