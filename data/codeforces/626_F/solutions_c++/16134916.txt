#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
void read(ll &v){
	v=0;char c=0;bool p=false;
	for(c=getchar();c<'0'||c>'9';c=getchar())if(c=='-')p=1;
	for(;c<='9'&&c>='0';c=getchar())v*=10,v+=c-'0';
	if(p)v=-v;
}
ll dp[207][107][1307],t,n,k,s[1007],ans;
int main(){
    read(n),read(k);
    for(ll i=1;i<=n;i++)read(s[i]);
    sort(s+1,s+1+n);
    s[n+1]=s[n];dp[1][0][0]=1;
    for(ll i=1;i<=n;i++){
        for(ll j=0;j<=(n/2)+2;j++){
            for(ll l=0;l<=k;l++){
                t=l+j*(s[i+1]-s[i]);if(t<=k)dp[i+1][j][t]+=dp[i][j][l]*(j+1),dp[i+1][j][t]%=mod;
				t=l+(j+1)*(s[i+1]-s[i]);if(t<=k)dp[i+1][j+1][t]+=dp[i][j][l],dp[i+1][j+1][t]%=mod;
				t=l+(j-1)*(s[i+1]-s[i]);if(j&&t<=k)dp[i+1][j-1][t]+=dp[i][j][l]*j,dp[i+1][j-1][t]%=mod;
			}
        }
    }
    for(ll i=0;i<=k;i++)ans+=dp[n+1][0][i],ans%=mod;
    printf("%I64d\n",ans);
}
