#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<iostream>
#define mod 1000000007

using namespace std;

int dp[2][210][2010];
int n,K;
int a[210];

void add(int &x,int y)
{
	x+=y;
	if (x>=mod) x-=mod;
}

int main()
{
	scanf("%d%d",&n,&K);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	memset(dp,0,sizeof(dp));
	dp[0][0][0]=1;
	for (int i=0,cur=0;i<n;i++,cur^=1)
	{
	  memset(dp[cur^1],0,sizeof(dp[cur^1]));
	  for (int j=0;j<=i;j++)
	    for (int k=0;k<=K;k++)
	    {
	    	int _k=k+j*(a[i+1]-a[i]),val=dp[cur][j][k];
	    	if (!val || _k>K) continue;
	    	add(dp[cur^1][j+1][_k],val);
	    	if (j) add(dp[cur^1][j-1][_k],(long long)val*j%mod);
	    	add(dp[cur^1][j][_k],(long long)val*(j+1)%mod);
	    }
	 }
	int ans=0;
	for (int i=0;i<=K;i++) add(ans,dp[n&1][0][i]);
	printf("%d\n",ans);
	return 0;
}
