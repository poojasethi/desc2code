#include<cmath>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>

#ifdef WIN32
#define orz "%I64d"
#else
#define orz "%lld"
#endif
#define N 210
#define K 1010
#define mod 1000000007
using namespace std;
typedef long long LL;
int f[N][N][K],n,k,a[N];
int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	f[0][0][0]=1;
	sort(a+1,a+n+1);
	for (int i=1;i<=n;i++)
		for (int j=0;j<i;j++)
			for (int l=0;l<=k;l++)
			{
				if (!f[i-1][j][l])
					continue;
				int cost=l+(a[i]-a[i-1])*j;
				if (cost>k)
					continue;
				f[i][j][cost]=(f[i][j][cost]+f[i-1][j][l])%mod;
				f[i][j+1][cost]=(f[i][j+1][cost]+f[i-1][j][l])%mod;
				f[i][j][cost]=(f[i][j][cost]+1LL*j*f[i-1][j][l]%mod)%mod;
				if (j)
					f[i][j-1][cost]=(f[i][j-1][cost]+1LL*j*f[i-1][j][l]%mod)%mod;
			}
	int ans=0;
	for (int i=0;i<=k;i++)
		ans=(ans+f[n][0][i])%mod;
	printf("%d\n",ans);
	return 0;
}
