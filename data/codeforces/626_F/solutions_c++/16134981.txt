#include<cstdio>
#include<cstring>
#include<algorithm>
#define ll long long
#define mod (int)(1e9+7)
using namespace std;
int n,m,a[205],f[2][205][1005],ans;
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
	scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	f[0][0][0]=1;
	int now=0;ll t;
	for(int i=1;i<=n;i++)
	{
		now^=1;
		memset(f[now],0,sizeof(f[now]));
		for(int j=0;j<i;j++)
		for(int k=0;k<=m;k++)
		if(t=f[now^1][j][k])
		{
			int tmp=k+j*(a[i]-a[i-1]);
			if(tmp>m)continue;
			(f[now][j][tmp]+=(ll)(j+1)*t%mod)%=mod;
			(f[now][j+1][tmp]+=t)%=mod;
			if(j)(f[now][j-1][tmp]+=(ll)j*t%mod)%=mod;
		}
	}
	for(int i=0;i<=m;i++)
	(ans+=f[now][0][i])%=mod;
	printf("%d",ans);
}
