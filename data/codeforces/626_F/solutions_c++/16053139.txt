#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
const int mm=1000000007;
int ans,lim,i,j,k,n,m,f[211][111][1101],a[211],tmp;
void add(int x,int y,int z,int d){
	f[x][y][z]=(f[x][y][z]+d)%mm;
}
int main(){
	scanf("%d%d",&n,&m);lim=n/2;
	for (i=1;i<=n;i++) scanf("%d",&a[i]);
	sort(a+1,a+1+n);memset(f,0,sizeof(f));
	f[1][0][0]=f[1][1][0]=1;
	for (i=1;i<n;i++)
	 for (j=0;j<=lim;j++)
	  for (k=0;k<=m;k++){
	  	if (!f[i][j][k]) continue;
	  	tmp=k+j*(a[i+1]-a[i]);
	  	if (tmp>m) continue;
	  	if (j){
	  		add(i+1,j,tmp,(ll)f[i][j][k]*(ll)j%mm);
			add(i+1,j-1,tmp,(ll)f[i][j][k]*(ll)j%mm);
		}
	  	add(i+1,j,tmp,f[i][j][k]);
	  	if (j<lim) add(i+1,j+1,tmp,f[i][j][k]);
	  }
	for (ans=i=0;i<=m;i++) ans=(ans+f[n][0][i])%mm;
	printf("%d\n",ans);
	return 0;
}
