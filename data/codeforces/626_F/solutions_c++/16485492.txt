#include <bits/stdc++.h>

using namespace std;

const long p=1000000007;

void add(long &x,long y)
{
    x+=y;
    if (x>=p)
        x-=p;
}

long n,m,mas[250],dp[250][250][1020];

int main()
{
    cin >> n >> m;
	for (long i=1;i<=n;i++)
        cin >> mas[i];
	sort(mas+1,mas+1+n);
	dp[0][0][m]=1;
	for (long i=0;i<n;i++)
        for (long j=0;j<=n;j++)
            for (long k=0;k<=m;k++)
            if (dp[i][j][k]!=0)
            {
                long temp=k-(mas[i+1]-mas[i])*j;
                if (temp<0)
                    continue;
                add(dp[i+1][j][temp],1LL*dp[i][j][k]*(j+1)%p);
                if (j>0)
                    add(dp[i+1][j-1][temp],1LL*dp[i][j][k]*j%p);
                add(dp[i+1][j+1][temp],dp[i][j][k]);
            }
	long ans=0;
	for (long k=0;k<=m;k++)
        add(ans,dp[n][0][k]);
	cout << ans;
	return 0;
}
