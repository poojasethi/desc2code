#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <iostream>
using namespace std;
#define mod 1000000007
int dp[202][202][1010];
int a[222],n,k;
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	a[0]=a[1];
	dp[0][0][0]=1;
	__int64 ans=0;
	for(int i=0;i<n;i++)
	{
		for(long long j=0;j<=i;j++)
		{
			for(int l=0;l<=k;l++)
			{
				if(dp[i][j][l])
				{
					long long tmpl=l+j*(a[i+1]-a[i]);
					if(tmpl<=k)
					{
						if(j>0)
						{
							(dp[i+1][j-1][tmpl]+=dp[i][j][l]*j%mod)%=mod;
						}
						(dp[i+1][j][tmpl]+=dp[i][j][l]*j%mod)%=mod;
						(dp[i+1][j][tmpl]+=dp[i][j][l])%=mod;
						(dp[i+1][j+1][tmpl]+=dp[i][j][l])%=mod;
					}	
				}
			}
		}
	}
	for(int i=0;i<=k;i++)
	{
		ans+=dp[n][0][i];
	}
	ans%=mod;
	printf("%I64d\n",ans);
}
