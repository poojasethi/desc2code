#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define int long long

signed main(){
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	int n,m;
	cin >> n >> m;
	int a[n+1];
	for(int i=1;i<=n;i++) cin >> a[i];
	a[0]=0;
	sort(a+1,a+n+1);
	int d[2][205][1005];
	memset(d,0,sizeof(d));
	d[0][0][0]=1;
	for(int i=1;i<=n;i++){
		memset(d[1],0,sizeof(d[1]));
		for(int j=0;j<=i;j++){
			for(int k=0;k<=m;k++){
				if(k + j*(a[i]-a[i-1]) > m) break;
				d[1][j][k + j*(a[i]-a[i-1])] += (j+1)*d[0][j][k];
				d[1][j][k + j*(a[i]-a[i-1])] %= MOD;
				d[1][j-1][k + j*(a[i]-a[i-1])] += j*d[0][j][k];
				d[1][j-1][k + j*(a[i]-a[i-1])] %= MOD;
				d[1][j+1][k + j*(a[i]-a[i-1])] += d[0][j][k];
				d[1][j+1][k + j*(a[i]-a[i-1])] %= MOD;
			}
		}
		for(int j=0;j<201;j++){
			for(int k=0;k<1001;k++) d[0][j][k] = d[1][j][k];
		}
	}
	int res = 0;
	for(int i=0;i<=m;i++) res += d[0][0][i];
	res %= MOD;
	cout << res << endl;
	return 0;
}
