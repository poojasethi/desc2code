#include<cstdio>
#include<cstring>
#include<algorithm>
#define mod 1000000007
using namespace std;
int n,K,a[210];
long long dp[2][210][1010];
#define upd(a,b) a=(a+b)%mod
int main()
{
	scanf("%d%d",&n,&K);
	for(int i=1; i<=n; i++) scanf("%d",&a[i]);
	sort(a+1,a+1+n);
	dp[0][0][0]=1;
	for(int i=0; i<n; i++)
	{
		int now=i&1,to=(i&1)^1;
		memset(dp[to],0,sizeof(dp[to]));
		for(int j=0; j<=i; j++)
		{
			long long add=1ll*j*(a[i+1]-a[i]);
			if(add>K) continue;
			for(int k=0; k+add<=K; k++)
				if(dp[now][j][k])
				{
					//new a group
					upd(dp[to][j+1][k+add],dp[now][j][k]);
					//new and close a group
					upd(dp[to][j][k+add],dp[now][j][k]);
					//insert a group
					upd(dp[to][j][k+add],1ll*j*dp[now][j][k]%mod);
					//close a group
					if(j) upd(dp[to][j-1][k+add],1ll*j*dp[now][j][k]%mod);
				}
		}
	}
	int now=n&1;
	long long ans=0;
	for(int k=0; k<=K; k++) upd(ans,dp[now][0][k]);
	printf("%I64d\n",ans);
	return 0;
}
