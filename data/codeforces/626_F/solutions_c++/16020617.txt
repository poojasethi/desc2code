#include <cstdio>
#include <cstdlib>
#include <ctime>
#include <cstring>
#include <cmath>
#include <algorithm>

using namespace std;

const int mod = 1e9 + 7;

void upd(int& a, int b)
{
	a += b;
	if (a >= mod) a -= mod;
}

int n, k;
int f[202][202][1005];
int ans = 0, a[1001];

int main( )
{
	scanf("%d %d", &n, &k);
	for (int i = 0; i < n; i ++)
		scanf("%d", &a[i]);

	sort(a, a + n);
	a[n] = 999999;
	f[0][0][0] = 1;
	for (int i = 0; i < n; i ++)
		 for (int j = 0; j <= i; j ++)
			for (int r = 0; r <= k; r ++)
			{
				int now = r - j * a[i];
				if (j && now + a[i] + (j - 1) * a[i + 1] <= k)
					upd(f[i + 1][j - 1][now + a[i] + (j - 1) * a[i + 1]], 1LL * f[i][j][r] * j % mod);
				if (now - a[i] + (j + 1) * a[i + 1] <= k)
					upd(f[i + 1][j + 1][now - a[i] + (j + 1) * a[i + 1]], f[i][j][r]);
				if (now + j * a[i + 1] <= k)
					upd(f[i + 1][j][now + j * a[i + 1]], 1LL * f[i][j][r] * (j + 1) % mod);
			}
	for (int i = 0; i <= k; i ++)
		upd(ans, f[n][0][i]);
	printf("%d\n", ans);
	return 0;
}
