#include<cstdio>
#include<algorithm>
using namespace std;
int a[1200],n,m;
int dp[220][220][1200];
const long long mod=1000000007;
int main()
{
	scanf("%d %d",&n,&m);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	sort(a+1,a+n+1);
	for (int i=n;i>=1;i--)
	{
		a[i]=a[i]-a[i-1];
	}
	dp[0][0][0]=1;
	for (int i=0;i<n;i++)
	{
		for (int j=0;j<=i;j++)
		{
			int cur=a[i+1]*j;
			for (int k=0;k<=m;k++)
			{
				if(k+cur>m)break;
				if(j!=0)
				{
					dp[i+1][j-1][k+cur]+=(long long)((long long)j*dp[i][j][k]%mod);
					dp[i+1][j-1][k+cur]%=mod;
				}
				dp[i+1][j+1][k+cur]+=dp[i][j][k];
				dp[i+1][j+1][k+cur]%=mod;
				dp[i+1][j][k+cur]+=(long long)((long long)(j+1)*dp[i][j][k])%mod;
				dp[i+1][j][k+cur]%=mod;
			}
		}
	}
	int ans=0;
	for (int i=0;i<=m;i++)
	{
		ans+=dp[n][0][i];
		ans%=mod;
	}
	printf("%d\n",ans);
	return 0;
}
