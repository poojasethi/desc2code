#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
typedef long long ll;
#define rep(i,a,b) for(int (i)=(a);(i)<=(b);(i)++)
#define per(i,a,b) for(int (i)=(a);(i)>=(b);(i)--)
template<class T>inline void rread(T&num){
	num=0;T f=1;char ch=getchar();
	while(ch<'0'||ch>'9'){if(ch=='-')f=-1;ch=getchar();}
	while(ch>='0'&&ch<='9')num=num*10+ch-'0',ch=getchar();
	num*=f;
}
const int mod=1000000007;
const int maxn=205;
int n,maxk,f[maxn][maxn][1005],a[maxn],res=0;
inline void update(int &x,ll y){x=(x+(ll)y%mod)%mod;}
int main(){
	rread(n);rread(maxk);
	rep(i,0,n-1)rread(a[i]);
	sort(a,a+n);reverse(a,a+n);
	f[0][0][0]=1;
	rep(i,0,n-1)rep(j,0,i)rep(k,0,maxk)if(f[i][j][k]){
		int s=j*a[i]+k,res=f[i][j][k];
		if(j>0&&s-a[i]-(j-1)*a[i+1]<=maxk)update(f[i+1][j-1][s-a[i]-(j-1)*a[i+1]],(ll)res*j);
		if(s+a[i]-(j+1)*a[i+1]<=maxk)update(f[i+1][j+1][s+a[i]-(j+1)*a[i+1]],res);
		if(s-j*a[i+1]<=maxk)update(f[i+1][j][s-j*a[i+1]],(ll)res*(j+1));
	}
	rep(i,0,maxk)update(res,f[n][0][i]);
	printf("%d\n",res);
	return 0;
}
