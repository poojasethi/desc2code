const int N=200+5;

int A[N],dp[N][N][1005];

int f(int x,int gr,int cost){
	if(cost>k or gr<0)
		return 0;
	if(x==n+1)
		return gr==0;
	int &r=dp[x][gr][cost];
	if(r!=-1)
		return r;
	return r=((f(x+1,gr+1,cost+gr*(A[x]-A[x-1]))+f(x+1,gr,cost+gr*(A[x]-A[x-1]))*(ll)(gr+1)%mod)%mod+f(x+1,gr-1,cost+gr*(A[x]-A[x-1]))*(ll)gr%mod)%mod;
}

main(){

	scanf("%d %d",&n,&k);

	for(i=1 ; i<=n ; i++)
		scanf("%d",A+i);

	sort(A+1,A+n+1);

	memset(dp,-1,sizeof dp);

	printf("%d",f(1,0,0));
