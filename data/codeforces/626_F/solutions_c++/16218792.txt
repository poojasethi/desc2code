#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#define N 205
#define M 1005
#define mo 1000000007
#define LL long long
#define oo (1<<30)
using namespace std;
int n,m,a[N],f[N][N][M];
int main()
{
	int i,j,k;
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	f[0][0][0]=1;
	for(i=1;i<=n;i++)
		for(j=0;j<=i;j++)
			for(k=0;k<=m;k++){
				if(k>=(j+1)*(a[i]-a[i-1])&&j+1<=i-1)
			  	f[i][j][k]+=1LL*(j+1)*f[i-1][j+1][k-(j+1)*(a[i]-a[i-1])]%mo;
				if(f[i][j][k]>=mo) f[i][j][k]-=mo;
				if(k>=j*(a[i]-a[i-1]))
	  			f[i][j][k]+=1LL*(j+1)*f[i-1][j][k-j*(a[i]-a[i-1])]%mo;
			  if(f[i][j][k]>=mo) f[i][j][k]-=mo;
				if(j>=1&&k>=(j-1)*(a[i]-a[i-1]))
				  f[i][j][k]+=f[i-1][j-1][k-(j-1)*(a[i]-a[i-1])];
				if(f[i][j][k]>=mo) f[i][j][k]-=mo;
			  }
	int ans=0;
	for(i=0;i<=m;i++)
		(ans+=f[n][0][i])%=mo;
	cout<<ans;
	return 0;
}
