#include<bits/stdc++.h>
#define N 205
#define M 1005
using namespace std;
const int P=1e9+7;
int a[N],f[N][N/2][M];
int n,L,i,c,j,k,p,now,ans;
inline void up(int &x){if (x>=P) x-=P;}
int main(){
  scanf("%d%d",&n,&L);
  for (i=1;i<=n;i++)
    scanf("%d",&a[i]);
  sort(a+1,a+n+1);
  f[0][0][0]=1;
  //到第i个点，还剩下j个max，和为k
  for (i=0;i<n;i++)
    for (j=0;j<=n/2;j++)
      for (k=0;k<=L;k++)
        if (now=f[i][j][k]){
          c=k+(a[i+1]-a[i])*j;
          if (c>L) continue;
          //单独一个点/随便放入一个min里 
          up(f[i+1][j][c]+=1ll*now*(j+1)%P);
          //增加一个min
          up(f[i+1][j+1][c]+=now);
          //减去一个min 
          if (j>0)
            up(f[i+1][j-1][c]+=1ll*now*j%P);
        }
  for (k=0;k<=L;k++)
    up(ans+=f[n][0][k]);
  printf("%d\n",ans);
}
