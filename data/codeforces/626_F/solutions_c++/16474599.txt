typedef long long LL;
typedef pair<int,int> pii;

const int mod = 1000000007;

int n,K;
int A[202];
int dp[202][102][1002];
int f(int k,int op,int s){
	if(s>K || op<0 || op>100) return 0;
	if(k==n+1) return op==0;
	if(dp[k][op][s]) return dp[k][op][s]-1;
	int sold=s;
	s+=(A[k]-A[k-1])*op;
	LL  ret=1LL*f(k+1,op,  s)*(op+1)+
		    f(k+1,op+1,s)+
		    1LL*f(k+1,op-1,s)*op;
	ret%=mod;
	dp[k][op][sold]=ret+1;
	return ret;
}

int main(){
	scanf("%d%d",&n,&K);
	fru(i,n) scanf("%d",&A[i+1]);
	sort(A+1,A+n+1);
	printf("%d\n",f(1,0,0));
	return 0;
}
