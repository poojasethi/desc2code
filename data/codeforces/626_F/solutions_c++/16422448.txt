#include <cstdio>
#include <cstring>
#include <algorithm>
const int RXD = 1e9+7;
using namespace std;
int a[1010],f[2][210][1010],n,ans,K;
int read(){
	int x = 0,f = 1,c = getchar();
	while (c<'0' || c>'9') {if (c=='-') f = -1;c = getchar();}
	while (c>='0' && c<='9') x = x*10+c-'0',c = getchar();
	return x*f;
}
void Update(int &x,int y) {(x+=y)%=RXD;}
int main(){
	n = read();K = read();
	for (int i=1;i<=n;i++) a[i] = read();
	sort(a+1,a+1+n);
	f[0][0][0] = 1;
	for (int i=1;i<=n;i++){
		int now = i&1,pre = now^1;
		memset(f[now],0,sizeof(f[now]));
		for (int j=0;j<=n;j++)
			for (int k=0;k<=K;k++)
				if (k+j*(a[i]-a[i-1])<=K) {
					Update(f[now][j][k+j*(a[i]-a[i-1])],f[pre][j][k]);
					if (j>0) {
						Update(f[now][j-1][k+j*(a[i]-a[i-1])],1ll*f[pre][j][k]*j%RXD);
						Update(f[now][j][k+j*(a[i]-a[i-1])],1ll*f[pre][j][k]*j%RXD);
					}
					Update(f[now][j+1][k+j*(a[i]-a[i-1])],f[pre][j][k]);
				}
	}
	for (int i=0;i<=K;i++) Update(ans,f[n&1][0][i]);
	printf("%d\n",ans);
}
