#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int mod=1000000007;
typedef long long ll;

int dp[1010][210],dp2[1010][210],a[210];

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int n,k;scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	sort(a+1,a+n+1);a[0]=a[1];
	memset(dp,0,sizeof(dp));dp[0][0]=1;
	for (int i=1;i<=n;i++)
	{
		memset(dp2,0,sizeof(dp2));
		for (int j=0;j<=k;j++)
			for (int p=0;p<=n;p++)
				if (dp[j][p])
				{
					int t=j+p*(a[i]-a[i-1]);
					if (t>k) continue;
					dp2[t][p]=(dp2[t][p]+dp[j][p])%mod;
					dp2[t][p+1]=(dp2[t][p+1]+dp[j][p])%mod;
					if (p)
					{
						dp2[t][p]=(ll(p)*dp[j][p]+dp2[t][p])%mod;
						dp2[t][p-1]=(ll(p)*dp[j][p]+dp2[t][p-1])%mod;
					}
				}
		for (int j=0;j<=k;j++)
			for (int p=0;p<=n;p++)
			{
				dp[j][p]=dp2[j][p];
			}
	}
	int ans=0;
	for (int i=0;i<=k;i++) ans=(ans+dp[i][0])%mod;	
	printf("%d\n",ans);
	return 0;
}
