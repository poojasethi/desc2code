#include <bits/stdc++.h>
#define ll int

using namespace std;

ll n,k;

ll arr[205];

ll dp[205][205][1005];

const long long int MOD = 1e9+7;

ll solve(ll idx,ll gr,ll imb)
{
    if(gr<0||imb<0)
        return 0;
    if(idx==n)
    {
        if(gr==0)
            return 1;
        return 0;
    }
    if(dp[idx][gr][imb]!=-1)
        return dp[idx][gr][imb];
    long long ans=0;
    ll nw=imb-gr*(arr[idx]-arr[idx-1]);
    if(nw>=0)
    {
        ans=(ans+1LL*(gr+1)*solve(idx+1,gr,nw))%MOD;
        ans=(ans+1LL*gr*solve(idx+1,gr-1,nw))%MOD;
        ans=(ans+1LL*(solve(idx+1,gr+1,nw)))%MOD;
    }
    return dp[idx][gr][imb]=ans;
}
int main()
{
    cin>>n>>k;
    ll i,j;
    memset(dp,-1,sizeof(dp));
    for(i=0;i<n;i++)
        cin>>(arr[i]);
    sort(arr,arr+n);
    cout<<solve(0,0,k);
    return 0;
}
