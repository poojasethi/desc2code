#include<iostream>
#include<cstdio>
#include<algorithm>
#define MOD 1000000007
using namespace std;
int n,K;
int dp[210][210][1010];
int a[205];
bool cmp(int x,int y)
{
	return x>y;
}
int v;
int main()
{
	scanf("%d%d",&n,&K);
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	sort(a,a+n,cmp);
	dp[0][0][0]=1;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<=i;j++)
		{
			for(int k=0;k<=K;k++)
			{
				if(!dp[i][j][k])continue;
				int s=j*a[i]+k,ret=dp[i][j][k];
				if(j>0 && s-a[i]-(j-1)*a[i+1]<=K)
				{
					dp[i+1][j-1][s-a[i]-(j-1)*a[i+1]]=(dp[i+1][j-1][s-a[i]-(j-1)*a[i+1]]+1ll*ret*j%MOD)%MOD;
				}
				if(s+a[i]-(j+1)*a[i+1]<=K)
				{
					dp[i+1][j+1][s+a[i]-(j+1)*a[i+1]]=(dp[i+1][j+1][s+a[i]-(j+1)*a[i+1]]+ret)%MOD;
				}
				if(s-a[i+1]*j<=K)
				{
					dp[i+1][j][s-a[i+1]*j]=(dp[i+1][j][s-a[i+1]*j]+1ll*ret*(j+1)%MOD)%MOD;
				}
			}
		}
	}
	int ans=0;
	for(int k=0;k<=K;k++)
	{
		ans=ans+dp[n][0][k];
		ans=ans%MOD;
	}
	printf("%d\n",ans);
	return 0;
}
