#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
const int base=1e9+7;
int f[202][202][1002];
int a[2000];
int n,m;
void add(int i, int j, int k, int x) {
    f[i][j][k]+=x;
    if (f[i][j][k]>=base) f[i][j][k]-=base;
}
int main() {
    ios_base::sync_with_stdio(0);
    cin >> n >> m;
    for (int i=1;i<=n;i++) cin >> a[i];
    sort(a+1,a+n+1);
    f[1][1][0]=1;
    f[1][0][0]=1;
    for (int i=1;i<n;i++)
    for (int j=0;j<=n;j++)
    for (int k=0;k<=m;k++) {
        if (f[i][j][k]==0) continue;
        int dif=a[i+1]-a[i];
        if (k+j*dif<=m) add(i+1,j+1,k+j*dif,f[i][j][k]);
        if (k+j*dif<=m) add(i+1,j,k+j*dif,f[i][j][k]);
        if (k+j*dif<=m && j!=0) add(i+1,j,k+j*dif,(1ll*f[i][j][k]*j)%base);
        if (k+j*dif<=m && j!=0) add(i+1,j-1,k+j*dif,(1ll*f[i][j][k]*j)%base);
    }
    int ans=0;
    for (int i=0;i<=m;i++) {
        ans+=f[n][0][i];
        if (ans>=base) ans-=base;
    }
    cout << ans << "\n";
    return 0;
}
