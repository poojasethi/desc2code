#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <queue>
#include <vector>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;
int n,k;
int MOD=1000000007;
long long a[555];
long long f[222][122][1111];
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	sort(a,a+n);
	f[0][0][0]=1;
	f[0][1][0]=1;
	for(int i=0;i<n-1;i++)
	{
		for(int j=0;j<=n/2+1;j++)
		{
			for(int l=0;l<=k;l++)
			{
				long long ll=l+1ll*j*(a[i+1]-a[i]);
				if(ll>k)continue;
				if(j!=0)
				{
					f[i+1][j-1][ll]+=1ll*j*f[i][j][l];
					f[i+1][j-1][ll]%=MOD;
				}
				f[i+1][j][ll]+=1ll*(j+1)*f[i][j][l];
				f[i+1][j][ll]%=MOD;
				f[i+1][j+1][ll]+=1ll*f[i][j][l];
				f[i+1][j+1][ll]%=MOD;
			}
		}
	}
	long long ans=0;
	for (int l=0;l<=k;l++)
	{
		ans+=f[n-1][0][l];
		ans%=MOD;
	}
	printf("%I64d",ans);
}
