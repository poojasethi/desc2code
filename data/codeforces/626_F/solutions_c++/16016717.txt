# include <cstdio>
# include <algorithm>
using namespace std;

# define REP(i, n) for (int i = 1; i <= n; ++ i)
# define FOR(i, a, b) for (int i = a; i <= b; ++ i)
# define NR 210
# define p 1000000007

int n, K, a[NR], f[NR][NR][1010], ans;

int main () {
	//freopen ("626E.in", "r", stdin);
	scanf ("%d%d", &n, &K);
	REP (i, n) scanf ("%d", &a[i]);
	sort (a + 1, a + n + 1);
	a[0] = a[1]; 
	f[0][0][0] = 1;
	REP (i, n) {
		FOR (j, 0, i) {
			FOR (k, 0, K) {
				int t0 = k - j * (a[i] - a[i - 1]), t1 = k - (j + 1) * (a[i] - a[i - 1]), t2 = k - (j - 1) * (a[i] - a[i - 1]);
				if (t0 >= 0) f[i][j][k] = 1ll * (j + 1) * f[i - 1][j][t0] % p; 
				if (t1 >= 0) f[i][j][k] = (f[i][j][k] + 1ll * (j + 1) * f[i - 1][j + 1][t1]) % p;
				if (j >= 1 && t2 >= 0) f[i][j][k] = (f[i][j][k] + f[i - 1][j - 1][t2]) % p;
				if (i == n && j == 0) ans = (ans + f[i][j][k]) % p; 
				//printf ("%d %d %d %d %d %d\n", i, j, k, f[i][j][k], f[i - 1][j][t], f[i - 1][j + 1][t]);
			}
		}
	}
	printf ("%d\n", ans);
	return 0;
}
