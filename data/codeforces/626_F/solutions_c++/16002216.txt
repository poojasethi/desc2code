#include <iostream>
#include <map>
#include <string>
#include <algorithm>
using namespace std;

long long mod=1000000007;
int n,kk;
long long dp[2][202][1010];
int arr[202];
int main(){
	cin>>n>>kk;
	for(int i=1;i<=n;i++){
		cin>>arr[i];
	}
	sort(arr+1,arr+1+n);
	dp[0][0][0]=1;
	for(int i=1;i<=n;i++){
		for(int j=0;j<202;j++){
			for(int k=0;k<1010;k++){
				dp[i%2][j][k]=0;
			}
		}
		for(int j=0;j<=n;j++){
			for(int k=0;k<=kk;k++){
				int cost=(arr[i]-arr[i-1])*j;
				int cost2=(arr[i]-arr[i-1])*(j-1);
				int cost3=(arr[i]-arr[i-1])*(j+1);
				if(k-cost>=0){
					dp[i%2][j][k]+=(j+1)*dp[(i+1)%2][j][k-cost];
				}
				if(j>0 && k-cost2>=0){
					dp[i%2][j][k]+=dp[(i+1)%2][j-1][k-cost2];
				}
				if(k-cost3>=0){
					dp[i%2][j][k]+=(j+1)*dp[(i+1)%2][j+1][k-cost3];
				}
				dp[i%2][j][k]%=mod;
			}
		}
	}
	long long sol=0;
	for(int i=0;i<=kk;i++){
		sol+=dp[n%2][0][i];
		sol%=mod;
	}
	cout<<sol<<endl;
}
