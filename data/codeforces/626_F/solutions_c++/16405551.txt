#include <bits/stdc++.h>
using namespace std;
long long dp[2][205][1050],t,ret=0;
int a[205];
const int mod = 1e9+7;
int main(){
	int x,n,k,i,j,temp;
	cin>>n>>k;
	for(i=1;i<=n;++i)cin>>a[i];
	int d=0;
	sort(a+1,a+1+n);
	dp[0][0][0]=1;
	for(i=0;i<n;++i){
		d^=1;memset(dp[d],0,sizeof(dp[d]));
		for(j=0;j<=i;++j)
			for(x=0;x<=k;++x)
				if(t = dp[d^1][j][x]){
					temp = x+j*(a[i+1]-a[i]);
					if(temp>k)break;
					(dp[d][j][temp]+=t*(j+1)%mod)%mod;
					if(j)
						(dp[d][j-1][temp]+=(t*j)%mod)%mod;
					(dp[d][j+1][temp]+=t)%mod;
				}
	}
	for(i=0;i<=k;++i)
		ret= (ret+dp[d][0][i])%mod;
	cout<<ret<<endl;
}
