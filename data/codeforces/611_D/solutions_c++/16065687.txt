#include <cstring>
#include <cstdio>
using namespace std;
const int mod  = 1e9+7;
const int maxn = 5010;

char c[maxn];
int n,nxt[maxn][maxn],s[maxn][maxn],dp[maxn][maxn];

void getnxt(){
    for(int i=n;i>=1;i--){
        for(int j=n;j>i;j--){
            if(c[i]==c[j]) nxt[i][j]=nxt[i+1][j+1]+1;
        }
    }
}

int main(){
	scanf("%d%s",&n,c+1);
	getnxt();
	for(int i=1;i<=n;i++) dp[i][i] = 1;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=i;j++){
			if(c[i-j+1]=='0') continue;
			dp[i][j]=(dp[i][j]+s[i-j][j-1])%mod;
			int l=i-2*j+1,r=i-j+1;
			if(l>=1 && nxt[l][r]<j && c[l+nxt[l][r]]<c[r+nxt[l][r]]){
				dp[i][j]=(dp[i][j]+dp[i-j][j])%mod;
			}
		}
		for(int j=1;j<=n;j++){
			s[i][j]=(s[i][j-1]+dp[i][j])%mod;
		}
	}
	int ans=0;
	for (int i=1;i<=n;i++){
		ans=(ans+dp[n][i])%mod;
	}
	printf("%d",ans);
	return 0;
}
