#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<vector>
using namespace std;
#define N 5005
#define lson l,m,x<<1
#define rson m+1,r,x<<1|1
#define ll long long
const ll mod = 1e9+7;
ll dp[N][N];
char s[N];
int a[N][N];
int main()
{
	int i,j;
	int n;
	cin>>n;
	memset(dp,0,sizeof(dp));
	scanf("%s",s+1);
	int len = strlen(s+1);
	for(i = len;i >= 1;i --)
	{
		for(j = len;j >= 1;j --)
		{
			if(s[i]==s[j]) a[i][j] = a[i+1][j+1]+1;
			else a[i][j] = 0;
		}
	}
	for(i = 0;i <= n;i ++)
		dp[0][i] = 1;
	//dp[1][0] = 1;
	for(i = 1;i <= len;i ++)
	{
		for(j = 1;j <= i;j ++)
		{
			dp[i][j] +=dp[i][j-1];
			if(s[i-j+1] =='0') continue;
			int ss = a[i-j+1][i-j-j+1];
	
			if(i-j-j+1<1) dp[i][j] = (dp[i][j]+dp[i-j][i-j])%mod;
			else if (j > ss&&s[i-j+1+ss]>s[i-j-j+1+ss])
			{
				
				dp[i][j] = (dp[i][j]+dp[i-j][j])%mod;
			}
			
			else
			dp[i][j] = (dp[i][j]+dp[i-j][j-1])%mod;
		//	printf("%d %d %I64d\n",i,j,dp[i][j]); 
		}
	}
	printf("%I64d\n",dp[n][n]); 
}
