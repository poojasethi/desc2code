#include<bits/stdc++.h>
#define ll long long
using namespace std;
struct t{int a,b;};
bool cmp(t x,t y){return (x.a<y.a || (x.a==y.a && x.b>y.b));}

t c[5010];
string s;
int n,m,d[5010];
ll f[5010];
int main(){
    scanf("%d",&n);
    cin>>s;
    f[0]=1;
   	for (int i=0; i<n; i++){
   		d[i]=s[i]-'0';
    	c[i].a=s[i]-'0';
    	c[i].b=i;
    }
    sort(c,c+n,cmp);
    m=n-1;
    for (int i=1; i<=n; i++){
    	for (int j=0; j<=m; j++) if (d[c[j].b]!=0 && c[j].b+i<=n){
    	    f[c[j].b+i]=(ll)(f[c[j].b+i]+f[c[j].b]) % 1000000007;
    	}
    	vector<t> v[11];
    	for (int j=0; j<=n-i; j++) if (c[j].b){
    		c[j].b--;
		    v[d[c[j].b]].push_back(c[j]);
		}
    	m=-1;
    	for (int j=0; j<=9; j++) for (int k=0; k<v[j].size(); k++) {
		    c[++m]=v[j][k];
		}
    }
    printf("%I64d",f[n]);
}
