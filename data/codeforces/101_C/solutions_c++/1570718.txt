using namespace std;

long long ax,ay,bx,by,cx,cy,d;

bool f(long long x,long long y){

  d=cx*cx+cy*cy;
  if(x==0&&y==0)return true;
  if(d==0)return false;
  return (x*cx+y*cy)%d==0&&(x*cy-y*cx)%d==0;

}

int main(){

  cin>>ax>>ay>>bx>>by>>cx>>cy;
  cout<<((f(bx+ax,by+ay)||f(bx-ay,by+ax)||f(bx+ay,by-ax)||f(bx-ax,by-ay))?"YES":"NO")<<endl;

}
