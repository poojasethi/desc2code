#include <cstdio>
long long A,B,x,y,p,q,Fa,mul;

bool c(int Ta,int Tb)
{
     if (p==0 || q==0)
        {
         if (p==0 && q==0) return (Ta==0 && Tb==0);
         if (p==0) return (Ta%q==0 && Tb%q==0);
         if (q==0) return (Ta%p==0 && Tb%p==0);
        }
     else
         {
          mul=(p*p+q*q);
          if ((Ta*p+Tb*q)%mul==0)
             {
              Fa=(Ta*p+Tb*q)/mul;
              if ((Fa*p-Ta)%q==0) return true;
             }
          return false;
         }
}

int main()
{
    scanf("%I64d%I64d%I64d%I64d%I64d%I64d",&A,&B,&x,&y,&p,&q);
    if (c(A-x,B-y) || c(A+x,B+y) || c(A-y,B+x) || c(A+y,B-x)) puts("YES");
    else puts("NO");
}
