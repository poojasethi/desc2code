#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

// equation
//  i cx + j cy = bx
//  i cy - j cx = by
// has a integer solution?
bool solve(ll bx,ll by,ll cx,ll cy){
	if(cx==0 && cy==0) return bx==0 && by==0;

	ll det=-cx*cx-cy*cy;
	return (-cx*bx-cy*by)%det==0 && (cx*by-cy*bx)%det==0;
}

int main(){
	int ax,ay,bx,by,cx,cy; scanf("%d%d%d%d%d%d",&ax,&ay,&bx,&by,&cx,&cy);
	if(solve(bx-ax,by-ay,cx,cy)
	|| solve(bx-ay,by+ax,cx,cy)
	|| solve(bx+ax,by+ay,cx,cy)
	|| solve(bx+ay,by-ax,cx,cy)) puts("YES"); else puts("NO");
	return 0;
}
