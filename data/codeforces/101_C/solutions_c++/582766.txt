#include <iostream>
#include <math.h>
#include <string>
#include <string.h>
#include <vector>

using namespace std;

int main(){
    long long a,b,c,d,x,y;
    cin>> a >> b>> c>> d>>x>>y;
    long long x2[5], y2[5], cx=a, cy=b;
    for(int i=0; i<4; i++){
	x2[i]=c-cx; y2[i]=d-cy;
	int tmp=cx;
	cx=-cy;
	cy=tmp;
    }
    long long anum, den,bnum; 
    bool ans=false;
    for(int i=0; i<4; i++){
	den=x*x+y*y;
	if(den==0){
	    if(x2[i]==0 and y2[i]==0){
		ans=true;
		break;
	    }
	    continue;
	}
	anum=x*x2[i]+y*y2[i];
	bnum=x*y2[i]-y*x2[i];
	if((anum%den==0)&&(bnum%den==0)){
	    ans=true;
	    break;
	}
    }
    if(ans){
	cout << "YES\n";
    }else{
	cout << "NO\n";
    }
}
