#include <iostream>
#include <utility>

using namespace std;

typedef long long ll;

ll mod;

struct pt {
	ll x, y;
	pt(ll x = 0, ll y = 0) : x(x), y(y) {}
	pt operator-(pt b) { return pt(x-b.x, y-b.y); }
	ll operator*(pt b) { return x*b.x + y*b.y; }
} A, B, C;

pt rot(pt a) {
	return pt(-a.y, a.x);
}

long long meumod(long long k)
{
	return (mod + (k%mod))%mod;
}

int main()
{
	cin >> A.x >> A.y >> B.x >> B.y >> C.x >> C.y;
	for(int i = 0; i < 4; i++) {
		A = rot(A);
		if((B-A).x == 0 && (B-A).y == 0) {
			cout << "YES" << endl;
			return 0;
		}
		mod = C*C;
		if(mod) {
			if(!(meumod((B-A)*C)) && !(meumod((B-A)*rot(C)))) {
				cout << "YES" << endl;
				return 0;
			}
		}
	}

	cout << "NO" << endl;
	return 0;
}
