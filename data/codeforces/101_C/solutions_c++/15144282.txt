#include <iostream>
#include <complex>

using namespace std;

typedef long long LL;

int main() {
    LL ar, ai, br, bi, cr, ci;
    cin >> ar >> ai >> br >> bi >> cr >> ci;
    complex<LL> a(ar, ai), b(br, bi), c(cr, ci);

    complex<LL> e(1, 0), i(0, 1);
    for (LL alpha = 0; alpha < 4; alpha++, e *= i) {
        complex<LL> n = b - e * a;
        if (norm(c) == 0)  {
            if (norm(n) == 0) {
                cout << "YES\n";
                return 0;
            }
        }
        else {
            complex<LL> d = n / c;
            if (c * d == n)
            {
               cout << "YES\n";
               return 0;
            }
        }
    }

    cout << "NO\n";
    return 0;
}
