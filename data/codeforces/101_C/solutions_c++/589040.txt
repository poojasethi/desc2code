#include <cstdio>
#define LL long long int
LL x,y,S,T,a,b;
int AC=0;

int f(LL S,LL T){
	if (x==S && y==T) return 1;
	if (a==0 && b==0) return 0;
	
	LL A = b, B = a, C = S-x, D = -a, E = b, F = T-y;
	
	LL DET = A*E-B*D , DX = B*F-C*E , DY = A*F-C*D;
	
	if (DX % DET == 0 && DY % DET == 0) return 1;
	else return 0;
	
}

int main(){
	scanf("%I64d%I64d%I64d%I64d%I64d%I64d",&x,&y,&S,&T,&a,&b);
	
	AC = 0;
	
	AC = f(S,T) || f(-T,S) || f(-S,-T) || f(T,-S);
	
	printf("%s\n",AC?"YES":"NO");
	
	//scanf("\n");
	return 0;
}
