#include <iostream>
using namespace std;

typedef long long ll;

void gira(ll& x, ll& y) {
  swap(x, y);
  x = -x;
}

int main() {
  ll ax, ay, bx, by, cx, cy;
  cin >> ax >> ay >> bx >> by >> cx >> cy;
  
  ll dx = cx, dy = cy;
  gira(dx, dy);
  
  bool ok = false;
  for (int i = 0; i < 4; ++i) {
    ll vx = bx - ax, vy = by - ay;
    
    if (vx == 0 and vy == 0) ok = true;
    
    ll d = cx*dy - dx*cy;
    ll d1 = vx*dy - dx*vy;
    ll d2 = cx*vy - vx*cy;
    
    if (d != 0 and d1%d == 0 and d2%d == 0) ok = true;
    
    gira(ax, ay);
  }
  
  if (ok) cout << "YES" << endl;
  else cout << "NO" << endl;
}
