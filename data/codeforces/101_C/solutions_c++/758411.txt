#include <iostream>

using namespace std;

long long Ax, Ay, Bx, By, Cx, Cy;

bool is_generated(long long a, long long b, long long x, long long y) {
  long long norm = x * x + y * y;
  if (norm == 0)
    return (a == 0 && b == 0);
  return ((a * x + b * y) % norm == 0) && ((b * x - a * y) % norm == 0);
}

int main() {
  cin >> Ax >> Ay >> Bx >> By >> Cx >> Cy;

  bool yes = (is_generated(Ax - Bx, Ay - By, Cx, Cy)
	      || is_generated(-Ay - Bx, Ax - By, Cx, Cy)
	      || is_generated(-Ax - Bx, -Ay - By, Cx, Cy)
	      || is_generated(Ay - Bx, -Ax - By, Cx, Cy));
  cout << (yes ? "YES\n" : "NO\n");
}
