#include <cstdio>
#include <iostream>
using namespace std;
long long x2,y2,x3,y3;
bool elven(long long x1,long long y1)
{
    x1=x2-x1; y1=y2-y1;
    long long det=-x3*x3-y3*y3;
    long long m1=-x1*x3-y1*y3;
    long long m2=x3*y1-x1*y3;
    if (!det)
    {
        if (x1==0 && y1==0) return 1;
        return 0;
    }
    if (m1%det==0 && m2%det==0) return 1;
    return 0;
}
int main()
{
    long long x1,y1;
    while (cin>>x1>>y1>>x2>>y2>>x3>>y3)
    {
        if ( !elven(x1,y1) && !elven(-y1,x1) && !elven(-x1,-y1) && !elven(y1,-x1))
            cout<<"NO"<<endl;
        else cout<<"YES"<<endl;
    }
    return 0;
}
