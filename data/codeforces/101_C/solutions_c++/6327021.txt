#include <iostream>
#include <cstdlib>

using namespace std;

typedef long long int ll;

void llega(ll x1,ll y1,
	   ll x2,ll y2,
	   ll x3,ll y3)
{
  ll x=x2-x1;
  ll y=y2-y1;
  if (x3==0 and y3==0) {
    if (x==0 and y==0) {
      cout<<"YES"<<endl;
      exit(0);
    }
    return;
  }
  ll nextx=x*x3+y*y3;
  ll nexty=-x*y3+y*x3;
  x=nextx;
  y=nexty;
  ll c=x3*x3+y3*y3;
  if (x%c==0 and y%c==0) {
    cout<<"YES"<<endl;
    exit(0);
  }
}

int main()
{
  ll x1,y1,x2,y2,x3,y3;
  cin>>x1>>y1>>x2>>y2>>x3>>y3;
  llega(x1,y1,x2,y2,x3,y3);
  llega(-y1,x1,x2,y2,x3,y3);
  llega(-x1,-y1,x2,y2,x3,y3);
  llega(y1,-x1,x2,y2,x3,y3);
  cout<<"NO"<<endl;
}
