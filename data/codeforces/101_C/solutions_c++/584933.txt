#include <iostream>
#include <string>
#include <sstream>
#include <vector>
#include <set>
#include <queue>
#include <cstring>

using namespace std;

typedef long long LL;

bool ok(LL x1,LL y1,LL x2,LL y2)
{
  LL d=x2*x2+y2*y2;
  LL n=x1*x2+y1*y2;
  if(d==0) return x1==0 && y1==0 ? 1:0;
  return (n%d)==0 ? 1:0;
}

int main()
{
  LL xa,xa2,xb,xc;
  LL ya,ya2,yb,yc;

  cin>>xa>>ya;
  cin>>xb>>yb;
  cin>>xc>>yc;

  for(int i=0;i<4;i++){

    if(ok(xb-xa,yb-ya,xc,yc) && ok(xb-xa,yb-ya,-yc,xc)){
      cout<<"YES"<<endl;
      return 0;
    }

    xa2=-ya; ya2=xa;
    swap(xa,xa2);
    swap(ya,ya2);
  }

  cout<<"NO"<<endl;

  return 0;
}
