#define int64 long long

using namespace std;
int i,j,k,n,m,xa,ya,xb,yb,xc,yc;
inline bool work(int c,int f) {
	if (!xb&&!yb) return !c&&!f;
	int a=xb,b=yb,d=yb,e=-xb;
	int64 s=1ll*c*d-1ll*a*f,v=1ll*b*d-1ll*a*e;
	if (s%v) return 0;
	s=1ll*c*e-1ll*b*f;
	if (s%(-v)) return 0;
	return 1;
}
int main() {
	scanf("%d%d%d%d%d%d",&xa,&ya,&xc,&yc,&xb,&yb);
	if (work(xc-xa,yc-ya)||work(xc+ya,yc-xa)||
		work(xc+xa,yc+ya)||work(xc-ya,yc+xa)) printf("YES\n");
	else printf("NO\n");
	return 0;
}
