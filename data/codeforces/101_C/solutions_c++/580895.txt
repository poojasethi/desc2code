
#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long ll;
bool gogo(ll a,ll b,ll c,ll d,ll e,ll f)
{
   ll mv=d*d+c*c;
   if(mv==0)
   {
      return (a==e && b==f);
   }
   else
   {
      if((d*(e-a)+c*(b-f))%mv==0 && (c*(e-a)+d*(f-b))%mv==0)
	 return true;
      else
	 return false;
   }
}
int main()
{

   ll ax,ay,bx,by,cx,cy;
   cin>>ax>>ay;//initial
   cin>>cx>>cy;//to get
   cin>>bx>>by;//to add
   // with A add B to get C
   if(gogo(ax,ay,bx,by,cx,cy) || gogo(-ay,ax,bx,by,cx,cy) || gogo(-ax,-ay,bx,by,cx,cy) || gogo(ay,-ax,bx,by,cx,cy))
      printf("YES\n");
   else
      printf("NO\n");
   return 0;
}
