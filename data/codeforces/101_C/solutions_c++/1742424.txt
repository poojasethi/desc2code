#include<cstdio>

long long a,b;

bool check(long long x1,long long y1)
{
    if(a==0&&b==0)  return x1==0&&y1==0;
    return (a*y1-b*x1)%(a*a+b*b)==0&&(a*x1+b*y1)%(a*a+b*b)==0;
}

int main()
{
    int x1,y1,x2,y2,x3,y3; 
    scanf("%I64d%I64d%I64d%I64d%I64d%I64d",&x1,&y1,&x2,&y2,&a,&b);
    if(check(x2-x1,y2-y1)||check(x2-y1,y2+x1)||check(x2+x1,y2+y1)||check(x2+y1,y2-x1))  printf("YES\n");
    else    printf("NO\n");
    return 0;
}
