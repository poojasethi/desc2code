#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define ll long long

ll ax,ay,bx,by,x,y,dx,dy,s;

bool f(ll bx,ll by)
{
  ll dx,dy;

  dx=bx-ax;
  dy=by-ay;
  return !((x*dx+y*dy)%s) && !((-y*dx+x*dy)%s);
}
int main()
{
  scanf("%I64d%I64d%I64d%I64d%I64d%I64d",&ax,&ay,&bx,&by,&x,&y);
  s=x*x+y*y;
  if(!s)
  {
    printf("%s",((ax==bx && ay==by) || (-ay==bx && ax==by) || (-ax==bx && -ay==by) || (ay==bx && -ax==by))?"YES":"NO");
	return 0;
  }
  printf("%s",(f(bx,by) || f(-by,bx) || f(-bx,-by) || f(by,-bx))?"YES":"NO");
  return 0;
}
