#include <stdio.h>

#define sq(x) x*x
#define LL long long
LL xa, xb, xc, ya, yb, yc;
LL f( LL a , LL b )
{
    LL r = sq(xc)+sq(yc);
    if( !r )
        return a == xb && b == yb;
    return ((xb-a)*xc + (yb-b)*yc ) % r == 0 && ((a-xb)*yc + (yb-b)*xc) % r == 0;
}
int main()
{
    scanf( "%I64D%I64D%I64D%I64D%I64D%I64D", &xa, &ya, &xb, &yb, &xc, &yc );
    if( f( xa , ya ) || f( ya , -xa ) || f( -xa , -ya ) || f( -ya , xa ) )
        puts("YES");
    else
        puts("NO");
    return 0;
}
