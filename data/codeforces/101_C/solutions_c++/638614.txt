#include<cstdio>
#include<iostream>

using namespace std;

#define llint long long

llint a,b,u,v,x,y,o;

bool check(llint a,llint b)
{
	if (!o) return a==u&&b==v;
	return ((u-a)*x+(v-b)*y)%o==0&&((a-u)*y+(v-b)*x)%o==0;
}

int main()
{
	cin>>a>>b>>u>>v>>x>>y;
	o=x*x+y*y;
	puts((check(a,b)||check(b,-a)||check(-a,-b)||check(-b,a))?"YES":"NO");
	
	return 0;
}
