#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a1,a2,b1,b2,c1,c2,det;
bool f(ll z1,ll z2)
{
    if(det==0  )
        return  (z1==0&&z2==0);
        else
        return   ((c1*z1+c2*z2)% det ==0&&(c1*z2-c2*z1)%det==0);
}
int main()
{
    cin>>a1>>a2>>b1>>b2>>c1>>c2;
    det= c1*c1+c2*c2;
    if(f(b1-a1,b2-a2) || f(b1-a2,b2+a1) || f(b1+a1,b2+a2) || f(b1+a2,b2-a1))
       cout<<"YES"<<endl;
       else
        cout<<"NO"<<endl;
    return 0;
}
