#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long LL;
LL X1,Y1,X2,Y2,X3,Y3;

bool sol(LL x,LL y)
{
    if(X2==0 && Y2==0)
    {
        if(x==X3 && y==Y3)return 1;
        return 0;
    }
    LL t = (X3-x)*Y2-(Y3-y)*X2;
    LL p = Y2*Y2+X2*X2;
    if(t%p)return 0;
    t = (X3-x)*X2+(Y3-y)*Y2;
    if(t%p)return 0;
    return 1;
}

int main()
{
    

    while(scanf("%I64d%I64d",&X1,&Y1)==2)
    {
        scanf("%I64d%I64d%I64d%I64d",&X3,&Y3,&X2,&Y2);
        if(sol(X1,Y1) || sol(Y1,-X1) || sol(-X1,-Y1) || sol(-Y1,X1))
            puts("YES");
        else puts("NO");
    }

    return 0;
}
