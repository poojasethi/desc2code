#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long i64;

i64 xa, ya, xb, yb, xc, yc;

bool tell(i64 dx, i64 dy) {
    i64 det = xc * xc + yc * yc;
    if (det != 0)
        return (dx * xc + dy * yc) % det == 0 && (dx * yc - dy * xc) % det == 0;
    else
        return dx == 0 && dy == 0;
}

int main() {
    scanf("%I64d%I64d%I64d%I64d%I64d%I64d", &xa, &ya, &xb, &yb, &xc, &yc);
    if (
        tell(xb - xa, yb - ya) ||
        tell(xb - ya, yb + xa) ||
        tell(xb + xa, yb + ya) ||
        tell(xb + ya, yb - xa)
    ) {
        puts("YES");
    } else {
        puts("NO");
    }
    return 0;
}
