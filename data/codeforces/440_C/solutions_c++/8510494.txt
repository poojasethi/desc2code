#include <algorithm>
#include <cstdio>

#define For(i, a, b) for (int i = a; i <= b; ++i)

using std::min;

typedef long long LL;

LL one[20];
int ans;

int dfs(LL x, const int num)
{
  int cnt = x / one[num];
  x %= one[num];
  int ret = cnt * num;
  if (x) ret += min(num + dfs(one[num] - x, num - 1), dfs(x, num - 1));
  return ret;
}

int main()
{
  For(i, 1, 16) one[i] = one[i - 1] * 10 + 1;
  LL n;
  scanf("%I64d", &n);
  printf("%d\n", dfs(n, 16));
  return 0;
}
