#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
long long n,f[20];
int dfs(long long x)
{
    int res,i=16;
    if(x<=11)
    return min(x,2+11-x);
    while(f[i-1]>=x)
        i--;
    res=((x/f[i-1])*(i-1))+dfs(x%f[i-1]);
    if(x<f[i]-x)
    return res;

    return min(res,i+dfs(f[i]-x));
}
int main()
{
    f[1]=1;
    for(int i=2;i<=16;i++)
        f[i]=10*f[i-1]+1;
    scanf("%I64d",&n);
    printf("%d\n",dfs(n));
    return 0;
}
