#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>

using namespace std;

long long f[20],x;

int dfs(long long x,int w)
{
if (!w) return 0;
int y=(x/f[w])*w;
x%=f[w];
return y+min(dfs(x,w-1),dfs(f[w]-x,w-1)+w);
}

int main()
{
scanf("%I64d",&x);
for (int i=1;i<=16;i++) f[i]=f[i-1]*10+1;
printf("%d",dfs(x,16));
}
