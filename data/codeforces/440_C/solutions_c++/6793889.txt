#include <cstdio>
#include <bitset>
#include <iostream>
using namespace std;
typedef long long LL;
LL x;
int ans;
void rec(LL x,int lev){
   if(lev>ans)return;
   if(x==0){
      ans = lev;
      return;
   }
   LL now=1;
   int tmp=1;
   while(now*10+1<=x){
      now = now*10LL+1;
      tmp++;
   }
   rec(x%now,x/now*tmp+lev);
   rec(now*10 + 1 - x,tmp+1+lev);
}
int main()
{
   ans = 1<<20;
   cin>>x;
   rec(x,0);
   printf("%d\n",ans);
}
