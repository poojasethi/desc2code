#include<iostream>

using namespace std ;

long long n;
long long d[401] ;

int f(long long n, int i)
{
	int p=(n/d[i])*i ;
	n%=d[i] ;
	if(n==0)
		return p ;
	else
		return p+min(i+f(d[i]-n,i-1), f(n,i-1)) ;
}

int main()
{
	cin >> n ;
	d[0]=0 ;
	for(int i=1 ; i<=16 ; i++)
		d[i]=d[i-1]*10+1 ;
	cout << f(n,16) << endl ;
	return 0 ;
}
