#include <iostream>
using namespace std;
long long ones[20];
int dfs(long long n, int i)
{int kol=i*(n/ones[i]);
 n=n%ones[i];
 if (n==0) return kol;
 return kol+min(i+dfs(ones[i]-n,i-1),dfs(n,i-1));
}
int main()
{ones[0]=0;
 for (int i=1; i<=16;i++) ones[i]=ones[i-1]*10+1;
 long long n;cin>>n;
 cout<<dfs(n,16)<<endl;
return 0;
}
