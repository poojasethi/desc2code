#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll dfs(ll x)
{
   if(x==0)
      return 0;
   ll bfr,aftr=1,d=1;
   while(aftr<=x)
      aftr=aftr*10+1,d++;
   bfr=aftr/10;
   ll a=dfs(x%bfr)+x/bfr*(d-1);
   ll b=dfs((aftr-x)%bfr)+d+(aftr-x)/bfr*(d-1);
   return min(a,b);
}
int main()
{
   ll n;
   cin>>n;
   ll ans=dfs(n);
   cout<<ans<<endl;
   return 0;
}
