#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef long long LL;
char a[100];
int n;
LL p[100];
int solve(LL x, int d)
{
	if(d == 0)
		return 0;
	LL ans = (x / p[d]) * (d);
	x %= p[d];
	return ans + min(solve(x, d-1), d + solve(p[d] - x, d-1));
}
int main()
{
	LL x;
	scanf("%I64d", &x);
	sprintf(a, "%I64d", x);
	n = strlen(a);
	for(int i = 1; i <= n+1; i++)
		p[i] = p[i-1] * 10 + 1;
	
	printf("%d\n", solve(x, n+1));
	return 0;
}
