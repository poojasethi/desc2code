#include<math.h>
#include<algorithm>
#include<stdio.h>
typedef long long ll;
using namespace std;

ll a[17] ;
ll f(ll n,ll i)
{
    ll num = n / a[i] ;
    n %= a[i] ;
    if(n == 0)
        return num*i ;
    else
        return num*i + min( i+f(a[i]-n,i-1),f(n,i-1) );
}
int main()
{
    ll n , i ;
    a[0] = 0 ;
    for(i = 1 ; i <= 16 ; i++)
        a[i] = a[i-1]*10 + 1 ;
    while(scanf("%I64d",&n) != EOF)
    {
        printf("%I64d\n",f(n,16));
    }
}
