#include "iostream"
#include "cstdio"
#include "cstring"

using namespace std;

long long n, f[18];

int cal(long long x) { 
    int ans, i = 16;
    if (x <= 11) return min(x, 13 - x);
    while (f[i - 1] >= x) i--;
    ans = ((x / f[i - 1]) * (i - 1)) + cal(x % f[i - 1]);
    if (x < f[i] - x) return ans;
    return min(ans, i + cal(f[i] - x));
}
int main() {
    f[1] = 1; for (int i = 2; i <= 16; i++) f[i] = 10 * f[i - 1] + 1;
    cin >> n; cout << cal(n) << endl;
}
