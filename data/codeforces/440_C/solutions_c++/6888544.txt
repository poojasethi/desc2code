#include<iostream>
#include<algorithm>
using namespace std;

long long one[20];

int dfs(long long int n,int i){
    int ans=i*(n/one[i]);
    n=n%one[i];
    if (n==0)
    {
        return ans;
    }
    return ans+min(i+dfs(one[i]-n,i-1),dfs(n,i-1));
}

int main(){
    one[0]=0;
    for (int i = 1; i < 17; ++i)
    {
        one[i]=one[i-1]*10+1;
    }
    long long int n;
    cin>>n;
    cout<<dfs(n,16);
}
