#include<iostream>
using namespace std;
long long int ones[20];
long long int dfs(long long int n,long long int i)
{
    long long int tot;
    tot=i*(n/ones[i]);
    n=n%ones[i];
    if(n==0)
    {
        return tot;
    }
    return tot+min(i+dfs(ones[i]-n,i-1),dfs(n,i-1));
}
int main()
{
    long long int n;
    ones[0]=0;
    cin>>n;
    for(int i=1;i<=16;++i)
    {
        ones[i]=ones[i-1]*10+1;
    }
    cout<<dfs(n,16)<<"\n";
    return 0;
}
