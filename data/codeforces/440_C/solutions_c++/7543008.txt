#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define LL __int64
LL a[17] ;
LL f(LL n,LL i)
{
    LL num = n / a[i] ;
    n %= a[i] ;
    if(n == 0)
        return num*i ;
    else
        return num*i + min( i+f(a[i]-n,i-1),f(n,i-1) );
}
int main()
{
    LL n , i ;
    a[0] = 0 ;
    for(i = 1 ; i <= 16 ; i++)
        a[i] = a[i-1]*10 + 1 ;
    scanf("%I64d", &n);
    printf("%I64d\n", f(n,16));

}
