#include <iostream>
#include <algorithm>
#include <cstdio>
#include <string>
#include <cmath>
#define L 18
typedef long long ll;
using namespace std;

ll a[L];

int dfs(ll n, int p) {
	int k = n / a[p];
	n = n % a[p];

	if (n == 0)
		return p * k;
	return p * k + min(p + dfs(a[p] - n, p - 1), dfs(n, p - 1));
}

int main() {
	a[0] = 0;
	for (int i = 1; i < L; i++)
		a[i] = a[i-1] * 10 + 1;

	ll n;
	while (cin >> n) {
		cout << dfs(n, L-1) << endl;
	}
}
