#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;
ll one[18]={0,1};
void init()
{
    for(ll i=2;i<=16;i++) one[i]=one[i-1]*10+1;
}
ll dfs(ll sum,ll n)
{
    if(n==1) return sum;
    ll k1=sum/one[n];
    ll k2=k1+1;
    ll key1=dfs(sum-k1*one[n],n-1),key2=dfs(k2*one[n]-sum,n-1);
    return min(k1*n+key1,k2*n+key2);
}
int main()
{
    init();
    ll n;
    while(~scanf("%I64d",&n))
    {
        cout<<dfs(n,16)<<"\n";
    }
    return 0;
}
