#include<cstdio>
#include<algorithm>
#include<iostream>
typedef long long LL;
using namespace std;
LL a[17];//a[i]ÓÐi¸ö1 

LL f(LL n, LL i) {
	LL num = n / a[i];
	n %= a[i];
	if(n==0) return num*i;
	return num*i +min(f(n,i-1), i+f(a[i]-n,i-1));
}

int main () {
	LL n, i;
	a[0] = 0;
	for(i=1; i<=16; i++) a[i] = a[i-1]*10+1;
	cin >> n;
	cout << f(n, 16) << endl;
	return 0;
}
