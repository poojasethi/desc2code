#include<cstdio>
#define min(x,y) ((x)<(y)?(x):(y))
long long a[20], n;
int dfs(long long n, int l){
    int cnt = n / (long long) a[l];
    if(!(n %= a[l])) return cnt * l;
    return cnt * l + min(l + dfs(a[l] - n, l - 1), dfs(n, l - 1));
}
int main(){
    for(int i = 1;i<17;i++) a[i] = a[i-1] * 10 + 1;
    scanf("%I64d", &n);
    printf("%d\n", dfs(n, 16));
    return 0;
}
