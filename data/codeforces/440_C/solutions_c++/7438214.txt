#include <bits/stdc++.h>
using namespace std;
#define int long long 
int arr[20]; 
int dfs (int num, int index)
{
    int r=(num/arr[index])*index; 
    num%=arr[index]; 
    if (num==0)
    {
        return r;
    }    
    else
    {
        return r+min(dfs(arr[index]-num, index-1)+index, dfs(num, index-1)); 
    
    }
}
