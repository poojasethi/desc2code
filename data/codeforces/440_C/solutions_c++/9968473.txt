#include <bits/stdc++.h>
using namespace std;

long long ones[20];

int dfs(long long n, int x) {
    int k = x * (n / ones[x]);
    n %= ones[x];
    if(!n)
        return k;
    return k + min(x + dfs(ones[x] - n, x - 1), dfs(n, x - 1));
}

int main() {
    for(int i = 1; i < 20; i ++)
        ones[i] = ones[i-1] * 10 + 1;
    long long x;
    while(cin >> x) {
        cout << dfs(x, 16) << endl;
    }
    return 0;
}
