#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long LL;
LL a[20], n;

LL dfs(LL n, int d)
{
    LL num = n / a[d];
    n = n % a[d];
    if(n == 0) return num * d;
    return num * d + min(d + dfs(a[d] - n, d - 1), dfs(n, d - 1));
}

int main()
{
    for(int i = 1; i <= 16; i++) a[i] = a[i - 1] * 10 + 1;
    scanf("%I64d", &n);
    printf("%I64d\n", dfs(n, 16));

    return 0;
}
