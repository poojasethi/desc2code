#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
ll a1[17];
int dfs(ll n,int a)
{
	ll m=n%a1[a];
	if(m)m=min(dfs(m,a-1),dfs(a1[a]-m,a-1)+a);
	return m+n/a1[a]*a;
}
int main()
{
	int a;ll n;
	for(int i=1;i<17;++i)
		a1[i]=a1[i-1]*10+1;
	scanf("%I64d",&n);
	for(int i=1;i<17;++i)
	if(a1[i]>=n)
	{
		printf("%d\n",dfs(n,i));
		break;
	}
}
