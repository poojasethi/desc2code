#include <bits/stdc++.h>
using namespace std;

typedef long long lint;
typedef pair < int , int > ii;

lint n,ans,ar[20];

lint solve( lint x , lint k ){
	lint t = k * (x / ar[k]);
	x %= ar[k];
	if( !x )
		return t;
	return t + min(k + solve(ar[k] - x , k-1) , solve(x , k-1));
}

int main(){
	
	scanf("%lld",&n);
	
	for( int i=1 ; i < 17 ; i++ )
		ar[i] = ar[i-1] * 10 + 1;
	
	printf("%lld\n",solve(n , 16));
	
	return 0;
}
