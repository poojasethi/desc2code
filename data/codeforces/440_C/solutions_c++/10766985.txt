long long unsigned n,s;
void f(long long unsigned x)
{
	if (x<=1){if (x==1)s++;return ;}
	long long  unsigned l,t=1,k=x;
	while(k)		k/=10,t*=10,t++;
	l=t/10;
	if (x<l)		l/=10,t/=10;
	k=(l+t)/2;
	k=(x<=k)?l:t;
	while(k)		s++,k/=10;
	(x<=(l+t)/2)?f(x-l):f(t-x);
}	
int main ()
{
