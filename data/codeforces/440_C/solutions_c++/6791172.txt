#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

#define LL __int64

using namespace std;

LL n;
LL an;

void dfs(LL x,LL m){
	if(m>an) return;
	if(!x){
		an=m;
		return;
	}
	LL tmp=0,st=0;
	while(tmp*10+1<=x) tmp=tmp*10+1,st++;
	dfs(x%tmp,m+x/tmp*st);
	dfs(tmp*10+1-x,m+st+1);
}

int main()
{
	an=1e9;
	cin>>n;
	dfs(n,0);
	cout<<an<<endl;
	return 0;
}
