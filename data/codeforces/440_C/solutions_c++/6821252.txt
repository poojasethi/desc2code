#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cmath>
using namespace std;
typedef long long ll;ll f[20],n;
int count(ll x){ 
    int res,i=17;
    if (x<=11) return min(x,2+11-x);
    while (f[i-1]>=x) i--;
    res=((x/f[i-1])*(i-1))+count(x%f[i-1]);
    if (2*x<f[i]) return res;
    return min(res,i+count(f[i]-x));
}
int main(){
    for (int i=1;i<=18;++i) f[i]=f[i-1]*10+1;
    cin>>n;cout<<count(n);
    return 0;
}
