#include <iostream>
#include<cstdio>

using namespace std;
#define LL long long
LL num[17];
LL cal(LL n,int i)
{
    int k=n/num[i];
    n%=num[i];
    if(n==0)return i*k;
    else return k*i+min(cal(n,i-1),i+cal(num[i]-n,i-1));
}
int main()
{
    int i;
    LL n;
    for(i=1;i<17;i++)
        num[i]=1+num[i-1]*10;
    scanf("%I64d",&n);
    printf("%I64d",cal(n,16));
    return 0;
}
