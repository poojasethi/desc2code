#include <iostream>
using namespace std;

long long a[16];

long long dfs(long long n, long long k)
{
	// cout<<"dfs("<<n<<","<<k<<")"<<endl;
	if (k==1) return n;
	else if (a[k]>n) return min(k+dfs(a[k]-n, k-1), dfs(n, k-1));
	else if (n%a[k]==0) return n/a[k]*k;
	else return min(n/a[k]*k+dfs(n%a[k],k-1), (n/a[k]+1)*k+dfs(a[k]-n%a[k],k-1));
}

int main()
{
	long long n;
	int i;
	cin>>n;
	for (a[1]=1,i=1; a[i]<n; i++) a[i+1]=a[i]*10+1;
	cout<<dfs(n, i)<<endl;
	return 0;
}
