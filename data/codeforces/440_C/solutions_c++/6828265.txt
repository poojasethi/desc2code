#include <iostream>
using namespace std;

#define F cin
#define G cout

long long n,a[16];

long long solve(long long n,int step)
{
    long long ans = (step+1LL)*(n/a[step]);
    n %= a[step];
    if ( n == 0 ) return ans;
    return ans + min(solve(n, step-1), step+1 + solve(a[step]-n, step-1));
}

int main()
{
    a[0] = 1;
    for (int i=1;i<16;++i)
        a[i] = a[i-1]*10 + 1;
    F>>n;
    G<<solve(n,15);
}
