#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
ll a1[17];
int dfs(ll n,int a)
{
	if(!a)return 0;
	ll b=(a1[a+1]-n);
	return min(dfs(n%a1[a],a-1)+n/a1[a]*a,dfs(b%a1[a],a-1)+b/a1[a]*a+a+1);
}
int main()
{
	int a;ll n;
	for(int i=1;i<17;++i)
		a1[i]=a1[i-1]*10+1;
	scanf("%I64d",&n);
	for(int i=16;i;--i)
	if(a1[i]<=n)
	{a=i;break;}
	printf("%d\n",dfs(n,a));
}
