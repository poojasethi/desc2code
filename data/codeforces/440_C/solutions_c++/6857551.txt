#include<bits/stdc++.h>
using namespace std;
long long ones[20];
int juggad(int i, long long n){
    long long a = i*(n/ones[i]);
    n = n%ones[i];
    if(n==0) return a;
    return a+min(i+juggad(i-1,ones[i]-n),juggad(i-1,n));
}
int main(){
    ones[0] = 0;
    for(int i=1;i<=16;i++)
    {
        ones[i] = 10*ones[i-1]+1;
    }
    long long n;
    cin>>n;
    cout<<juggad(16,n)<<endl;
    return 0;
}
