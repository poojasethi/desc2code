#include <iostream>
#include <vector>
#include <stdio.h>
#include <string>
#include <cmath>
#include <algorithm>

using namespace std;

int osn(long long a)
{   int p,q,l;
    long long k;
    if(a==0){return 0;}
    k=1;l=0;
    while(a>=k){l++;k=k*10+1;}
    p=osn((k-a)%(k/10))+l+1+l*((k-a)/(k/10));
    q=osn(a%(k/10))+l*(a/(k/10));
    return min(p,q);

}

int main ()
{
    long long n;
    cin>>n;
    int r;
    r=osn(n);
    cout<<r<<endl;

}
