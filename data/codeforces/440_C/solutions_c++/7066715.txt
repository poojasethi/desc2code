#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;
ll n;
ll a[17]={0,1};
int dfs(ll x)
{
	int sum1;
	int i=16;
	if(x<=11)
	{
		return  min(x,2+11-x);
	}
	while(a[i]>x) i--;
	
	sum1=x/a[i]*i+dfs(x%a[i]); 
	if(x<a[i+1]-x)
		return sum1; 
 
	return min( sum1,i+1+dfs(a[i+1]-x) );      
	
	
}
int main()
{
	for(int i=2;i<=16;i++)
	{
		a[i]=a[i-1]*10 +1;
	}
	
	while(cin>>n&&n>0)
	{
		cout<<dfs(n)<<endl;
		
	}
	return 0;
}
