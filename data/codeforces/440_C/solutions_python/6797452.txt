def solve(n,p):
	aparitii = n/a[p];
	if(n%a[p]==0):
		return aparitii*p;
	sol1 = aparitii * p + solve(n%a[p],p-1);
	sol2 = (aparitii+1) * p + solve(a[p]*(aparitii+1)-n,p-1);
	return min(sol1,sol2);
