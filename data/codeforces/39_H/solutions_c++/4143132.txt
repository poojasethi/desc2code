#include<iostream>
using namespace std;

long long chbase(int a,int base)
{
	long long ans=0;
	long long p=1;
	while(a)
	{
		ans+=(a%base)*p;
		a/=base;
		p*=10;
	}
	return ans;
}

int main()
{
	int k;
	cin>>k;
	for(int i=1;i<k;i++){
		for(int j=1;j<k;j++)
			cout<<chbase(i*j,k)<<' ';cout<<endl;}
	return 0;
}
