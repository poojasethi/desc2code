#include "iostream"
using namespace std;
void carp(int n,int t){
  if(t==0)return ;
  carp(n,t/n);
  cout<<t%n;
}
int main(){
  int n,i,j;
  cin>>n;
  for(i=1;i<n;i++,cout<<endl)
    for(j=1;j<n;j++){
     if(j==1)cout<<i<<" ";
     else if(i==1)cout<<j<<" ";
     else{ carp(n,i*j); cout<<" "; }
    }
  return 0;
}
