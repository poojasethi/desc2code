#include<iostream>
using namespace std;
int main()
{
    int k,i,j;
    cin>>k;
    for(i=1;i<k;i++)
    {
        for(j=1;j<k;j++)
        {
            int d=i*j;
            if(d>=k)
                cout<<d/k;
            cout<<d%k;
            cout<<' ';
        }
        cout<<endl;
    }
    //system("pause");
    return 0;
}
