#include<iostream>
#include<string>
using namespace std;
int main()
{
	int k,r;
	cin>>k;
	string s;
	for(int i = 1; i < k; i++)
	{
		for(int j = 1; j < k; j++)
		{
			r = i * j;
			s = "";
			while(r > 0)
			{
				s = char(r % k + int('0')) + s;
				r /= k;
			}
			cout<<s<<" ";
		}
		cout<<endl;
	}
	return 0;
}
