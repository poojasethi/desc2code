#include <iostream>

using namespace std;
int convert(int a,int b){
    int re=0;
    re=a/b;
    re=re*10+a%b;
    return re;
}
int main()
{
    int k;
    cin>>k;
    for(int i=1;i<k;i++){
        for(int j=1;j<k;j++){
            cout<<convert(j*i,k)<<" ";
        }
        cout<<endl;
    }
    return 0;
}
