#include <cstdlib>
#include <iostream>

using namespace std;
int k;
int q(int a)
{
     if(a>=k)
     cout<<(a/k);
     cout<<(a%k)<<' ';
}
int main(int argc, char *argv[])
{
    cin>>k;
    for(int i=1;i<k;i++){
    for(int j=1;j<k;j++)
    q(i*j);
    cout<<endl;
}
    //system("PAUSE");
    return EXIT_SUCCESS;
}
