#include <stdio.h>
#include <string.h>

int k;
void print(int n) {
	int a = n/k;
	int b = n%k;
	if(a)
		printf("%d",a);
	printf("%d",b);
}
int main() {
	scanf("%d", &k);
	for(int a=1; a<k; a++) {
		for(int b=1; b<k; b++) {
			print(a*b);
			printf(" ");
		}
		printf("\n");
	}
	printf("\n");
	
	return 0;
}
