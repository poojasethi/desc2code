#include<iostream>
using namespace std;
int main()
{
	int k;
	cin >> k;
	for(int i=1;i<k;i++)
	{
		for(int j=1;j<k;j++)
		{
			int ans;
				ans=i*j;
			int A[7],top=0;
			int ans2=ans;
			while(ans2!=0)
			{
			A[top]=ans2%k;
				top++;
				ans2=ans2/k;
			}
		for(int l=top-1;l>=0;l--)
				cout << A[l];
		cout << " ";
		}
		cout << endl;
	}
}
