#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n ;

    for(int i=1 ; i<n ; i++)
    {
        for(int j=1 ; j<n ; j++)
        {
            int x = i*j ;

            if( x < n ) cout << x << " ";
            else cout << x/n << x%n << " ";

        }
        cout << endl;
    }
}
