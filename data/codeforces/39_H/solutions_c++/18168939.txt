#include<cstdio>
int k;
int d(int x)
{
	int num=0;
	int l[10];
	int cnt=0;
    while(x)
	{
		l[++cnt]=x%k;
		x/=k;
	}
	for(int i=cnt;i>=1;i--)
	{
		num=num*10+l[i];
	}
	return num;
}
int main()
{
	scanf("%d",&k);
	for(int i=1;i<k;i++)
	{
		for(int j=1;j<k;j++)
		{
	
			printf("%d ",d(i*j));
		}
		printf("\n");
	}
}
