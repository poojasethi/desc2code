#! /usr/bin/python

from string import *

def decimal2other(x , base ):
    res = ""
    while x > 0 :
        t = x % base
        x /= base
        res = str(t) + res
    return res

def multable( n ) :
    for x in range(1,n):
        for y in range(1,n):
            t = x*y
            p = decimal2other(t,n)
            if y+1 < n :
                print p,
            else :
                print p


if __name__ =="__main__" :
    n = int(raw_input())
    multable(n)
