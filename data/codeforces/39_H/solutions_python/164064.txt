n = int(raw_input())

def basen(val, base):
    res = ""
    while val > 0:
        res += str(val % base)
        val -= val % base
        val /= base
    return res[::-1]

for x in range(n-1):
    for y in range(n-1):
        #Put x*y into base n
        print basen( (x+1) * (y+1), n), " ",
