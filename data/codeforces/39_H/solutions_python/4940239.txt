# -*- coding: utf-8 -*-
import sys

def func1(n,base):
    str1 = ""
    while True:
        if n/base == 0:
            str1 += str(n)
            break
        else:
            str1 += str(n%base)
            n /= base

    return str1[::-1]


if __name__ == '__main__':

    num = map(int,sys.stdin.readline().split())

    for v in range(1,num[0]):
        for u in range(1,num[0]):
            print func1(v*u,num[0]),
        print ""
