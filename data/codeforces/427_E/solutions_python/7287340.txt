import sys
from random import randint
from time import time

def readIn():
    return map(lambda x:int(x),sys.stdin.readline().strip('\n').split(" "))

n,m=readIn()
pos=readIn()
if n==1:
    print 0
else:
    if m>=n:
        m=n-1
    pickupLefts=pos[0::m]
    pickupRights=pos[-1::-m]
    index=0
    while pickupLefts[index]<=pickupRights[index]:
        index+=1
    distance=0
    for i in xrange(0,index):
        distance+=pickupRights[i]-pickupLefts[i]
    print distance*2
