#include<stdio.h>
__int64 a[1000010],b,sum;
int main(void){
	int i,j,k,m,n;

	while(scanf("%d %d",&n,&m)!=EOF){
		for(i=0;i<n;i++){
			scanf("%I64d",&a[i]);
		}
		sum=0;
		b=a[n/2];
		for(i=0;i<n/2;i=i+m){
			sum=sum+b-a[i];
		}
		for(i=n-1;i>=n/2;i=i-m){
			sum=sum+a[i]-b;
		}
		printf("%I64d\n",sum*2);
	}
	return 0;
}
