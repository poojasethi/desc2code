#include "bits/stdc++.h"
#define maxn 1000005
using namespace std;

long long x[maxn];

main(){
  int n, m;
  cin >> n >> m;
  for( int i = 0; i < n; i++ )  cin >> x[i];
  int mid = n/2;
  long long ans = 0LL;
  sort( x, x+n );
  for( int i = n-1; mid < i; i -= m )
    ans += 2 * ( x[i]-x[mid] );
  for( int i = 0; i < mid; i += m )
    ans += 2 * ( x[mid]-x[i] );
  cout << ans;
}
