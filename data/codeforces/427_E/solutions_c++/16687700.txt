#include <bits/stdc++.h>
#define MAX_N 1000100

using namespace std;

typedef long long ll; 

ll a[MAX_N],res; 
int n,m; 

int main () 
	{
		ios::sync_with_stdio(false); 
		
		cin >> n >> m; 
		for (int i = 1; i <= n; i++) cin >> a[i]; 
		
		int l = 1; int r = n; 
		
		while (l <= r) 
			{
				res += a[r] - a[l]; 
				l += m; r -= m;
			}
			
		cout << res * 2; 
		return 0;
	}
