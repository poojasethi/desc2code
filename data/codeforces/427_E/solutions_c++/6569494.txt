#include <fstream>
#include <iostream>
using namespace std;

long long int i,j,n,m,x,min1=0;
long long int d[1000001];
int main()
{
	cin>>n>>m;
	for(i=0;i<n;++i)
		cin>>d[i];

	for(i=0;i<n/2;i+=m)
		min1+=d[n/2]-d[i];

	for(i=n-1;i>n/2;i-=m)
		min1+=d[i]-d[n/2];

	cout<<min1*2<<"\n";
	return 0;
}
