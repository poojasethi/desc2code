#include <cstdio>

using namespace std;

typedef unsigned long long ull;

int main(void)
{
    int n, m;
    int p[1000001];
    scanf("%d%d", &n, &m);
    for (int i=0; i<n; i++)
        scanf("%d", p+i);
    ull ret=0;
    for (int i=0, j=n-1; i<j; ) {
        ret+=p[j]-p[i];
        i+=m;
        j-=m;
    }
    ret*=2;
    printf("%I64d\n", ret);
}
