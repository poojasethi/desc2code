#include <bits/stdc++.h>

using namespace std;

int *x, m, n;
int64_t res = 0;
int main(){
    scanf("%d %d", &n, &m);
    x = new int[n];
    for (int i = 0; i < n; ++i)
        scanf("%d", x + i);
    for (int i = 0, j = n - 1; j > i; j -= m, i += m)
        res += ((x[j] - x[i]));
    printf("%I64d\n", (res<<1));
    return 0;
}
