#include <iostream>
#include <cstdio>

using namespace std;
typedef long long ll;
ll d[11111111];

int main() {
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++) scanf("%I64d",d+i);
    ll ans=0;
    for(int i=0;i<n;i+=m) {
        ans+=(d[n-1]-d[i]);
        n-=m;
    }
    printf("%I64d\n",ans*2);
}
