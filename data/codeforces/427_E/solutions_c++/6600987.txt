#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

__int64 n,m,a[1001000];

int main(){
    while(~scanf("%d%d",&n,&m)){
        int i;
        for(i=0;i<n;++i)scanf("%I64d",&a[i]);
        __int64 ans = 0;
        for(i=0;i<n/2;i+=m)ans += (a[n/2]-a[i]);
        for(i=n-1;i>n/2;i-=m)ans += (a[i]-a[n/2]);
        printf("%I64d\n",ans<<1);
    }
    return 0;
}
