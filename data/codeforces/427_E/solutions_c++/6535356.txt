#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;
const int nMax = 1e6+10;
long long pos[nMax];
int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    for(int i = 1;i <= n;i++)
        scanf("%I64d",pos+i);
    long long dis = 0;
    for(int i = 1;i+i-1 <= n;i+=m)
        dis += pos[n-i+1]-pos[i];
    printf("%I64d\n",dis<<1);
}
