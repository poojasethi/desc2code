#include<stdio.h>
int n,m;
__int64 a[1001000];
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%I64d",a+i);
    __int64 ans=0;
    int pos=n/2+1;
    for(int i=n;i>=pos;i-=m)
        ans+=a[i]-a[pos];
    for(int i=1;i<pos;i+=m)
        ans+=a[pos]-a[i];
    ans*=2;
    printf("%I64d\n",ans);
    return 0;
}
