#include<cstdio>
using namespace std;
const int gg=10000001;
int n,m,a[gg];
long long ans;
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
      scanf("%d",&a[i]);
    int l=1,r=n;
    while(l<=r)
    {
        ans+=(long long)2*(a[r]-a[l]);
        l+=m;
        r-=m;
    }
    printf("%I64d\n",ans);
}
