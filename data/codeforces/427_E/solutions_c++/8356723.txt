#include<cstdio>
using namespace std;
int n,m,point[1000010];
long long ans=0;
void doit(){
		int i=1,j=n;
		while (i<j){
				ans=ans+(point[j]-point[i]);
				i=i+m;
				j=j-m;	
			}
	}
void init(){
		scanf("%d%d",&n,&m);
		for (int i=1;i<=n;i++)
			scanf("%d",&point[i]);
	}
int main(){
		init();
		doit();
		printf("%I64d",ans<<1);
	}
