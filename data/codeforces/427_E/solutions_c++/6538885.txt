#include<cstdio>

int n,m;
int a[1000000];

__int64 Cal (int l,int r)
{
    if(l>r)	return 0;
    int len = r-l+1;
    if (len<=m)	return (__int64)(a[r]-a[l]);
    return (__int64)(a[r]-a[l])+Cal(l+m, r-m);
}

int main ()
{
    scanf("%d%d",&n,&m);
    for (int i=0;i<n;i++)
        scanf("%d", a+i);
    printf("%I64d\n",Cal(0,n-1)*2);
    return 0;
}
