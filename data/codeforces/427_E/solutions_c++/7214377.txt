#include<iostream>
#include<set>
#include<vector>
#include<algorithm>
using namespace std;
int N , M ,a[int(1e6)+1000];
int main()
{
	ios_base::sync_with_stdio(false);
	cin >> N >> M;
	for(int i = 0 ; i< N;i++){
		cin >> a[i];
	}
	sort(a , a+N);
	long long ans = 0 ; 
	for(int i = 0 , j = N-1 ; i < j;i+= M , j-=M){		
		ans+=a[j] -  a[i];
	}
	cout << (ans << 1) << endl;
	return 0;
	
}
