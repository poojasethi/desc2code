#include<iostream>
#include<cstdio>
using namespace std;
int a[1000010];
int main()
{
    int mid,n,m;
    long long ans=0;
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;i++)
    scanf("%d",&a[i]);
    mid=n/2+1;
    for (int i=1;i<=mid;i+=m)
    ans+=a[mid]-a[i];
    for (int i=n;i>=mid;i-=m)
    ans+=a[i]-a[mid];
    ans*=2;
    cout<<ans<<endl;
    return 0;
}
