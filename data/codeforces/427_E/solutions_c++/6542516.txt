#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long LL;
const int N=1e6+10;
int m,n;
int a[N];

int main()
{
    scanf("%d%d",&n,&m);
    LL ans=0;
    for (int i=0;i<n;i++) scanf("%d",&a[i]);
    int l=0,r=n-1;
    while (l<=r) {
        ans+=a[r]-a[l];
        l+=m; r-=m;
    }

    printf("%I64d\n",ans*2);
    return 0;
}
