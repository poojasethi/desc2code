#include<iostream>
#include<cstdio>
using namespace std;
int n,m;
int a[1000005];
long long ans;
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    for(int i=1,j=n;i<j;i+=m,j-=m)ans+=a[j]-a[i];
    ans*=2;
    printf("%I64d\n",ans);
    //system("pause");
    return 0;
}
