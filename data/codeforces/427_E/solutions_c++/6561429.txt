#include <stdio.h>
#include <string.h>
__int64 road[1000010];
int main(void)
{
	int n,m;
	int i;

	while(scanf("%d%d",&n,&m)!=EOF)
	{
		for(i=0;i<n;i++)
			scanf("%I64d",&road[i]);
		int mid=(n-1)/2;
		__int64 ans=0;
		for(i=0;i<mid;i+=m)
			ans+=road[mid]-road[i];
		for(i=n-1;i>mid;i-=m)
			ans+=road[i]-road[mid];
		ans*=2;
		printf("%I64d\n",ans);
	}

	return 0;
}//6561323
