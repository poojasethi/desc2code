#include <cstdio>
#include <algorithm>
using namespace std;

int a[1000010],n,m;
long long ans = 0;

int main()
{
    //freopen("p.in","r",stdin);
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;++i)
        scanf("%d",&a[i]);
    //sort(a+1,a+n+1);
    for (int i=1,j=n;i<j;i+=m,j-=m)
        ans += a[j] - a[i];
    ans <<= 1;
    printf("%I64d\n",ans);
}
