#include<cstdio>
#include<cstring>
#include<algorithm>
#define N 1000002
using namespace std;
typedef long long ll;
ll a[N],ans;
int n,m;

int main(){
  int i;
  scanf("%d%d",&n,&m);
  for (i=0;i<n;++i) scanf("%I64d", &a[i]);
  for (i=0;i<=n-1-i;i+=m) ans+=a[n-1-i]-a[i];
  printf("%I64d\n",ans*2);
  return 0;
}
