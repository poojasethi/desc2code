#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,a[1000010],mid;
long long ans;

int main(){
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+1+n);
	mid=(n>>1)+1;
	for (int i=n;i>mid;i-=m)
		ans+=a[i]-a[mid];
	for (int i=1;i<mid;i+=m)
		ans+=a[mid]-a[i];
	ans*=2;
	printf("%I64d\n",ans);
	return 0;
}
