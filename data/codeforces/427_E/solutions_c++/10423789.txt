#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int n, m;
long long pos[1000000 + 5];
long long ans;
int main() {
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		scanf("%lld", &pos[i]);
	int l = 0, r = n-1;
	while (l < r) {
		ans += (pos[r] - pos[l]);
		l += m;
		r -= m;
	}
	printf("%I64d", ans*2);
	return 0;
}
