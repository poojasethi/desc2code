#include<stdio.h>

typedef long long int LD;

int ans;

struct {
  int a,b;
  LD ab;
  void output() {
    printf("%d %d\n",a,b);
  }
  void op1() {
    ab/=2;
    if(a%2==0)
      a/=2;
    else
      b/=2;
  }
  void op2() {
    ab=ab/3*2;
    if(a%3==0)
      a=a/3*2;
    else
      b=b/3*2;
  }
  void input() {
    scanf("%d %d",&a,&b);
    ab=a;
    ab*=b;
  }
  void fix1() {
    while(ab%3 ==0 ){
      ans++;
      op2();
    }
  }
  void fix2() {
    while(ab % 2== 0){
      ans++;
      op1();
    }
  }
}x,y;

int main() {
  x.input();
  y.input();
  while(x.ab % 3 ==0 && y.ab % 3 ==0) {
    x.ab /= 3;
    y.ab /= 3;
  }
  x.fix1();
  y.fix1();

  while(x.ab % 2 ==0 && y.ab % 2 ==0) {
    x.ab /= 2;
    y.ab /= 2;
  }
  x.fix2();y.fix2();
  if(x.ab!=y.ab){
    printf("-1\n");
    return 0;
  }

  printf("%d\n",ans);
  x.output();
  y.output();

  return 0;
}
