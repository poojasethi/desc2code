#include <stdio.h>
#include <iostream>
using namespace std;
#define long long ll

int  a1,b1,a2,b2,cnt,i;

int f(int num)
{
    if(num%i==0)
        return f(num/i)+1;
    else
        return 0;
}

int main()
{
    while(scanf("%d%d%d%d",&a1,&b1,&a2,&b2)!=EOF)
    {
        for(i=3;i>=2;i--)
        while(f(a1)+f(b1)-f(a2)-f(b2)!=0)
        {
            cnt++;
            if(f(a1)+f(b1)>f(a2)+f(b2))
            {
                if(f(a1)>0)
                    a1=a1*(i-1)/i;
                else
                    b1=b1*(i-1)/i;
            }
            else
            {
                if(f(a2)>0)
                    a2=a2*(i-1)/i;
                else
                    b2=b2*(i-1)/i;
            }
        }
        if(1LL*a1*b1!= 1LL*a2*b2)
            printf("-1\n");
        else
            printf("%d\n%d %d\n%d %d\n",cnt,a1,b1,a2,b2);
    }
    return 0;
}
