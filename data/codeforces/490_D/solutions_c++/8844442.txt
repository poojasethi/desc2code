#include<bits/stdc++.h>
using namespace std;
long long i,a,b,c,d,x,y,cnt2x,cnt3x,cnt2y,cnt3y;
long long ab(long long zz){if(zz>0)return zz; return -zz;}
int main()
{
    scanf("%lld %lld %lld %lld",&a,&b,&c,&d);
    x=a*b; y=c*d;
    while(x%2==0)x/=2,cnt2x++;
    while(x%3==0)x/=3,cnt2x++,cnt3x++;
    while(y%2==0)y/=2,cnt2y++;
    while(y%3==0)y/=3,cnt2y++,cnt3y++;
    if(x!=y)printf("-1\n");
    else
    {
        printf("%lld\n",ab(cnt2x-cnt2y)+ab(cnt3x-cnt3y));
        if(cnt3x>cnt3y)for(i=0;i<cnt3x-cnt3y;i++)
        {
            if(a%3)b=b/3*2;
            else a=a/3*2;
        }
        else for(i=0;i<cnt3y-cnt3x;i++)
        {
            if(c%3)d=d/3*2;
            else c=c/3*2;
        }
        if(cnt2x>cnt2y)for(i=0;i<cnt2x-cnt2y;i++)
        {
            if(a%2)b/=2;
            else a/=2;
        }
        else for(i=0;i<cnt2y-cnt2x;i++)
        {
            if(c%2)d/=2;
            else c/=2;
        }
        printf("%lld %lld\n%lld %lld\n",a,b,c,d);
    }
}
