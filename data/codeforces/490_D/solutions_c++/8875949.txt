#include<stdio.h>
int ans=0;
void y(int x0[],int x1[],int i)
{
    while(x0[i]>x1[i])
    {
        x0[i]--;ans++;
        if(i==3)
        {
            x0[2]++;
            x0[0]%i==0?x0[0]*=2:x0[1]*=2;
        }
        x0[0]%i==0?x0[0]/=i:x0[1]/=i;
    }
}
int x(int x0[],int i)
{
    int t=x0[i];
    while(t%2==0){t/=2;x0[2]++;}
    while(t%3==0){t/=3;x0[3]++;}
    return t;
}
int main()
{
    int a[4],b[4];
    scanf("%d%d%d%d",&a[0],&a[1],&b[0],&b[1]);
    a[2]=a[3]=b[2]=b[3]=0;
    if(x(a,0)*x(a,1)!=x(b,0)*x(b,1)){printf("-1");return 0;}
    y(a,b,3);y(b,a,3);y(a,b,2);y(b,a,2);
    printf("%d\n%d %d\n%d %d\n",ans,a[0],a[1],b[0],b[1]);
    return 0;
}
