#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a1,a2,b1,b2,k,l;
bool kiemtra(ll x,ll y)
{
    while(!(x%3)) x/=3;
	while(!(x%2)) x/=2;
	while(!(y%3)) y/=3;
	while(!(y%2)) y/=2;
	if(x==1&&y==1) return true;
	return false;
}
int main()
{
	cin >> a1 >> a2 >> b1 >> b2;
	ll ans=0;
	k=a1*a2; l=b1*b2;
	ll m=__gcd(k,l);
	if(!kiemtra(k/m,l/m)) { cout << -1 << endl; return 0; }
	while((k/m)%3==0)
	{ ans++;
		k=k*2/3;
		if(!(a1%3)) a1=a1*2/3;
		else a2=a2*2/3;
	}
	while((l/m)%3==0)
	{ ans++;
		l=l*2/3;
		if(!(b1%3)) b1=b1*2/3;
		else b2=b2*2/3;
	}
	m=__gcd(k,l);
	while(k>m)
	{ ans++;
		k/=2;
		if(!(a1%2)) a1/=2;
		else a2/=2;
	}
	while(l>m)
	{ ans++;
		l/=2;
		if(!(b1%2)) b1/=2;
		else b2/=2;
	}
		cout << ans << endl;
		cout << a1 << " " << a2 << " " << endl << b1 << " " << b2 << endl;
}
