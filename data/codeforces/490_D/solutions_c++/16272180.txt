#include<cstdio>
#define ll __int64
int main()
{
	ll a,b,c,d,s1,s2,s3,s4;
	scanf("%I64d%I64d%I64d%I64d",&a,&b,&c,&d);
	s1=a*b;s2=c*d;s3=s1,s4=s2;
	ll a1=0,b1=0,a2=0,b2=0,cnt=0;
	while(s3%3==0)
	{
		s3/=3;s3*=2;a1++; 
	}
	while(s3%2==0)
	{
		s3/=2;b1++;
	}
	while(s4%3==0)
	{
		s4/=3;s4*=2;a2++; 
	}
	while(s4%2==0)
	{
		s4/=2;b2++;
	}
	ll a3=a1-a2,b3=b1-b2;
	if(s3!=s4){printf("-1\n");return 0;}
	while(a3)
	{
		if(a3>0)
		{
			if(a%3==0)a=a/3*2;
			else b=b/3*2;
			a3--;
		}
		else
		{
			if(c%3==0)c=c/3*2;
			else d=d/3*2;
			a3++;
		}
		cnt++;
	}
	while(b3)
	{
		if(b3>0)
		{
			if(a%2==0)a/=2;
			else b/=2;
			b3--;
		}
		else
		{
			if(c%2==0)c/=2;
			else d/=2;
			b3++;
		}
		cnt++;
	}
	printf("%I64d\n",cnt);
	printf("%I64d %I64d\n%I64d %I64d\n",a,b,c,d);
	return 0;
}
