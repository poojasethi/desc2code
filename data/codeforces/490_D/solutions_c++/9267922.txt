#include <iostream>
using namespace std;
long long a[4],b[4];
long long minutes = 0;
long long checkposs(long long a[]) {
    long long m = a[0] * a[1];
    while(m%3==0) {
        m = m/3;
        a[3]++;
    }
    while(m%2==0) {
        m = m / 2;
        a[2]++;
    }
    return m;
}
void result(long long a[], long long b[], int r) {
    while (a[r] > b[r]) {
        a[r]--;
        minutes++;
        if(r == 3) {
            a[2]++;
            a[0] % r == 0 ? a[0] *= 2 : a[1] *= 2; 
        }
        a[0] % r == 0 ? a[0] /= r : a[1] /= r; 

    }
}
int main(int argc, char const *argv[])
{
    
    cin>>a[0]>>a[1];
    cin>>b[0]>>b[1];
    if(checkposs(a) != checkposs(b)) {
        cout<<"-1";
        return 0;
    }
    result(a,b,3);
    result(b,a,3);
    result(a,b,2);
    result(b,a,2);
    cout<<minutes<<endl;
    cout<<a[0]<<" "<<a[1]<<endl<<b[0]<<" "<<b[1];
    return 0;
}
