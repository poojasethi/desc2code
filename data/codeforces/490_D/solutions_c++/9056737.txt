#include <cstdio>
#include <iostream>

#define lint long long

int ar[2], br[2];

void process(lint x,int id){
 while(x % 2 == 0){
     x /= 2;
     ar[id]++;
 }
 while(x % 3 == 0){
     x /= 3;
     br[id]++;
 }
}

int main(){
 lint a1, b1, a2, b2;
 
 scanf("%I64d %I64d %I64d %I64d",&a1,&b1,&a2,&b2);
 process(a1,0);
 process(b1,0);
 process(a2,1);
 process(b2,1);
 
 int ans = 0;
 while(br[0] > br[1]){
     (a1 % 3 == 0) ? a1 = (a1 / 3) * 2 : b1 = (b1 / 3) * 2; 
     ar[0]++; br[0]--; ans++;
 }
 while(br[0] < br[1]){
     (a2 % 3 == 0) ? a2 = (a2 / 3) * 2 : b2 = (b2 / 3) * 2;
     ar[1]++; br[1]--; ans++;
 }
 while(ar[0] > ar[1]){
     (a1 % 2 == 0) ? a1 /= 2 : b1 /= 2;
     ar[0]--; ans++;
 }
 while(ar[0] < ar[1]){
     (a2 % 2 == 0) ? a2 /= 2 : b2 /= 2;
     ar[1]--; ans++;
 }
 
 if(a1 * b1 != a2 * b2) puts("-1");
 else{
     printf("%d\n",ans);
     printf("%I64d %I64d\n",a1,b1);
     printf("%I64d %I64d\n",a2,b2);
 }
 
 return 0;
}
