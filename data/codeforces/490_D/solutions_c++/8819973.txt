#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define LL long long
LL a1,a2,b1,b2,a,b,x,ans;
LL gcd(LL n,LL m){return m?gcd(m,n%m):n;}
int main()
{
	scanf("%d%d%d%d",&a1,&a2,&b1,&b2);
	a=a1*a2,b=b1*b2,x=gcd(a,b),a/=x,b/=x;
	if(b%2 && b%3 && b!=1)return printf("-1\n"),0;
	if(a%2 && a%3 && a!=1)return printf("-1\n"),0;
	while(b%3==0)
	  {
	  if(b1%3==0)b1/=3,b1*=2,b/=3,b*=2;
	  else b2/=3,b2*=2,b/=3,b*=2;ans++;
	  }
	x=gcd(a,b),a/=x,b/=x;
	while(a%3==0)
	  {
	  if(a1%3==0)a1/=3,a1*=2,a/=3,a*=2;
	  else a2/=3,a2*=2,a/=3,a*=2;ans++;
	  }
	x=gcd(a,b),a/=x,b/=x;
	while(b%2==0)
	  {
	  if(b1%2==0)b1/=2,b/=2;
	  else b2/=2,b/=2;ans++;
	  }
	while(a%2==0)
	  {
	  if(a1%2==0)a1/=2,a/=2;
	  else a2/=2,a/=2;ans++;
	  }if(a!=b)return printf("-1\n"),0;
	printf("%I64d\n%I64d %I64d\n%I64d %I64d\n",ans,a1,a2,b1,b2);return 0;
}
