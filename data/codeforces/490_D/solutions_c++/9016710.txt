#include <bits/stdc++.h>
using namespace std;
int ans, a[5], b[5];

long long f(int p[]) {
    long long s;
    s = (long long)p[0] * p[1];
    while (s % 2 == 0) {
        s /= 2;
        p[2]++;
    }
    while (s % 3 == 0) {
        s /= 3;
        p[3]++;
    }
    return s;
}

void g3(int p[], int q[]) {
    while (p[3] > q[3]) {
        ans++;
        p[3]--;
        p[2]++;
        if (p[0] % 3 == 0) p[0] = p[0] / 3 * 2;
        else p[1] = p[1] / 3 * 2;
    }
}

void g2(int p[], int q[]) {
    while (p[2] > q[2]) {
        ans++;
        p[2]--;
        if (p[0] % 2 == 0) p[0] /= 2;
        else p[1] = p[1] /= 2;
    }
}

int main() {
//    freopen("input.txt", "r", stdin);
    int i;
    cin >> a[0] >> a[1];
    cin >> b[0] >> b[1];
    if (f(a) != f(b)) {puts("-1"); return 0;}
    g3(a, b);
    g3(b, a);
    g2(a, b);
    g2(b, a);
    cout << ans << endl;
    cout << a[0] << ' ' << a[1] << endl;
    cout << b[0] << ' ' << b[1] << endl;
}
