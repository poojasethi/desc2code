#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
int main(){
	int a,b,c,d;
	int ans=0;
	scanf("%d%d%d%d",&a,&b,&c,&d);
	LL s1 = a*1LL * b;
	LL s2 = c*1LL * d;
	while( (s1%3==0) && (s2%3==0) ){
		s1 = s1*2/3;
		s2 = s2*2/3;
	}
	
	while( s1%3==0 ){
		s1=s1*2/3;
		if(a%3==0)	a=a*2/3;
		else	b=b*2/3;
		ans++;
	}
	
	while( s2%3==0 ){
		s2=s2*2/3;
		if(c%3==0)	c=c*2/3;
		else	d=d*2/3;
		ans++;
	}
	
	while( (s1%2==0) && (s2%2==0) ) s1/=2,s2/=2;

	while( s1%2==0 ){
		s1/=2;
		if(a%2==0)	a/=2;
		else	b/=2;
		ans++;
	}
	
	while( s2%2==0) {
		s2/=2;
		if(c%2==0)	c/=2;
		else	d/=2;
		ans++;
	}
	
	
	if(s1!=s2){
		puts("-1");
	}
	else{
		printf("%d\n",ans);
		printf("%d %d\n",a,b);
		printf("%d %d\n",c,d);
	}
	return 0;
}
