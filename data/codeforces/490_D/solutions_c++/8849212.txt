#include <iostream>
using namespace std;

int main(){
	long long a1, a2, b1, b2;
	cin >> a1 >> b1 >> a2 >> b2;
	
	long long s1 = a1 * b1;
	long long s2 = a2 * b2;
	
	while(s1 % 3 == 0 && s2 % 3 == 0) s1 = s1 / 3 * 2, s2 = s2 / 3 * 2;
	
	long sec = 0;
	while(s1 % 3 == 0){
		sec++;
		s1 = s1 / 3 * 2;
		if(a1 % 3 == 0)
			a1 = a1 / 3 * 2;
		else
			b1 = b1 / 3 * 2;
	}
	
	while(s2 % 3 == 0){
		sec++;
		s2 = s2 / 3 * 2;
		if(a2 % 3 == 0)
			a2 = a2 / 3 * 2;
		else
			b2 = b2 / 3 * 2;
	}
	
	while(s1 % 2 == 0 && s2 % 2 == 0) s1 = s1 /= 2, s2 /= 2;
	
	while(s1 % 2 == 0){
		sec++;
		s1 /= 2;
		if(a1 % 2 == 0)
			a1 /= 2;
		else
			b1 /= 2;
	}
	
	while(s2 % 2 == 0){
		sec++;
		s2 /= 2;
		if(a2 % 2 == 0)
			a2 /= 2;
		else
			b2 /= 2;
	}
	
	if(s1 == s2){
		cout << sec << endl
			 << a1 << " " << b1 << endl
			 << a2 << " " << b2;
	} else {
		cout << -1;
	}
	
	return 0;
}
