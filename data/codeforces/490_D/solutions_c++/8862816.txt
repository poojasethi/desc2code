#include <cstdio>

using namespace std;

typedef long long ll;
ll a, b, x, y, s, t, r;

ll step(ll a, ll b)
{
    ll s = 0ll;
    while (!(a % b)) a /= b, s++;
    return s;
}

int main()
{
    scanf("%I64d %I64d %I64d %I64d", &a, &b, &x, &y);

    s = step(a * b, 3ll); t = step(x * y, 3ll);

    while (s > t && !(a % 3ll)) a /= 3ll, a <<= 1ll, s--, r++;
    while (s > t && !(b % 3ll)) b /= 3ll, b <<= 1ll, s--, r++;
    while (t > s && !(x % 3ll)) x /= 3ll, x <<= 1ll, t--, r++;
    while (t > s && !(y % 3ll)) y /= 3ll, y <<= 1ll, t--, r++;

    s = step(a * b, 2ll); t = step(x * y, 2ll);

    while (s > t && !(a % 2ll)) a >>= 1ll, s--, r++;
    while (s > t && !(b % 2ll)) b >>= 1ll, s--, r++;
    while (t > s && !(x % 2ll)) x >>= 1ll, t--, r++;
    while (t > s && !(y % 2ll)) y >>= 1ll, t--, r++;

    if (a * b == x * y) printf("%I64d\n%I64d %I64d\n%I64d %I64d\n", r, a, b, x, y); else
    printf("-1\n");
    return 0;
}
