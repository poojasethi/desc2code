#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
void thanh(ll &x1,ll &x2,ll a)
{
	if(a==3){ 
	if(x1%3==0) { x1=x1*2/3; return;}
	else { x2=x2*2/3; return; }
	}
	else{
		if(x1%2==0) { x1/=2; return; }
		else { x2/=2; return; }
	}
}
main()
{
	ll a1,a2,b1,b2; cin>>a1>>a2>>b1>>b2;
	ll x=a1*a2,y=b1*b2;
	ll thanhdz=__gcd(x,y);
	x/=thanhdz; y/=thanhdz;
	while(x%2==0) x/=2;
	while(x%3==0) x/=3;
	while(y%2==0) y/=2;
	while(y%3==0) y/=3;
	if(x!=y) { cout<<-1; return 0;
	}
	ll ans=0;
	x=a1*a2/thanhdz; y=b1*b2/thanhdz;
	while(x%3==0)
	{
	thanh(a1,a2,3);
	x/=3;
	ans++;	
	}
	while(y%3==0)
	{
	thanh(b1,b2,3);
	y/=3;
	ans++;	
	}
	thanhdz=__gcd(a1*a2,b1*b2);
	x=a1*a2/thanhdz; y=b1*b2/thanhdz;
	while(x%2==0)
	{
	thanh(a1,a2,2);
	x/=2;
	ans++;	
	}
	while(y%2==0)
	{
	thanh(b1,b2,2);
	y/=2;
	ans++;	
	}
	cout<<ans<<endl<<a1<<" "<<a2<<endl<<b1<<" "<<b2<<endl;
}
