#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
typedef long long LL;

int main()
{
    LL a, b, c, d, s1, s2;
    cin>>a>>b>>c>>d;
    s1=a*b;
    s2=c*d;
    while(s1%3==0 && s2%3==0)
        s1=s1/3*2, s2=s2/3*2;
    int ans=0;
    while(s1%3==0)
    {
        ans++;
        s1=s1/3*2;
        if(a%3==0)a=a/3*2;
        else b=b/3*2;
    }
    while(s2%3==0)
    {
        ans++;
        s2=s2/3*2;
        if(c%3==0)c=c/3*2;
        else d=d/3*2;
    }
    while(s1%2==0 && s2%2==0)
        s1=s1/2, s2=s2/2;
    while(s1%2==0)
    {
        ans++;
        s1/=2;
        if(a%2==0)a/=2;
        else b/=2;
    }
    while(s2%2==0)
    {
        ans++;
        s2/=2;
        if(c%2==0)c/=2;
        else d/=2;
    }

//    cout<<s1<<" "<<s2<<" "<<ans<<endl;
    if(s1==s2)
        cout<<ans<<endl<<a<<" "<<b<<endl<<c<<" "<<d<<endl;
    else cout<<-1<<endl;
    return 0;
}
