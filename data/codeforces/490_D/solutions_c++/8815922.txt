#include<iostream>
#include<cstdio>
#define lld long long
using namespace std;
int arr[5],ans;
lld a1,a2,b1,b2,ta1,ta2,tb1,tb2;
void build(lld &x,int zz)
{
	while(x%2==0)
	{
		arr[2]+=zz;
		x/=2;
	}
	while(x%3==0)
	{
		arr[3]+=zz;
		x/=3;
	}
}
int main()
{
	cin>>a1>>b1>>a2>>b2;
	ta1=a1;ta2=a2;tb1=b1;tb2=b2;
	build(ta1,1);build(tb1,1);
	build(ta2,-1);build(tb2,-1);
	if(ta1*tb1!=ta2*tb2) {printf("-1");return 0;}
	while(arr[3]!=0)
	{
		ans++;
		if(arr[3]>0)
		{
			if(a1%3==0)	a1=a1/3*2;
			else b1=b1/3*2;
			arr[3]--;arr[2]++;
		}
		else
		{
			if(a2%3==0) a2=a2/3*2;
			else b2=b2/3*2;
			arr[3]++;arr[2]--;
		}
	}
	while(arr[2]!=0)
	{
		ans++;
		if(arr[2]>0)
		{
			if(a1%2==0) a1/=2;
			else b1/=2;
			arr[2]--;
		}
		else
		{
			if(a2%2==0) a2/=2;
			else b2/=2;
			arr[2]++;
		}
	}
	cout<<ans<<endl<<a1<<" "<<b1<<endl<<a2<<" "<<b2;
	return 0;
}
