#include <bits/stdc++.h>
using namespace std;
int ans, a[5], b[5];

long long f(int p[]) {
    long long s;
    s = (long long)p[0] * p[1];
    while (s % 2 == 0) {
        s /= 2;
        p[2]++;
    }
    while (s % 3 == 0) {
        s /= 3;
        p[3]++;
    }
    return s;
}

void g(int p[], int q[], int r) {
    while (p[r] > q[r]) {
        ans++;
        p[r]--;
        if (r == 3) {
            p[2]++;
            p[0] % r == 0 ? p[0] *= 2 : p[1] *= 2;
        }
        p[0] % r == 0 ? p[0] /= r : p[1] /= r;
    }
}

int main() {
//    freopen("input.txt", "r", stdin);
    cin >> a[0] >> a[1] >> b[0] >> b[1];
    if (f(a) != f(b)) {puts("-1"); return 0;}
    g(a, b, 3);
    g(b, a, 3);
    g(a, b, 2);
    g(b, a, 2);
    printf("%d\n%d %d\n%d %d", ans, a[0], a[1], b[0], b[1]);
}
