import sys

def dump(p, n):
	for i in xrange(0, n):
		print(p[i])

def count(dat, n, p, d, r, t):
	pre=10000000
	prer = ''
	pret = 1
	for i in xrange(0, n):
		x = dat[i]
		if x:
			current = 0
			while x % d == 0:
				x /= d
				current += 1
			if p[i] < pre:
				if t[i]:
					p[i] += current
				else:
					p[i] = 1
				r[i] += 'D'
			else:
				if pret:
					p[i] = pre + current
				else:
					p[i] = 1
				t[i] = pret
				r[i] = prer + 'R'
		else:
			t[i] = 0
			p[i] = 1
			r[i] = prer + 'R'
		pre = p[i]
		prer = r[i]
		pret = t[i]

def main():
	infile = sys.stdin
	outfile = sys.stdout
	n = int(infile.readline())
	p42 = [1000000] * n
	p45 = [1000000] * n
	p42[0] = 0
	p45[0] = 0
	t2 = [1] * n
	t5 = [1] * n
	p2 = [''] * n
	p5 = [''] * n
	for i in xrange(0, n):
		dat = [ int(x) for x in infile.readline().split() ]
		count(dat, n, p42, 2, p2, t2)
		count(dat, n, p45, 5, p5, t5)
	if p42[n-1] > p45[n-1]:
		outfile.write(str(p45[n-1]))
		outfile.write('\n')
		outfile.write(p5[n-1][1:])
	else:
		outfile.write(str(p42[n-1]))
		outfile.write('\n')
		outfile.write(p2[n-1][1:])
	

if __name__ == '__main__':
	main()
