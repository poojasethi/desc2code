#!/usr/bin/python

import sys

def CountMultipliers(v, m):
  if (v == 0): return 1
  result = 0
  while v % m == 0:
    result += 1
    v /= m
  return result


def Count(n, a):
  c = [None] * n
  f = [None] * n
  for i in range(n)[::-1]:
    c[i] = [None] * n
    f[i] = [None] * n
    for j in range(n)[::-1]:
      if i == n - 1 and j == n - 1:
        c[i][j] = a[i][j]
      else:
        if i == n - 1 or (j < n - 1 and c[i][j + 1] < c[i + 1][j]):
          c[i][j] = c[i][j + 1] + a[i][j]
          f[i][j] = False
        else:
          c[i][j] = c[i + 1][j] + a[i][j]
          f[i][j] = True
  return c[0][0], f


def PrintAnswer(n, c, f):
  print c
  i, j = 0, 0
  ans = []
  while (i, j) != (n - 1, n - 1):
    if f[i][j]:
      ans.append('D')
      i += 1
    else:
      ans.append('R')
      j += 1
  print ''.join(ans)


def main():
  n = int(sys.stdin.readline())
  a = []
  for i in range(n):
    a.append(map(int, sys.stdin.readline().split()))
  zi = -1
  for i in range(n):
    for x in a[i]:
      if x == 0:
        zi = i
        break

  ToMultipliers = lambda m: map(lambda x: map(lambda y: CountMultipliers(y, m), x), a)
  a2 = ToMultipliers(2)
  a5 = ToMultipliers(5)
  c2, f2 = Count(n, a2)
  c5, f5 = Count(n, a5)
  cm = min(c2, c5)
  if zi > -1 and cm > 1:
    print 1
    print 'D' * zi + 'R' * (n - 1) + 'D' * (n - 1 - zi)
  elif c2 < c5:
    PrintAnswer(n, c2, f2)
  else:
    PrintAnswer(n, c5, f5)


if __name__ == '__main__':
  main()
