from sys import stdin, maxint as inf
## from pprint import pprint as pp

def fpow(x, b):
    if not x: return 1

    result = 0
    while not x % b:
        x /= b
        result += 1
    return result

def main():
    readline = stdin.readline
    _fpow = fpow

    n = int(readline())
    dp2 = [inf] * (n+1); path2 = []
    dp5 = [inf] * (n+1); path5 = []
    dp2[0] = dp5[0] = 0

    ## A = [map(int, readline().split()) for i in xrange(n)]

    Z = None
    path2 = [['D']*n for _ in xrange(n)]
    path5 = [['D']*n for _ in xrange(n)]
    path2[0][0] = path5[0][0] = '0'
    for i in xrange(n):
        ## for j in xrange(n):
        for j, a in enumerate(map(int, readline().split())):
            ## a = A[i][j]
            if not a: Z = j, i

            if dp2[j-1] <= dp2[j]:
                dp2[j] = dp2[j-1]
                path2[i][j] = 'R'

            if dp5[j-1] <= dp5[j]:
                dp5[j] = dp5[j-1]
                path5[i][j] = 'R'

            dp2[j] += _fpow(a, 2)
            dp5[j] += _fpow(a, 5)
        ## print 'dp2 =', dp2
        ## print 'dp5 =', dp5

    # 0 and no free-zero way
    if Z and min(dp2[n-1], dp5[n-1]):
        x, y = Z
        zeros = 1
        path = [y*'D', x*'R', (n-1-y)*'D', (n-1-x)*'R']
    else:
        zeros, pathb = min((dp2[n-1], path2), (dp5[n-1], path5))
        ## print zeros
        ## [pp(' '.join(row)) for row in pathb]

        path = []
        i = j = n-1
        while i > 0 or j > 0:
            c = pathb[i][j]
            path.append(c)

            # exclusive
            j -= c == 'R'
            i -= c == 'D'
            ## print 'i, j =', i, j
        path.reverse()

    print zeros
    print ''.join(path)

if __name__ == '__main__':
    main()
