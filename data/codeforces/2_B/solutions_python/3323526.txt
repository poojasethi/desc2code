import sys
		
def translate(p, n):
	x = n-1
	y = n-1
	path = [0] * (2*n-2)
	while x or y:
		if p[x*n+y]:
			path[x+y-1] = 'R'
			y -= 1
		else:
			path[x+y-1] = 'D'
			x -= 1
	return ''.join(path)
		

def	count(dat, n, p, d, r, t):
	pre=10000000
	pret = 1
	for i in xrange(0, n):
		x = dat[i]
		if x:
			current = 0
			while x % d == 0:
				x /= d
				current += 1
			if p[i] < pre:
				if t[i]:
					p[i] += current
				else:
					p[i] = 1
				r.append(0)
			else:
				if pret:
					p[i] = pre + current
				else:
					p[i] = 1
				t[i] = pret
				r.append(1)
		else:
			t[i] = 0
			p[i] = 1
			r.append(1)
		pre = p[i]
		pret = t[i]

def main():
	infile = sys.stdin
	outfile = sys.stdout
	n = int(infile.readline())
	p42 = [1000000] * n
	p45 = [1000000] * n
	p42[0] = 0
	p45[0] = 0
	t2 = [1] * n
	t5 = [1] * n
	p2 = []
	p5 = []
	for i in xrange(0, n):
		dat = [ int(x) for x in infile.readline().split() ]
		count(dat, n, p42, 2, p2, t2)
		count(dat, n, p45, 5, p5, t5)
	if p42[n-1] > p45[n-1]:
		outfile.write(str(p45[n-1]))
		outfile.write('\n')
		outfile.write(translate(p5, n))
	else:
		outfile.write(str(p42[n-1]))
		outfile.write('\n')
		outfile.write(translate(p2, n))
	

if __name__ == '__main__':
	main()
