#include<bits/stdc++.h>
using namespace std;

const int N=1e3+10;

int f[N][N][2];
int g[N][N][2];
int n,k,x;

void print(int x,int y) {
  if(x==1&&y==1) {
    return;
  }
  if(g[x][y][k]) {
    print(x-1,y);
    putchar('D');
  } else {
    print(x,y-1);
    putchar('R');
  }
  return;
}
int main() {
  scanf("%d",&n);
  for(int i=2;i<=n;i++) {
    f[0][i][0]=f[0][i][1]=f[i][0][0]=f[i][0][1]=9999999;
  }
  for(int i=1;i<=n;i++) {
    for(int j=1;j<=n;j++) {
      scanf("%d",&k);
      if(k==0) {
        x=i;
      } else {
        for(;!(k&1);k>>=1) {
          f[i][j][0]++;
        }
        for(;!(k%5);k/=5) {
          f[i][j][1]++;
        }
      }
      for(k=0;k<2;k++) {
        if(g[i][j][k]=f[i-1][j][k]<f[i][j-1][k]) {
          f[i][j][k]+=f[i-1][j][k];
        } else {
          f[i][j][k]+=f[i][j-1][k];
        }
      }
    }
  }
  k=f[n][n][1]<f[n][n][0];
  if(x&&f[n][n][k]>1) {
    puts("1");
    for(int i=2;i<=x;i++) {
      putchar('D');
    }
    for(int i=2;i<=n;i++) {
      putchar('R');
    }
    for(int i=x+1;i<=n;i++) {
      putchar('D');
    }
  } else {
    printf("%d\012",f[n][n][k]);
    print(n,n);
  }
  return 0;
}
