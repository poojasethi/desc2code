#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;

int mul[1001][1001][2];
char g[1001][1001][2];
int n,k,x=0;

void prnt(int x,int y)
{
    if (x == 1 && y == 1) return ;
    if (g[x][y][k]) prnt(x - 1, y), putchar('D');
    else prnt(x, y - 1), putchar('R');
}

int main()
{
    cin>>n;
    memset(mul,0,sizeof(mul));
    for(int i=2;i<=n;i++)mul[0][i][0]=mul[0][i][1]=mul[i][0][1]=mul[i][0][0]=999999999;

    for(int i=1;i<=n;++i)
        for(int j=1;j<=n;++j)
    {
        cin>>k;
        if(!k)x=i;
        else
        {
            for(;!(k&1);k>>=1)mul[i][j][0]++;
            for(;!(k%5);k=k/5)mul[i][j][1]++;
        }

        for(int m=0;m<2;m++)
        if((g[i][j][m]= mul[i-1][j][m] < mul[i][j-1][m]))mul[i][j][m]+=mul[i-1][j][m];
        else mul[i][j][m] +=mul[i][j-1][m];

    }

        k= mul[n][n][1]<mul[n][n][0];
        if(x && mul[n][n][k] >1)
        {
            cout<<"1"<<endl;
            for(int i=2;i<=x;i++)cout<<"D";
            for(int i=2;i<=n;i++)cout<<"R";
            for(int i=x+1;i<=n;i++)cout<<"D";
        }
        else cout<<mul[n][n][k]<<endl, prnt(n,n);

    return 0;
}
