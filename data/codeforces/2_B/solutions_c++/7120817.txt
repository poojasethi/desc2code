#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int two[1001][1001],five[1001][1001],n,num,z=0;;
void out(int ar[][1001],int a,int b){string x;
    while(a&&b)
    if(ar[a-1][b]<ar[a][b-1]){a--;x="D"+x;}
    else {b--;x="R"+x;}
    x.erase(0,1);cout<<x<<endl;
}
int main(){
    memset(two,0,sizeof two);memset(five,0,sizeof two);
    scanf("%d",&n);n++;
    for (int i=2;i<n;i++) two[0][i] = five[0][i]=two[i][0] = five[i][0]= 10000;
    for(int i=1;i<n;i++)
    for(int j=1;j<n;j++){
    scanf("%d",&num);
    if(num){
            while(!(num&1)&&num){num>>=1;two[i][j]++;}
            while(!(num%5)&&num){num/=5;five[i][j]++;}
            }
    else {two[i][j]=five[z=i][j]=1;}
    if(two[i-1][j]<two[i][j-1])two[i][j]+=two[i-1][j];
    else two[i][j]+=two[i][j-1];
    if(five[i-1][j]<five[i][j-1])five[i][j]+=five[i-1][j];
    else five[i][j]+=five[i][j-1];
    }
    n--;
    if(z&&two[n][n]&&five[n][n]){cout<<'1'<<endl<<string(z-1,'D')+string(n-1,'R')+string(n-z,'D')<<endl;}
    else if(two[n][n]<five[n][n]){cout<<two[n][n]<<endl;out(two,n,n);}
    else {cout<<five[n][n]<<endl;out(five,n,n);}
    }
