#include <iostream>
#include <math.h>
#include <memory.h>
#include <string>
using namespace std;
int i,j,k,n,a[1111][1111][2],d[1111][1111][2],x,pr[1111][1111][2],ti;
bool t=false;
string s;
int main()
{
	cin>>n;
	memset(a,0,sizeof(a));
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
		{
			cin>>x;
			if (!x) { t=true; ti=i; continue; }
			while (!(x%2))
			{
				x/=2;
				a[i][j][0]++;
			}
			while (!(x%5))
			{
				x/=5;
				a[i][j][1]++;
			}
		}
	memset(d, 0x3F, sizeof(d));
	d[1][0][0]=d[1][0][1]=0;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			for(k=0;k<2;k++)
			{
				
					d[i][j][k]=d[i-1][j][k]+a[i][j][k];
					pr[i][j][k]=1;
					if (d[i][j][k]>d[i][j-1][k]+a[i][j][k])
					{
						d[i][j][k]=d[i][j-1][k]+a[i][j][k];
						pr[i][j][k]=0;
					}
			}
	if (d[n][n][1]>d[n][n][0])
		k=0;
	else k=1;
	if (t&&(d[n][n][k]>1))
	{
		cout<<1<<endl;
		for(i=1;i<ti;i++) cout<<'D';
		for(i=1;i<n;i++) cout<<'R';
		for(i=ti;i<n;i++) cout<<'D';
		return 0;
	}
	cout<<d[n][n][k]<<endl;
	i=n; j=n;
	while ((i>1)||(j>1))
	{
		if (pr[i][j][k])
		{
			i--;
			s=s+'D';
		}
		else
		{
			j--;
			s=s+'R';
		}
	}
	for(i=s.size()-1;i>=0;i--)
		cout<<s[i];
	return 0;
}
