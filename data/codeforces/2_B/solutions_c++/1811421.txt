#include <stdio.h>
#define for(i,b,e) for(i=b; i<=e; i++)
#define C 1001
main() {
  unsigned short d[C][C][2],i,j,l=0,n,zi;
  bool p[2*C],w[C][C][2],z=0,k;
  int t;
  scanf("%d",&n);
  for(i,0,n)
    for(j,0,n)
      d[i][j][0]=d[i][j][1]=0;
  for(i,2,n)
    d[0][i][0]=d[i][0][0]=d[0][i][1]=d[i][0][1]=65535;
  for(i,1,n)
    for(j,1,n) {
      scanf("%d",&t);
      if (t) {
        while (t%2==0)
          d[i][j][0]++,
          t/=2;
        while (t%5==0)
          d[i][j][1]++,
          t/=5;
        }
      else
        d[i][j][0]=d[i][j][1]=1,
        zi=i+1,
        z=1;
      for(t,0,1)
        d[i][j][t]+=d[i-1][j][t]<d[i][j-1][t]?d[i-1][j][t]:d[i][j-1][t],
        w[i][j][t]=d[i-1][j][t]<d[i][j-1][t];
    }
  k=d[n][n][0]>d[n][n][1];
  i=j=n;
  if (z && d[n][n][k]) {
    d[n][n][k]=1;
    for(t,zi,n)
      p[l++]=1;
    for(t,2,n)
      p[l++]=0;
    for(t,3,zi)
      p[l++]=1;
  }
  else
    while (i>1 || j>1)
      if (w[i][j][k])
        p[l++]=1,
        i--;
      else
        p[l++]=0,
        j--;
  printf("%d\n",d[n][n][k]);
  while (l--)
    putchar(p[l]?'D':'R');
  return 0;
}
