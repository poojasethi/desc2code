#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
char c1[12],c2[12],c3[12],id[1000];
int c[1000];
int main(){
	scanf("%s%s%s",c1,c2,c3);
	c[c1[0]]++; id[c1[0]]='F';
	c[c2[0]]++; id[c2[0]]='M';
	c[c3[0]]++; id[c3[0]]='S';
	if(c['r']==1 && c['s']==2)printf("%c\n",id['r']);
	else if(c['s']==1 && c['p']==2)printf("%c\n",id['s']);
	else if(c['p']==1 && c['r']==2)printf("%c\n",id['p']);
	else printf("?\n");
}
