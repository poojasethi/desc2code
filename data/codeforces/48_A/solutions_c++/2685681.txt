# include<stdio.h>
# include<string.h>
int main()
{
	int i,a[4];
	char str[10];
	for(i=1;i<=3;i++)
	{
		scanf("%s",str);
		if(str[0]=='r') a[i]=0;
		else if(str[0]=='s') a[i]=1;
		else a[i]=2;
	}

	if(a[2] == a[3] &&(a[2]==(a[1]+1)%3)) printf("F\n");
	else if(a[1] == a[3] &&(a[1]==(a[2]+1)%3)) printf("M\n");
	else if(a[1] == a[2] &&(a[1]==(a[3]+1)%3)) printf("S\n");
	else printf("?\n");
	return 0;
}
