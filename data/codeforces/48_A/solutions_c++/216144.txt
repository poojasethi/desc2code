#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

int a[3],x,y,z;
char s[3][10];
char name[3]={'F','M','S'};

int main(){
	scanf("%s%s%s",s[0],s[1],s[2]);
	for (int i=0;i<3;i++){
		if (s[i][0]=='r') a[i]=0;
		if (s[i][0]=='p') a[i]=1;
		if (s[i][0]=='s') a[i]=2;
	}

	for (int i=0;i<3;i++){
		x=(i+2)%3;
		y=(i+1)%3;
		z=(a[i]+2)%3;
		if (a[x]==z&&a[y]==z){
			printf("%c\n",name[i]);
			return 0;
		}
	}
	printf("?\n");

	return 0;
}
