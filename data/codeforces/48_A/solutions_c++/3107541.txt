#include <iostream>
using namespace std;
bool cmp(string s1, string s2)
{
 if (s1=="rock"&&s2=="scissors") return true;
 if (s1=="scissors"&&s2=="paper") return true;
 if (s1=="paper"&&s2=="rock") return true;
return false;
}
int main()
{string s1,s2,s3;
 cin>>s1>>s2>>s3;
 if (cmp(s1,s2)&&cmp(s1,s3)) {cout<<"F"; return 0;}
 if (cmp(s2,s1)&&cmp(s2,s3)) {cout<<"M"; return 0;}
 if (cmp(s3,s2)&&cmp(s3,s1)) {cout<<"S"; return 0;}
 cout<<"?";
return 0;
}
