#include <cstdio>
int main() {
    char s[20], v[3];
    for (int i = 0;i < 3;i++) gets(s), v[i] = (s[0] == 'r' ? 0 : (s[0] == 'p' ? 1 : 2));
    if (v[0] == (v[1] + 1) % 3 && v[0] == (v[2] + 1) % 3) puts("F");
    else if (v[1] == (v[0] + 1) % 3 && v[1] == (v[2] + 1) % 3) puts("M");
    else if (v[2] == (v[0] + 1) % 3 && v[2] == (v[1] + 1) % 3) puts("S");
    else puts("?");
}
