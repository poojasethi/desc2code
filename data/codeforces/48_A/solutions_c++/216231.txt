#include <iostream>

using namespace std;

bool win(string s1, string s2, string s3)
{
	return (s1[0] == 'r' && s2[0] == 's' && s3[0] == 's') || (s1[0] == 'p' && s2[0] == 'r' && s3[0] == 'r') || (s1[0] == 's' && s2[0] == 'p' && s3[0] == 'p');
}

int main()
{
	//freopen("a.in", "r", stdin);
	string s1, s2, s3;
	cin >> s1 >> s2 >> s3;
	if (win(s1, s2, s3)) cout << "F" << endl; else if (win(s2, s1, s3)) cout << "M" << endl; else if (win(s3, s1, s2)) cout << "S" << endl; else cout << "?" << endl;
}
