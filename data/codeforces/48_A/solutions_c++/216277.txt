#include <cstdio>

char jeu[3][20];

int main(){
	scanf("%s%s%s",jeu[0],jeu[1],jeu[2]);
	for (int i=0;i<3;i++){
		bool ok=true;
		for (int j=0;j<3;j++){
			if (i==j)
				continue;
			char c1=jeu[i][0],c2=jeu[j][0];
			if ((c1=='r' && c2=='s') || (c1=='s' && c2=='p') || (c1=='p' && c2=='r'))
				continue;
			ok=false;
		}
		if (ok){
			printf("%c\n","FMS"[i]);
			return 0;
		}
	}
	puts("?");
	return 0;
}
