#include <iostream>
using namespace std;
bool win(string a, string b, string c)
{
	if (a[0]=='p' && b[0]=='r' && c[0]=='r' ||
		a[0]=='r' && b[0]=='s' && c[0]=='s' ||
		a[0]=='s' && b[0]=='p' && c[0]=='p' ) return true;
	else return false;
}
int main()
{
	string F, M, S;
	cin>>F>>M>>S;
	if (win(F, M, S)) cout<<"F"<<endl;
	else if (win(M, S, F)) cout<<"M"<<endl;
	else if (win(S, F, M)) cout<<"S"<<endl;
	else cout<<"?"<<endl;
	return 0;
}
