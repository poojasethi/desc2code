#include <iostream>
#include <string>

using namespace std;

string F,M,S;

bool isWin(string s1,string s2)
{
	if (s1=="rock"&&s2=="scissors")
		return true;
	if (s1=="scissors"&&s2=="paper")
		return true;
	if (s1=="paper"&&s2=="rock")
		return true;
	return false;
}

string solve()
{
	if (isWin(F,M)&&isWin(F,S))
		return "F";
	if (isWin(M,F)&&isWin(M,S))
		return "M";
	if (isWin(S,F)&&isWin(S,M))
		return "S";
	return "?";
}

int main()
{
	cin >> F >> M >> S;
	cout << solve() << endl;
	return 0;
}
