#include <iostream>
#include <cstring>
using namespace std;
bool win(char a,char b,char c)
{
	if (a=='p')return b=='r' && c==b;
	if (a=='r')return b=='s' && b==c;
	if (a=='s')return b=='p' && b==c;
}
string s1,s2,s3;
int main()
{
	cin>>s1>>s2>>s3;
	if (win(s1[0],s2[0],s3[0]))cout<<"F";
	else if (win(s2[0],s1[0],s3[0]))cout<<"M";
	else if (win(s3[0],s1[0],s2[0]))cout<<"S";
	else cout<<"?";
	return 0;
}
