#include <iostream>
using namespace std;
bool beat(string a, string b){
    return (a == "scissors" && b=="paper" || a == "paper" && b=="rock" ||a == "rock" && b=="scissors");
}
int main(){
    string a,b,c;
    cin>>a>>b>>c;
    if (beat(a,b) && b==c)
        cout<<"F"<<endl;
    else if (beat(b,c) && a==c)
        cout<<"M"<<endl;
    else if (beat(c,a) && a==b)
        cout<<"S"<<endl;
    else
        cout<<"?"<<endl;
    //system("PAUSE");
    return 0;
}
