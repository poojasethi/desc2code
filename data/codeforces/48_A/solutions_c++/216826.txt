#include <iostream>
#include <string>

using namespace std;

bool w[200][200];
string s[10];

int main(){
	w['p']['r'] = 1;
	w['r']['s'] = 1;
	w['s']['p'] = 1;
	cin >> s[0] >> s[1] >> s[2];

	if((w[s[0][0]][s[1][0]])&&(w[s[0][0]][s[2][0]]))
		cout << "F" << endl;
	else if((w[s[1][0]][s[2][0]])&&(w[s[1][0]][s[0][0]]))
		cout << "M" << endl;
	else if((w[s[2][0]][s[1][0]])&&(w[s[2][0]][s[0][0]]))
		cout << "S" << endl;
	else
		cout << "?" << endl;

	return 0;
}
