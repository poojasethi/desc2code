#include <cstdio>
#include <map>
#include <iostream>

using namespace std;

map<string,long> str;

string str1,str2,str3;

int main()
{
 str["rock"]=1;
 str["paper"]=2;
 str["scissors"]=3;
 
 cin>>str1>>str2>>str3;
 
 if(str[str2]%3+1==str[str1]&&str[str3]%3+1==str[str1])
  cout<<"F";
 else if(str[str1]%3+1==str[str2]&&str[str3]%3+1==str[str2])
  cout<<"M";
 else if(str[str1]%3+1==str[str3]&&str[str2]%3+1==str[str3])
  cout<<"S";
 else
  cout<<"?";
 
 return 0;
}
