#include <stdio.h>
#include <iostream>
#include <string>
int main(void) {
char x[10], y[10], z[10];
int win[128][128] = {};
win['p']['r'] = 1;
win['r']['s'] = 1;
win['s']['p'] = 1;
scanf("%s%s%s", x, y, z);
if(y[0]==z[0] && win[x[0]][y[0]])
puts("F");
else if(x[0]==z[0] && win[y[0]][x[0]])
puts("M");
else if(x[0]==y[0] && win[z[0]][x[0]])
puts("S");
else
puts("?");
return 0;
}
