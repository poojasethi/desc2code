#include <iostream>
#include <stdio.h>
#include <map>
using namespace std;

map<string,string> m;
int main()
{
    m["scissors"]="paper";
    m["paper"]="rock";
    m["rock"]="scissors";
    string s1,s2,s3;
    cin>>s1>>s2>>s3;
    if(m[s1]==s2 && m[s1]==s3)cout<<"F\n";
    else if(m[s2]==s1 && m[s2]==s3)cout<<"M\n";
    else if(m[s3]==s1 && m[s3]==s2)cout<<"S\n";
    else cout<<"?\n";
    return 0;
}
