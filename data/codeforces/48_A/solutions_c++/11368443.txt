#include <iostream>
#include <map>
#include <string>
#include <cstdlib>

using namespace std;

map<string,int> mapa;

int v[3];
char c[4]="FMS";

int main()
{
  mapa["rock"]=0;
  mapa["paper"]=1;
  mapa["scissors"]=2;
  for (int i=0;i<3;i++) {
    string s;
    cin>>s;
    v[i]=mapa[s];
  }
  for (int i0=0;i0<3;i0++) {
    int i1=(i0+1)%3;
    int i2=(i0+2)%3;
    if (v[i1]==v[i2] and (v[i1]+1)%3==v[i0]) {
      cout<<c[i0]<<endl;
      exit(0);
    }
  }
  cout<<"?"<<endl;
}
