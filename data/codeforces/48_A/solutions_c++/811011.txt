#include<stdio.h>
bool win(char a,char b,char c)
{
    if(a=='r'&&b=='s'&&c=='s')return true;
    if(a=='s'&&b=='p'&&c=='p')return true;
    if(a=='p'&&b=='r'&&c=='r')return true;
    return false;
}
int main()
{
    char str1[10],str2[10],str3[10];
    gets(str1);gets(str2);gets(str3);
    if(win(str1[0],str2[0],str3[0]))printf("F\n");
    else if(win(str2[0],str1[0],str3[0]))printf("M\n");
    else if(win(str3[0],str1[0],str2[0]))printf("S\n");
    else printf("?\n");
    return 0;
}
