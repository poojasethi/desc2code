 #include <iostream>


 using namespace std;


 int main() {
     int a, b, c;
     char x[16], y[16], z[16];
     cin>>x>>y>>z;
     a = (x[0]=='s') + (x[0]=='r')*2;
     b = (y[0]=='s') + (y[0]=='r')*2;
     c = (z[0]=='s') + (z[0]=='r')*2;
     if (a == b && c == ((a+1)%3)) cout<<"S"<<endl;
     else if (a == c && b == ((a+1)%3)) cout<<"M"<<endl;
     else if (c == b && a == ((c+1)%3)) cout<<"F"<<endl;
     else cout<<"?"<<endl;
     return 0;
 }
