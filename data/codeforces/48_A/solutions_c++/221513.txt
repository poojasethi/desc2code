#include <iostream>
#include <string>
#include <map>
using namespace std;
/*---------------*/
map <string, string> A;
string x,y,z;
/*---------------*/
int main()
{
    cin >> x >> y >> z;
    A["paper"]="rock";
    A["rock"]="scissors";
    A["scissors"]="paper";
    if ((A[x]==y) && (A[x]==z)) cout << "F" << endl;
    else if ((A[y]==x) && (A[y]==z)) cout << "M" << endl;
    else if ((A[z]==x) && (A[z]==y)) cout << "S" << endl;
    else cout << "?" << endl;
}
