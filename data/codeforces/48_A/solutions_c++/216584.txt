#include <cstdio>

bool win(char s, char a, char b)
{
	if(a != b)
		return false;
	if(s == 'p' && a == 'r'
	|| s == 'r' && a == 's'
	|| s == 's' && a == 'p')
		return true;
	return false;
}

int main()
{
	char str[3][50];
	scanf("%s%s%s", str[0], str[1], str[2]);
	if(win(str[0][0], str[1][0], str[2][0]))
		puts("F");
	else if(win(str[1][0], str[0][0], str[2][0]))
		puts("M");
	else if(win(str[2][0], str[0][0], str[1][0]))
		puts("S");
	else
		puts("?");
	return 0;
}
