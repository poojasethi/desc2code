#include<iostream>
using namespace std;
int main()
{
  int a[3];
  string b1;
  for(int i=0;i<3;i++)
    {
      cin>>b1;
      if(b1=="rock")
	a[i]=2;
      else if(b1=="paper")
	a[i]=3;
      else
	a[i]=5;
    }
  int d=a[0]*a[1]*a[2];
  int d1=0;
  if(d==50)
    {
      d1=2;
    }
  else if(d==12)
    d1=3;
  else if(d==45)
    d1=5;
  if(d1!=0)
    {
      if(a[0]==d1)
	cout<<"F"<<endl;
      else if(a[1]==d1)
	cout<<"M"<<endl;
      else
	cout<<"S"<<endl;
    }
  else
    cout<<"?"<<endl;
}
