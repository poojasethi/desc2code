#include<iostream>
#include<map>
using namespace std;
map<string,string> mp;
int main(){
	mp["rock"]="scissors";
	mp["scissors"]="paper";
	mp["paper"]="rock";
	string a[3];
	char ans[3]={'F','M','S'};
	for(int i=0;i<3;i++)
		cin>>a[i];
	for(int i=0;i<3;i++){
		bool k=true;
		for(int j=0;j<3;j++)
			if(i!=j&&a[j]!=mp[a[i]])
				k=false;
		if(k){
			cout<<ans[i]<<endl;
			return 0;
		}
	}
	cout<<"?"<<endl;
}
