#include <iostream>
#include <string>
#include <map>
using namespace std;

map<string,string> rps;

string s1,s2,s3;

int main(){
    rps["rock"]="scissors";
    rps["paper"]="rock";
    rps["scissors"]="paper";
    cin>>s1>>s2>>s3;
    if (rps[s1]==s2 && s2==s3) cout<<"F"<<endl;
   else if (rps[s2]==s1 && s1==s3) cout<<"M"<<endl; else if (rps[s3]==s1 && s1==s2) cout<<"S"<<endl;
   else cout<<"?"<<endl;
    return 0;
}
