#include<cstdio>
char s[]="sprsprspr",a[22],b[22],c[22];
int find(char s1)
{
    for(int i=0;;i++)
    {
        if(s1==s[i]) return i;
    }
}
int main()
{
    int i,j,k,ans=-1;
    scanf("%s%s%s",a,b,c);
    int d=find(a[0]),e=find(b[0]),f=find(c[0]);
    if(d==e&&(f+1)%3==d)
        ans=3;
    if(d==f&&(e+1)%3==d) ans=2;
    if(e==f&&(d+1)%3==e) ans=1;
    if(ans==-1) puts("?");
    else if(ans==1) puts("F");
    else if(ans==2) puts("M");
    else puts("S");
}
