#include<cstdio>
char s[5][100];
int a[5];
int main(){
	scanf("%s%s%s",s[0],s[1],s[2]);
	
	for (int i=0;i<3;++i){
		if (s[i][0]=='p') a[i] = 2;
		else if (s[i][0]=='s') a[i] = 1;
		else if (s[i][0]=='r') a[i] = 0;
	}
	
	for (int i=0;i<3;++i){
		bool win = true;
		for (int j=0;j<3;j++){
			if (i==j) continue;
			if (a[j]!=(a[i]+1)%3) win=false;
		}
		
		if (win){
			if (i==0) puts("F");
			if (i==1) puts("M");
			if (i==2) puts("S");
			return 0;
		}
	}
	
	puts("?");

	return 0;
}
