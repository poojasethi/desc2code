#include <iostream>
using namespace std;

char s[3][9], *out = "FMS?";
int main() {
	cin >> s[0] >> s[1] >> s[2];
	int w = 3;
	for(int i = 0; i < 3; i++) {
		int ok = 1;
		for(int j = 0; j < 3; j++) if(j != i) {
			if(s[i][0] == 'r' && s[j][0] == 's') continue;
			if(s[i][0] == 'p' && s[j][0] == 'r') continue;
			if(s[i][0] == 's' && s[j][0] == 'p') continue;
			ok = 0;
		}
		if(ok) w = i;
	}
	cout << out[w] << endl;
	return 0;
}
