#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

Names = {0: 'F', 1: 'M', 2: 'S'}

def solve(M):
    d = len(set(M))
    
    if d != 2:
        return '?'
    
    if M.count('r') == 1 and 's' in M:
        return Names[M.index('r')]
    
    if M.count('s') == 1 and 'p' in M:
        return Names[M.index('s')]
    
    if M.count('p') == 1 and 'r' in M:
        return Names[M.index('p')]
    
    return '?' 

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

M = []
for _ in range(3):
    s = input.readline().strip()
    M.append(s[0])

a = solve(M)
output.write('%s\n' % a)
