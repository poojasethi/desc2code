f, m, s = raw_input(), raw_input(), raw_input()

wins = {'rock': {'rock': False, 'scissors': True, 'paper': False}, 'scissors': {'rock': False, 'scissors': False, 'paper': True}, 'paper': {'rock': True, 'scissors': False, 'paper': False}}

if (wins[f][m] and wins[f][s]):
    print 'F'
elif (wins[m][f] and wins[m][s]):
    print 'M'
elif (wins[s][f] and wins[s][m]):
    print 'S'
else:
    print '?'
