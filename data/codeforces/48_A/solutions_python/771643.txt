moves = [raw_input(), raw_input(), raw_input()]
beats = {}
beats['paper'] = 'rock'
beats['rock'] = 'scissors'
beats['scissors'] = 'paper'

if beats[moves[0]] == moves[1] and beats[moves[0]] == moves[2]:
    print 'F'
elif beats[moves[1]] == moves[0] and beats[moves[1]] == moves[2]:
    print 'M'
elif beats[moves[2]] == moves[1] and beats[moves[2]] == moves[0]:
    print 'S'
else:
    print '?'
