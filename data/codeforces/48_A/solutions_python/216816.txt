scissors = "scissors"
paper = "paper"
rock = "rock"

     F==paper and M==rock and S==rock or
     F==rock and M==scissors and S==scissors ):
	print "F"
elif ( M==scissors and S==paper and F==paper or
       M==paper and S==rock and F==rock or
       M==rock and S==scissors and F==scissors ):
	print "M"
elif ( S==scissors and M==paper and F==paper or
       S==paper and M==rock and F==rock or
       S==rock and M==scissors and F==scissors ):
	print "S"
else:
	print "?"
