import sys, string

a = []
m = {"rock\n":0, "scissors\n":1, "paper\n":2}
for line in sys.stdin:
    a.append(m[line])
if (a[0] + 1) % 3 == a[1] and (a[0] + 1) % 3 == a[2]:
    print "F"
elif (a[1] + 1) % 3 == a[2] and (a[1] + 1) % 3 == a[0]:
    print "M"
elif (a[2] + 1) % 3 == a[0] and (a[2] + 1) % 3 == a[1]:
    print "S"
else:
    print "?"
