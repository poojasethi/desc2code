#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    d = dict()
    for i in xrange(3):
        s = raw_input()
        if not d.has_key(s):
            d[s] = [1, i]
        else:
            d[s][0] += 1
    if len(d) == 2:
        d = d.items()
        i = 0 if d[0][1][0] == 1 else 1
        s1 = d[i][0]
        s2 = d[(i+1)%2][0]
        if (s1 == "rock" and s2 == "scissors") or (s1 == "paper" and s2 == "rock") or (s1 == "scissors" and s2 == "paper"):
            d = d[i][1][1]
            if d == 0:
                print "F"
            elif d == 1:
                print "M"
            else:
                print "S"
            return 0
    print "?"
    return 0

if __name__ == '__main__':
    main()
