rps = {'rock':1, 'paper':2, 'scissors':3}
pla = 'FMS'
def solve():
    a = [str(raw_input()) for i in range(3)]
    b = list(set(a))
    if len(b)==1 or len(b)==3:
        print '?'
        return
    x = a.count(b[0])
    if x==1 and (rps[b[0]]-rps[b[1]])%3==1:
        print pla[a.index(b[0])]
    elif x==2 and (rps[b[1]]-rps[b[0]])%3==1:
        print pla[a.index(b[1])]
    else:
        print '?'

solve()
