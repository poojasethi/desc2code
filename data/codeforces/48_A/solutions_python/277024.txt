import sys

a = map(lambda x: x[:-1], sys.stdin.readlines())
b = []
c = 'FMS'
d = {'rock':0, 'scissors':1, 'paper':2}

for i in xrange(3):
    cnt = 0
    for j in xrange(3):
        if (d[a[i]] + 1) % 3 == d[a[j]]:
            cnt += 1
    if cnt == 2:
        b.append(c[i])
if len(b) == 1: print b[0]
else: print '?'
