#!/usr/bin/env python

dic = {'rock': 0, 'paper': 1, 'scissors': 4}
names = {0: 'F', 1: 'M', 2: 'S'}

A = [0] * 3
A[0] = raw_input()
A[1] = raw_input()
A[2] = raw_input()

a = dic[A[0]]
a += dic[A[1]]
a += dic[A[2]]

if a == 0 or a == 3 or a == 12 or a == 2 or a == 9 or a == 5 or a == 4:
	print '?'
elif a == 1:
	print names[A.index('paper')]
elif a == 6:
	print names[A.index('scissors')]
elif a == 8:
	print names[A.index('rock')]
