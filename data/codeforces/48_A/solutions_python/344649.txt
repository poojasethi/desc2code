F = raw_input().strip()
M = raw_input().strip()
S = raw_input().strip()
if F == "rock":
    f = 0
elif F == "scissors":
    f = 1
else:
    f = 2
if M == "rock":
    m = 0
elif M == "scissors":
    m = 1
else:
    m = 2
if S == "rock":
    s = 0
elif S == "scissors":
    s = 1
else:
    s = 2
if m == s and m == (f+1)%3:
    print 'F'
elif f == s and f == (m+1)%3:
    print 'M'
elif m == f and m == (s+1)%3:
    print 'S'
else:
    print '?'
