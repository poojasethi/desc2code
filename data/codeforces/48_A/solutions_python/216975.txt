#!/usr/bin/python
import sys

def __main__ ():
	s = ['rock', 'paper', 'scissors']
	s += [s[0]]
	a = sys.stdin.readline ().strip ()
	b = sys.stdin.readline ().strip ()
	c = sys.stdin.readline ().strip ()
	for i in range (3):
		if a == s[i + 1] and b == s[i] and c == s[i]:
			print 'F'
			return 0
		if b == s[i + 1] and c == s[i] and a == s[i]:
			print 'M'
			return 0
		if c == s[i + 1] and a == s[i] and b == s[i]:
			print 'S'
			return 0
	print '?'
	return 0

if __name__ == "__main__":
	__main__ ()
