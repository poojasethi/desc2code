import sys
def win(x, y, z):
    if x == "rock" and y == "scissors" and z == "scissors": return True
    if x == "scissors" and y == "paper" and z == "paper": return True
    if x == "paper" and y == "rock" and z == "rock": return True
    return False

[f, m, s] = sys.stdin.readlines()
f = f.rstrip()
m = m.rstrip()
s = s.rstrip()
if win(f, m, s):
    print 'F'
elif win(m, f, s):
    print 'M'
elif win(s, f, m):
    print 'S'
else:
    print '?'
