#include<iostream>
#include<cstdio>
using namespace std;
int S[1001];
int main(){
    int n,m,h,tot=0;
    cin>>n>>m>>h;
    h--;
    for(int i=0;i<m;++i){
        cin>>S[i];
        tot+=S[i];
    }
    if(n>tot){puts("-1");return 0;}
    S[h]--;
    tot--;
    n--;
    double p=1.0;
    for(int i=0;i<n;++i){
        int x=tot-S[h]-i;
        p*=(x*1.0/(tot-i));
    }
    p=1.0-p;
    printf("%.8lf\n",p);
}
