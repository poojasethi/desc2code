#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,m,h,sum=0; scanf("%d %d %d",&n,&m,&h);
vector<int>v(m);
for(int i=0;i<m;i++)
    cin>>v[i],sum+=v[i];
if(sum<n)return printf("-1"),0;
h--;n--;v[h]--;sum--;
double res=1;
for(int i=0;i<v[h];i++)
    res*=double(sum-n-i)/(sum-i);
printf("%f",1-res);
return 0;
}
