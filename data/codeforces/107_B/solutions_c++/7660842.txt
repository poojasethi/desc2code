#include <cstdio>
 
int m, n, h;
int s[1001];

int main () {
	scanf("%d%d%d", &n, &m, &h);
	for (int i = 1; i <= m; ++i) 
		scanf("%d", s + i); 
	int S = 0;
	for (int i = 1; i <= m; ++i) 
		S = S + s[i]; 
	double p = 1;
	for (int i = 0; i < n-1; ++i) 
		p = p*((double(S-s[h] - i))/(S - 1 - i));
	if (S < n) puts("-1");
	else printf("%lf\n", 1 - p);
	return 0;
}
