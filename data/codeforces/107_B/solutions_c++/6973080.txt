#include <iostream>
#include <iomanip>
using namespace std;
int n,m,h, s, sh;
double ans = 1; 
int main()
{
    cin >> n >> m >> h;
    for(int k, i = 1; i <= m; i++)
    {
        cin >> k;
        if(i==h) sh = k;
        s += k;
    }  
    if( s < n) {cout << -1; return 0;}
    n--; sh--; s--;
    for(int i = 0; i < n; i++)
        ans*= 1.0*(s-sh-i)/(s-i);
    cout << setprecision(10) << 1-ans;
}
