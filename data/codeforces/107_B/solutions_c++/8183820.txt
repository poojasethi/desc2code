#include <cstdio>
int cnt[1001],n,h,m,tot;
int main(){
    scanf("%d%d%d",&n,&m,&h); h--;
    for (int i=0;i<m;i++)
        tot+=scanf("%d",cnt+i)-1+cnt[i];
    if (tot<n){
        printf("-1");return 0;
    }
    n--; cnt[h]--; tot--;
    double res=1;
    for (int i=0;i<cnt[h];i++) res*=double(tot-n-i)/double(tot-i);
    printf("%.9lf",1-res);
}
