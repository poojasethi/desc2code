#include <cstdio>
using namespace std;

int main()
{
	int N, M, H;
	scanf("%d %d %d", &N, &M, &H);

	int sum = 0, sh = 0;
	for (int i = 1; i <= M; i++)
	{
		int si;
		scanf("%d", &si);
		sum += si;
		if (i == H)
			sh = si;
	}

	if (sum < N)
		printf("-1\n");
	else
	{
		double prob = 1;
		for (int i = 0; i < sh - 1; i++)
			prob *= (double)(sum-N-i)/(sum-1-i);

		printf("%.9lf\n", 1-prob);
	}

	return 0;
}
