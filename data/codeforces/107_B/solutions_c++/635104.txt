#include <iostream>
#include <iomanip>
using namespace std;
int cnt[1001];
int n,h,m;
int tot;
int main()
{
	cin>>n>>m>>h; h--;
	for (int i=0;i<m;i++)
	{
		cin>>cnt[i]; tot+=cnt[i];
	}
	if (tot<n)
	{
		cout<<-1<<endl;
		return 0;
	}
	n--; cnt[h]--; tot--;
	double res=1;
	for (int i=0;i<cnt[h];i++) res*=double(tot-n-i)/double(tot-i);
	cout<<fixed<<setprecision(9)<<1-res<<endl;
}
