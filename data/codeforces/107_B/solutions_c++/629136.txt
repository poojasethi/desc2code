#include <cstdio>
#include <algorithm>
#define N 1005
using namespace std;
int n,m,h,s[N],TT=0;

int main(){
	scanf("%d%d%d",&n,&m,&h);
	
	for (int i=1;i<=m;i++){
		scanf("%d",&s[i]);
		TT+=s[i];
	}
	
	s[h]--;
	TT--;
	swap(s[1],s[h]);
	
	double AC = 1.0;
	
	for (int i=0;i<n-1;i++)
		AC *= (TT-s[1]-i)*1.0/(TT-i);
	
	if (TT+1<n) AC = 2.0;	
	printf("%.9lf\n",1.0-AC);
	
	//scanf("\n");
	return 0;
}
