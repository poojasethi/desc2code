#include <cstdio>
#include <cstring>

int main() {
	int n, m, h, i;
	int s=0, sh, t;
	scanf("%d%d%d",&n,&m,&h);
	for (i=1; i <= m; i++) {
		scanf("%d",&t);
		s += t;
		if (i == h)
			sh = t;
	}
	if (s < n) {
		printf("-1\n");
	}
	else if (s-sh < n-1) {
		printf("1\n");
	}
	else {
		double ans = 1;
		for (i=1; i < sh; i++) {
			ans *= 1.0*(s-sh-n+1+i)/(s-sh+i);
		}
		printf("%.8lf\n",1-ans);
	}
	return 0;
}
