#include <stdio.h>

int main() {
    int n, m, h, i;
    int s[1000];
    int sum = 0;
    double ans = 1;
    scanf("%d %d %d", &n, &m, &h);
    for (i = 0; i < m; i++) { scanf(" %d", s+i); sum += s[i]; }
    if (sum < n) { puts("-1"); return 0; }

    for (i = sum - 1; i > sum - s[h-1]; i--) ans *= (double)(i - n + 1) / i;
    printf("%.10f\n", 1 - ans);
    return 0;
}
