#include"stdio.h"
int main()
{
	int n,m,h;
	scanf("%d%d%d",&n,&m,&h);
	int nondept=0,dept=0;
	for(int x=1;x<=m;x++)
	{
		int tt=0;
		if(x==h)scanf("%d",&dept);
		else scanf("%d",&tt);
		nondept+=tt;
	}
	n--;
	dept--;
	if(dept+nondept<n){printf("-1.0");return 0;}
	//if(n==0){printf("0");return 0;}
	double ans=1.0L;
	for(int x=0;x<=n-1;x++)
	{
		ans*=((double)nondept-x)/(nondept+dept-x);
	}
	printf("%.15f",1.0-(double)ans);
}
