#include<cstdio>
#include<cmath>
int main(){
    int i,k,n,m,h,s0,s=0;
    double as=1;
    scanf("%d%d%d",&n,&m,&h);
    for(i=1;i<=m;i++){
        scanf("%d",&k);
        s+=k;
        if(i==h)s0=k;
    }
    if(s<n){puts("-1");return 0;}
    s0=s-s0;
    s--;n--;
    for(i=0;i<n;i++)
        as=as/(s-i)*(s0-i);
    printf("%.9lf\n",1-as);
    return 0;
}
