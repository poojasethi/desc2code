#include<cstdio>
int main()
{
	int i,n,k,s,a,b,d;
	double ans=1;
	scanf("%d%d%d",&n,&k,&s);
	for(a=b=0,i=1;i<=k;i++)
	{
		scanf("%d",&d);
		a+=d;
		if(i!=s)b+=d;
	}
	if(a<n)
		printf("-1\n");
	else if(b+1<n)
		printf("1\n");
	else
	{
		for(i=1;i<n;i++)
		{
			a--;
			ans*=1.0*b/a;
			b--;
		}
		printf("%lf\n",1-ans);
	}
	return 0;
}
