#include <iostream>
using namespace std;
int main () {
	int m, n, h;
	int s[1001];
	cin >> n >> m >> h;
	
	for (int i = 1; i <= m; i = i + 1) 
		cin >> s[i];
	int S = 0;
	for (int i = 1; i <= m; i = i + 1) 
		S = S + s[i]; 

	double p = 1;
	for (int i = 0; i < n-1; i = i + 1) 
		p = p*((double(S-s[h] - i))/(S - 1 - i));
	
	if (S < n) cout << "-1";
	else cout << 1 - p;
	return(0);
}
