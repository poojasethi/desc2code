#include <cstdio>
#include <vector>

using namespace std;

#define LL long long

int n, m, h, z, t, x;
double res = 1;

int main()
{
	scanf("%d%d%d", &n, &m, &h);
	
	for (int i = 1; i <= m; ++i)
	{
		scanf("%d", &x);
		t += x;
		if (i == h) z = x-1;
	}
	--n, --t;
	if (t < n) { printf("-1.0\n"); return 0; }
	
	for (int i = 0; i < n && t-i-z>=0; ++i)
	{
		res *= (t-i-z)/(double)(t-i);
	}
		
	printf("%lf\n", 1-res);
	
	return 0;
}
