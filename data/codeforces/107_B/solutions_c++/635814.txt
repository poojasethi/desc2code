#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
    int n,m,h,s[1000],ssum=0; scanf("%d%d%d",&n,&m,&h); h--;
    rep(i,m) scanf("%d",s+i), ssum+=s[i];
    s[h]--; ssum--; n--;

    double pr;
    if(ssum<n) pr=-1;
    else{
        pr=1;
        rep(i,n) pr*=(double)(ssum-s[h]-i)/(ssum-i);
        pr=1-pr;
    }

    printf("%.15f\n",pr);

    return 0;
}
