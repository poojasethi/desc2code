#include <iostream>
#include <cstdio>
using namespace std;
typedef long double ld;
int n,m,h;
int br[1010];
ld f,s;
int main(){
  cin>>n>>m>>h;
  for(int i=0;i<m;++i){
   cin>>br[i];
   s+=br[i];
   if(h==i+1)f=br[i];
                      }
  if(s<n)printf("-1.0\n");
  else{
   ld prob=1;
   for(int i=0;i<(f-1);++i)prob*=(s-n-i)/(s-1-i);
   printf("%.12lf\n",1-(double)(prob));
      }
  //system("pause");
  return 0;
}
