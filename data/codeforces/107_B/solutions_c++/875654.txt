#include "iostream"

using namespace std;

int main(){
	int n, m, h, q=0;
	int s[1001];
	cin >> n >> m >> h;
	for (int i=1; i<=m; i++){
		cin >> s[i];
		q=q+s[i];
	}
	if(n>q){
		cout << -1;
	}
	else{
		double p=1;
	for (int i=1; i<=n-1; i++){
		p=p*(q-s[h]+1-i)/(q-i);
	}
	cout << 1-p;
	}


	return 0;
}
