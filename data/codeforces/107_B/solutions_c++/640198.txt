#include <cstdio>
using namespace std;
int main(){
	int n, m, h;
	int t, ph;
	scanf( "%d %d %d", &n, &m, &h );
	t = 0;
	for( int i = 1; i <= m; i ++ ){
		int val;
		scanf( "%d", &val );
		t += val;
		if( i == h ){
			ph = val;
		}
	}

	if( t < n ){
		printf( "-1.0\n" );
	}else{
		double p = 1.0;
		for( int i = 0; i < n-1; i ++ ){
			p *= ( ( (double)( t - ph - i) ) / ( t - i - 1 ) ) ;
		}
		printf( "%.6f\n", 1-p );
	}

	return 0;
}
