#include<vector>
#include<iostream>

using namespace std;

int main(){
	int n,m,h,s1=0,i;
	cin>>n>>m>>h;
	vector <int> s(m+1);
	for (i=0; i<m; i++)
		cin>>s[i];
	for (i=0; i<m; i++)
	s1=s1+s[i];
	if (n<=s1)
	{
	double p=1;
	int a,b;
	for (i=0; i<n-1; i++)
	{
		a=s1-s[h-1]-i;
	b=s1-i-1;
		p=p*(double(a)/b);
	}
	cout<<(1-p);}
	else
		cout<<"-1";
return 0;
}
