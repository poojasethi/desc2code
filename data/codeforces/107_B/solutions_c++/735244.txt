#include<cstdio>
#include<cstdlib>

int n,m,p,k;
int s=0;

int main(){
    scanf("%d%d%d",&n,&m,&p);
    for(int i=1;i<=m;i++){
        int t;
        scanf("%d",&t);
        if(i==p) k=t;
        s+=t;
    }
    if(s<n){printf("-1.0");return 0;}
    n--;s--;k--;
    double ans = 1;
    for(int i=0;i<k;i++)
        ans*=(double) (s-n-i)/(s-i);
    printf("%.8llf",1-ans);
}
