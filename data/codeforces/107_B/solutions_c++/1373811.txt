#include <stdio.h>

int main (void) {
	double p;
	int n, m, h, t, s[1000];
	int i;
	scanf ("%d%d%d", &n, &m, &h);
	h--;
	for (t = i = 0; i < m; i++) {
		scanf ("%d", &s[i]);
		t += s[i];
	}
	if (t >= n) {
		p = 1;
		for (i = t-n+2; i <= t; i++)	p *= (i-s[h])/(double)(i-1);
	} else p = 2;
	printf ("%.7lf\n", 1.0-p);
	return 0;
}
