#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[1001];
int main()
{   int n,m,h,sum=0;
    cin>>n>>m>>h;
    for(int i=1;i<=m;i++)
    {   cin>>a[i];sum+=a[i];
    }
    if(sum<n) puts("-1");
    else if(sum-a[h]<n-1) puts("1");
    else
    {   double p=1;
        for(int i=1;i<a[h];i++)
        p*=(1-(n-1)*1.0/(sum-i));
        printf("%.8lf",1-p);
    }
    return 0;
}
