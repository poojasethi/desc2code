#include<iostream>

using namespace std;

int main()
{   int i,m,n,h;
    cin>>n;
    cin>>m;
    cin>>h;
    int s[1000],sum=0;
    double p=1;
    for(i=0;i<m;i++)
        {cin>>s[i];
        sum=sum+s[i];}
    for (i=0;i<n-1;i++)
        {p=p*(sum-s[h-1]-i)/(sum-i-1);}
    if(n>sum) 
        p=2;
    cout<<1-p;
    return 0;
}
