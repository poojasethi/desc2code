#include <cstdio>
const int N = 1005;
int a[N];
int main() {
	int n, m, h, sum = 0;
	scanf("%d%d%d", &n, &m, &h);
	for(int i = 1; i <= m; i ++) {
		scanf("%d", &a[i]);
		sum += a[i];
	}
	if(sum < n) puts("-1");
	else {
		if(sum - a[h] + 1 < n) puts("1");
		else {
			n --;
			double p = 1.0;
			for(int i = 1; i < a[h]; i ++) {
				p *= 1 - (double)n / (sum - i);
			}
			printf("%.8lf\n", 1 - p);
		}
	}
	return 0;
}
