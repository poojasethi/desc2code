#include<iostream>
#define maxn 1000
using namespace std;

int main()
{   int i,m,n,h;
    cin>>n>>m>>h;
    h--;
    int d[maxn],sum=0;
    long double p=1;
    for(i=0;i<m;i++)
    {
        cin>>d[i];
        sum=sum+d[i];
    }
    for (i=0;i<n-1;i++)
    {
        p=p*(sum-d[h]-i)/(sum-i-1);
    }
    if(n>sum)
        cout<<"-1";
    else
        cout<<1-p;
    return 0;
}
