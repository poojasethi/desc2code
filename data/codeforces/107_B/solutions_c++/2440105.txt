#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m;
int s[1010];
int main()
{
	int h;
	cin>>n>>m>>h;
	int sum=0;
	for(int i=1;i<=m;i++)
    {
		cin>>s[i];
		sum+=s[i];
	}
	if(n>sum) {
		return puts("-1.000"),0;
	}
	double ans=1.0;
	for(int i=0;i<n-1;i++)
	{
		//if(sum-s[h]-i>0)
	    ans=ans*(sum-s[h]-i)/(sum-1-i);
	}
	printf("%.9lf\n",1-ans);
	return 0;
}
