from fractions import *

def C(n, k):
  res = 1
  for i in xrange(1, k+1):
    res = res*(n-i+1)/i
  return res

def calc(a, b, n):
  s = a + b
  if s < n:
    return -1
  if b == 0:
    return 0

  good = 0
  for i in xrange(0, min(a+1, n)):
    good += C(a, i)*C(b, n-i) 
  return float(Fraction(good, C(s, n)))
