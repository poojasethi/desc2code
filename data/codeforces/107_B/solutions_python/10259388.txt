import sys
import math
import decimal

def choose(n, k):
    """
    A fast way to calculate binomial coefficients by Andrew Dalke (contrib).
    """
    if 0 <= k <= n:
        ntok = 1
        ktok = 1
        for t in xrange(1, min(k, n - k) + 1):
            ntok *= n
            ktok *= t
            n -= 1
        return ntok // ktok
    else:
        return 0

n, m, h = (int(x) for x in sys.stdin.readline().split())
values = [int(x) for x in sys.stdin.readline().split()]
rest = sum(values) - values[h-1]
total = sum(values) - 1
n_places_to_choose = n-1

if total < n_places_to_choose:
    print("-1")
    sys.exit(0)

top = choose(rest, n_places_to_choose)
bottom = choose(total, n_places_to_choose)
answer = 1 - decimal.Decimal(top)/bottom
print(round(answer, 6))
