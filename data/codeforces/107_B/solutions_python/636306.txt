n, m, h = map(int, raw_input().split())
s = map(int, raw_input().split())
n -= 1
h -= 1
s[h] -= 1


def negP(n, s, h):
  if (sum(s) - s[h]) < n:
    return 0.0
  else:
    all = sum(s)
    choosefrom = all - s[h]
    P = 1.0
    for _ in xrange(n):
      P *= 1.0 * choosefrom / all
      choosefrom -= 1
      all -= 1
    return P


if sum(s) < n:
  print -1
else:
  print 1.0 - negP(n, s, h)
