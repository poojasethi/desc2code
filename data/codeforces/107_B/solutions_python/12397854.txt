n,m,h=map(int,raw_input().split())
h-=1
a=map(int,raw_input().split())
n-=1
a[h]-=1
s=sum(a)
if s<n:
    print -1.0
else:
    p=1
    for i in range(n):
        p*=(s-a[h]-i)/1.0/(s-i)
    print 1-p
