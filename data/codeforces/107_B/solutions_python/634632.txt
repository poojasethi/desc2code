#!/usr/bin/env python
import sys;
n, m, h = [int(x) for x in sys.stdin.readline().split()]
v = [int(x) for x in sys.stdin.readline().split()][:m]
total = sum(v)

if total < n:
    print -1
else:
    x = total-v[h-1]
    y = v[h-1]-1

    res = 1
    for i in range(0, n-1):
        res *= float(x-i)/(x+y-i)
    print "{0:.9f}".format(1-res)
