#!/usr/bin/env python
import sys
from decimal import *

def C(n, k):
    res = Decimal(1)
    if n - k < k:
        k = n - k
    for i in range(k):
        res *= n - i
        res /= i + 1
    return res

def solve():
    getcontext().prec = 100
    n, _, h = map(int, sys.stdin.readline().split())
    n -= 1
    h -= 1
    counts = list(map(int, sys.stdin.readline().split()))
    counts[h] -= 1
    if sum(counts) < n:
        print -1
        return
    if sum(counts) - counts[h] < n:
        print 1
        return
    res = Decimal(0)
    A = counts[h]
    N = sum(counts)
    for i in range(1, min(n, A) + 1):
        a = C(A, i)
        a *= C(N - A, n - i)
        res += a
    res /= C(N, n)
    getcontext().prec = 10
    print +res

solve()
