#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout
import operator as op

def product(A,initial=1):
    return reduce(op.mul, A, initial)

def solve(n,m,h,S):
    n -= 1
    h -= 1
    S[h] -= 1
    sh = S[h]
    s = sum(S)
    
    if s < n:
        return -1
    if s-sh < n:
        return 1
    
    R = [float(a) / b for a,b in zip(range(s-sh,s-sh-n,-1),
                                     range(s,s-n,-1))]
    p = product(R)

    return 1-p

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

n,m,h = tuple(numbers_from_line())
S = numbers_from_line()

a = solve(n,m,h,S)
output.write('%s\n' % a)
