#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	vector< pair<int,string> > v[10005];
	int n,m,i,r,p;
	string s;
	cin>>n>>m;
    for(i=1;i<=n;i++)
    {
    	cin>>s>>r>>p;
    	v[r].push_back(make_pair(p,s));
	}
	int sz;
	for(i=1;i<=m;i++)
	{
		sort(v[i].begin(),v[i].end());
		sz=v[i].size();
		if(sz>2&&v[i][sz-2].first==v[i][sz-3].first)
		cout<<"?"<<endl;
		else
		cout<<v[i][sz-1].second<<" "<<v[i][sz-2].second<<endl;
	}
	return 0;
}
