#include<bits/stdc++.h>
using namespace std;
vector<pair<int,string> > v[10010];
int n,m;
int main(){
	cin >> n >> m;
	for(int i = 0 ; i < n ; i++){
		string name;
		int tm , pnt;
		cin >> name >> tm >> pnt;
		v[tm].push_back(make_pair(pnt,name));
	}
	for(int i = 1 ; i <= m ; i++) sort(v[i].rbegin(),v[i].rend());
	for(int i = 1 ; i <= m ; i++){
		if(v[i].size()>2&&v[i][1].first == v[i][2].first) cout << "?\n";
		else cout << v[i][0].second << " " << v[i][1].second << endl;
	}
}
