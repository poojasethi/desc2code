#include<cstdio>
#include<algorithm>
using namespace std;
char s[100005][12];

struct node {
	int i, t;
	node() {
		t = -1;
	}
}a[10005][4];

bool cmp(node a, node b) {
	return a.t > b.t;
}

int main() {
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i = 0; i < n; i++) {
		int t, tt;
		scanf("%s%d%d", s[i], &t, &tt);
		a[t][3].t = tt;
		a[t][3].i = i;
		sort(a[t], a[t]+4, cmp);
	}
	for(int i = 1; i <= m; i++) { 
		if(a[i][0].t == a[i][2].t || a[i][1].t == a[i][2].t) printf("?\n");
		else printf("%s %s\n", s[a[i][0].i], s[a[i][1].i]);
	}
	return 0;
}
