#include <bits/stdc++.h>

#define LL long long

using namespace std;

int main() {
	LL i,j,k,l,n,m,t;
	vector<pair<LL,string> > vec[10004];
	cin >> n >> m;
	string x;
	for(i=0;i<n;i++) {
		cin >> x >> j >> k;
		vec[j].push_back(make_pair(-1*k,x));
	}
	for(i=1;i<=m;i++) {
		sort(vec[i].begin(), vec[i].end());
		if (vec[i].size() == 2 || vec[i][1].first != vec[i][2].first ) {
			cout << vec[i][0].second << " " << vec[i][1].second << endl;
		}
		else {
			cout << "?" << endl;
		}
	}
	return 0;
}
