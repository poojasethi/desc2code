#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair

using namespace std;

int main()
{
	int n, m, i, a, p;
	string s;
	vector< pair<int,string> > v[10003];
	cin >> n >> m;
	while(n--)
	{
		cin >> s >> a >> p;
		v[a].pb(mp(p,s));
		
	}
	for(i=1;i<=m;i++)
	{
		sort(v[i].begin(),v[i].end());
		int sz=v[i].size();
		if(sz>2 && v[i][sz-2].first==v[i][sz-3].first) cout << "?" << endl;
		else cout << v[i][sz-1].second << " " << v[i][sz-2].second << endl;
	}
	return 0;
}
