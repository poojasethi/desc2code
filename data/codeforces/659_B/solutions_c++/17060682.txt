#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
using namespace std;
int main()
{
  int n,m,t1,t2;
  cin>>n>>m;
  string s;
  vector<pair<int,string> > v[m+1];
  for(int i=0;i<n;i++)
  {
  	cin>>s>>t1>>t2;
  	v[t1].pb(mp(t2,s));
  }
  for(int i=1;i<=m;i++)
  sort(v[i].begin(),v[i].end(),greater<pair<int,string> > ());
  for(int i=1;i<=m;i++)
  {
  	 if(v[i].size()!=2 &&v[i][1].first==v[i][2].first)
  	  {
  	   cout<<"?"<<endl;	
  	  }
  	  else
  	  cout<<v[i][0].second<<" "<<v[i][1].second<<endl;
  }
  return 0;
}
