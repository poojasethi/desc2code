#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
long long n,m,x,y;
string t;
vector <pair <long long,string> > r[100001];
int main() {
	cin >> n >> m;
	for(long long i=0;i<n;i++){
		cin >> t >> x >> y;
		r[x].push_back(make_pair(y,t));
	}
	for(long long i=1;i<=m;i++){
		sort(r[i].begin(),r[i].end());
		reverse(r[i].begin(),r[i].end());
		if(r[i].size()>2){
			if(r[i][1].first==r[i][2].first){
				cout << '?' << endl;
				continue;
			}
		}
		cout << r[i][0].second << " " << r[i][1].second << endl;
	}
	return 0;
}
