#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main() {
	ll i,j,k,l,n,m,t;
	vector<pair<ll,string> > v[10004];
	cin >> n >> m;
	string x;
	for(i=0;i<n;i++) {
		cin >> x >> j >> k;
		v[j].push_back(make_pair(-1*k,x));
	}
	for(i=1;i<=m;i++) {
		sort(v[i].begin(), v[i].end());
		if (v[i].size() == 2 || v[i][1].first != v[i][2].first ) {
			cout << v[i][0].second << " " << v[i][1].second << endl;
		}
		else {
			cout << "?" << endl;
		}
	}
	return 0;
}
