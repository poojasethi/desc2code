#include <bits/stdc++.h>

using namespace std;

bool comp(pair<string, int> &a, pair<string, int> &b)
{
	return a.second > b.second;
}

int main()
{
	int n, m, a, b;
	cin >> n >> m;
	string s;
	vector<pair<string, int> > vec[m+1];
	for(int i = 0; i < n; i++)
	{
		cin >> s >> a >> b;
		vec[a].push_back(make_pair(s, b));
	}
	for(int i = 1; i <= m; i++)
	{
		sort(vec[i].begin(), vec[i].end(), comp);
		if(vec[i].size() > 2 && vec[i][1].second == vec[i][2].second)
			cout << "?\n";
		else
			cout << vec[i][0].first << ' ' << vec[i][1].first << endl;
	}
}
