#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;

int n,m;
vector<pair<int,string> > v[1000000];

int main()
{
  ios::sync_with_stdio(false);
  cin>>n>>m;
  for (int i=0;i<n;i++) {
    string s;
    int r;
    int p;
    cin>>s>>r>>p;
    v[r].push_back(pair<int,string> (p,s));
  }
  for (int i=1;i<=m;i++) {
    vector<pair<int,string> > &w=v[i];
    sort(w.begin(),w.end());
    reverse(w.begin(),w.end());
    if (int(w.size())>=3 and w[1].first==w[2].first) cout<<"?"<<endl;
    else cout<<w[0].second<<" "<<w[1].second<<endl;
  }
}
