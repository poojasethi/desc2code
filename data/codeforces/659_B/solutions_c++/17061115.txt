#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,m;
	cin >> n >> m;
	vector<pair<int, string> >v[10005];
	
	for(int i=0;i<n;i++)
	{
		string s;
		int r, sc;
		cin >> s >> r >> sc;
		v[r].push_back(make_pair(sc, s));
	}
	for(int i=1;i<=m;i++)
	{
		sort(v[i].rbegin(), v[i].rend());
		if(v[i].size() == 2)
		{
			cout << v[i][0].second << " " << v[i][1].second << "\n";
		}
		else
		{
			if(v[i][1].first == v[i][2].first) cout << "?\n";
			else cout << v[i][0].second << " " << v[i][1].second << "\n";
		}
	}
}
