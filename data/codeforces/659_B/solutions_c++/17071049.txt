#include <bits/stdc++.h>

using namespace std;

typedef pair<int,string> pp;
vector<pp> ans[100001];
int n,m,t,pt;
string s;
int main()
{
    cin>>n>>m;
    for(int i=0;i<n;++i)
    {
        cin>>s>>t>>pt;
        ans[t].push_back(pp(pt,s));
    }
    for(int i=1;i<=m;++i)
    {
        vector<pp> &v=ans[i];
        sort(v.begin(),v.end(),greater<pp>());
        if(v.size()>2&&v[1].first==v[2].first) cout<<'?'<<endl;
        else cout<<v[0].second<<' '<<v[1].second<<endl;
    }
    return 0;
}
