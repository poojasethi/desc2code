#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n, m, a, b;
	cin >> n >> m;
	string s;
	vector<pair<int, string> > vec[m+1];
	for(int i = 0; i < n; i++)
	{
		cin >> s >> a >> b;
		vec[a].push_back(make_pair(-b, s));
	}
	for(int i = 1; i <= m; i++)
	{
		sort(vec[i].begin(), vec[i].end());
		if(vec[i].size() > 2 && vec[i][1].first == vec[i][2].first)
			cout << "?\n";
		else
			cout << vec[i][0].second << ' ' << vec[i][1].second << endl;
	}
}
