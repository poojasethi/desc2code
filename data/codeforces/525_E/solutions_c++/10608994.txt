#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <string>
#include <queue>
#include <set>
#include <utility>
#include <map>
#include <vector>

using namespace std;

long long f[20];
int n,k;
long long s,ans;
long long a[30];
map <long long,long long> mp[30];

inline void d1(int i,int t,long long sum)
{
	if (t>k||sum>s) return;
	if (i==n/2)
	{
		++mp[t][sum];
		return;
	}
	if (a[i]<=18) d1(i+1,t+1,sum+f[a[i]]);
	d1(i+1,t,sum);
	d1(i+1,t,sum+a[i]);
}

inline void d2(int i,int t,long long sum)
{
	if (t>k||sum>s) return;
	if (i==n)
	{
		for (int j=0;j+t<=k;j++)
		{
			if (mp[j].count(s-sum))
			ans+=mp[j][s-sum];
		}
		return;
	}
	if (a[i]<=18) d2(i+1,t+1,sum+f[a[i]]);
	d2(i+1,t,sum);
	d2(i+1,t,sum+a[i]);
}

int main()
{
	f[0]=1;
	for (int i=1;i<=18;i++)
	{
		f[i]=f[i-1]*i;
	}
	scanf("%d%d%lld",&n,&k,&s);
	for (int i=0;i<n;i++)
	{
		scanf("%lld",a+i);
	}
	ans=0;
	d1(0,0,0);
	d2(n/2,0,0);
	printf("%lld",ans);
	return 0;
}
