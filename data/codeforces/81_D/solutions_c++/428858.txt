#include <iostream>
#include <vector>

using namespace std;

int select_max(const vector<int>& v, int b1, int b2, int fst) {
  int res = -1;
  for (int i=0; i < v.size(); i++)
    if (i != b1 && i != b2 && v[i] && (res == -1 || v[res] < v[i] || 
        v[res] == v[i] && (res == fst) < (i == fst)))
      res = i;

  return res;
}

int main() {
  int N, M;
  cin >> N >> M;
  vector<int> cnt(M);
  for (int i=0; i < M; i++) {
    cin >> cnt[i];
  }

  vector<int> res;
  for (int i=0; i < N; i++) {
    int e = select_max(cnt, i ? res[i-1] : -1, i == N-1 ? res[0] : -1, i ? res[0] : -1);
    if (e == -1) {
      cout << -1 << endl;
      return 0;
    }
    cnt[e]--;
    res.push_back(e);
  }

  for (int i=0; i < res.size(); i++)
    cout << res[i]+1 << " ";
  cout << endl;

  return 0;
}
