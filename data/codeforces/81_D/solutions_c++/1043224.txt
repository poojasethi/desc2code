#include <iostream>
#include <vector>
#include <algorithm>

#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl

using namespace std;

int n,m,a[50],c[2000];
int d,e,f,g,h,i,j,k,l;

bool cmp(int x,int y){return a[x]>a[y];}

int main(){
  
  cin>>n>>m;
  rep(i,m)cin>>a[i];
  if(m==1){
    if(n==1)pr("1");
    else pr("-1");
    return 0;
  }


  rep(i,n){
    d=-1;
    for(j=0;j<m;j++){
      if((d==-1?a[j]>0:a[j]>=a[d])&&((i==0)||j!=c[i-1])&&((i==n-1&&j!=c[0])||i!=n-1)&&(a[j]>a[d]?true:j==c[0]))d=j;
    }
    c[i]=d;a[d]--;
    if(d==-1){pr("-1");return 0;}
  }

  rep(i,n)cout<<c[i]+1<<" ";
  pr("");


  return 0;

}
