#include<iostream>
#include<algorithm>
using namespace std;
int x[1001],y[45];
int main()
{
    int n,m;
    cin >> n >> m;
    for(int i=1; i<=m; i++)
        cin >> y[i];
    x[1] = 1;
    y[1]--;
    for(int i=2; i<=n; i++)
    {
        int mid=0;
        for(int j=1; j<=m; j++)
        {
            if(y[j]>y[mid] && j != x[i-1] && j != x[(i+1)%n])
                mid = j;
        }
        if(mid == 0)
        {
            cout << -1 << endl;
            return 0;
        }
        y[mid]--;
        x[i] =mid;
    }
    for(int i=1; i< n ;i++)
    {
        cout << x[i] << ' ';
    }
    cout << x[n] << endl;
    return 0;
}
