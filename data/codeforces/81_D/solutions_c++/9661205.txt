#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#define MAXN 1000
#define INF 1e9
using namespace std;
int n,m;
int a[MAXN +10];
int ans[MAXN +10];
int nr()
{
	int ans,k = 0;
	for(int i = 1;i <= m;i++)
		if(a[i] >= k)
		{
			ans = i;
			k = a[i];
		}
	return ans;
}
int main()
{
	int i,j;
	scanf("%d%d",&n,&m);
	for(i = 1;i <= m;i++)
		scanf("%d",&a[i]);
	ans[1] = 1;
	a[1]--;
	for(i = 2;i <= n;i++)
	{
		int pos = 0;
		for(j = 1;j <= m;j++)
			 if(a[pos] < a[j] && j != ans[i - 1] && j != ans[(i + 1) % n])
				pos = j;
		if(pos == 0)
		{
			printf("-1");
			return 0;
		}
		ans[i] = pos;
		a[pos]--;
	}
	printf("%d",ans[1]);
	for(i = 2;i <= n;i++)
		printf(" %d",ans[i]);
}
