#include <bits/stdc++.h>
#define MAXN 1000
#define MAXM 40
#define REP(i,f,t) for (int i=f; i<=t; i++)

using namespace std;

int n, m;
int A[MAXM + 10];
int P[MAXN + 10];

void read()
{
	scanf("%d%d", &n, &m);
	REP(i, 1, m) scanf("%d", &A[i]);
}

void work()
{
	//第一个位置直接放第一个1
	P[1] = 1, A[1]--;
	
	REP(i, 2, n)
	{
		int pos = 0;
		REP(j, 1, m)
			if (A[pos] < A[j] && j != P[i-1] && j != P[(i+1)%n])
				pos = j;
		if (pos == 0)
		{
			printf("-1");
			return;
		}
		P[i] = pos;
		A[pos]--;
	}
	REP(i, 1, n)
		printf(i == 1 ? "%d" : " %d", P[i]);
	return;
}

int main()
{
	read();
	work();
}
