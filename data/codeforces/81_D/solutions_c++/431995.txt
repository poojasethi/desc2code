#include <iostream>
#include <string.h>

using namespace std;

int a[50];
int answer[1010];

int main()
{
	
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < m; ++i)
		cin >> a[i];

	memset(answer, -1, sizeof(answer));

	for (int i = 0; i < n; ++i){
		int best = -1;
		for (int j = 0; j < m; ++j) {
			if (j == answer[(i - 1 + n) % n] || 
				j == answer[(i + 1) % n] || a[j] == 0)
				continue;
			if (best == -1 || a[j] > a[best])
				best = j;
			else if (a[j] == a[best] && j == answer[0])
					best = j;
		}

		if (best == -1) {
			cout << "-1";
			return 0;
		}
		answer[i] = best;
		--a[best];
	}
	for (int i = 0; i < n; ++i)
		cout << (answer[i] + 1) << " ";

	return 0;
}
