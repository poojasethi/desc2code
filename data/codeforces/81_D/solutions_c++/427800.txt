#include<cstdio>

int a[44];
int N,M,cur=0,pos=0;
int b[1005];

int main(){
	scanf("%d%d",&N,&M);
	
	for (int i=1;i<=M;++i) scanf("%d",&a[i]);
	
	pos = 0;
	int count = 0;
	while (count < N){
		if (a[cur]==0){
			cur = 1;
			for (int i=2;i<=M;++i){
				if (a[i]>a[cur]) cur = i;
			}
			pos = (pos+2)%N;
		}
		
		if (a[cur]==0){
			printf("-1\n");
			return 0;
		}
		
		int cnt = 0;
		int pre = pos;
		
		while (cnt<=N && (b[pos] || b[(pos+1)%N] == cur || b[(pos+N-1)%N]==cur)){
			pos = (pos+1)%N;
			++cnt;
		}
		
		if (cnt>N){
			pos = pre;
			a[cur] = 0;
		}else{
			b[pos] = cur;
			--a[cur];
			count++;
			//printf("%d %d\n",pos,cur);
		}
		
		
	}
	
	
	for (int i=0;i<N;++i){
		printf("%d",b[i]);
		if (i==N-1) puts("");
		else printf(" ");
	}
	return 0;
}
