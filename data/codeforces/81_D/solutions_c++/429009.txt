#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n,m,ret[1005],a[45],ok;

int main(){

	scanf("%d%d", &n, &m);
	for (int i=1; i<=m; i++) scanf("%d", &a[i]);

	ok = 1;
	for (int i=1; i<=n; i++){
		ok = 0;
		for (int j=1; j<=m; j++){
			if (a[j] && j!=ret[i-1] && (i<n || j!=ret[1])){
				if (i>1){
					if (j==ret[1] || a[j] > a[ok]) ok = j;
				}
				else{
					if (a[j] < a[ok] || !ok) ok = j;
				}
			}
		}
		if (!ok) break;
		a[ok]--;
		ret[i] = ok;
	}

	if (!ok) printf("-1\n");
	else{
		for (int i=1; i<=n; i++) printf("%d%c", ret[i], i<n?' ':'\n');
	}

	return 0;
}
