#include<cstdio>
#define MAXM 40
#define MAXN 1000
#include<algorithm>
using namespace std;
int n,m,ans[MAXN+10],cnt,a[MAXM+10];
int main()
{
	int i,j=0,k;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;i++){
		scanf("%d",&a[i]);
	}
	ans[1]=1;
	a[1]--;
	for(i=2;i<=n;i++){
		k=0;
		for(j=1;j<=m;j++)
			if(a[k]<a[j]&&j!=ans[(i-1+n)%n]&&j!=ans[(i+1)%n])
				k=j;
		if(!k){
			puts("-1");
			return 0;
		}
		ans[i]=k;
		a[k]--;
	}
	for(i=1;i<n;i++)
		printf("%d ",ans[i]);
	printf("%d\n",ans[i]);
}
