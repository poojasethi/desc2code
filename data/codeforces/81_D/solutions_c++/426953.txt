#include <cstdio>
#include <algorithm>
#include <utility>
using namespace std;

int main(){
	int n, m;
	pair<int, int> a[40];
	scanf("%d %d", &n, &m);
	for(int i=0; i<m; i++){
		scanf("%d", &a[i].first);
		a[i].second=i+1;
	}
	sort(a, a+m);
	reverse(a, a+m);
	int ord[40000], cnt=0;
	for(int i=0; i<m; i++){
		for(int j=0; j<a[i].first; j++)
			ord[cnt++]=a[i].second;
	}
	int ans[1000]={0};
	int i, j=0;
	for(i=0; i<n; i++){
		int pos= n&1 ? 2*i%n : 2*i%n + (2*i>=n);
		while(j<cnt && (ord[j]==ans[(pos+1)%n] || ord[j]==ans[(pos-1+n)%n]))
			j++;
		if(j==cnt)
			break;
		ans[pos]=ord[j];
		j++;
	}
	if(i==n){
		for(i=0; i<n; i++)
			printf("%d ", ans[i]);
		putchar('\n');
	}else{
		puts("-1");
	}
	return 0;
}
