#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <vector>
#include <algorithm>
#include <string>
#include <cstring>

using namespace std;

const int inf = 1<<30;

int n, m;
vector<int> a;
vector<int> ans;

int main()
{
  scanf("%d%d", &n, &m);
  a.resize(m);
  ans.assign(n, -1);
  for (int i=0; i<m; i++)
    scanf("%d", &a[i]);
  
  for (int i=0; i<n; i++)
  {
    int prev = ((i-1)+n)%n;
    int next = (i+1)%n;
    
    int maxid = -1;
    int maxval = -inf;
    
    for (int j=0; j<m; j++)
      if (j != ans[prev] && j != ans[next] && (a[j] > maxval || (i > 0 && a[j] == maxval && j == ans[0])))
      {
	maxid = j;
	maxval = a[j];
      }
      
    if (maxid == -1 || maxval == 0)
    {
      cout << -1;
      return 0;
    }
    
    a[maxid]--;
    ans[i] = maxid;
  }
  
  for (int i=0; i<n; i++)
    cout << ans[i] + 1 << ' ';
}
