#include<cstdio>
#define MAXN 1000
int A[MAXN+10];
int ans[MAXN+10];
int main()
{
	int n,m,i,j;
	scanf("%d%d",&n,&m);
	int sum=0;
	int max=0;
	int pos=0;
	for(i=1;i<=m;i++)
	{
		scanf("%d",&A[i]);
		sum+=A[i];
		if(A[i]>max) max=A[i];
	}
	if(sum<n)
	{
		printf("%d\n",-1);
		return 0;
	}
	sum-=max;
	int x=(n+1)/2;
	if(sum<x) 
	{
		printf("%d\n",-1);
		return 0;
	}
	ans[1]=1;
	A[1]--;
	for(i=2;i<=n;i++)
	{
		pos=0;
		for(j=1;j<=m;j++)
		{
			if(i+1>n) x=1;
			else x=i+1;
			if(A[pos]<A[j]&&j!=ans[i-1]&&j!=ans[x])
				pos=j;
		}
		if(pos==0)
		{
			printf("%d\n",-1);
			return 0;
		}
		ans[i]=pos;
		A[pos]--;
	}
	for(i=1;i<n;i++)
		printf("%d ",ans[i]);
	printf("%d\n",ans[i]);
}
