#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
#define fi first
#define se second
int a[10000], m, n;
pair<int, int> b[50];
void gao(int s){
    int i, j;
    for(i = s; i < n; i += 2){
        for(j = m - 1; j >= 0; j--)
            if(b[j].fi > 0 && b[j].se != a[(i - 1 + n) % n] && b[j].se != a[(i + 1) % n]){
                b[j].fi--;
                a[i] = b[j].se;
                break;
            }
    }
}
int main(){
    int i, j, k;
    scanf("%d%d", &n, &m);
    for(i = 0; i < m; i++){
        scanf("%d", &b[i].fi);
        b[i].se = i;
    }
    memset(a, -1, sizeof(a));
    sort(b, b + m);
    gao(0);
    gao(1);
    if(count(a, a + n, -1) > 0) puts("-1");
    else for(i = 0; i < n; i++) printf("%d%c", a[i] + 1, i < n - 1 ? ' ' : '\n');
    scanf("%*d");
}
