#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;

int srt[1005];
int a[1005];

int cmp(int x, int y) {
	return a[x] < a[y];
}

int main(void) {
	int n, m, i, j;
	int s=0;
	scanf("%d%d", &n, &m);
	for(i=0;i<m;i++) {
		scanf("%d", &a[i]);
		if(a[i]>n/2) a[i] = n/2;
		s += a[i];
		srt[i] = i;
	}
	sort(srt, srt+m, cmp);
	if(s < n) {
		puts("-1");
		return 0;
	}

	vector<int> p[1005];
	vector<int> w;
	for(i=0;i<m-1;i++)
		for(j=0;j<a[srt[i]];j++)
			w.push_back(srt[i]);
	j=0;
	for(i=0;i<n-a[srt[m-1]];i++) {
		p[j].push_back(w[i]);
		if(++j==a[srt[m-1]])
			j = 0;
	}
	for(i=0;i<a[srt[m-1]];i++) {
		printf("%s%d", i==0?"":" ", srt[m-1]+1);
		for(j=0;j<p[i].size();j++)
			printf(" %d", p[i][j]+1);
	}
	puts("");
	return 0;
}
