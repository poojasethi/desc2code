#include <iostream>
#include <algorithm>
using namespace std;
typedef pair<int, int> pii;

pii arr[1000];
int ans[1000];
int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < m; i++) {
		cin >> arr[i].first;
		arr[i].second = i + 1;
	}
	sort(arr, arr + m, greater<pii>());
	int pos = 0, loop = 0;
	for (int i = 0;;) {
		while (i < m && arr[i].first == 0) i++;
		if (i >= m) break;
		arr[i].first--;
		int cur = arr[i].second;
		if (ans[(pos + n - 1) % n] != cur && ans[(pos + 1) % n] != cur) {
			ans[pos] = cur;
			pos += 2;
			if (pos >= n) {
				if (loop == 0) {
					loop = 1;
					pos = 1;
				} else {
					for (int i = 0; i < n; i++) {
						cout << ans[i] << ' ';
					}
					return 0;
				}
			}
		}
	}
	cout << -1;
}
