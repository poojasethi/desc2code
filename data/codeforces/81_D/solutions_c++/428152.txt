#include <cstdio>
#include <algorithm>

using namespace std;

const int h = 1010;

int n, m;
int a[h], d[h];

int main() {
	scanf("%d%d", &n, &m);
	for (int i = 0; i < m; i++)
		scanf("%d", &d[i+1]);
	a[0] = 1;
	d[1] --;
	for (int i = 1; i < n; i++) {
		int k = 0;
		if (a[i-1] != 1 && a[(i+1) % n] != 1 && d[1] > 0)
			k = 1;
		else {
			for (int j = 1; j <= m; j++)
				if (j != a[i-1] && j != a[(i+1) % n] && d[j] > d[k])
					k = j;
		}
		if (k == 0) {
			puts("-1");
			return 0;
		}
		a[i] = k;
		d[k] --;
	}
	for (int i = 0; i < n; i++)
		printf("%d ", a[i]);
	puts("");
	return 0;
}
