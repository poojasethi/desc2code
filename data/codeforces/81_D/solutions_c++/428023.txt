#include<cstdio>
#include<cstring>

#define NMAX 1000
#define MMAX 40
int n,m,a[MMAX],b[NMAX],c[MMAX];

bool solve(int x) {
	int i,j,k;
	b[0]=x;
	a[x]--;
	for(i=1;i<n;i++) {
		k=-1;
		for(j=0;j<m;j++) {
			if(j==b[i-1])continue;
			if(i+1==n&&j==x)continue;
			if(!a[j])continue;
			if(k==-1||a[j]>a[k])k=j;
		}
		if(k==-1)return false;
		b[i]=k;
		a[k]--;
	}
	return true;
}

bool solve() {
	memcpy(c,a,sizeof(int)*m);
	for(int i=0;i<m;i++) {
		memcpy(a,c,sizeof(int)*m);
		if(solve(i))return true;
	}
	return false;
}

void output() {
	for(int i=0;i<n;i++) {
		if(i)putchar(' ');
		printf("%d",b[i]+1);
	}
	puts("");
}

void input() {
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++)scanf("%d",a+i);
}

int main() {
	input();
	if(solve())output();
	else puts("-1");
	return 0;
}
