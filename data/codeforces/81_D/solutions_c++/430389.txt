#include <iostream>
#include <cstdio>

#define MAX 1000

using namespace std;

int n, m, a[MAX + 5], c[MAX + 5];

int main() {

    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; i++)
        scanf("%d", &a[i]);

    for (int i = 1; i <= n; i++) {
        c[n + 1] = c[1];
        int k = 0;
        for (int j = 1; j <= m; j++)
            if (j != c[i - 1] && j != c[i + 1])
                if (k == 0 || a[j] > a[k] || (a[j] == a[k] && j == c[1]))
                    k = j;
        if (a[k] == 0) {
            printf("-1");
            return 0;
        }
        c[i] = k;
        a[k]--;
    }

    for (int i = 1; i <= n; i++) {
        printf("%d", c[i]);
        if (i < n)
            printf(" ");
    }

    return 0;
}
