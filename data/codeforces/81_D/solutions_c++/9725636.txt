#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn = 1010;
int a[maxn],path[maxn];
int n,m;
int main()
{
    while(cin>>n>>m)
    {
        bool flag=true;
        a[0]=0;
        for(int i=1;i<=m;i++)
            cin>>a[i];
        path[1]=1;a[1]--;
        for(int i=2;i<=n;i++)
        {
            int temp=0;
            for(int j=1;j<=m;j++)
                if(a[j]>a[temp]&&j!=path[i-1]&&j!=path[(i+1)%n])
                    temp=j;
            if(!temp)
            {
                puts("-1");
                flag=false;
                break;
            }
            path[i]=temp;a[temp]--;
        }
        if(!flag) continue;
        for(int i=1;i<=n;i++)
            cout<<path[i]<<" ";
    }
    return 0;
}
