#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
const int maxn = 1e3 + 5;
const int maxm = 40 + 5;
int a[maxm];
int n, m;
int ans[maxn];
void solve()
{
	ans[0] = 1;
	a[1]--;
	for (int i = 1; i < n; i++)
	{
		int pos = 0;
		for (int j = 1; j <= m; j++)
			if (a[j] > a[pos] && j != ans[i-1] && j != ans[(i+1)%n])
				pos = j;
		if (pos == 0)
		{
			cout << -1 << endl;
			return;
		}
		ans[i] = pos;
		a[pos]--;
	}
	for (int i = 0; i < n; i++)
		cout << ans[i] << ' ';
}
int main()
{
	//freopen("test.in", "r", stdin);
	cin >> n >> m;
	for (int i = 1; i <= m; i++)
		cin >> a[i];
	solve();
	return 0;
}
