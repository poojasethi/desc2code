#include <cstdio>
using namespace std;

int N,M,A[1001],r[1001];
int main () {
    scanf("%d %d",&N,&M);
    for (int i=1;i<=M;++i) scanf("%d",&A[i]);
    for (int i=1;i<=N;++i) {
        int b=0;
        for (int j=1;j<=M;++j) if (A[j] && (i!=N || j!=r[1]) && j!=r[i-1] && (A[j]>A[b] || A[j]==A[b] && j==r[1])) b=j;
        if (!b) { printf("-1"); return 0; }
        r[i]=b,--A[b];
    }
    if (r[N]==r[1]) printf("-1");
    else for (int i=1;i<=N;++i) printf("%d ",r[i]);
    return 0;
}
