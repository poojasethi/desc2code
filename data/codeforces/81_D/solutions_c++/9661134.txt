#include<bits/stdc++.h>
using namespace std;

const int MAXN = 1000;
const int MAXM = 40;

int n, m;
int A[MAXM + 10];
int Path[MAXN + 10];

int main()  {
    int i, j;

    scanf("%d%d", &n, &m);
    for(i = 1; i <= m; i++) scanf("%d", &A[i]);

    Path[1] = 1, A[1]--;

    for(i = 2; i <= n; i++) {
        int pos = 0;
        for(j = 1; j <= m; j++) {
            if(A[pos] < A[j] && j != Path[i - 1] && j != Path[(i + 1) % n])
                pos = j;
        }
        if(pos == 0)    {
            puts("-1");
            return 0;
        }
        Path[i] = pos;
        A[pos]--;
    }
    for(i = 1; i <= n; i++)
        printf(i == n ? "%d\n" : "%d ", Path[i]);

    return 0;
}
