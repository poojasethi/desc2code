#include <cstdio>
#include <algorithm>
#include <functional>
#define MAXN 1024
#define MAXM 64
using namespace std;
int main() {
	int n, m, c[MAXN];
	pair<int, int> cnt[MAXM];
	while(scanf("%d%d", &n, &m) != EOF) {
		for(int i = 0; i < m; ++i) {
			scanf("%d", &cnt[i].first);
			cnt[i].second = i + 1;
		}
		sort(cnt, cnt + m, greater<pair<int, int> >());
		int i, j;
		fill(c, c + n, -1);
		for(i = 0, j = 0; i < n; i += 2) {
			if(cnt[j].first == 0 || cnt[j].second == c[(i - 1 + n) % n] || cnt[j].second == c[(i + 1) % n])
				++j;
			if(j < m) {
				c[i] = cnt[j].second;
				--cnt[j].first;
			}
			else
				break;

			if(i % 2 == 0 && i + 2 >= n)
				i = -1;
		}
		if(i >= n)
			for(i = 0; i < n; ++i)
				printf("%d%c", c[i], (i == n - 1 ? '\n' : ' '));
		else
			puts("-1");
	}
	return 0;
}
