#include<cstdio>
#include<cstring>
#include<algorithm>

#define mp       make_pair
#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef	pair<int,int> pii;

int main(){
	int n,m; scanf("%d%d",&n,&m);
	int a[40];
	rep(i,m) scanf("%d",a+i);

	pii ord[40];
	rep(i,m) ord[i]=mp(-a[i],i);
	sort(ord,ord+m);

	int ans[1000],pos=0;
	memset(ans,-1,sizeof ans);
	rep(k,2){
		for(int i=k;i<n;i+=2){
			int now,j;
			for(j=0;j<m;j++){
				now=ord[pos].second;
				int prev=ans[(i-1+n)%n],next=ans[(i+1)%n];
				if(a[now]>0 && (prev!=now && next!=now)) break;
				pos=(pos+1)%m;
			}
			if(j==m){ puts("-1"); return 0; }
			ans[i]=now;
			a[now]--;
		}
	}

	rep(i,n) printf("%d%c",ans[i]+1,i<n-1?' ':'\n');

	return 0;
}
