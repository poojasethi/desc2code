#include <stdio.h>
#include <algorithm>
using namespace std;

#define rep(i,n) for (int i=0; i<n; i++)
#define MAXN 1000
#define MAXM 40

int n,m;
int a[MAXM+10];
int r[MAXN+10];
int prev,best,idx;

int main() {
	scanf("%d %d", &n,&m);
	rep(i,n) scanf("%d", &a[i]); 

	prev=-1;
	rep(i,n) {
		best=0;
		rep(j,m) if (best<a[j] && j!=prev) {
			if (i==n-1 && j==r[0]) continue;
			idx=j; best=a[j];
		}

		if (!best) {
			printf("-1\n");
			return 0;
		}

		if (i<n-1 && best==a[r[0]] && prev!=r[0]) idx=r[0];

		r[i]=idx;
		a[idx]--;
		prev=idx;
	}

	printf("%d", r[0]+1);
	rep(i,n) if (i) printf(" %d",r[i]+1);
	printf("\n");

	return 0;
}
