#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;
const int N=1005;
int a[N],res[N];

int main()
{
     int n,m,x,k,i,j;
     scanf("%d%d",&n,&m);
     for (i=1; i<=m; i++) scanf("%d",&a[i]);
     res[1]=1; a[1]--;
     for (i=2; i<n; i++)
     {
         x=0; k=0;
         if (res[i-1]!=1 && a[1]>0) k=1;
         else
         {
             for (j=1; j<=m; j++)
              if (j!=res[i-1] && a[j]>x)
               {x=a[j]; k=j;}
         }
         if (k==0) break;
         res[i]=k; a[k]--;
     }
     if (i<n) {printf("-1\n"); return 0;}
     for (j=1; j<=m; j++)
      if (a[j]>0 && j!=res[n-1] && j!=res[1])
      {res[n]=j; break;}
     if (res[n]==0) {printf("-1\n"); return 0;}
     for (i=1; i<=n; i++)
      printf("%d ",res[i]);
}
