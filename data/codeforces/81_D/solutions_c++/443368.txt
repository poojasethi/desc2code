int main(){

	int n,m,a[1005],res[1005];
	rep(i,1005) a[i]=res[i]=0;
	cin>>n>>m;
	rep(i,m) cin>>a[i];
	bool ok=true;
	rep(i,n){
		ok=false;
		rep(j,m)if(a[j] && (i==0 || res[i-1]!=j) && (i<n-1 || res[0]!=j)){
			if(i>0&&(!ok||res[0]==j||(res[0]!=res[i]&&a[res[i]]<a[j]))) res[i]=j;
			if(i==0&&(!ok || (a[res[i]]<a[j]))) res[i]=j;
			ok=true;
		}
		a[res[i]]--;
		if(!ok) break;
	}
	if(!ok) cout<<"-1"<<endl;
	else rep(i,n){
		cout<<res[i]+1;
		if(i==n-1) cout<<'\n';
		else cout<<' ';
	}
}
