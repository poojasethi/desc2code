#include<stdio.h>
int a[50],b[1050];
int main()
{
    int n,m,i,k,j;
    while(~scanf("%d%d",&n,&m))
    {
        for(i=1; i<=m; i++)
            scanf("%d",&a[i]);
         for(i=1;i<=n;i++)
         {
             k=0;
             for(j=1;j<=m;j++)
                if(a[j]&&j!=b[i-1]&&(i!=n||j!=b[1])) //保证数组a有数值，在输出项保证每项不同，保证首尾不相等
                if(a[j]>a[k])k=j;
             else if(a[j]==a[k]&&j==b[1])k=j;
             if(!k)
             {
                 puts("-1");
                 return 0;
             }
             b[i]=k;
             a[k]--;
         }
         for(i=1;i<=n;i++)
            printf("%d ",b[i]);
         printf("\n");

    }
}
