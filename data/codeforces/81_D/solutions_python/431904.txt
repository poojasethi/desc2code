r=lambda:map(int,raw_input().split())
n,m=r()
l=sorted(zip(r(),range(m)))[::-1]
p=[-1]*n
i=0
for j in range(m):
    mm=l[j][0]
    ind=l[j][1]
    lst = i
    for k in xrange(n):
        if mm==0: break
        if p[i]==-1 and p[(i+1)%n]!=ind and p[(i-1+n)%n]!=ind:
            p[i]=ind
            mm-=1
            lst = i
        i=(i+1)%n
    i = (lst+2)%n
if p.count(-1):
    print -1
else:
