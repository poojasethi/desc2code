R=lambda:map(int,raw_input().split())
n,m=R()
a=sorted(enumerate(R()),key=lambda x:x[1],reverse=1)
v,i=[-1]*n,0
for x,y in a:
  t=i
  for j in xrange(n+10):
    if not y:break
    if v[i]<0 and v[i-1]!=x+1 and v[(i+1)%n]!=x+1:
      y,v[i],t=y-1,x+1,i
    i=(i+1)%n
  i=(t+2)%n
print -1 if v.count(-1) else ' '.join(map(str,v))
