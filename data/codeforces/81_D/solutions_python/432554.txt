#!/usr/bin/env python

from sys import stdin

def solve(N, M, A):
   if N == 1:
      if M > 0:
         return [1]
      else:
         return [-1]

   P = [ (A[i], i+1) for i in xrange(M) ]
   P.sort()

   R = [0] * N
   i = M-1

   for n in xrange(0, N-1, 2):
      if P[i][0] <= 0: i -= 1
      if i < 0: return [-1]
      P[i] = (P[i][0]-1, P[i][1])
      R[n] = P[i][1]
   # print R

   for n in xrange(1, N, 2):
      if P[i][0] <= 0 or P[i][1] == R[n-1] or n < N-1 and P[i][1] == R[n+1]:
         i -= 1
      if i < 0: return [-1]
      P[i] = (P[i][0]-1, P[i][1])
      R[n] = P[i][1]
   # print R

   if N % 2 == 1:
      if P[i][0] <= 0: i -= 1
      if i < 0: return [-1]
   #  P[i] = (P[i][0]-1, P[i][1])
      if i > 0:
         R[N-1] = P[0][1]
      else:
         R[N-1] = R[0]
         R[0] = P[i][1]
         if R[0] == R[1]: return [-1]
   
   return R

if __name__ == '__main__':
   N, M = map(int, stdin.readline().split())
   A = map(int, stdin.readline().split())
   ans = solve(N, M, A)
   print ' '.join(map(str, ans))
