#include<cstdio>
#include<cstring>
#include<cmath>
#define nn 810
using namespace std;
struct nod{int end,next,c;double s;
}g[nn*nn*4];
int x[nn],pre[nn],y[nn],q[nn*nn*4],fa[nn];
int tem,n,t,j,i,S,T;
bool vis[nn];
double dis[nn],ans;

double D(int a,int b)
 {return sqrt(0.+(x[a]-x[b])*(x[a]-x[b])+(y[a]-y[b])*(y[a]-y[b]));}

void add(int a,int b,int c,double d)
 {g[t].end=b;g[t].next=fa[a];g[t].c=c;g[t].s=d;fa[a]=t++;
  g[t].end=a;g[t].next=fa[b];g[t].c=0;g[t].s=-d;fa[b]=t++;
 }

bool CM()
 {int u,i,t=0,w=1,x;
  for (i=S;i<=T;i++)dis[i]=1e20,vis[i]=0,pre[i]=-1;
  q[1]=S;dis[S]=0;vis[S]=1;
  for (;t<w;){
  x=q[++t];vis[x]=0;
  for (i=fa[x];i!=-1;i=g[i].next)if (g[i].c)
           {u=g[i].end;
      if (dis[u]>dis[x]+g[i].s+1e-8){
      dis[u]=dis[x]+g[i].s;pre[u]=i;
      if (!vis[u])vis[u]=1,q[++w]=u;
           }}}
  if (dis[T]>1e15)return 0;
  for (i=T;i!=S;i=g[i].end)
    i=pre[i],g[i].c--,g[i^=1].c++;
  tem++;ans+=dis[T];return 1;
 }
int main()
{
 scanf("%d",&n);
 for (i=1;i<=n;i++)scanf("%d%d",&x[i],&y[i]);
 S=0;T=2*n+1;memset(fa,-1,sizeof(fa));
 for (i=1;i<=n;i++){
 add(S,i,1,0);
 add(i+n,T,2,0);
 }
 for (i=1;i<=n;i++)for (j=1;j<=n;j++)
 if (y[i]>y[j])add(j,i+n,1,D(i,j));ans=0;
 while (CM());
 if (tem!=n-1)printf("-1\n");else printf("%.8lf\n",ans);
 return 0;  
}
