#include<cstdio>
#include<cmath>
#include<cstring>
#define N 333333
int x[N],y[N],ed[N],next[N],data[N],son[N],pre[N],f[N*10],n,S,T,l=1,cnt=1;
double d[N],cost[N],ans; bool b[N];
void add(int x,int y,int z,double c)
{ed[++l]=y,data[l]=z,cost[l]=c,next[l]=son[x],son[x]=l;}
#define Add(x,y,z,c) add(x,y,z,c),add(y,x,0,-c)
bool spfa()
{
    for (int i=S;i<=T;i++) d[i]=1e20,b[i]=0;
    int h=0,t=1; f[1]=S,d[S]=0;
    while (h<t){
        int i=f[++h]; b[i]=0;
        for (int p=son[i];p;p=next[p]) if (data[p]){
            int j=ed[p];
            if (d[i]+cost[p]+1e-8<d[j]){
                d[j]=d[i]+cost[p],pre[j]=p;
                if (!b[j]) b[j]=1,f[++t]=j;
                }
            }
        }
    if (d[T]>1e15) return 0;
    for (int p=T;p!=S;p=ed[p]) p=pre[p],data[p]--,data[p^=1]++;
    cnt++,ans+=d[T]; return 1;
}
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("check.txt", "w", stdout);
    scanf("%d",&n),T=n+n+1;
    for (int i=1;i<=n;i++)
        scanf("%d%d",x+i,y+i),Add(S,i,1,0),Add(i+n,T,2,0);
    for (int i=1;i<=n;i++) for (int j=1;j<=n;j++) if (y[i]<y[j])
        Add(i,j+n,1,sqrt(0.+(x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j])));
    while (spfa()){} if (cnt<n) puts("-1"); else printf("%.9lf\n",ans); return 0;
}
