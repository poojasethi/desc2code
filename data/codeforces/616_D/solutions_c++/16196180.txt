#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
#include<map>
using namespace std;
int a[500005];
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)
	scanf("%d",&a[i]);
	map<int,int> cnt;
	int ub=1,lb=0,s=0,e=0,sum=0;
	for(int lb=1;lb<=n;lb++)
	{
		cnt[a[lb]]++;
		if(cnt[a[lb]]==1)
		sum++;
		while(sum>k)
		{
			cnt[a[ub]]--;
			if(cnt[a[ub]]==0)
			sum--;
			ub++;
		}
		if(lb-ub>=e-s)
		{
			e=lb;
			s=ub;
		}
	}
	printf("%d %d\n",s,e);
	return 0;
}
