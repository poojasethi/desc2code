#include <cstdio>

const int M = 1000005;

int a[M], occ[M] = { 0 };
int k, n, l = 1, r = 1;
int countK = 0;

void addNumber(int x) {
  if (occ[x]++ == 0) ++countK;
}

void removeNumber(int x) {
  if (--occ[x] == 0) --countK;
}

int main( void ) {
  scanf("%d %d", &n, &k);
  int pointer = 1;

  for (int i = 1; i <= n; ++i) {
    scanf("%d", &a[i]);

    addNumber(a[i]);
    while(countK > k)
      removeNumber(a[pointer++]);

    if (i-pointer > r-l) {
      r = i; l = pointer;
    }
  }

  printf("%d %d\n", l, r);
  return 0;
}
