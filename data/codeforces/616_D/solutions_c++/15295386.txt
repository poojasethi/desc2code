#include <bits/stdc++.h>

using namespace std;

int d[1000007],a[1000007],n,k,res,rl,rr,slk;

int main()
{
    cin >> n >> k;
    for (int i=1; i<=n; ++i)
        scanf("%d",&a[i]);
    int j=1;
    rr=1; rl=1;
    d[a[1]]++; slk=1; res=1;
    for (int i=2; i<=n; ++i)
    {
        d[a[i]]++; if (d[a[i]]==1) slk++;
        while (slk>k) {d[a[j]]--; if (d[a[j]]==0) slk--; j++;}
        if (res<i-j+1)
        {
            res=i-j+1;
            rl=i;
            rr=j;
        }
    }
    cout << rr << " " << rl;

}
