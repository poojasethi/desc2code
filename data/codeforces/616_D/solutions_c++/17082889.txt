#include <stdio.h>
#include <string.h>

int n, k;
int lis[500010], ct[1000010];

int main()
{
	memset(ct, 0, sizeof ct);
	int i;
	scanf("%d%d", &n, &k);
	for(i = 0; i < n; i++)
		scanf("%d", &lis[i]);
	int l = 0, r = 0, x = 0, y = 1, cur = 1;
	ct[lis[0]] = 1;
	while(y < n)
	{
		ct[lis[y]]++;
		if(ct[lis[y]] == 1)
			cur++;
		if(cur > k)
		{
			ct[lis[x++]]--;
			while(ct[lis[x - 1]] != 0)
				ct[lis[x++]]--;
			cur--;
		}
		if(y - x > r - l)
			l = x, r = y;
		y++;
	}
	printf("%d %d\n", l + 1, r + 1);
	return 0;
}
