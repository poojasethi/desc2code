#include <bits/stdc++.h>
using namespace std;
const int maxn = 5e5+10;
int n,a[maxn],k,cnt[2*maxn],j,ans,l,r,now;
int main(){
    scanf("%d%d",&n,&k);
    for (int i=1; i<=n; i++) scanf("%d",&a[i]);
    j=0;a[n+1]=-1;
    for (int i=1; i<=n; i++){
        while (now<=k&&j<=n){
            j++;
            if (a[j]>=0) {cnt[a[j]]++;if (cnt[a[j]]==1) now++;}
        }
        if (ans<j-i){
            ans=j-i;
            l=i;r=j-1;
        }
        cnt[a[i]]--;
        if (cnt[a[i]]==0) now--;
    }
    cout<<l<<" "<<r;
}
