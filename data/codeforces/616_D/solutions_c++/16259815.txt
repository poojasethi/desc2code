#include<stdio.h>
#include<string.h>
const int big=600000,verybig=1100000;
int  a[big],b[verybig]; 
int main() 
{
	int n,k,i,l=1,r=0,al=1,ar=0,t=0;
	scanf("%d%d",&n,&k);
	
	for(i=1;i<=n;i++)	
		scanf("%d",&a[i]);
	memset(b,0,sizeof(b));
	
	for(i=1;i<=n;i++)	
	{
		r++;
		if(b[a[r]]==0)	t++;
		b[a[r]]++;
		while(t>k)
		{
			b[a[l]]--;
			if(b[a[l]]==0)	t--;
			l++;
		}
		if(r-l>ar-al)	
		{
			al=l;
			ar=r;
		}
	}
	printf("%d %d",al,ar);
	return 0;	
}
