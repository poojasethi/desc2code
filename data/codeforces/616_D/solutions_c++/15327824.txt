#include<iostream>
using namespace std;
int b[1000006],a[1000006],ll,rr,l,i,n,k,cn,mx;
int main()
{
    cin>>n>>k;
    for(i=0;i<n;i++){
        cin>>a[i];
        if(!b[a[i]])cn++;
        b[a[i]]++;
        while(cn>k){
            b[a[l]]--;
            if(!b[a[l]])cn--;
            l++;
        }
        if(i-l+1>mx){
            mx=i-l+1;
            ll=l;
            rr=i;
        }
    }
    cout<<ll+1<<' '<<rr+1;
}
