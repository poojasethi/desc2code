#include <iostream>

using namespace std;

int n,k;
int ultimo[2000000];
int v[2000000];

int main()
{
  ios::sync_with_stdio(false);
  cin>>n>>k;
  int i1=1;
  int sol1,sol2,maxsol=0;
  int cuantos=0;
  for (int i2=1;i2<=n;i2++) {
    cin>>v[i2];
    if (ultimo[v[i2]]<i1) cuantos++;
    ultimo[v[i2]]=i2;
    while (cuantos>k) {
      if (ultimo[v[i1]]==i1) cuantos--;
      i1++;
    }
    if (i2-i1+1>maxsol) {
      maxsol=i2-i1+1;
      sol1=i1;
      sol2=i2;
    }
  }
  cout<<sol1<<" "<<sol2<<endl;
}
