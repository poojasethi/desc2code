#include<bits/stdc++.h>
using namespace std;
int t[1000006], a[500005], n, m = 0, l=0, r=0, k;
int main()
{
	ios::sync_with_stdio(false);
	cin >> n >> k;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	m = 1;
	t[a[0]]++;
	for (int i = 0, j = 0; j < n;)
	{
		if (m <= k)
		{
			if (r - l < j - i)
				l = i, r = j;
			j++;
			if (t[a[j]] == 0)
				m++;
			t[a[j]]++;
		}
		else
		{
			t[a[i]]--;
			if (t[a[i]] == 0)
				m--;
			i++;
		}
	}
	cout << l+1 << " " << r+1 << endl;
	return 0;
}
