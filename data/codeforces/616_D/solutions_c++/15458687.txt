#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#define maxn 500005
using namespace std;
int i,j,k,n,m,a,b,c,fre[2*maxn],l,r,temp,cur,d[maxn];
int main(){
	scanf("%d %d",&n,&k);
	i=j=1;
	for(i=1;i<=n;i++){
		scanf("%d",&d[i]);
		if(!fre[d[i]])
		   cur++;
		fre[d[i]]++;
		while(cur>k){
			fre[d[j]]--;
			if(!fre[d[j]])
			cur--;
			j++;
		}
		if(i-j+1>temp){
		   temp=i-j+1;
		   l=j;
		   r=i;
		}
	}
	printf("%d %d\n",l,r);
	return 0;
}
