#include <bits/stdc++.h>
using namespace std;
int main() {
	int r1,r2,i,j,k,l=0,n,m,cnt[1000006]={0},b[1000006],cn=0,ma=0;
	cin>>n>>k;
	for(i=0;i<n;i++) {
		scanf("%d",&b[i]);
		if(!cnt[b[i]])
			cn++;
		cnt[b[i]]++;
		while(cn > k) {
			cnt[b[l]]--;
			if(cnt[b[l]]==0)
				cn--;
			l++;
		}
		if(i-l+1 > ma) {
			ma = max(ma,i-l+1);
			r1 = l;
			r2 = i;
		}
	}
	cout<<r1+1<<" "<<r2+1<<endl;
	return 0;
}
