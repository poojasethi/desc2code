#include <bits/stdc++.h>

using namespace std;

int N, K, a[500005], f[1000005];

int main(){

	ios_base::sync_with_stdio(false);

	cin >> N >> K;

	for (int i=0; i<N; i++) cin >> a[i];

	int lo = 0, hi = 0, cnt = 0, maxl = 0, lox, hix;
	while (hi < N){
		if (f[a[hi]] == 0) cnt++;

		f[a[hi]]++;

		while (cnt > K){
			f[a[lo]]--;

			if (f[a[lo]] == 0) cnt--;

			lo++;
		}

		if (hi-lo+1 > maxl){
			maxl = hi - lo + 1;

			lox = lo + 1;
			hix = hi + 1;
		}

		hi++;
	}

	cout << lox << " " << hix << endl;

	return 0;
}
