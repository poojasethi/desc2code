#include<bits/stdc++.h>
using namespace std;
int ha[1000005],a[1000005];
set<int> s;
int main()
{
int n,k,i,j,i1,j1,maxi=0;
scanf("%d %d",&n,&k);
for(i=0;i<n;i++)
scanf("%d",&a[i]);
for(i=0,j=0;j<n;)
{
while(s.size()<=k&&j<n)
{
if(ha[a[j]]==0&&s.size()==k)
break;
if(ha[a[j]]==0&&s.size()<k)
{
s.insert(a[j]);
ha[a[j]]++;
}
else
ha[a[j]]++;
j++;
}
if(j-i+1>maxi)
{
i1=i;j1=j;
maxi=j-i+1;
}
if(ha[a[i]]==1)
{
ha[a[i]]=0;
s.erase(a[i]);
}
else
ha[a[i]]--;
i++;

}
printf("%d %d\n",i1+1,j1);
return 0;
}
