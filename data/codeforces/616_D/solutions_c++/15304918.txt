#include <bits/stdc++.h>
using namespace std;
const int N = 5e5+5;
int cc[(int)1e6+3],ar[N];
int main(){
	int cur=0,rl=1,rr=1,i,l=1,r=1,n,k,ret=1;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;++i)scanf("%d",&ar[i]);
	while(r<=n){
		if(cc[ar[r]]==0)++cur;
		++cc[ar[r]];
		while(cur>k){
			--cc[ar[l]];
			if(cc[ar[l]]==0)--cur;
			++l;
		}
		if(r-l+1>ret){
			ret=r-l+1;
			rr=r,rl=l;
		}
		++r;
	}
	cout<<rl<<" "<<rr<<endl;
}
