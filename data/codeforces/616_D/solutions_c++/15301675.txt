#include <bits/stdc++.h>
using namespace std;

#define ll long long 

ll n, k, a[1000001];
ll lo=1,hi=1,l=1, r=1, cnt=1, i,len,h[1000001];
int main() 
{
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin>>n>>k;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	h[a[1]]++;
	for(i=2;i<=n;i++)
	{
		r=i;
		if(h[a[i]]==0)
			cnt++;
		h[a[i]]++;
		while(cnt>k)
		{
			h[a[l]]--;
			if(h[a[l]]==0)
				cnt--;
			l++;
		}
		if(r-l+1>len)
		{
			len=r-l+1;
			hi=r;
			lo=l;
		}
	}
	cout<<lo<<" "<<hi<<endl;
	return 0;
}
