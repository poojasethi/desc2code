#include<bits/stdc++.h>
using namespace std;
int A[(int)5e5+5];
int Map[(int)1e6+5];
int main(){
int n,k;cin>>n>>k;
int cnt=0,Max=1;
for(int i=0;i<n;i++)
    cin>>A[i];
int lo=0,hi=0;
int st=1,en=1;
while(hi<n){
    if(!Map[A[hi]])cnt++;
    Map[A[hi]]++;
    hi++;
    while(cnt>k){
        Map[A[lo]]--;
        if(!Map[A[lo]])cnt--;
        lo++;
    }
    if(hi-lo>Max){
        Max=hi-lo;
        st=lo+1;
        en=hi;
    }
}cout<<st<<' '<<en<<endl;

return 0;}
