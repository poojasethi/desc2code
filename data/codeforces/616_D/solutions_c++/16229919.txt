#include<iostream>
#include<algorithm>
#include<cstdio>
#include<map>
using namespace std;
int a[500005];
int main()
{
	int n,k;
	while(~scanf("%d%d",&n,&k))
	{
		for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
		map<int,int> cnt;
		int ub=1,lb=1,sum=0,r=1,l=1;
		for(lb=1;lb<=n;lb++)
		{
			if(cnt[a[lb]]++==0)
			sum++;
			while(sum>k)
			{
				if(--cnt[a[ub++]]==0)
				sum--;
			}
			if(lb-ub>r-l)
			l=ub,r=lb;
		}
		printf("%d %d\n",l,r);
	}
}
