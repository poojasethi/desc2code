#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int n,m,i,j,k,ans,l,y,z,t;
int L=1,R=1,best=0,r,x=1;
int a[500001];
int b[1000001];
int main()
{
std::ios_base::sync_with_stdio(false);
cin>>n>>k;
for (i=1;i<=n;i++) cin>>a[i];
for (i=1;i<=n;i++)
{
if (b[a[i]]==0) r++;
++b[a[i]];
if (i-x+1>best&&r<=k) {L=x;R=i;best=i-x+1;}
if (r>k)
{
for (j=x;j<=n;j++)
{
    --b[a[j]];
    if (b[a[j]]==0)
    {--r;x=j+1;break;}
x=j;
}
}
}
cout<<L<<" "<<R;
return 0;
}
