#include <bits/stdc++.h>
using namespace std;
int num[1000010],a[1000010],n,k;
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	int good=0,j=0;
	int l=0,r=0;
	int ans=0;
	for(int i=0;i<n;i++)
	{
		num[a[i]]++;
		if(num[a[i]]==1) good++;
		while(good>k)
		{
			num[a[j]]--;
			if(num[a[j]]==0) good--;
			j++;
		}
		if(i-j+1>ans)
		{
			ans=i-j+1;
			l=j+1;
			r=i+1;
		}
	}
	printf("%d %d\n",l,r);
	return 0;
}
