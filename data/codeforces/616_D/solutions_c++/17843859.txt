#include<cstdio>

int n, k;
int p, tot, x, l, r;
int a[1111111], s[1111111];

int main() {
	scanf("%d%d", &n, &k);
	p=tot=0;
	l=0, r=-1;
	for(int i=0;i<n;i++) {
		scanf("%d", &x);
		a[i]=x;
		
		s[x]++;
		if(s[x]==1) tot++;
		while(tot>k) {
			x=a[p++];
			s[x]--;
			if(s[x]==0) tot--;
		}
		if(i-p>r-l) {
			l=p;r=i;
		}
	}
	printf("%d %d\n", l+1, r+1);
	
	return 0;
}
