#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=500005;
int vis[2*N];
int a[N];

int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
	}
	int tl,tr,l,r;
	l=r=tl=tr=1;
	int Max=0,cnt=0;
	for(int i=1;i<=n;i++)
	{
		vis[a[i]]++;
		if(vis[a[i]]==1) cnt++;
		while(cnt>k){
			vis[a[tl]]--;
			if(vis[a[tl]]==0)
				cnt--;
			tl++;
		}
		if(i-tl+1>Max)
			Max=i-tl+1,r=i,l=tl;
	}
	printf("%d %d\n",l,r);
	return 0;
}
