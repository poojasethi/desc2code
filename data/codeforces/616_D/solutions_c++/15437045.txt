#include <bits/stdc++.h>
using namespace std;
const int maxn = 1000100;

int num[maxn];
int a[maxn];

int main()
{
	int n, k, l, r, len, type, L, R;
	while(cin >> n >> k)
	{
		type = 0;
		for(int i = 1; i <= n; i++)
			scanf("%d", a+i);
		memset(num, 0, sizeof num);

		l = 1; len = 0;
		for(int i = 1; i <= n; i++) {
			if(num[a[i]] == 0) type++;
			num[a[i]]++;
			while(type > k) {
				num[a[l]]--;
				if(num[a[l]] == 0) type--;
				l++;
			}
			if(i-l+1 > len) {
				len = i-l+1;
				L = l; R = i;
			}
		}
		printf("%d %d\n", L, R);
	}
	return 0;
}
