#include <stdio.h>
#include <stdlib.h>
#include <string.h>


int main(){
	int qtd[1000010], arr[1000010], a, k, kt, n, i, j, ri, rj, arrI, max;

	scanf("%d %d", &a, &k);

	memset(qtd, 0, sizeof(qtd));

	for(i = j = kt = arrI = max = 0; j < a; ++j){
		scanf("%d", &n);
		arr[arrI++] = n;
		if(qtd[n] == 0) ++kt;
		++qtd[n];

		if(kt <= k){
			if(max < (j - i)){
				max = j - i;
				ri = i;
				rj = j;
			}
		} else {
			for(; kt > k; ++i){
				if(--qtd[arr[i]] == 0) --kt;
			}
		}
	}

	printf("%d %d\n", ri + 1, rj + 1);

	return 0;
}
