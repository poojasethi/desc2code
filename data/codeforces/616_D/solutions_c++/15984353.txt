#include<bits/stdc++.h>
using namespace std;

#define maxn 1000005
int a[maxn];
int vis[maxn];
int main()
{
    int n,k;scanf("%d%d",&n,&k);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    int al=0,ar=0,ans=0,now=0,l=1;
    for(int i=1;i<=n;i++)
    {
        vis[a[i]]++;
        if(vis[a[i]]==1)now++;
        while(now>k)
        {
            vis[a[l]]--;
            if(vis[a[l]]==0)
                now--;
            l++;
        }
        if(i-l+1>=ar-al+1)
            ar=i,al=l;
    }
    cout<<al<<" "<<ar<<endl;
}
