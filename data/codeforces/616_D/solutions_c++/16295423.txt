#include<bits/stdc++.h>
using namespace std;
int a[500005],l[1000005];
int main()
{
int n,k;
scanf("%d%d",&n,&k);
int i,j;
for(i=1;i<=n;i++)
scanf("%d",&a[i]);

int lo,hi,si;
lo = hi=0;
si =0;
int di = 0;
i=1;
for(j=1;j<=n;j++)
{
l[a[j]]++;
if(l[a[j]]==1)di++;
if(di>k)
{
while(i<j && di>k)
{
l[a[i]]--;
if(l[a[i]]==0)di--;
i++;
}
}
int cur = j-i+1;
if(cur > si)
{
si = cur;
lo = i;
hi = j;
}
}
printf("%d %d\n",lo,hi);
return 0;
}
