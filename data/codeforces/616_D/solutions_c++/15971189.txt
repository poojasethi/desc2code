#include<cstdio>
int a[500000+5];
int visit[1000000+5];
int main(void){
	int n,k;
	scanf("%d%d",&n,&k);
	int l=1,r=1,sum=0,len=1,x=1,y=1;
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	
	while(r<=n){
		
		if(visit[a[r]]==0){
			visit[a[r]]++;
			sum++;
		}
		else if(visit[a[r]]!=0){
			visit[a[r]]++;
		} 
	
		while(sum>k){
			visit[a[l]]--;
			if(visit[a[l]]==0) sum--;
			l++;
		}
		if(r-l+1>len)
		{
			len=r-l+1;
			x=l;
			y=r;
		}
	   	r++;
	}
	printf("%d %d",x,y);
	return 0;
}
