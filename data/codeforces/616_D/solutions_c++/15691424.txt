#include <bits/stdc++.h>
using namespace std;
const int maxn=1000005;
int a[maxn],cnt[maxn];
int n,k;
int main() {
    scanf("%d %d",&n,&k);
    for(int i=0;i<n;i++) scanf("%d",a+i);
    int ans=0,l,r;
    int s=0,t=0;
    int tot=0;
    while(t<n) {
        while(t<n&&tot+(!cnt[a[t]])<=k) {
            tot+=!cnt[a[t]];
            cnt[a[t]]++;
            t++;
        }
        if(ans<t-s) {ans=t-s;l=s,r=t-1;}
        cnt[a[s]]--;
        if(!cnt[a[s]]) tot--;
        s++;
    }
    printf("%d %d\n",l+1,r+1);
    return 0;
}
