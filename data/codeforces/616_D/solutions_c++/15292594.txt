#include <bits/stdc++.h>

using namespace std;

int vet[500010];
int l, r;
int cont ;
int qt[1000100];

int main() {
	cont = 0;
	int n, k;
	scanf("%d %d", &n, &k);
	for (int i = 0; i < n; i++) scanf("%d", vet + i);
	l = 0; r = 0;	
	memset(qt,0,sizeof qt);
	int ult = 0;	
	for (int i = 0; i < n; i++) {
		qt[vet[i]]++;
		if(qt[vet[i]] == 1) cont++;
		while(cont > k) {
			qt[vet[ult]]--;
			if(!qt[vet[ult]])cont--;
			ult++;
		}
		if(r - l < i - ult)r = i,l = ult;
	}
	cout << l + 1 << " " << r + 1 << endl;
	return 0;
}
