#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int a[1111111] , c[1111111]={0};
int n , k ;
int l , r ;
int best=-1 , f ;
int L , R ;

int main()
{
scanf("%d %d",&n,&k);
for(int i=1;i<=n;i++)
    scanf("%d",&a[i]);

l=1;r=1;
f=0;
while( r<=n ){

  if( f<k || ( f<=k && c[a[r]]!=0 )  ){

  if( c[ a[r] ]==0 ){ f++; }
  c[a[r]]++;

   if( r-l+1 > best ){ L=l;R=r; best=r-l+1; }

    r++;


  }else {
  c[ a[l] ]--;

  if( c[ a[l] ]==0 )f--;

  l++;

  }
}

cout<<L<<" "<<R<<endl;
    return 0;
}
