#include <bits/stdc++.h>
using namespace std;
int n,k,p,q,l,r,mx=-1,cnt,vis[1000000],arr[1000000];
int main(){
	scanf("%i%i",&n,&k);
	for(int i=0;i<n;i++)scanf("%i",&arr[i]);
	for(p=0;p<n;p++){
		if(!vis[arr[p]])cnt++;
		vis[arr[p]]++;
		if(cnt<=k){
			if(p-q>mx){
				mx=p-q;
				l=q;
				r=p;
			}
		}
		while(cnt>k&&q<p){
			vis[arr[q]]--;
			if(!vis[arr[q]])cnt--;
			q++;
		}
	}
	printf("%i %i\n",l+1,r+1);

}
