#include<stdio.h>
#define N 1000005
using namespace std;
int a[N],c[N],ansl,ansr,n,m,i,l,r,number;
int main(){
    scanf("%d%d",&n,&m);
    for (i=1;i<=n;i++)
        scanf("%d",&a[i]);
    ansl=1;ansr=1;
    c[a[l=1]]++;number=1;
    for (r=2;r<=n;r++){
        number+=((++c[a[r]])==1);
        while (number>m)
            number-=((--c[a[l++]])==0);
        if (r-l>ansr-ansl)
            ansl=l,ansr=r;
  }printf("%d %d\n",ansl,ansr);
}
