#include<bits/stdc++.h>

using namespace std;

const int Maxn=5e5+7;

int a[Maxn];
int ted[Maxn*10];

int main() 
{
  ios_base::sync_with_stdio(0);
  int n,k,len=0,L=0,R=0;
  cin>>n>>k;
  for(int i=0;i<n;i++)
    cin>>a[i];
  for(int i=0,j=-1;i<n;i++)
    {
      for(;(ted[a[j+1]] or k) and j<n-1;)
	ted[a[++j]]++,k-=(ted[a[j]]==1);
      if(len<j-i)
	len=j-i,L=i,R=j;
      ted[a[i]]--;
      k+=(ted[a[i]]==0);
    }
  cout<<L+1<<" "<<R+1<<endl;
  return 0;
}
