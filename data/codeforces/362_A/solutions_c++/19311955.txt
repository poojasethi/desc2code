#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
using namespace std;
char map[10][10];
int a[10],b[10];
main()
{
	int t,i,j,k,s,q,w;
	scanf("%d",&t);
	while(t--)
	{
		k=0;s=0;
		for(i=0;i<8;i++)
		scanf("%s",map[i]);
		for(i=0;i<8;i++)
		{
			for(j=0;j<8;j++)
			{
				if(map[i][j]=='K')
				{
					a[k++]=i;
					b[s++]=j;
				}
		    }
	    }
	    q=abs(a[1]-a[0]);
	    w=abs(b[1]-b[0]);
	    if((q==4||q==0)&&(w==4||w==0))
	    printf("YES\n");
	    else
	    printf("NO\n");
}
}
