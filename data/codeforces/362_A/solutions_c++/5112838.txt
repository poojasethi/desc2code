#include <cstdlib>
#include <cstdio>
#include <iostream>
using namespace std;

int t,n,x[2],y[2];
char c;

int main()
{
	cin>>t;
	while (t--)
	{
		n=0;
		for (int i=1;i<=8;i++) for (int j=1;j<=8;j++)
		{
			cin>>c;
			if (c=='K') x[n]=i,y[n++]=j;
		}
		if ((x[0]-x[1])%2) goto NO;
		if ((y[0]-y[1])%2) goto NO;
		if ((x[0]-x[1])/2%2) goto NO;
		if ((y[0]-y[1])/2%2) goto NO;
		//if (((x[0]-x[1])/2%2)!=((y[0]-y[1])/2%2)) goto NO;
		cout<<"YES\n";continue;
		NO:cout<<"NO\n";
	}
	return 0;
}
