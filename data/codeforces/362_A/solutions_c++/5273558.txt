#include <iostream>
 
using namespace std;
 
int main()
{
char x[8][8];
int a,b,p,l,t,count1=0;
cin>>t;
for(int hh=0;hh<t;hh++){
count1=0;
for(int i=0;i<8;i++){
for(int j=0;j<8;j++){
cin>>x[i][j];
if((x[i][j]=='K'||x[i][j]=='k')&&count1==1){
a=i;
b=j;
}
if((x[i][j]=='K'||x[i][j]=='k')&&count1==0) {
p=i;
l=j;
count1=1;
}
}
 
}
 
 
if((a-p)%4==0&&(b-l)%4==0){
cout<<"YES"<<endl;
}
else {cout<<"NO"<<endl;}
}
 
 
return 0;
}
