#include <cstdio>
#include <cmath>

char cb[15][15];

int fuck(int x) { return x > 0 ? x : -x; }

int main() {
	int t;
	scanf("%d", &t);
	while (t--) {
		int sz = 0, nightx[2], nighty[2];
		for (int i = 0; i < 8; ++i) scanf("%s", cb[i]);
		for (int i = 0; i < 8; ++i)
			for (int j = 0; j < 8; ++j)
				if (cb[i][j] == 'K') { nightx[sz] = i; nighty[sz++] = j; }
		if (fuck(nightx[0] - nightx[1]) % 4 == 0 && fuck(nighty[0] - nighty[1]) % 4 == 0) puts("YES");
		else puts("NO");
	}
	return 0;
}
