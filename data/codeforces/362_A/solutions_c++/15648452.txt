#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main()
{
	int t,i,j,num=1;
	int x1,y1,x2,y2;
	char a[9][9];
	scanf("%d",&t);
	
	while(t--)
	{
		for(i=0;i<8;i++)
		{
			scanf("%s",a[i]);
			for(j=0;j<8;j++)
			{
				if(a[i][j]=='K')
				{
					if(num==1)
					{
						x1=i;y1=j;num++;
					}
					else
					{
						x2=i;y2=j;num=1;
					}
				}
			}
		}
		if((int)fabs(x1-x2)%4==0 && (int)fabs(y1-y2)%4==0) printf("YES");
		else printf("NO");
		if(t!=0) printf("\n"); 
	 } 
	 
	return 0;
}
