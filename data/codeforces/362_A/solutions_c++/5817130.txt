#include <iostream>
#include<cstring>
using namespace std;
char tb[16][16];int x[2],y[2];
int main() {
	int n;
	cin>>n;int fl=0;
	while(n>0){
		n--;fl=0;
		for(int i=0;i<8;i++){
			for(int j=0;j<8;j++){
				cin>>tb[i][j];
				if(tb[i][j]=='K'){
					x[fl]=i;
					y[fl]=j;fl++;
				}
			}
		}
		if((x[1]-x[0])%4==0&&(y[1]-y[0])%4==0) cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
	return 0;
}
