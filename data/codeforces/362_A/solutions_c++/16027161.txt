#include<cstdio>

int main()
{
	int T;
	char map[10];
	int x[2],y[2],t;
	scanf("%d",&T);
	while(T--)
	{
		t=0;
		for(int i=1;i<=8;i++)
		{
			scanf("%s",map+1);
			for(int j=1;j<=8;j++)
			{
				if(map[j]=='K')
				{
					x[t]=i;
					y[t]=j;
					t++;
				}
			}
		}
		if((x[0]-x[1])%4==0&&(y[0]-y[1])%4==0) printf("YES\n");
		else printf("NO\n");
	}
	return 0;
}
