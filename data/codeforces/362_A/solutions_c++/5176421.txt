#include<iostream>
using namespace std;
int main()
{
	int n;
	char a[9][9];
	while(cin>>n)
	{
		while(n--)
		{
			int s1=-1,s2=-1,q1,q2;
			for(int i=1;i<9;i++)
				for(int j=1;j<9;j++)
				{
					cin>>a[i][j];
					if(a[i][j]=='K'&&(s1==-1)&&(s2==-1))
					{
						s1=i;
						s2=j;
					}
					else if(a[i][j]=='K')
					{
						q1=i;
						q2=j;
					}
				}
				if((q1-s1)%4==0 && (q2-s2)%4==0)
				{
					cout<<"YES"<<endl;
				}
				else
				{
					cout<<"NO"<<endl;
				}
		}
		return 0;
	}
}
