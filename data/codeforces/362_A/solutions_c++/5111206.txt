#include<stdio.h>
int main()
{
	int t,i,j,l;
	int k[2][2];
	char s[8][8];
	scanf("%d",&t);
	while(t--)
	{
		l=0;
		for(i=0;i<8;i++)
			scanf("%s",&s[i]);
		for(i=0;i<8;i++)
			for(j=0;j<8;j++)
			{
				if(s[i][j]=='K')
				{
					k[l][0]=i+1;
					k[l][1]=j+1;
					l++;
				}
			}
		if((k[1][0]-k[0][0])%4==0 && (k[1][1]-k[0][1])%4==0)
			printf("YES\n");
		else
			printf("NO\n");
	}
}
