#include<stdio.h>
#include<stdlib.h>
#include<math.h>
int main()
{
	int t,i,j,m,n,a,b,x;char c[8][8];
	scanf("%d\n",&t);
	while(t)
	{	x=0;
	for(i=0;i<8;i++)
	{
	for(j=0;j<8;j++)
	{
	scanf("%c",&c[i][j]);
	if(c[i][j]=='K'&&x==0){m=i;n=j;x=1;}
	if(c[i][j]=='K'&&x==2){a=i;b=j;}
	if(x==1)x=2;	
	}
	getchar();
	}
    if(abs(m-a)%4==0&&abs(n-b)%4==0)printf("YES\n");
    else printf("NO\n");
	t--;
	if(t)getchar();
	}
return 0;
}
