#include<cstdio>
char map[10];
int main(){
	int t, x1, x2, y1, y2;
	scanf("%d", &t);
	while(t--){
		x1=-1;
		for(int i=0; i<8; i++){
			scanf("%s", map);
			for(int j=0; j<8; j++){
				if(map[j]=='K'){
					if(x1==-1){
						x1=i;
						y1=j;
					}
					else{
						x2=i;
						y2=j;
					}
				}
			}
		}
		if((x1-x2)%4==0 && (y1-y2)%4==0)	puts("YES");
		else	puts("NO");
	}
	return 0;
}
