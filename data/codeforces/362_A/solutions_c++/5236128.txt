#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
using namespace std;
int T,i,j,sx,sy,tx,ty;
char c[9];
int main(){
    for(scanf("%d",&T);T--;){
        sx=-1;
        for(i=1;i<=8;++i){
            scanf("%s",c+1);
            for(j=1;j<=8;++j)if(c[j]=='K'){
                if(sx==-1)sx=i,sy=j;
                else tx=i,ty=j;
            }
        }
        if((sx-tx)%4==0 && (sy-ty)%4==0)puts("YES");
        else puts("NO");
    }
}
