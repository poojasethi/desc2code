#include <bits/stdc++.h>
using namespace std;
int main(){
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    int n; cin >> n;
    for(int i = 0; i < n; i++){
        vector<int > x, y; string s[8]; for(int j= 0; j < 8; j++) cin >> s[j];
        for(size_t j = 0; j < 8; j++) for(size_t k = 0; k < 8; k++) if(s[j][k] == 'K') x.push_back(j), y.push_back(k); /*cout << j << " " << k << endl; */
        if(abs(x[0] - x[1]) % 4 == 0 && abs(y[0] - y[1]) % 4 == 0) cout << "YES" << endl; else cout << "NO" << endl;
    }
}
