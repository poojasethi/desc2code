#include<iostream>
using namespace std;
const int size=8;
int abs(int x){
  if(x<0) return -x;
  return x;
}
int main(){
 int n,i,j,k,posX[2],posY[2];
 cin>>n;
 while(n--){
  char str[10];
  k=0;
  for(i=0;i<size;++i){
   cin>>str;
   for(j=0;j<size;++j){
    if(str[j]=='K'){
     posX[k]=i,posY[k++]=j;
    }
   }
  }
  if( abs(posX[0]-posX[1])%4==0 && abs(posY[0]-posY[1])%4==0 ) cout<<"YES\n";
  else cout<<"NO\n";
 }

 return 0;
}
