#include <bits/stdc++.h>
#define MD(x) ((x)>=0)?(x):(-(x))

using namespace std;
int N;

int main ()
{
	scanf("%d",&N);
	char ch;
	for( int t=1 ; t<=N ; t++ )
	{
		int a1=0,a2=0,b1=0,b2=0;
		for( int i=1 ; i<=8 ; i++ )
			for( int j=1 ; j<=8 ; j++ )
			{
				scanf(" %c",&ch);
				if( ch=='K' ){
					if( a1 ){
						a2=i;
						b2=j;
					}
					else{
						a1=i;
						b1=j;
					}
				}
			}
			
		if( abs( (a1-a2) )%4==0 && abs( (b1-b2) )%4==0  )
			printf("YES\n");
			
		else printf("NO\n");
		
	}
	
}
