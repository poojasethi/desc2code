#include<bits/stdc++.h>

using namespace std;

int t,px,py,qx,qy;
char s[9][9];

int main(){
  scanf("%d",&t);
  while(t--){
    for(int i=0;i<8;i++) scanf("\n%s",s[i]);
    px=-1;
    for(int i=0;i<8;i++) for(int j=0;j<8;j++){
      if(s[i][j]=='K'){
        if(px==-1) px=i,py=j;
        else qx=i,qy=j;
      }
    }
    if((py-qy)%4==0 && (px-qx)%4==0) printf("YES\n");
    else printf("NO\n");
    scanf("\n");
  }
  return 0;
}
