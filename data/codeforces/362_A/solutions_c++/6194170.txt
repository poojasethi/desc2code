#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int x[10],y[10];
int main(){
	int t;scanf("%d",&t);
	for(int yo=1;yo<=t;yo++){
        int tot=0;
		for(int i=1;i<=8;i++)
			for(int j=1;j<=8;j++){char ch;
				for(;(ch=getchar())==' '||(ch==10););
				if(ch=='K')x[++tot]=i,y[tot]=j;
			}
		if(abs(x[1]-x[2])%4==0&&abs(y[1]-y[2])%4==0)printf("YES\n");
		else printf("NO\n");
	}
}
