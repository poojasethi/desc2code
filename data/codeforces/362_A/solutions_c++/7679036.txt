#include <iostream>
#include <stack>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <map>
#define ll __int64
using namespace std;
int x[3],y[3],cnt,i,j,len,t;
char s[11];
int main(){
	scanf("%d",&t);
	while(t--){
	cnt = 0;
	for(i = 1;i <= 8;i++){
		scanf("%s",s);
		for(j = 0;j < 8;j++)
			if(s[j] == 'K'){
				x[cnt] = i;
				y[cnt++] = j + 1; 
			}
	}
	sort(x,x + 2);
	sort(y,y + 2);
	if((x[1] - x[0]) % 4 == 0 && (y[1] - y[0]) % 4 == 0)
	puts("YES");
	else
	puts("NO");
	}
    return 0;
}
