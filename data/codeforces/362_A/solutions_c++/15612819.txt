#include <stdio.h>
#include <math.h>
 int main()
 {
     int n;
     scanf("%d",&n);
     while(n--)
    {
        int x1,x2,y1,y2,cur=0;
        int i,j;
        char s[10];
         for(i=0;i<8;i++){
             scanf("%s",s);
        for(j=0;j<8;j++){
           if(s[j]=='K')
            {
                if(cur) x2=i,y2=j;
                else x1=i,y1=j,cur=1;
            }
             }
         }
         printf(((x1-x2)%4==0&&(y1-y2)%4==0)?"YES\n":"NO\n");
     }
     return 0;
}
