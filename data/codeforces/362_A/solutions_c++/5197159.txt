#include <stdio.h>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int x1,x2,y1,y2,cur=0;
		char s[10];
		for(int i=0;i<8;i++){
			scanf("%s",s);
			for(int j=0;j<8;j++){
				if(s[j]=='K'){
					if(cur) x2=i,y2=j;
					else x1=i,y1=j,cur=1;
				}
			}
		}
		printf((abs(x1-x2)%4==0&&abs(y1-y2)%4==0)?"YES\n":"NO\n");
	}
	return 0;
}
