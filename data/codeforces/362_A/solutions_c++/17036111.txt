#include <cstdio>

int main()
{
	int T, x1, y1, x2, y2;
	scanf("%d", &T);
	char s[10];
	while(T--)
	{
		int k = 1;
		for(int i = 1; i <= 8; i++)
		{
			scanf("%s", s + 1);
			for(int j = 1; j <= 8; j++)
			{
				if(k == 1 && s[j] == 'K')
				{
					k++;
					x1 = i;
					y1 = j;
				}
				if(k == 2 && s[j] == 'K')
				{
					x2 = i;
					y2 = j;
				}
			}
		}
		if(((x2 -x1) % 4 == 0) && (y2 - y1) % 4 == 0)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
