#include <iostream>
using namespace std;
int t;

void init(){
	cin>>t;
	while (t--){
		int a,b;
		char ch;
		bool flag=false;
		for (int i=0;i<8;i++){
			for (int j=0;j<8;j++){
				cin>>ch;
				if (ch=='K'){
					if (!flag){
						a=i;
						b=j;
						flag=true;
					}
					else {
						if ((a-i) % 4 ==0 && (b-j) % 4 ==0)
							cout<<"YES"<<endl;
						else cout<<"NO"<<endl;
					}
				}
			}
		}
	}
}

int main(){
	init();
	return 0;
}
