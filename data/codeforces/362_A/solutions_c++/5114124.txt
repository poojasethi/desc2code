#include <iostream>

using namespace std;

#define abs(x) ((x) > 0 ? (x) : -(x))

char z;
int x,y,x2,y2;
int t;

int main()
{
	cin >> t;
	while(t -- )
	{
		x = y = -1;
		x2 = y2 = -1;
		for(int i = 0;i < 8;i ++ )
			for(int j = 0;j < 8;j ++ )
			{
				cin >> z;
				if(z == 'K')
					x == -1 ? x = i + 1,y = j + 1 : x2 = i + 1,y2 = j + 1;
			}
		cout << (abs(x - x2) % 4 || abs(y - y2) % 4 ? "NO" : "YES") << endl;
	}
	return 0;
}
