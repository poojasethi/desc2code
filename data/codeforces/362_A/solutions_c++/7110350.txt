// Two Semiknights Meet

#include <bits/stdc++.h>

using namespace std;

typedef long long int LL;

int main()
{
	int n,i,j,k,x1=0,x2,y1,y2;
	cin>>n;
	char arr[10][10];
	for(i=0;i<n;i++)
	{
		for(j=0;j<8;j++)
		{
			for(k=0;k<8;k++)
			{
				cin>>arr[j][k];
				if(arr[j][k]=='K')
				{
					if(x1==0)
					{
						x1=j+1;
						y1=k+1;
					}
					else
					{
						x2=j+1;
						y2=k+1;
					}
				}
			}
		}
		if((x1-x2)%4!=0 || (y1-y2)%4!=0)
		{
			cout<<"NO"<<endl;
		}
		else
		{
			cout<<"YES"<<endl;
		}
		x1=0;
	}
	return 0;
}
