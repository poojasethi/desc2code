#include <iostream>
using namespace std;
char board[10][10];
int main()
{
	int t;
	cin >> t;
	for(int T=0 ; T<t;++T)
	{
		pair<int,int> pos[2];
		int c=0;
		for(int i=1 ; i<=8 ;++i)
		{
			for(int j=1 ; j<=8 ; ++j)
			{
				cin >> board[i][j];
				if(board[i][j]=='K')
				{
					pos[c].first = i;
					pos[c++].second = j;
				}

			}
		}
		if(pos[0].first%4 != pos[1].first%4 || pos[0].second%4 != pos[1].second%4)
			cout << "NO\n";
		else
			cout << "YES\n";
	}
}
