#include <bits/stdc++.h>
using namespace std;
int t,x,y;
char arr[10][10];
bool cek;
int main(){
	cin>>t;
	for(int tc=0;tc<t;tc++){
		bool cek=false;
		for(int i=0;i<8;i++){
			for(int j=0;j<8;j++){
				cin>>arr[i][j];
				if(arr[i][j]=='K'){
					if(!cek){
						cek=true;
						x=i;
						y=j;
					}else
					if(cek){
						if((x-i)%4==0&&(y-j)%4==0) cout<<"YES"<<endl; else cout<<"NO"<<endl;
					}
				}
			}
		}
	}
}
