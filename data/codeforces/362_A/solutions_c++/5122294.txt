#include<iostream>
char A[9][9];
using namespace std;
int main()
{
	int a,x1,y1,x2,y2;
	cin>>a;
	while(a--)
	{
		x1=-1;
		for(int i=0;i<8;i++)
			for(int j=0;j<8;j++)
			{
				cin>>A[i][j];
				if(A[i][j]=='K' && x1==-1){x1=i;y1=j;}
				else
				{
					if(A[i][j]=='K'){x2=i;y2=j;}
				}
			}
		if((x2-x1)%4==0 && (y2-y1)%4==0)cout<<"YES"<<endl;
		else cout<<"NO"<<endl;
	}
  return 0;
}
