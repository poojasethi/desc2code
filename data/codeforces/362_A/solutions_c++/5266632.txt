#include <cstdio>
#include <iostream>

using namespace std;

char map[15][15];
int main()
{
	int n,i,j,x1,y1,x2,y2;
	scanf("%d",&n);
	
	while(n--)
	{
		x1=-1;
		for(i=0;i<8;i++)
		{
			scanf("%s", &map[i]);
			for(j=0; j<8; ++j)
			{
				if(map[i][j] == 'K')
				{
					if(x1==-1)
					{
						x1=i;y1=j;
					}
					else
					{
						x2=i;y2=j;
					}
				}	
			}
		}
		if((x1-x2)%4==0 && (y1-y2)%4==0) puts("YES");
		else puts("NO");
		
	}
//	system("pause");
	return 0;
}
