#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("in", "r", stdin);
#endif
	int t;
	while (cin >> t) {
		while (t--) {
			char c;
			vector<int> p;
			for (int i = 1; i <= 8; ++i)
				for (int j = 1; j <= 8; ++j) {
					cin >> c;
					if (c == 'K')
						p.push_back(i), p.push_back(j);
				}
			if ((p[2]-p[0])%4 == 0 && (p[3]-p[1])%4 == 0)
					printf("YES\n");
			else
				printf("NO\n");
		}
	}
	return 0;
}
