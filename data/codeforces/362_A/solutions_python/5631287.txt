
def test(r1, c1, r2, c2):
	if ((abs(r1 - r2) % 4 == 0) and (abs(c1 - c2) % 4 == 0)):
		return True
	else:
		return False

n = int(raw_input())
for k in xrange(0, n):

			if  (s[j] == 'K'):
				if (r1 == -1):

				else:
