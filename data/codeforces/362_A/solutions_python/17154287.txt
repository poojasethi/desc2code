def main():
	t = input()
	for k in xrange(t):
		x1, y1, x2, y2 = -1, -1, -1, -1
		for i in xrange(8):
			s = raw_input()
			for j in xrange(8):
				if 'K' == s[j]:
					if -1 == x1: x1, y1 = i, j
					else: x2, y2 = i, j
		print ("NO", "YES")[0 == (x1 - x2) % 4 and 0 == (y1 - y2) % 4]
		if k < t - 1: raw_input()

if __name__ == "__main__": main()
