n=input()

for i in range(n):
    paard=[]
    for j in range(8):
        s=raw_input()
        if s=='':
            s=raw_input()
        if 'K' in s:
            p=s.index('K')
            paard.append([j,p])
            if 'K' in s[p+1:]:
                paard.append([j,s[p+1:].index('K')+p+1])
    if abs(paard[0][0]-paard[1][0])%4==0 and abs(paard[0][1]-paard[1][1])%4==0:
        print 'YES'
    else:
        print 'NO'
