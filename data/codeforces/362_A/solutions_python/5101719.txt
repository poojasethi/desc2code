T = int(raw_input())
for t in xrange(T):
    a = [raw_input().strip() for _ in xrange(8)]
    b = []
    for i, r in enumerate(a):
        for j, c in enumerate(r):
            if c == 'K':
                b.append((i, j))
    c = [map(lambda x: abs(x) % 4, x) for x in b]
    if c[0] == c[1]:
        print "YES"
    else:
        print "NO"
    if t < T - 1:
        raw_input()
