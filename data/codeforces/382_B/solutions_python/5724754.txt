from math import ceil
a,b,w,x,c=map(int,raw_input().split())
d=w-x
l=0
h=10**20
while l!=h:
    m=(h+l)/2
    #print m,l,h
    t=b+(m*d)-(c-a)*x
    if h-l==1:
        if t<0:
            l=h
        else:
            h=l
        continue
    if t>=0:
        h=m
    else:
        l=m
print h+(c-a)
