def main():
    a, b, w, x, c = map(int, raw_input().split())
    if c <= a:
        print 0
        return
    lo, hi = 0, 10 ** 15
    while lo + 1 < hi:
        mid = (lo + hi) / 2
        d = b / w - (b - mid * x) / w
        if c - mid <= a - d:
            hi = mid
        else:
            lo = mid
    print hi
main()
