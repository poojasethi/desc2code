#!/usr/bin/env python

import sys
a, b, w, x, c = map(int, sys.stdin.readline().strip().split())
ans = 0

def iterate(count):
    global a, b, w, x, c, ans
    for i in range(count):
        if c <= a:
            return True
        ans += 1
        c -= 1
        if b >= x:
            b -= x
        else:
            a -= 1
            b = w - (x - b)
    return c <= a

iterate(10000)
bb = b
cycle_len = 0
aa = a
cc = c
while True:
    if iterate(1):
        break
    cycle_len += 1
    if b == bb:
        break
if c <= a:
    print ans
else:
    diff = (cc - c) - (aa - a)
    n = (c - a) / diff
    ans += cycle_len * n
    a -= (aa - a) * n
    c -= (cc - c) * n
    iterate(10000)
    print ans
