a, b, w, x, c = map(int, raw_input().split())
_a, _c = a, c
visited = set()
cycle = False
ans = 0
while c > a:
  visited.add(b)
  if b >= x:
    b -= x
  else:
    b = w - (x - b)
    a -= 1
  c -= 1
  ans += 1
  if not cycle and b in visited:
    la = _a - a
    lc = _c - c
    dca = c - a
    dl  = lc - la
    loops = (dca / dl) - 1
    a -= loops * la
    c -= loops * lc
    ans += loops * ans
    cycle = True
print ans
