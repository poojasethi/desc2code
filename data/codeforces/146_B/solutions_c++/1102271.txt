#include <iostream>
using namespace std;

int mask(int);

int main()
{
	int a, b;
	cin>>a>>b;
	for(int i = a+1; ; i++){
		if(mask(i) == b){
			cout<<i;
			return 0;
		}
	}
	return 0;
}

int mask(int n)
{
	int co=1;
	int res=0;
	while(n > 0){
		int c = n%10;
		n/=10;
		if(c == 7 || c== 4){
			res+=c*co;
			co*=10;
		}
	}
	return res;
}
