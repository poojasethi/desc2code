


#include <iostream>
using namespace std;
int mask(int x)
{
	int c,i=1,res=0;
	while(x!=0)
	{
		c=x%10;
		if(c==4 || c==7)
		{
			res=c*i+res;
			i=i*10;
		}
		x/=10;
	}
	return res;
}
int main(int argc, char **argv)
{
	int a,b,c;
	cin>>a>>b;
	for(c=a+1;;c++)
	if(mask(c)==b)
	{
		cout<<c;
		return 0;
	}
	return 0;
}
