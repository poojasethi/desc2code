#include<bits/stdc++.h>

using namespace std;
int i,n,m;

bool find(int a)
{
	int us=1,s=0;
	while(a)
	{
		if(a%10==7 or a%10 ==4)
		{
			s+=(a%10)*us;
			us*=10;
		}
		a/=10;
	}
	return s==m;
}
int main()
{
	cin>>n>>m;
	for(i=n+1; ; i++)
		if(find(i))
		{
			cout<<i;
			return 0;
		}


	return 0;
}
