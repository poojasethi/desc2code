#include<bits/stdc++.h>
using namespace std;
int mul(int x){
int sum=0,p=1;
while(x){
   int h=x%10;
    x=x/10;
if(h==4||h==7){
sum=sum+h*p;
p*=10;
}


}

return sum;
}
int main(){
int a,b;
//cout<<mul(23477794)<<endl;
cin>>a>>b;
a++;
while(mul(a)!=b)a++;
cout<<a<<endl;
return 0;
}
