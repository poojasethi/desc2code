#include <cstdio>

int gao(int a){
	int d, m, b;
	m = 0;
	b = 1;
	while(a){
		d = a % 10;
		a /= 10;
		if(d == 4 || d == 7){
			m += b * d;
			b *= 10;
		}
	}
	return m;
}

int main(){
	int a, b;
	scanf("%d%d", &a, &b);
	while(++a)
		if(gao(a) == b){
			printf("%d\n", a);
			break;
		}
	return 0;
}
