#include<stdio.h>

int check(int k){
	int i;
	int p;
	int j;
	int s;
	s=1;
	j=0;
	while(k>0){
		p=k%10;
		k=k/10;
		if(p==4||p==7){
			j=j+p*s;
			s=s*10;		
		}
	}
	return j;
}
int main(){
	int a,b;
	scanf("%d%d",&a,&b);
	a++;
	while(1){
		if(check(a)==b)break;
		a++;
		
	}
	printf("%d\n",a);
}
