#include <iostream>
#include <cmath>

using namespace std;

int mask(int x)
{
  int y=0;
  while (x>0) {
    if (x%10==4 or x%10==7)
      y=y*10+x%10;
    x/=10;
  }
  int z=0;
  while (y>0) {
    z=z*10+y%10;
    y/=10;
  }
  return z;
}

int main()
{
  int a,b;
  cin>>a>>b;
  a=max(a+1,b);
  while (mask(a)!=b) a++;
  cout<<a<<endl;
}
