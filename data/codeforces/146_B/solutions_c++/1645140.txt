#include <iostream>
#include <cstdio>
#include <time.h>
using namespace std;
int check(int n)
{
	int ans=0,u=1;
	while(n)
	{
		if(n%10==4||n%10==7)
			ans+=n%10*u,u*=10;
		n/=10;
	}
	return ans;
}
int main()
{
	int a,b,c;
	while(cin>>a>>b)
		for(c=a+1;;c++)
			if(check(c)==b)
			{
				cout<<c<<endl;
				break;
			}
	return 0;
}
