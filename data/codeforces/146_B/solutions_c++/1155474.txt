#include <iostream>
using namespace std;
int mask(int n) {
	int p=1, m=0;
	for (int i; n>0;) {
		i=n%10;
		n/=10;
		if (i==4 || i==7) {
			m+=p*i;
			p*=10;
		}
	}
	return m;
}

int main(int argc, char **argv)
{
	int a, b;
	cin >> a >> b;
	int c = a+1;
	while (mask(c)!=b)
		++c;
	cout << c;
	return 0;
}
