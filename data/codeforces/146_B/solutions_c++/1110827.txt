#include <iostream>

using namespace std;

int mask (int n )
{
	int res = 0;
	while (n > 0 ){
		int d = n % 10;
		if (d == 4 || d == 7 )
			res = 10*res + d;
		n /= 10;
	} // end while

	return res; 
}

int main()
{
	int a, b, c;
	cin >> a >> b;
	b = mask (b );
	for (c = a + 1; b != mask (c ); c++ );

	cout << c << endl;

	return 0;
}
