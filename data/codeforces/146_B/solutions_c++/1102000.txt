#include<cstdio>
int OK(int x){
//       Last Change:  2012-01-22 15:09:28
	int ret=0,id=1;
	while(x){
		int tmp=x%10;
		x/=10;
		if(tmp==4||tmp==7){
			ret+=tmp*id;
			id*=10;
		}
	}
	return ret;
}
int main(){
	int a,b;
	scanf("%d%d",&a,&b);
	for(++a;;++a)
		if(OK(a)==b)
			break;
	printf("%d\n",a);
	return 0;
}
