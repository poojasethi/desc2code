#include <iostream>
using namespace std;

int b;

int gao(int a)
{
	for(int j=a+1;j<7000000;j++)
	{
		int i=j;
		int tmp=0,tn=1;
		while(i)
		{
			int t=i%10;
			if(t==4||t==7)
			{
				tmp+=tn*t;
				tn*=10;
			}
			i/=10;
		}
		if(tmp==b)
			return j;
	}
	return -1;
}

int main()
{
	int a;
	cin>>a>>b;
	cout<<gao(a)<<endl;
	return 0;
}
