#include <iostream>
#include <cstdio>
using namespace std;
int solve(int a)
{
	int ret=0;
	for(;a>0;a/=10)
    {
		if(a%10==4||a%10==7)
			ret=ret*10+a%10;
	}
	return ret;
}
int main()
{
	int a,b;
	scanf("%d%d",&a,&b);
	b=solve(b),a++;
	while(solve(a)!=b)
        a++;
	printf("%d\n",a);
}
