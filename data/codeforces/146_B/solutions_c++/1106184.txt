#include<iostream>
#include<string>
using namespace std;
int mask(int x){
int s=0,l=1;
while(x!=0){
int g = x%10;
if((g == 4) || (g == 7) ){s+= g*l;l*=10;}
x/=10;
}
return s;
}
int main(){
int a,b;
cin >>a>>b;
//for(int v=1;v<100;v++){cout<<mask(v)<<endl;}
int j=a+1;
while(1){
if(mask(j)==b){cout<<j<<endl;break;}
j++;
}
return 0;
}
