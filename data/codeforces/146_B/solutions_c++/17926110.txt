#include <iostream>
using namespace std;

int chk(int x) {
	int v = 1, ret = 0;
	while (x) {
		int t = x % 10;
		if (t == 4 || t == 7) {
			ret += v*t; v *= 10;
		}
		x /= 10;
	}
	return ret;
}


int main(){
	int a, b;
	
	cin >> a >> b;
	for (a++; chk(a) != b; a++);
	cout << a;
//	system("pause");
	return 0;
}
