#include <iostream>
using namespace std;

int mask(int a)
{
	int ans=0,tmp=1;
	while(a)
	{
		if(a%10==4||a%10==7)
		{
			ans+=a%10*tmp;
			tmp*=10;
		}
		a/=10;
	}
	return ans;
}

int main()
{
	int a,b,c;
	cin >> a >> b;
	for(c=a+1;;c++)
	{
		if(b==mask(c))
		{
			cout << c << endl;
			break;
		}
	}
	return 0;
}
