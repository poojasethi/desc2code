import math

a, b = list(map(int, raw_input().split()))
b = str(b)
for i in range(a+1, 1000001):
    mask = ''
    for j in range(int(math.floor(math.log10(i)))+1):
        if (i % 10**(j+1))//(10**j) == 4 or  (i % 10**(j+1))//(10**j) == 7:
            mask = str((i % 10**(j+1))//(10**j) ) + mask
    if mask == b:
        print(i)
        exit(0)
