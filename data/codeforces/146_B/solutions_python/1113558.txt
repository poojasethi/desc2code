def mask(a):
    x = ""
    for char in str(a):
        if char in "47":
            x += char
    if x == "":
        return 0
    else:
        return int(x)

a,b = map(int,raw_input().split()) 
if b > a:
    print b

else:
    a += 1
    while mask(a) != b:
        a += 1
    print a
