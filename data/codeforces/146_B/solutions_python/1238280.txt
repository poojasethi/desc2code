def converta(a):
    a = str(a)
    lucky = '0'
    for each in a:
        if each == '4' or each == '7':
            lucky += each
    return int(lucky)

nums = raw_input().split()
a, b = int(nums[0]), int(nums[1])
while True:
    a += 1
    lucky = converta(a)
    if lucky == b:
        print a
        break
