#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
const int i=1<<19,n=200000;
int q,a[n],a1[n],a2[n],a3[n],s[n],s1[n],s2[n],s3[n];
int d[i],d1[i],f[n],f1[n];
int bp(int w){
  int c=-1,v;
  for(v=i>>2;v;v>>=1)
   if((c+v<q)&&(f[c+v]<=w))
    c+=v;
  return c;
}
void l(int q,int &r,int &r1){
  r=r1=0;
  int w=1,t;
  for(t=i>>2;t;t>>=1){
   w<<=1;
   if(q>=t){
    q-=t;
    if(r<d[w]){
     r=d[w];
     r1=d1[w];}
    w++;}}
}
void m(int e){
  e+=i>>1;
  d[e]=0;
  for(e>>=1;e;e>>=1)
   if(d[e<<1]<d[(e<<1)+1]){
    d[e]=d[(e<<1)+1];
    d1[e]=d1[(e<<1)+1];}
   else{
    d[e]=d[e<<1];
    d1[e]=d1[e<<1];}
}
void qs4(int q,int w){
  int e=q,r=w,t=f[q+(rand()%(w-q+1))],y;
  do{
  while(f[e]<t)e++;
  while(f[r]>t)r--;
  if(e<=r){
   y=f[e];f[e]=f[r];f[r]=y;
   y=f1[e];f1[e]=f1[r];f1[r]=y;
   e++;r--;}
  }while(e<=r);
  if(q<r)qs4(q,r);
  if(e<w)qs4(e,w);
}
void qs(int q,int w){
  int e=q,r=w,t=a1[a[q+(rand()%(w-q+1))]],y;
  do{
  while(a1[a[e]]<t)e++;
  while(a1[a[r]]>t)r--;
  if(e<=r){
   y=a[e];a[e]=a[r];a[r]=y;
   e++;r--;}
  }while(e<=r);
  if(q<r)qs(q,r);
  if(e<w)qs(e,w);
}
void qs1(int q,int w){
  int e=q,r=w,t=s1[q+(rand()%(w-q+1))],y;
  do{
  while(s1[e]<t)e++;
  while(s1[r]>t)r--;
  if(e<=r){
   y=s[e];s[e]=s[r];s[r]=y;
   y=s1[e];s1[e]=s1[r];s1[r]=y;
   y=s2[e];s2[e]=s2[r];s2[r]=y;
   y=s3[e];s3[e]=s3[r];s3[r]=y;
   e++;r--;}
  }while(e<=r);
  if(q<r)qs1(q,r);
  if(e<w)qs1(e,w);
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int w,e,r,t,t1,y1,y2,c,c1;
  long long y=0,v;
  scanf("%d%d",&q,&w);
  for(e=0;e<q;e++){
   scanf("%d%d",&a1[e],&a2[e]);
   f[e]=a2[e];
   f1[e]=a[e]=e;}
  qs4(0,q-1);
  for(e=i>>1;e<i;e++)
   d[e]=0;
  for(e=0;e<q;e++){
   a3[f1[e]]=e;
   d[e+(i>>1)]=a2[f1[e]]-a1[f1[e]];
   d1[e+(i>>1)]=f1[e];}
  for(e=(i>>1)-1;e;e--)
   if(d[e<<1]<d[(e<<1)+1]){
    d[e]=d[(e<<1)+1];
    d1[e]=d1[(e<<1)+1];}
   else{
    d[e]=d[e<<1];
    d1[e]=d1[e<<1];}
  qs(0,q-1);
  for(e=0;e<w;e++){
   scanf("%d%d%d",&s1[e],&s2[e],&s3[e]);
   s[e]=e;}
  qs1(0,w-1);
  e=r=c=c1=0;
  while(r<w)
   if((e<q)&&(a1[a[e]]<=s1[r])){
    if(c<a2[a[e]]){
     c=a2[a[e]];
     c1=a[e];}
    m(a3[a[e]]);
    e++;}
   else{
    v=min(c,s2[r])-s1[r];
    v*=s3[r];
    if(y<v){
     y=v;
     y1=c1;
     y2=s[r];}
    l(bp(s2[r])+1,t,t1);
    v=t;
    v*=s3[r];
    if(y<v){
     y=v;
     y1=t1;
     y2=s[r];}
    r++;}
  for(e=0;e<q;e++){
   r=(1<<30)-a1[e];
   a1[e]=(1<<30)-a2[e];
   a2[e]=r;}
  for(e=0;e<w;e++){
   r=(1<<30)-s1[e];
   s1[e]=(1<<30)-s2[e];
   s2[e]=r;}
  qs(0,q-1);
  qs1(0,w-1);
  e=r=c=c1=0;
  while(r<w)
   if((e<q)&&(a1[a[e]]<=s1[r])){
    if(c<a2[a[e]]){
     c=a2[a[e]];
     c1=a[e];}
    e++;}
   else{
    v=min(c,s2[r])-s1[r];
    v*=s3[r];
    if(y<v){
     y=v;
     y1=c1;
     y2=s[r];}
    r++;}
  printf("%I64d",y);
  if(y)
   printf("\n%d %d",y1+1,y2+1);
  return 0;}
