#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#define N 200001
using namespace std;
int n,m,xx,yy;
long long ans;
int a[N][2],b[N][3],c[N],d[N],f[N*8][2],p[N*2],help[2];
void init(){
    scanf("%d %d",&n,&m);
    for (int i=1;i<=n;i++)
        scanf("%d %d",&a[i][0],&a[i][1]),p[++p[0]]=a[i][0],p[++p[0]]=a[i][1],c[i]=i;
    for (int j=1;j<=m;j++)
        scanf("%d %d %d",&b[j][0],&b[j][1],&b[j][2]),d[j]=j;
}
bool cmp1(int x,int y){return a[x][0]<a[y][0];}
bool cmp2(int x,int y){return b[x][0]<b[y][0];}
bool cmp3(int x,int y){return a[x][1]>a[y][1];}
bool cmp4(int x,int y){return b[x][1]>b[y][1];}
void ins(int l,int r,int s,int ll,int v,int bz){
    if (f[s][0]<v)f[s][0]=v,f[s][1]=bz;
    if (l==r)return;
    static int ss;
    if (p[ss=(l+r)/2]>=ll)ins(l,ss,s+s,ll,v,bz);else ins(ss+1,r,s+s+1,ll,v,bz);
}
void find(int l,int r,int s,int ll,int rr){
    if (p[r]<ll||rr<p[l])return;
    if (help[0]>=f[s][0])return;
    if (ll<=p[l]&&p[r]<=rr){
        help[0]=f[s][0],help[1]=f[s][1];
        return;
    }
    find(l,(l+r)/2,s+s,ll,rr),find((l+r)/2+1,r,s+s+1,ll,rr);
}
void work(){
    static int x,y,z;
    static long long s;
    sort(p+1,p+p[0]+1);
    x=p[0],p[0]=1;
    for (int i=2;i<=x;i++)
        if (p[i]!=p[p[0]])p[++p[0]]=p[i];
    sort(c+1,c+n+1,cmp1),sort(d+1,d+m+1,cmp2);
    x=1,y=-1;
    for (int i=1;i<=m;i++){
        for (;a[c[x]][0]<=b[d[i]][0]&&x<=n;x++)
            if (y<a[c[x]][1])
                y=a[c[x]][1],z=c[x];
        if (ans<(s=(long long)(min(y,b[d[i]][1])-b[d[i]][0])*b[d[i]][2]))
            ans=s,xx=z,yy=d[i];
    }
    sort(c+1,c+n+1,cmp3),sort(d+1,d+m+1,cmp4);
    x=1,y=1000000001;
    for (int i=1;i<=m;i++){
        for (;a[c[x]][1]>=b[d[i]][1]&&x<=n;x++)
            if (y>a[c[x]][0])
                y=a[c[x]][0],z=c[x];
        if (ans<(s=(long long)(b[d[i]][1]-max(y,b[d[i]][0]))*b[d[i]][2]))
            ans=s,xx=z,yy=d[i];
    }
    x=n;
    for (int i=m;i;i--){
        for (;a[c[x]][1]<=b[d[i]][1]&&x;ins(1,p[0],1,a[c[x]][0],a[c[x]][1]-a[c[x]][0],c[x]),x--);
        help[0]=-1;
        find(1,p[0],1,b[d[i]][0],b[d[i]][1]);
        if (ans<(s=(long long)help[0]*b[d[i]][2]))
            ans=s,xx=help[1],yy=d[i];
    }
}
void write(){
    if (!ans)printf("0");
    else{
        printf("%I64d\n",ans);
        printf("%d %d",xx,yy);
    }
}
int main(){
    init();
    work();
    write();
    return 0;
}
