#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a;
	int n;
	cin>>n;
	vector<int> v;
	int c=0;
	for(int i=0;i<n;i++){
		int f=1;
		for(int j=0;j<n;j++)
			{cin>>a;if(a==1||a==3)f=0;}
		if(f){c++;v.push_back(i+1);}
		}

	cout<<c<<endl;
	for(int i=0;i<v.size();i++)cout<<v[i]<<" ";
	cout<<endl;
	return 0;
}
