#include <bits/stdc++.h>
using namespace std;
int main (){
	int i,j,n,f=0,kont=0,c=0,arr[10000]={0};
	int a;
	cin>>n;
	for(i=1;i<=n;i++){
		f=0;
		for(j=1;j<=n;j++){
			scanf("%d",&a);
			if(a==3||a==1)
				f++;
		}
		if(!f){
			kont++;
			arr[c++]=i;}
	}
	cout<<kont<<endl;
	for(i=0;i<c;i++)
		cout<<' '<<arr[i];
}
