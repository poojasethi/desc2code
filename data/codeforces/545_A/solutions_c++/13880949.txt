#include<iostream>
using namespace std;
int a[100],x[100][100],n,i,j,count;
int main()
{
cin>>n;
for(i=0;i<n;i++)
for(j=0;j<n;j++)
cin>>x[i][j];
for(i=0;i<n;i++)
for(j=0;j<n;j++)
if(x[i][j]==1)
a[i]=1;
else if(x[i][j]==2)
a[j]=1;
else if(x[i][j]==3)
a[i]=1,a[j]=1;
for(i=0;i<n;i++)
if(a[i]==0)
count++;
cout<<count<<"\n";
for(i=0;i<n;i++)
if(a[i]==0)
cout<<i+1<<" ";

}
