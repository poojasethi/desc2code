#include <bits/stdc++.h>
using namespace std;
#define ll long long

int n,ans,a[101],b,s;

int main()
{ cin>>n;
  for(int i=1;i<=n;i++) { s=0;
	for(int j=0;j<n;j++) {cin>>b; if (b==1) b=-1e6; if (b==3) b=-1e6; s+=b;}
	if (s>-2) a[ans++]=i;
  }	
  cout<<ans<<'\n';
  for(int i=0;i<ans;i++) cout<<a[i]<<' ';
}
