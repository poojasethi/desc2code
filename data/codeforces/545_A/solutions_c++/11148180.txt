#include <iostream>
#include <vector>
using namespace std;
int n,x;
vector<int>v;
bool q=1;
int main(){
  cin>>n;
  for(int i=0;i<n;i++,q=1){
    for(int j=0;j<n;j++){
      cin>>x;
      if(x==1 || x==3)
	q=0;
    }
    if(q)
      v.push_back(i+1);
  }
  cout<<v.size()<<endl;
  for(int i=0;i<v.size();i++)
    cout<<v[i]<<" ";
  return 0;
}
