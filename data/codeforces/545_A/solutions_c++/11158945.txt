#include<stdio.h>
int main()
{
	int n,i,j,a[110][110],s[110],k=0;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		int t=1;
		for(j=0;j<n;j++){
		    scanf("%d",&a[i][j]);
		    if(a[i][j]==0||a[i][j]==2)t++;
		}
		if(t==n){s[k]=i+1;k++;}
	}
	printf("%d\n",k);
	if(k!=0){
		for(i=0;i<k-1;i++)printf("%d ",s[i]);
	    printf("%d\n",s[i]);
	}
	return 0;
}
