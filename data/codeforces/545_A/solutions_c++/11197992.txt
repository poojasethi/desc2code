#include <bits/stdc++.h>
using namespace std;
int n,kon,k,st[105],top;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		kon=1;
		for(int j=1;j<=n;j++)
		{
			scanf("%d",&k);
			if(k==1||k==3)
			{
				kon=0;
			}
		}
		if(kon)
		{
			st[++top]=i;
		}
	}
	printf("%d\n",top);
	for(int i=1;i<=top;i++)
	{
		printf("%d ",st[i]);
	}
	return 0;
}
