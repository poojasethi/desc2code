#include<iostream>
using namespace std;
int main()
{
	int n,i,j;
	cin>>n;
	int cars[n][n];
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			cin>>cars[i][j];
	int ans[n],k=0;
	for(i=0;i<n;i++)
	{
		int f=1;
		for(j=0;j<n;j++)
		{
			if(cars[i][j]==1 || cars[i][j]==3)
				f=0;
		}
		if(f)
			ans[k++]=i+1;
	}
	cout<<k<<endl;
	for(int x=0;x<k;x++)
		cout<<ans[x]<<" ";
}
