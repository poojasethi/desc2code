#include<bits/stdc++.h>

using namespace std;

int b[100];

int main()
{
	int n,g=0,a,m=1;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			cin>>a;
			if(a==1 || a==3)
			{
				g=1;
			}
		}
		if(g==0)
		{
			b[m]=i;
			m++;
		}
		g=0;
	}
	cout<<m-1<<endl;
	for(int i=1;i<m;i++)
	{
		cout<<b[i]<<" ";
	}
}
