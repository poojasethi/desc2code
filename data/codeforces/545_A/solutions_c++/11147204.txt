#include<cstdio>
using namespace std;
int n,flag,k;
int st[105],top;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		flag=1;
		for (int j=1;j<=n;j++)
		{
			scanf("%d",&k);
			if (k==1||k==3)flag=0;
		}
		if (flag)st[++top]=i;
	}
	printf("%d\n",top);
	for (int i=1;i<=top;i++)printf("%d ",st[i]);
}
