#include <stdio.h>
int st[200],n;
int main(){
	int N;
	scanf("%d",&N);
	for(int i=1;i<=N;i++){
		int a,g=1;
		for(int j=1;j<=N;j++){
			scanf("%d",&a);
			if(a==1 || a==3)g=0;
		}
		if(g)st[++n]=i;
	}
	printf("%d\n",n);
	for(int i=1;i<=n;i++)printf("%d%c",st[i],i==n?'\n':32);
	return 0;
}
