#include <bits/stdc++.h>
int main()
{
	int i,j,n;
	scanf("%d",&n);
	int a[n][n];
	int b[n];
	int u=0;
	int c=0;
	for(i=0;i<n;i++)
	{
		int f=1;
		for(j=0;j<n;j++)
		{
			scanf("%d",&a[i][j]);
			if(a[i][j]==1||a[i][j]==3) f=0;
		}
		c+=f;
		if(f) b[u++]=i;
	}
	printf("%d\n",c);
	for(i=0;i<c;i++)
		printf("%d ",b[i]+1);
	if(c!=0) printf("\n");
	return 0;

}
