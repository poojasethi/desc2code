#include<iostream>

using namespace std;

int n,t,c;

int arr[105][105];
int a[105];

int main(){
  cin>>n;
  for(int i=0;i<n;i++){
    c=0;
    for(int j=0;j<n;j++){
      cin>>arr[i][j];
      if(arr[i][j]==1||arr[i][j]==3)
	c++;
    }
    if(c==0){
	a[t]=i+1;
	t++;
    }
  }
  cout<<t<<endl;
  for(int i=0;i<t;i++)
    cout<<a[i]<<" ";
  cout<<endl;
}
