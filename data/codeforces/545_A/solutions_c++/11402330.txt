#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
int n,i,j,m=0,k;
scanf("%d", &n);
int A[n][n];
int B[n];
for(i=0; i<n; i++)
{
for(j=0; j<n; j++)
{
cin>>A[i][j];
}
}
for(i=0; i<n; i++)
{
k=0;
for(j=0; j<n; j++)
{
if((A[i][j]==1)||(A[i][j]==3)) k=1;
}
if(k==0)
{
B[m]=i+1;
m++;
}
}
cout<<m<<endl;
for(i=0; i<m; i++) cout<<B[i]<<" ";
return 0;
}
