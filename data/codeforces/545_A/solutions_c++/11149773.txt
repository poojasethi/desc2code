#include <stdio.h>
using namespace std;
int main()
{
 int i,j,l,n,a[10000]={0},c=0;
  scanf("%d",&n);
  for(i=0;i<n;i++)
   for(j=0;j<n;j++)
    {
	 scanf("%d",&l);
	 if(l==1)a[i]=1;
	 if(l==2)a[j]=1;
	 if(l==3){a[i]=1,a[j]=1;}
	}
 for(i=0;i<n;i++)
  if(a[i]==0)c++;
  printf("%d\n",c);
 for(i=0;i<n;i++)
  if(a[i]==0)printf("%d ",i+1);
}
