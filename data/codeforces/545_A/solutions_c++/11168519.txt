#include <stdio.h>
#include <stdlib.h>

int main(){
	int i,j,k,n,a;

	scanf("%d", &n);

	int *s;
	s=(int*)calloc(n,sizeof(int));

	for(i=0,k=0;i<n;i++){
		for(j=0;j<n;j++){
			scanf("%d", &a);

			if(a==1 || a==3)
				s[i]++;
			else;
		}

		if(s[i]==0)
			k++;
	}

	printf("%d\n", k);

	for(i=0;i<n;i++)
		if(s[i]==0)
			printf("%d ", i+1);

	printf("\n");

	return 0;
}
