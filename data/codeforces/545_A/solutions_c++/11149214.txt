#include <cstdio>

int N, x, r;
bool cars[1000], f;

int main () {


	
	scanf("%d", &N);
	r = N;
	for (int i=0; i<N; ++i) {
		f = false;
		for (int j=0; j<N; ++j) {
			scanf("%d", &x);
			if ((x == 1 || x == 3) && !f) {
				f = cars[i] = true;
				--r;
			}
		}
	}
	printf("%d\n", r);
	for (int i=0; i<N; ++i)
		if (cars[i] == false)
			printf("%d ", i+1);
	return 0;
}
