#include <bits/stdc++.h>
using namespace std;
int a[101];
int main()
{
int i,j,n,t,k;
scanf("%d",&n);
for(i=0;i<n;i++) for(j=0;j<n;j++) {
    scanf("%d",&k);
    if(k==1) a[i]=1;
    else if(k==2) a[j]=1;
    else if(k==3) a[i]=a[j]=1;
}
j=0;
for(i=0;i<n;i++) if(!a[i]) j++;
printf("%d\n",j);
for(i=0;i<n;i++) if(!a[i]) printf("%d ",i+1);
return 0;
}
