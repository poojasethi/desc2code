#include <iostream>
#include <queue>
using namespace std;

int n,i,j,a,f,ans;
queue <int> q;

int main() {
	cin>>n;
	for(i=0;i<n;i++) {
		f=1;
		for(j=0;j<n;j++) {
			cin>>a;
			if(a==1||a==3) 
				f=0;
		}
		if(f) ans++, q.push(i+1);
	}
	cout<<ans<<endl;
	while(!q.empty()) {
		cout<<q.front()<<' ';
		q.pop();
	}
	return 0;
}
