#include <iostream>
#include <vector>
using namespace std;

int main() {
int n;
vector <int> a;
int c=0,ans=0;
cin>>n;
for(int i=0;i<n;i++){
    c=0;
    for(int j=0;j<n;j++){
    int x;
    cin>>x;
    if(x!=1&&x!=3){c++;}
    }
    if(c==n){a.push_back(i+1);ans ++;}
}
cout<<ans<<endl;
for(int i=0;i<a.size();i++)
cout<<a[i]<<" ";
    return 0;
}
