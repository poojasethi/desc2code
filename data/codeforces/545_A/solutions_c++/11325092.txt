#include<stdio.h>
#include<vector>
using namespace std;
vector<int>r;
int main(){
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		bool f=true;
		for(int j=1;j<=n;j++){
			int d;
			scanf("%d",&d);
			if(d==3||d==1)f=false;
		}
		if(f)r.push_back(i);
	}
	printf("%lu\n",r.size());
	for(int i=0;i<r.size();i++)
		printf("%d%c",r[i],i==r.size()-1?'\n':' ');
}
