#include <stdio.h>
int ans[200];
int main()
{
	int n,k;
	scanf("%d",&n);
	int cur=0;
	for(int i=0;i<n;i++)
	{
		int tag=0;
		for(int j=0;j<n;j++)
		{
			scanf("%d",&k);
			if(k==1||k==3)
			tag=1;
		}
		if(!tag)
		{
			ans[cur++]=i+1;
		}
	}
	printf("%d\n",cur);
	for(int i=0;i<cur;i++)
	printf("%d ",ans[i]);
	printf("\n");
}
