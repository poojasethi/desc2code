#include<iostream>
using namespace std;
int main()
{

int n=0,t=0;
cin>>n;
int arr[n][n],a[n];

for(int i=0;i<n;i++)
{
int c=0;
for(int j=0;j<n;j++)
{

cin>>arr[i][j];
if(arr[i][j]==1||arr[i][j]==3)
{
 c++;
}

}
if(c==0)
{
 a[t]=i+1;
 t++;
}

}
cout<<t<<"\n";
for(int i=0;i<t;i++)
{
    cout<<a[i]<<" ";
}

return 0;
}
