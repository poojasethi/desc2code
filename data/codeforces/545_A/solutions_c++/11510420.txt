#include<cstdio>
using namespace std;
int a1[100];
int main()
{
	int n,a,o=0;bool bl;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
	{
		bl=1;
		for(int j=0;j<n;++j)
		{
			scanf("%d",&a);
			if(a>0&&a!=2)bl=0;
		}
		if(bl)a1[o++]=i;
	}
	printf("%d\n",o);bl=0;
	for(int i=0;i<o;++i)
	{
		printf("%d",a1[i]+1);
		putchar(i==n-1?'\n':' ');
	}
}
