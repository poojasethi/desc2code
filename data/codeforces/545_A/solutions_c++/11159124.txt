#include<cstdio>
#include<algorithm>
int  a,i,m,t,j,x,c[105];
main() 
{
	scanf("%d",&a);
	for(i=0;i<a;i++)
	{
		for(j=0;j<a;j++)
		{
			scanf("%d",&x);
			if(x==3)c[i]=c[j]=1;
			if(x==2)c[j]=1;
			if(x==1)c[i]=1;
		}
	}
	for(i=0;i<a;i++)
	{
		if(c[i]==0)t++;
	}
	printf("%d ",t);
	for(i=0;i<a;i++)
	{
		if(c[i]==0)printf("%d ",i+1);
	}
}
