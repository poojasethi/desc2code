#include <cstdio>

const int N = 110;

int a[N];
int n, i, j, an;

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i++)
	{
		bool f = true;
		for (j = 0; j < n; j++)
		{
			int x;
			scanf("%d", &x);
			if (x == 1 || x == 3)
				f = false;
		}
		if (f)
			a[an++] = i;
	}

	printf("%d\n", an);
	for (i = 0; i < an; i++)
		printf("%d ", a[i] + 1);

	return 0;
}
