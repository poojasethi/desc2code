#include<cstdio>

int r[111],rn;

int main()
{
    int t,tt;
    int i,j,n;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        t=1;
        for(j=0;j<n;j++)
        {
            scanf("%d",&tt);
            if(tt==1||tt==3)t=0;
        }
        if(t)r[rn++]=i+1;
    }
    printf("%d\n",rn);
    for(i=0;i<rn;i++)printf("%d ",r[i]);
}
