#include <bits/stdc++.h>
using namespace std;
vector<int> v;
int main(){
int n; cin>>n; int x; for(int i=0;i<n;i++){
    bool ok=true;
    for(int j=0;j<n;j++){
        cin>>x;
        if(x==1||x==3) ok=false;
    }
    if(ok) v.push_back(i+1);
}
cout<<v.size()<<endl;
for(int i=0;i<v.size();i++){
    cout<<v[i]<<" ";
}
}
