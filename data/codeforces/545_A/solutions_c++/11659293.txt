#include <cstdio>
#include <algorithm>
using namespace std;
int n,f,x,ans1[110],ans;
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		f = 1;
		for (int j=1;j<=n;j++){
			scanf("%d",&x);
			if (x==1 || x==3) f = 0;
		}
		if (f) ans1[++ans] = i;
	}
	printf("%d\n",ans);
	for (int i=1;i<=ans;i++) printf("%d ",ans1[i]);
}
