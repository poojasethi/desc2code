#include<bits/stdc++.h>

int main()
{
	int t,i,j,k=0,n,p,u;
	scanf("%d",&n);
	int a,b[101];
	for(i=0;i<n;++i)
	{
		p=1;
		for(j=0;j<n;++j)
		{
			scanf("%d",&a);
			if(a==1 || a==3)
				p=0;
		}
		if(p)
		{
			k++;
			b[k-1]=i+1;
		}
	}
	printf("%d\n",k);
	for(i=0;i<k;++i)
		printf("%d ",b[i]);
	printf("\n");
	return 0;
}
