#include<bits/stdc++.h>

using namespace std;

int main(){
  int a[109][109],b[109],m,s,n,i,j;
  cin>>n;
  for(i=1;i<=n;i++)
    for(j=1;j<=n;j++){
		  cin>>a[i][j];
			if(a[i][j]==1 || a[i][j]==3) b[i]++;
		}
  s=0;
  for(i=1;i<=n;i++) 
	  if(b[i]==0) s++;
  cout<<s<<endl;
  for(i=1;i<=n;i++) 
	  if(b[i]==0) cout<<i<<" ";
}
