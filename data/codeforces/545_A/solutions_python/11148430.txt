n = int(raw_input())
grid = [map(int,str(raw_input()).split()) for _ in range(n)]

good_cars = []

for i in range(n):
	good = True
	for j in range(n):
		if grid[i][j] == 1 or grid[i][j] == 3: good = False
	if good: good_cars.append(i+1)

print len(good_cars)
if len(good_cars)>0: print " ".join(map(str,good_cars))
