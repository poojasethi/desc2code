#!/usr/bin/env python2
# author Jay

import sys

def main():
	infile = sys.stdin
	outfile = sys.stdout
	n, v = [int(x) for x in infile.readline().split()]
	pool = []
	inset = []
	for i in xrange(1, n+1):
		t, p = [int(x) for x in infile.readline().split()]
		pool.append((p*1.0/t, p, t, i))
	pool.sort(key=lambda item: item[0], reverse=True)
	sumary = 0
	for i in xrange(0, n):
		if v<=0:
			break;
		x = pool[i]
		sumary += x[1]
		v -= x[2]
		inset.append(x[3])
	
	if v==-1:
		found = False
		for j in xrange(-1, -i-1, -1):
			if pool[i+j][2]==1:
				found = True
				break
		if not found:
			j = -1
		sumary1 =  sumary - pool[i+j][1]
		found = False
		for k in xrange(i, n):
			if pool[k][2]==1:
				found = True
				break
		sumary2 = sumary - pool[i-1][1]
		if found:
			sumary2 += pool[k][1]
			
		if sumary1>sumary2:
			sumary = sumary1
			inset.pop(j)
		else:
			sumary = sumary2
			inset.pop()
			if found:
				inset.append(pool[k][3])
	
	outfile.write(str(sumary))
	outfile.write('\n')
	outfile.write(' '.join(str(x) for x in inset))
	

if __name__ == '__main__':
	main()
