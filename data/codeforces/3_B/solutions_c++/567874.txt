#include <iostream>
#include <algorithm>
#define MAX 100005
using namespace std;

struct node{
	   int c, pos;
}a[MAX], b[MAX];

bool cmp(node c, node d)
{
 return c.c > d.c;
}

int main()
{
 	int i, w, best, num, cost, k, n;
	int n1, n2;
	n1 = n2 = 0;
	cin >> n >> w;
	for (i = 1; i <= n; ++i)
	{
	 	cin >> num >> cost;
		if(num == 1)	a[++n1].c = cost, a[n1].pos = i;
		else	b[++n2].c = cost, b[n2].pos = i;
	}
	sort(a+1, a+n1+1, cmp);
	sort(b+1, b+n2+1, cmp);
	
	a[0].c = b[0].c = 0;
	for (i = 1; i <= n1; ++i)
		a[i].c += a[i-1].c;
	for (i = 1; i <= n2; ++i)
		b[i].c += b[i-1].c;

	best = k = 0;
	for (i = 0; i <= n2 && i+i <= w; ++i)
	{
	 	int sum = b[i].c + a[min(n1, w - i-i)].c;
		if(sum > best)
		{	best = sum; k = i;	}
	}

	cout << best << endl;
	for (i = 1; i <= k; ++i)
		cout << b[i].pos << " ";
	for (i = 1; i <= min(n1, w - k-k); ++i)
		cout << a[i].pos << " ";
	cout << endl;
	
//	system("pause");		
	return 0;
}
