#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <set>
using namespace std;
struct vel
{
	int t,c,id;
	vel(int t=0,int c=0):t(t),c(c){}
	bool operator<(const vel& tmp) const
	{
		return t*tmp.c<tmp.t*c;
	}
} a[100010];
int main()
{
	int n,v;
	scanf("%d%d",&n,&v);
	for(int i=0;i<n;i++) 
	{
		scanf("%d%d",&a[i].t,&a[i].c);
		a[i].id=i;
	}
	sort(a,a+n);
	int ans=0;
	set<int> opt;
	int last=-1;
	for(int i=0;i<n;i++) if(v>=a[i].t)
	{
		v-=a[i].t;
		ans+=a[i].c;
		opt.insert(a[i].id);
		if(a[i].t==1) last=i;
	}
	if(v==1 && last!=-1)
	{
		for(int i=0;i<n;i++) if(!opt.count(a[i].id) && a[i].t==2 && a[i].c>a[last].c)
		{
			ans+=a[i].c-a[last].c;
			opt.erase(a[last].id);opt.insert(a[i].id);
			break;
		}
	}
	printf("%d\n",ans);
	for(set<int>::iterator it=opt.begin();it!=opt.end();it++) printf("%d ",*it+1);
	puts("");
	return 0;
}
