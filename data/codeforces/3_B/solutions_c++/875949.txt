#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
struct boat
{
  int index, t, p;
  bool operator<(const boat &x) const
  {
    if (t != x.t)
      return t < x.t;
    else if (t == 1)
      return p < x.p;
    else
      return p > x.p;
  }
}boats[100002];
int n, v;
int main ()
{
  int i, j, max, maxi, mt, mp;
  scanf ("%d %d", &n, &v);
  for (i = 0; i < n; i++) {
    scanf ("%d %d", &boats[i].t, &boats[i].p);
    boats[i].index = i + 1;
  }
  sort (boats, boats + n);
  mt = mp = max = 0;
  for (i = j = 0; j < n; j++) {
    mt += boats[j].t;
    mp += boats[j].p;
    while (mt > v) {
      mt -= boats[i].t;
      mp -= boats[i].p;
      i++;
    }
    if (mp > max) {
      max = mp;
      maxi = i;
    }
  }
  printf ("%d\n", max);
  for (i = j = maxi, mt = mp = 0; j < n; j++) {
    mt += boats[j].t;
    mp += boats[j].p;
    if (mt > v) break;
    printf (" %d"+(i==j), boats[j].index);
  }
  printf ("\n");
  return 0;
}
