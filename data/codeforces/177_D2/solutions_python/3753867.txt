#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

int main(void)
{
	int n, m , c;
	cin >> n >> m >> c;
	vector <int> a(n);
	for (int i = 0; i < n; i++)
		scanf("%i",&a[i]);
	vector <int> C(m);
	for (int i = 0; i < m; i++)
		scanf("%i",&C[i]);
	int sum = 0;
	for (int i = 0; i < n; i++)
	{
		if (i < m)
			sum = (sum + C[i]) % c;
		if (i > n - m)
		{
			sum -= C[m-1-n+i];
			if (sum < 0)	sum += c;
		}
		a[i] = (sum + a[i]) % c;
		cout << a[i] << " ";
	}
	cout << endl;
	return 0;
}
