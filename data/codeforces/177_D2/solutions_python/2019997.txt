#include <iostream>
#include <cstdio>
#include <cstring>
#define FOR(i, n) for (int i = 0; i < (n); ++i)
#define MAX 100000
using namespace std;

int n, m, c, a[MAX], b[MAX + 1], tb;

inline int sumb(int i, int j) { return (b[j] - b[i - 1] + c) % c; }

int main() {
	scanf("%d%d%d", &n, &m, &c);
	FOR (i, n) scanf("%d", &a[i]);
	FOR (i, m) scanf("%d", &tb), b[i + 1] = (b[i] + tb) % c;
	FOR (i, n) printf("%d%c", (a[i] + sumb(m - min(n - i - 1, m - 1), min(i, m - 1) + 1)) % c, (i < n - 1) ? ' ' : '\n');
	return 0;
}
