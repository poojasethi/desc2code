#include <iostream>

using namespace std;

#define MAX 100005

int a[MAX], b[MAX];

int main()
{
	int n, m, c;
	cin >> n >> m >> c;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < m; i++)
		cin >> b[i];
	int sum = 0;
	for (int i = 0; i < n - m + 1; i++)
	{
		if (i < m)
			sum += b[i];
		a[i] = (a[i] + sum) % c;
	}
	for (int i = n - m + 1, j = 0; i < n; i++, j++)
	{
		if (i < m)
			sum += b[i];
		sum -= b[j];
		a[i] = (a[i] + sum) % c;
	}
	for (int i = 0; i < n; i++)
		cout << a[i] << " ";
	cout << endl;
}
