#include <iostream>
#include <cstdio>
#include <algorithm>
#define N 100010
using namespace std;
int a[N],b[N],n,m,c;
int main()
{
    scanf("%d%d%d",&n,&m,&c);
    for(int i=0;i<n;i++) scanf("%d",a+i);
    for(int i=0;i<m;i++) {scanf("%d",b+i);if(i) b[i]+=b[i-1];}
    for(int i=0;i<n;i++)
    {
        int r=min(m-1,i),l=max(i-n+m,0);
        a[i]+=(b[r]-b[l-1])%c;
        cout<<(a[i]+c)%c<<" ";
    }
    return 0;
}
