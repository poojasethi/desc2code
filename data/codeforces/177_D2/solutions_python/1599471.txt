#include <cstdlib>
#include <cstring>
#include <cstdio>

int n,m,c;
int i,j;
int a[100010],b[100010],now;

int main()
{
	scanf("%d%d%d",&n,&m,&c);
	for (i=0;i<n;i++)scanf("%d",&a[i]);
	for (i=0;i<m;i++)scanf("%d",&b[i]);
	now=0;
	for (i=0;i<n;i++)
	{
		if (i<m)
		{
			now+=b[i];
		}
		if (i+m-1>=n)
		{
			now-=b[i+m-1-n];
		}
		a[i]=(a[i]+now)%c;
	}
	for (i=0;i<n;i++)
	{
		printf("%d",a[i]);
		if (i<n-1)printf(" ");
		else printf("\n");
	}
	return 0;
}
