#include<iostream>
#include<algorithm>
using namespace std;

bool comp(int a,int b)
{
	return a < b;
}

int n,c,m;
int a[100005];
int b[100005];


int main()
{
	int add = 0;
	cin >> n >> m >> c;
	for(int i = 1; i <= n; i++)
		cin >> a[i];
	for(int i = 1; i <= m; i++)
		cin >> b[i];
	for(int i = 1; i <= n; i++)
	{
		if(i <= m)
			add = (add + b[i]);
		if(n - i + 1 < m)
			add = (add - b[m - n + i - 1]);
		a[i] = (a[i] + add) % c;
	}
	for(int i = 1; i <= n; i++)
		cout << a[i] << " ";
	return 0;		
}
