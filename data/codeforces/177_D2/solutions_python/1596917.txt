#include<cstdio>
int n,m,a[100005],b,s[100005],c,ac;
int main(){
	scanf("%d%d%d",&n,&m,&c);
	for (int i=1; i<=n+1; i++) s[i]=0;
	for (int i=1; i<=n; i++)
		scanf("%d",&a[i]);
	s[0]=0;
	for (int i=1; i<=m; i++){
		scanf("%d",&b);
		s[i]+=b; s[i]=s[i]%c;
		s[i+n-m+1]-=b; s[i+n-m+1]=(s[i+n-m+1]+c)%c;
	}
	for (int i=1; i<=n; i++){
		s[i]+=s[i-1];
		ac=(a[i]+s[i])%c;
		printf("%d",ac);
		if (i<n) printf(" "); else printf("\n");
	}
	return 0;
}
