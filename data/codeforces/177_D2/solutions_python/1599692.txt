#include <iostream>

using namespace std;

int main()
{
    long m, n, c, mas[200000] = {0}, i, j, b[200000] = {0}, sum = 0;
    
    cin >> m >> n >> c;
    
    for(i = 0; i < m; ++i)
        cin >> mas[i];
    
    for(i = 0; i < n; ++i)
        cin >> b[i];
    
    for(i = 0; i < m; ++i)
    {
        sum += b[i];
        if(i-(m-n+1) >= 0)
            sum -= b[i-(m-n+1)];
        cout << (mas[i]+sum)%c << ' ';
    }
    
    return 0;
}
