#include <iostream>
using namespace std;
int n,m,c,a[111111],b[111111],i;
int mi(int x, int y)
{
	if (x<y)
		return x;
	else return y;
}
int ma(int x, int y)
{
	if (x>y)
		return x;
	else return y;
}
int main()
{
	cin>>n>>m>>c;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=1;i<=m;i++)
	{
		cin>>b[i];
		b[i]+=b[i-1];
	}
	for(i=0;i<n;i++)
	{
		cout<<(a[i]+b[mi(m,i+1)]-b[ma(0,m-(n-i))])%c<<" ";
	}
    return 0;
}
