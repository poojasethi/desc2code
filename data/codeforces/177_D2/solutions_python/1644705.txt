#include<cstdio>
using namespace std;
int main()
{
    int n,i,m,c,a[100000],b[100000],s;
    scanf("%d%d%d",&n,&m,&c);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    for(i=0;i<m;i++)
    scanf("%d",&b[i]);
    s=0;
    for(i=0;i<n;i++)
    {
        if(i<m)s=(s%c+b[i]%c)%c;
        if(i-(n-m)-1>=0 && i-(n-m)-1<m)s=((s%c-b[i-(n-m)-1]%c)%c+c)%c;
        printf("%d",(s+a[i]%c)%c);
        if(i!=n-1)printf(" ");
    }
    printf("\n");
}
