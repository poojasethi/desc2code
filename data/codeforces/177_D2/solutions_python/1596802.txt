#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int a[100010],b[100010];
main()
{
    int n,m,c,x;
    cin >> n >> m >> c;
    for(int i=0;i<n;i++) cin >> a[i];
    for(int i=0;i<m;i++)
    {
        cin >> x;
        b[i+1]=(b[i]+x)%c;
    }
    for(int i=0;i<n;i++)
    {
        x=b[min(i+1,m)]-b[max(m+i-n,0)]+a[i];
        x%=c;
        x+=c;
        x%=c;
        cout << x << " ";
    }
    cout << endl;
    
    
    
    
    scanf(" ");
}
