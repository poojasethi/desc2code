#include <algorithm>
#include <iostream>
using namespace std;

int a[100000];
int b[100000];
int s[100000 + 1];

int main() {
	int n, m, c;

	ios_base::sync_with_stdio(false);

	cin >> n >> m >> c;
	for (int i = 0; i < n; ++i)
		cin >> a[i];
	for (int i = 0; i < m; ++i)
		cin >> b[i];

	s[0] = 0;
	for (int i = 1; i <= m; ++i)
		s[i] = (s[i - 1] + b[i - 1])%c;

	for (int i = 0; i < n; ++i)
		cout << (a[i] + (s[min(i, m - 1) + 1] - s[max(0, i - (n - m))]) + c)%c << " ";
}
