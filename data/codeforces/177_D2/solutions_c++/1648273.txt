#include <iostream>

using namespace std;

int n, m, c;
int a[100005];
int b[100005];

int main(){
	cin >> n >> m >> c;
	for (int i = 0; i < n; ++i)
		cin >> a[i];
	for (int j = 0; j < m; ++j){
		cin >> b[j];
		b[j] += b[j - 1];
		b[j] %= c;
	}
	
	for (int i = 0; i < n; ++i){
		int r = min(i, m - 1);
		int l = max(i - n + m, 0);
//cout << endl << l << " " << r << endl;
	 	int ans = a[i] + b[r];
		if (l > 0) ans -= b[l - 1];
		cout << (ans % c + c) % c << " ";
	}
	return 0;
}
