#include <iostream>
#include <cmath>
#include <cstdio>
#include <algorithm>
using namespace std;

int n, m, c, x, a[100000], b[300100];
int r;

int main() {
    scanf("%d%d%d",&n,&m,&c);
    r=n-m;
    for (int i=0; i<n; i++) scanf("%d",&a[i]);
    for (int i=r+1; i<m+r+1; i++) scanf("%d",&b[i]);
    for (int i=0; i<n; i++) {
        x+=b[i+r+1]; x-=b[i];
        a[i]+=x; a[i]%=c;
    }
    for (int i=0; i<n; i++) printf("%d ",a[i]);
    return 0;
}
