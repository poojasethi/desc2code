#include <iostream>
using namespace std;
const int MAXN=100000+100;
int a[MAXN];
int b[MAXN];
int s[MAXN];
int main()
{
	int n,m,c;
	cin>>n>>m>>c;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int i=1;i<=m;i++)
		cin>>b[i];
	for(int i=1;i<=m;i++)
		s[i]=s[i-1]+b[i];
	for(int i=1;i<=n;i++)
	{
		int r=m;
		if(i < m )
			r=i;
		int l=1;
		if(i >= n-m+1 )
			l=m-(n-i+1)+1;
		a[i]=( a[i]+s[r]-s[l-1] )%c;
	}
	for(int i=1;i<=n;i++)
		cout<<a[i]<<" ";
	cout<<endl;
	return 0;
}
