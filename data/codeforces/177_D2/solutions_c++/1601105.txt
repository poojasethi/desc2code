#include<stdio.h>
#define MAXN 100100

long a[MAXN], s[MAXN];

int main(){
	long n, m, c;
	scanf("%ld%ld%ld", &n, &m, &c);
	
	for(int i = 0; i < n; i++){
		scanf("%ld", &a[i]);
	}
	for(int i = 0; i < m; i++){
		scanf("%ld", &s[i]);
	}
	
	for(int i = 0; i < m; i++){
		s[i] += s[i - 1];
	}
	for(int i = 0; i < n; i++){
		long resIndex = (i < m ? i : m - 1);
		a[i] += s[resIndex];
		if(n - i < m)
			a[i] -= s[m - n + i - 1];
		while(a[i] < 0)
			a[i] += c;
		printf("%ld ", a[i] % c);
	}
}
