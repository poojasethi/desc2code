# include <iostream>
# include <fstream>
using namespace std;


int a[100000],b[100000];


int main(){


	//freopen ("input.txt", "r", stdin);
   // freopen ("output.txt", "w", stdout);

  int m,n,c,t;

  cin>>n>>m>>c;

  for(int i=1;i<=n;++i)
	  cin>>a[i];

  for(int i=1;i<=m;++i)
     cin>>b[i];


  t=n-m;

  for(int i=1; i<=m;++i)
	  for(int j=i;j<=i+t;++j)
		  a[j]=a[j]+b[i];


 


  for(int i=1;i<=n;++i)
	  cout<<a[i]%c<<' ';
	return 0;
}
