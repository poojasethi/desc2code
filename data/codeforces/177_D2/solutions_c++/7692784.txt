#include <iostream>
#include <cstdio>
#define N 100010
using namespace std;
int a[N],b[N],n,m,c;
int main()
{
    cin>>n>>m>>c;
    for(int i=0;i<n;i++) cin>>a[i];
    for(int i=0;i<m;i++) {cin>>b[i];if(i) b[i]+=b[i-1];}
    for(int i=0;i<n;i++){
        int r=min(m-1,i),l=max(i-n+m,0);
        a[i]+=b[r]-b[l-1];
        cout<<a[i]%c<<" ";
    }
    return 0;
}
