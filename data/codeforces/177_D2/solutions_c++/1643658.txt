#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
#define REP(i,n) for(int(i)=0;(i)<(int)(n);(i)++)
const int MAX=(int)1e5+5;
int a[MAX],b[MAX],n,m,c,ret[MAX];
int main()
{
	scanf("%d%d%d",&n,&m,&c);
	for (int i=1; i<=n; i++) scanf("%d",a+i);
	for (int i=1; i<=m; i++) {
		scanf("%d",b+i);
		b[i]=(b[i]+b[i-1])%c;
	}
	for (int i=1; i<=n; i++) {
		if (i<=n-m+1) ret[i]=(a[i]+(i>=m?b[m]:b[i]))%c;
		else ret[i]=(a[i]+b[(i>=m)?m:i]+(c-b[m-1-(n-i)]))%c;
		printf("%d ",ret[i]);
	}
	return 0;
}
