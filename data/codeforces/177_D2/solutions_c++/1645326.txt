# include <stdio.h>

int n,m,c;

int n1[100050];
int m1[100050];
int ps[100050];

int main()
{
	scanf("%d %d %d",&n,&m,&c);
	
	for(int i=0;i<n;i++) scanf("%d",&n1[i]),n1[i]%=c;
	for(int i=0;i<m;i++) scanf("%d",&m1[i]),m1[i]%=c;
	
	ps[0]=m1[0];
	
	for(int i=1;i<m;i++) ps[i]+=m1[i]+ps[i-1];
	
	for(int i=0;i<n;i++)
	{
		int k=i,l;
		
		if(k >= m) k=m-1;
		
		l=m-(n-i-1)-1; if(l <= 0) l=-1;
		
		n1[i]+=ps[k];
		
		if(l != -1) n1[i]-=ps[l-1];
		
		n1[i]%=c;
	}
	
	for(int i=0;i<n-1;i++)
	printf("%d ",n1[i]);
	printf("%d\n",n1[n-1]);
}
