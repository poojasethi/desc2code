#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e5+100;

int a[maxn],b[maxn];
int  s = 0;

int main(){
    int n,m,c;cin >> n >> m >> c;
    for(int i = 0;i < n;i++)
        cin >> a[i];
    for(int i = 0;i < m;i++)
        cin >> b[i];
    for(int i = 0;i < n;i++){
        if(i < m) s = (s + b[i]) % c;
        if(i > n-m) s = (s + c - b[i-1-(n-m)])%c;
        a[i] = (a[i] + s) % c;
    }
    for(int i=0;i<n;i++) 
        cout << a[i] << " ";
    cout << endl;
}
