#include<iostream>
using namespace std;

const int MAX = 100*1000+10;

int a[MAX+10];
int sum[MAX+10];
int b[MAX+10];
int n,m,D;

int main()
{
	cin >> n >> m >> D;
	for(int i=0; i<n; i++)
		cin >> a[i];
	for(int j=0; j<m; j++)
		cin >> b[j] , sum[j+1] = ( sum[j] + b[j] )%D;
	for(int i=0; i<n; i++)
	{
		int s = max( 0 , i-(n-m) );
		int e = min( m-1 , i );
		a[i] += sum[e+1] - sum[s];
		a[i] %= D;
		if( a[i] < 0 )
			a[i] += D;
		cout << a[i] << " ";
	}
	cout << endl;
	return 0;
}
