#include <stdio.h>

int main(void) {
    int n, m, c, i;
    scanf("%d %d %d", &n, &m, &c);

    int N[n], M[m], K[m+1];
    for(i = 0; i<n; i++){
        scanf("%d", &N[i]);
    }
    K[0] = 0;
    for(i = 0; i<m; i++){
       scanf("%d", &M[i]);
       K[i+1] = K[i] + M[i];
    }
    for(i = 0; i<n; i++){
        int min = (i+1) < m ? i+1 : m;
        int max = (i-n+m) > 0 ? i-n+m : 0;
        printf("%d ", (N[i] + K[min] - K[max])%c);
    }
    return 0;
}
