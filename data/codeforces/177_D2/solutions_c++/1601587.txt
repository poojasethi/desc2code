#include <iostream>
using namespace std;

int main()
{
	int a[100001], b[100001], c, n, m;
	cin>>n>>m>>c;
	for (int i=1; i<=n; ++i) cin>>a[i];
	for (int i=1; i<=m; ++i) cin>>b[i];
	for (int i=1; i<=m; ++i)
		for (int j=i; j<=i+n-m; ++j) a[j]=a[j]+b[i];
	for (int i=1; i<n; ++i) cout<<a[i]%c<<" ";
	cout<<a[n]%c<<endl;
	return 0;
}
