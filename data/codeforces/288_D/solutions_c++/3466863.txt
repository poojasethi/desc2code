#include <stdio.h>
#include <vector>
using namespace std;
vector<int> e[100000];
long long n;
long long s[100000];
long long ans;
void dfs(int fa, int x){
    int i, y;
    s[x] = 1;
    for (i = 0; i < (int)e[x].size(); i++){
        y = e[x][i];
        if (y == fa) continue;
        dfs(x, y);
        s[x] += s[y];
    }
    long long t = s[x] - 1;
    for (i = 0; i < (int)e[x].size(); i++){
        y = e[x][i];
        if (y == fa) continue;
        t += s[y] * (s[x] - s[y]);
    }
    t /= 2;
    ans -= t * t + t * s[x]  * (n - s[x]) * 2;
}
int main(){
    int i, x, y;
    scanf("%I64d", &n);
    for (i = 1; i < n; i++){
        scanf("%d%d", &x, &y);
        e[x].push_back(y);
        e[y].push_back(x);
    }
    ans = (n * (n - 1) / 2) * (n * (n - 1) / 2);
    dfs(0, 1);
    printf("%I64d\n", ans);
    return 0;
}
