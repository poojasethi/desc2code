#include <cstdio>
#include <cstring>
#include <vector>
#define ll long long

using namespace std;
const int MAX=80010;

ll a[MAX],n;
ll ans;
vector<int> edge[MAX];

void dfs(int u,int f)
{
    a[u]=1;
    ll in=0;
    for(int i=0;i<edge[u].size();i++)
    {
        int v=edge[u][i];
        if(v==f)
            continue;
        dfs(v,u);
        in+=a[u]*a[v];
        a[u]+=a[v];
    }
    ans=ans-in*(in+(n-a[u])*a[u]*2);
}

int main()
{
    scanf("%I64d",&n);

    for(int i=0;i<=n;i++)
        edge[i].clear();
    int a,b;
    for(int i=0;i<n-1;i++)
    {
        scanf("%d%d",&a,&b);
        edge[a].push_back(b);
        edge[b].push_back(a);
    }
    ans=n*(n-1)/2;
    ans=ans*ans;
    dfs(1,0);
    printf("%I64d\n",ans);
    return 0;
}
