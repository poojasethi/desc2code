#include<iostream>
using namespace std;
#define N 201000
typedef unsigned long long LL;
LL ans,s[N],n,pre[N],last[N],other[N],p,q,tot;
void add(LL p,LL q)
{
	pre[++tot]=last[p];
	last[p]=tot;
	other[tot]=q;
}

void dfs(LL x,LL f)
{
	s[x]=1LL;
	LL t=0LL;
	for (LL j=last[x];j;j=pre[j])
	{
		LL k=other[j];
		if (k==f) continue;
		dfs(k,x);
		t+=s[x]*s[k];
		s[x]+=s[k];
	}
	ans-=t*(t+2LL*s[x]*(n-s[x]));
}

int main()
{
	cin>>n;
	ans=(n*(n-1LL)/2LL)*(n*(n-1LL)/2LL);
	for (LL i=1;i<n;i++)
	{
		cin>>p>>q;
		add(p,q),add(q,p);
	}
	dfs(1,0);
	cout<<ans<<endl;
}
