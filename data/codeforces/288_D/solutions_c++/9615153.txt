#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define N 80005
#define Long long long
#define For(i,l,r) for(int i=l;i<=r;i++)
#define Rep(i,v,t) for(int i=son[t],v=ed[i];i;i=next[i],v=ed[i])
using namespace std;
int son[N],next[2*N],ed[2*N],cnt;
int size[N],n;
Long Ans;
void Tree_dp(int t,int f){
    Rep(i,v,t) if(v!=f) Tree_dp(v,t);
    
    Long path=0; size[t]=1;
    Rep(i,v,t) if(v!=f){
        path+=1LL*size[t]*size[v];
        size[t]+=size[v];
    }
    
    Ans-=path*(path+2LL*size[t]*(n-size[t]));
}
int main(){
    scanf("%d",&n);
    For(i,1,n-1){
        int x,y; scanf("%d%d",&x,&y);
        next[++cnt]=son[x]; son[x]=cnt; ed[cnt]=y;
        next[++cnt]=son[y]; son[y]=cnt; ed[cnt]=x;
    }
    Ans=1LL*n*(n-1)/2; Ans*=Ans;
    Tree_dp(1,0);
    
    printf("%I64d\n",Ans);
}
