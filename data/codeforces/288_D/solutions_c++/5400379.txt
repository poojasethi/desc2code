#include <cstdio>
#include <vector>

using namespace std;

typedef long long int ll;

ll wyn=0LL;

vector<int>graf[80005];

int n;

int licz(int w,int o)
{
  int g=1;
  ll suma=0LL,kwa=0LL;
  ll amus=0LL,awk=0LL;
  for(int i=0;i<graf[w].size();i++)
  {
    if(graf[w][i]==o) continue;
    int f=licz(graf[w][i],w);
    g+=f;
    suma+=f;
    kwa+=((ll)f)*f;
    ll cos=(((ll)f)*(f-1))/2;
    amus+=cos;
    awk+=cos*cos;
  }
  wyn+=((suma*suma-kwa)/2+suma)*((((ll)(n-g))*(n-g-1))/2);
  wyn-=(amus*amus-awk)/2;
  return g;
}

int main()
{
  scanf("%d",&n);
  for(int i=1;i<n;i++)
  {
    int a,b;
    scanf("%d%d",&a,&b);
    graf[a].push_back(b);
    graf[b].push_back(a);
  }
  licz(1,-1);
  printf("%lld\n",wyn*2);
  return 0;
}
