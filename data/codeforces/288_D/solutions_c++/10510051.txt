#include <bits/stdc++.h>

#define pb push_back

using namespace std;

typedef long long ll;
const int TAM = 88888;

ll ans, n, pend, cont;
vector < int > graph[TAM];
ll nh[88888];
ll npair[88888];

int go ( int node, int parent ) {
	nh[node] = 0;
	npair[node] = 0;

	ll aux, prevcont = cont;

	for ( int i = 0; i < graph[node].size(); i ++ ) {
		int v = graph[node][i];

		if ( v == parent ) continue;

		aux = go ( v, node );
		npair[node] += aux * nh[node];

		nh[node] += aux;
	}
	
	pend --;
	npair[node] += nh[node];
	nh[node] ++;

	ans += npair[node] * (n - nh[node]) * (n - nh[node] - 1) / 2;	
	ans -= npair[node] * ( prevcont );
	cont += npair[node];

	return nh[node];
}

int main() {

	int a, b;

	cin >> n;

	for ( int i = 1; i < n; i ++ ) {
		cin >> a >> b;

		graph[a].pb ( b );
		graph[b].pb ( a );
	}

	cont = 0;
	go ( 1, -1 );

	cout << 2ll * ans << '\n';

	return 0;
}
