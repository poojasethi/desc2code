#include <bits/stdc++.h>
#define PS system("pause");
#define maxn 100009
using namespace std;
vector<int>G[maxn];
int n,num[maxn];
long long ans;
void dfs(int u,int fa)
{
	num[u]=1;
	long long sum=0;
	for(int i=0;i<(int)G[u].size();i++)
	{
		int v=G[u][i];
		if(v==fa)
			continue;
		dfs(v,u);
		sum+=(long long)num[u]*num[v];
		num[u]+=num[v];
	}
	ans+=sum*sum+sum*(n-num[u])*num[u]*2;
}
int main()
{
	cin>>n;
	for(int i=1;i<n;i++)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		G[x].push_back(y);
		G[y].push_back(x);
	}
	dfs(1,-1);
	long long tot=(long long)n*(n-1)/2;
	cout<<tot*tot-ans<<endl;
	//PS;
	return 0;
}
