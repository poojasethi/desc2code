#include <bits/stdc++.h>

#define st first
#define nd second
#define mp make_pair
#define pb push_back
#define lli long long int
#define FP( ii,aa,bb ) for( int ii=aa;ii<=bb;ii++ )
#define FM( ii,aa,bb ) for( int ii=aa;ii>=bb;ii-- )

using namespace std;

vector<lli> V[100000];
lli n,res,child[100000];

void dfs( lli nod,lli pre ){
	
	vector<lli> v;
	lli len = V[nod].size();
	child[nod] = 1;
	FP( i,0,len-1 ){
		if( V[nod][i]==pre ) continue;
		v.push_back( V[nod][i] );
		dfs( v.back(),nod );
		child[nod] += child[ v.back() ];
	}

	lli r=0;
	len = v.size();
	FP( i,0,len-1 )
		r += child[v[i]]*(child[nod]-1-child[v[i]]);
	r /= 2;
	r += child[nod]-1;
	res += r*r;
	res += 2*r*(child[nod])*(n-child[nod]);

}

int main(){

	lli x,y;
	
	cin >> n;
	FP( i,1,n-1 ){
		cin >> x >> y;
		V[x].pb(y);
		V[y].pb(x);
	}

	dfs( 1,0 );
	cout << (n*(n-1)/2)*(n*(n-1)/2)-res << endl;

}
