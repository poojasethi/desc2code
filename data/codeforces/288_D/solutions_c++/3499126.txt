#include <iostream>
#include <stdio.h>
#include <list>
using namespace std; 
typedef long long LL;
const int N=200000;
list<int> h[N];
LL n; 
LL ans, s[N];
void init()
{
    cin >> n; 
    ans=n*(n-1)/2;
    ans=ans*(ans-1);
    int x, y; 
    for (int i=1; i<n; ++i)
    {
        scanf("%d%d", &x, &y);
        h[x].push_back(y); 
        h[y].push_back(x);
    }
}
void dfs(int x, int l)
{
    list<int>::iterator p;
    LL fs, ts=0, sum=0; 
    for (p=h[x].begin(); p!=h[x].end(); ++p)
        if (*p!=l)
        {
            dfs(*p, x);
            s[x]+=s[*p];
            ts+=sum*s[*p];
            sum+=s[*p];
        }
    ++s[x];
    ts+=sum; 
    fs=n-s[x];
    ans-=ts*(ts-1)+ts*s[x]*fs*2; 
}
int main()
{

    init();
    dfs(1, 0);
    cout << ans; 
}
