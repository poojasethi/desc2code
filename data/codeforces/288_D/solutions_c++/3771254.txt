#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 160005
#define ll long long
#define rep(i,n) for (int i=1;i<=n;i++)
int n,x,y,l,son[N],ed[N],next[N],fa[N];
ll s[N],res,ans;
ll sqr(ll x){return x*x;}
void dfs(int x,int last)
{
	s[x]=1; ll res=0;
	for (int p=son[x];p;p=next[p]){
		int y=ed[p]; if (y!=last) dfs(y,x),res+=s[x]*s[y],s[x]+=s[y];
		}
	ans-=res*(res+s[x]*(n-s[x])*2);
}
int main()
{
	scanf("%d",&n),ans=sqr(1LL*n*(n-1)/2);
	rep(i,n-1) scanf("%d%d",&x,&y),
		ed[++l]=y,next[l]=son[x],son[x]=l,
		ed[++l]=x,next[l]=son[y],son[y]=l;
	dfs(1,0),printf("%I64d\n",ans); return 0;
}
