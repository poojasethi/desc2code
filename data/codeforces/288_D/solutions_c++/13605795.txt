const int N=8e4+5;
const int mod=1e9+7;

int n,x,y,i,ans;
int ch[N];

vector<int> V[N];

void dfs(int x,int ata){

	ch[x]=1;

	int i,t=0;

	for(i=0 ; i<V[x].size() ; i++)
		if(V[x][i]!=ata){
			dfs(V[x][i],x);
			t+=ch[x]*ch[V[x][i]];
			ch[x]+=ch[V[x][i]];
		}
	
	ans-=t*(t+2*ch[x]*(n-ch[x]));

	return;
}

main(){

	scanf("%lld",&n);

	for(i=1 ; i<n ; i++){
		scanf("%lld %lld",&x,&y);
		V[x].pb(y);
		V[y].pb(x);
	}

	ans=n*(n-1)/2;
	ans*=ans;

	dfs(1,-1);

	cout<<ans;
}
