#include <iostream>
#include <cmath>
#include <vector>

#define ll long long
#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,t1,t2;
vector<int> g[100000];
ll res,c[100000];

void dfs(int x,int p){
  c[x] = 1;
  ll s = 0;
  rep(i,g[x].size()){
    int k = g[x][i];
    if(k==p)continue;
    dfs(k,x);
    s += c[x]*c[k];
    c[x] += c[k];
  }
  res -= s*(s+2*c[x]*(n-c[x]));
}


int main(){
  cin>>n;
  rep(i,n-1){
    cin>>t1>>t2;t1--;t2--;
    g[t1].push_back(t2);
    g[t2].push_back(t1);
  }
  res = (ll)n*(n-1)/2;
  res *=res;
  dfs(0,-1);
  cout<<res<<endl;
}
