#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
#include<iostream>
#define N 180000
#define int64 unsigned long long
using namespace std;
int i,n,x,y,next[N],head[N],len,e[N],s[N];
int64 sum[N],ans;
void dfs(int x,int ff){
	int i;
	int64 sum=0;
	s[x]++;
	for(i=head[x];i;i=next[i])if(e[i]!=ff){
		dfs(e[i],x);
		sum+=(int64)s[x]*s[e[i]];
		s[x]+=s[e[i]];
	}
	ans+=(int64)s[x]*(n-s[x])*sum*2+sum*sum;
}	
int main(){
	scanf("%d",&n);
	for(i=1;i<n;++i){
		scanf("%d%d",&x,&y);
		next[++len]=head[x];
		head[x]=len;
		e[len]=y;
		next[++len]=head[y];
		head[y]=len;
		e[len]=x;
	}
	dfs(1,0);
	int64 fans=(int64)n*(n-1)/2;
	fans*=fans;
	cout<<fans-ans;
}
