#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<vector>

using namespace std;
const int N=80010;
typedef long long ll;
vector<int> v[N];
int f[N],sum[N],n;
ll ans;

void dfs(int x,int fa)
{
    sum[x]++; f[x]=fa;
    ll s=0;
    for (int i=0; i<v[x].size(); i++)
    {
        int y=v[x][i];
        if (y!=fa) 
        {
            dfs(y,x);
            s+=(ll)sum[x]*(ll)sum[y]; 
            sum[x]+=sum[y];
        }
    }
    ans-=s*(s+2*(ll)sum[x]*(ll)(n-sum[x]));
}

int main()
{
    int i,x,y;
    scanf("%d",&n);
    for (i=1; i<n; i++) 
    {scanf("%d%d",&x,&y); v[x].push_back(y); v[y].push_back(x);}
    ans=(ll)n*(ll)(n-1)/2;
    ans*=ans;
    dfs(1,1);
    cout<<ans<<endl;
}
