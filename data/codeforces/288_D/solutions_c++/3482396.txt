#include<cstdio>
#include<vector>
using namespace std;

const int maxn=80005;
typedef __int64 lld;

int n;
lld a[maxn];
lld ans;
vector<int>edge[maxn];
lld tot[maxn];

void dfs(int u,int f){
    a[u]=1;
    for(int i=0;i<edge[u].size();i++){
        int v=edge[u][i];
        if(v==f)continue;
        dfs(v,u);
        tot[u]+=a[u]*a[v];
        a[u]+=a[v];
    }
    ans-=tot[u]*(tot[u]+2*a[u]*(n-a[u]));
}
int main(){
    int i,j;
    int u,v;
    scanf("%d",&n);
    for(i=1;i<n;i++){
        scanf("%d%d",&u,&v);
        edge[u].push_back(v);
        edge[v].push_back(u);
    }
    ans=(lld)n*(n-1)/2;
    ans*=ans;
    dfs(1,0);
    printf("%I64d\n",ans);
    return 0;
}
