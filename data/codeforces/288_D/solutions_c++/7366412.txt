#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>

using namespace std ;
typedef long long ll;
const int MAXN = 1e5;

vector<int> e[MAXN];
ll N,M;
ll sz[MAXN],ans;

void dfs(int u,int pre){
    sz[u]=1;
    ll sum=0;
    for (int i=0;i<e[u].size();i++){
        int v=e[u][i];
        if (v==pre) continue;
        dfs(v,u);
        sz[u]+=sz[v];
        sum+=sz[v]*sz[v];
    }
    ll cnt=(sz[u]*sz[u]-sum-1)/2;
    ans-=cnt*cnt+cnt*(N-sz[u])*sz[u]*2;
}

int main(){
    cin>>N;
    int x,y;
    ans=N*(N-1)/2;
    ans=ans*ans;
    for (int i=0;i<N-1;i++){
        scanf("%d%d",&x,&y);
        e[x].push_back(y);
        e[y].push_back(x);
    }
    dfs(1,-1);
    cout<<ans<<endl;
    return 0;
}
