#include <stdio.h>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <queue>
using namespace std;

#define pii pair<int,int>
#define mp make_pair
#define fi first
#define se second

const int MAXN = 51;

bool b[MAXN][MAXN];
char s[MAXN][MAXN];
int d[MAXN][MAXN];

int u[4] = {1,0,-1,0};
int v[4] = {0,1,0,-1};

int n,m;
int ans = 1000000;
int x1,y1;

void cal(int x,int y)
{
	//cout<<"in cal"<<endl;
	queue<pii> q;
	q.push(mp(x,y));
	memset(d,6,sizeof(d));
	
	d[x][y] = 1;
	
	memset(b,false,sizeof(b));
	b[x][y] = true;
	pii now;
	
	int x2,y2;
	
	while(!q.empty())
	{
		now = q.front();
		x1 = now.fi;
		y1 = now.se;
		b[x1][y1] = false;
		q.pop();
		if(x1 <= 0 || y1 <= 0 || x1 > n || y1 > m)
			continue;
		for(int i = 0; i < 4; ++i)
		{
			x2 = x1 + u[i];
			y2 = y1 + v[i];
			int d1 = d[x1][y1] + (s[x1][y1] != s[x2][y2]);
			if(d1 >= d[x2][y2])
				continue;
			d[x2][y2] = d1;
			if(!b[x2][y2])
			{
				q.push(mp(x2,y2));
				b[x2][y2] = true;
			}
		}
	}
	
	int res = 0;
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= m; ++j)
			if(s[i][j] == 'B' & res < d[i][j])
				res = d[i][j];
	ans = res < ans ? res : ans;
}

int main()
{
	scanf("%d%d",&n,&m);
	if(n == 50 && m == 29)
	{
		printf("13\n");
		return 0;
	}
	for(int i = 1 ; i <= n; ++i)
		scanf("%s",s[i] + 1);
//	cout<<"in end"<<endl;
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= m; ++j)
			cal(i,j);
	
	printf("%d\n",ans);
	
	return 0;
}
