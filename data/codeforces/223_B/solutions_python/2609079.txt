#!/usr/bin/python
import sys

s = raw_input()
t = raw_input()
n = len(s)
m = len(t)
nxt = [n for i in range(n)]
last = {}
for i in xrange(n-1, -1, -1):
	c = s[i]
	if c in last:
		nxt[i] = last[c]
	last[c] = i

cursors = [0 for i in range(m)]
for i in xrange(m):
	start = 0 if i == 0 else cursors[i-1]+1
	for j in xrange(start, n):
		if s[j] == t[i]:
			cursors[i] = j
			break
	else:
		print "No"
		sys.exit(0)

cursors2 = [0 for i in range(m)]
for i in xrange(m-1, -1, -1):
	start = n-1 if i == m-1 else cursors2[i+1]-1
	for j in xrange(start, -1, -1):
		if s[j] == t[i]:
			cursors2[i] = j
			break

selected = [0 for i in range(n)]
for i in xrange(m-1, -1, -1):
	c = cursors[i]
	while True:
		if selected[c] == 1:
			break
		selected[c] = 1
		if c == cursors2[i]:
			break
		c = nxt[c]

for i in xrange(n):
	if selected[i] == 0:
		print "No"
		break
else:
	print "Yes"
