#include<iostream>
#include<string>
using namespace std;
string s,p;
int n,m,i,j,k,x,y,L[200005],R[200005],a[200005][26],c[26];
bool T;
main()
{
    cin>>s>>p;
    s="="+s;
    p="+"+p;
    R[s.size()]=p.size();
    for (i=1;i<s.size();i++)
    L[i]=min((int)p.size()-1,(int)(s[i]==p[L[i-1]+1]?L[i-1]+1:L[i-1]));
    for (i=s.size()-1;i>=1;i--)
    R[i]=max(1,(int)(s[i]==p[R[i+1]-1]?R[i+1]-1:R[i+1]));
    for (i=1;i<p.size();i++)
    { a[i][p[i]-'a']=1;
      for (j=0;j<26;j++)
      a[i][j]+=a[i-1][j];}
    for (i=1;i<s.size();i++)
    {
      x=max(1,R[i+1]-1);
      y=min((int)p.size()-1,L[i-1]+1);
      if (x>y) T=1;
      if (a[y][s[i]-'a']-a[x-1][s[i]-'a']==0) T=1;
      }
      if (T) cout<<"No"<<endl; else cout<<"Yes"<<endl;
    }
