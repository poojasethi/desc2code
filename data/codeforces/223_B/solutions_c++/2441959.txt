#include <cstdio>
#include <iostream>
#include <cstring>

#define MAXN 222222

using namespace std;

char s[MAXN]="", t[MAXN]="";

int main()
{
	cin>>s>>t;
	int lens=strlen(s), lent=strlen(t), num[200];
	int left[MAXN], right[MAXN];
	memset(num, -1, sizeof(num));
	for(int i=0, j=0; i<lens; i++)
	{
		if(j<lent && s[i]==t[j]) { left[i]=num[s[i]]=j; j++; }
				  	 	 	else left[i]=num[s[i]];
	}
	memset(num, -1 , sizeof(num));
	for(int i=lens-1, j=lent-1; i>=0; i--)
	{
		if(j>=0 && s[i]==t[j]) { right[i]=num[s[i]]=j; j--; }
				  	 	  else right[i]=num[s[i]];
	}
	for(int i=0; i<lens; i++)
		if(left[i]==-1 || right[i]==-1 || left[i]<right[i]) { cout<<"NO"<<endl; return 0; }
	cout<<"YES"<<endl;
	return 0;
}
