#include<bits/stdc++.h>
using namespace std;
int pos[26],ls,lt,f[200005],b[200005];
char s[200005],t[200005];
int main(){
  cin.getline(s,200005);
  cin.getline(t,200005);
  ls=strlen(s);
  lt=strlen(t);
  int cp=-1;
  for(int i=0;i<26;i++) pos[i]=-1;
  for(int i=0;i<ls;i++){
    if(cp+1<lt && s[i]==t[cp+1]) pos[s[i]-'a']=f[i]=++cp;
    else f[i]=pos[s[i]-'a'];
  }
  cp=lt;
  for(int i=0;i<26;i++) pos[i]=lt;
  for(int i=ls-1;i>=0;i--){
    if(cp>0 && s[i]==t[cp-1]) pos[s[i]-'a']=b[i]=--cp;
    else b[i]=pos[s[i]-'a'];
  }
  for(int i=0;i<ls;i++){
    if(f[i]<b[i]){
      printf("No\n");
      return 0;
    }
  }
  printf("Yes\n");
  return 0;
}
