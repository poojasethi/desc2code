#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;
int L[200001],R[200001];
int C[27];
int main()
{
    string a,b;
    memset(C,-1,sizeof(C));
    cin>>a>>b;
    int j=0;
    for(int i=0;i<a.size();L[i]=C[a[i]-'a'],i++)
        if(a[i]==b[j]&&j<=b.size())
            C[a[i]-'a']=j++;
    memset(C,-1,sizeof(C));
    j=b.size()-1;
    for(int i=a.size()-1;i>=0;R[i]=C[a[i]-'a'],i--)
        if(a[i]==b[j]&&(j+1))
            C[a[i]-'a']=j--;
    for(int i=0;i<a.size();i++)
        if(R[i]==-1||L[i]==-1||L[i]<R[i])
            return puts("No"),0;
    puts("Yes");
    return 0;
}
