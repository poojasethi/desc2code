#include<iostream>
#include<cstdio>
#include<algorithm>
#include<string>
#define exits {printf("No\n");return 0;}
using namespace std;
const int N = 200005;
string S,T;
int occ[256];
int markL[N],markR[N];
int main()
{
    cin >> S >> T;
    int l,r;
    r = 0;
    for (l = 0;l<S.size();++l) if (r<T.size() && S[l] == T[r]) markL[l]++, ++r;
    if (r<T.size()) exits;
    r = T.size()-1;
    for ( l = S.size()-1;l>=0;--l) if (r>=0 && S[l] == T[r]) markR[l]-- , --r;
    for (int i = 0;i<S.size();++i)
    {
        occ[S[i]] += markL[i];
        if (occ[S[i]] == 0) exits;
        occ[S[i]] += markR[i];
    }
    printf("Yes\n");
    return 0;

}
