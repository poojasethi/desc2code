#include <cstdio>
#include <cstring>

char a[300000],b[300000];
int n,m,i,j,k,f[300000],g[300000],h[200];

int main()
{
    scanf("%s",a+1);
    scanf("%s",b+1);
    f[0]=0;
    j=1;
    n=strlen(a+1);
    m=strlen(b+1);
    memset(h,0,sizeof(h));
    for(i=1;i<=n;i++)
    {
        if(j<=m && a[i]==b[j])
        {
            h[a[i]]=j;
            j++;
        }
        f[i]=h[a[i]];
    }
    g[n+1]=0;
    j=m;
    memset(h,0,sizeof(h));
    for(i=n;i>0;i--)
    {
        if(j>0 && a[i]==b[j])
        {
            h[a[i]]=m-j+1;
            j--;
        }
        g[i]=h[a[i]];
    }
    for(i=1;i<=n;i++)
        if(f[i]+g[i]<=m)
        {
            puts("No");
            return 0;
        }
    puts("Yes");
    return 0;
}
