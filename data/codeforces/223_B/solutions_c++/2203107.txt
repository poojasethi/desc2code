#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int nMax = 200010;
char s[nMax],t[nMax];
int head[nMax],tail[nMax],last[30];
int main()
{
	int ls,lt,i,j,k;
	while(scanf("%s%s",s,t)!=EOF){
		ls=strlen(s);	lt=strlen(t);
		memset(last,-1,sizeof(last));
	for(i=0,k=0;i<ls;i++){
			if(s[i]==t[k])	k++;
			head[i]=k-1;
		}
		for(i=ls-1,k=lt-1;i>=0;i--){
			if(k>=0 && s[i]==t[k])	k--;
			tail[i]=k+1;
		}
		for(i=0;i<ls;i++){	
			if(head[i]>=0)	last[t[head[i]]-'a']=head[i];
			if(last[s[i]-'a']<0 || last[s[i]-'a']<tail[i] )	goto WA;
		}
		printf("Yes\n");	continue;
		WA:	printf("No\n");
	}
}
