#include<cstdio>
#include<cstring>
using namespace std;
#define M 200005
char s[M],t[M];
int l[M],r[M],num[35];
int main()
{
    int k;
    scanf("%s%s",s,t);
    int la=strlen(s),lb=strlen(t);
    memset(num,-1,sizeof(num));
    k=0;
    for(int i=0;i<la;++i)
    {
        if(s[i]==t[k]) ++k;
        l[i]=k-1;
    }
    k=lb-1;
    for(int i=la-1;i>=0;--i)
    {
        if(k>=0&&s[i]==t[k]) --k;
        r[i]=k+1;
    }
    bool flag=true;
    for(int i=0;i<la;++i)
    {
        if(l[i]>=0) num[t[l[i]]-'a']=l[i];
        if(num[s[i]-'a']<0||num[s[i]-'a']<r[i])
        {
            flag=false;
            break;
        }
    }
    printf("%s\n",flag?"Yes":"No");
    return 0;
}
