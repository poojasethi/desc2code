#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;
const int maxN=200000+10;

int n,m;
char s[maxN],t[maxN];
int f[maxN],last[256];

int main(){
    //freopen("b.inp","r",stdin);
    scanf("%s%s",s+1,t+1);
    n=strlen(s+1); m=strlen(t+1);
    int p=n;
    for (int i=m;i>=1;i--){
        while (p>0&&s[p]!=t[i]) p--;
        if (p==0){
            printf("No\n"); return 0;
        }
        f[i]=p; p--;
    }
    p=1;
    memset(last,0,sizeof last);
    for (int i=1;i<=n;i++){
        if (p<=m&&s[i]==t[p]){last[s[i]]=p; p++; continue;}
        if (last[s[i]]==0||f[last[s[i]]]<i){ printf("No\n"); return 0;}
    }
    printf("Yes\n");
    return 0;
}
