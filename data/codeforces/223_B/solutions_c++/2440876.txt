#include <iostream>
#include <cstdio>
#include <cstring>

#define Maxn 100000 << 1 | 1

using namespace std;

char str1[Maxn];
char str2[Maxn];
int ltr[Maxn],rtl[Maxn];
int num[27];

int main()
{
	cin>>str1>>str2;
	int k;
	int len1=strlen(str1);
	int len2=strlen(str2);
	k=0;
	memset(num,-1,sizeof(num));
	for(int i=0;i<len1;i++)
	{
		if(k<len2 && str1[i]==str2[k]) {k++;ltr[i]=num[str1[i]-'a']=k-1;}
		else ltr[i]=num[str1[i]-'a'];
	}
	k=len2-1;
	memset(num,-1,sizeof(num));
	for(int i=len1-1;i>=0;i--)
	{
		if(k>=0 && str1[i]==str2[k]) {k--;rtl[i]=num[str1[i]-'a']=k+1;}
		else rtl[i]=num[str1[i]-'a'];
	}
	bool flag=1;
	for(int i=0;i<len1;i++)
		if(ltr[i]==-1 || rtl[i]==-1 || ltr[i]<rtl[i]) {puts("No");return 0;}
	puts("Yes");
	return 0;
}
