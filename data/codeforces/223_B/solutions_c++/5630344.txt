#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#define N 300000
using namespace std;
int last[30],now,n,m,f1[N],f2[N];
char s[N],t[N];
int main(){
	scanf("%s%s",s,t);
	n=strlen(s);m=strlen(t);
	memset(last,-1,sizeof(last));
	for(int i=0;i<n;i++){
		if(s[i]==t[now])last[t[now]-'a']=f1[i]=now,now+=(now<n-1);
		else{
			f1[i]=last[s[i]-'a'];
		}
	}memset(last,-1,sizeof(last));
	now=m-1;
	for(int i=n-1;i>=0;i--){
		if(s[i]==t[now])last[t[now]-'a']=f2[i]=now,now-=(now>0);
		else{
			f2[i]=last[s[i]-'a'];
			if(f2[i]==-1)f2[i]=m;
		}
	}for(int i=0;i<n;i++){
		if(f1[i]-f2[i]<0)return puts("No"),0;
	}puts("Yes");
}
