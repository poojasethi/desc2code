#include<cstdio>
#include<cstdlib>
#include<cstring>
int l1,l2,pos[1001],front[200005],back[200005];
char a[200005],b[200005];
int main(){
	scanf("%s",a+1); l1=strlen(a+1);
	scanf("%s",b+1); l2=strlen(b+1);
	int j=1;
	for (int i=1;i<=255;i++) pos[i]=0; 
	for (int i=1;i<=l1;i++){
		if(a[i]==b[j]&&j<=l2){
			pos[a[i]]=j;
			j++; 
		}
		front[i]=pos[a[i]];
	}
	j=l2;
	for (int i=1;i<=255;i++) pos[i]=0;
	for (int i=l1;i>=1;i--){
		if(a[i]==b[j]&&j>=1){
			pos[a[i]]=l2-j+1;
			j--;
		}
		back[i]=pos[a[i]];
	}
	for (int i=1;i<=l1;i++)
	if(front[i]+back[i]<=l2){printf("No\n"); return 0;}
	printf("Yes\n");
}
