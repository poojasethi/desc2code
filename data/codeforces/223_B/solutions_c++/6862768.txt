#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 200010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f1[N],f2[N],a[N][26];
char p[N],s[N];
int main() {
	scanf("%s%s",p+1,s+1);
	n=strlen(p+1),m=strlen(s+1);
	For(i,1,m+1) {
		For(j,0,25) a[i][j]=a[i-1][j];
		if (i<=m) a[i][s[i]-'a']++;
	}
	For(i,1,n) {
		f1[i]=f1[i-1];
		if (p[i]==s[f1[i]+1]) f1[i]++;
	}
	f2[n+1]=m+1;
	for (i=n;i;i--) {
		f2[i]=f2[i+1];
		if (p[i]==s[f2[i]-1]) f2[i]--;
	}
	For(i,1,n) {
		int r=f1[i-1]+1,l=f2[i+1]-1,A=p[i]-'a';
		if (l) l--;
		if (r<l||!(a[r][A]-a[l][A])) {
			printf("No\n");
			return 0;
		}
	}
	printf("Yes\n");
	//for(;;);
	return 0;
}
