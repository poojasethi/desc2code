#include <iostream>
#include <string>
#include <cstring>
using namespace std;

string a,b;
int c[256];
int L[222222],R[222222];

int main()
{
	cin >> a >> b;
	memset(c,-1,sizeof(c));
	c[b[0]] = 0;
	for (int i = 0;i < a.length();++i)
	{
		L[i] = c[a[i]];
		if (c[a[i]] >= 0 && c[a[i]] < b.length() - 1)
		{
			c[b[c[a[i]]+1]] = max(c[b[c[a[i]] + 1]],c[a[i]]+1);
		}
	}
	memset(c,63,sizeof(c));
	c[b[b.length() -1]] = b.length() - 1;
	for (int i = a.length() - 1;i >= 0;--i)
	{
		R[i] = c[a[i]];
		if (c[a[i]] < b.length() && c[a[i]] > 0)
		{
			c[b[c[a[i]]-1]] = min(c[b[c[a[i]]-1]],c[a[i]]-1);
		}
	}
	for (int i = 0;i < a.length();++i)
	{
		if (L[i] < R[i])
		{
			cout << "No\n";
			return 0;
		}
	}
	cout << "Yes\n";
	return 0;
}
