#include <bits/stdc++.h>
using namespace std;
int main()
{
    int r=0,c=0,arr[26];
    char t[200001],s[200001];
    scanf("%s%s",t,s);
    int lent=strlen(t);
    int lens=strlen(s);
    for(int i=0;i<26;i++) arr[i]=-1;
    for (int i=0;i<lent;i++)
    {
        //printf("yes\n");
        if(t[i]==s[c])
        {
            //printf("in\n");
            c++;
            arr[t[i]-97]=c;
        }
        if(t[i]==s[r]) r++;
        if (arr[t[i]-97]<r)
        {
            //printf("less r\n");
            r=arr[t[i]-97];
        }
        if(r<0) break;
    }
    if (r<lens) printf("No\n");
    else printf("Yes\n");
    return 0;
}
