#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define N 220000
using namespace std;
char a[N],b[N];
int la,lb,f1[N],f2[N],h[300],i;
void work(int f[N]){
	int i,r=0;
	memset(h,0,sizeof(h));
	h[b[1]]=1;
	for(i=1;i<=la;++i){
		f[i]=h[a[i]];
		if(f[i]>r){
			r++;
			h[b[r+1]]=r+1;
		}
	}
}
int main(){
	scanf("%s%s",a+1,b+1);
	la=strlen(a+1); lb=strlen(b+1);
	work(f1);
	for(i=1;i+i<=lb;++i)swap(b[i],b[lb-i+1]);
	for(i=1;i+i<=la;++i)swap(a[i],a[la-i+1]);
	work(f2);
	for(i=1;i<=la;++i)if(f1[i]!=lb && f2[la-i+1]!=lb && f1[i]+f2[la-i+1]-1<lb)return puts("No");
	puts("Yes");
}
