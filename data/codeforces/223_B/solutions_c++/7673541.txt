#include <iostream>
#include <cstdio>
#include <cstring>
#define N 200020
using namespace std;
char s[N],t[N];
int l[N],r[N],v[27][N],flag=1;
int main()
{
    scanf("%s%s",s,t);
    int s_len=strlen(s),t_len=strlen(t);
    for(int i=1;i<=s_len;i++)l[i]=l[i-1]+(s[i-1]==t[l[i-1]]);
    for(int i=s_len;i>0;i--) r[i]=r[i+1]+(s[i-1]==t[t_len-1-r[i+1]]);
    for(int i=0;i<t_len;i++){
        for(int j=0;j<26;j++) v[j][i+1]=v[j][i];
        v[t[i]-'a'][i+1]=i+1;
    }
    for(int i=0;flag&&i<s_len;i++)
        if(v[s[i]-'a'][l[i+1]]+r[i+1]<=t_len) flag=0;
    cout<<(flag?"Yes":"No")<<endl;
    return 0;
}
