#include <bits/stdc++.h>
using namespace std;
string s,t;
int dp[30];
int main(){
	cin>>s>>t;
	memset(dp,-1,sizeof(dp));
	int i,l=0,ma=0;
	for(i=0;i<s.size();++i){
		if(s[i]==t[ma])
			dp[s[i]-'a']=++ma;
		if(s[i]==t[l])++l;

		if(dp[s[i]-'a']<l)
			l= dp[s[i]-'a'];
		if(l<0)break;
	}
	int lim = t.size();
	if(l<lim)
		cout<<"No\n";
	else
		cout<<"Yes\n";
}
