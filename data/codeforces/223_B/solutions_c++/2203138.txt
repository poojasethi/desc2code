#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int nMax = 200010;
char s[nMax],t[nMax];
int H[nMax],T[nMax],L[30];
int main()
{
	int ls,lt,i,j,k;
	scanf("%s%s",s,t);
	ls=strlen(s);	lt=strlen(t);
	memset(L,-1,sizeof(L));
	for(i=0,k=0;i<ls;i++){
		if(s[i]==t[k])	k++;
		H[i]=k-1;
	}
	for(i=ls-1,k=lt-1;i>=0;i--){
		if(k>=0 && s[i]==t[k])	k--;
		T[i]=k+1;
	}
	for(i=0;i<ls;i++){	
		if(H[i]>=0)	L[t[H[i]]-'a']=H[i];
		if(L[s[i]-'a']<0 || L[s[i]-'a']<T[i] )	goto WA;
	}
	printf("Yes\n");	return 0;
	WA:	printf("No\n");
}
