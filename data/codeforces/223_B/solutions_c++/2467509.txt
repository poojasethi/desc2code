#include <algorithm>
#include <cstring>

char base[200010];
char   fn[200010];
int   markp[200010],markm[200010];
int top[200010],bot[200010];
int occ[30];

int l1,l2;

int a,b;

int main()
{
    scanf("%s",base+1);
    scanf("%s",fn+1);
    l1=strlen(base+1);
    l2=strlen(fn+1);
    b=1;
    for (a=1; a<=l1; a++)
    if (base[a]==fn[b]) markp[a]++, b++;
    if (b<=l2)
    {
        printf("No");
        return 0;
    }

    b=l2;

    for (a=l1; a>=1; a--)
    if (base[a]==fn[b]) markm[a]--, b--;

    {
        occ[base[a]]+=markp[a];
        if (occ[base[a]]==0) {printf("No"); return 0; }
        occ[base[a]]+=markm[a];
    }

}
