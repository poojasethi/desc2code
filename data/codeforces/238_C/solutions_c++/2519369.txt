#include<cstdio>
#include<algorithm>
#define maxe 10000
using namespace std;
int p[maxe],e[maxe],ne[maxe],ec,n,isv[maxe];
bool fx[maxe][maxe];
int s,t;

void adde(int a,int b){
    e[++ec]=b;ne[ec]=p[a];p[a]=ec;
    e[++ec]=a;ne[ec]=p[b];p[b]=ec;
    fx[a][b]=1;
}

void dfs(int a,int z){
    isv[a]=1;
    t=max(t,z);
    for(int i=p[a];i;i=ne[i])
        if(!isv[e[i]]){
            s+=fx[e[i]][a];
            dfs(e[i],max(0,z+(fx[e[i]][a]?1:-1)));
        }
}

int main(){
    int i,j,ans=1000000,a,b;
    scanf("%d",&n);
    for(i=1;i<n;i++){
        scanf("%d%d",&a,&b);
        adde(a,b);
    }
    for(i=1;i<=n;i++){
        s=t=0;
        for(j=1;j<=n;j++)
            isv[j]=0;
        dfs(i,0);
        ans=min(ans,s-t);
    }
    printf("%d\n",ans);
}
