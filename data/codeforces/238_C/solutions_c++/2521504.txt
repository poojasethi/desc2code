#include<cstdio>
#define min(a,b) ((a)<(b)?(a):(b))
#define max(a,b) ((a)>(b)?(a):(b))
#define maxz 10000
int p[maxz],e[maxz],ne[maxz],ef[maxz],ec,n,cl[maxz],nowcl,s,t;
void adde(int a,int b){
        e[++ec]=b;ne[ec]=p[a];p[a]=ec;
        e[++ec]=a;ne[ec]=p[b];p[b]=ec;ef[ec]=1;
}
void dfs(int a,int nowpath){
        cl[a]=nowcl;
        t=max(nowpath,t);
        for(int i=p[a];i;i=ne[i]){
                if(cl[e[i]]!=nowcl){
                        s+=ef[i];
                        dfs(e[i],max(0,nowpath+(ef[i]?1:-1)));
                }
        }
}
int main(){
        int i,a,b,ans=9999;
        scanf("%d",&n);
        for(i=1;i<n;i++){
                scanf("%d%d",&a,&b);
                adde(a,b);
        }
        for(i=1;i<=n;i++){
                nowcl++;
                s=t=0;
                dfs(i,0);
                ans=min(ans,s-t);
        }
        printf("%d\n",ans);
}
