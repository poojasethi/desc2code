#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int head[10000], nxt[10000], vv[10000], st[10000];
int e;
void add( int u, int v, int w )
{
	vv[e] = v, st[e] = w;
	nxt[e] = head[u], head[u] = e++;
}

int sum, ans;
void dfs1( int root, int fa )
{
	for( int i = head[root]; i >= 0; i = nxt[i] )	if( vv[i] != fa )
	{
		sum += st[i];
		dfs1(vv[i], root);
	}
}

void dfs2( int root, int minv, int fa )
{	
	ans = min(ans, sum+minv);
	for( int i = head[root]; i >= 0; i = nxt[i] )	if( vv[i] != fa )
	{
		if( st[i] == 0 )
			dfs2(vv[i], min(0, minv+1), root);
		else
			dfs2(vv[i], minv-1, root);
	}
}

int main()
{
	int i, j, k, n, u, v;

	while( scanf("%d", &n) != EOF )
	{
		e = 0;
		memset(head, -1, sizeof(head));
		for( i = 1; i < n; ++i )
		{
			scanf("%d %d", &u, &v);
			add(u, v, 0);
			add(v, u, 1);
		}

		ans = n;
		for( i = 1; i <= n; ++i )
		{
			sum = 0;
			dfs1(i, -1);
			dfs2(i, 0, -1);
		}
		printf("%d\n", ans);
	}

	return 0;
}
