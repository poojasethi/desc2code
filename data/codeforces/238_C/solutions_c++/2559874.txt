#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n;
int h[3010],to[6010],nxt[6010],len[6010],index;
void init(){
    index=0;
    memset(h,-1,sizeof(h));
}
void addedge(int u,int v,int l){
    to[index]=v;
    len[index]=l;
    nxt[index]=h[u];
    h[u]=index++;
}
int x,y;
void dfs(int t,int p,int m){
    y=max(y,m);
    int k=h[t];
    while(k+1){
        int v=to[k];
        if(v==p){
            k=nxt[k];
            continue;
        }
        x+=len[k];
        dfs(v,t,max(0,m+2*len[k]-1));
        k=nxt[k];
    }
}
int main(){
    init();
    scanf("%d",&n);
    for(int i=1;i<n;i++){
        int a,b;
        scanf("%d%d",&a,&b);
        addedge(a,b,0);
        addedge(b,a,1);
    }
    int ans=999999999;
    for(int i=1;i<=n;i++){
        x=y=0;
        dfs(i,0,0);
        ans=min(ans,x-y);
    }
    printf("%d\n",ans);
}
