#include <bits/stdc++.h>

#define mp make_pair
#define pb push_back

using namespace std;

const int maxl = 3 * 1000 + 5;

vector < pair < int, bool > > v [ maxl ];
int n;

int dis [ maxl ], dp [ maxl ];
int ans;

int dfs ( int index, int par ){
	int res = 0;
	for ( int i = 0 ; i < v [ index ].size () ; i++ ){
		int node = v [ index ][ i ].first;
		bool dir = v [ index ][ i ].second;

		if ( node != par ){
			dis [ node ] = dis [ index ] + ((index < node)^(dir));
			dp [ node ] = min ( dis [ node ], dp [ index ] + ((node < index)^(dir)) );

			res += dfs ( node, index ) + ((index < node)^(dir));
		}
	}

	return res;
}

int main (){
	cin >> n;
	for ( int i = 1 ; i < n ; i++ ){
		int a, b;
		cin >> a >> b;
		a--; b--;

		v [ a ].pb ( mp ( b, a < b ) );
		v [ b ].pb ( mp ( a, a < b ) );
	}

	ans = n;
	for ( int i = 0 ; i < n ; i++ ){
		dis [ i ] = 0;
		dp [ i ] = 0;
		int tmp = dfs ( i, -1 );

		for ( int j = 0 ; j < n ; j++ )
			ans = min ( ans, tmp - dis [ j ] + dp [ j ] );
	}

	cout << ans << endl;
}
