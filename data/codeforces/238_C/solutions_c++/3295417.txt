#include <cstdio>
#include <cstring>
#include <iostream>
#include <vector>

#define fi first
#define se second
#define pb push_back
#define sz(a) (int)a.size()

using namespace std;

typedef pair<int,int> ii;

const int MAXN = 3003;

int N,mini=1e9,maxi,sum;

vector<ii> ways[MAXN];

void dfs( int nd, int dad, int cost )
{
	maxi = max( maxi, cost );
	
	for( int i=0 ; i<sz(ways[nd]) ; i++ )
		if( ways[nd][i].fi != dad ){
			sum += ways[nd][i].se;
			dfs( ways[nd][i].fi, nd, max(0,cost+2*ways[nd][i].se-1) );
		}
}

int main()
{
	scanf(" %d",&N);
	
	for( int i=1,a,b ; i<N ; i++ ){
		scanf(" %d %d",&a,&b);
		ways[a].pb(ii(b,0));
		ways[b].pb(ii(a,1));
	}
	
	for( int i=1 ; i<=N ; i++ ){
		sum=maxi=0;
		dfs( i, 0, 0 );
		mini = min( mini, sum-maxi );
	}
		
	printf("%d\n",mini);
	
	return 0;
}
