#include <cstdio>
#include <iostream>
using namespace std;

int l[3001][3001];
bool f[3001][3001];
int x,y;

void dfs(int t,int last,int m)
{
   y=max(y,m);
   for (int i=1;i<=l[t][0];i++)
      if (l[t][i]!=last)
      {
         x+=f[t][i];
         dfs(l[t][i],t,max(0,m+2*f[t][i]-1));
      }
}

int main()
{
   int n,a,b;
   cin>>n;
   for (int i=1;i<n;i++)
   {
      int a,b;
      scanf("%d%d",&a,&b);
      l[a][++l[a][0]]=b;
      l[b][++l[b][0]]=a;
      f[b][l[b][0]]=1;
   }
   int t=n;
   for (int i=1;i<=n;i++)
   {
      x=y=0;
      dfs(i,0,0);
      t=min(t,x-y);
   }
   cout<<t<<endl;
   return 0;
}
