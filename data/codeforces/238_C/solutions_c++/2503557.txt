#include<cstdio>
#include<cstdlib>
#include<cstring>
typedef long long int64;
struct da{int d,n;}b[6000];
int a[3005],f[3005],n,i,now,ans=~0U>>1,ed,s,t;
void link(int s,int t){b[++ed]=(da){t,a[s]};a[s]=ed;}
void dfs(int x,int p)
{
     for (int i=a[x];i;i=b[i].n)
       if (b[i].d!=p)dfs(b[i].d,x),f[x]+=f[b[i].d]+(~i&1);
 }
void dfs2(int x,int p,int ss)
{
     if (ss<0)ss=0;
     if (ans>now-ss)ans=now-ss;
     for (int i=a[x];i;i=b[i].n)
       if (b[i].d!=p)
         if (i&1)dfs2(b[i].d,x,ss-1);
         else dfs2(b[i].d,x,ss+1);
 }
void cal(int x,int p)
{
     dfs2(x,0,0);
     for (int i=a[x];i;i=b[i].n)
       if (b[i].d!=p)
         if (i&1)now++,cal(b[i].d,x),now--;
         else now--,cal(b[i].d,x),now++;
 }
int main()
{
    scanf("%d",&n);
    for (i=2;i<=n;i++)scanf("%d%d",&s,&t),link(s,t),link(t,s);
    dfs(1,0);now=f[1];
    cal(1,0);
    printf("%d",ans);
    return 0;
}
