#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<cstring>
#include<vector>
#define rep(i,s,t) for (int i=s;i<=t;i++)
using namespace std;
const int maxn = 3001;
vector <int> g[maxn];
int x,y,ans;
int f[maxn][maxn];
int n;
void dfs(int k,int last,int m)
{
    y = max(y,m);
    for (int i = 0;i < g[k].size();i++)
    if (g[k][i] != last)
    {
        if (f[k][g[k][i]]) x++;
        dfs(g[k][i],k,max(0,m + 2 * f[k][g[k][i]] - 1));
    }
}
int main()
{
    scanf("%d",&n);
    rep(i,1,n-1)
    {
        scanf("%d%d",&x,&y);
        g[x].push_back(y);
        g[y].push_back(x);
        f[y][x] = 1;
    }
    ans = n;
    rep(i,1,n)
    {
        x = y = 0;
        dfs(i,0,0);
        ans = min(ans,x-y);
    }
    printf("%d\n",ans);
    fclose(stdin);
    return 0;
}
