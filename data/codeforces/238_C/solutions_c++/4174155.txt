#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include <time.h>
#include <vector>
#include <set>
#include <map>
#include <algorithm>
#include <queue>
using namespace std;
vector < pair<int,int> > g[3010];
int Ans,Minn;
void dfs(int u,int pu,int tot,int pre)
{
    Minn=min(Minn,tot-pre);
    for (int i=0;i<g[u].size();i++)
    {
        int v=g[u][i].first;
        int w=g[u][i].second;
        if (v==pu) continue;
        if (w==1) Ans++;
        dfs(v,u,tot+(w==0?1:-1),max(pre,tot+(w==0?1:-1)));
    }
}
int main()
{
    int n;
    scanf("%d",&n);
    for (int i=0;i<n-1;i++)
    {
        int u,v;
        scanf("%d%d",&u,&v);
        g[u].push_back(make_pair(v,0));
        g[v].push_back(make_pair(u,1));
    }
    int anss=n+1;
    for (int i=1;i<=n;i++)
    {
        Ans=0;
        Minn=0;
        dfs(i,-1,0,0);
        anss=min(anss,Ans+Minn);
    }
    printf("%d\n",anss);
    return 0;
}
