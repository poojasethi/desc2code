#include <algorithm>
#include <stdio.h>
#include <vector>
#define  fi        first
#define  se        second
#define  maxn      3002
#define  inf       1 << 30
#define  pb        push_back
using    namespace std;
typedef  pair<int,int> pi;

int n,m,c1,c2;
int sz[maxn];
vector<pi>adj[maxn];

void dfs(int node , int bef , int c)
{
  c1=max(c1,c);
  for(int i=0 ; i<sz[node] ; i++)
    if(adj[node][i].fi!=bef)
    {
      int cc=max(0,c+2*adj[node][i].se-1);
      c2+=adj[node][i].se;
      dfs(adj[node][i].fi,node,cc);
    }
}

int main()
{
  scanf("%d",&n);
  m=n-1;
  while(m--)
  {
    int x,y;
    scanf("%d%d",&x,&y);
    adj[x].pb(pi(y,0));
    adj[y].pb(pi(x,1));
    sz[x]++,sz[y]++;
  }
  int ans=inf;
  for(int i=1 ; i<=n ; i++)
  {
    c1=c2=0;
    dfs(i,-1,0);
    //printf("%d %d\n",c1,c2);
    ans=min(ans,c2-c1);
  }
  printf("%d\n",ans);
  return 0;
}
