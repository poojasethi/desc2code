#include <cstdio>
#include <vector>
#include <iostream>

#define fi first
#define se second
using namespace std;

const int MAXN=3010;
typedef pair <int,int> ii;

int N,cnt,tmp,res=0x7fffffff;
vector <ii> ways[MAXN];

void f(int node,int dad,int rev){

	tmp=max(tmp,rev);

	for(int i=0;i<(int)ways[node].size();i++)
		if(ways[node][i].fi!=dad){
			cnt+=ways[node][i].se;
			f(ways[node][i].fi,node,max(0,rev+2*ways[node][i].se-1));
		}
}

int main(){

	scanf(" %d",&N);
	
	for(int i=1,a,b;i<N;i++){
		scanf(" %d %d",&a,&b);
		ways[a].push_back(ii(b,0));
		ways[b].push_back(ii(a,1));
	}
	
	for(int i=1;i<=N;i++){
		tmp=0;	cnt=0;
		f(i,0,0);
		res=min(res,cnt-tmp);
	}
	
	cout << res << endl;
	
	return 0;
}
