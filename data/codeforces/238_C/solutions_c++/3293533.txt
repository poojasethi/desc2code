#include <cstdio>
#include <iostream>
#include <vector>

#define fi first
#define se second
#define pb push_back

using namespace std;

typedef pair<int,int> ii;

const int MAXN=3003;

int N,res=1e8,cnt,mx;

vector<ii> way[MAXN];

void dfs( int nd , int dad , int cur ){
	
	mx=max(mx,cur);
	
	for( vector<ii>::iterator it=way[nd].begin() ; it!=way[nd].end() ; it++ )
		if( it->fi!=dad ){
			cnt+=it->se;
			dfs(it->fi,nd,max(0,cur+2*(it->se)-1));
		}
}

int main(){
	
	scanf(" %d",&N);
	
	for( int a,b,i=1 ; i<N ; i++ ){
		scanf(" %d %d",&a,&b);
		way[a].pb(ii(b,0));
		way[b].pb(ii(a,1));
	}
	
	for( int i=1 ; i<=N ; i++ ){
		cnt=0,mx=0;
		dfs(i,0,0);
		res=min(res,cnt-mx);
	}
	
	printf("%d\n",res);
	return 0;
}
