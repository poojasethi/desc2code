using namespace std;
int n,ans,k[2];
vector<pair<int,bool> >E[3000];
void dfs1(int x,int f){
	for(int i=-1;++i!=E[x].size();)if(E[x][i].first!=f){
		k[E[x][i].second]++;
		dfs1(E[x][i].first,x);
	}
}
int dfs2(int x,int f){
	int re=0;
	for(int i=-1;++i!=E[x].size();)if(E[x][i].first!=f){
		re=min(re,dfs2(E[x][i].first,x)+(E[x][i].second?-1:1));
	}
	ans=min(ans,k[1]+re);
	return re;
}
int main(){
	scanf("%d",&n);ans=n-1;
	for(int i=n;--i;){
		int x,y;
		scanf("%d%d",&x,&y);
		x--;y--;
		E[x].push_back(make_pair(y,0));
		E[y].push_back(make_pair(x,1));
	}
	for(int i=-1;++i!=n;){
		k[0]=k[1]=0;
		dfs1(i,-1);
		ans=min(ans,k[1]);
		dfs2(i,-1);
	}
	printf("%d\n",ans);
	return 0;
}
