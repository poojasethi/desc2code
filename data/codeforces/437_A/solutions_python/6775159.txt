fun=lambda:len(raw_input())-2
data=[fun() for i in xrange(4)]
ans=['A','B','C','D']
flag=0
index=0
for i in xrange(4):
    tmp1=0
    tmp2=0
    for j in xrange(4):
        if i==j:
            continue
        if data[i] * 2 <= data[j]:
            tmp1+=1
        if data[i] >= data[j] * 2:
            tmp2+=1
    if tmp1 >= 3 or tmp2 >= 3:
        index=i
        if flag:
            flag=2
        else:
            flag=1
if flag==2 or flag==0:
    print 'C'
else:
    print ans[index]
