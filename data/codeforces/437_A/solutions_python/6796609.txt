nums = map(lambda x: len(raw_input().strip())-2, range(4))
res = []

for e1, num1 in enumerate(nums):
    twice_shorter = twice_longer = True
    for e2, num2 in enumerate(nums):
        if e1 != e2:
            if num1 < 2*num2:
                twice_longer = False
            if (2*num1) > num2:
                twice_shorter = False
    if twice_shorter or twice_longer:
        res.append(e1)

if len(res) == 1:
    print 'ABCD'[res[0]]
else:
    print 'C'
