import sys

lines = sys.stdin.readlines()
lines = map(lambda x: x.strip(), lines)

lengths = []
mapping = {}
for line in lines:
    a, b = line.split('.')
    lengths.append(len(b))
    mapping[len(b)] = a
    
lengths.sort()

a = lengths[0] * 2 <= lengths[1]
b = lengths[-1] >= lengths[-2] * 2

if a and not b:
    print(mapping[lengths[0]])
elif b and not a:
    print(mapping[lengths[-1]])
else:
    print('C')
