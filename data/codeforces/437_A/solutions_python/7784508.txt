l = []

for i in range(4):
    s = raw_input()
    l.append((len(s[2:]), s[0]))

l.sort(key=lambda x : x[0])

choice = None

if 2*l[0][0] <= l[1][0]:
    choice = l[0][1]

if 2*l[-2][0] <= l[-1][0]:
    if choice is None:
        choice = l[-1][1]
    else:
        choice = None

if choice is None:
    choice = "C"

print choice
