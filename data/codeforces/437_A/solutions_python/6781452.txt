s = [[] for i in range(4)]
for i in range(4):
    s[i] = raw_input()
l = []
pos = f = 0
for i in range(4):
    l.append(len(s[i])-2)
for i in range(4):
    c1 = c2 = 0
    for j in range(4):
        if j != i and l[j] >= l[i]*2:
            c1 += 1
        if j != i and l[j]*2 <= l[i]:
            c2 += 1
    if c1 == 3 or c2 == 3:
        f += 1
        pos = i
if f == 1:
    print chr(65+pos)
else:
    print 'C'
