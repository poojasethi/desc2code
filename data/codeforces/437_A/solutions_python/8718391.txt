l=[]
ans=[1]*4
for i in range(4):
    s=raw_input()
    chk=len(s)-2
#    chk-=s.count('_')
    l.append(chk)
a=max(l)
i=min(l)
for x in range(4):
    if l[x]!=a and l[x]!=i:
        ans[x]=0
    else:
        for y in range(4):
            if x==y:
                pass
            elif l[x]==a and l[y]*2>a:
                ans[x]=0
            elif l[x]==i and l[y]<i*2:
                ans[x]=0
print 'C' if ans.count(1)!=1 else chr(ans.index(1)+65)
