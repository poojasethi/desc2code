#!/usr/bin/python

def read_one():
    s = raw_input()
    return s[2:]

s = [read_one(), read_one(), read_one(), read_one()]
s = [len(e) for e in s]

r = 'ABCD'
s = [(e, c) for e, c in zip(s, r)]

s.sort()


if 2*s[0][0]<=s[1][0]:
    zero_good = True
else:
    zero_good = False

if s[3][0]>=2*s[2][0]:
    last_good = True
else:
    last_good = False


if zero_good and last_good:
    print 'C'
elif zero_good:
    print s[0][1]
elif last_good:
    print s[3][1]
else:
    print 'C'
