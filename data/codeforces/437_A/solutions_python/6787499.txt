choice=[0,0,0,0]
string=' '
ans=65
Max,Min=0,1000
maxi,mini=0,0
for i in range(4) :
	string=raw_input();
	choice[i]=len(string)-2
	if choice[i]>Max :
		Max=choice[i]
		maxi=i
	if choice[i]<Min :
		Min=choice[i]
		mini=i

l,s=True,True
for i in range(4) :
	if maxi!=i and (choice[maxi]/choice[i])<2:
		l=False
	if mini!=i and (choice[i]/choice[mini])<2:
		s=False
if l and not s:
	print chr(maxi+ans)
elif s and not l:
	print chr(mini+ans)
else:
	print "C"
