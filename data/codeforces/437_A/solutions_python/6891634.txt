R = lambda:raw_input()[2:]
A = [R(), R(), R(), R()]
dbl, hlf, grt = [ True ] * 4, [ True ] * 4, []
for i in xrange(4):
	ilen = len(A[i])
	for j in xrange(4):
		if i != j:
			jlen = len(A[j])
			if jlen * 2 > ilen:
				dbl[i] = False
			if ilen * 2 > jlen:
				hlf[i] = False
	if dbl[i] or hlf[i]:
		grt.append(i)
if len(grt) == 1:
	print unichr(grt[0] + ord('A'))
else:
	print "C"
