def solve(q):
  l = [len(x) for x in q]
  g = []
  for i in range(4):
    f = True
    for j in range(4):
      if i == j: continue
      if l[i]*2 > l[j]: f = False
    if f:
      g.append(i)
      continue
    f = True
    for j in range(4):
      if i == j: continue
      if l[i] < l[j]*2: f = False
    if f:
      g.append(i)
      continue
  if len(g) == 1: return g[0]
  return 2

q = [raw_input()[2:] for i in range(4)]
print chr(ord('A')+solve(q))
