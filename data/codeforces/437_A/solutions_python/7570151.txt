a = []
for i in range(4):
    a.append(len(raw_input()) - 2)
b = ['A','B','C','D']
ans = []
for i in range(4):
    ok,ok1 = 0,0
    for j in range(4):
        if j != i:
            if a[i] >= 2*a[j]:
                ok += 1
            elif 2*a[i] <= a[j]:
                ok1 += 1
    if ok == 3 or ok1 == 3:
        ans.append(b[i])
print (ans[0] if len(ans) == 1 else 'C')
