op=[]
ans=['A','B','C','D']
for i in range(4):
    op.append(raw_input())
length=[]
for s in op:
    length.append(len(s)-2)
m=max(length)
M=length.index(m)
n=min(length)
N=length.index(n)
c=0
d=0
for i in range(4):
    if i!=M and length[i]<=m/2:
        c+=1
for i in range(4):
    if i!=N and length[i]>=2*n:
            d+=1
if  d==3 and c!=3:
        print str(ans[N])
elif c==3 and d!=3:
    print str(ans[M])
else:
        print 'C'
