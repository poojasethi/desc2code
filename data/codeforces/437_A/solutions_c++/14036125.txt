#include <bits/stdc++.h>
using namespace std;
string s[4];
bool cmp(string s1,string s2)
{
    return s1.length()<s2.length();
}
int main()
{
    for(int i=0;i<4;++i)
        cin>>s[i];
    sort(s,s+4,cmp);
    bool f1=((s[0].length()-2)*2<=s[1].length()-2);
    bool f2=((s[3].length()-2)>=(s[2].length()-2)*2);
    if(f1&&f2)
        return puts("C");
    if(!f1&&!f2)
        return puts("C");
    if(f1)
        return cout<<s[0][0],0;
    cout<<s[3][0],0;
    return 0;
}
