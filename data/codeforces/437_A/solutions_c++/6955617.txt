#include<stdio.h>
#include<string.h>
int main()
{
	char a[4][103];
	int b[4],i,max=0,min=101,m=0,n=0,mi,ni;
	for(i=0;i<4;i++)
		gets(a[i]);
	for(i=0;i<4;i++)
	{
		b[i]=strlen(a[i])-2;
		if(b[i]>max)
		{
			max=b[i];
			mi=i;
		}	
		if(b[i]<min)
		{
			min=b[i];
			ni=i;
		}	
	}
	for(i=0;i<4;i++)
	{
		if(max>=b[i]*2)
			m++;
		if(min*2<=b[i])
			n++;
	}
	if(m==3&n!=3)
		printf("%c\n",a[mi][0]);
	else if(m!=3&&n==3)
		printf("%c\n",a[ni][0]);
	else
		printf("C\n");
	return 0;
}
