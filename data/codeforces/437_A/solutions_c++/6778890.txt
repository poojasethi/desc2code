#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

int main() {
    int b[5], l[5], length=2;
    char tmp[200];
    for(int i=0; i<4; i++) {
        cin>>tmp;
        b[i] = l[i] = strlen(tmp) - 2;
    }

    sort(l, l+4);
    if((l[3]>=2*l[2])^(2*l[0]<=l[1])) {
        length=(2*l[0]<=l[1])?l[0]:l[3];
        for(int i=0; i<4; i++)
            if(b[i] == length) cout<<(char)('A'+i)<<endl;
    } else {
        cout<<"C"<<endl;
    }
}
