#include <bits/stdc++.h>
using namespace std;

pair<int,string> p[5];

int main () {

    string s;
    for (int i=0; i<4; i++) {
        cin >> s;
        p[i] = make_pair(s.size()-2, s);
    }

    sort(p,p+4);

    bool ok1=0,ok2=0;
    if ( p[0].first*2<=p[1].first ) ok1=1;
    if ( p[2].first*2<=p[3].first ) ok2=1;

    if (ok1 && !ok2) cout << p[0].second[0] << endl;
    else if (!ok1 && ok2) cout << p[3].second[0] << endl;
    else cout << "C" << endl;
}
