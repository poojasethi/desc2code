#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<set>
using namespace std;
const int maxn=500;
char s[5][maxn];
int a[5];
set<int>st;
int main()
{
	for (int i=1;i<=4;i++) scanf("%s",s[i]);
	for (int i=1;i<=4;i++) a[i]=strlen(s[i])-2;
	sort(a+1,a+5);
	for (int i=1;i<=4;i++)
	{
		if (strlen(s[i])-2<=a[2]/2) st.insert(i);
		if (strlen(s[i])-2>=a[3]*2) st.insert(i); 
	}
	if (st.size()==1)
	{
		set<int>::iterator p=st.begin();
		printf("%c",'A'+(*p)-1);
	}
	else printf("C\n");
	return 0;
}
