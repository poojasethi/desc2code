#include<cstdio>
#include<cstring>
char s[5][200],ans;
int l[5],a[5],b[5],anss;
int main()
{
	for(int i=1;i<=4;i++)
		scanf("%s",s[i]),l[i]=strlen(s[i])-2;
	for(int i=1;i<=4;i++){
		int a=1,b=1;
		for(int j=1;j<=4;j++)
			if(j!=i)
				a&=(l[j]>=l[i]*2),b&=(l[i]>=l[j]*2);
		if(a||b)anss++,ans='A'+i-1;
	}
	if(anss!=1)ans='C';
	printf("%c\n",ans);
	return 0;
}
