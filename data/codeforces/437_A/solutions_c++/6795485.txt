#include <algorithm>
#include <iostream>
#include <vector>
#include <cstdio>
using namespace std;

int main() {
	vector<pair<int, int> > a;
	for(int i = 0; i < 4; ++i) {
		string s; cin >> s;
		a.push_back(make_pair((int) s.size() - 2, i));
	}
	sort(a.begin(), a.end());
	bool x = a[0].first * 2 <= a[1].first;
	bool y = a[2].first * 2 <= a[3].first;
	putchar(x == y ? 'C' : x ? a[0].second + 'A' : a[3].second + 'A');
	return 0;
}
