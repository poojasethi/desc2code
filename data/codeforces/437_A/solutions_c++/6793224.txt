#include<iostream>
#include<string>
using namespace std;

int main(){
	int v[4],m,count=0,value=0;
	string str[4];
	for(int i=0;i<4;i++){
		cin >> str[i];
		v[i]=str[i].size()-2;
	}
	for(m=0;m<4;m++){
		int big=0,small=0;
		for(int n=0;n<4;n++){
			if(v[m]>=2*v[n])
				big++;
			else if(2*v[m]<=v[n])
				small++;
		}
		if(big==3 || small==3){
			count++;
			value=m;
		}
	}
	if(count==1)
		cout << (char)('A'+ value) <<endl;
	else
		cout << 'C' <<endl;
	return 0;
}
