#include <bits/stdc++.h>
using namespace std;

string s[4];
int a[4];
vector<char> v;

int main(){
cin >> s[0] >> s[1] >> s[2] >> s[3];
for(int i = 0; i < 4; i++) a[i] = s[i].size() - 2;
for(int i = 0; i < 4; i++){
int x = 1, y = 1;
for(int j = 0; j < 4; j++){
if(i != j){
x &= a[i] >= 2 * a[j], y &= 2 * a[i] <= a[j];
}
}
if(x || y) v.push_back('A' + i);
}
if(v.size() == 1) cout << v[0];
else cout << 'C';
}
