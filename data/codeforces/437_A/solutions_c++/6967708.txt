#include<iostream>
#define FOR(i,n) for(int i=0;i<n;i++)
#define SZ size()
using namespace std;
int main()
{
  string s[4],f="ABCD",ans; int n[4],c=0;
  FOR(i,4){cin>>s[i]; n[i]=s[i].SZ-2;}
  FOR(i,4)
  { int g=0,l=0;
    FOR(j,4)
    { if(i!=j)
      { if(n[i]>=2*n[j])g++;
        if(n[i]<=n[j]/2)l++;
      }
    }
    if(g==3||l==3){c++; ans=f[i];}
  }
  if(c==1)cout<<ans;
  else cout<<'C';
}
