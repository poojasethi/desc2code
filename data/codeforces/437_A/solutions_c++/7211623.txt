#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

pair<int,char> a[4];
bool ans[2];

int main()
{
	for(int i=0;i<4;i++)
	{
		string s;
		cin>>s;
		a[i]=make_pair(s.size()-2,s[0]);
	}
	sort(a,a+4);
	if(a[0].first*2<=a[1].first)
		ans[0]=true;
	if(a[3].first>=2*a[2].first)
		ans[1]=true;
	if((ans[0] && ans[1]) || (!ans[0] && !ans[1]))
		cout<<"C"<<endl;
	else if(ans[0])
		cout<<a[0].second<<endl;
	else
		cout<<a[3].second<<endl;
	return 0;
}
