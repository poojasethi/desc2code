#include<bits/stdc++.h>
using namespace std;
#define ff first
#define ss second

string s;
pair<int, int> l[4];

int main()
{
	for(int i = 0; i < 4; ++ i)
	{
		cin >> s;
		l[i] = make_pair(s.size() - 2, i);
	}
	sort(l, l + 4);
	int cnt = 0, ans;
	if(l[0].ff * 2 <= l[1].ff)
	{
		cnt ++;
		ans = l[0].ss;
	}
	if(l[2].ff * 2 <= l[3].ff)
	{
		cnt ++;
		ans = l[3].ss;
	}
	if(cnt == 1)
		cout << char('A' + ans) << endl;
	else
		cout << 'C' << endl;
	return 0;
}
