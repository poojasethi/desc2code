#include <bits/stdc++.h>

using namespace std;

pair<int,char> a[4];
bool ans1,ans2;
int main()
{
	for(int i=0;i<4;i++)
	{
		string s;
		cin>>s;
		a[i]=make_pair(s.size()-2,s[0]);
	}
	sort(a,a+4);

	if(a[0].first*2<=a[1].first)
		ans1=1;
	if(a[3].first>=2*a[2].first)
		ans2=1;
	if((ans1 && ans2) || (!ans1 && !ans2))
		cout<<"C"<<endl;
	else if(ans1)
		cout<<a[0].second<<endl;
	else
		cout<<a[3].second<<endl;
	return 0;
}
