#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <cstdio>

using namespace std;

int main()
{
	string s;
	vector<pair<int,string> > v;
	for(int i=0;i<4;i++){
		cin>>s;
		v.push_back(make_pair(s.length()-2,s));

	}
	sort(v.begin(),v.end());
	int odp=-10;
	if(v[0].first*2<=v[1].first) odp+=1;
	if(v[3].first>=2*v[2].first) odp-=1;
	if(odp==-11) cout<<v[3].second[0]<<endl;
	if(odp==-9) cout<<v[0].second[0]<<endl;
	if(odp==-10) cout<<"C\n";
	return 0;
}
