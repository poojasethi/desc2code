#include <bits/stdc++.h>
using namespace std;


int main(){
	//freopen("input.inp","r",stdin);
	string A;
	pair<int,int> a[4];
	for(int i=0;i<4;i++){
		cin >> A;
		a[i].first=A.size()-2;
		a[i].second=i;
	}
	sort(a,a+4);
	if(a[0].first*2<=a[1].first&&a[3].first<2*a[2].first) cout << char( a[0].second+'A');
	else if(a[3].first>=2*a[2].first&&a[0].first*2>a[1].first) cout << char( a[3].second+'A');
	else cout << "C";
}
