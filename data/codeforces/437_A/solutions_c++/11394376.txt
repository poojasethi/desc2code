#include <bits/stdc++.h>
using namespace std;

#define fr(i,n,m) for (int i = (n) ; i < (m) ; ++ i)

pair <int,int> a[4];
bool f1 , f2;

int main(){
	fr(i,0,4){
		string s;
		cin >> s;
		a[i].first = s.size() - 2;
		a[i].second = i;
	}
	sort(a,a+4);
	if (a[0].first * 2 <= a[1].first) f1 = 1;
	if (a[3].first >= a[2].first * 2) f2 = 1;
	if (f1 && !f2) return putchar('A'+a[0].second) , 0;
	if (!f1 && f2) return putchar('A'+a[3].second) , 0;
	return putchar('C') , 0;
}
