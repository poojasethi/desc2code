#include<bits/stdc++.h>
using namespace std;
#define f first
#define s second

string s;
pair<int, int> l[4];

int main(){
	for(int i = 0; i < 4; ++ i){
		cin >> s;
		l[i] = make_pair(s.size() - 2, i);
	}
	sort(l, l + 4);
	int cnt = 0, ans;
	if(l[0].f * 2 <= l[1].f){
		cnt ++;
		ans = l[0].s;
	}
	if(l[2].f * 2 <= l[3].f) { 
		cnt ++;
		ans = l[3].s;
	}
	if(cnt == 1)
		cout << char('A' + ans) << endl;
	else
		cout << 'C' << endl;
	return 0;
}
