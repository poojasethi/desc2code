#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;
int main() {
pair<int,int> a[4];
char s[1000];
int i;
for(i=0;i<4;i++)
{
cin>>s;
int l=strlen(s);
a[i]=make_pair(l-2,i);
}   
sort(a,a+4);
if(((a[0].first*2)<=a[1].first) &&!(a[3].first>=2*a[2].first))cout<<(char)('A'+a[0].second)<<endl;
else if((a[3].first>=2*a[2].first) && !((a[0].first*2)<=a[1].first))cout<<(char)('A'+a[3].second)<<endl;
else cout<<"C"<<endl;
return 0;
}
