#include <bits/stdc++.h>

using namespace std;
struct node
{
    int l;
    char c;
}a[4];
int cmp(node a,node b)
{
    return a.l<b.l;
}
int main()
{
    string s[4];
    int i;
    for (i=0;i<4;i++) {cin>>s[i];a[i].c=s[i][0];a[i].l=s[i].size()-2;}
    sort(a,a+4,cmp);
    if (a[0].l*2<=a[1].l&&a[3].l<a[2].l*2) cout<<a[0].c<<endl;
    else if (a[3].l>=a[2].l*2&&a[0].l*2>a[1].l) cout<<a[3].c<<endl;
    else cout<<"C"<<endl;
    return 0;
}
