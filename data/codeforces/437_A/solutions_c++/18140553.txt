#include <cstdio>
#include <cstring>
char s[1024];
int l[4],sel;
int main()
{
	sel=-1;
	for(int i=0;i<4;++i){gets(s);l[i]=strlen(s+2);}
	for(int i=0;i<4;++i)
	{
		int lt=1,gt=1;
		for(int j=0;j<4;++j)if(i-j)//l[i]>=2*l[j]//l[i]<=l[j]/2
		{if(l[i]<2*l[j])gt=0;if(l[j]/2<l[i])lt=0;}
		if(lt||gt){if(~sel){sel=-1;break;}else sel=i;}
	}
	putchar(~sel?sel+'A':'C');
	return 0;
}
