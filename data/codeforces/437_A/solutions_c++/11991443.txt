#include <bits/stdc++.h>
using namespace std;
int main(){
	int V[4],p;
	int sw=0;
	string CAD;
	char SOL[4]={'A','B','C','D'};
	for (int i = 0; i < 4; ++i){
		cin>>CAD;
		V[i]=CAD.length()-2;
	}
	
	for (int i = 0; i < 4; ++i){
		int m=1, n=1;
		for (int j = 0; j < 4; ++j){
			if(i!=j){
				m&=V[j] >= V[i]*2; 
				n&=V[i] >= V[j]*2;
			}
		}
		if(m || n)
			sw++, p=i;
	}
	if(sw!=1)
		cout<<"C"<<endl;
	else
		cout<<SOL[p]<<endl;

return 0;	
}
