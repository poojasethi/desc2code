# include <iostream>
# include <cstring>
# include <algorithm>
# define nmax 100005
# define fi cin
# define fo cout
using namespace std;
long long s[nmax];
int main(void)
{
    pair <int,int> s[7];
    char c[1000];
    for ( int i = 1 ; i <= 4 ; ++i ) fi>>c,s[i].first=strlen(c)-2,s[i].second=i-1;
    sort(s+1,s+5);
    bool a = ( s[1].first*2 <= s[2].first );
    bool b = ( s[3].first*2 <= s[4].first );
    fo<<char(a==b ? 'C':(a ? s[1].second+'A':s[4].second+'A') )<<"\n";
}
