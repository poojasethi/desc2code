#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxl=120;
int n=4,l[4],ans,anss=0;
char ch[4][maxl];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	for(i=0;i<n;i++)
		gets(ch[i]),l[i]=strlen(ch[i])-2;
	for(i=0;i<n;i++)
	{
		bool z1=1,z2=1;
		for(j=0;j<n;j++)
			if(j!=i)
				z1&=(l[j]>=l[i]*2),z2&=(l[i]>=l[j]*2);
		if(z1||z2)
			anss++,ans=i;
	}
	if(anss!=1) ans=2;
	printf("%c\n",ans+'A');
	return 0;
}
