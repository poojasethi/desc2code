#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int main() {
    string s;
    int len[4], ori[4];

    for (int i = 0; i < 4; ++i) {
        cin >> s;
        ori[i] = len[i] = s.length() - 2;
    }
    sort(len, len + 4);
    int ans=-1;
    if (len[0]*2<=len[1] ^ len[3]>=2*len[2])
        ans = len[0]*2<=len[1] ? len[0] : len[3];
    if (ans==-1)
        cout<<'C'<<endl;
    else
        cout<<char('A'+find(ori,ori+4,ans)-ori)<<endl;
    return 0;
}
