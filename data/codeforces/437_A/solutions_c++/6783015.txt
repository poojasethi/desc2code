#include<iostream>
using namespace std;
int main ()
{
  char n,a[4]={'A','B','C','D'};
  string s[4];  int co=0,c=0;
  for(int i=0;i<4;i++)cin>>s[i];

  for(int i=0;i<4;i++)
  { co=0;
    for(int j=0;j<4;j++)
    { if(i!=j)
      { if(s[i].size()-2>=(s[j].size()-2)*2)co++;
        if(s[i].size()-2<=(s[j].size()-2)/2)co--;
      }
    }
    if(co==3||co==-3){
        c++; n=a[i];
    }
  }
  if(c==1)cout<<n;
  else cout<<'C';
 return 0;
}
