#include <stdio.h>
int n,m,v;
int init()
{
	scanf("%d%d%d",&n,&m,&v);
	return 0;
}
int solve()
{
	int i,j,edge=1,end;
	if(m<n-1||m>(n*n-3*n+4)/2)
	{
		printf("-1");
		return 0;
	}
	if(v==n)
	{
		end=n-1;
		printf("%d %d\n",end,v);
	}
	else
	{
		end=v+1;
		printf("%d %d\n",v,end);
	}
	for(i=1;i<n;i++)
	{
		for(j=i+1;j<=n;j++)
		{
			if(i!=end&&j!=end&&edge!=m)
			{
				edge++;
				printf("%d %d\n",i,j);
			}
			else
			{
				if(edge==m)
				return 0;
			}
		}
	}
	return 0;
}
int main()
{
	init();
	solve();
	return 0;
}
