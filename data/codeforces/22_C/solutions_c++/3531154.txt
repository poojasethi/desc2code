#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

typedef long long LL;
LL n, m;
int v;

void deal(int s, int t, int edge)
{
	for(int k=1; edge; k++)
		for(int i=s; i+k<=t && edge; i++)
		{
			printf("%d %d\n", i, i+k);
			edge--;
		}
}

int main()
{
	scanf("%I64d%I64d%d", &n, &m, &v);

	if(m>(2-n+(n*n-n)/2) || m<n-1)
	{
		printf("-1\n");
		return 0;
	}
	if(v==1)
	{
		printf("%d %I64d\n", v, n);
		deal(1, n-1, m-1);
	}
	else
	{
		printf("%d %d\n", 1, v);
		deal(2, n, m-1);
	}
	return 0;
}
