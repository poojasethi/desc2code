#include <stdio.h>

using namespace std;

int main(){
	int n, m, v;
	scanf("%d %d %d", &n, &m, &v);
	if(m > ((long long)(n - 1)) * (n - 2) / 2 + 1 || m < n - 1) printf("-1");
	else{
		int is = (v % n) + 1;
		printf("%d %d\n", is, v); --m;
		for(int i = 1; i < n; i++)if(i != is)
			for(int j = i + 1; j <= n; j++)if(j != is)
			{
				printf("%d %d\n", i, j);
				if(!(--m)) return 0;
			}
	}
	return 0;
}
