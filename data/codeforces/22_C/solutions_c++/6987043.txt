#include <cstdio>
#include <cstring>
#include <vector>

using namespace std;

int n, m, v;

int main() {
	scanf("%d%d%d", &n, &m, &v);
	long long limit = (n - 1) * (n - 2) / 2 + 1;
	if (m > limit || m < n - 1) {
		puts("-1");
		return 0;
	}
	int end = v == n ? n - 1 : n;
	for (int u = 1; u <= n; ++u) {
		if (m <= 1) break;
		if (u == end) continue;
		for (int vv = u + 1; vv <= n; ++vv) {
			if (m <= 1) break;
			if (vv == end) continue;
			printf("%d %d\n", u, vv);
			--m;
		}
	}
	if (m == 1) printf("%d %d\n", end, v);
	return 0;
}
