#include<stdio.h>

int main(int agrc,char*argv[]){
  int N, M, V; scanf("%d %d %d", &N, &M, &V);
  if(M< N-1 || M > 1 + ((N-1)*(N-2))/2){
    printf("-1");
    return 0;
  }
  // star connection...
  for(int i=1; i<=N && (M!=0); i++)
    if(i!=V){
      printf("%d %d\n", V, i);
      M--;
    }
  int ncon[2] = {1, V};
  if(V==1) ncon[0] = 2;
  for(int j=1; j<=N && M!=0; j++)
    for(int k=j+1; k<=N && M!=0; k++)
      if(M!=0 && j!=ncon[1] && j!=ncon[0] && k!=ncon[1] && k!=ncon[0]){
        printf("%d %d\n", k, j);
        M--; 
      }
  return 0;
}
