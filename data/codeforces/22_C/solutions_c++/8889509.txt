# include <stdio.h>
# include <iostream>
using namespace std;

int main(){
	int n,m,v;

	cin>>n>>m>>v;

	int i,j;
	int edge=1;
	int end;

	if(m<n-1||m>(n*n-3*n+4)/2){
		cout<<"-1"<<" "<<"\n";
		return 0;
	}
	if(v==n){
		end=n-1;
		cout<<end<<" "<<v<<"\n";
	}
	else{
		end=v+1;
		cout<<v<<" "<<end<<"\n";
	}
	for(i=1;i<n;i++){
		for(j=i+1;j<=n;j++){
			if(i!=end&&j!=end&&edge!=m){
				edge++;
				cout<<i<<" "<<j<<"\n";
			}
			else{
				if(edge==m)
				return 0;
			}
		}
	}
	return 0;
}
