#include <cstdio>
using namespace std;
int n, m, v, bad;
int main()
{
	scanf("%d%d%d",&n,&m,&v);
	if (m<n-1 || m>((long long)(n-2)*(n-1)/2+1))
	{
		printf("-1\n");
		return 0;
	}
	int alone=(v%n)+1;
	printf("%d %d\n",v,alone);
	for (int i=1;i<=n;i++)
		if (i!=alone)
			for (int j=i+1;j<=n;j++)
				if (j!=alone)
				{
					if (--m==0) return 0;
					printf("%d %d\n",i,j);
				}
	return 0;
}
