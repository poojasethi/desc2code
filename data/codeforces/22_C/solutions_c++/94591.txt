#include <iostream>

using namespace std;

int main ()
{
	int n, m, v;
	cin >> n >> m >> v;
	if(m < n - 1 || m > (n * n - 3 * n + 4) / 2)
		cout << -1 << endl;
	else
	{
		if(v != n)
		{
			cout << n << ' ' << v << endl;
			m--;
				
			for(int i = 1; i < n && m; i++)
				for(int j = i + 1; j < n && m; j++)
				{
					cout << i << ' ' << j << endl;
					m--;
				}
		}
		else
		{
			cout << 1 << ' ' << v << endl;
			m--;
				
			for(int i = 2; i <= n && m; i++)
				for(int j = i + 1; j <= n && m; j++)
				{
					cout << i << ' ' << j << endl;
					m--;
				}
		
		}
	}
}
