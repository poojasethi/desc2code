#include<iostream>
#include<cstring>
using namespace std;
const int mm=100010;
int n,m,dex;
int main()
{
    while(cin>>n>>m>>dex)
    {
        if(m<n-1||m>((n-1)*(n-2))/2+1)
        {
            cout<<-1<<"\n";continue;
        }
        int bad;
        for(int i=1;i<=n;i++)
        if(i!=dex)
        {bad=i;cout<<i<<" "<<dex<<"\n";m--;}

        for(int i=1;i<=n&&m;i++)
        for(int j=i+1;j<=n&&m;j++)
        {
            if(i!=dex&&j!=dex&&m&&i!=bad&&j!=bad)
            cout<<i<<" "<<j<<"\n",m--;
        }
    }
}
