#include<cstdio>
typedef long long ll;
int main(){
	int n,m,v,h,x;
	scanf("%d%d%d",&n,&m,&v);
	if(m>n-1+ll(n-2)*(n-3)/2||m<n-1){
		puts("-1");
		return 0;
	}
	if(v==1)
		h=2;
	else
		h=1;
	printf("%d %d\n",h,v);
	for(int i=1;i<=n;i++){
		if(i!=v&&i!=h)
			printf("%d %d\n",v,i);
	}
	m-=n-1;
	if(m==0)
		return 0;
	for(int i=1;i<n;i++)
		for(int j=i+1;j<=n;j++){
			if(i!=h&&i!=v&&j!=h&&j!=v){
				printf("%d %d\n",i,j);
				m--;
				if(m==0)
					return 0;
			}
		}
	return 0;
}
