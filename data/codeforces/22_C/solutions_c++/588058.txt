#include <stdio.h>

int id[100005];
int main(void) {
	int i, j, n, m, v;
	scanf("%d%d%d", &n, &m, &v);
	if(m>1LL*(n-1)*(n-2)/2+1 || m<n-1) puts("-1");
	else{
		for(i=1;i<=n;i++)
			id[i] = i;
		id[1] = v;
		id[v] = 1;
		for(i=2;i<n;i++)
			printf("%d %d\n", id[i-1], id[i]);
		printf("%d %d\n", id[n], id[1]);
		m -= n-1;
		for(i=1;i<n && m>0;i++)
			for(j=i+2;j<n && m>0;j++, --m)
				printf("%d %d\n", id[i], id[j]);
	}
	return 0;
}
