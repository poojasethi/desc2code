#include <iostream>
#include <stdio.h>

using namespace std;

int n,m,v,i,j;
main()
 {
  cin>>n>>m>>v;
  if (((n-1)*(n-2))/2+1 < m || m < n-1)
   { cout<<-1<<endl; return 0; }
  for (i=1;i<=n;i++)
   if (i != v)
    cout<<i<<" "<<v<<endl;
  m-=n-1;
  for (i=3;i<=n;i++)
   if (i != v && m)
    for (j=i+1;j<=n;j++)
     if (j != v && m)
      cout<<i<<" "<<j<<endl,m--;
  for (j=3;j<=n;j++)
   if (m && j != v) cout<<2<<" "<<j<<endl,m--;
 }
