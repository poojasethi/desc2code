#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,v,d,x[100001];
int main(){
  scanf("%d%d%d",&n,&m,&v);
  if(m < n-1 || m > ((n-2)*(n-3))/2 + n - 1) printf("-1\n");
  else{
    for(int i=1;i<=n;i++) x[i] = i;
    if(v != 1) x[1] = 1, x[2] = v, x[v] = 2;
    else x[1] = 2, x[2] = 1;
    for(int i=1;i<n && m > 0;i++,m--) printf("%d %d\n",x[i],x[i+1]);
    for(int i=2;i<=n && m > 0;i++)
      for(int j=2;j<i-1 && m > 0;j++)
        printf("%d %d\n",x[i],x[j]), m--;
  }
  return 0;
}
