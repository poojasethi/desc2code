#include <cstdio>
using namespace std;

int n, m, v, u;

int main() {
	scanf("%d%d%d", &n, &m, &v);
	u = v==1? 2: 1;
	if (n-1 <= m && m <= n-1 + (n-2)*(n-3)/2) {
		for (int i=1; i<=n; ++i) {
			if (i == v) continue;
			printf("%d %d\n", v, i);
			m--;
		}
		for (int i=1; i<=n && m>0; ++i) {
			for (int j=i+1; j<=n && m>0; ++j) {
				if (i == u || i == v || j == u || j == v) continue;
				printf("%d %d\n", i, j);
				m--;
			}
		}
	}
	else printf("-1\n");
	return 0;
}
