#include<cstdio>
int main(){
    int m, n, v;
    scanf("%d%d%d", &n, &m, &v);
    if (m > (long long)(n - 2) * (n - 1) / 2 + 1 || m < n - 1)
        printf("-1\n");
    else
    {
        int a = v % n + 1;
        printf("%d %d\n", v, a);
        for (int i = 1; i <= n; i++)
        if (i != a)
            for (int j = i + 1; j <= n; j++)
                if (j != a){
                    if (--m == 0) return 0;
                    printf("%d %d\n", i, j);
                }
    }
}
