#include <cstdio>

using namespace std;

int main()
{
    int n,m,v;
    scanf("%d%d%d",&n,&m,&v);
    int spec=1;
    if(m<n-1||m>(n-1)*(n-2)/2+1)
    {
        printf("-1\n");
        return 0;
    }
    if(spec==v)spec++;
    printf("%d %d\n",v,spec);
    m--;
    for(int i=1;i<=n;i++)
    {
        if(i==spec)continue;
        for(int j=i+1;j<=n;j++)
        {
            if(j==spec)continue;
            printf("%d %d\n",i,j);
            m--;
            if(m==0)return 0;
        }
    }
    return 0;
}
