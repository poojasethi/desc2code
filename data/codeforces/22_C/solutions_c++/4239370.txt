#include<stdio.h>
int main()
{
	int n,m,v;
	scanf("%d%d%d",&n,&m,&v);
	int flag=0;
	int keep;
	int i,j;
	int save;
	if((m < n-1) || (m > ((((n-1)*(n-2))/2)+1)))
	{
		printf("-1\n");
	}
	else
	{
		if(v==1)
		{
			keep=2;
		}
		else
		{
			keep=1;
		}
		printf("%d %d\n",keep,v);


			
		for(i=1;i<=n;i++)
		{
			if(i==keep)
			{
				continue;
			}
			for(j=i+1;j<=n;j++)
			{
				if(j==keep)
				{
					continue;
				}
				printf("%d %d\n",i,j);
				m--;
				if(m==1)
				{
					flag=1;
					break;
				}
			}
			if(flag==1)
			{
				break;
			}
		}
	}

}
