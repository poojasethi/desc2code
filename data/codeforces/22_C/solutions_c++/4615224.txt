#include <iostream>
#include <cstdio>

using namespace std;

int main(){
  int n, m, v;
  cin >> n >> m >> v;
  if(n-1>m || ((n-1)*(n-2))/2 + 1 < m){cout << "-1" << endl; return 0;}
  int w = v%n + 1;
  for(int i=1; i<=n ; i++)
    if(i!=v){
      printf("%d %d\n", v, i);
      m--;
    }
  for(int i=1; i<=n && m>0; i++)
    for(int j=i+1; j<=n && m>0; j++)
      if(i!=v && i!=w && j!=v && j!=w){
        printf("%d %d\n", i, j);
        m--;
      }
  return 0;
}
