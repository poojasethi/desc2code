#include <iostream>

using namespace std;

int main() {
  int n, m, v;
  cin >> n >> m >> v;
  int max_con = (n-1) * (n-2) / 2 + 1;
  if (m < n - 1 || m > max_con) {
    cout << -1 << endl;
    return 0;
  }

  int alone = v + 1;
  if (alone == n + 1) {
    alone = 1;
  }
  cout << v << " " << alone << endl;

  int i = 1, j, step = 1;
  int edges = 1;
  while (edges < m) {
    j = i + step;
    cout << (i==alone?n:i) << " " << (j==alone?n:j) << endl;
    edges++;

    i++;
    if (i + step > n - 1) {
      i = 1;
      step++;
    }
  }

  return 0;
}
