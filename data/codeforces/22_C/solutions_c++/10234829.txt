#include<bits/stdc++.h>
using namespace std;
int n,m,v;
int main()
{
    cin>>n>>m>>v;
    if(m<n-1||(n-2)*(n-3)/2+n-1<m)return puts("-1");
    int cnt=0;
    int k=1;
    if(v==k)k++;
    cout<<v<<' '<<k<<endl;cnt++;
    for(int i=1;i<=n;++i)
        if(i!=v&&i!=k&&cnt<m)
            cout<<i<<' '<<v<<endl,cnt++;
    if(cnt==m)
        goto S;
    for(int i=k+1;i<=n;++i)
    for(int j=i+1;j<=n;++j)
        if(i!=v&&j!=v)
        {
            cout<<i<<' '<<j<<endl;
            cnt++;
            if(cnt==m)
                goto S;
        }
S:
    return 0;
}
