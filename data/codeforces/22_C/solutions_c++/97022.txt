#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
int n, m, k;
int main()
{
    cin >> n >> m >> k;
    if(m < n - 1 || ((long long)(n-1))*(n-2)/2 < m - 1) {
        cout << -1;
        return 0;
    }
    int bad = k % n + 1;
    printf("%d %d\n", k, bad);
    for(int i = 1; i <= n; i++)
        for(int j = i + 1; j <= n; j++)
            if(i != bad && j != bad) {
                if(!(--m))return 0;
                printf("%d %d\n", i, j);
            }
    return 0;
}
