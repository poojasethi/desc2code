#include<iostream>
#include<algorithm>
using namespace std;

int main(){
	int n, m, v;
	cin >> n >> m >> v;
	if(n == 1){
		if(m)
			cout << "-1" << endl;
		return 0;
	}
	if(m < n-1 || m > ((n-1)*(n-2))/2+1){
		cout << "-1" << endl;
		return 0;
	}
	m -= n-1;
	for(int i = 1; i <= n; i++)
		if(i != v)
			cout << i << " " << v << endl;
	if(v == 1)
		v = 2;
	for(int i = 2; i <= n && m > 0; i++)
		for(int j = i+1; j <= n && m > 0 ; j++)
			if(i != v && j != v){
				m--;
				cout << i << " " << j << endl;
			}
	return 0;
}
