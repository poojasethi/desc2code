#include<stdio.h>
int ch[1333133];
main()
{
    int v,e,tg,i,j,co=0,m;
    scanf("%d %d %d",&v,&e,&tg);
    if(e<v-1)printf("-1\n");
    else if(e>((v-1)*(v-2))/2+1)printf("-1\n");
    else
    {
        co++;
        m=v;
        if(tg==v)m--;
        printf("%d %d\n",tg,m);
        for(i=1;i<=v;i++)
        {
            if(i==m)continue;
            for(j=i+1;j<=v;j++)
            {
                if(j==m)continue;
                co++;
                if(co>e)break;
                printf("%d %d\n",i,j);
            }
        }
    }
}
