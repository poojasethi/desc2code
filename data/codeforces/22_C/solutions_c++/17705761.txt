#include <bits/stdc++.h>
using namespace std;

int n, k, m, v;

inline int chg(int i) {return (i + v - 3 + n) % n + 1;}

int main() {
//	freopen("c.in", "r", stdin);
	scanf("%d %d %d\n", &n, &m, &v); 
	if (m < n - 1 || m > 1ll * (n - 1) * (n - 2) / 2 + 1) {puts("-1"); return 0;}
	for (int i = 1; i < n; i ++) printf("%d %d\n", chg(i), chg(i + 1));
	m -= n - 1;
	for (int i = 2; i < n; i ++) {
		if (!m) break;
		for (int j = i + 2; j <= n; j ++) {
			printf("%d %d\n", chg(i), chg(j));
			m --;
			if (!m) break;
		}
	}
	return 0;
}
