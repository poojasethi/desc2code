#include <cstdio>
#include <algorithm>
using namespace std;
int n,m,v,k,k1;
int main()
{
	scanf("%d%d%d",&n,&m,&v);
	if ((n-1)*(n-2)/2+1<m || (m<n-1))printf("-1");else
	{
		k=v+1;k1=0;
		if (k>n)k-=n;
		for(int i=1;i<=n;i++,k1++)
			if (i!=v)printf("%d %d\n",v,i);
		for(int i=1;i<=n && k1<=m;i++)
			for(int j=i+1;j<=n && k1<=m;j++)
				if (i!=v && i!=k && j!= v && j!=k){printf("%d %d\n",i,j);k1++;}	
	}	
	return 0;
}
