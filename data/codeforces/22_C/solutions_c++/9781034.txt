#include<iostream>

using namespace std;

int main()
{
	int n, m, k;
	cin >> n >> m >> k;
	if(m < n - 1 || m > (n - 2) * (n - 3) / 2 + n - 1)
	{
		cout << -1 << endl;
		return 0;
	}
	int t = n;
	if(k == t)
		t--;
	for(int i = 1; i <= n; i++)
		if(i != k)
		cout << i << ' ' << k << endl;
	m -= n - 1;
	for(int i = n; i > 0; i--)
		if(i != k && i != t)
		for(int j = 1; j < i && m > 0; j++)
			if(j != k && j != t)
			{
				cout << i << ' ' << j << endl;
				m--;
			}
	return 0;
}
