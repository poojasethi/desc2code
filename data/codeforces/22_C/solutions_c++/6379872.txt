#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
int main()
{
	int n,m,v;
	while(~scanf("%d%d%d",&n,&m,&v))
	{
		long long temp = (n-1)*(n-2)/2 + 1;  
		if(m>=(n-1)&&m<=temp){ 
			int bu=v==n?n-1:n;
			printf("%d %d\n",v,bu);
			m--;
			for(int i=1;i<n&&m;i++)
			{
				if(i==bu)continue;
				for(int j=i+1;j<=n&&m;j++)
				{
					if(j==bu)continue;
					cout<<i<<' '<<j<<endl;
					m--;
				}
			}
		}
		else puts("-1");
	}
}
