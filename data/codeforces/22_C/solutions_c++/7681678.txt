#include <stdio.h>
#include <string.h>

int main() {
 int n, m, p;
 int i, j, k, l, x, y;
 scanf("%d%d%d", &n, &m, &p);

 long long t = n - 1;
 t *= n - 2;
 t >>= 1;
 if (m < n - 1 || m > t + 1) return puts("-1"), 0;

 if (p == n) {
  printf("1 %d\n", p);
  i = 2; j = 3; ++n;
 } else {
  printf("%d %d\n", n, p);
  i = 1; j = 2;
 }
 while (--m) {
  printf("%d %d\n", i, j);
  if (++j >= n) j = ++i + 1;
 }
 return 0;
}
