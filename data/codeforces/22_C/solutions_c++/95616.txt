#include <iostream>

using namespace std;

int main(){
	int n, m, v;
	
	cin >> n >> m >> v;
	
	if ( m<n-1 || m>(n-1)*(n-2)/2+1 ) cout << -1 << endl;
	else{
		int ff = v + 1;
		if ( ff > n ) ff -= n;
		for (int i=1; i<=n; i++)
			if ( i != v )
				cout << i << " " << v << endl;
		
		m -= n - 1;
		for (int i=1; i<=n && m; i++)
			for (int j=i+1; j<=n && m; j++)
				if ( i!=v && i!=ff && j!=ff && j!=v ){
					cout << i << " " << j << endl;
					m--;
				}
	}

	return 0;
}
