#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int m,n,k;
    scanf("%d%d%d",&n,&m,&k);
    int poc;
    if(k==1) poc=2;
    else poc=1;
    if(m<n-1||m>(n-1)*(n-2)/2+1){printf("-1\n");return 0;}
    printf("%d %d\n",poc,k);
    m--;
    while(true)
    {
        for(int i=1;i<=n;i++)
            for(int j=i+1;j<=n;j++)
                if(i!=poc&&j!=poc)
        {
            if(m==0) return 0;
            //printf("M je :%d\n ",m);
            m--;
            printf("%d %d\n",i,j);

        }
    }

    return 0;
}
