//C
#include <stdio.h>

int n, m, v;

int f(int i){
    return (i + v - 3 + n) % n + 1;
}

void add(int i, int j){
    printf("%d %d\n", f(i), f(j));
    m --;
}

int main(){
    scanf("%d %d %d", &n, &m, &v);
    long long t = (1ll * n * n - 3 * n + 4) / 2;
    if(1ll * m > t || m < n - 1) {puts("-1"); return 0;}
    for(int i = 1; i < n; i ++) add(i, i + 1);
    for(int i = 2; i <= n; i ++) for(int j = i + 2; j <= n; j ++) {
        if(!m) return 0;
        add(i, j);
    }
}
