#include <iostream>
#include <cstdio>
using namespace std;
#define ll long long
ll n, m, v, p;
int main()
{
  cin >> n >> m >> v;
  if (m < n - 1 || m > (n - 1) * (n - 2) / 2 + 1) return cout << -1, 0;
  p = v % n + 1;
  cout << v << " " << p << endl;
  for(int i = 1; i <= n; i++)
    if (i != v && i != p)
      cout << v << " " << i << endl;
  m -= n - 1;
  for(int i = 1; m && i <= n; i++)
    if (i != v && i != p)
      for(int j = i + 1; m && j <= n; j++)
        if (j != v && j != p)
          cout << i << " " << j << endl, m--;
  return 0;
}
