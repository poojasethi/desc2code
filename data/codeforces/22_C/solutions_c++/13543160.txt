#include<bits/stdc++.h>

using namespace std;

int n,w,i,j,u,v;

int main(){

	cin >> n >> w >> u;

	v = (u == 1 ? 2 : 1); 

	if(w < n-1 || w > (n-1) * (n-2) / 2 + 1){
		cout << -1;
	    return 0;
	}

	for(i=1;i<=n;i++)
		if(i != u)
			w--, cout << u << " " << i << endl;

	for(i=2;i<=n;i++)
		for(j=i+1;j<=n;j++){
			if(!w) break;
			if(i != u && i != v && j != u && j != v)
				w--,cout << i << " " << j << endl;
		}

	return 0;
}
