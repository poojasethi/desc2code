#include <iostream>
#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl
using namespace std;
int n,m,v,t;

int main(){

  cin>>n>>m>>v;
  if(m<n-1||m-n+1>(n-2)*(n-3)/2){pr("-1");return 0;  }
  rep(i,n){
    if(m==0){return 0;}
    if(i+1!=v){pr(v<<" "<<i+1);m--;}
  }

  if(v==n)t=n-1;else t=n;

  rep(i,n)for(int j=i+1;j<n;j++){
    if(m==0){return 0;}
    if(i+1!=v&&j+1!=v&&i+1!=t&&j+1!=t){
      pr(i+1<<" "<<j+1);m--;
    }
  }
  return 0;
}
