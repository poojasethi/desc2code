#include<iostream>
using namespace std;
typedef long long ll;

int main()
{

ll n,m,v;

cin>>n>>m>>v;

if( m < n-1 || m > (n-2)*(n-1)/2+1 )
{ cout<<"-1\n";
  return 0;
}

ll nx=v%n+1;

cout<<v<<" "<<nx<<endl;
m--;


for(int i=1;i<=n;i++)
{ if( i==nx )
  continue;

 for(int j=i+1;j<=n;j++)
   {  
        if( j==nx )
        continue;

    if(m==0)
      return 0;

     cout<<i<<" "<<j<<endl;
  
      m--;
    }

}

return 0;
}
