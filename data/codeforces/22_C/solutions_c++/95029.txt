#include <iostream>
using namespace std;
int main()
{
	int n, m, v;
	cin >> n >> m  >> v;
	if (m < n - 1 || m > (n - 2) * (n - 1) / 2 + 1)
	{		cout << -1 << endl;
			return 0;
	}
	for (int i = 1; i <= n; i++)
		if (i != v)
			cout << v << ' ' << i << endl;
	m -= n - 1;
	int x = 2, y = 3;
	if (v == 1)
	{
		x++;
		y++;
	}
	while(m > 0)
	{
		if (x != v && y != v)
		{
			cout << x << ' ' << y << endl;
			m--;
		}
		y++;
		if (y > n)
		{
			x++;
			if (x == v)
				x++;
			y = x + 1;
		}
	}
	return 0;
}
