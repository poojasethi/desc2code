#include <bits/stdc++.h>
using namespace std;

int main(){
	int n, m, v, x, y;

	// freopen("test.in", "rt", stdin);
	cin >> n >> m >> v;
	if (m > (n-1)*(n-2)/2+1){
		cout << -1;
		return 0;
	}
	if (m < n-1){
		cout << -1;
		return 0;
	}
	if (v == 1){
		cout << "1 " << n << endl;
		x = 1; y = 2;
		n--;
	}
	else{
		cout << 1 << " " << v << endl;
		x = 2; y = 3;
	}
	for (int i = 1; i < m; i++){
		cout << x << " " << y << endl;
		if (y == n){
			x++;
			y = x + 1;
		}
		else y++;
	}
	return 0;
}
