#include <iostream>
#include <vector>
#include <cmath>
using namespace std;
long long f(int n)
{
	return (n*(n-1))/2;
}
int main()
{
	int n,m,v;
	cin>>n>>m>>v;
	if(m < n-1 || m > f(n-2)+n-1)
	cout<<-1<<endl;
	else
	{
		int cnt=0;
		for(int i=1;i<=n;i++)
		if(i!=v)
		{
			m--;
			cout<<i<<" "<<v<<endl;
		}
		int p=0;
		if(v!=n)
		{
			p=n;
		}
		else
		p=n-1;
		if(m)
		for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
		{
			if(m && i!=v && v!=j && i!=p && j!=p)
			{
				cout<<i<<" "<<j<<endl;
				m--;
				if(!m)
				{
					i=n;j=n;
				}
			}
		}
	}
	return 0;
}
