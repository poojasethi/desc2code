from sets import Set

def main():
    n, m, v = [int(value) for value in raw_input().split()]
    if m < n - 1:
        print -1
        return;

    combSet = Set()

    def addToSet(x, y):
        if x < y:
            combSet.add(str(x) + ' ' + str(y))
        else:
            combSet.add(str(y) + ' ' + str(x))
            
    
    excludedElement = 1 if v != 1 else 2

    def isInvalid(i):
        return i == excludedElement or i == v or i > n

    addToSet(excludedElement, v)
    
    for i in range(1, n + 1):
        if not isInvalid(i):
            addToSet(v, i)

    # Minimum  Reached!

    currentIndex = 0
    while len(combSet) < m and currentIndex <= n:
        currentIndex += 1
        if isInvalid(currentIndex):
            continue
        
        nextIndex = currentIndex
        while len(combSet) < m and nextIndex <= n:
            nextIndex += 1
            if not isInvalid(nextIndex):
                addToSet(currentIndex, nextIndex)

    if currentIndex > n and len(combSet) != m:
        print -1
        return
    
    for x in combSet:
        print x

main()
