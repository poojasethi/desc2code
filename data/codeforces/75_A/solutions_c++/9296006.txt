#include <stdio.h>
#include <iostream>
using namespace std;

int a,b;

int f(int n){
  long long k=0;
  while (n)
  {
    if (n%10) k=10*k+n%10;
    n/=10;
  }
  while (k)
    n=10*n+k%10, k/=10;
  return n;
}

int main(){
  cin >> a >> b;
  if (f(a)+f(b)==f(a+b)) cout<<"YES";
  else cout << "NO";
  return 0;
}
