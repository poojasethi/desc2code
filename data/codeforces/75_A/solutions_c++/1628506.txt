#include <iostream>
#include <string>

using namespace std;

int A,B,C;

int remove(int x)
{
	int ret=0,ten=1;
	while (x)
	{
		if (x%10!=0)
		{
			ret+=x%10*ten;
			ten*=10;
		}
		x/=10;
	}
	return ret;
}

int main()
{
	cin >> A >> B;
	cout << (remove(A)+remove(B)==remove(A+B)?"YES":"NO") << endl;
	return 0;
}
