#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll f(ll x) {
	ll t=0;
	for(; x>0; x /= 10) if(x%10) t=10*t+x%10;
	for(; t>0; t /= 10) x=10*x+t%10;
	return x;
}

int main() {
	ll a, b;
	cin >> a >> b;
	cout << (f(a)+f(b)==f(a+b) ? "YES" : "NO") << endl;
	return 0;
}
