#include<iostream>
using namespace std;
int ans(int x)
{ int an=1,e=0;
    while(x)
    {
        if(x%10!=0)
        {e=e+x%10*an;
        an=an*10;

        }
        x=x/10;
    }
    return e;
}
int main()

{
    int a,s,sum;
    cin>>a>>s;
    sum=a+s;
    cout<<(ans(a)+ans(s)==ans(sum)?"YES":"NO");
}
