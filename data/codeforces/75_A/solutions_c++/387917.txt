#include <stdio.h>
#include <string.h>
int f(int x){
	int res=0,i,k=1;
	while (x>0){
		i=x%10;
		x/=10;
		if (i>0){
			res+=i*k;
			k*=10;
		}
	}
	return res;
}
int main(){
	int i,j,k;
	int a,b,c;
	scanf("%d %d",&a,&b);
	c=a+b;
	if (f(a)+f(b)==f(c)) printf("YES\n");else printf("NO\n");
	return 0;
}
