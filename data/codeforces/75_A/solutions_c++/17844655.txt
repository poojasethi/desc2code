#include<cstdio>
int shi(int x)
{
	int shu=0,y=x,ci=1;
	while(y)
	{
		int yu=y%10;
		if(yu)
		shu+=ci*yu,ci*=10;
		y=y/10;
	}
		return shu;
}

	{
			int a,b,c;

	c=a+b;
	if(shi(c)==shi(a)+shi(b))
	printf("YES\n");
	else
	printf("NO\n"); 
return 0;

	}
