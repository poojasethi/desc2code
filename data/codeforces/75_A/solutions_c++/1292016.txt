#include <algorithm>
#define _USE_MATH_DEFINES
#include <cmath>
#include <cstdio>

using namespace std;

int dc(int x) {
	if (x==0) return 0;
	return x%10 ? dc(x/10)*10+x%10 : dc(x/10);
}

int main() {
	int a,b,n;
	scanf("%d%d",&a,&b);
	n=a+b;
	printf(dc(n)==dc(a)+dc(b)?"YES\n":"NO\n");
	return 0;
}
