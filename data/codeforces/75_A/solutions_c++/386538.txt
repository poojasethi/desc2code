#include<iostream>
#include<vector>
using namespace std;

int no_zeros(int a) {
  if (a == 0) return 0;
  if (a % 10 == 0) return no_zeros(a / 10);
  return 10 * no_zeros(a / 10) + a % 10;
}

int main() {
  int a, b; cin >> a >> b;
  if (no_zeros(a)+no_zeros(b)==no_zeros(a+b)) cout << "YES" << endl;
  else cout << "NO" << endl;
}
