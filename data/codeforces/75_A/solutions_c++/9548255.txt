#include <iostream>
using namespace std;

int f(int x){
	int b=0,t=1;
	while (x!=0){
		while (x%10==0)
            x/=10;
		b+=(x%10)*t;
		t*=10;
		x/=10;
	}
	return b;
}
int main(){
	int a,b,c,xa=0,xb=0;
	cin>>a>>b;
	if ((f(a)+f(b))==f(a+b))
        cout<<"YES";
	else
	    cout<<"NO";
    return 0;
}
