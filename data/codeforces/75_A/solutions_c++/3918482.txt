#include<stdio.h>
long long rem(long long x){
	long long z=0;
	while (x!=0)
	{
		if(x%10!=0)
			z=z*10+x%10;
		x/=10;
	}
	x=0;
	while(z!=0){
	x=x*10+z%10;
	z/=10;
	}
return x;
}
int main () 
{
	int x,y;
	scanf("%d%d",&x,&y);
	if(rem(x+y)==rem(x)+rem(y))
		printf("YES");
	else
		printf("NO");
	return 0;
}
