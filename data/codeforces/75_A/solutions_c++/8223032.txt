#include <iostream>
using namespace std;
int f(int n){
    int ret = 0, ten = 1;
    while (n){
        if (n % 10)ret += n % 10 * ten, ten *= 10;
        n /= 10;
    }
    return ret;
}
int main(){
    int a, b;
    cin >> a >> b;
    cout << (f(a + b) == f(a) + f(b) ? "YES\n" : "NO\n");
    return 0;
}
