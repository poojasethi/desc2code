#include<iostream>
using namespace std;
int zero(int x)
{
    int w=1,e=0;
    while(x)
    {
        if(x%10!=0)
        {e=e+x%10*w;
        w=w*10;

        }
        x=x/10;
    }
    return e;
}
int main()

{
    int a,s,d;
    cin>>a>>s;
    d=a+s;
    cout<<(zero(a)+zero(s)==zero(d)?"YES":"NO");
}
