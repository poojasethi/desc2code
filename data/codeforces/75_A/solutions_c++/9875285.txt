#include<math.h>
int fun(int n)
{ int i=1,r,m=0;
 while(n)
 {
  r=n%10;
  if(r!=0)
  {
  m=m+r*i;
  i=i*10;
  }
  n=n/10;
 }
 return m;
}

{
