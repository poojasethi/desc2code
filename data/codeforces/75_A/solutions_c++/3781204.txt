#include<stdio.h>
#include<stdlib.h>
#include<iostream>
using namespace std;

int removec(int n){
	int r = 0, ten = 1;
	while(n){
		if(n%10){
			r = r + (n%10)*ten;
			ten *= 10;
		}
		n /= 10;
	}
	return r;
}

int main(){
	int n, m;
	cin>>n>>m;
	if(removec(n+m) == removec(n)+removec(m)) printf("YES\n");
	else printf("NO\n");
}
