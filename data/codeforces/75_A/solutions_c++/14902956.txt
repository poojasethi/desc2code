#include<bits/stdc++.h>

using namespace std;

int rem(int n)
{
  int c=0,r=1;
  while(n)
    (n%10?c=r*(n%10)+c,r*=10:c=c),n/=10;
  return c;
}

int main()
{
  int a,b;
  cin>>a>>b;
  int c=a+b;
  c=rem(c),b=rem(b),a=rem(a);
  if(a+b==c)
    cout<<"YES";
  else
    cout<<"NO";
}
