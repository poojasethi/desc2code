n, m, k = map(int, raw_input().split())

grid = [[0 for i in xrange(m)] for j in xrange(n)]

for l in xrange(k):

	i, j = map(int, raw_input().split())
	i -= 1
	j -= 1
	if grid[i][j] == 0:
		grid[i][j] = l+1

move = 10**9
for i in xrange(n-1):
	for j in xrange(m-1):

		if grid[i][j] > 0 and grid[i+1][j] > 0 and grid[i][j+1] > 0 and grid[i+1][j+1] > 0:
			move = min(move, max(grid[i][j], grid[i+1][j], grid[i][j+1], grid[i+1][j+1]))

if move == 10**9:
	print 0
else:
	print move
