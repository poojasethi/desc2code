n, m, k = map(int, raw_input().split())
S = set()
done = False
for i in xrange(k):
	x, y = map(int, raw_input().split())
	S.add((x, y))

	def ok(x, y):
		return (x, y) in S and (x + 1, y) in S and (x, y + 1) in S and (x + 1, y + 1) in S

	if ok(x, y) or ok(x - 1, y) or ok(x, y - 1) or ok(x - 1, y - 1):
		done = True
		print i + 1
		break

if not done: print 0
