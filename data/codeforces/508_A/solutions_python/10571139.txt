
def sq(a,x,y):
	if a[x][y] and a[x+1][y] and a[x][y+1] and a[x+1][y+1]:
		return True
	return False

n,m,k = [int(x) for x in raw_input().split()]
a = [[False]*(m+2) for x in xrange(n+2)]
move = 0
for i in xrange(k):
	x,y = [int(x) for x in raw_input().split()]
	a[x][y] = True
	if (sq(a,x-1,y-1) or sq(a,x-1,y) or sq(a,x,y-1) or sq(a,x,y)):
		move = i+1
		break

print move
