#include<iostream>
using namespace std;
int a[1001][1001];
int main(){
	int n, m, k, i, j, num=0;
	cin >> n >> m >> k;
	while( k-- ){
		cin >> i >> j;
		if( a[i][j+1]>0&&a[i+1][j]>0&&a[i+1][j+1]>0 || a[i][j+1]>0&&a[i-1][j]>0&&a[i-1][j+1]>0 || 
			a[i+1][j]>0&&a[i][j-1]>0&&a[i+1][j-1]>0 || a[i-1][j]>0&&a[i][j-1]>0&&a[i-1][j-1]>0 ){
			cout << ++num;
			return 0;
		}
		a[i][j]++;
		num++;
	}
	cout << "0";
	return 0;
}
