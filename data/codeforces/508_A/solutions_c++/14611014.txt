#include<stdio.h>
int ab[1010][1010];
int main()
{
	int n,m,k,a,b,t=0,i;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=k;i++)
	{
		scanf("%d%d",&a,&b);
		ab[a][b]=1;
	if(ab[a][b]&&ab[a-1][b-1]&&ab[a][b-1]&&ab[a-1][b]
	||ab[a][b]&&ab[a+1][b+1]&&ab[a+1][b]&&ab[a][b+1]
	||ab[a][b]&&ab[a-1][b+1]&&ab[a-1][b]&&ab[a][b+1]
	||ab[a][b]&&ab[a+1][b-1]&&ab[a+1][b]&&ab[a][b-1])
	{
		t=i;break;
	}
	
}
	printf("%d\n",t);
	return 0;
 }
