#include <bits/stdc++.h>
using namespace std;
int gp[1234][1234], ans;
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int n,m,i,j,k,a,b,x,y;
    cin >> n >> m >> k;
    for(i=1;i<=k;i++)
    {
        cin>> a >> b;
        gp[a][b] = 1;
        for(x=-1;x<2;x+=2)
            for(y=-1;y<2;y+=2)
                if( gp[a+x][b+y] && gp[a][b+y] && gp[a+x][b] && !ans ) ans = i;
    }
    cout << ans <<endl;
    return 0;
}
