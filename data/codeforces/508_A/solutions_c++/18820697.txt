#include <stdio.h>

int row, col, k, x, y, A[1010][1010];

int main() {
  scanf("%d%d%d", &row, &col, &k);
  for (int t = 1 ; t <= k ; ++t) {
    scanf("%d%d", &x, &y);
    A[x][y] = 1;
    for (int i = x - 1 ; i <= x ; ++i) {
      for (int j =  y - 1 ; j <= y ; ++j) {
        if (A[i][j] && A[i + 1][j] && A[i][j + 1] && A[i + 1][j + 1]) {
          printf("%d\n", t);
          return 0;
        }
      }
    }
  }
  puts("0");
}
