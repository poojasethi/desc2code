#include <iostream>
#include <cmath>

using namespace std;

int v[2000][2000];

bool cuad(int i,int j)
{
  return v[i][j]==1 and v[i+1][j]==1 and v[i][j+1]==1 and v[i+1][j+1]==1;
}

int main()
{
  ios::sync_with_stdio(false);
  int n,m,k;
  cin>>n>>m>>k;
  int sol=0;
  for (int d=0;d<k;d++) {
    int i,j;
    cin>>i>>j;
    v[i][j]=1;
    if (sol==0 and (cuad(i,j) or cuad(i-1,j) or cuad(i,j-1) or cuad(i-1,j-1)))
      sol=d+1;
  }
  cout<<sol<<endl;
}
