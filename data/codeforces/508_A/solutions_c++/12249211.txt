#include<iostream>
#include<string.h>
using namespace std;
int map[1005][1005];
int main()
{
	int n,m,k,a,b,step=0,i;
	cin>>n>>m>>k;
	memset(map,0,sizeof(map));
	for(i=0;i<k;i++)
	{
		cin>>a>>b;
		map[a][b]=1;
		for(int x=-1;x<2;x+=2)
		{
			for(int y=-1;y<2;y+=2)
			{
				if(map[x+a][b]&&map[a][y+b]&&map[x+a][y+b]&&step==0)
				{
					step=i+1;
				}
			}
		}
	}
	cout<<step<<endl;
}
