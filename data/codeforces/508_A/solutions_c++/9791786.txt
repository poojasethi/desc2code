#include<cstdio>
using namespace std;
bool s[1002][1002];
int main()
{
	int n,m,k,a,b,p;
	scanf("%d%d%d",&n,&m,&k);
	for(p=0;p<k;p++)
	{
		scanf("%d%d",&a,&b);
		s[a][b]=1;
		if(s[a-1][b-1]&&s[a][b-1]&&s[a-1][b] || s[a-1][b]&&s[a-1][b+1]&&s[a][b+1] || s[a][b-1]&&s[a+1][b-1]&&s[a+1][b] || s[a][b+1]&&s[a+1][b]&&s[a+1][b+1])
		{
			printf("%d\n",p+1);
			return 0;
		}
	}
	printf("0\n");
	return 0;
}
