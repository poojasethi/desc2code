#include<stdio.h>
int k,i,a,b,d[1005][1005],c;
int main(){

	scanf("%d %d %d",&a,&b,&k);

	for(i=1 ; i<=k ; i++){
		scanf("%d %d",&a,&b);
		d[a][b]=1;
		if(!c && d[a][b+1] && d[a+1][b] && d[a+1][b+1])
			c=i;
		if(!c && d[a][b-1] && d[a+1][b] && d[a+1][b-1])
			c=i;
		if(!c && d[a-1][b] && d[a][b+1] && d[a-1][b+1])
			c=i;
		if(!c && d[a][b-1] && d[a-1][b] && d[a-1][b-1])
			c=i;
	}
	printf("%d",c);

	return 0;
}
