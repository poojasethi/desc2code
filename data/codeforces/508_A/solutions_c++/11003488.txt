#include<iostream>
using namespace std;
int n,m,k,f[1005][1005],x,y,num;
int dian(int a,int b)
{
	if(f[a][b]==1 && f[a-1][b]==1 && f[a][b-1]==1 && f[a-1][b-1]==1)
	return 1;return 0;
} 
int main()
{
	cin>>n>>m>>k;
	for(int i=1;i<=k;i++)
	{
		cin>>x>>y;f[x][y]=1;
		if(num==0 && (dian(x,y)==1 || dian(x+1,y)==1 || dian(x,y+1)==1 || dian(x+1,y+1)==1))
		num=i; 
	}
	cout<<num;
	return 0;
}
