#include <iostream>
using namespace std;

int n,m,k;
int a[1100][1100];

int check(int x,int y)
{
	for(int i=x-1;i<x+1;i++)
		for(int j=y-1;j<y+1;j++)
			if(a[i][j]==1 && a[i+1][j]==1 && a[i][j+1]==1 && a[i+1][j+1]==1 )
				return 1;
return 0;
}

int main()
{
	cin>>n>>m>>k;
	for(int i=1;i<=k;i++)
	{
		int x,y;
		cin>>x>>y;
		a[x][y]=1;
		if(check(x,y))
		{
			cout<<i;
			return 0;
		}
	}
	cout<<0;
}
