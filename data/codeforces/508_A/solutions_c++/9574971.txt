#include <iostream>
#include <stdio.h>
using namespace std;

int n,m,x,y,i,k,a[1005][1005];

int main(){
//	freopen("input.txt","r",stdin);
//	freopen("output.txt","w",stdout);
	cin >> n >> m >> k;
	for (i=0; i<k; i++)
	{
		cin >> x >> y;
		a[x][y] = 1;
		if (a[x-1][y] && (a[x-1][y-1] && a[x][y-1] || a[x-1][y+1] && a[x][y+1]) || a[x+1][y] && (a[x+1][y-1] && a[x][y-1] || a[x][y+1] && a[x+1][y+1]))
		{
			cout << i+1;
			return 0;
		}
	}
	cout << 0;
	return 0;
}
