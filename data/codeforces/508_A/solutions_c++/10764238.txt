#include <bits/stdc++.h>
using namespace std;
bool mat[1005][1005];
int main()
{
	int n,m,k,a,b,knt=0;
	scanf("%d %d %d",&n,&m,&k);
	for(int i=1;i<=k;i++)
	{
		scanf("%d %d",&a,&b);
		mat[a][b]=1;
		for(int x=-1;x<2;x+=2)
		{
			for(int y=-1;y<2;y+=2)
			{
				if(mat[a+x][b+y]&&mat[a+x][b]&&mat[a][b+y]&&knt==0)
				{
					knt=i;
				}
			}
		}
	}
	printf("%d\n",knt);
	return 0;
}
