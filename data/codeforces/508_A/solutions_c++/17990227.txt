#include<cstdio>
#include<algorithm>
using namespace std;
int a[1005][1005];
int main()
{
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	int i,x,y;
	for(i=1;i<=k;i++)
	{
		scanf("%d%d",&y,&x);
		a[y][x]++;
		if(a[y][x-1]&&a[y-1][x-1]&&a[y-1][x])
		break;
		if(a[y-1][x]&&a[y][x+1]&&a[y-1][x+1])
		break;
		if(a[y+1][x]&&a[y+1][x+1]&&a[y][x+1])
		break;
		if(a[y][x-1]&&a[y+1][x]&&a[y+1][x-1])
		break;
	}
	printf("%d",i==k+1?0:i);
	return 0;
}
