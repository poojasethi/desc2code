#include <bits/stdc++.h>
using namespace std;
int n, m, k, x, y, a[1005][1005];
int main(){
	scanf("%d %d %d", &n, &m, &k);
	for(int i = 1; i <= k; i ++){
		scanf("%d %d", &x, &y);
		a[x][y] = 1;
		for(int j = max(x-1, 0); j <= min(x+1-1, n-1); j ++){
			for(int k = max(0, y-1); k <= min(y+1-1, m-1); k ++){
				if(a[j][k] && a[j][k+1] && a[j+1][k] && a[j+1][k+1]){
					printf("%d", i);
					return 0;
				}
			}
		}
	}
	puts("0");
}
