#include <iostream>
using namespace std;
int main(){
	int n , m , k , x , y;
	cin >> n >> m >> k;
	bool map[1010][1010] = {};
	for(int i = 1 ; i <= k ; i++){
		cin >> x >> y;
		map[x][y] = 1;
		for(int xx = x - 1 ; xx < x + 1 ; xx ++){
			for(int yy = y - 1 ; yy < y + 1 ; yy ++){
				if(map[xx][yy] && map[xx+1][yy] && map[xx][yy+1] && map[xx+1][yy+1]){
					cout << i;
					return 0;
				}
			}
		}
	}
	cout << 0;
}
