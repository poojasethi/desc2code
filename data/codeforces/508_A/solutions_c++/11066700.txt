#include<cstdio>
using namespace std;
int aa[1005][1005];
int main(){
	int n,m,k,ans=0,a,b;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=k;i++){
		scanf("%d%d",&a,&b);aa[a][b]=1;
		if(aa[a-1][b-1]&&aa[a-1][b]&&aa[a][b-1]||aa[a-1][b]&&aa[a-1][b+1]&&aa[a][b+1]||aa[a][b+1]&&aa[a+1][b+1]&&aa[a+1][b]||aa[a+1][b]&&aa[a+1][b-1]&&aa[a][b-1]){
			ans=i;break;
		}
	}
	printf("%d",ans);
	return 0;
}
