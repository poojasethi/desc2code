
def solve(N, M, vs, data):
    cost = 0
    for d in data:
        # print d
        cost += min(vs[d[0] - 1], vs[d[1] - 1])

    print cost



if __name__ == "__main__":
    N, M = map(int, raw_input().split())
    vs = map(int, raw_input().split())
    data = []
    for i in xrange(M):
        data.append(map(int, raw_input().split()))

    solve(N, M, vs, data)
