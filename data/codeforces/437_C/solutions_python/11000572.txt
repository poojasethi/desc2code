def plist(a, s = ' ') :
  print s.join(map(str, a))

def rint() :
  return int(sys.stdin.readline())

def rints() :
  return map(int, sys.stdin.readline().split())

def rfield(n, m = None) :
  if m == None :
    m = n
  
  field = []
  for i in xrange(n) :
    chars = sys.stdin.readline().strip()
    assert(len(chars) == m)
    field.append(chars)
  return field

def digits(x, p) :
  digits = []
  while x > 0 :
    digits.append(x % p)
    x /= p
  return digits
