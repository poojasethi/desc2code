
def IN():
    return map(int, raw_input().split()) 

n, m = IN()
v = [0] + IN()

graph = [[] for _ in range(n+1)] 
removed = [False for _ in range(n+1)] 

for _ in range(m): 
    s, t = IN() 
    graph[s] += [t] 
    graph[t] += [s] 

cost = 0 

sorted_v = sorted(zip(range(1, n+1), v[1:]), key=lambda x: -x[1]) 

for i, c in sorted_v: 
    adjs = graph[i]
    for a in adjs: 
        if removed[a]: 
            continue 
        cost += v[a] 

    removed[i] = True 

print cost
