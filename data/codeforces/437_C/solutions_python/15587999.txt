p = raw_input().split()
N = int(p[0])
M = int(p[1])

r = raw_input().split()
v = [int(i) for i in r]

min = lambda x, y: x if x < y else y
f = [[int(i) for i in raw_input().split()] for j in range(M)]
calc = lambda t: min(v[t[0]-1], v[t[1]-1])
print(sum([calc(t) for t in f]))
