n, m = map(int, raw_input().split())

# note that using a connectivty matrix makes removing nodes faster
conn = [[False]*n for i in range(n)]

vs = map(int, raw_input().split())

for _1 in range(m):
    i,j = map(int, raw_input().split())
    conn[i-1][j-1] = True
    conn[j-1][i-1] = True


def energy(i):
    return sum(vs[node] for node, v in enumerate(conn[i]) if v == True)

removed = [False]*n

def remove(i):
    global conn
    global removed
    for j in range(n):
        conn[i][j] = False
        conn[j][i] = False
    removed[i] = True
