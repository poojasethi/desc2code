def main():
    from collections import Counter
    n, cnt, l = int(raw_input()), Counter(), []
    for x, y in sorted((tuple(map(int, raw_input().split())) for _ in xrange(n)), reverse=True):
        for u, v in l:
            cnt[u - x, v - y] += 1
        l.append((x, y))
    print sum(i * (i - 1) for i in cnt.values()) // 4


if __name__ == '__main__':
    main()
