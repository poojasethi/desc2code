#include <cstdio>
#include <map>
#include <iostream>
using namespace std;

int x[2222];
int y[2222];

map<pair<int, int>, int> cnt;

int main() {
  int n;
  cin >> n;
  for (int i = 1; i <= n; i++) {
    cin >> x[i] >> y[i];
  }
  int ans = 0;
  for (int i = 1; i <= n; i++) {
    for (int j =1; j <= n; j++) {
      if(i==j)continue;
      int dx = x[i] - x[j];
      int dy = y[i] - y[j];
      ans += cnt[make_pair(dx, dy)]++;

    }
  }
  cout << ans / 4;
  return 0;
}
