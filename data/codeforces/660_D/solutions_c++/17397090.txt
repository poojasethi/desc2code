#include <stdio.h>
#include <iostream>
#include <string>
#include <algorithm>
#include <string>
#include <string.h>
#include <math.h>
#include <map>
#define MAXN 2005

using namespace std;

int main()
{
    //freopen("caicai.txt","r",stdin);
    map<pair<int,int>,int>point;
    int x[MAXN],y[MAXN];
    int n;
    scanf("%d",&n);
    int i,j,ans = 0;
    for(i = 0;i<n;i++)
    {
        scanf("%d%d",&x[i],&y[i]);
        for(j = 0;j<i;j++)
            ans += point[make_pair(x[i]+x[j],y[i]+y[j])]++;
    }
    cout<<ans<<endl;
    return 0;
}
