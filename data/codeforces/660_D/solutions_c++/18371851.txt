#include<bits/stdc++.h>
using namespace std;

typedef pair<int,int> ii;
typedef long long ll;
ii x[2222];
map<ii,int> mymap;
ll res;

int main() {
    int n;
    res = 0;
    scanf("%d",&n);
    for(int i = 0; i < n; ++i) scanf("%d%d",&x[i].first,&x[i].second);
    for(int i = 0; i < n; ++i)
        for(int j = 0; j < i; ++j)
            ++mymap[ii(x[i].first+x[j].first,x[i].second+x[j].second)];
    for(map<ii,int>::iterator it = mymap.begin(); it != mymap.end(); ++it) {
        ll tmp = it->second;
        res += (tmp*(tmp-1))/2;
    }
    cout<<res<<"\n";
}
