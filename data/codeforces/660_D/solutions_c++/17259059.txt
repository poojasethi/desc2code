#include<cstdio>
#include<map>
using namespace std;
int i,j,n,ans,x[2005],y[2005];
map<pair<int,int>,int>a;
inline int C(int x){return x*(x-1)/2;}
int main()
{
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&x[i],&y[i]);
		for(j=1;j<i;j++)
			a[make_pair(x[i]+x[j],y[i]+y[j])]++;
	}
	for(map<pair<int,int>,int>::iterator it=a.begin();it!=a.end();it++)
		ans+=C(it->second);
	printf("%d",ans);
}
