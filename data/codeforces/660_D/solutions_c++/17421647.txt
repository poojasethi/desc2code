#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<map>
using namespace std;
int x[2100];
int y[2100];
int main()
{
  int n;
  int ans=0;
  scanf("%d",&n);
  map<pair<int,int>,int>point;
  for(int i=0;i<n;i++)
  {
    scanf("%d%d",&x[i],&y[i]);
    for(int j=0;j<i;j++)
    ans+=point[make_pair(x[i]+x[j],y[i]+y[j])]++;

  }
  printf("%d\n",ans);
  return 0;
}
