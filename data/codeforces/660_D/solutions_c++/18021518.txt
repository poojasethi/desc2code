#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
#include<map>
#include<utility>
using namespace std;
int x[4000],y[4000];
pair<int,int>v[5000000];
map<pair<int,int>,int>m; 
int n;
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&x[i],&y[i]);
	}
	int t=0; 
	for(int i=1;i<n;i++)
		for(int j=i+1;j<=n;j++){
			t++;
			v[t].first=x[i]+x[j];
			v[t].second=y[i]+y[j];
			m[v[t]]++;
		} 
	int sol=0;
	map<pair<int,int>,int>::iterator ii;
	for(ii=m.begin();ii!=m.end();ii++){
		sol+=ii->second*(ii->second-1)/2;
	}
	printf("%d",sol);
	return 0;
}
