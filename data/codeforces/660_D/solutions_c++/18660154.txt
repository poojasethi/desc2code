#include<cstdio>
#include<map>
#define MG make_pair
using namespace std;
typedef long long ll;
typedef pair<int,int> pa;
int n,x[2010],y[2010];
ll ans;
map<pa,int> mp;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d%d",&x[i],&y[i]);
	for (int i=1;i<=n;i++)
		for (int j=i+1;j<=n;j++)
		{
			ans+=mp[MG(x[i]+x[j],y[i]+y[j])];
			mp[MG(x[i]+x[j],y[i]+y[j])]++;
		}
	printf("%I64d\n",ans);
}
