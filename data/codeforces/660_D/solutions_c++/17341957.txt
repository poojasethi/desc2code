#include <bits/stdc++.h>
const int maxn = 2000+10;
int n, ans, x[maxn], y[maxn];
using namespace std;
map<pair<int, int>, int> a;
int main()
{
	scanf("%d", &n);
	for(int i = 1; i <= n; i++)
	{
		scanf("%d%d", &x[i], &y[i]);
		for(int j = 1; j < i; j++)
		{
			ans += a[make_pair(x[i]+x[j], y[i]+y[j])]++;
		}
	}
	printf("%d",ans);
}
