#include <cstdio>
#include <map>
#include <iostream>
using namespace std;
const int maxn=2000+5; 
pair<long long ,long long>b[maxn];
map<pair<int,int>,int>sum;
int main(void)
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	scanf("%d%d",&b[i].first,&b[i].second);
	int k=0;
	k--;
	int count=0;
	for(int i=0;i<n-1;i++)
		for(int j=i+1;j<n;j++)
		{
			int x=b[i].first+b[j].first;
			int y=b[i].second+b[j].second;
			count+=sum[make_pair(x,y)]++;
		}
	cout<<count<<endl;
}
