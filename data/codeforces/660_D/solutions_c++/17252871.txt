#include<bits/stdc++.h>
using namespace std;
#define mid pair<double,double>
#define L long long int
map <mid,L> diag;
L x[2005],y[2005];
int n;
int main()
{
	int i;
	cin>>n;
	L ans=0;
	for(i=0;i<n;i++)
	{
		cin>>x[i]>>y[i];
		for(int j=0;j<i;j++)
		{
			mid m=make_pair((x[i]+x[j])/2.0,(y[i]+y[j])/2.0);
			ans+=diag[m];
			diag[m]++;
		}
	}
	cout<<ans<<"\n";
	return 0;

}
