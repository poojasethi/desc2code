#include <cstdio>
#include <iostream> 
#include <utility>
#include <map>
using namespace std;
struct thing 
{
	int x,y;
} p[2005];
map <pair<int,int>,int> f;
int n,ans;
int main()
{
	cin.sync_with_stdio(false);
	cin>>n;
	for(int i = 1;i <= n;i++) cin>>p[i].x>>p[i].y;
	for(int i = 1;i <= n-1;i++)
	 for(int j = i+1;j <= n;j++) 
	 {
	 	ans += f[make_pair(p[i].x+p[j].x,p[i].y+p[j].y)];
	 	f[make_pair(p[i].x+p[j].x,p[i].y+p[j].y)]++; 
  	 } 
  	cout<<ans<<endl;
}
