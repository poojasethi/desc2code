#include<bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int n,i,j;
    long int x[2000],y[2000];
    map< pair<int, long int>, long int > m;
    cin>>n;
    long long int c=0;

    for(i=0;i<n;i++)
        cin>>x[i]>>y[i];

    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            m[make_pair(x[i]+x[j],y[i]+y[j])]++;
        }
    }

    for(map< pair<int, long int>, long int >::const_iterator it = m.begin(); it!=m.end();it++)
    {
        c+=(it->second)*(it->second-1)/2;
    }

    cout<<c;

    return 0;
}
