#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> Pair;

int n, segmentSz;
Pair a[2222];
map<Pair, int> cnt;

int main()
{
    scanf("%d", &n);
    for(int i = 1; i <= n; ++i) scanf("%d%d", &a[i].first, &a[i].second);
    int res = 0;
    for(int i = 1; i <= n - 1; ++i)
        for(int j = i + 1; j <= n; ++j)
            res += cnt[Pair(a[i].first + a[j].first, a[i].second + a[j].second)]++;
    printf("%d", res);
    return 0;
}
