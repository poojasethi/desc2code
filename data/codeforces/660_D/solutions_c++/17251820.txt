#include<iostream>
#include<map>
#include<math.h>
#include<stdio.h>
using namespace std;
map<pair<int, int>, int> e;
int main(){
	int n,i,x[2005],y[2005],j,ans=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d%d",&x[i],&y[i]);
	}
	for(i=1;i<n;i++){
    	for(j=i+1;j<=n;j++){
			ans+= e[make_pair(x[i]+x[j],y[i]+y[j])];
			e[make_pair(x[i]+x[j],y[i]+y[j])]++;
		}
	}
  	cout<<ans<<endl;
	return 0;
}
