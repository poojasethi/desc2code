#include <bits/stdc++.h>
using namespace std;

int N, x[2002], y[2002];
map < pair <int, int>, int > cnt;
long long ans;

main()
{
    cin >> N;
    for (int i = 1; i <= N; i++)
        cin >> x[i] >> y[i];
    for (int i = 1; i < N; i++)
        for (int j = i + 1; j <= N; j++)
        {
            ans += cnt[make_pair(x[i] + x[j], y[i] + y[j])];
            cnt[make_pair(x[i] + x[j], y[i] + y[j])]++;
        }
    cout << ans;
}
