#include <bits/stdc++.h>
using namespace std;

#define ll unsigned long long
#define ld long double
#define MAX 1000000009
#define N 2003
int x[N];
int y[N];
map<ld, map<ll, int> > m;
int main()
{
	int n;
	cin >> n;

	for(int i = 0; i<n; i++)
	{
		scanf("%d%d", &x[i], &y[i]);
	}

	int ans = 0;

	for(int i = 0; i<n; i++)
	{
		for(int j = i+1; j<n; j++)
		{
			ld s;
			if(x[i]!=x[j])
				s = (ld)(y[i]-y[j])/(ld)(x[i]-x[j]);
			else
				s = MAX+1;
			ll len = (ll)(y[i]-y[j])*(ll)(y[i]-y[j]) + (ll)(x[i]-x[j])*(ll)(x[i]-x[j]);
			m[s][len]++;
			ans+=m[s][len]-1;
		}
	}

	cout << ans/2 << endl;
}
