#include <bits/stdc++.h>
#define inc(i,a,b) for (int i=a; i<=b; i++)
#define x first
#define y second
#define ll long long
using namespace std;

typedef pair<int,int> ii;

int n;
ll ans=0LL;
ii a[2002];
map<ii,ll> edge;

int main() {
	scanf("%d",&n);
	if (n<4) { printf("0\n"); return 0; }
	inc(i,1,n) scanf("%I64d%I64d",&a[i].x,&a[i].y);
	sort(a+1,a+1+n);
	inc(i,1,n-1) inc(j,i+1,n)
		edge[make_pair(a[j].x-a[i].x,a[j].y-a[i].y)]++;
	
	for (map<ii,ll>::iterator it=edge.begin(); it!=edge.end(); it++)
	    ans+=((it->second*(it->second-1)) >> 1LL);
	
	printf("%I64d\n",(ans>>1LL));
}
