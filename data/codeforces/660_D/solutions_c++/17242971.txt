#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair < int, int > ii;

const int N = 2000 + 5;

int n;
int a[N], b[N];
map < ii, int > M;

int main () {

    scanf("%d", &n);

    ll ans = 0;

    for(int i = 1; i <= n; i++) {
        scanf("%d %d", a + i, b + i);
        for(int j = 1; j < i; j++) {
            int x = a[i] + a[j];
            int y = b[i] + b[j];
            ans += M[make_pair(x, y)]++;
        }
    }

    //printf("%lld\n", ans);
cout<<ans<<endl;
    return 0;

}
