#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<map>
using namespace std;
#define LL long long
#define MAXN  2010
typedef pair<double,double> P;
int x[MAXN],y[MAXN];
map<P,int> mp;
int main()
{
	int n;
	cin>>n;
	int ans=0;
	for(int i=1;i<=n;i++) cin>>x[i]>>y[i];
	for(int i=1;i<=n;i++){
		for(int j=i+1;j<=n;j++){
			double tx=(x[i]+x[j])/2.,ty=(y[i]+y[j])/2.;
			ans+=mp[P(tx,ty)];
			mp[P(tx,ty)]++;
		}
	}
	cout<<ans<<endl;
	return 0;
}
