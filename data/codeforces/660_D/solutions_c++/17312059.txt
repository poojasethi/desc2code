#include<bits/stdc++.h>
using namespace std;

int n,ans,x,y,t=1;
struct coord{
	int x,y;
}a[2001],p,k[1999005];

int main(){
	map< pair<int,int>,int >b;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&a[i].x,&a[i].y);
	}
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++){
			p.x=(a[i].x+a[j].x);
			p.y=(a[i].y+a[j].y);
			b[make_pair(p.x,p.y)]++;
			if(b[make_pair(p.x,p.y)]==1){
				k[t]=p;
				t++;
			}
		}
	for(int i=1;i<=t;i++){
		ans+=b[make_pair(k[i].x,k[i].y)]*(b[make_pair(k[i].x,k[i].y)]-1)/2;
	}
	cout<<ans;
return 0;
}//
