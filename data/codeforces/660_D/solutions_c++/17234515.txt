#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
map<pair<int,int>,int>mp;
int n,x[2005],y[2005];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
        scanf("%d%d",x+i,y+i);
    for(int i=1;i<=n;++i)
        for(int j=i+1;j<=n;++j)
            ++mp[make_pair((x[i]+x[j]),((y[i]+y[j])))];
    LL ans=0,tmp;
    for(map<pair<int,int>,int>::iterator it=mp.begin();
        it!=mp.end();++it)
        {
            tmp=it->second;
            if(tmp>1)
                ans+=tmp*(tmp-1)>>1;
        }
    printf("%I64d\n",ans);
    return 0;
}
