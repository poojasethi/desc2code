#include<bits/stdc++.h>

using namespace std;
const int maxN=2100;
long long int n,ans;
double x[maxN],y[maxN];
map<pair<double,double>,int>cnt;
map<pair<double,double>,int>::iterator it;
#define MP make_pair
int main()
{
  cin>>n;
  for(int i=1;i<=n;i++)
    cin>>x[i]>>y[i];
  for(int i=1;i<=n;i++)
    for(int j=i+1;j<=n;j++)
      cnt[MP((x[i]+x[j])/2,(y[i]+y[j])/2)]++;
  
  for(it=cnt.begin();it!=cnt.end();it++)
    ans+=((it->second)*(it->second-1))/2;
  cout<<ans<<endl;
	  
  return 0;
}
