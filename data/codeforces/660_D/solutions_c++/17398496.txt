#include<bits/stdc++.h>

#define eps 0.000001

using namespace std;

double k[2010][2010],x[2010],y[2010];
int n,ans;

map < pair < double,double >,int> mid;
pair < double,double > tem;

int main()

{
    scanf("%d",&n);
    for (int i=1 ; i<=n ; i++)
        scanf("%lf%lf",&x[i],&y[i]);
    ans=0;
    for (int i=2 ; i<=n ; i++)
        for (int j=1 ; j<i ; j++)
        {
            tem.first=(x[i]+x[j])/2;
            tem.second=(y[i]+y[j])/2;
            ans+=mid[tem];
            mid[tem]++;
        }
    cout << ans;
    return 0;
}
