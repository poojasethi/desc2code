#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,x[2001],y[2001];
    cin>>n;
    map<pair<int,int>,int> a;
    for(int i=0;i<n;i++)
    {
        cin>>x[i]>>y[i];
    }
    int ans=0;
    for(int i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            ans+=a[make_pair(x[i]+x[j],y[i]+y[j])]++;
        }
    }
    cout<<ans;
    return 0;
}
