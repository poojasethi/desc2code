#include <bits/stdc++.h>
using namespace std;

#define IO ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);
#define endl "\n"
#define INF 2147483647
#define DBG 666
#define mp make_pair

typedef long long ll;

int n, result;
int x[2005], y[2005];
map< pair<int, int>, int > m;

void solve(){

	cin >> n;
	for(int i = 0; i < n; i++){

		cin >> x[i] >> y[i];
		for(int j = 0; j < i; j++){
			result += m[mp(x[i]+x[j], y[i]+y[j])];
			m[mp(x[i]+x[j], y[i]+y[j])]++;
		}
	}
	cout << result << endl;
}

int main(){

	IO;
	solve();

	return 0;
}
