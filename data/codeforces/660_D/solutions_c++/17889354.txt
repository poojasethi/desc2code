#include<iostream>
#include<stdio.h>
#include<map>
using namespace std;
map<pair<int, int>, int> mp;
int x[2005], y[2005];
int main(){
    int n;
    scanf("%d", &n);
    for(int i = 0; i < n; i++){
        scanf("%d%d", &x[i], &y[i]);
    }
    int ans = 0;
    for(int i = 0;i < n; i++){
        for(int k = i+1; k < n; k++){
            pair<int, int> a;
            a.first = x[i]+x[k];
            a.second = y[i]+y[k];
            mp[a] ++;
            ans += mp[a] - 1;
           // printf("%d %d %d %d\n", i, k, a.first, a.second);
        }
    }
    printf("%d\n", ans);
}
