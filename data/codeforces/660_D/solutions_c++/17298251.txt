#include<bits/stdc++.h>
#define MX 2005
using namespace std;

int x[MX], y[MX];
map< pair<double, double>, int > mp;

int main(){
     int i, j, k;
     int n, ans=0;
     double midX, midY;
     cin >> n;
     for(i=0; i<n; ++i) cin >> x[i] >> y[i];
     for(i=0; i<n; ++i){
          for(j=i+1; j<n; ++j){
               midX=(x[i]+x[j])/2.;
               midY=(y[i]+y[j])/2.;
               ans+=mp[make_pair(midX, midY)]++;
          }
     }
     cout << ans << endl;
}
