#include<map>
#include<cstdio>
using namespace std;
map<pair<int,int>,int> s;
int n,x[2010],y[2010],i,j,a,b,ans;
int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	  scanf("%d%d",&x[i],&y[i]);
	for (i=1;i<=n;i++)
	  for (j=1;j<i;j++)
	  {
	  	a=x[i]+x[j];b=y[i]+y[j];
	  	s[make_pair(a,b)]++;
	  }
	  map<pair<int,int>,int>::iterator it;
	  for (it=s.begin();it!=s.end();it++)
	  {
	  	ans+=(it->second)*((it->second)-1)/2;
	  }
      printf("%d\n",ans);

	return 0;
 }
