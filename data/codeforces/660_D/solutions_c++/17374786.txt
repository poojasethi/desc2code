#include <iostream>
#include <map>

using namespace std;
#define F first
#define S second


map<pair< int ,int > ,int> mmap;

pair < int , int > arr[2001];

int main()
{
    int n ,a , b;
    cin >> n;
    long long res = 0 ;
    for(int i = 0; i < n;i++){
        cin >> arr[i].first >> arr[i].second;
    }
    for(int i = 0; i < n ; i++)
    {
        for (int j = i+1 ; j < n ; j++)
        {
            res += mmap[make_pair(arr[i].F + arr[j].F , arr[i].S + arr[j].S)]++;
        }
    }
    cout << res << endl;
    return 0;
}
