#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	scanf("%d",&n);

	pair<int,int> p[n];
	for(int i=0;i<n;i++)
		scanf("%d%d",&p[i].first,&p[i].second);

	map< pair<int,int>, int> m;

	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			pair<int,int> ptemp = make_pair(p[i].first + p[j].first, p[i].second + p[j].second);
			m[ptemp]++;
		}
	}

	int res = 0;

	for(map< pair<int,int>, int>::iterator it = m.begin(); it != m.end(); ++it)
	{
		int r = it->second;
		res += (r*(r-1))/2;
	}

	printf("%d\n",res);

	return 0;
}
