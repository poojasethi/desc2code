#include <bits/stdc++.h>

#define NN 2010

using namespace std;
typedef pair <int, int> pii;

pii p[NN],q[NN*NN];

int main (){
	
	int N,i,j;
	
	scanf("%d",&N);
	for(i=0; i<N; i++) scanf("%d %d",&p[i].first, &p[i].second);
	sort(p,p+N);
	
	int M=0;
	for(i=0; i<N-1; i++){
		for(j=i+1; j<N; j++){
			q[M].first=p[j].first-p[i].first;
			q[M++].second=p[j].second-p[i].second;
		}
	}
	
	sort(q,q+M);
	
	long long ans=0;
	
	for(i=0; i<M; i++){
		int t=1;
		while(q[i]==q[i+1]) i++, t++;
		ans+=t*(t-1)/2;
	}
	
	ans/=2;
	printf("%I64d\n",ans);
	
	return 0;
}
