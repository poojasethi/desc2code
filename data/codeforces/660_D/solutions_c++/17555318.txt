#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define pii pair<int,int>
#define LL long long
#define F first
#define S second
using namespace std;
int x[2020],y[2020];
map<pair<double,double>,int> m;
LL ans;
int main(){
	int n; scanf("%d",&n);
	for(int i = 0 ; i < n ; i++) scanf("%d%d",&x[i],&y[i]);
	for(int i = 0 ; i < n ; i++){
		for(int j = i+1 ; j < n ; j++){
			double tx = (double)((x[i]+x[j]))/(double)2.0;
			double ty = (double)((y[i]+y[j]))/(double)2.0;
			ans+=m[mp(tx,ty)];
			m[mp(tx,ty)]++;
		}
	}
	cout << ans << endl;
}
