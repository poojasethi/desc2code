#include <bits/stdc++.h>
using namespace std;

int n, x[2001], y[2001], cnt = 0;
map<pair<int, int>, int> a;

int main () {
    //freopen("test01.in", "r", stdin);
    //freopen("", "w", stdout);
    scanf("%d", &n);
    for (int i = 1; i <= n; i++)
        scanf("%d%d", &x[i], &y[i]);

    for (int i = 1; i <= n; i++)
        for (int j = i+1; j <= n; j++)
            a[make_pair(x[i]+x[j], y[i]+y[j])]++;

    for (map<pair<int, int>, int>::iterator it = a.begin(); it != a.end(); it++)
        cnt += it->second * (it->second - 1) / 2;
    printf("%d\n", cnt);
    return 0;
}
