#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <map>
using namespace std;
pair<int,int> temp;
int a[2010][2];
int n;
map<pair<int,int>,int>hehe;
int main()
{
	cin >> n;
	for (int i = 1;i <= n;i++)
		scanf("%d%d",&a[i][0],&a[i][1]);
	long long ans = 0;
	for (int i = 1;i <= n;i++)
		for (int j = i + 1;j <= n;j++)
		{
			temp = make_pair(a[i][0] + a[j][0],a[i][1] + a[j][1]);
			ans += hehe[temp];
			hehe[temp]++;
		}
	cout << ans << endl;
}
