#include <cstdio>
#include <cstring>
#include <map>
#include <algorithm>
using namespace std;
typedef pair<int,int> Vector;
const int maxn=2010;

Vector Q[maxn*maxn];
int x[maxn],y[maxn];
int n,ans,qn;

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d%d",&x[i],&y[i]);
	for (int i=1;i<=n;i++) for (int j=1;j<=n;j++) if (i!=j) Q[++qn]=make_pair(x[i]-x[j],y[i]-y[j]);
	sort(Q+1,Q+1+qn);
	for (int l=1,r=1;l<=qn;l=r+1)
	{
		r=l;
		while (r<qn && Q[r+1]==Q[r]) r++;
		ans+=(r-l+1)*(r-l)/2;
	}
	printf("%d\n",ans/4);
	return 0;
}
