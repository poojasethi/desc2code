from time import mktime
import sys
import re
n,m = map(int,raw_input().split())
num, tim = [],[]
for ch in sys.stdin:
	num.append(ch[:19])
	tim.append(mktime(map(int, re.split(':|-|\s',ch[:19]) + [0]*3)))
r = 0
for i, val in enumerate(tim):
	while tim[r] + n <= val:
		r = r + 1
	if i - r + 1 >= m:
		print num[i]
		quit()
print -1
