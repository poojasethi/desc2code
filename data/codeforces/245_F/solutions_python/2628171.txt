import sys
from time import *
I, n, i = [], 0, 0
for s in sys.stdin:
	if n:
		I.append((int(mktime(map(int, s[ : 19].replace('-', ' ').replace(':', ' ').split()) + [0] * 3)), s[ : 19]))
	else:
		n, m = map(int, s.split())
for j, d in enumerate(I):
	while I[i][0] + n <= d[0]:
		i += 1
	if j - i + 1 >= m:
		print d[1]
		quit()
print -1
