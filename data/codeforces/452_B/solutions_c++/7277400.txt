#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;

int main()
{
    int n,m;
    while(cin>>n>>m)
    {
        if(n==0)
            printf("0 1\n0 %d\n0 0\n0 %d\n",m,m-1);
        else if(m==0)
            printf("1 0\n%d 0\n0 0\n%d 0\n",n,n-1);
        else if(n<m)
        {
            if(sqrt(n*n+m*m)+m<2*sqrt(m*m+(n-1)*(n-1)))
                printf("1 0\n%d %d\n0 0\n%d %d\n",n,m,n-1,m);
            else
                printf("0 0\n%d %d\n%d 0\n0 %d\n",n,m,n,m);
        }else
         {
            if(sqrt(n*n+m*m)+n<2*sqrt(n*n+(m-1)*(m-1)))
                printf("0 1\n%d %d\n0 0\n%d %d\n",n,m,n,m-1);
            else
                printf("%d %d\n0 0\n%d 0\n0 %d\n",n,m,n,m);
        }
    }
    return 0;
}
