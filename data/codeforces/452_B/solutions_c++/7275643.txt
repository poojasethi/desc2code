#include<iostream>
#include<cmath>
using namespace std;

double dis(int ax  , int ay , int bx ,int by){
	return sqrt((ax - bx)*(ax-bx) + (ay - by)*(ay-by));
}
void pd(int a , int b){
	cout << a <<" " << b << endl;
}
int main()
{
	int n , m ;
	cin >> n >>  m ;
	if(n == 0 )
		pd(0,1),pd(0,m),pd(0,0),pd(0,m-1);
	else if (m == 0 )
		pd(1,0),pd(n,0),pd(0,0),pd(n-1,0);
	else if(n >= m){
		if(dis(0,0,n,m)+dis(n,m,0,m)+dis(0,m,n,0) - dis(0,1,n,m)-dis(n,m,0,0)-dis(0,0,n,m-1) > -1e-7)
			pd(0,0),pd(n,m),pd(0,m),pd(n,0);
		else
			pd(0,1),pd(n,m),pd(0,0),pd(n,m-1);
	}
	else{
	  if(dis(n,m,0,0)+ dis(0,0,0,m)+dis(0,m,n,0) - dis(n-1,m,0,0)-dis(0,0,n,m)-dis(n,m,1,0) > -1e-7)
		  pd(n,m),pd(0,0),pd(0,m),pd(n,0);
	  else
		  pd(n-1,m),pd(0,0),pd(n,m),pd(1,0);
	}
	return 0;

	return 0;
}
