#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<math.h>

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	if(n==0)
	{
		printf("0 1\n0 %d\n0 0\n0 %d",m,m-1);
	}
	else if(m==0)
	{
		printf("1 0\n%d 0\n0 0\n%d 0",n,n-1);
	}	
	else if(n==1) printf("0 0\n1 %d\n1 0\n0 %d",m,m);
	else if(m==1) printf("0 0\n%d 1\n0 1\n%d 0",n,n);
	else if(m>n)
	{
		if(sqrt(m*m+n*n)+m<2*sqrt(m*m+(n-1)*(n-1)))
		printf("1 0\n%d %d\n0 0\n%d %d",n,m,n-1,m);
		else printf("0 0\n%d %d\n%d 0\n0 %d",n,m,n,m);
	}
	else 
	{
		if(sqrt(m*m+n*n)+n<2*sqrt(n*n+(m-1)*(m-1)))
		printf("0 1\n%d %d\n0 0\n%d %d",n,m,n,m-1);
		else printf("0 0\n%d %d\n0 %d\n%d 0",n,m,m,n);
	}
	return 0;
}
