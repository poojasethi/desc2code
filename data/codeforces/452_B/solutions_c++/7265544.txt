#include <bits/stdc++.h>
#define sqr(a) ((a)*(a))
#define X first
#define Y second
using namespace std;
typedef pair<int, int> ii;

double dis(int a, int b, int c, int d){
    return sqrt(sqr(a-c)+sqr(b-d));
}


int main() {
    int n, m;
    cin >> n >> m;
    if(n==0) printf("0 1\n0 %d\n0 0\n0 %d", m, m-1);
    else if(m==0) printf("1 0\n%d 0\n0 0\n%d 0", n, n-1);
    else{
        
        if(n > m){
            if(dis(0, 1, n, m)*2 > dis(0, 0, n, m) + n)
            printf("0 1\n%d %d\n0 0\n%d %d", n,m,n,m-1);
            else printf("0 0\n%d %d\n0 %d\n%d 0", n, m, m, n);
        }
        else{
            if(dis(1, 0, n, m)*2 > dis(0, 0, n, m) + m)
            printf("1 0\n%d %d\n0 0\n%d %d", n,m,n-1,m);
            else printf("0 0\n%d %d\n%d 0\n0 %d", n, m, n, m);
        }
    }
}
