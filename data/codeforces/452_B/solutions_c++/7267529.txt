#include<bits/stdc++.h>
#define __________ ios_base::sync_with_stdio(0),cin.tie(0);
#define ll long long
using namespace std;

int main(){
  __________;
  int n,m,x[4],y[4];
  bool f=0;
  scanf("%d%d",&n,&m);
  if(n<m){
    swap(n,m);
    f=1;
  }
  if(m==0){
    y[0]=y[1]=y[2]=y[3]=0;
    x[0]=1;
    x[1]=n;
    x[2]=0;
    x[3]=n-1;
  }
  else if(sqrt(m*m+n*n)+n>sqrt(n*n+(m-1)*(m-1))*2){
    x[0]=0,y[0]=m;
    x[1]=n,y[1]=0;
    x[2]=0,y[2]=0;
    x[3]=n,y[3]=m;
  }
  else{
    x[0]=0,y[0]=1;
    x[1]=n,y[1]=m;
    x[2]=0,y[2]=0;
    x[3]=n,y[3]=m-1;
  }
  if(f){
    for(int i=0;i<4;i++)
      swap(x[i],y[i]);
  }
  for(int i=0;i<4;i++)
    printf("%d %d\n",x[i],y[i]);
}
