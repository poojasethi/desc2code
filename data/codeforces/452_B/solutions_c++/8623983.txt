#include<cstdio>
#include<cmath>

int n,m;

int main() {
	scanf("%d%d", &n, &m);
	if(n==0) printf("%d %d\n%d %d\n%d %d\n%d %d\n", 0, 1, 0, m, 0, 0, 0, m-1);
	else if(m==0) printf("%d %d\n%d %d\n%d %d\n%d %d\n", 1, 0, n, 0, 0, 0, n-1, 0);
	else if(n>=m) {
		double a = 2.*sqrt(1.*n*n+m*m) + n;
		double b = sqrt(1.*n*n+m*m) + 2.*sqrt((m-1)*(m-1)+n*n);
		if(b>a)
			printf("%d %d\n%d %d\n%d %d\n%d %d\n", 0, 1, n, m, 0, 0, n, m-1);
		else printf("%d %d\n%d %d\n%d %d\n%d %d\n", 0, 0, n, m, 0, m, n, 0);
	} else {
		double a = 2.*sqrt(1.*n*n+m*m) + m;
		double b = sqrt(1.*n*n+m*m) + 2.*sqrt((n-1)*(n-1)+m*m);
		if (b>a)
			printf("%d %d\n%d %d\n%d %d\n%d %d\n", 1, 0, n, m, 0, 0, n-1, m);
		else printf("%d %d\n%d %d\n%d %d\n%d %d\n", 0, 0, n, m, n, 0, 0, m);
	}

	return 0;
}
