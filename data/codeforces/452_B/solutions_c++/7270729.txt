#include<stdio.h>
#include<math.h>
int main()
{
int n,m,max;
while(scanf("%d%d",&n,&m)!=EOF)
{
if(n>m)
max=n;
else
max=m;
if(n==0)
printf("%d %d\n%d %d\n%d %d\n%d %d\n",0,1,n,m,0,0,n,m-1);
else if(m==0)
printf("%d %d\n%d %d\n%d %d\n%d %d\n",1,0,n,m,0,0,n-1,m);
else if(sqrt(n*n+m*m)+max>=2*sqrt(n*n+(m-1)*(m-1))&&sqrt(n*n+m*m)+max>=2*sqrt((n-1)*(n-1)+m*m))
{
if(n>=m)
printf("%d %d\n%d %d\n%d %d\n%d %d\n",0,0,n,m,0,m,n,0);
else
printf("%d %d\n%d %d\n%d %d\n%d %d\n",0,0,n,m,n,0,0,m);
}
else if(2*sqrt(n*n+(m-1)*(m-1))>=sqrt(n*n+m*m)+max&&2*sqrt(n*n+(m-1)*(m-1))>=2*sqrt((n-1)*(n-1)+m*m))
printf("%d %d\n%d %d\n%d %d\n%d %d\n",n,m-1,0,0,n,m,0,1);
else
printf("%d %d\n%d %d\n%d %d\n%d %d\n",n-1,m,0,0,n,m,1,0);

}
return 0;
}
