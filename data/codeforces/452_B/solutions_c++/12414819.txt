#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
int a,b,c,d,e,f,n,m;
double pythagoreantheorem(int x,int y)
{
     return sqrt(x*x+y*y);
}
int main()
{
     cin>>n>>m;
     if (!n)
     {
          b=1;f=m-1;
     }
     else if (!m)
     {
          a=1;e=n-1;
     }
     else if (n>m) 
     {
          e=n;
          if (n+pythagoreantheorem(n,m)>2*pythagoreantheorem(n,m-1))
          {
               d=m;
          }
          else
          {
               b=1;f=m-1;
          }
     }
     else
     {
          f=m;
          if (m+pythagoreantheorem(n,m)>2*pythagoreantheorem(n-1,m))
          {
               c=n;
          }
          else 
          {
               a=1;e=n-1;
          }
     }
     cout<<a<<" "<<b<<endl<<n<<" "<<m<<endl<<c<<" "<<d<<endl<<e<<" "<<f;
     return 0;
}
