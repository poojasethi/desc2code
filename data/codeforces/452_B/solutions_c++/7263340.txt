#include<bits/stdc++.h>
using namespace std;
#define len(x,y) (sqrt(1.0*(x)*(x)+1.0*(y)*(y)))
#define pnt(x,y) printf("%d %d\n",x,y)

int n,m;

double cal0()
{
	if(n==0||m==0)
		return 0;
	return 2*len(n,m)+max(n,m);
}

double caln()
{
	if(n==0)
		return 0;
	return len(n,m)+2*len(n-1,m);
}

double calm()
{
	if(m==0)
		return 0;
	return len(n,m)+2*len(n,m-1);
}

int main()
{
	scanf("%d%d",&n,&m);
	double s0=cal0(),s1=caln(),s2=calm();
	if(s0>=s1&&s0>=s2)
	{
		if(n>m)
		{
			pnt(0,0);
			pnt(n,m);
			pnt(0,m);
			pnt(n,0);
		}
		else
		{
			pnt(0,0);
			pnt(n,m);
			pnt(n,0);
			pnt(0,m);
		}
	}
	else if(s1>=s2)
	{
		pnt(n-1,m);
		pnt(0,0);
		pnt(n,m);
		pnt(1,0);
	}
	else
	{
		pnt(n,m-1);
		pnt(0,0);
		pnt(n,m);
		pnt(0,1);
	}
	return 0;
}
