#include <iostream>
#include <cmath>
using namespace std;
int main()
{int n,m; cin>>n>>m;
if(n==0)
{cout<<"0 1"<<endl<<"0 "<<m<<endl<<"0 0"<<endl<<"0 "<<m-1<<endl;
 return 0;
 }
if(m==0)
{cout<<"1 0"<<endl<<n<<" 0"<<endl<<"0 0"<<endl<<n-1<<" 0"<<endl;
 return 0;
}
if(m>=n)
 {if(2*sqrt(m*m+n*n)+m<sqrt(m*m+n*n)+2*sqrt(m*m+(n-1)*(n-1)))
  cout<<"1 0"<<endl<<n<<" "<<m<<endl<<"0 0"<<endl<<n-1<<" "<<m<<endl;
   else cout<<"0 0"<<endl<<n<<" "<<m<<endl<<n<<" 0"<<endl<<"0 "<<m<<endl;
 }
 else
 {if(2*sqrt(m*m+n*n)+n<sqrt(m*m+n*n)+2*sqrt((m-1)*(m-1)+n*n))
   cout<<"0 1"<<endl<<n<<" "<<m<<endl<<"0 0"<<endl<<n<<" "<<m-1<<endl;
   else cout<<"0 0"<<endl<<n<<" "<<m<<endl<<"0 "<<m<<endl<<n<<" 0"<<endl;
 }
return 0;
}
