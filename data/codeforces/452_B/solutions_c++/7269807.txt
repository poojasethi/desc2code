#include <iostream>
#include <cmath>
#include <cstdio>
#define rep(i,a) for(int i=0; i<a; ++i)
#define gi(x) scanf("%d",&x)

using namespace std;
int x,y;

int main() 
{
	gi(x); gi(y);
	if(x==0) printf("0 1\n0 %d\n0 0\n0 %d\n",y,y-1);
	else if(y==0) printf("1 0\n%d 0\n0 0\n%d 0\n",x,x-1);
	else {
		bool swp=false;
		if(x<y) {
			swap(x,y);
			swp=true;
		}
		double f=2.0*sqrt(x*x+(y-1)*(y-1))+sqrt(x*x+y*y);
		double s=2.0*sqrt(x*x+y*y)+double(x);
		int ax[4],ay[4];
		
		if(s<f) 
		{
			ax[0]=0; ay[0]=1;
			ax[1]=x; ay[1]=y;
			ax[2]=0; ay[2]=0;
			ax[3]=x; ay[3]=y-1;
		}
		else {
			ax[0]=0; ay[0]=0;
			ax[1]=x; ay[1]=y;
			ax[2]=0; ay[2]=y;
			ax[3]=x; ay[3]=0;
		}
		rep(i,4) 
			if(swp) printf("%d %d\n",ay[i],ax[i]);
			else printf("%d %d\n",ax[i],ay[i]);
	}
	return 0;
}
