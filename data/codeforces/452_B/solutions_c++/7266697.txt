#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#include<algorithm>
using namespace std;
double eps = 1e-7;
#define ll long long
int n, m, x[5], y[5];;
int main()
{
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> n >> m;
	if(n == 0 || m == 0)
	{
		if(n == 0)
		{
			y[1] = 1, y[2] = m, y[3] = 0, y[4] = m - 1;
		}
		else
		{
			x[1] = 1, x[2] = n, x[3] = 0, x[4] = n - 1;
		}
	}
	else
	{
		int mn = min(n, m);
		if(mn < 4)
		{
			x[1] = n, y[1] = m; x[3] = (n >= m ? n : 0); y[3] = (n >= m ? 0 : m);
			x[4] = (n >= m ? 0 : n), y[4] = (n >= m ? m : 0);
		} 	
		else
		{
			x[1] = (n < m ? 1 : 0), y[1] = (n < m ? 0 : 1);
			x[2] = n, y[2] = m;
			x[3] = 0, y[3] = 0;
			x[4] = (n <= m ? n - 1 : n), y[4] = (n <= m ? m : m - 1);
		}
	}
	for(int i = 1; i <= 4; ++i)
	{
		cout << x[i] << ' ' << y[i] << "\n";
	}
}
