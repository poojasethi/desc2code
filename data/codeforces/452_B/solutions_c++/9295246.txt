#include<cstdio>
#include<iostream>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
int n,m;
double sq(int num){return num*num;}
int main(){
    scanf("%d %d",&n,&m);
    if(n==0){
        printf("0 %d\n0 %d\n0 %d\n0 %d\n",1,m,0,m-1);
    }else if(m==0){
        printf("%d 0\n%d 0\n%d 0\n%d 0\n",1,n,0,n-1);
    }else if(n==1 && m==1){
        printf("0 0\n1 1\n1 0\n0 1\n");
    }else if(n>m){
        if(sqrt(sq(n)+sq(m))+n>2*sqrt(sq(n)+sq(m-1))) printf("%d %d\n%d %d\n%d %d\n%d %d\n",0,0,n,m,0,m,n,0);
        else printf("%d %d\n%d %d\n%d %d\n%d %d\n",n,m-1,0,0,n,m,0,1);
    }else{
        if(sqrt(sq(n)+sq(m))+m>2*sqrt(sq(n-1)+sq(m))) printf("%d %d\n%d %d\n%d %d\n%d %d\n",0,0,n,m,n,0,0,m);
        else printf("%d %d\n%d %d\n%d %d\n%d %d\n",n-1,m,0,0,n,m,1,0);
    }
    return 0;
}
