#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    if(n==0&&m!=0)printf("0 1\n0 %d\n0 0\n0 %d",m,m-1);
    else if(n!=0&&m==0)printf("1 0\n%d 0\n0 0\n%d 0",n,n-1);
    else
    {
        if((double)sqrt(max(n*n+(m-1)*(m-1),(n-1)*(n-1)+m*m))*2<max(((double)sqrt(n*n+m*m)+n),((double)sqrt(n*n+m*m)+m)))
        {
            if(n>m)printf("0 0\n%d %d\n0 %d\n%d 0",n,m,m,n);
            else printf("0 0\n%d %d\n%d 0\n0 %d",n,m,n,m);
        }
        else if(n*n+(m-1)*(m-1)>(n-1)*(n-1)+m*m)printf("%d %d\n0 0\n%d %d\n0 1",n,m-1,n,m);
        else printf("%d 0\n0 %d\n%d 0\n1 %d",n-1,m,n,m);
    }
    fclose(stdin);
    fclose(stdout);
    return 0;
}
