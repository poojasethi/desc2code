#include<iostream>
#define f first
#define s second
#define mp make_pair
using namespace std;

int n, m;
bool swapped;

pair<int, int> point[4];

int main() {
	cin >> n >> m;
	
	if(n > m) {
		swap(n, m);
		swapped = true;
	}
	
	if(n == 0) {
		point[0] = mp(0, 1);
		point[1] = mp(0, m);
		point[2] = mp(0, 0);
		point[3] = mp(0, m-1);
	}
	
	else if( 2*(n*n + m*m) + m*m   >   2*((n-1) * (n-1) + m*m) + n*n + m*m) {
		point[0] = mp(0, 0);
		point[1] = mp(n, m);
		point[2] = mp(n, 0);
		point[3] = mp(0, m);
	}
	
	else{
		point[0] = mp(1, 0);
		point[1] = mp(n, m);
		point[2] = mp(0, 0);
		point[3] = mp(n-1, m);
	}
	
	if(swapped)
		for(int i=0;i<4;i++)
			swap(point[i].f, point[i].s);
	
	for(int i=0;i<4;i++)
		cout << point[i].f << ' ' << point[i].s << '\n';
	
	return 0;
}
