#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
	int a,b;double c,d;
	scanf("%d%d",&a,&b);
	if(!a){printf("0 %d\n0 0\n0 %d\n0 1\n",b-1,b);return 0;}
	if(!b){printf("%d 0\n0 0\n%d 0\n1 0\n",a-1,a);return 0;}
	c=sqrt(a*a+b*b);
	if(a>b)
	{
		d=c+2*sqrt(a*a+(b-1)*(b-1));
		c=2*c+a;
		if(c>d)printf("0 0\n%d %d\n0 %d\n%d 0\n",a,b,b,a);
		else printf("%d %d\n0 0\n%d %d\n0 1\n",a,b-1,a,b);
		return 0;
	}
	d=c+2*sqrt((a-1)*(a-1)+b*b);
	c=2*c+b;
	if(c>d)printf("0 0\n%d %d\n%d 0\n0 %d\n",a,b,a,b);
	else printf("%d %d\n0 0\n%d %d\n1 0\n",a-1,b,a,b);
}
