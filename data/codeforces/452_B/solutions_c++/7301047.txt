#include <iostream>
#include <cmath>
using namespace std;
int main(int argc, char *argv[])
{
    double n,m;
    cin>>n>>m;
    if(n==0)cout<<"0 1 \n0 "<<m<<"\n0 0\n0 "<<m-1<<endl;
    if(m==0)cout<<"1 0 \n"<<n<<" 0\n0 0\n"<<n-1<<" 0"<<endl;
    if(n&&m)
    {
        if(n<m)
        {
            double x=sqrt(n*n+m*m)+m;
            double y=2*sqrt((n-1)*(n-1)+m*m);
            if(x>y)
                cout<<n<<" "<<m<<"\n0 0\n0 "<<m<<"\n"<<n<<" 0"<<endl;
            else
                cout<<"1 0\n"<<n<<" "<<m<<"\n0 0\n"<<n-1<<" "<<m<<endl;
        }
        else
        {
            double x=sqrt(n*n+m*m)+n;
            double y=2*sqrt((m-1)*(m-1)+n*n);
            if(x>y)
                cout<<n<<" "<<m<<"\n0 0\n"<<n<<" 0\n0 "<<m<<endl;
            else 
                cout<<"0 1\n"<<n<<" "<<m<<"\n0 0\n"<<n<<" "<<m-1<<endl;
        }
    }
    else 
        return 0;
}
