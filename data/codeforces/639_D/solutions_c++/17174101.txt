#include <bits/stdc++.h>
using namespace std;

const int maxN=2e5+100;

int n,k,a[maxN];
long long xx,yy;
vector<int> b[5];

long long cal(int fin,int sta) {
	long long s=0;
	while ((fin-sta)%5) s+=yy, sta++;
	s+=min(xx*(fin-sta)/5,yy*(fin-sta));
	return s;
}

int main() {
	cin>>n>>k>>xx>>yy;
	for(int i=0;i<n;i++) scanf("%i",&a[i]);
	sort(a,a+n);
	for(int i=0;i<5;i++) b[i].push_back(-1e9-2);
	for(int i=0;i<n;i++) b[(a[i]%5+5)%5].push_back(a[i]);
	long long res=1e18;
	for(int du=0;du<5;du++) {
		int trai[5]={1,1,1,1,1} ,phai[5]={0,0,0,0,0};
		long long ht=0;
		int dem=0;
		//printf("du: %i\n",du);
		int pre=a[k-1];
		while ((pre%5+5)%5!=du) ++pre;
		
		for(int i=k-1;i<n;i++) {
			int gt=a[i];		
			while ((gt%5+5)%5!=du) ++gt;
			//printf("i: %i %i\n",pre,gt);
			for(int j=0;j<5;j++) 
				while (phai[j]+1<b[j].size() && b[j][phai[j]+1]<=gt) {					
					++phai[j];
					++dem;
					ht+=cal(gt,b[j][phai[j]]);
				}
			ht+=min(xx*((gt-pre)/5),yy*(gt-pre))*k;
			//cout<<ht<<endl;
			pre=gt;
			while (dem>k) {
				int pos=-1;
				for(int j=0;j<5;j++) 
					if (trai[j]<=phai[j] && (pos==-1 || cal(gt,b[j][trai[j]]) > cal(gt,b[pos][trai[pos]])))
						pos=j;
				ht-=cal(gt,b[pos][trai[pos]++]);
				dem--;
			}
			//cout<<ht<<endl;
			res=min(res,ht);
		}
	}
	cout<<res;
	return 0;
}
