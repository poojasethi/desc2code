#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<int, int> pii;
typedef pair<ll, int> pli;
typedef pair<int, ll> pil;
typedef pair<ll, ll> pll;
#define pb push_back
inline int ri() {
	int x=0, f=1, c=getchar();
	for(; c<48||c>57; f=c=='-'?-1:f, c=getchar());
	for(; c>47&&c<58; x=x*10+c-48, c=getchar());
	return x*f;
}
const int N=200005;
int t[N], r[N];
int main() {
	int n=ri(), k=ri(), b=ri(), c=ri();
	b=min(b, c*5);
	for(int i=0; i<n; ++i) {
		t[i]=ri();
	}
	sort(t, t+n);
	ll ans=~0ull>>1;
	for(int j=0; j<5; ++j) {
		multiset<ll> s;
		ll add=0, tot=0;
		for(int i=0; i<n; ++i) {
			r[i]=t[i]+(j-t[i]%5+5)%5;
		}
		for(int i=0; i<n; ++i) {
			if(i>0 && r[i]>r[i-1]) {
				ll tmp=(ll)(r[i]-r[i-1])/5*b;
				add+=tmp;
				tot+=tmp*s.size();
			}
			int dis=(r[i]-t[i])*c;
			s.insert(dis-add);
			tot+=dis;
			if(s.size()>k) {
				ll tmp=*(--s.end());
				tot-=add+tmp;
				s.erase(--s.end());
			}
			if(s.size()==k) {
				ans=min(ans, tot);
			}
		}
	}
	cout << ans << endl;
	return 0;
}
