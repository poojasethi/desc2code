#include <bits/stdc++.h>

using namespace std;

#define NN 210000
#define inf 0x3f3f3f3f
#define eps 1e-8
#define eprintf(...) fprintf(stderr, __VA_ARGS__)

typedef pair<int, int> pii;
typedef long long INT;

int N, K, B, C;
int a[NN], r[NN];
multiset<INT> cur;

int main() {
	scanf("%d%d%d%d", &N, &K, &B, &C);
	for (int i=0; i<N; i++) scanf("%d", &a[i]);
	
	sort(a, a+N);
	
	if (B>=5*C) {
		INT sum=0;
		for (int i=0; i<K; i++) sum+=a[i];
		INT ans=(INT)a[K-1]*K-sum;
		for (int i=K; i<N; i++) {
			sum+=a[i]-a[i-K];
			ans=min(ans, (INT)a[i]*K-sum);
		}
		ans*=C;
		printf("%I64d\n", ans);
		return 0;
	}
	
	INT tot=0, ans=-1;
	for (int i=0; i<5; i++) {
		cur.clear(); tot=0;
		for (int j=0; j<N; j++) {
			int x=((i-a[j])%5+5)%5;
			int y=a[j]+x-i;
			INT val=C*x-(INT)B*y/5;
			if (cur.size()<K) cur.insert(val), tot+=val;
			else if (val<*cur.rbegin()) {
				INT temp=*cur.rbegin();
				tot-=temp;
				cur.erase(cur.find(temp));
				cur.insert(val);
				tot+=val;
			}
			
			if (j>=K-1) {
				if (ans==-1) ans=tot+(INT)B*y/5*K;
				else ans=min(ans, tot+(INT)B*y/5*K);
			}
		}
	}
	
	printf("%I64d\n", ans);
	
	return 0;
}
