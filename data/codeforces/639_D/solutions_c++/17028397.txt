#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>

using namespace std;
typedef long long LL;
const int maxn = 2e5+5;
const int INF = 0x3f3f3f3f;

LL n,k,b,c,a[maxn];
LL d[maxn][5];
priority_queue<LL> que;

int main(){
    int i,j;
    scanf("%I64d%I64d%I64d%I64d",&n,&k,&b,&c);
    if(b > 5*c) b = 5*c;
    for(i = 1;i <= n;i++){
        scanf("%I64d",&a[i]);a[i] += 1e9;
    }
    sort(a+1,a+n+1);
    for(i = 1;i <= n;i++){
        for(j = 0;j < 5;j++)
            d[i][(a[i]+j)%5] = a[i]+j;
    }
    LL ans = 1e18,sum,tmp;
    for(j = 0;j < 5;j++){
        while(!que.empty()) que.pop();
        sum = 0;
        for(i = 1;i <= k;i++){
            tmp = 1LL*c*(d[i][j]-a[i])+1LL*b*((d[k][j]-d[i][j])/5);
            que.push(tmp);
            sum += tmp;
        }
        ans = min(ans,sum);
        for(i = k+1;i <= n;i++){
            tmp = 1LL*c*(d[i][j]-a[i])+1LL*b*((d[k][j]-d[i][j])/5);
            sum += tmp-que.top();que.pop();
            que.push(tmp);
            ans = min(ans,sum+1LL*k*b*((d[i][j]-d[k][j])/5));
        }
    }
    printf("%I64d\n",ans);

    return 0;
}
