#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	int wyn = 0;
	for(int i = 0;i < n;++i)
	{
		int mini = 1e9 + 1;
	 for(int j = 0;j < m;++j)
	 {
		 int a;
		 scanf("%d",&a);
		 mini = min(a,mini);
	 }
	 wyn = max(mini,wyn);
	}
	printf("%d",wyn);
	return 0;
}
