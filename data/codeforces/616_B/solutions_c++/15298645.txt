#include <cstdio>

int main() {
	int i, j, st_min, av_max = 0, N, M, x;
	scanf("%d %d", &N, &M);
	for (i=0; i<N; ++i) {
		st_min = 1000000000;
		for (j=0; j<M; ++j) {
			scanf("%d", &x);
			if (x < st_min) {
				st_min = x;
			}
		}
		if (st_min > av_max) {
			av_max = st_min;
		}
	}
	printf("%d\n", av_max);
}
