#include<iostream>
using namespace std;
int main()
{
	int n,m;
	while(cin>>n>>m)
	{
		int a,min,max=0;
		for(int i=0;i<n;i++)
		{
			min=1000000010;
			for(int j=0;j<m;j++)
			{
				cin>>a;
				if(a<min) min=a;
			}
			if(min>max) max=min;
		}
		cout<<max<<endl;
	}
	return 0;
}
