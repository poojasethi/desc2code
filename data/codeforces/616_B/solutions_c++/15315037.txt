#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n,m;
	cin>>n>>m;
	int min = 1000000000,temp = 0;
	int max = 0;
	for(int i =0;i<n;++i)
	{
		min = 1000000000;
		for(int j = 0;j<m;++j)
		{
			cin>>temp;
			if(temp<min)
				min = temp;
		}

		if(max<min)
			max = min;
	}
	cout << max << "\n";
}
