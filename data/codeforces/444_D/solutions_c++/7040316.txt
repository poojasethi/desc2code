#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<map>
#include<vector>

using namespace std;

map<string,vector<int> > loc; 
map<pair<string,string>,int> dp;
string s,s1,s2;
int q;
vector<int> v1,v2;

int ask()
{
	int renum=1<<30;
	if(dp.upper_bound(make_pair(s1,s2))!=dp.lower_bound(make_pair(s1,s2))) return dp[make_pair(s1,s2)];
	v1=loc[s1];v2=loc[s2];int i=0,j=0;
	while(i<v1.size()&&j<v2.size())
	{
		int now=max(v1[i]+s1.size(),v2[j]+s2.size())-min(v1[i],v2[j]);
		if(now<renum) renum=now;
		if(i+1==v1.size()) j++;
		else if(j+1==v2.size()) i++;
		else if(v1[i]<v2[j]) i++;
		else j++;
	}
	if(renum==(1<<30)) renum=-1;
	dp[make_pair(s1,s2)]=renum;
	dp[make_pair(s2,s1)]=renum;
	return renum;
}

int main()
{
	cin>>s;
	for(int i=0;i<s.size();i++)
	{
		string tmp="";
		for(int j=1;j<=4;j++)
		{
			tmp+=s[i+j-1];
			loc[tmp].push_back(i);
		}
	}
	cin>>q;
	while(q--)
	{
		cin>>s1>>s2;
		cout<<ask()<<endl;
	}
	return 0;
}
