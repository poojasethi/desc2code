#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <vector>
#include <map>
#include <algorithm>

using namespace std;

const int INF=99999999;

map<string, vector<int> > occ;
map<pair<string, string>, int> ans;
string s;

int main()
{
    cin>>s;
    int len = s.size();
    for(int i = 0; i < len; i++){
        string tmp = "";
        for(int j = 0; j < 4 && i + j < len; j++){
            tmp += s[i + j];
            occ[tmp].push_back(i);
        }
    }
    int p;
    cin>>p;
    while(p--){
        string a, b;
        cin>>a>>b;
        if(ans[make_pair(a, b)]){
            cout<<ans[make_pair(a, b)]<<endl;
            continue;
        }
        vector<int> va(occ[a]), vb(occ[b]);
        if(va.empty() || vb.empty()){
            ans[make_pair(a, b)]=-1;
            cout<<"-1"<<endl;
            continue;
        }
        int inf = INF;
        vector<int>::iterator ia=va.begin(), ib=vb.begin();
        while(ia!=va.end() && ib!=vb.end()){
            inf = min(inf, int(max(a.size() + *ia, b.size() + *ib)) - min(*ia, *ib));
            if(*ia < *ib) ia++;
            else ib++;
        }
        cout<<inf<<endl;
        ans[make_pair(a, b)]=inf;
    }
    return 0;
}
