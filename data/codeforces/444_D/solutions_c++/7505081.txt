#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<vector>
#include<algorithm>
#include<map>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define N 550005
#define pb push_back
typedef pair<int,int> pii;
vector <int> a[N];
map <pii,int> done;
char s[N],s1[6],s2[6];
int n,m,res,h1,h2,ans;
int main(){
    scanf("%s",s);
    n=strlen(s);
    rep(i,n){
        res=0;
        rep(j,4){
            if(i+j>=n) break;
            res=res*27+s[i+j]-'a'+1;
            a[res].pb(i);
        }
    }
    scanf("%d",&m);
    while (m--){
        scanf("%s",s1);
        scanf("%s",s2);
        h1=h2=0;
        rep(i,strlen(s1)) h1=h1*27+s1[i]-'a'+1;
        rep(i,strlen(s2)) h2=h2*27+s2[i]-'a'+1;
        if(a[h1].size()==0||a[h2].size()==0){
            printf("-1\n"); continue;
        }
        if(a[h1].size()>a[h2].size()) swap(h1,h2),swap(s1,s2);
        if(done[pii(h1,h2)]){
            printf("%d\n",done[pii(h1,h2)]);
            continue;
        }
        ans=0x7f7f7f7f;
        rep(i,a[h1].size()){
            int p=lower_bound(a[h2].begin(),a[h2].end(),a[h1][i])-a[h2].begin();
            for (int j=p-1;j<=p+1;j++){
                if(j<0||j>=a[h2].size()) continue;
                int l1=a[h1][i],r1=l1+strlen(s1)-1;
                int l2=a[h2][j],r2=l2+strlen(s2)-1;
                ans=min(ans,max(r1,r2)-min(l1,l2)+1);
            }
        }
        done[pii(h1,h2)]=ans;
        printf("%d\n",ans);
    }
}
