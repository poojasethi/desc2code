#include <map>
#include <vector>
#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;

#define MK  make_pair
#define Vec vector<int>
#define PA  pair<string,string>
#define min(a,b)    ((a)<(b)?(a):(b))
#define max(a,b)    ((a)>(b)?(a):(b))
#define For(i,a,b)  for(int i=a;i<b;i++)

int     N,Q;
string  S,s1,s2;

map<PA,int>G;
map<string,Vec >A;

int     main(){
        cin>>S; N=S.length();
        For(i,0,N){
            string  t="";
            For(j,0,4)  if  (i+j<N) A[t+=S[i+j]].push_back(i);
        }
        for (scanf("%d",&Q);Q--;){
            cin>>s1>>s2;
            if  (s1>s2) swap(s1,s2);
            if  (G.count(MK(s1,s2)))    {printf("%d\n",G[MK(s1,s2)]);continue;}

            Vec V1=A[s1],V2=A[s2];
            int L1=V1.size(),L2=V2.size(),ret=1<<30,l1=s1.length(),l2=s2.length();
            for (int i=0,j=0;i<L1&&j<L2;V1[i]<V2[j]?i++:j++)
                ret=min(ret,max(V1[i]+l1,V2[j]+l2)-min(V1[i],V2[j]));
            printf("%d\n",G[MK(s1,s2)]=(ret==1<<30)?-1:ret);
        }
}
