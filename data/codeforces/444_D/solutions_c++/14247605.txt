#include<bits/stdc++.h>
using namespace std;

string s,s1,s2;
int n;
map<int,vector<int> > a;
map<pair<int,int>, int> dp;

int main(){
	ios_base::sync_with_stdio(0); cin.tie(0);	
//	freopen("in.txt","r",stdin);
	clock_t now=clock();
	cin >> s;
	for(int i=0; i<s.length(); i++)a[s[i]-'a'+1].push_back(i);
	if(s.length()>1)for(int i=0; i<s.length()-1; i++)a[s[i]-'a'+1+31*(s[i+1]-'a'+1)].push_back(i);
	if(s.length()>2)for(int i=0; i<s.length()-2; i++)a[s[i]-'a'+1+31*(s[i+1]-'a'+1)+961*(s[i+2]-'a'+1)].push_back(i);
	if(s.length()>3)for(int i=0; i<s.length()-3; i++)a[s[i]-'a'+1+31*(s[i+1]-'a'+1)+961*(s[i+2]-'a'+1)+29791*(s[i+3]-'a'+1)].push_back(i);
//	cout << (double)(clock()-now)/CLOCKS_PER_SEC << " s\n";
	cin >> n;
	for(int i=0; i<n; i++){
		cin >> s1 >> s2;
		int h1=0,h2=0;
		for(int i=s1.length()-1; i>=0; i--)h1=31*h1+s1[i]-'a'+1;
		for(int i=s2.length()-1; i>=0; i--)h2=31*h2+s2[i]-'a'+1;
		if(dp[make_pair(h1,h2)]>0){
			cout << dp[make_pair(h1,h2)] << endl;
			continue;
		}
		if(a[h1].size()==0||a[h2].size()==0){
			cout << -1 << endl;
			dp[make_pair(h1,h2)]=-1;
			continue;
		}
		vector<int> va(a[h1]),vb(a[h2]);
		int ans=1000000000;
		int i=0,j=0;
		while(i<(int)va.size()&&j<(int)vb.size()){
			ans=min(ans,max((int)s1.size()+va[i],(int)s2.size()+vb[j])-min(va[i],vb[j]));
			if(va[i]<vb[j])i++;
			else j++;
		}
		cout << ans << endl;
		dp[make_pair(h1,h2)]=ans;
	}
	return 0;
}
