using namespace std;

char s[50001],s1[5],s2[5];
vector<int> v1[551881];
map<pair<int, int>, int> m1;
int cur;
int len[551881];

int f1(char *x){
    int ans=0;
    for(int i=0;x[i]!=NULL;i++)
        ans*=27,ans+=x[i]-'a'+1;
    return ans;
}

int main(){
    int q;
    scanf("%s", s);
    for(int i=0;s[i]!=NULL;i++,cur=0)
        for(int j=0;j<4&&s[i+j]!=NULL;j++){
            cur*=27,cur+=s[i+j]-'a'+1;
            v1[cur].push_back(i);
            len[cur]=j+1;
        }
    scanf("%d", &q);
    for(int i=1;i<=q;i++){
        scanf("%s %s", s1, s2);
        int t1=f1(s1),t2=f1(s2),i1=0,i2=0,ans=1e9;
        if(!len[t1]||!len[t2]){
            printf("-1\n");
            continue;
        }
        if(m1[make_pair(t1,t2)]){
            printf("%d\n", m1[make_pair(t1,t2)]);
            continue;
        }
        while(i1<v1[t1].size()&&i2<v1[t2].size()){
            if(v1[t1][i1]<v1[t2][i2]){
                while(i1+1<v1[t1].size()&&v1[t1][i1+1]<v1[t2][i2])
                    i1++;
            }
            else {
                while(i2+1<v1[t2].size()&&v1[t2][i2+1]<v1[t1][i1])
                    i2++;
            }
            ans=min(ans,max(v1[t1][i1]+len[t1], v1[t2][i2]+len[t2])
                -min(v1[t1][i1],v1[t2][i2]));
            if(v1[t1][i1]<v1[t2][i2])
                i1++;
            else i2++;
        }
        printf("%d\n",ans);
        m1[make_pair(t1,t2)]=ans;
    }
    return 0;
}
