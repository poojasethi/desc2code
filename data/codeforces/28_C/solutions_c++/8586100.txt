#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cmath>
using namespace std;

int n,m;
int a[51];
double f[51][51][51],c[51][51],ans;

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;i++) scanf("%d",&a[i]);
	for (int i=0;i<=n;i++)
	{
		c[i][0]=1;
		for (int j=1;j<=i;j++) c[i][j]=c[i-1][j-1]+c[i-1][j];
	}
	f[0][0][0]=1;
	for (int j=1;j<=m;j++)
		for (int i=0;i<=n;i++)
			for (int k=0;k<=n;k++)
				for (int l=0;l<=n-i;l++)
					f[i+l][j][max(k,l/a[j]+(l%a[j]!=0))]+=f[i][j-1][k]*c[n-i][l]*pow((double)(m-j)/(m-j+1),n-i-l)*pow((double)1/(m-j+1),l);
	for (int i=0;i<=n;i++) ans+=i*f[n][m][i];
	printf("%.9lf\n",ans);
}
