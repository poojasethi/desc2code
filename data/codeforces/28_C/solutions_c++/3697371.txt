#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <vector>
#include <cmath>
using namespace std;


#define fr(i,a,b) for(int i=a;i<b;++i)





int n,m;


int v[1000];
double dp[51][51][51];
double comb[51][51];
bool mk[51][51][51];

double go(int w, int q, int mai){
	if(mk[w][q][mai]) return dp[w][q][mai];
	mk[w][q][mai] = true;
	if(w == m-1) return dp[w][q][mai] = max(mai, (q+v[w]-1)/v[w]) * pow(1.0/m, q);
	dp[w][q][mai] = 0.0;
	double pot = 1.0;
	fr(i,0,q+1){
		dp[w][q][mai] += go(w+1, q-i, max(mai, (i+v[w]-1)/v[w])) * pot * comb[q][i];
		pot *= 1.0/m;
	}
	return dp[w][q][mai];
}






int main(){
	comb[0][0] = 1;
	fr(i,1,51){
		comb[i][0] = comb[i][i] = 1;
		fr(j,1,i) comb[i][j] = comb[i-1][j-1] + comb[i-1][j];
	}
	scanf("%d %d",&n,&m);
	fr(i,0,m) scanf("%d",&v[i]);
	printf("%.12lf\n",go(0,n,0));
	return 0;
}
