#include <iostream>
#include <cstring>
#include <iomanip>
using namespace std;
typedef long double ldb;
int st,room;
int a [55];
ldb dp[55][55],C[55][55];
inline ldb solve (int M){
    memset(dp,0,sizeof dp);
    dp[0][0] = 1;
    for (int i=1; i<=room; i++){
        for (int j=0; j<=st; j++)
            for (int z=0; z<=M*a[i] && z<=j; z++)
                dp[i][j]+= dp[i-1][j-z] * C[j][z];
    }
    return dp[room][st];
}
int main(){
    cin >> st >> room;
    for (int i=1; i<=room; i++)
        cin >> a[i];
    C[0][0]=1;
    for (int i=1; i<55; i++){
        C[i][0]=1;
        for (int j=1; j<55; j++)
            C[i][j] = C[i-1][j] + C[i-1][j-1];
    }
    ldb ret = 0;
    ldb TOT = solve(st);
    ldb past=0,ans=0;
    for (int MAX=1; MAX<=st; MAX++){
        ldb cur = solve(MAX);
        ans+= (ldb)MAX * (cur-past) / TOT;
        past=cur;
    }
    cout << fixed << setprecision(10) << ans << endl;   
    return 0;
}
