#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
double dp[51][51][51];
long long c[51][51];
int a[51];
double p[51];
int n,m;
double get(int x,int y,int z)
{
    if(x==m)
    {
        if(y==0)
        return z;
        return 0;
    }
    if(!(dp[x][y][z]<0))
    return dp[x][y][z];
    double ret=0;
    for(int i=0;i<y+1;i++)
    {
        int temp=(i+a[x+1]-1)/a[x+1];
        ret+=c[y][i]*p[i]*get(x+1,y-i,max(temp,z));
    }
    dp[x][y][z]=ret;
    return ret;
}
int main()
{
    scanf("%d%d",&n,&m);
    memset(c,0,sizeof(c));
    for(int i=1;i<=m;i++)
    scanf("%d",&a[i]);
    p[0]=1;
    for(int i=1;i<51;i++)
    p[i]=p[i-1]/m;
    for(int i=0;i<51;i++)
    c[i][0]=c[i][i]=1;
    for(int i=1;i<51;i++)
    {
        for(int j=1;j<i;j++)
        c[i][j]=c[i-1][j]+c[i-1][j-1];
    }
    fill(&dp[0][0][0],&dp[51][0][0],-1);
    printf("%.10lf\n",get(0,n,1));
}
