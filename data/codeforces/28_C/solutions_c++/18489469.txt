#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

#define N 100

int n, m;
int a[N];

double dp[N][N][N];
double C[N][N];
double p[N][N];

int main() {
	for(int i = 0; i < N; ++i) {
		C[i][0] = C[i][i] = 1;
		for(int j = 1; j < i; ++j)
			C[i][j] = C[i-1][j-1] + C[i-1][j];
	}
	p[0][0] = 1;
	for(int i = 1; i < N; ++i) {
		p[i][0] = 1;
		for(int j = 1; j < N; ++j) {
			p[i][j] = p[i][j-1] * i;
		}
	}

	scanf("%d%d", &n, &m);
	for(int i = 1; i <= m; ++i) scanf("%d", &a[i]);
	dp[0][0][0] = 1;

	for(int i = 0; i < m; ++i) {
		for(int j = 0; j <= n; ++j) {
			for(int k = 0; k <= n; ++k) {
				if(dp[i][j][k] < 1e-12) continue;
				for(int u = 0; u + j <= n; ++u) {
					int t = u / a[i+1];
					if(u % a[i+1]) ++t;
					t = max(k, t);
					dp[i+1][j+u][t] += dp[i][j][k] * C[j+u][u] * p[i][j] / p[i+1][u+j];
				}
			}
		}
	}
	double ans = 0;
	for(int k = 0; k <= n; ++k) {
		ans += dp[m][n][k] * k;
	}
	printf("%.12lf\n", ans);
	return 0;
}
