#include <bits/stdc++.h>
using namespace std; 


int n,m ; 
int a[55];
double dp[55][55][55];
double C[55][55];
int main() {
   scanf("%d%d", &n, &m); 
   for(int i=1; i<=m; i++) {
       scanf("%d", &a[i]); 
   }
   
   C[0][0] =1; 
   for(int i=1; i<=50; i++) {
       C[i][0] = 1; 
       for(int j=1; j<=i; j++) C[i][j] = C[i-1][j] + C[i-1][j-1]; 
   }
   
   for(int i=0; i<=m; i++) for(int j=0; j<=50; j++) dp[i][0][j] = j; 
   for(int i=1; i<=m; i++) {
       for(int j=1; j<=n; j++) {
            for(int k=0; k<=50; k++) {
                for(int l = 0; l<=j; l++) {
                    int cur = l / (a[i]) + (l % a[i] != 0); 
                    dp[i][j][k] += C[j][l] * dp[i-1][j-l][max(cur, k)]; 
                }
            }
       }
   }
   
 printf("%.9f\n", dp[m][n][0] / pow(m,n)); 
    
    return 0; 
}
