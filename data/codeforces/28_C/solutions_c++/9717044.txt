#include <cstdio>
#include <algorithm>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)

double A,B,f[60][60][60],p[60];
int i,j,k,l,n,m,a[60];

int main(){
        scanf("%d%d",&n,&m);
        REP(i,0,m) scanf("%d",&a[i]);
        p[0]=1;REP(i,1,n+1) p[i]=p[i-1]*i;
        f[0][0][0]=1;
        REP(i,0,m)REP(j,0,n+1)REP(k,0,n+1)REP(l,0,n+1-j)f[i+1][j+l][max(k,(l+a[i]-1)/a[i])]+=p[l+j]/p[l]/p[j]*f[i][j][k];
        A=B=0;
        REP(i,1,n+1) A+=i*f[m][n][i],B+=f[m][n][i];
        //printf("%.lf %.lf\n",A,B);
        printf("%.9lf",A/B);
}
