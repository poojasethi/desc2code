#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <vector>
#include <queue>
#define maxn 55
#define maxm
#define INF
#define MOD
using namespace std;
double C[maxn][maxn],dp[maxn][maxn][maxn];
int n,m,a[maxn];
int main()
{
	for(int i=0;i<maxn;i++)
	{
		C[i][i]=C[i][0]=1;
		for(int j=1;j<i;j++)
			C[i][j]=C[i-1][j]+C[i-1][j-1];
	}
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++)
		scanf("%d",&a[i]);
	memset(dp,0,sizeof(dp));
	dp[0][0][0]=1;
	for(int i=0;i<m;i++)
	{
		for(int j=0;j<=n;j++)
		{
			for(int k=0;k<=j;k++)
			{
				for(int l=j;l<=n;l++)
				{
					dp[i+1][l][max(k,(l-j+a[i+1]-1)/a[i+1])]+=C[l][j]*dp[i][j][k];
				}
			}
		}
	}
	double sum=0,cnt=0;
	for(int i=0;i<=n;i++)
	{
		sum+=dp[m][n][i]*i;
		cnt+=dp[m][n][i];
	}
	printf("%.20f\n",sum/cnt);
	//system("pause");
	return 0;
}
