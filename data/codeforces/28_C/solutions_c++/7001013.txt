#include <cstdio>
#include <algorithm>
using namespace std;
double dp[55][55][55],C[55][55];

void init(){
    C[0][0] = 1;
    for(int i = 1;i <= 50;i++){
        C[i][0] = C[i][i] = 1;
        for(int j = 1;j < i;j++)
            C[i][j] = C[i-1][j-1]+C[i-1][j];
    }
}

int main(){
    init();
    int n,m,x;
    scanf("%d%d",&n,&m);
    dp[0][0][0] = 1;
    for(int i = 1;i <= m;i++){
        scanf("%d",&x);
        for(int j = 0;j <= n;j++){
            for(int k = j;k <= n;k++){
                for(int t = 0;t <= n;t++){
                    dp[i][k][max(t,(k-j+x-1)/x)] += dp[i-1][j][t]*C[k][j];
                }
            }
        }
    }
    double cnt = 0,sum = 0;
    for(int i = 1;i <= n;i++){
        cnt += dp[m][n][i] * i;
        sum += dp[m][n][i];
    }
    printf("%.12f\n",cnt/sum);
    return 0;
}
