#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#define fou(i,j,k) for (int i=j;i<=k;i++)
#define fod(i,j,k) for (int i=j;i>=k;i--)
using namespace std;

const int maxn=55;
const double epsi=1e-12;

int m,n;
int a[maxn];
double c[maxn][maxn]; 
double f[maxn][maxn][maxn];

void init()
{
	scanf("%d%d",&n,&m);
	fou(i,1,m){
		scanf("%d",&a[i]);
	}
}

void solve()
{
	fou(i,0,n){
		c[i][0]=1;
		fou(j,1,i)
			c[i][j]=c[i-1][j-1]+c[i-1][j];
	}
	
	memset(f,0,sizeof(f));
	f[0][0][0]=1;
	fou(i,0,m-1)
		fou(j,0,n)
			fou(k,0,n)
			if (f[i][j][k]<-epsi || f[i][j][k]>epsi)
				fou(jj,0,n-j) 
					f[i+1][j+jj][max(k,(int)ceil((double)jj/a[i+1]))]+=c[j+jj][j]*f[i][j][k];
	double x,s;
	x=s=0;
	fou(i,0,n){
		x+=i*f[m][n][i];
		s+=f[m][n][i];
	}
	printf("%.10lf\n",x/s);
}

int main()
{
    init();
    solve();
	return 0;
}
