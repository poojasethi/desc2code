#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;

double dp1[51][51][100];
double dp2[100][100];

double power(double a,double b,int p)
{
if(a==0)
return 1.00;
double temp=1.00;
for(int i=1;i<=p;i++)
temp=temp*(a/b);
return temp;
}

double func(int *a,int m,int n)
{
dp2[0][0]=1.00;
for(int i=1;i<=50;i++)
{
dp2[i][0]=1.00;
for(int j=1;j<=i;j++)
dp2[i][j]=dp2[i-1][j]+dp2[i-1][j-1];
}
for(int i=0;i<=n;i++)
for(int j=0;j<=m;j++)
for(int k=0;k<=n;k++)
if(i==0&&j==0)
dp1[i][j][k]=k;
else if(j==0)
dp1[i][j][k]=0;
else
for(int p=0;p<=i;p++)
{
int temp=(int)(ceil((p+(long double).0)/a[j-1])+0.5);
dp1[i][j][k]+=dp2[i][p]*power(1,j,p)*power(j-1,j,i-p)*dp1[i-p][j-1][max(temp,k)];
}
return dp1[n][m][0];
}

int main()
{
ios_base::sync_with_stdio(false);
cout<<fixed;
cout.precision(10);
int n,m;
cin>>n>>m;
int a[m];
for(int i=0;i<m;i++)
cin>>a[i];
cout<<func(a,m,n)<<endl;
return 0;
}
