#include <bits/stdc++.h>
using namespace std;

const int N = 55;
double dp[N][N][N];
int a[N];
double c[N][N];

int divf(int x,int y)
{
    return x % y != 0 ? x / y + 1 : x / y;
}

int main()
{
    int n,m;
    scanf("%d%d",&m,&n);
    for(int i = 1; i <= n; i ++) scanf("%d",&a[i]);
    for(int i = 0; i < N; i ++) {
        c[i][i] = c[i][0] = 1;
        for(int j = 1; j < i; j ++)
            c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
    }
    dp[0][0][0] = 1;
    for(int i = 0; i < n; i ++) {
        for(int j = 0; j <= m; j ++) {
            for(int k = 0; k <= m; k ++) {
                for(int r = 0; r + j <= m; r ++) {
                    dp[i + 1][r + j][max(k,divf(r,a[i + 1]))] += dp[i][j][k] * c[m - j][r];
                }
            }
        }
    }
    double ans = 0;
    for(int i = 1; i <= m; i ++)
        ans += i * dp[n][m][i];
    for(int i = 1; i <= m; i ++) ans /= n;
    printf("%.20lf\n",ans);
    return 0;
}
