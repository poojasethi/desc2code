#include <cstdio>
#include <iostream>

#define CS					const static
#define Rep(i,l,r)				for(int i = (l) ; i <= (r) ; i++)
#define RepC(i,l,r)				for(int i = (l) , _ = (r) ; i <= _ ; i++)

using namespace std ;

inline int RD()		{int x ; scanf("%d",&x) ; return x ;}
inline int RD(int & x)	{return x = RD() ;}

CS int N = 60 ;
CS double EPS = 1e-25 ;

double f[N][N][N] ;

double P[N] ;

int a[N] ;

int n,m ;

int main()
{
	RD(n) , RD(m) ;
	Rep(i,1,m)	RD(a[i]) ;

	P[0] = 1 ;
	Rep(i,1,n)	P[i] = P[i-1] * i ;

	f[0][0][0] = 1 ;
	
	RepC(i,0,m-1)	Rep(j,0,n)	RepC(l,0,n)	if(f[l][i][j] > EPS)
		RepC(p,0,n-j)
		{
			int o = max(l , (p + a[i+1] -1)/a[i+1]) ;
			f[o][i+1][j+p] += f[l][i][j] * P[j+p] / P[j] / P[p] ;
		}

	double cnt = 0 ;
	Rep(l,1,n)	cnt += f[l][m][n] ;

	double len = 0 ;
	Rep(l,1,n)	len += f[l][m][n]/cnt * (double)l ;

	printf("%.20lf",len) ;

	return 0;
}
