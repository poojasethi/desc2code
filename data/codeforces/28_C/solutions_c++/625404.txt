#include <stdio.h>

int p[55];
int n, m;

double solve() {
	double dp[55] = {};
	int i, j, k;
	dp[0] = 1.0;
	for (i=1;i<=m;i++) {
		for (j=n-1;j>=0;--j) {
			double f = 1.0;
			for (k=1;j+k<=n && k<=p[i];k++) {
				f *= (j+k)/(double)k/m;
				dp[j+k] += f * dp[j];
			}
		}
	}
	return dp[n];
}

int main(void) {
	scanf("%d%d", &n, &m);
	int a[55], i, j;
	for(i=1;i<=m;i++)
		scanf("%d", &a[i]);
	double ans = 0;
	for(i=0;i<=n;i++) {
		for(j=1;j<=m;j++)
			p[j] = a[j] * i;
		ans += (1.0 - solve());
	}
	printf("%.15lf\n", ans);
	return 0;
}
