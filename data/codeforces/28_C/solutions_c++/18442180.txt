#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<cmath>
#include<iostream>
#define N 55
using namespace std;
double f[N][N][N],c[N][N];
int a[N],n,m;
double sum,tot;
int main()
{
	c[0][0]=1;
	for(int i=1;i<N;i++)
	{
		c[i][0]=1;
		for(int j=1;j<=i;j++) c[i][j]=c[i-1][j]+c[i-1][j-1];
	}
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++) scanf("%d",&a[i]);
	f[0][0][0]=1;
	for(int i=1;i<=m;i++)
		for(int k=0;k<=n;k++)
			for(int l=0;l<=n;l++)
			{
				if(f[i-1][k][l]==0) continue;
				for(int j=k;j<=n;j++)
				{
					int t=ceil((double)(j-k)/(double)a[i]);
					t=max(t,l);
					f[i][j][t]+=f[i-1][k][l]*c[n-k][j-k];
				}
			}
	for(int i=0;i<=n;i++) sum+=f[m][n][i]*i,tot+=f[m][n][i];
	printf("%.20lf\n",sum/tot);
	return 0;
}
