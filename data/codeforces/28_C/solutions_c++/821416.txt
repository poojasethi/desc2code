#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<iostream>
using namespace std;
double c[55][55],dp[55][55][55];
int a[55];
int n,m;
int main()
{
    c[0][0]=1;
    for(int i=1; i<=50; i++)
    {
        c[i][i]=c[i][0]=1;
        for(int j=1; j<i; j++) c[i][j]=c[i-1][j-1]+c[i-1][j];
    }
    while(scanf("%d%d",&n,&m)==2)
    {
        for(int i=0;i<m;i++) scanf("%d",a+i);
        memset(dp,0,sizeof(dp));
        dp[0][n][0]=1;
        for(int i=0;i<m;i++)
            for(int j=0;j<=n;j++)
                for(int k=0;k<=n;k++)
                {
                    for(int l=0;l<=j;l++)
                    {
                        int kk=max(k,(l+a[i]-1)/a[i]);
                        dp[i+1][j-l][kk]+=c[j][l]*dp[i][j][k];
                    }
                }
        double a=0,b=0;
        for(int i=0;i<=n;i++)
        {
            a+=i*dp[m][0][i];
            b+=dp[m][0][i];
        }
        printf("%.12f\n",a/b);
    }
    return 0;
}
