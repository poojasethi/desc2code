#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

int N, M, basin[55];
double comb[55][55];
double dp[55][55][55];
int main()
{
	scanf("%d%d", &N, &M);
	for (int i = 1; i <= M; i++)
		scanf("%d", &basin[i]);
	
	for (int i = 0; i <= N; i++)
	{
		comb[i][0] = 1;
		for (int j = 1; j <= i; j++)
			comb[i][j] = comb[i-1][j] + comb[i-1][j-1];
	}
	
	for (int x = 0; x <= N; x++)
		dp[0][0][x] = x;
	for (int m = 1; m <= M; m++)
	for (int n = 0; n <= N; n++)
	for (int x = 0; x <= N; x++)
	{
		for (int k = 0; k <= n; k++)
		{
			double prob = comb[n][k] * pow(1.0/m, k)* pow(1.*(m-1)/m, n-k);
			dp[m][n][x] += prob * dp[m-1][n-k][max(x, (k+basin[m]-1)/basin[m])];
		}
	}
	printf("%.20lf\n", dp[M][N][0]);
}
