#include<iostream>
#include<cstdio>
#include<cstring>
#define rep(i,a,b) for (int i=a;i<=b;i++)
using namespace std;
#define N 55
int n,m,d[N];
double f[N][N],c[N][N],p[N][N],last,now,ans;
double doit(int a,int b,int k)
{   if (b==1){
	   if (a==k)return 1;
	   return 0;
	}
	return c[a][k]*p[b-1][a-k]/p[b][a];
}
double solve(int limit)
{  memset(f,0,sizeof(f));
   f[0][0]=1.0;
   rep(i,1,m){
	   int maxn=limit*d[i];
	   rep(j,0,n)
  		   rep(k,0,n)
  		   	  if (j-k<=maxn) f[i][j]+=f[i-1][k]*doit(n-k,m-i+1,j-k);	
   }
   return f[m][n];
}
int main()
{     
	scanf("%d%d",&n,&m);
	rep(i,1,m)scanf("%d",&d[i]);
	c[0][0]=1;
	rep(i,1,n){
		c[i][0]=1;
		rep(j,1,n)
		   c[i][j]=c[i-1][j]+c[i-1][j-1];
	}
	rep(i,1,m){
		p[i][0]=1;
 		rep(j,1,n)
 		     p[i][j]=p[i][j-1]*(double)i;
	}
    last=0.0;ans=0.0;
	rep(i,1,n){
		 now=solve(i);
		 ans+=(now-last)*i;
		 last=now;
	}
	printf("%.20lf\n",ans);
}
