#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

double f[55][55][55],h[55],ans1,ans2;
int n,m,a[55];

int main()
{
	
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;i++) scanf("%d",&a[i]);
	h[0]=1;for (int i=1;i<=n;i++) h[i]=h[i-1]*i;
	f[0][0][0]=1;
	for (int i=0;i<m;i++)
	{
		for (int j=0;j<=n;j++)
		{
			for (int k=0;k<=n;k++)
			{
				for (int p=0;p<=n-j;p++)
				{
					f[i+1][j+p][max(k,int(ceil(double(p)/a[i+1])))]+=h[j+p]/h[p]/h[j]*f[i][j][k];
				}
			}
		}
	}
	for (int i=1;i<=n;i++) ans1+=f[m][n][i]*i,ans2+=f[m][n][i];
	printf("%.10f\n",ans1/ans2);
	
	return 0;
}
