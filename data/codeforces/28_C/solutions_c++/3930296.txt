#include <stdio.h>
int max(int a,int b){
    if(a<b)
        return b;
    return a;
}
int n,m,v[50];
double dp[51][50][51];
bool vis[51][50][51];
long long nCk[51][51];
double calc(int N,int i,int mx){
    if(i==m)
        return (double)mx*!N;
    if(!vis[N][i][mx]){
        vis[N][i][mx]=true;
        double p=1.0;
        dp[N][i][mx]=0;
        for(int j=0;j<=N;p/=m,++j)
            dp[N][i][mx]+=p*nCk[N][j]*calc(N-j,i+1,max(mx,(j+v[i]-1)/v[i]));
    }
    return dp[N][i][mx];
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;++i)
        scanf("%d",v+i);
    nCk[0][0]=1;
    for(int i=1;i<51;++i)
        for(int j=0;j<=i;++j)
            if(!i || !j)
                nCk[i][j]=1;
            else
                nCk[i][j]=nCk[i-1][j]+nCk[i-1][j-1];
    printf("%.16lf\n",calc(n,0,0));
    return 0;
}
