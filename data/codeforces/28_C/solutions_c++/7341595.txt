#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,a[51];
double dp[52][52][52],C[52][52],P[51];
int main(){
  scanf("%d%d",&n,&m);
  for(int i=1;i<=m;i++)
    scanf("%d",&a[i]);
  for(int i=0;i<=n;i++){
    C[i][0] = 1.0; 
    for(int j=1;j<=i;j++)
      C[i][j] = C[i-1][j-1] + C[i-1][j];
  }
  P[0] = 1.0;
  for(int i=1;i<=n;i++)
    P[i] = P[i-1]/m;
  dp[0][0][0] = 1.0;
  for(int i=1;i<=m;i++)
    for(int j=0;j<=n;j++)
      for(int k=0;k<=n;k++)
    for(int r1=0;r1<=n-j;r1++){
      dp[i][j+r1][max(k,(r1+a[i]-1)/a[i])] += dp[i-1][j][k]*C[j+r1][r1]*P[r1];
    }
  double ans = 0.0;
  for(int i=1;i<=n;i++)
    ans += dp[m][n][i]*i;
  printf("%.12lf\n",ans);
  return 0;
}
