#include <bits/stdc++.h>
using namespace std;

#define M 51

int a[M];
double f[M][M][M];

long long c[M][M];

void prepare(){
	for(int i = 0; i < M; i ++) c[i][0] = c[i][i] = 1;
	for(int i = 1; i < M; i ++) {
		for(int j = 1; j < i;j ++) c[i][j] = c[i-1][j] + c[i-1][j-1];
	}
}

int main() {
	///freopen("c.in", "r", stdin);
	prepare();
	int n, m;
	scanf("%d %d", &n, &m);
	for(int i = 1; i <= m; i ++) scanf("%d", &a[i]);
	///cout << pow(0., 0) << endl;
    for(int i = 1; i <= n; i ++) {
        f[i][0][0] = 1.;
		for(int j = 1; j <= m; j ++) {
            for(int k = 0; k <= n; k ++) {
                for(int r = 0; r <= k && r <= a[j] * i; r ++) {
                    f[i][j][k] += 1. * f[i][j-1][k-r] * pow(1./j, r) * pow(1 - 1./j, k - r) * c[k][r];
                }
            }
		}
		///printf("%lf\n", f[i][m][n]);
    }
    double ans = 0;
    for(int i = n; i >= 1; i --) ans += 1. * i * (f[i][m][n] - f[i-1][m][n]);
    printf("%.100lf\n", ans);
}
