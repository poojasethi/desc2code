#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 51

int i,n,m;
double F[lim],M[lim][lim][lim];
int A[lim];
bool B[lim][lim][lim];

double solve(int i,int j,int k)
{
  double &res=M[i][j][k];
	int p,s;
	
	if(!B[i][j][k])
	{
	  if(j==m)
		  res=k;
		else
		{
		  s=m-j;
	    for(res=p=0;p<=i;p++)
		    res+=solve(i-p,j+1,max(k,(p+A[j]-1)/A[j]))*F[i]*pow(s-1.,(double)i-p)/(F[p]*F[i-p]*pow((double)s,(double)i));
		}
		B[i][j][k]=1;
		//printf("%d %d %d %.2lf\n",i,j,k,res);
	}
	return res;
}
int main()
{
  F[0]=1;
  for(i=1;i<lim;i++)
	  F[i]=F[i-1]*i;
  scanf("%d%d",&n,&m);
  for(i=0;i<m;i++)
    scanf("%d",&A[i]);
  printf("%.20lf\n",solve(n,0,1));		
  return 0;
}
