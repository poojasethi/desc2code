#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
using namespace std;
double c[51][51],p[51][51],f[51][51];
int a[51],n,m;
void init()
{
	int i,j,nn;
	scanf("%d%d",&n,&m);
	nn=max(n,m);
	for(i=1;i<=m;i++)scanf("%d",&a[i]);
	for(i=0;i<=n;i++)
	{
		c[i][0]=1.0;
		for(j=1;j<=n;j++)
			c[i][j]=c[i-1][j-1]+c[i-1][j];	
	}	
	for(i=1;i<=nn;i++)
	{
		p[i][0]=1.0;
		for(j=1;j<=nn;j++)
			p[i][j]=p[i][j-1]*i;	
	}
}
double cal_2(int a,int b,int cc)
{
	if(b==1)
	{
		if(a==cc)return 1;
		return 0;	
	}		
	return c[a][cc]*p[b-1][a-cc]/p[b][a];
}
double cal_1(int limit)
{
	int i,j,k,zd;
	memset(f,0,sizeof(f));
	f[0][0]=1;
	for(i=1;i<=m;i++)
	{
		zd=limit*a[i];
		for(j=0;j<=n;j++)
			for(k=0;k<=j;k++)
				if(j-k<=zd)
					f[i][j]+=f[i-1][k]*cal_2(n-k,m-i+1,j-k);		
	}
	return f[m][n];	
}
void solve()
{
	int i,j,k;
	double last=0.0,now,ans=0.0;
	for(i=1;i<=n;i++)
	{
		now=cal_1(i);
		ans+=(now-last)*i;
		last=now;
	}
	printf("%.10lf\n",ans);	
}
int main()
{
	init();
	solve();
	return 0;	
}
