#include <stdio.h>
#include <algorithm>
#include <vector>
#include <math.h>
#include <iostream>
using namespace std;
#define ll long long int
double dp[60][60][60];
int ar[60];
ll combi[51][51];
void pre(){
	for(int i=0;i<=50;++i)combi[i][0]=combi[i][i]= 1;
	for(int i=2;i<=50;++i)
		for(int j=1;j<i;++j)
			combi[i][j]= combi[i-1][j]+combi[i-1][j-1];
}
int main(){
	pre();
	int n,m;cin>>n>>m;
	for(int i=0;i<m;++i)
		cin>>ar[i];
	dp[m][n][0]= 1;//index , remain , max
	for(int i=m-1;i>=0;--i)
		for(int maxi =0;maxi<=n;++maxi)
			for(int j=0;j<=n;++j)
				for(int k=0;k<=j;++k){
					int next= max(maxi,(int)(k-1)/ar[i] +1 );
					if(k==0) next= max(maxi,0);
					double mul = log(combi[j][k])-log(m)*k;
					dp[i][j-k][next]+=dp[i+1][j][maxi]*exp(mul);
				}
	double ret= 0;
	for(int i=1;i<=n;++i)		
		ret += i*dp[0][0][i];
	printf("%0.9f\n",ret);	
}
