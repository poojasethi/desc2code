#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<cstring>
using namespace std;
#define Rep(i,a) for(i=0;i<(a);i++)
#define For(i,a,b) for(i=(a);i<(b);i++)

double dp[100][100][100];
int key[10000];
int cell(double x){
	int y=(int)x;
	return y+((x-y)>1e-10);
}
int main(){
	int i,j,k,n,m,x;
	while(~scanf("%d%d",&n,&m)){
		Rep(i,m)scanf("%d",&key[i]);
		memset(dp,0,sizeof(dp));
		dp[0][0][0]=1;
		for(i=0;i<=n;i++)for(j=0;j<m;j++)for(k=0;k<=n;k++){
			for(x=0;i+x<=n;x++){
				int kk=max(k,cell((double)x/key[j]));
				double pp=1.0/(j+1);
				double p=1.0;
				int iii;
				Rep(iii,x)p*=(i+x-iii)*pp;
				Rep(iii,i)p*=(1.0-pp);
				Rep(iii,x)p*=(1.0/(iii+1));
				
				dp[i+x][j+1][kk]+=dp[i][j][k]*p;
			}
		}
		double ans=0;
		Rep(i,n+1)
			ans+=i*dp[n][m][i];
		printf("%.20lf\n",ans);
	}
}
