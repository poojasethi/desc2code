#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;

int m, a[55], n;
double nCr[51][51]={0};
double ans[51][51][51];

double dp(int l, int r, int mm){//number of students left, number of rooms left, maximum room size
	if(l==0) return mm;
	if(r==0) return 0;
	if(ans[l][r][mm]>=0) return ans[l][r][mm];
	ans[l][r][mm] = 0;
	for(int i = 0; i<=l;i++){
		ans[l][r][mm]+=((nCr[l][i]*pow(r-1,l-i))/pow(r,l))*dp(l-i, r-1, max(mm,(int)ceil((double)(i)/a[r])));
	}
	return ans[l][r][mm];
}

int main(){
	for (int i = 0; i <= 50; ++i)
		nCr[i][0] = 1;
	for (int i = 1; i <= 50; ++i)
		for (int j = 1; j <= i; ++j)
			nCr[i][j] = nCr[i-1][j]+nCr[i-1][j-1];
	cin>>n>>m;
	for(int i=0;i<51;i++)
		for(int j=0;j<51;j++)
			for(int k=0;k<51;k++)
				ans[i][j][k] = -1;
	for(int i=1;i<=m;i++)
		cin>>a[i];
	printf("%.10f\n", dp(n, m, 0));
}
