# -*- coding: utf-8 -*-
n = int(raw_input())

if n==1:
    print 1
else:
    p = ['1']
    e = n%2
    for i in range(3, n, 2):
        p.append(str(i+1))
        p.append(str(i))
    if e:
        p.insert(0, str(n))
        p.append(str(2))
    else:
        p.insert(0, str(2))
    print ' '.join(p)
