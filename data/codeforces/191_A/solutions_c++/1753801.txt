#include <iostream>
#include <string>
#define N 30

using namespace std;

int f[N][N],k;

int main()
{
    int i,j,n;
    ios_base::sync_with_stdio(0);
    cin>>n;
    string s;
    for (i=1;i<=n;i++)
    {
        cin>>s;
        char s1=s[0]-'a',sn=s[s.length()-1]-'a';
        for (j=0;j<26;j++) if (f[j][s1]) f[j][sn]=max(f[j][sn],f[j][s1]+(int)s.length());
        f[s1][sn]=max((int)s.length(),f[s1][sn]);
    }
    for (i=0;i<26;i++) k=max(k,f[i][i]);
    cout<<k;
    return 0;
}
