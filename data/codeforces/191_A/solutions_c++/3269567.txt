#include<iostream>
#include<cstring>
using namespace std;

int main()
{



int dp[30][30];

memset(dp,-63,sizeof(dp) );


for(int i=0;i<29;i++)
dp[i][i]=0;


int n;
cin>>n;

string in;

for(int i=1;i<=n;i++)
  { cin>>in;
   int s=in[0]-'a';
   int t=in[ in.size() -1 ]-'a';

    for(int j=0;j<26;j++)
      dp[j][t]=max( dp[j][t] , dp[j][s] + (int)in.size() );

  }

int ans=0;

for(int i=0;i<26;i++)
 ans=max( ans  , dp[i][i] );

cout<<ans<<endl;


return 0;
}
