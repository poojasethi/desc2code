#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int n,v[27][27],ans;
char s[12];

int main()
{
    scanf("%d",&n);
    while(n--)
    {
        scanf("%s",s);
        int l=strlen(s),h=s[0]-'a',w=s[l-1]-'a';
        for(int i=0;i<26;++i) if(v[i][h]) v[i][w]=max(v[i][w],v[i][h]+l);
        v[h][w]=max(v[h][w],l);
        ans=max(ans,v[w][w]);
    }
    printf("%d\n",ans);
    return 0;
}
