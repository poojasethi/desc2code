#include<iostream>
#include<cstring>
using namespace std;
int v[26][26];
char s[12];
int max(int a,int b)
{
	return a>b?a:b;
}
int main()
{
	int n,l,ans=0;
	int h,w;
	cin>>n;
	memset(v,0,sizeof(v));
	while(n--)
	{
		cin>>s;
		l=strlen(s);
		h=s[0]-'a';
		w=s[l-1]-'a';
		for(int i=0;i<26;i++)
		{
			if(v[i][h])
				v[i][w]=max(v[i][w],v[i][h]+l);			
		}
		v[h][w]=max(v[h][w],l);
		ans=max(ans,v[w][w]);

	}
	cout<<ans<<endl;
	return 0;
}
