#include <iostream>
using namespace std;

int dp[26][26];

int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		string s;
		cin >> s;
		int len=s.size();
		int f = s[0]-'a', b = s[len-1]-'a';
		for(int j = 0; j < 26; j++)
			if(dp[j][f]>0)
				dp[j][b]=max(dp[j][b],dp[j][f]+len);
		dp[f][b]=max(dp[f][b],len);
	}
	int mx=0;
	for(int i = 0; i < 26; i++)
		mx=max(mx,dp[i][i]);
	cout << mx << endl;
}
