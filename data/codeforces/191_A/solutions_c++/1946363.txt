#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int T,v[27][27],ans;
char str[12];

int main()
{
	scanf("%d",&T);
	while(T--)
	{
		scanf("%s",str);
		int l=strlen(str),s=str[0]-'a',e=str[l-1]-'a';
		
		for(int i = 0 ; i < 26 ; ++i) 
          if(v[i][s]) 
            v[i][e] = max(v[i][e],v[i][s]+l);
           
		v[s][e]=max(v[s][e],l);
		
        ans=max(ans,v[e][e]);
	}
	printf("%d\n",ans);
	//system("pause");
	return 0;
}
