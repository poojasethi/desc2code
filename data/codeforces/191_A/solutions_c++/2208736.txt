#include<stdio.h>
#include<string.h>

int f[26][26];
char s[20];

int main()
{
    int n;
    scanf("%d\n",&n);
    while (n--)
    {
        gets(s);
        int x=s[0]-'a',y=s[strlen(s)-1]-'a',l=strlen(s);
        for (int i=0;i<26;i++)
        if ((f[i][x]>0)&&(f[i][y]<f[i][x]+l)) f[i][y]=f[i][x]+l;
        if (f[x][y]<l) f[x][y]=l;
    }
    int ans=0;
    for (int i=0;i<26;i++) ans=ans>f[i][i]?ans:f[i][i];
    printf("%d\n",ans);
    return 0;
}
