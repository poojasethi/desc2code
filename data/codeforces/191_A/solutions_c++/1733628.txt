#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int n,i,l,q,h,ma[255][255],ans;
char ch[10];
int main()
{
	scanf("%d",&n);
	while (n--)
	{
		scanf("%s",&ch);
		l=strlen(ch)-1;
		h=ch[l];
		q=ch[0];
		for (i='a';i<='z';i++)
		if (ma[i][q])
		{
			if (ma[i][q]+l+1>ma[i][h])
				ma[i][h]=ma[i][q]+l+1;
		}
		if (ma[q][h]<l+1)
			ma[q][h]=l+1;
	}
	for (i='a';i<='z';i++)
	ans=max(ma[i][i],ans);
	printf("%d",ans);
}
