#include<iostream>
#include<algorithm>
#include<cstring>

using namespace std;
int dp[26][26];

int main()
{
	int i,j,N,k,l;
	char tmp[15];
	cin>>N;

	for(i=0;i<N;i++)
	{
		cin>>tmp;
		j=strlen(tmp);
		k=tmp[j-1]-'a';
		l=tmp[0]-'a';

		for(j=0;j<26;j++)
		{
			if(dp[j][l])
				dp[j][k]=max(dp[j][l] + (int)strlen(tmp),dp[j][k]);
		}
		dp[l][k]=max(dp[l][k],(int)strlen(tmp));
	}

	j=dp[0][0];
	for(i=1;i<26;i++)
	{
		j=max(j,dp[i][i]);
	}

	cout<<j<<endl;

	return 0;
}
