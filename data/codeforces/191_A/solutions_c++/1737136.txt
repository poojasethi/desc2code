#include<stdio.h>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
string s;
int dp[300][300];
int i,x,y,size,ans,n;
char j;

main()
{cin>>n;
 for(i=1;i<=n;i++){
  cin>>s;
  size=s.size();
  x=s[0];
  y=s[s.size()-1];
  for(j='a';j<='z';j++)
   if(dp[j][x]>0 || x==j)
    dp[j][y]=max(dp[j][y],dp[j][x]+size);
                  }   
 
 for(j='a';j<='z';j++)  
  ans=max(dp[j][j],ans);
 cout<<ans<<endl;
// system("pause");
}
