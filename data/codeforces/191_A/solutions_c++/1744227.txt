#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

int a[26][26], h, t,  n, l;
char s[11];
int main(){
	scanf("%d", &n);
	while(n--){
		scanf("%s", s);
		l = strlen(s);
		h = s[0]-'a';
		t = s[l-1]-'a';
		for(int i = 0; i < 26; ++i) {
			if(a[i][h]) a[i][t] = max(a[i][t], a[i][h]+l);
		}
		a[h][t] = max(a[h][t],l);
	}
	int ans = 0;
	for(int i = 0; i < 26; ++i)
			if(a[i][i] > ans)
					ans = a[i][i];
	printf("%d\n", ans);
}
