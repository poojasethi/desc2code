#include<bits/stdc++.h>
using namespace std;
int dp[27][27],n,i,j,l,r,len,mx;
string s;
int main()
{
   ios::sync_with_stdio(false);
   cin>>n;
   for(i=0;i<n;i++) 
   {
         cin>>s;
         l=s[0]-'a',len=s.length(),r=s[len-1]-'a';
         for(j=0;j<26;j++)
         	if(dp[j][l])
         		dp[j][r]=max(dp[j][r],dp[j][l]+len);
         dp[l][r]=max(dp[l][r],len);
   }
   for(i=0;i<26;i++)
     	 mx=max(mx,dp[i][i]);
    cout<<mx;
	return 0;
}
