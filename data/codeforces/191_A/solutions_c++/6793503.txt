#include <bits/stdc++.h>
using namespace std;
int dp[30][30];
int main(){
	int n;
	scanf("%d",&n);
	while(n--){
		#define MAXX 11
		char str[MAXX];
		scanf("%s",str);
		int siz=strlen(str);
		int s=str[0]-'a',l=str[siz-1]-'a';
		for(int i=0;i<26;i++)if(dp[i][s])dp[i][l]=max(dp[i][l],dp[i][s]+siz);
		dp[s][l]=max(dp[s][l],siz);
	}
	int ans=0;
	for(int i=0;i<26;i++)ans=max(ans,dp[i][i]);
	printf("%d\n",ans);
	return 0;
}
