#include<bits/stdc++.h>
using namespace std;
int n , ans[100][100] , shuf;

main(){
	ios_base::sync_with_stdio(false);
	cin >>n;
	for(int i=0;i<n;++i){
		string s;
		cin >>s;
		int x = s[0]-'a' , y = s[s.size()-1]-'a';
		for(int j=0;j<26;++j){
			if(ans[j][x] > 0 || j == x){
				int e= ans[j][x]+s.size();
				ans[j][y] = max(ans[j][y] , e);
			}
		}
	}
	for(int i=0;i<26;++i)
		shuf = max(shuf , ans[i][i]);
	cout <<shuf<<endl;
}
