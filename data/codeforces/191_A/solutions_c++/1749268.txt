#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int a[27][27];
int f[27][27];
#define max(a,b) a>b?a:b
int main()
{
	int n;
	char s[11];
	scanf("%d",&n);
	int ans=0,len;
	for(int i=0;i<n;i++)
	{
		scanf("%s",s);
		len=strlen(s);
		int x=s[0]-'a';
		int y=s[len-1]-'a';
		for(int j=0;j<26;j++)
			if(a[j][x])a[j][y]=max(a[j][y],a[j][x]+len);
		a[x][y]=max(a[x][y],len);	
		ans=max(ans,a[y][y]);
	}
	printf("%d\n",ans);
	return 0;
}
