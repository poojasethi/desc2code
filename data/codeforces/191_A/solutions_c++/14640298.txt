#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main ()
{
//	freopen("test.txt","r",stdin);
	int n,maxx=0,l,a[130][130];
	char g[15];
	cin>>n;
	while (n--){
		scanf("%s",g);
		l = strlen(g);
		char start=g[0],end=g[l-1];
		for (int j='a';j<='z';j++)
			if (a[start][j]) a[end][j] = max(a[end][j], a[start][j] + l);
		a[end][start] = max(a[end][start],l);
		maxx = max(maxx,a[end][end]);
	}
	cout<<maxx;
    return 0;
}
