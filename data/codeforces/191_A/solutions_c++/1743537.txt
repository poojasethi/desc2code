
#include <iostream>
#include <algorithm>

using namespace std;

int d[30][30];
int n;

int main() {
  cin>>n;
  int top=0;
  for (int i=0;i<n;i++) {
	string s;
	cin>>s;
	int first=s[0]-'a', last=s[s.length()-1]-'a';
	int m=s.length();
	for (int k=0;k<26;k++) {
	  int x=d[k][first]+m;
	  if (x>d[k][last] && (d[k][first]>0 || first==k)) {
		if (k==last) top=max(top,x);
		d[k][last]=x;
	  }
	}
  }
  cout<<top<<endl;
  return 0;
}
