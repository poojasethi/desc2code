#include <cstdio>
#include <cstring>
#define max(x, y) ((x)>(y) ? (x) : (y))

int n;
char buf[16];
int l, u, v, dp[26][26], ans = 0;

int main() {
  memset(dp, 0, sizeof dp);
  scanf("%d", &n);
  for (int i = 0; i < n; ++i) {
    scanf("%s", buf);
    l = strlen(buf), u = buf[0]-'a', v = buf[l-1]-'a';
    for (int j = 0; j < 26; ++j)
      if (dp[j][u] || j == u) dp[j][v] = max(dp[j][u]+l, dp[j][v]);
  }
  for (int i = 0; i < 26; ++i)
    ans = max(ans, dp[i][i]);
  printf("%d\n", ans);
  return 0;
}
