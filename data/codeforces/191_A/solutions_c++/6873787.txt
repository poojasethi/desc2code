#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
	int p[676] = {0}, i, n, l, c, d = 0;
	char t[12], a, b;
	scanf("%d", &n);
	for(i = 0; i < n; i++) {
		scanf("%s", t);
		l = strlen(t);
		a = t[0] - 'a';
		b = t[l - 1] - 'a';
		for(c = 0; c < 676; c += 26) if(p[c + a]) p[c + b] = max(p[c + b], p[c + a] + l);
		p[26 * a + b] = max(p[26 * a + b], l);
	}
	for(c = 0; c < 676; c += 27) if(p[c] > d) d = p[c];
	printf("%d", d);
	return 0;
}
