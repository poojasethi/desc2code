#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
int n;
int g[30][30];
char s[20];

int main(){
     scanf("%d",&n);
     for(int i = 0; i < n; i++){
	  scanf("%s",s);
	  int len = strlen(s);
	  for(int j=0;j<len;j++) s[j] -= 'a';
	  for(int j=0;j<26;j++){
	       if(g[j][s[0]] || j == s[0])
		    g[j][s[len-1]] = max(g[j][s[len-1]],g[j][s[0]]+len);
	  }
     }
     int ans=0; for(int i = 0; i < 26; i++) ans=max(ans,g[i][i]);
     printf("%d\n",ans);
}
