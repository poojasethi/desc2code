#include<iostream>
#include<cstring>
using namespace std;

const int maxn = 5e5 + 10;

int ans[30][30];

int main() {
	int n;
	cin>>n;
	memset(ans,0,sizeof ans);
	for(int i=0;i<n;i++) {
		string in;
		cin>>in;
		int f = in[0]-'a';
		int s = in[in.size()-1]-'a';
		for(int j=0;j<26;j++) {
			if(ans[j][f]>0)
				ans[j][s] = max(ans[j][s],ans[j][f]+(int)in.size());
		}
		ans[f][s] = max(ans[f][s],(int)in.size());
	}
	int fans = 0;
	for(int i=0;i<26;i++) {
		fans = max(fans,ans[i][i]);
	}
	cout<<fans<<endl;
}
