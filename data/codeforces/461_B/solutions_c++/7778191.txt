#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define MOD (1000000007)
#define ll long long
using namespace std;

const int maxn=100010;

int fa[maxn];
ll d[maxn][2];
int n;

void Init()
{
	int cnt;
	scanf("%d",&n);
	for(int i=2;i<=n;i++)
	{
		scanf("%d",&fa[i]);
		fa[i]++;
	}
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&cnt);
		if(cnt)	d[i][1]=1;
		else	d[i][0]=1;
	}
}
void doit()
{
	for(int j=n;j>1;j--)
	{
		int i=fa[j];
		d[i][1]=( d[i][0]*d[j][1]%MOD + d[i][1]*(d[j][0] + d[j][1])%MOD )%MOD;
		d[i][0]=d[i][0]*(d[j][1] + d[j][0])%MOD;
	}
	printf("%d\n",d[1][1]);
}
int main()
{
	Init();
	doit();
	return 0;
}
