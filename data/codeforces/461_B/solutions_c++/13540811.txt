#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007 
int c[100009];
vector<int> v[100009];
long long dp[100009][2];
void dfs(int x){
    dp[x][c[x]]=1;
    for(int i=0;i<v[x].size();i++){
        int y=v[x][i];
        dfs(y);
        //if(y==par) continue;
        dp[x][1]=((dp[x][0]*dp[y][1])%mod+(dp[x][1]*dp[y][1])%mod+(dp[x][1]*dp[y][0])%mod)%mod;
        dp[x][0]=((dp[x][0]*dp[y][0])%mod+(dp[x][0]*dp[y][1])%mod)%mod;
    }
}
int main(){
    int n,x;
    cin>>n;
    for(int i=2;i<=n;i++){
        cin>>x;
        x++;
        v[x].push_back(i);
    }
    for(int i=1;i<=n;i++) scanf("%d",&c[i]);
    dfs(1);
    cout<<dp[1][1]<<endl;
}
