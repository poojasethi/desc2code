#include<algorithm>
#include<climits>
#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<string>
#include<vector>
using namespace std;
const int p = 1e9 + 7;
int n, t, i, a[100000];
long long f[100000], g[100000];
vector <int> son[100000];
void DP(int x)
{
	g[x] = 1;
	int i, j;
	for (i = 0; i < son[x].size(); i++)
	{
		DP(j = son[x][i]);
		f[x] = (f[x] * g[j] + g[x] * f[j]) % p;
		g[x] = g[x] * g[j] % p;
	}
	a[x] ? f[x] = g[x] : g[x] = (f[x] + g[x]) % p;
}
int main()
{
	scanf("%d", &n);
	for (i = 1; i < n; i++)
	{
		scanf("%d", &t);
		son[t].push_back(i);
	}
	for (i = 0; i < n; i++)
		scanf("%d", &a[i]);
	DP(0);
	printf("%I64d\n", f[0]);
	return 0;
}
