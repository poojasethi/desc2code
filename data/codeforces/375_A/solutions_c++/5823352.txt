#include<stdio.h>

int s[8]={1869,1968,1689,6198,1698,1986,1896,1869};
int a[10]={0};

int main()
{
    int i,x=0;
    char c;
    while((c=getchar())!='\n')
        a[c-'0']++;
    a[1]--;a[6]--;a[8]--;a[9]--;
    for(i=1;i<10;i++)
    {
    while(a[i]>0)
    {
    printf("%d",i);
    x=(x*10+i)%7;             
    a[i]--;
    }
    }       
    x=(x*10000)%7;
    printf("%d",s[7-x]);
    while(a[0]>0)
    {
    printf("0");
    a[0]--;
    }
    return 0;
}
