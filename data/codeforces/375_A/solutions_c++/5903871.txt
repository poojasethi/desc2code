#include <cstdio>
#include <iostream>

int a[10], c, i, s[7] = {1869, 6198, 1896, 1689, 1986, 1968, 1698};

using namespace std;

int main()
{
    while((i = getchar()) != '\n')
        ++ a[i - 48];
    -- a[1], -- a[6], -- a[8], -- a[9];
    for (i = 1; i <= 9; i ++)
        while(a[i]--)
        {
            c = (c * 10 + i) % 7;
            cout<<i;
        }
    cout<<s[c];
    while(a[0]--)
        cout<<"0";
    return 0;
}
