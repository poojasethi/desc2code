#include <cstdio>
#include <cstring>
using namespace std;
int modi[]={1869,1968,1689,6198,1698,1986,1896,1869} ;
const int maxn=1e6 ;
int t[10] ;
char s[maxn] ;
int main()
{
	scanf("%s",s) ;
	int n=strlen(s) ;
	for(int i=0;i<n;i++)
		t[s[i]-'0']++ ;
	t[1]-- ;
	t[6]-- ;
	t[8]-- ;
	t[9]-- ;
	int sum=0 ;
	for(int i=1;i<10;i++)
		for(int j=0;j<t[i];j++)
		{
			printf("%d",i) ;
			sum=(sum*10+i)%7 ;
		}
	sum*=10000 ;
	sum%=7 ;
	printf("%d",modi[7-sum]) ;
	for(int i=0;i<t[0];i++)
		printf("0") ;
}
