#include<bits/stdc++.h>

#define FP( ii,aa,bb )	for( int ii=aa;ii<=bb;ii++ )
#define FM( ii,aa,bb )	for( int ii=aa;ii>=bb;ii-- )

using namespace std;

int  n,ind,h[15];
char s[2000000];
int  v[2000000];

int main(){

	scanf("%s",s+1);
	n = strlen( s+1 );
	
	int cnt,x;

	FP( i,1,n )
		if( s[i]!='0' )	v[++ind] = s[i]-'0';
		else						cnt++;

	FP( i,1,50 ){
		random_shuffle( v+1,v+ind+1 );
		x=0;
		FP( j,1,ind )	x = (x*10+v[j])%7;

			FP( j,1,n )	printf("%d",v[j]);
			break;
		}
	}

}
