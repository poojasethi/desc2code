#include <iostream>
using namespace std;
int mod, cnt, dig[10], arr[7]={1869,6189,1689,6198,1698,1986,1896}, set[4]={1,6,8,9};
int main()
{
	string s;
	cin >> s;
	for(int i = 0; i < s.length(); i++) dig[s[i]-'0']++;
	for(int i = 0; i < 4; i++) dig[set[i]]--;
	for(int i = 1; i <= 9; i++)
		while(dig[i]--)
		{
			mod = (mod*10+i)%7;
			cout << i;
		}
	mod = (mod*4)%7;
	cout << arr[(7-mod)%7];
	while(dig[0]--)
		cout << 0;
}
