#include <cstdio>

const int num[] = {1869, 1968, 1689, 6198, 1698, 1986, 1896};

char s[1000005];
int cnt[10];

int main() {
	gets(s);
	for(int i = 0; s[i]; ++i) {
		cnt[s[i]-'0']++;
	}
	cnt[1]--;
	cnt[6]--;
	cnt[8]--;
	cnt[9]--;
	int n = 0;
	for(int i = 1; i <= 9; ++i) {
		for(int j = 0; j < cnt[i]; ++j) {
			putchar(i+'0');
			n = (n*10+i)%7;
		}
	}
	n = n*10000%7;
	printf("%d", num[(7-n)%7]);
	for(int i = 0; i < cnt[0]; ++i) putchar('0');
	puts("");
}
