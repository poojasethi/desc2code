#include <iostream>
using namespace std;
int m[7]={1869,1968,1689,6198,1698,1986,1896};
int main()
{
	int d[10]={0},i,r=0;
	string s;
	cin>>s;
	for (i=0;i<s.size();i++) d[s[i]-'0']++;
	d[1]--;d[6]--;d[8]--;d[9]--;
	for (i=1;i<=9;i++)
		while (d[i]--)
		{
			cout<<i;
			r=(r*10+i)%7;
		}
	r=r*10000%7;
	cout<<m[(7-r)%7];
	while (d[0]--) cout<<0;
	return 0;
}
