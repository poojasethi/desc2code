#!/usr/bin/env python

import itertools

perm = [1869, 6891, 1896, 1689, 1986, 1968, 1698]
def get_divisible_number(number):
    div_num = '0'
    for x in ('1', '8', '6', '9'):
        i = number.index(x)
        number = number[:i] + number[i + 1:]
    rem = 0
    for x in number:
        rem = (rem * 10 + int(x)) % 7 
    div_num = number + str(perm[rem])
    lead_zeroes = 0
    for x in div_num:
        if x == '0':
            lead_zeroes += 1
        else:
            break
    div_num = div_num[lead_zeroes:] + lead_zeroes * '0'
 
    return div_num

if __name__ == '__main__':
    number = raw_input()
    print get_divisible_number(number)
