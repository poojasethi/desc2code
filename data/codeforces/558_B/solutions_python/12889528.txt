size = input()
array = map(int, raw_input().split())

counts = [0] * (10**6+5)
l = [-1] * (10**6+5)
r = [-1] * (10**6+5)
mc = -1
c = []

for j in range(len(array)):
	i = array[j]
	counts[i] += 1
	if l[i] == -1:
		l[i] = j
	r[i] = j
	if counts[i] > mc:
		c = []
		c.append(i)
		mc = counts[i]
	elif counts[i] == mc:
		c.append(i)

# print c
m = 10**9
lmin = 0
rmin = 0
for i in c:
	if r[i] - l[i] + 1 < m:
		lmin = l[i]
		rmin = r[i]
		m = r[i] - l[i] + 1
print lmin+1, rmin+1
