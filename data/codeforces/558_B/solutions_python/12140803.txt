n=int(raw_input())
a=map(int,raw_input().split())
cnt=[]
last=[]
first=[]
for i in range(1000005):
	cnt.append(0)
	last.append(-1)
	first.append(-1)
for i in range(len(a)):
	cnt[a[i]]+=1
	last[a[i]]=i
	if(first[a[i]]==-1):
		first[a[i]]=i
mx=0
for i in range(1000005):
	if(cnt[i]>mx):
		mx=cnt[i]
mn=999999999
ans=0
for i in range(1000005):
	if(cnt[i]==mx):
		if(last[i]-first[i]+1<mn):
			mn=last[i]-first[i]+1
			ans=i
print first[ans]+1,last[ans]+1
