n = input()
num = map(int, raw_input().split())
arr = {}
for i in xrange(len(num)):
    if arr.has_key(num[i]):
        arr[num[i]][2] = i
        arr[num[i]][0] += 1
    else :
        arr[num[i]] = [1,i,i]
mxk = num[0]
for key in arr:
    if arr[mxk][0] < arr[key][0]:
        mxk = key
    elif arr[mxk][0] == arr[key][0] and arr[mxk][2] - arr[mxk][1] > arr[key][2] - arr[key][1]:
        mxk = key
print arr[mxk][1] + 1, arr[mxk][2] + 1
