l=map(int,raw_input().split())
chk={}
for i,j in enumerate(l):
    if chk.has_key(j):
        chk[j][1]=i+1
        chk[j][2]+=1
    else:
        chk[j]=[i+1,i+1,1]
s=e=k=0
for i in chk:
    tmp=chk[i]
    if tmp[2]>k or (tmp[2]==k and e-s>tmp[1]-tmp[0]):
        s=tmp[0]
        e=tmp[1]
        k=tmp[2]
