n=int(raw_input())
a=map(int,raw_input().split())
log={}
memo=0
chk=[]
for i,j in enumerate(a):
    if log.has_key(j):
        log[j][2]=i
    else:
        log[j]=[0,i,i]
    log[j][0]+=1

memo,l,r=0,0,10000000000
for i in log:
    if log[i][0]>memo:
        memo=log[i][0]
        l=log[i][1]
        r=log[i][2]
    elif log[i][0]==memo:
        if r-l>log[i][2]-log[i][1]:
            l=log[i][1]
            r=log[i][2]
print l+1,r+1
