n = int(raw_input())
arr = map(int, raw_input().split())
freq = {}
for i, v in enumerate(arr):
	if v in freq:
		tmp = freq[v]
		freq[v] = [tmp[0] + 1, tmp[1], i]
	else:
		freq[v] = [1, i, i]
m = max(freq.viewvalues())[0]
dist = float('inf')
ans = [1, n]
for i in freq.viewvalues():
	if i[0] == m:
		if i[2] - i[1] < dist:
			dist = i[2] - i[1]
			ans[0] = i[1] + 1
			ans[1] = i[2] + 1
print ans[0], ans[1]
