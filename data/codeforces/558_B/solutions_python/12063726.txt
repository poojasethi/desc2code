#!/usr/bin/env python

def main():
	input()
	start = {}
	end = {}
	cnt = {}

	a = map(int, raw_input().split())
	for (i, ai) in enumerate(a):
		if ai not in start:
			start[ai] = end[ai] = i
			cnt[ai] = 1
		else:
			end[ai] = i
			cnt[ai] += 1
	
	max_cnt = max(cnt.values())
	min_len = len(a) + 1

	for (k, v) in cnt.items():
		if v == max_cnt and end[k] - start[k] < min_len:
			min_len = end[k] - start[k]
			s = start[k]
			e = end[k]
	
	print s + 1, e + 1

if __name__ == '__main__':
	main()
