n = int(raw_input())
a = map(int, raw_input().split())

cnt = {}
l = {}
r = {}

for i in xrange(n):
    x = a[i]
    if a[i] not in cnt:
        cnt[x] = 1
        l[x] = i
        r[x] = i
    else:
        cnt[x] += 1
        r[x] = i

bcnt = 0
bl = br = 0

for i in xrange(n):
    x = a[i]
    if cnt[x] > bcnt or cnt[x] == bcnt and r[x] - l[x] < br - bl:
        bcnt = cnt[x]
        bl = l[x]
        br = r[x]

print bl + 1, br + 1
