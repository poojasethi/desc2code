N = int(raw_input())

a = map(int, raw_input().split())
data = [0] * (10 ** 6 + 1)
L = [0] * (10 ** 6 + 1)
R = [0] * (10 ** 6 + 1)

mx = 0
for i, v in enumerate(a):
	if data[v] == 0:
		L[v] = i
	R[v] = i
	data[v] += 1
	mx = max(mx, data[v])

mn = N
index = 0
for i in xrange(len(data)):
	if data[i] == mx:
		if mn > R[i] - L[i]:
			mn = R[i] - L[i]
			index = i
print L[index] + 1, R[index] + 1
