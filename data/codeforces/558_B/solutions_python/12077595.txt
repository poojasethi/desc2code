n=input()
ar=map(int,raw_input().split())
maxn=1000005
tim=[0 for i in range(maxn)]
be=[0 for i in range(maxn)]
en=[0 for i in range(maxn)]

for i in range(n):
    e=ar[i]
    if tim[e]==0:
        be[e]=i
    tim[e]+=1
    en[e]=i

ans=0

for i in range(maxn):
    if tim[i]>tim[ans]:
        ans=i
    elif tim[i]==tim[ans]:
        if en[ans]-be[ans]>en[i]-be[i]:
            ans=i

print be[ans]+1,en[ans]+1
