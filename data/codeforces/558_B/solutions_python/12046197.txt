from sys import stdin
from collections import defaultdict
def main():
    n = int(stdin.readline())
    a = map(int, stdin.readline().split())
    d = defaultdict(int)
    r = defaultdict(int)
    for i, x in enumerate(a):
        d[x] += 1
        r[x] = i
    v = max(d.values())
    al, ar = 0, n - 1
    for i, x in enumerate(a):
        if d[x] == v and r[x] - i < ar - al:
            al, ar = i, r[x]
        d[x] -= 1
    print al + 1, ar + 1
main()
