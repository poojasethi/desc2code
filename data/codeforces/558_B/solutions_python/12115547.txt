n = int(raw_input())

x = map(int, raw_input().split())

cnt = {}
l = {}
r = {}

for i, v in enumerate(x, start=1):
    if v not in cnt:
        cnt[v] = 0
        l[v] = i
        r[v] = i
    cnt[v] += 1
    l[v] = min(l[v], i)
    r[v] = max(r[v], i)

M = max(cnt.values())
ans = [len(x), 1, n]

for k, v in cnt.iteritems():
    if v != M:
        continue

    if r[k]-l[k]+1 < ans[0]:
        ans = [r[k]-l[k]+1, l[k], r[k]]

print ans[1], ans[2]
