from collections import Counter
n = int(raw_input())
a = raw_input().split()
c = Counter()
l = {}
r = {}
i = 0
for k in a:
    c[k] += 1
    r[k] = i
    if c[k] == 1:
        l[k] = i
    i += 1
maxn = max(c.values())
d = sorted([(r[k] - l[k] + 1, l[k], r[k]) for k, v in c.items() if v == maxn])[0]
print d[1] + 1, d[2] + 1
