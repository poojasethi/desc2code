
n = input()

a = map(int, raw_input().split())

b = []

dict = {}
id = 0
for i in range(0,n):
	if not dict.has_key(a[i]):
		dict[a[i]] = id
		id +=1
		b.append([-1, 1, i+1, i+1])
	else:
		myId = dict[a[i]]
		b[myId][0] -=1
		b[myId][2] = min(i+1, b[myId][2])
		b[myId][3] = max(i+1, b[myId][3])
		b[myId][1] = b[myId][3] - b[myId][2] + 1

b.sort()

print b[0][2], b[0][3]
