n=int(raw_input())
M=10**6+10
first=[M for i in xrange(M)]
last=[-M for i in xrange(M)]
times=[0 for i in xrange(M)]
a=map(int,raw_input().split())
n=len(a)
for i in xrange(n):
    x=a[i];
    first[x]=min(first[x],i)
    last[x]=max(last[x],i)
    times[x] += 1
l,r,beauty=0,M,-1
for i in xrange(M):
    beauty=max(beauty,times[i])
for i in xrange(M):
    if times[i] == beauty:
        k = last[i]-first[i];
        if k < r-l:
            l,r=first[i],last[i]
l += 1
r += 1
print "%d %d\n"%(l,r)
