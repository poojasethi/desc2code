from collections import Counter
n = input()
a = map(int,raw_input().split())
s = [[-1,0,0] for _ in xrange(10**6+1)] #start,end,count
for ind,x in enumerate(a):
    s[x][2] += 1
    if s[x][0] == -1:
        s[x][0] = ind
    s[x][1] = ind
o = max(s,key=lambda x:(x[2],-(x[1]-x[0])))
print o[0]+1,o[1]+1
