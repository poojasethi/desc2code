n = input();
a = map(int, raw_input().split())
b = [0 for i in xrange(1000010)]
s = [0 for i in xrange(1000010)]
e = [0 for i in xrange(1000010)]

for i in xrange(n):
    x = a[i]
    if b[x] == 0:
        s[x] = i
    b[x] += 1
    e[x] = i

ans = 0
for i in xrange(1000001):
    if b[ans] < b[i]:
        ans = i
    elif b[ans] == b[i] and e[ans] - s[ans] > e[i] - s[i]:
        ans = i
print s[ans]+1, e[ans]+1
