n = int(raw_input())
a = map(int,raw_input().split(" "))
d = dict()
for i in range(n):
	if a[i] in d:
		d[a[i]][0] += 1
		d[a[i]][2] = i
		d[a[i]][3] = d[a[i]][1] - d[a[i]][2] - 1
	else:
		d[a[i]] = [1,i,i,-1]
result = sorted(d.values(), reverse = True, key = lambda x: (x[0],x[3]))
print result[0][1]+1, result[0][2]+1
