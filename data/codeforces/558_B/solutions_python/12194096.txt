n = int(raw_input())
As = map(int, raw_input().split())
count_dict = dict()
for i in range(len(As)):
    if As[i] not in count_dict: count_dict[As[i]] = [0, i+1, 0]
    count_dict[As[i]][0] += 1
    count_dict[As[i]][2] = i+1
ans, idx1, idx2 = 0, 0, 0
for num in count_dict.itervalues():
    if num[0] < ans: continue
    if num[0] == ans and idx2 - idx1 <= num[2] - num[1]: continue
    ans, idx1, idx2 = num[0], num[1], num[2]
print idx1, idx2
