#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int C[1000001], L[1000001], R[1000001], n, a, i, m, mC, mL, mR;
int main()
{
	scanf("%d", &n);
	for (i=1; i<=n; i++) {
		scanf("%d", &a);
		C[a]++;
		if (L[a]==0) L[a]=i;
		if (i>R[a]) R[a]=i; 
	}
	for (i=1; i<=1000000; i++) 
	if(C[i]>mC || C[i]==mC && R[i]-L[i]+1<m) {
		mC=C[i]; m=R[i]-L[i]+1; mL=L[i]; mR=R[i]; 
	}
	printf("%d %d\n", mL, mR);
	return 0;
}
