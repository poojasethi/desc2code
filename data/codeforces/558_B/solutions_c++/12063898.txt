#include<iostream>
#include<algorithm> 
using namespace std;
int st[1000100],en[1000100],a[1000100];
int main()
{
	int x,y,z,i,t,b=0;
	cin>>x;
	for(i=1;i<=x;i++)
	{
		cin>>y;
		if(st[y]==0)
		st[y]=i;
		else 
		en[y]=i;
		a[y]++;
		if(a[y]>b)
		{
			b=a[y];
			t=y; 
		}
		else if(a[y]==b&&en[y]-st[y]<en[t]-st[t])
		{
			t=y;
		}
	}
	if(en[t]==0)
	cout<<"1 1"<<endl;
	else 
	cout<<st[t]<<" "<<en[t]<<endl;
	return 0;
}
