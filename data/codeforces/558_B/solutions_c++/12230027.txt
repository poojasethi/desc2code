#include<cstdio>
int n,i,ml,mr,maxnum,a[100005],b[1000005],l[1000005];
int main()
{
    scanf("%d",&n);
    for (i=1;i<=n;i++) scanf("%d",&a[i]);
    maxnum=0; ml=1; mr=1;
    for (i=1;i<=n;i++)
    {
        if (b[a[i]]==0) l[a[i]]=i;
        b[a[i]]++;
        if (b[a[i]]>maxnum||b[a[i]]==maxnum&&i-l[a[i]]+1<mr-ml+1)
        {
            maxnum=b[a[i]];
            ml=l[a[i]];
            mr=i;
        }
    }
    printf("%d %d",ml,mr);
}
