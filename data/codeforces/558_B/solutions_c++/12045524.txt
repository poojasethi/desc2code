#include <cstdio>
using namespace std;

int N;
int fo[1000010], lo[1000010], tm[1000010];

int main(){
    int i,a,mx=1;
    scanf("%d",&N);
    for(i=1;i<=N;++i){
        scanf("%d",&a); lo[a]=i; ++tm[a];
        if(!fo[a]) fo[a]=i;
    }
    for(i=1;i<=1000000;++i){
        if(tm[i]>tm[mx]) mx=i;
        else if(tm[i]==tm[mx]){
            if(lo[i]-fo[i]<lo[mx]-fo[mx]) mx=i;
        }
    }
    printf("%d %d",fo[mx],lo[mx]);
}
