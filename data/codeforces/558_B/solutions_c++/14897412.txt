#include <stdio.h>
int a[1000100],start[1000100],end[1000010];
int main()
{
	int i,j,n,num;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&num);
		a[num]++;
		if(a[num]==1)
		{
			start[num] = i+1;
		}
		end[num] = i+1;
	}
	int max=0,index=0;
	for(i=0;i<=1000010;i++)
	{
		if(a[i]>max||a[i]==max&&end[i]-start[i]<end[index]-start[index])
		{
			max = a[i];
			index = i;
		}
	}
	printf("%d %d\n",start[index],end[index]);
	return 0;
}
