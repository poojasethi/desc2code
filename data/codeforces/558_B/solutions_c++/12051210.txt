#include <bits/stdc++.h>
using namespace std;

int n,sz,mx,l,r,a[100005]; vector<int> occ[1000005];

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> n; for (int i=1;i<=n;i++) cin >> a[i];
	for (int i=1;i<=n;i++) occ[a[i]].push_back(i);
	for (int i=1;i<=1000000;i++){
		sz=occ[i].size(); if (sz==0) continue;
		if (sz>mx||(sz==mx&&occ[i][sz-1]-occ[i][0]+1<r-l+1)){
			mx=sz; l=occ[i][0]; r=occ[i][sz-1];
		}
	} cout << l << " " << r << "\n";
}
