#include<cstdio>

int n, v, ans, ans_l, ans_r;
int s[1111111], l[1111111], r[1111111];

int main() {
	scanf("%d", &n);
	ans = 0;
	for(int i=0;i<n;i++) {
		scanf("%d", &v);
		if (!s[v]) l[v] = r[v] = i;
		else r[v] = i;
		s[v]++;
		
		if(s[v] > ans || s[v] == ans && ans_r-ans_l > r[v]-l[v]) {
			ans = s[v];
			ans_r = r[v];
			ans_l = l[v];
		}
	}
	printf("%d %d\n", ans_l+1, ans_r+1);
	
	return 0;
}
