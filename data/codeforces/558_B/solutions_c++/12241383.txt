#include<cstdio>
using namespace std;
int a1[1000001],a2[1000001],a3[1000001];
int main()
{
    int n,a,b,o=0;
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
    {
        scanf("%d",&a);
        if(!a1[a])a1[a]=i;
        a2[a]=i;++a3[a];
    }
    for(int i=1;i<1000001;++i)
    if(a3[i]>o)o=a3[i],a=a1[i],b=a2[i];
    else if(a3[i]==o&&a2[i]-a1[i]<b-a)a=a1[i],b=a2[i];
    printf("%d %d\n",a,b);
}
