#include <iostream>
#include<map>
typedef long ll;
using namespace std;
map<ll,ll> a,b;
ll n,i=0,c,o=0,m[2];
int main()
{
    cin>>n;
    m[0]=0;m[1]=n;
    for(i=1;i<=n;i++)
    {
       cin>>c;
       if(a[c]<=0)
       {
           a[c]=i;
       }
       b[c]++;
       if(b[c]>o || b[c]>=o && i-a[c]<m[1]-m[0])
       {
           m[1]=i;
           m[0]=a[c];
           o=b[c];
       }

    }
    cout<<m[0]<<" "<<m[1];
}
