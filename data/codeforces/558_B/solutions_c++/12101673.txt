#include <stdio.h>

int b[1000001],l[1000001],r[1000001],g[100001];
int i,n,max,min1,min2;

int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		scanf("%d",&g[i]);
		if (b[g[i]]==0)
		    l[g[i]]=i;
		b[g[i]]++;
		r[g[i]]=i;
		if (b[g[i]]>max)
		    max=b[g[i]];
	}
	min1=1;
	min2=n;
	for (i=1;i<=1000000;i++)
		if ((b[i]==max) && (min2-min1>r[i]-l[i]))
		{
			min1=l[i];
			min2=r[i];
		}
	printf("%d %d",min1,min2);
	return 0;
}
