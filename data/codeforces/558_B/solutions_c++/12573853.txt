#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;

int n;
int a[1000100]={0};
int b[1000100]={0};

int main()
{
	scanf("%d",&n);
	int l,r;
	int max=0,m;
	for(int i=1;i<=n;i++)
	{
		int k;
		scanf("%d",&k);
		if(a[k]==0)a[k]=i;
		b[k]++;
		if(b[k]>max)max=b[k],r=i,l=a[k];
		else if(b[k]==max && i-a[k]<r-l)r=i,l=a[k];
	}
	printf("%d %d",l,r);
	return 0;
}
