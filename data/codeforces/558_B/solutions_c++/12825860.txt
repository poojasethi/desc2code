#include<iostream>
#include<cstdio>
#include<cstring>

using namespace std; 

int fir[1000005],cnt[1000005]; 

int main(){
	int n,maxlen=0,l=0,r=-1;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){ 
		int k; 
		scanf("%d",&k); 
		if (++cnt[k]==1)  fir[k]=i; 
		if (cnt[k]>maxlen ||  cnt[k]==maxlen && r-l+1>i-fir[k]+1) {
			l=fir[k],r=i,maxlen=cnt[k]; 
		}
	}
	printf("%d %d\n",l,r); 
	return 0; 
}
