#include<bits/stdc++.h>
using namespace std;
#define m 1000010
int num[m],ini[m],termi[m];
int main()
{
    int n,res=100; cin>> n;
    for(int i=1; i<=n; i++){
        int x; cin>> x;
        if(num[x]==0) ini[x]=i;
        termi[x]=i; num[x]++;
        if(num[x]>num[res]) res=x;
        if(num[x]==num[res] && termi[x]-ini[x]<termi[res]-ini[res]) res=x;
    }
    cout<<ini[res]<<" "<<termi[res];
    return 0;
}
