#include<stdio.h>
int a[1000002][3];
int main()
{
	int n,i,z,max=0,l,r,dif;
	scanf("%d",&n);
//	for(i=0;i<1000002;i++)
//		a[i][0]=0;
	for(i=0;i<n;i++)
	{
		scanf("%d",&z);
		a[z][0]++;
		if(a[z][0]==1)
			a[z][1]=i+1;
		a[z][2]=i+1;
		if(max<a[z][0])
			max=a[z][0];
	}
	dif=999999999;
	for(i=0;i<1000002;i++)
	{
		if(a[i][0]==max)
		{
			if(dif>(a[i][2]-a[i][1]))
			{
				l=a[i][1];
				r=a[i][2];
				dif=r-l;
			}
		}
	}
	printf("%d %d\n",l,r);
	return 0;
}
