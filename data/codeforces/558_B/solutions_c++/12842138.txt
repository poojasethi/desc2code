#include <stdio.h>
int arr[1000001],s1[1000001];
int main(){
    int n,a[100001],s,e,max1=0,i;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
        arr[a[i]]++;
        if(arr[a[i]]==1) s1[a[i]]=i;
        if(arr[a[i]]>max1 || (arr[a[i]]==max1 && (i-s1[a[i]]<e-s))) {e=i;s=s1[a[i]];max1=arr[a[i]];}
    }
    printf("%d %d",s+1,e+1);
    return 0;
}
