#include<bits/stdc++.h>
using namespace std;

int n,a[1000002],b[1000002],c[1000002],x,ans;

int main()
{
    cin>>n;
    for(int i=1;i<=n;++i)
    {
        cin>>x;
        ++a[x];
        if(b[x]==0)b[x]=i;
        c[x]=i;
        if(a[x]>a[ans])ans=x;
            if(a[x]==a[ans]&&c[x]-b[x]<c[ans]-b[ans])ans=x;
    }
    cout<<b[ans]<<' '<<c[ans];
}
