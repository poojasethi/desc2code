#include <cstdio>
#include <algorithm>
using namespace std;
int a[100005],b,c,d,e,f[1000005],g,h,i,j,k,m,n,l[1000005],r[1000005];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++){
		if(!f[a[i]])l[a[i]]=i;
		f[a[i]]++;
		if(f[a[i]]>g)g=f[a[i]];
		r[a[i]]=i;
	}
	h=n;
	for(i=1;i<=1000000;i++)
	if(f[i]==g){
		if(r[i]-l[i]+1<=h){
			h=r[i]-l[i]+1;
			b=l[i];c=r[i];
		}
	}
	printf("%d %d\n",b,c);
	return 0;
}
