#include<stdio.h>
#define M 1000005
int L[M]={0},R[M],cnt[M]={0};
int main(){
	int n,i,j,k,mt=0;
	scanf("%d",&n); 
	for(i=1;i<=n;i++){
		scanf("%d",&k);
		if(L[k]==0)L[k]=i;
		R[k]=i;
		cnt[k]++;
		if(cnt[k]>mt)mt=cnt[k];
	}
	int ans=n+1,l,r;
	for(i=1;i<M;i++){
		if(cnt[i]!=mt)continue;
		int tmp=R[i]-L[i]+1;
		if(tmp<ans){
			ans=tmp,l=L[i],r=R[i];
		}
	}
	printf("%d %d\n",l,r);
	return 0;
}
