#include <iostream>
using namespace std;
int a[1000001], b[1000001], mx, c, l, r, n;
int main(){
	cin >> n;
	for (int i = 1; i <= n; i ++){
		cin >> c;
		if(!b[c])
		    b[c] = i;
		a[c] ++;
		if(a[c] > mx || a[c] == mx && i - b[c] < r - l){
			mx = a[c];
			l = b[c];
			r = i;
		}
	}
	cout << l << ' ' << r;
}
