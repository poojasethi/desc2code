#include <bits/stdc++.h>
using namespace std;
int main(){
  int n,mi;cin>>n;
  map<int,int> a,s,e,c;
  for(int i=1;i<=n;i++){
    cin>>a[i];
    c[a[i]]++;e[a[i]]=i;
    s[a[i]]=s[a[i]]?s[a[i]]:i;
  }
  mi=a[1];
  for(int i=2,m=c[a[1]];i<=n;i++){
    if(c[a[i]]>m||(c[mi]==c[a[i]]&&(e[a[i]]-s[a[i]]<e[mi]-s[mi]))){m=c[a[i]];mi=a[i];}
  }
  cout<<s[mi]<<" "<<e[mi]<<endl;
  return 0;
}
