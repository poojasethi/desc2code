#include<iostream>
using namespace std;
int a[1000005]={0};
int b[1000005]={0};
int main()
{
    long long n,q,d,l,i,ans=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>d;
		if(b[d]==0)
		{
			b[d]=i+1;
		}
		a[d]+=1;
		if(ans<a[d])
		{
			ans=a[d];
			q=d;
			l=i+1;
		}
	 
	if(ans==a[d])
	{
		if(b[q]-l<b[d]-(i+1))
		{
			q=d;
			l=i+1;
		}
	}
}
	cout<<b[q]<<" "<<l;

}
