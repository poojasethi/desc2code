#include<cstdio>

int n,a,x;
int c[1000010];
int l[1000010];
int r[1000010];

int main()
{
	scanf("%d",&n);
	for(int i=0;i<=1000000;i++)
	{
		l[i]=n+1;
		r[i]=-1;
	}
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a);
		if(i<l[a]) l[a]=i;
		if(i>r[a]) r[a]=i;
		c[a]++;
	}
	for(int i=0;i<=1000000;i++)	if(c[i]>c[x]||(c[i]==c[x]&&r[i]-l[i]<r[x]-l[x])) x=i;
	printf("%d %d\n",l[x],r[x]);
	return 0;
}
