#include <iostream>
using namespace std;
int A[1000001], B[1000001][2];
int main() {
	int n, c = 0, x1, x2, o;
	cin >> n;
	for(int i = 0; i < n; ++i) {
		cin >> o;
		if(!A[o]) B[o][0] = i;
		B[o][1] = i;
		A[o]++;
	}
	for(int i = 1; i <= 1000000; ++i) {
		if(c < A[i] || (c == A[i] && 
		   B[i][1] - B[i][0] < x2 - x1)) x1 = B[i][0], x2 = B[i][1], c = A[i];
	}
	cout << x1 + 1 << " " << x2 + 1 << endl;
}
