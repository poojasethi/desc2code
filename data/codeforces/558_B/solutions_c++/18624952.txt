#include<bits/stdc++.h>
using namespace std;

int start[1000001];
int num[1000001];

int main()
{
	int n;
	cin>>n;
	int c,ans1,ans2,Max=0;
	for(int i=1;i<=n;i++)
	{
		cin>>c;
		if(!start[c]) 
		{
			start[c]=i;
			
		}
		num[c]++;
		if(num[c]>Max || num[c]==Max && i-start[c]<ans2-ans1)
		{
			Max=num[c];
			ans1=start[c];
			ans2=i;
		}
	}
	cout<<ans1<<" "<<ans2<<endl;
}
