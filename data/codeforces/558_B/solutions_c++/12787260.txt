#include <stdio.h>
int a[1111111], b[1111111], c[1111111], l, r = 1e7, n, x, mx;
int main(){
	scanf("%d", &n);
	for(int i = 0; i < n; ++ i){
		scanf("%d", &x);
		if(a[x] == 0) b[x] = i;
		c[x] = i;
		a[x] ++;
		if(a[x] > mx) mx = a[x];
	}
	for(int i = 1; i <= 1000000; ++ i){
		if(a[i] == mx && c[i] - b[i] < r - l) r = c[i], l = b[i];
	}
	printf("%d %d", l + 1, r + 1);
	return 0;
}
