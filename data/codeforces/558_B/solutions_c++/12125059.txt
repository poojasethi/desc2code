#include <iostream>
using namespace std;
int n, b[1000005], a[1000005], mmax=0, k, pos, inp;
int main(){
    cin>>n;
    fill (b,b+1000005, -1);
    for (int i=0; i<n; i++){
        cin>>inp;
        if (b[inp]==-1)
            b[inp]=i;
        a[inp]++;
        if (a[inp]>mmax || (a[inp]==mmax && i-b[inp]<pos-b[k])){
            mmax=a[inp];
            k=inp;
            pos=i;
        }
    }
    cout<<b[k]+1<<" "<<pos+1;
    return 0;
}
