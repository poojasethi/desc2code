#include<iostream>
using namespace std;

int n,a,mx;
int t[1000005],l[1000005],r[1000005];

int main()
{
    cin>>n;
    
    for(int i=1;i<=n;++i)
    {
        cin>>a;
        t[a]++;
        if(!l[a])l[a]=i;
        r[a]=i;
        if(t[a]>t[mx])mx=a;
        else if((t[a]==t[mx]) && (r[a]-l[a]<=r[mx]-l[mx]))mx=a;
    }
    
    cout<<l[mx]<<" "<<r[mx];
    //system("pause");
    return 0;
}
