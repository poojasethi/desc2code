#include <bits/stdc++.h>
using namespace std;

const int maxv = 1e6+123;

int n,cnt[maxv],l[maxv],r[maxv];

int main(){
	
	scanf("%d",&n);
	
	for( int i=1,x ; i <= n ; i++ ){
		scanf("%d",&x);
		if( !cnt[x] )
			l[x] = i;
		cnt[x]++;
		r[x] = i;
	}
	
	int val = 1;
	
	for( int i=2 ; i < maxv ; i++ )
		if( cnt[val] < cnt[i] or (cnt[val] == cnt[i] and r[val] - l[val] > r[i] - l[i]))
			val = i;
	
	printf("%d %d\n",l[val],r[val]);
	
	return 0;
}
