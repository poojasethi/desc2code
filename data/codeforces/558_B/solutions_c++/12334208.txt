#include <bits/stdc++.h>
#define maxn 1000001
using namespace std;
int n,a,b,c,o;
int a1[maxn],a2[maxn];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	{
		scanf("%d",&c);
		if(!a2[c])
		{
			a2[c]=i;
		}
		a1[c]++;
		if(a1[c]>o||a1[c]==o&&i-a2[c]<b-a)
		{
			o=a1[c],a=a2[c],b=i;
		}
	}
	printf("%d %d\n",a,b);
	return 0;
}
