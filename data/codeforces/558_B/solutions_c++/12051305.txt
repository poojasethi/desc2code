#include<bits/stdc++.h>
using namespace std;
int a[1000001]={0};
int b[1000001]={0};
int main(){
    int n,ma=0,q,l;
    cin>>n;
    for(int i=0;i<n;i++){
        int k; cin>>k;
        if(b[k]==0) b[k]=i+1;
        a[k]+=1;
        if(ma<a[k]){
            ma=a[k];
            q=k; l=i+1;
        }
        if(ma==a[k]){
            if(b[q]-l<b[k]-(i+1)){q=k; l=i+1;}
        }
    }
    cout<<b[q]<<" "<<l;
}
