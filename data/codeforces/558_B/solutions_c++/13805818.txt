#include <iostream>
using namespace std;
const int maxn=1e7;
int a[maxn],b[maxn],s[maxn],e[maxn];
int main()
{
	int n,r,m=0,lp=1,rp=1;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		r=a[i];
		b[r]++;
		if(b[r]==1)
		{
			s[r]=i;
			e[r]=i;
		}
		else
		{
			e[r]=i;
			if(m<b[r])
			{
				m=b[r];
				lp=s[r];
				rp=e[r];
			}
			else if(m==b[r])
			{
				if(e[r]-s[r]+1<rp-lp+1)
				{
					lp=s[r];
					rp=e[r];
				}
			}
		}
	}
	cout<<lp<<" "<<rp;
	return 0;
}
