#include<iostream>
#define N 1000002
using namespace std;
int mas1[N], mas2[N];
int i, j, k, l, m, n, a, b, c, o;
int main(){
    cin >> n;
    for(i = 1; i <= n; ++i){
        cin >> c;
        if(!mas2[c]) mas2[c] = i; 
        ++mas1[c];
        if(mas1[c] > o || mas1[c] == o && i-mas2[c] < b-a)
            o = mas1[c], a = mas2[c], b = i;
    }
    cout << a << " " << b << " " << endl;
}
