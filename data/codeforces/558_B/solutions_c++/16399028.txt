#include <bits/stdc++.h>
using namespace std;
#define M 1000005

int n, a[M], l[M], r[M], b, c;

int main() {
  scanf("%d", &n);
  for (int i = 1, j; i <= n; i++) {
    scanf("%d", &j);
    a[j]++;
    if (l[j]) {
      r[j] = i;
    } else {
      l[j] = r[j] = i;
    }
    if (b == 0 || a[b] < a[j]) b = j;
    if (a[b] == a[j] && r[j]-l[j] < r[b]-l[b]) b = j;
  }
  printf("%d %d\n", l[b], r[b]);
	return 0;
}
