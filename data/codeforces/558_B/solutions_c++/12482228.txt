#include <iostream>
using namespace std;
int arr[1000001];
int st[1000001];
int main()
{
    int n,mx=0,x,s,e;
     cin >> n;
     for(int i = 1 ; i <= n ; i++)
     {   cin >> x;
         if(!st[x])st[x]=i;arr[x]++;
         if(arr[x]>mx||arr[x]==mx&&i-st[x]<e-s)
            {mx=arr[x];s=st[x];e=i;}}
     cout << s << " " << e;
    return 0;
}
