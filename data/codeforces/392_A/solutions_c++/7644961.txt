#include <iostream>
#include <algorithm>
using namespace std;

long long x, y, n, s;
int main() {
	cin >> n;
	if (n == 0) {
		cout << 1;
		return 0;
	}
	x = n;
	n = n*n;
	while (x > 0) {
		int k = 0;
		while ((y+1)*(y+1) <= n - x*x)
			++y, ++k;
		s += max(k-1, 0);
		while ((y+1)*(y+1) > n - x*x && x > 0)
			--x, ++s;
	}
	cout << 4*s;
}
