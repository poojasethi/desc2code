#include<stdio.h>
#include<string.h>
#include<math.h>
int main()
{
	__int64 n;
	scanf("%I64d",&n);
	__int64 i;
	__int64 max,temp;
	__int64 ans=0;
	max=n;
	for(i=1;i<=n;++i)
	{
		temp=(__int64)sqrt((double)(n*n-i*i));
		if(temp==max)ans+=1;
		else ans+=max-temp;
		max=temp;
	}
	if(n==0)printf("1\n");
	else printf("%I64d\n",ans*4);
	return 0;
}
