#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

int n,p;
char s[10010];

int main()
{
    cin>>n>>p>>s;
    int cnt=n-1;
    s[cnt]++;
    while(cnt>=0 && cnt<n){
        if(s[cnt]>'a'+p-1){
            s[cnt]='a';
            s[--cnt]++;
        }
        else if(cnt>0&&s[cnt]==s[cnt-1]){
            s[cnt]++;
        }
        else  if(cnt>1&&s[cnt]==s[cnt-2]){
            s[cnt]++;
        }
        else cnt++;
    }
    if(cnt<0) cout<<"NO";
    else cout<<s;

    return 0;
}
