#include <iostream>
#include <algorithm>
#include <cstdio>
#include <string>
#include <cstring>

using namespace std;

int n,p;
string s;

bool solve(int u)
{
	if(u<0) return false;
	s[u]++;
	if(s[u]-'a'==p){
		if(solve(u-1)){
			s[u] = 'a' - 1;
			return solve(u);
		}
	}
	else{
		if(u-1>=0 && s[u]==s[u-1]) return solve(u);
		if(u-2>=0 && s[u]==s[u-2]) return solve(u);
		return true;
	}
}

int main(void)
{
	cin>>n>>p;
	cin>>s;
	if(solve(n-1)) cout<<s<<endl;
	else printf("NO\n");
	return 0;
}
