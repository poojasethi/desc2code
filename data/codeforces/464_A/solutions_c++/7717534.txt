#include <bits/stdc++.h>
using namespace std;
int n,p;
char c[1010],s[1010];
void dfs(int u,bool fg) {
	if (u==n) {
		if (fg) puts(c),throw 0;
	} else {
		for (int i=(fg?'a':s[u]);i<'a'+p;i++) if ((u==0||i!=c[u-1])&&(u<=1||i!=c[u-2])) 
			c[u]=i,dfs(u+1,fg|(s[u]!=i));
	}
}
int main() {
	scanf("%d%d",&n,&p);
	scanf("%s",s);
	try { dfs(0,0); } catch(int e) { return 0;}
	puts("NO");
}
