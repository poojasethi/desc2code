#include <iostream>
#include <cstdio>
using namespace std;
#define pri printf
const int num = 1e3 + 5;
char g[num];
int p,n,i;

int main ()
{
//	freopen("test.txt","r",stdin);
	cin >> n >> p; p+='a';
	scanf ("%s", g);
	for (++g[ i = n - 1 ]; ~i && i < n ; ){
		if (g[i] == p)g[i] = 'a', ++g[--i];
		else if (g[i] == g[i-1] || i>1 && g[i] == g[i-2])++g[i];
		else i++;
	}
	pri("%s",!~i?"NO":g);
    return 0;
}
