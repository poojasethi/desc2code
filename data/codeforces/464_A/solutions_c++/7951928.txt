#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,m;
bool ch;
char s[1000],c[1000];
void dfs(int x)
{
	if (x==n) {
		if (ch) puts(c),throw 0;
		return ;
	}
	char st = ch?'a':s[x];
	for (int i=st;i<'a'+m;i++)
	if ((x==0||i!=c[x-1])&&(x<=1||i!=c[x-2])){
		if (!ch&&s[x]!=i) ch = true;
		c[x]=i;
		dfs(x+1);
	}
} 

int main()
{
//	freopen("1.in","r",stdin);
	scanf("%d%d",&n,&m);
	scanf("%s",s);
	try { dfs(0); } catch(int e) {
		return 0;
	}
	printf("NO");
}
