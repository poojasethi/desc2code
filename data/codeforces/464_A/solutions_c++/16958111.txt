#include <bits/stdc++.h>
using namespace std;
int x,y;
string s;
bool flag;
void tot(int i){
 if(i<0){
  cout<<"NO"<<endl;
  exit(0);
 }
 to:
 s[i]++;
 while((s[i]-97)<y){
  if((s[i]!=s[i-1])&&(s[i]!=s[i-2])){
   flag=true;
   break;
  }
  //cout<<s[i]<<endl;
  s[i]++;
 }
 if(flag){
  flag=false;
  return;
 }
 else{
  s[i]=96;
  tot(i-1);
  goto to;
 }
}
int main(){
 cin>>x>>y>>s;
 tot(x-1);
 cout<<s<<endl;
}
