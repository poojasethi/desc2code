

#include <stdio.h>
int n,p;
char s[1024],t[1024];
int go(int i,int f){
  if(i<n){
    char&c=t[i];
    for(c=f?'a':s[i];c<'a'+p;++c)
      if(c!=t[i-1]&&c!=t[i-2]&&go(i+1,f||c!=s[i]))
        return 1;
    return 0;
  }else return f;
}
int main(){
  scanf("%d%d%s",&n,&p,s+2),n+=2;
  puts(go(2,0)?t+2:"NO");
}
