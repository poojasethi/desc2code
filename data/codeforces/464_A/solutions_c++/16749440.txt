#include <bits/stdc++.h>
using namespace std;
string s;
int p,n,k,f;
int main(){
	cin>>n>>p>>s;
	k=n-1;
	s[k]++;
	while(k>=0&&k<n){
		if(s[k]-'a'>=p){
			s[k]='a';
			k--;
			if(k>=0)s[k]++;
		}
		else if(k&&(s[k]==s[k-1])||k>1&&(s[k]==s[k-2])){
			s[k]++;
		}
		else k++;
	}
	(k<0)?cout<<"NO\n":cout<<s<<endl;
}
