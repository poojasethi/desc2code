#include<bits/stdc++.h>
using namespace std;
#define no printf("NO\n");
main()
{
    char a[1005];
    int n,p;
    cin>>n>>p;
    cin>>a;
    int i=n-1;
    ++a[i];
    while(i>=0&&i<n)
    {
        int x=a[i]-'a';
        if(x>=p)
        {
            //cout<<"hi";
            a[i]='a';
            i--;
            if(i>=0){++a[i];}
        }
        else if(i-1>=0 && a[i]==a[i-1]){++a[i];}
        else if(i-2>=0 && a[i]==a[i-2]){++a[i];}
        else i++;
    }
    if(i<0){no}
    else {cout<<a<<endl;}
}
