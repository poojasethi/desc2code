#include<cstdio>
using namespace std;
char s[1003];
int main()
{
    int n,p,a,b;
    scanf("%d%d%s",&n,&p,s+2);
    n+=2;
    for(a=n-1;a>1;--a)
    {
        b=s[a]+1;
        while(b==s[a-1]||b==s[a-2])
            ++b;
        if(b!=97+p)
        {s[a]=b;break;}
    }
    if(a==1){puts("NO");return 0;}
    for(int i=a+1;i<n;++i)
    {
        s[i]=97;
        while(s[i]==s[i-1]||s[i]==s[i-2])
            ++s[i];
    }
    printf("%s\n",s+2);
}
