#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
const ll maxn=1000005,M=1000000007;
ll a[maxn],n,c[maxn],f[maxn],m,pre[maxn],cnt[maxn],ans;
ll get(ll i){ll ret=0;for(;i>0;i-=i&(-i))ret=(ret+c[i])%M;return ret;}
ll add(ll i,ll d){for(;i<=m;i+=i&(-i))c[i]=(c[i]+d)%M;}
int main()
{
	cin>>n;
	for(ll i=1;i<=n;i++)scanf("%d",&a[i]),m=max(a[i],m);
	for(ll i=1;i<=n;i++){
		f[i]=(get(a[i])*a[i]%M+a[i]-cnt[a[i]]+M)%M;
		add(a[i],f[i]);cnt[a[i]]=(cnt[a[i]]+f[i])%M;ans=(ans+f[i])%M;
	}
	cout<<ans;return 0;
}
