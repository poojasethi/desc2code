#include<iostream>
using namespace std;
typedef long long ll;

const ll D = 1000LL*1000*1000+7;
const int MAXN = 100*1000+1;
const int MAXL = 1000*1000+1+1;

int t[MAXL+10];
int l[MAXL+10];
ll tmp,n;

void add(int x,ll val)
{
	for(; x<=MAXL; x+=x&(-x))
	{
		t[x] += val;
		if( t[x] >= D )
			t[x] -= D;
	}
}

ll sum(int x)
{
	ll ret = 0;
	for(; x>0; x-=x&(-x))
	{
		ret += t[x];
		if( ret >= D )
			ret -= D;
	}
	return ret;
}

int main()
{
	ios::sync_with_stdio(false);
	cin >> n;
	for(int i=0; i<n; i++)
	{
		cin >> tmp;
		ll s = ( tmp*(sum(tmp)+1) )%D;
		add(tmp,((s-l[tmp])+D)%D);
		l[tmp] = s;
	}
	cout << sum(MAXL) << endl;
	return 0;
}
