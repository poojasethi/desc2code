import sys
K=int(raw_input())
S=raw_input()
if K>len(S):
    print'NO'
    sys.exit()
Ans=[]
Firsts=[]
for s in S:
    if not s in Firsts and not len(Ans)>=K:
        Firsts.append(s)
        Ans.append([s])
    else:
        Ans[-1].append(s)
if len(Ans)>=K:
    print'YES'
    for A in Ans:
        print ''.join(A)
else:
    print'NO'
