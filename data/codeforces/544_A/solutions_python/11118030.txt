n = int(raw_input())
q = raw_input()

m = len(q)

if n==1:
	print 'YES'
	print q
	exit()
		
if m < n:
	print 'NO'
	exit() 

mark = [0]*26
mark[ord(q[0]) - ord('a')] = 1

s = q[0]
v = []
t = n
for i in range(1,m):
	z = ord(q[i]) - ord('a')
	if not mark[z] and t-1:
		v.append(s)
		mark[z] = 1
		s = q[i]
		t-=1
	else:
		s = s + q[i]	

if len(s):
	v.append(s)		

if len(v) != n:
	print 'NO'
else:
	print 'YES'
	for _ in v:
		print _
