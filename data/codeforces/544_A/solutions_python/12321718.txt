from sys import stdin
n = int(stdin.readline())
a = stdin.readline().strip()
di = {}
l = len(a)
for i in xrange(l):
	ch = a[i]
	if di.get(ch,-1)==-1:
		di[ch]  =i
if len(di)< n:
	print "NO"
else:
	print "YES"
	b = []
	for i in di:
		b.append(di[i])
	b.sort()
	b=b[:n]
	for i in xrange(n-1):
		print a[b[i]:b[i+1]]
	print a[b[-1]:]
