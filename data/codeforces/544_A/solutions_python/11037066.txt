k = int(raw_input())
s = raw_input()

cnt = []
vis = [False]*26
now = 0
l = len(s)
i = 0
while i < l and now < k:
	index = ord(s[i]) - ord('a')
	if(vis[index] == False):
		cnt.append(i)
		vis[index] = True
		now += 1
	else:
		cnt[now - 1] = i;
	i += 1

if now < k:
	print "NO"
else:
	cnt[now - 1] = l - 1;
	print "YES"
	now = 0
	for i in range(0,k):
		print s[now:cnt[i]+1]
		now = cnt[i]+1
