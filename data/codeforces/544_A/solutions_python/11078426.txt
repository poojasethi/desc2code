
def main():
	k = int(raw_input())
	q = raw_input()

	count = 1
	L = list()
	LL = list()
	LL.append(q[0])

	prev = 0
	success = False

	for x in xrange(0,len(q)):
		
		if q[x] not in LL:
			count+=1
			L.append(q[prev:x])
			LL.append(q[x])
			prev = x
		if count==k:
			L.append(q[prev:])
			success = True
			break
			
	if success:
		print "YES"
		for x in L:
			print x
	else:
		print "NO"


if __name__ == '__main__':
	main()
