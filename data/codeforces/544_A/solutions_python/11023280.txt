n = int(raw_input())
q = raw_input()
S = set()
S.add(q[0])
res = []
p = 0
for i in range(1, len(q)):
    if len(res) + 1 == n:
        break
    if q[i] not in S:
        res.append(q[p:i])
        S.add(q[i])
        p = i

if len(res) + 1 == n:
    res.append(q[p:])
    print "YES"
    for x in res:
        print x
else:
    print "NO"
