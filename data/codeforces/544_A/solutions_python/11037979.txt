S = rr()
Slen = len(S)
seen = set([])
ix = []
for i in xrange(Slen):
    if S[i] not in seen:
        ix.append(i)
        seen.add(S[i])
if len(ix)>= N:
    print "YES"
    ix = ix[:N]+[-1]
    ans = []
    for i in xrange(N):
        if ix[i+1]==-1:
            ans.append(S[ix[i]:])
        else:
            ans.append(S[ix[i]:ix[i+1]])
    for i in ans: print i
