n = int(raw_input())
str = raw_input()

seen = set([str[0]])
starts = [0]
for i in range(1, len(str)):
    c = str[i]
    if c not in seen:
        seen.add(c)
        starts.append(i)
if len(starts) >= n:
    print("YES")
    starts.append(-1) # dummy
    starts[n] = len(str)
    for i in range(0, n):
        start = starts[i]
        end = starts[i + 1]
        print(str[start:end])
else:
    print("NO")
