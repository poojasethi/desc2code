k = int(raw_input())
S = raw_input()

X = [0]
C = [S[0]]

for i in xrange(1, len(S)):
    if len(X) == k:
        break
    if S[i] not in C:
        X.append(i)
        C.append(S[i])

if len(X) != k:
    print "NO"
else:
    X.append(len(S))
    print "YES"
    for i in xrange(k):
        print S[X[i]:X[i+1]]
