n = int(raw_input())
r = raw_input()
l = list(r)
a = l[0]
add = 1
dic = {}
dic[1] = []
l_1 = [l[0]]
for x in l:
	if add == n:
		dic[add].append(x)
	else:
		if (x in l_1)==True:
			dic[add].append(x)
		else:
			add = add+1
			dic[add] = [x]
			l_1.append(x)
if add == n:
	print 'YES'
	for p in range(1,add+1):
		print ''.join(dic[p])
else:
	print 'NO'
