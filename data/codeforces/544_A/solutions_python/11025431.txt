k = input()
q = raw_input()

if k == 1:
	print "YES" 
	print q
elif k > len(q):
	print "NO"
else:
	first = {k: 0 for k in range(0, 26)}
	s = []
	t = 0
	for i in range(0, len(q)):
		if first[ord(q[i]) - 97] == 0 and t < k:
			t += 1
			s.append(q[i])
			first[ord(q[i]) - 97] = 1
		else:
			s[t-1] += q[i]

	if k == t: 
		print "YES"
		for i in range(len(s)):
			print s[i]
	else:
		print "NO"
