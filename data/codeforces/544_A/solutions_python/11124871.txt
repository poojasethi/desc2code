#!/usr/bin/python
# -*-coding:utf-8 -*-

k = int(raw_input())
s = raw_input()
result = []
flag = set()
cursor = 0
while k and cursor < len(s):
    if s[cursor] in flag:
        result[-1] += s[cursor]
    else:
        flag.add(s[cursor])
        result.append(s[cursor])
        k -= 1
    cursor += 1
result[-1] += s[cursor:]
if k:
    print "NO"
else:
    print "YES"
    for item in result:
        print item
