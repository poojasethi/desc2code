n=input()
s=raw_input()
c=0
r=[]
a=set(s[0])
t=1
for i in range(1,len(s)):
    if s[i] not in a:
        #print i,s[i]
        if len(r) + 1 == n:
            r += [s[c:]]
            t=0
            break
        r += [s[c:i]]
        c=i
        a.add(s[i])
if t:
    r += [s[c:]]
#print r
if len(r) == n:
    print 'YES\n'+'\n'.join(r)
else:
    print 'NO'
