k = int(raw_input())
q = str(raw_input())
lets = []
last = 0
s = set()
s.add(q[0])
for i in range(1,len(q)):

    if q[i] != q[i-1] and q[i] not in s:
        lets.append(q[last:i])
        s.add(q[i])
        last = i
lets.append(q[last:len(q)])
if len(lets) >= k:
    print "YES"
    for i in range(k-1):
        print lets[i]
    print "".join(lets[k-1:])
else:
    print "NO"
