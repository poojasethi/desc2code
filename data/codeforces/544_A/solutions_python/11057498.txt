#!/usr/bin/python

import sys

def Ni(): return map(int, raw_input().split())
def Ns(): return raw_input().split()

k = Ni()[0]
q = Ns()[0]

letters = set()
answer = []
for qi, ql in enumerate(q):
	if k > 0 and ql not in letters:
		if qi != 0:
			answer.append(qi)
		k -= 1
		letters.add(ql)

if k == 0:
	print "YES"
	s = 0
	for a in answer:
		print q[s:a]
		s = a
	print q[s:]
else:
	print "NO"
