k = input()
s = raw_input()
hs = set()
for c in s: hs.add(c)
if len(hs) < k: print "NO"
else:
	print "YES"
	vis = [False] * 26
	pre = -1
	while k > 0:
		if 1 == k: print s[pre + 1:]
		else:
			for i in xrange(pre + 1, len(s)):
				if not vis[ord(s[i]) - ord('a')]:	
					vis[ord(s[i]) - ord('a')] = True
					if i != 0: 
						print s[pre + 1 : i]
						pre = i - 1
						break
		k -= 1
