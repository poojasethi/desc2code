#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	string s;
	cin>>s;
	int h[130]={0};
	for(int i=0;i<s.size();i++)
	h[s[i]]++;

	int c=0;
	for(int i=0;i<130;i++)
	if(h[i])c++;

	if(c<n){
	cout<<"NO"<<endl;
	return 0;
	}
	cout<<"YES"<<endl;
	int p[s.size()]={0};
	for(int i=0;i<s.size();i++)
	if(h[s[i]]&&n>0)
	{p[i]=1;h[s[i]]=0;n--;}

	for(int i=0;i<s.size();i++)
	{
		if(p[i]&&i!=0)cout<<endl;
		cout<<s[i];
	}
	cout<<endl;
	return 0;
}
