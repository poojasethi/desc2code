#include <bits/stdc++.h>
using namespace std;
int a[100],b[100],n,m,i;
string s;
int main()
{
	scanf("%d",&n);
	cin>>s;
	for(i=0;i<s.size()&&m<n;i++)
	{
		if(a[s[i]]==0)
		{
			a[s[i]]=1;
			b[m++]=i;
		}
	}
	if(m==n)
	{
		printf("YES");
	}
	else 
	{
		printf("NO");
		return 0;
	}
	m=0;
	for(i=0;i<s.size();i++)
	{
		if(b[m]==i)
		{
			m++;
			printf("\n");
		}
		printf("%c",s[i]);
	}
}
