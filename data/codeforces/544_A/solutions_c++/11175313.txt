#include <bits/stdc++.h>

using namespace std;

bool vis[30];

int main(){
    int k;
    string x;
    string ans = "YES";

    cin >> k >> x;

    for (int i=0;i<x.length();i++){
        if(k && vis[ x[i]-'a' ]==0)     {ans += "\n";k--;}

        vis[x[i]-'a'] = 1;
        ans += x[i];
    }

    if (k == 0){
//        cout << "YES" << endl;
        cout << ans << endl;
    }

    else cout << "NO" << endl;

return 0;
}
