#include <bits/stdc++.h>
using namespace std;

string s;
int k,f[1<<7];
vector<int> res;

int main()
{
	cin>>k>>s;
	int l=s.length();

	memset(f,-1,sizeof(f));
	for(int i=0;i<l;i++)
		if(f[s[i]]==-1)
		{
			res.push_back(i);
			f[s[i]]=i;
		}

	if(res.size()<k)
		puts("NO");
	else
	{
		puts("YES");
		for(int i=0;i<k-1;i++)
			cout<<s.substr(res[i],res[i+1]-res[i])<<endl;
		cout<<s.substr(res[k-1],l-res[k-1])<<endl;
	}
	return 0;
}
