#include <bits/stdc++.h>using namespace std;const int N=1e6, INF=1e9;int pos[N];int main(){	fill_n(pos,N, INF);	int n,cnt=0;	scanf("%d", &n);	string s;	cin>>s;	for(int i=0; s[i]; i++){		cnt+=pos[s[i]-'a']==INF;		pos[s[i]-'a']=min(pos[s[i]-'a'], i);	}	sort(pos, pos+N);	pos[n]=s.size();	puts(cnt>=n?"YES":"NO");	for(int i=0; cnt>=n&&i<n; i++){		int l=pos[i], r=pos[i+1]-pos[i];		puts(s.substr(l, r).c_str());	}	///de	return 0;}
