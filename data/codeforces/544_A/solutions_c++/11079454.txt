#include <bits/stdc++.h>
using namespace std;
char s[111];
bool vis[26], d[111];
int main() {
  int n; scanf("%d %s", &n, s);
	for (int i = 0; s[i] && n > 0; ++i) {
		if (!vis[s[i] - 'a']) {
			--n;
			vis[s[i] - 'a'] = 1;
			d[i] = 1;
		}
	}
	if (n == 0) {
		puts("YES");
		for (int i = 0; s[i]; ++i) {
			if (i > 0 && d[i]) puts("");
			putchar(s[i]);
		}
		puts("");
	} else {
		puts("NO");
	}
  return 0;
}
