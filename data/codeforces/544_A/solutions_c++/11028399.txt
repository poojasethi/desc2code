#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int m[129],i,d,n;
int main()
{
	char s[129];
	cin>>n>>s;
	for(i=-1;s[++i];)if(!m[s[i]])++d,m[s[i]]=1;
	if(d<n){
		cout<<"NO";
		return 0;
	}else{
		cout<<"YES";
		if(d>=n)for(i=-1;s[++i];){
			if(n&&m[s[i]]!=2){
				--n;
				m[s[i]]=2;cout<<"\n";
			}
			cout<<s[i];
		}	
	}
	return 0;
}
