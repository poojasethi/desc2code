#include<iostream>
#include<string>
using namespace std;
int main()
{
string s;
int j=0,k,k1=0;
cin>>k;
cin>>s;
int a[26];
int b[26];
for(int i=0;i<26;i++)
a[i] = 0;
for(int i = 0;i<s.length();i++)
{
if (a[s[i]-'a']==0)
{
k1++;
a[s[i]-'a'] = 1;
if(j<k)
b[j++] = i;
}
}
if(k1<k)
{
cout<<"NO";
}
else
{
cout<<"YES\n";
j = 1;
for(int i=0;i<s.length();i++)
{
if(i>0&&i==b[j]&&j<k)
{
cout<<"\n";
j++;
}
cout<<s[i];
}
}
return 0;
}
