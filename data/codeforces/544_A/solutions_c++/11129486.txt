#include <cstdio>
#include <cstring>

using namespace std;

int k;
int t[105], d[105];
char s[105];

int main() {
	scanf("%d%s", &k, s);
	int n = strlen(s);
	for (int i = 0; i < n && k > 0; i++) {
		if (!t[s[i]-'a']) {
			t[s[i]-'a'] = 1;
			d[i] = 1;
			k--;
		}
	}
	if (k) {
		puts("NO");
		return 0;
	}
	puts("YES");
	for (int i = 0; i < n; i++) {
		if (d[i] && i) putchar('\n'); 
		putchar(s[i]);
	}
	return 0;
}
