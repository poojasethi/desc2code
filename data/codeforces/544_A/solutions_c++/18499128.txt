#include <bits/stdc++.h>
using namespace std;
map<char,bool> m;
int main()
{
	int n,k;
	string s;
	cin>>k>>s;
	n=s.size();
	for(int i=0;i<n;i++) m[s[i]]=false;
	if(m.size()<k) cout<<"NO";
	else{
		int i=0;
		cout<<"YES"<<endl;
		m[s[0]]=true;
		while(k>1){
			cout<<s[i];
			i++;
			if(i && s[i-1]!=s[i] && !m[s[i]]){
				k--;
				m[s[i]]=true;
				cout<<endl;
			}
		}
		for(int k=i;k<n;k++){
			cout<<s[k];
		}
	}
}
