#include<cstdio>
using namespace std;
int a1[26];bool b1[26];
char s[101];
int main()
{
    int n=0,k;
    scanf("%d%s",&k,s);
    for(int i=0;s[i];++i)
    if(n<k&&!b1[s[i]-97])
        a1[n++]=i,b1[s[i]-97]=1;
    if(n<k){puts("NO");return 0;}
    puts("YES");n=1;
    for(int i=0;s[i];++i)
    {
        if(i==a1[n])puts(""),++n;
        putchar(s[i]);
    }
}
