# include<iostream>
# include<string.h>
# include<map>
using namespace std;
int main(){
map<char,int>ahmed;
char input[101];
int num,temp;
cin>>num;
cin>>input;
temp=strlen(input);
for(int k=0;k<temp;k++)
   ahmed[input[k]]=1;
if(num>ahmed.size())
   cout<<"NO";
else{
  cout<<"YES"<<endl;
  cout<<input[0];
  ahmed[input[0]]=0;
for(int k=1;k<temp;k++){
   if(ahmed[input[k]]==1&&num>1){
   cout<<endl;
   ahmed[input[k]]=0;
   num--;
   }
   cout<<input[k];
}
}
}
