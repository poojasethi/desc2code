#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	int n,l,a[26]={0},c=0;
	set<char> se;
	char s[110];
	cin>>n;
	scanf("%s",s);
	for(int i=0;i<strlen(s);i++)
		se.insert(s[i]);
	l=se.size();
	if(l<n)
		cout<<"NO\n";
	else
	{
		cout<<"YES\n";
		for(int i=0;i<strlen(s);i++)
		{
			if(!a[s[i]-97]&&c<n)
			{
				a[s[i]-97]=1;
				if(c>0)
					cout<<endl;
				c++;
			}
			cout<<s[i];
		}
		cout<<endl;
	}
	return 0;
}
