#include <cstdio>
#include <cstring>
using namespace std;

char ch[200];
int n,l;
int q[200],usd[256],cnt;

int main()
{
	scanf("%d",&n);
	scanf("%s",ch);
	l=strlen(ch);
	for (int i=0;i<l;i++) if (!usd[ch[i]] && cnt<n) q[cnt++]=i,usd[ch[i]]=1;
	if (cnt<n) printf("NO\n");
	else {
		printf("YES\n");
		q[cnt]=l;
		for (int i=0;i<cnt;i++)
		{
			for (int j=q[i];j<q[i+1];j++) putchar(ch[j]);
			putchar('\n');
		}
	}
	return 0;
}
