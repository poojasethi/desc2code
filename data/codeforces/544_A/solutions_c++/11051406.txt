#include<set>
#include<string>
#include<iostream>
using namespace std;
int main()
{
	int n,i,m=0;
	string a;
	set<char> s;
	cin>>n>>a;
	for(i=0;i<a.length();i++)s.insert(a[i]);
	if(s.size()<n){cout<<"NO";}
	else
	{
		s.clear();
		cout<<"YES";
		for(i=0;i<a.length();i++)
		{
			if(!s.count(a[i])&&m<n)
			{
				cout<<'\n';
				m++;
			}
			cout<<a[i];
			s.insert(a[i]);
		}
	}
	return 0;

}
