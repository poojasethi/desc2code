#include<iostream>
#include<string>
using namespace std;

string a ;
bool p[105] ;
bool alpha[26];
int n,x=0;

int main()
{
	cin>>n>>a;
	for(int i=0 ; i<a.size() && x<n ; i++)
		if(alpha[a[i]-'a'] == 0){
			p[i]=1;
			alpha[a[i]-'a'] =1;
			x++;
		}
	if(x<n)
		cout<<"NO\n";
	else{
		cout<<"YES\n";
		for(int i=0 ; a[i]!=0 ; i++){
			if(i && p[i]==1)
				cout<<" ";
			cout<<a[i];
		}
	}
	return 0;
}
