#include<bits/stdc++.h>
using namespace std;

int k , f[256] , mrk[101] , cnt=0 , done=0 ;
string a ;

int main(){
  memset(f,-1,sizeof(f)) ;
  cin >> k >> a ;
  for(int i=0 ; i<a.size() ; i++) if(f[a[i]]==-1) f[a[i]]=i , mrk[i]=1 , cnt++ ; 
  if(cnt<k) { printf("NO") ;  return 0 ;}

  printf("YES") ; 
  for(int i=0 ; i<a.size() ; i++) {
    if(mrk[i] && done<k) puts("") , done++ ;
    cout << a[i] ;  
  }   

  return 0;
}
