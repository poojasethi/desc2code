#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxl=120,maxk=30;
int K,l,ans[maxk];
char s[maxl];
bool vis[maxk];
int main()
{
	int i,j,k;
	scanf("%d%s",&K,s);
	l=strlen(s);
	for(i=0;i<l&&k<K;i++)
		if(!vis[s[i]-'a'])
			ans[k++]=i,vis[s[i]-'a']=1;
	if(k<K) printf("NO\n");
	else
	{
		printf("YES\n");
		ans[K]=l;
		for(i=0;i<K;i++)
		{
			for(j=ans[i];j<ans[i+1];j++)
				printf("%c",s[j]);
			printf("\n");
		}
	}
	return 0;
}
