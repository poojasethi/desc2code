#include<bits/stdc++.h>
using namespace std;
int main()
{
	string a;
	int n,i,enter[105]={0},mark[30]={0},ans=0,k=0;
	cin>>n>>a;
	int len=a.length();
	for(i=0;i<len;i++)
	{
		int x=a[i]-'a';
		if(mark[x]==0)
		{
			enter[i-1]=1;
			ans++;
			mark[x]=1;
		}
	}
	if(ans>=n)
	{
		cout<<"YES"<<endl;
		for(i=0;i<len;i++)
		{
			cout<<a[i];
			if(enter[i]==1&&k<n-1)
			{
				cout<<endl;
				k++;
			}
		}
	}
	else	cout<<"NO"<<endl;
}
