#include <bits/stdc++.h>
using namespace std;
char ind[101],hash[26];
int main() {
int i,c=0,k;char a[101];
scanf("%d%s",&k,a);
for(i=0;a[i]&&c<k;i++)
if(!hash[a[i]-'a']) {
ind[i]=1;
hash[a[i]-'a']=1;
c++;
}
if(c!=k) printf("NO");
else {	printf("YES\n");
	string s="";
	for(i=0;a[i];i++) 
	{	if(ind[i]&&s!="") cout<<s<<endl,s="";
		s+=a[i];
	}
	if(s!="") cout<<s<<endl;
}
return 0;
}
