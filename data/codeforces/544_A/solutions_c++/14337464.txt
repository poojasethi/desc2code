#include<cstdio>

int k, ans;
char s[333];
int tot[555], v[555];

int main() {
	scanf("%d%s", &k, s);
	ans = 0;
	for(int i=0;s[i];i++) {
		if (!tot[s[i]]) ans++;
		tot[s[i]] ++;
	}
	if(ans < k) printf("NO\n");
	else {
		printf("YES\n%c", s[0]);
		v[s[0]] = true;k--;
		for(int i=1;s[i];i++) {
			if (k && !v[s[i]]) {
				v[s[i]] = true;
				k--; printf("\n");
			}
			printf("%c", s[i]);
		}
		printf("\n");
	}
	
	return 0;
}
