#include<stdio.h>
#include<string.h>

int n,i,j,H[10000],t,W[10005];
char arr[100005];

int main(){

  scanf("%d %s",&n,arr);
  int u = strlen(arr);

  for(i = 0; i < u; i++){
	  if(!H[arr[i]]){
		  H[arr[i]]++;
		  W[i]++;
		  t++;
	     if(t >= n)
			  break;
	  }
  }
  
  if(t < n){ puts("NO"); return 0; }

  puts("YES");
  printf("%c",arr[0]);
  for(j = 1; j < u; j++){
	  if(W[j])
		  printf("\n");
     printf("%c",arr[j]);
  }

  return 0;
}
