#include <bits/stdc++.h>

using namespace std;
int t[27]={0};
bool used[27]={false};



int main()
{
int n,j=0;
cin>>n;
string s;cin>>s;
for(int i=0;i<s.length();i++)
{
 if(!used[s[i]-'a']){t[j++]=i;used[s[i]-'a']=true;}
}

if(j<n){cout<<"NO";return 0;}
cout<<"YES"<<endl;

for(int i=0;i<n-1;i++)cout<<s.substr(t[i],t[i+1]-t[i])<<endl;

cout<<s.substr(t[n-1],s.length()-t[n-1]);

    return 0;
}
