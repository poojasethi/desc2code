#include <iostream>
#include <string>
using namespace std;
int T[26], k, C[100];
string str;
int main(){
	cin >> k >> str;
	for (int i = 0; i < str.size() && k > 0; i++){
		if (!T[str[i] - 'a']){
			T[str[i] - 'a'] = 1;
			k--;
			C[i] = 1;
		}
	}
	if (k){
		cout << "NO";
		return 0;
	}
	cout << "YES" << endl;
	for (int i = 0; i < str.size(); i++){
		if (C[i] && i) cout << endl;
		cout << str[i];
	}
	return 0;
}
