#include <iostream>
#include <string>
using namespace std;
bool A[26];
int d[26];
int main() {
	string s; int k, cur = 0;
	cin >> k >> s;
	for(int i = 0; i < s.size() && cur < k; ++i)
		if(!A[s[i] - 'a']) {
			A[s[i] - 'a'] = 1;
			d[cur++] = i;
		}
	if(cur != k) { cout << "NO"; return 0; }
	cur = 1;
	cout << "YES" << endl;
	for(int i = 0; i < s.size(); ++i) {
		if(cur != k && d[cur] == i) { cout << endl; ++cur;}
		cout << s[i];
	}
}
