#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
char str[110],n;
bool vis[26];
int main(){
	scanf("%d\n",&n);
	gets(str);
	int begin[110]={0},j=0;
	for(int i=0;str[i]&&j<n;++i)
		if(!vis[str[i]-'a']) begin[j++]=i,vis[str[i]-'a']=1;
	if(j<n)puts("NO");
	else{
		puts("YES");
		begin[n]=strlen(str);
		for(int i=0;i<n;++i){
			for(int j=begin[i];j<begin[i+1];++j) putchar(str[j]);
			putchar('\n');
		}
	}
}
