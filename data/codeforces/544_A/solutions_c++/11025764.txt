#include<bits/stdc++.h>
using namespace std;
main()
{
    map<char,bool>used;
    int k;
    string s,a[27];
    cin>>k>>s;
    int f=0;
    for (int i=0;i<s.size();i++)
    {
        if (!used[s[i]] && f<k)
        {
            f++;
            used[s[i]]=true;
        }
        a[f]+=s[i];
    }
    if (f<k)
    {
        cout<<"NO"<<endl;
        return 0;
    }
    cout<<"YES"<<endl;
    for (int i=1;i<=k;i++)
    {
        cout<<a[i]<<endl;
    }
}
