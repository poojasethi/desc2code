#include<cstdio>
#include<cstring>
#define N 101000
#define M 3010
int n,len,i,f[N],g[N];
char s[N];
int main()
{
	scanf("%d",&n);
	scanf("%s",s);
	len=strlen(s);
	for (i=0;i<len;i++)
	if ((f[s[i]]==0)&&n)
	{
		g[i]=1;
		f[s[i]]=1;
		n--;
	}
	if (n)
	printf("NO");
	else
	{
		printf("YES\n");
		g[0]=0;
		for (i=0;i<len;i++)
		{
			if (g[i]) printf("\n");
			printf("%c",s[i]);
		}
	}
}
