#include <bits/stdc++.h>
using namespace std;
char s[105];
int freq[128], freq2[128];
int main() {
	int n, l, c = 0;
	scanf("%d %s", &n, s);
	l = strlen(s);
	for (int i = 0; i < l; i++) {
		c += (freq[s[i]]++ == 0);
	}
	if (n > c) {
		printf("NO\n");
		return 0;
	}
	printf("YES");
	for (int i = 0; i < l; i++) {
		if (n && (freq2[s[i]]++ == 0)) {
			printf("\n");
			n--;
		}
		printf("%c", s[i]);
	}return 0;
}
