#include <bits/stdc++.h>

using namespace std;

int main() {
  int k;
  string s;
  cin>>k>>s;
  string V[k];
  V[0]=s[0];
  int c=0;
  for(int i=1;i<s.size();i++) {
    bool sw=false;
    for(int j=0;j<=c;j++)if(V[j][0]==s[i])sw=true;
    if(sw || c==k-1) {
      V[c]+=s[i];
    }else {
      c++;
      V[c]=s[i];
    }
  }
  if(c==k-1) {
    cout<<"YES\n";
    for(int i=0;i<k;i++)cout<<V[i]<<"\n";
  }else cout<<"NO\n";
  return 0;
}
