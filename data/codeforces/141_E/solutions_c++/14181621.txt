#include<iostream>
#include<vector>
using namespace std;
const int N=100100;
int from[N],to[N],p[N];
char ch[N];
vector<int>ans;
int fnd(int a)
{
    return p[a]==a?a:p[a]=fnd(p[a]);
}
void mrg(int a,int b)
{
    p[fnd(a)]=fnd(b);
}
int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=0;i<m;i++)
    {
        cin>>from[i]>>to[i]>>ch[i];
        from[i]--;
        to[i]--;
    }
    if(n%2==0)
    {
        cout<<-1<<endl;
        return 0;
    }
    for(int i=0;i<n;i++)
        p[i]=i;
    for(int i=0;i<m;i++)
        if(ch[i]=='S')
            mrg(from[i],to[i]);
    for(int i=0;i<m;i++)
        if(ch[i]=='M' && fnd(from[i])!=fnd(to[i]))
            ans.push_back(i+1),mrg(from[i],to[i]);
    for(int i=0;i<n;i++)
        p[i]=i;
    for(int i=0;i<ans.size();i++)
        mrg(from[ans[i]-1],to[ans[i]-1]);
    for(int i=0;i<m && ans.size()<(n-1)/2;i++)
        if(ch[i]=='M' && fnd(from[i])!=fnd(to[i]))
            ans.push_back(i+1),mrg(from[i],to[i]);
    if(ans.size()!=(n-1)/2)
    {
        cout<<-1<<endl;
        return 0;
    }
    for(int i=0;i<m && ans.size()<n-1;i++)
        if(ch[i]=='S' && fnd(from[i])!=fnd(to[i]))
            ans.push_back(i+1),mrg(from[i],to[i]);
    cout<<ans.size()<<endl;
    for(int i=0;i<ans.size();i++)
        cout<<ans[i]<<" ";
    cout<<endl;
    return 0;
}
