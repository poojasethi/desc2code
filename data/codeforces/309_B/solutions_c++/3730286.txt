#include <iostream>
#include <cstdio>
#include <vector>
#include <string>

using namespace std;

const int MAXN = 1000001;
int a[MAXN];
string s[MAXN];
int f[MAXN][22];
int main()
{
    //freopen("input.txt", "r", stdin);
    int n, r, c;
    cin >> n >> r >> c;
    c++;
    for (int i = 0; i < n; i++)
    {
        cin >> s[i];
        a[i] = s[i].length() + 1;
    }
    int cur = 0;
    for (int i = 0; i < n; i++)
    {
        if (i == 0) f[i][0] = 0;
        else f[i][0] = f[i - 1][0];
        while ((f[i][0] < n) && (cur + a[f[i][0]] <= c))
        {
            cur += a[f[i][0]];
            f[i][0]++;
        }
        cur -= a[i];
    }
    f[n][0] = n;
    for (int j = 1; j < 22; j++)
    {
        f[n][j] = n;
        for (int i = 0; i < n; i++) f[i][j] = f[f[i][j - 1]][j - 1];
    }
    int mx = 0, mid = 0;
    for (int i = 0; i < n; i++)
    {
        int cr = r, cur = i;
        for (int j = 0; j < 22; j++)
        {
            if (cr % 2) cur = f[cur][j];
            cr /= 2;
        }
        if (cur - i > mx) mx = cur - i, mid = i;
    }
    cur = mid;
    for (int i = 0; i < r; i++)
    {
        for (int j = cur; j < f[cur][0]; j++)
        {
            cout << s[j];
            if (j == f[cur][0] - 1) cout << endl;
            else cout << ' ';
        }
        cur = f[cur][0];
    }
    return 0;
}
