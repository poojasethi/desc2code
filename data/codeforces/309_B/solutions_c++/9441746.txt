#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

const int N = 1000005;

#define fo(i , st , en) for (int i = st; i <= en; i++)
#define fd(i , st , en) for (int i = st; i >= en; i--)
#define Me(x , y) memset(x , y , sizeof(x))

int a[N] , sum[N] , f[N] , ans[N];
int n , r , c;
char ch[N * 6];

int main(){
	scanf("%d%d%d" , &n , &r , &c);
	fo (i , 1 , n){
		scanf("%s" , ch + sum[i - 1] + 2); a[i] = strlen(ch + sum[i - 1] + 2); sum[i] = sum[i - 1] + 1 + a[i];
	}
	fo (i , 1 , n)
		for (f[i] = max(f[i - 1] , i); f[i] <= n && sum[f[i]] - sum[i - 1] - 1 <= c; f[i]++);
	f[n + 1] = n + 1;
	fo (i , 1 , n) ans[i] = i;
	for (; r; r >>= 1)
		fo (i , 1 , n){
			if (r & 1) ans[i] = f[ans[i]];
			f[i] = f[f[i]];
		}
	int pos = 1;
	fo (i , 2 , n)
		if (ans[i] - i > ans[pos] - pos) pos = i;
	int last = 0;
	fo (i , pos , ans[pos] - 1){
		if (last + a[i] > c){
			last = 0; puts("");
		}
		if (last) putchar(' '); last += a[i] + 1;
		printf("%s" , ch + sum[i - 1] + 2);
	}
	return 0;
}
