const int N=1010000;
int nxt[N],f[N],ret[N],len[N];

int main() {
	scanf("%d%d%d",&n,&r,&c); ++c;
	int l=0;
	rep(i,1,n+1) {
		scanf("%s",s+l);

		while (t<=n&&len[t]-len[i-1]<=c) t++;

	for (int b=r;b;b>>=1) {
		if (b&1) rep(i,1,n+1) ret[i]=f[ret[i]];
		rep(i,1,n+1) f[i]=f[f[i]];
	}
	int rp=1,rc=ret[1]-1;

		rep(j,len[rp-1],len[nxt[rp]-1]-1) putchar(s[j]);
