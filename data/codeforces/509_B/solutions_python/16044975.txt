R = lambda: map(int, raw_input().split())
def solve():
    n,k = R()
    A = R()
    for i in range(n):
        for j in range(i, n):
            if abs(A[i]-A[j]) > k:
                print "NO"
                return
    print "YES"
    for e in A:
        if e <= k:
            print " ".join(map(str, range(1,e+1)))
        else:
            r = range(1, (k+1))
            t = e/k
            print " ".join(map(str, ((r * t) + (r[0:(e%k)]))))

solve()
