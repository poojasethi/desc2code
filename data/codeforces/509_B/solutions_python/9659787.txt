

from sys import stdin
import sys
import fileinput


piles, colors = map(int, stdin.readline().rstrip().split())
stones = map(int, stdin.readline().rstrip().split())

solutions = [[] for x in xrange(piles)]
for i in xrange(1, colors + 1):
	minst = min(stones)
	maxst = max(stones)
	for j in xrange(len(stones)):
		colored = minst + 1 if stones[j] > minst else minst
		stones[j] -= colored
		solutions[j] += [i]*colored
	if sum(stones) == 0:
		print "YES"
		for k in solutions:
			print " ".join(map(str, k))
		sys.exit()
print "NO"
