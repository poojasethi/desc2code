R=lambda:map(int,raw_input().split())
n,k=R()
a=R()
ans=[[k] * x  for x  in a]
a=sorted(zip(a,range(n)))
last=k
for i in range(1,n):
	d=a[i][0]-a[i-1][0]
	now,pre=a[i][1],a[i-1][1]
	for j in range(a[i-1][0]): ans[now][j]=ans[pre][j]
	if d:
		if last:
			d-=1
			ans[now][a[i-1][0]]=last
		for j in range(d): k-=1; ans[now][a[i-1][0]+j+(last>0)]=k
		last=0
		if k<=0: print 'NO';break
else: 
	print 'YES'
	for i in range(n):
		print ' '.join(map(str, ans[i]))
