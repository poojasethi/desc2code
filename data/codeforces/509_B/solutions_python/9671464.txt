#!/usr/bin/python


import sys


def color(piles, least):
    for p in piles:
        colors = [1] * least + list(range(1, p - least + 1))
        print " ".join(map(str, colors))


def solve(piles, k):
    least = min(piles)
    most = max(piles)
    if least + k < most:
        print "NO"
    else:
        print "YES"
        color(piles, least)
    

if __name__ == "__main__":
    n, k = map(int, sys.stdin.readline().split())
    piles = map(int, sys.stdin.readline().split())
    solve(piles, k)
