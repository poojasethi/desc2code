N, K = map(int, raw_input().split())
nums = map(int, raw_input().split())

minV = min(nums)
maxV = max(nums)
if maxV-minV > K:
    print 'NO'
else:
    print 'YES'
    for i in nums:
        tmp = [1]*minV
        for i in xrange(0, i-minV):
            tmp.append(i+1)
        print ' '.join(map(str,tmp))
