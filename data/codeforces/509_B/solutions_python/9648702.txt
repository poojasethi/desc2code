#!/usr/bin/env python

import sys
import math
import random
import operator
from fractions import Fraction, gcd
from decimal import Decimal, getcontext
from itertools import product, permutations, combinations
getcontext().prec = 100

MOD = 10**9 + 7
INF = float("+inf")

n, k = map(int, raw_input().split())
arr = map(int, raw_input().split())
src = arr[::]
arr.sort()

colored_values = {}
can_increase = set(range(1, k + 1))
if k >= arr[0]:
    prev = range(1, arr[0] + 1)
    prev_val = arr[0]
    colored_values[arr[0]] = prev
else:
    prev = range(1, k) + [k] * (1 + arr[0] - k)
    prev_val = arr[0]
    colored_values[arr[0]] = prev

for c in arr[1:]:
    prev = prev[::]
    if c - prev_val > len(can_increase):
        print "NO"
        quit()
    for i in xrange(c - prev_val):
        color = can_increase.pop()
        prev.append(color)
    prev_val = c
    colored_values[c] = prev

print "YES"
for c in src:
    print " ".join(map(str, colored_values[c]))
