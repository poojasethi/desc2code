n, k = [int(x) for x in raw_input().split()]
a = [int(x) for x in raw_input().split()]

mi = min(a)
ma = max(a)

if k < (ma - mi):
    print "NO"
else:
    print "YES"
    for i in xrange(n):
        res = []
        k = 1
        for j in xrange(a[i]):
            if j>mi:
                k+=1
            res.append(k)
        print " ".join([str(x) for x in res])
