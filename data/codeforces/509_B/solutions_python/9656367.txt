n, k = map(int,raw_input().split())
peb = map(int,raw_input().split())
fail = False
if max(peb) - min(peb) > k:
	fail = True

paint = [[] for i in range(n)]
paindiff = max(0,max(peb) - k) #Num to be painted in 1
for i in range(n):
	paint[i] += [1 for j in range(paindiff)]
	paint[i] += range(1,peb[i]-paindiff+1)
	
if not fail:
	print 'YES'
	for i in range(n):
		for j in paint[i]:
			print j,
		print
else:
	print 'NO'
