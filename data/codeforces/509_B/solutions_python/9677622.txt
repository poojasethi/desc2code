from math import ceil
#br = open('b.in')
f = lambda: map(int, raw_input().strip().split())
n, k = f()
a = f()
x, y = min(a), max(a)
if ceil(1.0 * (y - x)) / k > 1:
  print "NO"
else:
  print "YES"
  for i in range(n):
    c, v = 0, []
    for j in range(a[i]):
      v.append((c % k) + 1)
      c += 1
    v.sort()
    print ' '.join(map(str, v))
