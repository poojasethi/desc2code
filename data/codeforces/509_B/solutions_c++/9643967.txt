#include<iostream>
#include<cstdio>
using namespace std;
int mn=100000,mx;
int n,k;
int a[105];
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		mx=max(mx,a[i]);
		mn=min(mn,a[i]);
	}
	if(mx-mn>k)
	{
		puts("NO");
		return 0;
	}
	puts("YES");
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=a[i];j++)
		{
			printf("%d ",(j-1)%k+1);
		}
		puts("");
	}
	return 0;
}
