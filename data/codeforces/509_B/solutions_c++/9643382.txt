#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,i,j;
	cin>>n>>k;
	int a[n],mini=1000,maxi=0;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
	{
		mini=min(mini,a[i]);
		maxi=max(maxi,a[i]);
	}
	if((maxi-mini)>k)
		cout<<"NO\n";
	else
	{
		cout<<"YES\n";
		for(i=0;i<n;i++)
		{
			for(j=0;j<a[i];j++)
			{
				if((j+1)%k==0)
				cout<<k<<" ";
				else
				cout<<(j+1)%k<<" ";
			}
			cout<<endl;
		}
	}		
	return 0;		
}
