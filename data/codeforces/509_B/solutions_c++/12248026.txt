#include <cstdio>
const int MAX_N = 110;
int a[MAX_N];
int main()
{
	int n, k, mx = 0, mi = 101;
	scanf("%d %d", &n, &k);
	for (int i = 0;i < n;i++){
		scanf("%d", a+i);
		if (a[i] > mx)	mx = a[i];
		if (a[i] < mi)	mi = a[i];
	}
	if (mx-mi > k)	printf("NO\n");
	else{
		printf("YES\n");
		for (int i = 0;i < n;i++)
			for (int j = 0;j < a[i];j++)
				printf("%d%s", j%k+1, j == a[i]-1?"\n":" ");
	}
	return 0;
}
