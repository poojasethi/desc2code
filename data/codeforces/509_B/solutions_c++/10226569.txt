#include<cstdio>
#include<algorithm>
using namespace std;
int a1[100];
int main()
{
	int n,k,a=100,b=-1;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i)
	{
		scanf("%d",a1+i);
		a=min(a,a1[i]);
		b=max(b,a1[i]);
	}
	if(b-a>k)
	{puts("NO");return 0;}
	puts("YES");
	for(int i=0;i<n;++i,puts(""))
	for(int j=0;j<a1[i];++j)
	{
		if(j)putchar(' ');
		printf("%d",j<a?1:j-a+1);
	}
}
