#include <bits/stdc++.h>
int aa[100+10];
int Min =1000,Max=-1;
using namespace std;
int main()
{
	int a,b; cin >> a >> b;
	for(int x=0;x<a;++x)  { cin >> aa[x]; if(aa[x]<Min) Min=aa[x]; if(aa[x]>Max) Max=aa[x]; }
	if(Max-Min>b) cout << "NO" << endl;
	else{ cout << "YES" << endl;
	for(int y=0;y<a;++y)
	{
	for(int x=1;x<=aa[y];++x)
	cout << ((x%b==0)?b:(x%b)) << " ";
cout << endl;
} }
}
