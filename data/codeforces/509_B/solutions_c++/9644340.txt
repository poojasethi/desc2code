#include "bits/stdc++.h"
using namespace std;
int main()
{
    int n,k; cin>>n>>k;
    int a[1000000];
    for(int i=0;i<n;i++) cin>>a[i];

    if(*max_element(a,a+n)-*min_element(a,a+n) > k)
    {   cout<<"NO"<<endl; return 0; }
    else
    {
        cout<<"YES"<<"\n";
        for(int i=0;i<n;i++){
            for(int j=0;j<a[i];j++){
                cout<<(j%k)+1<<" ";
            }
            cout<<"\n";
        }
    }
    return 0;
}
