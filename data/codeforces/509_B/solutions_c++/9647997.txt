#include<bits/stdc++.h>

using namespace std;

int arr[500];
int mn,mx,n,m,k,i,j;

int main(){
	cin >> n >> m;

	for(i=1 ; i<=n ; i++){
		cin >> arr[i];
		if(i==1)
			mx = mn = arr[i];

		mx = max(mx,arr[i]);
		mn = min(mn,arr[i]);
	}

	if(mx - mn > m){
		cout << "NO" << endl;
		return 0;
	}

	cout << "YES" << endl;

	for(i=1 ; i<=n ; i++){
		for(j=1 ; j<=arr[i] ; j++)
			cout << (j-1)%m+1 << ' ';
		cout << endl;
	}

}
