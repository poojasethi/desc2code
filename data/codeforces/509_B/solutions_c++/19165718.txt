#include<stdio.h>
int arr[100];
int main(){
	int n,k,i,j,g=0,s=100,l;
	scanf("%d%d",&n,&k);
	for(i=0;i<n;i++){
		scanf("%d",&arr[i]);
		if(g<arr[i])
			g=arr[i];
		if(s>arr[i])
			s=arr[i];
	}
	if((g-s)<=k){
		printf("YES\n");
		for(i=0;i<n;i++){
			l=1;
			for(j=0;j<arr[i];j++){
				if(l>k)
					l=1;
				printf("%d ",l);
				l++;
			}
			printf("\n");
		}
	}
	else
		printf("NO\n");
	return 0;
}
