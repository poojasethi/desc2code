#include<iostream>
 
using namespace std;

int main()
{
 int n=4,k=4,A[107]={1,2,3,4};
 cin>>n>>k;
 for( int i =0; i< n; i++)
 cin>>A[i];
 int Min= A[0],Max=A[0];

 for( int i =0; i< n; i++)
 { Min = min(A[i] , Min);
  Max =max(A[i], Max);}
 if( Max - Min <= k)
 {
 cout<<"YES"<<endl;
 for( int i =0; i< n; i++)
 {int p =0;
 for(int j =0; j< A[i]; j++)
  cout<< p++%k +1<< ' ';
 cout<<endl;}
 }
 else cout<< "NO";
 return 0;
}
