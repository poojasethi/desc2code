#include <bits/stdc++.h>
using namespace std;
int n,k,p[1005],mini=10009,maxi=-10009;
int main() {

    scanf ("%d%d",&n,&k);
    for(int i=0; i<n; i++)   
	{
		 scanf ("%d",&p[i]);
		 mini=min(mini,p[i]);
		 maxi=max(maxi,p[i]);
	}
    if (maxi-mini > k)
    {
    	printf("NO");
    	return 0;
	}
	
printf("YES\n");
for(int i=0;i<n;i++) {
	while(p[i]--)
	printf("%d ",(p[i]%k)+1);
	printf("\n");
}

	
	
    return 0;
}
