#include<iostream>
#include<math.h>
using namespace std;
int a[150];
int main(){
	int n,k;
	cin >> n >> k;
	int max=-99 , min=150;
	for(int i = 1 ; i <= n ; i++){
		cin >> a[i];
		if( a[i] > max)
			max = a[i];
		if( a[i] < min)
			min = a[i];
	}
	if( max-min >= k+1){
		cout << "NO";
		return 0;
	}
	cout<<"YES"<<endl;
	for(int i = 1 ; i <= n ;i++){
		for(int j = 0 ; j < a[i] ; j++)
			cout<<(j%k)+1<<" ";
		cout<<endl;
	}
	return 0;
}
