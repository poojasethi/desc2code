#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,i,j,a[105],b[105],m;
	//freopen("in.txt","r",stdin);
	cin>>n>>k;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		b[i]=a[i];
	}
	sort(b,b+n);
	if(b[n-1]-b[0]<=k)
	{
		cout<<"YES"<<endl;
		for(i=0;i<n;i++)
		{
			m=1;
			for(j=0;j<a[i];j++)
			{
				cout<<m<<" ";
				m++;
				if(m>k)
					m=1;
			}
			cout<<endl;
		}
	}
	else
	cout<<"NO"<<endl;
	return 0;
}
