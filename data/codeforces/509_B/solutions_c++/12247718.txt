#include<cstdio>
int main(){
	int n,k,a[105],max_n=-200,min_n=200;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i){
		scanf("%d",&a[i]);
		if(max_n<a[i]) max_n=a[i];
		if(min_n>a[i]) min_n=a[i]; 
	}
	if(max_n-min_n>k) printf("NO\n");
	else{
		printf("YES\n");
		for(int i=0;i<n;++i){
			for(int j=0;j<a[i];++j)
				printf("%d%c",j%k+1,j<a[i]-1?' ':'\n');
		}
	}
	return 0;
}
