#include <cstdio>
int ar[105];
int main() {
	int n, k, min = 1000, max = 0;
	scanf("%d %d", &n, &k);
	for(int i = 0; i < n; ++i) {
		scanf("%d", &ar[i]);
		max = max < ar[i] ? ar[i] : max;
		min = min > ar[i] ? ar[i] : min;
	}
	if(max - min > k) { puts("NO"); return 0; }
	puts("YES");
	for(int i = 0; i < n; ++i) {
		for(int j = 0, o = 2; j < ar[i]; ++j)
			if(j < min + 1) printf("1 ");
			else printf("%d ", o++);
		puts("");
	}
}
