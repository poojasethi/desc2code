#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n, k;
	cin>>n>>k;
	int D[n];
	vector<int>F;
	for (int i=0; i<n; i=i+1)
	{
		cin>>D[i];
		F.push_back(D[i]);
	}
	sort(F.begin(), F.end());
	if (F[n-1]-F[0]>k)
	{
	    cout<<"NO";
	    return 0;
	}
	cout<<"YES"<<endl;
	for (int i=0; i<n; i=i+1)
	{
		for (int j=0; j<D[i]; j++)
		{
			cout<<j%k+1<<" ";
		}
		cout<<endl;
	}
}
