#include <bits/stdc++.h>
using namespace std;
 
int n,k,a[1000000],i,j;
 
int main()
{
 cin>>n>>k;
 int a[n];
 for(i = 0; i < n; i++) cin>>a[i];
 if(*max_element(a,a+n)-*min_element(a,a+n)>k){
 cout<<"NO"<<endl;return 0;
 }
 else
 {
    cout<<"YES"<<endl;
    for(i = 0 ;i < n; i++)
    {
        for(j = 0; j < a[i]; j++) cout<<(j%k)+1<<" ";
        cout<<endl;
    }
 }
 
 
  return 0;
}
