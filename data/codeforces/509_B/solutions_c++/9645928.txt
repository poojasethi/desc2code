#include <bits/stdc++.h>
using namespace std;
int main() {
	int n,k;
	cin >> n >> k;
	int i,a[n],j,b[n];
	for (i = 0; i < n; i++) {
		cin >> a[i];
		b[i] = a[i];
	}
	sort(a,a+n);
	if (a[n-1] > a[0] + k) {
		puts("NO");
		return 0;
	}
	puts("YES");
	for (i = 0; i < n; i++) {
		int d = 0,j = 1;
		while (d < b[i]) {
			if (j > k) {
				j = 1;
			} 
			cout << j << ' ';
			d++;
			j++;
		}
		cout << endl;
	}
	return 0;
}
