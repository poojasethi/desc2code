#include <iostream>
using namespace std;
int main()
{
	int a[1001],k,n,mn=99999,mx=-99999;
	cin>> n>> k;
	for(int i=0; i<n; ++i)
	{
		cin>> a[i];
		mn=min(mn,a[i]);
		mx=max(mx,a[i]);
	}
	if(mx-mn>k)
	{
		cout<<"NO\n";
		return 0;
	}
	else cout<<"YES\n";
	for(int i=0; i<n; ++i)
	{
		for(int j=0; j<mn; ++j)
		{
			cout<<1<<" ";
		}
		for(int j=mn; j<a[i]; ++j)
		{
			cout<<j-mn+1<<" ";
		}
		cout<<endl;
	}
	return 0;
}
