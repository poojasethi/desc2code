#include<iostream>
using namespace std;
int main(){
    int n,k,a[101],i,mi=0,ma=0,j,l=1;
    cin>>n>>k>>a[0]; mi=a[0]; ma=a[0];
    for(i=1;i<n;i++){
        cin>>a[i];
        mi=min(mi,a[i]); ma=max(ma,a[i]);
    } if(ma>k+mi){cout<<"NO"; return 0;}
    cout<<"YES"<<endl;
    for(i=0;i<n;i++){
        for(j=0;j<a[i];j++){
            if(l>k) l=1;
            cout<<l<<" "; l++;
        } cout<<endl; l=1;
    }//cout<<ma<< " "<<mi;
}
