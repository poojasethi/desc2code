#include<bits/stdc++.h>
using namespace std;
int main(){

int n,k;
cin>>n>>k;
int arr[105]={0},arr2[105]={0};
for(int i=0;i<n;i++){
    scanf("%d",&arr[i]);
    arr2[i]=arr[i];
}

sort(arr2,arr2+n);

if(arr2[0]+k>=arr2[n-1]){
    cout<<"YES\n";

 for(int i=0;i<n;i++){
    for(int j=0;j<arr[i];j++){
        printf("%d ",(j)%k+1);
    }
    printf("\n");
 }
}
else {
    cout<<"NO";
}

return 0;
}
