#include<bits/stdc++.h>

using namespace std;

int main() {
	int n,k,x;
	cin>>n>>k;
	vector<int>v,v1;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		v.push_back(x);
		v1.push_back(x);
	}
	
	sort(v1.begin(),v1.end());
	
	if(v1[n-1]-v1[0]>k)
	{
		cout<<"NO"<<endl;
		exit(0);
	}
	cout<<"YES"<<endl;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<v[i];j++)
		{
			cout<<(j%k) + 1<<" ";
		}
		cout<<endl;
	}	
}
