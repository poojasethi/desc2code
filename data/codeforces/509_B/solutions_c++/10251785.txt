#include <bits/stdc++.h>

using namespace std;


int main()
{
int n,k,maximum=-1,minimum=101;
cin>>n>>k;
int tab[n];
for(int i=0;i<n;i++)
{
    cin>>tab[i];
    maximum=max(tab[i],maximum);
    minimum=min(tab[i],minimum);
}
if(maximum-minimum>k){cout<<"NO";return 0;}
cout<<"YES"<<endl;
for(int i=0;i<n;i++)
{
  for(int j=0;j<tab[i];j++)
    cout<<(j%k)+1<<" ";
  cout<<endl;

}


return 0;
}
