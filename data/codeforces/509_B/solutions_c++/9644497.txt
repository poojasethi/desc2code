#include<iostream>
#include<algorithm>

using namespace std;

int a[1000];

int main()
{
	int n, k;
	cin >> n >> k;
	for(int i = 0; i < n; i++)
	{
		cin >> a[i];
		for(int j = 0; j < i; j++)
			if(a[i] > a[j] + k || a[i] + k < a[j])
			{
				cout << "NO" << endl;
				return 0;
			}
	}
	cout << "YES" << endl;
	for(int i = 0; i < n; i++)
	{
		int j;
		for(j = 0; j < a[i]; j++)
			cout << j % k + 1 << ' ';
		cout << endl;
	}
	return 0;
}
