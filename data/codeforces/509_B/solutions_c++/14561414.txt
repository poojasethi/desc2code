#include <bits/stdc++.h>
using namespace std;


int mn=100000,mx;
int n,k;
int a[105];




int main(){
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		mx=max(mx,a[i]);
		mn=min(mn,a[i]);
	}
	if(mx-mn>k){
		cout <<"NO"<< endl ;
		return 0;
	}
	cout << "YES" << endl ;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=a[i];j++){
			printf("%d ",(j-1)%k+1);
		}
		cout << " " ; 
	}
	return 0;
}
