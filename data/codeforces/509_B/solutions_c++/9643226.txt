#include "cstdio"
int n,k,tab[105],min=10000,maks=-1,x;
int main()
{
	scanf ("%d%d", &n, &k);
	for (int i=0; i<n; i++)
	{
		scanf ("%d", &tab[i]);
		if (tab[i]>maks) maks=tab[i];
		if (tab[i]<min) min=tab[i];
	}
	
	if (maks-min>k) printf ("NO\n");
	else
	{
		printf ("YES\n");
		for (int i=0; i<n; i++)
		{
			x=1;
			for (int j=1; j<=tab[i]; j++)
			{
				printf ("%d ", x);
				x++;
				if (x>k) x=1;
			}
			printf ("\n");
		}
	}

	return 0;
}
