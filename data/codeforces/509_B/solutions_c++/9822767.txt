#include<iostream>
using namespace std;
int main()
{
	int n,k;
	int min=1000,max=-1;
	int arr[100];
	cin>>n>>k;
	int i;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
		if(arr[i]<min)
			min=arr[i];
		if(arr[i]>max)
			max=arr[i];
	}
	int j;
	if(max>min+k)
		cout<<"NO"<<endl;
	else
	{
		cout<<"YES"<<endl;
		for(i=0;i<n;i++)
		{
			for(j=0;j<min;j++)
				cout<<1<<" ";
			for(j=min;j<arr[i];j++)
				cout<<j-min+1<<" ";
			cout<<endl;
		}
	}
	return 0;
}
