#include <iostream>
using namespace std;

long long arr[101];

int main(){
	long long n, k, m, M, i, j, t;
	cin >> n >> k;
	cin >> arr[0];
	m = M = arr[0];
	
	for(i=1;i<n;i++){
		cin >> arr[i];
		if(arr[i]>M) M=arr[i];
		if(arr[i]<m) m=arr[i];
	}
	
	if(M-m>k){
		cout << "NO\n";
	}
	else{
		cout << "YES\n";
		for(i=0;i<n;i++){
			t=1;
			for(j=0;j<arr[i];j++){
				cout << t << ' ';
				t++;
				if(t==k+1) t=1;
			}
			cout << '\n';
		}
	}
	return 0;
}
