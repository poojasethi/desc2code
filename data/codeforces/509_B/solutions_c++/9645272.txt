#include<iostream>

using namespace std;
int data[101];
int main(){
	int n,k;
	cin>>n>>k;
	int mi=12000,ma=0;
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		mi=min(x,mi);
		ma=max(x,ma);
		data[i]=x;
	}
	if((ma/k-1)*k+ma%k>mi){
		cout<<"NO"<<endl;
		return 0;
	}
	
	cout<<"YES"<<endl;
	for(int i=0;i<n;i++){
		for(int j=0;j<data[i];j++){
			if(j)cout<<" ";
			cout<<j%k+1;
		}
		cout<<endl;
	}
	
	
}
