#include<bits/stdc++.h>
using namespace std;

int main()
{
int n,m;
cin>>n>>m;
vector<int>vec,v(n);
for(int i=0;i<n;i++)
    cin>>v[i];

vec=v;
sort(vec.begin(),vec.end());
if(vec[vec.size()-1]-vec[0]>m)
{cout<<"NO";return 0;}

cout<<"YES"<<endl;
for(int i=0;i<n;i++)
{
for(int j=1;j<=v[i];j++)
{
    if(j%m==0)
        cout<<m<<" ";
    else
    cout<<j%m<<" ";
}
cout<<endl;
}

}
