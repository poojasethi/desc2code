#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int main(){
	int n,k;
	cin>>n>>k;
	ll arr[n+1], brr[n+1];;
	for(int i=0; i<n; ++i){
		cin>>arr[i];
		brr[i] = arr[i];
	}
	sort(brr, brr+n);
	if((brr[n-1]-brr[0])>k){
		cout<<"NO"<<endl;
	}else{
		cout<<"YES"<<endl;
		for(int i=0; i<n; ++i){
			for(int j=0; j<arr[i]; ++j){
				if((j+1)%k == 0){
					cout<<k<<" ";
				}else{
					cout<<(j+1)%k<<" ";
				}
			}
			cout<<endl;
		}
	}
}
