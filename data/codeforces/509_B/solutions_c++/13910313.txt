#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	int n,k,ans=0,f=1;
	cin>>n>>k;
	int a[n],b[n];
	for(int i=0;i<n;i++){cin>>a[i];b[i]=a[i];}
	sort(b,b+n);
	ans=b[n-1]-b[0];
	ans<=k?cout<<"YES":cout<<"NO";
	if(ans<=k){for(int i=0;i<n;i++){
			cout<<endl;f=1;
			for(int j=1;j<=a[i];j++){
				if(j<=b[0]+1)cout<<"1 ";
				else {f++;cout<<f<<" ";}
			}
		}}
}
