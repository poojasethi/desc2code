#include<bits/stdc++.h>


using namespace std;


int main()
{


	int n ,  k ;

	cin >> n >> k ;

	int a[101];


	for(int i =0 ;i <n ; i++)
		cin >> a[i] ; 

	if( *max_element(a , a + n ) - *min_element(a,a+n) > k)
	{
		cout << "NO" << endl;
		exit(0);
	}


	cout << "YES" << endl;
	for(int i=0; i< n; i++,cout << endl)
		for(int j =0; j< a[i] ; j++){
				cout << j%k+1<<" ";
		}



	return 0;
}
