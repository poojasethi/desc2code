#include<bits/stdc++.h>

using namespace std;

int main()
{

int n,k,i,j;
cin>>n>>k;

int a[n],mi=INT_MAX,ma=INT_MIN,c=1;

for(i=0;i<n;i++)
{
cin>>a[i];

if(ma<a[i])
ma=a[i];

if(mi>a[i])
mi=a[i];
}

if(ma-mi<=k)
{
 cout<<"YES"<<endl;

 for(i=0;i<n;i++)
 {
    c=1;
  for(j=1;j<=a[i];j++)
   {
    cout<<c<<" ";
    c++;
    if(c>k) c=1;
   }
  cout<<endl;
 }
}

else
cout<<"NO"<<endl;

return 0;
}
