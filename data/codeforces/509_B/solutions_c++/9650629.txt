#include<bits/stdc++.h>
using namespace std;
int main()
{
    
int n,k;
cin>>n>>k;
int a[n+1];
int max1=0, min1=101;
for(int i=1;i<=n;i++){cin>>a[i];min1=min(min1,a[i]);}
bool res=true;
for(int i=1;i<=n;i++)
{
    if(a[i]>min1+k)res=false;

}
if(res)
{
cout<<"YES\n";

for(int j=1;j<=n;j++)
{
    for(int i=1;i<=min1;i++)cout<<"1 ";
    for(int i=1;i<=a[j]-min1;i++)cout<<i<<" ";
    cout<<"\n";
}
}
else cout<<"NO";
}
