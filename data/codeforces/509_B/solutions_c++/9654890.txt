#include <iostream>

using namespace std;

int main()
{
    int n,k,i[100],a,ma=-1,mi=1000,delta,q;
    cin>>n>>k;
for (a=0;a<n;a++)
{
cin>>i[a];
if (i[a]>ma) ma=i[a];
if (i[a]<mi) mi=i[a];
}
if (mi+k<ma) cout<<"NO";
else
{
cout<<"YES"<<endl;
for (a=0;a<n;a++)
{
delta=0;
while (delta!=mi) {cout<<1<<' ';delta++;};
i[a]-=mi;
q=1;
while (i[a]!=0) {cout<<q<<' '; q++; i[a]--;}
cout<<endl;
}
}
return 0;
}
