#include<bits/stdc++.h>
using namespace std;
int a[105];
int main(){
int n,c,diff,max1=-1,min1=1e9;
int x=1;
cin>>n>>c;
for(int i=0;i<n;i++)
cin>>a[i];
for(int i=0;i<n;i++)
{
if(a[i]>max1)
max1=a[i];
if(a[i]<min1)
min1=a[i];

}
if(max1-min1>c)
{
cout<<"NO";
return 0;
}

else
{
cout<<"YES\n";
for(int i=0;i<n;i++)
{
x=1;
for(int j=0;j<a[i];j++)
{
cout<<x<<" ";
x++;
if(x==c+1)
x=1;
}
cout<<"\n";
}
}
return 0;
}
