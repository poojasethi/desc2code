#include<stdio.h>
#include<algorithm>
using namespace std;
int a[111];
int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	int mx = 0,mn = 111;
	for(int i = 0;i<n;i++){
		scanf("%d",&a[i]);
		mx = max(mx,a[i]);
		mn = min(mn,a[i]);
	}
	if(mx-mn > k){
		puts("NO");
	}else{
		puts("YES");
		for(int i = 0;i<n;i++){
			for(int j = 1;j<=a[i];j++){
				printf("%d ",(j-1)%k+1);
			}
			printf("\n");
		}
	}
	return 0;
}
