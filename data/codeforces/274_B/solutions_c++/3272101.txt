#include<stdio.h>
#include<string.h>
#include<list>
#include<algorithm>
#include<vector>
#include<math.h>
#define N 100010
using namespace std;
vector<int> e[N];
long long a[N],inc[N],dec[N];
void dfs(int u,int pre)
{
	int i,n=e[u].size();
	for(i=0;i<n;i++){
		int v=e[u][i];
		if(v==pre) continue;
		dfs(v,u);
		inc[u]=max(inc[u],inc[v]);
		dec[u]=max(dec[u],dec[v]);
	}
	a[u]+=inc[u]-dec[u];
	if(a[u]<0) inc[u]-=a[u];
	else dec[u]+=a[u];
}
int main()
{
	int n,x,y;
	scanf("%d",&n);
	for(int i=1;i<n;i++){
		scanf("%d %d",&x,&y);
		e[x].push_back(y);
		e[y].push_back(x);
	}
	for(int i=1;i<=n;i++)
		scanf("%I64d",&a[i]);
    dfs(1,0);
	printf("%I64d\n",inc[1]+dec[1]);
	return 0;
}
