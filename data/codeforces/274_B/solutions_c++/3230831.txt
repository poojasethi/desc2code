#include<iostream>
#include<vector>
#include<cstring>
using namespace std;
#define rep(i,n) for(int (i)=0;(i)<(n);(i)++)
typedef long long ll;
vector<int> ed[110000];
long long a[100100],b[100100],v[100100];
void dfs(int i,int p){
    for(int j=0;j<ed[i].size();j++){
        int k=ed[i][j];
        if(k==p) continue;
        dfs(k,i);
        a[i]=max(a[i],a[k]);
        b[i]=max(b[i],b[k]);
    }
    v[i]+=a[i];v[i]-=b[i];
    if(v[i]<0) a[i]+=-v[i];
    else b[i]+=v[i];
}
int main(){
    cin.sync_with_stdio(false);
    int n,x,y;
    cin>>n;
    rep(i,n-1){
        cin>>x>>y;
        ed[x].push_back(y);ed[y].push_back(x);
    }
    rep(i,n)cin>>v[i+1];
    dfs(1,0);
    cout<<a[1]+b[1];
    return 0;
}
