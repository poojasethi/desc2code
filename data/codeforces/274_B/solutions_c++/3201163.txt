#include<cstdio>
#include<cstring>
#include<cmath>
#include<vector>
#include<algorithm>
#define INF 0x3f3f3f3f
using namespace std;
const int MAXN=100000;
vector<int>map[MAXN];
long long w[MAXN];
long long d[MAXN];

void dfs(int p,int u)
{
	long long tmp1=0LL,tmp2=0LL;
	for(int i=0;i<map[u].size();i++)
	{
		int v=map[u][i];
		if(v==p)continue;
		dfs(u,v);
		tmp1=max(d[v],tmp1);
		tmp2=max(d[v]-w[v],tmp2);
	}
	d[u]=tmp1+max(w[u]-tmp1+tmp2,0LL);
}

int main()
{
int n;
scanf("%d",&n);
for(int i=1;i<n;i++)
{
	int u,v;
	scanf("%d%d",&u,&v);
	u--;
	v--;
	map[u].push_back(v);
	map[v].push_back(u);
}
for(int i=0;i<n;i++)
{
	scanf("%I64d",&w[i]);
}
dfs(-1,0);
printf("%I64d\n",2*d[0]-w[0]);
return 0;
}
