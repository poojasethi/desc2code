n = input()
a = map(int, raw_input().split())
A = []
A.append([a[1]-a[0],a[n-1]-a[0]])
for i in range(1,n-1):
    S = a[i] - a[i-1]
    T = a[i+1] - a[i]
    A.append([min(S,T), max(a[i]-a[0],a[n-1] - a[i])])
A.append([a[n-1] - a[n-2],a[n-1] - a[0]])
for j in range(0,n):
    print A[j][0],A[j][1]
