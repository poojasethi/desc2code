raw_input()
x = map(int, raw_input().split(' '))
a = len(x)

for i in xrange(a):
	if i == 0:
		print abs(x[i] - x[i+1]),  abs(x[i] - x[a - 1])
	elif i == a - 1:
		print abs(x[i] - x[i - 1]), abs(x[i] - x[0])
	else:
		print min(abs(x[i] - x[i+1]), abs(x[i] - x[i - 1])), max(abs(x[i] - x[0]), abs(x[i] - x[a - 1]))
