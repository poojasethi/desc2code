n = int(raw_input())
_ = raw_input().split()
a = [int(i) for i in _]

for idx, i in enumerate(a):
    if idx == 0:
        print a[idx+1] - i, a[len(a) - 1] - i

    elif idx == len(a)-1:
        print i - a[idx-1], i - a[0]

    else:
        print min(i - a[idx-1], a[idx+1] - i), max(i - a[0], a[len(a) - 1] - i)
