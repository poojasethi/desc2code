n = input()
cities = map(int,raw_input().split())

print abs(cities[0]-cities[1]),abs(cities[0]-cities[-1])
for i in xrange(1,n-1):
        print min(abs(cities[i]-cities[i-1]),abs(cities[i]-cities[i+1])),max(abs(cities[i]-cities[0]),abs(cities[i]-cities[-1]))
print abs(cities[-1]-cities[-2]),abs(cities[0]-cities[-1])
