#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	long long a[n+2];
	for(int i=1;i<=n;i++)cin>>a[i];
	a[0] = -10000000000;
	a[n+1] =10000000000;
	int mini, maxi;
	for(int i=1;i<=n;i++)
	{
		mini = min(a[i]-a[i-1], a[i+1]-a[i]);
		maxi = max(a[i]-a[1], a[n]-a[i]);
		cout<<mini<<" "<<maxi<<endl;
	}
	
}
