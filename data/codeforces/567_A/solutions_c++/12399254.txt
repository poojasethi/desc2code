#include<iostream>
using namespace std;
main()
{
    int n;
    cin>>n;
    int a[n+1];
    for(int i=1; i<=n; i++)
        cin>>a[i];
    cout<<a[2]-a[1]<<" "<<a[n]-a[1]<<endl;
    for(int i=2; i<n; i++)
        cout<<min(a[i]-a[i-1],a[i+1]-a[i])<<" "<<max(a[i]-a[1],a[n]-a[i])<<endl;
    cout<<a[n]-a[n-1]<<" "<<a[n]-a[1];

}
