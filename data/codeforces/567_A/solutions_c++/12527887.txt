#include <iostream>
#include <algorithm>
using namespace std;
long long n,a[100005];
int main()
{
     cin>>n;
     for (int i=1;i<=n;++i) cin>>a[i];
     a[0]=-1000000000000000LL;a[n+1]=1000000000000000LL;
     for (int i=1;i<=n;++i)
     {
          cout<<min(a[i]-a[i-1],a[i+1]-a[i])<<" "<<max(a[i]-a[1],a[n]-a[i])<<endl;
     }
     return 0;
}
