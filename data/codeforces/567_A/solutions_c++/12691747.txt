#include<iostream>
#include<fstream>
using namespace std;
long long a[10000011];
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	a[0]=-9999999999999;
	a[n+1]=9999999999999;
	for(int i=1;i<=n;i++){
		cout<<min(a[i]-a[i-1],a[i+1]-a[i])<<" "<<max(a[n]-a[i],a[i]-a[1])<<endl;
	}
	return 0;
}
