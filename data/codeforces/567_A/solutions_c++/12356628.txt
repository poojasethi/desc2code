#include <iostream>

using namespace std;
const int maxN=1e5+5;
int main()
{
   int n,x[maxN];
   cin>>n;
   for(int i=1;i<=n;i++) cin>>x[i];
   cout<<x[2]-x[1]<<" "<<x[n]-x[1]<<"\n";
   for(int i=2;i<n;i++)
        cout<<min(x[i]-x[i-1],x[i+1]-x[i])<<" "<<max(x[i]-x[1],x[n]-x[i])<<"\n";
   cout<<x[n]-x[n-1]<<" "<<x[n]-x[1]<<"\n";
}
