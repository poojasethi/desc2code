#include <bits/stdc++.h>
using namespace std;
int n, x[100005];
int main() {
	scanf("%d", &n);
	for(int i=0; i<n; ++i) {
		scanf("%d", &x[i]);
	}
	printf("%d %d\n", x[1]-x[0], x[n-1]-x[0]);
	for(int i=1; i<n-1; ++i) {
		printf("%d %d\n", min(x[i]-x[i-1], x[i+1]-x[i]), max(x[i]-x[0], x[n-1]-x[i]));
	}
	printf("%d %d\n", x[n-1]-x[n-2], x[n-1]-x[0]);
	return 0;
}
