#include<cstdio>
#include<algorithm>
using namespace std;
#define N   1000005

int a[N],n,i;

int main(){
    scanf("%d",&n);
    for (i=1;i<=n;i++) scanf("%d",&a[i]);
    printf("%d %d\n",a[2]-a[1],a[n]-a[1]);
    for (i=2;i<n;i++) printf("%d %d\n",min(a[i]-a[i-1],a[i+1]-a[i]),max(a[n]-a[i],a[i]-a[1]));
    printf("%d %d\n",a[n]-a[n-1],a[n]-a[1]);
}
