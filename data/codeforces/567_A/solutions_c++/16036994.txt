#include<iostream>
using namespace std;


int main(){
    long long int i,n;
	long long int d[100005];
	cin>>n;
	for(i=0;i<n;i++){
	   cin>>d[i];
	}
	cout<<d[1]-d[0]<<" "<<d[n-1]-d[0]<<endl;
	for(i=1;i<n-1;i++){
	   cout<<min(d[i]-d[i-1],d[i+1]-d[i])<<" "<<max(d[i]-d[0],d[n-1]-d[i])<<endl;
	}
	cout<<d[n-1]-d[n-2]<<" "<<d[n-1]-d[0];
}
