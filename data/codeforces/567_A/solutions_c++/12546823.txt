#include<bits/stdc++.h>
using namespace std;
int n,i,h[100005];
int main()
{
	for(cin>>n;i<n;i++)
		cin>>h[i];
	for(i=0;i<n;i++)
	{
		if(i&&(i==n-1 || h[i]-h[i-1]<h[i+1]-h[i]))cout<< h[i]-h[i-1];
		else cout << h[i+1]-h[i];
		cout << ' ';
		if(h[i]-h[0]>h[n-1]-h[i])cout<<h[i]-h[0];
		else cout<<h[n-1]-h[i];
		cout<<endl;
	}
}
