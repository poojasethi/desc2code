#include<iostream>
using namespace std;
int main()
 {
int i,n;
long x[200000];
cin>>n;
for(i=0;i<n;i++)
    {cin>>x[i];}
cout<<x[1]-x[0]<<" "<<x[n-1]-x[0]<<endl;
for(i=1;i<n-1;i++)
{
cout<<min(x[i]-x[i-1],x[i+1]-x[i])<<" "<<max(x[i]-x[0],x[n-1]-x[i])<<endl;
}
cout<<x[n-1]-x[n-2]<<" "<<x[n-1]-x[0];
return 0;
}
