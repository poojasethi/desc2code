#include <iostream>

using namespace std;

long long int a[100005];

int main(){
    int n,i;
    cin>>n;
    for(i=1;i<=n;i++)cin>>a[i];
    cout<<a[2]-a[1]<<" "<<a[n]-a[1]<<endl;
    for(i=2;i<n;i++)cout<<min(a[i+1]-a[i],a[i]-a[i-1])<<" "<<max(a[n]-a[i],a[i]-a[1])<<endl;
    cout<<a[n]-a[n-1]<<" "<<a[n]-a[1];
}
