#include <bits/stdc++.h>
using namespace std;

long long a[100015], n;
int main()
{
    cin >> n;
    for(int i=1;i<=n;++i) cin>>a[i];
    a[0]=-1e18;a[n+1]=1e18;
    for(int i = 1; i <= n; i++)
        cout << min(a[i]-a[i-1], a[i+1]-a[i]) << " " << max(a[i]-a[1], a[n]-a[i]) << endl;
    return 0;
}
