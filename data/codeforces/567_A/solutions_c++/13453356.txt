#include<bits/stdc++.h>
using namespace std;
long long n,a[100010],i;
int main()
 {
    cin>>n;
    for(i=1;i<=n;i++) cin>>a[i];
    a[0] =   -1000000000000;
    a[n+1] =  1000000000000;
    sort(a,a+n);
    for(i=1;i<=n;i++) cout<<min(a[i]-a[i-1],a[i+1]-a[i])<<" "<<max(a[i]-a[1],a[n]-a[i])<<"\n";
    return 0;
 }
