#include<iostream>
#include<algorithm>
using namespace std;
long long n, a[100002], i, t;
main(){
  cin>>n;
  for(i=0;i<n;i++)cin>>a[i];
  sort(a,a+n);
  cout<<a[1]-a[0]<<" "<<a[n-1]-a[0]<<endl;
  for(i=1;i<n-1;i++){
    cout<<min(a[i+1]-a[i],a[i]-a[i-1])<<" "<<max(a[i]-a[0],a[n-1]-a[i])<<endl;
  }
  cout<<a[n-1]-a[n-2]<<" "<<a[n-1]-a[0];
}
