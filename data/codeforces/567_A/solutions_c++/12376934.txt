#include<bits/stdc++.h>
using namespace std;
int mi , ma;
int main()
{
	int n;
	cin>>n;
	long long a[n];
	for(int i=0;i<n;i++)cin>>a[i];
	cout<<a[1]-a[0]<<" "<<a[n-1] - a[0]<<endl;

	for(int i=1;i<n;i++)
	{
		mi = min( a[i]-a[i-1], a[i+1]-a[i] );
		ma = max( a[i]-a[0], a[n-1]-a[i] );
		cout<<mi<<" "<<ma<<endl;
	}

}
