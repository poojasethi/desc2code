#include <bits/stdc++.h>
#define int long long

using namespace std;

const int maxn = 1e5 + 10;
int x[maxn];

main() {
	int n;
	cin >> n;
	for(int i = 1;i <= n;i ++)
		cin >> x[i];
	x[0] = -999999999999ll;
	x[n + 1] = -x[0];
	for(int i = 1;i <= n;i ++)
		cout << min(x[i] - x[i - 1], x[i + 1] - x[i]) << " " << max(x[i] - x[1], x[n] - x[i]) << endl;
	return 0;
}
