#include <bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	int a[n + 2];
	for (int i = 1; i <= n; i++) cin>>a[i];
	for (int i = 1; i <= n; i++) {
		int maxi = max(a[n] - a[i], a[i] - a[1]);
		int mini = 2e9 + 10;
		if(i > 1) mini = min(mini, a[i] - a[i - 1]);
		if(i < n) mini = min(mini, a[i + 1] - a[i]);
		cout<<mini<<" "<<maxi<<endl;
	} 
}
