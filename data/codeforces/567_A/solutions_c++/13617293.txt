#include<iostream>
#include<algorithm>
using namespace std;
int n,b[100040];
int main(){
  cin>>n;
  for(int i=0;i<n;i++) cin>>b[i];
  cout<<b[1]-b[0]<<' '<<b[n-1]-b[0]<<endl;
  for(int i=1;i<=n-2;i++) cout<<min(b[i+1]-b[i],b[i]-b[i-1])<<' '<<max(b[i]-b[0],b[n-1]-b[i])<<endl;
  cout<<b[n-1]-b[n-2]<<' '<<b[n-1]-b[0]<<endl;
  return 0;
}
