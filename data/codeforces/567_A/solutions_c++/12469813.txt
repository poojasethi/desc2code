#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin>>n;
    vector<int> v(n);
    for(int i=0;i<n;++i)
        cin>>v[i];
    for(int i=0;i<n;++i)
    {
        cout<<min((i>0)?abs(v[i]-v[i-1]):INT_MAX,(i<n-1)?abs(v[i+1]-v[i]):INT_MAX)<<" ";
        cout<<max(abs(v[i]-v[0]),abs(v[n-1]-v[i]))<<endl;
    }
    return 0;
}
