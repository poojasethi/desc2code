n, m, k = map(int, raw_input().split())

v = [x-1 for x in list(map(int, raw_input().split()))]
b = [x-1 for x in list(map(int, raw_input().split()))]


pos = [0 for x in v]
for i in range(0, len(v)):
    pos[v[i]] = i

sol = 0
for x in b:
    sol += 1 + pos[x]//k

    i = pos[x]
    if i == 0: continue

    v[i], v[i-1] = v[i-1], v[i]
    pos[v[i]] = i
    pos[v[i-1]] = i-1

print sol
