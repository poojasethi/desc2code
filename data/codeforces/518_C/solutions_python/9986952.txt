from sys import stdin
def main():
    n, m, k = map(int, stdin.readline().split())
    a = map(int, stdin.readline().split())
    b = map(int, stdin.readline().split())
    v = [0] * (n + 1)
    for i, x in enumerate(a):
        v[x] = i
    ans = 0
    for x in b:
        i = v[x]
        ans += i / k + 1
        if i:
            a[i - 1], a[i] = a[i], a[i - 1]
            v[a[i - 1]] = i - 1
            v[a[i]] = i
    print ans
main()
