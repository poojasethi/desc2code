n,m,k=map(int ,raw_input().split())

IconAtPos=map(int ,raw_input().split())
IconAtPos.insert(0,-1)
Querry=map(int ,raw_input().split())
l=len(IconAtPos)
PosOfIcon=[-1]
for i in range(1,l):
    PosOfIcon.append([])
for i in range(1,l):
    PosOfIcon[IconAtPos[i]]=i
    
ans=0   
for i in range(0,m):
    Icon=Querry[i]
    ans+=int((PosOfIcon[Icon]-1)/k)+1
    if PosOfIcon[Icon]>1:
        Previous=PosOfIcon[Icon]-1
        temp=PosOfIcon[Icon]
        PosOfIcon[Icon]=PosOfIcon[IconAtPos[Previous]]
        PosOfIcon[IconAtPos[Previous]]=temp
        
        temp=IconAtPos[Previous]
        IconAtPos[Previous]=IconAtPos[Previous+1]
        IconAtPos[Previous+1]=temp
print ans
