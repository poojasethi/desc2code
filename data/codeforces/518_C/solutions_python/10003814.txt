n,m,k=raw_input().split()
n=int(n)
m=int(m)
k=int(k)
a=[]
b=[]
index={}
def swap(a,b):
    return b,a
t=raw_input().split()
for i in range(n):
    a.append(int(t[i]))
    index[a[i]]=i
t=raw_input().split()
for i in range(m):
    b.append(int(t[i]))
    
gestures=0

for i in b:
    gestures+=(index[i]/k)+1
    if index[i]>0:
        a[index[i]],a[index[i]-1]=swap(a[index[i]],a[index[i]-1])
        index[i]=index[i]-1
        index[a[index[i]+1]]=index[i]+1
        
print gestures
