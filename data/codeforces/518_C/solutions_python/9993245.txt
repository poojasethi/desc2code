import math

n, m, k = map(int, raw_input().split())
a = list(map(int, raw_input().split()))
b = list(map(int, raw_input().split()))

por = {}
inv_por = {}

for i in range(1, len(a)+1):
    por[a[i-1]] = i
    inv_por[i] = a[i-1] 
    
s = 0

for j in b:
    s += int(math.ceil((float(por[j]) / k)))
    if por[j] != 1:
        jj = por[j]
        por[j] -= 1
        por[inv_por[jj-1]] +=1
        inv_por[jj] = inv_por[jj-1]
        inv_por[jj-1] = j

print s
