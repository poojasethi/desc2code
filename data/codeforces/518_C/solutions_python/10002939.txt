(n,m,k)=map(int,raw_input().split())
a=map(int,raw_input().split())
ind=[0 for i in range(10**5+1)]
b=map(int,raw_input().split())
j=1
for i in a:
    ind[i]=j
    j+=1
t=0
for i in range(m):
    n=ind[b[i]]
    if n%k==0:
        t+=n/k
    else:
        t+=1+n/k
    if n-1:
        a[n-1],a[n-2]=a[n-2],a[n-1]
        ind[a[n-2]],ind[a[n-1]]=ind[a[n-1]],ind[a[n-2]]
print t
