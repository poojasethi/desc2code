n,m,k = map(int, raw_input().split())
l = map(int, raw_input().split())
lm = map(int, raw_input().split())

p = [0]*(n)

for i in xrange(n):
	p[l[i]-1] = i
	
	
c = len(lm)
for elem in lm:
	c += p[elem-1]/k
	i = p[elem-1]
	if i != 0:
		l[i], l[i-1] = l[i-1], l[i]
		v = l[i]
		p[v-1] += 1
		p[elem-1] -= 1
		
print c
