#!/usr/bin/env python
# coding=utf-8

n, m, k = map(int, raw_input().split())
a = map(int, raw_input().split())
p = [0] * (n + 1)
for i in range(n):
    p[a[i]] = i
t = 0
for x in map(int, raw_input().split()):
    i = p[x]
    j = max(0, i - 1)
    y = a[j]
    a[i], a[j], p[x], p[y] = a[j], a[i], p[y], p[x]
    t += i / k + 1
print t
