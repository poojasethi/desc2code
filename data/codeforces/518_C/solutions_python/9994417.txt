n,m,k = map( int, raw_input().split() )
location = map( int, raw_input().split() )
plan = map( int, raw_input().split() )
d = dict()
g = 0
for i in range(n):
    d[location[i]] = i

for i in range(len(plan)):
    move = plan[i]
    g += (d[move]/k + 1)
    if( d[move] != 0 ):
        f1 = d[move]
        d[move] -= 1
        d[location[f1-1]] += 1
        location[f1], location[f1-1] = location[f1-1], location[f1]
print g
