import sys

appsPerScreen = [int(n) for n in sys.stdin.readline().strip().split()][2]
allApps = [int(n) for n in sys.stdin.readline().strip().split()]
launchApps = [int(n) for n in sys.stdin.readline().strip().split()]

appToIdx = {}
for n, app in enumerate(allApps):
	appToIdx[app] = n

t = 0
for currentApp in launchApps:
	currentAppIdx = appToIdx[currentApp]
	currentAppScreen = currentAppIdx / appsPerScreen
	currentAppPosition = currentAppIdx % appsPerScreen
	t += currentAppScreen + 1
	if currentAppScreen == 0 and currentAppPosition == 0:
		continue
	prevAppIdx = currentAppIdx - 1
	prevApp = allApps[prevAppIdx]
	allApps[prevAppIdx] = currentApp
	allApps[currentAppIdx] = prevApp
	appToIdx[prevApp] = currentAppIdx
	appToIdx[currentApp] = prevAppIdx

print t
