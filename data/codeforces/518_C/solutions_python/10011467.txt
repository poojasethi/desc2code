n,m,k=map(int,raw_input().split())
p=map(lambda x: int(x)-1,raw_input().split())
q=map(lambda x: int(x)-1,raw_input().split())
op=[-1]*n
for i in xrange(n):
    op[p[i]]=i
c=0
for cur in q:
    ci=op[cur]
    c+=ci/k+1
    if ci>0:
        pv=p[ci-1]
        p[ci]=pv
        p[ci-1]=cur
        op[cur]=ci-1
        op[pv]=ci
print c
