from sys import stdin

n, m, k = map(int, stdin.readline().split())
a = map(int, stdin.readline().split())
b = map(int, stdin.readline().split())

pos = [0]*(n+1)#position of numbers

for i in xrange(n):
    pos[a[i]] = i
count = 0
for item in b:
    count += pos[item]/k+1
    if pos[item]>0:
        prenum = a[pos[item]-1]
        a[pos[item]], a[pos[item]-1], pos[prenum], pos[item] = \
        a[pos[item]-1], item, pos[item], pos[item]-1
print count
