n, m, k = map(int, raw_input().split())

a = map(int, raw_input().split())
pos = {}
# telkens berekenen van index is telkens O(n)
#opzoeken is O(1)

for i in range(n):
    num = a[i]
    pos[num] = i

ans = 0
for app in map(int, raw_input().split()):
    ans += (pos[app] / k) + 1

    if pos[app] > 0:
        a[pos[app]], a[pos[app] - 1] = a[pos[app] - 1], a[pos[app]]
        pos[app] -= 1
        pos[a[pos[app] + 1]] += 1

print ans
