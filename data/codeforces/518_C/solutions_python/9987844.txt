def readlist(f): return map(f, raw_input().split())
def readint(): return int(raw_input())
def printlist(l): print ' '.join(map(str, l))

n, m, k = readlist(int)
A = readlist(lambda s: int(s) - 1)
B = readlist(lambda s: int(s) - 1)

revA = [-1]*n
for i in range(n):
  revA[A[i]] = i

res = 0

for b in B:
  if revA[b] == 0:
    res += 1
    continue
  res += (revA[b] / k) + 1
  prev = A[revA[b] - 1]
  revA[b] -= 1
  revA[prev] += 1
  A[revA[b]] = b
  A[revA[prev]] = prev

print res
