def main():
    n, m, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    b = map(int, raw_input().split())
    d = {v:i + 1 for i, v in enumerate(a)}
    pos = {i + 1:v for i, v in enumerate(a)}
    result = 0
    for i in b:
        result += d[i] / k
        if d[i] % k != 0:
            result += 1
        if d[i] > 1:
            pos[d[i]], pos[d[i] - 1] = pos[d[i] - 1], pos[d[i]]
            d[pos[d[i]]] += 1
            d[i] -= 1
    print result
main()
