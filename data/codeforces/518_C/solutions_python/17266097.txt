arr = [int(i) for i in raw_input().split()]
n = arr[0]
m = arr[1]
k = arr[2]
a = [int(i) for i in raw_input().split()]
dic1 = {}
dic2 = {}
for i in range(n):
 dic1[a[i]] = i
 dic2[i] = a[i]

l = [int(i) for i in raw_input().split()]
result = 0
for j in l:
 if dic1[j] != 0:
  result += (dic1[j]//k + 1)
  t = dic1[j] - 1
  t1 = dic1[j]
  t2 = dic2[t1]
  t3 = dic2[t]
  dic2[t] = t2
  dic1[t2] = t
  dic2[t1] = t3 
  dic1[t3] = t1
 else:
  result += 1

print (result)
