n, m, k = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
pos = [0] * (n + 1)
for i in xrange(n): pos[a[i]] = i
ans = 0
for i in xrange(m):
	p = b[i]
	ans += pos[p] / k + 1
	if pos[p] != 0:
		pre = a[pos[p] - 1]
		pos[pre] = pos[p]
		pos[p] -= 1
		a[pos[p]] = p
		a[pos[p] + 1] = pre
print ans
