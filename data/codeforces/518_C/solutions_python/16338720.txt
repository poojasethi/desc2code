m, n, k = map(int, raw_input().split())
apl = map(int, raw_input().split())
ld = map(int, raw_input().split())
dc = {}
cd = {}
for i in xrange(m):
    dc[apl[i]] = i
    cd[i] = apl[i]
total = 0
nld = ld
for l in ld:
    try:
        v = dc[l]
        if v !=0:
            dc[l] = v-1
            dc[cd[v-1]] = dc[cd[v-1]] + 1
            tmp = cd[v-1]
            cd[v-1] = l
            cd[v] = tmp
        total += int((v/k))
    except ValueError:
        pass
print(total + n)
