n, m, k = map(int, raw_input().split())
pointer = [0] * (n + 1)
buf = map(int, raw_input().split())
for i, v in enumerate(buf):
	pointer[v] = i

seq = map(int, raw_input().split())
total = 0
for s in seq:
	p = pointer[s]
	total += p / k + 1
	if p > 0:
		pointer[s] -= 1
		pointer[buf[p - 1]] += 1

		buf[p], buf[p - 1] = buf[p - 1], buf[p]

print total
