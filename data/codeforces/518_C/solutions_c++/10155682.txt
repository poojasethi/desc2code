#include <bits/stdc++.h>
using namespace std;

int n, m, k, x, p, a[100005], b[100005];
long long ans;
int main() {
	cin >> n >> m >> k;
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
		b[a[i]] = i;
	}
	while (m--) {
		scanf("%d", &x);
		p = b[x];
		ans += p / k + 1LL;
		if (p) {
			swap(a[p], a[p - 1]);
			b[x]--;
			b[a[p]]++;
		}
	}
	cout << ans;
}
