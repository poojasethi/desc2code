#include<bits/stdc++.h>
using namespace std;
int main()
{
long long int n,m,k,ans=0,x;
scanf("%I64d %I64d %I64d",&n,&m,&k);
long long int a[n+1],pos[n+1];
for(int i=1;i<=n;i++)
{
scanf("%I64d",&a[i]);
pos[a[i]]=i;
}
while(m--)
{
scanf("%I64d",&x);
ans=ans+(pos[x]-1)/k+1;
if(pos[x]>1)
{
swap(a[pos[x]],a[pos[x]-1]);
swap(pos[x],pos[a[pos[x]]]);
}
}
printf("%I64d\n",ans);
}
