#include<iostream>
using namespace std;
int a[100005],b[100005];
int main(){
	int n,m,k;long long tot=0;
	cin>>n>>m>>k;
	for(int i=1;i<=n;i++){
		int x;cin>>x;
		a[x]=i;
		b[i]=x;
	}
	for(int i=1;i<=m;i++)
	{
		
		int x;cin>>x;
		tot=tot+(a[x]-1)/k+1;
		if(a[x]!=1){
			a[x]--;
			a[b[a[x]]]++;
			int t=b[a[x]];
			b[a[x]]=b[a[x]+1];
			b[a[x]+1]=t;
		}
	}
	cout<<tot;
    return 0;
}
