#include <stdio.h>
#include <vector>
using namespace std;

int main() {

	int N, M, K;
	long long ans = 0;
	scanf("%d %d %d\n", &N, &M, &K);

	vector<int> A(N), B(N+1);

	for (int i = 0; i < N; i++) {
		scanf("%d", &A[i]);
		B[A[i]] = i;
	}

	for (int k, i = 0; i < M; i++) {
		scanf("%d", &k);
		ans += B[k] / K + 1;
		if (B[k] > 0) {
			A[B[k]] = A[B[k]-1];
			A[B[k]-1] = k;
			B[A[B[k]]]++;
			B[k]--;
		}
	}

	printf("%I64d\n", ans);
}
