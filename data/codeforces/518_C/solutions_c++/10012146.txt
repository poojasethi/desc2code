#include <bits/stdc++.h>  
using namespace std;   

const int maxn=100010;

int a[maxn];
int _a[maxn];

int main(){
	int n,m,k;
	cin>>n>>m>>k;
	
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		_a[a[i]]=i;
	}
	
	long long ans=0;
	for(int i=1;i<=m;i++){
		int b;
		scanf("%d",&b);
		int t=_a[b];
		ans+=((_a[b]-1)/k+1);
		if(t>1){
			swap(_a[b],_a[a[t-1]]);
			swap(a[t],a[t-1]);
		}
	}
	cout<<ans<<endl;
	return 0;
}
