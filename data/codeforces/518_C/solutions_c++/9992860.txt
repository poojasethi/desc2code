#include<iostream>
#define ll long long
ll a[100005],f[100005],inp[100005];
ll ans=0;
using namespace std;
int main()
{
	ll i,j,k,l,m,n;
	cin>>n>>m>>k;
	for(i=1;i<=n;i++)
	{
		cin>>j;
		a[i]=j;
		f[j]=i;
	}
	for(i=0;i<m;i++)
	{
		cin>>j;
		ans+=(((f[j]-1)/k));
		ll x=f[j];
		if(x!=1)
		{
			f[a[x-1]]=x;
			a[x]=a[x-1];
			a[x-1]=j;
			f[j]=x-1;
		}
	}
	cout<<ans+m;
}
