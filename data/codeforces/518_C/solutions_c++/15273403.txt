#include <bits/stdc++.h>
using namespace std;
const int N = 1e5+3;
map<int,int> hash,ord;
int main(){
	int next,cur,n,m,k,i,j,a;cin>>n>>m>>k;
	for(i=1;i<=n;++i)
		cin>>a,hash[a]= i,ord[i]= a;
	long long ret=0;
	for(i=0;i<m;++i){
		cin>>a;
		cur = hash[a];//order
		ret += (cur-1)/k+1;
		if(cur==1)continue;
		next= ord[cur-1];
		ord[cur-1]= a;ord[cur]= next;
		hash[a]= cur-1;
		hash[next]= cur;
	}
	cout<<ret<<endl;
}
