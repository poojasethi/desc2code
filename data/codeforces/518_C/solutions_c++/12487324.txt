#include <stdio.h>

int g[200001],w[200001];
int i,m,n,k,t,tt;
long long s;

int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for (i=1;i<=n;i++)
	{
	    scanf("%d",&g[i]);
	    w[g[i]]=i;
	}
	for (i=1;i<=m;i++)
	{
		scanf("%d",&t);
		s=s+w[t]/k;
		if (w[t]%k!=0)
		    s++;
		if (w[t]!=1)
		{
			tt=g[w[t]-1];
			g[w[t]]=tt;
			g[w[t]-1]=t;
			w[t]--;
			w[tt]++;
		}
	}
	printf("%I64d",s);
	return 0;
}
