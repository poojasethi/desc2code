#include<bits/stdc++.h>
using namespace std;

const int maxn=100000+10;
int n,m,k,a[maxn],b[maxn];

int main()
{
	ios_base::sync_with_stdio(false);
	cin>>n>>m>>k;
	long long ans=m,t;
	for(int i=0;i<n;i++){
		cin>>t;
		a[t]=i;
		b[i]=t;
	}
	while(m--){
		cin>>t;
		ans+=a[t]/k;
		if(a[t]!=0){
			a[t]--;
			a[b[a[t]]]++;
			swap(b[a[t]],b[a[t]+1]);
		}
	}
	cout<<ans<<endl;
}
