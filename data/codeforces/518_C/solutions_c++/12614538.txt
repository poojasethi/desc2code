 #include <iostream>
using namespace std ; 
int a[100005],b[100005],n , m , k ,c;
long long sum = 0   ; 
int main (){
cin >> n >>  m >> k ;
for(int i = 1 ; i <= n ;i++){
    cin >>a[i];
    b[a[i]] = i ; 
}
for (int i = 1 ; i <= m ; i++){
    cin>> c ; 
    sum += (( b[c]-1 )/ k)+1 ;
    if(b[c] != 1){
     b[c]--;
     b[a[b[c]]]++;
     swap(a[b[c]] , a[b[c]+1]);
    }
}
cout<< sum <<endl; 
return 0 ; 
}
