#include<iostream>
using namespace std;
__int64 a[100010],b[100010];
int main()
{
	__int64 x,y,z,i,t,ans=0;
	cin>>x>>y>>z;
	for(i=1;i<=x;i++)
	{
		cin>>a[i];
		b[a[i]]=i; 
	}
	for(i=0;i<y;i++)
	{
		cin>>t;
		if(b[t]!=1)
		{
			ans+=(b[t]-1)/z;
			swap(a[b[t]],a[b[t]-1]);
			swap(b[a[b[t]]],b[a[b[t]-1]]);
		}
	}
	cout<<ans+y<<endl;
	return 0;
}
