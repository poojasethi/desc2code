#include<bits/stdc++.h>

int main()
{
	int t,i,j,k,n,p,m,a[100001],b[100001];
	unsigned long long u;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;++i)
	{
		scanf("%d",&a[i]);
		b[a[i]]=i;
	}
	while(m--)
	{
		scanf("%d",&t);
		u+=(unsigned long long)(b[t]/k);
		if(b[t]%k==0)
			u-=1ULL;
		if(b[t]!=1)
		{
			b[t]--;
			b[a[b[t]]]++;
			a[b[t]+1]=a[b[t]];
			a[b[t]]=t;
		}
		u+=1ULL;
	}
	printf("%I64d\n",u);
	return 0;
}
