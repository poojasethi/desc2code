#include <bits/stdc++.h>
using namespace std;

int main(){
	long long int n,m,k,a[100001],ids[100001],gestures;
	cin >> n >> m >> k;
	for(int i=1;i<=n;i++){
		cin >> a[i];
		ids[a[i]]=i;
	}
		for(int i=1;i<=m;i++){
		int b;
		cin >> b;
		gestures +=(ids[b]-1)/k+1;
		b = ids[b];
		if (b>1){
			swap(a[b-1],a[b]);
			ids[a[b-1]]=b-1;
			ids[a[b]]=b;
			
			}
		
		}
cout << gestures;
}
