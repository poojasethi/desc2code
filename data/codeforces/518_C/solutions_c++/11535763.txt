#include<iostream>
using namespace std;
int main()
{
	long long int x,y,n,m,k,i,temp,a[100009],pos[100009],ans=0,t;
	cin >> n >> m >> k;
	for(i=1;i<=n;i++)
	{
		cin >> t;
		pos[t]=i;
		a[i]=t;
	}
	for(i=0;i<m;i++)
	{
		cin >> t;
		ans+=pos[t]/k;
		if(pos[t]%k!=0)
			ans++;
		if(pos[t]!=1)
		{
			pos[t]--;
			temp=a[pos[t]];
			a[pos[t]]=a[pos[t]+1];
			a[pos[t]+1]=temp;
			pos[temp]++;
		}
	}
	cout << ans << endl;
	return 0;
}
