#include <bits/stdc++.h>

using namespace std;
string s;long long i,n,t,k,a[100001],b[1000001],sol,x;

int main()
{cin>>n>>t>>k;
for(i=0;i<n;i++)
{
    cin>>a[i];
    b[a[i]]=i;
}

for(i=0;i<t;i++)
{
    cin>>x;//cout<<b[x]/k+1<<' ';
    sol+=b[x]/k+1;//cout<<sol;
    long long  p=b[x];
    if(p>0)
    {
        swap(a[p],a[p-1]);
b[x]--;
        b[a[p]]++;

    }
}cout<<endl<<sol;
}
