#include<iostream>
using namespace std;
int main()
{
    long long int n,m,k,t,count=0,p,i;
    cin>>n>>m>>k;
    long long int a[100001],b[n+1];
    for(i=1;i<=n;i++){cin>>p;b[i]=p;a[p]=i;}
    while(m--)
    {
        cin>>p;
        count+=(a[p]/k+(a[p]%k!=0));if(a[p]!=1){
        t=a[p];
        a[p]=a[p]-1;
        a[b[t-1]]=t;
        b[t]=b[t-1];
        b[t-1]=p;}
    }
    cout<<count<<endl;
}
