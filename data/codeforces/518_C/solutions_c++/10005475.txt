#include<iostream>
#include<cstring>
using namespace std;
long long int a[100002],h[100002];
int main()
{
  long long int n,m,k,t,s=0,i,f,j;
  cin>>n>>m>>k;
  for(i=0;i<n;i++)
    {
      cin>>t;
      a[i]=t;
      h[t]=i;
    }
  for(i=0;i<m;i++)
    {
      cin>>t;
      j=h[t];
      s=s+j/k+1;
      if(j==0)
	continue;

      f=a[j];
      a[j]=a[j-1];
      a[j-1]=f;
      h[a[j]]=j;
      h[a[j-1]]=j-1;

    }
  cout<<s<<endl;
}
