#include<cstdio>
using namespace std;
long long ans;
int n,i,m,k,a[100010],wz[100010],x;
int main()
{
    scanf("%d%d%d",&n,&m,&k);
    for(i=1;i<=n;i++)scanf("%d",&a[i]),wz[a[i]]=i;
    while(m--)
    {
        scanf("%d",&x);ans+=(wz[x]-1)/k+1;
        if(wz[x]>1)wz[x]--,wz[a[wz[x]]]++,a[wz[x]+1]=a[wz[x]],a[wz[x]]=x;
    }printf("%I64d\n",ans);
}
