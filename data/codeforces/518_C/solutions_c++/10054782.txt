#include <stdio.h>
__int64 a[100005], b[100005], n, m, k, res, x;
int main(){
	scanf("%I64d %I64d %I64d", &n, &m, &k);
	for(int i = 1; i <= n; ++ i){
		scanf("%I64d", a + i);
		b[a[i]] = i;
	}
	while(m --){
		scanf("%I64d", &x);
		res += (b[x] + k - 1) / k;
		if(b[x] != 1){
			int q = a[b[x] - 1];
			a[b[x] - 1] = x; a[b[x]] = q;
			b[x] --;	b[a[b[x] + 1]] ++;
		}
	}
	printf("%I64d", res);
	return 0;
}
