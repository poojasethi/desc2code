#include<bits/stdc++.h>
using namespace std;
int a[100010], b[100010];
int main()
{
	int N,M,K,i,x,t;
	long long int c;
	cin>>N>>M>>K;
	for(i = 0; i < N; ++i)
	{
		cin>>x;
		a[x] = i+1;
		b[i+1] = x;
	}

	c = 0;
	while(M--)
	{
		cin>>x;
		c = c + (long long int)(((a[x]-1)/ K)+1);

		if(a[x] > 1)
		{
			t = a[x];
			a[b[a[x]-1]] = a[x];
			--a[x];
			b[t] = b[t-1];
			b[t-1] = x;
		}
	}
	cout<<c;
	return 0;
}
