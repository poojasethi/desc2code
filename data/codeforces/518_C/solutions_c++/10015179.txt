#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	int a[n],i,x,y,j;
	long long int ans=0;
	int b[1000000]={0};
	for(i=0;i<n;i++){ cin>>a[i]; b[a[i]]=i;}
	for(i=0;i<m;i++)
	{
		cin>>x;
		int index=b[x];
		ans=ans+index/k+1;	
		if(index==0){ continue; ans++;}
		y=a[index];
		a[index]=a[index-1];
		a[index-1]=y;
		b[a[index]]=index;
		b[a[index-1]]=index-1;
	}
	cout<<ans<<endl;
	return 0;
}
