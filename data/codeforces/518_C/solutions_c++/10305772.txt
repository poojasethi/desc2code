#include <iostream>

using namespace std;

int a[100001],b[100001],i,j,k,m,n,x,y;
long long res=0;
int main()
{

cin>>n>>m>>k;
for(i=1;i<=n;i++)
{
     cin>>a[i];
     b[a[i]]=i;
}
for(i=1;i<=m;i++)
{
    cin>>x;
    res+=(b[x]-1)/k+1;
    if(b[x]>1)
    {
        swap(a[b[x]],a[b[x]-1]);
        swap(b[x],b[a[b[x]]]);
    }
}
  cout<<res;
}
