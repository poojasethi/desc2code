#include <bits/stdc++.h>

#define MAX 100010

using namespace std;

typedef long long int ll;

int n,m,k,b;
int a[MAX],p[MAX];


int main(){
	scanf("%d %d %d",&n,&m,&k);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);	
		p[a[i]]=i;	
	}

	ll resp=0;
	for(int i=0;i<m;i++){
		scanf("%d",&b);	
		resp+=(p[b]-1)/k + 1;
		if(p[b] == 1) continue;
		int ant=a[p[b]-1];
		swap(a[p[b]], a[p[b]-1]);
		p[b]--;
		p[ant]++;	
	}
	cout << resp << endl;

}
