#include <bits/stdc++.h>
using namespace std;
const int MAXN = 1e5+1;
long long int n, m, k, a[MAXN], pos[MAXN], ans;
int main()
{
	cin>>n>>m>>k;
	for (int i = 1;i <= n;i++)
		cin>>a[i], pos[a[i]] = i;
	for (int i = 1;i <= m;i++)
	{
		int x; cin>>x;
		ans += (pos[x]-1)/k+1;                 
		x = pos[x];
		if (x > 1)
		{
			swap(a[x-1], a[x]);
			pos[a[x-1]] = x-1;
			pos[a[x]] = x;
		}
	}
	cout<<ans;
	return  0;	
}
