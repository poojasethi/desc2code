#include<bits/stdc++.h>

using namespace std;

long long int i,j,k,n,m,ans,arr[100005],tarr[100005];

int main(){
	cin >> n >> m >> k;

	for(i=1 ; i<=n ; i++){
		cin >> arr[i];
		tarr[arr[i]] = i;
	}

	int x;

	for(i=1 ; i<=m ; i++){
		cin >> x;
		ans += ((tarr[x]-1)/k)+1;
		if(tarr[x]!=1){
			tarr[x]--;
			long long int k = arr[tarr[x]];
			arr[tarr[x]]=x;
			tarr[k]++;
			arr[tarr[k]]=k;
		}
	}

	cout << ans << endl;
}
