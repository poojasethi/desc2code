#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll p[100010],e[100010];

int main()
{
	ll n,m,k,ans=0;
	int x;
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&x);
		p[x]=i;
		e[i]=x;
	}
	for(int i=0;i<m;i++)
	{
		scanf("%d",&x);
		ans=(ll)ans+(ll)((ll)p[x]/k)+1;
		if(p[x]!=0)
		{
			p[e[p[x]-1]]=p[x];
			e[p[x]]=e[p[x]-1];
			p[x]-=1;
			e[p[x]]=x;
		}
	}
	cout<<ans<<endl;
	return 0;
}
