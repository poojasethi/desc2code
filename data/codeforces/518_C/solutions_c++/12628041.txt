#include<bits/stdc++.h>

using namespace std;

long long int n,m,k,i,H[100005],arr[100005],a,ans;
long long int temp;

int main(){

	cin >> n >> m >> k;
	for(i = 1; i <= n; i++){
		cin >> arr[i];
		H[arr[i]] = i;
	}

	for(i = 1; i <= m; i++){
		cin >> a;
		temp = H[a];
		ans += (temp-1)/k + 1;

		if(temp != 1){
			int x = arr[temp - 1];
			swap(arr[temp - 1],arr[temp]);
			swap(H[a],H[x]);
		}
	}

	cout << ans << endl;

	return 0;
}
