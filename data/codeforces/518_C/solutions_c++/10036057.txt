#include <bits/stdc++.h>
using namespace std;

map<int, int> mp;
int arr[100000+5];
int main(){
	int n, m, k, a;
	cin>>n>>m>>k;
	for(int i=0;i<n;i++){
		cin>>arr[i];
		mp[arr[i]]= i;
	}
	long long int ans= 0;
	for(int i=0;i<m;i++){
		cin>>a;
		ans+= (long long int)(mp[a]/k) + 1;
		if(mp[a]>0){
			int pos1= mp[a];
			int pos2= mp[a]-1;
			mp[arr[pos2]]= pos1;
			mp[a]= mp[a]-1;
			swap(arr[pos1], arr[pos2]);	
		}
	}	
	cout<< ans<<"\n";
}
