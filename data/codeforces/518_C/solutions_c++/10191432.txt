#include<iostream>
using namespace std;

const int maxn = 110*1000;

int a[maxn], ja[maxn];
int n, m, k;
long long J;

int main(){
	cin >> n >> m >> k;
	for (int i=0 ; i<n ; i++){
		cin >> a[i];
		a[i]--;
		ja[a[i]] = i;
	}	
	while(m--){
		int x;
		cin >> x;
		x--;
		J += ja[x]/k + 1;
		if (!ja[x]) continue;
		int y = a[ja[x]-1];
		ja[x]--;
		ja[y]++;
		a[ja[x]] = x;
		a[ja[y]] = y;
	}
	cout << J << endl;
}
