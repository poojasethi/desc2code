#include <iostream>

using namespace std;

int main()
{int n,m,k; cin>>n>>m>>k;
 int a[100005]={},poz[100005]={};
 for (int i=0;i<n;i++)
 {cin>>a[i]; poz[a[i]]=i;}


long long  ans=0;

 for (int j=0;j<m;j++)
 {int b; cin>>b;
  int i0=poz[b];

  ans+=(1LL*i0/k)+1;
  if (i0>0) {swap(a[i0],a[i0-1]);
              swap(poz[a[i0]],poz[a[i0-1]]);
            }

 }
 cout<<ans;
return 0;
}
