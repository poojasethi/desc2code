#include <iostream>
#include <algorithm>
using namespace std;
long long n, m, k, ans;
int order[100005], v[100005];
int main() {
	cin >> n >> m >> k;
	for(int i = 0; i < n; ++i) { cin >> v[i]; order[v[i]] = i; }
	for(int i = 0; i < m; ++i) {
		int o; cin >> o;
		ans += order[o] / k + 1;
		int orr = order[o];
		if(orr > 0) { ++order[v[orr - 1]]; --order[v[orr]]; swap(v[orr], v[orr - 1]); }
	}
	cout << ans;
}
