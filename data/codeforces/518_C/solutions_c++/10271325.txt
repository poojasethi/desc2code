#include<bits/stdc++.h>
using namespace std;
int o[100005], p[100005];
int main() {
  int n, m, k, x, s, t;
  long long g = 0;
  cin >> n >> m >> k;
  for(int i = 0; i < n; i++) {
    cin >> x;
    o[i] = x;
    p[x] = i;
  }
  for(int i = 0; i < m; i++) {
    cin >> x;
    g+= p[x]/k + 1;
    if(p[x]) {
      t = o[p[x]];
      o[p[x]] = o[p[x]-1];
      o[p[x]-1] = t;

      p[o[p[x]]]++;
      p[x]--;
    }
  }

  cout << g;
  return 0;
}
