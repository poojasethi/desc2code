#include <bits/stdc++.h>
using namespace std;

int pos[100010];
int a[100010];

int main()
{
	int n, m, k, x;

	scanf("%d%d%d", &n, &m, &k);

	for(int i = 0; i < n; i++) { scanf("%d", &a[i]); pos[a[i]] = i; }

	long long count = 0;
	while(m--) {
		scanf("%d", &x);
		count += pos[x]/k;
		if(pos[x]) {
			--pos[x];
			++pos[a[pos[x]]];
			swap(a[pos[x]], a[pos[x]+1]);
		}
		++count;
	}

	printf("%lld\n", count);

	return 0;
}
