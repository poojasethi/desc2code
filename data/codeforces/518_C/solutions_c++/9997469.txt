#include<iostream>
#define ll long long
using namespace std;
int main(){
	ll n,m,k;
	cin>>n>>m>>k;
	ll count=0;
	ll a[n],b[n+1];
	for(int i=0;i<n;i++){
		int tmp;
		cin>>tmp;
		a[i]=tmp;
		b[tmp]=i;
	}
	for(int i=0;i<m;i++){
		int tmp;
		cin>>tmp;
		int x=b[tmp];
		count+=(x/k)+1;
		if(x!=0){
			b[tmp]--;
			b[a[x-1]]++;
			int y=a[x-1];
			a[x-1]=a[x];
			a[x]=y;
		}
	}
	cout<<count;
}
