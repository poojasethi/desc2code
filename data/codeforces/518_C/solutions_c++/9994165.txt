#include <bits/stdc++.h>
using namespace std;

int a[100005];
int pos[100005];

int main(){
  
  int n, m, k, i, b;
  long long cnt = 0;

  scanf("%d %d %d", &n, &m, &k);
  for(i=0; i<n; i++){
    scanf("%d", &a[i]);
    pos[a[i]] = i;
  }
  for(i=0; i<m; i++){
    scanf("%d", &b);
    cnt += pos[b]/k + 1;
    if(pos[b]){
      swap(a[pos[b]], a[pos[b]-1]);
      pos[a[pos[b]]]++;
      pos[b]--;
    }
  }
  cout << cnt << endl;
  return 0;
}
