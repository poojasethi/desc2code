#include<iostream>

using namespace std;

int main()
{
	int A[300000];
	int B[300000];
	long long int temp,i,j,k,a,b,c,fin=0;
	int n,m;
	cin >> n >> m >> k;
	for(i=1;i<=n;i++)
	{
		cin >> c;
		A[c]=i;
		B[i]=c;
	}
	for(i=1;i<=m;i++)
	{
		cin >> c;
		fin+=A[c]/k;
		if(A[c]%k!=0)
			fin++;
		temp=c;
		if(A[c]>1)
		{
			A[c]--;
			temp=B[A[c]];
			B[A[c]]=B[A[c]+1];
			B[A[c]+1]=temp;
			A[temp]++;
		}
	}
	cout << fin << endl;
	return 0;
}
