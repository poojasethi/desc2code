#include<bits/stdc++.h>
using namespace std;

int main(){
	int n,m,k,i,j,tmp,var,locvar;cin>>n>>m>>k;
	int a[n],b[n];
	for(i=0;i<n;i++){cin>>a[i];b[a[i]-1]=i;}
	long long int ans=0;
	for(i=0;i<m;i++){
		cin>>j;var=b[j-1];
		ans+=(((var)/k)+1);//cout<<ans<<endl;
		if(var>0){
			b[a[var]-1]--;b[a[var-1]-1]++;
			tmp=a[var-1];a[var-1]=a[var];a[var]=tmp;
		}
		//~ for(int k=0;k<n;k++)	cout<<a[k]<<" ";cout<<endl;
	}
	cout<<ans<<endl;
	return 0;
}
