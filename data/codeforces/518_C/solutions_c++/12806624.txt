#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n, m, k;
	cin>>n>>m>>k;
	int D[n];
	map<int, int>M;
	for (int i=0; i<n; i++)
	{
		cin>>D[i];
		M[D[i]]=i;
	}
	long long ans=0;
	for (int i=0; i<m; i++)
	{
		int a;
		cin>>a;
		int x=M[a];
		if (M[a]==0)
		{
			ans++;
			continue;
		}
		else
		{
			ans=ans+x/k+1;
			swap(D[x], D[x-1]);
			M[a]=x-1;
			M[D[x]]=x;
		}
	}
	cout<<ans;
	return 0;
}
