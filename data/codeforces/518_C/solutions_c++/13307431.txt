#include <iostream>

using namespace std;

int main(){

long long u,n,m,k,i,p,ans=0,temp,a[100099],b[100099];

	cin>>n>>m>>k;

	for(i=0;i<n;i++)
	{
		cin>>b[i];
		a[b[i]]=i;
	}
	for(i=0;i<m;i++)
	{
		cin>>p;
		ans+=(a[p]/k)+1;
		if(a[p]!=0)
		{
			u=a[p]-1;
			a[p]--;
			a[b[u]]++;
			temp=b[u];
			b[u]=b[a[p]+1];
			b[a[p]+1]=temp;

		}
	}
	cout<<ans<<endl;
	return 0;
}
