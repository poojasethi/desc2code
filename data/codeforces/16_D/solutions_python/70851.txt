#!/usr/bin/env python

from sys import stdin
import re

p = re.compile('\[(\d\d):(\d\d) ([ap])\.m\.\]')

res = 0
prev_ts = 24*60
cnt_rep = 0

N = int(stdin.readline().strip())
for i in xrange(N):
   line = stdin.readline()
   m = p.match(line)
   if m is None:
#     print >>sys.stderr, "Invalid timestamp:", line
      continue
   hh = int(m.group(1)) % 12
   mm = int(m.group(2))
   if m.group(3) == 'p':
      hh += 12
   ts = hh * 60 + mm
#  print >>sys.stderr, hh, mm
   if ts < prev_ts:
      res += 1
      cnt_rep = 0
   elif ts == prev_ts:
      if cnt_rep == 10:
         res += 1
         cnt_rep = 0
   else:
      cnt_rep = 0
   prev_ts = ts
   cnt_rep += 1

print res
