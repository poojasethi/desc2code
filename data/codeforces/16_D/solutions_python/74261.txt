import sys

n = int(sys.stdin.readline().strip())

consec = 0
last = -1
ret = 1
for i in xrange(n):
    s = sys.stdin.readline().strip()
    h = int(s[1:3])
    if h == 12: h = 0
    m = int(s[4:6])
    x = s[7]
    if x == 'p': h += 12
    h *= 60
    h += m
    if h < last or h == last and consec == 10:
	consec = 1
	ret += 1
    elif h == last:
	consec += 1
    else:
	consec = 1
    last = h
print ret
