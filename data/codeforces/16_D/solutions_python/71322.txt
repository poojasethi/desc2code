list = []
for i in range(n):

    min = int(line[4:6])
    if line[7] == 'a':
	aft = 0
    else:
	aft = 1
    if hour == 12:
	hour = 0
    if aft:
	hour += 12
    list.append((hour,min))

days = 1
a = list[0]
number = 0
for i in range(1,n):

	days += 1

	number += 1

	a = list[i]
	number = 0

	days += 1
