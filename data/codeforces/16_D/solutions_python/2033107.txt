#!/usr/bin/python

import sys
import re

n = int(sys.stdin.readline())

p = re.compile('\[([0-9]*):([0-9]*) (a|p).*')
messages = map(lambda x: p.match(x).groups(), sys.stdin)
minutes = map(lambda x: int(x[0])%12*60 + int(x[1]) + 
	                    720*(1 if x[2]=='p' else 0), messages)

days = 1
dupCounter = 0
for i in range(1, len(minutes)):
	if minutes[i] < minutes[i-1]:
		dupCounter = 0
		days += 1
	elif minutes[i] == minutes[i-1]:
		dupCounter += 1;
		if dupCounter == 10:
			dupCounter = 0
			days += 1
	else:
		dupCounter = 0

print days
