def readint(): return int(raw_input())
def readfloat(): return float(raw_input())
def readarray(N, foo):
        res = []
        for _ in xrange(N):
                res.append(foo())
        return res
def readlinearray(foo): return map(foo, raw_input().split())

HOUR=0
MINUTE=1
def get_ndays(logs):
    
    nlogs = []
    for log in logs:
        time, x = log.split()
        hour, minute = map(int, time.split(':'))
        if x == 'p':
            if hour != 12:
                hour += 12
                hour %= 24
        else:
            hour %= 12
        nlogs.append((hour, minute))

    r = 1
    count = 1
    last_log = nlogs[0]
    for log in nlogs[1:]:
        
        if last_log == log:
            count += 1
            if count > 10:
                r += 1
                count = 1
        else:
            count = 1
        
        if log[HOUR] < last_log[HOUR]:
            r += 1
        
        if log[HOUR] == last_log[HOUR] and log[MINUTE] < last_log[MINUTE]:
            r += 1
        
        last_log = log
        
    return r

if __name__ == '__main__':
    n = readint()
    logs = []
    for _ in xrange(n):
        logs.append(raw_input()[1:8])
    print get_ndays(logs)
