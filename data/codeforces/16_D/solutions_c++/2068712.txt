#include<iostream>
#include<cstring>
#include<cstdio>
#include<cctype>

using namespace std;

int n , ans  = 1 , cnt = 1 , last , a , b;
char str[64] , ch;

int main() {
	scanf( "%d" , &n ); getchar();
	for ( int i = 1 ; i <= n ; i++ ) {
		gets( str );
		sscanf( str , "[%d:%d %c.m.]:" , &a , &b , &ch );
		int now = ( a % 12 ) * 60 + b;
		if ( ch == 'p' ) now += 12 * 60;
		if ( i != 1 ) {
			if ( now < last ) {
				ans++;
				cnt = 1;
			} else
			if ( now == last ) {
				if ( ++cnt > 10 ) {
					ans++;
					cnt = 1;
				}
			} else
				cnt = 1;
		}
		last = now;
	}
	printf( "%d\n" , ans );
	
	return 0;
}
