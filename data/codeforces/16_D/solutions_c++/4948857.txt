#include <cstdio>
#include <cstdlib>

int solve(char *s)
{
	int t = (atoi(s+1)%12) * 60 + atoi(s+4);
	if(s[7] == 'p')	t += 12 * 60;
	return t;
}

int main()
{
	char s[100];
	int n, last = -1, now, ans = 1, repeat = 1;
	fgets(s, 100, stdin);
	n = atoi(s);
	while(n--)
	{
		fgets(s, 100, stdin);
		now = solve(s);
		if(now < last)	ans++;
		if(now == last)
		{
			repeat++;
			if(repeat > 10)
				ans++, repeat = 1;
		}
		else	repeat = 1;

		last = now;
	}
	printf("%d\n", ans);
	return 0;
}
