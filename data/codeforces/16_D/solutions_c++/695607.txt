#include<iostream>
#include<cstdio>
using namespace std;

char mess[100], ch;
int n, h,m, tl, t, res = 0, d;

int main(){
    scanf("%d\n",&n);
    tl = -1;
    d = 0;
    res = 1;
    while(n--){
      scanf("[%d:%d %c.m.]:",&h,&m,&ch);
      gets(mess);
      if(h==12) h=0;
      t = h * 60 + m + (ch=='a'?0:12*60);
      if (t < tl) {res++;d=1;}
      if (t == tl) d++;
      if (t>tl) d = 1;
      if(d>10) {res++;d=1;}
      //cout<<t<<" "<<tl<<" "<<res<<endl;
      tl = t;
    }
    cout<<res;
    //system("pause");
}
