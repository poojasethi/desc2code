#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
int main(){
	int n;
	char s[100010];
	while(scanf("%d%*c",&n)!=EOF){
		int i;
		int time=99999,res=0;
		int tp;
		for(i=1;i<=n;i++){
			gets(s);
			int len=strlen(s);
			char ap;
			int q,w;
			sscanf(s,"[%d:%d %c.m.]",&q,&w,&ap);
			if(q==12)q=0;
			if(ap=='p')q+=12;
			int now=q*60+w;
			if(time!=now)tp=0;
			if(now<time){
				res++;
				}
			else if(now==time){
				tp++;
				if(tp>=10)res++,tp-=10;
				}
			time=now;
			}
		printf("%d\n",res);
		}
	return 0;
	}
