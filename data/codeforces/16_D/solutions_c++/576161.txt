#include <stdio.h>

char b[10005];

int gettime(void) {
	int hh, mm;
	sscanf(b+1, "%d:%d", &hh, &mm);
	if(hh==12) hh=0;
	if(b[7]!='a') hh+=12;
	return hh*60+mm;
}

int main(void) {
	int n;
	int days = 1, last = 0, cnt = 0, i;
	scanf("%d", &n);
	gets(b);
	for(i=0;i<n;i++){
		gets(b);
		int t = gettime();
		if(last==t && ++cnt>10) {
			++days;
			cnt = 1;
		}
		if(last > t)
			++days;
		if(last != t) {
			last = t;
			cnt = 1;
		}
	}
	printf("%d\n", days);
	return 0;
}
