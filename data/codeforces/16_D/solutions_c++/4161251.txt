#include <stdio.h>
char str[255];
int main()
{
	int n;
	scanf("%d\n",&n);
	int next=100000;
	int num=0;
	int rep=0;
	for (int i=0;i<n;i++)
	{
		gets(str);
		int hour=(str[1]-'0')*10+(str[2]-'0');
		int minute=(str[4]-'0')*10+(str[5]-'0');
		char ch=str[7];
		int time=hour*60+minute;
		if (ch=='p'&&hour!=12)
			time+=12*60;
		else if (ch=='a'&&hour==12)
			time-=12*60;
		if (time<next)
		{
			rep=0;
			num+=1;
		}
		else if (time==next)
			rep++;
		else if (time>next)
			rep=0;
		if (rep>=10)
		{
			rep-=10;
			num++;
		}
		next=time;
	}
	printf("%d",num);
}
