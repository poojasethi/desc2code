#include<iostream>
#include<cstdio>
#include<string.h>
#include<algorithm>
using namespace std;


int main(){
	int n,hh,mm,sum,last=-1,day=1,cnt=0,first;
	char str[10],str2[101];
	cin>>n;
	getchar();
	for(int i=0;i<n;i++){
		scanf("[%d:%d %s]:",&hh,&mm,str);
		gets(str2);
		sum=hh*60+mm;
		if(str[0]=='p'){
			sum+=720;
		}
		if(hh==12)
			sum-=720;
		if(last==sum)
			cnt++;
		else cnt=0;
		if(cnt>9) 
			day++,cnt=0;
		if(last>sum) 
			day++;
		last=sum;
	}
	printf("%d\n",day);
	return 0;
}
