# include <stdio.h>

int main(){
	
	//freopen("input.txt","r", stdin);
	
	int hh, mm;
	int a[128], n,i;
	char message[128], c;
	
	scanf("%d\n", &n);
	
	for(i=0;i<n;i++){
		scanf("[%d:%d %c.m.]: ", &hh, &mm, &c);
		gets(message);
		if(hh==12) hh=0;
		if(c=='p')	hh+=12;
		if(hh==24)	hh=0;
		a[i]=hh*60+mm;
	}
	
	int ans=1, cnt=1;
	
	for(i=1;i<n;i++){
		if(a[i-1]>a[i]){	ans++;cnt=1;}
		else if(a[i-1]==a[i]){cnt++;if(cnt>10){ans++;cnt-=10;}}
		else cnt=1;
	}
	
	printf("%d\n", ans);
	return 0;
}
