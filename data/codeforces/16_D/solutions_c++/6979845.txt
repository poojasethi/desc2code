#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int n;
	char s[10000];
	cin>>n;
	int time[n],ans=1,sum=-1;
	gets(s);
	for(int i=0;i<n;i++)
	{
		gets(s);
		if(s[1]=='1'&&s[2]=='2'&&s[7]=='a')time[i]=(s[4]-'0')*10+(s[5]-'0');
		else if(s[1]=='1'&&s[2]=='2'&&s[7]=='p')time[i]=(s[4]-'0')*10+(s[5]-'0')+720;
		else 
		time[i]=(s[1]-'0')*600+(s[2]-'0')*60+(s[4]-'0')*10+(s[5]-'0')+(s[7]=='p')*720;
	}
	for(int i=1;i<n;i++)
	{
		if(time[i]==time[i-1])sum++;
		else sum=0;
		if(time[i]<time[i-1])ans++;
		if(sum>9)ans++,sum=0;
	}
	cout<<ans<<endl;
}
