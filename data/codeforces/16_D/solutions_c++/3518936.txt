#include <iostream>

using namespace std;

int main() {
    int n,i,q=-1,t,s=1,f=0;
    string str;
    cin>>n;
    getline(cin,str);
    for(i=0;i<n;i++) {
        getline(cin,str);
        t=((str[1]-'0')*10+(str[2]-'0'))%12*60+(str[4]-'0')*10+(str[5]-'0');
        if(str[7]=='p') t+=720;
        if(t==q) f++; else f=0;
        if(f>9) s++, f=0;
        if(t<q) s++;
        q=t;
        }
    cout<<s;
    return 0;
}
