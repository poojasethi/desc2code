#include<stdio.h>
using namespace std;
int prev=1500,h,m,tm,N,ok;
char c;
int main(){
	scanf("%d\n",&N);
	for(int i=0;i<N;++i){
		scanf("[%d:%d %c",&h,&m,&c);
		while(!(getchar()=='\n'));
		h%=12;
		if(c=='p') h+=12;
		m+=h*60;
		if(m==prev){
			if(tm<9) ++tm;
			else{
				tm=0;
				++ok;
			}
		}
		else{
			if(m<prev){
				++ok;
			}
			tm=0;
		}
		prev=m;
	}
	printf("%d\n",ok);
	return 0;
}
