#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int x[111];

int read() {
	getchar();
	char x;
	int h, m;
	scanf("[%d:%d %c.m.]: %*[^\n]", &h, &m, &x);
	if (h == 12) h = 0;
	if (x == 'p') h += 12;
	return h * 100 + m + 1;
}

int main() {
	int n;

	scanf("%d", &n);
	for (int i=1; i<=n; ++i) x[i] = read();
	int tot = 0, cc = 0;
	for (int i=1; i<=n; ++i) {
//		printf("%d ", x[i]);
		if (x[i] == x[i-1]) cc == 9 ? (++tot,cc=0) : ++cc;
		else if (x[i] < x[i-1]) ++tot, cc = 0;
		else cc = 0;
	}
	printf("%d\n", tot+1);

	return 0;
}
