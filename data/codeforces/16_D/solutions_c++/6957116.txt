#include<cstdio>
#include<cstring>
char s[200004];
int gett(){
    int h,m;
    sscanf(s+1,"%d:%d %*c",&h,&m);
    if(h==12)h=0;
    if(s[7]!='a')h+=12;
    return h*60+m;
}
int n;
int main(){
    int d=1,lst=-1,cnt=0;
    scanf("%d\n",&n);
    for(int i=0;i<n;++i){
        gets(s);
        //printf("s:%s\n",s);
        int t=gett();
        if(lst==t && ++cnt>10){
            ++d;
            cnt=1;
        }
        if(lst>t){
            ++d;
            cnt=1;
        }
        if(lst!=t){
            lst=t;
            cnt=1;
        }
    }
    printf("%d\n",d);
    return 0;
}
