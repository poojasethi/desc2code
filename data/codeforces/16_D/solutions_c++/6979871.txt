#include <iostream>
#include <cstdio>
using namespace std;

int main()
{	
	int n, d=1, x, y=0, s=0;
	char str[100];
	cin>>n;
	gets(str);
	while (n--) {
		gets(str);
		x = (str[1]-'0')*10+(str[2]-'0');
		x = x*60+(str[4]-'0')*10+(str[5]-'0');
		if (str[7]=='p') x=x%720+720;
		else if (str[7]=='a') x%=720;
		// cout<<"x="<<x<<endl; //
		if (x<y) {
			d++; s=1;
		}
		else if (x==y) {
			s++;
			if (s>10) {
				d++; s=1;
			}
		}
		else s=1;
		y=x;
	}
	cout<<d<<endl;
	return 0;
}
