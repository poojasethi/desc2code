#include <iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#define MIN(x,y) ((x)<(y)?(x):(y))
#define MAX(x,y) ((x)>(y)?(x):(y))
using namespace std;
char s[10000];
int main()
{
	int m,n,i,x,y,prex=-1,prey=-1,ans,cnt=1;
	scanf("%d",&n);
	gets(s);
	ans=1;
	for(i=0;i<n;i++)
	{
		gets(s);
		x=(s[1]-'0')*10+s[2]-'0';
		y=(s[4]-'0')*10+s[5]-'0';
		if(x==12) x-=12;
		if(s[7]=='p') x+=12;
		if(x==prex&&y==prey) cnt++;
		else cnt=1;
		if(x<prex||x==prex&&y<prey||cnt==11)
		{
			ans++;
			cnt=1;
		}
		prex=x; prey=y;
	}
	printf("%d\n",ans);
}
