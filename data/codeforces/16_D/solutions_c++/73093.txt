#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  int n;
  cin >> n;
  int h, m, t;
  int prevt = -1;
  char tmpch;
  int res = 1;
  int same = 0;
  while (n--) {
    while (cin >> tmpch && tmpch != '[');
    cin >> h >> tmpch >> m >> tmpch;
    t = ((h % 12) + (tmpch == 'p' ? 12 : 0)) * 60 + m;
    if (prevt == t) {
      same++;
      res += same / 10;
      same %= 10;
    } else {
      if (prevt > t)
        res++;
      same = 0;
    }
    prevt = t;
  }
  cout << res << endl;
  return 0;
}
