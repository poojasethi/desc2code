#include <stdio.h>
#include <string.h>
int main()
{
//    freopen("test.in","r",stdin);
    int n,ans,t1,t2,t;
    char tod[100];
    scanf("%d",&n);
    ans=1;
    getchar();
    t=0;
    for(int i=0;i<n;i++){
        gets(tod);
        t1=t2;
        t2=(tod[1]-'0')*600+(tod[2]-'0')*60+(tod[4]-'0')*10+(tod[5]-'0');
        t2+=(tod[7]=='p'?(24*60):0);
        if(tod[1]=='1'&&tod[2]=='2')t2-=720;
        if(i==0) continue;
        if(t2==t1) t++;else t=0;
        if(t>9) t=0,ans++;
        if(t2<t1) ans++;
    }
    printf("%d\n",ans);
}
