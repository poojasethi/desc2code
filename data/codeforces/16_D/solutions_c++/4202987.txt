#include<cstdio>
using namespace std;

int prev=1500,h,m,mm,n,ok;
char s;

int main()
{
    scanf("%d\n",&n);
    for(int i=0; i<n; i++)
    {
        scanf("[%d:%d %c",&h,&m,&s);
        while(!(getchar()=='\n'));
        h%=12;
        if(s=='p') h+=12;
        m+=h*60;
        if(m==prev)
        {
            if(mm<9) ++mm;
            else
            {
                mm=0;
                ++ok;
            }
        }
        else
        {
            if(m<prev)
            {
                ++ok;
            }
            mm=0;
        }
        prev=m;
    }
    printf("%d\n",ok);
    return 0;
}
