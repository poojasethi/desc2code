#include <stdio.h>
int main()
{
	int n;
	scanf("%d\n",&n);
	int a[100];
	int i = 0;
	int h,m;
	char c;
	char message[50];
	int mints = 0;
	int days = 1;
	while (i < n)
	{
		scanf("[%d:%d %c",&h,&m,&c);
		gets(message);
		h = h%12;
		if (c == 'p')
		{
			h += 12;
		}
		mints = h*60+m;
		a[i] = mints;
		i++;
	}

	int cnt = 1;
	for (int i = 1; i < n; i++)
	{
		int a1 = a[i-1];
		int a2 = a[i];
		if (a2 < a1)
		{
			days++;
			cnt = 1;
		}
		else if (a2 == a1)
		{
			cnt++;
			if (cnt > 10)
			{
				days++;
				cnt -= 10;
			}
		}
		else
			cnt = 1;
	}
	printf("%d\n",days);
	return 0;
}
