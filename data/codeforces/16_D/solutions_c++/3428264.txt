#include <stdio.h>
using namespace std;

char buff [100];

int main() {
	int n, c = 1;
	scanf("%d\n", &n);
	int h, m;
	char l;
	int prev = 0;
	int t = 0;
	int logs = 0;

	while(n--) {
		gets(buff);
		sscanf(buff, "[%d:%d %c.m.]: ", &h, &m, &l);
		if (l == 'p') {
			if (h != 12) {
				h = h + 12;
			}
		} else if (h == 12) h = 0;
		t = h * 60 + m;
		if (prev == t) {
			if (++logs == 10) {
				logs = 0;
				c++;
			}
		} else {
			logs = 0;
			if (prev > t) c++;
		}

		prev = t;
	}

	printf("%d\n", c);
	return 0;
}
