#include <stdio.h>

int main()
{
	int k,c,time,t,h,m,r;
	char temp[205],ap;
	//freopen("in.txt","r",stdin);
	gets(temp);
	sscanf(temp,"%d",&k);
	c=0;
	time=24*60;
	while(k--)
	{
		gets(temp);
		sscanf(temp,"%c%d:%d %c",&ap,&h,&m,&ap);
		h%=12;
		if(ap=='p') h+=12;
		m+=h*60;
		if(m==time)
		{
			r++;
			if(r>10)
			{
				r=1;
				c++;
			}
		}
		else
		{
			r=1;
			if(m<time)
				c++;
		}
		time=m;
	}
	printf("%d",c);
	return 0;
}
