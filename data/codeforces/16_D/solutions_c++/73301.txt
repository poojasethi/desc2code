#include <stdio.h>

int main()
{
	int k,c=0,time,t,h,m,r;
	char temp[205],ap;
	scanf("%d",&k);
	time=24*60;
	while(k--)
	{
		scanf(" %c%d:%d %c%[^\n]",&ap,&h,&m,&ap,temp);
		if(ap=='p' && h!=12) h+=12;
		if(ap=='a' && h==12) h-=12;
		m+=h*60;
		if(m==time)
		{
			r++;
			if(r>10)
			{
				r=1;
				c++;
			}
		}
		else
		{
			r=1;
			if(m<time)
				c++;
		}
		time=m;
	}
	printf("%d",c);
	return 0;
}
