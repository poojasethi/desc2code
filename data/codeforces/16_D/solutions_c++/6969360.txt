#include <cstdio>
#include <cstring>
char a[1000];
int main() {
	int n;
	scanf("%d", &n);
	getchar();
	int t = -1, cnt = 0, ans = 0;
	while(n-- > 0) {
		gets(a);
		if('1' == a[1] && '2' == a[2]) a[1] = '0', a[2] = '0';
		int nt = ((a[1] - 48) * 10 + a[2] - 48) * 60 + (a[4] - 48) * 10 + a[5] - 48;
		if(a[7] == 'p') nt +=  12 * 60;
//		printf("		t=%d,nt=%d, ans=%d\n", t, nt, ans);
		if(nt < t) {
			ans ++;
			cnt = 0;
		} else if(nt == t) {
			cnt++;
			if(cnt >= 10) {
				ans++;
				cnt = 0;
			}
		} else cnt = 0;
		
		t = nt;
	}
	printf("%d\n", ans + 1);
	return 0;
}
