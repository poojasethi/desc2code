#include<cstdio>
int n,p,q=-1,ans,t,f;
char s[110];
int main()
{
//    freopen("in.txt","r",stdin);
	scanf("%d\n",&n);
	for (int i=0;i<n;i++)
	{
		gets(s);
		p=((s[1]-'0')*10+(s[2]-'0'))*60+(s[4]-'0')*10+(s[5]-'0');
		if (s[1]=='1'&&s[2]=='2') p-=s[7]=='a'?720:0;
		else p+=s[7]=='p'?720:0;
		if (p==q) t++; else t=0;
		if (t>9) ans++, t=0;
		if (p<q) ans++;
		q=p;
	}
	printf("%d\n",ans+1);
}
