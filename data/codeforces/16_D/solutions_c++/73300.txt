#include <iostream>
#include <cstring>
#include <string>
#include <cstdio>

using namespace std;

int main()
{
    int n;
    int i,h,m,pm,d,c;
    char s[1000],noon;
    while(scanf("%d", &n)!=EOF)
    {
        gets(s);pm=60000;d=0;
        for(i=0; i<n; ++i)
        {
            scanf("[%d:%d %c.m.]", &h, &m, &noon);gets(s);
            if('a'==noon && 12==h) h=0;
            else if('p'==noon && h<12) h+=12;
            m+=h*60;
            if(m<pm) ++d,c=1;
            else if(m==pm) {++c;if(c>10) ++d,c=1;}
            else c=1;
            pm=m;
        }
        printf("%d\n", d);
    }
    return 0;    
}
