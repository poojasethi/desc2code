s = raw_input()
n = len(s)
ans = ''
i = 0
while i < n:
	while (i < n) and (s[i] == ' '):
		i += 1
	if (i == n):
		print(ans)
		exit(0)
	if (s[i] == '"'):
		i += 1
		ans += '<'
		while (s[i] != '"'):
			ans += s[i]
			i += 1
		ans += '>\n'
		i += 1
	else:
		ans += '<'
		while (i < n) and (s[i] != ' '):
			ans += s[i]
			i += 1
		ans += '>\n'
		i += 1
print(ans)
