s = list(raw_input().strip())
sw = 0
for i in range(len(s)):
    if sw:
        if s[i] == '"':
            sw = 0
            s[i] = " "
        elif s[i] == " ":
            s[i] = "_"
    else:
        if s[i] == '"':
            sw = 1
            s[i] = "#"
for w in "".join(s).split():
    print "<%s>"%(w.replace("_", " ").replace("#", ""))
