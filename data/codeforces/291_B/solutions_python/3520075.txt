S = raw_input()
k = 0
S1 = ""
for i in range(len(S)):
	if (S[i] == '"'):
		k = (k + 1) % 2
		S1 += '"'
	elif (S[i] == ' '):
		if (k == 1):
			S1 += '_'
		else:
			S1 += ' '
	else:
		S1 += S[i]
A = S1.split(' ')
for i in A:
	if (len(i) == 0):
		continue
	Z = '<' + i.replace('_', ' ').replace('"', '') + '>'
	print(Z)
