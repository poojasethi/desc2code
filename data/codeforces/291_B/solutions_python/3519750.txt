import sys
import re

storage = []

def store(s):
    global storage
    storage += [s.groups(1)]
    return "&" + str(len(storage)-1)

inp = sys.stdin.read().strip()
inp = re.compile('"([^"]*)"').sub(store, inp)

inp = re.compile(' +').split(inp)

for i in inp:
    if i[0] == '&':
        s = storage[int(i[1:])]
    else:
        s = i
    sys.stdout.write('<%s>\n' % s)
