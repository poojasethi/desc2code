s = raw_input() + ' '
i = 0
n = len(s)
while i<n:
    j = i+1
    if s[i] == '\"':
        while s[j] != '\"':
            j += 1
        print '<%s>' % s[i+1:j]
        i = j+1
        continue
    if s[i] != ' ':
        while j < n and s[j] != ' ':
            j += 1
        print '<%s>' % s[i:j]
        i = j+1
        continue
    i += 1
