input=raw_input()
opQ=False
lex=''
for i in input:
	if opQ:
		if i <> '"':
			lex+=i
		else:
			print '<' + lex + '>'
			opQ=False
			lex=''
	else:
		if i == '"':
			opQ=True
		else:
			if i == ' ' and lex <> '':
				print '<' + lex + '>'
				lex=''
			else:
				if i <> ' ':
					lex+=i
if lex <> '':
	print '<' + lex + '>'
