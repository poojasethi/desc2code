s = raw_input() + " "

q = False

ls = []
l = ""

for c in s:
  if c == '"':
    q = not q
    if not q:
      ls += [l]
      l = ""
  else:
    if c == ' ':
      if q:
        l += c
      else:
        if l != "":
          ls += [l]
          l = ""
    else:
      l += c

for l2 in ls:
  print "<" + l2 + ">"
