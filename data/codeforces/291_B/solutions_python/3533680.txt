from itertools import cycle
import re
c=cycle(("@","#"))
strs=raw_input()
strs1="".join(next(c) if x=='"' else x for x in strs)
for x in re.findall(r'(@[\sA-Za-z0-9.,?!]{0,}#)|([A-Za-z0-9.,?!]{1,})',strs1):
	y="".join(x)
	if y.startswith("@"):
		print "{0}{1}{2}".format("<",y[1:-1],">")
	else:
		print "{0}{1}{2}".format("<",y,">")
