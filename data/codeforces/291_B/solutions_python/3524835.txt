s = raw_input().strip()

i = 0
while i < len(s):
	j = i + 1
	if s[i] == '"':
		while j < len(s):
			if s[j] == '"': break
			j += 1
		print('<{0}>'.format(s[i+1:j]))
		i = j
	elif s[i] != ' ':
		while j < len(s):
			if s[j] == ' ': break
			j += 1
		print('<{0}>'.format(s[i:j]))
		i = j

	i += 1
