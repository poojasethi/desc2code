import sys

def main():
    s = sys.stdin.readline()
    s.strip()
    l = s.split('"')
    i = 0
    for expr in l:
        if i % 2 == 1:
            print "<"+expr+">"
            i += 1
            continue
        for el in expr.split():
            print "<"+el+">"
        i+=1
main()
