s = raw_input()

i = 0
while i < len(s):
  if s[i] == '"':
    j = i + 1
    while s[j] != '"':
      j += 1
    print '<' + s[i+1:j] + '>'
    i = j + 1
  elif s[i] != ' ':
    j = i + 1
    while j < len(s) and s[j] != ' ':
      j += 1
    x = s[i:j]
    if x:
      print '<' + s[i:j] + '>'
    i = j
  else:
    i += 1
