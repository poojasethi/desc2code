import re

arg = re.compile("\"(.*?)\"")

if __name__ == "__main__":
    s = raw_input()
    args = arg.split(s)
    need_split = True
    for a in args:
        if need_split:
            for aa in a.split():
                print "<" + aa + ">"
        else:
            print "<" + a + ">"
        need_split = not need_split
