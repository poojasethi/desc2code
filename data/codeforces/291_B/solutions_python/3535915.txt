#-*- coding: utf-8 -*-
s=raw_input()

f=0
res=[]
i=0
while i<len(s):
    s1=''
    if s[i]=='"':
        f=f^1
        i+=1
    while i<len(s) and not s[i]=='"':
        s1+=s[i]
        i+=1
    if f==1:
        res.append(s1)
    else:
        if not s.strip()=='':
            res+=s1.strip().split()

for i in res:
    print '<'+i+'>'
