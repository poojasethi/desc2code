s=raw_input()
s2=""
q=False;
for c in s:
    if c=='"':
        q = not q
    if q and c==' ':
        s2=s2+'_'
    else:
        s2=s2+c
s3 = s2.split()
if s3.count('') > 0:
    s3.remove('')
for sc in s3:
    if sc.count('"') > 0:
        sc = sc[1:-1]
    sc = sc.replace('_',' ')
    print '<'+sc+'>'
