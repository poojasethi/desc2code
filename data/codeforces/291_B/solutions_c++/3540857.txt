#include <iostream>
#include <sstream>
#include <string>
using namespace std;
int main()
{
	string s;
	getline(cin, s);
	bool dq=false;
	for(int i=0;i<s.size();++i) {
		if (s[i]=='\"') {
			dq=!dq;
			s[i]=dq?'<':'>';
		}
		else if (s[i]==' ' && dq) s[i]='-';
	}
	istringstream is(s);
	string t;
	while (is>>t) {
		for(int i=0;i<t.size();++i) if(t[i]=='-') t[i]=' ';
		if(t[0]!='<') t='<'+t+'>';
		cout<<t<<endl;
	}
	return 0;
}
