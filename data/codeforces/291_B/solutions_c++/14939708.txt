#include <iostream>
using namespace std;

int main()
{
	string s;
	getline(cin, s);

	for(int i = 0; i < s.size(); i ++)
	{
		if(s[i] == '"')
		{
			i ++;
			string t;
			while(s[i] != '"')
			{
				t += s[i];
				i ++;
			}
			cout << "<" << t << ">" << endl;
		}
		else if(s[i] != ' ')
		{
			string t;
			while(s[i] != '"' && s[i] != ' ' && i < s.size())
			{
				t += s[i];
				i ++;
			}

			cout << "<" << t << ">" << endl;
			i --;
		}
	}
}
