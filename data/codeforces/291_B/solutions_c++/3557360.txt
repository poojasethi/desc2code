#include <iostream>
#include <string>

using namespace std;

int main()
{
	string s;
	getline(cin, s);
	for(int i=0,N=s.size();i<N;i++)
	{
		char c = s[i];
		if(c=='"')
		{
			int fi=++i;
			while(s[i]!='"') i++;
			if(s[i]=='"'&&s[i-1]=='"') cout<<"<>"<<endl;
			else cout<<"<"<<s.substr(fi, i-fi)<<">"<<endl;
		}
		else if(c!=' ')
		{
			int fi=i;
			while(i<N && s[i]!=' ') i++;
			cout<<"<"<<s.substr(fi, i-fi)<<">"<<endl;
		}
	}
	return 0;
}
