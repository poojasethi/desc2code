#include <iostream>
#include <string>
using namespace std;


int main()
{
    string s,c;
    getline(cin,s);
    char ch;
    for(int i=0; i<s.size(); i++)
    {
        if(s[i]==' '){continue;}
        ch=' ';
        if(s[i]=='"') i++,ch='"';
        c="<";
        while(s[i]!=ch && i<s.size()) c+=s[i],i++;
        c+=">";
        cout<<c<<endl;
    }
}
