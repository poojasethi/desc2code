#include <cstdio>
#include <cstring>

char s[100005];

int main()
{
	gets(s);
	int n = strlen(s);
	for(int i=0; i<n; ++i)
        {
		if(s[i] == ' ') continue;
		putchar('<');
		if(s[i] == '"') {
			for(++i; s[i]!='"'; ++i)
				putchar(s[i]);
		} else {
			for(; i<n && s[i]!=' '; ++i)
				putchar(s[i]);
		}
		puts(">");
	}
}
