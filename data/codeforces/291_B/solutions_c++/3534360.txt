#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
int main () {
	int i,j;
	char s[100001];
	gets(s);
	for(i=0;s[i];i++) {
		if(s[i]=='"') {
			printf("<");
			for(j=i+1;s[j]!='"';j++) printf("%c",s[j]);
			printf(">\n");
			i=j;
			continue;
		}
		if(s[i]!=' ') {
			printf("<");
			for(j=i;s[j]!=' '&&s[j]!='"'&&s[j];j++) printf("%c",s[j]);
			printf(">\n");
			i=j;
		}
	}
	return 0;
}
