#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char ch[100009];

int main()
{
	gets(ch);
	string s;
	int i=0,l=strlen(ch);
	while(i<l)
	{
		s="";
		if(ch[i]==' ')
			i++;
		else if(ch[i]=='"')
		{
			i++;
			while(ch[i]!='"')
			{
				s=s+ch[i];
				i++;
			}
			i++;
			cout<<"<"<<s<<">\n";
		}
		else
		{
			while(ch[i]!=' ' && i<l)
			{
				s=s+ch[i];
				i++;
			}
			cout<<"<"<<s<<">\n";
		}
	}
	
	return 0;
}
