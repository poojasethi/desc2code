#include<bits/stdc++.h>
using namespace std;
int main()
{
	char c;
	while(1)
	{
		c=getchar();
		if(c=='"')
		{
			printf("<");
			while(1)
			{
				c=getchar();
				if(c=='"')
				{
					printf(">\n");
					break;
				}
				printf("%c",c);
			}
		}
		else if(!isspace(c))
		{
			printf("<%c",c);
			while(1)
			{
				c=getchar();
				if(c==' '||c=='\n')
				{
					printf(">\n");
					break;
				}
				printf("%c",c);
			}
		}
		if(c=='\n')
			break;
	}
	return 0;
}
