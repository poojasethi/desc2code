#include <cstdio>

const int MAX_N = 100000;
char st[MAX_N+3];
int h;


inline void output()
{
	st[h] = '\0';
	printf("<%s>\n", st);
	h = 0;
}

int main()
{
	int c;
	while ((c=getchar())!='\n') {
		switch (c) {
			case '"':
				while ((c=getchar())!='"')
					st[h++] = c;
				output();
				break;
			case ' ':
				if (h)
					output();
				break;
			default:
				st[h++] = c;
		}
	}
	if (h)
		output();
	return 0;
}
