#include<cstdio>
using namespace std;
char s[100001];
int main()
{
	bool bl=0;
	gets(s);
	for(int i=0;s[i];++i)
	{
		if(s[i]==' ')
		{
			if(bl)puts(">");
			bl=0;continue;
		}
		if(s[i]=='"')
		{
			putchar('<');
			++i;
			while(s[i]!='"')
			{putchar(s[i]);++i;}
			puts(">");
			continue;
		}
		if(!bl)putchar('<');
		putchar(s[i]);
		bl=1;
	}
	if(bl)puts(">");
}
