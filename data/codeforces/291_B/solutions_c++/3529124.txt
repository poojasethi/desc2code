#include <iostream>
#include <string>
using namespace std;
int main()
{
	string s;
	getline(cin,s);
	for (int i=0;i<s.length();i++)
	{
		if (s[i] == '"')
		{
			i++;
			cout << '<';
			while (s[i] != '"')
			{
				cout << s[i];
				i++;
			}
			cout << '>' << endl;
		}
		else{
			if (s[i] != ' ')			{
				cout << '<';
				while (s[i] != ' ' && i < s.length())
				{
					cout << s[i];
					i++;
				}
				cout << '>' << endl;
			}
		}
	}
}
