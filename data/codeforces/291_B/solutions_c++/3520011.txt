#include<cstdio>
#include<iostream>
#include<string>
using namespace std;

string s;

int main()
{
	getline(cin, s);
	for(int i=0; i<s.size(); i++)
	{
		if(s[i]=='"')
		{
			i++;
			cout<<"<";
			while(s[i]!='"')
			{
				cout<<s[i];
				i++;
			}
			cout<<">\n";
		}
		if(s[i]!=' ' && s[i]!='"')
		{
			cout<<"<";
			while(s[i]!=' ' && i<s.size())
			{
				cout<<s[i];
				i++;
			}
			cout<<">\n";
		}
	}
	return 0;
}
