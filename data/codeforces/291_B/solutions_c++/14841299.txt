#include<cstdio>

int len, p;
char s[111111], buf[111111];

void add(char ch) {
	buf[len++] = ch;
}
void out() {
	printf("<");
	for(int i=0;i<len;i++)
		printf("%c", buf[i]);
	printf(">\n");
	len=0;
}

int main() {
	gets(s);
	len=p=0;
	for(p=0;s[p];p++) {
		if(s[p]=='"') {
			p++;
			while(s[p]!='"') {
				add(s[p]);
				p++;
			}
			out();
		}else if(s[p]==' ') {
			if(len) out();
		}else add(s[p]);
	}
	if(len)out();
	
	return 0;
}
