#include <iostream>
#include <string>

using namespace std;
string st;
int main()
{
  
  getline(cin, st);
  
   


for(int i=0; i<st.size(); i++){

  if(st[i]=='"'){
  i++;
  cout<<"<";
   while(st[i]!='"'){
    cout<<st[i];
    i++;
  }
   cout<<">"<<endl;
   i++;
  }else if(st[i]!=' '){
   cout<<"<";
   while(st[i]!=' ' && i!=st.size()){
   cout<<st[i];
   i++;
   }
   cout<<">"<<endl;
  }

}


 return 0;
}
