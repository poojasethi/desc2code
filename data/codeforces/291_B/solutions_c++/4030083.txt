#include <iostream>
#include <sstream>
using namespace std;
int main() {
	string s, t="";
	getline(cin,s);
	int i = 0;
	while (i < s.size()) {
		while (i < s.size() and s[i] == ' ') ++i;
		if (s[i] == '"') {
			++i;
			while (i < s.size() and s[i] != '"') {
				t+=s[i];
				++i;
			}
			++i;
		}
		else {
			while (i < s.size() and s[i] != ' ') {
				t+=s[i];
				++i;
			}
		}
		if (s[i-1] != ' ') cout << '<' + t + '>' << endl;
		t = "";
	}
}
