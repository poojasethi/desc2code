#include <iostream>
#include <string>
using namespace std;

int main() {
  string s;
  getline(cin,s);

  for (int i=0;i<s.size();++i) {
    if (s[i] == ' ') continue;
    if (s[i] == '"') {
      int j = i + 1;
      while (j < s.size() && s[j] != '"') j++;
      cout << '<' << s.substr(i+1,j-i-1) << '>' << endl;
      i = j;
    } else {
      int j = i + 1;
      while (j<s.size() && s[j] != ' ') j++;
      cout << '<' << s.substr(i,j-i) << '>' << endl;
      i = j;
    }
  }
}
