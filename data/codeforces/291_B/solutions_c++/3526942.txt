#include <iostream>

using namespace std;

bool open, radi;

string s;

int main()
{
  getline(cin, s);
  for (int i = 0; i < s.size(); i++)
  {
    if (s[i] == '\"')
    {
      open ^= 1;
      cout << (open ? "<" : ">\n");
    }
    else if (open) cout << s[i];
    else if (radi)
    {
      if (isspace(s[i])) radi = false, cout << ">\n";
      else cout << s[i];
    }
    else if (!isspace(s[i])) radi = true, cout << "<" << s[i];
  }
  if (radi) cout << ">\n";
  return 0;
}
