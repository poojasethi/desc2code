#include <iostream> 
#include <string> 
using namespace std; 
int main(){ 
int i,j,g,w; 
string a,b[100007]; 
char c; 
getline(cin, a);
j=0; 
g=0;
w=a.length();
i=0;
while(a[i]==' ') i++;
while (i<w){ 
c=a[i]; 
if (c=='"') {g=1-g; i++;} else
{
if (c!=' ' or g==1) {i++; b[j]=b[j]+c;} else { j++; while (a[i]==' ') i++; if (i==w) j--;};
} 
} 
for (i=0; i<=j; i++) cout<<"<"<<b[i]<<">\n"; 
return 0; 
}
