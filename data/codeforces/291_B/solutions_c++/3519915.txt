#include <cstdio>

int main()
{
	bool inside=false;
	bool printed=false;
	char c;
	while((c=getchar())!=EOF)
	{
		if(c=='\n') break;
		if(!inside&&c==' ')
		{
			if(printed)
			{
				putchar('>');
				puts("");
				printed=false;
			}
		}
		else
		{
			if(!printed)
			{
				printed=true;
				putchar('<');
			}
			if(c=='"')
			{
				inside=1-inside;
			}
			else putchar(c);
		}
	}
	if(printed) putchar('>');
}
