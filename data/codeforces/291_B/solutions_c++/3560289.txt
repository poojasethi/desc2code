#include<stdio.h>
#include<string.h>
#define N 100005
char a[N];
int main()
{
	int i,j;
	gets(a);
	for(i=0;a[i]!=0;)
	{
		if(a[i]!=' ')
		{
			printf("<");
            if(a[i]=='"')
			{
				for(j=i+1;a[j]!='"';j++)
					printf("%c",a[j]);
				printf(">\n");
				i=j+1;
				continue;
			}
			else
			{
				for(j=i;a[j]!=' '&&a[j]!=0;j++)
					printf("%c",a[j]);
				printf(">\n");
				i=j;
				continue;
			}
		}
		else
			i++;
	}
	return 0;
}
