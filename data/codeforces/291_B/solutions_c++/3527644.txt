#include<stdio.h>




int main()
{
	//freopen("in.txt","r",stdin);
	char c;
	while( scanf("%c",&c) && c != '\n' ){
		if( c == '"' ){
			printf("<");
			while( scanf("%c",&c) && c != '"' )
				printf("%c",c);
			printf(">\n");
		}else if( c != ' ' ){
			printf("<%c",c);
			while( scanf("%c",&c) != EOF && c != '\n' && c != ' ' )
				printf("%c",c);
			printf(">\n");
		}
	}
	return 0;
}
