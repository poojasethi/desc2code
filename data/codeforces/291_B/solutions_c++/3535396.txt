#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
char s[100100];
int main(){
    gets(s);
    int l = strlen(s);
    char e;
    for (int i = 0; i < l; i++){
        if (s[i] == ' ') continue;
        printf("<");
        if (s[i] == '"') e = '"';
        else{
            putchar(s[i]); e = ' ';
        }
        i += 1;
        for (; s[i] != e && i < l; i++) putchar(s[i]);
        printf(">\n");
    }
    return 0;
}
