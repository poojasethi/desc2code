#include <cstdio>
#include <cctype>
using namespace std;


char buf[100009];

int main()
{
    fgets(buf, sizeof(buf), stdin);

    for(int i = 0, j = 0, k = 0; buf[i]; i++)
    if(buf[i] == '"')
        printf(k? ">\n": "<"), k ^= 1;
        else if(k)
        printf("%c", buf[i]);
    else if(!isspace(buf[i]))
        printf("%s%c", j? "": "<", buf[i]), j = 1;
    else
        printf(j? ">\n": ""), j = 0;

    printf("\n");

    return 0;
}
