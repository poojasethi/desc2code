#include <iostream>
using namespace std;
int main(){
string x;
string g="";
int k=0;
getline(cin,x);
for(int i=0;i<x.size();i++){
if(x[i] == '"')
k++;
else if(k==1)
g+=x[i];
else if(k==0 && x[i] != ' ')
g+=x[i];
else if(x[i] == ' ' && g!="" && k!=1){
cout<<'<'<<g<<'>'<<endl;
g="";
}
if(k==2){
         cout<<'<'<<g<<'>'<<endl;
         g="";
         k=0;
         }
if(i==x.size()-1 && g!="")
cout<<'<'<<g<<'>'<<endl;
}
 return 0;   
}
