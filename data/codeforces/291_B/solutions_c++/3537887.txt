#include<cstdio>
#include<cstring>
#define N 1000000
char b[N];
int main(){
	gets(b);
	int n=strlen(b);
	for(int i=0,j;i<n;){
		while( i<n && b[i]==' ') ++i;
		if(i==n) break;

		if(b[i]=='"'){
			j=++i;	
			while(j<n&&b[j]!='"')++j;
			b[j]=0;
			printf("<%s>\n",b+i);
			i=j+1;
		}else{
			j=i;
			while(j<n&&b[j]!=' ')++j;
			b[j]=0;
			printf("<%s>\n",b+i);
			i=j+1;
		}
	}
	return 0;
}
