#include <stdio.h>
#include <string.h>

int main()
{
	int i,len;
	char str[100100];
	gets(str);
	len=strlen(str);
	for(i=0;i<len;i++){
		if(str[i]==' '){
			continue;
		}
		printf("<");
		if(str[i]=='"'){
			for(++i;str[i]!='"';i++){
				printf("%c",str[i]);
			}
		}else{
			for(;i<len&&str[i]!=' ';i++){
				printf("%c",str[i]);
			}
		}
		printf(">\n");
	}
	return 0;
}
