#include <stdio.h>
#include <stdlib.h>
#include "string.h"
#include "stdlib.h"
char s[100005];
int main()
{
	int n;
	int i;
	gets(s);
	n=strlen(s);
	for(i=0; i<n; i++)
	{
		if(s[i]==' ')
			continue;
		printf("<");
		if(s[i]=='"')
			for(i++; s[i]!='"';  i++)
				printf("%c",s[i]);
		else
			for(; i<n&&s[i]!=' '; i++)
				printf("%c",s[i]);
		printf(">\n");
	}
	return 0;
}
