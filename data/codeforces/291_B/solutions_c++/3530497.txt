#include <iostream>
using namespace std;

int main()
{
	string s;
	while (cin >> s) {
		if (s[0] == '\"') {
			s[0] = '<';
			if (s[s.size()-1] == '\"')
				s[s.size()-1] = '>';
			else {
				cout << s;
				getline(cin,s,'\"');
				s.push_back('>');
			}
			cout << s << "\n";
		}
		else cout << '<' << s << ">\n";
	}
	return 0;
}
