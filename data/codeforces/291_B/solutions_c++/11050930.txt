#include <cstdio>
char s[100005];
int main()
{
	char beg=0;
	gets(s);
	for(int i=0;s[i];i++)
	{
		if(!beg)
		{
			if(s[i]=='"')
			{
				putchar('<');
				beg=s[i];
			}
			else if(s[i]!=' ')
			{
				putchar('<');
				putchar(s[i]);
				beg=' ';
			}
		}
		else
		{
			if(s[i]==beg)
			{
				puts(">");
				beg=0;
			}
			else
				putchar(s[i]);
		}
	}
	if(beg)puts(">");
	return 0;
}
