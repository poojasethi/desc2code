#include <string.h>
#include <stdio.h>
#include <vector>

using namespace std;

typedef const char *cstr;

int main() {
    char str[102400];
    gets(str);

    for(cstr s = str, p = str; p; ) {
		p = strchr(p, ' ');
		int l = p ? p++ - s : strlen(s);

		if(l) {
			if(*s == '"') {
				p = strchr(++s, '"');

				l = p - s;
				printf("<%.*s>\n", l, s);

				p++;
			} else
				printf("<%.*s>\n", l, s);
		}

		s = p;
    }

    return 0;
}
