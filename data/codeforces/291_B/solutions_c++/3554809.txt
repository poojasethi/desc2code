#include<cstdio>
#include<cstring>
using namespace std;

char s[100000+5];

int main() {
	gets(s); strcat(s, " ");
	for(int pr = 0, i = 0, inbrkt = 0; s[i]; ++i) {
		if(s[i] == '\"') inbrkt = (inbrkt + 1) % 2;
		if(s[i] == ' ' && inbrkt == 0 && i != 0 && s[i-1] != ' ') {
			while(s[pr] == ' ') ++pr; if(s[pr] == '\"') ++pr;
			int end = i-1;
			while(s[end] == ' ') --end; if(s[end] == '\"') --end;
			printf("<%.*s>\n", end - pr + 1, s+pr);
			pr = i;
		}
	}
	return 0;
}
