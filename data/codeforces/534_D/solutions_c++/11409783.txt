#include <iostream>
#include <vector>
using namespace std;

int mas[200020], x, i, n;
vector<int> a[200020];

int main()
{
    cin >> n;
    for (i = 1; i <= n; i++){
        cin >> x;
        a[x].push_back(i);
    }
    x = 0;
    while (--i){
        while (x >= 0 && a[x].empty())   x -= 3;
        if (x < 0){
            cout << "Impossible";
            return 0;
        }
        mas[i] = a[x].back();
        a[x++].pop_back();
    }
    cout << "Possible\n";
    for (i = n; i >= 1; i--)
        cout << mas[i] << ' ';
    return 0;
}
