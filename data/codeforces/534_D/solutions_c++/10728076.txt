#include<cstdio>
#include<vector>
using namespace std;

#define MAXN 200005

int n;
int in,ans[MAXN];
vector<int> v[MAXN];

int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&in);
        v[in].push_back(i);
    }
    int now=0;
    for(int i=0;i<n;i++)
    {
        while(now>=0&&v[now].empty())now-=3;
        if(now<0){printf("Impossible\n");return 0;}
        ans[i]=v[now].back();
        v[now].pop_back();
        ++now;
    }
    printf("Possible\n");
    for(int i=0;i<n;i++)
        printf("%d ",ans[i]+1);
    return 0;
}
