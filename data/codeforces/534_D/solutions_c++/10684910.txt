#include <iostream>
#include <vector>
#include <algorithm>
#include <stdio.h>

#define pb push_back

using namespace std;

vector < int > a[200020];
int ans[200020],n,x,i,ind,st;

main()
 {
  cin>>n;
  for (i=1;i<=n;i++)
   {
    scanf("%d",&x);
    a[x].pb(i);
   }
  while (ind != n)
   {
    while (st >= 0 && a[st].size() == 0) st-=3;
    if (st < 0)
     {
      cout<<"Impossible"<<endl;
      return 0;
     }
    ans[++ind]=a[st].back();
    a[st].pop_back();
    st++;
   }
  cout<<"Possible"<<endl;
  for (i=1;i<=n;i++)
   printf("%d ",ans[i]);
 }
