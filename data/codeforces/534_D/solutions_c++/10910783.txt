#include <bits/stdc++.h>
#define maxn 200009
using namespace std;
vector<int>G[maxn],ans;
int n,sz[maxn];
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		int x;
		cin>>x;
		G[x].push_back(i);
	}
	int cur=0;
	while((int)ans.size()!=n)
	{
		while(cur>=0&&sz[cur]>=(int)G[cur].size())
			cur-=3;
		if(cur<0)
		{
			puts("Impossible");
			return 0;
		}
		ans.push_back(G[cur][sz[cur]]);
		sz[cur]++;
		cur++;
	}
	puts("Possible");
	for(int i=0;i<(int)ans.size();i++)
		printf("%d ",ans[i]);
	//system("pause");
	return 0;
}
