#define N (1<<18)
#include <bits/stdc++.h>
using namespace std;

int n,x,res[N];
multiset<pair<int,int> > st[3];

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		scanf("%d",&x),st[x%3].insert(make_pair(x,i));

	int last=-1;
	for(int i=1;i<=n;i++)
	{
		set<pair<int,int> >::iterator it=st[(i-1)%3].lower_bound(make_pair(last+2,0));
		if(it==st[(i-1)%3].begin())
		{
			puts("Impossible");
			return 0;
		}
		else
		{
			it--;
			res[i]=(*it).second;
			last=(*it).first;
			st[(i-1)%3].erase(it);
		}
	}
	puts("Possible");
	for(int i=1;i<=n;i++)
		printf("%d ",res[i]);
	cout<<endl;
	return 0;
}
