#include<bits/stdc++.h>
using namespace std;
int n,ans[200005];
vector<int> vec[200005];

int main(void){
	int i,j;
	scanf("%d",&n);
		for(i=1;i<=n;i++){
			scanf("%d",&j);
			vec[j].push_back(i);
		}
		int cur=0,cnt=0;
		while(cnt<n){
			if(vec[cur].size()){
				ans[++cnt]=vec[cur].back();
				vec[cur].pop_back();
				cur++;
			}
			else{
				cur-=3;
				if(cur<0) break;
			}
		}
		if(cnt<n) printf("Impossible\n");
    	else{
        	printf("Possible\n");
        	for(i=1;i<n;i++)
            	printf("%d ",ans[i]);
            printf("%d\n",ans[n]);
        }
	return 0;
}
