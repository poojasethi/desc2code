#include<stdio.h>
#include<vector>
using namespace std;

int N;
vector<int> S[200001], res;
int main() {
	int i, x;
	scanf("%d", &N);
	for (i = 0; i < N; i++) {
		scanf("%d", &x);
		S[x].push_back(i + 1);
	}
	int hand = 0;
	for (i = 0; i < N; i++) {
		while (true) {
			if (hand < 0 || !S[hand].empty()) break;
			hand -= 3;
		}
		if (hand < 0) break;
		res.push_back(S[hand].back());
		S[hand].pop_back();
		hand++;
	}
	if (i != N) printf("Impossible\n");
	else {
		printf("Possible\n");
		for (i = 0; i < N; i++)
			printf("%d ", res[i]);
	}
	return 0;
}
