#include<bits/stdc++.h>
#define pb push_back
using namespace std;
int main(){
int n;
cin>>n;
int i;vector<vector<int> > v(n);
for(i=0;i<n;i++){
	int a;
	cin>>a;
	v[a].pb(i+1);
}
int j=0;vector<int> ans;
for(i=0;i<n;i++){
	while(j>=3 && v[j].empty()){
		j-=3;
	}
	if(v[j].empty()){
		cout<<"Impossible";
		return 0;
	}
	ans.pb(v[j].back());
	v[j].pop_back();
	j++;
}
cout<<"Possible"<<endl;
for(i=0;i<n;i++)
cout<<ans[i]<<" ";
return 0;
}
