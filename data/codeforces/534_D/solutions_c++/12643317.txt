int n;
vector<int> t[222222];
int main(){cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i];
        t[a[i]].push_back(i+1);}
    vector<int> ans;
    int now=-1;
    for(int i=0;i<n;i++){
        now++;
        while(now>=0 && t[now].empty()) now-=3;
        if(now<0){
            puts("Impossible");
            return 0;}
        ans.push_back(t[now][t[now].size()-1]);
        t[now].pop_back();}
    puts("Possible");
    for(int i=0;i<n;i++)printf("%d ",ans[i]);}
