#include <bits/stdc++.h>
using namespace std;
vector<int> v[200002];
int n, a[200002], t, s;
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> n;
    for(int i=1; i<=n; i++){
        cin >> t;
        v[t].push_back(i);
    }
    for(int i=0; i<n; i++){
        while(s>=0&&v[s].size()==0){s-=3;}
        if(s<0){cout << "Impossible";return 0;}
        a[i]=v[s].back();
        v[s++].pop_back();
    }
    cout << "Possible\n";
    for(int i=0; i<n; i++){
        cout << a[i] << ' ';
    }

}
