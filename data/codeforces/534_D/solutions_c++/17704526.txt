#include<cstdio>
#include<vector>
using namespace std;
const int N=200005;
int cnt[N],n;
int x,ans[N],m;
vector<int>vec[N];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&x),cnt[x]++;
		vec[x].push_back(i);
	}
	int now=0;
	while(n)
	{
		while(cnt[now]==0&&now>=0) now-=3;
		if (now<0) {printf("Impossible\n");return 0;}
		while(cnt[now])
		{
			cnt[now]--;
			ans[++m]=vec[now][cnt[now]];
			now++;
			n--;
		}
	}
	printf("Possible\n");
	for (int i=1;i<=m;i++) printf("%d ",ans[i]);
}
