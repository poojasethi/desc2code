#include <iostream>
#include <list>
#include <vector>
#include <cstring>

using namespace std;

int main()
{
	int n, a, i, p;
	vector <int> order;
	list<int> *l;

	cin>>n;

	l = new list<int>[n+2];

	for (i = 1; i <= n; i++)
	{
		cin>>a;
		l[a].push_back(i);
	}

	p = 0;
	do
	{
		if (l[p].size())
		{
			order.push_back(l[p].front());
			l[p++].pop_front();
		}
		else
		{
			p -= 3;
		}
	}
	while (p >= 0);

	if (order.size() == n)
	{
		cout<<"Possible"<<endl;
		for (i = 0; i < n; i++)
		{
			cout<<order[i]<<" ";
		}
		cout<<endl;
	}
	else
	{
		cout<<"Impossible"<<endl;
	}

	return 0;
}
