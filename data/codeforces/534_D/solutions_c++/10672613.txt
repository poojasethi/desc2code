#include <cstdio>
#include <vector>

#define REP(i, n) for(int i = 0; i < n; i++)

using namespace std;

typedef vector<int> vi;

int main(){
	vi x[234567];
	int n;
	scanf("%d", &n);
	int a;
	REP(i, n){
		scanf("%d", &a);
		x[a].push_back(i);
	}
	int ok = 0;
	int now = 0;
	vi ret;
	while(1){
		if(now < 0 || ok == n)break;
		if(x[now].size() == 0){
			now-=3;
			continue;
		}
		ret.push_back(x[now].back());
		x[now].pop_back();
		now++;
		ok++;
	}
	if(ok == n){
		printf("Possible\n");
		REP(i, n)printf("%d ", ret[i]+1);
		printf("\n");
	}else{
		printf("Impossible\n");
	}
	
	return 0;
}
