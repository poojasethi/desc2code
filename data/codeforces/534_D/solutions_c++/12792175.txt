#include <bits/stdc++.h>

using namespace std;

int main()
{
  int n, aux;
  cin>>n;
  int sol[n+1];
  queue <int> v[n+1];
  for(int i=0;i<n;i++)
    cin>>aux, v[aux].push(i+1);
  aux=0;
  for(int i=0;i<n;i++)
  {
    while(v[aux].size()==0)
    {
      aux-=3;
      if(aux<0)
      {
        cout<<"Impossible"<<endl;
        return 0;
      }
    }
    sol[i]=v[aux].front();
    v[aux].pop();
    aux++;
  }
  cout<<"Possible"<<endl;
  for(int i=0;i<n;i++)
    cout<<sol[i]<<' ';
}
