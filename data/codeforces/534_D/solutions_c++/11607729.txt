#include <iostream>
#include <vector>
using namespace std;
int n,cur,ans[200000];
vector<int> data[200000];
int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        int a;cin>>a;
        data[a].push_back(i+1);
    }
    for(int i=0;i<n;i++){
        while(cur>=0&&data[cur].empty())cur-=3;
        if(cur<0){
            cout<<"Impossible";return 0;
        }
        ans[i]=data[cur].back();
        data[cur].pop_back();
        cur++;
    }
    cout<<"Possible\n";
    for(int i=0;i<n;i++)cout<<ans[i]<<" ";
}
