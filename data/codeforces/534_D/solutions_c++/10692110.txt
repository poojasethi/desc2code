#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;
vector<int> v[200005],ans;
int main()
{//freopen("i.txt","r",stdin);
int n;
cin>>n;
for(int i=0;i<n;i++){int x;
cin>>x;v[x].push_back(i+1);}
int cnt=0;
for(int i=0;i<n;i++){
 while(v[cnt].size()==0&&cnt>=0) cnt-=3;
 if(cnt<0){cout<<"Impossible";return 0;}
ans.push_back(v[cnt].back());
v[cnt].pop_back();
cnt++;
}
cout<<"Possible"<<endl;
for(int i=0;i<n;i++)cout<<ans[i]<<" ";
 return 0;
}
