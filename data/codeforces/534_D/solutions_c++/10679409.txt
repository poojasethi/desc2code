#include<bits/stdc++.h>
#define MP make_pair
#define R second
#define L first
using namespace std;
const int maxn=200000+1000;
vector<int> v[maxn];
vector<int> res;
int main(){
	int n;
	cin >> n;
	for(int i=0;i<n;i++){
		int x;
		cin >> x;
		v[x].push_back(i+1);
	}
	int cur=0;
	while(cur>=0){
		if(!v[cur].empty()){
			res.push_back(v[cur].back());
			v[cur].pop_back();
			cur++;
			continue;
		}else
			cur-=3;
	}
	if(res.size()==n){
		cout << "Possible" << endl;
		for(int i=0;i<n;i++)
			cout << res[i] << " ";
		cout << endl;
	}else
		cout << "Impossible" << endl;
	return 0;
}
