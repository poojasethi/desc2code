#include<iostream>
#include<vector>
#define rep(i,j) for(int i=1;i<=j;i++)
using namespace std;
int n,a[300000],ans[300000],len;
vector<int>g[300000];
int main()
{
	cin>>n;
	rep(i,n)
	{
		cin>>a[i];
		g[a[i]].push_back(i);
	}
	int t=0,pos=1;
	rep(i,n)
	{
		while(t>=0&&g[t].size()==0)
		t-=3;
		if(t<0)
		{
			pos=0;
			break;
		}
		ans[len++]=g[t].back();
		g[t++].pop_back();
	}
	if(pos)
	{
		cout<<"Possible"<<endl;
		cout<<ans[0];
		rep(i,len-1)
		cout<<" "<<ans[i];
		cout<<endl;
	}
	else
	cout<<"Impossible"<<endl;
	return 0;
}
