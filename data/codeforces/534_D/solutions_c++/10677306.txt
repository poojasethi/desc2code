#include <bits/stdc++.h>
using namespace std;
const int maxn = 200005;
queue <int> Q[maxn];
vector <int> sol;
int main() {
	int n, x;
	scanf("%d", &n);
	for(int i = 1; i <= n; i++) {
		scanf("%d", &x);
		Q[x].push(i);
	}
	int now = 0;
	do {
		if(!Q[now].empty()) {
			int u = Q[now].front();
			Q[now].pop();
			sol.push_back(u);
			now++;
			while(now >= 0 && Q[now].empty()) {
				now -= 3;
			}
			if(now < 0) break;
		} else break;
	} while(1);
	if(sol.size() == n) {
		printf("Possible\n");
		for(int i = 0; i < n; i++) printf("%d%c", sol[i], i + 1 == n ? '\n' : ' ');
	} else {
		printf("Impossible\n");
	}
	return 0;
}
