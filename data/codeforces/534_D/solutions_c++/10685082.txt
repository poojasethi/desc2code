#include <bits/stdc++.h>
using namespace std;

vector < int > a[200020];
int ans[200020],n,x,i,st;

main()
 {
  cin>>n;
  for (i=1;i<=n;i++)
   {scanf("%d",&x);
    a[x].push_back(i);}
  while (--i)
   {
    while (st >= 0 && a[st].size() == 0) st-=3;
    if (st < 0)
     { cout<<"Impossible"<<endl; return 0; }
    ans[i]=a[st].back();
    a[st++].pop_back();
   }
  cout<<"Possible"<<endl;
  for (i=n;i>=1;i--)
   printf("%d ",ans[i]);
 }
