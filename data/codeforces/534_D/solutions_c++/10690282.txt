#include <bits/stdc++.h>
using namespace std;

int n,x;
int ans[200005];
vector<int> room[200005];
int main(){
	cin>>n;
	for(int q=1;q<=n;q++){
		cin>>x;
		room[x].push_back(q);
	}

	int cnt=0;
	for(int q=0;q<n;q++){
		while(room[cnt].size()==0 && cnt>=0)cnt-=3;
		if(cnt<0){
			cout<<"Impossible\n";
			return 0;
		}
		ans[q]=room[cnt].back();
		room[cnt].pop_back();

		cnt++;
	}

	cout<<"Possible\n";
	for(int q=0;q<n;q++){
		cout<<ans[q]<<" ";
	}	
	cout<<endl;
	return 0;
}
