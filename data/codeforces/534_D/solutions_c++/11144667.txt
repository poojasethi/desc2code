#include <bits/stdc++.h>
#define ll long long int

using namespace std;

stack<int> v[200010];
vector<int> ans;

int main()
{
	int n,x,y,a;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a;
		v[a].push(i);
	}
	x=0;
	while(1)
	{
		while(v[x].size()==0&&x>=0)
			x-=3;
		if(x<0)
			break;
		y=v[x].top();
		ans.push_back(y);
		v[x].pop();
		x++;
	}
	if(ans.size()!=n)
		cout<<"Impossible\n";
	else
	{
		cout<<"Possible\n";
		for(int i=0;i<n;i++)
			cout<<ans[i]<<" ";
		cout<<endl;
	}
	return 0;
}
