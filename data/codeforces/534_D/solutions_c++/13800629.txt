#include<stdio.h>
#include<vector>
using namespace std;
const int maxn=200000+10;
vector<int>vec[maxn];
int cnt[maxn];
vector<int>ans;
int main()
{
	int n;
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		int x;
		scanf("%d",&x);
		vec[x].push_back(i);
		cnt[x]++;
	}
	int v=0;
	while (v>=0)
	{
		if (cnt[v])
		{
			ans.push_back(vec[v][cnt[v]-1]);
			cnt[v]--;
			v++;
		}
		else v-=3;
	}
	if (ans.size()<n) printf("Impossible\n");
	else
	{
		printf("Possible\n");
		for (int i=0;i<ans.size();i++)
		  printf("%d ",ans[i]);
	}
	return 0;
}
