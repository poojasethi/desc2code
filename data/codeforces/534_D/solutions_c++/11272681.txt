#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<vector>
using namespace std;
int a,now,q,n,p;
vector<int> v[200010];
int st[200010],ans[200010];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		scanf("%d",&a);
		v[a].push_back(i);
		st[a]=0;
	}
	int now=0;
	while (now>=0){
		p=v[now].size();
		if (st[now]<=p-1){
			ans[++q]=v[now][st[now]];
			if (q==n) break;
			st[now]++;now++;
		} else now=now-3;
	}
	if (q==n) {
	cout<< "Possible" <<endl;
	for	 (int i=1;i<=n;i++) printf("%d ",ans[i]);} else 
	cout<< "Impossible" <<endl;
	return 0;
}
