#include <bits/stdc++.h>
using namespace std;
vector<int> a[200005];
int b[200005];
int main() {
  int n, x;
  scanf("%d", &n);
  for (int i = 1; i <= n; i++) {
    scanf("%d", &x);
    a[x].push_back(i);
  }
  int current = 0;
  for (int i = 1; i <= n; i++) {
    while (a[current].empty()) {
      current -= 3;
      if (current < 0) {
        printf("Impossible\n");
        return 0;
      }
    }
    b[i] = a[current].back();
    a[current].pop_back();
    current++;
  }
  printf("Possible\n");
  for (int i = 1; i <= n; i++) {
    printf("%d ", b[i]);
  }
  return 0;
}
