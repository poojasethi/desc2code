#include<cstdio>
#include<cstring>
#include<cstring>
#include<algorithm>
#include<stack>
using namespace std;
#define maxn 201000
stack<int> s[maxn];
int ans[maxn];
int main(){
	int n,x;
	scanf("%d",&n);
	for (int i=1;i<=n;i++) {
		scanf("%d",&x);
		s[x].push(i);
	}
	int t=0,cnt=0;
	while (t>=0) {
		if (s[t].empty()) {t-=3;continue;}
		ans[++cnt]=s[t].top();s[t].pop();t++;
	}
	if (cnt!=n) {
		puts("Impossible");
		return 0;
	}
	puts("Possible");
	for (int i=1;i<=n;i++) printf("%d ",ans[i]);
	return 0;
}
