#include <cstdio>
#include <algorithm>
using namespace std;
int a[1000005],b,c,d,e,f,g,h,i,j,k,l,m,n,next[1000005],last[1000005],ans[1000005];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d",&a[i]);
		next[i]=last[a[i]];last[a[i]]=i;
	}
	for(i=0;j<n;){
		if(last[i]){ans[++j]=last[i];last[i]=next[last[i]];i++;}
		else if(i>=3)i-=3;
		else {printf("Impossible\n");return 0;}
	}
	printf("Possible\n");
	for(i=1;i<=n;i++)printf("%d ",ans[i]);
	return 0;
}
