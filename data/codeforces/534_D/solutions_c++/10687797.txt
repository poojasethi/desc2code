#include<stdio.h>
#include<vector>
using namespace std;
int main()
{
	int n;
	vector <int> a[200005];
	scanf("%d", &n);
	for(int i=1;i<=n;i++)
	{
		int tem;
		scanf("%d", &tem);
		a[tem].push_back(i);
	}
	int cur=0, done=0;
	vector <int> ans;
	while(cur>=0)
	{
		int s=a[cur].size();
		if(s>0)
		{
			done++;
			ans.push_back(a[cur][s-1]);
			a[cur].pop_back();
			cur++;
		}
		else cur-=3;
	}
	if(done==n)
	{
		printf("Possible\n");
		for(int i=0;i<ans.size();i++)printf("%d ", ans[i]);
		printf("\n");
	}
	else printf("Impossible\n");
	return 0;
}
