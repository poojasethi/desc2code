#include <iostream>
#include <vector>
using namespace std;
vector <int> a[200001];
vector <int> ans;
int main() {
	// your code goes here
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		a[x].push_back(i+1);
	}
	int cur=0;
	for(int i=0;i<n;i++){
		while(cur>=0 && a[cur].size() == 0 )cur-=3;
		if(cur<0){
			cout<<"Impossible"<<endl;
			return 0;
		}
		int x = a[cur].back();
		ans.push_back(x);
		a[cur].pop_back();
		cur++;
	}
	cout<<"Possible"<<endl;
	for(int i=0;i<n;i++){
		cout<<ans[i]<<" ";
	}
	cout<<endl;
	return 0;
}
