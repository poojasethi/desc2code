#include <cstdio>
#include <vector>
#define N 200005
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define fi(a, b, c) for(int a = (b); a < (c); a++)
using namespace std;

int n, c, ans[N], p, l[N];
vector<int> v[N];

int main(){
	scanf("%d", &n);
	FI(i, 1, n){
		int x;
		scanf("%d", &x);
		v[x].push_back(i);
	}
	
	fi(i, 0, n){
		while(c >= 0 && l[c] == v[c].size()) c -= 3;
		if(c < 0) break;
		
		ans[p++] = v[c][l[c]++];
		c++;
	}
	
	if(p < n) puts("Impossible");
	else{
		puts("Possible");
		fi(i, 0, n) printf("%d%c", ans[i], i == n - 1 ? '\n' : ' ');
	}
}
