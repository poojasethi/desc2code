#include<cstdio>
#include<vector>
using namespace std;

int n, x, cur;
int ans[222222];
vector<int> v[222222];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) v[i].clear();
	for(int i=1;i<=n;i++) {
		scanf("%d", &x);
		v[x].push_back(i);
	}
	cur = 0;
	for(int i=0;i<n;i++){
		while (cur >= 0 && !v[cur].size()) cur -= 3;
		if (cur < 0) {
			printf("Impossible\n");
			return 0;
		}
		int sz = v[cur].size();
		ans[i] = v[cur][sz - 1];
		v[cur].pop_back();
		
		cur++;
	}
	printf("Possible\n");
	for(int i=0;i<n;i++)
		printf("%d%c", ans[i], (i<n-1) ? ' ' : '\n');

	return 0;
}
