#include <bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	int a[n];
	for (int i = 0; i < n; i++) cin>>a[i];
	map<int,set<int> > m;
	for (int i = 0; i < n; i++) m[a[i]].insert(i + 1);
	int count = 0;
	vector<int> ans;
	while(ans.size() < n) {
		if(count < 0){
			cout<<"Impossible\n";
			return 0;
		}  
		if((int)m[count].size() == 0) {
			count -= 3;
			continue;
		}	
		ans.push_back((*m[count].begin()));
		m[count].erase((*m[count].begin()));
		count++;
	}  
	cout<<"Possible\n";
	for (int i = 0; i < n; i++) cout<<ans[i]<<" ";
	cout<<endl; 
}
