n = input()
a = [[] for i in xrange(n)]
r = [0 for i in xrange(n)]
for (i, x) in enumerate(map(int, raw_input().split())):
	a[x].append(i + 1)

ans = []
c = 0
for _ in xrange(n):
	while len(a[c]) == 0 and c >= 3:
		c -= 3
	if len(a[c]) == 0:
		break
	ans.append(a[c].pop())
	c += 1

if len(ans) == n:
	print "Possible"
	print " ".join(map(str, ans))
else:
	print "Impossible"
