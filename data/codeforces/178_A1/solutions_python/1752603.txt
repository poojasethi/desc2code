#include <stdio.h>

typedef __int64 int64;

int main() {
	int64 n, summa=0;
	scanf("%I64d", &n);
	int64 *a= new int64[n];
	for(int64 i=0; i<n; i++)
		scanf("%I64d", &a[i]);
	for(int64 i=0; i<n-1; i++){
		int64 t=1;
		while (i+(t<<1)<n){
			t<<= 1;
		}
		summa+=a[i];
		a[i+t]+=a[i];
		a[i]=0;
		printf("%I64d\n", summa);
	}
}
