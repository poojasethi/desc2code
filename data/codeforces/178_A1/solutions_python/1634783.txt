#include<iostream>

using namespace std;

typedef		unsigned long long ll;

#define		MAX		100010


ll x[MAX];


int main(){

	int n;
	cin >> n;
	for (int i=1;i<=n;i++) cin >> x[i];

	ll res = 0;

	ll k = 1;
	while(k<=n) k<<=1;
	k>>=1;
	for (int i=1;i<n;i++){
		while(i+k>n) k>>=1;
		res += x[i];
		cout << res << endl;
		x[i+k]+=x[i];
	}

	return 0;
}
