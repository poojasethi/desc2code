#include <stdio.h>
#include <string.h>
#define N 100003
int main()
{
	int n,a[N];
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%d",&a[i]);
	int step=0,t;
	for(int i=0;i<n-1;++i)
	{
		t=1<<17;
		if(a[i]!=0)
		{
			while(i+t>n-1)
				t>>=1;
			a[i+t]+=a[i];
			step+=a[i];
			a[i]=0;
		}
		printf("%d\n",step);
	}
	return 0;
}
