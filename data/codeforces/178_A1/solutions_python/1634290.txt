#include <stdio.h>

int n,a[100000],i,p,q;
long long sum;

int main()
{
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    sum=0;
    p=1<<20;
    q=n-1;
    for(i=0;i<q;i++)
    {
        sum+=a[i];
        printf("%I64d\n",sum);
        while(i+p>=n)p=p>>1;
        a[i+p]+=a[i];
    }
    return 0;
}
