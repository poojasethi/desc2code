#include <iostream>

#define N 100002

using namespace std;

int main()
{
    long long v[ N ]={},a;
    int n;
    cin >> n;
    long long ans = 0;
    for( int i = 0 ; i < n - 1 ; i++ )
    {
      cin >> a;
      v[ i ] += a;
      ans += v[ i ];
      cout << ans << endl;
      int r = 1;
      while( r + i < n ) r*=2;
      r/=2;
      v[ r + i ] += v[ i ];
      v[ i ] = 0;
    }
}
