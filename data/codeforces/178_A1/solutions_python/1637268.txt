#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>

using namespace std;

int n,a[100000],b=16384,k,ans;

int main()
{
	cin>>n;
	for(int i=0;i<n-1;i++)
	{
        cin>>k;
        a[i]+=k;
        while(i+b>=n)
            b/=2;
        a[i+b]+=a[i];
        ans+=a[i];
        cout<<ans<<endl;
    }
    cin>>n;
    return 0;
}
