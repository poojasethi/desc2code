#include <iostream>
using namespace std;

unsigned int a[100000];

int main() {
	int n;
	unsigned int r;
	int t;

	ios_base::sync_with_stdio(false);

	cin >> n;
	for (int i = 0; i < n; ++i)
		cin >> a[i];

	r = 0;
	for (int i = 0; i < n - 1; ++i) {
		r += a[i];

		cout << r << "\n";

		t = 0;
		while (i + (1 << (t + 1)) < n)
			++t;
		a[i + (1 << t)] += a[i];
	}

	return 0;
}
