#include<stdio.h>
int main()
{
    int n,i,t;
    long long a[200002],ans=0;
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%I64d",&a[i]);
    for(i=1;i<n;i++)
    {
        t=1;
        while(i+t<=n)t*=2;//gredd
        t>>=1;
        ans+=a[i];
        a[i+t]+=a[i];
        printf("%I64d\n",ans);
    }
    return 0;
}
