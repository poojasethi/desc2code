#include <bits/stdc++.h>
#define ll long long 
using namespace std;
int main() {
	ll n;
	cin >> n;
	ll A[n];
	for(int i = 0; i < n; i++) {
		cin >> A[i];
	}
	ll mov = 0;
	int p = log(n) / log(2);
	for(int i = 0; i < n - 1; ++i) {
		while( n - 1 < i + pow(2, p))p--;
		mov += A[i];
		ll aux = i + pow(2, p);
		A[aux] += A[i];
		cout << mov << endl;
	}
	return 0;
}
