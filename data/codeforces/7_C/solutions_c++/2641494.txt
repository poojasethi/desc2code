# include<iostream>
# include<cstdio>
# include<algorithm>
using namespace std;
typedef __int64 ll;

void gcd(ll a,ll b,ll &g,ll &x,ll &y)
{
    if(!b) {g=a; x=1;y=0;}
    else { gcd(b,a%b,g,y,x); y-=x*(a/b);}
}

int main()
{
    ll a,b,g,x,y,c;
    cin>>a>>b>>c;
    gcd(a,b,g,x,y);
    g = __gcd(a,b);
    //cout<<g<<endl;
    if(c%g) cout<<-1<<endl;
    else cout<<x*(-c)/g<<" "<<y*(-c)/g<<endl;
    return 0;
}
