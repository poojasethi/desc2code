#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
typedef long long LL;
void gcd(LL a,LL b,LL& d,LL& x,LL& y)
{
	if(!b) {d=a;x=1;y=0;}
	else
	{
		gcd(b,a%b,d,y,x);
		y -= x*(a/b);
	}
}
int main()
{
	LL A,B,C,d,x,y;
	cin>>A>>B>>C;
	gcd(A,B,d,x,y);
	if(C%d!=0) puts("-1");
	else
	{
		LL X=-C/d*x,Y=-C/d*y;
		cout<<X<<" "<<Y<<"\n";
	}
	return 0;
}
