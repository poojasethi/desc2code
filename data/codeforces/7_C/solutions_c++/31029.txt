#include<cstdio>
__int64 exgcd(__int64 a,__int64 b,__int64&x,__int64&y){
  if(!b){
    x=1;
    y=0;
    return a;
  }
  int t=exgcd(b,a%b,y,x);
  y-=(a/b)*x;
  return t;
}
int main(){
    int a,b,c;
    scanf("%d%d%d",&a,&b,&c);
    long long x,y;
    long long t=exgcd(a,b,x,y);
    c=-c;
    if(c%t)puts("-1");
    else{
        x*=c/t;
        y*=c/t;
        printf("%I64d %I64d",x,y);
    }
    return 0;
}
