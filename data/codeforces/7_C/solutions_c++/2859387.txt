#include <iostream>
#include <algorithm>

#define LL long long

using namespace std;

int a, b, c, g;
LL x, y;

    inline void exgcd(int a, int b, LL &x, LL &y)
    {
        if (b) {exgcd(b,a%b,y,x); y -= x*(a/b);}
        else x = 1, y = 0; 
    }

int main()
{
    cin >> a >> b >> c;
    exgcd(a,b,x,y);
    g = __gcd(a,b);
    if (c%g) cout << "-1"; else cout << -x*(c/g) << " " << -y*(c/g);
    return 0;
}
