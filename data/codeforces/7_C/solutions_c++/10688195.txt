#include <iostream>

using namespace std;
long long ex_gcd(long long a,long long b,long long &x,long long &y){
    if(b==0){x=1;y=0;return a;}
    else{
        int t=ex_gcd(b,a%b,y,x);
        y=y-(a/b)*x;
        return t;
    }
}
int main()
{
    long long a,b,c,x,y;
    cin >> a >> b >> c;
    long long d = ex_gcd(a,b,x,y);
    if(c%d!=0)cout << -1;
    else cout << -x*(c/d) << " " << -y*(c/d);
    return 0;
}
