#include <cstdio>
#include <iostream>
using namespace std;
long long x,y,a,b,c;

int gcd(int a,int b){
    if (b == 0){
        x = 1; y = 0; return a;
    }
    int d = gcd(b,a%b);
    int t = x;
    x = y;
    y = t - (a / b) * x;
    return d;
}

int main(){
    cin>>a>>b>>c;
    long long d = gcd(a,b);
    if (c % d == 0){
        cout<<x*(-c/d)<<" "<<y*(-c/d)<<endl;
    } else puts("-1");
    return 0;
}
