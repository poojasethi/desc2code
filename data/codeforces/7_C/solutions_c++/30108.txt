#include <iostream>
#include <utility>
using namespace std;
typedef long long LL;

LL xgcd(LL a, LL b, LL* x, LL* y) // ax+by=g
{
	if(b) {
		LL yy, g = xgcd(b,a%b,&yy,x);
		*y = yy - a/b**x;
		return g;
	}
	else {
		*x=1, *y=0;
		return a;
	}
}

int main()
{
	LL A, B, C;
	cin >> A >> B >> C;

	LL x, y;
	LL g = xgcd(A, B, &x, &y);
	if( C%g != 0 )
		cout << -1 << endl;
	else {
		cout << (x*(-C/g)) << " " << (y*(-C/g)) << endl;
	}
}
