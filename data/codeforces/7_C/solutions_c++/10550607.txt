#include<cstdio>
#include<cstring>
using namespace std;
typedef long long LL;
void gcd(LL a, LL b, LL &d,LL &x, LL &y)
{
    if(!b)
	{
	   d=a; x=1; y=0; 
     }
    else 
   {
	gcd(b,a%b,d,y,x);
	 y -= x*(a/b); 
   }
}

int main(){
    LL a,b,c,d,x,y;
    scanf("%lld%lld%lld",&a,&b,&c);
    gcd(a,b,d,x,y);
    if(c%d!=0)  printf("-1\n");
    else 
        printf("%lld %lld\n",-x*(c/d),-y*(c/d));
    return 0;
}
