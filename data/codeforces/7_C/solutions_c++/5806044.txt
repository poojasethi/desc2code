#include <iostream>

using namespace std;



int main()
{
    long long a,b,c,x=1,y=0,x1=0,y1=1,ga,gb;
    cin>>a>>b>>c;
    ga=a;
    gb=b;
    while(gb)
    {
        x=x-((ga/gb)*x1);
        y=y-((ga/gb)*y1);
        ga%=gb;
        swap(ga,gb);
        swap(x,x1);
        swap(y,y1);
    }
    if(c%ga!=0)
    cout<<-1;
    else{
    cout<<x*(c*(-1)/ga)<<" ";
    cout<<y*(c*(-1)/ga);}
    return 0;
}
