#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll ex_gcd(ll a,ll b,ll &x,ll &y){
     if(b==0){
          x=1;y=0;
          return a;
     }
     else
     {
          ll r=ex_gcd(b,a%b,y,x);
          y-=x*(a/b);
          return r;
     }
}
int main()
{
	ll a,b,c,x,y,d;
	cin>>a>>b>>c;
	d=ex_gcd(a,b,x,y);
	if(c%d){
		puts("-1");
	}else{
		x*=-c/d;
		y*=-c/d;
		cout<<x<<" "<<y;
	}
}
