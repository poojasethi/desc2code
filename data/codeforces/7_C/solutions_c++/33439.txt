#include <iostream>

using namespace std;

long long xgcd(long long a, long long b, long long *x, long long *y){
	if(b){
		long long g = xgcd(b, a%b, y, x);
		*y -= a/b**x;
		return g;
	} else {
		*x = 1, *y = 0; return a;
	}
}

int main(){
	long long A, B, C;
	while(cin >> A >> B >> C){
		long long x, y, g = xgcd(A, B, &x, &y);
		if(C%g) cout << -1 << endl;
		else    cout << (-C/g*x) << " " << (-C/g*y) << endl;
	}
}
