#include<iostream>
#include<cstdio>
using namespace std;
typedef long long ll;

ll a,b,c,x,y,g;

ll exgcd(ll a,ll b,ll &x,ll &y){
	if(b==0){
		x=1;y=0;
		return a;
	}
	ll g=exgcd(b,a%b,y,x);
	y=y-a/b*x;
	return g;
}

int main(){
	scanf("%I64d%I64d%I64d",&a,&b,&c);
	g=exgcd(a,b,x,y);
	if(c%g!=0){
		printf("-1\n");
		return 0;
	}
	printf("%I64d %I64d\n",-c/g*x,-c/g*y);
	return 0;
}
