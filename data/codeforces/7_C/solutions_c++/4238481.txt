#include<cstdio>
int extgcd(int a,int b,long long&x,long long&y)
{
    int d=a;
    if(b==0){x=1;y=0;}
    else{
       d=extgcd(b,a%b,y,x);
       y-=a/b*x;
    }
   return d;
}

int main()
{
   int A,B,C,d;
   long long x,y;
   scanf("%d %d %d",&A,&B,&C);
   d=extgcd(A,B,x,y);
   x*=(-C/d);
   y*=(-C/d);
   if(C%d) printf("-1\n");
   else printf("%I64d %I64d\n",x,y);
   return 0;
}
