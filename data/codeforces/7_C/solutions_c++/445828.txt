#include <stdio.h>
typedef long long LL;

LL rr, rs;
LL ext_gcd(LL x, LL y, LL p, LL q, LL r, LL s) {
	if(y==0) {
		rr=p;
		rs=q;
		return x;
	}
	return ext_gcd(y, x%y, r, s, p-r*(x/y), q-s*(x/y));
}

int main(void) {
	LL a, b, c;
	scanf("%I64d%I64d%I64d", &a, &b, &c);
	LL g;
	g = ext_gcd(a, b, 1, 0, 0, 1);
	if(c%g) printf("-1\n");
	else{
		LL k = (-c)/g;
		printf("%I64d %I64d\n", rr * k, rs * k);
	}
	return 0;
}
