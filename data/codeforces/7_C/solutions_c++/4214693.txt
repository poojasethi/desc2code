#include <stdio.h>
#include <string.h>
#define LL long long
LL egcd(LL a,LL b,LL &x,LL &y){
      LL x1,y1,gcd;
      if(!b){x=1, y=0;return a;}
      gcd=egcd(b,a%b,x1,y1);
      y=x1-(a/b)*y1;
      x=y1;
      return gcd;
}
int main(){
	LL A,B,C,x,y,d;
	scanf("%I64d %I64d %I64d",&A,&B,&C);
	d = egcd(A,B,x,y);
	if(C%d) printf("-1\n");
	else printf("%I64d %I64d\n",-x*C/d,-y*C/d);
	return 0;
}
