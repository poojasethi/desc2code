#include <cstdio>
typedef long long ll;
long long a,b,c,t,tt,x,y;
void exgcd(ll a,ll b){
	if(b==0){x=1;y=0;return;}
	exgcd(b,a%b);
	ll t=x;
	x=y;y=t-(a/b)*y;
}
int main() {
	scanf("%I64d%I64d%I64d",&a,&b,&c);
	exgcd(a,b);
	t=a*x+b*y;
	if(c%t==0){
		tt=-c/t;
		x=x*tt;y=y*tt;
		printf("%I64d %I64d\n",x,y);
	}
	else
        puts("-1");
    return 0;
}
