#include <stdio.h>
#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;
void gcd(ll a,ll b,ll &x,ll &y){
    if(!b){x=1;y=0;}
    else{gcd(b,a%b,y,x);y-=x*(a/b);}
}
int main(){
    ll a,b,c;
    ll x,y;
    cin>>a>>b>>c;
    if(c%__gcd(a,b)==0LL){
        gcd(a,b,x,y);
        cout<<-x*c/__gcd(a,b)<<" "<<-y*c/__gcd(a,b)<<endl;
    }else{
        cout<<"-1\n";
    }
    return 0;
}
