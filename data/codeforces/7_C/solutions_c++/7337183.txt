#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef long long LL;
const LL INF = 5*1e18;

void gcd(LL a, LL b, LL& d,LL& x, LL& y)
{
    if(!b){d=a; x=1; y=0; }
    else {gcd(b,a%b,d,y,x); y -= x*(a/b); }
}

int main()
{
    LL a,b,c,d,x,y;
    cin >> a >> b >> c;
    gcd(a,b,d,x,y);
    if(c%d != 0)
        puts("-1");
    else 
        cout<<-x*(c/d)<< " "<< -y*(c/d)<<endl;
    return 0;
}
