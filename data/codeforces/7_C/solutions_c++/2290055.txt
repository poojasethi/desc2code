#include<iostream>
using namespace std;
typedef long long ll;
ll a,b,c,x,y,t;
ll exdgcd(ll a,ll b)
{
	if(b==0)
	{
		x=1;y=0;
		return a;
	}
	ll tmp=exdgcd(b,a%b);
	swap(x,y);
	y-=(a/b)*x;
	return tmp;
}
int main()
{
	while(cin>>a>>b>>c)
	{
		c*=-1;
		t=exdgcd(a,b);
		if(c%t)
		{
			cout<<-1<<endl;
			continue;
		}
		if(t<0)
		x*=-1,y*=-1,t*=-1;
		x*=c/t;y*=c/t;
		cout<<x<<" "<<y<<endl;	
	}
}
