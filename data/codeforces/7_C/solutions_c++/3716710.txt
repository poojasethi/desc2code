#include <stdio.h>
typedef long long LL;	
LL x,y;

void init(LL a,LL b) {
	if(b==0) {
		x=1;
		y=0;
		return ;
	}
	
	init(b,a%b);
	LL t=x;
	x=y;
	y=t-(a/b)*y;
}

int main() {
	LL a,b,c;
	while(scanf("%lld%lld%lld",&a,&b,&c)!=EOF) {
		init(a,b);
		LL t=a*x+b*y;
		if(c%t==0) {
			LL tt=-c/t;
			x=x*tt;
			y=y*tt;
			printf("%lld %lld\n",x,y);
		}
		else
			printf("-1\n");
	}
	return 0;
}
