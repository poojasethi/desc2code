#include<stdio.h>
void ex_gcd(long long a, long long b, long long &x, long long &y, long long &g){
    if(b == 0){x = 1; y = 0; g = a; return;}
    ex_gcd(b, a%b, y, x, g);
    y-=x*(a/b);
}

int main(){
    long long a, b, c, g, x, y;
    scanf("%I64d%I64d%I64d", &a, &b, &c);
    ex_gcd(a, b, x, y, g);
    if(c%g != 0)
        printf("-1\n", g);
    else
        printf("%I64d %I64d\n", -x*(c/g), -y*(c/g));
    return 0;
}
