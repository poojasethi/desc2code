#include<iostream>
using namespace std;
typedef long long ll;
ll a,b,c,s,t;
ll gcd (ll a, ll b, ll & x, ll & y) {
	if (a == 0) {
		x = 0; y = 1;
		return b;
	}
	ll x1, y1;
	ll d = gcd (b%a, a, x1, y1);
	x = y1 - (b / a) * x1;
	y = x1;
	return d;
}
int main() {
    cin>>a>>b>>c;
    ll g=gcd(a,b,s,t);
    if(c%g)
    {
        cout<<"-1";
        return 0;
    }
    c=-c;
    cout<<s*c/g<<" "<<t*c/g;
}
