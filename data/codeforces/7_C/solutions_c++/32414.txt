#include<cstdio>
#include<cstdlib>
#include<cmath>
long long egcd(long long a,long long b,long long &x,long long &y){
	long long d;
	if (b==0LL){ x=1LL;y=0LL;return a;}
	else { d=egcd(b,a%b,y,x);
	y-=a/b*x; return d;}
}

long long a,b,c,g,x,y;

int main(){
	scanf("%I64d%I64d%I64d",&a,&b,&c);
	long long g=egcd(a,b,x,y);
	if (c%g==0) printf("%I64d %I64d\n",x*(-c/g),y*(-c/g));
	else printf("-1\n");
	return 0;
}
