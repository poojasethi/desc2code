#include <stdio.h>
#define LL __int64

LL exp_gcd(LL a,LL b,LL&x,LL&y){
	LL temp,p;
	if(b==0){
		x=1;y=0;
		return a;
	}
	p=exp_gcd(b,a%b,x,y);
	temp=x;x=y;y=temp-(a/b)*y;
	return p;
}
int main(){
	int a,b,c;
	LL x,y;
	scanf("%d%d%d",&a,&b,&c);
	LL gcd=exp_gcd(a,b,x,y);
	if((-c)%gcd) printf("-1\n");
	else printf("%I64d %I64d\n",x*(-c)/gcd,y*(-c)/gcd);
	
	return 0;
}
