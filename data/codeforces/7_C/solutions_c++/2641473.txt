# include<iostream>
# include<cstdio>
# include<algorithm>
using namespace std;
typedef __int64 ll;

void gcd(ll a,ll b,ll &d,ll &x,ll &y)
{
    if(!b) {d=a; x=1;y=0;}
    else { gcd(b,a%b,d,y,x); y-=x*(a/b);}
}

int main()
{
    ll a,b,g,x,y,c;
    cin>>a>>b>>c;
    gcd(a,b,g,x,y);
    //int g = __gcd(a,b);
    if(c%g) cout<<-1<<endl;
    else cout<<x*(-c)/g<<" "<<y*(-c)/g<<endl;
    return 0;
}
