#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL gcd(LL a,LL b,LL& x,LL& y)
{
    if(b==0) {x=1;y=0;return a;}
    else {LL t1=gcd(b,a%b,y,x); y=y-(a/b)*x;return t1;}
}
int main()
{
    LL a,b,c,d,x,y;
    scanf("%I64d %I64d %I64d", &a, &b, &c);
    d=gcd(a,b,x,y);
    if(c%d!=0)
        puts("-1");
    else
        printf("%I64d %I64d\n",-x * (c / d), -y * (c / d));
    return 0;
}
