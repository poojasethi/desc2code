#include<cstdio>
int a,b,c,d;
long long x,y;
void exgcd(int a,int b,int &d,long long &x,long long &y){
	if(!b){
		d=a;
		x=1;
		y=0;
		return;
	}
	exgcd(b,a%b,d,y,x);
	y-=a/b*x;
}
int main()
{
	scanf("%d%d%d",&a,&b,&c);
	c=-c;
	exgcd(a,b,d,x,y);
	if(c%d)
		puts("-1");
	else
		printf("%I64d %I64d",c/d*x,c/d*y);
}
