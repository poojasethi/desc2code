#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;
ll a,b,c;
ll x,y;
ll gcd(ll a,ll b)
{
	if (b==0)
	{
		x=1;y=0;
		return a;
	}
	ll tmp=gcd(b,a%b);
	swap(x,y);
	y-=ll(a/b)*x;
	return tmp;
}
int main()
{
	cin>>a>>b>>c;
	c*=-1;
	ll t=gcd(a,b);

	if (c%t!=0)
	{
		cout<<-1<<endl;
		return 0;
	}
	if (t<0)
		x*=-1,y*=-1,t*=-1;
	x*=c/t;
	y*=c/t;
	cout<<x<<" "<<y<<endl;
	return 0;
}
