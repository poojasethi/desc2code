#include<iostream>
using namespace std;
long long x,y,gcd;
void egcd(long long a,long long b)
{
  if(!b)
  { x=1;y=0;
    gcd=a;return;
  }
  egcd(b,a%b);
  long long z=y;
  y=x-(a/b)*y;x=z;
}
int main()
{
  long long a,b,c;
  while(cin>>a>>b>>c)
  {
    egcd(a,b);
    if(-c%gcd)cout<<"-1\n";
    else cout<<(x*(-c)/gcd)<<" "<<(y*(-c)/gcd)<<"\n";
  }
}
