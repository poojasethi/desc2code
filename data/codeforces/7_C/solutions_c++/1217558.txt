#include <iostream>
#include <cstdio>
using namespace std;
#define ll long long int
ll a,b,c,x,y;     
ll gcd (ll a, ll b, ll & x, ll & y) {
	if (a == 0) {
		x = 0; 
		y = 1;
		return b;
	}
	ll x1, y1;
	ll d = gcd (b%a, a, x1, y1);
	x = y1 - (b / a) * x1;
	y = x1;
	return d;
}
int main (){
	cin >>a>>b>>c;
	ll k = gcd(a,b,x,y);
	if ((-1*c)%k!=0){
		cout <<-1;
		return 0;
	}
	cout <<x*(-c/k)<<" "<<y*(-c/k);
	return 0;
}
