def gcdex(a, b):
	if b == 0:
		return (a, 1, 0)
	(d1, x1, y1) = gcdex(b, a % b)
	(d, x, y) = (d1, y1, x1 - (a / b) * y1)
	return (d, x, y)
	
a, b, c = [long(i) for i in raw_input().split()]
c *= -1
(d, x, y) = gcdex(a, b)
if c % d == 0:
	x *= c / d
	y *= c / d
	if (-5 * 10 ** 18 <= x <= 5 * 10 ** 18) and (-5 * 10 ** 18 <= y <= 5 * 10 ** 18):
		print x, y
	else: 
		print -1
else:
	print -1
