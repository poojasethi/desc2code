def extended_gcd(a,b):
    global gcd
    if b == 0:
        gcd = a       
        return (1,0)
    else:
        (s,t) = extended_gcd(b,a%b)
        return (t,s - (a/b)*t)

a,b,c = map(int, raw_input().split())
c = c*(-1)
solution = extended_gcd(a,b)
num = 5000000000000000000

if c % gcd != 0:
    print "-1"
else:
    if (-num,-num)<=solution<=(num,num) :
        print solution[0]*(c/gcd) , solution[1]*(c/gcd)
    else:
        print "-1"
