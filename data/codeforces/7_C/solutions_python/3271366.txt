#!/usr/bin/env python



def egcd(a, b):
    x,y, u,v = 0,1, 1,0
    while a != 0:
        q,r = b//a,b%a; m,n = x-u*q,y-v*q
        b,a, x,y, u,v = a,r, u,v, m,n
    return b, x, y


temp = raw_input()
tab = temp.split()
for i in xrange(len(tab)):
    tab[i] = int(tab[i])
a, b, c = egcd(tab[0],tab[1])
if tab[2]%a==0 and -5*10**18<=b*tab[2]/a<=5*10**18 and -5*10**18<=c*tab[2]/a<=5*10**18:
    print -b*tab[2]/a, -c*tab[2]/a
else:
    print "-1"
