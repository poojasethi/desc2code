a,b,c = map(int,raw_input().split());

def ext_gcd(a,b):
	if a == 0: return 0,1;
	x1,y1 = ext_gcd(b%a,a);
	x = y1-int(b/a)*x1;
	y = x1;
	return x,y;

def gcd(a,b):
	if (a==0): return b
	return gcd(b%a,a)

x,y = ext_gcd(a,b)

g = gcd(a,b)

if c%g==0:
	x *= int(-c/g)
	y *= int(-c/g)
	print x,y
else:
	print -1
