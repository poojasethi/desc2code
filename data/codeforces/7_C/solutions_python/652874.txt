def gcd(a,b):
	x=0
	lastx=1
	y=1
	lasty=0
	while b!=0:
		q= a/b
		a,b=b,a%b
		x,lastx=lastx-q*x,x
		y,lasty=lasty-q*y,y
	ans=[lastx]
	ans+=[lasty]
	return ans
