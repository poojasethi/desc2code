def gcdex (a, b):
    if a == 0:
        return 0, 1, b
    x, y, d = gcdex (b % a, a)
    return y - (b / a) * x, x, d
a, b, c = map (int, raw_input ().split ())
M = 5 * 10 ** 18
x, y, d = gcdex (a, b)
if c % d == 0:
    x *= -c / d
    y *= -c / d
    if -M <= x <= M and -M <= y <= M:
        print x, y
    else:   
        print -1
else:
    print -1
