def egcd(a, b):
    x,y, u,v = 0,1, 1,0
    while a != 0:
        q, r = b//a, b%a
        m, n = x-u*q, y-v*q
        b,a, x,y, u,v = a,r, u,v, m,n
    gcd = b
    return gcd, x, y


def linear_diophantine(a, b, c):
  d,x,y = egcd(a,b);
  if c%d:
    x=y=-1;
  else:
    x = c*x/d;
    y = c*y/d;
  return x,y
