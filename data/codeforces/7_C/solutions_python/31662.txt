#!/bin/env python

import sys

def gcd(a, b):
    x, last_x = 0, 1
    y, last_y = 1, 0

    while b:
        q = a/b
        a, b = b, a%b
        last_x, x = x, last_x - q*x
        last_y, y = y, last_y - q*y
    i = last_x
    j = last_y
    return a, i, j

def solve(A, B, C):
    if A == 0:
        if C%B: return False
        return (0, -C/B)

    if B == 0:
        if C%A: return False
        return (-C/A, 0)

    k, i, j = gcd(B, A)
    if C%k: return False
    y = i*-C/k
    x = -(B*y + C)/A
    return (x,y)

A,B,C = map(int, sys.stdin.readline().strip().split())
res = solve(A, B, C)
if res is False:
    print -1
else:
    x,y = res
    if x < -5e18 or x > 5e18 or y < -5e18 or y > 5e18:
        print -1
    else:
        print x,y
