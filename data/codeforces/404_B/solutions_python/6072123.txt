#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

a, d = map(float, raw_input().split())
n = input()

def floor(x): 
    return int(x)

def get_edges(l, a): 
    return floor(l/a)

for i in xrange(1, n+1): 
    l = i * d
    e = get_edges(l, a)

    rem = l - e * a
    pos = e % 4

    if pos == 0 : 
        print rem, 0.0
    elif pos == 1: 
        print a, rem
    elif pos == 2: 
        print a - rem, a
    else : 
        print 0, a - rem
