def get_point(a, dist):
    if dist < a:
        return dist, 0
    elif a <= dist < a * 2:
        return a, dist - a
    elif a * 2 <= dist < a * 3:
        return a - (dist - 2 * a), a
    elif a * 3 <= dist < a * 4:
        return 0, a - (dist - 3 * a)


a, d = map(float, raw_input().split())
n = input()
dist = 0.0
tmp = a * 4
for i in range(n):
    dist = (dist + d) % tmp
    print "%.10f %.10f" % get_point(a, dist)
