a, d = map(float, raw_input().split())
n = int(raw_input())

M = int(d/a)
delta = d - a*M

def print_coord(side, dist):
  if side == 0:
    print dist, 0.0
  elif side == 1:
    print a, dist
  elif side == 2:
    print a-dist, a
  else:
    print 0.0, a-dist

side = 0
dist = 0.0
for i in range(n):
  side = (side+M)%4
  dist += delta
  if dist >= a:
    dist -= a
    side = (side+1)%4
  print_coord(side, dist)
