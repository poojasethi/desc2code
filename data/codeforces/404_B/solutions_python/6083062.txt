def pos(a, d):
    d %= 4*a
    if d < a:
        return d, 0
    elif d < 2*a:
        return a, d-a
    elif d < 3*a:
        return a-(d % a), a
    else:
        return 0, a - (d % a)

a, d = map(float, raw_input().split())
n = int(raw_input())

for i in range(1, n+1):
    p = pos(a, i*d)
    print p[0], p[1]
