import sys
from random import randint
from time import time

last=time()

def interval():
    global last
    print time()-last
    last=time()

def readIn():
    return map(lambda x:float(x),sys.stdin.readline().strip("\n").split(" "))

def getCoordinate(x,a):
    section=int(x/a)%4
    dist=x-int(x/a)*a
    if section==0:
        return (dist,0)
    elif section==1:
        return (a,dist)
    elif section==2:
        return (a-dist,a)
    elif section==3:
        return (0,a-dist)


def main():
    a,d=readIn()
    n=int(sys.stdin.readline())
    for i in xrange(1,n+1):
        coord=getCoordinate(d*i,a)
        print str(coord[0]) + " " + str(coord[1])
    



main()
