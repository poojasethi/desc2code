a, d = list(map(float, raw_input().split(" ")))
n, x = int(raw_input()), float()
for _ in range(n):
    x += d
    x %= 4*a
    if x <= a:
        print(str(x) + " 0")
    elif x <= 2*a:
        print(str(a) + " " + str(x-a))
    elif x <= 3*a:
        print(str(3*a-x) + " " + str(a))
    else:
        print("0 " + str(4*a-x))
