#!/usr/bin/env python
import math
a, d = map(float, raw_input().split())
n = input()

edge, delta = 0, 0.0

def print_pos(edge, delta):
	if edge == 0:
		print delta, 0
	elif edge == 1:
		print a, delta
	elif edge == 2:
		print a-delta, a
	else:
		print 0, a-delta

for _ in range(n):
	new_delta = math.fmod(delta + d, a)
	turns = int((delta + d - new_delta) / a)
	new_edge = (edge + turns) % 4
	print_pos(new_edge, new_delta)
	edge, delta = new_edge, new_delta
