[a,d] = map(float,raw_input().split())
[n] = map(int,raw_input().split())
for i in range (1,n+1):
    k = int(i*d/a)
    r = i*d-k*a
    k = k%4
    if k == 0 :
        print '%.5f ' % r + '%.5f' % 0
    if k == 1 :
        print '%.5f ' % a + '%.5f' % r
    if k == 2 :
        print '%.5f ' % (a-r) + '%.5f' % a
    if k == 3 :
        print '%.5f ' % 0 + '%.5f' % (a-r)
