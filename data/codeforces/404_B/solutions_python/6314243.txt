from math import floor
a,d=map(float,raw_input().split())
n=input()
t=0
for i in range(1,n+1):
    rest=i*d%(4*a)
    if rest<a:
        print '{:f} {:f}'.format(rest,0)
    elif rest<2*a:
        print '{:f} {:f}'.format (a, rest-a)
    elif rest<3*a:
        print '{:f} {:f}'.format (3*a-rest,a)
    else:
        print '{:f} {:f}'.format (0,4*a-rest)
