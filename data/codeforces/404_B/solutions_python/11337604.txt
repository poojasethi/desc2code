a, d = map(float, raw_input().split())
n = input();

for i in range(n):
	x = (d*(i+1)) % (4.0*a);
	# print i, ": ", x;
	if x >= 0 and x < a:
		print "%f 0.0000"%x
	x -= a;
	if x >= 0 and x < a:
		print "%f %f"%(a,x)
	x -= a;
	if x >= 0 and x < a:
		print "%f %f"%(a-x, a)
	x -= a;
	if x >= 0 and x < a:
		print "0.0000 %f"%(a-x)
	x -= a;
