(a, d), n, acc = map(float, raw_input().split()), int(raw_input()), 0.0
for i in range(n):
	acc = (acc + d) % (4 * a)
	x = acc % a if acc < a else a if acc < 2 * a else a - acc % a if acc < 3 * a else 0
	y = 0 if acc < a else acc % a if acc < 2 * a else a if acc < 3 * a else a - acc % a
	print x, y
