from math import floor

a,d=map(float,raw_input().split())
n=int(raw_input())

def nlen(x):
    return int(floor(x/a))


for i in range(n):
   t=(i+1)*d
   b=nlen(t)
   
   if b%4==0:
       print t-a*b, 0
   elif b%4==1:
       print a, t-a*b
   elif b%4==2:
       print a-(t-a*b), a
   else:
       print 0, a-(t-a*b)
