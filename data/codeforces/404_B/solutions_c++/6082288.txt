#include <cstdio>

double a, d, r, t, x, y;
long long n, k, i;

int main()
{
	scanf("%lf%lf%d", &a, &d, &n);

	for (i = 1; i <= n; i++)
	{
		k = i * d / a;
		t = i * d - k * a;
		k %= 4;
		switch (k)
		{
			case 0: x = t, y = 0; break;
			case 1: x = a, y = t; break;
			case 2: x = a - t, y = a; break;
			case 3: x = 0, y = a - t; break;
		}
		printf("%.6lf %.6lf\n", x, y);
	}

	return 0;
}
