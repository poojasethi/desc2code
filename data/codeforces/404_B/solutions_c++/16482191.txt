#include<cstdio>
int main()
{
    double a,b=0,q;
    int k,z=0,w;
    scanf("%lf%lf%d",&a,&q,&k);
    while(k--)
    {
        b+=q;
        w=b/a;
        b-=a*w;
        z=(z+w)%4;
        printf("%.5lf %.5lf\n",z!=0?(z!=1?(z!=2?0.0:a-b):a):b,z!=0?(z!=1?(z!=2?a-b:a):b):0.0);
    }
}
