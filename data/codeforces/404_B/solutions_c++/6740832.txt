#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	double a,n;
	int m;
	cin>>a>>n>>m;
	double s=0;
	bool w=true;
	while(m--)
	{
		s+=n;
		s=s-int(s/(4*a))*4*a;
		if(s>=3*a)printf("%.10lf %.10lf\n",0.0,4*a-s);
		else if(s>=2*a)printf("%.10lf %.10lf\n",3*a-s,a);
		else if(s>=a)printf("%.10lf %.10lf\n",a,s-a);
		else printf("%.10lf %.10lf\n",s,0.0);
	}
	return 0;
}
