#include <cstdio>
#include <iostream>
using namespace std;
int main(int argc, char *argv[])
{
	double a,b,sum,x,y,t,z;
	int n;
	cin>>a>>b>>n;
	z=4*a;
	sum=0.0;
	while(n--){
		sum+=b;
		x=int(sum/z);
		x=sum-z*x;
		sum=x;
		t=int(x/a);
		y=x-t*a;
		if(t==0)	{t=y;y=0;}
		else if(t==1)	t=a;
		else if(t==2)	{t=a-y;y=a;}
		else if(t==3)	{t=0;y=a-y;}
		printf("%.10lf %.10lf\n",t,y);
	}
	return 0;
}
