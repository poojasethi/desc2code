#include<stdio.h>
#include<cmath>
#include<algorithm>
main()
{
 double a,d,x=0;
int n,i;
scanf("%lf%lf%d",&a,&d,&n);

while(d>4*a)
d-=4*a;

for(i=0;i<n;i++)
{
 x+=d;
while(x>=4*a)
  x-=4*a; 

 if(x<=a)
  printf("%lf 0.0000000000\n",x);
  
else if(x<2*a)
printf("%lf %lf\n",a,x-a);


else if(x<=3*a)
 printf("%lf %lf\n",a-(x-a*2),a);
 

else if(x<=4*a)
 printf("0.0000000000 %lf\n",a-(x-3*a));

}

return 0;
}
