#include<stdio.h>
int main()
{
	double a,b,s;
	int n,f;

	scanf("%lf%lf%d",&a,&b,&n);
	s=f=0;
	while(n--)
	{
		s+=b;
		f+=(int)(s/a)%4;
		f%=4;
		s-=(int)(s/a)*a;
		if(f==0)printf("%.10f 0.0000000000\n",s);
		else if(f==1)printf("%.10f %.10f\n",a,s);
		else if(f==2)printf("%.10f %.10f\n",a-s,a);
		else printf("0.0000000000 %.10f\n",a-s);
	}
	return 0;
}
