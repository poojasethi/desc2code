#include<stdio.h>

int main(){
	double a=0,x,y,p;
	long long int i,j,k,t=0,n;
	scanf("%lf %lf",&x,&y);
		p=x;
	scanf("%lld",&n);
	while(y>4*x)
		y-=4*x;
	for(i=0;i<n;i++){
		
		a+=y;
		while(a>=p*4)
			a-=p*4;
		if(a<=p)
			printf("%lf %lf\n",a,0.0);
		else if(a<=2*p)
			printf("%lf %lf\n",p,a-p);
		else if(a<=3*p)
			printf("%lf %lf\n",p-(a-2*p),p);
		else if(a<=4*p)
			printf("%lf %lf\n",0.0,p-(a-3*p));

	}

	return 0;
}
