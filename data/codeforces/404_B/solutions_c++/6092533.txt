#include <bits/stdc++.h>
using namespace std;

int main(){
    double a,d;
    int N;
    cin >> a >> d >> N;
    
    for (int i=1;i<=N;i++){
        double x = fmod(i*d,4*a);
        
        if (x<=a) printf("%lf %lf\n",x,0.0);
        else if (x<=2*a) printf("%lf %lf\n",a,x-a);
        else if (x<=3*a) printf("%lf %lf\n",3*a-x,a);
        else printf("%lf %lf\n",0.0,4*a-x);
    }
}
