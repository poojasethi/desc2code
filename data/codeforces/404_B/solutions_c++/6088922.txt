#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
  double a,d;
  int n;
  cin >> a >> d >> n;
  double r = 0;
  for(int i=0; i<n; i++){
	r += d;
	r -= (4 * a) * (int)(r / (4 * a));
	double x,y;
	if(r <= a){
	  x = r;
	  y = 0;
	}else if(r <= 2*a){
	  x = a;
	  y = r - a;
	}else if(r <= 3*a){
	  x = 3*a - r;
	  y = a;
	}else{
	  x = 0;
	  y = 4*a - r;
	}
	printf("%.10f %.10f\n",x,y);
  }
  return 0;
}
