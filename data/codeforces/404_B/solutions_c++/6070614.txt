#include<cstdio>
double a, d, s, x, y;
int n, p;
int main(){
	while(~scanf("%lf %lf", &a, &d)){
		scanf("%d", &n);
		x=0.0;
		s=a*4.0;
		while(n--){
			x+=d;
			p = (int)(x/s);
			x-=s*p;
			if(x<=a)	printf("%.6lf 0.000000\n", x);
			else if(x<=a*2.0)	printf("%.6lf %.6lf\n", a, x-a);
			else if(x<=a*3.0)	printf("%.6lf %.6lf\n", a*3.0-x, a);
			else	printf("0.000000 %.6lf\n", a*4.0-x);
		}
	}
	return 0;
}
