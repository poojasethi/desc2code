#include<stdio.h>
int main(){

	//freopen("in.txt","r",stdin);

	double a,d,s;
	int i,n,t,k;

	scanf("%lf %lf\n%d",&a,&d,&n);
	t=0;s=0;
	for(i=0;i<n;i++){
		k=(int)((s+d)/a);
		t=(t+k)%4;
		s=s+d-a*k;
		if(!t)printf("%lf %lf\n",s,0.0);
		if(t==1)printf("%lf %lf\n",a,s);
		if(t==2)printf("%lf %lf\n",a-s,a);
		if(t==3)printf("%lf %lf\n",0.0,a-s);
	}

	return 0;
}
