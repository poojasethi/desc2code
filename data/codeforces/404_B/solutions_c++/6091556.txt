#include <iostream>
#include <stdio.h>
#include <math.h>

using namespace std;

int main(){
    double a,d,n,i;
    cin>>a>>d>>n;
    for(i=1;i<=n;i++){
        double x=fmod(i*d,4*a);
        if(x<=a)
        cout<<x<<" "<<0<<endl;
        else if(x<=2*a)
        cout<<a<<" "<<(x-a)<<endl;
        else if(x<=3*a)
        cout<<a-(x-2*a) <<" "<<a<<endl;
        else
        cout<<0<<" "<<a-(x-3*a)<<endl;
    }
    return 0;
}
