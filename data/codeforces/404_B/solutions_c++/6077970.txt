#include<stdio.h>

int main(){

	double a=0,b=0,d=0;
	int n=0,i=0;
	double x=0;

	scanf("%lf %lf",&a,&d);
	scanf("%d",&n);
	while(d>4*a)
		d-=4*a;
	for( i=0 ; i<n ; i++){
		x+=d;
		while(x>=4*a)
			x-=4*a;

		if(x<=a) printf("%lf %lf\n",x,0.0);
		else if(x<=2*a) printf("%lf %lf\n",a,x-a);
		else if(x<=3*a) printf("%lf %lf\n",a-(x-2*a),a);
		else if(x<=4*a) printf("%lf %lf\n",0.0,a-(x-3*a));

	}

	return 0 ;

}
