#include<stdio.h>

int main(void){
	double m,n;
	scanf("%lf%lf",&m,&n);
	int t;
	scanf("%d",&t); 
	long long int x;
	double y;
	for(int i=1;i<=t;i++){
		x=(i*n/m);y=i*n-m*x;
		if(x%4==0) printf("%.10lf 0.0000000000\n",y);
		else if(x%4==1) printf("%.10lf %.10lf\n",m,y);
		else if(x%4==2) printf("%.10lf %.10lf\n",m-y,m);
		else if(x%4==3) printf("0.0000000000 %.10lf\n",m-y);
	}
	
	return 0;
}
