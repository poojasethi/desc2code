#include <cstdio>
#include <cmath>

using namespace std;

int main()
{
    double a,d,x,y,s=0.0;
    int n;
    scanf("%lf%lf%d",&a,&d,&n);
    for(int i=1; i<=n; i++){
        s+=d;
        s=fmod(s,4*a);
        if(s <= a)x=s,y=0;
        else if(s <=2*a)x=a,y=s-a;
        else if(s <=3*a)x=a-(s-2*a),y=a;
        else x=0,y=a-(s-3*a);
        printf("%lf %lf\n",x,y);
    }
    return 0;
}
