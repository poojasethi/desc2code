#include<cstdio>

double a, d;
int n;

int main() {
	scanf("%lf%lf%d", &a, &d, &n);
	if(d>=a*4) {
		int x = d/(a*4);
		d-=x*a*4;
	}
	double v = 0;
	for(int i=0;i<n;i++) {
		v+=d;
		if(v>=a*4)v-=a*4;
		if(v<a) printf("%.10lf %.10lf\n", v, 0.);
		else if(v<2*a) printf("%.10lf %.10lf\n", a, v-a);
		else if(v<3*a) printf("%.10lf %.10lf\n", 3*a-v, a);
		else printf("%.10lf %.10lf\n", 0., 4*a-v);
	}
	
	return 0;
}
