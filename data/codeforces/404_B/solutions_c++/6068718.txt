#include<bits/stdc++.h>
using namespace std;

int main() {
	double a, d; cin >> a >> d;
	int n; cin >> n;
	for(int i = 1; i <= n; ++i) {
		double x = fmod(i * d, 4 * a);
		if(x <= a) printf("%.9lf %.9lf\n", x, 0.0);
		else if(x <= 2 * a) printf("%.9lf %.9lf\n", a, x - a);
		else if(x <= 3 * a) printf("%.9lf %.9lf\n", 3 * a - x, a);
		else printf("%.9lf %.9lf\n", 0.0, 4 * a - x);
	}
	return 0;
}
