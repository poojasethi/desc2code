#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int main(){
  int n;
  double a,b,k;
  cin >>a>>b>>n;
  for(int i=1; i<=n; i++){
    k=fmod((i*b),(4*a));
    //cout<<k<<endl;
    if(k<=a) printf("%lf %lf\n",k,0.0000000000);
    else if(k<=2*a) printf("%lf %lf\n",a,k-a);
    else if(k<=3*a) printf("%lf %lf\n",a-(k-a-a),a);
    else if(k<=4*a) printf("%lf %lf\n",0.0000000000,a-(k-a-a-a));    
  }
}
