#include <stdio.h>
#include <math.h>

int main() {
	double a, d, cur=0;
	int n, i;
//	freopen ("in.txt", "r", stdin);
	scanf("%lf %lf", &a, &d);
	scanf("%d", &n);
	for (i=0; i<n; i++) {
		cur+=d;
		cur=fmod(cur, 4*a);
		if (cur>3*a) printf("0 %lf\n", 4*a-cur);
		else if (cur>2*a) printf("%lf %lf\n", 3*a-cur, a);
		else if (cur>a) printf("%lf %lf\n", a, cur-a);
		else printf("%lf 0\n", cur);
	}
}
