# include <bits/stdc++.h>
using namespace std;
# define fi cin
# define fo cout
int main(void)
{
    int n;
    double x,y,a,b;
    fi>>x>>y>>n;
    for (int i=1;i<=n;++i)
    {
        double r=fmod(i*y,4*x);
        if (r<=x) a=r,b=0;
        else if (r<=2*x) a=x,b=r-x;
        else if (r<=3*x) a=3*x-r,b=x;
        else             a=0,b=4*x-r;
        printf("%f %f\n",a,b);
    }
}
