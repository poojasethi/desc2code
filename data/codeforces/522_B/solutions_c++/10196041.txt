#include <iostream>

using namespace std;

int main()
{
	int n, s=0, mx1=0, mx2=0; // mx1 - the most maximum item
	cin >> n;
	int* w = new int[n];
	int* h = new int[n];
	for (int i=0; i<n; i++)
	{
		cin >> w[i] >> h[i];
		s+=w[i];
		if (h[i]>=mx1)
		{
			mx2=mx1;
			mx1=h[i];
		} else if (h[i]>=mx2)
			mx2=h[i];
	}
	for (int i=0; i<n; i++)
		cout << (s-w[i])*(h[i]==mx1 ? mx2 : mx1) << " ";
	return 0;
}
