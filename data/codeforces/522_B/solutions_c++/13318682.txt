#include <bits/stdc++.h>

int main()
{   int n,W[200005],H[200005],sw=0,m1=0,m2=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d%d",&W[i],&H[i]);  
        if(m1<=H[i]){m2=m1; m1=H[i];}
        else if(m2<H[i]) m2=H[i];
        sw+=W[i];
    } 

    for(int i=0,t;i<n;i++){
        t=H[i]==m1? m2: m1;     
        printf("%d%c",(sw-W[i])*t, (i==n-1)? '\n':' ');
    }
    return 0;
    
}
