#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define FO(i,n) for (int i=0;i<n;i++)
#define num 200005
int w[num],h[num],d[num];
int main ()
{
//	freopen("test.txt","r",stdin);
	int n,sum=0;
	cin>>n;
	FO(i,n){
		scanf("%d %d",&w[i],&h[i]);
		d[i]=h[i];
	}
	sort(d,d+n);
	FO(i,n)
		sum+=w[i];
	FO(i,n)
		printf("%d ",(sum-w[i]) * (h[i] == d[n-1]?d[n-2]:d[n-1]));
    return 0;
}
