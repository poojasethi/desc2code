#include<bits/stdc++.h>

using namespace std;

int w[200005], h[200005], n;
int m1 = 0, m2 = 0, tong = 0;


int main()
{
  	cin>>n;
  	for(int i = 1; i <= n; i++)
  	{
  		cin>>w[i]>>h[i];
  		if (h[i] > h[m1])
  			m2 = m1, m1 = i;
  		else if (h[i] > h[m2])
  			m2 = i;
  		tong += w[i];
	}
	for(int i = 1; i <= n; i++)
		if (i == m1)
			cout<<h[m2] * (tong - w[i])<<" ";
		else
			cout<<h[m1] * (tong - w[i])<<" ";
		
}
