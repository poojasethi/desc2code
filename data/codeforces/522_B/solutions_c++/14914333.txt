#include <iostream>
using namespace std;

const int MAXN = 2 * 100 * 1000 + 10;
int w[MAXN], h[MAXN];

int main()
{
	int n, W = 0, H1 = 0, H2 = 0;
	cin>> n;

	for(int i = 0; i < n; i ++)
	{
		cin>> w[i] >> h[i];
		W += w[i];

		if(H1 <= h[i])
		{
			H2 = H1;
			H1 = h[i];
		}
		else if(H2 <= h[i])
			H2 = h[i];

	}

	for(int i = 0; i < n; i ++)
	{
		int f = H1;
		if(h[i] == H1)
			f = H2;
		cout<< (W - w[i]) * f << " ";
	}

	cout<< endl;
}
