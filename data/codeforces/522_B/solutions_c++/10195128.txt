#include <bits/stdc++.h>

using namespace std;

long long n,a[200001],b[200001],c[200001],sum,mx,mxi[200001];

int main()
{
    cin>>n;
    for (int i=0; i<n; i++)
    {
        cin>>a[i]>>b[i];
        sum+=a[i];
        c[i]=b[i];
    }
    sort(c,c+n);
    for (int i=0; i<n; i++)
    {
        cout<<(c[n-1]==c[n-2]?(sum-a[i])*c[n-1]:(sum-a[i])*(b[i]==c[n-1]?c[n-2]:c[n-1]))<<" ";
    }
}
