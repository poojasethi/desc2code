#include<iostream>
using namespace std;

int a[200000],b[200000];
int n,max1,max2;
long long unsigned sumw;

int main()
{
	ios_base::sync_with_stdio(0);
	
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		sumw+=a[i];
		cin>>b[i];
		if(b[i]>max1)
		{
			max2=max1;
			max1=b[i];
		}
		else
			if(b[i]>max2)
				max2=b[i];
	}
	
	for(int i=0;i<n;i++)
		cout<<((sumw-a[i])*(max1==b[i]?max2:max1))<<" ";
	
	return 0;
}
