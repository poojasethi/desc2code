#include <iostream>

using namespace std;

int w[200000], h[200000];

int main()
{
	int n, i;
	cin >> n;
	long long summ_w = 0;
	int maxh1 = 0, maxh2 = 0;
	for (i = 0; i < n; ++i)
	{
		cin >> w[i] >> h[i];
		summ_w += w[i];
		if (h[i] >= maxh1)
		{
			maxh2 = maxh1;
			maxh1 = h[i];
		} else if (h[i] > maxh2)
		{
			maxh2 = h[i];
		}
	}
	for(i = 0; i < n; ++i)
	{
		cout << (summ_w - w[i]) * ((h[i] == maxh1) ? maxh2 : maxh1) << ' ';
	}
	return 0;
}
