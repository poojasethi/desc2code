# include<iostream>
using namespace std;
int main(){
int k,n;cin>>n;
int w[n],h[n],sumw=0,firsth=0,secondh=0;
for(k=0;k<n;k++){
cin>>w[k]; sumw+=w[k];
cin>>h[k];
if(h[k]>firsth){
  secondh=firsth; firsth=h[k];
}else if(h[k]>secondh)secondh=h[k];
}
for(k=0;k<n;k++){
h[k]==firsth?cout<<secondh*(sumw-w[k])<<'\t':cout<<firsth*((sumw-w[k]))<<'\t';
}}
