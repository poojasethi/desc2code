#include <iostream>

using namespace std;

int main()
{int ans=0,k,h=-999999,h1=-999999,n,a[500000],b[500000];
cin>>n;
for (int i=0;i<n;i++) {cin>>a[i]>>b[i];
ans+=a[i];
if (h<b[i]) {h1=h;h=b[i]; k=i;}
else {if (h1<b[i]) h1=b[i];}
}



//cout<<h<<" "<<h1<<" ";
for (int i=0;i<n;i++) {
if (k!=i) {cout<<(ans-a[i])*h<<" ";}
else cout<<(ans-a[i])*h1<<" ";}


    return 0;
}
