#include "stdio.h"

const int MAXN = 200005;

int main(){
	int n;
	int w[MAXN], h[MAXN];
	int sum=0,h1=0,h2=0;
	scanf("%d",&n);
	int a,b;
	for(int i=0;i<n;i++){
		scanf("%d%d",&a,&b);
		w[i]=a;
		h[i]=b;
		sum+=a;
		if(b>h1 && b>h2){
			h2=h1;
			h1=b;
		}else if(b>h2){
			h2=b;
		}
	}
	for(int i=0;i<n;i++){
		int hi;
		if(h[i] == h1) hi=h2;
		else hi=h1;
		printf("%d ",(sum-w[i]) * hi);
	}
	return 0;
}
