#include <iostream>

using namespace std;
long long int n,hh,sum,w[200100],h[200100],max1=0,max2=0;
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>w[i]>>h[i];
        sum+=w[i];
        if(max2<h[i])max2=h[i];
        if(max2>max1)swap(max1,max2);
    }
    for(int i=1;i<=n;i++){
        if(max1==h[i])hh=max2;
        else hh=max1;
        cout<<(sum-w[i])*hh<<" ";
    }
    
    return 0;
}
