#include<bits/stdc++.h>

using namespace std;

int n,i,a[200100],b[200100],m1,m2;
long long x,y,sum;

int main()
{
    cin>>n;
    for(i=0;i<n;++i)
    {
        cin>>a[i]>>b[i];
        if(b[i]>m1){if(m1>m2)m2=m1; m1=b[i];}
            else if(b[i]>m2)m2=b[i];
        sum+=a[i];
    }
    for(i=0;i<n;++i)
    {
        x=sum-a[i];
        y=(b[i]==m1?m2:m1);
        cout<<x*y<<' ';
    }
}
