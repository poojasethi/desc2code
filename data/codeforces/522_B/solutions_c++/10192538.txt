#include<iostream>



int main()
{
	using namespace std;
	int n,sumw,maxh,maxh2,h,ind_maxh;
	maxh=0;
	maxh2=0;
	sumw=0;
	int w[200000];
	cin>>n;
	for (int i=0;i<n;i++)
	{
		cin>>w[i]>>h;
		sumw+=w[i];
		maxh2=max(maxh2,h);
		if (h>maxh)
		{
			maxh2=maxh;
			maxh=h;
			ind_maxh=i;
		}
		
	
	}
	for (int i=0;i<n;i++)
	{
		if (i!=ind_maxh)	cout<<(sumw-w[i])*maxh<<" ";
		else cout<<(sumw-w[i])*maxh2<<" ";
	}
}
