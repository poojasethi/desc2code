#include <cstdio>
#include <algorithm>
using namespace std;
const int maxn=200002;

int n;
int h[maxn],w[maxn];
int sum;
int f[maxn][2];
int main() {
	scanf("%d",&n);
	for (int i=1;i<=n;i++) {
		scanf("%d%d",&w[i],&h[i]);
		sum+=w[i];
		f[i][0]=max(f[i-1][0],h[i]);
	}
	for (int i=n;i;i--)
		f[i][1]=max(f[i+1][1],h[i]);

	for (int i=1;i<=n;i++)
		printf("%d%c",max(f[i-1][0],f[i+1][1])*(sum-w[i]),i==n?10:32);
	return 0;
}
