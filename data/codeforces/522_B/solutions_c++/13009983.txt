#include<cstdio>
#include<algorithm>
using namespace std;
int a1[200000],a2[200000];
int main()
{
    int n,a=0,b=0,tt=0;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
    {
        scanf("%d%d",a1+i,a2+i);
        tt+=a1[i];b=max(b,a2[i]);
        if(b>a)swap(a,b);
    }
    for(int i=0;i<n;++i)
    {
        printf("%d",(tt-a1[i])*(a2[i]==a?b:a));
        putchar(i==n-1?'\n':' ');
    }
}
