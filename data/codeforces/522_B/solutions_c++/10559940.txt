#include<iostream>
using namespace std;

int main()
{
long int i,posmax,w[200000],n,max1=0,max2=0,h,sumw=0,d;
cin>>n;
for(i=0;i<n;i++)
{
cin>>w[i]>>h;
sumw+=w[i];

if(h>max1)
{
max2=max1;
max1=h;
posmax=i;
}
else if(h>max2)
{
max2=h;
}


}


for(i=0;i<n;i++)
{
if(i!=posmax)
{
d=sumw-w[i];
cout<<max1*d<<' ';
}
else
{
d=sumw-w[i];
cout<<max2*d<<' ';
}

}


return 0;
}
