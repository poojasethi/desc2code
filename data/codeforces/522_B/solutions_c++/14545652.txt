#include <bits/stdc++.h>

using namespace std;

struct A { int w, h; } a[212345];
int main()
{
	ios_base::sync_with_stdio(false);

	int n;
	cin>>n;
	
	int W = 0 , H = 0 , H2 = 0;
	for (int i=1 ; i<=n ; i++)
	{
		cin>>a[i].w>>a[i].h;
		W += a[i].w;
		if (a[i].h >= H) 		H2 = H , H = a[i].h;
		else if (a[i].h > H2) 	H2 = a[i].h;
	}
	
	for (int i=1 ; i<=n ; i++)
		cout<<( (W-a[i].w) * (a[i].h==H ? H2 : H) )<<" ";
	
	return 0;
}
