#include<bits/stdc++.h>
using namespace std;
int h[200005],w[200005];
int main()
{
	int n,sum=0;
	map<int,int> mp;
	map<int,int>::iterator it;
	cin>>n;
	for(int i=1;i<=n;i++)
	cin>>w[i]>>h[i],sum+=w[i],mp[h[i]]++;
	for(int i=1;i<=n;i++)
	{
		mp[h[i]]--;
		it=mp.end();
		it--;
		if(it->second==0)
		it--;
		int ans=(sum-w[i])*(it->first);
		mp[h[i]]++;
		cout<<ans<<" ";
	}
}
