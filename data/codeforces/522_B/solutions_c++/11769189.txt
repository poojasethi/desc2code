#include <bits/stdc++.h>
using namespace std;

int main() {
int i,j,k,n,sum=0,y;
vector<int> adj[1001];
int a[2000000];
scanf("%d",&n);
for(i=0;i<n;i++) {
	scanf("%d%d",a+i,&y);
	sum+=a[i];
	adj[y].push_back(i);
}
long long val;
for(i=0;i<n;i++)
{
 val=sum-a[i];
for(j=1000;j;j--) for(k=0;k<(int)adj[j].size();k++)
	if(adj[j][k]!=i) {val*=j;goto asd;}
asd:;
printf("%I64d ",val);
}
return 0;
}
