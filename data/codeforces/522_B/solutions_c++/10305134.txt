#include <iostream>

using namespace std;

int main()
{
	int n, *w, *h, sumw, h1, h2, i;

	cin>>n;

	w = new int[n];
	h = new int[n];

	sumw = h1 = h2 = 0;

	for (i = 0; i < n; i++)
	{
		cin>>w[i]>>h[i];

		if (h[i] >= h1)
		{
			h2 = h1;
			h1 = h[i];
		}
		else if (h[i] > h2)
		{
			h2 = h[i];
		}

		sumw += w[i];
	}

	for (i = 0; i < n; i++)
	{
		cout<<((sumw-w[i]) * (h[i]<h1 ? h1 : h2))<<" ";
	}
	cout<<endl;

	return 0;
}
