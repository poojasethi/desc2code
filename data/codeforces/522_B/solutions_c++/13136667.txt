#include<cstdio>

int n, tot_w, top, sec;
int w[222222], h[222222];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d%d", &w[i], &h[i]);
		tot_w += w[i];
		if (h[i] > top) {sec = top; top = h[i];}
		else if(h[i] > sec) sec = h[i];
	}
	
	for(int i=0;i<n;i++) 
		printf("%d%c", (tot_w - w[i]) * (h[i] == top ? sec : top), i<n-1 ? ' ' : '\n');
	
	return 0;
}
