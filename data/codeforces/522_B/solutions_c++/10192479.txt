#include <iostream>
#include <vector>
#include <set>
#define pii pair<int,int>

using namespace std;

int w[200010];
set<pii> s;

int main() {
	int n,h;
	long long W=0;
	cin>>n;
	for(int i=0;i<n;++i) {
		cin>>w[i]>>h;
		W+=w[i]; s.insert(pii(-h,i));
	}
	for(int i=0;i<n;++i) {
		set<pii>::iterator it=s.begin();
		if(it->second==i) it++;
		long long sz=(W-w[i])*(-it->first);
		cout<<sz<<" ";
	}
	
	
	return 0;
}
