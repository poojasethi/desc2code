#include <iostream>
#include <algorithm>
using namespace std;
int n,i,j,H,W,w[200005],h[200005],x[200005],k;

int main(){
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>w[i]>>h[i];
        x[i]=w[i]*h[i];
        W+=w[i];
    }
    sort(h+1,h+n+1);
    for(i=1;i<=n;i++)
        if(w[i]*h[n]!=x[i])
            cout<<(W-w[i])*h[n]<<endl;
        else
            cout<<(W-w[i])*h[n-1]<<endl;
    return 0;
}
