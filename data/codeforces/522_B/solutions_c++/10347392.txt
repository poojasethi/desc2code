#include<bits/stdc++.h>
using namespace std;
int main(){
int n;
cin>>n;
long long t=-1e18,b[n+1],s=0,k=0;
int w[n+1],h[n+1];
for(int i=1; i<=n; i++){
    cin>>w[i]>>h[i];
    s+=w[i];
    k++;
    b[k]=h[i];
}
sort(b+1,b+1+n);
for(int i=1; i<=n; i++){
    if(h[i]!=b[n]){
    cout<<b[n]*(s-w[i])<<" ";
    }
    if(h[i]==b[n]){
    cout<<b[n-1]*(s-w[i])<<" ";
    }
}
}
