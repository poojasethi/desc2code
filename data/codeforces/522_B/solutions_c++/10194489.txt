#include <bits/stdc++.h>
using namespace std;
long long n,w[10000001],h[10000001],i,j,m1=0,m2=0,k=0,l;
int main()
{
 cin>>n;
 for(i=1;i<=n;i++)
 {
    cin>>w[i]>>h[i];
   k+=w[i];
   if(h[i]>m1)
   {
       m2=m1;
       m1=h[i];
       l=i;
   }
   else
    if(m2<h[i])
    m2=h[i];
 }
 for(i=1;i<=n;i++)
    if(i!=l)
      cout<<(k-w[i])*m1<<" ";
    else
      cout<<(k-w[i])*m2<<" ";
 return 0;
}
