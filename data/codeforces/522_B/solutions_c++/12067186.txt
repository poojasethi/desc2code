#include<stdio.h>
#include<math.h>
int n,i,j,ww,h1,h2;
int w[200005],h[200005];
int main()
{
	scanf("%d",&n);
	h1=0;
	h2=0;
	for(i=1;i<=n;i++)
	{
		scanf("%d %d",&w[i],&h[i]);
		ww=ww+w[i];
		if(h[i]>=h1)
		{
			h2=h1;
			h1=h[i];
		}
		if((h[i]>h2)&&(h[i]<h1)) h2=h[i];
	}
	for(i=1;i<=n;i++)
	{
		if(h[i]==h1) printf("%d ",(ww-w[i])*h2);
		if(h[i]<h1) printf("%d ",(ww-w[i])*h1);
	}
	return 0;
}
