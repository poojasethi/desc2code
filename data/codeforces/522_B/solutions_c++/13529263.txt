#include <cstdio>
#include <algorithm>

using namespace std;

int w[200000],h[200000];
int sum,max1,max2;
int n;

int main(void) {
	scanf("%d",&n);
	for(int i=0; i<n;++i) {
		scanf("%d %d",&w[i],&h[i]);
		sum+=w[i];
		if(h[i]>h[max1]) max1=i;
	}
	for(int i=0;i<n;++i)
		if(max1==max2 || (h[i]>h[max2] && i!=max1))
			max2 = i;
	for(int i=0;i<n;++i)
		printf("%d ",(sum-w[i])*((i==max1)?h[max2]:h[max1]));
	return 0;
}
