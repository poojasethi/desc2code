#include<cstdio>
#define N 200005
using namespace std;
int w[N],h[N],n,i,max1,max2,sum;
long long ans;
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
  {
    scanf("%d%d",&w[i],&h[i]);
    sum+=w[i];
    if (h[i]>max1) max2=max1,max1=h[i];
    else if (h[i]>max2) max2=h[i];
  }
  for (i=1;i<=n;i++)
  {
    if (h[i]==max1) ans=max2*1ll*(sum-w[i]);
    else ans=max1*1ll*(sum-w[i]);
    printf("%I64d ",ans);
  }
}
