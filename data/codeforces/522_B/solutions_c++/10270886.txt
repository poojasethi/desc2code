#include<iostream>
#include<algorithm>
using namespace std; 
int a[199999],b[199999],k,j,n,i,mx,d,l;
int main()
{

cin>>n;
for(i=0;i<n;i++)
{
cin>>a[i]>>b[i];
k+=a[i];
if(b[i]>d)
{
	j=d;d=b[i];
	l=i;
}
else if(j<b[i])j=b[i];
}
for(i=0;i<n;i++)
{
if(l!=i)
{
	cout<<d*(k-a[i])<<" ";
}
else cout<<j*(k-a[i])<<" ";}
return 0;
}
