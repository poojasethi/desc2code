#include<iostream>

using namespace std;

int main()
{
ios_base::sync_with_stdio(false);

int n,w[200000],h[200000],i,H1=0,j,H2=0,W=0;
cin>>n;
for(i=0;i<n;i++)
{
cin>>w[i]>>h[i];
W+=w[i];
if(H1<h[i])
{
H1=h[i];
j=i;
}
}

for(i=0;i<n;i++)
if((i!=j)&&(h[i]>H2))
H2=h[i];

for(i=0;i<n;i++)
{
int c;
long b;
if(i!=j)
c=H1;
else
c=H2;
b=(W-w[i])*c;
cout<<b<<" ";
}


}
