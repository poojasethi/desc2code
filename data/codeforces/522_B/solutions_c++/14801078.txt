#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define s(n) scanf("%llu",&n)
#define p(n) printf("%llu ",n)
ll w[200005],h[200005];
int main()
{ll n,m=0,i,sum=0,pos=0;
s(n);
for(i=1;i<=n;++i){ s( w[i] );sum+=w[i];s(h[i]);if(m<h[i]){ m=h[i],pos=i;}}
sort(h,h+n+1,greater<ll>());
for(i=1;i<=n;++i){
if(i==pos)
cout<<( sum-w[i] )*h[1]<<" ";
else cout<<( sum-w[i] )*h[0]<<" ";}
}
