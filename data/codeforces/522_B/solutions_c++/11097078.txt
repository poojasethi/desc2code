#include<iostream>
using namespace std;
int n,b,i,j,w[200000],h[200000],max1=0,max2=0,total=0,total2=0,z=0;
int main()
{
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>w[i]>>h[i];
		total+=w[i];
		if(max1<h[i])
		{
			max2=max1;
			max1=h[i];
		}
		else if(max2<h[i]) max2=h[i];
	}
	for(i=0;i<n;i++)
	{
		if(h[i]==max1)z=max2;
		else z=max1;
		cout<<(total-w[i])*(z)<<' ';
	}
	return 0;
}
