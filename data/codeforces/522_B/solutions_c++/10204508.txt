#include<iostream>
using namespace std;
int h[200010],w[200010],n,m,m1,buf,sum,ma;
int main () {
	cin>>n;
	for (int i=0;i<n;i++) {
		cin>>w[i]>>h[i];
		sum+=w[i];
		if (h[i]>=m) {
			m=h[i];
			buf=i;
		}
	}
	for (int i=0;i<n;i++) {
		if (h[i]>=m1 && i!=buf) 
			m1=h[i];
	}
	for (int i=0;i<n;i++) {
		if (i==buf)
			ma=m1;
		else 
			ma=m;
		cout<<(sum-w[i])*ma<<" ";
	}
return 0;
}
