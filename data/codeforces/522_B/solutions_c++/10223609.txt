#include <bits/stdc++.h>
using namespace std;
int main ()
{
int long long a,b[200099],c[200099],d=0,d1=0,d2=0,i,j;
cin>>a;
for (i=1;i<=a;i++)
{
cin>>b[i]>>c[i];
if( c[i]>d1 ){
    d2=d1;
    d1=c[i];
}
else if( c[i]>d2 )d2=c[i];
d=d+b[i];
}
for (i=1;i<=a;i++)
{
if (c[i]==d1) {cout<<(d-b[i])*d2<<" ";}
else {cout<<(d-b[i])*d1<<" ";}
}
}
