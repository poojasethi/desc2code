n = int(raw_input())

w = [0] * n
h = [0] * n
sumw = 0
mx1 = 0
mx2 = 0

for i in range(0, n):
	x, y = raw_input().split()
	
	w[i] = int(x)
	h[i] = int(y)
	
	sumw += w[i]
	
	if h[i] >= mx1:
		mx2 = mx1
		mx1 = h[i]
	elif h[i] > mx2:
		mx2 = h[i]
		
ans = ""

for i in range(0, n):
	t = 0
	
	if h[i] == mx1:
		t = mx2
	else:
		t = mx1
		
	ans += str((sumw-w[i]) * t) + ' '
	
print ans
