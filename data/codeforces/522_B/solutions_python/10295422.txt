#!/usr/bin/python

import sys

n = int(sys.stdin.readline())

width = []
height = []
max_height = []
width_sum = 0

for i in xrange(n):
  l = sys.stdin.readline().rstrip()
  parts = l.split(" ")
  w = int(parts[0])
  h = int(parts[1])
  width.append(w)
  height.append(h)
  width_sum += w
  max_height.append(h)
  max_height.sort()
  if len(max_height) == 3:
    max_height = max_height[1:]

for i in xrange(n):
  h = max_height[1]
  if height[i] == h:
    h = max_height[0]
  area = (width_sum - width[i]) * h
  sys.stdout.write(str(area))
  sys.stdout.write(" ")
