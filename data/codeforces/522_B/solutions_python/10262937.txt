n = int(raw_input())

W = []
H = []

for i in xrange(n):
    w, h = map(int, raw_input().strip().split())
    W.append(w)
    H.append(h)

W_all = sum(W)
H_max = max(H)
max_index = H.index(H_max)
H2 = H[:]
H2[max_index] = 0
H2_max = max(H2)

results = []
for i in xrange(n):
    if i == max_index:
        results.append( (W_all - W[i]) * H2_max )
    else:
        results.append( (W_all - W[i]) * H_max )

print ' '.join(map(str, results))
