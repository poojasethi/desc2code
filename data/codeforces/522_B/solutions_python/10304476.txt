n = input()
w,h,s = [],[],0
for _ in xrange(n):
    x,y = map(int,raw_input().split())
    w.append(x); h.append(y); s+=x
m = sorted(h,reverse=True)[:2]
if m[0] == m[1]:
    for i in xrange(n): print (s-w[i])*m[0],
else:
    for i in xrange(n): 
        if h[i] == m[0]: print (s-w[i])*m[1],
        else: print (s-w[i])*m[0],
