R=lambda:map(int,raw_input().split(' '))
n = input()
w = 0
h = 0
h2 = 0
ind = 0
m = []
for i in xrange(n):
    a,b = R()
    m.append( (a,b) )
    w += a
    if b >= h:
        h2 = h
        h = b
        ind = i
    if b >= h2 and ind != i:
        h2 = b

if h2 == 0: h2 = m[1][1]
    
for i in xrange(n):
    if m[i][1] == h:
        print (w - m[i][0]) * h2
    else:
        print (w - m[i][0]) * h
