n = int(input())
w = []
h = []

for i in range(n):
    var1, var2 = raw_input().split()
    var1, var2 = [int(var1), int(var2)]
    w.append(int(var1))
    h.append(int(var2))
    
photo = 0
hmax = max(h)
wsum = sum(w)
ind_max = h.index(hmax)
hsecond_max = max(h[:ind_max]+h[(ind_max+1):])

for i in range(n):
    if h[i] < hmax:
        photo = hmax*(wsum - w[i])
    else: # == hmax
        photo = hsecond_max*(wsum - w[i])
    print photo
