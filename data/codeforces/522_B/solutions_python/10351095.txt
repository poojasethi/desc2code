import sys
import math

def get(f=int):
    return map(f, raw_input().split())

def main():
    n = input()
    friends = []
    width = 0
    h1 = h2 = 0
    for _ in range(n):
        w, h = get()
        width += w
        if h > h2:
            if h >= h1:
                h1, h2 = h, h1
            else:
                h2 = h

        friends.append((w, h))

    for w, h in friends:
        print (width - w) * (h1 if h != h1 else h2),

if __name__ == '__main__':
    main()
