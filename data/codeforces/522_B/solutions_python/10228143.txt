n = map(int, raw_input().split())
w, h = list(), list()
h_1, h_2, sum_ = 0, 0, 0

for i in range(n[0]):
	input_ = map(int, raw_input().split())
	w.append(input_[0])
	h.append(input_[1])

	sum_ = sum_ + input_[0]

	if input_[1] > h_1:
		h_2 = h_1
		h_1 = input_[1]
	elif input_[1] > h_2:
		h_2 = input_[1]

for i in range(n[0]):
	if h[i] != h_1:
		print (sum_ - w[i]) * h_1, 
	else:
		print (sum_ - w[i]) * h_2,
