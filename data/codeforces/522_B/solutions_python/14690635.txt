n = int(raw_input())
w = n * [0]
h = n * [0]
frst, scnd = 0, 0
for i in range(n):
    w[i], h[i] = map(int, raw_input().split())
    if h[i] > frst:
        scnd = frst
        frst = h[i]
    elif h[i] > scnd:
        scnd = h[i]
ch = h.count(frst)
sw = sum(w)
cx = h.index(frst) if ch == 1 else -1
for i in range(n):
    print (sw - w[i]) * (scnd if i == cx else frst),
