n = int(raw_input())
w, h = [], []
for _ in xrange(n):
    w1, h1 = map(int, raw_input().split())
    w.append(w1)
    h.append(h1)

s = sum(w)
h1, h2 = -1, -1
for i in h:
    if h1 <= i:
        h2 = h1
        h1 = i
    elif h2 < i:
        h2 = i
for width, height in zip(w, h):
    print (s - width) * (h1 if height != h1 else h2),
print
