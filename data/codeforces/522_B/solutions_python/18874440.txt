n = int(raw_input())
ws = []
hs = []
for i in range(n):
	w, h = [int(i) for i in raw_input().split()]
	ws.append(w)
	hs.append(h)
hs_sorted = sorted(hs)
dic = {}
for i in range(n):
	if i == n - 1:
		dic[hs_sorted[i]] = hs_sorted[n - 2]
	else:
		dic[hs_sorted[i]] = hs_sorted[n - 1]
sum_ws = sum(ws)
result = ""
for i in range(n):
	result += str((sum_ws - ws[i]) * dic[hs[i]]) + " "
print result
