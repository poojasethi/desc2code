from operator import itemgetter
FriendsNum=input()
DataFriends=[]
Answer=[]

for i in xrange(FriendsNum):
    DataFriends.append(map(int,raw_input().split()))
TmpFriends=[x[1] for x in DataFriends]

SumWidth=sum([x[0] for x in DataFriends])

MaxHeight=max(TmpFriends)
MaxHeightIndex=TmpFriends.index(MaxHeight)
TmpFriends.pop(MaxHeightIndex)
MaxHeightSecond=max(TmpFriends)

# print MaxHeight
# print MaxHeightSecond
# print MaxHeightIndex
# MaxHeight=max([x[1] for x in DataFriends])


for i in xrange(FriendsNum):
    TmpWidth=SumWidth-DataFriends[i][0]
    if i==MaxHeightIndex:
        TmpHeight=MaxHeightSecond
    else:
        TmpHeight=MaxHeight
    # print TmpHeight
    Answer.append(str(TmpWidth*TmpHeight))

print ' '.join(Answer)
