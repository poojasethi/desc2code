n = int(raw_input())
maxh = 0
maxn = -1
max2h = 0
sumw = 0
lst = list()
for i in range(n):
    m = raw_input()
    lst.append(m.split())
    sumw += int(m.split()[0])
    if maxh<int(m.split()[1]):
        maxp = maxh
        maxh = int(m.split()[1])
        maxn = i
    elif max2h<int(m.split()[1]): max2h = int(m.split()[1])
    if maxp>max2h: max2h = maxp
for i in range(n):
    if i!=maxn: print str((sumw-int(lst[i][0]))*maxh)+' ',
    else: print str((sumw-int(lst[i][0]))*max2h)+' ',
