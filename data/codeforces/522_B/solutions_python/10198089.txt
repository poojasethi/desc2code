import sys
num = int(raw_input())
sum, imax, max1, max2, w = 0, 0, 0, 0, bytes()
for i in range(0,num):
    wi, hi = raw_input().split()
    w += bytes(str(int(wi)-1))
    h = int(hi)
    sum += int(wi)
    if h > max1:
        max2 = max1
        max1 = h
        imax = i
    elif max2==0 or h > max2:
        max2 = h
for i in range(0,num):  
    sys.stdout.write(str((sum-int(w[i])-1)*(max1 if i != imax else max2))+' ')
