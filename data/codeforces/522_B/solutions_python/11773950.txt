n=int(raw_input())
width=0
h1=[0,0]
h2=[0,0]
l=[]
for i in range(n):
    w,h=map(int,raw_input().split())
    width+=w
    if h>h1[0]:
        h2=[h1[0],h1[1]]
        h1=[h,i]
    elif h>h2[0]:
        h2=[h,i]
    l.append((w,h))

for i,j in enumerate(l):
    if i==h1[1]:
        print (width-j[0])*h2[0],
    else:
        print (width-j[0])*h1[0],
