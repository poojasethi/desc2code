from __future__ import print_function
n = input()
w = [None] * n
h = [None] * n
imax = 0
hmax = None
wsum = 0
for i in xrange(n):
    (w[i],h[i]) = [int(x) for x in raw_input().split(' ')]
    if h[i]>hmax:
        imax = i
        hmax = h[i]
    wsum+=w[i]
hpremax = max([max(h[:imax]+[None]),max(h[imax+1:]+[None])])
for i in xrange(n):
   print((wsum-w[i])*(hmax if h[i]< hmax else hpremax), end=' ')
