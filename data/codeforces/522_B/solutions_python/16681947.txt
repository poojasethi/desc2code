def solve():
	n = input()
	m = list()
	max1 = max2 = -100500
	s = 0
	for _ in range(n):
		a, b = raw_input().split(' ')
		m.append([int(a), int(b)])
		s += int(a)
		if (int(b) >= max1):
			max2 = max1
			max1 = int(b)
		elif max1 > int(b) > max2:
			max2 = int(b)

	ans = list()
	for a, b in m:
		if b != max1:
			ans.append((s - a) * max1)
		else:
			ans.append((s - a) * max2)

	print " ".join(map(str, ans))

solve()
