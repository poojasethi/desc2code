n = int(raw_input())
maxh = -1
premaxh = -1
sum = 0
ws = []
hs = []
for i in xrange(n):
    w, h = map(int, raw_input().split())
    ws.append(w)
    hs.append(h)
    sum += w
    if h >= maxh:
        premaxh = maxh
        maxh = h
    elif h > premaxh:
        premaxh = h

for i in xrange(n):
    curw = sum - ws[i]
    curh = maxh
    if hs[i]==maxh:
        curh = premaxh
    if i < n-1:
        print curw*curh,
    else:
        print curw*curh
