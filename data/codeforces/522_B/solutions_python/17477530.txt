#!/usr/bin/env python

from collections import deque as dq

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()

ww = []
hh = []
W = 0
H1 = H2 = -1
for i in range(n):
    w, h = ia()
    ww.append(w)
    hh.append(h)
    W = W + w
    if h>=H1:
        H1, H2 = h , H1
    elif h>=H2:
        H2 = h

for w, h in zip(ww, hh):
    w0 = W - w
    if h<H1:
        h0 = H1
    else:
        h0 = H2
    print w0*h0
