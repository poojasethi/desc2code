
import sys
from collections import namedtuple

Photo = namedtuple("Photo", ["height", "width"])


def widths(photos):
    maxHeight = 0
    secondHeight = 0
    maxIndex = 0
    cumWidth = 0

    for i, p in enumerate(photos):
        if p.height >= maxHeight:
            secondHeight = maxHeight
            maxHeight = p.height
            maxIndex = i
        if p.height > secondHeight and p.height < maxHeight:
            secondHeight = p.height
        cumWidth += p.width
    result = []

    for i, p in enumerate(photos):
        if i != maxIndex:
            result.append((cumWidth-p.width)*(maxHeight))
        else:
            result.append((cumWidth-p.width)*(secondHeight))
    return result


def main():
    photos = []

    for i, line in enumerate(sys.stdin):
        if i < 1:
            continue
        width, height = line.split()
        photos.append(Photo(height=int(height), width=int(width)))
    print " ".join([str(x) for x in widths(photos)])


if __name__ == "__main__":
    main()
