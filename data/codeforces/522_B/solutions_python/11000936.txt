
n = input() 

P = [map(int, raw_input().split()) for _ in range(n)] 

W = [p[0] for p in P]
H = [p[1] for p in P]

width_sum = sum(W)

L = [0] * n
R = [0] * n

for i in range(1, n): 
    if H[i-1] > L[i-1]: 
        L[i] = H[i-1]
    else:
        L[i] = L[i-1]

for i in range(n-2, -1, -1): 
    if H[i+1] > R[i+1]: 
        R[i] = H[i+1]
    else:
        R[i] = R[i+1]

for i in range(n): 
    w = width_sum - W[i]
    print w * max(L[i], R[i]), 

print ''
