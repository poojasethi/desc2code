n = int(input())

W = []
H = []  
for i in range(n):
    string = raw_input().split(' ')
    W.append(int(string[0]))
    H.append(int(string[1]))

all_w = sum(W)

heights = list(H)

H.sort()
m_one = H[-1]
m_two = H[-2]
top_index = heights.index(m_one)

result = []
for i in range(n):
    sqale1 = all_w - W[i]

    if i == top_index: #m_one != m_two
        result.append(str(m_two * sqale1))
    else:
        result.append(str(m_one * sqale1))
print (" ".join(result))
