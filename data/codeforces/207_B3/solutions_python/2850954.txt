const int N=2*(250000+10);

int n, a[N], lg[N];
struct T{
	int dis, id;
	bool operator<(const T &b)const{
		return dis<b.dis;
	}
} st[20][N];
int f[20][N];

void init(int n){
	for(int i=1, j=0; i<=n; i++){
		if ( (1<<(j+1)) == i )
			j++;
		lg[i]=j;
	}
	Rep(i,1,n)
		st[0][i]=(T){max(i-a[i],1), i};
	RepC(d,1,lg[n])
		RepC(i,1,n-(1<<d)+1)
			st[d][i]=min(st[d-1][i], st[d-1][i+(1<<(d-1))]);
			
	Rep(i,1,n){
		int l=max(i-a[i],1), r=i, d=lg[r-l+1];
		f[0][i]=min(st[d][l], st[d][r-(1<<d)+1]).id;
	}
	RepC(d,1,lg[n])
		Rep(i,1,n)
			f[d][i]=f[d-1][f[d-1][i]];
}

int main(){
	
	cin>>n;
	Rep(i,1,n)
		scanf("%d",a+i), a[i+n]=a[i];
	
	init(2*n);	
	
	LL ans=0;
	Rep(i,n,2*n-1)
		if (st[0][i].dis<=i-n+1)
			ans++;
		else{
			int x=i;
			Repd(d,19,0)
				if ( st[0][f[d][x]].dis > i-n+1 ){
					x=f[d][x];
					ans+=1<<d;
				}
			ans+=2;
		}
	
	cout<<ans<<endl;

    return 0;
}
