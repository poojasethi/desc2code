#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n, m = ia()

a = [0 for i in range(m)]

def solve(n):
    while True:
        for i in range(m):
            a[i]+=1; n-=1
            if n==0: return

solve(n)            
a.sort()            
print ' '.join(map(str, a))
