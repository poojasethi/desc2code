# Codeforces Testing Round 6
# Problem A: Candies

def candies(n, m):
    partitions = []
    for i in xrange(m):
        partitions += [n/m]
    for j in xrange(n % m):
        partitions[m-1-j] += 1
    return partitions

x = raw_input("").split(' ')
n = int(x[0])
m = int(x[1])
answer = candies(n, m)
output = ""
for i in xrange(len(answer)):
    output += str(answer[i]) + " "
print output
