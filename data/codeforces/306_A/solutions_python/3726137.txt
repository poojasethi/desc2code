#!/usr/bin/env python
# -*- coding: UTF-8 -*-

import sys

if __name__ == '__main__':
    try:
        while True:
            n, m = [int(x) for x in raw_input().split()]
            if n % m:
                lower = n // m
                upper = lower + 1
                upperN = n % m
                lowerN = m - upperN
                print ' '.join([str(lower)] * lowerN + [str(upper)] * upperN)
            else:
                print ' '.join([str(n // m)] * m)

    except EOFError, e:
        pass
