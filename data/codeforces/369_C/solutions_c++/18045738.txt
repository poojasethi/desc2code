#include<bits/stdc++.h>
using namespace std;

int n,color[100005],d[100005];
vector<int>a[100005];
vector<int>ans;

void func(int u, int prev){
	int v;
	d[u]=color[u];
	for(v=0; v<a[u].size(); v++){
		if(a[u][v]==prev)
			continue;
		func(a[u][v],u);
		d[u]+=d[a[u][v]];
	}
}

void print(){
	for(int i=0; i<n; i++){
		if(color[i]==1 && d[i]==1)
			ans.push_back(i+1);
	}
	cout << ans.size() << endl;
	for(int i=0; i<ans.size(); i++)
		cout << ans[i] << " ";
	cout << endl;
}

main(){
	cin >> n;
	int u,v,w,i=n-1;
	while(i--){
		cin >> u >> v >> w;
		u--; v--;
		a[u].push_back(v);
		a[v].push_back(u);
		if(w==2){
			color[u]=1;
			color[v]=1;
		}
	}
	func(0,-1);
	print();
	return 0;
}
