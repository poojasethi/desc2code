#include "stdio.h"
#include "vector"
using namespace std;

vector<pair<int,int> > edge[100005];
vector<int> ans;
bool visited[100005];
int n;

bool dfs(int x,int mark){
	visited[x] = 1;
	bool tmp = 0;
	for(int i=0;i<edge[x].size();i++){
		if(!visited[edge[x][i].first]){
			tmp |= dfs(edge[x][i].first,edge[x][i].second);
		}
	}
	if(tmp) return 1;
	if(mark==2){
		ans.push_back(x);
		return 1;
	}else return 0;
}

int main(){
	scanf("%d",&n);
	for(int i=0;i<n-1;i++){
		int a,b,c;
		scanf("%d%d%d",&a,&b,&c);
		edge[a].push_back(make_pair(b,c));
		edge[b].push_back(make_pair(a,c));
	}
	dfs(1,1);
	printf("%d\n",ans.size());
	for(int i=0;i<ans.size();i++){
		printf("%d ",ans[i]);
	}
}
