#include<iostream>
#include<vector>

using namespace std;

int viz[100005];
int i,a,b,x,n;
vector<pair<int, int > > g[100005];
vector<int> ans;

int dfs(int t, int dead){
int a=0;

viz[t]=1;
for(int i=0; i<g[t].size(); i++){
if(viz[g[t][i].first]==0){
int k=dfs(g[t][i].first,g[t][i].second);
if(k==1) a=1;
}
}
if(dead==2 && a==0) ans.push_back(t);

if (dead==2 || a==1) return 1; else return 0;

}

main()
{
cin>>n;
for(i=0; i<n-1; i++){
cin>>a>>b>>x;
g[a].push_back(make_pair(b,x));
g[b].push_back(make_pair(a,x));
}
dfs(1,1);

cout<<ans.size()<<endl;

for(i=0; i<ans.size(); i++)
cout<<ans[i]<<" ";


//system("pause");
}
