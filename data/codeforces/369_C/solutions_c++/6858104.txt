//Valera and Elections
#include <iostream>
#include <vector>

using namespace std;

const int maxn=1e5+10;

int n,p[maxn];
vector <int> adj[maxn],ans;

int dfs(int now,int pre)
{

	int t=0;
	for(int i=0;i<adj[now].size();i++)
	{
		if(adj[now][i]!=pre)
		{
 			int s=dfs(adj[now][i],now);
 			if(s==1) t=s;
		}
	}
	if(t==1)
		return 1;
	if(t==0 && p[now]==2)
	{
		ans.push_back(now);
		return 1;
	}
	return 0;
}

int main(void)
{
	cin>>n;

	for(int i=1;i<n;i++)
	{
		int a,b,c;
		cin>>a>>b>>c;
		adj[a].push_back(b);
		adj[b].push_back(a);
		if(p[a]!=2) p[a]=c;
		if(p[b]!=2) p[b]=c;
	}
	dfs(1,1);
	int t=ans.size();
	cout<<t<<endl;
	for(int i=0;i<t;i++)
	{
		cout<<ans[i]<<" ";
	}
	return 0;
}
