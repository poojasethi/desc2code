#include "bits/stdc++.h"
using namespace std;
typedef long long LL;
vector<pair<int, int> > g[100001];
vector<int> ans;
int dfs(int cur, int pre, int broken) {
	int repaired = 0;
	for (size_t i = 0; i < g[cur].size(); i++) {
		if (g[cur][i].first == pre)
			continue;
		repaired |= dfs(g[cur][i].first, cur, g[cur][i].second);
	}
	if (broken && 0 == repaired) {
		ans.push_back(cur);
		repaired = 1;
	}
	return repaired;
}
int main() {
	int n, x, y, t;
	cin >> n;
	for (int i = 1; i < n; i++) {
		cin >> x >> y >> t;
		g[x].push_back(make_pair(y, t - 1));
		g[y].push_back(make_pair(x, t - 1));
	}
	dfs(1, 0, 0);
	cout << ans.size() << endl;
	for (size_t i = 0; i < ans.size(); i++)
		cout << ans[i] << " ";
	cout << endl;
	return 0;
}
