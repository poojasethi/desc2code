#include <iostream>
#include <vector>
using namespace std;
vector <int> v[100100],ans;
int n,m[100100],d[100100];
void poisk(int k,int p){
    //cout<<k<<endl;
    for(int i=0;i<v[k].size();i++){
        if(v[k][i]!=p){
            poisk(v[k][i],k);
            m[k]+=m[v[k][i]];
        }
    }
   if(m[k]==1)ans.push_back(k+1);
}
int main()
{
    cin>>n;
    for(int i=1;i<n;i++){
        int x,y,c;
        cin>>x>>y>>c;
        x--;y--;
        if(c){
            v[x].push_back(y);
            v[y].push_back(x);
            if(c==2)m[x]=m[y]=1;
        }
    }
    poisk(0,-1);
    cout<<ans.size()<<endl;
    for(int i=0;i<ans.size();i++){
        cout<<ans[i]<<" ";
    }




}
