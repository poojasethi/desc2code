#include <iostream>
#include <vector>
using namespace std;
const int N=100100;
int n;
vector<int> g[N],t;
int go(int u=1,int f=0,int c=0){
  int s=0;
  for(int i=0,v;i<g[u].size();++i)if((v=g[u][i]/2)!=f)s+=go(v,u,g[u][i]%2);
  if(c&&!s)t.push_back(u);
  return s+c;
}
int main(){
  cin>>n;
  for(int i=1,a,b,c;i<n&&cin>>a>>b>>c;++i)g[a].push_back(b*2+c-1),g[b].push_back(a*2+c-1);
  go();
  cout<<t.size()<<endl;
  for(int i=0;i<t.size();++i)cout<<t[i]<<" ";
}
