#include <cstdio>
#include <vector>

#define N 100000

using namespace std;

struct z
{
	int y, t;
} p;

int b[N + 3], w[N + 3], n, i, x, y, t, k = 0;
vector <z> a[N + 3];

void go(int v, bool f = false)
{
	int c = k;
	b[v] = 1;
	for (int i = 0; i < a[v].size(); i ++)
		if (!b[a[v][i].y])
			go(a[v][i].y, a[v][i].t);
	if (f && c == k)
		w[k ++] = v;
}

int main()
{
	scanf("%d", &n);

	for (i = 1; i < n; i ++)
	{
		scanf("%d%d%d", &x, &y, &t);
		p.y = y;
		p.t = t - 1;
		a[x].push_back(p);
		p.y = x;
		a[y].push_back(p);
	}

	go(1);

	printf("%d\n", k);
	for (i = 0; i < k; i ++)
		printf("%d ", w[i]);

	return 0;
}
