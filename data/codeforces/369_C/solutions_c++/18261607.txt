#include<bits/stdc++.h>
using namespace std;
int n;
const int N = 400000;
int white[N], d[N];
vector<int> adj[N];

void findit(int v, int p=-1) {
	d[v]=white[v];
	for(int i=0;i<adj[v].size();i++) {
		int to = adj[v][i];
		if(p==to)
			continue;
		findit(to,v);
		d[v]+= d[to];
	}
}

inline void solve() {
	findit(0);
	vector<int> ans;
	for(int i=0;i<n;i++) {
		if(white[i] && d[i]==1)
		{
			ans.push_back(i+1);
		}
	}
	printf("%d\n", (int) (ans.size()));
	for(int i=0;i<ans.size();i++) {
		cout << ans[i] << " ";
	}
	cout << endl;
}

int main() {
	scanf("%d",&n);
	for(int i=0;i<n-1;i++) {
		int x,y,t;
		cin >> x >> y >> t;
		x--,y--;
		adj[x].push_back(y);
		adj[y].push_back(x);
		if(t==2) {
			white[x]=1;
			white[y]=1;
		}
	}
	solve();
}
