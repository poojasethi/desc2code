#include <iostream>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <utility>
#include <algorithm>
#include <cmath>

using namespace std;

int n,x,y,t,k=0,a[100001],d[100001];
vector<int> v[100001];

int recurse(int node, int prev) {
	for (int i = 0; i < v[node].size(); i++)
		if (v[node][i] != prev)
			d[node] += recurse(v[node][i], node);
	if (d[node]==1) a[k++] = node;
	return d[node];
}

int main() {
	memset(d,0,sizeof(d));
	cin >> n;
	for (int i = 0; i < n-1; i++) {
		cin >> x >> y >> t;
		v[x].push_back(y);
		v[y].push_back(x);
		if (t==2) {
			d[x] = 1;
			d[y] = 1;
		}
	}
	recurse(1,0);
	cout << k << "\n";
	for (int i = 0; i < k; i++) cout << a[i] << " ";
}
