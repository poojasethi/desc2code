#include <stdio.h>
#include <iostream>
#include <vector>

using namespace std;

vector<vector<pair<int, int> > > g(100001);
vector<int> an;
int dfs(int s, int p, int c){
	int ans = 0;
	for(int i = 0 ; i < g[s].size() ; i++){
		if(g[s][i].first != p){
			ans = ans + dfs(g[s][i].first , s, g[s][i].second);
		}
	}
	if(c == 2 && ans == 0){
		an.push_back(s);
		return 1;
	}
	else
		return ans;

}
int main(){
	int n, x, y, t;
	cin >> n;
	for(int i = 1 ; i < n ; i++){
		cin >> x >> y >> t;
		g[x].push_back(make_pair(y, t));
		g[y].push_back(make_pair(x, t));	
	}
	cout << dfs(1, -1, 1) << endl;
	for(int i = 0 ; i < an.size() ; i++)
		cout << an[i] << " ";
	cout << endl;
	return 0;
}
