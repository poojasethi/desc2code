#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

#define ii pair <int, int>

int n;
vector <ii> adj[100005];
vector <int> ans;

inline bool dfs(int u, int p, int w) {
  bool already = false;
  for (int i = 0; i < adj[u].size(); i++) {
    int v = adj[u][i].first;
    if (v == p) continue;
    already |= dfs(v, u, adj[u][i].second);
  }
  if (!already && w == 2) ans.push_back(u), already = true;
  return already;
}

int main() {
  scanf("%d", &n);
  for (int i = 1; i < n; i++) {
    int u, v, t;
    scanf("%d %d %d", &u, &v, &t);
    adj[u].push_back(ii(v, t));
    adj[v].push_back(ii(u, t));
  }
  dfs(1, 0, 1);
  printf("%d\n", (int) ans.size());
  for (int i = 0; i < ans.size(); i++) printf("%d ", ans[i]);
}
