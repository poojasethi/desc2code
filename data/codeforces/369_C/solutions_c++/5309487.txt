#include <stdio.h>
#include <vector>
using namespace std;
vector<int> e[100010], re;
void dfs(int fa, int x, int t)
{
	int rn = re.size(), i, y, z;
	for (i = 0; i < (int)e[x].size(); i += 2)
	{
		y = e[x][i], z = e[x][i + 1];
		if (y == fa)
			continue;
		dfs(x, y, z);
	}
	if (t == 2 && rn == re.size())
		re.push_back(x);
}
int main()
{
	int n, x, y, z, i;
	scanf("%d", &n);
	for (i = 1; i < n; i++)
	{
		scanf("%d%d%d", &x, &y, &z);
		e[x].push_back(y);
		e[x].push_back(z);
		e[y].push_back(x);
		e[y].push_back(z);
	}
	dfs(0, 1, 1);
	printf("%d\n", re.size());
	for (i = 0; i < (int)re.size(); i++) printf("%d ", re[i]);
	printf("\n");
	return 0;
}
