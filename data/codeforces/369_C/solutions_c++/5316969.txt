#include <iostream>
#include <vector>
using namespace std;

typedef pair<int,int> pii;

int n;
vector<pii> adj[100005];
vector<int> v;

bool dfs(int cur, int prev, bool b){
	bool ok = 0;
	for(int i = 0; i<adj[cur].size(); i++) if(adj[cur][i].first!=prev) ok|=dfs(adj[cur][i].first, cur, adj[cur][i].second);
	if(!ok&&b) v.push_back(cur), ok = 1;
	return ok;
}

int main(){

	cin >> n;
	for(int i = 1; i<n; i++){
		int a, b, c;
		cin >> a >> b >> c;
		a--; b--; c--;
		adj[a].push_back(pii(b, c));
		adj[b].push_back(pii(a, c));
	}

	dfs(0, -1, 0);
	cout << v.size() << endl;
	for(int i = 0; i<v.size(); i++) cout << v[i]+1 << " ";

}
