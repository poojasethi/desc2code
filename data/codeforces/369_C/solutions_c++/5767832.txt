#include <stdio.h>
#include <iostream>
#include <vector>
using namespace std;
int n,a,b,c,p=0,t[200000];
vector <int> g[100005];
int make(int u=1,int f=0,int c=0){
	int s=0,w;
	for (int i=0;i<g[u].size();i++){
		w=g[u][i]/2;
		if (w!=f)
			s+=make(w,u,g[u][i]%2);
	}
	if (c && !s){
		p++;
		t[p]=u;
	}
	return s+c;
}
int main( ){
	cin>>n;
	for (int i=1;i<n;i++){
		cin>>a>>b>>c;
		g[a].push_back(b*2+c-1);
		g[b].push_back(a*2+c-1);
	}
	make();
	cout<<p<<endl;
	if (p>0){
		for (int i=1;i<p;i++)
			cout<<t[i]<<' ';
		cout<<t[p]<<endl;
	}
	return 0;
}
