#include<bits/stdc++.h>
using namespace std;
vector<int> edges[100001];
int white[100001],dp[100001];
int rec(int curr,int prev)
{
	dp[curr]=white[curr];
	for(int i=0;i<edges[curr].size();i++)
	{
		if(edges[curr][i]!=prev)
		dp[curr]+=rec(edges[curr][i],curr);
	}
	return dp[curr];
}

int main()
{
	int n,x,y,t;
	scanf("%d",&n);
	for(int i=1;i<n;i++)
	{
		scanf("%d %d %d",&x,&y,&t);
		edges[x].push_back(y);
		edges[y].push_back(x);
		if(t==2)
		{
			white[x]=1;
			white[y]=1;
		}
	}
	rec(1,-1);
	vector<int> ans;
	for(int i=1;i<=n;i++)
	{
		if(dp[i]==1)
		{
			ans.push_back(i);
		}
	}
	printf("%d\n",ans.size());
	for(int i=0;i<ans.size();i++)printf("%d ",ans[i]);
	return 0;
}
