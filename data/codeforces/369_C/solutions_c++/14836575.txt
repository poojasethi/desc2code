#include<bits/stdc++.h>
using namespace std;
int n,total=0;
int ans[100004];
vector <pair<int,int> > G[100052];
int visited[100005];
int dfs(int v,int repaired)
{
	visited[v]=1;
	int i,fl=0;
	for(i=0;i<G[v].size();i++)
	{
		if(!visited[G[v][i].first])
		{
			
			if(dfs(G[v][i].first,G[v][i].second))
			{

				fl=1;
			}
		}
	}
	if(fl==0 && repaired)
	{
		ans[total++]=v;
		return 1;
	}
	if(repaired || fl)
	{
		return 1;
	}
	return 0;

}
int main()
{
	int i,j,fl,test,z,x,y;
	scanf("%d",&n);
	for(i=0;i<n-1;i++)
	{
		scanf("%d%d%d",&x,&y,&z);

		G[x].push_back(make_pair(y,(z+1)%2));
		G[y].push_back(make_pair(x,(z+1)%2));
	}
	dfs(1,0);
	printf("%d\n",total );
	for(i=0;i<total;i++)
	{
		printf("%d ",ans[i] );
	}
	return 0;
}
