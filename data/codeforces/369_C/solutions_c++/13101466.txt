//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

const int maxn=100100;
int n,P;
vector<pair<int,int> > g[maxn];
vector<int> ans;
bool vis[maxn];

void dfs(int v,int c)
{
  vis[v]=true;
  for(int i=0;i<g[v].size();i++)
    if(!vis[g[v][i].first])
      P+=g[v][i].second,dfs(g[v][i].first,g[v][i].second);
  if(P && c)
    ans.push_back(v),P=0;
}

int main()
{
  cin>>n;
  for(int v,u,t,i=1;i<n;i++)
    {
      cin>>v>>u>>t;
      t=!(t&1);
      g[v].push_back(make_pair(u,t));
      g[u].push_back(make_pair(v,t));
    }
  dfs(1,0);
  cout<<ans.size()<<endl;
  for(int i=0;i<ans.size();i++)
    cout<<ans[i]<<" ";
  cout<<endl;
  return 0;
}
