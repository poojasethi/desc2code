#include <iostream>
#include <vector>
using namespace std;
vector <int> R[100001];
int n, x, y, t, k, i, D[100001];

int dfs(int x)
{
	int rc=0;
	D[x] |= 2;
	for (int i=0; i<R[x].size(); i++) {
		if (D[R[x][i]]<2) rc+=dfs(R[x][i]); 
	}
	if (rc) D[x]&=2;
	if (rc||(D[x]&1)) return 1;
	else return 0;
}

int main()
{
	cin>>n;
	for (i=1; i<n; i++) {
		cin>>x>>y>>t;
		R[x].push_back(y);
		R[y].push_back(x);
		if (t==2) D[x]=D[y]=1;
	}
	dfs(1);
	for (int i=2; i<=n; i++) if (D[i]&1) k++;
	cout<<k<<endl;
	for (int i=2; i<=n; i++) if (D[i]&1) cout<<i<<" ";
	cout<<endl;
	return 0;	
}
