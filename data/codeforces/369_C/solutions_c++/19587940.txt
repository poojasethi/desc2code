#include<bits/stdc++.h>
using namespace std;

vector < pair <int, bool> > V[100005];
bool visited[100005];
set <int> S;


int dfs(int x, bool a )
{
	visited[x] = 1;
	int cnt = 0;
	for(int i = 0 ;i < V[x].size(); i++)
	{
		if( visited[V[x][i].first] == 0)
		{
			cnt += dfs(V[x][i].first,V[x][i].second);
		}
	}
	if(cnt == 0 && a)
		{
			S.insert(x);
			cnt ++;
		}
		
	return cnt;

}

int main()
{
	int N;
	cin >> N;
	int x, y, t;
	for(int i = 0 ;i < N-1; i++)
	{
		cin >> x >> y >> t;
		V[x].push_back(make_pair(y,t-1));
		V[y].push_back(make_pair(x,t-1));
	}

	cout << dfs(1,0) << endl;

	set<int>::iterator it;

	for(it = S.begin(); it!= S.end(); it++)
	{
		cout << *it << " ";
	}

	return 0;
}
