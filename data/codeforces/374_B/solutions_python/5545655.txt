import sys



a = sys.stdin.read(1)
prev = 0
count = 1
while a != '\n':
  curr = int(a)
  if curr + prev == 9:
    a = sys.stdin.read(1)
    subcount = 0
    while a != '\n':
      if int(a) == prev:
        prev, curr = curr, prev
        subcount += 1
        a = sys.stdin.read(1)
      else:
        prev = int(a)
        a = sys.stdin.read(1)
        break
    if subcount % 2 != 0:
      count *= (subcount + 3) / 2

  else:
    prev = curr
    a = sys.stdin.read(1)


print count
