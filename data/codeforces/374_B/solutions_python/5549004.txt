a = list(raw_input())
length = len(a)
ans = 1
tot = 0

for i in range(1,length):
	if int(a[i])+int(a[i-1]) == 9:
		tot = tot + 1
	else:
		if tot == 0 or tot % 2 != 0:
			tot = 0
			continue
		val = 2
		val += (tot//2)-1
		ans *= val
		tot = 0

if tot != 0 and tot % 2 == 0:
	val = 2
	val += (tot//2)-1
	ans *= val

print ans
