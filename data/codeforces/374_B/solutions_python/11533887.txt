#!/usr/bin/python
# encoding: utf-8

a = raw_input()
n = len(a)

result = 1
l = 0
for i in xrange(n):
    if i < n-1 and int(a[i]) + int(a[i + 1]) == 9:
        if l == 0:
            l = 2
        else:
            l += 1
    else:
        if l % 2 == 1:
            result *= l / 2 + 1
        l = 0
print result
