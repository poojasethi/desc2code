n = map(int,list(raw_input()))
c,l = 1,0
for i in xrange(0,len(n)-1):
    if l == 0:
        if n[i] + n[i+1] == 9:
            l = 2
    else:
        if n[i] + n[i+1] == 9:
            l += 1
        else:
            if l % 2 == 1:
                c *= (l/2 + 1)
            l = 0
print c * (l/2 + 1) if l % 2 != 0 else c
