a = str(raw_input())
t = 0
res = 1

def num_ways(x):
    if x%2 == 0:
        return 1
    else:
        return (x+1)/2
    
for i in xrange(1,len(a)):
    if int(a[i]) + int(a[i-1]) == 9 and int(a[i]) not in [0,9]:
        t += 1
    else:
        res *= num_ways(t+1)
        t = 0
res *= num_ways(t+1)
print res
