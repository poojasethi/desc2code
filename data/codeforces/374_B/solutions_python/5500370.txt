#In the name of ALLAH
str = raw_input()
ans = 1
start = 0
end = 0
state = False
for ind, c in enumerate(str[:-1]):
	#print start, end, state
	i = int(c)
	ii = int(str[ind+1])
	if i+ii == 9:
		if not state:
			start = ind
			state = True
		end = ind + 1
	else:
		if state:
			if (end-start) % 2 == 0:
				ans *= (end-start+2)/2
			state = False
if state == True:
	if (end-start) % 2 == 0:
		ans *= (end-start+2)/2
print ans
