string = raw_input()

nines = [False]

digit = int(string[0])
for i in xrange(1, len(string)):
    next = int(string[i])
    if digit + next == 9:
        nines.append(True)
    else:
        nines.append(False)
    digit = next

nines.append(False)
count = 1
mult = 0
for i in xrange(len(nines)):
    if nines[i]:
            mult += 1
    else:
        if mult > 0 and mult % 2 == 0:
            count *= mult / 2 + 1
        mult = 0



print count
