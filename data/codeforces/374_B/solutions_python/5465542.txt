s = raw_input()+str(9)
t = 0
a = []
for i in range(1,len(s)):
    if(int(s[i]) + int(s[i-1]) == 9):
        t += 1
    else:
        a.append(t)
        t = 0
for i in range(len(a)):
    if a[i] % 2 == 1 or a[i] == 0:
        a[i] = 1
    else:
        a[i] = 1 + a[i]/2

c = 1

for i in range(len(a)):
    c*=a[i]
    
print c
