def main():
    k = raw_input()
    p = k[0]
    result = 1
    length = 0
    for i in k[1:]:
        if int(i) + int(p) == 9:
            length += 1
        else:
            if length and length % 2 == 0:
                result *= length / 2 + 1
            length = 0
        p = i
    else:
        if length and length % 2 == 0:
            result *= length / 2 + 1
    print result

if __name__ == '__main__':
    main()
