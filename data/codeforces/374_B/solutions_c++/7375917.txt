#include<iostream>
#include<cstring>
#include<cstdio>
#include<string>
using namespace std;
string str;
int main()
{
	__int64 sum,l,k;
	int i;
	cin>>str;
	l=str.size();
	k=1;
	sum=1;
	for(i=0;i<l;i++)
	{
		if(str[i]=='9')continue;
		if(str[i]-'0'+str[i+1]-'0'==9)k++;
		else if(k)
		{
			if(k%2==1)sum*=(k+1)/2;
			k=1;
		}
	}
	cout<<sum<<endl;
}
