#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

string s;

int main(){
    ll ans = 1,i,cnt = 0;
    cin >> s;
    for(i=1;i<=s.size();i++){
        if(s[i-1]-'0'+s[i]-'0' == 9) cnt++;
        else{
            if(cnt > 0 && cnt%2 == 0) ans *= (cnt/2+1);
            cnt = 0;
        }
    }
    cout << ans << '\n';
    return 0;
}
