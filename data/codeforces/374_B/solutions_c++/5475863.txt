#include <cstdio>
#include <iostream>
using namespace std;

char c, d;
long long a = 1, b=0;

int main()
{
    scanf("%c", &c);

    d = c;
    while (~scanf("%c",&c))
    {
        ++ b;
        if (c + d - 105)
        {
            a *= b % 2 ? b / 2 + 1 : 1;

    }

cout<<a;

}
