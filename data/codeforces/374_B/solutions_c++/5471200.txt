#include <cstdio>
#include <cstring>

long long ans=1;
char ch[111111];

int main(){
    scanf("%s",ch);
    int len=strlen(ch);
    int t=1;
    for(int i=1;i<len;i++){
        if(ch[i]+ch[i-1]-2*48==9)t++;
        else{
            if(t>1&&t%2)ans*=t/2+1;
            t=1;
        }
    }
    if(t>1&&t%2)ans*=t/2+1;
    printf("%I64d\n",ans);
    return 0;
}
