#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    char a[100005];
    scanf("%s",a);
    long long l=strlen(a),ans=1,p=0,q;
    while(p+1<l)
    {
        q=p+1;
        while((a[q]+a[q-1]-'0')=='9')
            ++q;
        if((q-p)%2==1)
            ans*=(q-p+1)/2;
        p=q;
    }
    printf("%I64d",ans);
}
