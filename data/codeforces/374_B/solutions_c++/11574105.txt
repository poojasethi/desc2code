# include <bits/stdc++.h>

using namespace std;

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(0);

	string s;
	cin >> s;
	s += "9";
	long long int cnt = 0, ans = 1;
	for(int i=1 ; i<s.size() ; i++){
		if((s[i]-'0') + (s[i-1]-'0') == 9)
			cnt++;
		else{
			cnt++;
			if(cnt % 2)
				ans *= (1 + int(cnt/2));
			cnt = 0;
		}
	}

	cout << ans << endl;

	return 0;
}
