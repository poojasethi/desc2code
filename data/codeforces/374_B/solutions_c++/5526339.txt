#include <stdio.h>

int main() {
  long long cnt = 1, ans = 1;
  char s[100005];
  gets(s);
  for(int i=1; s[i]!='\0'; ++i) {
    if(s[i]+s[i-1]-'0'-'0' == 9) ++cnt;
    else {
      if(cnt % 2LL) ans *= (cnt+1)/2;
      cnt = 1;
    }
  }
  if(cnt % 2LL) ans *= (cnt+1)/2;
  printf("%lld\n", ans);
  return 0;
}
