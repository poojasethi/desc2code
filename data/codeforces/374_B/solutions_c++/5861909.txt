//In the name of God
#include <iostream>
using namespace std;

long long ans = 1;
string s;

int main() {
	ios_base::sync_with_stdio(false);
	cin >> s;
	int p1 = 0, p2 = 1;
	while (p1 < s.size()) {
		while (p2 < s.size() && s[p2] + s[p2 - 1] == '0' + '9')
			p2++;
		if ((p2 - p1) % 2)
			ans *= (p2 - p1 + 1) / 2;
		p1 = p2++;
	}
	cout << ans << '\n';
	return 0;
}
