# include <stdio.h>
# include <iostream>
# include <string>
using namespace std;

int main(){
	string s;
	getline(cin,s);
	long long int i=0,ans=1;
	int c;
	while(s[i]!=0){
		c=0;
		while(s[i+1]!=0&&s[i]+s[i+1]-48=='9'){
			c++;
			i++;
		}
		if(c>0&&c%2==0) ans=ans*((long long)c/2+1);
		i++;
	}
	printf("%I64d\n",ans );
	return 0;
}
