#include <iostream>
#include <string>

using namespace std;

int main(int argc, char** argv) {
string S;
cin>>S;
int i,k,N=S.size(),prev=S[0]-'0'; long long ans=1;
for(i=k=1;i<N;i++)
{
    int c=S[i]-'0';
    if(prev+c==9)k++;
    else { if(k>1 && (k&1))ans*=(k+1)>>1; k=1;
    }
    prev=c;
}
if(k>1 && (k&1))ans*=(k+1)>>1;
    cout<<ans;
    return 0;
}
