#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    cin>>s;
    long long res=1;
    for(int i=0;i<s.size();i++)
    {
        int l=i,k=i+1;
        while(k<s.size()&&s[k-1]-'0'+s[k]-'0'==9)k++;
        k--;
        int d=k-i+1;
        if(d&1)res*=((d+1)>>1);
        i=k;
    }
    cout<<res;
}
