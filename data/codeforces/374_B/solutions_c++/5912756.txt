#include <iostream>
#include <string>

using namespace std;

string m;
long long s=1,t;

int main()
{
    cin >> m;
    for (int i=0;i<m.size();i++)
        if (m[i]+m[i+1] - 2*'0' == 9) 
        {
           t++;
        } 
        else 
        {
             s*=1+!(t%2)*((t+1)/2);
             t=0;
        }
    cout << s;
}
