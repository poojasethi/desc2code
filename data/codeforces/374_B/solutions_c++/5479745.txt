#include <stdio.h>

const int N = 100008;

char s [N];
__int64 ans = 1;
int main()
{
    int i,j;
    scanf("%s", s);
    for(i = 0; s[i];)
    {
    for(j = i + 1; s[j] && s[j] - '0' + s[j-1] - '0' == 9; j++);
    if((j - i) % 2)
        ans *= (__int64)((j - i + 1) / 2);
    i = j;
    }
    printf("%I64d\n", ans);
    return 0;
}
