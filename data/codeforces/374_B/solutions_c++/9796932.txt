#include<cstdio>

char s[111111];
int cur;
long long ans;

int main() {
	ans = 1LL;cur = 0;
	scanf("%s", s);
	for(int i=1;s[i];i++) {
		if (s[i] - '0' + s[i-1] - '0' == 9)
			cur ++;
		else if(cur) {
			if (!(cur & 1))
				ans*=(cur >> 1) + 1;
			cur = 0;
		}
	}
	if (cur && !(cur & 1)) ans*=(cur >> 1) + 1;
	printf("%I64d\n", ans);

	return 0;
}
