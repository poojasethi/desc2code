#include <cstdio>
char S[100001];
long long int ans = 1, cont;
int main()
{
    scanf("%s", S);
    for(int i = 0 ; S[i] ; i++ )
    {
        if( S[i]-'0' + S[i+1]-'0' == 9 ) cont++;
        else
        {
            if( cont % 2 == 0 ) ans *= (cont/2+1);
            cont = 0;
        }
    }
    printf("%lld", ans);
}
