#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pii;

int main(){
	string s;
	cin >> s;

	int c = 0; ll ans = 1;
	for (int i = 0; i < s.size(); i++)
		if (i < s.size() - 1 && s[i] + s[i+1] == 2 * '0' + 9)
			c++;
		else{
			if (c % 2 == 0 && c) ans *= (c/2) + 1;
			c = 0;
		}
		

	cout << ans << endl;
}
