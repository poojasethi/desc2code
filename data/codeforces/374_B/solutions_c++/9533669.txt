#include<iostream>
#include<string>
using namespace std;
typedef long long ll;
int main() {
 string s; cin>>s;
 int sz=s.size(), ct=1;
 ll ret=1;
 int p=s[0]-'0',q;
 for (int i=1; i<sz; i++) {
  q=s[i]-'0';
  if(p+q==9) ct++;
  else {
   if(ct%2) ret*=(ct+1)/2;
   ct=1;
  }
  p=q;
 }
 if(ct%2) ret*=(ct+1)/2;
 cout << ret << endl;
 return 0;
}
