#include<stdio.h>
int main()
{
	__int64 a,i;
	int b;
	char c;

	c=getchar();
	b='9'-c;
	i=0;
	a=1;
	while((c=getchar())!=EOF&&c!=10)
	{
		if(c==b+'0')i++;
		else
		{
			if(!(i%2)&&i)a*=i/2+1;
			i=0;
		}
		b='9'-c;
	}
	if(!(i%2)&&i)a*=i/2+1;
	printf("%I64d\n",a);
	return 0;
}
