#include <bits/stdc++.h>
using namespace std;

const int N = 111111;
typedef long long LL;
char s[N];

int main() {
	while (cin >> s) {
		LL ans = 1;
		int len = strlen(s);
		for (int l = 0, r; l < len; l = r) {
			r = l + 1;
			while (r < len && s[r - 1] + s[r] - '0' == '9') r++;
			if (r - l & 1) ans *= r - l + 1 >> 1;
		}
		cout << ans << endl;
	}
	return 0;
}
