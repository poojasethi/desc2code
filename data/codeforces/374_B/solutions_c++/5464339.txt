#include <iostream>
#include <string>
using namespace std;
string s;
long long ans=1ll;
int i,k;
int main() {
    cin >> s;
    for (i=1;i<s.length();i++) {
        if (s[i]+s[i-1]=='0'+'9') k++;
        else {
            if (k%2==0) ans*=1ll*(k/2+1);
            k=0;
        }
    }
    if (k%2==0) ans*=1ll*(k/2+1);
    cout << ans << endl;
}
