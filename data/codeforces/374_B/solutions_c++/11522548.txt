#include<bits/stdc++.h>
using namespace std;
#define LL long long int
int main()
{
    string s;
    cin >> s;
    LL ans=1;
    int k=0,i=0;
    for (i = 0;i<s.size()-1;i++)
    if (s[i]+s[i+1]=='9'+'0')
      k++;
    else
    {
        if(k%2==0)
        {ans*=k/2+1;}
        k=0;
    }
    if (k%2==0)
       {ans*=k/2+1;}
    cout << ans<<endl;;
}
