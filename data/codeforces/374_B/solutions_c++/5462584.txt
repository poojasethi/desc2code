#include<stdio.h>
#include<string.h>
char s[100011];
int main()
{
    scanf("%s",s);
    long long ans=1;
    int i=1;
    long long c=1;
    while(s[i])
    {
     if(s[i]+s[i-1]-'0'-'0'==9) c++;else
     {
      if(c%2) ans=ans*((c+1)/2);
      c=1;
     }
     i++;
    }
    if(c%2) ans=ans*((c+1)/2);
    printf("%I64d\n",ans);
}
