#include <iostream>
#define I long long
using namespace std;

int main() {
  string s;
  getline(cin, s);
  I ans = 1, i = 0;
  while(i < s.length() - 1) {
    I curr = 1;
    while(i < s.length() - 1 && (s[i] - '0') + (s[i + 1] - '0') == 9) {
      curr++; i++;
    }
    ans *= (curr % 2 == 0) ? 1 : (curr + 1) / 2;
    i++;
  }
  cout << ans << endl;
}
