#include <iostream>

using namespace std;

int main()
{
	string s;
	cin >> s;
	int i = 0;
	long long ans = 1;
	while (i < s.length())
	{
		int cnt = 1;
		while (i + 1 < s.length() && ((s[i] - '0') + (s[i + 1] - '0')) == 9)
		{
			i++;
			cnt++;
		}
		if ((cnt % 2 == 1))
			ans = (ans * (cnt + 1) / 2);
		i++;
	}
	cout << ans << endl;

}
