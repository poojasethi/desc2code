#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
	char a[1000005];
	long long int ans=1;
	int i=1,c=1;
	cin>>a;
	while(a[i]){
		if(a[i]+a[i-1]-'0'-'0'==9)c++;
		else{
			if(c%2)
				ans*=(c+1)/2;
			c=1;
		}
		i++;
	}
	if(c%2)ans*=(c+1)/2;
	cout<<ans<<endl;
	return 0;
}
