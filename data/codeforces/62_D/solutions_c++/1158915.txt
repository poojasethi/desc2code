#include<cmath>
#include<cstdio>
#include<string>
#include<vector>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<map>
using namespace std;
#define ll __int64
int mp[105][105],s[2005],n,m,ans[2005];
int dfs(int x,int u,int big)
{
    ans[x]=u;
    if(x==m)
    {
        return big;
    }
    int low=s[x+1];
    if(big)low=1;
    for(int v=low;v<=n;v++)
    if(mp[u][v])
    {
        --mp[u][v];
        --mp[v][u];
        int ret=dfs(x+1,v,v>s[x+1]||big);
        if(ret)return 1;
        ++mp[u][v];
        ++mp[v][u];
    }
    return 0;
}
int main()
{
    while(~scanf("%d%d",&n,&m))
    {
        memset(mp,0,sizeof(mp));
        scanf("%d",&s[0]);
        for(int i=1;i<=m;i++)
        {
            scanf("%d",&s[i]);
            mp[s[i-1]][s[i]]++;
            mp[s[i]][s[i-1]]++;
        }
        if(dfs(0,s[0],0))
        {
            for(int i=0;i<=m;i++)
            printf("%d ",ans[i]);
            puts("");
        }
        else puts("No solution");
    }
}
