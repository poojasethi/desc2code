#include <stdio.h>
#include <string.h>

char str[100010], model[100010];
int a[100010];

int main()
{
	int m, n, i, j, len, pos, mlen;
	scanf("%s", str);
	n=strlen(str);
	for (i=0; i<n; i++)
	  a[i]=n-i;
	scanf("%d", &m); 
	for (i=0; i<m; i++)
	{
		scanf("%s", model);
		mlen=strlen(model);
		int far=n;
		for (j=n-mlen; j>=0; j--)
		{
		  if (strncmp(str+j, model, mlen)==0)
		  	far=j+mlen-1;
		  if (far-j<a[j]) a[j]=far-j;	
		}
	}
	len=0;
	pos=0;
	for (i=0; i<n; i++)
	  if (a[i]>len)
	  {
	  	len=a[i];
	  	pos=i;
	  }
	printf("%d %d\n", len, pos);  
	return 0;
}
