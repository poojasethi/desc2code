#include <stdio.h>
#include <string.h>
#include <iostream>
using namespace std;
char ch[100010],b[11][11];
int n,w,l[100010],ans;
int main(){
    int i,j,k,p;
    scanf("%s",ch);
    scanf("%d",&n);
    for (i=0;i<n;i++)
        scanf("%s",&b[i]);
    w=0;
    ans=0;
    for (i=0;ch[i]!='\0';i++){
        if (i!=0) l[i]=l[i-1]+1;
        else l[i]=1;
        for (j=0;j<n;j++){
            k=strlen(b[j]);
            if (k>i+1) continue;
            p=1;
            while (p<=k && b[j][k-p]==ch[i-p+1]) p++;
            if (p>k && k-1<l[i]) 
               l[i]=k-1;
        }
        if (l[i]>ans){
           ans=l[i];
           w=i-l[i]+1;
        }
    }
    printf("%d %d\n",ans,w);
    return 0;
}
