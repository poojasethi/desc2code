#include <cstring>
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>

using namespace std;

#define maxn 100007
int n;
int dp[maxn];
string txt;
string a[11];

bool check(int i,int x){
  int j;
  for( j=a[i].size()-1;j>=0 && x>=0; --j,--x)
   if(a[i][j]!=txt[x]) return false;
  if(j>=0 && x<0) return false;
  return true;
}

int main(){
  cin>>txt>>n;
  for(int i=0;i<n;++i) cin>>a[i];
  int ln = txt.size();
  for(int i=0;i<ln;++i){
    int mx = -1;
    for(int j=0;j<n;++j)
     if(check(j,i)) mx = max(mx,i-(int)a[j].size()+1);
    dp[i] = (i?dp[i-1]:0)+1;
    if(mx!=-1)  dp[i] = min(i-mx,dp[i]);
  }
  int mx = 0; int s = 0;
  
  for(int i=0;i<ln;++i){ 
    if(mx<dp[i]){
        mx = dp[i]; s = i-dp[i]+1;
    }
  }
  cout<<mx<<" "<<s<<endl;
  return 0;
}
