#include<cstring>
#include<cstdio>

const int mx=100010,mx2=15,mx3=15;

int n;
char s[mx];
char t[mx2][mx3];
int leni[mx2];

int minn(int a,int b)
{
	return a<b?a:b;
}

int main()
{
	int i,j;
	scanf("%s",s);
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%s",t[i]);
		leni[i]=strlen(t[i]);
	}
	int len=strlen(s);
	int start,end=len-1;
	int ans_len=0;
	int ans_pos=0;
	for(start=len-1;start>=0;start--)
	{
		for(i=0;i<n;i++)
		{
			if(start+leni[i]>len)
				continue;
			for(j=0;j<leni[i];j++)
			{
				if(s[start+j]!=t[i][j])
					break;
			}
			if(j>=leni[i])
				end=minn(end,start+leni[i]-2);
		}
		if(ans_len<end-start+1)
		{
			ans_len=end-start+1;
			ans_pos=start;
		}
	}
	printf("%d %d\n",ans_len,ans_pos);

	return 0;
}
