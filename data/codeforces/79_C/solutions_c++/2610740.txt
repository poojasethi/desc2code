#include <iostream>
#include <string>
#define sz(a) ((int)(a).size())

using namespace std;

string wstr;
int n;
int d[ 110000 ];

int main(int argc, char *argv[]) {
  cin >> wstr;

  for (int i = 0; i < sz(wstr); ++i) {
    d[i] = sz(wstr) - i;
  }

  cin >> n;
  for (int i = 0; i < n; ++i) {
    string temp;
    cin >> temp;
    for (int j = 0; j < sz(wstr) - sz(temp) + 1; ++j) {
      if (wstr.substr(j, sz(temp)) == temp) {
        d[j] = min(d[j], sz(temp) - 1);
      }
    }
  }

  for (int i = sz(wstr) - 2; i >= 0 ; i --) {
    d[i] = min(d[i+1] + 1, d[i]);
  }
  int cmaxi = 0, maxi = 0;
  for (int i = 0; i < sz(wstr); ++i) {
    if (cmaxi < d[i]) {
      cmaxi = d[i];
      maxi = i;
    }
  }
  cout << cmaxi << " " << maxi << endl;
  return 0;
}
