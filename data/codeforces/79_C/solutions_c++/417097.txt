#include<stdio.h>
#include<string.h>
char s[100005];
char ss[15];
int a[100005],p[15];
int main()
{
	int l,m,n,i,j,k,temp;
	scanf("%s",&s[1]);
	n=strlen(&s[1]);
	scanf("%d",&m);
	memset(a,0,sizeof(a));
	while (m>0)
	{
		m--;
		scanf("%s",&ss[1]);
		p[1]=0;
		j=0;
		for (i=2;ss[i]!='\0';i++)
		{
			while (j>0&&ss[j+1]!=ss[i])
				j=p[j];
			if (ss[j+1]==ss[i])
				j++;
			p[i]=j;
		}
		l=i-1;
		j=0;
		temp=0;
		for (i=1;i<=n;i++)
		{
			while (j>0&&ss[j+1]!=s[i])
				j=p[j];
			if (ss[j+1]==s[i])
				j++;
			if (j==l)
			{
				temp=i-l+1;
				j=p[j];
			}
			if (a[i]<temp)
				a[i]=temp;
		}
	}
	k=1;
	temp=k-a[k];
	for (i=2;i<=n;i++)
	{
		if (i-a[i]>temp)
		{
			k=i;
			temp=i-a[i];
		}
	}
	printf("%d %d\n",temp,a[k]);
	return 0;
}
