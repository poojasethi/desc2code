#include <iostream>
#include <string>
#include <vector>

using namespace std;

bool essufix(string &b,string &s,int is)
{
  for (int i=0;i<int(b.size());i++)
    if (b[int(b.size())-1-i]!=s[is-i])
      return false;
  return true;
}

int main()
{
  string s;
  cin>>s;
  int n;
  cin>>n;
  vector<string> v(n);
  for (int i=0;i<n;i++) cin>>v[i];
  int maxpos=0;
  int maxlen=0;
  int pos=0;
  for (int i=0;i<int(s.size());i++) {
    for (int j=0;j<n;j++) {
      string b=v[j];
      int lenb=int(b.size());
      if (i-pos+1>=lenb and essufix(b,s,i))
	pos=i-lenb+2;
    }
    if(i-pos+1>maxlen) {
      maxlen=i-pos+1;
      maxpos=pos;
    }
  }
  cout<<maxlen<<" "<<maxpos<<endl;
}
