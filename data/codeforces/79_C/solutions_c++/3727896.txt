# include <stdio.h>
# include <iostream>
# include <algorithm>

using namespace std;

int b,z,mx,t,an;

string a,c[10];

int main()
{
	cin>>a>>b;
	for(int i=0;i<b;i++)	cin>>c[i];
	for(int i=0;i<a.size();i++)
	{
		for(int j=0;j<b;j++)
		if(i-(int)c[j].size()>-2)
		{
			z=0;
			for(int l=i;l>i-(int)c[j].size();l--)	if(c[j][c[j].size()+l-i-1]!=a[l])	{z=1;break;}
			if(z==0)	t=max(t,i-(int)c[j].size()+2);
		}
		if(mx<i-t+1)	{mx=i-t+1;an=t;}
	}
	
	printf("%d %d",mx,an);getchar();getchar();
}
