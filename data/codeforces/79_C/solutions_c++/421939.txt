#include<iostream>
#include<string>
using namespace std;
string pp[11];
int main()
{
    string s;
    cin >> s;
    int n;
    cin >> n;
    for(int i=0;i<n;i++)
        cin >> pp[i];
    int len=s.size();
    int a,b;
    a=b=0;
    int now=len;
    int ll;
    for(int i=len-1;i>=0;i--)
    {
        for(int j=0;j<n;j++)
        {
            ll=pp[j].size();
            if(i+ll <= now && s.substr(i,ll) == pp[j])
                now=i+ll-1;
        }
        if(now - i > a)
        {
            a=now-i;
            b=i;
        }
    }
    cout << a << " " << b << endl;
    return 0;
}
