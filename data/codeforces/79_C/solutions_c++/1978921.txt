#include <iostream>
#include <string>
#include <vector>
using namespace std;

string s;
vector<string> p;
int n;

int func(int prev, int j) {
	int mn = 1e8;
	for(int i = 0; i < p.size(); i++) {
		if(j+1 < p[i].size())
			continue;
		string tmp = s.substr(j-p[i].size()+1, p[i].size());
		if(tmp == p[i]) {
			mn = min<int>(mn, p[i].size()-1);
		}
	}
	return min<int>(mn, prev+1);
}

int main() {
	cin >> s >> n;
	for(int i = 0; i < n; i++) {string t; cin >> t; p.push_back(t); }
	int prev = 0, mx = 0, mx_pos = 0;
	for(int i = 0; i < s.size(); i++) {
		int d = func(prev, i);
		if(d > mx) {
			mx = d;
			mx_pos = i-mx+1;
		}
		prev = d;
	}
	cout << mx << " " << mx_pos << endl;
	return 0;
}
