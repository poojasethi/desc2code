#include<cstdio>
#include<iostream>
#include<string>
#include<cstring>
#include <vector>
#include <algorithm>
using namespace std;
char s[100005];
int tmt[100005];
char b[12][100];
void proc(char *t){
    int i,m=strlen(t);
    for(i=0;s[i];i++){
        if(strncmp(s+i,t,m)==0){
            tmt[i+m-1]=max(tmt[i+m-1],i);
        }
    }
}

int main(){
    memset(tmt,-1,sizeof(tmt));
    int n,i,j,len=0,st=0;
    cin>>s;
    cin>>n;
    for(i=0;i<n;i++){
        cin>>b[i];
        proc(b[i]);
    }
    for(i=j=0;s[i];i++){
       // cout<<tmt[i]<<" ";
        j=max(j,tmt[i]+1);
        if(i-j+1>len){
            len=i-j+1;
            st=j;
        }
    }
   //  cout<<endl;
    printf("%d %d\n",len,st);
   //  while(1);
}
