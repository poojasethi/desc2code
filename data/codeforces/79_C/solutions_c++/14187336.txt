#include<iostream>
using namespace std;
int main(){
   int n,l=-1,r=-1,i,j,k,tans=0,sans=0;
   string s,t[100];
   cin>>s>>n;
   for(i=0;i<n;i++) cin>>t[i];
   for(i=0;i<s.size();i++){
      for(j=max(i-10,0);j<=i;j++){
          string r;
          for(k=j;k<=i;k++) r+=s[k];
          for(k=0;k<=n;k++){
            if(r==t[k]&&l<(i-(int)r.size()+1)) {l=i-r.size()+1;r=j;}
          }  
      }
      if(j-l>tans) tans=(j-l),sans=l+1;
   }
   cout<<tans-1<<" "<<sans<<endl;  
   return 0; 
}
