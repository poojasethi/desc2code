#include<iostream>

using namespace std;

int E,W;

main(){
   int n,l=-1,q=-1,i,j;
   string s,d[100];
   cin>>s>>n;
   for(i=0 ; i<n ; i++) cin>>d[i];
   for(int i=0 ; i<s.size(); i++){
      for(j=max(i-10,0) ; j<=i ; j++){
		string q;
        	for(int k=j ; k<=i ; k++)
			q+=s[k];
          for(int k=0 ; k<=n ; k++)
            if(q==d[k] && l<(i-(int)q.size()+1)){
				l=i-q.size()+1;
				q=j;
			}
      }
      if(j-l>W) W=(j-l),E=l+1;
   }
   cout<<W-1<<" "<<E<<endl; 
   //system("pause"); 
   return 0; 
}
