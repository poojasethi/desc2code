#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cctype>
#include <cmath>
#include <cassert>
#include <algorithm>
using namespace std;

#define PROBLEM "c"

int main() {
#ifdef LAPTOP
  freopen(PROBLEM".in", "r", stdin);
#endif
  int i, j, x, L, W;
  int al=0, ap=0;
  int ln[16];
  char bf[100008];
  char wd[16][16];
  scanf("%s", bf);
  L=strlen(bf);
  scanf("%d", &W);
  for (i=0; i<W; i++) {
    scanf("%s", wd[i]);
    ln[i]=strlen(wd[i]);
  }
  for (i=x=L-1; i>=0; i--) {
    for (j=0; j<W; j++)
      if (!strncmp(&bf[i], wd[j], ln[j]))
        x=min(x, i+ln[j]-2);
    if (x-i+1>al) {
      al=x-i+1;
      ap=i;
    }
  }
  printf("%d %d\n", al, ap);
  return 0;
}
