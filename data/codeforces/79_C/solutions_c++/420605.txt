#include <cstdio>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int n;
string s;
int f[100009];

int main() {
	cin >> s >> n;
	int sl = s.size();
	for(int i=0;i<sl;i++) f[i] = sl-i;
	for(int i=1;i<=n;i++) {
		string b;
		cin >> b;
		int l = b.size();
		for(int j=0;j<sl;j++)
			if(l-1<f[j] && s.substr(j,l)==b)
				f[j] = l-1;
	}
	f[sl] = 0;
	for(int i=sl-1;i>=0;i--) f[i] = min(f[i],f[i+1]+1);
	int len=0, pos=0;
	for(int i=0;i<sl;i++)
		if(f[i]>len) len = f[i], pos = i;
	cout << len << " " << pos << endl;
}
