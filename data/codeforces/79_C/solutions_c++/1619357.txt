#include <iostream>
#include <string>
using namespace std;
int mx=-1,p=-1;
string list[11];
int n;
string s;
int main()
{
    cin>>s;
    cin>>n;
    for (int i=0;i<n;i++)
        cin>>list[i];
    int i=0,l=0,sz=s.size();
    while (i<sz)
    {
        for (int j=0;j<n;j++) if (list[j].size()<=i-l+1 && s.substr(i-list[j].size()+1,list[j].size())==list[j])
            l=i-list[j].size()+2;
        if (i-l+1>mx)
        {
            mx=i-l+1;
            p=l;
        }
        i++;
    }
    cout<<mx<<" "<<p<<endl;
    return 0;
}
