#include<cstdio>
#include<cstring>


char c[100005];
char a[15][15];
int b[100005];
int N;

int main(){
	scanf("%s%d",c,&N);
	
	for (int i=0;c[i];++i) b[i]=-1;
	for (int i=0;i<N;++i){
		scanf("%s",a[i]);
	}
	
	for (int i=0;c[i];++i){
		for (int j=0;j<N;++j){
			int k = 0;
			while (a[j][k] && c[i+k] && c[i+k]==a[j][k]) ++k;
			if (!a[j][k]){
				if (b[i+k-1] < i) b[i+k-1] = i;
			}
		}
	}
	
	int len = 0;
	int mx = 0;
	int sss = 0;
	int pre = -1;
	
	for (int i=0;c[i];++i){
		if (b[i]>=0 && pre<b[i]){
			pre = b[i];
		}
		
		len = i - pre;
		
		if (len>mx){
			mx = len;
			sss = pre+1;
		}
	}
	printf("%d %d\n",mx,sss);
	return 0;
}
