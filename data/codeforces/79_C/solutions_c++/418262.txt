#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

char buf[100000 + 10];
char son[100][100];

const int INF = 10000000;

int gao(int x, int y, int k) {
	for (int i = x; i < y; ++i) {
		if (son[k][i - x] == 0) {
			return i - 1;
		}
		if (buf[i] != son[k][i - x]) {
			return INF;
		}
	}
	if (son[k][y - x] == 0) {
		return y - 1;
	}
	return INF;
}

int main() {
	scanf("%s", buf);
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%s", son[i]);
	}
	int ans = 0, be = 0;
	int len = strlen(buf);
	int x , y = len;
	for (x = len - 1; x >= 0; --x) {
		int g = INF;
		for (int j = 0; j < n; ++j) {
			g = min(g, gao(x, y, j));
		}
		if (g != INF) {
			y = g;
		} 
		if (y - x > ans) {
			ans = y - x;
			be = x;
		}
	}
	printf("%d %d\n", ans, be);
	return 0;
}
