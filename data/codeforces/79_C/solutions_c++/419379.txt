#include <iostream>
#include <string>
#include <algorithm>

using namespace std;


int main(){
	string s;
	string b[10];
	int bsize;
	cin >> s >> bsize;
	int len=s.length();
	int found;
	for(int i=0; i<bsize; ++i){
		cin >> b[i];
	}
	int finala=0,finalb=0;
	int ansa=0, ansb=len;
	int next=len;
	while(ansa<len){
		for(int j=0; j<bsize; ++j){
			found=s.find(b[j],ansa);
			if(found==string::npos){ 
				b[j]="!";
				continue;
			}
			ansb=min(ansb,found+(int)b[j].length()-1);
			if(found<next) next=found;
		}
		if(ansb-ansa>finalb){
			finalb=ansb-ansa;
			finala=ansa;
		}
		ansa=next+1;
		next=ansb=len;
	}
	cout << finalb << " " << finala << "\n";
	return 0;
}
