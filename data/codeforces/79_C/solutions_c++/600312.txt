#include <iostream>
#include <string>

using namespace std;

int f[100001];

int main()
{
  string s;
  int n;
  cin >> s >> n;
  for(int i = 0; i < s.size(); i++) f[i] = s.size()-i;
  for(int i = 1; i <=n; i++)
  {
    string b;
    cin >> b;
    for(int j = 0; j <= (int)(s.size() - b.size());j++)
    {
      if ( f[j] > b.size() - 1 && s.substr(j,b.size()) == b)
        f[j] = b.size()-1;
    }
  }
  f[s.size()] = 0;
  for(int i = s.size()-1; i >= 0; i--) f[i] = min(f[i],f[i+1]+1);
  int len = 0;
  int pos = 0;
  for(int i = 0; i < s.size(); i++)
    if ( f[i] > len) { len = f[i]; pos = i;}
  cout << len << " " << pos;  
  //system("PAUSE");
}
