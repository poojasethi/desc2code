#include <iostream>
#include <string>
#include <map>
#include <vector>
#include <cassert>
#include <algorithm>

using namespace std;

int main()
{
	string s;
	//getline(cin,s);
	cin>>s;
	int n;
	cin >> n;
	vector<string> v(n);
	for(int i=0;i<n;++i)
		cin>>v[i];
	int l=0,pos=0;
	for(int i=0;i<s.size();++i)
	{
		int j=0;
		string tmp;
		bool ok=true;
		while(ok && i+j<s.size())
		{
			tmp+=s[i+j];
			for(int k=0;k<v.size();++k)
			{
				if(v[k].size()<=tmp.size() && tmp.substr(tmp.size()-v[k].size(),v[k].size())==v[k])
					ok=false;
			}
			if(ok)
				++j;
		}
		if(j>l)
		{
			l=j;
			pos=i;
		}
		if(j>10)
			i+=(j-10);
	}
	cout << l << " " << pos << endl;
	return 0;
}
