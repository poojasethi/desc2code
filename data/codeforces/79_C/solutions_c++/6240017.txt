#include<stdio.h>
#include<cstring>
#include<algorithm>
using namespace std;

int n , x , y[100001] ;
char a[100001] , b[11] ;

int  main()
{    
   scanf("%s%d",&a,&x) ;
  
   int n=strlen(a) ; memset(y,1,sizeof(y)) ;
   while(x--)
   { 
      scanf("%s",&b) ;
      int m=strlen(b) ; 
      for(int i=0 ; i<=n-m ; i++)
      {
         int sig=0 ; 
         for(int j=0 ; j<m ; j++) if(a[i+j]!=b[j]) { sig=1 ; break ; }
         if(!sig) y[i]=min(y[i],m-1) ;  
      } 
   } 
   for(int i=n-2 ; i>=0 ; i--) y[i]=min(y[i],y[i+1]+1) ; 
   
   int ans=0 , pos=0 ;  
   for(int i=0 ; i<n ; i++) if( ans<min(y[i],n-i)) ans=min(y[i],n-i) , pos=i ;
   printf("%d %d\n",ans,pos) ;    
     
 return 0;
}
