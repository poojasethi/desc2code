#include <bits/stdc++.h>
#include<cstring>
#define MOD 1000000007
#define MAX 100005
using namespace std;
string s,ps[10];
int f[100005];
int main()
{
    int k,i,j,n,len=0,pos=0;
    cin>>s;
    cin>>k;
    for(i=0;i<k;i++)
        cin>>ps[i];
    n=s.length();
    f[n]=0;
    for(i=n-1;i>=0;i--)
    {
        f[i]=f[i+1]+1;
        for(j=0;j<k;j++)
        {
            if(n-i>=ps[j].length()&&s.substr(i,ps[j].length())==ps[j])
            f[i]=min(f[i],(int)ps[j].length()-1);
        }
        if(f[i]>len)
        {
            len=f[i];
            pos=i;
        }
    }
    cout<<len<<" "<<pos;
    return 0;
}
