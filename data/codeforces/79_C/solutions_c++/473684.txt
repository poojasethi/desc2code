#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

const int maxn=100000+10;
char st[maxn],g[15][15];
int len[15];
int n;

int main()
{
    scanf("%s%d",st,&n);
    for (int i=0;i<n;i++)
    {
        scanf("%s",g[i]);
        len[i]=strlen(g[i]);
    }
    int last=-1,res=0,pos=0;
    for (int i=0;st[i];i++)
    {
        for (int j=0;j<n;j++)
        if (i+1>=len[j])
        {
            bool ok=1;
            for (int k=0;k<len[j] &&ok;k++)
                ok&=(st[i-len[j]+k+1]==g[j][k]);
            if (ok) last=max(last,i-len[j]+1);
        }
        if (i-last>res) 
        {
            res=i-last,pos=last+1;
        }
    }
    printf("%d %d\n",res,pos);
}
