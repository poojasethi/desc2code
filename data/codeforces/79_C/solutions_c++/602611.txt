#include <iostream>
#include <cstdio>
using namespace std;

#define REP(i,b,n) for(int i=b;i<(int)n;i++)
#define rep(i,n) REP(i,0,n)
#define foreach(i,c) for(__typeof((c).begin())i=(c).begin();i!=(c).end();i++)
#define mp make_pair

string in, w[12];
int n, lb, len, pos;

int main(int argc, char* argv[]) {
  cin>>in>>n;
  rep(i,n) cin>>w[i];
  rep(i,in.size()) {
    rep(j,n) {
      if (i - lb + 1 < w[j].size()) continue;
      if (!(in.substr(i - w[j].size() + 1, w[j].size()) == w[j])) continue;
      lb = i - w[j].size() + 2;
    }
    if (len < i - lb + 1) {
      len = i - lb + 1;
      pos = lb;
    }
  }
  printf("%d %d\n", len, pos);
  return 0;
}
