#include <iostream>
#include <string>

using namespace std;

string s;
string str[20];

int main() {
    int n;
    cin >> s;
    cin >> n;
    int ans = 0, flag = 0; 
    for (int i = 0; i < n; ++i) cin >> str[i];
    int l = 0; 
    for (int r = 0; r < s.size(); ++r) {
        for (int i = 0; i < n; ++i) {
            if (str[i].size() <= r - l + 1 && s.substr(r - str[i].size() + 1, str[i].size()) == str[i]) {
                l = r - str[i].size() + 2;
            }
        }
        if (r - l + 1 > ans) {
            ans = r - l + 1;
            flag = l;
        }
    }
    cout << ans << ' ' << flag << endl;
    return 0;
}
