#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
string s,b,c;
const int inf=1000000000;
int n,i,j,z[100001],y[100001],maxi,m,r,l,ss;
int main()
{
cin>>s>>n;
ss=s.size();
for (i=0;i<ss;i++)
y[i]=ss;
for (i=0;i<n;i++)
{cin>>b;
if (b.size()<=s.size())
for (j=0;j<=s.size()-b.size();j++)
	{
	c=s.substr(j,b.size()).c_str();
	if (c==b){
		if (y[j]>j+b.size()-1)
		y[j]=j+b.size()-1; 
		}
	}        
 }
y[ss]=ss;
for (i=ss-1;i>=0;i--)
{
if (y[i+1]<y[i])
	y[i]=y[i+1];
else y[i]--;
if (y[i]-i+1>maxi)
	{
	maxi=y[i]-i+1;
	m=i;
	}
}
cout<<maxi<<" "<<m;

return 0;
}
