#include <iostream>
#include <string>
using namespace std;
int mx=-1,p=-1;
string list[100];
int n;
string s;
int main()
{
	ios::sync_with_stdio(false);
	cin>>s;
	cin>>n;
	for (int i=0;i<n;i++)
		cin>>list[i];
	int i=0;
	int l=0;
	while (i<s.size())
	{
		for (int j=0;j<n;j++) if ((int)list[j].size()<=i-l+1 && s.substr(i-list[j].size()+1,list[j].size())==list[j])
			l=i-list[j].size()+2;
		if (i-l+1>mx)
		{
			mx=i-l+1;
			p=l;
		}
		i++;
	}
	cout<<mx<<" "<<p<<endl;
	return 0;
}
