import sys
#sys.stdin = file("input.txt", "r");

s, n = raw_input(), int(raw_input())
a = [0] * n

for i in xrange(n):
    a[i] = raw_input()
    
ln, ind, ls = 0, 0, 0
for i in xrange(len(s) - 1, -1, -1):
    ls += 1
    for t in a:
        if len(t) <= ls and s[i : i + len(t)] == t:
            ls = len(t) - 1
    if ls > ln:
        ln, ind = ls, i

print ln, ind
