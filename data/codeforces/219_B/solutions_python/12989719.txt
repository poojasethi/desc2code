p,d = map(int,raw_input().split())

digits = map(int,str(p)[::-1])

for i in xrange(len(digits)-1):
    if (digits[i]==9) : continue
    cost = (digits[i]+1)*(10**i)
    if cost <= d:
        d -= cost
        p -= cost
        digits = map(int,str(p)[::-1])

if digits[-1] == 0:
    digits = digits[:-1]
print ("".join(map(str,digits)))[::-1]
