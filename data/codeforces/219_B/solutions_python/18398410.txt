def my(): return map(int, raw_input().split())

def main():
    p, d = my()
    if d == 0:
        print p
        exit(0)
    
    k = len(str(d))
    res = 0
    for i in range(1, k+1):
        df = int(str(p+1)[-i:])
        if df <= d:
            res = df

    print p-res
    
if __name__ == "__main__":
    main()
