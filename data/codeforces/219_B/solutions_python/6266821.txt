def telnegen(x):
    s=x[::-1]
    
    i,n=0,0
    while i<len(s) and s[i]=='9':
        n+=1
        i+=1
    return n

g=[]
e,getal=1,p-p%10-1


while getal>=p-d:
    g.append(getal)
    e+=1
    getal=getal-getal%10**e-1
    
#print g    

m,j=telnegen(str(p)),-1
for i in g:
    z=telnegen(str(i))
    if z>m:
        j=g.index(i)
        m=z
