#!/usr/bin/env python

from sys import stdin

def f(P, d):
   x = P
   p10 = 1
   while True:
      dig = (x / p10) % 10
      if dig != 9:
         nx = x
         nx -= dig * p10
         nx += 9 * p10
         nx -= p10 * 10
         if nx <= 0 or P - nx > d: break
         x = nx
      p10 *= 10
   return x

def main():
   P, D = map(int, stdin.readline().split())
   print f(P, D)
   return 0

if __name__ == '__main__': main()
