n,d=map(int,raw_input().split())
s=str(n)
if (n-d)/10*10+9 >= n or n+1 == 2*d or set(str(n)[1:]) == set(['9']):
    print n
    exit()
m=str(n-d)
m='0'*(len(s)-len(m))+m
r=''
N=0
for (a,b) in zip(s,m):
    if N:
        r+= '9'
    elif a == b:
        r += a
    else:
        r += chr(ord(a)-1)
        N=1
print int(r)
