import sys
import math

p, d = [int(x) for x in (sys.stdin.readline()).split()]

i = 10
j = 1
k = 0
while(int(p / i) > 0):
    if(k + (int((p % i) / j) + 1) * j > d):
        break
    if(int(p % i / j)  != 9):
        k += (int((p % i) / j) + 1) * j
        p -= (int((p % i) / j) + 1) * j
        
    i *= 10
    j *= 10

print(p)
