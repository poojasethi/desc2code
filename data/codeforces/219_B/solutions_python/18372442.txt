#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

def f(p, k):
    return p - p % 10**k - 1

p, d = ia()

for k in range(100, 0, -1):
    np = f(p+1, k)
    if p - np<=d:
        print np; break
else:
    print p
