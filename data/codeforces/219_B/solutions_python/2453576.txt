a, b = map(str, raw_input().split())
rev_a = a[::-1]
l = len(a)
a = int(a)
_a = a
b = int(b)
win = a
i = 0
for k in rev_a:
    if int(k) != 9:
        break
    i = i + 1
while i < l - 1:
    n = pow(10, i + 1)
    k = a % n
    if k + 1 == n:
        win = a
        i = i + 1
        continue
    a -= (k+1)
    if _a - a > b:
        print win
        exit()
    win = a
    i = i + 1
print win
