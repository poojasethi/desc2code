s = raw_input().partition(" ")
p = str(int(s[0]))
d = str(int(s[2]))
if p.count("9") == len(p):
    print(p)
else:
    tmp = 1
    for i in range(len(p) - 1, 0, -1):
        curr = int(i * "9")
        s = p[len(p) - i : ]
        if int(s) < curr:
            s = "1" + s
        num = int(p) - (int(s) - curr)
        if num >= int(p) - int(d):
            print(num)
            tmp = 0
            break
    if tmp == 1:
        print(p);
