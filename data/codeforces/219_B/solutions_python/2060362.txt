def main():
    p, d = map(int, raw_input().split())
    ps, res, sdvig = str(p), '', 0

    for i in xrange(len(str(p))):
        val = 1 + int(ps[len(ps) - 1 - i]) - sdvig
        sdvig = int(val != 10)
        old, res = res or 0, str(val % 10) + res
        # print i, val, old, res
        if p - int(res) < p - d:
            return p - int(old)
    return p - int(res)

if __name__ == '__main__':
    print main()
