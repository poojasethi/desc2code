from math import *

table = [10 ** x for x in range(19)]
table9 = set([x - 1 for x in table if x != 1])
p, d = [int(x) for x in raw_input().split()]

n = int(floor(log10(p)))
done = False
while n > 0:
    if p % table[n] in table9:
        print p
        done = True
        break
    highest = p // table[n]
    target = highest * table[n] - 1
    if p - target <= d:
        print target
        done = True
        break
    n -= 1

if not done:
    print p
