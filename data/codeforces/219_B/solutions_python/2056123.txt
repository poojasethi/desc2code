def countLuck(n):
      ans = 0
      while n % 10 == 9:
            ans += 1
            n //= 10
      return ans


price, delta = map(int, raw_input().split())

best = price
for i in range(1, 25):
      decrease = (int('1' + str(price)[-i:]) - int('9' * i))
      if price < decrease or decrease > delta:
            break
      result = price - decrease
      if (countLuck(result) == countLuck(best) and result > best) or countLuck(result) > countLuck(best):
            best = result
print best
