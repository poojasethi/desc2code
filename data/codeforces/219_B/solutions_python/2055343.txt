import sys

input = sys.stdin

Ps, Ls = input.readline().split()
Pn = int(Ps)
Ln = int(Ls)
L = len(Ps)

def nines(s):
	L = len(s)
	cnt = 0
	i = 0
	while i < L:
		if s[L - 1 - i] != '9':
			break
		cnt += 1
		i += 1
	return cnt

results = [(nines(Ps), Pn)]
for i in xrange(1, L):
	Cs = Ps[:L - i] + '9' * i
	Cn = int(Cs)
	
	Dn = Cn - int('1' + '0' * i)
	Ds = str(Dn)
	if Pn - Dn <= Ln:
		results.append((nines(Ds), Dn))

print max(results)[1]
