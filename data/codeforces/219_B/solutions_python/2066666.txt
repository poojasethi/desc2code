def dgt(p, i):
    return (p / (10 ** i)) % 10

p, d = map(int, raw_input().split())

i = 0
k = 1
ind = True
while ind and i < len(str(p)):
    if dgt(p, i) != 9:
        d_dgt = dgt(p, i) + 1
        if (d >= k * d_dgt):
            d -= k * d_dgt
            p -= k * d_dgt
        else:
            ind = False
    i += 1
    k *= 10

print p
