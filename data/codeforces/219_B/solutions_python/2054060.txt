def cnt9(k):
    res = 0
    while k % 10 == 9:
        res += 1
        k /= 10
    return res

p, d = map(int, raw_input().split())
mod = 10
ans = p
while p % mod + 1 <= d:
    res = p - p % mod - 1
    if (ans is None) or (cnt9(ans) < cnt9(p - p % mod - 1)):
        ans = p - p % mod - 1
    mod *= 10
print ans
