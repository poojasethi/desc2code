n = int(raw_input())
prog = []
math = []
sport = []
t = map(int, raw_input().split())
for i in xrange(0, n):
	if (t[i] == 1):
		prog.append(i + 1)
	elif (t[i] == 2):
		math.append(i + 1)
	elif (t[i] == 3):
		sport.append(i + 1)
teams = zip(prog, math, sport)
print len(teams)
for x in teams:
	print " ".join(map(str, x))
