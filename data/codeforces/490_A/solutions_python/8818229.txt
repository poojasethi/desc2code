n=input()
arr=map(int,raw_input().split())
a=[]
b=[]
c=[]

for i in range(0,n):
	tmp=arr[i]
	if tmp==1:
		a.append(i+1)
	elif tmp==2:
		b.append(i+1)
	elif tmp==3:
		c.append(i+1)


x=a.__len__()
y=b.__len__()
z=c.__len__()
m=min(x,y,z)
print m
while(m):
	print a[x-1],b[y-1],c[z-1]
	x-=1
	y-=1
	z-=1

	m-=1
