#include <iostream>
using namespace std;
int a,i,n,v[5001][4];
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>a;
        v[0][a]++;
        v[v[0][a]][a]=i;
    }
    n=min(v[0][1],v[0][2]);
    n=min(n,v[0][3]);
    cout<<n<<'\n';
    for(i=1;i<=n;i++)
        cout<<v[i][1]<<" "<<v[i][2]<<" "<<v[i][3]<<'\n';
// system ("pause");
}
