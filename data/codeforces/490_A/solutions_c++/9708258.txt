#include <bits/stdc++.h>
using namespace std;

int main(){
	ios::sync_with_stdio(false);
	int i,n,x,G[4][6000],S[4]={0,0,0,0},ans;
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>x;
		S[x]++;
		G[x][S[x]]=i;
	}
	ans=min(S[1],min(S[2],S[3]));
	cout<<ans<<endl;
	for(i=1;i<=ans;i++){
		cout<<G[1][i]<<" "<<G[2][i]<<" "<<G[3][i]<<endl;
	}
}
