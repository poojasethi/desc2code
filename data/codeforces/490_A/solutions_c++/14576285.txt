#include <bits/stdc++.h>

using namespace std;

int M[5000][4];

int main(){
	int n,a;
	int V[4];
	while(cin>>n){
		memset(V,0,sizeof V);
		for(int i=0;i<n;i++){
			cin>>a;
			M[V[a]++][a]=i+1;
		}
		sort(V,V+4);
		cout<<V[1]<<"\n";
		for(int i=0;i<V[1];i++){
			cout<<M[i][1]<<" "<<M[i][2]<<" "<<M[i][3]<<"\n";
		}
	}
	return 0;
}
