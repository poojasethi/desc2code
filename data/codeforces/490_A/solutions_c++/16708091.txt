#include <bits/stdc++.h>
using namespace std;
vector<int>v[4];
int main()
{
	int n,ans,c;
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&c),v[c].push_back(i);
	ans=min(v[1].size(),min(v[2].size(),v[3].size()));
	printf("%d\n",ans);
	for (int i=0;i<ans;i++) printf("%d %d %d\n",v[1][i],v[2][i],v[3][i]);
	return 0;
}
