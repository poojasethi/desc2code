#include<iostream>
using namespace std;

int a[5000][4];

int main()
{
	int n,x,ans;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>x;
		a[++a[0][x]][x]=i;
	}
	ans=a[0][1];
	if(a[0][2]<ans) ans=a[0][2];
	if(a[0][3]<ans) ans=a[0][3];
	cout<<ans<<endl;
	for(int i=1;i<=ans;i++) cout<<a[i][1]<<" "<<a[i][2]<<" "<<a[i][3]<<endl;
	
	return 0;
}
