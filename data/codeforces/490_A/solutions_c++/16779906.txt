#include <bits/stdc++.h>
using namespace std;
int main(){
	vector<int>  v[4];
	int n,a;
	cin>>n;
	for (int i = 0; i < n; i++) 
	    cin>>a,v[a].push_back(i+1); 
	int ans=min(v[1].size(),min(v[2].size(),v[3].size()));
	cout<<ans<<endl;
	for (int i = 0; i < ans; i++) 
	    cout<<v[1][i]<<" "<<v[2][i]<<" "<<v[3][i]<<endl;
}
