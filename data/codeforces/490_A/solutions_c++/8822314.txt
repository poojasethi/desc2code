#include<cstdio>
#include<algorithm>
using namespace std;
int id[4][10005];
int n,i,ans,x;
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%d",&x),id[x][++id[x][0]]=i;
  ans=min(min(id[1][0],id[2][0]),id[3][0]);
  printf("%d\n",ans);
  for (i=1;i<=ans;i++)
    printf("%d %d %d\n",id[1][i],id[2][i],id[3][i]);
}
