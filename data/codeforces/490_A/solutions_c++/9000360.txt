#include<iostream>
#include<vector>
using namespace std;
vector <int> v[3];
int  main(){
	int n,temp;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>temp;
		temp--;
		v[temp].push_back(i);
	}
	int b=min(v[0].size(),min(v[1].size(),v[2].size()));
	cout<<b<<endl;
	for(int i=0;i<b;i++){
		cout<<v[0][i]<<" "<<v[1][i]<<" "<<v[2][i]<<endl;
	}
	return 0;
}
