#include <bits/stdc++.h>

using namespace std;

int n,x;
vector<int> v[5];

int main() {
	scanf("%d",&n);
	for(int i = 1; i<=n; i++) {
		scanf("%d",&x);
		v[x].push_back(i);
	}
	int ans = min(v[1].size(),min(v[2].size(),v[3].size()));
	printf("%d\n",ans);
	for(int i = 0; i<ans; i++)
		printf("%d %d %d\n",v[1][i],v[2][i],v[3][i]);
}
