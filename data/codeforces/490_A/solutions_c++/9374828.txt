#include<bits/stdc++.h>
using namespace std;

int main(){
	int n,i,m,p1=0,p2=0,p3=0;
	cin >> n ;
	int a1[n],a2[n],a3[n];
	for(i=1;i<=n;i++){
		cin >> m;
		if(m==1) a1[p1++]=i;
		if(m==2) a2[p2++]=i;
		if(m==3) a3[p3++]=i;
	}
	int ans = min(min(p1,p2),p3);
	cout << ans << endl;
	for(i=0;i<ans;i++){
		cout << a1[i] << " " << a2[i] << " " << a3[i] << endl;
	}
	return 0;
}
