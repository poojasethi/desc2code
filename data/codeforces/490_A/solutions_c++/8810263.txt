#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;

int main() {
	int n; vector<int> v[3];
	scanf("%d",&n);
	for(int j=1;j<=n;j++) {
		int i;
		scanf("%d",&i);
		v[i-1].push_back(j);
	}
	const int a=min(min(v[0].size(),v[1].size()),v[2].size());
	printf("%d\n",a);
	for(int i=0;i<a;i++)printf("%d %d %d\n",v[0][i],v[1][i],v[2][i]);
	return 0;
}
