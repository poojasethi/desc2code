#include <bits/stdc++.h>
using namespace std;
vector<int> a[3];
int main()
{
	int n,x;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		a[x-1].push_back(i+1);
	}
	int ans = min(min(a[0].size(),a[1].size()) , a[2].size());
	cout<<ans<<"\n";
	for(int i=0;i<ans;i++)cout<<a[0][i]<<" "<<a[1][i]<<" "<<a[2][i]<<"\n";
	return 0;
}
