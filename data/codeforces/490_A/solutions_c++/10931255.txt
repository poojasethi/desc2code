#include <bits/stdc++.h>
#define pb push_back
using namespace std;
vector <int> v[4];
int n,a;
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++) 
	{
		scanf("%d",&a);
		v[a].pb(i+1); 
	}
	int ans=min(v[1].size(),min(v[2].size(),v[3].size()));
	printf("%d\n",ans);
	for(int i=0;i<ans;i++) 
	{
		printf("%d %d %d\n",v[1][i],v[2][i],v[3][i]);
	}
	return 0;
}
