#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
main(){
	int n;
	cin>>n;
	vector<int>v[3];
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		v[x-1].push_back(i+1);
	}
	int a=min(v[0].size(),min(v[1].size(),v[2].size()));
	cout<<a<<"\n";
	for(int i=0;i<a;i++)cout<<v[0][i]<<" "<<v[1][i]<<" "<<v[2][i]<<"\n";
}
