#include <iostream>
#include <vector>

using namespace std;

vector<int> v[4];

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		int t;
		cin >> t;
		v[t].push_back(i + 1);
	}
	int ans = min(min(v[1].size(), v[2].size()), v[3].size());
	cout << ans << endl;
	for (int i = 0; i < ans; i++)
		cout << v[1][i] << " " << v[2][i] << " " << v[3][i] << endl;
	
}
