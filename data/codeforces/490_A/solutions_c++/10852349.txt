#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int n,i,j,b[4]={0},cnt;
cin>>n;
int a[n];
for(i=0;i<n;i++)
{
cin>>a[i];
b[a[i]]++;
}
sort(b,b+4);
cout<<b[1]<<"\n";
for(i=0;i<b[1];i++)
{
for(cnt=1;cnt<=3;cnt++)
for(j=0;j<n;j++)
{
if(a[j]==cnt)
{
cout<<j+1<<" ";
a[j]=0;
break;
}
}
cout<<"\n";
}
return 0;
}
