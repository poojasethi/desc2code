#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int n;
	cin>>n;
	vector <int> a(3,0);
	vector <vector <int> > v(3);
	int k;
	for(int i=0;i<n;i++)
	{
		cin>>k;
		v[k-1].push_back(i+1);
		a[k-1]++;
	}
	int u=min(a[0],a[1]);
	u=min(u,a[2]);
	cout<<u<<endl;
	for(int i=0;i<u;i++)
		cout<<v[0][i]<<" "<<v[1][i]<<" "<<v[2][i]<<endl;
	return 0;
}
