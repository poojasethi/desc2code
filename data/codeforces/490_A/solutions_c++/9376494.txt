#include<cstdio>
#include<vector>
using namespace std;

int n,x,len;
vector<int>a[5];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d", &x);
		a[x].push_back(i+1);
	}
	len = min(min(a[1].size(),a[2].size()),a[3].size());
	printf("%d\n", len);
	for(int i=0;i<len;i++)
		printf("%d %d %d\n", a[1][i],a[2][i],a[3][i]);

	return 0;
}
