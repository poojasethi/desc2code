#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,i,x;
vector<int> a,b,c;
scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%d",&x);
if(x==3)a.push_back(i+1);
else if (x==1)b.push_back(i+1);
else c.push_back(i+1);
}
n = min(a.size(),min(b.size(),c.size()));
printf("%d\n",n);
for(i=0;i<n;i++)
printf("%d %d %d\n",a[i],b[i],c[i]);
return 0;
}
