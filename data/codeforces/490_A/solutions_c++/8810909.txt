#include<iostream>

#include<vector>
using namespace std;
int main() {
	vector<int> a[3];
	int n;
	cin>>n;
	for(int i=1;i<=n;i++) {
		int x;
		cin>>x;
		a[x-1].push_back(i);
	}
	int ans = min(min(a[0].size(),a[1].size()),a[2].size());
	cout<<ans<<endl;
	for(int i=0;i<ans;i++)
		cout<<a[0][i]<<" "<<a[1][i]<<" "<<a[2][i]<<endl;
	
}
