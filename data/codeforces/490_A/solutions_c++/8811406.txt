#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	cin>>n;
	vector< vector<int> > t(3);
	int a;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		t[a-1].push_back(i);
	}
	int r=min(t[0].size(),min(t[1].size(),t[2].size()));
	cout<<r<<endl;
	for(int i=0;i<r;i++)
	{
		cout<<t[0][i]+1<<" "<<t[1][i]+1<<" "<<t[2][i]+1<<endl;
	}
	return 0;
}
