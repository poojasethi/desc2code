#include <cstdio>
int a[4][5000], x[4];
int main() {
  int n, k;
  scanf("%d", &n);
  for (int i = 1; i <= n; i++) {
    scanf("%d", &k);
    a[k][x[k]++] = i;
  }
  k = x[1];
  if (x[2] < k) k = x[2];
  if (x[3] < k) k = x[3];
  printf("%d\n", k);
  for (int i = 0; i < k; i++) {
    printf("%d %d %d\n", a[1][i], a[2][i], a[3][i]);
  }
  return 0;
}
