#include<iostream>
using namespace std;
int n, i, a, b[4][5001], c[4];
int main()
{
    cin>>n;
    for(i = 1; i <= n; i ++)
    {
        cin>>a;
        b[a][++ c[a]] = i;
    }
    cout<<min(c[1], min(c[2], c[3]))<<endl;
    while(c[1] && c[2] && c[3])
        cout<<b[1][c[1] --]<<" "<<b[2][c[2] --]<<" "<<b[3][c[3] --]<<endl;
    return 0;
}
