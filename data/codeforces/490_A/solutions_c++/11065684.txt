#include <bits/stdc++.h>
using namespace std;
int main()
{
    int a,i,n,v[5500][4];

    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>a;
        v[0][a]++;
        v[v[0][a]][a]=i;
    }
    n=min(v[0][1],v[0][2]);
    n=min(n,v[0][3]);
    cout<<n<<'\n';
    for(i=1;i<=n;i++)
        cout<<v[i][1]<<" "<<v[i][2]<<" "<<v[i][3]<<endl;;
return 0;
}
