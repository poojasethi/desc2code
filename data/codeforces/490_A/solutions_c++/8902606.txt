#import<bits/stdc++.h>
using namespace std;

vector<int> s[5];
int n, mn;
int main() {
	int i, t;
	cin>>n;
	for (i = 1; i <= n; i++) {
		cin>>t;
		s[t].push_back(i);
	}
	mn = min(min(s[1].size(), s[2].size()), s[3].size());
	cout<<mn<<endl;
	for (i = 0; i < mn; i++) {
		printf("%d %d %d\n", s[1][i], s[2][i], s[3][i]);
	}
}
