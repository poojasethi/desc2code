#include<bits/stdc++.h>

using namespace std;

int main(){
	int N,aux;
	cin>>N;
	vector<vector<int> >v(3);
	for(int i = 1;i<=N;++i){
		cin>>aux;
		v[aux-1].push_back(i);	
	}
	int teams = min(v[0].size(),min(v[1].size(),v[2].size()));
	cout<<teams<<endl;
	for(int i = 0;i<teams;++i)cout<<v[0][i]<<" "<<v[1][i]<<" "<<v[2][i]<<endl;
	
	

}
