#include <bits/stdc++.h>
using namespace std ;
int main(){
int n , x ;
vector<int> A[4] ;
cin >> n ; 
for(int i = 1 ; i <= n ; i++)
{
cin >> x ;
A[x].push_back(i) ;
}
int m = min(A[1].size() , min(A[2].size() , A[3].size() )) ;
cout << m << "\n" ;

for(int i = 0 ; i < m ; i++)
cout << A[1][i] << " " << A[2][i] << " " << A[3][i] <<"\n" ;

return 0 ;
}
