#include<iostream>
#include<cstdio>
using namespace std;
const int MaxV = 5000 + 10;
int n,t[5][MaxV],ar;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&ar);
		t[ar][++t[ar][0]]=i;
	}
	int ans=min(t[3][0],min(t[1][0],t[2][0]));
	printf("%d\n",ans);
	for(int i=1;i<=ans;i++) printf("%d %d %d\n",t[1][i],t[2][i],t[3][i]);
	return 0;
}
