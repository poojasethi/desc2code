#include <bits/stdc++.h>

using namespace std;

vector<int>v[4];
int a[4];

int main()
{
int n;
cin>>n;
for(int i=1;i<=n;i++)
{
int x;
cin>>x;
a[x]++;
v[x].push_back(i);
}

int l=min(a[1] , min(a[2],a[3])  );

cout<<l<<endl;
for(int i=0;i<l;i++)
{
cout<<v[1][i]<<' '<<v[2][i]<<' '<<v[3][i]<<endl;
}

return 0;
}
