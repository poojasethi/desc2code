#include <iostream>
using namespace std;
int k,i,n,a[6000][5];
main()
{
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>k;
        a[0][k]++;
        a[a[0][k]][k]=i;
    }
    n=min(a[0][1],a[0][2]);
    n=min(n,a[0][3]);
    cout<<n<<endl;
    for(i=1;i<=n;i++)
        cout<<a[i][1]<<" "<<a[i][2]<<" "<<a[i][3]<<endl;
}
