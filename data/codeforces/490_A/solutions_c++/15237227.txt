using namespace std;

int main(){
  int n,m,k,i,l,p,j,s,x;
  int a[1009]={0},b[4][10009];
  
  cin>>n;

    a[x]++;
    b[x][a[x]]=i;
  }

  cout<<m<<endl;

    cout<<b[1][i]<<" "<<b[2][i]<<" "<<b[3][i]<<endl;
}
