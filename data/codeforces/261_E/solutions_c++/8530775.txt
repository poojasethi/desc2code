#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int P[105];
int a[3000005];
int f[3000005];
bool v[3000005];
int top,n,L,R,ans;
void Dfs(int x,int y)
{
	if(x>top)
	{
		a[++n]=y;
		return ;
	}
	for(;y<=R/P[x];y*=P[x])
		Dfs(x+1,y);
	Dfs(x+1,y);
}
bool Check(int x)
{
	for(int i=2;i*i<=x;i++)
		if(x%i==0)
			return 0;
	return 1;
}
int main()
{
	int p;
	scanf("%d%d%d",&L,&R,&p);
	for(int i=2;i<=p;i++)
		if(Check(i))
			P[++top]=i;
	Dfs(1,1);
	sort(a+1,a+1+n);
	memset(f,127,sizeof(f));
	f[1]=0;
	for(int t=2;t<=p;t++)
		for(int i=t,l=1;i<=n;i++)
		{
			while(a[l+1]<=a[i]/t)l++;
			if(a[l]*t==a[i]&&f[i]>f[l]+1&&f[l]+t<p)
				f[i]=f[l]+1,v[i]=1;
		}
	for(int i=2;i<=n;i++)
		ans+=a[i]>=L&&a[i]<=R&&v[i];
	printf("%d\n",ans);
}
