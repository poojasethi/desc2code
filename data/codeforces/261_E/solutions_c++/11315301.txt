#include <bits/stdc++.h>

#define	st first
#define	nd second
#define	mp make_pair
#define	pb push_back
#define	lli long long int
#define	all( gg )	gg.begin(),gg.end()
#define	foreach( gg,itit )	for( typeof(gg.begin()) itit=gg.begin();itit!=gg.end();itit++ )
#define	FP( ii,aa,bb ) for( lli ii=aa;ii<=bb;ii++ )
#define	FM( ii,aa,bb ) for( lli ii=aa;ii>=bb;ii-- )
#define	debug(ccc)	cout << #ccc << " = " << ccc << endl;

#define	mod	1000000007LL

using namespace std;

int	l,r,p;

vector<int>	v,dp,can;

int main(){
	cin >> l >> r >> p;
	v.pb( 1 );
	FP( i,2,p ){
		int	ok=1;
		FP( j,2,i-1 )
			ok &= (i%j!=0);
		if( ok )
			for( int j=0;j<v.size();j++ )
				if( v[j]*i<=r )
					v.pb( v[j]*i );
	}
	sort( all(v) );
	dp .resize( v.size(),mod );
	dp[0] = 0;
	can.resize( v.size() );
	for( int P=2;P<=p;P++ ){
		int	div=0;
		for( int j=0;j<v.size();j++ )
			if( v[j]%P==0 ){
				if( dp[j]>dp[div]+1 ){
					dp[j] = dp[div]+1;
					if( dp[j]+P<=p )	can[j] = 1;
				}
				div++;
			}
	}
	int	res = 0;
	for( int j=0;j<v.size();j++ )
		if( can[j] and v[j]>=l )	res++;
	cout << res << endl;
}
