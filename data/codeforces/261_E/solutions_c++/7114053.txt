#include<cstdio>
#include<algorithm>
#include<cstring>
#define N 3000005
using namespace std;
int f[N],flag[N],a[N],check[N],prime[N];
int L,R,P,cnt,day,n,up,ans,i,j;
void dfs(int k,int now)
{
  if (!k) {a[++n]=now;return;}
  dfs(k-1,now);
  while (R/now>=prime[k])
    now*=prime[k],dfs(k-1,now);
}
int main()
{
  scanf("%d%d%d",&L,&R,&P);
  for (i=2;i<=P;i++) 
  {
    if (!flag[i]) prime[++cnt]=i;
    for (j=1;j<=cnt&&prime[j]*i<=P;j++) flag[prime[j]*i]=1;
  }
  dfs(cnt,1);
  sort(a+1,a+n+1);
  memset(f,60,sizeof(f));f[1]=0;
  for (day=2;day<=P;day++)
    for (i=1,up=1;i<=n;i++)
    {
      for (;a[i]*day>a[up]&&up<=n;up++);if (up>n) break;
      f[up]=min(f[up],f[i]+1);
      if (f[up]+day<=P) check[up]=1;
    }
  for (i=1;i<=n;i++)
    ans+=check[i]*(a[i]>=L);
  printf("%d",ans);
  return 0;
}
