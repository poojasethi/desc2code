#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
int n,a[1001],m,L,R,p,ans;
int g[3001000],h[3001000];
bool F[3001000];
bool isprime(int n){
    for(int i=2;i*i<=n;i++) if(n%i==0) return 0;
    return 1;
}
void dfs(int x,int sum){
    if(x>n){g[++m]=sum; return;}
    dfs(x+1,sum);
    while(1LL*a[x]*sum<=R) sum*=a[x],dfs(x+1,sum);
}
int main(){
    scanf("%d%d%d",&L,&R,&p);
    for(int i=2;i<=p;i++) if(isprime(i)) a[++n]=i;
    dfs(1,1);
    sort(g+1,g+1+m);
    for(int i=1;i<=m;i++) h[i]=p+1; h[1]=0;
    for(int i=2;i<=p;i++){
        int k=1;
        for(int j=1;j<=m;j++){
            if(1LL*g[j]*i>R) break;
            for(;g[k]<g[j]*i;k++);
            h[k]=min(h[k],h[j]+1);
        }
        for(int j=1;j<=m;j++) F[j]|=(i+h[j]<=p);
    }
    for(int j=1;j<=m;j++) ans+=((g[j]>=L)&F[j]);
    printf("%d\n",ans);
}
