#include <cmath>
#include <cstdio>
#include <iostream>
#include <cstdlib>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;
int a[1005],t[3000005],dp[3000005],ans,o,p,i,j,k,cnt,l,r,V[3000005];
bool FLAG;
void dfs(int x,int y)
{
    if (y==o+1) {t[++cnt]=x;return ;}
    while (1)
    {
        dfs(x,y+1);
        if (1000000000/x<a[y]) return;
        x*=a[y];
    }
}
int cmp(int i,int j) {return i<j;}
int main()
{
    cin>>l>>r>>p;
    for (i=2; i<=100; i++)
    {
        FLAG=false;
        for (j=2; j<i; j++) if (i%j==0) FLAG=true;
        if (!FLAG) a[++o]=i;
    }
    dfs(1,1);
    sort(t+1,t+cnt+1,cmp);
    dp[1]=0;
    for (i=2; i<=cnt; i++) dp[i]=453266144;
    for (i=2; i<=p; i++)
    {
        for (k=1,j=2; j<=cnt; j++)
        if (t[k]*i==t[j])
        {
            dp[j]=min(dp[j],dp[k]+1);
            if (dp[j]+i<=p) V[j]=1;
            k++;
        }
    }
    for (j=2; j<=cnt; j++) if (V[j] && t[j]>=l && t[j]<=r) ans++;
    cout<<ans;
    return 0;
}
