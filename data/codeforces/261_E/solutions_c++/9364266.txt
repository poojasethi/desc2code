#include<bits/stdc++.h>
using namespace std;
int l,r,p,a[4000001],f[4000001],pnum,pri[101],n,ans;
bool ck[4000001];
void dfs(long long x,int t)
{
	if(t>pnum)return;
	while(x<=r)
	{
		dfs(x,t+1);
		x=x*pri[t];
		if(x<=r)a[++n]=x;
	}
}
int main()
{
	scanf("%d%d%d",&l,&r,&p);
	for(int i=2;i<=p;i++)
	{
		bool ok=true;
		for(int j=2;j*j<=i;j++)
			if(i%j==0)
				ok=false;
		if(ok)pri[++pnum]=i;
	}
	a[++n]=1;dfs(1,1);sort(a+1,a+n+1);
	for(int i=2;i<=n;i++)f[i]=p+1;
	for(int t=1;t<=p;t++)
	{
		int now=1;
		for(int i=1;i<=n;i++)
		{
			long long tmp=a[i]*t;
			while(now<=n&&a[now]<tmp)++now;
			if(now>n)break;
			if(a[now]==tmp&&f[now]>f[i]+1)
			{
				f[now]=f[i]+1;
				if(f[now]+t<=p)ck[now]=true;
			}
		}
	}
	for(int i=1;i<=n;i++)if(a[i]>=l&&ck[i])++ans;
	cout<<ans<<endl;
}
