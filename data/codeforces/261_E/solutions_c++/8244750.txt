#include <bits/stdc++.h>
#define maxn 3000005
using namespace std;

int L,R,P,n,m,p[30],q[maxn],f[maxn];
bool b[105],c[maxn];

void dfs(int t,int s)
{
	if (t>n) {q[++m]=s; return;}
	dfs(t+1,s);
	for (;1ll*s*p[t]<=R;) dfs(t+1,s*=p[t]);
}

void doit()
{
	scanf("%d%d%d",&L,&R,&P);
	memset(b,1,sizeof(b));
	for (int i=2; i<=P; i++) if (b[i])
		for (int j=i*i; j<=P; j+=i) b[j]=0;
	for (int i=2; i<=P; i++) if (b[i]) p[++n]=i;
	memset(f,63,sizeof(f));
	dfs(1,1),sort(q+1,q+m+1),f[1]=0;
	for (int i=2; i<=P; i++)
		for (int j=1,k=2; j<=m; j++) if (f[j]+i<=P)
		{
			for (;k<=m&&q[k]<q[j]*i; k++);
			if (k<=m)
			{
				f[k]=min(f[k],f[j]+1);
				if (f[k]+i<=P) c[k]=1;
			}
		}
	int ans=0;
	for (int i=1; i<=m; i++) if (L<=q[i]&&c[i]) ans++; 
	printf("%d\n",ans);
}

int main()
{
	doit();
	return 0;
}
