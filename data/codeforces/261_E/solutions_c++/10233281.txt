#include <cstdio>
#include <iostream>
#include <algorithm>
#define P 101
#define N 3000000
using namespace std;
int num,prime[P],del[P];
int tot,a[N],f[N],g[N];
int main(){
	int l,r,p;
	scanf("%d%d%d",&l,&r,&p);
	a[tot=1]=1;
	for(int i=2;i<=p;++i){
		if(!del[i]){
			prime[++num]=i;
			for(int j=1;j<=tot;++j)
		        if((long long)a[j]*i<=(long long)r)
				    a[++tot]=a[j]*i;
		}
		for(int j=1;j<=num && i*prime[j]<=p;++j){
			del[i*prime[j]]=1;
			if(!(i%prime[j]))break;
		}
		
	}
	sort(a+1,a+tot+1);
	for(int i=2;i<=tot;++i)f[i]=P;
	for(int i=2;i<=p;++i){
		int k=1;
	    for(int j=2;j<=tot;++j)
		    if(a[j]%i==0){
		    	f[j]=min(f[j],f[k++]+1);
		    	if(a[j]>=l && i+f[j]<=p)g[j]=1;
		    }
    }
    int ans=0;
    for(int i=1;i<=tot;++i)ans+=g[i];
    printf("%d",ans);
}
