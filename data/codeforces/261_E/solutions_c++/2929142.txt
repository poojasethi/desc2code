#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
int l, r;
int dp[3000010], ans;
int c[3000010], n;
bool u[3000010];
int main(){
    int p, i, k;
    scanf("%d%d%d", &l, &r, &p);
    c[n++] = 1;
    for (k = 2; k < p; k++){
        for (i = 2; i * i <= k; i++){
            if (k % i == 0){
                break;
            }
        }
        if (i * i <= k) continue;
        for (i = 0; i < n; i++){
            if (1LL * c[i] * k <= r){
                c[n++] = c[i] * k;
            }
        }
    }
    sort(c, c + n);
    for (i = 0; i < n; i++) dp[i] = p + 1;
    dp[0] = 0;
    u[0] = true;
    for (k = 1; k < p; k++){
        for (i = 0; i < n; i++) dp[i]++;
        int x = 0;
        for (i = 0; i < n && 1LL * c[i] * k <= r; i++){
            if (dp[i] < p){
                while (c[x] < c[i] * k) x++;
                dp[x] = min(dp[x], dp[i] + 1);
                u[x] = true;
            }
        }
    }
    for (i = 0; i < n; i++) if (c[i] >= l && u[i]) ans++;
    printf("%d\n", ans);
    return 0;
}
