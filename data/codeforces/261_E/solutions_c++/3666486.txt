#include<cstdio>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=110,mm=5100000;
int l,r,p,n=0,tot=0;
int pr[mn],a[mm],f[mm];
bool fl[mn],g[mm];
void dfs(int k,int v)
{
    a[++n]=v;
    for(;k<=tot;++k)
    {
        if((long long)v*pr[k]>r)break;
        dfs(k,v*pr[k]);
    }
}
int main()
{
    scanf("%d%d%d",&l,&r,&p);
    int i,j,k;
    fo(i,2,p)
    {
        if(!fl[i])pr[++tot]=i;
        else continue;
        fo(j,2,p/i)fl[i*j]=1;
    }
    dfs(1,1);
    sort(a+1,a+1+n);
    fill(f+2,f+n+1,p+1);
    f[1]=0,g[1]=1;
    fo(i,2,p)
    {
        k=1;
        fo(j,2,n)
        {
            while(k<=n&&a[k]*i<a[j])++k;
            if(k>n)break;
            if(a[k]*i==a[j])
            {
                f[j]=min(f[j],f[k]+1);
                if(f[j]+i<=p)g[j]=1;
            }
        }
    }
    int ans=0;
    fo(i,1,n)
        if(a[i]>=l&&g[i])++ans;
    printf("%d\n",ans);
    return 0;
}
