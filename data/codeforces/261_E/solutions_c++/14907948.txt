#include <bits/stdc++.h>
using namespace std;

int l, r, p, pri[105], tot, n, c[3000010], f[3000010];
bool v[3000010];

void getpri(int n)
{
	for (int i = 2; i <= n; ++i)
	{
		if (!v[i]) pri[++tot] = i;
		for (int j = 1; i * pri[j] <= n; ++j)
		{
			v[i * pri[j]] = true;
			if (i % pri[j] == 0) break;
		}
	}
}

void add(int x, int now)
{
	c[++n] = x;
	for (int i = now; i <= tot; ++i)
		if (1ll * x * pri[i] <= r)
			add(x * pri[i], i);
}

int main()
{
//	freopen("schizophrenia.in", "r", stdin);
//	freopen("schizophrenia.out", "w", stdout);
	cin >> l >> r >> p;
	getpri(p); add(1, 1);
	sort(c + 1, c + n + 1);
	memset(f, 0x3f, sizeof f);
	f[1] = 0;
	for (int i = 2; i <= p; ++i)
	{
		int k = 1;
		for (int j = 1; j <= n; ++j)
		{
			if (1ll * c[j] * i > r) break;
			while (c[k] < c[j] * i) ++k;
			f[k] = min(f[k], f[j] + 1);
			if (f[k] + i <= p) v[k] = 1;
		}
	}
	int ans = 0;
	for (int i = 1; i <= n; ++i)
		if (c[i] >= l) ans += v[i];
	cout << ans << endl;
	return 0;
}
