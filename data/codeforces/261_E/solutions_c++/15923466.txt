#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<string>
#include<ctime>
using namespace std;
typedef long long LL; 
const int imax=3000000+229;
int n,tot; int l,r;
int a[imax]; 
int f[imax],ind[imax];

int main()
{
	scanf("%d%d%d",&l,&r,&n);
	a[++tot]=1;
	for(int i=2;i<=n;i++) 
	{
		bool flag=1;
		for(int j=2;j<i;j++) if((i%j)==0) { flag=false; break;}
		if(flag)
		{
			for(int z=1;z<=tot;z++) 
				if(a[z]*(LL)i<=r) a[++tot]=a[z]*i;	
		}	
	}
	for(int i=1;i<=tot;i++) f[i]=100000229;
	f[1]=0; 
	sort(a+1,a+tot+1); 
	for(int i=2;i<=n;i++)
		for(int j=1,now=1;j<=tot;j++)
			if((a[j]%i)==0)
			{
				if(f[j]>f[now]+1)
				{ 
					f[j]=f[now]+1;//a变化的次数 
					if(f[j]+i<=n) ind[j]=1; //b变化的次数 	
				}
				now++;		
			}
	int ans=0; //printf("%d\n",clock());
	for(int i=1;i<=tot;i++) if(ind[i] && a[i]>=l) ans++;
	printf("%d\n",ans);
	return 0;
}
