#include <bits/stdc++.h>

using namespace std;

const int N = 3100000;

int l,r,p,m,rs;
int a[N],f[N],g[N];

int main()
{
	scanf("%d%d%d",&l,&r,&p);
	a[0]=1;m=1;
	for(int i=2;i<=p;i++){
		int flg=1;
		for(int j=2;j<i && flg;j++) flg=i%j;
		if(!flg) continue;
		for(int j=0;j<m;j++) if(1ll*a[j]*i<=r) a[m++]=a[j]*i;
	}
	sort(a,a+m);
	memset(f,0x7f,sizeof(f));
	f[0]=0;
	for(int i=2;i<p;i++)
		for(int j=0,k=0;j<m;j++) if(a[j]%i==0){
			if(f[k]+1<f[j]){
				f[j]=f[k]+1;
				if(f[j]+i<=p) g[j]=1;
			}
			k++;
		}
	for(int i=0;i<m;i++) if(a[i]>=l && g[i]) rs++;
	printf("%d\n",rs);
	return 0;
}
