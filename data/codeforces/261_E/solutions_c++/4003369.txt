#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#define int64 long long
#define foru(i,a,b) for(int i=a;i<=b;++i)

using namespace std;
bool vi[210],can[3100000];
int x[3100000],pr[210],f[3100000],l,r,p,ans,tot,cnt,q;
void dfs(int a,int b)
{
	x[++tot]=a;
	for(;b<=cnt;++b)
	{
		if((int64)a*pr[b]>r) break;
		dfs(a*pr[b],b);
	}
}
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>l>>r>>p;
	memset(vi,1,sizeof(vi));
	foru(i,2,p)
	{
		if(vi[i]) pr[++cnt]=i;
		foru(j,1,cnt)
		{
			if(i*pr[j]>p) break;
			vi[i*pr[j]]=0;
			if(i%pr[j]==0) break;
		}
	}
	dfs(1,1);
	sort(x+1,x+tot+1);
	f[1]=0; foru(i,2,tot) f[i]=p+1;
	foru(i,2,p)
	{
		q=1;
		foru(j,1,tot)
		{
			if(x[j]<=r&&l<=x[j]&&f[j]+i<=p) can[j]=1;
			while(q<tot&&x[j]*i>x[q]) ++q;
			if(x[j]*i==x[q]) f[q]=min(f[q],f[j]+1);
		}
	}
	foru(i,1,tot) ans+=can[i];
	cout<<ans<<endl;
	return 0;
}
