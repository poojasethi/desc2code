#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <map>
#define N 110
#define M 3000010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,l,r,an;
int f[N],p[N],g[M],h[M],F[M];
void dfs(int x,int y) {
	if (x>*p) {
		g[++m]=y;
		return;
	}
	dfs(x+1,y);
	if (1ll*y*p[x]<=r) dfs(x,p[x]*y);
}
int main() {
	scanf("%d%d%d",&l,&r,&n);
	For(i,2,n) {
		if (!f[i]) f[i]=p[++*p]=i;
		for (j=1;j<=*p&&p[j]<=f[i]&&i*p[j]<=n;j++) f[i*p[j]]=p[j];
	}
	dfs(1,1);
	sort(g+1,g+m+1);
	For(i,1,m) h[i]=n+1;
	h[1]=0;
	For(i,2,n) {
		k=1;
		For(j,1,m) if (1ll*g[j]*i<=r) {
			if (h[j]>n) continue;
			int A=g[j]*i;
			for (;g[k]<A;k++);
			h[k]=min(h[k],h[j]+1);
		} else break;
		For(j,1,m) F[j]|=i+h[j]<=n;
	}
	For(j,1,m) if (g[j]>=l) an+=F[j];
	printf("%d\n",an);
	return 0;
}
