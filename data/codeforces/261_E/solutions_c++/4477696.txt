#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
int i,cnt,L,R,p,a[120],j,n,v[3001000],f[3001000],ans,can[3001000],tot,l;
bool isprime(int x){
	for(int i=2;i*i<=x;++i)if(x%i==0)return 0;
	return 1;
}
void dfs(int l,int x){
	if(l==0){
		v[++tot]=x;
		return;
	}
	for(;;){
		dfs(l-1,x);
		if(x>R/a[l])break;
		x*=a[l];
	}
}
int main(){
	scanf("%d%d%d",&L,&R,&p);
	for(i=2;i<=p;++i)if(isprime(i))
		a[++n]=i;
	dfs(n,1);
	sort(v+1,v+tot+1);
	for(i=1;i<=tot;++i)f[i]=1e9;
	f[1]=0;
	can[1]=1;
	for(i=2;i<=p;++i){
		l=1;
		for(j=1;j<=tot;++j){
			while(l<=tot && v[l]<v[j]*i)l++;
			if(l<=tot){
				f[l]=min(f[l],f[j]+1);
				if(f[l]+i<=p)can[l]=1;
			}
		}
	}
	for(i=1;i<=tot;++i)if(v[i]>=L && v[i]<=R && can[i])ans++;
	printf("%d\n",ans);
}
