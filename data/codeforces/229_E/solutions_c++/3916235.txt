#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
struct ppp{
	int x,be;
}a[1200];
double C[1200][1200],f[1200][1200];
int i,j,k,l,n,m,pnum[1200],num[1200],x,tot,bo[1200];
bool cmp(const ppp&a,const ppp &b){
	return a.x>b.x;
}
int main(){
	scanf("%d%d",&m,&n);
	for(i=0;i<=1000;++i)
		for(j=0;j<=i;++j)C[i][j]=j==0?1:C[i-1][j-1]+C[i-1][j];
	for(i=1;i<=n;++i){
		scanf("%d",&num[i]);
		for(j=1;j<=num[i];++j)scanf("%d",&x),a[++tot].x=x,a[tot].be=i;
	}
	sort(a+1,a+tot+1,cmp);
	for(i=1;i<=tot;i=j){
		for(j=i;j<=tot && a[j].x==a[i].x;++j);
		if(m>j-i){
			m-=j-i;
			for(k=i;k<j;++k)pnum[a[k].be]++;
		}
		else{
			double tmp=1;
			for(k=i;k<j;++k)bo[a[k].be]=1;
			for(k=1;k<=n;++k)if(!bo[k])tmp/=C[num[k]][pnum[k]];
			f[i][0]=tmp;
			for(k=i;k<j;++k){
				for(l=0;l<=m;++l){
					f[k+1][l]+=f[k][l]/C[num[a[k].be]][pnum[a[k].be]];
					f[k+1][l+1]+=f[k][l]/C[num[a[k].be]][pnum[a[k].be]+1];
				}
			}
			printf("%.10lf\n",f[j][m]/C[j-i][m]);
			return 0;
		}
	}
}
