#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define LL long long
#define pb push_back
using namespace std;
int n,m,x,y,nt;
vector <int> all,a[1005];
double c[1005][1005],f[1005][1005],g[1005][1005];
bool cmp(int x,int y){return x>y;}
int main(){
    cin>>n>>m;
    c[0][0]=1;
    fo(i,1,1000) fo(j,0,i){
        c[i][j]=(!j)?1:c[i-1][j]+c[i-1][j-1];
    }
    fo(i,0,m-1){
        scanf("%d",&nt);
        fo(j,1,nt) scanf("%d",&x),a[i].pb(x),all.pb(x);
    }
    sort(all.begin(),all.end(),cmp);
    f[0][0]=g[0][0]=1;
    fo(i,0,m-1){
        int x=y=0;
        int nt=a[i].size();
        fo(j,0,nt-1) 
            if (a[i][j]>all[n-1]) x++;
            else if (a[i][j]==all[n-1]) y++; 
        fo(j,0,n) fo(k,0,y) if(j+x+k<=n)
            g[i+1][j+x+k]+=g[i][j],
            f[i+1][j+x+k]+=f[i][j]/c[nt][x+k];
    }
    printf("%.10f\n",f[m][n]/g[m][n]);
    return 0;
}
