#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int cnt,tot,n,m,i,j,k,st[1002],p[1002],dl[1002],must[1002],num[1002],lim;
bool ex[1002];
double f[1002][1002],g[1002],a,b;
int main()
{
    cin >> n >> m;
    st[1]=1;
    for (i=1;i<=m;i++)
    {
        cin >> k;
        st[i+1]=tot+k+1;
        for (;k;k--)
            cin >> p[++tot],dl[tot]=p[tot];
        sort(p+st[i],p+st[i+1]);
    }
    sort(dl+1,dl+tot+1);
    lim=dl[tot-n+1];
    for (i=1;i<=m;i++)
    {
        for (j=st[i+1]-1;j>=st[i] && p[j]>lim;j--) cnt++,must[i]++;
        ex[i]=j>=st[i] && p[j]==lim;
        num[i]=st[i+1]-st[i];
    }
    for (i=m;i>=1;i--) g[i]=g[i+1]+ex[i];
    f[0][0]=1;
    for (i=1;i<=m;i++)
    {
        a=1;
        for (j=must[i];j>=1;j--) (a*=j)/=num[i]-(must[i]-j);
        if (!ex[i])
           for (j=0;j<=i;j++) f[i][j]=f[i-1][j]*a;
        else
        {
            b=a*(must[i]+1)/(num[i]-must[i]);
            for (j=0;j<i;j++)
                f[i][j]+=f[i-1][j]*a*(1-(n-cnt-j)/g[i]),f[i][j+1]+=f[i-1][j]*b*(n-cnt-j)/g[i];
        }
    }
    printf("%.10f\n",f[m][n-cnt]);
    return 0;
}
