#include<cstdio>
#include<vector>
#include<algorithm>
#define pb push_back
#define fo(i,a,b) for(i=a;i<=b;++i)
#define fd(i,a,b) for(i=a;i>=b;--i)
using namespace std;
const int mn=1010;
double f[mn],g[mn],c[mn][mn];
vector<int>P,p[mn];
int n,m,lo,sa,sb;
bool cmp(int a,int b){return a>b;}
void dp(vector<int>&p)
{
    int i,j,a=0,b=0;
    for(i=0;i<p.size();++i)
        if(p[i]>lo)
            ++a;
        else if(p[i]==lo)++b;
    sa+=a,sb+=b;
    double f0,g0;
    fd(i,sb,0)
    {
        f0=g0=0.0;
        fo(j,0,b)
        {
            if(j>i)break;
            g0+=g[i-j],f0+=f[i-j]/c[p.size()][a+j];
        }
        f[i]=f0,g[i]=g0;
    }
}
int main()
{
    scanf("%d%d",&m,&n);
    int i,j,k;
    fo(i,1,n)
    {
        scanf("%d",&k);
        while(k--)
        {
            scanf("%d",&j);
            P.pb(j),p[i].pb(j);
        }
    }
    sort(P.begin(),P.end(),cmp);
    lo=P[m-1];
    f[0]=g[0]=1,c[0][0]=1;
    sa=sb=0;
    fo(i,1,P.size())
    {
        c[i][0]=1;
        fo(j,1,i)c[i][j]=c[i-1][j-1]+c[i-1][j];
    }
    fo(i,1,n)
        dp(p[i]);
    printf("%.10lf\n",f[m-sa]/g[m-sa]);
    return 0;
}
