#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m,total,sum[1005],need[1005],a[1005];
long double c[1005][1005],f[1005][1005],g[1005][1005];
struct huo{
	int a,b;
}s[1000005];
bool cmp(huo a,huo b)
{
	return a.a>b.a;
}
int main()
{
	scanf("%d %d",&n,&m);
	int i,j,k;
	for(i=1;i<=m;i++)
	{
		scanf("%d",&a[i]);
		for(j=1;j<=a[i];j++)
		{
			scanf("%d",&s[++total].a);
			s[total].b=i;
		}
	}
	sort(s+1,s+total+1,cmp);
	int l=n-1,r=n,p=s[n].a;
	while(p==s[r].a)sum[s[r++].b]++;
	while(p==s[l].a)sum[s[l--].b]++;
	int x=n-l;
	for(i=1;i<=l;i++)
		need[s[i].b]++;
	c[0][0]=1;
	for(i=1;i<=1000;i++)
		for(j=0;j<=i;j++)
			c[i][j]=c[i-1][j-1]+c[i-1][j];
	f[0][0]=g[0][0]=1;
	for(i=1;i<=m;i++)
		for(j=0;j<=x;j++)
		{
			for(k=0;k<=sum[i]&&j-k>=0;k++)
			{
				f[i][j]+=f[i-1][j-k]/c[a[i]][need[i]+k];
				g[i][j]+=g[i-1][j-k];
			}
		}
	printf("%.9lf\n",(double)(f[m][x]/g[m][x]));
	return 0;
}
