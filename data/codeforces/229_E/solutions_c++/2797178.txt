#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;

double c[1010][1010],f[1010][1010];
int a[1010][1010],cnt[1010],flag[1010];
int tot,n,m,_n,num,s[1010],v[1000010];

bool cmp(const int i, const int j){return i>j;}

int main()
{
	c[0][0]=1;
	for (int i=1; i<=1000; i++)
	{
		c[i][0]=1;
		for (int j=1; j<=i; j++) c[i][j]=c[i-1][j-1]+c[i-1][j];
	}
	
	scanf("%d%d",&n,&m); _n=n;
	for (int i=1; i<=m; i++)
	{
		scanf("%d",&s[i]);
		for (int j=1; j<=s[i]; j++)
		{
			scanf("%d",&v[++tot]);
			a[i][j]=v[tot];
		}
	}
	
	sort(v+1, v+tot+1, cmp); int value=v[n];
	
	for (int i=1; i<=m; i++)
	{
		for (int j=s[i]; j; j--)
		{
			if (a[i][j]>value) ++cnt[i], n--;
			if (a[i][j]==value) flag[i]=1, num++;
		}
	}
	
	f[0][0]=1;
	for (int i=1; i<=m; i++)
	{
		if (flag[i])
		{
			f[i][cnt[i]]=f[i-1][0]*1.0/c[s[i]][cnt[i]];
			for (int j=cnt[i]+1; j<=_n; j++)
				f[i][j]=f[i-1][j-cnt[i]]/c[s[i]][cnt[i]]+ f[i-1][j-cnt[i]-1]/c[s[i]][cnt[i]+1];
		}		
		else
		{
			for (int j=cnt[i]; j<=_n; j++)
				f[i][j]=f[i-1][j-cnt[i]]/c[s[i]][cnt[i]];
		}
	}
	printf("%.10lf\n",f[m][_n]/c[num][n]);
	//for (;;);
}
