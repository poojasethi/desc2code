#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#define N 1010
using namespace std;
int n,m,k,tot,sum;
int c[N],d[N];
double C[N][N],f[N][N];
vector<int> p[N],v;
int main(){
    C[0][0]=1.;
    for(int i=1;i<N;i++){
        C[i][0]=1.;
        for(int j=1;j<=i;j++) C[i][j]=C[i-1][j-1]+C[i-1][j];
    }
    cin>>n>>m;    
    for(int i=1,tt,t;i<=m;i++){
        cin>>tt;
        while(tt--) cin>>t,p[i].push_back(t),v.push_back(t);
    }
    sort(v.begin(),v.end(),greater<int>() );    
    for(k=1;n-k-1>=0&&v[n-k-1]==v[n-1];k++);    
    for(int i=1;i<=m;i++)
        for(int j=0;j<p[i].size();j++)
            if(p[i][j]>=v[n-1]){
                c[i]++;
                if(p[i][j]==v[n-1]) d[i]=1,tot++;
            }        
    f[0][0]=1.;    
    for(int i=0;i<m;sum+=d[++i])
        for(int j=0;j<=k;j++){
            if(f[i][j]<1e-12) continue;
            double t=1.;
            if(d[i+1]) t=C[tot-sum-1][k-j]/C[tot-sum][k-j];            
            f[i+1][j]+=t*f[i][j]/C[p[i+1].size()][c[i+1]-d[i+1]];
            f[i+1][j+1]+=(1-t)*f[i][j]/C[p[i+1].size()][c[i+1]];
        }
    printf("%.12lf\n",f[m][k]);
}
