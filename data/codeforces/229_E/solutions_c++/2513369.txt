#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>
#include <algorithm>

using namespace std;

const int N=1012;
int n,m,k[N],now[N],tot=0,x;
struct Data {int price,name;} a[N];
inline bool operator < (Data a,Data b)
  {
    return a.price>b.price;
  }
double p=1,f[N][N];
  
int main()
  {
    scanf("%d%d",&n,&m);
    for (int i=1;i<=m;++i)
      {
        scanf("%d",&k[i]);
        for (int j=1;j<=k[i];++j)
          a[++tot].name=i,scanf("%d",&a[tot].price);
      }
    sort(a+1,a+tot+1);
    for (int i=1;i<=n;++i) 
      if (a[i].price!=a[n].price) 
        p=p*(now[a[i].name]+1)/(k[a[i].name]-now[a[i].name]),++now[a[i].name];
    int l=n,r=n;
    while (l>1 && a[l-1].price==a[n].price) --l;
    while (r<tot && a[r+1].price==a[n].price) ++r;
    for (int i=l;i<=r;++i) a[i-l+1]=a[i];tot=r-l+1;n=n-l+1;
    f[0][0]=p;
    for (int i=1;i<=tot;++i)
      {
        f[i][0]=f[i-1][0];
        for (int j=1;j<=n;++j)
          f[i][j]=
            f[i-1][j-1]*(now[a[i].name]+1)/(k[a[i].name]-now[a[i].name])*j/i
            +f[i-1][j]*(i-j)/i;
      }
    printf("%.9lf\n",f[tot][n]);
  }
