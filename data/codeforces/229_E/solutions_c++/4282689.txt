#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;
#define N 1005

double dp[N][N],f[N][N],c[N][N];
int n,m;
vector<int> v[N],w;

void ptm()
{
	c[0][0]=1;
    for(int i=1;i<N;i++)
	{
        c[i][0]=1;
        for(int j=1;j<=i;j++)
			c[i][j]=c[i-1][j]+c[i-1][j-1];
    }    
}

void init()
{
	scanf("%d%d",&n,&m);
    for(int i=0;i<m;i++)
	{
        int x,y;
        scanf("%d",&x);
        for(int j=0;j<x;j++)
		{
            scanf("%d",&y);
			v[i].push_back(y);
            w.push_back(-y);
        }
    }
}

void DP()
{
	dp[0][0]=f[0][0]=1;
    for(int i=0;i<m;i++)
	{
        int x=0,y=0;
        for(int j=0;j<v[i].size();j++)
		{
            if(v[i][j]>-w[n-1])x++;
            else if(v[i][j]==-w[n-1])y++;
		} 
        for(int j=0;j<=n;j++)
		{
            for(int k=0;k<=y&&j+x+k<=n;k++)
			{
                f[i+1][j+x+k]+=f[i][j];
                dp[i+1][j+x+k]+=dp[i][j]/c[v[i].size()][x+k];
            }
        }   
    }
}

int main()
{
   	ptm();
   	init();
    sort(w.begin(),w.end());
    DP();
    printf("%.10lf\n",dp[m][n]/f[m][n]);  
    return 0;
}
