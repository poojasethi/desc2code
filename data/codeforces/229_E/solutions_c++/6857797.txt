#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define db double
#define N 1010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int a,id;
} b[N];
int i,j,k,n,m,t;
int h[N],a[N][N],F[N];
db f[N][N],C[N][N];
inline bool cc1(const ww &a,const ww &b) {
	return a.a>b.a;
}
int main() {
	For(i,0,N-1) {
		C[i][0]=1;
		For(j,1,i) C[i][j]=C[i-1][j-1]+C[i-1][j];
	}
	scanf("%d%d",&m,&n);
	For(i,1,n) {
		scanf("%d",&a[i][0]);
		For(j,1,*a[i]) scanf("%d",&a[i][j]),b[++t]=(ww){a[i][j],i};
	}
	sort(b+1,b+t+1,cc1);
	int A=b[m].a,sum=0;
	For(i,1,t) if (b[i].a>A) h[b[i].id]++,m--;
	For(i,1,t) if (b[i].a==A) F[b[i].id]=1,sum++;
	f[0][0]=1;
	For(i,1,n) {
		memcpy(f[i],f[i-1],sizeof(f[i]));
		For(j,0,m) f[i][j]/=C[*a[i]][h[i]];
		if (F[i]) For(j,1,m) f[i][j]+=f[i-1][j-1]/C[*a[i]][h[i]+1];
	}
	printf("%.11lf\n",f[n][m]/C[sum][m]);
	//for(;;);
	return 0;
}
