#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
#define PB push_back
#define N 1005
vector<int> a[N],v;
double c[N][N],f[N][N],g[N][N];
int n,m,x,y,L;
bool cmp(const int i,const int j){return i>j;}
int main()
{
    for (int i=0;i<=N-5;i++)
        for (int j=0;j<=i;j++)
            c[i][j]=(i&&j)?c[i-1][j]+c[i-1][j-1]:1;
    scanf("%d%d",&n,&m);
    for (int i=0;i<m;i++){
        scanf("%d",&L);
        for (int j=1;j<=L;j++) scanf("%d",&x),a[i].PB(x),v.PB(x);
        }
    sort(v.begin(),v.end(),cmp); f[0][0]=g[0][0]=1;
    for (int i=0;i<m;i++){
        L=a[i].size(); x=y=0;
        for (int j=0;j<L;j++)
            if (a[i][j]>v[n-1]) x++;
            else if (a[i][j]==v[n-1]) y++;
        for (int j=0;j<=n;j++)
            for (int k=0;k<=y;k++) if (j+x+k<=n)
                g[i+1][j+x+k]+=g[i][j],f[i+1][j+x+k]+=f[i][j]/c[L][x+k];
        }
    printf("%.10lf\n",f[m][n]/g[m][n]); return 0;
}
