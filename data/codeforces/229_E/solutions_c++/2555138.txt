#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<vector>
#define pb push_back
using namespace std;
const int maxn=1010;
double f[maxn][maxn],g[maxn][maxn],c[maxn][maxn];
int n,m,s,v;
vector<int> a[maxn],b;
bool cmp(int a,int b){return a>b;}
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
    scanf("%d%d",&n,&m);
    for(int i=0;i<maxn;i++)
        for(int j=0;j<=i;j++)
            c[i][j]=(i&&j)?c[i-1][j-1]+c[i-1][j]:1;
    for(int i=1;i<=m;i++){
        scanf("%d",&s);
        for(int j=0;j<s;j++){
            scanf("%d",&v);
            a[i].pb(v);
            b.pb(v);
        }
    }
    sort(b.begin(),b.end(),cmp);
    f[1][0]=g[1][0]=1;
    for(int i=1;i<=m;i++){
        int x=0,y=0;
        s=a[i].size();
        for(int j=0;j<s;j++){
            if(a[i][j]>b[n-1])x++;
            if(a[i][j]==b[n-1])y++;
        }
        for(int k=0;k<=y;k++)
            for(int j=0;j<=n-x-k;j++){
                g[i+1][j+x+k]+=g[i][j];
                f[i+1][j+x+k]+=f[i][j]/c[s][x+k];
            }
    }
    printf("%.12lf\n",f[m+1][n]/g[m+1][n]);
    return 0;
}
