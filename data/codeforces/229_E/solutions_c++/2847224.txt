#include<cstdio>
#include<vector>
#include<algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
#define For(i,n) for(int i=1;i<=n;i++)
#define FOR(i,a,b) for(int i=a;i<=b;i++)
#define pb push_back
#define lb lower_bound
#define N 1010
using namespace std;
int n,m,k,x,t,p,sp,s,sz,tot,a[N];
double f[N][N],c[N][N],dc[N][N],ans;
vector <int> e[N];
int main()
{
	scanf("%d%d",&n,&m);
	For(i,m)
	{
		scanf("%d",&k);
		For(j,k)
		{
			scanf("%d",&x);
			e[i].pb(x);
			a[++t]=x;
		}
	}
	sort(a+1,a+1+t);
	sp=a[t-n+1];
	FOR(i,0,t) c[i][0]=dc[i][0]=1;
	For(i,t)
	    For(j,n)
	    {
			c[i][j]=c[i-1][j]+c[i-1][j-1];
			dc[i][j]=1/c[i][j];
		}
	t=0;
	f[0][0]=1;
	For(i,m)
	{
		sort(e[i].begin(),e[i].end());
		p=lb(e[i].begin(),e[i].end(),sp)-e[i].begin();
		sz=e[i].size();
		s=sz-p;
		if (s && e[i][p]==sp)
		{
		    t++;
			FOR(j,s,n)
			    f[i][j]=f[i-1][j-s]*dc[sz][s];
			s--;
		}
		tot+=s;
		FOR(j,s,n) f[i][j]+=f[i-1][j-s]*dc[sz][s];
	}
	ans=f[m][n]*dc[t][n-tot];
	printf("%.16lf\n",ans);
	return 0;
}
