#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m,K=0,E,H=0;
int a[1100][1100],b[1000100];
int A[1100];
bool u[1100];
double f[1100],C[1100][1100];
void P()
{
	C[0][0]=1;
	for(int i=1;i<=1000;++i)
	{
		C[i][0]=C[i][i]=1;
		for(int j=1;j<i;++j)
			C[i][j]=C[i-1][j]+C[i-1][j-1];
	}
}
int main()
{
	scanf("%d%d",&n,&m);P();
	for(int i=1;i<=m;++i)
	{
		scanf("%d",&a[i][0]);
		for(int j=1;j<=a[i][0];++j)
			scanf("%d",&a[i][j]),b[++K]=a[i][j];
		sort(a[i]+1,a[i]+1+a[i][0]);
	}
	sort(b+1,b+1+K);
	E=b[K-n+1];
	for(int i=1,j;i<=m;++i)
	{
		for(j=a[i][0];j>0 && a[i][j]>E;--j);
		A[i]=a[i][0]-j;
		u[i]=(j>0 && a[i][j]==E );
		n-=A[i];
		H+=u[i];
	}
	memset(f,0,sizeof(f));
	f[0]=1;
	for(int i=1;i<=m;++i)
		if ( !u[i] ) 
		{
			for(int j=0;j<=n;++j)
				f[j]/=C[a[i][0]][A[i]];
		}else{
			for(int j=n;j>0;--j)
				f[j]=f[j]/C[a[i][0]][A[i]]+f[j-1]/C[a[i][0]][A[i]+1];
			f[0]/=C[a[i][0]][A[i]];
		}
	printf("%.10lf\n",f[n]/C[H][n]);
	return 0;
}
