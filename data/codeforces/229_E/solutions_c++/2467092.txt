#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
double f[1001][1001],g[1001][1001],c[1001][1001];
vector<int> cc[1001];
int n,m,i,j,k,p[1000*1000+1],lp;
int main()
{
//	freopen("1.in","r",stdin);
//	freopen("1.out","w",stdout);
	int n,m;
	scanf("%d%d",&n,&m);
	for (i=1;i<=m;i++)
	{
		int le; scanf("%d",&le); 
		cc[i].resize(le);
		for(j=0;j<le;j++)
		scanf("%d",&cc[i][j]),p[lp++]=cc[i][j];	
	}
	sort(p,p+lp,greater<int>());
	int l=p[n-1];
	int num=count(p,p+n,l);
	for (i=0;i<=m;i++) g[i][0]=1;
	f[0][0]=1;
	c[0][0]=1;
	for(i=1;i<1000;i++)
	{
		c[i][0]=1; 
		for(j=1;j<=i;j++)
		c[i][j]=c[i-1][j]+c[i-1][j-1];
	}
	for(i=1; i<=m;i++)
	{
		int le; le=cc[i].size();
		int x,y; x=y=0;
		for (j=0;j<le;j++)
		{
			if (cc[i][j]>l) x++;
			if (cc[i][j]==l) y++;
		}
		f[i][0]=f[i-1][0]/c[le][x];
		for(j=1;j<=num;j++)
		{
			f[i][j]=0;g[i][j]=0;
			for (k=0;k<=y;k++)
				g[i][j]+=g[i-1][j-k];
			for (k=0;k<=y;k++)
				f[i][j]+=f[i-1][j-k]/c[le][x+k];
		}
	}
	printf("%.10f\n",f[m][num]/g[m][num]);
	//printf("%.2f",g[m][num]);
	return 0;
}
