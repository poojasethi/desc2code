#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++) 
int n,m,k,x,r,t,fg[2000],ans[2000];
double dp[2000][2000],c[2000][2000];
vector<int> q[2000],p;
int main() {
	scanf("%d%d",&m,&n);
	for(int i=1;i<=n;i++) {
		scanf("%d",&k);
		rep(j,k) scanf("%d",&x),q[i].push_back(x),p.push_back(x);
	}
	sort(p.rbegin(),p.rend());
	r=p[m-1];
	for (int i=1;i<=n;i++) {
		sort(q[i].rbegin(),q[i].rend());
		rep(j,q[i].size()) if (q[i][j]==r) fg[i]=1; else if (q[i][j]>r) ans[i]++,t++;
	}
	c[0][0]=1;
	for (int i=1;i<=1000;i++) {
		c[i][0]=c[i][i]=1;
		for (int j=1;j<i;j++) c[i][j]=c[i-1][j]+c[i-1][j-1];
	}
	dp[0][0]=1;k=0;
	for (int i=1;i<=n;i++) if (fg[i]) {
		dp[i][ans[i]]=dp[i-1][0]/c[q[i].size()][ans[i]];
		for (int j=ans[i]+1;j<=m;j++) 
			dp[i][j]=dp[i-1][j-ans[i]]/c[q[i].size()][ans[i]]+dp[i-1][j-ans[i]-1]/c[q[i].size()][ans[i]+1];
		k++;
	} else for (int j=ans[i];j<=m;j++) dp[i][j]=dp[i-1][j-ans[i]]/c[q[i].size()][ans[i]];
	printf("%.10f\n",dp[n][m]/c[k][m-t]);
}
