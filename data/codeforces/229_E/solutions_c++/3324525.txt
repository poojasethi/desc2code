#include<stdio.h>
#include<algorithm>
using namespace std;
int v[1005],d[1005],s[1000005],e[1000005],begin[1005],h[1005];
long double c[1005][1005],g[1005][1005],f[1005][1005];
bool cmp(int a,int b){return a>b;}
int main()
{
    int i,j,k,t,nd,n,m,p=0,l,r,qv,qd;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;i++)
    {
        scanf("%d",&t);h[i]=t;begin[i]=p+1;
        while(t--)
        {
            scanf("%d",&s[++p]);
            e[p]=s[p];
        }
    }
    begin[m+1]=p+1;l=r=n;
    sort(s+1,s+1+p,cmp);
    while(r<p&&s[r+1]==s[r])r++;
    while(l>1&&s[l-1]==s[l])l--;
    s[0]=1000000005;
    qv=s[l-1];qd=s[l];
    for(i=1;i<=m;i++)
    {
        for(j=begin[i];j<begin[i+1];++j)
            if(e[j]>=qv)++v[i];
            else if(e[j]>=qd)++d[i];
    }
    c[0][0]=1;
    for(i=1;i<=1000;i++)
        for(j=0;j<=i;j++)
            c[i][j]=c[i-1][j]+c[i-1][j-1];
    f[0][0]=1;g[0][0]=1;
    nd=n-l+1;
    for(i=1;i<=m;i++)
        for(j=0;j<=nd;j++)
            for(k=0;k<=d[i]&&j-k>=0;++k){f[i][j]+=f[i-1][j-k]/c[h[i]][v[i]+k];g[i][j]+=g[i-1][j-k];}
    printf("%0.10f",(double)(f[m][nd]/g[m][nd]));
    return 0;
}
