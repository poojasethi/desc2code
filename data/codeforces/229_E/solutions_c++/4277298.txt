#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
const int maxn=1000+5;
int n,m,k,maxk,cnt;
vector<int> g[maxn];
int h[maxn],d[maxn],s[maxn];
double C[maxn][maxn],f[maxn][maxn],p[maxn][maxn];
void prepare()
{
	for(int i=0;i<=maxk;i++)
	{
		C[i][0]=C[i][i]=1;
		for(int j=1;j<i;j++)
			C[i][j]=C[i-1][j]+C[i-1][j-1];
	}
}
bool cmp(int a,int b) { return a>b; }
int main()
{
	scanf("%d %d",&n,&m);
	for(int i=1;i<=m;i++)
	{
		scanf("%d",&k),maxk=max(maxk,k);
		for(int j=1;j<=k;j++)
		{
			scanf("%d",&h[++cnt]);
			g[i].push_back(h[cnt]);
		}
	}
	prepare();
	sort(h+1,h+1+cnt,cmp);
	int l=n,r=n;
	while(l>1 && h[l-1]==h[n]) l--;
	while(r<n && h[r+1]==h[n]) r++;
	for(int i=1;i<=m;i++)
		for(int j=0;j<g[i].size();j++)	
			d[i]+=g[i][j]>h[l],s[i]+=g[i][j]==h[l];
	f[0][0]=p[0][0]=1;
	for(int i=1;i<=m;i++)
		for(int j=0;j<=n-l+1;j++)
			for(int k=0;k<=s[i]&&j>=k;k++)
			{
				f[i][j]+=f[i-1][j-k]/C[g[i].size()][d[i]+k];
				p[i][j]+=p[i-1][j-k];
			}
	printf("%.10lf\n",f[m][n-l+1]/p[m][n-l+1]);
	return 0;
}
