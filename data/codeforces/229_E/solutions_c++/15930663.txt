#include<bits/stdc++.h>
using namespace std;

struct unit
{int l;
 int c[1010];
};
struct unit b[1010];

int m,n,n1,tot,inf=1e9;
int a[1010],w[1010],sa[1010];
double ans;
double f[1010][1010],c[1010][1010];

bool cmp(int i,int j)
{return i>j;
}

int main()
{int i,j,k;
 double s,s1;
 
 scanf("%d%d",&m,&n);
 for(i=1;i<=n;i++)
 {	scanf("%d",&b[i].l);
 	for(j=1;j<=b[i].l;j++)
 	{	scanf("%d",&b[i].c[j]);
 		sa[++tot]=b[i].c[j];
 	}
 	sort(b[i].c+1,b[i].c+b[i].l+1,cmp);
 }
 for(i=0;i<=tot;i++)
 {	c[i][0]=1;
 	for(j=1;j<=i;j++)
 		c[i][j]=c[i-1][j]+c[i-1][j-1];
 }
 sort(sa+1,sa+tot+1,cmp);
 sa[0]=inf;
 k=m;
 while(sa[k]==sa[k+1]) k--;
 for(i=1;i<=n;i++)
 {	j=0;
 	while(j+1<=b[i].l && b[i].c[j+1]>=sa[k]) j++;
 	a[i]=j;
 	if(j+1<=b[i].l && b[i].c[j+1]==sa[k+1])
	{	w[i]=1;
		n1++;
	}
 }
 f[0][0]=1;
 for(i=1;i<=n;i++)
 {	s=1/c[b[i].l][a[i]];
 	if(w[i]) s1=1/c[b[i].l][a[i]+1];
	for(j=0;j<=m-k;j++)
	{	f[i][j]=f[i-1][j]*s;
		if(w[i]) f[i][j]+=f[i-1][j-1]*s1;
	}
 }
 ans=f[n][m-k]/c[n1][m-k];
 printf("%.10lf\n",ans);
 return 0;
}
