#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
#define PB push_back
const int u=1005;
vector<int> a[u],b;
double c[u][u],f[u][u],g[u][u];
int n,m,i,j,k,x,y,z;
bool cmp(int i,int j){return i>j;}
int main()
{
	cin>>n>>m;
	for(i=0;i<m;i++)
	{
		scanf("%d",&k);
		for(j=1;j<=k;j++) scanf("%d",&x),a[i].PB(x),b.PB(x);
	}
	for(i=0;i<=u-5;i++)
		for(j=0;j<=i;j++)
			c[i][j]=(i&&j)?c[i-1][j]+c[i-1][j-1]:1;
	sort(b.begin(),b.end(),cmp);
	f[0][0]=g[0][0]=1;
	for(i=0;i<m;i++)
	{
		z=a[i].size();
		for(j=x=y=0;j<z;j++)
			x+=a[i][j]>b[n-1],y+=a[i][j]==b[n-1];
		for(j=0;j<=n;j++)
			for(k=0;k<=y;k++)
				if(j+x+k<=n) g[i+1][j+x+k]+=g[i][j],f[i+1][j+x+k]+=f[i][j]/c[z][x+k];
	}
	printf("%.12f\n",f[m][n]/g[m][n]);
	return 0;
}
