#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int z=1005;
int l[z],res[z],p[z][z];
double ans[z][z];
bool cmp(int a,int b){
    return a>b;
}
double c(double ans,int a,int b){
    for (int i=1;i<=b;i++) ans=ans/(a-b+i)*i;
    return ans;
}
int main(){
    int n,m,tot,i,j,t,cnt,ct;
//  freopen("229e.in","r",stdin);
//  freopen("229e.out","w",stdout);
    scanf("%d%d",&n,&m);
    tot=0;
    for (i=0;i<m;i++){
        scanf("%d",l+i);
        for (j=0;j<l[i];j++) res[tot++]=(scanf("%d",&p[i][j]),p[i][j]);
        sort(&p[i][0],&p[i][l[i]],cmp);
    }
    sort(res,res+tot,cmp);
    for (i=n-1;i>=0 && res[i]==res[n-1];i--);
    cnt=n-1-i;
    memset(ans,0,sizeof ans);
    ans[0][0]=1,ct=0;
    for (i=0;i<m;i++){
        for (t=0;t<l[i] && p[i][t]>=res[n-1];t++);
        if (t && p[i][t-1]==res[n-1]){
            ans[i+1][0]=c(ans[i][0],l[i],t-1);
            for (j=1;j<=cnt;j++)
                ans[i+1][j]=c(ans[i][j],l[i],t-1)+c(ans[i][j-1],l[i],t);
            ct++;
        } else{
            for (j=0;j<=cnt;j++) ans[i+1][j]=c(ans[i][j],l[i],t);
        }
    }
    printf("%.12lf\n",c(ans[m][cnt],ct,cnt));
    return 0;
}
