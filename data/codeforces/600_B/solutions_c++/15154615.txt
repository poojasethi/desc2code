#include <bits/stdc++.h>
using namespace std;

int main() {
    int n,m,a[200000],b[200000],i,j;

    scanf("%d %d",&n,&m);
    for (i=0; i<n; i++)
        cin >> a[i];
    sort(a,a+n);
    for (i=0; i<m; i++) {
        cin >> b[i];
        cout << upper_bound(a,a+n,b[i])-a << endl;
    }
    return 0;
}
