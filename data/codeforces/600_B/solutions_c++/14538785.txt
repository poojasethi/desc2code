#include<bits/stdc++.h>
using namespace std;
int n,m,ar[1000000],k;
int main(){
	scanf("%d %d",&n,&m);
	for(int i=1 ; i<=n ; i++){
		scanf("%d",&ar[i]);
	}
	sort(ar+1,ar+1+n);
	for(int j=1 ; j<=m ; j++){
		scanf("%d",&k);
		int it=lower_bound(ar+1,ar+1+n,k+1)-ar-1;
		printf("%d ",it);
	}
	return 0;
}
