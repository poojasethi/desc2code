#include<bits/stdc++.h>
using namespace std;
long A[200005]={0};
int main()
{
	int n,m;
	long b;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%ld",&A[i]);
	sort(A,A+n);
	for(int i=0;i<m;i++)
		{
			scanf("%ld",&b);
			 int up=upper_bound(A,A+n,b)-A;
			
			printf("%d ",up);
		}
	printf("\n");
	return 0;
}
