#include <algorithm>
#include <iostream>

using namespace std;

int main()
{
	int n, m, a[200000], b[200000];
	cin >> n >> m;
	for (int i=0; i<n; i++)
		cin >> a[i];
	for (int j=0; j<m; j++)
		cin >> b[j];
	//int w = 7;
	sort(a, a+n);
	for (int j=0; j<m; j++)
	{
		cout << upper_bound(a, a+n, b[j]) - a << " ";
	}
	return 0;
}
