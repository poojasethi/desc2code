#include<cstdio>
#include<algorithm>
using namespace std;

int n, m, v;
int a[222222];

int main() {
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++) scanf("%d", &a[i]);
	sort(a,a+n);
	for(int i=0;i<m;i++) {
		scanf("%d", &v);
		printf("%d%c", upper_bound(a, a+n, v) - a, i<m-1 ? ' ' : '\n');
	}
	
	return 0;
}
