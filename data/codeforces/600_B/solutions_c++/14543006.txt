#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long n,m,a[200005],b;
    cin>>n>>m;
    for(int i=0 ; i < n ; i++)cin>>a[i];
    sort(a,a+n);
    for(int i=0 ; i < m ; i++){
        cin>>b;
        cout<<upper_bound(a,a+n,b)-a;
        if(i!=m-1)cout<<" ";
    }cout<<endl;
    return 0;
}
