def GI(): return int(raw_input())
def GIS(): return map(int, raw_input().split())

def main():
    GIS()
    a = GIS()
    b = GIS()

    a.sort()
    c = sorted(b)

    d = {}
    i = 0
    for x in c:
        while i < len(a) and a[i] <= x:
            i += 1
        d[x] = i
    return ' '.join(str(d[x]) for x in b)

print main()
