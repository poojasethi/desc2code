import sys

[n, m] = map(lambda x: int(x), sys.stdin.next().split(' '))
a = map(lambda x: (int(x), 0, 0), sys.stdin.next().split(' '))
b = map(lambda x: (int(x[1]), 1, x[0]), enumerate(sys.stdin.next().split(' ')))

c = sorted(a + b)
solution = [0] * m

counter = 0
for i in range(n + m):
    if c[i][1] == 0:
        counter += 1
    else:
        solution[c[i][2]] = counter


print(' '.join(map(str, solution)))
