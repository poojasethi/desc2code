import bisect

def get_list():
    line = raw_input()
    arr = [int(item) for item in line.strip().split(' ')]
    return arr

def read_input():
    line = raw_input()
    return get_list(), get_list()

def main():
    a, b = read_input()
    a.sort()
    rez = []
    for x in b:
        val = bisect.bisect_right(a, x)
        rez.append(str(val))
    print ' '.join(rez)

if __name__ == '__main__':
    # print bisect.bisect_right([1, 4, 10, 15], 15)
    main()
