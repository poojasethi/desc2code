raw_input()
first = map(int, raw_input().split())
second = map(int, raw_input().split())
first.sort()
for i, j in enumerate(second):
    second[i] = (j, i)
second.sort()
ans = [0]*len(second)
a = 0
b = 0
while a < len(first) and b < len(second):
    if second[b][0] < first[a]:
        ans[second[b][1]] = a
        b += 1
    else:
        a += 1
if a == len(first):
    while b < len(second):
        ans[second[b][1]] = len(first)
        b += 1
print " ".join(map(str, ans))
