
n,m = map(int, raw_input().split())

ln = map(int, raw_input().split())
lm = map(int, raw_input().split())


def binarySearch(ln, element):
    start = 0
    end = n-1

    ans = 0

    while start<= end:
        mid = start + (end- start)/2

        if ln[mid]<=element:
            start = mid+1
            ans = start
        else:
            end = mid-1
    return ans




ln.sort()

for element in lm:
    print binarySearch(ln, element),
