def getarr():
    return map(int, raw_input().split())

def func(a, item):
    l = 0
    r = len(a)
    while(l + 1 < r):
        mid = l + (r - l) / 2
        if a[mid] <= item:
            l = mid
        else:
            r = mid
    if a[l] <= item:
        return l + 1
    else:
        return l
