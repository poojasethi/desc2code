n,m  = [int(i) for i in raw_input().split(' ')]
a = sorted([int(c) for c in raw_input().split(' ')])
b = sorted([(int(c),i) for i,c in enumerate(raw_input().split(' '))])

o = [0]*m

j = 0
for i in range(m):
     while j < n and b[i][0] >= a[j]:
             j+=1
     o[b[i][1]] = str(j)
print " ".join(o)
