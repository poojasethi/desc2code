#include <algorithm>
#include <iostream>
using namespace std;
int N, S, A[1010], PC, PJ, odd, val;
int main() {
    cin >> N;
    for (int i=0; i<N; ++i) {
	cin >> S;
	for (int i=0; i<S/2; ++i) {
	    cin >> val;
	    PC += val;
	}
	if (S % 2) {
	    cin >> val;
	    A[odd++] = val;
	}
	for (int i=0; i<S/2; ++i) {
	    cin >> val;
	    PJ += val;
	}
    }
    sort(A, A+odd);
    for (int i=odd-1; i>=0; i-=2) {
	PC += A[i];
	if (i-1 >= 0) PJ += A[i-1];
    }
    cout << PC << ' ' << PJ << endl;
}
