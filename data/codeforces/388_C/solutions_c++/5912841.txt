#include<cstdio>
#include<functional>
#include<vector>
#include<algorithm>
#define N 100
using namespace std;

int ans[2];

int main()
{
	int n; scanf("%d",&n);
	
	vector<int> m;
	for(int i=0;i<n;++i)
	{
		int q; scanf("%d",&q);
		for(int j=0;j<q;++j)
		{
			int v; scanf("%d",&v);
			if(q%2 && j==q/2)
				m.push_back(v);
			else
				ans[j>=q/2] += v;
		}		
	}
	sort(m.begin(), m.end(), greater<int>());
	for(int i=0;i<m.size();++i)
		ans[i%2] += m[i];
	printf("%d %d\n", ans[0], ans[1]);
	return 0;
}
