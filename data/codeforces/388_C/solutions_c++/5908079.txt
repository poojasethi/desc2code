#include<iostream>
#include<algorithm>
using namespace std;

int n,s[10001];
int ans1,ans2;

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		int m;
		cin>>m;
		for(int j=1;j<=m;j++)
		{
			int x;
			cin>>x;
			if(m&1){
				if(j<(m+1)/2) ans1+=x;
				else if(j>(m+1)/2) ans2+=x;
				else s[i]=x;
			}
			else{
				if(j<=m/2) ans1+=x;
				else ans2+=x;
				s[i]=0;
			}
		}
	}
	sort(s+1,s+n+1);
	for(int i=n;i>=1;i-=2){
		ans1+=s[i];ans2+=s[i-1];
	}
	cout<<ans1<<" "<<ans2<<endl;
	return 0;
}
