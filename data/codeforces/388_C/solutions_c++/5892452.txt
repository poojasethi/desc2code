#include <stdio.h>
#include <algorithm>
using namespace std;
int a[110], c[110];
int main(){
    int n, m, i, an = 0, p = 0, q = 0;
    scanf("%d", &n);
    while (n--){
        scanf("%d", &m);
        for (i = 0; i < m; i++) scanf("%d", &c[i]);
        for (i = 0; i < m / 2; i++){ p += c[i]; q += c[m - 1 - i]; }
        if (m % 2) a[an++] = -c[m / 2];
    }
    sort(a, a + an);
    for (i = 0; i < an; i++){
        if (i % 2) q -= a[i];
        else p -= a[i];
    }
    printf("%d %d\n", p, q);
    return 0;
}
