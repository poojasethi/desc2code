//In the name of God
#include <algorithm>
#include <iostream>
using namespace std;

int l, r, n, cnt, a[200], b[200];

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n;
	for (int i = 0; i < n; i++) {
		int m; cin >> m;
		for (int i = 0; i < m; i++) {
			cin >> a[i];
			if (2 * i + 1 < m)
				l += a[i];
			else if (2 * i + 1 > m)
				r += a[i];
			else
				b[cnt++] = a[i];
		}
	}
	sort(b, b + cnt, greater<int>());
	for (int i = 0; i < cnt; i++)
		(i % 2? r: l) += b[i];
	cout << l << ' ' << r << '\n';
	return 0;
}
