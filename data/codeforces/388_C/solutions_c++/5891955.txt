#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
int n,s[105];
int a,b;
int c[105],m;
int main(){
    //freopen("in","r",stdin);
    int x;
    cin>>n;
    for (int i=0;i<n;i++){
        cin>>x;
        for (int j=0;j<x;j++)
            cin>>s[j];
        for (int j=0;j<x/2;j++)
            a+=s[j],b+=s[x-1-j];
        if (x&1) c[m++]=s[x/2];
    }
    sort(c,c+m); reverse(c,c+m);
    for (int i=0;i<m;i++)
        if (i&1) b+=c[i]; else a+=c[i];
    cout<<a<<" "<<b;
    return 0;
}
