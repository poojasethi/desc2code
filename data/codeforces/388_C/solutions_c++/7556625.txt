#include<cstdio>
#include<algorithm>
int n,Ciel(0),Jiro(0),rem[105];
int main(){
	scanf("%d",&n);
	for (int s,i=0;i<n;i++){
		scanf("%d",&s);
		for (int c,j=0;j<s;j++){
			scanf("%d",&c);
			if (s%2&&j==s/2) rem[++rem[0]]=c;
			else if (j<s/2) Ciel+=c;
			else Jiro+=c;
		}
	}
	if (rem[0]){
		std::sort(rem+1,rem+rem[0]+1);
		for (int i=rem[0];i;i--)
			if ((rem[0]-i)%2) Jiro+=rem[i];
			else Ciel+=rem[i];
	}
	printf("%d %d\n",Ciel,Jiro);
}
