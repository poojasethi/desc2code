#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int nn=105;
int n,m,s,u,v,a[nn],b[nn];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;++i){
		scanf("%d",&s);
		for(int i=1;i<=s;++i)scanf("%d",&a[i]);
		if(s&1)b[++m]=a[s/2+1];
		for(int i=1;2*i<=s;++i)u+=a[i],v+=a[s-i+1];
	}
	sort(b+1,b+m+1);
	for(int i=m;i;--i)
		if((i-m)&1)v+=b[i];else u+=b[i];
	printf("%d %d\n",u,v);
	return 0;
}
