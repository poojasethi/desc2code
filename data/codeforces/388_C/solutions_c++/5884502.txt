#include<cstdio>
#include<algorithm>
using std::sort;

int v[105], sa, sb, mn;

int main(void){
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++){
		int l, x;
		scanf("%d", &l);
		for(int i=0;i<l/2;i++){
			scanf("%d", &x);
			sa += x;
		}

		if(l%2)
			scanf("%d", v + (mn++));

		for(int i=0;i<l/2;i++){
			scanf("%d", &x);
			sb += x;
		}
	}

	sort(v, v+mn);
	bool ciel=1;
	for(int i=mn-1;i>=0;i--, ciel=!ciel)
		if(ciel)
			sa += v[i];
		else
			sb += v[i];

	printf("%d %d", sa, sb);
	return 0;
}
