#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n;
int l, r;

int s[110], t;

int main()
{
    scanf("%d", &n);
    for (int i = 1, j, z; i <= n; ++i)
    {
	scanf("%d", &j);
	for (int k = 1; k <= (j >> 1); ++k)
	    scanf("%d", &z), l += z;
	if (j & 1) scanf("%d", &z), s[++t] = z;
	for (int k = 1; k <= (j >> 1); ++k)
	    scanf("%d", &z), r += z;
    }
    sort(s + 1, s + t + 1);
    for (int i = 0; i < t; ++i)
	if (i & 1) r += s[t - i]; else l += s[t - i];
    printf("%d %d\n", l, r);
    return 0;
}
