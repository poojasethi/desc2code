#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
vector<int> v;
int n,k,a,b,c;
int main( )
{
 scanf("%d",&n);
 for(int i=1;i<=n;i++){
  scanf("%d",&k);
  for(int j=1;j<=k;j++){ 
  scanf("%d",&c);
  if(k%2==1 && j==(k+1)/2) v.push_back(c);
  else
  if(j<=k/2) a+=c;
  else b+=c;
	
}
}
sort(v.begin(),v.end());
reverse(v.begin(),v.end());
for(int i=0;i<v.size();i++) if(i%2==0) a+=v[i]; else b+=v[i];
printf("%d %d\n",a,b);
return 0;
}
