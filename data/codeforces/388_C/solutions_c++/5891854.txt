#include<iostream>
#include<vector>
#include<cmath>
#include<string>
#include <algorithm>

using namespace std;
int w[101];

int main() {
	int n,sum = 0,total = 0,b=0;

	cin >> n;
	for(int i = 0,k,a; i < n; i++) {
		cin >> k;
		for(int j = 1; j <= k; j++) {
			cin >> a;
			total += a;
			if(2 * j - k <= 0)
				sum += a;
			else if(2 * j - k == 1)
				w[b++] = -a;
		}
	}
	
	sort(w,w+b);
	for(int i = 0; i < b; i+=2)
		sum -= w[i];
	
	cout << sum << " " << total - sum;
	
	return(0);
}
