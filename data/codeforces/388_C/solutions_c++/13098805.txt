#include<bits/stdc++.h>
using namespace std;
vector<int> a[105];
vector<int> v;
int main()
{
int n,m,i,j;
scanf("%d",&n);

for(i=0;i<n;i++)
{
scanf("%d",&m);
int x;
for(j=0;j<m;j++)
{
scanf("%d",&x);
a[i].push_back(x);
}
}
int fir ,sec;
fir = sec = 0;
for(i=0;i<n;i++)
{
m = a[i].size();
if(m%2)v.push_back(a[i][m/2]);
for(j=0;j<m/2;j++)
{
fir+=a[i][j];
sec+=a[i][m-1-j];
}
}
sort(v.begin(),v.end());
reverse(v.begin(),v.end());
for(i=0;i<v.size();i++)
if(i%2)sec+=v[i];
else fir+=v[i];
cout<<fir<<" "<<sec;

return 0;
}
