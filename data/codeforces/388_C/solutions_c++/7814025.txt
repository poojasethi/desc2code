#include <bits/stdc++.h>
#define FOR(i,a,b)  for(int i=a; i<=b; i++)
using namespace std;

int a[100],b[100], n, s, dem, kq, sum;
int main()
{
    cin>>n;
    while(n--)
    {
        cin>>s;
        FOR(i,1,s) cin>>a[i];
        if(s%2) b[++dem] = a[s/2+1];
        FOR(i,1,s/2) kq += a[i];
        FOR(i,1,s) sum += a[i];
    }
    sort(b+1,b+dem+1);
    for(int i= dem; i > 0; i-=2)    kq += b[i];
    cout<<kq<<" "<<sum-kq;
}
