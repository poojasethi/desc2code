#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
vector<int> v;
int main()
{
	int n;
	scanf("%d",&n);
	int sum1=0,sum=0;
	int x,num;
	while(n--)
	{
		scanf("%d",&x);
		for(int i=1;i<=x;i++)
		{
			scanf("%d",&num);
			sum+=num;
			if(i<=(x>>1))
				sum1+=num;
			if(x&1 && i==(x>>1)+1)
				v.push_back(num);
		}
	}
	sort(v.begin(),v.end());
	int i=v.size()-1;
	while(i>=0)
	{
		sum1+=v[i];
		i-=2;
	}
	printf("%d %d\n",sum1,sum-sum1);
	return 0;
}
