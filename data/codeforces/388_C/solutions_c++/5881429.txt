#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
  int n;
  cin>>n;
  int a=0,b=0;
  vector<int> v;
  for (int i=0;i<n;i++) {
    int m;
    cin>>m;
    for (int j=0;j<m;j++) {
      int x;
      cin>>x;
      if (m%2==0) {
	if (j<m/2) a+=x;
	else b+=x;
      } else {
	if (j<m/2) a+=x;
	else if (j>m/2) b+=x;
	else v.push_back(x);
      }
    }
  }
  sort(v.begin(),v.end());
  for (int i=0;i<int(v.size());i++)
    if (i%2==0) a+=v[int(v.size())-1-i];
    else b+=v[int(v.size())-1-i];
  cout<<a<<" "<<b<<endl;
}
