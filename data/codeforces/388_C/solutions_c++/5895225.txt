#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;
int n,l,r,a[105],x,y,s=0,f[105];
bool cmp(int x,int y){
	if (x<y) return false; else return true;
}
int main( ){
	cin>>n;
	for (int i=1;i<=n;i++){
		cin>>r;
		for (int j=1;j<=r;j++)
			cin>>a[j];
		l=0;
		r++;
		while (true){
			l++;
			r--;
			if (l>=r) break ;
			x+=a[l];
			y+=a[r];
		}
		if (l==r) s++,f[s]=a[l];
	}
	sort(f+1,f+s+1,cmp);
	for (int i=1;i<=s;i++)
		if (i%2>0) x+=f[i]; else y+=f[i];
	cout<<x<<' '<<y<<endl;
	return 0;
}
