#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int n, s, a, b;
int c[111];
vector<int>rem;

int main() {
	scanf("%d", &n);
	rem.clear();
	for(int i=0;i<n;i++) {
		scanf("%d", &s);
		for(int j=0;j<s;j++) scanf("%d", &c[j]);
		int l=0, r=s-1;
		for(; l<r; l++,r--)
			a+=c[l], b+=c[r];
		if(l==r) rem.push_back(c[l]);
	}
	sort(rem.begin(), rem.end());
	for(int i=rem.size()-1, f = 0;i>=0;i--) {
		if (!f) a+=rem[i];
		else b+=rem[i];
		f = 1-f;
	}
	printf("%d %d\n", a, b);
	
	return 0;
}
