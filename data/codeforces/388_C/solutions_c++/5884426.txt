#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<int> mids;
int sum[2];
int n, s, c;
int main() {
	cin >> n;
	for(int i=0;i<n;i++) {
		cin >> s;
		for(int j=0;j<s/2;j++) {
			cin >> c;
			sum[0] += c;
		}
		if(s%2 == 1) {
			cin >> c;
			mids.push_back(c);
		}
		
		for(int j=0;j<s/2;j++) {
			cin >> c;
			sum[1] += c;
		}
	}
	
	sort(mids.begin(), mids.end());
	reverse(mids.begin(), mids.end());
	
	for(int i=0;i<mids.size();i++)
		sum[i%2] += mids[i];
		
	cout << sum[0] << ' ' << sum[1] << '\n';
	return 0;
}
