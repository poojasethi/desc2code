#include <cstdio>
#include <algorithm>
using namespace std;

int N,A,B,a[1005],q[105];

int main()
{
	scanf("%d",&N);
	for (int i=1,s; i<=N; i++)
	{
		scanf("%d",&s);
		for (int j=1; j<=s; j++)
		{
			scanf("%d",&a[j]);
			if (j<=s/2) A+=a[j]; else if (s-j+1<=s/2) B+=a[j];
		}
		if (s&1) q[++q[0]]=-a[s/2+1];
	}
	sort(q+1,q+q[0]+1);
	for (int i=1; i<=q[0]; i++) if (i&1) A-=q[i]; else B-=q[i];
	printf("%d %d\n",A,B);
}
