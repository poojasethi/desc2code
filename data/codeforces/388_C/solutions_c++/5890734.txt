#include <cstdio>
#include <algorithm>

#define N 100

int a[N], n, m, i, j, k, t, c, s1, s2;

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i ++)
	{
		scanf("%d", &k);
		c = (k >> 1);
		for (j = 0; j < c; j ++)
			scanf("%d", &t), s1 += t;
		if (k & 1)
			scanf("%d", &t), a[m ++] = t;
		for (j = 0; j < c; j ++)
			scanf("%d", &t), s2 += t;
	}

	std::sort(a, a + m);

	for (i = 0; i < m; i ++)
		if (i % 2 == (m - 1) % 2)
			s1 += a[i];
		else
			s2 += a[i];

	printf("%d %d\n", s1, s2);

	return 0;
}
