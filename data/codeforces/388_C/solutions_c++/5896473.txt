#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int A,B,n,m,s,a,q[111];

int main()
{
    cin>>n;
    while(n--)
    {
        cin>>s;
        for(int i=1;i<=s;++i)
        {
            cin>>a;
            if(2*i<=s) A+=a;
            else if(2*i==s+1) q[m++]=a;
            else B+=a;
        }
    }
    sort(q,q+m,greater<int>());
    for(int i=0;i<m;++i)
    {
        if(i&1) B+=q[i];
        else A+=q[i];
    }
    cout<<A<<" "<<B<<endl;
    return 0;
}
