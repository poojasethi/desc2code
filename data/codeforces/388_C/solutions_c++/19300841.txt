#include <bits/stdc++.h>
using namespace std;
int n;
priority_queue<int> q;

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> n;
	int t, num, ans=0, res=0;
	while (n--){
		cin >> t;
		for (int i=0; i<t; ++i){
			cin >> num;
			if (i < t/2)
				ans += num;
			else if (i==t/2 && (t&1))
				q.push(num);
			else
				res += num;
		}
	}
	t = 1;
	while (!q.empty()){
		if (t)
			ans += q.top();
		else
			res += q.top();
		q.pop();
		t ^= 1;
	}
	cout << ans << " " << res << endl;
	return 0;
}
