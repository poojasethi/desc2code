#include <cstdio>
#include <algorithm>
using namespace std;
const int S=105,N=S;
int n,m,x,y,s,c,q[N];
int main(){
	scanf("%d",&n);
	while(n--){
		scanf("%d",&s);
		for(int i=1;i<=s;i++){
			scanf("%d",&c);
			if(i<=s/2) x+=c; else
			if(s-i+1<=s/2) y+=c; else
			q[m++]=c;
		}
	}
	sort(q,q+m);
	for(int i=m-1;i>=0;i--) if((m-i)&1) x+=q[i]; else y+=q[i];
	printf("%d %d\n",x,y);
}
