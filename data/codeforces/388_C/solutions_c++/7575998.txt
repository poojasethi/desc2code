#include<cstdio>
#include<algorithm>
#define nn 120
using namespace std;
bool cmp(int a,int b){return a>b;}
int n,j,k,i,m,h,g,b[nn],a[nn];
int main()
{
 scanf("%d",&n);
 for (j=1;j<=n;j++)
     {scanf("%d",&k);
      for (i=1;i<=k;i++)scanf("%d",&a[i]);
      for (i=1;i<=k/2;i++)h+=a[i];
      for (i=(k+1)/2+1;i<=k;i++)g+=a[i];
      if (k&1)b[++m]=a[k/2+1];
     }
 sort(b+1,b+m+1,cmp);
 for (i=1;i<=m;i++)
  if (i&1)h+=b[i];
  else g+=b[i];
 printf("%d %d\n",h,g);
 return 0;
}
