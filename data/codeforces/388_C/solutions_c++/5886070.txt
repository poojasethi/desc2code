#include <cstdio>
#include <algorithm>
using namespace std;

int n, s, i, len, ans1, ans2, a[1100], x[1100];

int main() {
	scanf("%d", &n);
	while (n--) {
		scanf("%d", &s);
		for (i = 1; i <= s; i++)	scanf("%d", &a[i]);
		for (i = 1; i <= s / 2; i++)	ans1 += a[i];
		for (i = (s + 1) / 2 + 1; i <= s; i++)	ans2 += a[i];
		if (s % 2 == 1)	x[++len] = a[s / 2 + 1];
	}
	sort(x + 1, x + len + 1);
	for (i = len; i; i--)
		if ((len - i) % 2 == 0)	ans1 += x[i];
		else	ans2 += x[i];
	printf("%d %d\n", ans1, ans2);
}
