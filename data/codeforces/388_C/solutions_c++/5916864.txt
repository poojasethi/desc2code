#include<stdio.h>
#include<algorithm>

int pn=0;
int p[110];
int S[2] = {0,0};

int main() {
  int n,s,c;
  scanf("%d", &n);
  for (int i=0; i<n; i++) {
    scanf("%d", &s);
    for (int j=0; j<s; j++) {
      scanf("%d", &c);
      if (j<s/2) { S[0] += c; }
      else if (j>s/2) { S[1] += c; }
      else if (s&1) { p[pn++] = c; }
      else { S[1] += c; }
    }
  }
  std::sort(p, p+pn);
  int a=0;
  for (int i=pn-1; i>=0; i--) {
    S[a] += p[i];
    a = (a+1)&1;
  }
  printf("%d %d", S[0], S[1]);
  return 0;
}
