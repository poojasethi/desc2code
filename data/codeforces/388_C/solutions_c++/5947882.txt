#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
int T,n,m,a[110],s1,s2,b[110];
int main(){
    scanf("%d",&T);
    while(T--){
      scanf("%d",&n);
      for(int i=1;i<=n;i++)scanf("%d",&a[i]);
      for(int i=1;i<=n/2;i++)s1+=a[i],s2+=a[n-i+1];
      if(n%2)b[++m]=a[n/2+1];
    }
    sort(b+1,b+m+1);
    reverse(b+1,b+m+1);
    for(int i=1;i<=m;i++)
      if(i%2)s1+=b[i];
      else s2+=b[i];
    cout<<s1<<" "<<s2<<endl;
    return 0;
}
