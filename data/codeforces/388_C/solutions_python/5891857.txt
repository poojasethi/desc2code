n,a,A,B = input(),[],0,0
for i in range(n):
        c = map(int,raw_input().split())
        A += sum(c[1:c[0]/2+1])
        B += sum(c[(c[0]+1>>1)+1:])
        if (c[0]&1): a.append(c[c[0]+1>>1])
a.sort(); a.reverse()
for i in range(len(a)):
        if i&1: B += a[i]
        else: A += a[i]
print A,B
