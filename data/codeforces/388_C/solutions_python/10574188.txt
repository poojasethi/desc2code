ans1 = 0
ans2 = 0
n = int(raw_input())
arr = []

for i in range(n):
	pile = list(map(int,raw_input().split()))
	j = 1
	while(j <= pile[0]//2):
		ans1 += pile[j]
		j+=1
	if (pile[0] %2 == 1):
		arr.append(pile[j])
		j+= 1
	while(j<=pile[0]):
		ans2+= pile[j]
		j+= 1

arr.sort()
arr.reverse()
l = len(arr)
for i in range(0,l):
	if ( i%2 == 0 ):
		ans1+= arr[i]
	else:
		ans2+= arr[i]
		
print("%r %r"%(ans1, ans2))
