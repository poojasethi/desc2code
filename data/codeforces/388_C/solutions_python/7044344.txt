n = int(raw_input())
rem = []
s1, s2 = 0, 0
for i in range(n):
	l = map(int , raw_input().split())
	cnt = l[0]
	l = l[1:]
	if cnt % 2:
		s1 += sum(l[:cnt/2]); s2 += sum(l[(cnt + 1)/2:])
		rem += [l[cnt/2]]
	else:
		s1 += sum(l[:cnt/2]); s2 += sum(l[cnt/2:])
rem.sort(); rem.reverse()
s1 += sum(rem[::2]); s2 += sum(rem[1::2])
print s1, s2
