def main():
    n = int(raw_input())
    a = [0, 0]
    c = []
    for i in xrange(n):
        l = map(int, raw_input().split())
        m, l = l[0], l[1:]
        a[0] += sum(l[:m/2])
        a[1] += sum(l) - sum(l[:m-m/2])
        if m % 2:
            c.append(l[m/2])
    c.sort(reverse=True)
    for i, x in enumerate(c):
        a[i%2] += x
    print a[0], a[1]
main()
