n = input()
cie = jir = 0
mid = []
for i in xrange(n):
	card = map(int, raw_input().split())
	s = card[0]
	card = card[1:]
	b = s / 2
	e = (s + 1) / 2
	if(b != e):
		mid.append(card[b])
	cie += sum(card[:b])
	jir += sum(card[e:])

l = (len(mid) + 1) / 2
mid = sorted(mid, reverse = True)
for i in xrange(len(mid)):
	if(i % 2 == 0):
		cie += mid[i]
	else:
		jir += mid[i]
print cie,jir
