#In the name of ALLAH
m,n,i,j,a,b = map(int, raw_input().split())
ans = []


for x in [1,m]:
	for y in [1,n]:
		dx = abs(x-i)
		dy = abs(y-j)
		if dx+dy ==0:
			ans.append(0)
		
		if dx%a==0 and dy%b ==0 and abs((dx/a)-(dy/b))%2 == 0:
			ans.append(max(dx/a, dy/b))

if len(ans) == 0:
	print 'Poor Inna and pony!'
elif ans[0] == 0:
	print 0
elif (i-1 < a and m-i < a) or (j-1 < b and n-j < b):
	print 'Poor Inna and pony!'
else:
	print int(min(ans))
