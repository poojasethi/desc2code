#!/usr/bin/python
# encoding: utf-8

n, m, i, j, a, b = map(int, raw_input().split())
if i == 1 and j == 1 or i == 1 and j == m or i == n and j == 1 or i == n and j == m:
    print 0
else:
    if i-a>=1 and j-b>=1 or i-a>=1 and j+b<=m or i+a<=n and j+b<=m or i+a<=n and j-b>=1:
        result = -1
        if (i - 1) % a == 0 and (j - 1) % b == 0:
            tmp = max((i-1)/a, (j-1)/b)
            tmpv = (i - 1) / a + (j - 1) / b;
            if tmpv % 2 == 0:
                if result == -1:
                    result = tmp
                else:
                    result = min(result, tmp)
        if (i - 1) % a == 0 and (m - j) % b == 0:
            tmp = max((i-1)/a, (m-j)/b)
            tmpv = (i - 1) / a + (m - j) / b;
            if tmpv % 2 == 0:
                if result == -1:
                    result = tmp
                else:
                    result = min(result, tmp)
        if (n - i) % a == 0 and (j - 1) % b == 0:
            tmp = max((n-i)/a, (j-1)/b)
            tmpv = (n - i) / a + (j - 1) / b;
            if tmpv % 2 == 0:
                if result == -1:
                    result = tmp
                else:
                    result = min(result, tmp)
        if (n - i) % a == 0 and (m - j) % b == 0:
            tmp = max((n-i)/a, (m-j)/b)
            tmpv = (n - i) / a + (m - j) / b;
            if tmpv % 2 == 0:
                if result == -1:
                    result = tmp
                else:
                    result = min(result, tmp)
        print "Poor Inna and pony!" if result == -1 else result
    else:
        print "Poor Inna and pony!"
