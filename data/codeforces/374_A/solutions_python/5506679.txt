
I=lambda:map(int, raw_input().split())
n, m, i, j, a, b = I()
ans = 99999999999999
for c, d in [(1, 1), (1, m), (n, 1), (n, m)]:
    dx, dy = abs(c-i), abs(d-j)
    if dx + dy == 0:
        print 0;exit()
    if dx%a == 0 and dy%b == 0 and  (dx/a - dy/b)%2 == 0:
        ans = min(ans, max(dx/a, dy/b))
if (n-i<a and i-1<a) or (m-j<b and j-1<b):
    print 'Poor Inna and pony!';exit()
if ans != 99999999999999:
    print ans
else: print 'Poor Inna and pony!'
