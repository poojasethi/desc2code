n,m,i,j,a,b = map(int,raw_input().split())
c = max(n,m)
for p1,p2 in ((1,1),(1,m),(n,1),(n,m)):
    dy = abs(p2-j); dx = abs(p1-i);
    if dy % b == 0 and dx % a == 0 and dy/b % 2 == dx/a % 2:
        if (a > max(i-1,n-i) and dy > 0) or (b > max(j-1,m-j) and dx > 0):
            break
        c = min(c,max(dy/b,dx/a))
print c if c < max(n,m) else 'Poor Inna and pony!'
