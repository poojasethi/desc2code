n, m, i, j, a, b = map(int, raw_input().split())
moves = max(n, m)
for d in [(n, m), (n, 1), (1, m), (1, 1)]:
	di, dj = abs(i - d[0]), abs(j - d[1])
	if di % a == 0 and dj % b == 0 and (di / a) % 2 == (dj / b) % 2:
		# need to make a move but have no room!
		if (a > max(i - 1, n - i) and dj > 0) or (b > max(j - 1, m - j) and di > 0):
			continue
		moves = min(moves, max(di / a, dj / b))
print moves if moves < max(n, m) else 'Poor Inna and pony!'
