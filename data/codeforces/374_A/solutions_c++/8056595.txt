#include<iostream>

using namespace std ;

int n , m , x , y , a , b ;

unsigned f(int x , int y)
{
	if(x%a!=0 || y%b!=0)
		return -1 ;
	if((x!=0 || y!=0) && (a>=n || b>=m))
		return -1 ;
	x/=a , y/=b ;
	if(x%2!=y%2)
		return -1 ;
	return max(x,y) ;
}

int main()
{
	cin >> n >> m >> x >> y >> a >> b ;
	x-- , y-- ;
	unsigned ans=min(min(f(x,y),f(n-x-1,y)), min(f(x,m-y-1),f(n-1-x,m-y-1))) ;
	if(ans==-1)
		cout << "Poor Inna and pony!" << endl ;
	else
		cout << ans << endl ;
	return 0 ;	
}
