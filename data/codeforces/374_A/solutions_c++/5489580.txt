#include<cstdio>
#include<algorithm>
int n,m,x,y,a,b,ans;
int max(int a,int b)
{
	return a>b?a:b;
}
int min(int a,int b)
{
	return a<b?a:b;
}
void go(int p,int q)
{
	if(x==p&&y==q) {ans=0;return;}
	if(x+a>n&&x-a<=0) return;
	if(y+b>m&&y-b<=0) return;
	if(abs(x-p)%a||abs(y-q)%b) return;
	int c=abs(x-p)/a;
	int d=abs(y-q)/b;
	if(c%2==d%2) 
	{
		int z=max(c,d);
		ans=min(ans,z);
	}
	return;
}
int main()
{
	scanf("%d%d%d%d%d%d",&n,&m,&x,&y,&a,&b);
	ans=1<<30;
	go(1,1);
	go(1,m);
	go(n,1);
	go(n,m);
	if(ans==1<<30) printf("Poor Inna and pony!");
	else printf("%d",ans);
	return 0;
}
