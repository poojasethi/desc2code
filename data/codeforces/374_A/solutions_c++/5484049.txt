#include <iostream>
#include <cmath>

using namespace std;

int n,m,a,b;

const int infinito=1000000000;

int llega(int i,int j)
{
  if (i%a!=0 or j%b!=0) return infinito;
  if ((i!=0 or j!=0) and (a>=n or b>=m)) return infinito;
  i/=a;
  j/=b;
  if ((i+j)%2==1) return infinito;
  return max(i,j);
}

int main()
{
  int i,j;
  cin>>n>>m>>i>>j>>a>>b;
  i--;j--;
  int sol=min(min(llega(i,j),llega(n-1-i,j)),
	      min(llega(i,m-1-j),llega(n-1-i,m-1-j)));
  if (sol==infinito)
    cout<<"Poor Inna and pony!"<<endl;
  else
    cout<<sol<<endl;
}
