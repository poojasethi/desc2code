#include<iostream>
#include<algorithm>
using namespace std;
int n,m,i,j,a,b, res = 1<<30;
int myabs(int x){if(x < 0)return -x; return x;}
void go(int x, int y)
{
    if(x == i && y == j){ res = 0; return;}
    if(i + a > n && i - a < 1) return;
    if(j + b > m && j - b < 1) return;
    x = myabs(x - i); y = myabs(y - j);
    if(x % a || y % b) return;
    x /= a; y /= b;
    if(x % 2 != y % 2) return;
    res = min(res,max(x, y));
}
int main()
{
    cin >> n >> m >> i >> j >> a >> b;
    go(1, 1); go(1,m); go(n, 1); go(n, m);
    if(res == 1<<30) cout << "Poor Inna and pony!";
    else cout << res;
    return 0;
}
