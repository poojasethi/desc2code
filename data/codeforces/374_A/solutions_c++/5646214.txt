#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
#define INF 10000000
int n,m,i,j,a,b;
int f(int l,int r)
{
	if(i==l&&j==r) return 0;
	if(i+a>n&& i-a<1 || j+b>m&& j-b<1) return INF;
	if(abs(i-l)%a==0 && abs(j-r)%b==0 && (abs(i-l)/a) %2== (abs(j-r)/b) %2 ) 
	{
		int t=max( abs( i-l)/a, abs(j-r)/b);
		return t;
	}
	else return INF;
}
int main()
{
	cin>>n>>m>>i>>j>>a>>b;
	int t=min( min(f(1,1), f(1,m)), min(f(n,1), f(n,m) ) ) ;
	if( t==INF ) cout<<"Poor Inna and pony!";
	else cout<<t<<endl;
}
