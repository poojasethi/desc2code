#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	int n, m, i, j, a, b, i0, j0, x, y, ans=1e9;
	cin>>n>>m>>i>>j>>a>>b;
	if (i==1&&j==1 || i==1&&j==m || i==n&&j==1 || i==n&&j==m) {
		cout<<0<<endl; return 0;
	}
	if (i+a>n&&i-a<1 || j+b>m&&j-b<1) {
		cout<<"Poor Inna and pony!"<<endl; return 0;
	} 
	for (i0=1; i0<=n; i0+=max(1,n-1)) for (j0=1; j0<=m; j0+=max(1,m-1)) {
		x=abs(i-i0); y=abs(j-j0);
		if (x%a==0 && y%b==0 && x/a%2==y/b%2 && max(x/a, y/b)<ans)
			ans=max(x/a, y/b);  
	}
	if (ans==1e9) cout<<"Poor Inna and pony!"<<endl; 
	else cout<<ans<<endl;
	return 0;
}
