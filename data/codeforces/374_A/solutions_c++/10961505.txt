#include <iostream>
#include <cstdio>
#include <cmath>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;
int n,m,a,b;
const int INF=10000000000;

int get(int i, int j){
int ans=0;
if (i%a!=0||j%b!=0) return INF;
ans=min(i/a,j/b);
i-=ans*a;
j-=ans*b;
if (i&&j) return INF;
if ((j/b)%2!=0||(i/a)%2!=0) return INF;
if (i==0&&j!=0&&a>=n) return INF;
if (j==0&&i!=0&&b>=m) return INF;
return ans+j/b+i/a;
}

int main(){
int i,j;
cin>>n>>m>>i>>j>>a>>b;
int ans=min(min(get(i-1,j-1),get(n-i,j-1)),min(get(i-1,m-j),get(n-i,m-j)));
if (ans<INF) printf("%d", ans); else printf("Poor Inna and pony!");

return 0;
}
