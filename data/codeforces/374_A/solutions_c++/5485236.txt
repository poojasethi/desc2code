#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
	int x[2],y[2],n,m,i,j,a,b,ans=90019001;
	cin>>n>>m>>i>>j>>a>>b;
	x[0]=1;x[1]=n;
	y[0]=1;y[1]=m;
	for(int k=0;k<2;k++){
		for(int l=0;l<2;l++){
			if(i==x[k] && j==y[l]){
				cout<<"0"<<endl;
				return 0;
			}
			if(abs(i-x[k])%a==0 && abs(j-y[l])%b==0){
				int xx=abs(i-x[k])/a; int yy=abs(j-y[l])/b;
				if((xx+yy)%2==0){
					ans=min(ans,max(xx,yy));
				}
			}
		}
	}
	if(ans==90019001 || a>=n || b>=m)
		cout<<"Poor Inna and pony!"<<endl;
	else
		cout<<ans<<endl;
return 0;
}
