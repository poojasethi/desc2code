#include<bits/stdc++.h>
using namespace std;
int x,y,a,b,n,m;int get(int c,int d){if(abs(x-c)%a||abs(y-d)%b)return 100000000;int x1=abs(x-c)/a,x2=abs(y-d)/b;if(c==x&&d==y)return 0;if((x+a>n&&x<=a)||(y+b>m&&y<=b)||(x1%2&&x2%2==0)||(x1%2==0&&x2%2))return 100000000;return max(x1,x2);}
int main(){cin>>n>>m>>x>>y>>a>>b;int res=min(get(1,m),min(get(n,1),min(get(1,1),get(n,m))));if(res==100000000)cout<<"Poor Inna and pony!";else cout<<res;}
