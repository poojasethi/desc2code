#include <cstdio>
#include <algorithm>
using namespace std;

const int inf = -1u/2;

int n, m, i, j, a, b;

inline int cal(int r, int c) {
	if (i == r && j == c) return 0;	
	if (abs(r - i) % a || abs(c - j) % b) return inf;
	if (abs(r - i) / a % 2 != abs(c - j) / b % 2) return inf;
	if (abs(i - 1) < a && abs(i - n) < a) return inf;
	if (abs(j - 1) < b && abs(j - m) < b) return inf;
	return max(abs(i - r) / a, abs(j - c) / b);
}

int main() {
	scanf("%d %d %d %d %d %d", &n, &m, &i, &j, &a, &b);
	int ans = min(min(cal(1, m), cal(n, 1)), min(cal(n, m), cal(1, 1)));
	printf(ans == inf ? "Poor Inna and pony!": "%d", ans);
}
