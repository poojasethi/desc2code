#include<cstdio>
#include<cmath>
#include<algorithm>

using namespace std;

int n,m,a,b,x,y;
const int inf=1e7;

int cal(int tx,int ty){
    int dx=abs(x-tx);
    int dy=abs(y-ty);
    if(dx%a||dy%b)return inf;
    dx/=a;dy/=b;
    if(dx%2!=dy%2)return inf;
    if(dy&&x+a>n&&x-a<=0)return inf;
    if(dx&&y+b>m&&y-b<=0)return inf; 
    return max(dx,dy);
}

int main(){
    int as=inf;
    scanf("%d%d%d%d%d%d",&n,&m,&x,&y,&a,&b);
    as=min(as,cal(1,1));
    as=min(as,cal(1,m));
    as=min(as,cal(n,m));
    as=min(as,cal(n,1));
    if(as==inf)puts("Poor Inna and pony!");
    else printf("%d\n",as);
    return 0;
}
