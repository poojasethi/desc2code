#include <iostream>
#include <algorithm>
using namespace std;

const int inf = 1e9 + 7;

int n, m, x, y, a, b;

int solve(int n, int m)
{
    int t = max(n - x, x - n), v = max(m - y, y - m);
    if (t % a || v % b || (t / a % 2) != (v / b % 2))  
        return inf;
    return  max(t / a, v / b); 
}

int main()
{
    cin >> n >> m >> x >> y >> a >> b;
    int ans = min(min(solve(1, 1), solve(n, 1)), min(solve(1, m), solve(n, m)));
    if (ans == inf || (ans && (n <= a || m <= b))) 
        cout << "Poor Inna and pony!" << endl;
    else
        cout << ans << endl;
    return 0;
}
