#include <bits/stdc++.h>
using namespace std;

int limit = 10000000;
int a,b,n,m,i,j;

int fun(int x,int y)
{
    if(x == i && y == j) return 0;
    if(i%a == x%a && j%b == y%b && (abs(x-i)/a)%2 == (abs(y-j)/b)%2)
        if((x == i && i-a < 1 && i+a > n) || (y == j && j-b < 1 && j+b > m));
        else return max(abs(x-i)/a,abs(y-j)/b);
    return limit;
}
int main() {
    cin>>n>>m>>i>>j>>a>>b;
    int ans = limit;
    ans = min(ans , fun(1,m));
    ans = min(ans , fun(1,1));
    ans = min(ans , fun(n,m));
    ans = min(ans , fun(n,1));
    if(ans == limit)cout<<"Poor Inna and pony!"<<endl;
    else cout<<ans<<endl;
    return 0;
}
