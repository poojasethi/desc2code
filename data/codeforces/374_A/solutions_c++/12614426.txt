#include<bits/stdc++.h>
using namespace std;

int n,m,i,j,a,b;

int P(int x,int y){
    int c = abs(x-i); int d = abs(y-j);
    if(c == 0 && d == 0) return 0;
    if(c%a != 0 || d%b != 0) return 2000000000;
    if(a+1 > n || b+1 > m) return 2000000000;
    c /= a; d /= b;
    if(c%2 == 0 && d%2 != 0) return 2000000000;
    if(c%2 != 0 && d%2 == 0) return 2000000000;
    return max(c,d);
}

int main(){
    cin >> n >> m >> i >> j >> a >> b;
    int ans = min(min(P(1,m),P(n,1)),min(P(n,m),P(1,1)));
    if(ans == 2000000000) cout << "Poor Inna and pony!\n";
    else cout << ans << '\n';
    return 0;
}
