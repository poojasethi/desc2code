#include <iostream>
#include<stdio.h>
#include<cmath>
using namespace std;

int n,m,i,j,a,b;
int ok(int x, int y){
	if((i-x)%a==0&&(j-y)%b==0&&((i-x)/a-(j-y)/b)%2==0) return max(abs(i-x)/a,abs(j-y)/b);
 return 1e9;	
}
int main() {
	scanf("%d%d%d%d%d%d",&n,&m,&i,&j,&a,&b);
	int ans=min(min(ok(1,1),ok(n,m)),min(ok(n,1),ok(1,m)));
	if(ans==1e9) printf("Poor Inna and pony!");
	else if(ans>0&&(n<=a||m<=b)) printf("Poor Inna and pony!");
	else printf("%d",ans);
	return 0;
}
