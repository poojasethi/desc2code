#include<iostream>
#include<algorithm>
using namespace std;
int n, m, i, j, a, b;

int cal(int c, int d)
{
    if(abs(i-c)%a!=0) return 1e9;
    if(abs(j-d)%b!=0) return 1e9;
    int x=abs(i-c)/a;
    int y=abs(j-d)/b;

    if(x+y>0 && ((i+a>n && i-a<1) || (j+b>m && j-b<1)))
        return 1e9;
    if(x%2==y%2) return max(x, y);
    else return 1e9;
}


int main()
{
    int ans;
    cin>>n>>m>>i>>j>>a>>b;
    ans=min(min(cal(1, 1), cal(1, m)), min(cal(n, m), cal(n, 1)));
    if(ans==1e9)
        cout<<"Poor Inna and pony!"<<endl;
    else
        cout<<ans<<endl;
    return 0;
}
