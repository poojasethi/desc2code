#include<cstdio>
#include<algorithm>
using namespace std;

int n, m, x, y, a, b, ans = 1e7, mx, my;

void solve(int ay, int ax) {
	int tx = ax / a;
	int ty = ay / b;
	if (ay % b == 0 && ax % a == 0 && (ay / b - ax / a) % 2 == 0) {
		if (tx && !ty && b>my) return ;
		if (ty && !tx && a>mx) return ;
		ans = min(ans, max(ay / b, ax / a));
	}
}

int main() {
	scanf("%d%d%d%d%d%d", &n, &m, &x, &y, &a, &b);
	mx = max(x-1, n-x);
	my = max(y-1, m-y);
	solve(m - y, n - x);
	solve(y - 1, x - 1);
	solve(m - y, x - 1);
	solve(y - 1, n - x);
	if (ans != 1e7)
		printf("%d\n", ans);
	else
		puts("Poor Inna and pony!");
	return 0;
}
