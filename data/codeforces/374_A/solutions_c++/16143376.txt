#include <bits/stdc++.h>

using namespace std;
int n , m , a , b;

unsigned solve(int i,int j)
{
    if(i%a!=0||j%b!=0)return -1;
    if((i!=0||j!=0)&&(a>=n||b>=m))return -1;
    i/=a,j/=b;
    if(i%2!=j%2)return -1;
    return max(i,j);
}




int main()
{
    int x1 , x2 , y1 , y2 , i , j  ;

    cin >> n >> m >> i >> j >> a >> b;
    i-- , j--;
    unsigned s=min(min(solve(i,j),solve(n-1-i,j)),min(solve(i,m-1-j),solve(n-1-i,m-1-j)));

     if(s==-1)cout<<"Poor Inna and pony!"<<endl;

    else cout<<s<<endl;

    return 0;
}
