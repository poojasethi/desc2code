#include<bits/stdc++.h>
using namespace std;
int n,m,i,j,a,b;
int getmin(int tx,int ty)
{
    if(tx%a!=i%a)return 9999999;
    if(ty%b!=j%b)return 9999999;
    if(((abs(tx-i))/a+abs(ty-j)/b)%2==0)return max((abs(tx-i))/a,(abs(ty-j))/b);
    return 9999999;
}
int main()
{
    cin>>n>>m>>i>>j>>a>>b;
    if((i==1&&j==1)||(i==1&&j==m)||(i==m&&j==1)||(i==m&&j==n)){cout<<0<<"\n";return 0;}
    if((i+a>n&&i-a<1)||(j+b>m&&j-b<1))
    {
        cout<<"Poor Inna and pony!\n";return 0;
    }
    int ans=min(min(getmin(1,1),getmin(1,m)),min(getmin(n,1),getmin(n,m)));
    if(ans>=9999999)cout<<"Poor Inna and pony!\n";
    else cout<<ans<<endl;
}
