#include<bits/stdc++.h>

using namespace std;
int n, m, i, j, a, b;

int go(int x, int y){
	int x1 = x - i;
	int y1 = y - j;
	if (x1%a!=0 || y1%b!=0) return 1000000000;
	x1 = abs(x1/a);
	y1 = abs(y1/b);
	int d = x1 + y1;
	if (d%2) return 1000000000;
	else return max(x1,y1);
	
}
int main(){
	cin >> n >> m >> i >> j >> a >> b;
	if ((i==1 && (j==1 || j==m))||((i==n)&&(j==1 || j== m))) cout << "0" << endl;
	else
	if(n<=a || m<=b) cout << "Poor Inna and pony!" << endl;
	else{
	
	int dmin = min(min(go(1,m), go(n,m)), min(go(n,1),go(1,1)));
	if (dmin < 1000000000) cout << dmin << endl;
	else
		cout << "Poor Inna and pony!" << endl;
	}
}
