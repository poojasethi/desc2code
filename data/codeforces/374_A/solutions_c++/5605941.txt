#include<iostream>
#include<cstdio>
#include<cmath>

using namespace std;

int n, m, i, j, a, b, inf=1000000000, ans=inf;

int abs(int x)
{
    if(x>0)return x; else return -x;
}

void go(int x1, int y1, int x2, int y2)
{
    int q=abs(x1-x2);
    int w=abs(y1-y2);
    if(q%a||w%b||((q/a)%2!=(w/b)%2))return;
    ans=min(ans, max(q/a, w/b));
}

int main()
{
    cin>>n>>m>>i>>j>>a>>b;
    go(i, j, 1, 1);
    go(i, j, 1, m);
    go(i, j, n, 1);
    go(i, j, n, m);
    if(ans==inf||(ans>0&&((i+a>n&&n-a<1)||(j+b>m&&m-b<1))))
    {
        cout<<"Poor Inna and pony!";
        return 0;
    }
    cout<<ans;
}
