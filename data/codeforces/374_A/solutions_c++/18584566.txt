#include <iostream>
#include <cmath>

using namespace std;

const int inf = 1e9;

int n, m, x, y, a, b, ans = inf;

int solve(int i, int j) {
	if(i == x && j == y) return 0;
	int x1 = abs(i - x);
	int y1 = abs(j-y);
	if(x1 % a || y1 % b || (x1 / a - y1 / b) % 2) return ans ;
	if(a >= n || b >= m) return ans;
	return max(x1 / a, y1 / b);
}

int main() {
    cin >> n >> m >> x >> y >> a >>b;
	ans = min(solve(1,m), solve(n,1));
	ans = min(ans, solve(n,m));
	ans = min(ans, solve(1,1));
	if (ans != 1e9)
		cout << ans;
	else
		cout <<"Poor Inna and pony!";
	return 0;
}
