#include<bits/stdc++.h>
using namespace std;

long long n,m,x,y,a,b;

long long go(long long edx,long long edy)
{
	if(edx==x&&edy==y)
		return 0;
	if(a>max(n-x,x-1)||b>max(m-y,y-1))
		return 1<<29;
	if(abs(edx-x)%a!=0)
		return 1<<29;
	if(abs(edy-y)%b!=0)
		return 1<<29;
	long long xstep=abs(edx-x)/a;
	long long ystep=abs(edy-y)/b;
	if(abs(xstep-ystep)%2)
		return 1<<29;
	return max(xstep,ystep);
}

int main()
{
	long long ans;
	while(cin>>n>>m>>x>>y>>a>>b)
	{
		ans=min(min(go(1,1),go(1,m)),min(go(n,1),go(n,m)));
		if(ans<1<<29)
			cout<<ans<<endl; 
		else
			cout<<"Poor Inna and pony!\n";
	}
}
