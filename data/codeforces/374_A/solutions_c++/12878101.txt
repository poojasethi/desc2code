#include <iostream>
#include <cmath>

using namespace std;
int n,m,i,j,a,b,ans;
int calc(int x, int y)
{
    if((i-x)%a == 0 && (j-y)%b == 0)
    {
        if(((i-x)/a - (j-y)/b) % 2 == 0)
            return max(abs((i-x)/a),abs((j-y)/b));
    }
    return INT_MAX;
}
int main()
{
    cin >> n >> m >> i >> j >> a >> b;
    ans = min(min(min(calc(1,1),calc(n,m)),calc(n,1)),calc(1,m));
    if((ans > 0) && (n <= a || m <= b))
        cout << "Poor Inna and pony!";
    else if(ans == INT_MAX)
        cout << "Poor Inna and pony!";
    else
        cout << ans;
}
