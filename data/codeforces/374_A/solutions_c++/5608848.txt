#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define inf 1000000000
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,a,b,an,f,h;
inline void work(int n,int m) {
	int A=n/a,B=m/b;
	if (n%a==0&&m%b==0) {
		if (A==B) an=min(an,A);
		else if ((A-B)%2==0) {
			if (A<B&&f) an=min(an,B);
			if (B<A&&h) an=min(an,A);
		}
	}
}
int main() {
	scanf("%d%d%d%d%d%d",&n,&m,&i,&j,&a,&b);
	an=inf; f=a<n,h=b<m;
	work(i-1,j-1),work(i-1,m-j);
	work(n-i,j-1),work(n-i,m-j);
	if (an==inf) printf("Poor Inna and pony!\n");
	else printf("%d\n",an);
	//for(;;);
	return 0;
}
