#include<cstdio>
#include<algorithm>
#include<cstdlib>

using namespace std;

int n, m, x, y, a, b, ans = 1e7;

int solve(int i, int j) {
	if(i==x && j==y) return 0;
	int tx = abs(i-x);
	int ty = abs(j-y);
	if(tx%a || ty%b || (tx/a-ty/b)%2) return ans ;
	if(a>=n || b>=m) return ans;
	return max(tx/a, ty/b);
}

int main() {
	scanf("%d%d%d%d%d%d", &n, &m, &x, &y, &a, &b);
	ans = min(solve(1,m), solve(n,1));
	ans = min(ans, solve(n,m));
	ans = min(ans, solve(1,1));
	if (ans != 1e7)
		printf("%d\n", ans);
	else
		puts("Poor Inna and pony!");
	return 0;
}
