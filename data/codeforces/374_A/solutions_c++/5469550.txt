int  po(int xx,int yy)
{
 if (xx==0 && yy==0 ) return 0;
 if (xx%a || yy%b) return -1;
 xx/=a;yy/=b;
 if (xx%2!=yy%2) return -1;

 return max(xx,yy);
}
