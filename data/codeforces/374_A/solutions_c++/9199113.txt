#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int n,m,a,b;
int inf = 1000000000;
int step(int i,int j)
{
    if (i%a!=0 || j%b!=0) return inf;
    if ((i!=0||j!=0) && (a>=n||b>=m)) return inf;
    i /= a, j/=b;
    if (i%2 != j%2) return inf;
    return max(i,j);    
}

int main()
{
    int i,j;
    cin >> n >> m >> i >> j >> a >> b;
    int ans;
    i--, j--;
    ans = min(min(step(i,j),step(n-i-1,j)),min(step(i,m-1-j),step(n-1-i,m-1-j)));
    if (ans == inf) cout << "Poor Inna and pony!\n";
    else cout << ans << endl;
    //system("pause");
    return 0;
}
