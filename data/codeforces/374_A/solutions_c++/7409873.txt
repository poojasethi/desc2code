#include<iostream>
using namespace std;
int n,m,a,b;
unsigned f(int i,int j)
{
    if(i%a!=0||j%b!=0)return -1;
    if((i!=0||j!=0)&&(a>=n||b>=m))return -1;
    i/=a,j/=b;
    if(i%2!=j%2)return -1;
    return max(i,j);
}
int main()
{
    int i,j;
    cin>>n>>m>>i>>j>>a>>b;
    i--,j--;
    unsigned ans=min(min(f(i,j),f(n-1-i,j)),min(f(i,m-1-j),f(n-1-i,m-1-j)));
    if(ans==-1)cout<<"Poor Inna and pony!";
    else cout<<ans<<endl;
    return 0;
}
