#include <bits/stdc++.h>
using namespace std;

int main(){
    int n,m,i,j,a,b;
    cin>>n>>m>>i>>j>>a>>b;

    int dr[4]={1,1,n,n};
    int dc[4]={1,m,1,m};



    int res=1<<30;
    for(int k=0;k<4;k++){
        int r=dr[k],c=dc[k];
        if(abs(r-i)%a==0 && abs(c-j)%b==0 && (abs(r-i)/a - abs(c-j)/b)%2==0 ){
              if( r==i && i-a<1 && i+a>n  && c!=j  )continue;
              if( c==j && j-b<1 && j+b>m  && r!=i )continue;
              res=min(res,max(abs(r-i)/a , abs(c-j)/b));
        }

    }

    if(res < (1<<30))cout<<res<<"\n";
    else cout<<"Poor Inna and pony!\n";
}
