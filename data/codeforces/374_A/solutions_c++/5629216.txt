#include <iostream>
#include <cmath> 
#include <algorithm>
#define INF 1000000000
using namespace std;
int n,m,i,j,a,b,res=INF;
int dist(int x, int y, int x2, int y2)
{
	int dx = abs(x-x2), dy = abs(y-y2);
	if(dx==0 && dy==0) return 0;
	if(dx%a!=0 || dy%b!=0) return INF;
	if( (dx/a)%2 != (dy/b)%2) return INF;	
	if((x+a>n && x-a < 1)|| (y+b>m && y-b < 1)) return INF;
	return max(dx/a,dy/b);
}
int main()
{
	cin >> n >> m >> i >> j >> a >> b;
	res=min(res,dist(i,j,1,m));
	res=min(res,dist(i,j,n,1));
	res=min(res,dist(i,j,1,1));
	res=min(res,dist(i,j,n,m));
	if(res>=INF)
		cout << "Poor Inna and pony!";
	else
		cout << res;
}
