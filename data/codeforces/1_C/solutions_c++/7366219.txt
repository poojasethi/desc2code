#include<cstdio>
#include<cmath>
#define sqr(x) ((x)*(x))
typedef double db;
db x[3],y[3],d[3],v[3],a=1e100;
int main(){
    int i,j,OK;
    for(i=0;i<3;++i)scanf("%lf%lf",x+i,y+i);
    for(i=0;i<3;++i)for(j=i+1;j<3;++j)d[3-i-j]=sqrt(sqr(x[i]-x[j])+sqr(y[i]-y[j]));
    db s1=2.0*d[0]*d[1]*d[2],s2=sqr(d[0])+sqr(d[1])+sqr(d[2]);
    for(i=0;i<3;++i)v[i]=acos((s2-2*sqr(d[i]))/(s1/d[i]));
    for(i=3;i<=200;++i){
        db vv=M_PI/i;
        for(OK=1,j=0;j<3;++j)if(fabs(int(v[j]/vv+0.5)*vv-v[j])>0.0001)OK=false;
        if(OK){
            db r=d[1]/sin(v[1])/2.0,rr=i*r*r*sin(2*M_PI/i)/2;
            if(rr<a)a=rr;
        }
    }
    printf("%.6lf\n",a);
    return 0;
}
