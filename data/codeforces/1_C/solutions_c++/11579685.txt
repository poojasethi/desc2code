#include<stdio.h>
#include<math.h>

using namespace std;

inline double sqr(double x){
  return x*x;
}

bool fit(double w, int N){
  double s = sin(w*N);
  double eps=1e-4;
  return (s<eps && s > - eps);
}

int main(int agrc,char*argv[]){
  double p[6];
  for(int i=0;i<6;i++)
    scanf("%lf",&p[i]);
  double s[3];
  for(int i=0;i<3;i++){
    int c = (i+1)%3;
    s[i] = sqrt(sqr(p[2*i] - p[2*c]) + sqr(p[2*i+1]-p[2*c+1]) );
  }
  double angle[3];
  for(int i=0;i<3;i++){
    int c1=(i+1)%3, c2=(i+2)%3;
    double cosangle=(sqr(s[c1])+sqr(s[c2])-sqr(s[i]))/(2*s[c1]*s[c2]);
    angle[i] = acos(cosangle);
  }
  int N=3;
  while(!fit(angle[0],N) || !fit(angle[1],N) || !fit(angle[2],N))
    N++;
  double R = s[0] /(2*sin(angle[0]));
  double A = 0.5*N*R*R * sin(2*acos(-1)/N);
  printf("%lf",A);
  return 0;
}
