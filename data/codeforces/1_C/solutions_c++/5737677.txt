#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
double point[3][2];
double pi=acos(-1);
double eps=1e-4;
double cal_dist(int i,int j)
{
double x,y;
x=point[i][0]-point[j][0];
y=point[i][1]-point[j][1];
return sqrt(x*x+y*y);
}
double cal_angle(double x,double y,double z)
{
return acos((x*x+y*y-z*z)/(2*x*y));
}
double gcd(double a,double b)
{
if(fabs(b)<eps)
 return a;
return gcd(b,fmod(a,b));
}
int main()
{
for(int i=0;i<3;i++)
scanf("%lf %lf",&(point[i][0]),&(point[i][1]));
double a,b,c;
a=cal_dist(0,1);
b=cal_dist(1,2);
c=cal_dist(2,0);
double A,B,C;
A=cal_angle(b,c,a);
B=cal_angle(c,a,b);
C=cal_angle(a,b,c);
int n=pi/gcd(A,gcd(B,C));
double R=0.5*a/(sin(A));//To calculate circum radius
double area=0.5*R*R*n*sin(2*pi/n);
printf("%.10lf \n",area);
}
