#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#define pi acos(-1)
int i;
double x[10],y[10],d[10],p,s,r,S,mm,rad[10];

double fgcd(double a,double b)
{
       if (fabs(a)<1e-2) return b;
       if (fabs(b)<1e-2) return a;
       return fgcd(b,fmod(a,b));
}

double sqr(double x) { return x*x; }

int main()
{
    for (i=0;i<3;i++) scanf("%lf%lf",&x[i],&y[i]);
    for (i=0;i<3;i++)
        d[i]=sqrt(sqr(x[i]-x[(i+1)%3])+sqr(y[i]-y[(i+1)%3]));
    p=(d[0]+d[1]+d[2])/2;
    s=sqrt(p*(p-d[0])*(p-d[1])*(p-d[2]));
    r=d[0]*d[1]*d[2]/(4*s);
    for (i=0;i<2;i++)
        rad[i]=2*asin(d[i]/2/r);
    rad[2]=2*pi-rad[1]-rad[0];
    rad[3]=fgcd(rad[0],fgcd(rad[1],rad[2]));
    printf("%.6lf\n",r*r*sin(rad[3])*pi/rad[3]);
    return 0;
}
