#include <iostream>
#include<cstdio>
#include<cmath>
#define D double
#define Six(x)    ((x)*(x))
#define Auf(t)    a[t]=x[t]-x[2];b[t]=y[t]-y[2];c[t]=(Six(x[t])+Six(y[t])-Six(x[2])-Six(y[2]))/2;
#define Mio(p,q)  (p[0]*q[1]-p[1]*q[0])
D g(D a,D b){return fabs(b)<1e-4?a:g(b,fmod(a,b));}
D x[3],y[3],a[3],b[2],c[2],A,X,Y;
int main(){
    for(int i=0;i<3;i++)scanf("%lf%lf",x+i,y+i);
    Auf(0);Auf(1);
    X=Mio(c,b)/Mio(a,b);
    Y=Mio(a,c)/Mio(a,b);
    for(int i=0;i<3;i++)a[i]=atan2(x[i]-=X,y[i]-=Y);
    A=g(M_PI*2,g(fabs(a[1]-a[0]),fabs(a[2]-a[0])));
    printf("%lf\n",(Six(x[0])+Six(y[0]))*sin(A)*M_PI/A);
    return 0;
}
