#include<cstdio>
#include<cmath>
#define eps 1e-4
#define pi 4*atan(1)
using namespace std;
double gcd(double x,double y)
{
	if(y>eps)
	return gcd(y,x-floor(x/y)*y);
	else
	return x;
}
double theta(double a,double b,double c)
{
	return acos((a*a+b*b-c*c)/(2*a*b));
}
int main(void)
{
	double ax,ay,bx,by,cx,cy;
	while(scanf("%lf%lf%lf%lf%lf%lf",&ax,&ay,&bx,&by,&cx,&cy)==6)
	{
		double a=sqrt((ax-bx)*(ax-bx)+(ay-by)*(ay-by));
		double b=sqrt((bx-cx)*(bx-cx)+(by-cy)*(by-cy));
		double c=sqrt((cx-ax)*(cx-ax)+(cy-ay)*(cy-ay));
		double p=(a+b+c)/2;
		double s=sqrt(p*(p-a)*(p-b)*(p-c));
		double r=(a*b*c)/(4*s);
		double A=theta(b,c,a);
		double B=theta(a,c,b);
		double C=theta(a,b,c);
		double n=pi/(gcd(A,gcd(B,C)));
		printf("%.8lf\n",0.5*r*r*n*sin(2*pi/n));
	}
	return 0;
}
