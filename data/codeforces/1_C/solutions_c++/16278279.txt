#include<iostream>
#include<cstdio>
#include<cmath>

double length(double a,double b)
{
	return a*a+b*b;
}

double gcd(double a,double b)
{
	return b<1E-4?a:gcd(b,fmod(a,b));
}

double xa,ya,xb,yb,xc,yc;
double a,b,c,A,B,C;
double x,y;

int main()
{
	while(~scanf("%lf%lf%lf%lf%lf%lf",&xa,&ya,&xb,&yb,&xc,&yc))
	{

		y=length(xa,ya);
		a=xb-xa,b=yb-ya,c=(length(xb,yb)-y)*0.5;
		A=xc-xa,B=yc-ya,C=(length(xc,yc)-y)*0.5;
		y=1.0/(a*B-A*b);
		x=(c*B-C*b)*y,y*=(a*C-A*c);


		a=atan2(ya-=y,xa-=x);
		b=atan2(yb-=y,xb-=x);
		c=atan2(yc-=y,xc-=x);

		a=gcd(gcd(fabs(b-a),fabs(c-a)),2*M_PI);

		printf("%.11lf\n",(length(xa,ya))*sin(a)*M_PI/a);
	}
	return 0;
}
