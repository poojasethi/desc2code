#include<stdio.h>
#include<math.h> 
const double Pi = acos(-1);
double Heron(double a, double b, double c)
{
	double p=(a+b+c)/2;
	return sqrt(p*(p-a)*(p-b)*(p-c));
}
bool IsInteger(double f)
{
	if(fabs(floor(f+0.5)-f)<1e-5)
		return true;
	else
		return false;
}
int main()
{
	double x1,y1,x2,y2,x3,y3;
	scanf("%lf%lf%lf%lf%lf%lf",&x1,&y1,&x2,&y2,&x3,&y3);
	double a=hypot(x1-x2,y1-y2);
	double b=hypot(x2-x3,y2-y3);
	double c=hypot(x3-x1,y3-y1);
	double S=Heron(a,b,c);
	double r=a*b*c*.25/S;
	
	double t1=.5*acos(1-.5*a*a/r/r)/Pi;
	double t2=.5*acos(1-.5*b*b/r/r)/Pi;
	double t3=.5*acos(1-.5*c*c/r/r)/Pi;
	
	int k;
	for(k=3;k<=100;k++)
		if(IsInteger(t1*k) && IsInteger(t2*k) && IsInteger(t3*k))
			break;
	printf("%f\n",.5*k*r*r*sin(2*Pi/k));
	return 0;
}
