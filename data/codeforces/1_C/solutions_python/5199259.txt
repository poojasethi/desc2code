from math import acos, sin, pi
def f(x, y): return sum((i - j) ** 2 for i, j in zip(x, y))
def g(a, b, c): return acos((b + c - a) / (2 * (b * c) ** 0.5)) / pi
def h(x, n): return 0.01 < (x * n) % 1 < 0.99
x, y, z = (tuple(map(float, raw_input().split())) for i in range(3))
a, b, c = f(x, y), f(x, z), f(y, z)
A, B = g(a, b, c), g(b, a, c)
r, n = a / (8 * sin(A * pi) ** 2), 3
while h(A, n) or h(B, n): n += 1
print n * r * sin(2 * pi / n)
