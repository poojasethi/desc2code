from math import *
p=[map(float,raw_input().split()) for i in [0]*3]
a,b,c=[hypot(x-X,y-Y) for (x,y),(X,Y) in [(p[0],p[1]),(p[1],p[2]),(p[0],p[2])]]
A,B,C=[acos((y*y+z*z-x*x)/2/z/y) for x,y,z in [(a,b,c),(b,c,a),(c,a,b)]]
R=a/2/sin(A)
def g(x,y):
    if y < pi/100: return x
    return g(y,fmod(x,y))
u=2*g(A,g(B,C))
print R*R*pi/u*sin(u)
