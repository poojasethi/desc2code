import math



def gcd(a,b):

	if b < 1e-3:

		return a

	else:

		return gcd(b,math.fmod(a,b))





p=[[float(x) for x in raw_input().split(' ')] for y in range(0,3)]

a,b,c=[math.hypot(x1-x2,y1-y2) for (x1,y1),(x2,y2) in [(p[0],p[1]),(p[1],p[2]),(p[0],p[2])]]

A,B,C=[math.acos((a*a+b*b-c*c)/(2*a*b)) for a,b,c in [(b,c,a),(a,c,b),(a,b,c)]]

r=a/(2*math.sin(A))

ang=2*gcd(A,gcd(B,C))

ans=2*math.acos(-1)/ang*1/2*r*r*math.sin(ang)

print(round(ans,12))
