from math import hypot,acos,pi,sin,fmod

def gcd(a,b):
    if b<pi/100: return a
    return gcd(b,fmod(a,b))

p=[map(float,raw_input().split()) for i in range(3)]

a,b,c=[hypot(x[0]-y[0],x[1]-y[1]) for x,y in[(p[0],p[1]),(p[1],p[2]),(p[2],p[0])]]
ua,ub,uc=[acos((y**2+z**2-x**2)/(2*y*z)) for x,y,z in [(a,b,c),(b,c,a),(c,a,b)]]

R=a/(2*sin(ua))
u=2*gcd(ua,gcd(ub,uc))

print R**2*pi/u*sin(u)
