#include <cstdio>
using namespace std;

int main()
{
    int n, m, k;
    scanf("%d%d%d", &n, &m, &k);
    if(n+k-m < -1) {
        puts("0");
        return 0;
    }
    double x = 1.0, y = 1.0;
    for(int i = 0; i <= k; ++i) {
        x *= m-i;
        y *= n+i+1;
    }
    printf("%lf\n", 1.0-x/y);
    return 0;
}
