#include <iostream>
#include <cstdio>
using namespace std;

int n,m,k;
double ans;

int main()
{
   // freopen("26D.txt","r",stdin);
	cout.setf(ios::fixed); 
    cout.precision(8);
    scanf("%d%d%d",&n,&m,&k);
    ans=1;
    for (int i=0;i<=k;i++) ans*=(double)(m-i)/(n+i+1);
	if (n+k<m) cout<<0<<endl;else cout<<1-ans<<endl;

    return 0;
}
