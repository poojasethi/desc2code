#include <cstdio>
#include <cstring>
#include <iostream>
#include <cctype>
using namespace std;
int n,  m, k;
double s = 1;

int main()
{
    int i;
	//freopen("1.out", "w" , stdout);
	scanf("%d%d%d", &n, &m, &k);

	for (i = 0; i <= k; ++ i)
		s = s * (m - i + 0.0) / (n + i + 1);
	printf("%.9lf\n", 1 - min(s, 1.0));
	return 0;
}
