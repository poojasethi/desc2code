#include <cstdio>
#include <cmath>
using namespace std;

int main(){
	int n, m, k, i;
	
	scanf("%d%d%d", &n, &m, &k);
	if (m > n+k){
		printf("%.6lf\n", 0.0);
		return 0;
	}
	if (m <= k){
		printf("%.6lf\n", 1.0);
		return 0;
	}
	
	double ans = 1;
	for (i=0; i<=k; i++)
		ans*=(m-i),ans/=(n+i+1);
	printf("%.6lf\n", 1-ans);
	return 0;
}
