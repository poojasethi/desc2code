#include<iostream>
#include<cstdio>
using namespace std;

int n, m, k;
double ans;

int main()
{
	scanf("%d %d %d", &n, &m, &k);
	if (n + k < m)
	{
		printf("0.0000\n");
		return 0;
	}
	if (k >= m)
	{
		printf("1.0000\n");
		return 0;
	}
	ans = 1;
	for (int i = 0; i <= k; i++)
		ans *= double(m - i) / (n + i + 1);
	printf("%.4lf\n", 1 - ans);
}
