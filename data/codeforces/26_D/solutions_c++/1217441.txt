using namespace std;
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cassert>
#define REP(i, n) for (int i (0); i < (n); i ++)
typedef long double D;
main()
{
	int n, m, k; cin >> n >> m >> k;
	
	D divi (1);
	REP(i, k + 1)
	{
		divi *= m - i, divi /= n + i + 1;
	}
	D p (1 - divi);
	cout.precision(9);
	cout << fixed << max(p, (D)0.0) << endl;
}
