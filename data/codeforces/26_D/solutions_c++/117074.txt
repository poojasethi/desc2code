#include <cstdio>

int main() {
    int n, m, k;
    scanf("%d%d%d", &n, &m, &k);
    if (k >= m) printf("%.6lf\n", 1.0);
    else if (n + k < m) printf("%.6lf\n", 0.0);
    else {
        double r = 1.0;
        for (int i = 0; i <= k; ++i)
            r *= (double) (m - i) / (n + i + 1);
        printf("%.6lf\n", 1 - r);
    }
    return 0;
}
