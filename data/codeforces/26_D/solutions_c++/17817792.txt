#include<stdio.h>

double ans;
int n, m, k;
int main(){
    scanf("%d %d %d", &n, &m, &k);
    if(k >= m){ puts("1"); return 0; }
    if(k + n < m){ puts("0"); return 0; }
    ans = 1;
    for(int i = 0; i <= k; i ++){
        ans = ans * (m-i) / (n + k + 1 - i);
    }
    printf("%lf", 1 - ans);
}
