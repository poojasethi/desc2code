#include<iostream>
using namespace std;
int main( )
{
    int n,m,k;
    cin>>n>>m>>k;
    double r=1.0;
    if( n+k<m) {
        cout<<"0.000000"<<endl;
        return 0;
    }
    if( k>=m){
        cout<<"1.000000"<<endl;
        return 0;
    }
    for(int i=0;i<=k;i++)
        r *= (1.0*(m-i))/(n+1.0+i);
    r = 1-r;
    cout<<r<<endl;
    return 0;
}
