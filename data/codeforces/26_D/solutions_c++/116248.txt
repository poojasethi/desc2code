#include <cstdio>
int main()
{
	int n,m,k;
	scanf("%d %d %d",&n,&m,&k);
	double res=1;
   if(m>n+k)
    {
        printf("%.10f\n",0.0);
		return 0;
    }
    if(k>=m)
    {
         printf("%.10f\n",1.0);
		 return 0;
    }

	for(int i=0;i<=k;i++)
	{
		res*=1.0*(m-i)/(n+i+1);
	}
	printf("%.10f\n",1-res);
	return 0;
}
