#include <iostream>

using namespace std;

int main(){
	int n,m,k;
	cin>>n>>m>>k;
	//C(n+m,n)-C(n+m,m-k-1) the ones which touch X-axis
	double res=1.0;
	if(n+k<m){
		cout<<0<<endl;
		return 0;
	}
	for(int i=0;i<=k;i++){
		res*=(m-i+0.0);
		res/=(n+i+1.0);
		//cout<<res<<endl;
	}
	cout<<1.0-res<<endl;
}
