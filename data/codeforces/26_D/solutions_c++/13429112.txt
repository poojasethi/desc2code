#include <stdio.h>
#include <math.h>
using namespace std;
int main(){
	int n,m,k,t,i,j,x;
	scanf("%d %d %d",&n,&m,&k);

	if(n+k<m) {
		printf("0\n");return 0;
	}
	if(k>=m) {printf("1\n");return 0;}
	double bot=0,top =0;
	for(i=m,j=n+1;i>=m-k;--i,++j)
		top += log(i),bot+= log(j);
	top-=bot;
	printf("%0.6f\n",1-exp(top));
}
