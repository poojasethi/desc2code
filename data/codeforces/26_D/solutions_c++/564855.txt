#include <iostream>
#include<stdio.h>
using namespace std;
double ans;
int i,n,m,k;
int main() {
	scanf("%d%d%d",&n,&m,&k);
	if (n+k<m) printf("0\n");
	else if (m-k-1<0)printf("1\n");
	else{
		ans=1;
		for (i=m-k;i<=m;i++) ans*=(double)i;
		for (i=n+1;i<=n+1+k;i++) ans/=(double)i;
		printf("%lf\n",1-ans);

	}
	return 0;
}
