#include <stdio.h>
#include <math.h>
using namespace std;
int main(){
	int n,m,k,i,j;
	scanf("%d %d %d",&n,&m,&k);
	if(n+k<m) {
		printf("0\n");return 0;
	}
	if(k>=m){printf("1\n");return 0;}
	double b=0,t =0;
	for(i=m,j=n+1;i>=m-k;--i,++j)
		t += log(i),b+= log(j);
	printf("%0.6f\n",1-exp(t-b));
}
