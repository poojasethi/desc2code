#include <cstdio>
#include <cstring>
#include <cstdlib>

using namespace std;

int i, N, M, K;
double ans = 1.0;

int main()
 {
    scanf("%d%d%d", &N, &M, &K);
    for (i = 0; i <= K; i ++)
        ans *= (double) (M-i) / (N+i+1.0);
    ans = 1.0 - ans;
    printf("%lf\n", (ans < 0 ? 0 : ans));
    return 0;
 }
