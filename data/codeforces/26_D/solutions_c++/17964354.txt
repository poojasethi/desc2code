#include<cstdio>

int main(){
	int n, m, k;
	scanf("%d%d%d", &n, &m, &k);
	double ans = 1, mi = m, ni = n + k + 1;
	if (m <= k)
		printf("1");
	else{
		for (int i = 0; i <= k; i++){
			ans = ans*mi / ni;
			mi--;
			ni--;
		}
		if (ans <= 1)
			printf("%lf", 1 - ans);
		else
			printf("0");
	}
	return 0;
}
