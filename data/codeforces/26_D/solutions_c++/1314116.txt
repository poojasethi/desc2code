#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

int n,m,k;

int main() {
    scanf("%d%d%d",&n,&m,&k);
    if (m-n > k) { printf("0\n"); return 0; }
    if (m <= k) { printf("1\n"); return 0; }
    double ret = 1;
    for (int i=0; i<=k; i++)
        ret *= (m-i)*1.0/(n+i+1);
    printf("%f\n",1-ret);
}
