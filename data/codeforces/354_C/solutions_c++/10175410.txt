#include<iostream>

using namespace std;

const int MAXN = 1e7, MAX = 1e6;
int cnt[MAXN];

int main()
{
	int n, k;
	cin >> n >> k;
	for(int i = 0; i < n; i++)
	{
		int a;
		cin >> a;
		cnt[a]++;
	}
	for(int i = 1; i < MAXN; i++)
		cnt[i] += cnt[i - 1];
	int ans = 1;
	for(int i = 2; i <= MAX; i++)
	{
		int sum = 0;
		for(int j = i; j <= MAX; j += i)
			sum += cnt[min(j + k, j + i - 1)] - (j > 0?cnt[j - 1]:0);
		if(sum == n)
			ans = i;
	}
	cout << ans << endl;
	return 0;
}
