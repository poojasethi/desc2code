#include <stdio.h>
#include <algorithm>
using namespace std;
int n, k, S[300001], m = 0x7fffffff;
int main()
{
    int i, p;
    scanf("%d%d",&n,&k);
    for( i = 0 ; i < n ; i++ )
    {
        scanf("%d", &S[i]);
        m = m > S[i] ? S[i] : m;
    }
    for( int c = 0 ; c < 2 ; c++ )
    {
        for( i = p = 0 ; i < n ; i++ )
            while( S[i]%m > k ) m--;
    }
    printf("%d", m);
}
