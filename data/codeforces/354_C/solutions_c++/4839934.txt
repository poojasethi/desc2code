#include<stdio.h>
#include<algorithm>
int n,k,a[300005],min=1000000000;
int main(){
    int i,l;
    scanf("%d%d",&n,&k);
    
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
        if(min>a[i])min=a[i];
    }
    int p=1;
    while(p)
    {
        p=0;
        for(i=0;i<n;i++)
        {
            while(a[i]%min>k)
            {
                min--;
                p=1;
            }
        }
    }
    printf("%d",min);
}
