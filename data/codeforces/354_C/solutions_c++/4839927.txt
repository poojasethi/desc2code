#include<stdio.h>
int f[1000010];
int min(int i,int j){ return i<j?i:j;}
int main(){
    int n,k;
    scanf("%d%d",&n,&k);
    int i,m=99999999,t;
    for(i=0;i<n;i++){
        scanf("%d",&f[i]);
        m=min(m,f[i]);
    }
    int p;
    while(1){
        t = m;
        for(i=0;i<n;i++){
            p = f[i]/m;
            if(!(p > 0 && p*m+k>=f[i])) m = f[i]/(p+1);
        }
        if(t == m) break;
    }
    printf("%d",t);
    return 0;
 }
