#include <bits/stdc++.h>
using namespace std;
int main(){
	int n,k;
	cin>>n>>k;
	long long a[n];
	long long maxgcd = 1000000000;
	for(int i=0;i<n;i++){
		cin>>a[i];
		maxgcd=min(maxgcd,a[i]);
	}
	sort(a,a+n);
	while(true){
		bool t  = true;
		for(int i=0;i<n;i++){
			t=t&&(a[i]%maxgcd<=k);
			if(t==false){
				maxgcd=a[i]/(a[i]/maxgcd+1);
				break;
			}
		}
		if(t){
			break;
		}
	}
	cout << maxgcd << endl;
}
