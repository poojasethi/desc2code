#include <cstdio>

#define M 1000000

int a[M + 1], m = M, n, k, i, t;

int main()
{
	scanf("%d%d", &n, &k);

	for (i = 0; i < n; i ++)
	{
		scanf("%d", &t);
		a[t] ++;
		if (t < m)
			m = t;
	}

	if (m <= k + 1)
	{
		printf("%d\n", m);
		return 0;
	}

	for (i = 1; i <= M; i ++)
		a[i] += a[i - 1];

	do
	{
		t = 0;
		for (i = m; i <= M; i += m)
			t += a[i + k > M ? M : i + k] - a[i - 1];
		m --;
	}
	while (t < n);

	printf("%d\n", m + 1);
}
