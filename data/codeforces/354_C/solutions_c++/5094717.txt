#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
using namespace std;
int n,k,i,cnt[2200000],x,maxa,m,ans,res;
int main(){
	scanf("%d%d",&n,&k);
	m=1<<30;
	maxa=0;
	for(i=1;i<=n;++i)scanf("%d",&x),m=min(m,x),cnt[x]++,maxa=max(maxa,x);
	for(i=1;i<=2000000;++i)cnt[i]+=cnt[i-1];
	if(m<=k)return printf("%d\n",m),0;
	for(ans=m;ans>k;--ans){
		res=0;
		for(i=ans;i<=maxa;i+=ans)res+=cnt[i+k]-cnt[i-1];
		if(res==n)break;
	}
	printf("%d\n",ans);
}
