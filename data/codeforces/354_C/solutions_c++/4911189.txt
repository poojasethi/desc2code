#include <iostream>
#include <algorithm>

using namespace std;

const int MAX_N = 3e5 + 35;

int n, k, ret, niz[MAX_N];

int main()
{
  cin >> n >> k;
  for (int i = 0; i < n; i++)
    cin >> niz[i];
  sort(niz, niz + n);
  ret = niz[0];
  for (bool flag = true; flag;)
  {
    flag = false;
    for (int i = 0; i < n; i++)
      while (niz[i] % ret > k)
        ret--, flag = true;
  }
  cout << ret;
  return 0;
}
