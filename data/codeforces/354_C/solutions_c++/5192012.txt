#include<iostream>
#include<cstdio>

#define INF 99999999
using namespace std;

const int MAX=300000+10;
int s[MAX];

int main()
{
	int n,k;
	while(cin>>n>>k)
	{
		int gcd=INF,temp;
		for(int i=0;i<n;++i)
		{
			scanf("%d",&s[i]);
			gcd=min(gcd,s[i]);
		}
		do
		{
			temp=gcd;
			for(int i=0;i<n;++i)
			{
				int num = s[i]/gcd;
				if(s[i]-num*gcd > k)
					gcd = s[i]/(num+1);
			}
		}while (gcd!=temp);
		cout<<gcd<<endl;
	}
	return 0;
}
