#include<stdio.h>
#include<algorithm>
using namespace std;
int a[333333];
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	int i,j;
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	sort(a,a+n);

	if(k+1>=a[0])printf("%d\n",a[0]);
	else
	{
	for(i=a[0];i>k+1;i--)
     	{
		int temp=0;
		for(j=0;j<=a[n-1]/i;j++)
		{
			int l,r;
		 l=lower_bound(a,a+n,j*i)-a;
	     r=upper_bound(a,a+n,j*i+k)-a;
	     temp+=r-l;
			
		}

		if(temp==n)break;
     	}
	printf("%d\n",i);
	}


}
