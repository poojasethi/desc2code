#include <cstdio>
#include <algorithm>
int N,K,a=1e9,b,s[2000005];
int main()
{
    scanf("%d%d",&N,&K);
    for (int i=1,x; i<=N; i++) scanf("%d",&x),s[x]++,a=std::min(a,x),b=std::max(b,x);
    if (a<=K+1) {printf("%d\n",a); return 0;}
    for (int i=1; i<=b+K; i++) s[i]+=s[i-1];
    for (int i=a; i>K; i--)
    {
        int t=0;
        for (int j=1; j<=b/i; j++) t+=s[j*i+K]-s[j*i-1];
        if (t==N) {printf("%d\n",i); return 0;}
    }
}
