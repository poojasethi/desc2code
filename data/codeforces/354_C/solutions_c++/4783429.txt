#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int a[300010],n,k;
int main()
{
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++)
    scanf("%d",&a[i]);
    sort(a,a+n);
    int ans=a[0],b;
    while(1)
    {
        for(b=0;b<n;b++)
        {
            if(a[b]%ans>k)
            {
                ans=a[b]/(a[b]/ans+1);
                break;
            }
        }
        if(b==n)break;
    }
    printf("%d\n",ans);
    return 0;
}
