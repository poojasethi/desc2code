#include <cstdio>
#include <algorithm>
#define FOR(i,x,y) for(i=x;i<=y;++i)
using namespace std;
int i,n,k,minn,a[300010];
int main()
{
    scanf("%d%d",&n,&k);
    FOR(i,1,n) scanf("%d",&a[i]);
    sort(a+1,a+1+n);
    minn=a[1];
    while(true)
    {
        FOR(i,1,n) if(a[i]%minn>k)
        {
            while(a[i]%minn>k) --minn;
            break;
        }
        if(i>n) break;
    }
    printf("%d\n",minn);
    return 0;
}
