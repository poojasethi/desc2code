#include <cstdio>
#define min(x,y) (x<y?x:y)
using namespace std;

const int N = 300010;

int n,k,a[N],b,t,mn=1<<30;

int main()
{
	scanf("%d%d",&n,&k);
	for (int i=0;i<n;++i)
	{
		scanf("%d",&a[i]);
		mn = min(mn,a[i]);
	}
	while ( 1 )
	{
        t = mn;
	    for (int i=0;i<n;++i)
	    {
	        b = a[i] / mn;
	        if ( ! ( b > 0 && b * mn >= a[i] - k ) )
                mn = a[i] / (b + 1);
	    }
	    if ( t == mn ) break;
	}
	printf("%d\n",mn);
}
