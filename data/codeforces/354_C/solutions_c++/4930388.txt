#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
	int n, k;
	cin >> n >> k;
	vector<int> a(n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &(a[i]));
	}
	int ans = *min_element(a.begin(), a.end());

	while (true) {
		bool flag = true;
		for (int i = 0; i < n; i++) {
			while (a[i] % ans > k) {
				ans --;
				flag = false;
			}
		}
		if (flag) break;
	}

	cout << ans << '\n';
	return 0;
}
