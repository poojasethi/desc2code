#include <iostream>
using namespace std;
int as[1<<20];
int add[1<<20];
int main() {
	int n,k;cin>>n>>k;
	int hi=0,low=1e9;
	for(int i=0; i<n; ++i) {
		cin>>as[i];
		int x = as[i];
		int v = max(0,x-k);
		++add[v];
		--add[x+1];
		hi = max(hi,x);
		low = min(low,x);
	}
	for(int i=0; i<hi; ++i) add[i+1] += add[i];
	for(int r=low; ; --r) {
		int cnt=0;
		for(int i=r; i<=hi; i+=r) {
			cnt += add[i];
		}
		if (cnt>=n) {
			cout<<r<<'\n';
			break;
		}
	}
}
