#include <bits/stdc++.h>
using namespace std;
int A[1000020];
#define MAXI 1000000
int main()
{
	int N,K,v;
	cin>>N>>K;
	int mini = 1e9;
	for(int i = 0;i<N;i++)
	{
		cin>>v;
		A[v+1]-=1;
		A[max(1,v-K)]+=1;
		mini = min(v,mini);
	}
	for(int i = 1;i<=MAXI;i++)
		A[i]+=A[i-1];
	int Ans = min(mini,K+1);
	for(int i = K+2;i<=mini;i++)
	{
		int ans = 0;
		for(int j = i;j<=MAXI;j+=i){
			ans+=A[j];
		}
		if(ans==N)
			Ans = i;
	}
	cout<<Ans<<endl;
	return 0;
}
