#include <iostream>
using namespace std;

int main() {
    int n, k, a[300005], ans = 1234567;
    cin >> n >> k;
    
    for(int i = 0; i < n; i++) {
        cin >> a[i];
        ans = min(ans, a[i]);
    }
    
    while(1) {
        int t = ans;
        for(int i = 0; i < n; i++) {
            if(a[i] % t > k ) t = a[i] / (a[i] / t + 1);
        }
        if(t == ans) break;
        else ans = t;
    }
    
    cout << ans;
}
