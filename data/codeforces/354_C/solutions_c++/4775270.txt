#include <iostream>
using namespace std;
int a[2000006];
int main(){
    int n;
    int k;
    cin >> n >> k;
    int x;
    for(int i=0;i<n;i++){cin >> x;a[x]++;}
    for(int i=1;i<=2000000;i++){a[i]+=a[i-1];}
    int ans=0;
    for(int i=1;i<=1000000;i++){
    int sum=0;
    int dk=min(k,i-1);
    for(int j=i;j<=1000000;j+=i)sum+=(a[j+dk]-a[j-1]);
    if(sum==n)ans=i;
    }
    cout <<  ans << endl;
}
