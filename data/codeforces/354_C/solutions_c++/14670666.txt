#include <bits/stdc++.h>
using namespace std;

int main () {
	
	int n=0, k=0, i=0, small=0, flag=0;

	scanf("%d %d", &n, &k);
	int arr[n];

	small = INT_MAX;

	for (i=0; i<n; i++) {
		scanf("%d", &arr[i]);
		small = min (small, arr[i]);
	}

	flag=1;
	while (flag != 0) {

		flag = 0;
		for (i=0; i<n; i++)
			while ( (arr[i]%small) > k )
				flag = small--;
	}

	printf("%d\n", small);


	return 0;
}
