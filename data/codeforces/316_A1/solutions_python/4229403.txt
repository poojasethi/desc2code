#include<stdio.h>
#include<stdlib.h>
#include<string.h>
using namespace std;
char s[1000002];
int main(void)
{
	long int i,ans=1,v=10,track[256]={};
	scanf("%s",s);
	if(s[0]=='?')
		ans*=9;//--1 to 9
	else if(s[0]>='A' && s[0]<='J')
	{
		ans*=9;//--1 to 9
		track[s[0]]++;//--track of duplicacy
		v--;
	}
	for(i=1;i<strlen(s);i++)
	{
		if(s[i]=='?')
			ans*=10;
		else if(s[i]>='A' && s[i]<='J' && track[s[i]]==0)
		{
			ans*=v;
			v--;
			track[s[i]]++;
		}
			
	}
	printf("%ld\n",ans);
	return 0;
}
