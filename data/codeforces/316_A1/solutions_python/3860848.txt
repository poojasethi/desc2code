#include <cstdio>
#include <algorithm>
using namespace std;

char s[10];
int c[26];

int main() {
	int tot = 1;

	scanf("%s", s);
	if (s[0] == '?') tot = 9;
	else if ('A' <= s[0] && s[0] <= 'J') ++c[s[0]-'A'];
	for (int i=1; s[i]; ++i) {
		if (s[i] == '?') tot *= 10;
		else if ('A' <= s[i] && s[i] <= 'J') ++c[s[i]-'A'];
	}
	int cc = 10;
	if ('A' <= s[0] && s[0] <= 'J') tot *= 9, cc = 9, c[s[0]-'A'] = 0;
	for (int i=0; i<26; ++i) if (c[i]) {
		tot *= cc;
		--cc;
	}
	printf("%d\n", tot);

	return 0;
}
