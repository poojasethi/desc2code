#include<iostream>
#include<string>
using namespace std;
bool b[10];
int main()
{
	int si=10;
	long long  ans=1;
	string n;
	cin>>n;
	int l=n.length();
	for(int i=0;i<l;i++)
	    if(i!=0){
	    	if(n[i]>='A' && n[i]<='J' && !b[n[i]-'A']) {ans*=si;si--;b[n[i]-'A']=true;}
	    	if(n[i]=='?') ans*=10;
	    }
	    else{
	    	if(n[i]>='A' && n[i]<='J') {ans*=--si;b[n[i]-'A']=true;}
	    	if(n[i]=='?') ans*=9;
	    }
	cout<<ans;
	return 0;
}
