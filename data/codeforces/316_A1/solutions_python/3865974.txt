#include <iostream>
using namespace std;

char ch;
bool f[100];
int t = 10;

int main()
{
	long long cnt = 1;
	bool flag = false;
	int inc = 0;
	while (cin >> ch)
	{
		if (ch == '?') cnt *= (flag) ? 10 : 9;
		else if ('A' <= ch && ch <= 'J')
			{
				if (!f[ch]) 
				{
					if (!flag) cnt *= 9;
					else cnt *= t;
					f[ch] = true;
					--t;
				}
			}
		flag = true;
	}
	cout << cnt;
	cout << endl;
}
