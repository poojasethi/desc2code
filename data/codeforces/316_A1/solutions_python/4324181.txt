#include<cstdio>
#include<iostream>
int main()
{
	unsigned long long int res=1;
	int i,seencount=0;
	bool seen[26]={0};
	char str[100001];
	scanf("%s",str);
	for (i=0;str[i];i++)
	{
		if (str[i]=='?')
			res*=i?10:9;
		else if (str[i]>='0'&&str[i]<='9')
			;
		else if (seen[ str[i]-'A' ]==false)
		{
			seen[ str[i]-'A' ]=true;
			if (i==0)
				res*=9;
			else
				res*=10-seencount;
			seencount++;
		}
	}
	std::cout<<res;
	return 0;
}
