#include<stdio.h>
#include<string.h>
#include<ctype.h>
int main()
{
	char a[100007];
	int used[26]{0};
	scanf("%s",a);
	
	int K=10;
	long long ways=1;
	//first letter: ?|alpha -> 9ways
	if(!isdigit(a[0]))
	{
		ways*=(K-1);
		if(isalpha(a[0]))
		{
			used[a[0]-'A']=1;
			K--;
		}
	}
	
	//a[1]..a[n-1] ?->10 & alpha->K 
	int l=strlen(a);
	for(int i=1;i<l;i++)
	if(a[i]=='?')
	ways*=10;
	else if(isalpha(a[i]))
	{
		if(!used[a[i]-'A'])
		{
			ways*=K;
			used[a[i]-'A']=1;
			K--;
		}
	}
	printf("%I64d\n",ways);
}
