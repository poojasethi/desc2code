#include <iostream>
#include <cmath>
#include <string>
using namespace std;

bool alpha[26];
int wcn=1, alpn=1, alpm=10;
char str[500001];

int main(){
	
	cin>>str;

	for(int i=0; str[i]!='\0'; ++i){
		
		if('0'<=str[i] && str[i]<='9') continue;
		
		if('A'<=str[i] && str[i]<='J' && !alpha[str[i]-'A']){
			alpn*=(alpm-(i==0));
			--alpm;
			alpha[str[i]-'A']=1;
			continue;
		}

		if(str[i]=='?'){
			wcn*=(10-(i==0));
		}

	}

	cout<<alpn*wcn<<endl;
	return 0;
}
