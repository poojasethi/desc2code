#include<cstdio>

char s[111111];
int tot,cnt,base;
int v[1111];
int ans;

int main() {
	scanf("%s", s);
	for(int i=0;s[i];i++){
		if(s[i]=='?')tot++;
		if(s[i]>='A'&&s[i]<='J')v[s[i]]++;
	}
	for(int i='A';i<='J';i++)
		if(v[i])cnt++;
	ans=base=1;
	if(cnt){
		if(s[0]>='A'&&s[0]<='J')base=9;
		else base=10;
		for(int i=9;i>10-cnt;i--)
			base*=i;
	}
	ans*=base;
	if(tot) {
		if(s[0]=='?')ans*=9;
		else ans*=10;
		for(int i=1;i<tot;i++)
			ans*=10;
	}
	printf("%d\n",ans);

	return 0;
}
