#include<iostream>
#include<cstdio>
#include<cctype>

using namespace std;
bool H[10000];
int main()
{
	int i,ol=1,ol2=10;
	char c;
	for(i=0; ; i++)
	{
	   scanf("%c",&c);
		if(c=='\n') break;
		if(isdigit(c))	
			continue;
		if(i==0 and c=='?'){ ol*=9; continue; }
		if(i==0 and !H[c-'0']) {H[c-'0']=true; ol*=(--ol2);  continue; }
		if(c=='?') ol*=10;
		else if(!H[c-'0']) { H[c-'0']=true; ol*=(ol2--); }
		
	}
		cout<<ol;
	return 0;
}
