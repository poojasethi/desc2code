#include<iostream>
#include<ctype.h>
#include<stdio.h>
using namespace std;
int main()
{
int str[150]={0};

char c='\0';
c=getchar();
long long int ans=1;
int i=0;
int j=0;
while(c!='\n')
{
	if(c=='?')
	{
		if(i==1)
			ans*=10;
		else
			ans*=9;
	}
	else if(isalpha(c) && str[c]==0)
	{
			if(i==1)
				ans*=(10-j);
			else
				ans*=9;
			str[c]=1;
			j++;
	}
	

	c=getchar();
	i=1;
}
if(ans>0)
cout<<ans<<endl;
else
cout<<"0"<<endl;
return 0;
}
