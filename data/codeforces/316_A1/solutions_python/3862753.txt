#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
char a[100005];int i,s[12]={0};
long long int p=1,c=0;
scanf("%s",a);
if(a[0]=='\0')
{
printf("0\n");return(0);
}
if(a[0]=='?' || (a[0]>='A' && a[0]<='J'))
{
p=9;
if(a[0]!='?')
{
s[a[0]-'A']=1;c++;
}
}
for(i=1;a[i]!='\0';i++)
{
if(a[i]=='?')p*=10;
else if(a[i]>='A' && a[i]<='J')
{
if(s[a[i]-'A']==0)
{
p*=(10-c);
c++;
s[a[i]-'A']=1;
}
}
}
cout<<p<<endl;
return(0);
}
