#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
char s[101000];
int len,cnt,fg[20],Cnt=10,ans=1;
int main(){
	scanf("%s",s);
	len=strlen(s);
	rep(i,0,len) {
		if (i&&s[i]=='?') cnt++;
		if (s[i]>='A'&&s[i]<='Z') fg[s[i]-'A']=1;
	}
	rep(i,0,10) if (fg[i]) ans*=(Cnt--);
	if (s[0]=='?') ans*=9; else if (s[0]>='A'&&s[0]<='Z') ans=ans/10*9;
	printf("%d",ans);
	rep(i,0,cnt) putchar('0');
}
