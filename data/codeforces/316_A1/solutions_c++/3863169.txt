#include <iostream>
#include <string>
#include <set>

using namespace std;

int main()
{
	string s;
	cin >> s;
	int pro = 1;
	set<char > c;
//	set<char > ss;
	int count = 0;
	for (int i = 0; i < s.length(); i++)
	{
		if (s[i] >= 'A' && s[i] <= 'J' && (!c.count(s[i])))
		{
			if (i == 0)
				pro *= 9-c.size()-count;
			else
				pro *= 10-c.size()-count;
			c.insert(s[i]);
		}
		else if (s[i] == '?')
		{
			if (i == 0)
				pro *= 9;
			else
				pro *= 10;	
		}

	}	
	cout << pro << endl;
}
