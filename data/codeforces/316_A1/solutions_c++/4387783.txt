#include <iostream>

using namespace std;
char s[100005];
int sol,i,j,cnt[11],used[11];
int main()
{
    cin>>s;
    for(i=0,j=10;s[i];i++)
    {
        if(s[i]=='?')cnt[10]++;
        if('A'<=s[i]&&s[i]<='J'&&!used[s[i]-'A'])
        {
            cnt[j--]++;
            used[s[i]-'A']=1;
        }
    }
    if(s[0]<'0'||'9'<s[0]){cnt[10]--;cnt[9]++;}
    sol=1;
    for(i=0;i<10;i++)
        for(;cnt[i];cnt[i]--)sol*=i;
    cout<<sol;
    for(;cnt[10];cnt[10]--)cout<<'0';
    return 0;
}
