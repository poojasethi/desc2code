#include <iostream>
#include <cstdio>
using namespace std;

int f[10],ans=1,show=0;
char c;

int main(){
    c=getchar();
    if (c=='?') ans=9;
    else if (c>='A' && c<='J') ans=9,f[c-'A']=9,show=1;
    while(c=getchar(),c!='\n'){
        if (c=='?') ans*=10;
        else if (c>='0' && c<='9') ans*=1;
        else if (f[c-'A']==0) f[c-'A']=10-show,ans*=f[c-'A'],show++;
        //cout<<ans<<endl;
    }
    cout<<ans<<endl;
    return 0;
}
