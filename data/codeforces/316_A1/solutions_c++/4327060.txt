#include <iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<cstdlib>
using namespace std;

int main()
{
	int a[300],i=1,j=10,k,c;
	memset(a,0,sizeof(a));
	c=getchar();
	if(c<'1'||c>'9')
	{
		i=i*9;
		a[c]++;
		if(c!='?')
		{
			j--;
		}
	}
	while((c=getchar())!='\n')
	{
		if(c=='?')
		{
			i=i*10;
		}
		else if(c>='A'&&c<='J'&&a[c]==0)
		{
			i=i*j;
			a[c]++;
			j--;
		}
	}
	printf("%d\n",i);
	return 0;
}
