#include <iostream>
using namespace std;
int main()
{string s; cin>>s;
int k[10]={0};
int ans=1;
int j=10;
if (s[0]>='A'&&s[0]<='J') {ans=9;k[s[0]-'A']++;j=9;}
if (s[0]=='?') ans=9;
for (int i=1;i<s.size();i++)
 {if (s[i]>='A'&&s[i]<='Z') k[s[i]-'A']++;
  if (s[i]=='?') ans*=10;
 }

for (int i=0;i<10;i++)
 if (k[i]&&s[0]!=i+'A') {ans*=j; j--;}
cout<<ans;
return 0;
}
