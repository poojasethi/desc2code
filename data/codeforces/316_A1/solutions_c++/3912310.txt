# include <iostream>
using namespace std;

	string s;
	int sum=1,q=10;

int main()
{
	cin>>s;
	
	for(int i=0; i<5; i++)
	{
		if(s[i]>='A' && s[i]<='J')
		for(int h=i+1; h<5; h++)
		{
			if(s[h]==s[i])
				s[h]='!';
		}
	}
	
	if((s[0]>='A' && s[0]<='J'))
		{sum=9;q--;}
	else if(s[0]=='?')
		sum=9;
	
	for(int i=1; i<5; i++)
	{
		if((s[i]>='A' && s[i]<='J'))
			{sum*=q,q--;}
		else if(s[i]=='?')
			sum*=10;
	}
	cout<<sum<<"\n";
}
