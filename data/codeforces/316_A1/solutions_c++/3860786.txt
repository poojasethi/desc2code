#include <iostream>
#include <string>
#include <cmath>

using namespace std;

bool mark[300];

int main()
{
	string s;
	cin>>s;
	long long ans = 1;
	long long m = 10;
	for (int i = 0; i<s.length(); i++)
		if (s[i] == '?')
		{
			if (i == 0)
				ans *= 9;
			else
				ans *= 10;
		}
		else if (s[i] >= 'A' && s[i] <= 'J' && !mark[s[i]])
		{
			if (i == 0)
				ans *= 9;
			else
				ans *= m;
			m--;
			mark[s[i]] = true;
		}
	cout<<ans<<endl;
	return 0;
}
