#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

string s;
int h[1000];
int main() {
	cin >> s;
	int c = 0, i;
	int n = s.length();
	long long r = 1;
	for(i = 0; i < n; i++) {
		if(isalpha(s[i])) h[s[i]]++;
		else if(s[i] == '?' and i > 0) r *= 10;
		else if(s[i] == '?' and i == 0) r *= 9;
	}

	for(i = 'A'; i <= 'J'; i++) {
		if(h[i] > 0) c++;
	}

	if(isalpha(s[0])){
		i = 1;
		r *= 9;
	} else {
		i = 0;
	}

	for(;i < c; i++) {
		r *= (10 - i);
	}

	cout << r << endl;
	return 0;
}
