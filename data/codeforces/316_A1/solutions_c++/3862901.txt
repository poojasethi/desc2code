#include<iostream>
#include<string>
#include<ctype.h>
using namespace std;
int main(){
string s;
cin>>s;
int freq[26],co=0,n=1,i;
int y=s.size();
for(i=0;i<y;++i){
if(isalpha(s[i])){
freq[s[i]-'A']++;
} }
for(i=0;i<y;++i){
if(i==0){
if(isdigit(s[i]))n=1;
else if(isalpha(s[i])) {n=(9);freq[s[i]-'A']=0;co=1;}
else n=9;
}
else if(isdigit(s[i]))continue;
else if(s[i]=='?')n*=10;
else{
if(freq[s[i]-'A']){
    n*=(10-co++);
    freq[s[i]-'A']=0;
}
}
}
cout<<n<<"\n";
return 0;
}
