# include <bits/stdc++.h>

using namespace std;

typedef long long int LL;

int main(){
	string s;
	cin >> s;
	int cnt = 0;
	set <char> st;
	for (int i = 0; i < s.size(); i++){
		if(s[i] == '?')
			cnt++;
		if('A' <= s[i] && s[i] <= 'J')
			st.insert(s[i]);
	}
	LL ans = 1;
	while(cnt--){
		ans *= 10;
	}
	for (int i = 0; i < st.size(); i++)
		ans *=  (10 - i);

	if(s[0] == '?' || ('A' <= s[0] && s[0] <= 'J'))
		ans /= 10, ans *= 9;

	cout << ans << endl;

	return 0;
}
