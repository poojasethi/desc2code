#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
int i;
long long k,ans=1,b[20];
char a[100001];
for(i=0;i<20;i++)
b[i]=0;
scanf("%s",a);
k=10;
if(a[0]=='?')
{ans=9;
k=10;
}
if(a[0]>='A'&&a[0]<='J')
{b[a[0]-'A']=1;
ans=9;
k=9;
}
for(i=1;a[i]!='\0';i++)
{
if(a[i]=='?')
ans=ans*10;
else if(a[i]>='A'&&a[i]<='J')
{
if(b[a[i]-'A']==0)
{
b[a[i]-'A']=1;
ans=ans*k;
k--;
}
}
}
cout<<ans<<endl;
return 0;
}
