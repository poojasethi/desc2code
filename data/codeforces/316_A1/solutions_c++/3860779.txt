#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int n=s.length();
    int a[26]={0},count=10,ans=1;
    if(s[0]=='?')ans*=9;
    if(s[0]>='A'&&s[0]<='J'){ans*=9;a[s[0]-'A']=1;count--;}
    for(int i=1;i<n;i++)
    {
        if(s[i]=='?')ans*=10;
        else if(s[i]>='A'&&s[i]<='J'&&a[s[i]-'A']==0)
        {
            ans*=count;
            count--;
            a[s[i]-'A']=1;
        }
    }
    printf("%d",ans);
}
