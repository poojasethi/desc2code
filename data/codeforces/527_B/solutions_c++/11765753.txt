#include<iostream>

using namespace std;

int dp[28][28];
int m[28];
int main(){
	int n;
	cin>>n;
	string s1,s2;
	cin>>s1>>s2;
	int count=0;
	for(int i=0;i<n;i++){ 
		if(s1[i]!=s2[i]){
			dp[s1[i]-'a'][s2[i]-'a']=i+1;
			count++;
			m[s1[i]-'a']=i+1;	
		}
	}
	for(int i=0;i<26;i++){
		for(int j=0;j<26;j++){
			if(dp[i][j]!=0 && dp[j][i]!=0){
				cout<<count-2<<"\n"<<dp[i][j]<<" "<<dp[j][i];
				return 0;
			}
		}
	}
	for(int i=0;i<26;i++){
		for(int j=0;j<26;j++){
			if(dp[i][j]!=0 && m[j]!=0){
				cout<<count-1<<"\n"<<dp[i][j]<<" "<<m[j];
				return 0;
			}
		}
	}
	cout<<count<<"\n"<<"-1 -1";
	return 0;
}
