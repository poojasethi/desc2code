#include <bits/stdc++.h>
using namespace std;
const int N=200005;
    
char s1[N],s2[N];
int a[27],b[27],c[27][27];
int main(){
    int i,j,n,s,x,y,t,ans;
    while(~scanf("%d%s%s",&n,s1,s2)){
        for(x=y=-1,i=s=0;i<n;i++)
            if(s1[i]!=s2[i]) s++,a[s1[i]-'a']=i+1,b[s2[i]-'a']=i+1,c[s1[i]-'a'][s2[i]-'a']=i+1;
        for(i=0;i<26;i++){
            if(a[i]&&b[i]) t=1,x=a[i],y=b[i];
        }
        for(i=0;i<26;i++){
            for(j=0;j<26;j++){
                if(c[i][j]&&c[j][i]) x=c[i][j],y=c[j][i],t=2;
            }
        }
        printf("%d\n%d %d\n",s-t,x,y);
    }
    return 0;
}
