
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,j,k,l;
string s,t;
cin>>n;
cin>>s>>t;
//n=s.size();

int arr[26][26];

for(i=0;i<26;i++)
	for(j=0;j<26;j++)
		arr[i][j]=0;
k=0;
for(i=0;i<n;i++)
{
if(s[i]!=t[i])
	{arr[s[i]-'a'][t[i]-'a']=i+1;
	k++;	
	}
}

for(i=0;i<26;i++)
	for(j=0;j<26;j++)
		{if(arr[i][j]!=0 && arr[j][i]!=0)
			{
			cout<<k-2<<endl<<arr[i][j]<<' '<<arr[j][i]<<endl;
			return 0;
			}
		}

for(i=0;i<26;i++)
	for(j=0;j<26;j++)
		{for(l=0;l<26;l++)
			if(arr[i][j]!=0 && arr[j][l]!=0)
				{cout<<k-1<<endl<<arr[i][j]<<' '<<arr[j][l]<<endl;
				return 0;
				}
		}

cout<<k<<endl<<-1<<' '<<-1<<endl;

}
