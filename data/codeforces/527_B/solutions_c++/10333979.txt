#include <bits/stdc++.h>
#define rep(i,j,k)	for (int i = (j); i <= (k); i++)
#define dep(i,j,k)	for (int i = (j); i >= (k); i--)
using namespace std;
typedef long long ll;
const int N = 200010;
int n, ans;
int a[30][30];
char s1[N], s2[N];
int main()
{
	scanf("%d%s%s", &n, s1+1, s2+1);
	rep(i,1,n)	if (s1[i] != s2[i])	a[s1[i]-'a'][s2[i]-'a'] = i, ans ++;
	rep(i,0,25)	rep(j,0,25)	if (a[i][j] && a[j][i])
	{
		cout << ans-2 << endl;
		cout << a[i][j] << ' ' << a[j][i];
		return 0;
	}
	rep(i,0,25)	rep(j,0,25)	if (a[i][j])
		rep(k,0,25)	if (a[j][k])
		{
			cout << ans-1 << endl;
			cout << a[i][j] << ' ' << a[j][k] << endl;
			return 0;
		}
	cout << ans << endl;
	puts("-1 -1");
}
