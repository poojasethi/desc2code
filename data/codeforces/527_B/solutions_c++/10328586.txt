#include <iostream>
#include <stdio.h>

using namespace std;
int x[256][256],n,i,j,ans,k;
string a,b;
main()
 {
  cin>>n;
  cin>>a>>b;
  for (i=0;i<n;i++)
  {
   if (a[i] != b[i])
    {
     x[a[i]][b[i]]=i+1;
     ans++;
    }
  }
  for (i='a';i<='z';i++)
   for (j=i+1;j<='z';j++)
    if (x[i][j] && x[j][i])
     {
     cout<<ans-2<<endl;
     cout<<x[i][j]<<" "<<x[j][i]<<endl;
     return 0;
    }
  for (i='a';i<='z';i++)
   for (j='a';j<='z';j++)
    for (k='a';k<='z';k++)
    if (x[i][j] && x[j][k])
     {
     cout<<ans-1<<endl;
     cout<<x[i][j]<<" "<<x[j][k]<<endl;
     return 0;
    }
 cout<<ans<<endl;
 cout<<-1<<" "<<-1<<endl;
 }
