#include<cstdio>
int N,Ans,tot;
char s[200010],t[200010];
int P[27][27];
int f[200001];
int main(){
    scanf("%d",&N);
    scanf("%s",s+1);
    scanf("%s",t+1);
    for (int i=1;i<=N;++i)
	    if (s[i]!=t[i]){
			Ans+=(f[i]=1);
			P[s[i]-'a'+1][0]=P[s[i]-'a'+1][t[i]-'a'+1]=i;		
        }
    int x=-1,y=-1;
    int ff=0;
	for (int i=1;i<=N;++i)
        if (f[i]){
			if (P[t[i]-'a'+1][s[i]-'a'+1]){
			    x=i,y=P[t[i]-'a'+1][s[i]-'a'+1];
			    ff=2;
			    break;
			}
			if (P[t[i]-'a'+1][0]){
			    x=i,y=P[t[i]-'a'+1][0];
			    ff=1;
			}
        }
    printf("%d\n%d %d\n",Ans-ff,x,y);
}
