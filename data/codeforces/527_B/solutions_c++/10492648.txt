#include <iostream>
#include <stdio.h>
#include <string>
using namespace std;
int alpha[26][26];
int main(){
int i,j,k,dist=0,len;scanf("%d",&len);
string s,t;
cin>>s>>t;
for(i=0;i<len;i++){
dist+=(s[i]!=t[i]);alpha[s[i]-'a'][t[i]-'a']=i+1;}
for(i=0;i<26;i++)
for(j=i+1;j<26;j++)
if(alpha[i][j]&&alpha[j][i])
{printf("%d\n%d %d",dist-2,alpha[i][j],alpha[j][i]);return 0;}
for(i=0;i<26;i++)
for(j=0;j<26;j++)
for(k=0;k<26;k++)
if(alpha[i][j]&&alpha[j][k]&&i!=j&&j!=k)
{printf("%d\n%d %d",dist-1,alpha[i][j],alpha[j][k]);return 0;}
printf("%d\n-1 -1",dist);
return 0;
}
