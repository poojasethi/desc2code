#include<bits/stdc++.h>
using namespace std;
int a[500][500];
int main()
{
            int n;
            cin>>n;
string s1,s2;
cin>>s1>>s2;
int place1=-1,place2=-1,hm=0,d,i,j,k;
for(int i=0;i<n;i++)
{
if(s1[i]!=s2[i])
{
a[s1[i]][s2[i]]=i+1;
hm++;
}
}
 for(d=hm,i='a';i<='z';++i)
   for(j='a';j<='z';++j)
   {
     for(k='a';k<='z';++k)
     if(a[i][j] && a[j][k]) d=hm-1,place1=a[i][j],place2=a[j][k];

    if(a[i][j] && a[j][i]) d=hm-2,place1=a[i][j],place2=a[j][i],i=j=k='z'+1;
   }




cout<<d<<endl;
cout<<place1<<" "<<place2<<endl;




return 0;
}
