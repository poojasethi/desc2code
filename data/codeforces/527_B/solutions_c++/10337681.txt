#include <iostream>
#include <string>
using namespace std;

long ns[26][26],n,k;

string s,t;
int main()
{
    cin>>n;
    cin>>s;
    cin>>t;

    for(long i=0;i<s.size();i++)
        if(s[i]!=t[i])
       {
         ns[s[i]-97][t[i]-97]=i+1;
         k++;
       }


    for(long i=0;i<26;i++)
     for(long j=0;j<26;j++)
      if(ns[i][j] && ns[j][i])
      {
      cout<<k-2<<endl<<ns[i][j]<<" "<<ns[j][i];
      return 0;
      }

    for(long i=0;i<26;i++)
     for(long j=0;j<26;j++)
      for(long g=0;g<26;g++)
      if(ns[i][j] && ns[g][i])
      {
      cout<<k-1<<endl<<ns[i][j]<<" "<<ns[g][i];
      return 0;
      }

      cout<<k<<endl<<"-1 -1";
    return 0;
}
