#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int A[200005];

int N,Ans,Min;
char S[200005],T[200005];
int xx=-1,yy=-1;

int main()
{
	scanf("%d",&N);
	cin >> S+1;
	cin >> T+1;
	for(int i=1;i<=N;i++)
	{
		if(S[i] != T[i])
		{
			Ans++;
			if(!A[S[i]-96])	A[S[i] - 96] = i;
		}
	}
	Min = Ans;
	for(int i=1;i<=N;i++)
	{
		if(S[i] != T[i])
		{
			int k = Ans;
			if(A[T[i] - 96])
			{
				k--;
				if(T[A[T[i]-96]] == S[i]) k--;
			}
			if(k < Min)
			{
				Min = k;
				xx = i;
				yy = A[T[i]-96];
			}
		}
	}
	printf("%d\n",Min);
	printf("%d %d\n",xx,yy);
}
