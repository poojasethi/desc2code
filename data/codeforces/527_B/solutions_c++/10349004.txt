#include<iostream>

using namespace std;

int a[300][300], ham;

int main()
{
	int n;
	cin >> n;
	string s, t;
	cin >> s >> t;
	for(int i = 0; i < n; i++)
	{
		a[s[i]][t[i]] = i + 1;
		ham += !(s[i] == t[i]);
	}
	for(int i = 'a'; i <= 'z'; i++)
		for(int j = i + 1; j <= 'z'; j++)
			if(a[i][j] && a[j][i])
			{
				cout << ham - 2 << endl;
				cout << a[i][j] << ' ' << a[j][i] << endl;
				return 0;
			}
	for(int i = 0; i < n; i++)
		if(s[i] != t[i])
		{
			for(int j = 'a'; j <= 'z'; j++)
				if(s[i] != j && a[j][s[i]])
				{
					cout << ham - 1 << endl;
					cout << i + 1 << ' ' << a[j][s[i]] << endl;
					return 0;
				}
		}
	cout << ham << endl;
	cout << "-1 -1\n";
	return 0;
}
