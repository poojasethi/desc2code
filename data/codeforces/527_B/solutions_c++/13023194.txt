#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int n,z,sum,ii,jj;
int v[30][30];
char s[200010],t[200010];
int main()
{
	scanf("%d",&n);
	scanf("%s%s",s,t);
	for(int i=0;i<n;i++)
	{
		if(s[i]!=t[i])
		{
			v[s[i]-'a'][t[i]-'a']=i+1;
			sum++;
		}
	}
	for(int i=0;i<26;i++)
	for(int j=0;j<26;j++)
	for(int k=0;k<26;k++)
	{
		if(v[i][j]>0&&v[k][i]>0)
		{
			ii=v[i][j];
			jj=v[k][i];
			z=1;
		}
	}
	for(int i=0;i<26;i++)
	for(int j=0;j<26;j++)
	{
		if(v[i][j]>0&&v[j][i]>0)
		{
			ii=v[i][j];
			jj=v[j][i];
			z=2;
		}
	}
	if(z==0)
	printf("%d\n-1 -1",sum);
	else
	printf("%d\n%d %d",sum-z,ii,jj);
	return 0;
}
