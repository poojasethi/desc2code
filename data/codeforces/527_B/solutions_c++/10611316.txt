#include<iostream>
#include<vector>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<cstdlib>

using namespace std;

int way[300][300];
int invs[300], invt[300];
char s[202000],t[202000];
int main()
{
	int n,i,j,sum=0;
	scanf("%d",&n);
	scanf("%s%s",s,t);
	for(i=0;i<n;++i)
		if(s[i]!=t[i])
		{
			sum++;
			way[s[i]][t[i]]=i+1;
			invs[s[i]]=invt[t[i]]=i+1;
		}
	for(i='a';i<='z';++i)
		for(j='a';j<='z';++j)
			if(way[i][j] && way[j][i]){printf("%d\n%d %d\n",sum-2,way[i][j],way[j][i]);return 0;}
	for(i='a';i<='z';++i)
		if(invs[i] && invt[i]){printf("%d\n%d %d\n",sum-1,invs[i],invt[i]);return 0;}
	
	printf("%d\n-1 -1\n",sum);	
			
			
 	return 0;
}
