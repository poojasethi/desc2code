#include <iostream>
#include <string>
#include <cstdio>
using namespace std;

int main() {
	//freopen("inp.txt", "r", stdin);
	int n, i, j, d = 0, x = -1, y = -1, k = 0;
 	int a[256][256] = {0};
	string s, t;
	cin >> n >> s >> t;
	for (i = 0; i<n; i++) 
		if (s[i]!=t[i]) {
			d++;
			if (a[t[i]][s[i]]) {
				k = 2;
				x = a[t[i]][s[i]];
				y = i+1;
			}
			if (k<2) 
				for (j = 0; j < 256; j++)
					if (a[j][s[i]]) {
						k = 1;
						x = a[j][s[i]];
						y = i+1;
					} else 
					if (a[t[i]][j]) {
						k = 1;
						x = a[t[i]][j];
						y = i+1;
					}
			a[s[i]][t[i]] = i+1;
		}
	d -= k;
	cout << d << endl << x << " " << y;
}
