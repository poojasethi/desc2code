#include <bits/stdc++.h>
using namespace std;

int n, h, a[30], b[30], v[30][30];
char s[200005], t[200005];
int main() {
	int i, j;
//	freopen("input.txt", "r", stdin);
	cin >> n;
	scanf("%s %s", s + 1, t + 1);
	for (i = 1; i <= n; i++) {
		if (s[i] != t[i]) {
			h++;
			a[s[i] - 'a'] = i;
			b[t[i] - 'a'] = i;
			v[s[i] - 'a'][t[i] - 'a'] = i;
		}
	}
	for (i = 0; i < 26; i++) {
		for (j = i + 1; j < 26; j++) {
			if (v[i][j] > 0 && v[j][i] > 0) {
				cout << h - 2 << endl;
				cout << v[i][j] << ' ' << v[j][i];
				return 0;
			}
		}
	}
	for (i = 0; i < 26; i++) {
		if (a[i] > 0 && b[i] > 0) {
			cout << h - 1 << endl;
			cout << a[i] << ' ' << b[i];
			return 0;
		}
	}
	cout << h << endl << "-1 -1";
}
