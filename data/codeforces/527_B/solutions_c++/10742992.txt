#include<iostream>
#include<string>
#include<set>
#include<cmath>
#include<algorithm>
using namespace std;
int n,a[26][26],i,j,k,ans;
char s[500001],t[500001];
main(){
cin>>n;
cin>>s>>t;
for(i=0;i<n;i++){
if(s[i]!=t[i]){
ans++;
a[s[i]-'a'][t[i]-'a']=i+1;}}
for(i=0;i<26;i++)
for(j=0;j<26;j++)
if(a[i][j] && a[j][i]){
cout<<ans-2<<endl;
cout<<a[i][j]<<" "<<a[j][i];
return 0;}
for(i=0;i<26;i++)
for(j=0;j<26;j++)
if(a[i][j]){
for(k=0;k<26;k++)
if(a[j][k]){
cout<<ans-1<<endl;
cout<<a[i][j]<<" "<<a[j][k];
return 0;}}
cout<<ans<<endl;
cout<<"-1 -1";
}
