#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=200050;
int n,ans1=0,ans2,ans3;
char S[maxn],T[maxn];
int pos[26][26];
int main()
{
	int i,j,k;
	bool z=0;
	scanf("%d%s%s",&n,S+1,T+1);
	for(i=1;i<=n;i++)
		ans1+=(S[i]!=T[i]),pos[S[i]-'a'][T[i]-'a']=i;
	for(i=0;i<26&&!z;i++)
			for(j=0;j<26&&!z;j++)
				if(i!=j&&pos[i][j]&&pos[j][i])
					ans1-=2,ans2=pos[i][j],ans3=pos[j][i],z=1;
	if(!z)
		for(i=0;i<26&&!z;i++)
			for(j=0;j<26&&!z;j++)
				if(i!=j&&pos[i][j])
					for(k=0;k<26&&!z;k++)
						if(k!=j&&pos[j][k])
							ans1--,ans2=pos[i][j],ans3=pos[j][k],z=1;
	if(!z) ans2=ans3=-1;
	printf("%d\n%d %d\n",ans1,ans2,ans3);
	return 0;
}
