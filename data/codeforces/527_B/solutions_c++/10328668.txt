#include <iostream>
#include <string>

using namespace std;

int main()
{
	int n, i, j, dist = 0, t[26][26] = {{0, }, };
	string S, T;

	cin>>n>>S>>T;

	for (i = 0; i < n; i++)
	{
		if (S[i] != T[i])
		{
			t[S[i]-'a'][T[i]-'a'] = i + 1;
			dist++;
		}
	}

	for (i = 0; i < n; i++)
	{
		if (S[i] != T[i] && t[T[i]-'a'][S[i]-'a'])
		{
			cout<<dist-2<<endl<<i+1<<" "<<t[T[i]-'a'][S[i]-'a']<<endl;
			return 0;
		}
	}

	for (i = 0; i < n; i++)
	{
		if (S[i] != T[i])
		{
			for (j = 0; j < 26; j++)
			{
				if (t[j][S[i]-'a'])
				{
					cout<<dist-1<<endl<<i+1<<" "<<t[j][S[i]-'a']<<endl;
					return 0;
				}
			}
		}
	}

	cout<<dist<<endl<<"-1 -1"<<endl;

	return 0;
}
