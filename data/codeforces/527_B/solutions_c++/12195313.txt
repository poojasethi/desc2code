#include <cstdio>

using namespace std;

int main (){
	
int n;
char s[200010],t[200010];
int a[30][30];
int poz[30];
int x,y,temp,udaljenost;

x=y=-1;
temp=udaljenost=0;

scanf("%d",&n);
scanf("%s%s",&s,&t);		
	
for(int i=0;i<n;i++){
		
	if(s[i]==t[i]) continue;	
	
	a[s[i]-97][t[i]-97]=i+1;
	
	poz[s[i]-97]=i+1;
	
	udaljenost+=1;	
}		
		
for(int i=0;i<26;i++){
for(int j=0;j<26;j++){

	if(a[i][j]!=0 && a[j][i]!=0){
		temp=2;
		x=a[i][j];
		y=a[j][i];
	}	
	
	if(temp!=2 && a[i][j]!=0 && poz[j]!=0){
		temp=1;
		x=a[i][j];
		y=poz[j];
	}

}
}

udaljenost-=temp;

printf("%d\n",udaljenost);
printf("%d %d",x,y);

return 0;
}
