#include <iostream>
#include <cmath>
#include <algorithm>

using namespace std;

typedef long long ll;

string s, t;

int c[26][26];
int b[26];

int main()
{
	int n;
	cin >> n;
	cin >> s >> t;
	

	int x = 0;
	int ind1 = -1, ind2 = -1;
	int ans = 0;

	for (int i=0; i<n; i++)
	{
		if (s[i] != t[i])
		{
			c[s[i]-'a'][t[i]-'a'] = i + 1;
			b[t[i]-'a'] = i + 1;
			ans++;
		}
	}

	for (int i=0; i<n; i++)
	{
		if (s[i] != t[i])
		{
			if (c[t[i]-'a'][s[i]-'a'] != 0)
			{
				x = -2;
				ind1 = c[t[i]-'a'][s[i]-'a'];
				ind2 = i+1;
			}
			if (x != -2 && b[s[i]-'a'] != 0)
			{
				x = -1;
				ind1 = b[s[i]-'a'];
				ind2 = i+1;	
			}
		}
	}
	cout << ans + x << "\n" << ind1 << " " << ind2 << "\n";
}
