#include <bits/stdc++.h>
using namespace std;

int n, h, x = -1, y = -1, z, v[30][30];
string s, t;
int main() {
	int i, j, k;
//	freopen("input.txt", "r", stdin);
	cin >> n >> s >> t;
	for (i = 0; i < n; i++) {
		if (s[i] != t[i]) {
			h++;
			v[s[i] - 'a'][t[i] - 'a'] = i + 1;
		}
	}
	for (i = 0; i < 26; i++) {
		for (j = 0; j < 26 && !v[i][j]; j++);
		for (k = 0; k < 26 && !v[k][i]; k++);
		if (j < 26 && k < 26) {
			z = 1;
			x = v[i][j];
			y = v[k][i];
		}
	}
	for (i = 0; i < 26; i++) {
		for (j = i + 1; j < 26; j++) {
			if (v[i][j] > 0 && v[j][i] > 0) {
				z = 2;
				x = v[i][j];
				y = v[j][i];
			}
		}
	}
	printf("%d\n%d %d", h - z, x, y);
}
