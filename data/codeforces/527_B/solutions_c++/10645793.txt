#include<iostream>
#include<vector>
#include<stdio.h>
using namespace std;
int a[100001];
int v[260][260];
int main()
{
	int n;
	cin>>n;
	char s[1000000],t[1000000];
	int h=0;
	scanf("%s %s",(s+1),(t+1));
	for(int i=1;s[i]!='\0';i++)
	{
		if(s[i]!=t[i])
		{
			v[s[i]][t[i]]=i;
			h++;
		}
	}

	for(int i=1;s[i]!='\0';i++)
	{
		if(s[i]!=t[i])
		{
			if(v[t[i]][s[i]]!=0)
			{
				cout<<h-2<<endl;
				cout<<v[s[i]][t[i]]<<" "<<v[t[i]][s[i]];
				return 0;
			}
		}
	}
	for(int i=1;s[i]!='\0';i++)
	{
		if(s[i]!=t[i])
		{
			for(int j=1;j<260;j++)
				if(v[t[i]][j]!=0)
				{
					cout<<h-1<<endl;
					cout<<v[s[i]][t[i]]<<" "<<v[t[i]][j];
					return 0;
				}
		}
	}
	cout<<h<<endl<<-1<<" "<<-1;
	return 0;
}
