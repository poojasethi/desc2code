#include<bits/stdc++.h>
using namespace std;
int main()
{
int n;
scanf("%d",&n);
char s[200001],t[200001];
cin>>s>>t;
int a[26][26],i,j,k,ham=0;
for(i=0;i<n;i++)
{
if(s[i]!=t[i])
{
ham++;
a[s[i]-'a'][t[i]-'a']=i+1;
}
}
for(i=0;i<26;i++)
for(j=0;j<26;j++)
if(a[i][j]&&a[j][i])
{
cout<<ham-2<<endl;
cout<<a[i][j]<<" "<<a[j][i];
return 0;
}
for(i=0;i<26;i++)
for(j=0;j<26;j++)
if(a[i][j])
{
for(k=0;k<26;k++)
if(a[j][k])
{
cout<<ham-1<<"\n";
cout<<a[i][j]<<" "<<a[j][k];
return 0;
}
}
cout<<ham<<"\n";
cout<<"-1 -1";
return 0;
}
