n = input()
s,t = raw_input(),raw_input()
c = sum(s[i]!=t[i] for i in xrange(n))
need = {};
for i in xrange(n):
    if s[i] != t[i]:
        if t[i] not in need: need[t[i]] = {}
        need[t[i]][s[i]] = i;
        if s[i] in need and t[i] in need[s[i]]:
            print c-2,'\n',need[s[i]][t[i]]+1,i+1; exit(0)
for i in xrange(n):
    if s[i] != t[i]:
        if s[i] in need:
            print c-1,'\n',need[s[i]].values()[0]+1,i+1; exit(0)
print c,'\n',-1,-1
