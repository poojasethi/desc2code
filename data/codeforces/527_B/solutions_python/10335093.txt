from string import lowercase

n = input()
s = raw_input()
t = raw_input()

d = 0
ts = {c: dict() for c in lowercase}
for i in xrange(n):
    if s[i] == t[i]: continue
    ts[t[i]][s[i]] = i
    d += 1

inc = 0
u, v = -1, -1
for i in xrange(n):
    if s[i] == t[i]: continue
    if t[i] in ts[s[i]]:
        inc = -2
        u, v = i + 1, ts[s[i]][t[i]] + 1
        break
    if ts[s[i]]:
        inc = -1
        u, v = i + 1, next(ts[s[i]].itervalues()) + 1

print d + inc
print u, v
