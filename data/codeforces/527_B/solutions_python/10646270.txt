#!/usr/bin/python
# -*- coding:utf-8 -*-

n = int(raw_input())
S = raw_input()
T = raw_input()
dic = {}
rlt = [None,None]
cnt = 0
for i in range(n):
    if S[i] != T[i]:
        dic[S[i]+T[i]] = i
        dic[S[i]] = i
        cnt += 1
for i in range(n):
    if S[i] != T[i]:
        k1 = S[i]+T[i]
        k2 = T[i]+S[i]
        if k2 in dic:
            rlt[1] = tuple([dic[k2],i])
        if T[i] in dic:
            rlt[0] = tuple([dic[T[i]],i])
if rlt[1]:
    print cnt - 2
    rlt[1] = (min(rlt[1][0],rlt[1][1]),max(rlt[1][0],rlt[1][1]))
    print '%s %s' % (rlt[1][0]+1,rlt[1][1]+1)
elif rlt[0]:
    print cnt - 1
    rlt[0] = (min(rlt[0][0],rlt[0][1]),max(rlt[0][0],rlt[0][1]))
    print '%s %s' % (rlt[0][0]+1, rlt[0][1]+1)
else:
    print cnt
    print '-1 -1'
