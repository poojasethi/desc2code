using namespace std;

double ans,rec;
int a[10],b[10],c[10],d[10],n,m,A;

void solve(int k,int sum,double pro,int ac)
{
	if (k>n)
	{
		if (ac>n/2) rec+=pro; else rec+=pro*A/(A+sum);
		return;
	}
	solve(k+1,sum,pro*d[k]/100,ac+1);
	solve(k+1,sum+a[k],pro*(100-d[k])/100,ac);
}

void check()
{
	rec=0;
	for(int i=1;i<=n;i++)
		d[i]=min(c[i]*10+b[i],100);
	solve(1,0,1,0);
	if (rec>ans) ans=rec;
}

void dfs(int k,int rem)
{
	if (k>n)
	{
		if (rem==0) check();
		return;
	}
	
	for(int i=0;i<=rem;i++)
	{
		c[k]=i;
		dfs(k+1,rem-i);
	}
}

int main()
{
	//freopen("b.in","r",stdin);
	
	scanf("%d%d%d",&n,&m,&A);
	for(int i=1;i<=n;i++)
		scanf("%d%d",&a[i],&b[i]);
	dfs(1,m);
	printf("%.9lf\n",ans);
	
	return 0;
}
