#include <iostream>
#include <iomanip>
#include <algorithm>
using namespace std;
double ans=0;
int p[10];
int r[10];
int n,m,A;
void find(int a,int b)
{
	if (a==n || b==0)
	{
		double res=0;
		for (int i=0;i<(1<<n);i++)
		{
			double tmp=1;
			int sum=0,cnt=0;
			for (int j=0;j<n;j++) 
				if ((i>>j)%2==1)
				{
					cnt++;
					tmp*=double(p[j])/100;
				}
				else
				{
					sum+=r[j];
					tmp*=double(100-p[j])/100;
				}
			if (cnt*2>n)
				res+=tmp;
			else
			{
				tmp*=double(A)/double(A+sum);
				res+=tmp;
			}
		}
		ans=max(ans,res);
		return ;
	}
	int tmp=p[a];
	for (int i=0;i<=b && tmp+10*i<=100;i++)
	{
		p[a]=tmp+10*i;
		find(a+1,b-i);
	}
	p[a]=tmp;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin>>n>>m>>A;
	for (int i=0;i<n;i++)
		cin>>r[i]>>p[i];
	find(0,m);
	cout<<fixed<<setprecision(100)<<ans<<endl;
	return 0;
}
