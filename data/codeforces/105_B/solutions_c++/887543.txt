#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int b[10], l[10];
int n, k, A;

double go(int x, int y) {
	if(x==n) {
		double r = 0.0;
		for(int i=0;i<(1<<n);i++) {
			double v = 1.0;
			int B = 0, C = 0;
			for(int j=0;j<n;j++)
				if(i&(1<<j)) v = v * l[j] / 100., ++C;
				else v = v * (100-l[j]) / 100., B += b[j];
			if (C > n/2) r += v;
			else r += v * A / (A+B);
		}
		return r;
	}
	int last = l[x];
	double ret = 0.0;
	for(int i=0;i+y<=k && l[x]<=100;i++) {
		ret = max(ret, go(x+1, y+i));
		l[x] += 10;
	}
	l[x] = last;
	return ret;
}

int main(void) {
	scanf("%d%d%d", &n, &k, &A);
	for(int i=0;i<n;i++)
		scanf("%d%d", &b[i], &l[i]);
	double ans = go(0, 0);
	printf("%.9lf\n", ans);
	return 0;
}
