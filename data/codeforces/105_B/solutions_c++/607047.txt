
#include<iomanip>
#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
#define pop_count(n) __builtin_popcount(n)
typedef long double LD;
int n,k,A;
int lev[10],loy[10];
LD ans=0;
void split(int at,int left)
{
   if(at==n)
   {
      //try all 2**n cases;
      LD tmp,cur=0;
      int op;
      for(int i=0;i<(1<<n);i++)
      {
	 tmp=1;
	 op=0;
	 for(int j=0;j<n;j++)
	 {
	    if(i&(1<<j))// positive vote
	       tmp= tmp * (LD(loy[j])/100);
	    else
	    {
	       tmp= tmp * (LD(100-loy[j])/100);
	       op+=lev[j];
	    }
	 }
	 if(pop_count(i)<=(n/2))
	    cur+=tmp*(A/LD(A+op));
	 else
	    cur+=tmp;
      }
      if(cur>ans)
	 ans=cur;
   }
   else
   {
      for(int i=0;i<=left;i++)
      {
	 if(loy[at]+i*10>100)
	    break;
	 loy[at]+=i*10;
	 split(at+1,left-i);
	 loy[at]-=i*10;
      }
   }
}
int main()
{
   scanf("%d%d%d",&n,&k,&A);
   int i;
   for(i=0;i<n;i++)
      scanf("%d%d",&lev[i],&loy[i]);
   split(0,k);
   cout<<fixed<<setprecision(10);
   cout<<ans<<endl;
   return 0;
}
