#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 10;
int n, k, a;
int x[N], y[N];
double check(int idx, int deep, int need) {
	if(deep == 0) return 1;
	if(idx == n) return (double)a / (double)(a + need);
	
	double ans = check(idx+1, deep, need+x[idx]) * (1-y[idx]/10.0);
	if(deep) ans += check(idx+1, deep-1, need) * (y[idx]/10.0);
	
	return ans;
	
}
double dfs(int idx, int d) {
	if(idx == n) {
		return check(0, n/2+1, 0);
	}
	double ans = 0;
	for(int i = 0; i <= d; i ++) {
		if(y[idx] + i > 10) break;
		y[idx] += i;
		ans = max(ans, dfs(idx+1, d-i));
		y[idx] -= i;
	}
	return ans;
}
int main() {
	scanf("%d%d%d", &n, &k, &a);
	for(int i = 0; i < n; i ++) {
		scanf("%d%d", &x[i], &y[i]);
		y[i] /= 10;
	}
	printf("%.10f\n", dfs(0, k));
	
	return 0;
}
