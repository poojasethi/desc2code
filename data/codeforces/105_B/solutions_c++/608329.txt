#include <cstdio>
#include <algorithm>
#define MAXN 16
using namespace std;
int n, k, A;
int l[MAXN], p[MAXN];
double ans;
void gao(int idx, int rem) {
	if(idx == n) {
		double tp = 0;
		for(int m = 0; m < (1 << n); ++m) {
			int cnt = 0, sum = 0;
			double term = 1;
			for(int i = 0; i < n; ++i)
				if(m & (1 << i)) {
					term *= 1.0 * p[i] / 100;
					++cnt;
				}
				else {
					sum += l[i];
					term *= 1.0 * (100 - p[i]) / 100;
				}
			if(2 * cnt <= n)
				term *= 1.0 * A / (A + sum);
			tp += term;
		}
		//for(int i = 0; i < n; ++i)
		//	printf("%d%c", p[i], (i == n - 1 ? '\n' : ' '));
		//printf("%.8lf %d\n", tp, rem);
		if(tp > ans)
			ans = tp;
		return;
	}
	for(int i = 0; i <= rem && p[idx] + i * 10 <= 100; ++i) {
		p[idx] += i * 10;
		gao(idx + 1, rem - i);
		p[idx] -= i * 10;
	}
}
int main() {
	while(scanf("%d%d%d", &n, &k, &A) != EOF) {
		for(int i = 0; i < n; ++i)
		       scanf("%d%d", &l[i], &p[i]);
		ans = 0;
		gao(0, k);
		printf("%.8lf\n", ans);
	}	
	return 0;
}
