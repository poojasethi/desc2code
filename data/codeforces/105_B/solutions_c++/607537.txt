#include <cstdio>
#include <algorithm>

using namespace std;

const int MAXN = 10;

int n, candies, skill;
int level[MAXN], prob[MAXN], realProb[MAXN];
double sol;

void back(int pos, int candies) {
	if (pos == n) {
		double res = 0;
		for (int i = 0; i < 1<<n; ++i) {
			int votes = 0, sum = 0;
			double current_prob = 1;
			for (int j = 0; j < n; ++j)
				if (i&(1<<j)) {
					++votes;
					current_prob *= (double) realProb[j]/100;
				} else {
					sum += level[j];
					current_prob *= (double) (100-realProb[j])/100;
				}

			if (votes > n/2) {
				res += current_prob;
			} else {
				res += current_prob*skill/(skill+sum);
			}

			sol = max(sol, res);
		}
	} else {
		for (int i = 0; i <= min(candies, (100-prob[pos])/10); ++i) {
			realProb[pos] = prob[pos] + i*10;
			back(pos+1, candies-i);
		}
	}
}

int main() {
//	freopen("b.in", "r", stdin);

	scanf("%d %d %d", &n, &candies, &skill);
	for (int i = 0; i < n; ++i) {
		scanf("%d %d", &level[i], &prob[i]);
	}

	back(0, candies);
	printf("%.10lf\n", sol);

	return 0;
}
