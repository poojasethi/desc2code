#include <iostream>
#include <iomanip>
using namespace std;

const int MAXN = 8;
int n, k, a;

int b[MAXN];
int l[MAXN];
double maxP;

void bt(int c, int lc)
{
	if(c == 0)
	{
		double sum = 0;
		for(int i = 0; i < (1 << n); i++)
		{
			int t = 0, sl = 0;
			double p = 1;
			for(int j = 0; j < n; j++)
			{
				if(((i >> j) & 1) == 1)
				{
					p *= double(l[j]) / 100;
					t++;
				}
				else
				{
					p *= double(100 - l[j]) / 100;
					sl += b[j];
				}
			}
			if(t > n / 2)
			{
				sum += p;
			}
			else
			{
				sum += p * (double(a) / (a + sl));
			}
		}
		if(sum > maxP)
		{
			maxP = sum;
		}
		return;
	}
	for(int i = lc; i < n; i++)
	{
		if(l[i] < 100)
		{
			int ll = l[i];
			l[i] = (l[i] + 10 > 100) ? 100 : (l[i] + 10);
			bt(c - 1, i);
			l[i] = ll;
		}
	}
	bt(c - 1, n);
}

int main()
{
	cin >> n >> k >> a;
	for(int i = 0; i < n; i++)
	{
		cin >> b[i] >> l[i];
	}
	bt(k, 0);
	cout << fixed << setprecision(10) << maxP << endl;
	return 0;
}
