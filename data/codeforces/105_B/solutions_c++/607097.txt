#include<cstdio>
#include<cstring>
#include<string>
#include<map>
#include<cmath>
#include<algorithm>
#include<iostream>
#define MAXN 10
using namespace std;
int n,m,a,p[MAXN],l[MAXN];
double pp[MAXN];
double ans;
bool flag[MAXN];
double anss;
void dfs2(int pos,int cnt,int b,double now)
{
	if(pos==n){
		if(cnt*2>n)
			anss+=now;
		else
			anss+=now*(a)/(a+b);
	}
	else{
		dfs2(pos+1,cnt+1,b,now*pp[pos]);
		dfs2(pos+1,cnt,b+l[pos],now*(1.0-pp[pos]));
	}
}
void dfs(int pos,int lv)
{
	if(pos==n){
		memset(flag,false,sizeof(flag));
		anss=0;
		dfs2(0,0,0,1.0);
		ans=max(ans,anss);
	}
	else{
		int lim=(100-p[pos])/10;
		for(int i=0;i<=lim&&i<=lv;i++){
			pp[pos]=(p[pos]+i*10.0)/100;
			if(pp[pos]>1.0)
				pp[pos]=1.0;
			dfs(pos+1,lv-i);
		}
	}
}
void solve()
{
	ans=0;
	dfs(0,m);
	if(ans>1)
		ans=1;
	cout<<ans<<endl;
}
int main()
{
	//freopen("input.txt","r",stdin);
	while(cin>>n>>m>>a){
		for(int i=0;i<n;i++)
			cin>>l[i]>>p[i];
		solve();
	}
	return 0;
}
