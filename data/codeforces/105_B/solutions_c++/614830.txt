#include <algorithm>
#include <iostream>
#include <vector>
#include <cstdio>

using namespace std;

int a, n;
vector<int> lev;
vector<int> loy;
vector<int> done;
double ans = 0.0;

double check() {
double cur = 0.0;
for(int mask = 0;mask < (1 << n);mask++) {
	double ver = 1.0;
	int za = 0;
	int b = 0;
	for(int i = 0;i < n;i++) {
		int v = min(loy[i] + done[i], 10);
		if((mask & (1 << i)) != 0) {
			za++;
			ver *= (double)v / 10.0;
		} else {
			b += lev[i];
			ver *= (double)(10 - v) / 10.0;
		}
	}
	if(za <= n - za) ver *= (double)a / (double)(a + b);
	cur += ver;
}
return cur;
}

void rec(int sen, int k) {
if(sen == n) {
	ans = max(ans, check());
} else {
	for(int i = 0;i <= k;i++) {
		done[sen] = i;
		rec(sen + 1, k - i);
	}
}
}

int main() {
int k;
cin >> n >> k >> a;
lev.resize(n);
loy.resize(n);
done.resize(n);
for(int i = 0;i < n;i++) {
	cin >> lev[i] >> loy[i];
	loy[i] /= 10;
}
rec(0, k);
printf("%.6lf\n", ans);
}
