#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;
int n,a;
int l[10],b[10];
double ans=0,temp;

void cal(int k,int s,double res,int sum)
{
   // if (l[1]==100 && l[2]==100 && l[3]==100) cout<<k<<' '<<s<<' '<<res<<' '<<sum<<endl;
    if (k>n)
    {
        if (s>n/2) temp+=res; else temp+=res*a/(a+sum);
        return;
    }
    cal(k+1,s+1,res*l[k]/100,sum);
    cal(k+1,s,res*(1-double(l[k])/100),sum+b[k]);
}

void dfs(int k,int m)
{
    if (k>n)
    {
        temp=0;
        cal(1,0,1,0);
        if (temp>ans) ans=temp;
        return;
    }
    for (int i=0; i<=m; i++)
    {
        int t=l[k];
        l[k]+=i*10; if (l[k]>100) l[k]=100;
        dfs(k+1,m-i);
        l[k]=t;
    }
}

int main()
{
    int i,m;
    cin>>n>>m>>a;
    for (i=1; i<=n; i++) cin>>b[i]>>l[i];
    dfs(1,m);
    printf("%.10lf\n",ans);
}
