using namespace std;

typedef long long LL;
int N,K,A;
int B[8],L[8];

double go(int l[8],int i,int k){
  double res=0;

  if(i==N){
    for(int m=0;m<(1<<N);m++){
      int t=0;
      int v=0;
      double p=1;
      for(int j=0;j<N;j++){
        if( m&(1<<j) ){
          p*=(double)l[j]/100;
          v+=1;
        }else{
          p*=(double)(100-l[j])/100;
          t+=B[j];
        }
      }

      res+=(v>N/2)?p:p*((double)A/(A+t));
    }
    return res;
  }
 
  for(int j=0;k+j<=K;j++){
    l[i]=L[i]+j*10;
    if(l[i]>100) break;
    res=max(res,go(l,i+1,k+j));
  }

  return res;
}

int main(){

  cin>>N>>K>>A;

  for(int i=0;i<N;i++){
    cin>>B[i]>>L[i];
  }

  int l[8];

  cout<<fixed<<setprecision(10)<<go(l,0,0)<<endl;

  return 0;
}
