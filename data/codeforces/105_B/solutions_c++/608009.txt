#include <cstdio>
int v[10],m[10];
int n,k,A,maxn,l=0;
double res;
void dfs(int step,int k)
{
    if(step==n){
        double ans=0;
        int have[10];
        for(int i=0;i<maxn;i++){
            int cnt=0,p=i,tol=0;
            double g=1;
            for(int j=0;j<n;j++){
                if(p&(1<<j)) { cnt++ ; g*=m[j]/100.0;}
                else {
                    g*=(1-m[j]/100.0);
                    tol+=v[j];
                }
            }
            if(cnt>n/2) {
                ans+=g;
            }
            else {
                ans+=g*A/(A+tol);
            }
        }
        if(res<ans) res=ans;
        return ;
    }
    for(int i=0;i<=k&&i<=(100-m[step])/10;i++){
        m[step]+=10*i;
        dfs(step+1,k-i);
        m[step]-=10*i;
    }
}


int main()
{
    scanf("%d%d%d",&n,&k,&A);
    for(int i=0;i<n;i++) {
        scanf("%d%d",&v[i],&m[i]);
    }
    res=0;
    maxn=(1<<n);
    dfs(0,k);
    printf("%.10f\n",res);
}
