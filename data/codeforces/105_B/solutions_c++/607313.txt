#include <algorithm>
#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
using namespace std;

int n,k,a;
int f[8][2],g[8];
double ans,tmp,t;

void find()
{
	int x,y,k,num,s;
	tmp=0; x=1<<n;
	for (int i=0; i<x; i++) {
		t=1; num=0; s=0;
		for (int j=0; j<n; j++) {
			k=1<<j;
			if ((k&i)==0) {
				s+=f[j][0];
				t=(t*(10-f[j][1]))/10;
			} else {
				++num;
				t=(t*f[j][1])/10;
			}
		}
		if (num*2>n) tmp+=t; else tmp+=(t*a)/(a+s);
	}
	ans=max(ans,tmp);
}

void work(int x,int y)
{
	int z;
	if (x==n) find(); else {
		z=min(y,10-f[x][1]);
		for (int i=0; i<=z; i++) {
			f[x][1]+=i;
			work(x+1,y-i);
			f[x][1]-=i;
		}
	}
}

int main()
{
	cin>>n>>k>>a;
	for (int i=0; i<n; i++) {
		cin>>f[i][0]>>f[i][1];
		f[i][1]/=10;
	}
	ans=0;
	work(0,k);
	cout<<ans<<endl;
	return 0;
}
