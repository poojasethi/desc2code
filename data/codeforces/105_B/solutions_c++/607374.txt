//#include "stdafx.h"
#include <iostream>
#include <algorithm>

using namespace std;

int W[8];
int L[8];
int K[8];
int n,a,wsum=0;
long double res=0.0;

void solve(int m,int k) {
	if (m==0) {
		long double prob=0;
		for (int s=(1<<n)-1;s>=0;--s) {
			long double p=1.0;
			int sum=0,r=0;
			for (int t=s,i=0;i<n;i++,t>>=1) {
				if (t&1)
					p*=(L[i]+K[i])/10.0,r++;
				else
					p*=(10-L[i]-K[i])/10.0,sum+=W[i];
			}
			if (2*r>n)
				prob+=p;
			else
				prob+=p*(long double)a/(a+sum);
		}
		if (res<prob)
			res=prob;
	} else {
		--m;
		int mk=min(k,10-L[m]);
		for (int i=0;i<=mk;i++) {
			K[m]=i;
			solve(m,k-i);
		}
	}
}

int main(void)
{
	int k;
	cin>>n>>k>>a;
	for (int i=0;i<n;i++) {
		cin>>W[i]>>L[i];
		L[i]/=10;
		wsum+=W[i];
	}
	solve(n,k);
	cout.precision(12);
	cout<<fixed<<res;
	return 0;
}
