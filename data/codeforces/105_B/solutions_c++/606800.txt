#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

int n;
int b[100],l[100];
double A,ans,sum;

void ssearch(int depth,double current,double beat,int votes)
{
  if(depth>n){if(votes*2<=n)sum+=current*A/(A+beat);else sum+=current;return;}
  double tmp=double(l[depth])/100;
  ssearch(depth+1,current*tmp,beat,votes+1);
  ssearch(depth+1,current*(1-tmp),beat+b[depth],votes);
}

double pos()
{
  sum=0;
  ssearch(1,1,0,0);
  return sum;
}

void dfs(int depth,int remain)
{
  if(depth>n){double tmp=pos();if(tmp>ans)ans=tmp;return;}
  int temp=l[depth];
  dfs(depth+1,remain);
  while(remain&&l[depth]<100)
    {
	  remain--;
	  l[depth]+=10;
	  dfs(depth+1,remain);
	}
  l[depth]=temp;
}

int main()
{
  int k,i;
  scanf("%d%d%lf",&n,&k,&A);
  for(i=1;i<=n;i++)scanf("%d%d",&b[i],&l[i]);
  ans=0;
  dfs(1,k);
  printf("%0.10lf\n",ans);
  return 0;
}
