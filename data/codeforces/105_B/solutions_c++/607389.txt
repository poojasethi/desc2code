#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <cmath>
#include <cctype>
#include <iostream>
#include <string>

using namespace std;

#define maxn (10)

int n,m;
double a[maxn];
double b[maxn];
double A;
double ans;

inline double calc(int step, int sum, double B, double x) {
	if (step>=n) {
		return sum>n/2 ? x : (A/(A+B))*x;
	}

	return calc(step+1,sum+1,B,x*b[step])+calc(step+1,sum,B+a[step],x*(1.-b[step]));
}

inline void dfs(int step,int rest) {
	if (step>=n) {
/*
		puts("");
		for (int i=0;i<n;i++) printf("%lf ",b[i]);
		puts("");
		printf("%lf\n",calc(0,0,0.,1.));
*/
		ans=max(calc(0,0,0.,1.),ans); return ;
	}

	dfs(step+1,rest);

	double tmp=b[step];
	for (int i=0;i<=rest;i++) {
		dfs(step+1,rest-i);
		b[step]+=0.1; if (b[step]>1.) b[step]=1.;
	}

	b[step]=tmp;
}

int main(){

	cin >> n >> m >> A;
	for (int i=0;i<n;i++) cin >> a[i] >> b[i];
	for (int i=0;i<n;i++) b[i]*=0.01;

	ans=0.;
	dfs(0,m);

	printf("%.10f",ans);

	return 0;
}
