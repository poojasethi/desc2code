#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <map>
#include <set>
#include <algorithm>

using namespace std;

int n,k,A,p[10],l[10],a[10];
double ans;

double pro(int x, int B, int ac){
    if (x>n)    return ac+ac>n?1:(double)A/(A+B);
    double tmp=pro(x+1, B, ac+1)*(l[x]+a[x]*10)/100;
    tmp+=pro(x+1, B+p[x], ac)*(100-l[x]-a[x]*10)/100;
    return tmp;
}

void calc(int x, int k){
    if (x>n){
        double tmp=pro(1, 0, 0);
        ans=max(ans, tmp);
    }else{
        for (int i=0; i<=min(k, (100-l[x])/10); i++){
            a[x]=i;
            calc(x+1, k-i);
        }
    }
}

int main(){
	//freopen("a.in","r",stdin);
    cin>>n>>k>>A;
    for (int i=1; i<=n; i++) cin>>p[i]>>l[i];
    calc(1, k);
    cout.setf(ios::fixed); cout.precision(10);
    cout<<ans<<endl;
}
