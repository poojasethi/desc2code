#!/usr/bin/python
import sys

(n, m) = map(lambda(x):int(x), sys.stdin.readline().split())
switched = False
if n > m:
	(n, m) = (m, n)
	switched = True

a = []
if n == 1:
	if m == 1:
		print 1
		sys.exit(0)
	elif m <= 3:
		print -1
		sys.exit(0)
	else:
		a.append([])
		for i in range(1, m//2+1):
			a[0].append(2 * i)
		for i in range(1, m-m//2+1):
			a[0].append(2 * i - 1)
elif n == 2:
	if m <= 2:
		print -1
		sys.exit(0)
	else:
		a.append([])
		a.append([])
		for i in range(1, m+1):
			a[0].append(2 * i - 1)
		for i in range(1, m+1):
			a[1].append(2 * ((i - 3) % m + 1))
else:
	for i in range(1, n+1):
		a.append([])
		for j in range(1, m+1):
			a[i-1].append(i + ((j-i)%m) * n)

if switched:
	for i in range(m):
		for j in range(n):
			print a[j][i],
		print
else:
	for i in range(n):
		for j in range(m):
			print a[i][j],
		print
