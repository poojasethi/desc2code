#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <set>
#include <algorithm>
#include <cmath>

using namespace std;

int n,m;
int a[105][105];

int main(){
    scanf("%d %d",&n,&m);

    if (n+m==3 or n+m==4)
        printf("-1\n");
    else{
        int num=2;
        if (n>=m){
            for (int i=1; i<=n; i++)
                for (int j=1; j<=m; j++){
                    if (num>m*n)
                        num=1;
                    a[i][j]=num;
                    num+=2;
                }
        }
        else
            for (int j=1; j<=m; j++)
                for (int i=1; i<=n; i++){
                    if (num>m*n)
                        num=1;
                    a[i][j]=num;
                    num+=2;
                }
        for (int i=1; i<=n; i++)
            for (int j=1; j<=m; j++){
                printf("%d",a[i][j]);
                if (j==m)
                    printf("\n");
                else
                    printf(" ");
            }
    }
    return 0;
}
