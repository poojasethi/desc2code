using namespace std;
#define li		long long
#define rep(i,to)	for(li i=0;i<((li)to);i++)
#define pb		push_back
#define sz(v)		((li)v.size())

int dx[]={1,0,-1,0};
int dy[]={0,1,0,-1};
int main(){
	li n,m;
	cin>>n>>m;
	vector<vector<li> > mp(n,vector<li>(m,-1));
	li mid=(n*m+1)/2;
	rep(i,n)rep(j,m)if((i+j)%2==0) mp[i][j]=mid--;
	mid=(n*m+1)/2+1;
	for(li i=n-1;0<=i;i--)for(li j=m-1;0<=j;j--)if(mp[i][j]==-1) mp[i][j]=mid++;
	rep(i,n)rep(j,m)rep(k,4){
		li x=i+dx[k];
		li y=j+dy[k];
		if(0<=x && x<n && 0<=y && y<m){
			if(abs(mp[i][j]-mp[x][y])==1){
				cout<<"-1"<<endl;
				return 0;
			}
		}
	}
	rep(i,n){ rep(j,m) cout<<mp[i][j]<<" "; cout<<endl; }
}
