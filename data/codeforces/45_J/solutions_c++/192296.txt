#include <algorithm>
#include <iostream>
#include <cstdio>
using namespace std;
int n, m, a[100][100], k = 1, p;
bool use[10010];
int main()
{
	cin >> n >> m;
	for(int i = 0; i < n; i++)
		for(int j = !(i % 2); j < m; j+=2)
			a[i][j] = k,
			k++;
	for(int i = 0; i < n; i++)
		for(int j = (i % 2); j < m; j+=2)
			for(int h = k; h <= n * m; h++)
				if(!use[h]) {
					int fl = 1;
					if(i - 1 >= 0 && abs(h - a[i-1][j]) <= 1)fl = 0;
					if(i + 1 < n && abs(h - a[i+1][j]) <= 1)fl = 0;
					if(j - 1 >= 0 && abs(h - a[i][j-1]) <= 1)fl = 0;
					if(j + 1 < m && abs(h - a[i][j+1]) <= 1)fl = 0;
					
					//cout << i << ' '<< j << ' ' << h << ' ' << fl << endl;
					if(fl == 1)
					{
						a[i][j] = h;
						use[h] = 1;
						break;
					}
				}
	for(int i = 0; i < n; i++)
		for(int j = 0; j < m; j++)
			if(i + 1 < n && abs(a[i][j] - a[i+1][j]) <= 1 || j + 1 < m && abs(a[i][j] - a[i][j+1]) <= 1) {
				cout << -1;
				return 0;
			}
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < m; j++)
			cout << a[i][j] << ' ';
		cout << endl;
	}
	return 0;
}
