#include <cstdio>

int main()
{
    int n, m, a, b;
    scanf("%d%d", &n, &m);
    if (n * m == 2 || n * m == 3 || (n == 2 && m == 2))
        {printf("-1\n"); return 0;}
    a = 1; b = 1 + n * m / 2;
    for (int i = 0; i < n; ++i){
        for (int j = 0; j < m; ++j) {
            if (j) printf(" ");
            if ((i + j) & 1) printf("%d", a++);
                else printf("%d", b++);
        }
        printf("\n");
    }
    return 0;
}
