#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int n, m;
int d[105][105]={};
void solve() {
	int c=0;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			if((i+j)%2==1)
				d[i][j] = ++c;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			if((i+j)%2==0)
				d[i][j] = ++c;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			printf("%d%c", d[i][j], j==m-1?'\n':' ');
}

int main(void) {
	scanf("%d%d", &n, &m);
	if (n*m==3 || n*m==2 || (n==2&&m==2)) puts("-1");
	else { solve();
	}
	return 0;
}
