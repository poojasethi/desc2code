#include <cstdio>
int abs(int x) { return x < 0 ? -x : x; }
int n, m, a[109][109], i, j, x, y;
int main () {
	scanf("%d %d", &n, &m);
    x = 1; y = (n * m) / 2 + 1;
    for (i = 1; i <= n; i++) {
        for (j = 1; j <= m; j++) {
            if ((i + j) % 2 == 0) {
                a[i][j] = y;
                y++;
            }
            else {
                a[i][j] = x;
                x++;
            }
            if ((i - 1 > 0 && abs(a[i - 1][j] - a[i][j]) == 1)
                || (j - 1 > 0 && abs(a[i][j - 1] - a[i][j]) == 1)) {
                    puts("-1");
                    return 0;
                }
        }
    }
    for (i = 1; i <= n; i++)
			for (j = 1; j <= m; j++)
				printf("%d%c", a[i][j], j == m ? '\n' : ' ');

}
