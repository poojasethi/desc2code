#include <bits/stdc++.h>

using namespace std;

int a[105][105];

int main()
{

    int n, m, r, c;

    cin>>r>>c;

    n = min(r,c);
    m = max(r,c);

    if(n*m==2 || n*m==3 || n==2 && m==2)
    {
        cout<<"-1"<<endl;
        return 0;
    }

    int key = 1, i = 0, j = 1;

    while(key<=n*m)
    {
        if(j>=m)
        {
            i++;
            j = j%m;
        }
        if(i>=n) i = 0, j = 0;
        else
        {
            a[i][j] = key++;
            j+=2;
        }
//        cout<<i<<' '<<j<<endl;
    }

    if(r==n)
    {
        for(int i = 0; i<n; i++)
        {
            for(int j = 0; j<m; j++)
            {
                if(j) cout<<" ";
                cout<<a[i][j];
            }
            cout<<endl;
        }
    }
    else
    {
        for(int i = 0; i<m; i++)
        {
            for(int j = 0; j<n; j++)
            {
                if(j) cout<<" ";
                cout<<a[j][i];
            }
            cout<<endl;
        }
    }

    return 0;
}
