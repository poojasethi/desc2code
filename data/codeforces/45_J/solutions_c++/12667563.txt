#include <bits/stdc++.h>

using namespace std;
#define FI "table.i2"
#define FO "table.o2"
#define NN 101
int m, n, a[NN][NN];

int main()
{
    memset(a,-1,sizeof(a[0])*NN);
    #ifndef ONLINE_JUDGE
    freopen(FI,"r",stdin);
    freopen(FO,"w",stdout);
    #endif // ONLINE_JUDGE
    scanf("%d %d", &m, &n);
    if (m == 1 && n == 1) {printf("1\n"); return 0;}
    if (m + n <= 4) {printf("-1\n"); return 0;}
    int x1 = 1, x2 = (m*n+1)/2+1;
    bool f = ((m+n) % 2 == 0 ? true : false);
    for(int i = 1; i <= m; i++){
        for(int j = 1; j <= n; j++){
            if(f) {a[i][j]= x1; printf("%d ",x1); x1++;}
            else  {a[i][j]= x2; printf("%d ",x2); x2++;}
            f = !f;
        }
        if(n % 2 == 0) f = !f;
        printf("\n");
    }
    return 0;
}
