int main() {
	int n , m;
	cin >> n >> m;
	if (n*m == 2 || (n == 2 && m == 2) || (n*m == 3)) {
		cout << "-1" << endl;
	} else {
		int a = 1;
		int b = 1 + n*m/2;
		for (int i = 0 ; i < n ; i ++) {
			for (int j = 0 ; j < m ; j ++) {
				if (j) cout << ' ' ;
				if ((i +j)&1) cout << a ++;
				else cout << b ++;
			}
			cout << endl;
		}
	}
	return 0;
}
