#include "fstream"
#include "iostream"
#include "utility"
#include <string.h>
#include <conio.h>

using namespace std;

int main(int argc, char *argv[]) {
    int a[100][100]={};
    int r, c, cur, cur2;
    cin >> r >> c;
    if (r==1 && c==1) cout << 1 << endl;
    else if (r==1 && c<4 || c==1 && r<4 || r==2 && c==2) cout << -1 << endl;
    else {
        cur=r*c/2+(r*c)%2;
        cur2=r*c;
        if (c==2) {
            for (int i=0; i<c; i++)
                for (int j=0; j<r; j++) {
                     if ((i+j)%2) {a[j][i]=cur2; cur2--;}
                     else {a[j][i]=cur; cur--;}
                }
        } else {
            for (int i=0; i<r; i++)
                for (int j=0; j<c; j++) {
                     if ((i+j)%2) {a[i][j]=cur2; cur2--;}
                     else {a[i][j]=cur; cur--;}
                }
        }
        for (int i=0; i<r; i++, cout << endl)
            for (int j=0; j<c; j++) cout << a[i][j] << ' ';
    }
//    getch();
    return 0;
}
