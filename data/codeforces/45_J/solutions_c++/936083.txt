#include <cstdio>

int tab[111][111];

int main() {
  int a,b;
  bool zamieniony=false;
  scanf("%d%d",&a,&b);
  if(a==1 && b==1) { printf("1\n"); return 0; }
  if((a==2 && b==1) || (a==1 && b==2) || (a==1 && b==3) || (a==3 && b==1) || (a==2 && b==2)) { printf("-1\n"); return 0; }
  if(a==1 && b==4) {printf("3 1 4 2\n"); return 0;}
  if(a==4 && b==1) {printf("3\n1\n4\n2\n"); return 0;}
  if(a<b){int t=a; a=b; b=t; zamieniony=true;}
  int fillnumber=1;
  if(b==1) {
    for(int i=0; i<a; i+=2) {
      tab[i][0]=fillnumber++;
    }
    for(int i=1; i<a; i+=2) {
      tab[i][0]=fillnumber++;
    }
  } else 
  for(int i=0; i<b; i++) {
    int x=i;
    for(int j=0; j<a; ++j) {
      tab[j][x]=fillnumber++;
      ++x;
      if(x>=b)x=0;
    }
  }
  
  if(zamieniony) {
    for(int i=0; i<b; ++i) {
      for(int j=0; j<a; ++j) {
	printf("%d ", tab[j][i]);
      }
      printf("\n");
    }
  } else {
    for(int i=0; i<a; ++i) {
      for(int j=0; j<b; ++j) {
	printf("%d ", tab[i][j]);
      }
      printf("\n");
    }
  }
  return 0;
}
