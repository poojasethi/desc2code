#include<stdio.h>
int col[101][101];
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    if(n*m==1)printf("1\n");
    else if(n*m==2||n*m==3)printf("-1\n");
    else if(n==2&&m==2)printf("-1\n");
    else if(n==2&&m==3)printf("3 6 2\n5 1 4\n");
    else if(n==3&&m==2)printf("3 5\n6 1\n2 4\n");
    else
    {
        int cnt=1;
        for(int i=1;i<=n;i++)
        {
            for(int j=1+i%2;j<=m;j=j+2)
            {
                col[i][j]=cnt++;
            }
        }
        for(int i=1;i<=n;i++)
        {
            for(int j=2-i%2;j<=m;j=j+2)
            {
                col[i][j]=cnt++;
            }
        }
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=m;j++)
            {
                if(j!=1)printf(" ");
                printf("%d",col[i][j]);
            }
            printf("\n");
        }
    }
    return 0;
}
