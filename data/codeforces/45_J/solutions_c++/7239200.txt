#include<cstdio>
#include<cstring>

const int MAX_N = 107;

int n, m;

int ans[MAX_N][MAX_N];

int main() {
	scanf("%d%d", &n, &m);
	if (n == 1 && m == 1) puts("1");
	else if (n + m < 5) puts("-1");
	else {
		int a = n * m / 2 + 1, b = 1;
		for (int i = 0; i < n; ++i) {
			for (int j = 0; j < m; ++j) {
				if ((i + j) & 1) ans[i][j] = b++;
				else ans[i][j] = a++;
			}
		}
		for (int i = 0; i < n; ++i) {
			for (int j = 0; j < m; ++j) {
				if (j == 0) printf("%d", ans[i][j]);
				else printf(" %d", ans[i][j]);
			} 
			puts("");
		}

	}
	return 0;
}
