#include <iostream>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)

int a[110][110];

int main(){
        int n,m,i,j,tt;
        cin>>n>>m;
        if (n*m==3||n*m==2||(n==2&&m==2)) cout<<-1<<endl;
        else{
                tt=0;
                REP(i,0,n)REP(j,0,m) if((i+j)%2==1) a[i][j]=++tt;
                REP(i,0,n)REP(j,0,m) if((i+j)%2==0) a[i][j]=++tt;
                REP(i,0,n){REP(j,0,m) cout<<a[i][j]<<' ';cout<<endl;}
        }
}
