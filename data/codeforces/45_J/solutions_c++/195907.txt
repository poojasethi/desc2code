#include<iostream>
#include<cstring>
#include<vector>
#include<map>

using namespace std;

int ans[101][101];

typedef pair<int,int> P;

int main(){
  int m,n;
  cin >> m >> n;
  P p(m,n);
  memset(ans,sizeof(ans),0);

  vector<P> excep;
  // excep.push_back(make_pair(1,1));
  excep.push_back(make_pair(1,2));
  excep.push_back(make_pair(2,1));
  excep.push_back(make_pair(1,3));
  excep.push_back(make_pair(3,1));
  excep.push_back(make_pair(2,2));

  for(int i = 0; i < excep.size(); i++)
    { if(excep[i] == p){cout << "-1" << endl; return 0;} }

  int k = 1;
  int j;

  for(int i = 1; i <= m; i++){
    for(i % 2 ?  j = 2 :  j = 1; j <= n; j += 2 ){
	ans[i][j] = k++;
      }}
    for(int i = 1; i <= m; i++ ){
      for( i % 2 ?  j = 1 :  j = 2; j <= n; j += 2){
	ans[i][j] = k++;
      }}

    for(int i = 1; i <= m; i++){
      for(int j = 1; j <= n; j++){
	if(j != n){ cout << ans[i][j] << " ";}
	else{ cout << ans[i][n] << endl;}
      }}

  return 0;
}
