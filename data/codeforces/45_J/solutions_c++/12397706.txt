#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;

int S[101][101];

int main()
{
    int n,m,r,c;

    cin>>r>>c;
    n = min(r,c);
    m = max(r,c);

    if(n*m==2||n*m==3||(n==2&&m==2))
    {
        cout<<"-1"<<endl;
        return 0;
    }

    int key = 1, i = 0, j = 1;
    while(key<=n*m)
    {
        if(j>=m)
        {
            i++;
            j = j%m;
        }
        if (i>=n)
        {
            i=0;
            j=0;
        }
        else
        {
            S[i][j] = key++;
            j+=2;
        }
    }

    if(r==n)
    {
        for(int i = 0; i < n; i++)
        {
            for(int j = 0; j < m; j++)
            {
                if (j) cout<<" ";
                cout<<S[i][j];
            }
            cout<<endl;
        }
    }
    else
    {
        for(int i = 0; i < m; i++)
        {
            for(int j = 0; j < n; j++)
            {
                if (j) cout<<" ";
                cout<<S[j][i];
            }
            cout<<endl;
        }
    }
    return 0;
}
