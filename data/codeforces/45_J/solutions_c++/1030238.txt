#include<cstdio>  
#include<cstring>  
#include<cstdlib>  
#include<iostream>  
#include<algorithm>  
#include<map>
#include<set>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=101;
int n,m,i,j,k=0,a[mn][mn];
int main()
{
    scanf("%d%d",&n,&m);
    if(n*m==2||n*m==3||(n==2&&m==2))
    {
        printf("-1\n");
        return 0;
    }
    fo(i,1,n)
        fo(j,1,m)
            if((i+j)&1)
                a[i][j]=++k;
    fo(i,1,n)
        fo(j,1,m)
        if(!((i+j)&1))
            a[i][j]=++k;
    fo(i,1,n)
    {
        fo(j,1,m-1)printf("%d ",a[i][j]);
        printf("%d\n",a[i][m]);
    }
    return 0;
}
