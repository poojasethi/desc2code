#include <iostream>
#include <conio.h>
#include <algorithm>
using namespace std;
int main ()
{
	int n,m,h=0;
	cin>>n>>m;
	if (m==2)
	{
		m=n;
		n=2;
		h=1;
	}
	int a[n][m];
	if (n==1 && m==2)
	{
		cout<<-1;
		return 0;
	}
	if (m==1 && n==2)
	{
		cout<<-1;
		return 0;
	}
	if (m==2 && n==2)
	{
		cout<<-1;
		return 0;
	}
	if (m==1 && n==3)
	{
		cout<<-1;
		return 0;
	}
	if (m==3 && n==1)
	{
		cout<<-1;
		return 0;
	}
	int p,q;
	p=n*m;
	if (p%2==1)
	{
		q=p/2;
		q++;
	}
	else
	q=p/2;
	int t=0;
	for (int i=0;i<n;i++)
	{
		for (int j=0;j<m;j++)
		{
			if (t%2==0)
			{
				a[i][j]=q;
				q--;
			}
			if (t%2==1)
			{
				a[i][j]=p;
				p--;
			}
			t++;
		}
	}
	if (h==1)
	{
		n=m;
		m=2;
	}
	if (m==2)
	{
		 for (int i=0;i<n;i++)
		 {
		 	for (int j=0;j<m;j++)
		 	{
		 		cout<<a[j][i]<<" ";
		 	}
		 	cout<<endl;
		 }
	}
	else
	{
		for (int i=0;i<n;i++)
		{
			for (int j=0;j<m;j++)
			cout<<a[i][j]<<" ";
			cout<<endl;
		}
	}
	return 0;
}
