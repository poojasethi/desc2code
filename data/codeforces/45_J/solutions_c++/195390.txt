#include <iostream>
#include <cstdio>
using namespace std;

int N,M,T;
int G[105][105];

int main(){
	scanf("%d%d",&N,&M);
	if(
		(N == 1 && M == 2) ||
		(N == 1 && M == 3) ||
		(M == 1 && N == 2) ||
		(M == 1 && N == 3) ||
		(N == 2 && M == 2)
	) printf("-1\n");
	else if(N == 1 && M == 4) printf("3 1 4 2\n");
	else if(N == 4 && M == 1) printf("3\n1\n4\n2\n");
	else{
		T = 1;
		for(int i=0;i<N;++i)
			for(int j=0;j<M;++j)
				if((i+j)%2 == 1)
					G[i][j] = T++;
		for(int i=0;i<N;++i)
			for(int j=0;j<M;++j)
				if((i+j)%2 == 0) G[i][j] = T++;
		for(int i=0;i<N;++i){
			for(int j=0;j<M;++j){
				printf("%d",G[i][j]);
				if(j<M-1) printf(" ");
				else printf("\n");
			}
		} 
	}
//	system("pause");
}
