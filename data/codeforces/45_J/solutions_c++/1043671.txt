#include <iostream>
using namespace std;

int n,m,t,a[101][101];

int main()
{
    cin>>n>>m;
    if ((n*m==2)||(n*m==3)||(n==2)&&(m==2))
    {
      cout<<-1;
      return 0;
    }
    t=0;
    for (int i=1;i<=n;i++)
    {
        if (i%2==0)
        {
          for (int j=1;j<=(m+1)/2;j++)
            {
                t=t+1;
                a[i][j*2-1]=t;
            }
        }
        else
        {
            for (int j=1;j<=m/2;j++)
            {
              t=t+1;
              a[i][j*2]=t;
            }
        }
    }
    for (int i=1;i<=n;i++)
    {
        if (i%2==0)
        {
          for (int j=1;j<=m/2;j++)
          {
              t=t+1;
              a[i][j*2]=t;
          }
        }
        else
        {
            for (int j=1;j<=(m+1)/2;j++)
            {
                t=t+1;
                a[i][j*2-1]=t;
            }
        }
    }
    for (int i=1;i<=n;i++)
    {
      for (int j=1;j<=m;j++)
      {
          cout<<a[i][j];
          if (j<m) cout<<" ";
      }
      cout<<endl;
    }
    return 0;
}
