#include<bits/stdc++.h>
using namespace std;
const int N = 110;
int n,m;
int a[N][N];

bool check()
{
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (i+1<n) {
                int k = abs(a[i][j]-a[i+1][j]);
                if (k==1)return 0;
            }
            if (j+1<m) {
                int k = abs(a[i][j]-a[i][j+1]);
                if (k==1)return 0;
            }
        }
    }
    return 1;
}

int main()
{
    scanf("%d%d", &n, &m);
    int k = n*m;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if ((i+j)%2==1) {
                a[i][j] = k--;
            }
        }
    }
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if ((i+j)%2==0) {
                a[i][j] = k--;
            }
        }
    }
    if (check()) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0;j < m; ++j){
                printf("%d%c",a[i][j],(j==m-1)?'\n':' ');
            }
        }
    }
    else {
        printf("-1\n");
    }
    return 0;
}
