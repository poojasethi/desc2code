#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<iostream>
#define MAXN 105
using namespace std;
int r,c,g[MAXN][MAXN];
void solve()
{
    int idx=0;
    memset(g,0,sizeof(g));
    for(int i=0;i<r;i++)
        for(int j=0;j<c;j++)
            if((i+j)%2)
                g[i][j]=++idx;
    for(int i=0;i<r;i++)
        for(int j=0;j<c;j++)
            if(!g[i][j])
                g[i][j]=++idx;
    for(int i=0;i<r;i++)
        for(int j=0;j<c;j++)
            if(i<r-1&&abs(g[i][j]-g[i+1][j])<=1 || j<c-1&&abs(g[i][j]-g[i][j+1])<=1){
                printf("-1\n");
                return;
            }
    for(int i=0;i<r;i++)
        for(int j=0;j<c;j++)
            printf("%d%c",g[i][j],j<c-1?' ':'\n');
}
                
int main()
{
    while(scanf("%d%d",&r,&c)!=EOF)
        solve();
    return 0;
}
