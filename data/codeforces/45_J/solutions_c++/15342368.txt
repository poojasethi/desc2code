#include<cstdio>
#include<algorithm>
using namespace std;

int n, m, f;
int g[111][111];

int main() {
	scanf("%d%d", &n, &m);
	f=0;
	if(n>m) {
		swap(n, m);
		f = 1;
	}
	if(m==1) {
		printf("1\n");
		return 0;
	}
	if(m+n<=4) {
		printf("-1\n");
		return 0;
	}
	if(m<=3) {
		if(n==2) {
			g[0][0] = 3; g[0][1] = 6; g[0][2] = 2;
			g[1][0] = 5; g[1][1] = 1; g[1][2] = 4;
		} else {
			g[0][0] = 1; g[0][1] = 4; g[0][2] = 7;
			g[1][0] = 6; g[1][1] = 8; g[1][2] = 3;
			g[2][0] = 9; g[2][1] = 2; g[2][2] = 5;
		}
	} else {
		for(int i=(m/2+1), j=0;i<=m;i++, j+=2)
			g[0][j] = i;
		for(int i=1, j=1;i<=(m/2);i++, j+=2)
			g[0][j] = i;
		for(int i=1;i<n;i++)
			for(int j=0;j<m;j++)
				g[i][j] = g[i-1][j] + m;
	}
	if(!f) {
		for(int i=0;i<n;i++)
			for(int j=0;j<m;j++)
				printf("%d%c", g[i][j], j<m-1 ? ' ' : '\n');
	} else {
		for(int i=0;i<m;i++)
			for(int j=0;j<n;j++)
				printf("%d%c", g[j][i], j<n-1 ? ' ' : '\n');
	}
	
	return 0;
}
