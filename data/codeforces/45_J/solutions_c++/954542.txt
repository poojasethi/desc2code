#include <iostream>

using namespace std;

int main ( ){

    int res[101][101], n, m;

    cin >> n >> m;

    if(n == 1 && m == 1) {
        cout << 1;
        return 0;
    }
    if(n == 1) {
        if(m < 4) {
            cout << -1;
            return 0;
        }
    }
    if(m == 1) {
        if(n < 4) {
            cout << -1;
            return 0;
        }
    }
    if(n == 2 && m == 2) {
        cout << -1;
        return 0;
    }

    int t = (m * n) / 2 + 1, s = 1;
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= m; j++) {
            if((i + j) % 2 == 1) res[i][j] = s++;
            else res[i][j] = t++;
        }
    }

    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= m; j++) {
            cout << res[i][j] << " ";
        }
        cout << endl;
    }

return 0;
}
