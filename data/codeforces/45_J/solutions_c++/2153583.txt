#include <iostream>
#include <stdio.h>
#include <string>
#include <algorithm>
using namespace std;

int main()
{//freopen("i.txt","r",stdin);
   int n,m,a[100][100];
   cin>>n>>m;
  if(n*m==2||(m*n==4&&n==2)||n*m==3) {cout<<"-1"; return 0;}
  if(n*m==1) {cout<<1; return 0;}
  int k=n*m/2+1; int x=1;
   for (int i=0; i<n; i++)
      for (int j=0; j<m; j++)
  {
      if ((i+j)%2==0) {a[i][j]=k; k++;} else
      {a[i][j]=x; x++;}
  }

   for (int i=0; i<n; i++)
      for (int j=0; j<m; j++)
      {
       cout<<a[i][j];
       if (j==m-1) cout<<endl; else cout<<" ";
      }

    return 0;
}
