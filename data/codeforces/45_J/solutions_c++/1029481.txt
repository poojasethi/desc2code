#include<cstdio>
#include<cstring>
using namespace std;

const int N = 105;

#define fo(i , st , en) for (int i = st; i <= en; i++)

int ans[N][N];
int nn , mm , n , m , tot , temp;

int main(){
    scanf("%d %d" , &n , &m); tot = 0; nn = n; mm = m;
    if (nn > mm){
        temp = nn; nn = mm; mm = temp;
    }
    if (nn == 1 && mm == 1){
        puts("1"); return 0;
    }
    if (nn == 1 && mm <= 3 || nn == 2 && mm == 2){
        puts("-1"); return 0;
    }
    fo (i , 1 , n)
        fo (j , 1 , m)
            if ((i + j) & 1)
                ans[i][j] = ++tot;
    fo (i , 1 , n)
        fo (j , 1 , m)
            if ((i + j) & 1 ^ 1)
                ans[i][j] = ++tot;
    fo (i , 1 , n){
        printf("%d" , ans[i][1]);
        fo (j , 2 , m)
            printf(" %d" , ans[i][j]);
        puts("");
    }
    return 0;
}
