#include <cstdio>

int main() {
	int n, m;
	scanf("%d%d", &n, &m);

	if (n == 1 && m == 1) {
		printf("1\n");
		return 0;
	}

	if (n + m < 5) {
		printf("-1\n");
		return 0;
	}

	int b = n * m / 2;
	int cb = 1;
	int cw = b + 1;

	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			int x;
			if ((i + j) % 2 == 0)
				x = cw++;
			else
				x = cb++;

			if (j)
				printf(" ");
			printf("%d", x);
		}
		printf("\n");
	}
	return 0;
}
