#include<bits/stdc++.h>
using namespace std;
long long int f[200100];
map<long long int,long long int>ans;
typedef pair<int,long long int>ii;
ii a[200100];
bool price(ii x, ii y)
{
    return x.second<y.second;
}
set<long long int>st;
int main()
{
    long long int d,n,m,res=0;
    scanf("%lld %lld %lld",&d,&n,&m);
    for(long long int i=1;i<=m;i++)
    {
        scanf("%lld %lld",&a[i].first,&a[i].second);
    }
    m++; a[m]=ii(d,0);
    sort(a+1,a+1+m,price);
    for(long long int i=1;i<=m;i++)
    {
        if(st.upper_bound(a[i].first)!=st.end()) ans[a[i].first]= (*st.upper_bound(a[i].first));
        else ans[a[i].first]=123456789;
        st.insert(a[i].first);
    }
    sort(a+1,a+1+m);
    f[0]=n;
    for(long long int i=1;i<=m;i++)
    {
        f[i]=f[i-1]-(a[i].first-a[i-1].first);
        if(f[i]<0) {printf("-1"); return 0;}
        long long int j=ans[a[i].first];
        if(j-a[i].first>f[i])
        {
            res+=(min(j-a[i].first,n)-f[i])*a[i].second;
            f[i]=min(j-a[i].first,n);
        }
        if(a[i].first==d)
        {
            printf("%lld",res); return 0;
        }
    }
    printf("%lld",res); return 0;
}
