#include <bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

#define PII pair<int,int>
#define mp make_pair
#define fi first
#define se second

using namespace std;

int dis,v,n;
LL ans;

struct pt{
	int p,c;
	bool operator<(const pt&k) const{
		return c>k.c;
	}
};

priority_queue<pt> Q;

vector<PII> V,V2;

int main(){
	scanf("%d %d %d",&dis,&v,&n);
	FI(i,1,n){
		int a,b;
		scanf("%d %d",&a,&b);
		V.push_back(mp(a,b));
		if(a+v<dis) V2.push_back(mp(a+v,-1));
	}
	V.push_back(mp(0,0));
	V2.push_back(mp(v,-1));
	V.push_back(mp(dis,0));
	sort(V.begin(),V.end());
	FI(i,1,(int)V.size()-1){
		if(V[i].fi-V[i-1].fi>v){
			printf("-1\n");
			return 0;
		}
	}
	FI(i,0,(int)V2.size()-1) V.push_back(V2[i]);
	
	sort(V.begin(),V.end());
	Q.push((pt){0,0});
	FI(i,1,(int)V.size()-1){
		while(!Q.empty()){
			pt cur=Q.top();
			if(V[i].fi-cur.p>v){
				Q.pop();
				continue;
			}
			ans+=1LL*(V[i].fi-V[i-1].fi)*cur.c;
			break;
		}
		if(V[i].se!=-1) Q.push((pt){V[i].fi,V[i].se});
	}
	printf("%lld\n",ans);
	return 0;
}
