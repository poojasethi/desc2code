#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<vector>
#include<queue>
#include<string>
using namespace std;
pair<__int64, __int64> a[200010], q[200010];
__int64 d, n, m, l = 1, r = 0, sum;

int main()
{
	cin >> d >> n >> m;
	for (int i = 1; i <= m; i++)
		scanf("%I64d%I64d", &a[i].first, &a[i].second);
	sort(a + 1, a + m + 1);
	a[++m].first = d;
	__int64 now = n, pos = 0, ans = 0;
	for (int i = 1; i <= m; i++)
	{
		while (a[i].first - pos > now)
		{
			if (l > r)
			{
				puts("-1");
				return 0;
			}
			__int64 dt = min(a[i].first - pos - now, q[l].second - sum);
			ans += dt*q[l].first;
			now += dt, sum += dt;
			while (l <= r && q[l].second <= sum) l++;
		}
		now -= a[i].first - pos;
		pos = a[i].first;
		while (l <= r && a[i].second <= q[r].first) r--;
		q[++r].first = a[i].second, q[r].second = n - now + sum;
	}
	cout << ans << endl;
	return 0;
}
