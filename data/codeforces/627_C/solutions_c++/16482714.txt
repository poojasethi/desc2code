#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int N=1000000;

struct node
{
	int x,y;
	bool operator <(node a)const {return x<a.x;}
}a[N+10];

typedef long long ll;

int c[N+10],next[N+10];

inline int lowbit(int x) {return x&(-x);}

void change(int x,int y)
{
	while (x<=N)
	{
		c[x]=min(c[x],y);x+=lowbit(x);
	}
}

int query(int x)
{
	int t=N;
	while (x)
	{
		t=min(t,c[x]);x-=lowbit(x);
	}
	return t;
}


int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int d,n,m;scanf("%d%d%d",&d,&n,&m);
	for (int i=1;i<=m;i++) scanf("%d%d",&a[i].x,&a[i].y);
	sort(a+1,a+m+1);
	a[0].x=0;a[m+1].x=d;
	for (int i=0;i<=m;i++) if (a[i+1].x-a[i].x>n) {puts("-1");return 0;}
	for (int i=1;i<=N;i++) c[i]=m+1;
	for (int i=m;i;i--)
	{
		next[i]=query(a[i].y);
		change(a[i].y,i);
	}
	ll ans=0;int now=n;
	for (int i=1;i<=m;i++)
	{
		now=now-a[i].x+a[i-1].x;
		int p=max(0,min(n,a[next[i]].x-a[i].x)-now);
		ans=ans+ll(p)*a[i].y;now+=p;
	}
	printf("%I64d\n",ans);
	return 0;
}
