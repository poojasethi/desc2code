#include <bits/stdc++.h>
using namespace std; 
typedef long long ll;
typedef pair<ll,ll> pp;
int n, m, d; 
pair<int,int> a[200005];
int arrive[200005];
set<pp> extra;
int nxt[200005];
int main() {
    
    scanf("%d%d%d", &d,&n,&m);
    for(int i=0; i<m; i++) scanf("%d%d", &a[i].first, &a[i].second);
    
    sort(a, a+m);
    a[m] = make_pair(d, 0);
    for(int i=0; i<=m; i++) {
        if (!i && a[i].first > n) { printf("-1\n"); return 0; } 
        if (a[i].first - a[i-1].first > n ) { printf("-1\n"); return 0; } 
    }
    stack<int> s; s.push(m);
    for(int i=m-1; i>=0; i--) {
        while (a[s.top()].second >= a[i].second) s.pop(); 
        nxt[i] = s.top(); 
        s.push(i);
    }
    
    ll ans = 0, gas = n-a[0].first;
    for(int i=0; i<m; i++) {
        int r = nxt[i];
        int buy = a[r].first - a[i].first;
        if (buy > gas) buy = min((ll)buy-gas, n-gas);
        else buy = 0;
        ans += (ll)buy * a[i].second;
        gas += buy;
        gas -= a[i+1].first - a[i].first;
    }
    
    printf("%I64d\n", ans);
    
    return 0; 
}
