#include <bits/stdc++.h>
using namespace std;

#define IO ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);
#define endl "\n"
#define INF 2147483647
#define DBG 666

typedef long long ll;

int d, n, m;
int parent[200005];
pair<int, int> p[200005];

void init(){

	cin >> d >> n >> m;
	for(int i = 0; i < m; i++)
		cin >> p[i].first >> p[i].second;
	sort(p, p+m);
}

void solve(){

	p[m] = make_pair(d, -1);
	stack<int> s;
	s.push(m);
	for(int i = m-1; i >= 0; i--){

		while(p[s.top()].second >= p[i].second)
			s.pop();

		parent[i] = s.top();
		s.push(i);
	}
	int t = n - p[0].first;
	ll result = 0;
	for(int i = 0; t >= 0 && i < m; i++){

		if(p[parent[i]].first - p[i].first > t){

			int tmp = min(p[parent[i]].first-p[i].first, n)-t;
			result += 1LL*tmp*p[i].second;
			t += tmp;
		}
		t -= p[i+1].first - p[i].first;;
	}
	cout << ((t < 0) ? -1 : result) << endl;
}

int main() {

	IO;
	init();
	solve();

	return 0;
}
