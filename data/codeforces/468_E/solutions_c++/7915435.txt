const int s_mod=100007;

struct has
{
   long long key;
   vector<int>w;
};
has hs[s_mod][107],now,nw;
int num[s_mod],fac[110000];
int tn,n,m,ans,mod=1000000007;

has find(long long state)
{
     int i,j,s,p,q,la;
     la=(int)(state%s_mod);
     for(i=0;i<num[la];i++)
     {
        if(hs[la][i].key==state)
            return hs[la][i];
     }
    // while(true)
      //  puts("no");
}
void dfs(int id,long long state,bool vis[],int list[],int &cnt_list)
{
     int i,j,s,p,q,ip;
     vis[id]=true;
     list[cnt_list++]=id;
     for(i=0;i<deg[id];i++)
     {
         ip=adj[id][i];
         if(!vis[ip]&&(state&(1LL<<ip)))
            dfs(ip,state,vis,list,cnt_list);
     }
}
void merge(has &gift,has now)
{
     int i,j,s,p,q,siz=gift.w.size();
     for(i=siz+(int)(now.w.size())-2;i>=siz;i--)
         gift.w.push_back(0);
     for(i=(int)(gift.w.size())-1;i>=0;i--)
     {
         if(gift.w[i]==0)
            continue;
         for(j=(int)(now.w.size())-1;j>=0;j--)
         {
             if(now.w[j]==0)
             {
                 if(j==0)
                    gift.w[i]=0;
                 continue;
             }
             if(j==0)
                 gift.w[i]=(int)((long long)gift.w[i]*(long long)now.w[j]%mod);
             else
             {
                 gift.w[i+j]+=(int)((long long)gift.w[i]*(long long)now.w[j]%mod);
                 gift.w[i+j]%=mod;
             }
         }
     }
  //   gift.len+=now.len-1;
}
void comb(has &gift,has now)
{
     int i,j,s,p,q;
     for(i=0;i<now.w.size();i++)
     {
        if(i<gift.w.size())
          gift.w[i]=(gift.w[i]+now.w[i])%mod;
        else
        {
            gift.w.push_back(now.w[i]);
          //gift.w[i]=now.w[i];
        }
     }
   //  gift.len=max(gift.len,now.len);
}
void solve(long long state)
{
     int i,j,s,p,q,id,la,ps,id1,id2;
     has nw,now;
     bool need;
     long long  nstate;
     //bool vis[55];
     la=(int)(state%s_mod);
     for(i=0;i<num[la];i++)
     {
         if(hs[la][i].key==state)
             return;
     }
     hs[la][num[la]].w.clear();
     hs[la][num[la]].w.push_back(1);;
     //hs[la][num[la]].len=1;
     hs[la][num[la]].key=state;
     nw=hs[la][num[la]];
     ps=num[la];
     num[la]++;
     while(num[la]>=107)
        puts("orz");
     bool vis[55];
     int list[51],cnt_list;
     for(i=0;i<m;i++)
        vis[i]=false;
     need=false;
     //printf("state=%I64d\n",state);
     for(i=0;i<m;i++)
     {
        if(state&(1LL<<i))
        {
            if(!vis[i])
            {
               cnt_list=0;
               dfs(i,state,vis,list,cnt_list);
               nstate=0;
               for(j=0;j<cnt_list;j++)
                  nstate|=((long long)1<<(long long)list[j]);
               if(nstate!=state)
               { 
                  // if((state&nstate)!=nstate) 
                    //  printf("orz1=%I64d,orz2=%I64d\n",state&nstate,nstate);// puts("orz");
                   solve(nstate);
                   now=find(nstate);
                   merge(hs[la][ps],now);
                //   printf("len1=%d,len2=%d,cnt_list=%d\n",hs[la][ps].w.size(),now.w.size(),cnt_list);
               }
               else
               {
                   need=true;
                   break;
               }
            }
        }
     }
    // printf("st=%I64d\n",state);
     //puts("orz");
     if(need)
     {
         int in=1000000000,ip,nc;
         for(i=0;i<cnt_list;i++)
         {
             ip=list[i];
             id1=e[ip].id1;
             id2=e[ip].id2;
             nc=0;
             for(j=0;j<m;j++)
             {
                if(state&(1LL<<j))
                {
                    if(e[j].id1==e[ip].id1)
                       nc++;
                    else if(e[j].id2==e[ip].id2)
                       nc++;
                }
             }
             if(in>nc)
             {
                 in=nc;
                 id=ip;
             }
         }
         //id=list[rand()%cnt_list];
        // while(!(state&(1LL<<id)))
          //  puts("orz");
         nstate=state^(1LL<<id);
         solve(nstate);
         now=find(nstate);
         merge(hs[la][ps],now);
         //printf("orz=%d\n",hs[la][num[la]].len);
         for(i=0;i<m;i++)
         {
             if(nstate&(1LL<<i))
             {
                 if(e[i].id1==e[id].id1||e[i].id2==e[id].id2)
                    nstate^=(1LL<<i); 
                // if(e[i].id2==e[id].id2)
                  //  nstate^=(1LL<<i);
             }
         }
         solve(nstate);
         now=find(nstate);
         merge(nw,now);
         //now.len=2;
         now.w.clear();
         now.w.push_back(0);
         now.w.push_back(e[id].w-1);
         if(now.w[1]<0)
            now.w[1]+=mod;
         merge(nw,now);
         comb(hs[la][ps],nw);
     }
     //printf("dst=%I64d\n",state);
     //printf("state=%I64d\n",state);
  //   printf("num=%d\n",num[la]);
}
