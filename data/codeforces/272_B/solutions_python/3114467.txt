def f(x):
    if x == 0: return 0
    elif (x%2 == 0): return f(x/2)
    return f(x/2)+1

def start():
    from collections import Counter
    n = input()
    indices = [f(int(x)) for x in raw_input().split()]
    totalCount = 0
    c = Counter(indices)
    totalCount = sum(c[v]*(c[v]-1)/2 for v in c)
    print totalCount

start()
