n = input()
a = map(int, raw_input().split())

d = {}

def f(x):
  if x in d:
    return d[x]
  if x == 0:
    result = 0
  elif x % 2 == 0:
    result = f(x / 2)
  else:
    result = f(x / 2) + 1
  d[x] = result
  return result

for i in range(len(a)):
  a[i] = f(a[i])
  
from collections import Counter
c = Counter(a)
print sum(((x * (x - 1)) / 2 for x in c.values()))
