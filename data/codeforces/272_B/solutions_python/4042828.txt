import sys
from collections import Counter

def f(x):
    return bin(x).count('1')

def C2(n):
    return n * (n - 1) / 2

def calc(a):
    cnt = Counter()
    for x in a:
        cnt[f(x)] += 1
    return sum(C2(n) for _, n in cnt.most_common())
    
def solve(lines, output):
    _ = int(lines.readline())
    a = [int(x) for x in lines.readline().split()]
    output.write('{0}\n'.format(calc(a)))

if __name__ == '__main__':
    solve(sys.stdin, sys.stdout)
