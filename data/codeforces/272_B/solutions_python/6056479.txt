def F(x):
    if(x==0):
        return 0
    return F(x//2)+x%2
import sys
n=int(sys.stdin.readline())

L=list(map(int,sys.stdin.readline().split()))

for i in range(n):
    L[i]=F(L[i])
L.sort()

cnt=1
p=L[0]
ans=0
for i in range(1,n):
    if(L[i]==p):
        cnt+=1
    else:
        ans+=(cnt*(cnt-1))//2
        cnt=1
        p=L[i]
ans+=(cnt*(cnt-1))//2
print(ans)
