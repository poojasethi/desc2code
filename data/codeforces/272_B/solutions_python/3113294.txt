import sys
import collections

inp = [int(i) for i in sys.stdin.read().splitlines()[1].split(' ')]

def ones(c):
    res = 0
    while c > 0:
        res += c % 2
        c /= 2
    return res

inp_ones = [ones(i) for i in inp]
inp_cnt = collections.Counter(inp_ones)

res = 0
for k, v in inp_cnt.iteritems():
    res += (v*(v-1)) / 2

sys.stdout.write(str(res))
