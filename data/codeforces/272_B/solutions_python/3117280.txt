n = int(raw_input())
array = [int(x) for x in raw_input().split()]

data = []
for x in array:
    ret = 0
    while True:
        ret += x%2
        x = x/2
        if x == 1 or x == 0:
            break
    data.append(x+ret)
   
dc = {}
for x in data:
    dc[x] = dc.get(x,0) + 1

sum = 0    
for v in dc.values():
    if v == 1:
        continue
    sum += v*(v-1)/2

print sum
