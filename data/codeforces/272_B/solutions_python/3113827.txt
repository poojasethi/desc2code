import collections
raw_input()
d = [int(x) for x in raw_input().split()]
def bitcount(x):
	if x == 0:
		return 0
	r = 0
	while x:
		r += x & 1
		x //= 2
	return r
assert bitcount(5) == bitcount(3)
assert bitcount(15) == 4
h = collections.defaultdict(lambda:0)
for x in d:
	h[bitcount(x)] += 1
s = 0
for v in h.itervalues():
	s += v*(v-1)/2
print s
