n = int( raw_input() )
a = [ int( x ) for x in raw_input().split()[ : n ] ]
b = {}

i = 0
for x in a:
    count = 0
    s = bin( x )[ 2 : ]
    for j in s:
        if j == '1':
            count += 1
    b[ count ] = b.get( count, 0 ) + 1
    i += 1
s = 0
for k in b:
    s += b[ k ] * ( b[ k ] - 1 ) / 2
print s
