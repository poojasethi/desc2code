def main():
    n=int(raw_input())
    num=map(int,raw_input().split())
    hash={}
    j=0
    while(j<30):
        hash[j]=0
        j+=1
    i=0
    ll=len(num)
    while(i<ll):
        x=num[i]
        one=0
        while(x):
            if(x%2==1):
                one+=1
            x/=2
        hash[one]+=1
        i+=1
    j=0
    ans=0
    while(j<30):
        ans+=((hash[j]*(hash[j]-1))/2)
        j+=1
    print ans
main()
