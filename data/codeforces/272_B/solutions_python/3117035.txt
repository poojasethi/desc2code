#!/usr/bin/env python

def func(pp):
	if pp == 0:
		return 0
	if pp == 1:
		return 1
	
	if pp % 2 == 0:
		return func(pp/2)
	else:
		return 1+func((pp-1)/2)


n = int(raw_input())
numbers = map(int,raw_input().split())

ans = {}

for p in numbers:
	s = func(p)
	if s in ans:
		ans[s] += 1
	else:
		ans[s] = 1

res = 0
for a in ans:
	res += (ans[a]*(ans[a]-1))/2

print res
