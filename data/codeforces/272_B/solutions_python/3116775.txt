n= input()
s= raw_input().split(' ')

def work():
    cnt= { }
    for z in s:
        t= int(z)
        x= 0
        while t>0:
            if t & 1:
                x= x+1
            t= t/2
        if x in cnt:
            cnt[x]= cnt[x] +1
        else:
            cnt[x]= 1
    ans= 0
    for x in cnt.values():
        ans= ans+ x*(x-1)/2
    return ans

print work()
