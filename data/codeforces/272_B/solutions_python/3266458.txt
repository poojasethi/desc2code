#!/usr/bin/python

def f(x):
	global cache
	if x == 0:
		return x
	if x < len(cache):
		if cache[x] != -1:
			return cache[x]
		if x % 2 == 1:
			cache[x] = f(x/2) + 1
		else:
			cache[x] = f(x/2)
		return cache[x]
	if x % 2 == 1:
		return f(x/2) + 1
	return f(x/2)

n = int(raw_input())
a = map(int, raw_input().split())

cache = [-1] * 1000000
map = {}
for i in a:
	v = f(i)
	if not (v in map):
		map[v] = 0
	map[v]+=1

ans = 0
for k,v in map.iteritems():
	ans+=v*(v-1)/2
print ans
