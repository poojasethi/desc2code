def f(x):
    y = 0
    while x:
        if x % 2 == 1:
            y += 1
        x /= 2
    return y
n = int(raw_input())
a = map(int, raw_input().split())
b = map(f, a)
b.sort()
j = 0
ans = 0
for i in xrange(1, n+1):
    if i == n or b[i-1] != b[i]:
        ans += (i - j)*(i - j - 1)/2
        j = i
print ans
