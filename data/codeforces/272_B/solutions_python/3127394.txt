n = int(raw_input())
a = map(int, raw_input().split())

mp = dict()

for key in a:
  b = 0
  while key != 0 and key != 1:
    if key % 2 == 0:
      key /= 2
    else:
      key = (key - 1) / 2
      b += 1
  key = key + b
  if key not in mp.keys():
    mp[key] = 1
  else:
    mp[key] += 1

sum = 0
for val in mp.values():
  sum += val * (val - 1) / 2

print sum
