def f(x):
    if x == 0:
        return 0;
    elif x % 2 == 0:
        return f(x / 2);
    else:
        return f(x / 2) + 1;

n = int(raw_input())
num = map(int,raw_input().split())
num = map(f,num)
num.sort()

s = 1
ans = 0
for i in range(1, n):
    if num[i - 1] == num[i]:
        s += 1
    else:
        ans += s * (s - 1) / 2
        s = 1
print ans + s * (s - 1) / 2
