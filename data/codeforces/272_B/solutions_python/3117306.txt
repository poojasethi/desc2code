#!/usr/bin/python

def func(n):
  if n==0:
    return 0
  elif n%2==0:
    return func(n/2)
  else:
    return (1+func(n/2))

def main():
  n=input()
  l=[]
  l=map(int,raw_input().split())
  d={}
  for i in l:
    temp=func(i)
    if temp in d.keys():
      d[temp]+=1
    else:
      d[temp]=1
  ans=0
  for i in d.values():
    ans+=((i*(i-1))/2)
  print ans

if __name__=='__main__':
  main()
