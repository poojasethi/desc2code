n = int(raw_input())

inp = raw_input().split()
a = [int(inp[i]) for i in range(n)]

def f(ai):
	if (ai == 0):
		return 0
	elif (ai % 2 == 0):
		return f(ai/2)
	else:
		return f(ai/2) + 1

fCount = [0 for i in range(32)]

for i in range(n):
	fCount[f(a[i])] += 1

ans = 0

for j in range(32):
	ans += (fCount[j] * (fCount[j] - 1))/2

print ans
