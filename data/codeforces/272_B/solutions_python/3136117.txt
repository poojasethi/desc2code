from collections import *

mem = {}

def f(x):
    if x in mem: return mem[x]
    if x == 0:
	res = 0
    elif x % 2 == 0:
	res = f(x / 2)
    else:
	res = f(x / 2) + 1
    mem[x] = res
    return res

n = input()
a = map(int, raw_input().split())
c = defaultdict(int)
for i in a:
    c[f(i)] += 1
res = 0
for i in c:
    res += c[i] * (c[i] - 1) / 2
print res
