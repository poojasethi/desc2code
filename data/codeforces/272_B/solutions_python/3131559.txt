def f(x):
    ret = 0
    while x:
        ret += (x & 1)
        x >>= 1
    return ret

n = int(raw_input())
arr = [int(i) for i in raw_input().split()]
arr.sort()
mp = {}
ret = 0
for i in arr:
    tmp = f(i)
    if tmp in mp:
        ret += mp[tmp]
    else:
        mp[tmp] = 0
    mp[tmp] += 1
print ret
