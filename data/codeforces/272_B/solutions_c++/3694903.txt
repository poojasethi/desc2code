#include <vector>
#include <iostream>

using namespace std;

const int MAX = (int)1e5 + 1;
typedef long long ll;

int tmp;
ll sol = 0;
int nbits[50] = {0};
int count;
int x;

int main() {
	int n;
	cin >> n;
	while(n--) {
		count = 0;
		cin >> x;
		while(x) count += (x & 1), x >>= 1;
		sol += nbits[count]++;
	}
	cout << sol;
}
