#include<iostream>
using namespace std;
long long s[50];
int main()
{
	long long n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		long long u;
		cin>>u;
		int o=0;
		while(u!=1)
		{
			if(u%2)
				o++;
			u/=2;
		}
		s[o]++;
	}
	long long sum=0;
	for(int i=0;i<50;i++)
	{
		sum+=s[i]*(s[i]-1)/2;
	}
	cout<<sum<<endl;
}
