#include <stdio.h>
int main()
{
	int n,i,x,y;
	__int64 num,a[100]={0};
	scanf("%d",&n);
	num=0;
	for(i=1;i<=n;i++){
		y=0;
		scanf("%d",&x);
		while(x){
			if(x%2)
				y++;
			x/=2;
		}
		a[y]++;
	}
	for(i=0;i<100;i++)
		num+=a[i]*(a[i]-1)/2;
	printf("%I64d\n",num);
	return 0;
}
