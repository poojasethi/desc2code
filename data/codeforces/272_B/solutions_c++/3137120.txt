#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    int a[32]={0};
    int n; cin>>n;
    for(int i=0;i<n;++i) {
        int t; cin>>t;
        a[__builtin_popcount(t)]++;
    }
    long long res=0;
    for(int i=0;i<32;++i)res+=a[i]*(a[i]-1LL)/2;
    cout<<res<<endl;
    return 0;
}
