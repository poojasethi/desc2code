#include<stdio.h>
#include<algorithm>
using namespace std;

int a[100];
int main()
{
	int n,m;
	int bit=0;
	long long sum=0;
	scanf("%d",&n);
	while (n--)
	{
		scanf("%d",&m);
        bit=0;
		while(m)
		{
			bit+=m&1;
			m>>=1;
		}
		sum+=a[bit]++;
	}
	printf("%I64d\n",sum);
        return 0;
}
