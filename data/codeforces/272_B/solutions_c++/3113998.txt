#include <iostream>
#include <stdio.h>
using namespace std;

long long int a[35];

int main()
{
	int n, x, y;
	cin>>n;

	for(int i=0;i<n;i++)
	{
		scanf("%d",&x);

		y = 0;
		while(x) y+=(x&1), x>>=1;
		a[y]++;
	}

	long long int result = 0;
	for(int i=1;i<35;i++) result += a[i]*(a[i]-1)/2;
	cout<<result;

	return 0;
}
