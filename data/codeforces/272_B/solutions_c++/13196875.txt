#include <bits/stdc++.h>
using namespace std;

int N, cnt[40];
long long ans;

main()
{
    scanf("%d", &N);
    for (int a; N--;)
    {
        scanf("%d", &a);
        cnt[__builtin_popcount(a)]++;
    }
    for (int i = 1; i <= 32; i++)
        ans += 1LL * cnt[i] * (cnt[i] - 1) / 2;
    printf("%I64d", ans);
}
