#include <iostream>
using namespace std;
long long xx[50];
long long a;
long long n;

int main()
{
cin >> n;
long long x = 0;
for(long long i=0;i<n;i++)
{
 cin >> a;
 x = 0;
 while(a>0){ x+=a%2; a/=2;}
 xx[x]+=1;
}
long long ans = 0;
for(int i=0;i<50;i++)
{
 ans+=xx[i]*(xx[i]-1)/2;
}
cout << ans << endl;
return 0;
}
