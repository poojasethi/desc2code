#include <iostream>
using namespace std;
int use[10000000];
int f(long long n)
{
	if(n==0) return 0;
	else if(n%2) return f((n-1)/2)+1;
	else return f(n/2);
}
int main()
{
	long long n,i,a,k,z=0;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>a;
		k=f(a);
		z+=use[k];
		use[k]++;
	}
	cout<<z<<endl;
	return 0;
}
