#include<iostream>
#include<cstdio>

using namespace std;
int main()
{
	int a[32]={0};
	int n,x;
	long long count=0;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		int bit=0;
		while(x)
		{
			
			bit+=x&1;
			x=x>>1;
		}
		count+=a[bit]++;
	 } 
	 cout<<count;
	return 0;	
}
