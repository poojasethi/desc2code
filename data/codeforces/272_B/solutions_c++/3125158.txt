#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int main() {
	int n;
	cin >> n;
	long long int sum=0;
	long long dp[32]={};
	int x;
	for(int i=0; i<n; i++) {
		cin >> x;
		dp[__builtin_popcount(x)]++;	
	}
	for(int i=0; i<32; i++) {
		sum += (dp[i]*(dp[i]-1))/2LL;
	}
	cout << sum << endl;
}
