#include <stdio.h>
int f(int x) {
 if (!x) return 0;
 if (x & 1) return f((x-1)>>1) + 1;
 else return f(x>>1);
}
int used[50];
int main () {
 int n;
 scanf("%d", &n);
 long long answer = 0;
 for(int i=0; i < n; i++) {
  int a;
  scanf("%d", &a); 
  answer += used[f(a)];
  used[f(a)]++;
 }
 printf("%I64d", answer);
}
