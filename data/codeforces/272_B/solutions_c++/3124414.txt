#include<iostream>
using namespace std;
int main(){
    int n,a,b,c[50];
    long long s=0;
    cin>>n;
    fill(c,c+50,0);
    for(int i=0;i<n;i++){
        cin>>a,b=0;
        while(a)
            b+=a%2,a/=2;
        c[b]++;
    }
    for(int i=0;i<50;i++)
        s+=(long long)c[i]*(c[i]-1)/2;
    cout<<s;
}
