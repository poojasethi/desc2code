#include <cstdio>
long long arr[50],cnt,ans;
int main(){
    int n,d;
    scanf("%d",&n);
    while(n--){
        scanf("%d",&d);
        cnt=0;
        while(d>0){
            if(d&1) cnt++;
            d/=2;
        }
     //   printf("%I64d ",cnt);
        ans+=arr[cnt]++;
    }
    printf("%I64d",ans);
   return 0;
}
