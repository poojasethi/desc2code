#include<cstdio>
#include<algorithm>
using namespace std;

int n, v;
int s[55];
long long ans;

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d", &v);
		s[__builtin_popcount(v)] ++;
	}
	ans = 0LL;
	for(int i=0;i<33;i++)
		ans += 1LL*s[i] * (s[i]-1)/2;
	printf("%I64d\n", ans);
	
	return 0;
}
