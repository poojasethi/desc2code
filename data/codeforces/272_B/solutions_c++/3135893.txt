#include<iostream>
using namespace std;

int T[10000],num,i,n,r;
long long nr;
int main ()
{
    cin>>n;
    for(i=1;i<=n;++i)
    {
        cin>>num; 
        r=0;
        while(num)
        {
            if(num%2)
                r++;
            num>>=1;
        }
        nr+=T[r];
        T[r]++;
    }
    cout<<nr;
    return 0;
}
