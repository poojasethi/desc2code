//In the name of God
#include <iostream>
using namespace std;

typedef long long ll;

ll a[40], ans;
int n;

int main() {
	cin.sync_with_stdio(false);
	cin >> n;
	for (int i = 0; i < n; i++) {
		int t;
		cin >> t;
		a[__builtin_popcount(t)]++;
	}
	for (int i = 0; i < 40; i++)
		ans += a[i] * (a[i] - 1) / 2;
	cout << ans << '\n';
	return 0;
}
