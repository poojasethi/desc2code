#include<cstdio>
int n,in,i;
long long cnt[30],ans;
int f(int n)
{
    if(n==0)return 0;
    return f(n/2)+n%2;
}
int main()
{
    scanf("%d",&n);
    while(n--)
    {
        scanf("%d",&in);
        cnt[f(in)]++;
    }
    for(i=1;i<30;i++)
    {
        ans+=cnt[i]*(cnt[i]-1)/2;
    }
    printf("%I64d\n",ans);
}
