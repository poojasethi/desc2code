const int MXL = 1010;

int len, sum;
char a[MXL];
inline void solve() {
	scanf("%s\n", a);
	len = strlen(a);
	for(int i = 0; i < len; i++) {
		if(a[i] == 'A') sum += 1;
		else if(a[i] >= '2') sum += a[i] - '0'; //
		else if (a[i] == '1') sum += 10;
	}
	printf("%d\n", sum);
}
int main() {
	//freopen("F.in","r",stdin);//freopen("F.out","w",stdout);
	solve();
	return 0;
