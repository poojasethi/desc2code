#include<stdio.h>
int main()
{
	int i;
	int sum;
	char iA[8];
	while(scanf("%s",&iA)!=EOF)
	{
		sum=1;
		for (i=1;i<7;i++)
		{
			if ((iA[i]=='1')&&(iA[i+1]=='0'))
			{
				sum=sum+10;
			}
			else
			{
				sum+=iA[i]-'0';
			}
		}
		printf("%d\n",sum);
	}
	return 0;
}
