#include<bits/stdc++.h>
#define INF ~0U>>1
using namespace std;

string s;

int main(){
	cin>>s;
	int ans=0;
	for(int i=0;i<7;i++){
		if(s[i]=='A')ans+=1;
		else if(s[i]=='J')ans+=11;
		else if(s[i]=='Q')ans+=12;
		else if(s[i]=='K')ans+=13;
		else if(s[i]=='1')ans+=10;
		else ans+=s[i]-'0';
	}
	cout<<ans;
	return 0;
}
