#include <bits/stdc++.h>

using namespace std;

int ans = 0;

int main() {
    string a;
    cin >> a;
    int tmp = 0;
    while(tmp != a.size()) {
        if(a[tmp] == 'A') {
            ans += 1;
            ++tmp;
            continue;
        }
        if(a[tmp] == '2') {
            ans += 2;
            ++tmp;
            continue;
        }
        if(a[tmp] == '3') {
            ans += 3;
            ++tmp;
            continue;
        }
        if(a[tmp] == '4') {
            ans += 4;
            ++tmp;
            continue;
        }
        if(a[tmp] == '5') {
            ans += 5;
            ++tmp;
            continue;
        }
        if(a[tmp] == '6') {
            ans += 6;
            ++tmp;
            continue;
        }
        if(a[tmp] == '7') {
            ans += 7;
            ++tmp;
            continue;
        }
        if(a[tmp] == '8') {
            ans += 8;
            ++tmp;
            continue;
        }
        if(a[tmp] == '9') {
            ans += 9;
            ++tmp;
            continue;
        }
        if(a[tmp] == '1') {
            ans += 10;
            ++tmp, ++tmp;
            continue;
        }
    }
    cout << ans << endl;
    return 0;
}
