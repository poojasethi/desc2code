#include <bits/stdc++.h>
using namespace std;

int main(){
    string str;
    cin>>str;
    int len=str.size();
    int total=0;

    for(int i=1;i<len;i++){
        int l=str[i]-48;
        if(str[i+1]=='0'){
            while(str[i+1]=='0'){
                l=(l*10);
                i++;
            }
            total+=l;
        }
        else{
            total+=l;
        }
    }
    cout<<total+1<<endl;


}
