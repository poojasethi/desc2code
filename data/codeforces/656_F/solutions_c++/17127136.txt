#include <bits/stdc++.h>
using namespace std;
char s[10];
int main() {
  int sum = 0;
  scanf("%s", s);
  int n = strlen(s);
  for (int i = 0; i < n; i++) {
    if (s[i] == '1') {
      sum += 10;
      i++;
    } else if (s[i] >= '2' && s[i] <= '9') {
      sum += s[i] - '0';
    }
  }
  printf("%d\n", sum + 1);
  return 0;
}
