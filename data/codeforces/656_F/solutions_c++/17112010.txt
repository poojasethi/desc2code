#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#define ll long long
using namespace std;
int main()
{
#ifndef ONLINE_JUDGE
    freopen("in.txt","r",stdin);
    //freopen("out.txt", "w",stdout);
#endif
    string s;
    while(cin>>s)
    {
        int len=s.length();
        int ans=1;
        for(int i=0;i<len;i++)
        {
            if(i){
                if(s[i]=='1')
                {
                    ans+=(s[i]-'0')*10+s[i+1]-'0';
                    i++;
                }
                else ans+=s[i]-'0';
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
