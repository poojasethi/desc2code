#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cstdio>
#include<string.h>
#include<map>
#include<queue>
#include<math.h>
#include<bitset>
using namespace std;
int main()
{
    char n;
    int ans=0;
    for(int i=0;i<7;i++)
    {
        scanf("%c",&n);
        if(n=='A')
        {
            ans+=1;
        }
        else if(n=='0')
        {
            ans+=9;
        }
        else
        {
            ans+=n-'0';
        }
    }
    printf("%d\n",ans);
    return 0;
}
