#include <bits/stdc++.h>
using namespace std;

typedef __int64 ll;
const int maxn=20;

char s[10];

int main() {
	while(~scanf("%s",s)){
		int res=0;
		for(int i=0;i<7;i++){
			if(s[i]=='A') res++;
			else if(s[i]=='J') res+=11;
			else if(s[i]=='Q') res+=12;
			else if(s[i]=='K') res+=13;
			else if(s[i]=='1') res+=10,i++;
			else res+=s[i]-'0';
		}
		printf("%d\n",res);
	}
	return 0;
}
