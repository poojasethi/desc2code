#include<bits/stdc++.h>
using namespace std;
int main(){
    char a[10];
    int i,b[10];
    scanf("%s",a);
    for (i=0;i<7;i++)
        if (a[i]=='A') b[i]=1;
        else b[i]=a[i]-'0';

    int sum=1;
    i=1;
    while (i<7){
        if (b[i]==1) {
            sum+=b[i]*10+b[i+1];
            i+=2;
        }
        else sum+=b[i++];
    }
    cout<<sum<<endl;
    return 0;
}
