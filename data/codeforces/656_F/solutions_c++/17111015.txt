void f() {
  char str[8];
  cin >> str;
  int i = 6;
  int sum = 1;
  while (i > 0) {
    if (str[i] == '0') {
      sum += 10;
      i-=2;
    }
    else {
      sum += str[i]-'0';
      i--;
    }
  }
  cout << sum << endl;
}

int main(int argc, const char * argv[]) {
  f();
  return 0;
}
