#include<iostream>
#include<cstring>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    char aa[10];
    scanf("%s",aa);
    int res=1;
    for(int i=1; i<strlen(aa); i++)
    {
        if(aa[i]=='1'&&aa[i+1]=='0')
        {
            res+=10;
            i++;
            continue;
        }
        else
            res+=((int)aa[i]-48);
    }
    cout<<res<<endl;
    return 0;
}
