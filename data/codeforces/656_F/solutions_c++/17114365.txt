#include <bits/stdc++.h>

typedef long long ll;
using namespace std;

int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  int i, sum = 0, x;
  string s;
  cin >> s;
  for (i = 1; i <= 6; i++) {
    x = s[i] - '0';
    if (x == 0)
      x = 9;
    sum += x;
  }
  cout << sum + 1 << '\n';
  return 0;
}
