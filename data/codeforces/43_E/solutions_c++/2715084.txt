#include<stdio.h>
#include<string.h>
int arr[110][30010];
int main()
{
	int n,s,i,j,k;
	int w,a,b,len,ans,mm;
	while(scanf("%d%d",&n,&s)!=EOF)
	{
		ans=0;mm=0;
		memset(arr,-1,sizeof(arr));
		for(i=0;i<n;i++)
		{
			scanf("%d",&w);
			arr[i][0]=0,len=1;
			while(w--)
			{
				scanf("%d%d",&a,&b);
				while(b--)
				{
					arr[i][len]=arr[i][len-1]+a;
					len++;
				}
				if(len>mm) mm=len;
			}
		}
		for(i=2;i<mm;i++)
		{
			for(j=0;j<n-1;j++)
			{
				if(arr[j][i]==-1) continue;
				for(k=j+1;k<n;k++)
				{
					if(arr[k][i]==-1) continue;
					if(arr[k][i]==arr[j][i])
					{
						if(arr[j][i+1]>arr[k][i+1]&&arr[j][i-1]<arr[k][i-1]) ans++;
						if(arr[j][i+1]<arr[k][i+1]&&arr[j][i-1]>arr[k][i-1]) ans++;
					}
					else 
					{
						if(arr[j][i]>arr[k][i]&&arr[j][i-1]<arr[k][i-1]) ans++;
						if(arr[j][i]<arr[k][i]&&arr[j][i-1]>arr[k][i-1]) ans++;
					}
				}
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
