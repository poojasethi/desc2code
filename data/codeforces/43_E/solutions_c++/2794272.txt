#include <cstdio>
int INFI = 99999999;
int k[200],sd[200][200],t[200][200],cnt,cnt2;
int main()
{
    //freopen("input.txt","r",stdin);
    //freopen("ouput.txt","w",stdout);
    int n,s,i,j,u,v,dt,s1,s2,time,state;
    scanf("%d%d",&n,&s);
    for(i=1;i<=n;i++)
    	for(j=1;j<=102;j++) t[i][j] = INFI;
    for(i=1;i<=n;i++)
    {
		scanf("%d",&k[i]);
		t[i][0] = 0;
		for(j=1;j<=k[i];j++)
		{
			scanf("%d%d",&sd[i][j-1],&dt);
			t[i][j] = t[i][j-1]+dt;
		}
	}
	for(u=1;u<=n;u++)
		for(v=u+1;v<=n;v++)
		{
			time = 0; i = j =0; s1 = s2 = 0; state = 0;
			while(s1 < s || s2 < s)
			{
				dt = t[v][j+1] < t[u][i+1] ? (t[v][j+1]-time) : (t[u][i+1]-time);
				s1 += dt*sd[u][i]; s2 += dt*sd[v][j];
				if(t[v][j+1] < t[u][i+1]) j++;
				else i++;
				time += dt; 
				if(s1 < s2)
				{
					if(state == 1) cnt++;
					state = -1;
				}
				else if(s1 > s2)
				{
					if(state == -1) cnt++;
					state = 1;	
				}
			}
		}
	printf("%d\n",cnt);
}
