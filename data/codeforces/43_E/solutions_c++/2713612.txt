#include<cstdio>
#include<cstring>
const int T=50000;
int c[100][T];
int main()
{
	int n,s,i,j,k,v,t,N;
	while(~scanf("%d%d",&n,&s))
	{
		memset(c,0,sizeof(c));
		N=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&k);
			j=0;
			while(k--)
			{
				scanf("%d%d",&v,&t);
				while(t--){j++;c[i][j]=c[i][j-1]+v;}
			}
			if(j>N)N=j;
		}
		int ans=0;
		for(i=2;i<=N;i++)
		{
			for(j=0;j<n-1;j++)
			{
				if(c[j][i]==0)continue;
				for(k=j+1;k<n;k++)
				{
					if(c[k][i]==0)continue;
					if(c[j][i]==c[k][i])
					{
						if(c[j][i-1]<c[k][i-1]&&c[j][i+1]>c[k][i+1])ans++;
					    if(c[j][i-1]>c[k][i-1]&&c[j][i+1]<c[k][i+1])ans++;
					}
					else
					{
						if(c[j][i-1]<c[k][i-1]&&c[j][i]>c[k][i])ans++;
					    if(c[j][i-1]>c[k][i-1]&&c[j][i]<c[k][i])ans++;
					}
				}
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
