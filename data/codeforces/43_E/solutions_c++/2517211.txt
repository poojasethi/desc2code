#include<cstdio>
int l[111],v[111][111],t[111][111];
int main(){
	int n,s,d0,d1,p0,p1,t0,t1,ti,ld,dt,cnt=0,i,j;
	scanf("%d%d",&n,&s);
	for (i=0;i<n;i++){
		scanf("%d",l+i);
		for (j=0;j<l[i];j++) scanf("%d%d",&v[i][j],&t[i][j]);
		for (j=0;j<i;j++){
			for (t0=t1=d0=d1=p0=p1=ti=0,ld=v[i][0]>v[j][0];d0!=s && d1!=s;){
				ti+=(dt=(t0+t[i][p0]<t1+t[j][p1]?t0+t[i][p0]:t1+t[j][p1])-ti);
				d0+=dt*v[i][p0],d1+=dt*v[j][p1];
				if (d0!=d1 && (ld^(d0>d1))) cnt++,ld=!ld;
				if (t0+t[i][p0]>t1+t[j][p1]) t1+=t[j][p1++]; else t0+=t[i][p0++];
			}
		}
	}
	printf("%d\n",cnt);
}
