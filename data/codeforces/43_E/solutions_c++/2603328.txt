#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,k,i,j,si,sj,fi,t,ans,now1,now2,c[101][101],v[101][101];
int main()
{
//	freopen("1.in","r",stdin);
//	freopen("1.out","w",stdout);
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&c[i][0]);
		for(j=1;j<=c[i][0];j++)
		scanf("%d%d",&v[i][j],&c[i][j]);
		for(j=2;j<=c[i][0];j++)
		c[i][j]+=c[i][j-1];
	}
	for(i=1;i<=n;i++)
		for(j=i+1;j<=n;j++)
		{
			//cout<<"     "<<i<<"        "<<j<<endl;
			for(si=sj=t=0,now1=1,now2=1,fi=-1;n;)
			{
		 		
				if (c[i][now1]<c[j][now2])
				{
					si+=(c[i][now1]-t)*v[i][now1];
					sj+=(c[i][now1]-t)*v[j][now2];
					t=c[i][now1];
					now1++;
				}	
				else
				{
					si+=(c[j][now2]-t)*v[i][now1];
					sj+=(c[j][now2]-t)*v[j][now2];
					t=c[j][now2];
					now2++;
				}
				if (si>sj)
				{
					if (fi==1) ans++;
					fi=0;
				}
				if (si<sj)
				{
					if (fi==0) ans++;
					fi=1;
				}
				if (now1>c[i][0]||now2>c[j][0]) break;
			}
		}
		cout<<ans;
}
