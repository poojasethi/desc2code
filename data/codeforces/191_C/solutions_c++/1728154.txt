#include<cstdio>
#include<cstring>
#define N 100005
int son[N],next[N*2],id[N*2],ed[N*2],ans[N];
int n,Q,x,y,z,l,f[N],d[N],c[N],num[N],sum[N],fa[N],top[N],heavy[N];
bool b[N];
void bfs()
{
	memset(d,0,sizeof(d));
	int h=0,t=1; f[1]=1; d[1]=1;
	while (h<t){
		int i=f[++h];
		for (int p=son[i];p;p=next[p]){
			int j=ed[p]; if (d[j]) continue;
			fa[j]=i; num[j]=id[p]; d[j]=d[i]+1; f[++t]=j;			
			}		
		}
	for (int k=n;k;k--){
		int i=f[k],q=0,m=0; sum[i]=1;
		for (int p=son[i];p;p=next[p]){
			int j=ed[p]; if (j==fa[i]) continue;
			if (sum[j]>m) m=sum[j],q=j; sum[i]+=sum[j];		
			}
		heavy[i]=q;
		}
	for (int k=1;k<=n;k++){
		int i=f[k]; if (b[i]) continue;
		for (int j=i;j;j=heavy[j]) b[j]=1,top[j]=i;
		}
}
int lca(int x,int y)
{
	while (top[x]!=top[y]){
		if (d[top[x]]<d[top[y]]){int tmp=x;x=y;y=tmp;}
		if (top[x]==x) x=fa[x];
		else x=top[x];		
		}
	return d[x]<d[y]?x:y;
}
int main()
{
	scanf("%d",&n); l=0;
	for (int i=1;i<n;i++){
		scanf("%d%d",&x,&y);
		ed[++l]=y; id[l]=i; next[l]=son[x]; son[x]=l;
		ed[++l]=x; id[l]=i; next[l]=son[y]; son[y]=l;
		}
	bfs(); scanf("%d",&Q);
	while (Q--){
		scanf("%d%d",&x,&y); z=lca(x,y);
		c[x]++; c[y]++; c[z]-=2;		
		}
	for (int k=n;k;k--){
		int i=f[k];
		ans[num[i]]=c[i];
		c[fa[i]]+=c[i];		
		}
	for (int i=1;i<n;i++)
		printf("%d%c",ans[i],i<n-1?' ':'\n');
	return 0;
}
