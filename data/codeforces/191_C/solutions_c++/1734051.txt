#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <vector>
using namespace std;

const int T = 100003;
const int rt = 1;
#define Cl(x,y) memset(x,y,sizeof(x))

int h[T],nxt[T*3],w[T*3],to[T*3],sb[T],s[T],cnt;
int pos[T],n,m;
int p[T],f[T*3];
bool vis[T],vv[T];
vector <int> a[T];

inline void add(const int x,const int y,const int z)
{
	nxt[cnt]=h[x],to[cnt]=y,w[cnt]=z,h[x]=cnt++;
}

inline int F(const int x)
{
	//printf("hjt: %d %d\n",x,f[x]);
	return f[x]==x?x:f[x]=F(f[x]);
}

inline void lca(const int u)
{
	//printf("%d %d\n",u,p[u]);
	//system("pause");
	f[u]=u;
	for(int i=h[u];~i;i=nxt[i])
		if(to[i]!=p[u] && !vis[to[i]])
			p[to[i]]=u,pos[to[i]]=w[i],lca(to[i]),f[to[i]]=u;
	vis[u]=1;
	int ss=a[u].size();
	for(int i=0;i<ss;i++)
		if(vis[a[u][i]])
		{
			int v=a[u][i],l=F(v);
			sb[v]++,sb[u]++,sb[l]-=2;
		}
}

inline void dfs(int u)
{
	for(int i=h[u];~i;i=nxt[i])
		if(to[i]!=p[u])
			dfs(to[i]),sb[u]+=sb[to[i]];
}
	
		

int main()
{
	Cl(h,-1);
	scanf("%d",&n);
	for(int i=1,j,k;i<n;i++)
		scanf("%d%d",&j,&k),add(j,k,i),add(k,j,i);
	
	
	
	scanf("%d",&m);
	while(m--)
	{
		int x,y; scanf("%d%d",&x,&y);
		a[x].push_back(y),a[y].push_back(x);
	}
	
	lca(rt);
	dfs(rt);
	
	
	//for(int i=1;i<=n;i++)
	//	printf("%d %d\n",i,sb[i]);
		
	for(int i=1;i<=n;i++)
		s[pos[i]]=sb[i];
	for(int i=1;i<n;i++)
		printf("%d ",s[i]);
	
	//system("pause");
	return 0;
}
