#include<bits/stdc++.h>
#define f first 
#define s second
using namespace std;
const int maxn=1e5+3;
int ans[maxn],mark[maxn],par[maxn][22],h[maxn];
vector<int>v[maxn];
pair<int,int> adj[maxn];
void dfs(int a)
{	
	mark[a]=1;
	for(int i=0;i<v[a].size();i++)
	{
		int child=v[a][i];
		if(!mark[child])
		{
			par[child][0]=a;
			h[child]=h[a]+1;
			dfs(child);
			ans[a]+=ans[child];
		}
	}
}			
int lca(int a,int b)
{
	int d=h[a]-h[b];
	for(int i=20;i>=0;i--)
		if(d & (1<<i))
			a=par[a][i];
	if(a==b)
		return a;
	for(int i=20;i>=0;i--)
		if(par[a][i]!=par[b][i])
			a=par[a][i],b=par[b][i];
	return par[a][0];
}
int main()
{
	int n,a,b;cin>>n;
	for(int i=0;i<n-1;i++)
	{
		cin>>a>>b;
		adj[i].f=a;
		adj[i].s=b;
		v[a].push_back(b);
		v[b].push_back(a);
	}
	h[1]=1;
	dfs(1);
	for(int i=1;i<=20;i++)
	{
		for(int j=1;j<=n;j++)
			par[j][i]=par[par[j][i-1]][i-1];//cout<<par[j][i];
		//cout<<endl;
	}
	int m;cin>>m;
	for(int i=0;i<m;i++)
	{
		cin>>a>>b;
		if(h[a]<h[b])
			swap(a,b);
		int p=lca(a,b);
	//	cout<<p;
		ans[a]++;
		ans[p]-=2;
		ans[b]++;
	}
	for(int i=1;i<=n;i++)
		mark[i]=0,h[i]=0;
	dfs(1);
	for(int i=0;i<n-1;i++)
	{
		if(h[adj[i].f]<h[adj[i].s])
			swap(adj[i].f,adj[i].s);
		cout<<ans[adj[i].f]<<" ";
	}
}
