#include <cstdio>
#include <vector>

using namespace std;

#define POW 17
#define N 1<<POW

int n, k, s[N], t[N], p[POW][N], * par = p[0], d[N], cnt[N];
vector<int> g[N];

void preprocess( int v )
{
	for( int i = 1; i < POW; i++ ) p[i][v] = p[i-1][p[i-1][v]];
	for( int i = 0; i < g[v].size(); i++ )
	{
		int u = g[v][i];
		if( u == par[v] ) continue;
		par[u] = v;
		d[u] = d[v] + 1;
		preprocess(u);
	}
}

int lca( int a, int b )
{
	if( d[a] > d[b] ) a ^= b, b ^= a, a ^= b;
	if( d[a] < d[b] )
	{
		int del = d[b] - d[a];
		for( int i = 0; i < POW; i++ ) if( del & ( 1 << i ) ) b = p[i][b];
	}
	if( a != b )
	{
		for( int i = POW-1; i >= 0; i-- ) if( p[i][a] != p[i][b] ) a = p[i][a], b = p[i][b];
		a = par[a], b = par[b];
	}
	return a;
}

void dfs( int v )
{
	for( int i = 0; i < g[v].size(); i++ )
	{
		int u = g[v][i];
		if( u == par[v] ) continue;
		dfs( g[v][i] );
		cnt[v] += cnt[u];
	}
}

int main()
{
	scanf( "%d", &n );
	for( int i = 0; i < n - 1; i++ )
	{
		int & a = s[i], & b = t[i];
		scanf( "%d%d", &a, &b );
		g[a].push_back(b), g[b].push_back(a);
	}
	preprocess(1);

	scanf( "%d", &k );
	for( int i = 0; i < k; i++ )
	{
		int a, b;
		scanf( "%d%d", &a, &b );
		int c = lca( a, b );
		cnt[a]++, cnt[b]++;
		cnt[c] -= 2;
	}

	dfs(1);
	for( int i = 0; i < n - 1; i++ ) printf( "%d ", cnt[ par[s[i]] == t[i] ? s[i] : t[i] ] );
	printf( "\n" );

	return 0;
}
