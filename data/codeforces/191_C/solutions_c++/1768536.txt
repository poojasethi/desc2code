#include <cstdio>
#define N 100005
#define fi(a, b, c) for(int a = (b); a < (c); a++)
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define fe(a, b) for(int a = (b); a; a = o[a])

int n, k, f[N], o[N * 4], v[N * 4], p;
int F[N], s[N], a[N], r[N], x, y;
bool u[N];

int fd(int x){
	return x == s[x] ? x : s[x] = fd(s[x]);
}

void bd(int F[], int a, int b){
	v[++p] = b; o[p] = F[a]; F[a] = p;
}

void dfs(int x){
	s[x] = x;
	fe(i, f[x]) if(!s[v[i]]){
		dfs(v[i]);
		a[x] += r[i + 1 >> 1] = a[v[i]];
		s[v[i]] = x;
	}
	u[x] = 1;
	fe(i, F[x]){
		a[x]++;
		if(u[v[i]]) a[fd(v[i])] -= 2;
	}
}

int main(){
	scanf("%d", &n);
	fi(i, 1, n){
		scanf("%d %d", &x, &y);
		bd(f, x, y);
		bd(f, y, x);
	}
	
	scanf("%d", &k);
	FI(i, 1, k){
		scanf("%d %d", &x, &y);
		if(x == y) continue;
		bd(F, x, y);
		bd(F, y, x);
	}
	
	dfs(1);
	fi(i, 1, n) printf("%d%c", r[i], i == n - 1 ? '\n' : ' ');
}
