#include <cstdio>
#include <algorithm>
#define N 200005
using namespace std;
int a,b,c,d,e,g,h,i,j,k,l,m,n,sum,tot;
int f[N],dis[N],T[N],fa[N],top[N],size[N],son[N];
int wa[N],ne[N],la[N],P[N],ans[N];
void make(int a,int b){
	wa[++tot]=b;ne[tot]=la[a];la[a]=tot;
	wa[++tot]=a;ne[tot]=la[b];la[b]=tot;
}
void calc(int a,int b){
	f[a]++;f[b+1]--;
}
void dfs(int x){
	size[x]=1;
	for(int i=la[x];i;i=ne[i])
	if(wa[i]!=fa[x]){
		fa[wa[i]]=x;
		dis[wa[i]]=dis[x]+1;
		dfs(wa[i]);
		size[x]+=size[wa[i]];
		if(size[wa[i]]>size[son[x]])son[x]=wa[i];
	}
	
}
void dfs2(int x,int Top){
	top[x]=Top;
	T[x]=++sum;
	P[sum]=x;
	
	if(son[x])dfs2(son[x],Top);
	for(int i=la[x];i;i=ne[i])
	if(wa[i]!=fa[x]&&wa[i]!=son[x]){
		dfs2(wa[i],wa[i]);
	}
}
int main(){
	scanf("%d",&n);
	for(i=1;i<n;i++){
		scanf("%d%d",&a,&b);
		make(a,b);
	}
	dfs(1);
	dfs2(1,1);
	scanf("%d",&m);
	for(;m;m--){
		scanf("%d%d",&c,&d);h=0;a=c;b=d;
		for(;top[a]!=top[b];){
			if(dis[top[a]]<dis[top[b]])swap(a,b);
			calc(T[top[a]],T[a]);
			a=fa[top[a]];
		}
		if(dis[a]>dis[b])swap(a,b);
		calc(T[a]+1,T[b]);
	}
	for(i=1;i<=n;i++)ans[P[i]]=ans[P[i-1]]+f[i];
	for(i=1;i<n;i++){a=wa[i*2];b=wa[i*2-1];if(dis[a]<dis[b])a=b;printf("%d ",ans[a]);}
	return 0;
}
