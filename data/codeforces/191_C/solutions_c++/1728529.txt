#include<iostream>
#include<cstdio>
#include<queue>
#include<cmath>
#include<algorithm>
using namespace std;
const int u=100010;
int f[u][20],d[u],v[u],ver[2*u],next[2*u],head[u],c[u],ans[u];
int n,m,tot,t,i,j,x,y,z;
queue<int> q;

void add(int x,int y)
{
	ver[++tot]=y; next[tot]=head[x]; head[x]=tot;
}

void bfs()
{
	q.push(1); v[1]=1; d[1]=1;
	while(q.size())
	{
		x=q.front(); q.pop();
		for(i=head[x];i;i=next[i])
			if(!v[y=ver[i]])
			{
				q.push(y); v[y]=1;
				d[y]=d[x]+1;
				f[y][0]=x;
				for(j=1;j<=t;j++) f[y][j]=f[f[y][j-1]][j-1];
			}
	}
}

int lca(int x,int y)
{
	if(d[x]>d[y]) swap(x,y);
	for(int i=t;i>=0;i--)
		if(d[f[y][i]]>=d[x]) y=f[y][i];
	if(x==y) return x;
	for(int i=t;i>=0;i--)
		if(f[x][i]!=f[y][i]) x=f[x][i],y=f[y][i];
	return f[x][0];
}

void dfs(int x)
{
	for(int i=head[x];i;i=next[i])
		if(f[ver[i]][0]==x)
		{
			dfs(ver[i]);
			ans[i+1>>1]=c[ver[i]];
			c[x]+=c[ver[i]]; 
		}
}

int main()
{
	cin>>n;
	t=(int)(log(n)/log(2)+0.1);
	for(i=1;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		add(x,y),add(y,x);
	}
	bfs();
	cin>>m; 
	for(i=1;i<=m;i++)
	{
		scanf("%d%d",&x,&y);
		c[x]++,c[y]++,c[lca(x,y)]-=2;
	}
	dfs(1);
	for(i=1;i<n;i++) printf("%d ",ans[i]);
	puts("");
	return 0;
}
