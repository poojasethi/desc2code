#include<stdio.h>
#include<string.h>

int fa[110000],ans[110000],ind[110000],first[110000],next[210000],vv[210000];
int first1[110000],next1[210000],vv1[210000];
int col[110000];

int fin(int n)
{
	if(n==fa[n])
		return n;
	return fa[n]=fin(fa[n]);
}
int vis[110000];
void dfs(int n,int e)
{
	vis[n]=1;
	ind[n]=e/2;fa[n]=n;
	for(e=first[n];e;e=next[e])if(!vis[vv[e]])
	{
		dfs(vv[e],e);
		ans[n]+=ans[vv[e]];
		fa[vv[e]]=n;
	}
	col[n]=1;
	for(e=first1[n];e;e=next1[e])
	{
		ans[n]++;
		if(col[vv1[e]])
			ans[fin(vv1[e])]-=2;
	}
}
int a[110000],n;
int main()
{
	int i,j,k,u,v,e=2;

	scanf("%d",&n);
	for(i=1;i<n;i++)
	{
		scanf("%d%d",&u,&v);
		next[e]=first[u],vv[e]=v,first[u]=e++;
		next[e]=first[v],vv[e]=u,first[v]=e++;
	}
	scanf("%d",&k);
	e=2;
	for(i=1;i<=k;i++)
	{
		scanf("%d%d",&u,&v);
		if(u==v)
			continue;
		next1[e]=first1[u],vv1[e]=v,first1[u]=e++;
		next1[e]=first1[v],vv1[e]=u,first1[v]=e++;
	}
	dfs((n+1)/2,0);
	for(i=1;i<=n;i++)
		a[ind[i]]=ans[i];
	for(i=1;i<n;i++)
		printf("%d%c",a[i],i==n-1?'\n':' ');
}
