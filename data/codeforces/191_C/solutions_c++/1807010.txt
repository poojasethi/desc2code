#include <fstream>
#include <iostream>
#include <vector>

#include <cstdio>

using namespace std;

#define N 100005
#define fe(a, b) for(int a = (b); a; a = o[a])

int n, k, f[N], o[N * 4], v[N * 4], p;
int F[N], s[N], a[N], r[N], x, y;

int fd(int x)
{
	return x == s[x] ? x : s[x] = fd(s[x]);
}

void bd(int F[], int a, int b)
{
	v[++p] = b;
	o[p] = F[a];
	F[a] = p;
}

void dfs(int x)
{
	s[x] = x;
	for (int i = F[x]; i; i = o[i])
	{
		a[x]++;
		if(s[v[i]])
			a[fd(v[i])] -= 2;
	}
	for (int i = f[x]; i; i = o[i])
		if(!s[v[i]])
		{
			dfs(v[i]);
			a[x] += r[i + 1 >> 1] = a[v[i]];
			s[v[i]] = x;
		}
}

int main()
{
	
	cin >> n;
	for (int i = 1; i < n; ++i)
	{
		cin >> x >> y;
		bd(f, x, y);
		bd(f, y, x);
	}
	
	cin >> k;
	for (int i = 1; i <= k; ++i)
	{
		cin >> x >> y;
		if(x == y) continue;
		bd(F, x, y);
		bd(F, y, x);
	}
	
	dfs(1);
	for (int  i = 1; i < n; ++i)
		printf("%d ", r[i]);
	return 0;
}
