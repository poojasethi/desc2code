//In His Name
#include<bits/stdc++.h>
using namespace std;
#define F first
#define S second
const int MAX=100100,MAXP=32;
int n,q;
vector<pair<int,int> >e;
vector<int>adj[MAX];
bool mark1[MAX],mark2[MAX];
int d[MAX],par[MAX][MAXP],sum[MAX],num[MAX];
void dfs1(int v){
	mark1[v]=true;
	for(int i=0;i<adj[v].size();i++){
		int u=adj[v][i];
		if(mark1[u]) continue;
		d[u]=d[v]+1;
		par[u][0]=v;
		for(int i=1;i<MAXP;i++)
			par[u][i]=par[par[u][i-1]][i-1];
		dfs1(u);
	}
	return;
}
int dfs2(int v){
	sum[v]=num[v];
	mark2[v]=true;
	for(int i=0;i<adj[v].size();i++)
		if(!mark2[adj[v][i]])
			sum[v]+=dfs2(adj[v][i]);
	return sum[v];
}
int parent(int x,int d){
	int ans=x;
	for(int i=0;i<MAXP;i++)
		if(((d>>i)&1))
			ans=par[ans][i];
	return ans;
}
int lca(int x,int y){
	if(d[x]>d[y]) swap(x,y);
	y=parent(y,d[y]-d[x]);
	if(x==y) return x;
	for(int i=MAXP-1;i>=0;i--)
		if(par[x][i]!=par[y][i]){
			x=par[x][i];
			y=par[y][i];
		}
	return par[x][0]; 
}
int main(){
	ios_base::sync_with_stdio(false);
	cin>>n;
	for(int i=0;i<n-1;i++){
		int x,y;
		cin>>x>>y;
		adj[x].push_back(y);
		adj[y].push_back(x);
		e.push_back(make_pair(x,y));
	}
	dfs1(1);
	cin>>q;
	while(q--){
		int x,y;
		cin>>x>>y;
		int z=lca(x,y);
		num[x]++;
		num[y]++;
		num[z]-=2;
	}
	dfs2(1);
	for(int i=0;i<e.size();i++){
		int x=e[i].F,y=e[i].S;
		if(d[x]<d[y]) swap(x,y);
		cout<<sum[x]<<endl;
	}
}
