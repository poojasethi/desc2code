#include <cstdio>
#include <iostream>
#include <vector>

#define fi first
#define se second
#define pb push_back

using namespace std;

typedef pair<int,int> ii;

const int MAXN=1e5+10;

int N,M;
int cnt[MAXN];
int dep[MAXN];
int par[MAXN][22];

ii e[MAXN];

vector<int> g[MAXN];

inline void dfs( int nd , int pre , int d ){
	
	dep[nd]=d;
	par[nd][0]=pre;
	
	for( vector<int>::iterator it=g[nd].begin() ; it!=g[nd].end() ; it++ )
		if( *it!=pre )
			dfs(*it,nd,d+1);
}

inline int dfs2( int nd ){
	
	for( vector<int>::iterator it=g[nd].begin() ; it!=g[nd].end() ; it++ )
		if( *it!=par[nd][0] )
			cnt[nd]+=dfs2(*it);
	
	return cnt[nd];
}

inline int lca( int a , int b ){
	
	if( dep[a]>dep[b] )
		swap(a,b);
	
	for( int i=0 ; i<22 ; i++ )
		if( ((dep[b]-dep[a])>>i)&1 )
			b=par[b][i];
	
	if( a==b )
		return a;
	
	for( int i=21 ; i>=0 ; i-- )
		if( par[a][i]!=par[b][i] ){
			a=par[a][i];
			b=par[b][i];
		}
	
	return par[a][0];
}

int main(){
	
	scanf("%d",&N);
	
	for( int a,b,i=1 ; i<N ; i++ ){
		scanf("%d%d",&a,&b);
		g[a].pb(b);
		g[b].pb(a);
		e[i]=ii(a,b);
	}
	
	dfs(1,0,0);
	
	for( int i=1 ; i<22 ; i++ )
		for( int j=1 ; j<=N ; j++ )
			par[j][i]=par[par[j][i-1]][i-1];
	
	scanf("%d",&M);
	
	for( int a,b,i=0 ; i<M ; i++ ){
		scanf("%d%d",&a,&b);
		cnt[a]++;
		cnt[b]++;
		cnt[lca(a,b)]-=2;
	}
	
	dfs2(1);
	
	for( int i=1 ; i<N ; i++ )
		if( dep[e[i].fi]>dep[e[i].se] )
			printf("%d ",cnt[e[i].fi]);
		else
			printf("%d ",cnt[e[i].se]);
	
	return 0;
}
