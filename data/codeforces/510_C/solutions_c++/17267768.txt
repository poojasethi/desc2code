#include<bits/stdc++.h>
using namespace std;

int n;
char s[102][102];
int in[26];
vector<int> v[26];

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i)
		scanf("%s", s[i]);
	for(int i=0 ; i+1<n ; ++i) {
		int j;
		for(j=0 ; s[i][j] && s[i+1][j] ; ++j) {
			if(s[i][j] != s[i+1][j]) {
				++in[s[i+1][j]-'a'];
				v[s[i][j]-'a'].push_back(s[i+1][j]-'a');
				break;
			}

		}
		if(s[i][j] && !s[i+1][j]) {
			puts("Impossible");
			return 0;
		}
	}
	queue<int> q;
	for(int i=0 ; i<26 ; ++i)
		if(!in[i]) q.push(i);
	int x ;
	string s = "";
	while(!q.empty()) {
		x = q.front();
		s+= x+'a';
		q.pop();
		for(int i=0 ; i<(int)v[x].size() ; ++i) {
			if(!--in[v[x][i]])
				q.push(v[x][i]);
		}
	}

	if(s.size() != 26) s = "Impossible";
	puts(s.c_str());
	return 0;
}
