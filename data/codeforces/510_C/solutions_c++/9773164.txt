#include <iostream>
#include <algorithm>
using namespace std;
string s[105];
int a[105][105];
int pos[105];
char rez[26];
int t;
int flag=true;
void fs(int x){
	
	if (pos[x]==1){
		flag=false;
		

		return;
	}
	else if (pos[x]==0){
		pos[x]=1;
		for (int i=1;i<=26;i++){
			if (a[x][i]==1){
				fs(i);
			}
		}
		pos[x]=2;
		t++;
		rez[t]='a'+x-1;
		
	}
	
}
int main(){
	int n;
	cin>>n;
	for (int i=1;i<=n;i++){
		cin>>s[i];
	}
	int f=0;
	for (int i=1;i<n;i++){
		int dx=s[i].size();
		int dy=s[i+1].size();
		for (int j=0;j<min(dx,dy);j++){
			if (s[i][j]!=s[i+1][j]){
				a[s[i][j]-'a'+1][s[i+1][j]-'a'+1]=1;
				f=1;
				break;
			}
		}
		if (f==0 && dx>dy){
			cout<<"Impossible";
			return 0;
		}
	}
	for (int i=1;i<=26;i++){
		if (pos[i]!=2){
			fs(i);
			
			if (flag==false){
				cout<<"Impossible";
				return 0;
			}
		}
	}
	for (int i=26;i>=1;i--){
		cout<<rez[i];
	}
	return 0;
}
