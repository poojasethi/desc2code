#include<iostream>
#include<string.h>
using namespace std;

int n;
int m[26][26];
int l[26];
char st[27],s[101],p[101];
int stn=0;

bool dfs(int v)
{
	l[v]=1;
	for(int i=0;i<26;i++)
		if(m[v][i])
			if(l[i]==1) return true;
			else
				if(l[i]==0)
					if(dfs(i)) return true;
	l[v]=2;
	st[stn++]=v+'a';
}

int main()
{
	int j;
	
	cin>>n;
	cin>>p;
	for(int i=1;i<n;i++)
	{
		cin>>s;
		for(j=0;s[j]==p[j];j++);
		if(s[j]==0)
		{
			cout<<"Impossible";
			return 0;
		}
		else
			if(p[j]!=0)
				 m[s[j]-'a'][p[j]-'a']=1;
			
		strcpy(p,s);
	}
	
	for(int i=0;i<26;i++)
		if(!l[i])
			if(dfs(i))
			{
				cout<<"Impossible";
				return 0;
			}
	
	st[26]=0;
	cout<<st;
	
	return 0;
}
