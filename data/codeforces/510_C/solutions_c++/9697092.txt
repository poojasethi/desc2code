#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

int n,m=0,b[200][2],dt[30][200]={0},l[200],du[30]={0}; 
char a[200][200],ans[30]={0};

int min(int x,int y)
{
	if(x<y) return x;
	return y; 
}

int main()
{
	int i,j;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		l[i]=strlen(a[i]);
	}
	for(i=1;i<n;i++)
	{
		int lt=min(l[i-1],l[i]);
		for(j=0;j<lt;j++)
			if(a[i-1][j] != a[i][j]) break;
		if(j>=lt)
		{
			if(l[i-1]>l[i])
			{
				cout<<"Impossible"<<endl;
				return 0;
			}
			else continue;
		}
		int t1=a[i-1][j]-'a',t2=a[i][j]-'a';
		du[t2]++;
		dt[t1][++dt[t1][0]]=t2;
	}
	int m=26;
	while(m)
	{
		for(i=0;i<26;i++) if(du[i]==0) break;
		if(i>=26) break;
		ans[++ans[0]]='a'+i;
		du[i]=-1;
		for(j=1;j<=dt[i][0];j++)
			du[dt[i][j]]--;
		m--;
	}
	if(m) cout<<"Impossible"<<endl;
	else {
		for(i=1;i<=26;i++)
			cout<<ans[i];
		cout<<endl;
	}
	return 0; 
}
