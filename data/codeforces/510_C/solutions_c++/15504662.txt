#include <bits/stdc++.h>
using namespace std;

#define pb                          push_back
#define forall(i,a,b)               for(int i=a;i<b;++i)

stack<int> ms;
int edges[30][30],col[30],temp;

bool tpSort(int x){
	col[x]=1;
	forall(i,0,26){
		if(edges[x][i]==0) continue;
		if(col[i]==1) return false;
		else if(col[i]==0){
			if(!tpSort(i)) return false;
		}
	}
	col[x]=2;
	ms.push(x);
	return true;
}

int main(){
	int n,m,x,y; cin >> n;
	string a,b; cin >> a;
	forall(i,1,n){
		bool happ=false;
		cin >> b;
		forall(i,0,min(a.size(),b.size())){
			if(a[i]!=b[i]){
				edges[a[i]-'a'][b[i]-'a']=1;
				happ=true;
				break;
			}
		}
		if(!happ && a.size()>b.size()){
			cout << "Impossible" << endl;
			return 0;
		}
		a=b;
	}
	forall(i,0,26){
		if(col[i]==2) continue;
		if(!tpSort(i)){
			cout << "Impossible" << endl;
			return 0;
		}
	}
	while(!ms.empty()){
		cout << (char)(ms.top()+'a');
		ms.pop();
	}
	cout << endl;
}
