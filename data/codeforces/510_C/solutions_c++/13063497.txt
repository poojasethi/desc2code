//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

int n,L;
string s,t,ans="";
vector<int> g[32];
bool vis[32],mark[32],Ans;

void dfs(int v)
{
  vis[v]=mark[v]=true;
  for(int i=0;i<g[v].size();i++)
    if(!vis[g[v][i]])
      dfs(g[v][i]);
    else if(mark[g[v][i]])
      Ans=true;
  ans=(char)(v+'a')+ans;
  mark[v]=false;
}

int main()
{
  cin>>n;
  cin>>s;
  n--;
  while(n--)
    {
      cin>>t;
      L=min(s.size(),t.size());
      for(int i=0;i<L;i++)
	if(s[i]!=t[i])
	  {
	    g[s[i]-'a'].push_back(t[i]-'a');
	    break;
	  }
	else if(i==L-1 && s.size()>t.size())
	  Ans=true;
      s=t;
    }
  for(int i=25;i>=0;i--)
    if(!vis[i])
      dfs(i);
  if(!Ans)
    cout<<ans<<endl;
  else
    cout<<"Impossible"<<endl;
  return 0;
}
