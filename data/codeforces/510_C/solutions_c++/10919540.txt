#include<stdio.h>
using namespace std;
int n;
bool p, o[300], g[300][300];
char i, j, k, a[300][300], b[300];
int main(){
    scanf("%d\n", &n);
    for (i=0; i<n; i++)
        for (j=0, scanf("%c", &a[i][j]); a[i][j]!='\n'; j++, scanf("%c", &a[i][j]));
    for (i=0; i<n-1; i++)
        for (j=i+1; j<n; j++){
            for (k=0; a[i][k]==a[j][k]; k++);
            g[a[i][k]][a[j][k]]=true;
        }
    n=0;
    for (i=0; i<26; i++)
        for (j='a'; j<='z'; j++){
            p=true;
            for (k='a'; k<='z'; k++) if (g[k][j]) p=false;
            if (p && !o[j]){
                o[j]=true;
                b[n]=j;
                n++;
                for (k='a'; k<='z'; k++) g[j][k]=false;
            }
        }
    p=false;
    for (i='a'; i<='z'; i++) if (g[i]['\n']) p=true;
    if (n!=26 || p) printf("Impossible");
        else
            for (i=0; i<n; i++) printf("%c", b[i]);
    return 0;
}
