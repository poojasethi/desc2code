#include<iostream>
#include<algorithm>
using namespace std;

int n,v[26];
bool adj[26][26];
string s,t,ans;

void dfs(int u){
	v[u]=1;
	for(int i=0; i<26; i++){
		if(adj[u][i]&&v[i]==1){
			cout << "Impossible\n";
			exit(0);
		}
		if(adj[u][i]&&v[i]!=-1)dfs(i);
	}
	v[u]=-1;
	ans=ans+(char)(u+'a');
}

int main(){
	cin >> n; 
	cin >> s;
	for(int j=1; j<n; j++){
		cin >> t; bool check=1;
		for(int i=0; i<t.length()&&i<s.length(); i++){
			if(s[i]!=t[i]){check=0; adj[t[i]-'a'][s[i]-'a']=1; break;}
		}
		if(check&&s.length()>t.length()){cout << "Impossible\n"; return 0;}
		s=t;
	}
	for(int i=0; i<26; i++){
		if(v[i]!=-1)dfs(i);
	}
	cout << ans << endl;
	return 0;
}
