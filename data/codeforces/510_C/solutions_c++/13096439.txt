#include<iostream>
#include<algorithm>
#include<string>
#include<vector>
#include<set>
using namespace std;
int j;
string n[110];
int k;
int ans[33];
int tot;
bool q=true;
bool v[33],w[33];
vector<int> a[33];
void dfs(int x)
{
	v[x]=1;
	w[x]=1;
	for(int i=0;i<a[x].size();i++)
	{
		int p;
		p=a[x][i];
		if(!w[p]) dfs(p);
		else if(v[p]) q=false;
		
	}
	v[x]=0;
	ans[tot++]=x;
}
int main()
{
	cin>>k;
	for(int i=0;i<k;i++) cin>>n[i];
	for(int i=1;i<k;i++)
	{
		int l1,l2;
		l1=n[i].length();
		l2=n[i-1].length();
		for(j=0;j<l1&&j<l2&&n[i][j]==n[i-1][j];j++);
		if(j<l1&&j<l2) a[n[i][j]-'a'].push_back(n[i-1][j]-'a');
		else if(l2>l1) q=false;
	}
	for(int i=0;i<26;i++)
	{
		if(!w[i]) dfs(i);
	}
	if(q)
	{
		for(int i=0;i<26;i++)
		cout<<char(ans[i]+'a');
	}
	else cout<<"Impossible";
	return 0;
}
