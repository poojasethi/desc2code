#include <map>
#include <set>
#include <queue>
#include <cstdio>
#include <vector>
#include <string>
#include <sstream>
#include <iostream>
#include <algorithm>
using namespace std;
int n;
bool rc=true;
string s[105];
queue <int> q;
vector <int> nei[200];
int in[200];
string ans="";
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	cin>>s[i];
	for(int i=1;i<=n-1;i++)
	for(int k=0;k<s[i].length();k++)
	{
	if(k>=s[i+1].length())
	rc=false;
	if(s[i][k]!=s[i+1][k])
	{
	nei[(int)s[i][k]].push_back((int)s[i+1][k]);
	in[(int)s[i+1][k]]++;
	break;
	}
	}
	for(int i=(int)'a';i<=(int)'z';i++)
	if(!in[i])
	q.push(i);
	while(!q.empty())
	{
	int a=q.front();
	q.pop();
	for(int i=0;i<nei[a].size();i++)
	{
	in[nei[a][i]]--;
	if(!in[nei[a][i]])
	q.push(nei[a][i]);
	}
	ans+=(char)a;
	}
	if(ans.length()==26)
	cout<<ans;
	else cout<<"Impossible";
	return 0;
}
