#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
char s[100][101];
vector<int>v[26];
int a1[26],o[26],lo;
bool f(int a)
{
    a1[a]=1;
    for(int i=0;i<v[a].size();++i)if(a1[v[a][i]]<2)
    if(a1[v[a][i]]==1||f(v[a][i]))return 1;
    a1[a]=2;o[lo++]=a;return 0;
}
int main()
{
    int n,a;
    scanf("%d%s",&n,s[0]);
    for(int i=1;i<n;++i)
    {
        scanf("%s",s[i]);
        for(a=0;s[i][a]&&s[i][a]==s[i-1][a];++a);
        if(s[i-1][a])
        {
            if(!s[i][a]){puts("Impossible");return 0;}
            v[s[i][a]-97].push_back(s[i-1][a]-97);
        }
    }
    for(int i=0;i<26;++i)if(!a1[i]&&f(i))
    {puts("Impossible");return 0;}
    for(int i=0;i<26;++i)putchar(o[i]+97);
}
