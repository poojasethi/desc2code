#include<bits/stdc++.h>

using namespace std;

int n, i, j, x, y, in[26];
vector<int> v[26];
char a[101], b[101];

void no() {
	puts("Impossible");
	exit(0);
}

int main(int argc, char **argv) {
	scanf("%d", &n);
	scanf("%s", a);
	for (i = 1; i < n; ++i) {
		scanf("%s", b);
		for (j = 0; a[j] && b[j]; ++j) {
			if (a[j] != b[j]) {
				x = a[j] - 'a';
				y = b[j] - 'a';
				v[x].push_back(y);
				++in[y];
				break;
			}
		}
		if (a[j] && !b[j]) no();
		strcpy(a, b);
	}
	queue<int> q;
	for (int i = 0; i < 26; ++i)
		if (!in[i]) q.push(i);
	string ans = "";
	while (!q.empty()) {
		x = q.front(), q.pop();
		ans += (x + 'a');
		for (int i = 0; i < (int) v[x].size(); ++i) {
			--in[v[x][i]];
			if (in[v[x][i]] == 0)
				q.push(v[x][i]);
		}
	}
	ans.size() == 26 ? cout << ans << endl : cout << "Impossible" << endl;
	return 0;
}
