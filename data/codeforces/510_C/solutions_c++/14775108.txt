#include <bits/stdc++.h>
using namespace std;
string s, p, ans;
int j, n, G[26][26], a[27], c[26];
bool dfs(int v){
    c[v]=1;
    for(int i=0; i<26; i++){
        if(G[v][i])
            if(c[i]==1)return 1;
            else if(c[i]==0) if(dfs(i)) return 1;
    }
    c[v]=2;
    ans+=(v+'a');
    return 0;
}
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> n;
    cin >> s;
    for(int i=1; i<n; i++){
        cin >> p;
        for(j=0; s[j]==p[j]; j++);
        if(p[j]==0){
            cout << "Impossible"; return 0;
        }
        else{
            if(s[j]){
                G[p[j]-'a'][s[j]-'a']=1;
            }
        }
        s=p;
    }
    for(int i=0; i<26; i++){
        if(!c[i])
        if(dfs(i)){
            cout << "Impossible"; return 0;
        }
    }
    cout << ans;
}
