

#include <iostream>
#include <cstdio>
using namespace std;

int d[30][30];
char c[30],t;
int main() {
	int n,i,j,k;cin>>n;
	char s[200][200];
	for(i=0;i<n;i++)	scanf("%s",s[i]);
	for(i=1;i<n;i++){
		for(j=0;s[i][j]&&s[i][j]==s[i-1][j];j++);
		if(s[i][j]&&s[i-1][j])	d[s[i][j]-'a'][s[i-1][j]-'a']=1;
		if(!s[i][j]&&s[i-1][j]){
			cout<<"Impossible"<<endl;
			return 0;
		}
	}
	for(i=0;i<26;i++)	c[i]=i+'a';
	for(i=0;i<500;i++)for(j=0;j<26;j++)for(k=j+1;k<26;k++)	if(d[c[j]-'a'][c[k]-'a'])
	{
		t=c[j];
		c[j]=c[k];
		c[k]=t;
	}
	for(j=0;j<26;j++)for(k=j+1;k<26;k++)if(d[c[j]-'a'][c[k]-'a']){
		cout<<"Impossible"<<endl;
		return 0;
	}
	cout<<c<<endl;
	return 0;
}
