#include<cstdio>

int n, l, r, cur;
int e[33][33], tot[33], q[33];
char s[111][111];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) scanf("%s",s[i]);
	
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++) {
			for(int k=0;s[i][k];k++) {
				if (!s[j][k]) {
					printf("Impossible\n");
					return 0;
				}
				if(s[i][k] != s[j][k]) {
					int x = s[i][k] - 'a', y = s[j][k] - 'a';
					if (!e[x][y]) tot[y] ++;
					e[x][y] = 1;
					break;	
				} 
			}
		}
	l = r = 0;
	for(int i=0;i<26;i++)
		if(!tot[i]) q[r++] = i;
	while(l<r) {
		cur = q[l++];
		for(int i=0;i<26;i++)
			if(e[cur][i]) {
				tot[i] --;
				if(!tot[i]) q[r++] = i;
			}
	}
	if(r<26) printf("Impossible\n");
	else {
		for(int i=0;i<26;i++)
			printf("%c", q[i] + 'a');
		printf("\n");
	}
	
	return 0;
}
