#include<bits/stdc++.h>
using namespace std;
int m[27][27];
int l[27],n;
int j,stn=0;
char s[103],p[103],st[27];
bool dfs(int v)
{
    l[v]=1;
    for (int i=0;i<26;i++)
        if (m[v][i])
            if (l[i]==1)
                return 1;
            else if (l[i]==0)
                    if (dfs(i)) return 1;
    l[v]=2;
    st[stn++]=v+'a';
}
main()
{
    cin>>n;
    cin>>p;
    for (int i=1;i<n;i++)
    {
        cin>>s;
        for (j=0;s[j]==p[j];j++);
        if (s[j]==0)
            {
                cout<<"Impossible";
                return 0;
            }
        else if (p[j]!=0)
            m[s[j]-'a'][p[j]-'a']=1;
        strcpy(p,s);
    }
    for (int i=0;i<26;i++)
        if (!l[i])
            if (dfs(i))
                {
                    cout<<"Impossible";
                    return 0;
                }
    st[26]=0;
    cout<<st;
}
