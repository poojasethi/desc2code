#include<cstdio>
#include<cstring>
#include<algorithm>
char a[26],s[102400],*p,w,x,y;
int l,z[26],i,ans,n,j,t;
int main(){
	l=strlen(gets(s+1));
	for(i=1;i<=l;++i)s[i]-='a';
	scanf("%d",&n);
	memset(a,0xff,sizeof(a));
	while(n--){
		scanf(" %c%c",&x,&y);
		x-='a',y-='a';
		a[x]=y;a[y]=x;
	}
	ans=0;
	memset(z,0,sizeof(z));
	for(i=1;i<=l;++i){
		for(j=t=0;j<26;++j)
		if(j!=a[s[i]])
			t=std::max(t,z[j]);
		ans=std::max(ans,(z[s[i]]=t+1));
	}
	printf("%d\n",l-ans);
	return 0;
}
