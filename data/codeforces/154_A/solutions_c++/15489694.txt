#include <iostream>
#include <string>
using namespace std;


bool forbid[30][30];
int n;
string in;
int k;
string tm;
int main(){
	cin>>in;
	n=in.length();
	cin>>k;
	long long sol=0;
	for(int i=0;i<k;i++){
		cin>>tm;
		int a=0;
		int b=0;
		for(int i=0;i<n;i++){
			if(tm[0]==in[i]){
				a++;
			} else if(tm[1]==in[i]){
				b++;
			} else {
				sol+=min(a,b);
				a=0;
				b=0;
			}
		}
		sol+=min(a,b);
	}

	cout<<sol<<endl;
}
