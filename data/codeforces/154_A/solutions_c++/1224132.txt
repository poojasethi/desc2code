#include<cstdio>
#include<cstring>
int i,j,k,l,Ans,f[100005],g[128];
char S[100005],s[10];
bool a[128][128];
int main()
{
	gets(S);
	l=strlen(S);
	scanf("%d",&k),gets(s);
	for(i=1;i<=k;i++)
	{
		gets(s);
		a[s[0]][s[1]]=a[s[1]][s[0]]=1;
	}
	Ans=l;
	for(i=0;i<l;i++)
	{
		f[i]=1;
		for(j='a';j<='z';j++)
			if(!a[j][S[i]])
				if(g[j]+1>f[i])
					f[i]=g[j]+1;
		if(l-f[i]<Ans) Ans=l-f[i];
		if(f[i]>g[S[i]]) g[S[i]]=f[i];
	}
	printf("%d\n",Ans);
	return 0;
}
