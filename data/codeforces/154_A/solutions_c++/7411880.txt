#include<algorithm>
#include<cstring>
#include<iostream>
using namespace std;
char s[100005];
int k;
char a,b;
int main()
{
	int ans=0;
	int l,r;
	cin>>s;
	cin>>k;
	for(int i=0;i<k;i++)
	{
		cin>>a;
		cin>>b;
		for(int j=0;j<strlen(s);j++)
		{
			l=r=0;
			while(s[j]==a||s[j]==b)
			{
				if(s[j]==a)	l++;
				if(s[j]==b)	r++;
				j++;
			}
			ans+=min(l,r);
		}
	}
	cout<<ans<<endl;
	return 0;
}
