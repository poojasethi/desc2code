#include<iostream>
#include<string>
using namespace std;
#define min(a,b) a<b?a:b
string s,t;
int main()
{
	int n,i,j;
	while(cin>>s)
	{
		cin>>n;int aa=0,bb=0;int ans=0;
		for(i=1;i<=n;i++)
		{
		    cin>>t;
		    for(j=0;s[j];j++)
		    {
		    	if(s[j]==t[0]) aa++;
		    	else 
	    	           if(s[j]==t[1]) bb++;
	    	           else {
			       ans+=min(aa,bb);
			       aa=0,bb=0;
	    	           }
		    }	
		   
		} ans+=min(aa,bb);
		cout<<ans<<endl;
	}
}
