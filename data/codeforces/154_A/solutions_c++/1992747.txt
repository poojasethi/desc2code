#include<iostream>
#include<string.h>
#include<algorithm>
using namespace std;

string s;char a,b;
int i,j,l,x,y,n,k,su;

int main()
{
    cin>>s>>n;
    l=s.size();
    for(i=1;i<=n;++i)
    {
        cin >> a >> b;
        for(j=0;j<l;++j)
        {
            x=0;y=0;
            for(;s[j]==a || s[j]==b;++j)
                s[j]==a?++x:++y;
            su+=min(x,y);
        }
    }
    cout <<su<<endl;
    return 0;
}
