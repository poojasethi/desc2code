#include<iostream>
#include<string.h>
#include<algorithm>
#include<stdio.h>
#include<string>
using namespace std;

int main()
{
	string s;
	cin>>s;
	int n;
	cin>>n;
	int ans=0,x,y;
	int l=s.size();
	char a,b;
	for(int i=0;i<n;i++)
	{
		cin>>a>>b;
		int j;
		for(j=0;j<l;j++)
		{
			x=y=0;
			for(;s[j]==a||s[j]==b ;j++)
			{
				if(s[j]==a) x++;
				else if(s[j]==b) y++;
			}
			ans+=min(x,y);
		}
	}
	printf("%d\n",ans);
}
