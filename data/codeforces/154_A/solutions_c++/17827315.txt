#include<iostream>
#include<cstdio>
#include<cstring>
#define Min(a,b) (a>b?b:a)
using namespace std;
int main()
{
	char str[100010];
	gets(str);
	int k;
	scanf("%d",&k);
	int ans = 0;
	for (int i = 0; i < k; i++)
	{
		char a,b;
		scanf(" %c%c",&a,&b);
		int j = 0;
		while(str[j])
		{
			int l = 0,r = 0;
			while(str[j] == a || str[j] == b)
			{
				if(str[j] == a) l++;
				else r++;
				j++;
			}
			ans += Min(l,r);
			j++;
		}
	}
	printf("%d\n",ans);
 }
