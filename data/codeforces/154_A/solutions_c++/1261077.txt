#include <iostream>
#include <string>
using namespace std;
int i, j, k, ans, k1, k2;
int main() {
    char a, b;
    string s;
    cin >> s >> k;
    for (; i < k; ++i) {
        cin >> a >> b;
        for (j = 0; j < s.size(); ++j) {
            for (; s[j] == a || s[j] == b; ++j)
                if (s[j] == a)
                    ++k1;
                else
                    ++k2;
            ans += min(k1, k2), k1 = 0, k2 = 0;
        }
    }
    cout << ans;
    return 0;
}
