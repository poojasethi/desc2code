#include <bits/stdc++.h>
using namespace std;

int main(){
  string str;
  int n, ret = 0;
  cin >> str >> n;
  while(n--){
    int c0, c1;
    string tmp;
    cin >> tmp;
    c0 = c1 = 0;
    for (int i = 0; i < str.size(); i++){
      if (str[i] == tmp[0])
        c0++;
      else if (str[i] == tmp[1])
        c1++;
      else
        ret += min(c0, c1), c0 = c1 = 0;
    }
    ret += min(c0, c1);
  }
  cout << ret << endl;
}
