#include<iostream>
#include<string.h>
#include<algorithm>
using namespace std;

char s[100100],a,b;
int i,j,l,x,y,n,k,ans;

int main()
{
	cin >> s >> n;
	l=strlen(s);
	for(i=1;i<=n;++i)
	{
		cin >> a >> b;
		for(j=0;j<l;++j)
		{
			x=0;y=0;
			for(;s[j]==a || s[j]==b;++j)
				s[j]==a ? ++x : ++y;
			ans+=min(x,y);
		}
	}
	cout << ans << endl;
	return 0;
}
