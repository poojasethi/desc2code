#include<bits/stdc++.h>
using namespace std;
int main()
{
string s,v; cin>>s;
int k; cin>>k;
int ret=0,c1=0,c2=0;
for(int i=0;i<k;i++)
{
    cin>>v;
int ind=0;
while(ind<s.size())
{
 while((ind<s.size())&&(s[ind]!=v[0]&&s[ind]!=v[1]))
    ind++;

c1=0;c2=0;
while((ind<s.size())&&(s[ind]==v[0]||s[ind]==v[1]))
{
    if(s[ind]==v[0])c1++;
    else c2++;
    ind++;
}

ret+=min(c1,c2);
}

}
cout<<ret<<endl;
return 0;}
