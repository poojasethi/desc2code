#include<bits/stdc++.h>
using namespace std;
string s;
char a,b;
int ans,i,n,l;
main()
{
    cin>>s>>n;
    l=s.size();
    for (i=0;i<n;i++)
    {
        cin>>a>>b;
        for (int j=0;j<l;j++)
        {
            int x=0,y=0;
            for (;s[j]==a||s[j]==b;j++)
                s[j]==a?x++:y++;
            ans+=min(x,y);
        }
    }
    cout<<ans;
}
