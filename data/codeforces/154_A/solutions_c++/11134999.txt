#include<bits/stdc++.h>
using namespace std;
int main()
{
	int k,i,ans=0,j,n,a,b;
	string s,s1;
	ios_base::sync_with_stdio(false);
	cin>>s;
	n=s.length();
	cin>>k;
	for(j=0;j<k;j++)
	{
		cin>>s1;
		a=b=0;
		for(i=0;i<n;i++)
		{
			if(s[i]==s1[0])
				a++;
			else if(s[i]==s1[1])
				b++;
			else
			{
				ans+=min(a,b);
				a=b=0;
			}		
		}
		ans+=min(a,b);
	}
	cout<<ans<<endl;	
	return 0;
}
