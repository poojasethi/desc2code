#include <cstdio>
#include <cstring>

#define MAX 100010

char s[MAX], sub[3];
int a[MAX];
int n, res, len, x, y;

int main(){
	scanf("%s", s);
	len = strlen(s);
	scanf("%d", &n);
	for (int i = 0; i < n; i++){
		scanf("%s", sub);
		
		for (int j = 0; s[j]; j++){
			x = 0; y = 0;
			while (s[j] == sub[0] || s[j] == sub[1]){
				if (s[j] == sub[0])
					x++;
				else
					y++;
				j++;
			}
			res += x < y ? x : y;
		}
	}
	
	printf("%d", res);
	return 0;
}
