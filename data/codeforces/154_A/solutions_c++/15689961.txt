#include<cstdio>
#include<cstring>
char s[100005],str[5];
int main()
{
	int	i,n,j;
	scanf("%s",s);
	int len=strlen(s);
	scanf("%d",&n);
	int ans=0;
	for(i=1;i<=n;i++)
	{
		scanf("%s",str);
		j=0;
		while(j<len)
		{
			int cnt1=0,cnt2=0;
			while(s[j]==str[0]||s[j]==str[1])
			{
				if(s[j]==str[0])
					cnt1++;
				else
					cnt2++;
				j++;
			}
			j++;
			ans=ans+(cnt1<cnt2?cnt1:cnt2);
		}
	}
	printf("%d\n",ans);
	return 0;
}
