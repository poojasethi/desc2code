#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char c[100001];
	cin>>c;
	int n=strlen(c),k,i=0,count=0;
	cin>>k;
	char d[k][2];
	for(int i=0;i<k;i++)cin>>d[i];
	for(int j=0;j<k;j++)
	{
		int i=0;
		while(i<n)
		{
			int a=0,b=0;
			while(c[i]==d[j][0] or c[i]==d[j][1] and i<n)
			{
				if(c[i]==d[j][0])a++;
				else b++;
				i++;
			}count+=min(a,b);i++;
		}
	}
	cout<<count;
}
