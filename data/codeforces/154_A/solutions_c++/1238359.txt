#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[100005];
int l, n;

int main() {
	scanf("%s%d%*c", s, &n);
	l = strlen(s);
	int ans = 0;
	for (int i = 1; i <= n; ++i) {
		char a, b;
		scanf("%c%c%*c", &a, &b);
		for (int j = 0; j < l; ++j) {
			int c1 = 0, c2 = 0;
			for (; s[j]==a || s[j]==b; ++j)
				s[j]==a ? c1++ : c2++;
			ans += min(c1, c2);
		}
	}
	printf("%d\n", ans);
}
