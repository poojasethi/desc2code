#include <iostream>
#include <string>

#define FOR(i,a,b) for (int i=a;i<=b;++i)

using namespace std;

int k,l,m,n=0;
string s,x;

int main()
{
    cin>>s>>k;
    s.push_back('_');
    while (k--)
    {
        cin>>x;
        l=m=0;
        FOR(i,0,s.size()-1)
            if (s[i]==x[0]) ++l;
            else if (s[i]==x[1]) ++m;
            else
            {
                n+=l>m?m:l;
                l=m=0;
            }
    }
    cout<<n<<endl;
    return 0;
}
