#include <cstdio>
#include <cstring>

char s[100010];
int n, k, an, bn;
char a, b;
int ans;

int main() {
  scanf("%s", s);
  n = strlen(s);
  scanf("%d", &k);
  ans = 0;
  for (int i = 0; i < k; ++i) {
    scanf(" %c%c", &a, &b);
    an = bn = 0;
    for (int j = 0; j <= n; ++j)
      if (s[j] == a)
	an++;
      else if (s[j] == b)
	bn++;
      else
	ans += an < bn ? an : bn, an = bn = 0;
  }

  printf("%d\n", ans);
  return 0;
}
