#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	cin>>s;
	int p;
	cin>>p;
	string t;
	int total=0;
		int a=0,b=0;
	for(int j=0;j<p;j++)
	{
		cin>>t;


	for(int i=0;i<s.length();i++)
	  {
	  	if(s[i]==t[0])
		{
			a++;
		}
		else
			if(s[i]==t[1])
		{
			b++;
		}
		else
		{
			total+=min(a,b);
			a=0;
			b=0;
		}
	  }
	  total+=min(a,b);
	  a=0;
	  b=0;
	}
	cout<<total<<endl;
}
