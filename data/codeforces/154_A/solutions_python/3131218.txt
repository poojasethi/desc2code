s = raw_input().strip()
s += '_'
k = int(raw_input())
ans = 0
for a in range(k):
    t = raw_input()
    cnt = [0, 0]
    for i in xrange(len(s)):
        if s[i] == t[0]:
            cnt[0] += 1
        elif s[i] == t[1]:
            cnt[1] += 1
        else:
            ans += min(cnt[0], cnt[1])
            cnt = [0, 0]
print ans
