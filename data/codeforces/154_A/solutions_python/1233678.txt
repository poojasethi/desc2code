

s = raw_input() + '1'

k = int(raw_input())

ks = []
for i in range(k):
    ks.append(tuple(raw_input()))


res = 0
for k in ks:
    a, b = 0, 0
    for i in range(len(s)):
        if   s[i] == k[0]: a += 1
        elif s[i] == k[1]: b += 1
        else:
            if a and b:
                res += min(a, b)
            a, b = 0, 0


print res
