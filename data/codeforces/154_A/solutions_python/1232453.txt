s = raw_input()
s += '_'
k = int(raw_input())
lens = len(s)

ans = 0
for i in range(k):
    p = raw_input().strip()
    cnt = [0,0]
    for i in range(lens):
        if s[i] == p[0]:
            cnt[0] += 1
        elif s[i] == p[1]:
            cnt[1] += 1
        else:
            ans += min(cnt)
            cnt = [0,0]

print ans
