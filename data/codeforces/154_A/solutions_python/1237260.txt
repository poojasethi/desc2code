s=raw_input()
n=input()
A=[list(raw_input()) for i in range(n)]
M={}
for x in A:
    M[x[0]]=x[1]
    M[x[1]]=x[0]
i=0
t=0
while i<len(s)-1:
    if s[i] in M and (M[s[i]]==s[i+1] or s[i]==s[i+1]):
        j=i
        a=0
        b=0
        while j<len(s):
            if s[j]==s[i]:
                a+=1
            elif s[j]==M[s[i]]:
                b+=1
            else:
                break
            j+=1
        t+=min(a,b)
        i=j
    else:
        i+=1
print t
