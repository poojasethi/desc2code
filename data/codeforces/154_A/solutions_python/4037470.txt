data = raw_input()
m = int(raw_input())
fb = {}
for i in xrange(0,m):
    tmp = raw_input()
    fb[tmp[0]] = tmp[1]
    fb[tmp[1]] = tmp[0]

data += '_'
ans = 0
i = 0 
while i < len(data):
    if data[i] in fb.keys():
        count = {data[i]:0,fb[data[i]]:0}
        while data[i] in count.keys():
            count[data[i]] += 1
            i += 1
        ans += min(count.values())
    else:
        i += 1
print ans
