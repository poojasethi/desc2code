r=raw_input
s=r()
n=input()
d={}
for i in n*[0]:
    p=r()
    d[p[0]]=p
    d[p[1]]=p
    
i=0
r=0
while i < len(s):
    if s[i] in d:
        c = d[s[i]]
        a,b=0,0
        while i < len(s) and s[i] in c:
            a+=s[i] == c[0]
            b+=1
            i+=1
        r+=min(a,b-a)
    else:
        i+=1
print r
