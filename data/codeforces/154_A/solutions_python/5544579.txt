p, t = {}, raw_input()
x = y = s = a = 0
q = [raw_input() for i in range(int(raw_input()))]
for a, b in q: p[a], p[b] = b, a
for i in t: 
    if a:
        if i == a: x += 1
        elif i == b: y += 1
        else:
            s += min(x, y)
            if i in p: a, b, x, y = i, p[i], 1, 0
            else: a = 0
    elif i in p: a, b, x, y = i, p[i], 1, 0
if a: s += min(x, y)
print(s)
