#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;
const int inf=2000000000;
const int N=100005;
int a[N],b[N],c[N],k[N];

int main()
{
    int n,m,x,y,l,res=0,i;
    scanf("%d%d%d%d",&n,&m,&x,&y);
    for (i=1; i<=n; i++) scanf("%d",&a[i]);
    for (i=1; i<=m; i++) {scanf("%d",&b[i]); c[i]=inf;}
    l=1;
    for (i=1; i<=n; i++)
    {
        while (l<m && abs(a[i]-b[l])>abs(a[i]-b[l+1])) l++;
        if (l<m && abs(a[i]-b[l])==abs(a[i]-b[l+1]))
        {
            if (c[l]==inf || c[l]==abs(a[i]-b[l]))
             {c[l]=abs(a[i]-b[l]); k[i]=l;}
            else
             {c[l+1]=min(c[l+1],abs(a[i]-b[l+1])); k[i]=l+1;}
        }
        else
        {
            {c[l]=min(c[l],abs(a[i]-b[l])); k[i]=l;}
        }
    }
    for (i=1; i<=n; i++) if (c[k[i]]==abs(a[i]-b[k[i]])) res++;
    printf("%d\n",n-res);
}
