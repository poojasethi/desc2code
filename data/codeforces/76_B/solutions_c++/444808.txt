#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<vector>
#include<queue>
#include<cctype>
#include<cmath>
#include<map>
using namespace std;

int x[111111],X[111111],n,m,nouse1,nouse2,match[111111],dis[111111];

int main(){
  
  scanf("%d %d %d %d",&n,&m,&nouse1,&nouse2);
  for (int i=1; i<=n; i++) scanf("%d",&x[i]);
  sort(x+1,x+n+1);
  for (int i=1; i<=m; i++) scanf("%d",&X[i]);
  sort(X+1,X+m+1);
  X[0]=-1000000000;
  X[m+1]=1000000000;
  
  memset(match,0,sizeof(match));
  
  int hungry=n;
  for (int i=1; i<=n; i++)
    {
      int l=0,r=m+1; 
      while (l<r-1)
        {
          int mid=(l+r)/2;
          if (X[mid]>x[i]) r=mid; else l=mid;
        }
      int y;
      if (X[l]>x[i]) y=l; else y=r;
      int d=min(X[y]-x[i],x[i]-X[y-1]);
      if (X[y]-x[i]>d || (x[i]-X[y-1]==d && (match[y-1]==0 || dis[y-1]==d))) y--;
      if (match[y]==0 || dis[y]==d) hungry--; 
      if (match[y]==0) {match[y]=1; dis[y]=d;} else dis[y]=min(dis[y],d);
    }
    
  cout<<hungry<<endl;
}
