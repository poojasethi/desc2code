#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 100010

using namespace std;

int t[maxn],dis[maxn];
int x0[maxn],x1[maxn];
int n,m,y0,y1;
inline int Dis(int p1,int p2){
	return abs(x0[p1]-x1[p2]);
}
int main(){
	int i,j,k;
	scanf("%d %d %d %d",&n,&m,&y0,&y1);
	for (i=0;i<n;i++) scanf("%d",x0+i);
	for (i=0;i<m;i++) scanf("%d",x1+i);
	k=0;
	for (i=0;i<m;i++) dis[i]=100000000;
	for (i=0;i<n;i++){
		while (k<m-1 && Dis(i,k)>Dis(i,k+1)) k++;
		if (k<m-1 && Dis(i,k)==Dis(i,k+1)){
			if (t[k]==0 || Dis(i,k)==dis[k]){
				t[k]++;
				dis[k]=Dis(i,k);
			}else{
				k++;
				t[k]=1;
				dis[k]=Dis(i,k);
			}
		}else{
			if (t[k]==0 || Dis(i,k)<dis[k]){
				t[k]=0;
				dis[k]=Dis(i,k);
			}
			if (Dis(i,k)==dis[k]) t[k]++;
		}
	}
	int ans=n;
	for (i=0;i<m;i++) ans-=t[i];
	printf("%d\n",ans);
	return 0;
}
