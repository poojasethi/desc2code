#include<stdio.h>
#include<stdlib.h>
#define INF 999999999
int main(){
	int n,m;
	while(scanf("%d%d%*d%*d",&n,&m)==2){
		int a[100005]={};
		int b[100005]={};
		int pos[100005]={};
		for(int i=0;i<n;i++)
			scanf("%d",&a[i]);
		for(int i=0;i<m;i++)
			scanf("%d",&b[i]);
		if(m==0){
			printf("%d\n",n);
			continue;
		}
		int now=0;
		for(int i=0;i<n;i++){
			while(now<m-1&&abs(b[now+1]-a[i])<abs(b[now]-a[i]))
				now++;
			pos[i]=now;
		}
		int ans=0,dis[100005];
		for(int i=0;i<m;i++)
			dis[i]=INF;
		for(int i=0;i<n;i++)
			if(pos[i]<m-1&&abs(b[pos[i]]-a[i])==abs(b[pos[i]+1]-a[i])){
				if(dis[pos[i]]!=INF&&dis[pos[i]]!=abs(b[pos[i]]-a[i]))
					dis[pos[i]+1]=abs(b[pos[i]+1]-a[i]);
				else dis[pos[i]]=abs(b[pos[i]]-a[i]);
			}
			else{
				if(dis[pos[i]]<abs(b[pos[i]]-a[i])||(dis[pos[i]]>abs(b[pos[i]]-a[i])&&dis[pos[i]]!=INF))
					ans++;
				if(dis[pos[i]]>abs(b[pos[i]]-a[i]))
					dis[pos[i]]=abs(b[pos[i]]-a[i]);
			}
		printf("%d\n",ans);
	}
}
