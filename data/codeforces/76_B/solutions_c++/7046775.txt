#include<iostream>
#include<algorithm>
using namespace std;
#define INF 1000000000
#define N 120000
int x[N],y[N],z[N];
int main()
{
	int n,m,y0,y1;
	while(cin>>n>>m>>y0>>y1)
	{
		int i;
		for(i=0;i<n;i++)
		cin>>x[i];
		y[0]=-INF;
		for(i=1;i<=m;i++)
		cin>>y[i];
		y[++m]=INF;
		for(i=0;i<m;i++)
		z[i]=INF;
		int ans=0;
		for(i=0;i<n;i++)
		{
			int j=lower_bound(y,y+m+1,x[i])-y;
			int d=min(y[j]-x[i],x[i]-y[j-1]);
			if(y[j]-x[i]>d||(x[i]-y[j-1]==d&&(z[j-1]==d||z[j-1]==INF)))j--;
			if(z[j]==d||z[j]==INF)
			ans++;
			z[j]=min(z[j],d);
		}
		cout<<n-ans<<endl;
	}
}
