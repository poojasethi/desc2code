#include<iostream>
#define abs(x) max(x,-(x))
using namespace std;
const int N=100100;
int xn[N],xm[N],c[N],md[N];
int main()
{
    int n,m;
    cin>>n>>m;
    if(!m)
    {
        cout<<n<<endl;
        return 0;
    }
    int trash1,trash2;
    cin>>trash1>>trash2;
    for(int i=0;i<n;i++)
        cin>>xn[i];
    for(int j=0;j<m;j++)
        cin>>xm[j];
    fill(md,md+m,1e8);
    int j=0;
    for(int i=0;i<n;i++)
    {
        while(j<m-1 && xm[j+1]<=xn[i])
            j++;
        if(j==m-1 || abs(xn[i]-xm[j])<abs(xn[i]-xm[j + 1]) || (abs(xn[i]-xm[j])==abs(xn[i]-xm[j+1]) && (md[j]==1e8 || md[j]==abs(xn[i]-xm[j]))))
            c[i]=j;
        else
            c[i]=j+1;
        md[c[i]]=min(md[c[i]],abs(xn[i]-xm[c[i]]));
    }
    int ans=0;
    for (int i=0;i<n;i++)
        if(md[c[i]]!=abs(xn[i]-xm[c[i]]))
            ans++;
    cout<<ans<<endl;
    return 0;
}
