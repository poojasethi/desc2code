#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;

#define MAX 100010

int ches[MAX];
int mice[MAX];
int n,m;

int main()
{
//  freopen("TEST.inp","r",stdin);
    
    int i,x,y,j;
    scanf("%d %d %d %d",&n,&m,&x,&y);
    for(i=0;i<n;i++){
        scanf("%d",&mice[i]);
    }
    for(j=0;j<m;j++){
        scanf("%d",&ches[j]);
    }
    
    int last=-1,res=0;
    j=0;
    for(i=0;i<n;i++){
        while(j<m && abs(mice[i]-ches[j])>abs(mice[i]-ches[j+1])){
            j++;
        }
        
        if(j!=last || abs(mice[i]-ches[j])==abs(mice[i-1]-ches[j])){
            last=j;
            res++;
        }
        else if(j+1<m && abs(mice[i]-ches[j])==abs(mice[i]-ches[j+1])){
            res++;
            last=++j;
        }
        else{
            last=j;
        }
    }
    printf("%d",n-res);
}
