#include<bits/stdc++.h>
using namespace std;
int aa[100005],bb[100005],cc[100005];

int s(int a,int b,int c){
    int ans=c,d=abs(aa[a]-bb[b]);
    for(int i=b+1;i<=c;i++){
        if(abs(aa[a]-bb[i])>=d){
            ans=i-1;break;
        }else d=abs(aa[a]-bb[i]);
    }
    return ans;
}

int main(){
//  freopen ("mice.inp","r",stdin);
//  freopen ("mice.out","w",stdout);
    int n,m,x1,Y1,ans,k=1,p=0,q=0;
    cin>>n>>m>>x1>>Y1;
    ans=n;
    for(int i=1;i<=n;i++) cin>>aa[i];
    for(int j=1;j<=m;j++) cin>>bb[j];
    if(k<=m){
        for(int i=1;i<=n;i++){
            k=s(i,k,m);
            if(cc[k]==0){ans--;cc[k]++;p=abs(aa[i]-bb[k]);}
            else{
                q=abs(aa[i]-bb[k]);
                if(p==q){ans--;cc[k]++;}
                else{
                    if(k!=m&&abs(aa[i]-bb[k+1])==q){ k++;cc[k]++;ans--;p=q;}
                    else if(q<p){ans=ans+cc[k]-1;cc[k]=1;p=q;}
                }
            }
        }
            
    }
    cout<<ans;
    return 0;
}
