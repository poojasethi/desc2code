#include <cstdio>
#define INF 1000000000
int i,j,n,m,x,y,pa;
int a[111111],b[111111],c[111111];
int main()
{
    //freopen("76b.in","r",stdin);
    //freopen("76b.out","w",stdout);
    scanf("%d%d%*d%*d",&n,&m);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    
    for(i=0;i<m;i++)
    {
        scanf("%d",&b[i]);
        c[i]=INF;
    }
    b[m]=INF;
    for(i=j=0;i<n;i++)
    {
        while(a[i]>b[j]) j++;
        if(j) x=a[i]-b[j-1]; else x=INF;
        y=b[j]-a[i];
        if(x>y)
        {
            if(c[j]==INF) pa++;
            c[j]=y;
        }
        if(x<y)
        {
            if(c[j-1]==INF || c[j-1]==x) pa++;
            if(c[j-1]>x) c[j-1]=x;
        }
        if(x==y)
        {
            if(c[j-1]==INF || c[j-1]==x)
            {
                pa++;
                c[j-1]=x;
            }else
            if(c[j]==INF)
            {
                pa++;
                c[j]=y;
            }else
            if(c[j-1]>x) c[j-1]=x;
        }
    }
    printf("%d\n",n-pa);
    return 0;
}
