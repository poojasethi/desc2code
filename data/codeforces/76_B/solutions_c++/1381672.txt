#include <iostream>
#include <cstdlib>
using namespace std;

int main()
{
	int n,m,y0,y1,i,j,ans,dist;
	bool ok;
	int a[100011],b[100011],min[100011];
	cin >> n >> m >> y0 >> y1;
	for(int i = 1; i <= n; i++) cin >> b[i];
	for(int i = 1; i <= m; i++) cin >> a[i];
	for(int i = 1; i <= m; i++) min[i]=0x7fffffff;
	j = 1;
	ans = n;
	for(int i = 1; i <= n; i++)
	{
		while (j < m && a[j+1] <= b[i]) j++;
		dist = 0x7fffffff;
		if(j > 0) dist = abs(a[j]-b[i]);
		ok = false;
		if (j+1 <= m)
			if(abs(a[j+1]-b[i]) <= dist)
			{
				dist = abs(a[j+1]-b[i]);
				ok = true;
			}
		if(j > 0)
			if(abs(a[j]-b[i]) == dist)
				if (min[j] == 0x7fffffff || min[j] == dist || !ok)
				{
					if(dist < min[j])
					{
						if(min[j] < 0x7fffffff) ans++;
						min[j] = dist;
					}
					if (dist == min[j]) ans--;
					continue;
				}
		if(dist < min[j+1])
		{
			if(min[j+1] < 0x7fffffff) ans++;
			min[j+1] = dist;
		}		
		if(dist == min[j+1]) ans--;
	}
	cout << ans << "\n";
}
