#include<cstdio>  
#include<cstring>  
#include<cstdlib>  
#include<iostream>  
#include<algorithm>  
using namespace std;
#define fo(i,a,b) for(i=a;i<=b;++i)
const int mn=100100;
int n,m,i,j,a[mn],b[mn],tmp,last,ans;
char ch;
void read(int &a)
{
    int p=1;
    for(ch=getchar();(ch<'0'||ch>'9') && (ch!='-');ch=getchar());
    if(ch=='-')a=0,p=-1;else a=ch-'0';
    for(ch=getchar();ch>='0'&& ch<='9';ch=getchar())a=a*10+ch-'0';
    a*=p;
}
int main()
{
    read(n),read(m),read(tmp),read(tmp);
    fo(i,1,n)read(a[i]);
    fo(i,1,m)read(b[i]);
    ans=last=0,j=1;
    fo(i,1,n)
    {
        while(j<m&&abs(a[i]-b[j+1])<abs(a[i]-b[j]))++j;
        if(last<j||abs(a[i-1]-b[j])==abs(a[i]-b[j]))++ans,last=j;
        else if(j<m&&abs(a[i]-b[j+1])==abs(a[i]-b[j]))++ans,last=++j;
        else last=j;
    }
    printf("%d\n",n-ans);
    return 0;
}
