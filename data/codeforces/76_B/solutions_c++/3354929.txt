using namespace std;

const int inf = 0x3f3f3f3f;
const int maxn = 100010;
int x0[maxn], x1[maxn], d[maxn], c[maxn];

int main() {
    int n, m, y0, y1;
    scanf("%d %d %d %d", &n, &m, &y0, &y1);
    for (int i = 0; i < n; ++i) {
        scanf("%d", &x0[i]);
    }
    memset(d, -1, sizeof(d));
    for (int i = 0; i < m; ++i) {
        scanf("%d", &x1[i]);
    }
    int p = 0, k = 0, last = -1;
    for (int i = 0; i < n; ++i) {
        while (p < m && abs(x0[i] - x1[p]) > abs(x0[i] - x1[p + 1])) ++p;
        if (last < p || abs(x0[i - 1] - x1[p]) == abs(x0[i] - x1[p])) ++k, last = p;
        else if (p < m && abs(x0[i] - x1[p]) == abs(x0[i] - x1[p + 1])) ++k, last = ++p;
        else last = p;
    }
    printf("%d\n", n - k);
    return 0;
}
