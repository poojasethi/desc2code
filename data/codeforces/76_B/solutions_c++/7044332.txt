#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m,y1,y2,i,j,ans,w;
int flag[110000],dis[110000],b[110000],a[110000];
int min(int a,int b)
{
    if (a<b) return a;return  b;
}
int main()
{
    scanf("%d%d%d%d",&n,&m,&y1,&y2);
    for (i=1;i<=n;i++)
    scanf("%d",&a[i]);
    for (i=1;i<=m;i++)
    scanf("%d",&b[i]);
    j=1;
    b[0]=-1000000000;
    b[m+1]=1000000000;
    for (i=1;i<=n;i++)
    {
        while (b[j]<a[i]) j++;
        w=min(b[j]-a[i],a[i]-b[j-1]);
        if (w==a[i]-b[j-1])
        {
        if ((dis[j-1]==w)||(flag[j-1]==0))
        {
               flag[j-1]++;
               dis[j-1]=w;
        }
        else
        if (b[j]-a[i]==w)
        {
              flag[j]=1;
              dis[j]=w;
        }
        else
        if (dis[j-1]>w)
        {
            flag[j-1]=1;
            dis[j-1]=w;
        }
        }
        else
        {
            flag[j]=1;
            dis[j]=w;
        }
    }
    for (i=1;i<=m;i++)
    ans=ans+flag[i];
    printf("%d",n-ans);
    
}
