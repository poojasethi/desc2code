#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#define Max 100005
#define INF 1e9
using namespace std;

int arr[Max], brr[Max];
int mat[Max];
int mice[Max];

int main()
{
	int n, m, i, j, ans, d;

	scanf("%d %d %*d %*d", &n, &m);
	
	for(i=0;i<n;i++)
		scanf("%d", &arr[i]);

	for(i=0;i<m;i++)
		scanf("%d", &brr[i]);

	for(i=0;i<m;i++)
		mat[i] = INF;

	for(i=0, j=0;i<n;i++)
	{
		while(j+1<m && abs(brr[j]-arr[i]) > abs(brr[j+1]-arr[i]))
			j++;

		if(j+1<m && mat[j]!=INF && 
			abs(brr[j]-arr[i]) != mat[j] && abs(brr[j]-arr[i])==abs(brr[j+1]-arr[i]) )
			j++;

		d = abs(brr[j]-arr[i]);
		mice[i] = j;


		mat[j] = min(d, mat[j]); 
	}

	ans = 0;
	for(i=0;i<n;i++)
	{
		j = mice[i];

		//cout<<i<<" -> "<<j<<"\n";
		//cout<<mat[j]<<"\n";

		if(mat[j] != abs(brr[j]-arr[i]))
			ans++;
	}
	printf("%d\n", ans);
}
