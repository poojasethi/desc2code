#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;

int t[101000],n,m,ya,yb,a[101000],b[101000],et[101000],de[101000];

int main()
{
  //  freopen("76B.txt","r",stdin);

    cin>>n>>m>>ya>>yb;
    for (int i=0;i<n;i++) cin>>a[i];
    for (int i=0;i<m;i++) cin>>b[i];

    memset(et,127,sizeof(et));
    int po=0;
    for (int i=0;i<n;i++)
    {
        while ( po<m-1 &&  (( abs(a[i]-b[po+1])<abs(a[i]-b[po]) ))) po++;
        t[i]=abs(a[i]-b[po]);
        if (et[po]==et[m+1]) de[i]=po;
        else if (po<m-1 && t[i]==abs(a[i]-b[po+1]))
			{
        	 if (et[po]!=t[i]) de[i]=po+1;
			 else de[i]=po;
			}
        else de[i]=po;
        et[de[i]]=min(et[de[i]],t[i]);
    }
    int ans=n;
    for (int i=0;i<n;i++) if (et[de[i]]==t[i]) ans--;
    if (m==0)cout<<n<<endl;else cout<<ans<<endl;

    return 0;
}
