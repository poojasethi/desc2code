#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int t[200000],d[200000];
int a[200000],b[200000];
int n,m;

int dis(int x,int y){
        return abs(a[x]-b[y]);
}
int main()
{
    //freopen("b.in","r",stdin);
    //freopen("b.out","w",stdout);
    
    int y0,y1;
    cin>>n>>m>>y0>>y1;
    for (int i=1;i<=n;i++) 
        scanf("%d",&a[i]);
    for (int i=1;i<=m;i++) 
        scanf("%d",&b[i]);
    int k=1;
    for (int i=1;i<=m;i++) 
        d[i]=100000000;
    for (int i=1;i<=n;i++) {
        while (k<m && dis(i,k)>dis(i,k+1)) 
              k++;
        if (k<m && dis(i,k)==dis(i,k+1)) {
           if (t[k]==0 || dis(i,k)==d[k]) {
              t[k]++;
              d[k]=dis(i,k);
           } else {
              k++;
              t[k]=1;
              d[k]=dis(i,k);
           }
        } else {
            if (t[k]==0 || dis(i,k)<d[k]) {
               t[k]=0;
               d[k]=dis(i,k);
               t[k]++;
            } else if (dis(i,k)==d[k]) t[k]++;
          }
    }
    int ans=0;
    for (int i=1;i<=m;i++) 
        ans+=t[i];
    cout<<n-ans<<endl;
    //system("pause");
    return 0;
}
