#include <bits/stdc++.h>
using namespace std;
const int mx = 100050;
int a[mx];
int b[mx];
int main()
{
    int n,m,y,yy;
    scanf("%d %d %d %d",&n,&m,&y,&yy);
    for(int x=0; x<n; ++x)
    {
        cin >> a[x];
    }
    for(int x=0; x<m; ++x)
    {
        cin >> b[x];
    }
    int i=0,j=0,last=-1;
    int kk=0;
    for(int x=0; x<n; ++x)
    {
        while(j<m&&abs(a[x]-b[j+1])<abs(a[x]-b[j])) j++;
        if(j!=last || abs(a[x]-b[j]) == abs(a[x-1]-b[j]) ) kk++, last=j;
        else if( j+1<m && abs(a[x]-b[j]) == abs(a[x]-b[j+1]) ) kk++, last=++j;
        else last = j;
    }
    cout << n - kk << endl;
}
