#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>

using namespace std;

const int MaxN=100005;
int N,M,Mice[MaxN],Ches[MaxN],d[MaxN],t[MaxN];
int dis(int x,int y) {return abs(Mice[x]-Ches[y]);}
int main() {
    scanf("%d%d%*d%*d",&N,&M);
    for (int i=1;i<=N;++i) scanf("%d",&Mice[i]);
    for (int i=1;i<=M;++i) scanf("%d",&Ches[i]);
    int k=1;
    for (int i=1;i<=N;++i) {
        while (k<M && dis(i,k)>dis(i,k+1)) ++k;
        if (k<M && dis(i,k)==dis(i,k+1)) {
           if (t[k]!=0 && d[k]!=dis(i,k)) ++k;
           ++t[k];
           d[k]=dis(i,k);
        }
        else if (t[k]==0 || dis(i,k)<d[k]) {
                t[k]=1;
                d[k]=dis(i,k);
             }
             else if (dis(i,k)==d[k]) ++t[k];
    }
    int ans=N;
    for (int i=1;i<=M;++i) ans-=t[i];
    cout<<ans<<"\n";
    //system("pause");
    return 0;
}
