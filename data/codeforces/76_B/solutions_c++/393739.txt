#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;

#define fr(i, n) for(int i=0;i<n;i++)
#define PB push_back
const int maxn=500*1000+1000, inf=(1<<31)-1;
int n, m, t, d[maxn], match[maxn];
bool mark[maxn], Mark[maxn];
vector<int> x, y;

inline int dis(const int &u, const int &v){
	int p=x[u]-y[v];
	if(p<0)
		return -p;
	return p;
}

int main(){
	scanf("%d%d", &n, &m);
	scanf("%d%d", &t, &t);
	t=0;
	x.resize(n);
	fr(i, n)
		scanf("%d", &x[i]);
	y.resize(m);
	fr(i, m){
		scanf("%d", &y[i]);
		d[i]=inf;
	}
	fr(i, n){
		while(t<m-1 && dis(i, t)>dis(i, t+1))
			t++;
		if(!mark[t]){
			d[t]=dis(i, t);
			mark[t]=1;
			match[i]=t;
		}else
			if(d[t]==dis(i, t))
				match[i]=t;
			else
				if(t<m-1 && dis(i, t)==dis(i, t+1)){
					match[i]=t+1;
					mark[t+1]=1;
					d[t+1]=dis(i, t+1);
				}else{
					match[i]=t;
					d[t]=min(d[t], dis(i, t));
				}
	}
	int sum=0;
	fr(i, n)
		if(d[match[i]]!=dis(i, match[i]))
			sum++;
	cout << sum << endl;
	return 0;
}
