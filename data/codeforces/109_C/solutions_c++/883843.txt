#include<iostream>
using namespace std;
int n,f[100001],s[100001];
long long ans=0;
int set(int x){return f[x]==x?x:f[x]=set(f[x]);}
int main()
{
	cin>>n;
	for(int i=1;i<=n;++i)f[i]=i,s[i]=1;
	for(int i=1,x,y,z,p,q;i<n;++i)
	{
		cin>>x>>y>>z;
		bool out=z>0;
		for(;z;z/=10)if(z%10!=4&&z%10!=7)out=0;
		if(out)continue;
		p=set(x),q=set(y);
		f[p]=q,s[q]+=s[p];
	}
	for(int i=1;i<=n;++i)ans+=(long long)(n-s[set(i)])*(n-s[set(i)]-1);
	cout<<ans<<endl;
	return 0;
}
