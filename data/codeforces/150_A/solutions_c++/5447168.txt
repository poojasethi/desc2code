#include <cstdio>
using namespace std;

int main(){
    long long n,f=0;
    scanf("%lld",&n);
    for(long long i=2;i*i<=n;i++){
        if(n%i==0 && !f) f=i,n/=i;
        if(n%i==0 && n>i) {
            printf("1\n%lld",f*i);
            return 0;
        }
    }
    if(!f) puts("1\n0"); else puts("2");
    return 0;
}
