#include <iostream>
using namespace std;
int main(){
  long long n,m,p,q;
  cin >> n;
  for(p=2; p*p<=n and n%p!=0; p++);
  if(p*p>n) cout << 1 << endl << 0 << endl;
  else{
    m=n/p;
    for(q=p; q*q<=m and m%q!=0; q++);
    if(q*q>m) cout << 2 << endl;
    else cout << 1 << endl << p*q << endl;  
  } 
  return 0;
}
