#include <iostream>
using namespace std;

typedef long long ll;


int main() {
	ll q, d[50], nd = 0, i;

	cin >> q;
	for (i = 2; q > 1; i++) {
		if (i * i > q) {d[nd++] = q; break;}
		while (q % i == 0) d[nd++] = i, q /= i;
	}

	if (nd <= 1) {
		cout << "1\n0";
	} else if (nd <= 2) {
		cout << "2";
	} else {
		cout << "1\n" << d[0] * d[1];
	}

	return 0;
}
