#include <iostream>

typedef long long ll;

using namespace std;

int main() {
	ll d = 0;
	ll n;
	cin >> n;
	ll ac = 1;
	for (ll i = 2; i * i <= n; i++) {
		while (n % i == 0) {
			n /= i;
			if (d < 2)
				ac *= i;
			d++;
		}
	}
	if (n > 1)
		d ++;
	if (d == 2)
		cout << 2 << endl;
	else {
		cout << 1 << endl;
		cout << (ac != 1 ? ac : 0) << endl;
	}
}
