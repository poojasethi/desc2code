#include<bits/stdc++.h>
using namespace std;
long long n,h[5],u,i,j,s,t,d;
int main()
{
	cin>>n;
	d=sqrt(n);
	for(i=2;i<=d;i++)
	{
		while(n%i==0)
		{
			h[u++]=i;
			n/=i;
			if(u>2){cout<<1<<endl<<h[0]*h[1]<<endl;return 0;}
		}
	}
	if(n>1 and u==2){cout<<1<<endl<<h[0]*h[1]<<endl;return 0;}
	if(!u){cout<<1<<endl<<0<<endl;}
	else cout<<2<<endl;
}
