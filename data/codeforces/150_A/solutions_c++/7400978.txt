#include<stdio.h>
#include<math.h>
__int64 n,x,tot,g[1000];
int main(){
	scanf("%I64d",&n);
	x=sqrt(n);
	tot=0;
	g[1]=g[2]=0;
	for(__int64 i=2;i*i<=n;i++)
	if(n%i==0){		
		while(n%i==0){n/=i;tot++;g[tot]=i;}		
	}
	if(n!=1)tot++;
	if(tot<=1||tot>=3)printf("1\n%I64d\n",g[1]*g[2]);
	else printf("2\n");	
}
