#include<cstdio>
#define ll __int64
int main()
{
	ll q,ans;
	int i,d,cnt,a[100];
	d=0;

	ans=1,cnt=0;
	for(i=2;(ll)i*i<=q;i++)
	{
		while(q%i==0)
		{
			ans*=i;
			cnt++;
			q/=i;
			if(cnt==2&&q!=1)
			{
				printf("1\n%I64d\n",ans);

			}
		}
	}
