#include <iostream>
#include <cmath>
using namespace std;
int main(){
  long long q,w,e,r=0,t=0,a[3];
  cin>>q;
  for(w=2,e=sqrt(q);w<=e;w++)
   while(q%w==0){
    if(t==3)
     goto stop;
    q/=w;
    a[t]=w;
    t++;}
  if(q>1)t++;
  stop:
  if(t<2)cout<<"1\n0";
  else if(t==2)cout<<2;
  else cout<<"1\n"<<a[0]*a[1];
  return 0;}
