#include <iostream>
using namespace std;

long long a[10];

int main() {
	long long n, copy;
	int cur=a[0]=a[1]=0;
	cin >> n;
	copy=n;
	for(long long i=2; i*i <= n; i++) 
		for(; n%i == 0 && cur<3; n /= i)
			a[cur++]=i;
	if(copy != n && n>1) a[cur++]=n;
	if(cur-2) cout << 1 << endl << a[0]*a[1] << endl;
	else cout << 2 << endl;  	
	return 0;
}
