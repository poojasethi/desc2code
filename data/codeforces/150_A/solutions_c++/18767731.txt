#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll x; int idx = 0, pf[111111];

int main(int argc, char **argv) {
	cin >> x;
	for(int i=2 ; (ll)i*i<=x ; ++i) while(x%i==0) pf[idx++] = i, x/=i;
	if(x > 1) pf[idx++] = x;
	if(idx == 2) puts("2");
	else {
		puts("1");
		printf("%lld\n", (ll)pf[0] * pf[1]);
	}
	return 0;
}
