#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
typedef long long LL;
LL a[100000],k=0,n,m;
int main()
{
    cin>>n;m=n;
    for(LL i=2;i*i<=n;i++)
        for(;n%i==0;n/=i)a[++k]=i;
    if(n!=m&&n!=1)a[++k]=n;
    if(k!=2)cout<<"1"<<endl<<a[2]*a[1]<<endl;else puts("2");
    return 0;
}
