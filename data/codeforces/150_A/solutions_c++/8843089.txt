#include<bits/stdc++.h>

__int64 fac[60],tot;
int main(){
	__int64 n;
	scanf("%I64d",&n);
	for(__int64 i = 2; i*i<=n;i ++)
		while(n%i==0)
			n/=i,fac[tot++]=i;
	if(n > 1)
		fac[tot++] = n;
	if(tot<=1)
		printf("1\n0\n");
	else if(tot == 2)
		printf("2");
	else {
		__int64 ans = fac[0] * fac[1];
		printf("1\n%I64d", ans);
	}
	return 0;
}
