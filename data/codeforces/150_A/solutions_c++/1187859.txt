#include<iostream>
using namespace std;
int main(){
	long long a,t,d[1024];
	int c=0;
	cin >> a;t=a;
	for(;!(t&1);d[c++]=2)t>>=1;
	for(long long i=3;i*i<=t;i+=2)
		for(;t%i==0;t/=i)
			d[c++]=i;
	if(t>1)d[c++]=t;
	if(c==2)return (cout<<2),0;
	cout << 1 << endl;
	if(c==0)cout<<0;
	else cout << d[0]*d[1];
	return 0;
}
