#include<iostream>
typedef long long ll;
using namespace std;

int main()
{
  ll n;
  cin >> n;
  ll i,j,tmp,ans=0;
  
  for(i=2;(i*i)<=n && n%i != 0;i++);
  
  if((i*i) > n)
    cout << "1\n0" << endl;
  else
  {
    tmp = n/i;

    for(j=2;(j*j)<=tmp && tmp%j!=0;j++);

    if((j*j)>tmp)
      cout << "2" << endl;
    else
      cout << "1\n" << j*i << endl; 
  }
return 0;
}
