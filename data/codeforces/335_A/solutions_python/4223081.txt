from collections import Counter
from math import ceil

s = raw_input()
c = Counter(s)
n = int(raw_input())
if n < len(c):
    print -1
else:
    d = {l: 1 for l in c}
    for _ in range(n - len(c)):
        _, l = max([(float(c[l]) / d[l], l) for l in c])
        d[l] += 1
    print int(max(ceil(float(c[l]) / d[l]) for l in c))
    print ''.join(l * d[l] for l in d)
