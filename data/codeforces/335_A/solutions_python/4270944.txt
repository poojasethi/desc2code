def ceil(a,b):
	return (a+b-1)/b

s = raw_input()
n = input()

cnts = {}
for l in set(s): cnts[l] = s.count(l)

if len(cnts) > n:
	print -1
	raise SystemExit
for i in range(1,1001):
	tot = 0
	for (l, c) in cnts.items():
		tot += ceil(c,i)
	if tot <= n:
		print i
		out = ''
		for (l, c) in cnts.items():
			out += l*ceil(c,i)
		if len(out) < n:
			out += 'a'*(n-len(out))
		print out
		raise SystemExit
