s = raw_input()
n = input()
f = {}
for c in s: f[c] = f.get(c, 0) + 1
if n < len(f):
	print -1
	quit()

l, r = 1, len(s)
while l < r:
	m = l + r >> 1
	if n < sum((v + m - 1) / m for v in f.itervalues()): l = m + 1
	else: r = m

print l
a, s = '', 0
for c, v in f.iteritems():
	t = (v + l - 1) / l
	s += t
	a += c * t

print a + 'a' * (n - s)
