#include <cstdio>
#include <cstring>

using namespace std;

const int maxlen=1000;
char s[maxlen],ans[maxlen];
int n,p,ls,la,t[26],st;

int main() {
	scanf("%s",s);
	scanf("%d",&n);
	ls=strlen(s);
	p=0;
	for (int i=0;i<26;i++) t[i]=0;
	for (int i=0;i<ls;i++) {
		if (t[s[i]-97]==0) p++;
		if (p>n) {
			printf("-1\n");
			return 0;
		}
		t[s[i]-97]++;
	}
	p=0;
	do {
		p++;
		st=0;
		for (int i=0;i<26;i++) st+=t[i]/p+(t[i]%p>0);
	}
	while (st>n);
	la=0;
	for (int i=0;i<26;i++) 
		while (t[i]>0) {
			t[i]-=p;
		 	ans[la]=i+97;
		 	la++;
		}
	while (la<n) {
		ans[la]='a';
		la++;
	}
	ans[n]=0;
	printf("%d\n%s\n",p,ans);
	return 0;
}
