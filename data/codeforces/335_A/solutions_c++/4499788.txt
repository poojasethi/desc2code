#include <iostream>
#include <cmath>
#include <string>
using namespace std;

string s;
int n; 
float alp[26];


int main(){

	cin>>s>>n;

	int dif=0, slen=s.length();
	for(int i=0; i<slen; ++i){
		if(alp[s[i]-'a']==0) ++dif;
		++alp[s[i]-'a'];
	}
	if(dif>n){
		cout<<-1<<'\n';
		return 0;
	}

	for(int res=1; res<=slen; ++res){
		
		int tot=0;
		for(int i=0; i<26; ++i){
			//cout<<alp[i]<<endl;
			tot+=ceil(alp[i]/res);
		}
		//cout<<tot<<'\n';
		if(tot>n) continue;

		cout<<res<<'\n';
		string r;
		for(int i=0; i<26; ++i){
			for(int o=0, O=ceil(alp[i]/res); o<O; ++o){
				r+=('a'+i);
			}
		}
		for(int i=r.length(); i<n; ++i)
			r+='a';
		cout<<r<<'\n';
		return 0;

	}

}
