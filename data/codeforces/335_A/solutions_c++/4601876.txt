#include <cstdio>
#include <cstring>

int N,K,a[30],b[30];
char s[1005];

int ok(int x)
{
	int s=0;
	for (int i=0; i<26; i++) if (a[i]) b[i]=(a[i]-1)/x+1,s+=b[i];
	return s;
}

void doit()
{
	scanf("%s%d",s+1,&K),N=strlen(s+1);
	for (int i=1; i<=N; i++) a[s[i]-97]++;
	int l=1,r=N;
	for (;l<=r;)
	{
		int m=(l+r)>>1;
		if (ok(m)<=K) r=m-1; else l=m+1;
	}
	if (l>N) {puts("-1"); return;}
	int x=ok(l);
	printf("%d\n",l);
	for (int i=0; i<26; i++)
		for (int j=1; j<=b[i]; j++) putchar(i+97);
	for (int i=x+1; i<=K; i++) putchar(s[1]);
	puts("");
}

int main()
{
	doit();
	return 0;
}
