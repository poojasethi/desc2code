#include <cstdio>
#include <cstring>

#define REP(i,n) for(int i=0; i<(n); ++i)
#define FOR(i,p,k) for(int i=(p); i<=(k); ++i)

enum { s_max = 2000 };

char S[s_max];
char P[s_max];

int A[300];

int main()
{
	int n;
	scanf(" %s%d",S,&n);
	int s = strlen(S);
	REP(i,s) A[S[i]]++;
	
	int d = 0; REP(i,300) if(A[i]) d++;
	if(d>n){ puts("-1"); return 0; }

	int l = 1, h = s;
	while(l<h)
	{
		int m = (l+h)/2;
		int r = 0;
		REP(i,300) r += (A[i]+m-1)/m;
		if(r<=n) h = m; else l = m+1;
	}

	int p = 0;
	REP(i,300) A[i] = (A[i]+l-1)/l;
	REP(i,300) while(A[i]--) P[p++] = i;
	while(p<n) P[p++] = 'a';
	printf("%d\n%s\n",l,P);

	return 0;
}
