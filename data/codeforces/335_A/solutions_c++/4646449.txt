#include<cstdio>
#include<cstring>

char s[5555],a[5555];
int n,ans,len,v[555];

bool check(int mul) {
	memset(a,0,sizeof a);
	int l=0;
	for(int i='a';i<='z';i++) {
		int x = v[i]?((v[i]-1)/mul+1):0;
		for(int j=0;j<x;j++)
			a[l++] = i;
	}
	while(l<n)a[l++]='a';
	return (l<=n);
}

int main() {
	scanf("%s%d", s, &n);
	len=strlen(s);
	memset(v,0,sizeof v);
	for(int i=0;i<len;i++)
		v[s[i]]++;
	int ans=-1;
	for(int i=1;i<=len;i++)
		if(check(i)) {
			ans = i;break;
		}
	printf("%d\n",ans);
	if(~ans)printf("%s\n",a);

	return 0;
}
