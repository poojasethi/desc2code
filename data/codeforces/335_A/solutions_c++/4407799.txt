#include <iostream>
#include <vector>
#include <string>

using namespace std;

int occ[30];

int main(void){
	string str;
	int n;
	cin >> str >> n;
	for(int i = 0; i < (int)str.length(); ++i)
		occ[str[i] - 'a']++;
	int chtype = 0;
	for(int i = 0; i < (int)30; ++i){
		if(occ[i] > 0) chtype++;
	}
	if(chtype > n){
		cout << -1 << endl;
		return 0;
	}
	int a = 1;
	string ret;
	while(true){
		ret = "";
		for(int i = 0; i < (int)26; ++i){
			int dl = 0;
			dl += occ[i] / a;
			if(occ[i] % a > 0) dl++;
			string st(dl, (char)i + 'a');
			ret += st;
		}
		if(ret.length() <= n) break;
		a++;
	}
	if(ret.length() < n){
		string st(n - ret.length(), 'z');
		ret += st;
	}
	cout << a << endl << ret << endl;
	return 0;
}
