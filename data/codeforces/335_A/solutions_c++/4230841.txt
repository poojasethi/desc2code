#include <iostream>
#include <cstring>

using namespace std;

string s;
int n, a[256];

main()
{
  cin >> s;
  cin >> n;
  for (int i = 0; i < s.size(); i++) a[s[i]]++;
  
  int x = 0;
  for (int i = 0; i < 256; i++) x += (a[i] > 0);
  if (x > n)
  {
    cout << -1;
  }
  else
  {
    for (int i = 1; i <= s.size(); i++) 
    {
      int y = 0;
      for (int j = 0; j < 256; j++) 
        if (a[j]) y += (a[j] - 1) / i + 1;
      if (y <= n)
      {
        cout << i << "\n";
        for (int j = 0; j < 256; j++)
          if (a[j]) 
            for (int k = 0; n && k < (a[j] - 1) / i + 1; k++, n--) cout << char(j);
        while (n--) cout << "a";    
      }
    }
  }
//  system("pause");
}
