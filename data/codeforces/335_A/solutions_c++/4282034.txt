#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
int len, a[111], ans = -1;
string s;
int main()
{
	cin >> s;
	cin >> len;
	for (int i = 0; i < s.length(); i++)
		a[s[i] - 'a']++;
	for (int i = s.length(); i >= 1; i--)
	{
		int cur = 0;
		for (int j = 0; j < 26; j++)
			if (a[j])
				cur += (a[j] - 1) / i + 1;
		if (cur <= len)
			ans = i;
	}
	cout << ans << endl;
	if (ans != -1)
	{
		int cur_len = 0;
		for (int i = 0; i < 26; i++)
			if (a[i])
			{
				for (int j = 0; j < (a[i] - 1) / ans + 1; j++)
					cout << char(i + 'a');
				cur_len += (a[i] - 1) / ans + 1;
			}
		for (int i = cur_len; i < len; i++)
			cout << 'a';
	}
	cin >> s;
}
