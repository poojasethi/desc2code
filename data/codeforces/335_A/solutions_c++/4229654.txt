#include <iostream>

using namespace std;

char ans[1005];
int mark[30];

int main()
{
	string s;
	int n;
	cin >> s;
	cin >> n;
	for (int i = 0; i < s.length(); i++)
		mark[s[i] - 'a']++;
	int d, j;
	for (d = 1; d <= 1000; d++)
	{
		j = 0;
		bool can = true;
		for (int i = 0; i < 26; i++)
		{
			int t = mark[i] / d;
			if (mark[i] % d != 0)
				t++;
			while (t != 0)
			{
				if (j == n)
				{
					can = false;
					break;
				}
				ans[j] = i + 'a';
				j++;
				t--;
			}
			if (!can)
				break;
		}
		if (can)
			break;
	}
	if (d > 1000)
	{
		cout << -1 << endl;
		return 0;
	}
	cout << d << endl;
	for (int i = 0; i < j; i++)
		cout << ans[i];
	for (int i = j; i < n; i++)
		cout << 'a';
	cout << endl;
}
