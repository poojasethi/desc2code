#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <vector>
using namespace std;

#define FOR(it, c) for(__typeof((c).begin()) it = (c).begin(); it != (c).end(); it++)
#define SZ(c) ((int)(c).size())

char s[1005];
int n;
int t[128];
int main(void) {
	scanf("%s%d", s, &n);
	int m = strlen(s);
	for(int i=0;i<m;i++) t[(int)s[i]]++;
	for(int i=1;i<=m;i++) {
		int r=0;
		for(int j='a';j<='z';j++)
			r += (t[j]+i-1)/i;
		if(r<=n) {
			printf("%d\n", i);
			for(int j='a';j<='z';j++)
				for(int k=0;k<(t[j]+i-1)/i;k++)
					printf("%c", j);
			while(r<n) { printf("z"); ++r; }
			printf("\n");
			return 0;
		}
	}
	puts("-1");
	return 0;
}
