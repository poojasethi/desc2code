#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;

int main()
{
	string s;
	int num[30];
	int i, j, k, n;
	
	cin >> s;
	for (i = 0; i < 30; ++i)
		num[i] = 0;
	for (i = 0; i < s.length(); ++i)
		num[s[i] - 'a']++;
	cin >> n;
	i = 1;
	while (i <= s.length())
	{
		j = 0;
		for (k = 0; k < 28; ++k)
		{
			if (num[k] > 0)
				j += (num[k] - 1) / i + 1;
		}
		if (j <= n)
		{
			cout << i << endl;
			for (k = 0; k < 28; ++k)
			{
				while (num[k] > 0)
				{
					cout << (char)(k + 'a');
					num[k] = num[k] - i;
				}
			}
			for (k = j; k < n; ++k)
				cout << 'a';
			cout << endl;
			return 0;
		}
		else
		{
			++i;
		}
	}
	
	cout << -1 << endl;

	return 0;
}
