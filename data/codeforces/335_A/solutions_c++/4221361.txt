#include <iostream>
#include <string>

using namespace std;

int cnt[30];
int main()
{
	string s, ans = "";
	cin >> s;
	int n;
	cin >> n;
	for (int i = 0; i < s.length(); i++)
		cnt[s[i] - 'a']++;
	int f = 1, e = 1010, b = -1;
	while (f < e)
	{
		int mid = (f + e) / 2;
		int tmp = 0;
		for (int i = 0; i < 26; i++)
			tmp += (cnt[i] + mid - 1) / mid;
		if (tmp <= n)
			b = e = mid;
		else
			f = mid + 1;
 	}
 	cout << b << endl;
 	if (b != -1)
 	{

		for (int i = 0; i < 26; i++)
		{
			int p = (cnt[i] + b - 1) / b;
			for (int j = 0; j < p; j++)
				ans += char('a' + i);
		}
		while (ans.length() < n)
			ans += 'a';
		cout << ans << endl;
 	}

}
