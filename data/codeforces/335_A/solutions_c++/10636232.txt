#include<cstdio>
using namespace std;
int a[26],n;
char c;
int ok(int k)
{
	int res=0;
	for(int i=0;i<26;i++) res+=(a[i]+k-1)/k;
	return res<=n;
}
int run()
{
	int m=0,s=0;
	for(int i=0;i<26;i++) if(a[i]) m++,s+=a[i];
	if(m>n) return -1;
	int l=1,r=1001,res=l+r>>1;
	while(l!=r)
	{
		if(ok(res)) r=res,res=l+r>>1;
		else l=res+1,res=l+r>>1;
	}
	return res;
}
void print(int k)
{
	for(int i=0;i<26;i++)
	{
		int j=0;
		for(;j<(a[i]+k-1)/k;j++) putchar('a'+i);
		n-=j;
	}
	while(n--) putchar('a');
	puts("");
}
int main()
{
	while(scanf("%c",&c),c!='\n') a[c-'a']++;
	scanf("%d",&n);
	int ans=run();
	printf("%d\n",ans);
	if(ans!=-1) print(ans);
	return 0;
}
