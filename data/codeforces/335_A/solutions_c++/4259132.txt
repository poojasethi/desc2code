#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<string>
#include<map>
#include<cmath>
#include<vector>
#include<queue>
using namespace std;

int cnt[30];
char s[10000];
string ans;

int jud(int n,int t)
{
	int i,j,c[30]={0};
	for(i=0;i<26;i++) c[i]=(cnt[i]+t-1)/t;
	int cc=0;
	for(i=0;i<26;i++) cc+=c[i];
	if(cc>n) return 0;
	ans.clear();
	for(i=0;i<26;i++) for(j=0;j<c[i];j++) ans.push_back('a'+i);
	for(i=ans.size();i<n;i++) ans.push_back('a');
	return 1;
}

int main()
{
	int n,t;
	cin>>s>>n;
	for(int i=0;s[i];i++) cnt[s[i]-'a']++;
	for(int i=1;i<2000;i++) if(jud(n,i))
	{
		cout<<i<<endl;
		cout<<ans<<endl;
		return 0;
	}
	cout<<-1<<endl;
	return 0;
}
