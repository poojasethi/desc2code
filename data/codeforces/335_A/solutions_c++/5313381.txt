#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int x,m,num[1200],i,j,n,k;
char c[1200];
int main(){
    scanf("%s%d",c+1,&m);
    n=strlen(c+1);
    for(i=1;i<=n;++i)num[c[i]-97]++;
    for(i=1;i<=1000;++i){
        x=0;
        for(j=0;j<26;++j)if(num[j]>0)x+=(num[j]-1)/i+1;
        if(x<=m){
            printf("%d\n",i);
            for(j=0;j<26;++j)if(num[j]>0)
                for(k=1;k<=(num[j]-1)/i+1;++k)printf("%c",j+97);
            for(j=1;j<=m-x;++j)printf("a");
            return 0;
        }
    }
    printf("-1\n");
}
