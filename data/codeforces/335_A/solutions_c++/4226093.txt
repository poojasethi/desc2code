#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char c[1005];
int cnt[30];
int main()
{
	scanf("%s",c);
	int len=strlen(c),i,j,n;
	scanf("%d",&n);
	for(i=0;i<len;i++)
	{
		cnt[c[i]-'a']++;
	}
	int high=1000,low=1,ans=-1;
	while(low<=high)
	{
		int mid=(low+high)/2;
		int tot_pos=n;
		for(i=0;i<26;i++)
		{
			if(cnt[i]==0)continue;
			tot_pos-=(cnt[i]-1)/mid+1;
		}
		if(tot_pos>=0)
		{
			high=mid-1;
			ans=mid;
		}
		else
		{
			low=mid+1;
		}
	}
	printf("%d\n",ans);
	if(ans!=-1)
	{
		int tot_pos=n;
		for(i=0;i<26;i++)
		{
			if(cnt[i]==0)continue;
			tot_pos-=(cnt[i]-1)/ans+1;
			for(j=(cnt[i]-1)/ans+1;j>0;j--)
			{
				printf("%c",'a'+i);
			}
		}
		for(i=0;i<tot_pos;i++)
		{
			printf("a");
		}
	}
	return 0;
}
