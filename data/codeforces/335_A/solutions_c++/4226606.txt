#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
string s;
int n;
int a[27];
bool che(int i)
{
	int ju[27];
	int sum=0;
	for(int j=0;j<26;j++)
	{
		ju[j]=(a[j]+i-1)/i;sum+=ju[j];
	}
	if (sum<=n)
	{
		cout<<i<<endl;
		int f=0;
		for(int j=0;j<26;j++)
		{
			if (ju[j]!=0)
			{
				if (!f)
				{
					ju[j]+=max(0,n-sum);
					f=1;
				}
				for(int k=0;k<ju[j];k++) cout<<char(j+'a');
			}
		}
		cout<<endl;
		return 1;
	}
	return 0;
}
int main()
{
	cin>>s>>n;
	int num=0;
	for(int i=0;i<s.length();i++) num+=(a[s[i]-'a']==0),a[s[i]-'a']++;
	if (n<num) return puts("-1")&0;
	int i=1;
	while(1)
	{
		if (i*n>=s.length() && che(i)) return 0;
		++i;
	}
}
