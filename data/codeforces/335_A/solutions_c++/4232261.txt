#include<cstdio>
int cnt[26];
int main(){
    char c;
    int i,sum,k,as;
    while(c=getchar()){
        if(c=='\n')break;
        cnt[c-'a']++;
    }
    scanf("%d",&k);
    for(as=1;as<=1000;as++){
        sum=0;
        for(i=0;i<26;i++)
            sum+=(cnt[i]+as-1)/as;
        if(sum<=k)break;
    }
    if(as>1000){puts("-1");return 0;}
    printf("%d\n",as);
    for(i=0;i<26;i++){
        sum=(cnt[i]+as-1)/as;
        k-=sum;
        while(sum--)putchar(i+'a');
    }
    while(k--)putchar('a');
    puts("");
    return 0;
}
