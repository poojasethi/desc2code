#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <string>
using namespace std;
string s;
int b[26],a[26],n;
int main()
{
    cin >>s;
    cin >>n;
    for (int i=0; i<s.size(); i++)
    {
        a[s[i]-'a']++;
    }
    for (int j=1; j<=2000; j++)
    {
        int res=0;
        for (int i=0; i<26; i++)
        {
            b[i]=a[i]/j+(a[i]%j==0?0:1);
            res+=b[i];
        }
        if (res<=n) 
        {
            b[0]+=n-res;
            cout <<j <<endl;
            for (int i=0; i<26; i++) cout <<string(b[i],'a'+i);
            cout <<endl;
            return 0;
        }
    }
    cout <<-1 <<endl;
    return 0;
}
