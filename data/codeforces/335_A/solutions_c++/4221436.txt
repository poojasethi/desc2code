#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;

int num[1100];
int ans[1100];
int jud(int n,int mid)
{
    int i;
    memset(ans,0,sizeof(ans));
    for(i='a';i<='z';i++)
	n-=(num[i]+mid-1)/mid,ans[i]=(num[i]+mid-1)/mid;
    return n>=0;
}
char s[1100];
int main()
{
    int i,j,n,len;
    scanf("%s",s);
    scanf("%d",&n);


    for(i=0;s[i];i++)
	num[s[i]]++;
    int l=1,r=110000,mid;
len=strlen(s);

    while(l<r)
    {
	mid=(l+r)/2;
	if(jud(n,mid))
	    r=mid;
	else
	    l=mid+1;
    }
    if(l>10000)
	puts("-1");
    else
    {
	printf("%d\n",l);
	jud(n,l);
	int cnt=0;
	for(i='a';i<='z';i++)
	{
	    cnt+=ans[i];
	    while(ans[i]--)
		printf("%c",i);
	}
	while(cnt<n)
	    cnt++,printf("%c",'a');
	puts("");
    }
}
