#include<stdio.h>
long i,j,k,x,r,y,n,f[150],z,R;
char s[1002],a[1002];
int main(){
 scanf("%s",s);
 scanf("%ld",&n);
 for(i=0;s[i];i++) f[s[i]]++;

 for(i='a';i<='z';i++) if(f[i]) j++;

 if(j>n){ printf("-1\n"); return 0; }



 x=1;
 y=1000000;

 dos:

 z=(x+y)/2;


  //           printf("%ld %ld %ld\n",x,y,z);


 r=0;
 for(i='a';i<='z';i++){ r+=f[i]/z; if(f[i]%z!=0) r++;  }

 if(r<=n){
  R=z;

  if(x!=y){
   y=z-1;
   if(y>0)
   goto dos;
  }

 }else{
  x=z+1;
  if(x<=y)
    goto dos;
 }

 k=0;
 for(j=0,i='a';i<='z';i++){ k+=f[i]/R; if(f[i]%R!=0) k++; while(j<k){ a[j]=i; j++; } }

 while(j<n){ a[j]='a'; j++; }

 printf("%ld\n%s\n",R,a);

}
