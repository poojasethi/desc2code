#include<iostream>
using namespace std;
int main()
{
	string s; int n; cin>>s>>n;
	int tot[200];
	for(int i = 0; i<200; ++i) tot[i] = 0;
	int sz = s.size();
	for(int i = 0; i<sz; ++i) ++tot[s[i]];
	int curr = 0;
	for(int i = 0; i<200; ++i) if(tot[i]) ++curr;
	if(curr > n) { cout<<-1<<endl; return 0; }
	int a[200], fin = 1;
	for(int l = 1; l <= 1000; ++l)
	{
		for(int i= 0; i<200; ++i) a[i] = 0;
		for(int i = 'a'; i<= 'z'; ++i)
			a[i] = ( tot[i]+l-1 )/l;
		int sum = 0;
		for(int i = 0; i<200; ++i) sum += a[i];
		if(sum <= n) { a['z'] += n - sum; fin = l; break; }
	}
	cout<<fin<<endl;
	for(int i = 0; i<200; ++i)
		for(int j = 0; j<a[i]; ++j)
			cout<<((char)(i));
	cout<<endl;
	return 0;
}
