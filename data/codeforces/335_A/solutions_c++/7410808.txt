#include <cstdio>
#include <iostream>
#include <string>
using namespace std;

int c[26];
string s;
int n;

int main(){
	//freopen("input.in", "r", stdin);
	cin >> s >> n;
	for(int i=0; i<s.size(); i++) c[s[i]-'a']++;
	int t=0;
	for(int i=0; i<26; i++) if(c[i]) t++;
	if(t > n){
		cout << -1;
		return 0;
	}
	int l = 1, r = 1000000, m;
	while(l < r){
		m = l+r>>1;
		int cnt = 0;
		for(int i=0; i<26; i++) cnt+=c[i]/m + (c[i]%m>0);
		if(cnt <= n) r=m;
		else l=m+1;
	}
	cout << l << '\n';
	for(int i=0; i<26; i++)
		for(int j=0; j<c[i]/l+(c[i]%l>0); j++){
			cout << char(i+'a');
			n--;
		}
	while(n--) cout << 'a';
}
