#include<iostream>
#include<string>
#include<vector>

using namespace std;

int main()
{
    string s;
    int n;
    cin>>s>>n;
    vector<int> count(128,0);
    for(unsigned i=0; i<s.size(); i++)   count[s[i]]++;
    //printf("a\n");
    for(int i=1; i<=1000 ; i++){
        string ans;
        for(int b='a'; b<='z'; b++){
            ans+=string((count[b]+i-1)/i,b);
        }
        if((int)ans.size()<=n){
            cout<<i<<endl;
            ans.resize(n,'a');
            cout<<ans<<endl;  
            //cin>>n;
            return 0;  
        }
        
    }
    cout<<"-1"<<endl;
    //cin>>s;
    return 0;    
}
