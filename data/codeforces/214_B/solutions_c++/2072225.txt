#include <cstdio>
#include <algorithm>

using namespace std;

int n;
int occ[10];
int p[3];
int sum;
int a,b;

void finish()
{
    bool ok=0;
    int i=9;
    while (i>=0)
    {
        if (occ[i]>0 && p[i%3]>0)
        {
            printf("%d",i);
            occ[i]--;
            p[i%3]--;
            if (i>0) ok=1;
            if (i==0 && !ok) exit(0);
            if (i==0 && !ok) exit(0);
        }
        else i--;
    }
    exit(0);
}

int main()
{
    scanf("%d",&n);
    for (a=1; a<=n; a++)
    {
        scanf("%d",&b);
        occ[b]++;
        p[b%3]++;
    }
    if (occ[0]==0) {printf("-1"); return 0;}

    for (a=1; a<=9; a++) sum+=occ[a]*a;

    if (sum%3==0) finish();
    if (p[sum%3]>0) p[sum%3]--, finish();
    if (p[1]>0) p[1]-=2; else p[2]-=2;
    finish();

}
