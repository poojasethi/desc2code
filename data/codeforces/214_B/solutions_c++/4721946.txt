#include<iostream>
using namespace std;
int n,d[10],a,u;
main()
{
    cin>>n;
    while(n--)cin>>a,++d[a],u+=a;
    if(u%3)
    {
        for(int i=1; i<10; ++i)
            if(i%3==u%3 && d[i])
                --d[i],u-=i,i=10;
        for(n=0; n<2; ++n)
            for(int i=1; i<10; ++i)
                if(u%3&&i%3&&d[i])
                    --d[i],u-=i,i=10;
    }
    if(!u)d[0]=1;
    if(!d[0]||u%3)cout<<-1;
    else
        for(int i=9; i>=0; --i)
            while(d[i]--)
                cout<<i;
    return 0;
}
