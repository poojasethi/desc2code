#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,v[100005];
    int suma=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>v[i];
        suma = suma + v[i];
    }
    sort(v,v+n);
    if(v[0]!=0){cout<<"-1\n";return 0;}
    if(suma%3==0){if(v[n-1]==0){cout<<"0\n";return 0;}for(int i=n-1;i>=0;i--)cout<<v[i];return 0;}
    int u=suma%3,k=3-u;
    int rr=0;
    for(int i=0;i<n;i++)
    {
        if(v[i]%3==u){v[i]=-1;rr=2;break;}
    }
    int i=0;
    while(rr!=2 && i<n ){if(v[i]%3==k){v[i]=-1;rr++;}i++;}
    int count=0;
    for(int i=0;i<n;i++){if(v[i]==0 || v[i]==-1){count++;}}
    if(count==n){cout<<"0\n";return 0;}
    for(int i=n-1;i>=0;i--){if(v[i]!=-1){cout<<v[i];}}
    cout<<"\n";
return 0;
}
