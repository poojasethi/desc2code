#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int n,c[10]={0},ans=0;

void Write()
{
	ans=0;
	for (int i=9;i>=0;i--)
		for (int j=0;j<c[i];j++)
		{
			printf("%d",i),ans++;
			if (ans==1&&i==0) break;
		}
}

void Dec(int a)
{
	for (int i=a;i<10;i+=3)
		if (c[i]>0){c[i]--; ans=0;break;}
	if (a==2&&ans!=0){
		for (int i=1;i<10;i+=3)
			for (;c[i]>0&&ans>0;c[i]--,ans--);} else 
	if (a==1&&ans!=0)
		for (int i=2;i<10;i+=3)
			for (;c[i]>0&&ans>=0;c[i]--,ans--);
}

int main()
{
	scanf("%d",&n);
	int s;
	for (int i=0;i<n;i++)
	{
		scanf("%d",&s);
		c[s]++;
		ans+=s,ans%=3;
	}
	if (c[0]==0) puts("-1"); else
		if (ans==0) Write(); else
	{
		if (ans==1)Dec(1); else Dec(2);
		Write();
	}
	return 0;
}
