#include <iostream>
using namespace std;

int main()
{
    int n, d[10] = {0}, i, c, sum = 0;
    cin >> n;
    for (i = 0; i < n; i++){
        cin >> c;
        d[c]++;
        sum += c;
    }
    if (sum % 3)
        for (i = 0; i < 10; i++)
            if (i%3 == sum%3 && d[i]) {d[i]--; sum-=i; break;}
    for (c = 0; c < 2; c++)
        if (sum % 3)
            for (i = 0; i < 10; i++)
                if (i % 3 && d[i])
                    {d[i]--; sum -= i;}
    if (sum == 0) d[0] = 1;
    if (!d[0] || sum%3) cout << -1;
    else 
        for (i = 9; i >= 0; i--)
            for (; d[i]--;)
                cout << i;
    cout<<endl;
}
