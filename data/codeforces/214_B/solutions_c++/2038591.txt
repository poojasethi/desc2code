#include <stdio.h>
int cnt[11],su,n;
int main() {
	int s=0,q=1,i,j;
	scanf("%d",&n);
	for(i=0; i<n; i++){
		scanf("%d",&su);
		cnt[su]++;
		s = s + su;
	}
	if(cnt[0] != 0){
		if(s%3 != 0){
			for(i=s%3; i<9; i+=3)
				if(cnt[i] != 0) {
					cnt[i]--;
					q=0;
					break;
				}
				if(q != 0) {
					su=2;
					for(j=s%3==2?1:2; j<9; j+=3){
						while(su && cnt[j]){
							cnt[j]--;
							su--;
						}
					}
				}
		}
		q=0;
		for(i=1; i<=9; i++){
			q=cnt[i]||q;
		}
		if(q != 0){
			for(i=9; i>=0; i--){
				for(j=0; j<cnt[i]; j++){
					printf("%d",i);
				}
			}
			printf("\n");
		}
		else{
			printf("0");
			printf("\n");
		}
		
	}
	else{
		printf("-1");
		printf("\n");
	}
	return 0;
}
