#include <stdio.h>
main() {
  int k[11]={},n,t,s=0,q=1;
  scanf("%d",&n);
  for(int i=0; i<n; i++)
    scanf("%d",&t),
    k[t]++,
    s+=t;
  if(k[0]) {
    if(s%3) {
      for(int i=s%3; i<9; i+=3)
        if(k[i]) {
          k[i]--;
          q=0;
          break;
        }
      if(q) {
        t=2;
        for(int i=s%3==2?1:2; i<9; i+=3)
          while(t && k[i])
            k[i]--,
            t--;
      }
    }
    q=0;
    for(int i=1; i<=9; i++)
      q=k[i]||q;
    if(q)
      for(int i=9; i+1; i--)
        for(int j=0; j<k[i]; j++)
          printf("%d",i);
    else
      puts("0");
  }
  else
    puts("-1");
  return 0;
}
