#include <iostream>
#include <string>
#include <string.h>
#include <math.h>
using namespace std;
int s[10],a[10],mlen=0,m[10];
void sub(int ind,int residue,int len)
{
	if (ind==0)
	{
		if (residue==0)
		{
			if (len>mlen)
			{
				mlen=len;
				for (int i=1;i<=9;i++) {m[i]=s[i];}
			}
		}
		return;
	}
	for (int i=a[ind];i>=max(0,a[ind]-2);i--)
	{
		s[ind]=i;
		sub(ind-1,(residue+ind*i)%3,len+i);
	}
}
int main()
{
	int n,i,j,k;
	memset(a,0,sizeof(a));
	memset(s,0,sizeof(s));
	cin>>n;
	for (i=1;i<=n;i++) {cin>>j; a[j]++;}
	if (a[0]==0)
	{
		cout<<-1<<endl;
	}
	else
	{
		sub(9,0,0);
		m[0]=a[0];
		bool f=1;
		for (j=9;j>=0;j--)
		{
			for (k=1;k<=m[j];k++) {f=0; cout<<j;}
			if (j==1 && f)
			{
				m[0]=1;
			}
		}
		cout<<endl;
	}
}
