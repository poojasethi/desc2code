# include <stdio.h>
# include <algorithm>

using namespace std;

int a,sum,k;
int b[100009];
int y[100009];
bool v[100009];

int main()
{
	scanf("%d",&a);
	
	for(int i=0;i<a;i++)
	{scanf("%d",&b[i]);sum+=b[i];}
	
	sort(b,b+a);
	
	if(b[0]!=0)
	{printf("-1");return 0;}
	
	for(int i=0;i<a;i++)
	if((sum-b[i])%3==0 && sum%3!=0)
	{
		for(int j=a-1;j>=0;j--)
		if(j!=i)
		{y[k]=b[j];k++;}
		
		if(y[0]==0)
		{printf("0");return 0;}
		
		for(int j=0;j<k;j++)
		printf("%d",y[j]);
		
		return 0;
	}
	
	for(int i=0;i<=a;i++)
	{
		if(sum%3==0)
		{
			for(int j=a-1;j>=0;j--)
			if(v[j]==0)
			{y[k]=b[j];k++;}
			
			if(y[0]==0)
			{printf("0");return 0;}
			
			for(int j=0;j<k;j++)
			printf("%d",y[j]);
			
			return 0;
		}
		
		if(b[i]%3!=0)
		{sum=sum-b[i];v[i]=1;}
	}
	
	getchar();
}
