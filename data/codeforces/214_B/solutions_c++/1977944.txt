#include<iostream>
using namespace std;

int c[10];
int main()
{
  int n;
  cin>>n;
  int sum = 0;
  for (int i = 0; i < n; i++) {
    int x;
    cin>>x;
    c[x]++;
    sum += x;
  }

  if (sum%3) for (int i = 0; i < 10; i++) if (c[i] && (i%3 == sum%3)) {sum -= i; c[i]--; break;}
  for (int n = 0; n < 2; n++) for (int i = 0; i < 10; i++) if (sum%3 && c[i] && i%3) sum -= i, c[i]--;
  if (sum == 0) c[0] = 1;
  if (!c[0] || sum%3) cout<<"-1";
  else for (int i = 9; i >= 0; i--) for (;c[i]; c[i]--) cout<<i;
  cout<<endl;
}
