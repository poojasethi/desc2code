#include <cstdio>
int main()
{
	int n,d,cnt[10]={0},sum=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&d);
		cnt[d]++;
		sum+=d;
	}
	if(cnt[0]==0)
		printf("-1\n");
	else
	{
		for(int i=sum%3;sum%3!=0&&i<10;i+=3)
			if(cnt[i]>=1)
			{
				cnt[i]--;
				sum-=i;
			}
		for(int i=3-(sum%3);sum%3!=0&&i<10;i+=3)
			if(cnt[i]>=1)
			{
				cnt[i]--;
				sum-=i;
				i-=3;
			}
		if(sum==0)
			printf("0\n");
		else
		{
			for(int i=9;i>=0;i--)
				for(int j=0;j<cnt[i];j++)
					printf("%d",i);
			printf("\n");
		}
	}
	return 0;
}
