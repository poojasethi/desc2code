#include <iostream>
using namespace std;

int main() {
	int n, sum = 0, t;
	int cnt[100010];
	cin>>n;
	for (int i = 0; i < n; i++) {
		cin>>t;
		cnt[t]++;
		sum += t;
	}
	if (cnt[0] == 0)
		cout<<-1<<endl;
	else {
		for (int i = sum%3; sum%3 != 0 && i < 10; i+=3) {
			if (cnt[i] > 0) {
				cnt[i]--;
				sum -= i;
			}
		}
		for (int i = 3-(sum%3); sum %3 != 0 && i < 10; i+=3) {
			if (cnt[i] > 0) {
				cnt[i]--;
				sum-=i;
				i-=3;
			}
		}
		if (sum == 0)
			cout<<0<<endl;
		else {
			for (int i = 9; i>=0; i--)
				for (int j = 0; j < cnt[i]; j++)
					cout<<i;
			cout<<endl;
		}
		
	}
	return 0;
}
