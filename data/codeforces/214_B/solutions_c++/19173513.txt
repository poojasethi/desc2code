#include<cstdio>
int m[2][3]={{1,4,7},{2,5,8}};
int d[10];
void solve(int k)
{
	int i,j;
	for(i=0;i<3;i++)
	{
		if(d[m[k][i]])
		{
			d[m[k][i]]--;
			return;
		}
	}
	for(i=0;i<3;i++)
	{
		if(d[m[!k][i]]>=2)
		{
			d[m[!k][i]]-=2;
			return;
		}
		for(j=0;j<i;j++)
		{
			if(d[m[!k][i]]&&d[m[!k][j]])
			{
				d[m[!k][i]]--;
				d[m[!k][j]]--;
				return;
			}
		}
	}
}
void print()
{
	int i,j,f=1;
	for(i=9;i>=0;i--)
	{
		for(j=0;j<d[i];j++)
		{
			printf("%d",i);
			if(!i&&f)return;
			f=0;
		}
	}
}
int main()
{
	int i,n,a,s;
	scanf("%d",&n);
	for(s=i=0;i<n;i++)
	{
		scanf("%d",&a);
		s+=a;
		d[a]++;
	}
	if(!d[0])printf("-1");
	else
	{
		if(s%3)solve(s%3-1);
		print();
	}
	return 0;
}
