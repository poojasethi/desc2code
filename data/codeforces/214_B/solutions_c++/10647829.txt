#include <bits/stdc++.h>
using namespace std;

bool ok(int sum, int i){
  return (sum % 3 != 0 && i < 10);
}

int main(){
  int n, arr[10] = {0}, sum = 0, foo;
  cin >> n;
  for (int i = 0; i < n; i++)
    cin >> foo, arr[foo]++, sum += foo;
  if (arr[0] > 0){
    int i = sum % 3;
    while (ok(sum, i)){
      if (arr[i])
        arr[i]--, sum -= i, i -= 3;
      i += 3;
    }
    //cout << sum << endl;
    i = 3 - sum % 3;
    while (ok(sum, i)){
      if (arr[i])
        arr[i]--, sum -= i, i -= 3;
      i += 3;
    }
    if (sum){
      for (int i = 9; i >= 0; i--)
        for (int j = 0; j < arr[i]; j++)
          cout << i;
      puts("");
    }
    else
      puts("0");
  }
  else
    puts("-1");
}
