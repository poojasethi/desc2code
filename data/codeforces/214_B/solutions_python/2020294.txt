#! -*-coding:utf-8 -*-

import sys
import math
import re

def solve():
    n=int(raw_input())
    a=map(int,raw_input().split(' '))
    # 最下位桁は0になる
    # 各桁の和が3の倍数になる
    dict={}
    sm=0
    for i in a:
        sm+=i
        if not i in dict:
            dict[i]=1
        else:
            dict[i]+=1
    if (not (0 in dict)):
        print -1
    elif (0 in dict) and dict[0]==n:
        print 0
    else:
        if sm % 3 == 0:
            a.sort()
            a.reverse()
            if a.count(0)==len(a):
                print 0
            else:
                print ("".join(map(str,a)))
        elif sm % 3 == 1:
            lt1=[i for i in a if i%3==1]
            lt2=[i for i in a if i%3==2]
            lt1.sort()
            lt2.sort()
            if len(lt1)>=1:
                a.remove(lt1[0])
                a.sort()
                a.reverse()
                if a.count(0)==len(a):
                    print 0
                else:
                    print ("".join(map(str,a)))
                return
            if len(lt2)>=2:
                a.remove(lt2[0])
                a.remove(lt2[1])
                a.sort()
                a.reverse()
                if a.count(0)==len(a):
                    print 0
                else:
                    print ("".join(map(str,a)))
                return
            print 0
        elif sm % 3 == 2:
            lt1=[i for i in a if i%3==1]
            lt2=[i for i in a if i%3==2]
            lt1.sort()
            lt2.sort()
            if len(lt2)>=1:
                a.remove(lt2[0])
                a.sort()
                a.reverse()
                if a.count(0)==len(a):
                    print 0
                else:
                    print ("".join(map(str,a)))
                return
            if len(lt1)>=2:
                a.remove(lt1[0])
                a.remove(lt1[1])
                a.sort()
                a.reverse()
                if a.count(0)==len(a):
                    print 0
                else:
                    print ("".join(map(str,a)))
                return
            print 0
solve()
