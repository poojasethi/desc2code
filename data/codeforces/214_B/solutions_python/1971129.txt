n = raw_input()
a = map(int, raw_input().split())
a.sort()

if a[0]!=0:
    print -1
    exit()

m = sum(a)%3
if m!=0:
    t = next((i for i, v in enumerate(a) if v%3==m), None)
    if t is None:
        del a[next((i for i, v in enumerate(a) if v%3==3-m), None)]
        del a[next((i for i, v in enumerate(a) if v%3==3-m), None)]
    else:
        del a[t]

a.reverse()
if a[0]==0:
    print 0
    exit()
print ''.join(map(str, a))
