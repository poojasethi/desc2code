n = input()
a = map(int, raw_input().split())
a.sort()

if a[0]!=0:
    print -1
    exit()

s = sum(a)%3

if s!=0:
    b = next((i for i, v in enumerate(a) if v%3==s), None)
    if b is None:
        del a[next((i for i, v in enumerate(a) if v%3==3-s), None)]
        del a[next((i for i, v in enumerate(a) if v%3==3-s), None)]
    else: del a[b]

a.reverse()
if a[0]==0: print 0
else: print ''.join(map(str,a))
