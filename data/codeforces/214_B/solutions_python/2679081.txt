def f(a):
	s = (len(a[1]) + len(a[2]) * 2) % 3
	if s:
		if len(a[s]):
			a[s].pop(0)
		else:
			a[3 - s] = a[3 - s][2 : ]
	r = ''.join(sorted(''.join(''.join(map(str, x)) for x in a)))[ : : -1]
	return [0, r][len(set(r)) > 1]
