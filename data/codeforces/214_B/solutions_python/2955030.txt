digit = sorted([int(i) for i in raw_input().split()])

if digit[0]:
	digit = [-1]
else:
	s = sum(digit)
	if s%3:
		for i in digit:
			if i%3 == s%3:
				digit.remove(i)
				break
		else:
			for i in digit:
				if i%3 :
					digit.remove(i)
					break
			for i in digit:
				if i%3 :
					digit.remove(i)
					break

while len(digit)>1 and digit[-1]==0:
	digit.pop()
print ''.join(map(str,digit[::-1]))
