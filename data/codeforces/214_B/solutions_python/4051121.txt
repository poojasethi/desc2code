import sys

n = int(raw_input())
line = raw_input()
strlist = line.split(' ', n)
intlist = map((lambda x : int(x)), strlist)

intlist.sort(reverse=True);   # max

if intlist[-1] != 0:        # 2, 5
    print -1
    sys.exit(0)
    
filterlist = filter((lambda x : x % 3), intlist)

residue = reduce((lambda x, y : x+y), filterlist, 0) % 3

if residue == 0:
    if intlist[0] == 0:        # leading zeros
        print 0
    else:
        for n in intlist:
            sys.stdout.write(str(n))
    sys.exit(0)
else:
    for n in reversed(filterlist):
        if n % 3 == residue:
            intlist.remove(n)
            if intlist[0] == 0:        # leading zeros
                print 0
            else:
                for n in intlist:
                    sys.stdout.write(str(n))
            sys.exit(0)
    tmp = 0
    for n in reversed(filterlist):
        if n % 3 == 3 - residue:
            if tmp == 0:
                tmp = n
            else:
                intlist.remove(tmp)
                intlist.remove(n)
                if intlist[0] == 0:        # leading zeros
                    print 0
                else:
                    for n in intlist:
                        sys.stdout.write(str(n))
                sys.exit(0)
                
print -1
