#! -*-coding:utf-8 -*-
def solve():
    n=int(raw_input())
    a=map(int,raw_input().split(' '))
    dict={}
    sm=sum(a)
    for i in a:
        if not i in dict:dict[i]=1
        else:dict[i]+=1
    if (not (0 in dict)):print -1
    elif ((0 in dict) and dict[0]==n):print 0
    else:
        lt=[]
        lt.append(sorted([i for i in a if i%3==1]))
        lt.append(sorted([i for i in a if i%3==2]))
        if sm % 3 == 0:
            a.sort(reverse=True)
            if a.count(0)==len(a):print 0
            else:print ("".join(map(str,a)))
        else:
            if len(lt[sm%3-1])>=1:
                a.remove(lt[sm%3-1][0])
                a.sort(reverse=True)
                if a.count(0)==len(a):print 0
                else:print ("".join(map(str,a)))
                return
            if len(lt[(sm%3-1)^1])>=2:
                a.remove(lt[(sm%3-1)^1][0])
                a.remove(lt[(sm%3-1)^1][1])
                a.sort(reverse=True)
                if a.count(0)==len(a):print 0
                else:print ("".join(map(str,a)))
                return
            print 0
solve()
