if __name__=="__main__":
    n, A = map(int, raw_input().split())
    d = map(int, raw_input().split())

    sumd = sum(d)

    if n==1:
        print d[0]-1
        exit()
    else:
        b = [str(x) for x in d]
        c=0
        while c < n:
            r = 0
            Ac = sumd-d[c]
            r += max(0,A-Ac-1)
            r += max(0,d[c]- (A-(n-1)))
            b[c] = str(r)
            c += 1
        print ' '.join(b)
