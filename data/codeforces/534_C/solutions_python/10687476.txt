

import sys
import math


n, A = map(int, sys.stdin.readline().rstrip().split())
dices = map(int, sys.stdin.readline().rstrip().split())

s = sum(dices)
dicemin = len(dices)

res = []
for i in dices:
	other_dice_sum = s-i
	if other_dice_sum >= A - 1:
		minimum = 1
	else:
		minimum = A - other_dice_sum
	maximum = min(A - (dicemin - 1), i)
	res.append(i - (maximum - minimum + 1))

print " ".join(map(str, res))
