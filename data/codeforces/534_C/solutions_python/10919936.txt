n, A = map(int, raw_input().split())
d = map(int, raw_input().split())

sum = reduce(lambda s, v: s + v, d)

result = []
for _d in d:
    max_d = min(_d, A - (n - 1))
    min_d = max(1, A - (sum - _d))
    result.append(_d - (max_d - min_d + 1))

s = ''
for i in xrange(0, n):
    s += str(result[i])
    if i < n - 1:
      s += ' '

print s
