inp = raw_input().split()
n = int(inp[0])
A = int(inp[1])
dice = [int(i) for i in raw_input().split()]
cs = dice[0]
for i in xrange(1,n):
    cs += dice[i]
for i in xrange(0,n):
    y = cs - dice[i]
    ans = 0
    if(y<A):
        ans += A-y-1
    if(dice[i]+n-1>A):
        ans+=dice[i]+n-1-A
    print ans,
