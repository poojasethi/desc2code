#!/usr/bin/env python

n, A = map(int, raw_input().split())
d = map(int, raw_input().split())

s = sum(d)

res = []
for i in xrange(n):
	res.append(0)

for i in xrange(n):
	if d[i] > A - (n - 1):
		res[i] += d[i] - A + n - 1

	if A - s + d[i] - 1 > 0:
		res[i] += A - s + d[i] - 1

for x in range(len(res)):
	print res[x],
