#!/usr/bin/python

import sys

def Nil(): return map(int, raw_input().split())

n, A = Nil()
d = Nil()

#print n, A
s = sum(d)
#print s
answer = []

for i in range(n):
    maxOfOthers = s - d[i]
    if maxOfOthers < A:
        Min = (A - maxOfOthers)
    else:
        Min = 0

    Min = max(Min, 1)

    minOfOthers = n - 1
    Max = min(d[i], A - minOfOthers)
       
    # print "min", "max", Min, Max, d[i]
    r = Min - 1 + (d[i] - Max)
    answer.append(str(r))

print " ".join(answer)
