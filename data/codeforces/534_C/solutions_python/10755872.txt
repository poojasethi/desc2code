n,A = list(int(foo) for foo in raw_input().split())

d = list(int(foo) for foo in raw_input().split())

total = sum(d)
dices = len(d)

def ans(i,total,dices):
    # which values would be too small?
    other_max = total - i

    # which values would be too big?
    other_min = dices - 1

    mini = A - other_max
    maxi = A - other_min
    
    if mini < 1:
        mini = 1
    if maxi > i:
        maxi = i
        
    if maxi < mini:
        return i
    else:
        return i - (maxi - mini + 1)
print ' '.join(str(ans(i,total,dices)) for i in d)
