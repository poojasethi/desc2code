import sys

inp = sys.stdin.readlines()
n, a = map(int, inp[0].split())
ds = list(map(int, inp[1].split()))

max_val = sum(ds)
min_val = len(ds)

ret = []
for i in range(n):
    max_without = max_val - ds[i]
    min_without = min_val - 1

    min_of = max(a - max_without, 1)
    max_of = min(a - min_without, ds[i])

    ret.append((ds[i] - max_of) + (min_of - 1))

print ' '.join(str(x) for x in ret)
