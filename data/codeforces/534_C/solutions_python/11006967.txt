n,A = raw_input().split()
n,A = int(n),int(A)

l = [int(i) for i in raw_input().split()]

s = sum(l)
ans = []
for i in l:
    temp = s-i
    max_val = i
    min_val = 1
    if temp < A:
        min_val = A-temp
    temp = A - n + 1
    if temp < i:
        max_val = temp
    ans.append(i - (max_val - min_val + 1))

print ' '.join(map(str,ans))
