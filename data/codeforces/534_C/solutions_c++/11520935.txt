#include<iostream>
#include<algorithm>
using namespace std;

int main(){
   long long int d[200005],i,sum=0,n,a,maxi,mini,y=1;
   cin>>n>>a;
   for(i=0;i<n;++i)
   { 
     cin>>d[i];
     sum+=d[i];
    }
   for(i=0;i<n;++i){
      mini=max(y,a-(sum-d[i]));
      maxi=min(d[i],a-(n-1));
      cout<<d[i]-(maxi-mini)-1<<" ";
    }
   
  return 0;
 }
