#include<bits/stdc++.h>
using namespace std;
typedef long long lint;
const int maxn=200000+100;
lint d[maxn];

int main(){
	lint n;
	lint A, s=0;
	cin >> n >> A;
	for(int i=0;i<n;i++)
		cin >> d[i], s+=d[i];
	for(int i=0;i<n;i++){
		lint l=max(1ll, A+d[i]-s);
		lint r=min(d[i], A-n+1);
		cout << d[i]-(r-l+1) << " ";
	}
	cout << endl;
}
