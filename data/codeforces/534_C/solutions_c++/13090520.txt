#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

ll d[200006];
int main(){

ll n,a,i,s,m;

cin>>n>>a;
m=0;
for(i=0;i<n;i++)
{cin>>d[i]; m+=d[i];}

for(i=0;i<n;i++)
{ s=0;
if(d[i]>a-n+1)
    s+=d[i]-a+n-1;
    if(m-d[i]<a)
    s+=a-m+d[i]-1;

    cout<<s<<" ";
}
cout<<endl;







    return 0;
}
