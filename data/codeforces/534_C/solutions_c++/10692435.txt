#include <bits/stdc++.h>
using namespace std;

long long n,A,summ,mas[9999999],p;

int main(){
    cin >> n >> A;

    for(long long i=0;i<n;++i){
        cin >> i[mas];
        summ += i[mas];
    }
    for (long long i=0;i<n;++i)
        cout << max(A-summ+i[mas]-1,p)+max(i[mas]-A+n-1,p) << " ";
    return 0;
}
