#include <iostream>

using namespace std;

#define LL long long
LL nums[200005];

int main()
{

	LL n, A;
	LL sum = 0;
	cin >> n >> A;

	for(int i=1; i<=n; i++)
	{
		cin >> nums[i];
		sum += nums[i];
	}

	for(int i=1; i<=n; i++)
	{
		int low = max(1ll, A-sum+nums[i]);
		int high = min(nums[i], A-n+1);

		cout << nums[i] - high+low-1 << " ";

	}






}
