#include <bits/stdc++.h>
using namespace std;
long long a[1<<18];
int main()
{
	long long n,m;
	long long s=0LL;
	cin >> n >> m;
	for(int x=0;x<n;++x)
	{
		cin >> a[x];
		s+=a[x];
	}
	for(int x=0;x<n;++x)
	{
		long long r=min(m-n+1LL,a[x]);
		long long l=max(1LL,m-(s-a[x]));
		cout << a[x]-(r-l+1LL) << " ";
	}

}
