#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long i,n,d,sum=0;
    cin>>n>>d;
    long long a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        sum+=a[i];
    }
    for(i=0;i<n;i++)
        cout<<a[i]-(min(a[i],d-(n-1))-max(1ll,d-(sum - a[i]))+1)<<" ";
    return 0;
}
