#include <stdio.h>
int a[200002];
__int64 sum;

int main(){
	int n,c;
	__int64 S;
	scanf("%d %I64d",&n,&S);
	for(int i  =0; i < n; i ++){
		scanf("%d",&a[i]);
		sum += a[i];
	}
	for(int i = 0; i < n; i ++){
		c = 0;
		if(n - 1 + a[i] > S) c += n - 1 + a[i] - S;
		if(S > sum - a[i]) c += S - (sum - a[i]) - 1;
		printf("%d ",c);
	}
}
