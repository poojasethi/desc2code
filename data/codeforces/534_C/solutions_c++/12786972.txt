#include <bits/stdc++.h>

using namespace std;

int main()
{
  long long n, A;
  cin>>n>>A;
  long long sum=0, a[n], aux;
  for(int i=0;i<n;i++)
    cin>>a[i], sum+=a[i];
  for(int i=0;i<n;i++)
  {
    aux=A-sum+a[i]-1;
    if(A-sum+a[i]-1<0)
      aux=0;
    if(a[i]>A-n+1)
      aux+=a[i]-A+n-1;
    cout<<aux<<' ';
  }
}
