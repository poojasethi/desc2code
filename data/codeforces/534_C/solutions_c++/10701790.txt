#define N (1<<18)
#define ll long long
#include <bits/stdc++.h>
using namespace std;

ll m,s;
int n,a[N];

int main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)
		scanf("%d",a+i),s+=a[i];
	for(int i=0;i<n;i++)
	{
		int l=max(1ll,m-(s-a[i]));
		int r=min((ll)a[i],m-n+1);
		printf("%d ",a[i]-(r-l+1));
	}
	cout<<endl;
	return 0;
}
