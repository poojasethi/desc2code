#include<bits/stdc++.h>
using namespace std;

int main()
{long long int n,s,a[200004],sum=0,x1,x2;
	cin>>n>>s;

	
	
	for(long long int i=0;i<n;i++)
	{
	cin>>a[i];
	sum+=a[i];
}
		

	for(long long int i=0;i<n;i++)
	{
		long long int x=0;
		x1=s-n+1;
		x2=s-(sum-a[i]);
		if(x1<a[i])
		x+=a[i]-x1;
		if(x2>0)
		x+=x2-1;
		cout<<x<<" ";
	}

	

}
