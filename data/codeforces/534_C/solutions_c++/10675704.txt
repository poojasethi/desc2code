#include <iostream>
using namespace std;

long long d[222222];
long long n, A, s;
int main()
{
	cin >> n >> A;
	for(long long i=0; i<n; i++)
	{
		cin >> d[i];
		s += d[i];
	}

	for(long long i=0; i<n; i++)
	{
		long long mn = max(1ll, A-s+d[i]);
		long long mx = min(d[i], A-n+1ll);

		cout << d[i]-mx+mn-1ll << ' ';
	}

	cout << endl;

	return 0;
}
