#include<iostream>
using namespace std;
#define NMAX 200005
long long n,s,d,A[NMAX],inf,sup;
int main()
{
    long long i;
    cin>>n>>s;
    for (i=1;i<=n;++i)
        cin>>A[i], d+=A[i];
    for (i=1;i<=n;++i)
    {
        inf=max(1LL,s-d+A[i]);
        sup=min(A[i],s-n+1);
        cout<<A[i]-(sup-inf+1)<<" ";
    }
    return 0;
}
