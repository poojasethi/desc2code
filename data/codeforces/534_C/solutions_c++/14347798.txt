#include <bits/stdc++.h>

using namespace std;

int main() 
{
	long long sum=0,n,A;
	scanf("%lld%lld",&n,&A); 
	long long arr[n];
	for(int i=0; i<n; i++) scanf("%lld",&arr[i]),sum+=arr[i];
	for (int i = 0; i < n; ++i) printf("%lld ",arr[i]-(min(arr[i],A-n+1)-max(1LL,A-sum+arr[i])+1) );
	printf("\n");

	return 0;
}
