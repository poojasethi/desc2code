#include<iostream>
using namespace std;
long long a[1000010];
int main()
{
	long long n,A,sum=0;
	cin>>n>>A;
	for(long long i=1;i<=n;i++)
	{
		cin>>a[i];
		sum+=a[i];
	}
	for(long long i=1;i<=n;i++)
	{
		long long m1=min(a[i],A-n+1);
		long long rest=sum-a[i];
		long long m2=max((long long)1,(A-rest));
//		cout<<m1<<" "<<m2;
		cout<<a[i]-m1+m2-1<<" ";
	}
	return 0;
}
