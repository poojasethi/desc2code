#include<cstdio>

long long n,A,t;
long long d[200010],s;

long long pos(long long r){ return r>0?r:0; }

int main(){
  scanf("%I64d%I64d",&n,&A);
  for(long long i=0;i<n;i++){
    scanf("%I64d",d+i);
    s=s+d[i];
  }
  for(long long i=0;i<n;i++) printf("%I64d ",pos(A-s+d[i]-1)+pos(d[i]-A+n-1));
  return 0;
}
