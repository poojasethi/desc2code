#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;


long long int n,s,a[300000],r,l,z=0,k;

int main()
{
cin >> n >> s;
for(int i=1;i<=n;i++)
{
cin >> a[i];
z+=a[i];
}
for(int i=1;i<=n;i++)
{
if(a[i]+n-1<=s)
r=a[i];
else
r=s-n+1;

if(z-a[i]+1>=s)
l=1;
else
l=s-z+a[i];

//cout<<r<<" "<<l<<endl;

cout<<a[i]-r+l-1<<" ";


}

return 0;
}
