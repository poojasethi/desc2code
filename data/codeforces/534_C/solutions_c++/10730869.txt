#include<iostream>
using namespace std;
int main(){
	long long n,A,sum=0;
	cin>>n>>A;
	long long d[200000]={};
	for(int i=0;i<n;i++){
		cin>>d[i];
		sum+=d[i];
	}
	long long ans;
	for(int i=0;i<n;i++){
		ans=0;
		if(-sum+A+d[i]-1 > 0)ans=-sum+A+d[i]-1;
		if(n-1+d[i]-A > 0)ans+=n-1+d[i]-A;
		cout<<ans<<" ";
	}
}
