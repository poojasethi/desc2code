#include<bits/stdc++.h>
using namespace std;
typedef long long lint;
const int N=2e5+20;
long long int d[N],n,a,sum=0,fi,se;
void input(){
	cin >> n >> a;
	for(int i=0;i<n;i++)
		cin >> d[i];
	return;
}
int main(){
	input();
	for(int i=0;i<n;i++)
		sum+=d[i];
	for(int i=0;i<n;i++){
		fi=max(1ll,a+d[i]-sum),se=min(d[i],a-n+1);
		cout <<d[i]-(se-fi+1) <<" ";
	}
}
