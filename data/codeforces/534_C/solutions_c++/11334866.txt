#include <bits/stdc++.h>
using namespace std;

long long n,a,arr[200005],ans,tot;

int main(){
	cin >> n >> a;
	for (int i=1;i<=n;i++){cin >> arr[i]; tot+=arr[i];}
	for (int i=1;i<=n;i++){
		ans=max((long long)0,a-(tot-arr[i])-1);
		ans+=max((long long)0,arr[i]-(a-(n-1)));
		cout << ans;
		if (i==n) cout << "\n"; else cout << " ";
	}
}
