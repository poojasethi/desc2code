#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll a[200010];

int main()
{
	ll n,s,sum=0,ans,v;
	cin>>n>>s;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		sum+=a[i];
	}
	for(int i=0;i<n;i++)
	{
		ans=0;
		v=s-(sum-a[i]);
		if(v>0)
			ans+=s-(sum-a[i])-1;
		if(a[i]>s-(n-1))
			ans+=a[i]-(s-(n-1));
		cout<<ans<<" ";
	}
	cout<<endl;
	return 0;
}
