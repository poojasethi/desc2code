#include <cstdio>
#include <algorithm>
using namespace std;
int n,i,a[1000005],b,c;
long long g,m;
int main(){
	scanf("%d%I64d",&n,&m);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++)g+=a[i];
	for(i=1;i<=n;i++){
		c=min(1ll*a[i],m-(n-1));
		b=max(1ll,m-(g-a[i]));
		printf("%d ",a[i]-(c-b+1));
	}
	return 0;
}
