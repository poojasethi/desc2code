#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
long long n,A,s,d[200005];
void nhap()
{
    cin>>n>>A;
    long long i;
    for(i=1;i<=n;i++) {cin>>d[i];s+=d[i];}
}
void xuli()
{
    long long i;
    for(i=1;i<=n;i++) cout<<max(LL(0),A-(s-d[i]+1))+max(LL(0),d[i]-(A-n+1))<<" ";
}
main()
{
    nhap();
    xuli();
}
