#include<stdio.h>
#include<algorithm>
using namespace std;
int x[200000];
int main(){
	int n;
	long long m;
	while(scanf("%d%I64d",&n,&m)==2){
		long long mm=m;
		for(int i=0;i<n;i++){
			scanf("%d",x+i);
			m-=x[i];
		}
		m*=-1;
		for(int i=0;i<n;i++){
			printf("%I64d%c",max(x[i]-m-1,0ll)+max(x[i]-mm+n-1,0ll),i==n-1?'\n':' ');
		}
	}
}
