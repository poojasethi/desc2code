#include<bits/stdc++.h>
using namespace std;
typedef long long ll;


int main()
{
	ios::sync_with_stdio(false);
	int n;
	ll req;
	cin>>n>>req;
	ll sum=0;
	ll a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		sum+=a[i];
	}
	for(int i=0;i<n;i++)
	{
		ll l=max(1ll,req-(sum-a[i]));
		ll r=min(0ll+a[i],req-(n-1));
		ll k=a[i]-(r-l+1);
		cout<<k<<" ";
	}
	cout<<"\n";
	return 0;
}
