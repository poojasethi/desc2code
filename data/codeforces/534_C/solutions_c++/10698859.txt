#include <bits/stdc++.h>

using namespace std;

const int MAXN = 5+2e5;

long long dice[MAXN];

int main(){
  int n;
  long long a, s;
  cin >> n >> a;
  s = 0;
  for(int i = 0; i < n; i++){
    cin >> dice[i];
    s += dice[i];
  }
  for(int i = 0; i < n; i++){
    cout << max(0ll, a - s + dice[i] - 1) + max(0ll,dice[i] - a + n - 1) << " ";
  }
  cout << endl;
}
