#include <iostream>
#include <cstdio>
using namespace std;
typedef long long ll;
int n,b[200001];
ll a,s;
int main()
{
	int i;
	scanf("%d%I64d",&n,&a);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&b[i]);
		s+=b[i];
	}
	for(i=1;i<=n;i++)
	{
		int l=max(a-(s-b[i]),1LL),r=min(a-(n-1),(ll)b[i]);
		printf("%d ",b[i]-(r-l+1));
	}
	puts("");
	return 0;
}
