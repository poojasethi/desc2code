#include<bits/stdc++.h>
using namespace std;
int main(){
	long long int n,top,ar[1000000],dp=0,mx,mn;
	scanf("%lld %lld",&n,&top);
	for(int i=1 ; i<=n ; i++){
		scanf("%lld",&ar[i]);
		dp+=ar[i];
	}
	for(int i=1 ; i<=n ; i++){
		mx=max(top-dp+ar[i],1ll);
		mn=min(top-n+1,ar[i]);
		printf("%lld ",ar[i]-mn+mx-1);
	}
	return 0;
}
