#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n,dum,A;
vector<ll>d;

int main()
{
	cin>>n>>A;
	d.push_back(0);
	ll s=0;
	for(ll i=0;i<n;i++)
	{
		cin>>dum; s+=dum;
		d.push_back(dum);
	}
	for(ll i=1;i<=n;i++)
	{
		ll l=1; if(A-(s-d[i])>l) l=A-(s-d[i]);
		ll u=d[i]; if(A-(n-1)<u) u=A-(n-1);
		ll pos = u-l+1;
		cout<<d[i]-pos<<" ";
	}
	return 0;
}
