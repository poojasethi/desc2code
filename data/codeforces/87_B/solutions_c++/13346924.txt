#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define MAXN 100005
#define mp make_pair
#define INF 400000000000LL

string a, b;
int n;
map<string, int> mapa;

void imprimir(int x) {
	if (x < 0) {
		cout << "errtype";
	} else {
		cout << "void";
		for (int i = 0; i < x; i++) {
			cout << "*";
		}
	}
	cout << endl;
}

int cant(string a, char car) {
	int cont = 0;
	for (int i = 0; i < a.length(); i++) {
		if (a[i] == car) {
			cont++;
		}
	}
	return cont;
}

string word(string x) {
	string sol = "";
	for (int i = 0; i < x.length(); i++) {
		if (x[i] != '*' && x[i] != '&') {
			sol += x[i];
		}
	}
	return sol;
}

int main() {
	cin >> n;
	mapa["void"] = 0;
	for (int i = 0; i < n; i++) {
		cin >> a;
		if (a == "typedef") {
			cin >> a >> b;
			string w = word(a);
			if (mapa.find(w) == mapa.end() || mapa[w] < 0) {
				mapa[b] = -1;
			} else
				mapa[b] = mapa[w] + (cant(a, '*') - cant(a, '&'));
		} else {
			cin >> a;
			string w = word(a);
			if (mapa.find(w) == mapa.end() || mapa[w] < 0) {
				imprimir(-1);
			} else {
				int x = mapa[w] + (cant(a, '*') - cant(a, '&'));
				imprimir(x);
			}
		}
	}
	return 0;
}
