#include<iostream>
#include<cstdio>
#include<map>
#include<string>
#include<cstring>

#define Max 1000005
#define INF Max

using namespace std;

char str[Max], tmp[Max];

map<string, int> m;

int analysis()
{
	int i, j, t, l;

	l = strlen(str);

	for(i=0;i<l;i++)
		if('a' <= str[i] && str[i]<='z')
			break;

	for(j=i;j<l;j++)
		if(!('a' <= str[j] && str[j]<='z'))
			break;

	for(int k=i;k<j;k++)
		tmp[k-i] = str[k];
	tmp[j-i] = 0;

	t = m.count(tmp) > 0 ? m[tmp] : -INF;

	t -= i;
	t += l - j;

	if(t < 0)
		t = -INF;

	return t;
}

void output(int t)
{
	//printf("!");
	if(t < 0)
	{
		printf("errtype\n");
		return ;
	}
	printf("void");
	for(int i=0;i<t;i++)
		printf("*");
	printf("\n");
}

int main()
{
	int n;

	m["void"] = 0;
	m["errtype"] = -INF;

	scanf("%d", &n);

	while(n--)
	{
		scanf("%s", str);
		if(str[4] == 'd')
		{
			scanf("%s", str);

			int t = analysis();

			scanf("%s", str);

			m[str] = t;
		}
		else
		{
			scanf("%s", str);

			int t = analysis();

			output(t);
		}
	}
}
