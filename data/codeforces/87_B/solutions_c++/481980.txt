#include<cstdio>
#include<cstring>
#include<string>
#include<iostream>
#include<map>
using namespace std;
int main() {
	int t;
	scanf("%d", &t);
	string st, st1, st2;
	map <string, string> mpstr;
	map <string, int> mp;
	mpstr["void"] = "void";
	while(t--) {
		cin >> st;
		if(st == "typedef") {
			cin >> st1 >> st2;
			int cnt = 0;
			int i = -1, j = st1.length();
			while(st1[++i] == '&') cnt--;
			while(st1[--j] == '*') cnt++;
			st1 = st1.substr(i,j - i + 1);
			mp[st2] = mp[st1] + cnt;
			if(mp[st2] < 0 || mpstr[st1] != "void") {
				mpstr[st2] = "";
			} else {
				mpstr[st2] = "void";
			}
		} else {
			cin >> st1;
			int cnt = 0;
			int i = -1, j = st1.length();
			while(st1[++i] == '&') cnt--;
			while(st1[--j] == '*') cnt++;
			st1 = st1.substr(i,j - i + 1);
			cnt += mp[st1];
			if(mpstr[st1] != "void" || cnt < 0) {
				cout<<"errtype\n";
				continue;
			}
			cout<<mpstr[st1];
			if(cnt > 0) {
				for(int i = 0; i < cnt; i++)
					printf("*");
			}
			cout<<"\n";
		}
	}
	return 0;
}
