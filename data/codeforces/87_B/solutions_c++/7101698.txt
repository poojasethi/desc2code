#include <bits/stdc++.h>
using namespace std;

int n;
map<string,int> m;
string t, a, b;

int main(){
  m["void"] = 0;
  cin >> n;
  for(int i = 0; i < n; i++){
    cin >> t >> a;
    if(t == "typedef"){
      cin >> b;
      string c = "";
      int ref = 0;
      for(int i = 0; i < a.size(); i++){
        if(a[i] == '&') ref--;
        else if(a[i] == '*') ref++;
        else c += a[i];
      }
      if(m.find(c) == m.end() || m[c] < 0) m[b] = -1;
      else m[b] = m[c] + ref;
    } else {
      string c = "";
      int ref = 0;
      for(int i = 0; i < a.size(); i++){
        if(a[i] == '&') ref--;
        else if(a[i] == '*') ref++;
        else c += a[i];
      }
      if(m.find(c) == m.end() || m[c] < 0 || m[c] + ref < 0) cout << "errtype";
      else {
        cout << "void";
        for(int i = 0; i < m[c] + ref; i++) cout << '*';
      }
      cout << '\n';
    }
  }
}
