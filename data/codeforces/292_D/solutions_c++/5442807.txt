#include <stdio.h>
#include <iostream>
using namespace std;
struct node{
	int p[600],s;
	void init(int n){
		s=n;
	}
	int find(int x){
		if (p[x])
			return p[x]=find(p[x]);
		return x;
	}
	void add(int x,int y){
		int a=find(x);
		int b=find(y);
		if (a!=b){
			p[a]=b;
			s--;
		}
	}
}l[10005],r[10005];
int n,m,x[10005],y[10005],a,b,q;
int main( ){
	cin>>n>>m;
	for (int i=1;i<=m;++ i)
		cin>>x[i]>>y[i];
	l[0].init(n);
	r[m+1].init(n);
	for (int i=1;i<=m;++i){
		l[i]=l[i-1];
		l[i].add(x[i],y[i]);
	}
	for (int i=m;i>=1;--i){
		r[i]=r[i+1];
		r[i].add(x[i],y[i]);
	}
	cin>>q;
	for (int i=0;i<q;++i){
		cin>>a>>b;
		node tmp=l[a-1];
		for (int j=1;j<=n;++j)
			if (r[b+1].p[j])
				tmp.add(j,r[b+1].p[j]);
		cout<<tmp.s<<endl;
	}
	return 0;
}
