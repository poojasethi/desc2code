#include <cstdio>
#include <vector>
using namespace std;
const int maxn=505;
int n,m;
vector< pair<int,int> > nei[maxn];
int k,l,r,ans;
bool taken[maxn];
void dfs(int v)
{
	 taken[v]=1;
	 for (int i=0;i<nei[v].size();i++)
     {
		  pair<int,int> u=nei[v][i];
		  if ((u.second<l||u.second>r) && !taken[u.first]) dfs(u.first);
	 }
}
int main()
{
	 scanf("%d%d",&n,&m);
	 for (int i=1;i<=m;i++)
     {
		  int a,b;
          scanf("%d%d",&a,&b);
	 	  nei[a].push_back(make_pair(b,i));
          nei[b].push_back(make_pair(a,i));
	 }
	 scanf("%d",&k);
	 while (k--)
     {
		  scanf("%d %d",&l,&r);
		  fill(taken+1,taken+n+1,0);
		  ans=0;
		  for (int i=1;i<=n;i++)
          {
               if (!taken[i])
               {
	                ans++;dfs(i);
		       }
          }
		  printf("%d\n",ans);
	 }
}
