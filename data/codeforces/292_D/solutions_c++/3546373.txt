#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int u=510,w=10010;
int a[w],b[w],qian[u],hou[u],fa[u],n,m,t,q,h,i,x,y,ans;

int get(int x)
{
	if(x==fa[x]) return x;
	return fa[x]=get(fa[x]); 
}

int main()
{
	cin>>n>>m;
	for(i=1;i<=m;i++)
		scanf("%d%d",&a[i],&b[i]);
	for(i=1;i<=n;i++) fa[i]=i;
	for(i=1;i<=m;i++)
	{
		x=get(a[i]),y=get(b[i]);
		if(x!=y) fa[y]=x,qian[++q]=i;
	}
	for(i=1;i<=n;i++) fa[i]=i;
	for(i=m;i;i--)
	{
		x=get(a[i]),y=get(b[i]);
		if(x!=y) fa[y]=x,hou[++h]=i;
	}
	cin>>t;
	while(t--)
	{
		scanf("%d%d",&x,&y);
		for(i=1;i<=n;i++) fa[i]=i;
		for(i=1;i<=q;i++)
			if(qian[i]<x) fa[get(a[qian[i]])]=get(b[qian[i]]); else break;
		for(i=1;i<=h;i++)
			if(hou[i]>y) fa[get(a[hou[i]])]=get(b[hou[i]]); else break;
		ans=0;
		for(i=1;i<=n;i++)
			if(fa[i]==i) ans++;
		printf("%d\n",ans);
	}
}
