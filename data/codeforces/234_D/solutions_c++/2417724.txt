#include<stdio.h>
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,m,k,i,j,like[200],w,films[200][2],d;
	char w1[100];
	scanf("%d%d",&m,&k);
	for(i=0;i<200;i++)
		like[i]=0;
	for(i=1;i<=k;i++)
	{
		scanf("%d",&w);
		like[w]=1;
	}
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%s%d",w1,&d);
		films[i][0]=films[i][1]=0;
		for(j=1;j<=d;j++)
		{
			scanf("%d",&w);
			if(w&&like[w])films[i][0]++;
			if(!like[w]&&w)films[i][1]++;
		}
		films[i][1]=d-films[i][1];
		if(films[i][1]>k)films[i][1]=k;
		if(films[i][0]<d-m+k)films[i][0]=d-m+k;
	}
	for(i=1;i<=n;i++)
	{
		int fl=1;
		for(j=1;j<=n;j++)
			if((i!=j)&&films[j][1]>films[i][0])
			{
				fl=0;
			}
		if(fl)
		{
			printf("0\n");
			continue;
		}
		fl=1;
		for(j=1;j<=n;j++)
			if((i!=j)&&films[i][1]<films[j][0])
			{
				fl=0;
			}
		if(fl)
		{
			printf("2\n");
			continue;
		}
		printf("1\n");
	}
	return 0;
}
