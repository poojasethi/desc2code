#include<fstream>
#include<string>
using namespace std;
ifstream cin("input.txt");
ofstream cout("output.txt");
int main()
{
    int n,k,t,i,j,u,v,w,p,o,m,min,max,l[102],r[102],d[102];
    bool bl[102];
    string name;
    cin>>n>>o;
    for(i=0;i<=n;i++) bl[i]=false;
    for(i=0;i<o;i++) cin>>p,bl[p]=true;
    cin>>k;
    for(i=0;i<k;i++)
    {
        cin>>name;
        cin>>m;
        w=u=v=0;
        while(m--)
        {
            cin>>p;
            if(p)
                if(bl[p]) u++;
                else v++;
            else w++;
        }
        min=v+w-n+o;
        if(min<0) min=0;
        max=(o-u<w?o-u:w);
        l[i]=u+min,r[i]=u+max;
    }
    for(i=0;i<k;i++) d[i]=2;
    for(i=0;i<k;i++)
    {
        bool b=true;
        for(j=0;j<k;j++)
        {
            if(i==j) continue;
            if(l[j]>r[i]) d[i]=1;
            if(r[j]>l[i]) b=false;
        }
        if(b) d[i]=0;
    }
    for(i=0;i<k;i++) cout<<d[i]<<endl;
    return 0;
}
