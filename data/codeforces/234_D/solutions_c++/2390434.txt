
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int maxn = 101;
int n, m, k, fav[maxn];
pair<int, int> pii[maxn];

int main() {
	char buf[64];
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d%d", &m, &k);
	for (int i = 1; i <= k; ++i) {
		int x;
		scanf("%d", &x);
		fav[x] = 1;
	}
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		scanf("%s", buf);
		int x, y, s1 = 0, s2 = 0;
		scanf("%d", &x);
		for (int j = 1; j <= x; ++j) {
			scanf("%d", &y);
			if (y == 0) {
				s2 += 1;
			} else {
				s1 += fav[y];
			}
		}
		int l = max(s1, x - m + k), r = min(s1 + s2, k);
		pii[i] = make_pair(l, r);
	}
	for (int i = 1; i <= n; ++i) {
		int u1 = 0, u2 = 0;
		for (int j = 1; j <= n; ++j) {
			if (i != j) {
				u1 = max(u1, pii[j].second);
				u2 = max(u2, pii[j].first);
			}
		}
		if (pii[i].first >= u1) {
			puts("0");
		} else if (pii[i].second < u2) {
			puts("1");
		} else {
			puts("2");
		}
	}
	return 0;
}
