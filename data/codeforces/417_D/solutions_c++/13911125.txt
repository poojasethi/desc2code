//In the Name of God

#include<iostream>
#include<algorithm>
#include<fstream>
#include<cstring>
#define int long long
#define nm first
#define cost second.first
#define prob second.second

using namespace std;

const int maxf=128,inf=1ll<<62;
int n,m,b,N,ans=inf,dp[2][1<<21];
pair<int,pair<int,int> > f[maxf];

main()
{
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  cin>>m>>n>>b;
  N=1ll<<n;
  for(int x,i=0;i<m;i++)
    {
      cin>>f[i].cost>>f[i].nm>>x;
      for(int pid;x--;)
	{
	  cin>>pid;
	  f[i].prob+=1ll<<(pid-1);
	}
    }
  sort(f,f+m);
  for(int i=1;i<N;i++)
    dp[0][i]=inf;
  dp[0][0]=0LL;
  for(int s,t,i=0;i<m;i++)
    {
      s=i&1;
      t=1-s;
      for(int mask=0;mask<N;mask++)
	{
	  dp[t][mask]=min(dp[s][mask],dp[s][mask&(~f[i].prob)]+f[i].cost);
	}
      ans=min(ans,dp[t][N-1]+f[i].nm*b);
    }
  if(ans<inf)
    cout<<ans<<endl;
  else
    cout<<-1<<endl;
  return 0;
}
