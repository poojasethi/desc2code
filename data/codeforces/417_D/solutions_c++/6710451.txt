#include <bits/stdc++.h>
#define X first
#define Y second
using namespace std;

typedef pair<int,int> ii;

int n,m,b,c[101],bit[101];
long long f[1<<20],g[1<<20];
ii x[101];

void minimize(long long &a,long long b){
	if(a>b) a=b;
}

int main(){
	//freopen("input.inp","r",stdin);
	scanf("%d%d%d",&n,&m,&b);
	for(int i=1;i<=n;i++){
		int k,t;
		x[i].Y=i;
		scanf("%d%d%d",&c[i],&x[i].X,&k);
		bit[i]=0;
		while(k--){
			scanf("%d",&t);
			bit[i]|=1<<(t-1);
		}
	}
	sort(x+1,x+n+1);
	memset(f,0x3f,sizeof f);
	for(int i=n+1;i>1;i--){
		memset(g,0x3f,sizeof g);
		g[bit[x[i-1].Y]]=1ll*x[i-1].X*b+c[x[i-1].Y];
		for(int k=1;k<1<<m;k++){
			if(f[k]!=0x3f3f3f3f3f3f3f3fll){
				minimize(g[k],f[k]);
				minimize(g[k|bit[x[i-1].Y]],f[k]+c[x[i-1].Y]);
			}
			f[k]=g[k];
		}
	}
	if(f[(1<<m)-1]==0x3f3f3f3f3f3f3f3fll) cout << -1;
	else cout << f[(1<<m)-1];
}
