#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=120,maxs=1050000;
const long long big=2e18;
struct Friend
{
	int x,k,s;
	bool operator < (const Friend &yy) const { return k<yy.k; }
}A[maxn];
int n,m,b,S;
long long ans=big;
long long dp[maxs];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,k,t;
	scanf("%d%d%d",&n,&m,&b);
	dp[0]=0;
	for(i=(1<<m)-1;i;i--)
		dp[i]=big;
	for(i=0;i<n;i++)
	{
		scanf("%d%d%d",&A[i].x,&A[i].k,&j);
		A[i].s=0;
		while(j--)
			scanf("%d",&t),A[i].s|=1<<(t-1);
	}
	sort(A,A+n);
	for(i=j=0;i<n;i++)
	{
		while(i<n-1&&A[i+1].k==A[i].k) i++;
		for(;j<=i;j++)
			for(k=(1<<m)-1;k>=0;k--)
				dp[k|A[j].s]=min(dp[k|A[j].s],dp[k]+A[j].x);
		ans=min(ans,dp[(1<<m)-1]+(long long)A[i].k*b);
	}
	printf("%I64d\n",ans==big?-1:ans);
	return 0;
}
