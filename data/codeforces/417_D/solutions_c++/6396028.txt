#include<cstdio>
#include<iostream>
#include<stdlib.h>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;

struct node{
	long long fe,bo,nu;
};

bool cmp(node a,node b){
	return a.bo<b.bo;
}

int main(){
	long long ans=2000000000000000000,g[30],f[2000000];
	int i,j,flag=0,n,m,mm,tt,b;
	node a[200];
	g[1]=1; for (i=2;i<=20;i++) g[i]=g[i-1]*2;

	cin>>n>>m>>b;
	for (i=1;i<=n;i++){
		cin>>a[i].fe>>a[i].bo>>mm;
		a[i].nu=0;
		for (j=1;j<=mm;j++){
			cin>>tt;
			a[i].nu+=g[tt];
		}
	}
	for (i=1;i<=20;i++) {g[i]+=g[i-1]; }

	sort(a+1,a+n+1,cmp);

	for (j=1;j<=g[m];j++) f[j]=ans;
	f[0]=0;
	for (i=1;i<=n;i++){
		for (j=0;j<=g[m];j++){
			if (f[j|a[i].nu]>f[j]+a[i].fe)
				f[j|a[i].nu]=f[j]+a[i].fe;
		}
		if (f[g[m]]+a[i].bo*b<ans){
			ans=f[g[m]]+a[i].bo*b;
			flag=1;
		}
	}
	if (flag) cout<<ans;
	else cout<<-1;
}
