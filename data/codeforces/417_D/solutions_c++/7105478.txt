#include<bits/stdc++.h>

using namespace std;

#define FOR(ii,aa,bb) for(lli ii=aa;ii<=bb;ii++)
#define lli long long int
#define N 105
#define M 25
#define ANS ((1<<m)-1)

lli n,m,b,a[N],c[N],q[N],s,d,dp[(1<<20)+5],ans=-1;

void MIN(lli &a,lli b) {

	if(a==-1 or b<a)
		a=b;

}

int main () {

	int x;

	memset(dp,-1,sizeof(dp));

	cin >> n >> m >> b;

	FOR(i,1,n)
	{
		cin >> a[i] >> c[i] >> s;

		FOR(j,1,s)
		{
			cin >> x;

			q[i]+=1<<x-1;
		}
	}

	FOR(i,1,n)
		FOR(j,1,n-i)
			if(c[j]>c[j+1])
			{
				swap(c[j],c[j+1]);
				swap(a[j],a[j+1]);
				swap(q[j],q[j+1]);
			}

	dp[0]=0;

	FOR(i,1,n)
	{
		FOR(j,0,ANS)
			if(dp[j]!=-1)
				MIN(dp[j|q[i]],dp[j]+a[i]);

		if(dp[ANS]!=-1)
			MIN(ans,dp[ANS]+c[i]*b);
	}

	cout << ans << endl;

	return 0;

}
