#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
#define int long long
#define inf 2000000000000000000

int n,m,b;
pair<pair<int,int>,int> a[200];
int d[(1<<20)+10];
main()
{
	cin>>n>>m>>b;
	for(int i=0;i<n;i++)
	{
		int k,u,v;
		cin>>u>>v>>k;
		a[i].first.second=u;
		a[i].first.first=v*b;
		for(int j=0;j<k;j++)
		{
			int y;
			cin>>y;
			a[i].second|=1<<(y-1);
		}
	}
	sort(a,a+n);
	fill(d,d+(1<<20)+10,inf);
	int sum=inf;
	d[0]=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<(1<<m);j++)
			d[j | a[i].second]=min(d[j|a[i].second],d[j]+a[i].first.second);
		sum=min(sum,d[(1<<m)-1]+a[i].first.first);
	}
	if(sum!=inf)
	cout<<sum<<endl;
	else
		cout<<-1<<endl;
	//system("pause");
}
