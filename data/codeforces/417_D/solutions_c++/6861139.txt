#include <stdio.h>
#include <algorithm>
using namespace std;
const long long oo = 1ll << 62;
int n, p, x[100], k[100], m[100], solve[100], o[100];
long long b, dp[1 << 20];
bool cmp(const int &a, const int &b){
	return k[a] < k[b];
}
int main()
{
	scanf("%d%d%lld", &n, &p, &b);
	for (int i = 0; i < n; ++i){
		o[i] = i;
		scanf("%d%d%d", x + i, k + i, m + i);
		for (int j = 0, c; j < m[i]; ++j){
			scanf("%d", &c);
			solve[i] |= 1 << (c - 1);
		}
	}
	sort(o, o + n, cmp);
	for (int j = (1 << p) - 1; j >= 0; --j)
		dp[j] = oo;
	dp[0] = 0;
	long long res = oo;
	for (int i = 0; i < n; ++i){
		for (int j = (1 << p) - 1; j >= 0;--j)
			dp[j | solve[o[i]]] = min(dp[j | solve[o[i]]], x[o[i]] + dp[j]);
		res = min(res, dp[(1 << p) - 1] + b*k[o[i]]);
	}
	if (res == oo)
		puts("-1");
	else
		printf("%lld\n", res);
	return 0;
}
