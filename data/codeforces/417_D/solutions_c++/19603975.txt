#include <bits/stdc++.h>
using namespace std;
const long long int INF = 2e18+1;
const int N = 100+5;
pair <int, pair <int,int> > p[N];
int n;
long long int b,m;
long long int ans;
long long int d[(1<<20)];
int main()
{
	cin >> n >> m >> b;
	int help1, help2;
	for (int i = 0 ;i < n; ++i)
	{
		cin >> p[i].second.first >> p[i].first;
		cin >> help1;
		for (int j = 0 ;j < help1; ++j)
		{
			cin >> help2;
			help2 --;
			p[i].second.second |= (1<<help2);
		}
	}
	sort (p, p+n);
	ans = INF;
	for (int i = 1 ;i < (1<<20); ++i) d[i] = INF;
	for (int i = 0; i < n; ++i)
	{
		for (int j = 0; j < (1<<m); ++j)
			d[(j | p[i].second.second)] = min (d[(j | p[i].second.second)], d[j] + p[i].second.first);
		ans = min (ans, d[(1<<m)-1] + p[i].first*b);
	}
	if (ans == INF) return cout << -1, 0;
	cout << ans;

}
