#include<stdio.h>
#include<algorithm>
using namespace std;
__int64 dp[1<<22];
struct data
{
	__int64 mo,mi;
	int sta;
}d[222];
bool cmp(data a,data b)
{
return a.mi<b.mi;	
}
int main()
{
	int n,m;
	__int64 b;
	scanf("%d%d%I64d",&n,&m,&b);
	int i;
	for(i=0;i<n;i++)
	{
		int t;
		scanf("%I64d%I64d%d",&d[i].mo,&d[i].mi,&t);
		d[i].sta=0;
		while(t--)
		{
			int a;
			scanf("%d",&a);
			d[i].sta+=1<<(a-1);
			
		}
		
	}
	sort(d,d+n,cmp);
	__int64 ans=1ll<<60;
	for(i=1;i<1<<m;i++)
	dp[i]=1ll<<60;
	dp[0]=0;
	int j;
	for(i=0;i<n;i++)
	{
		for(j=0;j<1<<m;j++)
	    if(dp[j|d[i].sta]>dp[j]+d[i].mo)
		dp[j|d[i].sta]=dp[j]+d[i].mo;
	    ans=min(dp[(1<<m)-1]+d[i].mi*b,ans);
		
	}
	if(ans==1ll<<60)printf("-1\n");
	else 
	printf("%I64d\n",ans);
}
