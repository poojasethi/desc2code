#include <iostream>
#include <utility>
#include <algorithm>
using namespace std;

pair<int, pair<int, int> > a[100];
long long d[1 << 20];

int main() {
int n, m; long long b; cin >> n >> m >> b;
for (int i = 0; i < n; ++i) {int mi; cin >> a[i].second.first >> a[i].first >> mi;
for (int j = 0; j < mi; ++j) {int l; cin >> l; a[i].second.second |= 1 << l - 1;}}
sort(a, a + n);
int pk = 1 << m;
long long ans = 1ll << 60;
for (int i = 1; i < pk; ++i) d[i] = ans;
for (int i = 0; i < n; ++i) {for (int j = 0; j < pk; ++j) {
d[j|a[i].second.second] = min(d[j|a[i].second.second], d[j] + a[i].second.first);
}ans = min(ans, d[pk - 1] + b * a[i].first);}
cout << (ans == 1ll << 60 ? -1 : ans) << endl;
}
