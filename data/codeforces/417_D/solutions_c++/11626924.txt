#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
typedef long long LL;
const LL INF = 1LL<<62;
#define two(x) (1<<(x))
LL f[1<<20];
int i,j,n,m,B,k,x;
struct arr
{
  int x,k,pro;
  bool operator <(const arr &a)const
  {
    return k>a.k;
  }
} a[105];

int main(){
  scanf("%d%d%d",&n,&m,&B);
  for (i=1;i<=n;i++){
    scanf("%d%d%d",&a[i].x,&a[i].k,&k);
    while (k--){
      scanf("%d",&x);
      a[i].pro|=two(x-1);
    }
  }
  sort(a+1,a+n+1);
  f[0] = 0;
  for (i=1;i<two(m);i++)
    f[i] = INF;
  for (i=1;i<=n;i++){
    for (j=two(m)-1;j>0;j--)
      if (f[j]<INF) f[j|a[i].pro] = min(f[j|a[i].pro], f[j]+(LL)a[i].x);
    f[a[i].pro] = min(f[a[i].pro], (LL)B*a[i].k+(LL)a[i].x);
  }
  if (f[two(m)-1]>=INF)
    f[two(m)-1] = -1;
  printf("%I64d\n",f[two(m)-1]);
  return 0;
}
