#include <bits/stdc++.h>
using namespace std;
long long dp[1<<20];
int x[101],k[101],mm[101],order[101],lim,m,b,n;
bool comp(int a,int y){
	return k[a]<k[y];
}
int main(){
	int a,cc,i,j;
	cin>>n>>m>>b;
	lim = (1<<m)-1;
	for(int i=0;i<n;++i){
		order[i]=i;
		cin>>x[i]>>k[i]>>cc;
		mm[i]=0;
		for(int j=0;j<cc;++j){
			cin>>a;--a;
			mm[i] |= (1<<a);
		}
	}
	sort(order,order+n,comp);
	memset(dp,-1,sizeof(dp));
	dp[0]=0;
	int cur,next;
	long long value;
	vector<int> q;
	q.push_back(0);
	for(i=0;i<n;++i){
		int o = order[i],size= q.size();
		if(dp[mm[o]]!= -1 && dp[mm[o]]<=x[o])
			continue;
		for(j=0;j<q.size();++j){
			cur = q[j],next= q[j]| mm[o];
			if(cur==next)continue;
			if(dp[next]==-1 && next!=lim)
				q.push_back(next);
			value = dp[cur]+x[o];if(next==lim)value += 1LL*k[o]*b;
			if( dp[next]==-1 || dp[next]>value)
				dp[next]= value;
		}
	}
	cout<<dp[lim]<<endl;
}
