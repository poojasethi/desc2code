#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define ll long long
#define inf 5000000000000000000LL
using namespace std;
int n,m,ed;
ll b,ans=inf;
ll f[1<<20];
struct data{ll x,k;int c;}a[101];
bool cmp(data a,data b){return a.k<b.k;}
void cal(int x)
{
    for(int i=0;i<=ed;i++)
       if(f[i]+a[x].x<f[i|a[x].c])
	      f[i|a[x].c]=f[i]+a[x].x;
    ans=min(ans,f[ed]+b*a[x].k);
}
int main()
{
    scanf("%d%d%I64d",&n,&m,&b);
    ed=(1<<m)-1;
    for(int i=1;i<=n;i++)
    {
       int t;
       scanf("%I64d%I64d%d",&a[i].x,&a[i].k,&t);
       for(int j=1;j<=t;j++)
       {
           int x;scanf("%d",&x);
           a[i].c|=1<<x-1;
       }
    }
    sort(a+1,a+n+1,cmp);
    for(int i=0;i<=ed;i++)f[i]=inf;
    f[0]=0;
    for(int i=1;i<=n;i++)cal(i);
    if(ans!=inf)printf("%I64d",ans);
    else printf("-1");
    return 0;
}
