#include <stdio.h>

int main()
{
	int N;
	scanf("%d", &N);
	if(N == 2)
	{
		puts("1\n1");
	} else if(N == 3)
	{
		puts("2\n1 3");
	} else if(N == 4)
	{
		puts("4\n3 1 4 2");
	} else
	{
		int i = 1;
		printf("%d\n", N);
		for(; i <= N; i += 2) printf("%d ", i);
		i = 2;
		for(; i <= N; i += 2) printf("%d ", i);
	}
	return 0;
}
