
#include <iostream>

using namespace std;

int main()
{
int n;
	cin>>n;
	
	if (n==1 || n==2) cout<<"1\n1";
	else if (n==3) cout<<"2\n1 3";
	else if (n==4) cout<<"4\n3 1 4 2";
	else 
	{
		cout<<n<<"\n";
		for (int i=1 ; i<=n ; i+=2) cout<<i<<" ";
		for (int i=2 ; i<=n ; i+=2) cout<<i<<" ";
	}
	
	return 0;
}
