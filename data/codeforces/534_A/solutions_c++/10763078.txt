#include<iostream>
using namespace std;
int main()
{
	int n,i;
	cin>>n;
	if(n>=5)
	{
		cout<<n<<endl;
		for(i=1;i<=n;i+=2)
			cout<<i<<" ";
		for(i=2;i<=n-2;i+=2)
			cout<<i<<" ";
		cout<<i;
	}
	if(n==1||n==2)
		cout<<1<<endl<<1;
	if(n==3)
		cout<<"2\n1 3";
	else if(n==4)
		cout<<"4\n3 1 4 2";
}
