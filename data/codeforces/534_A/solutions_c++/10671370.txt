#include<bits/stdc++.h>
using namespace std;
int main()
{
long long n;int i;
cin>>n;
if (n==1 || n==2)
cout<<"1\n1\n";
else
{
if (n==3)
cout<<"2\n1 3\n";
else
{
if (n==4)
cout<<"4\n2 4 1 3\n";
else
{
cout<<n<<"\n";
for(i=1;i<=n;i+=2)
cout<<i<<" ";
for(i=2;i<=n;i+=2)
cout<<i<<" ";
cout<<"\n";
}
}
}
return 0;
}//main
