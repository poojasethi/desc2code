#include<cstdio>

using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	if(n==2)
		printf("1\n2\n");
	else if(n==3)
		printf("2\n1 3\n");
	else
	{
		printf("%d\n",n);
		for(int i=n-1;i>0;i-=2)
			printf("%d ",i);
		for(int i=n;i>0;i-=2)
			printf("%d ",i);
	}
	return 0;
}
