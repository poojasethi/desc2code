#include<cstdio> 
int main()
{
	int n,i;
	scanf("%d",&n);
	if(n==2){printf("1\n1");return 0;} 
	if(n==3){printf("2\n1 3");return 0;} 
	printf("%d\n",n);
	for(i=(n-1)/2;i>=0;i--)printf("%d ",i*2+1);
	for(i=n/2;i>1;i--)printf("%d ",i*2);if(n>1)printf("2");
	return 0; 
}
