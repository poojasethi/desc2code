#include <bits/stdc++.h>

using namespace std;
int main()
{
	int n,i;
	scanf("%d",&n);
	switch(n)
	{
		case 1: cout<<"1\n1";break;
		case 2: cout<<"1\n1";break;
		case 3: cout<<"2\n1 3";break;
		default:
			cout<<n<<endl<<"2";	  
			for (i=4;i<=n;i+=2)
				cout<<" "<<i;
			for (i=1;i<=n;i+=2)
				cout<<" "<<i;

	}
}
