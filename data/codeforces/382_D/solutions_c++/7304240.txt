#include<stdio.h>
#include<algorithm>
#include<string.h>
using namespace std;
char map[2222][2222];
char s[10]="^v<>";
int st[4][2]={{1,0},{-1,0},{0,1},{0,-1}};
int n,m;
int fmax=0,smax=0;
int cont=0;
int change(int r)
{
	
	smax=max(smax,r);
	if(smax>fmax)
swap(smax,fmax);

}
int dfs(int x,int y)
{
cont++;
int ans=0;
int i;
for(i=0;i<4;i++)
  {
  	int q,p;
	q=x+st[i][0];
	p=y+st[i][1];

	if(q>=n||q<0||p>=m||p<0)continue;
	if(map[q][p]!=s[i])continue;
	if(map[x][y]=='#')change(dfs(q,p));
	else ans=max(ans,dfs(q,p));	
	//printf("%d %d %c %d\n",q,p,map[q][p],ans);
  }	
  return ans+1;
}
int main()
{
	int i,j;
scanf("%d%d",&n,&m);
for(i=0;i<n;i++)
scanf("%s",map[i]);
for(i=0;i<n;i++)
for(j=0;j<m;j++)
if(map[i][j]=='#')
dfs(i,j);
if(cont!=n*m)printf("-1\n");
else {
  if(fmax==smax)
  printf("%d\n",fmax*2);
  else printf("%d\n",fmax*2-1);	
}

	
}
