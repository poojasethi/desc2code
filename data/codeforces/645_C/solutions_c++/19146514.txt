#include<stdio.h>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;

char room[100003];
int fr[100003];

int main(){
	int i,num,n,k,m=100005,h,r;
	scanf("%d%d",&n,&k);
	getchar();
	for(i=num=0;i<n;i++){
		scanf("%c",&room[i]);
		if(room[i]=='0')
			fr[num++]=i;
	}
	for(h=0,r=k;r<num;h++,r++){
		for(int dis=(fr[r]-fr[h])/2;dis<=fr[r]-fr[h]&&dis<m;dis++){
			if(room[fr[h]+dis]=='0'||room[fr[r]-dis]=='0')
				m=min(m,max(fr[r]-fr[h]-dis,dis));
		}
	}
	printf("%d\n",m);
	return 0;
}
