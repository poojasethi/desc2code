#include <cstdio>

char s1[100010];
int a[100010],sum[100010],ans;
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	scanf("%s",s1);
	for (int i = 0;i < n;i++)
		a[i+1] = s1[i]-'0',sum[i+1] = sum[i]+a[i+1];
	for (int L = 1,R = n;L <= R;)
	{
		int mid = (L+R)>>1,flag = 0;
		for (int i = 1;i <= n;i++)
		if (!a[i])
		{
			int l = i-mid>1?i-mid:1,r = i+mid<=n?i+mid:n;
			if (r-l-sum[r]+sum[l-1]>=k)
			{
				flag = i;
				break;
			}
		}
		if (flag) ans = mid,R = mid-1;
		else L = mid+1;
	}
	printf("%d",ans);
	return 0;
}
