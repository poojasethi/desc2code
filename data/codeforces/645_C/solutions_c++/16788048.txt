#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <map>
#include <vector>
using namespace std;

#define N 400020



int n, k;
char s[N];

bool check(int t) {
	int cnt = 0;
	for(int i = 1, u = 1, v = 1; i <= n; ++i) {
		while(u <= n && u - i <= t) {
			if(s[u] == '0') ++cnt;
			++u;
		}
		while(v <= i && i - v > t) {
			if(s[v] == '0') --cnt;
			++v;
		}
		if(s[i] == '0' && cnt >= k + 1) return 1;
	}
	return 0;
}

int main() {
	scanf("%d%d", &n, &k);
	scanf("%s", s + 1);
	int l = 1, r = n;
	while(l < r) {
		int mid = (l + r) / 2;
		if(check(mid)) r = mid;
		else l = mid + 1;
	}
	printf("%d\n", l);
	return 0;
}
