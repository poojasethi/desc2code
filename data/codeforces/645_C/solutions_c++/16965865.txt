#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std;
const int MAXN=100100;
int n,m;
int a[MAXN];
char s[MAXN];

int check(int x)
{
	int xx,yy;
	for (int i=1;i<=n;i++)
	{
		if (s[i-1]=='1') continue;
		xx=i-x-1;
		yy=i+x;
		if (xx<=0) xx=0;
		if (yy>n) yy=n;
		if (a[yy]-a[xx]>=m+1) return 1;
	}
	return 0;
}

int main()
{
	cin>>n>>m;
	scanf("%s",s);
	for (int i=1;i<=n;i++)
	{
		a[i]=a[i-1];
		if (s[i-1]=='0') a[i]++;
	}
	
    int l=1,r=n,mid;
    while (l<=r)
    {
      mid=(l+r)/2;
      if (check(mid)) r=mid-1;
      else l=mid+1;
    }
    cout<<l;
}
