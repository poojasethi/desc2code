#include <bits/stdc++.h>

using namespace std;

#define mp make_pair

vector<int> a;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n, k;
    cin >> n >> k;
    string s;
    cin >> s;
    for (int i = 0; i < n; ++i) {
        if (s[i] == '0') a.push_back(i);
    }
    int nn = a.size();
    int m = 0;
    int ans = 1000 * 1000 * 1000;
    for (int l = 0; l < nn - k; ++l) {
        int r = l + k;
        while (a[m] - a[l] < a[r] - a[m]) m++;
        ans = min(ans, max(a[m] - a[l], a[r] - a[m]));
        ans = min(ans, max(a[m - 1] - a[l], a[r] - a[m - 1]));
    }
    cout << ans << endl;

    return 0;
}
