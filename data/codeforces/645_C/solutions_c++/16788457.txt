#include<bits/stdc++.h>
#define M 100000
using namespace std;
int s[M + 5] , n , k , ans;
char hotel[M + 5];
int cal(int x,int y){
	if(x > y)	return 0;
	return s[y] - s[x - 1];
}
main(){
//	freopen("test.inp","r",stdin);
	scanf("%d %d",&n,&k);
	ans = n;
	scanf("%s",hotel);
	for(int i = 1 ; i <= n ; i++)
		s[i] = s[i - 1] + (hotel[i - 1] == '0');
	for(int i = 1 ; i <= n ; i++){
		if(hotel[i - 1] == '1')	continue;
		int lo = 1 , hi = n;
		while(lo < hi){
			int mid = (lo + hi)/2;
			if(cal(max(i - mid , 1) , i - 1) + cal(i + 1 , min(i + mid , n)) >= k)	hi = mid;
			else lo = mid + 1;
		}
		ans = min(ans , lo);
	}
	printf("%d",ans);
}
