#include<iostream>
#include<algorithm>
using namespace std;
char room[100005];
int s[100005];
int main()
{
	int n,k;
	cin>>n>>k;
	for(int i=1;i<=n;i++)
	{
		cin>>room[i];
	}
	s[0]=0;
	for(int i=1;i<=n;i++)
	{
		if(room[i]=='0')
		{
			s[i]=s[i-1]+1;
		}
		else
		{
			s[i]=s[i-1];
		}	
	}
	int ans=999999999;
	//cout<<s[5]-s[1]<<endl;
	for(int i=1;i<=n;i++)
	{
		if(room[i]=='1')
		{
			continue;
		}
		int l=1,r=n;
		while(l<r)
		{
			int mid=(l+r)/2;
			int m_l=max(1,i-mid);
			int m_r=min(n,i+mid);
			if(s[m_r]-s[m_l-1]-1>=k)
			{
				r=mid;
			}
			else
			{
				l=mid+1;
			}
		}
		ans=min(ans,l);
	}
	cout<<ans<<endl;
}
