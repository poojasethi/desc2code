#include<bits/stdc++.h>

using namespace std;

int A[100005],a,n,k,i,j,l,x,asd = 100005;

string s;

int main(){

	cin >> n >> k >> s;  k++;

	for(i=j=0;i<n;i++){
		
		while(x + (s[j] == '0') <= k && j < n)  if(s[j++] == '0')  x++,  A[a++] = j-1;
		
		if(x == k && s[i] == '0') {

			l = lower_bound(A , A+a , i+A[a-1] >> 1) - A;

			asd = min( asd , max(A[l] - i , A[a-1] - A[l]) ); 
			
			if(l)
				asd = min( asd , max(A[l-1] - i , A[a-1] - A[l-1]) );

			x--;
		}
	}

	cout << asd << endl;
	return 0;
}
