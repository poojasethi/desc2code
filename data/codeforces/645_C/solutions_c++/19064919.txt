#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
typedef long long LL;
#define INF 0x3f3f3f3f
#define eps 1e-8

char s[100010];
int n, k, f[100010], ans, cur, x;

int main()
{
	scanf("%d%d%s", &n, &k, s);
	memset(f, 0, sizeof(f));
	f[0] = s[0] - '0';
	for (int i = 1; i < n; i++)
		f[i] = f[i-1] + s[i] - '0';
	ans = INF;
	for (int i = 0; i < n; i++)
	if (s[i] != '1' && i-f[i] >= k)
	{
		cur = i - k - 1;
		while (cur >= 0 && i-cur-f[i]+f[cur] < k+1)
			cur--;
		x = (i-cur) / 2;
		while (s[i-x] == '1' && s[cur+x+1] == '1')
			x++;
		if (x < ans) ans = x;
	}
	printf("%d\n", ans);
	return 0;
}
