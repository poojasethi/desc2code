#include <cstdio>
#include <algorithm>
#define N 100005
int n, k, sum[N];
char s[N];
bool check(int x) {
	bool ok = false;
	for (int i = 1; i <= n; ++i) {
		if (s[i] == '1') continue; 
		int l = std::max(i-x-1, 0), r = std::min(i+x, n);
		if (sum[r] - sum[l] >= k+1) ok = true;
	}
	return ok;
}
int main() {
	scanf("%d %d %s", &n, &k, s+1);
	for (int i = 1; i <= n; ++i) {
		sum[i] = sum[i-1] + (s[i] == '0' ? 1 : 0);
	}
	int l = 0, r = n, ans;
	while (l <= r) {
		int mid = (l + r) / 2;
		if (check(mid)) {
			ans = mid;
			r = mid - 1;
		} else {
			l = mid + 1;
		}
	}
	printf("%d\n", ans);
	return 0;
}
