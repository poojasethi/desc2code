#include <bits/stdc++.h>

#define NN 100010

using namespace std;

char str[NN];
int tot[NN];

int main (){
	
	int N,k,i;
	scanf("%d %d",&N,&k);
	scanf("%s",str+1);
	
	for(i=1; i<=N; i++){
		tot[i]=tot[i-1];
		if(str[i]=='0') tot[i]++;
	}
	
	int st=0, ed=N-1, mid;
	while(ed-st > 1){
		mid=(st+ed)/2;
		for(i=1; i<=N; i++){
			if(str[i]=='1') continue;
			if((tot[min(N,i+mid)]-tot[max(0,i-mid-1)])>=k+1) break;
		}
		if(i<=N) ed=mid;
		else st=mid;
	}
	
	printf("%d\n",ed);
	return 0;
}
