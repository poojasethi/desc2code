#include <cstdio>
#include <algorithm>
using namespace std; 
char s[100010];
int pos[100010],ans,cnt=0,p,n,k;
int main(){
    scanf("%d%d",&n,&k);
    scanf(" %s",&s); p=1;
    ans=n;
    for(int i=0;i<n;i++){
        if(s[i]=='0')pos[++cnt]=i;
        if(cnt>k){
            if(p<cnt-k)p++;
            while(p<cnt&&max(pos[p]-pos[cnt-k],pos[cnt]-pos[p])>max(pos[p+1]-pos[cnt-k],pos[cnt]-pos[p+1]))p++;
            ans=min(max(pos[p]-pos[cnt-k],pos[cnt]-pos[p]),ans);
        }
    }printf("%d\n",ans);
    return 0;
}
