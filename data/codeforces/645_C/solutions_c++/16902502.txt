#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[101000];
int sum[101000];
int n,m,L,R,mid;

int main() {
	scanf("%d %d",&n,&m);
	scanf("%s",s + 1);
	for (int i = 1; i <= n; i++)
		sum[i] = sum[i - 1] + (int) (s[i] == '0');
	L = 1; R = n;
	while (L < R) {
		mid = (L + R) >> 1;
		bool flag = false; 
		for (int i = 1; i <= n; i++) {
			int a = max(i - mid,1), b = min(i + mid,n);
			if (sum[b] - sum[a - 1] >= m + 1 && s[i] == '0') flag = true;
		}
		if (flag) R = mid;
		else L = mid + 1;
	}
	printf("%d",L);
	return 0;
}
