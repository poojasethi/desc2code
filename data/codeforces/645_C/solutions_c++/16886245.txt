#include<iostream>
#include<vector>
#include<string>
#include<cstring>
using namespace std;
int p[100010],n,k;
string s;
vector<int>v;
int main(){
	cin>>n>>k;
	cin>>s;
	for(int i=0;i<n;i++)p[i+1]=p[i]+(s[i]=='0');
	int lo=0,hi=n;
	while(hi-lo>1){ 
		int mid=(hi+lo)/2;
		bool good=false;
		for(int i=0;i<n;i++){
			if(s[i]!='0')continue;
			int l=max(0,i-mid),r=min(n-1,i+mid)+1;
			if(p[r]-p[l]>=k+1){good=true;break;}
		}
		if(good)hi=mid;
		else lo=mid;
	}
	cout<<hi;
	return 0;
}
