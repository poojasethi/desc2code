#include<stdio.h>
#include<string.h>
int a[100005];
char c[100005];
int n,k;
int div(int dis)
{
	int l=0,r=n,mid,i,j;
	while(r>l)
	{
		mid=(l+r)>>1;
		i=dis-mid>0?dis-mid:0;
		j=dis+mid<n-1?dis+mid:n-1;
		j++;
		if(a[j]-a[i]>k)
			r=mid;
		else
			l=mid+1;
	}
	return r;
}
int main()
{
	int i,j,sum,min,d;
	while(scanf("%d%d",&n,&k)!=EOF)
	{
		getchar();
		scanf("%s",&c);
		sum=0;
		for(i=0;i<=n;i++)
		{
			a[i]=sum;
			if(i<n&&c[i]=='0')
				sum++;
		}
		min=n;
		for(i=0;i<n;i++)
		{
			if(c[i]=='1')
				continue;
			d=div(i);
			//printf("ss%d\n",d);
			if(d<min)
				min=d;
		}
		printf("%d\n",min);
	}
	return 0;
}
