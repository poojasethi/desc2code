#include <iostream>
#include <string>

#define MIN(x,y) ((x)>(y)?(y):(x))
#define MAX(x,y) ((x)>(y)?(x):(y))

using namespace std;

int n,k,i,j;
string s;
int p[110000];

int main()
{
	cin>>n>>k;
	cin>>s;
	for (i=j=0;i<s.size();i++)
	{
		if (s[i]=='0')
		{
			p[j] = i;
			j++;
		}
	}
	int l=0,r=k,m=0,min=100000;
	while (r<j)
	{
		while (p[m]-p[l]<p[r]-p[m]) m++;
		if (p[m]-p[l]>p[r]-p[m] && m>0 && p[m]-p[l]>p[r]-p[m-1]) m--;
		min=MIN(min,MAX(p[m]-p[l],p[r]-p[m]));
		r++;l++;
	}
	cout<<min<<endl;
	return 0;
}
