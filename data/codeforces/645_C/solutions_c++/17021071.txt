#include <iostream>
#include <algorithm>
using namespace std;

int n, k, cnt[100005];
char room[100005];

bool C(int x) {
	for (int i = 1; i <= n; ++i)
		if (room[i - 1] == '0' && cnt[i - 1] - cnt[max(i - x - 1, 0)] + cnt[min(i + x, n)] - cnt[i] >= k) return 1;
	return 0;
}

int main() {
	cin >> n >> k >> room;
	for (int i = 0; i < n; ++i) cnt[i + 1] = cnt[i] + (room[i] == '0' ? 1 : 0);
	int l = 0, r = n, mid;
	while (r - l > 1) {
		mid = (l + r) >> 1;
		C(mid) ? r = mid : l = mid;
	}
	cout << r << endl;
	return 0;
}
