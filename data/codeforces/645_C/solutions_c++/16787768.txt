#include<iostream>
#include<cstdio>
#include<string>
#include<algorithm>
#include<queue>
using namespace std;

long long n, k, m, l;
string s;
int ind[100005];
int main()
{
	cin >> n >> k;
	cin >> s;
	l = s.length();
	for(int i=0;i<l;i++)
		if (s[i] == '0')
			ind[m++] = i;

	int c = 0;
	int best = 1000000000;
	for(int a=0,b=k;b<m;a++,b++)
	{
		while(c<b && ind[c+1]-ind[a] < ind[b]-ind[c+1] ) c++;
		best = min(best, min(max(ind[c]-ind[a],ind[b]-ind[c]), max(ind[c+1]-ind[a],ind[b]-ind[c+1])));		
	}
	cout << best << endl;
	
	return 0;
}
