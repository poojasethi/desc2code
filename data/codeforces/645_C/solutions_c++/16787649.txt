#include<bits/stdc++.h>
using namespace std;
int n,k;
int a[200010],sum[200010];
int getsum(int l,int r)
{
	l=max(l,1);r=min(r,n);
	return sum[r]-sum[l-1];
}
bool pend(int d)
{
	for (int i=1;i<=n;i++) if ((a[i]==0)&&(getsum(i-d,i+d)>=k+1)) return true;
	return false;
}
int main()
{
	scanf("%d%d\n",&n,&k);
	for (int i=1;i<=n;i++)
	{
		char ch=getchar();
		a[i]=ch-'0';
		sum[i]=sum[i-1]+(!a[i]);
	}
	int l=0,r=n;
	while (l+1!=r)
	{
		int mid=(l+r)>>1;
		if (pend(mid)) r=mid;else l=mid;
	}
	printf("%d\n",r);
}
