#include <bits/stdc++.h>
#define mx 100005
using namespace std;
int n, k, ans = 1e6;
int idx[mx], c;
int main()
{
	cin >> n >> k;
	for(int i = 0; i < n; i++)
	{
		char x;
		cin >> x;
		if(x == '0') idx[c] = i, c++;
	}
	int i = 0, j = k;
	while(i + k < c)	
	{
		int l = (idx[i + k] - idx[i]) / 2;
		int a= lower_bound(idx + i,idx + i + k, idx[i] + l) - idx;
		ans = min(ans, max(abs(idx[i]-idx[a]),abs(idx[i+k]-idx[a])));
		ans = min(ans, max(abs(idx[i]-idx[a-1]),abs(idx[i+k]-idx[a-1])));
		i++;
	}

	cout << ans;
}
