#include <bits/stdc++.h>
#define ll long long int
using namespace std;

string s;
vector < int > v;

int main()
{
    int n, k, i, j, ans = (int) 1e9, low;
    cin >> n >> k >> s;
    for(i=0; i<n; i++) if(s[i]=='0') v.push_back(i);
    for(i=k, j=0; i<v.size(); i++, j++){
        int mid = (v[i]+v[j])/2;
        low = lower_bound(v.begin(), v.end(), mid) - v.begin();
        if(v[low] != mid) low--;
        ans = min(ans, max(v[i] - v[low], v[low]-v[j]));
        ans = min(ans, max(v[i] - v[low+1], v[low+1]-v[j]));
    }
    cout << ans << endl;
    return 0;
}
