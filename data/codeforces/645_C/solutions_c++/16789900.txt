#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

#define MAXL 100005

char s[MAXL];
int sum[MAXL];

int main(){
  int n,k;

  scanf("%d %d %s",&n,&k,s);

  sum[0] = 0;

  for(int i = 0;i < n;++i)
    sum[i + 1] = sum[i] + (s[i] == '0'? 1 : 0);

  int ans = n;

  for(int i = 0;i < n;++i){
    if(s[i] == '0'){
      int lo = 0,hi = n,mi;

      while(lo < hi){
        mi = (lo + hi) >> 1;

        int start = max(0,i - mi),end = min(n - 1,i + mi);
        int cont = sum[end + 1] - sum[start];

        if(cont < k + 1) lo = mi + 1;
        else hi = mi;
      }

      ans = min(ans,lo);
    }
  }

  printf("%d\n",ans);

  return 0;
}
