#include<bits/stdc++.h>
using namespace std;

int n,k,a[100100],sum[100100];

int judge(int len)
{
	for(int i=1;i<=n;i++)
	{
		if(a[i]=='1')
			continue;
		if(sum[i-1]-sum[max(0,i-len-1)]+sum[min(i+len,n)]-sum[i]>=k)
			return 1;
	}
	return 0;
}

int main()
{
	int i,lef,rig,mid;
	while(cin>>n>>k)
	{
		for(i=1;i<=n;i++)
			scanf(" %c",&a[i]);
		sum[0]=0;
		for(i=1;i<=n;i++)
		{
			if(a[i]=='0')
				sum[i]=sum[i-1]+1;
			else
				sum[i]=sum[i-1];
		}
		lef=1;rig=n;
		while(lef<=rig)
		{
			mid=(lef+rig)>>1;
		//	cout<<lef<<" "<<mid<<" "<<rig<<endl;
			if(judge(mid))
				rig=mid-1;
			else
				lef=mid+1;
		}
		cout<<lef<<endl;
	}
}
