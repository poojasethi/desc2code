#include<cstdio>
#include<algorithm>
using namespace std;

const int INF=0x3f3f3f3f;
int n,k,ans;
char s[100005];
int num[100005];

int main()
{
	while(~scanf("%d%d",&n,&k))
	{
		scanf("%s",s+1);
		num[1]=1;
		for(int i=2;i<=n+1;i++)
		{
			num[i]=num[i-1];
			if(s[i-1]=='0') num[i]++;
		}
		ans=INF;
		for(int i=1;i<=n;i++)
		{
			if(s[i]=='1') continue;
			int l=1,r=n,m;
			while(l<r)
			{
				m=(l+r)>>1;
				int re=min((i+m),n);
				int le=max((i-m),1);
				if((num[re+1]-num[le])>=(k+1)) r=m;
				else l=m+1;
				//printf("%d %d\n",l,r);
			}
			if(ans>l) ans=l;
		}
		printf("%d\n",ans);
	}
	return 0;
}
