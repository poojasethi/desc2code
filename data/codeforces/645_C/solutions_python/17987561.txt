n, k = map(int, raw_input().split())
s = raw_input()

oc = [0]*(n + 1)
for i in xrange(1, n+1):
	oc[i] = oc[i - 1] + (s[i - 1] == '0')

ret = 1e9
for john in xrange(1, n+1):
	if s[john - 1] == '1':
		continue
	hi, low = n if 2*ret > n else 2*ret, 0
	while(hi >= low):
		mid = (hi+low)/2
		l = 0 if john - mid -1 < 0 else john - mid -1
		r = n if n < john + mid else john + mid
		if oc[r] - oc[l] >= k + 1:
			hi = mid - 1
			ret = ret if ret < mid else mid
			continue
		low = mid + 1
print ret
