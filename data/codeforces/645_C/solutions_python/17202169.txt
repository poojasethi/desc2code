#!/usr/bin/python2
# -*- coding: utf-8 -*-

import sys
import os

def main():
    n, k = map(int, sys.stdin.readline().split())
    a = sys.stdin.readline().rstrip()
    nextleft = [-1] * len(a)
    nextright = [-1] * (len(a) + 1)
    if a[0] == '0':
        nextleft[0] = 0
    if a[-1] == '0':
        nextright[len(a) - 1] = len(a) - 1
    for i in xrange(1, len(a)):
        nextleft[i] = i if a[i] == '0' else nextleft[i-1]
    for i in xrange(len(a)-2, -1, -1):
        nextright[i] = i if a[i] == '0' else nextright[i+1]
    ret = len(a)
    left = nextright[0]
    right = left
    for i in xrange(k):
        right = nextright[right+1]
    assert right != -1
    while right != -1:
        center = (left + right) / 2

        c = nextright[center]
        assert c != -1 and left <= c <= right
        ret = min(ret, max(c-left, right-c))

        c = nextleft[center]
        assert c != -1 and left <= c <= right
        ret = min(ret, max(c-left, right-c))

        left = nextright[left+1]
        right = nextright[right+1]
    print ret


if __name__ == '__main__':
    main()
