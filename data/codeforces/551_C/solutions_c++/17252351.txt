#include<iostream>
using namespace std;
#define ll long long
ll n,m,hi,lo,mid,ans=1e18;
ll a[100009],s[100009];
int main() 
{
	cin>>n>>m;
	hi=n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		hi+=a[i];
	}
	while(hi>=lo)
	{
		mid=(hi+lo)/2;
		int j=n,i;
		ll t=0;
		for(i=0;i<m&&j>0;i++)
		{
			while(a[j]==0)
				j--;
			ll left=mid-j+t;
			while(j>0&&left>=a[j])
			{
				left-=a[j];
				j--;
			}
			t=left;
		}
		if(j==0)
		{
			ans=mid;
			hi=mid-1;
		}
		else 
		    lo=mid+1;
	}
	cout<<ans<<endl;
	return 0;
}
