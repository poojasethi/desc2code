#include <bits/stdc++.h>
using namespace std;
#define ll long long
const int N = 1e5+1;
int a[N],tem[N],n,m;

bool ok(ll t){
	int i,j,last=n;
	for(i=1;i<=n;++i)tem[i]=a[i];
	ll c,minus;
	for(i=0;i<m;++i){
		c= t;
		while(last>0)if(tem[last]!=0)break;else --last;
		c-=last;
		while(true){
			minus = min((ll)tem[last],c);
			c-=minus,tem[last]-=minus;
			if(tem[last]==0)--last;
			else
				break;
			if(last<=0)return true;
		}
	}
	return false;
}
int main(){
	int i,j;
	cin>>n>>m;
	for(i=1;i<=n;++i)cin>>a[i];
	ll l=0,r= 1e15,mid;
	while(l<r){
		mid = (l+r-1)/2;
		if(ok(mid))
			r= mid;
		else
			l= mid+1;
	}
	cout<<l<<endl;
}
