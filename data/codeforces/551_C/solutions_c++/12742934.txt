#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
#define LL long long
int n,q;
LL s[100005],t[100005];
bool ok(LL time){
  for(int i=1;i<=n;i++) t[i]=s[i];
  int r=1;
  LL cur=0;
  for(int i=1;i<=q;i++){
    while(r<=n){
      if(r+cur+t[r]<=time){
        cur+=t[r];
        t[r]=0;
        r++;
      }else{
        t[r]+=r+cur-time;
        cur=0;
        break;
      }
    }
  }
  return r>n;
}
int main(){
  scanf("%d %d",&n,&q);
  for(int i=1;i<=n;i++) scanf("%lld",&s[i]);
  while(n>=1 && s[n]==0) n--;
  LL l=0,r=2E18;
  while(r-l>1){
    LL m=(l+r)>>1;
    if(ok(m)) r=m;
    else l=m;
  }
  printf("%lld\n",r);
  return 0;
}
