#include<cstdio>
#include<iostream>
using namespace std;
int m,n,k;
long long a[100005];
int f(long long x){
	
	int i,cnt=m;
	long long sum2=0;
	
	for(i=1;i<=k;i++){
		sum2+=a[i];
	//	cout<<a[i]<<" "<<sum<<endl;
		while(sum2+i>=x){
			sum2-=x-i;
			cnt--;
			if(cnt<0)return 0;
		}
	}
	if(cnt==0)return sum2<=0;
	return 1;
}
int main(){
	long long l,r,mi,ans;
	cin>>n>>m;
	long long sum=0;
	for(int i=1;i<=n;i++){
		scanf("%lld",&a[i]);
		sum+=a[i];
		if(a[i])k=i;
	}
	l=k;r=sum+k;

	while(r>=l){
		//cout<<1<<endl;
		mi=l+(r-l)/2;
		if(f(mi)){r=mi-1;ans=mi;}
		else l=mi+1;
	}
    cout<<ans<<endl;
	
	return 0;}
