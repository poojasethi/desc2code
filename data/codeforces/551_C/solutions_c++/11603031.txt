#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll ar[100009],r[100009],a,s,e=1e15+9,m=1e18+9,b,in;
int si(ll x)
{
	ll is=1,o=x-1;
	for(int i=1;i<=a;i++)r[i]=ar[i];
	for(int i=1;i<=a;i++,o--)
		while(r[i])
		{
			if(!o)o=x-i,is++;
			if(r[i]>=o)r[i]-=o,o=0;	
			else o-=r[i],r[i]=0;
		}
	if(is<=b&&m>x)m=x;	
	return (is<=b);
}
int main()
{
	cin>>a>>b;	
	for(int i=1;i<=a;i++){cin>>ar[i];if(ar[i])in=i;}
	while(s<e)
		((e+s)/2<=in||(si((e+s)/2))==0)?s=((e+s)/2)+1:e=(e+s)/2; 
	cout<<m<<endl;
	return 0;
}
