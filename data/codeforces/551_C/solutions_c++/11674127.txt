#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;
int n,m,a[100005];
long long time,sum;
long long from=0;
int m1,w,i;
bool can(long long t)
{
    from=0;
    m1=m;
    for(i=n;i;--i)
    {
        w=ceil(double(a[i]-from)/(t-i));
        m1-=w;
        if(m1<0)return 0;
        from=w*(t-i)-(a[i]-from);
    }
    return 1;
}

int main()
{
    cin>>n>>m;
    for(i=1;i<=n;++i)
    {
        scanf("%d",a+i);
        //sum+=a[i];
    }
    while(a[n]==0)--n;
    long long b=n+1,e=10000000000000000LL,mid;
    while(b!=e)
    {
        mid=(e+b)>>1;
        if(can(mid))e=mid;
        else b=mid+1;
    }
    cout<<b;
}
