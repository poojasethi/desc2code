#include <iostream>
#include <cstdio>
using namespace std;
int n, m, a[100010], b[100010];
bool ban(long long x)
{
	for (int i=1; i<=n; i++) b[i]=a[i];
	int p=0;
	for (int i=0; i<m; i++) {
		long long L=x-p;
		while (L>0 && p<=n) {
			while(b[p]==0 && L>0 && p<=n) {
				p++; L--;
			}
			if (L>b[p]) {
				L-=b[p]; b[p]=0;
			}
			else if (L>0) {
				b[p]-=L; L=0;
			}
		}
		if (p>n || p==n && b[p]==0) return 1;
	}
	return 0;
}
int main()
{
	scanf("%d%d", &n, &m);
	for (int i=1; i<=n; i++) scanf("%d", &a[i]);
	while (a[n]==0) n--;
	long long L=0, R=1e14+1e5, M;
	// while(!ban(R)) R*=2;
	while (R-L>1) {
		M=(R+L)/2;
		if (ban(M)) R=M;
		else L=M;
	}
	printf("%I64d\n", R);
	return 0;
}
