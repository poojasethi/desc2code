#include<iostream>
#include<cstdio>
using namespace std;
typedef long long ll;
ll n,m;
const int maxn=100010;
ll a[maxn],b[maxn];
bool check(ll x,ll n)
{	
	for(int i=1;i<=n;i++)b[i]=a[i];
	ll temp=m;
	while(temp--){
		ll t=x-n;
		while(t>0&&n>0){
			ll cnt=min(t,b[n]);
			t-=cnt;b[n]-=cnt;
			while(b[n]==0)n--;
		}
	}
	if(n>0)return false;
	else return true;
}
int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	while(!a[n])n--;
	ll l=0,r=1e15,ans;
	while(l<=r){
		ll mid=(l+r)/2;
		if(check(mid,n)){
			ans=mid;
			r=mid-1;
		}
		else l=mid+1;
	}
	cout<<ans<<endl;
}
