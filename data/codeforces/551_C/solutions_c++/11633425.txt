#include <cstdio>
#include <iostream>
#include <vector>
#include <map>

#include<climits>

using namespace std;

int main()
{
	long long int m,n,i,s;
	scanf("%lld %lld",&n,&m);
	long long int a[n],b[n];
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
		if(a[i] !=  0)
			s=i;
	}
	long long int l=1, r=LLONG_MAX-10,mid,iter,k;
	while(l<r)
	{
		mid=(l+r)/2;
		for(i=0;i<=s;i++)
			b[i] = a[i];
		iter = 0;
		for(i=0;i<m;i++)
		{
			k=mid-iter-1;
			while(k>0 && iter <= s)
			{
				if(b[iter] <= k)
				{
					k -= b[iter];
					k--;
					iter++;
				}
				else
				{
					b[iter] -= k;
					k=-1;
				}
			}
		}
		if(iter == s+1)
			r=mid;
		else
			l=mid+1;
		//cout << l << "   "<< r << "\n";
	}
	printf("%lld\n",l);
	return 0;
}
