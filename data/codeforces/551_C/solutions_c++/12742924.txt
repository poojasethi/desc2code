#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
#define LL long long
int n,q;
LL s[100005],t[100005];
bool ok(LL time){
  for(int i=1;i<=n;i++) t[i]=s[i];
  int r=n;
  LL cur=0;
  for(int i=1;i<=q;i++){
    cur=r;
    while(r>=1){
      if(cur+t[r]<=time){
        cur+=t[r];
        t[r--]=0;
      }else{
        t[r]=min(t[r],cur+t[r]-time);
        break;
      }
    }
  }
  return r<=0;
}
int main(){
  scanf("%d %d",&n,&q);
  for(int i=1;i<=n;i++) scanf("%lld",&s[i]);
  while(n>=1 && s[n]==0) n--;
  LL l=0,r=2E18;
  while(r-l>1){
    LL m=(l+r)>>1;
    if(ok(m)) r=m;
    else l=m;
  }
  printf("%lld\n",r);
  return 0;
}
