#include<bits/stdc++.h>
using namespace std;
 
int main()
{
	int n , m ;
	cin >> n >> m ; 
	int a[ n ] ; for ( int i = 0 ; i < n ; i++ ) cin >> a[ i ] ; 
	long long s = 0 ; 
	for ( int i = 0 ; i < n ; i++ ) s += a[ i ] ; 
	long long l = 2 , r = s + n ; 
	while ( l < r )
	{
		long long z = l + r >> 1 ;
		int b[ n ] ; for ( int i = 0 ; i < n ; i++ ) b[ i ] = a[ i ] ; 
		int p = n - 1 ; 
		for ( int i = 0 ; i < m ; i++ )
		{
			while ( p >= 0 && b[ p ] == 0 ) p-- ; 
			long long t = z - p - 1 ; 
			if ( t <= 0 ) break ; 
			while ( p >= 0 && b[ p ] <= t ) t -= b[ p-- ] ; 
			if ( p >= 0 ) b[ p ] -= t ; 
		}
		
		if ( p < 0 ) r = z ; 
		else l = z + 1 ; 
	}
	
	cout << r << endl;
    return 0;
}
