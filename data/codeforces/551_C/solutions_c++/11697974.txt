#include <stdio.h>
#include <iostream>

using namespace std;

inline bool check(int a[], int n, int m, long long t){
	int b,i,j;
	long long x;

	for(i=0,j=0,b=a[0];i<m && j<n;i++){
		x=t-1-j;
		
		while(j<n && x>0)
			if(b<=x){
				x-=(b+1);
				j++;
				if(j<n)
					b=a[j];
			}
			else{
				b-=x;
				x=0;
			}
	}

	if(j==n)
		return 1;
	else
		return 0;
}

int main(){
	int n,m,i,j,k;
	long long x,y,z,mid;

	scanf("%d%d", &n,&m);

	int a[n];

	for(i=0,y=n;i<n;i++){
		scanf("%d", &a[i]);
		y+=a[i];
	}

	while(a[n-1]==0 && n>0)
		n--;

	x=0;
	z=y;

	while(x<=y){
		mid=(x+y)/2;

		if(check(a,n,m,mid)==1){
			z=mid;
			y=mid-1;
		}
		else
			x=mid+1;
	}

	cout << z << '\n';

	return 0;
}
