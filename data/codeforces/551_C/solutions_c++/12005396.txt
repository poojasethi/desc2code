#include<cstdio>
#include<algorithm>

using namespace std;

long long  a[100010];
int i,j,k,n,m;
long long ans,tot,mid,l,r,s;

int Check(long long  x){

int i,cnt;
long long s;

  cnt=m;
  s=0;
  for (i=1;i<=tot;i++){
    s+=a[i];
    while (s+i>=x){
      s-=x-i;
      cnt--;
      if (cnt<0) return 0;
    }
  }

  if (cnt==0) return s<=0;
  return 1;

}



int main(){

  scanf("%d%d",&n,&m);
  for (i=1;i<=n;i++){
    scanf("%I64d",&a[i]);
    s+=a[i];
    if (a[i]) tot=i;
  }

  l=tot;
  r=tot+s;
  while (l<=r){
    mid=(l+r)>>1;
    if (Check(mid)){
      ans=mid;
      r=mid-1;
    }
    else l=mid+1;
  }

  printf("%I64d\n",ans);
  return 0;


}
