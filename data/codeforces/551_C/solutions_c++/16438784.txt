#include<iostream>
#include<cstdio>
using namespace std;
long long t,n,m,s,a[100010],sum,l,r,mid;
int i,st;
bool ch(long long x){
	int t=m;
	long long s=0;
	for (i=1;i<=st;i++){
		s+=a[i];
		while(s+i>=x){
			s-=x-i;
			t--;
			if(t<0)return 0;
		}
	}
	if(t==0)return s<=0;
	return 1;
}
int main(){
	cin>>n>>m;
	for (i=1;i<=n;i++) {
		scanf("%d",&a[i]);
		sum+=a[i];
		if (a[i]) st=i;
	}
	l=st+1;r=sum+st+1;
	while(l<r){
		mid=(l+r)>>1;
		if(ch(mid))r=mid;
		else l=mid+1;
	}
	cout<<l<<endl;
}
