#include<iostream>
#include<cmath>
using namespace std;
int n,m,a[100005];
long long time,sum;
bool can(long long t)
{
    long long from=0;
    int m1=m,w;
    for(int i=n;i>0;--i)
    {
        w=ceil(double(a[i]-from)/(t-i));
        if(m1<w)return 0;
        m1-=w;
        from=w*(t-i)-(a[i]-from);
    }
    return 1;
}
long long bin(long long b,long long e)
{
    if(b==e)return b;
    long long mid=(e+b)/2;
    if(can(mid))return bin(b,mid);
    return bin(mid+1,e);
}
int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;++i)
    {
        cin>>a[i];
        sum+=a[i];
    }
    while(a[n]==0)--n;
    cout<<bin(n+1,sum+n);

}
