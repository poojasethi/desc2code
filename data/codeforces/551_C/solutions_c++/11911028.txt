#include<cstdio>
#include<cstdlib>
using namespace std;
typedef long long LL;
int n,m,last;
LL a[100010];
int try1(LL x)
{
	int num=m;
	LL s=0;
	for(int i=1;i<=last;i++)
	{
		s+=a[i];
		while(s+i>=x)
		{
			s-=x-i;
			num--;
			if(num<0) return 0;
		}
	}
	if(num==0) return s<=0;
	return 1;
}
LL s,l,r,ans;
int i;
int main()
{
	scanf("%d%d",&n,&m);
	s=0;
	for(i=1;i<=n;i++)
	{
		scanf("%lld",&a[i]);
		if(a[i]!=0) last=i;
		s+=a[i];
	}
	l=last;r=last+s;
	while(l<=r)
	{
		LL mid=(l+r)/2;
		if(try1(mid))
		{
			ans=mid;
			r=mid-1;
		}else 
		l=mid+1;
	}
	printf("%lld\n",ans);
	//system("pause");
}
