#include <bits/stdc++.h>
using namespace std;

typedef long long lint;

const int maxn = 1e5+123;

int N,M,ar[maxn],er[maxn];

bool calc( lint goal ){
	
	memcpy( er , ar , sizeof er );
	
	int l=1;
	for( int i=1 ; l <= N and i <= M ; i++ ){
		
		lint val = goal - l;
		
		while( l <= N and val > 0 ){
			if( val >= er[l] )
				val -= er[l++] + 1;
			else{
				er[l] -= val;
				break;
			}
		}
	}
	
	return l > N;
}

int main(){
	
	scanf("%d%d",&N,&M);
	
	for( int i=1 ; i <= N ; i++ )
		scanf("%d",&ar[i]);
	
	while( !ar[N] )N--;
	
	lint bgn = 0 , mid , end = 1e18;
	
	while( bgn < end ){
		mid = (bgn + end) >> 1;
		if( calc( mid ) )
			end = mid;
		else
			bgn = mid+1;
	}
	
	printf("%lld\n",end);
	
	return 0;
}
