#include<iostream>
#include<string>
#include<cmath>
using namespace std;
int main(){
    int n,m;
    cin>>n>>m;
    long long a1[n],sum=0;
    for(int i=0;i<n;i++) cin>>a1[i],sum+=a1[i];
    long long l=2,r=sum+n;
    while(l<r){
        long long mid=(l+r)>>1;
        long long a[n];
        for(int i=0;i<n;i++)a[i]=a1[i];
        int x=n-1;
        for(int i=0;i<m;i++){
            while(x>=0 && a[x]==0)x--;
            long long int mi = mid-x-1;
            if(mi<=0) break;
            while(x>=0 && mi>=a[x])mi-=a[x--];
            if(x>=0 ) a[x]-=mi;
        }
        if(x<0) r=mid;
        else l=mid+1;
    }
    cout<<l<<endl;
}
