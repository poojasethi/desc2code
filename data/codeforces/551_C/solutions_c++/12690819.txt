#include <iostream>
#include <cstdio>
#include <cstring>

const int N=100100;
int a[N],p[N];
int n,m,i,j;
long long l,r,mid;

int ok(long long t,int *p)
{
	int i,j;
	long long s;
	for (j=n;p[j]==0&&j;j--);
	for (i=1;i<=m;i++)
	{
		s=t-j;
		if (s<=0) break;
		while (j&&s>=p[j]) s-=p[j--];
		if (j) p[j]-=s;
		if (j==0) break;
	}
	if (j==0) return 1; else return 0;
}

int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		r+=a[i];
	}
	l=1;
	r+=n;
	while (l<r)
	{
		for (i=1;i<=n;i++) p[i]=a[i];
		mid=(l+r)/2;
		if (ok(mid,p)) r=mid;
		else l=mid+1;
	}
	printf("%I64d\n",l);
	return 0;
}
