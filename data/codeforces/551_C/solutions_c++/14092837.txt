#include<cstdio>
#include<iostream>
#define N 100100
#define LL long long
using namespace std;

int n,m,a,w[N],t[N];
LL sum,ans;

int can(LL a)
{
	for (int i=1;i<=n;i++) t[i]=w[i];
	LL j=n;
	for (int i=1;i<=m;i++)
	{
		LL d=a-j;
		while (j&&t[j]<d) d-=t[j--];
		if (!j) return 1;
		if (t[j]>=d) t[j]-=d;
		while (j&&!t[j]) j--;
		if (!j) return 1;
	}
	return !j;
}

void find()
{
	LL l=0,r=1000000000,mid;r*=r;
	while (l+1<r)
	{
		mid=(l+r)>>1;
		if (can(mid)) r=mid;
		else l=mid;
	}
	cout<<r<<endl;
}

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a);
		w[i]=a;
		sum+=a;
	}
	while (!w[n]) n--;
	find();
	return 0;
}
