#include<cstdio>
#define MAXN 100005
#define LL long long
using namespace std;
LL n,m;
LL a[MAXN],b[MAXN];
LL L,R,M;
bool check(LL t)
{
	LL cnt=0,left=0;
	for (int i=1;i<=n;i++)b[i]=a[i];
	for (int i=n;i>=1;i--)
	{
		if (b[i]<=left)
		{
			left-=b[i];
			continue;
		}
		b[i]-=left;
		cnt+=b[i]/(t-i);
		if (b[i]%(t-i))
		{
			cnt++;
			left=(t-i)-b[i]%(t-i);
		}
		else left=0;
	}
	return cnt<=m;
}
int main()
{
	scanf("%I64d%I64d",&n,&m);
	for (int i=1;i<=n;i++)
	scanf("%I64d",&a[i]);
	while (!a[n])n--;
	L=n+1;R=0x3f3f3f3f3f3f3f3fLL;
	while (L<=R)
	{
		M=(L+R)>>1;
		if (check(M))R=M-1;
		else L=M+1;
	}
	printf("%I64d\n",L);
}
