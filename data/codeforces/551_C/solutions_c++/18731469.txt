#include<bits/stdc++.h>
using namespace std;
long long n,m,sm,a[100005],lw,hgh,md,st;
bool check(long long  vl)
{
	long long  mm=m,s=0;
	for(int i=1;i<=st;i++)
	{
         s+=a[i];while(s+i>=vl){s=s+i-vl;mm--;if(mm<0)return false;}
    }
    if(mm==0) return s<=0;
    return true;
}
int main()
{
     cin>>n>>m;
     for(int i=1;i<=n;i++)cin>>a[i],a[i]?st=i:1,sm+=a[i]+i;
     lw=0,hgh=sm;
     while(lw<hgh)
     	md=(lw+hgh)/2,check(md)?hgh=md:lw=md+1; 
     cout<<lw;                      
     return 0;
}
