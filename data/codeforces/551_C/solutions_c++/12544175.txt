#include <cstdio>
using namespace std;
int n,m,i,a[100010],k;
long long l,r,mid;
int check(long long x)
{
  long long sum=0;
  int tmp=m;
  for (i=1;i<=k;i++)
  {
    sum+=a[i];
    while (sum>=x-i)
    {
      sum=sum-(x-i);
      tmp--;
      if (tmp<0)
        return 0;
    }
  }
  if (tmp==0&&sum>0)
    return 0;
  return 1;
}
int main()
{
  scanf("%d%d",&n,&m);
  for (i=1;i<=n;i++)
  { 
    scanf("%d",&a[i]);
    if (a[i]>0)
      k=i;
  }
  l=k; r=1100000000; r=r*100000;
  while (l<r)
  {
    mid=(l+r)/2;
    if (check(mid))
      r=mid;
    else
      l=mid+1;
  }
  printf("%I64d\n",l);
  return 0;
}
