#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 5;

#define st first
#define nd second
#define make(a,b) make_pair(a,b)

typedef pair<int,int> pun;
typedef long long ll;

int tab[N];
int n,m;

bool mozna( long long x )
{
//	printf("mozna %lld\n",x);
	int need = 1;
	ll y = x - n;
	for (int i=n-1;i>=0;i--)
	{
		y -= tab[i];
//		printf("zos %lld\n",y);
		while( y < 0 )
		{
			y += x - i - 1;
			need ++;
		}
	}
	if ( need > m ) return 0;
	return 1;
}

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++) scanf("%d",tab+i);
	while( tab[n-1] == 0 ) n--;
	ll p = n+1, k = 1000000000000000LL;
	while( p < k )
	{
		if ( mozna( (p+k)/2 ) ) k = (p+k)/2;
		else p = (p+k)/2+1;
	}
	printf("%lld\n",p);
}
