#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int last,n,m;
long long a[100005],ans,num,l,r;
bool pd(long long x)
{
	int cnt=m;
	long long s=0;
	for (int i=1;i<=last;i++)
	{
		s+=a[i];
		while (s+i>=x)
		{
			s-=(x-i);
			cnt--;
			if (cnt<0) return 0;
		}
	}
	if (cnt==0) return s<=0;
	return 1;
}
int main()
{
	scanf("%d%d",&n,&m);
	num=0;
	for (int i=1;i<=n;i++)
	{
		scanf("%I64d",&a[i]);
		if (a[i]!=0) last=i;
		num+=a[i];
	}
	l=last;
	r=num+last;
	while (l<=r)
	{
		long long mid=(l+r)/2;
		if (pd(mid))
		{
			ans=mid;
			r=mid-1;
		}
		else 
			l=mid+1;
	}
	printf("%I64d\n",ans);
	return 0;
}
