#include <bits/stdc++.h>
using namespace std;
long long a[100000], b[100000], n, m, s, l, r, z, t, p;
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> n >> m;
    for(int i=0; i<n; i++){
        cin >> a[i];
        s+=a[i];
    }
    l=2; r=s+n;
    while(l<r){
        copy(a,a+n,b);
        z=l+r>>1;
        p=n-1;
        for(int i=0; i<m; i++){
            while(p>=0&&b[p]==0)p--;
            t=z-p-1;
            if(t<0)break;
            while(p>=0&&b[p]<=t)t-=b[p--];
            if(p>=0)b[p]-=t;
        }
        if(p<0){
            r=z;
        }
        else{
            l=z+1;
        }
    }
    cout << r;
}
