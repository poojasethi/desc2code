#define N (1<<17)
#define LL long long
#include <bits/stdc++.h>
using namespace std;

int n,m,a[N],b[N];

bool judge(LL t)
{
	for(int i=1;i<=n;i++)
		b[i]=a[i];

	LL ct=0,cnt=0;
	for(int i=n;i>=1;)
	{
		if(b[i]>0)
		{
			if(ct==0)
			{
				cnt++,ct=t-i;
				if(ct<=0 || cnt>m) return 0;
			}
			if(ct>b[i])
				ct-=b[i],b[i]=0;
			else
				b[i]-=ct,ct=0;
		}
		else
			i--;
	}
	return 1;
}

int main()
{
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		scanf("%d",a+i);

	LL Left=1,Right=(1ll<<60);
	for(;Left<Right;)
	{
		LL mid=(Left+Right)>>1;
		if(judge(mid))
			Right=mid;
		else
			Left=mid+1;
	}
	cout<<Left<<endl;
	return 0;
}
