#include<cstdio>
#include<algorithm>
#define rep(i,n) for(int i=1; i<=n; i++)
#define ll long long
using namespace std;
int main(){
    int n,m,a[100001],b[100001],v;
    scanf("%d%d",&n,&m);
    rep(i,n) {scanf("%d",&a[i]); if (a[i]) v=i;}
    ll l=1,r=100000000100000;
    while (l<r){
        ll d=(l+r)/2,t=v;
        rep(i,v) b[i]=a[i];
        rep(i,m){
            ll k=d-t; 
            while (k>=0 && t>0) {
                if (k>=b[t]) {k-=b[t]; t--;} 
                else {b[t]-=k; k=-1;};
            }
        }
        if (t==0) r=d;
        else l=d+1;
    }
    printf("%I64d",l);
}
