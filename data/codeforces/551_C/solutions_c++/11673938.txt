#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;
int n,m,a[100005];
long long time,sum;
bool can(long long t)
{
    long long from=0;
    int m1=m,w;
    for(int i=n;i>0;--i)
    {
        w=ceil(double(a[i]-from)/(t-i));
        if(m1<w)return 0;
        m1-=w;
        from=w*(t-i)-(a[i]-from);
    }
    return 1;
}

int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;++i)
    {
        scanf("%d",a+i);
        sum+=a[i];
    }
    while(a[n]==0)--n;
    long long b=n+1,e=sum+n,mid;
    while(b!=e)
    {
        mid=(e+b)>>1;
        if(can(mid))e=mid;
        else b=mid+1;
    }
    cout<<b;

}
