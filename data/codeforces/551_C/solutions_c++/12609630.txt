#include <bits/stdc++.h>

using namespace std;

const int maxn = 100 * 1000 + 100;

int n, m, a[maxn];
long long sum, cnt, l, mid, r = 1ll << 50;

bool check(long long x)
{
	sum = cnt = 0;
	for(int i = 1; i <= n; i++){
		sum += a[i];
		if(x - i <= 0){
			if(sum > 0){
				cnt += m;
				break;
			}
			else
				continue;
		}
		cnt += sum / (x - i);
		sum -= (sum / (x - i)) * (x - i);
	}
	if(sum > 0)
        cnt++;
	if(cnt > m)
		return false;
	return true;
}

int main()
{
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; i++)
		scanf("%d", &a[i]);

	while(l <= r){
		mid = (l + r) >> 1;
		if(check(mid))
			r = mid - 1;
		else
			l = mid + 1;
	}

	printf("%I64d", l);
	return 0;
}
