#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mp(x,y) make_pair(x,y)
#define pr pair<int,int>
ll n,m,i,last,start,end,mid,l,t;
ll a[100005],b[100005];
bool f(ll time) {
	l=last;
	for(i=1;i<=n;i++) b[i]=a[i];
	for(i=1;i<=m;i++) {
		t=time;
		t-=l;
		while(t>=b[l] && l>0) {
			t-=b[l];
			l--;
		}
		if(l>0)
			b[l]-=t;
	}
	if(l==0) return true;
	else return false;
}

int main()
{
	std::ios_base::sync_with_stdio(false);
	cin>>n>>m;
	for(i=1;i<=n;i++) { cin>>a[i]; if(a[i]!=0) last=i; }
	start=0,end=10e15;
	while(start<end) {
		mid = start+(end-start)/2;
		if(f(mid)==true)
			end=mid;
		else
			start=mid+1;
	}
	cout<<start<<endl;
	return 0;
}
