#include<bits\stdc++.h>
using namespace std;
long long int a[100010],b[100010];
int main()
{
	long long int n,m;
	cin>>n>>m;
	for(long long int i=0;i<n;i++)cin>>a[i];
	while(n&&!a[n-1])n--;
	long long int beg=0,end=2e14;
	do{
		long long int mid=(beg+end)/2,t,w=1,k=0;
		t=mid; 
		for(long long int i=0;i<n;i++)b[i]=a[i];
		for(long long int i=0;i<n;i++)
		{
			t--;
			bool l=true;
			if(t>=b[i])t-=b[i],b[i]=0,k=0;
			else
			{
				if(t>0)b[i]-=t;
				t=mid-i-1;
				l=false;
				t++;
				w++;
				k++;
			}
			if(k>a[i]+10)
			{
				w=m+1;
				break;
			}
			if(!l)i--;
		}
		if(w>m)beg=mid;
		else end=mid;
	}while(beg+1<end);
	cout<<end<<endl;
}
