#include<bits/stdc++.h>
#define inf (1<<30)
#define mx 100005
using namespace std;

int n,k,a[mx];
char inp[12];

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cin>>n>>k;
    for(int i=0;i<n;i++){
        cin>>inp;
        if(inp[0]=='?'){
            a[i]=inf;
        }
        else{
            a[i]=atoi(inp);
        }
    }

    for(int i=0;i<k;i++){
        int range=((n-i-1)/k)+1,prev=-inf;
        //cout<<"i:"<<i<<"\trange:"<<range<<"\tprev:"<<prev<<endl;
        for(int j=0;j<range;j++){
            int p=j,next=inf;
            while(p<range && a[i+k*p]==inf)p++;
            //cout<<"\tj:"<<j<<"\tp:"<<p<<endl;
            if(p<range){
                next=a[i+p*k];
            }
            if(next<=prev+p-j){
                cout<<"Incorrect sequence";
                return 0;
            }

            int up=(p-j+1)/2;
            up=max(up,prev+p-j+1);
            up=min(up,next);

            for(int x=j;x<p;x++){
                a[i+x*k]=up-p+x;
            }
            prev=next;
            j=p;
        }
    }

    for(int i=0;i<n;i++){
        cout<<a[i];
        if(i==n-1)cout<<"\n";
        else cout<<" ";
    }



    return 0;
}
