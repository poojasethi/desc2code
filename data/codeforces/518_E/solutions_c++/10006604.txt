#include <iostream>
#include <cstdio>

using namespace std;

const int N=2*100*1000+5;
const int MAX=1000*1000*1000+N;

int a[N],lb[N],qm[N];

int main() {
    int n,k;
    cin>>n>>k;
    for(int i=0;i<k;i++) {
        lb[i]=-MAX;
        qm[i]=0;
    }
    for(int i=0;i<n+k;i++) {
        string s;
        if (i<n) {
            cin>>s;
            if (s[0]!='?') {
                sscanf(s.c_str(),"%d", &a[i]);
            }
        } else {
            a[i] = MAX;
        }
        if (k==0)
            continue;
        int imk=i%k;
        if (s[0]=='?') {
            qm[imk]++;
        } else {
            if (a[i]-lb[imk]>qm[imk]) {
                int qs=lb[imk]+1;
                if(a[i]-1<=qm[imk]/2) {
                    qs=a[i]-qm[imk];
                } else if(qs<-(qm[imk]/2)) {
                    qs=-(qm[imk]/2);
                }
                for (int j=qm[imk]; j>0; j--) {
                    a[i-j*k]=qs++;
                }
            } else {
                //cerr<<"err"<<i<<endl;
                k=0;
            }
            lb[imk]=a[i];
            qm[imk]=0;
        }
    }
    if (k==0) {
        cout<<"Incorrect sequence"<<endl;
    } else {
        for (int i=0; i<n; i++) {
            cout<<a[i]<<" ";
        }
        cout<<endl;
    }
    //main();
    return 0;
}
