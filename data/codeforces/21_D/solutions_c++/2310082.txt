#include<stdio.h>
#include<string.h>
const int MAXN=15;
const int INF=1000000000;
int map[MAXN][MAXN];
int dp[1<<MAXN];
bool vis[MAXN];
int min(int a,int b)
{
    return a>b?b:a;
}
int main()
{
    int n,m,i,j,k;
    scanf("%d%d",&n,&m);
    int x,y,c;
    int ans=0;
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            map[i][j]=INF;
    for(i=0;i<n;i++)
        map[i][i]=0;
    int init=0;
    for(i=0;i<m;i++)
    {
        scanf("%d%d%d",&x,&y,&c);
        x--;y--;
        ans+=c;
        if(map[x][y]>c)
            map[x][y]=map[y][x]=c;
        vis[x]=1;
        vis[y]=1;
        init ^= 1<<x;
        init ^= 1<<y;
    }
    for(k=0;k<n;k++)
        for(i=0;i<n;i++)
            for(j=0;j<n;j++)
                if(map[i][k]+map[k][j]<map[i][j])
                    map[i][j]=map[i][k]+map[k][j];
    for(i=0;i<n;i++)
    {
        if(vis[i]&&map[0][i]==INF)
        {
            printf("-1\n");
            return 0;
        }
    }
    for(i=0;i<=init;i++)
        dp[i]=INF;
    dp[init]=0;
    for(i=init;i>0;i--)
    {
        if(dp[i]!=INF)
        {
            for(j=0;j<n;j++)if(i&(1<<j))
            {
                for(k=0;k<n;k++)if(i&(1<<k))
                {
                    if(j==k)
                        continue;
                    int t=i-(1<<j)-(1<<k);
                    dp[t]=min(dp[t],dp[i]+map[j][k]);
                }
            }
        }
    }
    printf("%d\n",ans+dp[0]);
    return 0;
}
