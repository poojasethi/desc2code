#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int INF = 0x7f7f7f7f;

int n,m,sum=0;
int d[16],u[2010],v[2010];
int g[16][16];
int f[1<<15];

void checkmin(int &a,int b){if(b<a) a=b;}
int main()
{
	scanf("%d%d",&n,&m);
	memset(g,0x7f,sizeof(g));
	memset(f,0x7f,sizeof(f));
	for(int i=1;i<=m;i++){
		int w;scanf("%d%d%d",&u[i],&v[i],&w);sum+=w;
		g[u[i]][v[i]]=min(g[u[i]][v[i]],w);
		g[v[i]][u[i]]=min(g[v[i]][u[i]],w);
		d[u[i]]++;d[v[i]]++;
	}
	for(int k=1;k<=n;k++)
		for(int i=1;i<=n;i++) if(g[i][k]<INF)
			for(int j=1;j<=n;j++) if(g[k][j]<INF)
				g[i][j]=min(g[i][j],g[i][k]+g[k][j]);
	for(int i=1;i<=m;i++) if(g[1][u[i]]>=INF || g[1][v[i]]>=INF) return puts("-1") && 0;
	int state=0;
	for(int i=1;i<=n;i++) if(!(d[i]&1)) state|=1<<i-1;
	f[state]=sum;
	for(int i=state;i<(1<<n);i++) if(f[i]<INF)
		for(int j=1;j<=n;j++) if(!(state&(1<<j-1)))
			for(int k=j+1;k<=n;k++) if(!(state&(1<<k-1)) && g[j][k]<INF)
				checkmin(f[i^(1<<j-1)^(1<<k-1)],f[i]+g[j][k]);
	printf("%d\n",f[(1<<n)-1]<INF ? f[(1<<n)-1] : -1);
	return 0;
}
