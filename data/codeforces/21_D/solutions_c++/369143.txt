#include <iostream>
#include <cstdio>
#include <algorithm>
#define FOR(a,b,c) for(long a=b;a<=c;a++)
using namespace std;
const long inf = 10000000;
long res[1000010];
long a[20][20];
long b[20],d[20];
long n,m,s,nn;
int main(){
scanf("%ld %ld",&n,&m);
FOR(i,1,n)
FOR(j,1,n)a[i][j]=inf*(i!=j);
FOR(i,1,m){
           long u,v,z;
           scanf("%ld %ld %ld",&u,&v,&z);
           d[u]++;
           d[v]++;
           s+=z;
           a[u][v]=min(a[u][v],z);
           a[v][u]=a[u][v];
           }
FOR(k,1,n)
FOR(i,1,n)
FOR(j,1,n)
if (a[i][k]!=inf && a[k][j]!=inf)
a[i][j]=min(a[i][j],a[i][k]+a[k][j]);
FOR(i,1,n)
if (a[1][i]==inf && d[i]>0){
                  printf("-1");
                  return 0;
                  }
FOR(i,1,n)
if (d[i]%2)b[++nn]=i;
if (nn>0){
res[0]=0;
FOR(i,1,(1<<nn)-1)res[i]=inf;
FOR(i,0,(1<<nn)-2)
if (res[i]!=inf)
FOR(u,1,nn-1)
if (((i>>(u-1))&1)==0)
FOR(v,u+1,nn)
if (((i>>(v-1))&1)==0)
res[i | (1<<(u-1)) | (1<<(v-1))]=min(res[i | (1<<(u-1)) | (1<<(v-1))],res[i]+a[b[u]][b[v]]);
printf("%ld",res[(1<<nn)-1]+s);
}else printf("%ld",s);
return 0;
};
