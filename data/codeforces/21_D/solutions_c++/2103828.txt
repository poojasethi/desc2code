#include<iostream>
#include<cstdio>

using namespace std;

const int Inf = 1000000000;

int n , m , f[2][( 1 << 15 ) + 1];
int par[15] , lev[15];

int anc( int t ) {
	if ( par[t] != t ) par[t] = anc( par[t] );
	return par[t];
}

int main() {
	scanf( "%d%d" , &n , &m );
	for ( int i = 0 ; i < n ; i++ ) par[i] = i;
	for ( int i = 1 ; i < ( 1 << n ) ; i++ ) f[0][i] = Inf;
	for ( int i = 1 ; i <= m ; i++ ) {
		int I = i & 1 , _I = I ^ 1;
		int u , v , r; scanf( "%d%d%d" , &u , &v , &r );
		lev[--u]++ , lev[--v]++;
		par[anc( u )] = anc( v );
		for ( int j = 0 ; j < ( 1 << n ) ; j++ )
			f[I][j] = min( f[_I][j] + 2 * r , f[_I][j ^ ( 1 << u ) ^ ( 1 << v )] + r );
	}
	
	for ( int i = 0 ; i < n ; i++ )
		if ( anc( i ) != anc( 0 ) && lev[i] > 0 ) {
			printf( "-1\n" );
			return 0;
		}
	printf( "%d\n" , f[m & 1][0] );
	return 0;
}
