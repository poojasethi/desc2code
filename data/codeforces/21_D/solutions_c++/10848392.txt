#include<bits/stdc++.h>

using namespace std;

int DST[15][15];
int DP[32768],ans;
int deg[15],x,y,c;
int n,m,M,i,j,k;

main()
{
 cin>>n>>m;

 for (i=0;i<n;i++)
  for (j=0;j<n;j++)
   DST[i][j]=1000000000;

 for (i=1;i<=m;i++)
  {
   cin>>x>>y>>c;
   deg[--x]++;
   deg[--y]++;
   ans+=c;
   DST[x][y]=min(DST[x][y],c);
   DST[y][x]=min(DST[y][x],c);
  }

 for (k=0;k<n;k++)
  for (i=0;i<n;i++)
   for (j=0;j<n;j++)
    DST[i][j]=min(DST[i][j],DST[i][k]+DST[k][j]);

  for (i=1;i<n;i++)
   if (deg[i] && DST[0][i] >= 1000000000)
    { cout<<-1<<endl; return 0; }

  for (i=1;i<(1<<n);i++)
   DP[i]=1000000000;

  for (i=1;i<(1<<n);i++)
   for (j=0;j<n;j++)
    for (k=0;k<n;k++)
     if (j != k && ((1<<k)&i) && ((1<<j)&i) )
      DP[i]=min(DP[i],DP[i-(1<<j)-(1<<k)]+DST[j][k]);

  for (i=0;i<n;i++)
   if (deg[i]%2) M+=(1<<i);

  cout<<ans+DP[M]<<endl;
}
