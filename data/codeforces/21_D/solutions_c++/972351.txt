#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define REP(i,n) for(i=0;i<n;i++)

const int INF=1e9;
int g[20][20],f[1<<15],d[16],a[2100];

int main(){
        int n,m,i,j,k,s=0;
        cin>>n>>m;
        if(!m) {puts("0");return 0;}
        REP(i,n)REP(j,n) g[i][j]=INF;
        REP(i,m){
                int x,y,z;
                cin>>x>>y>>z;
                x--;y--;g[x][y]=g[y][x]=min(g[x][y],z);
                d[x]++;d[y]++;s+=z;
                a[i]=x;
        }
        REP(i,n) g[i][i]=0;
        REP(k,n)REP(i,n)REP(j,n) g[i][j]=min(g[i][j],g[i][k]+g[k][j]);
        REP(i,m) if (g[0][a[i]]==INF) {puts("-1");return 0;}
        REP(i,1<<n) f[i]=INF;
        f[0]=0;
        REP(i,1<<n){
                //cout<<i<<' '<<f[i]<<endl;
                REP(j,n)if (((1<<j)&i)==0&&d[j]%2==0) f[i+(1<<j)]=min(f[i+(1<<j)],f[i]);
                REP(j,n)if (((1<<j)&i)==0&&d[j]%2==1)
                REP(k,n)if (j<k&&((1<<k)&i)==0&&d[k]%2==1) f[i+(1<<j)+(1<<k)]=min(f[i+(1<<j)+(1<<k)],f[i]+g[j][k]);
        }
        if (f[(1<<n)-1]==INF) cout<<-1<<endl;else cout<<f[(1<<n)-1]+s<<endl;
}
