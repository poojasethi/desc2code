#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 16
#define inf 100000000

using namespace std;

int ss[1<<maxn];
int deg[maxn];
int d[maxn][maxn];
int f[1<<maxn];
int n,m;
int ans;
int main(){
	for (int i=0;i<16;i++) ss[1<<i]=i;
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++)
		for (int j=0;j<n;j++)
			d[i][j]=(i==j?0:inf);
	for (int i=0;i<m;i++){
		int u,v,w;
		scanf("%d%d%d",&u,&v,&w);
		--u,--v;
		deg[u]++;
		deg[v]++;
		d[u][v]=min(d[u][v],w);
		d[v][u]=min(d[v][u],w);
		ans+=w;
	}
	for (int k=0;k<n;k++)
		for (int i=0;i<n;i++)
			for (int j=0;j<n;j++)
				if (d[i][k]+d[k][j]<d[i][j]) d[i][j]=d[i][k]+d[k][j];
	for (int i=1;i<n;i++)
		if (deg[i]>0 && d[0][i]==inf){
			puts("-1");
			return 0;
		}
	int tmask=0;
	for (int i=0;i<n;i++)
		if (deg[i]%2==1) tmask|=(1<<i);
	for (int i=0;i<(1<<n);i++) f[i]=inf;
	f[tmask]=0;
	for (int mask=tmask;mask>0;mask--)
		if (f[mask]<inf){
			int x=(mask&(mask^(mask-1)));
			x=ss[x];
			for (int j=x+1;j<n;j++)
				if ((mask^(1<<j))>0){
					int cmask=mask;
					cmask^=(1<<x);
					cmask^=(1<<j);
					f[cmask]=min(f[cmask],f[mask]+d[x][j]);
				}
		}
	printf("%d\n",ans+f[0]);
	return 0;
}
