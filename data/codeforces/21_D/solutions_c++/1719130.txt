#include<iostream>
#include<algorithm>
using namespace std;
#define N 15
#define inf 1987654321
int n,m;
int F[1<<N],dis[N][N],d[N];
bool v[N];
int main()
{
    int i,j,k;
    cin>>n>>m;
    if(m==0) { cout<<0<<endl; return 0;}
    for(i=0;i<n;++i) for(j=0;j<n;++j) if(i^j) dis[i][j]=inf;
    int res=0;
    for(i=0;i<m;i++)
    {
        int a,b,c;
        cin>>a>>b>>c;
        a--,b--;
        v[a]=true;
        d[a]++,d[b]++;
        res+=c;
        if(dis[a][b]>c) dis[a][b]=dis[b][a]=c;

    }
    for(k=0;k<n;++k)
    for(i=0;i<n;++i)
    for(j=0;j<n;++j)
    if(dis[i][k]<inf&&dis[k][j]<inf&&dis[i][j]>dis[i][k]+dis[k][j]) dis[i][j]=dis[i][k]+dis[k][j];
    for(i=0;i<n;i++) if(v[i]&&dis[0][i]>=inf) { cout<<-1<<endl; return 0;}
    for(int s=1;s<(1<<n);++s)
    {
        for(i=n-1;i>=0&&!(s&(1<<i));i--);
        F[s]=inf;
        for(j=i-1;j>=0;--j)
        {
            int temp=F[s^(1<<i)^(1<<j)];

            if(temp<inf&&dis[i][j]<inf)
               F[s]=min(F[s],dis[i][j]+temp);
        }
    }
    int s=0;
    for(i=0;i<n;i++) if(d[i]&1) s+=(1<<i);
    cout<<res+F[s]<<endl;

}
