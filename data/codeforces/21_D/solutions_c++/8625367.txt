#include <bits/stdc++.h>
using namespace std;
#define rep(i,x,y) for(i=x;i<=y;++i)
#define repd(i,x,y) for(i=x;i>=y;--i)
const int maxn=18;

int n,m,init,an,i,j,k,allset,x,y,z;
int D[maxn][maxn],F[1<<maxn],met[maxn];

int main(){
    //freopen("input.txt","r",stdin);
    scanf("%d%d",&n,&m);
    init=0;
    an=0;
    rep(i,1,n)rep(j,1,n)D[i][j]=1e9;
    rep(i,1,m){
        scanf("%d%d%d",&x,&y,&z);
        an+=z;
        met[x]=1;
        met[y]=1;
        D[x][y]=min(D[x][y],z);
        D[y][x]=min(D[y][x],z);
        init^=(1<<(x-1));
        init^=(1<<(y-1));
    }
    rep(k,1,n)rep(i,1,n)rep(j,1,n)
        D[i][j]=min(D[i][j],D[i][k]+D[k][j]);
    rep(i,1,n)if(met[i]&&D[1][i]==1e9){
        puts("-1");return 0;
    }
    allset=(1<<n)-1;
    fill(F,F+allset,1e9);
    F[init]=0;
    repd(i,init,0){
        rep(j,1,n)if(i&(1<<(j-1))){
            rep(k,1,n)if(i&(1<<(k-1)))
                F[i^(1<<(j-1))^(1<<(k-1))]=min(F[i^(1<<(j-1))^(1<<(k-1))],F[i]+D[j][k]);
        }
    }
    printf("%d\n",F[0]+an);
    return 0;
}
