#include <algorithm>
#include <iostream>
using namespace std;

typedef pair<int, int> pii;
const int oo = 1e9;

int n, m, ans, dp[(1 << 15) + 2], dis[20][20], deg[20], a[20];

void smin(int &a, int b) { a = min(a, b); }

int main() {
    cin.sync_with_stdio(false);
    cin >> n >> m;
    fill(dis[0], dis[n], oo);
    for (int i = 0; i < n; i++)
        dis[i][i] = 0;
    for (int i = 0; i < m; i++) {
        int v, u, w;
        cin >> v >> u >> w;
        deg[--u]++;
        deg[--v]++;
        ans += w;
        smin(dis[u][v], w);
        smin(dis[v][u], w);
    }
    for (int k = 0; k < n; k++)
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                smin(dis[i][j], dis[i][k] + dis[k][j]);
    for (int i = 1; i < n; i++)
        if (deg[i] && dis[0][i] == oo) {
            cout << -1 << '\n';
            return 0;
        }
    m = 0;
    for (int i = 0; i < n; i++)
        if (deg[i] % 2)
            a[m++] = i;
    int m2 = 1 << m;
    fill(dp + 1, dp + m2, oo);
    for (int mask = 0; mask < m2; mask++)
        for (int v = 0; v < m; v++)
            if (mask & 1 << v)
                for (int u = 0; u < v; u++)
                    if (mask & 1 << u)
                        smin(dp[mask], dis[a[u]][a[v]] + dp[mask - (1 << u) - (1 << v)]);
    cout << (dp[m2 - 1] == oo? -1: dp[m2 - 1] + ans) << '\n';
    return 0;
}
