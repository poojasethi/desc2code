s = '.' + raw_input().lower()
n = len(s)
a, b = [0] * n, [0] * n
for i in range(1,n):
	a[i] = a[i-1] + (-1 if s[i] in 'aeiou' else 2)
p, r = 0, 0
for i in range(n):
	if a[i] < p:
		p = a[i]
		b[p] = i
for i in range(n):
	r = i if a[i] > 0 else max(r,i-b[a[i]])
if r: print r, map(lambda i: a[i] >= a[i-r], range(r,n)).count(True)
else:
	print 'No solution'
