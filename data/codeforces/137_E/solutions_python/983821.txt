s=raw_input().lower()
a=[[0,0]]
for i,c in enumerate(s):
	a.append([a[i][0]+(-1 if c in "aeiou" else +2),i+1])
a.sort()
mn=100000000
ans,c=0,0
for i in a:
	if(mn<i[1]):
		if i[1]-mn > ans:
			ans=i[1]-mn
			c=1
		elif i[1]-mn == ans:
			c+=1
	mn=min(mn,i[1])
if ans==0:
	print "No solution"
else:
	print ans,c
