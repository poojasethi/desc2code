#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
const int N = 200009;
string q = "aeuio", s;
int n, i, j, a[N], b[N], m, d[N * 2], c, e, l;
int main () {
 	cin >> s;
 	n = s.length();
 	s = '&' + s;
 	for (i = 1; i <= n; i++) {
 		if (s[i] < 95) s[i] += 32;
 		a[i] = 2;
 		for (j = 0; j < 5; j++)
 			if (s[i] == q[j])
 				a[i] = -1;
 	}
 	m = 0;
 	for (i = 1; i <= n; i++) { 
 		b[i] = b[i - 1] + a[i];
 		if (b[i] < m) 
 			m = b[i];
 	}
 	for (i = 1; i <= n; i++) {
 		b[i] -= m; 
 		d[b[i]] = i;
 	}
 	c = -m;

 	m = e = 0;
 	for (i = 1; i <= n; i++) {
 		if (b[n] >= c) 
 			l = n - i + 1;
 	    else
 	    	l = d[c] - i + 1;
 	    if (m < l) {
 	    	m = l;
 	    	e = 0;
 	   	}	
 	   	if (m == l) e++;
 	   	c += a[i];
 	}
 	if (m != 0) cout << m << " " << e;
 	else cout << "No solution";   	
}
