#include <bits/stdc++.h>

using namespace std;

pair <int,int> cum[200005];

int chk(char p)
{
    if ( p == 'a' || p == 'e' || p == 'i' || p == 'o' || p == 'u' ) return 1;
    if ( p == 'A' || p == 'E' || p == 'I' || p == 'O' || p == 'U' ) return 1;
    return 0;
}

int main()
{
    string s;
    cin >> s;
    int n = (int)s.size();
    cum[0].first = 0;
    cum[0].second = 0;
    for ( int i = 1; i <= n; i++ ) {
        cum[i].second = i;
        if ( chk(s[i-1]) ) cum[i].first = cum[i-1].first - 1;
        else cum[i].first = cum[i-1].first + 2;
    }
    sort(cum, cum+n+1);
    int mn = 100000000,ans = 0,cnt = 0;
    for ( int i = 0; i <= n; i++ ) {
        if ( cum[i].second - mn > ans ) {
            ans = cum[i].second - mn;
            cnt = 1;
        }
        else if ( cum[i].second - mn == ans ) {
            cnt++;
        }
        mn = min(mn, cum[i].second);
    }
    if ( ans == 0 ) cout << "No solution" << endl;
    else cout << ans << " " << cnt << endl;
    return 0;
}
