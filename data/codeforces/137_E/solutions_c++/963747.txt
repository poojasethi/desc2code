#include <string.h>
#include <ctype.h>
#include <stdio.h>
#define REP(i,n) for (int i=0; i<n; ++i)

int sc[200100], m[200100], last=1<<30, res=0, cnt=0, n;

int main() {
   char s[200100]; gets(s); 
   n = strlen(s);
   REP(i,n) s[i] = tolower(s[i]);
   REP(i,n) sc[i+1] = sc[i] + ((s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u') ? -1:2);
   REP(i,n+1) if (sc[i]<last) last=sc[i], m[-sc[i]]=i;
   REP(i,n+1) {int x = i-(sc[i]>0?0:m[-sc[i]]); if (res<x) res=x;}
   REP(i,n-res+1) if (sc[i+res]>=sc[i]) cnt++;
   if (res==0) puts("No solution"); else printf("%d %d",res,cnt);   
}
