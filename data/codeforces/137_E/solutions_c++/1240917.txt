#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char s[200005];
vector<pair<int, int> > a;

bool isvowel(char x) {
	return x=='a' || x=='A' || x=='e' || x=='E' || x=='i'
		|| x=='I' || x=='o' || x=='O' || x=='u' || x=='U';
}

int main(void) {
	scanf("%s", s);
	int sum=0, mxlen=0, cnt=0;
	a.push_back(make_pair(0, -1));
	for(int i=0;s[i];i++) {
		if(isvowel(s[i])) ++sum; else sum -= 2;
		if(sum > a.back().first) {
			a.push_back(make_pair(sum, i));
		} else {
			int p=lower_bound(a.begin(), a.end(), make_pair(sum,-1))-a.begin();
			if(i-a[p].second > mxlen) { mxlen = i-a[p].second; cnt=0; }
			if(mxlen == i-a[p].second) ++cnt;
		}
	}
	if(cnt==0) puts("No solution");
	else printf("%d %d\n", mxlen, cnt);
	return 0;
}
