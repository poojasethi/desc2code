#include <bits/stdc++.h>
//Damned if I don't take what I want.

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN=  2e5 + 100, MAXA = 3 * MAXN, S = 2 * MAXN;

int n;
pii a[MAXN];

bool isvow(char c){ return (c == 'a' || c == 'o' || c == 'e' || c == 'i' || c == 'u'); }

int main(){
	ios::sync_with_stdio( false );
	cout.tie(0); cin.tie(0);

	string s; cin >> s; n = s.size();

	for (int i = 0; i < n; i++){
		if (i) a[i].first = a[i - 1].first; else a[i].first = 0;
		if (isvow(tolower(s[i])) ) a[i].first--;
		else a[i].first += 2;
		a[i].second = i + 1;
	}
	a[n++] = pii(0, 0);

	sort(a, a + n);

	int ans = 0, cnt = 0, ind = n;
	for (int i = 0; i < n; i++){
		//cerr << a[i].first << ", " << a[i].second << " : " << ind << endl;
		int x = a[i].second - ind;
		if (ans == x) cnt++;
		if (ans < x) ans = x, cnt = 1;
		ind = min(ind, a[i].second);
	}
		
	if (ans) cout << ans << " " << cnt << endl;
	else cout << "No solution\n";
}
