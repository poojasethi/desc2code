#include <iostream>
#include <vector>
#include <algorithm>
#include <map>

using namespace std;

const int Max=2*100*1000+10;
int a[Max];
map <int , int > m;
int jav[Max];
string s;
int n; 
int s1,s2;

int main()
{
	cin>>s;
	n=s.size();
	for(int i=0;i<n;++i)
	{
		if(int (s[i])<int ('a'))
			s[i]+=int ('a')-int ('A');
		if(s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u')
			++s1;
		else
			++s2;
		a[i]=s1-2*s2;	
	}
	int k=0;
	for(int i=0;i<n;++i)
	{
		if(a[i]<=0)
			jav[i]=i+1;
		else
		{
			if(m[a[i]]!=0)
				jav[i]=i-m[a[i]]+1;
			else
				m[a[i]]=i+1;
		}
	}
	int ma=0;
	int t=0;
	for(int i=0;i<n;++i)
	{
		if(jav[i]>ma)
		{
			ma=jav[i];
			t=1;
		}
		else
			if(ma==jav[i])
				++t;
	}
	if(ma==0)
		cout<<"No solution"<<endl;
	else
		cout<<ma<<" "<<t<<endl;
}
