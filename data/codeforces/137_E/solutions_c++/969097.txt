#include <iostream>
#include <algorithm>
#include <cstring>

#define FOR(a,b,c) for(int a=b;a<=c;a++)

using namespace std;

char str[200010];
int len,a[200010],s[200010],ans=1,c=0;
bool nosol=true;

int code(char c)
{
	if(c=='a' || c=='e' || c=='i' || c=='o' || c=='u') return -1;
	if(c=='A' || c=='E' || c=='I' || c=='O' || c=='U') return -1;
	return 2;
}

bool comp(int i, int j)
{
	if(a[i]==a[j]) return i<j;
	return a[i]<a[j];
}

int main()
{
	cin>>str;
	len=strlen(str);
	FOR(i,1,len) 
	{
		a[i]=a[i-1]+code(str[i-1]);
		if(a[i]>=0) ans=i;
		if(code(str[i-1])==2) nosol=false;
		s[i]=i;
	}
	if(nosol) 
	{
		cout<<"No solution\n";
		return 0;
	}
	sort(s+1,s+len+1,comp);
	int mi=s[1];
	FOR(i,2,len)
	{
		if(s[i]>mi) ans=max(ans,s[i]-mi); else mi=s[i];
	}
	FOR(i,ans,len) if((a[i]-a[i-ans])>=0) c++;
	cout<<ans<<" "<<c<<endl;
	return 0;
}
