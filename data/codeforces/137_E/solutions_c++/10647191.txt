#include <stdio.h>
#include <string.h>

int f[3000000], g[3000000], ff[3000000], gg[3000000];
char a[3000000];

int main()
{
    int i, n, x, y, ansx, ansy, p, m;
    scanf("%s", a);
    n=strlen(a);
    x=y=0;
    memset(ff, 63, sizeof(ff));
    p=3*n;
    while (p)
    {
        if (-1<ff[p]) ff[p]=-1;
        p-=(p & -p);
    }
    ansx=-1; ansy=0;
    for (i=0; i<n; i++) 
    {
        if (a[i]=='a' || a[i]=='e' || a[i]=='i' || a[i]=='o' || a[i]=='u' || a[i]=='A' || a[i]=='E' || a[i]=='I' || a[i]=='O' || a[i]=='U')
        x++; else y++;
        p=x-2*y+3*n; m=-1;
        while (p<7*n)
        {
            if (i-ff[p]>m) m=i-ff[p];
            p+=(p & -p);
        }
        p=x-2*y+3*n; 
        while (p)
        {
            if (i<ff[p]) ff[p]=i;
            p-=(p & -p);
        }
        if (m>ansx)
        {
            ansx=m;
            ansy=1;
        }   else
        if (m==ansx) ansy++;
    }
    if (ansx!=-1) printf("%d %d", ansx, ansy);
    else printf("No solution");
    return 0;
}
