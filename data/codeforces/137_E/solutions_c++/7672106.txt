#include<bits/stdc++.h>
using namespace std;
const int N=2e5+10;
struct node
{
    int id,k;
    bool operator <(const node&o)const
    {
        if (k==o.k) return id>o.id;
        else return k<o.k;
    }
}a[N];
char s[N];

inline bool yuanyin(char ch)
{
    if (ch<='Z') return (ch=='A'||ch=='E'||ch=='I'||ch=='O'||ch=='U');
    else return (ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u');
}

int main()
{
    scanf("%s",s);
    int n=strlen(s);
    a[0].id=0;
    a[0].k=0;
    for (int i=0,p=0,q=0;i<n;i++) {
        if (yuanyin(s[i])) p++; else q++;
        a[i+1].id=i+1;
        a[i+1].k=p-2*q;
    }
    n++;
    sort(a,a+n);
    int ans=0,num=0,t=0;
    for (int i=1;i<n;i++) {
        if (a[i].id<a[t].id) {
            int len=a[t].id-a[i].id;
            if (len>ans) {
                ans=len;
                num=1;
            }
            else if (len==ans) num++;
        }
        else t=i;
    }
    if (ans==0) printf("No solution\n");
    else printf("%d %d\n",ans,num);
    return 0;
}
