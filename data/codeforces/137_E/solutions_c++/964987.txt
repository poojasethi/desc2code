#include<iostream>
#include<cstdio>
#include<string.h>
#include<string>
#include<math.h>
#include<algorithm>
#include<vector>
using namespace std;
#define maxn 200009

string s;
int a[maxn],b[maxn];

bool isvow(char c){
    c = tolower(c);
    if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u') return true;
    return false; 
}

int main(){
    cin >> s;
    int n = s.size();
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    for(int i=0;i<n;i++){
        if(isvow(s[i])) a[i+1] = a[i]+1;
        else a[i+1] = a[i]-2;   
    }
    int ans = 0;
    int tmp = 0;
    for(int i=0;i<n+1;i++){
        if(a[i]>tmp){
            tmp = a[i];
            b[tmp] = i;
        }
    }
    for(int i=0;i<n+1;i++){
        if(a[i]<=0) ans = i;
        else ans = max(ans,i-b[a[i]]);    
    }
    int num=0;
    for(int i=0;i<n-ans+1;i++){
        if(a[i+ans]<=a[i]) num++;
    }
    if(ans) cout << ans << " " << num << endl;
    else cout << "No solution" << endl;
    return 0;
}
