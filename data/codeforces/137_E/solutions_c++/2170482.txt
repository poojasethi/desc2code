#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
char a[200001],v[]="AEIOUaeiou";
pair<int,int> p[200001];
int L,maxI,best,sum,cnt,i;
int main(){
	scanf("%s",a);
	for(;a[L];L++)a[L]=strchr(v,a[L])?-1:2;
	for(i=1;i<L+1;i++)p[i]=make_pair(p[i-1].first-a[i-1],-i);
	sort(p,p+L+1);
	for(i=0;i<L+1;i++){
		if(-p[i].second>maxI)maxI=-p[i].second;
		int c=maxI+p[i].second;
		if(c>best)best=c,cnt=0;
		if(c==best)cnt++;
	}
	best?printf("%d %d\n",best,cnt):printf("No solution\n");
}
