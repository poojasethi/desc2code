#include<stdio.h>
#include<algorithm>
#include<string.h>
using namespace std;
const int MAXN = 201000;
int res[MAXN], m[MAXN], v[MAXN];
int S;
char t[MAXN];
int main () {
    scanf("%s", &t[1]);
    int x = strlen(&t[1]);
    for (int c = 1; c <= x; c++) {
        if (t[c] >= 'A' && t[c] <= 'Z') t[c] = t[c]-'A'+'a';
        if (t[c] == 'a' || t[c] == 'e' || t[c] == 'i' || t[c] == 'o' ||  t[c] == 'u') v[c] = 1;
        else v[c] = -2;
    }
    S = 0;
    int topo = 0;
    for (int c = 1; c <= x; c++) {
        S += v[c];
        if (S <= 0) res[c] = c;
        else {
            if (S > topo) {
                topo++;
                m[S] = c;
            }
            res[c] = c-m[S];
        }
    }
    int maior = -1;
    int cnt = 0;
    for (int c = 1; c <= x; c++) maior = max(maior, res[c]);
    if (maior == 0) printf("No solution");
    else {
        for (int c = 1; c <= x; c++) if (res[c] == maior) cnt++;
        printf("%d %d", maior, cnt);
    }
    return 0;
}
