#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;

const int N = 200010;
const int D = 200010;

char str[N];

int c[N * 3];

void update(int p, int d) {
	p += D;
	for (int i = p; i < N * 3; i += i & -i)
		c[i] = min(c[i], d);
}

int query(int p) {
	p += D;
	int res = c[p];
	for (int i = p; i; i -= i & -i)
		res = min(res, c[i]);
	return res;
}

char CH[] = "aeiou";
bool isVowl(char ch) {
	ch = tolower(ch);
	return strchr(CH, ch) != NULL;
}

int main() {
	gets(str);
	int n = strlen(str);
	memset(c, 0x3f, sizeof(c));
	update(0, 0);
	int s = 0;
	int ans = 0, cnt = 0;
	for (int i = 0; i < n; i++) {
		s += isVowl(str[i]) ? -1 : 2;
		int tmp = i + 1 - query(s);
		if (tmp > ans) {
			ans = tmp;  
			cnt = 1;
		} else if (tmp == ans)
			cnt++;
		update(s, i + 1);
	}
	if (ans == 0)
		cout << "No solution" << endl;
	else
		cout << ans << ' ' << cnt << endl;
    return 0;
}
