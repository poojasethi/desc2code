#include<iostream>
#include<cstring>
#include<cstdio>
#include<vector>
#include<algorithm>
#define mod 1
using namespace std;

struct node{
	int s,num;
}p[200005];
char s[200005];
int ans[200005][2],res=1000000,q=0;
bool cmp(node x,node y){
	if(x.s!=y.s) return x.s<y.s;
	else return x.num<y.num;
}

int main()
{
	scanf("%s",s+1);
	int n=strlen(s+1);
	for(int i=1;i<=n;i++)
	{
		char x=s[i];
		if('A'<=x&&x<='Z') x=x-'A'+'a';
		if(x=='a'||x=='e'||x=='u'||x=='i'||x=='o')
			ans[i][0]=1;
		else ans[i][1]=1;
		ans[i][0]+=ans[i-1][0];
		ans[i][1]+=ans[i-1][1];
		p[i].num=i;
		p[i].s=-ans[i][0]+2*ans[i][1];
		//printf("%d\n",p[i].s);
		if(p[i].s>=0) q=max(q,i);
	}
	sort(p+1,p+n+1,cmp);
	for(int i=1;i<=n;i++)
	{
		//printf("%d\n",p[i].num);
		res=min(res,p[i].num);
		if(p[i].num-res>q) q=p[i].num-res;
	}
	int tot=0;
	for(int i=1;i<=n;i++)
		if(i+q-1<=n&&ans[i+q-1][0]-ans[i-1][0]<=2*(ans[i+q-1][1]-ans[i-1][1]))  tot++;
	if(!q) puts("No solution"); else printf("%d %d\n",q,tot);
}
