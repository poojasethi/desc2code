#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>

using namespace std;

char vowels[5] = {'a', 'e', 'i', 'o', 'u'};
int a[200005];
pair <int, int> p[200005];

bool vowel(char c)
{
    for (int i = 0; i < 5; i++)
        if (c == vowels[i] || char(c + 'a' - 'A') == vowels[i])
            return true;
    return false;
}

int main()
{
    string s;
    cin >> s;
    int n = s.length();
    for (int i = 1; i <= n; i++)
        a[i] = vowel(s[i-1]) ? (-1) : 2;
    for (int i = 1; i <= n; i++)
        p[i] = make_pair(p[i-1].first + a[i], i);
    sort(p, p+n+1);

    int Min = p[0].second;
    int ans = 0, rep = 1;

    for (int i = 1; i <= n; i++)
    {
        int now = p[i].second - Min;
        if (now > ans)
        {
            ans = now;
            rep = 1;
        }
        else if (now == ans)
            rep++;
        Min = min(Min, p[i].second);
    }
    if (ans == 0)
        cout << "No solution\n";
    else
        cout << ans << ' ' << rep << endl;
    return 0;
}
