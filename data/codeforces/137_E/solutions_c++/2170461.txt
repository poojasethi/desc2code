#include <cstdio>
#include <algorithm>
using namespace std;
char a[200001];
pair<int,int> p[200001];
int L;
int main(){
	scanf("%s",a);
	while(a[L]){
		if(a[L]=='a'||a[L]=='e'||a[L]=='i'||a[L]=='o'||a[L]=='u'||
		a[L]=='A'||a[L]=='E'||a[L]=='I'||a[L]=='O'||a[L]=='U')
			a[L]=-1;
		else
			a[L]=+2;
		L++;
	}
	for(int i=1;i<L+1;i++){
		p[i].first=p[i-1].first-a[i-1];
		p[i].second=-i;
	}
	sort(p,p+L+1);
	int maxI=0;
	int best=0;
	for(int i=0;i<L+1;i++){
		if(-p[i].second>maxI) maxI=-p[i].second;
		if(maxI+p[i].second>best) best=maxI+p[i].second;
	}
	int sum=0;
	int count=0;
	for(int i=0;i<best;i++) sum+=a[i];
	for(int i=0;i<L-best;i++){
		if(sum>=0) count++;
		sum+=a[i+best]-a[i];
	}
	if(sum>=0) count++;
	if(!best || !count){
		printf("No solution\n");
		return 0;
	}
	printf("%d %d\n",best,count);
}
