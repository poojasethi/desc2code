#include <cstdio>
#include <cstring>
#include <cctype>
#include <vector>
#include <algorithm>
using namespace std;

char s[200005]="0";
int n, L, cnt, c[200005];

bool vowel(char c) {
	c = tolower(c);
	return c=='a'||c=='e'||c=='i'||c=='o'||c=='u';
}

int main() {
	scanf("%s", s+1);
	n = strlen(s+1);
	for (int i = 1; i <= n; ++i)
		c[i] = c[i-1] + (vowel(s[i]) ? -1 : 2);
	vector<int> v(1, 0);
	for (int i = 1; i <= n; ++i) {
		int l = -1, r = v.size()-1;
		if (c[i] - c[v[r]] < 0) {
			v.push_back(i);
			continue;
		}
		while (l+1 < r) {
			int m = (l+r) / 2;
			c[i]-c[v[m]]>=0 ? r=m : l=m;
		}
		if (i-v[r] > L) L = i-v[r], cnt = 1;
		else if (i-v[r] == L) ++cnt;
	}
	if (!L) puts("No solution");
	else printf("%d %d\n", L, cnt);
}
