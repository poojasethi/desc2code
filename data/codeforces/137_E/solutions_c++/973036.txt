#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
#define SZ 1000000
#define shft 400100
#define INF 1000000000
#define pb(a) push_back(a)
int MP[SZ],val[SZ];
void init(){for(int i=0;i<SZ;++i)val[i]=MP[i]=INF;}
int get(int p){
  int ret=INF;
  for(;p;p-=(p&(-p)))ret=min(ret,MP[p]);
  return ret;
}
int add(int p,int w){
  for(;p<SZ;p+=(p&(-p)))MP[p]=min(w,MP[p]);
}
char V[200010]; int N;
vector<int> ps;
int main(){
  init();
  gets(V); N=strlen(V);
  int BL=0,BC=0;
  ps.pb(0);
  for(int i=0;i<N;++i){
   if(strchr("aeiouAEIOU",V[i]))ps.pb(ps.back()-1);
   else ps.pb(ps.back()+2); 
                      }
  for(int i=0;i<ps.size();++i){
   int op=get(ps[i]+shft);
   if(BL<i-op){BC=1; BL=i-op;}
   else if(BL==i-op)++BC;
   if(val[ps[i]+shft]>i)add(ps[i]+shft , i) , val[ps[i]+shft]=i;
                              }
  if(!BL)printf("No solution\n");
  else printf("%d %d\n",BL,BC);
  //system("pause");
  return 0;
}
