#include<stdio.h>
#include<string.h>
#include<ctype.h>
#define M 200005
int B[M],A[M];
char str[M];
int max(int a,int b){
	return (a>b?a:b);
}
int main(){
	int n,i,j,k;
	while(gets(str)){
		A[0]=B[0]=0;
		n=strlen(str);
		int Len=0;
		for(i=1;i<=n;i++){
			if(str[i-1]<='Z')
				str[i-1]+=32;
			int c=(strchr("aoeiu",str[i-1])!=NULL)?1:-2;
			A[i]=A[i-1]+c;
			B[i]=max(B[i-1],A[i]);
			int L=0,R=i-1;
			int res=-1;
			while(L<=R){
				int mid=(L+R)>>1;
				if(B[mid]>=A[i]){
					res=mid;
					R=mid-1;
				}else L=mid+1;
			}
			if(res==-1)continue;
			else if(i-res>=Len)
				Len=i-res;
		}
		if(Len==0){
			puts("No solution");
			continue;
		}
		int cnt=0;
		for(i=Len;i<=n;i++)
			if(A[i]<=A[i-Len])cnt++;
		printf("%d %d\n",Len,cnt);
	}
	return 0;
}
