#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
char a[200001],v[]="AEIOUaeiou";
pair<int,int> p[200001];
int L,maxI,best,sum,cnt;
int main(){
	scanf("%s",a);
	for(;a[L];L++) a[L]=strchr(v,a[L])?-1:2;
	for(int i=1;i<L+1;i++) p[i]=make_pair(p[i-1].first-a[i-1],-i);
	sort(p,p+L+1);
	for(int i=0;i<L+1;i++){
		if(-p[i].second>maxI) maxI=-p[i].second;
		if(maxI+p[i].second>best) best=maxI+p[i].second;
	}
	for(int i=0;i<best;i++) sum+=a[i];
	for(int i=0;i<L-best;i++){
		if(sum>=0) cnt++;
		sum+=a[i+best]-a[i];
	}
	if(sum>=0) cnt++;
	if(!best || !cnt) printf("No solution\n");
	else printf("%d %d\n",best,cnt);
}
