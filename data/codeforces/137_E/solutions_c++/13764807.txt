#include <bits/stdc++.h>
using namespace std;

const int maxN=2e5+100;
char s[maxN];
int a[maxN],b[maxN];

bool cmp(int i,int j) {
    if (3*a[i]-i*2!=3*a[j]-j*2) return 3*a[i]-i*2<3*a[j]-j*2;
    return i>j;
}

int main() {
    scanf("%s",s);
    int n=strlen(s);
    for(int i=0;i<n;i++) if (s[i]<='Z' && s[i]>='A') s[i]=s[i]-'A'+'a';
    for(int i=1;i<=n;i++) if (s[i-1]=='a' || s[i-1]=='e' || s[i-1]=='i' || s[i-1]=='o' || s[i-1]=='u') 
        a[i]=a[i-1]+1; else a[i]=a[i-1];
    for(int i=0;i<=n;i++) b[i]=i;
    sort(b,b+n+1,cmp);
//    for(int i=0;i<=n;i++) printf("%i ",3*a[b[i]]-b[i]*2);
//    cout<<"\n";    
//    for(int i=0;i<=n;i++) printf("%i ",b[i]);
//    cout<<"\n";
    int res=-1, cnt=0, vt=b[0];
    for(int i=1;i<=n;i++) {
        if (vt>b[i]) res=max(vt-b[i],res);
        vt=max(vt,b[i]);
//        cout<<vt;
    }
    if (res!=-1) 
        for(int i=res;i<=n;i++) if (a[i]-a[i-res]<=2*(res-a[i]+a[i-res])) ++cnt;
    if (res==-1) cout<<"No solution"; else cout<<res<<" "<<cnt;
}
