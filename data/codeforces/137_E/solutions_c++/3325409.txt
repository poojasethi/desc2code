#include <iostream>
#include <string>
#include <cstring>
#include <algorithm>
#include <vector>

#define rep(i,n) for(int i=0;i<n;i++)
#define pi       pair<int,int>

using namespace std;

int n,a[200010],sum[200010],mx[200010],res,cnt;
string str;
vector< pi > vp;

int main(){
  cin>>str; n=str.size();
  rep(i,n){
    int c=tolower(str[i]);
    a[i] = ((c=='a' || c=='e' || c=='i' || c=='o' || c=='u')?-1:2);
  }
  sum[0] = a[0];
  rep(i,n)sum[i+1] = sum[i]+a[i+1];
  rep(i,n)vp.push_back(pi(sum[i],i));
  vp.push_back(pi(0,-1));
  sort(vp.begin(),vp.end());

  mx[n] = vp[n].second;

  for(int i=n-1;i>=0;i--)mx[i] = max(mx[i+1],vp[i].second);
  rep(i,n+1){
    if(res<mx[i]-vp[i].second){
      res = mx[i]-vp[i].second;
      cnt = 0;
    }
    if(res==mx[i]-vp[i].second)cnt++;
  }
  if(res==0)cout<<"No solution"<<endl;
  else cout<<res<<" "<<cnt<<endl;


}
