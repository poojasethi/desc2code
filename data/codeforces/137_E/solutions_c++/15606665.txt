#include<bits/stdc++.h>
#define ll long long
#define pii pair<ll,ll>
#define X first
#define Y second
using namespace std;
pii p[200200];
ll t[200200];
ll f(char c)
{
    if(c=='a' || c=='e' || c=='i' || c=='o' || c=='u' || c=='A' || c=='E' || c=='I' || c=='O' || c=='U')
        return -1;
    return 2;
}
int main()
{
    string s;
    cin>>s;
    ll n=s.size(),ans1=0,ans2=0;
    p[0].X=0;
    for(ll i=1;i<=n;i++)
        p[i].X=p[i-1].X+f(s[i-1]),p[i].Y=i;
    sort(p,p+n+1);
    t[n]=p[n].Y;
    for(ll i=n-1;i>=0;i--)
    {
        t[i]=max(p[i].Y,t[i+1]);
        if(t[i]-p[i].Y>ans1)
            ans2=1,ans1=t[i]-p[i].Y;
        else if(t[i]-p[i].Y==ans1)
            ans2++;
    }
    if(ans1==0)
        cout<<"No solution"<<endl;
    else
        cout<<ans1<<" "<<ans2<<endl;
    return 0;
}
