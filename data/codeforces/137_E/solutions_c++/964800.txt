#include<iostream>
#include<string>
#include<ctype.h>
#include<utility>
#include<algorithm>
using namespace std;

string s;
pair<int,int> dp[200005];
int ANS=0,nANS=0;

int main()
{
     cin>>s;
     int n=s.length(),c=0,v=0;
     for (int i=0;i<n;i++)
     {
         s[i]=tolower(s[i]);
         if ( (s[i]=='a') || (s[i]=='e') || (s[i]=='i') || (s[i]=='o') || (s[i]=='u') ) v++;
         else c++;
         dp[i+1].first=2*c-v;
         dp[i+1].second=i+1;
     }
     dp[0].first=dp[0].second=0;
     sort(dp,dp+n+1);
     int MIN=dp[0].second;
     for (int i=1;i<=n;i++)
     {
         if (ANS<dp[i].second-MIN)
         {
             ANS=dp[i].second-MIN;
             nANS=0;
         }
         if (ANS==dp[i].second-MIN) nANS++;

         if (dp[i].second<MIN) MIN=dp[i].second;
     }
     if (ANS==0) cout<<"No solution"<<endl;
     else cout<<ANS<<" "<<nANS<<endl;
}
