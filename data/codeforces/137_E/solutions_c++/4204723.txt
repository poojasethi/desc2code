#include <stdio.h>
bool V[128];
char str[200001]="aeiou";
int n,v[200001];
int main()
{
    for(int i=0;str[i];++i)
        V[str[i]]=V[str[i]-32]=true;
    gets(str);
    for(n=0;str[n];++n)
        v[n+1]=v[n]+V[str[n]];
    int best=0,o,z;
    for(int i=1;i<=n;++i)
        for(int size=best+1;i+size-1<=n;++size){
            o=v[i+size-1]-v[i-1];
            z=size-o;
            if(o<=2*z)
                best=size;
            else
                size=o+(o+1)/2-1;
        }
    if(best==0){
        puts("No solution");
        return 0;
    }
    o=0;
    for(int i=1;i<=n-best+1;++i)
        o+=(v[i+best-1]-v[i-1])<=2*(best-(v[i+best-1]-v[i-1]));
    printf("%d %d",best,o);
    return 0;
}
