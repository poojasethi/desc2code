#include<cstdio>
#include<cstring>
char s[204800],v[]="AEIOUaeiou",*p;
int l,c,n,x[204800],L,R,M,i;
int main(){
	gets(s);
	for(l=0;s[l];++l)
		if(s[l]=!strchr(v,s[l]))x[n++]=l;
	if(n){
		if(n*3>=l)L=n;
		else{
			for(M=n;!L&&M;--M)
			for(i=0;i+M<=n;++i)
				if((x[i+M-1]-x[i]+1)<=M*3){L=M;break;}
		}
		M=L;L*=3;if(L>l)L=l;R=0;
		for(i=0;i+L<=l;++i){
			if(x[R]<i)R++;
			if(R+M-1==n)break;
			if(x[R]>=i+L)continue;
			if(x[R+M-1]<i+L)++c;
		}
	}
	L?printf("%d %d\n",L,c):puts("No solution"); 
	return 0;
}
