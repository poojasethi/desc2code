#include <iostream>
#include <cstring>
using namespace std;
int main() {
int vals[256];
int cumSum;
int *earliest = new int[200001];
int posn;
char sets[] = { "AEIOUbcdfghjklmnpqrstvwxyz" };
char c;
static int maxLength;
static int count;
  memset(earliest,0,200001*sizeof(int));
  memset(vals,0,sizeof(vals));
  for (char* p=sets; *p; ++p) vals[toupper(*p)]=vals[tolower(*p)]=toupper(*p)==*p?1:-2;
  for ( (cin>>c),posn=cumSum=maxLength=0; cin.good() && vals[c]; cin>>c ) {
  int L = ++posn;
    cumSum+=vals[c];
    if (cumSum>0) L -= earliest[cumSum] ? earliest[cumSum] : (earliest[cumSum]=posn);
    if ( L<maxLength ) continue;
    if ( L>maxLength ) { maxLength=L; count=0; }
    if ( maxLength) ++count;
  }
  if ( maxLength ) cout<<maxLength<<" "<<count<<endl;
  else cout<<"No solution"<<endl;
  return 0;
}
