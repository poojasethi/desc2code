#include <stdio.h>
#include <map>
#include <string.h>
using namespace std;
char s[200005];
int a[200005],record[200005];

map <int,int> mm;

int main()
{
    int k,v,c,i,j,mx,cnt;
    scanf("%s",s);
    k=strlen(s);
    v=c=0;
    for(i=0;i<k;i++)
    {
        if(int (s[i])<int ('a'))
                s[i]+=int ('a')-int ('A');
        if(s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u')
            v++;
        else
            c++;
        a[i]=v-2*c;
    }

    for(i=0;i<k;i++)
    {
        if(a[i]<=0)
            record[i]=i+1;
        else
        {
            if(mm[a[i]]==0)
                mm[a[i]]=i+1;
            else
                record[i]=i+1-mm[a[i]];

        }
    }
    mx=1;
    cnt=0;
    for(i=0;i<k;i++)
    {
       if(record[i]==mx)
            cnt++;
       if(record[i]>mx)
       {
           mx=record[i];
           cnt=1;
       }

    }
    if(cnt==0)
        printf("No solution\n");
    else
        printf("%d %d\n",mx,cnt);

}
