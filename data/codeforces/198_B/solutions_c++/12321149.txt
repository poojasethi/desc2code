#include<iostream>
#include<queue>
#define ii pair<int,int>
using namespace std;
char a[100000][2];
int d[100000][2];
int main ()
{
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++) cin>>a[i][0];
	for(int i=0;i<n;i++) cin>>a[i][1];
	queue <ii> que;
	if(a[0][0]=='-') que.push(ii(0,0));
	d[0][0]=0;
	while(!que.empty())
	{
		ii temp=que.front();
		que.pop();
		int x=temp.first;int lr=temp.second;
		if(x+1>=n||x-1>=n||x+k>=n)
		{
			cout<<"YES";
			return 0;
		}
		if(a[x+1][lr]=='-'&&!d[x+1][lr]) que.push(ii(x+1,lr)),d[x+1][lr]=d[x][lr]+1;
		if(a[x-1][lr]=='-'&&!d[x-1][lr]&&!(x==1&&lr==0)&&d[x][lr]+1<=x-1) 
			que.push(ii(x-1,lr)),d[x-1][lr]=d[x][lr]+1;
		if(a[x+k][!lr]=='-'&&!d[x+k][!lr]) que.push(ii(x+k,!lr)),d[x+k][!lr]=d[x][lr]+1;
	}
	cout<<"NO";
	return 0;
}
