#include <cstdio>
#include <iostream>
using namespace std;
int a[2][100005],p[200005],q[200005];
char c[2][100005];
int main ( ) {
	int n,k,i,f,r,x,y;
	scanf("%d%d",&n,&k);
	scanf("%s",c[0]);scanf("%s",c[1]);
	a[0][0]=1;
	f=0;r=1;
	while (f<r) {
		x=p[f];y=q[f];f++;
		if (y>=n-k) {printf("YES\n");return 0;}
		if (a[x][y-1]==0 && y>a[x][y] && c[x][y-1]=='-' && y>0) {
			a[x][y-1]=a[x][y]+1;
			p[r]=x;q[r]=y-1;r++;
		}
		if (a[x][y+1]==0 && c[x][y+1]=='-') {
			a[x][y+1]=a[x][y]+1;
			p[r]=x;q[r]=y+1;r++;
		}
		if (a[1-x][y+k]==0 && c[1-x][y+k]=='-') {
			a[1-x][y+k]=a[x][y]+1;
			p[r]=1-x;q[r]=y+k;r++;
		}
	}
	printf("NO\n");
	return 0;
}
