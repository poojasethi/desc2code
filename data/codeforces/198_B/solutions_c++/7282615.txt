#include<iostream>
#include<cstdlib>
#include<algorithm>
#include<cassert>
using namespace std;
char s[2][1000000];
int n,i,qx[1000000],qy[1000000],d[2][1000000],k,qst,qen;
void add(int x,int y,int D)
{
    if(y>n)
    {
        cout<<"YES";
        exit(0);
    }
    if(y<0 || d[x][y] || s[x][y]=='X')
    {
        return;
    }
    d[x][y]=D;
    if(D<=y)
    {
        qx[qen]=x;qy[qen++]=y;
    }
}
int main()
{
    cin>>n>>k;
    for(i=0;i<n;i++)cin>>s[0][i];
    for(i=0;i<n;i++)cin>>s[1][i];
    add(0,0,0);
    while(qst<qen)
    {
        int x=qx[qst];int y=qy[qst++];int D=d[x][y]+1;
        add(x^1,y+k,D);
        add(x,y+1,D);
        add(x,y-1,D);
    }
    cout<<"NO";
    return 0;
}
