#include <iostream>
#include <cstring>
using namespace std;
int main(void){
  int n,k,t[100100][2],q[100100*2]={0},a=0,b=1,y=0;
  cin>>n>>k;
  char s[2][100100];
  cin>>s[0]>>s[1];
  memset(t,0x3f,sizeof(t)),t[0][0]=0;
  while(!y&&a<b){
    int p=q[a++],w=p&1;p>>=1;
    int x=t[p][w]+1;
    if(p+k>=n)y=1;
    else{
      if('-'==s[w][p+1]&&t[p+1][w]>x)t[p+1][w]=x,q[b++]=(p+1<<1)+w;
      if('-'==s[1-w][p+k]&&t[p+k][1-w]>x)t[p+k][1-w]=x,q[b++]=(p+k<<1)+1-w;
      if(p>x&&'-'==s[w][p-1]&&t[p-1][w]>x)t[p-1][w]=x,q[b++]=(p-1<<1)+w;
    }
  }
  cout<<(y?"YES":"NO")<<endl;
  return 0;
}
