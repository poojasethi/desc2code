#include <bits/stdc++.h>
using namespace std ;

string s[3]  ;
int ck[3][100005] ;
int n, k , yo ;

void go( int x, int y, int wat )
{
    //cout << x << " "<< y << " " << wat << endl ;
    if(yo) return ;
    if( x < 0 )return ;
    if( x <= wat )return ;
    if( x >= n )
    {
        yo = 1 ;
        return ;
    }
    if( s[y][x] == 'X' )return ;
    if( ck[y][x]  )return ;
    //cout << x << " "<< y <<endl ;
    ck[y][x] = 1 ;
    go( x+k , ( (y+1)%2 ) , wat+1 ) ;
    go( x + 1 ,y , wat+1 ) ;
    go( x - 1 , y , wat + 1 ) ;

}

int main()
{
   cin>>n >> k ;
   cin>>s[0] >> s[1] ;

   go(0,0, -1 ) ;

   if(yo)cout <<"YES" ;
   else cout <<"NO" ;


return 0 ;
}
