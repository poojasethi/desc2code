#include <stdio.h>
#include <string.h>

int n, m, l, r;
char a[2][200000];
int d[200000], e[200000], f[2][200000], ans, i, j;

void updata(int p, int i, int j)
{
    if (j>n) ans=true;
    else if (j<1) return;
    else if (a[i][j-1]=='-' && f[i][j]==-1)
    {
        f[i][j]=p;
        if (f[i][j]<=j)
        {
            r++; d[r]=i; e[r]=j;
        }
    }
}

int main()
{
    scanf("%d%d%s%s", &n, &m, &a[0], &a[1]);
    memset(f, 255, sizeof(f));
    l=0; r=1; d[1]=0; e[1]=1; f[0][1]=1;
    while (l<r)
    {
        l++; i=d[l]; j=e[l];
        updata(f[i][j]+1, i, j+1);
        updata(f[i][j]+1, i, j-1);
        updata(f[i][j]+1, i^1, j+m);
    }
    if (ans) printf("YES\n");
    else printf("NO\n");
    return 0;
}
