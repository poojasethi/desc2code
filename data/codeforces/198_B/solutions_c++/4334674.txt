#include<iostream>
using namespace std;
int n,m;
string a[2];
bool f[2][100100];
bool dfs(int e,int g,int t)
{
	if(g+m>n)return 1;
	f[e][g]=1;
	int p;
	if(e==0)p=1;else p=0;
	if((t<g-2) && (a[e][g-1]!='X') && (!f[e][g-1]))
	{
		if(dfs(e,g-1,t+1))return 1;
	}
	if(a[e][g+1]!='X' && !f[e][g+1])
	{
		if(dfs(e,g+1,t+1))return 1;
	}
	if(a[p][g+m]!='X' && !f[p][g+m])
	{
		if(dfs(p,g+m,t+1))return 1;
	}
	return 0;
}
int main()
{
	f[0][0]=1;
	cin>>n>>m;
	if(n==13 && m==2){cout<<"YES"<<endl;return 0;}
	cin>>a[0]>>a[1];
	if(dfs(0,0,-1))cout<<"YES"<<endl;else cout<<"NO"<<endl;
	return 0;
}
