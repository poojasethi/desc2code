int n, k;
int visited[2][100001];
string walls[2];
char a, b;


bool dfs(int x, int y, int t){
    if(y >= n) return true;
    if(y < 0) return false;
    if(walls[x][y] == 'X' || (visited[x][y] <= t && visited[x][y] != 0) || y < t) return false;
    
    visited[x][y] = t;
    if(dfs(1-x, y+k, t+1)) return true;
    if(dfs(x, y+1, t+1)) return true;
    if(dfs(x, y-1, t+1)) return true;
    
    return false;
}

int main()
{
    cin>>n>>k;
    cin>>walls[0]>>walls[1];
    if(dfs(0, 0, 0))
        cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
}
