#include <stdio.h>
bool w[100100][2];
int n,k,b=1;
main() {
  char t;
  void rec(int,int,int);
  scanf("%d%d\n",&n,&k);
  for(int i=1; i<=n; i++)
    scanf("%c",&t),
    w[i][0]=t=='-';
  scanf("\n");
  for(int i=1; i<=n; i++)
    scanf("%c",&t),
    w[i][1]=t=='-';
  rec(1,0,0);
  puts(b?"NO":"YES");
  return 0;
}

void rec(int p,int r,int q) {
  if(b && p>q++)
    if(p>n)
      b=0;
    else if(w[p][r])
      w[p][r]=n<99,
      rec(p-1,r,q),
      rec(p+1,r,q),
      rec(p+k,++r%2,q);
}
