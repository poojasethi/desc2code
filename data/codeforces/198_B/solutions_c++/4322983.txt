#include <map>

using namespace std;
queue< int > q;
bool u[2][100005];
int n,k;
string s[2];
bool f;
inline void ex(int a,int b,int c)
{
	if(b>n)
		f=true;
	if(!u[a][b] && s[a][b-1]=='-' && b>c)
		q.push(a),q.push(b),q.push(c),u[a][b]=true;
}
int main()
{
//	ifstream fin("input.txt");
//	ofstream fout("output.txt");
	cin>>n>>k;
	cin>>s[0]>>s[1];
	q.push(0),q.push(1),q.push(0);
	u[0][1]=true;
	int x,y,wl;
	while(!q.empty())
	{
		x=q.front();
		q.pop();
		y=q.front();
		q.pop();
		wl=q.front();
		q.pop();
		ex(x,y-1,wl+1);
		ex(x,y+1,wl+1);
		ex(1-x,y+k,wl+1);
		if(f)
		{
			cout<<"YES"<<endl;
			return 0;
		}
	}
	cout<<"NO"<<endl;
	return 0;
}
