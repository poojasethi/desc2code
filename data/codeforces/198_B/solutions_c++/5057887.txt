#include<iostream>
using namespace std;
int f[2][100005],sx[100005],sy[100005];
char s[2][100005];
int fi,fr;
int main()
{
	int n,k;
	cin>>n>>k;
	cin>>s[0];
	cin>>s[1];
	f[0][0]=1;
	for(fr=1;fi<fr;)
	{
		int x=sx[fi];
		int y=sy[fi++];
		if (y>=n-k)
		{
			cout<<"YES";
			return 0;
		}
		if (y>0 && s[x][y-1]=='-' && f[x][y-1]==0 && y>f[x][y])
		{
			f[x][y-1]=f[x][y]+1;
			sx[fr]=x;
			sy[fr++]=y-1;
		}
		if (s[x][y+1]=='-' && f[x][y+1]==0)
		{
			f[x][y+1]=f[x][y]+1;
			sx[fr]=x;
			sy[fr++]=y+1;
		}
		if (s[1-x][y+k]=='-' && f[1-x][y+k]==0)
		{
			f[1-x][y+k]=f[x][y]+1;
			sx[fr]=1-x;
			sy[fr++]=y+k;
		}
	}
	cout<<"NO";
	return 0;
}
