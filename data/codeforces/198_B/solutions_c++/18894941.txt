#include<stdio.h>
const int maxn=1e5+1;
char map[2][maxn];
int check[2][maxn];
int n,k,level,cnt;
int dfs(int i,int j){
	if(j>n)return 1;
	if(map[i][j]=='X'||j<cnt||check[i][j])
	return 0;
	check[i][j]=1;
	cnt++;
	int t=dfs(i,j-1)||dfs(1-i,j+k)||dfs(i,j+1);
	cnt--;
	return t;
}
int main(){
	scanf("%d %d",&n,&k);
	getchar();
	for(int i=0;i<2;i++){
		scanf("%s",map[i]); 
		getchar();
	}
	n--;
	if(dfs(0,0)){
		printf("YES\n");
	}else printf("NO\n");
	return 0;
}
