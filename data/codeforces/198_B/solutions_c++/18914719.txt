#include<cstdio>
int n,k,level=0;
char map[2][100000];
bool used[2][100000];
bool dfs(int i,int j)
{
	if(j>n)
	return true;
	if(map[i][j]=='X'||j<level||used[i][j])
	return false;
	used[i][j]=true;
	level++;
	bool f=dfs(i,j-1)||dfs(1-i,j+k)||dfs(i,j+1);
	level--;
	return f;
}
int main()
{
	scanf("%d %d %s %s",&n,&k,map[0],map[1]);
	n--;
	printf(dfs(0,0)?"YES":"NO");
	return 0;
}
