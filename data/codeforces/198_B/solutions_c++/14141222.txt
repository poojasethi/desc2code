#include <iostream>
#include <cstdlib>
using namespace std;
const int ma=1e5+3;
int n,k,d[2][ma],qf,qe,qx[ma],qy[ma];
char s[2][ma];
inline void add(int x,int y,int D){
	if (y>n){cout<<"YES";exit(0);}
	if (s[x][y]=='X'||d[x][y]||y<0) return;
	d[x][y]=D;
	if (D<=y){
		qx[qe]=x;
		qy[qe++]=y;
	}
}
int main(){
	cin>>n>>k;
	cin>>s[0];
	cin>>s[1];
	add(0,0,0);
	while(qf<qe){
		int x=qx[qf],y=qy[qf++],D=d[x][y]+1;
		add(x^1,y+k,D);
		add(x,y+1,D);
		add(x,y-1,D);
	}
	cout<<"NO";
	return 0;
}
