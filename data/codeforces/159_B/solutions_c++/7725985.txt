#include<cstdio>
using namespace std;
int a1[1001][1001],a2[1001];
int main()
{
	int n,m,a,b,o1=0,o2=0;
	scanf("%d%d",&n,&m);
	while(n--)
	{
		scanf("%d%d",&a,&b);
		++a1[b][a],++a2[b];
	}
	while(m--)
	{
		scanf("%d%d",&a,&b);
		if(a2[b])--a2[b],++o1;
		if(a1[b][a])--a1[b][a],++o2;
	}
	printf("%d %d\n",o1,o2);
}
