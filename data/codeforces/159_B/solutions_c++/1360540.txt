#include<cstdio>
int n,m;
int g[1005][1005],p[1005],a1,a2;
int main()
{
    scanf("%d%d",&n,&m);
    int i,a,b;
    for(i=0;i<n;i++)
    {
        scanf("%d%d",&a,&b);
        p[b]++;
        g[a][b]++;
    }
    for(i=0;i<m;i++)
    {
        scanf("%d%d",&a,&b);
        if(p[b])
        {
            p[b]--;
            a1++;
        }
        if(g[a][b])
        {
            g[a][b]--;
            a2++;
        }
    }
    printf("%d %d",a1,a2);
}
