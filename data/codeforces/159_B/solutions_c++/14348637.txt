#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
int k,i,l,n,r,z,m,x,y,a[1001][1001];
vector < int > g[1001],q[1001]; 
main()
{
 cin>>n>>m;
 for(i=1;i<=n;i++){
  cin>>y>>x;
  a[x][y]++;
  g[x].push_back(y);
 }
 for(i=1;i<=m;i++){
  cin>>y>>x; q[x].push_back(y);
  if(a[x][y]) { k++; a[x][y]--; }
 }
 for(i=1;i<=1001;i++)
  z+=min(q[i].size(),g[i].size());
 cout<<z<<" "<<k<<endl;
}
