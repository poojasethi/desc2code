#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
using namespace std;
#define MAX 1002

int arr[MAX], mat[MAX][MAX], u, v;
int main() {
	int n, m, x, y;

	cin >> n >> m;

	for (int i = 0; i < n; ++i) {
		cin >> x >> y;
		arr[y]++;
		mat[x][y]++;
	}
	for (int i = 0; i < m; ++i) {
		cin >> x >> y;
		if (arr[y] > 0) {
			u++; arr[y]--;
		}
		if (mat[x][y] > 0) {
			v++; mat[x][y]--;
		}
	}

	cout << u << " " << v << endl;

}
