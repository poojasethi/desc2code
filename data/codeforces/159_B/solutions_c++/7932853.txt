#include <bits/stdc++.h>

using namespace std;

const int N=2000;
int n,m,a,b,x,y,pp[N][N],p[N],q[N],qq[N][N],k,i,v,u;

int main()
{
 cin>>n>>m;
 for ( i=0;i<n;i++ )
  {
   cin>>a>>b;
   qq[a][b]++;
   q[b]++;
  }
 for ( k=0;k<m;k++ )
  {
   cin>>x>>y;
   pp[x][y]++;
   p[y]++;
  }

 for ( i=1;i<=1000;i++ )
  for ( k=1;k<=1000;k++ )
   v+=min(qq[i][k],pp[i][k]);
 
 for ( k=1;k<=1000;k++ )
  u+=min(q[k],p[k]);
  
 cout<<u<<" "<<v<<endl;
 return 0;
}
