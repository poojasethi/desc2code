#include<iostream>
using namespace std;

const int oo=1010;
int n,m,x,y,a[oo][oo],b[oo],c[oo][oo],d[oo],C,BC;

int main(){
  cin>>n>>m;
  for(int i=0;i<n;i++){
    cin>>x>>y;
    a[x][y]++;
    b[y]++;
  }
  for(int i=0;i<m;i++){
    cin>>x>>y;
    c[x][y]++;
    d[y]++;
  }
  for(int i=0;i<oo;i++){
    C+=min(b[i],d[i]);
    for(int j=0;j<oo;j++)
      BC+=min(a[i][j],c[i][j]);
  }
  cout<<C<<' '<<BC;
  return 0;
}
