#include <cstdio>
int N, M;
int mat[1001][1001];
int di[1001];

int main() {
scanf("%d%d",&N,&M);
for(int i =0;i<N;++i) {
    int a, b;
    scanf("%d%d", &a,&b);
    mat[a][b]++;
    ++di[b];
}
int ans =0,a2=0;
for(int i =0;i<M;++i) {
    int a, b;
    scanf("%d%d", &a,&b);
    if(mat[a][b]) --mat[a][b], ++a2; 
    if(di[b]) --di[b], ++ans; 
}
printf("%d %d\n", ans, a2);
}
