#include <iostream>
using namespace std;

int n,m,a[1111][1111],u,v,x,y,len[1111],jb[1111];

int main(void)
{
//	freopen("159B.in", "r", stdin);

	cin >> n >> m;
	for (int i=1; i<=n; i++)
	{
		cin >> x >> y;
		a[x][y]++;
		len[y]++;
	}
	for (int i=1; i<=m; i++)
	{
		cin >> x >> y;
		if (a[x][y])
		{
			a[x][y]--;
			len[y]--;
			u++; v++;
		}
		else jb[y]++;
	}
	for (int i=1; i<=1000; i++) u += min(len[i], jb[i]);
	
	cout << u << ' ' << v << "\n";
	return 0;
}
