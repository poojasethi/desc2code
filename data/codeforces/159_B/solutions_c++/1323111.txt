#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;

int s1[1100]={0}, s2[1100][1100]={0};
int main(){
	int n, m, a, b;
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++){
		scanf("%d%d", &a, &b);
		s1[b]++;
		s2[a][b]++;
	}
	int x = 0, y = 0;
	for(int i=0;i<m;i++){
		scanf("%d%d", &a, &b);
		if(s1[b] > 0){ x++; s1[b]--; }
		if(s2[a][b] > 0){ y++; s2[a][b]--; }
	}
	printf("%d %d\n", x, y);
	return 0;
}
