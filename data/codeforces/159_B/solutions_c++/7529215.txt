#include<iostream>
using namespace std;
int main()
{
int n,m,i,j,a,b,x,y,ca[1001]={0},cb[1001][1001]={0},cx[1001]={0},cy[1001][1001]={0},u=0,v=0;
cin>>n>>m;
for(i=0; i<n; i++)
{
cin>>b>>a;
ca[a]++;
cb[a][b]++;
}
for(i=0; i<m; i++)
{
cin>>y>>x;
cx[x]++;
cy[x][y]++;
}
for(i=1; i<=1000; i++)
if(ca[i] && cx[i])
{
x=(ca[i]>cx[i])?cx[i]:ca[i];
u+=x;
for(j=1; j<=1000 && x; j++)
{
y=(cb[i][j]<cy[i][j])?cb[i][j]:cy[i][j];
x-=y;
v+=y;
}
}
cout<<u<<' '<<v;
return 0;
}
