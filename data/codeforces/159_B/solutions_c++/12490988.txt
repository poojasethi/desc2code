#include<bits/stdc++.h>
using namespace std;

int dia[1001],marker[1001][1001],caps[1001][1001];
int n,m,ans,beauty,a,b;

int main(){
	cin >> n >> m;
	for(int i=1;i<=n;i++){
		cin >> a >> b;
		marker[a][b]++;
		dia[b]++;
	}
	for(int i=1;i<=m;i++){
		cin >> a >> b;
		if(marker[a][b]){
			beauty++;
			marker[a][b]--;
		}
		if(dia[b]){
			ans++;
			dia[b]--;
		}
	}
	cout << ans <<' '<< beauty <<endl;
}
