#include<iostream>
using namespace std;
int ax[100001],ay[100001],a[1001][1001]={0};
int main()
{
  int n,m,x,y,r[1001]={0},k=0,s=0,i,l=0;
  cin>>n>>m;
  for(i=0;i<n;i++)
  {
    cin>>x>>y; a[x][y]++; r[y]++;
  }
  for(i=0;i<m;i++)
  {
    cin>>x>>y;
    if (a[x][y]){a[x][y]--; k++; r[y]--;}
    else
    {
      ax[l]=x; ay[l++]=y;
    }
  }
  for(i=0;i<l;i++)
  {
    if (r[ay[i]]) {s++; r[ay[i]]--;}
  }
  cout<<s+k<<" "<<k<<endl;
//  system("pause");
  return 0;
}
