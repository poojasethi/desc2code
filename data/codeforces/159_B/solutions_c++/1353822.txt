#include <iostream>
using namespace std;

int a[1001][1001], b[1001];
int match = 0;
int beautiful_match = 0;
int n, m;
int main() {
     cin>>n>>m;
     int l, r;
     for (int i = 0; i < n; i++) {
	  cin>>l>>r;
	  a[l][r]++;
	  b[r]++;
     }
     for (int i = 0; i < m; i++) {
	  cin>>l>>r;
	  if (b[r] > 0) {
	       b[r]--;
	       match++;
	  }
	  if (a[l][r] > 0) {
	       a[l][r]--;
	       beautiful_match++;
	  }
     }

     cout<<match<<' '<<beautiful_match<<endl;
     return 0;
}
