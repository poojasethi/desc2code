
#include <cstdio>

int main()
{   
	int s1[100005]={0},s2[1005][1005]={0};
    int n,m,a,b;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%d%d",&a,&b);
		s1[b]++;
		s2[a][b]++;
	}
	int u=0,v=0;
	for(int i=0;i<m;i++){
		scanf("%d%d",&a,&b);
		if(s1[b]>0){ 
			u++;
			s1[b]--;
		}
		if(s2[a][b]>0){
			v++;
			s2[a][b]--;
		}
	}
	printf("%d %d\n",u,v);
	return 0;
}
