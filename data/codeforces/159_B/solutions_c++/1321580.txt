#include<stdio.h>
int stat1[1001],stat2[1001][1001];
int main()
{
	int n,m,a,b,i,ans1=0,ans2=0;
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&m);
	for(i=0;i<n;++i)
	{
		scanf("%d%d",&a,&b);
		stat1[b]++;
		stat2[b][a]++;
	}
	for(i=0;i<m;++i)
	{
		scanf("%d%d",&a,&b);
		if(stat1[b])
			stat1[b]--,ans1++;
		if(stat2[b][a])
			stat2[b][a]--,ans2++;
	}
	printf("%d %d\n",ans1,ans2);
	return 0;
}
