#include <iostream>

using namespace std;

int a[1000][1000];
int b[1000][1000];

int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		int x,y;
		cin>>x>>y;
		x--;y--;
		a[y][x]++;
	}
	for(int j=0;j<m;j++)
	{
		int x,y;
		cin>>x>>y;
		x--;y--;
		b[y][x]++;
	}
	int ans1=0,ans2=0;
	for(int i=0;i<1000;i++)
	{
		int tota=0;
		int totb=0;
		for(int j=0;j<1000;j++)
		{
			ans2+=min(a[i][j],b[i][j]);
			tota+=a[i][j];
			totb+=b[i][j];
		}
		ans1+=min(tota,totb);
	}
	cout<<ans1<<' '<<ans2;
}
