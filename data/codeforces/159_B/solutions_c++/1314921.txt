#include <cstdio>
using namespace std;

int rs[1024];
int ms[1024][1024];

int main()
{
    int N, M;
    scanf("%d%d", &N, &M);

    int ans1 = 0, ans2 = 0;
    while(N--)
    {
    int x, y;
    scanf("%d%d", &x, &y);
    rs[y]++;
    ms[x][y]++;
    }
    while(M--)
    {
    int x, y;
    scanf("%d%d", &x, &y);
    if(rs[y]-- > 0)
        ans1++;
    if(ms[x][y]-- > 0)
        ans2++;
    }
    printf("%d %d\n", ans1, ans2);

    return 0;
}
