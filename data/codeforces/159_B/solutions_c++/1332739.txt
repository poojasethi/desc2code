# include <stdio.h>

int n,m,t[1000][1000],b[1000],d[1000][2],g,l,x1=0,x2=0;

int main()
{
	scanf("%d %d",&n,&m);
	
	for(int h=0; h<n; h++)
	{
		scanf("%d %d",&g,&l);
		t[g-1][l-1]++;
		b[l-1]++;
	}
	
	for(int h=0; h<m; h++)
	{
		scanf("%d %d",&g,&l);
		
		if(b[l-1]>0)
		{
			x1++;
			b[l-1]--;
		}
		
		if(t[g-1][l-1]>0)
		{
			t[g-1][l-1]--;
			x2++;
		}
		
	}
	
	printf("%d %d",x1,x2);
	
getchar();
}
