#include<bits/stdc++.h>
using namespace std;
int s[1010][1010],cnt[1010]; 
int main()
{
	int n,m,i,tot=0,bea=0,x,y;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		cnt[y]++;
		s[x][y]++;
	}
	for(i=0;i<m;i++)
	{
		scanf("%d%d",&x,&y);
		if(s[x][y])
		{
			s[x][y]--;
			bea++;
		}
		if(cnt[y])
		{
			cnt[y]--;
			tot++;
		}
	}
	cout<<tot<<' '<<bea<<endl;
	return 0;
}
