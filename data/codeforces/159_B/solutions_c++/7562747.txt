#include <iostream>

using namespace std;

int cap[1024][1024], u,v, T[1024];

int main(void){
	int n,m;
	cin>>n>>m;
	for(int i=0 ; i<n ; i++){
		int a,b;
		cin>>a>>b;
		cap[a][b]++;
		T[b]++;
	}
	for(int i=0 ; i<m ; i++){
		int a,b;
		cin>>a>>b;
		if(cap[a][b]){
			cap[a][b]--; v++;
		}
		if(T[b]){
			T[b]--; u++;
		}
	}
	cout << u << " " << v << endl;
}
