#include <iostream>

using namespace std;
int n,m,i,j,f,t,a[1001][1001],b[1001],c[1001];
int main()
{
    cin>>n>>m;
    for(;n;n--)
    {
        cin>>i>>j;
        a[i][j]++;
        b[j]++;
    }
    for(;m;m--)
    {
        cin>>i>>j;
        if(a[i][j]){b[j]--;a[i][j]--;f++;}
        else c[j]++;
    }
    for(i=1,t=f;i<=1000;i++)t+=c[i]<b[i]?c[i]:b[i];
    cout<<t<<' '<<f;
    return 0;
}
