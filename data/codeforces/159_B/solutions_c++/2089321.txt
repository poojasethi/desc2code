#include<iostream>
#include<cstring>
using namespace std;
int b[1050],a[1050][1050];
int main(){
	int n,m;
	while(cin>>n>>m){
		int x,y;
		int ans1=0,ans2=0;
		memset(a,0,sizeof(a));
		memset(b,0,sizeof(b));
		for(int i=0;i<n;++i){
			cin>>x>>y;a[x][y]++;b[y]++;
		}
		for(int i=0;i<m;++i){
			cin>>x>>y;
			if(a[x][y]){ans2++;a[x][y]--;}
			 if(b[y]){ans1++;b[y]--;}
		}
		cout<<ans1<<" "<<ans2<<endl;
	}
	return 0;
}
