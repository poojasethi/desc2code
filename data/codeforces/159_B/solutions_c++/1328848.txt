#include<stdio.h>

int cuplaj, cost, n, m;
int f[1005];
int f2[1005][1005];

int main ()
{
	int i, a, b; 
	
	//freopen("b.in","r",stdin);
	
	scanf("%d%d",&n,&m);
	
	for(i = 1; i <= n; i++)
	{
		scanf("%d%d", &a, &b);
		f[b]++;
		f2[b][a]++;
	}
		
	for(i = 1; i <= m; i++)	
	{
		scanf("%d%d", &a, &b);
		if(f[b])
		{
			cuplaj++;
			f[b]--;
		}	
		if(f2[b][a])
		{
			cost++;
			f2[b][a]--;
		}
	}	
	printf("%d %d\n",cuplaj,cost);
		
		
	return 0;
}
