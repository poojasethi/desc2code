#include<cstdio>
#include<algorithm>
using namespace std;

int n,m,a,b,ans1,ans2;
int s[1111][1111],tot[1111],req[1111];

int main() {
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%d%d",&a,&b);
		s[a][b]++;
		tot[b]++;
	}
	ans1=ans2=0;
	for(int i=0;i<m;i++){
		scanf("%d%d", &a, &b);
		if(s[a][b]){
			s[a][b]--;tot[b]--;ans2++;
		}else req[b]++;
	}
	for(int i=0;i<1111;i++)
		ans1+=min(tot[i],req[i]);
	printf("%d %d\n",ans1+ans2,ans2);

	return 0;
}
