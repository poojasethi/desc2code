#include <cstdio>
#include <algorithm>
#define FOR(x,y,z) for(int x=y;x<=z;++x)
using namespace std;
int n,m,a[1005][1005],b[1005][1005],x,y,p,q;
int main(){
	scanf("%d%d",&n,&m);
	FOR(i,1,1000) FOR(j,1,1000) a[i][j]=b[i][j]=0;
	FOR(i,1,n){
		scanf("%d%d",&x,&y);
		++a[y][x];
	}
	FOR(i,1,m){
		scanf("%d%d",&x,&y);
		++b[y][x];
	}
	p=q=0;
	FOR(i,1,1000){
		int s1=0,s2=0;
		FOR(j,1,1000) s1+=a[i][j],s2+=b[i][j],q+=min(a[i][j],b[i][j]);
		p+=min(s1,s2);
	}
	printf("%d %d\n",p,q);
	return 0;
}
