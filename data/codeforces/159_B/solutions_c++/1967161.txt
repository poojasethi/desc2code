#include <stdio.h>

int n, mi, x, y, c, b;
int m[1010][1010], cl[1010];

int main()
{
        scanf( "%d %d", &n, &mi );
        for ( int i = 0; i < n; ++i ) {
                scanf ( "%d %d", &x, &y );
                cl[y]++; m[x][y]++;
        }
        for ( int i = 0; i < mi; ++i ) {
                scanf ( "%d %d", &x, &y );
                if ( m[x][y] ) b++, m[x][y]--;
                if ( cl[y] ) c++, cl[y]--;
        }
        printf ( "%d %d\n", c, b );
}
