#include <iostream>
#include <map>
using namespace std;
int main(void)
{
    map<pair<int,int>,int >B;
    map<int,int> A;
    int i,c,d,n,m,a=0,b=0;
    cin>>n>>m;
    for (i=0;i<n;++i) {
        cin>>c>>d;
        ++A[d];
        ++B[make_pair(c,d)];
    }
    for (i=0;i<m;++i) {
        cin>>c>>d;
        if (A[d]>0)
            --A[d],a++;
        if (B[make_pair(c,d)]>0)
            --B[make_pair(c,d)],++b;
    }
    cout<<a<<" "<<b;
    return 0;
}
