#include "bits/stdc++.h"
using namespace std;
typedef long long LL;
int n, m, a[1001], b[1001][1001], u, v;
int main() {
    int i, x, y;
    scanf("%d%d", &n, &m);
    for (i = 0; i < n; i++) {
        scanf("%d%d", &x, &y);
        a[y]++;
        b[x][y]++;
    }
    for (i = 0; i < m; i++) {
        scanf("%d%d", &x, &y);
        if (a[y]) a[y]--, u++;
        if (b[x][y]) b[x][y]--, v++;
    }
    printf("%d %d\n", u, v);
	return 0;
}
