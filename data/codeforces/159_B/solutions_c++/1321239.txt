#include <iostream>
#include <stdio.h>
using namespace std;
int a[1111][1111];
int b[1111];
int n,m;
int main()
{
    cin>>n>>m;
    int ta,tb;
    int ans1=0,ans2=0;
    for(int i=0;i<n;i++) cin>>ta>>tb,a[ta][tb]++,b[tb]++;
    for(int i=0;i<m;i++)
    {
        cin>>ta>>tb;

        if(a[ta][tb]) ans2++,a[ta][tb]--;
        if(b[tb]) ans1++,b[tb]--;
    }
    cout<<ans1<< " "<<ans2<<endl;
}
