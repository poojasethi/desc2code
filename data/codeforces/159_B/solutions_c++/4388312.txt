#include <iostream>
int a[1000][1000],b[1000][1000];
using namespace std;
int main() {
    int x,y,ans1=0,ans2=0,n,m;
	cin>>n>>m;
	for (int i=0; i<n; i++) {
		cin>>x>>y;
		a[x-1][y-1]++;}
	for (int i=0; i<m; i++) {
		cin>>x>>y;
		b[x-1][y-1]++;	}
	for (int j=0; j<1000; j++) {
		x=0; y=0;
		for (int i=0; i<1000; i++) {
			x+=a[i][j]; y+=b[i][j];
			if (a[i][j] < b[i][j]) ans2+=a[i][j];
			else ans2+=b[i][j];	}
		if (x<y) ans1+=x;
		else ans1+=y; }
	cout<<ans1<<' '<<ans2;
}
