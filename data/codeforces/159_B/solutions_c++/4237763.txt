#include<cstdio>
int v[1020][1020];
int c[1020],n,m,x,y,a,b;
int main()
{
    scanf("%d %d",&n,&m);
    for(int i=0;i<n;i++)//can fit if y same
        scanf("%d %d",&x,&y),c[y]++,v[x][y]++;//x=color,y=diameter
    for(int i=0;i<m;i++)
    {
        scanf("%d %d",&x,&y);
        if(v[x][y])//both match
            b++,v[x][y]--;
        if(c[y])//hv this diameter
            a++,c[y]--;
    }
    printf("%d %d",a,b);
    return 0;
}
