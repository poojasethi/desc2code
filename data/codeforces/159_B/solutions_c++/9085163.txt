#include<iostream>
using namespace std;
int main(){
    int x,y,n,m;
    cin>>n>>m;
    int a[1024][1024];
    int b[1024]={0};
    for(int i=0;i<n;i++){
        cin>>x>>y;
        b[y]++;
        a[x][y]++;
    }
    int a1=0,a2=0;
    for(int j=0;j<m;j++){
        cin>>x>>y;
        if(a[x][y]>0){
            a[x][y]--;
            a1++;
        }
        if(b[y]>0){
            b[y]--;
            a2++;
        }
    }
    cout<<a2<<" "<<a1;

return 0;
}
