I = lambda:map(int, raw_input().split())
n,m = I()
f = [ I() for _ in range(n) ]

a = {}
b = {}

for _ in range(m):
  x, y = I()
  a[(x,y)] = a.get((x,y), 0) + 1
  b[y] = b.get(y, 0) + 1
ans1 = ans2 = 0
for t in f:
  if a.get((t[0], t[1])) > 0:
    ans2 += 1
    a[(t[0], t[1])] -= 1
  if b.get(t[1]) > 0:
    ans1 += 1
    b[t[1]] -= 1

print ans1, ans2
