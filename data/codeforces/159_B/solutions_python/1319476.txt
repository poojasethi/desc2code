n, m = map(int, raw_input().split())
d = {}
bu = [0] * 1024
bv = bu[:]
for _ in xrange(n):
    a, b = map(int, raw_input().split())
    d[(a, b)] = d.get((a, b), 0) + 1
    bu[b] += 1
y = 0
for _ in xrange(m):
    a, b = map(int, raw_input().split())
    if d.get((a, b), 0) > 0:
        d[(a, b)] -= 1
        y += 1
        bu[b] -= 1
    else:
        bv[b] += 1
x = sum(min(a, b) for a, b in zip(bu, bv))
print x + y, y
