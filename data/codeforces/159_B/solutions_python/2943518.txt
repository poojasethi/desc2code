I = lambda: map(int, raw_input().split())
def f(n):
    a = {}
    b = {}
    for x, y in [I() for i in range(n)]:
        a[y] = a.get(y, 0) + 1
        b[(x, y)] = b.get((x, y), 0) + 1
    return a, b

n, m = I()
a0, b0 = f(n)
a1, b1 = f(m)
x = 0
for k, v in a0.items():
    x += min(v, a1.get(k, 0))
y = 0
for k, v in b0.items():
    y += min(v, b1.get(k, 0))
print x, y
