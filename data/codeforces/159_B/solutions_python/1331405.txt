sizes = [0]*1000
colours = [[0]*1000 for z in range(1000)]
n,m = map(int,raw_input().split())

for _ in range(n):
    a,b = map(int,raw_input().split())
    a -= 1
    b -= 1
    sizes[b] += 1
    colours[b][a] += 1
    
closed = 0
beauty = 0
for _ in range(m):
    a,b = map(int,raw_input().split())
    a -= 1
    b -= 1
    
    if sizes[b]:
        closed += 1
        sizes[b] -= 1
    if colours[b][a]:
        beauty += 1
        colours[b][a] -= 1
        
print closed,beauty
