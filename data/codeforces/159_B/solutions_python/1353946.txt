n,m=map(int,raw_input().split())
p=[map(int,raw_input().split()) for _ in range(n)]
a={}
b={}
for _ in range(m):
    x,y=map(int,raw_input().split())
    a[(x,y)]=a.get((x,y),0)+1
    b[y]=b.get(y,0)+1
u=v=0
for x in p:
    if a.get((x[0],x[1])):
        u+=1
        a[(x[0],x[1])]-=1
    if b.get(x[1]):
        v+=1
        b[x[1]]-=1
print v,u
