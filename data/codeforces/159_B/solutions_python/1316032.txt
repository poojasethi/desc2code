#!/usr/bin/env python
import sys

(n, m) = map(int, raw_input().split(" "))
pen = {}

for i in range(n):
    (color, d) = map(int, raw_input().split(" "))
    if (color, d) not in pen:
        pen[(color, d)] = 0
    pen[(color, d)] += 1

beautiful = 0
closed = 0
cap = {}
for i in range(m):
    (color, d) = map(int, raw_input().split(" "))
    if (color, d) in pen and pen[(color, d)] > 0:
        pen[(color, d)] -= 1
        beautiful += 1
    else:
        if d not in cap:
            cap[d] = 0
        cap[d] += 1

marker = {}
for (c, d) in pen.keys():
    if pen[(c,d)] > 0:
        if d not in marker:
            marker[d] = 0
        marker[d] += pen[(c,d)]

for d in cap:
    if d in marker:
        closed += min(cap[d], marker[d])

print beautiful + closed, beautiful
