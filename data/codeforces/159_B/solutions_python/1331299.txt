#!/usr/bin/env python

from collections import defaultdict

markers = defaultdict(int)
caps = defaultdict(int)

n, m = map(int, raw_input().strip().split())
for i in xrange(n):
    x, y = map(int, raw_input().strip().split())
    markers[x, y] += 1
for i in xrange(m):
    a, b = map(int, raw_input().strip().split())
    caps[a, b] += 1

j, jb = 0, 0
for x, y in markers:
    if (x, y) in caps:
        q = min(markers[x, y], caps[x, y])
        markers[x, y] -= q
        caps[x, y] -= q
        j += q
        jb += q

markersd = defaultdict(int)
capsd = defaultdict(int)

for (x, y), q in markers.iteritems():
    markersd[y] += q
for (a, b), q in caps.iteritems():
    capsd[b] += q

for y in markersd:
    if y in capsd:
        q = min(markersd[y], capsd[y])
        markersd[y] -= q
        capsd[y] -= q
        j += q

print j, jb
