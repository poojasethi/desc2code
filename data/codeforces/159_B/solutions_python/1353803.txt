#!/usr/bin/env python
#coding:utf-8

n, m = map(int, raw_input().split(' '))

a = [[0 for x in range(1001)] for y in range(1001)]
b = [0 for x in range(10001)]
match = 0
beautiful_match = 0

for i in range(n):
    l, r = map(int, raw_input().split(' '))
    a[l][r] += 1
    b[r] += 1
for i in range(m):
    l, r = map(int, raw_input().split(' '))
    if b[r] > 0:
	b[r] -= 1
	match += 1
    if a[l][r] > 0:
	a[l][r] -= 1
	beautiful_match += 1



print match, beautiful_match
