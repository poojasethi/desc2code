
import collections, string
n, m = [int(j) for j in raw_input().split()]
marker_color = collections.defaultdict(collections.Counter)
marker_size = collections.defaultdict(collections.Counter)

for j in range(n):
    col, size = [int(j1) for j1 in raw_input().split()] 
    marker_color[col][size]+=1
    marker_size[size][col]+=1

bmatches = 0
matches = 0
for j in range(m):
    col, size = [int(j1) for j1 in raw_input().split()]
    if col in marker_color and size in  marker_color[col] and marker_color[col][size]:
        marker_color[col][size] -= 1
        marker_size[size][col] -= 1
        bmatches += 1
    elif size in marker_size:
        marker_size[size]['waitlist'] += 1

for size in marker_size:
    if 'waitlist' in marker_size[size]:
        n = marker_size[size]['waitlist']
        del marker_size[size]['waitlist']
        colCounter = marker_size[size]
        colCount = sum(colCounter.values())
        matches += min([n, colCount])
##        for col in colCounter:
##            del marker_color[col][size]

print matches+bmatches, bmatches
