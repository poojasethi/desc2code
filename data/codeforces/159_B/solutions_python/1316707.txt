from collections import defaultdict

[n, m] = map(int, raw_input().split())

markers = defaultdict(int)
caps = defaultdict(int)

bm = defaultdict(int)
bc = defaultdict(int)

for i in xrange(n):
  [x, y] = map(int, raw_input().split())
  markers[y] += 1
  bm[(x,y)] += 1 

for i in xrange(m):
  [a, b] = map(int, raw_input().split())
  caps[b] += 1
  bc[(a,b)] += 1 



u = 0
v = 0

for m in markers:
  u += min(markers[m], caps[m])

for m in bm:
  v += min(bm[m], bc[m])

print u, v
