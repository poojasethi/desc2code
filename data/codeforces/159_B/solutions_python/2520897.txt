def f(n):
	a = {}
	b = {}
	for x, y in [map(int, raw_input().split()) for i in range(n)]:
		a[y] = (a.get(y) or 0) + 1
		b[(x, y)] = (b.get((x, y)) or 0) + 1
	return a, b

n, m = map(int, raw_input().split())
a0, b0 = f(n)
a1, b1 = f(m)
x = 0
for k, v in a0.items():
	x += min(v, a1.get(k) or 0)
y = 0
for k, v in b0.items():
	y += min(v, b1.get(k) or 0)
print x, y
