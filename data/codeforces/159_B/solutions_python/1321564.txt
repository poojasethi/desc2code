#!/usr/bin/env python

pen = {}
(n, m) = map(int, raw_input().split())
for each in range(n):
	(color, diam) = map(int, raw_input().split())
	if pen.has_key(diam):
		if pen[diam].has_key(color):
			pen[diam][color]+=1
		else:
			pen[diam][color]=1
	else:
		pen[diam] = {}
		pen[diam][color] = 1

cap = {}
count_full_closed = 0
for each in range(m):
	(color, diam) = map(int, raw_input().split())
	if pen.has_key(diam):
		if pen[diam].has_key(color) and pen[diam][color] > 0:
			count_full_closed+=1
			pen[diam][color]-=1
			if pen[diam][color] == 0:
				del(pen[diam][color])
		else:
			if cap.has_key(diam):
				cap[diam]+=1
			else:
				cap[diam] = 1

count_closed = count_full_closed
for diam in cap:
	count_diam = cap[diam]
	if pen.has_key(diam):
		for color in pen[diam]:
			if pen[diam][color] >= count_diam:
				count_closed += count_diam
				count_diam = 0
			else:
				count_closed += pen[diam][color]
				count_diam -= pen[diam][color]
print str(count_closed)+' '+str(count_full_closed)
