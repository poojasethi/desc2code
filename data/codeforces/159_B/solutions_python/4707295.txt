n, m = map(int, raw_input().split())
x, y, u, v = {}, {}, 0, 0
for i in range(n):
    a, b = raw_input().split()
    if b in x: x[b][a] = x[b].get(a, 0) + 1
    else: x[b] = {a: 1}
for i in range(m):
    a, b = raw_input().split()
    if b in y: y[b][a] = y[b].get(a, 0) + 1
    else: y[b] = {a: 1}
for b in set(x.keys()) & set(y.keys()):
    u += min(sum(x[b].values()), sum(y[b].values()))
    v += sum(min(x[b][a], y[b][a]) for a in set(x[b].keys()) & set(y[b].keys()))
print u, v
