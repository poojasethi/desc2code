s= raw_input().split()

n= int(s[0])
m = int( s[1] )
a=[[] for i in range(1001)]
b=[[] for i in range(1001)]


sum=0
sumb=0

for i in range(n):
 s= raw_input().split()
 a[int(s[1])].append(int(s[0]))
 
for i in range(m):
 s= raw_input().split()
 b[int(s[1])].append(int(s[0]))

for i in range(1001):
 sum+=min(len(a[i]),len(b[i]))
  
  
for i in range(1001):
 nub=[0]*1001
 for j in a[i]:
  nub[j]+=1
 for j in b[i]:
  if nub[j]>0:
   nub[j]-=1
   sumb+=1 
 
 
print sum,sumb
