marks, close = map(int, raw_input().split())
markers = {}
for i in xrange(marks):
    b,a = map(int, raw_input().split())
    markers.setdefault(a, [])
    markers[a].append(b)
caps = {}
boootiful = 0
for i in xrange(close):
    b,a = map(int, raw_input().split())
    caps.setdefault(a, [])
    caps[a].append(b)


markset = set(markers)
regular = 0
bootiful = 0
while markset:
    current = min(markset)
    markset.remove(current)
    if current in caps:
        regular += min(len(caps[current]),len(markers[current]))
        caps[current].sort()
        markers[current].sort()
        i = 0
        j = 0
        lencap = len(caps[current])
        lenmar = len(markers[current])
        while i < lencap and j < lenmar :
            if caps[current][i] == markers[current][j]:
                bootiful += 1
                i,j = i+1,j+1
            elif caps[current][i] > markers[current][j]:
                j+=1
            elif caps[current][i] < markers[current][j]:
                i+=1
print regular, bootiful
