from collections import defaultdict
R=lambda:map(int,raw_input().split())
n,m=R()
A=[tuple(R()) for i in xrange(n)]
B=[tuple(R()) for i in xrange(m)]
u=0;v=0
da=defaultdict(int)
db=defaultdict(int)
for a,b in A: da[b]+=1
for a,b in B: db[b]+=1
for a in da.iterkeys():
    if a in db:
        u += min(da[a],db[a])
da.clear()
db.clear()
for a in A: da[a]+=1
for a in B: db[a]+=1
for a in da.iterkeys():
    if a in db:
       v += min(da[a],db[a])
print u,v
