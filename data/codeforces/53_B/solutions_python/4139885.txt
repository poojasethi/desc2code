import math
h,w = tuple(map(int, raw_input().split()))

hh = 1
ww = int(math.floor(w*1.25))
while hh*2<=min(h,ww):
    hh*=2

wwtop = min(w, int(math.ceil(hh*1.25)))
ans = (hh,wwtop)

ww = 1
hh = int(math.floor(h*1.25))
while ww*2<=min(w,h): 
    ww*=2

hhtop = min(int(math.floor(ww*1.25)), h)
if hhtop*ww > ans[0]*ans[1]:
    ans = (hhtop, ww)
elif hhtop*ww == ans[0]*ans[1] and hhtop > ans[0]:
    ans = (hhtop, ww)
print ans[0], ans[1]
