h, w = map(int, raw_input().split())

rh = 1
while rh * 2 <= h and rh * 1.6 <= w:
    rh *= 2
rw = min(rh * 1.25, w)

rw2 = 1
while rw2 * 2 <= w and rw2 * 1.6 <= h:
    rw2 *= 2
rh2 = min(rw2 * 1.25, h)

if rh * rw == rh2 * rw2:
    if rh > rh2:
        print "%d %d" % (rh, rw)
    else:
        print "%d %d" % (rh2, rw2)
elif rh * rw > rh2 * rw2:
    print "%d %d" % (rh, rw)
else:
    print "%d %d" % (rh2, rw2)
