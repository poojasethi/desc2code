def f(H, W):
    h = 1
    while h * 2 <= H and h * 2 < 1.25 * W:
        h *= 2
    w = W
    if h < 0.8 * w:
        w = int(h * 1.25)
    return (h, w)
h, w = map(int, raw_input().split())
a, b = f(h, w), f(w, h)
if a[0] * a[1] == b[0] * b[1]: print b[1], b[0]
else:
    if a[0] * a[1] > b[0] * b[1]: print a[0], a[1]
    else: print b[1], b[0]
