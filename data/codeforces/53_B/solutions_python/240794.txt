#!/usr/bin/env python
import string, math

# Input
x, y = map(int, string.split(raw_input()))

x_l = int(math.log(x, 2)) + 1
y_l = int(math.log(y, 2)) + 1

rx = 0
ry = 0

# Check by x
for i in range(0, x_l):
	new_x = 2 ** i
	
	max_y = 1.25 * new_x
	min_y = 0.8  * new_x

	if y >= max_y:
		new_y = max_y
	elif (y <= max_y) and (y >= min_y):
		new_y = y
	else:
		new_y = 0

	if (new_x * new_y > rx * ry) or ((new_x * new_y == rx * ry) and (new_x > rx)):
		rx = new_x
		ry = new_y

# Check by y
for i in range(0, y_l):
	new_y = 2 ** i
	
	max_x = 1.25 * new_y
	min_x = 0.8  * new_y

	if x >= max_x:
		new_x = max_x
	elif (x <= max_x) and (x >= min_x):
		new_x = x
	else:
		new_x = 0

	if (new_x * new_y > rx * ry) or ((new_x * new_y == rx * ry) and (new_x > rx)):
		rx = new_x
		ry = new_y

print str(int(rx)) + ' ' + str(int(ry))
