h,w=map(int,raw_input().split())
s,a,b=0,0,0
for i in range(33,-1,-1):
  y=1<<i
  if y<=h and (y*4+4)/5<=w:
    x=min(w,y*5/4)
    if x*y>s:
      s,a,b=x*y,y,x
    break
