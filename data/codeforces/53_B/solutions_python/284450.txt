#!/usr/bin/env python

from sys import stdin

H, W = map(int, stdin.readline().split())

besth, bestw, bestarea = 0, 0, 0

w = 1
while w <= W:
   h = min(H, w*5/4)
   if 8*w <= h*10 and h*4 <= w*5:
      area = w*h
      if area > bestarea or area == bestarea and h > besth:
         besth, bestw, bestarea = h, w, area
   #print 'w:', w, 'h:', h
   w *= 2

h = 1
while h <= H:
   w = min(W, h*10/8)
   if 8*w <= h*10 and h*4 <= w*5:
      area = w*h
      if area > bestarea or area == bestarea and h > besth:
         besth, bestw, bestarea = h, w, area
   h *= 2
   

print besth, bestw
