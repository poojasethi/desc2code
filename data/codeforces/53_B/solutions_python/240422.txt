H, W = map(int, raw_input().split())

h2 = 1
while h2 * 2 <= H and h2*2 < 1.25 * W:
    h2 *= 2
w = W
if h2 < 0.8 * w:
    w = int(h2 * 1.25)
    
w2 = 1
while w2 * 2 <= W and w2 * 2 < 1.25 * H:
    w2 *= 2

h = H
if w2 < 0.8 * h:
    h = int(w2 * 1.25)

if h2 * w == h * w2:
    print h, ' ', w2
else:
    if h2 * w > h * w2:
        print h2, w
    else:
        print h, w2
