#!/usr/bin/python
import sys
h, w = [int (x) for x in sys.stdin.readline ().split ()]
res, resh, resw = 0, 0, 0
for p in range (30):
	a = 2 ** p
	if a <= h:
		curh = a
		curw = min (w, (a * 5) / 4)
		cur = curh * curw
		if curw * 5 >= curh * 4:
			if res < cur or (res == cur and resh < curh):
				res, resh, resw = cur, curh, curw
	if a <= w:
		curw = a
		curh = min (h, (a * 5) / 4)
		cur = curh * curw
		if curh * 5 >= curw * 4:
			if res < cur or (res == cur and resh < curh):
				res, resh, resw = cur, curh, curw
print resh, resw
