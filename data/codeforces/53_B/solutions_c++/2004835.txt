#include<cstdio>
#include<algorithm>
#include<iostream>
int main(){
	long long h,w;
	std::cin>>h>>w;
	long long H=-1,W;
	for(long long i=1;i<=h;i*=2){
		long long j=std::min((5*i)/4,w);
		//std::cout<<i<<" "<<j<<"\n";
		if(5*i<4*j||4*i>5*j)continue;
		if(H==-1){
			H=i;
			W=j;
		}
		else{
			if(H*W<i*j||H*W==i*j&&H<i){
				H=i;
				W=j;
			}
		}
	}
	for(long long j=1;j<=w;j*=2){
		long long i=std::min((5*j)/4,h);
		if(5*i<4*j||4*i>5*j)continue;
		if(H==-1){
			H=i;
			W=j;
		}
		else{
			if(H*W<i*j||H*W==i*j&&H<i){
				H=i;
				W=j;
			}
		}
	}
	std::cout<<H<<" "<<W<<"\n";
}
