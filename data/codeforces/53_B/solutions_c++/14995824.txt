#include <bits/stdc++.h>
#define ll long long int
using namespace std;
int main(){
	ll h,w,hr,wr,f1,f2;
	ll mx=0,a[33]={0};
	cin>>h>>w;
	a[0]=1;
	for(int i=1;i<=30;i++){
		a[i]=2*a[i-1];
		if(a[i]<=h)f1=a[i];
		if(a[i]<=w)f2=a[i];
	}
	if(w==1)f2=1;if(h==1)f1=1;
	//cout<<f1<<" "<<f2<<endl;
	while(f2){
		double aux2=f2/0.8+0.0001;
		ll aux=(ll)(aux2);
		if(aux>h)aux=h;
		if(aux>=ceil(f2/1.25+0.0001)&&aux*f2>mx){
			hr=aux;
			wr=f2;
			mx=aux*f2;
			
		}
		f2/=2;
	}
	while(f1){
		double aux2=f1/0.8+0.0001;
		ll aux=(ll)(aux2);
		if(aux>w)aux=w;
		if(aux>=ceil(f1/1.25+0.0001)&&aux*f1>mx){
			hr=f1;
			wr=aux;
			mx=f1*aux;
		}
		f1/=2;
	}
	cout<<hr<<" "<<wr<<endl;
}
