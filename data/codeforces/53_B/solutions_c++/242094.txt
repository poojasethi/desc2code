#include <iostream>
#include <vector>
#include <map>
#include <set>
#include <list>
#include <algorithm>

using namespace std;

#define PB	push_back

int main()
{
	unsigned long long h, w;
	cin >> h >> w;
	unsigned long long x = 1;
	unsigned long long y = 1;
	while(w >= x && w >= y){
		x *= 2;
		y *= 2;
	}
	x /= 2;
	y /= 2;

	x = min(x, w);
	y = min(y, h);
	w = min(w, (unsigned long long)(y*1.25));
	h = min(h, (unsigned long long)(x*1.25));
	if(h*x >= y*w){
		cout << h << " " << x;
	}else{
		cout << y << " " << w;
	}

	return 0;
}
