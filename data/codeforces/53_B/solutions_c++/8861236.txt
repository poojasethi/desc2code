#include<iostream>

#define int long long

using namespace std;

int ans, a1, a2;

main()
{
	int h, w;
	cin >> h >> w;
	for(int i = 0; (1 << i) <= h; i++)
	{
		int x = (1 << i);
	    int l = (x * 4 + 4) / 5, r = x * 5 / 4;
		if(w < l)
			break;
		int t = min(r, w);
		if(x * t >= ans)
		{
			ans = x * t;
			a1 = x;
			a2 = t;
		}
	}
	for(int i = 0; (1 << i) <= w; i++)
	{
		int x = (1 << i);
	    int l = (x * 4 + 4) / 5, r = x * 5 / 4;
		if(h < l)
			break;
		int t = min(r, h);
		if(x * t > ans || (x * t == ans && t > a1))
		{
			ans = x * t;
			a1 = t;
			a2 = x;
		}
	}
	cout << a1 << ' ' << a2 << endl;
	return 0;
}
