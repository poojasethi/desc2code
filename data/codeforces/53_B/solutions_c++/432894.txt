#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    __int64 h, w, x, y;
    cin>>h>>w;
    x = y = 1;
    while( w >= x && w >= y )
        x *= 2, y *= 2;
    x /= 2; y /= 2;

    x = min( x, w );
    y = min( y, h );
    w = min( w, (__int64)( y * 1.25) );
    h = min( h, (__int64) (x * 1.25) );
    if( h * x >= y * w )
        cout<<h<<" "<<x<<endl;
    else cout<<y<<" "<<w<<endl;

    return 0;
}
