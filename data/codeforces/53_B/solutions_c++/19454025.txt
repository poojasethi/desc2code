#include<bits/stdc++.h>
#define f first
#define s second
#define mp make_pair
using namespace std;
typedef long long ll;

ll h, w, H, W;
pair<ll, ll> x;

void f(ll h, ll w) {
	if(h*4 > w*5 || w*4 > h*5) return ;
	if(h*w > x.f * x.s || (h*w == x.f * x.s && h > x.f))
		x.f = h, x.s = w;
}

int main(int argc, char **argv) {
	cin >> h >> w;
	for(H=1; H<=h ; H <<= 1) f(H, min(w, 5*H/4));
	for(W=1; W<=w ; W <<= 1) f(min(h, W*5/4), W);
	cout << x.f << " " << x.s << endl;
	return 0;
}
