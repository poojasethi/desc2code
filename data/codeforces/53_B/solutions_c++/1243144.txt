#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long LL;
LL a, b;
LL ax=0, ay=0;
int main(void) {
	scanf("%I64d%I64d", &a, &b);
	for(LL i=1;i<=a;i*=2) {
		LL j1 = (i*4+4)/5, j2=(i*5)/4;
		j2=min(b, j2);
		if(j2<j1) continue;
		if(ax*ay < i*j2 || (ax*ay==i*j2 && ax<i)) {
			ax=i; ay=j2;
		}
	}
	for(LL j=1;j<=b;j*=2) {
		LL i1 = (j*4+4)/5, i2=(j*5)/4;
		i2=min(a, i2);
		if(i2<i1) continue;
		if(ax*ay < i2*j || (ax*ay==i2*j && ax<i2)) {
			ax=i2; ay=j;
		}
	}
	printf("%I64d %I64d\n", ax, ay);
	return 0;
}
