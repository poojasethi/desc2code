
#include <iostream>
#include <cstdio>

using namespace std;

typedef long long ll;

ll h,w;

int main(){
    cin >> h >> w;
    
    ll n=0, m=0;

    for (ll i=1;i<=h;i*=2){
        ll mini= i*4/5 + 1;
        ll maxi= i*5/4;

        maxi= min (maxi, w);

        if (mini>w)
            break;

        n=i, m=maxi;
    }

    for (ll j=1;j<=w;j*=2){
        ll mini= j*4/5 + 1;
        ll maxi= j*5/4;

        maxi= min (maxi, h);

        if (mini>h)
            break;

        if ((n*m<maxi*j) || (n*m==maxi*j && maxi>n))
            n=maxi, m=j;
    }

    cout << n << ' ' << m << endl;

    return 0;
}
