#include <cstdio>

#include <algorithm>
using namespace std;
typedef long long ll;
ll go(ll x) {
	while( x - x&(-x) ) {
		x -= x&(-x);
	}
	return x;
}
int main() {
	ll n, m, nn, mm, a = 1, b = 1;
	scanf("%I64d%I64d", &n, &m);
//	if(n > 1.25 * m) n = 1.25 * m;
//	else if(n < 0.8 * m) m = 0.8 * m;
	nn = n, mm = m;
	while(a <= n && m > 0.8*a) a *= 2;
	a /= 2;
	
	if(m > 1.25 * a) mm = 1.25 * a;
	while(b <= m && n > 0.8*b) b *= 2;
	b /= 2;
	
	if(n > 1.25 * b) nn = 1.25 * b;
	if(a*mm>nn*b || (a*mm == nn*b && a > nn)) printf("%I64d %I64d\n", a, mm);
	else printf("%I64d %I64d\n", nn, b);
	
    return 0;
}
