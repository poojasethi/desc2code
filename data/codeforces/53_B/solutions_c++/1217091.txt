#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	long long h, w, ah = 0, aw = 0;
	cin >> h >> w;
	for(long long i = 1; i <= h; i <<= 1)
	{
		long long tw = (i*10)/8;
		tw = min(tw, w);
		if(tw >= (i*4 + 4)/5 && ah*aw <= tw*i)
			ah = i, aw = tw;
	}

	for(long long i = 1; i <= w; i <<= 1)
	{
		long long th = (i*5)/4;
		th = min(th, h);
		if(th >= (i*8 + 9)/10 && (ah*aw < th*i || (ah*aw==th*i && th > ah)))
			aw = i, ah = th;
	}
	cout << ah << ' ' << aw;

	return 0;
}
