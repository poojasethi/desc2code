#include<iostream>
using namespace std;
typedef long long LL;
LL H,W;
int main()
{
    cin >> H >> W;
    pair<LL,pair<LL,LL> > ans(-1,make_pair(-1,-1));
    for(LL h = 1; h <= H; h <<= 1)
    {
           LL hh = h * 100;
           LL ww = min(W, hh / 80);
           if(125 * ww >= hh)
                 ans = max(ans,make_pair(ww*h,make_pair(h,ww)));
    }   
    
    for(LL w = 1; w <= W; w <<= 1)
    {
           LL hh = min(H,(125 * w) / 100);
           if(100 * hh >= 80 * w)
                 ans = max(ans,make_pair(w*hh,make_pair(hh,w)));
    }   
    cout << ans.second.first << " " << ans.second.second << endl;
}
