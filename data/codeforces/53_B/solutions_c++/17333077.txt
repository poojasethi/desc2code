#include<bits/stdc++.h>
using namespace std;
int main()
{
    int h,w,m,x=-1,y=-1,s=-1;
    cin>>h>>w;
    int c=1;
    vector <int> v;
    v.push_back(c);
    m=max(h,w);
    while(2*c<=m)
    {
        c*=2;
        v.push_back(c);
    }
    int z=v.size(),k;
    bool f=0;
    for(int i=z-1;i>=0;i--)
    {
        k=v[i]*1.3;
        for(double j=k;j>=1;j--)
        {
            if(v[i]/j<=1.25&&v[i]/j+1e-15>=.8&&v[i]<=h&&j<=w&&(v[i]*j>s||v[i]*j==s&&v[i]>x)) f=1,x=v[i],y=j,s=x*y;
            if(j/v[i]<=1.25&&j/v[i]+1e-15>=.8&&j<=h&&v[i]<=w&&(v[i]*j>s||v[i]*j==s&&j>x))    f=1,x=j,y=v[i],s=x*y;
        if(f)
            {cout<<x<<" "<<y<<endl; return 0;}
        }
    }
	return 0;
}
