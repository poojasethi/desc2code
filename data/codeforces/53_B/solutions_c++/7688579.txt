#include<cstdio>
#include<algorithm>
using namespace std;
int h,w,sv1,sv2;
void chk(int a,int b){
    if(1.0*a <= b*1.25+1e-10 && a <= h && b <= w && (1ll*sv1*sv2 < 1ll*a*b || (1ll*sv1*sv2 == 1ll*a*b && a > sv1))) sv1 = a,sv2 = b;
}
int main(){
    scanf("%d%d",&h,&w);
    for(int i=1;i<=max(h,w);i*=2){
        chk((int)(min(1.0*i,1.25*w)-1e-10)+1,(int)(min(i/0.8,1.0*w)-1e-10)+1);
        chk((int)(min(1.0*h,1.25*i)-1e-10)+1,(int)(min(1.0*i,h/0.8)-1e-10)+1);
    }
    printf("%d %d\n",sv1,sv2);
    return 0;
}
