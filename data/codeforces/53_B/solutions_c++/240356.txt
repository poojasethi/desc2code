#include <iostream>
#include <vector>
#include <functional>
#include <algorithm>
using namespace std;

typedef long long ll;

int main()
{
    ll w, h;
    cin >> h >> w;
    
    ll j,k,l,m,t;
    j = min(h, w * 5 / 4);
    t = 1;
    while(t * 2 <= j) 
        t *= 2;
    j = t;
    k = min(w, j * 5 / 4);


    m = min(w, h * 5 / 4);
    t = 1;
    while(t * 2 <= m) 
        t *= 2;
    m = t;
    l = min(h, m * 5 / 4);


    if(j * k < l * m || j * k == l * m && l > j) {
        j = l;
        k = m;
    }

    cout << j << " " << k << endl;
    
}
