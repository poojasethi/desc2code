#include <iostream>
#include <cstdio>
using namespace std;
pair <long long, long long> ans;
int main()
{
  long long h, w, t, p = 1;
  cin >> h >> w;
  for(int i = 0; i <= 30; i++)
  {
    if (p <= h && 4 * p <= 5 * w)
    {
      t = min(w, 5 * p / 4);
      if (p * t > ans.first * ans.second || (p * t == ans.first * ans.second && ans.first < p))
        ans.first = p, ans.second = t;
    }
    if (p <= w && 4 * p <= 5 * h)
    {
      t = min(h, 5 * p / 4);
      if (p * t > ans.first * ans.second || (p * t == ans.first * ans.second && ans.first < t))
        ans.first = t, ans.second = p;
    }
    p *= 2;
  }
  cout << ans.first << " " << ans.second;
  return 0;
}
