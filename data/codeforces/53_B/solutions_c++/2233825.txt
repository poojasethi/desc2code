#include <iostream>
using namespace std;

typedef long long ll;

int main()
{
	ll h, w;
	cin >> h >> w;
	ll mxa = 0, mxh = 0;
	for(ll i = 1; i <= w; i*=2)
	{
		if(i * 8/10 < h)
		{
			ll H = min(h, i*125/100);
			ll a = H*i;
			if(a > mxa)
				mxa = a,
				mxh = H;
			else if(a == mxa)
				mxh = max(mxh, H);
		}
	}
	for(ll i = 1; i <= h; i*=2)
	{
		if(i*100/125 < w)
		{
			ll W = min(w, i*10/8);
			ll a = W*i;
			if(a > mxa)
				mxa = a,
				mxh = i;
			else
				mxh = max(mxh, i);
		}
	}
	cout << mxh << " " << mxa/mxh << endl;
	{int _;cin>>_;}
	return 0;
}
