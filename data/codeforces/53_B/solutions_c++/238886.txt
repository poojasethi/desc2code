#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	long long H,W,h,w;
	long long F = 5;
	cin >> H >> W;
	
	
	long long ba=-1,bw,bh;
	
	for(h = 1; h <= H; h <<= 1)
	{
		w = (h<<2)/F + 1;
		if(w <= W)
		{
			w = min(W,(F*h)>>2);
			if(h*w > ba || (h*w==ba && h > bh))
			{
				ba = h*w;
				bh = h;
				bw = w;
			}
		}
	}
		
	for(w = 1; w <= W; w <<= 1)
	{
		h = (w<<2)/F + 1;
		if(h <= H)
		{
			h = min(H,(F*w)>>2);
			if(h*w > ba || (h*w==ba && h > bh))
			{
				ba = h*w;
				bh = h;
				bw = w;
			}
		}
	}	
	cout << bh << " " << bw << endl;
}
