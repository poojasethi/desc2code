#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>
using namespace std;
int main() {
    long long H,W;
    cin >> H >> W;
    pair< long long, pair< long long, long long > > res;
    res.first=-1;
    for(long long h=1;h<=H;h<<=1) {
        long long w=min(W,5*h/4);
        if(4*h > 5*w) {
            continue;
        }
        res=max(res,make_pair(w*h,make_pair(h,w)));
    }
    
    for(long long w=1;w<=W;w<<=1) {
        long long h=min(H,5*w/4);
        if(5*h < 4*w) {
            continue;
        }
        res=max(res,make_pair(w*h,make_pair(h,w)));
    }
    cout << res.second.first << " " << res.second.second << endl;
    return 0;
}
