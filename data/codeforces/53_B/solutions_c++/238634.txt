#include <iostream>

using namespace std;

long long h,w;

long long gh, gw;

void check(long long nh, long long nw){
	if(4*nw > 5*nh)
		return;
	if(4*nh > 5*nw)
		return;

	if(nh*nw > gh*gw){
		gh = nh;
		gw = nw;
	}
	if((nh*nw == gh*gw)&&(nh > gh)){
		gh = nh;
		gw = nw;
	}

}

int main(){
	cin >> h >> w;

	for(long long i = 1; i <= h; i *= 2){
		check(i, min(i*5/4, w));
	}
	for(long long i = 1; i <= w; i *= 2){
		check(min(i*5/4, h), i);
	}

	cout << gh << " " << gw << endl;

	return 0;
}
