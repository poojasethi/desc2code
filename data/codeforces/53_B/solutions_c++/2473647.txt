#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <string>
#include <iostream>
#include <cmath>
using namespace std;
typedef long long ll;
long long h,w,s;
long long ansh,answ,anss;
void check(long long a,long long b)
{
///	printf("%I64d %I64d\n",a,b);
	if(a*b==anss&&a>ansh||a*b>anss)
		anss=a*b,ansh=a,answ=b;
}	
int main()
{
	scanf("%I64d%I64d",&h,&w);
	for(int i=0;h>=(1ll<<i);++i)
	{
		ll sb=1ll<<i;
		if(sb*1.0/w<=1.25)	check(sb,min(w,(ll)floor(sb*1.25)));
	}
//	printf("!!!\n");
	for(int i=0;w>=(1ll<<i);++i)
	{
		ll sb=1ll<<i;
		if(h*1.0/sb>=0.8) check(min(h,ll(floor(sb*1.25))),sb);
	}
	cout<<ansh<<" "<<answ<<endl;
}
