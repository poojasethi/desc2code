#include <bits/stdc++.h>

using namespace std;

long long p2[32];
long long a,b;
long long h,w;
long long area=0;

void tryto(long long x,long long y)
{
    if(x>a||y>b) return;
    if((x*y>=area)&&(x>h))
    {
        if((double)x/y>=0.7999999999&&(double)x/y<=1.25){
        area=x*y;
        h=x;w=y;
        }
    }
}

int main()
{
    cin >> a >> b;
    int i;
    p2[0]=1;
    for(i=1;i<30;i++) p2[i]=p2[i-1]*2;
    for(i=0;i<30;i++)
    {
        tryto(p2[i],min((long long)ceil(1.25*p2[i]),b));
        tryto(min((long long)floor(1.25*p2[i]),a),p2[i]);
    }
    cout << h << " " << w << endl;
}
