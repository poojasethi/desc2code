#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"

long long h,w,x,y,i,w2,h2;

int main()
{
  cin>>h>>w;
  for(i=0;1<<i<=h;i++)
	{
	  h2=1ll<<i;
    w2=min(w,h2*5/4); 
		if(w2*5/4<h2)
		  continue;
    if(w2*h2>x*y || (w2*h2==x*y && h2>x))
		{
		  x=1<<i;
			y=w2;
		}
	}
  for(i=0;1<<i<=w;i++)
	{
    h2=min(h,(1ll<<i)*5/4);
		if(h2*5/4<1<<i)
		  continue;  
    if((1<<i)*h2>x*y || ((1<<i)*h2==x*y && h2>x))
		{
		  x=h2;
			y=1<<i;
		}
	}
	cout<<x<<" "<<y;
	return 0;
}
