#include <iostream>
using namespace std;
int main()
{
	long long h, w, h1=0, w1, h2=0, w2, x, y, z;
	cin>>h>>w;
	for (x=0x40000000; h1==0||h2==0; x>>=1) {
		y = (x*4+4)/5;
		z = x*5/4;
		if (h1==0 && x<=w && y<=h) {
			w1 = x; h1 = min(h, z); 
		}
		if (h2==0 && x<=h && y<=w) {
			h2 = x; w2 = min(w, z);
		}
	}
	if (h1*w1>h2*w2 || h1*w1==h2*w2 && h1>h2) cout<<h1<<" "<<w1<<endl;
	else cout<<h2<<" "<<w2<<endl;
	return 0;
}
