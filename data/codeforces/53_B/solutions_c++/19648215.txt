#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll x, y, a, b, j;
ll ans;
int main() {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d%d", &x, &y);
	for (ll i = 1; i <= 1000000000; i *= 2) {
		if (i <= x) {
			j = i * 1.25;
			j = min(j, y);
			if (j < i * 0.8)
				continue;
			if (ans < i * j) {
				ans = i * j;
				a = i, b = j;
			}
			if (ans == i * j && i > a) {
				a = i, b = j;
			}
		}
		if (i <= y) {
			j = i * 1.25;
			j = min(j, x);
			if (j < i * 0.8)
				continue;
			if (ans < i * j) {
				ans = i * j;
				a = j, b = i;
			}
			if (ans == i * j && j > a) {
				a = j, b = i;
			}

		}
	}
	cout << a << " " << b << endl;
	return 0;
}
