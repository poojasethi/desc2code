#include <iostream>
using namespace std;
int main() {
    int a, b, h1, h2, w1, w2, p;
    while(cin >> a >> b) {
        h1 = min(a, (int)(b * 1.25));
        p = 1;
        while((p<<1) <= h1) p <<= 1;
        h1 = p;
        w1 = min(b, (int)(h1 * 1.25));

        w2 = min(b, (int)(a * 1.25));
        p = 1;
        while((p<<1) <= w2) p <<= 1;
        w2 = p;
        h2 = min(a, (int)(w2 * 1.25));

        if((long long)h1 * w1 < (long long)h2 * w2 || (long long)h1 * w1 == (long long)h2 * w2 && h1 < h2) {
            h1 = h2;
            w1 = w2;
        }

        cout << h1 << " " << w1 << endl;
    }
    return 0;
}
