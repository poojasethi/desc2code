#include<iostream>
using namespace std;
typedef long long int ll;
ll mini(ll a,long double b)
{
if(a<=b)
return a;
else
return (ll) b;
}
int main()
{
ll h,w,k,a,b;
long double al,ah,pp=1;
cin>>h>>w;
a=h,b=w;
k=1;
while(k<b)
k=k<<1;
if(k>b)
k/=2;
b=k;
while(true)
{
al=b*4.0*pp/5;
ah=b*5.0*pp/4;
if(a>=al){
a=mini(a,ah);
break;
}
b/=2;
}
ll c,d;
long double wl,wh;
c=h,d=w;
k=1;
while(k<c)
k=k<<1;
if(k>c)
k/=2;
c=k;
while(true)
{
wh=c*5*pp/4;
wl=c*4*pp/5;
if(d>=wl)
{
d=mini(d,wh);
break;
}
c/=2;
}
//cout<<a<<" "<<b<<" "<<c<<" "<<d<<endl;
if(a*b>c*d)
cout<<a<<" "<<b<<endl;
else if(a*b==c*d)
{
if(a>c)
cout<<a<<" "<<b<<endl;
else
cout<<c<<" "<<d<<endl;
}else
cout<<c<<" "<<d<<endl;
return 0;
}
