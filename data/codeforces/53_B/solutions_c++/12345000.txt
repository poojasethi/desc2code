#include <bits/stdc++.h>
using namespace std;

#define ll long long
ll h, w, rh, rw, p, t;

int main(){
	// freopen("test.in", "rt", stdin);
	cin >> h >> w;
	p = 1;
	for(int i = 0; i < 30; i++){
		if(p <= h && 4*p <= 5*w){ //xem p la h
			t = min(w, 5*p/4); //t la w
			if(p*t > rh*rw || (p*t == rh*rw && rh < p)){
				rh=p;
				rw=t;
			}
		}
		if(p <= w && 4*p <= 5*h){ //xem p la w
			t = min(h, 5*p/4); //t la h
			if(p*t > rh*rw || (p*t == rh*rw && rh<t)){
				rh = t;
				rw = p;
			}
		}
		p <<= 1;
	}
	cout << rh << " " << rw;
	return 0;
}
