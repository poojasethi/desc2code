#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
const int M=50000;
int n,m;
char s[1001000];
int hd[30],a[30],e[30];
unsigned char A[1<<26];
unsigned short B[1<<26];
bool cmp(const int &x,const int &y)
{
	return hd[x]>hd[y];
}
int main()
{
	scanf("%s",s+1);
	n=strlen(s+1);
	memset(hd,0,sizeof(hd));
	memset(A,0,sizeof(A));
	memset(B,0,sizeof(B));
	e[0]=1;
	for(int i=1;i<=30;++i) e[i]=e[i-1]<<1;
	s[n+1]='a'+26;
	for(int i=1;i<=n;++i)
	{
		int p=s[i]-'a',q=s[i+1]-'a',g=0;
		hd[p]=i;
		for(int j=0;j<26;++j) a[j]=j;
			sort(a,a+26,cmp);
		for(int j=0;j<26 && a[j]!=q && hd[a[j]]>0;++j)
		{
			g|=e[a[j]];
			++B[g];
			if ( B[g]==M ) ++A[g],B[g]=0;
		}
	}
	for(scanf("%d",&m);m--;)
	{
		int g=0;
		scanf("%s",s+1);
		n=strlen(s+1);
		for(int i=1;i<=n;++i) g|=e[s[i]-'a'];
		printf("%d\n",(int)A[g]*M+(int)B[g]);
	}
	return 0;
}
