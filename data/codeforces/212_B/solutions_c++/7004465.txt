#include <cstdio>
#include <cstring>
#include <map>

using namespace std;

int n,m;
char s[1000010];
char buf[100];
short c[1<<26];
int a[10010],b[10010];
int main()
{
	scanf("%s%d",s+1,&m);
	n=strlen(s+1);int cnt=0;
	for(int i=1;i<=m;i++){
		scanf("%s",buf);
		for(int j=0;buf[j];j++) a[i]|=1<<(buf[j]-'a');
		if(!c[a[i]]) c[a[i]]=++cnt;
	}
	for(int i=1;i<=n;i++)
		for(int j=i,k=0;j<=n;j++){
			if(s[j]==s[i-1]) break;
			k|=1<<(s[j]-'a');
			if(!(k&(1<<s[j+1]-'a'))) b[c[k]]++;
		}
	for(int i=1;i<=m;i++) printf("%d\n",b[c[a[i]]]);
	return 0;
}
