#include <stdio.h>
#include <iostream>
#include <vector>
#include <string.h>
using namespace std;
const int MAXQ=10000;
int q[MAXQ+10];
short map[1<<26];
int nodeN;
int ans[MAXQ+10];
int main(){
	static char s[1000010];
	int m;
	scanf("%s%d",s,&m);
	int n=strlen(s);
	for(int i=0;i<n;i++)s[i]-='a';
	s[n]=100;
	for(int i=1;i<=m;i++){
		static char tmp[100];
		scanf("%s",tmp);
		int hash=0;
		for(char*j=tmp;*j;j++)
			hash|=1<<*j-'a';
		q[i]=hash;
		if(!map[hash])map[hash]=++nodeN;
	}
	vector<int> v;
	for(int i=0;i<=n;i++){
		int pos=-1,hash=0;
		for(size_t j=0;j<v.size();j++){
			if(v[j]==s[i]){
				pos=j;
				break;
			}
			hash|=1<<v[j];
			ans[map[hash]]++;
		}
		if(pos!=-1)
			v.erase(v.begin()+pos);
		v.insert(v.begin(),s[i]);
	}
	for(int i=1;i<=m;i++)
		printf("%d\n",ans[map[q[i]]]);
	return 0;
}
