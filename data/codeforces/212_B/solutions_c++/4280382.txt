#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
char str[1000001], s[1000001];
int h[1000001];
short hash[67108865];
int to[1000001];
int n, m;
int main( )
{
	scanf("%s", str + 1); n = strlen(str + 1);
	scanf("%d", &m);
	for (int i = 1; i <= m; i ++)
	{
		scanf("%s", s);
		int len = strlen(s), get = 0;
		for (int j = 0; j < len; j ++)
		{
			int tmp = (1 << (s[j] - 'a'));
			get |= tmp;
		}
		hash[get] = i; to[i] = get;
	}
 	for (int i = 1; i <= n; i ++)
	{
		int get = 0;
		for (int j = i; j <= n; j ++)
		{
			if (str[j] == str[i - 1]) break;
			int tmp = (1 << (str[j] - 'a'));
			get |= tmp;
			if (!(get >> str[j + 1] - 'a' & 1)) h[hash[get]] ++;
		}
	}
	for (int i = 1; i <= m; i ++)
		printf("%d\n", h[hash[to[i]]]);
	return 0;
}
