#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>

using namespace std;

#define M 50005
#define N 1<<26

int n,m;
char s[1001000];
int hd[30],a[30],e[30];
unsigned char ff[N];
unsigned short f[N];

bool cmp(const int &x,const int &y)
{
	return hd[x]>hd[y];
}

void init()
{
	scanf("%s",s+1);
	n=strlen(s+1);
}

void work()
{
	e[0]=1;
	for(int i=1;i<=30;++i) 
		e[i]=e[i-1]<<1;
	s[n+1]='a'+26;
	for(int i=1;i<=n;++i)
	{
		int p=s[i]-'a',q=s[i+1]-'a',g=0;
		hd[p]=i;
		for(int j=0;j<26;++j) 
			a[j]=j;
		sort(a,a+26,cmp);
		for(int j=0;j<26&&a[j]!=q&&hd[a[j]]>0;++j)
		{
			g=g|e[a[j]];
			++f[g];
			if(f[g]==M) ++ff[g],f[g]=0;
		}
	}
}

int main()
{
	init();
	work();
	scanf("%d",&m);
	for(int i=1;i<=m;i++)
	{
		int g=0;
		scanf("%s",s+1);
		n=strlen(s+1);
		for(int i=1;i<=n;++i) 
			g=g|e[s[i]-'a'];
		printf("%d\n",(int)ff[g]*M+(int)f[g]);
	}
	return 0;
}
