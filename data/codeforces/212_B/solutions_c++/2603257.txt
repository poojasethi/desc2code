#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char s[1001111],c[111];
int n,m,a[11111],cur[26],b[26],num,ans[11111];
short pos[1<<26];

int main()
{
	scanf("%s",s);
	n=strlen(s);
	memset(pos,-1,sizeof pos);
	scanf("%d",&m);
	for (int i=0; i<m; i++)
	{
		scanf("%s",c);
		for (int j=0; c[j]!=0; j++)
			a[i]+=(1<<(c[j]-'a'));
		if (pos[a[i]]==-1)
			pos[a[i]]=num++;
	}
	memset(cur,-1,sizeof cur);
	for (int i=0; i<n; i++)
	{
		memcpy(b,cur,sizeof cur);
		sort(b,b+26);
		int x=(1<<(s[i]-'a'));
		for (int j=25; j>=0; j--)
		{
			if (s[b[j]]==s[i])
				break;
			if (pos[x]!=-1)
				ans[pos[x]]++;
			if (b[j]==-1)
				break;
			x+=(1<<(s[b[j]]-'a'));
		}
		cur[s[i]-'a']=i;
	}
	for (int i=0; i<m; i++)
		printf("%d\n",ans[pos[a[i]]]);
	return 0;
}
