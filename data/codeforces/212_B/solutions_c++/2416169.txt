#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
bool bo[50];
char s[1000001],w[50];

int n,m,x,y,q[50],qq,i,j,sw[20000],ww[20000],f[20000];
inline int se(int x)
{
	int y=lower_bound(sw+1,sw+m+1,x)-sw;
	return (sw[y]==x)?y:0;
}
int main()
{
//	freopen("1.in","r",stdin);
//	freopen("1.out","w",stdout);
	scanf("%s%d",&s,&m);  
	n=strlen(s);
	for (int i=1;i<=m;i++)
	{
		scanf("%s",w);x=0;
		for (int j=0;j<strlen(w);j++)
			x|=1<<(w[j]-'a');
		sw[i]=ww[i]=x;
	}
	sort(sw+1,sw+m+1);
	for(i=n-1;i>=0;i--)
	{
		x=1<<(s[i]-'a');
		int y=(i==0)?1<<27:1<<(s[i-1]-'a');
		for(j=0;j<26;j++)
		{
			if (s[i]==char('a'+j))
			q[j]=0,bo[j]=true;
			else
			{
				q[j]|=x;
				int yy=se(q[j]);
				if ((q[j]|y)!=q[j]&&bo[j]&&yy!=0)
				{	
				//	cout<<q[j]<<' '<<y<<' '<<i<<' '<<j<<endl;
				//	b.insert(q[j]);
				f[yy]++;
				}
			}
		}
		qq|=x;int yy=se(qq);
		if((qq|y)!=qq&&yy!=0)
		{
			f[yy]++;
		//	cout<<qq<<' '<<y<<' '<<i<<' '<<endl;
		}
	}
	for(i=1;i<=m;i++)
	printf("%d\n",f[se(ww[i])]);
}
