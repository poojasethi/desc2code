#include <cstdio>
#include <set>
#include <algorithm>
#include <cstring>
using namespace std;
char str[1000005],s1[100];
unsigned short int query[1<<26];
int ans[10005],same[10005],yet[10005],apd[100];
int main()
{
  //freopen("b.in","r",stdin);
  //freopen("b.out","w",stdout);
  int i,m,a,x,j,tail=0;
  scanf("%s",str);
  scanf("%d",&m);
  for(i=1;i<=m;i++)
    {
      scanf("%s",s1);
      x = 0;
      for(j=0;s1[j]!='\0';j++)
	x |= 1<<(s1[j]-'a');
      if(query[x] != 0)
	same[i] = query[x];
      else
	query[x] = i;
    }
  for(i=0;str[i]!='\0';i++)
    {
      a = str[i]-'a';
      for(j=1;j<=tail;j++)
	if(apd[j] == a)
	  break;
      if(j == tail+1)
	apd[++tail] = a;
      else
	{
	  while(j<tail)
	    apd[j] = apd[j+1],j++;
	  apd[tail] = a;
	}
      for(j=tail,x=0;j>0;j--)
	{
	  if(str[i+1]-'a' == apd[j]) break;
	  x |= (1<<apd[j]);
	  if(query[x])
	    ans[query[x]]++;
	}
    }
  for(i=1;i<=m;i++)
    {
      if(same[i] > 0)
	printf("%d\n",ans[same[i]]);
      else
	printf("%d\n",ans[i]);
    }
}
