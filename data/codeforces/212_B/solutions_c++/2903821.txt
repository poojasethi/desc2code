#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <map>

using namespace std;

enum {MAXN = 1000100,MAXC = 30,MAXM = 10010};
char str[MAXN],t[MAXC];
int query[MAXM];
map<int,int> cnt;

int main()
{
  scanf("%s",str);
  int n = strlen(str);
  
  int q;
  scanf("%d",&q);
  for(int i = 0;i < q;i++)
    {
      scanf("%s",t);
      int l = strlen(t),v = 0;
      for(int j = 0;j < l;j++)
	v |= 1 << (t[j] - 'a');
      query[i] = v;
      cnt[v] = 0;
    }

  for(int i = 0;i < n;i++) str[i] -= 'a';

  for(int i = 0;i < n;i++)
    for(int j = i,v = 0;j < n;j++)
      {
	if (i && str[j] == str[i-1]) break;
	v |= 1 << str[j];
	if (j + 1 == n || !((v >> str[j + 1]) & 1))
	  {
	    map<int,int>::iterator res = cnt.find(v);
	    if (res != cnt.end()) ++res->second;
	  }
      }

  for(int i = 0;i < q;i++) printf("%d\n",cnt[query[i]]);
  return 0;
}
