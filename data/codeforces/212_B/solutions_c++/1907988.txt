#include <cstdio>
#include <vector>
#include <map>
#include <cstring>
#include <iostream>
#include <algorithm>

#define pb push_back
#define fi first
#define se second
using namespace std;

const int MAXN = 1000010;

char s[MAXN],tmp[MAXN];

vector<int> query;

map<int,int> used;

int pos[30];
int ar[30];
const int inf = 0x7fffffff;

int cmp(const int &a,const int &b){
	return pos[a]>pos[b];
}

int main(){
	
	scanf(" %s",s);
	int n=strlen(s),m,mask;
	scanf(" %d",&m);
	for(int i=0;i<m;i++)
	{
		scanf(" %s",tmp);
		mask=0;
		for(int j=0;j<(int)strlen(tmp);j++)
			mask|=1<<(tmp[j]-'a');
		query.pb(mask);
		used[mask]=0;
	}
	for(int i=0;i<28;i++)
		pos[i]=-inf;
		
	for(int i=0;i<n;i++){
		pos[s[i]-'a']=i;
		for(int j=0;j<26;j++)
			ar[j]=j;
		sort(ar,ar+26,cmp);
		mask=0;
		
		for(int j=0;j<26;j++)
		{
			if(pos[ar[j]]==-inf)
				break;
			mask|=1<<ar[j];
			if(i<n-1 && (mask&(1<<(s[i+1]-'a'))))
				break;
			if(used.find(mask)!=used.end())
				used[mask]++;
		}
	}
	for(int i=0;i<m;i++)
		cout << used[query[i]] << endl;
	
	
	return 0;
	
}
