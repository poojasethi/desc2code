#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
char s[1000010];
int n,m,ex[30],no[30];
char jw[1<<26];
unsigned short ans[1<<26];
bool cmp(int x,int y){
    return ex[x]>ex[y];
}
int main(){
    //freopen("AC.in","r",stdin);
    scanf("%s",s);
    n=strlen(s);
    s[n]='a'+26;
    for(int i=0;i<n;i++){
        int p=s[i]-'a',q=s[i+1]-'a',sta=0;
        ex[p]=i+1;
        for(int j=0;j<26;j++)no[j]=j;
        sort(no,no+26,cmp);
        for(int j=0;j<26&&ex[no[j]]&&no[j]!=q;j++){
            sta|=1<<no[j];
            ans[sta]++;
            if(ans[sta]==60000){
                jw[sta]++;
                ans[sta]=0;
            }
        }
    }
    scanf("%d",&m);
    while(m--){
        scanf("%s",s);
        n=strlen(s);
        int sta=0;
        for(int i=0;i<n;i++)
            sta|=1<<(s[i]-'a');
        printf("%d\n",(int)jw[sta]*60000+ans[sta]);
    }
    return 0;
}
