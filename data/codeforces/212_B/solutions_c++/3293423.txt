#include<stdio.h>
#include<stdlib.h>
#include<string.h>
long n,i,j,l,p,t,last;
long ans[10010],a[10010];
char s[1000010],s1[30];
struct type{long pre,next;}point[30];
short hash[1<<26];
int main(){
    gets(s);
    scanf("%ld",&n);
    for (gets(s1),i=1;i<=n;i++){
        gets(s1);
        for (l=strlen(s1),p=j=0;j<l;j++)
            p|=1<<(s1[j]-'a');
        a[hash[p]=i]=p;
        }
    for (l=strlen(s),i=0;i<l;i++){
        t=s[i]-'a'+1;
        if (t!=last){
            point[point[t].pre].next=point[t].next;
            point[point[t].next].pre=point[t].pre;
            point[point[last].next=t].pre=last;
            last=t;
            }
        for (p=0,j=last;j;j=point[j].pre){
            p|=1<<(j-1);
            if ((p>>(s[i+1]-'a'))&1) break;
            ans[hash[p]]++;
            }
        }
    for (i=1;i<=n;i++)
        printf("%ld\n",ans[hash[a[i]]]);
    return 0;
}
