#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
const int maxn = 10010;
int b[maxn], c[maxn];
char s[1<<20], cmd[27];
short a[1<<26];

int main()
{
    int i, n, m, j, k;
    scanf("%s%d", s+1, &m);
    n = strlen(s+1);
    for(i = 1; i <= m; i++)
    {
        scanf("%s", cmd+1); int len = strlen(cmd+1);
        for(j = 1, k = 0; j <= len; j++) k |= 1<<(cmd[j]-'a');
        a[k] = i; b[i] = k;
    }
    for(i = 1; i <= n; i++) for(j = i, k = 0; j <= n; j++)
    {
        if(s[j] == s[i-1]) break;
        k |= 1<<(s[j]-'a');
        if(!(k>>(s[j+1]-'a') & 1)) ++c[a[k]];
    }
    for(i = 1; i <= m; i++) printf("%d\n", c[a[b[i]]]);
return 0;
}
