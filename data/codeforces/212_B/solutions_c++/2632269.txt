#include<cstdio>
#include<cstring>
#include<map>
#include<algorithm>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=1010000,mm=10100;
int n,m,i,j,k,t,a[mm],la[30],sa[30];
map<int,int>ans;
char s[mn],c[30];
bool cmp(int i,int j){return la[i]>la[j];}
int main()
{
    scanf("%s",s+1);
    n=strlen(s+1);
    scanf("%d",&m);
    fo(i,1,m)
    {
        scanf("%s",c+1);
        k=strlen(c+1);
        t=0;
        fo(j,1,k)
            t|=1<<(c[j]-97);
        ans[t]=0,a[i]=t;
    }
    fo(i,0,25)sa[i]=i;
    fo(i,1,n)
    {
        if(s[i]==s[i+1])continue;
        la[s[i]-97]=i;
        k=s[i+1]-97,t=0;
        sort(sa,sa+26,cmp);
        fo(j,0,25)
        {
            if(!la[sa[j]]||sa[j]==k)break;
            t|=1<<sa[j];
            if(t==5)
            {
                ++la[30];
            }
            if(ans.count(t))++ans[t];
        }
    }
    fo(i,1,m)printf("%d\n",ans[a[i]]);
    return 0;
}
