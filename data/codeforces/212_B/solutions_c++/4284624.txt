#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int next[1000005][30];
pair<int,int> a[30];
int seq[30000005];
char st[1000005];
char qr[35];
int n,c;
void pre()
{   int i,j,l,s;
	for (i=0;i<=25;++i)
		next[n+1][i]=n+1;
	for (i=n;i>0;--i)
	{	for (j=0;j<26;++j)
			next[i][j]=next[i+1][j];
		next[i][st[i]-'a']=i;
	}
	for (i=1;i<=n;++i)
	{	l=s=0;
		for (j=0;j<26;++j)
			if (next[i][j]!=n+1)
				a[++l]=make_pair(next[i][j],j);
		sort(&a[1],&a[l+1]);
		for (j=1;j<=l;++j)
		{	if (i>1&&st[i-1]-'a'==a[j].second)
				break;
			s+=1<<a[j].second;
			seq[++c]=s;
		}
	}
	sort(&seq[1],&seq[c+1]);
}
int query()
{	int i,s,l;
	s=0,l=strlen(qr+1);
	for (i=1;i<=l;++i)
		s+=1<<(qr[i]-'a');
	return upper_bound(&seq[1],&seq[c+1],s)-lower_bound(&seq[1],&seq[c+1],s);
}
int main()
{	int i,q;
	scanf("%s",st+1);
	n=strlen(st+1);
	pre();
	scanf("%d",&q);
	while (q--)
	{	scanf("%s",qr+1);
		printf("%d\n",query());
	}
	return 0;
}
