#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int i,j,k,l,n,m,g[12000],x,ans[12000];
char c[1200000],ch[27];
short f[1<<26];
int main(){
	scanf("%s",c+1);
	n=strlen(c+1);
	scanf("%d",&m);
	for(i=1;i<=m;++i){
		scanf("%s",ch+1);
		l=strlen(ch+1);
		x=0;
		for(j=1;j<=l;++j)x|=1<<ch[j]-97;
		f[x]=i;
		g[i]=x;
	}
	
	c[0]=c[n+1]='z'+1;
	for(i=1;i<=n;++i){
		x=0;
		for(j=i;j<=n;++j){
			if(c[j]==c[i-1])break;
			x|=1<<c[j]-97;
			if(!(x>>c[j+1]-97&1))ans[f[x]]++;
		}
	}
	
	for(i=1;i<=m;++i)printf("%d\n",ans[f[g[i]]]);
}
