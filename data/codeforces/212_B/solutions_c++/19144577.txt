#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=1000005;
const int M=10000;
char s[N];
int n,m,ha[M],q[M],ans[M],nxt[30];
int bin(int v)
{
	int l,r,mid;
	l=0,r=m-1;
	while(l<=r)
	{
		mid=(l+r)>>1;
		if(ha[mid]==v)return mid;
		if(ha[mid]<v)l=mid+1;
		else r=mid-1;
	}
	return -1;
}
int main()
{
	char c[50];
	int i,j,k,x,y,z;
	scanf("%s",s);
	n=strlen(s);
	scanf("%d",&m);
	for(i=0;i<m;i++)
	{
		k=0;
		scanf("%s",c);
		for(j=0;c[j];j++)
			k|=1<<(c[j]-'a');
		ha[i]=q[i]=k;
	}
	sort(ha,ha+m);
	memset(nxt,-1,sizeof(nxt));
	for(i=0;i<n;i++)
	{
		x=1<<(s[i]-'a');
		y=i==n-1?-1:(1<<(s[i+1]-'a'));
		for(j=0;j<26;j++)
		{
			if(nxt[j]==-1||nxt[j]==x)
			{
				for(k=j;k;k--)nxt[k]=nxt[k-1];
				nxt[0]=x;
				break;
			}
		}
		for(k=j=0;nxt[j]!=-1&&nxt[j]!=y;j++)
		{
			k|=nxt[j];
			if((z=bin(k))!=-1)
				ans[z]++;
		}
	}
	for(i=0;i<m;i++)
		printf("%d\n",ans[bin(q[i])]);
	return 0;
}
