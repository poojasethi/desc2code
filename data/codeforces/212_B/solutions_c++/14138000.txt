#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<map>
#include<bitset>
using namespace std;
const int maxn=1000000+10;
char s[maxn],f[maxn];
int n,m,k,cur;
bitset<70000000> g;
map<int,int> hash;
int ans[10010];
int main()
{
	scanf("%s",s+1),n=strlen(s+1);
	scanf("%d",&m);
	for(int i=0;i<m;i++)
	{
		scanf("%s",f),k=strlen(f);
		for(int j=0;j<k;j++)
			cur|=(1<<(f[j]-'a'));
		g[cur]=1,ans[i]=cur,cur=0;
	}
	for(int i=1;i<=n;i++,cur=0)
		for(int j=i;j<=n&&s[j]!=s[i-1];j++)
		{
			cur|=(1<<(s[j]-'a'));
			if((j==n)||((cur&(1<<(s[j+1]-'a')))==0&&g[cur]))
				hash[cur]++;
		}
	for(int i=0;i<m;i++)
		printf("%d\n",hash[ans[i]]);
	return 0;
}
