#include<cstdio>
#include<cstring>
#include<algorithm>
#define N 1000005
short p[1<<26];
int last[255],id[255],q[N],ans[N],n,m,x,L;
char s[N],a[N];
bool cmp(const int i,const int j){return last[i]>last[j];}
int main()
{
	scanf("%s",&s),n=strlen(s),scanf("%d",&m);
	for (int i=1;i<=m;i++){
		scanf("%s",&a),x=0,L=strlen(a);
		for (int j=0;j<L;j++) x|=1<<(a[j]-97);
		if (!p[x]) p[x]=q[i]=i; else q[i]=p[x];
		}
	for (int i=0;i<26;i++) id[i]=i;
	for (int i=0,k;i<n;i=k){
		last[s[i]-97]=i+1; int j=25; while (id[j]!=s[i]-97) j--; while (j) id[j]=id[j-1],j--;
		id[0]=s[i]-97; ans[p[x=1<<id[0]]]++; k=i+1; while (k<=n && s[k]==s[i]) k++;
		for (int j=1;j<26;j++){
			int now=id[j]; if (!last[now] || now==s[k]-97) break;
			x|=1<<now; ans[p[x]]++;
			}
		}
	for (int i=1;i<=m;i++) printf("%d\n",ans[q[i]]); return 0;
}
