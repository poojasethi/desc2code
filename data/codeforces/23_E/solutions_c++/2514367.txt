#include<cstdio>
#include<cstring>
#define MAXN 710

int x[2*MAXN],y[2*MAXN],f[MAXN],Peo[MAXN],F[MAXN],s[MAXN],T(2),n,b[MAXN];

struct Number
{
	int c[30],l;
	
	void Printf()
	{
		printf("%d",c[l-1]);
		for(int i=l-2;i>=0;i--)printf("%04d",c[i]);
		printf("\n");
	}
	
} p[MAXN][MAXN],Tmp;

bool Smaller(Number&a,Number&b)
{
	if(a.l!=b.l)return a.l<b.l;
	for(int i=a.l-1;i>=0;i--)
		if(a.c[i]!=b.c[i])
			return a.c[i]<b.c[i];
	return 0;
}

void Update(Number&A,Number&B,int C)
{	int i,j;
	memset(Tmp.c,0,sizeof Tmp.c);
	Tmp.l=B.l;
	for(i=0;i<Tmp.l;i++)Tmp.c[i]=B.c[i]*C;
	for(i=0;i<Tmp.l||Tmp.c[i];i++)
	if(Tmp.c[i]>=10000)
	{	Tmp.c[i+1]+=Tmp.c[i]/10000;
		Tmp.c[i]%=10000;
	}
	Tmp.l=i;
	for(;Tmp.l&&!Tmp.c[Tmp.l-1];Tmp.l--);
	if(Smaller(A,Tmp))
		A=Tmp;
}

void Update(Number&A,Number&B,Number&C)
{	int i,j;
	memset(Tmp.c,0,sizeof Tmp.c);
	Tmp.l=B.l+C.l-1;
	for(i=0;i<B.l;i++)
		for(j=0;j<C.l;j++)
			Tmp.c[i+j]+=B.c[i]*C.c[j];
	for(i=0;i<Tmp.l||Tmp.c[i];i++)
	if(Tmp.c[i]>=10)
	{	Tmp.c[i+1]+=Tmp.c[i]/10000;
		Tmp.c[i]%=10000;
	}
	Tmp.l=i;
	for(;Tmp.l&&!Tmp.c[Tmp.l-1];Tmp.l--);
	if(Smaller(A,Tmp))
		A=Tmp;
}



void IN(int a,int b){x[T]=f[a];y[T]=b;f[a]=T++;}

int main()
{
	int i,j,k;
	scanf("%d",&n);
	for(i=1;i<n;i++)
	{	scanf("%d%d",&j,&k);
		IN(j,k);
		IN(k,j);
	}
	for(b[s[i=0]=1]=j=1;i<j;i++)
		for(k=f[s[i]];k;k=x[k])
			if(!b[y[k]])
			{	b[s[j++]=y[k]]=1;
				F[y[k]]=s[i];
			}
	for(i=1;i<=n;i++){Peo[i]=1;p[i][1].l=p[i][1].c[0]=1;}
	for(i=n-1;i>=0;i--)
	{	for(j=1;j<=Peo[s[i]];j++)
			Update(p[s[i]][0],p[s[i]][j],j);
		Peo[F[s[i]]]+=Peo[s[i]];
		for(k=Peo[F[s[i]]];k>0;k--)
			for(j=0;j<k&&j<=Peo[s[i]];j++)if(p[F[s[i]]][k-j].l)
				Update(p[F[s[i]]][k],p[F[s[i]]][k-j],p[s[i]][j]);
	}
	p[1][0].Printf();
	return 0;
}
