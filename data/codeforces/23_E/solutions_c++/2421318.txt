using namespace std;
struct bigint
{
	int a[100];
	bigint(int x=0)
	{
		memset(a,0,sizeof(a));
		a[0]=1; a[1]=x;
	}
	bigint operator *(const bigint &x)
	{
		bigint b;
		int i,j,k;
		for(i=1;i<=a[0];i++)
			for(j=1;j<=x.a[0];j++)
			b.a[i+j-1]+=a[i]*x.a[j];
		k=a[0]+x.a[0]-1;
		for(i=1;i<=k;i++)
		{
			b.a[i+1]+=b.a[i]/10000;
			b.a[i]%=10000;
		}
		while (b.a[k+1]>0)
		{
		b.a[k+1]+=b.a[k]/10000;
		b.a[k]%=10000;
		k++;
		}
		b.a[0]=k; return b;
	}
	
	bool operator > (const bigint &x)
	{
		if (a[0]>x.a[0])return true;
		else
		if (a[0]==x.a[0])
		{
			for(int i=a[0];i>0;i--)
				if (a[i]>x.a[i]) return true;
				else if (a[i]<x.a[i]) return false;	
		}
		return false;
	}
	
	void print()
	{
		printf("%d",a[a[0]]);
		for (int i=a[0]-1;i>0;i--)
		printf("%04d",a[i]);
		printf("\n");
	}
};

bigint ma(bigint x,bigint y)
{
	return(x>y?x:y);
}

bigint f[706][706];

{
	f[x][1]=bigint(1);
	num[x]=1;

	for(tt=fir[x];tt!=0;tt=ne[tt])
	{
		if (gy[tt]==y) continue;
		se(gy[tt],x);
		for(i=num[x];i>0;i--)
			for(j=num[gy[tt]];j>=0;j--)
			{
			f[x][i+j]=ma(f[x][i+j],f[x][i]*f[gy[tt]][j]);
//			cout<<x<<' '<<i<<' '<<j<<endl;

//		cout<<x<<' '<<gy[tt]<<endl;

	for(i=1;i<=num[x];i++)
	f[x][0]=ma(f[x][0],f[x][i]*bigint(i));
	f[x][0]=ma(f[x][0],bigint(num[x]));
}
