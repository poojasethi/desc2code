#include <stdio.h>

int cnt[20000], msg[10];
int M[20000][10];

int main (void) {
	int n, m, k;
	int i, j, u, v;
	scanf ("%d%d%d", &n, &m, &k);
	for (i = 0; i < n; i++)
		for (j = 0; j < m; j++)
			scanf ("%d", &M[i][j]);
	for (i = 0; i < k; i++) {
		scanf ("%d%d", &u, &v);
		u--; v--;
		cnt[u]--;
		msg[v]++;
	}
	for (i = 0; i < n; i++) {
		for (j = 0; j < m; j++) {
			cnt[i] += M[i][j]?msg[j]:0;
		}
		printf ("%d ", cnt[i]);
	}
}
