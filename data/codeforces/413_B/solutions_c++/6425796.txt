#include<cstdio>
#include<cstring>
int main(){
int n,m,k,a[20010][12],h1[20010],h2[12],x,y,c;
memset(h1,0,sizeof(h1));
memset(h2,0,sizeof(h2));
scanf("%d %d %d",&n,&m,&k);
for(int i=1;i<=n;i++){
for(int j=1;j<=m;j++){
	scanf("%d",&a[i][j]);
}
}
for(int i=1;i<=k;i++){
	scanf("%d %d",&x,&y);
	h1[x]++;
	h2[y]++;
}
for(int i=1;i<=n;i++){c=0;
	for(int j=1;j<=m;j++){
		if(a[i][j]==1){
			c+=h2[j];
		}
	}
	c=c-h1[i];
	printf("%d ",c);
}
return 0;
}
