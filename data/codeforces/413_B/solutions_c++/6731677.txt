#include<cstdio>

const int N=20050,M=12;
int map[N][M];
int x[N],y[M];

int main(){
  int n,m,k;
  scanf("%d%d%d",&n,&m,&k);
  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
      scanf("%d",&map[i][j]);
  for(int i=1;i<=k;i++){
    int a,b;scanf("%d%d",&a,&b);
    x[a]--;y[b]++;
  }
  for(int i=1;i<=m;i++)
    for(int j=1;j<=n;j++)
      x[j]+=map[j][i]*y[i];
  for(int i=1;i<=n;i++)
    printf("%d ",x[i]);        
}
