#include<cstdio>
const int nn=20001;
int a[nn][11];
int c[nn];
int d[nn];
int main()
{
    int i,j,n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            scanf("%d",&a[i][j]);
    for(i=1;i<=k;i++)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        c[y]++;
        d[x]--;
    }
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
            d[i]+=a[i][j]*c[j];
        printf("%d ",d[i]);
    }
}
