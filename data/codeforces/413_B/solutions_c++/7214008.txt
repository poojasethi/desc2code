#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
int n,m,k,s[15],c[20005],g[20005][15];
int main()
{
    scanf("%d%d%d",&n,&m,&k); int x,y;
    for(int i=1;i<=n;++i)
        for(int j=1;j<=m;++j) scanf("%d",&g[i][j]);
    while(k--) {scanf("%d%d",&x,&y); ++s[y]; --c[x];}
    for(int i=1;i<=n;++i)
        for(int j=1;j<=m;++j) c[i]+=g[i][j]*s[j];
    for(int i=1;i<=n;++i) printf("%d ",c[i]);
    printf("\n"); return 0;
}
