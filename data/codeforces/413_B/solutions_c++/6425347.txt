#include<cstdio>
#include<cstdlib>
using namespace std;
int n,m,k,g[30000][20],a[20],ans[30000];
int main()
{
	int i,j,r1,r2;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;i++) for(j=1;j<=m;j++) scanf("%d",&g[i][j]);
	for(i=1;i<=k;i++)
	{
		scanf("%d%d",&r1,&r2);
		a[r2]++;
		if(g[r1][r2]==1) ans[r1]--;
	}
	for(i=1;i<=n;i++) for(j=1;j<=m;j++) if(g[i][j])ans[i]+=a[j];
	for(i=1;i<=n;i++) printf("%d ",ans[i]);
	return 0;
}
