#include<cstdio>
#define N 20005
#define M 15
using namespace std;
int n,m,k,i,j,x,y;
int a[N][M],f[N],cnt[M];
int main()
{
  scanf("%d%d%d",&n,&m,&k);
  for (i=1;i<=n;i++)
    for (j=1;j<=m;j++)
      scanf("%d",&a[i][j]);
  for (i=1;i<=k;i++)
  {
    scanf("%d%d",&x,&y);
    cnt[y]++;if (a[x][y]) f[x]--;
  }
  for (i=1;i<=n;i++)
    for (j=1;j<=m;j++)
      if (a[i][j]) f[i]+=cnt[j];
  for (i=1;i<=n;i++)
    printf("%d ",f[i]);
  return 0;
}
