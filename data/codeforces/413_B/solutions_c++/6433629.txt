#include<stdio.h>

#define N (int)1e5+10

int arr[N][11];
int A[N];
int B[N][11];
int main(){
	int n,m,k;
	int i,j,a,b,top=0;

	scanf("%d %d %d",&n,&m,&k);

	for(i=1 ; i<=n ; i++)
		for(j=1 ; j<=m ; j++)
			scanf("%d",&arr[i][j]);

	for(i=0 ; i<k ; i++){
		scanf("%d %d",&a,&b);
		A[b]++;
		B[a][b]++;
	}

	for(i=1 ; i<=n ; i++){
		top=0;
		for(j=1 ; j<=m ; j++)
			if(arr[i][j])
				top+=A[j]-B[i][j];
		printf("%d ",top);
	}

}
