#include<stdio.h>
int n,m,k;
int num[22222][11];
int l[11],r[22222];
int main(){
	int i,j;
	while(~scanf("%d %d %d",&n,&m,&k)){
		for(i=1;i<=n;i++){
			for(j=1;j<=m;j++){
				scanf("%d",&num[i][j]);
			} 
		}
		int x,y;
		for(i=0;i<k;i++){
			scanf("%d %d",&x,&y);
			r[x]++;
			l[y]++;
		}
		for(i=1;i<=n;i++){
			__int64 res=0; 
			for(j=1;j<=m;j++){
				if(num[i][j])res+=l[j];
			}
			if(i!=1)printf(" ");
			printf("%d",res-r[i]);	
		}
		printf("\n");
		
		
	}
	return 0;
}
