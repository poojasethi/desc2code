    #include <iostream>
using namespace std;
int n,m,k;
int a[20001][11];
int dem[20001][11];
int dem1[11];
void doc(){
cin>>n>>m>>k;
for (int i=1;i<=n;i++){
for (int j=1;j<=m;j++) cin>>a[i][j];
}
for (int i=1;i<=k;i++){
int x,y;
cin>>x>>y;
dem[x][y]++;
dem1[y]++;
}
}
void xuli(){
for (int i=1;i<=n;i++){
int ans=0;
for (int j=1;j<=m;j++){
if (a[i][j]==1) ans+=(dem1[j]-dem[i][j]);
}
cout<<ans<<" ";
}
}
int main(){
doc();
xuli();
}
