#include<iostream>
#include<vector>
#include<stdio.h>
using namespace std;
int n,m,k,mark[100002][12],x,y,i,j;
int q[12],answ[100002];
int main()
{
	scanf("%d%d%d", &n, &m, &k);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			scanf("%d", &mark[i][j]);
	for(i=1;i<=k;i++)
	{
		scanf("%d%d", &x, &y);
		q[y]++;
		answ[x]--;
	}
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			if(mark[i][j])
				answ[i]+=q[j];
		}
	for(i=1;i<=n;i++)
		printf("%d ", answ[i]);
	cout<<endl;
	return 0;
}
