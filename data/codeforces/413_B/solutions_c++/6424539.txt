#include<iostream>
using namespace std;
int ch[10],se[20000];
bool sk[20000][10];
int main() {
	int n,m,k,a,b;
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin>>sk[i][j];
	for(int i=0;i<k;i++) {
		cin>>a>>b;
		se[a-1]++;
		ch[b-1]++;
	}
	for(int i=0;i<n;i++) {
		int ans=0;
		for(int j=0;j<m;j++)
			ans+=sk[i][j]?ch[j]:0;
		cout<<ans-se[i]<<" ";
	}
	return 0;
}
