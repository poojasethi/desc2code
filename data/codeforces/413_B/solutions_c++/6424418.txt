#include <cstdio>
using namespace std;

int n,m,k,arr[20005][15],x,y,ans[20005],tmp[15];

int main(){
 scanf("%d %d %d",&n,&m,&k);
 for(int i=1;i<=n;i++)
  for(int j=1;j<=m;j++)
   scanf("%d",&arr[i][j]);
 while(k--){
  scanf("%d %d",&x,&y);
  tmp[y]++;
  ans[x]--;
 }
 for(int i=1;i<=n;i++)
  for(int j=1;j<=m;j++)
   if(arr[i][j]) ans[i]+=tmp[j];
 for(int i=1;i<=n;i++){
  printf("%d",ans[i]);
  if(i<n) printf(" ");
  else printf("\n");
 }
 return 0;
}
