#include<iostream>
using namespace std;
int n,m,k;
int a[20010][10];
int p[20];
int d[20010];
int main()
{
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		cin>>a[i][j];
	for(int i=0;i<k;i++)
	{
		int u,v;
		cin>>u>>v;
		u--;
		v--;
		d[u]--;
		p[v]++;
	}
	for(int i=0;i<m;i++)
	{
		for(int j=0;j<n;j++)
		if(a[j][i])
			d[j]+=p[i];
	}
	for(int i=0;i<n;i++)
		cout<<d[i]<<" ";
	cout<<endl;
}
