#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,K,x,y;
int a[20010][20],an[20010],b[20];
int main() {
	scanf("%d%d%d",&n,&m,&K);
	For(i,1,n)For(j,1,m) scanf("%d",&a[i][j]);
	For(i,1,K) {
		scanf("%d%d",&x,&y);
		an[x]--;
		b[y]++;
	}
	For(i,1,n)For(j,1,m) if (a[i][j]) an[i]+=b[j];
	For(i,1,n) printf("%d ",an[i]);
	printf("\n");
	//for(;;);
	return 0;
}
