#include<iostream>
#include<algorithm>
using namespace std;
int a[20004][13],ans[15],c[20004][13];
int main()
{
  int i,j,n,m,k,x,y,temp=0;
  cin>>n>>m>>k;
  for(i=1;i<=n;i++) {
    for(j=1;j<=m;j++) cin>>a[i][j],c[i][j]=0;
  }
  for(i=1;i<=k;i++) {
    cin>>x>>y;
    ans[y]++;
    c[x][y]++;
  }
  for(i=1;i<=n;i++) {
    temp=0;
    for(j=1;j<=m;j++) if(a[i][j]==1) temp+=ans[j]-c[i][j];
    cout<<temp<<" ";
  }
//  system("pause");
  return 0;
}
