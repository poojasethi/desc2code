#include<stdio.h>
int arr[200005][11]={{0}};
int main()
{
	int n,m,k,i,j,x,y;
	scanf("%d%d%d",&n,&m,&k);
	int a[n+1];
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			scanf("%d",&arr[i][j]);
		}
		a[i]=0;
	}
	int b[11]={0};
	
	for(i=1;i<=k;i++)
	{
		scanf("%d%d",&x,&y);
		a[x]++;
		b[y]++;
	}
	for(i=1;i<=n;i++)
	{
		int sum=0;
		for(j=1;j<=m;j++)
		{
		if(arr[i][j]==1)
			sum+=b[j];
		}
		
		printf("%d\t",sum-a[i]);
	}
	return 0;
}
