#include<cstdio>
using namespace std;

int i,j,k,n,m,x,y;
int a[21000][20],b[20],num[25000];
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;i++)
	  for(j=1;j<=m;j++)  scanf("%d",&a[i][j]);
	for(i=1;i<=k;i++)
	{
		scanf("%d%d",&x,&y);
		num[x]=num[x]-1;
		b[y]++;
	}
	for(i=1;i<=n;i++)  for(j=1;j<=m;j++)  if(a[i][j])  num[i]+=b[j];
	for(i=1;i<=n;i++)  printf("%d ",num[i]);
	return 0;
}
