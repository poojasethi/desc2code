#include<stdio.h>

int main()
{
	int n,m,k;
	scanf("%d %d %d",&n,&m,&k);
	int i,j;
	int a[n+1][m+1];
	int b[m+1];
	int c[n+1];
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			scanf("%d",&a[i][j]);
			b[j]=0;
		}
		c[i]=0;
	}
	int p,q;
	for(i=1;i<=k;i++)
	{
		scanf("%d %d",&p,&q);
		b[q]++;
		c[p]--;
	}
	for(i=1;i<=m;i++)
	{
		for(j=1;j<=n;j++)
		{
			c[j]=c[j]+b[i]*a[j][i];
		}
	}
	for(i=1;i<=n;i++)
		printf("%d ",c[i]);
	printf("\n");
	return 0;
}
