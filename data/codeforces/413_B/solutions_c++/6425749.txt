#include <iostream>
#include <algorithm>
using namespace std;
int a[20000][10], c[10], d[20000];
int main()
{
	int n, m, k, i, j, x, y, s;
	cin>>n>>m>>k;
	for (i=0; i<n; i++) {
		for (j=0; j<m; j++) cin>>a[i][j];
	}
	for (i=0; i<k; i++) {
		cin>>x>>y;
		x--; y--;
		d[x]++;
		c[y]++;
	}
	for (i=0; i<n; i++) {
		s = 0;
		for (j=0; j<m; j++) s+=c[j]*a[i][j];
		cout<<s-d[i]<<" ";
	}
	cout<<endl;
	return 0;
}
