#include<iostream>

using namespace std;

int main()
{
	long long n, m, k, i, j, c[20]={}, send[20010][20]={}, x, y, ans;
	bool chat[20010][20]={};
	cin>>n>>m>>k;
	for(i=1; i<=n; i++)
	{
		for(j=1; j<=m; j++)
		{
			cin>>chat[i][j];
		}
	}
	for(i=0; i<k; i++)
	{
		cin>>x>>y;
		c[y]++;
		send[x][y]++;	
	}
	for(i=1; i<=n; i++)
	{
		ans=0;
		for(j=1; j<=m; j++)
		{
			if(chat[i][j])
				ans+=c[j]-send[i][j];	
		}
		cout<<ans<<" ";	
	}
}
