#include <iostream>
using namespace std;
int a[100100][15],b[100100],c[15],n,m,k,i,j,x,y,ans;
int main()
{
	cin>>n>>m>>k;
	for (i=1; i<=n; i++)
	    for (j=1; j<=m; j++)
	        cin>>a[i][j];
	for (i=0; i<k; i++)
	{
		cin>>x>>y;
		b[x]++;
		c[y]++;
	}
	for (i=1; i<=n; i++)
	{
	    ans=-b[i];
	    for (j=1; j<=m; j++)
	        ans+=a[i][j]*c[j];
	    cout<<ans<<" ";    
	}
	cout<<endl;
	return 0;
}
