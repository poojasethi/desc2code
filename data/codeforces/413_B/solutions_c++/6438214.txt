#include<stdio.h>
int n,m,k,i,j,av[20005][15]={0},ta[15]={0},tb[20005]={0},xt,yt;
int main()
{
    scanf("%d %d %d",&n,&m,&k);
    for(i=1;i<=n;i++) for(j=1;j<=m;j++) scanf("%d",&av[i][j]);
    while(k--)
    {
        scanf("%d %d",&xt,&yt);
        ta[yt]++;
        tb[xt]--;
    }
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
            tb[i]+=(av[i][j]*ta[j]);
        printf("%d%c",tb[i],i==n?'\n':' ');
    }
    return 0;
}
