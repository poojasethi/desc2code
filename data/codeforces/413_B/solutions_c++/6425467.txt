#include <iostream>

using namespace std;

int a[20001][11]={0}, sCount[20001]={0}, dCount[11]={0}, res[20001]={0};

int main(){
	int n, m, k;
	
	cin >> n >> m >> k;
	
	int i, j;
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			cin >> a[i][j];
		}
	}
	
	for(i=0;i<k;i++){
		int s, d;
		cin >> s >> d;
		sCount[s]++;
		dCount[d]++;
	}
	
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			res[i]+=(a[i][j]*dCount[j]);
		}
		res[i]-=sCount[i];
		cout << res[i] << " ";
	}

	return 0;
}
