#include<iostream>
using namespace std;
long long n,m,k,i,j,a,b,s,num[15],snd[20005];
bool map[20005][15];
int main()
{
    cin>>n>>m>>k;
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
            cin>>map[i][j];
    for(i=0;i<k;i++)
    {
        cin>>a>>b;
        snd[a-1]++;
        num[b-1]++;
    }
    for(i=0;i<n;i++)
    {
        s=0;
        for(j=0;j<m;j++)
            if(map[i][j])
                s+=num[j];
        s-=snd[i];
        cout<<s<<' ';
    }
    return 0;
}
