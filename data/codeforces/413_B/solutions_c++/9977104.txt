#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int n,m,k;
int rea[100010];
int hs,ht;
int g[100010][20];
int logs[20];
int main()
{
	cin>>n>>m>>k;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			cin>>g[i][j];
	for(int i=1;i<=k;i++)
	{
		int x,y;
		cin>>x>>y;
		rea[x]--;
		logs[y]++;
	}
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			if(g[i][j]) rea[i]+=logs[j];
	for(int i=1;i<=n;i++) cout<<rea[i]<<" ";
	cout<<endl;
	return 0;
}
