#include<stdio.h>

char arr[20005][15];
int k,h[2005],y,x,n,m,sayac[20005],i,j;
int main(){
	scanf("%d %d %d",&n,&m,&k);
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			scanf(" %c",&arr[i][j]);
		}
	}
	for(i=0;i<k;i++){
		scanf("%d %d",&x,&y);
		x--,y--;
		sayac[x]--;
		h[y]++;
	}
	for(i=0;i<n;i++){
		
		for(j=0;j<m;j++){
			if(arr[i][j]=='1'&&h[j])
				sayac[i]+=h[j];
		}

	}
	for(i=0;i<n;i++)
		printf("%d ",sayac[i]);
		

	
	return 0;
}
