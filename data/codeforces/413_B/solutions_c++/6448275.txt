#include<iostream>
using namespace std;
long long n,m,k,i,j,a,b,nu[15],s[20005];
bool ma[20005][15];
int main()
{cin>>n>>m>>k;
for(i=0;i<n;i++)for(j=0;j<m;j++)cin>>ma[i][j];
for(i=0;i<k;i++){cin>>a>>b;s[a-1]--;nu[b-1]++;}
for(i=0;i<n;i++){for(j=0;j<m;j++)if(ma[i][j])s[i]+=nu[j];cout<<s[i]<<' ';}
return 0;}
