#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int sum[100000],mark[100000][11];
int f[100000][11];

int main(){
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	for (int i=1; i<=n; ++i)
		for (int j=1; j<=m; ++j)
			scanf("%d",&f[i][j]);
	for (;k;--k){
		int x,y;
		scanf("%d%d",&x,&y);
		++sum[y];
		++mark[x][y];
	}
	for (int i=1; i<=n; ++i){
		int k=0;
		for (int j=1; j<=m; ++j)
			if (f[i][j]) k+=sum[j]-mark[i][j];
		printf("%d%c",k," \n"[i==n]);
	}
	return 0;
}
