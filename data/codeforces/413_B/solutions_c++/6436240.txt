#include <iostream>

using namespace std;

const int MAXN = 30000;
bool a[MAXN][11];
int c[MAXN], b[11];

int main() {
	int n, m, k;
	cin >> n >> m >> k;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> a[i][j];
	for (int i = 0; i < k; i++) {
		int x, y;
		cin >> x >> y;
		x--; y--;
		c[x]--;
		b[y]++;
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++)
			c[i] += a[i][j] * b[j];
		cout << c[i] << " ";
	}
	return 0;
}
