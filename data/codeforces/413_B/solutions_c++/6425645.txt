#include<cstdio>
int num[20004][10];
int chat[10]={0};
int send[20004]={0};
int main () {
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			scanf("%d",&num[i][j]);
	for(int i=0;i<k;i++) {
		int a,b;
		scanf("%d %d",&a,&b);
		send[a-1]++;
		chat[b-1]++;
	}
	for(int i=0;i<n;i++) {
		int sum=0;
		for(int j=0;j<m;j++)
			if(num[i][j]==1) sum+=chat[j];
		sum-=send[i];
		printf("%d ",sum);
	}
	return 0;
}
