#include <bits/stdc++.h>
using namespace std;
int n,m,k,i,j,sent[200010],rec[15],part[200010][15],a,b;

int main() {
    cin>>n>>m>>k;
    for(i=1;i<=n;i++) for(j=1;j<=m;j++) scanf("%d",&part[i][j]);
    while (k--){
        scanf("%d%d",&a,&b);
        rec[b]++; sent[a]--;
    }
    for(i=1;i<=n;i++) for(j=1;j<=m;j++) if (part[i][j]) sent[i]+=rec[j];
    for(i=1;i<=n;i++) printf("%d ",sent[i]);
    return 0;
}
