#define rep(i,k,n) for(int i=k;i<n;i++)
#define mp make_pair

const int iinf = 2147483647;

using namespace std;
char input[limit];
int m,n,l,d;
vector<int> perm_comp(const vector<int> & b,const vector<int> & a){
    vector<int> ans;
    rep(i,0,a.size()) ans.pb(b[a[i]]);
    return ans;
}
vector<int> perm_pow(vector<int> a,int k){
    vector<int> ans;
    rep(i,0,n) ans.pb(i);
    while(k){
        if(k%2) ans=perm_comp(ans,a);
        a=perm_comp(a,a);
        k/=2;
    }
    return ans;
}
int main(){
    scanf("%s%d",input,&m);
    n=strlen(input);
    vector<int> rot(n),a(n);
    rep(i,0,m){
        scanf("%d%d",&l,&d);
        int ind=0;
        rep(j,0,d) rep(k,0,l/d+1) if(k*d+j<l){
            a[k*d+j]=(ind+n-1)%n;
            ind++;
        }
        rep(j,l,n) a[j]=(j+n-1)%n;
        rep(j,0,n) rot[j]=(j+n-l+1)%n;
        a=perm_comp(rot,perm_pow(a,n-l+1));
        char next[n];
        rep(i,0,n) next[a[i]]=input[i];
        rep(i,0,n) input[i]=next[i];
        printf("%s\n",input);
    }
    return 0;
}
