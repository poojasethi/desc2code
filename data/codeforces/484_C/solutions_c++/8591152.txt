#include <bits/stdc++.h>

#define mp make_pair
#define st first
#define nd second
#define pb push_back

using namespace std;

typedef long long ll;
typedef pair<int,int> pii;
typedef pair< ll,ll > pll;

const int inf = 1000000000;

int n,m;
string s;

struct perm {
	int* p;

	perm() {
		p = (int*) malloc(n * sizeof(int));
	}

	void shift(){
		int t = p[0];

		for (int i = 1; i < n; i++)
			p[i-1] = p[i];
		p[n-1] = t;
	}

	int& operator[] (int i) {
		return p[i];
	}

};

perm operator* (perm &a, perm &b) {
	perm res;
	for (int i = 0; i < n; i++)
		res[i] = a[b[i]];
	return res;
}

perm operator^ (perm &a,int x) {
	if (x == 1)
		return a;

	perm t = a ^ (x >> 1);

	t = t * t;

	if (x & 1)
		t = t * a;

	return t;
}


int main()
{
	int k,d;

	cin >> s >> m;

	n = s.size();
	
	perm p;

	while (m--) {	
		scanf("%d %d",&k,&d);

		int it = 0;

		for (int i = 0; i < d; i++) {
			for (int j = i; j < k; j += d)
				p[it++] = j;
		}

		for (int i = k; i < n; i++)
			p[it++] = i;
	
		p.shift();
		
		p = p ^ (n-k+1);

		string temp;

		for (int i = 0; i < n; i++)
			temp += s[p[i]];
		
		s = temp.substr(k-1,n-k+1) + temp.substr(0,k-1);

		puts(s.c_str());
	}
	

}
