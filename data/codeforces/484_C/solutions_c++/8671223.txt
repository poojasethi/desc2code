#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;

char s[1000001];
int n,m,d,K,id[1000001],now,key;
int tmp1[1000001],tmp2[1000001],f[1000001],tmp[1000001];

void work(int ans[],int b,int f[])
{
	for (;b;b>>=1)
	{
		if (b&1) 
		{
			for (int i=1;i<=n;i++) tmp[i]=ans[f[i]];
			for (int i=1;i<=n;i++) ans[i]=tmp[i];
		}
		for (int i=1;i<=n;i++) tmp[i]=f[f[i]];
		for (int i=1;i<=n;i++) f[i]=tmp[i];
	}
}

int main()
{
	scanf("%s",s+1);
	n=strlen(s+1);
	for (int i=1;i<=n;i++) id[i]=i;
	scanf("%d",&m);
	for (int i=1;i<=m;i++)
	{
		scanf("%d%d",&K,&d);
		now=1; key=1;
		for (int i=1;i<=K;i++)
		{
			tmp2[i]=key;
			key+=d;
			if (key>K) key=++now;
		}
		for (int i=K+1;i<=n;i++) tmp2[i]=i;
		for (int i=1;i<=n;i++) tmp1[i]=i%n+1;
		for (int i=1;i<=n;i++) f[i]=i;
		for (int i=1;i<=n;i++) tmp[i]=f[tmp2[i]];
		for (int i=1;i<=n;i++) f[i]=tmp[tmp1[i]];
		work(id,n-K+1,f);
		work(id,K-1,tmp1);
		for (int i=1;i<=n;i++) putchar(s[id[i]]);
		puts("");
	}
}
