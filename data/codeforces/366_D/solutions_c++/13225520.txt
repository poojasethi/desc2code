#include <bits/stdc++.h>
using namespace std;
#define ll long long int

struct node
{
    ll a, b, l, r;
    bool operator <(const node&B) const {return r > B.r;}
}a[3005];

ll n, m, f[1005];

typedef struct node node;

ll find(ll x)
{
    return (x==f[x] ? x : f[x] = find(f[x]));
}

int main() {

    cin>>n>>m;
    for(ll i = 1; i <= m; i++) cin>>a[i].a>>a[i].b>>a[i].l>>a[i].r;
    sort(a+1, a+m+1);
    ll res = 0;
    for(ll i = 1; i <= m; i++)
    {
        for(ll j = 1; j <= n; j++) f[j] = j;
        for(ll j = 1; j <= m; j++)
        {
            if(a[j].l <= a[i].l)
            {
                ll x = find(a[j].a), y = find(a[j].b);
                if(x != y) f[x] = y;
                if(find(1) == find(n)) {res = max(res, a[j].r - a[i].l + 1); break;}
            }
        }
    }
    if(!res)
        cout<<"Nice work, Dima!"<<endl;
    else
        cout<<res<<endl;
    return 0;
}
