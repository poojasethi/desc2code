#include <cstdio>
#include <algorithm>
using namespace std;
const int N=3100;
int n,m,L,R,M,i,x[N],y[N],l[N],r[N],f[N];
int get(int x) {return (f[x]==x)?x:f[x]=get(f[x]);}
bool check(int L,int R) {
    for (int i=1;i<=n;i++) f[i]=i;
    for (int i=0;i<m;i++)
        if ((l[i]<=L)&&(R<=r[i])) {
            int u=get(x[i]),v=get(y[i]);
            f[u]=v;
        }
    return get(f[1])==get(f[n]);
}
int main() {
    scanf("%d%d",&n,&m);
    for (i=0;i<m;i++) {
        scanf("%d%d%d%d",x+i,y+i,l+i,r+i);
        R=max(R,r[i]-l[i]+1);
    }
    L=0;
    while (L<R) {
        M=(L+R)/2+1;
        for (i=0;i<m;i++)
            if (check(l[i],l[i]+M-1)) break;
        if (i<m) L=M;
        else R=M-1;
    }
    if (L==0) printf("Nice work, Dima!\n");
    else printf("%d\n",L);
    return 0;
}
