};
int fa[maxn];

int f(int x){return x==fa[x]?x:fa[x]=f(fa[x]);}
int main()
{

    for(int i=0;i<m;i++)cin>>a[i].a>>a[i].b>>a[i].l>>a[i].r;
    sort(a,a+m);
    int ans=0;
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<=n;j++)fa[j]=j;
        for(int j=0;j<m;j++)if(a[i].l>=a[j].l)
        {
            int x=f(a[j].a);
            int y=f(a[j].b);
            if(x!=y)fa[x]=fa[y];
            if(f(1)==f(n)){ans=max(ans,a[j].r-a[i].l+1);break;}
        }
    }
    if(ans==0)puts("Nice work, Dima!");
