#include<cstdio>
#define N 5005
using namespace std;
int f[N],x[N],y[N],a[N],b[N];
int L,R,ans,n,m,i,ma;
int get(int u){return f[u]==u?u:f[u]=get(f[u]);}
inline int check(int R)
{
  for (int i=1;i<=n;i++) f[i]=i;
  for (int i=1;i<=m;i++)
    if (a[i]<=L&&b[i]>=R) f[get(x[i])]=get(y[i]);
  return (get(1)==get(n));
}
int erfen(int l,int r)
{
  if (l==r) return l;int mid=((l+r)>>1)+1;
  if (check(mid)) 
    return erfen(mid,r);
  return erfen(l,mid-1);
}
int main()
{
  scanf("%d%d",&n,&m);
  for (i=1;i<=m;i++)
    scanf("%d%d%d%d",&x[i],&y[i],&a[i],&b[i]),ma=(b[i]>ma)?b[i]:ma;
  for (i=1;i<=m;i++)
  {
    L=a[i];R=erfen(L,ma);
    if (R-L+1>ans&&check(R)) ans=R-L+1;
    L=b[i];R=erfen(L,ma);
    if (R-L+1>ans&&check(R)) ans=R-L+1;
  }
  if (!ans) puts("Nice work, Dima!");else printf("%d\n",ans);
}
