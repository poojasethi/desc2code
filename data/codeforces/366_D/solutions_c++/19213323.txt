#define LL long long
#define INF 0x7f7f7f7f
#define PI acos(-1.0)
using namespace std;

const int maxn = 3e3 + 5;
struct Edge {
	int u, v, l, r;
	bool operator < (const Edge &t) const {
		return r < t.r;
	}
} e[maxn];
int f[maxn];

int find(int x) {
	return f[x] == x ? x : f[x] = find(f[x]);
}

int main()
{
	int n, m;
	while (~scanf("%d%d", &n, &m)) {
		for (int i = 0; i < m; i++) {
			scanf("%d%d%d%d", &e[i].u, &e[i].v, &e[i].l, &e[i].r);
		}
		sort(e, e+m);
		
		int ans = 0;
		for (int i = 0; i < m; i++) {
			for (int j = 1; j <= n; j++) f[j] = j;
			for (int p = m-1; p >= 0; p--) {
				if (e[p].l > e[i].l)	continue;
				if (e[p].r < e[i].l)	break;
				if (find(e[p].u) != find(e[p].v))	f[f[e[p].u]] = f[e[p].v];
				if (find(1) == find(n))	{
					ans = max(ans, e[p].r-e[i].l+1);
					break;
				}
			}
		}
		if (ans == 0)	puts("Nice work, Dima!");
		else printf("%d\n", ans);
	}
	return 0;
}
