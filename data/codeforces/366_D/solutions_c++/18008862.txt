#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
typedef unsigned long long int ull;
#define inf 1000000000
struct node{
	int a,b,l,r;
}arr[40000];
int par[40000];
bool cmp(node a,node b)
{
	return a.r>b.r;
}
int find(int x)
{
	return (par[x]==x)? x : par[x] = find(par[x]);
}
int n,m;
int ans;
void solve(int x)
{
	int i,a,b;
	for(i=1;i<=n;i++)
	par[i] = i;
	for(i=1;i<=m;i++)
	{
		if(arr[i].l>x)
		continue;
		
		a = find(arr[i].a);
		b = find(arr[i].b);
		//cout<<"ok "<<a<<" "<<b<<endl;
		par[a] = b;
		if(find(1)==find(n))
		{
			ans = max(ans,arr[i].r-x+1);
		}
		//cout<<"ok "<<a<<" "<<b<<endl;
	}
}
int main()
{ 
   scanf("%d %d",&n,&m);
   int i;
   for(i=1;i<=m;i++)
   scanf("%d %d %d %d",&arr[i].a,&arr[i].b,&arr[i].l,&arr[i].r);
   ans = 0;
   sort(arr+1,arr+1+m,cmp);
   for(i=1;i<=m;i++)
   {
   	  //cout<<i<<endl;
   	   solve(arr[i].l);
   }
   if(ans==0)
   printf("Nice work, Dima!\n");
   else
   printf("%d\n",ans);
}
