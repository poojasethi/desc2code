#include <cstdio>
#include <algorithm>
using namespace std;

struct st
{
	int a,b,l,r;
	bool operator <(const st&B) const {return r>B.r;}
} a[3005];
int N,M,f[1005];

int find(int x) {return x==f[x]?x:f[x]=find(f[x]);}

int main()
{
	scanf("%d%d",&N,&M);
	for (int i=1; i<=M; i++) scanf("%d%d%d%d",&a[i].a,&a[i].b,&a[i].l,&a[i].r);
	sort(a+1,a+M+1);
	int ans=0;
	for (int i=1; i<=M; i++)
	{
		for (int j=1; j<=N; j++) f[j]=j;
		for (int j=1,x,y; j<=M; j++) if (a[j].l<=a[i].l)
		{
			x=find(a[j].a),y=find(a[j].b);
			if (x!=y) f[x]=y;
			if (find(1)==find(N)) {ans=max(ans,a[j].r-a[i].l+1); break;}
		}
	}
	if (!ans) puts("Nice work, Dima!"); else printf("%d\n",ans);
}
