#include <bits/stdc++.h>
using namespace std;

typedef struct given{
	int a,b,l,r;	
}node;

node inp[3000 + 5];
int path[3000 + 5];


bool waytosort(node a,node b){
	return a.r > b.r;
}

int find(int x){
	return x == path[x] ? x : path[x] = find(path[x]); 
}

int main(){
	int n,m;
	cin>>n>>m;

	for(int i = 1;i <= m;i++){
		cin>>inp[i].a>>inp[i].b>>inp[i].l>>inp[i].r;
	}
	sort(inp + 1,inp + m + 1,waytosort);
	int ans = -1;
	for(int i = 1;i <= m;i++){
		for(int k = 1;k <= n;k++)
			path[k] = k;
		for(int j = 1;j <= m;j++){
			if(inp[j].l <= inp[i].l){
				int p1 = find(inp[j].a);
				int p2 = find(inp[j].b);
				if(p1 != p2)
					path[p1] = p2;
				if(find(1) == find(n)){
					ans = max(ans,inp[j].r - inp[i].l + 1);
					break;
				}
			}
		}
	}
	if(ans == -1 || ans == 0)
		cout<<"Nice work, Dima!";
	else 
		cout<<ans;
}
