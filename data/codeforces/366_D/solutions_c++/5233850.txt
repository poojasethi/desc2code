#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<vector>
#include<set>

using namespace std;

#define sf scanf
#define pf printf
#define rep(i,n) for(int (i)=0;(i)<(n);(i)++)
#define nMax 5000

struct T1{
	int u,v;
	int l,r;
	void read() { sf("%d%d%d%d",&u,&v,&l,&r);u-=1,v-=1; }
} a[nMax];

bool operator < (const T1& a,const T1& b) {
	return a.r > b.r;
}

int f[nMax];
int inline find(int x) {
	return f[x] == x ? x : f[x]=find(f[x]);
}
 
int n,m;

int main() {
#ifndef ONLINE_JUDGE
	freopen("in.txt","r",stdin);
#endif

	while(~sf("%d%d",&n,&m)){
		rep(i,m) a[i].read();
		sort(a,a+m);
		int ans=0,x,y;
		rep(i,m) {
			rep(j,n) f[j]=j;
			rep(j,m) if(a[j].l <= a[i].l) {
				x = find(a[j].u), y = find(a[j].v);
				if(x != y) f[x]=y;
				if(find(0) == find(n-1)) {
					ans = max(ans,a[j].r-a[i].l+1);
					break;
				}
			}
		}
		if(ans == 0) cout << "Nice work, Dima!" << endl;
		else cout << ans << endl;
	}
	
	return 0;
}
