#include<bits/stdc++.h>
using namespace std;
const int Maxn=1010;
const int Maxm=3030;

struct EDGE
{
	int a,b,l,r;
	bool operator < (const EDGE& a) const
	{
		return l<a.l;
	}
}edge[Maxm];

int fa[Maxn],n,m;

void clr()
{
	for(int i=1;i<=n;i++)
	  fa[i]=i;
}

int getFa(int x)
{
	if(fa[x]!=x)
	  fa[x]=getFa(fa[x]);
	return fa[x];
}

void merge(int x,int y)
{
	int fx=getFa(x),fy=getFa(y);
	fa[fx]=fa[fy];
}

int main()
{
	cin>>n>>m;
	for(int i=0;i<m;i++)
	 scanf("%d%d%d%d",&edge[i].a,&edge[i].b,&edge[i].l,&edge[i].r);
	sort(edge,edge+m);
	
	int ans=0;
	for(int i=0;i<m;i++)
	{
		clr();
		for(int j=0;j<m;j++)
		{
			if(edge[j].l>edge[i].r)
			  break;
			if(edge[j].r<edge[i].r)
			  continue;
			merge(edge[j].a,edge[j].b);
			if(getFa(1)==getFa(n))
			{
				ans=max(ans,edge[i].r-edge[j].l+1);
				break;
			}
		}
	}
	
	if(ans==0)
	  cout<<"Nice work, Dima!"<<endl;
	else
	  cout<<ans<<endl;

	return 0;
}
