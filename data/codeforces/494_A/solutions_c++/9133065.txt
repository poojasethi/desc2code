#include <iostream>
#include <string>
using namespace::std;
string s;

int main()
{
	int i=0,a=0,b=0,l=0,lpos=-1;
	cin >> s;
	for(i=0; i<(int)s.size(); i++){
		if(s[i]=='(') a++;
		else if(s[i]==')'){ a--; b++; }
		else if(s[i]=='#'){ a--; l++; lpos=i; }
	}
	if(a<0){
	cout << "-1\n";
	return 0;
	}
	int x=0,y=0;
	for(i=0; i<(int)s.size(); i++){
		if(s[i]=='(') x++;
		else if(s[i]==')') y++;
		else if(s[i]=='#') y++;
		if(i==lpos) y+=a;
		
		if(y>x){ cout << "-1\n"; return 0;}
	}
		
	for(i=0; i<l-1; i++) cout << "1\n";
	cout << a+1 << endl;		
	return 0;
}
