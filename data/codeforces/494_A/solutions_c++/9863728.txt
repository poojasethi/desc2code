#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    cin >> s;
    int x=0,d,q;
    int h=0,p=0;
    for(int i=0;i<s.size();i++){
        if(s[i]=='(') h++;
        else p++;
        if(h<p) {cout << -1; return 0;}
        if(s[i]=='#') {x++; q=h-p;}
        else if(s[i]==')') d=h-p;
    }
    d-=h-p;
    q-=h-p;
    if(d<0||q<0) cout << -1;
    else{
        for(int i=1;i<x;i++) cout << 1 << endl;
        cout << h-p+1 ;
    }
}
