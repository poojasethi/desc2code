#include <iostream>
using namespace std;

int main()
{
	char d;
	int a = 0, b = 0, c = 0;
	while (cin >> d){
		if (d == '(') 
			++a, ++c;
		else{
			a--;
			if (c) c--;
			if (a < 0){
				cout << -1; 
				return 0;
			}
			if (d == '#')
				c = 0, b++;
		}
	}
	if (c > 0){
		cout << -1;
		return 0;
	}
	for (int i = 1; i < b; i++)
		cout << 1 << '\n';
	cout << a + 1;
	return 0;
}
