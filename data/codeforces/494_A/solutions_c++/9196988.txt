#include<string>
#include<iostream>

using namespace std;

typedef long long ll;

int main(){
	string s;
	cin>>s;
	
	int a0=0,b0=0,k=0,b;
	
	for(int i=0;i<s.length();i++){
		if(s[i]=='(')a0++;
		if(s[i]==')')a0--;
		if(s[i]=='#')b0++;
	}
	if(a0<b0) {cout<<-1<<endl;return 0;}
	b=b0;
	for(int i=0;i<s.length();i++){
		if(s[i]=='(')k++;
		if(s[i]==')')k--;
		if(s[i]=='#'){
			b--;
			k--;
		}
		if(b==0){k-=(a0-b0);b--;}
		//cout<<k<<b<<endl;
		if(k<0) {cout<<-1;return 0;}
	}
	
	for(int i=0;i<b0-1;i++){
		cout<<1<<endl;
	}
	cout<<1+a0-b0<<endl;
}
