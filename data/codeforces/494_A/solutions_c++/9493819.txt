
 #include<cstdio>
#include<cstring>
#define N 100005
#define GO {puts("-1");return 0;}
using namespace std;
char a[N];int ans[N],num[N],Left[N];
int n,i,L,cnt,last;
int main()
{
  scanf("%s",a+1);
  n=strlen(a+1);
  for (i=1;i<=n;i++)
  {
    if (a[i]=='(') Left[++L]=i;
    if (a[i]==')') if ((--L)<0) GO;
    if (a[i]=='#')
    {
      ans[++cnt]=i;
      num[cnt]=1;if ((--L)<0) GO;
      last=i;
    }
  }
  if (L&&Left[L]>last) GO;
  for (i=1;i<cnt;i++) printf("%d\n",num[i]);
  printf("%d\n",L+1);
  return 0;
}
