#include <iostream>
#include <string>
#include<cstdio>
using namespace std;

int main()
{
	string s;
	int x=0,n=0,c=0;
	cin >> s;
	for (int i = 0; i < s.length(); i++){
		if (s[i] == '(') 
			++x, ++c;
		else{
			if (x==0){
				puts("-1"); 
				return 0;
			}
			x--;
			if (c) c--;
			if (s[i] == '#')
			{
				c=0;
				n++;
			}
		}
	}
	if (c > 0){
		puts("-1");
		return 0;
	}
	for (int i=1;i<n;i++)
		puts("1");
	printf("%d\n",x+1);
	return 0;
}
