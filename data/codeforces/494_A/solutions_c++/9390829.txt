#include<cstdio>
using namespace std;
char s[100001];
int main(){
    int a=0,b=0,c=0;bool ok=1;
    scanf("%s",s);
    for(int i=0;s[i];++i)
        if(s[i]=='(')++a,++c;
        else{
            --a;if(c)--c;
            if(a<0){puts("-1");ok=0;break;}
            if(s[i]=='#')c=0,++b;
        }
    if(ok && c>0){puts("-1");ok=0;}
    if(ok){
        for(int i=1;i<b;++i)puts("1");
        printf("%d\n",a+1);
    }
}
