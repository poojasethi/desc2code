#include <bits/stdc++.h>
using namespace std;
char ch[100020];
int main() {
	cin>>ch;
	int len=strlen(ch);
	int bobo=0,count=0,lolo=0;
	for(int i=0;i<len;i++)
	{
		if(ch[i]=='(')
		{
			bobo++;
			lolo++;
		}
		else if(ch[i]==')')
		{
			bobo--;
			lolo--;
			if(lolo<0)
				lolo=0;
		}
		else if(ch[i]=='#')
		{
			bobo--;
			lolo=0;
			count++;
		}
		if(bobo<0)
		{
			cout<<-1;
			return 0;
		}
	}
	if(lolo>0)
		cout<<-1;
	else
	{
		for(int i=1;i<count;i++)
			cout<<1<<endl;
		cout<<bobo+1;
	}
	return 0;
}
