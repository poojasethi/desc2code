#include <iostream>
#include <string>
using namespace std;

int main()
{
	string d;
	int a = 0, b = 0, c = 0;
	cin >> d;
	for (int i = 0; i < d.length(); i++){
		if (d[i] == '(') 
			++a, ++c;
		else{
			a--;
			if (c) c--;
			if (a < 0){
				cout << -1; 
				return 0;
			}
			if (d[i] == '#')
				c = 0, b++;
		}
	}
	if (c > 0){
		cout << -1;
		return 0;
	}
	for (int i = 1; i < b; i++)
		cout << 1 << '\n';
	cout << a + 1;
	return 0;
}
