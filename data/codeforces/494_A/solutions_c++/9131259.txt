#include <cstdio>
#include <cstring>
using namespace std;

char s[100003];

int main()
{
  gets(s);
  int l=strlen(s);
  int c=0,last=0,cnt=0;
  for(int i=0;i<l;i++)
  {
    if(s[i]=='(') c++;
	else if(s[i]==')') c--;
	else cnt++,c--,last=i;
	if(c<0) {printf("-1");return 0;}
  }
  int left=c;c=0;
  for(int i=0;i<l;i++)
  {
    if(s[i]=='(') c++;
	else c--;
	if(i==last) c-=left;
	if(c<0) {printf("-1");return 0;}
  }
  for(int i=1;i<=cnt;i++)
    printf("%d\n",i==cnt?left+1:1);
  return 0;
}
