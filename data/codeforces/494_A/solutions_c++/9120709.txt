#include<stdio.h>
#include<stdlib.h>
char s[100005];
int main(void){
	scanf("%s",s);
	int i;
	int ok=1;
	int lst=0;
	int rr=0;
	int gg=0;
	int cnt=0;
	int ff=0;
	for(i=0;s[i];i++)
	  if(s[i]=='(') ff++;
	  else ff--;
	ff++;
	for(i=0;s[i];i++)
	  if(s[i]=='#') lst=i,cnt++;
	for(i=0;s[i];i++){
      if(i==lst) gg-=ff; 
	  else if(s[i]=='#')  gg--;
	  else if(s[i]=='(') gg++;
	  else if(s[i]==')') gg--;
	  if(gg<0) ok=0;
	}
	if(ok==0||ff<=0) printf("-1\n");
	else{
		for(i=0;i<cnt-1;i++) printf("1\n");
		printf("%d\n",ff);
	}
	return 0;
}
