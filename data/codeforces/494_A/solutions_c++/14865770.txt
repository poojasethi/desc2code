#include <iostream>
#include <algorithm>
#include <queue>
using namespace std;
int main(){
	char a[100001];
	cin>>a;
	int x=0;
	priority_queue <int> k,l;
	for(int i=0;a[i];i++){
		if(a[i]=='('){
			k.push(i);
		}
		if(a[i]==')'){
			if(k.empty()){
				cout<<-1;
				return 0;
			}
			k.pop();
		}
		if(a[i]=='#'){
			l.push(i);
		}
		if(k.size()<l.size()){
			cout<<-1;
			return 0;
		}
	}
	if(k.top()>l.top()){
		cout<<-1;
		return 0;
	}
	for(int i=l.size()-1;i;i--){
		cout<<1<<endl;
		k.pop();
	}
	cout<<k.size();
	
}
