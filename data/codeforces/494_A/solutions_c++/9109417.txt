#include <iostream>
#include <string>
using namespace std;

string s;
int n;
int a, c;
int pos;
int val[111111];

int main(){
	cin >> s;
	n = s.size();
	
	for(int i=0; i<n; i++){
		if(s[i] == '(') {a++; val[i] = 1;}
		if(s[i] == ')') {a--; val[i] = -1;}
		if(s[i] == '#') {c++; a--; val[i] = -1; pos = i;}
		if(a < 0) {cout << -1; return 0;}
	}
	
	int x=0;
	for(int i=n-1; i>pos; i--){
		x+=val[i];
		if(x > 0) {cout << -1; return 0;}
	}
	
	for(int i=1; i<c; i++) cout << 1 << '\n';
	cout << a+1;
	return 0;
}
