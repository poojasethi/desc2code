#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
	string cad;

	cin >> cad;

	int sum = 0, con = 0, p;
	for(int i = 0; i < cad.size(); i++){
		if(cad[i] == ')') sum++;
		if(cad[i] == '(') sum--;
		if(cad[i] == '#'){
			p = i;
			con++;
			sum++;
		}
		if(sum > 0){cout << "-1"; return 0;}
	}
	int sum2 = 0;
	for(int i = cad.size() - 1; i > p; i--){
		if(cad[i] == '(') sum2++;
		if(cad[i] == ')') sum2--;
		if(sum2 > 0) {cout << "-1"; return 0;}
	}
	for(int i = 0; i < con - 1; i++)
		cout << "1 ";
	cout << -1 * sum + 1;
	return 0;
}
