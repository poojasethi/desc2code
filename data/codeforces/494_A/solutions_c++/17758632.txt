#include <bits/stdc++.h>
using namespace std;

string s; int len,ct,last,a[100005],no;

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> s; len=s.length();
	for (int i=0;i<len;i++){
		if (s[i]=='(') ct++;
		else if (s[i]==')') ct--;
		else{
			a[i]=1; ct--; last=i;
		}
	}
	if (ct<0) no=1;
	a[last]+=ct; ct=0;
	for (int i=0;i<len;i++){
		if (s[i]=='(') ct++;
		else if (s[i]==')') ct--;
		else ct-=a[i];
		if (ct<0) no=1;
	}
	if (ct) no=1;
	if (no) cout << "-1\n";
	else{
		for (int i=0;i<len;i++)
		if (s[i]=='#') cout << a[i] << "\n";
	}
}
