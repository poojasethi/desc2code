#include<bits/stdc++.h>
using namespace std;
int main(){
    int i,t=0,last=0,cnt=0;
    string s;
    cin>>s;
    for(i=0;i<s.size();i++){
        if(s[i]=='(') t++;
        else if(s[i]==')') t--;
        else cnt++,t--,last=i;
        if(t<0) {puts("-1");return 0;}
    }
    int left=t;t=0;
    for(i=0;i<s.size();i++){
        if(s[i]=='(') t++;
        else t--;
        if(i==last) t-=left;
        if(t<0) {puts("-1");return 0;}
    }
    for(i=1;i<=cnt;i++)
        printf("%d\n",i==cnt?left+1:1);
}
