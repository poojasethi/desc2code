#include<cstdio>
#include<cstring>
using namespace std;
char s[100001];
int n,last,gs;
int main()
{
	int i,j,k;
	scanf("%s",s);
	n=strlen(s);
	for(i=n-1;i>=0;i--)if(s[i]=='#'){last=i;break;}
	int sum=0,sum2=0;
	for(i=0;i<last;i++)
	{
		if(s[i]=='(')sum++;else sum--;
		if(s[i]=='#')gs++;
		if(sum<0){printf("-1\n");return 0;}
	}
	for(i=n-1;i>last;i--)
	{
		if(s[i]==')')sum2++;else sum2--;
		if(sum2<0){printf("-1\n");return 0;}
	}
	if(sum-sum2<=0){printf("-1\n");return 0;}
	for(i=1;i<=gs;i++)printf("1\n");
	printf("%d\n",sum-sum2);
	return 0;
}
