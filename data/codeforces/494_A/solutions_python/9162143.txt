#!/usr/bin/env python
#-*- coding:utf-8 -*-

s = raw_input()

c1 = 0
c2 = 0
cx = 0
for c in s:
    if c == "(":
        c1 += 1
    elif c == ")":
        c2 += 1
    elif c == "#":
        cx += 1
    if c2 + cx > c1:
        print -1
        quit()

diff = c1 - c2
c1 = 0
c2 = 0
i = 0
res = []
for c in s:
    if c == "(":
        c1 += 1
    elif c == ")":
        c2 += 1
    elif c == "#":
        i += 1
        if i != cx:
            res.append(1)
            diff -= 1
            c2 += 1
        else:
            res.append(diff)
            c2 += diff
    if c2 > c1:
        print -1
        quit()

for a in res:
    print a
