#!/usr/bin/env python

def main():
    s = raw_input()
    bal = 0
    a = []
    b = []
    for c in s:
        if c == '(':
            bal += 1
        elif c == ')':
            bal -= 1
        else: # c == '#'
            bal -= 1
            a.append(1)
        if bal < 0:
            return [-1]
    a[-1] += bal

    # check subexpr after last #))...)
    lhi = s.rfind('#')
    sfxi = s.find('(', lhi)
    if sfxi != -1:
        if s.count('(', sfxi) > s.count(')', sfxi):
            return [-1]

    return a

print ''.join(str(b) + "\n" for b in main())
