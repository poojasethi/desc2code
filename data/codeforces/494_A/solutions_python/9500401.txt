#!/usr/bin/env python
def work(s):
    p = s.rfind('#')
    k = s.count('(')*2 - len(s) + 1
    if k <= 0:
        return False
    t = 0
    for i in range(len(s)):
        if s[i] == '(':
            t += 1
        elif s[i] == ')':
            t -= 1
        elif i == p:
            ans.append(k)
            t -= k
        else:
            ans.append(1);
            t -= 1
        if t < 0:
            return False
    return True
ans = []
s = raw_input()
if work(s):
    for i in ans:
        print i
else:
    print -1
