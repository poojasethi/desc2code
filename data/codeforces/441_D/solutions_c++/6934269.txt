#include <iostream>
#include <cmath>
using namespace std;

const int MAXN = 3005;
int n, a[MAXN], k, c, cn[MAXN];

int cyc(){
  bool vis[MAXN] = {};
  int res = 0, c = 0;
  for(int i = 0; i < n; i++){
    if(!vis[i]){
      int cur = i;
      while(!vis[cur]){
        vis[cur] = 1;
        cn[cur] = c;
        cur = a[cur];
        res++;
      }
      res--;
      c++;
    }
  }
  return res;
}

int main(){
  cin >> n;
  for(int i = 0; i < n; i++){
    cin >> a[i];
    a[i]--;
  }
  cin >> k;
  c = cyc();
  cout << abs(c - k) << '\n';
  while(c != k){
    if(c > k){ // sort
      int q = 0;
      while(a[q] == q) q++;
      for(int i = q + 1; i < n; i++){
        if(cn[i] == cn[q]){
          cout << q + 1 << " " << i + 1 << " ";
          swap(a[q], a[i]);
          break;
        }
      }
    } else {
      for(int i = 0; i < n; i++){
        if(cn[i] != cn[0]){
          cout << 1 << " " << i + 1 << " ";
          swap(a[0], a[i]);
          break;
        }
      }
    }
    c = cyc();
  }
}
