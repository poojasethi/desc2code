#include<stdio.h>

int x[3001], y[3001];
main()
{
	int a,b,c;
	int n,m;
	int i,t,j;
	int xx;
	while(scanf("%d",&n)!=EOF)
	{
		for (i=1;i<=n;i++) 
		{
			scanf("%d",&x[i]);
		}
		scanf("%d",&m);
		xx=0;
		for (i=1;i<=n;i++) 
		{
			if (y[i]==0) 
			{
				y[i]=i;
				for (j=x[i];j!=i;j=x[j]) 
				{
					y[j]=i;
				}
				xx++;	
			}
		}
		if(n-xx==m) printf("0\n");
		else if(n-xx>m) 
		{
			i=n-xx-m;
			printf("%d\n",i);
			while(i--) 
			{ 
				j=1;
				while(x[j]==j)
				{
					j++;
				}
				
				for (t=j,b=x[j];x[t]!=j;t=x[t]) 
				{
					if (x[t]<b) 
					{	
						b=x[t];
					}
				}
				printf("%d %d ",j,b);
				a=x[j]; x[j]=x[b]; x[b]=a;
			}
			printf("\n");
			}
			else 
			{
				i=m+xx-n;
				printf("%d\n",i);
				while(i--) 
				{ 
				for (j=1;j<=n;j++)
				{
					y[j]=0;
				}
				for (y[1]=1,j=x[1]; j!=1; j=x[j]) 
				{
					y[j]=1;
				}
				j=1;
				while(y[j]==1)
				{
					j++;
				}
				printf("1 %d ",j);
				a=x[j]; x[j]=x[1]; x[1]=a;
			}
			printf("\n");
		}
	}
}
