#include<bits/stdc++.h>
using namespace std;

int a[3005];
bool f[3005];

int main()
{
	int n,m,i,j,c,x,y;
	for(scanf("%d",&n),i=1;i<=n;i++)
		scanf("%d",a+i);
	scanf("%d",&m);
	for(c=0,i=1;i<=n;i++)
		if(!f[i])
		{
			f[i]=1;
			for(j=a[i];j!=i;j=a[j],c++)
				f[j]=1;
		}
	printf("%d\n",abs(c-m));
	while(c!=m)
	{
		if(c>m)
		{
			for(i=1;i<=n;i++)
				if(a[i]!=i)
				{
					x=i;
					break;
				}
			for(y=j=a[i];j!=i;j=a[j])
				y=min(y,j);
			c--;
		}
		else
		{
			x=1;
			if(a[1]==1)
				y=2;
			else
			{
				memset(f,0,sizeof(f));
				f[1]=1;
				for(i=a[x];i!=1;i=a[i])
					f[i]=1;
				for(i=1;f[i];i++);
				y=i;
			}
			c++;
		}
		swap(a[x],a[y]);
		printf("%d %d%c",x,y,c==m?'\n':' ');
	}
	return 0;
}
