#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

const int N = 3005;

int n, m, cnt;
int a[N];
bool used[N];
int pos[N];

void cycle(int v){
	for(int i = v; !used[i]; i = a[i])
		used[i] = 1;
}

int main(){
	#ifndef ONLINE_JUDGE
	freopen ("in","r",stdin);
	freopen ("out","w",stdout);
	#endif
	cin >> n;
	for(int i = 1;i <= n;++ i)
		cin >> a[i];
	cin >> m;
	m = n - m;
	for(int i = 1;i <= n;++ i){
		if(!used[i]){
			++cnt;
			cycle(i);
		}
	}			
	printf("%d\n",abs(cnt - m));
	if(cnt > m){
		memset(used, 0,sizeof used);
		cycle(1);
		for(int i = 2;i <= n && cnt > m;++ i){
			if(!used[i]){
			    --cnt;
				printf("%d %d ",1, i);
				cycle(i);				
			}
		}
	}
	else {
		for (int i = 1; i <= n && cnt < m; ++ i) {
			for(int j = 1;j <= n;++ j)pos[j] = -1;
			int cur = 0;
			for (int j = a[i]; j != i; j = a[j])
				pos[j] = cur++;
			pos[i] = cur;
			cur = 0;
			for (int j = i + 1; j <= n && cnt < m; ++ j)
				if (pos[j] >= cur){
					printf("%d %d ", i, j);
					++cnt;
					cur = pos[j] + 1;
					swap(a[i], a[j]);
				}
		}	
	}
	return 0;
}
