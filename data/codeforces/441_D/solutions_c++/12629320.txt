#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
const int sz = 3030;

int n, m;
int a[sz], b[sz], c[sz];
int gao(){
     int i, j, t = 0;
     for (i = 1; i <= n; i++) b[i] = c[i] = 0;
     for (i = 1; i <= n; i++)
      if (!b[i]){
           t++;
           for (j = i; !b[j]; j = a[j]) b[j] = t, c[t]++;
      }
     return t;
}

int main(){
     int i, j, t;
     scanf("%d", &n);
     for (i = 1; i <= n; i++) scanf("%d", &a[i]);
     scanf("%d", &m);
     t = n - gao();
     if (t > m){
      printf("%d\n", t - m);
      t = t - m;
      while (t--){
           for (i = 1; i <= n; i++) if (c[i] != 1) break;
           for (j = i + 1; j <= n; j++) if (b[i] == b[j]) break;
           printf("%d %d ", i, j);
           swap(a[i], a[j]);
           gao();
      }
     } else if (t < m){
      printf("%d\n", m - t);
      t = m - t;
      while (t--){
           for (i = 1; i <= n; i++) if (b[i] != b[1]) break;
           printf("%d %d ", 1, i);
           swap(a[1], a[i]);
           gao();
      }
     } else {
      printf("0");
     }
     return 0;
}
