#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N = 3005;
int n, m, a[N], p[N], v[N];

int cal(int a[]) {
	int ret = n, itr = 0;
	memset(v, 0, sizeof(v));
	for(int i = 0; i < n; i ++) {
		if (!v[i]){
			++itr;
			int x = i;
			while (!v[x]){v[x] = itr, x = a[x];}
			--ret;
		}
	}
	return ret;
}
int main() {
	scanf("%d", &n);
	for(int i = 0; i < n; i ++) {
		scanf("%d", &a[i]);
		a[i] --;
	}
	scanf("%d", &m);
	
	int cur = cal(a);
	printf("%d\n", abs(cur - m));
	do{
		cur = cal(a);
		for(int i = 0; i < n; i ++) p[a[i]] = i;
		if (cur > m){
			for(int i = 0; i < n; i ++) if (a[i] != i){
				int t;
				for(int j = i + 1; j < n; j ++) {
					if (v[i] == v[j]){
						t = j;
						break;
					}
				}
				printf("%d %d ", i + 1, t + 1);
				swap(a[i], a[t]);
				break;
			}
		}
		else if (cur < m){
			for(int i = 0; i < n; i ++) {
				if (v[i] != v[0]){
					printf("%d %d ", 1, i + 1);
					swap(a[0], a[i]);
					break;
				}
			}
		}
	} while (cur != m);
	puts("");
	return 0;
}
