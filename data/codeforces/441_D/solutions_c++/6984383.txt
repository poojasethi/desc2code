#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 3010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,s,A;
int a[N],f[N],id[N];
inline void work1(int x) {
	int i,j;
	printf("%d\n",x);
	k=1;
	For(i,1,n) if (id[i]>k) {
		if (!x) break;
		printf("%d %d ",1,i);
		k++;
		swap(a[1],a[i]);
		if (k-1==x) break;
	}
	printf("\n");
}
inline void work2(int x) {
	int i,j;
	printf("%d\n",x);
	for (i=1;i<=n;) if (a[i]!=i) {
		if (!x) break;
		int mi=a[i];
		for (j=a[i];j!=i;j=a[j]) mi=min(mi,j);
		printf("%d %d ",i,mi);
		swap(a[i],a[mi]);
		if (!(--x)) break;
	} else i++;
	printf("\n");
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d",&a[i]);
	For(i,1,n) if (!f[i]) {
		for (k=a[i];k!=i;k=a[k]) f[k]=1;
		id[i]=++s;
	}
	scanf("%d",&A);
	A=n-A;
	if (s<A) work2(A-s);
	else work1(s-A);
	//for(;;);
	return 0;
}
