#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
int n, m, q;
int p[3111];
int C[3111];

int main() {
  int nxt, mn, k;
  while (scanf("%d", &n) > 0) {
    for(int i=1; i<=n; i++)
      scanf("%d", &p[i]);
    scanf("%d", &m);
    memset(C, -1, sizeof(int)*(n+1));
    q = n;
    for(int i=1; i<=n; i++)
      if (C[i] == -1) {
        int c = i;
        q --;
        while (C[c] == -1) {
          C[c] = i;
          c = p[c];
        }
      }
    q -= m;
    if (q >= 0) {
      printf("%d\n", q);

      nxt = 1;
      for(int j=0; j<q; j++) {
        if (j > 0) printf(" ");
        while (p[nxt] == nxt) {
          nxt ++;
        }
        mn = p[nxt];
        k = p[nxt];
        while (k != nxt) {
          mn = min(mn, k);
          k = p[k];
        }
        printf("%d %d", nxt, mn);
        swap(p[nxt], p[mn]);
      }
      printf("\n");
    } else {
      q *= -1;
      printf("%d\n", q);
      nxt = 2;
      mn = 1;
      for(int i=0; i<q; i++){
        if (i>0) printf(" ");
        while (C[nxt] < nxt)
          nxt ++;
        printf("1 %d", nxt);
        nxt ++;
      }
      printf("\n");
    }
  }
  return 0;
}
