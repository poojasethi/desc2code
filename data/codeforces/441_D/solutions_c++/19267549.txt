#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
const int maxn = 200000+ 100;
int num[maxn],a[maxn],sum[maxn],n;
bool vis[maxn];
void dfs(int x,int k)
{
	num[x]=k;vis[x]=1;sum[k]++;
	if(!vis[a[x]])dfs(a[x],k);
	return ;
}
int maintain()
{
	memset(vis,0,sizeof(vis));
	memset(sum,0,sizeof(sum));
	int cnt=0;
	for(int i=1;i<=n;i++)
	if(!vis[i])dfs(i,++cnt);
	return cnt;
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	int k=maintain();
	int m;cin>>m;m=n-m;
	cout<<abs(m-k)<<endl;
	while(m!=k)
	{
//		cout<<k<<endl;system("pause");
		if(m>k)
		{
			int j=1;
			while(sum[num[j]]==1)j++;
			for(int i=j+1;i<=n;i++)
			if(num[i]==num[j])
			{
				swap(a[i],a[j]);
				printf("%d %d ",j,i);
				break;
			}
		}
		else
		{
			for(int i=2;i<=n;i++)
			if(num[i]!=num[1])
			{
				swap(a[i],a[1]);
				printf("1 %d ",i); 
				break;
			}
		}
		k=maintain();
	}
	cout<<endl;
	return 0;
}
