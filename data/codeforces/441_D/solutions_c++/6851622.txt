#include <iostream>
using namespace std;
const int N = 5000;

int n, m, cnt, a[N], col[N];

inline void change(int k, int cnt) {
	int x = k;
	do {
		col[x] = cnt;
		x = a[x];
	} while (x != k);
}
int main() {
	ios_base::sync_with_stdio(false);
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i], a[i]--;
	cin >> m;
	m = n - m;
	for (int i = 0; i < n; i++)
		if (!col[i]) {
			change(i, ++cnt);
		}
	if (cnt > m) {
		cout << cnt - m << '\n';
		for (int j = 0; j < n && cnt > m; j++)
			for (int k = j + 1; k < n && cnt > m; k++)
				if (col[j] != col[k]) {
					cout << j + 1 << ' ' << k + 1 << ' ';
					swap(a[j], a[k]);
					change(j, col[j]);
					cnt--;
				}
	}
	else {
		cout << m - cnt << '\n';
		for (int j = 0; j < n && cnt < m; j++)
			for (int k = j + 1; k < n && cnt < m; k++)
				if (col[j] == col[k]) {
					cout << j + 1 << ' ' << k + 1 << ' ';
					swap(a[j], a[k]);
					change(k, ++cnt);
				}
	}
	cout << '\n';
	return 0;
}
