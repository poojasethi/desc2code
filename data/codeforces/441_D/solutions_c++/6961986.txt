#include<stdio.h>
#include<string.h>
int main()
{
	int i,j,sum,n,m,x,y;
	int a[3001],b[3001];
	while(scanf("%d",&n)!=EOF){
		memset(b,0,sizeof(b));
		for(i=1;i<=n;i++)
			scanf("%d",&a[i]);
		scanf("%d",&m);
		sum=0;
		for(i=1;i<=n;i++)
			if(a[i]!=i&&!b[i]){
				for(j=i;a[j]!=i;j=a[j]){
					sum++;
					b[j]=1;
				}
				b[j]=1;
			}
		if(sum==m)
			printf("0\n");
		else if(sum>m){
			x=sum-m;
			printf("%d\n",x);
			while(x--){
				i=1;
				while(a[i]==i)
					i++;
				for(y=j=a[i];j!=i;j=a[j])
					if(j<y&&j!=i)
						y=j;
				printf("%d %d ",i,y);
				j=a[i];
				a[i]=a[y];
				a[y]=j;
			}
			printf("\n");
		}
		else {
			x=m-sum;
			printf("%d\n",x);
			memset(b,0,sizeof(b));
			for(i=a[1];i!=1;i=a[i])
				b[i]=1;
			while(x--){
				i=2;
				while(b[i])
					i++;
				printf("1 %d ",i);
				b[i]=1;
				for(j=a[i];j!=i;j=a[j])
					b[j]=1;
			}
			printf("\n");
		}
	}
}
