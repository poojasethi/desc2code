#include <cstdio>
#include <cstring>
#include <cmath>
#include <vector>
#pragma GCC optimize("O2")
using namespace std;

typedef vector<int> vin;

int n, m;
int a[3002];
int b[3002];

void ucyc(int x) {
    for (int i=x; !b[i]; i=a[i])
        b[i]=1;
}

int main(void) {
    scanf("%d", &n);
    int t=0;
    for (int i=1; i<=n; i++)
        scanf("%d", a+i);
    for (int i=1; i<=n; i++) if (!b[i]) {
        t++;
        ucyc(i);
    }
    memset(b, 0, sizeof(b));
    scanf("%d", &m);
    m=n-m;
    printf("%d\n", (int)abs(t-m));
    if (t>m) {
        ucyc(1);
        for (int i=2; i<=n && t>m; i++) if (!b[i]) {
            printf("%d %d ", 1, i);
            t--;
            ucyc(i);
        }
    }
    if (t<m) {
        for (int i=1; i<=n && t<m; i++) {
            vin p(n+1, -1);
            int c=1;
            for (int j=a[i]; j!=i; j=a[j])
                p[j]=c++;
            p[i]=c;
            c=1;
            for (int j=i+1; j<=n && t<m; j++) if (p[j]>=c) {
                printf("%d %d ", i, j);
                t++;
                c=p[j];
                swap(a[i], a[j]);
            }
        }
    }
}
