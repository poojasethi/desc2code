#include <stdio.h>
#include <string.h>
#define N 3000
int p[N+20],f[N+20],num[N+20],fa[N+20],pre[N+20];
int main()
{
	int n,m,x,y;
	scanf("%d",&n);
	for (int i=1;i<=n;i++) 
	{
		scanf("%d",&x);
		//p[x] = i;
		//pre[i] = x;
		p[i] = x;
		pre[x] = i;
	}
	int cnt = 0;
	for (int i=1;i<=n;i++)
	{
		if (f[i]==0)
		{
			cnt++;
			f[i]=cnt;
			num[cnt]++;
			fa[cnt] = i;
			y = p[i];
			while (y!=i)
			{
				f[y] = cnt;
				num[cnt]++;
				y = p[y];
			}
		}
	}
	int sum = 0;
	for (int i=1;i<=cnt;i++) sum = sum + num[i];
	sum = sum - cnt;
	scanf("%d",&m);
	if (sum>=m)
	{
		printf("%d\n",sum-m);
		int x,y;
		for (int i=1;i<=n;i++)
		{
			if (sum==m) break;
			while (p[i]!=i)
			{
				if (sum==m) break;
				x = p[i]; y = p[i];
				while (y!=i)
				{
					if (y<x) x = y;
					y = p[y];
				}
				printf("%d %d ",i,x);
				sum--;
				int tmp = p[i];
				p[i] = p[x];
				p[x] = tmp;
			}
		}
	}
	else
	{
		printf("%d\n",m-sum);
		for (int i=2;i<=cnt;i++)
		{
			if (m==sum) break;
			printf("%d %d ",fa[1],fa[i]);
			sum++;
		}
	}
	return 0;
}
