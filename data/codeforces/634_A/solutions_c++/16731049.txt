#include <cstdio>

int main() {
    int n, ia, ib;
    int a[200005], b[200005];
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
        if (a[i] == 1) ia = i;
    }
    for (int i = 0; i < n; i++) {
        scanf("%d", &b[i]);
        if (b[i] == 1) ib = i;
    }
    bool f = true;
    for (int i = 0; i < (n-1); i++) {
        if (a[(ia+i)%n] == 0) ia++;
        if (b[(ib+i)%n] == 0) ib++;
        if (a[(ia+i)%n] != b[(ib+i)%n]) {
            f = false;
            break;
        }
    }
    printf("%s\n", f ? "YES" : "NO");
}
