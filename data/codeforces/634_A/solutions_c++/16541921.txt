#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,i,j,l,flag=0;
	scanf("%d",&n);
	int a[n],b[n];
	for(i=0;i<n-1;i++){
		scanf("%d",&a[i]);
		if(a[i]==0)
		i=i-1;
	}
	for(i=0;i<n-1;i++){
		scanf("%d",&b[i]);
		if(b[i]==0)
		i=i-1;
	}
	for(i=0;i<n-1;i++){
		if(a[i]==b[0]){
			l=i;
			break;
		}
	}
	for(i=0;i<n-1;i++){
		if(b[i]!=a[(l+i)%(n-1)]){
			flag=1;
			break;
		}
	}
	if(flag==1){
		printf("No\n");
	}
	else
		printf("Yes\n");
	return 0;
}
