#include<stdio.h>
int main()
{
	int i,j,n,a[200010],b[200010],first,flag;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	
	for(i=0;i<n;i++)
	scanf("%d",&b[i]);
	
	for(i=0;i<n;i++)
	if(a[i]!=0)
	{
		first=i;
		break;
	}
	for(i=0,flag=0;i<n;i++)
	if(b[i]==a[first])
	{
		for(j=1;j<n;j++)
		{
			if(b[(i+j)%n]==0)
			i++;
			if(a[(first+j)%n]==0)
			first++;
			if(b[(i+j)%n]!=a[(first+j)%n])
			{
				flag=1;
				break;
			}
		}
	}
	if(flag)
	printf("NO\n");
	else
	printf("YES\n");
}
