	int n;
	
	scanf("%d", &n);
	
	for (int i = 0; i < n; i++) {
		int v;
		scanf("%d", &v);
		if (v) {
			a.push_back(v);
		}
	}
	
	for (int i = 0; i < n; i++) {
		int v;
		scanf("%d", &v);
		if (v) {
			b.push_back(v);
		}
	}
	
	rotate(a.begin(), min_element(a.begin(), a.end()), a.end());
	rotate(b.begin(), min_element(b.begin(), b.end()), b.end());

	} else {

	}

}
