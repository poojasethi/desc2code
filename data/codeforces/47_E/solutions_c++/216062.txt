#include <cmath>
#include <iostream>
#include <iomanip>
#include <algorithm>
#include <set>
using namespace std;

int n, m, V;
double alpha[1<<18], x[1<<18], y[1<<18], g = 9.8;
int ord_alpha[1<<18], ord_x[1<<18];
pair<double,double> ans[1<<18];

int cmp_alpha(int a, int b) {
	return alpha[a] < alpha[b];
}

int cmp_x(int a, int b) {
	return x[a] < x[b];
}

int main() {
	cin >> n >> V;
	for(int i = 0; i < n; i++) cin >> alpha[i], ord_alpha[i] = i;
	sort(ord_alpha,ord_alpha+n,cmp_alpha);
	cin >> m;
	for(int i = 0; i < m; i++) cin >> x[i] >> y[i], ord_x[i] = i;
	sort(ord_x,ord_x+m,cmp_x);
	int px = 0;
	for(int i = 0; i < n; i++) {
		int now = ord_alpha[i], ok = 0;
		while(px < m) {
			int loc = ord_x[px];
			double t = x[loc] / (V * cos(alpha[now]));
			double yt = V * sin(alpha[now]) * t - g * t * t / 2;
			if(yt <= y[loc]) {
				ans[now] = pair<double,double>(x[loc],yt);
				if(yt >= 0) ok = 1;
				break;
			}
			else px++;
		}
		if(!ok) {
			double t = 2 * V * sin(alpha[now]) / g;
			double xt = V * cos(alpha[now]) * t;
			ans[now] = pair<double,double>(xt,0.0);
		}
	}
	for(int i = 0; i < n; i++) {
		cout << fixed << ans[i].first << " " << ans[i].second << endl;
	}
	return 0;
}
