#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;

const int N = 100005;
int p[N], q[N], n, m, V;
double a[N], x[N], y[N], ax[N], ay[N], vx, vy;
inline bool cmp1(int i, int j) {return a[i]<a[j];}
inline bool cmp2(int i, int j) {return x[i]<x[j];}
double pos(double t) {return vy*t-4.9*t*t;}
int main() {
	scanf("%d%d", &n, &V);
	for (int i=0; i<n; i++)
		scanf("%lf", &a[i]), q[i]=i;
	scanf("%d", &m);
	for (int i=0; i<m; i++)
		scanf("%lf%lf", &x[i], &y[i]), p[i]=i;
	sort(q,q+n,cmp1);
	sort(p,p+m,cmp2);
	int j=0;
	double t, w;
	for (int k=0;k<n;k++) {
		int i=q[k];
		vx = V*cos(a[i]), vy = V*sin(a[i]), t=vy/4.9, w=vx*t;
		for (; j<m && pos(x[p[j]]/vx)>y[p[j]]; ++j);
		if (j<m && x[p[j]]<w)
			ax[i] = x[p[j]], ay[i] = pos(x[p[j]]/vx); else
			ax[i] = w, ay[i] = 0;
	}
	for (int i=0;i<n;i++)
		printf("%.9lf %.9lf\n", ax[i]+1e-7, ay[i]+1e-7);
	return 0;
}
