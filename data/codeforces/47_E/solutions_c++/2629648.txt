#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
#define ft first
#define sd second
#define sqr(x) ((x)*(x))
using namespace std;
typedef pair<double,double> PR;
const int N=100100;
int n,V,m;
PR a[N],b[N],ans[N];
int main()
{
    scanf("%d%d",&n,&V);
    for(int i=1;i<=n;++i)
        scanf("%lf",&a[i].ft),a[i].sd=i+1e-7;
    sort(a+1,a+1+n);
    scanf("%d",&m);
    for(int i=1;i<=m;++i)
        scanf("%lf%lf",&b[i].ft,&b[i].sd);
    sort(b+1,b+1+m);
    for(int i=1,j=1;i<=n;++i)
    {
        double B=tan(a[i].ft),A=4.9/sqr(cos(a[i].ft)*V),y;
        while ( j<=m && (y=B*b[j].ft-A*sqr(b[j].ft))>=b[j].sd+1e-9 ) ++j;
        int k=a[i].sd;
        if ( j>m || y<=0 ) ans[k]=make_pair(B/A,0);
            else ans[k]=make_pair(b[j].ft,B*b[j].ft-A*sqr(b[j].ft));
    }
    for(int i=1;i<=n;++i) printf("%.10lf %.10lf\n",ans[i].ft,ans[i].sd);
    return 0;
}
