#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<iostream>
using namespace std;
const double g=4.9;
const double eps=1e-8;
int N,M;
double v;
struct Point
{
	double x,y;
	bool operator<(const Point &b)const
	{
		return x<b.x;
	}
}p[100005];
struct In
{
	double a;
	int id;
	bool operator<(const In &b)const
	{
		return a<b.a;
	}
}q[10005];
double x[10005],y[10005];
int main()
{
	while(~scanf("%d%lf",&N,&v))
	{
		for(int i=0;i<N;i++)
		{
			scanf("%lf",&q[i].a);
			q[i].id=i;
		}
		scanf("%d",&M);
		for(int i=0;i<M;i++)
		scanf("%lf%lf",&p[i].x,&p[i].y);
		sort(p,p+M);sort(q,q+N);
		int j=0;
		for(int i=0;i<N;i++)
		{
			int f=0;
			while(j<M)
			{
				double t=p[j].x/v/cos(q[i].a);
				double Y=v*sin(q[i].a)*t-g*t*t;
				if(Y<eps)break;
				if(Y<p[j].y+eps)
				{
					f=1;
					x[q[i].id]=p[j].x;
					y[q[i].id]=Y;
					break;
				}
				j++;
			}
			if(!f)
			{
				x[q[i].id]=v*v*sin(q[i].a)*cos(q[i].a)/g;
				y[q[i].id]=0.0;
			}
		}
		for(int i=0;i<N;i++)
		printf("%.6f %.6f\n",x[i],y[i]);
	}
}
