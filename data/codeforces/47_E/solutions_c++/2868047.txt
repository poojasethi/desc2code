#include <cstdio>
#include <cmath>
#include <utility>
#include <algorithm>
using namespace std;

const int N = int(1e4) + 50, M = int(1e5) + 50;

#define Px first
#define Py second

double v;
pair<double, double> fort[M], res[N];
pair<double, int> bullet[N];
int n, m;

int main ()
{
#ifdef LOCAL
	freopen("in", "r", stdin);
#endif
	scanf("%d%lf", &n, &v);
	for (int i = 0; i < n; ++i)
	{
		scanf("%lf", &bullet[i].Px);
		bullet[i].Py = i;
	}
	sort(bullet, bullet + n);
	scanf("%d", &m);
	for (int i = 0; i < m; ++i)
		scanf("%lf%lf", &fort[i].Px, &fort[i].Py);
	sort(fort, fort + m);
	for (int i = 0, j = 0; i < n; ++i)
	{
		double vx = v * cos(bullet[i].Px), vy = v * sin(bullet[i].Px), w = vy / 4.9 * vx, t, h;
		for (; j < m && fort[j].Px < w; ++j)
		{
			t = fort[j].Px / vx;
			h = vy * t - 4.9 * t * t;
			if (h <= fort[j].Py) break;
		}
		if (j < m && fort[j].Px < w) res[bullet[i].Py] = make_pair(fort[j].Px, h);
		else res[bullet[i].Py] = make_pair(w, 0);
	}
	for (int i = 0; i < n; ++i)
		printf("%.9f %.9f\n", res[i].Px, res[i].Py);
	return 0;
}
