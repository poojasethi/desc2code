#include <cstring>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <ctime>
using namespace std;
double t,x,y,V;
int N,M;
class rec1{public:double al,x,y;int t;}a[2000000];
class rec2{public:double x,y;}p[2000000];
double g=9.8;
bool cmp(rec2 a,rec2 b){return a.x<b.x;}
bool cmp2(rec1 a,rec1 b){return a.al<b.al;}
bool cmp3(rec1 a,rec1 b){return a.t<b.t;}
double floor(double al){t=V*sin(al)*2/g;return V*cos(al)*t;}
double calc(double x,double al){t=x/V/cos(al);y=V*sin(al)*t-g*t*t/2;return y;}
bool ok(int i,int j){t=calc(p[j].x,a[i].al);return t>p[j].y;}
int main(){
    scanf("%d%lf",&N,&V);for (int i=1;i<=N;i++)scanf("%lf",&a[i].al),a[i].t=i;
    scanf("%d",&M);for (int i=1;i<=M;i++)scanf("%lf%lf",&p[i].x,&p[i].y);
    sort(p+1,p+M+1,cmp);
    sort(a+1,a+N+1,cmp2);
    int j=1;
    for (int i=1;i<=N;i++){
        a[i].x=floor(a[i].al);a[i].y=0;
        if (j<=M&&ok(i,j))for(;j<=M&&ok(i,j);j++);
        if (j<=M&&a[i].x>p[j].x){a[i].x=p[j].x;a[i].y=calc(p[j].x,a[i].al);}
        }
    sort(a+1,a+N+1,cmp3);
    for (int i=1;i<=N;i++)printf("%0.7lf %0.7lf\n",a[i].x,a[i].y);
}
