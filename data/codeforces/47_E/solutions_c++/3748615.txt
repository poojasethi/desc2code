#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define x first
#define y second
using namespace std;
const double g=9.8;
double v,al[10005],ss[10005],cc[10005];
int d[10005];
pair<double,double> w[100005];
double x[10005];
double run(double x,double alpha)
{
	return (v*sin(alpha)+v*sin(alpha)-g*x/(v*cos(alpha)))*x/(v*cos(alpha))/2;
}
bool cmp(const int &a,const int &b)
{
	return al[a]<al[b];
}
int main()
{
	int n,m;
	scanf("%d%lf",&n,&v);
	for(int i=1;i<=n;i++)
		scanf("%lf",&al[d[i]=i]),cc[i]=cos(al[i]),ss[i]=sin(al[i]);
	sort(d+1,d+1+n,cmp);
	scanf("%d",&m);
	for(int i=1;i<=m;i++)
		scanf("%lf%lf",&w[i].x,&w[i].y);
	sort(w+1,w+1+m);
	int j=1;
	for(int i=1;i<=n;i++)
	{
		while(j<=m&&run(w[j].x,al[d[i]])>w[j].y)j++;
		if(j>m||run(w[j].x,al[d[i]])<0)
		{
			x[d[i]]=v*ss[d[i]]/g*2*v*cc[d[i]];
			continue;
		}
		x[d[i]]=w[j].x;
	}
	for(int i=1;i<=n;i++)
		printf("%.9lf %.9lf\n",x[i],run(x[i],al[i]));
	return 0;
}
