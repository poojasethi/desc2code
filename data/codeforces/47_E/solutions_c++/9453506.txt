#include<bits/stdc++.h>
using namespace std;
typedef double lf;
#define x first
#define y second
#define mp make_pair
const int maxn=100000+10;
const lf inf=INT_MAX;
pair<lf,int> a[maxn];
pair<lf,lf> b[maxn],ans[maxn];
lf v;
int n,m;
bool cmp(lf a,lf b){
	return fabs(a-b)<=1e-9 || a<b;
}
int main(){
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	scanf("%d%lf",&n,&v);
	for(int i=1;i<=n;i++)
		scanf("%lf",&a[i].x),a[i].y=i;
	sort(a+1,a+1+n);
	scanf("%d",&m);
	for(int i=1;i<=m;i++)
		scanf("%lf%lf",&b[i].x,&b[i].y);
	b[++m]=mp(inf,inf);
	sort(b+1,b+1+m);
	for(int i=1,j=1;i<=n;i++){
		lf e=a[i].x;
		lf mx=2*v*v*sin(e)*cos(e)/9.8;
		for(;;j++){
			if(mx<b[j].x){
				ans[a[i].y]=mp(mx,0.0);
				break;
			}
			lf t=b[j].x/(v*cos(e));
			lf my=v*sin(e)*t-4.9*t*t;
			if(cmp(my,b[j].y)){
				ans[a[i].y]=mp(b[j].x,my);
				break;
			}
		}
	}
	for(int i=1;i<=n;i++)
		printf("%.9lf %.9lf\n",ans[i].x,ans[i].y);
}
