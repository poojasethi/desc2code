
// Codeforces Beta Round #44 (Div. 2) E Cannon

#include <algorithm>
#include <cstring>
#include <cstdio>
#include <cmath>

using namespace std;

const double g = 9.8;

pair<double, double> wall[100000], ans[10000];
pair<double, int> angle[10000];

int main() {
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	
	int n;
	double V;
	scanf("%d%lf", &n, &V);
	for (int i = 0; i < n; i ++) {
		scanf("%lf", &angle[i].first);
		angle[i].second = i;
	}
	sort(angle, angle + n);
	int m;
	scanf("%d", &m);
	for (int i = 0; i < m; i ++)
		scanf("%lf%lf", &wall[i].first, &wall[i].second);
	sort(wall, wall + m);
	
	int j = 0;
	for (int i = 0; i < n; i ++) {
		double x = V * V * sin(angle[i].first * 2) / g;
		bool flag = false;
		while (j < m) {
			if (wall[j].first > x) break;
			double t = wall[j].first / (V * cos(angle[i].first));
			double y = V * sin(angle[i].first) * t - g * t * t / 2;
			if (y <= wall[j].second) {
				flag = true;
				ans[angle[i].second] = make_pair(wall[j].first, y);
				break;
			}
			j ++;
		}
		if (! flag) {
			ans[angle[i].second] = make_pair(x, 0.0);
		}
	}
	
	for (int i = 0; i < n; i ++)
		printf("%.9lf %.9lf\n", ans[i].first, ans[i].second);
	
	return 0;
}
