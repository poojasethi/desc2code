#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

const double g = 9.8;
int n,m;
double v0;
const int MAXN = 10101;
const int MAXM = 101010;

double a[MAXN],x[MAXN],y[MAXN];

pair<double , double >wall[MAXM];

int pos[MAXN];

bool cmp(int x,int y)
{
	return a[x] < a[y];
}

int main()
{
	scanf("%d",&n);
	//cout<<v0<<endl;
	scanf("%lf",&v0);
	for(int i = 0; i < n; ++i){
		scanf("%lf",&a[i]);
		pos[i] = i;
	}
	sort(pos,pos + n,cmp);
	scanf("%d",&m);
	for(int i = 0; i < m; ++i)
	{
		scanf("%lf%lf",&wall[i].first,&wall[i].second);
	}
	sort(wall,wall + m);
	
//	cout<<n<<" "<<v0<<" "<<m<<endl;
	
	int j = 0;
	for(int it = 0; it < n; ++it)
	{
		int i = pos[it];
		double vx = v0 * cos(a[i]);
		double vy = v0 * sin(a[i]);
		double w = vy * vx * 2 / g;
		double t,h;
		while(j < m && wall[j].first < w)
		{
			t = wall[j].first / vx;
			h = vy * t - t * t * g / 2;
			if(h <= wall[j].second)
				break;
			++j;
		}
		if(j < m && wall[j].first < w){
			x[i] = wall[j].first;
			y[i] = h;
		}
		else
		{
			x[i] = w;
			y[i] = 0;
		}
	}
	for(int i = 0; i < n; ++i)
		printf("%.9lf %.9lf\n",x[i],y[i]);
	
	return 0;
}
