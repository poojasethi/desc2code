#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
struct rec
{
	double thta;int id;
	double x,y;
}p[10010];
struct point
{
	double x,y;
}q[100010];
int n,m;
double V;
inline bool cmp1(rec a,rec b){return a.thta<b.thta;}
inline bool cmp2(rec a,rec b){return a.id<b.id;}
inline bool cmp3(point a,point b){return a.x<b.x;}
inline double sqr(double x){return x*x;}
inline double Y(double x,double thta)
{
	return -4.9/sqr(V)/sqr(cos(thta))*sqr(x)+tan(thta)*x;
}
int main()
{	scanf("%d%lf",&n,&V);
	for(int i=1;i<=n;i++)
	{
		scanf("%lf",&p[i].thta);p[i].id=i;
	}
	sort(p+1,p+n+1,cmp1);
	scanf("%d",&m);
	for(int i=1;i<=m;i++) scanf("%lf%lf",&q[i].x,&q[i].y);
	sort(q+1,q+m+1,cmp3);
	int j=1;
	for(int i=1;i<=m;i++)
	{
		for(;j<=n&&Y(q[i].x,p[j].thta)<=0;j++)
		{
			p[j].x=sqr(V)*cos(p[j].thta)*sin(p[j].thta)/4.9;p[j].y=0;
		}
		if (j>n) break;
		double t;
		for(;j<=n&&(t=Y(q[i].x,p[j].thta))<=q[i].y;j++)
		{
			p[j].x=q[i].x;p[j].y=t;
		}
		if (j>n) break;
	}
	for(;j<=n;j++)
	{
		p[j].x=sqr(V)*cos(p[j].thta)*sin(p[j].thta)/4.9;p[j].y=0;
	}
	sort(p+1,p+n+1,cmp2);
	for(int i=1;i<=n;i++) printf("%lf %lf\n",p[i].x,p[i].y);
	return 0;
}
