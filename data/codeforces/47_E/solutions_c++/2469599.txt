#include<cstdio>
#include<cmath>
#include<utility>
#include<algorithm>
#define fi first
#define se second
#define mp make_pair
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=101000;
const double eps=1e-8,g=9.8;
typedef pair<double,double> node;
typedef pair<double,int> canno;
int n,m;
double v;
canno a[mn];
node w[mn],ans[mn];
int main()
{
    scanf("%d%lf",&n,&v);
    int i,j,k;
    fo(i,1,n)scanf("%lf",&a[i].fi),a[i].se=i;
    scanf("%d",&m);
    fo(i,1,m)scanf("%lf%lf",&w[i].fi,&w[i].se);
    sort(a+1,a+1+n);
    sort(w+1,w+1+m);
    double vx,vy,t,y;
    node now;
    j=1;
    fo(i,1,n)
    {
        vx=v*cos(a[i].fi),vy=v*sin(a[i].fi);
        k=a[i].se;
        for(;j<=m;++j)
        {
            t=w[j].fi/vx,y=vy*t-g*t*t/2;;
            if(y<eps){ans[k]=mp(2*vx*vy/g,0);break;}
            if(y<w[j].se+eps){ans[k]=mp(w[j].fi,y);break;}
        }
        if(j>m)ans[k]=mp(2*vx*vy/g,0);
    }
    fo(i,1,n)printf("%.8lf %.8lf\n",ans[i].fi,ans[i].se);
    return 0;
}
