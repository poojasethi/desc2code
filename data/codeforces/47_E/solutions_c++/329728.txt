#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>

using namespace std;
#define mp make_pair
#define forn(i,n) for(int i=0;i<(int)n;++i)
#define X first
#define Y second
typedef double ld;
const int N = (int)1e5+10;
pair<ld, int> p[N];
pair<ld,ld> ans[N],wall[N];
int n,v,m,j;
const ld g = 9.8;
inline pair<ld,ld> st(int i, int j)
{
  ld t = wall[j].X/v/cos(p[i].X);
  return mp(wall[j].X,v*sin(p[i].X)*t-g*t*t/2);
}
int main()
{
  scanf("%d%d",&n,&v);
  forn(i,n)
    scanf("%lf",&p[i].X),p[i].Y=i;
  sort(p,p+n);
  forn(i,n)
    ans[p[i].Y]=mp(2*v*sin(p[i].X)/g*v*cos(p[i].X),0.);
  scanf("%d",&m);
  forn(i,m)
    scanf("%lf%lf",&wall[i].X,&wall[i].Y);
  sort(wall,wall+m);
  forn(i,n)
  {
    while (j<m && st(i,j).Y > wall[j].Y+1e-8)
      j++;
    pair<ld,ld> a=st(i,j);
    if (j<m && a.Y>=0)
      ans[p[i].Y]=min(ans[p[i].Y],a);
  }
  forn(i,n)
    printf("%.20lf %.20lf\n",ans[i].X,ans[i].Y);
  return 0;
}
