#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
struct wall{double x,y;};
const double pi=acos(-1);
const double g=9.8;
const double eps=1e-12;
const int nn=10011;
const int mm=100011;
int n,t;wall b[mm];
double a[nn],v,c[mm];
inline bool cmp(const wall &x,const wall &y){
	return x.x<y.x;
}
double get(double x,double s){
	return tan(s)*x-g*x*x/(2*v*v*cos(s)*cos(s));
}
void init(){
	scanf("%d%lf",&n,&v);
	for(int i=1;i<=n;++i)scanf("%lf",&a[i]);
	scanf("%d",&t);
	for(int i=1;i<=t;++i)scanf("%lf%lf",&b[i].x,&b[i].y);
	sort(b+1,b+t+1,cmp);
	for(int i=1;i<=t;++i){
		double l=0,r=pi/4;
		while(r-l>eps){
			double m=(l+r)/2;
			if(get(b[i].x,m)<b[i].y+eps)l=m;else r=m;
		}
		c[i]=max(c[i-1],l);
	}
}
void work(){
	for(int i=1;i<=n;++i){
		int l=1,r=t;
		while(l<r){
			int m=(l+r)/2;
			if(a[i]<c[m]+eps)r=m;else l=m+1;
		}
		if(get(b[l].x,a[i])<-eps||a[i]>c[l]+eps)
			printf("%.9lf %.9lf\n",v*v*sin(2*a[i])/g,0.0);
		else printf("%.9lf %.9lf\n",b[l].x,get(b[l].x,a[i]));
	}
}
int main(){
	init();
	work();
	return 0;
}
