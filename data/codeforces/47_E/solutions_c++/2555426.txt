#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<cmath>

using namespace std;

const int maxn=100010;
const double g=4.9;
const double eps=1e-8;

int n,m;

double v;

pair<double,int> z[maxn];

pair<double,double> y[maxn],ans[maxn];

int main()
{
	scanf("%d%lf",&n,&v);
	for (int a=1;a<=n;a++)
		scanf("%lf",&z[a].first),z[a].second=a;
	scanf("%d",&m);
	for (int a=1;a<=m;a++)
		scanf("%lf%lf",&y[a].first,&y[a].second);
	sort(z+1,z+n+1);
	sort(y+1,y+m+1);
	int nowp=1;
	for (int a=1;a<=n;a++)
	{
		double vx=v*cos(z[a].first);
		double vy=v*sin(z[a].first);
		while (nowp<=m)
		{
			double nowt=y[nowp].first/vx;
			double nowy=(vy-nowt*g)*nowt;
			if (nowy<-eps)
			{
				ans[z[a].second]=make_pair(vx*vy/g,0.0);
				break;
			}
			else
			{
				if (nowy<y[nowp].second+eps)
				{
					ans[z[a].second]=make_pair(y[nowp].first,nowy);
					break;
				}
				else nowp++;
			}
		}
		if (nowp>m) ans[z[a].second]=make_pair(vx*vy/g,0.0);
	}
	for (int a=1;a<=n;a++)
		printf("%.9lf %.9lf\n",ans[a].first,ans[a].second);

	return 0;
}
