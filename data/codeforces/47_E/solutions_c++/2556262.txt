#include<cstdio>
#include<cmath>
#define MAXN 101000
#include<algorithm>

int n,q[MAXN],m;

struct pt{double x,y;}p[MAXN],Ans[MAXN];

double V,G,a[MAXN];

double Height(int k,double x)
{
	double t=x/(V*cos(a[k]));
	return V*sin(a[k])*t-G*t*t/2;
}

bool cmp(pt a,pt b){return a.x<b.x;}

bool cmp2(int x,int y){return a[x]<a[y];}

int main()
{
	int i,j;
	scanf("%d%lf",&n,&V);
	for(i=0;i<n;q[i]=i++)scanf("%lf",a+i);
	scanf("%d",&m);
	for(i=0;i<m;i++)scanf("%lf%lf",&p[i].x,&p[i].y);
	std::sort(p,p+m,cmp);
	std::sort(q,q+n,cmp2);
	G=9.8;
	for(i=j=0;i<n;i++)
	{
		for(;j<m&&Height(q[i],p[j].x)-p[j].y>(1e-10);j++);
		if(Height(q[i],p[j].x)<(1e-10)||j==m)Ans[q[i]]=(pt){V*V*sin(2*a[q[i]])/G,0};
			else Ans[q[i]]=(pt){p[j].x,Height(q[i],p[j].x)};
	}
	for(i=0;i<n;i++)
		printf("%.6lf %.6lf\n",Ans[i].x,Ans[i].y);
	return 0;
}
