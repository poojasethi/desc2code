#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<map>
#define N 100001
using namespace std;
typedef long long LL;
typedef pair<int,int> p;
int maxx,maxy,w,ansx,ansy;
p xr[N],yr[N];
map<p,int> xt,yt;

int rev(int x){
	int ret=0;
	while (x){
		ret=ret*10+x%10;
		x/=10;
	}
	return ret;
}

int gcd(int a,int b){
	while (b){
		int temp=a%b; a=b; b=temp;
	}
	return a;
}
p getp(int a,int b){
	int temp=gcd(a,b);
	return p(a/temp,b/temp);
}

int main(){
	scanf("%d %d %d",&maxx,&maxy,&w);
	int maxi=maxx>maxy?maxx:maxy;
	for (int i=1;i<=maxi;i++){
		int revi=rev(i);
		xr[i]=getp(i,revi);
		yr[i]=getp(revi,i);
	}
	for (int i=1;i<=maxx;i++) xt[xr[i]]++;
	yt[yr[1]]++; ansx=-100010,ansy=-100100;
	int cnt=xt[yr[1]];
	for (int x=maxx,y=1;x>=0;x--){
		while (cnt<w && y<=maxy){
			y++;
			yt[yr[y]]++;
			cnt+=xt[yr[y]];
		}
		if (cnt>=w && y<=maxy && LL(x)*LL(y)<LL(ansx)*LL(ansy)) ansx=x,ansy=y;
		if (y>maxy) break;
		xt[xr[x]]--;
		cnt-=yt[xr[x]];
	}
	if (ansx>0) printf("%d %d",ansx,ansy); else printf("-1");
	puts(""); 
}
