#include <cstdio>
#include <algorithm>

using namespace std;

int r[100001],s[100001];

int gcd(int x,int y)
{
  if(y==0)
    return x;
  return gcd(y,x%y);
}

int main(void)
{
  int n,m,w;
  scanf("%d %d %d",&n,&m,&w);
  for(int i=1;i<=max(n,m);++i)
  {
    int k=i;
    while(k>0)
    {
      r[i]=r[i]*10+k%10;
      k/=10;
    }
  }
  int x=n,y=m,tot=0;
  bool flag=false;
  for(int i=1;i<=n;++i)
  {
    int k=r[i]/gcd(i,r[i]);
    for(int j=k;j<=m;j+=k)
    {
      if((long long)i*j==(long long)r[i]*r[j])
      {
        ++tot;
        ++s[j];
      }
    }
    while(tot-s[m]>=w)
    {
      tot-=s[m];
      --m;
    }
    if(tot>=w)
    {
      if(!flag)
        flag=true;
      if((long long)i*m<(long long)x*y)
      {
        x=i;
        y=m;
      }
    }
  }
  if(flag)
    printf("%d %d\n",x,y);
  else
    printf("-1\n");
  return 0;
}
