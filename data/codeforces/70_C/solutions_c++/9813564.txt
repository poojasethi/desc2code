#include<cstdio>
#include<iostream>

#include<string>

#include<set>
#include<vector>

#include<cstdlib>
#include<ctime>
#include<cmath>
using namespace std;
typedef long long LL;
#define N 100010

LL ans=-1;
int fi,fj;
int n,m,w;
pair<int,int> f1[N],f2[N];
map<pair<int,int>,int> d1,d2;

int rev(int x){
	int y=0;
	while(x){
		y=y*10+x%10;
		x/=10;
	}
	return y;
}

int gcd(int a,int b){
	return (!b)?a:gcd(b,a%b);
}

int main(){
	for(int i=1;i<=100000;i++){
		int r=rev(i);
		int d=gcd(i,r);
		f1[i]=make_pair(i/d,r/d);
		f2[i]=make_pair(r/d,i/d);
	}
	scanf("%d%d%d",&n,&m,&w);
	for(int i=1;i<=m;i++) d2[f2[i]]++;
	int j=m,sum=0;
	for(int i=1;i<=n;i++){
		sum+=d2[f1[i]];
		d1[f1[i]]++;
		while(sum>=w&&j){
			if(ans==-1||ans>(LL)(i)*j){
				ans=(LL)(i)*j;
				fi=i;
				fj=j;
			}
			sum-=d1[f2[j]];
			d2[f2[j]]--;
			j--;
		}
		if(!j) break;
	}
	if(ans==-1) puts("-1");
	else printf("%d %d\n",fi,fj);
	return 0;
}
