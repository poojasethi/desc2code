#include <iostream>
#include <map>
#define ll long long
using namespace std;
const ll inf = 1000000000;
typedef long double ld;
ll w,xx,yy,x,y,sum,p = inf,q = inf;
map <ld,int> mx,my;
int rev(int a) { int b = 0; while(a > 0) { b = (b*10)+(a%10); a /= 10; } return b; }
ld num(int a) { return ld(ld(a)/ld(rev(a))); }
ld num2(int a) { return ld(ld(rev(a))/ld(a)); }
int main () {
	cin >> xx >> yy >> w;
	for(y = 1 ; y <= yy ; y++) my[num2(y)]++; y--;
	for(x = 1 ; x <= xx ; x++) { mx[num(x)]++; sum += my[num(x)]; ld f = mx[num2(y)];
		while(sum-f >= w && y > 0) { sum -= f; my[num2(y)]--; y--; f = mx[num2(y)]; }
		if(sum >= w && p*q > x*y) { q = x; p = y; }
	} if(q == inf) cout << -1 << endl;
	else cout << q << ' ' << p << endl;
	return 0;
}
