#include <cstdio>
#include <cstdlib>

#include <iostream>
#include <map>

using namespace std;

int maxs, maxn, w;

double rev(int n)
{
    int answer = 0;
    while (n) answer *= 10, answer += n % 10, n /= 10;
    return answer;
}

int main()
{
#ifndef ONLINE_JUDGE
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif
    cin >> maxs >> maxn >> w;
    map<double, int> series, nums;
    for (int i = 1; i <= maxs; i++) series[i / rev(i)]++;
    int happy = 0, n;
    for (n = 1; n <= maxn && happy < w; n++)
        nums[rev(n) / n]++, happy += series[rev(n) / n];
    n--;
    if (happy < w)
    {
        cout << -1;
        return 0;
    }
    int anss = maxs, ansn = n;
    for (int s = maxs - 1; s >= 1; s--)
    {
        int t = s + 1;
        happy -= nums[t / rev(t)], series[t / rev(t)]--;
        for (; n < maxn && happy < w; n++)
        {
            t = n + 1;
            nums[rev(t) / t]++, happy += series[rev(t) / t];
        }
        if (happy < w) break;
        if (s * n < anss * ansn) anss = s, ansn = n;
    }
    cout << anss << " " << ansn;
    return 0;
}
