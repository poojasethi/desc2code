#include<cstdio>
#include<cstdlib>
#include<map>
#include<algorithm>
using namespace std;
int n,m,W,x,y,a1,a2,all;
map < pair<int,int> , int> M,M2;
int re(int t){
    int s = 0;
    while(t){ s = s*10+(t%10); t/= 10; }
    return s;    
}
int main(){
    scanf("%d%d%d",&n,&m,&W);
    for(int i=1;i<=n;i++){
       x = re(i); y = __gcd(i,x);
       M[ make_pair(i/y, x/y) ]++;         
    }
    a1 = a2 = 1000000; all = 0;
    for(int i=n,j=0;i>0;i--){
        x = re(i); y = __gcd(x,i);   
        while(j<m && all < W){
            ++j;
            x = re(j); y = __gcd(j,x);
            all += M[ make_pair(x/y,j/y) ];          
            M2[ make_pair(x/y,j/y) ]++;
        }    
        if(all >= W && (long long)(j)*(long long)(i) < (long long)(a1)*(long long)(a2)) a1 = i,a2 = j;
        x = re(i); y = __gcd(x,i);
        all -= M2[ make_pair(i/y,x/y) ];
        M[ make_pair(i/y,x/y) ]--;
    }
    if(a1 == 1000000) printf("-1\n");
    else printf("%d %d\n",a1,a2);
    //system("pause");
    return 0;
}
