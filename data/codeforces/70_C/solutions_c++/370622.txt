#include <stdio.h>
#include <iostream>
#include <map>
#include <algorithm>

#define oo 100009
//(1<<20)

using namespace std;

map<pair<int,int> ,int>mapx,mapy;

int rev(int n)
{
    int mod=0;
    while(n)
    {
        mod=mod*10 + n%10;
        n/=10;
    }
    return mod;
}
pair<int,int> gen(int n)
{
    int r=rev(n);
    int g=__gcd(n,r);
    return make_pair(n/g,r/g);
}
pair<int,int> flip(pair<int,int>p)
{
    return make_pair(p.second,p.first);
}

int bestx=oo,besty=oo;
long long prod;

int main(void)
{
    int X,Y,w,i,j,lucky;
    scanf("%d %d %d",&X,&Y,&w);
    for(i=1;i<=Y;i++)
    {
        mapy[flip(gen(i))]++;
    }
    for(i=1,j=Y,lucky=0;i<=X;i++)
    {
        lucky+=mapy[gen(i)];
        mapx[gen(i)]++;

        while(lucky-mapx[flip(gen(j)) ] >=w)
        {
            mapy[flip(gen(j))]--;
            lucky-=mapx[flip(gen(j)) ];
            j--;
        }

        if(lucky>=w && (long long) i*j < (long long)bestx*besty )
        {
            bestx=i;besty=j;
        }
    }
    if(bestx==oo) puts("-1");
    else cout<<bestx<<" "<<besty<<endl;


}
