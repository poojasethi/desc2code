#include <cstdio>
#include<iostream>
#include<map>
using namespace std;

typedef long long LL;
typedef pair<int, int> PI;
typedef map<PI,int> MI;

PI p[110000];
MI mx,my;
int r[110000];
int Gcd(int a,int b)
{
	return b==0? a:Gcd(b,a%b);
}
int Riv(int a)
{
	int i=0;
	while(a)
	{
		i=i*10+a%10;
		a/=10;
	}
	return i;
}
PI Tre(PI p)
{
	PI q(p.second,p.first);
	return q;
}
int main()
{
	int maxx,maxy,w,i,j,ii,jj,k,d;
	bool flag;
	while(scanf("%d%d%d",&maxx,&maxy,&w)!=EOF)
	{
		my.clear();
		mx.clear();

		j=maxx;
		if(maxy>j) j=maxy;
		for(i=1;i<=j;i++)
		{
			r[i]=Riv(i);
			d=Gcd(i,r[i]);
			p[i]=make_pair(i/d,r[i]/d);
		}
		for(i=1;i<=maxy;i++)
			my[p[i]]++;

		ii=maxx,jj=maxy;
		flag=false;
		for(i=1,k=0,j=maxy;i<=maxx;i++)
		{
			mx[p[i]]++;
			k+=my[Tre(p[i])];
			while(j>0&&k>=w)
			{
				flag=true;
				if((LL)ii*jj>(LL)i*j)
				{
					ii=i;
					jj=j;
				}
				my[p[j]]--;
				k-=mx[Tre(p[j])];
				j--;
			}
		}

		if(!flag) printf("-1\n");
		else printf("%d %d\n",ii,jj);
	}
	return 0;
}
