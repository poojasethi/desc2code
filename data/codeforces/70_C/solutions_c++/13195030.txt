#include<bits/stdc++.h>
using namespace std;

int rev( int x ){
    int ans = 0, y;
    while( x ){
        y = x % 10;
        ans = ans *10 + y;
        x /= 10;
    }
    return ans;
}
pair< int , int > f( int x ){
    int u = __gcd( x , rev( x ) );
    return make_pair( x / u , rev( x )/u  );
} 
pair< int , int > g( int x ){
    int u = __gcd( x , rev( x ) );  
    return make_pair( rev( x )/u  , x / u );
} 

int main() {
    int xm , ym, w , ci , cj;
    cin>>xm>>ym>>w;
    map< pair< int , int >, int> M1, M2;
    int cnt = 0;
    long long mini = LLONG_MAX;
    
    for ( int i = 1 ; i <= ym ; i++ ) M1[ g( i ) ]++;
    
    for ( int i = 1, j = ym; i <= xm ; i++ ){
        cnt += M1[ f( i ) ];
        M2[ f(i) ]++;
        while ( j - 1 >= 0 && cnt - M2[ g( j ) ] >= w ){
            cnt -= M2[ g( j ) ];
            M1[ g( j ) ]--;
            j--;
        }
        if( mini > i*j && cnt >= w ){
            mini = i*j;
            ci = i;
            cj = j;
        } 
    }
    if( mini == LLONG_MAX ) cout<<-1<<endl;
    else cout<<ci<<" "<<cj<<endl;
    return 0;
}
