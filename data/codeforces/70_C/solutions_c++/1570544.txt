#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<map>
#include<vector>
using namespace std;
const int N = 100005;
#define mp(x,y) make_pair(x,y)
typedef pair<int,int> per;
typedef __int64 ll;
int a[10],m,n,k;
map<per,int>lzs,cnt;
per xs[N],ys[N];
int gcd(int x,int y)
{
	int z;
	while(y)
	{
		z=x%y;
		x=y;
		y=z;
	}
	return x;
}
int f(int x)
{
	int v=0;
	while(x){v=v*10+x%10;x/=10;}
	return v;
}
int main()
{
	int i,j,sz=1,c,d,e;
	scanf("%d%d%d",&m,&n,&k);
	for(i=1;i<=m;i++)
	{
		int tp=f(i);
		e=gcd(i,tp);
		c=i/e;
		d=tp/e;
		xs[i]=mp(c,d);
		ys[i]=mp(d,c);
		lzs[xs[i]]++;
	}
	for(;i<=n;i++)
	{
		int tp=f(i);
		e=gcd(i,tp);
		c=i/e;
		d=tp/e;
		ys[i]=mp(d,c);
	}
	int r1=-1,r2,t1,t2,all=0;
	for(i=m,j=0;i>0;i--)
	{
		while(all<k&&j<n)
		{
			j++;
			all+=lzs[ys[j]];
			cnt[ys[j]]++;
		}
		//printf("i:%d all:%d \n",i,all);
		if(all>=k)
		{
			//printf("i:%d j:%d all:%d \n",i,j,all);
			if(r1==-1||(ll)r1*r2>(ll)i*j)
			{
				r1=i;
				r2=j;
			}
		}
		lzs[xs[i]]--;
		all-=cnt[xs[i]];
	}
	if(r1==-1)puts("-1");
	else printf("%d %d\n",r1,r2);
	return 0;
}
