#include<algorithm>
#include<cstdio>
//             Last Change:  2011-03-30 15:08:50
#include<map>
#define MX 100003
#define MP make_pair
using namespace std;
map< pair<int,int>,int >h,d;
int mx,my,w,a1,a2;
pair<int,int> fx[MX],fy[MX];
int rev(int x){
	int ret=0;
	while(x)ret=(ret<<1)+(ret<<3)+x%10,x/=10;
	return ret;
}
pair<int,int> frac(int x,int y){
	int tmp=__gcd(x,y);
	return MP(x/tmp,y/tmp);
}
int main(){
	scanf("%d%d%d",&mx,&my,&w);a1=a2=MX;
	for(int i=1;i<=mx;++i)fx[i]=frac(i,rev(i));
	for(int i=1;i<=my;++i)fy[i]=frac(rev(i),i);
	for(int i=1;i<=mx;++i)++h[fx[i]];
	for(int i=mx,j=0,v=0;i;--i){
		for(;v<w&&j<=my;){
			++j;
			v+=h[fy[j]];
			++d[fy[j]];
		}
		if(v>=w&&(long long)i*j<(long long)a1*a2)a1=i,a2=j;
		--h[fx[i]],v-=d[fx[i]];
	}
	if(a1<=mx&&a2<=my)printf("%d %d\n",a1,a2);
	else puts("-1");
	return 0;
}
