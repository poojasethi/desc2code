#include <iostream>
#include <map>
#include <algorithm>

using namespace std;

typedef pair<int,int> pii;
const int MN=100000+10;
pii xs[MN],ys[MN];
map<pii,int> xc,yc;
int mx,my,w;
int bx=MN,by=MN;
long long tt;

int rev(int a){
	int res=0;
	while(a)
		res=res*10+a%10,a/=10;
	return res;
}

pii make(int a,int b){
	int temp=__gcd(a,b);
	return pii(a/temp,b/temp);
}

int main(){
	cin>>mx>>my>>w;
	for(int i=1;i<=mx;i++) 
		xs[i]=make(i,rev(i));
	for(int i=1;i<=my;i++) 
		ys[i]=make(rev(i),i);
	for(int i=1;i<=mx;i++) 
		xc[xs[i]]++;
	yc[ys[1]]++;
	tt=xc[ys[1]];
	for(int x=mx,y=1;x>=1;x--){
		while(tt<w && y<=my){
			y++;
			yc[ys[y]]++;
			tt+=xc[ys[y]];
		}
		if(tt>=w && y<=my &&(long long)x*y<(long long)bx*by)
			bx=x,by=y;
		xc[xs[x]]--;
		tt-=yc[xs[x]];
	}
	if(bx==MN)
		cout<<-1<<endl;
	else
		cout<<bx<<" "<<by<<endl;
	//system("pause");
	return 0;
}
