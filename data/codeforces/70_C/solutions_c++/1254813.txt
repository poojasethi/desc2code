#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int rev(int x) {
	int v=0;
	while(x%10==0) x/=10;
	while(x>0) {v=v*10+x%10; x/=10; }
	return v;
}
int rr[100005];
int cnty[100005];
int main(void) {
	int x, y, w, ax, ay, total=0, flag=0;
	scanf("%d%d%d", &x, &y, &w);
	ax=x, ay=y;
	for(int i=1;i<=x||i<=y;i++) rr[i]=rev(i);
	for(int i=1;i<=x;i++) {
		int k = rr[i]/__gcd(i, rr[i]);
		for(int j=k;j<=y;j+=k) {
			if(i*1LL*j==rr[i]*1LL*rr[j]) {
				total++;
				cnty[j]++;
			}
		}
		while(y>0 && total-cnty[y]>=w) {
			total -= cnty[y--];
			//printf("(%d,%d) total=%d nxt=%d\n", i, y, total, cnty[y]);
		}
		if(total>=w) {
			flag=1;
			if(ax*1LL*ay > i*1LL*y) {
				ax=i; ay=y;
			}
		}
	}
	if(!flag) printf("-1\n");
	else printf("%d %d\n", ax, ay);
	return 0;
}
