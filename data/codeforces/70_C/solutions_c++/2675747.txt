#include<iostream>
#include<cstdio>
#include<map>

using namespace std;

int n , m , w , p , x = -1 , y;
long long ans = ( long long ) 1e18;

map<pair<int,int>,int> A , B;

int gcd( int a , int b ) {
	if ( b == 0 ) return a;
	return gcd( b , a % b );
}

pair<int,int> caln_A( int r ) {
	int x = r , c = 0;
	while ( x > 0 ) {
		c = c * 10 + x % 10;
		x /= 10;
	}
	int d = gcd( r , c );
	r /= d;
	c /= d;
	return make_pair( r , c );
}

pair<int,int> caln_B( int r ) {
	pair<int,int> ret = caln_A( r );
	swap( ret.first , ret.second );
	return ret;
}

int main() {
	scanf( "%d%d%d" , &n , &m , &w );
	
	for ( int i = 1 ; i <= n ; i++ )
		A[caln_A( i )]++;
	
	for ( int i = n , j = 0 ; i >= 1 ; i-- ) {
		while ( p < w && j < m ) {
			j++;
			pair<int,int> now = caln_B( j );
			B[now]++;
			p += A[now];
		}
		if ( p < w ) continue;
		if ( ( long long ) i * j <= ans ) {
			ans = ( long long ) i * j;
			x = i;
			y = j;
		}
		pair<int,int> now = caln_A( i );
		p -= B[now];
		A[now]--;
	}
	if ( x == -1 ) printf( "-1\n" );
		else printf( "%d %d\n" , x , y );
	return 0;
}
