using namespace std;
const int MAX_N=100*1000;
const LL INF=(1LL<<30)*(1LL<<30);
map <double,LL> M1;
map <double,LL> M2;
LL Mx,My;
LL W;
LL rev(LL x)
{
	LL ret=0;
	while(x)
	{
		ret*=10;
		ret+=x%10;
		x/=10;
	}
	return ret;
}
int main()
{
	cin >> Mx >> My >> W;
	for(LL i=1;i<=Mx;i++)
		M1[(double)i/(double)rev(i)]++;
	
	LL Ansa=-1;
	LL Ansb=-1;
	LL X=Mx;
	LL Y=0;
	LL NOW=0;
	while(X>=1 && Y<=My)
	{
		if(NOW<W)
		{
			Y++;
			double TEMP=(double)rev(Y)/(double)Y;
			M2[TEMP]++;
			NOW+=M1[TEMP];
		}
		else
		{
			if(Ansa==-1 || Ansa*Ansb>X*Y)
					Ansa=X,Ansb=Y;
			double TEMP=(double)X/(double)rev(X);
			NOW-=M2[TEMP];
			M1[TEMP]--;
			X--;
		}
	}
	if(Ansa==-1)
			cout<<-1<<endl;
	else
			cout<<Ansa<<" "<<Ansb<<endl;
}
