#include<stdio.h>
#include<algorithm>
#include<string>
#include<map>
using namespace std;
#define rep(i,n) for (int i=1;i<=n;i++)
typedef long long ll;

map<double,int> a,b;
const ll oo=10000000000000000;
int maxx,maxy,q,kqx,kqy;
long long kq;

int dao(int i)
{
	int j=0,l=i;
	while (i) j=j*10+i%10,i/=10;
	return j;
}

void xuli()
{
	int i=0,j=maxy,dem=0;
	rep(j,maxy) ++b[dao(j)/double(j)];
	rep(i,maxx)
	{
		dem+=b[double(i)/dao(i)];
		a[double(i)/dao(i)]++;
		while (j>0&&dem-a[dao(j)/double(j)]>=q)
		{
			dem-=a[dao(j)/double(j)];
			b[dao(j)/double(j)]--;
			j--;
		}
		if (dem>=q&&ll(i)*j<kq) kq=ll(i)*j,kqx=i,kqy=j;
	}
}

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("test.inp","r",stdin);
	#endif // ONLINE_JUDGE
	scanf("%d%d%d",&maxx,&maxy,&q);
	kq=oo;
	xuli();
	if (kq!=oo) printf("%d %d",kqx,kqy);
	else printf("-1");
}
