#include<stdio.h>
#include<math.h>
#include<string.h>
#include<algorithm>
#include<map>
using namespace std;
#define LL long long
map<double,LL>mp1,mp2;
LL rev(LL x)
{
  LL ans=0;
  while (x){
    ans=ans*10+x%10;
    x/=10;
  }
  return ans;
}
int main()
{
  LL maxx,maxy,w,cnt,i,tx,ty,ax,ay;
  double tmp;
  scanf("%I64d%I64d%I64d",&maxx,&maxy,&w);
  cnt=0;
  for (i=1;i<=maxx;i++) mp1[1.0*i/rev(i)]++;
  for (i=1;i<=maxy;i++) cnt+=mp1[1.0*rev(i)/i];
  if (cnt<w) {printf("-1\n"); return 0; }
  tx=maxx; ty=1; 
  cnt=mp1[1.0]; mp2[1.0]++;
  ax=maxx; ay=maxy;
  while (tx>=1&&ty<=maxy)
  {
    if (cnt>=w){
      if (tx*ty<ax*ay) ax=tx,ay=ty;
      tmp=1.0*tx/rev(tx);
      cnt-=mp2[tmp]; mp1[tmp]--; tx--;
    }
    else{
      ty++;
      tmp=1.0*rev(ty)/ty;
      cnt+=mp1[tmp]; mp2[tmp]++;
    }
  }
  printf("%I64d %I64d\n",ax,ay);
  return 0;
}
