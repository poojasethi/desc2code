#include <iostream>
#include <map>
using namespace std;
typedef pair<int,int> pie;
const int MN=100000+10;
pie xs[MN],ys[MN];
map<pie,int> xc,yc;
int mx,my,w;
int bx=MN,by=MN;
long long total;
int rev(int a)
{
	int res=0;
	while (a)
		res=res*10+a%10,a/=10;
	return res;
}
int gcd(int a,int b)
{
	return b==0?a:gcd(b,a%b);
}
pie make(int a,int b)
{
	int temp=gcd(a,b);
	return pie(a/temp,b/temp);
}
int main()
{
	cin>>mx>>my>>w;
	for (int i=1;i<=mx;i++) xs[i]=make(i,rev(i));
	for (int i=1;i<=my;i++) ys[i]=make(rev(i),i);
	for (int i=1;i<=mx;i++) xc[xs[i]]++;
	yc[ys[1]]++;
	total=xc[ys[1]];
	for (int x=mx,y=1;x>=1;x--)
	{
		while (total<w && y<=my)
		{
			y++;
			yc[ys[y]]++;
			total+=xc[ys[y]];
		}
		if (total >=w && y<=my && (long long)x*y<(long long) bx*by)
			bx=x,by=y;
		xc[xs[x]]--;
		total-=yc[xs[x]];
	}
	if (bx==MN)
		cout<<-1<<endl;
	else
		cout<<bx<<" "<<by<<endl;
	return 0;

}
