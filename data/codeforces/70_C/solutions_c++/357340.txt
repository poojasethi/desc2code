#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <map>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define mp make_pair
#define ll long long
#define lim 100001

ll xmx,ymx,w,i,x,y,x2,y2,n,d,s;
map <pair<ll,ll>,ll> M,M2;
ll R[lim];

int main()
{
  cin>>xmx>>ymx>>w;
  for(i=1;i<lim;i++)
  {
    for(x=i,R[i]=0;x;x/=10)
      R[i]=10*R[i]+x%10;
  }
  for(x=1;x<=xmx;x++)
  {
    n=R[x]/__gcd(x,R[x]);
    d=x/__gcd(x,R[x]);
    M[mp(n,d)]++;
  }
  x2=-1;
  for(x=xmx,y=s=0;x>=1;x--)
  {
    while(s<w && y<ymx)
    {
      y++;
      n=y/__gcd(y,R[y]);
      d=R[y]/__gcd(y,R[y]);
      s+=M[mp(n,d)];
      M2[mp(n,d)]++;
    }
    if(s<w)
      break;
    if(x==xmx || x*y<x2*y2)
    {
      x2=x;
      y2=y;
    }
    n=R[x]/__gcd(x,R[x]);
    d=x/__gcd(x,R[x]);
    M[mp(n,d)]--;
    s-=M2[mp(n,d)];
  }
  if(x2==-1)
    printf("-1");
  else
    cout<<x2<<" "<<y2;
  return 0;
}
