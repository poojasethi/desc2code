def readint():
    return map(int, raw_input().split())


def readstring():
    return raw_input().split()

n, L, x, y = readint()
array = readint()
marks = {}
todo = {}
for i in array:
    marks[i] = 1

findx = False
findy = False
for i in array:
    if i + x in marks:
        findx = True
    elif i + x <= L:
        todo[i + x] = 1
    if i - x in marks:
        findx = True
    elif i - x >= 0:
        todo[i - x] = 1

for i in array:
    if i + y in marks:
        findy = True
    elif i + y <= L:
        if i + y in todo:
            todo[i + y] += 2
        else:
            todo[i + y] = 2
    if i - y in marks:
        findy = True
    elif i - y >= 0:
        if i - y in todo:
            todo[i - y] += 2
        else:
            todo[i - y] = 2

if findx and findy:
