#include <iostream>
#include <cstdio>
#include <map>
using namespace std;
typedef long long int l;
#define FO(i,n) for (int i=0;i<n;i++)
#define pri printf
int main ()
{
//	freopen("test.txt","r",stdin);
	int n,le,x,y,t,c,ans=0 ; cin >> n >> le >> x >> y;
	bool xx=0,yy=0,found=0;
	map<int,bool> m;
	FO(i,n){
		scanf ("%d", &t);
		if (m[t-x]) xx = 1;
		if (m[t-y]) yy = 1;
		if (!found){
			if (t - y == x) ans = y;
			if (m[t - y + x]){
				if (t + x < le) ans = t + x;
				if (t - y > 0) ans = t - y;
			}
			if (m[t - x - y]) ans = t - y;
			if (ans)found = 1;
		}
		m[t] = 1;
	}
	if (xx && yy) return pri("0"),0;
	if (!xx && yy) return pri("1 %d",x),0;
	if (xx && !yy) return pri("1 %d",y),0;
	if (found) return pri("1 %d",ans),0;
	pri("2 %d %d",x,y);
    return 0;
}
