#include <iostream>
#include <string.h>
#include <stdio.h>
#include <algorithm>
#include <map>
using namespace std;
int n,l,x,y,a[110000];
map<int,int>mp;
int main()
{
	scanf("%d%d%d%d",&n,&l,&x,&y);
	mp.clear();
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		mp[a[i]]=1;
	}
	sort(a+1,a+1+n);
	int sx=0,sy=0;
	for (int i=1;i<=n;i++)
	{
		sx|=(mp[a[i]-x] || mp[a[i]+x]);
		sy|=(mp[a[i]-y] || mp[a[i]+y]);
	}
	if (sx && sy)
		printf("0\n");
	else if (sx || sy)
		printf("1\n%d\n",sx?y:x);
	else
	{
		int ans=0;
		for (int i=1;i<=n;i++)
		{
			if (a[i]+x>0 && a[i]+x<l)
			{
				if (mp[a[i]+x-y] || mp[a[i]+x+y])
					ans=a[i]+x;
			}
			if (a[i]-x>0 && a[i]-x<l)
			{
				if (mp[a[i]-x-y] || mp[a[i]-x+y])
					ans=a[i]-x;
			}
		}
		if (!ans)
			printf("2\n%d %d\n",x,y);
		else
			printf("1\n%d\n",ans);
	}
	return 0;
}
