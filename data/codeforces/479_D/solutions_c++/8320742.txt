#include <cstdio>
#include <map>

int v[100001]; std::map <int, int> existe;
bool teste (int n, int a){for (int i = 0; i < n; i++) if (existe[v[i] - a]) return true; return false;}
void print (int n, int m, int a, int b){

    for (int i = 0; i < n; i++){
        int k = v[i] - a; if (k >= 0) if (existe[k + b] || existe[k - b] ){ printf ("%d\n%d", 1, k); return; }
        k = v[i] + a;
        if (k <= m) if (existe[k + b] || existe[k - b]){ printf ("%d\n%d", 1, k); return; }
    }
    printf ("%d\n%d %d", 2, a, b);
}
main(){
    int n, m, a, b;
    scanf ("%d %d %d %d", &n, &m, &a, &b);
    for (int i = 0; i < n; i++){ scanf ("%d", v + i); existe[v[i]] = 1; }
    bool x = teste (n, a), y = teste (n, b);
    if (x && y) printf ("0"); else if (x && !y) printf ("%d\n%d", 1, b); else if (!x && y) printf ("%d\n%d", 1, a); else{ print(n, m, a, b);}
}
