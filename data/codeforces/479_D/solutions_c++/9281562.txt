#include <bits/stdc++.h>

#define PB push_back
 

typedef long long int LL;

using namespace std;

int main()
{
	int n,l,x,y,ans=2,i,num,flag=0,f;
	cin>>n>>l>>x>>y;
	if(x>y) swap(x,y);
	map<int,int> M;
	for(i=0;i<n;i++)
	{
		cin>>num;
		M[num]=1;
		if(!flag && M[num+x-y]==1) {flag=1; if(num+x<l) f=num+x; else if(num-y>0) f=num-y; else flag=0;}
		if(!flag && M[num-x-y]==1) {flag=1; if(num-x>0) f=num-x; else flag=0;}
		if(M[num-x]==1) x=0;
		if(M[num-y]==1) y=0;
	}
	vector<int> V;
	if(y==0) ans--;
	else V.PB(y);
	if(x==0 || M[y-x]==1) ans--;
	else V.PB(x);
	if(ans==2 && flag==1) {cout<<1<<endl<<f<<endl; return 0;}
	cout<<ans<<endl;
	for(i=0;i<ans;i++) cout<<V[i]<<" ";
	cout<<endl;
	return 0;
}
