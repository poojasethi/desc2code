#include<cstdio>
#include<algorithm>
#include<cstring>
#include<map>
using namespace std;
map<int,int>mp;
int a[100010];
int main()
{
	int n,l,x,y,i,l1,l2;
	int fx=0,fy=0,fxy=0,fyx=0;
	scanf("%d%d%d%d",&n,&l,&x,&y);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		mp[a[i]]=1;
	}
	for(i=1;i<=n;i++)
	{
		if(mp[a[i]+x]==1)
			fx=1;
		if(mp[a[i]+y]==1)
			fy=1;
		if(!fxy && mp[a[i]+x+y]==1)
		{
			fxy=1;
			l1=a[i]+x;
		}
		if(!fyx && mp[a[i]+y-x]==1)
		{
			if(a[i]+y<=l)
			{
				l2=a[i]+y;
				fyx=1;
			}
			else if(a[i]-x>=0)
			{
				l2=a[i]-x;
				fyx=1;
			}
		}
	}
	if(fx&&fy)	printf("0\n");
	else if(fy)	printf("1\n%d\n",x);
	else if(fx) printf("1\n%d\n",y);
	else if(fxy) printf("1\n%d\n",l1);
	else if(fyx) printf("1\n%d\n",l2);
	else printf("2\n%d %d\n",x,y);
	return 0;
}
