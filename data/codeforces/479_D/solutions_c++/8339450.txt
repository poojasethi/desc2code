#include <cstdio>
#include <iostream>
#include <set>

using namespace std;

set <int> s;

int main() {
    int n, l, x, y, a, dl;
    scanf("%d%d%d%d", &n, &l, &x, &y);
    bool f1 = 0, f2 = 0, f3 = 0;
    for (int i = 0; i < n; i++) {
        scanf("%d",&a);
        if (a-x >= 0 && s.count(a-x)) f1 = 1;
        if (a-y >= 0 && s.count(a-y)) f2 = 1;
        if (a-x-y >= 0 && s.count(a-x-y)) {f3 = 1; dl = a-y;}
        if (a-y >= 0 && s.count(a-y+x)) {f3 = 1; dl = a-y;}
        if (a+x <= l && s.count(a+x-y)) {f3 = 1; dl = a+x;}
        s.insert(a);
    }
    if (f1 == 1 && f2 == 1) {printf("0"); return 0;}
    if (f1) {printf("1\n%d",y); return 0;}
    if (f2) {printf("1\n%d",x); return 0;}
    if (f3) {printf("1\n%d",dl); return 0;}
    printf("2\n%d %d",x,y);
    return 0;
}
