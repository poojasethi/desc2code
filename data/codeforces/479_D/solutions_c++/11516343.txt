#include<bits/stdc++.h>
using namespace std;
int i,n,la,a,f1=0,f2=0,f3=0,dl,x,y;
set <int> s;
main()
{
    scanf("%d%d%d%d",&n,&la,&x,&y);
    for (i=1;i<=n;i++)
        {
            scanf("%d",&a);
            if (a-x>=0&&s.count(a-x)) {f1=1;}
            if (a-y>=0&&s.count(a-y)) f2=1;
            if (a-x-y>=0&&s.count(a-x-y)) {f3=1;dl=a-x;}
            if (a-y>=0&&s.count(a-y+x)) {f3=1;dl=a-y;}
            if (a+x<=la&&s.count(a+x-y)) {f3=1;dl=a+x;}
            s.insert(a);
        }
    if (f1&&f2) return printf("0"),0;
    if (f1) {cout<<1<<endl<<y;return 0;}
    if (f2) {cout<<1<<endl<<x;return 0;}
    if (f3) {cout<<1<<endl<<dl;return 0;}
    cout<<2<<endl<<x<<' '<<y;
}
