#include <iostream>
#include <cstdio>
#include <set>
using namespace std;
set <int> s;
int main()
{
    int n,l,x,y,a,res;
    scanf("%d%d%d%d",&n,&l,&x,&y);
    bool f1=false,f2=false,f3=false;
    for (int i=1; i<=n; i++)
	{
        scanf("%d",&a);
        if (a>=x && s.count(a-x)) f1=true;
        if (a>=y && s.count(a-y)) f2=true;
        if (a>=x+y && s.count(a-x-y)) {f3=true; res=a-x;}
        if (a>=y && s.count(a-y+x)) {f3=true; res=a-y;}
        if (a<=l-x && s.count(a-y+x)) {f3=true; res=a+x;}
        s.insert(a);
    }
    if (f1 && f2) {printf("0"); return 0;}
    if (f1) {printf("1\n%d",y); return 0;}
    if (f2) {printf("1\n%d",x); return 0;}
    if (f3) {printf("1\n%d",res); return 0;}
    printf("2\n%d %d",x,y);
    return 0;
}
