#include<iostream>
#include<set>
using namespace std;
set<int>s;
int main()
{int n,la,x,y,dl,a;
 cin>>n>>la>>x>>y;
 bool f1=0,f2=0,f3=0;
  for(int i=1;i<=n;i++)
   {cin>>a;
    if(a-x>=0 && s.count(a-x)){f1=1;}
    if(a-y>=0 && s.count(a-y)){f2=1;}
    if(a-x-y>=0 && s.count(a-x-y)){f3=1;dl=a-y;}
    if(a-y>=0 && s.count(a-y+x)){f3=1;dl=a-y;}
    if(a+x<=la && s.count(a+x-y)){f3=1;dl=a+x;}
    s.insert(a);
  }
if(f1 && f2){cout<<0<<endl; return 0;}
if(f1){cout<<1<<endl<<y;return 0;}
if(f2){cout<<1<<endl<<x;return 0;}
if(f3){cout<<1<<endl<<dl;return 0;}
cout<<2<<endl<<x<<" "<<y;
return 0;
}
