#include<iostream>
#include<cstdio>
#include<cstring>
#include<map>
#include<algorithm> 
#define N 400100
using namespace std;
int x,y,n,l;
map<int,int> ma;
int a[N];bool fx,fy;
int main()
{
    scanf("%d%d%d%d",&n,&l,&x,&y);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]),ma[a[i]]=1;
    sort(a+1,a+1+n);
    for(int i=1;i<=n;i++)
    {
        if(ma[a[i]-x])fx=1;
        if(ma[a[i]-y])fy=1;
    }
    
    if(fx&fy){puts("0");return 0;}
    if(fx){printf("1\n%d",y);return 0;}
    if(fy){printf("1\n%d",x);return 0;}
    
    for(int i=1;i<=n;i++)
    {
        if(ma[a[i]+x-y]&&a[i]+x<=l)
        {printf("1\n%d",a[i]+x);return 0;}
        if(ma[a[i]-x+y]&&a[i]-x>0)
        {printf("1\n%d",a[i]-x);return 0;}
        if(ma[a[i]-x-y])
        {printf("1\n%d",a[i]-x);return 0;}
    }
    printf("2\n%d %d",x,y);
    return 0;
}
