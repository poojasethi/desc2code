#include<iostream>
#include<set>
using namespace std;

set<int> s;

int main()
{
	int n, l, x, y, a,d;
	cin >> n >> l >> x >> y;
	bool f1 = 0, f2 = 0, f3 = 0;
	for (int i = 0; i < n; i ++)
	{
		cin >> a;
		if (a - x >= 0 && s.count(a - x)) f1 = 1;
		if (a - y >= 0 && s.count(a - y)) f2 = 1;
		if (a - x - y >= 0 && s.count(a - x - y)) { f3 = 1; d = a - y; }
		if (a - y >= 0 && s.count(a - y + x)){ f3 = 1; d = a - y; }
		if (a + x <= l && s.count(a - y + x)){ f3 = 1; d = a + x; }
		s.insert(a);
	}
	if (f1 && f2){ cout << 0 << endl; return 0; }
	if (f1){ cout << 1 << endl << y<<endl; return 0; }
	if (f2){ cout << 1 << endl << x << endl; return 0; }
	if (f3){ cout << 1 << endl << d << endl; return 0; }
	cout << 2 << endl << x << " " << y << endl;
	return 0;
}
