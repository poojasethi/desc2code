#include <cstdio>
#include <algorithm>
#include <map>
using namespace std;
int a[100005],b,c,d,e,g,h,i,j,k,l,m,n;
map<int,int>f;
int main(){
	scanf("%d%d%d%d",&n,&m,&b,&c);
	for(i=1;i<=n;i++){
		scanf("%d",&a[i]);
		f[a[i]]=1;
	}
	for(i=1;i<=n;i++){
		if(f[a[i]+b]||f[a[i]-b]){g=1;break;}
	}
	for(i=1;i<=n;i++){
		if(f[a[i]+c]||f[a[i]-c]){h=1;break;}
	}
	if(g&&h){printf("0");return 0;}
	if(g&&!h){printf("%d\n%d",1,c);return 0;}
	if(!g&&h){printf("%d\n%d",1,b);return 0;}
	for(i=1;i<=n;i++){
		l=a[i]+b;
		if(l<=m&&(f[l-c]||f[l+c])){printf("%d\n%d",1,l);return 0;}
		l=a[i]-b;
		if(l>=0&&(f[l-c]||f[l+c])){printf("%d\n%d",1,l);return 0;}
	}
	printf("%d\n%d %d",2,b,c);
	return 0;
}
