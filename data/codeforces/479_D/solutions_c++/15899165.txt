#include <cstdio>
#include <set>

using namespace std;
set<int>s;

int main()
{
    int n,l,x,y,d3,a;
    scanf("%d%d%d%d",&n,&l,&x,&y);
    bool f1=0,f2=0,f3=0;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a);
        if(a-x>=0&&s.count(a-x)){f1=1;}
        if(a-y>=0&&s.count(a-y)){f2=1;}
        if(a-(x+y)>=0&&s.count(a-x-y)){f3=1;d3=a-y;}
        if(a-y>=0&&s.count(a-y+x)){f3=1;d3=a-y;}
        if(a+x<=l&&s.count(a-y+x)){f3=1;d3=a+x;}
        s.insert(a);
    }
    if(f1&&f2){printf("0");return 0;};
    if(f1){printf("1\n%d\n",y);return 0;}
    if(f2){printf("1\n%d\n",x);return 0;}
    if(f3){printf("1\n%d\n",d3);return 0;}
    printf("2\n%d %d\n",x,y);
}
