#include<iostream>
#include<set>
using namespace std;
int main () {
    int n,l,x,y,z,z1=-1;
    bool x1=1,y1=1;
    set <int> a;
    cin>>n>>l>>x>>y;
    for (int i=0;i<n;i++) {
        cin>>z;
        if (x1) { if (z-x>=0 && a.count(z-x)) x1=0; }
        if (y1) { if (z-y>=0 && a.count(z-y)) y1=0; }
        if (z1==-1 && x1 && y1) {
            if (z-x-y>=0 && a.count(z-x-y)) z1=z-y;
            else if (z-y>=0 && a.count(z+x-y)) z1=z-y;
            else if (z+x<=l && a.count(z+x-y)) z1=z+x;
        }
        a.insert(z);
    }
    if (x1==0 && y1==0) cout<<"0"<<endl;
    else if (x1==0) cout<<"1"<<endl<<y<<endl;
    else if (y1==0) cout<<"1"<<endl<<x<<endl;
    else if (z1!=-1) cout<<"1"<<endl<<z1<<endl;
    else cout<<"2"<<endl<<x<<" "<<y<<endl;
    return 0;
}
