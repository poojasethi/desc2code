#include "cstdio"
#include "cstring"
#define maxn 2010
#define mod 1000000007
#define LL long long
char s[maxn],e[maxn];
LL dp[100005][2],k,cnt=0,len;
int main()
{
    //freopen("in.txt","r",stdin);
    scanf("%s%s%I64d",&s,&e,&k);
    len=strlen(s);
    for(int i=0;i<len;i++)
    {
        s[i+len]=s[i];
        if(!strncmp(s+i,e,len)) cnt++;
    }
    dp[0][strncmp(s,e,len)!=0]=1;
    for(int i=1;i<=k;i++)
    {
        dp[i][0]=(dp[i-1][0]*(cnt-1)+dp[i-1][1]*cnt)%mod;
        dp[i][1]=(dp[i-1][0]*(len-cnt)+dp[i-1][1]*(len-cnt-1))%mod;
    }
    printf("%I64d\n",dp[k][0]);

}
