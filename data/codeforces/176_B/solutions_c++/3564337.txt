#include <iostream>
#include <string>

using namespace std;

string start, end;
int K;

typedef long long ll;
const ll MOD = 1000000007;
ll t;

int main(){
	cin >> start >> end >> K;
	
	if (!K) {cout << (start==end) << endl; return 0;}
	
	int n = (int)start.length();
	
	t = 0;
	for (int i=1; i<K; i++){
		t = (t * (n-1));
		if (i % 2) t += n-1;
		else t -= n-1;
		t = (t+MOD)%MOD;
	}
	int fc;
	if (K % 2) fc = 1;
	else fc = -1;
	
	ll ans = 0;
	for (int i=0; i<n; i++){
		string s = start.substr(i, n-i) + start.substr(0,i);
		if (s == end){
			if (!i)	ans += t;
			else ans += t + fc;
			ans %= MOD;
		}
	}
	
	cout << ans << endl;
	
	return 0;
}
