#include<iostream>
#include<cstdio>
#include<string.h>
#include<string>
#include<math.h>
#include<algorithm>
#include<queue>
#include<map>
#include<time.h>
using namespace std;
#define mod 1000000007

string s1,s2;
long long f[100009][2];
int k;

int main(){
	cin >> s1 >> s2 >> k;
	f[0][s1==s2] = 1;
	int n = s1.size();
	s1.append(s1);
	int cnt = 0;
	for(int i=0;i<n;i++)
		if(s1.compare(i,n,s2)==0)
			cnt++;
	for(int i=1;i<=k;i++){
		f[i][1] = (f[i-1][1]*(cnt-1)+f[i-1][0]*cnt)%mod;
		f[i][0] = (f[i-1][0]*(n-cnt-1)+f[i-1][1]*(n-cnt))%mod;
	}
	cout << f[k][1] << endl;
	return 0;
}
