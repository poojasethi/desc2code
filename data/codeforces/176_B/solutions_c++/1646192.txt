#include<cstdio>
#include<cstring>
const int N = 1005;
const int MOD = 1000000007;
bool u[N];
char s1[N],s2[N];
typedef __int64 ll;
ll dp[100005][2];
int main()
{
	int n,k,i,j;
	scanf("%s%s%d",s1,s2,&k);
	n=strlen(s1);
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(s1[(i+j)%n]!=s2[j])break;
		}
		if(j==n)u[i]=true;
		//printf("i:%d u:%d \n",i,u[i]);
	}
	dp[0][0]=1;
	for(i=1;i<=k;i++)
	{
		dp[i][0]=dp[i-1][1]*(n-1)%MOD;
		dp[i][1]=(dp[i-1][0]+dp[i-1][1]*(n-2))%MOD;
	}
	int r=0;
	for(i=0;i<n;i++)
	{
		if(u[i])
		{
			if(!i)r=(r+dp[k][0])%MOD;
			else r=(r+dp[k][1])%MOD;
		}
	}
	printf("%d\n",r);
	return 0;
}
