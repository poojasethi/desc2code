#include <cstdio>
#include <cstring>

using namespace std;

const int p=1000000007;
char s[1001],t[1001];

int main(void)
{
  int m;
  scanf("%s %s %d",s,t,&m);
  int n=(int)strlen(s);
  long long s1=1,s2=0;
  for(int i=1;i<=m;++i)
  {
    long long t1=s2*(n-1)%p,t2=(s1+s2*(n-2))%p;
    s1=t1;
    s2=t2;
  }
  long long ans=0;
  if(strcmp(s,t)==0)
    ans+=s1;
  for(int i=1;i<n;++i)
    if((strncmp(s+i,t,n-i)==0)&&(strncmp(s,t+n-i,i)==0))
      ans=(ans+s2)%p;
  printf("%I64d\n",ans);
  return 0;
}
