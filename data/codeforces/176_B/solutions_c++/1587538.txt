#include<cstdio>
#include<cstring>
#define Mod 1000000007

long long X,Y,XX,YY,Ans;

int N,K;

char Start[2100],End[1100];

bool cmp(int x)
{	int i;
	for(i=x;i<N;i++)if(End[i]!=Start[i-x])return 0;
	for(i=0;i<x;i++)if(End[i]!=Start[N-x+i])return 0;
	return 1;
}

int main()
{	int i;
	//freopen("b.in","rb",stdin);
	//freopen("b.out","wb",stdout);
	scanf("%s%s%d",Start,End,&K);
	N=strlen(Start);
	X=1;
	Y=0;
	for(i=1;i<=K;i++)
	{	XX=((N-1)*Y)%Mod;
		YY=((N-2)*Y+X)%Mod;
		X=XX;
		Y=YY;
	}
	for(i=0;i<N;i++)
	if(cmp(i))
	{	if(!i)Ans=(Ans+X)%Mod;
		else Ans=(Ans+Y)%Mod;
	}
	printf("%I64d\n",Ans);
	return 0;
}
