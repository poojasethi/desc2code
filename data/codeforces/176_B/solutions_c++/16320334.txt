#include <bits/stdc++.h>
using namespace std;

const int base=1e9+7;
const int maxN=2e3+100;
const int maxM=1e5+100;

int n,k;
char s[maxN],t[maxN];
int f[maxN];

int main() {
	scanf("%s\n%s\n%i",s,t,&k);
	n=strlen(s);
	for(int i=0;i<n;i++) s[i+n]=s[i];
	f[0]=1;
	for(int i=1;i<=k;i++) {
		int sum=0;
		for(int j=0;j<n;j++) sum=(sum+f[j])%base;
		for(int j=0;j<n;j++) f[j]=(sum-f[j]+base)%base;
	}
	long long res=0;
	for(int i=0;i<n;i++)  {
		bool ok=true;
		for(int j=0;j<n;j++) if (s[i+j]!=t[j]) ok=false;
		if (ok) res=(res+f[i])%base;
	}
	cout<<res;
}
