#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1010;
const int mod = 1e9+7;

char ar[MAXN*2],er[MAXN];
int N,K,dn[MAXN],pre[MAXN],sum,presum;

int main(){
	
	scanf(" %s %s %d",ar+1,er+1,&K);
	N=strlen(ar+1);
	
	for(int i=1;i<=N;i++)
		ar[N+i]=ar[i];
	
	for(int i=1;i<=N;i++){
		int cnt=0;
		for(int j=1;j<=N;j++)
			if(ar[i+j-1]==er[j])
				cnt++;
		if(cnt==N){
			pre[i]=1;
			presum++;
		}
	}
	
	for(int i=1;i<=K;i++){
		for(int j=1;j<=N;j++){
			dn[j]=(presum-pre[j]+mod)%mod;
			sum=(sum+dn[j])%mod;
		}
		for(int j=1;j<=N;j++)
			pre[j]=dn[j];
		memset(dn,0,sizeof dn);
		presum=sum;
		sum=0;
	}
	
	cout<<pre[1]<<endl;
	
	return 0;
}
