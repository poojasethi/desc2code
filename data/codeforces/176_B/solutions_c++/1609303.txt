#include<cstdio>
#include<cstring>
const long long oo=1000000007;
long long f0,f1,f2,f3,p0,p1,Ans;
int flag,i,j,N,M,K;
char A[1005],B[1005];
int main()
{
	scanf("%s",A),N=strlen(A);
	scanf("%s",B),M=strlen(B);
	scanf("%d",&K);
	if(N!=M) printf("0\n");
	else
	{
		for(i=0;i<N;++i)
		{
			flag=1;
			for(j=0;j<N;++j)
				if(A[(j+i)%N]!=B[j])
				{
					flag=0;
					break;
				}
			if(flag) if(!i) p0=1; else ++p1;
		}
		f0=1;
		for(i=1;i<=K;++i)
		{
			f2=f1*(N-1)%oo;
			f3=(f1*(N-2)+f0)%oo;
			f0=f2,f1=f3;
		}
		Ans=(f0*p0+f1*p1)%oo;
		printf("%I64d\n",Ans);
	}
	return 0;
}
