#include <iostream>
#include <string>
#include <cstring>

#define rep(i,n) for(int i=0;i<n;i++)
#define ll long long
#define mod 1000000007

using namespace std;

ll dp[10],tmp[10],cnt;
int n,k;
string str,end;

int main(){

  cin>>str>>end>>k;
  n = str.size();
  if(str==end)dp[0]=1;
  else dp[1] = 1;

  str.append(str);
  rep(i,n)if(str.substr(i,n)==end)cnt++;

  rep(i,k){
    tmp[0] = (dp[0]*(cnt-1)+dp[1]*cnt)%mod;
    tmp[1] = (dp[0]*((ll)n-cnt)+dp[1]*((ll)n-1-cnt))%mod;
    dp[0] = tmp[0];
    dp[1] = tmp[1];
  }
  cout<<dp[0]<<endl;
}
