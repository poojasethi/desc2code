#include<cstdio>
#include<cstring>
typedef __int64 lld;
const lld mod = 1000000007;
lld dp[100010][2];
char a[2010],b[1010];
int main()
{
    int i,m;
    gets(a);gets(b);
    int n=strlen(a);
    scanf("%d",&m);
    dp[0][strcmp(a,b)!=0]=1;
    for(i=0;i<n;i++) a[i+n]=a[i];
    lld cnt=0;
    for(i=0;i<n;i++) if(strncmp(a+i,b,n)==0) cnt++;
    for(i=0;i<m;i++)
    {
        dp[i+1][0]=(dp[i][0]*(cnt-1)+dp[i][1]*cnt)%mod;
        dp[i+1][1]=(dp[i][0]*(n-cnt)+dp[i][1]*(n-cnt-1))%mod;
    }
    printf("%I64d\n",dp[m][0]);
    return 0;
}
