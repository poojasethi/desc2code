#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>
#include<cmath>
#include<vector>
#include<set>

using namespace std;
int n,m,k,l,ch = 0;
long long  g[100005],b[100005];
int main()
{
	string s,s1;
	cin>>s>>s1>>n;
	for(int i = 0;i<s.length();++i)
	{
		for(int j = 0;j<s.length();++j)
		{
			if(s[(j+i)%s.length()]!=s1[j])
			{
				ch = 1;break;
			}
		}
		if(!ch)++m;
		else ++k;
		ch = 0;
	}
	if(s==s1)g[0] = 1;
	else b[0] = 1;
	for(int i = 1;i<=n;++i)
	{
		g[i] = (g[i-1]*(m-1)+b[i-1]*m)%1000000007;
		b[i] = (b[i-1]*(k-1)+g[i-1]*k)%1000000007;
	}
	cout<<g[n];
	return 0;
}
