#include<stdio.h>
#include<string.h>
#define mod 1000000007
#define LL long long

char a[2005],b[1002];
LL dp[100003][2];
int main(void)
{
    int n,x=0,length;

    scanf("%s%s",a,b);
    scanf("%d",&n);

    length=strlen(a);
    dp[0][strcmp(a,b)!=0]=1;

    for(int i=0;i<length;i++)a[i+length]=a[i];
    for(int i=0;i<length;i++)
    if(strncmp(a+i,b,length)==0)x++;

    for(int i=0;i<n;i++)
    {
        dp[i+1][0]=(x*dp[i][1]+(x-1)*dp[i][0])%mod;
        dp[i+1][1]=((length-x)*dp[i][0]+(length-x-1)*dp[i][1])%mod;
    }

    printf("%lld\n",dp[n][0]);

    return 0;
}
