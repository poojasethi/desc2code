#include<stdio.h>
#include<string.h>
#define MOD 1000000007
char a[10000];
char b[10000];
int main(){
	int n;
	scanf("%s %s",a,b);
	n = strlen(a);
	int m;
	scanf("%d",&m);
	long long dif = 0;
	long long sam = 1;
	for( int i = 0; i < m; i++ ){
		long long tdif, tsam;
		tdif = dif*(n-2);
		tdif += sam;
		tdif %= MOD;
		tsam = dif*(n-1);
		tsam %= MOD;
		dif = tdif;
		sam = tsam;
	}
	long long ans = 0;
	for(int k = 0 ; k < n ; k++ ){
		int chk = 1;
		for(int i = 0; i < n;i++ ){
			if( a[(k+i)%n] != b[i] ){
				chk = 0;
				break;
			}
		}
		if( chk ){
			if( k == 0 ) ans += sam;
			else ans += dif;
			ans %= MOD ;
		}
	}
	printf("%I64d\n",ans);
}
