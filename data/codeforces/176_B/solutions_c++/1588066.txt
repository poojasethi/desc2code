#include <algorithm>
#include <cstdio>
#include <iostream>
#include <map>
#include <vector>

using namespace std;

typedef long long int64 ;

int64 modul = 1000000007LL ;

int main() {
  string sa, sb ;
  cin >> sa >> sb ;
  int64 base = (sa == sb) ? 1 : 0, next = 0 ;
  for (int i = 1 ; i < sa.size() ; i ++) {
    if (sa.substr(i) + sa.substr(0, i) == sb) next ++ ;
  }
  int64 K, n = sa.size() ;
  cin >> K ;
  int64 a = 1LL, b = 0LL, tmp ;
  for (int k = 0 ; k < K ; k ++) {
    tmp = ((n - 2) * b + a) % modul ;
    a = ((n - 1) * b) % modul ; b = tmp ;
  }
  cout << (base * a + next * b) % modul << endl ;

  return 0 ;
}
