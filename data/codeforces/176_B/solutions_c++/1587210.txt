#include <iostream>
#include <algorithm>
using namespace std;

const long long mod = 1000000007;

string a , b;
int k;

void read() {
	cin >> a >> b >> k;
}

void solve() {
	long long c , d , nc , nd , ans = 0;
	int i;
	
	c = 1;
	d = 0;
	
	for (i = 1; i <= k; i++) {
		nc = (((long long)a.size() - 1) * d) % mod;
		nd = (((long long)a.size() - 2) * d + c) % mod;
		
		c = nc;
		d = nd;
	}
	
	if ( a == b ) {
		ans += c;
	}
	a += a;
	for (i = 1; i < (int)b.size(); i++) {
		if ( a.substr ( i , (int)b.size() ) == b ) {
			ans = (ans + d) % mod;
		}
	}
	
	cout << ans << '\n';
}

int main() {
	read();
	solve();
	
	return 0;
}
