#include <iostream>
#include <string>
#include <string.h>
using namespace std;
int main()
{
    string s,e;
    cin>>s>>e;
    int n,i,k,l,m=s.length();
    long long j,f0,f1;
    cin>>n;
    f0=1;
    f1=0;
    for (i=1;i<=n;i++)
    {
        j=f0;
        f0=((m-1)*f1)%1000000007;
        f1=(((m-2)*f1)%1000000007+j)%1000000007;
    }
    j=0;
    for (i=0;i<m;i++)
    {
        string t1=s.substr(i)+s.substr(0,i);
        if (t1==e)
        {
            if (i==0) {j+=f0;} else {j+=f1;}
            j=j%1000000007;
        }
    }
    cout<<j<<endl;
}
