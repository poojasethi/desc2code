#include<cstdio>
#include<cstring>
#define mod 1000000007
#define LL long long
const int N= 2222;
char a[N],b[N];
LL dp[111111][2];
int main(void){
    int k,x=0,len;
    scanf("%s%s",a,b);
    scanf("%d",&k);
    len=strlen(a);
    dp[0][strcmp(a,b)!=0]=1;//[1]mean bad,[0] mean good 
    for(int i=0;i<len;i++)a[i+len]=a[i];
    for(int i=0;i<len;i++)if(strncmp(a+i,b,len)==0)x++;
    for(int i=0;i<k;i++){
        dp[i+1][0]=(x*dp[i][1]+(x-1)*dp[i][0])%mod;// ->cannot rotate to itself
        dp[i+1][1]=((len-x)*dp[i][0]+(len-x-1)*dp[i][1])%mod;
    }
    printf("%I64d\n",dp[k][0]);
    return 0;
}
