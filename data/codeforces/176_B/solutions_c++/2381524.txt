#include <iostream>
#include <string.h>
using namespace std;
#define N 100010
#define MOD 1000000007
int main(){
        string s,e;
        cin>>s>>e;
        long long k,dpg[N],dpb[N],g=0,b=0;
        cin>>k;
        for(int i=0; i<s.length(); i++){
                if((s.substr(i)+s.substr(0,i))==e)g++;
                else b++;
        }
        dpg[0]=(s==e);
        dpb[0]=(s!=e);
        for(int i=1; i<=k; i++){
                dpg[i]=((dpg[i-1]*(g-1))%MOD+(dpb[i-1]*g)%MOD)%MOD;
                dpb[i]=((dpg[i-1]*b)%MOD+(dpb[i-1]*(b-1))%MOD)%MOD;
        }
        cout<<dpg[k]<<endl;
        return 0;
}
