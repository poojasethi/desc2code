#include<cstdio>
#include<cstring>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

const ll M=1000000007;

int main(){
	char s[1001],t[1001]; scanf("%s%s",s,t);
	int n=strlen(s);

	int n_cut=0;
	rep(i,n) if(strncmp(s,t+i,n-i)==0 && strncmp(s+n-i,t,i)==0) n_cut++;

	int k; scanf("%d",&k);
	ll good=(strcmp(s,t)==0?1:0),bad=1-good;
	rep(i,k){
		ll good2=(n_cut-1)*good+n_cut*bad;
		ll bad2=(n-n_cut)*good+(n-n_cut-1)*bad;
		good=good2%M;
		bad=bad2%M;
	}
	printf("%d\n",good);

	return 0;
}
