#include <bits/stdc++.h>
using namespace std;
int dp[1000],n;
const int mod = 1e9+7;
int main(){
	string s,e;cin>>s>>e;
	n=s.size();
	s+=s;int tot=0;
	for(int i=0;i<n;++i){
		bool ok=true;
		for(int j=0;j<n;++j)
			if(s[i+j]!=e[j]){ok=false;break;}
		if(ok){dp[i]=1;++tot;}
	}
	int x,i,j,cand=0;cin>>x;
	while(x--){
		cand=0;
		for(i=0;i<n;++i){
			dp[i] = (tot-dp[i]+mod)%mod,cand+=dp[i];
			if(cand>=mod)cand-=mod;
		}
		tot=cand;
	}
	cout<<dp[0]<<endl;
}
