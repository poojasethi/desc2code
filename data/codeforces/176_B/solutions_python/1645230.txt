
MOD = 1000000007

if __name__ == '__main__':
    a, b, k = raw_input(), raw_input(), int(raw_input())
    
    str_len = len(a)
    cut_pos = [i for i in xrange(str_len) if a[i:] + a[:i] == b]
    cut_pos_len = len(cut_pos)
    sum = 1
    x, y = 1, 0
    for _ in xrange(k):
        x = sum - x
        y = sum - y
        sum *= str_len-1
        sum %= MOD
    print (x+y*(cut_pos_len-1))%MOD if 0 in cut_pos else y*cut_pos_len%MOD
