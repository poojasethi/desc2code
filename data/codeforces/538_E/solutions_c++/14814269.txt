#include <bits/stdc++.h>
using namespace std;
const int N = 2e5+2;
vector<int> adj[N];
int R=0,dp[N][2];
void dfs(int root){
	if(adj[root].empty()){
		++R;dp[root][0]=dp[root][1]=1;
		return;
	}
	dp[root][0]= N+5;int i,v;
	for(i=0;i<adj[root].size();++i){
		v= adj[root][i];
		dfs(v);
		dp[root][0]= min(dp[root][0],dp[v][1]);
		dp[root][1]+=dp[v][0];
	}
}
int main(){
	int n,i,a,b;
	cin>>n;
	for(i=0;i<n-1;++i){
		cin>>a>>b;
		adj[a].push_back(b);
	}
	dfs(1);
	printf("%d %d\n",R-dp[1][0]+1,dp[1][1]);
}
