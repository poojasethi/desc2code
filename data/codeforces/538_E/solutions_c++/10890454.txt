#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=200050;
int n,m=0,ans0,ans1;
int es=0,he[maxn],ne[maxn],to[maxn];
int f0[maxn],f1[maxn];
void ins(int x,int y)
{
	ne[++es]=he[x],he[x]=es,to[es]=y;
}
void dfs(int x,int d)
{
	int i,y;
	bool z=0;
	if(!d) f0[x]=n,f1[x]=-1;
	else f0[x]=-1,f1[x]=n;
	for(i=he[x];i;i=ne[i])
	{
		z=1;
		dfs(y=to[i],!d);
		if(!d)
			f0[x]=min(f0[x],f0[y]),f1[x]+=f1[y]+1;
		else
			f0[x]+=f0[y]+1,f1[x]=min(f1[x],f1[y]);
	}
	m+=!z;
	if(!z) f0[x]=f1[x]=0;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
	freopen("data.out","w",stdout);
#endif
	int i,j,x,y;
	scanf("%d",&n);
	for(i=1;i<n;i++)
		scanf("%d%d",&x,&y),ins(x,y);
	dfs(1,0);
	//for(i=1;i<=n;i++) printf("%d %d\n",f0[i],f1[i]);
	ans0=m-f0[1],ans1=f1[1]+1;
	printf("%d %d\n",ans0,ans1);
	return 0;
}
