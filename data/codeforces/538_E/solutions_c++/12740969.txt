#include <iostream>
#include <vector>
#include <cstdio>

using namespace std;

int c[300100];
vector<int> e[300100];

int go1(int);

int t1(int u)
{
    if(c[u]==0) return 1;
    int m=0;
    for(int i=0;i<e[u].size();i++)
        m+=go1(e[u][i]);
    return m;
}

int go1(int u)
{
    if(c[u]==0) return 1;
    int m = 1e9;
    for(int i=0;i<e[u].size();i++)
        m = min(m,t1(e[u][i]));
    return m;
}

int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n-1;i++)
    {
        int u,v;
        scanf("%d%d",&u,&v);
        u--;v--;
        e[u].push_back(v);
        c[u]++;
    }
    int m=0;
    for(int i=0;i<n;i++)
        if(c[i]==0) m++;

    cout << m-go1(0)+1<<' '<<t1(0);
    return 0;
}
