#include<iostream>
#include<cstdio>
#include<cstring>
#define maxn 400009
using namespace std;
int next[maxn],head[maxn],point[maxn],now=0,leaf;
bool visit[maxn];
void add(int x,int y)
{
	next[++now]=head[x];
	head[x]=now;
	point[now]=y;
}
int dfs(int k,bool f)
{
	visit[k]=1;
	int minx=0x3f3f3f3f,ret=0,flag=1;
	for(int i=head[k];i;i=next[i])if(!visit[point[i]])
	{
        flag=0;
		int u=point[i];
		if(f)
		{
			minx=min(minx,dfs(u,f^1));
		}
		else
		{
			ret+=dfs(u,f^1);
		}
	}
	if(flag)
	{
        leaf++;
        return 1;
	}
	if(f)return minx;else return ret;
}
int main()
{
	int n,x,y;
	scanf("%d",&n);
	for(int i=1;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		add(x,y);
		add(y,x);
	}
	int u=dfs(1,1);
	leaf=0;
	memset(visit,0,sizeof(visit));
	int v = dfs(1,0);
	printf("%d %d\n",leaf-u+1,v);
	return 0;
}
