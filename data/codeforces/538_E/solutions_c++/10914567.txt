#include <cstdio>
#include <algorithm>
using namespace std;

int m, n, a[200001], e[400000], b[400000];
int dp[200001];

void dfs(int v, bool first) {
    if (a[v] == 0) {
        m++;
        dp[v] = 1;
        return;
    }
    int p = (1<<31)-1, q = 0;
    for (int i = a[v]; i; i = b[i]) {
        dfs(e[i], !first);
        p = min(p, dp[e[i]]);
        q += dp[e[i]];
    }
    dp[v] = first ? p : q;
}

int main() {
    scanf("%d", &n);
    for (int i = 1; i < n; i++) {
        scanf("%d%d", e + i, e + i + n);
        b[i + n] = a[e[i]];
        a[e[i]] = i + n;
    }
    dfs(1, true);
    int p = m - dp[1] + 1;
    dfs(1, false);
    printf("%d %d\n", p, dp[1]);
}
