#include <cstdio>
#include <vector>
using namespace std;
int n,a[200100],b[200100],c[200100];
vector<int> g[200100];
void dfs(int i, int p, int d){
	bool was=false;
	int ansa=d?n:1,ansb=d?0:n;
	for(int j=0;j<g[i].size();j++){
		int k=g[i][j];
		if(k==p) continue;
		dfs(k,i,d^1);
		c[i]+=c[k];
		if(d){
			ansa=min(ansa,c[k]-a[k]);
			ansb+=b[k];
		}else{
			ansa+=a[k]-1;
			ansb=min(ansb,b[k]);
		}
		was=true;
	}
	if(!was){
		a[i]=b[i]=c[i]=1;
	}else if(d){
		a[i]=c[i]-ansa;
		b[i]=ansb;
	}else{
		a[i]=ansa;
		b[i]=ansb;
	}
}
int main(){
	scanf("%d",&n);
	for(int i=1;i<n;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		g[x].push_back(y);
		g[y].push_back(x);
	}
	dfs(1,0,1);
	printf("%d %d\n",a[1],b[1]);
	return 0;
}
