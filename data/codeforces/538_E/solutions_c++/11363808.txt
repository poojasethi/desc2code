#include<cstdio>
#include<vector>
using namespace std;
#define MIN(a,b) a<b?a:b
vector<int>edge[200001];
const int INF=0x3f3f3f3f;
int dfs(int x,int k)
{
    int l=edge[x].size();
    if(l==0)return 1;
    int t1=INF,t2=0;
    for(int i=0;i<l;i++)
    {
        int v=edge[x][i];
        int num=dfs(v,k^1);
        if(k==0)t1=MIN(t1,num);
        else t2+=num;
    }
    if(k==0)return t1;
    return t2;
}
int main()
{
    int n,u,v;
    scanf("%d",&n);
    for(int i=0;i<n-1;i++){
        scanf("%d%d",&u,&v);
        edge[u].push_back(v);
    }
    int m=0;
    for(int i=1;i<=n;i++){
        if(edge[i].size()==0)m++;
    }
    printf("%d %d\n",m+1-dfs(1,0),dfs(1,1));
    return 0;
}
