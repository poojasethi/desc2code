#include <algorithm>
#include <cstdio>
int f[200010][2],to[400010],ne[400010],fi[200010],
	num[200010],i,n,x,y,tot;
void add(int x,int y)
{
	to[++tot]=y;ne[tot]=fi[x];fi[x]=tot;
}
void dfs(int fa,int x,int y)
{
	int j=0,k=0,l=n;
	for (int i=fi[x];i;i=ne[i])
	if (to[i]!=fa)
	{
		dfs(x,to[i],y+1);
		num[x]+=num[to[i]];
		++j;
		if (y&1)
		{
			l=std::min(l,num[to[i]]-f[to[i]][0]);
			k+=f[to[i]][1];
		}
		else
		{
			l=std::min(l,f[to[i]][1]);
			k+=f[to[i]][0];
		}
	}
	if (!j) f[x][0]=f[x][1]=num[x]=1;
		else if (y&1) f[x][0]=num[x]-l,f[x][1]=k;
			else f[x][0]=k-j+1,f[x][1]=l;
}
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d\n",&n);
	for (i=1;i<n;++i)
	{
		scanf("%d%d\n",&x,&y);
		add(x,y);add(y,x);
	}
	dfs(0,1,1);
	printf("%d %d\n",f[1][0],f[1][1]);
}
