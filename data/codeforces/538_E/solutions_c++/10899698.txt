#include <bits/stdc++.h>

using namespace std;

vector<int> adj[200010];


int maior(int);
int menor(int);


int maior(int at){
    if(adj[at].size() == 0)
        return 1;
    int resp = 1000000000;
    for(int i = 0;i<adj[at].size();i++)
        resp = min(resp,menor(adj[at][i]));
    return resp;
}

int menor(int at){
    if(adj[at].size() == 0)
        return 1;
    int resp = 0;
    for(int i = 0;i<adj[at].size();i++)
        resp+= maior(adj[at][i]);
    return resp;
}


int main(){

    int n;
    cin>>n;
    for(int i = 0;i<n;i++)
        adj[i].clear();
    for(int i = 0;i<n-1;i++)
    {
        int a, b;
        cin>>a>>b;
        a--;b--;
        adj[a].push_back(b);
    }
    int qtd = 0;
    for(int i = 0;i<n;i++)
        qtd+=(adj[i].size()==0);

    cout<<qtd - maior(0) + 1<<" "<<menor(0)<<endl;
    return 0;
}
