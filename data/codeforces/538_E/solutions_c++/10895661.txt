#include <stdio.h>
#include <vector>

using namespace std;

const int MAX=1000000000;

vector<int> edge[200005];

int go(int wh,int typ)
{
	if(edge[wh].size()==0){
		return 1;
	}
	if(typ==0){
		int mmin=MAX;
		for(int i=0; i<edge[wh].size(); i++){
			int tmp=go(edge[wh][i],1);
			if(mmin>tmp)
				mmin=tmp;
		}
		return mmin;
	}else{
		int sum=0;
		for(int i=0; i<edge[wh].size(); i++){
			int tmp=go(edge[wh][i],0);
				sum+=tmp;
		}
		return sum;
	}
}

int main()
{
	int n;
	int u,v;
	scanf("%d",&n);
	int cnt=n;
	for(int i=1; i<n; i++){
		scanf("%d %d",&u,&v);
		if(edge[u].size()==0)
			cnt--;
		edge[u].push_back(v);
	}
	printf("%d %d\n",cnt-go(1,0)+1,go(1,1));
	return 0;
}
