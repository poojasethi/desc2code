#include<bits/stdc++.h>

using namespace std;

#define pb push_back
#define inf 1e8+7

int DP[200005];

vector<int> V[200005];

int f(int x,int k){
	
	int &r=DP[x],i;

	if(V[x].empty())
		return r=1;

	if(k){
		r=inf;
		for(i=0 ; i<V[x].size() ; i++){
			f(V[x][i],!k);
			r=min(r,DP[V[x][i]]);
		}
		if(r==inf)
			r=0;
	}

	else {
		r=0;
		for(i=0 ; i<V[x].size() ; i++){
			f(V[x][i],!k);
			r+=DP[V[x][i]];
		}
	}
}

int main(){

	int n,i,leaf=0,x,y;

	cin>>n;

	for(i=1 ; i<n ; i++){
		scanf("%d %d",&x,&y);
		V[x].pb(y);
	}

	for(i=1 ; i<=n ; i++)
		if(V[i].empty())
			leaf++;

	f(1,1);

	cout<<leaf-DP[1]+1<<' ';

	f(1,0);

	cout<<DP[1]<<endl;

	return 0;
}
