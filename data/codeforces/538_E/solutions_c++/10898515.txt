#include <bits/stdc++.h>
#define N 212345
using namespace std;

vector<int> edges[N];
int dfs(int i,int p)
{
    if(edges[i].size()==0) return 1;
    if(p==1)
    {
        int ans=dfs(edges[i][0],2);
        for(int j=1;j<edges[i].size();++j) ans=min(ans,dfs(edges[i][j],2));
        return ans;
    }
    int ans=0;
    for(int j=0;j<edges[i].size();++j) ans+=dfs(edges[i][j],1);
    return ans;
}
int leaves(int i)
{
    if(edges[i].size()==0) return 1;
    int ans=0;
    for(int j=0;j<edges[i].size();++j) ans+=leaves(edges[i][j]);
    return ans;
}
int main()
{
    int n,u,v;
    scanf("%d",&n);
    for(int i=1;i<n;++i)
    {
        scanf("%d %d",&u,&v);
        edges[u].push_back(v);
    }
    int l=leaves(1);
    int ma=l-dfs(1,1)+1;
    int mi=dfs(1,2);
    printf("%d %d\n",ma,mi);
    return 0;
}
