//Hei Ho Heidi Ho
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
vector <int> edge[201010];
int node,leafcount=0;

int DFS (int node,bool bad)
{
    if(edge[node].empty())  return 1; //Leaf Node
    int x = 999999999,y=0;
    for(int i=0;i<edge[node].size();i++)
    {
        if(bad)
            y += DFS(edge[node][i],!bad);
        else //Good
            x = min (x,DFS(edge[node][i],!bad));
    }
    if(bad) return y; else return x;
}

int main()
{
    scanf("%d",&node);
    for(int i=1;i<node;i++)
    {
        int to,from;
        scanf("%d%d",&from,&to);
        edge[from].push_back(to);
    }
    for(int i=1;i<=node;i++)
        if(edge[i].empty()) leafcount++;
    printf("%d %d",leafcount+1-DFS(1,false),DFS(1,true));
    return 0;
}
