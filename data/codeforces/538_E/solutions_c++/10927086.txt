#include <bits/stdc++.h>
using namespace std;

const int maxn=2e5+10;
const int INF=1e9;

int n;
int head[maxn];
int pre[maxn];
int to[maxn];
int f[maxn];
int g[maxn];
int tote;

void addedge(int u,int v){
    to[tote]=v;
    pre[tote]=head[u];
    head[u]=tote++;
}


static int leafcnt=0;
void dfs(int u){
    f[u]=INF;
    for(int i=head[u];~i;i=pre[i]){
        int v=to[i];
        dfs(v);
        if(g[v]<f[u])f[u]=g[v];
        g[u]+=f[v];
    }
    if(!g[u]){
        leafcnt++;
        g[u]=f[u]=1;
    }
}


int main(){
    memset(head,-1,sizeof(head));
    tote=0;
    
    cin>>n;
    for(int i=1;i<n;i++){
        int u,v;
        scanf("%d%d",&u,&v);
        addedge(u,v);
    }
    dfs(1);
    
    cout<<leafcnt-f[1]+1<<" "<<g[1]<<endl;
    return 0;
}
