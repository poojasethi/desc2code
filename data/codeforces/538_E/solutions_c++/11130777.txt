#include <iostream>
#include <cstdio>
#include <vector>
#define N 200001
#define INF 0x7fffffff
using namespace std;
int n,m;
vector <int> v[N];
int dfs(int u,bool f)
{
    if(!v[u].size())return 1;
    int i,t1=INF,t2=0;
    for(i=0;i<v[u].size();i++)
    {
        int to=v[u][i];
        if(f)t2+=dfs(to,0);
        else t1=min(t1,dfs(to,1));
    }
    if(f)return t2;
    else return t1;
}
int main()
{
    int i,x,y;
    scanf("%d",&n);
    for(i=1;i<n;i++)
    {
        scanf("%d%d",&x,&y);
        v[x].push_back(y);
    }
    for(i=1;i<=n;i++)
        if(!v[i].size())
            m++;
    printf("%d %d\n",m+1-dfs(1,0),dfs(1,1));
    return 0;
}
