#include <bits/stdc++.h>
using namespace std;
int n;
vector<int>son[200005];
int f[200005][2];
int leaves;
void DFS(int loc)
{
	if(!son[loc].size()){leaves++,f[loc][0]=f[loc][1]=1;return;}
	f[loc][0]=0x3f3f3f3f;
	for(int i=0;i<son[loc].size();i++)
	{
		DFS(son[loc][i]);
		f[loc][0]=min(f[loc][0],f[son[loc][i]][1]);
		f[loc][1]+=f[son[loc][i]][0];
	}
}
int main()
{
	scanf("%d",&n);
	for(int u,v,i=2;i<=n;i++)
		scanf("%d%d",&u,&v),son[u].push_back(v);
	DFS(1);
	printf("%d %d\n",leaves-f[1][0]+1,f[1][1]);
	return 0;
}
