#include <iostream>
#include <cstdlib>
#include <vector>
#include <map>
using namespace std;
#define mp make_pair
const int MAXN=2e5+10,INF=1e9;
vector <int> g[MAXN];
int dp[MAXN][2],m;

void dfs(int v,int par=0)
{
  if(g[v].empty())
    {
      m++;
      dp[v][1]=dp[v][0]=1;
      return;
    }
  dp[v][0]=INF;
  for(int i=0;i<g[v].size();i++)
    {
      int u=g[v][i];
      dfs(u,v);
      dp[v][0]=min(dp[v][0],dp[u][1]);
      dp[v][1]+=dp[u][0];
    }
}
  

int main()
{
  int n;
  cin>>n;
  for(int i=1;i<n;i++)
    {
      int x,y;
      cin>>x>>y;
      g[x].push_back(y);
    }
  dfs(1);
  cout<<m-dp[1][0]+1<<" "<<dp[1][1]<<endl;
  return 0;
}
