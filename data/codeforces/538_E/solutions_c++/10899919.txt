#include <iostream>
#include <vector>
using namespace std;

#define MAXN 200005
#define MAXVAL 10000000

vector<int> adj[MAXN];

int dfs(int cur, bool player){
	if (adj[cur].size() == 0) return 1;
	int player2 = MAXVAL;
	int player1 = 0;
	for (int i = 0; i<adj[cur].size(); i++) {
		if (player) player1 += dfs(adj[cur][i], !player);
		else player2 = min(player2, dfs(adj[cur][i], !player));
	}
	return player ? player1: player2;
}

int main() {
	int N, M;
	M=0;
	cin>>N;
	for (int i=0; i<N-1; i++) {
		int a,b;
		cin>>a>>b;
		adj[a].push_back(b);
	}
	for (int i=1; i<=N; i++) {
		if (adj[i].size() == 0) M++;
	}
	int player1 = M+1 - dfs(1, false);
	int player2 = dfs(1,true);
	cout<<player1<<" "<<player2<<endl;
	return 0;
}
