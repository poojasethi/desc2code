#include<iostream>
#include<string.h>
#include<stdio.h>
#include<vector>
using namespace std;

vector<int> G[200005];
int dp[200005][2];

void dfs(int p){
	if(G[p].size()==0){
		dp[p][0]=1;
		dp[p][1]=1;
		return;
	}
	dp[p][0]=0;dp[p][1]=1000000000;
	for(int i=0;i<G[p].size();i++){
		int to=G[p][i];
		dfs(to);
		dp[p][0]+=dp[to][1];
		dp[p][1]=min(dp[p][1],dp[to][0]);
	}
}

int main(){
	int n;
	int i,j;
	int x,y;
	scanf("%d",&n);
	for(i=1;i<n;i++){
		scanf("%d%d",&x,&y);
		G[x].push_back(y);
	}
	int c=0;
	for(i=1;i<=n;i++){
		if(G[i].size()==0){
			c++;
		}
	}
	dfs(1);
	printf("%d %d\n",c+1-dp[1][1],dp[1][0]);
	return 0;
}
