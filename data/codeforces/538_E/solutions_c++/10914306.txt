#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
#define N 300100
#define INF 0x3f3f3f3f
#define ll long long
using namespace std;
int n,m;
vector<int> e[N];
int dfs(int u,bool flag){
	if(e[u].size()==0) return 1;
	int a=0,b=INF;
	for(int i=0;i<e[u].size();i++){
		if(flag) a+=dfs(e[u][i],flag^1);
		else b=min(b,dfs(e[u][i],flag^1));
	}
	if(flag) return a;
	else return b;
}
int main(){
	int a,b;
	scanf("%d",&n);
	for(int i=1;i<n;i++){
		scanf("%d%d",&a,&b);
		e[a].push_back(b);
	}
	for(int i=1;i<=n;i++) if(e[i].size()==0) m++;
	printf("%d %d",m+1-dfs(1,0),dfs(1,1));
	return 0;
}
