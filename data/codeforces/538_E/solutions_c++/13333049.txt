const long long A=100000000000000LL,N=228228;

vector<long long> List,a[N];
long long i,j,n,m,x,y;

struct dp_xa{
    long long Size,Min;
} dp[N];

void go(int v){
    if(!a[v].size()){
        dp[v].Size=dp[v].Min=1;
        return;
    }
    dp[v].Min=A;
    for(int i=0;i<a[v].size();i++){
        go(a[v][i]);
        dp[v].Min=min(dp[v].Min,dp[a[v][i]].Size);
        dp[v].Size+=dp[a[v][i]].Min;
    }
}
