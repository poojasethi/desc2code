#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N=(int)2e5+5;

int n;

struct node{
	node *son,*next;
	int size;
	int f[2];
	void dfs(){
		if(!son){
			size=1;
			f[0]=f[1]=1;
			return;
		}
		for(node *v=son;v;v=v->next) v->dfs(),size+=v->size;
		for(node *v=son;v;v=v->next) f[0]=max(f[0],size-(v->size-v->f[1]));
		int sum=0;
		for(node *v=son;v;v=v->next) sum+=v->size-v->f[0]+1;
		f[1]=size-sum+1;
	}
} V[N];

int main(){
	scanf("%d",&n);
	for(int i=1;i<n;i++){
		int u,v; scanf("%d%d",&u,&v);
		V[v].next=V[u].son; V[u].son=V+v;
	}
	V[1].dfs();
	printf("%d %d\n",V[1].f[0],V[1].size+1-V[1].f[1]);
	scanf("\n");
}
