#include <bits/stdc++.h>
using namespace std;
#define fo(i,a,b) for (int i = (a); i < (b); i++)
#define N 234567
#define pb push_back

int n, m, a, b, ans[2][N];
vector<int> al[N];
void slv(int at, int par) {
    if (al[at].size() == 1 && at != 1) {
        m++;
        ans[0][at] = ans[1][at] = 1;
        return;
    }
    ans[0][at] = N;
    for (int i = 0; i < al[at].size(); i++) {
        int to = al[at][i];
        if (to == par) continue;
        slv(to, at);
        ans[0][at] = min(ans[0][at], ans[1][to]);
        ans[1][at] += ans[0][to];
    }
}
int main() {
    scanf("%d", &n);
    if (n==1) {
        puts("1 1"); return 0;
    }
    fo(i,0,n-1) {
        scanf("%d %d", &a, &b);
        al[a].pb(b); al[b].pb(a);
    }
    slv(1, 0);
    printf("%d %d\n", m - ans[0][1] + 1, ans[1][1]);

    return 0;
}
