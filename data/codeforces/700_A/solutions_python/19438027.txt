n, L, v1, v2, K = map(int, raw_input().split())

C = ((n-1) / K) + 1
bb = 0.0
ee = float(1e15)
LL = 0.0

# print 'C:', C

for i in range(100):
    T = (bb + ee) * 0.5
    LL = (T - T*(v2-v1)/(v1+v2)) * v2 * (C-1) + T * v2
    if LL > L:
        ee = T
    else:
        bb = T
    # print "i:%d, T:%.11f, bb:%.11f, ee:%.11f, LL:%.11f" % (i, T, bb, ee, LL)

print '%f' % (T + (L - v2*T) / v1)
