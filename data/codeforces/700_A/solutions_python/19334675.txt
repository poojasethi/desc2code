def main():
    n, l, v1, v2, k = map(int, raw_input().split())
    lo = 0
    hi = 1.0 * l / v1
    m = (n + k - 1) / k
    dv = v2 - v1
    for i in xrange(100):
        mid = (lo + hi) / 2
        s = (l - mid * v1) / dv
        p = s * v2 * m - l
        t = s * m + p / v2
        if t < mid:
            hi = mid
        else:
            lo = mid
    print "%.12f" %  hi
main()
