count, distance, student_speed, bus_speed, bus_cap = (int(x) for x in raw_input().split())
distance, student_speed, bus_speed = (float(x) for x in (distance, student_speed, bus_speed))
trip_count = (count + bus_cap - 1) / bus_cap

optimal_trip_distance = distance / (
    1 + student_speed *
        (trip_count / (bus_speed - student_speed) +
        ((trip_count - 1)) / (bus_speed + student_speed)))

print "%.9f" % (optimal_trip_distance * (trip_count / (bus_speed - student_speed) + (trip_count - 1) / (bus_speed + student_speed)))
