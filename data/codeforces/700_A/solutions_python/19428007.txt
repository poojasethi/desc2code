#!/usr/bin/env python


def calculate_time_to_excursion(pupil_count, length, pupil_speed, bus_speed, bus_capacity):
    if bus_speed < pupil_speed:
        return 1.0 * length / pupil_speed

    pass_count = (pupil_count - 1) / bus_capacity
    optimal_time_on_bus = (1.0 * length / bus_speed) / (1.0 + 2.0 * pass_count * pupil_speed / (pupil_speed + bus_speed))
    overall_time = (length - optimal_time_on_bus * (bus_speed - pupil_speed)) / pupil_speed
    return overall_time


def main():
    pupil_count, length, pupil_speed, bus_speed, bus_capacity = map(int, raw_input().split())
    time_to_excursion = calculate_time_to_excursion(pupil_count, length, pupil_speed, bus_speed, bus_capacity)
    print round(time_to_excursion, 7)


if __name__ == "__main__":
    main()
