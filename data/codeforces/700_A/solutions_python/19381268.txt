import math

a = raw_input().split()
n = float(a[0])
l = float(a[1])
v1 = float(a[2])
v2 = float(a[3])
k = float(a[4])
alfa = math.ceil(n/k) - 1

l1 = l * (v1 + v2) / (2 * v1 * alfa + v1 + v2)
t = l1/v2 + (l - l1)/v1

print(t)
