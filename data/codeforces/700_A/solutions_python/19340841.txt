if __name__ == "__main__":

    s = raw_input().split()
    n = int(s[0])
    l = float(s[1])
    v1 = float(s[2])
    v2 = float(s[3])
    k = int(s[4])

    groups = (n / k)
    if (n % k != 0):
        groups += 1
    t_small = l * 1.0 / (2.0 * v2 * v1 / (v1 + v2) * (groups - 1) + v2)
    t = l * 1.0 / v1 - (v2 - v1) * 1.0 / v1 * t_small
    print t
