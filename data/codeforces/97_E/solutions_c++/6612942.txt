const int N=100010;
const int M=2000010;
int q,n,m,i,j,k,l,p,a[M][2];
struct graph
{
	int edge,e[M],b[M],w[M],fir[N];
	void add(int x,int y,int z){e[++edge]=y;w[edge]=z;b[edge]=fir[x];fir[x]=edge;}
}G,H;

int p0[N],p1[N],fa[N];bool o[N];
int get(int*f,int x){return (f[x]==x)?x:f[x]=get(f,f[x]);}
int dep[N],lca[N];

void dfs(int i)
{
	for(int k=G.fir[i];k;k=G.b[k])
	{
		int u=G.e[k];
		if(!dep[u])
		{
			fa[u]=i;dep[u]=dep[i]+1;dfs(u);p1[u]=i;
			if(get(p0,i)==get(p0,u))o[i]|=o[u];
		}
		else if(dep[u]+1<dep[i])
		{
			int z=get(p0,i);
			while(dep[z]>dep[u]+1)
				p0[z]=fa[z],z=get(p0,z);
			if((dep[i]-dep[u])%2==0)o[i]=1;
		}
	}
	for(int k=H.fir[i];k;k=H.b[k])
		lca[H.w[k]]=get(p1,H.e[k]);
}

int tot[N];
void go(int i)
{
	if(o[get(p0,i)])tot[i]++;
	for(int k=G.fir[i];k;k=G.b[k])
	{
		int u=G.e[k];
		if(dep[u]==dep[i]+1)
		{
			if(get(p0,i)==get(p0,u))o[u]|=o[i];
			tot[u]=tot[i];go(u);
		}
	}
}
bool check(int x,int y,int w)
{
	//printf("chk= %d %d %d\n",x,y,w);
	if(get(p1,x)!=get(p1,y))return 0;
	if((dep[x]-dep[y])%2)return 1;
	return tot[x]+tot[y]-tot[w]*2>0;
}

	rep(i,1,n)p0[i]=p1[i]=i;
	rep(i,1,n)if(!dep[i])dep[i]=1,dfs(i),go(i);
	rep(i,1,q)
	printf("%s\n",check(a[i][0],a[i][1],lca[i])?"Yes":"No");
	
    return 0;
}
