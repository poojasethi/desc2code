n = int(raw_input())
friend = [ map(int, list(raw_input())) for _ in range(n)]
lgor = map(int, raw_input().split(' '))
participant=[]
while 0 in lgor:
    newer = lgor.index(0)
    participant.append(newer)
    lgor = [lgor[m] - friend[newer][m] for m in range(n)]
print len(participant)
if len(participant) > 0:
    A = [str(m + 1) for m in participant]
    print ' '.join(A)
