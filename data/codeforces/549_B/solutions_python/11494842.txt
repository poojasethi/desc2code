n=int(raw_input())
matrix = []
for i in range(n): matrix.append(raw_input())
guess=map(int, raw_input().split())
ans=[]
while 0 in guess:
    index0=guess.index(0)
    ans.append(index0)
    for i in range(n): guess[i]-=int(matrix[index0][i])
print len(ans)
if len(ans)>0:
    print " ".join([str(x+1) for x in ans])
