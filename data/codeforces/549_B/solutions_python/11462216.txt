
import sys
stdin = sys.stdin

n = int(stdin.readline().strip())
grid = [[0 if c == '0' else 1 for c in stdin.readline().strip()]
        for i in range(n)]
aa = map(int, stdin.readline().strip().split())
should = [False] * n
multiple = [0] * n
while True:
    for i in range(n):
        multiple[i] = 0
        for k in range(n):
            multiple[i] += should[k] * grid[k][i]

    updated = False
    for i in range(n):
        if multiple[i] == aa[i]:
            should[i] = True
            updated = True

    if not updated: break

print sum(should)
who = [index for coming, index in zip(should, range(1, n+1)) if coming]

print ' '.join(map(str, who))
