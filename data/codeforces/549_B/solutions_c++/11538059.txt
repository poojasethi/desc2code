#include<bits/stdc++.h>
using namespace std;
string g[102];
int main()
{
  vector<int> l;
  int ans=0,i,j,n,a[102],f[102]={0};
  cin>>n;
  for(i=0;i<n;i++)
    cin>>g[i];
  for(i=0;i<n;i++)
    cin>>a[i];
  while(1)
  {
    int k=0;
    for(i=0;i<n;i++)
    {
      if(!f[i] && !a[i])
      {
        k=1;
        ans++;
        f[i]=1;
        l.push_back(i+1);
        for(j=0;j<n;j++)
        {
          if(g[i][j]=='1')
            a[j]--;
        }
      }
    }
    if(!k)
      break;
  }
  cout<<ans<<endl;
  for(i=0;i<ans;i++)
    cout<<l[i]<<" ";
  return 0;
}
