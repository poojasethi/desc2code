#include <bits/stdc++.h>
using namespace std;

string grid[105];
int a[105],n;
vector<int>res;

void solve(){

    int e=-1;
    for(int i=0;i<n;i++)if(!a[i]){e=i; break;}
    if(e==-1)return ;

    res.push_back(e);
    for(int i=0;i<n;i++)if(grid[e][i]=='1')a[i]--;
    solve();

}

int main(){
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>grid[i];
     for(int i=0;i<n;i++)cin>>a[i];

     solve();
     cout<<(int)res.size()<<"\n";
     for(int i=0;i<(int)res.size();i++){
        cout<<res[i]+1;
        if(i==(int)res.size()-1)cout<<"\n"; else cout<<" ";
     }
}
