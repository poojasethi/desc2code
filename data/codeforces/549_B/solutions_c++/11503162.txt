#include<bits/stdc++.h>
using namespace std;

string s[100];
int a[100], go[100], n, cnt, cont;
int main() {
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for (int i = 0; i < 2 * n; i++)
		if (i < n)
			cin >> s[i];
		else
			cin >> a[i - n];

	for (int i = 0; i < n * n; i++)
		if (a[i % n] == 0) {
			cont = true;
			go[i % n] = true;
			cnt++;
			for (int j = 0; j < n; j++)
				if (s[i % n][j] == '1')
					a[j]--;
		}
	cout << cnt << endl;
	for (int i = 0; i < n; i++)
		if (go[i])
			cout << i + 1 << " ";
	cout << endl;

	return 0;
}
