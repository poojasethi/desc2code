#include <bits/stdc++.h>
using namespace std;
const int maxn = 105;
char str[maxn][maxn];
int a[maxn];
bool vis[maxn];
vector <int> sol;
int main() {
	int n;
	scanf("%d", &n);
	for(int i = 0; i < n; i++) {
		scanf("%s", str[i]);
	}
	for(int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	do {
		int flag = -1;
		for(int i = 0; i < n; i++) if(!vis[i] && a[i] == 0) flag = i;
		if(flag < 0) break;
		sol.push_back(flag);
		vis[flag] = 1;
		for(int i = 0; i < n; i++) a[i] -= (str[flag][i]) - '0';
	} while(1);

	printf("%d\n", (int)sol.size());
	for(int i = 0; i < sol.size(); i++) printf("%d%c", sol[i] + 1, i + 1 == sol.size() ? '\n' : ' ');
	return 0;
}
