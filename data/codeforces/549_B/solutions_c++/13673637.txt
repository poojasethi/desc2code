#include <cstdio>

const int MaxN = 105;

int n, li_n = 0;
char contacts[MaxN][MaxN];
int number[MaxN];
int li[MaxN];

int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; ++i)
		scanf("%s", contacts[i]);
	for (int i = 0; i < n; ++i)
		scanf("%d", &number[i]);

	while (true)
	{
		int zero = -1;
		for (int i = 0; i < n; ++i)
		{
			if (number[i] == 0)
				zero = i;
		}
		if (zero == -1)
			break;

		li[li_n++] = zero + 1;
		for (int i = 0; i < n; ++i)
		{
			if (contacts[zero][i] == '1')
				--number[i];
		}
	}

	printf("%d\n", li_n);
	for (int i = 0; i < li_n; ++i)
		printf("%d ", li[i]);

	return 0;
}
