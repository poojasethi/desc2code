#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int n,a[101][101],b[101],m,ans[101];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	    for(int j=1;j<=n;++j)scanf("%1d",&a[i][j]);
    for(int i=1;i<=n;++i)scanf("%d",&b[i]);
	while(true){
		int t=0;
		for(int i=1;i<=n;++i)
		    if(!b[i]){ans[++m]=t=i;break;}
		if(!t)break;
		for(int i=1;i<=n;++i)b[i]-=a[t][i];
	}
	sort(ans+1,ans+m+1);
	printf("%d\n",m);
	for(int i=1;i<=m;++i)printf("%d ",ans[i]);
}
