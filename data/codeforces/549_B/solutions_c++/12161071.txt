#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;

int n, x;
char g[111][111];
int exp[111], s[111];
vector<int> ans;

int same() {
	for(int i=0;i<n;i++)
		if(exp[i] == s[i]) return i;
	return -1;
}

int main() {
	scanf("%d", &n);
	for (int i=0;i<n;i++) scanf("%s", g[i]);
	for(int i=0;i<n;i++) scanf("%d", &exp[i]);
	
	ans.clear();
	memset(s, 0, sizeof s);
	while((x = same()) >= 0) {
		ans.push_back(x);
		for(int i=0;i<n;i++)
			s[i] += g[x][i] - '0';
	}
	
	printf("%d\n", ans.size());
	for(int i=0;i<ans.size();i++) {
		if(i) printf(" ");
		printf("%d", ans[i]+1);
	}
	printf("\n");
	
	return 0;
}
