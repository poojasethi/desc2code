    cin >> n;
    f(i,2*n)if (i < n)cin >> s[i];else cin >> a[i - n];
    f(i,n*n)if(a[i%n]==0){
            g[i%n]=1;c++;f(j,n)if(s[i%n][j]=='1')a[j]--;
        }
    cout<<c<<endl;
    f(i,n)if(g[i])cout<<i+1<<' ';
}
