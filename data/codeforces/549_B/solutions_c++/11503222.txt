#include<iostream>
using namespace std;
string s[100];
int a[100],g[100],n,c,i,j;
int main(){
    cin >> n;
    i=-1;
    while(++i<2*n)
        if (i < n)
            cin >> s[i];
        else
            cin >> a[i - n];
    i=-1;
    while(++i<n*n)
        if(a[i%n]==0){
            g[i%n]=1;c++;j=-1;
            while(++j<n)
                if(s[i%n][j]=='1')
                    a[j]--;
        }
    cout<<c<<endl;i=-1;
    while(++i<n)if(g[i])cout<<i+1<<' ';
    cout<<endl;
}
