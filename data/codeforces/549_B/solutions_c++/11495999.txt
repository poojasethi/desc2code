#include<bits/stdc++.h>
#define maxi 105
using namespace std;

int arr[maxi];
char ch[maxi][maxi];
int n;
vector<int> temp;
int main()
{
	cin >> n;
	for(int i=0;i<n;i++)
	{
		cin >> ch[i];
	}

	for(int i=0;i<n;i++)
	{
		cin >> arr[i];
	}

    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		if(arr[j]==0)
    		{
    			for(int k=0;k<n;k++)
    			{
    				if(ch[j][k]=='1')
    				{
    					arr[k]--;
    				}
    			}
    			temp.push_back(j+1);
    		}

    	}
    }

    int l = temp.size();
    cout<<l<<endl;
    for(int i=0;i<l;i++)
    {
    	cout<<temp[i]<<" ";
    }
    cout<<endl;

	return 0;
}
