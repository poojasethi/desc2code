#include<iostream>
#include<algorithm>
#include<string.h>
#include<vector>
using namespace std;

char c[105][105];
int a[105];
vector< int >ans;

int main()
{
    int i,j,n,m,temp;
    cin>>n;
    for(i=1;i<=n;i++) {
        for(j=1;j<=n;j++) {
            cin>>c[i][j];
        }
    }
    for(i=1;i<=n;i++) cin>>a[i];
    while(1) {
        for(i=1;i<=n;i++) {
            if(a[i]==0) break;
        }
        if(i>n) break;
        ans.push_back(i);
        for(j=1;j<=n;j++) {
            if(c[i][j]=='1') a[j]--;
        }
    }
    cout<<ans.size()<<"\n";
    for(i=0;i<ans.size();i++) cout<<ans[i]<<" ";
    return 0;
}
