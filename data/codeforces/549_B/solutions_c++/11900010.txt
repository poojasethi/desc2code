# include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	char input[n][n+1];
	int guess[n];
	int out[n];
	for (int i = 0; i < n; ++i)
		scanf("%s",input[i]);
	for (int i = 0; i < n; ++i)
		scanf("%d",&guess[i]);
	int count=0;
	while(1)
	{
		int i;
		for (i = 0; i < n; ++i)
			if(guess[i]==0)
				break;
		if(i==n)
			break;
		for (int j = 0; j < n; ++j)
			if(input[i][j]=='1')
				guess[j]--;
		out[count]=i;
		count++;
	}
	printf("%d\n",count);
	for (int i = 0; i < count; ++i)
		printf("%d ",out[i]+1);
	printf("\n");
	return 0;
}
