#include <iostream>
#include <vector>
using namespace std;

int n, a[111];
vector<int> v;
bool mark[111];
string s[111];

int main()
{
	cin >> n;
	for(int i=0; i<n; i++)
		cin >> s[i];

	for(int i=0; i<n; i++)
		cin >> a[i];

	for(int i=0; i<n; i++)
	{
		int id = -1;
		for(int j=0; j<n; j++)
			if(a[j] == 0 && !mark[j])
				id = j;

		if(id == -1)
			break;

		v.push_back(id);
		mark[id] = 1;
		for(int j=0; j<n; j++)
			if(s[id][j] == '1')
				a[j]--;
	}

	cout << v.size() << endl;
	for(int i=0; i<v.size(); i++)
		cout << v[i]+1 << ' ';
	cout << endl;	

	return 0;
}
