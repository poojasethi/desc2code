#include<iostream>
#include<cstring>
#include<vector>
using namespace std;

const int maxn = 200;
string s[maxn];
vector<int> l;
int d[maxn], n;

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> n;
	for (int i=0 ; i<n ; i++)
		cin >> s[i];
	for (int i=0 ; i<n ; i++)
		cin >> d[i];
	while(1){
		int I = -1;
		for (int i=0 ; i<n ; i++)
			if (d[i]==0)
				I = i;
		if (!~I) break;		
		l.push_back(I);
		for (int i=0 ; i<n ; i++)
			if (s[I][i]=='1')
				d[i]--;
	}
	cout << l.size() << endl;
	for (int i=0 ; i<l.size() ; i++)
		cout << l[i]+1 << " ";
	cout << endl;	
}
