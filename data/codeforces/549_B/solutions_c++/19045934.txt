#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[101][101],f[101],an[101],o,n;
void pan(int i)
{if(f[i]==0)
{int h=0,g[100]={0};
f[i]--;
an[o++]=i;
for(int j=1;j<=n;j++)
if(a[i][j]&&i!=j){f[j]--;g[h++]=j;}
for(int j=0;j<h;j++)
pan(max(1,g[j]));
}
}
int main()
{std::ios::sync_with_stdio(false);
cin>>n;
for(int i=1;i<=n;i++)
for(int j=1;j<=n;j++)
{char u;
cin>>u;
a[i][j]=u-'0';
}
for(int i=1;i<=n;i++)
cin>>f[i];
for(int i=1;i<=n;i++)
pan(i);
cout<<o<<endl;
for(int i=0;i<o;i++)
cout<<an[i]<<' ';
return 0;
}
