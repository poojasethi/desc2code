#include <cstdio>
#include <vector>

using namespace std;

const int N = 111;

int n;
int a[N], b[N];
char e[N][N];

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%s", e[i]);
	}
	for (int i = 0; i < n; ++i) {
		scanf("%d", &a[i]);
	}
	vector<int> ans;
	for (int loop = 0; loop < n; ++loop) {
		for (int i = 0; i < n; ++i) {
			if (b[i] == a[i]) {
				ans.push_back(i);
				for (int j = 0; j < n; ++j) {
					if (e[i][j] == '1') {
						b[j]++;
					}
				}
			}
		}
	}
	printf("%d\n", (int)ans.size());
	for (int i = 0; i < (int)ans.size(); ++i) {
		printf("%d\n", ans[i] + 1);
	}
	return 0;
}
