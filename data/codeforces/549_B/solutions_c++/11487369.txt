#include <iostream>
#include <cstdio>
#include <vector>
#define N 101
using namespace std;
int n,d[N];
bool g[N][N];
vector <int> ans;
int main()
{
	int i,j;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			scanf("%1d",&g[i][j]);
	for(i=1;i<=n;i++)
		scanf("%d",&d[i]);
	while(1)
	{
		int p;
		for(i=1;i<=n;i++)
			if(!d[i])
			{
				p=i;
				break;
			}
		if(i>n)break;
		ans.push_back(p);
		for(i=1;i<=n;i++)
			if(g[p][i])
				d[i]--;
	}
	printf("%d\n",ans.size());
	for(i=0;i<ans.size();i++)
		printf("%d ",ans[i]);
	puts("");
	return 0;
}
