#include<bits\stdc++.h>
using namespace std;
char a[200][200];
int main()
{
	int n;
	cin>>n;
	int b[n];
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n;i++)for(int j=0;j<n;j++)a[i][j]-='0';
	for(int i=0;i<n;i++)cin>>b[i]; 
	vector<int> ans;
	bool w=true;
	while(w)
	{
		w=false;
		for(int i=0;i<n;i++)if(b[i]==0)
		{
			ans.push_back(i+1);
			for(int j=0;j<n;j++)b[j]-=a[i][j];
			w=true;
		} 
	}
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)cout<<ans[i]<<' ';
	return 0;
}
