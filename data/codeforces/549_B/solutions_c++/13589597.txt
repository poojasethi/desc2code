#include<cstdio>
#include<iostream>
#define N 1000
using namespace std;
int a[N][N];
int b[N];
int ans[N];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	{
		char ch[N];
		scanf("%s",ch);
		for(int j=0;j<n;++j)
			a[i][j+1]=ch[j]-'0';
	}
	int num1=0;
	for(int i=1;i<=n;++i)
	{
		scanf("%d",&b[i]);
		num1+=(b[i]>0);
	}
	if(num1==n)
	{
		cout<<0;
		return 0;
	}
	int tot=0;
	for(;;)
	{
		int flag=0;
		for(int i=1;i<=n;++i)
		if(!b[i])
		{
			flag=1;
			for(int j=1;j<=n;++j)
				if(a[i][j])--b[j];
			ans[++tot]=i;
		}
		if(!flag)break;
	}
	cout<<tot<<"\n";
	for(int i=1;i<=tot;++i)
		cout<<ans[i]<<" ";
}
