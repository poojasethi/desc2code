#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
char a[100][101];
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t=0,s[100],d[100],f[100];
  cin>>q;
  for(e=0;e<q;e++)
   cin>>a[e];
  for(w=0;w<q;w++){
   cin>>s[w];
   d[w]=0;}
  bool b=1;
  while(b){
   b=0;
   for(w=0;w<q;w++)
    if(d[w]==s[w]){
     f[t]=w;
     t++;
     for(e=0;e<q;e++)
      if(a[w][e]=='1')
       d[e]++;
     b=1;
     break;}}
  cout<<t<<"\n";
  for(w=0;w<t;w++)
   cout<<f[w]+1<<" ";
  return 0;}
