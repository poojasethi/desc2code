#include<iostream>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<cstdio>
using namespace std;
int b[110],n,ans,an[110],now;string a[110];
bool judge() {
	for(now=1;now<=n;now++) 
		if(b[now]==0) return true;
	return false;
}
int main() {
	scanf("%d",&n);
	for(int i=1;i<=n;i++) cin>>a[i];
	for(int i=1;i<=n;i++) scanf("%d",&b[i]);
	while(judge()) {
		ans++;an[ans]=now;
		for(int i=1;i<=n;i++) 
			if(a[now][i-1]=='1') b[i]--;
	}
	printf("%d\n",ans);sort(an+1,an+ans+1);
	for(int i=1;i<=ans;i++) printf("%d ",an[i]);
 	return 0;
}
