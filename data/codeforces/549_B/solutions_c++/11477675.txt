#include <cstdio>

int N, in[100], sol, A[100];
bool graf[100][100], inv[100];

void ukljuci(int x) {
	sol++;
	inv[x] = 1;
	for(int i = 0; i < N; ++i)
		in[i] += graf[x][i];
	for(int i = 0; i < N; ++i)
		if(in[i] == A[i]) {
			ukljuci(i);
			i = 0;
		}
	return;
}

int main() {
	scanf("%d", &N);
	for(int i = 0; i < N; ++i) {
		char C[102];
		scanf("%s", C);
		for(int j = 0; j < N; ++j)
			if(C[j] == '1')
				graf[i][j] = 1;
	}
	for(int i = 0; i < N; ++i)
		scanf("%d", A+i);
	for(int i = 0; i < N; ++i)
		if(A[i] == in[i]) {
		 	ukljuci(i);
			i = 0;
		}
	printf("%d\n", sol);
	for(int i = 0; i < N; ++i)
		if(inv[i] == 1)
			printf("%d ", i+1);
	return 0;
}
