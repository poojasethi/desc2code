#include<iostream>
#include<string>
using namespace std;
int n,b[101],i,j,l,c[101],pos;
char a[101][101];
int main()
{
cin>>n;
for(i=1;i<=n;i++)
for(j=1;j<=n;j++)
cin>>a[i][j];
for(i=1;i<=n;i++)
cin>>b[i];
for(i=1;i<=n;i++)
{
for(j=1;j<=n;j++)
{
if(b[j]==0)
{
for(l=1;l<=n;l++)  
b[l]-=(a[j][l]=='1');                  
c[++pos]=j;
}          
}
}   
cout<<pos<<endl;
for(i=1;i<=pos;i++)
cout<<c[i]<<" ";
return 0;
}
