#include<stdio.h>

int con[110],n,i,j,k,in[110],c;
char mat[110][110],ch[110];

int main(){
    scanf("%d",&n);
    for(i=0;i<n;i++)scanf("%s",mat[i]);
    for(i=0;i<n;i++)scanf("%d",&in[i]);
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if(con[j]==in[j]){
                ch[j]=1;
                c++;
                for(k=0;k<n;k++){
                    con[k]+=mat[j][k]-'0';
                }
                break;
            }
        }
    }
    printf("%d\n",c);
    for(i=0;i<n;i++){
        if(ch[i]==1)printf("%d ",i+1);
    }
    printf("\n");
}
