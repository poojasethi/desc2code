#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int n,a[105],q[105],tail,sum;
bool bo[105];
char s[105][105];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i)	scanf("%s",s[i]+1);
	for (int i=1;i<=n;++i) scanf("%d",a+i);
	tail=0;
	for (int i=1;i<=n;++i) 
		if (a[i]==0) q[++tail]=i;
	int x;
	for (int j=1;j<=tail;++j)
	{
		x=q[j];
		if (a[x]<0) continue;
		++sum;
		bo[x]=true;
		for (int i=1;i<=n;++i)
			if (a[i]>0&&s[x][i]=='1')
			{
				--a[i];
				if (a[i]==0) q[++tail]=i;
			}
	}
	printf("%d\n",sum);
	for (int i=1;i<=n;++i)
		if (bo[i]) printf("%d ",i);
	printf("\n");
}
