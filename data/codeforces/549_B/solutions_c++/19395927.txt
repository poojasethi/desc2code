#include<cstdio>
#include<cstring>

int n,cans,a[111];
bool pd[111];
char s[111][111];

int main()
{
//	freopen("test.in","r",stdin);
	scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%s",s[i]);
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)if(a[j]==0){
			cans++;
			pd[j]=true;
			for(int k=0;k<n;k++)if(s[j][k]=='1'){
				a[k]--;
			}
		}
	}
	printf("%d\n",cans);
	for(int i=0;i<n;i++)if(pd[i])printf("%d ",i+1);
}
