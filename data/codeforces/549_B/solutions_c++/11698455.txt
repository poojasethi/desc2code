#define N (1<<7)
#include <bits/stdc++.h>
using namespace std;

char s[N][N];
int n,x,a[N];
vector<int> res;

int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>(s[i]+1);

	for(int i=1;i<=n;i++)
		scanf("%d",a+i);

	bool flag=1;
	for(;flag;)
	{
		flag=0;
		for(int i=1;i<=n;i++)
			if(a[i]==0)
			{
				flag=1;
				for(int j=1;j<=n;j++)
					if(s[i][j]=='1') a[j]--;
				res.push_back(i);
			}
	}

	cout<<res.size()<<endl;
	for(int i=0;i<res.size();i++)
		cout<<res[i]<<" ";
	cout<<endl;
	return 0;
}
