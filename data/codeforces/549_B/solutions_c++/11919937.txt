#include<cstdio>
#include<cstring>
#include<algorithm>
#define rep(i,n) for(int i=1; i<=n; i++)
int n,l=0,r=0,b[200],q[200],ans[200];
char a[200][200];
int main(){
    scanf("%d\n",&n);
    memset(ans,0,sizeof(ans));
    rep(i,n){
        rep(j,n) a[i][j]=getchar();
        getchar();
    }
    rep(i,n){
        scanf("%d",&b[i]);
        if (b[i]==0) q[++r]=i;
    }
    while (l<r){
        l++;
        ans[q[l]]=1;
        rep(i,n) if (a[q[l]][i]=='1'){
            b[i]--;
            if (b[i]==0) q[++r]=i;
        }
    }
    printf("%d\n",r);
    rep(i,n) if (ans[i]==1) printf("%d ",i);
}
