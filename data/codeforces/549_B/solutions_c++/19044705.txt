#include <cstdio>
#include <vector>
bool a[105][105],u[105];
int lim[105],ans[105],n,ansr=0;

int main()
{
	int i,j,k,pr;
	char c[105];
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		scanf("%s",c+1);
		for (j=1;j<=n;j++) a[i][j]=c[j]-'0';
	}
	for (i=1;i<=n;i++)
		scanf("%d",&lim[i]);
	for (i=1;i<=n;i++)
	{
		for (pr=1;pr<=n;pr++)
			if (ans[pr]==lim[pr]) break;
		if (pr>n) break;
		for (j=1;j<=n;j++)
			if (a[j][pr] && !u[j])
			{
				ansr++; u[j]=true;
				for (k=1;k<=n;k++) if (a[j][k]) ans[k]++;
				break;
			}
	}
	printf("%d\n",ansr);
	for (i=1;i<=n;i++)
		if (u[i]) printf("%d ",i);
}
