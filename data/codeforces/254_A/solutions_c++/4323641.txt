#include <cstdio>

using namespace std;
int a[300005],b[300005],c[5005],n,k,i,j;
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);n<<=1;
    for(i=1;i<=n;i++)
    {
        scanf("%d",&j);
        if(c[j]){a[++k]=c[j];b[k]=i;c[j]=0;}
        else c[j]=i;
    }
    n>>=1;
    if(k<n){printf("-1");return 0;}
    for(i=1;i<=n;i++)
        printf("%d %d\n",a[i],b[i]);
    return 0;
}
