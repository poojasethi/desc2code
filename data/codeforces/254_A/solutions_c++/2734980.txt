#include<fstream>
#include<map>
#include<sstream>

using namespace std;

int main(){
  ifstream fin("input.txt");
  ofstream fout("output.txt");
  stringstream ss;
  int n;
  fin>>n;
  map<int,int> m;
  for(int i=1;i<=n*2;i++){
    int c;
    fin>>c;
    if(m.count(c)){
      ss<<i<<' '<<m[c]<<endl;
      m.erase(c);
    }else{
      m[c]=i;
    }
  }
  if(m.empty()){
    fout<<ss.str();
  }else{
    fout<<-1<<endl;
  }
  return 0;
}
