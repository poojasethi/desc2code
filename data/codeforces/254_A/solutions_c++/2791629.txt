#include <cstdio>
#include <vector>

using namespace std;

int n;
vector<int>a[5050];
int main(void) {
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for(int i=1,x;i<=n*2;++i)scanf("%d",&x),a[x].push_back(i);
    for(int i=0;i<=5000;++i)if(a[i].size()&1){puts("-1");return 0;}
    for(int i=0;i<=5000;++i)for(int j=a[i].size();j>0;j-=2)printf("%d %d\n",a[i][j-2],a[i][j-1]);
    
    return 0;
}
