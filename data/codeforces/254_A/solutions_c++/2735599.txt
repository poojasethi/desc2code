#include<stdio.h>
using namespace std;
int P[2][1000000],B[5001]={0};
int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,a;
	int i,j;
	j=0;
	scanf("%d",&n);
	for(i=1;i<=2*n;i++) {
		scanf("%d",&a);
		if(B[a]) {
			P[0][j]=B[a];
			P[1][j]=i;
			j++;
			B[a]=0;
		} else {
			B[a]=i;
		}
	}
	if(j<n ) {
		printf("-1");
		return 0;
	}
	for(i=0;i<j;i++) printf("%d %d\n",P[0][i],P[1][i]);
	return 0;
}
