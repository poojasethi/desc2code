#include <stdio.h>

int flag[50001];
int ans[3 * 100000 + 1][2];

int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n;
	scanf("%d",&n);
	int now = 0;
	for (int i = 1;i <= 2 * n;i++){
		int x;
		scanf("%d",&x);
		if (flag[x]){
			ans[now][0] = flag[x];
			ans[now++][1] = i;
			flag[x] = 0; 	
		}
		else
			flag[x] = i;
	}
	if (now != n)
		printf("-1\n");
	else
		for (int i = 0;i < now;i++)
			printf("%d %d\n",ans[i][0],ans[i][1]);
}
