#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
vector<int>card[5001];
int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,a;
	cin>>n;
	n+=n;
	for(int i=1;i<=n;i++) {
		cin>>a;
		card[a].push_back(i);
	}
	for(int i=1;i<=5000;i++)
		if(card[i].size()&1){cout<<-1;return 0;}
	for(int i=1;i<=5000;i++)
		for(int j=0;j<card[i].size();j+=2)
			cout<<card[i][j]<<" "<<card[i][j+1]<<endl;
	return 0;
}
