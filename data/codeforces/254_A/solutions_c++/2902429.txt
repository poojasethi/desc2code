#include<cstdio>
int n,s[5005];
int ans[300005][2],p=0;
int main(){
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
for(int i=1;i<=5000;i++) s[i]=-1;
scanf("%d",&n);
for(int i=0;i<n*2;i++){
int in;
scanf("%d",&in);
if(s[in]==-1) s[in]=i+1;
else{
ans[p][0]=s[in];
ans[p][1]=i+1;
p++;
s[in]=-1;
}
}
if(p!=n) printf("-1\n");
else{
for(int i=0;i<p;i++){
printf("%d %d\n",ans[i][0],ans[i][1]);
}
}
return 0;
}
