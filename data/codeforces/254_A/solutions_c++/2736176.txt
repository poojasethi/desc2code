#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int A[600001],O[600001];
int N;

bool cmp( int a, int b)
{
	return A[a]<A[b];
}

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&N);
	for (int i=1;i<=2*N;i++) scanf("%d",A+i),O[i]=i;
	sort(O+1,O+2*N+1,cmp);
	bool f=0;
	for (int i=1;i<=N;i++)
		if (A[O[2*i]]!=A[O[2*i-1]]) f=1;
	if (f) puts("-1");
	else for (int i=1;i<=N;i++) printf("%d %d\n",O[i*2],O[i*2-1]);
}
