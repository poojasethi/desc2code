#include <bits/stdc++.h>
using namespace std;

int n,i,f[6666];
pair<int,int>a[900000];

int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);n*=2;
    for(i=1;i<=n;++i)scanf("%d",&a[i].first),a[i].second=i,++f[a[i].first];
    for(i=1;i<=5000;++i)if(f[i]&1)return puts("-1"),0;
    sort(a+1,a+n+1);for(i=1;i<=n;++i)if(i&1)printf("%d %d\n",a[i].second,a[i+1].second);
    return 0;
}
