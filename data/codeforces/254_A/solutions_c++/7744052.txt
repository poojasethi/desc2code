#include<cstdio>
using namespace std;
int a1[5001],a2[300000],a3[300000];
bool b1[5001];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,m,a,l=0;
	scanf("%d",&n);
	m=2*n;
	for(int i=1;i<=m;++i)
	{
		scanf("%d",&a);
		if(b1[a])
		{
			a2[l]=a1[a];
			a3[l++]=i;
		}
		else a1[a]=i;
		b1[a]^=1;
	}
	if(l!=n){puts("-1");return 0;}
	for(int i=0;i<n;++i)
		printf("%d %d\n",a2[i],a3[i]);
}
