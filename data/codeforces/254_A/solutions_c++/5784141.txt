#include<fstream>

using namespace std;
ifstream fin("input.txt");
ofstream fout("output.txt");
int k[300005],l[300005],a[5005];

int main()
{
    int n,i,f=0,x;
    fin>>n;
    for (i=1; i<=2*n; i++)
    {
        fin>>x;
        if (a[x]==0) a[x]=i;
           else {k[++f]=a[x]; l[f]=i; a[x]=0;}             
    }    
    if (f==n) for (i=1; i<=f; i++) fout<<k[i]<<" "<<l[i]<<endl;
       else fout<<-1;
    return 0;
}
