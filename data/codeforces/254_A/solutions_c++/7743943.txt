#include<cstdio>

int a[300005][2];
int b[5005] = {0};


int main(){
	int n, i, k, t;
	
	freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
	scanf("%d", &n);
	
	for(i = 0, k = 0; i < 2*n ; i++){
		scanf("%d", &t);
		
		if(b[t] == 0){
			b[t] = i + 1;
		}else{
			a[k][0] = b[t];
			a[k][1] = i + 1;
			b[t] = 0;
			k++;
		}
	}
	
	if( k < n){
		printf("-1\n");
	}else{
		for(i = 0; i < n; i++){
			printf("%d %d\n", a[i][0], a[i][1]);
		}
	}
	
	return 0;
}
