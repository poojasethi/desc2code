#include <iostream>
#include <vector>
#include <stdio.h>

using namespace std;

int main()
{
vector <int> a[5001];
int kol[5001]={0};
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
int n,x;
cin>>n;
for (int i=1;i<=2*n;i++)
 {
  cin>>x; a[x].push_back(i); kol[x]++;
 }

for (int i=1;i<=5000;i++)
 if (kol[i]%2!=0) {cout<<-1; return 0;}

for (int i=1;i<=5000;i++)
 for (int j=0;j<kol[i];j+=2)
  cout<<a[i][j]<<" "<<a[i][j+1]<<endl;
return 0;
}
