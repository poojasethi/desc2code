#include <stdio.h>
int r[2][300000];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	int n,a[5000],i,j=0,t,te;
	for(i=0;i<5000;i++) a[i]=0; 
	scanf("%d",&n);
	for(i=1;i<=2*n;i++){
		scanf("%d",&t);
		if(a[t-1]){
			r[0][j]=a[t-1];
			r[1][j++]=i;
			a[t-1]=0;
		}else a[t-1]=i;
	}
	te=0;
	while(te<5000)if(a[te++]){
		puts("-1");
		j=0;
		break;
	}
	for(i=0;i<j;i++){
		printf("%d %d\n",r[0][i],r[1][i]);
	}
	return 0;
}
