#include<stdio.h>
#include<vector>

using std::vector;

const int MAXA=5000;

vector<int> card[MAXA+1];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,i,j;
	scanf("%d",&n);
	for(i=1;i<=n<<1;++i)
	{
		scanf("%d",&j);
		card[j].push_back(i);
	}
	for(i=1;i<=MAXA;++i)
		if(card[i].size()&1)
		{
			puts("-1");
			return 0;
		}
	for(i=1;i<=MAXA;++i)
		for(unsigned k=0;k<card[i].size();k+=2)
			printf("%d %d\n",card[i][k],card[i][k+1]);
	return 0;
}
