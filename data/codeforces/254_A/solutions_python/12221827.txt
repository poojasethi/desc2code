from itertools import *
from collections import *

input  = file('input.txt')
output = file('output.txt','w')

n = int(input.readline())
d = defaultdict(list)
for i,x in enumerate(imap(int, input.readline().split())):
  d[x].append(i+1)

if all(len(x)%2==0 for x in d.itervalues()):
  for x in d:
    for i in xrange(0,len(d[x]),2):
      print >>output, d[x][i],d[x][i+1]
else:
  print >>output, -1
