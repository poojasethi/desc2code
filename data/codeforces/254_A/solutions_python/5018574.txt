import sys
from collections import defaultdict
sys.stdout=open("output.txt",'w')


C = defaultdict(list)
for (i,e) in enumerate(open("input.txt").read().split()[1:],1):
    C[e].append(i)


for x in C.values():
    if len(x)%2:
        print -1
        exit()
for x in C.values():
    print '\n'.join("{0} {1}".format(*y) for y in zip(x[0::2],x[1::2]))
