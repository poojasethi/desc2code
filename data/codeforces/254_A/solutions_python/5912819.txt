import sys
sys.stdin=open('input.txt','r')
sys.stdout=open('output.txt','w')

n=int(input())
a = map(int,raw_input().split())

b={}

for x in range(2*n):
    if a[x] in b:
        b[a[x]].append(x+1)
    else:
        b[a[x]]=[x+1]

l=1        
for x in b.keys():
    if len(b[x])%2 !=0:
        l=0

if l:
    print "\n".join("\n".join("{0} {1}".format(*y) for y in zip(x[0::2],x[1::2])) for x in b.values())
else:
    print -1
