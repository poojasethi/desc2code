from collections import Counter
import sys
sys.stdin = open("input.txt", "r")
sys.stdout = open("output.txt", "w")
t = 5010 * [0]
n = int(raw_input())
a = map(int, raw_input().split())
if any(i % 2 for i in Counter(a).values()):
    print -1
else:
    for i in xrange(2*n):
        x = a[i]
        if t[x]:
            print t[x], i + 1
            t[x] = 0
        else:
            t[x] = i + 1
sys.stdout.close()
