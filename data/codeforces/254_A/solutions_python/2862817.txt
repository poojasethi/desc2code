import sys
from collections import defaultdict
sys.stdin = open('input.txt','r')
sys.stdout = open('output.txt','w')
n = input()
mp = defaultdict(list)
a = raw_input().split()
for i,v in enumerate(a):
	mp[v].append(i+1)
for x in mp.values():
	if len(x)%2:
		print -1
		exit()
for x in mp.values():
		print '\n'.join("{0} {1}".format(*y) for y in zip(x[0::2],x[1::2]))
