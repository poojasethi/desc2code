import sys
sys.stdin = open('input.txt','r')
sys.stdout = open('output.txt','w')

R = lambda:map(int,raw_input().split(' '))
n = input()
num = R()
vis = {}
ans = []
for i in xrange(len(num)):
    loc = i + 1
    if num[i] not in vis:
        vis[num[i]] = loc
    else:
        ans.append((vis[num[i]],loc))
        vis.pop(num[i])

if len(ans) == n:
    for x,y in ans:
        print x,y
else:
    print -1
