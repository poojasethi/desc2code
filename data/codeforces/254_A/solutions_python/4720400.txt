from collections import defaultdict
fout = open('output.txt','w')
fin = open('input.txt','r')

t = fin.read().split()
p = defaultdict(int)
n = int(t.pop(0))
k, ans = 0, [''] * n

for i, x in enumerate(t, 1):
    if p[x]:
        ans[k] = '{0} {1}'.format(i, p[x])
        p[x] = 0
        k += 1
    else: p[x] = i
if k == n: fout.write('\n'.join(ans))
else: fout.write('-1')

fout.close()
fin.close()
