import sys
from collections import defaultdict

def main():
    f = open('input.txt')
    sys.stdout = open('output.txt','w')
    n = int(f.readline())
    d = defaultdict(list)
    for i,v in enumerate(f.readline().strip('\n').split()):
        d[v].append(i+1)
    for v in d.values():
        if len(v) % 2:
            print(-1)
            return
    print('\n'.join('\n'.join("{0} {1}".format(*y) for y in zip(x[0::2], x[1::2])) for x in d.values()))

main()
