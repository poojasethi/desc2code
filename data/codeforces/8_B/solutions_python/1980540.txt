import sys

path = sys.stdin.read()[:-1]
grid = [ [0,] * 200 for i in range(200) ]

#start
i = 100
j = 100
grid[i][j] = 1

bug = 0

for d in path:
  if   d == 'L': j -= 1
  elif d == 'R': j += 1
  elif d == 'U': i -= 1
  elif d == 'D': i += 1

  neib = grid[i-1][j] + grid[i+1][j] + grid[i][j-1] + grid[i][j+1]

  if grid[i][j] or neib > 1:
    bug = 1
    break
  else:
    grid[i][j] = 1

if bug:
  print 'BUG'
else:
  print 'OK'
