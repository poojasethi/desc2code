I=lambda:map(int, raw_input().split())
s=raw_input()
dr = ([0,1],[1,0],[-1,0],[0,-1])
d=set()
d.add((0,0))
x = y = 0
bug = 0

for i in s:
    if i == 'U':x -= 1
    elif i == 'D':x += 1
    elif i == 'L':y -= 1
    else:y += 1
    cnt=0
    for x0,y0 in dr:
        if (x+x0,y+y0) in d: 
            cnt+=1
           
    if (x,y) in d or cnt!=1:
        bug = 1
        break
    d.add((x,y))
    
print ['OK', 'BUG'][bug]
