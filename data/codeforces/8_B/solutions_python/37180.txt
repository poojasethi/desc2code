path = raw_input()
pos = (0,0)
visit = set()
visit.add(pos)

for c in path:
	if c == "L":
		t = (pos[0]+1,pos[1])
	if c == "R":
		t = (pos[0]-1,pos[1])
	if c == "U":
		t = (pos[0],pos[1]+1)
	if c == "D":
		t = (pos[0],pos[1]-1)
	
	s = set()
	s.add( t )
	s.add( (t[0]+1,t[1]) )
	s.add( (t[0]-1,t[1]) )
	s.add( (t[0],t[1]+1) )
	s.add( (t[0],t[1]-1) )
	s -= set([pos])
	
	for x in s:
		if x in visit:
			print "BUG"
			exit(0)
	
	visit.add(t)
	pos = t

print "OK"
