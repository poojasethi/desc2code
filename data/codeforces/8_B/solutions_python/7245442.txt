import sys

def shortest(s):
    x,y = 0,0
    p = set([(x,y)])
    
    for c in s:
        q = [(x+1,y),(x-1,y),(x,y+1),(x,y-1)] #buren 
        if c=='L':
            x,y = x-1,y
        elif c=='R':
            x,y = x+1,y
        elif c=='U':
            x,y = x,y+1
        elif c=='D':
            x,y = x,y-1
        if (x,y) in p:
            return "BUG"
        for i in q:
            p.add(i)
    return "OK"

s = sys.stdin.readline().strip()
print shortest(s)
