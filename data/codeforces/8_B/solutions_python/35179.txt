path = raw_input()
pos = set()
p = (0,0)
pos.add(p)
f = True
for c in path:
	if c == "L":
		t = (p[0]+1,p[1])
	if c == "R":
		t = (p[0]-1,p[1])
	if c == "U":
		t = (p[0],p[1]+1)
	if c == "D":
		t = (p[0],p[1]-1)
	
	s = set()
	s.add( t )
	s.add( (t[0]+1,t[1]) )
	s.add( (t[0]-1,t[1]) )
	s.add( (t[0],t[1]+1) )
	s.add( (t[0],t[1]-1) )
	s -= set([p])
	
	for x in s:
		if x in pos:
			f = False
	pos.add(t)
	p = t

print "OK" if f else "BUG"
