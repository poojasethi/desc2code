dxy = {'L':(-1, 0), 'R': (1, 0), 'U':(0, 1), 'D':(0, -1)}
def before(a, now):
	for i in dxy:
		if [now[0] + dxy[i][0], now[1] + dxy[i][1]] in a[0: len(a) - 1:] :
				return True
	return False
mov = raw_input()
a = [[0, 0]]
now = [0, 0]
for i in mov:
	now[0] += dxy[i][0]
	now[1] += dxy[i][1]
	if now in a or before(a, now):
		print 'BUG'
		exit()
	a.append(list(now))
print 'OK'
