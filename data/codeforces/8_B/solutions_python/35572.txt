import sys
s = sys.stdin.readline().strip()
d = {'L': (-1, 0), 'U': (0, 1), 'R': (1, 0), 'D': (0, -1)}
n = len(s)

a = [[False for x in xrange(n+n+10)] for y in xrange(n+n+10)]

x, y = n + 5, n + 5

a[x][y] = True

for c in s:
    px, py = x, y
    x += d[c][0]
    y += d[c][1]
    for q in d:
        nx = x + d[q][0]
        ny = y + d[q][1]
        if ((nx, ny) != (px, py) and a[nx][ny]) or a[x][y]:
            print "BUG"
            exit(0)
    a[x][y] = True
print "OK"
