Path=list(raw_input())
alreadyVisited=[(0,0)]
flag=False
dic={'L':(-1,0),'R':(1,0),'U':(0,-1),'D':(0,1)}
for i in Path:
	x,y=alreadyVisited[-1]
	a,b=x+dic[i][0],y+dic[i][1]
	for c,d in dic.values():
		if (a+c,b+d) in alreadyVisited and (a+c,b+d)!=(x,y):
			flag=True
	if flag or (a,b) in alreadyVisited:
		print 'BUG'
		break
	alreadyVisited.append((a,b))
else:
	print 'OK'
