d=[[0,1],[0,-1],[1,0],[-1,0],[0,0]]

def solve(path):
	vis = []
	cur = [0,0]
	for p in path:
		prev = cur
		if p == 'L': index = 0
		elif p == 'R': index = 1
		elif p == 'U': index = 2
		else: index = 3
		cur = [cur[0]+d[index][0], cur[1]+d[index][1]]
		if cur in vis:
			return 'BUG'
		for dx,dy in d:
			vis.append([prev[0]+dx, prev[1]+dy])
	return 'OK'

print solve(raw_input())
