S = raw_input()
dr = [[0,1],[1,0],[-1,0],[0,-1]]
mp = [[0]*300 for _ in range(300)]
x = y = 150
bug = 0
def sr(x,y):
	r = 0
	for i in xrange(4):
		tx = x + dr[i][0]
		ty = y + dr[i][1]
		r += mp[tx][ty]
	return r
for i in S:
	mp[x][y] = 1
	if i == 'U':
		x -= 1
	elif i == 'D':
		x += 1
	elif i == 'L':
		y -= 1
	else:
		y += 1
	if mp[x][y] == 1 or sr(x,y) >= 2:
		bug = 1	
print ['OK', 'BUG'][bug]
