#include <iostream>
#include <string>
using namespace std;

bool m[202][202];
bool check(int x, int y) {
	int n = 0;
	if (m[y][x+1]) ++n;
	if (m[y][x-1]) ++n;
	if (m[y+1][x]) ++n;
	if (m[y-1][x]) ++n;
	return (n <= 1);
}

int main (){
	string s;
	cin >> s;
	int x = 100;
	int y = 100;
	m[y][x] = true;
	bool exists = true;
	int i = 0;
	while (exists and i < int(s.size())){
		if (s[i] == 'L') --x;
		else if(s[i] == 'R') ++x;
		else if(s[i] == 'U') ++y;
		else --y;
		exists = (check(x, y) and !m[y][x]);
		m[y][x] = true;
		++i;
	}

	if (exists) cout << "OK" << endl;
	else cout << "BUG" << endl;
}
