#include<bits/stdc++.h>
using namespace std;
int a[300][300]={0};
int x=150,y=150;
int main()
{

    a[x][y]=1;
    char c;
    while(cin>>c)
    {
        if(c=='L')--x;
        else if(c=='R')++x;
        else if(c=='U')--y;
        else ++y;
        int num=a[x-1][y]+a[x+1][y]+a[x][y-1]+a[x][y+1]+a[x][y];
        if(num>1){cout<<"BUG";return 0;}
        a[x][y]=1;
    }
    cout<<"OK";

}
