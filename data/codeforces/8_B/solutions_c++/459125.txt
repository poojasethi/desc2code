#include <stdio.h>

int p[205][205];
int dx[4]={0, 1, 0, -1}, dy[4]={1, 0, -1, 0};

int main(void) {
	char a[105];
	scanf("%s", a);
	int x=100, y=100, i, j;
	for(i=0;a[i];i++){
	p[x][y] =1;
		int c=0;
		if(a[i]=='L')--x;
		else if(a[i]=='U') ++y;
		else if(a[i]=='R') ++x;
		else if(a[i]=='D') --y;
		if(p[x][y]) break;
		for(j=0;j<4;j++)
			c+=p[x+dx[j]][y+dy[j]];
		if(c>1)
			break;
	}
	if(a[i])
		puts("BUG");
	else
		puts("OK");
	return 0;
}
