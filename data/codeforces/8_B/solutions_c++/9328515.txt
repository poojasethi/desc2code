#include <bits/stdc++.h>
using namespace std;

const int maxn=666;

int n,i,x,y,a[maxn][maxn];
char s[maxn];

int main(){
    //freopen("input.txt","r",stdin);
    x=y=200,scanf("%s",s+1),n=strlen(s+1);
    for(i=1;i<=n;++i){
        a[x][y]=1;
        if(s[i]=='L')--x;
            if(s[i]=='R')++x;
                if(s[i]=='U')--y;
                    if(s[i]=='D')++y;
        if(a[x][y]+a[x-1][y]+a[x+1][y]+a[x][y-1]+a[x][y+1]>1)
            return puts("BUG"),0;
    }puts("OK");
    return 0;
}
