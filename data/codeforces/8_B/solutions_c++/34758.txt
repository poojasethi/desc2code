#include<cstdio>
#include<cstring>

int b[505][505];
int x=255,y=255;
char cmd[105];
int N;
int main(){
	gets(cmd);	
	N = strlen(cmd);
	
	memset(b,0,sizeof(b));
	
	b[x][y] = 1;
	for (int i=0;i<N;++i){
		if (cmd[i]=='L') --x;
		if (cmd[i]=='R') ++x;
		if (cmd[i]=='U') ++y;
		if (cmd[i]=='D') --y;
		
		if ((b[x-1][y] + b[x+1][y] + b[x][y-1] + b[x][y+1] >= 2) || (b[x][y])){
			puts("BUG");
			return 0;
		}
		b[x][y] = 1;
	}
	puts("OK");
	return 0;
}
