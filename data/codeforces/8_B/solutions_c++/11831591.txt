#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(){
	string s;
	cin >> s;
	int x[105],y[105];
	x[0] = 0;
	y[0] = 0;
	int n = 1;
	for(int i=0;i<s.length();i++){
		x[n] = x[n-1];
		y[n] = y[n-1];
		if(s[i]=='L') x[n]--;
		if(s[i]=='R') x[n]++;
		if(s[i]=='U') y[n]++;
		if(s[i]=='D') y[n]--;
		n++;
	}
	for(int i=0;i<n;i++){
		for(int j=i+2;j<n;j++){
			if(abs(x[i]-x[j])+abs(y[i]-y[j])<=1){
				cout << "BUG";
				return 0;
			}
		}
	}
	cout << "OK";

	return 0;
}
