#include<stdio.h>
int main(){
    int vis[255][255];
    char str[11111];
    while(scanf("%s",&str)!=EOF){
        int x=100,y=100;
        vis[x][y]=1;
        int flag=0;
        for(int i=0;str[i];i++){
            if(str[i]=='U')x++;
            if(str[i]=='D')x--;
            if(str[i]=='L')y++;
            if(str[i]=='R')y--;
            if(vis[x][y]||(vis[x-1][y]+vis[x+1][y]+vis[x][y-1]+vis[x][y+1]!=1))flag=1;
            vis[x][y]=1;
        }
        if(flag)printf("BUG\n");
        else printf("OK\n");
    }
    return 0;
}
