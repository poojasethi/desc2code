#include <bits/stdc++.h>
using namespace std;

bool m[1000][1000];

int main () {
    string s; cin >> s;
    int n = s.size();
    int x=500,y=500,c; char v;
    m[x][y] = 1;
    for (int i=0; i<n; i++) {

        v = s[i];
        if (v == 'U') y++;
        else if (v == 'D') y--;
        else if (v == 'L') x++;
        else x--;

        c = m[x][y]+m[x+1][y]+m[x-1][y]+m[x][y+1]+m[x][y-1];

        if (c>=2) {
            cout << "BUG" << endl; return 0;
        }
        m[x][y] = 1;
    }
    cout << "OK" << endl;
}
