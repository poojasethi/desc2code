#include<iostream>
#include<vector>
#include<stdio.h>
using namespace std;
int f[205][205]={0};
int main()
{//freopen("i.txt","r",stdin);
string s; cin>>s;
f[100][100]=1;
int x=100,y=100;
for (int i=0;i<s.size();i++)
{if (s[i]=='L')  x--;
 if (s[i]=='R')  x++;
 if (s[i]=='D')  y--;
 if (s[i]=='U')  y++;
 int k=f[x+1][y]+f[x-1][y]+f[x][y+1]+f[x][y-1];
 if (k>=2||f[x][y]==1) {cout<<"BUG"; return 0;}
    else f[x][y]=1;
}
cout<<"OK";
return 0;
}
