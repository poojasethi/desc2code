#include<cstdio>
char s[1<<10];
bool vis[210][201];
int main()
{
    bool f=1;
    int i,x=101,y=101;
    gets(s);
    for(i=0;s[i];i++)
    {
        vis[x][y]=1;
        if(s[i]=='L') x--;
        else if(s[i]=='R') x++;
        else if(s[i]=='U') y++;
        else y--;
        if(vis[x][y]+vis[x-1][y]+vis[x][y-1]+vis[x+1][y]+vis[x][y+1]>1)
        {
            f=0;break;
        }
    }
    puts(f?"OK":"BUG");
    return 0;
}
