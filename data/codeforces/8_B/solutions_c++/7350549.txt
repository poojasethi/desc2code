#include <cstdio>
using namespace std;
char s[102];
bool Map[204][204];
int px[]={0,1,0,-1},py[]={1,0,-1,0};
int main()
{
	scanf("%s",s);
	bool ff=true;
	Map[102][102]=1;
	for (int i=0,x=102,y=102,cnt;s[i] && ff;i++)
	{
		switch(s[i])
		{
			case 'U':
				x++;
				break;
			case 'L':
				y--;
				break;
			case 'R':
				y++;
				break;
			case 'D':
				x--;
		}
		if (Map[x][y]) ff=false;
		cnt=0;
		for (int j=0;j<4;j++)
			if (Map[x+px[j]][y+py[j]]) cnt++;
		if (cnt>=2) ff=false;
		Map[x][y]=true;
	}
	if (ff) puts("OK");
	else puts("BUG");
	return 0;
}
