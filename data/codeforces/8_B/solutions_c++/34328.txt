#include <iostream>
#include <string>
#include <algorithm>
#include <complex>
#include <vector>
using namespace std;

typedef complex<double> point;

int main()
{
	string s;
	getline(cin, s);

	vector<point> log;

	point p(0,0);
	for(int i=0; i<s.size(); ++i) {
		point q = p;
		if( s[i]=='L' )
			q += -1;
		else if( s[i]=='R' )
			q += 1;
		else if( s[i]=='D' )
			q += point(0, -1);
		else if( s[i]=='U' )
			q += point(0, 1);

		for(int j=0; j<log.size(); ++j)
			if( abs(q-log[j])<=1 )
				{cout << "BUG" << endl; return 0;}

		log.push_back(p);
		p = q;
	}
	cout << "OK" << endl;
}
