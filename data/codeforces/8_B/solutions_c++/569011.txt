#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
char s[111];
bool visit[250][250],flag=0;
int main(void)
{
    scanf("%s",s);
    int x = 130, y = 130;
    for ( int i = 0;s[i];i++){
        visit[x][y] = 1;
        if ( s[i] == 'L')x--;
        else if ( s[i] == 'R')x++;
        else if ( s[i] == 'U')y++;
        else if ( s[i] == 'D')y--;
        if(visit[x][y]+visit[x-1][y]+visit[x][y-1]+visit[x][y+1]+visit[x+1][y] > 1){flag=1;break;}
    }
    if ( flag )
       puts("BUG");
    else
        puts("OK");
    return 0;
}
