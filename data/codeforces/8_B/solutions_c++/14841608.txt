#include<stdio.h>
#include<cmath>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int a[250][250]={0};
int main()
{
	char s[110],c;
	int i,j,k,m,n,l,x,y,x1[4]={0,0,1,-1},y1[4]={1,-1,0,0};
	bool t;
	scanf("%s",s);
	l=strlen(s);
	x=100; y=100; t=true;
	for (i=0;i<=l-1;i++)
	{
		m=x; n=y;
		if (s[i]=='U') x=x+1;
		if (s[i]=='D') x=x-1;
		if (s[i]=='L') y=y-1;
		if (s[i]=='R') y=y+1;
		if (a[x][y]==1) t=false;
		for (j=0;j<=3;j++) a[m+x1[j]][n+y1[j]]=1;
		a[m][n]=1;
	}
	if (t) printf("OK\n"); else printf("BUG\n");
}
