#include<stdio.h>
#include<string.h>
bool v[500][500];
char c[100];
int main()
{
	int x,y,i,t;
  	while(~scanf("%s",c))
  	{
  		t=strlen(c);
  		x=y=200;
	  	for(i=0;i<t;i++)
	  	{
    		v[x][y]=1;
    		switch(c[i])
			{
      			case 'L': --x; break;
      			case 'R': ++x; break;
      			case 'U': ++y; break;
      			case 'D': --y; break;
    		}
    		if(v[x][y]+v[x-1][y]+v[x][y-1]+v[x][y+1]+v[x+1][y]>1) break;
		}
  		if(i<t) puts("BUG");
  		else puts("OK");
  	}
}
