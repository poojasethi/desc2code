#include <iostream>
#include <algorithm>
#include <math.h>
#include <stdio.h>

using namespace std;

string s;
int stx,sty,i;
int f[2000][2000];

main()
 {
  cin>>s;
  stx=100;
  sty=100;
  f[stx][sty]=1;
  for (i=0;i<s.size();i++)
   {
    if (s[i] == 'U') stx--;
    if (s[i] == 'D') stx++;
    if (s[i] == 'L') sty--;
    if (s[i] == 'R') sty++;
    f[stx][sty]++;
    if (f[stx-1][sty]+f[stx+1][sty]+f[stx][sty-1]+f[stx][sty+1]+f[stx][sty] >= 3)
     { cout<<"BUG"<<endl; return 0; }
   }
  cout<<"OK"<<endl;
 }
