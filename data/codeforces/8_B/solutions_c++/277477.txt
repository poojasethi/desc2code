#include <stdio.h>
#define SZ 501


int main()
{
	int x = SZ / 2, y = SZ / 2,
		mat[SZ][SZ] = {0};
	char buf[SZ];
	char* p = buf;
	char* res = "OK";
	
	scanf("%s", buf);
	while(*p)
	{
		char ch = *p++;
		mat[x][y] = 1;
		
		switch(ch)
		{
			case 'L':
				--x;
				break;
			case 'R':
				++x;
				break;
			case 'U':
				++y;
				break;
			case 'D':
				--y;
				break;
		}
		
		if(mat[x][y] == 1 || mat[x + 1][y] + mat[x - 1][y] + mat[x][y - 1] + mat[x][y + 1] > 1)
		{
			res = "BUG";
			break;
		}
	}
	
	printf("%s\n", res);
	return 0;
}
