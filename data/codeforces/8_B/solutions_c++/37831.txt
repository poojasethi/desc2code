#include<iostream>
#include<algorithm>
#include<string>

using namespace std;

int main(){
    int a[205][205]={0};
    int x=101,y=101;
    string s;
    cin>>s;
    a[x][y]=1;
    for(int i=0;i<s.size();i++){
        if(s[i]=='L')
            x--;
        else if(s[i]=='R')
            x++;
        else if(s[i]=='U')
            y--;
        else 
            y++;
        
        if(a[x][y]==1 || ( a[x-1][y]+a[x+1][y]+a[x][y-1]+a[x][y+1]>=2)){
            cout<<"BUG"<<endl;
            return 0;
        }
        a[x][y]=1;  
    }
    cout<<"OK"<<endl;
    return 0;
}
