#include <iostream>
using namespace std;
int a[400][400];
int i=200,j=200,k;
string s,res="OK";
int main()
{
    cin>>s;
    a[i][j]=1;
    for (k=0; k<s.length(); k++) {
        if(s[k]=='L') j--;
        if(s[k]=='R') j++;
        if(s[k]=='U') i--;
        if(s[k]=='D') i++;
        if(a[i][j]){res="BUG"; break;}
        if(a[i-1][j]+a[i+1][j]+a[i][j-1]+a[i][j+1]>=2) {res="BUG"; break;}
        a[i][j]=1;
    }
    cout << res;
    return 0;
}
