#include <iostream>
#include <string>
using namespace std;
long x=102,y=102,i,f[210][210]={};
string s;
int main() {
	cin >> s;
	for (i=0;i<s.length();i++) {
		f[x][y]=1;
		switch (s[i]) {
			case 'L':y--;break;
			case 'R':y++;break;
			case 'U':x--;break;
			case 'D':x++;break;
		}
		if (f[x-1][y]+f[x+1][y]+f[x][y-1]+f[x][y+1]+f[x][y]>1) break;
	}
	cout << ((i<s.length())?"BUG":"OK") << endl;
	return 0;
}
