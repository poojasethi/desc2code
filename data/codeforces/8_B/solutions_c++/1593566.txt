#include <stdio.h>
#include <stdlib.h>
char map[202][202]={0};
int check(int i, int j)
{
	int t=0;
	if(map[i-1][j]) t++;
	if(map[i+1][j]) t++;
	if(map[i][j-1]) t++;
	if(map[i][j+1]) t++;
	return t;
}
int main()
{
	char move;
	int i=101, j=101, f=1;
	map[101][101]=1;
	while(1) {
		scanf("%c", &move);
		if(move=='\n') break;
		if(move=='R')  j++;
		else if(move=='L') j--;
		else if(move=='U') i++;
		else i--;
		map[i][j]=1;
		if(check(i, j)>=2) {f=0; break;}
	}
	move=check(101,101); if(move>1) f=0;
	if(f) printf("OK"); else printf("BUG");
	printf("\n");
}
