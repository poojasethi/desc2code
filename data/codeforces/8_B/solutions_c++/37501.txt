#include<cstdio>
#include<cstring>
bool vis[222][222];
char cmd[77];
bool fail(int x,int y){
  int cnt=0;
  cnt+=vis[x][y];
  cnt+=vis[x-1][y];
  cnt+=vis[x][y-1];
  cnt+=vis[x][y+1];
  cnt+=vis[x+1][y];
  return cnt>1;
}
int main(){
  memset(vis,0,sizeof(vis));
  scanf("%s",cmd);
  char*p;
  int x=111,y=111;
  for(p=cmd;*p;++p){
    vis[x][y]=1;
    switch(*p){
      case'L':--x;break;
      case'R':++x;break;
      case'U':++y;break;
      case'D':--y;break;
    }
    if(fail(x,y))break;
  }
  if(*p)puts("BUG");
  else puts("OK");
  return 0;
}
