#include<stdio.h>
#include<string.h>

char z[105];
int z1[500][500];


int main()
{
	int a,b,c,i,j,k,n,m,x,y;

	scanf("%s",z);
	x=200;
	y=200;

	for(i=0;i<strlen(z);i++)
	{
		z1[x][y]=1;

		switch(z[i])
		{
			case 'R':x++;break;
			case 'L':x--;break;
			case 'U':y++;break;
			case 'D':y--;break;
		}



		if(z1[x-1][y]+z1[x][y-1]+z1[x][y]+z1[x][y+1]+z1[x+1][y]>1)
		{
			printf("BUG\n");
			return 0;
		}

	}

	printf("OK\n");
	return 0;

	return 0;
}
