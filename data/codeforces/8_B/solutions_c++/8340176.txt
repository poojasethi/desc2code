#include <iostream>

using namespace std;

int seen[205][205] = { };

int main()
{
	string x;
	cin >> x;
	seen[100][100];
	int r = 100, c = 100;
	for(int i = 0; i < x.size(); i++)
	{
		seen[r][c] = true;

		switch(x[i])
		{
			case 'U': r++; break;
			case 'D': r--; break;
			case 'L': c--; break;
			case 'R': c++; break;
		}

		if (seen[r][c]
		|| (seen[r - 1][c] && x[i] != 'U')
		|| (seen[r + 1][c] && x[i] != 'D')
		|| (seen[r][c - 1] && x[i] != 'R')
		|| (seen[r][c + 1] && x[i] != 'L'))
		{
			cout << "BUG";
			return 0;
		}
	}

	cout << "OK";
}
