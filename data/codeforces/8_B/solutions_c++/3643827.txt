#include<stdio.h>
int a[2005][2005];
int dif(int a,int b){
if(a>=b)return a-b;
return b-a;
}
int main()
{

int i;
int j;
for(i=0;i<2000;i++)
for(j=0;j<2000;j++)
a[i][j] = 0;
int valid=1;
char s[1005];
scanf("%s",&s);
int x,y;
x=y=1000;
a[x][y] = 1;
for(i=0;s[i];i++)
{
if(s[i] == 'R')x++;
else if(s[i] =='L') x--;
else if(s[i] == 'U') y--;
else y++;
if(a[x][y] + a[x-1][y] + a[x+1][y] + a[x][y-1] + a[x][y+1] > 1){
valid =0; break;
}


a[x][y] = 1;
}
if(valid)printf("OK\n");
else printf("BUG\n");

return 0;
}
