#include<cstdio>
#include<cmath>
#include<algorithm>
#include<cstring>
using namespace std;
int map[205][205];
int main(){
	char t[105];
	while(gets(t)!=NULL){
		memset(map,0,sizeof(map));
		int x=102,y=102;
		int len=strlen(t);
		int i,j,flag=1;
		for(i=0;i<len;i++)
		{	map[x][y]=1;
			if(t[i]=='L') y--;
			else if(t[i]=='R') y++;
			else if(t[i]=='U') x--;
			else if(t[i]=='D') x++;
			if(map[x][y]+map[x-1][y]+map[x+1][y]+map[x][y-1]+map[x][y+1]>1) {flag=0; break;}
		}
		if(flag) printf("OK\n");
		else printf("BUG\n");
	}
return 0;}
