#include<cstdio>
char s[105];
bool g[206][206];
int main() {
    gets(s);
    int x=102,y=102;
    int i;
    for(i=0;s[i];i++) {
        g[x][y]=1;
        if(s[i]=='L')y--;
        else if(s[i]=='R')y++;
        else if(s[i]=='U')x--;
        else x++;
        if(g[x+1][y]+g[x-1][y]+g[x][y+1]+g[x][y-1]+g[x][y]>1)break;
    }
    if(s[i])puts("BUG");
    else puts("OK");
    return 0;
}
