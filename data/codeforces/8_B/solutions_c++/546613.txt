#include <iostream>
#include <string>
using namespace std;
string str;
int N;
bool solve(int i,int j)
{
	int x=0,y=0;
	//cout << i << " " << j << endl;
	for(int k=i;k<=j;k++)
	{
		if(str[k]=='L') y--;
		if(str[k]=='R') y++;
		if(str[k]=='U') x--;
		if(str[k]=='D') x++;
	}
	if(j-i==1) if(!x && !y) return 0;
	if((x==1 || x==-1)&&!y ||(y==1 || y==-1)&&!x) return 0;
	return 1;
}
int main()
{
	cin >> str;
	N=str.length();
	for(int i=0;i<N;i++)
		for(int j=i+1;j<N;j++)
			if(!solve(i,j))
			{
				cout << "BUG" <<endl;
				return 0;
			}
	cout << "OK" << endl;
	return 0;
}
