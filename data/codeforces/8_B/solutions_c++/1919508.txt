#include <iostream>
#include<string>
using namespace std;


int main()
{
   string s;cin>>s;int k=100,l=100,a[202][202];;a[k][l]=1;
   if(s.size()<3) {cout<<"OK";return 0;}

   
     
   for(int i=0;i<s.size();i++)
   {
       if(s[i]=='L') k--;
       else if(s[i]=='R')k++;
       else if(s[i]=='U') l++;
       else l--;
       if(a[k][l]==1) {cout<<"BUG";return 0;}
       a[k][l]=1;
       if(a[k-1][l]+a[k+1][l]+a[k][l+1]+a[k][l-1]==2)
       {
           cout<<"BUG";return 0;
       }
   }
   cout<<"OK";
         return 0;
}
