#include<cstdio>
bool vis[222][222];
char cmd[77];
int main(){
  scanf("%s",cmd);
  char*p;
  int x=111,y=111;
  for(p=cmd;*p;++p){
    vis[x][y]=1;
    switch(*p){
      case'L':--x;break;
      case'R':++x;break;
      case'U':++y;break;
      case'D':--y;break;
    }
    if(vis[x][y]+vis[x-1][y]+vis[x][y-1]+vis[x][y+1]+vis[x+1][y]>1)break;
  }
  if(*p)puts("BUG");
  else puts("OK");
  return 0;
}
