#include<bits/stdc++.h>
#define M 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
int a[210][210];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int x=105,y=105,i;
	string s;
	cin>>s;
	for(i=0;i<s.size();i++)
	{
		a[x][y]=1;
		if(s[i]=='L')
			x--;
		if(s[i]=='R')
			x++;
		if(s[i]=='U')
			y++;
		if(s[i]=='D')
			y--;
		if(a[x][y] + a[x-1][y] + a[x+1][y] + a[x][y-1] + a[x][y+1] > 1)
			break;
	}
	if(i<s.size())
		cout<<"BUG\n";
	else
		cout<<"OK\n";
	return 0;
}
