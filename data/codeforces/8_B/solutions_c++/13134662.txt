#include<iostream>
#include<string>
using namespace std;

string a;

int b[400][400];

int main()
{
    int l , x, y;
    x = y = 200;
    cin >> a;
    b[x][y] = 1;
    for(int i = 0 ; i < a.size() ; i++){
        if( a[i] == 'L' ) x--;
        if( a[i] == 'R' ) x++;
        if( a[i] == 'U' ) y++;
        if( a[i] == 'D' ) y--;
        l = b[x][y] + b[x-1][y] + b[x+1][y] + b[x][y-1] + b[x][y+1];
        if( l > 1 ){
            cout << "BUG";
            return 0;
        }
        b[x][y] = 1;
    }
    cout << "OK\n";
}
