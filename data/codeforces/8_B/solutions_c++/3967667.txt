using namespace std;
string s;
int b[1005][1005];
int main(){
	cin>>s;
	int x,y;
	x=y=500;
	b[500][500]=1;
	for(int i=0; i<s.length(); ++i){
		if (s[i]=='L') --y;
		else if (s[i]=='R') ++y;
		else if (s[i]=='U') --x;
		else ++x;
		if (b[x][y]+b[x-1][y]+b[x+1][y]+b[x][y-1]+b[x][y+1]>1){
			cout<<"BUG\n";
			return 0;
		}
		b[x][y]=1;
	}
	cout<<"OK\n";
	return 0;
}
