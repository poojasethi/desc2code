#include <stdio.h>

bool land[300][300];
char ins[105];

int main(){	
	scanf("%s",ins);
	int x = 150, y=150;
	
	char *p = ins;
	while ( *p ){
		land[x][y]++;
		
		switch ( *p ){
			case 'U': y++; break;
			case 'D': y--; break;
			case 'L': x--; break;
			case 'R': x++; break;	
		}
		
		if ( land[x][y] + land[x-1][y] + land[x][y-1] + land[x+1][y] + land[x][y+1] > 1 ) break;
		p++;
	}
	
	if ( *p ) puts("BUG");
	else puts("OK");

	return 0;
}
