k, a, b = raw_input().split(' ')
k = long(k)
a = long(a)
b = long(b)
if a == 0 and b != 0 :
    ans = b / k + 1
elif a != 0 and b == 0 :
    ans = abs(a) / k + 1
elif a <= 0 and b >= 0 :
    ans = abs(a) / k + b / k + 1
elif b < 0 :
    ans = abs(a) / k - (abs(b) - 1) / k
else :
    ans = abs(b) / k - (abs(a) - 1) / k
print ans
