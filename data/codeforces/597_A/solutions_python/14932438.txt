from math import *
from Queue import *
from sys import *
from datetime import *




def nd(a,k):
    if a >= 0:
        return ((a+k-1)/k)*k
    else:
        return -pd(-a,k)

def pd(a,k):
    if a >= 0:
        return (a/k)*k
    else:
        return -nd(-a,k)

def my(a,b,k):
    a = ((a+k-1)/k)*k
    b = (b/k)*k
    return (b-a)/k + 1

k, a, b = map(int, raw_input().split())
print(my(nd(a,k), pd(b,k), k))
