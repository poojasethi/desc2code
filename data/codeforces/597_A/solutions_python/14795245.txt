# http://codeforces.com/problemset/problem/597/A


def calculate(numbers):
    k, a, b = map(int, numbers.split())
    left = a + (k - a % k if a % k else 0)
    right = b - b % k
    return (right - left) / k + 1


def main():
    print(calculate(raw_input()))


if __name__ == '__main__':
    main()
