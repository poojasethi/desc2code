#include<iostream>
#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
#define X first
#define Y second
#define rep(i,s,t) for (int i=s;i<=t;++i)
typedef long long LL;
typedef pair<int, int> PII;
typedef map<int, int> mi;
typedef mi::iterator mii;
int n,m;
int a[1111111],b[1111111],ok[1111111];PII q[1111111];
mi M;LL T[1111111],ans;
void add(int i, int d){ for (;i<=n;i+=i&(-i)) T[i]+=d; }
LL  query(int i){LL s=0;
    for (;i;i-=i&(-i)) s+=T[i];return s;
}
int main(){
    scanf("%d%d",&n,&m);
    rep(i,1,n) scanf("%d", &a[i]);
    rep(i,1,m) scanf("%d", &b[i]);
    int p=1;
    rep(i,1,n) if (a[i]==b[p]&&p<=n) p+=ok[i]=1;
    rep(i,1,n) q[i]=make_pair(a[i],i);
    sort(q+1,q+1+n);
    rep(i,1,n) add(i,1);
    M[0]=M[n+1]=1;
    rep(i,1,n){
        add(q[i].Y,-1);
        M[q[i].Y]=1;
        if (ok[q[i].Y]==0){
            mii c=M.find(q[i].Y);
            mii d=c--; d++;
            ans += query(d->X-1)-query(c->X)+1;
            c++;M.erase(c);
        }
    }
    cout<<ans<<endl;
}
