#include <bits/stdc++.h>

#pragma comment(linker, "/STACK:102400000,102400000")
using namespace std;

#define LL long long
#define pii pair<int,int>
#define MP make_pair
#define ls i << 1
#define rs ls | 1
#define md (ll + rr >> 1)
#define lson ll, md, ls
#define rson md + 1, rr, rs
#define Pi acos(-1.0)
#define mod 1000000007
#define eps 1e-10
#define inf 0x3f3f3f3f
#define N 2000010
#define M 2000020

int n, k, id[N], vis[N], sum[N];
set<int> s;
set<int>::iterator it;
void add(int x, int v){
	for(; x <= n; x += x & -x) sum[x] += v;
}
int query(int x){
	int ret = 0;
	for(; x > 0; x -= x & -x) ret += sum[x];
	return ret;
}
int main(){
	scanf("%d%d", &n, &k);
	for(int i = 1; i <= n; ++i){
		int a;
		scanf("%d", &a);
		id[a] = i;
		add(i, 1);
	}
	for(int i = 1; i <= k; ++i){
		int a;
		scanf("%d", &a);
		vis[a] = 1;
	}
	LL ans = 0;
	s.insert(0), s.insert(n + 1);
	for(int i = 1; i <= n; ++i){
		if(vis[i]){
			s.insert(id[i]);
			continue;
		}
		it = s.upper_bound(id[i]);
		int r = (*it) - 1;
		int l = *(--it);
		ans += query(r) - query(l);
		add(id[i], -1);
	}
	cout << ans << endl;
	return 0;
}
