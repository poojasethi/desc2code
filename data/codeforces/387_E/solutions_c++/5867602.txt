#include <set>
#include <cstdio>
#include <algorithm>
#define maxn 1000005
using namespace std;

int N,K,p[maxn],s[maxn];
bool b[maxn];
set<int>F;

void add(int x,int d) {for (;x<=N; x+=x&-x) s[x]+=d;}
int sum(int x) {int d=0; for (;x; x-=x&-x) d+=s[x]; return d;}

void init()
{
	scanf("%d%d",&N,&K),F.insert(0),F.insert(N+1);
	for (int i=1,x; i<=N; i++) scanf("%d",&x),p[x]=i,add(i,1);
	for (int i=1,x; i<=K; i++) scanf("%d",&x),b[x]=1;
}

void doit()
{
	long long ans=0;
	for (int i=1; i<=N; i++) if (b[i]) F.insert(p[i]);
	else
	{
		set<int>::iterator x=F.upper_bound(p[i]),y=x--;
		ans+=sum(*y-1)-sum(*x),add(p[i],-1);
	}
	printf("%I64d\n",ans);
}

int main()
{
	init();
	doit();
}
