#include<iostream>
#include<vector>
#include<set>
using namespace std;
#define int long long
const int maxN=1200000;
int n;
int k;
int t[maxN];
int a[maxN];
bool m[maxN];
set<int> s;
void update(int c , int r)
{
	for(;r<=1100000;r+=(r& -r))
		t[r]+=c;
}
int sum(int r)
{
	int u=0;
	for(;r>0;r-=(r&-r))
		u+=t[r];
	return u;
}
main()
{
	ios::sync_with_stdio ( false );
	cin>>n>>k;
	for(int i=1;i<=n;i++)
	{
		int u;
		cin>>u;
		a[u]=i;
	}
	for(int i=0;i<k;i++)
	{
		int b;
		cin>>b;
		m[b]=true;
	}
	int w=0;
	s.insert ( 0 );
	s.insert ( n+1 );
	for(int i=1;i<=n;i++)
	{
		if(m[i])
			s.insert(a[i]);
		else
		{
			int p,q;
			set<int>::iterator x=s.lower_bound(a[i]),y=x;
			p=*(--x);
			q=*y;
			int z=sum(q),e=0;
			if(p>0)
				e=sum(p);
			//  cout<<q<<" "<<p<<" "<<z<<" "<<e<<endl;
			w+=q-p-z+e-1;
			update(1,a[i]);
		}
	}
	cout<<w<<endl;
	// system("pause");
}
