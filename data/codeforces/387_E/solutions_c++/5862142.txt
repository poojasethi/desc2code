# include <stdio.h>
# include <algorithm>
# include <set>

# define MAXN 1000009

using namespace std;

typedef long long lld;

int n, k;
pair<int,int> A[MAXN];
bool need[MAXN];
set<int> st;
int T[MAXN];
lld ans;

int main(){
	
	scanf("%d %d",&n,&k);
	
	for(int i=0; i<n; i++){
		scanf("%d",&A[i].first);
		A[i].second = i;
	}
	
	sort(A, A+n);
	
	for(int i=0; i<k; i++){
		int a;
		scanf("%d",&a);
		need[a] = 1;
	}
	
	st.insert(-1);
	st.insert(n);
	
	for(int i=0; i<n; i++){
		pair<int,int> cur = A[i];
		
		st.insert(cur.second);
		
		if(!need[cur.first]){
			__typeof((st).begin()) it = st.find(cur.second);
			
			it--;
			int l = *it;
			it++, it++;
			int r = *it;
			
			ans += r - l - 1;
			
			st.erase(cur.second);
			
			l += 2;
			r += 2;
			
			for(int j=r; j>0; j-=j&-j)	ans -= T[j];
			for(int j=l; j>0; j-=j&-j)	ans += T[j];
			for(int j=cur.second+2; j<=n+2; j+=j&-j)	T[j]++;
		}
	}
	
	printf("%lld",ans);
}
