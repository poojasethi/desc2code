#include <cstdio>
#include <set>
#include <algorithm>
using namespace std;

int n, k;
int p[1000001], r[1000001], c[1000001], s[1000001], z[1000001], w[1000001];
long long A;
set<int> S;

int main() {
	scanf("%d%d", &n, &k);
	for (int i = 1; i <= n; ++i) {
		scanf("%d", p + i);
		r[p[i]] = i;
	}
	for (int i = 1; i <= k; ++i) {
		int b;
		scanf("%d", &b);
		c[b] = 1;
	}
	S.insert(0);
	S.insert(n + 1);
	for (int i = 1; i <= n; ++i) {
		if (!c[i]) {
			int left = *--S.lower_bound(r[i]);
			int right = *S.lower_bound(r[i]);
			A += right - left - 1;
			for (int i = right - 1; i; i ^= i & -i)
				A -= s[i];
			for (int i = left; i; i ^= i & -i)
				A += s[i];
			for (int j = r[i]; j <= n; j += j & -j)
				++s[j];
		} else {
			S.insert(r[i]);
		}
	}
	printf("%I64d\n", A);
}
