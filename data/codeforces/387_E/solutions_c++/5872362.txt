#include <iostream>
#include <cstdio>
#include <string>
#include <cstring>
#include <ctime>
#include <cmath>
#include <cstdlib>
#include <vector>
#include <stack>
#include <algorithm>
#include <set>
#define INF 15000
#define N 1000009
using namespace std;
int n,k;
int x[N];
int place[N];
bool save[N];
set<int> s;
void ad(int i,int o)
{
    for(;i<=n;i+=(i&-i))
        x[i]+=o;
    return;
}
int sum(int i)
{
    int ans=0;
    for(;i>0;i-=(i&-i))
        ans+=x[i];
    return ans;
}
int main()
{
    scanf("%d%d",&n,&k);
    for(int i=1,ol;i<=n;i++){
        scanf("%d",&ol);
        place[ol]=i;
        ad(i,1);
    }
    for(int i=0,ol;i<k;i++){
        scanf("%d",&ol);
        save[ol]=true;
    }
    s.insert(0);
    s.insert(n+1);
    long long ans=0;
    for(int i=1;i<=n;i++)
    {
        if(save[i])s.insert(place[i]);
        else{
            int l,r;
            set<int>::iterator j=s.upper_bound(place[i]);
            r=*j;
            l=*(--j);
            ans+=sum(r-1)-sum(l);
            ad(place[i],-1);
        }
    }
    printf("%I64d\n",ans);
    return 0;
}
