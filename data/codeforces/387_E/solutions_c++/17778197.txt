#include<bits/stdc++.h>

using namespace std;

#define sgmn int v, int tl, int tr
#define tm   (tl + tr >> 1)
#define sol  v*2, tl, tm
#define sag  v*2+1, tm+1, tr
#define mp make_pair
#define N    1000006

pair < int , int > S[N];

int s,A[N],L[N],R[N],T[N],F[N],n,k,i,x,t;

long long asd;

bool H[N];

int qry(int x){
	t = 0;
	for(; x ; x -= x & -x)
		t += F[x];
	return t;
}

void up(int x){
	for(; x <= n ; x += x & -x)
		F[x]++;
}

int main(){

	cin >> n >> k;

	for(i=1;i<=n;i++){
		scanf("%d",&A[i]);
		T[ A[i] ] = i;
	}

	for(i=1;i<=k;i++){
		scanf("%d",&x);
		H[x] = 1;
	}

	for(i=1;i<=n;i++){

		if(H[ A[i] ]){
			while(s > 0 && A[i] < S[s].first) s--;
			S[++s] = mp(A[i] , i);
		}

		else
			L[i] = S[ (lower_bound(S+1 , S+s+1 , mp(A[i] , 0)) - S - 1) ].second;
	}

	S[0].second = n+1;

	for(s=0,i=n; i ;i--){

		if(H[ A[i] ]){
			while(s > 0 && A[i] < S[s].first) s--;
			S[++s] = mp(A[i] , i);
		}

		else
			R[i] = S[ (lower_bound(S+1 , S+s+1 , mp(A[i] , 0)) - S - 1) ].second - 1;
	}

	for(i=1;i<=n;i++){
		
		if(H[i])
			continue;

		x = T[i];

		asd += R[x] - L[x] - qry(R[x]) + qry(L[x]);

		up(x);
	}

	cout << asd;
	return 0;
}
