#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <string.h>
#include <algorithm>
#include <vector>
#include <string>
#include <stack>
#include <queue>
#include <set>
#include <map>
#include <iostream>

using namespace std;

int n,m, a[1100000], b[1100000], pos[1100000], us[1100000], t[1100000];

int sum (int r)
{
    int result = 0;
    for (; r >= 0; r = (r & (r+1)) - 1)
        result += t[r];
    return result;
}

void inc (int i, int delta)
{
    for (; i < n+10; i = (i | (i+1)))
        t[i] += delta;
}

int sum (int l, int r)
{
    return sum (r) - sum (l-1);
}

long long ans;

int main () {
	cin >> n >> m;
	for(int i=0;i<n;i++) { cin >> a[i]; pos[a[i]] = i; }
	for(int i=0;i<m;i++) { cin >> b[i]; us[b[i]] = 1;  }
	
	set <int> is;
	for(int i=1;i<=n;i++){
		if(us[i])
			is.insert(pos[i]);
		else {
			set<int>::iterator it = is.upper_bound(pos[i]);
			int l=-1, r=n;
			if(it!=is.end()) r = *it;
			if(it!=is.begin()) {
				--it;
				l = *it;
			}
			ans += r-l-1-sum(l, r);
			inc(pos[i], 1);
		}
	}
	cout << ans << endl;
	return 0;
}
