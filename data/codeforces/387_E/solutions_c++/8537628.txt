#include<bits/stdc++.h>
#include<functional>
#include<string.h>
using namespace std;
#define ll   long long  int
#define mp make_pair
#define s second
#define f first
#define mod 1000000007
#define pi pair<ll,pair<ll,ll> >
#define pb push_back
#define INF 1000000
#define yup ios_base::sync_with_stdio(0)
ll a[INF+10],b[INF+10];
ll bt[INF+10];
ll ind[INF+10];
bool need[INF+10];
ll n;
void update(ll ind,ll val){
	while(ind<=n&&ind>0){
	bt[ind]+=val;
	ind+=(ind&-ind);	
	}
}
ll query(ll ind){
ll sum=0;
while(ind>0){
	sum+=bt[ind];
	ind-=(ind&-ind);	
}	
return sum;	
}
int main(){
	yup;
	ll k;
	cin>>n>>k;
	set<ll>s;
	for(int i=1;i<=n;i++){
	 cin>>a[i];
	 ind[a[i]]=i; 
    }
	for(int i=1;i<=k;i++){
	 cin>>b[i];
     need[b[i]]=1;
    }
    s.insert(0);
    s.insert(n+1);
    ll ans=0;
    for(int i=1;i<=n;i++){
	if(need[i]) s.insert(ind[i]);
	else{
		set<ll>::iterator it=s.upper_bound(ind[i]);
	    ll r=*it;
	    r--;
	    --it;
	    ll l=*it;
	    l++;
	    ans=ans+r-l+1-query(r)+query(l-1); 
	    update(ind[i],1);
	}
	}
	cout<<ans;
}
