#include <cstdio>
#include <algorithm>
#include <set>
#define LL long long
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

const int MAXN=1000010;
set<int> st;
bool boo[MAXN];
pair<int,int> a[MAXN];
int n,m,x,l,r;
int t[MAXN];
LL ans;

int main() {
	scanf("%d%d",&m,&n);
	REP(i,m) {
		scanf("%d",&x);
		a[i].first=x;
		a[i].second=i;
	}
	sort(a+1,a+m+1);
	REP(i,n) {
		scanf("%d",&x);
		boo[x]=true;
	}
	st.insert(0);
	st.insert(m+1);
	REP(i,m)
		if (boo[a[i].first])
			st.insert(a[i].second);
		else {
			st.insert(a[i].second);
			__typeof(st.begin()) it=st.find(a[i].second);
			++it; r=(*it)-1;
			--it; --it; l=(*it)+1;
			ans+=(LL)(r-l+1);
			st.erase(a[i].second);
			for (int j=r;j;j-=(j&(-j))) ans-=(LL)t[j];
			for (int j=l-1;j;j-=(j&(-j))) ans+=(LL)t[j];
			for (int j=a[i].second;j<=m;j+=(j&(-j))) ++t[j];
		}
	printf("%I64d\n",ans);
	return 0;
}
