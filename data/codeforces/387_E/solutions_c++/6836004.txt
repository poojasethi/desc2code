#include <iostream>
#include <cstdio>
#include <set>
using namespace std;
int n,m,st,en,in,x,pos[1000010],bit[1000010];
bool vis[1000010];
set<int>s;
set<int>::iterator it;
void Add(int ind,int val)
{
    while(ind<=n)
    {
        bit[ind]+=val;
        ind+=ind&(-ind);
    }
}
int Query(int ind)
{
    int ret=0;
    while(ind)
    {
        ret+=bit[ind];
        ind-=ind&(-ind);
    }
    return ret;
}
int main()
{
    //freopen("r.txt","r",stdin);
    scanf("%d %d",&n,&m);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&in);
        pos[in]=i;
    }
    for(int i=0;i<m;i++)
    {
        scanf("%d",&x);
        vis[x]=1;
    }
    s.insert(0) , s.insert(n+1);
    long long int res=0;
    for(int i=1;i<=n;i++)
    {
        if(vis[i]==0)
        {
            it=s.lower_bound(pos[i]) , it-- , st=(*it);
            it=s.upper_bound(pos[i])  , en=(*it);
            res+=en-st-1- ( Query(en-1)-Query(st) );
            Add(pos[i],1);
        }
        else
        {
            s.insert(pos[i]);
        }
    }
    printf("%I64d",res);

    return 0;
}
