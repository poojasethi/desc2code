#include<cstdio>
#include<algorithm>
#include<set>
using namespace std;
int a[1000001],n,k,p[1000001],b[1000001],m[1000001],pos[1000001],ct,f[1000001];
long long ans;
set <int> S,S2;
set <int>::iterator it1,it2;
void upd(int x,int val){
    for(int i=x;i<=n;i+=(i&(-i)))
        f[i] += val;
}
int fi(int x){
    int sum =0;
    for(int i=x;i>0;i-=(i&(-i)))
        sum += f[i];
    return sum;
}
int main(){
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;i++) scanf("%d",&p[i]), pos[p[i]] = i;
    for(int i=1;i<=k;i++) scanf("%d",&b[i]), m[b[i]] = 1;
    
    for(int i=1;i<=n;i++){
        if(m[i] == 0){
            int a,b,c,d;
            it1 = S.lower_bound(pos[i]);
            it2 = S.upper_bound(pos[i]);
            a = (it2 == S.end() ? n : *it2 -1);
            b = (it1 == S.begin() ? 0 : *(--it1));
            ans += (long long)(a - b - (fi(a)-fi(b)));
            upd(pos[i],1);
        }
        else 
            S.insert(pos[i]);
    }
    printf("%I64d\n",ans);
    return 0;
}
