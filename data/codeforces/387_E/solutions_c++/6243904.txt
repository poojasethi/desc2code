#include <iostream>
#include <cstdio>
#include <set>
using namespace std;
#define ll long long
int pos[1000010];
bool need[1000010];
int t[1000010];
int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i = 0;i < n;i++){
        int a;
        scanf("%d",&a);
        pos[a] = i + 1;
    }
    for(int i = 0;i < k;i++){
        int a;
        scanf("%d",&a);
        need[a] = true;
    }
    set<int> s;
    s.insert(0);
    s.insert(n + 1);
    ll ans = 0;
    for(int i = 1;i <= n;i++){
        if(need[i]){
            s.insert(pos[i]);
        }
        else{
            set<int>::const_iterator x = s.upper_bound(pos[i]);
            int up = *x;
            x--;
            int low = *x;
            ans += up - low - 1;
            //cout<<i<<endl;
            for(int j = up - 1;j > 0;j -= (j&(-j)))
                ans -= t[j];
            for(int j = low;j > 0;j -= (j&(-j)))
                ans += t[j];
            for(int j = pos[i];j <= n;j += (j&(-j)))
                t[j]++;
        }
    }
    cout<<ans<<endl;
}
