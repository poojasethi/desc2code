#include<set>

#define rep(i,n) for (int i=1;i<=n;++i)

{

	char ch; x=0; for (ch=CH;ch<'0' || ch>'9';ch=CH);
	for (;ch>='0' && ch<='9';x=x*10+ch-48,ch=CH);
}
int get(int x){int s=0; for (;x;x-=x&-x) s+=c[x]; return s;}
void add(int x){for (;x<=n;x+=x&-x) ++c[x];}
