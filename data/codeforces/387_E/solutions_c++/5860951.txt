#include <cstdio>
#include <iostream>
#include <algorithm>
#include <set>
using namespace std;
int n,k;
int c[1000005],v[1000005],p[1000005],a[1000005],b[1000005];
set<int> L,R;

void add(int k){
    for (int i=k;i<=n;i+=i&-i)
        c[i]++;
}

int get(int k){
    int s=0;
    for (int i=k;i>0;i^=i&-i)
        s+=c[i];
    return s;
}

int main(){
    cin>>n>>k;
    for (int i=1;i<=n;i++){
        scanf("%d",&a[i]);
        v[a[i]]=i;
    }
    for (int i=0;i<k;i++){
        scanf("%d",&b[i]);
        p[b[i]]=1;
    }
    L.insert(0);
    L.insert(n+1);
    long long ans=0;
    for (int i=1;i<=n;i++)
        if (p[i]==1) {
            L.insert(v[i]);
        } else {
            int l,r;
            r=*(L.lower_bound(v[i])); r--;
            l=*(--L.lower_bound(v[i])); l++;
            ans+=r-l+1;
            ans-=get(r)-get(l-1);
            add(v[i]);
        }
    cout<<ans;
    return 0;
}
