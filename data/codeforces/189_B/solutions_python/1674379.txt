#!/usr/bin/env python
import sys

h,w = map(int, raw_input().split())

res = 0
if h < 2 or w < 2:
    res = 0
else:
    hh = []
    ww = []
    for i in xrange(2, h+1, 2): hh.append(i)
    for i in xrange(2, w+1, 2): ww.append(i)

    for a in hh:
        for b in ww:
            res += (h-a+1) * (w-b+1)

print res
