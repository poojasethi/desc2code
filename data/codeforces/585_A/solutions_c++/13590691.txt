#include <cstdio>
#include <vector>
#define ll long long
using namespace std;

ll n,p[4000],v[4000],d[4000];
ll x[4000];
vector<ll> solved;
int main(void) {
	scanf("%I64d",&n);
	for(ll i=0;i<n;++i) {
		scanf("%I64d %I64d %I64d",&v[i],&d[i],&p[i]);
		x[i]=i+1;
	}
	for(ll i=0;i<n;++i) {
		if(!x[i]) continue;
		solved.push_back(x[i]);
		
		ll noise=v[i],extra=0;
		for(ll j=i+1;j<n;++j) {
			if(x[j]==0) continue;
			p[j]-=(noise+extra);
			if(p[j]<0){
				extra+=d[j];
				x[j]=0;
			}
			if(noise>0) noise--;
		}
	}
	
	printf("%d\n",solved.size());
	for(ll i=0;i<solved.size();++i)
		printf("%I64d ",solved[i]);
		
	return 0;
}
