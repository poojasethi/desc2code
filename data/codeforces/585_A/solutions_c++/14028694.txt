#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
#include<cmath>
using namespace std;
int ans[4010],n;
long long v[4010],d[4010],p[4010];
bool mark[4010];
int main(){
	memset(mark,0,sizeof(mark));
	int i,j,k,cnt=0,x=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++) cin>>v[i]>>d[i]>>p[i];
	for(i=1;i<=n;i++){
		if(mark[i]) continue;
		ans[x++]=i;
		long long w=v[i],e=0;
		for(j=i+1;j<=n;j++){
			if(mark[j]) continue;
			p[j]-=w+e;w--;
			if(w<1) w=0;
			if(p[j]<0) {e+=d[j];mark[j]=1;}
		}
	}
	printf("%d\n",x);
	for(i=0;i<x;i++) printf("%d ",ans[i]);
	return 0;
}
