#include <bits/stdc++.h>
using namespace std;
long long v1[4005],v2[4005],a[4005];
vector<int>v;
int main()
{
	int n;
	while (~scanf("%d",&n))
	{
		for (int i=1;i<=n;i++) scanf("%lld%lld%lld",v1+i,v2+i,a+i);
		v.clear();
		for (int i=1;i<=n;i++)
		{
			if (a[i]<0) continue;
			v.push_back(i);
			long long s1=0,s2=v1[i];
			for (int j=i+1;j<=n;j++)
			{
				if (a[j]<0) continue;
				a[j]-=s1+s2;
				if (s2>0) s2--;
				if (a[j]<0) s1+=v2[j];
			}
		}
		printf("%d\n",v.size());
		for (int i=0;i<v.size();i++) 
		{
			if (i) printf(" %d",v[i]);
			else printf("%d",v[i]);
		}
		printf("\n");
	}
	return 0;
}
