#include <iostream>
#include <vector>
using namespace std;
int main(){
	vector < long long > v;
	int n;
	cin>>n;
	int ans=0;
	long long a[n],b[n],c[n];
	bool used[n];
	for(int i=0;i<n;i++){
		cin>>a[i]>>b[i]>>c[i];
	}
	for(int i=0;i<n;i++){
		if(c[i]>=0){
			ans++;
			v.push_back(i+1);
		
		long long t=0;
		for(int j=i+1;j<n;j++){
			if(c[j]>=0){
			c[j]=c[j]-a[i]-t;
		
			if(a[i]>0){
				a[i]--;
			}
			if(c[j]<0){
				t=t+b[j];
			}
			}
			
		}
		}
			
	}
	cout<<ans<<endl;
	
		for(int j=0;j<v.size();j++){
			cout<<v[j]<<" ";
		}
}
