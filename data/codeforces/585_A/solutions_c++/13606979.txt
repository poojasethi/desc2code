#include<iostream>
#include<stdio.h>
using namespace std;
long long  n,v[4005],d[4005],p[4005];
int vis[4005];
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	 scanf("%d%d%d",&v[i],&d[i],&p[i]);
	for(int i=0;i<n;i++)
	if(p[i]>=0)
	{
		vis[i]=1;
		long long sum=0,de=v[i];
		for(int j=i+1;j<n;j++)
		if(p[j]>=0)
		{
			p[j]-=de+sum;
			if(p[j]<0) sum+=d[j];
			if(de>0) de--;
		}
	}
	int num=0;
	for(int i=0;i<n;i++)
	if(vis[i]) num++;
	cout<<num<<endl;
	for(int i=0;i<n;i++)
	if(vis[i]) printf("%d ",i+1);
	cout<<endl;
}
