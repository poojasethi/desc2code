#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int n,i,j,a,b,c,co=0,k;
	cin>>n;
	long long int ar[n][3];
	for ( i = 0; i < n; ++i)
	{
		cin>>ar[i][0]>>ar[i][1]>>ar[i][2];

	}
	for ( i = 0; i < n; ++i)
	{

		if(ar[i][2]>=0)
		{	a=ar[i][0];
			co++;
			b=0;
			for ( j = i+1; j < n && a>0; ++j)
			{

				if(ar[j][2]>=0)
				{
					ar[j][2]-=a;
					if(ar[j][2]<0)
					{
						a+=ar[j][1];
						b+=ar[j][1];
					}

						if(a>b)
						   a--;

				}



			}
		}
		 	}
	cout<<co<<endl;
	for ( i = 0; i < n; ++i)
	{
		if(ar[i][2]>=0)
			cout<<i+1<<" ";
	}

    return 0;
}
