#include<iostream>
#include<vector>
using namespace std;

int main(){
	int n;
	long long v[4005],d[4005],p[4005];
	vector<int> ans;
	cin >> n;
	for(int i=0; i<n; i++)cin >> v[i] >> d[i] >> p[i];
	for(int i=0; i<n; i++){
		if(p[i]<0)continue;
		ans.push_back(i+1);
		long long x=0;
		for(int j=i+1; j<n; j++){
			if(p[j]<0)continue;
			p[j]-=x+max(v[i],0LL);
			if(p[j]<0)x+=d[j];
			v[i]--;
		}
	}
	cout << ans.size() << endl;
	for(int i=0; i<ans.size(); i++)cout << ans[i] << " ";
	return 0;
}
