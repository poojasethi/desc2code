#include<iostream>
#include<stdio.h>
using namespace std;
int n,j,tot;
long long rd,ans,sc[4005],vis[4005],v[4005],d[4005],p[4005];
int main(void)
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		cin>>v[i]>>d[i]>>p[i];
	for(int i=0;i<n;i++)
	{
		if(p[i]<0)continue;
		sc[ans]=i+1;ans++;rd=0;j=v[i];
		for(tot=i+1;tot<n;tot++)
		{
			if(p[tot]<0)continue;
			p[tot]-=j+rd;
			if(j)j--;
			if(p[tot]<0)rd+=d[tot];
		}
	}
	cout<<ans<<endl;
	for(int i=0;i<ans;i++)
	{
		cout<<sc[i];
		if(i==ans-1)printf("\n");
		else printf(" ");
	}
	return 0;
}
