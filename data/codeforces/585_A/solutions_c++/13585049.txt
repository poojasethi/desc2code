#include<iostream>
#define LiangJiaJun main
using namespace std;
int LiangJiaJun(){
    long long n,ans=0;
    cin>>n;
    long long v[n],d[n],p[n];

    for(int i=0;i<n;i++)cin>>v[i]>>d[i]>>p[i];
    for(int i=0;i<n;i++){
        if(p[i]>=0){
            long long d1=0;
            for(int j=i+1;j<n;j++){
                if(p[j]>=0){
                    p[j]-=v[i]+d1;
                    if(p[j]<0)d1+=d[j];
                    if(v[i])v[i]--;
                }
            }
        ans++;
        }
    }
    cout<<ans<<endl;
    for(int i=0;i<n;i++)if(p[i]>=0)cout<<i+1<<" ";
    return 0;
}
