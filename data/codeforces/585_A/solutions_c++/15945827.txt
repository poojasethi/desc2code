#include<bits/stdc++.h>
using namespace std;
int v[4001],n,cnt,ans[4001],d[4001],p[4001];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d%d%d",&v[i],&d[i],&p[i]);
	for(int i=1;i<=n;i++)
	{
		if(p[i]>=0)
		{
			ans[++cnt]=i;
			int c=0;
			for(int j=i+1;j<=n;j++)
			{
				if(p[j]<0) continue;
				p[j]-=(v[i]+c);
				if(v[i]>0) v[i]--;
				if(p[j]<0&&c<1e7)
					c+=d[j];
			}
		}
	}
	cout<<cnt<<endl;
	for(int i=1;i<=cnt;i++) printf("%d ",ans[i]);
	return 0;
}
