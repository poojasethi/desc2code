#include <iostream>
using namespace std;
#include<bits/stdc++.h>

int main()
{
    long long n,i,j,k=0,v[4001],d[4001],p[4001];
    vector<int> ans;
    cin>>n;
    for(i=1;i<=n;++i)
    {
        cin>>v[i]>>d[i]>>p[i];
    }
    
    for(int i=1;i<=n;i++)
    {
		if(p[i]>=0){
			k++;
			ans.push_back(i);
		
		long long t=0;
		for(int j=i+1;j<=n;j++){
			if(p[j]>=0){
			p[j]=p[j]-v[i]-t;
		
			if(v[i]>0){
				v[i]--;
			}
			if(p[j]<0){
				t=t+d[j];
			}
			}
			
		}
		}
			
	}
    cout<<k<<endl;
    for(i=0;i<ans.size();++i)
    cout<<ans[i]<<" ";
	return 0;
}
