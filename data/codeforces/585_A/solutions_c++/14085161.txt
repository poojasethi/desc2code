#include <cstdio>

int n, v[4010], d[4010];
int K, ans[4010], next[4010], last;
long long sum, p[4010];

int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++)
	{
		scanf("%d%d%I64d", &v[i], &d[i], &p[i]);
		next[i] = i + 1;
	}
	for (int i = 1; i <= n; i = next[i])
	{
		ans[++K] = i;
		sum = 0;
		last = i;
		for (int j = next[i], k = v[i]; j <= n; j = next[j])
		{
			p[j] -= k + sum;
			if (k > 0) k--;
			if (p[j] < 0)
			{
				sum += d[j];
				next[last] = next[j];
			} else last = j;
		}
	}
	printf("%d\n", K);
	for (int i = 1; i <= K; i++)
	{
		printf("%d ", ans[i]);
	}
}
