#include<iostream>
#include<vector>
using namespace std;
long long  v[4001],d[4001],p[4001];
vector< int > ann;
int main()
{
	long long n,i,j,k,r,ans = 0,dm=0;
	cin>>n;
	for (i=1;i<=n;i++)cin>>v[i]>>d[i]>>p[i];
	for (i=1;i<=n;i++)
	{
		if (p[i]>=0)
		{

			ann.push_back(i);
			for (j=i+1,r=v[i],dm=0;j<=n;j++)
			{
				if (p[j] < 0 )continue;
				if (r>0)p[j] -= r;
				p[j] -= dm;
				if (p[j] < 0)dm+=d[j];
				r--;
			}
		}
	}
	cout<<ann.size()<<endl;
	for(i=0;i<ann.size();i++)cout<<ann[i]<<' ';
	//system("pause");
	return 0;
	
}
