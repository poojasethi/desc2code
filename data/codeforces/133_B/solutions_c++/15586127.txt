#include<bits/stdc++.h>
using namespace std;
int a;
map<char,int>v;

int main()
{
    v['>'] = 0,v['<'] = 1,v['+'] = 2,v['-'] = 3,v['.'] = 4,v[','] = 5,v['['] = 6,v[']'] = 7;
    string s;
    cin>>s;
    for(int i=0;i<s.length();i++)
        a=(a*16+8+v[s[i]])%1000003;
    cout<<a;
    return 0;
}
