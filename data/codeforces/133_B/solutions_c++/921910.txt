#include <cstdio>
#include <cstring>
char All[] = "><+-.,[]";
int Map[305];
char Inp[105];
int main()
{
	int i,Ans = 0;
	for (i = 0;i < 8;++i)
		Map[All[i]] = i + 8;
	gets(Inp);
	for (i = 0;i < strlen(Inp);++i)
	{
		Ans <<= 4;
		Ans += Map[Inp[i]];
		Ans %= 1000003;
	}
	printf("%d\n",Ans);
	return 0;
}
