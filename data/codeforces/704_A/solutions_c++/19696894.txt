#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int n, q, a, b, br, sol, l;
bool in[300100];
vector<int> v[300100];

int main(){
	scanf("%d %d", &n, &q);
	while ( q-- ){
		scanf("%d %d", &a, &b);
		
		if ( a == 1 ){
			v[b].push_back(br);
			in[br] = true;
			++br; ++sol;
		}
		
		if ( a == 2 ){
			for ( int i = 0; i < int(v[b].size()); i++ )
				if ( in[ v[b][i] ] ){
					in[ v[b][i] ] = false;
					--sol;
				}
			v[b].clear();
		}
		
		if ( a == 3 ){
			int cap = min(br, b);
			for ( ; l < cap; l++ )
				if ( in[l] ){
					in[l] = false;
					--sol;
				}
		}
		
		printf("%d\n", sol);
		
	}
	
	return 0;
}
