z, k = map(int,raw_input().split())
seq = list(map(int,raw_input().split()))

i = 0
j = 0
s = 0
sta = 0
sto = 0
while j < z:
    if seq[j] == 0:
        s += 1
    j += 1
    if (s>k):
        if seq[i] == 0:
            s -= 1
        i += 1
    else:
        if (j-i)>(sto-sta):
            sta = i
            sto = j

L = sto-sta
seq[sta:sto] = [1]*(L)

print L
