n, k = map(int, raw_input().split())
v = list(map(int, raw_input().split()))
res, start = 0, 0
left, changed = 0, 0
for right in xrange(n):
    if v[right] == 0: changed += 1
    while changed > k:
        if v[left] == 0:
            changed -= 1
        left += 1
    elements = right - left + 1
    if elements > res:
        res = elements
        start = left
print res
for i in xrange(n):
    if i >= start and i < start + res:
        v[i] = 1
print " ".join(map(str, v))
