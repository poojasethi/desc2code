#include <bits/stdc++.h>

using namespace std;

int n,k,a[300010],v,ans,vt,d;
int main()
{
    scanf("%d %d",&n,&k);
    for (int i=1; i<=n; i++) scanf("%d",&a[i]);
    v=1; d=0; ans=0;
    for (int i=1; i<=n; i++)
    {
        if (a[i]==0) d++;
        while (d>k)
        {
            if (a[v]==0) d--;
            v++;
        }
        if (i-v+1>ans)
        {
            ans=i-v+1;
            vt=v;
        }

    }
    printf("%d\n",ans);
    for (int i=1; i<=n; i++)
        if ((i>=vt) && (i<=vt+ans-1)) printf("1 "); else printf("%d ",a[i]);
}
