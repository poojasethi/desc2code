#include<cstdio>

int n,k;
int a[300005];

int main()
{
	int l,r,num,ans,index;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	l=index=1,num=ans=0;
	for(int r=1;r<=n;r++)
	{
		if(a[r]==0) num++;
		while(num>k)
		{
			if(a[l++]==0) num--;
		}
		if(ans<(r-l+1))
		{
			ans=r-l+1;
			index=l;
		}
	}
	printf("%d\n",ans);
	for(int i=1;i<=n;i++)
	{
		if(i!=1) printf(" ");
		if(i>=index&&k>0&&a[i]==0)
		{
			printf("1");
			k--;
		}
		else printf("%d",a[i]);
	}
	printf("\n");
	return 0;
}
