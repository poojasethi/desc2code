#include<bits/stdc++.h>
using namespace std;
#define MAXN 300010
bool num[MAXN];
int main()
{
	int n,k;
	cin>>n>>k;
	for(int i=1;i<=n;i++) cin>>num[i];
	int r=1,l=1,ansr=0,ansl=0,cntk=0,ans=0;
	while(l<=n){
		while(r<=n){
			if(num[r]==0){
				if(cntk==k){
					break;
				}
				cntk++;
			}
			r++;
		}
		if(r-l>ans){
			ans=r-l;
			ansl=l;
			ansr=r-1;
		}
		while(l<=n&&num[l]) l++;
		l++;cntk--;
		
	}

	printf("%d\n",ans);
	for(int i=1;i<=n;i++){
		if(i>=ansl&&i<=ansr) 
			printf("1 ");
		else
			printf("%d ",num[i]);
	}
	return 0;
}
