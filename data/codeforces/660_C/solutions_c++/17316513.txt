#include <iostream>
#include <stdio.h>
using namespace std;
int a[300010];
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	int l=0,x=0,ans=0,s=0;
	for(int r=0;r<n;r++){
		x+=1-a[r];
		while(x>m) x+=a[l++]-1;
		if(r-l+1>ans){
			ans=r-l+1;
			s=l;
		}
	}
	printf("%d\n",ans);
	for(int i=0;i<n;i++)
	if(a[i]) printf("1 ");
		else if(m&&i>=s) {printf("1 ");m--;}
			else printf("0 ");
	puts("");
}
