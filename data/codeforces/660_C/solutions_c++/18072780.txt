#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int sum[300003],a[300003];
pair<int,int> ans;
int main(){
	int n,k,j = 1;
	cin>>n>>k;
	for(int i=1;i<=n;i++){
		cin>>a[i];
		sum[i] = sum[i-1] + !a[i];
	}
	for (int i=1;i<=n;i++){
		j = max(j,i);
		for(; j<=n && sum[j] - sum[i-1] <= k; j++);
        j--;
        if (j - i + 1 > ans.second) ans = make_pair(i, j - i + 1);
        
	}
	cout<<ans.second<<"\n";
    for(int i=ans.first; i < ans.first + ans.second; i++) a[i] = 1;
    for(int i=1; i<=n; i++)cout<<a[i]<<" ";
	return 0;
}
