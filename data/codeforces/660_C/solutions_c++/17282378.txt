#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int n,k,i,a[1000000];
	cin>>n>>k;
	for(i=0;i<n;i++)
	cin>>a[i];
	int l=0,r=0,zcount=0,bestseg=0,bestl=0;
	while(r<n)
	{
		if(zcount<=k)
		   {
		   	 if(a[r]==0)
		   	    zcount++;
		   	  r++;  
		   }
	   if(zcount>k)
		   {
		   	 if(a[l]==0)
		   	   zcount--;
		   	 l++;
		   }
		   if(r-l>bestseg)
		   {
		   	bestseg=r-l;
		   	bestl=l;
		   }
	}
	for(i=bestl;i<bestseg+bestl;i++)
	if(a[i]==0)
	a[i]=1;
	cout<<bestseg<<endl;
	for(i=0;i<n;i++)
	cout<<a[i]<<" ";
	
	return 0;
}
