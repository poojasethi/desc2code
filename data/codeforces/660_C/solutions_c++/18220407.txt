#include<bits/stdc++.h>
using namespace std;
int a[(int)(3e5+6)];
int main(){
    int n,k;
    cin>>n>>k;
     for(int i=1;i<=n;i++)cin>>a[i];
     int r=0,tr=0,tl=0,cnt=0,ans=0;
     for(int i=1;i<=n;cnt=cnt-!a[i++]){
         while(cnt<=k && r<=n)
         cnt+=(!a[++r]);
       // printf("#####count is %d#########i is %d#####r is %d#####\n",cnt,i,r);
         r-i>ans?(ans=r-i,tr=i,tl=r):1;
     }
     cout<<ans<<endl;
     for(int i=1;i<=n;i++)
      cout<<(a[i]||i>=tr && i< tl)<<" ";
    return 0;
}
