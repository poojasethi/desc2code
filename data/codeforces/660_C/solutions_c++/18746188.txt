#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>
#define maxn 300010

using namespace std;

int a[maxn];
int n,k,ans,ansl,ansr;

int main()
{
	scanf("%d%d",&n,&k);
	ans=0;ansl=1;ansr=0;
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	int l=1,num=0;
	for (int r=1;r<=n;r++)
	{
		num+=a[r]==0;
		while (num>k && l<=r) num-=a[l++]==0;
		if (r-l+1>ans) ans=r-l+1,ansl=l,ansr=r;  
	}
	printf("%d\n",ans);
	for (int i=1;i<=n;i++) if (i>=ansl && i<=ansr) printf("1 "); else printf("%d ",a[i]);printf("\n");
	return 0;
}
