#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstdlib>
#include <set>
using namespace std;
int n,k,ans=0,i,p,q,a,b,maxi;
int st[300005];
int main()
{
	scanf("%d %d",&n,&k);
	for(int i=0;i<n;++i)
	    scanf("%d ",&st[i]);
	for(i=p=q=a=b=0;i<n;++i)
	{
		p+=st[i]==0;
		while(a<=i&&p>k)
		    p-=st[a++]==0;
		if(p<=k)
		    if(i-a+1>ans)
		        ans=i-a+1,maxi=i;
	}
	printf("%d",ans);
	puts("");
	for(i=maxi;i>maxi-ans;--i)
	    st[i]=1;
	for(int i=0;i<n;++i)
	    cout<<st[i]<<" ";
}
