#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k;
	scanf("%d %d",&n,&k);

	int a[n+100];
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);

	int s=0,c=0,c1=0;
	int ans=0,anss=-1,anss1=-1;

	for(int i=0;i<n;i++)
    {
       if(a[i]==0)
       	c1++;
       while(c1>k)
       {
        	if(a[s]==0)
       		c1--;
       	    s++;
       }
        if(i-s+1>ans)
        {
       	    ans=i-s+1;
        	anss=s;
        	anss1=i;
        }
	}
	for(int i=anss;i<=anss1;i++)
		a[i]=1;

	printf("%d\n",ans);

	for(int i=0;i<n;i++)
    {
		printf("%d ",a[i]);
	}
	return 0;
}
