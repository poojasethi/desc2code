#include<bits/stdc++.h>
using namespace std;

int A[300005];

int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=0;i<n;i++)
        cin>>A[i];
    int l=0,r=-1,ll=-1,rr=-2,k=0;
    while(r<n)
    {
        if(rr-ll<r-l)
        {
            rr=r;
            ll=l;
        }
        if(A[++r]==0)
            k++;
        while(k>m)
        {
            if(A[l++]==0)
                k--;
        }
    }
    cout<<rr-ll+1<<endl;
    for(int i=0;i<n;i++)
    {
        if(i) cout<<" ";
        if(i>=ll&&i<=rr) cout<<1;
        else cout<<A[i];
    }
    cout<<endl;
    return 0;
}
