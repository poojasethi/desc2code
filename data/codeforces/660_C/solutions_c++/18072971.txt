#include<cstdio>
#define N 300010
using namespace std;

int n,k,a[N],sum[N];

int main(){
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	for(int i=1;i<=n;i++) sum[i]=sum[i-1]+a[i];
	int l=1,r=0;
	for(int i=1;i<=n;i++){
		if(i-sum[i]==k){
			r=i;
			break;
		}
	}
	int mx=r-l+1,mxl=l;
	for(r=r+1;r<=n;r++){
		for(;(r-l+1)-(sum[r]-sum[l-1])>k;l++) continue;
		if(r-l+1<mx) continue;
		mx=r-l+1;mxl=l;
	}
	printf("%d\n",mx);
	for(int i=1;i<=n;i++){
		if(mxl<=i&&i<mxl+mx){
			printf("1 ");
		}else{
			printf("%d ",a[i]);
		}
	}
	return 0;
}
