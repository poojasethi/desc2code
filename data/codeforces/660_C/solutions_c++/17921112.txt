#include<stdio.h>
#include<string.h>
const int maxn=300000+10;
int n,k;
int a[maxn];
int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	int ans=0,ansl,ansr;
	int l,r=0,cnt=1;
	for (l=1;l<=n;l++)
	{
		if (a[l-1]==0) cnt--;
		while (true)
		{
			if (r==n) break;
			if (a[r+1]==1) r++;
			else
			{
				if (cnt==k) break;
				cnt++;r++;
			}
		}
		if (r-l+1>ans)
		{
			ans=r-l+1;
			ansl=l;ansr=r;
		}
	}
	printf("%d\n",ans);
	if (k!=0) for (int i=ansl;i<=ansr;i++) a[i]=1;
	for (int i=1;i<=n;i++) printf("%d ",a[i]);
	printf("\n");
	return 0;
}
