#include <bits/stdc++.h>

using namespace std;

const int kMaxN=400001;
int a[kMaxN];
int sum[kMaxN];

int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++){
		scanf("%d",a+i);
		sum[i]=sum[i-1]+!a[i];
	}
	int p=1,q=0;
	int ans=0;
	int l=0,r=0;
	while(p<=n && q<n){
		q++;
		while(p<=q && sum[q]-sum[p-1]>k){
			p++;
		}
		if(q-p+1>=ans){
			ans=q-p+1;
			l=p,r=q;
		}
	}
	for(int i=l;i<=r;i++){
		a[i]=1;
	}
	printf("%d\n",ans);
	for(int i=1;i<=n;i++){
		printf("%d ",a[i]);
	}
	return 0;
}
