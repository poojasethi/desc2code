#include<bits/stdc++.h>


using namespace std;


int ar[3*100005];

int main(){
	
	int n,k;
	scanf("%d %d",&n,&k);	
	for(int i=0;i<n;i++)
		scanf("%d",&ar[i]);

	int j=0;
	int maxi=-1;
	int x,y;
	for(int i=0;i<n;i++){
		if(ar[i]==0 && k>0){
			k--;
		}else if(ar[i]==0 && k==0){
			while(k==0){
				if(ar[j]==0) k++;
				j++;
			}
			k--;
		}
		if(maxi<i-j+1){
			x=j,y=i;
			maxi=i-j+1;
		}
	}
	
	printf("%d\n",maxi);
	for(int w=x;w<=y;w++)
		ar[w]=1;
	for(int i=0;i<n-1;i++)
		printf("%d ",ar[i]);
	printf("%d\n",ar[n-1]);
	return 0;
}
