#include<algorithm>
#include<iostream>
#include<cstdlib>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;

#define N 300010

int a[N],b[N];
int n,k;
int ans,cnt;

int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if (!a[i]) 
			b[i]=1;
        b[i]+=b[i-1];
	}
	for (int i=1,j=1;i<=n;i++)
	{
		while (j<=n && b[j]-b[i-1]<=k)
			j++;
		int k=j-i;
		if (ans<k)
			ans=k,cnt=i;
	}
	for (int i=0;i<ans;i++)
		a[cnt+i]=1;
	printf("%d\n",ans);
	for (int i=1;i<=n;i++)
		printf("%d ",a[i]);
	return 0;
}
