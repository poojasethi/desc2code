#include <stdio.h>
#include <iostream>
#include <cstring>
const int N=300005;
int n,k,a[N];
int ans=0,ansl,ansr;
using namespace std;

int main()
{
	//freopen ("C.in","r",stdin);
	//freopen ("C.out","w",stdout);
	scanf ("%d %d",&n,&k);
	int i;
	for (i=1;i<=n;i++) scanf ("%d",&a[i]);
	int l,r,sum=0;
	for (l=1,r=0;l<=n;l++)
	{
		while (r<=n && sum<=k)
		{
			r++;
			if (!a[r]) sum++;
		}
		if (ans<r-l) ans=r-l,ansl=l,ansr=r-1;
		if (!a[l]) sum--;
	}
	printf ("%d\n",ans);
	for (i=1;i<=n;i++)
	if (i>=ansl && i<=ansr) printf ("1 ");
	else printf ("%d ",a[i]);
	return 0;
}
