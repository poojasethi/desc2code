#include <bits/stdc++.h>

using namespace std;

const int maxn=1e6;
int n,k;
int a[maxn];
int main(){
	cin>>n>>k;
	for(int i=1;i<=n;i++) cin>>a[i];
	int cnt=0;
	int ans=0;
	int j=1;
	int s=0,t=0;
	for(int i=1;i<=n;i++){
		while(cnt<k&&j<=n){
			if(a[j]==0) cnt++;
			j++;
		}
		while(a[j]==1) j++;
		if(j-i>ans){
			ans=j-i;
			s=i,t=j;
		}
		cnt-=a[i]^1;
	}
	cout<<ans<<endl;
	for(int i=1;i<=n;i++){
		if(i>=s&&i<t) cout<<1<<" ";
		else cout<<a[i]<<" ";
	}
	return 0;
}
