#include <bits/stdc++.h>
using namespace std;

int main() {
	int n, k; cin >> n >> k;
	int items[n];
	for(int i = 0; i < n; i++) cin >> items[i];
	
	int l = 0, r = 0;
	int zeros = 0;
	int max_l = -1, max_r = -1;
	for(int i = 0; i < n; i++) {
		l = i;
		while(r < n) {
			if(zeros + !items[r] > k)
				break;
			zeros += !items[r];
			r++;
		}
		
		if(r - l > max_r - max_l) {
			max_r = r;
			max_l = l;
		}
		
		if(items[l] == 0) zeros--;
	}
	

	for(int i = max_l; i < max_r; i++)
		items[i] = 1;
		
	cout << max_r - max_l << endl;
	for(int i = 0;  i < n; i++)
		cout << items[i] << " ";
}
