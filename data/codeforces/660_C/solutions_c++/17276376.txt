#include <cstdio>
const int N=300001;
int sum[N]={0,},a[N];
int main(){
    int n,k,i,j=0,ans=0,tt=0;
    scanf("%d %d",&n,&k);
    for(i = 1; i <= n; i++) {
        scanf("%d",&a[i]);
        sum[i]=sum[i-1]+a[i];
    }
    for(i = 1; i <= n; i++){
        if(a[i]==0){
            while(sum[i]-sum[j]+k <i-j)
                j++;
        }
        if((i - j) > ans) {
            ans=i-j;
            tt=i;
        }
    }
    printf("%d\n",ans);
    for(i=tt; i >= 1 && k; i--){
        if(a[i]==0)
            a[i]=1,k--;
    }
    for(i=1;i<=n;i++)
        printf("%d ",a[i]);
    return 0;
}
