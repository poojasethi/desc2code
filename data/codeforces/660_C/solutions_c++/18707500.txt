#include <iostream>
using namespace std;
int arr[300005];
int main()
{
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	int start=0;
	int end=0;
	int ans_start=start,ans_end=end,ans=-1;
	while(end<n)
	{
		if(arr[end]==0)
			k--;
		while(k<0)
		{
			
			if(arr[start]==0)
			{
				k++;
			}
			start++;
		}
		if(end-start+1>ans)
		{
			ans=end-start+1;
			ans_end=end;
			ans_start=start;
		}
		end++;
	}
	cout<<ans<<endl;
	//cout<<ans_start<<" "<<ans_end<<endl;
	for(int i=0;i<n;)
	{
		if(i==ans_start && ans_start<=ans_end)
			for(;i<=ans_end;i++)
				cout<<1<<" ";
		else
			cout<<arr[i]<<" ",i++;
	}
	cout<<endl;
}
