#include <bits/stdc++.h>

using namespace std;

int a[(int)3e5+1], b[(int)3e5+1];

int main()
{
	int n, k, ans = 0, r, l;
	cin >> n >> k;
	for(int i = 1; i <= n; i++)
	{
		cin >> a[i];
		b[i] = b[i-1]+(a[i] ? 0 : 1);
	}
	for(int i = 1; i <= n; i++)
	{
		int x = upper_bound(b+i, b+n+1, b[i-1]+k)-b;
		if(x-i > ans)
			ans = x-i, r = i, l = x;
	}
	cout << ans << endl;
	for(int i = 1; i <= n; i++) cout << (a[i] || i >= r && i < l) << ' ';
	cout << endl;
}
