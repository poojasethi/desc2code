#include <cstdio>
int a[300300],N,M;
int main(){
    scanf("%d%d", &N,&M);
    for(int i=1; i<=N; i++)
        scanf("%d",&a[i]);
    int mxL=-1; int mxR=-1;
    for(int L=1,R=0; L<=N; L++){
        while(M>=0 && R<=N){
            R++; 
            if(a[R]==0) M--;
        }
        if(R-L>mxR-mxL) mxR=R, mxL=L;
        if(a[L]==0) M++;
    }
    printf("%d\n", mxR-mxL);
    for(int i=1; i<=N; i++) 
        if(i>=mxL&&i<mxR) printf("1 ");
            else printf("%d ",a[i]);
}
