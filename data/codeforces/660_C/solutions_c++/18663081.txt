#include<cstdio>
using namespace std;
int n,K,num,ans,l,r,a[300010];
int main()
{
	scanf("%d%d",&n,&K);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	int j=1;num=0;
	for (int i=1;i<=n;i++)
	{
		if (!a[i]) num++;
		while (num>K){if (!a[j]) num--;j++;}
		if (i-j+1>ans)
		{
			ans=i-j+1;
			l=j;r=i;
		}
	}
	for (int i=l;i<=r;i++) a[i]=1;
	printf("%d\n",ans);
	for (int i=1;i<n;i++) printf("%d ",a[i]);printf("%d\n",a[n]);
}
