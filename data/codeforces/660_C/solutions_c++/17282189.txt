#include<iostream>
using namespace std;
int main(){
	int n,k;
	cin>>n>>k;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	int cnt=0,ansl=0,ansr=0,j=0;
	for(int i=0;i<n;i++){
		if(j<i)
		{
		j=i;
		cnt=0;
	    }
	    while(j<n){
	    	int ncnt=cnt+!a[j];
	    	if(ncnt>k)
	    	break;
	    	cnt+=!a[j];
	    	j++;
		}
		if (j - i > ansr - ansl)
			ansl = i, ansr = j;

		if (cnt > 0) cnt -= !a[i];
	}
	cout << ansr - ansl << endl;
	for(int i=ansl;i<ansr;i++)
	a[i]=1;
	for(int i=0;i<n;i++)
	cout<<a[i]<<" ";
	return 0;
}
