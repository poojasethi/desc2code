#include <iostream>

using namespace std;

int n,k;
int v[1000000];

int maxsol=-10;
int i1sol,i2sol;

int main()
{ 
  ios::sync_with_stdio(false);
  cin>>n>>k;
  for (int i=1;i<=n;i++)
    cin>>v[i];
  int i1=1;
  int amount=0;
  for (int i2=1;i2<=n;i2++) {
    amount+=v[i2]==0;
    while (amount>k) {
      amount-=v[i1]==0;
      i1++;
    }
    if (i2-i1>maxsol) {
      maxsol=i2-i1;
      i1sol=i1;
      i2sol=i2;
    }
  }
  for (int i=i1sol;i<=i2sol;i++)
    v[i]=1;
  cout<<i2sol-i1sol+1<<endl;
  for (int i=1;i<=n;i++) {
    if (i>1) cout<<" ";
    cout<<v[i];
  }
  cout<<endl;
}
