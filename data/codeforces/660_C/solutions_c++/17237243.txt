#include <bits/stdc++.h>
using namespace std;

const int N=300010;
int a[N],c[N],ans,pos,n,k;

int main()
{
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++) scanf("%d",a+i);
	int last=0,tot=0;
	for(int i=0;i<n;i++) if(a[i]==1)
	{
		if(i-last+1>ans) 
		{
			ans=i-last+1;
			pos=i;
		} 
	}
	else
	{
		tot++;
		if(tot>k) 
		{
			while(tot>k)
			{
				if(a[last]==0) tot--;
				last++;
			}
		}
		if(i-last+1>ans)
		{
			ans=i-last+1;
			pos=i;
		}
	}
	for(int i=pos-ans+1;i<=pos;i++) a[i]=1;
	printf("%d\n",ans);
	for(int i=0;i<n;i++) printf("%d ",a[i]);
	printf("\n");
	return 0;
}
