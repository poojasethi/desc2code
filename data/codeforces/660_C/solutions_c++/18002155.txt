#include <cstdio>
#include <cstring>

int n,k,l,r,li,ri,ans,ansl,ansr,maxn;
int a[300001],q[300001];

int main()
{
	int i;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++) scanf("%d",&a[i]);
	l=1;
	r=0;
	li=1;
	ri=0;
	for(i=1;i<=n;i++)
	{
		if(a[i]) ri++;
		else
		{
			if(ri-li+1>ans)
			{
				ans=ri-li+1;
				ansl=li;
				ansr=ri;
			}
			q[++r]=++ri;
			while(r-l+1>k)
			{
				li=q[l]+1;
				l++;
			}
		}
	}
	if(ri-li+1>ans)
	{
		ans=ri-li+1;
		ansl=li;
		ansr=ri;
	}
	printf("%d\n",ans);
	for(i=ansl;i<=ansr;i++) a[i]=1;
	for(i=1;i<=n;i++) printf("%d ",a[i]);
	return 0;
}
