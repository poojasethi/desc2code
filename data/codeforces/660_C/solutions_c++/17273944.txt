#include<iostream>
using namespace std;
int main(void)
{
	int n,k,temp=0;
	cin>>n>>k;
	int a[n];
	int i;
	for(i=0;i<n;i++)
	cin>>a[i];
	int res=0;
	int l=0,r=0,count0=0,count1=0;
	int p=0; 
	while(r<n)
	{
		while(r<n)
		{
		if(a[r++]==0)
		count0++;
		else count1++;
		if(count0<=k)
		{
		res++;
		p=l;
		}
		else break;
		}
		if(a[l++]==0)
		count0--;
		else count1--;
	}
	for(i=p;i<p+res;i++)
	a[i]=1;
	cout<<res<<endl;
	for(i=0;i<n;i++)
	{
		if(i==0)
		cout<<a[i];
		else cout<<" "<<a[i];
	}
	return 0;
}
