#include <bits/stdc++.h>
using namespace std;
#define M 301010

int n, m;
int s[M], a[M];
int I, J, res;

int main() {
    scanf("%d",&n);
    scanf("%d",&m);
    for (int j = 0, i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
        s[i] = s[i-1] + (!a[i]);
        while (s[i] - s[j] > m) {
            j++;
        }
        if (i - j > res) {
            res = i - j;
            I = i; J = j;
        }
    }
    printf("%d\n", res);
    for (int i = 1; i <= n; i++) {
        printf("%d ", a[i] || (J < i && i <= I));
    }
    return 0;
}
