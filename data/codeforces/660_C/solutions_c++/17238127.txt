#include <bits/stdc++.h>
using namespace std;
vector <int> u, v;

int main() {
	int n, k;
	cin >> n >> k;

	for(int i = 1; i <= n; i++) {
		int x;
		cin >> x;
		u.push_back(x);
	}
	int ans = 0;
	int b, e; 
	v.push_back(-1);

	for(int i = 0; i < n; i++) {
		if(u[i] == 0) v.push_back(i);
		int start = max(1, int(v.size()) - k);
		int B, E;
		B = v[start - 1] + 1;
		E = i;
		if(ans < E - B + 1) {
			ans = E - B + 1;
			b = B;
			e = E;
		}
	}
	for(int i = b; i <= e; i++) u[i] = 1;
	cout << ans << endl;
	for(int i = 0; i < n; i++) cout << u[i] << " ";
	cout << endl;
	return 0;
}
