#include <iostream>
using namespace std;
int main()
{
    long long int n,k;
     int a[300000];
    cin>>n>>k;
    for (int i=0; i<n; i++) cin>>a[i];
    int ansl = 0, ansr = 0;
	int j = 0, cnt = 0;
	for(int i=0; i<n; i++) {
		if (j < i) {
			j = i;
			cnt = 0;
		}

		while (j < n) {
			int ncnt = cnt + !a[j];
			if (ncnt > k) break;
			cnt += !a[j];
			j++;
		}
		
		if (j - i > ansr - ansl)
			ansl = i, ansr = j;

		if (cnt > 0) cnt -= !a[i];
	}

	cout << ansr - ansl << endl;
	for(int i=ansl; i<ansr; i++) a[i] = 1;
	for(int i=0; i<n; i++) {
	cout<<a[i]<<' ';
	}

}
