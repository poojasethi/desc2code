#include<bits/stdc++.h>

using namespace std;

#define sgmn int v, int tl, int tr
#define tm   (tl + tr >> 1)
#define sol  v*2, tl, tm
#define sag  v*2+1, tm+1, tr
#define ll long long
#define N 300005

int n,k,i,j,x,l,r;

bool A[N];

int main(){

	cin >> n >> k;

	for(i=1;i<=n;i++)
		cin >> A[i];

	for(i=j=1;i<=n;i++){
		
		while(x + !A[j] <= k && j <= n)  x += !A[j++];
		
		x -= !A[i];

		if(r - l < j - i)  l = i,  r = j;
	}

	cout << r - l << endl;

	for(i=1;i<=n;i++)
		cout << (A[i] || (i >= l && i < r)) << " ";

	return 0;
}
