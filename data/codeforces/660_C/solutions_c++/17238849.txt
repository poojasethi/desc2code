#include <cstdio>
#include <cstdlib>
using namespace std;
#define MN 100100
int N,K,R,a[3*MN];
int main(){
	scanf("%d%d", &N,&K);
	for(int i=1; i<=N; i++)
		scanf("%d", &a[i]);
	for(R=1; R<=N; R++)	{
		if(!a[R]) K--;
		if(K<0) break;
	}
	int MAXR=R-1; int MAXL=0;
	if(R>N){
		printf("%d\n", N);
		for(int i=1; i<=N; i++)
			printf("1 ");
		return 0;
	}
	for(int L=0; L<=N&&R<=N; R++){
		if(!a[R]){
			while(a[L+1]&&L<=N) L++;
			L++;
		}
		if(R-L>MAXR-MAXL)MAXL=L,MAXR=R;
	}
	printf("%d\n", MAXR-MAXL);
	for(int i=1; i<=N; i++)
		if(i<=MAXL||i>MAXR) printf("%d ",a[i]);
						else printf("1 ");
}
