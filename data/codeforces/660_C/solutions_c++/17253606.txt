#include<bits/stdc++.h>
using namespace std;
int arr[1000000];
int main() 
{
	int n,k,l,ans,leftarrr,rightarr,element;
	cin>>n>>k;
	for(int i=0;i<n;i++) 
	cin>>arr[i];
	l=0;
	ans=0;
	element=0;
	leftarrr=1;
	rightarr=-1;
	for(int i=0;i<n;i++)
	{
		if(arr[i]==0)
		 element++;
		while(element>k)
		{
			if(arr[l]==0) 
			element--;
			l++;
		}
		if(i-l+1>ans)
		{
			leftarrr = l;
			rightarr = i;
			ans = i-l+1;
		}
	}
	cout<<ans<<endl;
	for(int i=0;i<n;i++)
	{
		if(i>=leftarrr && i<=rightarr) 
		cout<<"1";
		else 
		cout<<arr[i];
		if(i!=n-1)
	    cout<<" ";
	}
return 0;
}
