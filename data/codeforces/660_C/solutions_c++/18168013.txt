#include<bits/stdc++.h>

using namespace std;

int n,k;
int str[300005];

int v1=-1,v2=-1;

int solve()
{
    int num=0,ans=0;

   for(int i=0,j=0;i<n;i++)
   {
    while(j<n&&((num+(str[j]!=1))<=k))
    {
        num+=(str[j]!=1);
        j++;
    }

   if(j-i>ans)
    {
    	ans=j-i;
    	v1=i;
    	v2=j;
    }
    num-=(str[i]!=1);
   }

   return ans;
}

int main()
{

cin>>n>>k;

for(int i=0;i<n;i++)
 cin>>str[i];

cout<<solve()<<endl;

//cout<<v1<<" "<<v2;

for(int i=0;i<n;i++)
{
	if(i>=v1&&i<v2)
	 cout<<"1 ";
	else
	 cout<<str[i]<<" ";
}


return 0;
}
