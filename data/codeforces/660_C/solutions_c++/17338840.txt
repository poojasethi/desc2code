#include<cstdio>
#define N 400000
using namespace std;
int a[N],p1,p2,num,k,n,pos,mx;
void go(){
	while(p1<=p2&&num>k){
		if(a[p1]==0) num--;
		p1++;
	}
}
int main(){
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	p1=1;p2=0;num=0;
	a[0]=-1;
	while(p2<n){
		p2++;
		if(a[p2]==0) num++;
		if(num>k) go();
		if(p2-p1+1>mx) mx=p2-p1+1,pos=p2;
	}
	printf("%d\n",mx);
	for(int i=pos;i>=pos-mx+1;i--) a[i]=1;
	for(int i=1;i<=n;i++) printf("%d ",a[i]);
}
