#include <cstdio>
bool a[300000];
int n, k;
int left = 0, ans = 0, from;
int main (){
	scanf("%d %d", &n, &k);
	for (int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	for (int right = 0; right < n; right++){
		if (!a[right])
			k--;
		while (k < 0){
			if (!a[left])
				k++;
			left++;
		}
		if (ans < right - left + 1){
			ans = right - left + 1;
			from = left;
		}
	}
	printf("%d\n", ans);
	for (int i = 0; i < n; i++)
		printf("%d%c", i >= from && i < from + ans ? 1 : a[i], i != n - 1 ? ' ' : '\n');
	return 0;
}
