#include<stdio.h>
#define N 300100
int a[N];
int n,s;
int main()
{
  scanf("%d%d",&n,&s);
  for(int i=1;i<=n;i++)
    scanf("%d",&a[i]);
  int l,r,bl,br,ans,sum;
  l=r=1;
  ans=0;
  bl=0;
  br=0;
  sum=0;
  while(l<=n)
  {
  while(r<=n&&sum<=s)
  {
    if(!a[r])
    {
      if(sum==s)
        break;
      else
        sum++;
    }
    r++;
  }
  if(r-l>ans) ans=r-l,br=r-1,bl=l;
  while(l<=n&&a[l]) l++;
  sum--;
  l++;
  }
  printf("%d\n",ans);
  for(int i=1;i<=n;i++)
  {
   if(i<=br&&i>=bl)
      printf("1 ");
   else
    printf("%d ",a[i]);
  }
  return 0;
}
