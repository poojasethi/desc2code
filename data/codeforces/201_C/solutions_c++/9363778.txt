#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int a[N];
int64 f[N],h[N],g[N],an;
int main() {
	scanf("%d",&n); n--;
	For(i,1,n) scanf("%d",&a[i]);
	For(i,1,n) {
		if (a[i]>1) f[i+1]=max(f[i+1],f[i]+a[i]-(a[i]&1));
		h[i+1]=max(h[i+1],max(h[i],f[i])+a[i]-(a[i]%2==0));
	}
	for (i=n;i;i--) if (a[i]>1) g[i]=max(g[i],g[i+1]+a[i]-(a[i]&1));
	For(i,1,n+1) {
		an=max(an,h[i]+g[i]);
		an=max(an,max(f[i],g[i]));
	}
	printf("%I64d\n",an);
	return 0;
}
