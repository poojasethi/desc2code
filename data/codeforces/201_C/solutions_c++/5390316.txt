#include <iostream>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n;
long long a[100010],b[100010],kl[100010],kr[100010],nl[100010],nr[100010],res;

int main(){
  cin>>n; rep(i,n-1)cin>>a[i],b[n-i-2]=a[i];
  rep(i,n){
    kl[i+1] = (a[i]==1?0:kl[i]+a[i]/2*2);
    nl[i+1] = max(kl[i+1],nl[i]+(a[i]-1)/2*2+1);
    kr[i+1] = (b[i]==1?0:kr[i]+b[i]/2*2);
    nr[i+1] = max(kr[i+1],nr[i]+(b[i]-1)/2*2+1);
  }

  rep(i,n){
    res = max(res, kl[i]+nr[n-1-i]);
    res = max(res, kr[i]+nl[n-1-i]);
  }
  cout<<res<<endl;
}
