#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#include<vector>
using namespace std;
#define go(i,n) for(size_t i=0;i<n;++i)
typedef unsigned u32;
u32 a[102400],n;
unsigned long long f[102400][3],ans;
void ck(__typeof(ans) x){if(x>ans)ans=x;}
u32 odd(u32 x){return x-((x^1u)&1u);}
u32 even(u32 x){return x&~1u;}
int main(){
	scanf("%d",&n);--n;
	go(i,n)scanf("%u",a+i);
	go(i,n){
		int i1=i+1;
		if(a[i]>1){
			ck(f[i1][0]=even(a[i])+ f[i][0]);
			ck(f[i1][1]=odd(a[i])+max(f[i][0],f[i][1]) );
			ck(f[i1][2]=even(a[i])+max(f[i][1],f[i][2]) );
		}else{
			ck(f[i1][1]=1+max(f[i][0],f[i][1]));
			f[i1][0]=f[i1][2]=0;
		}
	}
	printf("%I64u\n",ans);
}
