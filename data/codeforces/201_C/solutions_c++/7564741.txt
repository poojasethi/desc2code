#include <iostream>
using namespace std;
unsigned long long a[100010];
unsigned long long dpL[100010][2];
unsigned long long dpR[100010][2];
int main()
{
	unsigned long long s=0,n;
	cin>>n;
	for(int i=1;i<n;i++)
	{
		cin>>a[i];
	}
	for(int i=1;i<n;i++) 
	{
		if(a[i]==1)
		{
			dpL[i][0]=0;
		}
		else
		{
			dpL[i][0]=dpL[i-1][0]+a[i]-(a[i]%2);
		}
		dpL[i][1]=max(dpL[i-1][1]+a[i]+(a[i]%2)-1,dpL[i][0]);
	}
	for(int i=n-2;i>0;i--)
	{
		if(a[i+1]==1)
		{
			dpR[i][0]=0;
		}
		else
		{
			dpR[i][0]=dpR[i+1][0]+a[i+1]-(a[i+1]%2);
		}
		dpR[i][1]=max(dpR[i+1][1]+a[i+1]+(a[i+1]%2)-1,dpR[i][0]);
	}
	for(int i=0;i<n;i++)
	{
		s=max(s,max(dpL[i][1]+dpR[i][0],dpL[i][0]+dpR[i][1]));
	}
	cout<<s<<endl;
	return 0;
}
