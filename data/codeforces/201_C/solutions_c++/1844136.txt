#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[111111];
long long f[111111][2],g[111111][2];
int main(){
	int n,i;
	long long ans;
	scanf("%d",&n);
	for (i=1;i<n;i++) scanf("%d",a+i);
	f[0][0]=f[0][1]=0;
	for (i=1;i<n;i++){
		f[i][0]=a[i]>1?f[i-1][0]+a[i]/2*2:0;
		f[i][1]=max(f[i-1][0],f[i-1][1])+a[i]-((a[i]-1)&1);
	}
	g[n-1][0]=g[n-1][1]=0;
	for (i=n-2;i>=0;i--){
		g[i][0]=a[i+1]>1?g[i+1][0]+a[i+1]/2*2:0;
		g[i][1]=max(g[i+1][0],g[i+1][1])+a[i+1]-((a[i+1]-1)&1);
	}
	ans=0;
	for (i=0;i<n;i++)
		ans=max(ans,max(f[i][0]+max(g[i][0],g[i][1]),g[i][0]+max(f[i][0],f[i][1])));
	printf("%I64d\n",ans);
	return 0;
}
