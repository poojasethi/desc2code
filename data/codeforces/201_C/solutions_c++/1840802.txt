#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int a[100000];
long long b[100000][3];
int main()
{
	int n,i,x;
	long long ans;
	scanf("%d",&n);
	for (i=1;i<n;i++)
		scanf("%d",a+i);
	b[0][0]=b[0][1]=b[0][2]=0;
	ans=0;
	for (i=1;i<n;i++)
	{
		if (a[i]%2==0)
			x=a[i]-1;
		else
			x=a[i];
		b[i][0]=max(b[i-1][0],b[i-1][1])+x;
		if (a[i]==1)
			b[i][1]=b[i][2]=0;
		else
		{
			if (a[i]%2==0)
				x=a[i];
			else
				x=a[i]-1;
			b[i][1]=b[i-1][1]+x;
			b[i][2]=max(b[i-1][2],b[i-1][0])+x;
		}
		if (b[i][0]>ans)
			ans=b[i][0];
		if (b[i][1]>ans)
			ans=b[i][1];
		if (b[i][2]>ans)
			ans=b[i][2];
	}
	printf("%I64d\n",ans);
	return 0;
}
