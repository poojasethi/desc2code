#include<stdio.h>
#include<algorithm>
using namespace std;
__int64  map[111111];
__int64 l[111111][2];
__int64 r[111111][2];
int main()
{
int n,i;
scanf("%d",&n);
for(i=2;i<=n;i++)
scanf("%I64d",&map[i]);
l[1][0]=0;
l[1][1]=0;
r[n][0]=0;
r[n][1]=0;
for(i=2;i<=n;i++)
 {
	l[i][0]=max(l[i-1][0],l[i-1][1])+map[i]-(map[i]+1)%2;
	if(map[i]>1)
	l[i][1]=l[i-1][1]+map[i]/2*2;
	else l[i][1]=0;
 }
 for(i=n-1;i>=1;i--)
 {
	r[i][0]=max(r[i+1][0],r[i+1][1])+map[i+1]-(map[i+1]+1)%2;
	if(map[i+1]>1)
	r[i][1]=r[i+1][1]+map[i+1]/2*2;
	else r[i][1]=0;
 }
 __int64 maxx=0;
 for(i=1;i<=n;i++)
 {
 	maxx=max(l[i][0]+r[i][1],maxx);
 	maxx=max(l[i][1]+r[i][0],maxx);
 	maxx=max(l[i][1]+r[i][1],maxx);
 }
	printf("%I64d\n",maxx);
}
