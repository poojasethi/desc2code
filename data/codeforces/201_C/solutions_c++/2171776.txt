#include <cstdio>
#include <iostream>
#define MAXN 100000
using namespace std;
int n;
long long left1[MAXN+1];
long long left2[MAXN+1];
long long right1[MAXN+1];
long long right2[MAXN+1];
long long ar[MAXN];
int main()
{
	int i;
	long long res=0;
	scanf(" %d",&n);
	for(i=1;i<n;i++)
		cin >> ar[i];
	
	for(i=2;i<=n;i++)
	{
		if(ar[i-1]!=1) 
			left1[i]=left1[i-1]+ar[i-1]/2*2;
		left2[i]=max(left1[i],ar[i-1]-(!(ar[i-1]&1))+left2[i-1]);
	}
	
	for(i=n-1;i>=1;i--)
	{
		if(ar[i]!=1) 
			right1[i]=right1[i+1]+ar[i]/2*2;
		right2[i]=max(right1[i],ar[i]-(!(ar[i]&1))+right2[i+1]);
	}
	for(i=1;i<=n;i++)
		res=max(res,max(left1[i]+right2[i],right1[i]+left2[i]));
	cout << res << endl;
	return 0;
}
