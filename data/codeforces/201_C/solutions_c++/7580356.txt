#include<iostream>
#include<algorithm>
using namespace std;
long long a[100000],dpl[100000][2],dpr[100000][2],j=0;
int main(){
	int n;
	cin>>n;
	for(int i=1;i<n;i++) cin>>a[i];
	for(int i=1;i<n;i++){
		if(a[i]==1) dpl[i][0]=0;
		else dpl[i][0]=dpl[i-1][0]+a[i]-a[i]%2;
		dpl[i][1]=max(dpl[i-1][1]+a[i]-!(a[i]%2),dpl[i][0]);
	}
	for(int i=n-2;i>0;i--){
		if(a[i+1]==1) dpr[i][0]=0;
		else dpr[i][0]=dpr[i+1][0]+a[i+1]-a[i+1]%2;
		dpr[i][1]=max(dpr[i+1][1]+a[i+1]-!(a[i+1]%2),dpr[i][0]);
	}
	for(int i=0;i<n;i++){
		j=max(j,max(dpr[i][1]+dpl[i][0],dpr[i][0]+dpl[i][1]));
	}
	cout<<j<<endl;
	return 0;
}
