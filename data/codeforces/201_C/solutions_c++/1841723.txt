#include <cstdio>
#include <algorithm>

using std::max;

int n, a[111111];
long long f[111111][4];
long long ans;

int main() {
	scanf("%d", &n);
	-- n;
	for (int i = 1; i <= n; ++ i) scanf("%d", &a[i]); 
	f[0][0] = 0LL;
	for (int i = 0; i < n; ++ i) {
		if (a[i + 1] > 1) f[i + 1][1] = max(f[i + 1][1], max(f[i][0], f[i][1]) + (a[i + 1] - (a[i + 1] & 1)));
		f[i + 1][2] = max(f[i + 1][2], max(f[i][2], max(f[i][0], f[i][1])) + (a[i + 1] - (~a[i + 1] & 1)));
		if (a[i + 1] > 1) f[i + 1][3] = max(f[i + 1][3], max(f[i][2], f[i][3]) + (a[i + 1] - (a[i + 1] & 1)));
		ans = max(ans, max(f[i + 1][1], max(f[i + 1][2], f[i + 1][3])));
	}
	printf("%I64d\n", ans);
	return 0;
}
