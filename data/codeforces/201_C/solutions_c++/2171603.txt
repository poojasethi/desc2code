#include <cstdio>
#include <iostream>
#define  MAXN 100010
#define  tek(a) ((a%2)?(a):(a-1))
#define  cift(a) ((a%2)?(a-1):(a))
using namespace std;
typedef long long ll;

int N;
ll ar[MAXN];
ll  L[MAXN][2],R[MAXN][2];

int main()
{
	int i;
	
	scanf(" %d",&N);
	for(i=1;i<N;i++)
		scanf(" %I64d",&ar[i]);
	
	for(i=2;i<=N;i++)
	{
		L[i][0]=max(L[i-1][0],L[i-1][1])+tek(ar[i-1]);
		if(ar[i-1]>1)
			L[i][1]=L[i-1][1]+cift(ar[i-1]);
	}
	for(i=N-1;i;i--)
	{
		R[i][0]=max(R[i+1][0],R[i+1][1])+tek(ar[i]);
		if(ar[i]>1)
			R[i][1]=R[i+1][1]+cift(ar[i]);
	}
	
	ll t,res=0;
	for(i=1;i<=N;i++)
	{
		t=max(L[i][1]+R[i][1],L[i][1]+R[i][0]);
		t=max(t,L[i][0]+R[i][1]);
		res=max(res,t);
	}
	
	printf("%I64d\n",res);
	
	return 0;
}
