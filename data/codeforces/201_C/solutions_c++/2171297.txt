#include <cstdio>
#include <iostream>
#define MAXN 100010
using namespace std;
typedef long long ll;

ll N;
ll Res;
ll Nar[MAXN];
ll Sol[MAXN][2];
ll Sag[MAXN][2];

int main(){
	scanf("%I64d",&N);
	
	for(ll i=1;i<N;i++)
		scanf("%I64d",Nar+i);
		
	for(ll i=N-1;i>=1;i--){
		Sag[i][0]=(max(Sag[i+1][0],Sag[i+1][1])+Nar[i]-(1-(Nar[i]&1)))*(Nar[i]>0);
		Sag[i][1]=(Sag[i+1][1]+Nar[i]-(Nar[i]&1))*(Nar[i]>1);
	}
	
	for(ll i=2;i<=N;i++){
		Sol[i][0]=(max(Sol[i-1][0],Sol[i-1][1])+Nar[i-1]-(1-(Nar[i-1]&1)))*(Nar[i-1]>0);
		Sol[i][1]=(Sol[i-1][1]+Nar[i-1]-(Nar[i-1]&1))*(Nar[i-1]>1);
	}
	
	for(ll i=1;i<=N;i++){
		Res=max(Res,max(Sol[i][0],Sol[i][1])+Sag[i][1]);
		Res=max(Res,max(Sag[i][0],Sag[i][1])+Sol[i][1]);
	}
	
	printf("%I64d\n",Res);
	
	return 0;
}
