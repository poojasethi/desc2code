 #include <iostream>
#include <cstdio>
#include <cstring>
typedef long long ll;
using namespace std;
const int maxn = 100005;
ll l[maxn], r[maxn], sum[maxn];
ll a[maxn], q[maxn];
ll ans;
int n;
int main() {
    int i, j;
    cin >> n;
    for(i = 2; i <= n; i++) {
        scanf("%I64d", &a[i]);
        sum[i] = sum[i-1] + ((a[i]&1) ? a[i] : a[i]-1);
        if(a[i] == 1) continue;
        l[i] =  l[i-1] + a[i]/2*2;
    }
    for(i = n-1; i >= 1; i--) {
        if(a[i+1] == 1) continue;
        r[i] = r[i+1] + a[i+1]/2*2;
    }
    int s = 0,t = -1;
    ll mx = 0;
    for(i = 1; i <= n; i++) {
        mx = max(mx, l[i]-sum[i]);
        ans = max(ans, mx+r[i]+sum[i]);
    }
    cout << ans << endl;
    return 0;
}
