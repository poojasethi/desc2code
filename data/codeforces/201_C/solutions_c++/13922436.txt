#include<bits/stdc++.h>

using namespace std;

#define N 100005

long long A[N],L[N],R[N],L2[N],R2[N],n,i,mx;

int main(){

	cin >> n;

	for(i=1;i<n;i++)
		cin >> A[i];

	for(i=2;i<=n;i++)
		if(A[i-1] != 1)
			L[i] = L[i-1] + A[i-1] - (A[i-1]&1);

	for(i=n-1; i ;i--)
		if(A[i] != 1)
			R[i] = R[i+1] + A[i] - (A[i]&1);


	for(i=2;i<=n;i++)
		L2[i] = max(L[i] , L2[i-1] + A[i-1] - !(A[i-1]&1) + (!(A[i-1]&1) && L[i-1] == L2[i-1]));

	for(i=n-1; i ;i--)
		R2[i] = max(R[i] , R2[i+1] + A[i] - !(A[i]&1) + (!(A[i]&1) && R[i+1] == R2[i+1]));

	for(i=1;i<=n;i++)
		mx = max(mx , max(L[i] + R2[i] , R[i] + L2[i]));

	cout << mx;
	return 0;
}
