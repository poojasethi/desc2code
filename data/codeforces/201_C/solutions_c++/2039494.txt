#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long ll;
ll d[4][100001],a[100001],A;
int main(){
    int N;
    scanf("%d",&N);
    for(int i=1;i<N;i++)scanf("%I64d",a+i);
    for(int j=2;j<=N;j++){
        d[0][j]=d[0][j-1]+a[j-1]-!(a[j-1]&1);
        d[2][j]=a[j-1]<2?0:d[2][j-1]+a[j-1]/2*2;
        d[0][j]=max(d[0][j],d[2][j]);
    }
    for(int j=N-1;j>0;j--){
        d[1][j]=d[1][j+1]+a[j]-!(a[j]&1);
        d[3][j]=a[j]<2?0:d[3][j+1]+a[j]/2*2;
        d[1][j]=max(d[1][j],d[3][j]);
    }
    for(int i=1;i<=N;i++){
        A=max(A,max(d[0][i],d[1][i]));
        A=max(A,d[0][i]+d[3][i]);
        A=max(A,d[1][i]+d[2][i]);
    }
    printf("%I64d",A);
    return 0;
}
