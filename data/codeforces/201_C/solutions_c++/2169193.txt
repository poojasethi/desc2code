#include <stdio.h>
#include <algorithm>
using namespace std;

typedef long long int big;

int arr[100001];

big left[100001][2];
big right[100001][2];

main()
{
	int i,j,k;
	
	int n;
	
	scanf("%d",&n);
	
	for (i=1;i<n;i++)
		scanf("%d",arr+i);
	
	
	for (i=2;i<=n;i++)
	{
		if (arr[i-1]>1)
			left[i][0]=left[i-1][0]+arr[i-1]-arr[i-1]%2;
		if (arr[i-1])
			left[i][1]=max(left[i][0],left[i-1][1]+arr[i-1]-!(arr[i-1]%2));
	}
	
	for (i=n-1;i>=1;i--)
	{
		if (arr[i]>1)
			right[i][0]=right[i+1][0]+arr[i]-arr[i]%2;
		if (arr[i])
			right[i][1]=max(right[i][0],right[i+1][1]+arr[i]-!(arr[i]%2));
	}
	
	big ans=0,a;
	
	for (i=1;i<=n;i++)
	{
		a=max(right[i][0]+left[i][1],left[i][0]+right[i][1]);
		ans=max(ans,a);
	}
	printf("%I64d\n",ans);
}
