#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define MAXN 111111
int N,A;
long long Dp[3],Ans;
int main()
{
    scanf("%d",&N);
    for(int n=1;n<N;++n)
    {
        scanf("%d",&A);
        Dp[0]=max(Dp[0]+A/2*2,Dp[1]+A);
        Dp[1]=max(Dp[1],Dp[2])+(A-1)/2*2+1;
        Dp[2]=Dp[2]+A/2*2;
        Ans=max(Ans,Dp[0]);
        Ans=max(Ans,Dp[1]);
        Ans=max(Ans,Dp[2]);
        if(A==1)
        {
            Dp[0]=Dp[1];
            Dp[2]=0;
        }
//        printf("%d %I64d %I64d %I64d\n",n,Dp[0],Dp[1],Dp[2]);
    }
    printf("%I64d\n",Ans);
    return 0;
}
