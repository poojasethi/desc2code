#include<iostream>
#include<algorithm>
#include<set>
#include<queue>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#define N 100005
using namespace std;
#define ll __int64
ll l[N],r[N],a[N];
int main()
{
  int n,i;
  while(~scanf("%d",&n))
  {
   for(i=0;i<n-1;i++)
   scanf("%d",&a[i]);
   l[0]=0;
   for(i=1;i<n;i++)
   if(a[i-1]>=2)
   l[i]=l[i-1]+a[i-1]/2*2;
   else l[i]=0;
   r[n-1]=0;
   for(i=n-2;i>=0;i--)
   if(a[i]>=2)
   r[i]=r[i+1]+a[i]/2*2;
   else r[i]=0;
   ll ans,m=l[0]+r[0];
   ans=m;
   for(i=1;i<n;i++)
   {
    m-=r[i-1];
    m+=r[i];
    m+=(a[i-1]-1)/2*2+1;
    m=max(m,l[i]+r[i]);
    ans=max(ans,m);
   }
   printf("%I64d\n",ans);
  }
  return 0;
}
