#include <iostream>
#define FOE(i,s,e) for(int i=(s);i<=(int)(e);i++)
using namespace std;

const int N = 2036000;
long long a[N], dp[N][3], c, ans;
int n;

int main() {
    cin >> n; n--;
    FOE(i, 1, n) cin >> a[i];
    
    FOE(i, 1, n) {
        FOE(j, 0, 2) {
            c = a[i];
            if ((j+c)%2) c--;
            if (c) {
                FOE(k, 0, j) {
                    dp[i][j] = max(dp[i][j], dp[i-1][k] + c);
                }
            }
            ans = max(ans, dp[i][j]);
        }
    }
    
    cout << ans << endl;

    return 0;
}
