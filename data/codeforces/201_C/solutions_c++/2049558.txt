#include <cstdio>
#include <algorithm>
using namespace std;

#define rep(I,L,R) for(int I=L;I<=R;++I)

const int N = 100050;

typedef long long LL;

LL dp[N][3], res;
int n;

int main ()
{
	scanf("%d", &n);
	rep(i, 2, n)
	{
		int a, lev, lod;
		scanf("%d", &a);
		if (a & 1) lev = a - 1, lod = a; else lev = a, lod = a - 1;
		if (lev) dp[i][0] = dp[i - 1][0] + lev;
		if (lod) dp[i][1] = max(dp[i - 1][0], dp[i - 1][1]) + lod;
		if (lev) dp[i][2] = max(dp[i - 1][0], max(dp[i - 1][1], dp[i - 1][2])) + lev;
		res = max(res, max(dp[i][0], max(dp[i][1], dp[i][2])));
	}
#ifndef unix
	printf("%I64d\n", res);
#else
	printf("%lld\n", res);
#endif
	return 0;
}
