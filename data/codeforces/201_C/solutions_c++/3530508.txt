#include<cstdio>
#include<cstring>
#include<algorithm>
#define LL long long
using namespace std;

LL b[100001],d[100001][2],f[100001][2];

int main()
{
    int n;
    scanf("%d",&n);
    for(int i=2;i<=n;i++) scanf("%I64d",b+i);
    d[1][0]=d[1][1]=0;
    for(int i=2;i<=n;i++)
    {
        d[i][1]=b[i]>1?d[i-1][1]+b[i]/2*2:0;
        d[i][0]=b[i]&1?d[i-1][0]+b[i]:max(d[i-1][0]+b[i]-1,d[i][1]);
    }
    f[n][0]=f[n][1]=0;
    for(int i=n-1;i>=1;i--)
    {
        f[i][1]=b[i+1]>1?f[i+1][1]+b[i+1]/2*2:0;
        f[i][0]=b[i+1]&1?f[i+1][0]+b[i+1]:max(f[i+1][0]+b[i+1]-1,f[i][1]);
    }
    LL ans=0;
    for(int i=1;i<=n;i++)
    {
        ans=max(ans,max(d[i][1]+f[i][0],f[i][1]+d[i][0]));
    }
    printf("%I64d\n",ans);
    return 0;
}
