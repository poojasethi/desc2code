#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int a[200000];
long long g1[200000],g2[200000],f[200000];

long long max(long long x,long long y){
	return x>y?x:y;
}

int main(){
	int n;
	scanf("%d",&n);
	for (int i=1; i<n; ++i) scanf("%d",&a[i]);
	for (int i=1; i<n; ++i){
		int x=a[i];
		if (x & 1) --x;
		if (x==0) g1[i]=x;
		else g1[i]=g1[i-1]+x;
	}
	for (int i=n-1; i>=1; --i){
		int x=a[i];
		if (x & 1) --x;
		if (x==0) g2[i]=x;
		else g2[i]=g2[i+1]+x;
	}
	long long ans=0;
	for (int i=1; i<n; ++i){
		int x=a[i];
		if ((x & 1)==0) --x;
		f[i]=max(f[i-1]+x,g1[i]);
		ans=max(f[i]+g2[i+1],ans);
		//cout<<f[i]<<" "<<g2[i+1]<<endl;
	}
	cout<<ans<<endl;
}
