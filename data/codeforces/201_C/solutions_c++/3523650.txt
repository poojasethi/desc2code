#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define MAXN 111111
int N,A;
long long Dp[MAXN][3],Ans;
int main()
{
    scanf("%d",&N);
    for(int n=1;n<N;++n)
    {
        scanf("%d",&A);
        Dp[n][2]=Dp[n-1][2]+A/2*2;
        Dp[n][1]=max(Dp[n-1][1],Dp[n-1][2])+(A-1)/2*2+1;
        Dp[n][0]=max(Dp[n-1][0]+A/2*2,Dp[n-1][1]+A);
        Ans=max(Ans,Dp[n][0]);
        Ans=max(Ans,Dp[n][1]);
        Ans=max(Ans,Dp[n][2]);
        if(A==1)
        {
            Dp[n][0]=Dp[n][1];
            Dp[n][2]=0;
        }
//        printf("%d %I64d %I64d %I64d\n",n,Dp[n][0],Dp[n][1],Dp[n][2]);
    }
    printf("%I64d\n",Ans);
    return 0;
}
