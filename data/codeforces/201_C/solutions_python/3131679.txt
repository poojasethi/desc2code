n=input()
a=map(int,raw_input().split())
l=[[0,0] for _ in range(n)]
for i in range(1,n):
  x=a[i-1]
  p,q=l[i-1]
  l[i][0]=(x-1)/2*2+1+max(p,q)
  if x>1:l[i][1]=x/2*2+q
r=[[0,0] for _ in range(n)]
for i in range(n-2,-1,-1):
  x=a[i]
  p,q=r[i+1]
  r[i][0]=(x-1)/2*2+1+max(p,q)
  if x>1:r[i][1]=x/2*2+q
print max(map(lambda i:max(l[i][0]+r[i][0],l[i][0]+r[i][1],l[i][1]+r[i][0],l[i][1]+r[i][1]), range(n)))
