#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#define maxn 10010
using namespace std;
const int inf=1<<29;

int n;
int a[maxn],a1,d;

int main()
{
    while(scanf("%d",&n)!=EOF)
    {
       for(int i=0; i<n; i++)
       {
           scanf("%d",&a[i]);
       }
       sort(a,a+n);
       int ans=inf;
       for(int k=0; k<=20002; k++)
       {
          int max1=-inf,min1=inf;
          for(int i=0; i<n; i++)
          {
              min1=min(min1,a[i]-i*k);
              max1=max(max1,a[i]-i*k);
          }
          int x=(min1+max1)/2;
          if(ans>max(x-min1,max1-x))
          {
            ans=max(x-min1,max1-x);
            a1=x;
            d=k;
          }
       }
       printf("%d\n",ans);
       printf("%d %d\n",a1,d);
    }
    return 0;
}
