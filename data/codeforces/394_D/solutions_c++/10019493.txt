#include<cstdio>
#include<cstring>
#include<algorithm>
#define mem(name,value) memset(name,value,sizeof(name))
#define FOR(i,n) for(int i=1;i<=n;i++)
using namespace std;

const int maxn = 1000+10;
const int inf = 0x3f3f3f3f;
int num[maxn];

void solve(int n){
    int ans = inf;
    int d = 0, s = 0;
    for(int i=0;i<=20000;i++){
        int maxv = -inf, minv = inf;
        for(int j=0;j<n;j++){
            maxv = max(maxv,i * j - num[j]);
            minv = min(minv,i * j - num[j]);
        }
        int tmp = (maxv - minv + 1)/2;
        if(tmp < ans){
            ans = tmp;
            s = - (minv + tmp);
            d = i;
        }
    }
    printf("%d\n%d %d\n",ans,s,d);
}
int main(){
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&num[i]);
    sort(num,num+n);
    solve(n);
    return 0;
}
