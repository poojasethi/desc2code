#include<stdio.h>
#include<algorithm>

using namespace std;

int students[1010];

int main() {
  int n;
  scanf("%d", &n);

  int mine=20000;
  int ma0, md;

  int mins=20000,maxs=-20000;
  for (int i=0; i<n; i++) {
    scanf("%d", &students[i]);
    mins = min(mins, students[i]);
    maxs = max(maxs, students[i]);
  }
  sort(students, students+n);
  int bound = maxs-mins;
  for (int d=0; d<=bound; d++) {
    int a0 = students[0];
    int minx=0, maxx=0;
    int x;
    int e;
    for (int i=1; i<n; i++) {
      x = students[i]-(a0 + i*d);
      minx = min(minx,x);
      maxx = max(maxx,x);
    }
    e = (maxx-minx+1)/2;
    //printf("%d : %d\n", d, e);
    if (e < mine) {
      mine = e;
      md = d;
      ma0 = a0 + (maxx + minx)/2;
    }
  }

  printf("%d\n%d %d", mine, ma0, md);

  return 0;
}
