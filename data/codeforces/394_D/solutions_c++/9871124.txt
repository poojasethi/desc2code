#include<stdio.h>
#include<map>
#include<algorithm>
#define M 10000000
using namespace std;
typedef pair<int,int> P;
int a[1000];

int max(int a,int b){return a>b?a:b;}
int min(int a,int b){return a<b?a:b;}

P cross(P a,P b){
    return P(max(a.first,b.first),min(a.second,b.second));
}

P solve(int n,int w){
    for(int d=0;d<=20000;d++){
        P r=P(a[0]-w,a[0]+w);
        for(int i=1;i<n;i++){
            int t=a[i]-i*d;
            r=cross(r,P(t-w,t+w));
        }
        if(r.first<=r.second)return P(r.first,d);
    }
    return P(0,-1);
}

int main(){
    int n,i;
    scanf("%d",&n);
    for(i=0;i<n;i++)scanf("%d",&a[i]);
    sort(a,a+n);
    int l=-1,r=M;
    while(r-l>1){
        int mid=(l+r)/2;
        if(solve(n,mid).second!=-1)r=mid;
        else l=mid;
    }
    P ans=solve(n,r);
    printf("%d\n",r);
    printf("%d %d\n",ans.first,ans.second);
    return 0;
}
