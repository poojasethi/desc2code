#include <bits/stdc++.h>

using namespace std;

const int maxl = 1000 + 1;
const int oo = 2 * 1000 * 10 + 5;

int a [ maxl ];
int n;

int step, x, mx = 1000 * 1000 * 1000;

int val ( int x, int stp ){
	int res = 0;

	for ( int i = 0 ; i < n ; i++ )
		res = max ( res, max ( a [ i ] - x - stp * i, x + stp * i - a [ i ] ) );

	return res;
}

bool chk ( int i, int stp ){
	return (val ( i, stp ) < val ( i + 1, stp ));
}

int main (){
	cin >> n;
	for ( int i = 0 ; i < n ; i++ )
		cin >> a [ i ];
	sort ( a, a + n );

	for ( int stp = 0 ; stp < oo ; stp++ ){
		int lo = -oo, hi = oo;

		while ( hi - lo != 1 ){
			int mid = (hi + lo) >> 1;

			if ( chk ( mid, stp ) )
				hi = mid;
			else
				lo = mid;
		}

		int tmp = val ( hi, stp );
		if ( mx >= tmp ){
			step = stp;
			x = hi;
			mx = tmp;
		}
	}

	cout << mx << endl << x << " " << step << endl;
}
