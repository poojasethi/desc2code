#include <bits/stdc++.h>
using namespace std;
int step;
int low;
int L[1010],R[1010],A[1010];
int N;
int calc(int maxd)
{
	for(int i = 0;i<N;i++)
	{
		L[i] = A[i] - maxd;
		R[i] = A[i] + maxd;
	}
	int diffmax = (1<<30);
	for(int i = 1;i<N;i++)
	{
		diffmax = min(diffmax,R[i] - L[i-1]);
	}
	for(int i = 0;i<=diffmax;i++)
	{
		int left = L[0];
		int right = R[0];
		int F = 0;
		for(int j = 1;j<N;j++)
		{
			left = max(left,L[j] - j*i);
			right = min(right,R[j] - j*i);
			if(left>right)
			{
				F = 1;
				break;
			}
		}
		if(F==0)
		{
			low = left;
			step = i;
			return 1;
		}
	}
	return 0;
}
int main()
{
	cin>>N;
	for(int i = 0;i<N;i++){
		cin>>A[i];
	}
	sort(A,A+N);
	int l = 0;
	int r = 2e4;
	int res = -1;
	while(l<=r)
	{
		int m = (l+r)/2;
		if(calc(m))
		{
			r = m-1;
			res = m;
		}
		else
			l = m+1;
	}
	cout<<res<<endl;
	cout<<low<<" "<<step<<endl;
	return 0;
}
