#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=1050;
int n,A[maxn],ans;
int A0[maxn],dif;
bool judg(int x)
{
	int i,j,min0,max0,z=0;
	for(i=-2*x;i<=2*x;i++)
	{
		dif=A[n-1]+i-A[0];
		if(dif%(n-1)) continue;
		dif/=n-1;
		A0[0]=0,A0[n-1]=i;
		min0=min(0,i),max0=max(0,i);
		for(j=1;j<n-1;j++)
		{
			A0[j]=A[j-1]+A0[j-1]+dif-A[j];
			min0=min(min0,A0[j]);
			max0=max(max0,A0[j]);
		}
		if(max0-min0<=2*x) {z=1;break;}
	}
	if(z)
	{
		int k=min0+x;
		for(i=0;i<n;i++)
			A0[i]-=k;
	}
	return z;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,l=0,r=0,mid;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&A[i]),r=max(r,abs(A[i]));
	sort(A,A+n);
	while(r>l)
	{
		mid=(l+r)/2;
		if(judg(mid)) r=mid;
		else l=mid+1;
	}
	judg(l);
	printf("%d\n%d %d\n",l,A[0]+A0[0],dif);
	return 0;
}
