#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int64 a;
	int id;
} b[N];
int i,j,k,n,m,st,mi,x,a1,a2;
int a[N];
int64 s[N];
inline bool cc1(const ww &a,const ww &b) {
	return a.a<b.a;
}
int main() {
	scanf("%d",&n); x=mi=20000;
	For(i,1,n) scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	For(st,0,x) {
		if (-10000+st*(n-1)>30000) break;
		For(i,1,n) b[i].a=a[i]-st*(i-1),b[i].id=i;
		sort(b+1,b+n+1,cc1);
		int A=(b[1].a+b[n].a)/2;
		b[0].a=-x;
		For(i,1,n) if (b[i-1].a<A&&b[i].a>=A) {
			int64 S=max(A-b[1].a,b[n].a-A);
			int64 x=b[i].id,y=a[x]-(x-1)*st-(b[i].a-A);
			if (S<mi)
			mi=S,a1=st,a2=y;
		}
	}
	printf("%d\n%d %d\n",mi,a2,a1);
	return 0;
}
