#include <bits/stdc++.h>
using namespace std;

const int N=1005, INF=10000000;
int n, arr[N], pr1=INF, pr2, pr3;

int main()
{
   //freopen("r.txt","r",stdin);
   cin>>n;
   for(int i=0;i<n;i++)
     cin>>arr[i];
   sort(arr, arr+n);
   for(int i=0;i<=20000;i++)
   {
       int val=arr[0], mini=INF, maxi=-INF, cur=0;
       for(int j=0;j<n;j++)
       {
           mini=min(mini, arr[j]-val);
           maxi=max(maxi, arr[j]-val);
           val+=i;
       }
       int mid=(mini+maxi)/2;
       val=arr[0]+mid;
       for(int j=0;j<n;j++)
       {
           cur=max(cur, abs(arr[j]-val));
           val+=i;
       }
       if(cur<pr1)
       {
           pr1=cur;
           pr2=arr[0]+mid;
           pr3=i;
       }
   }
   cout<<pr1<<"\n"<<pr2<<" "<<pr3;

    return 0;
}
