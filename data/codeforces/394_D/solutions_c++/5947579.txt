#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN = 1010;
const int MAXD = 20001;
const int INF = 1 << 30;

#define rep(i, n) for (int i = 0, _##i = (n); i <_ ##i; ++i)

int main() {
  int N; cin >> N;
  int ar[N];
  rep(i, N) cin >> ar[i]; 
  sort(ar, ar + N);
  int ans = INF;
  int st, cd;
  rep(diff, MAXD) {
    int lo = INF;
    int hi = -INF;
    rep(i, N) {
      lo = min(i * diff - ar[i], lo); 
      hi = max(i * diff - ar[i], hi); 
    } 
    int cand = (hi - lo + 1) / 2;
    if (cand < ans) {
      ans = cand;
      st = cand - hi; 
      cd = diff;
    }
  }
  cout << ans << endl << st << ' ' << cd << endl;
  return 0; 
}
