#include <bits/stdc++.h>
using namespace std;
#define FREP(b) for(int i=0;i<b;i++)
#define REP(a,b,c) for(int a=b;a<c;a++)
#define asd(x) cout << "Here for " << #x << " :" <<x << endl;
#define asdf(x, y)  cout << "Here FOR " << #x << " :" << x << " " << #y  << " :" << y << endl;
typedef pair<int,int> ii;
typedef long long LL;
int n, A[1003];
int MAXS = 20002;
int ans = INT_MAX, a, D;
int main(){
    cin >> n;
    REP(i,0,n) cin >> A[i];
    sort(A, A+n);

    for(int d = 0; d <= MAXS; d++){
        // got to do something .. 
        int mini = INT_MAX, maxi = INT_MIN;

        for(int i = 0; i < n; i++){
            mini = min(mini, A[i] - i*d);
            maxi = max(maxi, A[i] - i*d);
        }
        int mid = (maxi + mini)/2;
        int here = max(maxi - mid, mid - mini);
        if(ans > here){
            ans = here, a = mid, D = d;
        }
    }
    cout << ans << endl;
    cout << a << " "  << D << endl;
    return 0;
}
