#include<cstdio>
#include<algorithm>
using namespace std;

const int inf=999999999;
const int mmax=1003;
int a[mmax];

int main()
{
    int n;
    while(scanf("%d",&n)!=EOF)
    {
        for(int i=0;i<n;i++)
            scanf("%d",&a[i]);
        sort(a,a+n);
        int ans=inf,st,d;
        for(int i=0;i<=20000;i++)
        {
            int l=0,r=0;
            for(int j=1;j<n;j++)
            {
                l=min(l,a[0]+j*i-a[j]);
                r=max(r,a[0]+j*i-a[j]);
            }
//            printf("%d %d\n",l,r);
            int dd=(r-l)/2;
            dd=max(r-l-dd,dd);
//            printf("%d\n",dd);
            if(dd<ans)
            {
                ans=dd;
                st=a[0]+(dd-r);
                d=i;
            }
        }
        printf("%d\n%d %d\n",ans,st,d);
    }
    return 0;
}
