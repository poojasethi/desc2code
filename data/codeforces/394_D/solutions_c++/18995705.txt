#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	vector <int> arr (n);
	for (int i = 0; i < n; i++) scanf("%d",&arr[i]);
	sort(arr.begin(),arr.end());
	int anw0 = -1;
	int anw1, anw2;
	for (int i = 0; i <= 20000; i++)
	{
		int mx = -arr[0];
		int mn = -arr[0];
		for (int j = 1; j < n; j++)
		{
			int td = j*i-arr[j];
			mx = max(mx,td);
			mn = min(mn,td);
		}
		int tr = 0;
		int tb = (mx+mn)/2;
		tb*= -1;
		for (int j = 0; j < n; j++) tr = max(abs((j*i-arr[j])+tb),tr);
		if (anw0 == -1 || anw0 > tr)
		{
			anw0 = tr;
			anw1 = tb;
			anw2 = i;
		}
	}
	printf("%d\n%d %d",anw0,anw1,anw2);
}
