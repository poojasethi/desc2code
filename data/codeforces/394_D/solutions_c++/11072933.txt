using namespace std;

typedef long long ll;
typedef pair <int, int> ii;
const int N=500005;

int a[N],n;
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++)scanf("%d",&a[i]);
    sort(a,a+n);
    if(n==2){
        cout<<"0\n"<<a[0]<<' '<<a[1]-a[0];
        return 0;
    }
    int total=10000;
    int mx=N,step,mn;
    for(int i=a[0]-total;i<=a[0]+total;i++){
        for(int j=0;j<=total;j++){
            int cur_mx=abs(a[0]-i);//cout<<i<<' ';
            //if(cur_mx>=mx)break;
            for(int k=1;k<n;k++){//cout<<i+k*j<<' ';
                cur_mx=max(cur_mx,abs(i+k*j-a[k]));
                if(cur_mx>=mx)break;
            }//cout<<cur_mx<<endl;
            if(cur_mx<mx)mx=cur_mx,mn=i,step=j;
        }
    }
    cout<<mx<<endl<<mn<<' '<<step;
    return 0;
}
