#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
const int MaxN = 1000 + 10, INF = 0x3f3f3f3f;
int n, arr[MaxN], MaxDerta, kd, kg;

void init()
{
	scanf("%d", &n);
	for(int i = 1; i <= n; ++i) scanf("%d", &arr[i]);
}

bool ok(int v)
{
	int low, upr;
	bool yy;
	for(int i = 0; i <= MaxDerta; ++i)
	{
		low = -INF;
		upr = INF;
		yy = true;
		for(int j = 1; j <= n; ++j)
		{
			low += i;
			upr += i;
			if(arr[j] - v > low) low = arr[j] - v;
			if(arr[j] + v < upr) upr = arr[j] + v;
			if(low > upr)
			{
				yy = false;
				break;
			}
		}
		if(yy)
		{
			kd = i;
			kg = upr - kd * (n - 1);
			return true;
		}
	}
	return false;
}

void solve()
{
	int st = -1, en = MaxDerta * n + 1, mid;
	while(st + 1 < en)
	{
		mid = (st + en) >> 1;
		if(ok(mid)) en = mid;
		else st = mid;
	}
	ok(en);
	printf("%d\n", en);
	printf("%d %d\n", kg, kd);
}

int main()
{
	init();
	sort(arr + 1, arr + n + 1);
	MaxDerta = arr[n] - arr[1];
	
	solve();
	
	return 0;
}
