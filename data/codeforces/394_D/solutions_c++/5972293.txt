#include <iostream>
#include <fstream>
#include <algorithm>
#include <cstring>
#include <cmath>
using namespace std;
#define N 1005
#define INF 0x3f3f3f3f

int num[N],n;

void solve()
{
    int ans=INF,start,dd;
    sort(num,num+n);
    for (int d=0;d<=20000;d++)
    {
        int up = -INF,down= INF;
        for (int i=0;i<n;i++)
        {
            up=max(up,i*d-num[i]);
            down=min(down,i*d-num[i]);
        }
        int res=(up-down+1)/2;
        if (ans>res){
            ans=res; start=-up+res; dd=d;
        }
    }

    printf("%d\n%d %d\n",ans,start,dd);
}

int main()
{
    //freopen("1.in","r",stdin);
    scanf("%d",&n);
    for (int i=0;i<n;i++) scanf("%d",&num[i]);
    solve();

    return 0;
}
