#include <iostream>
#include <stdlib.h>
#include <cstdio>
#include <algorithm>
#include <string>
#include <memory.h>
#define maxn 1010
#define BC 20001
using namespace std;
int a[maxn],Max,Min;
int main()
{
//    freopen("in.txt","r",stdin);
    int n;
    while(~scanf("%d",&n))
    {
        for(int i=0;i<n;i++)scanf("%d",a+i);
        sort(a,a+n);
        int ans=BC,ans2,gongcha;
        for(int i=0;i<=BC;i++)
        {
            Min=BC;Max=-BC;
            for(int j=0;j<n;j++)
            {
                Min=min(Min,a[j]-j*i);
                Max=max(Max,a[j]-j*i);
            }
            if((Max-Min+1)/2<ans)
            {
                ans=(Max-Min+1)/2;
                ans2=Min+ans;
                gongcha=i;
            }
        }
        cout<<ans<<"\n"<<ans2<<" "<<gongcha<<endl;
    }
}
