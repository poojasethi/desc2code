#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN = 1001, MAXW = 20001;
int a[MAXN], n;

int main()
{
	cin >> n;
	for (int i = 0; i < n; i++) cin >> a[i];
	sort(a, a + n);
	int best = 1e9, d = -1, st = -1e9;
	for (int i = 0; i < MAXW; i++)
	{
		int mx = -1e9, mn = 1e9;
		for (int j = 0; j < n; j++)
		{
			mx = max(i * j - a[j], mx);
			mn = min(i * j - a[j], mn);
		}
		int cst = -1e9;
		for (int j = 0; j < n; j++)
			if (i * j - a[j] == mn)
				cst = a[j] - (mx - mn) / 2 - i * j;
		int tmp = (mx - mn + 1) / 2;
		if (best > tmp) best = tmp, st = cst, d = i;
	}
	cout << best << endl << st << " " << d << endl;
	return 0;
}
