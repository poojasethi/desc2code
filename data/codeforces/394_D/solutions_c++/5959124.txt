#include <iostream>
#include <algorithm>

using namespace std;

int n, m[1005];
int x, y;

bool check(int mid){
    for(int i = m[0] - mid; i <= m[0] + mid; i++){
        int mind = 0, maxd = 100000;
        for(int j = 1; j < n; j++){
            mind = max(mind, (m[j] - mid - i - 1 + 123 * j) / j + - 122);
            maxd = min(maxd, (m[j] + mid - i + 123 * j) / j - 123);
        }
        if(mind <= maxd){
            x = i, y = mind;
            return 1;
        }
    }
  return 0;
}

int main(){
    cin >> n;
    
    for(int i = 0; i < n; i++) cin >> m[i];
    for(int i = 0; i < n; i++) m[i] += 100000;
    
    sort(m, m + n);
    
    int lower = -1;
    int higher = 20001;
    int middle = (lower + higher) / 2;

    while(lower < middle){
        if(check(middle))
            higher = middle;
        else
            lower = middle;
        middle = (lower + higher + 2) / 2 - 1;
    }
    check(higher);
    
    cout << higher << '\n' << x - 100000 << " " << y;

    return 0;
}
