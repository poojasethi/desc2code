#include <iostream>
#include <stdio.h>
#include <string>
#include <cstring>
#include <algorithm>
#include <cmath>
#define N 10009
#define INF 0x3f3f3f
using namespace std;

int a[N];
int n;

int main()
{
    while(~scanf("%d",&n))
    {
        for(int i=0;i<n;i++)
        scanf("%d",&a[i]);

        sort(a,a+n);
        int ans=INF;
        int a1;
        int d;

        for(int i=0;i<=20000;i++)
        {
            int x=INF,y=-INF;
            for(int j=0;j<n;j++)
            {
                x=min(x,i*j-a[j]);
                y=max(y,i*j-a[j]);
            }
            int ff=(y-x+1)/2;
            if(ans>ff)
            {
                ans=ff;
                a1=-x-ff;
                d=i;
            }
        }

        printf("%d\n%d %d\n",ans,a1,d);
    }
    return 0;
}
