#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#define N 111111
#define inf 0x7f7f7f7f
using namespace std;

int n;
int hei[N];
int ans = inf, ansa, ansd;

int main()
{
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i)
        scanf("%d", &hei[i]);
    sort(hei+1, hei+1+n);

    if (n == 1)
    {
        printf("0\n");
        printf("%d 0\n", hei[1]);
        return 0;
    }
    
    for (int i = 0; i*(n-1) <= 4000000; ++i)
    {
        int maxn = -inf, minn = inf;
        for (int j = 1; j <= n; ++j)
        {
            maxn = max(maxn, hei[j]);
            minn = min(minn, hei[j]);
        }

        int nowuse = (maxn-minn+1) / 2;
        if (nowuse < ans)
        {
            ans = nowuse;
            ansa = maxn - nowuse;
            ansd = i;
        }

        for (int j = 1; j <= n; ++j)
            hei[j] -= j-1;
    }

    printf("%d\n", ans);
    printf("%d %d\n", ansa, ansd);

    return 0;
}
