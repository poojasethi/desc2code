#include <stdio.h>
#include <algorithm>
#include <string.h>
using namespace std;
#define M 1010
#define MAX(a,b) a>b?a:b
#define MIN(a,b) a<b?a:b
int a[M], n, b[M];
int D=0, Sta=0;
bool able(int m){
	int lef , rig;
	for(int d=0; (n-1)*d<=40000; d++){
		lef=-50000, rig=50000;
		for(int i=0; i<n; i++){
			lef = MAX(lef, a[i]-m-i*d);
			rig = MIN(rig, a[i]+m-i*d);
		}
		if(lef>rig)continue;
		D = d;
		Sta = b[0] + lef;
		return 1;
	}
	return 0;
}
int main(){
	int i, j, k;
	scanf("%d", &n);
	for(i=0; i<n; i++)scanf("%d", a+i);
	sort(a, a+n);
	for(i=2; i<n; i++)if(a[i]!=2*a[i-1]-a[i-2])break;
	if(i==n){
		puts("0");
		printf("%d %d\n", a[0], a[1]-a[0]);
		return 0;
	}

	int d, sta;
	int mn=0, mx=10000, mid;
	while(mn<mx-1){
		mid=(mn+mx)/2;
		if(able(mid))mx=mid;
		else mn=mid;
	}
	printf("%d\n", mx);
	printf("%d %d\n", Sta, D);
	return 0;
}
