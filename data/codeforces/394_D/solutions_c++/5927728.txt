#include <iostream>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <utility>
#include <algorithm>
#include <cmath>
#define MAXN 1010
#define MAXH 10005

using namespace std;

int n,h[MAXN];
int best=2*MAXH,start,bestStep;

int main() {
	cin >> n;
	for (int i = 0; i < n; i++) cin >> h[i];
	sort(h,h+n);
	for (int step = 0; step <= 2*MAXH; step++) {
		int hi = -h[0], lo = -h[0];
		for (int i = 1; i < n; i++) {
			int diff = step*i - h[i];
			hi = max(hi,diff);
			lo = min(lo,diff);
		}
		int d = (hi - lo + 1) / 2;
		if (d < best) {
			best = d;
			bestStep = step;
			start = d - hi;
		}
	}
	cout << best << '\n' << start << ' ' << bestStep;
}
