#include <stdio.h>
#include <string.h>
#include <iostream>
#include <math.h>
#include <algorithm>
#include <vector>
#include <queue>
#include <map>
#define PI acos(-1.0)
#define M 1000005  //10^6
#define eps 1e-8
#define LL long long
#define moo 1000000007
using namespace std;
const int N=1005;
const int INF=999999999;
int n, num[N];

int main()
{
	while(~scanf("%d",&n))
    {
        for(int i=0;i<n;i++)
            scanf("%d",&num[i]);
        sort(num,num+n);
        int ans=INF;
        int s;
        int temp;
        int d;
        for(int step=0;step<20005;step++)
        {
            int Min=INF;
            int Max=-INF;
            for(int i=0;i<n;i++)
            {
                Min=min(Min,num[i]-step*i);
                Max=max(Max,num[i]-step*i);
            }
            temp=(Max-Min+1)/2;
            if(ans>temp)
            {
                ans=temp;
                s=Min+ans;
                d=step;
            }
        }
        printf("%d\n%d %d\n",ans,s,d);
    }
	return 0;
}
