#include <iostream>
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <vector>
#include <map>
#include <set>
#include <cstdio>
#include <ctime>

#define mp make_pair
#define pb push_back

using namespace std;

typedef long long ll;
typedef long double ld;

const int N = 5e5 + 7;
const ll INF = 20000;

int a[N];

int main()
{
    int n, ans = INF * INF, an1 = 0, an2 = 0;
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> a[i];
    sort(a, a + n);
    for (int i = 0; i <= INF; i++){
        int mi = INF * INF, ma = - INF * INF;
        for (int j = 0; j < n; j++){
            int t = a[j] - i * j;
            mi = min(t, mi), ma = max(t, ma);
        }
        int t = (ma - mi + 1) / 2;
        if (ans > t)
            ans = t, an1 = (ma + mi) / 2, an2 = i;
    }
    cout << ans << endl << an1 << " " << an2 << endl;
    return 0;
}
