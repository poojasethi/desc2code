#include <iostream>
#include <algorithm>
using namespace std;

int n, a[1005];
int x, y;

bool check(int mid){
  for(int i = a[0] - mid; i <= a[0] + mid; i++){
    int mind = 0, maxd = 100000;
    for(int j = 1; j < n; j++){
      // a[j] - mid <= i + j * d <= a[j] + mid
      mind = max(mind, (a[j] - mid - i - 1 + 123 * j) / j + - 122);
      maxd = min(maxd, (a[j] + mid - i + 123 * j) / j - 123);
    }
    if(mind <= maxd){
      x = i, y = mind;
      return 1;
    }
  }
  return 0;
}

int main(){
  cin >> n;
  for(int i = 0; i < n; i++) cin >> a[i];
  for(int i = 0; i < n; i++) a[i] += 100000;
  sort(a, a + n);
  int lo = -1, hi = 20001, mid = (lo + hi) / 2;
  while(lo < mid){
    if(check(mid)) hi = mid;
    else lo = mid;
    mid = (lo + hi + 2) / 2 - 1;
  }
  check(hi);
  cout << hi << '\n' << x - 100000 << " " << y;
}
