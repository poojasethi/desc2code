#include<cstdio>
#include<algorithm>
using namespace std;
const int MAXN=1005;
const int INF=0x3f3f3f3f;
int main()
{
    int n,a[MAXN],d,ans=INF,a1;
    scanf("%d",&n);
    for(int i=0; i<n; ++i)
        scanf("%d",&a[i]);
    sort(a,a+n);
    for(int i=0; i<=20000; ++i)
    {
        int mi=INF,ma=-INF;
        for(int j=0; j<n; ++j)
        {
            mi=min(mi,j*i-a[j]);
            ma=max(ma,j*i-a[j]);
        }
        int no=(ma-mi+1)/2;
        if(ans>no)
        {
            ans=no;
            a1=-mi-no;
            d=i;
        }
    }
    printf("%d\n%d %d",ans,a1,d);
}
