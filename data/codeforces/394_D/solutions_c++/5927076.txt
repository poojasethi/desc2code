#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n;
	cin >> n;

	int hei[1005];
	for (int i = 0; i < n; i++)
		cin >> hei[i];

	sort(hei, hei + n);

	int eat = (hei[n-1] - hei[0] + 1)/2;
	int start = (hei[0] + hei[n-1])/2;
	int diff = 0;

	for (int i = 1; i <= 20000; i++)
	{
		int mineat = hei[0], maxeat = hei[0];
		for (int j = 1; j < n; j++)
		{
			int e = hei[j] - j*i;
			if (e < mineat)
				mineat = e;
			else if (e > maxeat)
				maxeat = e;
		}

		int n = (maxeat - mineat + 1)/2;
		if (n < eat)
		{
			eat = n;
			start = (maxeat + mineat)/2;
			diff = i;
		}
	}

	cout << eat << "\n";
	cout << start << " " << diff << "\n";

	return 0;
}
