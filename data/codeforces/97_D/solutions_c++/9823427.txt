const int MAXN=160;

int N,M,K;
char Field[MAXN][MAXN];
bitset<MAXN*MAXN> Cur,Wall,Wall2; 
bitset<MAXN*MAXN> Final;
string cmd;

int main()
{
    ios::sync_with_stdio(false);
    cin>>N>>M>>K;
    rep(i,N)
    {
        rep(j,M)
        {
            cin>>Field[i][j];
            if(Field[i][j]=='#') Wall.set(i*M+j);
            else{
                Cur.set(i*M+j);
                if(Field[i][j]=='E'){
                    Final.set(i*M+j);
                }
            }
        }
    }
    Wall2=Cur;
    cin>>cmd;
    rep(i,K)
    {
        if(Cur==Final){
            cout<<i<<endl;
            return 0;
        }
        if(cmd[i]=='L'){
            Cur=(Cur>>1)&Wall2|(Wall<<1)&Cur;
        }
        if(cmd[i]=='R'){
            Cur=(Cur<<1)&Wall2|(Wall>>1)&Cur;
        }
        if(cmd[i]=='D'){
            Cur=(Cur<<M)&Wall2|(Wall>>M)&Cur;
        }
        if(cmd[i]=='U'){
            Cur=(Cur>>M)&Wall2|(Wall<<M)&Cur;
        }
    }
    if(Cur==Final) cout<<K<<endl;
    else cout<<-1<<endl;
    return 0;
}
