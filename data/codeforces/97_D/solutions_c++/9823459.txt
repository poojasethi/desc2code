		if(c!=d){
			char t=order[i];

				c=((c<<1)&a)|((b>>1)&c);
			else if(t=='D')
				c=((c<<m)&a)|((b>>m)&c);
			else if(t=='L')
				c=((c>>1)&a)|((b<<1)&c);

}
