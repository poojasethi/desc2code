# include <bits/stdc++.h>

# define ff first
# define ss second
# define mp(x,y) make_pair(x,y)
# define ma(x,y) (x>y ? x : y)
# define mi(x,y) (x<y ? x : y)
# define tr(i,c) for(__typeof((c).begin()) i=(c).begin(); i!=(c).end(); i++)
# define MAXN 2009

using namespace std;

typedef long long lld;

int n, m, k;
int A[MAXN][MAXN];

char S[MAXN];

pair<int,int> Q[MAXN];

int up[MAXN][MAXN];
int down[MAXN][MAXN];

int ans[MAXN];

priority_queue < pair<int,int>, vector< pair<int,int> >, greater< pair<int,int> > > lft, rght, tt;

bool change(int pos, int mx){
	lft = tt;
	rght = tt;
	
	for(int j=1; j<=m; j++){
		lft.push(mp(up[pos][j], j));
		rght.push(mp(down[pos][j], j));
		
		while(lft.top().ss <= j-mx)
			lft.pop();
		
		while(rght.top().ss <= j-mx)
			rght.pop();
		
		if(j >= mx  &&  lft.top().ff + rght.top().ff - 1 >= mx)
			return 1;
	}
	
	return 0;
}

int main(){
	scanf("%d %d %d",&n, &m, &k);
	
	for(int i=1; i<=n; i++){
		scanf("%s",S);
		
		for(int j=1; j<=m; j++)
			A[i][j] = (S[j-1] == '.');
	}
	
	for(int i=0; i<k; i++){
		scanf("%d %d",&Q[i].ff, &Q[i].ss);
		
		A[Q[i].ff][Q[i].ss] = 0;
	}
	
	for(int i=1; i<=n; i++)
		for(int j=1; j<=m; j++)
			up[i][j] = up[i-1][j] * A[i][j] + A[i][j];
	
	for(int i=n; i>0; i--)
		for(int j=1; j<=m; j++)
			down[i][j] = down[i+1][j] * A[i][j] + A[i][j];
	
	int cur = 0;
	
	for(int i=1; i<=n; i++)
		while(change(i, cur+1))
			cur++;
	
	for(int i=k-1; i>=0; i--){
		ans[i] = cur;
		
		int x = Q[i].ff;
		int y = Q[i].ss;
		
		A[x][y] = 1;
		
		for(int j=1; j<=n; j++)
			up[j][y] = up[j-1][y] * A[j][y] + A[j][y];
		
		for(int j=n; j>0; j--)
			down[j][y] = down[j+1][y] * A[j][y] + A[j][y];
		
		/*for(int i1=1; i1<=n; i1++){
			for(int j1=1; j1<=m; j1++)
				printf("%d ",A[i1][j1]);
			printf("\n");
		}*/
		
		while(change(x, cur+1))
			cur++;
	}
	
	for(int i=0; i<k; i++)
		printf("%d\n",ans[i]);
}
