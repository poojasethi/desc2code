#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main()
{
	bool l=0;
	string a;
	cin>>a;
	for(int i=0;i<=a.size()-2&&l==0;i++)
	if(a[i]%2==0)
	if(a[i]<a[a.size()-1])
	swap(a[i],a[a.size()-1]),l=1;
	for(int i=a.size()-2;i>=0&&l==0;i--)
	if(a[i]%2==0)
	swap(a[i],a[a.size()-1]),l=1;
	if(l==0)
	cout<<-1;
	else
	cout<<a;
}
