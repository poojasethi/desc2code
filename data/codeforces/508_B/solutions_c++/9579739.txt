#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main(){
	string s;
	int ans = -1;
	cin >> s;
	int n = s.length();
	for(int i = 0; i < n-1; i++){
		if((s[i]-'0') % 2 == 1)
			continue;
		ans = i;
		if(s[n-1] > s[i])
			break;
	}
	if(ans != -1)
		swap(s[ans], s[n-1]);
	cout << (ans == -1 ? "-1" : s) << endl;
	return 0;
}
