#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    string s;
    cin>>s;
    for(ll i=0;i<s.size()-1;i++) if ((s[i]%2==0)&&(s[i]<s[s.size()-1])) {swap(s[i],s[s.size()-1]);cout<<s;return 0; }
    for(ll i=s.size()-2;i>=0;i--) if (s[i]%2==0) {swap(s[i],s[s.size()-1]);cout<<s;return 0; }
    cout<<-1;
}
