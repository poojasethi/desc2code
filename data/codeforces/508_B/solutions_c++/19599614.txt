#include <iostream>
using namespace std;

int main() {
	// your code goes here
	
	string s;
	cin>>s;
	
	int n =s.length(),idx=-1;
	
	for(int i=0;i<n;i++)
	{
		if(((s[i]-'0')%2)==0)
		{
			idx = i;
			if(s[i]<s[n-1])
			break;
		}
		
	}
	
	if(idx==-1)
	cout<<idx;
	
	else
	{
		
		swap(s[idx],s[n-1]);
	 	cout<<s;
	}
	return 0;
}
