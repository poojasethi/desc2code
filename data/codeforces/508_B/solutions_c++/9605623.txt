#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int main()
{
    string s;
    cin>>s;
    int n=s.size(),t1=-1;
    for(int i=0;i<n-1;i++)
    {
        if((s[i]-'0')%2==0)
        if(s[i]<s[n-1]) {t1=i; break;}
        else t1=i;
    }
    if(t1==-1) cout<<-1;
    else
    {
        swap(s[t1],s[n-1]);
        cout<<s;
    }
}
