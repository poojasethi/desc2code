#include <iostream>

using namespace std;

int main()
{string a;long k=-1,i,n;
    cin>>a;n=a.size();char b=a[a.size()-1];
    for(i=0;i<n;i++)
    {
        if(a[i]%2==0)
         {if(a[i]<b)
        {k=i;
           break;
        }
       else  {k=i;}

    }}
    if(k==-1){cout<<-1;return 0;}

   else {swap(a[a.size()-1],a[k]);cout<<a;}
    return 0;
}
