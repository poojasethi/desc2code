#include<bits/stdc++.h>
using namespace std;
string a;
long long i,j,m,n,l;
int main()
{
    cin>>a;
    l=a.size();
    m=-1;
    for(i=0;i<l;i++)
    {
       if(a[i]%2==0)
       {
        m=i;
        if(a[i]<a[l-1])
            break;
       }
    }
    swap(a[m],a[l-1]);
    cout<<(m==-1?"-1":a);

}
