#include<bits/stdc++.h>
using namespace std;
int main()
{
string s;
long long int f,i;
cin>>s;
f=-1;
for(i=0;i<s.size()-1;i++)
{
    if((s[i]-'0')%2)
        continue;
    f=i;
    if(s[i]<s[s.size()-1])
    {
        break;
    }
}
if(f!=-1)
{
  swap(s[f],s[s.size()-1]);
   cout<<s;
}
else
    cout<<-1;
  return 0;
}
