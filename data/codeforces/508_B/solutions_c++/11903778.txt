#include <bits/stdc++.h>
using namespace std;
main()
{
    int x=-1;
    string s;
    cin >> s;
    for(int i=0;i<s.length()-1;i++)
    {
    if((s[i])%2==0)
    {
       x = i;
       if(s[s.length()-1]>s[i])
            break;
    }
    }
     swap(s[x],s[s.length()-1]);
     cout<<(x==-1? "-1" : s);
}
