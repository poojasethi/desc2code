#include <bits/stdc++.h>

using namespace std;
int a=-1;string s;
int main()
{
    cin>>s;
    for(int i=0;i<s.length()-1;i++){
      if((s[i]-'0')&1)continue;
      a=i;
      if(s[i]<s[s.length()-1])break;
    }
    
    if(a!=-1)swap(s[a],s[s.length()-1]);
    
    cout<<(a==-1?"-1":s);
    return 0;
}
