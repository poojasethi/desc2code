#include<iostream>
using namespace std;
string x;
main()
{
    
    cin>>x;
    int i,p=-1,l=x.length();
    for(i=0;i<l;i++)
    if((x[i]-'0')%2==0)
    {
    p=i;
    if(x[i]<x[l-1])break;
    }
    if(p==-1)cout<<-1<<endl;
    else
    {
        char t=x[p];x[p]=x[x.length()-1];x[x.length()-1]=t;
        cout<<x;
    }
}
