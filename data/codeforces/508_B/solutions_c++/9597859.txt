#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
string s;
cin>>s;
int l=s.length();
int temp=0,t;
for(int i=0;i<l-1;i++)
{
if((s[i]-'0')%2==0)
{
if(s[l-1]>s[i])
{
t=i;
temp=1;
break;
}
else
{
t=i;
temp=1;
}
}
}

if(temp)
{
swap(s[t],s[l-1]);
cout<<s<<endl;
}
else
cout<<-1<<endl;
return 0;
}
