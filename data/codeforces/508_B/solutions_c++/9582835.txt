#include <iostream>

using namespace std;

int main() {
	string s;
	cin >> s;
	int n = s.size();
	int mini = -1;
	for (int i = n - 1; i >= 0; --i) {
		if ((s[i] - '0') % 2 == 0) {
			if (mini == -1)
				mini = i;
			else if (s[i] < s[n - 1])
				mini = i;
		}
	}
	if (mini == -1) {
		cout << "-1\n";
	} else {
		swap(s[n - 1], s[mini]);
		cout << s << '\n';
	}
	return 0;
}
