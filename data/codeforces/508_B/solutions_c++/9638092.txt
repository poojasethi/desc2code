#include <bits/stdc++.h>
using namespace std;
const int N = 1000*1000;
char slo[N];

int main()
{
	scanf("%s",slo);
	int n = strlen(slo);
	int pos = -1;
	for (int i=n-2;i>=0;i--)
		if ( slo[i]%2 == 0 && ( slo[i] < slo[n-1] || pos == -1 ) )
			pos = i;
	if ( pos == -1 ) printf("-1");
	else
	{
		swap(slo[pos],slo[n-1]);
		printf("%s",slo);
	}
}
