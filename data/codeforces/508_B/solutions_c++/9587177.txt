#include<stdio.h>
#include<iostream>
#include<cstring>
using namespace std; 

int main()
{

string s;
cin>>s;
int max=-1,pos=-1;
for(int i=0;i<s.length()-1;i++)
{
int d=s[i]-'0';
if((d)%2==0)
{
pos=i,max=d;
if(d<(s[s.length()-1]-'0'))
break;
}

}
if(pos!=-1)
{
char d=s[s.length()-1];
s[s.length()-1]=s[pos];

}
else
printf("-1\n");
return 0;
}
