#include <bits/stdc++.h>
using namespace std;
char str[1000000];
int main()
{
	scanf("%s",str);
	int n=strlen(str);
	int yer=-1;
	for(int i=n-2;i>=0;i--)
	{
		if(str[i]%2==0&&(str[i]<str[n-1]||yer==-1))
		{
			yer=i;
		}
	}
	if(yer==-1)
	{
		printf("-1");
	}
	else
	{
		swap(str[yer],str[n-1]);
		printf("%s",str);
	}
	return 0;
}
