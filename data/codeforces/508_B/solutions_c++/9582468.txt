#include <bits/stdc++.h>
using namespace std;

char N[100001];

int main() {
	cin >> N;
	int d = strlen(N);
	int s = -1;
	for (int i = d - 1; i >= 0; --i) {
		if (N[i] % 2 == 0) {
			if (N[i] < N[d - 1]) s = i;
			else if (s == -1) s = i;
		}
	}
	if (s == -1) puts("-1");
	else {
		swap(N[s], N[d - 1]);
		puts(N);
	}
	return 0;
}
