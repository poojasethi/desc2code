#include<bits/stdc++.h>
using namespace std;

char s[100005];
int main()
{
	int i,l;

	scanf("%s",s);
	l=strlen(s);
	
	for(i=0;i<l;i++)
	{
		if(s[l-1]>s[i] && s[i]%2==0)break;
	}
	if(i==l)
	{
		for(i=l-2;i>=0;i--)
		{
			if(s[i]%2==0)break;
		}
	}
	if(i==-1)
	{
		printf("-1");
		return 0;
	}
	s[l-1]^=s[i]^=s[l-1]^=s[i];
	printf("%s",s);
	return 0;
}
