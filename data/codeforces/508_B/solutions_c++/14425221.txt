#include <bits/stdc++.h>

using namespace std;

int main()
{
    int i;
    int a=-1;
    string s;
    cin>>s;
    for( i=0;i<s.size();i++){
      if((s[i]-'0')&1)continue;
      a=i;
      if(s[i]<s[s.length()-1])break;
    }
    
    if(a!=-1)swap(s[a],s[s.length()-1]);
    
    cout<<(a==-1?"-1":s);
    return 0;
}
