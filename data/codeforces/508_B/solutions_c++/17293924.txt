#include <string.h>
using namespace std;
int main(){
	char arr[100005]={0};
	int i,f=15978,u,g;

	u=strlen(arr);
	for(i=0;i<u;i++){
		g=arr[i]-'0';
		if(!(g%2)){
			f=i;
			if(arr[i]<arr[u-1])
				break;}}
	if(f!=15978){
		swap(arr[f],arr[u-1]);
		puts(arr);}
	else
		puts("-1");
	return 0;
}
