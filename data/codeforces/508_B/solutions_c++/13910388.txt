#include <iostream>
using namespace std;

string s;
int id(-1);
int main() {
    cin >> s;
    for (int i = 0; i < s.length() - 1; i++) {
        if (s[i] % 2 == 0) {
                id = i;
            if (s[i] < s[s.length() - 1]) break;
        }
    }
    if (id == -1) cout << -1;
    else swap(s[id], s[s.length() - 1]), cout << s;
}
