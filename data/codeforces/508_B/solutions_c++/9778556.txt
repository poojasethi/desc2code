#include<stdio.h>
#include<string.h>
char a[100100];
main()
{
	int f=-1,l=0;
	char t;
	scanf("%s",a);
	l=strlen(a);
	for(int i=l-1;i>=0;i--)
	{
		if((a[i]-'0')%2)
		continue;
		if(f==-1)
		f=i;
		else if(a[i]<a[l-1])
		f=i;
	}
	if(f==-1)
	printf("-1");
	else
	{
		t=a[f];a[f]=a[l-1];a[l-1]=t;
		printf("%s",a);
	}
	
}
