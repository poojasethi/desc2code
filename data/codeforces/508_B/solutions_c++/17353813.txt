#include <bits/stdc++.h>
using namespace std;
int main()
{string s;
int k=-1;
cin>>s;
for (int i=0; i<s.size();i++)
 {if ((s[i]-'0')%2==0  && (s[i]<s[s.size()-1])) {swap(s[i],s[s.size()-1]);cout<<s;return 0;}
  else if ((s[i]-'0')%2==0  && (s[i]>s[s.size()-1])) {k=i;}
 }
 if(k!=-1) {swap(s[k],s[s.size()-1]);cout<<s; return 0;}
 else cout<<-1;
}
