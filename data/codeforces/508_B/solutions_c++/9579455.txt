#include <bits/stdc++.h>
using namespace std;

int main() {
	char a[100005];
	scanf("%s", a);
	int k = -1, n = strlen(a) - 1;
	for (int i = n; i >= 0; i--) {
		if (a[i] % 2 == 0) {
			if (k == -1 || a[i] < a[n])
				k = i;
		}
	}
	if (k == -1)
		printf("%d\n", k);
	else {
		char c = a[k];
		a[k] = a[n];
		a[n] = c;
		printf("%s\n", a);
	}
	return 0;
}
