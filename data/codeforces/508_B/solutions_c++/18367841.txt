#include <cstdio>
#include <cstring>
char c[100005],ch;
int n,ans=-1;
int main()
{
	int i;
	scanf("%s",c);
	n=strlen(c); ch=c[n-1];
	for (i=0;i<n-1;i++)
		if (!(c[i]&1)) 
		{
			if (c[i]<ch){ans=i; break;}
			else {ans=i;} 
		}
	if (ans<0) printf("-1");
	else
	{
		c[n-1]=c[ans]; c[ans]=ch;
		printf("%s",c);
	}
}
