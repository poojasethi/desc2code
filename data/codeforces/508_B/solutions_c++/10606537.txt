#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int main()
{
    string x;
    cin >> x;
    int c = -1;
    for (int i = 0; i < x.length();++i)
    {
        if ((x[i] - '0') % 2 == 0)
        if (c == -1 || x[c]> x[x.length() - 1])c = i;
    }
    if (c == -1)cout << -1;
    else{ swap(x[c], x[x.length() - 1]); cout << x; }
}
