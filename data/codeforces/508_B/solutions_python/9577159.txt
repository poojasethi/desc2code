n = map(int, list(raw_input()))
l = len(n)

for i in xrange(l - 1):
	if n[i] % 2 == 0 and n[i] < n[l - 1]:
		n[i], n[l - 1] = n[l - 1], n[i]
		print "".join(map(str, n))
		exit()

for i in xrange(l - 1, -1, -1):
	if n[i] % 2 == 0:
		n[i], n[l - 1] = n[l - 1], n[i]
		print "".join(map(str, n))
		exit()

print -1
