n = map(int,list(raw_input()))
for i in xrange(len(n)):
    if n[i]%2 == 0 and n[-1] > n[i]:
        n[i],n[-1] = n[-1],n[i]
        print ''.join(map(str,n))
        exit(0)
for i in xrange(len(n)-1,-1,-1):
    if n[i]%2 == 0 and n[-1] < n[i]:
        n[i],n[-1] = n[-1],n[i]
        print ''.join(map(str,n))
        exit(0)
print -1
