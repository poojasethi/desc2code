n = str(raw_input())
l = len(n)
for i in range(l):
	if int(n[i])%2 == 0:
		if int(n[i])<int(n[l-1]):
			print n[:i]+n[l-1]+n[i+1:l-1]+n[i]
			break
		else:
			m = i;
			for j in range(i+1,l):
				if int(n[j])%2 == 0 and int(n[j])<int(n[l-1]):
					m = j
					break
				if int(n[j])%2 == 0 and int(n[j])>int(n[l-1]):
					m = j
			print n[:m]+n[l-1]+n[m+1:l-1]+n[m]
			break
	if (i==l-1):
		print "-1"
