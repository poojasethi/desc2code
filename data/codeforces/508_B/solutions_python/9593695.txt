import sys
s = raw_input()
s = list(s)
lst = -1
for i in range(len(s)):
    if int(s[i]) % 2 == 0:
        if s[i] < s[-1]:
            tmp = s[i]
            s[i] = s[-1]
            s[-1] = tmp
            print ''.join(s)
            sys.exit(0)
        else:
            lst = i
