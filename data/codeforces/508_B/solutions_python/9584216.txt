INF = float('+inf')

n = raw_input()
t = map(int, list(n))
f = False

for i in xrange(len(n) - 1):
    if t[i] % 2 != 0:
        continue
    if t[i] < t[-1]:
        f = True
        break
if not f:
    for i in xrange(len(n) - 2, -1, -1):
        if t[i] % 2 == 0:
            f = True
            break
t[-1], t[i] = t[i], t[-1]
if not f:
    print -1
else:
    print ''.join(map(str, t))
