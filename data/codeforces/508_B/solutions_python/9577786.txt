s = list(raw_input())
l = len(s)

def val(c): return ord(c) - ord('0')
def even(c): return val(c)%2 == 0
e = val(s[-1])
p = -1

for i in xrange(l):
    if even(s[i]):
        if val(s[i]) < e:
            s[i], s[-1] = s[-1], s[i]
            print ''.join(s)
            exit(0) 
        p = i
if p == -1:
    print -1
else:
    s[p], s[-1] = s[-1], s[p]
    print ''.join(s)
