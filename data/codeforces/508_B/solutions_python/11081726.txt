n=list(raw_input())
for i in range(len(n)-1):
    if int(n[i]) % 2==0 and int(n[i]) < int(n[-1]):
        n[i],n[-1]=n[-1],n[i]
        print "".join(n)
        exit()
for i in range(len(n)-2,-1,-1):
    if int(n[i]) % 2==0 :
        n[i],n[-1]=n[-1],n[i]
        print "".join(n)
        exit()
print -1
