input = raw_input
n = input()
z = list(n)
k = 0
iid = -1
ind = len(n)-1
for i in n:
    if int(i) % 2 == 0:
        if i < n[ind]:
            iid = k
            break
        else:
            iid = k
    k += 1

if iid > -1:
    z[iid] = n[ind]
    z[ind] = n[iid]
    print "".join(z)
else:
    print -1
