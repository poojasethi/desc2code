s = raw_input()
l = len(s)
last = int(s[l-1])
x = None
for i in xrange(l-1) :
    if int(s[i]) % 2 == 0 :
        if int(s[i]) < last :
            x = i
            break
        else :
            x = i
if x == None :
    print '-1'
else :
    c = ''
    for j in xrange(l-1) :
        if j == x :
            c += str(last)
        else :
            c += s[j]
    c += s[x]
    print c
