n = raw_input()

max_even_i = None
last_n = int(n[len(n) - 1])
for i in range(len(n)):
    n_val = int(n[i])
    if n_val % 2 == 0:
        max_even_i = i
        if last_n > n_val:
            break

if max_even_i is not None:
    print n[:max_even_i] + n[len(n)-1] + n[max_even_i + 1: -1] + n[max_even_i]
else:
    print - 1
