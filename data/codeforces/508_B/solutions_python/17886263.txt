eveninteger=['2','4','6','8','0']
n=raw_input()
nlen=len(n)
evenindex=[i for i in xrange(nlen) if n[i] in eveninteger]
if len(evenindex)==0:
    print -1
else:
    for i in evenindex:
        if int(n[i])<int(n[-1]):
            print n[:i]+n[-1]+n[i+1:-1]+n[i]
            break
    else:
        for i in evenindex[::-1]:
            if int(n[i])>int(n[-1]):
                print n[:i]+n[-1]+n[i+1:-1]+n[i]
                break
