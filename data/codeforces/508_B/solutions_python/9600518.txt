max=0
swap=-1

for i in range(len(number)):
	if number[i]%2==0:
		if number[i]<number[len(number)-1]:
			swap=i
			break

if swap==-1:
	for i in range(len(number)):
		if number[len(number)-1-i]%2==0:
			swap=len(number)-1-i
			break

if swap==-1:
	number=[-1]

number[swap],number[len(number)-1]=number[len(number)-1],number[swap]

result=''

	result+=str(n)
