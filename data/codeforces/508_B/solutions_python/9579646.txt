n=raw_input()
n=list(n)
n=map(int,n)
for i in xrange(len(n)-1):
    if n[i]&1==0:
        if n[i]<n[-1]:
            n[i],n[-1]=n[-1],n[i]
            n=map(str,n)
            print ''.join(n)
            break
else:
    for i in xrange(len(n)-2,-1,-1):
        if n[i]&1==0:
            n[i],n[-1]=n[-1],n[i]
            n=map(str,n)
            print ''.join(n)
            break
    else:
        print -1
