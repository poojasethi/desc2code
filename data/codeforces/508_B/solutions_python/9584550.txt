s=raw_input()
s=list(s)
l=len(s)-1
no=int(s[l])
index=-1
for i in xrange(0,l):
    temp=int(s[i])
    if(temp%2==0):
        if temp<no:
            index=i
            break
        else:
            index=i
if index==-1:
    print -1
else:
    s[index],s[l]=s[l],s[index]
    s=''.join(s)
    print s
