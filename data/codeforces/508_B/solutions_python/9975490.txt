#!/usr/bin/python

def swap(s, a, b):
	lst=list(s)
	lst[a],lst[b]=lst[b],lst[a]
	print ''.join(lst)

def main():
	even, n=-1, raw_input()
	for i in range(len(n)):
		if int(n[i])%2==0 and int(n[i])<int(n[-1]):
			swap(n,i,-1)
			exit()
		elif int(n[i])%2==0:
			even=i
	if even==-1:
		print -1
 	else:
 		swap(n,even,-1)

if __name__=='__main__':
	main()
