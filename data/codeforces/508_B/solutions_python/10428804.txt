S = raw_input()

L = []
for i in xrange(len(S)):
    if S[i] in '02468':
        L.append(i)

if L == []:
    print -1
else:
    last = S[-1]
    for k in L:
        if S[k] < last:
            print S[:k]+last+S[k+1:-1]+S[k]
            break
            
    else:
        print S[:k]+last+S[k+1:-1]+S[k]
