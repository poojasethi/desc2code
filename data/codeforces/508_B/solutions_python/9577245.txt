def readlist(f): return map(f, raw_input().split())
def readint(): return int(raw_input())
def printlist(l): print ' '.join(map(str, l))

n = raw_input()
l = ord(n[-1]) - ord('0')
j = -1
for i in range(len(n)-1):
  d = ord(n[i]) - ord('0')
  if d%2 == 1: continue
  j = i
  if l > d: break
if j == -1: print -1
else: print (n[:j] + n[-1] + n[j+1:-1] + n[j])
