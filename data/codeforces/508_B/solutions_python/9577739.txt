
num = list(raw_input())
n = len(num) 

pivot = num[-1]

p = -1

for i in range(n-1): 
    if int(num[i]) % 2 == 0:
        if num[i] < pivot: 
            p = i
            break
        else: 
            p = i

if p >= 0:
    num[p], num[-1] = num[-1], num[p]
    print ''.join(num)
else: 
    print -1
