n = map(int, list(raw_input()))
nn = n[-1]
for i in range(len(n)):
    if n[i] % 2 == 0 and n[i] < nn:
        n[-1] = n[i]
        n[i] = nn
        print ''.join([str(x) for x in n])
        exit()
for i in range(len(n)-1, -1, -1):
    if n[i] % 2 == 0:
        n[-1] = n[i]
        n[i] = nn
        print ''.join([str(x) for x in n])
        exit()
print -1
