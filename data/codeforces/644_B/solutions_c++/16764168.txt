#include "stdio.h"
#include "string.h"
int t[200000],d[200000],q[200000];
long long tt,a[200000];
int main(){
    int i,n,dd,l=0,r=0;
    scanf("%d%d",&n,&dd);
    for(i=0;i<n;i++)
        scanf("%d%d",&t[i],&d[i]);
    tt=0;
    i=0;
    while(l<r||i<n){
        while(l<r&&(i==n||tt+d[q[l]]<=t[i])){
            tt+=d[q[l]];
            a[q[l++]]=tt;
        }
        if(l==r&&tt<=t[i])tt=t[i];
        while(i<n&&(l==r||t[i]<tt+d[q[l]])){
            if(r-l>dd)
                 a[i++]=-1;
            else q[r++]=i++;
        }
    }
    for(i=0;i<n-1;i++)
        printf("%lld ",a[i]);
    printf("%lld\n",a[i]);
    return 0;
}
