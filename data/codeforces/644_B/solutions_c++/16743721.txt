#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<vector>
#include<queue>
using namespace std;

typedef long long ll;
int c[200010],d[200010],p,q,n,b;
ll t[200010];

int main(){
	scanf("%d%d",&n,&b);
	for(int i=1;i<=n;i++) scanf("%I64d%d",&t[i],&d[i]);
	p=1; q=0;
	for(int i=1;i<=n;i++){
		while(p<=q&&t[c[p]]<=t[i]) p++;
		if(q-p+1>b) printf("-1 ");
		else {
			t[i]=max(t[i],t[c[q]])+d[i];
			c[++q]=i;
			printf("%I64d ",t[i]);
		}
	}
	return 0;
}
