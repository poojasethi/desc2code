#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 201000
using namespace std;
long long ans[maxn],que[maxn],t[maxn],d[maxn];
long long n,m,T,L,R,pas;
int main() {
	scanf("%I64d %I64d",&n,&m);
	T = 0; pas = 0;
	for (long long i = 1; i <= n; i++) {
		scanf("%I64d %I64d",&t[i],&d[i]);
		for (;L < R && max(T,t[que[L + 1]]) <= t[i]; L++) {
			T = max(T,t[que[L + 1]]) + d[que[L + 1]];
			ans[que[L + 1]] = T;
		}
		if (R - L < m) que[++R] = i;
		else ans[i] = -1;
	}
	for (; L < R; L++) {
		T = max(T,t[que[L + 1]]) + d[que[L + 1]];
		ans[que[L + 1]] = T;
	}
	for (long long i = 1; i <= n; i++) printf("%I64d ",ans[i]);
	return 0;
}
