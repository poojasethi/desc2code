#include <stdio.h>
#include <queue>
using namespace std;
int main(){
	queue<long long> q;
	int n, b;
	int t, x;
	scanf("%d%d", &n, &b);
	while (n--){
		scanf("%d%d", &t, &x);
		while (!q.empty() && q.front() <= t) q.pop();
		if (q.size()>b) printf("-1 ");
		else if (q.empty()){
			printf("%d ", t + x);
			q.push(t + x);
		}
		else{
			printf("%I64d ", q.back() + x);
			q.push(q.back() + x);
		}
	}
	puts("");
	return 0;
}
