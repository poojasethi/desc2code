#include <stdio.h>
using namespace std;
queue <long long> q;
int main()
{//freopen("i.txt","r",stdin);

long long b; cin>>b;

 for (int i=1;i<=n;i++)
 {long long t,d; cin>>t>>d;
  while (q.front()<=t && q.size()) q.pop();
   if (q.size()>b) {cout<<-1<<" "; continue;}

 }
