#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[200010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n,b;
	cin>>n>>b;
	memset(a,0,sizeof(a));
	ll int i,j=0,cur=0,old,t,d;
	bool flag;
	for(i=0;i<n;i++)
	{
		flag=0;
		cin>>t>>d;
		if(t<cur)
		{
			if(a[j]<=t)
			{
				a[j]=cur;
				cur=a[j]+d;
				flag=1;
			}
			else
			{
				old=j;
				j=(j+1)%b;
				if(a[j]<=t)
				{
					a[j]=cur;
					cur=a[j]+d;
					flag=1;
				}
			}
			if(!flag)
			j=old;
		}
		else
		{
			cur=t+d;
			flag=1;
		}
		if(flag)
		cout<<cur<<" ";
		else
		cout<<-1<<" ";
	}
	cout<<endl;
	return 0;
}
