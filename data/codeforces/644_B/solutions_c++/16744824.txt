#include<cstdio>
using namespace std;
const int N=200005;
int i,n,b,H=1,T=1,x,t[N],d[N],q[N];
long long now,ans[N];
int main()
{
	scanf("%d%d",&n,&b);
	scanf("%d%d",&t[1],&d[1]);
	ans[1]=now=t[1]+d[1];
	for(i=2;i<=n;i++)
	{
		scanf("%d%d",&t[i],&d[i]);
		while(t[i]>=now&&H<T)x=q[H++],now=ans[x]=now+d[x];
		if(T-H>=b){ans[i]=-1;continue;}
		if(t[i]<now)q[T++]=i;else ans[i]=now=t[i]+d[i];
	}
	while(H<T)
		x=q[H++],now=ans[x]=now+d[x];
	for(i=1;i<=n;i++)
		printf("%I64d ",ans[i]);
}
