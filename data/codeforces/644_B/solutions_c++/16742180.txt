#include <stdio.h>
#include <iostream>
#include <queue>
using namespace std;
long long x,y,i,a,b,d[200300],t=0;
queue <int>p1;
queue <int>p2;
int main(){
	scanf("%d%d",&x,&y);
	for(i=0;i<x;i++){
		scanf("%d%d",&a,&b);
		while(a>=p2.front() && p1.size()>=1){
			p1.pop();
			p2.pop();
		}
		if(p1.size()==0)t=a;
		if(p1.size()<=y){
			p1.push(a);
			p2.push(t+b);
			t+=b;
			d[i]=t;
		}
		else{
			d[i]=-1;
		}
	}
	for(i=0;i<x;i++)cout<<d[i]<<" ";
    return 0;
}
