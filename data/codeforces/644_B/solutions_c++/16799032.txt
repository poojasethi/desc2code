#include<iostream>
#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
int n,b;
queue <long long> q;
int main()
{
	scanf("%d%d",&n,&b);
	for(int i=1;i<=n;i++)
	{
		long long t,d;
		scanf("%I64d%I64d",&t,&d);
		while(!q.empty() && q.front()<=t)q.pop();
		if(q.size()>b)
		{
			puts("-1");
		}
		else
		{
			long long ans;
			if(q.empty())ans=t;else ans=q.back();
			ans+=d;
			printf("%I64d ",ans);
			q.push(ans);
		}
	}
	puts("");
	return 0;
}
