#include <bits/stdc++.h>
using namespace std;

typedef long long lint;

int n,b;

int main(){
	
	scanf("%d%d",&n,&b);
	b++;
	queue < lint > q;
	lint last = 0;
	for( int i = 1,t,d ; i <= n ; i++ ){
		scanf("%d%d",&t,&d);
		last = max(last , 1ll*t);
		while( !q.empty() and q.front() <= t ) q.pop();
		if( (int)q.size() == b )
			printf("-1 ");
		else {
			last = last + d;
			printf("%lld ",last);
			q.push(last);
		}
	}
	
	return 0;
}
