#include <cstdio>
#include <cstring>
#include <queue>

using namespace std;

#define MAXN 200000

int n,b,t[MAXN],d[MAXN];
long long e[MAXN];

int main(){
  scanf("%d %d",&n,&b);

  for(int i = 0;i < n;++i){
    scanf("%d %d",&t[i],&d[i]);
  }

  memset(e,-1,sizeof e);

  deque<int> Q;
  int qlen = 0;

  for(int i = 0;i < n;++i){
    while(!Q.empty() && e[Q.front()] <= t[i]){
      Q.pop_front();
      --qlen;
    }

    if(qlen < b + 1){
      if(qlen == 0) e[i] = t[i] + d[i];
      else e[i] = e[Q.back()] + d[i];
      Q.push_back(i);
      ++qlen;
    }
  }

  for(int i = 0;i < n;++i)
    printf("%I64d ",e[i]);
  printf("\n");

  return 0;
}
