#include <iostream>
#include <stdio.h>

#define MAXN 200100 

using namespace std;

long long q[MAXN], iq, nq, m, n;

int main() {
    scanf("%I64d%I64d", &m, &n);
    iq = 1; 
    for ( int i = 1; i <= m; i++ ) {
        long long s, t;
        scanf("%I64d%I64d", &s, &t);

        while ((iq <= nq) && (q[iq] <= s))
            iq++;

        if (nq - iq <  n) {
            nq++;
            if ( nq - 1 >= iq )
                q[nq] = q[nq-1] + t;
            else
                q[nq] = s + t;
            printf("%I64d ", q[nq]);
        } else {
            printf("-1 ");
        }
    }
    return 0;
}
