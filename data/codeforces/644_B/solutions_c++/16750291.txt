#include <bits/stdc++.h>

using namespace std;

long long q[200200];

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int n, b;
    cin >> n >> b;
    int l = 0, r = 0;
    for (int i = 0; i < n; ++i) {
        int t, d;
        cin >> t >> d;
        while (l < r && q[l] <= t) ++l;
        if (r - l > b) cout << -1 << ' ';
        else {
            if (l == r) q[r] = t + d;
            else q[r] = q[r - 1] + d;
            cout << q[r] << ' ';
            r++;
        }
    }

    return 0;
}
