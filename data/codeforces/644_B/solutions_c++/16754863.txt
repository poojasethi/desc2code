#include <cstdio>
#include <queue>

using std::queue;

int main() {
  int n, b;
  scanf("%d%d", &n, &b);
  queue<long long> q;
  long long time = 0;
  for (int i = 0; i < n; ++i) {
    int x, y;
    scanf("%d%d", &x, &y);
    // check queue first
    while (!q.empty() && x >= q.front()) {
      q.pop();
    }
    // execute immediately
    if (time <= x) {
      time = x + y;
      printf("%I64d ", time);
    } else {
      if (q.size() < b) {
        q.push(time);
        time += y;
        printf("%I64d ", time);
      } else {
        printf("-1 ");
      }
    }
  }
  printf("\n");
}
