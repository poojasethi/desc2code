#include<bits/stdc++.h>

int n, b;
long long f, s[200005], t[200005], d[200005];
std::queue<int> q;

int main() {
	scanf("%d %d", &n, &b);
	for (int i = 0; i < n; i++) scanf("%d %d", &t[i], &d[i]);
	for (int i = 0; i < n; i++) {
		while (q.size() && s[q.front()] <= t[i]) q.pop();
		f = std::max(f, t[i]);
		if (q.size() < b) {
			s[i] = f;
			printf("%I64d", f = s[i] + d[i]);
			q.push(i);
		}
		else printf("-1");
		printf("%c", i == n - 1 ? '\n' : ' ');
	}
	scanf("\n");
	return 0;
}
