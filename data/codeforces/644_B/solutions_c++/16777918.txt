#include <bits/stdc++.h>
#define f first
#define s second
#define mp make_pair
using namespace std;

long long n, b, T;
queue<pair<long long, long long > > q;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin >> n >> b;
    for(int i = 1; i <= n; ++i)
    {
        long long t, d;
        cin >> t >> d;
        while(q.size()>0 && q.front().f <= t)
            q.pop();
        if(q.size() < b)
        {
            q.push(mp(max(T, t), max(T, t) + d));
            cout << max(T, t) + d << ' ';
            T = max(T, t) + d;
        }
        else
            cout << -1 << ' ';
    }
}
