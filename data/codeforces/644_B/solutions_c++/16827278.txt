#include <stdio.h>

long long end[200010];

int main()
{
    int b,d,i,n,t,num,first,last;
    scanf("%d %d",&n,&b);
    scanf("%d %d",&t,&d);
    first=0; last=1; num=1;
    end[0]=t+d;
    printf("%I64d ",end[0]);
    for (i=1;i<n;i++)
    {
        scanf("%d %d",&t,&d);
        while (t>=end[first] && first<i)
        {
            first++;
            num--;
        }
        if (num>b)
        {
            printf("-1 ");
            continue;
        }
        if (first==i)
            end[last]=t+d;
        else end[last]=end[last-1]+d;
        printf("%I64d ",end[last]);
        num++; last++;
    }
    printf("\n");
}
