#include<cstdio>
using namespace std;
typedef long long ll;
const int N = 200000;
ll q[N];
int f,r;
int main()
{
	int n,l; scanf("%d%d",&n,&l);
	while(n--)
	{
		ll a,b;
		scanf("%I64d%I64d",&a,&b);
		while(r-f>0 && a>=q[f]) f++;
		if(r-f==l+1) {
			printf("-1 ");
		}
		else {
			if(r-f==0) q[r] = a+b;
			else q[r] = q[r-1]+b;
			printf("%I64d ",q[r++]);
		}
	}
}
