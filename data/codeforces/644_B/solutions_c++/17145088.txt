#include<cstdio>  
#include<iostream>  
#include<algorithm>  
#include<cstdlib>  
#include<cstring>
#include<string>
#include<climits>
#include<cmath>

using namespace std;

int n,m;
struct data
{
	long long x,y;
}a[200010];

int main()
{
	scanf("%d%d",&n,&m);
	long long x,y;
	int l=0,r=0;
	for (int i=1;i<=n;i++)
	{
		scanf("%I64d%I64d",&x,&y);
		while (a[l].y<x) 
		{
			l++;
			if (l>r) break;
		}
		if (r-l+1<=m)
		{
			r++;
			if (r==0) a[r].x=x;else a[r].x=max(a[r-1].y+1,x);
			a[r].y=a[r].x+y-1;
			if (i<n) printf("%I64d ",a[r].y+1);else printf("%I64d\n",a[r].y+1);
		}
		else 
		{
			if (i<n) printf("-1 ");else printf("-1\n");
		}
	}
	return 0;
}
