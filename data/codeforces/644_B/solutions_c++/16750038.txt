#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
int n, m;
long long now, t[200010], d[200010], ans[200010];
queue<int> q;

int main()
{
	cin >> n >> m;
	for (int i = 1; i <= n; i++)
	{
		scanf("%I64d%I64d", &t[i], &d[i]);
		while (now <= t[i] && q.size())
		{
			int j = q.front();
			q.pop();
			now = max(now, t[j]);
			now += d[j];
			ans[j] = now;
		}
		if (q.size() < m) q.push(i); else ans[i] = -1;
	}
	while (q.size())
	{
		int j = q.front();
		q.pop();
		now = max(now, t[j]);
		now += d[j];
		ans[j] = now;
	}
	for (int i = 1; i < n; i++) printf("%I64d ", ans[i]);
	printf("%I64d\n", ans[n]);
}
