#include <iostream>
#include <queue>
#define ll long long 
using namespace std;
const int MX = 2e5+1;

int main() {
	ll n, b;
	cin >> n >> b;
	queue<ll> q;
	ll ans[MX], fin = 0;
	for (int i = 0; i < n; i++) {
		ll t, l;
		cin >> t >> l;
		while (!q.empty() && q.front() <= t) {
			q.pop();
		}
		if (q.size() <= b) {
			if (fin < t) fin = t;
			q.push(fin+l);
			ans[i] = fin+l;
			fin = fin+l;
		}
		else {
			ans[i] = -1;
		}
	}
	for (int i = 0; i < n; i++) {
		cout << ans[i] << " ";
	}
	cout << '\n';
	return 0;
}
