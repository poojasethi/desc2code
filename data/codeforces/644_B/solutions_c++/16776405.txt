#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>

using namespace std;

struct data
{
    long long t,d;
}a[300005];

int n,b;
long long e[300005];

int main()
{
    memset(e,0,sizeof(e));
    int h=0, r=0;
    int cnt=0;
    scanf("%d%d",&n,&b);
    for(int i=0;i<n;i++) {scanf("%d%d",&a[i].t,&a[i].d);}
    long long t=0;
    while(h<n)
    {
        for(;e[h]==-1;h++,cnt--);
        if(h==r) r++;

        t=t<a[h].t?a[h].t:t;
        t=t+a[h].d;
        e[h]=t;
        for(;r<n&&a[r].t<t;r++)
            if(r-h-cnt>b) {e[r]=-1;cnt++;}

        h++;
    }

    for(int i=0;i<n-1;i++) printf("%I64d ",e[i]);
    printf("%I64d\n",e[n-1]);
}
