#include <iostream>
#include <queue>

#define ll long long 
using namespace std;

deque<ll> que;

ll last, b, n, st, lg;

void process(ll st, ll lg)
{
	while (que.size() && que.front() <= st)
		que.pop_front();

	if (que.size() == b +1)
	{
		cout << -1 << ' ';
		return;
	}

	last = max(st, que.size() ? que.back() : 0);

	cout << last + lg<<' ';
	
	que.push_back(last + lg);

}

int main()
{

	cin >> n >> b;

	for (ll i = 1; i <= n; i++)
	{
		cin >> st >> lg;
		process(st, lg);
	}

    return 0;
}
