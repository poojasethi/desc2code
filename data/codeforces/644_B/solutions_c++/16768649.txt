#include <stdio.h>
#include <iostream>
using namespace std;
const long long NMAX = 200010;

long long print[NMAX+2];
int in[NMAX+2][2];
int main()
{
	int C, D;
	long long q;
	long long time, nt;
	long long mi;
	scanf("%d%d",&C,&D);
	for(long long i=1;i<=C;i++) {
		scanf("%d%d",&in[i][0], &in[i][1]);
	}
	time = 0;
	mi = q = 0;
	for(long long i=1;i<=C;i++) {
		if( print[i] == -1 ){ printf("-1 "); continue; }

		time = time > in[i][0] ? time : in[i][0];
		time = time + in[i][1];
		cout << time << " ";
		mi-=1;

		while( in[q][0] < time && q <= C ) {
			if( mi <= D ) mi += 1;
			else print[q] = -1;
			q += 1;
		}
	}
	printf("\n");
	return 0;
}
