#include <bits/stdc++.h>
using namespace std;
#define M 200005

long long n, b, f[M];
queue <long long> q;

int main() {
  scanf("%d%d", &n, &b);
  for (int i = 0, t, d; i < n; i++) {
    scanf("%d%d", &t, &d);
    while (q.size() && q.front() <= t) {
      q.pop();
    }
    if (q.empty()) {
      f[i] = t + d;
      q.push(f[i]);
    } else if (q.size() <= b) {
      f[i] = q.back() + d;
      q.push(f[i]);
    } else {
      f[i] = -1;
    }
  }
  for (int i = 0; i < n; i++) {
    printf("%I64d%c", i && f[i] == f[i-1] ? -1 : f[i], i + 1 < n ? ' ' : '\n');
  }
  return 0;
}
