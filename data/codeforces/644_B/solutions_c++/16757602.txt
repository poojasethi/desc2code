#include <bits/stdc++.h>
#define K push_back
#define L long long int
using namespace std;
L n, b, i, t, d;
vector < L > v;
int main()
{
    cin >> n >> b;
    for( i = 1; i < b + 2; i++ )v.K(0);
    for( i = 1; i <= n; i++ ){
        cin >> t >> d;
        if( v[v.size() - b - 1] > t )cout << "-1 ";
        else{
            v.K(max(v.back(), t) + d);
            cout << v.back() << " ";
        }
    }
}
