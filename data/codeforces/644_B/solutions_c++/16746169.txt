#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

#define M 200005

int n,sz;
int l=1,r;
LL res[M],q[M],qi[M],ct;

void POP(int &L){
  ct+=q[L];
  res[qi[L]]=ct;
  L++;
  return;
}

int main(){
  scanf("%d %d",&n,&sz);
  FI(i,1,n){
    int a,b;
    scanf("%d %d",&a,&b);
    while(l<=r && ct+q[l]<=a) POP(l);
    if(r-l==sz) res[i]=-1;
    else{
      if(l>r) ct=a;
      q[++r]=b;
      qi[r]=i;
    }
  }
  while(l<=r) POP(l);
  FI(i,1,n) printf("%lld%c",res[i],i==n?'\n':' ');
  return 0;
}
