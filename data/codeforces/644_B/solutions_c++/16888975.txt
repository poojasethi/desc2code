#include<stdio.h>
#define LL long long
LL a[200005];
int main()
{
    LL n,b,t,d;
    LL head=0, tail=0, time=0;
    scanf("%I64d%I64d",&n,&b);
    while(n--){
        scanf("%I64d%I64d", &t, &d);
        if(time <= t) {
            time = t + d;
            printf("%I64d ",time);
            a[tail++] = time;
            head = tail-1;
        }
        else {
            while(a[head] <= t) head++;
            if(tail-head-1 < b) {
                time += d;
                printf("%I64d ", time );
                a[tail++] = time;
            }
            else {
                printf("-1 ");
            }
        }
    }
    return 0;
}
