#include<iostream>
#include<set>
#include<stack>
#include<queue>
using namespace std;
const int MAXN = 2 * 1e5 + 5;
long long ans[MAXN] , t[MAXN] , d[MAXN];

queue<int> S;

long long tm = 0;

void goanddo(int i)
{	
	while(!S.empty())
	{
		int id = S.front();
		long long st = max(tm , t[id]);
		if(st + d[id] > t[i])
			break;
		tm = st + d[id];
		ans[id] = tm;
		S.pop();
	}
}
int main()
{
	int n , b;
	cin >> n >> b;

	for(int i = 1;i <= n;i ++)
	{
		cin >> t[i] >> d[i];	
		goanddo(i);
		if((int)S.size() <= b)
			S.push(i);
		else
			ans[i] = -1;
	}

	t[n + 1] = (long long)1e15;
	goanddo(n + 1);

	for(int i = 1;i <= n;i ++)
		cout << ans[i] << " ";
	cout << endl;
	return 0;
}
