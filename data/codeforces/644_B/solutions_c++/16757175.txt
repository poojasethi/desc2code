#include<cstdio>
#define maxn 200010
long long t[maxn],d[maxn],ans[maxn];
int sta[maxn],n,b;
int main()
{
	scanf("%d%d",&n,&b);
	int ta=1,fr=0;
	long long nt=0;
	for(int i=1; i<=n; i++)
	{
		scanf("%I64d%I64d",&t[i],&d[i]);
		while(ta<=fr && nt<=t[i])
		{
			nt+=d[sta[ta]];
			ans[sta[ta]]=nt;
			ta++;
		}
		if(nt<=t[i]) nt=t[i];
		if(fr-ta+1<b) sta[++fr]=i;
		else ans[i]=-1;
	}
	while(ta<=fr)
	{
		nt+=d[sta[ta]];
		ans[sta[ta]]=nt;
		ta++;
	}
	for(int i=1; i<n; i++) printf("%I64d ",ans[i]);
	printf("%I64d\n",ans[n]);
	return 0;
}
