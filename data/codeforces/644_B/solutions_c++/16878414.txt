#include <iostream>
#include <stdio.h>
long long int a[200010],t[200010],d[200010],e[200010];
int main(){
long long n,b,br,poc,kraj,l;
scanf("%I64d",&n);
scanf("%I64d",&b);
scanf("%I64d",&t[0]);
scanf("%I64d",&d[0]);
a[0]=t[0]+d[0];
e[0]=a[0];
br=1;
for(int i=1;i<n;i++){
scanf("%I64d",&t[i]);
scanf("%I64d",&d[i]);
poc=0;
kraj=br-1;
l=-1;
while(poc<=kraj){
if(a[(poc+kraj)/2]<=t[i]){
l=(poc+kraj)/2;
poc=l+1;}
else{
kraj=(poc+kraj)/2-1;}}
if(br-l-1<b+1){
if(t[i]>a[br-1]){
a[br]=t[i]+d[i];}else{
a[br]=a[br-1]+d[i];}
br+=1;
e[i]=a[br-1];}
else{
e[i]=-1;}
}
for(int i=0;i<n;i++){
printf("%I64d ",e[i]);}
return 0;}
