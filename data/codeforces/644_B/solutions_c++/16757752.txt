#include <iostream>
#include <vector>
#include <queue>
using namespace std;

queue<long long> q;

int main() {
	int n, b;
	cin >> n >> b;
	for (int i = 0; i < n; i ++) {
		long long t, d;
		long long ans;
		cin >> t >> d;
		while (!q.empty() && q.front() <= t) {
			q.pop();
		}
		if (q.size() > b) {
			ans = -1;
		} else {
			if (q.empty()) ans = t + d; else ans = q.back() + d;
			q.push(ans);
		}
		cout << ans;
		if (i == n - 1) cout << endl; else cout << " ";
	}
	return 0;
}
