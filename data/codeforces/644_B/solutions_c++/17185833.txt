#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll n,b,t,d,ans=0;
queue<long long> q;
int main() {
	cin>>n>>b;
	while (n--) {
		cin>>t>>d;
		while (q.size() && q.front()<=t)
            q.pop();
		if (q.empty())
            ans=t;
		if (q.size()<=b){
            ans+=d;
            q.push(ans);
            cout<<ans<<" ";}
		else cout<<-1<<" ";
	}
	return 0;
}
