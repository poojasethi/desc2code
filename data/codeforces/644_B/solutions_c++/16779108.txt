#include <iostream>
#include <cmath>

using namespace std;

long long D[200001];

int main(int argc, char** argv) {
	int n,b,t,d,L,R;
	cin>>n>>b;
	cin>>t>>d;
	D[0]=t+d;
	cout<<D[0];
	L=0;R=1;
	while(--n)
	{
	 cin>>t>>d;
	 while(L<R)
	 	{
	 	 if(D[L]<=t)L++;
	 	 else break;
		}
	 if(R-L-1==b)cout<<' '<<-1;
	 else
	 	{
	 	 if(t>D[R-1])D[R]=t; else D[R]=D[R-1];
                 D[R]+=d; 
		 cout<<' '<<D[R]; 
		 R++;
		}
	}
	return 0;
}
