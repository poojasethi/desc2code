#include <iostream>
#include <queue>
using namespace std;

long long n, b,t,d;

queue<long long> a;
int main()
{
	cin >> n >> b;
	while (n--){
		cin >> t >> d;
		while (!a.empty() && a.front() <= t) a.pop();
		if (a.empty()){
			cout << t + d << " ";
			a.push(t + d);
		}
		else if (a.size() > b)
			cout << "-1 ";
		else{
			cout << a.back() + d << " ";
			a.push(a.back() + d);
		}
	}
	return 0;
}
