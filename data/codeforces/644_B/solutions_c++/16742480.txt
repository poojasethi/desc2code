#include<cstdio>
#include<set>
using namespace std;
set<long long>S;
int main(){
	int n,b;scanf("%d%d",&n,&b);
	int t,d;
	long long ans;
	S.clear();b++;
	while(n--){
		scanf("%d%d",&t,&d);
		while(!S.empty()&&*S.begin()<=t)S.erase(S.begin());
		if(S.size()==b){printf("-1 ");continue;}
		if(S.empty())ans=t+d;
		else ans=*S.rbegin()>t?*S.rbegin()+d:t+d;
		printf("%I64d ",ans);
		S.insert(ans);
	}
	return 0;
}
