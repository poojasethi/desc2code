#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<iostream>
#define maxn 200010

using namespace std;

long long t[maxn],d[maxn],ans[maxn];
int q[maxn];
int n,b;

int main()
{
	scanf("%d%d",&n,&b);
	for (int i=1;i<=n;i++) {scanf("%I64d%I64d",&t[i],&d[i]);ans[i]=-1;}
	int l=0,r=0,i=0;
	long long T=0;
	while (i<=n)
	{
		if (l<r) l++,ans[q[l]]=T+d[q[l]],T=ans[q[l]];
		else i++,ans[i]=t[i]+d[i],T=ans[i];
		while (i<n && t[i+1]<T)
		{
			i++;
			if (r-l<b) q[++r]=i;
		}
	}
	for (int i=1;i<n;i++) printf("%I64d ",ans[i]);printf("%I64d\n",ans[n]);
	return 0;
}
