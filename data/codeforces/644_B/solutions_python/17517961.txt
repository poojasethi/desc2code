#!/usr/bin/env python

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n, b = ia()
N = 1234567
q = [0 for e in range(N)]

t, d, res = [], [], []
for i in range(n):
    tt, dd = ia()
    t.append(tt); d.append(dd); res.append(-1)
t.append(1e18)

fr, to = 0, 0
ct = 0 # current time
for i in range(n+1):
    # processing queue
    while (fr < to and ct <= t[i]):
        j = q[fr] # current task
        ct = ct + d[j]
        res[j] = ct
        fr = fr + 1
    ct = max(ct, t[i])
    if to - fr < b:
        q[to] = i; to = to + 1
