n, b = map(int, raw_input().split(' '))
a = []
st = 0
en = 0

for i in range(n):
    t, d = map(int, raw_input().split(' '))
    while(st < en and a[st][0] <= t):
        st += 1

    if (en-st < b):
        mx = t
        if(en > 0):
            mx = max(mx, a[en-1][1])
        a.append((mx, mx + d))
        en += 1
        print mx+d,
    else:
        print -1,

print;
