def main():
	
	import sys
	#import math
	from Queue import PriorityQueue
	#from operator import itemgetter
	#from sets import Set
 
	n, b = map(int,sys.stdin.readline().strip().split())

	ex = [0]*n
	q_len = 0
	p_q = PriorityQueue()
	next_out = 0
	last_out = 0
	for i in range(n):
		t,d = map(int,sys.stdin.readline().strip().split())
		
		while (q_len > 0) and (t >= next_out):
			next_out = p_q.get()
			q_len -= 1
		
		if (q_len >=b):
			ex[i] = str(-1)
		else:
			q_len += 1
			last_out = max(last_out,t) + d
			p_q.put(last_out)
			ex[i] = str(last_out)

	
	print " ".join(ex)


		
if __name__ == "__main__":

	main()
