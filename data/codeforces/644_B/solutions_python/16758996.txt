R=lambda:map(int,raw_input().split())
n,s=R()
q=[]
t,d=R()
T,I=t+d,0
ans=[0]*n
j=0
for i in range(1,n):
    t,d=R()
    if len(q)<s:
        q.append((i,t,d))
    elif T<=t:
        ans[I]=T
        I,ht,hd=q.pop(0)
        T=max(ht,T)+hd
        q.append((i,t,d))
    else:
        ans[i]=-1
ans[I]=T
for I,t,d in q:
    T=max(t,T)+d
    ans[I]=T
print ' '.join(map(str,ans))
