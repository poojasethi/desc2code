from collections import deque
n, b = map(int, raw_input().split())
ans = [-1] * n
q = deque()
T = 0
b += 1
for i in xrange(n):
    t, d = map(int, raw_input().split())
    while q and t >= T + q[0][0]:
        ans[q[0][1]] = T + q[0][0]
        T += q[0][0]
        q.popleft()
    if len(q) < b:
        if q:
            q[0][0] -= t - T
        T = t
        q.append([d, i])
while q:
    ans[q[0][1]] = T + q[0][0]
    T += q[0][0]
    q.popleft()
