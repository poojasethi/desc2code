import collections

n,b = map(int, raw_input().split())

queue = collections.deque()
inqueue = 0
tprev = 0

for i in range(n):	
	t,d = map(int, raw_input().split())

	while queue and tprev + queue[0] <= t:
		tprev += queue[0]
		if queue[0] > 0:
			print tprev,
			inqueue -= 1
		else:
			print -1,
		queue.popleft()

	if queue and tprev < t:
		queue[0] -= t - tprev
	if inqueue <= b:
		queue.append(d)
		inqueue += 1
	else:
		queue.append(0)
	tprev = t

while queue:
	tprev += queue[0]
	if queue[0] > 0:
		print tprev,
		inqueue -= 1
	else:
		print -1,
	queue.popleft()
