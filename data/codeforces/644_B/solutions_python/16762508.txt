#!/usr/bin/env python
# encoding: utf-8

from collections import deque

n, b = map(int, raw_input().split())
queries = [map(int, raw_input().split()) for _ in xrange(n)]

que, cur = deque(), 0
ans = [-1 for i in xrange(n)]
i, cur = 0, 0
while i < n:
    if len(que) == 0:
        while i < n and len(que) < b:
            que.append(i)
            i += 1
    while len(que) > 0:
        j = que.popleft();
        a = queries[j]
        ans[j] = max(cur, a[0]) + a[1]
        cur = ans[j]
        while i < n and queries[i][0] < cur:
            if len(que) == b:
                pass
            else:
                que.append(i)
            i += 1
print ' '.join(map(str, ans))
