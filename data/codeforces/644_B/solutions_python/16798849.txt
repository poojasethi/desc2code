n,m=R()
x=0
h,e=0,0

v=[-1]*n
for i in range(n):
  t,d=R()
  if x<=t and h>=e:
    x=max(x,t)+d
    v[i]=x
  else:
    while x<=t and h<e:
      a,b,c=q[h]
      h+=1
      x=max(x,b)+c
      v[a]=x
    if e-h<m:
      q[e]=(i,t,d)
      e+=1
