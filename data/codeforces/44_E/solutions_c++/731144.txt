#include <iostream>
#include <cstring>

using namespace std;

char str[205]={0};
long r[205]={0};

int main()
{
 long k,a,b,l,i,j;
 
 cin>>k>>a>>b;
 cin>>str+1;
 
 l=strlen(str+1);
 for(i=1;l;i=i%k+1,l--)
   r[i]++;
 for(i=1;i<=k;i++)
   if(r[i]<a||r[i]>b)
    {
     cout<<"No solution"<<endl;
     return 0;
    }
 for(i=1,j=1;i<=k;i++)
   {
    for(;r[i];r[i]--,j++)
      cout<<str[j];
    cout<<endl;
   }

 return 0;
}
