#include<bits/stdc++.h>
using namespace std;
int k,a,b,len;
char s[202];
int main(){
  scanf("%d %d %d\n%s",&k,&a,&b,s);
  len=strlen(s);
  if(len>k*b || k*a>len){
    printf("No solution\n");
    return 0;
  }
  int lol=len%k;
  for(int i=1;i<=len;i++){
    printf("%c",s[i-1]);
    if(lol>0){
      if(i%(len/k+1)==0){
        lol--;
        printf("\n");
      }
    }else{
      if((len-i)%(len/k)==0){
        printf("\n");
      }
    }
  }
  return 0;
}
