#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

char s[100010];

int main()
{
    int k,a,b;
    char ch;
    scanf("%d%d%d%c",&k,&a,&b,&ch);
    scanf("%s",s);
    int l=strlen(s);
    if (k*a>l || k*b<l) printf("No solution\n");
    else
    {
        int pos=0;
        for (int i=0;i<k;i++)
        {
            for (int j=0;j<l/k+((i<l%k)?1:0);j++,pos++)
                printf("%c",s[pos]);
            printf("\n");
        }
    }
    return 0;
}
