#include <string>
#include <iostream>

using namespace std;

int main() {
    int a, b, c, n, m;
    string s;

    cin >> c >> a >> b >> s;
    n = s.length();
    m = 0;
    if (a * c <= n && n <= b * c) {
        for (int i = 0; i < c; ++i) {
            cout << s.substr(m, (n - m) / (c - i)) << endl;
            m += (n - m) / (c - i);
        }
    } else {
        cout << "No solution" << endl;
    }

    return 0;
}
