#include <iostream>
#include <string>

using namespace std;

int k, a, b;
string s;

int f[210];

int main(){
	cin >> k >> a >> b;
	cin >> s;

	if ((k*a > (int)s.size())||(k*b < (int)s.size())){
		cout << "No solution";
		return 0;
	}


	for (int i = 0; i < (int)s.size(); i++){
		f[i%k]++;
	}

	int poi = -1;
	for (int i = 0; i < k; i++){
		for (int j = 0; j < f[i]; j++){
			poi++;
			cout << s[poi];
		}
		cout << endl;
	}

	return 0;
}
