#include<bits/stdc++.h>

using namespace std;

int M[202][202],m,n,i,j,k,a,b;

string s;

void f(int x){

	if(!x) return;
	f(x - M[m--][x]);

	m++;

	for(i=x - M[m][x];i < x;i++)
        cout << s[i];
	cout << endl;
}
int main(){

	cin >> m >> a >> b >> s;

    n = s.size();

    M[0][0] = 1;

	for(i=1;i<=m;i++)
		for(j=1;j<=n;j++)
			for(k=a;k<=b;k++)
				if(j >= k && M[i-1][j-k]){
					M[i][j] = k;
					break;
				}

	if(M[m][n]) f(n);

	else cout << "No solution";

	return 0;
}
