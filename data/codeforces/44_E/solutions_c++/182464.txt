#include <cstdio>
using namespace std;

int main()
{
//	freopen("e.in","r",stdin);
	int k,a,b;
	char buf[300];
	scanf("%d%d%d\n",&k,&a,&b);
	gets(buf);
	int len = 0;
	while(buf[len]) len++;
	double tmp = (double) len / k;
	if(tmp < a || tmp > b) 
	{
		puts("No solution");
		return 0;
	}
	int l = len / k;
	int mod = len % k;
	int cur = 0;
	for(int i=0;i<k;i++)
	{
		for(int p = 0;p < l;p++)
			putchar(buf[cur++]);
		if(i < mod) putchar(buf[cur++]);
		puts("");
	}
		
	
	return 0;
}
