#include <iostream>
#include <string>
using namespace std;

int main()
{
	int k,a,b;
	string s;
	cin>>k>>a>>b;
	cin>>s;
	int n = s.length();
	int p = n/k;
	if (p>b || p<a)
		cout<<"No solution";
	else
	{
		if (n%k!=0 && p==b)
		{
			cout<<"No solution";
			return 0;
		}
		for (int i = 0; i<n%k;i++)
		{
			cout<<s.substr(0,p+1)<<endl;
			s.erase(0,p+1);
		}
		while (s.length()!=0)
		{
			cout<<s.substr(0,p)<<endl;
			s.erase(0,p);
		}
	}
	return 0;
}
