#include<iostream>
#include<string>
using namespace std;
int main(){
    int k,l,r,z,c,t=0;
    string s;
    cin>>k>>l>>r>>s;
    z=s.size();
    if(k*l>z||k*r<z)
        cout<<"No solution";
    else
    while(k--){
        for(c=l;c<=r;c++)
            if(k*l+c<=z&&k*r+c>=z)break;
        z-=c;
        while(c--)cout<<s[t++];
        cout<<endl;
    }
}
