#include<bits/stdc++.h>
using namespace std;

int k , a , b ;
string str ;

int main(){
  cin >> k >> a >> b >> str ;
  
  int n = str.length() , len[2] , tot[2] ; 
  len[0] = a + (n-k*a)/k  ;
  len[1] = len[0] + 1 , tot[1] = (n-k*a)%k  ;
  tot[0] = k - tot[1] ;
  
  if(n > k*b || n < k*a)  { cout << "No solution\n" ; return 0 ; }
  
  int cnt=0 ;
  for(int l=0 ; l<2 ; l++) for(int i=0 ; i<tot[l] ; i++) {
    for(int j=0 ; j<len[l] ; j++) cout << str[cnt++] ; 
    puts("") ;
  }  
  
  return 0;
}
