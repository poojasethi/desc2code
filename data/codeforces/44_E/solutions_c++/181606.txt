#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main()
{
	int k, a, b;
	string str;
	cin >> k >> a >> b >> str;

	int n = str.size();
	vector<int> line(k);
	for(int i=0; n--; i=(i+1)%k)
		line[i] ++;

	for(int i=0; i<k; ++i)
		if( line[i]<a || b<line[i] )
			{cout << "No solution" << endl; return 0;}

	for(int i=0,off=0; i<k; off+=line[i++])
		cout << str.substr(off,line[i]) << endl;
}
