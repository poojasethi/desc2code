#include<cstdio>
#include<cstring>
int l, k, a, b, i, j, t, x, y;
char s[210];
int main(){
	while(~scanf("%d %d %d", &k, &a, &b)){
		scanf("%s", s);
		l = strlen(s);
		if(l>=(k*a) && l<=(k*b)){
			t = l/k;
			x = l-t*k;
			y=0;
			for(i=0; i<x; i++){
				for(j=0; j<=t; j++)	putchar(s[y++]);
				puts("");
			}
			for(i=x; i<k; i++){
				for(j=0; j<t; j++)	putchar(s[y++]);
				puts("");
			}
		}
		else	puts("No solution");
	}
	return 0;
}
