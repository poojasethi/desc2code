#include <iostream>
#include <string>
using namespace std;

int main(){
	int k,a,b;
	string str;
	cin>>k>>a>>b>>str;
	if(str.length()<k*a || str.length()>k*b){
		cout<<"No solution"<<endl;
		return 0;
	}
	int each=str.length()/k;
	int left=str.length()%k;

	int p=0;
	for(int i=0;i<k;++i){
		int sum=each;
		if(i<left)++sum;
		for(int j=0;j<sum;++j){
			cout<<str[p++];
		}
		cout<<endl;
	}
	return 0;
}
