#include<stdio.h>
#include<string.h>
int main(){
	
	int k,a,b;
	char m[300];
	while(scanf("%d%d%d",&k,&a,&b)!=EOF){
		scanf("%s",m);
		int len=strlen(m);
		if(len*1.0/a<k||len*1.0/b>k)printf("No solution\n");
		else{
			int p=len*1.0/k;
			int i,j=0;
			for(i=1;i<k;i++)
			{
				for(;;j++)
				{
					printf("%c",m[j]);
					if((j+1)%p==0){
						printf("\n");break;
					}
				}
				j++;
			}
			for(;j<len;j++)
			{
				printf("%c",m[j]);
			}
			printf("\n");
		}
	}
}
