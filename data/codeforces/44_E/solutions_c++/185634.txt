#include <iostream>
#include <string>
using namespace std;

int K,a,b;
int main(){
	string s;
	cin>>K>>a>>b;
	cin>>s;
	int x,y,L = s.length();
	x = L / K;
	y = L % K;
	if(x<a)cout<<"No solution"<<endl;
	else{
		if(x + (y>0)>b)cout<<"No solution"<<endl;
		else{
			for(int i=0,L = 0;i<K;i++){
				cout<<s.substr(L,x + (i<y))<<endl;
				L += x + (i<y);
			}
		}
	}
	return 0;
}
