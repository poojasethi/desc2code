//In the name of God
#include <iostream>
#include <string>
using namespace std;

int a, b, k;
string s;

int main() {
	ios::sync_with_stdio(false);
	cin >> k >> a >> b >> s;
	int q = s.size() / k, r = s.size() % k;
	if (q < a || b < q || (r && q == b)) {
		cout << "No solution\n";
		return 0;
	}
	int p = 0;
	for (int i = 0; i < r; i++) {
		for (int j = 0; j < q + 1; j++)
			cout << s[p++];
		cout << '\n';
	}
	for (int i = 1; p < s.size(); i++) {
		cout << s[p++];
		if (!(i % q))
			cout << '\n';
	}
	return 0;
}
