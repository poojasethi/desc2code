#include<iostream>
#include<algorithm>
#include<vector>
#include<string>

using namespace std;

int main()
{
    int k,a,b;
    string str;
    
    cin>>k>>a>>b>>str;
    int n = str.size();
    int l = n/k;
    if (l < a || (l == b && n%k != 0) || l > b )
    {
        cout<<"No solution"<<endl;
        return 0;
    }   
    
    int c = 0;
    int m = n%k;
    for (int i=0;i<k;i++)
    {
        cout<<str.substr(c,l + (m && i < m))<<endl;
        c += l + (m && i < m);
    }
    return 0;   
}
