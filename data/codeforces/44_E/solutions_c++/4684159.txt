#include<bits/stdc++.h>

#define FOR(i,a,b) for (int i=(a);i<(b);++i)
#define REP(i,n) FOR(i,0,n)

using namespace std;

int main()
{
    int k,a,b;
    string seq;
    cin>>k>>a>>b;
    cin>>seq;
    int len=seq.size();
    if(len>=k*a && len<=k*b){
        int m=len/k,n=len%k;
        REP(i,n)cout<<seq.substr(i*(m+1),m+1)<<endl;
        REP(i,k-n)cout<<seq.substr(n*(m+1)+i*m,m)<<endl;
    }
    else cout<<"No solution"<<endl;
    return 0;
}
