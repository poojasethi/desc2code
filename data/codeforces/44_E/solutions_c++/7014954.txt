#include<stdio.h>
#include<string.h>

#define maxn 300

int k,a,b,l;
int key;
char s[maxn];
int cnt=0;

int main()
{
	int j;
	scanf("%d %d %d",&k,&a,&b);
	getchar();
	scanf("%s",s);
	l=strlen(s);
	if(k*a>l||k*b<l)
	{
		printf("No solution");
		return 0;
	}
	else
	{
		key=0;
		for(int i=a;i<=b;i++)
		{
			
			if((k-1)*a<=l-i&&(k-1)*b>=l-i)
			{
				k--;
				l=l-i;
				for(j=key;j<key+i;j++)
					printf("%c",s[j]);
				printf("\n");
				key=j;
				i=a-1;
			}
		}
	}	
	return 0;
}
