#include <cstdio>
#include <iostream>

using namespace std;

const int MAXN=202;

int N,M,a,b,c;
int dn[MAXN];

string s;

int main(){
	
	cin >> c >> a >> b;
	cin >> s;
	N=(int)s.size();
	
	if( a*c<=N && N<=b*c ){
		for( int i=0 ; i<c ; i++ ){
			cout << s.substr(M,(N-M)/(c-i)) << endl;
			M+=(N-M)/(c - i);
		}
	}
	else
		cout << "No solution" << endl;
	
	return 0;
}
