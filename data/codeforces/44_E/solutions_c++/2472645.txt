#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
char s[500];
int main()
{
	int k,a,b;
	scanf("%d%d%d",&k,&a,&b);
	scanf("%s",s);
	int len=strlen(s);
	if(k*a > len || k*b < len)  {
		return puts("No solution"),0;
	}
	int tmp=len/k;
	int tt=len%k;
	int pt=0;
	int i,j;
	for(i=0;i<k;i++)
	{
		for(j=pt;j<pt+tmp;j++)
		{
			printf("%c",s[j]);
		}
		pt = j;
		if(tt) {
			printf("%c",s[pt++]);
			tt--;
		}
		puts("");
	}
	return 0;
}
