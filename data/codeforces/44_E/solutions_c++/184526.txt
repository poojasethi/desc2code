#include <iostream>
#include <string>

using namespace std;

int main()
{
	int k,a,b;
	cin>>k>>a>>b;
	string str;
	cin>>str;
	int size=str.length();
	if(k*b<size || k*a>size)
	{
		cout<<"No solution"<<endl;
		return 0;
	}
	int x = int(b-(k*b-size)/k);
	int strt=0;
	for(int i=0;i<k;i++)
	{
		int ar = x;
		if(i<k*x-size)
			ar--;
		cout<<str.substr(strt,ar)<<endl;
		strt+=ar;
	}
	return 0;
}
