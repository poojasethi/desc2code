#include <bits/stdc++.h>
using namespace std;
int main ()
{
int long long k,n,m,i,j,d;
string c;
cin>>k>>n>>m;
cin>>c;
d=c.size();
if (k*m<d || k*n>d){cout<<"No solution\n";return 0;}
d/=k;
for (i=0;i<k-1;i++)
{
    for (j=0;j<d;j++)
    {
    cout<<c[i*d+j];
    }
    cout<<endl;
}
for (i=(k-1)*d;i<c.size();i++)
{
cout<<c[i];
}
}
