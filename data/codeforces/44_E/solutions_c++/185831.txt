#include <cstdio>
#include <cstring>

using namespace std;

int k, A, B, n, m, r, u, j;
char s[500];

int main ()
{
    scanf ("%d%d%d%s", &k, &A, &B, s);
    n = strlen (s);

    m = n / k;
    r = n % k;
    if (k * A > n || k * B < n)
    {
        puts ("No solution");
        return 0;
    }

    for (int i = 0; i < k; i++)
    {
        u = m;
        if (r) u++, r--;
        while (u--) printf ("%c", s[j++]);
        printf ("\n");
    }

    return 0;
}
