#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main(){
    //freopen("10.inp","r",stdin);
    //freopen("10.out","w",stdout);

    int k,a,b;
    cin>>k>>a>>b;
    string s;
    cin>>s;

    int n=s.size();

    if(a*k>n || b*k<n) {
        cout<<"No solution";
        return 0;
    }

    int cur=0;
    while(cur<n && k){
        int nx=cur+(n-cur-1)/k;
        for(int i=cur;i<=nx;i++) cout<<s[i];
        cout<<endl;
        k--;
        cur=nx+1;
    }


}
