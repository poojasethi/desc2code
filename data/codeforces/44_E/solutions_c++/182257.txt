#include <iostream>
#define maxN (200+10)

using namespace std;

int k, a, b;
int l [maxN];
string s;

int main()
{
	cin>>k>>a>>b>>s;
	int n=s.size();

	if (n<a*k || n>b*k)
	{
		cout<<"No solution"<<endl;
		return 0;
	}

	n-=a*k;
	for (int i=0; i<k; i++)
		l[i]=a;

	for (int i=0; i<k; i++)
		l[i]+=n/k;

	n%=k;
	
	for (int i=0; i<k; i++)
		l[i]+=(i<n);

	for (int i=0, z=0; i<k; i++,cout<<endl)
		for (int j=0; j<l[i]; j++, z++)
			cout<<s[z];

	return 0;
}
