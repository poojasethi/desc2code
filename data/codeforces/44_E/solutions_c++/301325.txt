#include <vector>
#include <string>
#include <iostream>

using namespace std;

int main(){
  int k,a,b;cin >> k >> a >> b;
  string s;cin >> s;
  int n =(int )  s.size();
  if(n < k * a  || k * b < n ){
    cout << "No solution" << endl;
    return 0;
  }
  int len = ((n+1)/k);
  for(int i = 0;i<k-1;i++){
    cout << s.substr(i*len,len) << endl;
  }
  cout << s.substr((k-1)*len, n -(k-1)*len+1);
  return 0;
}
