#include <cstdio>
#include <cstring>
int k,a,b,t=0;
char s[1000]; 
int main(){
     scanf("%d%d%d%s",&k,&a,&b,s);
     int len=strlen(s);
     if(a*k>len||b*k<len)puts("No solution");
     else{
          int l=len/k;
          for(int i=0;i<k-1;i++){
               for(int j=0;j<l;j++)printf("%c",s[j+t]);
               puts("");t+=l;
          }for(int j=0;j<len-t;j++)printf("%c",s[j+t]);puts("");
     }return 0;
}
