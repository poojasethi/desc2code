#include <bits/stdc++.h>
using namespace std;

string st;

int main()
{
    int k, a, b, n;

    // freopen("D:\\1.in", "r", stdin);
    // freopen("D:\\1.out", "w", stdout);

    cin >> k >> a >> b >> st;
    n = st.length();

    if(k * a <= n && n <= k * b)
    {
        for(int i = 0, j = 0; i < k; i ++)
            cout << st.substr(j, n / k + (i < n % k)) << endl, j += n / k + (i < n % k);
    }
    else
        printf("No solution\n");

	return 0;
}
