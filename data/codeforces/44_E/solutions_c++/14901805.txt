#include<iostream>
#include<string>

using namespace std;
string s;
int n,k,a,b;

int main()
{
  cin>>k>>a>>b>>s;
  n=s.size();
  int n1=n/k;
  int n2=n/k+1;
  int t2=n%k;
  int t1=k-t2;
  if(k*a>n || k*b<n){
    cout<<"No solution";
    return 0;}
  int now=0;
  for(int i=1;i<=t1;i++){
    for(int j=0;j<n1;j++)
      cout<<s[now++];
    cout<<endl;}
    for(int i=1;i<=t2;i++){
    for(int j=0;j<n2;j++)
      cout<<s[now++];
    cout<<endl;}
  
  return 0;
}
