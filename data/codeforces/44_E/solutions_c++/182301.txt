#include<stdio.h>
#include<string.h>
char s[201];

int main()
{
scanf("%d%d%d",&n,&a,&b);
gets(s);gets(s);
l=strlen(s);
if ((a*n<=l)&&(l<=b*n))
{
m=l/n;k=l%n;c=0;
for (int i=1;i<=n;i++)
{
int u=m;if (k>0) {k--;u++;}

{

}

}
