#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char s[100005];
int t[205];

int main(void) {
	int n, a, b;
	scanf("%d%d%d", &n, &a, &b);
	scanf("%s", s);
	int l=strlen(s);
	for(int j=0,i=0;i<l;i++,j=(j+1)%n) {
		t[j]++;
	}
	for(int i=0;i<n;i++)
		if(t[i]<a || t[i]>b) {
			puts("No solution");
			return 0;
		}
	for(int j=0,i=0;i<n;i++) {
		for(int k=j;k<j+t[i];k++)
			printf("%c", s[k]);
		j += t[i];
		puts("");
	}
	return 0;
}
