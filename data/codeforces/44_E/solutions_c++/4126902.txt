#include<iostream>
#include<cstring>
using namespace std;
char seg[210];
int main()
{
	int k,a,b;
	while(cin>>k>>a>>b)
	{
		cin>>seg;
		int len=strlen(seg);
		if(len<k*a||len>k*b)
			cout<<"No solution"<<endl;
		else
		{
			int start=0;
			for(int i=k;i>=1;i--)
			{
				for(int j=start;j<start+len/i;j++)
					cout<<seg[j];
				start+=len/i;
				len=len-len/i;
				cout<<endl;
			}
		}
	}
	
	return 0;
}
