#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
using namespace std;
int main()
{
	int k,a,b;
	string s;
	scanf("%d%d%d",&k,&a,&b);
	cin>>s;
	int len=s.length();
	vector<int>res;
	while(len)
	{
		if(a <= len/k && b>=len/k)
		{
			res.push_back(len/k);
			len-=len/k;
			k--;
		}
		else
		{
			printf("No solution");
			break;
		}	
	}
	if(!k)
	{
		int j=0,prev=0;
		for(int i=0;i<res.size();i++)
		{
			for(;j<prev+res[i];j++)
				cout<<s[j];
			printf("\n");
			prev=prev+res[i];
		}
	}
}
