#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
int k,a,b;
string st;
int main(){
  scanf("%d%d%d",&k,&a,&b);
  cin>>st;
  if (k*a>st.length() || k*b<st.length()) printf("No solution\n");
  else{
  	int l=0,j;
    for (int i=1;i<=k;i++){
      for (j=a;j<=b;j++)
	    if (st.length()-l-j>=(k-i)*a && st.length()-l-j<=(k-i)*b) break;
	  for (int k=l;k<l+j;k++) cout<<st[k];
	  l+=j;
	  printf("\n");    
    }  
  }
  return 0;
}
