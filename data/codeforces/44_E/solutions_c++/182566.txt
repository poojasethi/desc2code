#include <iostream>
#include <string>
using namespace std;
int main(){
int n,a,b,len,x,y,i,j,z;
string s,t;
cin>>n>>a>>b;
cin>>s;
len=s.length();
if((n*b<len)||(n*a>len))
cout<<"No solution";
else{
x=len/n;
y=len%n;
//cout<<x<<" "<<y<<endl;
z=0;
int e;
for(i=1;i<=n-y;i++){
	e=1;t="";
	while(e<=x){
	t+=s[z];
	z++;
	e++;
	}
	cout<<t<<endl;
}
for(i=1;i<=y;i++){
	e=1;t="";
	while(e<=x+1){
	t+=s[z];
	z++;
	e++;
	}
	cout<<t<<endl;
}
}
return 0;
}
