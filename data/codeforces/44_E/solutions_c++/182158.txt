#include<iostream>

using namespace std;

int n,a,b;

int main(){
	cin >> n >> a >> b;
	string s;
	cin >> s;
	int tmp=s.length()/n;
	if(tmp < a || (tmp*n < s.length() && tmp+1 > b) || tmp>b){
		cout << "No solution" << endl;
		return 0;
	}
	int r=s.length()-tmp*n;
	int st=0;
	for(int i=0;i<n;i++){
		if(r > 0){
			cout << s.substr(st,tmp+1) << endl;
			st+=tmp+1;
			r--;
		}else{
			cout << s.substr(st,tmp) << endl;
			st+=tmp;
		}
	}
	return 0;
}
