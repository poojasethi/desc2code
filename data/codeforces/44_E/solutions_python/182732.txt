#!/usr/bin/env python

import sys

(k, a, b) = map(int, sys.stdin.readline().split(' '))
sir = sys.stdin.readline().rstrip('\n')
l = len(sir)

x = [0] * k

for i in xrange(l):
    x[i % k] += 1

for i in xrange(k):
    if x[i] < a or x[i] > b:
        sys.stdout.write("No solution")
        exit(0)

y = 0
for i in xrange(k):
    now = ""
    for j in xrange(x[i]):
        now += sir[y]
        y += 1
    sys.stdout.write(now)
    if i != k - 1:
        print ""
