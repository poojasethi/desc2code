import sys

k_lines, min_length, max_length = [int(x) for x in sys.stdin.readline().split()]
line = sys.stdin.readline().strip()
mem = set()

def split_str(string, n_lines, sol=[]):
    if (string, n_lines) in mem:
        return
    mem.add((string, n_lines))
    if string == "" and n_lines == 0:
        for l in sol:
            print(l)
        sys.exit(0)
        return
    if n_lines == 0:
        return
    if len(string) < min_length:
        return

    for i in range(min_length, max_length+1):
        cut_out, rest = string[0:i], string[i:]
        split_str(rest, n_lines-1, sol+[cut_out])

split_str(line, k_lines)
print("No solution")
