#!/usr/bin/python
import sys

(k, a, b) = map(lambda(x):int(x), sys.stdin.readline().split())
s = sys.stdin.readline().strip()
l = len(s)
if l < k * a or l > k * b:
	print "No solution"
	sys.exit(0)

if a == b:
	for i in range(k):
		print s[i*a:(i+1)*a]
	sys.exit(0)

na = (k*b-l) // (b-a)
nb = k - na
if (k*b-l) % (b-a) != 0:
	nb -= 1
for i in range(na):
	print s[i*a:(i+1)*a]
for i in range(nb):
	print s[na*a+i*b: na*a+(i+1)*b]
if (na*a+nb*b) < l: 
	print s[na*a+nb*b:]
