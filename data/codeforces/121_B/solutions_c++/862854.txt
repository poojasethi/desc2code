#include <iostream>

using namespace std;

int main()
{
	int n, k;
	cin >> n >> k;
	string str = "";
	cin >> str;

	for (int i = 1; i < n; i++ ){
		if (k == 0 ) break;
		if (str[i-1] == '4' && str[i] == '7' ){
			if (i % 2 == 1 ){
				str[i] = '4';
			}else{
				if (i - 2 >= 0 && str[i-2] == '4' ){
					if (k % 2 == 1 ) str[i-1] = '7';
					break;
				}else{
					str[i-1] = '7';
				} // end if						
			} // end if
			k--;
		} // end if
	} // end for

	cout << str << endl;	
		
	return 0;
}
