#include<cstdio>
#include<cstring>
#define maxn 100005
using namespace std;

int		n,k; 
char	s[maxn]; 

int		main()
{
	scanf("%d%d%s", &n, &k, s );
	if ( k == 0 ) { printf("%s\n" , s );  return 0; }
	for (int i = 0; s[i]; i++) {
		if ( s[i] == '4' && s[i+1] == '7' ) {
			k--; 
			if ( i%2 ) {
			//	s[i] = '7';
				if ( s[i-1] == '4' ){
					if ( k%2 == 0 ) s[i] = '7';
					break; 
				}
				s[i] = '7';
			}
			else s[i+1] = '4';
			if ( k == 0 ) break;
		}
	}
	printf("%s\n" , s );
	return 0;
}
