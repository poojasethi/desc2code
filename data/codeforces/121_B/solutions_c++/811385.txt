#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
int main(void)
{
	int n,k;
	string s;
	cin >> n >> k >> s;
	for ( int i = 0; i+1 < n && k>0; i++ )
		if ( s[i] == '4' && s[i+1] == '7' ){
			k--;
			if ( i%2 == 0 )s[i+1] = '4';
			else{
				s[i] = '7';
				if ( i > 0 ){
					i--;
					if ( s[i] == '4' )k%=2;
					i--;
				}
			}
		}
	cout << s << endl;
	return 0;
}
