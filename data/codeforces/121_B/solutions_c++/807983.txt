#include <cstdio>
#include <cstring>

int n, K;
char s[100001];

int main() {
	scanf("%d%d%s", &n, &K, s);
	for (int i = 0; i < n && K > 0; ++i) {
		if (s[i] == '4' && s[i + 1] == '7') {
			if (i % 2 == 0) {
				s[i] = s[i + 1] = '4';
			} else if (i > 0 && s[i - 1] == '4') {
				s[i] = K % 2 == 0 ? '4' : '7';
				break;
			} else {
				s[i] = s[i + 1] = '7';
			}
			--K;
		}
	}
	printf("%s\n", s);
	return 0;
}
