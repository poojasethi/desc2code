#include<cstdio>
int main()
{
	int i,n,k;
	char a[100005];
	scanf("%d%d",&n,&k);
	scanf("%s",a);
	for(i=0;i<n&&k;i++)
	{
		if(a[i]=='4'&&i%2==0)
		{
			if(a[i+1]=='7'&&a[i+2]=='7')
			{
				if(k&1)a[i+1]='4';
				k=0;
			}
			else if(a[i+1]=='4'&&a[i+2]=='7')
			{
				if(k&1)a[i+1]='7';
				k=0;
			}
			else if(a[i+1]=='7')
				a[i+1]='4',k--;
		}
		else if(a[i]=='4'&&a[i+1]=='7')
			a[i]='7',k--;
	}
	printf("%s\n",a);
	return 0;
}
