#include<iostream>
#include<string>
using namespace std;
char a[100010];
int n,k;
int main(){
    cin>>n>>k>>a;
    for(int i=0;i<n&&k>0;i++)
    {
        if(a[i]=='4'&&a[i+1]=='7')
        {
            if(i%2==0) a[i]=a[i+1]='4';
            else if(i>0&&a[i-1]=='4')
            {
                if(k%2) a[i]='7';
                else a[i]='4';
                break;
            }
            else a[i]=a[i+1]='7';
            k--;
        }
    }
    cout<<a;
}
