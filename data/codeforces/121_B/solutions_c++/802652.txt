#include<cstdio>

int n,k;
char A[100010];

int main()
{
	scanf("%d%d",&n,&k);
	scanf("%s",A+1);
	for(int c=1;c<n;c++)
	{
		if( c != n-1 and c%2 == 1 and A[c] == '4' and ( A[c+1] == '7' or A[c+1] == '4' ) and A[c+2] == '7' ) k %= 2;
		if( A[c] == '4' and A[c+1] == '7' and k != 0 )
		{
			k --;
			if( c%2 == 1 ) A[c] = A[c+1] = '4';
			else A[c] = A[c+1] = '7';
			c --;
		}
		if( k == 0 ) break;
	}
	printf("%s\n",A+1);
	return 0;
}
