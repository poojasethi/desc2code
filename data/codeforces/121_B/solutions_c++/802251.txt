#include <iostream>
#include <cstdio>
#include <vector>
#include <string>
using namespace std;

string v;
int main()
{
//	freopen("input.txt", "r", stdin);
//	freopen("output.txt", "w", stdout);

	int n, k;
	cin >> n >> k;
	cin >> v;

	int c = 0;
	for (int i = 0; i + 1 < n && c < k; ++i)
	{
		if (v[i] == '4' && v[i+1] == '7')
		{
			v[i] = v[i+1] = (i & 1 ? '7' : '4');
			c++;
		}
		if (i && v[i-1] == '4' && v[i] == '7')
		{
			v[i] = ((k-c) & 1 ? '4' : '7');
			c = k;
		}
	}
	cout << v;

	return 0;
}
