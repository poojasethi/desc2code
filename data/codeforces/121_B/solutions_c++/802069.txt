#include<iostream>
using namespace std;
int main()
{
int n,m,i;
string s;
cin>>n>>m>>s;
for(i=0;i<n;i++)
{
if(m==0) break;
if(i>=0 && s[i]=='4' && s[i+1]=='7')
{
if(s[i-1]=='4' && i%2==1 && m>1) m%=2;
else if(s[i+2]=='7' && i%2==0 && m>1) m%=2;
else
{
if(i%2==1) s[i]='7';
else s[i+1]='4';
m--;
}
i-=2;
}
}
cout<<s;
return 0;
}
