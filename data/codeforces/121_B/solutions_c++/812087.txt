#include<string>
using namespace std;
int main()
{
	string s;int i,j,k,flag=0,n;
	cin>>n>>k>>s;
	for(i=0;i<n&&k>0;i++)
	{
		if(s[i]=='4'&&s[i+1]=='7')
		{
            if(i%2==0)s[i+1]='4';
			else if(i>0&&s[i-1]=='4')

			}

		}		
	}

}
