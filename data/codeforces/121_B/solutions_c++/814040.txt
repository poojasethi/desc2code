#include <iostream>
#include <string>
using namespace std;
string inp;
int n,t;
int now,used;
int main()
{
	ios_base::sync_with_stdio(false);
	cin>>n>>t>>inp;
	while (now<n-1 && used<t)
	{
		if (inp[now]=='4' && inp[now+1]=='7')
		{
			if (now%2==0)
				inp[now+1]='4';
			else
			{
				inp[now]='7';
				if (now>0 && inp[now-1]=='4')
				{
					if ((t-used)%2==0)
						inp[now]='4';
					cout<<inp<<endl;
					return 0;
				}
			}
			used++;
		}
		else
			now++;
	}
	cout<<inp<<endl;
	return 0;
}
