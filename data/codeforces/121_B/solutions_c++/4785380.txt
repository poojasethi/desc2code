#include <cstdlib>
#include <cstdio>
char s[100100];
int main(){
	//freopen("B.in","r",stdin);
	int n,k;
	scanf("%d%d%s",&n,&k,s);
	int i = 0;
	while (k-- && i < n - 1) {
		for (; i < n - 1; i++)
			if (s[i]=='4' && s[i+1] == '7')
				break;
		if (i < n - 1 && s[i]=='4' && s[i+1] == '7')
			if (i & 1){
				s[i] = '7';
				if (i && s[i-1] == '4'){
					if (k % 2) 
						s[i] = '4';
					break;
				}
			} else s[i+1] = '4';
		else
			break;
	}
	puts(s);
}
