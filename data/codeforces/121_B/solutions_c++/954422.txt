#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
int vis[100000];
int main()
{
int n,k,i=0,c=0;
cin>>n>>k;
string s;
cin>>s;
for(int i=0;i<s.size();i++)
vis[i]=-1;
while(c<k && i+1<s.size())
{
if(s[i]=='4' && s[i+1]=='7')
{
if(vis[i]!=-1 && c%2 == k%2)
break;
if(i%2)
s[i]='7';
else
s[i+1]='4';
vis[i]=c;
i--;
c++;
if(i<0)
i=0;
else
if(vis[i]!=-1 && c%2 == k%2)
break;
}
else
i++;
}
cout<<s<<endl;
}
