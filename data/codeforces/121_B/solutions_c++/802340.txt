#include <stdio.h>

char a[100010];
int n, k;

int main()
{
	int i;
	scanf("%d%d", &n, &k);
	scanf("%s", a);
	if (k>0)
	for (i=0; i<n-1; i++)
	  if (a[i]=='4' && a[i+1]=='7')
	  {
	  	if (i%2==0 && i+2<n && a[i+2]=='7')
	  	{
	  		if (k&1) a[i+1]='4';
	  		break;
	  	}
	  	if (i%2==1 && a[i-1]=='4')
	  	{
	  		if (k&1) a[i]='7';
	  		break;
	  	}
	  	if (i%2==0) a[i+1]='4';
	  	else a[i]='7';
	  	k--;
	  	if (k==0) break;
	  }
	printf("%s\n", a);
	return 0;
}
