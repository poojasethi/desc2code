#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std ;

int main() {
	int N, D ;
	string S ;
	int t=0, pos ;
	
	cin >> N >> D >> S ;
	
	for( pos=0; pos+1<N && (S[pos]!='4' || S[pos+1]!='7');  pos++  )  ;
	for( t=0; t<D && pos+1<N; t++ )  {
		if( pos%2  )  {
			S[pos] = '7' ;
			if( pos-1>=0 && S[pos-1]=='4' ) {
				if( (D-t)%2==0   )
					S[pos] = '4' ;
				break ;
			}
		}
		else 
			S[pos+1] = '4' ;
		for( ; pos+1<N && (S[pos]!='4' || S[pos+1]!='7');  pos++  )  ;
	}
	
	cout << S << endl ;
	
	return 0 ;
}
