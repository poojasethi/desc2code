#include <stdio.h>
#define TT 100000+2
char a[TT];
void solve (int n,int k) {
	for (int i=0;i<n-1&&k;i++) {
		if (a[i]=='4'&&a[i+1]=='7') {
			if (i>0&&a[i-1]=='4'&&i%2)
				k=k%2;
			if (k==0)
				break;
			if (i%2==0)
				a[i+1]='4';
			else
				a[i]='7';
			--k;
		}
	}
	printf("%s\n",a);
}

int main ( ) {
	int n,k;
	while (scanf("%d%d",&n,&k)!=EOF) {
		getchar ( );
		gets (a);
		solve (n,k);
	}
	return 0;
}
