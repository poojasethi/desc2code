#include <algorithm>
#include <stdio.h>
#define  maxn      100005
using    namespace std;

int n,k,c=1;
char s[maxn];

int main()
{
  scanf("%d%d%s",&n,&k,s+1);
  while(k--)
  {
    while(c<=n-1 && !(s[c]=='4' && s[c+1]=='7')) c++;
    if(c==n) break;
    if(c%2==1)
    {
      s[c+1]='4';
      if(c+2<=n && s[c+2]=='7') k%=2;
    }
    else
    {
      s[c]='7';
      if(c-1>=1 && s[c-1]=='4') k%=2;
    }
    c=max(1,c-2);
  }
  printf("%s\n",s+1);
  return 0;
}
