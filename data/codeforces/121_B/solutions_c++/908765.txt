#include <stdio.h> 
#include <iostream> 
using    namespace std; 

int N,K;
char s[100005]; 
 
int main()
{ 
  cin >> N >> K;
  cin >> s;

  for(int i=0 ; i<N-1 && K ; i++)
    if(s[i]=='4' && s[i+1]=='7')
    { 
      K--; 
      if(i%2==0)  s[i+1]='4'; 
      else
      {
        s[i]='7';
        if(!i || s[i-1]!='4')  continue;
        else
        {
          K &= 1; 
          if(K) { K--;  s[i]='4'; }
        }
      }
    }

  cout << s << endl;
  return 0; 
}
