#include <cstdio>

using namespace std;

char s[100010];

int main() {
	int n, k, i, cond = -1;
	scanf("%d%d%s", &n, &k, s);
	
	for (i = 0; i < n - 1 && k; ++i)
		if (s[i] == '4' && s[i + 1] == '7')
			if (i % 2) {
				if (s[i - 1] == '4') {
					cond = i - 1;
					break;
				}
				s[i] = '7';
				--k;
			} else {
				s[i + 1] = '4';
				--k;
			}
	
	if (k && cond != -1) if (k % 2) s[cond + 1] = '7';
	
	printf("%s\n", s);
	
	return 0;
}
