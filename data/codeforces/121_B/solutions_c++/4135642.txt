#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int main(){
	int d[100010];
	int n,k,i;
	while(cin>>n>>k){
		memset(d,0,sizeof(d));
		for(i=1;i<=n;i++)scanf("%1d",&d[i]);
		if(n!=1&&k){
			for(i=1;i<=n&&k;i++){
				if(i%2==1&&d[i]==4&&d[i+1]==4&&d[i+2]==7)k%=2;
				if(d[i]==4&&d[i+1]==7){
					if(i%2==1)d[i+1]=4;
					else {
						d[i]=7;
						i--;
						}
					i--;
					k--;
					}
				}
			}
		for(i=1;i<=n;i++)cout<<d[i];
		cout<<endl;
		}
	return 0;
	}
