#include<cstdio>
using namespace std;

char s[100010];
int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	scanf("%s",s);
	int i=-1;bool f=1;
	while(i<n && k){
		i++;
		if(s[i]=='4' && s[i+1]=='7'){
			if(i&1)s[i]='7';else s[i+1]='4';
			k--;
		}
		if(k==0)break;
		if(i>0 && s[i]=='7' && s[i-1]=='4'){
			f=0;break;
		}
	}
	k%=2;
	if(i<n && k>0 && f==0)
		if(i&1)s[i]='4';else s[i]=7;
	printf("%s\n",s);
	return 0;
}
