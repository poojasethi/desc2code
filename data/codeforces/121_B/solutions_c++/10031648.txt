#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<int, int> pii;

int main() {
	ios_base::sync_with_stdio(false);
	int n, k;
	cin >> n >> k;
	string s; cin >> s;
	for (int i = 0; i < n - 1 && k > 0; i++)
		if (s[i] == '4' && s[i + 1] == '7') {
			if (i % 2) {
				if (i && s[i - 1] == '4') {
					s[i] = (k % 2? '7': '4');
					cout << s << '\n';
					return 0;
				}
				s[i] = '7';
			}
			else {
				s[i + 1] = '4';
			}
			k--;
		}
	cout << s << '\n';
	return 0;
}
