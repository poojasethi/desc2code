using namespace std;

int main(){
	int n,k;
	char s[100001]; scanf("%d%d%s",&n,&k,s);

	for(int i=1;i<n;i++){
		if(k==0) break;
		if(s[i-1]=='4' && s[i]=='7'){
			if(i%2==1){
				s[i]='4';
			}
			else{
				if(i-2>=0 && s[i-2]=='4'){
					if(k%2==1) s[i-1]='7';
					break;
				}
				else{
					s[i-1]='7';
				}
			}
			k--;
		}
	}

	puts(s);

	return 0;
}
