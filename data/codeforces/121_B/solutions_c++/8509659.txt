#include<cstdio>
using namespace std;
int n,k;
char a[100100];
int main(){
	scanf("%d%d", &n, &k);
	gets(a);scanf("%s", a);
	int i,pre=-1;
	for(i=0;i<n-1 && k;i++){
		if (a[i]=='4' && a[i+1]=='7'){
			if (i+(!(i&1))==pre) break;
			if (!(i&1)) a[i+1]='4',pre=i+1;
			else{
				a[i]='7';pre=i;
				while(a[i]=='7') i--;
				i--;
			}
			k--;
		}
	}
	if (i<n-1 && k&1){
		if (!(i&1)) a[i+1]='4';
		else a[i]='7';
	}
	printf("%s\n", a);
	return 0;
}
