#include<iostream>
using namespace std;

const int N =110000;

int n, k;
char a[N];

int main() {
	int i;
	//freopen("ttt", "r", stdin);
	
	cin >> n >> k;
	cin.get();
	cin.getline(a, N);
	
	i = 0;
	
	while(k) {
		while(i < n && (a[i] != '4' || a[i + 1] != '7'))
			++i;
		if(i == n)
			break;
		
		if((i & 1) == 0) {
			a[i + 1] = '4';
			
		}
		else {
			if(i && a[i - 1] == '4') {
				if(k & 1)
					a[i] = '7';
				break;
			}
			a[i] = '7';
		}
		--k;
	}
	
	cout << a;
	
	return 0;
}
