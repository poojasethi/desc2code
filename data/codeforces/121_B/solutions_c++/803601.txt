//#include "stdafx.h"
#include <iostream>
#include <algorithm>
using namespace std;

char X[200000];

int main(void)
{
	int i,n,k;
	cin>>n>>k;
	cin>>X+1;
	X[n+1]=X[n+2]=0;
	for (i=1;i<n && k;i++) {
		if ((i&1) && X[i]=='4' && (X[i+1]=='4' || X[i+1]=='7') && X[i+2]=='7')
			k&=1;
		if (k && X[i]=='4' && X[i+1]=='7') {
			if (i&1)
				X[i+1]='4';
			else
				X[i]='7';
			k--;
		}
	}
	cout<<X+1;
	return 0;
}
