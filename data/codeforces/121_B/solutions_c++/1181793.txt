#include <iostream>
#include <string>
using namespace std;
string s;
int i,n,k,i1=-1;
int main()
{
	cin>>n>>k;
	getline(cin,s);getline(cin,s);
	i=0;
	while (k)
	{
		i--;
		if (i<0) i=0;
		while ((i+1<s.size())&&((s[i]!='4')||(s[i+1]!='7'))) i++;
		if (i+1==s.size())
		{
			cout<<s;
			return 0;
		}
		if (i%2)
			s[i]='7';
		else
		{
			i++;
			s[i]='4';
		}
		k--;
		if (i==i1) k%=2;
		i1=i;
	}
	cout<<s;
	return 0;
}
