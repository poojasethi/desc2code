n, m = map(int, raw_input().split())
s = list(raw_input())
i = 1
count = 0
while i != n and count != m:
	if s[i-1] == '4' and s[i] == '7':
		if i >= 2 and s[i-2] == '4' and i%2 == 0 and m - count != 1:
			count += (m - count) - (m - count)%2
		else:
			if i%2 == 1:
				s[i] = '4'
			else:
				s[i-1] = '7'
			count += 1
		i -= 1
		#print ''.join(s)
	else:
		i += 1

print ''.join(s)
