#!/usr/bin/python
import sys

a = []

def gen (n, d):
	global a
	if n > 0:
		a += [n]
	if d > 0:
		gen (n * 10 + 4, d - 1)
		gen (n * 10 + 7, d - 1)

def go (l, r, pos, c):
	l = max (l, pos)
	r = min (r, c)
#	print l, r, c
	return max (r - l + 1, 0) * c

gen (0, 10)
a.sort ()

while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		s = sys.stdin.readline ().strip ()
		if s == '':
			break
	n, k = [int (x) for x in s.split ()]
	s = sys.stdin.readline ().strip ()
	a = [int (c) for c in s]
	b = [-1] * n
	i = 0
	j = 0
	while j < k:
		while i < n - 1 and (a[i] != 4 or a[i + 1] != 7):
			i += 1
		if i >= n - 1:
			break
		if i & 1 == 0:
			a[i] = 4
			a[i + 1] = 4
		else:
			a[i] = 7
			a[i + 1] = 7
		if b[i] == -1:
			b[i] = j
		else:
			j += ((k - 1 - j) >> 1) << 1
		if i > 0:
			i -= 1
		j += 1
	s = ''.join (str (c) for c in a)
	print s
