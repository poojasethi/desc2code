n,k=map(int,raw_input().split())
d=map(int,raw_input())
c=0
while k>0:
    k-=1
    while c<n-1:
        if d[c]==4 and d[c+1]==7:
            break
        c+=1
    if c==n-1:
        break
    if c%2==0:
        d[c+1]=4
    else:
        d[c]=7
        if c>0:
            if d[c-1]==4:
                k=k%2
                c=c-1
print "".join(map(str,d))
