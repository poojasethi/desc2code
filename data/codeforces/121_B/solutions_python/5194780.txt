n,m=map(int, raw_input().split())
s=['']+list(raw_input())
#print s
for i in range(1,n):
#    print i, s, m
    if s[i] == '4' and s[i+1] == '7' and m > 0:
        if not i%2:        
            if i > 0 and s[i-1] == '4':
                if m % 2:
                    s[i]='7'
                break
            else:
                s[i]='7'
           
        else:
            s[i+1] = '4'
        m-=1       
print ''.join(s)
