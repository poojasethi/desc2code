n, k = map(int, raw_input().split())
t = list(raw_input())
i = 0
while k and i < n - 1:
    if t[i] == '4' and t[i + 1] == '7':
        k -= 1
        if i & 1 == 0: t[i + 1] = '4'
        else:
            if i > 0 and t[i - 1] == '4':
                if k & 1 == 0: t[i] = '7'
                break
            t[i] = '7'
            i -= 2
    i += 1
print ''.join(t)
