line = raw_input().split(' ');
n = int (line[0]);
k = int (line[1]);

arr = range (1,n+1)

if k==n:
    print -1
elif k==n-1:
    print str(arr).replace(',','').replace('[','').replace(']','');
else:
    iters = (n-k-1)/2
    ind = n-1

    while iters>0:
        arr[ind], arr[ind-1] = arr[ind-1], arr[ind]
        ind = ind -2
        iters = iters-1

    if (n-k)%2==0:
        arr[0],arr[ind] = arr[ind], arr[0]
        
    print str(arr).replace(',','').replace('[','').replace(']','');
