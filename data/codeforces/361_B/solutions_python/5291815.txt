r = lambda: map(int, raw_input().split())
n, k = r()
p = n - k
if n == k: print -1
else:
    if p & 1:
        print 1,
        for i in xrange(p / 2):
            print (i + 1) * 2 + 1, (i + 1) * 2,
        for i in xrange(p + 1, n + 1):
            print i,
    else:
        for i in xrange(p / 2):
            print (i + 1) * 2, (i + 1) * 2 - 1,
        for i in xrange(p + 1, n + 1):
            print i,
