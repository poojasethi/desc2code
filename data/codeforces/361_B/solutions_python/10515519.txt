#!/usr/bin/env python
# coding=utf-8
def main():
    n, k = map(int, raw_input().split())
    if k > n - 1:
        print - 1
        return
    else:
        if (n - k) % 2 == 0:
            print n,
            for i in range(2, k + 2):
                print i, 
            for i in range(k + 2, n):
                if (i - k - 2) % 2 == 0:
                    print i + 1,
                else:
                    print i - 1,
            print 1
        else:
            for i in range(1, k + 2):
                print i,
            for i in range(k + 2, n + 1):
                if (i - k - 2) % 2 ==0:
                    print i + 1,
                else:
                    print i - 1,
            print
main()
