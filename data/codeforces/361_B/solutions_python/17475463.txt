def main():
	n, k = map(int, raw_input().split())
	if k == n:
		print -1
		return
	a = [i for i in xrange(1, n + 1)]
	if 1 == (n - 1 - k) % 2: a[0], a[-1] = a[-1], a[0]
	for i in xrange((n - 1 - k) / 2):
		a[2 * i + 1], a[2 * i + 2] = a[2 * i + 2], a[2 * i + 1]
	print " ".join(map(str, a))

if __name__ == "__main__": main()
