def gcd (a, b):
    if a < b:
        return gcd(b, a)
    else:
        if b == 0:
            return a
        else:
            return gcd(b, a % b)

n, k = map(int,raw_input().split())
if k == n:
    print -1
else:
    m = [str(i) for i in range(1, n + 1)]
    x2 = n - 1 - k
    for i in range(x2/2):
        m[2*(i + 1)],m[2*i + 1] = m[2*i + 1],m[2*(i + 1)]
    if x2 & 1 > 0:
        m[0], m[-1] = m[-1], m[0]
    print ' '.join(m)
