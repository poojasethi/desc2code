from fractions import gcd

def solve(n, k):
    if n == k:
        print -1
        return

    if k == n-1:
        print " ".join(map(str,range(1, n+1)))
        return

    s = range(1, n+1)
    for i in range(n-1-k):
        temp = s[0]
        s[0] = s[-i-1]
        s[-i-1] = temp

    print " ".join(map(str, s))
    return



n, k = map(int, raw_input().split())
solve(n, k)
