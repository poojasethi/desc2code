from sys import exit
n,k=map(int,raw_input().split(' '))
if (n==1 and k>0) or (n==k):
    print '-1'
    exit(0)
elif k==n-1:
    print ' '.join(map(str,[i for i in range(1,n+1)]))
else:
    # make permutation
    perm=[n]
    perm.extend([i for i in range(2,2+k)])
    perm.append(1)
    perm.extend([i for i in range(k+2,n)])
    print ' '.join(map(str,perm))
