def print_beauty_perms(n, k):
    if k == n:
        print -1
        return
    non_beauty = n - k
    i = 0
    if non_beauty % 2 == 1:
        print 1,
        i += 1
    for num in xrange(i, non_beauty, 2):
        print num+2, num+1,
        i += 2
    for num in xrange(i, n):
        print num + 1,

nums = map(int, raw_input('').split())
print_beauty_perms(*nums)
