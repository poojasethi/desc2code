n, k = map(int, raw_input().split())

seq = range(1, n+1)

goods = n - 1

if k > goods :
    print -1
else :
    swaps = (goods - k)/2
    i = n - 1
    while swaps > 0 :
        tmp = seq[i]
        seq[i] = seq[i-1]
        seq[i-1] = tmp
        i -= 2
        swaps -= 1
    if (goods - k) % 2 == 1 :
        seq[0] = 2
        seq[1] = 1

    print " ".join(map(str, seq))
