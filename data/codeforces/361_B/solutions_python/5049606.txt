n,k = map(int,raw_input().split())

if n==k:
    print(-1)
    exit(0)
if k==0:
    print(n),
    for i in range(1,n):
       print(i),
    print('')
    exit(0)
if n==2:
    print(1),
    print('2'+'\n') 
else:
    if k==n-1:
       for i in range(1,n+1):
           print(i),
       print('')
       exit(0)
    print(n-k),
    for i in range(1,n+1):
        if i!=n-k:
            print(i),
