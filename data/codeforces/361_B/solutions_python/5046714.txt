n,k = map(int, raw_input().split())
if k > n-1:
	print -1
else:		
	if n == 2 and k == 1:
		print "1 2"
	elif n == 1 and k == 0:
		print "1"
	elif k == n-1:
		t1 = ""
		for i in xrange(n):
			t1 += str(i+1) + " "
		print t1.strip()
	else:			
		tmp = str(n) + " "
		for i in xrange(2,k+2):
			tmp += str(i) + " "
		tmp += "1 "
		for i in xrange(k+2,n):
			tmp += str(i) + " "
		print tmp.strip()
