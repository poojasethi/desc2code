import fileinput
import sys

for line in fileinput.input():
    strippedInput = (line.rstrip().lstrip()).split(' ')
    n = int(strippedInput[0])
    k = int(strippedInput[1])
    if k >= n:
        print '-1'
        break
    result = ''
    i = 1
    while i < n-k:
        result += str(i+1) + ' '
        i += 1
    result += '1'
    i+=1
    while i <= n:
        result += ' ' + str(i)
        i+=1
    print result
