n,k=map(int,raw_input().split())
l=range(1,n+1)
k=n-k
if (k==0):
    print -1
elif (k%2==0):
    for g in xrange(k/2):
        l[2*g],l[2*g+1]=l[2*g+1],l[2*g]
    print ' '.join(map(str,l[0:]))
elif (k%2==1):
    for g in xrange(k/2):
        l[2*g+1],l[2*g+2]=l[2*g+2],l[2*g+1]
    print ' '.join(map(str,l[0:]))
