#!/usr/bin/env python

n, k = map(int, raw_input().split())

perm = range(1, n+1)

p = n - k - 1
	
if p % 2 == 1:
	perm[0], perm[p] = perm[p], perm[0]
	p -= 1

i = 1
while i+1 < n:
	if not p: break
	perm[i], perm[i+1] = perm[i+1], perm[i]
	p -= 2
	i += 2

if p: print -1
else:
	for x in perm: print x,
	print
