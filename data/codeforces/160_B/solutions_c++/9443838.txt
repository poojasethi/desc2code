#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int main(){
    int n, i;
    char tmp[200];
    scanf("%d%s", &n, tmp);
    sort(tmp, tmp+n);
    sort(tmp+n, tmp+(n*2));
    int x=1, y=1;
    for (i=0; i<n; i++){
        if (tmp[i] <= tmp[i+n]) x = 0;
        if (tmp[i] >= tmp[i+n]) y = 0;
    }
    cout << ((x || y) ? "YES" : "NO") << "\n";
}
