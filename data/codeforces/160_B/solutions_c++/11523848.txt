#include <iostream>
#include <algorithm>
using namespace std;
int n;
string s;

int main() 
{
	cin >> n >> s;
	sort(s.begin(), s.begin() + n);
	sort(s.begin() + n, s.end());
	int i = 0;
	while(i < n && s[i] < s[i + n]) i++;
	if(i == n) cout << "YES";
	else
	{
		i = 0;
		while(i < n && s[i] > s[i + n]) i++;
		if(i == n) cout << "YES";
		else cout << "NO";
	}

}
