#include <iostream>
#include <algorithm>

using namespace std;

int main() {
	char p[2 * 100 + 1];
	int n;

	cin >> n;
	cin >> p;
	sort(p, p + n);
	sort(p + n, p + 2 * n);

	int f = 3;
	for (int i = 0; i < n; i++) {
		f &= 2 | (p[i] < p[n + i]);
		f &= 1 | ((p[i] > p[n + i]) << 1);
	}

	cout << (f ? "YES\n" : "NO\n");

	return 0;
}
