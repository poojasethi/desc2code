#include "iostream"
#include "string"
#include "algorithm"
using namespace std;
int main()
{
int i,k,a[100],b[100],d=0,c=0,n;
char l;
cin>>n;
for(i=0;i<n;i++)
{cin>>l;
a[i]=l-'0';
}
for(i=0;i<n;i++)
{
cin>>l;
b[i]=l-'0';}
sort(a,a+n);
sort(b,b+n);
for(i=0;i<n;i++)
{if(a[i]<b[i])
d++;
if(a[i]>b[i])
c++;}
if((d==n)||(c==n))
cout<<"YES";
else
cout<<"NO";
}
