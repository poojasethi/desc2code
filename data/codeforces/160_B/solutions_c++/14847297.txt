#include <bits/stdc++.h>
using namespace std;

int n;
string s;

bool f(int x, int y) {
   for(int i = 0; i < n; i++) {
      if(s[x+i] <= s[y+i]) return false;
   }
   return true;
}

int main() {
   cin >> n >> s;
   sort(s.begin(), s.begin()+n);
   sort(s.begin()+n, s.end());
   bool res = f(0,n) or f(n,0);
   cout << (res?"YES":"NO") << endl;
}
