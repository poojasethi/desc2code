#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int n,k=0,c=0;
	cin>>n;
	char a[2*n];
	for(int i=0;i<2*n;i++){
		cin>>a[i];
	}
	sort(a,a+n);
	sort(a+n,a+2*n);
	for(int i=0;i<n;i++){
		if(a[i]>a[i+n])
		c++;
		else if(a[i]<a[i+n])
		k++;

	}
     if(c==n||k==n)cout<<"YES";
	 else cout<<"NO";	
	
	return 0;
}
