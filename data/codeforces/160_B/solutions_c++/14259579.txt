#include<iostream>
#include<algorithm>
using namespace std;
main()
{
    char s[1000000];
    int i,j,n,c1=0,c2=0;
    cin>>n>>s;
    sort(s,s+n);
    sort(s+n,s+(2*n));
    for(i=0;i<n;i++) {
                if(s[i]>s[n+i]) c1++;
                else if(s[i]<s[n+i]) c2++;
        } if(c1==n || c2==n) cout<<"YES"<<endl;
          else cout<<"NO"<<endl;
     return 0;
}
