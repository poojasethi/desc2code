#include <bits/stdc++.h>
using namespace std;
int main(){
  int n,x[205],y[205],i;
  bool f1=true,f2=true;
  string s;
  cin>>n>>s;
  for(i=0;i<n;i++){
    x[i]=s[i]-'0';
    y[i]=s[i+n]-'0';
  }
  sort(x,x+n);
  sort(y,y+n);
  for(i=0;i<n;i++){
    if(x[i]>=y[i]) f1=false;
    if(x[i]<=y[i]) f2=false;
  }
  if(f1|f2) puts("YES");
  else puts("NO");
  return 0;
}
