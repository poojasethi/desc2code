#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	int n;
	cin >> n;
	string a,b;
	cin >> a;
	b=a.substr(n,n);
	a.resize(n);
	sort(a.begin(),a.end());
	sort(b.begin(),b.end());
	bool more=true, less = true;
	for(int i = 0; i < n; ++i){
		if(a[i]>=b[i])
			less=false;
		if(a[i]<=b[i])
			more=false;
	}
	if(more||less)
		cout << "YES\n";
	else
		cout << "NO\n";
}
