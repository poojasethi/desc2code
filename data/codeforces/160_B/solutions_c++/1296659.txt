#include <stdio.h>
#include <algorithm>

char s[300];

int main()
{
  int n,i;
  bool f1=1,f2=1;
  scanf("%d %s",&n,s);
  std::sort(s,s+n);
  std::sort(s+n,s+2*n);
  for (i=0;(f1||f2)&&i<n;i++)
  {
    if (s[i]<=s[i+n])f1=0;
    if (s[i]>=s[i+n])f2=0;
  }
  if (f1||f2)puts("YES");
  else puts("NO");
}
