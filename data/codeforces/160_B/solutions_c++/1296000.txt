#include<cstdio>
#include<algorithm>
using namespace std;
char s[1<<10];
int main()
{
    int n,i;
    bool f1=1,f2=1;
    scanf("%d%s",&n,s);
    sort(s,s+n);
    sort(s+n,s+n+n);
    for(i=0;i<n;i++)
    {
        if(s[i]<=s[i+n]) f2=0;
        if(s[i]>=s[i+n]) f1=0;
    }
    if(f1||f2) puts("YES");
    else puts("NO");
    return 0;
}
