#include <bits/stdc++.h>
using namespace std;

char d[205];
int n, f, b;

int main() {
  scanf("%d%s", &n, d);
  sort(d, d+n);
  sort(d+n, d+n+n);
  f = b = 1;
  for (int i = 0; i < n; i++) {
    if (d[i] >= d[i+n]) f = 0;
    if (d[i] <= d[i+n]) b = 0;
  }
  printf("%s\n", f||b ? "YES" : "NO");
	return 0;
}
