#include <stdio.h>
#include <algorithm>
#include<stdlib.h>
#include<iostream>
using namespace std;
char s[300];

int main()
{
  int n,i;
  bool f1=1,f2=1;
  cin>>n>>s;
  sort(s,s+n);
  sort(s+n,s+2*n);
  for (i=0;(f1||f2)&&i<n;i++)
  {
    if (s[i]<=s[i+n])f1=0;
    if (s[i]>=s[i+n])f2=0;
  }
  if (f1||f2)cout<<"YES";
  else cout<<"NO";
}
