#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

int main(){
	int n, s1=0,s2=0;
	string s;
	cin>>n;
	cin>>s;
	sort(s.begin(),s.end()-n);
	sort(s.begin()+n,s.end());
	for(int i=0;i<n;i++){
		if(s[i]>s[i+n]) s1++;
		else if(s[i]<s[i+n]) s2++;
	}
	if(s1==n||s2==n){ cout<<"YES"<<endl;}
	else cout <<"NO"<<endl;
	
	return 0;
}
