#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,j;
    char s[210];
    cin>>n>>s;
    sort(s,s+n);
    sort(s+n,s+2*n);
    for(i=0;i<n;i++)
    if (s[i]<=s[i+n])
    break;
    for(j=0;j<n;j++)
    if (s[j]>=s[j+n])
    break;
    if(i==n || j==n)
    cout<<"YES";
    else
    cout<<"NO";
    return 0;
}
