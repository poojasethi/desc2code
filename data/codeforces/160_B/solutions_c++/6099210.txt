#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main()
{
int nu,s1=0,s2=0;
string s;
cin>>nu;
cin>>s;
sort(s.begin(),s.end()-nu);
//cout<<s;
sort(s.begin()+nu,s.end());

for(int i=0;i<nu;i++)
    if(s[i]<s[nu+i])
        s1++;
    else if (s[i]>s[nu+i])
        s2++;
if(s1==nu||s2==nu)
    cout<<"YES";
else
    cout<<"NO";


}
