#include<bits/stdc++.h>
#include<string.h>
using namespace std;
int main()
{
char a[555];
int n;
scanf("%d",&n);
scanf("%s",&a);
n = n*2;
int m = n/2;
sort(a,a+m);
sort(a+m,a+n);
int inc,dec ;
inc =1;
dec = 1;
int i;
for(i=0;i<m;i++)
if(a[i]>a[i+m])dec=0;
else if(a[i]<a[i+m])inc = 0;
else inc=dec=0;
if(inc||dec)printf("YES\n");
else printf("NO\n");

return 0;
}
