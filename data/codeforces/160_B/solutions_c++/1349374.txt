#include<cstdio>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n;
	char s[201]; scanf("%d%s",&n,s);

	sort(s,s+n);
	sort(s+n,s+n+n);

	bool ok1=true;
	rep(i,n) if(s[i]>=s[i+n]) ok1=false;
	bool ok2=true;
	rep(i,n) if(s[i]<=s[i+n]) ok2=false;

	puts(ok1||ok2?"YES":"NO");

	return 0;
}
