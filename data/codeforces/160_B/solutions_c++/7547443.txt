#include<iostream>
#include<algorithm>
using namespace std ;
int n , x=1 , y=1 ;
string s ;
int main()
{
	cin >> n >> s;
	sort(&s[0],&s[n]) ;
	sort(&s[n],&s[n*2]) ;
	for(int i=0 ; i<n ; i++)
	{
		if(s[i]>=s[i+n])
			x=0 ;
		if(s[i]<=s[i+n])
			y=0 ;
	}
	if(x==1 || y==1)
		cout << "YES" << endl ;
	else
		cout << "NO" << endl ; 
	return 0 ;
}
