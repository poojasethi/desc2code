#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,i,s,p,flag=1;
    cin>>n;
    char ara[2*n+1];
    scanf("%s",ara);
    sort(ara,ara+n);
    sort(ara+n,ara+2*n);

    s=ara[n]-ara[0];
    if(s==0) flag=0;
    for(i=1;i<n;i++){
        p=ara[n+i]-ara[i];
        if(s*p<=0) flag=0;}

    if(flag==1) printf("YES");
    else printf("NO");
}
