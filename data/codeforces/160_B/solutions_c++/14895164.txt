#include <bits/stdc++.h>
using namespace std;

char s[201];

int main()
{
    int n, num = 0, i;
    scanf("%d\n%s", &n, s);
    sort(s, s + n);
    sort(s + n, s + 2*n);
    for(i = 0; i < n; i++)
        num += s[i] < s[i + n] ? 1 : (s[i] > s[i + n] ? -1 : 0);
    if(num == n || num == -n) printf("YES");
    else printf("NO");
    return 0;
}
