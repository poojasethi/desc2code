#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int n,h1=0,h2=0,i;
char str[260];
cin>>n>>str;
sort(str,str+n);
sort(str+n,str+2*n);
for(i=0;i<n;i++)
{
    if(str[i]>str[i+n])
       h1++;
    else if(str[i]<str[i+n]) 
       h2++; 
} 
if(h1==n || h2==n)
    cout<<"YES\n";
else
    cout<<"NO\n"; 
return 0;
}
