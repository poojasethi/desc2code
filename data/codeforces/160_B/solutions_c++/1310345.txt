#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

int n;
char s[1111];

int main() {
    scanf("%d%s",&n,s);
    sort(s,s+n);
    sort(s+n,s+n*2);
    int ok = 1;
    for (int i=0; i<n; i++)
        if ((s[i]-s[i+n])*(s[0]-s[n]) <= 0) ok = 0;
    if (ok) printf("YES\n");
    else printf("NO\n");
}
