#include<iostream>
#include<algorithm>
using namespace std ;
int n ;
string s ;
bool al=true , ab=true ;
int main()
{
	cin >> n >> s;
	sort(&s[0],&s[n]) ;
	sort(&s[n],&s[n*2]) ;
	for(int i=0 ; i<n ; i++)
	{
		al &= s[i]<s[i+n] ;
		ab &= s[i]>s[i+n] ;
	}
	cout << (al || ab ? "YES" : "NO") <<endl ;
	return 0 ;
}
