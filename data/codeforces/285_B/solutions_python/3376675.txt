def raw():
    return raw_input().strip()


def ints():
    return map(int, raw().split())


n, s, t = ints()
ps = ints()


current = s
c = 0
while True:
    if current == t:
        print c
        exit(0)
    if current == s and c > 0:
        print -1
        exit(0)
    current = ps[current - 1]
    c += 1
