ar=[1]
(n,s,t)=map(int,raw_input().split())
ar+=map(int,raw_input().split())
mark=[0]*(n+4)
pos=s
flag=True
count=0
while(True):
    if(pos==t):
        break
    if(mark[ar[pos]]==1):
        flag=False
        break
    mark[pos]=1
    pos=ar[pos]
    count+=1
if(flag==False): print -1
else: print count
