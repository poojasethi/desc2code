n,s,t = map(int,raw_input().split())
ls = map(int,raw_input().split())
ls.insert(0, 0)
check = False
i=s
count = 0
if s==t:
    print(0)
    exit(0)
while ls[i]!=t:
    i = ls[i]
    count+=1
    if ls[i]==s:
        check = True
        break
count+=1

if check:
    print(-1)
else:
    print(count)
