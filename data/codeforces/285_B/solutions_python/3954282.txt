#!/usr/bin
from sys import stdin

def IMP( n, s, t, shuffle ):
    d = s
    for w in xrange(-~n):
        if s == t:
            return w
        s = shuffle[s-1]
        if s == d:
            return -1

n, s, t = map( int, stdin.readline().split() )
print IMP( n, s, t, map(int, stdin.readline().split()) )
