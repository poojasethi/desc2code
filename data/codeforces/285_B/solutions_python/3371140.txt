import sys
n, s, t = map(int, raw_input().split())
p = raw_input().split(' ')

psn = s-1
kol = 0
bilo = set()
while True:
    if psn == t-1:
        break
    
    bilo.add(psn)
    
    psn = int(p[psn])-1
    if psn in bilo:
        print -1
        sys.exit(0)
    else:
        kol+=1

print kol
