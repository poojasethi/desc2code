#include<cstdio>
#include<algorithm>
using namespace std;

int p[1<<22];
int q[1<<22];
int cost[25][25];
char st[200010];
int lnk[200010][25];
int n,m,t;
int delta[25];

int main()
{
//  freopen("input.txt","r",stdin);

    
    int XXX=0;

    scanf("%d%d%d",&n,&m,&t);
    scanf("%s",st+1);
    for (int i=1;i<=n;i++)
        XXX|=(1<<(st[i]-'A'));

    for (int i=0;i<m;i++) scanf("%d",&delta[i]);
    for (int i=0;i<m;i++)
    for (int j=0;j<m;j++)
        scanf("%d",&cost[i][j]);
    
    for (int i=2;i<=n;i++)
    for (int j=0;j<m;j++)
    if (st[i-1]-'A'==j) lnk[i][j]=i-1;
    else lnk[i][j]=lnk[i-1][j];

    for (int i=2;i<=n;i++)
    {
        pair<int,int> g[25];
        int cnt=0;
        for (int j=0;j<m;j++)
        if (lnk[i][j]) g[cnt++]=make_pair(-lnk[i][j],j);
        sort(g,g+cnt);

        int tmp=0;
        for (int j=0;j<cnt;j++)
        {
            p[tmp]+=cost[g[j].second][st[i]-'A'];
            p[tmp|(1<<g[j].second)]-=cost[g[j].second][st[i]-'A'];
            if (g[j].second!=st[i]-'A')
            {
                p[tmp|(1<<(st[i]-'A'))]-=cost[g[j].second][st[i]-'A'];
                p[tmp|(1<<(st[i]-'A'))|(1<<g[j].second)]+=cost[g[j].second][st[i]-'A'];
            }
            if (g[j].second==st[i]-'A') break;
            tmp|=(1<<g[j].second);
        }
    }

    //printf("- -\n");

    for (int i=0;i<m;i++)
    {
        for (int j=0;j<(1<<m);j++)
    //  if (p[j])
        {
            if (j&(1<<i)) q[j]+=p[j];
            else {q[j]+=p[j];q[j|(1<<i)]+=p[j];}
        }
        for (int j=0;j<(1<<m);j++) {p[j]=q[j];q[j]=0;}
    }

    //printf("- -\n");

    int res=0;
    for (int i=0;i<(1<<m);i++)
    if ((i|XXX)==XXX)
    {
        if ((i^XXX)==0) continue;
        if (p[i]>t) continue;
        for (int j=0;j<m;j++)
        if (i&(1<<j)) p[i]+=delta[j];
        res+=(p[i]<=t);
    }
    printf("%d\n",res);
}
