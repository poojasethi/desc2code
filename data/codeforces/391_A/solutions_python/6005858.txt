#!/usr/bin/python

def main():
    dnk = raw_input()
    cur_nukl = dnk[0]
    counter = 0
    inner_c = 0
    for nukl in dnk:
        if nukl == cur_nukl:
            inner_c += 1
        else:
            if inner_c % 2 == 0:
                counter += 1
            cur_nukl = nukl
            inner_c = 1
    if inner_c % 2 == 0: # if last seq even
        counter += 1
    print counter
    
if __name__ == '__main__':
  main()
