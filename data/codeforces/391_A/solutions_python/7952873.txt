import sys

def main():
    dna = raw_input()
    cha_count = 0
    seq_start = 0
    for i in range(1, len(dna)):
        if i == len(dna)-1 and dna[i] == dna[i-1]:
            if ((i+1)-seq_start)%2 == 0:
                cha_count += 1
        if dna[i-1] != dna[i]:
            if (i - seq_start)%2 == 0:
                cha_count += 1
            seq_start = i
            
    print(cha_count)
    return 0
main()
