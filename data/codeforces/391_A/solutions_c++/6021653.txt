#include<stdio.h>
#include<string.h>
#include<stdlib.h>
char c[200];
int i,j,k,m,n,ans;
int main()
{
    scanf("%s",c);
    n=strlen(c);
    i=0;
    ans=0;
    while (i<n)
    {
       k=i;
       while (k+1<n&&c[k+1]==c[i])k++;
       if ((k-i)&1)ans++;
       i=k+1;
    }
    printf("%d\n",ans);
   // system("pause");
    return 0;
}
