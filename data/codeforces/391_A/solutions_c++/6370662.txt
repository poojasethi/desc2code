#include <iostream>
#include <string>
using namespace std;
int main(){
	string h,sub;
	int total = 0;
	cin >> h;
	sub = h[0];
	for (int i = 1; i < h.length(); ++i)
	{
		if(h[i]==sub[0]){
			sub+=h[i];
		}
		else{
			if(sub.length()%2==0){
				total++;
			}
			sub = h[i];
		}
	}
	if(sub.length()%2==0){
		total++;
	}
	cout << total;
}
