#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	string s;
	cin>>s;
	int n = s.size();
	int count=1,total = 0;
	
	for(int i = 0;i<n;i++)
	{
		if(s[i]!=s[i+1])
		{
			if(count%2==0)
			{
				++total;
				count = 1;
			}
		}
		
		else
		{
			++count;
		}
			 
	}
	
	cout<<total;
	return 0;
}
