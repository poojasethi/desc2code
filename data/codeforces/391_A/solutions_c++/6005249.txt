#include <iostream>
#include <string>
using namespace std;
 
int main() {
	int maincount=0;
	int count=0;
	string s;
	cin >> s;
	for (int i = 0 ; i <= s.length() ; i++)
	{
		if( s[i] == s[(i+1)] )
			count++;
		else 
		{
			if (count%2 != 0 )
				maincount++;
			count = 0;
		}
	}
	cout << maincount;
	return 0;
}
