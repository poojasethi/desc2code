#include <cstdio>
using namespace std;
#define fo(i,a,b) for (int i = (a); i < (b); i++)

char str[123]; int ans;
int main() {
	scanf("%s", str);
	for (int i = 1, j = 1; str[i]; i++) {
		if (str[i]==str[i-1]) j++;
		else {
			ans += !(j%2);
			j = 1;
		}
		if (!str[i+1]) ans += !(j%2);
	}
	printf("%d\n", ans);

	return 0;
}
