#include<vector>
#include<iostream>
#include<algorithm>
#include<math.h>
#include<cmath>
#include<stdio.h>
using namespace std;
int main()
{
int i,j,n,ans=0,f=0;
string a;
cin>>a;
n=a.size();
for(i=0;i<n;i++)
{
f=0;
for(j=i+1;j<n;j++)
{
if(a[j]==a[i])
{
f++;
}
else
break;
}
if(f%2==1)
ans++;
i=j-1;
}
cout<<ans;
return 0;
}
