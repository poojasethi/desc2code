#include<iostream>
using namespace std;
int main()
{
	
	string s;
	cin>>s;
	int count=1;
	int tot=0;
	int i=0;
	for(i=0;i<s.size();i++)
	{
		if(s[i]==s[i+1])
		{
			count++;
		}
		else
		if(s[i]!=s[i+1])
		{
			if(count%2==0&&count>0)
			{
				tot++;
			}
			count=1;
		}
	}
	cout<<tot;
	return 0;
}
