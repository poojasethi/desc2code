#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int MaxN = 100 + 10;
char s[MaxN];int n;
int main()
{
	scanf("%s", s);
	n = strlen(s);
	int l = 1, sum = 0;
	for(int i = 1; i <= n; ++i)
		if(s[i] != s[i - 1])
		{
			if((l & 1) == 0) ++sum;
			l = 1;
		}
		else ++l;
	cout << sum << endl;
	return 0;
}
