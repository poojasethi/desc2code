#include <iostream>
#include <string>
using namespace std;


int main() {
	string s;
	cin >> s;
	int count = 0;
	int l, r;
	l = r = 0;
	int n = s.length();
	while (r < n) {
		if (s[l] == s[r]) r++;
		else {
			if ((r-l)%2 == 0) count++;
			l = r;
		}
	}
	if ((r-l)%2 == 0) count++;
	cout << count << endl;
}
