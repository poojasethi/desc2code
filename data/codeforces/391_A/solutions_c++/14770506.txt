#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
	char s[150];
	int i,n=0,k=1,j;
	gets(s);
	for(i=0;i<strlen(s);i+=k)
	{
		k=1;j=i+1;
		while(s[i]==s[j]&&s[j]!='\0')
		{
			j++;
			k++;
		}
		if((k)%2==0)
		n++;
	}
	printf("%d\n",n);
	return 0; 
}
