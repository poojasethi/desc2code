#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main ()
{
	string s;
	int sum=0,sum1=0;
	cin>>s;
	for(int i=0;i<s.length();i++)
	{
		sum++;
	    if(s[i]!=s[i+1])
	    {
	    	
	    	if(sum%2==0)
	    	{
	    		sum1++;
	    	}
	    	sum=0;
	    }
	    
	}
	cout<<sum1;
	return 0;
}
