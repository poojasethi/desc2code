#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int n;
char s[105];
int cur=1,ans=0;
int main(){
	cin.getline(s,105);
	n=strlen(s);
	for(int i=1;i<n;i++){
		if(s[i]!=s[i-1]){
			if(cur%2==0) ans++;
			cur=1;
		}else cur++;
	}
	if(cur%2==0) ans++;
	printf("%d\n",ans);
	return 0;
}
