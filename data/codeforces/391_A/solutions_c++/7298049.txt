#include<iostream>
#include<cstring>
using namespace std;
char a[100];
int main()
{
	int b=1,d=0;
	cin>>a;
	for(int i=0;i<strlen(a);i++)
	{
		for(int c=i;a[c]==a[c+1];c++)
		{
			b++;
		}
		if(b%2==0)
		{
			d++;
			i+=b-1;
			b=1;
		}
		else
		{
			i+=b-1;
			b=1;
		}
	}
	cout<<d<<endl;
	return 0;
}
