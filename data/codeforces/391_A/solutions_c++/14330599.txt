#include <bits/stdc++.h>
using namespace std;

const int maxn = 110;

int n;
char ar[maxn];

int main(){
	
	scanf("%s",ar+1);
	n = strlen(ar+1);
	int ans = 0;
	
	for( int i=1 ; i <= n ; i++ ){
		int j = i;
		while( j+1 <= n and ar[j+1] == ar[i] ) j++;
		//~ printf("%d %d\n",i,j);
		ans += (j-i) & 1;
		i = j;
	}
	
	printf("%d\n",ans);
	
	return 0;
}
