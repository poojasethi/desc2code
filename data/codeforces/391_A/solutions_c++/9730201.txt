#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main(){ 
  ios_base::sync_with_stdio(0);
  string s;
  cin>>s;
  int n = s.size();
  int tot = 0;
  for(int i=0; i<n;) {
    int j = i;
    while(j<n && s[j]==s[i]) j++;
    tot+=((j-i)%2==0);
    i = j;
  }
  cout<<tot<<endl;
  return 0;
}
