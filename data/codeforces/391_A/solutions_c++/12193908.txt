#include<stdio.h>
#include<string.h>
int main(){
	char s[102];
	int len,i,ans[102],k=0,sum=0;
	scanf("%s",s);
	len=strlen(s);
	while(i<len-1){
		ans[k]=1;
		while(s[i]==s[i+1]){
			ans[k]++;
			i++;
		}
		k++;
		i++;
	}
	for(i=0;i<k;i++){
		if(ans[i]%2==0) sum++;
	}
	printf("%d",sum);
	return 0;
}
