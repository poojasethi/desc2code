
#include <iostream>
#include <string>

using namespace std;

int main()
{
	
	int n = 0, t = 0, res = 0;
	string s;
	cin>>s;
	
	for (unsigned int i = 0; i < s.length(); i++) 
	{
		
		t = i;
		while (s[t] == s[i]) 
		{

			++i;
			++n;

		}

		if (n % 2 == 0) ++res;

		n = 0;
		--i;

	}

	cout<<res;

	return 0;

}
