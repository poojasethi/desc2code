#include<iostream>
#include<string>
using std::string;
using namespace std;
string s;
int res,temp;
int main()
{
	cin>>s;
	for(int i=0;i<=s.size();i++)
		{
			temp=temp+1;
			if(s[i]!=s[i+1])
				{
					if(temp % 2==0)
						res=res+1;
					temp=0;
				}
			//cout<<t['A']<<"   "<<t['C'];
		}
	cout<<res;
	return 0;
}
