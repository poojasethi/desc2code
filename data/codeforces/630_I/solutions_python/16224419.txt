s = raw_input()
l = s.split(" ")
n = long(l[0])
m = 2 * n - 2
dp = []
for i in range(m + 1):
	dp.append([0] * 2)

dp[0][0] = 1
for i in range(m):
	dp[i + 1][0] += dp[i][0] * 4
	dp[i + 1][1] += dp[i][1] * 4
	if i + n + 1 <= m:
		if i == 0:
			dp[i + n + 1][1] += dp[i][0] * 4 * 3
		else:
			dp[i + n + 1][1] += dp[i][0] * 3 * 3
	if i + n == m:
		if i == 0:
			dp[i + n][1] += dp[i][0] * 4
		else:
			dp[i + n][1] += dp[i][0] * 3

# print dp
print dp[m][1]
