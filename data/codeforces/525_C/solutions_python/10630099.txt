import sys 
import cProfile
import random

def process(lines):
    numberOfSticks = int(lines[0].strip())
    stickLengths = map(int, lines[1].split(' '))
    stickLengths.sort()
    area = 0
    previousStick = None
    previousSide = None
    while len(stickLengths) > 0:
        currentStick = stickLengths.pop()
        if previousStick is not None and previousStick - currentStick < 2:
            currentSide = min([currentStick, previousStick])
            if previousSide is not None:
                area += previousSide * currentSide
                previousSide = None
            else:
                previousSide = currentSide
            previousStick = None
        else:
            previousStick = currentStick

    return str(area)

def start():
    #cProfile.run('print(process([\'10000\', \' \'.join([\'1000000\']*10000)]))')
    print(process(sys.stdin.readlines()))

if __name__ == '__main__':
    start()
