n = int(raw_input())
m = [int(x) for x in raw_input().split()]

m = sorted(m, reverse=True)
l = None
i = 0
s = 0
while i < len(m) - 1:

    if m[i] == m[i + 1]:
        if l is None:
            l = m[i]
        else:
            s += m[i] * l
            l = None
        i += 1
    elif m[i + 1] == m[i] - 1:
        if l is None:
            l = m[i] - 1
        else:
            s += (m[i] - 1) * l
            l = None
        i += 1
    i += 1
print s
