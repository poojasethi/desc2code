n = input()
sticks = map(int, raw_input().split())
sticks.sort()
sticks.reverse()
sides = []
i = 0
while i < len(sticks) - 1:
    if sticks[i] == sticks[i+1] or sticks[i] == sticks[i + 1] + 1:
        sides.append(sticks[i + 1])
        i+=2
    else:
        i+=1

ans = 0
if len(sides) >= 2:
    i = 0
    while i < len(sides) - 1:
        ans += sides[i] * sides[i+1]
        i+=2
print ans
