n = int(raw_input().strip())
numbers = map(int, raw_input().strip().split(' '))
numbers.sort()
prev = -1
sides = []
for i in xrange(len(numbers) - 1, -1, -1):
    num = numbers.pop()
    if num == prev or num == prev - 1:
        sides.append(num)
        prev = -1
    else:
        prev = num
result = 0
for i in xrange(0, len(sides) - 1, 2):
    result += sides[i] * sides[i + 1]
print result
