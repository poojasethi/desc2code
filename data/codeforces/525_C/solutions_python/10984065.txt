n=input()
ls =raw_input().split(' ')
ls=map(int,ls)
ls=sorted(ls)
ans=[]
i=n-2
tmp=0
while i>=0:
    if(abs(ls[i]-ls[i+1])<=1):
        ans.append(min(ls[i],ls[i+1]))
        i-=2
    else:
        i-=1
l=len(ans)
#print ans
if l<2:
    print '0'
else:
    i=0
    while i<=l-2:
        tmp+=ans[i]*ans[i+1]
        i+=2
    print tmp
