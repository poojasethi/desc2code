n=input()
num=sorted(map(int,raw_input().split()),reverse=True)
length=len(num)
pair=[]
i=0
while i<length-1:
    if(num[i]-num[i+1]<=1):
        pair.append(num[i+1])
        i=i+1;
    i=i+1
pair_length=len(pair)
if(pair_length%2!=0):
    pair.append(0)
i=0
area=0
while i<pair_length-1:
    area=area+pair[i]*pair[i+1]
    i=i+2

print area
