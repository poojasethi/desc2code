n = int( raw_input())

s = map(int,raw_input().split()  )

s = sorted( s )

ans =0
i = n-1
#print i
#print s
pre = 0
while i >= 1:
    if s[i]-s[i-1]<=1 :
        if pre == 0 :
            pre = s[i-1]
        else :
            ans += pre * s[i-1]
            pre = 0
        i -= 2
    else:
        i -= 1

print ans
