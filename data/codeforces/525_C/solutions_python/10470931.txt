from collections import Counter
n = input()
sticks = map(int, raw_input().split())
cnt = Counter()
for s in sticks:
	cnt[s] += 1
T = sorted(map(list, cnt.items()))
for i in range(len(T))[1:][::-1]:
	if T[i][1] % 2 and T[i-1][0] == T[i][0] - 1:
		T[i][1] -= 1
		T[i-1][1] += 1

pairs = []
for length, count in T:
	pairs += count/2*[length]

sumArea = 0
while len(pairs) >= 2:
	sumArea += pairs.pop()*pairs.pop()

print sumArea
