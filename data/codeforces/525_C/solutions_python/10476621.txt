



import sys
from collections import defaultdict

sticks = int(sys.stdin.readline().rstrip())
lens = map(int, sys.stdin.readline().rstrip().split())

lens.sort(reverse=True)

p = None
totalArea = 0
i = 0
while i < len(lens) - 1:
    if lens[i] == lens[i+1] or lens[i] - 1 == lens[i+1]:
        if p != None:
            totalArea += p*lens[i+1]
            p = None
            i += 1
        else:
            p = lens[i+1]
            i += 1
    i += 1
    
print totalArea
