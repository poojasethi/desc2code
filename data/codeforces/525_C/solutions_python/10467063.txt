import sys

n = int(sys.stdin.next())
l = sorted(map(int, sys.stdin.next().split(" ")))[::-1]

ss = []

i = 0
while i < n - 1:
    if l[i] - l[i + 1] <= 1:
        ss.append(l[i + 1])
        i += 2
    else:
        i += 1

summa = 0
if len(ss) >= 2:
    for i in range(len(ss) - 1)[::2]:
        summa += ss[i] * ss[i + 1]

print(summa)
