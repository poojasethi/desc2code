n = input()
a = sorted(map(int, raw_input().split()))[::-1]
lengths = []
i=0
while i<n-1:
    if a[i] == a[i+1]:
        lengths.append(a[i])
        i += 2
    elif a[i]-1 == a[i+1]:
        lengths.append(a[i+1])
        i += 2
    else:
        i += 1
if len(lengths)>=2:
    if len(lengths)%2==1:
        lengths = lengths[:-1]
    s=0
    for i in range(len(lengths)/2):
        s += lengths[2*i]*lengths[2*i+1]
    print s
else:
    print 0
