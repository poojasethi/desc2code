def sticks(n, lengths):
    lengths.sort(reverse=True)
    w = 0
    unmatched = set()
    res = 0
    for l in lengths:
        matched = 0
        if l in unmatched:
            matched = l
            unmatched.remove(l)
        elif l + 1 in unmatched:
            matched = l
            unmatched.remove(l+1)
        else:
            unmatched.add(l)

        if matched != 0:
            if w == 0:
                w = matched
            else:
                res += w * matched
                w = 0
    return res

n = int(raw_input())
lengths = map(int, raw_input().split())

print sticks(n, lengths)
