#!/usr/bin/env python
# coding=utf-8
def main():
    n = int(raw_input())
    p = map(int, raw_input().split())
    p = sorted(p, reverse = True)
    l = len(p)
    rc_e = []
    i = 0
    while i < l - 1:
        if p[i] == p[i + 1]:
            rc_e.append(p[i])
            i = i + 1
        elif p[i] - 1 == p[i + 1]:
            rc_e.append(p[i + 1])
            i = i + 1
        i += 1
    r = 0
    i = 0
    while i < len(rc_e):
        if i + 1 < len(rc_e):
            t = rc_e[i] * rc_e[i + 1]
            r += t
            i += 1
        i += 1
            
    print r
main()
