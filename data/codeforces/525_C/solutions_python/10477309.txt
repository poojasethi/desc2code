n=int(raw_input())
t=raw_input()
t=t.split(' ')
s=[int(i) for i in t]
s.sort()
s.reverse()
mult={}
for i in s:
    mult[i]=0
    mult[i-1]=0
f=[]
for i in s:
    mult[i]+=1
    if mult[i]&1==0:
        f.append(i)
        mult[i-1]=0
        continue
    mult[i-1]+=1
max=0
if len(f)>1:
    for i in range(0,len(f),2):
        if i<len(f)-1:
            max+=f[i]*f[i+1]
else:
    max=0
print max
