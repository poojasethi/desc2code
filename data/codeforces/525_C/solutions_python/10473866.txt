import sys

N = int(sys.stdin.readline())
s = map(lambda x : int(x), sys.stdin.readline().split())

pairs = []

s.sort(reverse=True)

buf = -1
for i in s:
  if buf == i:
    pairs.append(i)
    buf = -1
  elif buf - 1 == i:
    pairs.append(i)
    buf = -1
  else:
    buf = i

buf = -1
count = 0
for i in pairs:
  if buf == -1:
    buf = i
  else:
    count += buf * i
    buf = -1


print count
