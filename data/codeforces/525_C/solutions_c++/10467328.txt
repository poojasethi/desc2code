#include<bits/stdc++.h>
using namespace std;
long long a[1000000],i,n,ans=0,k=0,x;
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)
        cin>>a[i];
    sort(a+1,a+n+1);
    for(i=n;i>1;i--)
    {
        if ((a[i]-1==a[i-1]) || (a[i]==a[i-1]))
        {
            k++;
            if (k==2) {ans+=x*a[i-1];k=0;}
            else
            x=a[i-1];
            i--;
        }
    }
    cout<<ans<<endl;
    return 0;
}
