#include<bits\stdc++.h>
using namespace std;
int L[100005];
int main(){
	int n;
	long long int ans = 0, tmp = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d", L + i);
	sort(L, L + n);
	for (int i = n - 1; i - 1 >= 0;){
		if (L[i] - L[i - 1] > 1)
			i--;
		else if (tmp == 0)
			tmp = L[i - 1], i -= 2;
		else
			ans += tmp * L[i - 1], i -= 2, tmp = 0;
	}
	printf("%lld\n", ans);
	return 0;
}
