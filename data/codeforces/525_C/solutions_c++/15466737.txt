
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,A=0;
cin>>n;

vector <LL> v(n),ma;

for(i=0;i<n;i++)
	cin>>v[i];

sort(v.begin(),v.end());


for(i=(n-1);i>0;i--)
	if(v[i]==v[i-1] || v[i]==(v[i-1]+1))
		{ma.push_back(v[i-1]);v[i-1]=-1;}


for(i=0;i<ma.size();i+=2)
	if((i+1)<ma.size())
		A+=ma[i]*ma[i+1];


cout<<A<<endl;
}
