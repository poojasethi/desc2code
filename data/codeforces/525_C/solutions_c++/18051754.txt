#include <bits/stdc++.h>
using namespace std;
int a[100005];
int main() {
	int n; cin >> n; 
	for (int i = 0; i < n; ++i)
		cin >> a[i];
	sort(a, a + n);
	long long area = 0;
	long long total = 0;
	int choose = 0;
	for (int i = n - 1; i > 0; --i) {
		if (a[i] == a[i-1] || a[i] == a[i-1] + 1) {
			if (choose % 2 == 0)
				area = a[i-1];
			else
				total += area * a[i-1];
			choose++;
			--i;
		}
	}
	cout << total << endl;
}
