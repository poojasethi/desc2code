#include <bits/stdc++.h>

using namespace std;

long long int N,ans,ar[100100];

int main(){
	scanf("%lld",&N);
	
	for(long long int i=1;i<=N;i++)
		scanf("%lld",&ar[i]);
	
	sort(ar+1,ar+N+1);
	reverse(ar+1,ar+N+1);
	
	for(long long int i=2,last=0;i<=N;i++){
		if(ar[i-1]-ar[i]<=1){
			ans+=last*ar[i];
			if(!last)
				last=ar[i];
			else
				last=0;
			i++;
		}
	}
	
	printf("%lld\n",ans);
	
	return 0;
}
