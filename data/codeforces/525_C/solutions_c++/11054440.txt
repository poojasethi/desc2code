#include<cstdio>
#include<algorithm>
using namespace std;
long long v[100001],v2[100001];
int main ()
{long long n,i,x=0,k=0,q;
scanf("%I64d",&n);
for(i=1;i<=n;i++)
	scanf("%d",&v[i]);
sort(v+1,v+n+1);
for(i=n;i>1;i--)
	{if(v[i]==v[i-1])
		{x++;
		v2[x]=v[i];
		i--;
		}
	else
		if(v[i]==v[i-1]+1)
			{x++;
			v2[x]=v[i]-1;
			i--;
			}
	}
for(i=1;i<=x;i+=2)
	k+=v2[i]*v2[i+1];
printf("%I64d",k);
return 0;
}
