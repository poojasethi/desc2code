#include <bits/stdc++.h>
using namespace std;
int main() {

ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	int n;
	cin >> n;
	long long a[n];
	for(int i = 0; i < n; i++)
		cin >> a[i];
	long long ans = 0;
	sort(a, a + n);
	int Q = 0;
	for(int i = n - 1; i > 0; i--){
		if(a[i] <= a[i - 1] + 1){
			if(Q == 0){
				Q = a[i - 1];
			}else{
				ans += Q * a[i - 1];
				Q = 0;
			}
			i--;
		}
	}

	cout << ans;


}
