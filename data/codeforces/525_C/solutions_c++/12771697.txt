#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
	int n;
	cin>>n;
	std::vector<int> v(n);
	int i;
	for(i=0;i<n;i++)
	{
		cin>>v[i];
	}
	sort(v.begin(),v.end());
	std::vector<int> x;
	i=n-1;
	while(i>=1)
	{
		if(v[i]-v[i-1]<=1)
		{
			x.push_back(v[i-1]);
			i=i-2;
		}
		else
		{
			i--;
		}
	}
	ll ans=0;
	for(i=0;i+1<x.size();i+=2)
	{
		ans=ans+(ll)x[i]*(ll)x[i+1];
	}
	cout<<ans<<endl;
}
