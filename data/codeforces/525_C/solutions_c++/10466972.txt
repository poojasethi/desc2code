#include<cstdio>
#include<algorithm>
using namespace std;
long long ans;
int l[100005],n;

int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;++i)scanf("%d",l+i);
    sort(l+1,l+n+1);
    for(int i=n-1,x=0,y=0;i>0;--i,x=y=0){
        if(l[i]==l[i+1]||l[i]==l[i+1]-1)x=l[i];
        else continue;
        for(i-=2;i>0&&!y;i--)if(l[i]==l[i+1]||l[i]==l[i+1]-1)y=l[i];
        ans+=(long long)x*y;
    }
    printf("%I64d",ans);
}
