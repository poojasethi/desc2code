#include <cstdio>

const int N = 100100, M = 1000100;

int a[M], c[N];
int n, i, k;
long long s;

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i++)
	{
		int x;
		scanf("%d", &x);
		++a[x];
	}

	for (i = M - 2; i > 0; i--)
		while (a[i] + a[i + 1] >= 2)
		{
			a[i] -= 2 - a[i + 1];
			a[i + 1] = 0;
			c[k++] = i;
		}

	for (i = 0; i + 1 < k; i += 2)
		s += 1LL * c[i] * c[i + 1];

	printf("%I64d\n", s);

	return 0;
}
