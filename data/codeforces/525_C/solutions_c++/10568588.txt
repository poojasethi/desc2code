#include<iostream>
#include<algorithm>
using namespace std;
int a[100001];
int main()
{
	int n;
	long long ans=0,w=0,h=0;
	cin>>n;
	int i;
	for(i=1;i<=n;i++)
		cin>>a[i];
	sort(a+1,a+n+1);
	for(i=n;i>=1;i--)
	{
		if(a[i]-a[i-1]<=1)
		{
			if(w==0)
				w=a[i-1];
			else 
			{
				h=a[i-1];
				ans+=w*h;
				w=0;
				h=0;	
			}
			i--;
		}
	}
	cout<<ans;
	return 0;
}
