#include <stdio.h>
#include <algorithm>

using namespace std;
typedef long long lld;
const int MAXN = 100009;

int n, d;
lld ans, A[MAXN], D[MAXN];

int main()
{
	scanf("%d" ,&n);
	
	for (int i=0; i<n; i++)
		scanf("%lld" ,&A[i]);
	
	sort(A, A+n);
	
	for (int i=n-1; i>0; i--)
		if (A[i]-A[i-1]<2)
			D[d++] = A[i-1], i--;
	
	for (int i=0; i<d; i+=2)
		ans += D[i] * D[i+1];
	
	printf("%lld" ,ans);
}
