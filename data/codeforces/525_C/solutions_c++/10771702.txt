#include<iostream>
#include<algorithm>
using namespace std;
int stick[100100];
int main()
{
	int n, x = 0, i;
	long long sum = 0;
	cin >> n;
	for(i = 0; i < n; ++i)
		cin >> stick[i];
	sort(stick , stick + n);
	for(i = n - 1; i > 0; --i)
		if(stick[i] - stick[i-1] <= 1)
			if(x == 0)
			{
				x = stick[i-1];
				--i;
			}
			else
			{
				sum += (long long) x * stick[i-1];
				--i;
				x = 0;
			}	
	cout << sum;
 	return 0;
}
