#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int i,n;
long long x,ans;
int main(){
    cin>>n;
    vector<int>v(n);
    for(i=0;i<n;++i)
        cin>>v[i];
    sort(v.begin(),v.end());
    for(i=n-2;i>=0;--i)
    if(v[i+1]-v[i]<2){
        if(x){ans+=x*v[i];x=0;}
        else x=v[i];
        i--;
    }
    cout<<ans<<endl;
    return 0;
}
