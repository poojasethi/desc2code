#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,i,j;
	scanf("%lld",&n);
	long long int b[n];
	for(i=0;i<n;i++)
		scanf("%lld",&b[i]);
	sort(b,b+n);
	long long int t[n];
	long long int u=0;
	for(i=n-1;i>0;i--)
	{
		if(b[i]-b[i-1] < 2) 
		{
			t[u]=b[i-1];
			u++;
			i--;
		}
	}
	long long int si=0;
	for(j=0;j<u-1;j+=2)
		si += (long long int) t[j]*t[j+1];
	printf("%lld\n",si);
	return 0;
}
