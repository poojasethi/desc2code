#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,i,j,ans,arr[100001];
	cin>>n;
	for(i=0;i<n;i++)
		cin>>arr[i];
	sort(arr,arr+n);
	i=n-1;
	ans=0;
	while(i>=1)
	{
		
		if(arr[i]==arr[i-1] || arr[i]==arr[i-1]+1)
		{
			for(j=i-2;j>=1;j--)
			{
				if(arr[j]==arr[j-1] || arr[j]==arr[j-1]+1)
				{
					ans+=arr[i-1]*arr[j-1];
					break;	
				}
			}
			i=j-2;
		}
		else
			i--;
	}
	cout<<ans;
}
