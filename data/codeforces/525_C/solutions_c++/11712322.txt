#include <stdio.h>
#include<stdlib.h>
#include<string.h>
#include<algorithm>
using namespace std;
long long n, a[100100], i, m, s;
int main() {
    scanf("%I64d",&n);
    for (i=0;i<n;i++) 
		scanf("%I64d",&a[i]);
    sort(a,a+n);
    for (i=n-2;i>=0; i--)
        if (a[i+1]-a[i] < 2) {
            if (m) 
				s += m*a[i], m = 0;
            else m = a[i];
            i--;
        }
    printf("%I64d\n",s);
}
