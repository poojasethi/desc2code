#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=100010;
long long a[maxn];
int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	sort(a,a+n);
	long long l=0,ans=0;
	for(int i=n-1;i>=0;){
		if((a[i]-a[i-1]==1)||(a[i]==a[i-1])){
				if(l==0)l=a[i-1];
				else ans=ans+l*a[i-1],l=0;
				i=i-2;
		}else --i;
	}
	printf("%I64d",ans);
}
