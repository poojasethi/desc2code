#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n;
	cin >> n;
	long long int arr[n];
	for(int i=0;i<n;++i)
		cin >> arr[i];
	sort(arr,arr+n);
	long long int area=1;
	long long int total=0;
	int cnt=0;
	int i=n-1;
	while(i>0)
{
	if(arr[i]-arr[i-1]>1)
		--i;
	else
{
	area*=arr[i-1];
	++cnt;
	i=i-2;
	
}	
	if(cnt==2)
{
		total+=area;
		cnt=0;
		area=1;
}
}
	cout << total << "\n";
}
