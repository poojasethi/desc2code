#include<cstdio>
#include<algorithm>
using namespace std;
int a[100005],c[100005],cnt,n,i;long long ans;
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%d",&a[i]);
  sort(a+1,a+n+1);
  reverse(a+1,a+n+1);
  for (i=1;i<=n;i++)
    if (a[i+1]==a[i]) c[++cnt]=a[i],i++;
    else if (a[i+1]+1==a[i]) c[++cnt]=a[i+1],i++;
  if (cnt>1)
  for (i=1;i<=cnt;i+=2)
    ans+=c[i]*1ll*c[i+1];
  printf("%I64d\n",ans);
}
