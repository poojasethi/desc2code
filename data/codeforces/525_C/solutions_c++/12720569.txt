#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
#define LL long long
LL ans=0,cur=0;
int n,s[100005],a1=-1;
int main(){
  scanf("%d",&n);
  for(int i=1;i<=n;i++) scanf("%d",&s[i]);
  sort(s+1,s+n+1);
  for(int i=n-1;i>=1;i--){
    if(s[i+1]-s[i]<=1){
      if(a1==-1) a1=0,cur=s[i];
      else if(a1==0) a1=-1,ans+=cur*s[i];
      s[i]+=29;
    }
  }
  printf("%lld\n",ans);
  return 0;
}
