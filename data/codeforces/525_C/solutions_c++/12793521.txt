#include<bits/stdc++.h>
using namespace std;
typedef long long ll;


int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin>>n;
	ll a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	bool flag=false;
	ll x=0,ans=0;
	for(int i=n-1;i>0;i--)
	{
		if(a[i]-a[i-1]<=1)
		{
			if(flag)
			{
				ans+=(x*a[i-1]);
				i--;
				flag=false;
			}
			else
			{
				flag=true;
				x=a[i-1];
				i--;
			}
		}
	}
	cout<<ans<<"\n";
	return 0;
}
