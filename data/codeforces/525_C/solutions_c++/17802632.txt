#include<cstdio>
#include<algorithm>
using namespace std;
const int N=100005;
int a[N];
int n;
int b[N],cnt;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	for (int i=n;i>=2;i--)
	{
		if (a[i]<=a[i-1]+1)
		{
			b[++cnt]=min(a[i],a[i-1]);
			i--;
		}
	}
	long long ans=0;
	for (int i=2;i<=cnt;i+=2)
	{
		ans+=(long long) b[i]*b[i-1];
	}
	printf("%I64d\n",ans);
}
