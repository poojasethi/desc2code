#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

long long n, l[111111], ans;
vector<long long> v;

int main()
{
	cin >> n;
	for(int i=0; i<n; i++)
		cin >> l[i];
	sort(l, l+n);

	for(int i=n-1; i > 0; i--)
	{
		if(l[i] == l[i-1] || l[i]-1 == l[i-1])
		{
			v.push_back(l[i-1]);
			i--;
		}
	}

	for(int i=1; i<v.size(); i+=2)
		ans += v[i]*v[i-1];

	cout << ans << endl;
	
	return 0;
}
