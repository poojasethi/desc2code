#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long LL;

int main()
{
	LL n,l[100005],k[100005];
	scanf("%I64d",&n);
	for(int i=1;i<=n;i++)
		scanf("%I64d",&l[i]);
	sort(l+1,l+n+1);
	int p=0;
	for(int i=n;i>=2;i--)
	{
		if(l[i]-l[i-1]<=1)
		{
			p++;
			k[p]=l[i-1];
			i--;
		}
	}
	LL ans=0;
	for(int i=1;i<p;i++)
	{
		ans+=k[i]*k[i+1];
		i++;
	}
	printf("%I64d\n",ans);
	
	return 0;
}
