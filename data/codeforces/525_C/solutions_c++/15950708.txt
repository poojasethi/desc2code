#include<algorithm>
#include<cstdio>
using namespace std;
int n,a[100010];
long long l,ans;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%d",&a[i]);
    sort(a,a+n);
    for(int i=n-1;i>=1;i--)
        if(a[i]==a[i-1] || a[i]==a[i-1]+1){
            if(l)ans+=l*a[--i],l=0;
            else
                l=a[--i];
        }
    printf("%lld",ans);
    return 0;
}
