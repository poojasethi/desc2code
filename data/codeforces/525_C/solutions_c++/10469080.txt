#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int maxn=100010;
ll a[maxn],n;
ll Max()
{
	ll res=a[--n];
	while(n--)
		if(res-1<=a[n]) return a[n];
		else res=a[n];
	return 0;
}
int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0);
	cin>>n;
	for(int i=0;i<n;i++) cin>>a[i];
	sort(a,a+n);
	ll ans=0,x,y;
	while(n>3)
	{
		x=Max();
		y=Max();
		if(y) ans+=x*y;
	}
	cout<<ans<<endl;
	return 0;
}
