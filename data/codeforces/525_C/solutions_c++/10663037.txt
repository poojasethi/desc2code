#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int i,j,n;
	unsigned long long f=0,b=0;
	int s[100001]={0};
	cin>>n;
	for(i=0;i<n;i++)cin>>s[i];
	sort(s,s+n);
	for(j=n-2;j>=0;j--)
		if(s[j]==s[j+1]||s[j]+1==s[j+1])
		{
			if(b){f+=b*s[j];b=0;j--;}
			else{b=s[j];j--;}
		}
	cout<<f;
	return 0;
}
