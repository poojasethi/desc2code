#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
int a1[100000];
int main()
{
    int n,a;bool bl=0;ll o=0;
    scanf("%d",&n);
    for(int i=0;i<n;++i)scanf("%d",a1+i);
    sort(a1,a1+n);
    for(int i=n-1;i>0;i-=2,bl^=1)
    {
        while(i&&a1[i]-a1[i-1]>1)--i;
        if(!i)break;
        if(bl)o+=(ll)a*a1[i-1];
        else a=a1[i-1];
    }
    printf("%I64d\n",o);
}
