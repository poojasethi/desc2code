#include<bits/stdc++.h>
using namespace std;
typedef __int64 ll;
int n; ll l[1000005];

int main(void){
	int i,j;
	while(~scanf("%d",&n)){
		for(i=0;i<n;i++) scanf("%I64d",&l[i]);
		sort(l,l+n);
		int flag=-1; ll ans=0,temp=0;
		for(i=n-1;i>-1;){
			if(l[i]==l[i-1] || l[i]==l[i-1]+1){
				if(!temp) temp=l[i-1];
				else ans+=temp*l[i-1],temp=0;
				i-=2;
			}
			else i--;
		}
		printf("%I64d\n",ans);
	}
	return 0;
}
