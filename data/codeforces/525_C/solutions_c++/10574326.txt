#include <iostream>
#include <algorithm>
using namespace std;
long long n,a[100002],m,s;
int main(){
    cin>>n;
    for (int i=1;i<=n;i++) 
    	cin>>a[i];
    sort(a+1,a+n+1);
    for (int i=n-1;i>=1;i--)
        if (a[i+1]-a[i]<2) {
            if (m){s+=m*a[i];m=0;} else m=a[i];
            i--;
        }
    cout<<s;
    return 0;
}
