#include <bits/stdc++.h>




using namespace std;



int main()
{
    long long n , s[200010],i;

    long long ans=0,c=0;
    cin >> n;
    for(i=0;i<n;i++) cin >> s[i];

    sort(s,s+n);
    for(i=n-2;i>=0;i--)
    {
        if(s[i]==s[i+1]||s[i]+1==s[i+1])
        {

            if(c) {ans+=c*s[i]; c=0; }
            else {c=s[i]; }

            i--;

        }


    }


    cout << ans;





    return 0;
}
