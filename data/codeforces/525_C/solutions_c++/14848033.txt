#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	long long int n;
	cin>>n;
	long long int a[n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	long long int area=0;
	long long int max1=0,max2=0;
	for(int i=n-1;i>=0;i--)
	{
		if(a[i]==a[i-1]||a[i]-1==a[i-1])
		{	max2=max1;
			max1=a[i-1];

			i--;
		}
		if(max2>0)
		{	area+=max1*max2;
			max1=0;
			max2=0;}
		
	}

	cout<<area;
	return 0;
}
