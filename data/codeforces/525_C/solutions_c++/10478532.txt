# include <bits/stdc++.h>
using namespace std;
# define fi cin
# define fo cout
long long s[1234567];
int main(void)
{
    int n;
    fi>>n;
    for (int i = 1;i <= n;++i) fi>>s[i];
    sort(s+1,s+1+n);
    long long mx = 0,ans = 0;
    for (int i = n;i>1;--i)
    if (s[i] == s[i-1] || s[i-1] == s[i]-1)
    {
        if (mx) ans += mx * s[i-1],mx=0;else mx = s[i-1];--i;
    }
    return fo << ans << '\n',0;
}
