#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int j=0,i,n,f1=0,f2=0;
	cin>>n;
	long long int sum=0,a[n],b[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		b[i]=0;
	}
	sort(a,a+n);
	for(i=n-1;i>0;i--)
	{
		if((a[i]-a[i-1]==1)||(a[i]-a[i-1]==0))
		{
			b[j]=a[i-1];
			j++;
			i--;
		}
	}
	int f=j-1;
	for(i=0;i<f;i=i+2)
	{
		sum+=b[i+1]*b[i];
	}
	cout<<sum;
	return 0;
}
