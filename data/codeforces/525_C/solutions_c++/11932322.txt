#include <cstdio>
#include <algorithm>
using namespace std;
long long n,ans,a[100010],l;
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%I64d",&a[i]);
	sort(a+1,a+1+n);
	long long l = 0,ans = 0;
	for (int i=n;i;){
		if (a[i]==a[i-1] || a[i]-a[i-1]==1){
			if (l==0) l = a[i-1];
			else {ans+=l*a[i-1];l = 0;}
			i-=2;
		}
		else i--;
	}
	printf("%I64d",ans);
}
