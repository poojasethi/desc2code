#include<iostream>
#include<stdio.h>
#include<cstring>
#include<algorithm>
#include<cctype>
using namespace std;
int n,m,i,j,pri[100000]={0},e=0,matr,sum[505][2]={0},ans=0x7fffffff;
bool h[110006]={0};
int main(){
	cin>>n>>m;
	for(i=2;i<=110005;i++)
		if (!h[i]){
			h[i]=1;
			pri[++e]=i;
			for(j=i+i;j<=110005;j+=i)
				h[j]=1;
		}
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++){
			cin>>matr;
			int f=pri[lower_bound(pri+1,pri+e,matr)-pri]-matr;
			sum[i][1]+=f;
			sum[j][0]+=f;
		}
	for(i=1;i<=n;i++)
		ans=min(sum[i][1],ans);
	for(i=1;i<=m;i++)
		ans=min(sum[i][0],ans);
	cout<<ans;
	return 0;
}
