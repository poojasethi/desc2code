#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<map>
#include<vector>
#include<cmath>
using namespace std;

int b[120005];
int a[505][505];

int main()
{
	int i,j,n,m;
	b[1]=1;
	for(i=2;i<=110000;i++)
		if(!b[i])
		{
			for(j=i+i;j<=110000;j+=i)
				b[j]=1;
		}
	cin>>n>>m;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			int x,c=0;
			cin>>x;
			for(int k=x;;k++,c++) if(!b[k]) break;
			a[i][j]=c;
		}
	int ans=10000000;
	for(i=0;i<n;i++)
	{
		int t=0;
		for(j=0;j<m;j++) t+=a[i][j];
		if(t<ans) ans=t;
	}
	for(j=0;j<m;j++)
	{
		int t=0;
		for(i=0;i<n;i++) t+=a[i][j];
		if(t<ans) ans=t;
	}
	cout<<ans<<endl;
}
