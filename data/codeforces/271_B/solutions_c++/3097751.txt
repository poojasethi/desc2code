#include<stdio.h>
int a[501][501],p[100010];
int next(int n)
{
    int i;
    for(i=n;p[i];i++);
    return i;
}
int main()
{
    int m,n,i,j,s,min=10000000;
    p[0]=p[1]=1;
    for(i=2;i*i<100010;i++)
    if(p[i]==0)
    for(j=i*2;j<100010;j+=i)
    p[j]=1;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++)
    for(j=0;j<m;j++)
    {
        scanf("%d",&a[i][j]);
        a[i][j]=next(a[i][j])-a[i][j];
        a[n][j]+=a[i][j];
        a[i][m]+=a[i][j];
    }
    for(i=0;i<n;i++)
    if(a[i][m]<min)
    min=a[i][m];
    for(i=0;i<m;i++)
    if(a[n][i]<min)
    min=a[n][i];
    printf("%d\n",min);
    return 0;
}
