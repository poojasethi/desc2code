#include <iostream>
#include <cstdio>
using namespace std;
#define MAX 100010
int n, m, x, F[MAX], R[MAX], C[MAX], ans;

int main() {
	F[0] = F[1] = 1;
	for (int i = 2; i * i < MAX; i++) {
		if (F[i] == 0) {
			for (int j = i * i; j < MAX; j += i) {
				F[j] = 1;
			}
		}
	}
	
	scanf("%d %d", &n, &m);
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			scanf("%d", &x);
			int k = 0;
			while (F[x + k]) {
				k++;
			}
			R[i] += k;
			C[j] += k;
		}
	}
	ans = R[0];
	for (int i = 0; i < n; i++) {
		if (ans > R[i])
			ans = R[i];
	}
	for (int i = 0; i < m; i++) {
		if (ans > C[i])
			ans = C[i];
	}
	printf("%d", ans);
	return 0;
}
