#include<cstdio>
#include<algorithm>
using namespace std;

int f[111111], p;
int n, m, v, ans;
int r[555], c[555];

int main() {
	f[0]=f[1]=2;
	for(int i=2;i<111111;i++)
		if (!f[i])
			for(int j=i+i;j<111111;j+=i)
				f[j]=true;
	p = 111110;
	while(f[p])p--;
	for(int i=p-1;i>=2;i--) {
		if(!f[i]) p = i;
		f[i] = p;
	}
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++) {
			scanf("%d", &v);
			r[i] += f[v]-v;
			c[j] += f[v]-v;
		}
	ans = r[0];
	for(int i=0;i<n;i++) ans = min(ans, r[i]);
	for(int i=0;i<m;i++) ans = min(ans, c[i]);
	printf("%d\n", ans);
	
	return 0;
}
