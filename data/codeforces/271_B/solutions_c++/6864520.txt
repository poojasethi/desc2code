#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int a1[500],a2[500];
const int maxn=100004;
int p[maxn],l=0;
bool num[maxn];
void prime()
{
	int sqt=(int)sqrt(maxn);
	for(int i=2;i<=sqt;++i)
	if(!num[i])
	{
		p[l++]=i;
		for(int j=i*i;j<maxn;j+=i)
			num[j]=1;
	}
	for(int i=sqt;i<maxn;++i)
	if(!num[i])p[l++]=i;
}
int main()
{
	prime();
	int n,m,x,o=2147483647;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;++i)
	for(int j=0;j<m;++j)
	{
		scanf("%d",&x);
		x=*lower_bound(p,p+l,x)-x;
		a1[i]+=x;a2[j]+=x;
	}
	for(int i=0;i<n;++i)
	if(a1[i]<o)o=a1[i];
	for(int i=0;i<m;++i)
	if(a2[i]<o)o=a2[i];
	printf("%d\n",o);
}
