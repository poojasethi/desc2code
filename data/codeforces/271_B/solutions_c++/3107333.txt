#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
int n,m,a[505][505],ans=1000000000;
const int N=100050;
int p[N]={0};
//freopen("i.txt","r",stdin);
p[1]=1;
for (int i=2;i<N;i++)
  if (p[i]==0)  for (int j=i+i;j<N;j+=i) p[j]=1;
cin>>n>>m;
for (int i=0;i<n;i++)
 for (int j=0;j<m;j++)
  { int temp; cin>>temp;
    int k=0;
    while (p[temp+k]==1) k++;
   a[i][j]=k;
  }
for (int i=0;i<n;i++)
{ int s=0;
 for (int j=0;j<m;j++) s+=a[i][j];
  ans=min(ans,s);
}
for (int j=0;j<m;j++)
{ int s=0;
  for (int i=0;i<n;i++) s+=a[i][j];
  ans=min(ans,s);
}
cout<<ans;
return 0;
}
