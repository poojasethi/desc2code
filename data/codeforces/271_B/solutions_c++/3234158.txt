#include<stdio.h>
const int N=200000;
int f[200010];
int a[510][510];
int main()
{
	int i,j,n,m,x,y,min,count;
	f[0]=1;f[1]=1;
	for(i=2;i<1000;i++)
		if(f[i]==0)
		{
			for(j=i+i;j<=N;j+=i)
				f[j]=1;
		}
	scanf("%d %d",&n,&m);
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			scanf("%d",&a[i][j]);
			if(a[i][j]>1)
			{
				y=a[i][j];
				while(f[y]==1)
				{
					y++;
				}
				a[i][j]=y-a[i][j];
			}
		}
	min=9999999;
	for(i=0;i<n;i++)
	{
		count=0;
		for(j=0;j<m;j++)
			count+=a[i][j];
		if(count<min)
			min=count;
	}
	for(j=0;j<m;j++)
	{
		count=0;
		for(i=0;i<n;i++)
			count+=a[i][j];
		if(count<min)
			min=count;
	}
	printf("%d\n",min);
	return 0;
}
