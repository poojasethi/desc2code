#include <iostream>
#include <algorithm>
using namespace std;

const int MAXN=500+10, N=1000*100+10;
bool p[N];
int row[MAXN], column[MAXN];

int main() {
    p[0]=p[1]=true;
    for (int i=2; i*i<N; i++)
        if (!p[i])
            for (int j=i*i; j<N; j+=i)
                p[j]=true;
    int n, m;
    cin>>n>>m;
    for (int i=0; i<n; i++)
        for (int j=0; j<m; j++){
            int x, k=0;
            cin>>x;
            for (x; p[x+k]; ++k);
            row[i]+=k;
            column[j]+=k;
        }
    cout<<min(*min_element(row, row+n), *min_element(column, column+m))<<endl;
    return 0;
}
