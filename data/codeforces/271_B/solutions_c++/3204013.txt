#include <iostream>
#include <algorithm>
using namespace std;

#define rep(i,n) for (int i = 0; i < int(n); ++i)
const int N = 2*1e5;
bool p[N] = { true, true };
int u[500], v[500];

int main() {
    for (int i = 2; i*i < N; ++i)
        if (!p[i]) for (int j = i*i; j < N; j += i) p[j] = true;

    int n, m;
    cin >> n >> m;
    rep (i,n) rep (j,m) {
        int x, k = 0;
        for (cin >> x; p[x+k]; ++k) ;
        u[i] += k;
        v[j] += k;
    }
    cout << min(*min_element(u,u+n), *min_element(v,v+m)) << endl;
}
