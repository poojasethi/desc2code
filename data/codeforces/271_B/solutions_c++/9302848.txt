#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
bool p[100010];
int row[510],col[510],i,j,n,m,a,k;
int main(){
	memset(row,0,1020*sizeof(int));
	memset(col,0,1020*sizeof(int));
	p[0]=p[1]=true;
	for(i=2;i*i<100010;i++){
		if(!p[i]){
			for(j=i*i;j<100010;j+=i)
				p[j]=true;
		}
	}
	cin>>n>>m;
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			k=0;
			cin>>a;
			while(1){
				if(p[a+k]==false){
					row[i]+=k;
					col[j]+=k;
					break;
				}
				else
					k++;
			}
		}
	}
	cout<<min(*min_element(row,row+n),*min_element(col,col+m))<<endl;
	return 0;
}
