#include <bits/stdc++.h>
#define N 100100
using namespace std;

int a[505], b[505],n ,m, x;
bool t[N] = {1,1};

int main()
{
    ios_base::sync_with_stdio(false);
    for(int i = 2; i*i < N; i++) if(!t[i])
        for(int j = i*i; j <N; j+=i) t[j] =1;

    cin >> n >> m;
    for(int i = 0; i < n; i++)
        for(int j = 0; j < m; j++)
        {
            int k = 0;
            for(cin >> x; t[x+k]; k++);
            a[i]+=k;
            b[j]+=k;
        }
   cout << min(*min_element(a, a+n), *min_element(b, b+m)) << endl;
}
