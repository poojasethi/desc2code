#include<stdio.h>
int main()
{
	bool prime[110005];
	int n,m,i,j,a[505][505],ans[505][505],min=1000000000,w;
	for(i=1;i<=110000;i++)
		prime[i]=true;
	prime[1]=false;
	for(i=2;i<=110000;i++)
	{
		if(!prime[i])continue;
		for(j=2;j<=110000/i;j++)
			prime[i*j]=false;
	}
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			scanf("%d",&a[i][j]);
			ans[i][j]=0;
			while(!prime[a[i][j]+ans[i][j]])ans[i][j]++;
		}
	for(i=1;i<=n;i++)
	{
		w=0;
		for(j=1;j<=m;j++)
			w+=ans[i][j];
		if(min>w)min=w;
	}
	for(j=1;j<=m;j++)
	{
		w=0;
		for(i=1;i<=n;i++)
			w+=ans[i][j];
		if(min>w)min=w;
	}
	printf("%d",min);
	return 0;
}
