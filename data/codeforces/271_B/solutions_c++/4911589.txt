#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    const int N = 100100;
    bool t[N] = { true, true };
    int a[500], b[500], n, m, i, j, x;
    for (i = 2; i * i < N; ++i)
        for (j = i * i; j < N; j += i)
            t[j] = true;//composite numb
    cin >> n >> m;
    for (i = 0; i < n; ++i)
        for (j = 0; j < m; ++j)
        {
            int  k = 0;
            for ( cin >> x; t[x + k]; ++k);//until reach a prime
            a[i] += k;//how many step we need,row
            b[j] += k;//for cloum
        }
    cout << min(*min_element(a, a + n), *min_element(b, b + m)) << endl;
}
