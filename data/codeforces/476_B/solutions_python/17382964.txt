import math
a=raw_input()
b=raw_input()
one=0
for i in a:
	if i=='+':
		one+=1
req=0
for i in b:
	if i=='?':
		req+=1
	if i=='+':
		one-=1
if(req-one<0 or one<0):
	print("0")
else:		
	nCr=math.factorial(req)/(math.factorial(one)*math.factorial(req-one))
	total=2**req
	ans=nCr/float(total)
	print(ans)
