def pos(a):
  return sum([1 if c == '+' else -1 for c in a])

def fill(b, c, n):
  ins = bin(n)[2:].zfill(c)
  res = []
  i = 0
  for x in b:
    if x != '?':
      res.append(x)
    else:
      res.append('+' if ins[i] == '1' else '-')
      i += 1
  return ''.join(res)

def solve(a, b):
  pa = pos(a)
  cb = b.count('?')
  t, c = 2**cb, 0
  for i in range(t):
    if pa == pos(fill(b, cb, i)):
      c += 1
  return c / float(t)

a = raw_input()
b = raw_input()
print solve(a, b)
