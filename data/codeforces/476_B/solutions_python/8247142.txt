s1=raw_input()
n=len(s1)
x=0
for i in s1:
    if i=='-':
        x-=1
    else:
        x+=1
s2=raw_input()
dp=[[0]*25 for i in range(25)]
dp[0][12]=1
for i in range(1,1+n):
    for j in range(1,24):
        if s2[i-1]=='-':
            dp[i][j]=dp[i-1][j+1]
        if s2[i-1]=='+':
            dp[i][j]=dp[i-1][j-1]
        if s2[i-1]=='?':
            dp[i][j]=0.5*dp[i-1][j-1]+0.5*dp[i-1][j+1]
print dp[n][12+x]
