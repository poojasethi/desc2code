import math
def nCr(n,r):
    f = math.factorial
    ret = f(n) / f(r) / f(n-r)
    return ret
def solve():
	x= raw_input()
	y= raw_input()
	n = len(x)
	posx = x.count('+') - x.count('-')
	posy = y.count('+') - y.count('-')
	un = y.count('?')
	if un==0 and posx==posy :
		return 1.0
	delta = abs(posx - posy)
	if un < delta:
		return 0.0
	p=(delta+un)/2
	q= un-p
	return nCr(un,p)*(1.0/(2.0**un))
print "%0.12f"%solve()
