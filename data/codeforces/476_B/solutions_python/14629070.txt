s1 = raw_input()
s2 = raw_input()

pos1 = sum([1 if c == '+' else -1 for c in s1])
pos2 = sum([{'+': 1, '-': -1, '?': 0}[c] for c in s2])
num = sum([1 if c == '?' else 0 for c in s2])

diff = abs(pos1 - pos2)

def fact(n):
    f = 1
    for i in range(1, n+1):
        f *= i
    return f

def choose(n, k):
    return fact(n)/fact(k)/fact(n-k)

if diff + num % 2 == 1 or diff > num:
    print 0
else:
    x = (diff + num) / 2
    print choose(num, x) * 1.0 / (2 ** num)
