from sys import stdin

def f(x):
    return x * f(x - 1) if x > 1 else 1.0

s1, s2 = [x.strip() for x in stdin.readlines()]
n = len(s1)
p, q = s2.count("?"), abs((s1.count("+") - s1.count("-")) - (s2.count("+") - s2.count("-")))
a, b = (p + q) / 2, (p - q) / 2
print "%.10f" % (f(a + b) / f(a) / f(b) / (2.0 ** p) if p >= q and (p + q) % 2 == 0 else 0)
