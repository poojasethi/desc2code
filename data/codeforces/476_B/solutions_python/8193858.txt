f = raw_input().strip()
s = raw_input().strip()

p = 0
for c in f:
  if c == '+':
    p += 1
  else:
    p -= 1
    
r = {0: 1}

for c in s:
  d = {}
  for k,v in r.iteritems():
    if c == '+':
      d[k+1] = d.get(k+1, 0) + v
    elif c == '-':
      d[k-1] = d.get(k-1, 0) + v
    else:
      d[k+1] = d.get(k+1, 0) + v*0.5
      d[k-1] = d.get(k-1, 0) + v*0.5

  r = d
  
print d.get(p, 0)
