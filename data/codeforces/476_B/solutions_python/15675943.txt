cmd = raw_input()
rec = raw_input()

cmdP = cmd.count('+')
recP = rec.count('+')
recQ = rec.count('?')

if recP + recQ < cmdP or recP > cmdP:
    ans = 0.0
else :
    ans = 1.0
    for i in range(1, 1 + cmdP - recP):
        ans *= 1.0 * (recQ - i + 1) / i
    ans /= 1 <<  recQ

print '{:.9f}'.format(ans)
