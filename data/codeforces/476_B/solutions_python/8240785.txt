from math import factorial

s = raw_input()
s_ = raw_input()

n = len(s)
a = sum([1 for x in s if x == '+'])
b = n - a

a_ = sum([1 for x in s_ if x == '+'])
b_ = sum([1 for x in s_ if x == '-'])

if(a_ > a or b_ > b):
    print(float(0))
else:
    c = n - a_ - b_
    ans = factorial(c) / factorial(a - a_) / factorial(b - b_)
    print(float(ans) / float(pow(2, c)))
