from sys import stdin
def fact(x):
	res = 1
	for i in range(1, x + 1):
		res *= i
	return res

def C(n, m):
	return fact(m) / (fact(n) * fact(m - n))
n, m = [x.strip() for x in stdin.readlines()]
sn = n.count('+') - n.count('-')
sm = m.count('+') - m.count('-')
sq = m.count('?')
diff = sn - sm
if (diff + sq) % 2 or abs(diff) > sq:
	print 0.0
else:
	a = (diff + sq) / 2
	print float(C(a, sq)) / float(2 ** sq)
