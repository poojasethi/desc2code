import sys

COM = sys.stdin.readline()
REC = sys.stdin.readline()

N = len(COM)

com_pos = COM.count('+') - COM.count('-')
rec_pos = REC.count('+') - REC.count('-')

to_move = abs(com_pos - rec_pos)
q_count = REC.count('?')

def c(n, k):
  if k == 0 or k == n:
    return 1
  else:
    return c(n-1, k-1) + c(n-1, k)

if q_count - to_move % 2 == 1 or q_count < to_move:
  print 0
else:
  k = abs(q_count - to_move) / 2
  print (0.5 ** q_count) * c(q_count, k)
