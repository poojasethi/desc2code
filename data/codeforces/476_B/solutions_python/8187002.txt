import math
s1 = raw_input()
s2 = raw_input()

def nCr(n,r):
    f = math.factorial
    return f(n) / f(r) / f(n-r)

c1=0
for i in s1:
    if i == "+":
        c1+=1
    else:
        c1-=1
c2=0
x = 0
for i in s2:
    if i == "+":
        c2+=1
    elif i == "-":
        c2-=1
    else:
        x+=1
if c2+x < c1 or c2-x > c1:
    print 0
else:
    if not((x+c1-c2)%2 == 0):
        print 0
    else:
        print nCr(x,(x+c1-c2)/2)/float(2**x)
