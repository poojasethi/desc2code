def comb(n, k):
    if 0 <= k <= n:
        ntok = 1
        ktok = 1
        for t in xrange(1, min(k, n - k) + 1):
            ntok *= n
            ktok *= t
            n -= 1
        return ntok / ktok
    else:
        return 0
s1 = raw_input().strip()
s2 = raw_input().strip()
p1 = s1.count("+") - s1.count("-")
p2 = s2.count("+") - s2.count("-")
k = s2.count("?")
n = (abs(p1 - p2) + k) / 2
print "%.12f" % (comb(k, n) / float(2 ** k))
