from operator import mul    # or mul=lambda x,y:x*y
from fractions import Fraction

def choose(n,k):
  return int( reduce(mul, (Fraction(n-i, i+1) for i in range(k)), 1) )

a, b = raw_input(), raw_input()
# a, b = '+++', '??-'
q = b.count('?')
va = a.count('+')-a.count('-')
vb = b.count('+')-b.count('-')
if q == 0 and va == vb:
  print 1
else:
  r = (q + va - vb)/2
  if abs(va-vb) > q:
    print 0
  else:
    print float(choose(q,r))/(2**q)
