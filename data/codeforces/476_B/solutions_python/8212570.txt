from math import factorial

def choose(n,k):
    return factorial(n)/(factorial(n-k)*factorial(k))

s1 = raw_input()
s2 = raw_input()
a,b = s1.count('+'),s1.count('-')
c,d,e = s2.count('+'),s2.count('-'),s2.count('?')
if a-c < 0 or b-d < 0:
    print 0.0
elif a-c == b-d == e == 0:
    print 1.0 
else:
    print choose(e,a-c)/2.0**e
