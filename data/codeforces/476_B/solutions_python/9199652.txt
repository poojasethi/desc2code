dp = [[0] * 30 for _ in range(30)]

s1 = raw_input()
s2 = raw_input()

p = s1.count('+') - s1.count('-')

dp[0][10] = 1
for i in range(len(s2)):
    for j in range(1, 29):
        if s2[i] in ['-', '?']:
            dp[i+1][j-1] += dp[i][j]
        if s2[i] in ['+', '?']:
            dp[i+1][j+1] += dp[i][j]

print 1.0 * dp[len(s2)][p+10] / sum(dp[len(s2)])
