def C(n, a):
    ans = 1
    for i in range (1, a+1, 1):
        ans = ans * (n-i+1) / i
    return ans

s1, s2 = raw_input(), raw_input()
n = s2.count('?')
k = abs(s1.count('+') - s1.count('-') - s2.count('+') + s2.count('-'))
if (n + k) % 2 != 0:
    print 0
else:
    a = (n + k) / 2
    print C(n, a) / 1.0 / 2**n
