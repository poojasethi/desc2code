def fact(a):
    res = 1
    while a > 0:
        res *= a
        a -= 1
    return res
    
s1 = raw_input()
s2 = raw_input()
p1 = s1.count("+")-s1.count("-")
p2 = s2.count("+")-s2.count("-")
q = s2.count("?")
d = abs(p1-p2)

if d > q or d%2 != q%2:
    print "%.9f"%0
else:
    f,b = d+(q-d)/2,(q-d)/2
    print "%.9f"%((fact(q)/fact(f)/fact(b))*1.0/2**q)
