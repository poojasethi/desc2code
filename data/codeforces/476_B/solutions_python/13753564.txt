s1 = raw_input()
s2 = raw_input()
fp = 0
for c in s1:
	if c == '+': fp += 1
	else: fp -= 1
c = 0
def f(i,p):
	global c
	if i == len(s2):
		c += 1
		return int(p == fp)
	if s2[i] == '+': return f(i+1, p + 1)
	if s2[i] == '-': return f(i+1, p - 1)
	if s2[i] == '?': return f(i+1, p - 1) + f(i+1, p+1)
print "%.10lf" % (float(f(0,0))/float(c))
