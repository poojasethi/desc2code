s1 = raw_input()
s2 = raw_input()
pos = s1.count("+") - s1.count("-")
x = [0]
y = []

for char in s2:
    for p in x:
        if   char == '+':
            y.append(p+1)
        elif char == '-':
            y.append(p-1)
        else:
            y.append(p+1)
            y.append(p-1)
    x = y
    y = []

print (x.count(pos) > 0) * x.count(pos) * .5 ** s2.count("?")
