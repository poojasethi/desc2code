from sys import stdin
fa = [1]
for i in xrange(1,55):
 fa.append(i*fa[i-1])
def c(n,r):
 if n<r:
  return 0
 return fa[n]/(fa[r]*fa[n-r])
a=stdin.readline().strip()
b=stdin.readline().strip()
co = 0
for i in a:
 if i=='+':
  co+=1
my = 0
qu = 0
for i in b:
 if i=='+':
  my+=1
 elif i=='?':
  qu+=1
l = len(a)
total = 1<<qu

if my>co:
 ans = 0
else:
 ans = c(qu,co-my)

print (ans*1.0)/total
