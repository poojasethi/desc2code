def fact(x):
	res = 1
	for i in range(1, x + 1):
		res *= i
	return res

def C(n, m):
	return fact(m) / (fact(n) * fact(m - n))
n = raw_input()
m = raw_input()
sn = 0
for c in n:
	if c == '+':
		sn += 1
	else:
		sn -= 1
sm, sq = 0, 0
for c in m:
	if c == '+':
		sm += 1
	elif c == '-':
		sm -= 1
	else:
		sq += 1
diff = sn - sm
if (diff + sq) % 2 or abs(diff) > sq:
	print 0.0
else:
	a = (diff + sq) / 2
	print float(C(a, sq)) / float(2 ** sq)
