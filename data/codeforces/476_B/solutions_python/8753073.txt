sent=raw_input()
received=raw_input()

possible_positions = [0]
end_position = 0
for i in range(len(received)):
	pp = []
	for p in possible_positions:
		if received[i] == '+' or received[i] == '?':
			pp.append(p+1)
		if received[i] == '-' or received[i] == '?':
			pp.append(p-1)
	if sent[i] == '+': end_position += 1
	if sent[i] == '-': end_position -= 1
	possible_positions = pp

print float(possible_positions.count(end_position))/float(len(possible_positions))
