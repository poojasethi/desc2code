import sys

S = sys.stdin.readline()
T = sys.stdin.readline()

c0, c1 = 0, 0
for k in S :
    if (k == '+') :
        c0 += 1 
    if (k == '-') : 
        c1 += 1

x, y, z = 0, 0, 0
for k in T : 
    if (k == '+') : 
        x += 1
    if (k == '-') : 
        y += 1
    if (k == '?') : 
        z += 1

if x > c0 or y > c1 or x + z < c0 or y + z < c1: 
    print 0.0
    exit()

ans = 0.5 ** z
for i in range(1, c0 - x + 1) : 
    ans *= 1.0 * (z - i + 1) / i
print ans
