

def f(s):
    return (sum( i=='+' for i in s),sum( i=='-' for i in s))

def F(s):
    return (sum( i=='+' for i in s)-sum( i=='-' for i in s))
P = [""]
x = raw_input()
y = raw_input()
L = len(x)




a,b = f(x)
c,d = f(y)
z = L-(c+d)
SUM = (a-b)-(c-d)
for i in xrange(z):
    T = list(P)
    P = []
    for i in T:
        P+=[i+"+"]
        P+=[i+"-"]

P = map(F,P)
print sum(i==SUM for i in P)*1.0/len(P)
