#include<cstdio>
#include<cstring>
char a[20],b[20];
int t0,i,j,x,n,m,t;double t1,t2;
int main(){
    scanf(" %s %s",a,b);
    n=strlen(a);
    for (i=0;i<n;i++){
        if (a[i]=='+') t++;else t--;
        if (b[i]=='-') t++;
        else if (b[i]=='+') t--;
        else m++;
    }
    for (i=0;i<(1<<m);i++){
        x=0;
        for (j=0;j<m;j++) x+=(i&(1<<j))?1:-1;
        if (x==t) t0++;
    }
    t1=t0;t2=(1<<m);
    printf("%.12lf\n",t1/t2);
    return 0;
}
