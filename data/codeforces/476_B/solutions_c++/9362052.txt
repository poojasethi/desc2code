#include <bits/stdc++.h>

using namespace std;

int main() {
  string s1, s2;
  cin >> s1 >> s2;
  int n = (int) s1.size();
  double g = 0;
  for (int i = 0; i < (1 << n); i++) {
    int at1 = 0, at2 = 0;
    for (int j = 0; j < n; j++) {
      if (s1[j] == '+') at1++;
      else at1--;
      if (s2[j] == '+') at2++;
      else if (s2[j] == '-') at2--;
      else if ((1<<j) & i) at2++;
      else at2--;
    }
    g += (at1 == at2);
  }
  printf ("%.30f\n", g / (1 << n));
}
