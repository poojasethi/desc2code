#include<cstdio>
#include<algorithm>
using namespace std;
char s1[11],s2[11];
int main()
{
	int a=0,b=0;double o=1;
	scanf("%s%s",s1,s2);
	for(int i=0;s1[i];++i)
	{
		if(s1[i]=='+')++a;
		else if(s1[i]=='-')--a;
		else ++b;
		if(s2[i]=='-')++a;
		else if(s2[i]=='+')--a;
		else ++b;
	}
	a=(abs(a)+b)/2;
	if(a>b)
	{puts("0.000000000000");return 0;}
	for(int i=1;i<=a;++i)
		o=o*(b-i+1)/i;
	while(b--)o/=2;
	printf("%.12f\n",o);
}
