#include<bits/stdc++.h>
using namespace std;
int fact(int);
long double f=0.0000000000,g=0.0000000000;
int main()
{
int n1=0,n2=0,m1=0,m2=0,i;

string s1,s2;
cin>>s1>>s2;
for(i=0;i<s1.size();i++){
if(s1[i]=='+')
m1++;
else if(s1[i]=='-')
n1++;
}
for(i=0;i<s1.size();i++){
if(s2[i]=='+')
m2++;
else if(s2[i]=='-')
n2++;
}
if(m1-m2<0||n1-n2<0)
cout<<0;
else{
f=fact(m1+n1-m2-n2)/(fact(m1-m2)*fact(n1-n2));
g=f/pow(2,(m1-m2+n1-n2));
cout.precision(50); 
cout<<g;
}
}
int fact(int n){
int i,s=1;
for(i=2;i<=n;i++)
s*=i;
return s;
}
