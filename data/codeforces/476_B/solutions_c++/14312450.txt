#include <bits/stdc++.h>

using namespace std;

int count(int a){
	int b=0;
	for(int i=0;i<32;i++){
		b+=(1<<i)&a?1:0;
	}
	return b;
}

int main(){
	string a,b;
	cin >> a >> b;
	int u=0,m=0;
	for(int i=0;i<a.size();++i){
		u+=(a[i]=='+'?1:-1)+((b[i]=='-')?1:((b[i]=='+')?-1:0));
		m+=b[i]=='?'?1:0;
	}
	double cont=0;
	for(int i=0;i<(1<<m);i++){
		int j=count(i);
		cont+=(j-(m-j))==u?1:0;
	}
	printf("%0.16lf",(cont/float(1<<m)));
	return 0;
}
