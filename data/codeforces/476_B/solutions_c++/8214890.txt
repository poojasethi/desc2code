#include <bits/stdc++.h>
using namespace std;
int fact(int n){
	return n ? n * fact(n - 1) : 1;
}
int main(){
	int plus = 0, minus = 0;
	double eh = 0;
	char c;
	while((c = getchar()) != '\n'){
		if(c == '+') plus ++;
		else minus ++;
	}
	while((c = getchar()) != '\0'){
		if(c == '\n') break;
		if(c == '+') plus --;
		else if(c == '-')minus --;
	}
	if(plus >= 0 && minus >= 0){
		eh = 1.0 * fact(plus + minus) / fact(plus) / fact(minus) / (1 << (plus + minus));
	}
	printf("%.9lf\n", eh);
	return 0;
}
