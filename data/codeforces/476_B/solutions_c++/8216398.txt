#include<iostream>
#include<stdio.h>
using namespace std;
char s1[11],s2[11];
int tot=0,cnt=0,c=0;
void find(char *s,int i,int t)
{
	if(s[i]=='\0')
	{
		if(t==tot)
			cnt++;
		c++;
		return;
	}
	else if(s[i]=='+')
	{
		find(s,i+1,t+1);
	}
	else if(s[i]=='-')
	{
		find(s,i+1,t-1);
	}
	else
	{
		find(s,i+1,t+1);
		find(s,i+1,t-1);
	}
}
int main()
{
	scanf("%s%s",s1,s2);
	int i;
	for(i=0;s1[i]!='\0';i++)
	{
		if(s1[i]=='+')
			tot++;
		else
			tot--;
	}
	find(s2,0,0);
	double x=double(cnt)/(double)c;
	printf("%.12lf",x);
}
