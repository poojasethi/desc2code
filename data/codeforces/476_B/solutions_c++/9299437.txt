#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define MAX 101

char s[MAX];
int sum, cnt;
double ans;

void dfs(int i, int sm)
{
	if (s[i] == 0) {
		cnt++;
		if (sm == sum)
			ans++;
		return;
	}
	if (s[i] == '+' || s[i] == '?')
		dfs(i + 1, sm + 1);
	if (s[i] == '-' || s[i] == '?')
		dfs(i + 1, sm - 1);
}

int main() {
	scanf("%s", s);
	for (int i = 0; s[i]; i++) {
		sum += (s[i] == '+') ? 1 : -1;
	}
	scanf("%s", s);
	dfs(0, 0);
	printf("%.12lf", ans / cnt);
	return 0;
}
