#include <cstdio>

char s1[100001], s2[100001];
int pos = 0, cnt = 0, ans = 0;

void f(int k, int p)
{
    if( s2[k] == 0 )
    {
        cnt++;
        if( p == pos ) ans++;
    }
    if( s2[k] == '+' ) f(k+1, p+1);
    if( s2[k] == '-' ) f(k+1, p-1);
    if( s2[k] == '?' ) f(k+1, p+1), f(k+1, p-1);
}

int main()
{
    scanf("%s%s", s1, s2);
    for(int i = 0 ; s1[i] ; i++ )
        if( s1[i] == '+' ) pos++;
        else pos--;
    f(0, 0);
    printf("%.10lf\n", (double)ans/cnt);
    
}
