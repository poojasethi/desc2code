#include <bits/stdc++.h>
using namespace std;
int main(){
  string a,b;
  int ida=0,idb=0,i,t=0,aux,x,ans=0;
  cin>>a>>b;
  for(i=0;i<a.size();i++){
    if(a[i]=='+')ida++;
    else ida--;
  }
  for(i=0;i<b.size();i++){
    if(b[i]=='+')idb++;
    else if(b[i]=='-')idb--;
    else t++;
  }
  for(i=0;i<(1<<t);i++){
    aux=i;
    x=idb;
    for(int j=0;j<t;j++){
      if(aux&1) x++;
      else x--;
      aux>>=1;
    }
    if(x==ida) ans++;
  }
  printf("%.12f\n",(ans*1.00)/(double)(1<<t));
  return 0;
}
