#include <cstdio>
#include <iostream>
#include <cmath>
#include <cstring>
using namespace std;
char a[100],b[100];
int n,t,sum,ans,now;
double p;
int main()
{
	scanf("%s%s",a,b);
	n=strlen(a);
	int t=0;
	for (int i=0;i<n;++i)
	{
		if (a[i]=='+') t++;else t--;
		if (b[i]=='+') t--;else if (b[i]=='-') t++;else sum++;
	}
	int ans=0;
	for (int i=0;i<1<<sum;++i)
	{
		int now=0;
		for (int j=0;j<sum;++j) if (1<<j&i) now++;else now--;
		if (now==t) ans++;
	}
	p=double(ans)/double(1<<sum);
	printf("%.10f\n",p);
}
