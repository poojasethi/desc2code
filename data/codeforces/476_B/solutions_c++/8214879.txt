#include <bits/stdc++.h>
using namespace std;

int res = 0;
int coin = 0;

void rec(int k, int a, int b){
	if(k == 0){
		coin ++;
		if(a == b) res ++;
		return ;
	}
	rec(k - 1, a, b + 1);
	rec(k - 1, a, b - 1);
}
int main(){
	int a = 0, b = 0, s = 0;
	char c;
	while((c = getchar()) != '\n'){
		if(c == '+') a ++;
		else a --;
	}
	while((c = getchar()) != '\0'){
		if(c == '\n') break;
		if(c == '+') b ++;
		else if(c == '-')b --;
		else s ++;
	}
	rec(s, a, b);
	printf("%.9lf\n",1.0 * res / coin);
	return 0;
}
