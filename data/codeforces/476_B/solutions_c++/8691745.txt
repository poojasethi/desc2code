#include<stdio.h>
#include<iostream>

using namespace std;
int target,len;
string s2;
double reach(int index,int cur)
{
	if(index==len)
	{
		if(cur==target)return 1;
		else return 0;
	}
	if(s2[index]=='+')return reach(index+1,cur+1);
	if(s2[index]=='-')return reach(index+1,cur-1);
	return 0.5*(reach(index+1,cur+1)+reach(index+1,cur-1));
}
int main()
{
	int i;
	string s1;
	cin>>s1;
	cin>>s2;
	len=s1.size();
	target=0;
	for(i=0;i<len;++i)
	{
		if(s1[i]=='+')++target;
		else --target;
	}
	printf("%.9f\n",reach(0,0));
	return 0;
}
