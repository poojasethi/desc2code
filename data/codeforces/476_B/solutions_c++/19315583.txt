#include <bits/stdc++.h>

using namespace std;

typedef __int64 ll;

string s, ss;
double o, c, ct;

void rec (int p=0, int ans=0)
{
	if (p==ss.size())
	{
		o++;
		if (c==ans)
			ct++;
		return;
	}
	if (ss[p]=='+')
		rec(p+1,ans+1);
	if (ss[p]=='-')
		rec(p+1,ans-1);
	if (ss[p]=='?')
		rec(p+1,ans+1), rec(p+1,ans-1);
}

main()
{
	cin >> s;
	cin >> ss;
	for (int i=0;i<s.size();i++)
	{
		if (s[i]=='+')
			c++;
		else
			c--;
	}
	rec();
	double res=ct/o;
	cout.precision(15);
	cout << fixed << res;
}
