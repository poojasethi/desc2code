#include <iostream>
#include <algorithm>
#include <stdio.h>

using namespace std;
string s,t;
double a,b;
int ss = 0;

void Do(int i,int k)
{
	if (i==t.size()) { (k==ss ? a : b)++; return ; }
	
	if (t[i]!='?') Do(i+1,(t[i]=='+' ? k+1 : k-1));
	else
	{
		Do(i+1,k+1);
		Do(i+1,k-1);
	}
}

int main()
{
	cin>>s>>t;

	for (int i=0 ; i<s.size() ; i++) ss+=(s[i]=='+' ? 1 : -1);
	
	Do(0,0);
	
	printf("%.12f",a/(a+b));
	
	return 0;
}
