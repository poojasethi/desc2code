#include<bits/stdc++.h>

using namespace std;

string a,b;
int sz,cnt,spec,sample;

void recurse(int pos, int carry){
	if(pos==sz){
		sample++;
		if(carry==cnt) spec++;
		return;
	}
	if(b[pos]=='+') recurse(pos+1,carry+1); else
	if(b[pos]=='-') recurse(pos+1,carry-1); else{
		recurse(pos+1,carry+1);
		recurse(pos+1,carry-1);
	}
}

int main(){
		cin>>a;
		cin>>b;
		sz=a.size();
		for(int x=0;x<sz;x++) 
		if(a[x]=='+') cnt++; else cnt--;
		recurse(0,0);
	printf("%.12lf\n",spec*1.00/sample);
}
