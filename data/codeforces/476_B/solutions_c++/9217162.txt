#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxl=15;
char s1[maxl],s2[maxl];
int l,cnt,sup=0;
double ans;
void dfs(int x,int p)
{
	if(x==l) { ans+=(p==sup);return ; }
	if(s2[x]=='?')
		dfs(x+1,p-1),dfs(x+1,p+1);
	else dfs(x+1,p+(s2[x]=='+'?1:-1));
}
int main()
{
	int i;
	scanf("%s%s",s1,s2);
	l=strlen(s1);
	for(i=0;i<l;i++)
		sup+=(s1[i]=='+'?1:-1),cnt+=(s2[i]=='?');
	dfs(0,0);
	while(cnt--)
		ans/=2;
	printf("%.10lf",ans);
	return 0;
}
