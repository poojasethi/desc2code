#include<bits/stdc++.h>
using namespace std;
using namespace std;
long long f(long long n)
{
		if(n==0)
					return 1;
			return n*f(n-1);
}
int main()
{
	char s1[12],s2[12];
	int i,j,l1,l2;
	cin>>s1>>s2;
	int l=strlen(s1);
	long long 	int v1=0,v2=0;
	for(i=0;i<l;i++)
	{
		if(s1[i]=='+')
			v1++;
		if(s2[i]=='+')
			v1--;
		if(s2[i]=='?')
			v2++;
	}
	if(v1<0 or v1>v2)
		printf("%0.12f",0.0);
	else{
		double ans=(double)(f(v2)/(f(v2-v1)*f(v1)))/(1<<v2);
		printf("%.12f",ans);
	}
	return 0;
}
