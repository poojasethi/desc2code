#include<bits/stdc++.h>
using namespace std;
int fac[15];
void fact(){
	fac[1]=fac[0]=1;
	for(int i=2;i<=12;i++)
		fac[i]=fac[i-1]*i;
}
int main(){
	char s[15],t[15];
	cin>>s>>t;
	int n=strlen(s),c=0,a=0,b=0,k;
	for(int i=0;i<n;i++)
		if(s[i]=='+') a++;
		else a--;
	for(int i=0;i<n;i++)
		if(t[i]=='+') b++;
		else if(t[i]=='-') b--;
		else c++;
	k=abs(b-a);
	double fc=0,tc;
	c-=k;
	if(c%2 || c<0){
		printf("%lf\n",fc);
		return 0;
	}
	fact();
	tc=pow(2,k+c);
	fc=fac[k+c]/(fac[k+c/2]*fac[c/2]);
	printf("%.12lf\n",fc/tc);
	return 0;
}
