#include <bits/stdc++.h>
using namespace std;

char s[20], t[20];
int a, b, c, n;

void walk(int i, int j) {
  if (i < n) {
    if (t[i] == '+' || t[i] == '?') walk(i+1, j+1);
    if (t[i] == '-' || t[i] == '?') walk(i+1, j-1);
  } else {
    if (j == a) b++;
  }
}

int main () {
  scanf("%s%s", s, t);
  n = strlen(s);
  for (int i = 0; i < n; i++) {
    if (s[i] == '+') a++;
    if (s[i] == '-') a--;
    if (t[i] == '?') c++;
  }
  walk(0, 0);
  printf("%.10f\n", (double) b / (1 << c));
	return 0;
}
