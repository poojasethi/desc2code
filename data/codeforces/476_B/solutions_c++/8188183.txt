#include<bits/stdc++.h>
using namespace std;
string s1,s2;
int n,pos,res,ans,cnt,now;
int main(){
	//freopen("fr.cpp","r",stdin);
	cin>>s1>>s2;
	for(int i=0;i<s1.size();++i)  if(s1[i]=='+')  ++pos; else --pos;
	
	for(int i=0;i<s2.size();++i)  if(s2[i]=='+') ++now; else if(s2[i]=='-') --now; else ++cnt;
	
	now=pos-now;
	for(int mask=0;mask<1<<cnt;++mask){
		ans=0;
		for(int i=0;i<cnt;++i)
			if(mask&1<<i) ++ans;
			else --ans;
		if(ans==now) ++res;
	}
	int k=1<<cnt;
	printf("%.9lf",1.0*res/k);
}
