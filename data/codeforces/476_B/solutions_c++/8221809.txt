#include<bits/stdc++.h>

using namespace std;

string s1,s2;
int x,l;

double recur(int xi,int y)
{
    if(xi == l) return (y == x);
    else {
        double ans = 0.;
        if(s2[xi] == '?') return (recur(xi+1,y-1)+recur(xi+1,y+1))/2.;
        if(s2[xi] != '+') return recur(xi+1,y-1);
        if(s2[xi] != '-') return recur(xi+1,y+1);
    }
}

int main()
{
    cin >> s1 >> s2;
    l = s1.size();
    x = 0;
    for(int i=0;i<l;i++) if(s1[i] == '+') x++; else x--;

    printf("%.9lf",recur(0,0));
}
