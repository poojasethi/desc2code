#include <bits/stdc++.h>

int main()
{
	long long int f[11];
	f[0]=1;
	for(int i=1;i<11;i++)
		f[i]=i*f[i-1];
	double a;
	int p=0,q=0,r=0;
	char c;
	while((c=getchar())!='\n')
	{
		if(c=='+') p++;
		else p--;
	}
	while((c=getchar())!='\n')
	{
		if(c=='+') q++;
		else if(c=='-') q--;
		else r++;
	}
	int t=abs(p-q);
	if((r>=t)&&(((r-t)%2))==0) 
	{
		int al=(r+t)/2;
		int ar=(r-t)/2;
		printf("%.12f\n",(double)f[ar+al]/(f[ar]*f[al]*pow(2,ar+al)));
	}
	else printf("0.000000000000\n");
	return 0;
}
