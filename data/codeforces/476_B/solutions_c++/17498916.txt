#include <bits/stdc++.h>
#define rss(a,b) scanf("%s %s",a,b)
#define FOR(i,l,r) for(int i=l;i<r;i++)
using namespace std;

int N, A=0, B=0, C=0;
char S1[11], S2[11];

int BC(int n, int k) {
	if(k==0||k==n) return 1;
	return BC(n-1,k)+BC(n-1,k-1);
}

int main() {
	rss(S1,S2); N=strlen(S1);
	FOR(i,0,N) A+=S1[i]=='+'?1:-1;
	FOR(i,0,N) S2[i]=='?'?C++:B+=S2[i]=='+'?1:-1;
	if(C>=abs(A-B) && (C-abs(A-B))%2==0)
		printf("%.9lf\n",BC(C,abs(A-B)+(C-abs(A-B))/2)/pow(2.0,C));
	else
		printf("0\n");
}
