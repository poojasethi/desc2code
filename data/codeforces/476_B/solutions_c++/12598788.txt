#include<bits/stdc++.h>
using namespace std;
int dis,len,pos1,pos2,s;
double ans;
string s1,s2;
int main ()
{
	cin>>s1>>s2;
	len=s1.length();
	for(int i=0; i<len; i++)
		pos1+=(s1[i]=='+')?1:-1;
	for(int i=0; i<len; i++)
		if(s2[i]=='?') s++;
		else pos2+=(s2[i]=='+')?1:-1;
	dis=pos1-pos2;
	if((s+dis)%2 || s<abs(dis))
	{
		cout<<0;
		return 0;
	}
	int m=(s+dis)/2,c=1;
	for(int i=0; i<m; i++)
		c*=s-i;
	for(int i=1; i<=m; i++)
		c/=i;
	ans=(double)c/(1<<s);
	printf("%.12lf",ans);
}
