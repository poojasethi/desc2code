#include <cstdio>
#include <cstring>
using namespace std;

char s[11], t[11];
int pos = 0, neg = 0, quest = 0, fact[11];

int main() {
	fact[0] = 1;
	for (int i = 1; i < 11; ++i)
		fact[i] = fact[i - 1] * i;

	gets(s);
	gets(t);

	for (int i  = 0; i < strlen(s); ++i) {
		s[i] == '+' ? pos++ : neg++;
		t[i] == '+' ? pos-- : t[i] == '-' ? neg-- : quest++; 
	}
	if (pos >= 0 && neg >= 0) {
		printf("%.16lf\n", fact[quest] * 1.0 / fact[pos] / fact[neg]/ (1 << quest));
	} else {
		printf("%.16lf\n", 0);
	}

	return 0;
}
