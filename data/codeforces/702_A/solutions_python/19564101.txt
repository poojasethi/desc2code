n = input()
maxlen = 0

inp = map(int, raw_input().split())

count = 1
for i in range(n):
	if i == 0:
		temp = inp[i]

	if inp[i] > temp:
		count += 1

	else:

		if count > maxlen:
			maxlen = count
		count = 1

	temp = inp[i]

if inp[i] > temp:
	count += 1
if count > maxlen:

	maxlen = count

print maxlen
