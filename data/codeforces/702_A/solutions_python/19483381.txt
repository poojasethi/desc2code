

def run():
    n = int(input())
    a = map(int, raw_input().split())

    max_len = 0
    cur_len = 1
    for i in xrange(1, len(a)):
        if a[i] <= a[i-1]:
            if max_len < cur_len:
                max_len = cur_len
            cur_len = 1
        else:
            cur_len += 1
    if max_len < cur_len:
        max_len = cur_len
    print max_len


if __name__ == '__main__':
    run()
