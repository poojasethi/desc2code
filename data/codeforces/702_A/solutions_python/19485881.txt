def maiorSeqCrescent(vetor):
    contTrue = 1
    contTemp = 1
    for i in xrange(1, len(vetor)):
        if vetor[i] > vetor[i-1]:
            contTemp += 1
            if contTemp > contTrue:
                contTrue = contTemp
        else:
            contTemp = 1
    return contTrue

tam = int(raw_input())
numeros = [0] * tam
numsInput = raw_input().split()
for i in xrange(tam):
    numeros[i] = int(numsInput[i])
print maiorSeqCrescent(numeros)
