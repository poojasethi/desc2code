from sys import stdin

n = int(stdin.readline())
a = map(int, stdin.readline().split(' '))
max_sequence = 1
current_sequence = 1
for i in range(1, n):
  if a[i] > a[i - 1]:
    current_sequence += 1
  else:
    max_sequence = max(max_sequence, current_sequence)
    current_sequence = 1
max_sequence = max(max_sequence, current_sequence)
print max_sequence
