k = int(raw_input())
a = map(int,raw_input().split())
i=0
size = []
count=0
if k == 1:
    print 1
else:
    while i < len(a) - 1:
        if a[i] < a[i + 1]:
            count = count + 1
            if i == len(a) - 2:
                size.append(count + 1)
                break
            i = i + 1
        else:
            size.append(count + 1)
            count = 0
            i = i + 1
    print max(size)
