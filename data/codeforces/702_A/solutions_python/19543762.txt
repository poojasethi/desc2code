import sys

n = int(raw_input().strip())
l = raw_input()
a = [int(i) for i in l.strip().split()]
if n == 1:
    print '1'
    sys.exit(0)
curr = 1
res = 1
for i in range(1, n):
    if a[i] > a[i-1]:
        curr += 1
    else:
        res = max(res, curr)
        curr = 1
res = max(res, curr)
print res
