def longest(a,n):
    subSeqLength = 1
    lon = 1
    for i in xrange(1,n):
        if a[i-1] < a[i]  :
            subSeqLength += 1               
        else:
            if subSeqLength > lon:
                lon = subSeqLength
            subSeqLength = 1
    if subSeqLength > lon :
        lon = subSeqLength
    return lon
n = int(raw_input())
a = map(int,raw_input().split())
print longest(a,n)
