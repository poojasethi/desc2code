#!/usr/bin/python

import sys

prev = 0
answer = 0
max_answer = 0
count = int(input())
i = 0

lst = sys.stdin.readline().split()
for a in lst:
    i += 1
    if i > count:
        break
    next = int(a)
    if (next > prev) :
        answer += 1
    else:
        if answer > max_answer:
            max_answer = answer
        answer = 1
    prev = next

if answer > max_answer:
    max_answer = answer

print max_answer
