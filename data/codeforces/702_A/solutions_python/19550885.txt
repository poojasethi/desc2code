def find_max_increasing (list, n):
    increasing_list = [0] * n
    increasing_list[0] = 1
    for i in xrange(1,n):
        if(list[i]>list[i-1]):
            increasing_list[i] = increasing_list[i-1] + 1
        else :
            increasing_list[i] = 1
    return max(increasing_list)


def main():
    n = int(raw_input())
    list = map(int, raw_input().split())
    max_inc_length = find_max_increasing(list, n)
    print max_inc_length

if __name__ == '__main__':
    main()
