#!/usr/bin/python

import sys

def Ni(): return map(int, sys.stdin.readline().split())

# find 3-clique with minimum sum of degrees
n, m = Ni()
deg = [0] * n
adj = [set() for i in range(n)]
edges = []

for i in range(m):
    a, b = Ni()
    a, b = a-1, b-1
    deg[a] += 1
    deg[b] += 1
    adj[a].add(b)
    adj[b].add(a)
    edges.append((a, b))

best = 1e8
for a, b in edges:
    for neighbor in adj[a]:
        if neighbor in adj[b]:
            # print "clique", a, b, neighbor
            best = min(best, deg[a] + deg[b] + deg[neighbor] - 6)

if best == 1e8:
    print -1
else:
    print best
