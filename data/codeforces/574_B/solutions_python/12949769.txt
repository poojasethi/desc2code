MAX = 4000

n, m = [int(x) for x in raw_input().split()]

adj = {}
for i in range(m):
	a, b = [int(x) for x in raw_input().split()]
	l = adj.get(a, [])
	l.append(b)
	adj[a] = l
	l = adj.get(b, [])
	l.append(a)
	adj[b] = l

#print adj

candidates = [k for k,v in adj.items() if len(v)>=2]

#print candidates

recognition = MAX
found = False
for c in candidates:
	neighbors_l1 = adj[c]
	for n1 in neighbors_l1:
		neighbors_l2 = adj[n1]
		for n2 in neighbors_l2:
			if c in adj[n2]:
				new_recognition = len(adj[c]) + len(adj[n1]) + len(adj[n2]) - 6
				if new_recognition < recognition:
					recognition = new_recognition
					found = True

if found:
	print recognition
else:
	print '-1'
