def main():
    len_w, num_pairs = map(int, raw_input().split())
    ws = [set() for i in xrange(len_w+1)]
    min_recognition = -1

    for i in xrange(num_pairs):
        w1, w2 = map(int, raw_input().split())
        ws[w1].add(w2)
        ws[w2].add(w1)
    bands = set()
    for w, friends in enumerate(ws):
        #print '*' * 20,
        #print w, friends
        if len(friends) < 2:
            continue
        for friend in friends:
            friend_friends = ws[friend]
            #print friend, friend_friends
            for ff in friend_friends:
                if ff == w:
                    continue
                if ff in friends and w in ws[ff]:
                    band = w, friend, ff
                    if band in bands:
                        continue
                    bands.add(tuple(sorted(list(band))))
                    recognition = sum(len(ws[w]) for w in band) - 6
                    if min_recognition == -1 or recognition < min_recognition:
                        min_recognition = recognition
    return min_recognition

print main()
