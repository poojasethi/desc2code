n, m = [int(a) for a in raw_input().split()]
arr = [[] for a in range(n)]
for a in range(m):
	x, y = [int(b) for b in raw_input().split()]
	arr[x-1].append(y-1)
	arr[y-1].append(x-1)
mi = -1
#print arr
for a in range(n):
	x = arr[a]
	if (len(x)<2):
		continue
	for y in arr[a]:
		for z in arr[a]:
			if z in arr[y]:
				an = (len(arr[a])+len(arr[y])+len(arr[z]))-6
			#	print an, a, y, z
				if an<mi or mi==-1:
					mi = an
