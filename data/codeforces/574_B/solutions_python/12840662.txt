n,m = map(int,raw_input().split())
recog = [-2]*n
l = [set() for i in xrange(n)]
know = [sorted(map(lambda x: int(x)-1,raw_input().split())) for i in xrange(m)]
know.sort(key = lambda x:x[0])
ans = 4000
for a,b in know:
	recog[a] += 1
	recog[b] += 1
	l[a].add(b)
for a,b in know:
	for i in xrange(a):
		if a in l[i] and b in l[i]:
			tmp = recog[i] + recog[a] + recog[b]
			if ans > tmp:
				ans = tmp
print -1 if ans == 4000 else ans
