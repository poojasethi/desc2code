n,m = map(int,raw_input().split())

edge = []
E = [set() for i in xrange(n+1)]

for i in xrange(m):
    u,v = map(int,raw_input().split())
    edge.append((u,v))
    E[u].add(v)
    E[v].add(u)

ret = 1e19
for u,v in edge:
    tmp = E[u] & E[v]
    for h in tmp:
        if h in E[u] and h in E[v]:
            ret = min(ret,len(E[h])+len(E[v])+len(E[u])-6)
            
if ret == 1e19:ret=-1
print ret
