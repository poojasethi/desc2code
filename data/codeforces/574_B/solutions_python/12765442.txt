from sys import stdin
n, m = map(int, stdin.readline().split())
rec = [[] for j in xrange(n)]
ans = 3*n
for i in xrange(m):
   a, b = map(int, stdin.readline().split())
   rec[a-1].append(b-1)
   rec[b-1].append(a-1)
tot = [len(rec[i]) for i in xrange(n)]
for i in xrange(n):
    for j in rec[i]:
        for k in set(rec[i]).intersection(set(rec[j])):
            ans = min(ans, tot[i]+tot[j]+tot[k]-6)
if ans == 3*n:
    ans = -1
print ans
