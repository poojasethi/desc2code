#include<bits/stdc++.h>
int ans = 30000;
using namespace std;
int main()
 {
   ios_base::sync_with_stdio(false);
   int n,m,x[4010],y[4010],i,k,j,s[4010]; 
   bool a[4010][4010];
   cin>>n>>m;
   for(i=1;i<=m;i++)
   {
    cin>>x[i]>>y[i];
    a[x[i]][y[i]] = 1;
    a[y[i]][x[i]] = 1;
    s[x[i]]++;
    s[y[i]]++;
   } 
   for(i=1;i<=m;i++) for(j=1;j<=n;j++) if(a[x[i]][j]&&a[y[i]][j]) ans = min(ans,s[x[i]] + s[y[i]] + s[j] - 6);
  if(ans<30000)
   cout<<ans;
  else
   cout<<-1;
  return 0;
}
