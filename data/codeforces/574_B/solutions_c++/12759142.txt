#include<bits/stdc++.h>
using namespace std;
long long n,i,otvet,matrix[4001][4001],a[5000],m,j,x[5000],y[5000];
int main()
{
 ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
cin>>n>>m;


 otvet=-1;
 for (i=0;i<m;++i)
 {
     cin>>x[i]>>y[i];
     a[x[i]]++;
     a[y[i]]++;
     matrix[x[i]][y[i]]=1;
       matrix[y[i]][x[i]]=1;
 }
 otvet=12345678;
 for (i=0;i<m;++i)
 {
   for (j=0;j<n;++j)
    if (matrix[x[i]][j]==1&&matrix[j][y[i]]==1) otvet=min(otvet,a[x[i]]+a[y[i]]+a[j]);

 }
if (otvet==12345678) cout<<-1; else cout<<otvet-6;
}
