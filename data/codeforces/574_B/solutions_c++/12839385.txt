#include <bits/stdc++.h>
using namespace std ;
const int maxn=4e3+5;

int a[maxn][maxn],d[maxn];
int x[maxn],y[maxn];
int n,m;

int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++){
        scanf("%d%d",&x[i],&y[i]);
        a[x[i]][y[i]]=a[y[i]][x[i]]=1;
        d[x[i]]++,d[y[i]]++;
    }
    int ans=1e9;
    for(int i=1;i<=m;i++){
        for(int j=1;j<=n;j++)
            if(a[j][x[i]]&&a[y[i]][j])
                ans=min(ans,d[x[i]]+d[y[i]]+d[j]);
    }
    if(ans==1e9) printf("-1\n");
    else printf("%d\n",ans-6);
    return 0;
}
