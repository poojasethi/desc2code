#include<iostream>
#include<algorithm>
using namespace std;
int adj[4010][4010], wr[4010], a, b, ans=100000000, n, m;

int main()
{
    cin>>n>>m;
    for(int i=0;i<m;i++)
    {
        cin>>a>>b;
        a--, b--;
        adj[a][b]=adj[b][a]=1;
        wr[a]++;
        wr[b]++;
    }

    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++) if(adj[i][j])
            for(int k=j+1;k<n;k++) if(adj[i][k] && adj[j][k])
            ans=min(ans, wr[i]+wr[j]+wr[k]-6);

    if(ans==(100000000)) ans=-1;
    cout<<ans<<endl;
    return 0;
}
