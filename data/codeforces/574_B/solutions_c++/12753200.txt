#include<iostream>
#include<cstdio>
using namespace std;
const int N=4004;
bool mp[N][N];int n,m,sum[N];
struct st{int x,y;}a[N];
int main(){
	ios::sync_with_stdio(1);
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		cin>>a[i].x>>a[i].y;
		mp[a[i].x][a[i].y]=mp[a[i].y][a[i].x]=1;
		sum[a[i].x]++;sum[a[i].y]++;
	}
	int ans=0x7fffffff;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=n;j++)
			if(mp[a[i].x][j]&&mp[a[i].y][j]){
				if(sum[a[i].x]+sum[a[i].y]+sum[j]<ans)ans=sum[a[i].x]+sum[a[i].y]+sum[j];
			}
	if(ans==0x7fffffff)cout<<-1<<endl;
	else cout<<ans-6<<endl;
	return 0;
}
