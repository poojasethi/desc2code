#include <bits/stdc++.h>
using namespace std;
bool adj[4001][4001];int cc[4001];
int main(){
	int i,n,m,a,b,c;cin>>n>>m;
	for(i=0;i<m;++i)
		cin>>a>>b,adj[a][b]=1,adj[b][a]=1,++cc[a],++cc[b];
	int ret=1e9;
	for(a=1;a<=n;++a)
		for(b=a+1;b<=n;++b)
			if(adj[a][b])
				for(c=b+1;c<=n;++c)
					if(adj[a][c] &&adj[b][c])
						ret = min(cc[a]+cc[b]+cc[c],ret);
	printf("%d\n",ret==1e9? -1 : ret-6);
}
