#include<bits/stdc++.h>
using namespace std;
int g[4001][4001];
int deg[4001];
int main()
{
	int i,j,k,n,m,x,y;
	cin>>n>>m;
	for(i=0;i<m;i++)
	{
		cin>>x>>y;
		deg[x]++;
		deg[y]++;
		g[x][y]++;
		g[y][x]++;
	}
	int mx=81994;
	for(i=1;i<=n;i++)
	{
		for(j=i+1;j<=n;j++)
		{
			if(g[i][j])
			for(k=j+1;k<=n;k++)
			{
				if(g[i][k]&&g[k][j] )
				mx=min(mx,deg[i]+deg[j]+deg[k]);
			}
		}
	}
	if(mx>10000)
	cout<<"-1"<<endl;
	else
	cout<<mx-6<<endl;
	return 0;
}
