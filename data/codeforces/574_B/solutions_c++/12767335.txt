#include <bits/stdc++.h>
using namespace std;
int c[4001];
int a[4001][4001];
int main()
{
	int n,m;
	cin >> n >>m;
	while(m--)
	{
		int p,q;
		cin >> p >> q;
		p--;
		q--;
		a[p][q]=1;
		a[q][p]=1;
		c[p]++;
		c[q]++;
	}
	int min = 1000000000;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
			if(a[i][j])
				for(int k=j+1;k<n;k++)
					if(a[j][k] and a[i][k])
						if(min > c[i]+c[j]+c[k]) min = c[i] + c[j] + c[k];
	if(min == 1000000000) cout << -1 << endl;
	else cout << min -6  <<  endl;
	return 0;
}
