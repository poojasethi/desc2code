#include<bits/stdc++.h>
using namespace std;
int n,m;
int a[4010],b[4010];
int deg[4010];
bool adj[4010][4010];
int main(){
	ios::sync_with_stdio(0);cin.tie(0);
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>a[i]>>b[i];
		adj[a[i]][b[i]]=1;
		adj[b[i]][a[i]]=1;
		deg[a[i]]++;
		deg[b[i]]++;
	}
	int ANS=1e9;
	bool k=0;
	for(int i=0,A,B,jam;i<m;i++){
		A=a[i];
		B=b[i];
		for(int j=1;j<=n;j++){
			if(j!=B&&j!=A){
				if(adj[A][j]==1&&adj[B][j]==1){
					k=1;
					jam=deg[A]+deg[B]+deg[j]-6;
					ANS=min(ANS,jam);
				}
			}
		}
	}
	if(k==1)cout<<ANS;
	else cout<<-1;
}
