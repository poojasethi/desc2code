#include<cstdio>
#include<algorithm>
using namespace std;
int a1[4000][2],a2[4000];
bool b1[4000][4000];
int main()
{
    int n,m,a,b,o=100000;
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;++i)
    {
        scanf("%d%d",&a,&b);
        a1[i][0]=--a;a1[i][1]=--b;
        ++a2[a];++a2[b];
        b1[a][b]=b1[b][a]=1;
    }
    for(int i=0;i<n;++i)for(int j=0;j<m;++j)
    if(b1[i][a1[j][0]]&&b1[i][a1[j][1]])
        o=min(o,a2[i]+a2[a1[j][0]]+a2[a1[j][1]]);
    printf("%d\n",o==100000?-1:o-6);
}
