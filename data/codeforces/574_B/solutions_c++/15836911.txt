#include <bits/stdc++.h>
using namespace std;
int g[4001][4001];
int main() 
{
	int n,m,a,b,i,j,k,min=1e6;
	cin>>n>>m;
	for(i=0;i<m;++i)
	{
		cin>>a>>b;
		g[a][b]=g[b][a]=1;
		g[a][0]++;g[b][0]++;
	}
	
	for(i=1;i<=n;++i)
	for(j=i+1;j<=n;++j)
		if(g[i][j])
		{
			for(k=1;k<=n;++k)
				if(g[i][k]&&g[j][k])
					if(min>g[i][0]+g[j][0]+g[k][0])
					min=g[i][0]+g[j][0]+g[k][0];
		}
		
	if(min==1e6)
	min=5;
	cout<<min-6;
	// your code goes here
	return 0;
}
