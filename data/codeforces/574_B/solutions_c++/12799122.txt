#include <iostream>
#include <cstdio>
using namespace std;

int n, m, mn = 2e9, b[4005];
bool a[4005][4005];
int main()
{
	int i, j, k, t1, t2;
//	freopen ("input.txt", "r", stdin);
	cin >> n >> m;
	for (i = 0; i < m; i++) {
		scanf("%d %d", &t1, &t2);
		a[t1][t2] = true;
		a[t2][t1] = true;
		b[t1]++;
		b[t2]++;
	}
	for (i = 1; i <= n; i++) {
		for (j = i + 1; j <= n; j++) {
			if (a[i][j]) {
				for (k = j + 1; k <= n; k++) {
					if (a[i][k] && a[j][k]) mn = min(mn, b[i] + b[j] + b[k]);
				}
			}
		}
	}
	if (mn == 2e9) mn = 5;
    cout << mn - 6 << endl;
    return 0;
}
