#include<bits/stdc++.h>
using namespace std;
int a[4005][4005];
int s[4005];
int main()
{
	int n,m;
	int ans=12000000;
	scanf("%d %d",&n,&m);
	for(int i=0;i<m;i++)
	{
		int x,y;
		scanf("%d %d",&x,&y);
		a[x][y]=a[y][x]=1;
		s[x]++;
		s[y]++;
	}
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			if(a[i][j]==1)
			{
				for(int k=j+1;k<=n;k++)
				{
					if(a[i][k]==1 && a[j][k]==1)
						ans=min(ans,s[i]+s[j]+s[k]-6);
				}
			}
		}
	}
	if(ans==12000000) printf("-1\n");
	else printf("%d\n",ans);
	return 0;
}
