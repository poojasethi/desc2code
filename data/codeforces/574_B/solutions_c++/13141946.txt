#include <cstdio>
int n,m,i,j,x[5000],y[5000],u[5000],ans;
bool a[4100][4100];

int main()
{
	scanf("%d%d", &n, &m);
	for (i=1; i<=m; ++i)
	{
		scanf("%d%d", &x[i], &y[i]);
		a[x[i]][y[i]] = a[y[i]][x[i]] = 1;
		++u[x[i]];
		++u[y[i]];
	}
	ans = 1000000007;
	for (i=1; i<=n; ++i)
	for (j=1; j<=m; ++j)
	if (a[i][x[j]] && a[i][y[j]])
	if (u[i]+u[x[j]]+u[y[j]] < ans) ans = u[i]+u[x[j]]+u[y[j]];
	if (ans > 1000000000) ans = 5;
	printf("%d\n", ans-6);
	return 0;
}
