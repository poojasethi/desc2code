#include<bits/stdc++.h>
using namespace std;

int v[4004][4004];
map<int,int>sm;

int main(){
	
	int n,m,c=4000000,a,b;
	cin>>n>>m;
	for(int i=1 ; i<=m ; i++)
	{
		cin>>a>>b;
		v[a][b]=1;
		v[b][a]=1;
		sm[a]++;
		sm[b]++;
	}
	for(int i=1 ; i<=n ; i++)
	{
		for(int j=1 ; j<=n ; j++)
		{
			if(v[i][j])
				for(int k=j+1 ; k<=n ; k++)
				{
					if(v[i][k] && v[j][k])
					{
						c=min(c,sm[i]+sm[j]+sm[k]-6);
					}
				}
		}
	}
	if(c==4000000)	c=-1;
	cout<<c<<endl;
	
	
	
	
}
