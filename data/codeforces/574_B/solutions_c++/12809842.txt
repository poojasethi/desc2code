#include <bits/stdc++.h>
using namespace std;

int a[4005][4005];
vector <int> c(4005,0);

int main(){
	int n,m; cin >> n >> m;
	while(m--){
		int x,y;
		cin >> x >> y;
		a[x-1][y-1]=1;
		a[y-1][x-1]=1;
		c[x-1]++;
		c[y-1]++;
	}
	int minn=1e9+1;
	for(int i=0; i<n; i++){
		for(int j=i+1; j<n; j++){
			if(a[i][j]==1){
				for(int k=j+1; k<n; k++){
					if(a[i][k]==1 && a[j][k]==1){
						minn=min(minn,c[i]+c[j]+c[k]);
					}
				}
			}
		}
	}
	if(minn==(int)1e9+1) cout << -1 << endl;
	else cout << minn-6 << endl;
}
