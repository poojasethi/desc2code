#include<iostream>
#define fr(i,a,b) for(int i=a;i<b;i++)
using namespace std;
const int nmax=4005;
bool flag[nmax][nmax];
int recog[nmax]{};
int main(){
    int na,nb;
    cin>>na>>nb;
    for(int i=0;i<nb;i++){
	int a,b;
	cin>>a>>b;
	flag[a][b]=flag[b][a]=true;
	recog[a]++;
	recog[b]++;
    }
    int result=123456;
    fr(i,1,na+1){
	fr(j,i+1,na+1){
	    if(flag[i][j]){
		fr(k,j+1,na+1){
		    if(flag[i][k]&&flag[j][k]){
			result=min(result,recog[i]+recog[j]+recog[k]);
		    }
		}
	    }
	}
    }
    if(result==123456) cout<<"-1";
    else cout<<result-6;
    return 0;
}
