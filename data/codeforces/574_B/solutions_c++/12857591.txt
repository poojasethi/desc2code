#include <bits/stdc++.h>
using namespace std;

int n,m,a[4001],b[4001],c[4001],d=20000;
bool know[4001][4001];

int main()
{
    cin>>n>>m;
    for(int i=1; i<=m; i++)
    {
        cin>>a[i]>>b[i];
        c[a[i]]++;
        c[b[i]]++;
        know[a[i]][b[i]]=1;
        know[b[i]][a[i]]=1;
    }
    for(int i=1; i<=m; i++)
        for(int j=1; j<=n; j++)
            if(know[a[i]][j]&&know[b[i]][j])
                d=min(c[a[i]]+c[b[i]]+c[j]-6,d);
    if(d<20000)
        cout<<d<<endl;
    else cout<<-1<<endl;
}
