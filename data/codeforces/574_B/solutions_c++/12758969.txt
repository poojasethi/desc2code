#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll n,i,ans,b[4001][4001],a[5000],m,j,x[5000],y[5000];
int main()
{
 ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
cin>>n>>m;


 ans=-1;
 for (i=0;i<m;++i)
 {
     cin>>x[i]>>y[i];
     a[x[i]]++;
     a[y[i]]++;
     b[x[i]][y[i]]=1;
       b[y[i]][x[i]]=1;
 }
 ans=12345678;
 for (i=0;i<m;++i)
 {
   for (j=0;j<n;++j)
    if (b[x[i]][j]==1&&b[j][y[i]]==1) ans=min(ans,a[x[i]]+a[y[i]]+a[j]);

 }
if (ans==12345678) cout<<-1; else cout<<ans-6;
}
