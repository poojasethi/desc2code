#include<bits/stdc++.h>

using namespace std;

int n,m,a[4001][4001],b[4001],x,y,ans=1000000;

int main()
{
    cin>>n>>m;
    for(int i=0;i<m;++i)
    {
        cin>>x>>y;
        a[x][y]=a[y][x]=true;
        ++b[x];
        ++b[y];
    }
    for(int i=1;i<=n;++i)
        for(int j=i+1;j<=n;++j)
            if(a[i][j])
                for(int k=j+1;k<=n;++k)
                    if(a[i][k]&&a[j][k])ans=min(ans,b[i]+b[j]+b[k]-6);
    cout<<(ans==1000000?-1:ans);
}
