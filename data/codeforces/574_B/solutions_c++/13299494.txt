#include <bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int n,m;cin>>n>>m;
	static bool e[4003][4003];memset(e,false,sizeof(e));
	int x,y,d[4003],ans=4000*3;memset(d,0,sizeof(d));
	while(m--)
	{
		cin>>x>>y;
		e[x][y]=true;e[y][x]=true;
		d[x]++;d[y]++;
	}
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
		{
			if(i!=j && e[i][j])
				{
					for(int k=1;k<=n;k++)
						if(e[i][k] && e[k][j])
							ans=min(ans,d[i]+d[j]+d[k]-6);
				}
			
		}
	if(ans==4000*3)ans=-1;
	cout<<ans<<endl;
	return 0;
}
