#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n,m,x,y,ans=123456789;
	cin>>n>>m;
	vector<int>a[n];
	for(int i=0;i<m;i++)
	{
		cin>>x>>y;
		x--,y--;
		a[x].push_back(y);
		a[y].push_back(x);
	}
	for(int i=0;i<n;i++)
		for(int j=0;j<a[i].size();j++)
			for(int k=0;k<a[a[i][j]].size();k++)
				for(int l=0;l<a[a[a[i][j]][k]].size();l++)
					if(a[a[a[i][j]][k]][l]==i)
					{
						int sum=a[i].size()+a[a[i][j]].size()+a[a[a[i][j]][k]].size()-6;
						ans=min(ans,sum);
					}
	if(ans!=123456789)cout<<ans<<endl;
	else cout<<-1<<endl;
}
