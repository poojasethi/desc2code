#include <bits/stdc++.h>
using namespace std;
int g[4004][4004];
int n,m;
int deg[4004];
int main()
{
	cin>>n>>m;
	int x,y;
	for(int i=0;i<m;i++)
	{
		cin>>x>>y;
		x--;
		y--;
		g[x][y] = g[y][x]  = 1;
		deg[x]++;
		deg[y]++;
	}
	int ans = 1e9;
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			if(g[i][j])
			{
				for(int k=j+1;k<n;k++)
				{
					if(g[i][k] && g[j][k])
						ans = min(ans , deg[i] + deg[j] + deg[k] - 2*(g[i][j] + g[i][k] + g[j][k]));			
				}
			}
		}
	}
	if(ans == 1e9)ans=-1;
	cout<<ans;
	return 0;
}
