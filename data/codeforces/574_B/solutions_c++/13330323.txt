#include<iostream>
using namespace std;

const int oo = 4010;
int n, m, x, y, a[oo], ans = oo * oo, p[oo], q[oo];
bool edg[oo][oo];

int main(){
  cin >> n >> m;
  for(int i = 1; i <= m; i++)
    cin >> x >> y, a[x]++, a[y]++, p[i] = x, q[i] = y, edg[x][y] = edg[y][x] = true;
  for(int i = 1; i <= n; i++)
    for(int j = 1; j <= m; j++)
      if((p[j] != i) && (q[j] != i) && (edg[i][p[j]]) && (edg[i][q[j]]))
        ans = min(ans, a[i] + a[p[j]] + a[q[j]] - 6);
  if(ans == oo * oo)
    cout << -1;
  else
    cout << ans;
  return 0;
}
