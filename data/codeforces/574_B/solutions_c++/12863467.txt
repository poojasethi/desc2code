#include <bits/stdc++.h>
using namespace std;
const int inf=100000000;
const int nmax=5000;
bool t[nmax][nmax];
int degree[nmax];
int main()
{
	int n,m,res=inf,a,b;
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++)
	{
		scanf("%d%d",&a,&b);
		t[a][b]=t[b][a]=true;
		degree[a]++;
		degree[b]++;
	}
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
		{
			if(t[i][j])
			{
				for(int k=j+1;k<=n;k++)
					if(t[i][k] && t[k][j])
						res=min(res,degree[i]+degree[k]+degree[j]-6);
			}
		}
	if(res==inf) printf("-1\n");
	else printf("%d\n",res);
}
