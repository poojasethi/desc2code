#include <bits/stdc++.h>
using namespace std;

int n,m,a,b,f[4005],ct1,ct2,ans,l,r,mid;
bool ok[4005][4005];
pair<int,int> arr[4005];
 
int main(){
	cin >> n >> m;
	for (int i=1;i<=m;i++){
		cin >> a >> b;
		arr[i]=make_pair(a,b);
		ok[a][b]=ok[b][a]=true;
		f[a]++; f[b]++;
	}
	ans=1000000000;
	for (int i=1;i<=m;i++){
		for (int j=1;j<=n;j++){
			if (ok[arr[i].first][j]&&ok[arr[i].second][j]){
				ans=min(ans,f[arr[i].first]+f[arr[i].second]
				+f[j]-6);
			}
		}
	}
	cout << (ans==1000000000 ? -1 : ans) << "\n";
}
