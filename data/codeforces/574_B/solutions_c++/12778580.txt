#include<bits/stdc++.h>

using namespace std;

const int INF = 100000005;
const int MAXN = 5005;
int d[MAXN];
bool t[MAXN][MAXN];

int main() {
	int n, m;
	cin >> n >> m;
	for(int i = 0; i < m; i++) {
		int v, u;
		cin >> v >> u;
		t[v][u] = t[u][v] = true;
		d[v]++;
		d[u]++;
	}
	int s = INF;
	for(int i = 1; i <= n; i++)
		for(int j = i + 1; j <= n; j++)
			if(t[i][j])
				for(int k = j + 1; k <= n; k++)
					if(t[i][k] && t[j][k])
						s = min(s, d[i]+d[j]+d[k]);
	if(s == INF)
        cout << "-1";
	else
        cout << s-6 << endl;
	return 0;
}
