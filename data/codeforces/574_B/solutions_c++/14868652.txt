#include <iostream>
using namespace std;
#define inf  1000000000

int lista[4002];
bool ady[4002][4002];

int main()
{
	int n,m;
	cin>>n>>m;
	
	for (int i=0; i<m; i++)
	{
		int x, y;
		cin>>x>>y;
		ady[x][y] = true;
		ady[y][x] = true;
		lista[x]++;
		lista[y]++;
	}
	
	int sum=inf;
	for (int i=1; i<=n; i++)
		for (int j=1; j<=n; j++)
			if (ady[i][j])
				for (int T=1; T<=n; T++)
					if (ady[j][T] && ady[T][i])
						sum = min(sum, lista[i] + lista[j] + lista[T]);
	
	if (sum == inf)
		cout<<-1;
	else cout<<sum-6;
	
	return 0;
}
