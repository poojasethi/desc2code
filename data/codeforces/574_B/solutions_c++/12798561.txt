#include<iostream>
using namespace std;
bool a[4001][4001]={0};
int d[4001]={0};
int ans = 99999999;
int main()
{
	int n,m;
	cin>>n>>m;
	for (int i=0;i<m;i++)
	{
		int p1,p2;
		cin>>p1>>p2;
		a[p1][p2] = true;
		a[p2][p1] = true;
		d[p1]++;
		d[p2]++;
	}
	for (int i=1 ; i <= n;i++)
	{
		for (int j=i+1;j<=n;j++)
			if (a[i][j])
			{
				for (int k=j+1;k<=n;k++)
				{
					if (a[k][j] && a[k][i])
					ans = min (ans ,d[i]+d[j]+d[k]-6);
				}
			}
	}
	if (ans == 99999999)
		cout<<"-1\n";
	else 
		cout<<ans<<endl;
	return 0;

}
