#include<cstdio>
#include<algorithm>
using namespace std;

int n, m, a, b, ans;
int e[4444][4444], s[4444];

int main() {
	scanf("%d%d", &n, &m);
	for(int i=0;i<m;i++) {
		scanf("%d%d", &a, &b);
		e[a][b] = e[b][a] = 1;
		s[a]++; s[b]++;
	}
	
	ans = 22222;
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			if(e[i][j])
				for(int k=j+1;k<=n;k++)
					if(e[i][k] && e[j][k])
						ans = min(ans, s[i]+s[j]+s[k]);
	if(ans <= 12000) printf("%d\n", ans - 6);
	else printf("-1\n");
	
	return 0;
}
