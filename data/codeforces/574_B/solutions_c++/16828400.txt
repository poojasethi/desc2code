#include<iostream>
using namespace std;
long x[10003];long y[10004];int a[4004][4004];long v[10004];
int main(){
	long n,m;
	cin>>n>>m;
	for(long i=0;i<m;i++){
		cin>>x[i]>>y[i];v[x[i]]=v[x[i]]+1;v[y[i]]=v[y[i]]+1;
		a[x[i]][y[i]]=1;a[y[i]][x[i]]=1;
	}
	long long min=100000000;int u=0;
	for(long i=0;i<m;i++){
		long q=x[i];long p=y[i];
		for(long j=1;j<=n;j++){
			if(j!=p&&j!=q){
				if(a[j][q]==1&&a[j][p]==1){u=1;
					long h=v[p]+v[q]+v[j]-6;
					if(h<min)min=h;
				}
			}
		}
	}
if(u==1)cout<<min<<endl;else cout<<"-1"<<endl;
}
