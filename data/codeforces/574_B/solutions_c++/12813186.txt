#include<bits/stdc++.h>
using namespace std;
int degree[4001];
bool vis[4001][4001];
int main()
{
int n,m,a,b,i,j,k;
scanf("%d%d",&n,&m);
for(i=0;i<m;i++)
{
scanf("%d%d",&a,&b);
vis[a][b]=vis[b][a]=1;
degree[a]++;
degree[b]++;
}
int min=INT_MAX;
for(i=1;i<=n;i++)
{
for(j=i+1;j<=n;j++)
{
if(vis[i][j])
{
for(k=j+1;k<=n;k++)
{
if(vis[i][k] && vis[j][k])
{
if(degree[i]+degree[j]+degree[k]-6<min)
min=degree[i]+degree[j]+degree[k]-6;
}
}
}
}
}
if(min==INT_MAX)
printf("-1\n");
else
printf("%d\n",min);
return 0;
}
