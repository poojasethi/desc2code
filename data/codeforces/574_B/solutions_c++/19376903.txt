#include<bits/stdc++.h>
using namespace std;
const int inf=1000000005;
int degree[4005];
bool vis[4005][4005];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		int a,b;
		cin>>a>>b;
		vis[a][b]=true;
		vis[b][a]=true;
		degree[a]++;
		degree[b]++;
	}
	int res=inf;
	for(int i=1;i<=n;i++)
	{
		for(int j=i+1;j<=n;j++)
		{
			if(vis[i][j])
			{
				for(int k=j+1;k<=n;k++)
				{
					if(vis[i][k]&&vis[j][k])
					res=min(res,degree[i]+degree[j]+degree[k]);
				}
			}
		}
	}
	if(res==inf)
	cout<<"-1";
	else
	cout<<res-6<<endl;
	return 0;
	}
