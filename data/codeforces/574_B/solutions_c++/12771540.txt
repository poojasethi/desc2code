#include <iostream>
using namespace std;

int a[4005][4005];

int main()
{

int n,m,s[4005]={0},x[4005],y[4005],mn=1000000,i,j;

cin>>n>>m;

 for(i=1;i<=m;i++)
 {
  cin>>x[i]>>y[i];

  a[x[i]][y[i]]=1;
  a[y[i]][x[i]]=1;

  s[x[i]]++;
  s[y[i]]++;
 }

 for(i=1;i<=m;i++)
  for(j=1;j<=n;j++)
  {
   if(a[x[i]][j]==1 && a[y[i]][j]==1)
    if(s[x[i]]+s[y[i]]+s[j]-6<mn)
     mn=s[x[i]]+s[y[i]]+s[j]-6;
  }

 if(mn<1000000)
  cout<<mn<<endl;
 else
  cout<<-1<<endl;

return 0;
}
