#include <bits/stdc++.h>
#define pb push_back
using namespace std;
vector <int> v[4001];
int n,m,d[4001],a,b,mi=1000000,dp[4001][4001],dat;

int main()
{cin>>n>>m;
for (int i=1;i<=m;i++) {cin>>a>>b; v[a].pb(b); v[b].pb(a); d[a]++; d[b]++;dp[a][b]++;dp[b][a]++;}
for (int first=1;first<=n;first++) for (int i=0;i<v[first].size();i++) {int two=v[first][i];
for (int j=0;j<v[two].size();j++) {int three=v[two][j]; if (dp[first][three]>0) {dat=1;mi=min(mi,d[first]+d[two]+d[three]);}}}
if (dat==0) cout<<"-1"<<endl; else cout<<mi-6<<endl;
    return 0;
}
