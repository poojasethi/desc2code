import copy
tab = [list(map(int,raw_input().split())) for i in range(3)]
ans =copy.deepcopy(tab)
M = [(-1,0),(0,-1),(1,0),(0,1)]
for i in range(3):
    for j in range(3):
        for dx,dy in M:
            if 0<=i+dx<3 and 0<=j+dy<3:
                ans[i+dx][j+dy]+=tab[i][j]
for i in range(3):
    for j in range(3):
        if ans[i][j]%2==0:
            ans[i][j]=1
        else: ans[i][j]=0
for i in range(3):
    print(''.join(map(str,ans[i])))
