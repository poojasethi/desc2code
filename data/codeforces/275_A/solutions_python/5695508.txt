
inp = []
res = []
for i in xrange(0, 3):
	inp.append(raw_input().split())
for i in xrange(0, 3):
	tmp = ""
	for j in xrange(0, 3):
		switches = 0
		for k in xrange(max(0, i - 1), min(3, i + 2)):
			switches += int(inp[k][j])
		for l in xrange(max(0, j - 1), min(3, j + 2)):
			switches += int(inp[i][l])
		switches -= int(inp[i][j])
		#print switches			
		tmp += str((switches + 1) % 2)
	res.append(tmp)
for s in res:
	print s
