a = [map(int , raw_input().split()) for i in range(3)]
b = [[0 , 0 , 0] for i in range(3)]
for i in range(3):
    for j in range(3):
        if i: b[i][j] += a[i - 1][j]
        if j: b[i][j] += a[i][j - 1]
        if i < 2: b[i][j] += a[i + 1][j]
        if j < 2: b[i][j] += a[i][j + 1]
        b[i][j] += a[i][j]


for i in range(3):
     print str(b[i][0] % 2 ^ 1) + str(b[i][1] % 2 ^ 1) + str(b[i][2] % 2 ^ 1)
