a = [[]] * 3

for i in range(3):
    a[i] = map(int, raw_input().split())

b = [[1] * 3 for i in range(3)]

def f(i, j):
    if i >= 0 and i < 3 and j >= 0 and j < 3:
        b[i][j] = 1 - b[i][j]

for i in range(3):
    for j in range(3):
        if a[i][j] % 2 == 1:
            f(i, j), f(i - 1, j)
            f(i + 1, j)
            f(i, j - 1)
            f(i, j + 1)

for i in range(3):
    print str(b[i]).translate(None, '[, ]')
