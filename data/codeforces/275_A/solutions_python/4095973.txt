state, a = [[1]*5 for _ in [0]*5], [1,2,3]
press = [map(int, raw_input().split()) for _ in [0]*3]
for i in a:
    for j in a:
        if press[i-1][j-1]&1:
            for x, y in zip([0,-1,1,0,0],[0,0,0,-1,1]):
                state[i+x][j+y] ^= 1
for i in state[1:-1]:print ''.join(map(str, i[1:-1]))
