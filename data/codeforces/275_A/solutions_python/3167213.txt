m = [[] for i in range(3)]

for i in range(3):
	a = raw_input().split()
	m[i] = [int(a[j]) for j in range(3)]

for i in range(3):
	l = ""
	for j in range(3):
		sum = m[i][j]
		if (i > 0):
			sum += m[i-1][j]
		if (i < 2):
			sum += m[i+1][j]
		if (j > 0):
			sum += m[i][j-1]
		if (j < 2):
			sum += m[i][j+1]
		l += str(1 - (sum % 2))
	print l
