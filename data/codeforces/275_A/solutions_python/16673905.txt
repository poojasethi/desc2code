a = []
for i in xrange(3): a.append(map(int, raw_input().split()))
dx = [0, 0, 1, 0, -1]
dy = [0, -1, 0, 1, 0]
for i in xrange(3):
	ans = ""
	for j in xrange(3):
		s = 0
		for k in xrange(5):
			iNext = i + dy[k]
			jNext = j + dx[k]
			if 0 <= iNext < 3 and 0 <= jNext < 3: s += a[iNext][jNext]
		ans += str(1 - s % 2)
	print ans
