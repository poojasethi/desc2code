m = [[1] * 5 for _ in range(5)]
for r in range(1,4):
    a = map(lambda x: int(x) % 2, raw_input().split())
    for j in range(3):
        c = j + 1
        if a[j]:
            m[r-1][c] ^= 1
            m[r][c-1] ^= 1
            m[r][c] ^= 1
            m[r][c+1] ^= 1
            m[r+1][c] ^= 1
for r in range(1, 4):
    print "".join(str(i) for i in m[r][1:4])
