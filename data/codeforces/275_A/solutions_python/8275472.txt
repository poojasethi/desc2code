grid = [map(int, raw_input().split()) for i in range(3)]
res = [[1]*3 for i in range(3)]
dr = [-1, 0, 1, 0]
dc = [0, 1, 0, -1]
for i in range(3):
  for j in range(3):
    if grid[i][j] % 2 != 0:
      res[i][j] = 1-res[i][j]
      for k in range(4):
        r, c = i+dr[k], j+dc[k]
        if r < 0 or r >= 3: continue
        if c < 0 or c >= 3: continue
        res[r][c] = 1-res[r][c]
print '\n'.join(''.join(map(str, r)) for r in res)
