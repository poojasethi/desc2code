a = [map(int, raw_input().split()) for i in xrange(3)]
b = [[1 for j in xrange(3)] for i in xrange(3)]
for i in xrange(3):
    for j in xrange(3):
        for x0,y0 in (0, 0), (1, 0), (0, 1), (0, -1), (-1, 0):
            x, y = i+x0, j+y0
            if 0 <= x < 3 and 0 <= y < 3:
                b[x][y] = (b[x][y]+a[i][j])%2
print "\n".join("".join(map(str, i)) for i in b)
