a = [map(int, raw_input().split()) for _ in range(3)]
b = [[0] * 5 for _ in range(5)]
for i in range(1, 4):
  for j in range(1, 4):
    k = a[i - 1][j - 1]
    b[i - 1][j] += k
    b[i + 1][j] += k
    b[i][j - 1] += k
    b[i][j + 1] += k
    b[i][j] += k
print '\n'.join([''.join(map(str, [1 if b[i][j] % 2 == 0 else 0 for j in range(1, 4)])) for i in range(1, 4)])
