import sys
import itertools
a = [map(int, line.split()) for line in sys.stdin]
b = [[0]*3 for i in (1,2,3)]
P = list(itertools.product(range(3), range(3)))
for i, j in P:
    c = a[i][j]
    for i1, j1 in [(i+1, j), (i, j+1), (i-1, j), (i, j-1)]:
        if (i1, j1) in P:
            c += a[i1][j1]
    b[i][j] = (1+c)%2
for line in b:
    print ''.join(map(str, line))
