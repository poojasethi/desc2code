from sys import stdin

switches = [map(int, stdin.readline().split()) for x in xrange(3)]
grid = [[1] * 3 for x in xrange(3)]

d = ((0, 0), (0, 1), (1, 0), (0, -1), (-1, 0))

for x in xrange(3):
    for y in xrange(3):
        for dd in d:
            xx, yy = x + dd[0], y + dd[1] 
            if 3 > xx >= 0 and 3 > yy >= 0:
                grid[xx][yy] ^= (switches[x][y] & 1)

for row in grid:
    print ''.join(map(str, row))
