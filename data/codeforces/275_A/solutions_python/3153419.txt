t=[map(int,raw_input().split()) for _ in range(3)]
a=[[1,1,1],[1,1,1],[1,1,1]]
for i in range(3):
  for j in range(3):
    t[i][j] %= 2
for i in range(3):
  for j in range(3):
    if t[i][j]:
      for x in range(i-1,i+2):
        if 0<=x<=2:
          a[x][j] = 0 if a[x][j] else 1
      for y in (j-1,j+1):
        if 0<=y<=2:
          a[i][y] = 0 if a[i][y] else 1
for i in range(3):
  print ''.join(map(str,[a[i][j] for j in range(3)]))
