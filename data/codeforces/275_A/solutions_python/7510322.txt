#all around
d=[(0,0),(1,0),(-1,0),(0,1),(0,-1)]
def f(m,i,j):
    ret=0
    for dx,dy in d:
#ignore the out of range
        if i+dx>=0 and i+dx<=2 and j+dy>=0 and j+dy<=2:
            ret+=m[i+dx][j+dy]
    return ret

m=[map(int, raw_input().split()) for _ in xrange(3)]
ans=[[0]*3 for _ in xrange(3)]
for i in xrange(3):
    for j in xrange(3):
        ans[i][j]=(f(m,i,j)+1)%2
for i in xrange(3):
    print ''.join(map(str,ans[i]))
