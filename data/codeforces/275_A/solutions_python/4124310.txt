ri = lambda: map(int, raw_input().split())

m = []
for _ in range(3):
    m.append(map(lambda k: k%2, ri()))

def neibors(i,j):
    for di, dj in [(0,1), (0,-1), (1,0), (-1,0)]:
        if i+di >= 0 and j+dj >= 0 and i+di < 3 and j+dj < 3:
            yield m[i+di][j+dj]

for i in range(3):
    print "".join(map(str, ((m[i][j] + sum(neibors(i,j)) + 1) % 2 for j in range(3))))
