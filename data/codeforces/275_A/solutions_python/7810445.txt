b = [map(int,raw_input().split()) for x in xrange(3)]
c = [[1,1,1],[1,1,1],[1,1,1]]
for x in xrange(3):
    for y in xrange(3):
        if b[x][y] % 2 == 1:
            c[x][y] ^= 1
            if x+1 < 3:
                c[x+1][y] ^= 1
            if x-1 >= 0:
                c[x-1][y] ^= 1
            if y+1 < 3:
                c[x][y+1] ^= 1
            if y-1 >= 0:
                c[x][y-1] ^= 1
for x in c:
    print "".join(map(str,x))
