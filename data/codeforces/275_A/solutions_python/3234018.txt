N = 3
result = list()
for i in range(N) : 
	result.append([1] * 3)
press = list()
for i in range(N) : 
	press.append(map(int, raw_input().split()))
for i in range(N) : 
	for j in range(N) : 
		if press[i][j] % 2 == 1 : 
			result[i][j] ^= 1
			if i - 1 >= 0 : 
				result[i - 1][j] ^= 1
			if j - 1 >= 0 : 
				result[i][j - 1] ^= 1
			if i + 1 <  N : 
				result[i + 1][j] ^= 1
			if j + 1 <  N :
				result[i][j + 1] ^= 1
for i in range(N) : 
	print(''.join(map(str, result[i])))
