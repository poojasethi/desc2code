l=lambda:map(int, raw_input().split())
a=[[1,1,1],[1,1,1],[1,1,1]]
m=[l(),l(),l()]

for i in range(3):
    for j in range(3):
        if m[i][j]%2:
           
            for k in range(3):
                if abs(k - i) <= 1:
                    a[k][j] = 1 - a[k][j]
              
                if abs(k - j) == 1:
                    a[i][k] = 1 - a[i][k]
                    
       
print '\n'.join(''.join(map(str, e))for e in a)
