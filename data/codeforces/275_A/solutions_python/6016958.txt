l=[map(int,raw_input().split()) for i in xrange(3)]
r=[[1]*3 for i in xrange(3)]
for i in xrange(3):
    for j in xrange(3):
        if not l[i][j]%2: continue
        for (dx,dy) in ((0,0),(0,1),(0,-1),(1,0),(-1,0)):
            if 0 <= i + dx < 3 and 0 <= j + dy < 3 :
                r[i + dx][j + dy] = 1 - r[i + dx][j + dy]
                
for i in r:
    print ''.join(str(j) for j in i)
