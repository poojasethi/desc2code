m = [[] for i in range(3)]

for i in range(3):
	inp = raw_input().split()
	m[i] = [int(inp[j]) for j in range(3)]

for i in range(3):
	outp = ""
	for j in range(3):
		sum = m[i][j]
		if (i > 0):
			sum += m[i-1][j]
		if (i < 2):
			sum += m[i+1][j]
		if (j > 0):
			sum += m[i][j-1]
		if (j < 2):
			sum += m[i][j+1]
		outp += str(1 - (sum % 2))
	print outp
