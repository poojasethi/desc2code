r = [];
a = [];
for i in range(3):
    t = map(int, raw_input().split());
    r.append(t);
    a.append([1,1,1]);


def add(i, j, val):
    if 2>=i>=0 and 2>=j>=0:
        a[i][j]+= val;
        
for i in range(3):
    for j in range(3):
        add(i-1,j,r[i][j]);
        add(i+1,j,r[i][j]);
        add(i,j-1,r[i][j]);
        add(i,j+1,r[i][j]);
        add(i,j,r[i][j]);
for i in range(3):
    print ''.join(map(str, [[0,1][a[i][j] % 2] for j in range(3)]))
