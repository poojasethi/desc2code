a=[[x&1 for x in map(int,raw_input().split(' '))] for _ in range(3)]
b=[[1,1,1,1,1] for _ in range(5)]
for i in range(3):
     for j in range(3):
          if a[i][j]:
               b[i+1][j+1]=not b[i+1][j+1]
               b[i+2][j+1]=not b[i+2][j+1]
               b[i][j+1]=not b[i][j+1]
               b[i+1][j+2]=not b[i+1][j+2]
               b[i+1][j]=not b[i+1][j]
print '\n'.join([''.join(map(str,map(int,(b[i+1][1:4])))) for i in range(3)])
