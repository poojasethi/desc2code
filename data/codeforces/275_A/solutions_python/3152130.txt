A = []

for i in range(3):
    A.append(map(int, raw_input().split()))

for i in range(3):
    s = ''

    for j in range(3):
        res = A[i][j]

        if i > 0:
            res += A[i - 1][j]
        if j > 0:
            res += A[i][j - 1]
        if i < 2:
            res += A[i + 1][j]
        if j < 2:
            res += A[i][j + 1]

        if res % 2 == 0:
            s = s + str(1)
        else:
            s = s + str(0)
    print(s)
