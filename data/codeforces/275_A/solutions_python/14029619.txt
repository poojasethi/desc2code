grid=[0]*3
for i in xrange(3):
	grid[i]=map(int, raw_input().split())
for i in xrange(3):
	ans=""
	for j in xrange(3):
		cnt=grid[i][j]
		if i-1>=0:
			cnt+= grid[i-1][j]
		if i+1<3:
			cnt+= grid[i+1][j]
		if j-1>=0:
			cnt+= grid[i][j-1]
		if j+1<3:
			cnt+= grid[i][j+1]
		if cnt&1:
			ans+="0"
		else:
			ans+="1"
	print ans
