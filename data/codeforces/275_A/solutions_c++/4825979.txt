#include <iostream>
using namespace std;

int i,j,w,f[5][5];

int main()
{
	for (i=1; i<=3; i++) for (j=1; j<=3; j++)
	{
		cin>>w;
		f[i][j]+=w+1;
		f[i][j+1]+=w;
		f[i-1][j]+=w;
		f[i][j-1]+=w;
		f[i+1][j]+=w;
	}
	for (i=1; i<=3; i++) 
	{
		for (j=1; j<=3; j++) cout<<f[i][j]%2;
		cout<<endl;
	}
	return 0;
}
