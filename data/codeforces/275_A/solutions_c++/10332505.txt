#include <bits/stdc++.h>
using namespace std;
int lights[5][5];
int main()
{
	for(int i=1; i<4; i++) for(int j=1; j<4; j++) scanf("%d",&lights[i][j]);
	for(int i=1; i<4; i++)
	{
		for(int j=1; j<4; j++)
			printf("%d",((lights[i][j]+lights[i-1][j]+lights[i+1][j]+lights[i][j-1]+lights[i][j+1])%2) ? 0:1);
		printf("\n");
	}
	return 0;
}
