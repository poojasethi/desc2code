#include<cstdio>
using namespace std;
int A[5][5];

int main() {
	for(int i = 1; i <= 3; ++i)
		for(int j = 1; j <= 3; ++j)
			scanf("%d", &A[i][j]);
	for(int i = 1; i <= 3; ++i) {
		for(int j = 1; j <= 3; ++j)
			printf("%d", (A[i][j] + A[i - 1][j] + A[i][j - 1] + A[i + 1][j] + A[i][j + 1] + 1) & 1);
		printf("\n");
	}
	return 0;
}
