#include<bits/stdc++.h>
using namespace std;
int a[5][5];
int main()
{
  int i,j;
  for(i=1;i<4;i++)
    scanf("%d%d%d",&a[i][1],&a[i][2],&a[i][3]);
  for(i=1;i<4;i++)
  {
    for(j=1;j<4;j++)
      printf("%d",(a[i][j-1]+a[i-1][j]+a[i][j]+a[i+1][j]+a[i][j+1]+1)%2);
    puts("");
  }
  return 0;
}
