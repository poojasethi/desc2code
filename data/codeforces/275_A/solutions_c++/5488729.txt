#include <iostream>
using namespace std;
int main()
{
    int a,b,c,d,e,f,j,i,h;
    cin>>a>>b>>c>>d>>e>>f>>j>>i>>h; 
    cout<<(a+b+d+1)%2<<(b+a+e+c+1)%2<<(c+b+f+1)%2<<endl;
    cout<<(d+a+e+j+1)%2<<(e+b+f+i+d+1)%2<<(f+c+e+h+1)%2<<endl;
    cout<<(j+d+i+1)%2<<(i+j+e+h+1)%2<<(h+i+f+1)%2<<endl;    
    return 0;
}
