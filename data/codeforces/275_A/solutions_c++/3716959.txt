#include <iostream>
using namespace std;

int n = 3, v[4][4];

int main()
{
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= n; ++j)
			cin >> v[i][j];
	for (int i = 1; i <= n; ++i)
	{
		for (int j = 1; j <= n; ++j)
			cout << (v[i][j] + v[i][j+1] + v[i+1][j] + v[i-1][j] + v[i][j-1] + 1) % 2;
		cout << endl;
	}
}
