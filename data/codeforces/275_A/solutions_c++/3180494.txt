#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
int i,j,a[5][5];
int main()
{
	for(i=1;i<=3;++i)
	for(j=1;j<=3;++j)scanf("%d",&a[i][j]);
	for(i=1;i<=3;++i)
	{
		for(j=1;j<=3;++j)printf("%d",(a[i][j+1]+a[i][j-1]+a[i][j]+a[i-1][j]+a[i+1][j]+1)%2);
		printf("\n");
	}
	return 0;
}
