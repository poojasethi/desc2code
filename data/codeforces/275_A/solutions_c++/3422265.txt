#include<cstdio>
using namespace std;
int ile[5][5];
int main(){
    for(int i=1;i<4;i++)
    for(int j=1;j<4;j++)
        scanf("%d", &ile[i][j]);
    for(int i=1;i<4;i++){
    for(int j=1;j<4;j++)
        printf("%d", 1-(ile[i][j]+ile[i+1][j]+ile[i-1][j]+ile[i][j+1]+ile[i][j-1])%2);
    printf("\n");
    }
    return 0;
}
