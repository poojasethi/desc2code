#include <algorithm>
#include <iostream>
using namespace std;
typedef pair<int, int> pii;
typedef pair<pii, int> ord;
#define X first
#define Y second
const int N = 1e5 + 5;

int n, p, k;
ord a[N];
pii b[N];

bool cmp(const ord &x, const ord &y) {
	if (x.X.Y != y.X.Y)
		return x.X.Y > y.X.Y;
	return x.X.X < y.X.X;
}
int main() {
	ios_base::sync_with_stdio(false);
	cin >> n >> p >> k;
	for (int i = 0; i < n; i++)
		cin >> a[i].X.X >> a[i].X.Y, a[i].Y = i;
	sort(a, a + n, cmp);
	for (int i = 0; i < n; i++)
		b[i] = pii(-a[i].X.X, i);
	sort(b, b + n - (p - k));
	int m = -1;
	for (int i = 0; i < k; i++)
		cout << a[b[i].Y].Y + 1 << ' ', m = max(m, b[i].Y + 1);
//	cerr << m << '\n';
	for (int i = 0; i < p - k; i++)
		cout << a[i + m].Y + 1 << ' ';
	cout << '\n';
	return 0;
}
