#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int maxn=1e5+10;

int i,j,n,p,k,d,a1,a2,b1,b2,c1,c2;

struct Node{
	int x,y,Id,pos;
}A[maxn],B[maxn];

bool cmp1(Node a,Node b){
	return (a.y<b.y)||(a.y==b.y && a.x>b.x);
}
bool cmp2(Node a,Node b){
	return (a.x>b.x)||(a.x==b.x && a.y>b.y);
}

int main(){
	
	scanf("%d%d%d",&n,&p,&k);
	for(i=1;i<=n;i++){
		scanf("%d%d",&A[i].x,&A[i].y);
		A[i].Id=i;
	}
	
	sort(A+1,A+n+1,cmp1);
	
	for(i=p-k+1;i<=n;i++)B[i]=A[i],B[i].pos=i;
	sort(B+p-k+1,B+n+1,cmp2);
	
	j=n+1;
	for(i=p-k+1;i<=p;i++)j=min(j,B[i].pos),printf("%d ",B[i].Id);
	for(i=j-1;i>=j-p+k;i--)printf("%d ",A[i].Id);
	
	return 0;
}
