#include <algorithm>
#include <cstdio>
#include <queue>
using namespace std;

struct s{
	int a, b, i;
};
s l[100005];
bool c(s a, s b){
	return (a.b == b.b)?a.a<b.a:a.b>b.b;
}
bool operator<(s const &a, s const &b){
	return a.a > b.a;
}

priority_queue<s> pq;

int main(){
	int n, p, k;
	scanf("%d %d %d", &n, &p, &k);
	for(int i = 0; i < n; i++){
		scanf("%d %d", &l[i].a, &l[i].b);
		l[i].i = i+1;
	}
	sort(l, l+n, c);
	int at = k;
	for(int i = 0; i < n-(p-k); i++){
		if(pq.size() < k) pq.push(l[i]);
		else
			if(l[i].a > pq.top().a){
				pq.pop();
				pq.push(l[i]);
				at = i+1;
			}
	}
	while(!pq.empty()){
		printf("%d ", pq.top().i);
		pq.pop();
	}
	for(int j = 0; j < p-k; j++){
		printf("%d ", l[at].i);
		at++;
	}
	return 0;
}
