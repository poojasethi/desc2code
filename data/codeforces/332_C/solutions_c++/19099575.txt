#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int maxn=1e7+10;

int i,j,n,p,k,d,a1,a2,b1,b2,c1,c2;

struct Node{
	int x,y,Id;
	bool c;
}A[maxn];

bool cmp1(Node a,Node b){
	return (a.y<b.y)||(a.y==b.y && a.x>b.x)||(a.x==b.x&&a.y==b.y&&a.c<b.c);
}
bool cmp2(Node a,Node b){
	return (a.x>b.x)||(a.x==b.x && a.y>b.y);
}

int main(){
	
	scanf("%d%d%d",&n,&p,&k);
	for(i=1;i<=n;i++){
		scanf("%d%d",&A[i].x,&A[i].y);
		A[i].Id=i;
	}
	
	sort(A+1,A+n+1,cmp1);
//	for(int i=1;i<=n;i++)cout<<A[i].Id<<' ';cout<<endl;
	sort(A+p-k+1,A+n+1,cmp2);
	for(i=p-k+1;i<=p;i++)A[i].c=true;
	sort(A+1,A+n+1,cmp1);
//	for(int i=1;i<=n;i++)cout<<A[i].Id<<' ';cout<<endl;
	j=0;
	for(i=n;i;i--)if(A[i].c==true)j=i;
	for(i=j-1;i>=j-p+k;i--)A[i].c=true;
	for(i=1;i<=n;i++)if(A[i].c)printf("%d ",A[i].Id);
	
	return 0;
}
