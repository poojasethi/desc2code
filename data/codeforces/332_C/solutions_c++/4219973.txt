#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
int n,p,k;
struct Ord
{
    int a,b;
    int id;
}px[100100];
int id[100100];
bool cmp1(const Ord &x, const Ord &y)
{
    if (x.b==y.b) return x.a<y.a;
    return x.b>y.b;
}
bool cmp2(int x, int y)
{
    if (px[x].a==px[y].a) return px[x].b>px[y].b;
    return px[x].a>px[y].a;
}
int main()
{
    scanf("%d%d%d",&n,&p,&k);
    for (int i=0; i<n; ++i)
    {
        scanf("%d%d",&px[i].a,&px[i].b);
        px[i].id=i+1;
        id[i]=i;
    }
    sort(px,px+n,cmp1);
    sort(id,id+n-p+k,cmp2);
    int maxid=0;
    for (int i=0; i<k; ++i)
    {
        printf("%d ",px[id[i]].id);
        maxid=max(maxid,id[i]);
    }
    for (int i=1; i<=p-k; ++i)
    {
        printf("%d ",px[maxid+i].id);
    }
    return 0;
}
