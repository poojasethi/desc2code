#include <cstdio>
#include <algorithm>

using namespace std;

int n,p,k;

struct ps
{
    int hr,dp,id;
};

inline bool bydp (ps a,ps b)
{
    if (a.dp!=b.dp) return a.dp>b.dp;
    return a.hr<b.hr;
}

inline bool byhr(ps a,ps b)
{
    if (a.hr!=b.hr) return a.hr>b.hr;
    return a.dp>b.dp;
}

ps pr[100005];
char tak[100005];

int main()
{
    scanf("%d%d%d",&n,&p,&k);
    for (int i=1; i<=n; i++)
    {
        scanf("%d%d",&pr[i].hr,&pr[i].dp);
        pr[i].id=i;
    }
    sort(pr+1,pr+1+n,bydp);
    sort(pr+1,pr+1+n-p+k,byhr);
    for (int i=1; i<=k; i++)
    {
        printf("%d\n",pr[i].id);
        tak[pr[i].id]=1;
    }
    int mv=p-k;
    if (mv==0)
        return 0;
    sort(pr+1,pr+1+n-p+k,bydp);
    int lst=n;
    while (tak[pr[lst-1].id]==0)
        lst--;
    for (int i=lst; i<=lst+mv-1; i++)
        printf("%d\n",pr[i].id);
}
