#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
using namespace std;
struct order
{
	int id,id2;
	int a;
	int b;
}z[100001];
bool cmp1(order x,order y)
{
	if(x.b!=y.b) return x.b>y.b;
	else return y.a>x.a; 
}
bool cmp2(order x,order y)
{
	if(x.a==y.a) return x.b>y.b;
	return x.a>y.a;
}
bool cmp3(int x,int y){return z[x].b>z[y].b;}

int main()
{
	int n,p,k;
	while(scanf("%d%d%d",&n,&p,&k)==3)
	{
		for(int i=0;i<n;i++) 
		{
			scanf("%d%d",&z[i].a,&z[i].b);
			z[i].id=i+1;
		}
		std::sort(z,z+n,cmp1);
		for(int i=0;i<n;i++) z[i].id2=i;
		std::sort(z,z+n-p+k,cmp2);
		int m=0;
		for(int i=0;i<k;i++) 
		{
			printf("%d ",z[i].id);
			m=max(m,z[i].id2);
		}
		vector<int>ans;
		for(int i=k;i<n;i++) if(z[i].id2>m) ans.push_back(i);
		std::sort(ans.begin(),ans.end(),cmp3);
		for(int i=0;i<p-k;i++) printf("%d ",z[ans[i]].id);
		printf("\n");
	}
}
