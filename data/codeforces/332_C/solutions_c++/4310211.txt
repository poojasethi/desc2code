#include <iostream>
#include <algorithm>
#include <queue>
#include <cstdio>
#include <string.h>
#include <vector>
using namespace std;
#define rep(i, n) for (int i=0, _##i=(n); i<_##i; i++)
vector<pair<int,pair<int,int> > >v1;
vector<pair<int,int> >v2;
int N,P,K;
int main(){
    cin>>N>>P>>K;
    int a,b;
    rep (i,N){
        cin>>a>>b;
        v1.push_back(make_pair(-b, make_pair(a,i+1)));
    }
    sort(v1.begin(), v1.end());
    rep(i,N+K-P) v2.push_back(make_pair(-v1[i].second.first, i));
    sort(v2.begin(), v2.end());
    int curmax=0;
    rep (i,K){
        cout<<v1[v2[i].second].second.second<<" ";
        curmax = max(curmax,v2[i].second);
    }
    for (int i = curmax+1; i<=curmax+P-K; i++){
        cout<<v1[i].second.second<<" ";
    }
    return 0;
}
