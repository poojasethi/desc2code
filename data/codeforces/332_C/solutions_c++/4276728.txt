#include <cstdio>
#include <algorithm>

using namespace std;

struct up
{
    int hr,dp,id,tak;
};

inline bool bydp(up a,up b)
{
    if (a.dp!=b.dp)
        return a.dp>b.dp;
    if (a.hr!=b.hr)
        return a.hr<b.hr;
    return a.tak>b.tak;
}

inline bool byhr(up a,up b)
{
    if (a.hr!=b.hr)
        return a.hr>b.hr;
    return a.dp>b.dp;
}

int n,p,k;
up st[100005];

int main()
{
    scanf("%d%d%d",&n,&p,&k);
    for (int i=1; i<=n; i++)
    {
        scanf("%d%d",&st[i].hr,&st[i].dp);
        st[i].id=i;
    }
    sort(st+1,st+1+n,bydp);
    sort(st+1,st+1+n-p+k,byhr);
    for (int i=1; i<=k; i++)
        st[i].tak=1;
    sort(st+1,st+1+n,bydp);
    int lp=n;
    while (st[lp-1].tak==0)
        lp--;
    while (k++<p)
        st[lp++].tak=1;
    for (int i=1; i<=n; i++)
        if (st[i].tak)
        printf("%d ",st[i].id);

}
