#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
struct xx{
	int x,y,z,num;
}s1[110000],s2[110000];
int n,p,k;
bool cmp1(const xx a,const xx b){
	return a.y>b.y || (a.y==b.y  && a.x<b.x);
}
bool cmp(const xx a,const xx b){
	return a.x>b.x || (a.x==b.x && a.z<b.z);
}
bool cmp2(const xx a,const xx b){
	return a.y>b.y;
}
int main(){
	scanf("%d%d%d",&n,&p,&k);
	for (int i=1;i<=n;i++){
	  scanf("%d%d",&s1[i].x,&s1[i].y);
	  s1[i].num=i;
	}
	sort(s1+1,s1+1+n,cmp1);
	for (int i=1;i<=n;i++){
		s2[i]=s1[i];
		s2[i].z=i;
	}
	sort(s2+1,s2+1+n-(p-k),cmp);
	int ma=0;
	for (int i=1;i<=k;i++){
		cout<<s2[i].num<<' ';
		ma=max(ma,s2[i].z);
	}
	ma++;
	if (ma<=n)sort(s1+1+ma,s1+1+n,cmp2);
	for (int i=ma;i<ma+p-k;i++){
		cout<<s1[i].num<<' ';
	}
}
