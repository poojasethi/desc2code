#include<cstdio>
#include<cstring>
#include<vector>
#include<algorithm>
using namespace std;
const int maxn=100005;
int n,p,k;
struct order {
    int id,a,b;
};
order s[maxn];
bool cmpx(const order &t1,const order &t2) {
    return t1.a>t2.a || (t1.a==t2.a && t1.b>t2.b);
}
bool cmpy(const order &t1,const order &t2) {
    return t1.b>t2.b || (t1.b==t2.b && t1.a<t2.a);
}
int main() {
    scanf("%d%d%d",&n,&p,&k);
    for (int i=0; i<n; i++) {
        scanf("%d%d",&s[i].a,&s[i].b);
        s[i].id=i+1;
    }
    sort(s,s+n,cmpy);
    sort(s,s+n-(p-k),cmpx);
    order c=s[0];
    for (int i=0; i<k; i++) {
        printf("%d ",s[i].id);
        if (cmpy(c,s[i])) c=s[i];
    }
    sort(s+k,s+n,cmpy);
    for (int i=k,j=0; i<n && j<p-k; i++)
        if (!cmpy(s[i],c)) {
            printf("%d ",s[i].id);
            j++;
        }
    return 0;
}
