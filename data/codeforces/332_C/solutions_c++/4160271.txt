#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

#define MAXN 100010

struct eugin
{
  int a,b,v,y;
};

int n,p,k;
eugin s1[MAXN];
eugin s2[MAXN];

bool cmp1(eugin q,eugin w)
{
  if (q.b>w.b) return true;
  if (q.b<w.b) return false;
  if (q.a<w.a) return true;
  return false;
}

bool cmp2(eugin q,eugin w)
{
  if (q.a>w.a) return true;
  if (q.a<w.a) return false;
  if (q.b>w.b) return true;
  return false;
}

int main()
{
  int i,top;
  scanf("%d%d%d",&n,&p,&k);
  for (i=1;i<=n;i++)
  {
    scanf("%d%d",&s1[i].a,&s1[i].b);
    s1[i].v=i;
  }
  sort(s1+1,s1+1+n,cmp1);
  for (i=1;i<=n;i++)
    s1[i].y=i;
  memcpy(s2,s1,sizeof(s1));
  sort(s2+1,s2+1+n-p+k,cmp2);
  top=0;
  for (i=1;i<=k;i++)
  {
    printf("%d ",s2[i].v);
    if (s2[i].y>top)
      top=s2[i].y;
  }
  for (i=top+1;i<=top+p-k;i++)
    printf("%d ",s1[i].v);
  puts("");
  return 0;
}
