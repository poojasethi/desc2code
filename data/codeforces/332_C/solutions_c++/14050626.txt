#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
    int a,b,c,id;
} a[N];
int i,j,k,n,m;
inline bool cc1(const ww &a,const ww &b) {
    return a.b<b.b||a.b==b.b&&a.a>b.a;
}
inline bool cc2(const ww &a,const ww &b) {
    return a.a>b.a||a.a==b.a&&a.c>b.c;
}
inline bool cc3(const ww &a,const ww &b) {
    return a.c<b.c;
}
int main() {
    scanf("%d%d%d",&n,&m,&k);
    For(i,1,n) scanf("%d%d",&a[i].a,&a[i].b),a[i].id=i;
    sort(a+1,a+n+1,cc1);
    m-=k;
    For(i,1,n) a[i].c=i;
    sort(a+m+1,a+n+1,cc2);
    int B=n;
    For(i,m+1,m+k) printf("%d ",a[i].id),B=min(B,a[i].c);
    sort(a+1,a+n+1,cc3);
    For(i,1,m) printf("%d ",a[B-i].id);
    printf("\n");
    return 0;
}
