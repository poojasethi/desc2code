#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 100010;

struct node {
    int x, y;
    int id;
}a[N];

bool cmpx(const node &a, const node &b) {
    return a.x > b.x || (a.x == b.x && a.y > b.y);
}
bool cmpy(const node &a, const node &b) {
    return a.y > b.y || (a.y == b.y && a.x < b.x);
}

int main() {
    int n, p, k;
    scanf("%d%d%d", &n, &p, &k);
    for (int i = 0; i < n; i++) {
        scanf("%d%d", &a[i].x, &a[i].y);
        a[i].id = i + 1;
    }
    sort(a, a + n, cmpy);
    sort(a, a + n - p + k, cmpx);
    node c = a[0];
    for (int i = 0; i < k; i++) {
        printf("%d ", a[i].id);
        if (cmpy(c, a[i])) c = a[i];
    }
    sort(a + k, a + n, cmpy);
    for (int i = k, j = 0; i < n && j < p - k; i++) 
        if (!cmpy(a[i], c)) {
            printf("%d ", a[i].id);
            j++;
        }
    return 0;
}
