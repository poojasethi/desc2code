#include <iostream>
#include <vector>
#include <fstream>
#include <algorithm>

using namespace std;
#define N 100001
#define ee 1000000007


struct node {
    int x, y, h, hh;
} a[N], b[N] ;

bool cmp1 ( node i, node j){
    return ( i.y < j.y || (i.y == j.y && i.x > j.x));
}

bool cmp2 ( node i, node j){
    return ( i.x > j.x || (i.x == j.x && i.y > j.y ) || ( i.x == j.x && i.y == j.y && i.h < j.h ) );
}

int n, p, k;

int main()
{
    cin >> n >> p >> k;
    for (int i = 1 ; i <=n; i++){
        cin >> a[i].x >> a[i].y;
        a[i].h = i;
    }
    sort ( a + 1 , a + n + 1, cmp1 );
    for (int i = 1; i <= n; i++) { a[i].hh = i; b[i] = a[i]; }
    sort ( a + p - k + 1 , a + n + 1 , cmp2);
    int j = n;
    for (int i = p-k+1; i <=p; i++){
        cout << a[i].h << " ";
        j = min ( a[i].hh, j);
    }
    for (int i = j-1; i >= j - p + k ; i-- ) cout << b[i].h << " ";
    return 0;
}
