#include<cstdio>
#include<iostream>
#define MaxN 1000000 + 1
#include<algorithm>

using namespace std;

struct node{
	int x,y,z,u;
}a[MaxN],b[MaxN];
bool compare1(node l,node r){
    return( l.y > r.y || (l.y == r.y && l.x < r.x) );
    }
bool compare2(node l,node r){
    return( l.x > r.x || (l.x == r.x && l.u < r.u) );
    }
int n,p,k,st;
void read()
{
    scanf("%d%d%d",&n,&p,&k);

	for (int i=1; i<=n; i++)
	 {
	  scanf("%d%d",&a[i].x,&a[i].y);
      a[i].z=i;
	 }
}
void solve(){
    sort(a+1,a+1+n,compare1);
    for (int i=1; i<=n; i++)
     {
	  a[i].u=i;
	  b[i]=a[i];
     }
	sort(a+1,a+1+(n-p+k),compare2);
     int st=0;
	for(int i=1;i<=k;i++){
		st=max(st,a[i].u);
		printf("%d ",a[i].z);
	}
	for(int i=st+1;i<=st+p-k;i++)
		printf("%d ",b[i].z);
}
int main(){
        read();
        solve();
}
