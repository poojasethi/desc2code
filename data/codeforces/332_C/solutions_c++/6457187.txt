#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
struct node
{int a,b,id,id2;};
bool cmp(node x ,node y)
{
    if(x.b!=y.b) return x.b>y.b;
    return x.a<y.a;
}
bool cmp2(node x ,node y)
{
    if(x.a!=y.a) return x.a>y.a;
    return x.b>y.b;
}
int n,p,k,endi;
node in[100010],in2[100010];
int main()
{
    //freopen("r.txt","r",stdin);
    scanf("%d %d %d",&n,&p,&k);
    for(int i=0;i<n;i++)
    {
        scanf("%d %d",&in[i].a,&in[i].b);
        in[i].id=i;
    }
    sort(in,in+n,cmp);
    for(int i=0;i<n;i++) in[i].id2=i;
    memcpy(in2,in,sizeof(in));
    sort(in,in+n-(p-k),cmp2);
    for(int i=0;i<k;i++)
    {
        printf("%d ",in[i].id+1);
        endi=max(endi,in[i].id2);
    }
    sort(in2+endi+1,in2+n,cmp);
    for(int i=endi+1;i<=endi+(p-k);i++)
        printf("%d ",in2[i].id+1);

    return 0;
}
