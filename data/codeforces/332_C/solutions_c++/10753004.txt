#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
const int maxx=1e5+5;
struct node
{
    int a,b,pos1,pos2;
}node1[maxx],node2[maxx];
bool cmp1(node x,node y)
{
    if(x.b!=y.b) return x.b>y.b;
    return x.a<y.a;
}
bool cmp2(node x,node y)
{
    if(x.a!=y.a) return x.a>y.a;
    return x.b>y.b;
}
int main()
{
    int n,p,k,e=0;
    scanf("%d%d%d",&n,&p,&k);
    for(int i=0;i<n;i++)
    {
        scanf("%d%d",&node1[i].a,&node1[i].b);
        node1[i].pos1=i;
    }
    sort(node1,node1+n,cmp1);
    for(int i=0;i<n;i++) node1[i].pos2=i;
    memcpy(node2,node1,sizeof(node1));
    sort(node1,node1+n-(p-k),cmp2);
    for(int i=0;i<k;i++)
    {
        printf("%d ",node1[i].pos1+1);
        e=max(e,node1[i].pos2);
    }
    sort(node2+e+1,node2+n,cmp1);
    for(int i=e+1;i<=e+(p-k);i++)
        printf("%d ",node2[i].pos1+1);
}
