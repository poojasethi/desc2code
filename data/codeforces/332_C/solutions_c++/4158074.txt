#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
struct nod
{
	int a,b,ind;
	nod() {a=b=ind=0;}
};
nod s[100005];
nod ans[100005];
bool cmp(nod i,nod j)
{
	if (i.b!=j.b) return i.b<j.b;
	return i.a>j.a;
}
bool cmp2(nod i,nod j)
{
	if (i.a==j.a) return i.b>j.b;
	return i.a>j.a;
}
bool cmp3(nod i,nod j)
{
	return i.b>j.b;
}
int main()
{
	int i,j,k,l,n,m,M=0;
	cin>>n>>m>>k;
	for (i=1;i<=n;i++) {cin>>s[i].a>>s[i].b; s[i].ind=i;}
	sort(s+1,s+1+n,cmp);
	sort(s+m-k+1,s+1+n,cmp2);
	j=1000000005;
	for (i=m-k+1;i<=m;i++)
	{
		ans[++M]=s[i];
		j=min(s[i].b,j);
	}
	sort(s+m+1,s+1+n,cmp3);
	for (i=m+1;i<=n && M<m;i++)
	{
		if (s[i].b<j)
		{
			ans[++M]=s[i];
		}
	}
	for (i=m-k;i>=1 && M<m;i--)
	{
		ans[++M]=s[i];
	}
	for (i=1;i<=m;i++) {cout<<ans[i].ind<<" ";}
	cout<<endl;
}
