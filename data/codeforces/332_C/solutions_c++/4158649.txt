#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;
#define N 100001
#define ee 1000000007

struct node{
       int x , y , h , hh;
}a[N] , b[N];

bool cmp1(node i , node j){
     return (i.y < j.y || (i.y == j.y && i.x > j.x));
}

bool cmp2(node i , node j){
     return (i.x > j.x || (i.x == j.x && i.y > j.y) || (i.x == j.x && i.y == j.y && i.h < j.h));
}

int n , p , k;

int main(){
    int i , j;
    cin >> n >> p >> k;
    for (i=1;i<=n;++i){
        cin >> a[i].x >> a[i].y;
        a[i].h = i;
    }
    sort(a+1,a+n+1,cmp1);
    for (i=1;i<=n;++i) { a[i].hh = i; b[i] = a[i];}
    sort(a+p-k+1,a+n+1,cmp2);
    j = n;
    for (i=p-k+1;i<=p;++i){
        cout << a[i].h << " ";
        j = min(a[i].hh,j);
    }
    for (i=j-1;i>=j-(p-k);i--) cout << b[i].h << " ";
    cout << endl;
    cin >> n;
    return 0;
}
