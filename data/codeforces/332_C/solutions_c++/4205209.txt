#include <iostream>
#include <algorithm>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

struct order{
    int dex;
    int a;
    int b;
};

int n,p,k,a,b,mv,ma,t;
order s[100010];

bool cmp1(order x,order y){
    if(x.b==y.b)return x.a<y.a;
    else return x.b>y.b;
}

bool cmp2(order x, order y){
    if(x.a==y.a)return x.b>y.b;
    return x.a>y.a;
}

int main(){
    cin>>n>>p>>k;
    rep(i,n){
    cin>>s[i].a>>s[i].b;
    s[i].dex = i+1;
    }
    sort(s,s+n,cmp1);
    sort(s,s+n-p+k,cmp2);
    sort(s+k,s+n,cmp1);

    rep(i,k)if(mv>s[i].b || i==0){mv = s[i].b; ma=i;}
    for(t=k;t<n;t++){
    if(mv>s[t].b)break;
    if(mv==s[t].b && s[t].a>=s[ma].a)break;
    }
    
    rep(i,k)cout<<s[i].dex<<" ";
    rep(i,p-k)cout<<s[t+i].dex<<" ";cout<<endl;

}
