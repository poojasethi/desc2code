#include<stdio.h>
#include<algorithm>
using namespace std;
struct node{
	int x,y;
	int num;
};
int n,p,k;
int label[102456];
node a[102456];
inline bool cmp1(node x,node y){
	return x.y>y.y || (x.y==y.y && x.x<y.x);
}
inline bool cmp2(int x,int y){
	return a[x].x>a[y].x || (a[x].x==a[y].x && a[x].y>a[y].y);
}
int main(){
	scanf("%d%d%d",&n,&p,&k);
	int x,y;
	for(int i=0;i<n;i++){
		scanf("%d%d",&x,&y);
		a[i].x=x,a[i].y=y;
		a[i].num=i+1;
		label[i]=i;
	}
	sort(a,a+n,cmp1);
	sort(label,label+n-p+k,cmp2);
	int maxx=-1;
	printf("%d",a[label[0]].num);
	maxx=label[0];
	for(int i=1;i<k;i++){
		printf(" %d",a[label[i]].num);
		maxx=max(maxx,label[i]);
	}
	for(int i=1;i<=p-k;i++)
		printf(" %d",a[maxx+i].num);
	puts("");
	return 0;
}
