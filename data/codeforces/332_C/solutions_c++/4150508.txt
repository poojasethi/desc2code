#include <algorithm>
using namespace std;
ifstream in("input.txt");

long n,p,k,t;

struct lol {
long ind,val1,val2;
} a[123456],b[123456];

bool cmp(lol q,lol w){
if (q.val2!=w.val2)
return q.val2<w.val2; // !!!!!!!!!!!!!!!!!!!!!!!!!!!
return q.val1>w.val1;
}

long tek=0;

bool cmp1(lol q,lol w){
if (q.val1!=w.val1)
return q.val1<w.val1;
return q.val2<w.val2;
}

bool ac[123456];

int main()
{
    cin>>n>>p>>k;
    for (long i=0;i<n;i++){
        cin>>a[i].val1>>a[i].val2;
        a[i].ind=i+1;
    }
  sort(a,a+n,cmp);
for (long i=p-k;i<n;i++)
b[tek++]=a[i];

sort(b,b+n-p+k,cmp1);

for (long i=n-p+k-1;i>n-p-1;i--){
    cout<<b[i].ind<<" ";
    ac[b[i].ind]=1;
    }
long sch=0;
long i=n-1;
while (sch!=k){
    if (ac[a[i].ind])
        sch++;
    i--;
}
for (long j=i;j>i-p+k;j--)
    cout<<a[j].ind<<" ";

    return 0;
}
