#include<cstdio> 
#include<iostream>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
struct xx{
	int x,y,z,zz;
}a[1000001],b[1000001];
int n,p,k,st;
bool cmp1(xx l,xx r){return(l.y>r.y||(l.y==r.y&&l.x<r.x));}
bool cmp2(xx l,xx r){return(l.x>r.x||(l.x==r.x&&l.zz<r.zz));}
int main(){
//	freopen("xx.in","r",stdin);
//	freopen("xx.out","w",stdout);
	scanf("%d%d%d",&n,&p,&k);
	for (int i=1; i<=n; i++)
	 { 
	  scanf("%d%d",&a[i].x,&a[i].y);
      a[i].z=i;
	 }
	sort(a+1,a+1+n,cmp1);
    for (int i=1; i<=n; i++)
     {
	  a[i].zz=i;
	  b[i]=a[i];
     }
	sort(a+1,a+1+(n-p+k),cmp2);  
     int st=0;
	for(int i=1;i<=k;i++){
		st=max(st,a[i].zz);
		printf("%d ",a[i].z);
	}
	for(int i=st+1;i<=st+p-k;i++)
		printf("%d ",b[i].z);
//	fclose(stdin);
//	fclose(stdout);
}
