#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct Node
{
    int a,b,id;
}e[100005];
bool cmp1(Node x,Node y)
{
    if(x.b!=y.b)return x.b>y.b;
    return x.a<y.a;
}
bool cmp2(Node x,Node y)
{
    if(x.a!=y.a)return x.a>y.a;
    return x.b>y.b;
}
int main()
{
    int n,p,k;
    while(scanf("%d%d%d",&n,&p,&k)!=EOF)
    {
        for(int i=0;i<n;++i)
        {
            e[i].id=i+1;
            scanf("%d%d",&e[i].a,&e[i].b);
        }
        sort(e,e+n,cmp1);
        sort(e,e+n-p+k,cmp2);
        Node c=e[0];
        for(int i=0;i<k;++i)
        {
            printf("%d ",e[i].id);
            if(cmp1(c,e[i]))c=e[i];
        }
        sort(e+k,e+n,cmp1);
        for(int i=k,j=0;i<n&&j<p-k;++i)
            if(!cmp1(e[i],c))
            {
                printf("%d ",e[i].id);
                ++j;
            }
        printf("\n");
    }
}
