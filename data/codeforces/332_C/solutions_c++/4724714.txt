#include <cstdio>
#include <algorithm>
#define maxn 100005
using namespace std;

struct st
{
	int a,b,c,p;
	bool operator <(const st&B) const {return b<B.b||(b==B.b&&a>B.a)||(b==B.b&&a==B.a&&c<B.c);}
} a[maxn];
int N,P,K,q[maxn];

bool cmp(const st&A,const st&B) {return A.a>B.a||(A.a==B.a&&A.b>B.b);}

void init()
{
	scanf("%d%d%d",&N,&P,&K);
	for (int i=1; i<=N; i++) scanf("%d%d",&a[i].a,&a[i].b),a[i].p=i;
}

void doit()
{
	sort(a+1,a+N+1);
	for (int i=1; i<=P-K; i++) a[i].c=1;
	sort(a+1,a+N+1,cmp);
	for (int i=1,t=K; i<=N&&t; i++) if (!a[i].c) a[i].c=2,t--,q[++q[0]]=a[i].p;
	sort(a+1,a+N+1);
	for (int i=N,s=0,t=P-K; i&&t; i--) if (a[i].c!=2&&s>=K) t--,q[++q[0]]=a[i].p; else if (a[i].c==2) s++;
	for (int i=1; i<=P; i++) printf("%d%c",q[i],i==P?'\n':' ');
}

int main()
{
	init();
	doit();
	return 0;
}
