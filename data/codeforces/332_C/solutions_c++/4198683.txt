#include <cstdio>
#include <algorithm>
#include <iostream>
#include <string>
#include <vector>
#include <set>

using namespace std;

int n, p, k;

struct q{
	int a, b, n;
	q() {}
	q(int a, int b, int n) : a(a), b(b), n(n) {}
	friend bool operator < (q x, q y)
	{
		if (x.b != y.b)
			return x.b > y.b;
		return x.a < y.a;
	}
};

q x[100005];
set<q> mx;

int main()
{
	scanf("%d%d%d", &n, &p, &k);
	for (int i = 0; i < n; ++i)
	{
		scanf("%d %d", &x[i].a, &x[i].b);
		x[i].n = i + 1;
	}
	sort(x, x + n);
	for (int i = 0; i < n - p + k; ++i)
		mx.insert(q(i, x[i].a, x[i].n));
	int t = 0, r = 0;
	for (set<q>::iterator qi = mx.begin(); (qi != mx.end()) && t < k; ++qi, ++t)
	{
		printf("%d ", qi->n);
		if (qi->a > r)
			r = qi->a;
	}
	for (int i = 1; i <= p - k; ++i)
		printf("%d ", x[i + r].n);
	return 0;
}
