#include <cstdio>
#include <queue>
#include <utility>
#include <algorithm>
#define X first
#define Y second
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

int n,k,p,low;
pair< pair<int,int>,int > a[100010];
priority_queue< pair<int,pair<int,int> > > pq;

int main() {
	scanf("%d%d%d",&n,&k,&p);
	REP(i,n) {
		scanf("%d%d",&a[i].X.Y,&a[i].X.X);
		a[i].X.Y=-a[i].X.Y;
		a[i].Y=i;
	}
	sort(a+1,a+n+1);
	low=n;
	for (int i=k-p+1;i<=n;++i)
		pq.push(make_pair(-a[i].X.Y,make_pair(i,a[i].Y)));
	REP(i,p) {
		printf("%d ",pq.top().Y.Y);
		low=min(low,pq.top().Y.X-1);
		pq.pop();
	}
	for (int i=low;i>low-k+p;--i)
		printf("%d ",a[i].Y);
	return 0;
}
