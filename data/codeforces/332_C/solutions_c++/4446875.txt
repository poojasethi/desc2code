#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct event
{
       int a,b,id,you;
}A[100001],temp[100001];
int n,p,k;
bool cmp1(event a,event b)
{
     return (a.b>b.b)||((a.b==b.b)&&(a.a<b.a));
}
bool cmp2(event a,event b)
{
     return (a.a>b.a)||((a.a==b.a)&&(a.you<b.you));
}
bool cmp3(event a,event b)
{
     return a.b>b.b;
}
int main()
{
    int i,cur;
    scanf("%d%d%d",&n,&p,&k);
    for (i=1;i<=n;i++)
    {
        scanf("%d%d",&A[i].a,&A[i].b);
        A[i].id=i;
    }
    sort(A+1,A+n+1,cmp1);
    memcpy(temp,A,sizeof(A));
    for (i=1;i<=n;i++) A[i].you=i;
    sort(A+1,A+n-(p-k)+1,cmp2);
    cur=0;
    for (i=1;i<=k;i++)
    {
        printf("%d ",A[i].id);
        if (A[i].you>cur) cur=A[i].you;
    }
    sort(temp+cur+1,temp+n+1,cmp3);
    for (i=1;i<=p-k;i++) printf("%d ",temp[i+cur].id);
    printf("\n"); return 0;
}
