v = [map(int, raw_input().split(' ')) + [i] for i in range(n)];
v1 = sorted(v, key = lambda x : (x[1], -x[0]));
v2 = sorted([v1[i] + [i] for i in range(p - k, n)], key = lambda x : (-x[0], -x[3]));
min_id = n + 1;
for x in v2[:k]:
    print x[2] + 1, ;
    min_id = min(min_id, x[3]);
