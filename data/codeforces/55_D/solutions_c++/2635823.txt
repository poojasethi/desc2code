#include <stdio.h>
#include <string.h>
#define forn(i,n) for(int i=0;i<n;i++)
#define clr(x,y) memset(x,y,sizeof(x))
#define ll __int64
ll dp[25][2530][50];int d[25],m[2530],s[50],top,mm[50][50];ll u,v;
int gcd(int a,int b){return b?gcd(b, a%b):a;}
int lcm(int a,int b){return a/gcd(a, b)*b;}
ll dfs(int p,int y,int lc,bool f){
if(p<0)return y%s[lc]==0;
if(f&&dp[p][y][lc]!=-1) return dp[p][y][lc];
ll a=0;int lim=f?9:d[p];
forn(i,lim+1) a+=dfs(p-1,(y*10+i)%2520,(i==0)?lc:mm[lc][m[i]],(f||i<lim));
if(f)dp[p][y][lc]=a;return a;
}
ll g(ll v){int l=0;for(;v;v/=10)d[l++]=v%10;return dfs(l-1,0,0,0);}
int main(){
int t,i,j;top=0;
for(i=1;i<=2520;i++)if(2520%i==0){s[top]=i;m[i]=top++;}
for(i=0;i<top;i++)for(j=0;j<top;j++)mm[i][j]=m[lcm(s[i],s[j])];
clr(dp,-1);g(1000000000000000000);
scanf("%d", &t);
while(t--){scanf("%I64d%I64d", &u, &v);printf("%I64d\n", g(v)-g(u-1));}
return 0;
}
