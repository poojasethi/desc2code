#include <iostream>
#include <stdio.h>
#include <string.h>
#define  ll  long long
#define  mod 2520
using namespace std;
ll t,dig[20],id[2555],s2;
ll n,m,f[20][2555][50];
ll gcd(int a,int b){return !b?a:gcd(b,a%b);}
ll lcm(int a,int b){return a/gcd(a,b)*b;}
ll dp(int pos,int sum,int pp,bool flag){
   if(pos==-1)return !(sum%pp);
   if(!flag&&f[pos][sum][id[pp]]!=-1)return f[pos][sum][id[pp]];
   ll res=0;
   for(int i=0,l=flag?dig[pos]:9;i<=l;i++)
      res+=dp(pos-1,(sum*10+i)%mod,!i?pp:lcm(pp,i),flag&&i==l);
   return flag?res:f[pos][sum][id[pp]]=res;
}
ll go(ll x){
   int cnt=0;
   while(x)dig[cnt++]=x%10,x/=10;
   return dp(cnt-1,0,1,1);
}
int main(){
    scanf("%d",&t);
    for(int i=1;i<=mod;i++)
       if(!(mod%i))id[i]=s2++;
    memset(f,-1,sizeof(f));
    while(t--){
       cin>>n>>m;
       cout<<go(m)-go(n-1)<<endl;
    }
    return 0;
}
