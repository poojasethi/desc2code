#include <cstdio>
using namespace std;
long long l,r,f[20][2520][52];
int x[2524],digit[24];
int gcd(int x,int y) {if (!y) return x; return gcd(y,x%y);}
int lcm(int x,int y) {return x*y/gcd(x,y);}
long long calc(int now,int MOD,int LCM,bool free)
{
	if (!now) return (MOD%LCM==0);
	if ((free)&&(f[now][MOD][x[LCM]]>=0)) return f[now][MOD][x[LCM]];
	int max=(free)?9:digit[now]; long long ret=0;
	for (int i=0;i<=max;i++) ret+=calc(now-1,(MOD*10+i)%2520,(i)?lcm(LCM,i):LCM,(free)||(i<max));
	if (free) f[now][MOD][x[LCM]]=ret;
	return ret;
}
long long get(long long n)
{
	int len=0;
	while (n) digit[++len]=n%10,n/=10;
	return calc(len,0,1,false);
}
int main()
{
	int T,tot=0;
	for (int i=1;i<=2520;i++) if (2520%i==0) x[i]=++tot;
	for (int i=0;i<20;i++) for (int j=0;j<2520;j++) for (int k=0;k<52;k++) f[i][j][k]=-1;
	scanf("%d",&T);
	while (T--)
	{
		scanf("%I64d%I64d",&l,&r);
		printf("%I64d\n",get(r)-get(l-1));
	}
	return 0;
}
