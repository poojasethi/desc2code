#include<iostream>
#include<memory.h>
#include<algorithm>
using namespace std;
typedef long long W;
W i,M[9999],F[22][3333][55],T,D[22],s,L,R,Tr[999][99],j;
W S(W r,W f,W l,W d){
	if(!r)return l%d==0;
	l%=252;W rt=0,i;
	if(!f){
		if(F[r][l][M[d]]>=0)return F[r][l][M[d]];
		for(i=0;i<10;i++)rt+=S(r-1,0,Tr[l][i],d*(i+!i)/__gcd(d,(i+!i)));
		return (F[r][l][M[d]]=rt);
	}else{
		for(i=0;i<=D[r];i++)rt+=S(r-1,i==D[r],Tr[l][i],d*(i+!i)/__gcd(d,(i+!i)));
		return rt;
	}
}
W C(W u){
	s=0;
	for(;u;u/=10)D[++s]=u%10;
	return S(s,1,0,1);
}
int main(){
	memset(F,128,sizeof(F));
	for(i=1;i<=2520;i++)if(2520%i==0)M[i]=++L;
	for(i=0;i<253;i++)for(j=0;j<10;j++)Tr[i][j]=i*10+j;
	cin>>T;
	for(;T;T--){
		cin>>L>>R;
		cout<<C(R)-C(L-1)<<endl;
	}
	return 0;
}
