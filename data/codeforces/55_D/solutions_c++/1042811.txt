#include <cstdio>
#include <cstring>
typedef long long LL;
int h[2521],a[20];
LL f[20][48][2520];

int gcd(int a,int b) {return b? gcd(b,a%b):a;}

LL dp(int p,int m,int x,bool same)
{
	if (!p) return x%m==0;
	if (!same && f[p][h[m]][x]>=0) return f[p][h[m]][x];
	LL s=0;
	for (int i=0; !same && i<10 || i<=a[p]; ++i)
		s+=dp(p-1,i>1? m/gcd(m,i)*i:m,(x*10+i)%2520,same && i==a[p]);
	if (!same) f[p][h[m]][x]=s;
	return s;
}

LL calc(LL num)
{
	int len=0;
	while (num) a[++len]=num%10, num/=10;
	return dp(len,1,0,1);
}

int main()
{
	int t=0;
	for (int i=1; i<=2520; ++i)
		if (2520%i==0) h[i]=t++;
	scanf("%d",&t);
	LL l,r;
	memset(f,-1,sizeof f);
	while (t--)
	{
		scanf("%I64d%I64d",&l,&r);
		printf("%I64d\n",calc(r)-calc(l-1));
	}
	return 0;
}
