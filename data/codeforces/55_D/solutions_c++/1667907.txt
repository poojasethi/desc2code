#include<stdio.h>
#include<memory.h>
#include<algorithm>
using namespace std;
typedef long long W;
W i,F[20][252][55],T,L,R;
short Tr[999][99],j,Px[2522][11];
char M[9999],D[22],s;
W S(char r,bool f,short l,short d){
	if(!r)return l%d==0;
	if(l>=252)l%=252;W rt=0;
	char i;
	if(!f){
		if(F[r][l][M[d]])return F[r][l][M[d]]-1;
		for(i=0;i<10;i++)rt+=S(r-1,0,Tr[l][i],Px[d][i]);
		return (F[r][l][M[d]]=rt+1)-1;
	}else{
		for(i=0;i<=D[r];i++)rt+=S(r-1,i==D[r],Tr[l][i],Px[d][i]);
		return rt;
	}
}
W C(W u){
	s=0;
	for(;u;u/=10)D[++s]=u%10;
	return S(s,1,0,1);
}
int main(){
//	freopen("in.txt","r",stdin);
//	freopen("out.txt","w",stdout);
	for(i=1;i<=2520;i++)if(2520%i==0)M[i]=++L;
	for(i=0;i<253;i++)for(j=0;j<10;j++)Tr[i][j]=i*10+j;
	for(i=0;i<=2520;i++)if(M[i])for(j=0;j<10;j++)Px[i][j]=i*(j+!j)/__gcd(i,1ll*j+(!j));
	scanf("%I64d",&T);
	for(;T;T--){
		scanf("%I64d%I64d",&L,&R);
		printf("%I64d\n",C(R)-C(L-1));
	}
	return 0;
}
