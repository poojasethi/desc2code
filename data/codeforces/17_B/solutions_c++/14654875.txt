#include <iostream>

using namespace std;

long int n,pr[1001],m,temp,a,b,c,goh,cnt,all;

int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>goh;
        pr[i]=100000000;
    }
    cin>>m;
    for(int i=1;i<=m;i++){
        cin>>a>>b>>c;
        pr[b]=min(pr[b],c);
    }
    for (int i=1;i<=n;i++){
        if(pr[i]!=100000000){
            all=all+pr[i];
        }
        else{
            cnt++;
        }
    }
    if(cnt<=1){
        cout<<all;
    }
    else{
        cout<<-1;
    }
    return 0;
}
