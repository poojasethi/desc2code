#include <iostream>
#define INF 10000000
using namespace std;

int main() {
	int n;
	cin >> n;
	int useless;
	int min[1000];
	for(int i = 0; i < n; i++) {
		cin >> useless;
		min[i] = INF;
	}
	int e;
	cin >> e;
	for(int i = 0; i < e; i++) {
		int src, dest, weight;
		cin >> src >> dest >> weight;
		dest--;
		if(weight < min[dest])
			min[dest] = weight;
	}
	int total = 0, noups = 0;
	for(int i = 0; i < n; i++) {
		if(INF == min[i]) {
			noups++;
		} else {
			total += min[i];
		}
	}
	if(1 == noups) {
		cout << total << endl;
	} else {
		cout << -1 << endl;
	}
	return 0;
}
