#include <iostream>
#include <stdio.h>

#define INF 1000000200

using namespace std;

long long n,a,b,c,res,i,m;
long long ans[1000020];

main()
 {
  cin>>n;
  for (i=1;i<=n;i++)
   ans[i]=INF;
  for (i=1;i<=n;i++)
   cin>>ans[0]; 
  cin>>m;
  for (i=1;i<=m;i++)
   {
    cin>>a>>b>>c;
    ans[b]=min(ans[b],c);
   }
  for (i=1;i<=n;i++)
   res+=ans[i];
  res-=INF;
  if (res >= INF) res=-1;
  cout<<res<<endl;
 }
