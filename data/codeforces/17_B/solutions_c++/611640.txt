#include<stdio.h>
#include<algorithm>
#include<iostream>

using namespace std;

int main(void)
{
    int n,m,q,a,b,c,ans=0,cnt=0,t[1111];
    scanf("%d", &n);
    for ( int i = 0; i < n; i++)scanf("%d", &q);
    scanf("%d", &m);
    //memset(t,-1,sizeof(t));
    for ( int i = 0; i < 1111; i++)t[i] = -1;
    while ( m--){
          scanf("%d%d%d", &a,&b,&c);
          if(t[b] == -1 || c < t[b]) t[b] = c;
    }
    for ( int i = 1; i <= n; i++){
        if ( t[i] == -1) cnt++;
        else ans += t[i];
    }
    if ( cnt > 1) puts("-1");
    else printf("%d\n", ans);
    
    return 0;
    
}
