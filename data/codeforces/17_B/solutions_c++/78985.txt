#include <stdio.h>
#include <string.h>

#define M 100000000

int cost[1000];
int n;

int main()
{
	scanf("%d", &n);
	int i;
	for (i=0; i<n; i++) cost[i] = M;
	
	int a;
	for (i=0; i<n; i++) scanf("%d", &a);
	
	int m;
	int b, c;
	scanf("%d", &m);
	while (m--)
	{
		scanf("%d %d %d", &a, &b, &c);
		a--; b--;
		if (cost[b] > c)
			cost[b] = c;
	}
	
	int tot = 0;
	int nroot = 0;
	for (i=0; i<n; i++)
	{
		if (cost[i] == M) nroot++;
		else tot += cost[i];
	}
	
	if (nroot != 1) printf("-1\n");
	else printf("%d\n", tot);
}
