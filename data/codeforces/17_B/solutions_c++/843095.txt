#include<stdio.h>
#define inf 10000000
int cost[1005],q[1005];
int main()
{
	int n,i,a,b,m,c;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&q[i]);
	for(i=0;i<n;i++)
		cost[i+1]=10000000;
	scanf("%d",&m);
	while(m--)
	{
		scanf("%d%d%d",&a,&b,&c);
        if(q[a]>q[b]&&c<cost[b])cost[b]=c;
	}
	int sum=0;
	bool first=true;
	for(i=1;i<=n;i++)
		if(cost[i]==inf)
		{
			if(first)first=false;
			else{printf("-1\n");break;}
		}
		else sum+=cost[i];
	if(i>n)printf("%d\n",sum);
	return 0;
}
