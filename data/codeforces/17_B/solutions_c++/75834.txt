#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main(){
	int n;
	cin >> n;
	int q;
	for (int i = n; i > 0; i--) cin >> q;
	vector<int>a(n);
	for (int i = 0; i < n; i++) a[i] = 2000000000;
	int v1, v2, w;
	int m;
	cin >> m;
	for (int i = 0; i < m; i++){
		cin >> v1 >> v2 >> w;
		a[v2-1] = min(a[v2-1], w);
	}
    int kol = 0;
	int ans = 0;
	for (int i = 0; i < n; i++){
		if (a[i] == 2000000000) kol++;
		if (kol == 2) {
			cout << "-1" << endl;
			return 0;
		}
		if (a[i] < 2000000000) ans += a[i];
	}
	cout << ans << endl;
	return 0;
}
