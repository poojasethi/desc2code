#include <cstdio>
#include <iostream>
#include <cstring>
int p[1005];
int main ( )  {
	int n,m,i,x,a,b,c,s=0,t=0;
	scanf("%d",&n);
	for (i=1;i<=n;i++) scanf("%d",&x);
	for (i=1;i<=n;i++) p[i]=-1;
	scanf("%d",&m);
	for (i=1;i<=m;i++) {
		scanf("%d%d%d",&a,&b,&c);
		if (p[b]==-1 || c<p[b]) p[b]=c;
	}
	for (i=1;i<=n;i++) {
		if (p[i]==-1) s++; else t+=p[i];
	}
	if (s>1) printf("-1\n"); else printf("%d\n",t);
	return 0;
}
