#include <iostream>
using namespace std;
int bMin[2000];
const int INF = (1<<30);
int main()
{
   int n;
   cin>>n;
   for(int i=0;i<n;i++) {
      int v;cin>>v;
   }
   for(int i=1;i<=n;i++) {
      bMin[i]=INF;
   }
   int ret = 0;
   int m;cin>>m;
   for(int i=0;i<m;i++) {
      int a,b,c;
      cin>>a>>b>>c;
      bMin[b]=min(bMin[b],c);

   }
   int count=0;
   for(int i=1;i<=n;i++) {
      if(bMin[i]==INF) count++;
      else ret+=bMin[i];
   }
   if(count==1) cout<<ret<<endl;
   else cout<<-1<<endl;
}
