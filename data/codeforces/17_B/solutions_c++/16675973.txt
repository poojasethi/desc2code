#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#include <cstdio>
#include <cstring>
using namespace std;
int boss[1005];
int ans;
int cost[1005];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
	int a;
	cin>>a;
	cost[i]=1000005;
	}
	int m;
	cin>>m;
	for(int i=1;i<=m;i++)
	{
	int a,b,c;
	cin>>a>>b>>c;
	if(boss[b]++==0)
	ans++;
	cost[b]=min(cost[b],c);
	}
	if(n-ans>1)
	cout<<"-1";
	else{
	ans=0;
	for(int i=1;i<=n;i++)
	if(boss[i])
	ans+=cost[i];
	cout<<ans;
	}
	return 0;
}
