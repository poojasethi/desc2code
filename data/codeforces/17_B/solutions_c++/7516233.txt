#include <cstdio>
#include <iostream>
using namespace std;
int f[1003];
const int INF=~0U>>2;
int n,m;
int main()
{
	scanf("%d",&n);
	for (int i=1,x;i<=n;i++) scanf("%d",&x),f[i]=INF;
	scanf("%d",&m);
	for (int i=1,u,v,c;i<=m;i++)
	{
		scanf("%d%d%d",&u,&v,&c);
		f[v]=min(f[v],c);
	}
	for (int i=1,t=0;i<=n;i++)
		if (f[i]==INF)
		{
			t++;
			if (t>=2)
			{
				puts("-1");
				return 0;
			}
		}
	int ans=0;
	for (int i=1;i<=n;i++)
		if (f[i]!=INF) ans+=f[i];
	printf("%d\n",ans);
	return 0;
}
