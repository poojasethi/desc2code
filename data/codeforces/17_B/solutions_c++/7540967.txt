#include <iostream>
#include <stdio.h>
#include <algorithm>
#define INF 1000000000
using namespace std;
int f[1003];
int main()
{
  int k,h,g;
  cin>>k;
  for (int i=1;i<=k;i++)
  cin>>h,f[i]=INF;
  cin>>g;
  for (int i=1,a,b,c;i<=g;i++)
   cin>>a>>b>>c,f[b]=min(f[b],c);
  for (int i=1,t=0;i<=k;i++)
    if (f[i]==INF)
    {
     t++;
     if (t>=2) {cout<<"-1";return 0;}
    }
  int ans=0;
  for (int i=1;i<=k;i++)
   if (f[i]!=INF)  ans+=f[i];
   cout<<ans;
return 0;
}
