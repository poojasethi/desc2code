#include <bits/stdc++.h>
using namespace std;
int q[1111];
const int inf = (int)1e9;
int cnt[1111];
int main()
{
	int n,m,a,b,c,ans=0;
	cin>>n;
	for(int i=0;i<n;i++)cin>>q[i];
	cin>>m;
	for(int i=0;i<=n;i++)cnt[i] = inf;
	while(m--)
	{
		cin>>a>>b>>c;
		cnt[b] = min(cnt[b],c);
	}
	int tot = 0;
	for(int i=1;i<=n;i++)
	{
		if(cnt[i] == inf)tot++;
		else ans += cnt[i];
	}
	if(tot >=2)cout<<-1;
	else cout<<ans;
	return 0;
}
