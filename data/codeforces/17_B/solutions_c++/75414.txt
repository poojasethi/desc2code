#include <cstdio>
#include <vector>

using namespace std;

int main()
{
	int n, m;
	int a, b, c;
	scanf("%d", &n);
	vector<int> q(n);
	vector<int> sp(n, 99999999);

	for (int i = 0; i < n; ++i)
		scanf("%d", &q[i]);
	scanf("%d", &m);
	for (int i = 0; i < m; ++i)
	{
		scanf("%d%d%d", &a, &b, &c);
		a--; b--;
		if (q[a] > q[b])
			if (sp[b] > c) sp[b] = c;
	}

	int tt = 0;
	int sum = 0;
	for (int i = 0; i < n; ++i)
	{
		if (sp[i] == 99999999) tt++; else sum += sp[i];
	}
	if (tt == 1) printf("%d\n", sum); else printf("-1\n");
}
