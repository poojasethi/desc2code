#include<cstdio>
#include<algorithm>
#define maxn 1001
using namespace std;
int cost[maxn];
int main()
{
    __int64 inf=1<<30,ans;
    int i,a,b,c,n,m;
    scanf("%d",&n);
    for(i=1;i<=n;i++) cost[i]=inf;
    ans=-inf;
    for(i=1;i<=n;i++) scanf("%d",&a);
    scanf("%d",&m);
    while(m--)
    {
        scanf("%d%d%d",&a,&b,&c);
        cost[b]=min(cost[b],c);
    }
    for(i=1;i<=n;i++) ans+=cost[i];
    if(ans>=inf) puts("-1");
    else printf("%I64d\n",ans);
    return 0;
}
