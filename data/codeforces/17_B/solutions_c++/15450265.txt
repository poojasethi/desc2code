#include <bits/stdc++.h>

const int INF=1 << 27;
using namespace std;
int n,m,a,b,c,temp,cost[1005];
set<int> s;

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&temp);
		cost[i]=INF;
	} 
	cin >> m;
	
	for (int i=1;i<=m;i++)
	{
		scanf("%d %d %d",&a,&b,&c);
		s.insert(b);
		if (cost[b]>c) cost[b]=c;
	}
	
//	for (int i=1;i<=n;i++) cout << cost[i] << "   ";
	
	if (s.size()!=n-1) cout << -1;
	else
	{
		int res=0;
		for (int i=1;i<=n;i++) if (cost[i]!=INF) res+=cost[i];
		printf("%d",res);
	}
	
	return 0;
}
