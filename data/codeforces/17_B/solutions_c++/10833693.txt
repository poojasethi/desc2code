#include<bits/stdc++.h>
using namespace std;
bool exist[1005]={0};
long long int cost[10005];
int main()
{
    int n,x,y,z,m;  
    cin>>n;
    int i,j,k,l,xx;
    for(i=1;i<=n;i++) {
    scanf("%d",&x);cost[i]=INT_MAX;}
    cin>>m; 
    for(i=1;i<=m;i++)
    {   
        scanf("%d%d%d",&x,&y,&z);
        exist[y]=1;
        if(cost[y]>z) cost[y]=z;
    }
    int coun=0;
    for(i=1;i<=n;i++) {  coun += exist[i]; }
    if(coun<n-1){cout<<"-1";return 0;}
    long long int ans=0;
    for(i=1;i<=n;i++) ans+=cost[i];
    ans=ans-*max_element(cost+1,cost+n+1);
    cout<<ans<<endl;
    return 0;
}
