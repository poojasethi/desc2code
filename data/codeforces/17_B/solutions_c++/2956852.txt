#include<iostream>
using namespace std;

const long maxn=1000+1;
const long maxm=10000+1;
const long inf=1000000000;

long n,tt,m,f[maxn],a,b,c,ans=0,sum=0;

int main()
{
    cin>>n;
    for (int i=1;i<=n;i++)cin>>tt;
    cin>>m;
    for (int i=1;i<=n;i++)f[i]=inf;
    for (int i=1;i<=m;i++)
    {
        cin>>a>>b>>c;
        if (c<f[b])f[b]=c;
    }
    for (int i=1;i<=n;i++)
    {
        if (f[i]==inf) {sum++;continue;}
        ans+=f[i];
    }
    if (sum>=2)cout<<-1<<endl;else cout<<ans<<endl;
    //system("pause");
    return 0;
}
