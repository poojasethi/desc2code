#include <cstdio>
#include <algorithm>
using namespace std;

const int INF = 0x3f3f3f3f;
const int MAXN = 1024;

int x[MAXN];

int main() {
	int n, m;

	scanf("%d", &n);
	for (int i=0; i<n; ++i) scanf("%*d");
	fill(x, x+n, INF);
	scanf("%d", &m);
	for (int i=0,a,b,c; i<m; ++i) {
		scanf("%d%d%d", &a, &b, &c), --a, --b;
		x[b] = min(x[b], c);
	}
	int c = 0, tot = 0;
	for (int i=0; i<n; ++i) x[i] == INF ? ++c : tot += x[i];
	printf("%d\n", c == 1 ? tot : -1);

	return 0;
}
