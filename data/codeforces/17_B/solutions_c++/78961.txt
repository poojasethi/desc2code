#include <iostream>
#include <cstring>
#include <algorithm>
#include <climits>
using namespace std;
int main()
{
	int cost[1005],m,n,i,j,a,b,c;
	cin>>m;
	for(i=0;i<m;i++)
	cin>>a;
	for(i=1;i<=m;i++)
	cost[i]=INT_MAX;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a>>b>>c;
		cost[b]=min(cost[b],c);
	}
	c=a=0;
	for(i=1;i<=m;i++)
	{
		if(cost[i]==INT_MAX)
		c++;
		else
		a+=cost[i];
		if(c>1)
		{
			cout<<"-1"<<endl;
			return 0;
		}
	}
	cout<<a<<endl;
	return 0;
}
