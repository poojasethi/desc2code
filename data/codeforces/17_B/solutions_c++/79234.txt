#include <cstdio>

int main() {

	int m,n,q,a,b,c,mq,mi,i,j,s;
	
	scanf("%d ", &n);
	
	int bla[n+1];
	
	mq = 0;
	for (i=0;i<n;i++) {
		bla[i+1]=9000000;
		scanf("%d ", &q);
		if (q>mq) { mq = q; mi=i+1; }
	}
	
	scanf("%d ", &m);
	
	bla[mi] = 0;
	for (i=0;i<m;i++) {
		scanf("%d %d %d ", &a, &b, &c);
		if (bla[b]>c) bla[b]=c;
	}
	
	int fodeu=0; s=0;
	for(i=1;!fodeu && i<=n;i++) {
		if (bla[i]==9000000) fodeu=1;
		else s += bla[i];
	}
	
	if (!fodeu)
		printf("%d\n", s);
	else
		printf("-1\n");

}
