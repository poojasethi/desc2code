#include <iostream>
#include <vector>
#include <set>
using namespace std;

int main()
{
	int n;
	cin >> n;

	vector<int> q(n);
	for(int j=0; j<n; ++j) cin >> q[j];

	int m;
	cin >> m;

	vector<int> a(m), b(m), c(m);
	for(int i=0; i<m; ++i) cin >> a[i] >> b[i] >> c[i];

	vector<int> pc(n, 0x7fffffff);
	for(int i=0; i<m; ++i)
		pc[b[i]-1] = min( pc[b[i]-1], c[i] );

	int nop = 0, sum = 0;
	for(int i=0; i<n; ++i)
		if( pc[i] == 0x7fffffff )
			nop += 1;
		else
			sum += pc[i];

	if( nop >= 2 )
		cout << -1 << endl;
	else
		cout << sum << endl;
}
