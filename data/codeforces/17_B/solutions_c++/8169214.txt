#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;
int n,m,a,b,c,sum,sv[1001],x;
int main(){
  cin>>n;
  for(int i=1;i<=n ;i++){
   // scanf("%d",&m), sv[i] = -1;
    cin>>m;
    sv[i]=-1;
  }
  cin>>m;
  for(int i=0;i<m;i++){
    cin>>a>>b>>c;
    if(sv[b]==-1 ||sv[b]>c)
      sv[b]=c;
  }
  for(int i=1;i<=n;i++){
    if(sv[i]!=-1) 
        sum+=sv[i];
    else 
        x++;
  }
  if(x==1)
    cout<<sum;
  else
    cout<<"-1"<<endl;
  return 0;
}
