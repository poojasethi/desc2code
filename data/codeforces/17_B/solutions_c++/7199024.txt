#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,a,b,c,sum,sv[1001],x;
int main(){
  scanf("%d",&n);
  for(int i=1;i<=n ;i++) scanf("%d",&m), sv[i] = -1;
  scanf("%d",&m);
  for(int i=0;i<m;i++){
    scanf("%d%d%d",&a,&b,&c);
    if(sv[b] == -1 || sv[b]>c)
      sv[b] = c;
  }
  for(int i=1;i<=n;i++){
    if(sv[i] != -1) sum += sv[i];
    else x++;
  }
  if(x == 1)  printf("%d",sum);
  else printf("-1\n");
  return 0;
}
