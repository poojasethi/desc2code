#include <iostream>
#include <vector>

using namespace std;

int main()
{
  int n;
  cin>>n;
  for (int i=0;i<n;i++) {
    int q;
    cin>>q;
  }
  vector<int> v(n,-1);
  int definits=0;
  int m;
  cin>>m;
  for (int i=0;i<m;i++) {
    int a,b,c;
    cin>>a>>b>>c;
    a--;b--;
    if (v[b]==-1) {
      v[b]=c;
      definits++;
    } else if (c<v[b]) {
      v[b]=c;
    }
  }
  if (definits==n-1) {
    int cost=0;
    for (int i=0;i<n;i++) {
      if (v[i]!=-1) {
	cost+=v[i];
      }
    }
    cout<<cost<<endl;
  } else {
    cout<<-1<<endl;
  }
}
