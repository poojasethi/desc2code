#include <iostream>
#include <vector>
using namespace std;
#define INF 9999999

int main(){
  int n, q, m, a, b, c, cc[1001];
  while(cin >> n){
    for(int i=0; i<n; i++){
      cin >> q;
      cc[i] = INF;
    }
    cin >> m;
    for(int i=0; i<m; i++){
      cin >> a >> b >> c;
      if(cc[b-1]>c)
	cc[b-1] = c;
    }
    
    int cost = 0, count = 0;
    for(int i=0; i<n; i++){
      if(cc[i]==INF) count++;
      else cost += cc[i];
    }
    if(count>=2) cout << -1 << endl;
    else cout << cost << endl;
  }
  return 0;
}
