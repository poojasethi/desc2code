#include<stdio.h>
#define inf 1000000000
int main()
{
    int in[1003],n,m,i,t,x,y,c,ans;
    while(~scanf("%d",&n))
    {
        for(i=1;i<=n;i++)
            scanf("%d",&x),in[i]=inf;
        scanf("%d",&m);
        for(i=0;i<m;i++)
        {
            scanf("%d%d%d",&x,&y,&c);
            in[y]=(c<in[y] ? c:in[y]);
        }
        for(t=0,ans=0,i=1;i<=n;i++)
        {
            if(in[i]<inf)
                ans+=in[i];
            else
                t++;
        }
        if(t==1)
            printf("%d\n",ans);
        else
            printf("-1\n");
    }
    return 0;
}
