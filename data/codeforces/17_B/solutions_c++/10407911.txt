#include<bits/stdc++.h>
using namespace std;
int n,q[1005],m;

int main(void){
	int i,j;
	while(~scanf("%d",&n)){
		for(i=0;i<n;i++) scanf("%d",&q[i]),q[i]=-1;
		scanf("%d",&m);
		while(m--){
			int a,b,w;
			scanf("%d%d%d",&a,&b,&w);
			a--,b--;
			if(q[b]<0 || w<q[b]) q[b]=w;
		}
		int r=0,ans=0;
		for(i=0;i<n;i++){
			if(q[i]<0) r++;
			else ans+=q[i];
		}
		if(r>1) puts("-1");
		else printf("%d\n",ans);
	}
	return 0;
}
