#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int n,m,res=0,u,v,w,k=0,a[2000];
vector <int> V[1000];
int main()
{
	cin>>n;
	for(int i=0; i<n; i++)
		cin >> a[i];
	cin >> m;
	for(int i=0; i<m; i++)
		cin >> u >> v >> w,
		V[--v].push_back(w);
	for(int i=0; i<n; i++){
		int ans = 1<<30;
		for(int j=0; j<(int)V[i].size(); j++)
			ans = min(ans,V[i][j]);
		if(ans!=1<<30)res+=ans;
			else k++; 
	}
	if(k>1)cout<<-1; else	
		cout<<res;
	return 0;
}
