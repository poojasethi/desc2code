#include<stdio.h>
int main()
{
	int n,k,i,m,chk,ck,max,s,p,q1,c;
	int q[1010],a[1010];
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&q[i]);
		a[i]=-1;
	}

	scanf("%d",&m);
	for(i=1;i<=m;i++)
	{
		scanf("%d%d%d",&p,&q1,&c);
		if(a[q1]>c || a[q1]==-1)
		a[q1]=c;
	}

	max=-1;
	for(i=1;i<=n;i++)
	if(q[i]>max)
	{
		max=q[i];
		ck=i;
	}

	chk=0;
	for(i=1;i<=n;i++)
	if(i!=ck)
	if(a[i]==-1)
	chk=1;

	if(chk==1)
	printf("-1\n");
	else
	{
		for(i=1;i<=n;i++)
		if(i!=ck)
		s=s+a[i];

	printf("%d\n",s);
	}

return 0;
}
