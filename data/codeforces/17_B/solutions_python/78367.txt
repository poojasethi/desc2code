n = input()
raw_input()
m = input()
app = []
for i in range(m):
	app += [[int(x) for x in raw_input().split()]]

cost = [0]*n
for i in range(n):
	t = [a[2] for a in app if a[1]==i+1]
	if len(t) == 0:
		cost[i] = -1
	else:
		cost[i] = min(t)

if cost.count(-1) >= 2:
	print -1
elif cost.count(-1) == 1:
	print sum(cost)+1
else:
	print sum(cost)-max(cost)
