from collections import defaultdict

def main():
    n = int(raw_input())
    arr = map(int, raw_input().split())

    n1 = int(raw_input())
    m = defaultdict(lambda: 1e10)
    for x in xrange(n1):
        a, b, c = map(int, raw_input().split())
        m[b] = min(m[b], c)

    if len(m) != n - 1:
        print -1
    else:
        ret = 0
        for k in m:
            ret += m[k]
        print ret


if __name__ == '__main__':
    main()
