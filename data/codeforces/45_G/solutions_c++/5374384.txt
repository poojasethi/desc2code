#include<stdio.h>
#include<cmath>

using namespace std;

int N,M,Num[6010],i;
bool Judge_Prime(int X){
    if (X<2) return 0;
    for (int i=2;i<=sqrt(X);i++)
      if (X%i==0) return 0;
    return 1;
}
int main(){
    scanf("%d\n",&N);
    M=N*(N+1)/2;
    for (i=1;i<=N;i++) Num[i]=1;
    if (!Judge_Prime(M)){
       if ((M&1) && !Judge_Prime(M-2)){
          M-=3;
          Num[3]=3;
       }
       for (i=2;i<M;i++)
         if (Judge_Prime(i) && Judge_Prime(M-i)){
            Num[i]=2;
            break;
         }
    }
    for (i=1;i<=N;i++) printf("%d ",Num[i]);
    printf("\n");
    return 0;
}
