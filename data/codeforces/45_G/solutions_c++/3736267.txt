#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <ctime>
#include <algorithm>
using namespace std;
int ans[10001];
int n;
bool check(int x)
{
	if (x < 2) return false;
	for (int i = 2; i <= (int )(sqrt(x)); i ++)
		if (x % i == 0) return false;
	return true;
}
int main( )
{
	scanf("%d", &n);
	int tmp = n * (n + 1) / 2;
	if (!check(tmp))
	{
		if ((tmp & 1) && !check(tmp - 2)) tmp -= 3, ans[3] = 3;
		for (int i = 2; i < tmp; i ++)
			if (check(i) && check(tmp - i)) {ans[i] = 2; break; }
	}
	for (int i = 1; i < n; i ++)
		printf("%d ", ans[i] ? ans[i] : 1);
	printf("%d\n", ans[n] ? ans[n] : 1);
	return 0;
}
