#include <cstdio>
int c[1000000],p[1000000],cnt; bool no[19000000];
bool prime(int x)
{
	for(int j=1;p[j]<x && j<=cnt;j++)
		if(x%p[j]==0) return false;
	return true;
}
void find(int n)
{
	for(int i=2;i<=n;i++)
	{
		if(!no[i]) p[++cnt] = i;
		for(int j=1;i*p[j]<=n&&j<=cnt;j++)
		{
			no[p[j]*i] = true;
			if(i%p[j] == 0) break;
		}
	}
}
int main()
{
	//freopen("input.txt","r",stdin);
	int n,sum,i;
	scanf("%d",&n); 
	sum = (1+n)*n/2;find(n);
	if(!prime(sum))
	{
		if(sum % 2)
		{
			for(i=1;p[i]<=n && i<=cnt;i++)
				if(prime(sum-p[i])) 
				{sum -= p[i]; c[p[i]] = 1; goto label;}
			sum -= 3; c[3] = 2;
		}
		for(i=2;i<=n;i++)
			if(prime(i)&&prime(sum-i)) break;
		c[i] = 1;
	}
	label:
	for(i=1;i<=n;i++)
		printf("%d ",c[i]+1);
}
