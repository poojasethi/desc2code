#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
int m,n,a[7000];
bool prime(int x){
    if(x==1)return 0;
    for(int i=2;i*i<=x;i++)
        if(x%i==0)return 0;
    return 1;
}
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
    scanf("%d",&n);
    m=n*(n+1)/2;
    for(int i=1;i<=n;i++)a[i]=1;
    if(!prime(m)){
        if(!prime(m-2)&&(m%2==1)){m-=3;a[3]=3;}
        for(int i=2;i<=n;i++)
            if(prime(i)&&prime(m-i)){a[i]=2;break;}
    }
    for(int i=1;i<=n;i++)printf("%d ",a[i]);
    return 0;
}
