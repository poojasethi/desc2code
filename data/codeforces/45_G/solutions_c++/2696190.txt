#include <cstdio>
#include <algorithm>
#include <string.h>
#include <cstdlib>
#include <iostream>
using namespace std;
int n, a[6010];
bool ok;

bool p(int x)
{
    if (x == 1) return false;
    for (int i = 2; i * i <= x; i++)
        if ((x / i) * i == x) return false;
    return true;
}

int main()
{
    scanf("%d", &n);
    int s = n * (n+1) / 2;
    if ((s & 1) && s > 3 && !p(s-2)) a[3] = 2, s -= 3;
    for (int i = 1; i <= n; i++)
         if (p(i) && p(s - i) && !a[i])
         {
              a[i] = 1;
              break;
         }
    for (int i = 1; i <= n; i++) printf("%d ", a[i]+1);
    printf("\n");
    return 0;
}
