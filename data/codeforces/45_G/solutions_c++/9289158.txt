#include <bits/stdc++.h>
#define rep(i,j,k)	for (int i = (j); i <= (k); i++)
#define dep(i,j,k)	for (int i = (j); i >= (k); i--)
using namespace std;
const int N = 6111;
typedef long long ll;
int n, k, a[N];
bool pri(int x)
{
	if (x == 1)	return 0;
	rep(i,2,x/i)	if (!(x%i))	return 0;
	return 1;
}
int main()
{
	cin >> n;
	k = n*(n+1)/2;
	if ((k&1) && n>2 && !pri(k-2))	a[3] = 2, k -= 3;
	rep(i,2,n)	if (!a[i] && pri(i) && pri(k-i))
	{
		a[i] = 1;
		break;
	}
	rep(i,1,n)	printf("%d ", a[i]+1);
}
