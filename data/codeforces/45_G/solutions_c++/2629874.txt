#include <cstdio>

using namespace std;

const int maxn = 6005;

int ans[maxn];
int n;

bool Isprime(int x)
{
	if (x < 2) return 0;
	for (int i = 2; i * i <= x; i ++)
		if (x % i == 0) return 0;
	return 1;
}

void init()
{
	scanf("%d", &n);
}

void work()
{
	int sum = n * (n + 1) >> 1;
	for (int i = 1; i <= n; i ++) ans[i] = 1;
	if (Isprime(sum)) return;
	if ((sum & 1) && !Isprime(sum - 2))
		ans[3] = 3, sum -= 3;
	for (int i = 2; i < sum; i ++)
		if (Isprime(i) && Isprime(sum - i)){
			ans[i] = 2; break;
		}
}

void print()
{
	for (int i = 1; i <= n; i ++) printf("%d ", ans[i]);
	printf("\n");
}

int main()
{
	init();
	work();
	print();
	return 0;
}
