#include <stdio.h>
#include <memory.h>
int c[10000]={0};
bool p(int x){
    if (x==1)return false;
    for (int i=2;i*i<=x;i++)
        if (x%i==0)
            return false;
    return true;
}
int main(){
    int n;
    scanf("%d",&n);
    int sum=(n)*(n+1)/2;
    if (sum&1&&n>2&&!p(sum-2)){c[3]=2;sum-=3;}
    for (int i=0;i<=n;i++)
        if (c[i]==0&&p(i)&&p(sum-i))
        {
            c[i]=1;
            break;
        }
    for (int i=1;i<=n;i++)
        printf("%d ",c[i]+1);
        
}
