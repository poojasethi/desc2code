#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <ctime>
#include <cmath>
#define maxn 6020
#define rep(i,n) for(int i=1;i<=n;++i)
using namespace std;
int n,s;
int c[maxn];
bool get_zs(int v)
{
    for(int i=2;i*i<=v;++i)
    {
        if(!(v%i))return false;
    }
    return true;
}
int main()
{
    scanf("%d",&n);
    s=(n+1)*n/2;
    if(!get_zs(s))
    {
        if(s%2&&(!get_zs(s-2)))
        {
            s-=3;c[3]=2;
        }
        for(int i=2;i<=s;++i)
        {
            if(get_zs(i)&&get_zs(s-i))
            {
                c[i]=1;
                break;
            }
        }
    }
    rep(i,n)printf("%d ",c[i]+1);
    return 0;
}
