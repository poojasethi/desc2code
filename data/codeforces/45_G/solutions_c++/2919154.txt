#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxn = 6005;
int p[maxn], n;

bool isprime(int x) {
    for (int i = 2; i * i <= x; ++i)
        if (x % i == 0) return false;
    return true;
}

int main() {
    int i, m;
    scanf("%d", &n); m = n * (n + 1) / 2;
    for (i = 1; i <= n; ++i) p[i] = 1;
    if (!isprime(m)) {
        if ((m & 1) && !isprime(m - 2))
            m -= 3, p[3] = 3;
        for (i = 2; i < m; ++i)
            if (isprime(i) && isprime(m - i)) {
                p[i] = 2; break;
            }
    }
    for (i = 1; i <= n; ++i)
        printf("%d%c", p[i], i == n ? '\n' : ' ');
    return 0;
}
