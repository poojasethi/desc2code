#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;

int n,z,v[6005];

bool Pd(int x) {
	if(x==1) return 0;
	for(int i=2;i*i<=x;i++)
        if(x%i==0) return 0;
	return 1;
}
int main() {
	scanf("%d",&n);
	z=n*(n+1)/2;
	if((z&1) && (z>2) && !Pd(z-2)) v[3]=2,z-=3;
	for(int i=0;i<=n;i++)
        if(!v[i] && Pd(i) && Pd(z-i)) {
            v[i]=1;
            break;
        }
	for(int i=1;i<=n;i++) printf("%d ",v[i]+1);
	printf("\n");
    return 0;
}
