#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=6000+10;
int n,m,a[maxn],i;
bool prime(int x)
{
	if(x<2) return 0;
	for(int i=2;i*i<=x;i++)
		if(x%i==0) return 0;
	return 1;
}
int main()
{
	scanf("%d",&n);
	m=n*(n+1)/2;
	for(i=1;i<=n;i++) a[i]=1;
	if(!prime(m))
	{
		if((m&1)&&!prime(m-2)) m-=3,a[3]=3;
		for(i=2;i<m;i++)
			if(prime(i)&&prime(m-i)) {a[i]=2; break;}
	}
	for(i=1;i<=n;i++) 
		printf("%d ",a[i]);
	printf("\n");
}
