#include <cstdio>
int i, a[6001], s, n;
bool pr(int x) {
	int j;
	for (j = 2; j * j <= x; j++)
		if (x % j == 0)
			return 0;
	return 1;
}
int main ()
{
	scanf("%d", &n);
	s = n * (n + 1) / 2;
	for (i = 1; i <= n; i++) a[i] = 1;
	if (pr(s) == 0) {
		if (s % 2 == 0) {
			for (i = 3; i < s; i++) { 
				if (pr(i) && pr(s - i)) { 
					a[i] = 2;
					break;
				}
			}
		}
		else {
			if (pr(s - 2)) a[2] = 2;
			else {
				a[3] = 3;
				s -= 3;
				for (i = 4; i < s; i++)
					if (pr(i) && pr(s - i)) {
						a[i] = 2;
						break;
					}
			}
		}
	}
	for (i = 1; i <= n; i++)
		printf("%d ", a[i]);
}
