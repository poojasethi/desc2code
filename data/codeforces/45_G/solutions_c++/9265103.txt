#include<cstdio>
using namespace std;
int prime[20000010],n,s,p[5000010],l,i,j,a[6010];
int main()
{
    scanf("%d",&n);s=n*(n+1)/2;prime[1]=1;p[0]=1e9;
    for(i=2;i<=s;i++)
    {
        if(!prime[i])p[++l]=i;
        for(j=1;j<=l&&i*p[j]<=s&&i%p[j-1];j++)prime[i*p[j]]=1;
    }for(i=1;i<=n;i++)a[i]=1;
    if(!prime[s])s=s;else
    if(!prime[s-2])a[2]=2;else
    {
        if(s&1)for(i=n;i>=1;i--)if(!prime[i]){s-=i,a[i]=3;break;}
        for(i=2;i<s;i++)
            if((!prime[i])&&(!prime[s-i]))
                {s=i;for(i=n;i>=1;i--)if(s>=i&&a[i]==1)s-=i,a[i]=2;break;}
    }for(i=1;i<=n;i++)printf("%d ",a[i]);
}
