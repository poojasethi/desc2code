#include <cstdio>
using namespace std;

const int N = 10000;

int C[N];

int isprime (int x)
{
	if (x == 1) return 0;
	for (int i = 2; i * i <= x; ++i) if (x % i == 0) return 0;
	return 1;
}

int main ()
{
	int n, s;
	scanf("%d", &n);
	s = n * (n + 1) / 2;
	if ((s & 1) && n > 2 && !isprime(s - 2)) C[3] = 2, s -= 3;
	for (int i = 1; i <= n; ++i)
		if (!C[i] && isprime(i) && isprime(s - i)) { C[i] = 1; break; }
	for (int i = 1; i <= n; ++i)
		printf("%d ", C[i] + 1);
	puts("");
	return 0;
}
