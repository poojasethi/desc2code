#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>

bool isPrime(int a) {
	if (a < 2) return false;
	for (int i = 2; i * i <= a; i++)
		if (a % i == 0) return false;
	return true;
}

int ans[6001];
int n;

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) ans[i] = 1;
	int m = n * (n + 1) / 2;
	if (!isPrime(m)) {
		if (m % 2 == 1 && !isPrime(m - 2)) {
			ans[1] = ans[2] = 3;
			m -= 3;
		}
		for (int i = 2; i <= n; i++) if (ans[i] == 1 && isPrime(i) && isPrime(m - i)) {
			ans[i] = 2;
			break;
		}
	}
	for (int i = 1; i <= n; i++) printf("%d ", ans[i]);
	return 0;
}
