#include<stdio.h>
int n,s;
bool isprime(int x)
{
	if(x<2)return false;
	for(int i=2;i*i<=x;i++)if(x%i==0)return false;
	return true;
}
char ans[6010];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)ans[i]='1';
	s=n*(n+1)/2;
	int t=3;
	if(!isprime(s))
	{
		if(s&1)
			if(isprime(s-2))s-=2,ans[2]='2';
				else ans[3]='3',s-=3,t+=2;
		if(!isprime(s))for(int i=t;i<=s;i+=2)if(isprime(i)&&isprime(s-i)){ans[i]='2';break;}
	}
	for(int i=1;i<n;i++)putchar(ans[i]),putchar(' ');
	putchar(ans[n]),putchar('\n');
	return 0;
}
