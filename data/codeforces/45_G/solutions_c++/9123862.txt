#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n, i, A[6060];
bool Prime(int x)
{
	if (x == 1) return false;
	for (int i = 2; i * i <= x; i++)
		if (x % i == 0) return false;
	return true;
}
int main()
{
	scanf("%d", &n);
	if (n == 2) {printf("1 1"); return 0;}
	int s = n * (n + 1) / 2;
	if (s & 1 && !Prime(s - 2)) A[3] = 2, s -= 3;
	for (i = 1; i <= n; i++) 
		if (!A[i] && Prime(i) && Prime(s - i)) {A[i] = 1; break;}
	for (i = 1; i <= n; i++) printf("%d ", A[i] + 1);
	return 0;
}
