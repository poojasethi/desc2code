#include <cstdio>
#include <cmath>
int n,i,j,k,m;
bool ss(int x)
{
	for (int i=2;i<=int(sqrt(x));++i) if (x%i==0) return false;return true;
}
int main()
{
	scanf("%d",&n);m=n*(n+1)/2;
	if (ss(m))
	{
		for (i=1;i<=n;++i) printf("%d ",1);printf("\n");return 0;
	}
	for (j=2;j<=n;++j) if ((ss(j))&&(ss(m-j)))
	{
		for (i=1;i<=n;++i) if (i!=j) printf("%d ",1);else printf("%d ",2);
		printf("\n");return 0;
	}
	for (j=2;j<=n;++j) if (ss(j)) for (k=j+1;k<=n;++k) if ((ss(k))&&(ss(m-j-k)))
	{
		for (i=1;i<=n;++i)
		{
			if (i==j) {printf("%d ",2);continue;}
			if (i==k) {printf("%d ",3);continue;}
			printf("%d ",1);
		}
		printf("\n");return 0;
	}
	printf("-1\n");
}
