#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cmath>
#include<algorithm>
#include<cstring>

using namespace std;

#define N 6005
int n,ans[N];
bool prime(int x)
{
	int k=sqrt((double)(x));
	for(int i=2;i<=k;i++)
		if(x%i==0) return false;
	return true;
}
int main()
{
	scanf("%d",&n);
	int m;
	m=n*(n+1)/2;
	for(int i=1;i<=n;i++)
		ans[i]=1;
	if(prime(m))
	{
		for(int i=1;i<=n;i++)
			printf("1 ");
		printf("\n");
		exit(0);
	}
	if(m&1&&!prime(m-2))
        ans[3]=3,m-=3;
    for(int i=2;i<=n;i++)
        if(prime(i)&&prime(m-i))
        {ans[i]=2;break;}
    for(int i=1;i<=n;i++)
    	printf("%d ",ans[i]);
    printf("\n");
    return 0;
}
