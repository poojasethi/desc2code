#include<cstdio>
#define fo(i,a,b) for(i=a;i<=b;++i)
const int mn=6010;
int a[mn];
int n,s,i,t=1;
bool pr(int n)
{
    int i;
    for(i=2;i*i<=n;++i)
        if(n%i==0)return 0;
    return 1;
}
int main()
{
    scanf("%d",&n);
    fo(i,1,n)a[i]=1;
    s=n*(n+1)/2;
    if(!pr(s))
    {
        if(s%2==1)if(!pr(s-2))s-=3,a[3]=++t;
        fo(i,2,s)
            if(a[i]==1&&pr(i)&&pr(s-i))
            {
                a[i]=++t;
                break;
            }
    }
    printf("%d",a[1]);
    fo(i,2,n)printf(" %d",a[i]);
    printf("\n");
    return 0;
}
