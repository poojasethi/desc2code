#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int top,a[6001],n,m;
bool Prime(int x)
{
    int tmp=floor(sqrt(x));
    for(int i=2;i<=tmp;i++)
        if(!(x%i))
            return false;
    return true;
}
void Print()
{
    for(int i=1;i<=n;i++)
        printf("%d ",a[i]);
    exit(0);
}
int main()
{
    scanf("%d",&n);
    m=(n+1)*n/2;
    for(int i=1;i<=n;i++)
        a[i]=1;
    if(Prime(m))
        Print();
    if(m&1&&!Prime(m-2))
        a[3]=3,m-=3;
    for(int i=2;i<=n;i++)
        if(Prime(i)&&Prime(m-i))
            {a[i]=2;break;}
    Print();
}
