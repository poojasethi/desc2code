#include<stdio.h>
char ans[6010];
bool pan(int x)
{
    if(x<2)return false;
    for(int i=2;i*i<=x;i++)if(x%i==0)return false;
    return true;
}
int main()
{
    int i,n,s,t=3;
    scanf("%d",&n);
    for(i=1;i<=n;i++)ans[i]='1';
    s=n*(n+1)/2;
    if(!pan(s))
    {
        if(s&1)
            if(pan(s-2)){s-=2;ans[2]='2';}
            else{ans[3]='3';s-=3;t+=2;}
        if(!pan(s))
            for(i=t;i<=s;i+=2)if(pan(i)&&pan(s-i)){ans[i]='2';break;}
    }
    for(i=1;i<n;i++)printf("%c ",ans[i]);
    printf("%c\n",ans[n]);
    return 0;
}
