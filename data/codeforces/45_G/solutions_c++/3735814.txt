#include<cstdio>
using namespace std;
int a[6005];
int n;
int isprime(int x)
{   if (x<2)
        return 0;
    int i;
    for (i=2;i*i<=x;++i)
        if (x%i==0)
            return 0;
    return 1;
}
void work()
{   int i,m;
    m=n*(n+1)/2;
    for (i=1;i<=n;++i)
        a[i]=1;
    if (!isprime(m))
    {   if ((m&1)&&!isprime(m-2))
            m-=3,a[3]=3;
        for (i=2;i<m;++i)
            if (isprime(i)&&isprime(m-i))
            {   a[i]=2;
                break;
            }
    }
}
int main()
{   int i;
    scanf("%d",&n);
    work();
    for (i=1;i<=n;++i)
        printf("%d ",a[i]);
    printf("\n");
}
