using namespace std;
const int maxn = 6001;
static int A[maxn],n,m;
bool isprim(int x){
	for (int i=2,sx=int(sqrt(x));i<=sx;i++)
		if (x%i==0) return false;
	return true;
}
int main(){
//freopen("45G.in","r",stdin);
//	freopen("45G.out","w",stdout);
	scanf("%d",&n);
	if (n==1){
		printf("-1\n");
		return 0;
	}
	if (n==2){
		printf("1 1\n");
		return 0;
	}
	m=n*(n+1)/2;
	for (int i=1;i<=n;i++) A[i]=1;
	if (isprim(m-2))
		A[2]=2;
	else{
		if (m&1){
			A[3]=3;
			m-=3;
		}
		for (int i=2;i<=n;i++) if (isprim(i)&&isprim(m-i)) {A[i]=2;break;}
	}
	for (int i=1;i<n;i++) printf("%d ",A[i]);printf("%d\n",A[n]);
	return 0;
}
