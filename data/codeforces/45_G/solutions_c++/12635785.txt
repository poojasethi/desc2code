#include<iostream>
using namespace std;
const int maxn = 6007;
int a[maxn];
bool prime(int x){
	if (x == 1)return false;
	for (int i = 2; i*i <= x; i++)if (x%i == 0)return false;
	return true;
}
int main(){
	//freopen("in.txt", "r", stdin);
	int n; cin >> n;
	for (int i = 1; i <= n; i++)a[i] = 1;
	int s = (n + 1)*n / 2;
	if (!prime(s)){
		if (prime(s - 2))a[2] = 2;
		else{ 
			if (s & 1){
				a[3] = 3;
				s -= 3;
			}
			for (int i = 3; i < s; i++){
				if (prime(i) && prime(s - i)){
					//cout << i << " " << s - i << endl;
					for (int j = n; j>0; j--){
						if (a[j] == 1 && i >= j){
							a[j] = 2;
							i -= j;
						}
					}
					break;
				}
			}
		}
	}
	cout << a[1];
	for (int i = 2; i <= n; i++)cout << " " << a[i];
	return 0;
}
