
// School Team Contest #3 (Winter Computer School 2010/11) G Prime Problem

#include <cstring>
#include <cstdio>

using namespace std;

int color[6000];

int isPrime(int x) {
	for (int i = 2; i * i <= x; i ++)
		if (x % i == 0) return false;
	return true;
}

int main() {
	int n;
	scanf("%d", &n);
	int t = n;
	n = (n + 1) * n / 2;
	
	for (int i = 0; i < t; i ++) color[i] = 1;
	if (! isPrime(n)) {
		if (n % 2 == 1 && ! isPrime(n - 2)) {
			color[2] = 3;
			n -= 3;
		}
		for (int i = 2; i <= t; i ++)
			if (isPrime(i) && isPrime(n - i) && color[i - 1] == 1) {
				color[i - 1] = 2;
				break;
			}
	}
	
	for (int i = 0; i < t; i ++)
		if (i < t - 1) printf("%d ", color[i]); else printf("%d\n", color[i]);
	
	return 0;
}
