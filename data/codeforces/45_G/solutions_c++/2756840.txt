#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
int n,i,j,k,xx,tot,p[10000];
bool ha(int x)
{
	if (x==1) return false;
	for (int i=2;i<=int(sqrt(x));i++) 
	if (x%i==0) return false;
	return true;
}
int main()
{
	//freopen("1.in","r",stdin);
	//freopen("1.out","w",stdout);
	scanf("%d",&n);
	xx=(n+1)*n/2;
	if (n==1)
	{
		printf("-1");
		return 0;
	}
	if (n==2)
	{
		printf("1 1");
		return 0;
	}
	for (i=1;i<=n;i++)
	if ((ha(i))&&(ha(xx-i)))
	{
		for(j=1;j<=n;j++)
		if (j!=i) printf("1 ");
		else printf("2 ");
		return 0;
	}
	for (i=1;i<=n;i++)
		for(j=i+1;j<=n;j++)
		if ((ha(i))&&(ha(j))&&(ha(xx-i-j)))
		{
			for(k=1;k<=n;k++)
			if (k==i) printf("2 ");
			else
			if (k==j) printf("3 ");
			else printf("1 ");
			return 0;
		}
}
