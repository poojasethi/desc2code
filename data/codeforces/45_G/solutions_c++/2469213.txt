#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxn = 10001;

int n,s;
int A[maxn];
int num;

int prime(int x)
{
    if (x == 1) return 0;
    for (int i = 2; i*i <= x; ++i)
        if (x % i == 0) return 0;
    return 1;
}

int main()
{
    int i;

    scanf("%d",&n);
    s = n*(n+1)/2;

    for (i = 1; i <= n; ++i) A[i] = 1;
    num = 1;

    if (!prime(s))
    {
        if ((s & 1) && !prime(s-2)) s -= 3,A[3] = 3;
        for (i = 2; i <= n; ++i) 
            if (prime(i) && prime(s-i))
            {
                A[i] = 2;
                break;
            }
    }
    for (i = 1; i <= n; ++i) printf("%d ",A[i]);

    return 0;
}
