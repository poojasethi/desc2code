#include<cstdio>
int n,sum,ans[6666];

bool Prime(int x)
{
    if (x==2)
        return true;
    for (int i=2; i*i<=x; i++)
        if (x%i==0)
            return false;
    return true;
}

int main()
{
    scanf("%d",&n);
    for (int i=1; i<=n; i++)
        ans[i]=1;
    sum=n*(n+1)/2;
    if (!Prime(sum))
    {
        if (sum&1 && !Prime(sum-2))
            sum-=3,ans[3]=3;
        for (int i=2; i<=n; i++)
            if (Prime(i) && Prime(sum-i))
            {
                ans[i]=2; break;
            }
    }
    for (int i=1; i<=n; i++)
        printf("%d ",ans[i]);
    return 0;
}
