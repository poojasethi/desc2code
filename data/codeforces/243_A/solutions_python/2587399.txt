
def main():
    n = input()
    a = map(int, raw_input().split())
    b = [0]*1050000
    s = set()
    t = set()
    for i in xrange(n):
        t = set()
        for x in s:
            t.add(x|a[i])
        t.add(a[i])
        s = t
        for x in s:
            b[x] = 1
    ans = 0
    for i in xrange(1<<20):
        ans += b[i]
    print ans

if __name__ == '__main__':
    main()
