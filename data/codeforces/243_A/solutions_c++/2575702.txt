#include <iostream>
#define G 2000000
int a,i,j,b,c,f[G],t[2],q[2][G];
int main (){
    std::cin>>i;
    do{
        std::cin>>a;
        c=i&1;t[!c]=1;
        f[a]=i;q[!c][0]=a;
        for(j=0;j<t[c];){
            b=a|q[c][j++];
            if(f[b]!=i)q[!c][t[!c]++]=b;
            f[b]=i;
        }
    }while(--i);
    for(j=G;j>0;)if(f[--j])++i;
    std::cout<<i;
}
