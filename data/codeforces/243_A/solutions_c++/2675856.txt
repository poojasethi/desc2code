#include <iostream>
#include <set>
using namespace std;

set<int> all;

int main()
{
  int n;
  cin>>n;
  set<int> l;
  while(n--)
    {
      int x; cin>>x;
      set<int> c;
      c.insert(x);
      for(set<int>::iterator i=l.begin();i!=l.end();++i)
	c.insert(*i|x);
      l=c;
      all.insert(c.begin(),c.end());
    }
  cout<<all.size()<<endl;
  return 0;
}
