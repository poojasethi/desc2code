#include<iostream>
#include<set>
using namespace std;
set<int> all;
int main()
{
	set<int> l;
	int n;
	cin>>n;
	while (n--)
	{
		set<int> c;
		int x;
		cin>>x;
		c.insert(x);
		for (set<int>::iterator i=l.begin();i!=l.end();i++)
		c.insert(*i|x);
		l=c;
		all.insert(c.begin(),c.end());
	}
	cout<<all.size()<<endl;
	return 0;
}
