using namespace std;

#include <iostream>
#include <set>

int main () {
	int n, a[100100];
	set <int> ans;

	cin >> n;

	for (int i = 0; i < n; ++i) {
		cin >> a[i];
		ans.insert (a[i]);
		int k = 0;
		for (int j = i - 1; j >= 0; --j) {
			k |= a[j];
			if (!(a[i] & ~k)) break;
			ans.insert (a[i] | k);
		}
	}

	cout << ans.size () << endl;
}
