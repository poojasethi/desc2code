#include<iostream>
#include<set>
using namespace std;
set<int> t,ans;
set<int> :: iterator it;
int main()
{
	int n, x;
	cin>>n;
	while(n--)
	{
		set<int> c;
		cin>>x;
		c.insert(x);
		for(it=t.begin() ; it!=t.end() ; it++ )
		{
			c.insert(*it|x);
		}
		t = c;
		ans.insert(t.begin(),t.end());
	}
	cout<<ans.size()<<endl;
}
