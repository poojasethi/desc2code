#include<iostream>
#include<set>
using namespace std;
set<int> ans;
int main()
{
	set<int> l;
	int n;
	cin>>n;
	while (n--)
	{
		set<int> c;
		int x;
		cin>>x;
		c.insert(x);
		for(set<int>::iterator it=l.begin();it!=l.end();it++)
			c.insert(x | *it);
		l=c;
		ans.insert(c.begin(),c.end());
	}
	cout<<ans.size()<<endl;
	//system("pause");
	return 0;
}
