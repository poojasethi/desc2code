#include <iostream>
using namespace std;
int set [1<<20];

int main ()
{int n,a[100100];
 cin >> n;
 for (int i=0;i<n;i++)
 {cin>>a[i]; set[a[i]]=1;
  int k=0;
  for (int j=i-1;j>=0;j--)
   {k|=a[j];
   if ((k|a[i])==k) break;
    set [a[i]|k]=1;
   }
}

 int cnt = 0;
 for (int i=0; i<(1<<20);i++) if (set [i]) ++cnt;
 cout<<cnt<<endl;
}
