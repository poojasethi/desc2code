#include<iostream>
#include<set>
using namespace std;

int n,x,y,a[555555];
set <int> p;

int main()
{
    while(cin>>n)
	{
		p.clear();
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
			x=a[i];
			y=0;
			p.insert(a[i]);
			for(int j=i-1;j>=0;j--)
			{
				x=x|a[j];
				p.insert(x);
				y=y|a[j];
				if(x==y) break;
			}
		}
		int num=p.size();
		cout<<num<<endl;
	}
	return 0;
}
