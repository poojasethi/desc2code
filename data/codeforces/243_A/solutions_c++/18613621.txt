#include<bits/stdc++.h>
using namespace std;

int arr[100005];
set<int> s;
int main()
{
	int x,i,j,k,m,n,t;
	cin>>n;
	for(i=0;i<n;i++)
	{	cin>>arr[i];
		s.insert(arr[i]);
		for(j=i-1;j>=0;j--)
		{
			x=arr[i]|arr[j];
			if(x==arr[j])break;
			arr[j]=x;
			s.insert(arr[j]);
		}
	}
	cout<<s.size();	
}
