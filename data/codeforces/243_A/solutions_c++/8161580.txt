#include<bits/stdc++.h>
using namespace std;
int n, a;
set<int> ans, aux, s;
set<int> :: iterator it;

int main(){
	cin>>n;
	
	for(int i = 0; i < n; i++){
		aux.clear();
		cin>>a;
		aux.insert(a);
		
		for(it = s.begin(); it != s.end(); it++)
			aux.insert(a | *it);
		
		s = aux;
		ans.insert(s.begin(), s.end());
	}
	cout<<ans.size()<<endl;
	
	return 0;
}
