#include <cstdio>

using namespace std;

int f[1<<20],a[100001];

int main(void)
{
  int n,ans=0;
  scanf("%d",&n);
  for(int i=1;i<=n;++i)
  {
    scanf("%d",&a[i]);
    ans+=1-f[a[i]];
    f[a[i]]=1;
    int k=0;
    for(int j=i-1;j>0;--j)
    {
      k|=a[j];
      if((k|a[i])==k)
        break;
      ans+=1-f[k|a[i]];
      f[k|a[i]]=1;
    }
  }
  printf("%d\n",ans);
}
