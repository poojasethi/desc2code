#include<iostream>
#include<set>
using namespace std;
int x,n;
set<int> l,all;
int main()
{
    cin>>n;
    while(n--)
    {
        cin>>x;
        set<int> now;
        now.insert(x);
        for(set<int>::iterator it=l.begin();it!=l.end();++it)
        now.insert(*it|x);
        l=now;
        all.insert(now.begin(),now.end());
    }
    cout<<all.size()<<endl;
    return 0;
}
