#include <iostream>

using namespace std;

int n,q[1<<20];
bool c[1<<20]={false};
int main(void) {
  cin >> n;
  for (int i = 0,t=0,a; i < n && cin >> a; ) {
    c[a]=true;
    for (int v = 0, j = t; j-->0&&(a&~(v|=q[j]));c[a|v]=true);
    while(t>0&&a==q[t-1])--t;
    q[t++]=a;
  }
  int s=0;
  for(int i=1<<20;i-->0;)if(c[i])++s;
  cout<<s<<endl;
  return 0;
}
