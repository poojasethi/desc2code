#include<iostream>
#include<set>
using namespace std;
int n;
set < int > ans;
set < int > st;
int main()
{
	cin>>n;
	while(n--)
	{
		int tmp;
		cin>>tmp;
		set < int > use;
		use.insert(tmp);
		for(set<int>::iterator i=st.begin();i!=st.end();i++)
		{
			use.insert(*i|tmp);
		}
		st=use;
		ans.insert(use.begin(),use.end());
	}
	cout<<ans.size()<<endl;
	return 0;
}
