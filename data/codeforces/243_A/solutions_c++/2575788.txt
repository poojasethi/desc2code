#include <iostream>
#define G 1100000
int a,i,j,b,c,f[G],t[2],q[2][G];
int main (){
    std::cin>>i;
    do{
        std::cin>>a;
        c=i&1;t[c]=1;
        f[a]=i;q[c][j=0]=a;
        while(j<t[!c]){
            b=a|q[!c][j++];
            if(f[b]!=i)q[c][t[c]++]=b;
            f[b]=i;
        }
    }while(--i);
    for(j=G;j;)i+=!!f[--j];
    std::cout<<i;
}
