#include <iostream>
#include <set>
using namespace std;
const int maxn = 1000 * 100 + 10;
int n,a[maxn],k;
set<int>s;
int main () { cin >> n; for(int i = 0 ; i < n ; i++) { cin >> a[i]; s.insert(a[i]); k = 0; for(int j = i-1 ; j >= 0 ; j--) { k |= a[j]; if((a[i] & ~k) == 0) break; s.insert(a[i]|k); } } cout << s.size() << endl; return 0;
}
