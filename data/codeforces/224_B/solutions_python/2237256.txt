n, k = map(int, raw_input().split())
a = map(int,raw_input().split())
d = {}
if len(set(a)) < k: print -1,-1
for i in xrange(n):
    if a[i] not in d: d[a[i]] = 1
    else: d[a[i]] += 1
    if len(d) == k:
        for j in xrange(i+1):
            d[a[j]]-=1
            if d[a[j]] == 0:
                print j+1,i+1
                break
        break
