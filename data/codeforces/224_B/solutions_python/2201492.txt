n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
cnt = {}
for i in xrange(len(a)):
    this_ele = a[i]
    if this_ele not in cnt:
        cnt[this_ele] = 1
    else:
        cnt[this_ele] += 1
    if len(cnt) == k:
        for j in xrange(i + 1):
            cnt[a[j]] -= 1
            if cnt[a[j]] == 0: 
                print j + 1, i + 1
                break
        break
else: print -1, -1
