[n,k] = map(int, raw_input().split())
a = map(int, raw_input().split())
d = dict()
q = 0
i = 0
while q < k and i < n:
    if a[i] not in d:
        d[a[i]] = 1
        q += 1
    else:
        d[a[i]] += 1
    i += 1
if q < k:
    print "-1 -1\n"
    exit(0)
j = 0
while a[j] in d and d[a[j]] > 1:
    d[a[j]] -= 1
    j += 1
print str(j + 1) + ' ' + str(i) + '\n'
