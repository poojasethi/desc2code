n,k = map(int, raw_input().split())
array = map(int, raw_input().split())

if len(set(array)) < k:
    print -1, -1
else:
    diff = [0]*100008
    i = 0;
    count = 0;
    while count < k:
        if diff[array[i]] == 0:
            count += 1
        diff[array[i]] += 1
        i += 1
    j = 0
    while diff[array[j]] > 1:
        diff[array[j]] -= 1
        j += 1

    print j+1, i
