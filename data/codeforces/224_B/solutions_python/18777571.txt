r = lambda:map(int,raw_input().split())
n,k = r(); a = r()
c = [0]*(max(a)+1)
s = range(n)
dif = 0
for x in a:
    dif += (c[x] == 0)
    c[x] += 1
i = 0
j = n-1
l = -2
r = -2
while ((dif >= k) and (i <= j)):
    if (dif == k):
        l = i
        r = j
    if (c[a[i]] >= c[a[j]]):
        c[a[i]] -= 1
        dif -= (c[a[i]] <= 0)
        i += 1
    else:
        c[a[j]] -= 1
        dif -= (c[a[i]] <= 0)
        j -= 1
print l+1,r+1
