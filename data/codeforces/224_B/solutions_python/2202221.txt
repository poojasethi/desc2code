def main():
    n, k = map(int, raw_input().split())
    ar = map(int, raw_input().split())

    i = 0
    data = {}
    while i < n and len(data) < k:
        data[ar[i]] = data.get(ar[i], 0) + 1
        i += 1

    if len(data) != k:
        return "-1 -1"

    j = 0
    while len(data) == k:
        data[ar[j]] = data[ar[j]] - 1
        if not data[ar[j]]:
            data.pop(ar[j])
        j += 1

    return "%s %s" % (j, i)


if __name__ == '__main__':
    print main()
