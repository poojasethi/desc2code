#!/usr/bin/env python
a,b = map(int,raw_input().split())
l = map(int,raw_input().split())
L = 0
R = a-1
d = 0
t = [0 for i in xrange(0,100000)]
for i in l:
    if t[i-1] is 0:
            d += 1
    t[i-1] += 1
if d < b:
    print -1, -1
    exit()
change = True
while L < R and change:
    change = False
    while t[l[R]-1] > 1 or d > b:
            t[l[R]-1] -= 1
            if t[l[R]-1] == 0:
                d -= 1
            R -= 1
            change = True
    while t[l[L]-1] > 1 or d > b:
            t[l[L]-1] -= 1
            if t[l[L]-1] == 0:
                 d -= 1
            L += 1
            change = True
print L+1, R+1
