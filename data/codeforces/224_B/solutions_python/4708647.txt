r=lambda:map(int,raw_input().split())
n,k=r()
e=r()
l=-1
for i in xrange(n):
    if i==n-1 or e[i] != e[i+1]:
        l=i
        break
S=set()
r=-1
for i in xrange(l,n):
    S.add(e[i])
    if len(S) == k:
        r = i
        break
S=set()
for i in xrange(r,l-1,-1):
    S.add(e[i])
    if len(S) == k:
        l=i
        break
if r ==-1:
    print -1, -1
else:
    print l+1,r+1
