n, k = map(int, raw_input().split())
a = map(int, raw_input().split())

cnt = dict()
for x in a:
	if x in cnt:
		cnt[x] += 1
	else:
		cnt[x] = 1
if len(cnt) < k:
	print '-1 -1'
	exit(0)

L, R = 0, n - 1
while len(cnt) >= k:
	cnt[a[L]] -= 1
	if cnt[a[L]] == 0:
		cnt.pop(a[L])
	L += 1
L -= 1
if a[L] in cnt:
	cnt[a[L]] += 1
else:
	cnt[a[L]] = 1
while len(cnt) >= k:
	cnt[a[R]] -= 1
	if cnt[a[R]] == 0:
		cnt.pop(a[R])
	R -= 1
R += 1
print L + 1, R + 1
