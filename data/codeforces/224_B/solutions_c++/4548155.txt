#include<iostream>
using namespace std;
int  main()
{int n,m,d[100002]={},k,kol=0; cin>>n>>m;
for(int i=1;i<=n;i++)
{cin>>k;
 if(d[k]==0) kol++;
 d[k]=i;
 if(kol==m)
 {int g=0;
  for(int j=1;j<100002;j++)
   if(d[j]>0 && (g==0 || d[j]<g)) g=d[j];
 cout<<g<<" " <<i;return 0;
 }
}
cout<<-1<< " "<<-1;
return 0;
}
