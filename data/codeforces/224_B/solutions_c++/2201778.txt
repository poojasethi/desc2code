#include<cstdio>
#include<algorithm>
#define N 100010
using namespace std;
int n, k, a[N], ap[N], cnt, len, x = -2, y = -1;
int main(){
	scanf("%d%d", &n, &k);
	for(int i = 0; i < n; i++) scanf("%d", &a[i]);
	for(int i = 0, j = 0; i < n; ){
		
		while(cnt < k && j < n){
			ap[a[j]]++;
			if(ap[a[j]] == 1) cnt++;
			j++;
		}

		if(cnt == k) len = j-i+1, x = i, y = j;
		
		ap[a[i]]--;
		if(ap[a[i]] == 0) cnt--;
		i++;
	}
	printf("%d %d\n", x+1, y);
}
