#include<iostream>
using namespace std;
int n,k,a[100001],b[100001],i,count,x=1;
int main()
 {
  cin>>n>>k;
  for(i=1;i<=n;i++)
   {
   cin>>a[i];
   if(!b[a[i]])
   count++;
   b[a[i]]++;
   if(count==k)
    {
    while(count==k)
      {
       b[a[x]]--;
       if(!b[a[x]])
       count--;
       x++;
      }



      cout<<x-1<<" "<<i;
      return 0;

      }
   }
   cout<<"-1"<<" "<<"-1";
   return 0;
 }
