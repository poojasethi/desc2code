#include <cstdio>

const int N = 100005;

int a[N], f[N];

int main()
{
  int n, k, num = 0;
  scanf("%d%d", &n, &k);
  for (int i = 0; i < n; ++i)
    scanf("%d", &a[i]);

  for (int i = 0; i < n; ++i)
    if (f[a[i]]++ == 0 && ++num == k)
    {
      int j = 0;
      while (true)
	if (--f[a[j++]] == 0) break;
      printf("%d %d\n", j, i + 1);
      return 0;
    }

  printf("-1 -1\n");
  return 0;
}
