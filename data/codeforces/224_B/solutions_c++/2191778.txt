#include <stdio.h>

int b[100100];
int main()
{
int a[100100];

int n,k,i,c=0,l,r;
 scanf("%d%d",&n,&k);    
 for(i=1;i<=n;i++)
 {
 scanf("%d",&a[i]);
 if(b[a[i]]==0)
  c++;
 b[a[i]]++;                
 if(c==k) {
   r=i;
   break;
   }               
 }
 if(c<k) printf("-1 -1");
 else{
   for(i=1;i<r;i++){
     b[a[i]]--;
     if(b[a[i]]==0) break;                
   }     
 printf("%d %d",i,r);
 }
 
}
