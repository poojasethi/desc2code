#include <stdio.h>
#define F(I,B,E) for(int I=B; I<E; I++)
main() {
  int n,k,a[100100],q[100100]={};
  scanf("%d%d",&n,&k);
  F(i,0,n)
    scanf("%d",a+i),
    q[a[i]]++;
  int d=0,l=0,r=n;
  while(a[--r]==a[n-1]);
  F(i,0,100100)
    d+=q[i]>0;
  if(k>d)
    puts("-1 -1");
  else {
    while(k<=d)
      d-=--q[a[l++]]==0;
    printf("%d %d",l,r+2);
  }
  return 0;
}
