#include <cstdio>

using namespace std;

int a[100005],d[100005];

int main() {
	int n, k, cnt = 0, x, l, r;
	scanf("%d%d", &n, &k);

	for(int i = 1; i <= n; ++ i) {
		scanf("%d", &x);
		a[i] = x;
		if(!d[x])
			++ cnt;
		++ d[x];
		
		if(cnt == k) {
			r = i;
			for(l = 1; l <= r && d[a[l]] > 1; ++ l)
				-- d[a[l]];
			printf("%d %d\n", l, r);
			return 0;
		}
	}

	printf("-1 -1\n");

	return 0;
}
