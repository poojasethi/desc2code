#include<iostream>
using namespace std;
int a[100001]={0};
int b[100001];
int main()
{
    int l=0,r,ans=0,n,i,k;
    cin>>n>>k;
    for(i=0;i<n;i++){
       cin>>b[i];
       if(!a[b[i]])ans++;
       a[b[i]]++;
       if(ans==k){
          r=i;
          break;
       }
    }
    if(ans==k){
       while(l<r&&a[b[l]]>1){a[b[l]]--;l++;}
       cout<<l+1<<" "<<r+1;
    }
    else cout<<"-1 -1";
    return 0;
}
