#include <cstdio>
#include <cstring>

using namespace std;

int n,m;
int a[100010];
int cnt[100010];

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	int l,r=0,sum=0;
	for(r=1;r<=n;r++){
		if(!(cnt[a[r]]++)) sum++;
		if(sum==m) break;
	}
	if(r>n) {puts("-1 -1");return 0;}
	for(l=1;l<=r;l++){
		if(!(--cnt[a[l]])) sum--;
		if(sum<m) break;
	}
	printf("%d %d\n",l,r);
	return 0;
}
