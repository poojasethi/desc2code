#include<cstdio>

int n, k, tot, l, r;
int a[111111];
int f[111111];

int main() {
	scanf("%d%d", &n, &k);
	for(int i=0;i<n;i++) scanf("%d", &a[i]);
	tot=l=r=0;
	while(r<n&&tot<k) {
		if(!f[a[r]]) tot++;
		f[a[r]]++;
		if(tot<k)r++;
	}
	if(r==n) {
		printf("-1 -1\n");
		return 0;
	}
	while(l<r&&f[a[l]]>1) {
		f[a[l]] --; l++;
	}
	printf("%d %d\n", l+1, r+1);
	
	return 0;
}
