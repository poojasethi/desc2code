#include <iostream>
using namespace std;

const int MAXN=1000*100+10;
int a[MAXN], flag[MAXN];  

int main(){
	int n, k, ans=0, l, r;
	cin>>n>>k;
	for (int i=1; i<=n; i++) 
		cin>>a[i];
	for (r=1; r<=n; r++){
		if (!flag[a[r]])
			ans++;
		flag[a[r]]++;
		if (ans==k)
			break;
	}
	for (l=1; l<=r && flag[a[l]]>1; l++)
		flag[a[l]]--;
	if (ans<k)
		l=r=-1;
	cout<<l<<" "<<r<<endl;
	return 0;
}
