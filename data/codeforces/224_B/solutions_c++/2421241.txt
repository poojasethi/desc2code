#include <iostream>

using namespace  std;

int arr1[100001],arr2[100001];

int main()
{
	int n,k;
	int l,r;
	cin>>n>>k;
	for(int i=0;i<n;i++)
		cin>>arr1[i];
	int cnt=0;
	for(r=0;r<n;r++)
	{
		if(!arr2[arr1[r]]) cnt++;
		arr2[arr1[r]]++;
		if(cnt==k) break;
	}	
	for(l=0;l<r && arr2[arr1[l]]>1;l++)
	{
		arr2[arr1[l]]--;
	}
	if(cnt<k) {l=-2;r=-2;}
	cout<<l+1<<" "<<r+1<<endl;
	return 0;
}
