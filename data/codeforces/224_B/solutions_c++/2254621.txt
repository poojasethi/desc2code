#include<cstdio>
#define N 100100

int a[N],u[N] = {0};

int main(){
    int n,k,l,r,sum,i;
    for (scanf("%d%d",&n,&k),i = 1; i <= n; ++i) scanf("%d",&a[i]);
    for (r = 1,sum = 0; r<=n && sum<k; ++r) {sum += (!u[a[r]]); ++u[a[r]];}
    if (sum < k){  printf("-1 -1\n");  return 0;  }
    for (l = 1; u[a[l]]>1; ++l) --u[a[l]];
    printf("%d %d\n",l,r-1);
    return 0;
}
