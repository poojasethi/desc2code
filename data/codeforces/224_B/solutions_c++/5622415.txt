#include <iostream>
using namespace std;

int b[100001];
int main()
{
int i,j,n,k,K,c=0;
cin>>n>>k;
int a[n];
for(i=0;i<n;i++) {cin>>a[i];b[a[i]]++;}
for(i=0;i<=100000;i++) if(b[i]) c++;
if(c<k) cout<<-1<<" "<<-1;
else
{
i=0;j=n-1;
while(1)
{
for(;(j>=i)&&(c>=k);j--)
{
  b[a[j]]--;K=j;
 if(!b[a[j]]) c--;
}
j=K;c++;b[a[j]]=1;
for(;(c>=k)&&(i<=j);i++)
{
 b[a[i]]--;K=i;
if(!b[a[i]]) c--;
}
i=K; c++;b[a[i]]=1;
if(c==k) break;
}
cout<<(i+1)<<" "<<(j+1);
}
return 0;
}
