#include <stdio.h>
#define N 100005
int a[N], used[N], n, k, i, l, r, cnt = 0;
int main()
{
	scanf("%d %d", &n, &k);
	for(i = 1; i <= n; i++)
		scanf("%d", &a[i]);
	for(r = 1; r <= n; r++)
	{
		if(!used[a[r]])
			cnt++;
		used[a[r]]++;
		if(cnt == k)
			break;
	}
	for(l = 1; l <= r && used[a[l]] > 1; l++)
			used[a[l]]--;
	if(cnt < k)
		l = r = -1;
	printf("%d %d\n", l, r);	
	return 0;
}
