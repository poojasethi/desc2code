#include <iostream>
#include <set>
using namespace std;

int main()
{int n,k,a[100005],kol[100005]={0}; cin>>n>>k;
int s=0;
int j=0;
for (int i=1;i<=n;i++)
 {cin>>a[i];
  kol[a[i]]++;
  if (kol[a[i]]==1) s++;
  if (s==k)
  {int j=1;
    while (kol[a[j]]>1) {kol[a[j]]--;j++;}
    cout<<j<<" "<<i; return 0;
  }
 }
cout<<"-1 -1";
return 0;
}
