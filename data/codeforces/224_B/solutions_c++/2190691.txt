#include <iostream>
#define MAXN 100005
using namespace std;

int n,k,v[MAXN],cnt,uz[MAXN];

int main()
{
    int i,j;
    cin>>n>>k;
    for(i=1;i<=n;i++)
        cin>>v[i];
    for(i=1;i<=n&&cnt<k;i++){
        if(!uz[v[i]])
            cnt++;
        uz[v[i]]++;}
    if(cnt!=k){
        cout<<"-1 -1\n";
        return 0;}
    uz[v[1]]--;
    for(j=2;uz[v[j-1]];j++)
        uz[v[j]]--;
    cout<<j-1<<' '<<i-1<<'\n';
    return 0;
}
