#include<iostream>
using namespace std;
int a[100001],b[100001]={0};
int main()
{
  int i,n,k,l=0,r=0,ans=0;
  cin>>n>>k;
  for(i=0;i<n;i++) cin>>a[i];
  for(i=0;i<n;i++)
  {
    if (!b[a[i]]) ans++;
    b[a[i]]++;
    if (ans==k) {r=i; break;}
  }
  if (k==ans)
  {
    while(b[a[l]]>1){b[a[l]]--;  l++;}
    cout<<l+1<<" "<<r+1;
  }else cout<<"-1 -1";
//  system("pause");
  return 0;
}
