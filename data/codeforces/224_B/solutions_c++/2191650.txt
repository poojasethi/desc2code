#include <iostream>
#define maxn 100000
using namespace std;

int a[maxn];
int vals[maxn + 1];
int main(){
	int n,k,l,r;
	cin>>n>>k;
	for(int i=0;i<n;i++) cin>>a[i];
	
	int kdiff=0;
	for(r=0;r<n && kdiff < k;r++){
		if(vals[a[r]] == 0) kdiff++;
		vals[a[r]]++;
	}
	
	if(kdiff < k) cout<<"-1 -1\n";
	else{
		for(l=0;l<r;l++)
			if(vals[a[l]] == 1) break;
			else vals[a[l]]--;
		cout<<l + 1<<" "<<r<<"\n";	
	}
	return 0;
}
