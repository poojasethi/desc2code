#include <bits/stdc++.h>
using namespace std;
int hash[100005],a[100005];
int main() {
	int n,k,cnt=0,j;
	cin>>n>>k;
	for(int i=1;i<=n;i++){
	    cin>>a[i];
	    if(hash[a[i]]==0)cnt++;
	    hash[a[i]]++;
	    if(cnt==k)
	    {
	     for(j=1;j<=i && hash[a[j]]>1;j++){
	         hash[a[j]]--;
	      }
	      cout<<j<<" "<<i;
	      return 0;
	    }
	}
	cout<<-1<<" "<<-1;
	return 0;
}
