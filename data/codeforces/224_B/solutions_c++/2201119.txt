#include<iostream>
#include<cstdio>
using namespace std;
long n,k,aw=1,ak,x[110000],bol[110000],ctr=0;
bool ketemu=false;


main(){
	cin>>n>>k;
	for (int i=1;i<=n;i++){
		scanf("%d",&x[i]);
		if (bol[x[i]]==0) ctr++;
		bol[x[i]]++;
		if (ctr==k){ketemu=true; ak=i; break;}
	}
	while (bol[x[aw]]>1){
		bol[x[aw]]--;
		aw++;
	}
	if (ketemu) printf("%d %d\n",aw,ak);
	else printf("-1 -1");
}
