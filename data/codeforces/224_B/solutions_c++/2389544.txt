#include <stdio.h>

int a[100500];
int n,k;
int rst=999999,r,l;
int ck[100005];

int main(){
int i,j;
r=l=-1;
scanf("%d%d",&n,&k);
for(i=1;i<=n;i++)scanf("%d",&a[i]);
int p=1;
int cnt=0;
for(i=1;i<=n;i++){
if(ck[a[i]]==0)cnt++;
ck[a[i]]++;

while(cnt>=k){
ck[a[p]]--;
if(ck[a[p]]==0)cnt--;
if(cnt==k-1){
ck[a[p]]++;
cnt++;
break;
}
p++;
}
if(rst>i-p+1 && cnt==k){
rst=i-p+1;
r=p;
l=i;
}

}
printf("%d %d",r,l);
}
