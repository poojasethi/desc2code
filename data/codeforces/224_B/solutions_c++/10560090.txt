#include <iostream>
#include <set>
using namespace std;
int n,k,a[100005],b[100005],s,m,i;
int main()
{
cin>>n>>k;

for (i=1;i<=n;i++)
{
 cin>>a[i];
  b[a[i]]++;
  if (b[a[i]]==1) s++;
  if (s==k)
  {
    m=1;
    while (b[a[m]]>1)
 {
  b[a[m]]--;
  m++;
}
    cout<<m<<endl<<i;
    return 0;
  }
 }
cout<<"-1 -1";
return 0;
}
