#include <iostream>
#include <cmath>

using namespace std;

long i,j,n,k,a[101010],d[101010],b,e;

int main()
{
	cin >> n >> k;
	for (i=1;i<=n;i++) cin >> a[i];
	for (i=1;i<=n && k>0;i++){
		if (d[a[i]]==0) {
			k--;
		}
		d[a[i]]++;
	}
	e=i-1;
	if (k>0) {
		cout << -1 << " " << -1;
		return 0;
	}
	b=1;
	while (d[a[b]]>1){
		d[a[b]]--;
		b++;
	}	
	cout << b << " " << e;
	return 0;	
}
