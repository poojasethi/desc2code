k,b,n,t = map(int,raw_input().split())
def get_p():
	global k,b,n,t,z
	if k == 1:
		z = 1 + n * b
		if t >= z:
			return 0
		return   (z-t)/b if (z-t)%b==0 else(z-t)/b+1
	tt =  (k-1) + b
	z = t * (k-1) + b
#	print tt,z
	ans = 0
	for i in range(0,n+1):
		if (tt * (k ** i) <= z):
			ans = i
		else:
			break
	return max(n - ans,0)
print get_p()
