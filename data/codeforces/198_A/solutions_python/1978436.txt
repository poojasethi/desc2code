import math

arr = [ int(x) for x in raw_input().split(' ') ]
k = arr[0]
b = arr[1]
n = arr[2]
t = arr[3]

if k == 1:
    tmp = (1 + n*b - t) / float(b)
else:
    n1 = math.log(k + b - 1) / math.log(k)
    n2 = math.log(t * k - t + b) / math.log(k)
    tmp = n1 - n2 + n

if tmp < 0:
    tmp = 0
print '%.0f' % (math.ceil(tmp))
