import sys
twopow128 = 2**128-1
def bp(a, n):
	res = 1
	while n:
		if (n & 1):
			res *= a
			res %= twopow128
		a *= a
		a %= twopow128
		n >>= 1
	return res

k,b,n,t=sys.stdin.readline().split(' ')
k,b,n,t=int(k),int(b),int(n),int(t)
if t==1:
	print n
	sys.exit(0)
kn=bp(k,n)
kxt=kn*t
knn=kn
x=n
while x>=0:
	if (kxt < knn): break
	knn+=(kn/k)*b
	kn/=k
	kxt/=k
	x-=1
print x+1
