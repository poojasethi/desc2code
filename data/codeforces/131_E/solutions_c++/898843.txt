#include<cstdio>
#include<set>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

set<int> H[100000],V[100000],D[200000],*E,_E[200000]; // -, |, /, \

int main(){
	E=_E+100000;

	int n,m; scanf("%d%d",&n,&m);
	int x[100000],y[100000];
	rep(i,m){
		scanf("%d%d",y+i,x+i), x[i]--, y[i]--;
		H[y[i]].insert(x[i]);
		V[x[i]].insert(y[i]);
		D[x[i]+y[i]].insert(x[i]);
		E[x[i]-y[i]].insert(x[i]);
	}

	int ans[9]={};
	set<int>::iterator it;
	rep(i,m){
		int c=0;
		it=H[y[i]].find(x[i]);
		if(it!=H[y[i]].begin()) c++;
		if(it!=--H[y[i]].end()) c++;

		it=V[x[i]].find(y[i]);
		if(it!=V[x[i]].begin()) c++;
		if(it!=--V[x[i]].end()) c++;

		it=D[x[i]+y[i]].find(x[i]);
		if(it!=D[x[i]+y[i]].begin()) c++;
		if(it!=--D[x[i]+y[i]].end()) c++;

		it=E[x[i]-y[i]].find(x[i]);
		if(it!=E[x[i]-y[i]].begin()) c++;
		if(it!=--E[x[i]-y[i]].end()) c++;

		ans[c]++;
	}

	rep(i,9) printf("%d%c",ans[i],i<8?' ':'\n');

	return 0;
}
