#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
#define maxn 100001
using namespace std;

int x[maxn],y[maxn],cnt[maxn],res[9],n,m;
pair<int,pair<int,int> > v[maxn];

void count()
{
    sort(v, v+m);
    for(int i=0; i<m; i++)
    {
        if(i>0 && v[i].first == v[i-1].first)++cnt[v[i].second.second];
        if(i<m-1 && v[i].first == v[i+1].first)++cnt[v[i].second.second];
    }
}


int main()
{
    //freopen("input1.txt","r",stdin);
    cin>>n>>m;
    for(int i=0;i<m;i++)cin>>x[i]>>y[i];
    
    for(int i=0; i<m; i++)v[i] = make_pair(x[i], make_pair(y[i],i));count();
    for(int i=0; i<m; i++)v[i] = make_pair(y[i], make_pair(x[i],i));count();
    for(int i=0; i<m; i++)v[i] = make_pair(x[i]-y[i], make_pair(y[i],i));count();
    for(int i=0; i<m; i++)v[i] = make_pair(x[i]+y[i], make_pair(y[i],i));count();
    
    for(int i=0;i<m;i++)++res[cnt[i]];
    for(int i=0; i<9; i++)cout<<res[i]<<" ";

    return 0;
}
