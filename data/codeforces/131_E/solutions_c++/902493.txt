#include <cstdio>
#include <iostream>
#include <map>
#include <algorithm>

#define fi first
#define se second
using namespace std;

typedef pair<int,int> ii;

const int MAXN = 100010;

const int arti = 100010;

int n,m;

ii ar[MAXN];

map<int,int>X[MAXN],Y[MAXN],C[MAXN*2],T[MAXN*2];

int say[15];

int main(){
	
	int x,y;
	
	scanf("%d %d",&n,&m);
	
	for(int i=1;i<=m;i++)
	{
		scanf("%d %d",&x,&y);
		ar[i]=ii(x,y);
		X[x][y]=1;
		Y[y][x]=1;
		C[x-y+arti][x]=1;
		T[x+y][x]=1;
	}
	
	map<int,int>::iterator it;
	
	int tmp=0;
	
	for(int i=1;i<=m;i++){
		x=ar[i].fi;		y=ar[i].se;
		tmp=0;
		it=X[x].begin();
		if(it->fi!=y)
			tmp++;
			
		if(X[x].upper_bound(y)!=X[x].end())
			tmp++;
		
		it=Y[y].begin();
		if(it->fi!=x)
			tmp++;
		
		if(Y[y].upper_bound(x)!=Y[y].end())
			tmp++;
		
		it=C[x-y+arti].begin();
		if(it->fi!=x)
			tmp++;
		
		if(C[x-y+arti].upper_bound(x)!=C[x-y+arti].end())
			tmp++;
		
		it=T[x+y].begin();
		if(it->fi!=x)
			tmp++;
		
		if(T[x+y].upper_bound(x)!=T[x+y].end())
			tmp++;
		
		say[tmp]++;
	}
	
	for(int i=0;i<9;i++)
		cout << say[i] << " ";
	puts("");
	return 0;
	
}
