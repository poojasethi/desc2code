#include<bits/stdc++.h>
using namespace std;
const int maxN =200010;

int x[maxN],y[maxN],n,m,ans[9];
vector<int> v[4][maxN];
int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;++i){
		scanf("%d%d",&x[i],&y[i]);
		int a=x[i],b=y[i];
		v[0][a].push_back(b);
		v[1][b].push_back(a);
		v[2][a+b].push_back(a);
		v[3][a-b+n].push_back(a);
	}
	
	for(int i=0;i<4;++i)
	for(int j=0;j<maxN;++j) sort(v[i][j].begin(),v[i][j].end());
	
//	memset(ans,0,sizeof ans);
	for(int i=0;i<m;++i){
		vector<int>::iterator it;
		int cnt=0;
		int p[]={x[i],y[i],x[i]+y[i],x[i]-y[i]+n};
		int s[]={y[i],x[i],x[i],x[i]};
		for(int j=0;j<4;++j){
			it=lower_bound(v[j][p[j]].begin(),v[j][p[j]].end(),s[j]);
			if(++it!=v[j][p[j]].end())cnt++;
			if(--it!=v[j][p[j]].begin())cnt++;
		}
		ans[cnt]++;
	}
	
	for(int i=0;i<9;++i) printf("%d ",ans[i]);
	return 0;
}
