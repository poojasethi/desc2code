#include<iostream>
#include<algorithm>
using namespace std;
struct queen {
int x,y,z;
} a[100005];
bool c1(queen i,queen j)
{
return (i.x<j.x)||(i.x==j.x && i.y<j.y);
}
bool c2(queen i,queen j)
{
return (i.y<j.y)||(i.y==j.y && i.x<j.x);
}
bool c3(queen i,queen j)
{
return (i.x+i.y<j.x+j.y)||(i.x+i.y==j.x+j.y && i.x<j.x);
}
bool c4(queen i,queen j)
{
return (i.x-i.y<j.x-j.y)||(i.x-i.y==j.x-j.y && i.x<j.x);
}
int main()
{
int n,m,i,j,k;
cin>>n>>m;
for(i=1;i<=m;i++)
{
cin>>a[i].x>>a[i].y;
a[i].z=0;
}
sort(a+1,a+m+1,&c1);
for(i=1;i<m;i++)
{
if(a[i].x==a[i+1].x) a[i].z++,a[i+1].z++;
}
sort(a+1,a+m+1,&c2);
for(i=1;i<m;i++)
{
if(a[i].y==a[i+1].y) a[i].z++,a[i+1].z++;
}
sort(a+1,a+m+1,&c3);
for(i=1;i<m;i++)
{
if(a[i].x+a[i].y==a[i+1].x+a[i+1].y) a[i].z++,a[i+1].z++;
}
sort(a+1,a+m+1,&c4);
for(i=1;i<m;i++)
{
if(a[i].x-a[i].y==a[i+1].x-a[i+1].y) a[i].z++,a[i+1].z++;
}
for(i=0;i<=8;i++)
{
k=0;
for(j=1;j<=m;j++)
{
if(a[j].z==i) k++;
}
cout<<k<<" ";
}
return 0;
}
