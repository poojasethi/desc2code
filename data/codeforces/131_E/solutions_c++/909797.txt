#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<vector>
#include<string>
#include<queue>
#define X first
#define Y second
#define pb push_back
#define mp make_pair
using namespace std;
const int N=200010;
int n,m,tmp=100001;
const int inf=2121212121;
int a1[N],a2[N],b1[N],b2[N],c1[N],c2[N],d1[N],d2[N];
struct
{
	int x,y;
}a[N];
int sum[N],s[N];
int main()
{
	int i,j,x,y;
	scanf("%d%d",&n,&m);
	for(i=1;i<=N-1;i++)
	{
		a1[i]=b1[i]=c1[i]=d1[i]=inf;
		a2[i]=b2[i]=c2[i]=d2[i]=-inf;
	}
	for(i=1;i<=m;i++)
	{
		scanf("%d%d",&x,&y);a[i].x=x,a[i].y=y;
		a1[y]=min(a1[y],x),a2[y]=max(a2[y],x);
		b1[x]=min(b1[x],y),b2[x]=max(b2[x],y);
		c1[x+y]=min(c1[x+y],x-y),c2[x+y]=max(c2[x+y],x-y);
		d1[x-y+tmp]=min(d1[x-y+tmp],x+y),d2[x-y+tmp]=max(d2[x-y+tmp],x+y);
	}
	for(i=1;i<=m;i++)
	{
		sum[i]+=(a[i].x>a1[a[i].y])+(a[i].x<a2[a[i].y]);
		sum[i]+=(a[i].y>b1[a[i].x])+(a[i].y<b2[a[i].x]);
		sum[i]+=(a[i].x-a[i].y)>c1[a[i].x+a[i].y],sum[i]+=(a[i].x-a[i].y)<c2[a[i].x+a[i].y];
		sum[i]+=(a[i].x+a[i].y)>d1[a[i].x-a[i].y+tmp],sum[i]+=(a[i].x+a[i].y)<d2[a[i].x-a[i].y+tmp];
		s[sum[i]]++;
	}
	for(i=0;i<=8;i++)cout<<s[i]<<' ';
}
