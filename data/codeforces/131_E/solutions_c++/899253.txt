#include<cstdio>
#include<algorithm>
using namespace std;
const int inf=1<<30;
int n,m;
struct node
{
    int x,y;
}arr[100005];
int mxh[100005],mih[100005],mxl[100005],mil[100005];
int mx1[200005],mi1[200005],mx2[200005],mi2[200005];
int ans[10];
int main()
{
    scanf("%d%d",&n,&m);
    int i,x,y;
    for(i=1;i<=n;i++)mih[i]=mil[i]=inf;
    for(i=n+n;i>=0;i--)mi1[i]=mi2[i]=inf;
    for(i=0;i<m;i++)
    {
        scanf("%d%d",&arr[i].x,&arr[i].y);
        x=arr[i].x;
        y=arr[i].y;
        mxh[x]=max(y,mxh[x]);
        mih[x]=min(y,mih[x]);
        mxl[y]=max(x,mxl[y]);
        mil[y]=min(x,mil[y]);
        mx1[x+y]=max(x,mx1[x+y]);
        mi1[x+y]=min(x,mi1[x+y]);
        mx2[x-y+n]=max(x,mx2[x-y+n]);
        mi2[x-y+n]=min(x,mi2[x-y+n]);
    }
    for(i=0;i<m;i++)
    {
        int sum=8;
        x=arr[i].x;
        y=arr[i].y;
        if(mxh[x]==y)sum--;
        if(mih[x]==y)sum--;
        if(mxl[y]==x)sum--;
        if(mil[y]==x)sum--;
        if(mx1[x+y]==x)sum--;
        if(mi1[x+y]==x)sum--;
        if(mx2[x-y+n]==x)sum--;
        if(mi2[x-y+n]==x)sum--;
        ans[sum]++;
    }
    for(i=0;i<9;i++)
    printf("%d ",ans[i]);
    return 0;
}
