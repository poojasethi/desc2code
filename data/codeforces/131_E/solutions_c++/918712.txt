#include <cstdio>
typedef struct unit{int x,y;}unit;
const int N=100100;
unit t[N];
int ans[9],m;
int Rmin[N],Rmax[N],Cmin[N],Cmax[N],RUmin[2*N],RUmax[2*N],RDmin[2*N],RDmax[2*N];
int calc(int x,int y)
{
	int ans=0;
	if( y<Rmax[x]&&y>Rmin[x])ans+=2;
	else if(y<Rmax[x]||y>Rmin[x])ans++;
	if( x<Cmax[y] && x>Cmin[y])ans+=2;
	else if(x<Cmax[y] || x>Cmin[y])ans++;
	if( x<RUmax[x+y] && x>RUmin[x+y])ans+=2;
	else if( x<RUmax[x+y] || x>RUmin[x+y])ans++;
	if( x<RDmax[x-y+m] && x>RDmin[x-y+m])ans+=2;
	else if( x<RDmax[x-y+m] || x>RDmin[x-y+m])ans++;
	return ans;
}
void deal(int &max,int &min,int k)
{
	if(max<k)max=k;
	if(min>k)min=k;
}
int main()
{
	int n,x,y;
	for(int i=0;i<N;i++)
		Rmin[i]=Cmin[i]=1<<30;
	for(int i=0;i<2*N;i++)
		RUmin[i]=RDmin[i]=1<<30;
	scanf("%d%d",&m,&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		t[i]=(unit){x,y};
		deal(Rmax[x],Rmin[x],y);
		deal(Cmax[y],Cmin[y],x);
		deal(RUmax[x+y],RUmin[x+y],x);
		deal(RDmax[x-y+m],RDmin[x-y+m],x);
	}
	for(int i=0;i<n;i++)
		ans[calc(t[i].x,t[i].y)]++;
	for(int i=0;i<=8;i++)
		printf("%d%c",ans[i],i==8?'\n':' ');
	scanf(" ");
}
