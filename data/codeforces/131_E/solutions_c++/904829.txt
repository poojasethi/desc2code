using namespace std;
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#define MAXN 100010
int c1[MAXN * 2], c2[MAXN * 2], c3[MAXN];
struct Point {
    int x, y, c;
    bool operator < (const Point &a) const {
        return x < a.x || x == a.x && y < a.y;
    }
}
p[MAXN];
int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    for(int i = 1; i <= m; ++i) {
        p[i].c = 0;
        scanf("%d%d", &p[i].x, &p[i].y);
    }
    sort(p + 1, p + 1 + m);
    p[0].x = -1;
    for(int i = 1; i <= m; ++i) {
        if(p[i - 1].x == p[i].x) {
            ++p[i].c, ++p[i - 1].c;
        }
        if(c1[p[i].x + p[i].y] != 0) {
            ++p[i].c, ++p[c1[p[i].x + p[i].y]].c;
        }
        if(c2[p[i].x - p[i].y + MAXN] != 0) {
            ++p[i].c, ++p[c2[p[i].x - p[i].y + MAXN]].c;
        }
        if(c3[p[i].y] != 0) {
            ++p[i].c, ++p[c3[p[i].y]].c;
        }
        c1[p[i].x + p[i].y] = i;
        c2[p[i].x - p[i].y + MAXN] = i;
        c3[p[i].y] = i;
    }
    int ans[9] = {0};
    for(int i = 1; i <= m; ++i) {
        ++ ans[p[i].c];
    }
    for(int i = 0; i <= 8; ++i) {
        printf("%d ", ans[i]);
    }
    return (0);
}
