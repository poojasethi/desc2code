#include <iostream>
#include <algorithm>
#include <vector>
#define pb push_back
#define all(a)   (a).begin(), (a).end()
using namespace std;
typedef pair <int,int>  pii;
const int MAXN=1000*100+1000;
int n,m,t[20],x,y;
pii p[MAXN];
vector <int> sa[MAXN],so[MAXN],go1[MAXN*2],go2[MAXN*2];
int main()
{
    ios::sync_with_stdio(false);
    cin>>n>>m;
    for(int i=0;i<m;i++){
        cin>>x>>y;
        p[i]=pii(x,y);
        go1[n-x+y].pb(n-x+1);
        go2[n-x+n-y+1].pb(n-x+1);
        sa[x].pb(y);
        so[y].pb(x);
    }
    for(int i=0; i<2*n; i++){
        sort(all(go1[i]));
        sort(all(go2[i]));
        if(i>n)  continue;
        sort(all(sa[i]));
        sort(all(so[i]));
    }
    for(int i=0;i<m;i++)
    {
        int x=p[i].first,y=p[i].second,temp=0;
        if(go1[n-x+y][0]<n-x+1) temp++;
        if(go1[n-x+y].back()>n-x+1)  temp++;
        if(go2[n-x+n-y+1][0]<n-x+1) temp++;
        if(go2[n-x+n-y+1].back()>n-x+1) temp++;
        if(sa[x][0]<y)  temp++;
        if(sa[x].back()>y)  temp++;
        if(so[y][0]<x)  temp++;
        if(so[y].back()>x)  temp++;
        t[temp]++;
    }
    for(int i=0;i<9;i++)
        cout<<t[i]<<' ';
}
