#include <cstdio>
#include <cstdlib>
#include <set>
#include <algorithm>
#include <iostream>
using namespace std;

int u[100000];
int v[100000];
set<int> a[100001];
set<int> b[100001];
set<int> c[200001];
set<int> d[200001];

int main(){
  int n,m; 
  scanf("%d%d", &n, &m);
  int i,j;
  for(int k = 0; k < m; k++){
    scanf("%d%d", &i,&j);
    i--, j--;
    u[k]=i; v[k] = j;
    a[i].insert(j); b[j].insert(i); 
    c[i-j+n-1].insert(i+j); d[i+j].insert(i-j+n-1);
  }
  int res[9] = {0,0,0,0,0,0,0,0,0};
  set<int>::iterator bound;
  for(int k = 0; k < m; k++){
    int r=0;
    i=u[k], j=v[k];
    r += (j > *(a[i].begin()));
    bound=a[i].end(); bound--;
    r += (j < *(bound));
    
    r += (i > *(b[j].begin()));
    bound = b[j].end(); bound--;
    r += i < *(bound);

    r += (i+j > *(c[i-j+n-1].begin()));
    bound = c[i-j+n-1].end(), bound--;
    r += (i+j < *(bound));

    r += (i-j+n-1 > *(d[i+j].begin()));
    bound = d[i+j].end(), bound--;
    r += (i-j+n-1 < *(bound));
    
    res[r]++;
  }
  for(int i = 0; i <= 8; i++)
    printf("%d ",res[i]);
  printf("\n");
  return 0;
}
