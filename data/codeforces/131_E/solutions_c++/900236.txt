#include <cstdio>
#include <set>
using namespace std;

int n, m, r[100005], c[100005], ans[10];
set<int> row[100005], col[100005], dia1[200005], dia2[200005];


int main() {
	scanf("%d%d", &n, &m);
	for (int i = 0; i < m; ++i) {
		scanf("%d%d", r + i, c + i);
		row[r[i]].insert(c[i]);
		col[c[i]].insert(r[i]);
		dia1[c[i]-r[i]+n].insert(r[i]);
		dia2[c[i]+r[i]-1].insert(r[i]);
	}
	for (int i = 0; i < m; ++i) {
		int cnt = 0;
		if (*row[r[i]].begin() < c[i]) ++cnt;
		if (*row[r[i]].rbegin() > c[i]) ++cnt;
		if (*col[c[i]].begin() < r[i]) ++cnt;
		if (*col[c[i]].rbegin() > r[i]) ++cnt;
		if (*dia1[c[i]-r[i]+n].begin() < r[i]) ++cnt;
		if (*dia1[c[i]-r[i]+n].rbegin() > r[i]) ++cnt;
		if (*dia2[c[i]+r[i]-1].begin() < r[i]) ++cnt;
		if (*dia2[c[i]+r[i]-1].rbegin() > r[i]) ++cnt;
		++ans[cnt];
	}
	for (int i = 0; i <= 8; ++i) {
		printf("%d%c", ans[i], (i==8?'\n':' '));
	}
}
