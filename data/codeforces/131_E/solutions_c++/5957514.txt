#include <iostream>
#include <set>
#include <vector>

using namespace std;

const int limite=1000000;

int mata(set<int> &s,int val)
{
  int c=0;
  set<int>::iterator it=s.upper_bound(val);
  if (it!=s.end()) c++;
  it=s.find(val);
  if (it!=s.begin()) c++;
  return c;
}


int n,m;
set<int> fil[limite],col[limite],diagsum[limite],diagres[limite];

int sol[9];

int main()
{
  cin>>n>>m;
  vector<pair<int,int> > input;
  for (int k=0;k<m;k++) {
    int i,j;
    cin>>i>>j;
    input.push_back(pair<int,int> (i,j));
    fil[i].insert(j);
    col[j].insert(i);
    diagsum[i+j].insert(i-j);
    diagres[i-j+n].insert(i+j);
  }
  for (int k=0;k<m;k++) {
    int i=input[k].first;
    int j=input[k].second;
    int c=mata(fil[i],j)+mata(col[j],i)+mata(diagsum[i+j],i-j)+mata(diagres[i-j+n],i+j);
    sol[c]++;
  }
  for (int i=0;i<9;i++) {
    if (i>0) cout<<" ";
    cout<<sol[i];
  }
  cout<<endl;
}
