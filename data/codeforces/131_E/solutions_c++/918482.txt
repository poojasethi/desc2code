#include <cstdio>
#include <vector>
#include <string>
#include <cstring>
#include <queue>
#include <algorithm>
using namespace std;

int n,m,r[100000],c[100000];
vector<int> row[100001];
vector<int> col[100001];
vector<int> scp[200002];
vector<int> scq[200002];

int getcnt(vector<int>& u, int x){
	size_t i=lower_bound(u.begin(),u.end(),x)-u.begin();
	int ret=0;
	if(i+1<u.size()) ret++;
	if(i) ret++;
	return ret;
}

int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++){
		scanf("%d%d",r+i,c+i);
		row[r[i]].push_back(c[i]);
		col[c[i]].push_back(r[i]);
		scp[r[i]+c[i]].push_back(c[i]-r[i]+n);
		scq[c[i]-r[i]+n].push_back(r[i]+c[i]);
	}
	for(int i=1;i<=n;i++){
		sort(row[i].begin(),row[i].end());
		sort(col[i].begin(),col[i].end());
	}
	for(int i=0;i<=2*n+1;i++){
		sort(scp[i].begin(),scp[i].end());
		sort(scq[i].begin(),scq[i].end());
	}
	int ans[9]={};
	for(int i=0;i<m;i++){
		int cnt=0;
		cnt+=getcnt(row[r[i]],c[i]);
		cnt+=getcnt(col[c[i]],r[i]);
		cnt+=getcnt(scp[r[i]+c[i]],c[i]-r[i]+n);
		cnt+=getcnt(scq[c[i]-r[i]+n],r[i]+c[i]);
		ans[cnt]++;
	}
	for(int i=0;i<=8;i++) printf("%d ",ans[i]);
}
