#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
struct c{int x,y;}s[100002];
pair<pair<int,int>,int> p[100002];
int ans[100002];
int l,n;
void f(){
    sort(p,p+n);
    for(int i=1;i<n;i++){
        if(p[i].first.first == p[i-1].first.first){
            ans[p[i].second]++;
            ans[p[i-1].second]++;
        }
    }
}
int main(){
    scanf("%d %d",&l,&n);
    for(int i=0;i<n;i++) ans[i]=0;
    for(int i=0;i<n;i++) scanf("%d %d",&s[i].x,&s[i].y);
    for(int i=0;i<n;i++){
        p[i].second=i;
        p[i].first=make_pair(s[i].x,s[i].y);
    }
    f();
    for(int i=0;i<n;i++){
        p[i].second=i;
        p[i].first=make_pair(s[i].y,s[i].x);
    }
    f();
    for(int i=0;i<n;i++){
        p[i].second=i;
        p[i].first=make_pair(s[i].x+s[i].y,s[i].x-s[i].y);
    }
    f();
    for(int i=0;i<n;i++){
        p[i].second=i;
        p[i].first=make_pair(s[i].x-s[i].y,s[i].x+s[i].y);
    }
    f();
    int pp[9];
    memset(pp,0,sizeof(pp));
    for(int i=0;i<n;i++) pp[ans[i]]++;
    for(int i=0;i<=8;i++) printf("%d%c",pp[i],i==8?'\n':' ');
    return 0;
}
