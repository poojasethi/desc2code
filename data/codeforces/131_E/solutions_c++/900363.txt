#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <string>
using namespace std;

#define MAX 111111
int N , M , X[ MAX ] , Y[ MAX ] , res[ 10 ] , cnt[ MAX ];
pair< int , pair< int , int > > lis[ 4 ][ MAX ];

int main(){
	scanf( "%d%d" ,&M ,&N );

	for( int q = 0 ; q < N ; q++ ){
		scanf( "%d%d" ,&X[ q ] ,&Y[ q ] );
		lis[ 0 ][ q ] = make_pair( X[ q ] , make_pair( Y[ q ] , q ) );
		lis[ 1 ][ q ] = make_pair( Y[ q ] , make_pair( X[ q ] , q ) );
		lis[ 2 ][ q ] = make_pair( X[ q ] + Y[ q ] , make_pair( X[ q ] , q ) );
		lis[ 3 ][ q ] = make_pair( X[ q ] - Y[ q ] , make_pair( X[ q ] , q ) );
	}

	for( int dim = 0 ; dim < 4 ; dim++ ){
		sort( lis[ dim ] , lis[ dim ] + N );
	}

	for( int q = 0 ; q < N ; q++ ){
		for( int dim = 0 ; dim < 4 ; dim++ ){
			int i = lis[ dim ][ q ].second.second;
			if( q && lis[ dim ][ q - 1 ].first == lis[ dim ][ q ].first ){
				cnt[ i ] ++;
			}
			if( q < N - 1 && lis[ dim ][ q + 1 ].first == lis[ dim ][ q ].first ){
				cnt[ i ] ++;
			}
		}
	}

	for( int q = 0 ; q < N ; q++ ){
		res[ cnt[ q ] ] ++;
	}

	for( int q = 0 ; q <= 8 ; q++ ){
		if( q ) printf( " " );
		printf( "%d" ,res[ q ] );
	}
	printf( "\n" );

	return 0;
}
