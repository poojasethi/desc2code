#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

vector<int> h[102000],l[102000],zd[302000],fd[302000];
int ax[100000],ay[100000],ans[10];
int main(){
	int n,m;scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++){
		int x,y;scanf("%d%d",&x,&y);x--,y--;
		h[x].push_back(y);
		l[y].push_back(x);
		zd[x-y+n].push_back(x);
		fd[x+y].push_back(x);
		ax[i]=x;ay[i]=y;
	}
	for(int i=0;i<n;i++)sort(h[i].begin(),h[i].end()),sort(l[i].begin(),l[i].end());
	for(int i=0;i<n*2+1;i++)sort(zd[i].begin(),zd[i].end()),sort(fd[i].begin(),fd[i].end());
	for(int i=0;i<m;i++){
		int tmp=0;
		if(h[ax[i]][0]<ay[i])tmp++;
		if(h[ax[i]][h[ax[i]].size()-1]>ay[i])tmp++;
		if(l[ay[i]][0]<ax[i])tmp++;
		if(l[ay[i]][l[ay[i]].size()-1]>ax[i])tmp++;
		if(zd[ax[i]-ay[i]+n][0]<ax[i])tmp++;
		if(zd[ax[i]-ay[i]+n][zd[ax[i]-ay[i]+n].size()-1]>ax[i])tmp++;
		if(fd[ax[i]+ay[i]][0]<ax[i])tmp++;
		if(fd[ax[i]+ay[i]][fd[ax[i]+ay[i]].size()-1]>ax[i])tmp++;
		ans[tmp]++;
	}
	for(int i=0;i<=8;i++)printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
