#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
const int MAXN = 200100;
int d[8][MAXN];
int cnt[MAXN];
int ans[MAXN];
struct data {
    int x, y;
    bool operator <(const data& d) const {
        return x == d.x ? y < d.y : x < d.x;
    }
}p[MAXN];
int main() {
    int n, m, x, y;
    scanf("%d%d", &n, &m);
    memset(d, -1, sizeof(d));
    for (int i = 0; i < m; i++) {
        scanf("%d%d", &x, &y);
        x--, y--;
        p[i].x = x, p[i].y = y;
    }
    sort(p, p + m);
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < 4; j++) {
            int t;
            if (j == 0) t = p[i].x;
            if (j == 1) t = p[i].x - p[i].y + n - 1;
            if (j == 2) t = p[i].y;
            if (j == 3) t = p[i].x + p[i].y;
            if (d[4 + j][t] != -1) {
                cnt[d[4 + j][t]]++;
                cnt[i]++;
            }
            d[4 + j][t] = i;
        }
    }
    for (int i = 0; i < m; i++) {
        ans[cnt[i]]++;
    }
    for (int i = 0; i <= 8; i++) {
        if (i) putchar(' ');
        cout << ans[i];
    }
    puts("");
}
