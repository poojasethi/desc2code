#include <stdio.h>
#include <algorithm>

using namespace std;

struct d{
	long long a, b;
	d(){}
	d(long long a, long long b, int c):a(a),b(b),c(c){}
	int c;
} a[ 100005 ];

bool cmpA( d x, d y ){
	return x.a < y.a || ( x.a == y.a && x.b < y.b );
}

bool cmpB( d x, d y ){
	return x.b < y.b || ( x.b == y.b && x.a < y.a );
}

int ans[9];

int main(){
	int n, m;
	
	scanf("%d %d",&n,&m);
	for (int i=0; i<m; i++){
		scanf("%lld %lld",&a[i].a,&a[i].b);
		a[i].c = 0;
	}
	
	sort( a, a+m, cmpA ); for (int i=1; i<m; i++) if ( a[i].a == a[i-1].a ) a[i].c++, a[i-1].c++;
	sort( a, a+m, cmpB ); for (int i=1; i<m; i++) if ( a[i].b == a[i-1].b ) a[i].c++, a[i-1].c++;
	for (int i=0; i<m; i++) a[i] = d( a[i].a+a[i].b, a[i].a-a[i].b+n, a[i].c );
	sort( a, a+m, cmpA ); for (int i=1; i<m; i++) if ( a[i].a == a[i-1].a ) a[i].c++, a[i-1].c++;
	sort( a, a+m, cmpB ); for (int i=1; i<m; i++) if ( a[i].b == a[i-1].b ) a[i].c++, a[i-1].c++;
	
	for (int i=0; i<m; i++) ans[ a[i].c ]++;
	for (int i=0; i<9; i++) printf("%s%d",i?" ":"",ans[i]);
	printf("\n");

	return 0;
}
