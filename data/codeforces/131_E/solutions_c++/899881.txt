#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>

#include <set>
#define fi "CF95_2_5.INP"

#define nmax 100000+5
#define INF 1000000000
using namespace std;
typedef pair<int,int> II;
typedef pair<II,int> III;
typedef set<III> SIII;
//VARIABLES
int n,m,d[nmax],c[10];
SIII my_set[5];
//FUNCTION PROTOTYPES

void Init();
void Process();

//MAIN
int main()
{
//       freopen(fi,"r",stdin);
//       freopen(fo,"w",stdout);
       Init();
       Process();
       return 0;
}

//FUNCTIONS AND PROCEDURES
void Init()
{
	int p,q;
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;++i)
	{
		scanf("%d%d",&p,&q);
		my_set[1].insert(III(II(p,q),i));
		my_set[2].insert(III(II(q,p),i));
		my_set[3].insert(III(II(p+q,p),i));
		my_set[4].insert(III(II(p-q,q),i));
	}
}
void Process()
{
	SIII::iterator it,it2;
	for (int i=1;i<=4;++i)
	{
		it=my_set[i].begin();it2=it;++it2;
		for (;it2!=my_set[i].end();++it,++it2)
		if (it->first.first==it2->first.first)
		{
			++d[it->second];++d[it2->second];
		}
	}
	for (int i=1;i<=m;++i) ++c[d[i]];
	for (int i=0;i<=8;++i) printf("%d ",c[i]);
	putchar('\n');
}
