#include <bits/stdc++.h>
using namespace std;
map<int,set<int> > M[4];
int r[100001],c[100001];
int t[10];
int func(set<int> &S,int x)
{
      set<int>::iterator it=S.find(x);
      if(S.size()==1)
            return 0;
      assert(it!=S.end());
      if(it==S.begin())
            return 1;
      it++;
      if(it==S.end())
            return 1;
      return 2;
}
int main()
{
      std::ios::sync_with_stdio(false);
      cin.tie(0);
      int n;
      cin>>n>>n;
      for(int i=0;i<n;++i)
      {
            int x,y;
            cin>>x>>y;
            r[i]=x;
            c[i]=y;
            M[0][x].insert(y);
            M[1][y].insert(x);
            M[2][x+y].insert(y);
            M[3][x-y].insert(y);
      }
      for(int i=0;i<n;++i)
      {     
            int x=r[i];
            int y=c[i];
            int cnt=0;
            cnt+=func(M[0][x],y);
            cnt+=func(M[1][y],x);
            cnt+=func(M[2][x+y],y);
            cnt+=func(M[3][x-y],y);
            ++t[cnt];
      } 
      for(int i=0;i<9;++i)
            cout<<t[i]<<" ";
}
