#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

typedef pair<int,int> ii;
typedef pair<ii,int> iii;

const int N = 100123;
const int K = 8;
int n,m;
int r[N], c[N], nattack[N];
int ans[K+1];

iii v[N];

void f() {
  sort(v, v+m);
  for(int i=0,j; i<m; i=j) {
    j=i+1;
    while(j<m && v[j].first.first == v[i].first.first) {
      ++nattack[v[j-1].second];
      ++nattack[v[j].second];
      ++j;
    }
  }
}

int main() {
  scanf(" %d %d", &n, &m);
  for(int i=0; i<m; ++i) {
    scanf(" %d %d", &r[i], &c[i]);
    nattack[i] = 0;
  }

  // line
  for(int i=0; i<m; ++i)
    v[i] = iii( ii(r[i], c[i]), i );
  f();

  // column
  for(int i=0; i<m; ++i)
    v[i] = iii( ii(c[i], r[i]), i );
  f();

  // diag 1
  for(int i=0; i<m; ++i)
    v[i] = iii( ii(r[i]-c[i], r[i]+c[i]), i );
  f();

  // diag 2
  for(int i=0; i<m; ++i)
    v[i] = iii( ii(r[i]+c[i], r[i]-c[i]), i );
  f();

  memset(ans,0,sizeof(ans));
  for(int i=0; i<m; ++i)
    ++ans[nattack[i]];
  for(int i=0; i<=K; ++i) {
    if(i) putchar(' ');
    printf("%d", ans[i]);
  }
  putchar('\n');

  return 0;
}
