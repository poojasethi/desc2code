#include <cstdio>
#include <cstring>
int main () {
	char x[32], y[32];
	int n, cnt = 63, j;

	scanf("%d", &n);
	scanf("%s", y);
	for (int i = 0; i != n; i++) {
		scanf("%s", x);
		for (j = 0; x[j] && y[j] && (x[j] == y[j]);j++); 
		if (j < cnt) cnt = j;
		strcpy(x, y);
	}
	printf("%d\n", cnt);
	
	return 0;
}
