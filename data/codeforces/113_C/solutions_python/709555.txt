f = lambda x: int(x**.5)+1
h,z = map(bytearray,'1\0')

L,R = map(int,raw_input().split())

n = f(R); b = n/2*h; b[0] = 0
for k in xrange(1,f(n)/2):
    if b[k]: p = 2*k+1; s = k*(p+1); b[s::p] = z*len(b[s::p])

g = ((i*(i+1),2*i+1) for i,v in enumerate(b) if v)

r = (R+3)/4*h; r[0] = 0
for s,p in g:
    r[s::p] = z*len(r[s::p])

print r.count(h,(L+2)/4)+(L<=2<=R)
