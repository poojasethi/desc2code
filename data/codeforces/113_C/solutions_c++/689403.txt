#include <stdio.h>
#include <bitset>
int l,r,N,ans,i,j;
std::bitset<300000001> p;
int main()
{
    scanf("%d%d",&l,&r);
    p.set();
    for(i=3;(j=i*i)<=r;i+=2)
        if( p[i] )
            for(j=i*i;j<=r;j+=2*i)
                p[j] = false;
    for(i=5;i<=r;i+=4)
        if( i >= l && p[i] )
            ans++;
    printf("%d",(l<=2 && r >=2 ? ++ans : ans));
    return 0;
}
