#include<cstdio>
#include<bitset>
#include<iostream>
using namespace std;
bitset<300000005>pri;
int main()
{
   int a,b;
   scanf("%d%d",&a,&b);
   for(int i=3;i*i<=b;i+=2)
     if(!pri[i])
     {
     	for(int j=i*i;j<=b;j+=i)
     	{
     	   pri[j] = 1;	
		}
	 }
    int sum = (2>=a&&2<=b)?1:0;
    while(a%4!=1)
    {
    	a++;
	}
	a = max(5,a);
    for(int i=a;i<=b;i+=4)
    {
       if(!pri[i])sum++;
	}
	printf("%d\n",sum);
}
