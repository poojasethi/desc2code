#include <stdio.h>
#include <iostream>
#include <bitset>
using namespace std;
bitset <300000005> bst;
int v=0,l,r,t;
int main( ){
	cin>>l>>r;
	if (l<=2 && 2<=r) v++;
	bst.set();
	for (int i=3;i*i<=r;i+=2){
		t=i*i;
		if (bst[i])
			while (t<=r){
				bst[t]=false;
				t+=i*2;
			}
	}
	for (int i=5;i<=r;i+=4)
		if (bst[i] && i>=l)
			v++;
	cout<<v<<endl;
	return 0;
}
