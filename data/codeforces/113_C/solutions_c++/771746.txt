#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
#include <bitset>
using namespace std;

int l,r,res,j;
bitset <300000001> prime;

int main () {
	scanf("%d%d",&l,&r);
	prime.set();
	for(int i=3;(j = i*i) <= r; i += 2)  if(prime[i]){
		for(;j<=r;j+=(i<<1))	
			prime[j] = false;
	}
	for(int i=5;i<=r;i+=4) {
		if(prime[i]  &&  l <= i)  res++;
	}
	if(l <= 2  &&  r >= 2)  res++; 
	printf("%d",res);
	return 0;
}
