#include <bitset>
#include <cstdio>
using namespace std;
const int maxn = 300000000;
bitset<maxn+1> p;
int cnt = 0, l, r, i, j;
int main() {
    scanf("%d%d", &l, &r);
    p.set();
    for (i = 3; (j = i*i) <= r; i += 2)
    if (p[i]) for (; j <= r; j += i << 1) p[j] = 0;
    for (i = 5; i <= r; i += 4) if (p[i] && i >= l) cnt++;
    printf("%d\n", cnt+(l <= 2 && 2 <= r));
}
