#include<cstdio>
#define S 20000
unsigned P[S+1],l,r,Z[76000000>>5],i,j,*d,b,c;
inline int a(){
	if(l>r||r<2)return 0;
	for(i=2;i<=S;++i)
	if(!P[i]){
			P[c++]=i;
			for(j=i*i;j<=S;j+=i)P[j]=true;
	}
	b=(r-1)>>2;Z[0]=1;
	for(d=P+1;(j=((*d)*(*d)))<=r;++d)
		for(j/=4;j<=b;j+=*d)
        Z[j>>5]|=(1<<(j&31));
	j=(l<3&&r>1);
	l=(l+2)>>2;
	r=(r-1)>>2;
	for(i=l;i<=r;++i)if(!(Z[i>>5]&(1<<(i&31))))++j;
	return j;
}
int main(){
	scanf("%d%d",&l,&r);
	printf("%d\n",a());
	return 0;
}
