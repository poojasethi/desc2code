#include<cstdio>
const int N=300000001;
int f[N/64+1];
int main()
{
	int i,j,l,r,ans=0;
	for(i=3;i<N;i+=2)
	{
		if(f[i/64]&(1<<( (i%64)>>1 ) ))continue;
		for(j=i*i;i<20000&&j<N;j+=(i<<1))
			f[j/64]|=(1<<( (j%64)>>1 ) );
	}
	scanf("%d%d",&l,&r);
	i=l;
	while((i-1)%4)i++;
	if(i==1)i+=4;
	for(;i<=r;i+=4)
	{
		if(!( f[i/64]&(1<<( (i%64)>>1 ) ) ) )
			ans++;
	}
	if(l<=2&&r>1)ans++;
	printf("%d\n",ans);
	return 0;
}
