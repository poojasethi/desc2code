#include<stdio.h>
#include<bitset>
using namespace std;
const int maxn=1e8+1e8+1e8;
bitset<maxn>p;
int main()
{
	int R,L;
	scanf("%d %d",&R,&L);
	p.set();
	for(int i=3;i*i<=maxn;i+=2)
	if(p[i])
	for(int j=i*i;j<=L;j+=2*i)
	p[j]=false;
	int ans=0;
	for(int i=5;i<=L;i+=4)
	if(p[i]&&i>=R)
	ans++;
	ans+=(R<=2&&L>=2)?1:0;
	printf("%d\n",ans);
	return 0;	
	
}
