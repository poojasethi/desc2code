#include <bitset>
#include <cstdio>
using namespace std;

const int MAXN = 300000000 + 86;
bitset <MAXN/2> v;

int main() {
	int l, r;
	scanf("%d%d", &l, &r);
	for (int i=1,k=2*i+1; k*k<=r; ++i,k+=2) if (!v.test(i)) {
		for (int j=(k-1)*(i+1); 2*j+1<=r; j+=k) v.set(j);
	}
	int tot = l <= 2 && 2 <= r;
	for (int i=2,k=2*i+1; k<=r; i+=2,k+=4) if (!v[i] && k >= l) ++tot;
	printf("%d\n", tot);
	return 0;
}
