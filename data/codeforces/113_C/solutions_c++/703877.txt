#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<bitset>
using namespace std;
const int N=300000100;
bitset<300000100>v;
int main()
{
    int i,ans=0,j,l,r;
    scanf("%d%d",&l,&r);
    ans+=(l<=2&&r>=2);
    v.set();
    for(i=3;(j=i*i)<=N;i+=2)
    if(v[j])for(;j<=N;j+=i<<1)v[j]=false;
    for(i=5;i<=r;i+=4)if(i>=l&&v[i])ans++;
    printf("%d\n",ans);
    return 0;
}
