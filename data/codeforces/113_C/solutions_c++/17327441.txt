#include<cstdio>

int p[16252326],r[10000001];

int main()
{
   int i,j,t,a,b,ans=0,o=0;
   scanf("%d%d",&a,&b);
   for (i=2;i<=b;i++)
   {
      if (!(r[i>>5]&(1<<(i&31))))
      {
         p[++o]=i;
         if (i>=a&(i&3)!=3)
            ans++;
      }
      for (j=1;(t=p[j]*i)<=b;j++)
      {
         r[t>>5]|=1<<(t&31);
         if (j==1&&!(i&1)||j>1&&i%p[j]==0)
            break;
      }
   }
   printf("%d\n",ans);
   return 0;
}
