#include<iostream>
#include<vector>
using namespace std;
const int L(300000000);
int i,j,k,t,a,b,ans;
vector<bool> bo(L/2+10);
int main()
{
    bo[0]=1;
    for (i=3;i*i<=L;i+=2) 
        if (!bo[i/2]) {
           for (j=i*i/2;j<=L/2;j+=i) bo[j]=1;
        }
    cin>>a>>b;
    if (a<=2&&2<=b) ++ans;
    for (i=a;i<=b;++i)
        if (i%4==1&&!bo[i/2]) ++ans;
    cout<<ans<<endl;
    return 0;
}
