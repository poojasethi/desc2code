#include<stdio.h>
int n,m,l,r,i,j,res,p;
char use[37500000],P[2] = {2,4};
void Criba()
{
    res=0;
    if(l<=2)res=1;
    p = 0;
  for(i=5;i<=r;i+=P[p&1],p++)
  {
      if((use[i>>3]&(1<<(i&7)))==0)
      {
          res+=((i&3)==1)*(i>=l);
          if(i<=r/i)
          for(j=i*i;j<=r;j+=2*i)
              use[j>>3]=(use[j>>3]|(1<<(j&7)));
      }
  }
}
int main()
{
 scanf("%d%d",&l,&r);
 if(!(l==1 && r==1))Criba();
 printf("%d\n",res);
}
