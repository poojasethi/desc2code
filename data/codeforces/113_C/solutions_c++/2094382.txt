#include<stdio.h>
#include<bitset>
using namespace std;
const int maxn=300000008;
bitset<maxn>b;

int main(){
    int i,j;
    int l,r;
    int ans=0;
    scanf("%d%d",&l,&r);
    b.set();
    for(i=3;i*i<=r;i+=2){
        if(b[i])
            for(j=i*i;j<=r;j+=(i<<1))
                b[j]=0;
    }
    if(l<=2 && r>=2)ans++;
    for(i=5;i<=r;i+=4){
        if(i>=l && b[i])
            ans++;
    }
    printf("%d\n",ans);
    return 0;
}
