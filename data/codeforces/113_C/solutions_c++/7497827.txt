#include<bits/stdc++.h>
using namespace std;

bitset<300000005> f;
int p[16500000],pn;

int main()
{
	int l,r,s,i,j;
	scanf("%d%d",&l,&r);
	s=l<=2&&r>=2;
	for(i=2;i<=r;i++)
	{
		if(!f[i])
		{
			p[pn++]=i;
			if(i>=l&&i%4==1)
				s++;
		}
		for(j=0;j<pn&&i*p[j]<=r;j++)
		{
			f[i*p[j]]=1;
			if(i%p[j]==0)
				break;
		}
	}
	printf("%d\n",s);
	return 0;
}
