#include <cstdio>
#include <bitset>
using namespace std;

const int MAX = (int)(3e8 + 1);

bitset<MAX> p;
int L, R, i, j, res = 0;

int main() {
    scanf("%d%d", &L, &R);
    res = (L <= 2 && 2 <= R);
    while (L % 4 != 1) L++; L = max(L, 5);

    p.set();
    for (i = 3; (j = i * i) <= R; i += 2) if (p[i])
        for (; j <= R; j += (i << 1)) p[j] = 0;

    for (i = L; i <= R; i += 4)
        if (p[i]) res++;

    printf("%d", res);
}
