#include<cstdio>
#include<bitset>
using namespace std;
bitset<300000005> f;
int main()
{
    int l,r;
    scanf("%d%d",&l,&r);
    for(int i=3;i*i<=r;i+=2)
    if (!f[i]) for(int j=i*i;j<=r;j+=i<<1) f[j]=1;
    f[1]=1;
    int cnt=0,i=l-l%4+1;
    if (i<l) i+=4;
    for(;i<=r;i+=4) if (!f[i]) cnt++;
    printf("%d\n",cnt+(l<=2&&r>=2));
}
