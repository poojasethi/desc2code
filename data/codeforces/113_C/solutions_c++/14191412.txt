#include<cstdio>
#include<iostream>
#include<cmath>
#include<algorithm>
#include<bitset>
using namespace std;
typedef int l;
bitset<300000000> b;

int main(){
    l le,ri; cin>>le>>ri;

    l res=0;
    if(le<=2 and ri>=2) ++res;
    b[1]=1;
    for(int i=3;i*i<=ri;i+=2){
        if(!b[i]){
            for(int j=i*i;j<=ri;j+=2*i) b[j]=1;
        }
    }
    for(int i=5;i<=ri;i+=4) if(!b[i] and i>=le) ++res;
    cout<<res;

}
