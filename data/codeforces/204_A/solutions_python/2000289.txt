#!/usr/bin/python

def ini():
        global a
        a[1] = 9
        a[2] = 9
        for i in range(3,19):
                a[i] = a[i - 1] * 10
#	print a

def get_number(n):
        global a
        num = len(str(n))
#	print 'num === ',num
        ans = 0
        for i in range(1,num):
                ans += a[i]
#	print ans
        if(num == 1):
                ans += n
        elif num == 2:
                if(n // 10 > n % 10) :
                        ans += (n // 10 - 1)
                else:
                        ans += (n // 10)
        else:
                m = str(n)
                start = int(m[0:1])
                end = n % 10
                ans += (start - 1) * (a[num] / 9)
                if(start <= end):
                        ans += int(m[1:-1]) + 1
                else:
                        ans += int(m[1:-1])
        return ans

def main():
        left,right = map(int,raw_input().split())
        print get_number(right) - get_number(left - 1)

if __name__ == '__main__':
        global a
        a = [0 for i in range(20)]
        ini()
        main()
