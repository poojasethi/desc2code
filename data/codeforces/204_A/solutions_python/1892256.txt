def f(s):
  if s == '0':
    return 0
  n = len(s)
  if n > 2:
    sum = '9' * (n-2)
    sum = int(sum) + 9
  elif n == 2:
    sum = 9
  else:
    sum = int(s)

  ss = s[0]
  cs = int(s)
  cs -= 10**(n-1)
  s = str(cs)
  n = len(s)

  for si in s[:-1]:
    sum += int(si) * 10**(n-2)
    n -= 1
  if int(s[-1]) >= int(ss):
    sum += 1
  return sum


l, r = map(int, raw_input().split())
print f(str(r)) - f(str(l-1))
