from math import (log10, ceil, floor)

def solve(x):
    if x < 10:
        return x
    ret = 0
    w = 0
    y = x / 10
    while y > 0:
        w = w + 1
        y = y / 10
    f = x / (10 ** w)
    if x % 10 >= f:
        ret = ret + 1
    for i in xrange(2, w + 1):
        ret = ret + 9 * (10 ** (i - 2))
    ret = ret + (f - 1) * (10 ** (w - 1))
    ret = ret + x % (10 ** w) / 10
    return ret + 9


if __name__ == "__main__":
    x, y = map(int, raw_input().split())
    print solve(y) - solve(x - 1)
