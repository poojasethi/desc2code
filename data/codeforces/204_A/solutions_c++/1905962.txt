#include<iostream>
#include<sstream>
using namespace std;
long long cal(long long n)
{
   if(n<10) return n;
   string str;
   stringstream strn;
   strn<<n;
   strn>>str;
   return n/10-1+(str[0]<=str[str.size()-1]?1:0)+9;
}
int main()
{
   long long l,r;
   cin>>l>>r;
   cout<<cal(r)-cal(l-1)<<endl;
   return 0;
}
