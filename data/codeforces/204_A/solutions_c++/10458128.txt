#include<stdio.h>
long long fun(long long m)
{
    if(m<10) return m;
    long long r=m%10,Sum=m/10+9;
    while(m>=10) m/=10;
    if(m>r) Sum--;
    return Sum;
}
int main()
{
    long long l,r;
    scanf("%I64d%I64d",&l,&r);
    printf("%I64d\n",fun(r)-fun(l-1));
    return 0;
}
