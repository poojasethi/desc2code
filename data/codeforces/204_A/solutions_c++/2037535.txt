#include <iostream>
using namespace std;

long long calc(long long a) {
	if (a < 10) return a;
	long long r = a / 10;
	r += 8;
	long long f = a;
	while (f >= 10) f /= 10;
	if (f <= a % 10) r++;
	return r;
}

int main() {
	long long a, b;
	cin >> a >> b;
	cout << calc(b) - calc(a - 1) << endl;
	return 0;
}
