#include <iostream>
using namespace std;

long long solve (long long num)
{
    if (num<10) return num;
    long long last = num;
    while (last>=10)
        last/=10;
    return 9+num/10-(num%10<last);
}

int main ()
{
    long long a,b;
    cin >> a >> b;
    cout << solve (b) - solve(a-1);
    return 0;
}
