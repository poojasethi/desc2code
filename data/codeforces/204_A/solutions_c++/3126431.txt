#include <iostream>
#define LL long long

using namespace std;

LL f(LL n)
{
    LL r = 0;
    if(n<10)
        return n;
    r = n/10 + 8;
    LL temp = n;
    while(n>9)
        n/=10;
    if(temp%10>=n)
        r++;
    return r;
}

int main()
{
    LL n ,m;
    cin>>n>>m;
    cout<<f(m)-f(n-1)<<endl;
}
