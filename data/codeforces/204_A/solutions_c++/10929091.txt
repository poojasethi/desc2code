#include<cstdio>
#include<cstring>
using namespace std;
__int64 solve(__int64 n)
{
    __int64 k=n%10,ans=n/10+9;
    if(n<10) return n;
    while(n>=10) n/=10;
    if(n>k) ans--;
    return ans;
}
int main()
{
    __int64 l,r;
    scanf("%I64d%I64d",&l,&r);
    printf("%I64d\n",solve(r)-solve(l-1));
    return 0;
}
