#include<iostream>
#include<string>
#include<sstream>
using namespace std;
long long l,r;
long long f(long long n)
{if(n<10) return(n);
    string s;
    stringstream str;
    str<<n;
    str>>s;
    return(n/10-1+(s[0]<=s[s.size()-1]?1:0)+9);


}
int main()
{
    cin>>l>>r;
    cout<<f(r)-f(l-1);
return 0;
}
