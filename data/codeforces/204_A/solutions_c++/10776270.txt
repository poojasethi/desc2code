#include<stdio.h>
#include<algorithm>
using namespace std;
__int64 f(__int64 x){
	if(x<10)
		return x;
	__int64 n = x % 10;
	__int64 m = x / 10 + 9;
	while(x>=10)
		x /= 10;
	if(x>n)
		m--;
	return m;
}
int main(){
	__int64 l, r;
	scanf("%I64d%I64d", &l, &r);
	printf("%I64d\n",f(r)-f(l-1));
	return 0;
}
