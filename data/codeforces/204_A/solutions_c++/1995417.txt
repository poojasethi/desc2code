#include <iostream>
using namespace std;
long long r,l;
long long f(long long x) {
    if(x < 10) return x;
    long long k = x/10;
    k += 8;
    long long q = x ;
    while(q > 9) q /= 10;
    if(q <= x % 10) k++;
    return k;
}

int main() {
    cin >> l >> r;
    cout << f(r) - f(l-1) << endl ;
}
