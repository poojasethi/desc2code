#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll first(ll a)
{
    while(a >= 10)
        a/=10;
    return a;
}
ll find(ll a)
{
    if(a < 10) return a;
    return (a/10 - 1) + 9 + (a%10 >= first(a) ? 1:0) ;
}
int main()
{
    ll a,b;
    cin>>a>>b;
    cout<<find(b)-find(a-1);
    return 0;
}
