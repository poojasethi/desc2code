#include<bits/stdc++.h>
using namespace std;

long long cal(long long n){
    if(n < 10) return n;
    long long res = n/10 + 9,p = n;
    while(p > 9) p /= 10;
    if(p > n%10) res--;
    return res;

}
int main(){
    long long l,r; cin >> l >> r;
    cout << cal(r) - cal(l-1) << endl;
    //main();
    return 0;
}
