#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef long long ll;

int a[2010][2010];
int l[2010][2010];
int r[2010][2010];
int u[2010][2010];
int d[2010][2010];

int n,m;

void init()
{
    for(int i=2;i<n;i++)
    {
        for(int j=1;j<=m && a[i][j];j++)
            l[i][j]=1;
        for(int j=m;j>=1 && a[i][j];j--)
            r[i][j]=1;
    }
    for(int j=2;j<m;j++)
    {
        for(int i=1;i<=n && a[i][j];i++)
            u[i][j]=1;
        for(int i=n;i>=1 && a[i][j];i--)
            d[i][j]=1;
    }
    
}

ll solve0()
{
    ll ret=0;
    for(int j=2;j<m;j++)
        ret+=u[n][j];
    for(int i=2;i<n;i++)
        ret+=l[i][m];
//  cout<<ret<<endl;
    return ret;
}

ll solve1()
{
    ll ret=0;
    for(int i=2;i<n;i++)
        for(int j=2;j<m;j++)
            ret+=(l[i][j]+r[i][j])*(u[i][j]+d[i][j]);
//  cout<<ret<<endl;
    return ret;
}

ll solve2()
{
    ll ret=0;
    int now;
    for(int i=2;i<n;i++)
    {
        now=0;
        for(int j=2;j<m;j++)
        {
            if (a[i][j])
                ret+=u[i][j-1]*d[i][j]+u[i][j]*d[i][j-1]+now*(u[i][j]+d[i][j]);
            now+=u[i][j-1]+d[i][j-1];
            if (!a[i][j-1] || !a[i][j])
                now=0;
        }
    }
    for(int j=2;j<m;j++)
    {
        now=0;
        for(int i=2;i<n;i++)
        {
            if (a[i][j])
                ret+=l[i-1][j]*r[i][j]+l[i][j]*r[i-1][j]+now*(l[i][j]+r[i][j]);
            now+=l[i-1][j]+r[i-1][j];
            if (!a[i-1][j] || !a[i][j])
                now=0;
        }
    }
//  cout<<ret<<endl;
    return ret;
}

int main()
{
    ios::sync_with_stdio(false);
    cin>>n>>m;
    char c;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
        {
            cin>>c;
            if (c=='.') a[i][j]=1;
        }
            
    ll ans=0;
    init();
    ans=solve0()+solve1()+solve2();
    cout<<ans<<endl;
    return 0;
}
