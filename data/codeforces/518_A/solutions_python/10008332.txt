s1 = raw_input()
s2 = raw_input()

lst = list(s1)
x=False

i=len(s1)-1
while( i >= 0 ):
    if( lst[i] != 'z' ):
        lst[i] = chr(ord(lst[i])+1)
        break
    else:
        lst[i] = 'a'
    i -= 1

for i in range(len(lst)):
    if( lst[i] != s2[i] ):
        print "".join(lst)
        x = True
        break

if x == False:
    print "No such string"
