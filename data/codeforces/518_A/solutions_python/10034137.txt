def f(s):
    return map(lambda x: ord(x) - 97, s)

s = f(raw_input())
t = f(raw_input())

s.reverse()

for i, v in enumerate(s):
    s[i] = (v + 1) % 26
    if v != ord('z') - 97:
        break

s.reverse()

if s == t:
    print "No such string"
else:
    print ''.join(map(lambda x: chr(97+x), s))
