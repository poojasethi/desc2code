s = raw_input()
t = raw_input()

next = ""
qts = 0
for i in range(len(t)-1, -1, -1):
  qts += 1
  if s[i] == "z":
    next = next + "a"
  else : 
    next = chr(ord(s[i])+1) + next
    break

prefix = ""
for i in range(len(s) - qts):
  prefix = prefix + s[i]
next = prefix + next


if next == t: print "No such string"
else : print next
