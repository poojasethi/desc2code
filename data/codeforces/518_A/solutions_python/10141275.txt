import sys

s = list(sys.stdin.readline().strip())
t = sys.stdin.readline().strip()

z = len(s) - 1
s[z] = chr(ord(s[z]) + 1)
while z > 0 and ord(s[z]) > ord('z'):
	s[z] = 'a'
	z -= 1
	s[z] = chr(ord(s[z]) + 1)

s = "".join(s)
if s < t:
	print s
else:
	print "No such string"
