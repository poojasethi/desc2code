first = raw_input('')
second = raw_input('')

n = len(first)
i = n - 1
while first[i] == 'z' and i >= 0:
    i -= 1
if i == -1:
    print 'No such string'
else:
    next_string = first[0: i] + chr(ord(first[i]) + 1) + 'a' * (n - 1 - i)
    if next_string == second:
        print 'No such string'
    else:
        print next_string
