a=raw_input()
b=raw_input()
n=len(a)
if a[:n-1]==b[:n-1] and ord(b[-1]) -ord(a[-1]) == 1:
    print "No such string"
else:
    for i in range(n-1,-1,-1):

        if a[i] != 'z':
            r= a[:i] + chr(ord(a[i])+1) + 'a'*(n-i-1)
            if r == b:
                print "No such string"
            else:
                print r
            exit(0)
