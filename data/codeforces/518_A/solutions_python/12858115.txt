import sys

def nextStr(s):
	s = [ord(c) for c in list(s)]
	idx = len(s) - 1
	s[idx] += 1
	while s[idx] > 122:
		s[idx - 1] += 1
		s[idx] = 97
		idx -= 1
	return ''.join([chr(c) for c in s])

s1 = sys.stdin.readline().strip()
s2 = sys.stdin.readline().strip()

s3 = nextStr(s1)

if s2 == s3:
	print 'No such string'
else:
	print s3
