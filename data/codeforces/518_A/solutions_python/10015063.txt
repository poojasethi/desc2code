a=raw_input()
b=raw_input()
c=list(a)
d=list(b)
for i in range(len(a)-1,-1,-1):
    if c[i]<'z':
        c[i]=chr(ord(c[i])+1);break;
for i in range(len(b)-1,-1,-1):
    if d[i]>'a':
        d[i]=chr(ord(d[i])-1);break;
bb=''.join(c)
aa=''.join(d)
if bb>=b and aa<=a:
    print "No such string"  
elif bb>=b:
    print aa 
else: print bb
