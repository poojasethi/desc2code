from sys import stdin
s = stdin.readline().split()[0]
t = stdin.readline().split()[0]
ans = []
temp = 1
for item in s[::-1]:
    cur = ord(item)-ord("a")+temp
    if cur > 25:
        temp = 1
    else:
        temp = 0
    ans.append(chr(cur%26+ord("a")))
ans = "".join(ans[::-1])
if ans<t:
    print ans
else:
    print "No such string"
