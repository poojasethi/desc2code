#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

def solve(a,b,c,d,e,f):
    x = b*d*f
    y = a*c*e
    if x+y > 0:
        if x > y:
            return 'Ron'
        else:
            return 'Hermione'
    else:
        if c==0 and d>0:
            return 'Ron'
        if a==0 and b>0 and d>0:
            return 'Ron'
        return 'Hermione'

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

A = numbers_from_line()
a,b,c,d,e,f = tuple(A[:6])

a = solve(a,b,c,d,e,f)
output.write('%s\n' % a)
