from __future__ import division

a, b, c, d, e, f = [int(x) for x in raw_input().split()]
if c == 0 and d != 0 or a == 0 and b != 0 and d != 0:
    winner = 'Ron'
elif b == 0 or d == 0 or f == 0:
    winner = 'Hermione'
else:
    x = a / b;
    y = c / d;
    z = e / f;
    factor = x * y * z;
    winner = 'Ron' if factor < 1 else 'Hermione'

print winner
