
def calc(a, b, c, d, e, f):
    if c == 0 and d > 0: return True
    if a == 0 and b > 0 and d > 0: return True
    if b == 0 or d == 0 or f == 0: return False
    
    if a*c*e < b*d*f:
        return True
    return False
        
        
    
def main():
    a, b, c, d, e, f = map(int, raw_input().split())

        print "Ron"

main()
