s = raw_input().split()
for i in range(len(s)):
	s[i] = int(s[i])

st = 1
for x in s[::2]:
	st *= x

if s[3] == 0:
	print 'Hermione'
	quit()
if s[2] == 0:
	print 'Ron'
	quit()
if s[1] == 0:
	print 'Hermione'
	quit()
if s[0] == 0:
	print 'Ron'
	quit()
if s[5] == 0:
	print 'Hermione'
	quit()
if s[4] == 0:
	print 'Ron'
	quit()

tmp1 = st*s[1]/s[0]	
tmp2 = tmp1*s[3]/s[2]
tmp3 = tmp2*s[5]/s[4]

if tmp3 > st:
	print 'Ron'
else:
	print 'Hermione'
