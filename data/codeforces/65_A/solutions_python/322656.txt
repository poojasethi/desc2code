import sys
a, b, c, d, e, f = [int(x) for x in raw_input().split()]
if a == 0 and b != 0 and d != 0:
    sys.stdout.write('Ron')
elif c == 0 and d != 0:
    sys.stdout.write('Ron')
elif e == 0 and b != 0 and d != 0 and f != 0:
    sys.stdout.write('Ron')
elif b == 0 or d == 0 or f == 0:
    sys.stdout.write('Hermione')    
else:
    if b *d * f - a * c * e > 0:
        sys.stdout.write('Ron')
    else:
        sys.stdout.write('Hermione')
