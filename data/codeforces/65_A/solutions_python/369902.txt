#!/usr/bin/python

a = raw_input().split()
for i in xrange(len(a)):
	a[i] = float(a[i])
eps = 1e-9
inf = 1e+9
p = 1. 
if a[0] < eps:
	pb = a[1] * inf
else:
	pb = a[1]/a[0] * p

if a[2] < eps:
	gold = a[3] * inf
else:
	gold = a[3]/a[2] * pb 

if a[4] < eps:
	p2 = a[5] * inf
else:
	p2 = a[5]/a[4] * gold

if (p2 > p and gold > eps) or gold > inf/2:
	print 'Ron'
else:
	print 'Hermione'
