#!/usr/bin/env python
# -*- coding: utf-8 -*-

def make_k(n1, n2):
    if n2 == 0:
        return 0
    elif n1 == 0:
        return -1
    else:
        return n2/n1

def main():
    n = [float(i) for i in raw_input().split()]
    k1 = make_k(n[0], n[1])
    k2 = make_k(n[2], n[3])
    k3 = make_k(n[4], n[5])
    if k2 == -1 or (k1 == -1 and k2 != 0) or (k3 == -1 and k1 != 0 and k2 != 0) or (k1 > 0 and k2 > 0 and k3 > 0 and k1*k2*k3 > 1):
        print "Ron"
    else:
        print "Hermione"
    return 0

if __name__ == '__main__':
    main()
