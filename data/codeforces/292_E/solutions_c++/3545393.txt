#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int u=1000010;
int a[u],b[u],dat[u],num[u],n,m,x,y,z,i,t;

void build(int x,int l,int r)
{
	dat[x]=-1;
	if(l==r) {num[x]=b[l]; return;}
	int mid=l+r>>1;
	build(x*2,l,mid);
	build(x*2+1,mid+1,r);
}

inline void update(int x,int y) {dat[x]=y,num[x]=a[y];}
inline void spread(int x,int l,int r)
{
	if(dat[x]<0) return;
	int mid=l+r>>1;
	update(x*2,dat[x]);
	update(x*2+1,dat[x]+mid-l+1);
	dat[x]=-1;
}

int ask(int x,int l,int r,int t)
{
	if(l==r) return num[x];
	int mid=l+r>>1;
	spread(x,l,r);
	if(t<=mid) return ask(x*2,l,mid,t);
	else return ask(x*2+1,mid+1,r,t);
}

void change(int x,int l,int r,int L,int R,int t)
{
	if(L<=l&&R>=r) {update(x,t+l-L); return;} 
	spread(x,l,r);
	int mid=l+r>>1;
	if(L<=mid) change(x*2,l,mid,L,R,t);
	if(R>mid) change(x*2+1,mid+1,r,L,R,t);
}

int main()
{
	cin>>n>>m;
	for(i=1;i<=n;i++) scanf("%d",&a[i]);
	for(i=1;i<=n;i++) scanf("%d",&b[i]);
	build(1,1,n);
	for(i=1;i<=m;i++)
	{
		scanf("%d",&t);
		if(t==1) scanf("%d%d%d",&x,&y,&z),change(1,1,n,y,y+z-1,x);
		else scanf("%d",&x),printf("%d\n",ask(1,1,n,x));
	}
	return 0;
}
