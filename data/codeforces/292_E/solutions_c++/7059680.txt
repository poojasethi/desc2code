#include <cstdio>
#include <algorithm>
using namespace std;
int N,M;
int a[100005],b[100005];
int x,y,q,k,tim;
struct yp{
	int x,y,t;
};
yp lz[100005*3];

void upd(int lvl,int bas,int son)
{
	//son = (y+k-1)
	if(bas > (y+k-1) || son  < y)	return ;
	if(bas >= y && son <= (y+k-1))
	{
		lz[lvl].x = x;
		lz[lvl].y = y;
		lz[lvl].t = tim;
	}
	else
	{
		upd(lvl*2,bas,(bas+son)/2);
		upd(lvl*2+1,(bas+son)/2+1,son);
	}
}

int sor(int lvl,int bas,int son)
{
	if(bas > x || son < x)	return -1;
	if(bas==x && son==x)	return lvl;
	int mx = lvl;
	int l = sor(lvl*2,bas,(bas+son)/2);
	int r =	sor(lvl*2+1,(bas+son)/2+1,son);
	int gec = max(l,r);
	if(lz[mx].t < lz[gec].t)
		mx = gec;
	return mx;
}

int main()
{
	//freopen("at.in","r",stdin);
	scanf("%d %d",&N,&M);
	for(int i=0;i<N;i++)
		scanf("%d",&a[i]);
	for(int i=0;i<N;i++)
		scanf("%d",&b[i]);
	for(int i=0;i<M;i++)
	{
		scanf("%d",&q);
		if(q == 2)
		{
			scanf("%d",&x);
			x--;
			int deg = sor(1,0,N-1);
			if(!lz[deg].t)	printf("%d\n",b[x]);
			else
				printf("%d\n",a[x - lz[deg].y + lz[deg].x]);
		}
		else
		{
			scanf("%d %d %d",&x,&y,&k);
			x--,y--;
			tim = i+1;
			upd(1,0,N-1);
		}
	}
	return 0;
}
