#include <iostream>
using namespace std;
int n,m,a[100001],b[100001],x[100001],y[100001],k[100001],der[400001],Two;
void Update(int l,int r,int k)
{
	if (l==r) der[l]=k;
	else 
	{
	  if (l % 2 ==1) { der[l]=k;l++; }
	  if (r % 2 ==0) { der[r]=k;r--; }
	  if (l<r) 
	  Update(l/2,r/2,k);
	}
}
int main()
{
	cin>>n>>m;
	for (Two=1;Two<n;Two*=2);
	for (int i=1;i<=n;i++) cin>>a[i];
	for (int i=1;i<=n;i++) cin>>b[i];
	for (int i=1;i<=m;i++)
	{
		int t;
		cin>>t;
		if (t==1)
		{
			cin>>x[i]>>y[i]>>k[i];
			Update(y[i]+Two-1,y[i]+k[i]+Two-2,i);
		}
		else 
		{
			int num,num1;
			cin>>num; num1=num;
			num+=Two-1;
			int ans=0;
			 while (num!=0)
			 {
				 if (ans<der[num]) ans=der[num];
				 num/=2;
			 }
			 if (ans!=0)
			 cout<<a[num1-y[ans]+x[ans]];
			 else cout<<b[num1];
			 cout<<'\n';
		}
	}
}
