#include <iostream>
using namespace std;
int n,m,a[2][1<<17],p[1<<18];
void upd(int y1,int y2,int x,int i=1,int s=0,int w=1<<17){
  if(y1<=s&&y2>=s+w)p[i]=x+s-y1;
  else if(w>>=1){
    if(p[i]>=0)p[i*2]=p[i],p[i*2+1]=p[i]>0?p[i]+w:p[i],p[i]=-1;
    if(y1<s+w)upd(y1,y2,x,i*2,s,w);
    if(y2>s+w)upd(y1,y2,x,i*2+1,s+w,w);
  }
}
int qry(int x,int i=1,int s=0,int w=1<<17){
  if(p[i]<0){
    w>>=1;
    return x<s+w?qry(x,i*2,s,w):qry(x,i*2+1,s+w,w);
  }
  if(p[i]<1)return 0;
  return p[i]+x-s;
}
int main(void){
  cin>>n>>m;
  for(int j=0;j<2;++j)for(int i=1;i<=n;++i)cin>>a[j][i];
  for(int i=0,c,x,y,k;i<m&&cin>>c>>x;++i)if(1==c){
    cin>>y>>k;
    upd(y,y+k,x);
  }
  else{
    int j=qry(x);
    cout<<(j>0?a[0][j]:a[1][x])<<endl;
  }
}
