#include <iostream>
using namespace std;

long i,j,n,m,a[1000000],b[1000000],t,x,y,k,r,l,an[1000000],dr[1000000],ans[1000000],dro[1000000],ko[1000000],o;

main()
{
cin>>n>>m;
 
 for (i=1;i<=n;i++)
   cin>>a[i];
   	
 for (i=1;i<=n;i++)
   cin>>b[i];
 
 while (n%700!=0) n++;  
   
   
   for (t=1;t<=m;t++)
   {
   	 cin>>o;
   	   
   	   if (o==1)
   	    {
   	    	cin>>x>>y>>k;
   	    	l=(y+699)/700;
   	        r=(y+k-1+699)/700;
			   
			   if (r-j<=1)
			    {
			       for (i=y;i<=y+k-1;i++)
				    dr[i]=t,an[i]=a[x+i-y];	
			    }  else
				{
				  for (i=y; (i%700!=1 || i==y) && i<=y+k-1;i++)
				    dr[i]=t,an[i]=a[x+i-y];
				  
				  for (i=y+k-1; (i%700!=0 || i==y+k-1) && i>=y ; i--)
				    dr[i]=t,an[i]=a[ x+k-1 -(y+k-1-i)];
					
				  for (i=l+1;i<=r-1;i++)
				   dro[i]=t,ans[i]=x,ko[i]=y;	  
				}    	
   	    	
        } else 
         {
         	cin>>x;
         	 r=(x+699)/700;
         if (dr[x]==0 && dro[r]==0)  {cout<<b[x]<<endl; continue;}
		  
		  if (dr[x]>dro[r])
		     	cout<<an[x]<<endl;  else cout<<a[ ans[r]+x-ko[r] ]<<endl;
		     	 	 
         	 
         }       
        
        
        
        
  }
  
  
  
}
