#include<iostream>
using namespace std;

int p[1048576], c[1048576], aa[100010], bb[100010];

void dod(int a, int b, int w, int s)
{
	while(a<=b)
	{
		if(a%2)
		{
			c[a]=s;
			p[a]=w;
		}
		if(b%2==0)
		{
			c[b]=s;
			p[b]=w;
		}
		a=(a+1)/2;
		b=(b-1)/2;
	}
}

int spr(int i)
{
	int pp, cc=0;
	while(i)
	{
		if(c[i]>cc)
		{
			cc=c[i];
			pp=p[i];
		}
		i/=2;
	}
	if(cc)
	return pp;
	else
	return 100000000;
}
		

int main()
{
	ios_base::sync_with_stdio(0);
	int n, m, x, y, t, k, d=524288;
	cin>>n>>m;
	for(int i=1; i<=n; i++)
	{
		cin>>aa[i];
	}
	for(int i=1; i<=n; i++)
	{
		cin>>bb[i];
	}
	for(int i=0; i<m; i++)
	{
		cin>>t;
		if(t==1)
		{
			cin>>x>>y>>k;
			dod(y+d, y+d+k-1, x-y, i+1);
		}
		else
		{
			cin>>x;
			y=spr(x+d);
			if(y<=n)
			{
				cout<<aa[x+y]<<"\n";
			}
			else
			{
				cout<<bb[x]<<"\n";
			}
		}
	}
	return 0;
}
