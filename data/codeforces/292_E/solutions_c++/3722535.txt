#include <cstdio>
#include <iostream>

#define fi first
#define se second

using namespace std;

typedef pair<int,int> ii;

const int MAXN=100100;

int N,M;
int ar[MAXN];
int as[MAXN];
int seg[MAXN*3];

ii q[MAXN];

void upd( int k , int b , int e , int a1 , int a2 , int v ){
	
	if( b>a2 || e<a1 )
		return;
	if( b>=a1 && e<=a2 ){
		seg[k]=v;
		return;
	}
	
	if( seg[k]!=-1 ){
		seg[k*2]=seg[k];
		seg[k*2+1]=seg[k];
		seg[k]=-1;
	}
	
	upd(k*2,b,(b+e)/2,a1,a2,v);
	upd(k*2+1,(b+e)/2+1,e,a1,a2,v);
}

int query( int k , int b , int e , int pl ){
	
	if( b>pl || e<pl )
		return 0;
	if( seg[k]!=-1 )
		return seg[k];
	if( b==e )
		return 0;
	
	return max(query(k*2,b,(b+e)/2,pl),query(k*2+1,(b+e)/2+1,e,pl));
}

int main(){
	
	scanf(" %d %d",&N,&M);
	
	for( int i=1 ; i<=N ; i++ )
		scanf(" %d",ar+i);
	for( int i=1 ; i<=N ; i++ )
		scanf(" %d",as+i);
	
	for( int a,b,i=1 ; i<=M ; i++ ){
		scanf(" %d",&a);
		if( a==1 ){
			scanf(" %d %d %d",&q[i].fi,&q[i].se,&b);
			upd(1,1,N,q[i].se,q[i].se+b-1,i);
		}
		else{
			scanf(" %d",&b);
			int t=query(1,1,N,b);
			if( t==0 )
				printf("%d\n",as[b]);
			else
				printf("%d\n",ar[q[t].fi+b-q[t].se]);
		}
	}
	
	return 0;
}
