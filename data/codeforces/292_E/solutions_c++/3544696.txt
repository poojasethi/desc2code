#include <iostream>
#include <cstdio>
#include <memory.h>
using namespace std;

int a[100500], b[100500];
const int K = 500;
int g[100500 / K + 10];
int n;

void norm(int pos){
	if (g[pos] == -1)
		return;
	for (int i = 0; i < K; ++i)
		b[pos * K + i] = a[g[pos] + i];
	g[pos] = -1;
}

void mcopy(int x, int y, int q){
	if (q != 0 && y % K != 0)
		norm(y / K);
	while (q != 0 && y % K != 0){
		b[y] = a[x];
		y++, x++, q--;
	}
	while (q >= K){
		g[y / K] = x;
		x += K, y += K, q -= K;
	}
	if (q != 0)
		norm(y / K);
	while (q != 0){
		b[y] = a[x];
		y++, x++, q--;
	}
}

int ask(int x){
	if (g[x / K] != -1)
		return a[g[x / K] + x % K];
	else
		return b[x];
}

int main(){
	//freopen("input.txt", "r", stdin);
	memset(g, -1, sizeof(g));
	int m;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; ++i)
		scanf("%d", &a[i]);
	for (int i = 0; i < n; ++i)
		scanf("%d", &b[i]);
	for (int i = 0; i < m; ++i){
		int t, x, y, q;
		scanf("%d", &t);
		if (t == 1){
			scanf("%d%d%d", &x, &y, &q);
			x--, y--;
			mcopy(x, y, q);
		}
		else{
			scanf("%d", &q);
			q--;
			printf("%d\n", ask(q));
		}
	}
}
