#include<stdio.h>

int a[100500],b[100500];
const int SIZE=1<<17;
int xx[SIZE*2],yy[SIZE*2];

void push(int x){
	if(x<SIZE && xx[x]){
		xx[x*2]=xx[x];
		yy[x*2]=yy[x];
		xx[x*2+1]=xx[x];
		yy[x*2+1]=yy[x];
		xx[x]=0;
		yy[x]=0;
	}
}
void set(int x,int l,int r,int ll,int rr,int v){
	if(rr<l || r<ll)
		return;
	if(l<=ll && rr<=r){
		xx[v]=x;
		yy[v]=l;
		return;
	}
	push(v);
	int m=(ll+rr)/2;
	set(x,l,r,ll,m,v*2);
	set(x,l,r,m+1,rr,v*2+1);
}
void set(int x,int y,int k){
	set(x,y,y+k-1,0,SIZE-1,1);
}
int get(int i){
	int ii=i+SIZE;
	i--;
	int res=b[i];
	while(ii){
		if(xx[ii])
			res=a[xx[ii]+i-yy[ii]];
		ii/=2;
	}
	return res;
}

int n,m;

int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%d",a+i);
	for(int i=0;i<n;i++)
		scanf("%d",b+i);
	for(int i=0;i<m;i++){
		int g;
		scanf("%d",&g);
		if(g==1){
			int x,y,k;
			scanf("%d%d%d",&x,&y,&k);
			set(x,y,k);
		}else{
			int x;
			scanf("%d",&x);
			printf("%d\n",get(x));
		}
	}
}
