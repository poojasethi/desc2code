#include <stdio.h>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <math.h>
#include <string>
#include <vector>
#include <map>
using namespace std;

const int H=(1<<20);
struct jim { int x,y,t; } T[2*H];
int n,m,i,j,x,y,k,a[H],b[H],t,o;

void upd(int i,int l,int r,int x,int y,int xx,int yy,int tt)
{
 if ( r<x || y<l ) return;
 if ( x<=l && r<=y )
  {
   T[i].x=xx;
   T[i].y=yy;
   T[i].t=tt;
   return;
  }
 upd(i*2,l,(l+r)/2,x,y,xx,yy,tt);
 upd(i*2+1,(l+r)/2+1,r,x,y,xx,yy,tt);
}

int ans(int i)
{
 int id=i;
 i+=H-1;
 int xx,yy,tt=-1;
 while ( true )
  {
   if ( T[i].t>tt )
    {
     xx=T[i].x;
     yy=T[i].y;
     tt=T[i].t;
    }
   if ( i==1 ) break;
   i=i/2;
  }

 if ( tt==0 ) return b[id];
 return a[xx+(id-yy)];
}

main()
{
 scanf("%d%d",&n,&m);
 for ( i=1;i<=n;i++ )
  scanf("%d",&a[i]);
 for ( i=1;i<=n;i++ )
  scanf("%d",&b[i]);

 for ( t=1;t<=m;t++ )
  {
   scanf("%d",&o);
   if ( o==2 )
    {
     scanf("%d",&x);
     printf("%d\n",ans(x));
    }
   else
    {
     scanf("%d%d%d",&x,&y,&k);
     upd(1,1,H,y,y+k-1,x,y,t);
    }
  }
}
