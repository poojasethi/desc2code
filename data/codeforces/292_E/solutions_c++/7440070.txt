#include <cstdio>
#include <cstring>
int t[300001],mask,n;
inline void pushDown(int pos){
	if(t[pos]!=0x3F3F3F3F){
		for(int i=pos<<1;i<=(pos<<1)+1&&i<=n+mask;i++){
			t[i]=t[pos];
		}
		t[pos]=0x3F3F3F3F;
	}
}
void set(int l,int r,int v,int pos,int lv){
	if(l<=(pos<<lv)&&r>=((pos+1)<<lv)){
		t[pos]=v;
		return;
	}
	if(l>=((pos+1)<<lv)||r<=(pos<<lv)){
		return;
	}
	pushDown(pos);
	set(l,r,v,pos<<1,lv-1);
	set(l,r,v,(pos<<1)+1,lv-1);
}
int get(int pos){
	int r=0x3F3F3F3F;
	while(pos){
		if(t[pos]!=0x3F3F3F3F){
			r=t[pos];
		}
		pos>>=1;
	}
	return r;
}
int a[100000],b[100000];
int main(){
	int q;
	scanf("%d%d",&n,&q);
	int lv=0;
	while((n-1)>>lv){
		lv++;
	}
	mask=1<<lv;
	memset(t,'?',(n+mask)*sizeof(int));
	for(int i=0;i<n;i++){
		scanf("%d",a+i);
	}
	for(int i=0;i<n;i++){
		scanf("%d",b+i);
	}
	for(int i=0;i<q;i++){
		int op;
		scanf("%d",&op);
		if(op==1){
			int apos,bpos,l;
			scanf("%d%d%d",&apos,&bpos,&l);
			apos--;
			bpos--;
			set(bpos+mask,bpos+l+mask,apos-bpos,1,lv);
		}else if(op==2){
			int pos;
			scanf("%d",&pos);
			pos--;
			int got=get(pos+mask);
			printf("%d\n",got==0x3F3F3F3F?b[pos]:a[pos+got]);
		}
	}
	return 0;
}
