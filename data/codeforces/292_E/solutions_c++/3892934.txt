#include <iostream>
#include <cstring>
using namespace std;

const int MAX = 100005;
int n, m, a[MAX], b[MAX], loc[4*MAX], pow = (1 << 18), diff[MAX];

int main(){

    cin >> n >> m;
    for(int i = 0; i<n; i++) cin >> a[i];
    for(int i = 0; i<n; i++) cin >> b[i];

    memset(loc, 0, sizeof(loc));

    int t, x, y, k;
    for(int i = 1; i<=m; i++){
        cin >> t;
        if(t==1){
            cin >> x >> y >> k;
            diff[i] = x-y;
            int lo = pow+y, hi = pow+y+k-1;
            while(lo<=hi){
                loc[lo] = i;
                loc[hi] = i;
                lo = (lo+1)/2;
                hi = (hi-1)/2;
            }
        } else {
            cin >> x;
            k = 0;
            for(int i = pow+x; i>0; i/=2){
                k = max(k, loc[i]);
            }
            if(k==0) cout << b[x-1] << endl;
            else cout << a[x+diff[k]-1] << endl;
        }
    }
    
}
