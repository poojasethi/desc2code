#include<stdio.h>
#define M5 500000
int REC[M5],a[M5],b[M5];
int X[M5],Y[M5],K[M5];
int n,m,i,k,x,y,t,p;

void UPDATE(int now,int l,int r,int L,int R)
 {if (r<L || l >R) return;
  if (L <= l && r <= R){if (i>REC[now]) REC[now]=i;return;}
  UPDATE(now*2,l,(l+r)/2,L,R);
  UPDATE(now*2+1,(l+r)/2+1,r,L,R);
 }

main()
{scanf("%d%d",&n,&m);
 for (i=1;i<=n;i++) scanf("%d",&a[i]);
 for (i=1;i<=n;i++) scanf("%d",&b[i]);
 k=1;while (k<=n) k*=2;k--;
 for (i=1;i<=m;i++)
  {scanf("%d",&t);
   if (t == 2)
    {scanf("%d",&x);p=0;y=x+k;
     while (y>=1) {if (p<REC[y])p=REC[y];y/=2;}
     if (p == 0) printf("%d\n",b[x]); else
                 printf("%d\n",a[X[p]+x-Y[p]]);}else
   scanf("%d%d%d",&X[i],&Y[i],&K[i]);
   UPDATE(1,k+1,2*k+1,Y[i]+k,Y[i]+K[i]+k-1);
  }
}
