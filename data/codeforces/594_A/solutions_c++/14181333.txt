#include<bits/stdc++.h>
using namespace std;
int n, a[200100], ans= 1e9+7;
int main(int argc, char **argv) {
	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i) scanf("%d", a+i);
	sort(a, a+n);
	int del = (n-2)/2, i=0, j=n-1;
	for(int k=0 ; k<=del ; ++k) ans = min(ans,a[j-k]-a[i+(del-k)]);
	printf("%d\n", ans);
	return 0;
}
