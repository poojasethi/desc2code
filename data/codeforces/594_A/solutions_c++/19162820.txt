#include <iostream>
#include <bits/stdc++.h>
#include <limits.h>
using namespace std;
int main()
{
	int N;
	cin>>N;
	int a;
	vector<int > A;
	for(int i=0;i<N;i++)
	{	cin>>a;
		A.push_back(a);
	}
	sort(A.begin(),A.end());
	
	long long int min=INT_MAX;
	for(int i=0;i<N/2;i++)
		if(A[i+N/2]-A[i] < min)
			min=A[i+N/2]-A[i];

	cout<<min;
	

}
