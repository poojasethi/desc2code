#include <iostream>
#include <algorithm>
using namespace std;
const int maxn=2*1e6;
int n,arr[maxn],ans = 1e9+10;
int main(){
    cin >> n;
    for(int i=0 ; i<n ; i++) cin >> arr[i];
    sort(arr,arr+n);
    for(int i=0 ; i<(n/2) ; i++)
        ans = min(ans,arr[i + n/2] - arr[i]);
    cout << ans << endl;
}
