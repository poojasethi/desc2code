#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,ar[200009],i,ma;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    scanf("%d",&ar[i]);
    sort(ar,ar+n);
    ma=1000000009;
    for(i=0;i<n/2;i++)
    {
        if(ar[n/2+i]-ar[i]<ma)
        ma=ar[n/2+i]-ar[i];
    }
    printf("%d\n",ma);
    return 0;
}
