#include <stdio.h>
#include <algorithm>

using namespace std;

int v[200000];

int main (void) {
	int n;
	scanf ("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf ("%d", &v[i]);
	}
	sort(v, v+n);
	int ans = v[n-1]-v[0];
	int k = (n-2)/2;
	for (int i = 0; i <= k; i++) {
		ans = min(ans, v[n-1-k+i] - v[i]);
	}
	printf ("%d\n", ans);
}
