#include <bits/stdc++.h>
using namespace std;

const int maxn = 2e5+5;

int n,ar[maxn];

int main(){
	
	scanf("%d",&n);
	
	for( int i=0 ; i < n ; i++ )
		scanf("%d",&ar[i]);
	
	sort(ar , ar+n);
	int ans = 1e9;
	for( int i=0 ; i < n/2 ; i++ )
		ans = min(ans , ar[i + n/2] - ar[i]);
	printf("%d\n",ans);
	
	return 0;
}
