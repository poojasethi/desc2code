#include<bits/stdc++.h>
using namespace std;
int a[200005];
int main()
{
int n,i,x;
scanf("%d",&n);
for(i=0;i<n;i++)
scanf("%d",&a[i]);
int ans = 2e9;
sort(a,a+n);
for(i=0;i<n/2;i++){
//printf("for %d check %d and %d\n",i,a[n/2+i],a[i]);
ans = min(ans,a[n/2+i]-a[i]);
}
printf("%d\n",ans);
return 0;
}
