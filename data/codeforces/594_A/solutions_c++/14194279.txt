#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int inf=1000000005;
int a[210001],i,j,k,n,m,ans;
int main(){
	scanf("%d",&n);k=(n-2)/2;ans=inf;
	for (i=1;i<=n;i++) scanf("%d",&a[i]);
	sort(a+1,a+1+n);
	for (i=1;i<=n-k-1;i++) ans=min(ans,a[i+k+1]-a[i]);
	printf("%d\n",ans);
	return 0;
}
