#include <bits/stdc++.h>
using namespace std;
int x[200005];
int main()
{
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i)
        cin >> x[i];
    sort(x, x + n);
    int ans = 1e9+1;
    for (int i = 0; i < (n + 1) / 2; ++i)
        ans = min(ans, x[i + n / 2] - x[i]);
    cout << ans << "\n";
    return 0;
}
