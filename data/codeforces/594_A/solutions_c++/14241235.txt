#include <algorithm>
#include <cstdio>
using namespace std;
const int N = 200005;
int n, a[N];
int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) scanf("%d", a+i);
	sort(a, a+n);
	int t = (n-2)/2, ans = a[n-1]-a[0];
	for (int i = 0; i+t+1 < n; i++)
		ans = min(ans, a[i+t+1]-a[i]);
	printf("%d\n", ans);
	return 0;
}
