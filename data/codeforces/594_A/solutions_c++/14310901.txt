/*Warrior and Archer*/

#include <cstdio>
#include <algorithm>
using namespace std;

int n;
int x[220000];

int main()
{
	scanf("%d",&n);
	int k=n/2;
	for (int i=0;i<n;i++)
		scanf("%d",x+i);
	sort(x,x+n);
	int min=0x7FFFFFFF;
	for (int i=0;i<k;i++)
		if (x[i+k]-x[i]<min) min=x[i+k]-x[i];
	printf("%d\n",min);
}
