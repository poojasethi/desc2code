#include<stdio.h>
#include<math.h>
#define sq(a) ((a)*(a))
#define esp 0.000001
double x[3],y[3],xi,yi,r[3],dis[3];
double chk(double a,double b)
{
    double rt=0;
    for(int i=0;i<3;i++)
    {
        dis[i] = sqrt( sq(a-x[i]) + sq(b-y[i]) )  ;
    }
    for(int i=0;i<3;i++)
    {
        rt+=sq((dis[i]/r[i] - dis[(i+1)%3]/r[(i+1)%3] ));
    }
    return rt;
}
int main()
{
    xi=yi=0;
    for(int i=0;i<3;i++)
    {
        scanf("%lf%lf%lf",x+i,y+i,r+i);
        xi+=x[i];
        yi+=y[i];
    }
    xi/=3;
    yi/=3;
    double s=1;

    for(;s>esp;)
    {
        if(chk(xi,yi)>chk(xi-s,yi)) xi-=s;
        else if(chk(xi,yi)>chk(xi+s,yi))xi+=s;
        else if(chk(xi,yi)>chk(xi,yi-s))yi-=s;
        else if(chk(xi,yi)>chk(xi,yi+s))yi+=s;
        else s*=0.5;

    }
    if(chk(xi,yi)<esp)printf("%lf %lf\n",xi,yi);
    return 0;
}
