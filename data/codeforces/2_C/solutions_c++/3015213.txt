#include <cstdio>
#include <cmath>
double x[3],y[3],r[3],t[3],dx,dy,s,f;
double F(double _x,double _y){
    double ret=0;
    for(int i=0;i<3;i++) t[i]=sqrt((_x-x[i])*(_x-x[i])  + (_y-y[i])*(_y-y[i]))/r[i];
    for(int i=0;i<3;i++) ret += (t[i]-t[(i+1)%3])*(t[i]-t[(i+1)%3]);
    return ret;
}
int main(){
    for(int i=0;i<3;i++) scanf("%lf %lf %lf",&x[i],&y[i],&r[i]);
    for(int i=0;i<3;i++) dx += x[i]/3.0, dy += y[i]/3.0;
    for(s=1;s>1e-6;f=0){
             if(F(dx,dy) > F(dx+s,dy))   dx+=s, f=1;
        else if(F(dx,dy) > F(dx-s,dy))   dx-=s, f=1;
        else if(F(dx,dy) > F(dx,dy+s))   dy+=s, f=1;
        else if(F(dx,dy) > F(dx,dy-s))   dy-=s, f=1;
        if(!f) s *= 0.7;
    }
    if(F(dx,dy)<1e-5) printf("%.5lf %.5lf\n",dx,dy);
    return 0;
}
