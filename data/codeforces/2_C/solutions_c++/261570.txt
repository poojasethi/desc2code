#include<cstdio>
#include<cmath>
#define sqr(x) ((x)*(x))
double x[3],y[3],r[3],d[4];
//    Last Change:  2011-01-25 20:20:27
double calc(double a,double b)
{
	for(int i=0;i<3;++i)d[i]=sqrt(sqr(x[i]-a)+sqr(y[i]-b))/r[i];
	double ret=0;
	d[3]=d[0];
	for(int i=0;i<3;++i)ret+=sqr(d[i]-d[i+1]);
	return ret;
}
int main()
{
	{
		double a=0,b=0;
		for(int i=0;i<3;++i)
		{
			scanf("%lf%lf%lf",x+i,y+i,r+i);
			a+=x[i],b+=y[i];
		}
		a/=3.0,b/=3.0;
		for(double d=1;d>1e-5;)
		{
			double aa=a,bb=b,v=calc(a,b);
			double tmp;
			tmp=calc(a-d,b);
			if(tmp<v)v=tmp,aa=a-d,bb=b;
			tmp=calc(a+d,b);
			if(tmp<v)v=tmp,aa=a+d,bb=b;
			tmp=calc(a,b+d);
			if(tmp<v)v=tmp,aa=a,bb=b+d;
			tmp=calc(a,b-d);
			if(tmp<v)v=tmp,aa=a,bb=b-d;
			if(a==aa&&b==bb)d*=0.7;
			else a=aa,b=bb;
		}
		if(calc(a,b)<1e-7)printf("%.2f %.2f\n",a,b);
	}
	return 0;
}
