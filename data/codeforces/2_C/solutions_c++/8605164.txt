double x[4],y[4],r[4],s[4];
double check(double tx,double ty)
{
	double ans=0;
	int i;
	for(i=0;i<3;i++)
	s[i]=sqrt((tx-x[i])*(tx-x[i])+(ty-y[i])*(ty-y[i]))/r[i];
	for(i=0;i<3;i++)
	ans+=(s[i]-s[(i+1)%3])*(s[i]-s[(i+1)%3]);
	return ans;
}

int main()
{
	int i,j;
	double x0=0,y0=0;
	for(i=0;i<3;i++)
	{
		scanf("%lf%lf%lf",&x[i],&y[i],&r[i]);
		x0+=x[i];
		y0+=y[i];
	}
	x0/=3;
	y0/=3;
	double d=1;
	while(d>eps)
	{
		if(check(x0,y0)>check(x0+d,y0))
		x0+=d;
		else if(check(x0,y0)>check(x0-d,y0))
		x0-=d;
		else if(check(x0,y0)>check(x0,y0+d))
		y0+=d;
		else if(check(x0,y0)>check(x0,y0-d))
		y0-=d;
		else
		d*=0.5;
	}
	if(check(x0,y0)<eps)
	printf("%.5f %.5f\n",x0,y0);
}
