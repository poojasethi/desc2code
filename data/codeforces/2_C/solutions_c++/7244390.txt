#include<bits/stdc++.h>
#define eps 1e-6
#define db double
using namespace std;
db x[3],y[3],r[3],dx=0,dy=0;
db F(db dx, db dy)
{
	int i;
	db t[3],res=0;
	for(i=0;i<3;i++)
		t[i]=sqrt((dx-x[i])*(dx-x[i]) + (dy-y[i])*(dy-y[i]))/r[i];
	for(i=0;i<3;i++)
		res+=(t[i] - t[(i+1)%3])*(t[i] - t[(i+1)%3]);
	return res;
}
int main()
{
	int i;
	bool v;
	for(i=0;i<3;i++)
	{
		cin>>x[i]>>y[i]>>r[i];
		dx+=x[i];
		dy+=y[i];
	}
	dx/=3;
	dy/=3;
	for(db e=1;e>eps;v=0)
	{
		if(F(dx,dy) > F(dx+e,dy))
		{
			dx+=e;
			v=1;
		}
		else if(F(dx,dy) > F(dx-e,dy))
		{
			dx-=e;
			v=1;
		}
		else if(F(dx,dy) > F(dx,dy+e))
		{
			dy+=e;
			v=1;
		}
		else if(F(dx,dy) > F(dx,dy-e))
		{
			dy-=e;
			v=1;
		}
		if(!v)
			e*=0.5;
	}
	if(F(dx,dy) < eps)
		printf("%0.5lf %0.5lf\n",dx,dy);
	return 0;
}
