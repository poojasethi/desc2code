#include<cstdio>
#include<cstring>
#include<vector>
#include<cmath>
using namespace std;
#define L(x) (int)(x).size()
#define PB push_back
#define ll long long
#define P2(x) ((x)*(x))
#define REP(x,y) for(int x=0; x<y; x++)
double x[3],y[3],r[3],px,py,ctg[3];
double F(double fx,double fy){
	double ret = 0;
	REP(i,3)ctg[i] = sqrt(P2(fx-x[i])+P2(fy-y[i]))/r[i];
	REP(i,3)ret+= P2(ctg[i]-ctg[(i+1)%3]);
	return ret;
	}
int main(){
	REP(i,3) scanf("%lf%lf%lf",&x[i],&y[i],&r[i]);
	REP(i,3){
		px+= x[i]/3;
		py+= y[i]/3;
		}
	bool f=false;
	for(double i=1 ;i>1e-6; f=false){
		if(F(px,py)>F(px+i,py)){px+=i;f=true;}
		if(F(px,py)>F(px-i,py)){px-=i;f=true;}
		if(F(px,py)>F(px,py+i)){py+=i;f=true;}
		if(F(px,py)>F(px,py-i)){py-=i;f=true;}
		if(!f)i*=0.75;
		}
	if(F(px,py)<1e-6)printf("%.5lf %.5lf\n",px,py);
	}
