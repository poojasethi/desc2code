#include <stdio.h>
#include <math.h>

const double EPS = 1e-7;

#define sqr(a) ((a)*(a))

double X[3], Y[3], R[3], cx, cy;

double F(double x, double y){
    double r = 0, A[3];
    for (int i=0; i<3; i++) A[i] = sqrt(sqr(X[i]-x) + sqr(Y[i]-y))/R[i];
    for (int i=0; i<3; i++) r += sqr(A[i]-A[(i+1)%3]);
    return r;
}

int main(){
    cx = cy = 0;
    for(int i=0; i<3; i++){
        scanf("%lf%lf%lf",X+i,Y+i,R+i);
        cx += X[i];
        cy += Y[i];
    }
    cx/=3.0; cy/=3.0;
    for (double d = 1.0; d>1e-5;){
        bool f = 0;
        if (F(cx-d,cy)<F(cx,cy)) cx-=d, f=1;
        else if (F(cx+d,cy)<F(cx,cy)) cx+=d, f=1;
        if (F(cx,cy-d)<F(cx,cy)) cy-=d, f=1;
        else if (F(cx,cy+d)<F(cx,cy)) cy+=d, f=1;
        if (!f) d*=0.7;
    }
    if (F(cx,cy)<EPS) printf("%.5lf %.5lf",cx,cy);
    return 0;
}
