#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

const double eps=1e-5;
double x[4],y[4],r[4];

double gao(double dx, double dy){
    double l[4],ret=0;
    for(int i=1;i<=3;i++) l[i]=sqrt(pow(x[i]-x[0]-dx,2)+
                                    pow(y[i]-y[0]-dy,2))/r[i];
    l[0]=l[3];
    for(int i=1;i<=3;i++) ret+=pow(l[i]-l[i-1],2);
    return ret;
}

int main(){
    x[0]=y[0]=0;
    for(int i=1;i<=3;i++){
        scanf("%lf%lf%lf",&x[i],&y[i],&r[i]);
        x[0]+=x[i]/3;
        y[0]+=y[i]/3;
    }
    double d=1.0;
    while(d>eps){
        int u=1;
        if(gao(d,0)<gao(0,0)) x[u=0]+=d; else
            if(gao(-d,0)<gao(0,0)) x[u=0]-=d;
        if(gao(0,d)<gao(0,0)) y[u=0]+=d; else
            if(gao(0,-d)<gao(0,0)) y[u=0]-=d;
        if(u) d*=0.75;
    }
    if(gao(0,0)<eps) printf("%.5f %.5f\n",x[0],y[0]);
}
