#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
struct qry
{
    int a,b,id;
    void input(int x)
    {
        scanf("%d%d",&a,&b);
        id=x;
    }
    bool operator<(const qry&tt)const
    {
        return b<tt.b;
    }
}q[200005];
int n,ans[200005];
int stk[200005],top=0;
void solve(int l,int r)
{
    int mid=l+r>>1;
    if(l==r)
        return;
    solve(l,mid);
    for(int i=l;i<=mid;++i)
        stk[top++]=q[i].a;
    sort(stk,stk+top);
    int tot=mid-l+1;
    for(int i=mid+1;i<=r;++i)
        ans[q[i].id]+=tot-(upper_bound(stk,stk+top,q[i].a)-stk);
    top=0;
    solve(mid+1,r);
}
int main()
{
    int i,j;
    scanf("%d",&n);
    for(i=0;i<n;++i)
        q[i].input(i+1);
    sort(q,q+n);
    solve(0,n-1);
    for(i=1;i<=n;++i)
        printf("%d\n",ans[i]);
    return 0;
}
