#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

#define ll long long
#define inf 1e9
#define eps 1e-10
#define md
#define N 400010
using namespace std;
struct PP { int l,r,id;}p[N];
int c[N],ans[N];
int n;
bool cmp_r(PP a,PP b) { return a.r<b.r;}
bool cmp_l(PP a,PP b) { return a.l==b.l?a.r<b.r:a.l>b.l;}
void lisanhua()
{
	sort(p+1,p+n+1,cmp_r);
	for (int i=1;i<=n;i++) p[i].r=i;
	sort(p+1,p+n+1,cmp_l);
}

int qsum(int x)
{
	int ans=0;
	for (;x;x-=x&(-x)) ans+=c[x];
	return ans;
}
void add(int x) { for (;x<=n;x+=x&(-x)) c[x]++;}

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d%d",&p[i].l,&p[i].r);
		p[i].id=i;
	}
	lisanhua();
	for (int i=1;i<=n;i++)
	{
		ans[p[i].id]=qsum(p[i].r);
		add(p[i].r);
	}
	for (int i=1;i<=n;i++) printf("%d\n",ans[i]);
	return 0;
}
