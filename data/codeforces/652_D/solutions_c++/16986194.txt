#include<cstdio>
#include<algorithm>
using namespace std;
#define maxn 200010
int n,l[maxn],R[maxn],od[maxn],b[maxn],ans[maxn];
bool cmp(const int &p,const int &q)
{
	return l[p]<l[q];
}
//sort it by the right axis, in range l..r
void cdq(int l,int r)
{
	if(l==r) return;
	int mid=(l+r)>>1;
	cdq(l,mid);
	cdq(mid+1,r);
	int lp=mid,rp=r,top=r+1;
	while(lp>=l && rp>mid)
	{
		int cl=od[lp],cr=od[rp];
		if(R[cl]>R[cr])
		{
			ans[cl]+=rp-mid;
			b[--top]=cl;
			lp--;
		}
		else
		{
			b[--top]=cr;
			rp--;
		}
	}
	while(lp>=l)
	{
		int cl=od[lp];
		b[--top]=cl; lp--;
	}
	while(rp>mid)
	{
		int cr=od[rp];
		b[--top]=cr; rp--;
	}
	for(int i=l; i<=r; i++) od[i]=b[i];
}
int main()
{
	scanf("%d",&n);
	for(int i=1; i<=n; i++)
	{
		od[i]=i;
		scanf("%d%d",&l[i],&R[i]);
	}
	sort(od+1,od+1+n,cmp);
	cdq(1,n);
	for(int i=1; i<=n; i++) printf("%d\n",ans[i]);
	return 0;
}
