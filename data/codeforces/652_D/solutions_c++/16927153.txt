#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cstdlib>

#define maxn 400010

using namespace std;
struct Node{
	int l,r,id;
	bool operator<(const Node &t)const{return l<t.l;}
}A[maxn];
int c[maxn];
int hash[maxn],tot;
void add(int x){
	for(;x<=tot;x+=x&(-x))c[x]++;
}
int query(int x){int ans=0;
	for(;x;x-=x&(-x))ans+=c[x];
	return ans;
}
int n;
int Ans[maxn];
void Work(){
	for(int i=n;i;i--){
		Ans[A[i].id]=query(A[i].r);
		add(A[i].r);
	}
	for(int i=1;i<=n;i++)printf("%d\n",Ans[i]);
}

void Init(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d%d",&A[i].l,&A[i].r),A[i].id=i,hash[++tot]=A[i].l,hash[++tot]=A[i].r;
	sort(hash+1,hash+tot+1);
	for(int i=1;i<=n;i++)A[i].l=lower_bound(hash+1,hash+tot+1,A[i].l)-hash,A[i].r=lower_bound(hash+1,hash+tot+1,A[i].r)-hash;
	sort(A+1,A+n+1);
}

int main(){
	Init();
	Work();
	return 0;
}
