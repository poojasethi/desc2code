#include<cstdio>
#include<queue>
#include<algorithm>
#include<map>
#define mxidx 524288
using namespace std;
map<int,int> mapping;
int bit[524289];
int pnt[400001];
int ans[200001];
pair< pair<int,int>, int> line[200001];

int getbit(int idx){
	int res=0;
	while(idx>0){
		res+=bit[idx];
		idx-= idx & (-idx);
	}
	return res;
}
int setbit(int idx){
	while(idx^mxidx){
		bit[idx]++;
		idx+=idx & (-idx);
	}
	return 0;
}
int main(){
	int n,a,b;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d %d",&a,&b);
		line[i]=make_pair(make_pair(b,a),i);
		pnt[i*2]=a;
		pnt[i*2+1]=b;
	}
	sort(pnt,pnt+(2*n));
	for(int i=0;i<2*n;i++){
		mapping[pnt[i]]=i+1;
	}
	for(int i=0;i<n;i++){
		line[i].first.first=mapping[line[i].first.first];
		line[i].first.second=mapping[line[i].first.second];
	}
	sort(line,line+n);
	for(int i=0;i<n;i++){
		ans[line[i].second]=i-getbit(line[i].first.second-1);
		setbit(line[i].first.second);
	}
	for(int i=0;i<n;i++) printf("%d\n",ans[i]);
}
