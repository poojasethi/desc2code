#include <cstdio>
#include <algorithm>
#define maxn 200010
using namespace std;
struct cool{int l,r,id;}a[maxn];
int ans[maxn],g[maxn],gg[maxn],d[maxn],n;
void update(int x,int y)
{
	for (;x <= n;x += x&-x)
		g[x] += y;
}
int query(int x)
{
	int y = 0;
	for (;x;x -= x&-x)
		y += g[x];
	return y;
}
bool cmp(cool a,cool b)
{
	return a.l==b.l?a.r>b.r:a.l<b.l;
}
int main()
{
	scanf("%d",&n);
	for (int i = 1;i <= n;i++)
		scanf("%d%d",&a[i].l,&a[i].r),a[i].id = i;
	sort(a+1,a+n+1,cmp);
	for (int i = 1;i <= n;i++)
		gg[i] = a[i].r;
	sort(gg+1,gg+n+1);
	for (int i = 1;i <= n;i++)
		d[i] = lower_bound(gg+1,gg+n+1,a[i].r)-gg;
	for (int i = n;i >= 1;i--)
	{
		int t = query(d[i]);
		ans[a[i].id] = t;
		update(d[i],1);
	}
	for (int i = 1;i <= n;i++)
		printf("%d\n",ans[i]);
	return 0;
}
