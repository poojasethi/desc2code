#include<bits/stdc++.h>
#define f first
#define s second
#define l (p<<1)
#define r (p<<1|1)
#define mid ((x+y)>>1)
using namespace std;

const int N = 200005;
int n, ans[N], v[N], bit[N];
bool ok[N];
pair<int, pair<int, int> > a[N];
map<int, int> mp;

void update(int idx, int v) {
	while(idx <= n) {
		bit[idx] += v;
		idx += idx & -idx;
	}
}

int get(int idx) {
	int ret = 0;
	while(idx > 0) {
		ret += bit[idx];
		idx -= idx & -idx;
	}
	return ret;
}

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i) {
		scanf("%d%d", &a[i].f, &a[i].s.f);
		a[i].s.s = i;
		v[i] = a[i].s.f;
	}
	sort(v, v+n);
	for(int i=0 ; i<n ; ++i)
		mp[v[i]] = i+1;
	memset(ok, true, sizeof ok);
	for(int i=1 ; i<=n ; ++i)
		update(i, 1);
	sort(a, a+n);
	for(int idx, i=0 ; i<n ; ++i) {
		idx = mp[a[i].s.f];
		ans[a[i].s.s] = get(idx-1);
		update(idx, -1);
	}
	for(int i=0 ; i<n ; ++i)
		printf("%d\n", ans[i]);
	return 0;
}
