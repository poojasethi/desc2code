#include<bits/stdc++.h>
#include<ext/pb_ds/assoc_container.hpp>
#include<ext/pb_ds/tree_policy.hpp>
using namespace std;
using namespace __gnu_pbds;
int n,ans[200000];
struct seg {
 int s,e,id;
 void read(int i) {
  scanf("%d%d",&s,&e);id=i;
 }
 bool operator<(const seg&b) {
  return e<b.e;
 }
}sg[200000];

tree<int,char,less<int>,rb_tree_tag,tree_order_statistics_node_update> st;
int main() {
 scanf("%d",&n);for(int i=0;i<n;++i)(sg+i)->read(i);
 sort(sg,sg+n);
 for(int i=0;i<n;++i) {
  ans[sg[i].id]=i-st.order_of_key(sg[i].s);
  st[sg[i].s];
 }
 for(int i=0;i<n;++i)printf("%d\n",ans[i]);
}
