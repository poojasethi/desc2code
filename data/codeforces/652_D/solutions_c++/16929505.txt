#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<cstdlib>
#define MAXN 200010
using namespace std;
struct B
{
    int x,y,num;
}b[MAXN];

int a[MAXN],ans[MAXN];
int n;

bool cmp1(B p,B q)
{
    return p.y<q.y;
}

bool cmp2(B p,B q)
{
    if (p.x==q.x) return p.y<q.y;
    else return p.x<q.x;
}

void add(int x)
{
    for (int i=x; i<=n; i+=i&(-i))
        a[i]++;
}

int sum(int x)
{
    int s=0;
    for (int i=x; i>0; i-=i&(-i))
        s+=a[i];
    return s;
}

int main()
{
    //freopen("xl_in.txt","r",stdin);
    scanf("%d",&n);
    for (int i=1; i<=n; i++)
    {
        scanf("%d%d",&b[i].x,&b[i].y);
        b[i].num=i;
    }

    sort(b+1,b+1+n,cmp1);

    for (int i=1; i<=n; i++)
        b[i].y=i;

    sort(b+1,b+1+n,cmp2);

    for (int i=n; i>=1; i--)
    {
        ans[b[i].num]=sum(b[i].y-1);
        add(b[i].y);
    }

    for (int i=1; i<=n; i++)
        printf("%d\n",ans[i]);
    return 0;
}
