#include <cstdio>
#include <algorithm>
#include <iostream>
#include <vector>
#include <queue>
#include <stack>
#include <cstring>
#include <map>
#define maxn 200005
#define fi first
#define se second
#define mp make_pair
#define pb push_back
#define pii pair < int,int >
using namespace std;
map < int,int > val;
pair < pii,int > d[maxn];
int n,m,i,j,k,a,b,c,tree[maxn],syc,ans[maxn];
int query(int x){
	int res=0;
	for(;x;x-=x&-x)
	res+=tree[x];
	return res;
}
void update(int x){
	 for(;x<=n;x+=x&-x)
	 tree[x]+=1;
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d %d",&d[i].fi.fi,&d[i].fi.se);
		val[d[i].fi.se]=1;
		d[i].se=i;
	}
	sort(d+1,d+n+1);
	map < int ,int > :: iterator it;
	it=val.begin();
	while(it!=val.end()){
		val[it->fi]=++syc;
		it++;
	}
	for(i=n;i;i--){
		int l=val[d[i].fi.se];
		ans[d[i].se]=query(l-1);
		update(l);
	}
	for(i=1;i<=n;i++)
	printf("%d\n",ans[i]);
	return 0;
}
