#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
const int MAXN=200005;
int n,N,BIT[MAXN*2],coor[MAXN*2],ans[MAXN];
pair<pair<int,int>,int>  vals[MAXN];
void update(int index)
{
    for (;index<MAXN*2;index+=index&-index)BIT[index]++;
}
int query(int index)
{
    int s=0;
    for (;index>0;index-=index&-index)s+=BIT[index];
    return s;
}
int main()
{
    //freopen("DATA.txt","r",stdin);
    scanf("%i",&n);
    for (int i=0;i<n;i++)
    {
        scanf("%i%i",&vals[i].first.first,&vals[i].first.second);
        vals[i].second=i;
        coor[N++]=vals[i].first.first;
        coor[N++]=vals[i].first.second;
    }
    sort(coor,coor+N);
    N=unique(coor,coor+N)-coor;
    sort(vals,vals+n);
    for (int i=n-1;i>=0;i--)
    {
        ans[vals[i].second]=query(lower_bound(coor,coor+N,vals[i].first.second)-coor+1);
        update(lower_bound(coor,coor+N,vals[i].first.second)-coor+1);
    }
    for (int i=0;i<n;i++)printf("%i\n",ans[i]);
}
