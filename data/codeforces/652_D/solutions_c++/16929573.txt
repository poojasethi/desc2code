#include<bits/stdc++.h>
using namespace std;
struct node
{
    int x,y,xu;
}p[200005],h[200005];
int n,i,k,maxn,f[200005],t[400005],ans[200005];
bool cmp(const node&a,const node&b)
{
    return a.x<b.x;
}
void update(int x)
{
    while(x<=maxn)
    {
        t[x]++;
        x+=x&-x;
    }
}
int solve(int x)
{
    int s=0;
    while(x>0)
    {
        s+=t[x];
        x-=x&-x;
    }
    return s;
}
int main()
{
    cin>>n;
    maxn=-1e9-1;
    for(i=1;i<=n;i++)
    {
        scanf("%d%d",&p[i].x,&p[i].y);
        p[i].xu=i;
        h[i].x=p[i].y;
        h[i].xu=i;
    }
    sort(p+1,p+n+1,cmp);
    sort(h+1,h+n+1,cmp);
    h[0].x=-1e9-1;
    int k=0;
    for(i=1;i<=n;i++)
    {
        if(h[i].x!=h[i-1].x) k++;
        f[h[i].xu]=k;
    }
    maxn=k;
    for(i=n;i>=1;i--)
    {
        ans[p[i].xu]=solve(f[p[i].xu]);
        update(f[p[i].xu]);
    }
    for(i=1;i<=n;i++)
        printf("%d\n",ans[i]);
    return 0;
}
