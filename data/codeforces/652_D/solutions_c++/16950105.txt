#include<bits/stdc++.h>
using namespace std;
#define f first
#define s second
int bit[400005],fin[400005];
vector<int> v;
int j;
map<int,int> mp;
pair<pair<int,int>,int> p[200005];
void update(int i,int v)
{
	while(i<=j)
	bit[i]+=v,i+=i&(-i);
}
int ans(int i)
{
	int sum=0;
	while(i)
	{
		sum+=bit[i];
		i-=i&(-i);
	}
	return sum;
}
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>p[i].f.f>>p[i].f.s;
		p[i].s=i;
		v.push_back(p[i].f.f);
		v.push_back(p[i].f.s);
	}
   sort(v.begin(),v.end());
   j=1;
   for(int i=0;i<v.size();i++)
   {
   	  mp[v[i]]=j++;
   }
   for(int i=1;i<=n;i++)
   {
   	p[i].f.f=mp[p[i].f.f];
   	p[i].f.s=mp[p[i].f.s];
   }
   sort(p+1,p+n+1);
   for(int i=n;i>=1;i--)
   {
   	fin[p[i].s]=ans(p[i].f.s);
   	update(p[i].f.s,1);
   }
   for(int i=1;i<=n;i++)
   cout<<fin[i]<<endl;
}
