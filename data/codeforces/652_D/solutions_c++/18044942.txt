#include<iostream>
#include<algorithm>
#include<map>

using namespace std;

map<long long,long long>m1,m2,m3;

long long n,fen[200005],ans[200005],b[200005];

void update(long long pos,long long val){

    for(;pos<=n;pos+= pos&-pos){
        fen[pos]+=val;
    }


}

long long getsum(long long pos){

    long long sum=0;
    for(;pos>0;pos-= pos&-pos){
        sum+=fen[pos];
    }
    return sum;

}

int main(){

    ios_base::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    long long i,a[200005];
    cin>>n;
    for(i=1;i<=n;i++){
        cin>>a[i]>>b[i];
        m1[a[i]]=b[i];
        m2[a[i]]=i;
    }
    sort(a+1,a+n+1);sort(b+1,b+n+1);
    for(i=1;i<=n;i++){
        m3[b[i]]=i;
        update(i,1);
    }
    for(i=1;i<=n;i++){
        ans[m2[a[i]]]=getsum(m3[m1[a[i]]]-1);
        update(m3[m1[a[i]]],-1);
    }
    for(i=1;i<=n;i++)cout<<ans[i]<<"\n";
    return 0;







}
