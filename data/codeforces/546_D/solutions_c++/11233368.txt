#include <bits/stdc++.h>
using namespace std;
const int N = 5e6 + 1;
int sum[N]={0} , t , a , b;
int main(){
    for (int i = 2 ; i < N ; sum[i] += sum[i-1] , ++i)
        if (!sum[i])
            for (long long j = i ; j < N ; j *= i)
                for (int k = j ; k < N ; sum[k] ++ , k += j);
    scanf("%d",&t);
    while(~scanf("%d %d",&a,&b))
        printf("%d\n",sum[a] - sum[b]);
}
