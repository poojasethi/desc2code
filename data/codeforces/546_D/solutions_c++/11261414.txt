#include <cstdio>
const int maxn = 5000000;

int d[maxn + 10];

int main()
{
    for(int i = 2; i <= maxn; i++) if(!d[i])
        for(int j = i; j <= maxn; j += i) d[j] = d[j / i] + 1;
    for(int i = 2; i <= maxn; i++) d[i] += d[i - 1];

    int t; scanf("%d", &t);
    while( t-- )
    {
        int b, a; scanf("%d%d", &b, &a);
        printf("%d\n", d[b] - d[a]);
    }

    return 0;
}
