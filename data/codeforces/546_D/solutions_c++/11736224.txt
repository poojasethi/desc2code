#include <stdio.h>
const int N = 5000000;
int c[N+4];
int main(){
  for(int i=2;i<=N;++i)if(!c[i])for(int j=i;j<=N;j+=i)c[j]=i;
  for(int i=2;i<=N;++i)c[i]=1+c[i/c[i]];
  for(int i=1;i<=N;++i)c[i]+=c[i-1];
  int n,a,b;
  scanf("%d",&n);
  while(n-->0&&2==scanf("%d%d",&a,&b))printf("%d\n",c[a]-c[b]);
}
