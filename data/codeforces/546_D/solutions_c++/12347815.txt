#include <cstdio>
#include <cstring>

const int maxn = 5000010;
int t, a, b, tab[maxn];

int main() {
	memset(tab, 0, sizeof(tab));
	for(int i=2; i <= maxn; i++) {
		if(!tab[i])
			for(int j=1; i*j <= maxn; j++)
				for(int k=i*j; k%i == 0; k /= i)
					tab[i*j]++;
		tab[i] += tab[i-1];
	}
	for(scanf("%d", &t); t-- && scanf("%d%d", &a, &b); printf("%d\n", tab[a]-tab[b]));
	return 0;
}
