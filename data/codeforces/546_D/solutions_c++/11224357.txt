#include <cstdio>
const int MAX = 5000005;
int dv[MAX];
int main()
{
	for (int i = 2; i < MAX; i++)
		if (!dv[i])
			for (int j = i; j < MAX; j += i)
			{
				int tmp = j;
				while (tmp % i == 0)
				{
					tmp /= i;
					dv[j]++;
				}
			}
	for (int i = 1; i < MAX; i++)
		dv[i] += dv[i - 1];
	int q;
	scanf("%d", &q);
	while (q--)
	{
		int l, r;
		scanf("%d %d", &r, &l);
		printf("%d\n", dv[r] - dv[l]);
	}
	return 0;
}
