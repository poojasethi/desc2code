#include <bits/stdc++.h>
using namespace std;
#define mxx 5000015
int A[mxx];
int main() 
{
    int i,t,n,m,j;
    for(i=2;i<mxx;i++){
        if(A[i]==0){
            for(j=i;j<mxx;j+=i){
                 A[j]=A[j/i]+1;
            }
        }
    }
    for(i=1;i<mxx;i++)
    A[i]=A[i-1]+A[i];
    scanf("%d",&t);
    while(t--){
        scanf("%d %d",&n,&m);
        printf("%d\n",A[n]-A[m]);
    }
    return 0;
}
