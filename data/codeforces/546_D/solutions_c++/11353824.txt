#include <iostream>
#include <cstdio>
using namespace std;
int a[5000001],b[5000001],sum[5000001];
int main(){
int t;
cin>>t;
for(long long i=2;i<=5000001;i++){
if(a[i]==0)a[i]=i;
if(a[i]==i){
for(long long j=i*i;j<=5000001;j+=i){
if(a[j]==0)a[j]=i;
}
}
b[i]=b[i/a[i]]+1;
sum[i]=(b[i]+sum[i-1]);
}
int x,y;
for(int i=1;i<=t;i++){
scanf("%d %d",&x,&y);
printf("%d \n",sum[x]-sum[y]);
}
return 0;
}
