#include <cstdio>
using namespace std;
const int maxa=5e6+1 ;
int dsi[maxa] ;
int main()
{
    for(int i=2;i<maxa;i++)
	if(dsi[i]==0)
	{
	    dsi[i]=1 ;
	    for(int j=i*2;j<maxa;j+=i)
		dsi[j]=i ;
	}
	else
	    dsi[i]=dsi[i/dsi[i]]+1 ;
    for(int i=2;i<maxa;i++)
	dsi[i]+=dsi[i-1] ;
    int n;
    scanf("%d",&n) ;
    int a,b ;
    for(int i=0;i<n;i++)
    {
	scanf("%d %d",&a,&b) ;
	printf("%d \n",dsi[a]-dsi[b]) ;
    }
}
