#include<cstdio>
#include<iostream>
using namespace std;

int s[5000001];
int main(){
    int i,j,t,a,b;
    for(i=2;i<=5000000;i++){
        if(s[i]==0) s[i]=1;
        for(j=2;j<=i && 1LL*j*i<=5000000;j++)
               s[j*i]=s[j]+s[i];
    }

    for(i=2;i<=5000000;i++) s[i]=s[i-1]+s[i];
    scanf("%d",&t);
    while(t--)
    {  scanf("%d%d",&a,&b);
       printf("%d\n",s[a]-s[b]);
    }
    return 0;
}
