#include <bits/stdc++.h>
#define f(i,a,b) for(i=a;i<b;i++)
#define S 5000001
int arr[S];
using namespace std;
int main()
{
    int n,i,j,a,b;
    scanf("%d",&n);
    f(i,2,S){
        if(!arr[i])for(j=i;j<S;j+=i)arr[j]=arr[j/i]+1;
    }
    f(i,2,S)arr[i]+=arr[i-1];
    f(i,0,n){
        scanf("%d%d",&a,&b);
        printf("%d\n",arr[a]-arr[b]);
    }
}
