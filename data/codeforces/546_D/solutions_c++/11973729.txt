#include <iostream>
#include <cstdio>
using namespace std;

int isPrime[5000001];

int main() {
	for(int i=2; i<5000001; i++)
	{
		if(isPrime[i]==0)
		{
			for(int j=1;i*j<=5000000;j++)
				for(int k=i*j;k%i==0;k/=i)
					isPrime[i*j]++;
		}
		isPrime[i]+=isPrime[i-1];
	}
	int cases;
	scanf("%d", &cases);
	while(cases--)
	{
		int a,b;
		scanf("%d %d", &a,&b);
		printf("%d\n",isPrime[a]-isPrime[b]);
	}
}
