#include <iostream>
#include <cstdio>
using namespace std;
int a[5000001], t, x, y;
int main()
{
	for (int i=2; i<=5000000; i++) {
		a[i]=1;
		for (int j=2; j*j<=i; j++)
			if (i%j==0) {
				a[i]+=a[i/j];
				break;
			}
	}
	for (int i=1; i<=5000000; i++) a[i]+=a[i-1];
	scanf("%d", &t);
	while (t--) {
		scanf("%d%d", &x, &y);
		printf("%d\n", a[x]-a[y]);
	}
	return 0;
}
