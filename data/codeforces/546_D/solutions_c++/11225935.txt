#include <cstdio>
int least[5000001], cnt[5000001];
int main() {
	for(int i = 2; i <= 5000000; ++i) 
		for(int j = i; j <= 5000000; j += i)
			if(!least[j]) least[j] = i;
	for(int i = 2; i <= 5000000; ++i)
		cnt[i] = cnt[i / least[i]] + 1;
	for(int i = 1; i <= 5000000; ++i)
		cnt[i] += cnt[i - 1];
	int t;
	scanf("%d", &t);
	while(t--) {
		int a, b;
		scanf("%d %d", &a, &b);
		printf("%d\n", cnt[a] - cnt[b]);
	}
}
