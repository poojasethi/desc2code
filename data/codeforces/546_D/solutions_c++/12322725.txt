#include <stdio.h>
#define MAXN 5000001

long long sum[MAXN];

int main ()
{
	int i, k, t;
	long long j;
	for (i=2; i<MAXN; sum[i] += sum[i-1], i++)
		if (!sum[i] && (j=i))
			while (j < MAXN)
			{
				for (k=j; k<MAXN; k+=j)
					sum[k]++;
				j *= i;
			}
	scanf ("%d", &t);
	while (t-- && scanf ("%d%d", &i, &k))
		printf ("%I64d\n", sum[i] - sum[k]);
	return 0;
}
