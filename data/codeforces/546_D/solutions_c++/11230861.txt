#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll ps[5000010];
int main()
{
	for(int i=2;i<5000000;i++)
		if(ps[i]==0)
			for(int j=1;i*j<=5000000;j++)
				for(int k=i*j;k%i==0;k/=i,ps[i*j]++);
	for(int i=1;i<=5000000;i++)
		ps[i]+=ps[i-1];
	int n;
	cin>>n;
	while(n--)
	{
		int a,b;
		scanf("%d%d",&a,&b);
		printf("%lld\n",ps[a]-ps[b]);
	}
}
