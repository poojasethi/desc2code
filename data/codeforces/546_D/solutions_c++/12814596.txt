#include<stdio.h>
#define N 5000000
int n,a,b,i,prime[500001],d[5000001],p,s,j,dp[N+1];
long long r;
int main(){
	for(i=2;i<=N;i++){
		if(d[i]==0){
			prime[p++]=i;
			r=i;
			while(r<=N){
				for(j=r;j<=N;j+=r)d[j]++;
				r*=i;
			}
		}
	}
	for(i=2;i<=N;i++)
		dp[i]=dp[i-1]+d[i];
	scanf("%d",&n);
	for(i=0;i<n;i++){
		s=0;
		scanf("%d%d",&a,&b);
		printf("%d\n",dp[a]-dp[b]);
	}
}
