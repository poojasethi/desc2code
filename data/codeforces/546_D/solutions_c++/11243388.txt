#include<bits/stdc++.h>
using namespace std;
#define M2H 5000050
long dp[M2H],i;
int main()
{
    int t; cin>>t;
    
    for(i=2;i<=M2H;i++)
    {
    
    if(dp[i]==0)for(long j=i;j<=M2H;j+=i)dp[j]=1+dp[j/i];
    
    }
    //for(int i=1;i<=20;i++)cout<<i<<' '<<dp[i]<<endl;
    for(long i=1;i<=M2H;i++)dp[i]+=dp[i-1];
while(t--)
{
    int a,b;
    scanf("%d%d",&a,&b),printf("%d\n",dp[a]-dp[b]);

}
}
