#include <bits/stdc++.h>
#define l 5555555
int d[l];
int main(void) {
	for(int i=2;i<l;i++)
		if(d[i]==0)
			for(int j=i;j<=l;j+=i)
				d[j]=d[j/i]+1;
	for(int i=1;i<=l;i++)
		d[i]+=d[i-1];
		int t;
	scanf("%d",&t);
	while(t--)
	{
	int a,b;
	scanf("%d %d",&a,&b);
	printf("%d \n",(d[a]-d[b]));
	}
	return 0;
}
