#include <bits/stdc++.h>
using namespace std;
int ans[5000002],sum[5000002];
int main ()
{
int a,b,c,d=0,i,j;
for (i=2;i<=5000000;i++)
{
int x=i,f=0;
while (x>1)
{
int h=0;
if (ans[x]!=0){f+=ans[x];x=1;break;}
for (j=2;j<=sqrt(x);j++)
{
if (x%j==0){x=x/j;f++;h=1;break;}
}
if (h==0){f=1;break;}
}
ans[i]=f;
sum[i]=sum[i-1]+ans[i];
}
scanf("%d",&a);
for (i=1;i<=a;i++)
{
scanf("%d%d",&b,&c);
printf("%d\n",sum[b]-sum[c]);
}
}
