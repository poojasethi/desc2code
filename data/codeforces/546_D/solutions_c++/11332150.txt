#include <bits/stdc++.h>
using namespace std;
const int maxn=5000000;
int d[maxn+228],i,j,n,m;
int main()
{
	cin>>n;                
	for(i=2,d[i]=1;i<=maxn;i++,d[i]=1)
	{
		for(j=2;j*j<=i;j++)
		{
			if(i%j==0)
			{
				d[i]+=d[i/j];
				break;
			}
		}
	}
	for(i=1;i<=maxn;i++)
	{
		d[i]+=d[i-1];
	}
	while(n--)
	{
		scanf("%d",&i);
		scanf("%d",&j);
		printf("%d\n",d[i]-d[j]);
	}
	return 0;
}
