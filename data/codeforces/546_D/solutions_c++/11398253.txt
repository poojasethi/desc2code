#include<iostream>
#include<stdio.h>
using namespace std;

int ps[5000010];
int main()
{
    for(int i=2;i<5000000;i++)
        if(ps[i]==0)
            for(int j=i;j<=5000000;j+=i)
                ps[j]=ps[j/i]+1;
    for(int i=1;i<=5000000;i++)
        ps[i]+=ps[i-1];
    int n;
    cin>>n;
    while(n--)
    {
        int a,b;
        scanf("%d%d",&a,&b);
        printf("%d\n",ps[a]-ps[b]);
    }
}
