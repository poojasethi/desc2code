#include<algorithm>
#include<iostream>
#include<cstdio>
using namespace std;
int s[5000100];
int main()
{
	for(int i=2;i<=5000000;i++)
	{
		for(int j=2;j*j<=i;j++)
		{
			if(i%j==0)
			{
				s[i]=s[i/j]+1;
				break;
			}
		}
		if(s[i]==0)s[i]=1;
	}
	for(int i=2;i<=5000000;i++)s[i]+=s[i-1];
	int n;
	cin>>n;
	while(n--)
	{
		int a,b;
		scanf("%i%i",&a,&b);
		printf("%i\n",s[a]-s[b]);
	}
	return 0;
}
