#include<stdio.h>

int num[5000001] = {0};

int main()
{
	for(int i = 2; i <= 5000000; i++)
	{
		if(num[i] == 0)
            for(int j = 1; i * j <= 5000000; j++)
                for(int k = i * j; k % i == 0; k /= i)
                    num[i * j]++;
		num[i] += num[i - 1];
	}

    int t;

	scanf("%d", &t);
	while(t--)
	{
	    int a, b;
		scanf("%d%d", &a, &b);
		printf("%d\n", num[a] - num[b]);
	}

	return 0;
}
