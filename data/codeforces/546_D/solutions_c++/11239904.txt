#include <cstdio>
#include <vector>
using namespace std;

int num[5000005];
void procPrime(){
	for(int i=2;i<=5000000;++i){
		if( num[i]!=0 ) continue;
		for(int j=i;j<=5000000;j+=i){
			num[j] = num[j/i] +1;
		}
	}
	for(int i=2;i<=5000000;++i)
		num[i] += num[i-1];
}

int main(){
	procPrime();
	int t;
	scanf(" %d\n",&t);
	int a,b;
	while( t-- ){
		scanf(" %d%d",&a,&b);
		printf("%d\n",num[a]-num[b]);
	}
}
