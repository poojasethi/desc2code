#include<cstdio>
#define MAX 5000000
using namespace std;
int t,a,b;
int p[MAX+5];
long long s[MAX+5];
int main()
{
	for (int i=1;i<=MAX;i++)
	p[i]=i;
	for (int i=2;i<=MAX;i++)
	if (p[i]!=1)
	{
		t=p[i];
		for (int j=i;j<=MAX;j+=i)
		s[j]++,p[j]/=t;
	}
	for (int i=2;i<=MAX;i++)
	s[i]+=s[i-1];
	scanf("%d",&t);
	while (t--)
	{
		scanf("%d%d",&a,&b);
		printf("%d\n",s[a]-s[b]);
	}
}
