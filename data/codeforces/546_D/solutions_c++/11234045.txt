#include <cstdio>
const int maxn=5000001;
int f[maxn];
int main()
{
    for(int i=2;i<maxn;++i) {
        if (!f[i]) {
            f[i]=1;
            for(int j=i+i;j<maxn;j+=i)
                for(int k=j;k%i==0;k/=i) f[j]++;
        }
        f[i]+=f[i-1];
    }

    int t; scanf("%d", &t);
    while (t--) {
        int a, b; scanf("%d%d", &a, &b);
        printf("%d\n", f[a]-f[b]);
    }
}
