#include<stdio.h>
#include<string.h>
#define N 5000001
int ct[N],fac[N];
void ss(){
	memset(fac,-1,sizeof(fac));
	ct[0]=ct[1]=0;
	for(int i=2;i<N;i++){
		if(fac[i]==-1){
			for(int j=i;j<N;j+=i)
				fac[j]=i;
		}
		int bf=i/fac[i];
		ct[i]=ct[i-1]+ct[bf]-ct[bf-1]+1;
	}
}
int main(){
	ss();
	int ca;
	scanf("%d",&ca);
	while(ca--){
		int a,b;
		scanf("%d%d",&a,&b);
		printf("%d\n",ct[a]-ct[b]);
	}
}
