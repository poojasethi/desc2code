# include <cstdio>
# include <iostream>
using namespace std;
# define N 5000005
int dp[N];
int main() {
	int t, a, b;
	scanf("%d", &t);
	for (int i = 2; i < N; i++)
		if (!dp[i])
			for (int j = i; j < N; j += i)
				dp[j] = dp[j / i] + 1;
	for (int i = 2; i < N; i++)
		dp[i] += dp[i - 1];
	while (t--) {
		scanf("%d%d", &a, &b);
		printf("%d\n", dp[a] - dp[b]);
	}
	return 0;
}
