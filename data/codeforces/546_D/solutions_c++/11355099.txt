#include<bits/stdc++.h>
using namespace std;
#define N 5000001
int main()
{
	int t,a,b;
	cin>>t;
	vector<int> s(N);
	for(int i=2;i<N;i++)
	{
		if(!s[i]) for(int j=i;j<N;j+=i) s[j]=s[j/i]+1;
	}
	for(int i=2;i<N;i++) s[i]+=s[i-1];
	for(int i=0;i<t;i++)
	{
		scanf("%d %d",&a,&b);
		printf("%d\n",s[a]-s[b]);
	}
	return 0;
}
