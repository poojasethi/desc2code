#! /usr/bin/python

from sys import stdin

if __name__=='__main__':
	stdin.readline()
	digits=stdin.readline().strip()
	rem=len(digits)%2
	if rem==0:
		arr=[digits[i:i+2] for i in xrange(0,len(digits),2)]
	else:
		arr=[digits[i:i+2] for i in xrange(0,len(digits)-1,2)]
		arr[-1]=arr[-1]+digits[-1]
	print "-".join(arr)
