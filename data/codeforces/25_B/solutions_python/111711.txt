import sys
write = sys.stdout.write

n = int(raw_input())
code = raw_input()

if n%2==0 :
	grupos2 = n/2
	grupos3 = 0
else :
	grupos2 = n/2-1
	grupos3 = 1

if grupos3==1 :
	write(code[:3])
	code = code[3:]
	if len(code)>0 :
		write('-')

for i in xrange(0,len(code),2) :
	if i>0 : 
		write('-')
	write(code[i:i+2])

write('\n')
sys.stdout.flush()
