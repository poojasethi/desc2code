#!/usr/bin/python
n=input()
a=raw_input()
y=""
if  n%3==0 :
    for i in range(n) :
        if i%3==0 and i!=0 :
          y+="-"
        y+=a[i]
elif n%2==0 :
    for i in range(n):
       if i%2==0 and i!=0 :
          y+="-"
       y+=a[i]
else:
    y+=a[0]
    y+=a[1]
    y+="-"
    d=0
    for i in range(2,n-2):
       y+=a[i]
       d=d+1
       if(d%3==0):
         y+="-"
         d=0
    y+=a[n-2]
    y+=a[n-1]
print y
