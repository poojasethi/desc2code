count = int(raw_input())
phone = raw_input()
newphone = ""
if count % 2 == 0:
    for i in range(0,count,2):
        newphone += "-"+phone[i:i+2]
elif count % 3 == 0:
    for i in range(0,count,3):
        newphone += "-"+phone[i:i+3]
else:
    newphone += "-"+phone[:3]
    for i in range(3,count,2):
        newphone += "-" + phone[i:i+2]

print newphone[1:]
