import sys

data = sys.stdin.read().split()

ls = []
st = data[1]

gr = len(st) / 6
if (len(st) % 6 == 1):
  gr -= 1

for i in range(gr * 2):
  ls.append(st[3*i:3*i+3])

st = st[gr*6:]

if len(st) % 2 == 0:
  for i in range(len(st) / 2):
    ls.append(st[2*i:2*i+2])
elif len(st) == 3:
  ls.append(st)
else:
  if (len(st) > 5):
    ls.append(st[:2])
    st = st[2:]
  ls.append(st[:2])
  ls.append(st[2:])

print '-'.join(ls)
