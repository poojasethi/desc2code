#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	char c;
	int i,j,w=0,b=0;
	for(i=0;i<8;i++)
	for(j=0;j<8;j++){
		cin>>c;
		if(c=='P')w++;
		if(c=='R')w+=5;
		if(c=='N')w+=3;
		if(c=='B')w+=3;
		if(c=='Q')w+=9;
		
		if(c=='p')b++;

		if(c=='n')b+=3;
		if(c=='b')b+=3;
		if(c=='q')b+=9;
	}
	if(b>w)cout<<"Black";
	else
	if(b==w)cout<<"Draw";
	else
	cout<<"White";
}
