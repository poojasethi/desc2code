#include<cstdio>
using namespace std;
int a1[26];char s[20];
int main()
{
    int a=0,b=0;
    a1[16]=9,a1[17]=5,
    a1[1]=3,a1[13]=3,a1[15]=1;
    for(int i=0;i<8;++i)
    {
        scanf("%s",s);
        for(int j=0;j<8;++j)
        if(s[j]!='.')
        {
            if(s[j]>96)a+=a1[s[j]-97];
            else b+=a1[s[j]-65];
        }
    }
    puts(a>b?"Black":a<b?"White":"Draw");
}
