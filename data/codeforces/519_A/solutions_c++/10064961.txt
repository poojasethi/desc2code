#include <bits/stdc++.h>

using namespace std;

int ans,kar[1000000];

int main(){
	kar['Q']=9;
	kar['q']=-9;

	kar['R']=5;
	kar['r']=-5;
	
	kar['B']=kar['N']=3;
	kar['b']=kar['n']=-3;
	
	kar['P']=1;
	kar['p']=-1;
	
	for(int i=1;i<=8;i++)
		for(int j=1;j<=8;j++){
			char a;
			scanf(" %c",&a);
			ans+=kar[a];
		}
	
	if(ans>0) printf("White\n");
	else if(ans<0) printf("Black\n");
	else printf("Draw\n");
	
	return 0;
}
