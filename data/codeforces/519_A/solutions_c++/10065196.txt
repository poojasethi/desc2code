#include <cstdio>

int s;

int main()
{
	for (int i=1;i<=8;++i) for (int j=1;j<=9;++j)
	{
		switch (getchar())
		{
			case 'Q':s+=9;break;
			case 'R':s+=5;break;
			case 'B':s+=3;break;
			case 'N':s+=3;break;
			case 'P':s+=1;break;
			case 'q':s-=9;break;
			case 'r':s-=5;break;
			case 'b':s-=3;break;
			case 'n':s-=3;break;
			case 'p':s-=1;break;
		}
	}
	if (s<0) puts("Black");
	else if (s>0) puts("White");
	else puts("Draw");
	return 0;
}
