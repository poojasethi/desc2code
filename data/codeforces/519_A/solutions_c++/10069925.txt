#include<cstdio>
using namespace std;
int i,j,k,l,n,m,w,b;
char ch;
int main(){
	while(scanf("%c",&ch)!=EOF){
		if(ch=='Q')w+=9;
		if(ch=='R')w+=5;
		if(ch=='B')w+=3;
		if(ch=='N')w+=3;
		if(ch=='P')w+=1;
		if(ch=='q')b+=9;
		if(ch=='r')b+=5;
		if(ch=='b')b+=3;
		if(ch=='n')b+=3;
		if(ch=='p')b+=1;
	}
	if(w>b)printf("White\n");
	else if(b>w)printf("Black\n");
	else printf("Draw\n");
}
