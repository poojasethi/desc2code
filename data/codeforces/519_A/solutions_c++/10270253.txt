#include <iostream>
using namespace std;
char map[128];
int main()
{
	map['Q']=9,
	map['R']=5,
	map['B']=3,
	map['N']=3,
	map['P']=1;
	map['q']=-9,
	map['r']=-5,
	map['b']=-3,
	map['n']=-3,
	map['p']=-1;
	int res=0;
	char ch;
	for(int i=0;i<8*8;i++)
	{
		cin >> ch;
		res+=map[ch];
	}

	cout << (res>0?"White":res==0?"Draw":"Black") << endl;
	return 0;
}
