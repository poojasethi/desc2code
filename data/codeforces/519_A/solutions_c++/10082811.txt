#include <iostream>
using namespace std;
int p[30];
int main() {
	int w = 0, b = 0;
	p['q' - 'a'] = 9, p['r' - 'a'] = 5, p['b' - 'a'] = 3, p['n' - 'a'] = 3, p['p' - 'a'] = 1;
	for(int i = 0; i < 8; ++i)
		for(int j = 0; j < 8; ++j) {
			char c; cin >> c;
			if(c >= 'a') b += p[c - 'a'];
			else if(c >= 'A') w += p[c - 'A'];
		}
	cout << (b > w ? "Black" : b == w ? "Draw" : "White");
}
