#include <iostream>
using namespace std;

int m[256];
int tot;

int main()
{
	m['q'] = -(m['Q'] = 9);
	m['r'] = -(m['R'] = 5);
	m['b'] = -(m['B'] = 3);
	m['n'] = -(m['N'] = 3);
	m['p'] = -(m['P'] = 1);

	for(int i=0; i<8; i++)
	{
		string s;
		cin >> s;
		for(int j=0; j<8; j++)
			tot += m[s[j]];
	}

	if(tot == 0)
		cout << "Draw";
	else
		cout << (tot > 0 ? "White" : "Black");
	cout << endl;

	return 0;
}
