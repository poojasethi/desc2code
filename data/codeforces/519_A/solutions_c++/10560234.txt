#include <bits/stdc++.h>

using namespace std;


int main() {
  char c;
  int w = 0, b = 0, arr[] = {9, 5, 3, 3, 1, 0};
  string str1 = "QRBNPK", str2 = "qrbnpk";
  for(int i = 0; i < 64; ++i) {
    cin >> c;
    if(c == '.') continue;
    if(isupper(c)) w += arr[str1.find(c)];
    else b += arr[str2.find(c)];
  }
    
  if(w > b) cout << "White";
  else if(b > w) cout << "Black";
  else cout << "Draw";

  return 0;
}
