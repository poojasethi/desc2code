#include <bits/stdc++.h>
using namespace std;

int main() {
	int a=0, b=0;
	for(int i=1; i<=8; ++i) {
		char c[10];
		scanf("%s", c+1);
		for(int j=1; j<=8; ++j) {
			char s=c[j];
			if(s=='Q') a+=9; if(s=='q') b+=9;
			if(s=='R') a+=5; if(s=='r') b+=5;
			if(s=='B') a+=3; if(s=='b') b+=3;
			if(s=='N') a+=3; if(s=='n') b+=3;
			if(s=='P') a+=1; if(s=='p') b+=1;
		}
	}
	if(a>b) puts("White");
	if(a<b) puts("Black");
	if(a==b) puts("Draw");
	return 0;
}
