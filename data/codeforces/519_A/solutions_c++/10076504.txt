#include<bits/stdc++.h>
using namespace std;
struct pii{char a;int b;}P[10]={'Q',9,'R',5,'B',3,'N',3,'P',1,'q',-9,'r',-5,'b',-3,'n',-3,'p',-1};
int main()
{
    //freopen("a.txt","r",stdin);
    int ch,ans=0;
    while(~(ch=getchar())) {
        for(int i=0;i<10;i++) {
            if(P[i].a==ch) {
                ans+=P[i].b;
            }
        }
    }
    printf("%s\n",ans?(ans>0?"White":"Black"):"Draw");
    return 0;
}
