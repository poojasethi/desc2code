#include <iostream>
using namespace std;
int cc[255];
int main()
{
    int w,b;
    char c;
    for(int i=0;i<8;i++)for(int j=0;j<8;j++){cin>>c;cc[c]++;}
    w=cc['Q']*9+cc['R']*5+cc['B']*3+cc['N']*3+cc['P'];
    b=cc['q']*9+cc['r']*5+cc['b']*3+cc['n']*3+cc['p'];
    if(w==b)cout<<"Draw";
    if(w>b)cout<<"White";
    if(w<b)cout<<"Black";
    return 0;
}
