#include<iostream>
#include<algorithm>

using namespace std;

int p[300], ans;

int main()
{
	p['Q'] = 9;
	p['R'] = 5;
	p['B'] = 3;
	p['N'] = 3;
	p['P'] = 1;
	for(int i = 0; i <= 26; i++)
		p[i + 'a'] = -p[i + 'A'];
	for(int i = 0; i < 8; i++)
	{
		string s;	
		cin >> s;
		for(int j = 0; j < 8; j++)
			ans += p[s[j]];
	}
	if(ans > 0)
		cout << "White" << endl;
	else if(ans < 0)
		cout << "Black" << endl;
	else
		cout << "Draw" << endl;
	return 0;
}
