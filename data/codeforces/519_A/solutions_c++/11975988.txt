#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
map<char,int> wt;
int main()
{
	wt['q']=9;wt['r']=5;wt['b']=3;wt['n']=3;wt['p']=1;wt['k']=0;
	int w=0,b=0;
	for(int i=0;i<8;i++)
		for(int j=0;j<8;j++)
			{char c;cin>>c;
				if(c>'A' && c<'Z')w+=(wt[c+'a'-'A']);
				if(c>'a' && c<'z')b+=(wt[c]);
				}
	if(w==b)cout<<"Draw"<<endl;
	if(w>b)cout<<"White"<<endl;
	if(w<b)cout<<"Black"<<endl;
	return 0;
}
