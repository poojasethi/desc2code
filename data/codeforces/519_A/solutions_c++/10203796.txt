#include<bits/stdc++.h>

using namespace std;

int w[255];
int main()
{
    w['Q']=9;
    w['R']=5;
    w['B']=3;
    w['N']=3;
    w['P']=1;
    w['K']=0;
    char c;
    int ans=0;
    while(~(c=getchar()))
    {
        if(c>='a'&&c<='z')
            ans-=w[c-32];
        else if(c>='A'&&c<='Z')
            ans+=w[c];
    }
    puts(ans>0?"White":ans<0?"Black":"Draw");
    return 0;
}
