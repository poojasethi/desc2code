#include<iostream>
#include<algorithm>
using namespace std;

int i,a[300],n,m;
char c;

int main()
{
  a['q']=a['Q']=9;
  a['r']=a['R']=5;
  a['b']=a['B']=3;
  a['n']=a['N']=3;
  a['p']=a['P']=1;

  for(i=1;i<=64;++i)
  {
    cin>>c;
    if(c>='a' && c<='z') n+=a[c];
    if(c>='A' && c<='Z') m+=a[c];
  }

  if(n==m) cout<<"Draw\n";
  else cout<<(n<m ? "White\n":"Black\n");

 return 0;
}
