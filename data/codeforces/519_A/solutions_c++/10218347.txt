#include<bits/stdc++.h>
using namespace std;
int main()
{
    int w,b;w=b=0;char c;
    map<char, int > m;

    m['.']=0;m['q']=9;m['r']=5;
    m['b']=3;m['n']=3;m['p']=1;

    for(int i=0;i<64;i++){
        cin>>c;
        if(islower(c))
            b+=m[c];
        else w+=m[c+'a'-'A'];
    }

    if(b>w)cout<<"Black"<<endl;
    else if (w>b)cout<<"White"<<endl;
    else cout<<"Draw"<<endl;
}
