#include <bits/stdc++.h>
using namespace std;
map<char,int>m;
char c;
int ans=0;
int main()
{
	m['q']=9; m['r']=5; m['b']=3; m['n']=3;m['p']=1;m['k']=0;
	m['Q']=-9; m['R']=-5; m['B']=-3; m['N']=-3;m['P']=-1;m['k']=0;
	m['.']=0;
	for (int i=1;i<=64;i++)
	{
		scanf(" %c",&c);
		ans+=m[c];
	}
	if (ans<0) printf("White\n");
	else if (ans>0) printf("Black\n");
	else printf("Draw\n");
	return 0;
}
