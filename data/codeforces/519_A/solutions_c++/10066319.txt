#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;

int main()
{
	int a=0,b=0;
	for(int i=0;i<8;i++)
	{
		string s;
		cin>>s;
		for(int j=0;j<8;j++)
		{
			a+=s[j]=='Q'?9:s[j]=='R'?5:s[j]=='B' or s[j]=='N'?3:s[j]=='P'?1:0;
			b+=s[j]=='q'?9:s[j]=='r'?5:s[j]=='b' or s[j]=='n'?3:s[j]=='p'?1:0;
		}
	}
	if(a==b) printf("Draw\n");
	else
		if(a>b) printf("White\n");
		else printf("Black\n");
	return 0;
}
