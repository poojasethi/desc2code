#include <iostream>
#include <map>
#include <cstdio>

using namespace std;

map<char, int> w;

int main() {
	w['Q'] = 9;
	w['R'] = 5;
	w['B'] = 3;
	w['N'] = 3;
	w['P'] = 1;
	w['q'] = -9;
	w['r'] = -5;
	w['b'] = -3;
	w['n'] = -3;
	w['p'] = -1;
	int ans = 0;
	for (int i = 0; i < 8; i++)
		for (int j = 0; j < 8; j++) {
			char c;
			cin >> c;
			ans += w[c];
		}
	cout << (ans == 0 ? "Draw" : (ans < 0 ? "Black" : "White")) << endl;
	return 0;
}
