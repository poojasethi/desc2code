#include <bits/stdc++.h>
using namespace std;

int main()
{
	int a=0,b=0;
	char c;
	while (cin>>c)
	{
		if (c=='Q') a+=9;
		if (c=='q') b+=9;
		if (c=='R') a+=5;
		if (c=='r') b+=5;
		if (c=='B') a+=3;
		if (c=='b') b+=3;
		if (c=='N') a+=3;
		if (c=='n') b+=3;
		if (c=='P') a++;
		if (c=='p') b++;
	}
	if (a>b) printf("White\n");
	else if (a<b) printf("Black\n");
	else printf("Draw\n");
	return 0;
}
