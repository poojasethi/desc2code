#include<iostream>
using namespace std;

char a[30] = "30000000000030195";
int main() {
	char str[10];
	int count=0;
	for(int j=0;j<8;j++) {
		cin>>str;
		for(int i=0;i<8;i++) {
			if(str[i]> '.') {
				str[i]<'a' ? (count+= (a[str[i]-'B'] - '0')): (count-=(a[str[i]-'b'] -'0'));
			}
		}
	}
	if(count>0) {
		cout<<"White";
	}
	else if(count ==0) {
		cout<<"Draw";
	}
	else cout<<"Black";
	return 0;
}
