#include <cstdio>

char c;
int k;

int main()
{
	do
	{
		c = getchar();
		if (c == 'Q')
			k += 9;
		if (c == 'R')
			k += 5;
		if (c == 'B' || c == 'N')
			k += 3;
		if (c == 'P')
			++k;
		if (c == 'q')
			k -= 9;
		if (c == 'r')
			k -= 5;
		if (c == 'b' || c == 'n')
			k -= 3;
		if (c == 'p')
			--k;
	}
	while (c != EOF);

	puts(k == 0 ? "Draw" : k > 0 ? "White" : "Black");

	return 0;
}
