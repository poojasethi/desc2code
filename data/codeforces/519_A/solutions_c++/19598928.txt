#include<iostream>
#include<map>

using namespace std;

int main()
{
	map<char,int> s;
	char c;
	int i,hasil=0;
	
	s['Q']=9;
	s['R']=5;
	s['B']=3;
	s['N']=3;
	s['P']=1;
	s['q']=-9;
	s['r']=-5;
	s['b']=-3;
	s['n']=-3;
	s['p']=-1;
	
	for(i=0;i<64;i++)
	{
		cin >> c;
		hasil+=s[c];
	}
	
	hasil>0?cout << "White" : hasil==0? cout << "Draw" : cout << "Black";
}
