#include<bits/stdc++.h>



int main(){
	int b=0, w=0;
	for(int i=0;i<8;++i){
		for(int j=0;j<8;++j){
			char x;
			scanf(" %c", &x);
			if(x=='q')b+=9;
			else if(x=='Q')w+=9;
			else if(x=='r')b+=5;
			else if(x=='R')w+=5;
			else if(x=='n' ||x=='b')b+=3;
			else if(x=='N'||x=='B')w+=3;
			else if(x=='p')b+=1;
			else if(x=='P')w+=1;
		}
	}

	if(b>w)
		printf("Black");
	else if(w>b)printf("White");
	else printf("Draw");
}
