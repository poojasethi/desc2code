#include <stdio.h>
int main()
{
	char i,k;
	short int d=0;
	for (i=0;i<64;i++)
	{
		scanf(" %c",&k);
		switch (k)
		{
			case 'Q':d+=9;break;
			case 'R':d+=5;break;
			case 'B':d+=3;break;
			case 'N':d+=3;break;
			case 'P':d++;break;
			case 'q':d-=9;break;
			case 'r':d-=5;break;
			case 'b':d-=3;break;
			case 'n':d-=3;break;
			case 'p':d--;break;
		}
	}
	printf(d==0?"Draw":d>0?"White":"Black");
	return 0;
}
