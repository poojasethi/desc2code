#include <iostream>
#include <cstdio>
using namespace std;

int s, a[99];
char c;
int main()
{
    int i, j;
//    freopen ("input.txt", "r", stdin);
	a['Q'] = 9;
	a['R'] = 5;
	a['B'] = 3;
	a['N'] = 3;
	a['P'] = 1;

    for (i = 0; i < 8; i++) {
        for (j = 0; j < 8; j++) {
            cin >> c;
            c < 'a' ? s += a[c] : s -= a[c - 32];
        }
    }
    puts(s > 0 ? "White" : s ? "Black" : "Draw");
    return 0;
}
