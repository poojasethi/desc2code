point = [0] * 2
table = {"q": 9, "r": 5, "b": 3, "n": 3, "p": 1, "k": 0}

for _ in xrange(8):
	for s in raw_input():
		if s == ".":
			continue
		which = s.isupper()
		point[which] += table[s.lower()]
if point[0] < point[1]:
	print "White"
elif point[0] > point[1]:
	print "Black"
else:
	print "Draw"
