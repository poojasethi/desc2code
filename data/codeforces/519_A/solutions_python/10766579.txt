d = {'q': 9, 'r': 5, 'b': 3, 'n': 3, 'p': 1, 'k': 0}
white = 0
black = 0
for c in ''.join(raw_input() for i in xrange(8)):
    if c.isupper():
        white += d[c.lower()]
    elif c.islower():
        black += d[c]

if white > black:
    print "White"
elif white < black:
    print "Black"
else:
    print "Draw"
