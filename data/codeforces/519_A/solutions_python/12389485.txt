wh = bl = 0
for i in range(8):
	l =raw_input()
	
	Q = l.count('Q')
	R = l.count('R')
	B = l.count('B')
	N = l.count('N')
	P = l.count('P')
	
	q = l.count('q')
	r = l.count('r')
	b = l.count('b')
	n = l.count('n')
	p = l.count('p')

	wh += Q*9 + R*5 + (B+N)*3 + P
	bl += q*9 + r*5 + (b+n)*3 + p
if (wh == bl):
	print "Draw"
else:
	print ("White","Black")[wh < bl]
