
D = {
	'Q':9,'q':-9,
	'R':5,'r':-5,
	'B':3,'b':-3,
	'N':3,'n':-3,
	'P':1,'p':-1,
	'K':0,'k':-0,
	'.':0
}

def main():
	sum = 0
	n = 8
	for i in range(n):
		for c in raw_input():
			sum += D[c]
	if sum < 0:
		print 'Black'
	elif sum > 0:
		print 'White'
	else:
		print 'Draw'


if __name__ == '__main__':
	main()
