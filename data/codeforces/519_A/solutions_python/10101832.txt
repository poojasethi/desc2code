from sys import stdin
l=[]
for i in stdin:
	x=str(i)
	l.append(x)

s=0
dic={}
dic['Q']=9
dic['R']=5
dic['B']=3
dic['N']=3
dic['P']=1
dic['q']=-9
dic['r']=-5
dic['b']=-3
dic['n']=-3
dic['p']=-1
dic['K']=0
dic['k']=0
dic['.']=0
dic['\n']=0
for i in l:
	for j in i:
		s+=dic[j]
if s>0:
	print "White"
elif s<0:
	print "Black"
else:
	print "Draw"
