chess = []
for i in xrange(8):
    temp = raw_input()
    chess.append(temp)

chess = str(chess)
a=0
b=0
res = ""
power = "qrbnp"
val = [9, 5, 3, 3, 1]

for i in xrange(5):
	b += val[i] * chess.count(power[i])
	a += val[i] * chess.count(power[i].upper())

if( a > b): res = "White"
elif(a < b): res = "Black"
else: res ="Draw"

print res
