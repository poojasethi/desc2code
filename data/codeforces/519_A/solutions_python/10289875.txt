points_map = {'q': 9, 'p': 1, 'r': 5, 'b': 3, 'n': 3, 'k': 0}
w_points = b_points = 0

for i in range(0,8):
	str = raw_input()
	for c in str:
	 if c!='.':
		 point = points_map[c.lower()]
		 if ord(c) <= 97:
		  w_points += point
		 else:
		  b_points += point

if w_points>b_points:
	print "White"
elif b_points>w_points:
	print "Black"
else:
	print "Draw"
