def f(X):
    ans = 0
    for i,w in [['Q',9],['R',5],['B',3],['N',3],['P',1],
              ['q',-9],['r',-5],['b',-3],['n',-3],['p',-1]]:
        ans += w*X.count(i)
    return ans

A = ''
for i in xrange(8):
    A = A+raw_input()
z=f(A)
if z>0: print "White"
elif z==0: print "Draw"
else: print "Black"
