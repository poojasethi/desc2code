chars = []
for i in xrange(8):
    chars += list(raw_input())

bal = 0
for i in chars:
    bal += (i=="Q")*9
    bal += (i=="R")*5
    bal += (i=="B")*3
    bal += (i=="N")*3
    bal += (i=="P")*1
    bal -= (i=="q")*9
    bal -= (i=="r")*5
    bal -= (i=="b")*3
    bal -= (i=="n")*3
    bal -= (i=="p")*1

print "White" if bal>0 else "Black" if bal<0 else "Draw"
