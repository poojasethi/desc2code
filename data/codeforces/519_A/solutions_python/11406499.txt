wht = {
	'Q':9,
	'R':5,
	'B':3,
	'N':3,
	'P':1,
	'K':0
}

blk = {
	'q':9,
	'r':5,
	'b':3,
	'n':3,
	'p':1,
	'k':0
}

if __name__ == '__main__':
	ws = 0
	bs = 0
	for _ in xrange(8):
		for i in raw_input():
			if i in wht:
				ws += wht[i]
			elif i in blk:
				bs += blk[i]
	if ws > bs:
		print 'White'
	elif ws < bs:
		print 'Black'
	else:
		print 'Draw'
