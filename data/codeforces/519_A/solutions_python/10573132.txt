sco = dict(q=9, r=5, b=3, n=3, p=1, k=0)
A = 0
B = 0
while True:
	try:
		s = raw_input()
		for ch in s:
			if ch.isalpha():
				if ch.islower():
					A = A + sco[ch]
				else:
					B = B + sco[ch.lower()]
	except EOFError:
		break
if A > B:
	print 'Black'
elif A == B:
	print 'Draw'
else:
	print 'White'
