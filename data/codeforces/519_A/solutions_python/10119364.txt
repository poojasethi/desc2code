a = 0
b = 0
ax = {'Q': 9,'R': 5,'B': 3,'N': 3,'P': 1}
bx = {'q': 9,'r': 5,'b': 3,'n': 3,'p': 1}
for _ in xrange(8):
    s = raw_input()
    for x in s:
        if x in ax.keys():
            a += ax[x]
        elif x in bx.keys():        
            b += bx[x]
if a==b:
    print "Draw"
elif a>b:
    print "White"
else:
    print "Black"
