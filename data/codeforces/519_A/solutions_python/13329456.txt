w = {"q":9, "r":5, "b":3, "n":3, "p":1, "k":0}
ww = bw = 0
for _ in range(8):
    for c in raw_input():
        if c == ".":
            continue
        elif c.isupper():
            ww += w[c.lower()]
        else:
            bw += w[c]
if ww > bw:
    print "White"
elif ww < bw:
    print "Black"
else:
    print "Draw"
