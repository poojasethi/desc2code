
v = {
    'q': 9,
    'r': 5,
    'b': 3,
    'n': 3,
    'p': 1,
    'k': 0,
    '.': 0
}

w, b = 0, 0

for _ in range(8):
    r = raw_input()
    for c in r:
        if c.islower():
            b += v[c]
        else:
            w += v[c.lower()]

print 'White' if w > b else 'Black' if b > w else 'Draw'
