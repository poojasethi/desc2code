#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,a[100005];
    cin>>n;
    cin>>a[0];
    int cnt=0,ind=0;
    for(int i=1;i<n;++i)
    {
        cin>>a[i];
        if(a[i]<a[i-1]){++cnt;ind=i;}
    }
    if(a[0]<a[n-1]){++cnt;ind=0;}
    if(cnt<2)cout<<(n-ind)%n;
    else cout<<"-1";
}
