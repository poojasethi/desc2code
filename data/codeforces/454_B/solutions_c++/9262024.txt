#include<iostream>
using namespace std;
int main(){
    int n,i,j,k=-1,a[100005];
    cin >> n;
    for(i=0;i<n;++i)cin>>a[i];
    for(i=1;i<n;++i)if(a[i]<a[i-1])break;
    for(j=n-1;j>0;j--)if(a[j]<a[j-1])break;
    if(i==j&&a[n-1]<=a[0])cout<<n-i<<endl;
    else if(i==n)cout<<0<<endl;
    else cout<<-1<<endl;
}
