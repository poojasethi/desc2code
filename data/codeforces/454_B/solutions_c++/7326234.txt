#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int n,ans=0,pos=1;
	cin>>n;
	int a[n+1];
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int j=1;j<n;j++)
	{
		if(a[j]>a[j+1])
		{
			ans++;
			pos=j;
		}
	}
	if(ans==0)
		cout<<"0";
	else if(ans==1 && a[n]<=a[1])
		cout<<(n-pos);
	else
		cout<<"-1";
	return 0;
}
