#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,a[100010],i,c=0,ind;
	cin>>n;
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<n;i++)
	{
		if(a[i] > a[(i+1)%n])
		{
			c++;
			ind=i;
		}
	}
	if(c==1)
		cout<<n-ind-1<<endl;
	else if(!c)
		cout<<"0\n";
	else
		cout<<"-1\n";
	return 0;
}
