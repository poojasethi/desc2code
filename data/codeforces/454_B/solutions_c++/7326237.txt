#include<stdio.h>
int main()
{
	int a,n,s,i,p,c = 0,b;
	scanf("%d",&n);
	scanf("%d",&s);
	a = s;
	for(i = 1 ; i < n ; i++) {
		scanf("%d",&b);
		if(a > b) {
			c++;
			p = i;
		}
		a = b;
	}
	if(c == 0) {
		printf("0");
	}else if(c == 1 && s >= b) {
		printf("%d",n - p);
	}else{
		printf("-1");
	}
	return 0;
}
