#include<bits/stdc++.h>

using namespace std;
int main()
{
	int i,j,k,l,m,n;
	cin>>n;
	vector<int> vect(n,0);
	for(i=0;i<n;i++)
	{
		cin>>vect[i];
	}
	k=0;
	l=n;
	for(i=1;i<n;i++)
	{
		if(vect[i]<vect[i-1])
		{
			k+=1;
			l=i;
		}
	}
	if(k>1 ||((k==1) && vect[n-1]>vect[0]))
	{
		cout<<-1;//<<" S";
		return 0;
	}
	cout<<n-l;
	return 0;
}
