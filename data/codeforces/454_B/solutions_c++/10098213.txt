#include<iostream>
using namespace std;
int a[1000*1000];
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	a[0]=a[n];
	int t=0;
	int cnt=0;
	for(int i=0;i<=n;i++){
		if(a[i]<a[i-1]){
			cnt++;
			t=i;
		}
	}
	if(cnt<=1){
		if(n-t+1>=n)
			cout<<0;
		else	
			cout<<n-t+1;
	}
	else
		cout<<-1;
}
