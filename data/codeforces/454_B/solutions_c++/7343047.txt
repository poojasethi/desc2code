#include <iostream>

using namespace std;
int a[100005];
int main()
{
    int n,s=0,k=0; cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        if(a[i]<a[i-1] && i>0){ s++; k=n-i;}
    }
    if(a[0]<a[n-1]) {s++; k=0;}
    if(s<=1) cout<<k<<endl;
    else cout<<-1<<endl;

    return 0;
}
