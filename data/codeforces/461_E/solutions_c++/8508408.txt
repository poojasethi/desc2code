#define pii pair<int,int>

#define B second
#define mp make_pair
#define pb push_back
long long n,f,l,m;
int tmp,M[1<<20];
long long dp[5][5],a[5][5],c[5][5],u;
char s[100002];
void Do(long long x){
    if(x == 1){
        for(int i=0;i<4;i++)
            for(int j=0;j<4;j++)
                a[i][j] = dp[i][j];
    }
    else{
        Do(x>>1);
        if(u) return ;
        for(int i=0;i<4;i++) for(int j=0;j<4;j++) c[i][j] = 1e18+1ll;
        for(int i=0;i<4;i++)
            for(int j=0;j<4;j++)
                for(int k=0;k<4;k++)
                    c[i][j] = min(c[i][j],a[i][k]+a[k][j]);
        for(int i=0;i<4;i++)
            for(int j=0;j<4;j++)
                a[i][j] = c[i][j], c[i][j] = 1e18+1ll;
        if(x&1){
            for(int i=0;i<4;i++)
                for(int j=0;j<4;j++)
                    for(int k=0;k<4;k++)
                        c[i][j] = min(c[i][j],a[i][k]+dp[k][j]);
            for(int i=0;i<4;i++)
                for(int j=0;j<4;j++)
                    a[i][j] = c[i][j];
        }
    }
    long long sv = 1e18+1ll;
    for(int i=0;i<4;i++)
        for(int j=0;j<4;j++)
                sv = min(sv,a[i][j]);
    if(sv >= n) u = 1;
    return ;
}
int main(){
    scanf("%I64d",&n);
    scanf("%s",s);
    for(int j=10;j>=2;j--){
        tmp = 0;
        for(int i=0;s[i];i++){
            tmp *= 4; tmp += s[i]-'A';  
            if(i >= j-1){
                if(i != j-1) tmp -= (1<<(2*j))*(s[i-j]-'A');
                M[tmp] = j;
            }   
        }
        for(int i=0;i<(1<<(2*j));i++)
            if(M[i] != j)
                dp[(i>>(2*(j-1)))&3][i&3] = j-1;
    }
    f = 1,l = n;
    while(l>=f){
        m = (f+l)>>1; u = 0;
        Do(m);
        if(u) l = m-1;
        else f = m+1;
    }
    printf("%I64d\n",f);
    return 0;
}
