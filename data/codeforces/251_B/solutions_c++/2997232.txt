#include<cstdio>
#include<cstring>
int a[101],b[101],s[101],c[101],d[101],cc[101],dd[101];
int n,k;
bool cmp(int a[101],int b[101])
{
     int i;
     for (i=1;i<=n;i++)
         if (a[i]!=b[i]) return false;
     return true;
}
int main()
{
    int i,j;
    scanf("%d%d",&n,&k);
    for (i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        b[a[i]]=i;
    }
    for (i=1;i<=n;i++) scanf("%d",&s[i]);
    for (i=1;i<=n;i++)
    {
        c[i]=i; d[i]=i;
    }
    if (cmp(c,s)||((k>1)&&cmp(a,b)))
    {
       printf("NO\n"); return 0;
    }
    if ((k%2==0)&&cmp(c,s))
    {
       printf("YES\n"); return 0;
    }
    for (i=1;i<=k;i++)
    {
        for (j=1;j<=n;j++)
        {
            cc[j]=c[a[j]]; dd[j]=d[b[j]];
        }
        memcpy(c,cc,sizeof(cc));
        memcpy(d,dd,sizeof(dd));
        if (((k-i)%2==0)&&(cmp(c,s)||cmp(d,s)))
        {
           printf("YES\n"); return 0;
        }
    }
    printf("NO\n"); return 0;
}
