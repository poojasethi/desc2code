#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<string.h>
using namespace std;
int n,b[105];
bool xd(int x[])
{
	for(int i=1;i<=n;i++)
	{
		if(x[i]!=b[i])
		return 0;
	}
	return 1;
}
int main()
{
	int a[105],k,c[105],i,j,s=0,t=0,d[105];
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)
	c[i]=i;
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
	for(i=1;i<=n;i++)
	scanf("%d",&b[i]);
	for(i=1;i<=k;i++)
	{
		for(j=1;j<=n;j++)
		d[j]=c[a[j]];
		for(j=1;j<=n;j++)
		c[j]=d[j];
		if(xd(c)) 
		{
			s=i;
			break;
		}
	}
	for(i=1;i<=n;i++)
	c[i]=i;
	for(i=1;i<=k;i++)
	{
		for(j=1;j<=n;j++)
		d[a[j]]=c[j];
		for(j=1;j<=n;j++)
		c[j]=d[j];
		if(xd(c)) 
		{
			t=i;
			break;
		}
	}
	for(i=1;i<=n;i++)
	c[i]=i;
	if(xd(c))
	printf("NO\n");
	else if(s==1&&t==1&&k>1)
	printf("NO\n");
	else if((k-s)%2==0&&s)
	printf("YES\n");
	else if((k-t)%2==0&&t)
	printf("YES\n");
	else
	printf("NO\n");
}
