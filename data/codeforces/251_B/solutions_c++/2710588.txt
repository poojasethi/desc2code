#include <iostream>

using std::cin;
using std::cout;
using std::endl;

int n, k;
const int MAXN = 100;
int q[MAXN], p[MAXN], s[MAXN], t[MAXN];

void foo()
{
	for (int i = 0; i != n; ++i) t[q[i] - 1] = p[i];
	for (int i = 0; i != n; ++i) p[i] = t[i];
}

void foo2()
{
	for (int i = 0; i != n; ++i) t[i] = p[q[i] - 1];
	for (int i = 0; i != n; ++i) p[i] = t[i];
}

bool chk()
{
	for (int i = 0; i != n; ++i) if (p[i] != s[i]) return false;
	return true;
}

int main()
{
	int x, y;
	cin >> n >> k;
	for (int i = 0; i != n; ++i) cin >> q[i];
	for (int i = 0; i != n; ++i) cin >> s[i];
	for (int i = 0; i != n; ++i) p[i] = i + 1;
	if (chk()) { cout << "NO" << endl; return 0; }
	for (x = 1; x <= k; ++x) { foo(); if (chk()) break; }
	for (int i = 0; i != n; ++i) p[i] = i + 1;
	for (y = 1; y <= k; ++y) { foo2(); if (chk()) break; }
	if ((k == 1 || x != 1 || y != 1) &&
	    ((x <= k && (x & 1) == (k & 1)) ||
	    (y <= k && (y & 1) == (k & 1))))
		cout << "YES" << endl;
	else
		cout << "NO" << endl;
}
