#include<stdio.h>
int b[105],n;
int pan(int c[])
{
	int i;
	for(i=1;i<=n;i++)
	if(b[i]!=c[i])return 0;
	return 1;
}
int main()
{
	int a[105],c[105],d[105],i,j,m,k,t;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		t=k=0;
		for(i=1;i<=n;i++)
		c[i]=i;
		for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
		for(i=1;i<=n;i++)
		scanf("%d",&b[i]);
		if(pan(c))
		{
			printf("NO\n");
			continue;
		}
		for(i=1;i<=m;i++)
		{
			for(j=1;j<=n;j++)
			d[j]=c[a[j]];
			for(j=1;j<=n;j++)
			c[j]=d[j];
			if(pan(c))
			{
				t=i;
				break;
			}
		}
		for(i=1;i<=n;i++)
		c[i]=i;
		for(i=1;i<=m;i++)
		{
			for(j=1;j<=n;j++)
			d[a[j]]=c[j];
			for(j=1;j<=n;j++)
			c[j]=d[j];
			if(pan(c))
			{
				k=i;
				break;
			}
		}
		if(t==1&&k==1&&m>1)printf("NO\n");
		else if((m-t)%2==0&&t)printf("YES\n");
		else if((m-k)%2==0&&k)printf("YES\n");
		else printf("NO\n");
	}
}
