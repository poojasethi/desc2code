#include <iostream>

#define rep(i,n) for(i= 0;i<n;i++)

using namespace std;

int a[110],p[110],q[110],s[110],n,k,i,j,ii,f[2];

int main(){

  cin>>n>>k;
  rep(i,n){cin>>q[i]; q[i]--; p[i] = i;}
  rep(i,n){cin>>s[i];s[i]--;}
  rep(ii,k+1){
    rep(i,n)if(p[i]!=s[i])break;
    if(i>=n) break;
    rep(i,n)a[i] = p[q[i]];
    rep(i,n)p[i] = a[i];
  }
  f[0] = ii;

  rep(i,n)p[i] = i;
  rep(ii,k+1){
    rep(i,n)if(p[i]!=s[i])break;
    if(i>=n)break;
    rep(i,n)a[q[i]] = p[i];
    rep(i,n)p[i] = a[i];
  }
  f[1] = ii;

  if(f[0] == k ||  f[1] == k )cout<<"YES"<<endl;
  else if(((f[0]%2==k%2) || (f[1]%2 == k%2)) && f[0]+f[1]>2)cout<<"YES"<<endl;
  else cout<<"NO"<<endl;

  return 0;



}
