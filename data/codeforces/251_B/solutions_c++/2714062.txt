#include<iostream>
#define pr 101
#define mod 1000000007
using namespace std;
int main()
{
int n,m,i,j=1,a[105],b[105],c[105],d[105],w=0,r,k=0,p=0;
cin>>n>>m;
for(i=1;i<=n;i++) cin>>c[i],a[i]=i;
for(i=1;i<=n;i++) cin>>d[i],w=(w*pr+d[i])%mod,j&=(d[i]==i);
if(j)
{
cout<<"NO";
return 0;
}
do{
k++;
r=0;
for(i=1;i<=n;i++) b[i]=a[c[i]],r=(r*pr+b[i])%mod;
for(i=1;i<=n;i++) a[i]=b[i];
if(k>m) break;
}while(w!=r);
for(i=1;i<=n;i++) a[i]=i;
do{
p++;
r=0;
for(i=1;i<=n;i++) b[c[i]]=a[i];
for(i=1;i<=n;i++) r=(r*pr+b[i])%mod;
for(i=1;i<=n;i++) a[i]=b[i];
if(p>m) break;
}while(w!=r);
if(m>=k && (m-k)%2==0 && (m==k || k>1 || p>1)) cout<<"YES";
else if(m>=p && (m-p)%2==0 && (m==p || k>1 || p>1)) cout<<"YES";
else cout<<"NO";
return 0;
}
