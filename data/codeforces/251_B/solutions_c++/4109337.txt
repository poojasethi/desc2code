#include<stdio.h>
#include<string.h>
int a[1000],b[1000],c[1000],bian[1000],mubiao[1000],n;
int bi(int x[],int y[]){
	int t;
	for(t=1;t<=n;t++){
		if(x[t]!=y[t])return 0;
	}
	return 1;	
}
int main (){
	int i,j,m,p,q,f,s,t,v;
	while(scanf("%d%d",&n,&m)!=EOF){
		s=t=0;
		for(i=1;i<=n;i++){
			scanf("%d",&bian[i]);
			a[i]=i;
		}
		for(i=1;i<=n;i++){
			scanf("%d",&mubiao[i]);
		}
		for(i=1;i<=m;i++){
			for(j=1;j<=n;j++){
				b[j]=a[bian[j]];
			}
			for(j=1;j<=n;j++){
				a[j]=b[j];
			}
			if(bi(a,mubiao)){
				s=i;
				break;
			}
		}
		for(j=1;j<=n;j++){
			a[j]=j;
		}
		for(i=1;i<=m;i++){
			for(j=1;j<=n;j++){
				b[bian[j]]=a[j];
			}
			for(j=1;j<=n;j++){
				a[j]=b[j];
			}
			if(bi(a,mubiao)){
				t=i;
				break;
			}
		}
		for(j=1;j<=n;j++){
			a[j]=j;
		}
	//	printf("%d %d\n",s,t);
		if(bi(a,mubiao)){
			printf("NO\n");
		}
		else if(t==1&&s==1&&m>1)
			printf("NO\n");
		else if(t>0&&(m-t)%2==0)
			printf("YES\n");
		else if(s>0&&(m-s)%2==0)
			printf("YES\n");
		else printf("NO\n");
	}
}
