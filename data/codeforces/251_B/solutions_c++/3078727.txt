#include<stdio.h>
int isEq(int*,int*,int);
int isEq(int*a,int*b,int n)
{
	int i,eq=1;
	for(i=1;i<=n;i++)
		if(a[i]!=b[i])eq=0;
	return eq;
}
int main()
{
	int n,k,q[1000],s[1000],q1[1000],st[1000],w[1000],i,j;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&q[i]);
		q1[q[i]]=i;
		st[i]=i;
	}
	for(i=1;i<=n;i++)
		scanf("%d",&s[i]);
	if(isEq(s,st,n))
	{
		printf("NO");
		return 0;
	}
	if(isEq(q1,q,n)&&isEq(q,s,n))
	{
		if(k==1)
			printf("YES");
		else printf("NO");
		return 0;
	}
	for(i=1;i<=k;i++)
	{
		for(j=1;j<=n;j++)
			w[j]=q[st[j]];
		for(j=1;j<=n;j++)
			st[j]=w[j];
		if(isEq(st,s,n))
		{
			if((k-i)%2==0)
			{
				printf("YES");
				return 0;
			}
			else break;
		}
	}
	for(i=1;i<=n;i++)
		st[i]=i;
	for(i=1;i<=k;i++)
	{
		for(j=1;j<=n;j++)
			w[j]=q1[st[j]];
		for(j=1;j<=n;j++)
			st[j]=w[j];
		if(isEq(st,s,n))
		{
			if((k-i)%2==0)
			{
				printf("YES");
				return 0;
			}
			else break;
		}
	}
	printf("NO");
	return 0;
}
