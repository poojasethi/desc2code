#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
#define N 21
int f[N][1<<N],n,m,ans=1e9;
char s[N][1000010];
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) scanf("%s",s[i]+1);
	for (int i=1;i<=m;i++) {
		int x=0;
		for (int j=1;j<=n;j++)
			if (s[j][i]=='1') x += 1<<(j-1);
		f[0][x]++;
	}
	for (int i=1;i<=n;i++)
		for (int j=i;j>=1;j--)
			for (int k=0;k<(1<<n);k++)
				f[j][k] += f[j-1][k^(1<<(i-1))];
	for (int i=0;i<(1<<n);i++) {
		int now = 0;
		for (int j=0;j<=n;j++) now += min(j,n-j)*f[j][i];
		ans = min(ans,now);
	}
	printf("%d",ans);
	return 0;
}
