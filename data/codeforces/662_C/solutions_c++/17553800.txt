#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define maxn 25
#define maxm 100005
#define maxs 1025*1025
#define mem(a,b) memset(a,b,sizeof(a))
#define mcp(a,b) memcpy(a,b,sizeof(a))
using namespace std;

int f[maxs][maxn];

int n,m;

int a[maxn][maxm];

int main(){
	scanf("%d%d",&n,&m);
	fo(i,1,n)
		fo(j,1,m) {
			char c=getchar();
			while (c!='0' && c!='1') c=getchar();
			a[i][j]=c-'0';
		}
	fo(j,1,m) {
		int now=0;
		fo(i,1,n) now|=(a[i][j] * (1 << (i-1)));
		f[now][0]++;
	}
	fo(i,1,n) {
		int g[maxs][maxn];
		mcp(g,f);
		mem(f,0);
		fo(s,0,(1 << n)-1)
			fo(j,0,n) {
				f[s][j]+=g[s][j];
				f[s ^ (1 << (i-1))][j+1]+=g[s][j];
			}
	}
	int ans=n*m;
	fo(s,0,(1 << n)-1) {
		int now=0;
		fo(i,0,n) now=now+min(i,n-i) * f[s][i];
		ans=min(ans,now);
	}
	printf("%d",ans);
	return 0;
}
