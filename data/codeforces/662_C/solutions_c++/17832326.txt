#include <iostream>
#include <cstdio>
#define MAXN 100005
using namespace std;

char w[MAXN];
int n, m;
int col[MAXN], dp[21][1<<21];

int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;++i)
    {
        scanf("%s",w);
        for(int j=0;j<m;++j)
            col[j]|=(w[j]-'0')<<i;
    }

    for(int j=0;j<m;++j)++dp[0][col[j]];

    int End=1<<n;
    for(int k=1, s, j;k<=n;++k)
        for(s=0;s<End;++s)
        {
            dp[k][s]=k>1?(k-2-n)*dp[k-2][s]:0;
            for(j=0;j<n;++j)dp[k][s]+=dp[k-1][s^(1<<j)];
            dp[k][s]/=k;
        }

    int ans=n*m, cnt;
    for(int s=0;s<End;++s)
    {
        cnt=0;
        for(int i=0;i<n;++i)cnt+=min(i,n-i)*dp[i][s];
        ans=min(ans,cnt);
    }
    printf("%d\n",ans);
    return 0;
}
