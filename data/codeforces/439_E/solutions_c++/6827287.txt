#include<cstdio>
#include<cmath>
#include<cstring>
#include<vector> 
#define Mod 1000000007
#define mn 100000
#define pb push_back
#define rep(I,A,B) for(int I=A;I<=(B);I++)
#define dep(I,A,B) for(int I=A;I>=(B);I--)
#define re(I,B) for(int I=0;I<(B);I++)
#define de(I,A) for(int I=(A)-1;I+1;I--)
using namespace std;
typedef long long ll;
int T,n,f;vector<int>fact[mn+11];
ll fac[mn+11],ifct[mn+11],dp[mn+11];

inline ll _pow(ll a,ll b){ll tmp=1;a%=Mod;for(;b;b>>=1){if(b&1)tmp=tmp*a%Mod;a=a*a%Mod;}return tmp;}
inline void R(int &xx){xx=0;char ch=getchar();while(ch<'0'||ch>'9')ch=getchar();while(ch>='0'&&ch<='9')xx=xx+xx+(xx<<3)+ch-48,ch=getchar();}
inline ll C(int n,int m){return (m>n)?0:fac[n]*ifct[m]%Mod*ifct[n-m]%Mod;}
inline void add(ll &a,ll b){a+=b;if(a>=Mod)a-=Mod;}
inline void dec(ll &a,ll b){a-=b;if(a<0)a+=Mod;}

int main()
{
    fac[0]=1;rep(i,1,mn)fac[i]=fac[i-1]*i%Mod;
    rep(i,0,mn)ifct[i]=_pow(fac[i],Mod-2);
    rep(i,1,mn)
        for(int j=i;j<=mn;j+=i)fact[j].pb(i);
    for(R(T);T--;){
        R(n);R(f);vector<int>a;
        re(i,fact[n].size())if((ll)fact[n][i]*f<=n)a.pb(fact[n][i]);
        de(i,a.size()){
            dp[i]=C(n/a[i]-1,f-1);
            rep(j,i+1,a.size()-1)
                if(a[j]%a[i]==0)dec(dp[i],dp[j]);
        }
        printf("%I64d\n",dp[0]);
    }
    return 0;
}
