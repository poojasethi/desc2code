// C++
#include <bits/stdc++.h>
#define fi first
#define se second
#define endl '\n'

using namespace std;

typedef long long lint;
typedef pair<int,int> ii;
const int MAXN = 2e5+222;
const int mod = 1e9+7;

int T,n,f,fakt[MAXN],inv[MAXN];

int mul(int a,int b){
	return (long long)a*b%mod;
}

int power(int a,int b){
	int res=1;
	for(int i=0;i<=30;i++){
		if(b&(1<<i))
			res=mul(res,a);
		a=mul(a,a);
	}
	return res;
}

int komb(int a,int b){
	if(a<b) return 0;
	return mul(fakt[a],mul(inv[b],inv[a-b]));
}

int main(){
	
	fakt[0]=1;
	for(int i=1;i<=2e5;i++)
		fakt[i]=mul(fakt[i-1],i);
	for(int i=0;i<=2e5;i++)
		inv[i]=power(fakt[i],mod-2);
	
	scanf(" %d",&T);
	while(T--){
		scanf(" %d %d",&n,&f);
		vector<int> pr;
		int nn=n;
		for(int i=2;i*i<=nn;i++)
			if(nn%i == 0){
				while(nn%i == 0) nn/=i;
				pr.push_back(i);
			}
		if(nn>1)
			pr.push_back(nn);
		int sz=pr.size();
		int ans=komb(n-1,f-1);
		for(int i=1;i<(1<<sz);i++){
			int m=1,cnt=0;
			for(int j=0;j<sz;j++)
				if(i&(1<<j))
					cnt++,m*=pr[j];
			if(n%m == 0)
				ans=(ans+(cnt%2?-1:1)*komb(n/m-1,f-1))%mod;
		}
		if(ans<0) ans+=mod;
		printf("%d\n",ans);
	}
	
	return 0;
}
