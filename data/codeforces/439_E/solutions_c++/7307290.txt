#include <cstdio>
#include <cstdlib>
#define limit 100000
#define P 1000000007
using namespace std;

long long q,n,m;
long long miu[100001],p[100001],primes,prime[100001];
long long jc[100001],ny[100001],ans;

long long calc(long long a,int b)
{
	long long ans=1;
	while (b)
	{
		if (b&1) ans=(ans*a)%P;
		a=(a*a)%P;
		b>>=1;
	}
	return ans;
}

long long work(long long d)
{
//	printf("	work  %I64d = ",d);
	d=(n-m*d)/d;
	if (d<0) return 0;
//	printf(" C ( %I64d , %I64d )= %I64d\n",d,d+m-1,((jc[d+m-1]*ny[d])%P*ny[m-1])%P);
	return ((jc[d+m-1]*ny[d])%P*ny[m-1])%P;
}

int main()
{
	for (int i=1;i<=limit;i++) p[i]=i; miu[1]=1;
    for (int i=2;i<=limit;i++)
    {
        if (p[i]==i) prime[++primes]=i,miu[i]=-1;
        for (int j=1;j<=primes&&prime[j]*i<=limit&&prime[j]<=p[i];j++)
        {
            p[i*prime[j]]=prime[j];
            miu[i*prime[j]]=prime[j]==p[i]?0:-miu[i];
        }
    }
    jc[0]=ny[0]=1;
    for (int i=1;i<=limit;i++) jc[i]=(jc[i-1]*i)%P;
    for (int i=1;i<=limit;i++) ny[i]=calc(jc[i],P-2);
	scanf("%I64d",&q);
	while (q--)
	{
		scanf("%I64d%I64d",&n,&m); ans=0;
		for (int i=1;i*i<=n;i++)
			if (n%i==0)
			{
				ans=(ans+miu[i]*work(i))%P;
				if (n/i!=i) ans=(ans+miu[n/i]*work(n/i))%P;
			}
		printf("%I64d\n",(ans+P)%P);
	}
}
