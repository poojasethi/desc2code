#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define FOR(i,a,b) for(int i=a; i<b; i++)
#define rep(i,n) FOR(i,0,n)
#define ff first
#define ss second
#define mod 1000000007

using namespace std;
typedef long long LL;
typedef vector<int> VI;
typedef vector< vector<int> > VVI;
typedef pair< int, int > PII;

int n,f;
const int N=100005;
LL fact[N],invfact[N];

LL modpow(LL a, LL x)
{
	LL ans=1;
	while(x){if(x&1)ans = ans*a%mod; a = (a*a)%mod;x /= 2;}
	return ans;
}

void init()
{
	fact[0] = 1;
	for(int i=1; i<N; i++)
		fact[i] =  (fact[i-1]*i)%mod;
	for(int i=0; i<N; i++)
		invfact[i] = modpow(fact[i],mod-2);
}

LL C(int n, int r)
{
	if(n<r)return 0;
	if(n==r)return 1;
	LL ans = (fact[n]*invfact[r])%mod;
	ans = (ans*invfact[n-r])%mod;
	return ans;
}

int main()
{
	init();
	int q;
	cin>>q;
	while(q--)
	{
		cin>>n>>f;
		VI p;
		int m=n;
		for(int i=2; i*i<=n; i++)
		{
			if(n%i==0)
				p.pb(i);
			while(n%i==0)
				n/=i;
		}
		if(n>1)
			p.pb(n);
		n=m;
		int cnt=p.size();
		LL ans=0;
		for(int i=0; i<(1<<cnt); i++)
		{
			int bitcnt=0;
			LL x=1;
			for(int j=0; j<cnt; j++)
			{
				if(i&(1<<j))
				{
					bitcnt++;
					x *= p[j];
				}
			}
			if(bitcnt%2)
				ans -= C(n/x-1,f-1);
			else
				ans += C(n/x-1,f-1);
			if(ans<0)
				ans += mod;
			ans %= mod;
		}
		cout<<ans<<endl;
	}
	return 0;
}
