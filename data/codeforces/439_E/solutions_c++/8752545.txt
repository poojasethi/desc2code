#include<iostream>
using namespace std;
#define mod 1000000007
typedef long long ll;
ll n,p[11],npr,ans,f,t,fa[111111]={1},fi[111111];
ll bin(ll a, ll n)
{
    ll ans=1;
    while(n)
    {
        if(n&1)
            ans*=a,ans%=mod;
        a*=a,a%=mod;
        n/=2;
    }
    return ans;
}
void dfs(ll pos, ll pr, ll cnt)
{
    if(pos>npr)
    {
        if(n/pr<f)
            return;
        if(cnt&1)
            ans-=(((fa[n/pr-1]*fi[f-1])%mod)*fi[n/pr-f])%mod,ans=(ans+mod)%mod;
        else
            ans+=(((fa[n/pr-1]*fi[f-1])%mod)*fi[n/pr-f])%mod,ans%=mod;
        return;
    }
    dfs(pos+1,pr,cnt);
    dfs(pos+1,pr*p[pos],cnt+1);
}
int main()
{
    for(ll i=1;i<=100000;i++)
    {
        fa[i]=fa[i-1]*i,fa[i]%=mod;
        fi[i]=bin(fa[i],mod-2);
    }
    fi[0]=bin(fa[0],mod-2);
    cin>>t;
    for(ll q=1;q<=t;q++)
    {
        cin>>n>>f;
        ll x=n;
        npr=ans=0;
        for(ll i=2;i*i<=x;i++)
            if(x%i==0)
            {
                while(x%i==0)
                x/=i;
                npr++;
                p[npr]=i;
            }
        if(x>1)
            npr++,p[npr]=x;
        dfs(1,1,0);
        cout<<ans<<endl;
    }
}
