def g():
 M=1000000007
 L=100002
 F=[1]
 for i in range(1,L):F.append(F[-1]*i%M)
 P=[0]*L
 for i in range(2,L):
  if P[i]<1:P[i::i]=[i]*((L-1)//i)
 s={}
 for q in[0]*input():
  n,f=map(int,raw_input().split())
  def c(n):
   if(n,f)not in s:s[n,f]=F[n-1]*pow(F[f-1]*F[n-f]%M,M-2,M)%M
   return s[n,f]
  a=[n];m=n
  while m>1:
	p=P[m];a+=[-i/p for i in a]
	while m%p<1:m/=p
  print(sum(v/abs(v)*c(abs(v))for v in a if abs(v)>=f))%M
g()
