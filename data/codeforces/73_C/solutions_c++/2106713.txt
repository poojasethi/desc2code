#include<iostream>
#include<memory.h>
#include<string>
#include<algorithm>

using namespace std;

string str;
int c,cost[27][27];
int mem[101][27][101];
bool mrk[101][27][101];

int solve(int r,int l,int k)
{
    if (r == str.size())
        return 0;
    if (mrk[r][l][k])
        return mem[r][l][k];
    mrk[r][l][k] = 1;
    int ans = solve(r+1,str[r] - 'a',k) + cost[l][str[r]-'a'];
    if (k > 0)
        for (int i=0;i<26;i++)
            ans = max(ans,solve(r+1,i,k-1) + cost[l][i]);
    return mem[r][l][k] = ans;
}

int main()
{
    int n,k;
    char a,b;
    cin>>str>>k>>n;
    for (int i=0;i<n;i++)
    {
        cin>>a>>b>>c;
        cost[a-'a'][b-'a'] = c;
    }
    
    cout<<solve(0,26,k)<<endl;
    return 0;
}
