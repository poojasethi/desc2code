#include <stdio.h>
#include <algorithm>
using namespace std;
main()
{
	int n,r[1000],i;
	double pi=3.14159265,ar,sum=0.0;
	scanf("%d",&n);
	r[0]=0;
	for(i=1;i<=n;i++)
	scanf("%d",&r[i]);
	sort(r,r+n+1);
	for(i=n;i>=1;i-=2)
	{
		ar=pi*((r[i]*r[i])-(r[i-1]*r[i-1]));
		sum+=ar;
	}
	printf("%lf",sum);
	return 0;
}
