#include<bits/stdc++.h>

using namespace std;

int main(){
	int n,i;
	double ans = 0;

	cin>>n;

	int arr[n+1];
	
	for(i=0;i<n;i++){
		cin>>arr[i];
	}

	arr[n] = 0;
	n++;

	sort(arr,(arr+n));

	for(i=n-1;i>0;i-=2){
		ans += (double)(pow(arr[i], 2) - pow(arr[i-1], 2));
	}

	ans = ans * 3.1415926536;

	cout<<ans<<endl;

	return 0;
}
