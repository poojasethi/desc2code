#include <iostream>
#include <algorithm>
#define pi 3.141592653589793
using namespace std;

main()
{
	int n,a[1007];
	float ans=0;
	cin>>n;
	for(int i=0;i<n;i++) cin>>a[i];
	sort(a,a+n);
	for(int i=n-1;i>=0;i--)
	{
		ans+=a[i]*a[i];
		i--;
		if(i>=0)
		ans-=a[i]*a[i];
	}
	cout<<pi*ans<<endl;
	return 0;
}
