#include <iostream>
#include <cstdio>
#include <algorithm>

#define PI 3.141592654

using namespace std;

int N,r[1010];
double area;

int main()
{
	cin >> N;
	for (int i=0;i<N;i++)
		cin >> r[i];
	N+=N%2;
	sort(r,r+N);
	for (int i=N-1;i>=0;i-=2)
		area+=(r[i]*r[i]-r[i-1]*r[i-1])*PI;
	printf("%.4lf\n",area);
	return 0;
}
