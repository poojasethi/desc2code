#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int n,a[1001],ans;
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n); reverse(a,a+n);
	for(int i=0;i<n;i+=2)
		ans += a[i]*a[i]-a[i+1]*a[i+1];
	printf("%.10lf\n",acos(-1.0)*ans);
	return 0;
}
