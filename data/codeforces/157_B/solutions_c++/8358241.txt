#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int n,a[1000];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	double ans=0,p=M_PI;
	for(int i=n;i;i--)
		if(((n-i)&1)==0)ans+=p*a[i]*a[i];else ans-=p*a[i]*a[i];
	printf("%.10lf\n",ans);
	return 0;
}
