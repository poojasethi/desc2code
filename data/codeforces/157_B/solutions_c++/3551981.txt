#include <iostream>
#include <algorithm>
using namespace std;
double a[101],area,pi=3.14159265359;
int i,n;
int main() 
{
    cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);
    for(i=n-1;i>=0;i-=2)
     {
       area+=(a[i]*a[i])-(a[i-1]*a[i-1]);
     }
 
     cout<<area*pi<<endl;
    return 0;

}
