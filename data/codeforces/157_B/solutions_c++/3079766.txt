#include <iostream>
#include <algorithm>
#define pi 3.14159265
using namespace std;
int main()
{
int n, a[103];
double S1;
cin>>n;
a[0]=0;
for (int i=1;i<=n;i++) cin>>a[i];
sort(a,a+n+1);
S1=pi*a[n]*a[n];
for (int i=n-1;i>0;i-=2) S1-=pi*(a[i]*a[i]-a[i-1]*a[i-1]);
cout.precision(15);
cout<<S1;
return 0;
}
