#include <cstdio>
#include <algorithm>

using namespace std;

int a[100];

int main() {
	int n;
	scanf("%i", &n);
	for(int i = 0; i < n; i++) scanf("%i", &a[i]);
	sort(a, a + n);
	int r = 0, k = 1;
	while(n--) {
		r += a[n] * a[n] * k;
		k = -k;
	}
	printf("%.4lf\n", r * static_cast<double>(3.141592653589793));
}
