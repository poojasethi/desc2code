#include<bits/stdc++.h>
using namespace std;

int n;
int r[1010];
int main(){
	cin >> n;
	for(int i=0;i<n;i++)
		cin >> r[i];
	sort(r,r+n);
	double area = 0.0;
	for(int i=n-1;i>=0;i-=2){
		area += M_PI * ((double)r[i]*r[i]);
		if(i)
			area-=M_PI * ((double)r[i-1]*r[i-1]);

	}

	cout << area << endl;
	

	return 0;
}
