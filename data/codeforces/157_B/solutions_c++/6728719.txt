#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int a1[101];
bool cmp(int a,int b)
{return a>b;}
int main()
{
	int n,o=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%d",a1+i);
	sort(a1,a1+n,cmp);
	o=0;
	for(int i=0;i<n;i+=2)
		o+=a1[i]*a1[i]-a1[i+1]*a1[i+1];
	printf("%.6lf",o*M_PI);
}
