#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n;
	double pi=acos(-1.0),res=0;
	scanf("%d",&n);
	double a[n];
	for(i=0;i<n;i++)
		scanf("%lf",&a[i]);
	sort(a,a+n);
	for(i=n-1;i>=0;i-=2)
	{
		res+=a[i]*a[i];
	}
	for(i=n-2;i>=0;i-=2)
	{
		res-=a[i]*a[i];
	}
	printf("%lf\n",pi*res);
}
