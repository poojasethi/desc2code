#include<bits/stdc++.h>
using namespace std;
double ans=0.0,pi=3.1415926536;
int main()
{
int n,x,a[105];cin>>n;
for(int i=0;i<n;i++)cin>>a[i];
sort(a,a+n);
if(n%2==1)
x=1;
else
x=-1;
for(int i=n-1;i>=0;i--)
{
if(i%2==0)
ans+=x*(a[i]*a[i]);
else
ans-=x*(a[i]*a[i]);
}
ans=ans*pi;
printf("%.9f",ans);
}
