#include <bits/stdc++.h>
using namespace std;

int main ( )
{
	int n;
	cin >> n;
	vector<int> r(n);
	for ( int i = 0; i < n; ++i )
		cin >> r[i];
	sort ( r.rbegin(), r.rend() );
	double ans = 0;
	for ( int i = 0; i < n; i++ )
		ans += r[i]*r[i] * (i&1?-1:1);
	printf ( "%.10lf\n", acos(-1)*ans );
	return 0;
}
