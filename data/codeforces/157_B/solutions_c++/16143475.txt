#include<bits/stdc++.h>
using namespace std;
#define pi 3.1415926536

int main()
{
	long s = 0;
	int n;
	cin >> n;
	int a[n + 1];
	for(int i=1;i<=n;i++)
	cin >> a[i];
	sort(a + 1, a + 1 + n);
	for(int i=n;i>=1;i--)
	s = s + ( ( (n - i + 1) & 1 ) ? (a[i] * a[i]) : -(a[i] * a[i]));
	printf("%0.10f", s * pi);
}
