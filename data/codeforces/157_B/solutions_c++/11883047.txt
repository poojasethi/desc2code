#include<iostream>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;

double pi=3.1415926535,z=0;
int a[222];

int main()
{
int n;
cin >> n;
for(int i=1;i<=n;i++)
cin >> a[i];
sort(a+1,a+n+1);
for(int i=1;i<=n;i++)
if(i%2)
z+=a[i]*a[i];
else
z-=a[i]*a[i];


if(!(n%2))
z*=-1;

cout<<z*pi;
return 0;
}
