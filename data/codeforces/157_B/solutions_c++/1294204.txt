#include<cstdio>
#include<cmath>
#include<algorithm>
using std::sort;

const double pi=4.0*atan(1.0);

int n,r[100];

int main(){
	scanf("%d",&n);
	for(int i=0;i<n;++i) scanf("%d",r+i);
	sort(r,r+n);
	double sum=0;
	for(int i=n-1,j=1;i>=0;--i,j=-j)
		sum+=r[i]*r[i]*pi*j;
	printf("%lf\n",sum);
	return 0;
}
