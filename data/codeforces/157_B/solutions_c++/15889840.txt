#include<cstdio>
#include<algorithm>
using namespace std;
int a[105],n;
float ans=0;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    sort(a+1,a+n+1);
        for(int i=2-(n%2);i<=n;i+=2)
            ans+=a[i]*a[i]-a[i-1]*a[i-1];
    printf("%f\n",ans*3.1415926536);
}
