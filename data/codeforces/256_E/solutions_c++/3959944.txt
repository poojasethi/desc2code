#include<stdio.h>
long long sg[400000][4][4];
int mt[4][4];
void upd(int nd,int a,int b,int x,int v)
{
    int i,j,k,l;
    if(x < a || b < x)
        return;
    if(a == b)
    {
        for(i=1;i<4;i++)
        for(j=1;j<4;j++)
            sg[nd][i][j]=0;
        if(v == 0)
        {
            for(i=1;i<4;i++)
            {
                sg[nd][i][i]=1;
            }
        }
        else
        {
            sg[nd][v][v]=1;
        }
        return;
    }
    int m=(a+b)>>1;
    upd(nd*2,a,m,x,v);
    upd(nd*2+1,m+1,b,x,v);
    for(i=1;i<4;i++)
    {
        for(j=1;j<4;j++)
        {
            sg[nd][i][j]=0;
            for(k=1;k<4;k++)
            {
                for(l=1;l<4;l++)
                if(mt[k][l] == 1)
                    sg[nd][i][j]+=sg[nd*2][i][k]*sg[nd*2+1][l][j];
                sg[nd][i][j]%=777777777;
            }
        }
    }
}
int main()
{
    int i,j,k,l,n,m;
    //freopen("in.in","r",stdin);
    scanf("%d%d",&n,&m);
    for(i=1;i<=3;i++)
    for(j=1;j<=3;j++)
    {
        mt[j][0]=1;
        mt[0][j]=1;
        scanf("%lld",&mt[i][j]);
    }
    mt[0][0]=1;
    //for(i=0;i<)
    long long sm=0;
    int idx,x;
    for(i=1;i<=n;i++)
        upd(1,1,n,i,0);
    for(i=0;i<m;i++)
    {
        scanf("%d%d",&idx,&x); 
        upd(1,1,n,idx,x);
        sm=0;
        for(j=1;j<4;j++)
        for(k=1;k<4;k++)
            sm+=sg[1][j][k];
        sm%=777777777;
        printf("%lld\n",sm);
    }
}
