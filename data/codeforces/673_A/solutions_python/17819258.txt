def input_list():
    s = raw_input()
    s = s.split(' ')
    l = [0 for i in range(len(s))]
    for i in range(len(s)):
        l[i] = int(s[i])
    return l

n = int(raw_input())
a = input_list()

a = [0] + a + [90]

t = 90
for i in range(n + 1):
    if a[i + 1] - a[i] > 15:
        t = a[i] + 15
        break
