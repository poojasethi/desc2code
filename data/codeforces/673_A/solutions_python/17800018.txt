num = int(raw_input())
mins = [int(x) for x in raw_input().split(' ')]
current = mins[0]
last = 0
i = 0
while current-last <= 15 and i < len(mins)-1:
	last = mins[i]
	current = mins[i+1]
	i += 1

if i == len(mins)-1:
	if current-last > 15:
		print last+15
	else:
		print min(current+15, 90)
else:
	print last+15
