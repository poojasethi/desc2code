intervals = int(raw_input())
minutes = map(int, raw_input().split())
minutes.insert(0,0)
minutes.insert(len(minutes), 90)


counter = 0


for i in range(intervals + 1):

    m = abs(minutes[i] - minutes[i+1])
    
    if m <= 15:
        counter += m
    else:
        counter += 15
        break

print counter
