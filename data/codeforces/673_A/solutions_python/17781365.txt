import math
import sys

def how_long_will_watch(imins):

	last_imin = 0

	for m in imins:
		if m-last_imin > 15:
			return last_imin + 15
		else:
			last_imin = m

	return last_imin + 15 if last_imin+15 < 90 else 90

lines = sys.stdin.readlines()
imins = map(int,lines[1].split())
print how_long_will_watch(imins)
