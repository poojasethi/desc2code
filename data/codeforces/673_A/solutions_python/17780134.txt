n = int(raw_input())
T = map(int, raw_input().split())

times = [False]*90

for t in T:
    times[t-1] = True

end = None
count = 0
for i, t in enumerate(times):
    if not t:
        count += 1
        if count == 15:
            end = i+1
            break
    else:
        count = 0

if end is None:
    print 90
else:
    print end
