#!/usr/bin/env python

import sys
import pdb

s = sys.stdin.readline().strip()
t = sys.stdin.readline().strip()

max = 0

def matches(a,b):
	"""docstring for matches"""
	num = 0
	for i in xrange(min(len(a), len(b))):
		if a[i] == b[i]: num += 1
	
	return num

for i in xrange(len(s)):
	temp = matches(s[i:], t)
	if temp > max:
		max = temp
	if max == len(t):
		break

for j in xrange(len(t)):
	temp = matches(s, t[j:])
	if temp > max:
		max = temp
	if max == len(t):
		break

print len(t)-max
