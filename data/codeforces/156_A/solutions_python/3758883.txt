import sys
s=raw_input()
u=raw_input()
max=0
def match(a,b):
    num=0
    for i in range(min(len(a),len(b))):
        if a[i]==b[i]:num+=1
    return num
for i in range(len(s)):
    temp=match(s[i:],u)
    if temp>max:
        max=temp
    if max==len(u):
        break
for i in range(len(u)):
    temp=match(s,u[i:])
    if temp>max:
        max=temp
    if max==len(u):
        break
