#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
ifs = sys.stdin
ofs = sys.stdout

def number_of_differences(A,B):
    return sum(1 for a,b in zip(A,B) if a!=b)

def solve(s,u):
    n = len(s)
    m = len(u)
    
    s = '%s%s%s' % ('.'*m, s, '.'*m)
    n += 2*m
    
    d_min = sys.maxint
    for i in range(n-m+1):
        ss = s[i:i+m]
        d = number_of_differences(ss, u)
        if d_min > d:
            d_min = d
        if d_min == 0:
            break

    return d_min

def numbers_from_line(d=' '):
    return [int(s) for s in ifs.readline().strip().split(d) if len(s.strip())>0]

s = ifs.readline().strip()
u = ifs.readline().strip()

a = solve(s,u)

ofs.write('%d\n' % (a))
