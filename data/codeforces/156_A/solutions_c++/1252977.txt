#include <iostream>

using namespace std;

string s1, s2;
int dp[2005][2005];

int main()
{
	cin>>s1>>s2;
	
	int L1 = s1.size(), L2 = s2.size(), ans = 0;
	
	for(int i=L1-1; i>=0; i--)
	{
		for(int j=L2-1; j>=0; j--)
		{
			dp[i][j] = (s1[i] == s2[j]) + dp[i+1][j+1];
			
			ans = max(ans, dp[i][j]);
		}
	}
	
	cout<<L2 - ans<<endl;
	
	return 0;
}
