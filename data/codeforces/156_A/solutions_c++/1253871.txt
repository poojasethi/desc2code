#include <iostream>
using namespace std;
int main() {
    string s,u;
    int i,j,c,r=1e4,l;
    cin>>s>>u;
    l=u.size();
    for (i=0;i<l;++i) s="_"+s+"_";
    for(i=1;i<s.size()-l;++i){
        c=0;
        for(j=0;j<l;++j)
            c+=(s[i+j]!=u[j]);
        r=min(r, c);
    }
    cout<<r;
}
