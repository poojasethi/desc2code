#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

string S,U;
int result=2000;

int main()
{
	cin >> S >> U;
	for (int i=0;i<U.size()-1;i++)
		S='#'+S+'#';
	for (int i=0;i<S.size()-U.size()+1;i++)
	{
		int cnt=0;
		for (int j=0;j<U.size();j++)
			cnt+=S[i+j]!=U[j];
		result=min(result,cnt);
	}
	cout << result << endl;
	return 0;
}
