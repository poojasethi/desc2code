#include<bits/stdc++.h>
using namespace std;
main()
{
    string s,t;int n,m,r,cur;
    cin>>s>>t;
    n=s.size();
    r=m=t.size();
    for (int i=-m;i<n;i++)
    {   cur=m;
        for (int j=0;j<m;j++)
            if (i+j>=0&&i+j<n&&s[i+j]==t[j])
                cur--;
        r=min(r,cur);
    }
    cout<<r;
}
