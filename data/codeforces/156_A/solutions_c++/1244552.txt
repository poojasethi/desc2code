#include<iostream>
using namespace std;

int main() {
	string a,b;
	cin>>a>>b;
	a=string(b.size(),'#')+a+string(b.size(),'#');
	int ans = b.size();
	for(int i=0;i+b.size()<=a.size();i++) {
		int cur = 0;
		for(int j=0;j<b.size();j++)
			cur+=b[j]!=a[i+j];
		ans = min(ans,cur);
	}
	cout<<ans<<endl;
}
