#include <stdio.h>
#include <algorithm>
using namespace std;

char x[8000];
char y[3000];

int main() {
	for (int i = 0; i < 8000; i++)
		x[i] = '0';
	scanf("%s %s", x+2000, y);
	int erg = 100000;
	for (int i = 0; x[i]; i++) {
		int anz = 0;
		for (int j = 0; y[j]; j++) {
			if (x[i+j] != y[j])
				anz++;
		}
		erg = min(erg,anz);
	}
	printf("%d\n", erg);
	return 0;
}
