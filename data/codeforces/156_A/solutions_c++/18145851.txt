#include <cstdio>
#include <cstring>
char s[6010],u[4010];
int main(){
	gets(s);
	gets(u);
	int n=strlen(s),m=strlen(u);
	for(int i=n-1;i>=0;i--)s[i+m]=s[i];
	for(int i=0;i<m;i++)s[i]=0;
	int ans=0;
	for(int i=0;i<m+n;i++){
		int cnt=0;
		for(int j=0;j<m;j++)if(u[j]==s[j+i])cnt++;
		if(cnt>ans)ans=cnt;
	}
	printf("%d",m-ans);
}
