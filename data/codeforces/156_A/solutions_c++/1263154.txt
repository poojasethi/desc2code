#include<iostream>
#include<string>
using namespace std;



int main(){

	string s,u;
	cin>>s>>u;
	for(int i=0;i<2000;i++) s="!"+s+"!";

	int value=5000;

	for(int i=0;i<s.length()-u.length()+1;i++){
		int count=0;
		for(int j=0;j<u.length();j++)
			if(u[j]!=s[i+j]) count++;
		value=min(value,count);

	}

	cout<<value<<endl;



return 0;
}
