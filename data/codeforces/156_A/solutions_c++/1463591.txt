#include <iostream>

using namespace std;


int main () {
  string a, b;
  cin >> a >> b;
  a = string(b.size()-1, '-') + a + string(b.size()-1, '-');
  int bst = 1000000;
  for (int i = 0; i <= a.size()-b.size(); i++) {
    int cur = 0;
    for (int j = 0; j < b.size(); j++) {
      if (a[i+j] != b[j]) cur++;
    }
    bst = min(bst,cur);
  }
  cout << bst << endl;
}
