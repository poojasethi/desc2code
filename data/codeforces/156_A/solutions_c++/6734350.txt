#include <iostream>

using namespace std;

string s, t;
int dp[2005][2005];

int main()
{
    cin>>s>>t;
    
    int L1 = s.size(), L2 = t.size(), ans = 0;
    
    for(int i=L1-1; i>=0; i--)
        for(int j=L2-1; j>=0; j--)
            ans = max(ans, dp[i][j] = (s[i] == t[j]) + dp[i+1][j+1]);
    
    cout<<L2 - ans<<endl;
    
    return 0;
}
