#include <iostream>
using namespace std;
string s,u;
int ans=9999999,cnt;
int main() {
	cin>>s>>u;s=string(4000,'$')+s+string(4000,'$');
	for(int i=0;i<(int)s.size()-(int)u.size();i++){
		cnt=0;
		for(int j=0;j<(int)u.size();j++){
			if(u[j]!=s[i+j])cnt++;
		}
		ans=min(ans,cnt);
	}
	cout<<ans<<endl;
	return 0;
}
