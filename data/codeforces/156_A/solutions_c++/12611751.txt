#include <cstdio>
#include <cstring>

int n, m, cnt, ans;
char s[2010], u[2010];

int main() {
	scanf("%s%s", s, u); 
	n=strlen(s);
	ans=m=strlen(u);
	for(int i=-m; i<n; i++) {
		cnt=m;
		for(int j=0; j<m; j++) if(i+j >= 0 && i+j<n && s[i+j ]== u[j]) cnt--;
		if(cnt<ans) ans=cnt;
	}
	printf("%d\n",ans);
	return 0;
}
