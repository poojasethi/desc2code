m, n = map(int,raw_input().split())

x = [[0] * (n + 1) for _ in xrange(m + 1)]

for mp in xrange(1, m + 1):
    t = map(int, raw_input().split())
    
    for p in xrange(1, n + 1):
        x[mp][p] = max(x[mp -1 ][p], x[mp][p - 1]) + t[p - 1]

print ' '.join(map(str, (x[mp][n] for  mp in xrange(1, m + 1))))
