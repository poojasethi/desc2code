npic, npainter = [int(x) for x in raw_input().strip().split()]

time = []
for i in range(npic):
  time.append([int(x) for x in raw_input().strip().split()])

now = [0 for x in range(npic)]
for i in range(npainter):
  now[0] += time[0][i]
  for k in range(1, npic):
    now[k] = time[k][i] + max(now[k-1], now[k])

print " ".join([str(x) for x in now])
