#!/usr/bin/env python
# -*- coding: UTF-8 -*-
__author__ = 'firago'

from sys import stdin, stdout

(m,n) = map(int,stdin.readline().strip().split())
mtrx = []
for line in stdin:
    mtrx.append(map(int,line.strip().split()))

for master in xrange(1,n):
    mtrx[0][master] += mtrx[0][master-1]

for picture in xrange(1,m):
    mtrx[picture][0] += mtrx[picture-1][0]


for picture in xrange(1,m):
    for master in xrange(1,n):
        if mtrx[picture][master-1] > mtrx[picture-1][master]:
            mtrx[picture][master] += mtrx[picture][master-1]
        else:
            mtrx[picture][master] +=mtrx[picture -1][master]

time = [0]*m
for picture in xrange(m):
    time[picture] = str(mtrx[picture][n-1])

stdout.write(' '.join(time))
