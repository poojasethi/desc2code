R=lambda:map(int, raw_input().strip().split(' '))

m, n =  R()

cost = [] 

cnt = 0
while cnt < m:
	cost.append(R())
	cnt += 1

ret = [0 for i in range(m)]

for i in range(0, n):
	tmp = [0] * m
	tmp[0] = ret[0] + cost[0][i]
	for j in range(1, m):
		if tmp[j - 1] >= ret[j]:
			tmp[j] = tmp[j - 1] + cost[j][i]
		else:
			tmp[j] = ret[j] + cost[j][i]

	ret = tmp
