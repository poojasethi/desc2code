pictures, painters = map(int, raw_input().split())
times = []
for i in range(pictures):
    times.append(list(map(int, raw_input().split())))
# print times
ans = [0] * pictures
for i in range(painters):
    ans[0] += times[0][i]
    for j in range(1, pictures):
        if painters == 0:
            ans[j] = ans[j-1] + times[j][i]
        else:
            ans[j] = max(ans[j-1], ans[j]) + times[j][i]
    # print ans

a = ' '.join(map(str, ans))
print a
