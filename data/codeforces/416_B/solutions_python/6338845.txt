#!/usr/bin/env python

import sys
import logging
import traceback

m, n = map(int, raw_input().split())
T = [map(int, raw_input().split()) for _ in range(m)]

# print T 

last = [0] * m

for i in range(n): 
    cur = 0
    for j in range(m): 
        last[j] = max(last[j], cur) + T[j][i]
        cur = last[j]
    # print last 

print ' '.join(map(str, last))
