"""
4 2
2 5
3 1
5 3
10 1
"""

m, n = map(int, raw_input().split())
ns = []
times = [0] * m

for _ in xrange(m):
    ns.append(map(int, raw_input().split()))

for i in xrange(n):
    for j in xrange(m):
        if j == 0:
            times[j] = times[j] + ns[j][i]
        else:
            times[j] = max(times[j], times[j - 1]) + ns[j][i]

print ' '.join(map(str, times))
