pictures, painters = map(int, raw_input().split())

inp = []
for i in range(pictures):
	inp.append(map(int, raw_input().split()))
inp = zip(*inp)

p = inp[0]
last = [0] * pictures
last[0] = p[0]
for i in range(1, pictures):
	last[i] = last[i-1] + p[i]

for j in range(1, painters):
	p = inp[j]
	this = [0] * pictures
	this[0] = last[0] + p[0]
	for i in range(1, pictures):
		this[i] = max(this[i-1], last[i]) + p[i]
	last = this

print ' '.join(map(str, last))
