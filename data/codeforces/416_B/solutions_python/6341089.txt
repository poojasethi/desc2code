import sys

inData={}
count=1
trig=True
for line in sys.stdin:
        l=line.rstrip().split()
	if trig:
		paint=int(l[-1])
		pic=int(l[0])
		art=[[0 for j in range(pic+1)] for i in range(paint+1)]
		trig=False
	else:
		inData[count]=[int(i) for i in l]
		count+=1

for p in range(1,paint+1):
	for pi in range(1,pic+1):
		art[p][pi]=max([art[p-1][pi],art[p][pi-1]])+inData[pi][p-1]
print " ".join([str(i) for i in art[-1][1:]])
