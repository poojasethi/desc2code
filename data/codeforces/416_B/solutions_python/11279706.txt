#!/usr/bin/env python

m, n = map(int, raw_input().split())

finish = [[0] * n]

result = []
for _ in range(m):
	times = map(int, raw_input().split())

	ready = [0] * n
	ready[0] = finish[-1][0] + times[0]

	for painter in range(1, n):
		ready[painter] = max(ready[painter-1], finish[-1][painter]) + times[painter]

	#print ready
	result.append(ready[-1])
	finish.append(ready)

for x in result:
	print x,
print ''
