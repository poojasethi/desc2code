m, n = map(int, raw_input().split())
T = []
for i in range(m):
    T.append(map(int, raw_input().split()))
memo = [[0] * n for _ in range(m)]
memo[0][0] = T[0][0]
for i in range(1, n):
    memo[0][i] = memo[0][i - 1] + T[0][i]
for i in range(1, m):
    for j in range(n):
        if j == 0:
            memo[i][j] = memo[i - 1][j] + T[i][j]
        else:
            memo[i][j] = max(memo[i - 1][j], memo[i][j - 1]) + T[i][j]
print ' '.join(map(str, zip(*memo)[-1]))
