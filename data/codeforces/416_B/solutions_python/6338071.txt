m, n = map(int, raw_input().strip().split(' '))

ret = []
last = [0 for _ in xrange(n)]

for i in xrange(1, m + 1):
    curr = map(int, raw_input().strip().split(' '))
    finish = [0 for _ in xrange(n)]
    for j in xrange(n):
        if j == 0:
            finish[j] = last[j] + curr[j]
        else:
            finish[j] = max(last[j], finish[j - 1]) + curr[j]
    ret.append(finish[-1])
    last = finish

print " ".join(map(str, ret))
