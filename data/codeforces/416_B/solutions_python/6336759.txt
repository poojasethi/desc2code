m, n = map(int, raw_input().split())

T = []
for i in range(m):
  T.append(map(int, raw_input().split()))

time = []
for j in range(n):
  time.append([])
  time[j].append(0)
  for i in range(m):
    if j == 0:
      time[j].append(time[j][-1]+T[i][j])
    else:
      time[j].append(max(time[j][-1], time[j-1][i+1])+T[i][j])
print ' '.join(map(str, time[n-1][1:]))
