m, n = map(int, raw_input().strip().split())
t = [map(int, raw_input().strip().split()) for i in xrange(m)]
#t[painting][painter]
totalt = [0 for i in xrange(m)]
for j in xrange(n):
    totalt[0] += t[0][j]
    for i in xrange(1, m):
        totalt[i] = max(totalt[i], totalt[i - 1]) + t[i][j]

print ' '.join(map(str, totalt))
