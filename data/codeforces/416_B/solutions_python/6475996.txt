import sys

m, n = map( int, raw_input().split() )
freeAt = [0]*n
for i in xrange(m):
    ti = map( int, raw_input().split() )
    timei = 0
    for j in xrange(n):
        startAt = max(timei, freeAt[j])
        timei = startAt + ti[j]
        freeAt[j] = timei
    if i > 0:
        sys.stdout.write(' ')
    sys.stdout.write(str(timei))
