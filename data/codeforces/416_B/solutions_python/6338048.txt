import math
from fractions import gcd

def main():
    n, m = [int(i) for i in raw_input().strip().split(" ")]

    whencan = [0] * m
    R = [0] * n

    for i in range(n):
        times = [int(k) for k in raw_input().strip().split(" ")]

        T = 0

        for j in range(m):
            T = max(T,whencan[j])
            T = T + times[j]
            whencan[j] = T

        R[i] = T

    print " ".join([r.__str__() for r in R])

    return 0

if __name__ == "__main__":
    main()
