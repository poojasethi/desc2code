# n: number of painters; m: number of paints
[m, n] = [int(x) for x in raw_input().split()]
painting_time = []
for i in range(m):
    painting_time.append([int(x) for x in raw_input().split()])

finish_time = [0] * m
for i in range(n):
    free_at = 0
    for j in range(m):
        start_time = max(free_at, finish_time[j])
        finish_time[j] = start_time + painting_time[j][i]
        free_at = finish_time[j]

print " ".join(map(str, finish_time))
