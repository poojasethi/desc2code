R = lambda : map(int, raw_input().split())
m,n = R()
prev = [0 for i in xrange(n)]
for i in xrange(m):
    current = R()
    for j in xrange(n):
        if j == 0:
            current[j] += prev[j]
        else:
            if max(current[j-1], prev[j]) == current[j-1]:
                current[j] += current[j-1]
            else:
                current[j] += prev[j]
    print current[-1],
    prev = current
