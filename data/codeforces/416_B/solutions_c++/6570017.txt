#include<iostream>
#include<string>
#include<cstdio>
using namespace std;
int main() {
    int m, n, a[8]={};
    cin>>m>>n;
    for(int i=0; i<m; ++i) {
        int x;
        for(int j=1; j<=n; ++j) {
            cin>>x;
            a[j]=max(a[j], a[j-1])+x;
        }
        cout<<a[n]<<" ";
    }
}
