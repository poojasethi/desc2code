#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int m,n,t[50000+10][6],f[50000+10][6];
int main(void)
{
	cin>>m>>n;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=n;j++)
		{
			cin>>t[i][j];
			f[i][j] = max(f[i-1][j],f[i][j-1]) + t[i][j];
		}
	for(int i=1;i<=m;i++)
		cout<<f[i][n]<<" ";
	return 0;
}
