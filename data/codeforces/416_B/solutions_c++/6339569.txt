#include<iostream>
#include<algorithm>
#define Nmax 50010
using namespace std;

int A[Nmax][6];
int N, M, i, a;
int main()
{
	cin >> M >> N;
	for (i = 1; i <= M; ++i)
	for (int j = 1; j <= N;++j)
	{
		cin >> a;
		A[i][j] = max(A[i - 1][j], A[i][j - 1]) + a;
	}
	for (i = 1; i <= M; ++i)
		cout << A[i][N] << ' ';

	return 0;
}
