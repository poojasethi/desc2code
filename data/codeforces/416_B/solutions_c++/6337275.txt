#include <cstdio>
#include <algorithm>
using namespace std;

const int M = 50005;

int m, n;
long long a[M][10];

int main() {
	scanf("%d%d", &m, &n);
	for (int i = 1; i <= m; ++i) {
		for (int j = 1; j <= n; ++j) {
			scanf("%I64d", &a[i][j]);
			a[i][j] += max(a[i-1][j], a[i][j-1]);
		}
		printf("%I64d ", a[i][n]);
	}
}
