#include<iostream>
using namespace std;
int m,n;
int t[50010][7];
int d[50010][7];
int main()
{
	cin>>m>>n;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=n;j++)
			cin>>t[i][j];
	for(int i=1;i<=m;i++)
	{
		for(int j=1;j<=n;j++)
		{
			d[i][j]=max(d[i-1][j],d[i][j-1])+t[i][j];
		}
	}
	for(int i=1;i<=m;i++)
		cout<<d[i][n]<<" ";
	cout<<endl;
}
