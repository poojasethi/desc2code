#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, m, t, p[9], i;
    cin >> m >> n;
    for (i = 0 ; i < 9; ++i)p[i]=0;
    while(m --)
    {
        for(i = 1; i <= n; ++i)
        {
            cin >> t;
            p[i] = max(p[i - 1], p[i]) + t;
        }
        cout << p[n] << " ";
    }
    return 0;
}
