#include<cstdio>
#include<algorithm>
using namespace std;
int a1[6];
int main()
{
    int n,m,a;
    scanf("%d%d",&m,&n);
    while(m--)
    {
        for(int i=1;i<=n;++i)
        {
            scanf("%d",&a);
            a1[i]=max(a1[i],a1[i-1])+a;
        }
        printf("%d",a1[n]);
        putchar(m?' ':'\n');
    }
}
