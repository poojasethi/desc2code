#include <cstdio>
#include <algorithm>

int M,N,T[50000+7][6];
int F[50000+7][6];
int main()
{
	scanf("%d %d", &M, &N);
	for (int i=1; i<=M; i++)
	{
		for (int j=1; j<=N; j++)
		{
			scanf("%d", &T[i][j]);
			F[i][j]=std::max(F[i-1][j], F[i][j-1])+T[i][j];
		}
	}for(int i=1;i<=M;i++)printf("%d ", F[i][N]);
	return 0;
}
