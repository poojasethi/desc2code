#include<iostream>
#include<cstdio>
using namespace std;
int n,m,t,ft[50002][6];
int main()
{
	cin>>m>>n;
	for(int i=1;i<=m;++i)
	{
		for(int j=1,t;j<=n;++j)
		{
			scanf("%d",&t);
			ft[i][j]=(ft[i][j-1]>ft[i-1][j]?ft[i][j-1]:ft[i-1][j])+t;
		}
	}
	cout<<ft[1][n];
	for(int i=2;i<=m;++i)
		cout<<" "<<ft[i][n];
	cout<<endl;

}
