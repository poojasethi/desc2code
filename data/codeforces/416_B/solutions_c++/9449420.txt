#include<bits/stdc++.h>

using namespace std;

int main()
{
    int m,n,i,j;
    cin>>m>>n;
    int t[50001][6];
    int a[50001][6]={0};
    for(i=1;i<=m;i++) for(j=1;j<=n;j++) cin>>t[i][j];
    for(i=1;i<=n;i++) for(j=1;j<=m;j++) a[j][i]=max(a[j-1][i],a[j][i-1])+t[j][i];
    for(i=1;i<=m;i++) cout<<a[i][n]<<" ";
    return 0;
}
