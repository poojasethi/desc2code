#include<bits/stdc++.h>

using namespace std;

const int Maxn=1e5+7;

int end[Maxn];

main()
{
  ios_base::sync_with_stdio(0);
  int m,n;
  cin>>m>>n;
  for(int i=0;i<m;i++)
    {
      for(int a,j=1;j<=n;j++)
	{
	  cin>>a;
	  end[j]=max(a+end[j-1],end[j]+a);
	}
      cout<<end[n]<<endl;
    }
}
