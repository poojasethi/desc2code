#include<cstdio>
#include<iostream>
using namespace std;
int main() {
    int m, n, a[8]={};
    scanf("%d%d", &m, &n);
    for(int i=0; i<m; ++i) {
        int x;
        for(int j=1; j<=n; ++j) {
            scanf("%d", &x);
            a[j]=max(a[j], a[j-1])+x;
        }
        printf("%d%c", a[n], i<m?' ':'\n');
    }
}
