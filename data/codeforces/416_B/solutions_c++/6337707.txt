#include<cstdio>
#include<algorithm>
using namespace std;
int o[60000][10];
int main()
{
	int m,n,t,i,j;
	scanf("%d%d",&m,&n);
	for(i=1;i<=m;i++)
		for(j=1;j<=n;j++)
		{
			scanf("%d",&t);
			o[i][j]=max(o[i][j-1],o[i-1][j])+t;
		}
	for(i=1;i<m;i++)
		printf("%d ",o[i][n]);
	printf("%d\n",o[i][n]);
}
