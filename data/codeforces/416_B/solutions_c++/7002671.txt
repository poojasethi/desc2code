#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;
int a[55000][7],i,j,n,m,x;
int main()
{
	scanf("%d%d",&n,&m);
    for (i = 1; i <= n;cout<<a[i++][m]<<' ')
	{
		for (j = 1; j <= m; cin >> x, a[i][j++] =x+ max(a[i - 1][j], a[i][j - 1]));
	}
}
