#include <iostream>
using namespace std;

int a[50001][6];
int main()
{
	int m, n, i, j;
	cin>>m>>n;
	for (i=1; i<=m; i++) for (j=1; j<=n; j++) cin>>a[i][j];
	for (i=1; i<=n; i++) for (j=1; j<=m; j++) a[j][i]+=max(a[j-1][i],a[j][i-1]);
	for (i=1; i<=m; i++) cout<<a[i][n]<<" ";
	cout<<endl;	 
	return 0;
}
