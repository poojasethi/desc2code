#include<stdio.h>
int main()
{
	int a[6];
	int m,n,i,b;

	scanf("%d%d",&m,&n);
	for(i=1;i<=n;i++)
		a[i]=0;
	while(m--)
	{
		for(i=1;i<=n;i++)
		{
			scanf("%d",&b);
			if(i-1&&a[i-1]>a[i])a[i]=a[i-1];
			a[i]+=b;
		}
		printf("%d",a[n]);
		if(m)putchar(' ');
		else putchar('\n');
	}
	return 0;
}
