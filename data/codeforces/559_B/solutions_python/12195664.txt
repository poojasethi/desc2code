def check_equiv(s1,s2):
    l=len(s1)
    if s1==s2:
        return True
    elif l%2==0:
        if check_equiv(s1[:l/2],s2[l/2:]) and check_equiv(s1[l/2:],s2[:l/2]):
            return True
        elif check_equiv(s1[:l/2],s2[:l/2]) and check_equiv(s1[l/2:],s2[l/2:]):
            return True
    else:
        return False
