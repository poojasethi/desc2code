def is_equal(s0, s1):
    if s0 == s1:
        return True
    L = len(s0)
    if L % 2 > 0:
        return False
    s00, s01 = s0[:L/2], s0[L/2:]
    s10, s11 = s1[:L/2], s1[L/2:]
    return ((is_equal(s00, s11) and is_equal(s01, s10)) or
            (is_equal(s00, s10) and is_equal(s01, s11)))

if is_equal(raw_input(), raw_input()):
    print('YES')
else:
    print('NO')
