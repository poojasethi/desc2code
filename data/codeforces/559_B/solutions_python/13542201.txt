a = raw_input('')
b = raw_input('')

def neq(s,e,l,w):
    if l % 2 == 1:
        return a[s:e] if w == 'a' else b[s:e]
    h = l/2
    s1 =  neq(s,s+h,h,w)
    s2 =  neq(s+h,e,h,w)
    return s1+s2 if s1 < s2 else s2+s1

l = len(a)

if neq(0,l,l,'a') == neq(0,l,l,'b'):
    print 'YES'
else:
    print 'NO'
