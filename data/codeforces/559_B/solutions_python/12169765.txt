def f(s):
    s = list(s)
    L = l = len(s)
    c = 0
    while l % 2 == 0:
        l /= 2
        c += 1
    for i in xrange(c):
        for j in xrange(0, L, l * 2):
            if s[j:j+l] > s[j+l:j+l+l]:
                s[j:j+l], s[j+l:j+l+l] = s[j+l:j+l+l], s[j:j+l]
        l *= 2
    return ''.join(s)

def main():
    s = raw_input().strip()
    t = raw_input().strip()
    print "YES" if f(s) == f(t) else "NO"
main()
