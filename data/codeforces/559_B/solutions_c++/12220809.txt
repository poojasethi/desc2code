#include<iostream>
#include<string>
using namespace std;
string is(string a)
{
	int n=a.length();
	if(n&1)return a;
	string a1=is(a.substr(0,n/2));
	string a2=is(a.substr(n/2));
	if(a1>a2)swap(a1,a2);
	return a1+a2;
}
int main()
{
	string s,d;
	cin>>s>>d;
	if(is(s)==is(d))cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
	//system("pause");
	return 0;
};
