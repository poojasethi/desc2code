#include <bits/stdc++.h>
using namespace std;
string f(string s){
	int n=s.size();
	if(n%2) return s;
	n/=2;
	string a="",b="";
	for(int i=0;i<n;i++){
		a+=s[i];
		b+=s[i+n];
	}
	a=f(a);
	b=f(b);
	return (a<b)? a+b:b+a;
}
int main(){
	string a,b;
	cin>>a>>b;
	if(a==b) puts("YES");
	else{
		a=f(a);
		b=f(b);
		puts((a==b)?"YES":"NO");
	}
	return 0;
}
