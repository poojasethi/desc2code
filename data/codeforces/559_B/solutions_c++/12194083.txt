#include<iostream>
#include<cstdio>
using namespace std;
string pd(string s) {
    if (s.size()%2==1) return s;
    string s1=pd(s.substr(0,s.size()/2));
    string s2=pd(s.substr(s.size()/2));
    if (s1<s2) return s1+s2;
    else return s2+s1;
}
int main(){
    string a,b;
    ios::sync_with_stdio(false);
    cin>>a>>b;
    puts(pd(a)==pd(b)?"YES":"NO");
    return 0;
}
