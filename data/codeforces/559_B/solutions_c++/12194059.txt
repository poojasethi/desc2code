#include <iostream>
using namespace std;
string pd(string s) {
    if (s.size()%2==1) return s;
    string s1=pd(s.substr(0,s.size()/2));
    string s2=pd(s.substr(s.size()/2));
    if (s1<s2) return s1+s2;
    else return s2+s1;
}
int main(){
    string a,b;
    cin>>a>>b;
    if (pd(a)==pd(b)) cout<<"YES";
    else cout<<"NO";
    return 0;
}
