#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
string f(string s)
{
    int n=s.length();
    if(n&1)return s;
    string s1=f(s.substr(0,n/2)),s2=f(s.substr(n/2));
    return s1<s2?s1+s2:s2+s1;
}
int main()
{
    string s1,s2;
    cin>>s1>>s2;
    puts(f(s1)==f(s2)?"YES":"NO");
}
