#include <iostream>
#include <string>
using namespace std;
string canon(string s)
{
	if (s.size()&1)
		return s;
	else
	{
		string l=canon(s.substr(0,s.size()>>1)),r=canon(s.substr(s.size()>>1));
		return l<r?(l+r):(r+l);
	}
}
int main()
{
	string s1,s2;
	cin>>s1>>s2;
	cout<<(canon(s1)==canon(s2)?"YES":"NO");
	return 0;
}
