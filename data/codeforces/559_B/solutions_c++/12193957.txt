#include <bits/stdc++.h>
using namespace std;
string m(string s) {
  if (s.length() % 2) return s;
  string a = m(s.substr(0, s.length() / 2));
  string b = m(s.substr(s.length() / 2));
  return a < b ? a + b : b + a;
}
string s, t;
int main() {
  cin >> s >> t;
  printf(m(s) == m(t) ? "YES\n" : "NO\n");
  return 0;
}
