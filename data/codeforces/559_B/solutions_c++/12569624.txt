#include<bits/stdc++.h>
using namespace std;
string hsh(string s){
	if (s.size()&1) return s;
	string s1=hsh(s.substr(0,s.size()>>1)),s2=hsh(s.substr(s.size()>>1));
	return s1<s2?(s1+s2):(s2+s1);
}
int main(){
	string s1,s2;
	cin>>s1>>s2;
	if (hsh(s1)==hsh(s2)) cout<<"YES"; else cout<<"NO";
}
