#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
string cal(string s)
{
	if(s.size()%2==1)return s;
	int t=s.size()/2;
	string s1=cal(s.substr(0,t));
	string s2=cal(s.substr(t,t));
	if(s1<s2)return s1+s2;
	else return s2+s1;
}
int main()
{
	string a,b;
	cin>>a>>b;
	if(cal(a)==cal(b))puts("YES");
	else puts("NO");
	return 0;
}
