#include<bits/stdc++.h>
using namespace std;
string sort(string s)
{
    if(s.size()%2!=0)return s;
    string s1 = sort(s.substr(0,s.size()/2));
    string s2 = sort(s.substr(s.size()/2,s.size()/2));
    if(s1>s2)return s1+s2;
    return s2+s1;
}
int main ()
{
    string a,b;cin >> a >> b;
    puts((sort(a)==sort(b))?"YES":"NO");
}
