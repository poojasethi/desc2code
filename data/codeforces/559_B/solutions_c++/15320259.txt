#include<bits/stdc++.h>
using namespace std;
string minn(string s)
{
	if(s.size()&1)
	return s;
	string s1 = minn(s.substr(0,s.size()/2));
	string s2 = minn(s.substr(s.size()/2,s.size()));
	if(s1<s2)
	return s1+s2;
	return s2+s1;
}
int main()
{
	string s1;
	string s2;
	cin>>s1>>s2;
	s1=minn(s1);
	s2=minn(s2);
	cout<<((s1==s2)?"YES\n":"NO\n");
}
