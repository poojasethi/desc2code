#include <bits/stdc++.h>

using namespace std;

string s1, s2;

string cal(const string &s) {
	if (s.size() % 2)
		return s;
		
	int m = s.size() >> 1;
	
	string x = cal(s.substr(0, m)), y = cal(s.substr(m, m));
	
	if (x < y)
		return x + y;
	else
		return y + x;
}

int main() {
	cin >> s1 >> s2;
	s1 = cal(s1);
	s2 = cal(s2);
	cout << ((s1 == s2) ? "YES" : "NO");
}
