#include <iostream>
#include <string>
using namespace std;
#define s string 

s f(s a) {
    if (a.length() % 2 == 1) return a;
    s a1 = f(a.substr(0, a.length()/2));
    s a2 = f(a.substr(a.length()/2));
    if (a1 < a2) return a1 + a2;
    else return a2 + a1;
}

int main() {
    s a,b;
    cin>>a>>b;
    (f(a)==f(b)) ? cout<<"YES":cout<<"NO";
    return 0;
}
