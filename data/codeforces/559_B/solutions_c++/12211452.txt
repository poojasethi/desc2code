#include<bits/stdc++.h>
using namespace std;
string f(string x){
    int siz = x.size();
    if(siz % 2 == 1) return x;
    string a = f(x.substr(0, siz / 2));
    string b = f(x.substr(siz / 2, siz / 2));
    if(a < b) return a + b;
    else return b + a;
}
int main(){
    string s, t;
    cin>>s>>t;
    if(f(s) == f(t)) cout<<"YES\n";
    else cout<<"NO\n";
}
