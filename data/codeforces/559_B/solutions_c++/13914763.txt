using namespace std;
string func(string s)
{
    int n=s.size();
    if(n&1)return s;
    string a,b;
    a=func(s.substr(0,n/2));
    b=func(s.substr(n/2));
    return min(a+b,b+a);
}
int main()
{
    string s,r;
    cin>>s>>r;
    if(func(s)==func(r))cout<<"YES";
    else cout<<"NO";
}
