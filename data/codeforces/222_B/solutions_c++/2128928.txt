#include <iostream>
using namespace std;
main()
{ long long int i,j,n,m,k,a[1001][1001],r[1001],c[1001],x,y,t;
char z;
cin>>n>>m>>k;
 for(i=1;i<=n;i++)
   for(j=1;j<=m;j++)
        { cin>>a[i][j];     
            r[i]=i;c[j]=j;}
 for(i=1;i<=k;i++)
{   cin>>z>>x>>y;   
       if(z=='g')
        cout<<a[r[x]][c[y]]<<endl;
        else if (z=='r')
             {t=r[x]; r[x]=r[y];r[y]=t;}               
          else  
            {t=c[x]; c[x]=c[y];c[y]=t;} 
}
return 0;}
