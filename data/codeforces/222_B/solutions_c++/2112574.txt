#include <iostream>

using namespace std;

int main()
{
	int m,n,k;
	cin >>m >>n >>k;
	int arr[m+1][n+1];
	int r[m+1],c[n+1];
	
	for(int i=1;i<=m;i++)
	{
		r[i]=i;
		for(int j=1;j<=n;j++)
		{
			if(i==1)
				c[j]=j;
			cin >> arr[i][j];
		}
	}

	while(k--)
	{
		char q;
		int i,j;
		cin >> q >>i>>j;
		if(q == 'c')
		{
			int temp = c[i];
			c[i]=c[j];
			c[j]=temp;
		}
		else if(q == 'r')
		{
			int temp = r[i];
			r[i]=r[j];
			r[j]=temp;
		}
		else
		{
			cout << arr[r[i]][c[j]]<<endl;
		}
	}
	return 0;
}
