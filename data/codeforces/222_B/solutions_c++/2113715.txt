#include <algorithm>
#include <cstdio>
using namespace std;
const int MaxN= 1111;
int N, M, K, x[MaxN], y[MaxN], a[MaxN][MaxN], n, m;
char s[5];
int main(){
	scanf("%d%d%d", &N, &M, &K);
	for(int i= 1;i<= N;++ i){
		for(int j= 1;j<= M;++ j){
			scanf("%d", a[i]+j);
			y[j]= j;
		}
		x[i]= i;
	}
	while(K --){
		scanf("%s%d%d", s, &n, &m);
		if(*s=='r') swap(x[n], x[m]);
		if(*s=='c') swap(y[n], y[m]);
		if(*s=='g') printf("%d\n", a[x[n]][y[m]]);
	}
	return 0;
}
