#include <iostream>
using namespace std;
int n,m,k,i,j,a[1001][1001],x,y,t,v[1001],g[1001];
char c;
int main()
{
   cin>>n>>m>>k;
   for(i=0;i<n;i++)
	   for(j=0;j<m;j++)
		   cin>>a[i][j];
	for(i=0;i<n;i++)
		g[i]=i; 
	for(i=0;i<m;i++)
		v[i]=i;
	while (k--)
   {
		cin>>c>>x>>y; x--; y--;
		if (c=='c')
			swap(v[x],v[y]);
		if (c=='r')
			swap(g[x],g[y]);
		if (c=='g')
		{
			x=g[x];
			y=v[y];
			cout<<a[x][y]<<endl;
		}
   }
}
