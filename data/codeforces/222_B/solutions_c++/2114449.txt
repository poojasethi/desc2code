#include<iostream>
using namespace std;
int main()
{
int n, m, k, x[1009], y[1009], l, r, a[1009][1009], i, j;
char c;
cin>>n>>m>>k;
for(i = 1; i <= n; i ++)
{
    for(j = 1; j <= m; j ++)
    {
    cin>>a[i][j];
    }
}

for(i = 1; i <= n; i ++)
    x[i] = i;

for(i = 1; i <= m; i ++)
    y[i] = i;

for(i = 1; i <= k; i ++)
{
    cin>>c>>l>>r;
    if(c == 'c') swap(y[l], y[r]);
    if(c == 'r') swap(x[l], x[r]);
    if(c == 'g') cout<<a[x[l]][y[r]]<<endl;
}
return 0;
}
