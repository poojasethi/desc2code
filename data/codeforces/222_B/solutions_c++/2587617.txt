#include<stdio.h>

int map[1001][1001];
int swapr[1001],swapc[1001];
int main()
{
	int n,m,ri,i,j,x,y;
	char a[2];
	scanf("%d%d%d",&n,&m,&ri);
	for(i=0;i<n;i++){
		swapr[i]=i;
		for(j=0;j<m;j++){
			scanf("%d",&map[i][j]);
			swapc[j]=j;
		}
	}
	while(ri--){
		scanf("%s%d%d",a,&x,&y);
		x--;y--;
		if(a[0]=='g')
			printf("%d\n",map[swapr[x]][swapc[y]]);
		else if(a[0]=='r')
			swapr[x]^=swapr[y]^=swapr[x]^=swapr[y];
		else
			swapc[x]^=swapc[y]^=swapc[x]^=swapc[y];
	}
	return 0;
}
