#include<cstdio>
#include<iostream>
using namespace std;
int main(){
int x,y,q,s[1001][1001],r[1001],c[1001];
scanf("%d %d %d",&x,&y,&q);
for(int i=1;i<=x;i++){
for(int j=1;j<=y;j++){
scanf("%d",&s[i][j]);
}
}
for(int i=1;i<=x;i++) r[i]=i;
for(int i=1;i<=y;i++) c[i]=i;
while(q--){
char cmd;
int px,py;
scanf("\n%c %d %d",&cmd,&px,&py);
if(cmd=='g') printf("%d\n",s[r[px]][c[py]]);
if(cmd=='r') swap(r[px],r[py]);
if(cmd=='c') swap(c[px],c[py]);
}
return 0;
}
