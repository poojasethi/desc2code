# include <iostream>
# include <algorithm>
# define DIM 1003
using namespace std;
int n, m, k, l[DIM], c[DIM], a[DIM][DIM];

int main ()
{
	cin>>n>>m>>k;
	for(int i=1;i<=n;++i)
		for(int j=1;j<=m;++j)
			cin>>a[i][j];
	for(int i=1;i<=n;++i)	
		l[i]=i;
	for(int i=1;i<=m;++i)
		c[i]=i;
	char ch;
	int x, y;
	
	for(;k--;)
	{
		cin.get();
		cin>>ch>>x>>y;
		if (ch=='r')
			swap(l[x],l[y]);
		else if (ch=='c')
			swap(c[x],c[y]);
		else 
			cout<<a[l[x]][c[y]]<<"\n";
	}
	return 0;
}
