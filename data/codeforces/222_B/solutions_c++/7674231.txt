#include<cstdio>
#include<algorithm>
using namespace std;
int a1[1000][1000],a2[1000],a3[1000];
int main()
{
	int n,m,k,a,b,c;char op;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;++i)a2[i]=i;
	for(int i=0;i<m;++i)a3[i]=i;
	for(int i=0;i<n;++i)
	for(int j=0;j<m;++j)
		scanf("%d",a1[i]+j);
	while(k--)
	{
		getchar();
		scanf("%c%d%d",&op,&a,&b);
		--a,--b;
		if(op=='r')swap(a2[a],a2[b]);
		else if(op=='c')swap(a3[a],a3[b]);
		else printf("%d\n",a1[a2[a]][a3[b]]);
	}
}
