#include <iostream>

using namespace std;

int a[1003][1003], s[1003], d[1003], n, m, x, y, k;
char ch;

int main()
{
    cin >> n >> m >> k;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            cin >> a[i][j], s[i] = i, d[j] = j;

    for (int i = 1; i <= k; i++)
    {
        cin >> ch >> x >> y;
        if (ch == 'r') swap(s[x], s[y]);
        if (ch == 'c') swap(d[x], d[y]);
        if (ch == 'g') cout << a[s[x]][d[y]] << endl;
    }

    return 0;
}
