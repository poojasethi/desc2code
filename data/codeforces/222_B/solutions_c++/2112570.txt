#include <iostream>
#include <string>
#include <cmath>
#include <algorithm>
using namespace std;

int a[1000][1000], n, m, k;
char c;
int x, y, p[1000], q[1000];

int main() {
	cin>>n>>m>>k;
	for (int i=0; i<n; i++)
		for (int j=0; j<m; j++) cin>>a[i][j];
	for (int i=0; i<1000; i++) p[i]=q[i]=i;
	for (int i=0; i<k; i++) {
		cin>>c>>x>>y;
		if (c=='c') swap(q[x-1],q[y-1]);
		if (c=='r') swap(p[x-1],p[y-1]);
		if (c=='g') cout<<a[p[x-1]][q[y-1]]<<endl;
	}
	return 0;
}
