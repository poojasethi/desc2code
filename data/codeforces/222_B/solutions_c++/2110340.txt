#include<stdio.h>
#include<string.h>

char s[20];
int a[1100][1100];
int rr[1100],cc[1100];

int main()
{
	int n,m,q,i,j,k;

	scanf("%d%d%d",&n,&m,&q);
	for(i=1;i<=n;i++)
	{
		rr[i]=i;
		for(j=1;j<=m;j++)
			scanf("%d",&a[i][j]);
	}
	for(i=1;i<=m;i++)
		cc[i]=i;
	while(q--)
	{
		scanf("%s%d%d",s,&i,&j);
		if(s[0]=='g')
			printf("%d\n",a[rr[i]][cc[j]]);
		else if(s[0]=='r')
			k=rr[i],rr[i]=rr[j],rr[j]=k;

		else
			k=cc[i],cc[i]=cc[j],cc[j]=k;
	}
}
