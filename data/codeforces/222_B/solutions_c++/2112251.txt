#include<iostream>
using namespace std;
int a[1000][1000];
int us[1000]={0},qa[1000]={0};
int main()
{
    int n,m,i,j,k,k1,k2;
    char cr;
    cin>>n>>m>>k;
    for(i=0;i<n;i++){
      qa[i]=i;
      for(j=0;j<m;j++){cin>>a[i][j];}
    }
    for(i=0;i<m;i++)us[i]=i;
    for(i=0;i<k;i++){
       cin>>cr>>k1>>k2;
       k1--;
       k2--;
       if(cr=='c')swap(us[k1],us[k2]); else
       if(cr=='r')swap(qa[k1],qa[k2]); else
       cout<<a[qa[k1]][us[k2]]<<endl;
    }
    return 0;
}
