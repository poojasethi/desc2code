#include<iostream>
using namespace std;
int n,m,k,a[1005][1005],i,j,x,y,b[1005],c[1005];
char s;
int main()
{
    cin>>n>>m>>k;
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            cin>>a[i][j];
    for(i=1;i<=n;i++) b[i]=i;
    for(i=1;i<=m;i++) c[i]=i;
    for(i=1;i<=k;i++) {
        cin>>s>>x>>y;
        if(s=='r') swap(b[x],b[y]);
        else if(s=='c') swap(c[x],c[y]);
        else cout<<a[b[x]][c[y]]<<endl;
    }
    return 0;
}
