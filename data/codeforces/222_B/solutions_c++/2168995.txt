#include<iostream>
using namespace std;
int l[1001],c[1001],x[1001][1001],n,m,k,i,j,a,b;
char cc;
int main()
{
	cin>>n>>m>>k;
	for(i=1;i<=n;i++)
	{
		l[i]=i;
		for(j=1;j<=m;j++){ cin>>x[i][j]; c[j]=j;}
	}
	int aux;
	for(i=1;i<=k;i++)
	{
		cin>>cc>>a>>b;
		if(cc=='g') cout<<x[l[a]][c[b]]<<endl;
		if(cc=='c') {aux=c[a]; c[a]=c[b]; c[b]=aux;}
		if(cc=='r') {aux=l[a]; l[a]=l[b]; l[b]=aux;}
	}
	return 0;
}
