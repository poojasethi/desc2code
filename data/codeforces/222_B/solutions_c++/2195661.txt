#include <iostream>
using namespace std;
main()
{
static long long int n,m,k,a[1001][1001],b[1001],i,temp,x,y,j,c[1001];
char p;
cin>>n>>m>>k;
for(i=1;i<=n;i++)
{
for(j=1;j<=m;j++)
{
cin>>a[i][j];
c[i]=i;b[j]=j;
}}
for(i=1;i<=k;i++)
{
cin>>p>>x>>y;
if(p=='g')
cout<<a[c[x]][b[y]]<<endl;
if(p=='r')
{
temp=c[y];
c[y]=c[x];
c[x]=temp;
}
if(p=='c')
{
temp=b[y];
b[y]=b[x];
b[x]=temp;
}
}
return(0);
}
