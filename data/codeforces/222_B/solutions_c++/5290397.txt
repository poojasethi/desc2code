#include <iostream>
#include <cstdio>
using namespace std;
const int MAXN=1010;
int a[MAXN][MAXN];
int main()
{
	int n,m,k;
	int c[MAXN],r[MAXN];
	cin>>n>>m>>k;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			scanf("%d",&a[i][j]);
			c[j]=j;
		}
		r[i]=i;
	}
	char cmd;
	int x,y;
	for(int i=1;i<=k;i++)
	{
		cin>>cmd;
		scanf("%d%d",&x,&y);
		if(cmd=='r')
			swap(r[x],r[y]);
		else if(cmd=='c')
			swap(c[x],c[y]);
		else
			cout<<a[r[x]][c[y]]<<endl;
	}
}
