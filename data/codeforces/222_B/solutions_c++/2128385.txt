#include <cstdio>
#include <algorithm>
using namespace std;
int n,m,q,r[1024],c[1024],a[1024][1024];
int main(void){
  scanf("%d %d %d",&n,&m,&q);
  for(int i=1;i<=n;++i)for(int j=1;j<=m;++j)scanf("%d",&a[i][j]);
  for(int i=max(n,m);i>0;--i)r[i]=c[i]=i;
  char s[32];
  for(int i=0,x,y;i<q&&3==scanf("%s %d %d",s,&x,&y);++i)
    if('r'==*s)swap(r[x],r[y]);else if('c'==*s)swap(c[x],c[y]);else printf("%d\n",a[r[x]][c[y]]);
  return 0;
}
