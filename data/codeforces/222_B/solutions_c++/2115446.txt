#include <stdio.h>
#define F(i,b,e) for(int i=b; i<=e; i++)
int n,m,k,a[1010][1010],p[1010];
main() {
  scanf("%d%d%d",&n,&m,&k);
  F(i,1,n)
    F(j,1,m)
      scanf("%d",&a[i][j]);
  F(i,1,m)
    p[i]=i;
  F(i,1,k) {
    char s;
    int x,y,t;
    scanf("\n%c%d%d",&s,&x,&y);
    if(s=='c')
      t=p[x], p[x]=p[y], p[y]=t;
    else if(s=='r')
      F(j,1,m)
        t=a[x][j], a[x][j]=a[y][j], a[y][j]=t;
    else
      printf("%d\n",a[x][p[y]]);
  }
  return 0;
}
