#include<bits/stdc++.h>
using namespace std;
int a[1010][1010],row[1010],col[1010];
int main()
{
	int n,m,k,i,j,x,y;
	char c;
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;i++)
	{
		row[i]=i;
		for(j=1;j<=m;j++)
			scanf("%d",&a[i][j]);
	}
	for(i=1;i<=m;i++)
		col[i]=i;
	while(k--)
	{
		scanf("%*c%c%d%d",&c,&x,&y);;
		if(c=='c')
			swap(col[x],col[y]);
		else if(c=='r')
			swap(row[x],row[y]);
		else if(c=='g')
			printf("%d\n",a[row[x]][col[y]]);
	}
	return 0;
}
