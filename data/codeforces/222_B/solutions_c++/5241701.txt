#include<stdio.h>
#include<iostream>
using namespace std;

int main(){
	int a[1001][1001],l[1001],r[1001],n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	for (int i=1;i<=n;++i)
		for (int j=1;j<=m;++j)
			scanf("%d",&a[i][j]);
	for (int i=1;i<=n;++i) l[i] = i;
	for (int i=1;i<=m;++i) r[i] = i;
	for (int i=1;i<=k;++i){
		char z;
		int x,y;
		cin>>z>>x>>y;
		if (z == 'g') printf("%d\n",a[l[x]][r[y]]);
		if (z == 'c') swap(r[x],r[y]);
		if (z == 'r') swap(l[x],l[y]);
	}
	
}
