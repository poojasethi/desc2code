#include <cstdio>
#define REP(i,N) for(int i=0;i<N;i++)
#define SWP(x,y) t=x,x=y,y=t;
int N,M,K,T[1001][1001],r[1001],c[1001],a,b,t;
char C;
int main(){
    scanf("%d%d%d",&N,&M,&K);
    REP(i,N)REP(j,M)scanf("%d",T[i]+j);
    REP(i,N)r[i]=i;
    REP(i,M)c[i]=i;
    REP(i,K){
        scanf(" %c%d%d",&C,&a,&b);
        if(C=='c') SWP(c[a-1],c[b-1]);
        if(C=='r') SWP(r[a-1],r[b-1]);
        if(C=='g') printf("%d\n",T[r[a-1]][c[b-1]]);
    }
}
