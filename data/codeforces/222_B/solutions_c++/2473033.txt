#include<iostream>
#include<string>
using namespace std;
int main()
{
string s;
int n,m,k,i,j,p,q,f=0;
int A[1001][1001],X[1001],Y[1001];
cin>>n>>m>>k;
for(i=1; i<=n; i++)
{
for (j=1; j<=m; j++)
{
cin>>A[i][j];
Y[j]=j;
}
X[i]=i;
}
for (i=1; i<=k; i++)
{
cin>>s>>p>>q;
if (s == "g")
{
cout<<A[X[p]][Y[q]]<<endl;
continue;
}
if (s == "c")
{
swap(Y[p],Y[q]); 
continue;
}
if (s == "r")
{
swap(X[p],X[q]); 
continue;
}
}
return 0;
}
