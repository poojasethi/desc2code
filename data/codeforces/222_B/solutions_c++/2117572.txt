#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	int A[n][m];
	int R[n],C[m];
	for(int i=0;i<n;i++)R[i]=i;
	for(int j=0;j<m;j++)C[j]=j;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			cin>>A[i][j];
	char c;int x,y;
	for(int i=0;i<k;i++){
		cin>>c>>x>>y;
		if(c=='c')swap(C[x-1],C[y-1]);
		else if(c=='r')swap(R[x-1],R[y-1]);
		else printf("%d\n",A[R[x-1]][C[y-1]]);
	}
}
