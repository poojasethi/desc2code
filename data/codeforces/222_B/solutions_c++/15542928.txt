#include<iostream>
#include<cstdio>
using namespace std;
int c,r,q,y,z;
char x;
int a[1005][1005],C[1005],R[1005];





int main()
{
cin>>r>>c>>q;
for(int i=1;i<=r;i++)
 for(int j=1;j<=c;j++)
 scanf("%d",&a[i][j]);
 for(int i=0;i<1001;i++)
    C[i]=i,R[i]=i;

for(int k=0;k<q;k++)
{
    getchar();
scanf("%c%d%d",&x,&y,&z);
if(x=='c')
swap(C[y],C[z]);
else if(x=='r')
swap(R[y],R[z]);
else
printf("%d\n",a[R[y]][C[z]]);
}
 return 0;
}
