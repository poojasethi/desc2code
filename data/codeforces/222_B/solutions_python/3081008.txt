n, m, k = map(int, raw_input().split())
t = [map(int, raw_input().split()) for _ in xrange(n)]
row = range(n)
col = range(m)
for _ in xrange(k):
  line = raw_input().split()
  x = int(line[1])-1
  y = int(line[2])-1
  if line[0] == 'r':
    tmp = row[x]
    row[x] = row[y]
    row[y] = tmp
  elif line[0] == 'c':
    tmp = col[x]
    col[x] = col[y]
    col[y] = tmp
  else:
    print t[row[x]][col[y]]
