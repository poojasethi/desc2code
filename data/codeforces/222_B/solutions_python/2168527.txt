n,m,q=map(int,raw_input().split())
r,c=range(n+1),range(m+1)
table=[raw_input().split() for _ in xrange(n)]
while q:
    s=raw_input().split()
    s1,s2=int(s[1]),int(s[2])
    if s[0]=='r':
        r[s1],r[s2]=r[s2],r[s1]
    elif s[0]=='c':
        c[s1],c[s2]=c[s2],c[s1]
    else:
        print table[r[s1]-1][c[s2]-1]
    q-=1
