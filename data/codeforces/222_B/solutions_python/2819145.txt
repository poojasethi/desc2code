n, m, k = map(int, raw_input().split())
b = [raw_input().split() for _ in range(n)]
r = range(n)
c = range(m)

for _ in range(k):
    o, s, t = raw_input().split()
    x = int(s) - 1
    y = int(t) - 1
    if o == 'c':
        c[x], c[y] = c[y], c[x]
    elif o == 'r':
        r[x], r[y] = r[y], r[x]
    else:
        print b[r[x]][c[y]]
