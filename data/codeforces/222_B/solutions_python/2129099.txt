# 222B - Cosmic Tables
n, m, k = map(int, raw_input().split())
t = []
for i in range(n):
	 x = raw_input().split()
	 t.append(x)
row = range(n+1)
col = range(m+1)
for i in range(k):
	f, x, y = raw_input().split()
	x = int(x)
	y = int(y)
	if f == "c":
		col[x], col[y] = col[y], col[x]
	elif f == "r":
		row[x], row[y] = row[y], row[x]
	else:
		print t[row[x] - 1][col[y] - 1]
