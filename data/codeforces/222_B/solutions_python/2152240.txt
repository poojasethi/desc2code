i=lambda:raw_input().split()
n,m,k=map(int,i())
t=[i() for _ in xrange(n)]
c=list(xrange(m))
r=list(xrange(n))
while k >0:
    k-=1
    s,x,y=i()
    x=int(x)-1
    y=int(y)-1
    if s == 'g':
        print t[r[x]][c[y]]
    elif s == 'c':
        c[x],c[y]=c[y],c[x]
    else:
        r[x],r[y]=r[y],r[x]
