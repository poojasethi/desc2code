n, m, k = map(int, raw_input().split())
t = [raw_input().split() for i in range(n)]
c = [0] + list(range(m))
r = [0] + list(range(n))

for i in range(k):
    s, x, y = raw_input().split()
    x, y = int(x), int(y)
    if s == 'c': c[x], c[y] = c[y], c[x]
    elif s == 'r': r[x], r[y] = r[y], r[x]
    else: print(t[r[x]][c[y]])
