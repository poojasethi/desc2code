import re
sp = re.compile(' ').split
n, m, k = map(int, sp(raw_input()))
table = [sp(raw_input()) for i in xrange(n)]
r = range(n)
c = range(m)
ans = []
for i in xrange(k):
    s,a,b = sp(raw_input())
    a = int(a)-1
    b = int(b)-1
    if s == 'c': c[a], c[b] = c[b], c[a];
    elif s == 'r': r[a], r[b] = r[b], r[a];
    else: ans.append(table[r[a]][c[b]])
print '\n'.join(ans)
