n, k, q = map(int, raw_input().split())
mp = []
[mp.append(raw_input().split()) for i in range(n)]
rs, cs = [i for i in range(n)], [i for i in range(k)]

for k in range(q):
	s, x, y = raw_input().split()	
	x, y = int(x) - 1, int(y) - 1
	if(s == 'c'): cs[x], cs[y] = cs[y], cs[x]
	if(s == 'r'): rs[x], rs[y] = rs[y], rs[x]
	if(s == 'g'): print(mp[rs[x]][cs[y]])
