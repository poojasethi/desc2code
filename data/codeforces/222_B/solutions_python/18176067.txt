def main():
	n, m, k = map(int, raw_input().split())
	board = [raw_input().split() for i in xrange(n)]
	row = range(n)
	col = range(m)
	for i in xrange(k):
		s = raw_input().split()
		x, y = int(s[1]) - 1, int(s[2]) - 1
		if "r" == s[0]: row[x], row[y] = row[y], row[x]
		elif "c" == s[0]: col[x], col[y] = col[y], col[x]
		else: print board[row[x]][col[y]]

if __name__ == "__main__": main()
