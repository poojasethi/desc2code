#!/usr/bin/python

str1 = raw_input()

n1, m1, k1 = str1.split()
n = int(n1)
m = int(m1)
k = int(k1)

matrix = [raw_input().split() for i in xrange(n)]

row = [i for i in xrange(n)]
col = [i for i in xrange(m)]

# deal with queries
for i in xrange(k):
	qw_str = raw_input()
	q = qw_str.split()
	x = int(q[1]) - 1
	y = int(q[2]) - 1
	if q[0] == "c":
		col[x], col[y] = col[y], col[x]
	elif q[0] == "r":
		row[x], row[y] = row[y], row[x]
	else:
		print matrix[row[x]][col[y]]
