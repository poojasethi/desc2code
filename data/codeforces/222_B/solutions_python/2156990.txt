n,m,q=map(int,raw_input().split())
tables = []
[tables.append(raw_input().split()) for i in xrange(n)]
row, col=[i for i in xrange(n)], [j for j in xrange(m)]
for k in xrange(q):
    s,x,y=raw_input().split()
    x,y=int(x)-1,int(y)-1
    if s=='r':
        row[x],row[y]=row[y],row[x]
    elif s=='c':
        col[x],col[y]=col[y],col[x]
    else :
        print tables[row[x]][col[y]]
