n, m, k = map(int, raw_input().split())
a, r, c = [raw_input().split() for i in xrange(n)], range(n), range(m)
for i in xrange(k):
    t, x, y = raw_input().split()
    x, y = int(x) - 1, int(y) - 1
    if t == 'r': r[x], r[y] = r[y], r[x]
    elif t == 'c': c[x], c[y] = c[y], c[x]
    else: print a[r[x]][c[y]]
