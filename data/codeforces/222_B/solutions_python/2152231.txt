i=lambda:raw_input().split()
l=lambda:map(int,i())
n,m,k=l()
t=[]
[t.append(i()) for _ in xrange(n)]
c=[j for j in xrange(m)]
r=[j for j in xrange(n)]
for j in xrange(k):
    s,x,y=raw_input().split()
    x=int(x)-1
    y=int(y)-1
    if s == 'g':
        print t[r[x]][c[y]]
    elif s == 'c':
        c[x],c[y]=c[y],c[x]
    else:
        r[x],r[y]=r[y],r[x]
