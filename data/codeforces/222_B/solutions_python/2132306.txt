table = []
n, m, k = map(int, raw_input().split())
for i in xrange(n):
    table.append(raw_input().split())
rs, cs = [i for i in xrange(n)], [i for i in xrange(m)]
for i in xrange(k):
    s, x, y = raw_input().split()
    x, y = int(x) - 1, int(y) - 1
    if s == "c":
        cs[x], cs[y] = cs[y], cs[x]
    elif s == "r":
        rs[x], rs[y] = rs[y], rs[x]
    else:
        print table[rs[x]][cs[y]]
