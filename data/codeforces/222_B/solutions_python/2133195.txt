n, m, k = map(int, raw_input().split())
mp = []

pr = range(n)
pc = range(m)
for i in xrange(n):
    mp.append(raw_input().split())


for i in xrange(k):
    s, x, y = raw_input().split()
    x, y = int(x) - 1, int(y) - 1
    if (s == 'c'):
        pc[x], pc[y] = pc[y], pc[x]
    elif (s == 'r'):
        pr[x], pr[y] = pr[y], pr[x]
    else:
        print mp[pr[x]][pc[y]]
