#include<bits/stdc++.h>
using namespace std;
int last[100001],num[100001],p[100001];
bool f[100001];

int main()
{
    int n,a,ans=0;
	cin>>n;
	while(n--)
	{
		cin>>a;
		num[a]++;
		if(last[a])
		{
			if(num[a]==2)
                p[a]=last[a]-n;
			if(n+p[a]!=last[a])
                f[a]=true;
		}
		last[a]=n;
	}
	for(int i=1;i<100001;i++)
        if(!f[i]&&num[i])
            ans++;
	cout<<ans<<endl;
	for(int i=1;i<100001;i++)
        if(!f[i]&&num[i])
            cout<<i<<" "<<p[i]<<endl;
	return 0;
}
