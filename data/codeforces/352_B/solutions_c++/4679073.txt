#include <iostream>
using namespace std;
long long n,p[100010];
long long pr[100010];
int main()
{
	cin >> n;
	int k=0;

	for (int a,i=1; i<=n; ++i)
	{
		cin >> a;
		if (p[a]==0 && pr[a]==0) {++k; pr[a]=i; continue;}
		if (p[a]==0 && pr[a]> 0) {p[a]=i-pr[a]; pr[a]=i; continue;}
		if (p[a]!=-1)
			if (p[a]==i-pr[a]) {pr[a]=i;} else {--k; p[a]=-1;}
	}

	cout << k << endl;
	for (int i=1; i<100010; ++i)
		if (p[i]>-1 && pr[i]>0) cout << i << ' ' << p[i] << endl;

	return 0;
}
