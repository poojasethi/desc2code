#include<iostream>
using namespace std;
int last[100001],num[100001],cha[100001];
bool w[100001];
int main()
{
	int n,a,ans=0;
	cin>>n;
	while(n--)
	{
		cin>>a;
		num[a]++;
		if(last[a])
		{
			if(num[a]==2)cha[a]=last[a]-n;
			if(n+cha[a]!=last[a])w[a]=true;
		}
		last[a]=n;
	}
	for(int i=1;i<100001;i++)if(!w[i]&&num[i])ans++;
	cout<<ans<<endl;
	for(int i=1;i<100001;i++)if(!w[i]&&num[i])cout<<i<<" "<<cha[i]<<endl;
	return 0;
}
