#include <stdio.h>
#include <string.h> 
int N , d[100001] , S[100001];
int main(){
	int tmp;
	int cnt = 0;
	scanf("%d",&N);
	for ( int i = 1 ; i <= N ; i++ ){
		scanf("%d",&tmp);
		if ( !S[tmp] ) S[tmp] = i , cnt++;
		else{
			if ( d[tmp] == 0 ){
				d[tmp] = i - S[tmp];
				S[tmp] = i;
				
			}
			else{
				if ( d[tmp] == -1 ) continue;
				if ( i - S[tmp] != d[tmp] )
					d[tmp] = -1 , cnt--;
				
				else
					S[tmp] = i;
				
			}
		}
	}


	printf("%d\n",cnt);
	for ( int i = 1;  i <= 100000 ; i++ )
		if ( S[i] && d[i] != -1 )
			printf("%d %d\n",i,d[i]);

	return 0;
}
