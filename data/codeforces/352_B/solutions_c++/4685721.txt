#include<iostream>
#include <string.h>
#include<vector>
#include<list>
#include<cstdlib>
#include<algorithm>
#define MAX 100001
using namespace std;
int main(){
	int a,n,i,t=0,p[MAX],last[MAX];
	bool bo[MAX];
	for(i=0;i<MAX;i++){
		p[i]=0;bo[i]=0;
	}
	cin>>n;
	for(i=0;i<n;i++){
	cin>>a;
		if(!bo[a]){
			bo[a]=1;
			last[a]=i;
			t++;
		}
		else{
			if(p[a]<0) continue;
			if(!p[a]) p[a]=i-last[a];
			else if ((i-last[a])!=p[a]) p[a]=-1,t--;
			last[a]=i;
		}
	}
	cout<<t<<endl;
	for(i=0;i<MAX;i++)
	if ((bo[i])&&(p[i]>=0)) cout<<i<<" "<<p[i]<<endl;
	return 0;
}
