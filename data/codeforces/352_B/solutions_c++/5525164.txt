#include<iostream>
using namespace std;
int main() {
	int diff[100001], last[100001], n, p, nv=0;
	cin >> n;
	for(int i=0;i<=100000;i++)
		last[i]=-1;
	for(int i=0;i<n;i++) {
		cin >> p;
		if(last[p] == -1) {
			diff[p] = 0;
			nv++;
		}
		else if(diff[p] == 0) 
			diff[p] = i - last[p];
		else if(diff[p] != -1 && i - last[p] != diff[p]) {
			diff[p] = -1;
			nv--;
		}
		last[p]=i;
	}
	cout << nv << '\n';
	for(int i=0;i<=100000;i++)
		if(last[i] != -1 && diff[i] != -1)
			cout << i << ' ' << diff[i] << '\n';
	return 0;
}
