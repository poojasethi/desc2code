#include<bits/stdc++.h>

using namespace std;


int arr[100001]={0},t[100001]={0};
int main()
{
    int n,x,cnt=0;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    cin>>x;
        if(arr[x])
            {
                if(!t[x])
                t[x]=i-arr[x];

                else
                if(i-arr[x]!=t[x])
                arr[x]=-1;

            }
        if(arr[x]!=-1)
        arr[x]=i;
    }
    for(int i=1;i<=100000;i++)
    if(arr[i]>0)cnt++;

    cout<<cnt<<endl;

    for(int i=1;i<=100000;i++)
    if(arr[i]>0)
    cout<<i<<" "<<t[i]<<endl;

return 0;
}
