#include<cstdlib>
#include<cstdio>
using namespace std;

int n,m,a;
int p[100001],d[100001];

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a);
		if (p[a])
		{
			if (p[a]!=-1)
			{
				if (d[a])
				{
					if (i-p[a]==d[a]) p[a]=i;
					else p[a]=-1;
				}
				else
				{
					d[a]=i-p[a];
					p[a]=i;
				}
			}
		}
		else p[a]=i;
	}
	for (int i=1;i<=100000;i++) if (p[i] && p[i]!=-1) ++m;
	printf("%d\n",m);
	for (int i=1;i<=100000;i++) if (p[i] && p[i]!=-1) printf("%d %d\n",i,d[i]);
	return 0;
}
