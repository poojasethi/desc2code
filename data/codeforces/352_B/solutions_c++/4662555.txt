#include <cstdio>
int ls[100005],cd[100005];
bool flag[100005];
int main()
{
	int i,n,t,sum = 0;
	scanf("%d",&n);
	for (i = 1;i <= n;++i)
	{
		scanf("%d",&t);
		if (ls[t] == 0)
		{
			flag[t] = true;
			++sum;
		}
		else
		{
			if (flag[t])
			{
				if (cd[t] == 0)
					cd[t] = i - ls[t];
				else
					if (cd[t] != i - ls[t])
					{
						flag[t] = false;
						--sum;
					}
			}
		}
		ls[t] = i;
	}
	printf("%d\n",sum);
	for (i = 1;i <= 100000;++i)
		if (flag[i] == true)
			printf("%d %d\n",i,cd[i]);
	return 0;
}
