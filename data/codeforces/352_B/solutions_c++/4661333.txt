int n;
int a[100001];
int b[100001];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=100000;i++) {a[i]=-1;b[i]=-1;}
    for(int i=0;i<n;i++)
    {
     int x;
     scanf("%d",&x);
     if(a[x]<0)
     {
      a[x]=i;
      b[x]=0;
     }else if(!b[x])
     {
      b[x]=i-a[x];
      a[x]=i;
     }else if(b[x]==i-a[x]) a[x]=i;
     else b[x]=-1;
    }
    int t=0;
    for(int i=1;i<=100000;i++)
    if(b[i]>=0) t++;
    printf("%d\n",t);
    for(int i=1;i<=100000;i++)
