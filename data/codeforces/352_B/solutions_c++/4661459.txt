#include <cstdio>
#include <algorithm>

using namespace std;

int N;
int mem[100005], cnt[100005], lst[100005];

int main(){
  scanf("%d",&N);
  for(int c = 1 ; c <= N ; c++){
    int a;
    scanf("%d",&a);
    if(cnt[a] == 0)
      cnt[a]++;
    else if(cnt[a] == 1)
      cnt[a]++, mem[a] = c-lst[a];
    else if(cnt[a] > 1){
      if(c-lst[a] == mem[a])
	cnt[a]++;
      else
	cnt[a] = -1;
    }
    lst[a] = c;
  }
  int sol = 0;
  for(int c = 1 ; c <= 100000 ; c++)
    if(cnt[c] > 0)
      sol++;
  printf("%d\n",sol);
  for(int c = 1 ; c <= 100000 ; c++)
    if(cnt[c] > 0)
      printf("%d %d\n",c,mem[c]);
}
