#include <stdio.h>

int M[100001], R[100001], A[100001], N, P;

int main()
{
	int i;
	scanf("%d", &N);
	for(i=1; i<=N; ++i)
	{
		scanf("%d", &A[i]);
		if (!M[A[i]])
		 M[A[i]] = i, ++P;
		else if (M[A[i]] != -1)
		{
			if (R[A[i]] != -1)
			{
				if (!R[A[i]])
				 R[A[i]] = i - M[A[i]];
				else if (R[A[i]] != i - M[A[i]])
				{
					R[A[i]] = -1, --P;
				}
			}
			M[A[i]] = i;
		}
	}
	printf("%d\n", P);
	for(i=1; i<=100000; ++i) {
		if (M[i] && R[i] != -1) {
			if (R[i]) printf("%d %d\n", i, R[i]);
			else printf("%d 0\n", i);
		}
	}
	return 0;
}
