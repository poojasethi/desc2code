#include <iostream>

using namespace std;

int main() {
	int n;
	cin >> n;
	int top[1001000];
	int diff[1001000];
	int x;
	int tot=0;
	for(int i=1; i<n+1; ++i) {
		cin >> x;
		if(top[x]==0 && diff[x]==0) {
			top[x]=i;
			tot++;
		} else if(top[x]!=0 && diff[x]==0) {
			diff[x]=i-top[x];
			top[x]=i;
		} else if(top[x]!=0 && diff[x]!=0) {
			if(diff[x]==(i-top[x])) {
				top[x]=i;
			} else {
				top[x]=0;
				tot--;
			}
		}
	}
	cout << tot << endl;
	for(int i=0; i<1001000; ++i) {
		if(top[i]!=0) {
			cout << i << " " << diff[i] << endl;
		}
	}
	return 0;
}
