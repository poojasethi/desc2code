#include<bits/stdc++.h>
using namespace std;
int a[100005],b[100005],last[100005],ans[100005],book[100005];
int main()
{
	int n,num=0;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if(!last[a[i]])
			last[a[i]]=i;
		else
		{
			if(!ans[a[i]])
				ans[a[i]]=i-last[a[i]];
			else
			{
				if(i-last[a[i]]!=ans[a[i]])
				book[a[i]]++;
			}
			last[a[i]]=i;
		}
	}
	for(int i=1;i<=n;i++)
	{
		if(!book[a[i]])
		{
			b[num++]=a[i];
			book[a[i]]++;
		}
	}
	sort(b,b+num);
	printf("%d\n",num);
	for(int i=0;i<num;i++)
		printf("%d %d\n",b[i],ans[b[i]]);
	return 0;
}
