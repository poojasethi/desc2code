#include <bits/stdc++.h>
using namespace std;

const int MAX = 1e5 + 5;
int a[MAX], ch[MAX], f[MAX];

int main () {
	
	int n, t;
	cin >> n;

    //int tot = 0;
	for (int i = 1; i <= n; ++i) {

		cin >> t; 
		f[t] = 1;
		if (ch[t] != 0) {

			if (ch[t] != i - a[t])
				ch[t] = -1;
		}else if (a[t] != 0){

			ch[t] = i - a[t];
		}

		a[t] = i;
	}	

    vector<int> v;
	for (int i = 1; i < MAX; ++i) {

		if (ch[i] != -1 && f[i])
		v.push_back(i);
    }

    cout << v.size() << "\n";
    for (int i = 0; i < v.size(); ++i)
    	cout << v[i] << " " << ch[v[i]] << "\n";
return 0;	
}
