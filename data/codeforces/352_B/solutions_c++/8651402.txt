#include <iostream>
#include<algorithm>
using namespace std;

int main() {
int c=0,n,i,j,k,z=0;
	cin>>n;
	int a[1000001],b[1000001];
for(i=1;i<=n;i++)
{cin>>k;
z=max(k,z);
if(a[k]==0 &&b[k]==0)
{a[k]=i;c++;}
else if(a[k]!=0 &&b[k]==0)
{b[k]=i-a[k];a[k]=i;}
else if(a[k]!=0 && b[k]!=(-1))
{if(b[k]!=i-a[k])
{b[k]=-1;c--;}
else
a[k]=i;
}}
cout<<c<<endl;
for(i=0;i<=z;i++)
{
if(b[i]!=-1 &&a[i]!=0)
cout<<i<<" "<<b[i]<<endl;
else if(a[i]!=0 &&b[i]==0)
cout<<i<<" "<<"0"<<endl;
}
	return 0;
}
