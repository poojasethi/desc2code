#include<iostream>
#include<algorithm>

using namespace std;
int main()
{
    int n,k=0;
    cin>>n;
    int a,d[100001]={0},prev[100001]={0},j=0,x[100001];
    for(int i=1;i<=n;i++)
    {
        cin>>a;
        if(prev[a]==0&&d[a]==0){k++;prev[a]=i;x[j]=a;j++;continue;}
        if(d[a]==0&&prev[a]>0){d[a]=i-prev[a];prev[a]=i;continue;}
        if(d[a]!=-1)
        {
            if(d[a]==i-prev[a]){prev[a]=i;} else {k--;d[a]=-1;}
        }
    }
    sort(x,x+j);
    cout<<k<<endl;
    for(int i=0;i<j;i++)
    {
        if(d[x[i]]>-1&&prev[x[i]]>0) cout<<x[i]<<' '<<d[x[i]]<<endl;
    }

}
