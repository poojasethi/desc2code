#include<iostream>
using namespace std;
int main()
{
int n,i,a,c[100001]={0},d[100001],b[100001],m=0;
cin>>n;
for(i=0; i<n; i++)
{
cin>>a;
if(c[a]!=-1)
{
c[a]++;
if(c[a]==1)
{
b[a]=i+1;
m++;
}
else if(c[a]==2)
{
d[a]=i+1-b[a];
b[a]=i+1;
}
else
if(d[a]==i+1-b[a])
b[a]=i+1;
else
{
c[a]=-1;
m--;
}
}
}
cout<<m<<'\n';
for(i=1; i<=100000; i++)
if(c[i]>0)
if(c[i]==1)
cout<<i<<" 0\n";
else
cout<<i<<' '<<d[i]<<'\n';
return 0;
}
