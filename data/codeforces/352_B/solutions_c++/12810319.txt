#include<stdio.h>
int aa[111111];
int bb[111111];
int n;
int main(){
	int i,j,k,x,y;
	while(~scanf("%d",&n)){
		y=0;
		for(i=1;i<=n;i++){
			
			scanf("%d",&x);
			
			if(aa[x]==-1)continue;
			
			
			if(aa[x]==0){
				aa[x]=i;
				y++;
				continue;
			}
			if(bb[x]==0){
				
				bb[x]=i-aa[x];
				
				
			}
			
			if(i-aa[x]!=bb[x]){
				aa[x]=-1;
				y--;
				continue;
			}
			
			
			
			
			
			aa[x]=i;
			
			
			
			
			 
			
		}
		
		
		printf("%d\n",y);
		for(i=1;i<111111;i++){
			if(aa[i]>0){
				
				printf("%d",i);
				printf(" %d\n",bb[i]);
			}	
		}
		
		
	}
	
	return 0;
}
