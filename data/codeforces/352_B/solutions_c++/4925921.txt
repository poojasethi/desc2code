#include <iostream>
#include <algorithm>
using namespace std;
int n;
int cal=0;
int l;
int m[100001];
int u[100001];
int main()
{
for (int i=0;i<100001;i++)
m[i]=-2;
int cal=0;
cin >> n;
for (int i=0;i<n;i++)
{
cin >> l;
if (m[l]>-1) { if (m[l]+u[l]==i) {m[l]=i;} else {m[l]=-3;cal--;}}
if (m[l]==-1) {m[l]=i;u[l]=i-u[l];}
if (m[l]==-2) {m[l]=-1;u[l]=i;cal++;}
}
cout << cal << "\n";
for (int i=0;i<100001;i++)
{
if (m[i]==-1)  {cout << i << " " << 0 << "\n";}
if (m[i]>=0)  {cout << i << " " << u[i] << "\n";}
}
}
