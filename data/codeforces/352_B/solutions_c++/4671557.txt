#include<stdio.h>

int main()
{

int i,n,c,a[100005],b[100005][3]={0};

scanf("%d",&n);

for(i=0;i<n;i++)
{
scanf("%d",&a[i]);
if(b[a[i]][0]==-5)
continue;
else if((b[a[i]][1]==0)&&(b[a[i]][1]==0))
{
b[a[i]][1]=i+1;
b[a[i]][0]=0;
c++;
}
else if(i+1-(b[a[i]][1])==b[a[i]][0])
{
b[a[i]][1]=i+1;
}
else if(b[a[i]][0]==0)
{
b[a[i]][0]=i+1-b[a[i]][1];
b[a[i]][1]=i+1;
}
else
{
b[a[i]][0]=-5;
c--;
}

}


printf("%d\n",c);
for(i=1;i<=100000;i++)
{
if((b[i][0]==-5)||((b[i][1]==0)&&(b[i][0]==0)))
{
continue;
}
else
{
printf("%d %d\n",i,b[i][0]);
}
}

return 0;
}
