#include <cstdio>
#include <cstdlib>
using namespace std;

int n,ans,x;
int last[200001],gc[200001];

int main()
{
    scanf("%d",&n);
    for (int i=1;i<=n;i++)
    {
        scanf("%d",&x);
        if (!last[x]) last[x]=i;
        else if (!gc[x]) gc[x]=i-last[x];
        else {
             if (i-last[x]!=gc[x]) gc[x]=-1;
        }
        last[x]=i;
    }
    for (int i=1;i<=100000;i++)
        if (last[i]&&gc[i]!=-1) ans++;
    printf("%d\n",ans);
    for (int i=1;i<=100000;i++)
        if (last[i]&&gc[i]!=-1) printf("%d %d\n",i,gc[i]);
}
