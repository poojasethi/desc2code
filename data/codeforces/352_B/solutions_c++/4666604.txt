#include<stdio.h>

int n,nr,x,p[100001],v[100001],w[100001],sol[100001];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	{
		scanf("%d",&x);
		if(!v[x])
		{
			v[x]=1;
			p[x]=i;
			sol[x]=i;
			++nr;
		}else if(v[x]==1)
		{
			++v[x];
			w[x]=i-p[x];
			p[x]=i;
		}else if(sol[x])
		{
			if(p[x]+w[x]!=i)
			{
				--nr;
				sol[x]=0;
			}
			p[x]=i;
		}
		
	}
	
	printf("%d\n",nr);
	for(int i=1;i<=100000;++i)
		if(sol[i])
			printf("%d %d\n",i,w[i]);
	
	
	
	return 0;
}
