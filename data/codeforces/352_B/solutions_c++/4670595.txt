#include<stdio.h>
#include<string.h>
int a[100001];
int b[100001];
int c[100001];
int d[100001];
int main()
{
memset(a,0,sizeof(a));
memset(c,0,sizeof(c));
int n,i,x,p,tot=0,q;
scanf("%d",&n);
for(i=1;i<=n;i++) {
scanf("%d",&x);
if(a[x]==0) {
	tot++;
}
if(a[x]!=-1) {
a[x]=x;
if(c[x]>=1) {
p=b[x];
}
b[x]=i;
c[x]++;
if(c[x]==1) {
	d[x]=0;
}
else {
	q=d[x];
	d[x]=b[x]-p;
	if(c[x]>2)
	{
		if(d[x]!=q) {
			tot--;
			a[x]=-1;
		}
	}
}
}

	
}
printf("%d\n",tot);
for(i=1;i<=100001;i++) {
	if(a[i]!=-1&&a[i]!=0) {
		printf("%d %d\n",a[i],d[i]);
	}
}
return 0;	
}
