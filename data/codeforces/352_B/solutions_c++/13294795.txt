#include <stdio.h>
const int maxn = 100001;
int a[maxn];
int ok[maxn];
int d[maxn];
int main(){
    int n;
    scanf("%d",&n);
    int t;
    for(int i = 1; i <= n; i ++){
        scanf("%d",&t);
        if(a[t]){
            if(d[t] != 0 && i - a[t] != d[t]) ok[t] = -1;
            d[t] = i - a[t];
        }
        a[t] = i;
    }
    t = 0;
    for(int i = 1; i < maxn; i ++) if(a[i] && ok[i] != -1) t ++;
    printf("%d\n", t);
    for(int i = 1; i < maxn; i ++) if(a[i] && ok[i] != -1) printf("%d %d\n", i, d[i]);
    return 0;
}
