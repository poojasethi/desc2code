#include <cstdio>
#define N 100001

int n,i,j,k,a[N],b[N],c[N],t;
bool f[N];

int main(){
	scanf("%d",&n);
	for(i=0;i<N;i++) b[i]=c[i]=-1;
	for(i=0;i<n;i++){
		scanf("%d",&a[i]);
		if (f[a[i]]) continue;
		if (c[a[i]]<0){
			c[a[i]]=0;
			b[a[i]]=i;
		}else{
			if (c[a[i]]){
				if (i-b[a[i]]!=c[a[i]]){
					f[a[i]]=1;
				}else{
					b[a[i]]=i;
				}
			}else{
				c[a[i]]=i-b[a[i]];
				b[a[i]]=i;
			}
		}
	}
	for(i=0;i<N;i++){
		if (c[i]>=0&&!f[i]) t++;
	}
	printf("%d\n",t);
	for(i=0;i<N;i++){
		if (c[i]>=0&&!f[i]){
			printf("%d %d\n",i,c[i]);
		}
	}
	return 0;
}
