#!/usr/bin/python
import sets
from collections import defaultdict
l=[]
n = int(raw_input(''))
t = raw_input('').split(' ')
if len(t) == n:
	for item in t:
		l.append(int(item))
D = defaultdict(list)
for i,item in enumerate(l):
	D[item].append(i+1)
D = {k:v for k,v in D.items() if len(v)>0}

zero = 0
l1=[]
l2=[]
for key,value in sorted(D.iteritems()):
	if len(value) > 1:
		m = [x - value[i - 1] for i, x in enumerate(value)][1:]
		if len(sets.Set(m)) == 1:
			l1.append(int(key))
			l2.append(int(m[0]))
	else:
		l1.append(int(key))
		l2.append(int(zero))

print len(l1)
for i in range(0,len(l1)):
	print l1[i],l2[i]
