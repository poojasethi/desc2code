import sys

raw_input()
numbers = map(int, raw_input().split())
m = {}

for i,n in enumerate(numbers):
  if m.has_key(n):
    a,b,c = m[n]
    if a == 0:
      continue

    if b:
      if i - c == b:
        m[n] = (1, b, i)
      else:
        m[n] = (0,0,0)
    else:
      m[n] = (1, i - c, i)
  else:
    m[n] = (1, 0, i)


ks = [k for k in m.keys() if m[k][0]]
ks.sort()
print len(ks)
for k in ks:
  print k, m[k][1]
