#!/usr/bin/python

n=int(raw_input())
inp=raw_input()
l=inp.split()
l=map(lambda x: int(x),l)
inp=[]
for i in xrange(len(l)):
        m=[]
        m.append(i)
        m.append(l[i])
        inp.append(m)
inp=sorted(inp,key=lambda x:x[1])
ans=[]
l=[]
l.append(inp[0][1])
l.append(0)
ans.append(l)
flag=0;
for i in range(1,len(inp)):
	if(inp[i][1]==inp[i-1][1] and flag!=1):
		diff=inp[i][0]-inp[i-1][0]
		if(ans[-1][1]==0):
			ans[-1][1]=diff
		if(diff!=ans[-1][1]):
			ans.pop()
			flag=1;
	elif(inp[i][1]!=inp[i-1][1]):
		l=[]
		l.append(inp[i][1])
		l.append(0)
		ans.append(l)
		flag=0
	
if(len(inp)==1):
	print "1"
	print str(inp[0][1])+" "+"0"
else:
	print len(ans)
	for i in xrange(len(ans)):
		print str(ans[i][0])+" "+str(ans[i][1])
