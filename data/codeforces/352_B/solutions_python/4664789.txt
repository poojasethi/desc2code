from collections import defaultdict
import itertools

raw_input()

a = map(int, raw_input().split())
d = defaultdict(list)

for i, x in enumerate(a):
    d[x].append(i)

ret = []

def check(l):
    ''' check wether the list is an arithmetic
        progression '''
    if len(l) == 1:
        return 0

    p = l[1] - l[0]
    if any(y - x - p for x, y in itertools.izip(l, l[1:])):
        return 'kakapo' 
    return p

for x, l in d.iteritems():
    p = check(l)
    if p != 'kakapo':
        ret.append((x, p))
print len(ret)
for t in sorted(ret):
    print t[0], t[1]
