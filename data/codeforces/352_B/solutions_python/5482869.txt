n = int(raw_input())
s = raw_input().split()
d = {}
r = {}

for i in range(n):
    j = int(s[i])
    if j in d:
        d[j].append(i)
    else:
        d[j] = [i]

for k in d.keys():
    v = d[k]
    diff = []
    for i in range(len(v)):
        if i!=0:
            diff.append(v[i] - v[i-1])
    if len(set(diff)) == 1:
        r[k] = list(set(diff))[0]
    elif len(set(diff)) == 0:
        r[k] = 0
        
print len(r)
for key in sorted(r):
    print key, r[key]
