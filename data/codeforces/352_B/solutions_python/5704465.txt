import sys

v={}

n=int(raw_input())

rij=map(int,raw_input().split())


for i in range(n):
  x=rij[i]
  if x in v:
    v[x]=(i,-1 if v[x][1] and i-v[x][0]!=v[x][1] else i-v[x][0])
  else:
    v[x]=(i,0)
