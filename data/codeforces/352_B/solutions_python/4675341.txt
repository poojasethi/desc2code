n = int(raw_input())
a = list(map(int, raw_input().split()))

d = {}
for i,x in enumerate(a):
	if x in d:
		p, last, diff = d[x]
		if not p:
			continue
		if diff:
			if i - diff == last:
				d[x] = True, i, diff
			else:
				d[x] = 0,0,0
		else:
			d[x] = True, i, i - last
	else:
		d[x] = True,i,0

elements = [k for k in d.keys() if d[k][0]]
elements.sort()
print len(elements)
for k in elements:
	print k, d[k][2]
