n=input()
r=sorted(enumerate(map(int, raw_input().split())),key=lambda x:x[1])+[(0,0)]

from collections import defaultdict as d
c=d(set)
for p in [ (a[1], b[0]-a[0]) if a[1] == b[1] else (a[1], 0) for a, b in zip(r, r[1:])]:
    c[p[0]].add(p[1])
    
t=[]
for x,s in  c.items():
    if len(s)==2:
        t.append((x, max(list(s))))
    if len(s)==1:
        t.append((x,0))
print len(t)
print '\n'.join(map(lambda x:' '.join(map(str, x)),sorted(t)))
