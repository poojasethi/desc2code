n = int(raw_input())
l = map(int,raw_input().split())
d = {}
for i in range(len(l)):
    k = l[i]
    if k in d:
        if d[k][1] and(i-d[k][0]!=d[k][1]):
            d[k] = (i,-1)
        else:
            d[k] = (i,i-d[k][0])
    else:
        d[k] = (i,0)

b = []

for k in sorted(d.keys()):
    if d[k][1]>=0:
        b.append((k,d[k][1]))

print len(b)

for x in b:
    print x[0],x[1]
