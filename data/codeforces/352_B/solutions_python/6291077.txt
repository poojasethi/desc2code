def series(l):
  if len(l) == 1:
    return 0
  d = l[1] - l[0]
  for i in range(1, len(l) - 1):
    if l[i+1] - l[i] != d:
      return None
  return d

raw_input()
ns = map(int, raw_input().split())
ixs = {}
for i, n in enumerate(ns):
  if n not in ixs:
    ixs[n] = []
  ixs[n].append(i)
out = []
for k, v in ixs.items():
  d = series(v)
  if d != None:
    out.append((k, d))
