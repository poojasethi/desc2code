from collections import *

n = int(raw_input())
d = defaultdict(list)
A = map(int, raw_input().split())
for i,j in enumerate(A):
  d[j].append(i)

solutions = []
for j in sorted(d):
  inds = d[j]
  if len(inds) == 1:
    solutions.append((j,0))
  else:
    diff = inds[1]-inds[0]
    if all(inds[i+1]-inds[i]==diff for i in xrange(len(inds)-1)):
      solutions.append((j,diff))

print len(solutions)
for i,j in solutions:
  print i,j
