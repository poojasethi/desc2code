n=int(raw_input())
a=raw_input()
a=a.split()
pos=[0]*100001
d={}
for i in range(n):
	num=int(a[i])
	if pos[num]==-1:
		continue
	if num not in d:
		pos[num]=i
		d[num]=0
	elif d[num]==0:
		d[num]=i-pos[num]
		pos[num]=i
	else:
		if i-pos[num]==d[num]:
			pos[num]=i
		else:
			del d[num]
			pos[num]=-1
print len(d)
for key in sorted(d.iterkeys()):
	print key,d[key]
