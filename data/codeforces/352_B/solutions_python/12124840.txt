n = input()
arr = map(int,raw_input().split())
d = {}

for i,x in enumerate(arr):
    if x not in d:
        d[x] = [i,0,1] # index of last, slope, count
    elif d[x][2] == 1:
        d[x][1] = i - d[x][0]; d[x][0] = i; d[x][2] += 1
    elif d[x][2] > 1 and d[x][1] == i - d[x][0]:
        d[x][0] = i; d[x][2] += 1
    elif d[x][2] > 1 and d[x][1] != i - d[x][0]:
        d[x] = [-1,-1,-1]

ans = [(x,d[x][1]) for x in sorted(d.keys()) if d[x][2] >= 1]
print len(ans)
for x,y in ans:
    print x,y
