#!/usr/bin/env python

raw_input()
a = sorted((int(ai), i) for i, ai in enumerate(raw_input().split()))
a.append((0, 0))
ans = []
prev_x, prev_i = -1, -1
d = -1
for x, i in a:
    if prev_x == x:
        if d == 0:
            d = i - prev_i
        elif i - prev_i != d:
            d = -1
    else:
        if d != -1:
            ans.append((prev_x, d))
        d = 0
    prev_x, prev_i = x, i

print len(ans)
for x, d in ans:
    print x, d
