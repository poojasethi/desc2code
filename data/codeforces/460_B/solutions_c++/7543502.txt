#include<bits/stdc++.h>
#define ll  long long
#define MOD 1000000000
using namespace std;
int sum1(ll m){
int sum=0;
while(m!=0){
sum+=m%10;
m/=10;
}
return sum;
}

int main()
{   ll i,n,m,k;
cin>>n>>m>>k;
vector<ll> v;
for(i=1;i<=81;i++){
ll val=1;
for(int j=0;j<n;j++)
val*=i;
ll q=m*val+k;
int y=sum1(q);
if(y==i&&q<MOD){
v.push_back(q);
}
}
cout<<v.size()<<endl;
for(i=0;i<v.size();i++)
cout<<v[i]<<" ";

return 0;
}
