#include <cstdio>

int R, tab[1000];

int main()
{
	int a,b,c;
	scanf("%d%d%d",&a,&b,&c);
	for (int i=1;i<100;i++)
	{
		long long int x = i;
		for (int j=1;j<a;j++) x*=i;
		x*=b;
		x+=c;
		if ( x >= 1000*1000*1000 ) continue;
		int suma = 0;
		tab[R] = x;
		while (x)
		{
			suma+=x%10;
			x/=10;
		}
		if (suma==i) R++;
	}
	printf("%d\n",R);
	for (int i=0;i<R;i++) printf("%d ",tab[i]);
	printf("\n");
	return 0;
}
