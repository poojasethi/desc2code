#include<cstdio>
#include<algorithm>
using namespace std;
long long a,b,c,i,j,x,sum,cnt,y,A,ans[1005];
int main()
{
  scanf("%I64d%I64d%I64d",&a,&b,&c);
  for (i=1;i<=81;i++)
  {
    x=i;
    for (j=1;j<a;j++) x*=i;
    x=b*x+c;y=x;
    for (cnt=0,sum=0;x;x/=10ll)
      sum+=x%10ll;
    if (sum==i&&y<1000000000) ans[++A]=y;
  }
  sort(ans+1,ans+A+1);
  printf("%I64d\n",A);
  for (i=1;i<=A;i++) printf("%I64d ",ans[i]);
  return 0;
}
