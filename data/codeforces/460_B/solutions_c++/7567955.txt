#include<bits/stdc++.h>
using namespace std;

const int MAX = 1e9;
int sum(long long n)
{
	int ret = 0;
	while(n)
	{
		ret += n%10;
		n /= 10;
	}	
	return ret ;
}
	
int main()
{
	int a,b,c,count=0;
	cin>>a>>b>>c;

	long long  d[100];

	for(int i=1;i<=81;i++)
	{
		long long x=b*(pow(i*1.0,a))+c;
		if(x<MAX && x>0 && i==sum(x))
			d[count++]=x;
	}
	sort(d,d+count);
	cout<<count<<endl;
	for(int i=0;i<count;i++)
			cout<<d[i]<<" ";
	
	cout<<endl;	
	return 0;
		
}
