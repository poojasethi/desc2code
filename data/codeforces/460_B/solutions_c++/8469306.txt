#include<cstdio>
using namespace std;
long long a,b,c,t,i,tmp,as,dn[1010],l,tt,pp;
int main()
{
	scanf("%I64d%I64d%I64d",&a,&b,&c);
	for(i=1;i<=81;i++)
	{
		for(t=1,tmp=1;t<=a;t++)tmp=tmp*i;
		as=tmp*b+c;if(as>1e9)break;tt=0;pp=as;
		while (as)tt+=as%10,as/=10;
		if(tt==i)dn[++l]=pp;
	}printf("%I64d\n",l);
	for(i=1;i<=l;i++)printf("%I64d ",dn[i]);
}
