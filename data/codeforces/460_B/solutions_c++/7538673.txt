#include<cstdio>
#include<vector>
using namespace std;
const int mx = 1e9;
long long ans[100];
int f(int n){
int i=0;
while(n){
    i+=n%10;
    n=n/10;
}
return i;
}
int main(){
int i,j,a,b,c,pos=0;
long long d;
scanf("%d%d%d",&a,&b,&c);
for(i=1;i<100;i++){
    d=b;
    for(j=1;j<=a;j++)
        d*=i;
    d=d+c;
    if(f(d)==i&&d>0&&d<mx)
        ans[pos++]=d;
}
printf("%d\n",pos);
for(i=0;i<pos;i++)
    printf("%I64d ",ans[i]);
}
