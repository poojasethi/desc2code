#include <iostream>
#include <cmath>
#include <set>

using namespace std;

set<long long> s;

int main() {
	int a,b,c;
	cin>>a>>b>>c;
	for(int i=1;i<=81;i++) {
		double x=b*pow((double)i,a)+c;
		if(x>0&&x<1000000000)
		{
			int sx=x,k=0;
			while(sx) {
				k+=sx%10; sx/=10;
			}
			if(k==i) s.insert(x);
		}
	}
	cout<<s.size()<<endl;
	for(set<long long>::iterator it=s.begin();it!=s.end();it++) {
		cout<<*it<<" ";
	}
	
	
	return 0;
}
