#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll dig(ll n)
{
	ll s=0;
	while(n>0)
	{
		s=s+n%10;
		n=n/10;
	}
	return s;
}
ll pwr(ll a,ll b)
{
	ll i,ans=1;
	for(i=1;i<=b;i++)
		ans=ans*a;
	return ans;
}

int main()
{
	ll a,b,c,i;
	cin>>a>>b>>c;
	int arr[100004]={0};
	ll cnt=0,k=0;
	for(i=1;i<=81;i++)
	{
		ll x;
		x=b*pwr(i,a)+c;
		if(dig(x)==i and x>0 and x<1000000007)
		{
			cnt++;
			arr[k++]=x;
		}
	}
	cout<<cnt<<endl;
	for(i=0;i<k;i++)
		cout<<arr[i]<<' ';
	return 0;
}
