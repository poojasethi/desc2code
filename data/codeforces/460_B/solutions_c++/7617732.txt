int he(long long x)
{ int sum=0;
while(x)
{ sum+=x%10;x/=10;}

 return sum;
}

int main()
{ long long a,b,c,x,s,ans[82];
while(cin>>a>>b>>c)
{ s=0;
int i;
    for(i=1;i<82;i++)
    {x=b*(long long)(pow(i,a)+1e-10)+c;
    //cout<<x<<" "<<i<<" "<<(long long)(pow(i,a)+1e-10)<<endl;
    if(x<1e9&&i==he(x)) ans[s++]=x;
    }
    cout <<s<< endl;
    for(i=0;i<s;i++) cout<<ans[i]<<" ";

}
    return 0;
}
