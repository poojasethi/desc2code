#include <bits/stdc++.h>

using namespace std;

int sumdig(int n)
{
	int ans=0;
	while(n>0)
	{
		ans+=n%10;
		n/=10;
	}
	return ans;
}

int main()
{
	long long a,b,c,q,i;
	cin>>a>>b>>c;
	vector<long long> v;
	for(i=1;i<=81;i++)
	{
		q=1;
		for(int j=0;j<a;j++)
		{
			q*=i;
		}
		q = q*b+c;
		if(q<1000000000&&sumdig(q)==i)
		{
			v.push_back(q);
		}
	}
	cout<<v.size()<<endl;
	for(i=0;i<v.size();i++)
		cout<<v[i]<<" ";
	printf("\n");
	return 0;
}
