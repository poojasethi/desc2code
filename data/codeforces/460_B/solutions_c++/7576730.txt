#include<stdio.h>
main()
{
    int a,b,c,s,n=0,z[100];
    scanf("%d%d%d",&a,&b,&c);
    for(s=1;s<=82;s++)
    {
        long long int q,m,x,t;
        q=1;
        for(int i=1;i<=a;i++)   q=q*s;
        x=b*q+c;
        m=x;
        t=0;
        while(m>0)
        {
            t+=m%10;
            m/=10;
        }
        if (t==s && x>0 && x<1000000000)    z[++n]=x;
    }
    printf("%d\n",n);
    for(int i=1;i<=n;i++)   printf("%d ",z[i]);
}
