#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long count=0,s[100]{},arr[100000]{},a,b,c,p=1;
	cin>>a>>b>>c;
	for(int i=1;i<=81;i++)
	{
		p=1;
		for(int j=0;j<a;j++) p*=i;
		s[i]=b*p+c;
		if(s[i]>0&&s[i]<1000000000)
		{
			int sum=0,x;
			x=s[i];
			for(int j=0;j<=9;j++)
			{
				sum+=x%10;
				x/=10;
			}
			if(sum==i) arr[count++]=s[i];
		}
	}
	cout<<count<<endl;
	for(int i=0;i<count;i++)
	{
		cout<<arr[i]<<" ";
	}
	return 0;
}
