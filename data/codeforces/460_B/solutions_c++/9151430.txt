#include<stdio.h>
int s(int x)
{
	int v = 0;
	while(x)
	{
		v += x % 10;
		x = x / 10;
	}
	return v;
}
int main()
{
	int a, b, c, i, d[100], j = 0;
	long long int x;
	scanf("%d%d%d", &a, &b, &c);
	for(i = 1; i <= 81; i++)
	{
		x = 1;
		int l = a;
		while(l--)
		{
			x *= i;
		}
		x = b * x + c;
		if( x > 0 && x < 1000000000 && s((int ) x) == i)
			d[j++] = (int ) x;
	}
	printf("%d\n", j);
	int f;
	for(f = 0; f < j; f++)
		printf("%d ", d[f]);
	return 0;
}
