#include<iostream>
#include<set>
#include <algorithm>
using namespace std;
long long n,ma,c,a,b,q,w,e,r,h[1000000],y,o,t;

int main(){
cin>>a>>b>>c;
for(int w=1;w<=81;w++){
	y=1;
	for(int j=1;j<=a;j++){
		y*=w;
	}
	q=b*y+c;
	if(q<=0)continue;
	if(q>999999999)continue;
	e=q;
	r=0;
	while(e>0){
		r+=e%10;
		e=e/10;
	}
	if(r==w)o++,h[o]=q;

}
cout<<o<<endl;;
for(int k=1;k<=o;k++){
	cout<<h[k]<<" ";
}	
}
