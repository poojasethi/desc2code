#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll s(ll x, ll r = 0){
for( ; x; x /= 10) r += x % 10;
return r;
}

ll a, b, c;

ll p(ll x, ll r = 1){
for(int i = 0; i < a; i++) r *= x;
return r;
}

vector<ll> res;

int main(){
cin >> a >> b >> c;
for(int i = 1; i < 10000; i++){
ll x = b * p(i) + c;
if(s(x) == i && x < 1e9) res.push_back(x);
}
cout << res.size() << '\n';
for(int i = 0; i < res.size(); i++) cout << res[i] << ' ';
}
