#include<cstdio>
#include<iostream>
#define LL long long 
LL s,a,b,c,q[100];
using namespace std;
int num = 0;
int main()
{
    cin>>a>>b>>c;
	for (LL i = 1;i<=81;i++)
	{
		s = b;
        for (int j = 0;j<a;j++)
		{
			s*=i;
		}
		s+=c;
		if (s>=1000000000) break;
		LL ans = 0;
		LL now = s;
		while (s) {ans+=(s % 10); s/=10;}
		if (ans==i) q[num++] = now; 
	}

    printf("%d\n",num);
	for (int i = 0;i<num;i++)
		cout<<q[i]<<" ";
	return 0;
}
