#include<stdio.h>
#include<math.h>
int q[100];
int f(long long a)
{
int s=0;;
while(a/10||a%10)
{
s+=a%10;
a=a/10;
}
return s;
}
int main()
{
	long long t;
int i;
int a,b,c,n;
n=0;
scanf("%d%d%d",&a,&b,&c);
for(i=1;i<=81;i++)
{
t=b*(long long)(pow(i,a)+0.5)+c;
if(t>1000000000)
break;
if(f(t)==i)
{q[n++]=t;}
}
printf("%d\n",n);
if(n)
{
for(i=0;i<n;i++)
printf("%d ",q[i]);
printf("\n");
}
return 0;
}
