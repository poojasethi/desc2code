#include <bits/stdc++.h>

using namespace std;

unsigned long long re[10000];

int main()
{
   long long a,b,c,d,f,t=1,y=0;
   cin>>a>>b>>c;
   for(int i=1;i<=81;i++)
   {
		t=1;
	  for(int j=1;j<=a;j++)
	  {
		 t*=i;
	  }
	  d=0;
	  f=t*b+c;
	  if(f>999999999)
	  continue;
	  while(f>0)
	  {
		  d+=f%10;
		  f/=10;
	  }
	  if(i==d)
	  re[++y]=t*b+c;
   }	
   cout<<y<<endl;
   for(int i=1;i<=y;i++)
   cout<<re[i]<<" ";
   return 0;
}
