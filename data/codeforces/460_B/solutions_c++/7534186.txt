# include<iostream>

using namespace std;

long long int ma,me[100];

int main()
{
	
	int a,b,c;
	cin>>a>>b>>c;
	
	for(int i=1;i<=100;i++)
	{
		long long int f=1;
		
		for(int j=1;j<=a;j++)
		  f*=i;
		 
		long long int q=b*f+c;
		long long int fff=q,ff=0;
		
		while(q>0)
		{
			ff+=q%10;
			q/=10;
		}
		
		if(ff==i && fff>0 && fff<=1000000000)
		{
		   ma++;
		   me[ma]=fff;
		}
	}
	
	cout<<ma<<endl;
	
	for(int i=1;i<=ma;i++)
	  cout<<me[i]<<" ";
	return 0;
}
