#include<cstdio>
typedef long long LL;
int a,b,c,i,j,ans;
LL f[200];
int get(LL x){return x == 0 ? 0 : get(x / 10) + x % 10;}
int main()
{
	scanf("%d%d%d", &a, &b, &c);
	for (i=1; i<=81; i++){
		f[i]=1;
		for (j=1; j<=a; j++)
			f[i]=1LL*f[i]*i;
	}
	for (i=1; i<=81; i++){
		LL w=1LL*f[i]*b+1LL*c;
		if (i==get(w) && w<=(LL)1e9) ++ans,f[99+ans]=w;
	}
	printf("%d\n",ans);
	for (i=100; i<=99+ans; i++) printf("%d ",f[i]);
}
