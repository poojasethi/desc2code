#include <iostream>

using namespace std;
long long a,b,c,p,sx,SX,x,y,i,cnt,sol[100];
int main()
{
    cin>>a>>b>>c;
    for(sx=0;sx<=81;sx++)
    {
        for(p=1,i=1;i<=a;i++)
            p*=sx;
        x=b*p+c;
        if(x<=0) continue;
        if(x>=1000000000)continue;
        for(y=x,SX=0;y;y/=10)SX+=y%10;
        if(SX==sx)
        {
            sol[++cnt]=x;
        }
    }
    cout<<cnt<<'\n';
    for(i=1;i<=cnt;i++)
        cout<<sol[i]<<' ';

    return 0;
}
