#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
typedef long long ll;
vector<ll> ans;

int main(){
	ll a,b,c;
	cin>>a>>b>>c;
	for (ll i=1;i<82;i++){
		ll x=1;
		for (int j=0;j<a;j++) x*=i;
		x=x*b+c;
		ll t=0,xx=x;
		while (x){
			t+=x%10;
			x/=10;
		}
		if (t==i && xx<1000000000ll) ans.push_back(xx);
	}
	sort(ans.begin(),ans.end());
	cout<<ans.size()<<endl;
	for (int i=0;i<ans.size();i++) cout<<ans[i]<<' ';
}
