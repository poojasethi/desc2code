#include <bits/stdc++.h>
using namespace std;

long long a , b , c , k[100] , l = 0;

int main(){
	cin >> a >> b >> c;
	for (int i = 1 ; i <= 81 ; ++ i){
		long long n;
		int j;
		for (n = 1 , j = 0 ; j < a ; ++ j) n *= i;
		n = (n * b) + c; long long xxx = n , sum = 0;
		if (n > 1e9) break;
		while (xxx){
			sum += xxx % 10;
			xxx /= 10;
		}
		if (sum == i) k[l++] = n;
	}
	cout << l << endl;
	for (int i = 0 ; i < l ; ++ i){
		if (i) cout << " ";
		cout << k[i];
	}
}
