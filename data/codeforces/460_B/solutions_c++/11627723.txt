#include <bits/stdc++.h>
using namespace std;
vector<long long>res;
int main(){
  long long a,b,c;
  cin>>a>>b>>c;
  for(int i=1;i<81;i++){
      long long ans=1;
      for(int j=0;j<a;j++)
	ans*=i;
      ans=ans*b+c;
      if(ans>=1e9 || ans<=0)
	continue;
      long long ans2=0,temp=ans;
      while(ans>0)
	ans2+=ans%10,ans/=10;
      if(ans2==i)
	res.push_back(temp);
    }
  cout<<res.size()<<endl;
  sort(res.begin(),res.end());
  for(int i=0;i<res.size();i++)
    cout<<res[i]<<" ";
  return 0;
}
