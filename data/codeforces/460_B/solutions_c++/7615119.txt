#include<cstdio>
#include<algorithm>
using namespace std;
int ans[100];
int main()
{
	int cnt=0,a,b,c;
	scanf("%d%d%d",&a,&b,&c);
	for(int i=1;i<81;++i){
		int x=b;
		for(int j=0;j<a;++j)x*=i;
		x+=c;
		if(x>=1e9)break;
		if(x<=0)continue;
		int d=0,z=x;
		while(x){
			d+=x%10;
			x/=10;
		}
		if(d==i)ans[cnt++]=z;
	}
	sort(ans,ans+cnt);
	printf("%d\n",cnt);
	if(cnt)printf("%d",ans[0]);
	for(int i=1;i<cnt;++i)printf(" %d",ans[i]);
	if(cnt)puts("");
	return 0;
}
