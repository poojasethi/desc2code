#include<iostream>
#include<vector>
using namespace std;
int main()
{
	long long a,b,c;
	cin>>a>>b>>c;
	vector<long long>v;
	for(long long int i=1;i<=81;i++)
	{
		long long m=1;
		for(long long int j=1;j<=a;j++)
			m*=i;
		long long x=b*m+c;
		if(x>0&&x<1000000000)
		{
			long long p=x,sum=0;
			while(p!=0)
			{
				sum+=p%10;
				p/=10;
			}
			if(sum==i)
				v.push_back(x);
		}
	}
	long long len=v.size();
	cout<<len<<endl;
	for(int i=0;i<len;i++)
		cout<<v[i]<<" ";
	return 0;
}
