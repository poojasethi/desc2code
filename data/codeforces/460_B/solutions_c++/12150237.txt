#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long a, b, c;
	cin>>a>>b>>c;
	long long s;
	vector<long long>ans;
	for (int i=1; i<82; i++)
	{
		s=1;
		for (int j=0; j<a; j++)
		{
			s=s*i;
		}
		long long z=b*s+c;
		long long t=0;
		long long y=z;
		while (y>0)
		{
			t=t+y%10;
			y=y/10;
		}
		if (t==i && z>0 && z<1000000000)
		{
			ans.push_back(z);
		}
	}
	cout<<ans.size()<<endl;
	for (int i=0; i<ans.size(); i++)
	{
		cout<<ans[i]<<" ";
	}
}
