#include <bits/stdc++.h>
using namespace std;
int main ()
{
vector <int long long> my;
int long long a,b,c,i,j;
cin>>a>>b>>c;
for (i=1;i<=81;i++)
{
    int long long d=1;
    for (j=1;j<=a;j++)
    {
    d*=i;
    }
    int long long sum=0,g=d*b+c;
    while (g>=1)
    {
    sum+=g%10;
    g/=10;
    }
    if (sum==i && d*b+c>0 && d*b+c<1000000000){my.push_back(d*b+c);}
}
cout<<my.size()<<endl;
for (i=0;i<my.size();i++)
{
cout<<my[i]<<" ";
}
}
