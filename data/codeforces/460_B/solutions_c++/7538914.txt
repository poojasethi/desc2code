#include <iostream>
using namespace std;
long long s(long long x)
{
	int rc=0;
	while (x>0) {
		rc+=x%10; x/=10;
	}
	return rc;
}
int main()
{
	long long a, b, c, i, x, y, z, ans[81], cnt=0;
	cin>>a>>b>>c;
	for (y=1; y<=81; y++) {
		for (z=1,i=0; i<a; i++) z*=y;
		x=b*z+c;
		if (x<1e9 && y==s(x)) ans[cnt++]=x;
	}
	cout<<cnt<<endl;
	for (i=0; i<cnt; i++) cout<<ans[i]<<" ";
	cout<<endl;
	return 0;
}
