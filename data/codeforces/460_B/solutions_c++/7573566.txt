#include<iostream>
#include<math.h>
using namespace std;
long long a,b,c,s,e,x,t[100],n=0,i,j,m,f;
main()
{
    cin>>a>>b>>c;
    for(s=0;s<=81;s++)
       {
         f=1;
         for(j=1;j<=a;j++)
            f*=s;
         x=b*f+c;
         m=x;
         e=0;
         while(m>0)
         {
             e=e+m%10;
             m=m/10;
         }
         if(e==s&&x>0&&x<1000000000) t[++n]=x;
       }
    cout<<n<<endl;
    for(i=1;i<=n;i++)
       cout<<t[i]<<" ";
}
