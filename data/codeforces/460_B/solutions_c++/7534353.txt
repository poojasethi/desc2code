#include<bits/stdc++.h>
using namespace std;
long long power(int a, int b)
{
	long long res=1;
	for(int i=1;i<=b;i++)
		res*=a;
	return res;
}
int main()
{
	int a,b,c,i,cnt=0,sum;
	long long x,n;
	vector<int> vec;
	cin>>a>>b>>c;
	for(i=1;i<=90;i++)
	{
		x=b*power(i,a) + c;
		n=x;
		if(n>=1000000000)
			break;
		sum=0;
		while(x)
		{
			sum+=x%10;
			x/=10;
		}
		if(sum==i)
		{
			cnt++;
			vec.push_back(n);
		}
	}
	cout<<cnt<<endl;
	for(i=0;i<vec.size();i++)
		cout<<vec[i]<<' ';
	return 0;
}
