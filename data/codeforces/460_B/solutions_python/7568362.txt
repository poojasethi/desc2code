def c_c(a):
    f = 0;
    while a != 0:
        f += (a % 10);
        a /= 10;
    return f;

a, b, c = map(int, raw_input().split())
ans = []
for i in range(1, 82):
    x = b * i**a + c;
    if (x > 0 and x < 1000000000 and c_c(x) == i):
        ans.append(x)
print len(ans)
print " ".join(map(str, ans))
