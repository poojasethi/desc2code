a,b,c = map(int, raw_input().split())
def getx(x):
    ans = 0
    while x != 0:
        ans += x%10
        x /= 10
    return ans
ans = []
for j in range (1, 82):
    x = b*(j**a) + c
    if x < 10**9 and x > 0 and j == getx(x):
         ans.append(x)
print len(ans)
print ' '.join(map(str, ans))
