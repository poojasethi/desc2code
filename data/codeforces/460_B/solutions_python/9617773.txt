def summa(x):
    k = 0
    for i in x:
        k += int(i)
    return k

a, b, c = map(int, raw_input().strip().split())
sx = range(1, 82)
d = 10**9
ch = []
for i in sx:
    f = b*(i**a)+c
    if f > 0 and summa(str(f)) == i and f < d:
        ch.append(f)

print len(ch)
print ' '.join(str(x) for x in ch)
