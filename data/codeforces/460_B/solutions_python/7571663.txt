a, b, c = map(int, raw_input().split())
s = []
for i in range(1, 90):
    x = (i ** a) * b + c
    if x >= 1000000000: break
    if x <= 0: continue
    ans = 0 
    tempx = x
    while x:
        ans += x % 10
        x = x / 10 
    if ans == i:
        s.append(tempx)
print len(s)
for i in s:
    print i
