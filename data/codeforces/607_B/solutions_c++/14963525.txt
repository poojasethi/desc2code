#include<bits/stdc++.h>
using namespace std;


int a[600],n;
int dp[600][600];

int main()
{
    int n;
    scanf("%d",&n);
    for (int i=1;i<=n;i++) {
            scanf("%d",&a[i]);
            dp[i][i]=1;
    }

    for (int i=n;i>=1;i--)
    for (int j=i+1;j<=n;j++)
    {
        dp[i][j]=1<<28;
        for (int k=i;k<j;k++)
        dp[i][j]=min(dp[i][j],dp[i][k]+dp[k+1][j]);
        if (a[i]==a[j])
        dp[i][j]=min(dp[i][j],max(1,dp[i+1][j-1]));
    }
    cout<<dp[1][n]<<endl;
}
