#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;
int a[520],f[520][520];
int dp(int x,int y)
{
	int ans=100000;
	if(f[x][y])return f[x][y];
	if(x==y)return f[x][y]=1; 
	if(a[x]==a[y])
	{
		if(x+1==y)return f[x][y]=1;
		else ans=dp(x+1,y-1);
	}
	for(int k=x;k<y;k++)
	{
		int buf=dp(x,k)+dp(k+1,y);
		if(buf<ans)ans=buf; 
	}
	return f[x][y]=ans;
}
int main()
{
	memset(f,0,sizeof(f));
	int n;cin>>n;
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	cout<<dp(1,n)<<endl;
	return 0; 
}
