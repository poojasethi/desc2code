#include <bits/stdc++.h>
using namespace std;
int n;
int a[505];
int f[505][505];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",a+i);
	for(int i=1;i<=n;i++)
		f[i][i]=f[i][i-1]=1;
	for(int l=2;l<=n;l++)
		for(int st=1;st<=n-l+1;st++)
		{
			if(a[st]==a[st+l-1])
				f[st][st+l-1]=f[st+1][st+l-2];
			else f[st][st+l-1]=0x3f3f3f3f;
			for(int k=st;k<=st+l-1;k++)
				f[st][st+l-1]=min(f[st][st+l-1],f[st][k]+f[k+1][st+l-1]);
		}
	printf("%d\n",f[1][n]);
	return 0;
}
