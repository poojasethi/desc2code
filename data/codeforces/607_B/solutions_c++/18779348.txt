#include<bits/stdc++.h>
using namespace std;

int b[503][503];
int a[503];
int n,i,j;
int ruby(int l,int r)
{
    if(l>=r)
        return 1;

    if(b[l][r]!=-1)
        return b[l][r];
    int e=100000000;
    if(a[l]==a[r])
        e=ruby(l+1,r-1);
    for(int j=r-1; j>=l; j--)
        e=min(e,ruby(l,j)+ruby(j+1,r));
    return b[l][r]=e;

}
int main()
{
    cin>>n;
    memset(b,-1,sizeof(b));
    for(i=1; i<=n; i++)
        cin>>a[i];
    cout<<ruby(1,n)<<endl;
    return 0;
}
