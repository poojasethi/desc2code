#include <bits/stdc++.h>
#define sp ' '
using namespace std;

int n, a[501], dp[510][510];
int main() {
	cin >> n;
	for(int i = 1; i <= n; i++) cin >> a[i];
	for(int i = n; i >= 1; i--) {
		dp[i][i] = 1;
		for(int j = i+1; j <= n; j++) {
			dp[i][j] = dp[i+1][j]+1;
			if(a[i] == a[i+1]) dp[i][j] = min(dp[i+2][j]+1, dp[i][j]);
			for(int k = i+2; k <= j; k++) {
				if(a[i] == a[k]) dp[i][j] = min(dp[i+1][k-1]+dp[k+1][j], dp[i][j]);
			}
			//cout << i << sp << j << sp << dp[i][j] << endl;
		}
	}
	cout << dp[1][n];
}
