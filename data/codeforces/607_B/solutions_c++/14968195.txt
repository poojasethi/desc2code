#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxn=510;

int f[maxn][maxn];
int c[maxn];
int n;

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&c[i]);
	for (int i=1;i<=n;i++) f[i][i]=1;
	for (int i=n-1;i>=1;i--)
	for (int j=i+1;j<=n;j++)
	{
		f[i][j]=1+f[i+1][j];
		if (c[i]==c[i+1]) f[i][j]=min(f[i][j],1+f[i+2][j]);
		for (int k=i+2;k<=j;k++) if (c[i]==c[k]) f[i][j]=min(f[i][j],f[i+1][k-1]+f[k+1][j]);
	}
	printf("%d\n",f[1][n]);
	return 0;
}
