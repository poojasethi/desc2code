#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
ll n,a[555],dp[505][505];
ll f(ll l,ll r)
{
    ll i;
    if(r<=l)
        return 1;
    if(dp[l][r])
        return dp[l][r];
    ll v=550;
    if(a[l]==a[r])
        v=min(v,f(l+1,r-1));
    for(i=l;i<r;i++)
        v=min(v,f(l,i)+f(i+1,r));
    dp[l][r]=v;
    return dp[l][r];

}
int main()
{
    ll i;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        dp[i][i]=1;
    }
    cout<<f(1,n);
}
