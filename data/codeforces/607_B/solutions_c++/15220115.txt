#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int n,f[505][505],a[505];
int main()
{
	scanf("%d",&n);
	memset(f,63,sizeof(f));
	for (int i=1;i<=n;++i) {
		f[i][i]=1;
		scanf("%d",a+i);
	}
	for (int l=1,j;l<=n;++l)
		for (int i=1;i<n;++i) {
			j=i+l;
			if (j>n) break;
			for (int k=i;k<j;++k)
				f[i][j]=min(f[i][j],f[i][k]+f[k+1][j]);
			if (a[i]==a[j])
				if (i<j-2) f[i][j]=min(f[i][j],f[i+1][j-1]);
				else f[i][j]=1;
		}
	printf("%d\n",f[1][n]);
}
