#include <iostream>
#include "bits/stdc++.h"
using namespace std;
int n,a[505];
int dp[505][505];
int main() {
	cin>>n;
	for(int i=0;i<n;i++)
	  cin>>a[i];
	 
	for(int i=0;i<n;i++)
	  dp[i][i]=1;
	  
	  for(int i=n-1;i>=0;i--)
	  {
	  	for(int j=i+1;j<n;j++)
	  	{
	  		dp[i][j]=n+1;
	  		for(int k=i;k<j;k++)
	  		{
	  			dp[i][j]=min(dp[i][j],max(1,dp[i][k]+dp[k+1][j]));
	  		}
	  		if(a[i]==a[j])
	  		{
	  			dp[i][j]=min(dp[i][j],max(1,dp[i+1][j-1]));
	  		}
	  	}
	  }
	 cout<<dp[0][n-1]<<endl;
	return 0;
}
