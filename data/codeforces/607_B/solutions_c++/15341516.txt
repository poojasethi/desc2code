#include<bits/stdc++.h>

using namespace std;

const int Maxn=507;

int a[Maxn];
int dp[Maxn][Maxn];

main() 
{
  ios_base::sync_with_stdio(0);cin.tie(0);
  int n;
  cin>>n;
  for(int i=0;i<n;i++)
    cin>>a[i];
  for(int i=n-1;i>=0;i--)
    for(int j=i;j<n;j++)
      {
	dp[i][j]=(a[i]==a[j])*(max(1,dp[i+1][j-1]));
	if(!dp[i][j])
	  dp[i][j]=500;
	for(int k=i;k<j;k++)
	  dp[i][j]=min(dp[i][j],max(1,dp[i][k]+dp[k+1][j]));
      }
  cout<<dp[0][n-1]<<endl;
  return 0;
}
