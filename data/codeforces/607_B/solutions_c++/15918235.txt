#include<bits/stdc++.h>
using namespace std;
int n,a[1000],dp[1000][1000];
int main(){
	cin>>n;
	for(int i=1;i<=n;i++)
	cin>>a[i];
	for(int j=1;j<=n;j++){
		for(int i=1;i<=j;i++){
			dp[i][j]=dp[i][j-1]+1;
			for(int k=i;k<j;k++)
			if(a[k]==a[j])
			dp[i][j]=min(dp[i][j],dp[i][k-1]+max(dp[k+1][j-1],1));
		}
	}
	cout<<dp[1][n]<<endl;
}
