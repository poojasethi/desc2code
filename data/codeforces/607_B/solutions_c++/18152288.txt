#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <queue>
#include <map>
using namespace std;
const int M = 510;
int f[M][M];
int n,a[M];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(int j=1;j<=n;j++)
	{
		for(int i=1;i<=j;i++)
		{
			f[i][j]=f[i][j-1]+1;
			for(int k=i;k<j;k++)
				if(a[k]==a[j])
					f[i][j]=min(f[i][j],f[i][k-1]+max(f[k+1][j-1],1));
		}
	}
	printf("%d\n",f[1][n]);
    return 0;
}
