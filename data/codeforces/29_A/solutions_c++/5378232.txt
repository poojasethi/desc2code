#include <iostream>
using namespace std;
#define MAXN 1000
int a[MAXN],d[MAXN];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i]>>d[i];
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
			if(a[i]+d[i]==a[j] && a[j]+d[j]==a[i])
			{
				cout<<"YES\n";
				return 0;
			}
	cout<<"NO\n";
	return 0;
}
