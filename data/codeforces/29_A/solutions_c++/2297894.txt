#include <stdio.h>
int main(){
	int n,x[1000],d[1000];
	scanf("%d",&n);
	int mark=0;
	for(int i=1;i<=n;i++){
		scanf("%d%d",&x[i],&d[i]);
		for(int j=1;j<i;j++){
			if(x[i]+d[i]==x[j]&&x[j]+d[j]==x[i]){mark=1;break;}
			if(mark)break;
		}	
	}
	if(mark)printf("YES\n");
	else printf("NO\n");
	return 0;
}
