#include<stdio.h>
int main()
{
	int n,i,x[101],y[101],d,j,dd=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x[i],&d);
		y[i]=x[i]+d;
	}
	for(i=0;dd==0&&i<n;i++)
	{
		for(j=0;dd==0&&j<n;j++)
		{
			if(x[i]==y[j]&&x[j]==y[i]&&i!=j)
			dd=1;
		}
	}
	if(dd)
	printf("YES\n");
	else
	printf("NO\n");
}
