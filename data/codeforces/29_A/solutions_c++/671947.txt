#include <iostream>
#include <map>
using namespace std;
map <int,int> mp;
bool solve()
{
	int i=0,N,x,y;
	cin >> N;
	for(; i<N; i++)
	{
		cin >> x >> y;
		mp[x]=y+=x;
		if(mp.find(y)!=mp.end() && mp[y]==x)
			return 1;
	}
	return 0;
}
int main()
{
	solve()? cout << "YES" : cout << "NO";
	return 0;
}
