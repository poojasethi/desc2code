#include<cstdio>
int n,d[101],b[101];
int main( )
{
 scanf("%d",&n);	
	 for(int i=1;i<=n;i++) {
	  scanf("%d %d",&d[i],&b[i]);	
	 	
	 	
	 }
	  for(int i=1;i<=n;i++)
	   for(int j=i+1;j<=n;j++){
	   	if(d[i]+b[i]==d[j] && d[j]+b[j]==d[i]) {printf("YES\n"); return 0; }
	   	
	   	}
	   	printf("NO\n");
	   	return 0;
}
