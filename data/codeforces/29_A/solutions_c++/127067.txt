#include <cstdio>

int n, x[128], d[128];
int main(void) {
	int i, j;
	scanf("%d",&n);
	for(i = 0; i < n; i++)
		scanf("%d %d",&x[i],&d[i]);
	for(i = 0; i < n; i++)
		for(j = i+1; j < n; j++)
			if(x[i] + d[i] == x[j] && x[j] + d[j] == x[i]) {
				printf("YES\n");
				return 0;
			}
	printf("NO\n");
	return 0;
}
