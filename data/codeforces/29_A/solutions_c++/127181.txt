#include<cstdio>

using namespace std;

int main(void){
	bool jud=false;
	int n,x[128],d[128];
	scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%d%d",&x[i],&d[i]);
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			if(x[i]+d[i]==x[j] && x[j]+d[j]==x[i])jud=true;
		}
	}
	if(jud)printf("YES\n");else printf("NO\n");
	return 0;
}
