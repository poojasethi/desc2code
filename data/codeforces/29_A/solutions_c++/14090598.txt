#include <bits/stdc++.h>

using namespace std;

set<pair<int,int> > s;
int n ;
int x , d ;

int main()
{
cin>>n;
while(n--){

cin>>x>>d;

if( s.find( make_pair( x+d , -d ) )!= s.end() ){
        cout<<"YES"<<endl;
        return 0;
         }
s.insert(make_pair(x,d));

}
    cout<<"NO"<<endl;

    return 0;
}
