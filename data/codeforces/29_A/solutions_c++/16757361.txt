#include<iostream>
using namespace std;
int a[20005];
int main(){
	int n,i,x,d,flag=0;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>x>>d;
		a[x+10000]=x+d+10000;
		if(x+d+10000<=20000&&x+d+10000>=0){
			if(a[x+d+10000]==x+10000)flag=1;
		}
	}
	if(flag)cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
	return 0;
}
