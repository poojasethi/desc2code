#include<iostream>
using namespace std;
int main()
{
int n,i,j;
cin>>n;
int *x= new int [n];
int *d= new int [n];
for(i=0; i<n; i++)
cin>>x[i]>>d[i];
for(i=0; i<n-1; i++)
for(j=i+1; j<n; j++)
if((x[i]+d[i]==x[j]) && (x[j]+d[j]==x[i]))
{
cout<<"YES";
return 0;
}
cout<<"NO";
delete d;
delete x;
return 0;
}
