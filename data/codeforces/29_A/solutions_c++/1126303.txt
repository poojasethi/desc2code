#include<iostream>
using namespace std;
int x[101],d[101];
int main()
{
	int n,flag=0,i,j;
	cin>>n;
	for(i=0;i<n;i++) cin>>x[i]>>d[i];

	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(x[i]+d[i]==x[j]&&x[j]+d[j]==x[i])
				flag=1;

		}

	}
	if(flag)
		cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
	return 0;

}
