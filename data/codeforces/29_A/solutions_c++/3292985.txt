#include <iostream>
using namespace std;

int main(void)
{
	int n, x[100], d[100];

	cin >> n;
	for (int i=0; i<n; ++i)
		cin >> x[i] >> d[i];
	bool flg=false;
	for (int i=0; i<n; ++i)
		for (int j=i+1; j<n; ++j)
			if ( x[i]+d[i]==x[j] && x[i]==x[j]+d[j] )	flg=true;
	cout << (flg?	"YES" : "NO") << endl;
	return 0;
}
