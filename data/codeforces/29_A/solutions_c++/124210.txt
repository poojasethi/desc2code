#include<iostream>
using namespace std;
int main()
{
	int n, x[100], d[100];
	cin>>n;
	for(int i = 0; i < n; i++) cin>>x[i]>>d[i];
	for(int i = 0; i < n - 1; i++)
		for(int j = i + 1; j < n; j++)
			if (x[i] + d[i] == x[j] && x[i] == x[j] + d[j])
			{
				cout<<"YES";
				return 0;
			}
	cout<<"NO";
	return 0;
}
