#include <iostream>
#include <cstdio>
using namespace std;
int n,a[1000],d[1000];
int main(){
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i]>>d[i];
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            if((a[i]+d[i]==a[j]) && (a[j]+d[j]==a[i])) {cout<<"YES"; return 0;}
    cout<<"NO";
    
    return 0;
}
