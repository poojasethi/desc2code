#include <iostream>
using namespace std;
int p[100],d[100];
int main(){
	int n,pos=0;
	for(cin>>n;pos<n;++pos)
		cin>>p[pos]>>d[pos];
	bool ok = false;
	for(int i=0;i<n;++i)
		for(int j=i+1;j<n;++j)
			if(p[i]+d[i]==p[j] && p[i] == d[j]+p[j])
				ok = true;
	if(ok)cout<<"YES\n";
	else  cout<<"NO\n";
}
