#include <stdio.h>

int cnt[40006]={1000000};
int main()
{
    int n,x,y;
    int r=0;
    scanf("%d",&n);
    while(n--)
    {
        scanf("%d%d",&x,&y);
        cnt[x+20001]=x+y+20001;
        if(cnt[x+y+20001]==x+20001)
            r=1;
    }
    if(r) printf("YES\n");
    else printf("NO\n");
    return 0;
}
