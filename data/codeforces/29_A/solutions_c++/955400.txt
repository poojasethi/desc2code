#include<iostream>
#include<set>
using namespace std;
const int maxn=110;
set<pair<int,int> > st;
int main(){
	int n;
	bool ans=false;
	cin>>n;
	while(n--){
		int x,d;
		cin>>x>>d;
		if(st.find(make_pair(x+d,x))!=st.end())
			ans=true;
		else
			st.insert(make_pair(x,x+d));
	}
	cout<<(ans?"YES":"NO")<<endl;
}
