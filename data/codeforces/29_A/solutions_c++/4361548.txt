#include<stdio.h>
int main()
{
	int n,i,j;
	int g[200],x[200],d[200];
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&x[i],&d[i]);
		g[i]=x[i]+d[i];
	}
	for(i=1;i<=n;i++)
		for(j=i;j<=n;j++)
			if((g[i]==x[j])&&(g[j]==x[i]))
			{
				printf("YES");
				return 0;
			}
	printf("NO");
	return 0;
}
