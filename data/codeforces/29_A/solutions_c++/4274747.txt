#include<iostream>
using namespace std;
int main () {
	int n,a[150]={},b[150]={},i,j,x,y;
	cin >> n;
	for(i=1;i<=n;i++)
		cin >> a[i] >> b[i];
	for(i=1;i<=n;i++)
	{
		x=a[i]+b[i];
		for(j=1;j<=n;j++)
		{
			y=a[j]+b[j];
			if(x==a[j] && a[i]==y)
			{
				cout << "YES" << endl;
				return 0;
			}
		}
	}
	cout << "NO" << endl;
	return 0;
}
