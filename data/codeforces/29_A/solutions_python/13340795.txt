n = int(raw_input())
x = []
d = []
for i in range(n):
    s = [int(i) for i in raw_input().split()]
    x.append(s[0])
    d.append(s[1])
ans = False
for i in range(n-1):
    for j in range(i+1, n):
        if (x[i] + d[i] == x[j]) and (x[j] + d[j] == x[i]):
            ans = True
print "YES" if ans else "NO"
