import sys

def sol():
	n = int (sys.stdin.readline())
	x = [0] * n
	xt = [0] * n
	for i in xrange (n):
		(x[i], d) = [int(it) for it in sys.stdin.readline().split()]
		xt[i] = x[i] + d
	for i in xrange (n):
		for j in xrange (i+1, n):
			if x[i] == xt[j] and xt[i] == x[j]:
				return True
	return False
	
	
def main():
	if sol():
		print 'YES'
	else:
		print 'NO'


if len (sys.argv) > 1:
	sys.stdin = open (sys.argv[1], 'r')
main()
