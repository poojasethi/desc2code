def main():
    n = int(raw_input())
    camels = []
    for _ in range(0,n):
        s = raw_input().split()
        camels.append((int(s[0]), int(s[1])))
    camels.sort()
    #print camels
    for i in range(0,n):
        for j in range(i+1,n):
            if (camels[i][1] == camels[j][0]-camels[i][0] and
                camels[j][1] == camels[i][0]-camels[j][0]):
                print 'YES',
                return
    print "NO",
if __name__ == '__main__':
    main()
