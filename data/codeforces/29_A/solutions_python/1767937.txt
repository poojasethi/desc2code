import sys

n = input()

x = []
d = []
for i in xrange(n):
    line = raw_input().split()
    x.append(int(line[0]))
    d.append(int(line[1]))

ans = "NO"
for i in xrange(n):
    for j in xrange(i + 1, n):
        if (x[i] + d[i] == x[j] and x[j] + d[j] == x[i]):
            ans = "YES"

print ans
