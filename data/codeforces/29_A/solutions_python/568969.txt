#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

def solve(C):
    for x,d in C.iteritems():
        if (x+d) in C:
            if C[x+d] == -d:
                return 'YES'
    return 'NO'

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

n = int(input.readline())

C = {}
for _ in range(n):
    A = numbers_from_line()
    x = A[0]
    d = A[1]
    C[x] = d

a = solve(C)
output.write('%s\n' % a)
