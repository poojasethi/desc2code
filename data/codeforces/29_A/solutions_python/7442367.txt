x=[0 for i in range(0,100000)]
d=[0 for i in xrange(100000)]

n=input()

for i in xrange(n):
	a,b=map(int,raw_input().split())
	a+=10000
#	print "aa  ",a,b,x[a],d[a]
	x[a]=a;
	d[a]=b;
#	print "aa  ",a,b,x[a],d[a]
	
isok=False
	
for i in xrange(100000):
	if d[i]!=0:
#		print i,x[i],d[i]
		to=x[i]+d[i];
		if to>=0 and to <=20000 and d[i]!=0:
			if x[to]+d[to] == x[i]:
				isok=True
				break
			
if isok:
	print "YES"
else:
	print "NO"
