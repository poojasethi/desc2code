n = int(raw_input())
arr = []
for i in range(n):
    arr.append(raw_input().split())
answer = 'NO'
for i in range(n):
    for j in range(i + 1, n):
        if int(arr[i][0]) + int(arr[i][1]) == int(arr[j][0]) and int(arr[j][0]) + int(arr[j][1]) == int(arr[i][0]):
            answer = 'YES'
            break
    if answer == 'YES':
        break
print answer
