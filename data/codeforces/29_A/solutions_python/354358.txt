#!/usr/bin/python

n = int(raw_input())
x = []
for i in xrange(n):
    x.append(raw_input().split())
    for j in xrange(len(x[i])):
        x[i][j] = int(x[i][j])

for i in xrange(len(x)):
    y = x[i][0] + x[i][1]
    for j in xrange(len(x)):
        if( y == x[j][0] and x[j][0] + x[j][1] == x[i][0]):
            print "YES"
            exit( 0)

print "NO"
