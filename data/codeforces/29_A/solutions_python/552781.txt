#coding: utf-8

def main():
    N = input()
    inp = [map(int, raw_input().split()) for i in xrange(N)]
    if N == 1:
        print "NO"
        return 0
    for i in xrange(N-1):
        for j in xrange(i+1, N):
            cmp1 = sum(inp[i])
            cmp2 = sum(inp[j])
            if cmp1 == inp[j][0] and cmp2 == inp[i][0]:
                print "YES"
                return 0
    print "NO"
    return 0
main()
