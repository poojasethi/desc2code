n = int(raw_input())
x = [0]
d = [0]
for i in range(0 , n):
    a , b = map(int , raw_input().split())
    x.append(a)
    d.append(b)
ans = "NO"
for i in range(1 , n+1):
    for j in range(i+1 , n+1):
        if x[i]+d[i] == x[j]:
            if x[j]+d[j] == x[i]:
                ans = "YES"
print(ans)
