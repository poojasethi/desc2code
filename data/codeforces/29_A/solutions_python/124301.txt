#! /usr/bin/python

from sys import stdin
NO_ONE=-10**100

if __name__=='__main__':
	n=int(stdin.readline().strip())
	camels=[map(int,stdin.readline().split()) for case in xrange(n)]
	camels.sort()
	pos=[NO_ONE]*50000
	camels=[(camel+25000,spit) for (camel,spit) in camels]
	
	for (camel,spit) in camels:
		pos[camel]=spit
		
	found=False
	for (camel,spit) in camels:
		to_spit=camel+spit
		to_to_spit=to_spit+pos[to_spit]
		if to_to_spit==camel:
			found=True
			break
	
	if found:
		print "YES"
	else:
		print "NO"
