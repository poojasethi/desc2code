n = input()
a = []
d = []
for _ in xrange(n):
    x, y = map(int, raw_input().split(' '))
    a.append(x)
    d.append(y)
f = False
for i in xrange(n):
    for j in xrange(i + 1, n):
        if a[i] + d[i] == a[j] and a[i] == a[j] + d[j]:
            f = True
print ["NO", "YES"][f]
