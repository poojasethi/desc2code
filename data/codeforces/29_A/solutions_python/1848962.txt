import sys
import math

file = sys.stdin
cs_cnt = int(file.readline().rstrip())
p = [0]*(10**4*2 + 5)
for _ in range(cs_cnt):
    a, b = map(int, file.readline().rstrip().split())
    a += 10**4
    p[a] = b

for i in range(len(p)):
    if i != 0 and p[i] != 0:
        if 0 <= i + p[i] <= 10**4*2 and p[i + p[i]] != 0:
            if 0 <= p[i + p[i]] + i + p[i] <= 10**4*2 and p[i + p[i]] + i + p[i] == i:
                print "YES"
                break
else:
    print "NO"
