n = int(raw_input())
a = [[] * 2] * n
fl = 0
for i in range(n):
    if fl == 1:
        break
    a[i] = [int(x) for x in raw_input().split()]
    for j in range(i):
        if a[i][0] + a[i][1] == a[j][0] and a[j][0] + a[j][1] == a[i][0]:
            print "YES"
            fl = 1
            break
if fl == 0:
    print "NO"
