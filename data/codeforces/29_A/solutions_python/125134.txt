camels = int(raw_input())
MAX    = 2*10**4 + 5
array  = [-1]*MAX
did_spit = False
offset = 10**4
for camel in xrange(camels):
    index, d = [ int(x) for x in raw_input().split() ]
    array[index + offset] = index + d
    spit_index = index + d + offset
    if spit_index >= 0 and spit_index < MAX and array[spit_index] == index:
        did_spit = True
print "YES" if did_spit else "NO"
