#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()

s = dict()

def solve():
    for i in range(n):
        x, d = ia()
        s[x] = x + d
        if x + d in s and s[x+d] ==  x:
            print 'YES'
            return
    print 'NO'

solve()
