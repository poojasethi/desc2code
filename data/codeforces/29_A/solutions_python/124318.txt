import sys

input = sys.stdin
n = int(input.readline().strip())

animals = []
for i in xrange(n):
  animals.append(map(lambda x:int(x), input.readline().strip().split(' ')))


for i, (x, d) in enumerate(animals):
  a = x + d
  for xi, di in animals[i+1:]:
    if a == xi and xi+di == x:
      print 'YES'
      sys.exit(0)

print 'NO'
