from math import *
from Queue import *


n = int(raw_input())
x = []
d = dict()
for i in range(n):
    l = map(int, raw_input().split())
    x.append(l[0])
    d[l[0]] = l[1]
ret = False
for i in range(n):
    new = x[i] + d[x[i]]
    if (new in x) and (new + d[new] == x[i]):
        ret = True
        print('YES')
        break
if not ret:
    print('NO')
