#!/usr/bin/python
import sys
n = int(sys.stdin.readline())
L = []
for i in xrange(n):
    (x, d)   = map(int, sys.stdin.readline().split())
    L.append((x,d))
done = False
for i in xrange(n):
    for j in xrange(i+1,n):
        if L[i][0]+L[i][1] == L[j][0] and\
           L[j][0]+L[j][1] == L[i][0]:
            done = True
            break
if done: 
    print 'YES' 
else: 
    print 'NO'
