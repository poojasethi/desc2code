import sys

n = raw_input()
x = []
for i in range(int(n)):
    ps = raw_input().split(' ')
    x.append((int(ps[0]),int(ps[1])))

for i in range(int(n)):
    for j in range(i+1,int(n)):
        if x[i][0] == x[j][0]+x[j][1] and x[j][0] == x[i][0]+x[i][1]:
            print 'YES'
            sys.exit(0)
print 'NO'
