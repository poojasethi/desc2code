#include<iostream>
using namespace std;

int main(){
	int k;
	cin >> k;
	int timing[10]={0};	
	char di;
	for(int i=0;i<16;i++){
		cin >> di;
		if (di != 46){
//			cout << di-48;
			timing[di-48]++;
			if(timing[di-48] > 2*k){
				cout << "NO" << endl;
				return 0;
			}
		}
	}
	cout << "YES" << endl;
}
