
#include <iostream>

using namespace std;

int p[11];

int main()
{

	int k;
	char temp;

	cin>>k;
	for(int i = 0; i  < 16; i++)
	{

		cin>>temp;
		if(temp != '.')	++p[temp - '0'];

	}
	
	for(int i = 0; i <11; i++)
	{

		if(p[i] > 2 * k)
		{

			cout<<"NO"; 
			return 0;

		}

	}

	cout<<"YES";

	return 0;

}
