#include <iostream>
using namespace std;
int k,a[10];
char c[10][10];
int main(){
	cin >> k;
	for(int i=1; i<=4; i++)
		for(int j=1; j<=4; j++){
			cin >> c[i][j];
			if(isdigit(c[i][j])) a[ c[i][j]-48 ]++;
		}
	for(int i=1; i<=9; i++)
		if(k*2 < a[i]) {
			cout << "NO";
			return 0;
		}
	cout << "YES";
}
