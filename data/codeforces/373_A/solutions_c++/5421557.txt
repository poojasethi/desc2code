#include<iostream>
using namespace std;
int k,i,j,c[100];
string a[10];
int main()
{
    cin>>k;
    for(i=0;i<4;i++)cin>>a[i];
    for(i=0;i<4;i++)for(j=0;j<4;j++)
    {
        if(a[i][j]!='.')c[a[i][j]-'0']++;
    }
    for(i=0;i<12;i++){if(c[i]>2*k){cout<<"NO"<<endl;return 0;}}
    cout<<"YES"<<endl;
}
