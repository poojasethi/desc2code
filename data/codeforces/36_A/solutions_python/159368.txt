import sys

sys.stdin  = open ('input.txt',  'r')
sys.stdout = open ('output.txt', 'w')

def main():
	n = int (sys.stdin.readline())
	a = sys.stdin.readline().strip()
	i1 = a.index ('1', 0)
	i2 = a.index ('1', i1 + 1)
	delta = i2 - i1
	i1 = i2
	while True:
		try:
			i2 = a.index ('1', i1 + 1)
		except ValueError:
			print 'YES'
			break			
		else:
			if i2 - i1 != delta:
				print 'NO'
				break
			else:
				i1 = i2

main()
