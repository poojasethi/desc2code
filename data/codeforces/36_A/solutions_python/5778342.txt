        flg+=1
    elif flg==1 and l[i]!='1':
        cnt+=1
    elif l[i]=='1' and flg==1:
        flg+=1
        god=cnt
        cnt=0
    elif flg==2 and l[i]=='1' and cnt==god:
        cnt=0
    elif flg==2 and l[i]=='0':
        cnt+=1
    elif flg==2 and l[i]=='1' and cnt!=god:
        print 'NO'
