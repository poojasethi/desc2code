inp = open('input.txt', 'r')
out = open('output.txt', 'w')

n = int(inp.readline())
sir = inp.readline()
x = []
for i in range(0, n):
    if sir[i] == '1':
        x.append(i)
for i in range(2, len(x) - 1):
    if x[i] - x[i - 1] != x[1] - x[0]:
        print >> out, "NO"
        exit(0)
print >> out, "YES"
