fin = open('input.txt', 'r')
fout = open('output.txt', 'w')

n = int(fin.readline())
s = fin.readline()
y = z = -1
f = True
i = 0
for x in s:
	i += 1
	if x == '1':
		if y > 0:
			if z < 0:
				z = i-y
			elif z != i-y:
				f = False
				break
		y = i

if f:
	fout.write('YES\n')
else:
	fout.write('NO\n')
