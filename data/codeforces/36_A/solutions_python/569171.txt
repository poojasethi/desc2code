input = open('input.txt','r')
output = open('output.txt','w')

def solve(s):
    first = s.find('1',0)
    if first == -1:
        return 'YES'
    second = s.find('1',first+1)
    if second == -1:
        return 'YES'
    
    d = second - first
    
    p = second
    while p != -1:
        i = s.find('1',p+1)
        if i!=-1 and i-p != d:
            return 'NO'
        p = i
    
    return 'YES'

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]
