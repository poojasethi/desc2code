fin = open("input.txt")
fout = open("output.txt", "w")

n = fin.readline()

s = fin.readline()

f = 0

a = s.find('1')
s = s[a + 1: ]
n = s.find('1')

while not a == -1:
    a = s.find('1')
    s = s[a + 1: ]
    if not a == n and not a == -1:
        f = 1

if f == 0:
    fout.write("YES")
else:
    fout.write("NO")
