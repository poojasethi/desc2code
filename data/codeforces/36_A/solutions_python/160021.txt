import sys
import string

sys.stdin = open("input.txt","r")
sys.stdout = open("output.txt","w+")


n = input()
s = raw_input()
r = []
for i in range(n):
	if s[i] == '1':
		r.append(i)
d = r[1] - r[0]
L = len(r)
for i in range(2,L,1):
	if r[i] - r[i-1] != d:
		break
if i == L - 1:
	print 'YES'
else:
	print 'NO'
