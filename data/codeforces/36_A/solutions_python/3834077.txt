f = open("input.txt")
lines = f.readlines()
f.close()
n = int(lines[0])
a = []
b = set()
for i in range(n):
    number = int(lines[1][i])
    if number == 1:
        a.append(i)
for i in range(1, len(a)):
    b.add(a[i] - a[i -1])

ff = open("output.txt", "w")
ff.write("YES") if len(b) == 1 else ff.write("NO")
ff.close()
