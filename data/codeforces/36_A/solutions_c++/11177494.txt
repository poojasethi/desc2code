#include<cstdio>
using namespace std;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,b[200];
	scanf("%d",&n);
	char a;
	int j=0,ok=1;
	for(int i=1;i<=n;i++){
		scanf("%c",&a);
		if(a=='1') b[j++]=i; 
	}
	int tmp=b[1]-b[0]; 
	for(int i=1;i<j-1;i++){
		if(tmp!=b[i+1]-b[i]){
			ok=0;break;
		}
	}
	if(ok) printf("YES\n");
	else printf("NO\n");
}
