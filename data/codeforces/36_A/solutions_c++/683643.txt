#include <iostream>
#include <fstream>
using namespace std;
ifstream in("input.txt");
ofstream out("output.txt");
int A[1000];
int solve()
{
	int i=0,l,k=0;
	char a;
	in >> l;
	for(; i<l; i++)
	{
		in >> a;
		if(a-'0')
			A[k++]=i;
	}
	for(i=1; i<k-1; i++)
		if(A[i]-A[i-1]!=A[i+1]-A[i])
			return 0;
	return 1;
}
int main()
{
	solve()? out << "YES" : out << "NO";
	return 0;
}
