#include <fstream>
#include <string>
using namespace std;
int i,n,k=-1,k1,d=-1;
ifstream cin("input.txt");
ofstream cout("output.txt");
int main()
{
    string s;
    cin>>n>>s;
    k=s.find("1");
    s[k]='0';
    k1=s.find("1"); 
    s[k1]='0';
    d=k1-k;
    for(i=0;i<n;i++)
     if(s[i]=='1')
     if(i-k1==d)k1=i;else {cout<<"NO\n";return 0;}
     cout<<"YES\n";

    return 0; 
}
