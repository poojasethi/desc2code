#include<iostream>
#include<fstream>
using namespace std;
int main(){
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int l;
	cin>>l;
	char arr[110];
	cin>>arr;
	int a[110];
	int s=0;
	for(int i=0;i<l;i++)
		if(arr[i]=='1'){
			a[s]=i;
			s++;
		}
	for(int i=1;i<s-1;i++)
		if(a[i]-a[i-1]!=a[i+1]-a[i]){
			cout<<"NO"<<endl;
			return 0;
		}
	cout<<"YES"<<endl;
}
