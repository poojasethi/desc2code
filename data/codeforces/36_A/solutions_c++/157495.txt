#include <iostream>
#include <fstream>
using namespace std;
int main()
{
	ifstream fin("input.txt");
	ofstream fout("output.txt");
	int i, j, num, re[101]={0};
	char q[101]={0};
	fin>>num;
	fin>>q;
	j = 0;
	for (i = 0; i < num; i++)
		if (q[i]=='1')
			re[j++]=i;
	for (i = 1; i < j-1; i++)
		if (2*re[i]!=re[i-1]+re[i+1])
			break;
	if (i == j-1)
		fout<<"YES\n";
	else
		fout<<"NO\n";
}
