#include <stdio.h>
char s[105];
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,i,a[105],c,m=0;
    scanf("%d",&n);
    scanf("%s",s);
    for(i=0;i<n;i++)if(s[i]=='1')a[m++]=i;
    c=a[1]-a[0];
    for(i=2;i<m;i++)
     if(a[i]-a[i-1]!=c)
        {
            puts("NO");
            return 0;
        }
    puts("YES");
    return 0;
}
