#include <cstdio>

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w+", stdout);
	int n;
	char s[105];
	scanf("%d%s", &n, s);
	int dist = -1, pre = -1;
	bool res = true;
	for (int i = 0; i < n && res; i++) {
		if (s[i] == '1') {
			if (pre == -1) {
				pre = i;
			} else if (dist == -1) {
				dist = i - pre;
				pre = i;
			} else {
				res = (dist == i - pre);
				pre = i;
			}
		}
	}
	puts(res ? "YES" : "NO");
	return 0;
}
