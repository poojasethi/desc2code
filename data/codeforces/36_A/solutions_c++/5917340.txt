#include <stdio.h>

int main(void) {
	int n;
	char s[105];
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d", &n);
	scanf("%s", s);
	int last=-1, len=-1, flag=0;
	for(int i=0;i<n;i++) {
		if(s[i]=='1') {
			if(last==-1) last = i;
			else if(len==-1) len = i-last;
			else if(len != i-last) flag=1;
			last = i;
		}
	}
	if(flag) puts("NO");
	else puts("YES");
	return 0;
}
