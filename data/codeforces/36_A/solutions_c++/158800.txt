#include<cstdio>
int main()
{
    freopen("input.txt","r",stdin); 
    freopen("output.txt","w",stdout);
    int n,num[500],laber=0,c;char s[500];
    scanf("%d\n%s",&n,s);
    for(int i=0;s[i];i++)
        if(s[i]=='1')
            num[laber++]=i;
    c=num[1]-num[0];
    for(int i=2;i<laber;i++)
        if(num[i]-num[i-1]!=c)
        {
            puts("NO");return 0;
        }
    puts("YES");return 0;
}
