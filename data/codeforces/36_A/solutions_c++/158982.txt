#include <string>
#include <fstream>
#include <vector>
using namespace std;

int main()
{
	ifstream fin("input.txt");
	ofstream fout("output.txt");
	int n;
	fin >> n;
	string s;
	fin >> s;

	vector <int> one;
	for (int i = 0; i < n; i++) if (s[i]=='1') one.push_back(i);
	int d = one[1]-one[0];
	bool yes = true;
	for (int i = 2; i < one.size(); i++) if ((one[i]-one[i-1]) != d) { yes = false; break; }
	if (yes) fout << "YES\n";
	else fout << "NO\n";
	return 0;
}
