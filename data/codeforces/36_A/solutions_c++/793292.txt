#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[105];

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	int n,ok=1;
	scanf("%d%s",&n,s);
	int pre = -1, mx = -1;
	for(int i = 0; i < n; i++)
	{
		if(s[i]=='1')
		{
			if(mx==-1)
			{
				if(s[pre]=='1')
					mx = i-pre;
			}
			else if(mx!=i-pre)ok=0;
			pre = i;
		}
	}
	puts(ok?"YES":"NO");

	return 0;
}
