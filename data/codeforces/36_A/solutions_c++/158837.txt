#include<stdio.h>
int main()
{
	int n,c=0,dif,c1[1000],i,j=0;
	char ch;
        FILE *fr,*fw;
        fr=fopen("input.txt","r");
        fw=fopen("output.txt","w");
	fscanf(fr,"%d",&n);
	i=0;j=0;
	while(i<n)
	{
		fscanf(fr,"%c",&ch);
		if(ch=='1')c1[j++]=i;
		i++;
	}
	dif=c1[1]-c1[0];
	for(i=2;i<j;i++)
	{
		if((c1[i]-c1[i-1])!=dif)
		{
			fprintf(fw,"NO\n");
			return 0;
		
		}
	}
	fprintf(fw,"YES\n");
	return 0;
}
