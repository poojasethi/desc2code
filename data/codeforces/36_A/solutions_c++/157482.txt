#include <fstream>
using namespace std;

ifstream cin("input.txt");
ofstream cout("output.txt");

main()
{
	int n,i,j,razn=-1;
	char a[200];
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n-1;i++)
		if(a[i]=='1')
			for(j=i+1;j<n;j++)
				if(a[j]=='1')
				{
					if(razn!=-1 && razn!=j-i)
					{
						cout<<"NO"<<endl;
						return 0;
					}
					razn=j-i;
					break;

				}
	cout<<"YES"<<endl;
	return 0;
}
