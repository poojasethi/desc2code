#include<cstdio>

int main(){
freopen("input.txt", "r", stdin);
freopen("output.txt", "w", stdout);

	int n;
	scanf("%d",&n);
	char s[110];
	scanf("%s",s);
	int prim=1;
	int val=-1;
	int erro=0;
	int cnt=0;
	for(int i=0;s[i];i++){
		
		if(s[i]=='1'&&prim==0){
			if(val==-1)
				val=cnt;
			else if(cnt!=val){
				erro=1;
				break;
			}
		}
		if(s[i]=='0')cnt++;
		if(s[i]=='1'){prim=0;cnt=0;}
	}
	if(erro)
		printf("NO\n");
	else printf("YES\n");
	return 0;
}
