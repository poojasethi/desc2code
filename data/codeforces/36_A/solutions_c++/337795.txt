#include <iostream>
#include <cstdio>
#include <cassert>
using namespace std;

#define rep(i,n) for (int i = 0; i < (n); i++)

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	int n;
	cin >> n;

	int a[100], an = 0;
	rep(i,n) {
		char c;
		cin >> c;
		if (c == '1') {
			a[an++] = i;
		}
	}

	rep(i,an-1) {
		if (a[i+1]-a[i] != a[1]-a[0]) {
			cout << "NO" << endl;
			return 0;
		}
	}
	cout << "YES" << endl;
	return 0;
}
