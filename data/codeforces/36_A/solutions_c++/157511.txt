



#include<stdio.h>
int N[1000],n,p;
char c;
int main()
{
FILE * fi = fopen("input.txt","r");
FILE * fo = fopen("output.txt","w");
 fscanf(fi,"%d",&p);
 fscanf(fi,"%c",&c);
 for(int i = 0 ; i < p ; i++)
 {
  fscanf(fi,"%c",&c);
  if(c == '1')
   N[n++]=i;
 }
 bool is = 1;
 int d = N[1] - N[0];
 for(int i =1 ;is && i < n ; i++)
  is = (d == N[i]-N[i-1]);
 if(is)
  fprintf(fo,"YES\n");
 else
  fprintf(fo,"NO\n");

  return 0;
}
