#include <stdio.h>

using namespace std;
int main()
{
    int n,x[120],k = 0;
    char inp[120];
    freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
    scanf("%d\n%s",&n,inp);
    for(int i = 0;i < n;i++)
    if(inp[i] == '1'){x[k] = i;k++;}
    for(int i = 1;i < k - 1;i++)
    if(x[i] - x[i - 1] != x[i + 1] - x[i]){printf("NO");return 0;}
    printf("YES");
}
