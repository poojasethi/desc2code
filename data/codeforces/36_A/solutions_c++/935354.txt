#include<iostream>
#include<fstream>
#include<vector>
using namespace std;
int main(){
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int n,x=-1;
	cin>>n;
	string a;
	cin>>a;
	vector<int> v;
	for(int i=0;i<n;i++)
		if(a[i]=='1')
		v.push_back(i);
	if(v.size()>2){
		x=v[1]-v[0];
		for(int i=1;i<v.size();i++)
			if(v[i]-v[i-1]!=x){
				//cout<<i<<endl;
				cout<<"NO"<<endl;
				return 0;
			}
	}
	cout<<"YES"<<endl;

}
