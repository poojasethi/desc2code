#include<stdio.h>
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt ","w",stdout);
	char nRec;
	int n,i,j,flag[105];
	while(scanf("%d",&n) != EOF)
	{
		getchar();
		for(i=0,j=0;i<n;i++)
		{
			scanf("%c",&nRec);
			if(nRec-'0'==1)
				flag[j++]=i;
		}
		for(i=0;i<j-2;i++)
		{
			if(flag[i+1]-flag[i]!=flag[i+2]-flag[i+1])
				break;
		}
		if(i>=j-2)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
