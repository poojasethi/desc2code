#include <iostream>
#include <fstream>
using namespace std;
ifstream fin("input.txt");  ofstream fout("output.txt");
int n,a[1100],ja,x;
string s;
bool check(){
for(int i=2,x=a[1]-a[0];i<ja;i++)  if(a[i]-a[i-1]!=x) return 0;
return 1;}
int main(){
    fin>>n>>s;
    for(int i=0;s[i];i++) if(s[i]=='1') a[ja++]=i;
    if(check()) fout<<"YES";
    else fout<<"NO";
}
