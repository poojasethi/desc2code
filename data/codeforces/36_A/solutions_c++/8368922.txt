#include<bits/stdc++.h>
#define N 1005
using namespace std;

int n,t,qw,f,answ;
int S[N];


int main(){
	freopen("input.txt","r",stdin);
  	freopen("output.txt","w",stdout);
	cin>>n;
	string st;cin>>st;
	for(int i=0;i<n;i++){
		S[i]=st[i]-'0';
	}
	answ=-1;
	for(int i=0;i<n;i++){
		if(S[i]==1 && t>=2 && i-qw !=answ) f++;
		if(S[i]){
			t++;
			if(t==1) qw=i;
			if(t==2) answ=i-qw;
			qw=i;
		}
	}
	if(!f) cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
	return 0;
}
