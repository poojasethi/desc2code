#include <fstream>
using namespace std;
int i, k;
char n;
int main() {
	ifstream I("input.txt");
	ofstream O("output.txt");
	int A[100];

	for (I >> n; I >> n; i++)
		if (n == '1') 
			A[k++] = i;

	int t = A[1] - A[0];

	for (i = 2 ; i < k ; i++) {
		if (A[i] - A[i - 1] != t) {
			O << "NO";
			return 0;
		}
	}
	if (k == 0) O << "NO";
	else 
	O << "YES";
}
