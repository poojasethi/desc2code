#include<stdio.h>
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int ar,in[105],i,n,j=0;
	char a[105];

	scanf("%d %s",&n,a);
	for(i=0;i<n;i++)
	{
		if(a[i]=='1')
			in[j++]=i;
	}
	for(i=1;i<j;i++)
	{
		if(i!=1 && in[i]-in[i-1]!=ar)
		{
			printf("NO\n"); return 0;
		}
		ar=in[i]-in[i-1];

	}
	printf("YES\n");
	return 0;
}
