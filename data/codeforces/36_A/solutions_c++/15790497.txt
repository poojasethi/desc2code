#include <bits/stdc++.h>

using namespace std;

string s;
vector<int>r;
set <int> g;
int main(){
	ifstream fin("input.txt");
	ofstream fout("output.txt");
	fin >> s >> s;
	for(int i=0;i<s.size();i++)
		if (s[i]=='1')
			r.push_back(i);
	for (int i=1;i<r.size();i++)
		g.insert(r[i]-r[i-1]);
	if (r.size()==0)
		return fout << "NO",0;
	if (g.size()<2)
		fout << "YES";
	else
		fout << "NO";
}
