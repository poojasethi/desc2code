#include <fstream>
#include <string>
using namespace std;

int main(void)
{
	ifstream ifs("input.txt");
	ofstream ofs("output.txt");

	int n, a[100], d, count=0;
	string str;

	ifs >> n;
	ifs >> str;
	for (int i=0; i<n; ++i)
		if (str[i] == '1')	a[count++]=i;
	d=a[1]-a[0];
	bool flg=true;
	for (int i=2; i<count; ++i)
		if ( d != a[i]-a[i-1] )	flg=false;
	ofs << (flg?	"YES" : "NO") << endl;
	return 0;
}
