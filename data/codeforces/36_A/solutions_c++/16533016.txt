#include <iostream>
#include <cstdio>
#include <set>
using namespace std;
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n, i, cnt = 0, f = 0;
	char a[150];
	set <int> zero;
	cin >> n;
	for (i = 0; i < n; i++){
		cin >> a[i];
		if (a[i] == '1') {
			if (!f) f = 1;
			else {
				zero.insert(cnt);
				cnt = 0;
			}
		}
		else if (f) cnt++;
	}
	if (zero.size() == 1) cout << "YES";
	else cout << "NO";
}
