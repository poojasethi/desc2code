#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
char str[1001];
int a[1001],i,j,n;
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n;
	cin>>str;
	i=0;
	for(j=0;j<n;j++){
		if(str[j]=='1')
			a[i++]=j+1;
	}
	for(j=2;j<i;j++){
		if((a[j]-a[j-1])!=(a[j-1]-a[j-2]))
			break;
	}
	if(i!=j) cout<<"NO"<<endl;
	else cout<<"YES"<<endl;
	return 0;
}
