results = ['Daenerys', 'Stannis']

def result(n, k, odd, even):
    if k == 0 :
        return results[odd & 1]
    if odd <= k/2 :
        return results[0]
    if even <= k/2 :
        return results[(n-k) & 1]
    return results[k & 1]

if __name__ == '__main__':
    n, k = map(int, raw_input().split())
    odd = sum(int(x) & 1 for x in raw_input().split())
    print result(n, n-k, odd, n-odd)
