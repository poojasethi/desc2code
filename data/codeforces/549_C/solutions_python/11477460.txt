n,k = map(int,raw_input().split())

store = map(int,raw_input().split())

"Daenerys"  "Stannis"
"even"        "odd" "first"

diff = n-k

even = 0
odd = 0

for i in store:
    if i%2 == 0:
        even +=1
    else:
        odd += 1

x = diff # x moves

if diff % 2 == 1: # stan moves first and last
    if even > (x-1)/2 and odd > (x-1)/2 :
        print "Stannis" # free move for stannis
    elif even > (x-1)/2 : # not enough odds
        # Dae will take all odds and sum is even
        print "Daenerys"
    elif odd > (x-1)/2:
        # Dae has to take all evens or stan has choice
        odds_taken = (x-even)
        if (odd-odds_taken) % 2 == 0:
            print "Daenerys"
        else:
            print "Stannis"
    else: # nothing is left
        print "Error"


else:# stan moves first and Daenerys last
    if diff == 0:
        if odd %2 == 1:
            print "Stannis"
        else:
            print "Daenerys"
    elif even > x/2 and odd > x/2 :
        print "Daenerys" # free move at last
    elif even > x/2 : # not enough odds
        # stan has to take all odds or de has choice
        print "Daenerys"
    elif odd > x/2:
        # stan has to take all evens or de has choice
        odds_taken = (x-even)
        if (odd-odds_taken) % 2 == 0:
            print "Daenerys"
        else:
            print "Stannis"
    else: # nothing is left
        print "Error"
