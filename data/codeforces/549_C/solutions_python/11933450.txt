n , k  = map(int, raw_input().split(' '))
population = map(lambda x: int(x) % 2, raw_input().split(' '))
num_zero = population.count(0)
num_one = population.count(1)
Astep = (n - k) // 2 + (n - k) % 2
Bstep = (n - k) // 2

if n == k:
    print "Daenerys" if num_one % 2 == 0 else "Stannis"
elif  Bstep >= num_one:
    print "Daenerys"
elif Bstep >= num_zero:
    print "Daenerys" if (num_zero + num_one - (n - k)) % 2 == 0 else "Stannis"
else:
    print "Stannis" if Astep > Bstep else "Daenerys"
