I=lambda:map(int,raw_input().split())
n,k=I()
t = n-k
a=map(1 .__and__, I())
odd=a.count(1)
eve=n-odd
last = (t&1)^1
nl = 1^last
nlgoal = 1^nl

lastlose = 0
nlt = t/2

#print last, nl, nlgoal

if nlt >= odd and nlgoal == 0: lastlose = 1
elif nlt >= eve and (odd - (t-eve)   )%2 == nlgoal: lastlose = 1

#print lastlose
winner = last ^ lastlose
if t==0: winner = (odd&1)^1
print ['Stannis','Daenerys'][winner]
