#include<bits/stdc++.h>
using namespace std;
int n,k,s_even,s_odd;
int main ()
{
	int i,x;
	cin>>n>>k;
	for(i=0; i<n; i++)
	{
		scanf("%d",&x);
      (x%2)?s_odd++:s_even++;
	}
	if(n==k)
	{
		puts((s_odd%2)?"Stannis":"Daenerys");
		return 0;
	}
	if((n-k)%2)
		if(k%2)
			puts(((n-k)/2>=s_odd)?"Daenerys":"Stannis");
		else
			puts(((n-k)/2>=s_odd || (n-k)/2>=s_even)?"Daenerys":"Stannis");
	else
		if(k%2)
			puts(((n-k+1)/2>=s_even)?"Stannis":"Daenerys");
		else
			puts("Daenerys");
}
