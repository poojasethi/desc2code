#include <cstdio>

using namespace std;

int n, k, x, a, b;

int work() {
	if (n==k) return a&1;
	if ((n-k)&1) {
		if ((n-k)/2>=a) return 0;
		if ((n-k)/2>=b && !(k&1)) return 0;
		return 1;
	}
	 else {
		if ((n-k)/2>=b && (k&1)) return 1;
		return 0;
	}
}

int main() {
	scanf("%d%d",&n,&k);
	for(int i = 1; i <= n; i++) {
		scanf("%d",&x);
		if (x&1) ++a;
		else ++b;
	}

	if (work()) printf("Stannis\n");
	else printf("Daenerys\n");

	return 0;
}
