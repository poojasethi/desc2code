#include<stdio.h>
int a[200005];
int main()
{
	int n,k,i,num1,num2;
	scanf("%d%d",&n,&k);
	num1=num2=0;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]%2)num1++;
		else num2++;
	}
	if(num1+num2==k)
	{
		if((num1+num2*2)%2)
		printf("Stannis\n");
		else 
		printf("Daenerys\n");
	}
	else if(num1<=(n-k)/2)
		printf("Daenerys\n");
	else if(num2<=(n-k)/2)
	{
		if(k%2)
		printf("Stannis\n");
		else 
		printf("Daenerys\n");
	}
	else if((n-k)%2)
	printf("Stannis\n");
 	else if((n-k)%2==0)
	printf("Daenerys\n");
	return 0;
}
