#include<cstdio>
#include<cstdlib>
using namespace std;
int i,n,K,n1,n2,x;
void go1() {printf("Stannis\n");exit(0);}
void go2() {printf("Daenerys\n");exit(0);}
int main()
{
	//freopen("1.in","r",stdin);freopen("1.out","w",stdout);
	scanf("%d%d",&n,&K);
	for (i=1;i<=n;i++)
	{
		scanf("%d",&x);
		if (x&1) n1++;else n2++;
	}
	if (n==K)
	{
		if (n1&1) go1();
		else go2();
	}
	if ((n-K)&1)
	{
		if ((n-K)/2>=n1) go2();
		if ((n-K)/2>=n2&&!(K&1)) go2();
		go1();
	}
	else
	{
		if ((n-K)/2>=n2&&(K&1)) go1();
		go2();
	}
	return 0;
}
