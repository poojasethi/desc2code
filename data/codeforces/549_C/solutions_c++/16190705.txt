#include<cstdio>
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	int d=(n-k-1)/2,s=n-k-1-d;
	int j=0,o=0;
	int tmp;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&tmp);
		if(tmp%2)
		j++;
		else
		o++;
	}
	if(n==k)
	{
		if(j%2)
		printf("Stannis\n");
		else
		printf("Daenerys\n");
	}
	else
	{
		if(d>=j)
		printf("Daenerys\n");
		else if(d>=o)
		{
			if(k%2)
			printf("Stannis\n");
			else
			printf("Daenerys\n");
		}
		else if(s>=o&&k%2)
		printf("Stannis\n");
		else 
		{
			if((n-k)%2)
			printf("Stannis\n");
			else
			printf("Daenerys\n");
		}
	} 
}
