#include <iostream>
using namespace std;
int n,k,e,o,t;
bool s;
int main(){
    ios_base::sync_with_stdio(0);
    cin>>n>>k;
    for(int i=0;i<n;i++){
        cin>>t;
        if(t%2)o++;
        else e++;
    }
    if(n==k)s=o%2;
    else if(k%2){
        if(((n-k+1)/2>=e&&e<=o)||((n-k)/2<o&&(n-k)%2))s=true;
    }
    else{
        if((n-k)/2<min(o,e)&&(n-k)%2)s=true;
    }
    cout<<(s?"Stannis":"Daenerys");
}
