#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,k;
	cin>>n>>k;
	int odd=0,even=0;
	int i,j;
	for(i=0;i<n;i++)
	{
		cin>>j;
        odd += j & 1;
        even += !( j & 1 );
	}
	int flag = odd & 1;
	int t = n - k;
	if(t>0)
	{
		if(t & 1)
		{
			flag = 0;
			if( min(odd,even) > t/2 || (odd > even && k & 1) ) flag = 1;
		}
		else
		{
			flag = 1;
			if( min(odd,even) > (t+1)/2 || (odd > even && !(k & 1)) || even > odd) flag = 0;
		}
	}
	flag ? cout<<"Stannis" : cout<<"Daenerys";
	cout<<endl;
	return 0;
}
