#include<stdio.h>
int main()
{
	int n,k,j=0,o=0,i,num,ds,ss,f;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++){
		scanf("%d",&num);
		if(num%2==0)o++;
		else j++;}
		ds=(n-k)/2,ss=(n-k+1)/2,f=j%2;
	if(k==n){
		if(j%2==0)puts("Daenerys");
		else puts("Stannis");	}
		else {
		if(ds>=j)puts("Daenerys");
		else if(ds>=o&&k%2==0)puts("Daenerys");
		else if(ds<ss)puts("Stannis");
		else {
			if(ss>=o&&k%2==1)puts("Stannis");
			else puts("Daenerys");
		}	
	}	
	return 0;
}
