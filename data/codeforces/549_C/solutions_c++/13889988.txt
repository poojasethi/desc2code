#include<stdio.h>
int n,k,i,odd,even,s,d,v,t,sum;
int main(){
	scanf("%d%d",&n,&k);
	for(i=0;i<n;i++){
		scanf("%d",&t);
		if(t%2==0)even++;
		else odd++;
		sum+=t;
	}
	d=(n-k)/2;
	s=n-k-d;
	if(s>d){
		if(d>=odd)v=1;
		else if(k%2==0&&even<=d)v=1;
		else v=0;
	}
	else{
		if(even<=s&&k%2==1)v=0;
		else v=1;
	}
	if(k==n)if(sum%2==0)v=1;else v=0;
	printf("%s",v?"Daenerys":"Stannis");
}
