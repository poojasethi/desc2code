#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[200000],o,e;
int main()
{std::ios::sync_with_stdio(false);
int n,k;
cin>>n>>k;
for(int i=0;i<n;i++)
{cin>>a[i]; 
if(a[i]%2)o++;else e++;
}
if(n==k)
{if(o%2)cout<<"Stannis";
else cout<<"Daenerys";return 0;}
if(k==0){cout<<"Daenerys";return 0;}
if(e<=(n-k+1)/2&&k%2){cout<<"Stannis";return 0;}
if(o<=(n-k)/2){cout<<"Daenerys";return 0;}
if(e<=(n-k)/2&&k%2==0){cout<<"Daenerys";return 0;}
if((n-k)%2)cout<<"Stannis";else cout<<"Daenerys";
return 0;
}
