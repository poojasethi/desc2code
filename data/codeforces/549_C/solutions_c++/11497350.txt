#include<bits/stdc++.h> 
using namespace std;
string SD[]={"Daenerys","Stannis"};
int n,k,e,o,b,x=-1,s;
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++)scanf("%d",&b),(!(b%2) ? e++ : o++);
	if(n==k)x=o;	
	n-=k,s=(o-(n-e));
	if(o<=n/2)x=0;
	if((n%2==0&&n/2>=e&&s%2)||(n%2&&n/2>=e&&s%2==0))n++;
	cout<<((x!=-1) ? SD[x%2] : SD[n%2])<<endl;
}
