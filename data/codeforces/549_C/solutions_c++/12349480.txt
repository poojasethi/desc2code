#include <iostream>
#include <cstring>
using namespace std;
string abc[2]={"Daenerys","Stannis"};
int n,k,e,m,b,x=-1,s;
int main()
{
	 cin>>n>>k;
	 for (int i=0;i<n;i++)
     {
          cin>>b;
          if (b%2==0) e++; else m++;
     }
	 if (n==k) x=m;	
	 n-=k;s=(m-(n-e));
	 if (m<=n/2) x=0;
	 if ((n%2==0 && n/2>=e && s%2==1) || (n%2==1 && n/2>=e && s%2==0)) n++;
	 if (x!=-1) cout<<abc[x%2]; else cout<<abc[n%2];
	 return 0;
}
