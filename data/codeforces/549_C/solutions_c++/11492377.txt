#include<bits/stdc++.h>
using namespace std;

int main()
{
	int ans=0,i,n,k,a,b,c;
	cin>>n>>k;
	b=0;
	for(i=0;i<n;i++)
	{
		cin>>a;
		if(a%2)
			b++;
	}
	c=n-k;
	if((c%2)&&(k%2))
	{
		if(b>(c/2))
			ans=1;
	}
	else if((c%2)&&!(k%2))
	{
		if(b>(c/2)&&(b<(n-c/2)))
			ans=1;
	}
	else if(!(c%2)&&(k%2))
	{
		if((k==n)&&(b%2))
			ans=1;
		else if(b>=(n-c/2))
			ans=1;
	}
	else
	{
		if((n==k)&&(b%2))
			ans=1;
	}
	if(!ans)
		cout<<"Daenerys"<<endl;
	else
		cout<<"Stannis"<<endl;
	return 0;
}
