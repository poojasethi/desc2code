#include <iostream>

using namespace std;

int n, k, a[3], ans;

int main()
{
    cin >> n >> k;
    for (int i=1; i<=n; i++)
    {
        int x;
        cin >> x;
        a[x%2] ++;
    }
    int len = (n - k) / 2;
    if (n == k)
    {
        ans = (a[1] % 2) ? 1 : 2;
    }
    else
    {
        if (a[1] <= len) ans = 0;
        else if (a[0] <= len) ans = (k % 2) ? 1 : 2;
             else ans = ((n - k) % 2) ? 1 : 2;
    }
    if (ans == 1) cout << "Stannis";
    else cout << "Daenerys";
    return 0;
}
