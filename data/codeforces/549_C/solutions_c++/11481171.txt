#include <bits/stdc++.h>
using namespace std;

bool gao() {
    int n, k, p, t[2] = {}, c[2] = {};
    scanf("%d%d", &n, &k);
    for (int x, i = 0; i < n; ++i) {
        scanf("%d", &x);
        ++c[x & 1];
    }
    p = n - k;
    t[1] = p / 2;
    t[0] = p - t[1];
    if (p & 1) {
        if (k & 1) return t[1] >= c[1];
        else return t[1] >= min(c[0], c[1]);
    } else if (!p) {
        return !(c[1] & 1);
    } else {
        if (k & 1) return t[0] < c[0];
        else return 1;
    }
}
int main() {
    puts(gao() ? "Daenerys" : "Stannis");
}
