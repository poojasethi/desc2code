#include <cstdio>

int n,m,x,y,z;
int main() {
	scanf("%d%d",&n,&m);
	x = y = 0;
	for (int i = 1;i <= n;i++) {
		scanf("%d",&z);
		if (z%2) 
			x++;
		else
			y++;
	}
	if (n == m) {
		if (x%2)
			printf("Stannis");
		else
			printf("Daenerys");
	} else {
		if ((n-m)/2 >= x)
			printf("Daenerys");
		else
		if ((n-m)/2 < y) {
			if ((n-m)%2)
				printf("Stannis");
			else
				printf("Daenerys");
		} else {
			if (m%2)
				printf("Stannis");
			else
				printf("Daenerys");
		}
	}
	return 0;
}
