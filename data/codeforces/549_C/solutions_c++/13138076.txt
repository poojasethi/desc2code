#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	int odd=0,even=0;
	for(int i=1;i<=n;++i)
	{
		int x;
		scanf("%d",&x);
		if(x%2)++odd;
		else   ++even;
	}
	int t=n-m;
	int ans;
    
	if(!t)ans=odd%2;
	else
	{
		if(t/2<min(odd,even)) ans=t%2;
		else
		{
			if(odd>even)
			{
				if( ( odd-(t-even) ) % 2 )ans=1;
				else ans=0;
			}
			else ans=0;
		}
    }
    printf(ans ? "Stannis" : "Daenerys");
}
