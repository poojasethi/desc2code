#include<stdio.h>

int a,c1,c0,m,n,i;

int main(){
    scanf("%d %d",&m,&n);
    for(i=0;i<m;i++){
        scanf("%d",&a);
        if(a%2==1)c1++;
        else c0++;
    }
    if(m==n){
        if(c1%2==0)printf("Daenerys\n");
        else printf("Stannis\n");
    }
    else if((m-n)%2==0){
        if((m-n)/2>=c0&&n%2==1)printf("Stannis\n");
        else printf("Daenerys\n");
    }
    else{
        if((m-n)/2>=c1)printf("Daenerys\n");
        else if((m-n)/2>=c0&&n%2==0)printf("Daenerys\n");
        else printf("Stannis\n");
    }
}
