#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int n,k,v,sum,x;
inline void p1()
{
	printf("Daenerys\n");
}
inline void p2()
{
	printf("Stannis\n");
}
int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;++i)
	{
		scanf("%d",&x);
		sum+=x&1;
	}
	v=n-k>>1;
	if (n==k) {
		if (sum&1) p2();
		else p1();
	}
	else if (v<sum&&v<n-sum) {
		if ((n-k&1)) p2();
		else p1();
	}
	else if ((v)>=sum) p1();
	else {
		if (!(k&1)&&(v)>=n-sum) p1();
		else if ((k&1)&&(n-k+1>>1)>=n-sum) p2();
		else p1();
	}
}
