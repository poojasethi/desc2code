#include <cstdio>

#define rep(i, a, b) for (int i = a; i <= b; i++)

bool win(int a, int b, int k) {
    if (a + b == k) return a & 1;
    if (a - b >= k) return k & 1;
    if (b - a >= k) return 0;
    return (a + b + k) & 1;
}

int main() {
    int n, k, a, odd = 0, even = 0;
    scanf("%d %d", &n, &k);
    rep(i, 0, n - 1) scanf("%d", &a), a & 1 ? odd++ : even++;
    printf("%s", win(odd, even, k) ? "Stannis" : "Daenerys");
    return 0;
}
