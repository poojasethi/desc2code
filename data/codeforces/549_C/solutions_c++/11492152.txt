#include<stdio.h>
int main()
{
	int n,k,p,ec=0,oc=0,x;
	scanf("%d %d",&n,&k);
	p=n;
	while(p--)
	{
		scanf("%d",&x);
		if(x%2)oc++;
		else ec++;
	}
	if(n==k)
	{
		if(oc%2==0)printf("Daenerys\n");
		else printf("Stannis\n");
	}
	else if((n-k)/2<oc && (n-k)/2<ec)
	{
		if((n-k)%2==0)printf("Daenerys\n");
		else printf("Stannis\n");
	}
	else if((n-k)/2>=oc)
	{
		printf("Daenerys\n");
	}
	else
	{
		if(k%2==0)printf("Daenerys\n");
		else printf("Stannis\n");
	}
	return 0;
}
