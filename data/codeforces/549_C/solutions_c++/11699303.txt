#include <bits/stdc++.h>
using namespace std;

int n,m,k,x;

int main()
{
	cin>>n>>k,k=n-k;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&x);
		if(x&1) m++;
	}

	int u=-1,v=-1,w;
	if(k/2>=m)
		u=m&1;
	if(k/2>=n-m)
		v=(k-(n-m))&1;

	if(k==0)
		w=!(m&1);
	else if(k&1)
	{
		if(m&1) w=u==1||v==1;
		else w=u==0||v==0;
	}
	else
	{
		if(m&1) w=!(u==0||v==0);
		else w=!(u==1||v==1);
	}

	puts(w?"Daenerys":"Stannis");
	return 0;
}
