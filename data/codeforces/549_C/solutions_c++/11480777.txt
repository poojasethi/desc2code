#include<iostream>
using namespace std;

int n, k;
int t1, t2;

int f(){
	int t = n - k;
	
	if (k==n){
		if (t1%2==1) return 1;
		return 2;
	}
	
	if (t%2==1){
		if (t/2 >= t1) return 2;
		if (t/2 >= t2 && k%2==0) return 2;
		return 1;
	}
	
	if (t/2 < t2) return 2;
	if (k%2==0) return 2;
	return 1;
}

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> n >> k;
	for (int i=0 ; i<n ; i++){
		int x;
		cin >> x;
		if (x%2==0) t2++;
		else t1++;
	}
	if (f()==1) cout << "Stannis" << endl;
	else cout << "Daenerys" << endl;
}
