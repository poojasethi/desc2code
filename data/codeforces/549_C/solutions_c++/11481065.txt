#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
int n,k,odd = 0,even = 0,a,b,x;
bool win;

int main(){
	scanf("%d%d",&n,&k);
	for(int i = 0;i < n;++ i){
		scanf("%d",&x);
		if(x & 1) ++ odd;
		else ++ even;
	}
	a = (n - k + 1) >> 1; b = n - k - a;
	if(n - k == 0) win = odd & 1;
	else if((n - k) & 1)
		win = (k + 1) & 1 ? b < even && b < odd : b < odd;
	else 
		win = !((k + 1) & 1) && a >= even;
	puts(win ? "Stannis" : "Daenerys");
	return 0;
}
