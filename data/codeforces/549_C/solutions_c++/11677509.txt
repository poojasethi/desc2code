#include <cstdio>
#define MAX 100000

using namespace std;

long int i,k,n,b,counter=0;

bool winner()
{
    if(n==k) return counter&1;
    if(k&1)
    {
        if(n&1) return 2*counter>=n+k;
        return n-k<2*counter;
    }
    if(n&1) return n-k<2*counter && 2*counter<n+k;
    return 0;
}

int main()
{
    scanf("%ld%ld",&n,&k);
    for(i=0;i<n;i++)
    {
        scanf("%ld",&b);
        counter+=b&1;
    }
    winner() ? printf("Stannis"):printf("Daenerys");
    return 0;
}
