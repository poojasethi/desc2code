#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <string>
#include <cmath>
using namespace std;
int n,k,x,odd,even;
int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		x&1?odd++:even++;
	}
	if(k==n)
		odd&1?printf("Stannis\n"):printf("Daenerys\n");
	else
		if(odd<=(n-k)/2||(even<=(n-k)/2&&(!(k&1))))
			printf("Daenerys\n");
		else
			if(even<=(n-k)/2&&(k&1))
				printf("Stannis\n");
			else
				(n-k)&1?printf("Stannis\n"):printf("Daenerys\n");
	return 0;
}
