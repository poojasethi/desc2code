#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>

using namespace std;

int main()
{
	int n, m; scanf("%d%d", &n, &m);
	
	int x = 0, y = 0;
	for (int i = 0; i < n; i++) {
		int a; scanf("%d", &a);
		a & 1 ? x++ : y++;
	}
	
	int ans = 0;
	if (m == n)
		ans = x & 1;
	else if (m & 1)
		ans = (n - m) & 1 ? x > (n - m) / 2 : x >= n - (n - m) / 2;
	else if ((n - m) & 1)
		ans = x > (n - m) / 2 && x < n - (n - m) / 2;
	else ans = 0;
	
	printf("%s", ans ? "Stannis" : "Daenerys");
	
	return 0;
}
