#include<stdio.h>
int main()
{
	int jishu=0,oushu=0;
	int n,k;
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++)
	{
		int x;
		scanf("%d",&x);
		if (x&1) jishu++;
		else oushu++;
	}
	if (n==k)
	{
		if (jishu&1) printf("Stannis\n");
		else printf("Daenerys\n");
		return 0;
	}
	if (jishu<=(n-k)/2)
	{
		printf("Daenerys\n");
		return 0;
	}
	if (oushu<=(n-k)/2)
	{
		if (k&1) printf("Stannis\n");
		else printf("Daenerys\n");
		return 0;
	}
	if ((n-k)&1) printf("Stannis\n");
	else printf("Daenerys\n");
	return 0;
}
