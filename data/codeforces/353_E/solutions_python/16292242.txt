def foo(s):
	a = []
	n = len(s)
	for i in range(n):
		if 0 == i or s[i] != s[i - 1]: a.append(1)
		else: a[-1] = 0
	if s[0] == s[-1]:
		a[0] = 0
		del a[-1]
	cnt = 0
	ans = 0
	for x in a:
		if 1 == x: cnt += 1
		else:
			ans += cnt / 2 + 1
			cnt = 0
	return ans + cnt / 2

s = raw_input()
print foo(s)
