#include <bits/stdc++.h>
#define SZ( x ) (( (int)x.size() ))
#define INF 1e9 + 100

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pii;

const ll N = 5000 * 1000 + 1000;

string s;
int n;

int main(){
    ios::sync_with_stdio(false);
    cin >> s;
    n = s.length();
    if(n == 2){
        cout << 1 << endl;
        return 0;
    }
    s.push_back(s[0]);
    s.push_back(s[1]);
    s.push_back(s[2]);
    int ans = 0;
    for(int i = 0; i < n; i++)
        if(s[i] != s[i+1])
            if(s[i] != s[i+2])
                ans++;
            else if(s[i] != s[i+3])
                ans++, i++;
    cout << ans << endl;
    return 0;
}
