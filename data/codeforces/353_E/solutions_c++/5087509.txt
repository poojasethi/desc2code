#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int n,i,bo[1200000],ans;
char c[3200000];
void work(int x){
	int i,j,res=0;
	memset(bo,0,sizeof(bo));
	for(i=x;i<x+n;++i){
		bo[i%n]=1;
		res++;
		do{
			if(bo[(i+1)%n]==1)res--;
			i++;
		}while(c[i]==c[i-1]);
	}
	ans=max(res,ans);
}
int main(){
	scanf("%s",c);
	n=strlen(c);
	for(i=0;i<n;++i)c[i+n]=c[i+n+n]=c[i];
	work(0);
	work(1);
	work(n-1);
	printf("%d\n",max(1,ans));
}
