#include <iostream>
#include <fstream>
#include <cstring>
using namespace std;

#define F cin
#define G cout

const int N = 1000010;

string a;
int b[N],n,now,k,ans;

int main()
{
    F>>a;
    n = a.size();

    for (int i=0;i<n;++i)
        if ( a[i] != a[i-1] )
            b[now++] = 1;
        else
            b[now-1]++;
    if ( a[0] == a[n-1] )
        b[0] += b[--now];

    for (int i=0;i<now;++i)
        if ( b[i] == 1 ) //
            k++;
        else
        {
            ans += k/2+1;
            k=0;
        }
    G<<ans+k/2<<'\n';
}
