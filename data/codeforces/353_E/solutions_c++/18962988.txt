#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
using namespace std;
int m,i,j,k,ans;
const int n=1e6+1e3;
int s[n],t;
char a[n];
int main()
{
    cin>>a;
    for(i=0;a[i];i++)if(a[i]-a[i-1])s[t++]=1;else s[t-1]++;
    if(a[0]==a[i-1])s[0]+=s[--t];
    for(i=0;i<t;i++)if(s[i]==1)k++;else ans+=k/2+1,k=0;
    cout<<ans+k/2;
}
