#include  <stdio.h>
#include  <stdlib.h>
#include  <string.h>
char in[1000010];

int main(int argc, char *argv[])
{
    int i, len, max;
    memset(in, 0, sizeof(in));
    scanf("%s", in);
    len = strlen(in);
    in[len] = in[0];
    in[len+1] = in[1];
    in[len+2] = in[2];
    max = 0;
    for ( i = 0; i < len; i++ ) {
        if ( in[i] != in[i+1] ) {
            if ( in[i] != in[i+2] ) {
                max++;
            } else if ( in[i+3] != in[i] ) {
                max++;
                i++;
            }
        }
    }
    printf("%d\n", max);
}
