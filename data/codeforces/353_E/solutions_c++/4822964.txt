#include <stdio.h>
#include <algorithm>
#include <cstring>
#include <string>
#include <map>
#define For(i,b,e) for(int i=b; i<=e; i++)
using namespace std;
char s[1000010];
int main()
{
	scanf("%s",s);
	int n=strlen(s),st=-1,ans=1;
	For(i,0,n-1){
		if(s[i]==s[(i+1)%n]){
			st=i;
			break;
		}
	}
	if(st==-1){
		printf("%d",n/2);
		return 0;
	}
	char nc=s[st];
	bool flag=false;
	for(int i=st+2; i!=st&&i!=(st+1)%n; i=(i+1)%n){
		if(s[i]==nc)continue;
		ans++;
		i=(i+1)%n;
		nc=s[i];
		if(st==i)
			flag=true;
		//printf("%c %d %d\n",s[i],i,ans);
	}
	if(flag)ans--;
	else if(nc==s[st])ans--;
	printf("%d",ans);
	return 0;
}
