#include <algorithm>
#include <iostream>
using namespace std;

int n, dsu;
string s, t;

int start(int m) {
	int res = 1;
	for (int i = m; i + 2 < n + m - 1; i++)
		if (t[i] != t[i + 1])
			res++, i++;
	return res;
}
int main() {
	ios_base::sync_with_stdio(false);
	cin >> s; n = s.size();
	t = s + s;
	for (int i = 0; i < n; i++)
		if (t[i] != t[i + 1] && t[i + 1] != t[i + 2]) {
			int ans = start((i + 2) % n);
			s[(i + 1) % n] = '2';
			reverse(s.begin(), s.end());
			t = s + s;
			for (int i = 0; i < n; i++)
				if (s[i] == '2') {
					cout << max(ans, start((i + 1) % n)) << '\n';
					return 0;
				}
		}

	for (int i = 0; i < n - 1; i++)
		if (s[i] != s[i + 1])
			dsu++;
	cout << dsu + (s[n - 1] != s[0]) << '\n';
	return 0;
}
