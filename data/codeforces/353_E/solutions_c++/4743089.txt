#include <cstdio>
#include <cstring>
using namespace std;
const int maxn=1000010;
char s[maxn];
int n;
bool f[maxn];
int prev(int x){if (x>1) return x-1;return n;}
int nxt(int x){if (x<n) return x+1;return 1;}
int main(){
    int d=0,z=0,x,i,g=0;
    scanf("%s",s+1);
    n=strlen(s+1);
    for (i=1;i<=n;i++) f[i]=false;
    for (i=1;i<=n;i++) if (s[prev(i)]!=s[i]) z++,f[i]=true;
    if (z==n){printf("%d\n",n>>1);return 0;}
    for (i=n;i>0;i--) if (!f[i]&&f[prev(i)]) break;
    x=i;z=0;
    i=prev(x);
    while (i!=x){
        if (!f[i]){
            if (g>0) z+=((g-1)>>1);g=0;
        } else{
            if (!f[nxt(i)]) d++;g++;
        }
        i=prev(i);
    }
    if (g>0) z+=((g-1)>>1);
    z+=d;
    printf("%d\n",z);
    return 0;
}
