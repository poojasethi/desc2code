#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=1000002;
char a[maxn];
int f[maxn];
bool v[maxn];
int n,s,ans;

int main(){
  int i,j;
  scanf("%s",a);
  n=strlen(a);
  for (i=0;i<n;++i)
    if (a[i]=='1') ++f[i];
    else ++f[(i+1)%n];
  for (i=0;i<n;++i)
    if (!f[i]){
      s=0;
      j=(i+1)%n;
      if (f[j]!=2) ++s;
      else {
        if (v[j]) ++s;
        v[j]=1;
      }
      j=(i-1+n)%n;
      if (f[j]!=2) ++s;
      else {
        if (v[j]) ++s;
        v[j]=1;
      }
      ans+=s;
      if (!s){
        ++ans;
        v[j]=0;
        v[(i+1)%n]=0;
      }
    }
  printf("%d\n",ans);
  return 0;
}
