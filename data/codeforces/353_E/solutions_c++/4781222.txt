#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

char s[1000005];

int sum[1000005],t[1000005];

int main()
{
    scanf("%s",s);
    int n=strlen(s);
    int ans=0,cnt=0,m=0;
    sum[cnt++]=1;
    for(int i=1;i<n;++i)
    {
        if(s[i]!=s[i-1]) sum[cnt++]=1;
        else ++sum[cnt-1];
    }
    if(s[0]==s[n-1]) sum[0]+=sum[--cnt];
//    for(int i=0;i<cnt;++i) cout<<sum[i]<<"**"<<endl;
    for(int i=0;i<cnt;++i)
    {
        if(sum[i]==1) ++t[m];
        else
        {
            ++ans;
            ++m;
        }
    }
//    cout<<m<<endl;
    if(m&&sum[cnt-1]==1&&sum[0]==1) t[0]+=t[m--];
//    for(int i=0;i<=m;++i) cout<<t[i]<<"^^"<<endl;
    for(int i=0;i<=m;++i) ans+=t[i]/2;
    printf("%d\n",ans);
    return 0;
}
