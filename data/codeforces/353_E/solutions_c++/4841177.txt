#include <cstdlib>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <iostream>
#include<algorithm>
#include<set>
using namespace std;
char str[1010000];
int len;
int cul(char *str)
{
	int i=0;
	int ans=0;
	while(i<len)
		{
			if((str[i]=='0'&&str[i+1]=='1')||(str[i]=='1'&&str[i+1]=='0'))
				{
				ans++;
				i+=2;
				}
			else
				{
				i++;
				}
		}
return ans;
}
int main()
{
	scanf("%s",str);
	len=strlen(str);
	int ans=cul(str);

	char *p=str+1;
	p[len-1]=str[0];
	p[len]=0;
	ans=max(ans,cul(p));
	printf("%d\n",ans);
}
