#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>
#include <vector>
#include <cmath>

using namespace std;
#define N 1000100
char a[N];
int s[N],n,i,t,k,ans;

int main() {
    scanf("%s",a+1);
    a[0]='$';
    for(i=1;a[i];i++)
        if(a[i]!=a[i-1]) s[t++]++;
        else s[t-1]++;
    if(a[i-1]==a[1]) s[0]+=s[--t];
    for(i=0;i<t;i++) {
        //printf("%d ", s[i]);
        if(s[i]==1) k++;
        else ans+=k/2+1,k=0;
    }
    printf("%d\n",ans+k/2);
    return 0;
}
