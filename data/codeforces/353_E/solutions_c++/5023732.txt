#include <stdio.h>
#include <string.h>

char s[1000005];

int main()
{
long long int n,i,ans;
while(scanf("%s",s)!=EOF)
{
   n=strlen(s);
   s[n]=s[0];
   s[n+1]=s[1];
   s[n+2]=s[2];
   ans=0;
   for(i=0;i<n;i++)
   {
      if(s[i+1]!=s[i] && s[i+2]!=s[i]) 
	  {
         ans++;
      }
   }
   for(i=0;i<n;i++)
   { 
      if(s[i+1]!=s[i] && s[i+2]==s[i] && s[i+3]!=s[i]) 
	  {
         ans++;
		 i++;
      }
   }
   printf("%I64d\n",ans);
}
return 0;
}
