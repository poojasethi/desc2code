#include<stdio.h>
char str[1000005];
int  ct,set[1000005]={1},ctONE,sol;
main()
{
    int i,j;
    scanf("%s",str);
    for (i=1;str[i];i++)
    {
        if (str[i]==str[i-1])
            set[ct]++;
        else
            set[++ct]=1;
    }
    if (str[i-1]==str[0])
    {   set[0]+=set[ct]; ct--; }
    //for (i=0;i<=ct;i++) printf("%d ",set[i]); printf("\n");
    for (i=0;i<=ct;i++) 
    {
        if (set[i]==1)
            ctONE++;
        else 
        {
            sol += 1+ (ctONE/2);
            ctONE=0;
        }
        //printf("sol=%d , ctONE=%d\n",sol,ctONE);
    }
    sol += ctONE/2;
    printf("%d",sol);
}
