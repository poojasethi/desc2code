#include <stdio.h>
#include <string.h>

char input[1000001];
bool few[1000000];

int main()
{
    scanf("%s", input);
    int n = strlen(input);

    int t = 0;

    while (input[n-1] == input[0]) {
        few[t] = true;
        n--;
    }

    for (int i = 1; i < n; i++) {
        if (input[i] != input[i-1])
            t++;
        else
            few[t] = true;
    }

    int cnt = 0;

    bool take = true;
    for (int i = 0; i < t; i++) {
        if (take)
            cnt++;
        take = few[i] || !take;
    }
    if (take && few[t]) cnt++;

    printf("%d", cnt);
}
