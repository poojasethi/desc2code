#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<vector>
using namespace std;
vector<pair<int,int> > V;
char s[2000001];
int main(){
scanf("%s",s);
int len=strlen(s);
for(int i=0;i<len-1;++i){
if(s[i]!=s[i+1]) {rotate(s,s+i+1,s+len);break;}
}
for(int i=0;i<len;){
int j=i;
while(j<len&&s[j]==s[i]) ++j;
if(j-i>1) V.push_back(make_pair(i,j));
i=j;
}
if(!V.size()) printf("%d\n",len/2);
else{
int ans=V.size();
sort(V.begin(),V.end());
V.push_back(make_pair(V[0].first+len,V[0].second+len));
for(int i=1;i<int(V.size());++i){
if(V[i].first>V[i-1].second+1) ans+=(V[i].first-V[i-1].second)/2;
}
printf("%d\n",ans);
}
return 0;
}
