#include <cstdio>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstring>

using namespace std;

char s[1000005];
int len,q0,q[10000005],dq,ans;
int main()
{
	scanf("%s",s);
	len=strlen(s);
	++q0;q[q0]=1;
	for (int i=1;i<len;++i)
	{
		if (s[i]==s[i-1])
		{
			q[q0]=0;
		}
		else
		{	
			++q0;
			q[q0]=1;
		}
	}
	if (s[0]==s[len-1]) q[1]=0,--q0;
	dq=0,ans=0;
	for (int i=1;i<=q0;++i)
	{
		if (q[i]>0) ++dq;
		else ans+=dq/2+1,dq=0;
	}
	ans+=dq/2;
	printf("%d\n",ans);
	return 0;
}
