#include <stdio.h>
#include <string.h>

char str[1000006];
int N;
int main()
{
	scanf("%s", str);
	N = strlen(str);

	int ans = 1;
	int f0 = 0;
	int f1 = 0;

	for (int i = 0; i < N; ++i)
	{
		if (str[i] == '0')
			f0 = 1;
		else
			f1 = 1;
		if (f0 == 1 && f1 == 1)
		{
			ans++;
			f0 = 0;
			f1 = 0;
		}
	}
	ans--;
	if (ans == 0)
		ans = 1;

	int ans2 = 1;
	int st = str[0] - '0';
	f0 = f1 = 0;
	for(int k = 1; k < N; ++k)
	{
		if (str[k] == '0')
			f0 = 1;
		else
			f1 = 1;
		if (f0 == 1 && f1 == 1)
		{
			ans2++;
			f0 = 0;
			f1 = 0;
		}
	}
	if (str[0] == '0')
	{
		if (f1 == 0)
			ans2--;
	}
	else
	{
		if (f0 == 0)
			ans2--;
	}

	if (ans2 > ans)
		ans = ans2;

	
	printf("%d\n", ans);
}
