#include <iostream>

using namespace std;

int main() {
    string s;
    cin >> s;
    int start = 0, n = s.length();
    for (int i = 0; i < n; i++) {
        if (s[i] == s[(i - 1 + n)%n] && s[i] != s[(i - 2 + n)%n]) {
            start = i;
            break;
        }
    }
    //cout << "start: " << start << endl;
    int cur = start, count = 0;
    do {
        //cout << "cur: " << cur << endl ;
        int next = cur;
        while (s[next] == s[cur]) next = (next + 1)%n;
        cur = (next + 1)%n;
        count++;
    } while ((start - cur + n)%n > 1);
    cout << count;
}
