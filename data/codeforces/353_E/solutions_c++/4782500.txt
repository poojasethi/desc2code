#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
#include <string.h>
#include <iomanip>
#include <math.h>
using namespace std;
const int N = 1111111;

int a[N];
char s[N];

int main(){
    while (scanf("%s",s)==1){
        int n = strlen(s), m = 1;
        a[1] = 1;
        for (int i=1;i<n;++i)
            if (s[i]==s[i-1]) a[m]++;
            else a[++m] = 1;
        if (s[0] == s[n-1]) a[1] += a[m--];

        int ans = 0, ret = 0;
        for (int i=1;i<=m;++i){
            if (a[i] == 1) ++ret;
            else{
                ans += 1;
                ans += ret / 2;
                ret = 0;
            }
        }
        if (ret) ans += ret / 2;
        cout << ans << endl;
    }
    return 0;
}
