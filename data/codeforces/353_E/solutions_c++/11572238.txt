#include<cstdio>
#include<cstring>
#define MAXN 2000005
using namespace std;
char s[MAXN];
int len,ans;
int main()
{
	scanf("%s",s+1);
	len=strlen(s+1);
	if (len==2)
	{puts("1");return 0;}
	strcat(s+1,s+1);
	for (int i=1;i<=len;i++)
	if (s[i]!=s[i+1])
	{
		if (s[i]!=s[i+2])ans++;
		else if (s[i]!=s[i+3])ans++,i++;
	}
	printf("%d\n",ans);
}
