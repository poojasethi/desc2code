#include <cstdlib>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <iostream>
#include<algorithm>
#include<set>
using namespace std;
char str[1010000];
int main()
{
	scanf("%s",str);
	int len=strlen(str);
	int i=0;
	int ans=0;
	while(i<len)
		{
			if((str[i]=='0'&&str[i+1]=='1')||(str[i]=='1'&&str[i+1]=='0'))
				{
				ans++;
				i+=2;
				}
			else
				{
				i++;
				}
		}
	char *p=str+1;
	p[len-1]=str[0];
	p[len]=0;
	i=0;
	int ans1=0;
	while(i<len)
		{
			if((p[i]=='0'&&p[i+1]=='1')||(p[i]=='1'&&p[i+1]=='0'))
				{
				ans1++;
				i+=2;
				}
			else
				{
				i++;
				}
		}
	printf("%d\n",max(ans1,ans));
}
