#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#define maxn 1000010
using namespace std;
char str[maxn];
int work()
{
    static int len[maxn],tot;
    int n=strlen(str+1),sum=0,ans=0;
    for(int i=1;i<=n;++i)
    {
        if(str[i]!=str[i-1])
            len[++tot]=1;
        else
            ++len[tot];
    }
    if(str[1]==str[n])
        len[1]+=len[tot--];
    for(int i=1;i<=tot;++i)
    {
        if(len[i]==1)
            ++sum;
        else
            ans=ans+sum/2+1,sum=0;
    }
    return ans+sum/2;
}
int main()
{
    scanf("%s",str+1);
    printf("%d\n",work());
    return 0;
}
