#define N 1000050
#define mp(a,b) make_pair(a,b)
#define rep(i,l,r) for (int i = l;i <= r;i ++)
using namespace std;

int l[N],r[N],n;
char s[N];

void solve(){
	n = strlen (s);
	rep(i,0,n - 1){
		if (s[i] == '1'){
			r[i] = 1;
			l[(i + 1) % n] = 0;
		}
		else {
			r[i] = 0;
			l[(i + 1) % n] = 1;
		}
	}
	int cnt = 0;
	rep(i,0,n - 1){
		if (l[i] != 2 && l[i] == r[i]){
			cnt ++;
//			printf ("i = %d\n",i);
			if (i - 1 < 0){
				r[n - 1] = 2;
			}
			else r[i - 1] = 2;
			l[(i + 1) % n] = 2;
		}
	}
	printf ("%d\n",cnt);
}

int main (){
//	freopen ("cf353e.txt","r",stdin);
	while (~scanf ("%s",s)){
		solve();
	}
	return 0;
}
