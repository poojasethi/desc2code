#include <cstdio>
#include <cstring>
int N,s[1000005];
char a[1000005];
int main()
{
	scanf("%s",a),N=strlen(a);
	int t=0,k=0,ans=0;
	for (int i=0; i<N; i++) if (a[i]!=a[i-1]) s[t++]=1; else s[t-1]++;
	if (a[0]==a[N-1]) s[0]+=s[--t];
	for (int i=0; i<t; i++) if (s[i]==1) k++; else ans+=k/2+1,k=0;
	printf("%d\n",ans+k/2);
}
