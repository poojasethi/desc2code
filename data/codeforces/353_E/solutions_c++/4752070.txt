#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <fstream>

using namespace std;

const int MAXN = 1000000 + 10;

char s[MAXN];
int sum[MAXN],cnt,n,ans,k;

int main()
{
	scanf("%s",s);
	n=strlen(s);
	for(int i=0;i<n;i++) if(s[i]!=s[i-1]) sum[cnt++]=1;else sum[cnt-1]++;
	if(s[0]==s[n-1]) sum[0]+=sum[--cnt];
	for(int i=0;i<cnt;i++) if(sum[i]==1) ++k;else {ans+=k/2+1;k=0;}
	printf("%d\n",ans+k/2);
	
	return 0;
}
