#include <cstdio>
using namespace std;

const int N = 200000+10;
int x[N];
long long s[N];

int main()
{
	int n, k;
	scanf("%d%d", &n, &k);
	for(int i = 1; i<=n; ++i) scanf("%d", &x[i]);

	s[1] = 0;
	for(int i = 1; i<=k; ++i) s[1] += x[i];
	for(int i = 2; i+k-1<=n; ++i) s[i] = s[i-1]-x[i-1]+x[i+k-1];
	long long maxs = 0, sa = 0;
	int a, b, ansa, ansb;
	for(b = k+1; b+k-1<=n; ++b) {
		if(s[b-k]>sa) sa = s[a = b-k];
		if(sa+s[b]>maxs) {
			maxs = sa+s[b];
			ansa = a; ansb = b;
		}
	}

	printf("%d %d\n", ansa, ansb);
}
