#include<stdio.h>
#include<stdlib.h>
typedef long long LL;
long i,n,ansj,ansi,k;
LL to[200010],a[200010],f[200010],f1[200010],ans;
int main(){
	scanf("%ld%ld",&n,&k);
	for (i=1;i<=n;i++)
		scanf("%I64d",&a[i]);
	for (i=n;i;i--)
		f[i]=f[i+1]-a[i+k]+a[i];
	for (i=n;i;i--){
		to[i]=i;f1[i]=f[i];
		if (f1[i]<f1[i+1]){
			to[i]=to[i+1];
			f1[i]=f1[i+1];
		}
	}
	for (i=1;i<=n;i++)
		if (i+2*k-1<=n && f[i]+f1[i+k]>ans){
			ans=f[i]+f1[i+k];
			ansi=i;
			ansj=to[i+k];
		}
	printf("%ld %ld\n",ansi,ansj);
	return 0;
}
