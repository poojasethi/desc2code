#include<cstdio>
#include<algorithm>
using namespace std;

int b, x, bb, n, k, a[200001];

long long s[200001], M, R;

int main()
{
	scanf("%d %d", &n, &k);
	s[1] = 0;
	for(int i = 1; i <= n; i++)
		scanf("%d", &a[i]);
	for(int i = 1; i <= k; i++)
		s[1] += a[i];
	for(int i = k+1; i <= n; i++)
		s[i-k+1] = s[i-k] - a[i-k] + a[i];
	
	for(int i = n-2*k+1; i >= 1; i--)
	{
		if(M <= s[i+k]) M = s[i+k], b = i+k;
		if(R <= M + s[i]) R = M + s[i], x = i, bb = b;
	}
	
	printf("%d %d\n", x, bb);
	
	
	return 0;
}
