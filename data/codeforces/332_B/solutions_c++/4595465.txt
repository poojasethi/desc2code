#include <iostream>
#include <stdio.h>
using namespace std;
long long a[200005]={0},b[200005]={0},im[200005]={0};
int main()
{//freopen("i.txt","r",stdin);
int n,k,x; cin>>n>>k;

 for (int i=1;i<=n;i++) {cin>>x; a[i]=a[i-1]+x;}
 for (int i=k;i<=n;i++) {b[i]=a[i]-a[i-k];}
 int i0,j0=n;
 long long s;
im[n]=n;
 for (int j=n-1;j>=2*k;j--)
  {if (b[j]>=b[j0]) j0=j;
   im[j]=j0;
  }
 i0=0,j0=0,s=0;
 for (int i=k;i<=n-k;i++)
   if (b[i]+b[im[i+k]]>s) {s=b[i]+b[im[i+k]]; i0=i; j0=im[i+k];}

cout<<i0-k+1<<" "<<j0-k+1;
return 0;
}
