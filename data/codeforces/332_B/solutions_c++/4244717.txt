#include<iostream>
using namespace std;
#define space " "
int main(){
unsigned long long i,n,k,a[200002],b[200002],c[200002],A,B,amax,bmax,p;
cin>>n>>k;
for(i=0;i<n;i++)cin>>a[i];
b[k-1]=0;
for(i=0;i<k;i++)b[k-1]+=a[i];
for(i=k;i<n;i++)b[i]=b[i-1]-a[i-k]+a[i];
c[n-1]=n-1;
for(i=n-2;i>=k;i--)
 c[i]=(b[i]>=b[c[i+1]]?i:c[i+1]);
p=0;
for(A=k-1;A<n-k;A++){

    if(b[A]+b[c[A+k]]>p){
        p=b[A]+b[c[A+k]];
        amax=A;
        bmax=c[A+k];
    }
}
cout<<amax-k+2<<space<<bmax-k+2<<endl;
return 0;
}
