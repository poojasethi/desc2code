#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
const long N=1100000;
long long i,n,k,ans,x,a[N],f[N],s[N];
int main() {
	scanf("%I64d%I64d",&n,&k);
	for (i=1;i<=n;i++) {
		scanf("%I64d",&a[i]);
		s[i]=s[i-1]+a[i];
	}
	for (i=1;i+k<=n+1;i++)
		a[i]=s[i+k-1]-s[i-1];
	for (i=n-k+1;i>0;i--) f[i]=max(f[i+1],a[i]);
	for (i=1;i+2*k<=n+1;i++)
		if (a[i]+f[i+k]>ans) {
			x=i;
			ans=a[i]+f[i+k];
		}
	for (i=x+k;i+k<=n+1;i++)
		if (ans==a[i]+a[x]) break;
	printf("%I64d %I64d\n",x,i);
	return 0;
}
