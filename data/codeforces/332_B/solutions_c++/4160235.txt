#include<cstdio>
#include<iostream>

using namespace std;;

int main(){
	int a, b, i, j, k, n;
	long long int v[200001];
	int ind[200001];
	scanf("%d %d", &n, &k);
	v[0]=0;
	for(i=1; i<=n; i++){
		cin >> v[i];;
		v[i]+=v[i-1];}
	long long int max=0;
	for(i=n-k; i>=0; i--){
		if(v[i+k]-v[i] >= max){
			max=v[i+k]-v[i];
			ind[i]=i;}
		else{
			ind[i]=ind[i+1];}}
	max=0;
	for(i=0; i+2*k<=n; i++){
		j=ind[i+k];
		long long int tmp = (v[i+k]-v[i]) + (v[j+k]-v[j]);
		if(tmp > max){
			max = tmp;
			a=i;
			b=j;}}
	printf("%d %d\n", ++a, ++b);
	return 0;}
