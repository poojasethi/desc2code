#include<iostream>
#include <string.h>
#include<vector>
using namespace std; 
int main(){
	long long n,k,q1,q2,a,b=0,fi=0,x,y,best=0;
	int i;
	vector<long long> sum(1,0);
	cin>>n>>k;
	for(i=1;i<=n;i++){
		cin>>a;
		sum.push_back(sum[i-1]+a);
	}
	for(i=n-2*k;i>=0;i--){
	q2=sum[i+k]-sum[i];
	q1=sum[i+2*k]-sum[i+k];
		if(q1>=b)
		{
			b=q1;
			fi=i+k+1;
		}
    	if(q2+b>=best)
		{
			best=q2+b;
			x=i+1;
			y=fi;
		}
	}
	cout<<x<<" "<<y;
	return 0;
}
