#include<iostream>
using namespace std;
long long a[200001],m[200001],bm[200001];
int main() {
	int n,k;
	cin>>n>>k;
	for(int i=1;i<=n;i++) {
		cin>>a[i];
		a[i]+=a[i-1];
	}
	long long max=0,bmax=0;
	for(int i=n-k+1;i>0;i--) {
		if(a[i+k-1]-a[i-1]>=max) {
			max=a[i+k-1]-a[i-1];
			bmax=i;
		}
		m[i]=max;
		bm[i]=bmax;
	}
	max=0;
	long long aa=0,bb=0;
	for(int i=1;i+k+k-1<=n;i++)
		if(a[i+k-1]-a[i-1]+m[i+k]>max) {
			max=a[i+k-1]-a[i-1]+m[i+k];
			aa=i;
			bb=bm[i+k];
		}
	cout<<aa<<" "<<bb;
	return 0;
}
