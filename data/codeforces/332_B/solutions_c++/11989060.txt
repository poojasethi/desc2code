#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll x[200005],n,i,v=0,a,b,l=0,k,g=0,bb;
main()
{
    scanf("%I64d%I64d",&n,&k);
    for (i=1;i<=n;i++)
        scanf("%I64d",&x[i]),x[i]+=x[i-1];
    for (i=n-2*k+1;i>=1;i--)
    {
        ll u=x[i+k-1]-x[i-1];
        ll s=x[i+2*k-1]-x[i+k-1];
        if (l<=s) l=s,bb=i+k;
        if (u+l>=g)  g=u+l,b=bb,a=i;
    }
    printf("%I64d %I64d",a,b);
}
