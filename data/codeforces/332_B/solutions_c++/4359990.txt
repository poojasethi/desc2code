#include<iostream>
#include<stdio.h>
#include<cstring>
#include<algorithm>
using namespace std;
long long  n,k,i,c,m=0,a,b,sum[200005]={0},f[200005]={0},p[200005]={0};
int main(){
	cin>>n>>k;
	for(i=1;i<=n;i++){
		scanf("%I64d",&c);
		sum[i]=sum[i-1]+c;
	}	
	for(i=k;i<=n;i++){
		if (sum[i]-sum[i-k]>f[i-1])
			f[i]=sum[i]-sum[i-k],p[i]=i-k+1;
		else 
			f[i]=f[i-1],p[i]=p[i-1];
		if (i>=2*k)
			if (f[i-k]+sum[i]-sum[i-k]>m)
				m=f[i-k]+sum[i]-sum[i-k],a=p[i-k],b=i-k+1;
	}
	cout<<a<<" "<<b;
	return 0;
}
