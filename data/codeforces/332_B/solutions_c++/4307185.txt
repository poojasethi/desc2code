#include<cstdio>
int i,a,b,n,k,bb;
int w[200005];
long long d[200005],m,r;
int main()
{
    scanf("%d%d",&n,&k);
    for(i=1;i<=n;i++)
    scanf("%d",&w[i]);
    for(i=1;i<=k;i++)
    d[1]+=w[i];
    for(i=k+1;i<=n;i++)
    {
        d[i-k+1]=d[i-k]-w[i-k]+w[i];
    }
    for(i=n-2*k+1;i>=1;i--)
    {
        if(m<=d[i+k]) m=d[i+k],b=i+k;
        if(r<=m+d[i]) r=m+d[i],a=i,bb=b;
    }
    printf("%d %d",a,bb);
}
