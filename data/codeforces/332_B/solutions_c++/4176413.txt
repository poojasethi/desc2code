#include <iostream>

using namespace std;
int n,i,j,k,a,b;
long long sol,Max,x[200010];
int main()
{
    cin>>n>>k;
    for(i=1;i<=n;i++){cin>>x[i];x[i]+=x[i-1];}
    i=1;j=k+1;sol=x[2*k];a=1;Max=x[k];
    for(b=k+2;b+k-1<=n;b++)
    {
        if(x[b-1]-x[b-k-1]>Max){a=b-k;Max=x[b-1]-x[b-k-1];}
        if(x[b+k-1]-x[b-1]+Max>sol){sol=x[b+k-1]-x[b-1]+Max;i=a;j=b;}
    }


    cout << i << ' ' << j << endl;
    return 0;
}
