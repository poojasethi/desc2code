#include<iostream>

int i,j,n,k,a,b,bb,x[200005];
long long tot=0,tempb=0,sum[200005];

using namespace std;

int main()
{
    cin>>n>>k;
    for(i=0;i<n;i++)
        cin>>x[i];
    sum[0]=0;
    for(i=0;i<k;i++)
        sum[0]+=x[i];
    for(i=1;i<n-k+1;i++)
        sum[i]=sum[i-1]-x[i-1]+x[i+k-1];
    for(i=n-2*k;i>=0;i--)
    {
        if(tempb<=sum[i+k])
            tempb=sum[i+k],bb=i+k+1;
        if(tempb+sum[i]>=tot)
            tot=tempb+sum[i],a=i+1,b=bb;
    }
    cout<<a<<" "<<b;
    return 0;
}
