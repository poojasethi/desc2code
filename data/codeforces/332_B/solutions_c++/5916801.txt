#include <cstdio>
#define LL long long
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

int n,k,ans;
LL a[200020];
int b[200020];

inline LL F(int x) {
	return a[x+k-1]-a[x-1];
}

int main() {
	scanf("%d%d",&n,&k);
	REP(i,n) scanf("%I64d",&a[i]);
	REP(i,n) a[i]+=a[i-1];
	for (int i=n-k-k+1;i;--i) {
		b[i]=b[i+1];
		if ((!b[i])||(F(i+k)>=F(b[i])))
			b[i]=i+k;
	}
	for (int i=n-k-k+1;i;--i)
		if ((!ans)||(F(i)+F(b[i])>=F(ans)+F(b[ans])))
			ans=i;
	printf("%d %d\n",ans,b[ans]);
	return 0;
}
