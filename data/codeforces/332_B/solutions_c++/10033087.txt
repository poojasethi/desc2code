#include<bits/stdc++.h>
using namespace std;
long long a[200009],s[200009],M,R;
int main()
{
    int n,k;
    cin>>n>>k;
    int i;
    s[1]=0;
    for(i=1; i<=n; i++)
    cin>>a[i];
    for(i=1; i<=k; i++)
    s[1]+=a[i];
    int b,x,bb;
    for(i=k+1; i<=n; i++)
        s[i-k+1]=s[i-k]-a[i-k]+a[i];
    for(i=n-2*k+1; i>=1; i--)
    {
        if(M<=s[i+k])M=s[i+k],b=i+k;
        if(R<=M+s[i])R=M+s[i],x=i,bb=b;
    }
    cout<<x<<" "<<bb<<endl;
    return 0;
}
