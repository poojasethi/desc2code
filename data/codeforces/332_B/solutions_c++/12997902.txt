#include <iostream>
using namespace std;
long long S[200006],n,k,a,b,f,c,i;
int main(){            
  cin>>n>>k;
  for (i=a=f=1,b=k+1,S[1]=0;i<=n;i++){
    cin>>c; S[i+1]=S[i]+c; c=i-k+1;
    if (i>=k) S[c]=S[i+1]-S[c];
    if (i>=k*2 && S[c-k]>S[f]) f=c-k;
    if (i>=k*2 && S[f]+S[c]>S[a]+S[b]) a=f, b=c;
  }
  cout<<a<<" "<<b<<endl;
}
