#include <cstdio>
#include <iostream>
using namespace std;
typedef long long ll;
ll a[200001],s[200001],m,r;
int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    s[1]=0;
    for (int i=1; i<=n; i++) scanf("%d",a+i);
    for (int i=1; i<=k; i++) s[1]+=a[i];
    int b,x,y;
    for (int i=k+1; i<=n; i++) s[i-k+1]=s[i-k]-a[i-k]+a[i];
    for (int i=n-2*k+1; i>=1; i--)
    {
        if (m<=s[i+k]) m=s[i+k],b=i+k;
        if (r<=m+s[i]) r=m+s[i],x=i,y=b;
    }
    printf("%d %d\n",x,y);
    return 0;
}
