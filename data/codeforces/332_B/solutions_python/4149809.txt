
N, K = map(int, raw_input().split())
A = [0]+(map(int, raw_input().split()))

for i in xrange(1, N+1):
	A[i] += A[i-1]

def getS(l):
	return A[l+K-1]-A[l-1]

ans = (N-K-K+1, N-K+1)
ma = N-K+1

for i in xrange(N-K-K, 0, -1):
	#print '#i', i
	if getS(i+K)>=getS(ma):
		ma = i+K
	#print '#ma', ma
	if getS(i)+getS(ma)>=getS(ans[0])+getS(ans[1]):
		ans = (i, ma)
	#print '#ans', ans


print ans[0], ans[1]
