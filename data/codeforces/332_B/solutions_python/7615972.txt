#!/usr/bin/env python

n,k = map(int, raw_input().split(' '))
arr = map(int, raw_input().split(' '))

partial_sum = [sum(arr[0:k])]
counter = 0
while counter < len(arr)-k:
    partial_sum.append(partial_sum[counter]-arr[counter] + arr[k+counter])
    counter+=1


max_abs = [0]*(n-k+1)
#partial_sum[n-k]
i = n-k
prev = (-1, -1)
while i >=0:
    #max_abs[i] = max(partial_sum[i], prev)
    if partial_sum[i]>prev[0]:
        max_abs[i] = (partial_sum[i], i+1)
    elif partial_sum[i]==prev[0] and i+1 < prev[1]:
        max_abs[i] = (partial_sum[i], i+1)
    else:
        max_abs[i] = prev
    
    prev = max_abs[i]
    i-=1

counter = 0
sol = [-100000, -100, -100]
while counter < len(partial_sum)-k:
    abs1 = partial_sum[counter]
    abs2 = max_abs[counter+k]
    if abs1+abs2[0] > sol[0]:
        sol[0]=abs1+abs2[0];sol[1]=counter+1;sol[2]=abs2[1]
    counter+=1
    
print sol[1], sol[2]
