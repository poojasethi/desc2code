R=lambda:map(int,raw_input().split())
n,k=R()
a=[0]+R()
b=[n-k]*(n+1)
for i in range(n):
  a[i+1]+=a[i]
for i in range(n-k-1,-1,-1):
  b[i]=b[i+1] if a[b[i+1]+k]-a[b[i+1]]>a[i+k]-a[i] else i
s,u,v=0,0,0
for i in range(n-k-k,-1,-1):
  t=a[i+k]-a[i]+a[b[i+k]+k]-a[b[i+k]]
  if t>=s:
    s,u,v=t,i,b[i+k]
print u+1,v+1
