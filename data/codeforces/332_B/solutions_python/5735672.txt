#!/usr/bin/python

if __name__ == '__main__':
	n, k = map(int, raw_input().split())
	x = map(int, raw_input().split())
	#print x

	# y[i] = sum(j=0)(k-1} x[i+j])
	y = [0]*n
	y[0] = sum(x[:k])
	for i in xrange(1, n-k+1):
		# print '%d - %d + %d' % (y[i-1], x[i-1], x[i+k-1])
		y[i] = y[i-1] - x[i-1] + x[i+k-1]
	#print y

	# z[i] = arg max j>=1 {y[j]}
	# z[i] = (y[j], j)
	z = [0]*n
	z[n-k] = n-k
	for i in xrange(n-k-1, -1, -1):
		if y[i] >= y[z[i+1]]:
			z[i] = i
		else:
			z[i] = z[i+1] 
	#print z

	a, b = 0, k
	for i in xrange(n-2*k+1):
		if y[i] + y[z[i+k]] > y[a] + y[b]:
			a, b, = i, z[i+k]

	print a+1, b+1
