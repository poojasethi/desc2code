R = map (int, raw_input().split())
n, k = R[0], R[1]
x = map(int, raw_input().split())
D = [0 for i in xrange(n)]

for i in xrange(k):
    D[0] += x[i]
for i in xrange(1,n - k+1):
    D[i] = D[i-1] - x[i-1] + x[i+k-1]

l,r = 0,0
a,b, bb = 0,0,0
for i in xrange(n - 2*k, -1, -1):
    if r <= D[i+k]:
        b = i + k
        r = D[i+k]
    if l <= D[i] + r:
        a = i
        l = D[i] + r
        bb = b
print a+1, bb+1
