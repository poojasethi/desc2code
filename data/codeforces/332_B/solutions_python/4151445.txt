from sys import stdin
def main():
    n, k = map(int, stdin.readline().split())
    a = map(int, stdin.readline().split())
    s = [0]
    for x in a:
        s.append(s[-1] + x)
    ans = (0, 0, 0)
    dp = [(0, 1)]
    for i in xrange(n-k):
        x = s[i+k] - s[i]
        dp.append(max(dp[-1], (x, -i)))
    for i in xrange(k, n-k+1):
        ans = max(ans, (dp[i-k+1][0] + s[i+k] - s[i], dp[i-k+1][1], -i))
    print 1-ans[1], 1-ans[2]
main()
