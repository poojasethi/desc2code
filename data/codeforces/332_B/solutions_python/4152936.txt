#!/usr/bin/env python
import sys

if __name__ == '__main__':
    n, k = map(int, sys.stdin.readline().strip().split())
    absu = map(int, sys.stdin.readline().strip().split())

    acc  = [0] * n
    for i, x in enumerate(absu):
        if i == 0:
            acc[i] = absu[i]
        else:
            acc[i] = absu[i] + acc[i-1]

    maxi  = [0] * n
    index = [None] * n
    for i in reversed(xrange(n - k + 1)):
        if i != 0:
            maxi[i] = acc[i + k - 1] - acc[i - 1]
        else:
            maxi[i] = acc[i + k - 1]

        index[i] = i
        if i + 1 < n:
            if maxi[i + 1] > maxi[i]:
                index[i] = index[i + 1]

            maxi[i] = max(maxi[i], maxi[i + 1])

    #print acc
    #print maxi
    #print index

    a, b  = None, None
    stval = float("-inf")
    for i in xrange(n - k + 1):
        if i + k >= len(maxi):
            continue
        val = acc[i + k - 1] - acc[i - 1] + maxi[i + k] if i != 0  else acc[i + k - 1] + maxi[i + k]
        #val = maxi[i] + maxi[i + k]
        if (a is None and b is None) or (val >= stval):
            x, y = i, index[i + k]

            if a is not None and b is not None and val == stval and (x > a or (x == a and y > b)):
                continue

            a, b = x, y
            stval = val

    print a + 1, b + 1
