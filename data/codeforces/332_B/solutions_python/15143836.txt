n,k = map(int,raw_input().split())
ar = raw_input().split()
dp = [0]
for i in xrange(n):
	dp += [dp[i]+int(ar[i])]

fullabs = 0	
maxa = 0
maxb = 0
b = 0
lastb = 0
for i in xrange(n-2*k+1,0,-1):
	#print i
	curb = dp[i+2*k-1]-dp[i+k-1]
	if curb>=b:
		b = curb
		lastb = i+k
	cura = dp[i+k-1]-dp[i-1]
	if cura + b >= fullabs:
		fullabs = cura + b
		maxa = i
		maxb = lastb
		
	#print cura, b
print maxa, maxb
