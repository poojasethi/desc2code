n,k = map(int, raw_input().split())
a = map(int, raw_input().split())

n+=1
a = [0] + a
for i in range(1, n):
	a[i] += a[i-1]

#print a
b = []
for i in range(k, n):
	b.append(a[i] - a[i-k])
#print b

'''
lets assume the answer is x = 0 and y = k initially
'''
x = 0
y = k
m = 0
for i in range(k, len(b)):
#	print "i is " + str(i)
	if b[i-k] > b[m]:
		m = i - k
	if b[m] + b[i] > b[x] + b[y]:
	#	print "updating x, y"
		x = m
		y = i
	#	print x, y

print x+1, y+1
