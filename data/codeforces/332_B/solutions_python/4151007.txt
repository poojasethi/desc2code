n, k = [int(i) for i in raw_input().strip().split()]
absurdities = [int(i) for i in raw_input().strip().split()]
totalsfromi = [0]*n

currtotal = sum(absurdities[0:k])
totalsfromi[0] = currtotal
i = 1
while i < (n-k+1): 
    currtotal += absurdities[i+k-1]
    currtotal -= absurdities[i-1]
    totalsfromi[i] = currtotal
    i += 1

i -= 1
maxbeyond = [0]*n
maxbeyondindices = [-1]*n
currmax = totalsfromi[i]
currmaxindex = i

while i >= 0:
    curr = totalsfromi[i]
    if curr >= currmax:
        currmax = curr
        currmaxindex = i
    maxbeyond[i] = currmax
    maxbeyondindices[i] = currmaxindex
    i -= 1

besttotal = -1
besta = -1
bestb = -1
for i in range(0, n-2*k+1):
    maxj = maxbeyond[i+k]
    maxjindex = maxbeyondindices[i+k]
    t = totalsfromi[i] + maxj
    if t > besttotal:
        besttotal = t
        besta = i
        bestb = maxjindex

print str(besta+1) + " " + str(bestb+1)
