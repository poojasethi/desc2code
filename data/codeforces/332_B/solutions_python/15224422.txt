n,k = map(int,raw_input().split(" "))
z = map(int,raw_input().split(" "))

ham = [0]
egg = 0

for zzz in z:
    egg += zzz
    ham.append(egg)

spam = []
python = []
boo = -10000
moo = -10000
q = 0
for i in range(n-k+1):
    poo  = ham[i+k] - ham[i]
    if (poo > boo):
        boo = poo
        q = i
    spam.append(boo)
    python.append(q)
    if i >= k:
        if (spam[i-k] + poo ) > moo:
            moo = spam[i-k] + poo
            ans = (python[i-k]+1,i+1)
print ans[0],ans[1]
