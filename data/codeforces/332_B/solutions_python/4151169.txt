def main():
    n, k = map(int, raw_input().split())
    x = map(int, raw_input().split())
    s = [0]
    for e in x:
        s.append(s[-1] + e)
    a, b = 0, k
    maxa, maxb = 0, k
    def f(t): return s[t+k]-s[t]
    while b <= n-k:
        if f(b-k) > f(a):
            a = b-k
        if f(a) + f(b) > f(maxa) + f(maxb):
            maxa, maxb = a, b
        b += 1
    print maxa+1, maxb+1

if __name__=='__main__':
    main()
