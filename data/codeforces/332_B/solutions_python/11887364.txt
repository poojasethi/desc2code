n,k = map(int,raw_input().split())
a = map(int,raw_input().split())
sums = [0]*n; Sums = []; left = 0; right = k; c = 0
for i in xrange(n):
    sums[i] += sums[i-1] + a[i]
sums = [0] + sums
for i in xrange(k,n+1):
    Sums.append(sums[i] - sums[i-k])

for i in xrange(k,n-k+1):
    if Sums[c] < Sums[i-k]:
        c = i - k
    if Sums[left] + Sums[right] < Sums[c] + Sums[i]:
        left,right = c,i

print left+1,right+1
