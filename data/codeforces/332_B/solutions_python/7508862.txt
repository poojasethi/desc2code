#!/usr/bin/python2
# -*- coding : utf-8 -*-

def main():
    n, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    s = [0] * (n + 1)
    mx = [-1] * (n + 1)
    pos = [-1] * (n + 1)
    mxSum = 0
    ans = [-1, -1]
    for i in xrange(n - 1, -1, -1):
        s[i] = s[i + 1] + a[i]

    for i in xrange(n):
        if i == 0:
            s[i] = a[i]
        else:
            s[i] = s[i - 1] + a[i]

        if i >= k - 1:
            if s[i] - s[i - k] > mx[i - 1]:
                mx[i] = s[i] - s[i - k]
                pos[i] = i - k + 1
            else:
                mx[i] = mx[i - 1]
                pos[i] = pos[i - 1]

        if i >= 2 * k - 1:
            if s[i] - s[i - k] + mx[i - k] > mxSum:
                mxSum = s[i] - s[i - k] + mx[i - k]
                ans[0] = pos[i - k]
                ans[1] = i - k + 1

    print ans[0] + 1, ans[1] + 1

if __name__ == '__main__':
    main()
