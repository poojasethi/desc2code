#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int mo=1000000007;
int n,b[120],ans,i,pow10[12],x,p,j,k;

int C(int n,int k){
	int res=1,i;
	for(i=n-k+1;i<=n;++i)res*=i;
	for(i=1;i<=k;++i)res/=i;
	return res;
}
int pow(int x,int y){
	int res=1;
	while(y){
		if(y&1)res=res*x;
		x=x*x;
		y>>=1;
	}
	return res;
}
void dfs(int dep,int limit,int s){
	if(dep==7){
		ans=(ans+s)%mo;
		return;
	}
	for(int i=0;i<limit;++i){
		b[i]--;
		dfs(dep+1,limit-i,(long long)s*(b[i]+1)%mo);
		b[i]++;
	}
}
int main(){
	scanf("%d",&n);
	pow10[0]=1;
	for(i=1;i<=9;++i)pow10[i]=pow10[i-1]*10;
	x=0;
	for(i=9;i>=0;--i){
		p=n/pow10[i]%10;
		for(j=0;j<p;++j){
			if(j==4 || j==7)x++;
			for(k=0;k<=i;++k)b[x+k]+=C(i,k)*pow(2,k)*pow(8,i-k);
			if(j==4 || j==7)x--;
		}
		if(p==4 || p==7)x++;
	}
	b[x]++;
	b[0]--;
	for(i=1;i<=10;++i){
		dfs(1,i,b[i]);
	}
	printf("%d\n",ans);
}
