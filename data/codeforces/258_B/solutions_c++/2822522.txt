#include <algorithm>
#include <cstdio>
#include <cstring>
#define mod 1000000007
using namespace std;
typedef long long LL;
LL dp[11];
char a[11];
int len;
LL ans;

void dfs(int dep,int sum,int tot){
  if(sum>=len) return;
  if(dep == 6){
    for(int i=sum+1;i<=len;++i) ans = (ans+tot*dp[i])%mod;
    return;
  }
  for(int i=0;i+sum<=len;++i){
    dp[i]--;
    dfs(dep+1,sum+i,(tot*(dp[i]+1))%mod);
    dp[i]++;
  }
}

void add(int dep,int dig,int num){
  if(dep==0){
    dp[dig]+=num;
    return;
  }
  add(dep-1,dig+1,num*2);
  add(dep-1,dig,num*8);
}

int main(){
  scanf("%s",a);
  len = strlen(a);
  int t = 0;
  for(int i=0;i<len;++i){
    for(int j=0;j<a[i]-'0';++j){
      if(j==4 || j == 7) add(len-i-1,t+1,1);
      else add(len-i-1,t,1);
    }
    if(a[i]=='4' || a[i]=='7') ++t;
  }
  dp[0]--;dp[t]++;
  ans = 0;
  dfs(0,0,1);
  printf("%lld\n",ans);
  return 0;
}
