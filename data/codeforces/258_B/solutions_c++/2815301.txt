#include<stdio.h>

long long mod(long long a){
	return a%1000000007;
}
long long mod2(long long a){
	return mod(mod(a)+1000000007);
}

long long c[15][15];
long long cnt[15];
long long p2[15],p8[15];
char n[15];
int len;

bool good(char c){
	return c=='4' || c=='7';
}

long long go(int i,int g){
	if(i==6)
		return cnt[g];
	long long res=0;
	for(int j=0;j<=g;j++){
		cnt[j]--;
		res=mod(res+(cnt[j]+1)*go(i+1,g-j));
		cnt[j]++;
	}
	return res;
}

int main(){
	for(int i=0;i<13;i++)
		c[i][0]=c[i][i]=1;
	for(int i=2;i<13;i++)
		for(int j=1;j<i;j++)
			c[i][j]=c[i-1][j]+c[i-1][j-1];
	p2[0]=p8[0]=1;
	for(int i=1;i<13;i++)
		p2[i]=p2[i-1]*2;
	for(int i=1;i<13;i++)
		p8[i]=p8[i-1]*8;

	gets(n);
	while(n[len])
		len++;
	int g=0;
	for(int i=0;i<len;i++){
		int lim=len-i-1;
		for(int j=0;j<=lim;j++)
			for(char k='0';k<n[i];k++)
				cnt[g+j+good(k)]+=c[lim][j]*p2[j]*p8[lim-j];
		g+=good(n[i]);
	}
	cnt[g]++;
	cnt[0]--;
	long long ans=0;
	for(int i=1;i<=len;i++)
		for(int j=0;j<i;j++)
			ans=mod(ans+go(1,j)*cnt[i]);
	printf("%d\n",(int)ans);
}
