#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
using namespace std;
#define pb push_back
typedef long long ll;
const int mo=1e9+7;
int n,m;
int cnt[30];
int f[11][11];
vector<int> a;
ll ans;
int get(int pos,int sum,int e){
    if (sum>m) return 0;
    if (pos<0) return sum==m;
    if (!e&&f[pos][sum]!=-1) return f[pos][sum];
    int ans=0,end=e?a[pos]:9;
    for (int i=0;i<=end;i++)
        ans+=get(pos-1,sum+(i==4||i==7),e&&i==end);
    if (!e) f[pos][sum]=ans;
    return ans;
}
void dfs(int dep,int sum,ll tot){
    if (sum>=10||cnt[sum+1]==0) return ;
    if (dep>=7){
        for (int i=sum+1;i<=10;i++)
        ans=(ans+tot*cnt[i]%mo)%mo;
        return;
    }
    for (int i=0;i<=10;i++){
        cnt[i]--;
        dfs(dep+1,sum+i,tot*(cnt[i]+1)%mo);
        cnt[i]++;
    }
}
int main(){
    //freopen("input.txt","r",stdin);
    cin>>n;
    for (;n;n=n/10)a.pb(n%10);
    for (int i=0;i<=10;i++){
        memset(f,-1,sizeof(f));
        m=i; cnt[i]=get(a.size()-1,0,1)-(i==0);
    }

    dfs(1,0,1);
    cout<<ans<<endl;
}
