#include <cstdio>
#include <cstring>
#include <algorithm>
#define FOR(i,s,e) for (int i=(s); i<(e); i++)
using namespace std;

int m, t, cnt[15];
char s[15];
long long ret, MOD = 1000000007;

void add(int x, int y, int S){
	if (!x) cnt[y] += S;
	else add(x-1, y, S*8), add(x-1, y+1, S*2);
}

void dfs(int x, int S, long long C){
	if (S > 9) return;
	if (x == 6){
		FOR(i,S+1,11) ret = (ret + C * cnt[i]) % MOD;
	}
	else FOR(i,0,11) if (cnt[i]){
		cnt[i]--;
		dfs(x + 1, S + i, C * (cnt[i] + 1) % MOD);
		cnt[i]++;
	}
}

int main(){
	scanf("%s", s);
	m = strlen(s);
	FOR(i,0,m){
		FOR(j,0,s[i]-'0') add(m - i - 1, t + (j == 4 || j == 7), 1);
		t += s[i] == '4' || s[i] == '7';
	}
	++cnt[t], --cnt[0];
	
	dfs(0, 0, 1);
	printf("%I64d\n", ret);
	return 0;
}
