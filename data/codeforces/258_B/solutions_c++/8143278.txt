long long a[10][10],p=1000000007;
int l(int q,int w){
  if(q==0)
   return 1;
  long long e,r=0;
  for(e=w;e>=0;e--){
   a[0][e]--;
   r=(r+(a[0][e]+1)*l(q-1,w-e))%p;
   a[0][e]++;}
  return r;
}
