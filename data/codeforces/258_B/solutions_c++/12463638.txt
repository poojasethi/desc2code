#include<stdio.h>
#include<cstring>
#define MOD 1000000007
__int64 ans,dp[11],Dp[11][11];
void dfs(int shu,int l,__int64 tans)
{
	if(shu==6)
	{
		ans=(ans+tans)%MOD;
		return ;
	}
	for(int i=0;i<l;i++)
	if(dp[i])
	{
		dp[i]--;
		dfs(shu+1,l-i,(tans*(dp[i]+1))%MOD);
		dp[i]++;
	}
}
int main()
{
	for(int i=0;i<10;i++)
	for(int j=0;j<=i;j++)
	Dp[i][j]=i&&j?Dp[i-1][j]+Dp[i-1][j-1]:1;
	__int64 m,len;
	while(~scanf("%I64d",&m))
	{
		m++;
		int shu[33],n=0,cut=0;
		while(m)
		{
			shu[++n]=m%10;
			m/=10;
		}
		memset(dp,0,sizeof(dp));
		dp[0]--;
		for(int i=n;i>0;i--)
		{
			for(int j=0;j<shu[i];j++)
			{
				len=(__int64)1<<(i-1);
				for(int k=i-1;k>=0;k--)
				{
					dp[cut+k+(j==4||j==7)]=(dp[cut+k+(j==4||j==7)]+Dp[i-1][k]*len)%MOD;
					len<<=2;
				}
			}
			cut+=(shu[i]==4||shu[i]==7);
		}
		ans=0;
		for(int i=0;i<10;i++)
		if(dp[i])
		{
			dp[i]--;
			dfs(0,i,dp[i]+1);
			dp[i]++;
		}
		printf("%I64d\n",ans);
	}
}
