#include <iostream>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)
#define MOD      1000000007

using namespace std;

int n,t,s,k,a[20],lim;
long long c[110][100],f[20],res;

long long dfs(int d, int w){
  long long res = 0;
  if(d>=6)return 1;
  rep(i,lim-w){
    f[i]--; res =(res + (f[i]+1)*dfs(d+1,w+i)%MOD)%MOD;  f[i]++;
  }
  return res;
}

int main(){
  cin>>n; t = n;
  while(t){  a[k] = t%10; k++;t/=10; }

  rep(i,100)rep(j,i+1){
    c[i][j] = (i&&j)?(c[i-1][j-1]+c[i-1][j])%MOD:1;
  }

  for(int i=k-1;i>=0;i--){
    rep(j,a[i])rep(l,i+1){
      f[l+s+(j==4||j==7)] += c[i][l] * (1<<(3*i-2*l));
    }
    s+=(a[i]==4 || a[i]==7);
  }
  f[s]++;f[0]--;

  for(int i=1;i<=k;i++){
    lim = i;
    res = (res+dfs(0,0)*f[i])%MOD;
  }

  cout<<res<<endl;


  return 0;

}
