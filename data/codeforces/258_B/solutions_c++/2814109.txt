#include<cstdio>
#define ll long long
#define oo 1000000007
int m,n,cnt,a[15];
ll ans,c[15],f[18][18];
void dfs(int t,int x,int y,ll s)
{
	if (t>5) {ans=(ans+s)%oo; return;}
	for (int i=0;i+y<x;i++)
		c[i]--,dfs(t+1,x,y+i,s*(c[i]+1)%oo),c[i]++;
}
int main()
{
	scanf("%d",&m),m++,c[0]--; while (m) a[++n]=m%10,m/=10;
	for (int i=0;i<=15;i++) for (int j=0;j<=i;j++)
		f[i][j]=i&&j?(f[i-1][j-1]+f[i-1][j])%oo:1;
	for (int i=n;i;i--){
		for (int j=0;j<a[i];j++){
			ll now=1LL<<i-1;
			for (int k=i-1;k>=0;k--)
				c[cnt+(j==4 || j==7)+k]+=now*f[i-1][k],now<<=2;
			}
		cnt+=(a[i]==4 || a[i]==7);
		}
//	for (int i=0;i<=9;i++) printf("%d\n",c[i]);
	for (int i=1;i<=9;i++) if (c[i]) c[i]--,dfs(0,i,0,c[i]+1),c[i]++;
	printf("%d\n",ans); return 0;
}
