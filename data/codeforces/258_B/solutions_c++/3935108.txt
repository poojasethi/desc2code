#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
#define int64 long long
#define MOD 1000000007
#define foru(i,a,b) for(int i=a;i<=b;++i)
#define ford(i,a,b) for(int i=a;i>=b;--i)

using namespace std;
int64 f[21][21],cnt[21],ans;
int b[21],len,tot,n;
void dp(int a)
{
	for(len=0;a>0;++len,a/=10) b[len+1]=a%10;
	f[0][0]=1;
	foru(i,0,len-2)
	  foru(j,0,i)
	    f[i+1][j+1]+=f[i][j]*2,f[i+1][j+1]%=MOD,
		f[i+1][j]+=f[i][j]*8,f[i+1][j]%=MOD;
	tot=0;
	ford(i,len,1)
	{
		foru(j,0,b[i]-1)
		  foru(ii,0,i-1)
		    cnt[ii+tot+(j==4||j==7)]+=f[i-1][ii];
	    tot+=(b[i]==4||b[i]==7);
	}
	--cnt[0],++cnt[tot];
}
int64 dfs(int a,int b)
{
	if(a==7) return 1;
	int64 res=0;
	foru(i,0,b)
	  if(cnt[i]) --cnt[i],res+=(cnt[i]+1)*dfs(a+1,b-i),res%=MOD,++cnt[i];
	return res;
}
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>n; dp(n);
	//foru(i,0,len) cout<<i<<' '<<cnt[i]<<endl;
	foru(i,1,len)
	  if(cnt[i])
	    --cnt[i],ans+=(cnt[i]+1)*dfs(1,i-1),ans%=MOD,++cnt[i];
	cout<<ans<<endl;
	return 0;
}
