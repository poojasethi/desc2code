#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

#define mod 1000000007LL

int n,wn,w[110],i,j,k,q;
long long F[110][110][10],kol[110],ret,cur;

void get(){
	int m=n;
	while(m){
		w[++wn]=m%10;
		m/=10;
	}
	reverse(w+1,w+wn+1);
	F[0][0][1]=1;
	for(i=0;i<wn;i++)for(j=0;j<=10;j++)for(k=0;k<2;k++)if(F[i][j][k]){
		if(k==0)for(q=0;q<10;q++)F[i+1][j+(q==4||q==7)][k]+=F[i][j][k];
		if(k==1){
			for(q=0;q<w[i+1];q++)F[i+1][j+(q==4||q==7)][0]+=F[i][j][k];
			F[i+1][j+(w[i+1]==4||w[i+1]==7)][1]+=F[i][j][k];
		}
	}
	for(i=0;i<=10;i++)kol[i]=F[wn][i][0]+F[wn][i][1];
	--kol[0];
}

void rec(int remain,int left){
	if(remain==0){
		ret=(ret+cur)%mod;
		return ;
	}
	int _cur=cur,i;
	for(i=0;i<left;i++)if(kol[i]){
		cur=(cur*1LL*kol[i])%mod;
		--kol[i];
		rec(remain-1,left-i);
		++kol[i];
		cur=_cur;
	}
}

int main (int argc, char * const argv[]) {
	cin>>n;
	get();
	cur=1;
	for(i=1;i<=10;i++)if(kol[i]){
		cur=(1LL*cur*kol[i])%mod;
		--kol[i];
		rec(6,i);
		++kol[i];
		cur=1;
	}
	cout<<ret<<endl;
    return 0;
}
