#include <cstdio>
#include <cstring>
#include <algorithm>
#define M 1000000007
using namespace std;
typedef long long LL;

LL n, m, a[20], cnt[20], f[20][20], ans, c;

void dfs (LL p, LL tt, LL sum){
	if(p == 6){
		for(LL i = tt+1; i <= n; i++)
			ans = (ans + cnt[i] * sum) % M;
		return;
	}
	for(LL i = 0; i <= n; i++){
		if(cnt[i] == 0) continue;
		cnt[i] --;
		dfs (p+1, tt+i, (sum * (cnt[i]+1)) % M);
		cnt[i] ++;
	}
}

int main (){
	scanf("%I64d", &m), m++, cnt[0]--;
	while(m) a[++n] = m % 10, m /= 10;
	for (LL i = 0; i <= 15; i++) for (LL j = 0; j <= i; j++)
			f[i][j] = i && j? (f[i-1][j-1]+f[i-1][j]) % M: 1;
	for (int i = n; i; i--){
		for (int j = 0; j < a[i]; j++){
			LL now = 1LL << i-1;
			for (int k = i-1; k >= 0; k--){
				cnt[c+(j == 4 || j == 7)+k] += now * f[i-1][k];
				now <<= 2;
			}
		}
		c += (a[i] == 4 || a[i] == 7);
	}
	//for (int i = 0; i <= 9; i++) printf("%I64d\n",cnt[i]);
	dfs(0, 0, 1);
	printf("%I64d\n", ans);
}
