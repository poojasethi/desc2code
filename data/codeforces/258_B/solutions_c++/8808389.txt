#include<cstdio>
#include<cstdlib>
#define mo 1000000007
int f[20],g[20],ans,top,n;
void work(int n){
	int val=0;
	for(int d=1000000000;d>=1;d/=10){
		int p=n/d;
		
//		printf("work d %d p %d\n",d,p);
		
		for(int k=9;k>=0;k--) g[k]=0;
		
		for(int k=9;k>=0;k--) if(f[k])
		for(int j=0;j<=9;j++)
		g[k+((j==4)||(j==7))]+=f[k];
		
		for(int j=0;j<p;j++) g[val+((j==4)||(j==7))]++;
		
		for(int k=9;k>=0;k--) f[k]=g[k];
		
		val+=((p==4)||(p==7));
		
		n%=d;
//		for(int k=0;k<=9;k++) printf("%d ",f[k]); printf("fff\n");
	}
	f[val]++; f[0]--;
//	for(int i=0;i<=9;i++) printf("%d ",f[i]); printf("ff\n");
}
void dfs(int x,int limit,int sum){
	if(x>6){ans=(ans+1LL*sum*f[top])%mo; return;}
	for(int i=0;i<limit;i++)
	if(f[i]){
		dfs(x+1,limit-i,1LL*sum*(f[i]--)%mo);
		f[i]++;
	}
}
int main(){
	scanf("%d",&n);
	
	work(n);
	
	for(top=9;top>=1;top--)
	if(f[top]) dfs(1,top,1);
	printf("%d\n",ans);
}
