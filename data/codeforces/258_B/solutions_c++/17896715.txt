#include <iostream>
#include <cstdio>
#include <cstring>
#define ll long long
const int mod = 1e9+7;
int dp[20][12][12];
int du[12];
int data[12];
int dfs(int len,int num_4,int num_7,int sum,int limit){
    if(!len)return num_4+num_7==sum;
    if(!limit&&dp[len][num_4][num_7]!=-1)return dp[len][num_4][num_7];
    int ed=limit?data[len]:9;
    int ans=0;
    for(int i=0;i<=ed;++i)
        ans+=dfs(len-1,num_4+(i==4?1:0),num_7+(i==7?1:0),sum,limit&&i==ed);
    return limit?ans:dp[len][num_4][num_7]=ans;
}
ll DFS(ll now,ll res){
    ll ans=0;
    if(now==7) return 1;
    for(int i=0;i<=res;++i)
        if(du[i]>0){
            du[i]--;
            ans=(ans+(du[i]+1)*DFS(now+1,res-i))%mod;
            du[i]++;
        }
    return ans;
}
void doit(int n){
    int len=0;
    while(n){
        data[++len]=n%10;
        n/=10;
    }
    for(int i=0;i<=9;++i){
        memset(dp,-1,sizeof(dp));
        du[i]=dfs(len,0,0,i,1);
    }
    du[0]--;
    ll ans=0;
    for(int i=1;i<=9;++i)
        ans=(ans+du[i]*DFS(1,i-1))%mod;
    printf("%I64d\n",ans);
}
int main(){
    int n;
    while(~scanf("%d",&n))doit(n);
    return 0;
}
