#include <vector>
#include <iostream>

using namespace std;

int main() {
	int n, k;
	cin >> n;
	cin >> k;

	if (k*3>n || k==1) {
		cout << -1;
		return 0;
	}

	for (int i=0; i<n; ++i) {
		if (i<k*2)
			cout << i/2 +1 << ' ';
		else if (i<k*3)
			cout << i%k +1 << ' ';
		else
			cout << 1 << ' ';
	}

}
