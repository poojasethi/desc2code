import sys
n, k = map(int, raw_input().split(' '))

if k*3 > n:
  print(-1)
  exit(0)

if k % 2 == 0:
  for i in xrange(0, k/2):
    x = i*2 + 1
    y = x + 1
    print("%i %i %i %i %i %i " % ( x, x, y, x, y, y))
 
  for i in xrange(k*3, n): print(1)

else:
  print("1 1 2 2 3 3 1 2 3 ")
  for i in xrange(0, k/2 - 1):
    x = i*2 + 1 + 3
    y = x + 1 
    print("%i %i %i %i %i %i " % (x, x, y, x, y, y) )
 
  for i in xrange(k*3, n): print(1)
