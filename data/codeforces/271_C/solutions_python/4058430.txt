import sys

def calc(n, k):
    if n < k * 3:
        return []
    return range(1, k + 1) + [i / 2 + 1 for i in range(k * 2)] + [1] * (n - k * 3)
    
def solve(lines, output):
    n, k = (int(i) for i in lines.readline().split())
    a = calc(n, k)
    output.write('{0}\n'.format((' '.join(str(i) for i in a) if len(a) > 0 else -1)))

if __name__ == '__main__':
    solve(sys.stdin, sys.stdout)
