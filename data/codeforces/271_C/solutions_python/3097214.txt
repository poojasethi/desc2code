n, k = map(int, raw_input().split())
if 3 * k > n:
    print -1
else:
    ans = []
    for i in range(1, k + 1):   
        ans.append(i)
    for i in range(k, 0, -1):
        ans.append(i)
    for i in range(n // k + 2):
        ans.extend(range(k, 0, -1))
    if k % 2 == 1:
        i = (k + 1) // 2
        ans[2 * k - i], ans[i - 2] = ans[i - 2], ans[2 * k - i]
    print ' '.join(map(str, ans[:n]))
