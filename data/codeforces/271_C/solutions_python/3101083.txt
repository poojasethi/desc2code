n,k = [int(x) for x in raw_input().split()]

ret = []

if n < 3*k:
    print -1
else:
    for i in xrange(n):
        j = i+1
        if j <= 2*k:
            ret.append(j%k+1)
        elif j <= 3*k:
            ret.append((j+1)%k+1)
        else:
            ret.append(1)
    print " ".join([str(x) for x in ret])
