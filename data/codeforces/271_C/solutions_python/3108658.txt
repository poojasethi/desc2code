if n / k <= 2:
	print -1

	m = n/k - 1
	a = range(n+1)
	end1 = n - k
	end2 = n
	for i in xrange(1, k+1):
		a[end2] = i
		end2 -= 1
		for j in xrange(m):
			a[end1] = i
			end1 -= 1
	while end1 >= 1:
		a[end1] = k
		end1 -= k
