n, k = map(int, raw_input().split())
if n / k < 3:
  print -1
else:
  a = [0 for i in range(n)]
  s, d = 0, 1
  for i in range(n):
    if i % k == 0:
      d = [-1, 1][s == 0]
      s = [k + 1, 0][s == 0]
    s += d
    a[i] = s
  a[k / 2], a[k / 2 + 1] = a[k / 2 + 1], a[k / 2]
  print " ".join(map(str, a))
