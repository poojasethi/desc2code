import sys

n, k = [int(i) for i in sys.stdin.read().strip().split(' ')]
if n/k < 3:
    sys.stdout.write('-1')
else:
    res = []
    for i in range(1, n + 30, 2):
        res += [i, i, i+1, i, i+1, i+1]
    res = [(i if i <= k else (i%k + 1)) for i in res[0:n]]
    if n == k*3 and n%2 == 1:
        m = max(res[-9:])
        res[-9:] = [m-2, m-1, m-2, m-1, m, m, m-2, m-1, m]
    sys.stdout.write(' '.join([str(i) for i in res]))
