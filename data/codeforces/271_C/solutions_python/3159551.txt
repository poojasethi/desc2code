n,k = map(int, raw_input().split())
if __name__ == "__main__":
    if n < 3*k: print(-1)
    else:
        L1 = list(range(1,k+1))
        for i in range(1,n+1): 
            if i <= k: print(L1[i-1])
            else: 
                if i <= 3*k: print(L1[int((i-k-1)/2)]) 
                else: print(1)
