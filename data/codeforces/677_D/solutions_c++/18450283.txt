#include<stdio.h>
#include<string.h>
#include<iostream>
#include<vector>
#include<queue>
#include<math.h>
#include<set>
#include<stdlib.h>
#include<map>
#include<string>
#include<algorithm>
#define ll  __int64
#define inf 1e8
using namespace std;
vector< pair<int,int> > v[90010];
int dis1[310][310],dis2[310][310];
int fg[310][310];
int main()
{
    int n,m,p;
    while(scanf("%d%d%d",&n,&m,&p)!=EOF)
    {
        for(int i=0;i<=n*m;i++) v[i].clear();
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=m;j++)
            {
                int x;
                scanf("%d",&x);
                v[x].push_back(make_pair(i,j));
                dis1[i][j]=inf;
                dis2[i][j]=0;
                fg[i][j]=0;
            }
        }
        v[0].push_back(make_pair(1,1));
        for(int i=0;i<=p;i++)
        {
            int len=v[i].size();
            for(int j=0;j<len;j++)
            {
                int r=v[i][j].first,c=v[i][j].second;
                if(r==1&&c==1&&dis1[r][c]==0) dis1[r][c]=inf;
                for(int k=1;k<=n;k++)
                {
                    if(fg[k][c]==i)
                    dis1[r][c]=min(dis1[r][c],dis2[k][c]+abs(k-r));
                }
            }
            for(int j=0;j<len;j++)
            {
                int r=v[i][j].first,c=v[i][j].second;
                for(int k=1;k<=m;k++)
                {
                    if(fg[r][k]!=i+1)
                    {
                        fg[r][k]=i+1;
                        dis2[r][k]=dis1[r][c]+abs(k-c);
                    }
                    else dis2[r][k]=min(dis2[r][k],dis1[r][c]+abs(k-c));
                }
            }
        }
        printf("%d\n",dis1[v[p][0].first][v[p][0].second]);
    }
    return 0;
}
