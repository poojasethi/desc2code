#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<string.h>
#include<algorithm>
#include<cmath>
using namespace std;

int n,m,p,aa,s[90109],a[303][303],ddd[303][303],i,k,j,c,d,l,v[303][303],top;

struct edge{
	int cc,dd,next;
}e[90109];

void add(int bb,int cc,int dd){
	e[top].cc=cc; e[top].dd=dd; e[top].next=s[bb]; s[bb]=top++;
}

int main(){
    //freopen("1.in","r",stdin);
    //freopen("1.out","w",stdout);
	scanf("%d%d%d",&n,&m,&p);
	memset(s,-1,sizeof(s));
	memset(a,63,sizeof(a));
	top=0;
	for(i=1;i<=n;i++)
	for(j=1;j<=m;j++){
	scanf("%d",&aa);
	add(aa,i,j);
    }
    add(0,1,1);
    for(int i=0;i<=p;i++)  
    {
    	for(j=s[i];j!=-1;j=e[j].next){
    		c=e[j].cc;d=e[j].dd;
    		if(c==1&&d==1&&a[c][d]==0)a[c][d]=1000000000;
    		for(k=1;k<=n;k++)
    		if(v[k][d]==i)a[c][d]=min(a[c][d],ddd[k][d]+abs(k-c));
		}
		for(j=s[i];j!=-1;j=e[j].next){
    		c=e[j].cc;d=e[j].dd;
    		for(k=1;k<=m;k++)
    		if(v[c][k]!=i+1){
    			v[c][k]=i+1;
    			ddd[c][k]=a[c][d]+abs(k-d);
    		}else ddd[c][k]=min(ddd[c][k],a[c][d]+abs(k-d));
    	}
		}
	printf("%d\n",a[e[s[p]].cc][e[s[p]].dd]);
	return 0;
}
