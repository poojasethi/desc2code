#include <bits/stdc++.h>
#define maxint 1000000007

using namespace std;

int n, m, p;
vector<pair<int, pair<int, int> > > a[123456];

int kc(pair<int, int> a, pair<int, int> b){
    return abs(a.first - b.first) + abs(a.second - b.second);
}

int main()
{
    ios_base::sync_with_stdio(false);
    //freopen("inp.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    cin >> n >> m >> p;
    int c;
    for(int i = 1; i <= n; i++)
        for(int j = 1; j <= m; j++){
            cin >> c;
            if(c != 1) a[c].push_back(make_pair(maxint, make_pair(i, j)));
                else a[c].push_back(make_pair(i - 2 + j, make_pair(i, j)));
        }
    for(int i = 1; i < p; i++){
        sort(a[i].begin(), a[i].end());
        for(int j = 0; j < min(int(a[i].size()), n + m + 1); j++)
            for(int k = 0; k < a[i+1].size(); k++){
                a[i+1][k].first = min(a[i+1][k].first, a[i][j].first + kc(a[i][j].second, a[i+1][k].second));
            }
    }
    cout << a[p][0].first;
    return 0;
}
