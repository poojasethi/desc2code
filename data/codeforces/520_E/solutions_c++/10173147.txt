#include<cstdio>
#include<iostream>
#define MAXN 100000
using namespace std;
const int mod=1000000007;
int sum[MAXN+10],f[MAXN+10],g[MAXN+10],A[MAXN+10];
char s[MAXN+10];
int powmod(int a,int n)
{
	int b=a%mod,ans=1;
	while(n)
	{
		if(n%2==1) ans=(1LL*ans*b)%mod;
		b=(1LL*b*b)%mod;
		n/=2;
	}
	return ans;
}
void prepare(int n)
{
	int i;
	for(i=1;i<=n;i++)
		A[i]=s[i-1]-'0';
	for(i=1;i<=n;i++)
		sum[i]=sum[i-1]+A[i];
	f[0]=1;
	for(i=1;i<=n;i++)
		f[i]=(1LL*f[i-1]*i)%mod;
	g[n]=powmod(f[n],mod-2);
	for(i=n-1;i>=0;i--)
		g[i]=(1LL*g[i+1]*(i+1))%mod;
}
int C(int n,int k)
{
	return 1LL*(1LL*f[n]*g[k]%mod)*g[n-k]%mod;
}
int main()
{
	int i,n,k;
	int ans=0;
	scanf("%d%d",&n,&k);
	scanf("%s",s);
	prepare(n);
	int p=1;
	for(i=1;i<=n-k;i++)
	{
		ans=(ans+1LL*(1LL*sum[n-i]*p%mod)*C(n-1-i,k-1)%mod)%mod;
		ans=(ans+1LL*A[n-i+1]*p*C(n-i,k)%mod)%mod;
		p=1LL*p*10%mod;
	}
	printf("%d\n",ans);
}
