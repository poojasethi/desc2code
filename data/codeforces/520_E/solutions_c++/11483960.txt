#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<algorithm>
#include<iostream>
#define maxn 100000+5
#define ll long long
#define for0(i,n) for(int i=0;i<=(n);i++)
#define for1(i,n) for(int i=1;i<=(n);i++)
#define mod 1000000007
using namespace std;
int n,k,fac[maxn],inv[maxn],p[maxn],s[maxn];
inline int c(int x,int y)
{
    return x<y?0:(ll)fac[x]*inv[y]%mod*inv[x-y]%mod;
}
int main()
{
    scanf("%d",&n);scanf("%d",&k);
    fac[0]=1;
    for1(i,n)fac[i]=(ll)fac[i-1]*i%mod;
    inv[0]=inv[1]=1;
    for1(i,n)inv[i]=(ll)inv[i-mod%i]*(mod/i+1)%mod;
    for1(i,n)inv[i]=(ll)inv[i]*inv[i-1]%mod;
    p[0]=1;
    for1(i,n)p[i]=(ll)p[i-1]*10%mod;
    for1(i,n)
    {
        char ch=getchar();
        while(ch<'0'||ch>'9')ch=getchar();
        s[i]=s[i-1]+ch-'0';
    }
    int ans=0;
    for1(i,n)(ans+=(ll)p[n-i]*(s[i]-s[i-1])%mod*c(i-1,k)%mod)%=mod; 
    for0(i,n-1)(ans+=(ll)p[i]*c(n-1-i-1,k-1)%mod*s[n-i-1]%mod)%=mod;
    cout<<ans<<endl;
    return 0;
}
