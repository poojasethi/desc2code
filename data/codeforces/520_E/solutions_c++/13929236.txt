#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

#define mod 1000000007
int n,k;
char s[100005];
LL fact[100005],inv[100005],p10[100005];
LL ans=0,cur=0;
LL po(LL a,LL b){
    if(b==0) return 1;
    if(b==1) return a%mod;
    LL cur=po(a,b>>1);
    cur=cur*cur%mod;
    return cur*po(a,b&1)%mod;
}
LL C(int a,int b){
    if(a<b || a<0 || b<0) return 0;
    else return fact[a]*inv[b]%mod*inv[a-b]%mod;
}
int main(){
    fact[0]=inv[0]=p10[0]=1;
    FI(i,1,100000) fact[i]=(fact[i-1]*i)%mod;
    FI(i,1,100000) inv[i]=po(fact[i],mod-2)%mod;
    FI(i,1,100000) p10[i]=p10[i-1]*10%mod;
    scanf("%d %d\n%s",&n,&k,s);
    //prec[i] = 10^i * C(n-2-i,k-1)
    FD(i,n-1,0){
        ans=(ans+(cur+p10[n-1-i]*C(i,k))*(s[i]-'0'))%mod;
        cur=(cur+p10[n-1-i]*C(i-1,k-1))%mod;
    }
    printf("%lld\n",ans);
    return 0;
}
