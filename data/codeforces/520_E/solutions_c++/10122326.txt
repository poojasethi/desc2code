#include <cstdio>
#define LL long long
#define N 100005
#define M 1000000007
#define fi(a, b, c) for(int a = (b); a < (c); a++)
#define FD(a, b, c) for(int a = (b); a >= (c); a--)

int n, k;
char s[N];
LL t, f, c, c2, a;

LL inv(int x){
	LL ret = 1, mul = x;
	for(int i = M - 2; i; i >>= 1){
		if(i & 1) ret = ret * mul % M;
		mul = mul * mul % M;
	}
	return ret;
}

int main(){
	scanf("%d %d %s", &n, &k, s);
	if(!k){
		fi(i, 0, n) a = (a * 10 + s[i] - '0') % M;
		printf("%I64d\n", a);
		return 0;
	}
	
	t = c = c2 = 1;
	fi(i, 0, k) c = c * (n - 1 - i) % M * inv(i + 1) % M;
	fi(i, 0, k - 1) c2 = c2 * (n - 2 - i) % M * inv(i + 1) % M;
	
	FD(i, n - 1, 0){
		a = (a + (f + c * t % M) * (s[i] - '0')) % M;
		f = (f + c2 * t) % M;
		c = c * inv(i) % M * (i - k) % M;
		t = t * 10 % M;
		c2 = c2 * inv(i - 1) % M * (i - k) % M;
	}
	
	printf("%I64d\n", a);
}
