#include<bits/stdc++.h>
using namespace std;
#define NN 100010
typedef long long INT;
const int P=1e9+7;
int paw(int a,int k){
	if(!k) return 1;
	INT ans=paw(a,k/2);
	ans=ans*ans%P;
	if(k&1) ans=ans*a%P;
	return ans;
}
char s[NN];
INT a[NN],rst[NN],r,S,jin,fac[NN];
int main(){
//	freopen("e.in","r",stdin);
	int n,k,i,j;
	rst[0]=1;
	for(i=1;i<NN;i++) rst[i]=rst[i-1]*10%P;
	fac[0]=1;
	for(i=1;i<NN;i++) fac[i]=fac[i-1]*i%P;
	scanf("%d%d",&n,&k);
	scanf("%s",s);
	for(i=0;i<n;i++) a[i]=s[i]-'0';
	r=1;
	for(i=k;i<n;i++){
		jin+=r*a[i]%P*rst[n-i-1]%P;
		jin%=P;
		r=r*(i+1)%P*paw(i+1-k,P-2)%P;
	}
	if(!k) {printf("%I64d\n",jin);return 0;}
	S=fac[n-2]*paw(fac[k-1],P-2)%P*paw(fac[n-k-1],P-2)%P;
	r=S;
	for(i=n-2;i>=0;i--){
		jin+=a[i]*S%P;
		jin%=P;
		if(i<k) r=0;
		else r=r*10%P*(i-k+1)%P*paw(i,P-2)%P;
		S=(S+r)%P;
	}
	printf("%I64d\n",jin);
}
