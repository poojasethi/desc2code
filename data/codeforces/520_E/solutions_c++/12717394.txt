#include<iostream>
using namespace std;
#define ll long long
const ll mod=1e9+7;
const int N=100005;
ll n,pc[N],ten[N],k,pinv[N],fac[N],res=0; string s;

ll power(ll s, ll p){
	if(p==0)return 1;
	if(p==1)return s;
	ll ans=1;
	if(p%2==1)ans*=s;
	ans=(ans*power((s*s)%mod,p/2))%mod;
	return ans;
}

ll f(ll x, ll y){
	if(x<y||y<0)return 0;
	if(y==0||y==x)return 1;
	return (pinv[y]*pinv[x-y]%mod)*fac[x]%mod;
}

void init(){
	fac[0]=pinv[0]=ten[0]=1; pc[0]=0;
	for(int i=1; i<n+3; i++){fac[i]=fac[i-1]*i%mod; pinv[i]=pinv[i-1]*power((ll)i,mod-2)%mod; ten[i]=10*ten[i-1]%mod;}
	for(int i=1; i<n-1; i++)pc[i]=(pc[i-1]+f(i,k-1)*ten[n-2-i]%mod)%mod;
	return;
}

int main(){
	cin >> n >> k;
	cin >> s;
	init();
	for(int i=0; i<n; i++){
		res+=(s[i]-'0')*(pc[n-2]-pc[i-1]+f(i,k)*ten[n-1-i]%mod)%mod;
		res%=mod;
		//cout << res << "\n";
	}
	if(k==1)res+=(s[0]-'0')*ten[n-1-k]%mod;
	res%=mod;
	while(res<0)res+=mod;
	cout << res << endl;
	return 0;
}
