#include <cstdio>
#include <iostream>
#define N 100010
#define P 1000000007ll
#define LL long long
using namespace std;
int n,k,bit[N];
LL jc[N],a[N],sum[N];
LL pow(LL x,LL y){
	LL res=1;
	for(;y;x=x*x%P,y>>=1)
	    if(y&1)res=res*x%P;
	return res;
}
LL C(LL x,LL y){
	if(x<y)return 0;
	return jc[x]*pow(jc[y]*jc[x-y]%P,P-2)%P;
}
int main(){
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;++i)scanf("%1d",&bit[i]);
	if(!k){
		LL ans=0;
		for(int i=1;i<=n;++i)ans=(ans*10+bit[i])%P;
		printf("%I64d",ans);
		return 0;
	}
	jc[0]=1;
	for(int i=1;i<=n;++i)jc[i]=jc[i-1]*i%P;
	for(int i=0;i<=n-2;++i){
		a[i]=pow(10,i)*C(n-1-(i+1),k-1)%P;
		if(i)sum[i]=sum[i-1]+a[i];
		else sum[i]=a[i];
	}
	LL ans=0;
	for(int i=1;i<=n;++i)ans=(ans+bit[i]*pow(10,n-i)*C(i-1,k)%P)%P;
	for(int i=1;i<n;++i)ans=(ans+bit[i]*sum[n-i-1]%P)%P;
	printf("%I64d",ans);
}
