read = lambda:map(int,raw_input().split())
n = input()
a = read()
ans = 2;
cnt = 0;
for i in range(1,n):
    if a[i] == 0:
        cnt+=1
    else:
        ans = max(ans,cnt)
        cnt = 0;
cnt = 2
for i in range(2,n):
    if a[i-1] + a[i - 2] == a[i]:
        cnt+=1
    else:
        ans = max(ans,cnt)
        cnt = 2
ans = max(ans,cnt)
ans = min(ans,n)
print ans
