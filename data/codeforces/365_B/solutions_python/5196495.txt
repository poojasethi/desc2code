n = int(raw_input())
t = map(int, raw_input().split())

def isFib(a, b, c):
    return c==a+b

lm = 2
k = 0
if n == 1 or n == 2:
    print n
else:
    while n-3 >= k:
        l = 2
        while isFib(t[k], t[k+1], t[k+2]):
            l += 1
            k += 1
            if k == n-2:
                break
        if l > lm:
            lm = l
        k += 1
    print lm
