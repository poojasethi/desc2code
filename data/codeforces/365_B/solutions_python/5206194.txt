#!/usr/env/bin python
#coding:utf8

import sys

n = int(sys.stdin.readline())
arr = map(int, sys.stdin.readline()[:-1].split())

if n == 1:
	print 1
	exit()

maxlength = 0
length = 0

for i in range(2, n):
	if arr[i-2] + arr[i-1] == arr[i]:
		length += 1
	else:
		if length > maxlength:
			maxlength = length
		length = 0

if length > maxlength:
	maxlength = length

print maxlength + 2
