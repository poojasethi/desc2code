n = input()
A = map(int, raw_input().split())

if n <= 2 :
    print n

else :
    cur = 2
    cnt = 2
    maxCnt = 2
    for cur in range(2, n) :
        if A[cur] == A[cur-1] + A[cur-2] :
            cnt += 1
            if maxCnt < cnt :
                maxCnt = cnt
        else :
            cnt = 2

    print maxCnt
