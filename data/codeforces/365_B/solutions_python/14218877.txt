def m2(n, nums):
    if n <= 2:
        return n
    i = 0
    ml = 0
    l = 2
    while i < n - 2:
        if nums[i] + nums[i+1] == nums[i+2]:
            l += 1            
        else:            
            l = 2
        ml = max(ml, l)    
        i += 1
    return ml                   
    
n = int(raw_input())    
nums = map(int, raw_input().split())  

print m2(n, nums)
