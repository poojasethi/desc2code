n = int(raw_input())
t= map(int,raw_input().split())

lmax=1
if n>1:
    lmax=2
    a=t[0]
    b=t[1]
    i=2
    l=2
    while i<n:
        while i<n and t[i]==a+b:
            a,b=b,t[i]
            l+=1
            i+=1
            if lmax<l:
                lmax=l
        if i<n:
            l=2
            a,b=b,t[i]
            i=i+1
    
print lmax
