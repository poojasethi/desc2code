N = int(raw_input())
A = map(int,raw_input().split())
if N == 1:
    print 1
elif N == 2 :
    print 2
else:
    X = 2
    count = 2
    for i in range (2,N):
        if A[i] == A[i-1] + A[i-2] :
            count += 1
        else:
            if count > X :
                X = count
            count = 2
    if count > X :
        X = count
    print X
