n = int(raw_input())
a = map(int, raw_input().split())
good = [False, False]
for i in xrange(2, n):
	if (a[i] == a[i - 1] + a[i - 2]):
		good.append(True)
	else:
		good.append(False)
#print good
res = 0
cur = 0
for i in xrange(0, n):
	if good[i]:
		cur += 1	
	else:		
		if res < cur:
			res = cur
		cur = 0
if res < cur:
	res = cur

res = min(res + 2, n)
print res
