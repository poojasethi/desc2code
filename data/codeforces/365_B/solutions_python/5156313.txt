n=int(s())
arr=map(int,s().split())
if n == 1:
    print 1
else:
    max_till_now = 2
    max_here = 2
    for i in range(2, n):
        if arr[i] == arr[i-1] + arr[i-2]:
            max_here += 1
        else:
            max_here = 2
        if max_here > max_till_now:
            max_till_now = max_here
    print max_till_now
