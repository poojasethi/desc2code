arr = map(int, raw_input().split())
if len(arr) == 1:
    print 1
    exit(0)

lengths = [1, 2] + [2] * (len(arr)-2)

def isFib(a, b, c):
    return c == a + b

for i in xrange(2, len(arr)):
    if isFib(arr[i-2], arr[i-1], arr[i]):
        lengths[i] = lengths[i-1] + 1
    else:
        lengths[i] = 2

# print lengths

print max(lengths)
