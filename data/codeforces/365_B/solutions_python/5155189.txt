
def main():
  n = input()
  if n == 1:
    print 1
    return
  elif n ==2:
    print 2
    return
  xs = [int(i) for i in raw_input().split()]
  fs = [0,]*n
  for i in xrange(2,n):
    if xs[i]==xs[i-1]+xs[i-2]:
      if (fs[i-1]):
        fs[i] = fs[i-1] + 1
      else:
        fs[i] = 3
  print max(2,*fs)


    


main()
