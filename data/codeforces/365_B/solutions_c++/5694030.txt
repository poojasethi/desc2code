#include<iostream>
#include<cmath>
#include<cstdio>
#include<vector>
using namespace std;
int main(){
int n,a[100000]={-1},b,c=0,d,e,j,i,maxs=0;
cin>>n;
for(i=1;i<=n;i++){
cin>>a[i];
}
for(i=3;i<=n;i++){
if(a[i]==a[i-1]+a[i-2])c++;
else c=0;
if(c>maxs)maxs=c;
}
if(n==1)cout<<1;
else cout<<maxs+2;
return 0;
}
