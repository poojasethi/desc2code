#include<iostream>
using namespace std;
int a[100010];
int d[100010];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	cin>>a[i];
	d[0]=1;
	d[1]=2;
	for(int i=2;i<n;i++)
	{
		if(a[i]==a[i-1]+a[i-2])
			d[i]=d[i-1]+1;
		else
			d[i]=2;
	}
	int maxx=0;	
	for(int i=0;i<n;i++)
	if(d[i]>maxx)
		maxx=d[i];
	cout<<maxx<<endl;
}
