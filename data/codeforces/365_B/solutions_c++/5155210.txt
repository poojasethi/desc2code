#include<iostream>
#include <algorithm>
using namespace std;
long long int m,a[111111],l=0,max1=0;
int main(){
 cin>>m;
 for(int i=1;i<=m;i++){
  cin>>a[i];
 }
 for(int i=3;i<=m;i++){
  if(a[i]==(a[i-1]+a[i-2]))l++;
  else l=0;
  if(l>=max1)max1=l;
 }
 if(m==1)cout<<1;
 else 
 cout<<max1+2;
 return 0;
}
