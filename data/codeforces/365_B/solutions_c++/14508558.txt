#include <iostream>

using namespace std;

const int N=100005;
int n,ans,k,a[N];
int main(){
  cin>>n;
  for(int i=0;i<n;i++)
    cin>>a[i];
  for(int i=0;i<n-2;i++){
    if(a[i]+a[i+1]==a[i+2])
      k++;
    else{
      ans=max(ans,k);
      k=0;
    }
  }
  ans=max(ans,k);
  if(n==1)
    ans=-1;
  cout<<ans+2<<endl;
}
