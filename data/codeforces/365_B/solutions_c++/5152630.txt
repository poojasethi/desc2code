#include<iostream>
using namespace std;

int a[100010],f[100010],n;
int main(){
	cin>>n;
	for (int i=0;i<n;i++) cin>>a[i];
	f[0]=1;
	f[1]=2;
	for (int i=2;i<n;i++){
		f[i]=2;
		if (a[i] == a[i-1]+a[i-2]) f[i]=max(f[i],f[i-1]+1);
	}
	int ans=0;
	for (int i=0;i<n;i++) ans=f[i]>ans?f[i]:ans;
	cout<<ans<<endl;
	return 0;
}
