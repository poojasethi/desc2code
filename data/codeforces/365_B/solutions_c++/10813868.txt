#include <iostream>
using namespace std;
int mmax=0, res=0, n, a[100005];
int main(){
    cin>>n;
    for (int i=0; i<n; i++) cin>>a[i];
    for (int i=2; i<n; i++){
        if (a[i]==a[i-1]+a[i-2]) mmax++;
        else mmax=0;
        res=max(res,mmax);
    }
    cout<<(n==1 ? res+1:res+2);
    return 0;
}
