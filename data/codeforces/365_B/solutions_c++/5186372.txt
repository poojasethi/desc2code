#include <iostream>
#include <cmath>

using namespace std;

int n;
int a[1000000];

int main()
{
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>a[i];
  int maximo=min(2,n);
  int len=2;
  for (int i=2;i<n;i++) {
    if (a[i]==a[i-1]+a[i-2])
      len++;
    else
      len=2;
    maximo=max(maximo,len);
  }
  cout<<maximo<<endl;
}
