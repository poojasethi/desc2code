#include<iostream>
#include<cstdio>
using namespace std;

int main(){
	int n,arr[100003];
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	int res=min(n,2),len=min(n,2);
	for(int i=2;i<n;i++){
		if(arr[i]==arr[i-1]+arr[i-2]){
			len++;
			res=max(res,len);
		}	
		else{
			len=2;
		}	
	}
	cout<<res<<endl;
	return 0;
}
