#include <cstdio>

int a[100005], n;

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) scanf("%d", a + i);
	int ans = 1, len = 1;
	for (int i = 2; i <= n; ++i) {
		if (a[i] == a[i - 1] + a[i - 2]) ++len;
		else len = 2;
		if (len > ans) ans = len;
	}
	printf("%d\n", ans);
	return 0;
}
