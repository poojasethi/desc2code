#include<iostream>
using namespace std;
int main()
{
	int n; cin>>n;
	int a[n];
	for(int i=0;i<n;i++) cin>>a[i];

	if(n<3) cout<<n<<endl;
	else {
		int ans=2;
		int tmp=2;
		for(int i=2;i<n;i++)
		{
			if(a[i-2]+a[i-1]==a[i]) tmp++;
			else tmp=2;

			ans=max(ans,tmp);
		}
		cout<<ans<<endl;
	}
	return 0;
}
