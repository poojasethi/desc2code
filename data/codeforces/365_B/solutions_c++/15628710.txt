#include<stdio.h>
int main()
{
	int n,a[100010],i;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	int cnt=2;int max=2;
	for(i=2;i<n;i++)
	{
		if(a[i]==a[i-1]+a[i-2])
		{
			cnt++;
		}
		else
		cnt=2;
		if(cnt>max)max=cnt;
	}
	if(n==1)max=1;
	printf("%d\n",max);
	return 0;
 }
