#include<Cstdio>

int a[100100];

int main(){
	int n,l=1,r=1,ac=0;
	a[0]=0;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		
		if(i>1){
			if(a[i]==a[i-1]+a[i-2]) r=i;
			else{
				l=i-1; r=i;
			}
		}
		
		if(r-l+1>ac) ac=r-l+1;
	}
	
	printf("%d",ac);
	return 0;
}
