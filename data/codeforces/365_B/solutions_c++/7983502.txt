#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int a[100050];
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	int res=n<2?n:2;
	int t=2;
	for(int i=2;i<n;i++){
		if(a[i]==a[i-1]+a[i-2])t++;
		else t=2;
		res=max(t,res);
	}
	cout<<res<<endl;
}
