#include <iostream>
#include <stdio.h>
using namespace std;

int main() {
int n,i;
cin>>n;
int mx=n,tl;
int a[3];
if(n>2) {
  tl=mx=2;
 cin>>a[0]>>a[1]; 
 for(i=2;i<n;i++) 
 {
  cin>>a[i%3];
  if(a[i%3]==a[(i+1)%3]+a[(i+2)%3]) tl++;
  else  tl=2;    
  if(tl>mx) mx=tl; 
  
 }
}  
cout<<mx;
 return 0; 
}
