#include<iostream>

using namespace std;
long long n,a[100002],b[100001],o,ma,k;
int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	for(int i=3;i<=n;i++){
		if(a[i]==a[i-1]+a[i-2])b[i-2]=1;
	}
	for(int i=1;i<=n;i++){
		if(b[i]==1)o++;
		else o=0;
		if(ma<o)ma=o;
	}
	if(n==1)ma=-1;
	cout<<ma+2;
}
