#!env python

with open('input.txt') as f:
    n = []
    t = int(f.readline())
    for x in xrange(t):
        n.append(int(f.readline()))

result = []
for x in n:
    if x % 2 == 0:
        result.append("1")
    else:
        result.append("0")

p = '\n'.join(result)

with open('output.txt', 'w') as f:
    f.write(p)
