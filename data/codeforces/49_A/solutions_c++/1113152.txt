#include <iostream>
#include <cctype>
#include <string>

using namespace std;
int main () {
	char p = ' ';
	
	while (true) {
		char c; cin >> c;
		if (c == '?') break;
		if (c != ' ') p = c;
	}
	
	p = tolower(p);
	
	string s ("aeiouy");
	
	if(s.find(p) != -1) {
		cout << "YES";
	} else {
		cout << "NO";
	}
	
	return 0;
}
