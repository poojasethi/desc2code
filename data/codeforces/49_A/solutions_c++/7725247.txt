#include <iostream>

using namespace std;

int main() {
    string s;
    getline(cin, s);
    int i = s.find_last_not_of(" ?");
    s[i] |= 1 << 5;
    if( s[i] == 'a' || s[i] == 'e' || s[i] == 'i' ||
        s[i] == 'u' || s[i] == 'o' || s[i] == 'y' )
        cout << "YES";
    else
        cout << "NO";
}
