#include <iostream>
#include <string>

using namespace std;

string g = "AEOIUYaeoiuy";

int main()
{
	string s;
	getline(cin, s);
	int i = s.length()-1;
	while (s[i] == '?' || s[i] == ' ')
		i--;
	char c = s[i];
	if (g.find(c) != -1)
		cout << "YES" << endl;
	else
		cout << "NO" << endl;
	
	return 0;
}
