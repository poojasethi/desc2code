#include<iostream>
#include<ctype.h>
#include<stdio.h>
using namespace std;
int main()
{
char line[101],ch;
int i;
gets(line);
for(i=0; line[i]!='?'; i++)
if(isalpha(line[i]))
ch=line[i];
ch=tolower(ch);
if(ch=='a' || ch=='e' || ch=='i' || ch=='o' || ch=='u' || ch=='y')
cout<<"YES";
else
cout<<"NO";
return 0;
}
