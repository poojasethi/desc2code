#include <bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	cin>>s;
	for(int i=0;i<=s.length();i++)
	{
		for(int j=s.length();j>=0;j--)
		{
			string str;
			str=s.substr(0,i)+s.substr(j,s.length());
			if(str=="CODEFORCES")
			{
				cout<<"YES";
				exit(0);
			}
		}
	}
	cout<<"NO";
	return 0;
}
