#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

const int Sn = 1010;
#define addans(_y, _x) Ans ++, An1[Ans] = _x, An2[Ans] = _y

int n, A[Sn];
int a, b, c;
int Ans, An1[Sn*Sn], An2[Sn*Sn];

void move(int i, int j)
{
     A[i] -= A[j];
     A[j] += A[j];
     addans(i, j);
}
void solve(int a, int b, int c)
{
     while(A[a] && A[b] && A[c]) {
	  if(A[b] > A[c]) swap(b, c);
	  if(A[a] > A[b]) swap(a, b);
	  if(A[b] > A[c]) swap(b, c);
	  for(int t = A[b] / A[a]; t; t >>= 1)
	       if(t & 1) move(b, a);
	       else  move(c, a);
     }
}
int check()
{
     int cnt = 0;
     for(int i = 1; i <= n; i ++)
	  cnt += !!A[i];
     return cnt;
}

int main()
{
#ifndef ONLINE_JUDGE
     freopen("cf.in" , "r", stdin);
     freopen("cf.out", "w", stdout);
#endif
     cin >> n;
     for(int i = 1; i <= n; i ++)
	  cin >> A[i];
     
     for(int i = 1; i <= n; i ++) if(A[i]) {
	  if(!A[a]) a = i;
	  else if(!A[b]) b = i;
	  else if(!A[c]) c = i;
	  solve(a, b, c);
     }

     if(check() != 2) printf("-1\n");
     else {
	  printf("%d\n", Ans);
	  for(int i = 1; i <= Ans; i ++)
	       printf("%d %d\n", An1[i], An2[i]);
     }
}
