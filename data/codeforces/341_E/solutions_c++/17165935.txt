#include <iostream>
#include <cstdio>
using namespace std;
int a[1010],b[1010],ans[1000010][2],i,n,tot=0;
void work(int &x,int &y,int &z)
{
	if (a[x]>a[y]) swap(x,y);
	if (a[x]>a[z]) swap(x,z);
	int s;
	for (;a[x];)
	{
		if (a[y]>a[z]) swap(y,z);
		s=a[y]/a[x];
		for (;s;s>>=1,a[ans[tot][0]=x]<<=1)
			if (s&1) a[y]-=a[x],ans[++tot][1]=y;else a[z]-=a[x],ans[++tot][1]=z;
		swap(x,y);
	}
}
int main()
{
		scanf("%d\n",&n);
		for (i=1;i<=n;++i)
		{
			scanf("%d",&a[i]);
			if (a[i]) b[++b[0]]=i;
			if (b[0]==3)
			{
				work(b[1],b[2],b[3]);
				b[1]=b[2];b[2]=b[3];b[0]=2;
			}
		}
		if (b[0]!=2) tot=-1;
		printf("%d\n",tot);
		for (i=1;i<=tot;++i) printf("%d %d\n",ans[i][0],ans[i][1]);
}
