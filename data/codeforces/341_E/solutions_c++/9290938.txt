#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
#include <vector>
using namespace std;
const int Maxn=1005;
int n,num[Maxn];
vector < pair<int,int> > ans;

void move(int s,int t) {
	num[s]-=num[t];
	num[t]<<=1;
	ans.push_back(make_pair(t,s));
}

void del(int p1,int p2,int p3) {
	while (num[p1]&&num[p2]&&num[p3]) {
		if (num[p1]<num[p2]) swap(p1,p2);
		if (num[p1]<num[p3]) swap(p1,p3);
		if (num[p2]<num[p3]) swap(p2,p3);
		for (int x=num[p2]/num[p3];x;x>>=1)
			if (x&1) move(p2,p3);
			else move(p1,p3);
	}
}

int main() {
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",num+i);
	for (int i=1,p1=0,p2=0,p3=0;i<=n;i++) {
		(num[p1]==0?p1:(num[p2]==0?p2:p3))=i;
		del(p1,p2,p3);
	}
	int cnt=0;
	for (int i=1;i<=n;i++) cnt+=num[i]!=0;
	if (cnt!=2) puts("-1");
	else {
		printf("%u\n",ans.size());
		for (int i=0;i<ans.size();i++)
			printf("%d %d\n",ans[i].first,ans[i].second);
	}
	return 0;
}
