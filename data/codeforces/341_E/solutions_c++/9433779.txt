#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
using namespace std;
const int maxn=1000005;
int a[maxn],b[maxn],tot;
int n,u[maxn],v[maxn],cnt;
void change(int i,int j)
{
	a[i]-=a[j];
	a[j]*=2;
	u[++tot]=j;
	v[tot]=i;
}
void work(int i,int j,int k)
{
	if(a[i]>a[j])
	swap(i,j);
	if(a[j]>a[k])
	swap(j,k);
	if(a[i]>a[j])
	swap(i,j);
	if(a[i]==a[j])
	{
		change(i,j);
		return;
	}
	if(a[j]==a[k])
	{
		change(j,k);
		return;
	}
	int t=a[j]/a[i];
	for(int x=1;x<=t;x<<=1)
	{
		if(t&x)
		change(j,i);
		else
		change(k,i);
	}
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		if(a[i])
		b[++cnt]=i;
	}
	if(cnt<=1)
	printf("-1\n");
	else
	{
		for(int i=cnt-2;i>=1;i--)
		{
			work(b[i],b[i+1],b[i+2]);
			if(!a[b[i+1]])
			swap(b[i+1],b[i+2]);
			else if(!a[b[i]])
			swap(b[i],b[i+2]);
			else if(!a[b[i+2]]);
			else
			i++;
		}
		cout<<tot<<endl;
		for(int i=1;i<=tot;i++)
		printf("%d %d\n",u[i],v[i]);
	}
	return 0;
}
