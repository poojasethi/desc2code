#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>

#include <vector>

using namespace std;

#define REP(i,n) for(int i=0;i<n;++i)
#define FOR(i,a,b) for(int i=a;i<=b;++i)
#define FORD(i,a,b) for(int i=a;i>=b;--i)
#define pb push_back
#define mk(a,b) make_pair(a,b)
#define fi first
#define se second
#define LL long long

#define SORT(a,b,c) if(data[a]>data[b]) swap(a,b);\
					if(data[b]>data[c]) swap(b,c);\
					if(data[a]>data[b]) swap(a,b);

#define N 1010

int n;
int data[N];

int a, b, c, tmp;
vector<pair<int,int> > ans;


int main(){
	scanf("%d",&n);
	
	FOR(i,1,n) scanf("%d ",data+i);
	
	FOR(i,1,n){
		if(!data[a]){a=i;continue;}
		if(!data[b]){b=i;continue;}
		c=i;
		
		for(;;){
			SORT(a,b,c);
			if(!data[a]) break;
			for(int i=data[b]/data[a]; i; i>>=1){
				tmp=(i&1)?b:c;
				ans.pb(mk(a,tmp));
				data[tmp]-=data[a];
				data[a]<<=1;
			}
		}
		
		a=b, b=c;
	}
	
	if(!!data[a] + !!data[b] != 2) printf("-1\n");
	else{
		printf("%d\n",ans.size());
		for(__typeof(ans.begin()) cur=ans.begin(); cur!=ans.end(); ++cur)
			printf("%d %d\n",cur->fi,cur->se);
	}
	
	return 0;
}
