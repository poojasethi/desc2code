#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

typedef pair<int, int> PI;

#define fi first
#define se second

const int N = 1005;
const int M = 1e6 + 7;

int n, tot, cnt;
PI x, y, z[N], step[M];

void move(PI &a, PI &b) { b.fi -= a.fi, a.fi *= 2, step[++cnt] = PI(a.se, b.se); }

void through(PI &a, PI &b, PI &c)
{
	if(a.fi > b.fi) swap(a, b);
	if(b.fi > c.fi) swap(b, c);
	if(a.fi > b.fi) swap(a, b);
	
	if(!a.fi) return;
	
	int p = b.fi / a.fi;
	for(; p; p>>=1) if(p&1) move(a,b); else move(a,c);
	through(a, b, c);
}

int main()
{
	
	scanf("%d", &n);
	for(int i=1, a; i<=n; i++)
	{
		scanf("%d", &a);
		if(a>0) z[++tot] = PI(a, i);
	}
	
	if(tot < 2) puts("-1");
	else
	{
		x = z[1], y = z[2];
		for(int i=3; i<=tot; i++) through(x, y, z[i]), swap(z[i], x);
		
		printf("%d\n", cnt);
		for(int i=1; i<=cnt; i++) printf("%d %d\n", step[i].fi, step[i].se);
	}
	
	return 0;
}
