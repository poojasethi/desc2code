#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
#include <string.h>
#include <cstdio>
using namespace std;
pair < int, int > a[1111], b[1111111];
int cont; 
void f(int i)
{
	sort(a + i, a + i + 3);
	int x = a[i].first, y = a[i + 1].first, z = a[i + 2].first;
	if (x == 0)
		return;
	int q = y / x;
	for (int j = 0; (1 << j) <= q; j++)
	{
		if ((q & (1 << j)) > 0)
		{
			y -= x;
			b[cont++] = make_pair(a[i].second, a[i + 1].second);
		}
		else 
		{
			z -= x;
			b[cont++] = make_pair(a[i].second, a[i + 2].second);
		}
		x *= 2;
	}
	a[i].first = x; a[i + 1].first = y; a[i + 2].first = z;
	if (a[i].first != 0)
		f(i);
}

int main()
{
	int n, kol = 0;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		int x;
		cin >> x;
		if (x > 0)
			a[kol++] = make_pair(x, i + 1);
	}
	if (kol < 2)
	{
		cout << -1; 
		return 0;
	}
	sort(a, a + kol);
	for (int i = 0; i + 2 < kol; i++)
		f(i);
	cout << cont << "\n";
	for (int i = 0; i < cont; i++)
		cout << b[i].first << " " << b[i].second << "\n";
	cin >> n;
	return 0;
}
