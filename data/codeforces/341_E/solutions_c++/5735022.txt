#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>
#include <string>
#include <cctype>
#include <algorithm>
#define maxn 1005

using namespace std;

typedef long long LL;

struct Tnode
 {
  int num;LL a;
 }a[maxn];

int n,ans,ax[1000005],ay[1000005];

void init()
 {
  scanf("%d",&n);
  int i;
  for (i=1;i<=n;i++)
   {
    scanf("%I64d",&a[i].a);a[i].num=i;
   }
 }

int cmp(const Tnode &a,const Tnode &b)
 {
  return a.a<b.a;
 }

void calc(int step)
 {
  sort(a+step,a+step+3,cmp);
  if (!a[step].a)return;
  int times=a[step+1].a/a[step].a;
  while (times)
   {
    ans++;
    if (times&1){ax[ans]=a[step].num;ay[ans]=a[step+1].num;a[step+1].a-=a[step].a;}
     else {ax[ans]=a[step].num;ay[ans]=a[step+2].num;a[step+2].a-=a[step].a;}
    a[step].a+=a[step].a;times>>=1;
   }
  calc(step);
 } 
 
void work()
 {
  sort(a+1,a+n+1,cmp);
  int i;
  for (i=1;i<=n;i++)if (a[i].a)break;
  if (n-i+1<2){puts("-1");return;}
  for (;i<=n-2;i++)
   {
    calc(i);
   }
  printf("%d\n",ans);
  for (i=1;i<=ans;i++)printf("%d %d\n",ax[i],ay[i]);
 }
 
int main()
 {
  init();
  work();
  return 0;
 }
