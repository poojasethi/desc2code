#include <cstdio>
#include <algorithm>
using namespace std;

int n, i, x, j, p1, p2, p3, c, a[2000], ans1[1100000], ans2[1100000];

void put(int x) {
	a[x] -= a[p1];
	a[p1] *= 2;
	c++;
	ans1[c] = p1;
	ans2[c] = x;
}

int main() {
	scanf("%d", &n);
	for (i = 1; i <= n; i++)	scanf("%d", &a[i]);
	for (p2 = 1, p3 = 2, i = 3; i <= n; i++) {
		if (!a[p1])	p1 = i;
		else	if (!a[p2])	p2 = i;
		else	p3 = i;
		while (a[p1] && a[p2] && a[p3]) {
			if (a[p1] > a[p2])	swap(p1, p2);
			if (a[p2] > a[p3])	swap(p2, p3);
			if (a[p1] > a[p2])	swap(p1, p2);
			for (x = a[p2] / a[p1], j = 1; j <= x; j *= 2)	if (x & j)	put(p2);	else	put(p3);
		}
	}
	if ((a[p1] == 0) + (a[p2] == 0) + (a[p3] == 0) > 1)	printf("-1\n");
	else for (printf("%d\n", c), i = 1; i <= c; i++)	printf("%d %d\n", ans1[i], ans2[i]);
}
