#include <cstdio>
#include <algorithm>
#define push(i,j,k) a[i]+=k,a[j]-=k,x[++S]=b[i],y[S]=b[j]
using namespace std;

int S,N,M,a[1005],b[1005],x[1000005],y[1000005];

void work(int i,int j,int k)
{
	for (;;)
	{
		if (a[i]>a[j]) swap(i,j);
		if (a[i]>a[k]) swap(i,k);
		if (a[j]>a[k]) swap(j,k);
		if (!a[i]) {swap(a[N],a[i]),swap(b[N--],b[i]); return;}
		int t=a[j]/a[i];
		for (int l=0,z=a[i]; 1<<l<=t; l++) if ((t>>l)&1) push(i,j,z<<l); else push(i,k,z<<l);
	}
}

void doit()
{
	scanf("%d",&N);
	for (int i=1,x; i<=N; i++)
	{
		scanf("%d",&x);
		if (x) a[++M]=x,b[M]=i;
	}
	if (M<2) {puts("-1"); return;}
	for (int i=1; i<=N-2;) work(i,i+1,i+2);
	printf("%d\n",S);
	for (int i=1; i<=S; i++) printf("%d %d\n",x[i],y[i]);
}

int main()
{
	doit();
	return 0;
}
