#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define N1 1000010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int a,id;
} a[N];
int i,j,k,n,m,s,l[N1],r[N1];
inline bool cc1(const ww &a,const ww &b) {
	return a.a<b.a;
}
inline void Do(ww &a,ww &b) {
	b.a-=a.a; a.a<<=1;
	l[++s]=a.id,r[s]=b.id;
}
void work(ww &a,ww &b,ww &c) {
	int A=b.a%a.a,B=b.a/a.a;
	for (;B;B>>=1) {
		if (B&1) Do(a,b);
		else Do(a,c);
	}
	swap(a,b);
	if (b.a>c.a) swap(b,c);
	if (!A) a=b,b=c;
	else work(a,b,c);
}
int main() {
	//freopen("341e.in","r",stdin);
	//freopen("341e.out","w",stdout);
	scanf("%d",&n);
	For(i,1,n) scanf("%d",&k),k>0?a[++m].a=k,a[m].id=i:0;
	if (m<2) printf("-1\n"),exit(0);
	sort(a+1,a+m+1,cc1); n=0;
	For(i,3,m) {
		a[3]=a[i];
		if (a[3].a<a[2].a) swap(a[2],a[3]);
		if (a[2].a<a[1].a) swap(a[1],a[2]);
		work(a[1],a[2],a[3]);
	}
	printf("%d\n",s);
	For(i,1,s) printf("%d %d\n",l[i],r[i]);
	//for(;;);
	return 0;
}
