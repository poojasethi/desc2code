#include <iostream>
using namespace std;
int a[1001];
int x[1000000];
int y[1000000];
int tot=0;
int tz(int i,int j,int k){
    if(a[i]>a[j])return tz(j,i,k);
    if(a[i]>a[k])return tz(k,j,i);
    if(a[j]>a[k])return tz(i,k,j);
    if(a[i]==0)return 1;
    int q=a[j]/a[i];

    while(q){
        x[tot]=i;
        if(q&1){a[j]-=a[i];y[tot++]=j;}
        else {a[k]-=a[i];y[tot++]=k;}
        q>>=1;
        a[i]<<=1;
    }
    return tz(i,j,k);
}
int main() {
    int n;
    int f=0;
    cin >> n;
    for(int i=1;i<=n;i++)
        cin >> a[i];
    for(int i=1;i<=n;i++)
        if(a[i]!=0)
        for(int j=1;j<=n;j++)
            if(i!=j&&a[j]!=0){f=1;
            for(int k=1;k<=n;k++)
                if(i!=k&&j!=k&&a[k]!=0)tz(i,j,k);
            }
    if(f!=0)
    cout << tot<<endl;
    else cout << -1 << endl;
    if(f)for(int i=0;i<tot;i++)
        cout << x[i] <<' '<< y[i] << endl;
}
