import sys
R=lambda:map(int,raw_input().split())
n,maxhp,reg=R()
s=[(b,a,i+1)  for i in xrange(n) for a,b in [R()]]
s.sort(reverse=True)
dmg=0;hp=maxhp
ans=[]
for t in xrange(2000):
    hp=min(maxhp,hp-dmg+reg)
    if hp<=0:
        print 'YES'
        print t,len(ans)
        print '\n'.join(ans)
        sys.exit()
    for i,(d,p,id) in enumerate(s):
        if p*maxhp>=hp*100:
            dmg+=d
            ans+=["%d %d"%(t,id)]
            del(s[i])
            break

print 'NO'
