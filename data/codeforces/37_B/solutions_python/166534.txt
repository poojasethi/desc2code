scrollnum, maxhealth, regen = [int(x) for x in raw_input().split(" ")]

health = maxhealth


scrolls = []

for x in range(scrollnum):
    scrolls.append([int(x) for x in raw_input().split(" ")])

for x in range(scrollnum):
    scrolls[x].append(x+1)

def sorter(x, y):
    if x[0] != y[0]:
        return y[0]-x[0]
    return y[1]-x[1]

scrolls = sorted(scrolls, cmp = sorter)

used = []

time = -1
while health > 0:
    time += 1

    health += regen
    health = min(health, maxhealth)
    
    if health < 0: break

    #use best scroll
    
    if len(scrolls) == 0:
        if regen < 0: continue
        print "NO"
        break

    bestindex = -1
    popindex = -1
    bestloss = 0

    for i, s in enumerate(scrolls):
        if float(health) * 100.0 / float(maxhealth) > s[0]: continue

        if s[1] > bestloss:
            bestindex = s[2]
            bestloss = s[1]
            popindex = i
        
    if bestindex == -1:
        if regen >= 0:
            print "NO"
            break
        continue
    
    regen -= bestloss

    scrolls.pop(popindex)


    used.append( "%d %d" % (time, bestindex))
    #remove from list

#print out order
if health <= 0:
    print "YES"
    print time, scrollnum - len(scrolls)
