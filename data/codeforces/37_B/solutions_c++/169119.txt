#include <stdio.h>
const int maxn = 1005;
int p[maxn], d[maxn], l[maxn][2], nl, ti;
int main() {
    int n, h, r, c;
    scanf("%d%d%d", &n, &h, &r);
    for(int i = 0; i != n; ++i) scanf("%d%d", p + i, d + i);
    c = h;
    while(true) {
        c += r;
        if(c > h) c = h;
        if(c <= 0) break;
        
        int t = 0, pos = -1;
        for(int i = 0; i != n; ++i) 
            if(c * 100 <= h * p[i] && t < d[i]) t = d[i], pos = i + 1;

        if(~pos) {
            r -= t;
            l[nl][0] = ti;
            l[nl ++][1] = pos;
            d[pos - 1] = -1;
        } else if(r >= 0) { puts("NO"); return 0; }
        ti ++;
    }
    puts("YES");
    printf("%d %d\n", ti, nl);
    for(int i = 0; i != nl; ++i) printf("%d %d\n", l[i][0], l[i][1]);
    return 0;
}
