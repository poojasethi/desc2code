#include<cstdlib>
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<string>

using namespace std;

int n,m0,r,m,p[3000],d[3000],ans[3000][2];
bool used[3000];

int main()
{
	cin>>n>>m0>>r;
	m=m0*100;
	r*=100;
	for(int i=1;i<=n;i++)
	{
		cin>>p[i]>>d[i];
		d[i]*=100;
	}
	int t=0,s=0,curd=0;
	for(int i=0;i<2000;i++)
	{
		int md=0,msn=0;
		for(int j=1;j<=n;j++)
			if(!used[j])
			{
				if(p[j]*m0>=m)
				{
					if(d[j]>md)
					{
						md=d[j];
						msn=j;
					}
				}
			}
		curd+=md;
		int curm=m-curd+r;
		if(!md)
		{
			if(curm>=m)
			{
				cout<<"NO";
				return 0;
			}
		}
		else
		{
			used[msn]=1;
			ans[s][0]=t;
			ans[s++][1]=msn;
		}
		t++;
		m=min(curm,m0*100);
		if(m<=0)
			break;
	}
	cout<<"YES"<<endl<<t<<" "<<s<<endl;
	for(int i=0;i<s;i++)
		cout<<ans[i][0]<<" "<<ans[i][1]<<endl;
	return 0;
}
