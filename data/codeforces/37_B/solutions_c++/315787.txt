#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#define REP(i,n) for(int i=0;i<(int)(n);i++)
using namespace std;
typedef long long ll;

int main()
{
  int n,max_hp,reg,pow[1000],dmg[1000];
  cin>>n>>max_hp>>reg;
  REP(i,n) cin>>pow[i]>>dmg[i];

  int cur = max_hp, attack = 0;
  vector<pair<int,int> > scr;
  bool used[1000] = {0};
  for(int turn = 0;;turn++){
    int pre = cur;
    cur = min(max_hp,cur - attack + reg);
    if (cur <= 0) {
      printf("YES\n%d %d\n",turn,scr.size());
      REP(i,scr.size())
        printf("%d %d\n",scr[i].first,scr[i].second);
      break;
    }

    int mx = 0, mx_i = -1;
    REP(i,n){
      if (!used[i] && cur * 100 <= pow[i] * max_hp && mx < dmg[i]){
        mx = dmg[i];
        mx_i = i;
      }
    }

    if (mx_i != -1){
      used[mx_i] = true;
      attack += dmg[mx_i];
      scr.push_back(make_pair(turn,mx_i + 1));
    }
    else if (pre == cur){
      puts("NO");
      break;
    }
  }

  return 0;
}
