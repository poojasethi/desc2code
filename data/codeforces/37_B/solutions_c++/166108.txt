#include <stdio.h>

int main()
{
  int n, max, reg;
  scanf("%d %d %d", &n, &max, &reg);
  int pow[1001] = {0}, dmg[1001] = {0};
  int used[1001] = {0};
  for (int i=0; i<n; i++)
    scanf("%d %d", &pow[i], &dmg[i]);
  int remain = max;
  int tm = 0;
  int used_cnt = 0;
  int used_time[1001] = {0}, used_num[1001] = {0};
  int damage_sum = 0;
  while (remain>0) {
    remain -= damage_sum - reg;
    if (remain > max) remain = max;
    if (remain <= 0) break;
    int idx = -1;
    int dmg_now = -1;
    for (int i=0; i<n; i++) {
      if (pow[i]*max<remain*100) continue;
      if (dmg_now<dmg[i] && !used[i]) { idx = i; dmg_now = dmg[i]; }
    }
    if (idx == -1 && damage_sum <= reg) break;
    if (idx > -1) {
      used_time[used_cnt] = tm;
      used_num[used_cnt++] = idx+1;
      used[idx] = 1;
      damage_sum += dmg[idx];
    }
    tm ++;
  }
  if (remain>0) printf("NO\n"); else {
    printf("YES\n");
    printf("%d %d\n", tm, used_cnt);
    for (int i=0; i<used_cnt; i++) printf("%d %d\n", used_time[i], used_num[i]);
  }
  return 0;
}
