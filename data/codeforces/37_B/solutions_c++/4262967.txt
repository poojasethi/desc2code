#include <iostream>
#include <cstdio>
using namespace std;
int n,maxi,reg,dmg[1001],powr[1001],e,dm,us[1001],ans[1001],ans2[1001],t,uk,i,d,dm2;
int main()
{
cin>>n>>maxi>>reg;
for (i=1;i<=n;i++)
cin>>powr[i]>>dmg[i];
d=maxi;
while (1)
{
	e=0;
	dm=0;
	for (i=1;i<=n;i++)
	{
	if ((powr[i]*maxi>=(d*100)) && us[i]==0)
		{
                if (dmg[i]>=dm)
			{
                        dm=dmg[i];
			e=i;
			}
		}
	}
	if (e==0 && dm2<=reg)
		{
		cout<<"NO";	
		return 0;
		}
	if (e>0) {
//	cout<<1<<endl;
	us[e]=1;
	dm2+=dm;
	ans[uk]=t;
	ans2[uk++]=e;
	}
	t++;
	d=min(maxi,d-dm2+reg);
	if (d<=0)
		{cout<<"YES"<<endl;
		break;}


}
cout<<t<<" "<<uk<<endl;
for (i=0;i<uk;i++)
cout<<ans[i]<<" "<<ans2[i]<<endl;




return 0;
}
