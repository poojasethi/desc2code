#include<iostream>
#include<vector>

using namespace std;

int main()
{
	int d,f=0,t,n,m,r,i,pow[1000],dmg[1000];
	vector<pair<int,int> >   rez;

	cin>>n>>m>>r;	
	for(i=0;i<n;i++)
	{
		cin>>pow[i]>>dmg[i];
		if(pow[i]==100)   f+=dmg[i];
	}

	if(f<=r)    
	{
		cout<<"NO";
		return 0;
	}

	t=0;
	d=m;
	while(m>0)
	{
		f=-1;
		for(i=0;i<n;i++)
			if(m*100<=pow[i]*d)
				if((f==-1)||(dmg[i]>dmg[f]))  f=i;

		if(m+r>0)
		if(f!=-1)
		{
			r-=dmg[f];
			pow[f]=-1;
			rez.push_back(make_pair(t,f+1));
		}

		m+=r;
		m=min(m,d);
		t++;
	}

	cout<<"YES\n";
	cout<<t<<" "<<rez.size()<<"\n";
	for(i=0;i<rez.size();i++)
		cout<<rez[i].first<<" "<<rez[i].second<<"\n";

	return 0;
}
