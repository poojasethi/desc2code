#include <iostream>
#include <string>
#include <algorithm>
#include <cstdio>
using namespace std;
int maxh=0,reg=0,h,stre=0,i,t,n,pp[2000],dmg[2000],ans_t[2000],ans_n[2000],ans_k=0;
bool use[2000];
int main() {
        scanf("%d%d%d",&n,&maxh,&reg);
        h=maxh;
        for (i=0;i<n;i++)scanf("%d%d", &pp[i], &dmg[i]);
        for (t=0;t<3000;t++) {
                h+=-stre+reg;
                h=min(maxh,h);
                if (h<=0)
                {
                        printf("YES\n");
                        printf("%d %d\n",t,ans_k);
                        for(i=0;i<ans_k;i++)printf("%d %d\n",ans_t[i],ans_n[i]);
                        return 0;
                }
                int x=-1;
                for(i=0;i<n;i++)if(!use[i]&&(h*100<=maxh*pp[i])&&(x<0||dmg[i]>dmg[x]))x=i;
                if (x>=0)
                {
                        ans_t[ans_k]=t;
                        ans_n[ans_k]=x+1;
                        ans_k++;
                        use[x]=1;
                        stre+=dmg[x];
                }

        }
        printf("NO\n");
        return 0;
}
