#include<stdio.h>
int a[1001][4],b[1001][2],lim[1001];
int n,m,k,h,uu;

void swap(int i,int j)
{
    for (int l=2;l<4;l++) {int u=a[i][l];a[i][l]=a[j][l];a[j][l]=u;}
}

int main()
{
    scanf("%d%d%d",&n,&m,&k);
    int t,hh;
    h=0;
    for (int i=1;i<=n;i++) 
    {
        a[i][2]=-1;scanf("%d%d",&a[i][0],&a[i][1]);if (a[i][0]==100) {h+=a[i][1];}a[i][3]=i;
        double v,u;
        u=a[i][0];u=u/100.0;v=m;u=u*v;u+=1e-10;
        lim[i]=u;lim[i]++;
    }
    if (h>k)
    {
        hh=m;t=0;h=0;int ll=1;
        while (hh>0)
        {
            hh-=h;
            hh+=k;if (hh>m) {hh=m;}
            if (hh<=0) {break;}
            int kk,uv;kk=0;uv=0;
            for (int i=ll;i<=n;i++) if (a[i][2]<0) {if (hh<lim[i]) {if (a[i][1]>kk) {kk=a[i][1];uv=i;}}}
            h+=kk;
            a[uv][2]=t;
            t++;
            if (uv!=0) {uu++;b[uu][0]=a[uv][2];b[uu][1]=uv;}
        }
        printf("YES\n");
        printf("%d %d\n",t,uu);
        for (int i=1;i<=uu;i++) {printf("%d %d\n",b[i][0],b[i][1]);}
    } else {printf("NO\n");}
    return 0;
}
