#include<cstdio>
#include<algorithm>
#include<vector>
#include<queue>
using namespace std;

const int N=1002;
int n,mx,reg,hp;
int pow[N],dmg[N];

vector<pair<int,int> >history;

int choose_spell(){
	int nr = -1;
	//printf("hp = %d\n",hp);
	for(int i=0; i<n; i++){
		if(hp*100 <= pow[i]*mx){
			if(nr==-1 || dmg[i]>dmg[nr]) nr=i;
		}
	}
	return nr;
}

main(){
	scanf("%d %d %d",&n,&mx,&reg);
	for(int i=0; i<n; i++){
		scanf("%d %d",&pow[i], &dmg[i]);
	}
	int dmgr=0, round;
	hp=mx;
	for(round=0;;round++){
		//printf("%d) hp %d, reg %d, dmgr %d\n",round,hp,reg,dmgr);
		hp-=dmgr;
		hp+=reg;
		hp=min(hp,mx);
		int nr = choose_spell();
		if(hp<=0) break;
		if(nr>=0){
			dmgr+=dmg[nr];
			pow[nr]=-1;
			history.push_back(make_pair(round, nr));
		} else if(dmgr<=reg){
			break;
		}
	}
	if(hp<=0){
		printf("YES\n%d %d\n",round, history.size());
		for(int i=0; i<history.size(); i++){
			printf("%d %d\n", history[i].first, history[i].second+1);
		}
	} else {
		printf("NO\n");
	}
}
