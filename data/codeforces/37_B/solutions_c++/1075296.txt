#include <iostream>
#include <vector>

using namespace std;

#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl
#define pi pair<int,int>

int n,mi,mm,reg,pow[1010],dmg[1010],used[1010],t,c,inf;
vector< pi > vi;

int main(){

  cin>>n>>mi>>reg;mm=mi;
  rep(i,n)cin>>pow[i]>>dmg[i];

  while(mi>0){
    mi=min(mi+reg-inf,mi);
    if(mi<=0)break;
    int tm=-1,ti=-1;
    rep(i,n){
      if(mm*pow[i]/100>=mi&&!used[i]&&dmg[i]>tm){
    tm=dmg[i];ti=i;
      }
    }
    if(ti==-1&&inf<=reg)break;
    if(ti!=-1){
      vi.push_back(pi(t,ti+1));
      used[ti]=true;
      inf+=tm;
    }
    t++;
  }

  if(mi>0)pr("NO");
  else {
    pr("YES");
    pr(t<<" "<<vi.size());
    rep(i,vi.size()){
      pr(vi[i].first<<" "<<vi[i].second);
    }
  }

  return 0;

}
