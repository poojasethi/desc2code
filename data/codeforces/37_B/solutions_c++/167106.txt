#include<cstdio>

using namespace std;

int n, maxhp, reg, hp, deb, time, nous;

struct scroll {
  int dps, ihp; 
} ta[1004] ;

char use[1004];

int w1[1004], w2[1004], wl=0;

int main() {
  scanf("%d%d%d",&n,&maxhp,&reg);
  for(int i=1; i<=n; i++) {
    scanf("%d%d",&ta[i].ihp,&ta[i].dps);
  }
  
  hp=maxhp;
  while(1) {
    int cn=0, mp=0;
    for(int i=1; i<=n; i++) if(!use[i]) if(ta[i].ihp*maxhp>=100*hp) if(ta[i].dps>mp) {
      mp=ta[i].dps;
      cn=i;
    }
    if(cn) {
      deb+=ta[cn].dps;
      use[cn]=1;
      w1[wl]=time; w2[wl]=cn; wl++;
    }
    hp-=deb;
    hp+=reg;
    if(hp>maxhp) hp=maxhp;
    time++;
    if(hp<=0) { printf("YES\n%d %d\n",time,wl); break; }
    if(time>10000) { printf("NO\n"); break; }
  }
  if(hp<=0) for(int i=0; i<wl; i++) printf("%d %d\n",w1[i],w2[i],wl);
  return 0; 
}
