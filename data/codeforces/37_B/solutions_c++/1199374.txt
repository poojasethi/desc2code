#include<stdio.h>
#include<algorithm>
#include<string>
#include<iostream>
#include<math.h>
using namespace std;

#define exp 1e-10

struct node
{
	int a,b,c,xu;
}p[1005];

bool cmp(node aa,node bb)
{
	if(aa.a==bb.a) return aa.b>bb.b;
	return aa.a>bb.a;
}

bool cmp1(node aa,node bb)
{
	return aa.b>bb.b;
}

int main()
{
	int n,i,j,s,k,t,ss,time,r,l;
	double pi;
	while(~scanf("%d%d%d",&n,&s,&k))
	{
		ss=s;time=0;
		for(i=0;i<n;i++)
		{
			scanf("%d%d",&p[i].a,&p[i].b);
			p[i].xu=i+1;
		}
		sort(p,p+n,cmp);
		pi=100;t=0;
		for(i=0;i<n;i++)
		if(p[i].a>pi||fabs(pi-p[i].a)<exp) t=t+p[i].b;
		else break;		
		if(t<=k) printf("NO\n");		
		else 
		{
			t=p[0].b;
			l=1;r=i;
			while(1)
			{
				s=s-t+k;
				time++;
				if(s>ss) s=ss;			
				if(s<=0) break;	
				pi=s*100.0/ss;
				for(r;r<n;r++)
				if(p[r].a<pi) break;
				if(l<r) {sort(p+l,p+r,cmp1);t=t+p[l].b;p[l++].c=time;}
			}
			printf("YES\n");
			printf("%d %d\n",time,l);
			for(j=0;j<l;j++)
			printf("%d %d\n",p[j].c,p[j].xu);
		}		
	}
}
