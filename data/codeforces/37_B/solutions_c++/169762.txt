#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define MAXN 1005

struct node
{
	int pow,dmg,id;
	node(){}
	node(int id,int p,int d):pow(p),dmg(d){}
}scroll[MAXN];
int tme[MAXN];
int id[MAXN];
bool flag[MAXN];
int main()
{
	int n,max,reg,tim,i,init;
	while(scanf("%d%d%d",&n,&init,&reg)!=EOF)
	{
		for(i=0;i<n;i++){
			scanf("%d%d",&scroll[i].pow,&scroll[i].dmg);
			scroll[i].id=i+1;
			flag[i]=false;
		}
		max=init;
		int tmp,who,cnt=0;
			for(tim=0;max>0;tim++)
			{	
				tmp=-1;
				for(i=0;i<n;i++)
				if(scroll[i].pow>=(double)(max*100.0/init))
				{
					if(!flag[i]&&tmp<scroll[i].dmg)
					{
						tmp=scroll[i].dmg;
						who=i;
					}
				}
				if(tmp==-1&&reg>=0)
					break;
				else if(tmp!=-1)
				{
					flag[who]=1;
					tme[cnt]=tim;
					id[cnt]=scroll[who].id;
					cnt++;
					reg-=tmp;
				}	
				max+=reg;
				if(max>init)	max=init;
			}
		if(max<=0)
		{
			printf("YES\n");
			printf("%d %d\n",tim,cnt);
			for(int j=0;j<cnt;j++)
				printf("%d %d\n",tme[j],id[j]);
		}
		else
			printf("NO\n");
	}
	return 0;
}
