#include<stdio.h>
int main()
{
	int ac,zong,n,max,reg,pow[1200][3],dmg[1200],i,kou,jishu,jishi,zhi,a[2000][5],qiy;
	double sheng;
	while(scanf("%d%d%d",&n,&max,&reg)!=EOF)
	{
		zong=0;
		kou=0;jishu=0;jishi=0;
		dmg[0]=0;
		for(i=1;i<=n;i++)
		{
			scanf("%d%d",&pow[i][1],&dmg[i]);
			pow[i][2]=0;
			if(pow[i][1]==100)zong+=dmg[i];
			if(pow[i][1]==100&&dmg[i]>kou){qiy=i;kou=dmg[i];}
		}
		if(zong-reg<=0)printf("NO\n");
		else
		{
			a[0][1]=0;a[0][2]=1;a[0][3]=qiy;   jishu=1;   zong=max; jishi=1;
			pow[qiy][2]=1;
			printf("YES\n");
			zhi=0;ac=0;
			while(max>0)
			{
				qiy=0;
				max=max-kou+reg;
				if(max>zong)max=zong;
				if(max<=0)break;
				sheng=max*1.0/(zong*1.0)*100.0;
				//printf("shijian:=%d;sheng=%.4lf\n",jishi,sheng);
				for(i=1;i<=n;i++)
				if(pow[i][1]>=sheng&&pow[i][2]==0&&dmg[i]>dmg[qiy])qiy=i;
				if(qiy!=0){a[++ac][2]=++jishu;a[ac][1]=jishi;a[ac][3]=qiy;}
				jishi++;
				kou=kou+dmg[qiy];
				pow[qiy][2]=1;
			}
			printf("%d %d\n",jishi,jishu);
			for(i=0;i<=ac;i++)
			printf("%d %d\n",a[i][1],a[i][3]);
		}
	}
	return 0;
}
