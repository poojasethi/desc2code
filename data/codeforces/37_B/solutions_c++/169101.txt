#include <iostream>
#include <vector>

using namespace std;

#define REP(i,a,b) for(i = (int)(a); i < (int)(b); i++)
#define rep(i,n) REP(i,0,n)

int main() {
  int n, max_hp, reg; cin >> n >> max_hp >> reg;
  int i,t,hp,dmg;
  vector<bool> used(n, false);
  vector<pair<int,int> > v(n), his;
  rep(i,n) { int a, b; cin >> a >> b; v[i] = make_pair(a,b); }

  hp = max_hp, dmg = 0;
  rep(t,1<<29) {
    hp -= dmg; hp = min(hp+reg, max_hp);
    if(hp <= 0) {
      break;
    }
    int atk = 0, idx = -1;
    rep(i,n) if(!used[i] && v[i].first*max_hp >= hp*100 && atk < v[i].second) {
      atk = v[i].second;
      idx = i;
    }
    if(idx != -1) {
      used[idx] = true;
      dmg += atk;
      his.push_back(make_pair(t,idx+1));
    } else if(reg >= dmg) {
      cout << "NO" << endl;
      return 0;
    }
  }
  cout << "YES" << endl;
  cout << t << " " << his.size() << endl;
  rep(i,his.size()) {
    cout << his[i].first << " " << his[i].second << endl;
  }
  return 0;
}
