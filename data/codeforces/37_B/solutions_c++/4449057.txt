#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#include <vector>
#include <utility>

#define pb push_back
#define mp make_pair

using namespace std;

int n,reg,dmg[1111],sec,used[1111],ok,total,mx,pos;
double h,pc[1111],health;
vector < pair < int ,int > > v;

void print()
{
	puts("YES");
	printf("%d %d\n",sec+1,v.size());
	for ( int i = 0; i < v.size(); i++ )
		printf("%d %d\n",v[i].first,v[i].second);
	
	exit(0);
}

int main()
{
	scanf("%d%lf%d",&n,&h,&reg);
	health = h;
	for ( int i = 1; i <= n; i++ )
		scanf("%lf%d",&pc[i],&dmg[i]);
	
	for ( ;; ) {
		mx = 0,pos = -1,ok = 0;
		for ( int j = 1; j <= n; j++ ) {
			if ( !used[j] ) ok = 1;
			if ( !used[j] && health*pc[j]/100 >= h && dmg[j] > mx ) mx = dmg[j],pos = j;
		}
		if ( pos != -1 ) {
			total+=dmg[pos];
			h-=total;
			h+=reg;
			v.pb(mp(sec,pos));
			if ( h <= 0 ) print();
			h=min(h,health);
			used[pos] = 1;
		}
		else {
			if ( reg >= total ) { puts("NO"); return 0; }
			h-=total;
			h+=reg;
			if ( h <= 0 ) print();
		}
		sec++;
	}
}
