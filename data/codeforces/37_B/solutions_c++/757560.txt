#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<iostream>
using namespace std;
typedef long long LL;
int ans[1200][2];
int x[1200],d[1200],vis[1200];
int n,Hp,ad,now,dam,t,ct,i,id;
int main()
{
	while(~scanf("%d%d%d",&n,&Hp,&ad))
	{
		for(i=0;i<n;i++)
		scanf("%d%d",&x[i],&d[i]);
		memset(vis,0,sizeof(vis));
		bool can=true;
		ct=0;now=Hp;dam=0;
		for(t=0;;t++)
		{
			now-=dam;
			now=min(Hp,now+ad);
			if(now<=0)break;
			id=-1;
			for(i=0;i<n;i++)
			{
				if(vis[i])continue;
				if(x[i]*Hp<now*100)continue;
				if(id==-1||d[i]>d[id])
				id=i;
			}
			if(id==-1&&dam<=ad)
			{can=false;break;}
			if(id!=-1)
			{
				vis[id]=1;dam+=d[id];
				ans[ct][0]=t;ans[ct++][1]=id;
			}
		}
		if(!can){puts("NO");continue;}
		puts("YES");
		printf("%d %d\n",t,ct);
		for(i=0;i<ct;i++)
		printf("%d %d\n",ans[i][0],ans[i][1]+1);
	}
}
