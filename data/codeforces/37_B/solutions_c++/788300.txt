#include <stdio.h>
#include <vector>
using namespace std;

int m, reg, n;
int u[1005],pow[1005], dmg[1005];
vector<pair<int, int> > sol;

int main(void) {
	scanf("%d%d%d", &n, &m, &reg);
	for(int i=0;i<n;i++) {
		scanf("%d%d", &pow[i], &dmg[i]);
	}
	int flag=0, t, now=m, pdm=-reg;
	for(t=0;now >0;t++) {
		int i, nxt=-1;
		for(i=0;i<n;i++) {
			if(!u[i] && (now*100+m-1)/m <= pow[i] && (nxt==-1||dmg[nxt] < dmg[i])) {
				nxt = i;
			}
		}
		if(nxt>=0) {
			sol.push_back(make_pair(t, nxt+1));
			u[nxt] = 1;
			pdm += dmg[nxt];
		}
		if(nxt==-1 && pdm <= 0) { flag=1;break; }
		now -= pdm;
		now = min(now, m);
	}
	if(flag) puts("NO");
	else {
		puts("YES");
		printf("%d %d\n", t, (int)sol.size());
		for(int i=0;i<(int)sol.size();i++)
			printf("%d %d\n", sol[i].first, sol[i].second);
	}
	return 0;
}
