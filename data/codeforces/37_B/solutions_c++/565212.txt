#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;

int main(void)
{
	int n,reg,max;
	scanf("%d %d %d",&n,&max,&reg);
	int s[1010][3];
	for(int i=1;i<=n;i++)
		scanf("%d %d",&s[i][0],&s[i][1]);
	int HP=max;
	int pre=0;
	int flag=0;
	int use[1010][3];
	int top=0;
	int dmg=0;
	memset(use,0,sizeof(use));
	int time;
	for(time=0;!flag;time++)
	{
		if(HP==pre)flag=1;
		pre=HP;
		HP-=dmg;
		HP=min(HP+reg,max);
		double per=(double)(HP*100.0)/max;
		int m=-1;
		int id=-1;
		for(int i=1;i<=n;i++)
		{
			if(s[i][0]>=per)
				if(m<s[i][1])
				{
					m=s[i][1];
					id=i;
				}
		}
		if(id>0&&HP>0)
		{
			s[id][0]=-1;
			use[top][0]=time;
			use[top][1]=id;
			top++;
			dmg+=s[id][1];
			flag=0;
			pre=0;
		}
		if(HP<=0)
		{
			break;
		}
	}
	if(flag==1)
	{
		puts("NO");
	}
	else 
	{
		puts("YES");
		printf("%d %d\n",time,top);
		for(int i=0;i<top;i++)
		{
			printf("%d %d\n",use[i][0],use[i][1]);
		}
	}
	return 0;
}
