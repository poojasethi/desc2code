#include <stdio.h>
#include <map>
#include <vector>
#include <algorithm>

using namespace std;

#define REP(i,n) for (int i=0,_n=n; i<_n; i++)
#define FORE(i,a) for (__typeof(a.begin()) i=a.begin(); i!=a.end(); i++)

int N,mx,rg,pw[1001],dm[1001],u[1001];

int main(){
	scanf("%d %d %d",&N,&mx,&rg);
	REP(i,N) scanf("%d %d",&pw[i],&dm[i]);

	int dps = 0, t = 0, H = mx;
	vector<pair<int,int> > S;
	while (true){
		int s = -1;
		REP(i,N) if (H*100 <= pw[i]*mx && !u[i])
			if (s==-1 || dm[s] < dm[i]) s = i;
	
		if (s==-1 && dps<=rg){
			puts("NO");
			return 0;
		}

		if (s!=-1){
//			printf("try %d %d\n",s,dm[s]);
			S.push_back(make_pair(t,s));
			dps += dm[s];
			u[s] = 1;
		}

		H -= dps;
		H = min(mx, H+rg);
		t++;
		if (H<=0) break;

//		printf("%d : %d %d %d\n",t,H,dps,rg);
	}
	puts("YES");
	printf("%d %d\n",t,S.size());
	REP(i,S.size()) printf("%d %d\n",S[i].first,S[i].second+1);
}
