#include <cstdio>
#include <vector>

using namespace std;

int n, hp, reg;
int per[1000], dmg[1000];

vector <pair<int, int> > ans;
bool isVisit[1000];

int main(void)
{
	scanf("%d %d %d", &n, &hp, &reg);
	for(int i=0;i<n;i++) scanf("%d %d", per+i, dmg+i);
	
	int maxHp=hp, curDmg=0;
	int t;
	for(t=0;t<10000;t++)
	{
		hp-=curDmg;
		hp+=reg;
		if(hp>maxHp) hp=maxHp;

		if(hp<=0) break;

		int curP=hp*100/maxHp;
		int ind=-1, maxVal=0;

		for(int j=0;j<n;j++)
		{
			if(isVisit[j]) continue;
			if(per[j]<curP || (per[j]==curP && hp*100%maxHp)) continue;

			if(maxVal<dmg[j])
			{
				maxVal=dmg[j];
				ind=j;
			}
		}

		if(ind>=0)
		{
			isVisit[ind]=true;
			curDmg+=maxVal;
			ans.push_back(make_pair(t, ind+1));		
		}
	}

	if(t==10000) printf("NO\n");
	else
	{
		printf("YES\n");
		printf("%d %d\n", t, (int)ans.size());
		for(int i=0;i<ans.size();i++) printf("%d %d\n", ans[i].first, ans[i].second);
	}
	return 0;
}
