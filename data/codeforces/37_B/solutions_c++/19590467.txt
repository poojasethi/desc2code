#include <iostream>
#include <stdio.h>
#include <string.h>
#define MAXN 1010

using namespace std;

int n , mx , reg , m , dmg , cnt , times , x[ MAXN ] , y[ MAXN ] , use[ MAXN ];

int main()
{
	cin >> n >> mx >> reg , m = mx;
	for( register int i = 1 ; i <= n ; i++ ) cin >> x[i] >> y[i];
	memset( use , -1 , sizeof( use ) );
	while( 1 )
	{
		m -= dmg;
		m = min( mx , m + reg );
		if( m <= 0 ) break;
		int now = 0;
		for( register int i = 1 ; i <= n ; i++ )
			if( x[i] * mx >= m * 100 )
				if( y[i] > y[ now ] )
					now = i;
		if( !now )
			if( dmg <= reg )
			{
				cout << "NO" << endl;
				return 0;
			}
		use[ now ] = times++ , dmg += y[ now ] , x[ now ] = -1 , cnt += !!now;
	}
	cout << "YES" << endl;
	cout << times << " " << cnt << endl;
	for( register int i = 0 ; i <= times ; i++ )
		for( register int j = 1 ; j <= n ; j++ )
			if( use[j] == i )
				cout << i << " " << j << endl;
	return 0;
}
