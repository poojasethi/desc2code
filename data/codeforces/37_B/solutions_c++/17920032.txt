#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
#define N 1010

int n, mx, reg, dag, health, p[N], d[N], cur, T;
vector <pair<int, int> > v;
bool vis[N];

int main(){
   scanf("%d %d %d", &n, &mx, &reg);
   for(int i = 1; i <= n; i ++) scanf("%d %d", p+i, d+i);
   health = mx;
   for(T = 0; health > 0; T++){
      dag = 0; cur = 0;
      for(int i = 1; i <= n; i ++)if(!vis[i]){
         if(p[i] * mx >= 100 * health and dag < d[i]){ cur = i; dag = d[i]; }
      }
      if(cur != 0){
         v.push_back(make_pair(T, cur));
         vis[cur] = 1;
         reg -= dag;
      }
      health = min(health + reg, mx);
      if(cur == 0 and reg >= 0){ puts("NO"); return 0; }
   }
   puts("YES");
   printf("%d %d\n", T, v.size());
   for(int i = 0; i < v.size(); i ++) printf("%d %d\n", v[i].first, v[i].second);
}
