#include <stdio.h>
#include <algorithm>

using namespace std;

int n, i, j, it, mi, maa, u[11111], sv[11111], vv[11111], kol, ur;
double cur, ma, rg;
pair<double, double> a[1111];

int main(){
	scanf("%d %lf %lf", &n, &ma, &rg);
	cur = ma;
	for(i = 0; i<n; i++) scanf("%lf %lf", &a[i].first, &a[i].second);
	for(it = 0; it<100000; it++){
		if ((it & 1) != 2){
			mi = -1;
			maa = -1;
			for(i = 0; i<n; i++) if (!u[i] && (a[i].first >= (cur/ma)*100) && a[i].second>maa){
				maa = a[i].second;
				mi = i;
			}
			if (mi != -1){
				++kol;
				sv[kol] = mi+1;
				vv[kol] = it;
				u[mi] = 1;
				ur += maa;
			}
			cur -= ur;
			cur += rg;
			cur = min(cur, ma);
			if (cur<=0){
				puts("YES");
				printf("%d %d\n", it+1, kol);
				for(i = 1; i<=kol; i++) printf("%d %d\n", vv[i], sv[i]);
				return 0;
			}
		}
	}
	puts("NO");
	return 0;
}
