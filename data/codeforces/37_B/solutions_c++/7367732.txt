#include<cstdio>
#include<algorithm>
#include<vector>
#define pii pair<int,int>
#define A first
#define B second
using namespace std;
pii P[10001];
int ti = 0,sc,h,n,m,k,tag[10001];
vector <pii> ans;
int main(){
  scanf("%d%d%d",&n,&m,&k); h = m;
  for(int i=1,a,b;i<=n;i++){
    scanf("%d%d",&a,&b);
    P[i] = pii(a,b);
  }
  while(h > 0){
    int sv = -1;
    for(int i=1;i<=n;i++)
      if(tag[i] == 0 && h <= P[i].A*m/100 && (sv == -1 || P[i].B > P[sv].B)) 
    sv = i;
    if(sv != -1)
      ans.push_back(pii(ti,sv)), k -= P[sv].B, tag[sv] = 1;
    h += min(k,0); ti++;
    if(sv == -1 && k >= 0) break;
  }
  if(h <= 0){
    printf("YES\n%d %d\n",ti,ans.size());
    for(int i=0;i<ans.size();i++)
      printf("%d %d\n",ans[i].A,ans[i].B);
  }
  else printf("NO");
  return 0;
}
