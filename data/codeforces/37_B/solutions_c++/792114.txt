#include <iostream>
using namespace std;
int n,Max,reg,i,sum,m,ans,mt,ind,a[1001],b[1001],x[2002],y[2002];
int main()
{
    //ifstream cin("input.txt");
    //ofstream cout("output.txt");
	cin >> n >> Max >> reg;
	for (i=1;i<=n;i++)
		cin >> a[i] >> b[i];
    sum=Max;
    while (sum>0)
	{
        m++;
        mt=-1;
        for(i=1;i<=n;i++)
			if (sum*100<=a[i]*Max)
        if (b[i]>mt) {mt=b[i]; ind=i;}
        if (mt==-1 && reg>=0) {cout << "NO" << endl;return 0;}
        if (mt==-1) {sum+=reg;continue;}
		reg-=mt;
        a[ind]=-1;
        x[++ans]=m-1;
        y[ans]=ind;
        sum+=reg;
        if (sum>Max) sum=Max;
    }
	cout << "YES" << endl << m << " " << ans << endl;
    for (i=1;i<=ans;++i)
        cout << x[i] << " " << y[i] << endl;
    return 0;
}
