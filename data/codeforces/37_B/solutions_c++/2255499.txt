#include <iostream>
#include <algorithm>
using namespace std;
int main(void) {
  int n,health,regen,when=0,dec=0,dmg[1024],powr[1024],k=0,used[1024],heap[1024],top=0;
  cin>>n>>health>>regen;
  for(int i=0;i<n;++i)cin>>powr[i]>>dmg[i],(powr[i]<<=10)+=i;
  sort(powr,powr+n);
  for(int rest=health,x;when<12345&&rest>0;++when){
    rest=min(health,rest-dec+regen);
    if(rest<1)break;
    while(n>0&&(powr[n-1]>>10)*health>=rest*100)--n,heap[top++]=(dmg[x=(powr[n]&0x3ff)]<<10)+x,push_heap(heap,heap+top);
    if(top>0)dec+=heap[0]>>10,used[k++]=(when<<10)+(heap[0]&0x3ff)+1,pop_heap(heap,heap+top--);
  }
  if(when<10000){
    cout<<"YES"<<endl;
    cout<<when<<" "<<k<<endl;
    for(int i=0;i<k;++i)cout<<(used[i]>>10)<<" "<<(used[i]&0x3ff)<<endl;
  }
  else cout<<"NO"<<endl;
  return 0;
}
