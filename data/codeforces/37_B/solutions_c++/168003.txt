#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
#define fi first
#define se second
int n,hmax,nr,i,j,maxhp,hp,dmg,aux,reg,dmgmax,poz;
pair<int,pair<int,int> > a[1050];
pair<int,int> log[1050];
int folosit[1050];
int main()
{
	scanf("%d",&n);
	scanf("%d%d",&hp,&reg);
	maxhp=hp;
	for (i=1;i<=n;i++)
	{
		scanf("%d %d",&a[i].fi,&a[i].se.fi);
		a[i].se.se=i;
	}
	sort(a+1,a+n+1);
	reverse(a+1,a+n+1);
	nr=1;
	for (i=1;;i++)
	{
		aux=hp;
		hp-=dmg;
		hp+=reg;
		if (hp>maxhp)
			hp=maxhp;
		if (hp<=0)
		{
			printf("YES\n%d %d\n",i-1,nr-1);
			for (i=1;i<nr;i++)
				printf("%d %d\n",log[i].fi-1,log[i].se);
			return 0;
		}
		dmgmax=-1;
		for (j=1;j<=n;j++)
			if (!folosit[j])
			{
				if ((double)maxhp*a[j].fi/100>=hp)
					if (dmgmax<a[j].se.fi)
					{
						dmgmax=a[j].se.fi;
						poz=j;
					}

			}
		if (dmgmax!=-1)
		{
			dmg+=a[poz].se.fi;
			log[nr].fi=i;
			log[nr].se=a[poz].se.se;
			folosit[poz]=1;
			nr++;
			continue;
		}
		else
			if (aux<=hp&&dmg<=reg)
			{
				printf("NO\n");
				return 0;
			}
	}
	return 0;
}
