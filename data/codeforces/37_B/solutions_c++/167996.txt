#include<iostream>
#include<vector>
using namespace std;
int n,r,m;
vector< pair<int, int> > v(1000);
vector<bool> ok(1000,true);
int main()
{
	int p,d,dd = 0,t = 0,mm,nom,hb,h,k = 0;
	cin>>n>>m>>r;
	h = m;
	for(int i = 0; i < n; i++)
	{
		cin>>p>>d;
		v[i] = make_pair(m * p / 100,d);
	}
	vector< pair<int, int> > ans;
	while(true)
	{
		hb = h;
		h -= dd;
		h += r;
		if (h <= 0) 
		{
			break;
		}
		h = min(h,m);
		mm = 0;
		nom = -1;
		for(int i = 0; i < n; i++)
			if (ok[i] && v[i].first >= h && v[i].second > mm)
			{
				mm = v[i].second;
				nom = i;
			}
		if (nom != -1)
		{
			ans.push_back(make_pair(t,nom + 1));
			k++;
			ok[nom] = false;
			dd += mm;
		} else
		if (hb <= h)
		{
			cout<<"NO";
			return 0;
		}
		
		
		t++;
	}
	cout<<"YES"<<endl;
	cout<<t<<" "<<k<<endl;
	for(int i = 0; i < k; i++)
		cout<<ans[i].first<<" "<<ans[i].second<<endl;
	return 0;
}
