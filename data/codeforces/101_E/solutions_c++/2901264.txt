#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,p,x[22222],y[22222],cur,f[22222],g[22222],ans;
char s[44444];

long long solve(int x1,int y1,int x2,int y2)
{
    long long res=0;
    if (x1==x2)
    {
        for (int j=y1; j<=y2; j++)
        {
            res+=(x[x1]+y[j])%p;
            if (j>y1)
                s[cur++]='S';
        }
        return res;
    }
    int mid=(x1+x2)/2;
    for (int i=y1; i<=y2; i++)
        f[i]=g[i]=0;
    for (int i=x1; i<=mid; i++)
        for (int j=y1; j<=y2; j++)
        {
            if (j>y1 && f[j-1]>f[j])
                f[j]=f[j-1];
            f[j]+=(x[i]+y[j])%p;
        }
    for (int i=x2; i>mid; i--)
        for (int j=y2; j>=y1; j--)
        {
            if (j<y2 && g[j+1]>g[j])
                g[j]=g[j+1];
            g[j]+=(x[i]+y[j])%p;
        }
    int pos=y1;
    for (int j=y1; j<=y2; j++)
        if (f[j]+g[j]>res)
            res=f[j]+g[j],pos=j;
    solve(x1,y1,mid,pos);
    s[cur++]='C';
    solve(mid+1,pos,x2,y2);
    return res;
}

int main()
{
    scanf("%d%d%d",&n,&m,&p);
    for (int i=0; i<n; i++)
        scanf("%d",&x[i]);
    for (int i=0; i<m; i++)
        scanf("%d",&y[i]);
    ans=solve(0,0,n-1,m-1);
    printf("%d\n",ans);
    printf("%s\n",s);
    return 0;
}
