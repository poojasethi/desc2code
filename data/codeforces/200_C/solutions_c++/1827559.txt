#include <cstdio>
#include <string>
#include <iostream>
#include <map>
#include <algorithm>
using namespace std;
int cnt=0,s[15],sg[15],mg[15],B,rB,ts[15],tsg[15],tmg[15],r[15];
string a,b,c,nam[15];
map<string,int> lbl;
bool comp(int x,int y){
    return ts[x]<ts[y]||(ts[x]==ts[y]&&(tsg[x]-tmg[x]<tsg[y]-tmg[y]||(tsg[x]-tmg[x]==tsg[y]-tmg[y]&&(tsg[x]<tsg[y]||(tsg[x]==tsg[y]&&nam[x]>nam[y])))));
}
int main(){
    for(int i=1; i<=6; ++i) s[i]=sg[i]=mg[i]=0,r[i]=3;
    for(int i=1; i<=5; ++i){
        cin >> a >> b >> c;
        if (lbl.find(a)==lbl.end()) lbl[a]=++cnt,nam[cnt]=a;
        if (lbl.find(b)==lbl.end()) lbl[b]=++cnt,nam[cnt]=b;
        --r[lbl[a]],--r[lbl[b]];
        int x=(c[0]-'0'),y=(c[2]-'0');
        if (x>y) s[lbl[a]]+=3;
        else if (x<y) s[lbl[b]]+=3;
        else ++s[lbl[a]],++s[lbl[b]];
        sg[lbl[a]]+=x,mg[lbl[a]]+=y;
        sg[lbl[b]]+=y,mg[lbl[b]]+=x;
    }
    for(int i=1; i<=6; ++i)
        if (nam[i]=="BERLAND") B=i;
    for(int i=1; i<=6; ++i)
        if (r[i]==1&&i!=B) rB=i;
    int m=1,m2=1;
    for(int i=1; i<=6; ++i)
        if (s[m]<s[i]) m=i;
    for(int i=1; i<=6; ++i)
        if (s[m2]<s[i]||m2==m) m2=i;
    int X,Y;
    for(int diff=1;diff<=30;++diff)
        for(Y=0;Y<=30;++Y){
            X=Y+diff;
            for(int i=1; i<=6; ++i) ts[i]=s[i],tsg[i]=sg[i],tmg[i]=mg[i];
            ts[B]+=3;
            tsg[B]+=X,tmg[B]+=Y;
            tsg[rB]+=Y,tmg[rB]+=X;
            int u[15];
            for(int i=1; i<=6; ++i) u[i]=i;
            for(int i=1; i<=6; ++i)
                for(int j=1; j<=6-i; ++j){
                    if (comp(u[j],u[j+1])) swap(u[j],u[j+1]);
                }
            if (u[1]==B||u[2]==B){
                printf("%d:%d\n",X,Y);
                return 0;
            }
        }
    printf("IMPOSSIBLE\n");
    return 0;
}
