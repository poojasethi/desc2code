#include <iostream>
#include <algorithm>
#include <string.h>
#include <cstring>
#include <string>
#include <map>
#include <cstdio>
using namespace std;
map<string,int> fs,jq,sq,ch;
int X,Y;
int main(){
	for (int i=1;i<=5;i++){
		string name1,name2,c;
		cin>>name1>>name2>>c;
		int score1,score2;
		sscanf(c.c_str(),"%d:%d",&score1,&score2);
		if (score1>score2){
			fs[name1]+=3;
			jq[name1]+=score1;
			sq[name1]+=score2;
			jq[name2]+=score2;
			sq[name2]+=score1;
		}
		if (score1<score2){
			fs[name2]+=3;
			jq[name2]+=score2;
			sq[name2]+=score1;
			jq[name1]+=score1;
			sq[name1]+=score2;
		}
		if (score1==score2){
			fs[name1]+=1;
			fs[name2]+=1;
			jq[name2]+=score2;
			sq[name2]+=score1;
			jq[name1]+=score1;
			sq[name1]+=score2;
		}
		ch[name1]++;
		ch[name2]++;
	}
	string me="BERLAND";
	string other;
	for (map<string,int>::iterator it=ch.begin();it!=ch.end();it++){
		if (it->second==2 && it->first!="BERLAND"){
			other=it->first;
		}
	}
	for (int DIF=1;DIF<=100;DIF++){
		for (Y=0;Y<=100;Y++){
			
		
			//cout<<"DIF="<<DIF<<" "<<"Y="<<Y<<endl;
			X=Y+DIF;
			//cout<<"X="<<X<<" "<<"Y="<<Y<<endl;
			fs[me]+=3;
			jq[me]+=X;
			sq[me]+=Y;
			jq[other]+=Y;
			sq[other]+=X;
			int cnt=0;
			for (map<string,int>::iterator it=ch.begin();it!=ch.end();it++){
				if (it->first==me){
					continue;
				}
				if (fs[it->first]>fs[me]){
					cnt++;
					continue;
				}
				if (fs[it->first]<fs[me]){
					continue;
				}
				int v=jq[it->first]-sq[it->first];
				int vme=jq[me]-sq[me];
				if (v>vme){
					cnt++;
					continue;
				}
				if (v<vme){
					continue;
				}
				v=jq[it->first];
				vme=jq[me];
				if (v>vme){
					cnt++;
					continue;
				}
				if (v<vme){
					continue;
				}
				if (it->first<me){
					cnt++;
				}
			}
			//cout<<cnt<<endl;
			if (cnt<=1){
				cout<<X<<":"<<Y<<endl;
				return 0;
			}
			fs[me]-=3;
			jq[me]-=X;
			sq[me]-=Y;
			jq[other]-=Y;
			sq[other]-=X;
		}
	}
	cout<<"IMPOSSIBLE";
	return 0;
}
