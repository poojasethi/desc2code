    #include <iostream>
    #include <string>
    #include <set>  
    #include <algorithm>
    #include <vector>
    #include <cmath>
    #include <queue>
    #include <cstring>
    #include <cstdio>
    #include <map>
using namespace std;
map <string,int> sc,diff,ball,cnt;
int main()
{
	string a,b,c;
	for(int i=0;i<5;i++)
	{
		cin>>a>>b>>c;
		int g1,g2;
		sscanf(c.c_str(),"%d:%d",&g1,&g2);
		if(g1>g2)sc[a]+=3;
		else if(g1==g2)sc[a]++,sc[b]++;
		else sc[b]+=3;
		diff[a]+=g1-g2;diff[b]+=g2-g1;
		ball[a]+=g1;ball[b]+=g2;
		cnt[a]++;cnt[b]++;
	}
	string m="BERLAND",o;
	for(map<string,int>::iterator it=cnt.begin();it!=cnt.end();it++)
		if(it->second<3&&it->first!=m)o=it->first;
	for(int d=1;d<20;d++)
		for(int y=0;y<20;y++)
		{
			int x=y+d;
			sc[m]+=3;
			diff[m]+=d;diff[o]-=d;
			ball[m]+=x;ball[o]+=y;
			int ccnt=0;
			for(map<string,int>::iterator it=cnt.begin();it!=cnt.end();it++)
			{
				string s=it->first;
				if(s==m)continue;
				if(sc[s]>sc[m]){ccnt++;continue;}
				if(sc[s]<sc[m])continue;
				if(diff[s]>diff[m]){ccnt++;continue;}
				if(diff[s]<diff[m])continue;
				if(ball[s]>ball[m]){ccnt++;continue;}
				if(ball[s]<ball[m])continue;
				if(s<m){ccnt++;continue;}
			}
			if(ccnt<=1){cout<<x<<":"<<y;return 0;}
			sc[m]-=3;
			diff[o]+=d;diff[m]-=d;
			ball[o]-=y;ball[m]-=x;
		}
	cout<<"IMPOSSIBLE";
	return 0;
}
