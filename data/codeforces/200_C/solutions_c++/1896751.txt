#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <cstdlib>
#include <algorithm>
#include<map>
using namespace std;
map<string,int> mp;
char s1[30],s2[30];
int win[5],mis[5] ;
int num[5] ;
int goal[5] ;
int ans1, ans2 , x , y  ;
bool can ;
string str[5];
int top = 0 ;
int hash( char s[] ) {
	string ns = "" ;
	for ( int i = 0 ; i < strlen(s) ; i ++ ){
		ns += s[i] ;
	}
	if ( mp[ns] )
		return mp[ns] ;
	top ++ ;
	mp[ns] = top ;
	str[top] = ns ;
	return top ;
}
bool pass( int x , int y , int a , int b ){
	int txw = win[x] , txm = mis[x] ;
	int tyw = win[y] , tym = mis[y] ;
	win[x] += a ; mis[x] += b ;
	win[y] += b ; mis[y] += a ;
	int ans = 0 ;
	for ( int i = 1 ; i <= 4 ; i ++ ){
		if ( i == x ) continue ;
		if ( goal[x] < goal[i] )
			ans ++ ;
		else if ( goal[x] == goal[i] ) {
			if ( win[x] - mis[x] < win[i] - mis[i] )
				ans ++ ;
			else if ( win[x] - mis[x] == win[i] - mis[i] ){
				if ( win[x] < win[i] )
					ans ++ ;
				else if ( win[x] == win[i] ) {
					if ( str[x] > str[i])
						ans ++ ;
				}
			}
		}
	}
	win[x] = txw , mis[x] = txm ;
	win[y] = tyw , mis[y] = tym ;
	if ( ans <= 1 )
		return true ;
	return false ;
}
void cal( int x , int y ){
	for ( int d = 1 ; d <= 100 ; d ++ ) {
		for ( int a = d ; a <= 100 ; a ++ ){
			int b = a - d ;
			if ( pass( x , y , a, b ) ) {
				can = true ;
				ans1 = a ; ans2 = b ;
				return ;
			}
		}
	}
	return ;
}
int main(){
	for ( int i = 1 ; i <= 5 ;  i ++ ){
		int a , b ;
		scanf( "%s %s %d:%d" , s1 , s2 , &a , &b ) ;
		int x = hash( s1 ) ; int y = hash( s2 ) ;
		num[x] ++ ; num[y] ++ ;
		if ( a > b ) goal[x] +=3 ;
		if ( b > a ) goal[y] +=3 ;
		if ( a == b ){ goal[x] ++ ; goal[y] ++ ; }
		win[x] += a ; mis[x] += b ;
		win[y] += b ; mis[y] += a ;
	}
	for ( int i = 1 ; i <= 4 ; i ++ ){
		if ( str[i] == "BERLAND" )
			x = i ;
		else if ( num[i] == 2 )
			y = i ;
	}
	goal[x] += 3 ;
	cal( x , y ) ;
	if ( can )
		printf("%d:%d\n" , ans1 , ans2 ) ;
	else
		puts("IMPOSSIBLE");
}
