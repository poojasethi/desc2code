#include<iostream>
#include<cstdio>
#include<memory.h>
using namespace std;
struct aa{int q,m,p;};
aa f[200];
int i,j,p,su,ss1,ss2,t1,t2,sum[200],len;
string ch1,ch2,s[10];
bool da(int t)
{
	if (t==ss1)
		return false;
	if (f[t].p!=f[ss1].p)
		return f[t].p>f[ss1].p;
	if (f[t].m!=f[ss1].m)
		return f[t].m>f[ss1].m;
	if (f[t].q!=f[ss1].q)
		return f[t].q>f[ss1].q;
	return s[t]<s[ss1];
}
int get(string a)
{
	for (int i=1;i<=len;i++)
	if (s[i]==a)
	return i;
	s[++len]=a;
	return len;
}
void read(int &k)
{
	char q;
	for (q=getchar();q<'0'||q>'9';q=getchar());
	k=q-'0';
}
int main()
{
	for (i=1;i<=5;i++)
	{
		cin>>ch1>>ch2;
		ss1=get(ch1);
		ss2=get(ch2);
		sum[ss1]++;
		sum[ss2]++;
		read(t1);
		read(t2);
		f[ss1].m+=t1-t2;
		f[ss2].m+=t2-t1;
		f[ss1].q+=t1;
		f[ss2].q+=t2;
		if (t1>t2)
			f[ss1].p+=3;
		else
		if (t1<t2)
			f[ss2].p+=3;
		else
		{
			f[ss1].p++;
			f[ss2].p++;
		}
	}
	for (i=1;i<=4;i++)
	if (s[i]=="BERLAND")
		ss1=i;
	else
	if (sum[i]<3)
		ss2=i;
	for (i=1;i<=35;i++)
	for (j=0;j<=35;j++)
	{
		t2=j;
		t1=j+i;
		f[ss1].m+=t1-t2;
		f[ss2].m+=t2-t1;
		f[ss1].q+=t1;
		f[ss2].q+=t2;
		if (t1>t2)
			f[ss1].p+=3;
		else
		if (t1<t2)
			f[ss2].p+=3;
		else
		{
			f[ss1].p++;
			f[ss2].p++;
		}
		su=0;
		for (p=1;p<=4;p++)
		if (da(p))
			su++;
		if (su<2)
		{
			printf("%d:%d",t1,t2);
			return 0;
		}
		f[ss1].m-=t1-t2;
		f[ss2].m-=t2-t1;
		f[ss1].q-=t1;
		f[ss2].q-=t2;
		if (t1>t2)
			f[ss1].p-=3;
		else
		if (t1<t2)
			f[ss2].p-=3;
		else
		{
			f[ss1].p--;
			f[ss2].p--;
		}
	}
	printf("IMPOSSIBLE");
}
