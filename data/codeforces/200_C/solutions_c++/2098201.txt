#include<cstring>
#include<cstdio>
#include<algorithm>
#include<string>
using namespace std;
struct team{
	int score;
	int miss,point;
	string name;
}a[4];
int len=0;
int find(string s){
	for(int i=0;i<len;i++) if(a[i].name == s) return i;
	a[len].score = a[len].miss = a[len].point = 0;
	a[len++].name = s;
	return len-1;
}
int id[4];
int cmp(const int &i,const int &j){
	if(a[i].point!=a[j].point) return a[i].point > a[j].point;
	if(a[i].score!=a[j].score) return a[i].score > a[j].score;
	if(a[i].miss!=a[j].miss) return a[i].miss > a[j].miss;
	return a[i].name < a[j].name;
}
int check(int x){
	for(int i=0;i<4;i++) id[i]=i;
	sort(id,id+4,cmp);
	if(id[0]==x||id[1]==x) return 1;
	return 0;
}
int main(){
//	freopen("in.txt","r",stdin);
	char s1[100],s2[100];
	int k1,k2;
	int cnt[4]={0};
	for(int i=0;i<5;i++){
		scanf("%s %s %d:%d",s1,s2,&k1,&k2);
		int t1 = find(string(s1));
		int t2 = find(string(s2));
		cnt[t1]++; cnt[t2]++;
		a[t1].score += (k1-k2);
		a[t2].score += (k2-k1);
		a[t1].miss += k1;
		a[t2].miss += k2;
		if(k1>k2) a[t1].point +=3;
		else if(k1==k2) a[t1].point++,a[t2].point++;
		else a[t2].point +=3; 
	}
	int c,b;
	for(int i=0;i<4;i++) if(a[i].name == "BERLAND") c = i;
	for(int i=0;i<4;i++) if(cnt[i]==2&&i!=c) b = i;
	int ok=0;
	for(int k=1;!ok&&k<=1000;k++){
		for(int j=0;j<=1000&&!ok;j++){
			int i = k+j;
			a[c].point +=3;
			a[c].score += (i-j);
			a[b].score += (j-i);
			a[c].miss += i;
			a[b].miss += j;
			if(check(c)){
				printf("%d:%d",i,j);
				ok=1;
			}else {
				a[c].point -=3;
				a[c].score -= (i-j);
				a[b].score -= (j-i);
				a[c].miss -= i;
				a[b].miss -= j;
			}
		}
	}
	if(!ok) printf("IMPOSSIBLE");
	return 0;
}
