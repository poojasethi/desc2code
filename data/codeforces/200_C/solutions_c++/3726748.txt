#include <iostream>
#include <map>
#include <string>
#include <vector>
#include <algorithm>
#include <cstdlib>

using namespace std;

map<string,int> p,gm,gr,d;

vector<string> v;

bool compara(string s1,string s2)
{
  if (p[s1]>p[s2]) return true;
  if (p[s1]<p[s2]) return false;
  if (gm[s1]-gr[s1]>gm[s2]-gr[s2]) return true;
  if (gm[s1]-gr[s1]<gm[s2]-gr[s2]) return false;
  if (gm[s1]>gm[s2]) return true;
  if (gm[s1]<gm[s2]) return false;
  return s1<s2;;
}

int main()
{
  for (int i=0;i<5;i++) {
    string s1,s2;
    int g1,g2;
    char c;
    cin>>s1>>s2>>g1>>c>>g2;
    gm[s1]+=g1;
    gr[s1]+=g2;
    gm[s2]+=g2;
    gr[s2]+=g1;
    d[s1]++;
    d[s2]++;
    if (g1>g2)
      p[s1]+=3;
    else if (g1<g2)
      p[s2]+=3;
    else {
      p[s1]++;
      p[s2]++;
    }
  }
  string rival;
  for (map<string,int>::iterator it=d.begin();it!=d.end();it++) {
    v.push_back(it->first);
    if (it->first!="BERLAND" and it->second==2)
      rival=it->first;
  }
  p["BERLAND"]+=3;
  for (int diferencia=1;diferencia<40;diferencia++) {
    for (int grival=0;grival<40;grival++) {
      gm["BERLAND"]+=grival+diferencia;
      gm[rival]+=grival;
      gr["BERLAND"]+=grival;
      gr[rival]+=grival+diferencia;
      sort(v.begin(),v.end(),compara);
      if (v[0]=="BERLAND" or v[1]=="BERLAND") {
	cout<<grival+diferencia<<":"<<grival<<endl;
	exit(0);
      }
      gm["BERLAND"]-=grival+diferencia;
      gm[rival]-=grival;
      gr["BERLAND"]-=grival;
      gr[rival]-=grival+diferencia;
    }
  }
  cout<<"IMPOSSIBLE"<<endl;
}
