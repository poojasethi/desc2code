using namespace std;
map<string,int>fs,jq,sq,cs;
int main(){
	string a,b,c;
	for(int p=0;p<5;p++)
	{
		cin>>a>>b>>c;
		int s,t;
		char f=1;
		sscanf(c.c_str(),"%d:%d",&s,&t);
		if(s>t)fs[a]+=3;
		if(s<t)fs[b]+=3;
		if(s==t){fs[a]++;fs[b]++;}
		jq[a]+=s;sq[a]+=t;
		jq[b]+=t;sq[b]+=s;
		cs[a]++;cs[b]++;
	}
	string me="BERLAND",other;
	for(map<string,int>::iterator it=cs.begin();it!=cs.end();it++)
	if(it->second==2&&it->first!="BERLAND")other=it->first;
	for(int dif=1;dif<=40;dif++)
	for(int y=0;y<=40;y++){
		int x=y+dif;
		fs[me]+=3;
		jq[me]+=x;sq[me]+=y;
		jq[other]+=y;sq[other]+=x;
		int cnt=0;
		for(map<string,int>::iterator it=cs.begin();it!=cs.end();it++)
		{
			if(it->first==me)continue;
			if(fs[it->first] > fs[me]){++cnt;continue;}
			if(fs[it->first] < fs[me])continue;
			int v=(jq[it->first]-sq[it->first]);
			int vm=jq[me]-sq[me];
			if(v>vm){++cnt;continue;}
			if(v<vm)continue;
			v=jq[it->first];
			vm=jq[me];
			if(v>vm){++cnt;continue;}
			if(v<vm)continue;
			if(it->first<me){++cnt;continue;}
		}
		if(cnt<=1){
			cout<<x<<":"<<y;
			return 0;
		}
		jq[me]-=x;sq[me]-=y;
		jq[other]-=y;sq[other]-=x;
		fs[me]-=3;
	}
	cout<<"IMPOSSIBLE";
	return 0;}
