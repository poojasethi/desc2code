// C. Football Championship
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <string>
#include <iostream>
#include <functional>
#include <algorithm>
#include <vector>
#include <map>

using namespace std;

struct team {
    int c, p, g, m;
    string name;
    team() { c = p = g = m = 0; }
    bool operator<(const team &t) const {
        if (p != t.p) return p > t.p;
        int d1 = g - m, d2 = t.g - t.m;
        if (d1 != d2) return d1 > d2;
        if (g != t.g) return g > t.g;
        return name < t.name;
    }
};

int main(int argc, char *argv[])
{
    char name1[32], name2[32], s[32];

    map<string, team> t;
    for (int i = 0; i < 5; i++) {
        cin >> name1 >> name2 >> s;
        int s1, s2;
        sscanf(s, "%d:%d", &s1, &s2);
        team &t1 = t[name1], &t2 = t[name2];
        t1.c++; t2.c++;
        t1.g += s1, t1.m += s2;
        t2.g += s2, t2.m += s1;
        if (s1 > s2) t1.p += 3;
        else if (s2 > s1) t2.p += 3;
        else t1.p += 1, t2.p += 1;
    }
    
    vector<team> a;
    for (map<string, team>::iterator it = t.begin(); it != t.end(); ++it) {
        it->second.name = it->first;
        a.push_back(it->second);
    }
    
    int ii, jj;
    for (int i = 0; i < a.size(); i++)
        if (a[i].name == "BERLAND") ii = i;
        else if (a[i].c == 2) jj = i;
    
    a[ii].p += 3;
    for (int i = 1; i < 100; i++)
        for (int j = 0; j < 100; j++) {
            vector<team> t = a;
            t[ii].g += j + i, t[ii].m += j;
            t[jj].g += j, t[jj].m += i + j;
            sort(t.begin(), t.end());
            if (t[0].name == "BERLAND" || t[1].name == "BERLAND") {
                cout << (i + j) << ":" << j << endl;
                return 0;
            }
        }
    
    cout << "IMPOSSIBLE" << endl;

    return 0;
}
