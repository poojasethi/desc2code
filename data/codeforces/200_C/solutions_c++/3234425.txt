#include <map>
#include <iostream>
#include <string>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int sc[6], gl[6], ms[6], in[6];
map<string, int> chg;
map<int, string> inv;

bool comp(int a, int b){
	if(sc[a] != sc[b]) return sc[a] > sc[b];
	if(gl[a]-ms[a] != gl[b]-ms[b]) return gl[a]-ms[a] > gl[b]-ms[b];
	if(gl[a] != gl[b]) return gl[a] > gl[b];
	return inv[a].compare(inv[b]) < 0;
}

int main (){
	string O = "BERLAND";
	chg[O] = 1; int cnt = 2;
	for(int i = 0; i < 5; i++){
		string A, B; int x, y;
		cin >> A >> B;
		scanf("%1d%*c%1d", &x, &y);
		if(!chg[A]) chg[A] = cnt++;
		int a = chg[A]; inv[a] = A;
		if(!chg[B]) chg[B] = cnt++;
		int b = chg[B]; inv[b] = B;
		in[a]++, in[b]++;
		gl[a] += x, ms[a] += y;
		gl[b] += y, ms[b] += x;
		if(x == y) sc[a] ++, sc[b] ++;
		if(x > y) sc[a] += 3;
		if(x < y) sc[b] += 3;
	}
	int a = 1, b = 0;
	for(int i = 1; i <= 4; i++) if(in[i] != 3) b = i;
	for(int dif = 1; dif <= 100; dif++){
		for(int y = 0; y <= 100; y++){
			int x = y + dif; if(x < 0) continue;
			if(x == y) sc[a] ++, sc[b] ++;
			if(x > y) sc[a] += 3;
			if(x < y) sc[b] += 3;
			gl[a] += x, ms[a] += y;
			gl[b] += y, ms[b] += x;
			int per[4] = {1, 2, 3, 4};
			sort(per, per+4, comp);
			if(per[0] == 1 || per[1] == 1)
				return printf("%d:%d\n", x, y), 0;
			if(x == y) sc[a] --, sc[b] --;
			if(x > y) sc[a] -= 3;
			if(x < y) sc[b] -= 3;
			gl[a] -= x, ms[a] -= y;
			gl[b] -= y, ms[b] -= x;
		}
	}
	puts("IMPOSSIBLE");
}
