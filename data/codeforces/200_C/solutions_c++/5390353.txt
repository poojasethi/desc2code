#include<iostream>
#include<cstdio>
#include<string> 
#include<algorithm>
#include<map>
using namespace std;
map<string,int> hash;
string s[10],sa,sb;
int a,b,c[10],d[10],e[10],cnt[10],n,i,j,x,y,p,sc;
char ch;

bool win()
{
    int i,k=0,w;
    for(i=2;i<=4;i++)
        if(c[i]>c[1]) k++;
    if(k>1) return 0;
    w=1-k; k=0;
    for(i=2;i<=4;i++)
        if(c[i]==c[1]&&d[1]<d[i]) k++;
    if(k>w) return 0;
    w-=k; k=0;
    for(i=2;i<=4;i++)
        if(c[1]==c[i]&&d[1]==d[i]&&e[1]<e[i]) k++;
    if(k>w) return 0;
    w-=k; k=0;
    for(i=2;i<=4;i++)
        if(c[1]==c[i]&&d[1]==d[i]&&e[1]==e[i]&&s[1]>s[i]) k++;
    return k<=w;
}

int main()
{
    hash["BERLAND"]=n=1; s[1]="BERLAND";
    for(i=1;i<6;i++)
    {
        cin>>sa>>sb>>x>>ch>>y;
        if(!hash[sa]) hash[sa]=++n,s[n]=sa;
        if(!hash[sb]) hash[sb]=++n,s[n]=sb;
        a=hash[sa],b=hash[sb];
        if(x>y) c[a]+=3;
        else if(x==y) c[a]++,c[b]++;
        else c[b]+=3;
        d[a]+=x-y; d[b]+=y-x;
        e[a]+=x; e[b]+=y;
        cnt[a]++,cnt[b]++;
    }
    for(i=2;i<=4;i++)
        if(cnt[i]<3) p=i;
    for(i=2;i<=4;i++)
        if(i!=p) sc=max(sc,c[i]);
    c[1]+=3;
    for(i=1;i<=100;i++)
    {
        d[1]+=i,d[p]-=i;
        for(j=i;j<=100;j++)
        {
            e[1]+=j,e[p]+=j-i;
            if(win()) {cout<<j<<':'<<j-i<<endl; return 0;}
            e[1]-=j,e[p]-=j-i;
        }
        d[1]-=i,d[p]+=i;
    }
    puts("IMPOSSIBLE");
    return 0;
}
