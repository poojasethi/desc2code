#include <vector>
#include <iostream>
#include <algorithm>
#include <iomanip>
#include <cmath>
#include <string>
using namespace std;
int x1,x2,i,j,p[4],g1[4],g2[4],k=1,ans1=100,ans2=0,m[4];
string s1,s2,s[4];
bool ok(int x, int y)
{
	int j1,j2;
	for(int i=0;i<4;i++)
		if (s[i]=="BERLAND")
			j1=i;
		else if (m[i]==2)
				j2=i;
	p[j1]+=3;
	g1[j1]+=x; g1[j2]+=y;
	g2[j1]+=y; g2[j2]+=x;
	int mj1=0,mj2=1;
	for(int i=2;i<4;i++)
	{
		if (p[i]>p[mj1])
			mj1=i;
		if (p[i]==p[mj1]&&g1[i]-g2[i]>g1[mj1]-g2[mj1])
			mj1=i;
		if (p[i]==p[mj1]&&g1[i]-g2[i]==g1[mj1]-g2[mj1]&&g1[i]>g1[mj1])
			mj1=i;
		if (p[i]==p[mj1]&&g1[i]-g2[i]==g1[mj1]-g2[mj1]&&g1[i]==g1[mj1]&&s[i]<s[mj1])
			mj1=i;
	}
	for(int i=0;i<4;i++)
	{
		if (i==mj1) continue;
		if (p[i]>p[mj2])
			mj2=i;
		if (p[i]==p[mj2]&&g1[i]-g2[i]>g1[mj2]-g2[mj2])
			mj2=i;
		if (p[i]==p[mj2]&&g1[i]-g2[i]==g1[mj2]-g2[mj2]&&g1[i]>g1[mj2])
			mj2=i;
		if (p[i]==p[mj2]&&g1[i]-g2[i]==g1[mj2]-g2[mj2]&&g1[i]==g1[mj2]&&s[i]<s[mj2])
			mj2=i;
	}
	p[j1]-=3;
	g1[j1]-=x; g1[j2]-=y;
	g2[j1]-=y; g2[j2]-=x;
	if (mj1==j1||mj2==j1)
		return true;
	else return false;
}
int main()
{
	s[0]="BERLAND";
	for(i=0;i<5;i++)
	{
		char c;
		cin>>s1>>s2>>x1>>c>>x2;
		int j1=-1,j2=-1;
		for(j=0;j<k;j++)
		{
			if (s1==s[j])
				j1=j;
			if (s2==s[j])
				j2=j;
		}
		if (j1==-1)
		{
			s[k]=s1;
			j1=k;
			k++;
		}
		if (j2==-1)
		{
			s[k]=s2;
			j2=k;
			k++;
		}
		m[j1]++; m[j2]++;
		if (x1>x2)
			p[j1]+=3;
		if (x1==x2)
		{
			p[j1]++;
			p[j2]++;
		}
		if (x1<x2)
			p[j2]+=3;
		g1[j1]+=x1; g1[j2]+=x2;
		g2[j1]+=x2; g2[j2]+=x1;
	}
	for(j=0;j<100;j++)
		for(i=j+1;i<100;i++)
			if (ok(i,j)&&i-j<ans1-ans2)
			{
				ans1=i;
				ans2=j;
			}
	if (ans1==100)
		cout<<"IMPOSSIBLE"<<endl;
	else cout<<ans1<<":"<<ans2<<endl;
	return 0;
}
