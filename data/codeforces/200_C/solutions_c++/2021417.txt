#include<cstdio>
#include<algorithm>
#include<string>
#include<iostream>
using namespace std;
int s,ans[2];
struct Team
{
	string Name;
	int w,m,p,g;
	bool operator<(const Team &X)const
	{
		if (p<X.p)return 0; else if (p>X.p)return 1;
		if (w-m<X.w-X.m)return 0; else if (w-m>X.w-X.m)return 1;
		if (w<X.w)return 0; else if (w>X.w)return 1;
		if (Name>X.Name)return 0; else if (Name<X.Name)return 1;
	}
	void operator=(const Team &X)
	{
		Name=X.Name; p=X.p; w=X.w; m=X.m; g=X.g;
	}
}T[4],TT[4];

void Add(string Name,int a,int b)
{
	int pos=-1;
	for (int j=0;j<s;j++)
		if (T[j].Name==Name) pos=j;
	if (pos==-1) {T[s].Name=Name,pos=s++;}
	T[pos].w+=a; T[pos].m+=b;
	if (a>b) T[pos].p+=3; else if (a==b)T[pos].p++;
	T[pos].g++;
}

int main()
{
	string x,y; int a,b; char t; 
	for (int i=0;i<5;i++)
	{
		cin>>x>>y;
		scanf("%d%c%d",&a,&t,&b);
		Add(x,a,b); Add(y,b,a);
	}
	for (int i=0;i<4;i++)
		if (T[i].g==2&&T[i].Name=="BERLAND") x=T[i].Name; else 
		if (T[i].g==2) y=T[i].Name;
	ans[0]=18; ans[1]=0;
	int flag=1;
	for (int i=0;i<4;i++)TT[i]=T[i];
	for (int i=1;i<=18;i++)
		for (int j=i-1;j>=0;j--)
		{
			for (int k=0;k<4;k++) T[k]=TT[k];
			Add(x,i,j);
			Add(y,j,i);
			sort(T,T+4);
			if (T[0].Name==x||T[1].Name==x)
				if (ans[0]-ans[1]>i-j)
				{
					ans[0]=i; ans[1]=j; flag=0;
				}
		}
	if (flag)printf("IMPOSSIBLE"); else printf("%d:%d",ans[0],ans[1]);
	return 0;
}
