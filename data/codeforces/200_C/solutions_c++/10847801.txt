#include<iostream>
#include<cstdio>
#include<map>
#define b "BERLAND"
using namespace std;
map<string,int>p,goal,miss,t;
string x,y,r;
int m,n;
int main()
{
	for(int i=0;i<5;i++)
	{
		cin>>x>>y;
		scanf("%d:%d",&m,&n);
		t[x]++;t[y]++;
		goal[x]+=m;miss[x]+=n;
		goal[y]+=n;miss[y]+=m;
		if(m>n)p[x]+=3;
		else if(m==n)p[x]++,p[y]++;
		else p[y]+=3;
	}
	for(map<string,int>::iterator it=t.begin();it!=t.end();it++)
	{
		if(it->first=="BERLAND")continue;
		if(it->second==2)r=it->first;
	}
	for(int dif=1;dif<43;dif++)
	for(int Y=0;Y<43;Y++)
	{
		int X=dif+Y;
		p[b]+=3;
		goal[b]+=X;
		miss[b]+=Y;
		goal[r]+=Y;
		miss[r]+=X;
		int num=0;
		for(map<string,int>::iterator it=t.begin();it!=t.end();it++)
		{
			if(p[it->first]>p[b]){num++;continue;}
			if(it->first==b)continue;
			if(p[it->first]<p[b])continue;
			int v=goal[b]-miss[b];
			int vr=goal[it->first]-miss[it->first];
			if(v<vr){num++;continue;}
			if(v>vr)continue;
			if(goal[b]>goal[it->first])continue;
			if(goal[b]<goal[it->first]){num++;continue;}
			if(it->first<b){num++;continue;}
		}
		if(num<=1)
		{
			cout<<X<<":"<<Y<<endl;
			return 0;
		}
		p[b]-=3;
		goal[b]-=X;
		miss[b]-=Y;
		goal[r]-=Y;
		miss[r]-=X;
	}
	cout<<"IMPOSSIBLE"<<endl;
	return 0;
}
