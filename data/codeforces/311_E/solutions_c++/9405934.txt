#include<bits/stdc++.h>
using namespace std;
#define INF 2147483647
#define S n+m+1
#define T n+m+2
int n,m,g,fir[20001],next[100001],to[100001],flow[100001],sz=1;
int sex[10001],d[20001],q[20001],cur[20001],ans;
void add(int x,int y,int f)
{
	next[++sz]=fir[x];fir[x]=sz;to[sz]=y;flow[sz]=f;
	next[++sz]=fir[y];fir[y]=sz;to[sz]=x;flow[sz]=0;
}
bool bfs()
{
	for(int i=1;i<=n+m+2;i++)d[i]=INF;
	d[S]=0;int l=0,r=0;q[++r]=S;
	while(l<r)
	{
		int x=q[++l];cur[x]=fir[x];
		for(int i=fir[x];i;i=next[i])
		if(flow[i]&&d[to[i]]==INF)
		{
			d[to[i]]=d[x]+1;
			q[++r]=to[i];
		}
	}
	return d[T]!=INF;
}
int dfs(int x,int p)
{
	if(x==T)return p;int o=p;
	for(;cur[x];cur[x]=next[cur[x]])
	if(flow[cur[x]]&&d[to[cur[x]]]==d[x]+1)
	{
		int delt=dfs(to[cur[x]],min(p,flow[cur[x]]));
		flow[cur[x]]-=delt;flow[cur[x]^1]+=delt;
		p-=delt;if(!p)break;
	}
	return o-p;
}
int main()
{
	scanf("%d%d%d",&n,&m,&g);
	for(int i=1;i<=n;i++)scanf("%d",&sex[i]);
	for(int i=1,v;i<=n;i++)
	{
		scanf("%d",&v);
		if(sex[i])add(i,T,v);
		else add(S,i,v);
	}
	for(int i=1;i<=m;i++)
	{
		int type,w,k;
		scanf("%d%d%d",&type,&w,&k);
		ans+=w;
		while(k--)
		{
			int t;scanf("%d",&t);
			if(type)add(t,i+n,INF);
			else add(i+n,t,INF);
		}
		int p;scanf("%d",&p);
		if(p)w+=g;
		if(type)add(i+n,T,w);
		else add(S,i+n,w);
	}
	while(bfs())ans-=dfs(S,INF);
	cout<<ans<<endl;
}
