#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;

int lev[1111111];
int q[1111111];
int S,T,edges;
int fir[1111111];
int next[1111111];
int tar[1111111];
int flw[1111111];

inline void edgeins(int u,int v,int f)
{
	next[edges]=fir[u];
	fir[u]=edges;
	tar[edges]=v;
	flw[edges]=f;
	edges++;
	next[edges]=fir[v];
	fir[v]=edges;
	tar[edges]=u;
	flw[edges]=0;
	edges++;
}

bool bfs()
{
	int h,t,u,v,pt;
	memset(lev,0,sizeof(lev));
	q[0]=T; t=1;
	lev[T]=1;
	for (h=0;h<t;h++)
		for (u=q[h],pt=fir[u];pt!=0;pt=next[pt])
			if (flw[pt^1])
				if (lev[v=tar[pt]]==0)
				{
					q[t++]=v;
					lev[v]=lev[u]+1;
				}
	return lev[S]!=0;
}

int dinic(int nd,int del)
{
	if (nd==T) return del;
	int ret=0,cur,pt;
	for (pt=fir[nd];del!=0&&pt!=0;pt=next[pt])
		if (cur=min(del,flw[pt]))
			if (lev[nd]==lev[tar[pt]]+1)
			{
				cur=dinic(tar[pt],cur);
				flw[pt]-=cur;
				flw[pt^1]+=cur;
				del-=cur;
				ret+=cur;
			}
	return ret;
}

int gend[1111111];
int pc[111];
int main()
{
//	freopen("311E.in","r",stdin);
//	freopen("311E.out","w",stdout);
	
	int n,m,g,i,h,j,wantsex;
	
	scanf("%d%d%d",&n,&m,&g);
	S=n+1; T=n+2; edges=2;
	for (i=1;i<=n;i++)
		scanf("%d",gend+i);
	for (i=1;i<=n;i++)
	{
		scanf("%d",&h);
		if (gend[i]==0)
			edgeins(S,i,h);
		else
			edgeins(i,T,h);
	}
	
	int ans=0,pcs;
	for (i=n+3;i<=n+m+2;i++)
	{
		scanf("%d%d",&wantsex,&j);
		ans+=j;
		scanf("%d",&pcs);
		for (h=0;h<pcs;h++)
			scanf("%d",pc+h);
		scanf("%d",&h);
		if (h==1) j+=g;
		if (wantsex==0)
		{
			edgeins(S,i,j);
			for (h=0;h<pcs;h++)
				edgeins(i,pc[h],j);
		}
		else
		{
			edgeins(i,T,j);
			for (h=0;h<pcs;h++)
				edgeins(pc[h],i,j);
		}
	}
	
	while (bfs())
		ans-=dinic(S,~0U>>1);
	
	printf("%d",ans);
	
	return 0;
}
