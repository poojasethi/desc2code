#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 20010
#define N1 10000000
#define inf 1000000000
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,S,T,t,i1,i2,num,an,G;
int a1[N],a2[N],exist[N],w[N],K[N],a3[N][20],a[N1],b[N1],next[N1],head[N],g[N],gg[N],F[N];
inline void add(int x,int y,int z) {
    a[++t]=y,b[t]=z,next[t]=head[x],head[x]=t;
    a[++t]=x,b[t]=0,next[t]=head[y],head[y]=t;
}
inline bool bfs() {
    int l,v;
    g[g[0]=l=1]=S; gg[S]=0; F[S]=++num;
    for (;l<=g[0];l++)
    for (v=head[g[l]];v;v=next[v])
    if (b[v]&&F[a[v]]!=num) {
        F[a[v]]=num,g[++g[0]]=a[v],gg[a[v]]=gg[g[l]]+1;
        if (a[v]==T) return 1;
    } return 0;
}
int dfs(int x,int y) {
    int v,re,w=0;
    if (x==T) return y;
    for (v=head[x];v&&w<y;v=next[v])
    if (F[a[v]]==num&&b[v]&&gg[a[v]]==gg[x]+1&&(re=dfs(a[v],min(b[v],y-w))))
        b[v]-=re,b[v^1]+=re,w+=re;
    if (!w) gg[x]=0; return w;
}
int main() {
    scanf("%d%d%d",&n,&m,&G);
    For(i,1,n) scanf("%d",&a1[i]);
    For(i,1,n) scanf("%d",&a2[i]);
    S=n+m+1,T=S+1,t=1;
    For(i,1,n) if (a1[i]) add(i,T,a2[i]); else add(S,i,a2[i]);
    For(i,1,m) {
        scanf("%d%d%d",&exist[i],&w[i],&K[i]);
        For(j,1,K[i]) {
            scanf("%d",&a3[i][j]);
            if (exist[i]) add(a3[i][j],n+i,inf); else add(n+i,a3[i][j],inf);
        }
        scanf("%d",&k); an+=w[i]; w[i]+=k?G:0;
        if (exist[i]) add(n+i,T,w[i]); else add(S,n+i,w[i]);
    }
    For(i,1,m) if (!exist[i]) For(j,i+1,m) if (exist[j]) {
        int F=0;
        For(i1,1,K[i])For(i2,1,K[j]) if (a3[i][i1]==a3[j][i2]) {
            F=1; break;
        }
        if (F) add(n+i,n+j,inf);
    } //printf("%d\n",an);
    for (;bfs();an-=dfs(S,inf));
    printf("%d",an); return 0;
}
