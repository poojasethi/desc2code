#include <iostream>
#include <cstdio>
using namespace std;

const long long MOD = 1000000007;

long long N, M;

long long qpow(long long a, long long b)
{
    long long res = 1;
    for (; b; b >>= 1)
    {
        if (b&1) res = res * a % MOD;
        a = a * a % MOD;
    }
    return res;
}

int main()
{
    cin >> N >> M;
    long long C = 1;
    long long ans = 0;
    long long tmp = 0;
    for (long long len = 1; len <= N; ++len)
    {
        //(M - 1) ^ (N - len) * C(N, len)
        tmp = ( tmp + qpow(M - 1, N - (len - 1)) * C % MOD ) % MOD;
        ans = ( ans + ( qpow(M , N) - tmp + MOD ) % MOD * qpow(M, len) % MOD ) % MOD;
       // cout << ans << endl;
        if (len < N)
           C = C * (N - len + 1) % MOD * qpow(len, MOD - 2) % MOD;
    }
    ans = (ans + qpow(M, N)) % MOD;
    cout << ans << endl;
    return 0;
}
