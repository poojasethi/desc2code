#include<bits/stdc++.h>
using namespace std;

int n,m;
typedef long long ll;
const int MOD = (int)1e9+7;

ll mypow(int base, int n) {
    if(n == 0) return 1;
    ll t = mypow(base,n/2)%MOD;
    t = t*t%MOD;
    if(n&1) t = t*base%MOD;
    return t;
}

int main() {
    scanf("%d%d",&n,&m);
    ll res = mypow(2*m-1,n);
    for(int i = 0; i < n; ++i)
        res = (res+mypow(m,i)*mypow(2*m-1,n-1-i)%MOD)%MOD;
    cout<<res<<"\n";
}
