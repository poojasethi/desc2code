#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
typedef long long i64;
const int N = 1e6+10, MOD = 1e9+7;

int n, m;
i64 pwr[N];

int main() {

	scanf("%d%d", &n, &m);
	pwr[0] = 1;
	for (int i = 1; i <= n; ++i)
		pwr[i] = pwr[i-1] * m % MOD;
	i64 ans = pwr[n], prod = 1;
	for (int j = 1; j <= n; ++j) {
		i64 sum = pwr[n-j+1];
		sum = sum * prod % MOD;
		prod = prod * (2*m-1) % MOD;
		ans = (ans + sum) % MOD;
	}
	printf("%I64d\n", ans);

	return 0;
}
