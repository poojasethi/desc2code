#include<iostream>
using namespace std;
typedef long long ll;

const int mod=1e9+7;
const int N=1000010;

ll gcd(ll a,ll b,ll & x,ll & y)
{
	if(b==0) 
	{
		x=1;
		y=0;
		return a;
	}
	ll ret=gcd(b,a%b,x,y);
	ll t=y;
	y=x-(a/b)*y;
	x=t;
	return ret;
}

ll pow(ll a,int b)
{
	if(b==0) return 1;
	ll ret=pow(a,b/2);
	ret=ret*ret%mod;
	if(b%2==1) ret=ret*a%mod;
	return ret;
}

ll inv(ll a)
{
	ll x,y;
	gcd(a,mod,x,y);
	return (x%mod+mod)%mod;
}
