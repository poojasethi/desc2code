#include<cstdio>
using namespace std;
long long P=1e9+7,f[1000010],g[1000010],ans,inv[1000010],fac,n,m,i;
long long sqr(long long x){return x*x%P;}
long long pow(long long a,long long n)
{
	if (n==0) return 1;
	if (n%2==0) return sqr(pow(a,n/2));
	return sqr(pow(a,n/2))*a%P;
}
int main()
{
	scanf("%I64d%I64d",&n,&m);
	f[0]=g[0]=inv[0]=1;fac=1;
	for (i=1;i<=n;i++)
	{
		f[i]=f[i-1]*m%P;
		g[i]=g[i-1]*(m-1)%P;
		fac=fac*i%P;
		inv[i]=inv[i-1]*pow(i,P-2)%P;
	}
	ans=f[n];
	for (i=1;i<=n;i++) ans=(ans+f[i]*g[n-i]%P*fac%P*inv[i-1]%P*inv[n-i+1]%P)%P;
	printf("%I64d\n",ans);
	return 0;
 }
