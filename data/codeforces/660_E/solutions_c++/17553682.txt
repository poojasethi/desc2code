#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define N 1000005
#define mo 1000000007
#define LL long long
#define oo (1<<30)
using namespace std;
int n,m; LL ans,fc[N],xf[N];
LL fpm(LL x,LL y)
{
	LL s=1;
	while(y){
		if(y&1) s=(s*x)%mo;
		y>>=1 , x=(x*x)%mo;
	  }
	return s;
}
LL C(int n,int m){ return fc[n]*xf[n-m]%mo*xf[m]%mo;}
int main()
{
	int i;
	scanf("%d %d",&n,&m);
	fc[0]=xf[0]=1;
	for(i=1;i<=n;i++){
		fc[i]=(fc[i-1]*i) %mo;
		xf[i]=fpm(fc[i],mo-2);
	  }
	ans=fpm(m,n);
	for(i=1;i<=n;i++)
		(ans+=fpm(m,i)*fpm(m-1,n-i)%mo*C(n,i-1))%=mo;
	cout<<ans;
	return 0;
}
