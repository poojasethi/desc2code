#include <cstdio>
#define ll long long
#define maxn 1000010
#define zhl  1000000007
ll n,m,ans,tmp,i;
ll Pow(ll x,ll y)
{
	if (!x) return 1;
	for (tmp=1;y;y>>=1,x=(x*x)%zhl) if (y&1) tmp = tmp*x % zhl;
	return tmp;
}

int main()
{
	scanf("%I64d%I64d",&n,&m);
	if (m == 1){printf("%I64d",n+1);return 0;}
	ans = Pow(m,n);
	ll R = Pow(m-1,n),con = Pow(m-1,zhl-2);
	for (i = 1;i <= n;i++) ans = (ans+(R=R*con%zhl*m%zhl))%zhl,R = R*((n-i+1)*Pow(i,zhl-2)%zhl)%zhl;
	printf("%I64d",ans);
	return 0;
}
