#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define N 1000005
#define mo 1000000007
#define LL long long
#define oo (1<<30)
using namespace std;
int n,m,up; LL ans,fc[N],xf[N],x[N],S1,S2;
LL fpm(LL x,LL y)
{
	LL s=1;
	while(y){
		if(y&1) s=(s*x)%mo;
		y>>=1 , x=(x*x)%mo;
	  }
	return s;
}
LL C(int n,int m){ return fc[n]*xf[n-m]%mo*xf[m]%mo;}
int main()
{
	int i;
	scanf("%d %d",&n,&m);
	up=max(n,m),fc[0]=xf[0]=1;
	for(i=1;i<=up;i++)
		fc[i]=(fc[i-1]*i)%mo;
	xf[up]=fpm(fc[up],mo-2);
	for(i=up-1;i>=1;i--){
		xf[i]=(xf[i+1]*(i+1))%mo;
		x[i] =(fc[i-1]*xf[i])%mo;
	  }
	x[up]=xf[up]*fc[up-1]%mo;
	ans=fpm(m,n),S1=1,S2=fpm(m-1,n);
	for(i=1;i<=n;i++){
		S1=(S1*m)%mo,S2=(S2*x[m-1])%mo;
		if(m==1&&i==n) S2=1;
		(ans+=S1*S2%mo*C(n,i-1))%=mo;
	  }
	cout<<ans;
	return 0;
}
