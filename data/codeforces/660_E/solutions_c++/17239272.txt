#include <bits/stdc++.h>
using namespace std;

#define mp(x, y) make_pair((x), (y))

typedef long long ll;

ll mod=1000000007;

ll n, m;
ll pm[1000005], pmm[1000005];
ll ans;

int main()
{
	scanf("%I64d%I64d", &n, &m);
	pm[0]=1;
	for(ll k=1; k<=n; k++) pm[k]=(pm[k-1]*m)%mod;
	pmm[0]=1;
	for(ll k=1; k<=n; k++) pmm[k]=(pmm[k-1]*(2*m-1))%mod;
	ans=0;
	for(ll k=1; k<=n; k++) ans=(ans+pm[n-k+1]*pmm[k-1])%mod;
	ans=(ans+pm[n])%mod;
	printf("%I64d\n", ans);

	return 0;
}
