#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> pii;
typedef long long INT;

#define MOD 1000000007

#define NN 1100000

int fac[NN];

int pow(int a, int b) {
	if (b==0) return 1;
	INT ans=pow(a, b/2);
	ans=ans*ans%MOD;
	if (b&1) ans=ans*a%MOD;
	return ans;
}

int main() {
	INT ans=0;
	int N, M;
	
	fac[0]=1;
	for (int i=1; i<NN; i++)
		fac[i]=(INT)fac[i-1]*i%MOD;
	
	scanf("%d%d", &N, &M);
	for (int i=1; i<=N; i++) {
		INT temp=(INT)fac[N]*pow(fac[N-i+1], MOD-2)%MOD;
		temp=temp*pow(fac[i-1], MOD-2)%MOD;
		temp=temp*pow(M, i)%MOD*pow(M-1, N-i)%MOD;
		ans+=temp;
	}
	
	ans=(ans+pow(M, N))%MOD;
	
	printf("%I64d\n", ans);
	
	return 0;
}
