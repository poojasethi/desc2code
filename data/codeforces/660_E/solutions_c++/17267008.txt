#include <iostream>
#include <vector>
#include <cstdio>
#define mod 1000000007
using namespace std;
int n,m;

int main()
{
    //freopen("in.txt","r",stdin);
    scanf("%d%d",&n,&m);
    long long f2=2*m,f1=m;
    for(int i=1;i<n;i++)
    {
        f2=((2*m-1)*f2%mod+f1)%mod;
        f1=f1*m%mod;
    }
    printf("%lld\n",f2);
    return 0;
}
