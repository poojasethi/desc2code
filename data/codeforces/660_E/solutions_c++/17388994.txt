#include <bits/stdc++.h>
#define rep(i,n) for(int i = 0; i < n; i++)
#define INF 100000000
#define EPS 1e-10
#define MOD 1000000007
using namespace std;
typedef pair<int,int> P;

long long n, m;
long long c[1000001];

long long modpow(long long x, long long y){
	long long ret = 1;
	while(y){
		if(y&1) ret = ret*x%MOD;
		x = x*x%MOD;
		y >>= 1;
	}
	return ret;
}

int main(){
	cin >> n >> m;
	c[0] = 1;
	for(int i = 1; i <= n; i++){
		c[i] = c[i-1]*(n-i+1)%MOD;
		c[i] = c[i]*modpow(i,MOD-2)%MOD;
	}
	long long ans = 0;
	for(int i = 1; i <= n; i++){
		long long tmp;
		tmp = modpow(m,i)*modpow(m-1,n-i)%MOD;
		tmp = tmp*c[i-1]%MOD;
		ans += tmp;
		ans %= MOD;
	}
	ans += modpow(m,n);
	ans %= MOD;
	cout << ans << endl;
}
