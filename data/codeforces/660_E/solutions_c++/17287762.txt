#define ll long long
#define mod 1000000007

ll modPow(ll base, ll exponent, int modulus=mod){
    ll result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}
