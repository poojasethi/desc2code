#define INF 0x3f3f3f3f
#define ull unsigned long long
#define ll long long
#define IN __int64
#define N 2010
#define M 1000000007
using namespace std;
int main()
{
	int n,m;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		ll ans=2*m;
		ll tmp=1;
		for(int i=2;i<=n;i++)
		{
			tmp=tmp*m%M;
			ans=(ans*(2*m-1)%M+tmp+M)%M;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
