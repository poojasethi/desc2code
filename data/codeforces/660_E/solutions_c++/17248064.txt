#include<algorithm>
#include<iostream>
#include<cstdlib>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;

typedef long long LL;

#define MOD 1000000007

int n,m;

LL ans,lastans;
LL k;

int main() 
{
	scanf("%d%d",&n,&m);
	ans=1;  
	for (int i=1;i<=n;i++)
	{
		lastans=ans;
		ans=((lastans*2-k)%MOD+MOD)*m%MOD;
		k=(lastans+k*(m-1))%MOD;
	}
	printf("%lld\n",ans);
	return 0;
}
