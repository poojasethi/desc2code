#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

long long power_up(long long a, long long n){
    if(n==1) return a;
    long long ret = power_up(a,n/2);
    ret = (ret*ret)%mod;
    if(n%2==1) ret = (ret*a)%mod;
    return ret;
}

int main(){
    long long m, n;
    cin >> n >> m;
    if(m==1){
        cout << n+1 << endl;
        return 0;
    }
    long long ans = power_up(m,n);
    long long temp = (m * power_up(m-1,mod-2))%mod;
    long long vemp = power_up(m+m-1,n);
    vemp = (vemp - ans)%mod;
    if(vemp < 0) vemp += mod;
    temp = (temp*vemp)%mod;
    ans = (ans+temp)%mod;
    cout << ans << endl;
    return 0;
}
