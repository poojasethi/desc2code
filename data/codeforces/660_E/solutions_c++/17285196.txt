#include <iostream>
using namespace std;
const int mod = 1e9 + 7;
int quick_pow(int a, int b)
{
    int ans = 1;
    for(;b;b >>= 1, a = a * 1ll * a % mod){
        if(b & 1) ans = ans * 1ll * a % mod;
    }
    return ans;
}
int main()
{
	int n, m;cin>>n>>m;
	int ans = quick_pow(m, n);
	for(int s = 1; s <= n; s ++)
		ans = (ans + quick_pow(m, n - s+ 1) * 1ll * quick_pow( 2 * m - 1, s - 1)) % mod;
	cout<<ans<<endl;
	return 0;
}
