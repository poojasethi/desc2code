#include <bits/stdc++.h>
using namespace std;

#define mp(x, y) make_pair((x), (y))

typedef long long ll;

ll mod=1000000007;

ll n, m;
ll ans;

ll pow(ll b, ll e)
{
	ll ret=1;
	while(e) {
		if(e&1) ret=(ret*b)%mod;
		e>>=1;
		b=(b*b)%mod;
	}
	return ret;
}

int main()
{
	scanf("%I64d%I64d", &n, &m);
	if(m==1) {
		ans=n+1;
	} else {
		ans=((m*pow(2*m-1, n)-pow(m, n+1)+mod)%mod*pow(m-1, mod-2)+pow(m, n))%mod;
	}
	printf("%I64d\n", ans);

	return 0;
}
