#include <bits/stdc++.h>

using namespace std;

const int MOD = 1e9+7;

inline int fpm(int a, int b) {
    int ret = 1;
    while(b) {
        if(b&1) ret = 1ll*ret*a%MOD;
        a = 1ll*a*a%MOD, b >>= 1;
    }
    return ret;
}

int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    int ans = fpm(m, n);
    for(int j = 1; j <= n; ++j)
        (ans += 1ll*fpm(m, n-j+1)*fpm(2*m-1, j-1)%MOD) %= MOD;
    cout << ans << endl;
    return 0;
}
