#include <bits/stdc++.h>

using namespace std;

const long long mod = (long long)1e9 + 7;

int n, m, i;
long long p, coeff;
long long f[2000020];

int main() {
  ios_base::sync_with_stdio(0);
  cin >> n >> m;
  coeff = 2 * m - 1;
  f[0] = p = 1ll;
  for (int i = 1; i <= n; i++) {
    f[i] = f[i - 1] * coeff;
    f[i] %= mod;
    f[i] += p;
    f[i] %= mod;
    p = (m * p) % mod;
  }
  cout << f[n];

  return 0;
}
