using namespace std;

const int MAXN = 1e6+7;
const int MOD = 1e9+7;
int n, m;
LL sumv[MAXN];
LL dp[MAXN];
// LL pow_mod(int a, int b) {
// 	if (!b) return 1;
// 	LL ans = pow_mod(a, b/2);
// 	ans = ans*ans % MOD;

int main()
{
    //freopen("input.txt", "r", stdin);
	scanf("%d%d", &n, &m); 
	dp[0] = 1;
	for (int i = 1; i <= n; i++) {
		dp[i] = m * (dp[i-1]*2 - sumv[i-1]) % MOD;
		sumv[i] = (sumv[i-1]*(m-1) + dp[i-1]) % MOD;
		//cout << dp[i] << endl;
	}
	dp[n] = (dp[n]+MOD) % MOD;
	cout << dp[n] << endl;
    return 0;
}
