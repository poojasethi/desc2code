#include <iostream>
#include <cstdio>
using namespace std;
long long mod=1e9+7;
long long n,m;

long long quick_pow(long long a,long long b){
    long long ret=1;
    while(b){
        if(b&1) ret=(ret*a)%mod;
        a=(a*a)%mod;
        b>>=1;
    }
    return ret;
}

int main(){
    scanf("%I64d %I64d",&n,&m);
    long long ans=0;
    for(int i=1;i<=n;i++){
        ans=(ans+(quick_pow(m,n-i+1)*quick_pow(2*m-1,i-1))%mod)%mod;
    }
    ans=(ans+quick_pow(m,n))%mod;
    printf("%I64d\n",ans);
    return 0;
}
