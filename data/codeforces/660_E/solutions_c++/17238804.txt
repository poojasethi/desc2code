#include <iostream>
using namespace std;

int N, M;
const int mod = 1e9 + 7;
int f[1000001];

int main() {
	cin >> N >> M;
	f[N] = 1;
	int k = 0;
	for (int i = N - 1; i >= 0; --i) {
		f[i] = 1LL * M * ((2 * f[i + 1] - k) % mod + mod) % mod;
		k = (1LL * (M - 1) * k % mod + f[i + 1]);
	}
	cout << f[0] << endl;
}
