#include <bits/stdc++.h>
using namespace std;
const int N = 1000010, mod = (int)1e9 + 7;

int n, f[N], m;
int sum;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(0);
    cin >> n >> m;
    f[0] = sum = 1;
    for (int i = 1; i <= n; i++) {
        int x = (1LL * m * sum) % mod;
        f[i] = ((1LL * m * f[i-1]) % mod + x) % mod;
        sum = (1LL * (m - 1) * sum) % mod; sum = (sum + x) % mod;
    }
    cout << f[n] << endl;
}
