#include <iostream>

using namespace std;

const int Mo = int(1e9) + 7;

int Pow(int a,int b)
{
	int tmp = 1;
	for(;b;b >>= 1,a = a * 1ll * a % Mo)
		if (b & 1) tmp = tmp * 1ll * a % Mo;
	return  tmp;
}

int main()
{
	int n,m;
	cin >> n >> m;
	int ans = Pow(m,n);
	for(int j = 1;j <= n;j ++)
		ans = (ans + Pow(m,n - j + 1) * 1ll * Pow(m * 2 - 1,j - 1) % Mo) % Mo;
	cout << ans << endl;
}
