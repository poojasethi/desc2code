#include<sstream>
#include<iostream>

using namespace std;

void spaces(int n)
{
	for(int i=0;i<n;i++)
		cout<<" ";
}

int main()
{
	string s;
	cin>>s;
	for(int i=0;i<s.size();i++)
		if(s[i]=='<' or s[i]=='>')
			s[i]=' ';
	stringstream ss(s);
	int dep=0;
	while(ss>>s)
	{
		if(s[0]=='/')
		{
			dep-=2;
			spaces(dep);
			cout<<"<"<<s<<">\n";
		}
		else
		{
			spaces(dep);
			cout<<"<"<<s<<">\n";
			dep+=2;
		}
	}
}
