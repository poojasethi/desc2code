#include <iostream>
#include <string>
using namespace std;

int main(){
    string XML;
    cin>>XML;
    int level = 0;
    for (int i = 0; i<XML.length(); i++){
        cout<<XML[i];        
        if (XML[i] == '>') {
            if ( XML[i-2] != '/' )level += 2;
            if ( XML[i+2] == '/') level -= 2;
            cout<<endl;
            for (int j = 0; j < level; j++) cout<<" ";
        }
    }
}
