#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;
int i,n,h,j;
char S[1005];

int main()
{
	gets(S);
	n=strlen(S);
	i=h=0;
	while (i<n)
	{
		i++;
		if (S[i]=='/')
		{
			i++;
			h--;
			for (j=1;j<=2*h;++j)
				cout<<" ";
			cout<<"</"<<S[i]<<">";
			cout<<endl;
			i++;
		}
		else
		{	
			i++;
			for (j=1;j<=2*h;++j)
				cout<<" ";
			cout<<"<"<<S[i-1]<<">";
			cout<<endl;
			h++;
		}
		i++;
	}
	return 0;
}
