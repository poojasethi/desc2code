#include <iostream>
using namespace std;
int main() {
    char x,c;
    int i,y=0;
    while(cin>>x){
        if(x=='<'){
            cin>>c;
            y++;
            if(c=='/')y--;
            for (i=1;i<y;i++)
                cout<<"  ";
            if(c=='/')y--;
            cout<<x<<c;
            }
        else if(x=='>')cout<<x<<endl;
        else cout<<x;
    }
return 0;}
