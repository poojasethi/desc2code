#include <iostream>
using namespace std;

int main()
{
  int h=0;
  string s;
  cin>>s;
  for(int i=0;i<s.length();i++)
    {
      if(s[i]=='<' && s[i+1]!='/')
	{
	  for(int j=1;j<=2*h;j++)
	    cout<<" ";
	  cout<<s[i]<<s[i+1]<<s[i+2]<<endl;
	  h++;
	  i+=2;
	}
      else if(s[i]=='<' && s[i+1]=='/')
	{
	  h--;
	  for(int j=1;j<=2*h;j++)
	    cout<<" ";
	  cout<<s[i]<<s[i+1]<<s[i+2]<<s[i+3]<<endl;
	}
    }
  return 0;
}
