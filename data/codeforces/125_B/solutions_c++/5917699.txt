#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int h, e;
char s[1005], c;

int main(){
	scanf("%s", s);
	for (int i=0; s[i]; i++){
		if (s[i] == '>'){
			if (e) h++;
			for (int j=0; j<2*h-2; j++) printf(" ");
			if (e) printf("<%c>\n", c);
			else printf("</%c>\n", c);
			if (!e) h--;
		}
		else{
			if (s[i] == '<') e = 1;
			if (s[i] >= 'a' && s[i] <= 'z') c = s[i];
			if (s[i] == '/') e = 0;
		}
	}
	return 0;
}
