#include <iostream>
#include <string.h>
#include <stdio.h>
using    namespace std;

int N;

int main()
{
  char s[1005];
  cin >> s;
  N=strlen(s);
  int x=-1;

  for(int i=0 ; i<N ; i++)
  { 
    if(s[i]=='<' || s[i]=='>') continue;
    if(s[i]=='/') { for(int t=0 ; t<x ; t++) printf("  "); printf("</%c>\n",s[i+1]); x--; i++; continue;} 
    x++;
    for(int t=0 ; t<x ; t++) printf("  ");
    printf("<%c>\n",s[i]);
  }
  return 0;
}
