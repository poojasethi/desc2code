#include<stdio.h>

char buf[1050]={0};

void printtag(char c,int depth,int isstart)
{
	for(int i=0;i<depth;++i) printf("  ");
	putchar('<');if(!isstart)putchar('/');putchar(c);putchar('>');putchar('\n');
}

int main()
{
	int depth;
	scanf("%s",buf);
	char* p=buf+1;
	depth=0;
	while(*p)
	{
		if(*p=='/')
		{
			printtag(*++p,--depth,0);
		}
		else
		{
			printtag(*p,depth++,1);
		}
		p+=3;
	}


	return 0;
}
