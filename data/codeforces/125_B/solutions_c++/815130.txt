#include<cstdio>
#include<cstring>

char s[2000];
int l,p,h;

int main()
{
    scanf("%s",s);
    l = strlen(s);
    p = h = 0;
    for(int i = 0;i < l;i++) if (s[i] == '>')
    {
        if (s[i-2] == '/') h--;
        for(int i = 0;i < h;i++) printf("  ");
        for(int j = p;j <= i;j++) printf("%c",s[j]);
        printf("\n");
        if (s[i-2] != '/') h++;
        p = i+1;
    }
    return 0;
}
