#include <bits/stdc++.h>

using namespace std;

stack < char > s;
int h;
string n;

void print ( bool flg ){
	for ( int i = 0 ; i < h ; i++ )
		printf ( "  " );

	printf ( ( flg ? "</%c>\n" : "<%c>\n" ), s.top () );
}

int main (){
	cin >> n;
	for ( int i = 0 ; i < n.size () ; i++ )
		if ( n [ i ] == '/' ){
			h--;
			print ( true );
			s.pop ();

			i++; continue;
		} else if ( n [ i ] != '>' && n [ i ] != '<' ){
			s.push ( n [ i ] );
			print ( false );
			h++;
		}
}
