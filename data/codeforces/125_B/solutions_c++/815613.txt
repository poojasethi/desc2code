#include<iostream>
#include<string>

using namespace std;

int main()
{
	string s;
	cin >> s;

	int count=0;
	int pos = 0;
	while(s.size() != pos){
		int i = 0;

		if( s[pos+1] != '/' ){
			while( i < 2*count){
				cout << " ";
				i++;
			}
			cout << s.substr(pos,3);
			pos = pos + 3;
			count ++;
		} else {
			count --;
			while( i < 2*count){
				cout << " ";
				i++;
			}
			cout << s.substr(pos,4);
			pos = pos + 4;
		}
		cout << endl;
	}
	return 0;
}
