#include <stdio.h>

char w[10005];
int x=0;

void go(int lev) {
	for(int i=0;i<lev;i++) {
		printf("  ");
	}
	char me = w[x+1];
	printf("<%c>\n", me);
	x += 3;
	while(w[x+1]!='/')
		go(lev+1);
	x += 4;
	for(int i=0;i<lev;i++) printf("  ");
	printf("</%c>\n", me);
}

int main(void) {
	scanf("%s", w);
	while(w[x]) go(0);
	return 0;
}
