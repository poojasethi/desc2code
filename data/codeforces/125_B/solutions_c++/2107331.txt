#include <iostream>
#include<string>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int k=0,i,j;
    for(i=0;i<s.size();i++)
    {

        if(s[i]=='<'&&s[i+1]!='/')
{for(j=0;j<2*k;j++) cout<<' ';
    cout<<s[i]<<s[i+1]<<s[i+2];
i+=2;
k++;}
else
{k--;
    for(j=0;j<2*k;j++) cout<<' ';
    cout<<s[i]<<s[i+1]<<s[i+2]<<s[i+3];i+=3;
}
    cout<<endl;}

    return 0;
}
