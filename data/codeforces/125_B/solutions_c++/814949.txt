#include<cstdio>
#include<cstring>
using namespace std;

char str[1005];
int main(){
    gets(str);
    int ss=0;
    for(int i=0;i<strlen(str);i++){
        if(str[i]=='<'){
            if(str[i+1]=='/') ss-=1;
            for(int j=0;j<ss;j++) printf("  ");
            if(str[i+1]=='/') ss-=1;
            ss++;
            for(;;i++){
                putchar(str[i]);
                if(str[i]=='>') break;
            }
        }puts("");
    }
}
