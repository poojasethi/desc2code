#include <iostream>
#include <cstring>
using namespace std;
const int MAXN=1000+10;
char a[MAXN];
int tag;
int main()
{
	cin >> a;
	int counter=strlen(a);
	tag=1;
	for(int i=0;i<counter;i++)
	{
		if(a[i+1]!='/')
		{
			for(int j=1;j<tag;j++)
				cout << "  ";
			cout << a[i] << a[i+1] << a[i+2] << endl;
			tag++;
			i+=2;
		}
		else
		{
			tag--;
			for(int j=1;j<tag;j++)
				cout << "  ";
			cout << a[i] << a[i+1] << a[i+2] << a[i+3] << endl;
			i+=3;
		}
	}
	return 0;
}
