#include <string>
#include <iostream>

using namespace std;

int main () {
	int h = 0;
	
	while (!cin.eof()) {
		string t ("");
		while (true) {
			char c;
			cin >> c;
			if (!cin.eof()) t.append(1, c);
			if (c=='>') break;
		}
		if (t[1]=='/') h--;
		for (int i = 0 ; i < h ; i++) cout << "  ";
		cout << t << endl;
		if (t[1]!='/') h++;
	}
}
