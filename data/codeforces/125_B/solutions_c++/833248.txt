#include<cstdio>
char s[1<<10];
int main()
{
    int i,j,bf=0;
    gets(s);
    for(i=0;s[i];)
    {
        if(s[i+1]=='/') bf-=2;
        for(j=0;j<bf;j++) printf(" ");
        for(j=i;;j++)
        {
            printf("%c",s[j]);
            if(s[j]=='>') break;
        }
        if(s[i+1]!='/') bf+=2;
        puts("");
        i=j+1;
    }
    return 0;
}
