#include <cstdio>
#include <cstring> 

char s[1024];

int print(int i, int l) {
	for(int d = 0; d < 2 * l; ++d)
		printf(" ");
	while(1) {
		printf("%c", s[i]);
		if(s[i] == '>') break;
		++i;
	}
	printf("\n");
	return ++i;
}

int main() {
	//freopen("b.in", "r", stdin);
	gets(s);
	int i = 0, l = -1, n = strlen(s);
	
	while(i < n - 1) 
		if(s[i + 1] == '/')
			i = print(i, l--);
		else
			i = print(i, ++l);
}
