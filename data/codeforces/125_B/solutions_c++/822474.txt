#include <stdio.h>
#include <string.h>

int main()
{
	char s[1001]={0};
	int N;

	scanf("%s",&s);
	N=strlen(s);
	int k=0;
	for (int i=0;i<N-1;)
	{
		if (s[i]=='<'&&s[i+1]!='/')
		{
			for (int j=0;j<k;j++) printf(" ");
			printf("<%c>\n",s[i+1]);
			i+=3;
			k+=2;
		}
		else
		{
			k-=2;
			for (int j=0;j<k;j++) printf(" ");
			printf("</%c>\n",s[i+2]);
			i+=4;
		}
	}

	return 0;
}
