{
  string s;
  cin>>s;
  int sz=s.size();
  int x=-1;

  for(int i=0;i<sz;i++)
  { 
if(s[i]=='<' || s[i]=='>') continue;
if(s[i]=='/') 
{ 
for(int j=0;j<x;j++) cout<<"  "; 
printf("</%c>\n",s[i+1]); 
x--; 
i++; 
continue;
} 
    x++;
    for(int j=0;j<x;j++) cout<<"  ";
    printf("<%c>\n",s[i]);
  }
  return 0;
}
