#include<iostream>
#include<string>
using namespace std;
int main()
{
	for(string s;cin>>s;)
	{
		int l=s.length();int i;int n=1;
		for(i=0;i<3;i++)
			cout<<s[i];
		cout<<endl;
		while(i<l)
		{
			if(s[i+1]=='/')
			{
				cout<<string(2*(n-1),' ');
				for(int j=0;j<4;j++)
					cout<<s[i+j];
				n--;cout<<endl;i+=4;
			}
			else
			{
				cout<<string(2*n,' ');
				for(int j=0;j<3;j++)
					cout<<s[i+j];
				n++;cout<<endl;i+=3;
			}
		}
	}
}
