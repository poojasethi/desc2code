#include<stdio.h>
#include<string.h>
int main()
{
	char s[1000];
	scanf("%s",s);
	int n=strlen(s);
	int i,j;
	int f=0;
	for(i=0;i<n-1;)
	{
		if(s[i]=='<'&&s[i+1]!='/')
		{
			for(j=0;j<f;j++)
			printf(" ");
			printf("<%c>\n",s[i+1]);
			i=i+3;
			f=f+2;
		}
		else
		{
			f=f-2;
			for(j=0;j<f;j++)
			printf(" ");
			printf("</%c>\n",s[i+2]);
			i=i+4;
		}
	}
}
