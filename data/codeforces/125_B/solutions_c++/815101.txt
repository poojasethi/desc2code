#include <cstdio>
#include <cstring>

char line[1005];
int i, len, h;

int main() {
	scanf("%s", line);
	i = 1, len = strlen(line), h = 0;
	while (i < len) {
		if (line[i] != '/') {
			for (int j = 0; j < 2 * h; j++) printf(" ");
			printf("<%c>\n", line[i]);
			h++, i += 3;
		} else {
			h--;
			for (int j = 0; j < 2 * h; j++) printf(" ");
			printf("</%c>\n", line[i+1]);
			i += 4;
		}
	}
}
