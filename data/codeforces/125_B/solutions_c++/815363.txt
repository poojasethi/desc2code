#include<iostream>
#include<string>

using namespace std;

int main()
{
	string str;
	int h=0;
	cin>>str;
	for(int i=0;i<str.size()-1;i++){
		if(str[i]=='<' && str[i+1]!='/'){
			for(int j=1;j<=2*h;j++){
				cout<<" ";
			}
			cout<<"<"<<str[i+1]<<">"<<endl;
			h++;
		}
		if(str[i]=='<' && str[i+1]=='/'){
			h--;
			for(int j=1;j<=2*h;j++){
				cout<<" ";
			}
			cout<<"</"<<str[i+2]<<">"<<endl;
		}
	}
	return 0;
}
