#include <iostream>

using namespace std;

int main() {
	char * chars = new char[1001];
	cin >> chars;
	int indent = 0, j;
	for (int i = 1; chars[i - 1] != '\0'; i++) {
		if (chars[i] == '/') {
			indent--;
			for (j = 0; j < indent; j++)
				cout << "  ";
			cout << "</";
			i++;
		} else {
			for (j = 0; j < indent; j++)
				cout << "  ";
			cout << "<";
			indent++;
		}
		cout << chars[i];
		i += 2;
		cout << ">\n";
	}
	return 0;
}
