#include <iostream>
#include <string>
using namespace std;

int main()
{
	string t;
	cin>>t;
	int p  = 3;
	int ci = 2;
	cout<<t.substr(0, 3)<<endl;
	while(p!=t.size()){
		if(t[p+1] == '/'){
			ci-=2;
			for(int i = 0; i < ci; i++) cout<<" ";
			cout<<t.substr(p, 4)<<endl;
			p+=4;
			continue;
		}
		for(int i = 0; i < ci; i++) cout<<" ";
		cout<<t.substr(p, 3)<<endl;
		ci+=2;
		p+=3;
	}
	return 0;
}
