#include <stdio.h>
#include <string>
#include <iostream>
using namespace std;

int main()
{
    string str;
    cin>>str;
    int now=0;
    for(int i=0;i<str.size();i++)
    {
        int a=0;
        if(str[i+1]=='/')a=1;
        if(!a)now++;
        for(int j=2;j<=now;j++)
            cout<<"  ";
        cout<<'<';
        i++;

        for(;str[i]!='>';i++)
            cout<<str[i];
        cout<<'>'<<endl;
        if(a)now--;
    }
}
