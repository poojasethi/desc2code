#include <iostream>
#include <string>


using namespace std;

int main(){
    string str;
    cin >> str;
    for (int i = 0 ,p = 0 ;i <str.size();i++  )
    {
        if (str[i+1] == '/')
            p -=2;
        cout << string(p ,' ');
        if (str[i+1] != '/')
            p +=2;
        for (;;i++)
        {
            cout <<str[i];
            if (str[i] == '>')
                break;

        }
        cout <<"\n";
    }
    return 0;
}
