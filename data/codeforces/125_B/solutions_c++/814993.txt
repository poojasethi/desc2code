#include <iostream>
#include <string>
using namespace std;

int main(){
	string s;
	cin>>s;
	string tag;
	int o=s.find('<',0);
	int h=0;
	while (o!=string::npos){
		int c=s.find('>',o+1);
		tag=s.substr(o,c-o+1);
		if (tag.size()==3){
			for (int i=0;i<h;i++)
				cout<<" ";
			cout<<tag<<endl;
			h+=2;
		}
		else{
			h-=2;
			for (int i=0;i<h;i++)
				cout<<" ";
			cout<<tag<<endl;
		}
		o=s.find('<',o+1);
	}
	return 0;
}
