#include<iostream>

using namespace std;

int main()
{
	string s;
	cin >> s;
	int count=-1;
	int i=1;
	while(i<s.length())
	{
		if(s[i]!='/')
		{
			count++;
			for(int j=0;j<2*count;j++)
				cout<<" ";
			cout<<s[i-1]<<s[i]<<s[i+1]<<endl;
			i+=3;	
		}
		else
		{
			for(int j=0;j<2*count;j++)
				cout<<" ";
			count--;
			cout<<s[i-1]<<s[i]<<s[i+1]<<s[i+2]<<endl;
			i+=4;
		}
		
	}
	return 0;
}
