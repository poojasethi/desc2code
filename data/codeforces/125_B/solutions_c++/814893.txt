#include <iostream>
#include <cstdio>
using namespace std;
char a;
int n = 0;
int main()
{
	while (cin >> a)
	{
		if (a == '>')
		{
			puts(">");
		} else
		if (a == '<')
		{
			cin >> a;
			if (a == '/') --n;
			
			for (int i = 0; i < n; ++i)
				printf("  ");

			if (a != '/') ++n;
			printf("<");
			printf("%c", a);
		} else
		if (a >= 'a' && a <= 'z')
			printf("%c", a);
	}
}
