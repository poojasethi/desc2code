import re
import sys

s0 = sys.stdin.read()

tags = []
p = 0
patt = re.compile(r'<\w+>|</\w+>', re.M)
patt_open = re.compile(r'<\w+>')
patt_close = re.compile(r'</\w+>')
while p < len(s0):
	g = patt.search(s0[p:])
	if not g: break
	tag = g.group()
	p += g.span()[1]
	tags.append(tag)

depth = 0
for tag in tags:
	if patt_open.match(tag):
		# is open tag
		print '  '*depth + tag
		depth += 1
	else:
		depth -= 1
		print '  '*depth + tag
