import re

s = re.sub('[<>]', '', raw_input())
stack = []
indent = 0
res = ''
while s:
    if s[0] != '/':
        line = ' ' * indent + '<%s>' % s[0]
        res = res + ("\n" if res else '') + line
        stack.append(indent)
        indent += 2
        s = s[1:]
    else:
        position = stack.pop()
        line = ' ' * position + '</%s>' % s[1]
        res = res + ("\n" if res else '') + line
        indent = position
        s = s[2:]
print res
