import re

def solve(xml):
    stack = []
    for s in re.findall("(</{0,1}.>)", xml):
        ending = s.find('/') == 1
        l = s[len(s)-2]
        if len(stack) == 0 or (not ending):
            stack.append(l)   
            print '  '*(2+len(stack)-3)+'<'+l+'>'
        else:
            print '  '*(2+len(stack)-3)+'</'+l+'>'
            stack = stack[:len(stack)-1]
            
            

xml = raw_input()
solve(xml)
