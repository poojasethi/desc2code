import sys
import re

file = sys.stdin
line = file.readline().rstrip()
p = re.compile("<[^>]+>")
match = p.findall(line)
indent = 0
print match[0]
for t in match[1:]:
    if t[1] != '/':
        indent += 2
    line = [" " for _ in range(indent)]
    line += t
    print "".join(line)
    if t[1] == '/':
        indent -= 2
