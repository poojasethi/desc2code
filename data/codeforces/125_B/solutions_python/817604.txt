import sys

#sys.stdin = open('in.txt', 'r')
s = sys.stdin.readline().split('><')

#print s
spaces = 0
for it in s:
	ss = it
	if ss.count('<'):
		ss = str(it[1:])
	if ss.count('>'):
		ss = str(it[:-2])
	if ss.count('/') > 0:
		spaces -= 2
		print ' ' * spaces + '<' + ss + '>'
	else:
		print ' ' * spaces + '<' + ss + '>'
		spaces += 2
