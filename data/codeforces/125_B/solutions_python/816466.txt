import re

def pr(x, l):
    print ' ' * 2 * l + x

st = []
for x in re.findall('<.*?>', raw_input()):
    if len(st) == 0:
        st = st + [x]
        print(x)
    else:
        if len(x) == 4 and x[-2] == st[-1][-2]:
            st = st[:-1]
            pr(x, len(st))
        else:
            pr(x, len(st))
            st = st + [x]
