from sys import stdin
import re

xml = stdin.read()

p = re.compile('<[\\a-z]{1,2}>')

start = 0
indent = 0

while start < len(xml):
    r = p.search(xml, start)
    if not r:
        break

    tag = xml[r.start():r.end()]

    if len(tag) == 3:
        print ' '*indent + tag
        indent += 2
    else:
        indent -= 2
        print ' '*indent + tag

    start = r.end()
