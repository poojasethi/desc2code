def tokenize(text):
  return text.replace("<", " <").split()

def is_open(tag):
  return len(tag) == 3

def print_tag(tag, indent):
  text = (' ' * indent) + tag
  print text

def solve(text):
  text = tokenize(text)
  indent = 0
  
  for tag in text:
    if is_open(tag):
      print_tag(tag, indent)
      indent += 2
    else:
      indent -= 2
      print_tag(tag, indent)

      
solve(raw_input())
