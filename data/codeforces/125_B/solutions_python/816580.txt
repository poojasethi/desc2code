import sys

if __name__ == '__main__':
	
	#cin = open("1.in", "r")
	#cout = sys.stdout
	cin = sys.stdin
	
	arr = cin.read().split(">")
	h = 0
	for idx, tag in enumerate(arr):
		if len(tag) == 0 or tag[0] != "<":
			continue
		if tag[1] == "/":
			h-=1
		print " "*(2*h) + tag + ">"
		if tag[1] != "/":
			h+=1
