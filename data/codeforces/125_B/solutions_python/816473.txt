S = raw_input().strip().split('>')
tab = 0
for i, x in enumerate(S[:-1]):
    y = S[i - 1] if i > 0 else '///'
    if x[1] == '/':
        if y[1] == '/':
            tab -= 1
        for _ in xrange(tab):
            print ' ',
        print x + '>'
    else:
        if y[1] != '/':
            tab += 1
        for _ in xrange(tab):
            print ' ',
        print x + '>'
