#include <stdio.h>
#include <algorithm>
using namespace std;
int main()
{
	int n,i,j;
	int a[5050],b[5050];
	__int64 dp[5050],t;
	while (scanf("%d",&n)!=EOF){
		for (i=0; i<n; i++){
			scanf("%d",a+i);
			b[i]=a[i];
		}
		sort(b,b+n);
		for (i=0; i<n; i++){
			dp[i]=abs(b[i]-a[0]);
			if (i) dp[i]=min(dp[i],dp[i-1]);
		}
		for (i=1; i<n; i++)
			for (j=0; j<n; j++){
				dp[j]+=abs(b[j]-a[i]);
				if (j) dp[j]=min(dp[j],t);
				t=dp[j];
			}
		printf("%I64d\n",dp[n-1]);
	}
	return 0;
}
