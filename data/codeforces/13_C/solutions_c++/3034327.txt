#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
LL n,a[5050],b[5050],t[2][5050],*p=t[0],*q=t[1],I=1LL<<60;
int main(void) {
  cin>>n;
  for(int i=0;i<n&&cin>>a[i];++i)b[i]=a[i];
  sort(b,b+n);
  for(int i=0;i<n;++i){
    q[0]=I;
    for(int j=0;j<n;++j)q[j+1]=min(q[j],p[j+1]+abs(b[j]-a[i]));
    swap(p,q);
  }
  cout<<p[n]<<endl;
}
