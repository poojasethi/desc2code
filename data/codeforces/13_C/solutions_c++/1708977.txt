# include <cstdio>
# include <algorithm>
# include  <iostream>
using namespace std;
int s[5000];int b[5000];long long num[5000]={0};
int main()
{int n;scanf("%d",&n);for(int i=0;i<n;i++){
		scanf("%d",&s[i]);
		b[i]=s[i];
	}
	sort(b,b+n);
	int m=unique(b,b+n)-b;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++){
			num[j] +=  abs (s[i]-b[j]);
			if(j)
				num[j] = min(num[j],num[j-1]);
		}
	printf("%I64d",num[m-1]);
	return 0;
}
