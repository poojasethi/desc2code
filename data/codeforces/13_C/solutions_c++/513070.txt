#include<iostream>

long long b1[5005],b2[5005],minm;


main()
{cin>>n;
 for(i=0;i<n;i++){cin>>a[i];c[i]=a[i];}
 sort(c,c+n);
 for(i=0;i<n;i++)
 {minm=1000000000000000000;
  for(j=0;j<n;j++)
  {if(i-1>=0)minm=min(b1[j],minm);else minm=0;
   b2[j]=minm+abs(a[i]-c[j]);} 
  for(j=0;j<n;j++)b1[j]=b2[j];
 } 
 
 minm=b2[0];
