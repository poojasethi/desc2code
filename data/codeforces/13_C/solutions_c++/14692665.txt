#include <bits/stdc++.h>
using namespace std;
const int maxn = 5010;
int n,a[maxn],v[maxn];
long long f[maxn],g[maxn];
int main(){
    scanf("%d",&n);
    for (int i=1; i<=n; i++) scanf("%d",&a[i]);
    for (int i=1; i<=n; i++) v[i]=a[i];
    sort(v+1,v+n+1);
    memset(g,0,sizeof g);
    for (int i=1; i<=n; i++){
        for (int j=1; j<=n; j++){
            f[j]=g[j]+abs(a[i]-v[j]);
        }
        g[0]=1e18;
        for (int j=1; j<=n; j++) g[j]=min(g[j-1],f[j]);
    }
    cout<<g[n];
}
