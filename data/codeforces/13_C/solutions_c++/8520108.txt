#include<bits/stdc++.h>
using namespace std;
int n;
int a[5005],b[5005];
long long c[5005];
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		b[i]=a[i];
	}
	sort(b,b+n);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			c[j]+=abs(a[i]-b[j]);
			if(j>0)
				c[j]=min(c[j],c[j-1]);
		}
	}
	cout<<c[n-1]<<endl;
	return 0;
}
