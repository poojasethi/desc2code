#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
	__int64 n,i,j,a[5005],b[5005],dp[5005];
	while(scanf("%I64d",&n) != EOF)
	{
		for(i=0;i<n;i++)
		{
			scanf("%I64d",&a[i]);
			b[i]=a[i];
		}
		sort(b,b+n);
		memset(dp,0,sizeof(dp));
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				dp[j]+=abs(a[i]-b[j]);
				if(j>0)
				{
					dp[j]=dp[j-1]<dp[j]?dp[j-1]:dp[j];
				}
			}
		}
		printf("%I64d\n",dp[n-1]);
	}
	return 0;
}
