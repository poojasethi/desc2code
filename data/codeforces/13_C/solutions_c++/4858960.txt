#include <iostream>
#include <algorithm>
using namespace std;

long long dp[2][5009], *f = dp[0], *g = dp[1], ans;
int a[5009], b[5009];
int n;

int main()
{
	cin >> n;
	for(int i = 0; i < n; i++)	cin >> a[i];
	for(int i = 0; i < n; i++)	b[i] = a[i];
	sort(b, b + n);
	
	for(int i = 0; i <= n; i++)
	{
		ans = f[0];
		for(int j = 0; j < n; j++)
		{
			ans = min(ans, f[j]);
			g[j] = ans + abs(b[j] - a[i]);
		}
		swap(f, g);
	}
	cout << ans << endl;
	return 0;
}
