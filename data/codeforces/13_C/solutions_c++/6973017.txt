#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cstdlib>
using namespace std;
const int maxn = 5010;
long long dp[maxn];
int a[maxn],b[maxn];

int main(){
    int n;
    scanf("%d",&n);
    for(int i = 0;i < n;i++){
        scanf("%d",&a[i]);
        b[i] = a[i];
    }
    sort(b,b+n);
    for(int i = 0;i < n;i++){
        for(int j = 0;j < n;j++){
            dp[j] += abs(a[i]-b[j]);
            if(j)   dp[j] = min(dp[j],dp[j-1]);
        }
    }
    printf("%I64d\n",dp[n-1]);
    return 0;
}
