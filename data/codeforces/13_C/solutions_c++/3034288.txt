#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
LL n,a[5050],b[5050],t[2][5050],*p=t[0],*q=t[1];
int main(void) {
  cin>>n;
  for(int i=0;i<n&&cin>>a[i];++i)b[i]=a[i];
  sort(b,b+n);
  for(int i=0;i<n;++i)p[i]=abs(b[i]-a[0]);
  for(int i=1;i<n;++i){
    for(int j=1;j<n;++j)p[j]=min(p[j],p[j-1]);
    for(int j=0;j<n;++j)q[j]=p[j]+abs(b[j]-a[i]);
    swap(p,q);
  }
  LL t=1LL<<60;
  for(int i=0;i<n;++i)t=min(t,p[i]);
  cout<<t<<endl;
}
