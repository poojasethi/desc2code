#include <cstdio>
#include <algorithm>
using namespace std;
int n;
long long a[5003],b[5003],f[5003];
inline long long Abs(long long x)
{
	if (x<0) return -x;
	return x;
}
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%I64d",a+i),b[i]=a[i];
	sort(b+1,b+n+1);
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++)
		{
			f[j]+=Abs(a[i]-b[j]);
			if (j>1) f[j]=min(f[j],f[j-1]);
		}
	printf("%I64d\n",f[n]);
	return 0;
}
