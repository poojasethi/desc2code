#include <cstdio>
#include <algorithm>
#include <cstdlib>
#include <cmath>
#include <iostream>

using namespace std;

long long int f[2][5005], n, a[5005], b[5005];

int main(){
	ios_base::sync_with_stdio(false);
	long long int i, j;
	cin >> n;
	for(i=1;i<=n;i++){
		cin >> a[i];
		b[i]= a[i];
	}
	sort(b+1,b+1+n);
	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++){
			f[i%2][j]= f[(i+1)%2][j] + abs(a[i]-b[j]);
			if(j>1) f[i%2][j]= min(f[i%2][j], f[i%2][j-1]);
		}
	}
	cout << f[n%2][n] << endl;
	return 0;
}
