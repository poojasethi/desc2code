#include <bits/stdc++.h>
using namespace std;

int N, a[5005], b[5005], id;
long long f[2][5005];

main()
{
    scanf("%d", &N);
    for (int i = 1; i <= N; i++) scanf("%d", &a[i]);

    for (int i = 1; i <= N; i++) b[i] = a[i];
    sort(b + 1, b + N + 1);

    for (int i = 1; i <= N; i++, id ^= 1)
    {
        f[id][0] = LONG_LONG_MAX;
        for (int j = 1; j <= N; j++)
            f[id][j] = min(f[id][j - 1], f[1 - id][j] + abs(a[i] - b[j]));
    }

    printf("%I64d", f[1 - id][N]);
}
