#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int a[5005];
int b[5005];
long long dp[5005];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		b[i]=a[i];
	}
	sort(b+1,b+n+1);
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			if(a[i]>b[j])
			    dp[j]+=a[i]-b[j];
			else
				dp[j]+=b[j]-a[i];
			if(j>1)
				dp[j]=min(dp[j],dp[j-1]);
		}
	}
	printf("%I64d\n",dp[n]);
	return 0;
}
