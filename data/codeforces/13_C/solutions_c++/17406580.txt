#include<bits/stdc++.h>
using namespace std;
long long int n,ar[100000],er[100000],dp[100000];
int main(){
	scanf("%lld",&n);
	for(int i=1 ; i<=n ;i++){
		scanf("%lld",&ar[i]);
		er[i]=ar[i];
	}
	sort(er+1,er+n+1);
	for(int i=1 ; i<=n ; i++){
		for(int j=1 ; j<=n ; j++){
			dp[j]+=abs(ar[i]-er[j]);
			if(j>1){
				dp[j]=min(dp[j],dp[j-1]);
			}
		}
	}
	printf("%lld\n",dp[n]);
	return 0;
}
