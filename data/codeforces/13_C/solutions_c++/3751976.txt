#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
long long f[5005];
int a[5005], b[5005];
int main()
{
	int n;
	while (cin >> n)
	{
		memset(f, 0, sizeof(f));
		for (int i = 1; i <= n; i++)
		{
			cin >> a[i];
			b[i] = a[i];
		}
		sort(b + 1, b + 1 + n);
		for (int i = 1; i <= n; i++)
			for (int j = 1; j <= n; j++)
			{
				f[j] += abs(a[i] - b[j]);
				if (j > 1)
					f[j] = min(f[j - 1], f[j]);
			}
		cout << f[n] << endl;
	}
	return 0;
}
