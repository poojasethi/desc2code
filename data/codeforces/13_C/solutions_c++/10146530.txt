#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
long long f[5001];
int a[5001],b[5001];
int main()
{
	int i,j,n;
	scanf("%d",&n);
    memset(f,0,sizeof f);
    for(i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        b[i]=a[i];
    }
    sort(b+1,b+n+1);
    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++)
        {
            f[j]+=abs(a[i]-b[j]);
            if(j>1)
            f[j]=min(f[j-1],f[j]);
        }
        printf("%I64d\n",f[n]);
    return 0;
}
