#include<iostream>
#include<algorithm>
#include<cmath>
#define MAXINT 5000
using namespace std;

int main()
{
	int n,i,j;
	int num[MAXINT+3],b[MAXINT+3];
	long long dp[MAXINT+3];
	while(cin>>n)
	{
		for(i=1;i<=n;i++)
		{
			cin>>num[i];
			b[i]=num[i];
		}
		sort(b+1,b+n+1);
		for(i=1;i<=n;i++)
			dp[i]=0;
		for(i=1;i<=n;i++)
			for(j=1;j<=n;j++)
			{
				dp[j]+=abs(num[i]-b[j]);
				if(j>1) dp[j]=min(dp[j-1],dp[j]);
			}
		cout<<dp[n]<<endl;
	}
}
