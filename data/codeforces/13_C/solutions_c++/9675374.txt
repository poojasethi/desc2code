#include <stdio.h>
#include <math.h>
#include <string.h>
#include <algorithm>
using namespace std;

int main()
{
	int a[5555],b[5555];
	__int64 dp[5555];
	int n;
	while(~scanf("%d",&n))
	{
		memset(dp,0,sizeof(dp));
		for(int i=1;i<=n;i++)
		{
			scanf("%d",&a[i]);
			b[i]=a[i];
		}
		sort(b+1,b+1+n);
		for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
		{
			dp[j]+=abs(a[i]-b[j]);
			if(j>1)
			dp[j]=min(dp[j-1],dp[j]);
		}
		printf("%I64d\n",dp[n]);
	}
	return 0;
}
