#include <iostream>
#include <algorithm>
using namespace std;

long long MAX = 9999999999999999;
int n, a[5001], b[5001];

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        cin >> a[i];
        b[i] = a[i];
    }
    sort(b + 1, b + n + 1); 
    int k = unique(b + 1, b + n + 1) - (b + 1);
    long long dp[5001] = {MAX};
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= k; ++j)
            dp[j] = min(dp[j-1], dp[j] + abs(a[i] - b[j])); 
    }
    cout << dp[k] << endl;
}
