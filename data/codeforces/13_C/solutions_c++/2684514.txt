# include<cstdio>
# include<iostream>
# include<algorithm>
# include<cmath>
using namespace std;

int a[5001],b[5001];
__int64 f[5001];

int main(void)
{
    int n;
    cin>>n;
    for(int i=1;i<=n;++i)
    {
        scanf("%d",&a[i]);
        b[i]=a[i];
    }
    sort(b+1,b+n+1);
    for(int i=1;i<=n;++i)
    {
        for(int j=1;j<=n;++j)
        {
          f[j]+=abs(a[i]-b[j]);
          if(j>1) f[j]=min(f[j],f[j-1]);
        }
    }
    printf("%I64d\n",f[n]);
return 0;
}
