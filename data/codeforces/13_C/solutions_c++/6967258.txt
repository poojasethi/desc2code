#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int MAX_N = 5007;

int n;
int a[MAX_N], b[MAX_N];
long long dp[MAX_N];

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		scanf("%d", &a[i]);
		b[i] = a[i];
	}
	sort(b + 1, b + 1 + n);
	dp[0] = 1LL<<61;
	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= n; ++j) {
			dp[j] = min(dp[j] + abs(a[i] - b[j]), dp[j - 1]);
		}
	}
	long long ans = dp[n];
	printf("%I64d\n", ans);
	return 0;
}
