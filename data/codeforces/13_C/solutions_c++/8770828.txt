#include<iostream>
#include<cstdlib>
#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;
int main()
{
    int n,i,j;
    scanf("%d",&n);

    long long  f[5005]= {0};
    int a[5005],b[5005];
    for(i=1; i<=n; i++)
    {
        scanf("%d",&a[i]);
        b[i]=a[i];
    }
    sort(b+1,b+1+n);
    for(i=1; i<=n; i++)
        for(j=1; j<=n; j++)
        {

            f[j]+=abs(a[i]-b[j]);
            if(j>1)
                f[j]=min(f[j-1],f[j]);
        }
    printf("%lld\n",f[n]);
}
