#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long a[5001],b[5001],f[5001]={0};
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        b[i]=a[i];
    }
    sort(b,b+n);
    for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
    {
        f[j]+=abs(a[i]-b[j]);
        if(j>0&&f[j-1]<f[j]) 
        f[j]=f[j-1];
    }
    cout<<f[n-1];
    return 0;
}
