#include<bits/stdc++.h>
using namespace std;
#define sp ' '
#define endl '\n'
int n;
int a[5005],b[5005];
long long f[5005];
int main()
{
    cin >> n;
    for (int i=1; i<=n; i++)
    {
        cin >> a[i];
        b[i]=a[i];
    }
    sort(b+1,b+n+1);
    for (int i=1; i<=n; i++)
        for (int j=1; j<=n; j++)
        {
            f[j]+=abs(a[i]-b[j]);
            if (j>1) f[j]=min(f[j],f[j-1]);
        }
    cout << f[n];
    return 0;
}
