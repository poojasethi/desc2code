#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <cstdlib>
using namespace std;
#define MAXN 5005
long long dp[MAXN];
int a[MAXN],b[MAXN];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]),b[i]=a[i];
	sort(b+1,b+n+1);
		for(int i=1;i<=n;i++)
			for(int j=1;j<=n;j++)
			{
			  if(j==1)dp[j]+=abs(a[i]-b[j]);
			  else
		      dp[j]=min(dp[j-1],dp[j]+abs(a[i]-b[j]));
			}
		printf("%lld\n",dp[n]);
	return 0;
}
