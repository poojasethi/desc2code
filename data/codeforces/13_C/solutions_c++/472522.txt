#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
int n;
ll s[5000],t[5000];
ll dp[5000];
int main(){
  while(cin>>n){
    for(int i = 0 ; i < n ; i++){
      cin>>t[i];
      s[i] = t[i];
      dp[i] = 0;
    } 
    sort(s , s+n);
    for(int i = 0 ; i < n ; i++){
      for(int j = 0 ; j < n ; j++){
	dp[j] += llabs(t[i] - s[j]);
	if(j > 0) dp[j] = min(dp[j-1],dp[j]);
      }
    }
    cout<<dp[n-1]<<endl;
  }
}
