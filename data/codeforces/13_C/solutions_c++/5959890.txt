#include <iostream>
#include <stdlib.h>
#include <algorithm>

// key observation: optimal solution transforms to sequence
// with same elements in different order
int main() {
	int n; std::cin >> n;
	long long a[n], b[n], c[n];
	for(int i = 0; i < n; ++i) {
		std::cin >> a[i];
		b[i] = a[i];
		c[i] = 0;
	}
	std::sort(b, b + n);
	for(int i = 0; i < n; ++i) {
		for(int j = 0; j < n; ++j) {
			c[j] += std::abs(a[i] - b[j]);
			if(j > 0)
				c[j] = std::min(c[j], c[j - 1]);
		}
	}
	std::cout << c[n - 1] << std::endl;
	return 0;
}
