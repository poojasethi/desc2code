#include <stdio.h>
#include <algorithm>
int a[6666],b[6666],n;
long long f[6666];
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){scanf("%d",&a[i]);b[i]=a[i];}
    std::sort(b,b+n);//改to那位
    for(int i=0;i<n;i++)for(int j=0;j<n;j++){
        f[j]+=abs(a[i]-b[j]);//全部數可加/減，目前的位置前所有數改至一個位置最少的cost我就可以不改那麼多
        f[j]=j&&f[j-1]<f[j]?f[j-1]:f[j];
    }
    printf("%I64d\n",f[n-1]);
}
