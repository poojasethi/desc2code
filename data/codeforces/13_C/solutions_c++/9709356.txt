#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

long long A[5005];
long long B[5005];
long long dp[5005];

int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%lld",&A[i]);
		B[i]=A[i];
	}
	sort(B,B+n+1);
	memset(dp,0,sizeof(dp));
	for(int i=1;i<=n;i++)
	{
		dp[1]+=abs(A[i]-B[1]);
		for(int j=2;j<=n;j++)
		dp[j]=min(dp[j]+abs(A[i]-B[j]),dp[j-1]);
	}
	printf("%lld\n",dp[n]);
	
	return 0;
}
