#include <algorithm>
#include <cstdio>
int a[5005],b[5005],i,j,n,m;
long long f[5005],inf=1e18;
int main()
{
	scanf("%d\n",&n);
	for (i=1;i<=n;++i) scanf("%d",&a[i]),b[i]=a[i];
	std::sort(b+1,b+n+1);
	m=std::unique(b+1,b+n+1)-b-1;
	for (i=1;i<=n;++i)
		for (f[0]=inf,j=1;j<=m;++j)
			f[j]=std::min(f[j-1],f[j]+abs(a[i]-b[j]));
	printf("%I64d\n",f[m]);
}
