n, m = map(int, raw_input().split())
mp, total = [""]*n, 0
rsum, csum = [0]*n, [0]*m
res = (-1, -1)

for i in xrange(n):
	s = raw_input()
	mp[i] = s
	for c in xrange (m):
		x = s[c] == '*'
		total += x
		rsum[i] += x
		csum[c] += x

for r in xrange(n):
	for c in xrange(m):
		sub = 1 if mp[r][c] == '*' else 0
		if rsum[r] + csum[c] - sub == total:
			res = (r + 1, c + 1)
			break

if res != (-1, -1):
	print "YES"
	print res[0], res[1]
else:
	print "NO"
