n, m = map(int, raw_input().split())
grid = [raw_input() for _ in xrange(n)]
rows = [r.count('*') for r in grid]
cols = [c.count('*') for c in zip(*grid)]
total = sum(rows)
ans = None
for r in xrange(n):
    for c in xrange(m):
        count = rows[r] + cols[c]
        if grid[r][c] == '*': count -= 1
        if count == total: ans = (r+1, c+1)
if ans is None:
    print 'NO'
else:
    print 'YES'
    print ans[0], ans[1]
