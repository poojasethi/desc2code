n, m = (int(x) for x in raw_input().split())
a = [raw_input() for _ in range(n)]
k = sum(a[i].count('*') for i in range(n))
r = [R.count('*') for R in a]
c = [C.count('*') for C in zip(*a)]
for i in range(n):  
    for j in range(m):
        if r[i] + c[j] - (a[i][j] == '*') == k:
            print 'YES'
            print i + 1, j + 1
            exit(0)
print 'NO'
