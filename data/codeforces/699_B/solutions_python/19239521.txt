import sys
n, m = tuple(map(int, raw_input().split()))

rs = [0]*n
cs = [0]*m
d = []
counter = 0

for i in range(n):
	temp = raw_input()
	d.append(temp)
	for j in range(m):
		if temp[j] == '*':
			rs[i] += 1
			cs[j] += 1
			counter += 1

for r in range(n):
	for c in range(m):
		if rs[r] + cs[c] == counter + 1 and d[r][c] == '*' or rs[r] + cs[c] == counter and d[r][c] == '.':
			print 'YES'
			print '%d %d' % (r+1, c+1)
			sys.exit()

print 'NO'
