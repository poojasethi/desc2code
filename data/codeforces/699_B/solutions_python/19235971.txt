L, W = (int(n) for n in raw_input().strip().split())
grid = [raw_input().strip() for i in range(L)]
rows = []
cols = []
for i in xrange(L):
    rows.append(grid[i].count('*'))
bombs = sum(rows)
for j in xrange(W):
    cols.append(''.join(grid[i][j] for i in xrange(L)).count('*'))
for i in xrange(L):
    for j in xrange(W):
        if rows[i] + cols[j] + (-1 if grid[i][j] == '*' else 0) == bombs:
            print "YES"
            print i+1,j+1
            exit()
print "NO"
