n, m = (int(x) for x in raw_input().split())
a = [raw_input() for _ in range(n)]
k = sum(a[i].count('*') for i in range(n))
r = [0] * n
c = [0] * m
for i in range(n):
    r[i] = sum([a[i][j] == '*' for j in range(m)])
for j in range(m):
    c[j] = sum([a[i][j] == '*' for i in range(n)])
for i in range(n):
    for j in range(m):
        if r[i] + c[j] - (a[i][j] == '*') == k:
            print 'YES'
            print i + 1, j + 1
            exit(0)
print 'NO'
