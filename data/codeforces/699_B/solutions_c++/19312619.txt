#include <bits/stdc++.h>
#include <iostream>

using namespace std;
int r[1000],c[1000];
int main() 
{
	int n,m;
	cin>>n>>m;
	int cur=0,cnt;
	string s[1000];
	for(int i=0;i<n;i++)
	{
		cin>>s[i];
		for(int j=0;j<m;j++)
		{
			if(s[i][j]=='*')
			{
				cur++;
				r[i]++;
				c[j]++;
			}
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			cnt=r[i]+c[j];
			if(s[i][j]=='*')
			cnt--;
			if(cnt==cur)
			{cout<<"YES"<<endl; cout<<i+1<<" "<<j+1<<endl; return 0;}
			
		}
	}
	
	cout<<"NO"<<endl;
	// your code goes here
	return 0;
}
