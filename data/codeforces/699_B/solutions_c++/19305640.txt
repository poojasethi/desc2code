#include <iostream>

using namespace std;

int main()
{
	char c[1001][1001];
	int n, m, a[1001] = {0}, b[1001] = {0}, k = 0, s;
	cin >> n >> m;
	for(int i = 0; i < n; i++)
		for(int j = 0; j < m; j++)
		{
			cin >> c[i][j];
			if(c[i][j] == '*')
				a[i]++, b[j]++, k++;
		}
	for(int i = 0; i < n; i++)
		for(int j = 0; j < m; j++)
		{
			s = a[i]+b[j];
			if(c[i][j] == '*') s--;
			if(s == k)
			{
				cout << "YES\n" << i+1 << ' ' << j+1 << endl;
				return 0;
			}
		}
	cout << "NO\n";
}
