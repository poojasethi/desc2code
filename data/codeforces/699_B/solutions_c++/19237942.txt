#include <stdio.h>
#include <iostream>
#include <cstring>
const int N=1005;
int n,m;
char s[N][N];
int x[N],y[N];
using namespace std;

int main()
{
	//freopen ("A.in","r",stdin);
	scanf ("%d %d",&n,&m);
	int i,j,sum=0;
	for (i=1;i<=n;i++) scanf ("%s",s[i]+1);
	for (i=1;i<=n;i++)
	for (j=1;j<=m;j++)
	x[i]+=(s[i][j]=='*'),y[j]+=(s[i][j]=='*'),sum+=(s[i][j]=='*');
	for (i=1;i<=n;i++)
	for (j=1;j<=m;j++)
	if ((x[i]+y[j]-(s[i][j]=='*'))==sum){printf ("YES\n%d %d",i,j);return 0;}
	printf ("NO");
	return 0;
}
