#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

const int N=1005;
char a[N][N];
int n,m,r[N],c[N],sum;

int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%s",&a[i]);
		for(int j=0;j<m;j++){
			if(a[i][j]=='*'){
				r[i]++;c[j]++;
				sum++;
			} 
		}
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			int num=r[i]+c[j]-(a[i][j]=='*'?1:0);
			if(num==sum){
				printf("YES\n%d %d\n",i+1,j+1);
				return 0;
			}
		}
	}
	printf("NO\n");
	return 0;
}
