#include <iostream>
using namespace std;
int main()
{
	typedef long long ll;
	ll n,m,r[1006]={0},c[1006]={0},t=0;
	cin>>n>>m;
	char a[n+2][m+2];
	for (int i=0; i<n; i++)
	for (int j=0; j<m; j++)
	{
		char p;
		cin>>p;
		if(p=='*') 
		{
			r[i]++;
			c[j]++;	
			t++;
		}
		a[i][j]=p;
	}
	for (int i=0; i<n; i++)
	for (int j=0; j<m; j++) if(r[i]+c[j]-1==t or t==0 or (r[i]+c[j]==t and a[i][j]!='*')) 
	{
		cout<<"YES"<<endl;
		cout<<i+1<<' '<<j+1;
		return 0;
	}
	cout<<"NO";
}
