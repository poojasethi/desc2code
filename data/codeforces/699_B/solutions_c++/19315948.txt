#include<cstdio>
using namespace std;
const int N = 1010;
int n,m,cx[N],cy[N],tot;
bool G[N][N];
char s[N];
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%s",s);
		for(int j=1;j<=m;j++){
			if(s[j-1]=='*') G[i][j]=1,cx[i]++,cy[j]++,tot++;
		}
	}
	int res=0;
	for(int x=1;x<=n;x++){
		for(int y=1;y<=m;y++){
			res=cx[x]+cy[y];
			if(G[x][y]==1) res--;
			if(tot==res){
				printf("YES\n");
				printf("%d %d\n",x,y);
				return 0;
			}
		}
	}
	printf("NO\n");
	return 0;
}
