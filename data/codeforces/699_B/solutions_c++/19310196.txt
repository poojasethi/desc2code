#include <cstdio>
#define N 1005

int n, m, k, cnt, r[N], c[N];
char a[N][N];
int main() {
	int i, j;
	scanf("%d%d", &n, &m);
	for(i = 1; i <= n; i++) {
		scanf("%s", &a[i][1]);
		for(j = 1; j <= m; j++) {
			if(a[i][j] == '*') {
				r[i]++;
				c[j]++;
				cnt++;
			}
		}
	}
	for(i = 1; i <= n; i++) {
		for(j = 1; j <= m; j++) {
			k = r[i] + c[j];
			if(a[i][j] == '*') k--;
			if(k == cnt) {
				puts("YES");
				return !printf("%d %d", i, j);
			}
		}
	}
	puts("NO");
	return 0;
}
