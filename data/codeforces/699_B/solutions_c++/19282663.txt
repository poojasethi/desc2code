#include <iostream>
using namespace std;
int n,m,sx[1010],sy[1010],cnt=0,ans=0,x,y;
char s[1010][1010];
int main(){
	cin>>n>>m;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			cin>>s[i][j];
			if(s[i][j]=='*'){
				sx[i]++;
				sy[j]++;
				cnt++;		
			}
		}
		
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			if(sx[i]+sy[j]-(s[i][j]=='*')==cnt){
				cout<<"YES"<<endl;
				cout<<i+1<<" "<<j+1;
				return 0;
			
			}
		}
		
	}
	cout<<"NO";
			
}
