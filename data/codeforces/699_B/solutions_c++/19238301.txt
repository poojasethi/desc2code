#include <cstdio>

int n, m, ansx, ansy, cntx, cnty, tot;
char cell[1005][1005];
int row[1005], col[1005];

int main() {
	scanf("%d %d", &n, &m);
	for (int i = 0; i < n; ++i) scanf("%s", cell[i]);
	for (int i = 0; i < n; ++i) for (int j = 0; j < m; ++j) if (cell[i][j] == '*') {
		++tot; ++row[i]; ++col[j];
	}
	for (int i = 0; i < n; ++i) for (int j = 0; j < m; ++j)
		if (row[i] + col[j] - (cell[i][j] == '*') == tot) {
			puts("YES");
			printf("%d %d\n", i + 1, j + 1);
			return 0;
		}
	puts("NO");
	return 0;
}
