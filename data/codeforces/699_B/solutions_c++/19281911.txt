#include <cstdio>

char c[1200][1200];
int d1[1200],d2[1200];
int i,j,m,n,t;

int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++) scanf("%s",c[i]+1);
	for (i=1;i<=n;i++) for (j=1;j<=m;j++) if (c[i][j]=='*') d1[i]++,d2[j]++,t++;
	for (i=1;i<=n;i++) for (j=1;j<=m;j++) if (d1[i]+d2[j]-(c[i][j]=='*')==t)
	{ printf("YES\n%d %d\n",i,j); return 0; }
	printf("NO\n");
	return 0;
}
