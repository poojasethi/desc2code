#include<stdio.h>
int x[1010];
int y[1010];
int main()
{
	int n,m;
	int i,j;
	char a[1010][1010];
	int num;
	scanf("%d%d",&n,&m);
	num=0;
	for(i=1;i<=n;i++)
	{
		scanf("%s",a[i]+1);
		for(j=1;j<=m;j++)
			if(a[i][j]=='*')
			{
				x[i]++;
				y[j]++;
				num++;
			}
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
			if(a[i][j]!='*'&&x[i]+y[j]==num||x[i]+y[j]==num+1)
				break;
		if(j<=m)
			break;
	}
	if(i<=n)
		printf("YES\n%d %d\n",i,j);
	else
		printf("NO\n");
	return 0;
}
