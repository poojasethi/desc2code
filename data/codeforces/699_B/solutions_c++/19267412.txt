#include <map>
#include <set>
#include <queue>
#include <cstdio>
#include <vector>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
char c[1005][1005];
int sum,l[1005],r[1005];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	for(int j=1;j<=m;j++)
	{
	cin>>c[i][j];
	if(c[i][j]=='*')
	{
	r[i]++;
	l[j]++;
	sum++;
	}
	}
	for(int i=1;i<=n;i++)
	for(int j=1;j<=m;j++)
	if(r[i]+l[j]-(c[i][j]=='*')==sum)
	{
	cout<<"YES"<<endl<<i<<" "<<j;
	return 0;
	}
	cout<<"NO";
	return 0;
}
