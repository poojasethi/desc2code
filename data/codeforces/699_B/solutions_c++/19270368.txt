#include <iostream>
using namespace std;
int row[1000000],col[100000];
char a[1000][1000];
int main() {
	int n,m;
	cin>>n>>m;
	int count=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			cin>>a[i][j];
			if(a[i][j]=='*')
			{row[i]++;col[j]++;
			count++;}
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if((row[i]+col[j]-(a[i][j]=='*'))==count)
			{
				cout<<"YES"<<endl;
				cout<<i+1<<" "<<j+1;
				return 0;
			}
		}
	}
	cout<<"NO";
	// your code goes here
	return 0;
}
