#include <iostream>

using namespace std;

int n, m, k, f, a[1001], b[1001];
char c[1000][1000];

int main()
{
	cin >> n >> m;

	for(int i = 0; i < n; ++i)
		for(int j = 0; j < m; ++j)
		{
			cin >> c[i][j];
			if(c[i][j] == '*')
			{
				k++;
				a[i]++;
				b[j]++;
			}
		}

	for(int i = 0; i < n; ++i)
		for(int j = 0; j < m; ++j)
		{
			f = a[i] + b[j];
			if(c[i][j] == '*')
				f--;
			if(f == k)
			{
				cout << "YES" << endl;
				cout << i + 1 << ' ' << j + 1;
				return 0;
			}
		}

	cout << "NO";
}
