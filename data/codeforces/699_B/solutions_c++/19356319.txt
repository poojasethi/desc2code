#include <cstdio>
using namespace std;
int main()
{
	int n,m,h[1010]={0},l[1010]={0},ans=0;
	char a[1010][1010];
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
	{
	    scanf("%s",a[i]);
		for(int j=0;j<m;j++)
		{

			if(a[i][j]=='*')
			{
				h[i]++;
				l[j]++;
				ans++;
			}
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			int tmp=h[i]+l[j];
			if(a[i][j]=='*')tmp--;
			if(tmp==ans)
			{
				printf("YES\n%d %d",i+1,j+1);
				return 0;
			}
		}
	}
		printf("NO\n");
	return 0;
}
