#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
bool w[1010][1010];
int n,m,row[1010],col[1010],cnt;
int main(){
	cin>>n>>m;
	for(int i=0;i<n;i++){
		string s;
		cin>>s;
		for(int j=0;j<m;j++){
			if (s[j]=='*'){
				w[i][j]=true;
				row[i]++;
				col[j]++;
				cnt++;
			}
		}
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			int res=cnt-row[i]-col[j];
			if (w[i][j]) res++;
			if(res==0) {
				cout<<"YES"<<endl<<i+1<<' '<<j+1;
				return 0;
			}
		}
	}
	cout<<"NO";
	return 0;
}
