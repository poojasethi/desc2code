#include<cstdio>
char L[1003][1003];
int N[1003],M[1003],i,j,m,n,t;
inline int min(int a,int b){return a<b?a:b;}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)scanf("%s",&L[i][1]);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if(L[i][j]=='*')N[i]++,M[j]++,t++;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if(N[i]+M[j]==t+(L[i][j]=='*')){
				printf("YES\n%d %d",i,j);
				return 0;
			}
	printf("NO");
}
