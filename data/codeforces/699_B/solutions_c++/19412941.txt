#include<stdio.h>
using namespace std;

{
	int n,m,count=0,i,k,j,r[1005],c[1005];
	char str[1005][1005];
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;i++)
	scanf("%s",&str[i][1]);

	{
		for(j=1;j<=m;j++)
		{
			if(str[i][j]=='*')
			{
				r[i]++;c[j]++;count++;
			}
		}
	}
	
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			if(r[i]+c[j]==count+(str[i][j]=='*'))

				return 0;
			}
		
		}
	}

}
