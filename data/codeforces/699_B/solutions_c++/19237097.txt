#include <stdio.h>
int row[1010],col[1010];
char a[1010][1010];
int main()
{
	// freopen("test.out","w",stdout);
	// freopen("test.in","r",stdin);
	int i,j,n,m,sum=0;
	scanf("%d %d",&n,&m);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			scanf(" %c",&a[i][j]);
			if(a[i][j]=='*')
			{
				sum++;
				row[i]++;
				col[j]++;
			}
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			if(row[i]+col[j]-(a[i][j]=='*')==sum)
			{
				printf("YES\n%d %d\n",i+1,j+1);
				return 0;
			}
		}
	}
	printf("NO\n");
	return 0;
}
