#include<bits/stdc++.h>
using namespace std;
	int row[1010];
	int col[1010];
	
int main()
{
	int n,m;
	cin>>n>>m;
	string str[n];
	for(int i=0;i<n;i++)
	{
		cin>>str[i];
	}

	int cnt=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(str[i][j]=='*')
			{
				row[i]++;
				col[j]++;
				cnt++;
			}
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if((row[i]+col[j]-(str[i][j]=='*'))==cnt)
			{
				printf("YES\n%d %d",i+1,j+1);
				return 0;
			}
		}
	}
	printf("NO");
}
