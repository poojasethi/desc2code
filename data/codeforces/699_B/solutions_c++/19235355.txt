#include <stdio.h>
#include <algorithm>
using namespace std;
char a[1005][1005];
int n, m, row[1005], line[1005], cnt;
main()
{
	scanf("%d %d\n", &n, &m);
	for(int i=0; i<n; i++)
	{
		scanf("%s", a[i]);
		for(int j=0; j<m; j++)
		if(a[i][j]=='*')
		{
			cnt++;row[i]++;line[j]++;
		}
	}
	for(int i=0; i<n; i++)
	{
		for(int j=0; j<m; j++)
		{
			int temp=row[i]+line[j]-(a[i][j]=='*');
			if(temp==cnt)
			{
				printf("YES\n%d %d", i+1, j+1);
				return 0;
			}
		}
	}
	puts("NO");
}
