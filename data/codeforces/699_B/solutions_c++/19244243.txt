#include<cstdio>
char map[1005][1005];
int cntx[1005]={0},cnty[1005]={0};
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	int ans=0;
	int ansx=-1,ansy=-1;
	for(int i=1;i<=n;i++)scanf("%s",map[i]+1);
	for(int i=1;i<=n;i++)for(int j=1;j<=m;j++)if(map[i][j]=='*')
	{
		ans++;
		cntx[i]++;
		cnty[j]++;
	}
	for(int i=1;i<=n;i++)for(int j=1;j<=m;j++)
	{
		int cnt=cntx[i]+cnty[j];
		if(map[i][j]=='*')cnt-=1;
		if(cnt==ans){ansx=i;ansy=j;}
	}
	if(ansx==-1&&ansy==-1)puts("NO");
	else printf("YES\n%d %d\n",ansx,ansy);
	return 0;
}
