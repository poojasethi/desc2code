#include<iostream>
using namespace std;
bool f[1001][1001];
char s;
int hang[1001],lie[1001],n,m,sum;
int main()
{
	cin>>n>>m;
	for (int i=1;i<=n;i++)
		for (int j=1;j<=m;j++)
		{
			cin>>s;
			if (s=='*') {f[i][j]=1;hang[i]++;lie[j]++;sum++;} 
		}
	for (int i=1;i<=n;i++)
		for (int j=1;j<=m;j++)
			if (hang[i]+lie[j]-f[i][j]==sum) {cout<<"YES"<<endl;cout<<i<<" "<<j<<endl;return 0;}
	cout<<"NO";
}
