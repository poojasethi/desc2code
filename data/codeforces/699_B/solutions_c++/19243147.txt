#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int n,m,i,j,f[1010],f1[1010],cnt;
char s[1010][1010];
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)scanf("%s",s[i]+1);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if(s[i][j]=='*')f[i]++,f1[j]++,cnt++;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if((f[i]+f1[j]-(s[i][j]=='*'))==cnt){
				puts("YES");
				printf("%d %d",i,j);
				return 0;
			}
	puts("NO");
	return 0;
}
