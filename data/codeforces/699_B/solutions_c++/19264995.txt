#include<cstdio>
using namespace std;
const int MAX=1005;
char a[MAX][MAX];
int row[MAX],column[MAX];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	int sum=0;
	for(int i=0;i<n;i++)
	{
		scanf("%s",&a[i]);
		for(int j=0;j<m;j++)if(a[i][j]=='*')
		{
			row[i]++;column[j]++;
			sum++;
		}
	}
	
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			{
				int cnt=row[i]+column[j]+(a[i][j]=='*'?-1:0);
				if(cnt==sum)
				{
					printf("YES\n%d %d",i+1,j+1);
					return 0;
				}
			}
	printf("NO");
	return 0;
}
