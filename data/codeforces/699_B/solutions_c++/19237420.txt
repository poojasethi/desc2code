#include<cstdio>
#define MAXN 1000

int n,m,row[MAXN+10],col[MAXN+10],tot;
char mat[MAXN+10][MAXN+10];
void read()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		scanf("%s",mat[i]+1);
		for(int j=1;j<=m;j++)
			if(mat[i][j]=='*'){
				tot++;
				row[i]++,col[j]++;
			}
	}
}
int main()
{
	read();
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++){
			int t=col[j]+row[i]-(mat[i][j]=='*'?1:0);
			if(t==tot){
				puts("YES");
				printf("%d %d\n",i,j);
				return 0;
			}
		}
	puts("NO");
}
