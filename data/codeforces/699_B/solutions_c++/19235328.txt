#include <bits/stdc++.h>
using namespace std;
char f[1010][1010];
int col[1010],row[1010],n,m,sum;
int main(){
	//freopen("B.in","r",stdin);
	cin >> n >> m;
	for(int i = 1; i <= n; i++){
		for(int j = 1; j <= m; j++){
			cin >> f[i][j];
			if(f[i][j]=='*') col[j]++,row[i]++,sum++;
		}
	}
	for(int i = 1; i <= n; i++){
		for(int j = 1; j <= m; j++){
			int minus = ((f[i][j]=='*')?1:0);
			if(col[j]+row[i]-minus==sum){
				cout << "YES\n";
				cout << i << " " << j;
				return 0;
			}
		}
	}
	cout << "NO";
}
