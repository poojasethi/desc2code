#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long n, m, i, j, t=0, a[1005]={0}, b[1005]={0}, s;
    char c[1005][1005];
    cin>>n>>m;
    for(i=0; i<n; i++)
    for(j=0; j<m; j++){
        cin>>c[i][j];
        if(c[i][j]=='*') {a[i]++, b[j]++, t++;}
    }
    for(i=0; i<n; i++)
    for(j=0; j<m; j++){
        s=a[i]+b[j];
        if(c[i][j]=='*') s--;
        if(t==s) {cout<<"YES"<<endl<<i+1<<" "<<j+1<<endl; return 0;}
    }
    cout<<"NO"<<endl;
    return 0;
}
