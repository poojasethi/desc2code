#include<iostream>
#include<cstdio>
using namespace std;
int col[1005],row[1005];
char g[1005][1005];
int main(){
	int n,m;
	int sum=0;
	scanf("%d %d",&n,&m);
	for (int i=0;i<n;i++){
		scanf("%s",g[i]);
		for (int j=0;j<m;j++){
			if (g[i][j]=='*') col[j]++,row[i]++,sum++;
		}
	}
	for (int i=0;i<n;i++){
		for (int j=0;j<m;j++){
			int S=col[j]+row[i];
			if (g[i][j]=='*') S--;
			if (S==sum){
				printf("YES\n");
				printf("%d %d\n",i+1,j+1);
				return 0;
			}
		}
	}
	printf("NO\n");
	return 0;
}
