#include<bits/stdc++.h>
using namespace std;
int i,j,n,m,t,xx,yy,x[1000005],y[1000005],cx[1005],cy[1005];
char s[1005][1005];
int main()
{
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	{
		scanf("%s",s[i]+1);
		for(j=1;j<=m;j++)
			if(s[i][j]=='*')x[++t]=i,y[t]=j;
	}	
	for(i=1;i<=t;i++)
		++cx[x[i]],++cy[y[i]];
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if(cx[i]+cy[j]-(s[i][j]=='*')==t)xx=i,yy=j;
	if(xx)printf("YES\n%d %d",xx,yy);else
	puts("NO");
}
