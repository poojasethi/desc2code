#include <iostream>
using namespace std;

char field[1010][1010];
int x[1010], y[1010];

int main() {
	int n, m, cnt=0;
	cin>>n>>m;
	for  (int i=0; i<n; i++) {
		for (int j=0; j<m; j++) {
			cin>>field[i][j];
			if (field[i][j]=='*') {
				x[i]++, y[j]++;
				cnt++;
			}
		}
	}
	for (int i=0; i<n; i++) {
		for (int j=0; j<m; j++) {
			if (x[i]+y[j]-(field[i][j]=='*')==cnt) {
				cout<<"YES\n"<<i+1<<' '<<j+1<<'\n';
				return 0;
			}
		}
	}
	cout<<"NO\n";
	return 0;
}
