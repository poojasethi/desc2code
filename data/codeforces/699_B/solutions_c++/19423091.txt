#include<iostream>
using namespace std;
char x;
bool q[1010][1010];
int a[1010],b[1010];
int s,n,m;
int main(){
	cin>>n>>m;
	for(int i = 1; i <= n; i++)
	{
		for (int j = 1; j <=m; j++)
		{
			cin >> x;
			if (x == '*'){
				q[i][j] = 1;
				s++;
				a[i]++;
				b[j]++;
			}
		}
	}
	for (int i =1; i <=n ; i++){
		for (int j = 1; j <= m; j++){
			if (a[i]+ b[j] - q[i][j] == s){
				cout << "YES\n" << i << " " << j;
				return (0);
			}
		}
	}
	cout <<"NO";
}
