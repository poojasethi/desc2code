#include <bits/stdc++.h>
using namespace std;
int row[1010], col[1010];
string s[1010];
int total;
int main() {
	int n, m;
	cin >> n >> m;
	for(int i = 0; i < n; i++) cin >> s[i];
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < m; j++) {
			if(s[i][j] == '*') row[i]++, col[j]++, total++;
		}
	}
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < m; j++) {
			int sum = row[i] + col[j];
			if(s[i][j] == '*') sum--;
			if(sum == total) {
				cout << "YES\n";
				cout << i + 1 << " " << j + 1 << endl;
				return 0;
			}
		}
	}
	cout << "NO\n";
}
