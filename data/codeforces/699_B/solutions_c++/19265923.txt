#include <stdio.h>
char ma[1005][1005];
int c[1005],w[1005],b=0,n,m;
int main()
{
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;i++)
        scanf("%s",ma[i]+1);
    for (int i=1;i<=n;i++)
        for (int j=1;j<=m;j++)
            if (ma[i][j]=='*')
                b++,c[i]++,w[j]++;
    for (int i=1;i<=n;i++)
        for (int j=1;j<=m;j++)
        if(c[i]+w[j]-(ma[i][j]=='*')==b)
        {
            printf("YES\n%d %d\n",i,j);
            return 0;
        }
    printf("NO\n");
    return 0;
}
