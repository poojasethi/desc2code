
 #include <cstdio>
char A[1010][1010];
int Sr[1010],Sc[1010];
int main(){
	int n,m,cnt=0;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%s",A[i]);
		for(int j=0;j<m;j++){
			if(A[i][j]=='*')Sr[i]++,Sc[j]++,cnt++;
		}
	}
	for(int i=0;i<n;i++)for(int j=0;j<m;j++){
		if(Sr[i]+Sc[j]-(A[i][j]=='*')==cnt)return!printf("YES\n%d %d",i+1,j+1);
	}
	printf("NO");
}
