#include <iostream>
using namespace std;

int main() {
	int a[1001][1001],n,m,x[1001]={0},y[1001]={0},count=0;
	cin>>n>>m;
	char c;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			cin>>c;
			a[i][j]=c;
			if(c=='*')
			{
				count++;
				x[i]++;
				y[j]++;
			}
		}
	}
	for(int i=1;i<=1000;i++)
	{
		for(int j=1;j<=1000;j++)
		{
			int val=x[i]+y[j];
			if(a[i][j]=='*')
			val-=1;
			if(val==count)
			{
				cout<<"YES\n";
				cout<<i<<" "<<j;
				return 0;
				
			}
		}
	}
	cout<<"NO";
	return 0;
}
