#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,i,j,val,cnt=0;
	cin>>n>>m;
	char ch;
	char a[n+1][m+1];
	int arr[n+1],brr[m+1];
	memset(arr,0,sizeof(arr));
	memset(brr,0,sizeof(brr));
	for(i=1;i<=n;++i)
	{
		for(j=1;j<=m;++j){
			cin>>a[i][j];
			if(a[i][j]=='*')
			{	cnt++;
				arr[i]++; brr[j]++;
			}
		}
	}
	for(i=1;i<=n;++i)
	{
		for(j=1;j<=m;++j)
		{
			val=arr[i]+brr[j];
			if(a[i][j]=='*') val-=1;
			if(val==cnt) { cout<<"YES\n"<<i<<" "<<j; return 0;}
		}
	}
	 cout<<"NO";
	return 0;
}
