#include <bits/stdc++.h>
using namespace std;

const int N=1005;
char tp[N][N],sum1[N],sum2[N],sum;
int n,m;

int main()
{
	scanf("%d%d",&n,&m);
	int i,j;
	for(i=1;i<=n;i++)
	{
	    scanf("%s",tp[i]+1);
	    for(j=1;j<=m;j++)
	        if(tp[i][j]=='*')
	            sum1[i]++,sum2[j]++,sum++;
	}
	for(i=1;i<=n;i++)
	    for(j=1;j<=m;j++)
	        if(sum1[i]+sum2[j]-((tp[i][j]=='*')?1:0)==sum)
	        {
	        	printf("YES\n%d %d",i,j);
	        	return 0;
	        }
	printf("NO");
}
