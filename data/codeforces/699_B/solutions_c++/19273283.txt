#include<cstdio>
#define fo(i,a,b) for(int i=a;i<=b;i++)
int n,m,x[1200],y[1200],tot;
char s[1200][1200];
int main(){
	scanf("%d%d",&n,&m);
	fo(i,1,n){
		scanf("%s",s[i]+1);
		fo(j,1,m)
			if (s[i][j]=='*'){
				tot++;
				x[i]++;
				y[j]++;
			}
	}
	fo(i,1,n)
		fo(j,1,m)
			if ((x[i]+y[j]-(s[i][j]=='*'))==tot){
				printf("YES\n%d %d\n",i,j);
				return 0;
			}
	printf("NO\n");
	return 0;
}
