#include<bits/stdc++.h>

using namespace std;

char p[1000][1000]; int a[1000], b[1000]; int i, j, k, n, m;

int main(){
    scanf("%d%d", &n, &m);

    for (i=0; i<n; i++){
            scanf("%s", p[i]);
            for(j=0;j<m; j++) if(p[i][j]=='*'){ a[i]++; b[j]++; k++;}
    }

    for (i=0; i<n; i++)
        for(j=0;j<m; j++)
            if ((a[i] + b[j] - (p[i][j]=='*') ) == k) {
                printf("%s%d %d", "YES\n", i+1, j+1);
                return 0;
            }
    printf ("%s", "NO");
    return 0;
}
