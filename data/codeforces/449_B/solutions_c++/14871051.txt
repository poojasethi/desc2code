#include<bits/stdc++.h>
#define mp make_pair
#define pb push_back

using namespace std;
typedef long long ll;
typedef pair<int,int>pii;
typedef pair<ll,int>pli;
#define X first
#define Y second
#define N 100005
vector< pii > adj[N];
int n,m,k,ans=0;
bool vis[N];

int main()
{
    scanf("%d%d%d",&n,&m,&k);
    for(int i=0,u,v,w;i<m;i++)
    {
        scanf("%d%d%d",&u,&v,&w);
        adj[u].pb(mp(v,w));
        adj[v].pb(mp(u,w));
    }
    priority_queue<pli>pq;
    for(int i=0,v,w;i<k;i++)
    {
        scanf("%d%d",&v,&w);
        pq.push(mp(-w,-v));
    }
    pq.push(mp(0,1));
    while(!pq.empty())
    {
        pli cur=pq.top();
        pq.pop();
        int u=cur.Y;
        ll d= cur.X;
        if(u<0 && vis[u=-u]) ans++;
        if(vis[u])continue;
        vis[u]=true;
        for(int i=0;i<adj[u].size();i++)
        {
            int v=adj[u][i].X;
            if(!vis[v]) pq.push(mp(d-adj[u][i].Y,v));
        }
    }
    printf("%d\n",ans);
    return 0;
}
