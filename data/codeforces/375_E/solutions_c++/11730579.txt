#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <cstdlib>
using namespace std;
const int N(1021);
typedef double LD;
const LD eps(1e-6);
struct hh{
    int v,w,nxt;
}edges[N*2];
int n,m,D,l,en,cnt,point[N];
LD a[N][N];
inline int dcmp(LD x){
    return (x>eps)-(x<-eps);
}
inline void read(int &x){
    char c=0;int f=1;
    for(;c<'0'||c>'9';c=getchar())if(c=='-')f=-1;
    for(x=0;c>='0'&&c<='9';c=getchar())x=x*10+(c-'0');
    x*=f;
}
inline void addedge(int u,int v,int w){
    edges[++en]=(hh){v,w,point[u]};
    point[u]=en;
}
inline void dfs(int x,int fa,long long dis){
    if(dis<=D)a[l][x]=-1;
    else return;
    for(int i(point[x]),v;i;i=edges[i].nxt)
        if((v=edges[i].v)!=fa)
            dfs(v,x,dis+edges[i].w);
}
inline void pivot(int l,int e){
    for(int i(0);i<=m;i++)
        if(i!=e)
            a[l][i]/=a[l][e];
    a[l][e]=1/a[l][e];
    for(int i(0);i<=n;i++)
        if(i!=l){
            if(!dcmp(a[i][e]))continue;
            for(int j(0);j<=m;j++)
                if(j!=e)
                    a[i][j]-=a[i][e]*a[l][j];
            a[i][e]*=-a[l][e];
        }
}
inline int simplex(){
    for(;;){
        int l=1,e=1;
        for(int i(1);i<=n;i++)if(a[i][0]<a[l][0])l=i;
        if(dcmp(a[l][0])>=0)break;
        for(int i(1);i<=m;i++)if(a[l][i]<a[l][e])e=i;
        if(dcmp(a[l][e])>=0){puts("-1");exit(0);}
        pivot(l,e);
    }
    for(;;){
        int l=1,e=1;
        for(int i(1);i<=m;i++)if(a[0][i]>a[0][e])e=i;
        if(dcmp(a[0][e])<=0)return -a[0][0];
        for(int i(1);i<=n;i++)if(dcmp(a[i][0])>0)l=i;
        for(int i(1);i<=n;i++)
            if(dcmp(a[i][e])>0&&a[i][0]*a[l][e]<a[l][0]*a[i][e])
                l=i;
        if(dcmp(a[l][e])<=0){puts("-1");exit(0);}
        pivot(l,e);
        
    }
}
int main(){
//  freopen("test.in","r",stdin);
    read(n);read(D);
    for(int i(1),t;i<=n;i++){
        read(t);cnt+=t;
        a[0][i]=t-1;
    }
    for(int i(1),u,v,w;i<n;i++){
        read(u);read(v);read(w);
        addedge(u,v,w);
        addedge(v,u,w);
    }
    for(int i(1);i<=n;i++){
        l=i;
        a[i][0]=-1;
        dfs(i,0,0);
    }
    m=n;n<<=1;
    for(int i(m+1);i<=n;i++)a[i][0]=1,a[i][i-m]=1;
    a[n+1][0]=cnt;for(int i(1);i<=m;i++)a[n+1][i]=1;
    a[n+2][0]=-cnt;for(int i(1);i<=m;i++)a[n+2][i]=-1;
    n+=2;
    printf("%d\n",(int)(-simplex()+.5));
    return 0;
}
