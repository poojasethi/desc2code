#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <vector>
#include <ctime>
#define fi first
#define se second
#define PA pair<int,int>
#define VI vector<int>
#define VP vector<PA >
#define mk(x,y) make_pair(x,y)
#define N 501
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,s,t;
int co[N],si[N],L[N],R[N];
int64 dis[N][N];
short f[N][N][N],g[N][N];
VI zi[N];
VP a[N];
void sou(int x,int y,int id) {
	int i;
	for (i=0;i<a[x].size();i++) {
		PA A=a[x][i];
		if (A.fi==y) continue;
		dis[id][A.fi]=dis[id][x]+A.se;
		sou(A.fi,x,id);
	}
}
int Cnt;
void Dfs(int x,int y) {
	int i,j,k,u,r;
	zi[x].push_back(x);
	for (i=0;i<a[x].size();i++) {
		PA A=a[x][i];
		if (A.fi==y) continue;
		Dfs(A.fi,x);
		for (j=0;j<zi[A.fi].size();j++) zi[x].push_back(zi[A.fi][j]);
	}
	For(j,1,n) {
		
		if (dis[x][j]>m) continue;
		si[x]=1;
		f[x][j][j==x]=j==x&&!co[x];
		for (i=0;i<a[x].size();i++) {
			PA A=a[x][i];
			int w=A.fi;
			if (w==y) continue;
			int h[N];
			memset(h,120,sizeof(h));
			int t=0;
			for (u=si[x];u>=0;u--) //if (f[x][j][u]<N)
				for (r=si[w];r>=0;r--) {Cnt++;
					t++;
					short zhi=f[w][j][r];
					if (dis[j][w]>dis[j][x]) zhi=min(zhi,g[w][r]);
					if (zhi<N) h[u+r]=min(h[u+r],f[x][j][u]+zhi);
				}
//			printf("%d t %d\n",x,t);
			si[x]+=si[w];
			For(k,0,si[x]) f[x][j][k]=h[k];
		}
	}
	
	for (j=0;j<zi[x].size();j++) {
		int A=zi[x][j];
		For(i,0,si[x]) g[x][i]=min(g[x][i],f[x][A][i]);
	}
}
int main() {
//	freopen("e.in","r",stdin);
//	freopen("bc.out","w",stdout);
	scanf("%d%d",&n,&m);
	For(i,1,n) scanf("%d",&co[i]),s+=co[i];
	For(i,1,n-1) {
		int x,y,z;
		scanf("%d%d%d",&x,&y,&z);
		a[x].push_back(mk(y,z));
		a[y].push_back(mk(x,z));
	}
	For(i,1,n) sou(i,0,i);
	memset(f,120,sizeof(f));
	memset(g,120,sizeof(g));
	Dfs(1,0);
	short an=N;
	For(i,1,n)For(j,0,s) an=min(an,f[1][i][j]);
	printf("%d\n",an==N?-1:an);
//	printf("%d\n",Cnt);
	return 0;
}
