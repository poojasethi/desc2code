#include <iostream>
#include <vector>
#include <algorithm>
#include <iterator>

using namespace std;

int toInt(string &s, int d) {
  int r = 0;
  for (int i = 0; i < s.size(); i++) {
    r *= d;
    int p = s[i] - '0';
    if (p >= 10 && d > 10)
      p = s[i] - 'A' + 10;
    if (p >= d)
      return -1;
    r += p;
  }
  return r;
}

int main() {
  string s;
  cin >> s;
  int pos = s.find(':');
  string h = s.substr(0, pos);
  string m = s.substr(pos+1);
  vector<int> res;
  for (int i = 2; i <= 60; i++) {
    int ih = toInt(h, i);
    int im = toInt(m, i);
    if (ih >= 0 && ih < 24 && im >= 0 && im < 60)
      res.push_back(i);
  }
  if (res.size() == 0)
    cout << 0;
  else if (res.back() == 60)
    cout << -1;
  else
    copy(res.begin(), res.end(), ostream_iterator<int>(cout, " "));
  return 0;
}
