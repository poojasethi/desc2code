#include <bits/stdc++.h>

using namespace std;

char s[1005];
int h, w;

int v(char c) {
  return c >= 'A' ? c - 'A' + 10 : c - '0';
}

bool ok(int x, char *a, char *b) {
  int h = 0, m = 0;
  for (int i = 0; a[i]; i++) {
    h = h * x + v(a[i]);
  }
  for (int i = 0; b[i]; i++) {
    m = m * x + v(b[i]);
  }
  return h < 24 && m < 60;
}

int main() {
  scanf("%s", s);
  for (int i = 0; s[i]; i++) {
    if (s[i] == ':') h = i;
    else w = max(w, v(s[i]) + 1);
  }
  s[h] = 0;
  if (ok(60, s, s + h + 1)) {
    puts("-1");
    return 0;
  }
  for (int x = 59; x >= w; x--) {
    if (ok(x, s, s + h + 1)) {
      for (int i = w; i <= x; i++) printf("%d ", i);
      return 0;
    }
  }
  puts("0");
}
