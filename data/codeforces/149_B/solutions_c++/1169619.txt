#include <iostream>
#include <vector>
#include <sstream>
using namespace std;

int base;
int conv(char c) {
	if (c >= '0' && c <= '9') return c - '0';
	return c - 'A' + 10;
}
int conv(string s) {
	int res = 0;
	for (int i = 0; i < s.size(); ++i) {
		int t = conv(s[i]);
		if (t >= base) return 1 << 30;
		res = res * base + t;
		if (res > 100) return 1 << 30;
	}
	return res;
}
int main() {
	string s;
	cin >> s;
	for (int i = 0; i < s.size(); ++i) {
		if (s[i] == ':') s[i] = ' ';
	}
	stringstream ss(s);
	string a, b;
	ss >> a >> b;
	vector<int> v;
	for (base = 2; base < 1000; ++base) {
		if (conv(a) < 24 && conv(b) < 60) {
			v.push_back(base);
		}
	}
	if (v.size() > 100) {
		cout << -1;
		return 0;
	}
	for (int i = 0; i < v.size(); ++i) {
		cout << v[i] << ' ';
	}
	if (v.size() == 0) {
		cout << 0;
		return 0;
	}
}
