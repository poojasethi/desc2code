#include<cstdio>
#include<cstring>
char str[20];
int a[5],b[5];
int max(int a,int b){if(a>b)return a;return b;}
int main()
{
	int i,j,d1,d2,g,f,f1,f2,s,h,m;
	scanf("%s",str);
	for(f=1,d1=d2=i=0;str[i];i++)
	{
		if(str[i]>='0'&&str[i]<='9')
		{
			if(f)a[d1++]=str[i]-'0';
			else b[d2++]=str[i]-'0';
		}
		else if(str[i]==':')
			f=0;
		else
		{
			if(f)a[d1++]=str[i]-'A'+10;
			else b[d2++]=str[i]-'A'+10;
		}
	}
	for(f1=g=i=0;i<d1;i++)
	{
		s=max(s,a[i]);
		if(g)f1=0;
		if(g==0&&a[i])f1=g=1;
	}
	if(!g)f1=1;
	for(f2=g=i=0;i<d2;i++)
	{
		s=max(s,b[i]);
		if(g)f2=0;
		if(g==0&&b[i])f2=g=1;
	}
	if(!g)f2=1;
	s++;
	for(i=s;;i++)
	{
		h=m=0;
		for(j=0;j<d1;j++)
			h=h*i+a[j];
		for(j=0;j<d2;j++)
			m=m*i+b[j];
		if(h>=24||m>=60)
		{
			if(i==s)
				printf("0\n");
				return 0;
		}
		if(f1==1&&f2==1)
		{
			printf("-1\n");
			return 0;
		}
		printf("%d ",i);
	}
	return 0;
}
