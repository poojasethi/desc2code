
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
using namespace std;

int decode(string s, int r) {
  int n = 0;
  for (int i = 0; i < s.size(); i++) {
    char c = s[i];
    int t = (c <= '9' ? c - '0' : c - 'A' + 10);
    if (t >= r)
      return -1;
    n = n * r + t;
  }
  return n;
}


bool test(string h, string m, int r) {
  int nh = decode(h, r);
  int nm = decode(m, r);
  return (0 <= nh && nh < 24 && 0 <= nm && nm < 60);
}

int main(void) {
  string s;
  cin >> s;
  int idx1 = s.find(":");
  string h = s.substr(0, idx1);
  string m = s.substr(idx1+1);
  if (test(h, m, 64)) {
    cout << -1 << endl;
    return 0;
  }
  int ans = 0;
  for (int r = 2; r < 64; r++)
    if (test(h, m, r)) {
      if (ans != 0) {
        cout << " ";
      }
      cout << r;
      ans++;
    }
  if (ans == 0)
    cout << 0;
  cout << endl;
  return 0;
}
