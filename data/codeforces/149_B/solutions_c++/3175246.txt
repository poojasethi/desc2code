#include <cstdio>

inline int num(char c) {
    if (c <= '9') return c - '0';
    return c - 'A' + 10;
}

long long val(char *str, int dig) {
    long long sum = 0;
    for (int i = 0; str[i]; ++i)
	sum = sum * dig + num(str[i]);
    return sum;
}

int max_char(char *str) {
    int max = 0;
    for (int i = 0; str[i]; ++i) {
	int gg = num(str[i]);
	if (gg > max) max = gg;
    }
    return max;
}

bool chk(char *h, char *m, int i) {
    if (val(h, i) >= 24 || val(m, i) >= 60) return false;
    if (max_char(h) < i && max_char(m) < i)
	return true;
    return false;
}

int main() {
    char h[180], *m;
    scanf("%s", h);

    for (m = h; *m != ':'; ++m) ;
    *m = '\0';
    m += 1;

    if (chk(h, m, 62)) {
	puts("-1");
	return 0;
    }

    int cnt = 0;
    for (int i = 2; i < 62; ++i)
	if (chk(h, m, i))
	    printf("%d ", i), cnt += 1;

    if (cnt == 0) puts("0");

    return 0;
}
