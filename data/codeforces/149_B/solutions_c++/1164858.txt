#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char a[12];
int b[12];
int s,t,len;

bool check(int k){
  int i,t,n,m;
  n=m=0;
  t=1;
  for (i=s-1;i>=0;i--){
    n+=b[i]*t;
    t*=k;
  }
  t=1;
  for (i=len-1;i>s;i--){
    m+=b[i]*t;
    t*=k;
  }
  if (n<24 && m<60) return 1;
  return 0;
}

int main(){
  int i,j;
  bool flag=true;
  scanf("%s",a);
  len=strlen(a);
  for (j=i=0;i<len;i++)
    if (a[i]==':') s=i;
    else {
         if (a[i]>='A') b[i]=a[i]-'A'+10;
         else b[i]=a[i]-'0';
         if (b[i]>b[j]) j=i;
         }
  t=b[j]+1;
  if (check(60)){
    printf("-1\n");
    return 0;
  }
  while (t<=60){
    if (check(t)){
      if (!flag) printf(" ");
      printf("%d",t);
      flag=false;
    }
    t++;
  }
  if (flag) printf("0");
  printf("\n");
  return 0;
}
