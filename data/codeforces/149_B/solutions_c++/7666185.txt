int parser(char *str,int base)
{int ans=0,i=0;
while(str[i]!=0)
{ans*=base;
	if(str[i]<='9'&&str[i]>='0')
	{if(base<=str[i]-'0')
			return -1;
	ans+=str[i]-'0';}
		else
		{int pm=str[i]-'A'+10;
			if(base<=pm)
				return -1;
			ans+=pm;}
			i++;}
			return ans;}
