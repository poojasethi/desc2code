#include<bits/stdc++.h>
using namespace std;
int num[255];
vector<int> ans;
int checkbase(string s,int b)
{
    int i=0,m=1,value=0;
    while(i<s.size())
    {
        if(num[s[s.size()-1-i]]>=b)
            return 100000;
        value+=num[s[s.size()-1-i]]*m;
        i++;
        m*=b;
    }
    return value;
}
int main()
{
    string s;
    cin>>s;
    for(int i='0';i<='9';i++)
    num[i]=i-'0';
    for(int i='A';i<='Z';i++)
    num[i]=i-'A'+10;
    int pos=s.find(":");
    string h=s.substr(0,pos);
    string w=s.substr(pos+1);
    for(int i=2;i<60;i++)
    {
        if(checkbase(h,i)<24&&checkbase(w,i)<60)
            ans.push_back(i);
    }
    if(!ans.size())
        cout<<"0";
    else if(checkbase(h,70)<24&&checkbase(w,70)<60)
        cout<<"-1";
    else
    {
        for(int i=0;i<ans.size();i++)
            cout<<ans[i]<<" ";
    }
    return 0;
}
