#include<bits/stdc++.h>
using namespace std;

int i, tmp, m, ct, r[81];
string s, hr, mt;

int ctod(char c){
    return (c >= '0' && c <= '9') ? c - '0' : c - 'A' + 10;
}

int check(int base){
    int sum, i;
    sum = i = 0;
    while(i < hr.size()) {sum *= base; sum += ctod(hr[i++]);}
    if(sum >= 24) return 0;
    sum = i = 0;
    while(i < mt.size()) {sum *= base; sum += ctod(mt[i++]);}
    if(sum >= 60) return 0;
    return 1;
}

int main(){
    cin >> s;
    while(s[i] == '0') i++;
    tmp = i;
    while(s[i] != ':') {if(ctod(s[i]) > m) m = ctod(s[i]); i++;}
    hr = s.substr(tmp, i-tmp);
    i++;
    while(s[i] == '0') i++;
    tmp = i;
    while(s[i] != '\0') {if(ctod(s[i]) > m) m = ctod(s[i]); i++;}
    mt = s.substr(tmp, i-tmp);
    for(i = m+1; i <= 61; i++)
        if(check(i)) r[ct++] = i;
    if(ct == 0)
        cout << "0";
    else if(r[ct-1] == 61)
        cout << "-1";
    else for(i = 0; i < ct; i++)
        cout << r[i] << " ";
    return 0;
}
