# include <stdio.h>
# include <string.h>

int ans[80],w=0;

int max1=-243;

int q=0,q1=0,q2=0;

char c[20];

char a[15],b[15];

main()
{
	scanf("%s",&c);
	
	int s=strlen(c);
	
	for(int i=0;i<s;i++)
	{
		if(c[i]==':') {q=1;continue;}

		c[i]-=48;
	
		if(c[i] > 9) c[i]-=7;

		if(q==0) a[q1++]=c[i];
		else b[q2++]=c[i];		

		if(c[i] > max1) max1=c[i];
	}		
	
	q=0;
	
	int k=0,l=0;
	
	for(int i=max1+1;i<61;i++)
	{
		int kl=i;
		k=0,l=0;
		k+=a[q1-1];
		for(int j=q1-2;j>=0;j--)
		{k+=a[j]*kl;kl*=i;}
		
		kl=i;
		
		l+=b[q2-1];
		for(int j=q2-2;j>=0;j--)
		{l+=b[j]*kl;kl*=i;}
		
		if(k < 24 && l < 60)
		{
			ans[w++]=i;		
			q=1;
			if(i==60) q=2;
		}
		
		if(i==max1+1 && q==0) {printf("0");return 0;}
	}
	
	if(q==2) printf("-1");
	else{
		printf("%d",ans[0]);
		
		for(int i=1;i<w;i++)
		printf(" %d",ans[i]);
		
		printf("\n");
	}
}
