f = lambda b, s : sum(v * b ** (len(s) - 1 - i) for i, v in enumerate(s))
h, m = map(lambda x : [ord(v) - [ord('A') - 10, ord('0')][v.isdigit()] for v in x], raw_input().split(':'))
l = [b for b in range(max(h + m) + 1, 99) if f(b, h) < 24 and f(b, m) < 60]
print 0 if not len(l) else -1 if max(l) > 90 else ' '.join(map(str, l))
