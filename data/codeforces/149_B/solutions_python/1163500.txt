import sys
a,b = sys.stdin.readline().strip().split(":")
def conv(s,b):
  if not s: return 0
  if b <= 36:
    return int(s,b)
  #print s, (s[:-1],s[-1])
  return conv(s[:-1],b)*b+int(s[-1],36)
res = []
for k in range(2,61):
  try:
    c,d = conv(a,k), conv(b,k)
    if c <= 23 and d <= 59:
      res.append(k)
  except ValueError:
    pass

if not res:
  print 0
else:
  if res[-1] == 60:
    print -1
  else:
    print " ".join(map(str,res))
