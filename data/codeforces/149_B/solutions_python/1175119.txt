def value(n, b):
	a = 1
	r = 0
	for i in range(-1, -len(n) - 1, -1):
		r += int(n[i], 36) * a
		a *= b
	return r

h, m = [i for i in raw_input().split(":")]
if int(h, 36) < 24 and int(m, 36) < 36:
	print -1
else:
	a = []
	tmp = list(h)
	tmp.extend(list(m))
	mb = max((int(x, 36)+1) for x in tmp)
	for i in range(mb, 60):
		if value(h, i) < 24 and value(m, i) < 60: a.append(i)
	if (len(a) == 0):
		print 0
	else:
		print ' '.join(map(str, a))
