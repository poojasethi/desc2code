def conv(n, b):
    if not n:
        return 0
    if int(n[-1], 36) >= b: return 100
    return int(n[-1], 36) + conv(n[:-1], b) * b

def good(n):
    return conv(h, n) < 24 and conv(m, n) < 60

h, m = raw_input().split(':')
ans = filter(good, range(100))
if ans == []:
    print 0
elif ans[-1] == 99:
    print -1
else:
    print ' '.join(map(str, ans))
