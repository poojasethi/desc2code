#include<cstdio>
#include<cstring>

char s[3000];

int m,N,f[3000],Mod(1000003);

int main()
{	int i,j;
//	freopen("115D.in","rb",stdin);
//	freopen("115D.out","wb",stdout);
	scanf("%s",s);
	N=strlen(s);
	for(i=0;i<N;i++)
	if((s[i]=='*'||s[i]=='/')&&(!i||s[i-1]<'0'||s[i-1]>'9'))
	{	printf("0\n");
		return 0;
	}
	if(s[N-1]<'0'|s[N-1]>'9')
	{	printf("0\n");
		return 0;
	}
	f[0]=1;
	for(i=0;i<N;i++)
	if('0'>s[i]||s[i]>'9')
	{	m++;
		for(j=m;j>0;j--)f[j]=f[j-1];
		f[0]=0;
	}
	else if(i&&('0'>s[i-1]||s[i-1]>'9'))
		for(j=m;j>=0;j--)f[j]=(f[j]+f[j+1])%Mod;
	printf("%d\n",f[0]);
	return 0;
}
