#include <cstdio>
#include <cstring>

int N,f[2005];
char s[2005];

int main()
{
	scanf("%s",s+1),N=strlen(s+1),f[0]=1;
	if (s[1]=='*'||s[1]=='/') goto bg;
	for (int i=1; i<=N; i++) if (s[i]<48)
	{
		if (s[i+1]=='*'||s[i+1]=='/') goto bg;
		for (int j=N; j>=0; j--) f[j+1]=f[j];
		f[0]=0;
	}
	else if (s[i+1]<48) for (int j=N,t=0; j>=0; j--) (t+=f[j])%=1000003,f[j]=t;
	printf("%d\n",f[0]); return 0;
	bg:puts("0");
}
