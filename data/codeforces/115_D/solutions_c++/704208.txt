#include <cstdio>
#include <cstring>

int len, nag[2048], dp[2048];
char s[2048];

bool check(int& id, int x) {
    while (id < len && !(s[id] >= '0' && s[id] <= '9')) {
        if (s[id] != '-' && s[id] != '+') return false;
        ++nag[x]; ++id;
    }
    if (id >= len) return false;
    while (id < len && (s[id] >= '0' && s[id] <= '9')) {
        ++id;
    }
    return true;
}

int main() {
    scanf ("%s", s);
    len = strlen(s);
    int id = 0, cnt = 0;
    while (id < len) {
        if (id != 0) ++id; 
        if (!check(id, cnt++)) {
            puts ("0");
            return 0;
        }
    }
    for (int i = 0; i < len; ++i) {
        dp[i] = 1;
    }
    int off = 0;
    for (int i = cnt - 1; i > 0; --i) {
        off += nag[i] + 1;
        for (int j = off + 1; j < len; ++j) {
            dp[j] = (dp[j] + dp[j - 1]) % 1000003;
        }
    }
    printf ("%d\n", dp[off + nag[0]]);
    
    return 0;
}
