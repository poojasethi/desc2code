//Light My Path

#include<iostream>
#include<algorithm>
using namespace std;

const int Mn=5010;
int mark[Mn],set[Mn][2],ans,n,c[Mn],mark2[Mn];

int op(int a,int b){
	if(set[a][0]<set[b][0]){
		return 1;
	}
	else{
		return 0;
	}
}

int main(){
	int x,y,k=0;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>set[i][0]>>set[i][1];
	}
	for(int i=0;i<n;i++){
		c[i]=i;
	}
	sort(c,c+n,op);
	for(int i=0;i<n;i++){
		mark[c[i]]=1;
		for(int j=0;j<n-1;j++){
			k=j+1;
			if(mark[c[j+1]]==1 && j+1<n-1){
				k++;
			}
			if(set[c[k]][0]<set[c[j]][1] && mark[c[k]]!=1 && mark[c[j]]!=1){
				mark[c[i]]=0;
				break;
			}
		}
		if(mark[c[i]]==1){
			ans++;
			mark2[c[i]]=1;
			mark[c[i]]=0;
		}
	}
	cout<<ans<<endl;
	if(ans!=0){
		for(int i=0;i<n;i++){
			if(mark2[i]!=0){
				cout<<i+1<<" ";
			}
		}
		cout<<endl;
	}
	return 0;
}
