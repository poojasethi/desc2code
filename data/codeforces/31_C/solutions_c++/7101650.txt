#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int l[n],r[n],s=0,sum[n];
	vector<int> ans;
	for(int i=0;i<n;i++)cin>>l[i]>>r[i],sum[i]=0;
	for(int i=0;i<n;i++)for(int j=0;j<i;j++)
	{
		if(l[i]<r[j]&&l[j]<r[i])sum[i]++,sum[j]++,s++;
	}
	for(int i=0;i<n;i++)if(sum[i]==s)ans.push_back(i+1);
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)cout<<ans[i]<<" ";
	return 0;
}
