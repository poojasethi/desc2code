#include<iostream>
#include<cstdlib>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<string>
#define LiangJiaJun main
using namespace std;
int ans=0,res[50004],n;
struct data{int L,R,sit;}a[5004];
int sum=0;
inline bool dex(data a,data b){return a.L<b.L;}
int LiangJiaJun(){
    memset(res,0,sizeof(res));
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d%d",&a[i].L,&a[i].R),a[i].sit=i;
    sort(a+1,a+n+1,dex);
    for(int i=1;i<=n;i++){
        int k=i,t=0;bool r=1;
        for(int j=1;j<=n;j++){
            if(j==k)continue;
            if(t>a[j].L){r=0;break;}
            else t=a[j].R;
        }
        if(r)res[++ans]=a[k].sit;
    }
    sort(res+1,res+ans+1);
    printf("%d\n",ans);
    for(int i=1;i<=ans;i++)printf("%d ",res[i]);
    return 0;
}
