#include <bits/stdc++.h>
using namespace std;
vector<pair<pair<int,int>,int> > vec;
std::vector<int> ans;
int main()
{
	#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
	#endif
	std::ios_base::sync_with_stdio(false);
	int i,j,k,l;
	cin>>i;
	for(j=0;j<i;j++)
	{
		cin>>k>>l;
		vec.push_back(make_pair(make_pair(l,k),j));
	}
	sort(vec.begin(), vec.end());
	for(j=0;j<i;j++)
	{
	//	cout<<j<<" "<<vec[j].first<<" "<<vec[j].second<<endl;
		l=-1;
		for(k=0;k<i;k++)
		{
			if(k==j)
				continue;
			if(vec[k].first.second<l)
				break;
			l=vec[k].first.first;
		}
		if(k==i)
			ans.push_back(vec[j].second);
	}
	sort(ans.begin(),ans.end());
	cout<<ans.size()<<endl;
	for(j=0;j<ans.size();j++)
		cout<<ans[j]+1<<" ";
	cout<<endl;
	return 0;
}
