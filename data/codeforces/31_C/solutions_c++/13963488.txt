#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main(){
	int n,a,b;cin>>n;
	vector<pair<pair<int,int>,int> >v;
	for(int i=0;i<n;++i){
		cin>>a>>b;
		v.push_back(make_pair(make_pair(a,b),i+1));
	}
	sort(v.begin(),v.end());
	vector<int> ret;
	for(int i=0;i<v.size();++i){
		int prev=-1;
		bool ok= true;
		for(int j=0;j<v.size();++j){
			if(i!=j){
				if(prev==-1){ prev= v[j].first.second;continue;}
				if(prev>v[j].first.first){ok = false;	break;}
				prev= v[j].first.second;
			}
		}
		if(ok)ret.push_back(v[i].second);
	}
	cout<<ret.size()<<endl;
	sort(ret.begin(),ret.end());
	for(int i=0;i<ret.size();++i)
		cout<<ret[i]<<" ";
	cout<<endl;
}
