#include <cstdio>
#define NN 5010
int n, k = 0, t = 0;
int s[NN], f[NN], x[NN];

int main(void) {
	int i, j;
	for(i = 0; i < NN; i++) x[i] = 0;
	scanf("%d",&n);
	for(i = 0; i < n; i++)
		scanf("%d %d",&s[i],&f[i]);
	for(i = 0; i < n; i++)
		for(j = i+1; j < n; j++)
			if(f[i] > s[j] && f[j] > s[i]) {
				x[i]++;
				x[j]++;
				k++;
			}
	for(i = 0; i < n; i++)
		if(x[i] == k) t++;
	printf("%d\n",t);
	for(i = 0; x[i] != k && t; i++);
	if(t) { printf("%d",i+1); t--; }
	for(i++;t;i++)
		if(x[i] == k) {
			printf(" %d",i+1);
			t--;
		}
	if(t) printf("\n");
	return 0;
}
