#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>
using namespace std;

int main()
{
	int n;
	cin >> n;
	vector <pair <int, pair <int, int> > > v(n);
	for (int i = 0; i < n; ++i)
	{
		cin >> v[i].first >> v[i].second.first;
		v[i].second.second = i;
	}
	sort(v.begin(), v.end());
	
	vector <int> ans;
	for (int i = 0; i < n; ++i)
	{
		bool good = true;
		int last = 0;
		for (int j = 0; j < n; ++j)
			if (j != i)
			{
				if (v[j].first < last)
				{
					good = false;
					break;
				}
				last = v[j].second.first;
			}
		if (good)
			ans.push_back(v[i].second.second + 1);
	}
	
	sort(ans.begin(), ans.end());
	cout << (int)ans.size() << '\n';	
	for (size_t i = 0; i < ans.size(); ++i)
		cout << ans[i] << ' ';
	cout << '\n';
	
	return 0;
}
