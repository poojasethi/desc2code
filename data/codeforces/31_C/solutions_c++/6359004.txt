#include<bits/stdc++.h>
using namespace std;
struct node{
    int l,r,idx;
};
bool cmp(node p1,node p2){
    return p1.l<p2.l;
}
vector<node> v;
int main(){
    node nt;
    int n,x,y,i,j;
    cin>>n;
    x=0;
    while(x<n){
        cin>>nt.l>>nt.r;
        x++;
        nt.idx=x;
        v.push_back(nt);
    }
    vector<int> sol;
    sort(v.begin(),v.end(),cmp);
    for(i=0;i<n;i++){

        bool flag=true;
        int last=0;
        for(j=0;j<n;j++){
            if(j==i)
            continue;
            if(v[j].l>=last) last=v[j].r;
            else
                flag=false;

        }
    if(flag)
        sol.push_back(v[i].idx);
    }
    cout<<sol.size()<<endl;
    sort(sol.begin(),sol.end());
    for(i=0;i<sol.size();i++)
        cout<<sol[i]<<" ";
}
