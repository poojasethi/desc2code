#include <bits/stdc++.h>
using namespace std;

const int maxN = 1e5+100;
const int base = 1e9+7;

struct a_ {
	int l,r,cs;
} a[maxN];
int n;
vector<int> ans;

bool cmp(a_ x, a_ y) {
	if (x.l != y.l) return x.l < y.l;
	return x.r<y.r;
}

int main() {
	cin >> n;
	for(int i=0; i<n; i++) scanf("%i%i", &a[i].l, &a[i].r);
	for(int i=0; i<n; i++) a[i].cs = i;
	sort(a,a+n,cmp);
	for(int i=0; i<n; i++) {
		a_ tam = a[i];
		for(int j=i; j<n-1; j++) 
			a[j] = a[j+1];
		ans.push_back(tam.cs);
		for(int j=1; j<n-1; j++) 
			if (a[j].l < a[j-1].r) {
				ans.erase(ans.end()-1);
				break;
			}
		for(int j=n-1; j>i; j--) 
			a[j] = a[j-1];
		a[i] = tam;
	}
	sort(ans.begin(), ans.end());
	cout << ans.size() << endl;
	for(int i=0; i<ans.size(); i++) 
		printf("%i ", ans[i]+1);
	return 0;
}
