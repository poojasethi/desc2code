#include<iostream>
#include<vector>

using namespace std;

int main () {
    int graph[5000];
    int l[5000], r[5000];
    int n;   
    cin >> n;
    for (int i=0; i<n; i++)
        cin >> l[i] >> r[i];
    
    int total = 0;
    for (int i=0; i<n; i++)
        for (int j=i+1; j<n; j++)
            if (l[i] < r[j] && r[i] > l[j]) {
               graph[i] += 1;
               graph[j] += 1;
               total += 2;
            }
    vector<int> ans;
    for (int i=0; i<n; i++) {
        if (2*graph[i] == total)
            ans.push_back(i);   
    }   
    cout << ans.size() << endl;
    for (int i=0; i<ans.size(); i++)
        cout << ans[i]+1 << " ";    
    return 0;
}
