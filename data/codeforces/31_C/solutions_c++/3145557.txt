#include <iostream>

using namespace std;


int main()
{
    int n;
int st[5001],end[5001];
int a[5001]={0};
int f=0,t=0;

    cin>>n;
    for (int i=1;i<=n;i++)
    {
        cin>>st[i]>>end[i];
    }
    for (int i=1;i<=n;i++)
    {
        for (int j=i+1;j<=n;j++)
        {
            if (st[i]<end[j]&&st[j]<end[i])
            {
                a[i]++;
                a[j]++;
                f++;
            }
        }
    }
    for (int i=1;i<=n;i++)
    {
        if (a[i]==f)
        {
            t++;
        }
    }
    cout<<t<<endl;
    for (int i=1;i<=n;i++)
    {
        if (a[i]==f)
        {
            cout<<i<<" ";
        }
    }
    cout<<endl;
    return 0;
}
