#include<iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 5050
using namespace std;
struct node
{
    int r,l;
} z[maxn];
int c[maxn]={0};
int s[maxn];
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
        scanf("%d%d",&z[i].l,&z[i].r);
    int cnt=0;
    for(int i=1;i<=n;i++)
        for(int j=i+1;j<=n;j++)
            if (!(z[i].r<=z[j].l||z[i].l>=z[j].r))
            cnt++,c[i]++,c[j]++;
    int m=0;
    for(int i=1;i<=n;i++)
    if (c[i]==cnt)
        s[m++]=i;
    cout<<m<<endl;
    for(int i=0;i<m;i++)
    printf("%d%c",s[i],i==m-1?'\n':' ');
    return 0;
}
