#include<iostream>
#include<vector>
#include<set>
using namespace std;

multiset<pair<int, int> > s;
int N;
pair<int, int> arr[5000];

bool ok(){
	if(s.size() < 2)return true;
	multiset<pair<int, int> >::iterator itr, last;
	itr = last = s.begin();
	itr++;
	while(itr != s.end()){
		if((*itr).first < (*last).second)
			return false;
		last = itr;
		itr++;
	}
	return true;
}

int main(){

	//freopen("1.in", "rt", stdin);
	cin >> N;
	for(int i = 0 ; i < N ; i++){
		cin >> arr[i].first >> arr[i].second;
		s.insert(arr[i]);
	}

	vector<int> res;
	for(int i = 0 ; i < N ; i++){
		s.erase(s.find(arr[i]));
		if(ok())
			res.push_back(i+1);
		s.insert(arr[i]);
	}
	cout << res.size() << endl;
	for(int i = 0 ; i < (int)res.size() ; i++){
		if(i)cout << " ";
		cout << res[i];
	}
	return 0;
}
