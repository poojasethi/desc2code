#include<iostream>
#include<string>
#include<cstring>
using namespace std;

int s[5010], e[5010];
int d[5010];
int main( )
{
	int n;
	while (cin >> n)
	{
		memset(d, 0, sizeof(d));
		int inter = 0;
		for (int i = 0; i < n; i++)
		{
			cin >> s[i] >> e[i];
			for (int j = 0; j < i; j++)
			{
				if (s[j] < e[i] && e[j]>s[i] || s[i] < e[j] && e[i]>s[j])
				{
					inter++;
					d[i]++;
					d[j]++;
				}
			}
		}
		int ans = 0;
		for (int i = 0; i < n; i++)
		{
			if (d[i] == inter)
				ans++;
		}
		cout << ans << endl;
		for (int i = 0; i < n; i++)
		{
			if (d[i] == inter)
				cout << i+1 << " ";
		}
		if (ans)
		cout << endl;
	}
	return 0;
}
