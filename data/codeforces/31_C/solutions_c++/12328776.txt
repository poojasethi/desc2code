//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

int n,l[5050],r[5050],g[5050],ng,mg,k;

int main()
{
  cin>>n;
  for(int i=1;i<=n;i++)
    cin>>l[i]>>r[i];
  for(int i=1;i<=n;i++)
    for(int j=i+1;j<=n;j++)
      if(l[j]<r[i] && r[j]>l[i])
	g[i]++,g[j]++;
  for(int i=1;i<=n;i++)
    {
      if(g[i]>1)
	k++;
      if(g[i]>mg)
	ng=1,mg=g[i];
      else if(g[i]==mg)
	ng++;
    }
  if((mg==0 || ng<=2) && k<2)
    {
      cout<<ng<<endl;
      for(int i=1;i<=n;i++)
	if(g[i]==mg)
	  cout<<i<<" ";
      cout<<endl;
    }
  else
    cout<<0<<endl;
  return 0;
}
