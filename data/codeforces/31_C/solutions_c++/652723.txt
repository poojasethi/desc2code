#include <stdio.h>
#include <algorithm>
using namespace std;

int L[5005], R[5005];
int srt[5005];

bool cmp(int x, int y) {
	return L[x]<L[y];
}

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d%d", &L[i], &R[i]);
		srt[i] = i;
	}
	sort(srt, srt+n, cmp);
	int ok[5005], okn=0;
	for(int i=0;i<n;i++) {
		int last = -1, flag=1;
		for(int j=0;j<n;j++) {
			if(srt[j]==i) continue;
			if(L[srt[j]]<last) {flag=0;break;}
			last=R[srt[j]];
		}
		if(flag) ok[okn++] = i+1;
	}
	printf("%d\n", okn);
	for(int i=0;i<okn;i++)
		printf("%d ", ok[i]);
	puts("");
	return 0;
}
