#include <cstdio>  
#include <cstring>  
#include <algorithm>  
#define maxn 5005  
using namespace std;  
struct node  
{  
    int r,l;  
} a[maxn];  
int main()  
{  
    int n,c[maxn]= {0},s[maxn];  
    scanf("%d",&n);  
    for ( int i=1; i<=n; i++ )  
        scanf("%d%d",&a[i].l,&a[i].r);  
    int cnt=0;  
    for ( int i=1; i<=n; i++ )   
        for ( int j=i+1; j<=n; j++ )   
            if ( !(a[i].r<=a[j].l||a[i].l>=a[j].r) ) cnt++,c[i]++,c[j]++;//求出相交的区间数  
    int m=0;  
    for ( int i=1; i<=n; i++ ) if ( c[i]==cnt ) s[m++]=i;//如果去掉那个区间之后无区间相交，说明那个区间参与了所有次数的相交  
    printf("%d\n",m);  
    for ( int i=0; i<m; i++ ) printf("%d%c",s[i],i==m-1?'\n':' ');  
    return 0;  
}
