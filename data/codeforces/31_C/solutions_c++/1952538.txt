#include<stdio.h>
#include<algorithm>
using namespace std;

int ans[5005];

struct pnt
{
	int l,r,id;
}p[5005];

int cmp(pnt a,pnt b)
{
	if(a.l==b.l)
		return a.r<b.r;
	return a.l<b.l;
}

int main()
{
	int n,i,j,c,f,max;
	while(scanf("%d",&n)!=EOF)
	{
		for(i=0;i<n;i++)
		{
			scanf("%d%d",&p[i].l,&p[i].r);
			p[i].id=i+1;
		}
		sort(p,p+n,cmp);
		c=0;
		for(i=0;i<n;i++)
		{
			f=1;
			max=1;
			for(j=0;j<n;j++)
				if(i!=j)
				{
					if(p[j].l>=max)
					{
						max=p[j].r;
					}
					else
					{
						f=0;
						break;
					}
				}
			if(f)
			{
				ans[c]=p[i].id;
				c++;
			}
		}
		printf("%d\n",c);
		if(c)
		{
			sort(ans,ans+c);
			printf("%d",ans[0]);
			for(i=1;i<c;i++)
				printf(" %d",ans[i]);
			printf("\n");
		}
	}
	return 0;
}
