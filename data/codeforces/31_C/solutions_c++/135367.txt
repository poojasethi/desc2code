#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct S {
	int id,beg,end;
	
	bool operator < (const S& q) const {
		if(beg!=q.beg) return (beg<q.beg);
		if(end!=q.end) return (end<q.end);
		return (id<q.id);
	}
} a[5005];

int main() {
	
	int n,i,j;
	
	cin>>n;
	for(i=0;i<n;i++) {
		a[i].id=i+1;
		cin>>a[i].beg>>a[i].end;
	}
	
	sort(a,a+n);
	
	vector <int> res;
	for(i=0;i<n;i++) {
		int prev=0;
		bool b=true;
		for(j=0;j<n;j++)if(j!=i) {
			if(a[j].beg<prev) {
				b=false;
				break;
				
			}
			prev=a[j].end;
		}
		if(b) res.push_back(a[i].id);
	}
	
	sort(res.begin(),res.end());
	
	cout<<res.size()<<endl;
	for( typeof(res.begin()) itr=res.begin(); itr!=res.end(); itr++) {
		cout<<*itr<<" ";
	}
	
	return 0;
}
