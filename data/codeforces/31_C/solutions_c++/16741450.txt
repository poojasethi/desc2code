#include <bits/stdc++.h>
#define pb push_back
#define MAX 5100
using namespace std;
pair <int , pair <int , int> > a[MAX];
vector <int> ans;
vector < pair < int ,int > > v;
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
    {
		cin>>a[i].first>>a[i].second.first;
		a[i].second.second=i;
	}
	sort(a+1,a+n+1);
	for(int i=1;i<=n;i++)
	{
		v.clear();
		for(int j=1;j<=n;j++)
			if(i!=j) v.pb(make_pair(a[j].first,a[j].second.first));
		bool check=false;
		int mx=v[0].second;
		for(int j=1;j<v.size();j++)
		{
			if(v[j].first<mx) check=true;
			mx=max(mx,v[j].second);
		}
		if(!check) ans.pb(a[i].second.second);
	}
	sort(ans.begin(),ans.end());
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)
        cout<<ans[i]<<" ";
    return 0;
}
