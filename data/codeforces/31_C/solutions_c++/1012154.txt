#include<iostream>
#include<algorithm>
using namespace std;
pair<pair<int,int>,int> seg[5010];
#define fi first.first
#define se first.second
#define id second

int stk[5010];
int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>seg[i].fi>>seg[i].se;
		seg[i].id=i+1;
	}
	seg[n].fi=-1;
	seg[n].se=-1;
	n++;
	sort(seg,seg+n);
	int cn=0;
	for(int i=1;i<n;i++) {
		int p=0;
		bool f=true;
		for(int j=1;j<n;j++) {
			if(i!=j) {
				if(seg[j].fi<seg[p].se) {
					f=false;
					break;
				}
				p=j;
			}
		}
		if(f) {
			stk[cn++]=seg[i].id;
		}
	}
	sort(stk,stk+cn);
	cout<<cn<<'\n';
	for(int i=0;i<cn;i++) {
		cout<<stk[i]<<' ';
	}
}
