#include<stdio.h>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;
struct node{
	int x,y,num;
};
node a[100010];
bool cmp(node x,node y)
{
	return x.x<y.x;
}
int main()
{
	int n,i,j,i1,j1,i2,j2,m,t,k,ans,b[10000],c[10000];
	bool tt;
	scanf("%d",&n); ans=0; k=1; 
	for (i=1;i<=n;i++) 
	{
		scanf("%d%d",&a[i].x,&a[i].y);
		a[i].num=i;
	}
	sort(a+1,a+n+1,cmp);
	a[0].x=0; a[0].y=0;
	for (i=1;i<=n;i++)
	{
		tt=true;
		for (j=1;j<=n-1;j++)
		{
			if (i==j) if (a[j-1].y>a[j+1].x) tt=false;
			if (i-1!=j && i!=j) if (a[j].y>a[j+1].x) tt=false;
		}
		if (tt)
		{
			ans++; b[ans]=i;
		}
	}
	printf("%d\n",ans);
	for (i=1;i<=ans;i++) c[i]=a[b[i]].num;
	sort(c+1,c+ans+1);
	for (i=1;i<=ans-1;i++) printf("%d ",c[i]);
	if (ans>0) printf("%d\n",c[ans]);
}
