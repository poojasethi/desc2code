#include<bits/stdc++.h>
using namespace std;

#define LL long long
#define mod 1000000007

int l[5000],r[5000],sum[5000];

int main()
{
    int n,j,i,s = 0;

    cin>>n;
    for(i=0;i<n;i++)
        cin>>l[i]>>r[i];

    for(i=1;i<n;i++)
        for(j=0;j<i;j++)
            if(l[i]<r[j] && l[j]<r[i])
            {
                sum[i]++;
                sum[j]++;
                s++;
            }
    vector<int> ans;
    for(i=0;i<n;i++)
        if(sum[i]==s)
            ans.push_back(i+1);

    cout<<ans.size()<<endl;
    for(i=0;i<ans.size();i++)
        cout<<ans[i]<<' ';
    return 0;
}
