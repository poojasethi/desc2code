#include <stdlib.h>
#include <vector>
#include <stdio.h>
#include <utility>
#include <algorithm>


#define x first
#define y second

using namespace std;

const int N  = 5500;
pair<  pair<int,int> ,int> arr[N];
int n; 


int main()
{
	scanf("%d", &n);
	
	for(int i = 0; i < n; ++i)
	{
		scanf("%d %d", &arr[i].x.x, &arr[i].x.y);
		arr[i].y = i + 1;
	}
	
	sort(arr, arr + n);
	
	vector<int> ans;
	
	for(int i = 0 ; i < n; ++i)
	{
		int j, now = 0;
		for( j = 0; j < n; ++j)
		{
			if( i == j) continue;
			if(arr[j].x.x < now) break;
			now = arr[j].x.y;	
		}
		if( j == n)
			ans.push_back(arr[i].y);
	}
	
	printf("%d\n", (int)ans.size());
	if(ans.size())
	{
		sort(ans.begin(), ans.end());
		for(int i = 0; i < (int)ans.size(); ++i)
			printf("%d ", ans[i]);
		
	}
	
	return 0;
}
