#include <bits/stdc++.h>

using namespace std;
const int maxn=5007;
int n, m;
int L[maxn], R[maxn], SUM[maxn];
vector<int> kq;
int main(){
    #ifndef ONLINE_JUDGE
    freopen("schedule.i5","r",stdin);
    freopen("schedule.o5","w",stdout);
    #endif // ONLINE_JUDGE
    scanf("%d",&n);
    for (int i = 1; i <= n; i++) scanf("%d%d",&L[i],&R[i]), SUM[i]=0;
    int cnt = 0;
    for (int i = 1; i <= n; i++)
        for(int j = 1 ; j < i; j++){
            if (L[i] < R[j] && L[j] < R[i]) SUM[i]++, SUM[j]++, cnt++;
        }
    for(int i = 1; i <= n ; i++)
        if(SUM[i] == cnt) kq.push_back(i);
    sort(kq.begin(), kq.end());
    printf("%d\n",kq.size());
    for (int i = 0; i < kq.size(); i++) printf("%d ",kq[i]);
    return 0;
}
