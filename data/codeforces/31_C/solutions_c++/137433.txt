#include<iostream>
#include<algorithm>
using namespace std;

const int maxn = 5050;
int a[maxn], l[maxn], r[maxn], n, num, ans;

int main(){
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> l[i] >> r[i];
	for(int i = 0; i< n; i++)
		for(int j = i+1; j < n; j++)
			if(l[i] < r[j] && r[i] > l[j]){
				a[i]++;
				a[j]++;
				//cerr << i << " --> " << j << endl;
				num++;
			}
	cerr << num << endl;
	for(int i = 0; i < n; i++)
		if(a[i] == num)
			ans++;
	cout << ans << endl;
	for(int i = 0; i < n; i++)
		if(a[i] == num)
			cout << i+1 << " ";
	cout << endl;
	return 0;
}
