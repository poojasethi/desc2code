#include<cstdio>
int n,k,i,j,g[5001][2],c[5001],o,s,a[5001];
int main()

{
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		c[i]=0;
		scanf("%d%d",&g[i][0],&g[i][1]);
	}
	for(i=0,o=0;i<n-1;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(((g[j][0]>g[i][0])&&(g[j][0]<g[i][1])||(g[j][1]>g[i][0])&&(g[j][1]<g[i][1]))||((g[i][0]>g[j][0])&&(g[i][0]<g[j][1])||(g[i][1]>g[j][0])&&(g[i][1]<g[j][1])))
			{
				c[i]++;c[j]++;
				o++;
			}
		}
	}
	for(i=0,s=0;i<n;i++)
	{
		if(c[i]==o)
		{
			s++;
			a[s]=i+1;
		}
	}
	printf("%d\n",s);
	for(i=1;i<=s;i++)
	{
		printf("%d ",a[i]);
	}
	return 0;
}
