#include <bits/stdc++.h>
using namespace std;
int ar[1010];
int main(void){
  ios_base::sync_with_stdio(false); cin.tie(0);
  int n;
  cin >> n;
  vector < pair < pair <int, int >, int > > v; // l, r, i
  v.push_back(make_pair(make_pair(0,0), 0));
  for(int i=1; i<=n; i++) {
    int l, r;
    cin >> l >> r;
    v.push_back(make_pair(make_pair(l, r),i));
  }
  sort(v.begin(), v.end());
  vector<int>ans;
  for(int i=1; i<=n; i++) {
    bool ok=true;
    int pos=0;
    for(int j=1; j<=n; j++) {
      if(i!=j) {
         if(v[j].first.first<pos) ok=false;
        pos=v[j].first.second;
      }
    }
    if(ok) ans.push_back(v[i].second);
  }
  sort(ans.begin(),ans.end());
  cout << ans.size() << "\n";
  for(size_t i=0; i<ans.size(); i++) {
    cout << ans[i] << " ";
  }
  return 0;
}
