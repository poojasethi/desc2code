#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 5005
using namespace std;
struct node
{
    int r,l;
} a[maxn];
int main()
{
    int n,c[maxn]= {0},s[maxn];
    scanf("%d",&n);
    for ( int i=1; i<=n; i++ )
        scanf("%d%d",&a[i].l,&a[i].r),a[i].r--;
     
    int cnt=0;
    for ( int i=1; i<=n; i++ ) 
        for ( int j=i+1; j<=n; j++ ) 
            if ( !(a[i].r<a[j].l||a[i].l>a[j].r) ) cnt++,c[i]++,c[j]++;
    int m=0;
    for ( int i=1; i<=n; i++ ) if ( c[i]==cnt ) s[m++]=i;
    printf("%d\n",m);
    for ( int i=0; i<m; i++ ) printf("%d%c",s[i],i==m-1?'\n':' ');
    return 0;
}
