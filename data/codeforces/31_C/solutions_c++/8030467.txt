#include <iostream>
#include <vector>
#define Maxn 5000+10
using namespace std;
int l[Maxn],r[Maxn],Sum[Maxn],Cnt;
int main(){
    ios_base::sync_with_stdio(0);
    vector<int>Ans;
    int n;
    cin >> n;
    for(int i=0;i<n;i++)
        cin >> l[i] >> r[i];

    for(int i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            if(l[i]<r[j] && l[j]<r[i]){
                Cnt++;
                Sum[i]++;
                Sum[j]++;
            }
        }
    }

    for(int i=0;i<n;i++){
        if(Sum[i]==Cnt)
            Ans.push_back(i);
    }
    cout << Ans.size() << endl;
    for(int i=0;i<Ans.size();i++)
        cout << Ans[i]+1 << " ";

    //system("pause");
}
