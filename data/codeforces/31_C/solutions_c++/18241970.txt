#include<iostream>
#include<algorithm>
using namespace std;
struct sajal{long long r,l,i;};
int mysort(sajal x,sajal y){
if(x.r<y.r)return 1;
return 0;
}
sajal a[100003];long b[5003];
int main(){
int n;
cin>>n;
for(int i=0;i<n;i++){
    cin>>a[i].l>>a[i].r;a[i].i=i;
}
sort(a,a+n,mysort);long ans=0;
for(long i=0;i<n;i++){long h=0;int e=0;
    for(long j=0;j<n&&e==0;j++){
        if(j!=i){
            long p=a[j].l;if(p<h)e=1;else{h=a[j].r;}
        }}
        if(e==0){
            ans++;b[ans]=a[i].i+1;
        }

}
cout<<ans<<endl;
sort(b+1,b+ans+1);
for(long i=1;i<=ans;i++)cout<<b[i]<<" ";
cout<<endl;










}
