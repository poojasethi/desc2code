#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int mm=5050;
class node
{
  public:int l,r,id;
}f[mm];
int s[mm],pos;
int m;
int main()
{
  while(cin>>m)
  { memset(s,0,sizeof(s));
    int num=0;
    for(int i=0;i<m;i++)
      {cin>>f[i].l>>f[i].r;f[i].id=i+1;}
    for(int i=0;i<m;i++)
      for(int j=i+1;j<m;j++)
      if(f[i].l<f[j].r&&f[i].r>f[j].l)
    {
      s[i]++;s[j]++;num++;
    }
    int ans=0;
    for(int i=0;i<m;i++)
      if(s[i]==num)ans++;
      cout<<ans<<"\n";
      for(int i=0;i<m;i++)
        if(s[i]==num)
        cout<<i+1<<" ";
      cout<<"\n";
  }
}
