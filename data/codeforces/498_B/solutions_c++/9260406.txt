#include <stdio.h>
#include <math.h>

double p[5005];
int t[5005];

double dp[5005][5005];
double mpow(double d,int n)
{
	double ret=1;
	while(n){
		if(n&1)
			ret*=d;
		d*=d;
		n>>=1;
	}
	return ret;
}

int main()
{
	int n,T;
	double ans=0;
	scanf("%d %d",&n,&T);
	for(int i=1; i<=n; i++){
		scanf("%lf %d",&p[i],&t[i]);
		p[i]/=100;
	}
	for(int i=0; i<=T; i++)
		dp[0][i]=0;
	dp[0][0]=1;
	for(int i=1; i<=n; i++){
		double tmp=dp[i-1][0];
		double prpow=1;
		prpow=mpow(1-p[i],t[i]);
		dp[i][0]=0;
		for(int j=1; j<=T; j++){
			dp[i][j]=tmp*p[i];
			tmp=tmp*(1-p[i])+dp[i-1][j];
			if(j>=t[i]){
				tmp-=dp[i-1][j-t[i]]*prpow;
				dp[i][j]+=dp[i-1][j-t[i]]*prpow;
			}
			ans+=dp[i][j];
		}
	}
	printf("%f\n",ans);
	return 0;
}
