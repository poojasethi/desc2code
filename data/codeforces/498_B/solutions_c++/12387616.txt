#include <bits/stdc++.h>
using namespace std;
const int N=5000;


int t[N+5];
double p[N+5],np[N+5];

double dp[N+5][N+5];
int main(){
    int n,T,x,y;
    scanf("%d%d",&n,&T);
    for(int i=0;i<n;i++){
        cin>>p[i]>>t[i];
        np[i]=100-p[i];

        p[i]/=100;
        np[i]/=100;
    }

    for(int i=n-1;i>=0;i--){

        double acc=0;

        for(int j=1;j<=T;j++){

             acc+=(dp[i+1][j-1]+1)*p[i];
             if(j>=t[i])acc-=( dp[i+1][j-t[i]] + 1 )*p[i]*pow(np[i],t[i]-1);

             dp[i][j]=acc;
             if(j>=t[i]) dp[i][j]+=( dp[i+1][j-t[i]] + 1 )*pow(np[i],t[i]-1);


             acc*=np[i];

        }

    }
    printf("%.9lf\n",dp[0][T]);
}
