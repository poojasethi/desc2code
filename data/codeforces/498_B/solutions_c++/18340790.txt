#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
using namespace std;

#define N 5020


int n, T, t[N];
double p[N];

double dp[N][N];
double f[N];

int main() {
	scanf("%d%d", &n, &T);
	for(int i = 1; i <= n; ++i) {
		scanf("%lf%d", &p[i], &t[i]);
		p[i] /= 100;
		f[i] = pow(1 - p[i], t[i] - 1);
	}
	dp[0][0] = 1;
	double ans = 0;
	for(int i = 1; i <= n; ++i) {
		double add = 0;
		for(int j = 1; j <= T; ++j) {
			add *= 1 - p[i];
			if(j >= t[i]) add -= dp[i-1][j-t[i]] * f[i] * p[i];
			add += dp[i-1][j-1] * p[i];
			dp[i][j] = add;
			if(j >= t[i]) dp[i][j] += dp[i-1][j-t[i]] * f[i];
			ans += dp[i][j];
			
		}
	}
	printf("%.12lf\n", ans);
	return 0;
}
