#include <iostream>
#include <cstdlib>
#include <cmath>
#include <iomanip>
#define MAXN 5005

using namespace std;

int n,t;
int p[MAXN],k[MAXN];
double dp[MAXN][MAXN],ans;

int main() {
    cin >> n >> t;
    for (int i = 1; i <= n; i++) {
        cin >> p[i] >> k[i];
    }
    dp[0][0] = 1.0;
    for (int i = 1; i <= n; i++) {
        double pp = pow(1.0-p[i]/100.0,k[i]-1);
        double cur = 0.0;
        for (int j = 1; j <= t; j++) {
            cur *= 1.0 - p[i] / 100.0;
            if (j - k[i] >= 0) {
                dp[i][j] += dp[i-1][j-k[i]] * pp;
                cur -= dp[i-1][j-k[i]] * pp * p[i] / 100.0;
            }
            cur += dp[i-1][j-1] * p[i] / 100.0;
            dp[i][j] += cur;
            ans += dp[i][j];
        }
    }
    cout << fixed << setprecision(15) << ans << '\n';
}
