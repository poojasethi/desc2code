#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007
#define N 100005

double p[5000];
int tt[5000];

double dp[5001][5001];

int main()
{
    int n,T,i,x,j,k;

    cin>>n>>T;
    for(i=1;i<=n;i++)
    {
        cin>>x>>tt[i];
        p[i] = 0.01*x;
    }

    double ans = 0;
    dp[0][0] = 1;
    for(i=1;i<=n;i++)
    {
        double x=0, k=pow(1-p[i],tt[i]-1);
        for(j=i;j<=T;j++)
        {
            x=x*(1-p[i])+dp[i-1][j-1];
            if(j>=tt[i])
            {
                x-=dp[i-1][j-tt[i]]*k;
                dp[i][j]=x*p[i]+dp[i-1][j-tt[i]]*k;
            }
            else
                dp[i][j] = x*p[i];
            ans+=dp[i][j];
        }
    }
    printf("%0.9lf",ans);
    return 0;
}
