#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
const int MAXN = 5003;
double f[MAXN],s[MAXN],p[MAXN],ans = 0.0,product,tmp;
int n,T,t[MAXN],x;

int main(){
	scanf("%d%d",&n,&T);
	for(int i = 0;i < n;++ i){
		scanf("%d%d",&x,&t[i]);
		p[i] = (100.0 - x) / 100.0;
	}
	f[0] = 1.0;
	for(int i = 0;i < n;++ i){
		product = 1.0;
		for(int j = 1;j < t[i];++ j) product *= p[i];
		s[0] = f[0];
		for(int j = 1;j <= T;++ j)
			s[j] = s[j - 1] * p[i] + f[j];
		for(int j = T;j;-- j){
			tmp = s[j - 1] - (j >= t[i] ? s[j - t[i]] * product : 0.0);
			f[j] = tmp * (1.0 - p[i]);
			if(j >= t[i])
				f[j] += product * f[j - t[i]];
		}
		for(int j = 1;j <= T;++ j)
			ans += f[j];
		f[0] = 0.0;
	}
	printf("%.9lf\n",ans);
	return 0;
}
