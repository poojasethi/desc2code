#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<math.h>
using namespace std;
double dp[5005][5005];
int main()
{
    int N,T,i,j,t[5005];
    double p[5005],temp,sum;
    scanf("%d%d",&N,&T);
        dp[0][0]=1.0;
        sum=0.0;
        for(i=1;i<=N;i++){
            scanf("%lf%d",&p[i],&t[i]);
            p[i]=p[i]*0.01;
            double vp=pow(1-p[i],t[i]);
            temp=0;
            for(j=0;j<=T;j++){
                if(j-1>=0) temp+=dp[i-1][j-1];
                if(j-t[i]-1>=0) temp-=dp[i-1][j-t[i]-1]*vp;
                dp[i][j]+=temp*p[i];
                if(j-t[i]>=0) dp[i][j]+=dp[i-1][j-t[i]]*vp;
                temp=temp*(1-p[i]);
                sum+=dp[i][j];
            }
        }
        printf("%.11f\n",sum);
    return 0;
}
