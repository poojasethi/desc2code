#include <bits/stdc++.h>
using namespace std;

const int MAXN = 5005;
int N, T, B[MAXN];
double dp[MAXN][MAXN], A[MAXN], pw[MAXN];

double go(double x, int p){
  if(p == 0) return 1;
  double s = go(x, p / 2);
  s *= s;
  if(p % 2) s *= x;
  return s;
}

int main(){
  cin >> N >> T;
  for(int i = 0; i < N; i++){
    cin >> A[i] >> B[i];
    A[i] /= 100;
    pw[i] = go(1 - A[i], B[i] - 1);
  }
  for(int i = N - 1; i >= 0; i--){
    for(int j = T - 1; j >= 0; j--){
      double t = dp[i][j + 1];
      if(j + B[i] + 1 <= T) t -= pw[i] * (dp[i + 1][j + B[i] + 1] + 1);
      if(j + B[i] <= T) t += pw[i] * (dp[i + 1][j + B[i]] + 1);
      dp[i][j] = t * (1 - A[i]) + A[i] * (dp[i + 1][j + 1] + 1);
    }
  }
  cout << fixed << setprecision(6) << dp[0][0] << '\n';
}
