#include <cmath>
#include <cstdio>
using namespace std;

#define n   5005
#define DB  double
#define For(i,a,b)  for(int i=a;i<=b;i++)

int     N,M,t[n];
DB      Ans,p[n],F[n][n];

int     main(){
        scanf("%d%d",&N,&M);F[0][0]=1;
        For(i,1,N)  scanf("%lf%d",&p[i],&t[i]),p[i]/=100.;
        For(i,1,N)  {
            DB  now=F[i-1][0],k=pow(1-p[i],t[i]-1);
            For(j,1,M){
                if  (j>=t[i])   F[i][j]+=F[i-1][j-t[i]]*k,now-=F[i-1][j-t[i]]*k;
                F[i][j]+=now*p[i];  now=now*(1-p[i])+F[i-1][j];
            }   Ans+=now*(i-1);
        }   For(i,1,M)  Ans+=F[N][i]*N;
        printf("%.9lf\n",Ans);
}
