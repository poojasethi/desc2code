#include <bits/stdc++.h>
using namespace std;

double pi[5000];
int ti[5000], N, Tt;

double dp[5001][5001];

int main() {
	cin >> N >> Tt;
	for (int n = 0; n < N; n++) {
		cin >> pi[n] >> ti[n];
		pi[n] *= 0.01;
	}
	for (int n = N-1; n >= 0; n--) {
		double ans = 0.0, rp = pi[n], cache[5001] = {};
		for (int T = 1; T <= Tt; T++) {
			double np = (T > 1 && n != N-1) ? dp[T-1][n+1] : 0;
			ans = (1.0 - rp)*ans + rp * (1+np);
			cache[T] = dp[T][n] = ans;
			if (T >= ti[n]) {
				np = (T-ti[n] >= 1 && n != N-1) ? dp[T-ti[n]][n+1] : 0;
				dp[T][n] += pow(1.0-rp, ti[n]) * ((1+np) - cache[T-ti[n]]);
			}
		}
	}
	cout << fixed << setprecision(10) << dp[Tt][0] << endl;
}
