#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <cstdio>
#include <algorithm>
using namespace std;
int t[5500],n,T;
double p[5500],f[5500][5500];
int main()
{
    scanf("%d%d",&n,&T);
	for (int i=1;i<=n;i++)
	{
		scanf("%lf%d",&p[i],&t[i]);
		p[i]=(double)(p[i]/100);
	}
	f[0][0]=1.0;
	double ans=0.0;
	for (int i=1;i<=n;i++)
	{
		double x=0.0,k=pow(1.0-p[i],t[i]-1);
		for (int j=i;j<=T;j++)
		{
			x=x*(1.0-p[i])+f[i-1][j-1];
			if (j>=t[i]) 
			{
				x-=f[i-1][j-t[i]]*k;
				f[i][j]=x*p[i]+f[i-1][j-t[i]]*k;
			}
			else 
				f[i][j]=x*p[i];
			ans+=f[i][j];
		}
	}
	printf("%.9lf\n",ans);
	return 0;
}
