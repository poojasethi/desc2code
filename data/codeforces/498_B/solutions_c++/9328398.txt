#include <stdio.h>

double dp[5001][5001];

double p, k;
int t;

int main()
{
    int n, T;
    scanf("%d %d", &n, &T);

    dp[0][0] = 1.000;
    double ans = 0.0;

    for (int i = 1; i <= n; i++) {

        scanf("%lf %d", &p, &t);

        p /= 100.0;

        k = 1.0;
        for (int j = 1; j <= t-1; j++) k *= (1.0 - p);

        double C = dp[i-1][0];

        for (int j = 1; j <= T; j++) {

            dp[i][j] = C*p;
            if (j >= t) {
                dp[i][j] += dp[i-1][j-t]*k*(1.0 - p);
                C -= dp[i-1][j-t]*k;
            }
            C = C * (1.0 - p) + dp[i-1][j];

            ans += dp[i][j];

        }

    }

    printf("%.9lf", ans);
}
