#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
double dp[2][5010];
int n,T;
int main()
{
   int cs=0;
   scanf("%d%d",&n,&T);
   dp[cs][0]=1;
   double ans=0;
   for(int i=1;i<=n;i++)
   {
      int a,b;scanf("%d%d",&a,&b);
      double p=a/100.0,t=pow(1-p,b-1);
      double cur=0;
      for(int j=0;j<=T;j++)
      {
         if(j<i){dp[cs^1][j]=0;continue;}
         cur=cur*(1-p)+dp[cs][j-1]*p;
         if(j-b>=0)cur+=dp[cs][j-b]*(1-p)*t;
         if(j-b-1>=0)cur-=dp[cs][j-b-1]*t*(1-p);
         dp[cs^1][j]=cur;
         ans+=dp[cs^1][j];
         //printf("i=%d j=%d dp[i][j]=%.2f\n",i,j,dp[cs^1][j]);
      }
      cs^=1;
   }
   printf("%.10f\n",ans);
}
