#include <bits/stdc++.h>
using namespace std;

const int N = 5003;

double dp[N][N];
int p[N],t[N];

int main()
{
	int n,T;
	dp[0][0] = 1;
	scanf("%d%d",&n,&T);
	for (int i=1;i<=n;i++)
		scanf("%d%d",p+i,t+i);
	double guess, prob,inst;
	double wynik = 0;
	for (int x=1;x<=n;x++)
	{
		guess = 1;
		inst = 1;
		prob = 0;
		for (int i=1;i<t[x];i++) guess*=(100-p[x])/100.0;
//		printf("dla %d guess %lf\n",x,guess);
		for (int i=1;i<=T;i++)
		{
			prob *= (100-p[x])/100.0;
			prob += dp[x-1][i-1];
			dp[x][i] = prob*p[x]/100.0;
			if ( i >= t[x] ) prob -= guess*dp[x-1][i-t[x]];
			if ( i >= t[x] ) dp[x][i] += guess*(100-p[x])*dp[x-1][i-t[x]]/100.0;
//			printf("%d %d -- %lf\n",x,i,dp[x][i]);
			wynik += dp[x][i];
		}
	}
	printf("%.10lf\n",wynik);
}
