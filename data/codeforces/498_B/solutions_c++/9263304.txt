#include <cstring>
#include <cstdio>
const int N = 5005;
double f[2][N][2], p[N];
int cur, n, m, t[N];
int main() {
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) {
		scanf("%lf%d", p+i, t+i);
		p[i] /= 100;
	}
	double ans = 0;
	f[0][0][1] = 1;
	for (int i = 1; i <= n; i++) {
		double pp = 1;
		cur ^= 1;
		memset(f[cur], 0, sizeof(f[cur]));
		for (int j = 0; j < t[i]; j++) pp *= 1-p[i];
		for (int j = 1; j <= m; j++) {
			double temp = f[cur][j-1][0]+f[cur^1][j-1][1];
			f[cur][j][1] = temp*p[i];
			if (j >= t[i])
				f[cur][j][1] += f[cur^1][j-t[i]][1]*pp;
			f[cur][j][0] = temp-f[cur][j][1];
		}
		ans += (i-1)*f[cur][m][0]+i*f[cur][m][1];
	}
	for (int j = 1; j < m; j++)
		ans += n*f[cur][j][1];
	printf("%.9lf\n", ans);
	return 0;
}
