#include<cstdio>
#include<cmath>
using namespace std;
int n,T,t[5010],i,j;
double f[5010],g[5010],p[5010],tmp,cdq,ans,tot,sgtot;
int main()
{
    scanf("%d%d",&n,&T);g[0]=1;sgtot=1;
    for(i=1;i<=n;i++)scanf("%lf%d",&p[i],&t[i]),p[i]/=100.0;
    for(i=1;i<=n;i++)
    {
        tmp=g[i-1]*p[i];
        cdq=pow(1-p[i],t[i]-1);
        for(j=i;j<=T;j++)
        {
            if(j>=t[i])tmp-=g[j-t[i]]*cdq*p[i];
            f[j]=tmp;if(j>=t[i])f[j]+=g[j-t[i]]*cdq*1;
            tmp=tmp*(1-p[i]);tmp+=g[j]*p[i];
        }for(tot=0,j=i;j<=T;j++)tot+=(g[j]=f[j]);g[i-1]=0;
        ans+=(sgtot-tot)*(i-1);sgtot=tot;
    }for(i=n;i<=T;i++)ans+=f[i]*n;
    printf("%lf\n",ans);
}
