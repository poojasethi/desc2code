#include<cstdio>
#include<algorithm>
using namespace std;
double tab[5001];
double tab2[5001];
int main(){
	double wyn=0;
	int n, t;
	scanf("%d%d", &n, &t);
	tab[0]=1;
	for(int s=1; s<=n; s++){
		double p,k=1;
		int cz;
		scanf("%lf%d", &p, &cz);
		p/=100;
		double sum=tab[0];
		for(int i=1; i<cz; i++){
			k*=(1-p);
			tab2[i]=sum*p;
			wyn+=tab2[i];
			sum*=(1-p);
			sum+=tab[i];
		}
		for(int i=cz; i<=t; i++){
			sum-=tab[i-cz]*k;
			tab2[i]=sum*p+tab[i-cz]*k;
			sum*=1-p;
			sum+=tab[i];
			wyn+=tab2[i];
		}
		for(int i=0; i<=t; i++)
			tab[i]=tab2[i];
	}
	printf("%.20lf", wyn);
}
