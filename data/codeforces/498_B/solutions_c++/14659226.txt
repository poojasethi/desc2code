//In the Name of God

#include<iostream>
#include<iomanip>
#define dd double

using namespace std;

const int maxn=5050;
int n,T;
dd p,q;
dd Q,Q_;
dd C;
int t;
dd dp[2][maxn];
dd ans;

int main()
{
  ios_base::sync_with_stdio(0);cin.tie(0);
  cin>>n>>T;
  dp[0][0]=1;
  for(int a=0,b=1,i=1;i<=n;i++,swap(a,b))
    {
      cin>>p>>t;
      p/=100;q=1-p;C=0;Q=q;Q_=1;
      for(int j=t-2;j-->0;Q*=q,Q_*=q);
      dp[b][0]=0;
      for(int j=1;j<=T;j++)
	{
	  if(t!=1 and p!=1)
	    {
	      if(j-t>=0)
		C-=dp[a][j-t]*Q_;
	      C=(C*q)+dp[a][j-1];
	      if(j-t>=0)
		dp[b][j]=C*p+dp[a][j-t]*Q;
	      else
		dp[b][j]=C*p;
	    }
	  else
	    {
	      dp[b][j]=dp[a][j-1];
	    }
	  ans+=dp[b][j];
	}
    }
  cout<<setprecision(15)<<fixed<<ans<<endl;
  return 0;
}
