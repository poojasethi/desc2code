#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
double *a[5001];
int t;
double l(int e,int r){
  if(r>t)
   return 0;
  return a[e][r]+1;
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,d[5000];
  double c,v,c1,v1,v3,s[5000];
  cin>>q>>t;
  for(w=0;w<=q;w++)
   a[w]=new double[t+1];
  for(w=0;w<=t;w++)
   a[q][w]=0;
  for(w=0;w<q;w++){
   cin>>s[w]>>d[w];
   s[w]/=100.0;}
  for(w=q-1;w>=0;w--){
   a[w][t]=0;
   c=1-s[w];
   v=s[w];
   v3=1;
   for(e=1;e<d[w];e++)
    v3*=c;
   c1=0;
   for(e=t-1;e>=0;e--){
    c1*=c;
    c1+=(l(w+1,e+1)-l(w+1,e+d[w])*v3)*v;
    a[w][e]=c1+l(w+1,e+d[w])*v3;}}
  printf("%.7f",a[0][0]);
  return 0;}
