#include <iostream>
#include <cstdio>
#include <math.h>
#define for(i,a,b) for(int i=a;i<=b;i++)
#define ford(i,b,a) for(int i=b;i>=a;i--)
using namespace std;
const int nmax=5001;
int t,n,ti[nmax];
double dp[nmax][nmax],p[nmax],ans=0;
int main()
{
    //freopen("input.inp","r",stdin);
    scanf("%d%d",&n,&t);
    for(i,1,n) scanf("%lf%d",&p[i],&ti[i]),p[i]/=100;
    dp[0][0]=1;
    for(i,1,n){
        double h=pow(1-p[i],ti[i]-1),s=0;
        for(j,i,t){
            s=s*(1-p[i])+dp[i-1][j-1];
            if (j>=ti[i]){
                s-=dp[i-1][j-ti[i]]*h;
                dp[i][j]=s*p[i]+dp[i-1][j-ti[i]]*h;
            }
            else dp[i][j]=s*p[i];
            ans+=dp[i][j];
        }
    }
    printf("%.9lf",ans);
}
