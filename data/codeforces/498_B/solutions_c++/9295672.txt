#include <cstdio>
typedef double db;
int n,m,F,t[5005];
db p[5005],f[5005][2],pw[5005];
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%lf",&p[i]),p[i]/=100,scanf("%d",&t[i]);
	for(int i=n;i;i--){
		db w=1;
		for(int j=1;j<t[i];j++) w*=1-p[i];
		db now=0;
		for(int j=m-1;j>=0;j--){
			now=(1-p[i])*now+p[i]*(1+f[j+1][F]);
			if(j+t[i]<=m)
				now-=w*p[i]*(1+f[j+t[i]][F]),
				f[j][!F]=now+w*(1+f[j+t[i]][F]);
			else
				f[j][!F]=now;
		}
		F^=1;
	}
	printf("%.9lf\n",f[0][F]);
}
