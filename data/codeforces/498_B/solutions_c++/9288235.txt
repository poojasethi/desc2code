#include <bits/stdc++.h>
using namespace std;
const int maxn = 5010;

double dp[maxn][maxn], p[maxn];
int t[maxn], n, T;

int main()
{
    //freopen("e:\\in.txt", "r", stdin);
    int n, T;
    cin >> n >> T;
    for(int i = 1; i <= n; i++) {
        cin >> p[i] >> t[i];
        p[i] *= 0.01;
    }
    memset(dp, 0, sizeof(dp));
    dp[0][0] = 1;

    double ans = 0;
    for(int i = 1; i <= n; i++) {
        double P = pow(1-p[i], t[i]), tmp = 0;
        for(int j = i; j <= T; j++) {
            tmp += dp[i-1][j-1];
            if(j - t[i] - 1 >= 0)   tmp -= dp[i-1][j-t[i]-1] * P;
            dp[i][j] = tmp * p[i];
            if(j - t[i] >= 0)       dp[i][j] += dp[i-1][j-t[i]] * P;
            tmp *= (1-p[i]);
            ans += dp[i][j];
        }
    }
    printf ( "%.6f\n" , ans ) ;
}
