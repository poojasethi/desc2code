#include <stdio.h>
#include <string.h>
#include <math.h>

#define N 5050

struct node
{
	double p;
	int t;
}f[N];

double dp[N][N];

int main()
{
	int i,j,n,m;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		for(i=1;i<=n;i++)
		{
			scanf("%lf %d",&f[i].p,&f[i].t);
			f[i].p/=100.0;
		}
		for(i=0;i<=m;i++)
		  dp[0][i]=0;
		  dp[0][0]=1;
		  double sum=0;
		  for(i=1;i<=n;i++)
		  {
		  	double s=0,q;
		  	q=pow(1-f[i].p,1.0*f[i].t);
		  	for(j=0;j<=m;j++)
		  	{
		  		if(j-1>=0)
		  		s+=dp[i-1][j-1];
		  		if((j-f[i].t-1)>=0)
		  		s-=dp[i-1][j-f[i].t-1]*q;
		  		dp[i][j]=s*f[i].p;
		  		if((j-f[i].t)>=0)
		  		dp[i][j]+=dp[i-1][j-f[i].t]*q;
		  		s*=(1.0-f[i].p);
		  		sum+=dp[i][j];
			  }
		  }
		  printf("%.10lf\n",sum);
	}
	return 0;
}
