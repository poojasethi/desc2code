#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

string S;
int K,cnt[26],id[26],flag[26],result;

bool cmp(int x,int y)
{
	return cnt[x]<cnt[y];
}

int main()
{
	cin >> S >> K;
	for (int i=0;i<S.size();i++)
		cnt[S[i]-'a']++;
	for (int i=0;i<26;i++)
		id[i]=i;
	sort(id,id+26,cmp);
	for (int i=0;i<26;i++)
	{
		if (K<cnt[id[i]])
			break;
		K-=cnt[id[i]];
		flag[id[i]]=1;
	}
	for (int i=0;i<26;i++)
		if (!flag[i])
			result++;
	cout << result << endl;
	for (int i=0;i<S.size();i++)
		if (!flag[S[i]-'a'])
			cout << S[i];
	return 0;
}
