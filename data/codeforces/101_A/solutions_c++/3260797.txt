#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
char str[100005];
int c[26];
pair<int,char> A[26];
int main(){
  int k;
  scanf("%s%d",str,&k);
  int size=strlen(str);
  for(char ch='a';ch<='z';ch++)     A[ch-'a'].second=ch;
  for(int i=0;i<size;i++)    A[str[i]-'a'].first++;
  sort(A,A+26);
  int cont=0,p=-1;
  for(int i=0;i<26;i++){
    cont+=A[i].first;
    c[A[i].second-'a']=i;
    if(cont<=k) p=i;
  }
  printf("%d\n",25-p);
  for(int i=0;i<size;i++) if(c[str[i]-'a']>p) putchar(str[i]);
  return 0;
}
