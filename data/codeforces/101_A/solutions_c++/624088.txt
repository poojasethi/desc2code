#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<iostream>
using namespace std;
int n,k,a[30];
char s[100050];
int main()
{
	scanf("%s%d",s,&k);
	memset(a,0,sizeof(a));
	for(int i=0;s[i];i++)
		a[s[i]-'a']++;
	a[26]=10000000;
	while(k){
		int idx=26;
		for(int i=0;i<26;i++)
			if(a[i]&&a[i]<a[idx])
				idx=i;
		if(idx==26)
			break;
		if(a[idx]>=k){
			a[idx]-=k;
			k=0;
		}
		else{
			k-=a[idx];
			a[idx]=0;
		}
	}
	int ans=0;
	for(int i=0;i<26;i++)
		if(a[i])
			ans++;
	cout<<ans<<endl;
	for(int i=0;s[i];i++)
		if(a[s[i]-'a'])
			printf("%c",s[i]);
	cout<<endl;
	return 0;
}
