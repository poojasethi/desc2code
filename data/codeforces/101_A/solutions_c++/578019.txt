#include <stdio.h>
char dat[100004];
int k;
int cnt[26];
int main(){
	scanf("%s %d",dat,&k);
	for(int i=0; i< 26;i++) cnt[i] = 0;
	for(int i=0; dat[i] != '\0'; i++){
		cnt[ dat[i] - 'a' ] ++;
	}
	int sum = 0;
	while(1){
		int mn = 999999, who = -1;
		for(int i=0;i<26;i++){
			if(cnt[i] > 0){
				if(mn > cnt[i]) {
					mn = cnt[i]; who = i;
				}
			}
		}
		if(mn == 999999) break;
		if(sum + mn > k) break;
		sum += mn;
		cnt[who] = 0;
	}
	int m = 0;
	for(int i=0; i<26; i++){
		if(cnt[i] > 0) m ++;
	}
	printf("%d\n",m);
	for(int i=0; dat[i] != '\0'; i++){
		if(cnt[ dat[i] - 'a' ] > 0) printf("%c", dat[i]);
	}
	printf("\n");

}
