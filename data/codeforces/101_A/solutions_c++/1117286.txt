#include<iostream>
#include<conio.h>
#include<map>
#include<vector>
#include<cstring>
#include<algorithm>
using namespace std;
pair<int,char> p[30];
map<char,bool> mp;
int main()
{
    vector<char> str,del;
    for(int i=1;i<=26;i++)
    p[i].second=i+'a'-1;
    string s;
    cin>>s;
    int k,m=0;
    cin>>k;
    for(int i=0;i<s.length();i++)
    p[s[i]-'a'+1].first++;
    sort(p+1,p+27);
    int j=0;
    while(k-p[j+1].first>=0&&j<26)
    {
    j++;
    k-=p[j].first;
    mp[p[j].second]=true;
}
    m=26-j;
    cout<<m<<endl;
    for(int i=0;i<s.length();i++)
    if(!mp[s[i]])
    cout<<s[i];
    return 0;
}
