#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
int cnt[26],i,n,k,ans,u[26];
char ch[2000000];
bool cmp(int i,int j) {return cnt[i]<cnt[j];}
int main()
{
  scanf("%s\n",ch+1);
  scanf("%d\n",&k);
  n=strlen(ch+1);
  for (i=1;i<=n;i++) cnt[ch[i]-'a']++;
  for (i=0;i<=25;i++) u[i]=i;
  sort(u,u+25+1,cmp);
  for (i=0;i<=25;i++)
    if (cnt[u[i]]<=k) k-=cnt[u[i]],cnt[u[i]]=0;
  ans=0;
  for (i=0;i<=25;i++) ans+=(cnt[i]!=0);
  printf("%d\n",ans);
  for (i=1;i<=n;i++)
    if (cnt[ch[i]-'a']) printf("%c",ch[i]);
  return 0;
}
