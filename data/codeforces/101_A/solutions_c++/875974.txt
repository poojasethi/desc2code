#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
pair<int,int>p[26];
char s[1<<18];
bool vis[1<<8];
int main()
{
    int n,k,i;
    for(i=0;i<26;i++)
    {
        p[i].first=0; p[i].second=i;
    }
    scanf("%s%d",s,&n);
    for(i=0;s[i];i++)
    {
        p[s[i]-'a'].first++;
    }
    sort(p,p+26);
    for(i=0;i<26;i++)
    {
        if(p[i].first<=n)
        {
            n-=p[i].first; vis[p[i].second+'a']=1;
        }
        else break;
    }
    printf("%d\n",26-i);
    for(i=0;s[i];i++)
    {
        if(vis[s[i]]) continue;
        printf("%c",s[i]);
    }
    puts("");
    return 0;
}
