#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <algorithm>
using namespace std ;

#define fo(i,a,b) for(int i=a;i<=b;i++)

string S ;
int K ;
pair<int,int> cnt[26] ;
bool b[26] ;

int	main() {
	cin >> S >> K ;
	fo(i,0,25) {
		cnt[i].first = 0 ;
		cnt[i].second = i ;
	}
	fo(i,0,S.size()-1) cnt[S[i]-'a'].first ++ ;
	
	sort( cnt , cnt+26 ) ;
	fo(i,0,25) b[i] = false ;
	int c = 0 ;
	fo(i,0,25) {
		if ( K >= cnt[i].first ) {
			b[cnt[i].second] = true ;
			K -= cnt[i].first ;
		} else c ++ ;
	}

	cout << c << "\n" ;
	fo(i,0,S.size()-1) if ( !b[S[i]-'a'] ) cout << S[i] ;
	cout << "\n" ;
}
