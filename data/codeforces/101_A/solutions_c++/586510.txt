#include <stdio.h>
#include <algorithm>

using namespace std;

char str[100010];
int a[100010];

int main()
{
	int i, j, dec, k, ch, ans;
	gets(str);
	scanf("%d", &k);
	for (i=0; str[i]!='\0'; i++)
	  a[str[i]-'a']++;
	for (i=0; i<26; i++)
	{
		ch=-1;
		for (j=0; j<26; j++)
		  if (a[j]>0)
		  	if (ch==-1 || a[j]<a[ch]) ch=j;
		dec=min(a[ch], k);  	
	  	a[ch]-=dec;
	  	k-=dec;
	  	if (k==0) break;
	}
	ans=0;
	for (i=0; i<26; i++)
	  if (a[i]>0) ans++;
	printf("%d\n", ans);  
	for (i=0; str[i]!='\0'; i++)
	  if (a[str[i]-'a']>0)
	  {
	  	a[str[i]-'a']--;
		printf("%c", 	str[i]);
	  }
	return 0;	
}
