#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

char s[100010];
int k,l;

bool ok[30];

struct arr
{
	int x,y;
} cnt[30];

bool comp(arr a,arr b)
{
	return a.x<b.x;
}

int main()
{
	scanf("%s%d",s,&k);
	l=strlen(s);
	for(int i=0;i<l;i++) cnt[s[i]-97].x++;
	for(int i=0;i<26;i++) cnt[i].y=i;
	sort(cnt,cnt+26,comp);
	int i;
	for(i=0;i<26;i++) if (k>=cnt[i].x) k-=cnt[i].x,ok[cnt[i].y]=true; else break;
	printf("%d\n",26-i);
	for(i=0;i<l;i++) if (!ok[s[i]-97]) printf("%c",s[i]);
	puts("");
	
	return 0;
}
