#include<bits/stdc++.h>
using namespace std;
struct st{
int id;
int cnt;
};
vector<st>v;
int a[100];
int p[100];
bool cmp(st x,st y)
{
	return x.cnt<y.cnt;
}
int main()
{
	string s;
	int k;
	cin>>s>>k;
	for(int i=0;i<s.length();i++)
	{
		a[s[i]-'a']++;
	}
	for(int i=0;i<26;i++)
	{
		if(a[i])
		{
			st t;
			t.id=i;
			t.cnt=a[i];
			v.push_back(t);
		}
	}
	sort(v.begin(),v.end(),cmp);
	int cn=0;
	for(int i=0;i<v.size();i++)
	{
		if(v[i].cnt<=k)
		{
			k-=v[i].cnt;
			p[v[i].id]=1;
		}
		else
			cn++;
	}
	cout<<cn<<endl;
	for(int i=0;i<s.length();i++)
	{
		if(p[s[i]-'a']==0)
			cout<<s[i];
	}
	cout<<endl;

}
