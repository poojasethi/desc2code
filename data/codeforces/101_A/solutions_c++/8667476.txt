#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int a[N],id[N],f[N];
char p[N];
inline bool cc1(const int &A,const int &B) {
	return a[A]<a[B];
}
int main() {
	scanf("%s%d",p+1,&m);
	n=strlen(p+1);
	For(i,1,n) a[p[i]]++;
	For(i,1,200) id[i]=i;
	sort(id+1,id+201,cc1);
	For(i,1,200) {
		int A=id[i];
		if (m<a[A]) break;
		m-=a[A];
	}
	for (j=i;j<=200;j++) f[id[j]]=1;
	printf("%d\n",200-(i-1));
	For(i,1,n) if (f[p[i]]) printf("%c",p[i]);
	printf("\n");
	return 0;
}
