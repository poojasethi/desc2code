#include<iostream>
#include<algorithm>

using namespace std;

pair<int,int> cnt[26];
bool bad[26];
string str;

int main()
{
	int k;
	cin>>str>>k;
	for (int i=0;i<26;i++)
		cnt[i].second = i;
	for (int i=0;i<str.size();i++)
		cnt[str[i] - 'a'].first++;
	int di = 0;
	for (int i=0;i<26;i++)
		if (cnt[i].first > 0)
			di++;
			
	sort(cnt,cnt+26);
	int sum = 0;
	for (int i=0;i<26;i++)
	{
		if (cnt[i].first != 0 && cnt[i].first <= k)
		{
			sum++;
			k-=cnt[i].first;
			bad[cnt[i].second] = 1;
		}
	}
	cout<<di - sum<<endl;
	for (int i=0;i<str.size();i++)
		if (!bad[str[i] - 'a'])
			cout<<str[i];
	cout<<endl;
	return 0;
}
