#include<stdio.h>
#include<algorithm>
using namespace std;
struct letter
{
	int count;
	int index;
}l[27];
int cmp(letter a,letter b)
{
	return a.count<b.count;
}
int main()
{
	int i,k,n;
	bool flag[26]={0};
	char str[100005];
	scanf("%s%d",str,&k);
	n=0;
    for(i=0;str[i];i++)
	{
		if(l[str[i]-'a'].count==0){l[str[i]-'a'].index=str[i]-'a';flag[str[i]-'a']=1;n++;}
		l[str[i]-'a'].count++;
	}
	sort(l,l+26,cmp);
	for(i=0;i<26;i++)
		if(l[i].count<=k&&l[i].count){k-=l[i].count;flag[l[i].index]=0;n--;}
		printf("%d\n",n);
	for(i=0;str[i];i++)
		if(flag[str[i]-'a'])printf("%c",str[i]);
		printf("\n");
	return 0;
}
