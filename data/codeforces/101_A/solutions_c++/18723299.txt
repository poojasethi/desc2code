#include<bits/stdc++.h>
#define newl '\n'
using namespace std;
const int N=1e5+5,M=26;
char s[N];
int b[M],sm,pos;
pair<int,char> p[M];
int main()
{

    int n,k,i;
    cin>>s>>k;
    for(i=0;i<26;i++)p[i].second='a'+i;
    for(i=0;s[i];i++)p[s[i]-'a'].first++;
    sort(p,p+M);
    for(i=0,pos=-1;i<26;i++)
    {
    	sm+=p[i].first;
        b[p[i].second-'a']=i;
        if(sm<=k)pos=i;  
    } 
    cout<<25-pos<<newl;
    for(i=0;s[i];i++)
    	if(b[s[i]-'a']>pos)
    		cout<<s[i];
	return 0;
}
