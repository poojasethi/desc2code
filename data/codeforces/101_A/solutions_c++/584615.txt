#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 100010
int num[26];
int tag[26];
int p[26];
int n,m;
char s[maxn];
bool cmp(int x,int y){
	return num[x]<num[y];
}
int main(){
	scanf("%s",s);
	n=strlen(s);
	scanf("%d",&m);
	for (int i=0;i<n;i++) num[s[i]-'a']++;
	for (int i=0;i<26;i++) p[i]=i;
	std::sort(p,p+26,cmp);
	for (int i=0;i<26;i++)
		if (m>=num[p[i]]){
			m-=num[p[i]];
			tag[p[i]]=1;
		}
	int ans=0;
	for (int i=0;i<26;i++)
		if (!tag[i] && num[i]) ans++;
	printf("%d\n",ans);
	for (int i=0;i<n;i++)
		if (!tag[s[i]-'a']) printf("%c",s[i]);
	puts("");
	return 0;
}
