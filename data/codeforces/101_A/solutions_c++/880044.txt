#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char a[100005];
int w[26];
int r[26];
int srt[26];
bool cmp(int x, int y) {
	return w[x]<w[y];
}

int main(void) {
	int k;
	scanf("%s%d", a, &k);
	for(int i=0;a[i];i++)
		w[a[i]-'a']++;
	for(int i=0;i<26;i++) srt[i]=i;
	sort(srt, srt+26, cmp);
	int d=0;
	for(int i=0;i<26;i++) {
		int p=min(w[srt[i]], k);
		k-=p;
		r[srt[i]] = w[srt[i]] - p;
		if(r[srt[i]]) ++d;
	}
	printf("%d\n", d);
	for(int i=0;a[i];i++) {
		if(r[a[i]-'a']-->0) printf("%c", a[i]);
	}
	puts("");
	return 0;
}
