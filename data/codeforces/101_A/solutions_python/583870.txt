#!/usr/bin/env python
# -*- coding: utf-8 -*-
import collections, sys

s = raw_input()
k = int(raw_input())
chrs = collections.defaultdict(int)
for c in s:
    chrs[c] += 1
lchrs = sorted([(y, x) for x, y in chrs.items()])[::-1]
while lchrs and k >= lchrs[-1][0]:
    k -= lchrs[-1][0]
    lchrs.pop()
kit = set([x for _, x in lchrs])
print len(kit)
for c in s:
    if c in kit:
        sys.stdout.write(c)
sys.stdout.write('\n')
