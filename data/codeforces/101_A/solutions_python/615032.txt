s = raw_input().strip()
k = int(raw_input().strip())
length = 0
all_cha = {}
for c in s:
  length+=1
  try:
    all_cha[c]+=1
  except KeyError:
    all_cha[c]=1

delete = []
breaker = True
while k and breaker:
  for i in sorted(all_cha, key=all_cha.get):
    temp = all_cha[i]
    if temp<=k:
      k=k-temp
      delete.append(i)
    else:
      breaker = False
new_s=""
for l in s:
  if l not in delete:
    new_s=new_s+l
print max(0,len(all_cha.keys())-len(delete))
print new_s
