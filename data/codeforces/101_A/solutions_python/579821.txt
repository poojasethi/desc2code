
s = list(raw_input())
k = input()

cnt = [0] * 26
for c in s:
  val = ord(c) - ord('a')
  cnt[val] += 1

a = sorted([(cnt[i], i) for i in range(26)])
replace = ''
while a and k >= a[0][0]:
  c = chr(ord('a') + a[0][1])
  replace += c
  k -= a[0][0]
  a = a[1:]

for i in xrange(len(s)):
  if s[i] in replace:
    s[i] = ''

ans_str = ''.join(s)
ans_cnt = len(set(list(ans_str)))

print ans_cnt
print ans_str
