import operator
#def gcd(a,b):
#   while b > 0: a,b = b, a%b
#   return a
#def lcm(a, b):
#   return a*b/gcd(a,b)

T = raw_input()
k = input()

lets = set(T)
cnt = {}

for i in lets:
    cnt[i] = T.count(i)
sor = sorted(cnt.iteritems(), key=operator.itemgetter(1))

fuldel = ""
for i in sor:
    if i[1] <= k:
        k -= i[1]
        fuldel += i[0]
            
for i in fuldel:
    T = T.replace(i, "")
    
print len(set(T))
print T
