#!/usr/bin/env python
import sys

s = sys.stdin.readline().strip()
k = int(sys.stdin.readline().strip())
counts = []
for i in xrange(0, 26):
    c = chr(i + ord('a'))
    counts.append( [s.count(c), c] )
counts.sort(key=lambda x: x[0])

for i in xrange(len(counts)):
    if counts[i][0] <= k:
        k -= counts[i][0]
        counts[i][0] = 0
counts.sort(key=lambda x: x[1])
print sum(map(lambda x: 1 if x[0] else 0, counts))
for c in s:
    if counts[ord(c) - ord('a')][0] > 0:
        sys.stdout.write(c)
