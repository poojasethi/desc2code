from collections import defaultdict
s = raw_input()
m = input()


a = defaultdict(int)
for c in s:
    a[c] += 1

counts = sorted( [ [v, k] for k, v in a.items() ] )

dis = len(counts)
for (v, k) in counts:
    if m < v:
        break
    m -= v
    dis -= 1
    s = s.replace(k, '')

print dis
print s
