import sys
def read_values():
  return map(int,raw_input().split())
s=raw_input()
k=read_values()[0]

h = {}
for c in s:
  if c not in h:
    h[c]=0
  h[c]+=1

t = h.keys()
t.sort(key=h.__getitem__)

res1=len(t)
deleted=0
skip={}
for c in t:
  if deleted+h[c]<=k:
    skip[c]=True
    deleted += h[c]
    res1 -= 1
  else:
    skip[c]=False

res2=''
for c in s:
  if skip[c]:
    continue
  res2 += c

print res1
print res2
