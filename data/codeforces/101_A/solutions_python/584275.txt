import sys

s = raw_input()
n = int(raw_input())
a = {}
for k in s:
    if not a.has_key(k):
        a[k] = 1
    else:
        a[k] += 1
b = [(v, k) for k, v in a.iteritems()]
b.sort()
t = len(b)
a = []
for k in b:
    if n >= k[0]:
        n -= k[0]
        t -= 1
    else:
        break
    a.append(k[1])
a = set(a)
print t
for k in s:
    if not k in a:
        sys.stdout.write(k)
