
s = raw_input()
k = input()

T = {}
for c in s:
   if c in T: T[c] += 1
   else: T[c] = 1

L = [(v, x) for x, v in T.iteritems()]
L.sort()
i, diffr = 0, len(L)
for v, x in L:
   if k < v: break
   k -= v
   T[x] = 0
   diffr -= 1

print diffr

sub = ''
if diffr > 0:
   for c in s:
      if T[c] > 0:
         T[c] -= 1
         sub += c

print sub
