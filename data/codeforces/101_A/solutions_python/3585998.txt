#!/usr/bin/env python
import sys
def read_str(): return sys.stdin.readline().strip()
def read_int(): return int(sys.stdin.readline().strip())

def create_char_map(sentence):
    char_map = {}
    for ch in sentence:
        char_map[ch] = char_map.get(ch, 0) + 1
    return char_map

def create_deletable_map(char_map, count):
    deletable = {}

    for key in sorted(char_map, key=lambda x:char_map[x]):
        if count <= 0:
            break
        val = char_map[key]
        if val <= count:
            deletable[key] = val
            count -= val
        else:
            deletable[key] = count
            break

    return deletable

def solve(sentence, count):
    ch_map = create_char_map(sentence)
    del_map = create_deletable_map(ch_map, count)

    ans_str = ''
    for ch in sentence:
        if ch in del_map:
            del_map[ch] -= 1
            if (del_map[ch] == 0):
                del del_map[ch]
        else:
            ans_str += ch

    print(len(create_char_map(ans_str)))
    print(ans_str)

if __name__ == '__main__':
    solve(read_str(), read_int())
