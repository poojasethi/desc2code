line = raw_input()
num = int(raw_input())
liters = []
freq = []
difer = 0
for a in line:
    if not a in liters:
        difer+=1
        liters.append(a)
        freq.append([line.count(a),a])
freq.sort()
deleted = 0
for i in freq:
    if deleted+i[0]<=num:
        deleted+=i[0]
        difer-=1
        line = line.replace(i[1],'')
    else: break
print difer
if line: print line
