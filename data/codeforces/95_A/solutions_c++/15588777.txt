#include<cstdio>
#include<cstring>
char fbd[110][110];
char s[110];
bool vis[110];
int main(){
	int n,i;scanf("%d",&n);
	for(i=0;i<n;i++)scanf("%s",fbd[i]);
	scanf("%s",s);
	char let[2];scanf("%s",let);
	int lt=let[0]-'a';
	int j,k,l=strlen(s),ll;
	for(i=0;i<n;i++){
		ll=strlen(fbd[i]);
		for(j=0;j<l;j++){
			for(k=0;k<ll&&j+k<l;k++){
				if(('a'<=s[j+k]&&s[j+k]<='z'?
				s[j+k]-'a':s[j+k]-'A')!=
				('a'<=fbd[i][k]&&fbd[i][k]<='z'?
				 fbd[i][k]-'a':fbd[i][k]-'A'))break;
			}
			if(k==ll){
				for(k=0;k<ll;k++)vis[j+k]=true;
			}
		}
	}
	for(ll=0;ll<26;ll++)if(ll!=lt)break;
	for(i=0;i<l;i++)if(vis[i]){
		if('a'<=s[i]&&s[i]<='z')j=s[i]-'a',k='a';
		else j=s[i]-'A',k='A';
		if(j==lt)s[i]=k+ll;
		else s[i]=k+lt;
	}
	printf("%s",s);
	return 0;
}
