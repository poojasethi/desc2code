#include <cstdio>
#include <cstring>
int N;
char s[105][105];
char w[105];
int len[105], lw;
bool b[105];
char let[10];

int main(){
	scanf("%d",&N);
	for (int i=0;i<N;++i){
		scanf("%s",s[i]);
		len[i] = strlen(s[i]);
	}
	scanf("%s",w);
	scanf("%s",let);
	if (let[0]<='Z') let[1]=let[0]+32;
	else{
		let[1] = let[0];
		let[0] -= 32;
	}
	lw = strlen(w);
	for (int i=0;w[i];++i){
		for (int j=0;j<N;++j){
			bool ok = true;
			for (int k=0;k<len[j];++k){
				char ca = w[i+k];
				char cb = s[j][k];
				if (ca<='Z') ca+=32;
				if (cb<='Z') cb+=32;
				if (ca!=cb) ok = false;
			}
			if (ok) for (int k=0;k<len[j];++k){
				b[i+k] = true;
			}
		}
		
		if (b[i]==true){
			if (w[i]<='Z'){
				if (w[i]!=let[0])
					w[i] = let[0];
				else{
					if (w[i]=='A') w[i] = 'B';
					else w[i] = 'A';
				}
			}else{
				if (w[i]!=let[1])
					w[i] = let[1];
				else{
					if (w[i]=='a') w[i] = 'b';
					else w[i] = 'a';
				}
			}
		}
	}
	
	puts(w);
	return 0;
}
