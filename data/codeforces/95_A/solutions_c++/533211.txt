#include <string.h>
#include <stdio.h>

int n;
char s[200][200];
char w[200], l, tmp[4];
int length[200];
bool check[200];

inline char small(char x)
{
	return ('a'<=x && x<='z') ? x : (x-'A'+'a');
}

int main()
{
	int i, j,k;
	scanf("%d", &n);
	for (i=0; i<n; i++) scanf("%s", s[i]);
	scanf("%s", w);
	scanf("%s", tmp);
	l = tmp[0];
	if ('A'<=l && l<='Z') l = l-'A'+'a';

	for (i=0; i<n; i++) {
		for (j=0; j<strlen(w); j++) {
			for (k=0; k<strlen(s[i]) && j+k<strlen(w); k++) {
				if (small(w[j+k]) != small(s[i][k]))
					break;
			}
			if (k==strlen(s[i])) {
				for (k=0; k<strlen(s[i]); k++)
					check[j+k] = true;
			}
		}
	}

	for (i=0; i<strlen(w); i++) {
		if (check[i]) {
			if (w[i] == l || w[i]-'A'+'a' == l) {
				if ('A' <= w[i] && w[i] <= 'Z') {w[i] = (l!='a'?'A':'B');}
				else {w[i] = (l!='a'?'a':'b');}
			}
			else {
				if ('A' <= w[i] && w[i] <= 'Z') w[i] = l-'a'+'A';
				else w[i] = l;
			}
		}
	}

	printf("%s", w);
	return 0;
}
