#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char a[105][105];
char str[105];
char l[105];
int change[105];

void assign(char &x, char y) {
	if(x>='a') x= y<='Z'? y+32:y;
	else if(x<='Z') x= y>='a'? y-32:y;
}

bool same(char x, char y) {
	if(!x||!y) return false;
	return (x-y)%32==0;
}

bool match(char *x, char *y) {
	for(int i=0;y[i];i++)
		if(!same(x[i], y[i])) return false;
	return true;
}

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++)
		scanf("%s", a[i]);
	scanf("%s%s", str, l);
	for(int i=0;str[i];i++) {
		for(int j=0;j<n;j++)
			if(match(str+i, a[j])) {
				for(int k=0;a[j][k];k++)
					change[i+k] = 1;
			}
	}
	for(int i=0;str[i];i++) {
		if(change[i]) {
			if(same(str[i], l[0])==false) assign(str[i], l[0]);
			else assign(str[i], 'a'+(l[0]=='a' || l[0]=='A'));
		}
	}
	puts(str);
	return 0;
}
