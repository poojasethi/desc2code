#include "bits/stdc++.h"
using namespace std;
char s[111][111], t[111], letter[10];
int n, i, j, k, lt, l[111];
bool b, v[111];
int main() {
	memset(v, 0, sizeof v);
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		scanf("%s", s[i]);
		l[i] = strlen(s[i]);
	}

	scanf("%s", t);
	lt = strlen(t);

	scanf("%s", letter);

	for (i = 0; i < n; i++) {
		for (j = 0; j <= lt - l[i]; j++) {
			for (k = 0, b = true; k < l[i] && b; k++) if (tolower(t[j + k]) != tolower(s[i][k])) b = false;
			if (b) for (k = 0; k < l[i]; k++) v[j + k] = true; 
		}
	}

	for (i = 0; i < lt; i++) if (v[i]) {
		if ('A' <= t[i] && t[i] <= 'Z') letter[0] = toupper(letter[0]); else letter[0] = tolower(letter[0]);
		if (t[i] != letter[0]) t[i] = letter[0];
		else if (letter[0] == 'a' || letter[0] == 'A') {
			if ('A' <= t[i] && t[i] <= 'Z') t[i] = 'B'; else t[i] = 'b';
		} else {
			if ('A' <= t[i] && t[i] <= 'Z') t[i] = 'A'; else t[i] = 'a';
		}
	}
	printf("%s\n", t);
	return 0;
}
