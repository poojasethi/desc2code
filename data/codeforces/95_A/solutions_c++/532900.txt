#include <iostream>
#include <string>
#include <cstring>
#include <cstdio>
#include <vector>
using namespace std;

char tolower(char c)
{
	if('A'<=c&&c<='Z') return c-'A'+'a';
	return c;
}

bool isequal(int start,string & str,string &cmp)
{
	for(int i=0;i<cmp.size();i++)
	{
		if(tolower(str[i+start])!=tolower(cmp[i]))
			return false;
	}
	return true;
}

int main()
{
	int n;
	cin>>n;
	vector<string> str(n);
	string w,t,res;
	char good,bad;
	for(int i=0;i<n;i++)
	{
		cin>>str[i];
	}
	cin>>w;
	cin>>t;
	good=t[0];
	if(good=='a')
		bad='b';
	else
		bad='a';
	res=w;
	for(int i=0;i<res.size();i++)
	{
		for(int j=0;j<n;j++)
		{
			if(isequal(i,w,str[j]))
			{
				for(int k=0;k<str[j].size();k++)
				{
					if(tolower(str[j][k])==good)
					{
						if('A'<=w[i+k]&&w[i+k]<='Z')
							res[i+k]=bad-'a'+'A';
						else
							res[i+k]=bad;
					}
					else
					{
						if('A'<=w[i+k]&&w[i+k]<='Z')
							res[i+k]=good-'a'+'A';
						else
							res[i+k]=good;
					}
				}
			}
		}
	}
	cout<<res<<endl;
	return 0;
}
