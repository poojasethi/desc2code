#include<iostream>
using namespace std;

int main(){
  int n,s;
  string f[100],str,ans;
  char l;

  cin >> n;
  for(int i=0;i<n;i++)cin >> f[i];
  cin >> str;
  cin >> l;

  s = (int)str.size();
  for(int i=0;i<s;i++)ans.push_back('*');
  for(int i=0;i<s;i++){
    for(int j=0;j<n;j++){
      bool flag = true;
      for(int k=0;k<(int)f[j].size();k++){
	if(f[j][k] != str[i+k] && f[j][k] - str[i+k] != 32 && f[j][k] - str[i+k] != -32){
	  flag = false;
	  break;
	}
      }
      if(flag){
	for(int k=0;k<(int)f[j].size();k++){
	  if(str[i+k] > 'Z' || str[i+k] < 'A'){
	    if(str[i+k] != l)ans[i+k] = l;
	    else if(str[i+k] != 'a')ans[i+k] = 'a';
	    else ans[i+k] = 'b';
	  }else{
	    if(str[i+k] - l != -32)ans[i+k] = l-32;
	    else if(str[i+k] != 'A')ans[i+k] = 'A';
	    else ans[i+k] = 'B';
	  }
	}
      }
    }
    if(ans[i] == '*')ans[i] = str[i];
  }
  cout << ans << endl;
}
