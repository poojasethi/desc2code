#include <algorithm>
#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>

using namespace std;

int n;
int a[200];
string w, s[200], ans;
char ch, ch1;

int main ()
{
	cin >> n;
	
	for (int i = 0; i < n; i++)
	{
		cin >> s[i];
		for (int j = 0; j < s[i].size (); j++)
			if (s[i][j] <= 'Z')
				s[i][j] = s[i][j]-'A'+'a';
	}
	
	cin >> w;
	
	for (int i = 0; i < w.size (); i++)
		if (w[i] <= 'Z')
			a[i] = 1,
			w[i] = w[i]-'A'+'a';
	cin >> ch;
	ch1 = 'a';
	if (ch1 == ch) ch1 = 'b';

	ans = w;
	for (int i = 0, best; i < w.size (); i++)
	{
		best = 0;
		for (int j = 0; j < n; j++)
		{
			if (w.size ()-i >= s[j].size () && w.substr (i, s[j].size ()) == s[j])
				best = max (best, (int)s[j].size ());
		}
//		printf ("best = %d\n", best);
		for (int j = 0; j < best; j++)
			if (w[i+j] == ch)
				ans[i+j] = ch1;
			else
				ans[i+j] = ch;
	}
			
	for (int i = 0; i < ans.size (); i++)
		if (a[i])
			ans[i] = ans[i]-'a'+'A';
	cout << ans << endl;
	
	return 0;
}
