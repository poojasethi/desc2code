#include <cstdio>
#include <cstdlib>
#include <cctype>
#include <cstring>
#include <algorithm>
using namespace std;
bool match(const char* s1,const char* s2) {
    if(!*s2)return true;
    return (tolower(*s1)==tolower(*s2)&&match(s1+1,s2+1));
}
int main(void) {
    int n;
    scanf("%d",&n);
    char f[128][128],s[128],c[32];
    for(int i=0;i<n;++i)scanf("%s",f[i]);
    scanf("%s %s",s,c);
    *c=tolower(*c);
    bool v[128]={false};
    for(int i=0;s[i];++i)for(int j=0;j<n;++j)if(match(s+i,f[j]))for(int k=0;f[j][k];++k)v[i+k]=true;
    for(int i=0;s[i];++i)if(v[i]){
        char x=tolower(s[i]);
        x=(*c!=x)?*c:('a'==x)?'b':'a';
        s[i]=islower(s[i])?x:toupper(x);
    }
    puts(s);
    return 0;
}
