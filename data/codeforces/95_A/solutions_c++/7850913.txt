#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 110
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f[N],len[N];
char p[N][N],s[N],h[N];
inline bool pan(int x,int y) {
	if (x>='a') x=x-'a'+'A';
	if (y>='a') y=y-'a'+'A';
	return x==y;
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%s",p[i]+1),len[i]=strlen(p[i]+1);
	scanf("%s",s+1);
	m=strlen(s+1);
	For(i,1,m) {
		For(j,1,n) {
			int i1,i2;
			for (i1=i,i2=1;i1<=m&&i2<=len[j];) {
				if (!pan(s[i1],p[j][i2])) break;
				i1++,i2++;
			}
			if (i2>len[j]) f[i]++,f[i+len[j]]--;
		}
	}
	For(i,1,m) f[i]+=f[i-1];
	scanf("%s",h);
	For(i,1,m) if (f[i]) {
		int A='A';
		if (s[i]>='a') A='a';
		int B=s[i]-A;
		if (!pan(s[i],h[0])) s[i]=A+h[0]-'a';
		else if (B) s[i]=A;
		else s[i]=A+1;
	}
	printf("%s\n",s+1);
	return 0;
}
