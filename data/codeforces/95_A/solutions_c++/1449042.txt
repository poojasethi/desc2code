#include <cctype>
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

inline bool cmp(char a, char b) { return tolower(a) == tolower(b); }
int main() {
	int n;
	cin >> n;
	vector<string> sv;
	string s;
	while (n--) {
		cin >> s;
		sv.push_back(s);
	}
	char l, ul, ll;
	cin >> s >> l;
	string ans(s);
	ul = toupper(l), ll = tolower(l);
	for (int i = 0, S = s.size(); i < S; ++i) {
		for (int id = 0, size = sv.size(); id < size; ++id) {
			int ss = sv[id].size();
			if (i + ss > S)
				continue;
			if (equal(&s[i], &s[i + ss], sv[id].begin(), cmp)) {
				for (int k = i, kk = i + ss; k < kk; ++k)
					if (isupper(s[k])) {
						if (s[k] == ul) {
							if (ul == 'A') ans[k] = 'B';
							else ans[k] = 'A';
						} else ans[k] = ul;
					} else {
						if (s[k] == ll) {
							if (ll == 'a') ans[k] = 'b';
							else ans[k] = 'a';
						} else ans[k] = ll;
					}
			}
		}
	}
	cout << ans << endl;
	return 0;
}
