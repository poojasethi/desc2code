#include<string>
#include<iostream>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
    int n; cin>>n;
    string sub[100],w;
    char letter;
    rep(i,n){
        cin>>sub[i];
        rep(j,sub[i].length()) sub[i][j]=tolower(sub[i][j]);
    }
    cin>>w>>letter;

    bool upper[100]={};
    rep(i,w.length()) if(isupper(w[i])) {
        w[i]=tolower(w[i]);
        upper[i]=true;
    }

    bool change[100]={};
    rep(i,n){
        int slen=sub[i].length();
        rep(j,(int)w.length()-slen+1){
            if(sub[i]==w.substr(j,slen)) fill(change+j,change+j+slen,true);
        }
    }

    rep(i,w.length()){
        if(change[i]){
            if(w[i]!=letter) w[i]=letter;
            else             w[i]=(w[i]!='a'?'a':'b');
        }
        if(upper[i]) w[i]=toupper(w[i]);
    }
    cout<<w<<endl;

    return 0;
}
