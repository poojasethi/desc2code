#include<iostream>
#include<cctype>
#include<string>
#include<cstring>
#include<map>
using namespace std;
string s[104];
string w;
int A[103];
map <string, bool> my;
int main()
{
	memset(A, 0, sizeof A);
	int T;
	cin>>T;
	for(int i=0;i<T;i++)
	{
		cin>>s[i];
		string tmp;
		for(int j=0;j<s[i].size();j++)
		{
			tmp+=tolower(s[i][j]);
		}
		my[tmp] = true;
	}

	cin>>w;
	char ch;
	cin>>ch;
        int sz = w.size();

	for(int i=0;i<sz;i++)
	{
	 string k;
	for(int j=i; j<sz; j++)
	{
		k+=tolower(w[j]);
		if(my.count(k)!=0)
		{
			A[i] +=1;
			A[j+1] -=1;
		}
	}
	}
        int s = A[0];
        for(int i=1;i<102;i++)
	{
		s+=A[i];
		A[i] = s;
	}

	for(int i=0;i<sz;i++)
	{
		char tmp = w[i];
		if(A[i])
		{
			if(tolower(w[i]) == ch)
			{
				if(ch!='a')
				w[i]='a';
				else
			        w[i]='b';
			}
			else
				w[i] = ch;

			if(isupper(tmp))
				w[i] = toupper(w[i]);


		}
	}


	cout<<w<<endl;




	return 0;
}
