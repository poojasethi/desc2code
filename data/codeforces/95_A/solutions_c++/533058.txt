#include <string>
#include <iostream>
using namespace std;
string inp,res;
string list[100];
int n;
char l;
char nx;
string tol(string a)
{
	string res;
	for (int i=0;i<a.size();i++)
	{
		if (a[i]>='A' && a[i]<='Z')
			res+=a[i]-'A'+'a';
		else
			res+=a[i];
	}
	return res;
}
void replace(int a,int b)
{
	for (int i=0;i<b;i++)
	{
		char now;
		if (inp[a+i]==l || inp[a+i]==l-'a'+'A')
			now=nx;
		else
			now=l;
		if (inp[a+i]>='A' && inp[a+i]<='Z')
			res[a+i]=now-'a'+'A';
		else
			res[a+i]=now;
	}
}
int main()
{
	ios::sync_with_stdio(false);
	cin>>n;
	for (int i=0;i<n;i++)
		cin>>list[i];
	cin>>inp;
	cin>>l;
	for (char i='a';i<='z';i++) if (i!=l)
	{
		nx=i;
		break;
	}
	res=inp;
	for (int i=0;i<inp.size();i++)
		for (int j=0;j<n;j++) if (i<=inp.size()-list[j].size())
		{
			if (tol(inp.substr(i,list[j].size()))==tol(list[j]))
				replace(i,list[j].size());
		}
	cout<<res<<endl;
	return 0;
}
