#include<cstdio>
#include<string.h>
#include<cstring>
#include<iostream>
using namespace std;

char s[200][103];
char inp[3000];
char inp2[3000];
int chn[3000];
char cs(char s,char to){
	if(to >= 'a'){
		if(s >= 'a')
			return s;
		return s + 'a'-'A';
	}
	else{
		if(s >= 'a')
			return s +'A'-'a';
		return s;
	}
}
int main(){
	char ch;
	int i,n,j,k;
	cin >> n;
	for(i=0;i<n;i++){
		scanf(" %s",s[i]);
	}
	scanf(" %s",inp);
	strcpy(inp2,inp);
	cin >> ch;
	int nn = strlen(inp);
	for(i=0;i<n;i++){
		int len = strlen(s[i]);
		for(j=0;j<nn;j++){
			if(strncasecmp(s[i],inp+j,len) == 0){
				for(k=j;k<j + len;k++)
					chn[k] = 1;
			}
		}
	}
	for(i=0;i<nn;i++){
		if(chn[i]){
			if(tolower(inp[i]) != tolower(ch))
				inp[i] = cs(ch,inp[i]);
			else if(inp[i] == 'a' || inp[i] == 'A'){
				inp[i] = cs('b',inp[i]);
			}
			else
				inp[i] = cs('a',inp[i]);
		}
	}
	printf("%s\n",inp);
}
