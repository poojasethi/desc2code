#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
#include <cstring>
using namespace std;
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,q1,w,e,r,t;
  char c,v,a[100][101],s[101];
  bool d[101],f[101];
  cin>>q;
  for(w=0;w<q;w++){
   cin>>a[w];
   for(e=0;a[w][e];e++)
    if(a[w][e]<97)
     a[w][e]+=32;}
  cin>>s>>c;
  q1=strlen(s);
  bool b;
  for(w=0;w<q1;w++){
   f[w]=d[w]=0;
   if(s[w]<97){
    s[w]+=32;
    d[w]=1;}}
  for(w=0;w<q1;w++){
   for(e=0;e<q;e++){
    b=1;
    for(r=0;a[e][r];r++)
     if(s[w+r]!=a[e][r]){
      b=0;break;}
    if(b)
     for(r=0;a[e][r];r++)
      f[w+r]=1;}}
  for(v=97;v==c;v++);
  for(w=0;w<q1;w++){
   if(f[w])
    if(s[w]==c)
     s[w]=v;
    else
     s[w]=c;
   if(d[w])
    s[w]-=32;
   cout<<s[w];}
  return 0;}
