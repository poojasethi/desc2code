#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#include<cmath>
#include<vector>
#include<map>
#include<queue>
using namespace std;
typedef long long ll;
const int N=1e2+10;
char s[N][N],w[N],l;
bool cmp(char a,char b){return abs(a-b)%32;}
bool b[N];
int main()
{
    int n,i,j,k;
    cin>>n;memset(b,0,sizeof b);
    for(i=0;i<n;i++)scanf("%s",s[i]);
    scanf("%s\n%c",w,&l);
    if(l>='a')l-=32;
    for(i=0;i<strlen(w);i++)
        for(j=0;j<n;j++)
            if(strlen(w)-i<strlen(s[j]))continue;
            else
            {
                bool bo=1;
                for(k=0;k<strlen(s[j]);k++)
                    if(cmp(w[i+k],s[j][k])){bo=0;break;}
                if(bo)
                    for(k=0;k<strlen(s[j]);k++)b[i+k]=1;
            }
    for(i=0;i<strlen(w);i++)
        if(b[i])
            if(cmp(l,w[i]))w[i]=w[i]>='a'?l+32:l;
            else w[i]=(w[i]>='a'?'a':'A')+(l=='A');
    printf("%s",w);
    return 0;
}
