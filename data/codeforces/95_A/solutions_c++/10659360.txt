#include <bits/stdc++.h>
using namespace std;

int main() {
	char s[105][105] , t[105] , c;
	int n , k , j; int temp[105]={0};
	
	scanf("%d" , &n);
	
	for(int i=0 ; i<n ; i++) scanf("%s" , &s[i]);
		
	scanf("%s %c" , &t , &c);
	
	int l=strlen(t);
	
	for(int i=0 ; i<n ; i++){
		if(strlen(s[i])>l) continue;
		for( j=0 ; j<=l-strlen(s[i]) ; j++){
			if(tolower(s[i][0]) == tolower(t[j])){
				for( k=1 ; k<strlen(s[i]) ; k++){
					if(tolower(s[i][k]) != tolower(t[j+k])) break;
				}
			if(k == strlen(s[i])) {
				for(int h=j ; h<j+strlen(s[i]) ; h++) temp[h]=1;
			}
			}
		}
	}
;
	
	for(int i=0 ; i<l ; i++){
		if(temp[i]){
			if(tolower(t[i]) != tolower(c)) { if(t[i]<='Z') t[i]=c-'a'+'A'; else t[i]=c; }
			else {
				if(tolower(t[i]) == 'a'){ if(t[i]>='a') t[i]='b'; else t[i]='B'; }
				else { if(t[i]>='a') t[i]='a'; else t[i]='A'; }
			
			}
		}
	}
	printf("%s\n" , t);
	return 0;
}
