#include <cstdio>
#include <cstring>
#include <cmath>
#include <ctype.h>
#include <algorithm>

using namespace std;

#define MAXN 110
typedef long long LL;

bool vis[MAXN];
char w[MAXN],s[MAXN][MAXN];

int main()
{
    int i,j,k,n,a,b; char c;
    scanf("%d",&n);
    for(i=1;i<=n;++i) scanf("%s",s[i]);
    scanf("%s %c",w,&c);
    b=strlen(w);
    for(i=1;i<=n;++i){
        for(j=0,a=strlen(s[i]);j<b;++j){
            for(k=0;k<a&&j+k<b;++k)
                if(tolower(s[i][k])!=tolower(w[j+k])) break;
            if(k>=a){
                for(k=0;k<a;++k) vis[j+k]=true;
            }
        }
    }
    for(i=0;i<b;++i){
        if(!vis[i]) continue;
        if(tolower(w[i])!=w[i]) k=1;
        else k=0;
        w[i]=tolower(w[i]);
        if(w[i]==c){
            if(c=='a') w[i]='b';
            else w[i]='a';
        }
        else w[i]=c;
        if(k) w[i]=toupper(w[i]);
    }
    printf("%s",w);
    return 0;
}
