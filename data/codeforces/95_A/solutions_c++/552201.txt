#include <cstdio>
#include <cstring>
#include <cctype>
#include <cstdlib>

int n;
char r[110];
char s[110][110];
char c[2];
bool mk[110];

int main ()
{
	scanf("%d",&n);
	for(int i = 0; i < n; i ++ )
	{
		scanf("%s", s[i]);
	}
	memset(mk,0,sizeof(mk));
	scanf("%s", r);
	scanf("%1s", c);
	for(int i = 0; r[i]; i ++ )
	{
		for(int j = 0; j < n; j ++ )
		{
			bool flag = true;
			for(int k = 0; s[j][k]; k ++ )
				if( toupper(r[i+k]) != toupper(s[j][k]) ) { flag = false; break; }
			if( flag ) {
				for(int k = 0; s[j][k]; k ++ )
					mk[i+k] = true;
			}
		}
	}
	for(int i = 0; r[i]; i ++ ) if( mk[i] ) 
	{
		if( toupper(r[i]) == toupper(c[0]) ) {
			if( islower(r[i]) ) {
				r[i] = 'a'; 
			} else r[i] = 'A';
			while( toupper(r[i]) == toupper(c[0]) ) r[i]++;
		} else
		{
			if( islower(r[i]) ) r[i] = tolower(c[0]); else
				r[i] = toupper(c[0]);
		}
	}	
	printf("%s\n", r);
//	system("pause");
	return 0;
}
