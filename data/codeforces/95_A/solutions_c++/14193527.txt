#include<algorithm>  
#include<iostream>  
#include<stdio.h>
#include<string.h>
#include<stdlib.h>

using namespace std;


char str[205][205]={0};
char opt[205]={0};

long hash[205]={0};

int main()
{
	long n,i,j,k;
	char ch;
	scanf("%ld",&n);
	for(i=1;i<=n;i++)
		scanf("%s",str[i]);
	scanf("%s\n",opt);
	for(i=0;opt[i];i++)
	{
		for(j=1;j<=n;j++)
		{
			for(k=0;str[j][k]&&(str[j][k]==opt[i+k]||str[j][k]==(opt[i+k]^32));k++);
			if(!str[j][k])
			{
			for(k=0;str[j][k];k++)
				hash[i+k]=1;
			}
		}
	}
	scanf("%c",&ch);
	if(ch>='a')
		ch^=32;
	for(i=0;opt[i];i++)
	if(hash[i])
    {
		if(opt[i]!=ch&&opt[i]!=(ch^32))
			printf("%c",opt[i]>='a'?(ch^32):ch);
		else if(opt[i]>='a')
			printf("%c",opt[i]=='a'?'b':'a');
		else
			printf("%c",opt[i]=='A'?'B':'A');
    }
	else
		printf("%c",opt[i]);
 return 0;
}
