#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char low(char c){
	if (c <= 'Z') return c + 32;
	return c;
}

int n, m, same, l[105], cover[105];
char s[105][105], S[105], luck[5];

int main(){

	scanf("%d", &n);
	for (int i=0; i<n; i++){
		scanf("%s", s[i]);
		l[i] = strlen(s[i]);
	}
	
	scanf("%s", S);
	scanf("%s", luck);
	m = strlen(S);
	
	for (int i=0; i<m; i++){
		for (int j=0; j<n; j++){

			same = 1;
			for (int k=0; k<l[j]; k++){
				if (i+k >=m || low(S[i+k]) != low(s[j][k])) same = 0;
			}
			
			if (same){
				for (int k=0; k<l[j]; k++) cover[i+k] = 1;
			}
		}
	}
	
	for (int i=0; i<m; i++){
		if (!cover[i]) printf("%c", S[i]);
		else{
			if (low(S[i]) != luck[0]){
				if (S[i] <= 'Z') printf("%c", luck[0] - 32);
				else printf("%c", luck[0]);
			}
			else{
				char c = 'a';
				if (c == luck[0]) c++;
				if (S[i] <= 'Z') printf("%c", c - 32);
				else printf("%c", c);
			}
		}
	}
	
	printf("\n");

	return 0;
}
