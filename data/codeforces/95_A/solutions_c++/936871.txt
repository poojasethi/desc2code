#include <iostream>
using namespace std ;

const int maxn = 100 ;

int n ;
string s , ans ;
string list[maxn] ;
char l ;
char f = 'a' ;

string low (string a) {
    for (int i = 0 ; i < (int)a.size() ; ++i)
        if ('A' <= a[i] && a[i] <= 'Z')
            a[i] = a[i]-'A'+'a' ;
    return a ;
}

void replace (int a , int len) {
    for (int i = 0 ; i < len ; ++i) {
        char c = l ;
        if (low(string("")+s[a+i]) == string("")+l)
            c = f ;
        ans[i+a] = c ;
        if ('A' <= s[a+i] && s[a+i] <= 'Z')
            ans [i+a] = c-'a'+'A' ;
    }
}

int main() {
    cin >> n ;
    for (int i = 0 ; i < n ; ++i)
        cin >> list[i] ;
    cin >> s >> l ;
    if (l == 'a')
        f = 'b' ;
    ans = s ;
    int len = s.size() ;
    for (int i = 0 ; i < len ; ++i) {
        for (int j = 0 ; j < n ; ++j) {
            int l2 = list[j].size() ;
            if (i<=len-l2 && ( low(s.substr(i,l2)) == low(list[j]) ) )
                replace (i,l2) ;
        }
    }
    cout << ans << endl ;
    return 0 ;
}
