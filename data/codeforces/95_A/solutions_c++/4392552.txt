#include<iostream>
#include<string.h>
#include<string>
#include<vector>
#include<stdio.h>
using namespace std;
int main()
{
	int n,i,k=0,j;
	char c;
	vector < string > sub;
	string str;
	scanf("%d",&n);
	sub.resize(n);
	for(i=0; i<n; i++)
		cin >> sub[i];
	cin >> str >> c;
	int size=str.size();
	int arr[109]={};
	for(i=0; i<size; i++)
	{
		for(j=0; j<n; j++)
		{
			k=0;
			while(k<sub[j].size() && k<size && ((str[k+i]==sub[j][k]+'a'-'A')|| (str[k+i]==sub[j][k]+'A'-'a')|| (str[k+i]==sub[j][k])))k++;
			if(k==sub[j].size())
			{
				for(k=i; k<i+sub[j].size(); k++)
					arr[k]=1;
			}
		}
	}
	for(i=0; i<size; i++)
	{
		if(arr[i])
		{
			if(str[i]==c || str[i]==c+'a'-'A' || str[i]==c+'A'-'a')
			{
				if(str[i]=='a' || str[i]=='A')
					putchar(str[i]+1);
				else if(str[i]<='Z' && str[i]>='A')putchar('A');
				else putchar('a');
			}
			else
			{
				if(str[i]>='A' && str[i]<='Z')
					putchar(c+'A'-'a');
				else putchar(c);
			}
		}
		else putchar(str[i]);
	}
	putchar('\n');
	return 0;
}
