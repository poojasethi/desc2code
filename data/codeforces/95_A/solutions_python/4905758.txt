n = input()
s = [raw_input().lower() for i in xrange(0,n)]
w = raw_input()
replace = raw_input().lower()

md = w.lower()
bad = [False] * len(w)
for x in s:
	i = -1
	while 1:
		i = md.find(x, i+1)
		if i == -1:
			break
		for j in xrange(i, i + len(x)):
			bad[j] = True
def work(ch,mk):
	if mk:
		tr = replace
		if tr == ch.lower():
			if ch.lower() != 'a':
				tr = 'a'
			else :
				tr = 'b'
		if ch.isupper():
			return tr.upper()
		return tr
	return ch
print "".join(map(work,list(w),bad))
