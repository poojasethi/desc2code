#!/usr/bin/env python
# -*- coding: utf-8 -*-

n = int(raw_input())
s = list()
for _ in xrange(n):
    s.append(raw_input().lower())
w = raw_input()
luck = raw_input().lower()

cw = w.lower()
u = [False]*len(cw)
for es in s:
    i = -1
    while True:
        i = cw.find(es, i+1)
        if i == -1:
            break
        for j in xrange(i, i + len(es)):
            u[j] = True
w = list(w)
for i in xrange(len(w)):
    if u[i]:
        sym = luck
        if sym == w[i].lower():
            sym = 'a' if w[i].lower() != 'a' else 'b'
        w[i] = sym.upper() if w[i].isupper() else sym
print ''.join(w)
