import sys

def main():
	sys.stdin = open("input.txt", "r")
	sys.stdout = open("output.txt", "w")
	n = input()
	a = [0] + map(int, raw_input().split())
	r1, c1, r2, c2 = map(int, raw_input().split())
	f = lambda x, y : min(a[x : y + 1] if x <= y else a[y : x + 1]) + 1
	print min(abs(r1 - i) + abs(i - r2) + abs(c2 - min(c1, f(r1, i), f(i, r2))) for i in xrange(1, n + 1))
	
if __name__ == "__main__": main()
