import sys
sys.stdin=open("input.txt",'r')
sys.stdout=open("output.txt",'w')
n = input()
R = lambda : map(int,raw_input().split())
lent = R()
x1,y1,x2,y2=R()
x1 -=1
x2 -=1
lent = map(lambda x: x+1,lent)
ans = int(1e7)
for i in xrange(n):
	y = min(lent[min(x1,i,x2):max(x2,i,x1)+1])
	y = min(y,y1)
	ans = min(ans , abs(x1-i) + abs(x2-i) + abs(y - y2))
print ans
