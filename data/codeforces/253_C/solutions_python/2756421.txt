import sys
sys.stdin = open('input.txt', 'r')
sys.stdout = open('output.txt', 'w')
n = input()
a = [0] + map(int, raw_input().split())
f = lambda x, y : min(a[x : y + 1] if x < y else a[y : x + 1]) + 1
u, v, x, y = map(int, raw_input().split())
r = 10 ** 9
for i in range(1, n + 1):
	r = min(r, abs(u - i) + abs(i - x) + abs(y - min(f(u, i), f(i, x), v)))
print r
