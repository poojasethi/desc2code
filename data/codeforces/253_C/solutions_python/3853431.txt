import sys
sys.stdin = open("input.txt", "r")
sys.stdout = open("output.txt", "w")

r = lambda: map(int, raw_input().split())
n = input()
a = [0] + r()
r1, c1, r2, c2 = r()
def f(x, y):
    if x < y: return min(a[x:y + 1]) + 1
    else: return min(a[y:x + 1]) + 1
print min(abs(r1 - i) + abs(r2 - i) + abs(c2 - min(f(r1, i), f(i, r2), c1)) for i in xrange(1, n + 1))
