import sys
sys.stdin=open('input.txt','r')
sys.stdout=open('output.txt','w')
R=lambda:map(int,raw_input().split())
n=input()
a=[0]+R()
r1,c1,r2,c2=R()
t=10**9
def go(i,j):
  r,c=r1,c1
  d=1 if i>r else -1
  while r!=i:
    r+=d
    c=min(c,a[r]+1)
  d=1 if j>i else -1
  while r!=j:
    r+=d
    c=min(c,a[r]+1)
  d=1 if r2>j else -1
  while r!=r2:
    r+=d
    c=min(c,a[r]+1)
  return abs(i-r1)+abs(j-i)+abs(r2-j)+abs(c-c2)
for i in range(n):
  for j in range(n):
    t=min(t,go(i+1,j+1))
print t
