#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int n,s[105],sx,sy,tx,ty;
int f(int L,int R,int S,int E){
  return min(abs(S-L)+abs(L-R)+abs(R-E),abs(S-R)+abs(R-L)+abs(L-E));
}
int main(){
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
  scanf("%d",&n);
  for(int i=1;i<=n;i++) scanf("%d",&s[i]);
  scanf("%d %d %d %d",&sx,&sy,&tx,&ty);
  sy--,ty--;
  int ans=999999999;
  for(int i=1;i<=n;i++){
    int mini=sy;
    for(int j=i;j<=n;j++){
      mini=min(mini,s[j]);
      if(i<=min(sx,tx) && max(sx,tx)<=j)
      ans=min(ans,f(i,j,sx,tx)+abs(ty-mini));
    }
  }
  printf("%d\n",ans);
  return 0;
}
