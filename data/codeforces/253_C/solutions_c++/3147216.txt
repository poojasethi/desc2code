#include <iostream>
#include <string>
#include <cstring>
#include <algorithm>
#include <cstdio>
using namespace std;


int a[110];
int n;


int find_min(int r1,int r2,int tmin)
{
    int i,j;
    for(i=min(r1,r2);i<=max(r1,r2);i++)
    {
        tmin=min(tmin,a[i]);
    }
    return tmin;
}

int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int i,j;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }

    int r1,r2,c1,c2;
    cin>>r1>>c1>>r2>>c2;
    r1--,c1--,r2--,c2--;

    int ans=1000100;

    for(i=0;i<n;i++)
    {
        int x,y;
        x=find_min(r1,i,c1);
        y=find_min(i,r2,x);

        int temp=abs(r1-i)+abs(r2-i)+abs(y-c2);

        ans=min(ans,temp);
    }
    cout<<ans<<endl;
}
