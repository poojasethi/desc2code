#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

const int tope=100000;
int n;
int a[tope];

int absoluto(int x)
{
  if (x<0) return -x;
  return x;
}

int main()
{
  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>a[i];
  int r1,c1,r2,c2;
  cin>>r1>>c1>>r2>>c2;
  r1--;c1--;r2--;c2--;
  int minimo=1000000000;
  for (int i=0;i<n;i++) {
    int mov=absoluto(r1-i)+absoluto(r2-i);
    int inf=min(min(r1,i),r2);
    int sup=max(max(r1,i),r2);
    int c=c1;
    for (int k=inf;k<=sup;k++)
      c=min(c,a[k]);
    mov+=absoluto(c2-c);
    minimo=min(minimo,mov);
  }
  cout<<minimo<<endl;
}
