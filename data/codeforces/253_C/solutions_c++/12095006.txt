#include<bits/stdc++.h>

using namespace std;

int n, a[111], r, c, r2, c2, x, y, res, ans = 1e6;

inline void f(int &y, int r) {
	if (y > a[r] + 1)
		y = a[r] + 1;
}

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i)
		scanf("%d", a + i);
	scanf("%d%d%d%d", &r, &c, &r2, &c2);
	for (int j = 1; j <= n; ++j) {
		x = r, y = c, res = 0;
		while (x < j) {
			++res, ++x;
			f(y, x);
		}
		while (x > j) {
			++res, --x;
			f(y, x);
		}
		while (x < r2) {
			++res, ++x;
			f(y, x);
		}
		while (x > r2) {
			++res, --x;
			f(y, x);
		}
		res += abs(y - c2);
		ans = min(ans, res);
	}
	printf("%d", ans);
	return 0;
}
