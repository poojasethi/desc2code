#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,a[100010];
int dp[100010];
int dd[100010];
int main()
{
    freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    int r1,c1,r2,c2;
    scanf("%d %d %d %d",&r1,&c1,&r2,&c2);
    dp[r1]=a[r1]+1;
    for(int i=r1-1;i>=1;i--)dp[i]=min(dp[i+1],a[i]+1);
    for(int i=r1+1;i<=n;i++)dp[i]=min(dp[i-1],a[i]+1);
    dd[r2]=a[r2]+1;
    for(int i=r2-1;i>=1;i--)dd[i]=min(dd[i+1],a[i]+1);
    for(int i=r2+1;i<=n;i++)dd[i]=min(dd[i-1],a[i]+1);
    int ans=1<<30;
    for(int i=1;i<=n;i++)ans=min(ans,abs(r1-i)+abs(r2-i)+abs(c2-min(dd[i],min(c1,dp[i]))));
    printf("%d",ans);
    return 0;
}
