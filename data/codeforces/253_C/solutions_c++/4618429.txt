#include <bits/stdc++.h>

using namespace std;

int n,bx,by,sx,sy;
int R[105];

void read()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int i;
	cin >> n ;
	for(i=1 ; i<=n ; i++)
	{
		cin >> R[i];
		R[i]++;
	}
	cin >> bx >> by >> sx >> sy ;
	R[bx]=by;
}

void solve()
{
	int k,i,minn=INT_MAX,costx,costy,b,s;
	int *mn;
	for(i=1 ; i<=n ; i++)
	{
		b=min(min(i,bx),sx);
		s=max(max(i,bx),sx);
		mn=min_element(R+b,R+s+1);
		costx=abs(sx-i)+abs(bx-i);
		costy=abs((*mn)-sy);
		minn=min(minn,costx+costy);
	}
	cout << minn << endl ;
}

int main()
{
	read();
	solve();
}
