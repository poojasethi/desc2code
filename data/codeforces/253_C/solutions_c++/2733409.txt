#include<iostream>
#include<string>
#include<stdio.h>
#include<algorithm>
#include<vector>

using namespace std;

long long n,i,j,l,ma,res=200000,k,ans;
long long a[200],kop[200][200],c1,r1,c2,r2;

main()
 {
  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
  cin>>n;
  for (i=1;i<=n;i++)
   cin>>a[i];
  for (j=1;j<=n;j++)
   for (l=j;l<=n;l++)
    {
     ma=2000000;
     for (i=j;i<=l;i++)
      ma=min(ma,a[i]);
     kop[j][l]=ma;
     kop[l][j]=ma;
    }
  cin>>c1>>r1>>c2>>r2;
  r1--;r2--;
  for (i=1;i<=n;i++)
   {
    ans=abs(c1-i)+abs(c2-i);
    k=min(r1,kop[c1][i]);
    k=min(k,kop[i][c2]);
    res=min(res,ans+abs(r2-k));
   }
  cout<<res<<endl;
 }
