#include <cstdio>
#include <algorithm>

#define ABS(a) ((a) < 0 ? -(a) : (a))

using namespace std;

int main()
{
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    int n, a[101], r1, c1, r2, c2, ans = 1 << 30;
    scanf("%d", &n);
    for(int i = 1;i <= n;++i)
    {
        scanf("%d", a + i);
        a[i]++;
    }
    scanf("%d%d%d%d", &r1, &c1, &r2, &c2);
    for(int i = 1;i <= n;++i)
    {
        int pos = min(c1, *min_element(a + min(i, min(r1, r2)), a + max(i, max(r1, r2)) + 1));
        ans = min(ans, ABS(r1 - i) + ABS(i - r2) + ABS(pos - c2));
    }
    printf("%d\n", ans);
    return 0;
}
