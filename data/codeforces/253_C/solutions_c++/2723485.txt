#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstdio>
#include <string>
using namespace std;
int r1,c1,r2,c2,n,a[111],ans=11111111;
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    cin>>n;
    for (int i=1;i<=n;i++)
        cin>>a[i];
    cin>>r1>>c1>>r2>>c2;
    for (int i=1;i<=n;i++){
        int kol=abs(i-r1);
        int cur=c1;
        for (int j=min(r1,i);j<=max(r1,i);j++)
            cur=min(cur,a[j]+1);
        for (int j=min(r2,i);j<=max(r2,i);j++)
            cur=min(cur,a[j]+1);
        kol+=abs(i-r2)+abs(cur-c2);
        if (kol<ans)ans=kol;
    }
    cout<<ans;
    return 0;
}
