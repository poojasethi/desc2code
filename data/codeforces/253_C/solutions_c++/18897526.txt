#include<bits/stdc++.h>
#define mp make_pair
#define f first
#define s second
using namespace std;
const int N=105;
int a[N],n,r1,c1,r2,c2,m,mm=0x3f3f3f3f;
int main()
{

  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
   int i,j;
   cin >> n;
   for( i = 0 ; i < n ;i++ )cin >> a[i];
   cin >> r1 >> c1 >> r2 >>c2;
   r1--,r2--;
   if(r1<r2)swap(r2,r1);
   for(i = 0 ; i < n ; i++ )
   {
         m = c1;
         for(j = min(r2,i) ; j <= max(r1,i) ; j++ )
         {
             m = min(a[j]+1,m);
         } 
         m = c2 - m;
         mm = min( mm, r1-r2+ abs(m) + (i<r2?2*(r2-i):0 )+ (i > r1 ? 2*(i-r1) : 0));
   }
   cout<<mm;
   return 0; 
}
