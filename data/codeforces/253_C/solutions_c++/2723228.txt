#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <vector>
#include <set>
#include <queue>

using namespace std;

int n,a[101],r1,r2,c1,c2;

int go(int s,int c,int t)
{
    while (s!=t)
    {
        s += s>t ? -1 : 1;
        c = min(c,a[s]);
    }
    return c;
}

int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for (int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        a[i]++;
    }
    scanf("%d%d%d%d",&r1,&c1,&r2,&c2);
    int ans=100000000;
    for (int i=1;i<=n;i++)
    {
        int t=abs(go(i,go(r1,c1,i),r2)-c2) + abs(r1-i) + abs(r2-i);
        ans = min(ans,t);
    }
    printf("%d\n",ans);
    return 0;
}
