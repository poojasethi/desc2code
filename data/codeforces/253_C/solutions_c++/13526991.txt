#include<bits/stdc++.h>
using namespace std;
int n,x,y,x1,y2,inp[200];
int main()
{
freopen("input.txt","r",stdin);
freopen ("output.txt","w",stdout);
    cin >> n;
    for(int i=1;i<=n;i++)cin >> inp[i],inp[i]++;
    cin >> x >> y >> x1 >> y2;
    int ans = INT_MAX;
    for(int i=1;i<=n;i++)
    {
        int con=y;
        for(int j=i;j<=n;j++)
        {
            con = min(con,inp[j]);
            if(i<=min(x1,x)&&j>=max(x1,x))
            {
                ans = min(ans,min(j-i+abs(x1-j)+abs(x-i)+abs(con-y2),j-i+abs(x1-i)+abs(x-j)+abs(con-y2)));
            }
        }
    }
    cout << ans << endl;
}
