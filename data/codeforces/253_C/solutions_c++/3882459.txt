#include<iostream>
#include<algorithm>
#include<utility>
#include<queue>
#include<cstdio>
#include<cstring>
using namespace std;
int dx[]={0,1,0,-1};
int dy[]={1,0,-1,0};
int d[110][100010],a[110];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int i,n,sx,sy,fx,fy;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		a[i]++;
	}
	cin>>sx>>sy>>fx>>fy;
	memset(d,-1,sizeof(d));
	d[sx][sy]=0;
	queue<pair<int,int> >q;
	q.push(make_pair(sx,sy));
	while (!q.empty())
	{
		pair<int,int> u=q.front();
		q.pop();
		int x=u.first,y=u.second;
		for(i=0;i<4;i++)
		{
			int x1=x+dx[i],y1=a[x1]<y+dy[i]?a[x1]:y+dy[i];
			if(x1 && y1 && x1<=n && y1<=a[x1] && d[x1][y1]<0)
			{
				d[x1][y1]=d[x][y]+1;
				q.push(make_pair(x1,y1));
			}
		}
	}
	cout<<d[fx][fy]<<endl;
}
