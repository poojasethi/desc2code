#include<bits/stdc++.h>
using namespace std;
#define ll  int
#define pi pair<ll,ll>
#define f first
#define s second
#define mp make_pair
ll a[105+1];
void solve(ll r1,ll r2,ll &mi){
    ll d=1;
    if(r1>r2) d=-1;
while(r1!=r2){
    mi=min(mi,a[r1]);
    r1+=d;
    mi=min(mi,a[r1]);
}   
}
int main(){
ll n;
freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
cin>>n;

for(int i=1;i<=n;i++) {cin>>a[i];
a[i]++;
}
ll r1,c1,r2,c2;
cin>>r1>>c1>>r2>>c2;
ll mi;
ll ans=INT_MAX;
for(int i=1;i<=n;i++){
mi=c1;
solve(r1,i,mi);
solve(r2,i,mi);
ans=min(ans,abs(i-r2)+abs(i-r1)+abs(mi-c2));
}
cout<<ans;
}
