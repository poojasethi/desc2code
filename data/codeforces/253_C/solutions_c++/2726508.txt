#include <vector>
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;

int N;
int v[102];
int X1, Y1, X2, Y2;

int dest(int fromL, int fromC, int to) {
	int c = fromC;
	for (int i = min(fromL, to); i <= max(fromL, to); i++) c = min(c, v[i]);
	return c;
}

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	
	scanf("%d", &N);
	for (int i = 0; i < N; i++) scanf("%d", &v[i]);
	scanf("%d %d %d %d", &X1, &Y1, &X2, &Y2);
	X1--; X2--; Y1--; Y2--;
	int res = 1000000;
	for (int i = 0; i < N; i++) {
		int aux = 0;
		int c1 = dest(X1, Y1, i);
		int c2 = dest(i, c1, X2);
		aux = abs(i - X1) + abs(X2 - i);
		aux += abs(c2 - Y2);
		res = min(res, aux);
	}
	printf("%d\n", res);
	return 0;
}
