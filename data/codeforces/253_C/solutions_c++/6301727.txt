#include <algorithm>
#include <stdio.h>
#include <memory.h>

using namespace std;
#define NN 200

typedef long long int64;
typedef pair<int, int> pii;

int a[NN];

int go(int &r, int &c, int tr) {
	int ret=0;
	while (r!=tr) {
		if (r<tr) r++; else r--;
		ret++;
		c=min(c, a[r]+1);
	}
	return ret;
}

int main() {
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);

	int n, sr, sc, tr, tc, ans, i, r, c, tmp;
	scanf("%d", &n);
	for (i=1; i<=n; i++) scanf("%d", a+i);
	scanf("%d%d%d%d", &sr, &sc, &tr, &tc);
	ans=1000000000;
	for (i=1; i<=n; i++) {
		r=sr; c=sc;
		tmp=go(r, c, i);
		tmp+=go(r, c, tr);
		ans=min(ans, tmp+abs(tc-c));
	}
	printf("%d\n", ans);
	
	return 0;
}
