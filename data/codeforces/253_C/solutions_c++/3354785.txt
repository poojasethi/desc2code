#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int inf = 111111111;
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,x1,x2,y1,y2,s,t,a[105],st,ans;
    while(scanf("%d",&n)!=EOF)
    {
        for(int i = 1; i <= n; i++) scanf("%d",&a[i]);
        scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
        ans = inf;
        for(int i = 1; i <= n; i++)
        {
            s = min(i,min(x1,x2));
            t = max(i,max(x1,x2));
            st = y1;
            for(int j = s; j <= t; j++)
                st = min(st,a[j]+1);
            ans = min(ans,abs(i-x1)+abs(i-x2)+abs(st-y2));
        }
        printf("%d\n",ans);
    }
    return 0;
}
