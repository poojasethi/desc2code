#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#define foru(i,a,b) for(int i=a;i<=b;++i)

using namespace std;
int n,sx,sy,tx,ty,gmin[210][210],ans;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&n);
	foru(i,1,n) scanf("%d",&gmin[i][i]);
	scanf("%d%d%d%d",&sx,&sy,&tx,&ty);
	foru(i,1,n) ++gmin[i][i];
	foru(i,1,n)
	  foru(j,i+1,n) gmin[i][j]=min(gmin[i][j-1],gmin[j][j]),gmin[j][i]=gmin[i][j];
	ans=210000000;
	foru(i,1,n)
	  ans=min(ans,abs(sx-i)+abs(tx-i)+abs(ty-min(sy,min(gmin[sx][i],gmin[tx][i]))));
	printf("%d\n",ans);
	return 0;
}
