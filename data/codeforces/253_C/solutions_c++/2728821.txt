#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;

int n,a[1000],tmp,ans,r1,r2,c1,c2;

int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for (int i=1;i<=n;i++) {scanf("%d",&a[i]);a[i]++;}
    scanf("%d%d%d%d",&r1,&c1,&r2,&c2);ans=99999999;
    for (int i=1;i<=n;i++){
        tmp=c1;
        for (int j=min(r1,i);j<=max(r1,i);j++) tmp=min(tmp,a[j]);
        for (int j=min(r2,i);j<=max(r2,i);j++) tmp=min(tmp,a[j]);
        ans=min(ans,abs(r1-i)+abs(r2-i)+abs(tmp-c2));
    }
    cout<<ans<<endl;
    return 0;
}
