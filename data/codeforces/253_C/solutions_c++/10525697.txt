#include <iostream>
#include <fstream>
#include <vector>
#include <algorithm>
using namespace std;
ifstream fin("input.txt");
ofstream fout("output.txt");
int main(){
	int n;
	fin >> n;
	vector<int> arr(n+1);
	for(int i = 1; i <= n; ++i)
		fin >> arr[i];
	int r1,c1,r2,c2;
	fin >> r1 >> c1 >> r2 >> c2;
	int ans = 2e9;
	for(int i = 1; i <= n; ++i){
		int rmin = min(min(r1,r2),i);
		int rmax = max(max(r1,r2),i);
		int cur = 0;
		int c = c1;
		for(int j = rmin; j <= rmax; ++j)
			c=min(c,arr[j]+1);
		if(c>c2)
			cur+=c-c2;
		else
			cur-=c-c2;
		if(i>r1)
			cur+=i-r1;
		else
			cur-=i-r1;
		if(i>r2)
			cur+=i-r2;
		else
			cur-=i-r2;
		ans=min(ans,cur);
	}
	fout << ans << '\n';
}
