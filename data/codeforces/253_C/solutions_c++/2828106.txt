# include <stdio.h>
# include <stdlib.h>

FILE * fin = fopen("input.txt","r");
FILE * fout = fopen("output.txt","w");

int n;

int ans=9999999;

int n1[150];

int a,b,c,d;

int main()
{
	fscanf(fin,"%d",&n);

	for (int i=1;i<=n;i++) fscanf(fin,"%d",&n1[i]);

	fscanf(fin,"%d %d %d %d",&a,&b,&c,&d);

	for (int i=1;i<=n;i++)
	{int k=b,q=0;

		if(i < a)
		for (int j=a;j>=i;j--) if(n1[j]+1 < k) k=n1[j]+1;
		if(i > a)
		for (int j=a;j<=i;j++) if(n1[j]+1 < k) k=n1[j]+1;

		q+=abs(a-i);

		if(i < c)
		for (int j=i;j<=c;j++) if(n1[j]+1 < k) k=n1[j]+1;
		if(i > c)
		for (int j=i;j>=c;j--) if(n1[j]+1 < k) k=n1[j]+1;

		q+=abs(c-i);

		//printf("%d -> %d",i,k);getchar();

		if(q+abs(d-k) < ans) ans=q+abs(d-k);
	}

	fprintf(fout,"%d\n",ans);

	fclose(fin);fclose(fout);
}
