#include<stdio.h>
int min(int a,int b){return a<b?a:b;}
int abs(int a){return a>0?a:-a;}
int c[110],r[110];
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int i,n,c1,c2,r1,r2,as;
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&r[i]);
	scanf("%d%d%d%d",&r1,&c1,&r2,&c2);
	c[r1]=c1;
	for(i=r1-1;i>=1;i--)c[i]=min(c[i+1],r[i]+1);
	for(i=r1+1;i<=n;i++)c[i]=min(c[i-1],r[i]+1);
	as=abs(r1-r2)+abs(c[r2]-c2);
	if(c[r2]<=c2){printf("%d\n",as);return 0;}
	for(i=1;i<=n;i++)as=min(as,abs(r1-i)+abs(r2-i)+abs(c[i]-c2));
	printf("%d\n",as);
	return 0;
}
