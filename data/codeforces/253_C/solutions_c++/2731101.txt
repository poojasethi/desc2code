#include <cstdio>
#include <algorithm>
using namespace std;

int ps[128];


int main()
{
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);

    int N;
    scanf("%d", &N);
    for(int i = 1; i <= N; i++)
        scanf("%d", ps + i), ps[i]++;

    int r1, c1, r2, c2;
    scanf("%d%d%d%d", &r1, &c1, &r2, &c2);

    
    int ans = 2000000000;
    for(int i = 1; i <= N; i++)
    {
        int s = min(i, min(r1, r2));
        int t = max(i, max(r1, r2));

        ans = min(ans, abs(c2 - min(c1, *min_element(ps + s, ps + t + 1))) + abs(i - r1) + abs(i - r2));
    }

    printf("%d\n", ans);
        


    


    return 0;
}
