#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
const int  Maxn=20000;
int n,i,x1,y1,x2,y2,sum,ans,a[Maxn];
int Abs(int a)
{
return a>0? a:-a;
}
int cnt(int x1,int y1,int x2)
{
sum+=max(x1,x2)-min(x1,x2);
if (x1<x2)
for (int i=x1+1;i<=x2;i++)
y1=min(y1,a[i]+1);

if (x1>x2)
for (int i=x1-1;i>=x2;i--)
y1=min(y1,a[i]+1);
return y1;
}
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
cin>>n;
for (i=1;i<=n;i++) cin>>a[i];
cin>>x1>>y1>>x2>>y2;
ans=99999999;
for (i=1;i<=n;i++)
{
sum=0;
int p=y1;
p=cnt(x1,p,i);
p=cnt(i,p,x2);
sum+=Abs(y2-p);
ans=min(ans,sum);
} 
cout<<ans<<endl;
}
