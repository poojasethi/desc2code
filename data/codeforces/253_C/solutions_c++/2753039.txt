#include<cstdio>
#include<algorithm>
#include<climits>
#include<cstdlib>
#define NN 100
using namespace std;
int a[NN];

int mv_r( int r1 , int c1 , int r2 , int& fc ){
	int ans = abs(r2-r1);
	if( r1 > r2){
		for( ;r1 > r2; --r1 )
			if( a[r1-1]+1 < c1 )
				c1 = a[r1-1]+1;

	}else{
		for( ;r1 < r2; ++r1 )
			if( a[r1+1]+1 < c1 )
				c1 = a[r1+1]+1;

	}
	fc = c1;
	return ans ; 
}
int main(){
	freopen("input.txt","r" ,stdin);
	freopen("output.txt","w" ,stdout);
	
	int n ; scanf("%d", &n );
	for( int i = 0 ; i < n ; ++i)
		scanf("%d", a+i);
	int r1 , c1 , r2 , c2;
	scanf("%d%d%d%d", &r1 ,&c1 ,&r2 , &c2 ); --r1,--r2;
	int ans = INT_MAX;
	for( int i = 0 , ic , fc; i < n; ++i ){
		ans = min( ans,  mv_r(r1,c1,i,ic) + mv_r(i,ic,r2,fc) + abs(fc-c2) );
	}
	printf("%d\n", ans);
	return 0;
}
