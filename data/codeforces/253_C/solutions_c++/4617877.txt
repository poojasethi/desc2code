#include <bits/stdc++.h>
#define dosya(s1,s2) freopen(s1".txt","r",stdin); freopen(s2".txt","w",stdout); 
#define FP( ii,aa,bb ) for( int ii = aa; ii <= bb; ii++ )
#define FM( ii,aa,bb ) for( int ii = aa; ii >= bb; ii-- )
#define st first
#define nd second

using namespace std;

typedef pair<int,int> pii;
typedef long long ll; 

int N,D[105];
int basx,basy,sonx,sony;
int mn,res,x,y,t;

int main()
{
	dosya("input","output");

	cin >> N;
	FP( i,1,N )
	{
		cin >> D[i];
		D[i]++;
	}

	cin >> basx >> basy >> sonx >> sony;


	res = (int(1e9));

	FP( i,1,N )
	{
		t = abs(basx-i) + abs(sonx-i);
	
		x = min(i,min(basx,sonx));
		y = max(i,max(basx,sonx)); 

		mn = basy;
		FP( i,x,y )
			mn = min( mn , D[i] );

		res = min( res , t + abs(mn-sony) );
	}

	cout << res << endl;
}
