#include <cstdio>
#include <cmath>
#include <cstdlib>

int n,f[105],i,j,k,r1,r2,c1,c2,x,ans;

int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for(i=1;i<=n;i++)
        scanf("%d",&f[i]);
    scanf("%d%d%d%d",&r1,&c1,&r2,&c2);
    ans=100000000;
    for(i=1;i<=n;i++)
    {
        x=c1;
        for(j=i;j<=n;j++)
        {
            if(f[j]+1<x)x=f[j]+1;
            if(r1<i || r1>j || r2<i || r2>j)continue;
            if(ans>(j-i)*2-abs(r1-r2)+abs(x-c2))
            {
                ans=(j-i)*2-abs(r1-r2)+abs(x-c2);
            }
        }
    }
    printf("%d\n",ans);
    return 0;
}
