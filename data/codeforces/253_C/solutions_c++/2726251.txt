#include <cstdio>
#include <cstring>
#define maxn 102
using namespace std;
int n, a[maxn],r1,c1,r2,c2,res,min[maxn][maxn];
int abs(int x){ if (x<0) return (-x); return x;}
int getmin(int a,int b){if (a<b) return a; return b;}
void process(){
	int i,j;
	for (i=1;i<=n;i++)	min[i][i]=a[i];
	for (i=1;i<=n;i++) for (j=i+1;j<=n;j++) {
		min[i][j]=getmin(min[i][j-1],min[j][j]);
		min[j][i]=min[i][j];
	}
	res=1000000000;
	for (int r=1;r<=n;r++){
		res=getmin(res,abs(r1-r)+abs(r2-r)+abs(c2-getmin(c1,getmin(min[r1][r],min[r][r2]))));
	}
}
int main(){
	freopen("input.txt","r",stdin);		freopen("output.txt","w",stdout);
	scanf("%d",&n); for (int i=1;i<=n;i++){scanf("%d",&a[i]);a[i]++;}	scanf("%d%d%d%d",&r1,&c1,&r2,&c2);
	process();
	printf("%d",res);
	return 0;
}
