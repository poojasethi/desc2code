#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstdlib>
#include <cmath>

using namespace std;

const int MAXN = 200010;

int a[MAXN];

int main() {
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    int r1, c1, r2, c2;
    cin >> r1 >> c1 >> r2 >> c2;
    r1--;
    r2--;
    int ans = MAXN;
    for (int i = 0; i < n; i++) {
        int minn = c1;
        for (int j = min(i, min(r1, r2)); j <= max(i, max(r1, r2)); j++) {
            minn = min(minn, a[j] + 1);
        }
        ans = min(ans, abs(i - r1) + abs(i - r2) + abs(c2 - minn));
    }
    cout << ans;
    fclose(stdout);
}
