#include <fstream>
#include <cmath>
#include <algorithm>

using namespace std;

ifstream fin("input.txt");
ofstream fout("output.txt");

int n, a[110], r1, c1, r2, c2;

int main(){
    fin >> n;
    for (int i = 0; i < n; i++) fin >> a[i];
    fin >> r1 >> c1 >> r2 >> c2;
    r1--; c1--; r2--; c2--;

    int ans = -1;
    a[r1] = c1;
    for (int i = r1 + 1; i < n; i++)
        a[i] = min(a[i], a[i - 1]);
    for (int i = r1 - 1; i >= 0; i--)
        a[i] = min(a[i], a[i + 1]);

    for (int i = 0; i < n; i++)
        if (ans == -1 ||
            ans > abs(i - r1) + abs(r2 - i) + abs(min(a[i], a[r2]) - c2))
            ans = abs(i - r1) + abs(r2 - i) + abs(min(a[i], a[r2]) - c2);

    fout << ans;
}
