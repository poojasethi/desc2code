n, m, k = map(int, raw_input().split())
a = map(int, raw_input().split())
b = [map(int, raw_input().split()) for _ in range(m)]
aa = [0] * (m + 1)

for _ in range(k):
    x, y = map(int, raw_input().split())
    aa[x-1] += 1
    aa[y] -= 1
for i in range(m):
    aa[i+1] += aa[i]

bb = [0] * (n + 1)

for i in range(m):
    l, r, d = b[i]
    d *= aa[i]
    bb[l-1] += d
    bb[r] -= d
    
for i in range(n):
    bb[i+1] += bb[i]
print ' '.join(map(str, [a[i] + bb[i] for i in range(n)]))
