R = lambda:map(int, raw_input().split())
n,m,k=R()
seq=R()
opera = [0]*(m+1)
lrd=[R() for i in range(m)]
query = [0]*(n+1)
xy= [R() for i in range(k)]
for (x,y) in xy:
    opera[x-1]+=1
    opera[y]-=1
temp,i=0,0
for (l,r,d) in lrd:
    temp+=opera[i]
    query[l-1]+=temp*d
    query[r]-=temp*d
    i+=1;
temp=0
for i in range(n):
    temp+=query[i]
    seq[i]+=temp
print ' '.join(map(str, seq))
