read = lambda: map(int, raw_input().split())
n, m, k = read()
a = read()
b = [read() for _ in xrange(m)]
s, f = [0] * (m + 1), [0] * (n + 1)
for x, y in [read() for _ in xrange(k)]:
    s[x - 1] += 1
    s[y] -= 1 
c = 0
for i in xrange(m):
    x, y, z = b[i]
    c += s[i]
    f[x - 1] += z * c
    f[y] -= z * c
x = 0
for i in xrange(n):
    x += f[i]
