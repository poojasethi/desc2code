n, m, k = map(int, raw_input().split())
a = map(int, raw_input().split())
b = [map(int, raw_input().split()) for _ in range(m)]
A = [0] * (m + 1)
for _ in range(k):
	x, y = map(int, raw_input().split())
	A[x - 1] += 1
	A[y] -= 1
for i in range(m):
	A[i + 1] += A[i]
B = [0] * (n + 1)
for i in range(m):
	l, r, d = b[i]
	d *= A[i]
	B[l - 1] += d
	B[r] -= d
for i in range(n):
	B[i + 1] += B[i]
print ' '.join(map(str, [a[i] + B[i] for i in range(n)]))
