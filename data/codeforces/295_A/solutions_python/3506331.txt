inp = raw_input().split()

n = int(inp[0])
m = int(inp[1])
k = int(inp[2])

a = [int(inp_i) for inp_i in raw_input().split()]

Op = []

for i in range(m):
	Op += [[int(inp_i) for inp_i in raw_input().split()]]

times_diff = [0 for i in range(m+1)]

for i in range(k):
	inp = raw_input().split()
	x = int(inp[0])
	y = int(inp[1])
	times_diff[x-1] += 1
	times_diff[y] -= 1

diff_arr = [0 for i in range(n+1)]
cur_times = 0

for i in range(m):
	cur_times += times_diff[i]
	diff_arr[Op[i][0]-1] += cur_times*Op[i][2]
	diff_arr[Op[i][1]] -= cur_times*Op[i][2]

cur_diff = 0

for i in range(n):
	cur_diff += diff_arr[i]
	print a[i] + cur_diff,
