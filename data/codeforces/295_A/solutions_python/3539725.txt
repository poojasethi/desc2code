R = lambda:map(int, raw_input().split())
n, m, k = R()
a = R()
lrd = [R() for i in range(0,m)]
xy = [R() for i in range(0,k)]
opers = [0 for i in range(0,m+1)]
adds = [0 for i in range(0,n+1)]
for (x, y) in xy:
    opers[x - 1] += 1
    opers[y] -= 1
t, i = 0, 0
for (l,r,d) in lrd:
    t += opers[i]
    adds[l-1] += d * t
    adds[r] -= d * t
    i += 1
d = 0
for i in range(0,n):
    d += adds[i]
    a[i] += d
print " ".join(map(str, a))
