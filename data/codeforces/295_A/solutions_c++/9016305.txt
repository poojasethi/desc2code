#include <iostream>
#define MAX 100005
int K,M,N;
long long A[MAX],Aux[MAX],Op[MAX],D[MAX],L[MAX],R[MAX],x,y;
using namespace std;
int main()
{
  cin>>N>>M>>K;
  for(int i=1;i<=N;i++) cin>>A[i];
  for(int i=1;i<=M;i++) cin>>L[i]>>R[i]>>D[i];
  for(int i=1;i<=K;i++) cin>>x>>y,Op[x]++,Op[y+1]--;
  for(int i=1;i<=M;i++) Op[i]+=Op[i-1],Aux[L[i]]+=D[i]*Op[i],Aux[R[i]+1]-=D[i]*Op[i];
  for(int i=1;i<=N;i++) Aux[i]+=Aux[i-1],cout<<Aux[i]+A[i]<<" ";
}
