#include <iostream>

using namespace std;
long long int c[100100],a[100100],s[100100],l[100100],r[100100],d[100100],cnt,cur,x,y,n,m,k;
int main(){
    cin>>n>>m>>k;
    for(int i=1;i<=n;i++)cin>>a[i];
    
    for(int i=1;i<=m;i++)cin>>l[i]>>r[i]>>d[i];
    
    for(int i=1;i<=k;i++){
        cin>>x>>y;
        s[x]++;
        s[y+1]--;
    }
    for(int i=1;i<=m;i++){
        cnt+=s[i];
        c[l[i]]+=cnt*d[i];
        c[r[i]+1]-=cnt*d[i];
    }
    for(int i=1;i<=n;i++){
        cur+=c[i];
        a[i]+=cur;
    }
    for(int i=1;i<=n;i++)cout<<a[i]<<" ";
    
    return 0;
}
