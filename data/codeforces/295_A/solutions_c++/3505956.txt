#include <cstdio>
#define LL long long
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

int n,m,k,l,r,q[100010][3];
LL num,delta,a[100010],b[100010],c[100010];

int main() {
	scanf("%d%d%d",&n,&m,&k);
	REP(i,n) scanf("%I64d",&a[i]);
	REP(i,m) scanf("%d%d%d",&q[i][0],&q[i][1],&q[i][2]);
	REP(i,k) {
		scanf("%d%d",&l,&r);
		++c[l]; --c[r+1];
	}
	REP(i,m) {
		num+=c[i];
		b[q[i][0]]+=num*(LL)q[i][2];
		b[q[i][1]+1]-=num*(LL)q[i][2];
	}
	REP(i,n) {
		delta+=b[i];
		printf("%I64d ",a[i]+delta);
	}
	return 0;
}
