#include<cstdio>

static long long v[100005], ww[100005], w[100005], ss[100005], s[100005], l[100005], r[100005], d[100005];

int main(void)
{
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);

	for(int i=1;i<=n;i++)
		scanf("%I64d",v+i);

	for(int i=1;i<=m;i++)
		scanf("%I64d%I64d%I64d",l+i,r+i,d+i);

	for(int i=0;i<k;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		ww[x]++;
		ww[y+1]--;
	}

	for(int i=1;i<=m;i++)
		w[i]=ww[i]+w[i-1];

	for(int i=1;i<=m;i++){
		ss[l[i]]+=w[i]*d[i];
		ss[r[i]+1]-=w[i]*d[i];
	}

	for(int i=1;i<=n;i++){
		s[i]=ss[i]+s[i-1];
		printf("%I64d ",v[i]+s[i]);
	}
	return 0;
}
