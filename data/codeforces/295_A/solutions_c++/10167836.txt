#include <iostream>

using namespace std;
long long int c[100005],a[100005],v[100005],l[100005],r[100005],d[1000005],h,w,x,y,n,m,k,i;
int main(){
    cin>>n>>m>>k;
    for(i=1;i<=n;i++)
  cin>>a[i];
    for(i=1;i<=m;i++)cin>>l[i]>>r[i]>>d[i];
    for(i=1;i<=k;i++){
        cin>>x>>y;
        v[x]++;
        v[y+1]--;
    }
    for(i=1;i<=m;i++){
        h+=v[i];
        c[l[i]]+=h*d[i];
        c[r[i]+1]-=h*d[i];
    }
    for(i=1;i<=n;i++){
        w+=c[i];
        a[i]+=w;
    }
    for(i=1;i<=n;i++)cout<<a[i]<<" ";
    return 0;
}
