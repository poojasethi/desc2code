#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
const long N=110000;
long long i,x,y,n,m,k,a[N],l[N],r[N],d[N],c[N],b[N];
int main() {
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	for (i=1;i<=n;i++) scanf("%I64d",&a[i]);
	for (i=1;i<=m;i++) scanf("%I64d%I64d%I64d",&l[i],&r[i],&d[i]);
	for (i=1;i<=k;i++) {
		scanf("%I64d%I64d",&x,&y);
		c[x]++;c[y+1]--;
	}
	for (i=1;i<=m;i++) {
		c[i]+=c[i-1];
		b[l[i]]+=d[i]*c[i];
		b[r[i]+1]-=d[i]*c[i];
	}
	for (i=1;i<n;i++) {
		printf("%I64d ",a[i]+b[i]);
		b[i+1]+=b[i];
	}
	printf("%I64d\n",a[n]+b[n]);
	return 0;
}
