using namespace std;
long long int a[100004],l[100002],r[100003],d[100004],w[100003],z[100003];
int main()
{
 long long  n,m,k;
 cin>>n>>m>>k;
 for(long i=1;i<=n;i++)cin>>a[i];
 for(long i=1;i<=m;i++){cin>>l[i]>>r[i]>>d[i];}
 long x,y;
 for(long i=1;i<=k;i++){
     cin>>x>>y;
     z[x]++;z[y+1]--;}long long int h=0;
     for(long i=1;i<=m;i++){
        h=h+z[i];
        w[l[i]]+=h*d[i];w[r[i]+1]-=h*d[i];
     }
     for(long i=1;i<=n;i++){w[i]=w[i]+w[i-1];a[i]+=w[i];
     cout<<a[i]<<" ";}
