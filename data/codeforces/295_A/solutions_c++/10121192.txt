#include <bits/stdc++.h>
using namespace std;

const int N=1e5+10;
typedef long long ll;
int add[N], n, m, k, L[N], R[N], D[N];
ll a[N], b[N];
int main() {
	scanf("%d%d%d", &n, &m, &k);
	for(int i=1; i<=n; ++i) scanf("%I64d", &a[i]);
	for(int i=1; i<=m; ++i) scanf("%d%d%d", &L[i], &R[i], &D[i]);
	for(int i=1; i<=k; ++i) {
		int l, r;
		scanf("%d%d", &l, &r);
		add[l]++;
		add[r+1]--;
	}
	ll sum=0;
	int ci=0;
	for(int i=1; i<=m; ++i) {
		ci+=add[i];
		b[L[i]]+=(ll)ci*D[i];
		b[R[i]+1]-=(ll)ci*D[i];
	}
	for(int i=1; i<=n; ++i) {
		sum+=b[i];
		printf("%I64d ", sum+a[i]);
	}
	return 0;
}
