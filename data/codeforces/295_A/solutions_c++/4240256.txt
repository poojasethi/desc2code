#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
const int inf=100001;
struct def{
   int f,s,t;
}st;
def b[inf];
long long e[inf],a[inf],c[inf],d;
int n,m,k,i,x,y;
int main()
{
cin>>n>>m>>k;
for (i=1;i<=n;i++)
	cin>>a[i];
for (i=1;i<=m;i++)
{
	cin>>b[i].f>>b[i].s>>b[i].t;
}
for (i=1;i<=k;i++)
	{
	cin>>x>>y;
	c[x]++;
	c[y+1]--;
	}
for (i=1;i<=m;i++)
{
d+=c[i];
e[b[i].f]+=1ll*b[i].t*d;
e[b[i].s+1]-=1ll*b[i].t*d;
}
d=0;
for (i=1;i<=n;i++)
{
	d+=e[i];
	a[i]+=d;
	cout<<a[i]<<" ";
}
return 0;
}
