#include <cstdio>
using namespace std;
#define MAX 100005
int K,M,N,x,y;
long long A[MAX],Aux[MAX],Op[MAX],D[MAX],L[MAX],R[MAX];
int main()
{
     scanf("%d%d%d",&N,&M,&K);
     for(int i=1;i<=N;i++) scanf("%I64d",&A[i]);
     for(int i=1;i<=M;i++) scanf("%I64d%I64d%I64d",&L[i],&R[i],&D[i]);
     for(int i=1;i<=K;i++) scanf("%d%d",&x,&y),Op[x]++,Op[y+1]--;
     for(int i=1;i<=M;i++) Op[i]+=Op[i-1],Aux[L[i]]+=D[i]*Op[i],Aux[R[i]+1]-=D[i]*Op[i];
     for(int i=1;i<=N;i++) Aux[i]+=Aux[i-1],printf("%I64d ",Aux[i]+A[i]);
 }
