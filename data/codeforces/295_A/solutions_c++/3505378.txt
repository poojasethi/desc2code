#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int u=100010; 
struct{int x,y; __int64 z;}op[u];
int a[u],n,m,q,i,x,y;
__int64 b[u],c[u],j;
int main()
{
	cin>>n>>m>>q;
	for(i=1;i<=n;i++) scanf("%d",&a[i]);
	for(i=1;i<=m;i++) scanf("%d%d%I64d",&op[i].x,&op[i].y,&op[i].z);
	for(i=1;i<=q;i++)
	{
		scanf("%d%d",&x,&y);
		b[x]++,b[y+1]--;
	}
	for(i=1;i<=m;i++)
	{
		j+=b[i];
		op[i].z*=j;
		c[op[i].x]+=op[i].z,c[op[i].y+1]-=op[i].z;
	}
	for(i=1,j=0;i<=n;i++)
	{
		j+=c[i];
		printf("%I64d ",a[i]+j);
	}
}
