#include<cstdio>
using namespace std;
long long a[111111],b[111111][3],c[111111],d[111111];
int n,m,k,x,y;
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(int i=1;i<=m;i++)
		scanf("%d%d%d",&b[i][0],&b[i][1],&b[i][2]);
	for(int i=1;i<=k;i++){
		scanf("%d%d",&x,&y);
		c[x]++;
		c[y+1]--;
	}
	for(int i=1;i<=m;i++){
		c[i]+=c[i-1];
		d[b[i][0]]+=c[i]*b[i][2];
		d[b[i][1]+1]-=c[i]*b[i][2];
	}
	for(int i=1;i<=n;i++){
		d[i]+=d[i-1];
		printf("%I64d ",a[i]+d[i]);
	}
	return 0;
}
