#include <iostream>
#include <algorithm>
#include <cmath>

#define rep(i,n) for(int i=1;i<=n;i++)
#define ll long long

using namespace std;

int n,m,k,l[100010],r[100010],x,y;
ll a[100010],d[100010],cnt[100010],res[100010];


int main(){
  cin>>n>>m>>k;
  rep(i,n)cin>>a[i];
  rep(i,m){cin>>l[i]>>r[i]>>d[i];}
  rep(i,k){
      cin>>x>>y;y++;
      cnt[x]++;cnt[y]--;
  }
  rep(i,m){
      cnt[i]+=cnt[i-1];
      res[l[i]]+=d[i]*cnt[i];
      res[++r[i]]-=d[i]*cnt[i];
  }
  rep(i,n){
      res[i]+=res[i-1];
      cout<<res[i]+a[i]<<" ";
  }
  cout<<endl;

}
