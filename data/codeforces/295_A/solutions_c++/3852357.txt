#include<stdio.h>
__int64 i,j,n,m,k,x,y,a[100001],l[100001],r[100001],d[100001],cnt[100001];
int main()
{
scanf("%I64d%I64d%I64d",&n,&m,&k);
for (i=1;i<=n;i++)
scanf("%I64d",&x),a[i]+=x,a[i+1]-=x;
for (i=1;i<=m;i++)
scanf("%I64d%I64d%I64d",l+i,r+i,d+i);
for (i=1;i<=k;i++)
scanf("%I64d%I64d",&x,&y),cnt[x]++,cnt[y+1]--;
for (i=1;i<=m;i++)
{
cnt[i+1]+=cnt[i];
a[l[i]]+=d[i]*cnt[i];
a[r[i]+1]-=d[i]*cnt[i];
}
for (i=1;i<n;i++)
{
printf("%I64d ",a[i]);
a[i+1]+=a[i];
}
printf("%I64d\n",a[n]);
}
