#include<cstdio>
#define ll long long
#define rep(i,n) for (int i=1;i<=n;i++)
#define N 100005
ll a[N],b[N],c[N],d[N]; int n,m,k,x,y,l[N],r[N];
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	rep(i,n) scanf("%I64d",a+i);
	rep(i,m) scanf("%d%d%I64d",l+i,r+i,d+i);
	rep(i,k) scanf("%d%d",&x,&y),b[x]++,b[y+1]--;
	rep(i,m) b[i]+=b[i-1],c[l[i]]+=b[i]*d[i],c[r[i]+1]-=b[i]*d[i];
	rep(i,n) c[i]+=c[i-1],printf("%I64d%s",c[i]+a[i],i<n?" ":"\n");
	return 0;
}
