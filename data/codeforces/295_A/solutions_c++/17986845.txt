#include<bits/stdc++.h>
#define fi(x,y,z) for(int x = y;x <= z; x++)
using namespace std;
long long int n, m, k, l[100002], r[100002], d[100002], x, y, s[100002], a[100002];
long long int ans[100002];
main ()
{
	cin>> n >> m >> k;
	fi(i, 1, n)
		cin>>a[i];
	fi(i, 1, m)
		cin>>l[i]>>r[i]>>d[i];
	fi(i, 1, k)
	{
		cin>>x>>y;
		s[x]++; s[y+1]--;
	}
	fi(i, 2, m+1)
		s[i]=s[i-1]+s[i];
	fi(i, 1, m)
	{
		ans[l[i]]+=s[i]*d[i];
		ans[r[i]+1]-=s[i]*d[i];
	}
	fi(i, 2, n)
		ans[i]=ans[i-1]+ans[i];
	fi(i, 1, n)
		cout<<ans[i]+a[i]<<" ";
}
