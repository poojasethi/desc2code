const double inf=1e20;
using namespace std;
char c[1111][1111];
int s,vis[1111][1111];
	int n,m;
int cnout[1000010];
void dfs(int x,int y)
{
	if (vis[x][y]||x<0||x>=n||y<0||y>=m||c[x][y]=='*') return;
	vis[x][y]=s;
	cnout[s]++;
	dfs(x-1,y);dfs(x+1,y);dfs(x,y-1);dfs(x,y+1);
}
int main(){
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++)
	scanf("%s",&c[i]);
	for (int i=0;i<n;i++)
	for (int j=0;j<m;j++)
	if (c[i][j]=='.'&&!vis[i][j]) s++,dfs(i,j);
	for (int i=0;i<n;i++)
	{
	for (int j=0;j<m;j++)
	if (c[i][j]=='*')
	{
		int ans=0;
		set<int> f;
		f.insert(vis[i-1][j]);
		f.insert(vis[i][j-1]);
		f.insert(vis[i+1][j]);
		f.insert(vis[i][j+1]);
		for (set<int>::iterator it=f.begin();it!=f.end();it++)
		ans=ans+cnout[*it];
		printf("%c",(ans+1)%10+'0');
	}
	else printf("%c",'.');
	puts("");
	}
	return 0;
}
