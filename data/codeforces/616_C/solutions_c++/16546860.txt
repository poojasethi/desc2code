#include <stdio.h>
#define id(x,y) ((x)*m+y)
char map[1001][1001],res[1001];
int P[1010001],c[1010001];
int v[1010001],n,m;
int dx[]={0,1,0,-1};
int dy[]={1,0,-1,0};
int find(int p)
{return p==P[p]?p:(P[p]=find(P[p]));}
void un1on(int x,int y)
{
	int p=find(x),q=find(y);
	if(p!=q){P[p]=P[q];c[q]=c[p]+c[q];c[p]=0;}
}
int main()
{
	scanf("%d%d%*c",&n,&m);
	for(int i=0;i<n;++i)scanf("%s",map[i]);
	for(int i=0;i<=id(n,m)+5;++i)c[i]=1,P[i]=i;
	for(int i=0;i<n;++i)
	for(int j=0;j<m;++j)
	{
		if(map[i][j]=='*')continue;
		for(int k=0;k<4;++k)
		{
			int x=i+dx[k],y=j+dy[k];
			if(x<0||x>=n||y<0||y>=m)continue;
			if(map[x][y]=='*')continue;
			un1on(id(x,y),id(i,j));
		}
	}
	for(int i=0;i<n;++i)
	{
		for(int j=0;j<m;++j)
		if(map[i][j]=='.')res[j]='.';
		else
		{
			int ret=1;
			for(int k=0;k<4;++k)
			{
				int x=i+dx[k],y=j+dy[k];
				if(x<0||x>=n||y<0||y>=m)continue;
				if(map[x][y]=='*')continue;
				int idx=find(id(x,y));
				if(!v[idx]){v[idx]=1;ret+=c[idx];}
			}
			res[j]=ret%10+'0';
			for(int k=0;k<4;++k)
			{
				int x=i+dx[k],y=j+dy[k];
				if(x<0||x>=n||y<0||y>=m)continue;
				if(map[x][y]=='*')continue;
				v[find(id(x,y))]=0;
			}
		}
		res[m]=0;puts(res);
	}
	return 0;
}
