#include <bits/stdc++.h>

using namespace std ;
#define fi first 
#define se second
#define aval false

string s[1010];
int cnt[1010][1010];
int cnt2[1010101];

int tmp = 1 , n , m ;

int dfs(int i, int j, int c){
	if ( i < 0 || j < 0 || i >= n || j >= m ) return -1;
	if ( s[i][j] == '*' ) return -1 ;
	if ( cnt[i][j] ) return cnt[i][j];
	cnt[ i ] [ j ] = c ;
	cnt2 [ c ] ++ ;
	dfs ( i + 1 , j , c ) ;
	dfs ( i - 1 , j , c ) ;
	dfs ( i , j + 1 , c ) ;
	dfs ( i , j - 1 , c ) ;
	return c ;
}

int main(){
	ios_base::sync_with_stdio(0);
	cin >> n >> m ;
	for ( int i = 0 ; i < n ; i++ ) 
		cin >> s[i] ;
	for ( int i = 0 ; i < n ; i++ ) {
		for ( int j = 0 ; j < m ; j++ ) {
			dfs ( i , j , tmp ) ;
			tmp ++ ;
		}
	}
	for ( int i = 0 ; i < n ; i++ ) {
		for ( int j = 0 ; j < m ; j++ ) {
			if ( s[i][j] == '*' ) {
				vector < int > v ;
				v.push_back ( dfs ( i - 1 , j , 0 ) ) ;
				v.push_back ( dfs ( i + 1 , j , 0 ) ) ;
				v.push_back ( dfs ( i , j-1 , 0 ) ) ;
				v.push_back ( dfs ( i , j+1 , 0 ) ) ;
				int sum = 1 ;
				v.push_back ( -1 ) ;
				sort ( v.begin() , v.end() ) ;

				for ( int i = 1 ; i < v.size() ; i++ ) 
					if ( v[i] != v[i-1] ) 
						sum += cnt2[v[i]] ; 
				cout << ( char ) ( '0' + ( sum % 10 ) ) ;
			}
			else {
				cout << '.' ;
			}
		}
		cout<< endl ;
	}
}
