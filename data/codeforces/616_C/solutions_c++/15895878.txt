#include <cstdio>
#include <map>
#include <set>

using namespace std;

int mp[1002][1002],type[1002][1002],M[1000000],n,m,i,j,cnt;
void dfs(int x,int y,int cnt)
{
	if(mp[x][y]!='.'||type[x][y]) return;
	type[x][y]=cnt;
	++M[cnt];
	dfs(x-1,y,cnt);
	dfs(x+1,y,cnt);
	dfs(x,y-1,cnt);
	dfs(x,y+1,cnt);
}
int main()
{
    scanf(" %d %d ",&n,&m);
    for(i=1;i<=n;++i)
	{
		for(j=1;j<=m;++j)
		{
			mp[i][j]=getchar();
		}
		getchar();
	}
	for(i=1;i<=n;++i)
	{
		for(j=1;j<=m;++j)
		{
			if(mp[i][j]=='.'&&!type[i][j])
			{
				dfs(i,j,++cnt);
			}
		}
	}
	for(i=1;i<=n;++i)
	{
		for(j=1;j<=m;++j)
		{
			if(mp[i][j]=='*')
			{
				set<int> st;
				int ans=1;
				if(type[i-1][j]) st.insert(type[i-1][j]);
				if(type[i+1][j]) st.insert(type[i+1][j]);
				if(type[i][j-1]) st.insert(type[i][j-1]);
				if(type[i][j+1]) st.insert(type[i][j+1]);
				set<int>::iterator it,b=st.begin(),e=st.end();
				for(it=b;it!=e;++it) ans+=M[*it];
				mp[i][j]=ans%10;
			}else{
				mp[i][j]=-1;
			}
		}
	}
	for(i=1;i<=n;++i)
	{
		for(j=1;j<=m;++j)
		{
			if(mp[i][j]>=0) printf("%d",mp[i][j]);
			else putchar('.');
		}
		putchar('\n');
	}
	return 0;
}
