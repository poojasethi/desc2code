#include<cstdio>
#include<set>
using namespace std;
int color[1000][1000],count[1000010],cnt,n,m;
char mat[1000][1000];
int dfs(int i,int j,int c){
	if(!(i>=0&&i<n&&j>=0&&j<m)) return -1;
	if(mat[i][j]=='*') return -1;
	if(color[i][j]) return color[i][j];
	count[c]++;
	color[i][j]=c;
	dfs(i-1,j,c);
	dfs(i,j-1,c);
	dfs(i,j+1,c);
	dfs(i+1,j,c);
	return c;
}
int main(){
	scanf("%d %d\n",&n,&m);
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++) scanf("%c",&mat[i][j]);
		getchar();	
	}
	int nc=0;
	for(int i=0;i<n;i++) 
		for(int j=0;j<m;j++) dfs(i,j,++nc);
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++)
			if(mat[i][j]=='.') putchar('.');
			else{
				int sum=0;
				set<int> col;
				col.insert(dfs(i-1,j,0));
				col.insert(dfs(i,j-1,0));
				col.insert(dfs(i,j+1,0));
				col.insert(dfs(i+1,j,0));
				for(set<int>::iterator it=col.begin();it!=col.end();it++)
					sum+=(*it>-1?count[*it]:0);
				printf("%d",(sum+1)%10);
			}
		printf("\n");
	}
  return 0;
}
