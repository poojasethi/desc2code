#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair < ll, ll > pi;
typedef vector < int > vi;

int n,m;
char a[1010][1010];
int cc[1000001]={0};
int ncc;
int vst[1000][1000]={0};

int k1[]={0,0,1,-1};
int k2[]={1,-1,0,0};
void dfs(int x,int y){
    if(!vst[x][y]){
        cc[ncc]++;
        vst[x][y]=ncc;
        for(int i=0;i<4;i++){
            int xx=x+k1[i];
            int yy=y+k2[i];
            if(xx>=0 && xx<n && yy>=0 && yy<m && !vst[xx][yy] && a[xx][yy]=='.')
                dfs(xx,yy);
        }
    }
}
int main(){
    cin>>n>>m;
    for(int i=0;i<n;i++)scanf("%s",&a[i]);
    ncc=0;
    for(int i=0;i<n;i++)for(int j=0;j<m;j++){
        if(a[i][j]=='.' && !vst[i][j]){
            ncc++;
            dfs(i,j);                     
        }
    }
    for(int i=0;i<n;i++)for(int j=0;j<m;j++)if(a[i][j]=='*'){
        set<int> s;
        for(int k=0;k<4;k++){
            int x=i+k1[k];
            int y=j+k2[k];
            if(x>=0 && x<n && y>=0 && y<m && a[x][y]=='.')s.insert(vst[x][y]);
        }
        int temp = 1;
        for(set<int>::iterator it=s.begin();it!=s.end();it++){
            temp+=cc[*it];
        }
        a[i][j]=(temp%10)+'0';
    }
    for(int i=0;i<n;i++)printf("%s\n",a[i]);
}
