#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <set>
using namespace std;
char map[1005][1005];
int dx[4]={0,0,-1,1};
int dy[4]={-1,1,0,0};
int vis[1005][1005];
int cnt[1000005];
//int cnt;
int n,m;
int col;
int dfs(int x,int y,int col){
	if (vis[x][y]){
		return vis[x][y];
	}
	if (x<1 || y<1 || x>n || y>m){
		return -1;
	}
	if (map[x][y]=='*'){
		return -1;
	}
	vis[x][y]=col;
	cnt[col]++;
	for (int i=0;i<4;i++){
		int nx=x+dx[i];
		int ny=y+dy[i];
		dfs(nx,ny,col);
	}
	return col;
}
int main(){
	cin>>n>>m;
	for (int i=1;i<=n;i++){
		for (int j=1;j<=m;j++){
			cin>>map[i][j];
		}
	}
	for (int i=1;i<=n;i++){
		for (int j=1;j<=m;j++){
			if (vis[i][j]==0 && map[i][j]=='.'){
				col++;
				dfs(i,j,col);
			}
		}
	}
	for (int i=1;i<=n;i++){
		for (int j=1;j<=m;j++){
			if (map[i][j]=='*'){
				set<int> f;
				f.insert(dfs(i,j-1,0));
				f.insert(dfs(i,j+1,0));
				f.insert(dfs(i+1,j,0));
				f.insert(dfs(i-1,j,0));
				int ans=1;
				for (set<int>::iterator it=f.begin();it!=f.end();it++){
					if (*it>-1){
						ans+=cnt[*it];
					}
					
				}
				cout<<ans%10;
			}
			else{
				cout<<'.';
			}
		}
		cout<<endl;
	}
	return 0;
}
