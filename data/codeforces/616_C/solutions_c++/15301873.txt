//In The Name Of God ;

#include<bits/stdc++.h>

using namespace std;

const int Maxn=1007;

int t=0;
int a[Maxn][Maxn];
int X[]={1,0,-1,0};
int Y[]={0,1,0,-1};
bool mark[Maxn][Maxn],vis[Maxn][Maxn];
bool M[Maxn*Maxn];
int col[Maxn][Maxn];
int Ans[Maxn*Maxn];

void dfs(int x,int y,int c)
{
  t++;
  mark[x][y]=true;
  col[x][y]=c;
  for(int i=0;i<4;i++)
    {
      int xx=X[i]+x,yy=Y[i]+y;
      if(!mark[xx][yy] && a[xx][yy])
	dfs(xx,yy,c);
    }
}

int main() 
{
  ios_base::sync_with_stdio(0);
  int n,m;
  char c;
  cin>>n>>m;
  m++,n++;
  for(int i=1;i<n;i++)
    for(int j=1;j<m;j++)
      cin>>c,a[i][j]=(c=='*')?0:1;
  int C=0;
  t=0;
  for(int i=1;i<n;i++)
    for(int j=1;j<m;j++,t=0)
      if(a[i][j] && !mark[i][j])
	dfs(i,j,++C),Ans[col[i][j]]=t;
  for(int i=1;i<n;cout<<endl,i++)
    for(int j=1;j<m;j++)
      {
	if(a[i][j])
	  {
	    cout<<'.';
	    continue;
	  }
	int A=0;
	for(int k=0;k<4;k++)
	  {
	    int x=i+X[k],y=j+Y[k];
	    if(a[x][y] && !M[col[x][y]])
	      (A+=Ans[col[x][y]])%=10,M[col[x][y]]=true;
	  }
	for(int k=0;k<4;k++)
	  {
	    int x=i+X[k],y=j+Y[k];
	    if(a[x][y])
	      M[col[x][y]]=false;
	  }
	cout<<(A+1)%10;
      }
  return 0;
}
