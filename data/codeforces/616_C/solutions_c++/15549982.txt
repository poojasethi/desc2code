#include <bits/stdc++.h>
using namespace std;
int visited[1010][1010];
char mat[1010][1010];
int sz[1010][1010],mark[1010][1010];
int n,m;
vector<pair<int,int> > v;
int a[]={0,0,1,-1};
int b[]={1,-1,0,0};
void dfs(int x,int y)
{
	v.push_back(make_pair(x,y));
	visited[x][y]=1;
	for(int i=0;i<4;i++)
	{
		int xx=x+a[i];
		int yy=y+b[i];
		if(xx>=0&&yy>=0&&xx<n&&yy<m&&!visited[xx][yy]&&mat[xx][yy]=='.') dfs(xx,yy);
	}
}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
	{
		scanf("%s",mat[i]);
	}
	int comp=1;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(!visited[i][j]&&mat[i][j]=='.')
			{
				v.clear();
				dfs(i,j);
				for(int l=0;l<v.size();l++)
				{
					sz[v[l].first][v[l].second]=v.size();
					mark[v[l].first][v[l].second]=comp;
				}
				comp++;
			}
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(mat[i][j]=='*')
			{
				int ans=0;
				set<int> s;
				for(int l=0;l<4;l++)
				{
					int xx=i+a[l];
					int yy=j+b[l];
					if(xx<0||xx>=n||yy<0||yy>=m||mat[xx][yy]=='*') continue;
					if(s.find(mark[xx][yy])!=s.end()) continue;
					ans=ans+sz[xx][yy];
					s.insert(mark[xx][yy]);
				}
				mat[i][j]='0'+(1+ans)%10;
			}
			printf("%c",mat[i][j]);
		}
		printf("\n");
	}
	return 0;
}
