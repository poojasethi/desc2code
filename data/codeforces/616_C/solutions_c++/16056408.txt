#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long
#define mod 1000000007

using namespace std;

int n,m;
char s[1005][1005];
int id[1005][1005],sz[1000005],p;

int dx[4]={0,1,0,-1};
int dy[4]={1,0,-1,0};

bool inr(int x,int y){
  return x>0 && x<=n && y>0 && y<=m;
}
void dfs(int x,int y){
  id[x][y]=p;
  sz[p]++;
  FI(i,0,3){
    int tx=x+dx[i];
    int ty=y+dy[i];
    if(inr(tx,ty) && !id[tx][ty] && s[tx][ty]=='.') dfs(tx,ty);
  }
  return;
}
int main(){
  scanf("%d %d",&n,&m);
  FI(i,1,n) scanf("\n%s",s[i]+1);
  FI(i,1,n) FI(j,1,m) if(!id[i][j] && s[i][j]=='.'){
    ++p;
    dfs(i,j);
  }
  FI(i,1,n){
    FI(j,1,m){
      if(s[i][j]=='.') printf(".");
      else{
        int summ=1;
        int tx[4],ty[4],idd[4];
        FI(k,0,3){
          tx[k]=i+dx[k];
          ty[k]=j+dy[k];
          idd[k]=id[tx[k]][ty[k]];
        }
        FI(k,0,3){
          if(inr(tx[k],ty[k]) && s[tx[k]][ty[k]]=='.'){
            FI(l,0,k-1){
              if(inr(tx[l],ty[l]) && idd[l]==idd[k]) goto gg;
            }
            summ+=sz[idd[k]];
            gg:;
          }
        }
        printf("%d",summ%10);
      }
    }
    printf("\n");
  }
  return 0;
}
