#include<iostream>
#include<cstring>
using namespace std;
char c;
int r[1005][1005];
int dc[1000005];
string s;
int i,j,n,m,cc=0,ii,jj,ss;
void dfs(int x,int y,int col){
	r[x][y]=col;
	dc[col]++;
	if(x+1<=n&&r[x+1][y]==0)dfs(x+1,y,col);
	if(x-1>0&&r[x-1][y]==0)dfs(x-1,y,col);
	if(y+1<=m&&r[x][y+1]==0)dfs(x,y+1,col);
	if(y-1>0&&r[x][y-1]==0)dfs(x,y-1,col);
}
int main(){
	cin>>n>>m;
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			cin>>c;
			if(c=='.')r[i][j]=0;else r[i][j]=-1;
		}
	}
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			if(r[i][j]==0){
				cc++; 
				dfs(i,j,cc);
			}
		}
	}
	cout<<endl;
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			if(r[i][j]==-1){
				ss=1;
				int zuo=r[i][j-1],you=r[i][j+1],shang=r[i-1][j],xia=r[i+1][j];
				ss+=dc[zuo];
				if(you!=zuo)ss+=dc[you];
				if(shang!=zuo&&shang!=you)ss+=dc[shang];
				if(xia!=shang&&xia!=zuo&&xia!=you)ss+=dc[xia];
				cout<<ss%10;
			}
			else
			cout<<'.';
		}
		cout<<endl;
	}
}
