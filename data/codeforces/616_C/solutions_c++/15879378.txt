#include<bits/stdc++.h>
using namespace std;
int s[1002][1002],vis[1002][1002];
int n,m,sum;
set<pair<int,int> > v;
void visit(int x,int y)
{
    if(x<0||y<0||x>=n||y>=m||vis[x][y]) return;
    if(s[x][y]!=-1){
        v.insert(make_pair(x,y));return;
    }
    vis[x][y]=1;
    sum++;
    visit(x+1,y);
    visit(x,y+1);
    visit(x-1,y);
    visit(x,y-1);

}
void dfs()
{
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            sum=0;
            if(s[i][j]==-1&&!vis[i][j]){
                v.clear();
                visit(i,j);
                for(set<pair<int,int> >::iterator it= v.begin();it!=v.end();it++){
                    s[it->first][it->second] += sum;
                }
            }

        }
    }
}
int main()
{
    char c;
    scanf("%d %d",&n,&m);
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            cin>>c;
            if(c=='.')s[i][j]=-1;
        }
    }
    dfs();
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            if(s[i][j]==-1)printf(".");
            else printf("%d",(s[i][j]+1)%10);
        }
        printf("\n");
    }
}
