#include <iostream>
#include <algorithm>
using namespace std;
const int N=1024,dr[]={-1,0,1,0},dc[]={0,1,0,-1};
int n,m,id[N][N],na,sz[N*N],t[4];
char s[N][N];
void fill(int r,int c){
  if(0<=r&&r<n&&0<=c&&c<m&&'.'==s[r][c]&&!id[r][c]){
    id[r][c]=na;
    ++sz[na];
    for(int d=0;d<4;++d)fill(r+dr[d],c+dc[d]);
  }
}
int main(){
  cin>>n>>m;
  for(int i=0;i<n;++i)cin>>s[i];
  for(int i=0;i<n;++i)for(int j=0;j<m;++j)if('.'==s[i][j]&&!id[i][j])++na,fill(i,j);
  for(int i=0;i<n;++i,cout<<endl)for(int j=0;j<m;++j)if('.'==s[i][j])cout<<'.';else{
    for(int d=0;d<4;++d){
      int r=i+dr[d],c=j+dc[d];
      t[d]=(0<=r&&r<n&&0<=c&&c<m)?id[r][c]:0;
    }
    sort(t,t+4);
    int z=1;
    for(int k=0,x=-1;k<4;++k){
      if(x!=t[k])z+=sz[t[k]];
      x=t[k];
    }
    cout<<z%10;
  }
}
