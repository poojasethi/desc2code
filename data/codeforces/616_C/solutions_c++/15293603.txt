#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

char a[1111][1111];
int n , m ;
int dx[4]={1,-1,0,0};
int dy[4]={0,0,-1,1};
int c[1111111]={0};
int color[1111][1111]={{0}};
int p=1;

bool inside(int i , int j){ return 1<=i && i<=n && 1<=j && j<=m ; }

void f(int i ,int j,int P){
color[i][j]=P;
c[P]++;
for(int k=0;k<4;k++)
{
   int x = dx[k]+i , y= dy[k]+j ;
    if( inside(x,y) && a[x][y]=='.' && color[x][y]==0 )
        f(x,y,P);
}

}


int solve(int i , int j ){
set<int> s;
set<int> :: iterator it;

for(int d=0;d<4;d++){
    int x=dx[d]+i , y=dy[d]+j ;
    if( inside(x,y) && a[x][y]=='.' && color[x][y]>0 ){
        s.insert(color[x][y]);
    }
}
int ans=0;
for(it=s.begin();it!=s.end();it++){
    int x=*it;
    ans+=c[x];
}
return (ans+1)%10;
}


int main()
{
cin>>n>>m;

for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    cin>>a[i][j];

for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
if( a[i][j]=='.' && color[i][j]==0){
    f(i,j,p);
     p++;
}



for(int i=1;i<=n;i++){
for(int j=1;j<=m;j++){
   if( a[i][j]=='.' )cout<<a[i][j];
   else cout<<solve(i,j);
}
cout<<endl;
}


    return 0;
}
