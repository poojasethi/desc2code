#include<stdio.h>
#include<iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;
const int N=1008;
char str[N];
int n,m,a[N][N],c[N*N];
int b[][2]={{1,0},{-1,0},{0,1},{0,-1}};
int yun(int x,int y,int k)
{
	a[x][y]=-k;c[k]++;
	int i,xx,yy;
	for(i=0;i<=3;i++)
	{
		xx=x+b[i][0];yy=y+b[i][1];
		if(1<=xx&&xx<=n&&1<=yy&&yy<=m&&a[xx][yy]==1)yun(xx,yy,k);
	}
	return 0;
}
int main(void)
{
	int i,j,k;
	scanf("%d%d",&n,&m);
	memset(a,0,sizeof(a));
	for(i=1;i<=n;i++)
	{
		memset(str,0,sizeof(str));
		scanf("%s",str+1);
		for(j=1;j<=m;j++)if(str[j]=='*')a[i][j]=0;else a[i][j]=1;
	}
	memset(c,0,sizeof(c));
	int flag=1;
	for(i=1;i<=n;i++)for(j=1;j<=m;j++)if(a[i][j]==1)yun(i,j,flag++);
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		if(a[i][j]<0)printf(".");
		else
		{
			k=0;
			if(a[i][j+1])k+=c[-a[i][j+1]];
			if(a[i][j-1]&&a[i][j-1]!=a[i][j+1])k+=c[-a[i][j-1]];
			if(a[i+1][j]&&a[i+1][j]!=a[i][j+1]&&a[i+1][j]!=a[i][j-1])k+=c[-a[i+1][j]];
			if(a[i-1][j]&&a[i-1][j]!=a[i][j+1]&&a[i-1][j]!=a[i][j-1]&&a[i-1][j]!=a[i+1][j])k+=c[-a[i-1][j]];
			printf("%d",(k+1)%10);
		}
		printf("\n");
	}
	return 0;
}
