#include <bits/stdc++.h>
using namespace std;

int dx[]={-1,0,1,0},dy[]={0,-1,0,1};
int belong[1005][1005];
int sz[1000*1000+5];
int cnt;
int n,m;
char g[1005][1005],ans[1005][1005];
int tot;
void dfs(int x,int y) {
    belong[x][y]=cnt;
    tot++;
    int nx,ny;
    for(int i=0;i<4;i++) {
        nx=x+dx[i],ny=y+dy[i];
        if(nx<1||nx>n||ny<1||ny>m||g[nx][ny]=='*'||belong[nx][ny]) continue;
        dfs(nx,ny);
    }
}
void calc(int x,int y) {
    int nx,ny;
    set<int> s;
    for(int i=0;i<4;i++) {
        nx=x+dx[i],ny=y+dy[i];
        if(nx<1||nx>n||ny<1||ny>m||g[nx][ny]=='*') continue;
        s.insert(belong[nx][ny]);
    }
    set<int>::iterator it=s.begin();
    int tmp=1;
    while(it!=s.end()) tmp+=sz[*it],it++;
    tmp%=10;
    ans[x][y]='0'+tmp;
}
int main() {
    scanf("%d %d",&n,&m);
    memset(ans,'.',sizeof(ans));
    for(int i=1;i<=n;i++) scanf("%s",g[i]+1),ans[i][m+1]='\0';
    for(int i=1;i<=n;i++) for(int j=1;j<=m;j++) if(g[i][j]=='.'&&!belong[i][j]) tot=0,++cnt,dfs(i,j),sz[cnt]=tot;
    for(int i=1;i<=n;i++) for(int j=1;j<=m;j++) if(g[i][j]=='*') calc(i,j);
    for(int i=1;i<=n;i++) printf("%s\n",ans[i]+1);

    return 0;

}
