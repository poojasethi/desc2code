#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<set>
using namespace std; 
int visit[1111][1111];
char a[1111][1111];
int ans[1111111];
int n,m;
int c[4]={1,-1,0,0};
int d[4]={0,0,-1,1};
void dfs(int x,int y,int index)
{
	if(x<0||y<0||x>=n||y>=m)return;
	if(a[x][y]=='*'||visit[x][y]!=0)return;
	visit[x][y]=index;
	dfs(x+1,y,index);
	dfs(x-1,y,index);
	dfs(x,y+1,index);
	dfs(x,y-1,index);	
}
int aaa(int u,int v)
{
	int p,i,x,y;
	set<int>q;
	set<int>::iterator it;
p=0;
	for(i=0;i<4;i++)
	{
		x=u+c[i];
		y=v+d[i];
		if(x<0||y<0||x>=n||y>=m);
		else
		{
			q.insert(visit[x][y]);
		}
	}
	for(it=q.begin();it!=q.end();it++)
	{
	p=p+ans[*it];
	}
	return p;
}
int main()
{   

int i,index,j;
memset(visit,0,sizeof(visit));
memset(ans,0,sizeof(ans));
scanf("%d%d",&n,&m);
for(i=0;i<n;i++)
scanf("%s",a[i]);


index=0;
for(i=0;i<n;i++)
for(j=0;j<m;j++)
{
	if(a[i][j]=='.'&&visit[i][j]==0)
	dfs(i,j,++index);
}
for(i=0;i<n;i++)
for(j=0;j<m;j++)
{
	if(a[i][j]=='.')
	ans[visit[i][j]]++;
}
for(i=0;i<n;i++)
{
for(j=0;j<m;j++)
{
	if(a[i][j]=='.')
	printf(".");
	else
	{
		printf("%d",(aaa(i,j)+1)%10);
	}
	
}
printf("\n");
}
