#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<set>
using namespace std;
char map[1010][1010];
int vis[1010][1010],c[1010][1010],sum[1010];
int cnt;
void dfs(int i,int j)
{
	if(vis[i][j]!=0||map[i][j]!='.') return;
	vis[i][j]=cnt;
	sum[cnt]++;
	dfs(i-1,j);
	dfs(i,j-1);
	dfs(i,j+1);
	dfs(i+1,j);
}
int main()
{
	int n,m;
	while(scanf("%d %d\n",&n,&m)!=EOF)
	{
		cnt=0;
		memset(map,'*',sizeof(map));
		memset(vis,0,sizeof(vis));
		memset(sum,0,sizeof(sum));
		memset(c,0,sizeof(c));
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=m;j++)
			{
				scanf("%c",&map[i][j]);
			}
			getchar();
		}
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=m;j++)
			{
				if(map[i][j]=='.'&&vis[i][j]==0)
				{
					cnt++;
					dfs(i,j);
				}
			}
		}
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=m;j++)
			{
				if(map[i][j]=='.') printf(".");
				else
				{
					int cnt2=1;
					set<int>s;
					if(map[i-1][j]=='.') s.insert(vis[i-1][j]);
					if(map[i][j-1]=='.') s.insert(vis[i][j-1]);
					if(map[i][j+1]=='.') s.insert(vis[i][j+1]);
					if(map[i+1][j]=='.') s.insert(vis[i+1][j]);
					for(set<int>::iterator it=s.begin();it!=s.end();it++)
					cnt2+=sum[*it];
					printf("%d",cnt2%10);
				}
			}
			printf("\n");
		}
	}
	return 0;
}
