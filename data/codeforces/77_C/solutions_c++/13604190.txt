const int N = 1e5+7;
int n,m,i,j,k,t,s,x,y,z,h[N],d,l,ii,jj,u;
vector<int> V[N];
int dfs(int x,int root)
{
	int i,j,u,t=0,k,s=d=0;
	vector<int> v;
	u=V[x].size();
	for(i=0;i<u;i++)
		if(V[x][i]!=root and h[V[x][i]])
		{
			--h[V[x][i]];
			j=dfs(V[x][i],x);
			d=1;
			v.pb(j);
			s+=h[V[x][i]];
		}
	if(!d)return 0;
	sort(v.rbegin(),v.rend());
	u=v.size();
	for(i=0;i<u;i++)
		if(h[x])
		{
			t+=v[i]+2;
			h[x]--;
		}
	k=min(s,h[x]);
	h[x]-=k;
	return t+2*k;
}
main()
{
	cin>>n;
	for(i=1;i<=n;i++)scanf("%lld",h+i);
	for(i=1;i<n;i++)
	{
		scanf("%lld%lld",&x,&y);
		V[x].pb(y);
		V[y].pb(x);
	}
	cin>>k;
	cout<<dfs(k,0)<<endl;
}
