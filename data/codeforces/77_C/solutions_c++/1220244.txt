#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define pb push_back
#define pi pair<long long,long long> 

vector<int> g[101000];
int s[101000],n,a,b,st,sum;

long long f(int at,int pre){

  vector<pi> v;
  long long sum=0;

  rep(i,g[at].size())if(g[at][i]!=pre&&s[g[at][i]]!=0){
    int ti=g[at][i];
    s[ti]--;
    v.pb(pi(f(ti,at),ti));
  }

  sort(v.begin(),v.end());
  reverse(v.begin(),v.end());

  if(v.size()<s[at]){
    rep(i,v.size())sum+=v[i].first+2;
    s[at]-=v.size();
    rep(i,g[at].size()){
      if(s[at]==0)break;
      if(g[at][i]!=pre&&s[g[at][i]]!=0){
    if(s[g[at][i]]>s[at]){s[g[at][i]]-=s[at];sum+=2*s[at];s[at]=0;}
    else{s[at]-=s[g[at][i]];sum+=2*s[g[at][i]];s[g[at][i]]=0;}
      }
    }
  }
  else {
    rep(i,s[at]) sum+=v[i].first+2;
    s[at]=0;
  }

  return sum;

}

int main(){

  cin>>n;
  rep(i,n)cin>>s[i];
  rep(i,n-1){
    cin>>a>>b;a--;b--;
    g[a].pb(b);g[b].pb(a);
  } 
  cin>>st;
  cout<<f(st-1,-1)<<endl;

}
