#include<iostream>
#include<cstdio>
#include<cstring> 
#include<vector>
#include<algorithm>
#define M 100100
using namespace std;
typedef long long ll;

int n,root,c[M];
vector<int> adj[M];

ll dfs(int x,int fa) 
{
	ll re=0;
	vector<ll>temp;
	int cnt=0;
	for(int y,i=0;i<adj[x].size();i++) 
	{
		y=adj[x][i];
		if(y!=fa && c[y]) 
		{
			c[y]--;
			temp.push_back(dfs(y,x));
			cnt+=c[y];
		}
	}
	sort(temp.begin(),temp.end());	
	while(c[x] && temp.size()) 
	{
		re+=temp.back()+2;
		c[x]--;
		temp.pop_back();
	}
	int k=min(c[x],cnt);
	re+=k*2,c[x]-=k;
	return re;
}
int main() 
{
	cin>>n;
	for(int i=1;i<=n;i++) scanf("%d",&c[i]);
	for(int x,y,i=1;i<n;i++) 
	{
		scanf("%d%d",&x,&y);
		adj[x].push_back(y);
		adj[y].push_back(x);
	}
	cin>>root;
	cout<<dfs(root,-1)<<endl;
	return 0;
}
