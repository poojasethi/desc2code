#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
#include<string.h>
using namespace std;
int n,a[100005],s;
__int64 dp[100005];
vector<int>pp[100005];
bool cmp(int a,int b){
	return a>b;
}
int min(int a,int b){
	return a>b?b:a;
}
void dfs(int u,int fa){
 dp[u]=1;
	 int i;
	 vector<__int64>ss;
	 for(i=0;i<pp[u].size();i++){
	  if(a[pp[u][i]]==0||pp[u][i]==fa)continue;
	  a[pp[u][i]]--;
	  dfs(pp[u][i],u);
	  ss.push_back(dp[pp[u][i]]);
	 }
	 sort(ss.begin(),ss.end(),cmp);
	 for(i=0;i<ss.size()&&a[u];i++){
		 a[u]--;
	  dp[u]+=ss[i]+1;
	 }
	 for(i=0;i<pp[u].size()&&a[u];i++){
	  if(pp[u][i]==fa||a[pp[u][i]==0])continue;
	  dp[u]+=2*min(a[u],a[pp[u][i]]);
	  a[u]-=min(a[u],a[pp[u][i]]);
	 }
}
int main(){
	while(cin>>n){
	 int i;
	 memset(dp,0,sizeof(dp));
	 for(i=1;i<=n;i++)
		 scanf("%d",&a[i]),pp[i].clear();
	 for(i=1;i<n;i++){
	  int x,y;
	  scanf("%d%d",&x,&y);
	  pp[x].push_back(y);
	  pp[y].push_back(x);
	 }
	 scanf("%d",&s);
	 dfs(s,s);
	 printf("%I64d\n",dp[s]-1);
	}
	return 0;
}
