#include <algorithm>
#include <cstdio>
#include <iostream>
#include <vector>

#define pb push_back

using namespace std;

typedef long long int Lint;

const int MAXN=1e5+10;

int N,root;
int cnt[MAXN];

vector<int> g[MAXN];

inline bool compare( const int &a , const int &b ){
	return a>b;
}

inline Lint dfs( int nd , int pre ){
	
	int lf=0;
	Lint rev=0LL;
	vector<Lint> vec;
	
	for( vector<int>::iterator it=g[nd].begin() ; it!=g[nd].end() ; it++ )
		if( *it!=pre && cnt[*it] ){
			cnt[*it]--;
			vec.pb(dfs(*it,nd));
			lf+=cnt[*it];
		}
	
	sort(vec.begin(),vec.end(),compare);
	
	for( vector<Lint>::iterator it=vec.begin() ; it!=vec.end() ; it++ )
		if( cnt[nd]){
			cnt[nd]--;
			rev+=*it+2LL;
		}
	
	int tmp=min(lf,cnt[nd]);
	rev+=tmp*2LL;
	cnt[nd]-=tmp;
	
	return rev;
}

int main(){
	
	scanf("%d",&N);
	
	for( int i=1 ; i<=N ; i++ )
		scanf("%d",cnt+i);
	
	for( int a,b,i=1 ; i<N ; i++ ){
		scanf("%d%d",&a,&b);
		g[a].pb(b);
		g[b].pb(a);
	}
	
	scanf("%d",&root);
	printf("%I64d\n",dfs(root,0));
	
	return 0;
}
