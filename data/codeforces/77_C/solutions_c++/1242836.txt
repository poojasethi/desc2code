#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> a[100005];
int p[100005];
typedef long long LL;

LL go(int x, int from=-1) {
	vector<LL> r;
	LL ret=0;
	for(int i=0;i<(int)a[x].size();i++) {
		if(a[x][i]!=from && p[a[x][i]]) {
			--p[a[x][i]];
			LL v = go(a[x][i], x);
			r.push_back(v);
		}
	}
	sort(r.begin(), r.end());
	while(!r.empty() && p[x]>0) {
		--p[x]; ret += 2+r.back(); r.pop_back();
	}
	for(int i=0;i<(int)a[x].size();i++) {
		if(a[x][i]==from) continue;
		LL m = min(p[x], p[a[x][i]]);
		p[x] -= m; p[a[x][i]] -= m; ret += 2*m;
	}
	return ret;
}

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d", &p[i]);
	for(int i=0,x,y;i<n-1;i++) {
		scanf("%d%d", &x, &y);
		a[x].push_back(y);
		a[y].push_back(x);
	}
	int s;
	scanf("%d", &s);
	LL val = go(s);
	printf("%I64d\n", val);
	return 0;
}
