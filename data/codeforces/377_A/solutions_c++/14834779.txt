#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
char a[505][505];
int b[505][505];
int n,m,k;
void dfs(int x,int y) 
{
	if(x>=n||y>=m||x<0||y<0)
	return;
	if(a[x][y]!='.')return;
	if(b[x][y]==1)return;
	else b[x][y]=1;
	dfs(x-1,y);
	dfs(x+1,y);
	dfs(x,y+1);
	dfs(x,y-1);
	if(k!=0)
{
	a[x][y]='X';k--;
}	
}
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;i++)
	scanf("%s",a[i]); 
	for(int i=0;i<n&&k!=0;i++)
	for(int j=0;j<m&&k!=0;j++)
	dfs(i,j);
	for(int i=0;i<n;i++)
	puts(a[i]);
}
