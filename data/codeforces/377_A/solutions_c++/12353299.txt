#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
#define maxn 512
int n,m,k;
char map[maxn][maxn];
bool vis[maxn][maxn];
void dfs(int x,int y)
{
	if(x<0||x>=n||y<0||y>=m||map[x][y]!='.'||vis[x][y])
		return ;
	vis[x][y]=true;
	dfs(x+1,y);
	dfs(x-1,y);
	dfs(x,y+1);
	dfs(x,y-1);
	if(k)
	{
		map[x][y]='X';
		k--;
	}
}
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;i++)
		scanf("%s",map[i]);
	for(int i=0;i<n&&k;i++)
		for(int j=0;j<m&&k;j++)
			dfs(i,j);
	for(int i=0;i<n;i++)
		puts(map[i]);
	return 0;
}
