#include<stdio.h>
int i, j, m, n, k, b[510][510];
char a[510][510];
void find(int i, int j)
{
	if (a[i][j] != '.' || i<0 || i >= m || j<0 || j >= n || b[i][j] == 1)
		return;
	b[i][j] = 1;
	find(i + 1, j);
	find(i - 1, j);
	find(i, j + 1);
	find(i, j - 1);
	if (k != 0)
	{
		a[i][j] = 'X';
		k--;
	}
}
int main()
{
	scanf("%d%d%d", &m, &n, &k);
	for (i = 0; i<m; i++)
		scanf("%s", a[i]);
	for (i = 0; i<m&&k; i++)
		for (j = 0; j<n&&k; j++)
			find(i, j);
	for (i = 0; i<m; i++)
		printf("%s\n", a[i]);
}
