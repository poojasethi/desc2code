#include<bits/stdc++.h>
using namespace std;
char maze[555][555];
bool v[555][555];
int n,m,k,u[]={1,0,-1,0},w[]={0,1,0,-1};
void dfs(int i,int j) {
if(maze[i][j]!='.'||v[i][j]) return;
v[i][j]=true;

for(int d=0;d<4;d++) {
    int x=i+u[d],y=j+w[d];
dfs(x,y);
}
if(k>0){
    maze[i][j]='X';
    k--;
}
}

int main() {
int i,j;
cin>>n>>m>>k;

for(int i=0;i<n;i++) {
cin>>maze[i];
}

for( i=0;i<n;i++)
    for(j=0;j<m;j++){
    if(maze[i][j]=='.');
    dfs(i,j);
    }

for( i=0;i<n;i++) {
    for( j=0;j<m;j++)
    cout<<maze[i][j];
    puts("");
}
return 0;
}
