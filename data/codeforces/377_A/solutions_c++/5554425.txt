#include <iostream>
using namespace std;

const int nmax = 500;
int n, m, k;
char a[nmax][nmax + 1];
bool v[nmax][nmax];
const int dx[] = {1, 0, -1, 0}, dy[] = {0, -1, 0, 1};

void dfs(int x, int y) {
	if (x >= 0 && x < m && y >= 0 && y < n && a[y][x] == '.' && !v[y][x]++) {
		for (int id = 0; id < 4; id++) dfs(x + dx[id], y + dy[id]);

		if (k) a[y][x] = 'X', k--;
	}
}

int main() {
	int i, j;

	cin >> n >> m >> k;

	for (i = 0; i < n; i++) cin >> a[i];

	for (i = 0; i < n; i++) for (j = 0; j < m; j++) dfs(j, i);

	for (i = 0; i < n; i++) cout << a[i] << '\n';
}
