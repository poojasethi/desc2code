#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int n, m, k, i, j;
char inp[505][505];
bool check[505][505];

void dfs(int x, int y) 
{
	if (x < 0 || x >= n || y < 0 || y >= m || inp[x][y] != '.' || check[x][y] == true) return;
	check[x][y] = true;
	dfs(x+1, y);
	dfs(x-1, y);
	dfs(x, y+1);
	dfs(x, y-1);

	if (k) {
		inp[x][y] = 'X';
		k--;
	}
}
int main()
{
	cin >> n >> m >> k;
	for (i = 0; i < n; i++) cin >> inp[i];
	for (i = 0; i < n; i++) {
		for (j = 0; j < m; j++) 
			if (inp[i][j] == '.') break;
		if (j < m) break;
	}
	dfs(i, j);
	for (i = 0; i < n; i++) cout << inp[i] << endl;
}
