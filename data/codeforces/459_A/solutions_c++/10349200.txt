#include <bits/stdc++.h>
using namespace std;

int main() {
	int a,b,c,d;
	scanf("%d%d%d%d",&a,&b,&c,&d);
	if(a==c||b==d||abs(a-c)==abs(b-d)){
		if(a==c)printf("%d %d %d %d",d-b+a,b,d-b+a,d);
		else if(b==d)printf("%d %d %d %d",a,b+a-c,c,b+a-c);
		else printf("%d %d %d %d",c,b,a,d);
	}
	else printf("-1");
	return 0;
}
