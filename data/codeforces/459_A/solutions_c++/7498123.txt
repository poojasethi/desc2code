#include <cstdio>
#include <cstdlib>
int main(){
    int a,b,c,d;
    scanf("%d%d%d%d",&a,&b,&c,&d);
    if(a==c)
        printf("%d %d %d %d",a+d-b,b,c+d-b,d);
    else if(b==d)
        printf("%d %d %d %d",a,b+c-a,c,d+c-a);
    else if(abs(d-b)==abs(c-a))
        printf("%d %d %d %d",a,d,c,b);
    else printf("-1");
    return 0;
}
