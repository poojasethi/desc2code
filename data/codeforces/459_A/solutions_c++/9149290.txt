#include<iostream>
#include<stdlib.h>
using namespace std;
int main()
{
 int a,b,c,d;
 cin>>a>>b>>c>>d;
 if(a==c)
  cout<<a+abs(b-d)<<" "<<b<<" "<<c+abs(b-d)<<" "<<d;
 else if(b==d)
    cout<<a<<" "<<b+abs(c-a)<<" "<<c<<" "<<d+abs(c-a);
 else if(abs(c-a)==abs(b-d))
    cout<<a<<" "<<d<<" "<<c<<" "<<b;
 else
    cout<<"-1";
 return 0;
}
