#include<bits/stdc++.h>
using namespace std;
int main()
{
 int a,b,x,y;
 cin>>a>>b>>x>>y;
 if(a==x)
    cout<<a+y-b<<" "<<y<<" "<<x+y-b<<" "<<b<<endl;
 else if(b==y)
           cout<<a<<" "<<y+x-a<<" "<<x<<" "<<b+x-a<<endl;
 else if(a-x==b-y||a-x==y-b)
        cout<<a<<" "<<y<<" "<<" "<<x<<" "<<b<<endl;
 else
      cout<<-1<<endl;
}
