#include<bits/stdc++.h>
using namespace std;
int main()
{
    int x1,x2,y1,y2;
    cin >> x1 >> y1 >> x2 >> y2;
    if(x1==x2||y1==y2) cout << y2-y1+x1 << " " << x2-x1+y1 << " " << y2-y1+x2 << " " << x2-x1+y2;
    else if(x1!=x2&&y1!=y2&&abs(x2-x1)==abs(y2-y1)) cout << x2 << " " << y1 << " " << x1 << " " << y2;
    else cout << "-1";
}
