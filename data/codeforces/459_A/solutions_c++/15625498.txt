#include<bits/stdc++.h>

using namespace std;

int a,b,c,d;

int main(){
    cin>>a>>b>>c>>d;
    if(a!=c && b!=d && abs(a-c)!=abs(b-d)) cout<<-1; else
    if(a==c) cout<<a+abs(b-d)<<" "<<b<<" "<<c+abs(b-d)<<" "<<d; else
    if(b==d) cout<<a<<" "<<b+abs(a-c)<<" "<<c<<" "<<d+abs(c-a); else
    cout<<a<<" "<<d<<" "<<c<<" "<<b;
    return 0;
}
