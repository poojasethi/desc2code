#include<iostream>
#include<cmath>
using namespace std;

int main() 
{
	int a,b,c,d;
	cin>>a>>b>>c>>d;
	if(abs(a-c)==abs(b-d))
	    cout<<a<<" "<<d<<" "<<c<<" "<<b;
	else if(a==c)
	    cout<<a+abs(d-b)<<" "<<b<<" "<<c+abs(d-b)<<" "<<d;
	else if(b==d)
	    cout<<a<<" "<<b+abs(a-c)<<" "<<c<<" "<<d+abs(a-c);
	else
	    cout<<-1;
	return 0;
}
