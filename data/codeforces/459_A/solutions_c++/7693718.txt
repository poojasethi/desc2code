#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
	int a,b,c,d;
	scanf("%d%d%d%d",&a,&b,&c,&d);
	c -= a;
	d -= b;
	if ( c*c==d*d ) printf("%d %d %d %d\n",a,b+d,a+c,b);
	else if ( c == 0 ) printf("%d %d %d %d\n",a+d,b,a+d,b+d);
	else if ( d == 0 ) printf("%d %d %d %d\n",a,b+c,a+c,b+c);
	else printf("-1\n");
}
