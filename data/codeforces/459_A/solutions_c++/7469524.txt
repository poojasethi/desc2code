#include<stdio.h>
#include<stdlib.h>
int main()
{
    int a,b,x,y;
    scanf("%d %d %d %d",&a,&b,&x,&y);
    if(a==x) printf("%d %d %d %d\n",a+abs(b-y),b,x+abs(b-y),y);
    else if(b==y) printf("%d %d %d %d\n",a,b+abs(a-x),x,y+abs(a-x));
    else if(abs(a-x)==abs(b-y))printf("%d %d %d %d\n",a,y,x,b);
    else printf("-1\n");
    return 0;
}
