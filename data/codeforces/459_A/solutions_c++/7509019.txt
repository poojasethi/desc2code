#include<stdio.h>
main()
{
    int x,y,a,b;
    scanf("%d %d %d %d",&a,&b,&x,&y);
    if(a==x)
        printf("%d %d %d %d\n",a+y-b,y,x+y-b,b);
    else if(b==y)
        printf("%d %d %d %d\n",a,y+x-a,x,b+x-a);
    else if(a-x==b-y||a-x==y-b)
        printf("%d %d %d %d\n",a,y,x,b);
    else
        printf("-1\n");
    return 0;
}
