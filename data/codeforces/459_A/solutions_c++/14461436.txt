#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a, b, c, d, e, f; cin>>a>>b>>c>>d;e=abs(a-c), f=abs(b-d);
    if(a==c)cout<<a+f<<" "<<b<<" "<<c+f<<" "<<d<<endl;
    else if(b==d)cout<<a<<" "<<b+e<<" "<<c<<" "<<d+e<<endl;
    else if(e==f)cout<<a<<" "<<d<<" "<<c<<" "<<b<<endl;
    else cout<<"-1"<<endl;
    return 0;
}
