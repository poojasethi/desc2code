#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int a,b,c,d,x,y;
    scanf("%d%d%d%d",&a,&b,&c,&d);
    x=abs(a-c);y=abs(b-d);
    if(!x)printf("%d %d %d %d\n",a+y,b,c+y,d);
    else if(!y)printf("%d %d %d %d\n",a,b+x,c,d+x);
    else if(x!=y)puts("-1");
    else printf("%d %d %d %d\n",a,d,c,b);
}
