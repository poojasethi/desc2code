#include<stdio.h>
#include<string.h>

char a[1010000],b[1010000];

int main()
{
	int la,lb,fa,fb,i;
	gets(a);
	gets(b);
	la=strlen(a);
	lb=strlen(b);
	fa=fb=0;
	for(i=0;i<la;i++) 
	{
		if(a[i]=='1') fa=1;
		if(b[i]=='1') fb=1;
	}
	if(la==lb && fa==fb) puts("YES");
	else puts("NO");
	return 0;
}
