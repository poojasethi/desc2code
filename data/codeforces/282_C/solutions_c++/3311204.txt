#include <iostream>
#include <algorithm>

using namespace std;

string a, b;

int main()
{
  cin >> a >> b;
  int cnta = count(a.begin(), a.end(), '1');
  int cntb = count(b.begin(), b.end(), '1');
  if (a.size() != b.size() || (bool(cnta) != bool(cntb))) cout << "NO";
  else cout << "YES";
  return 0;
}
