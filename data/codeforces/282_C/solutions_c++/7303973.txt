#include <iostream>
#include <fstream>

using namespace std;

string a, b;

int main()
{
    cin >> a >> b;
    if (a == b) cout << "YES";
    else if (a.size() != b.size() ||
        a.find('1') == string::npos ||
        b.find('1') == string::npos) cout << "NO";
    else cout << "YES";
}
