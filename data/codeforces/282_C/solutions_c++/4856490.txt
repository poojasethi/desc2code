#include <iostream>
#include <string>

using namespace std;

int main ()
{	
	string a, b;
	cin >> a >> b;

	size_t n1 = a.find('1'), n2 = b.find('1');
	cout << (a.length() != b.length() || n1 == string::npos && n2 != string::npos || n1 != string::npos && n2 == string::npos ? "NO" : "YES");

	return 0;
}
