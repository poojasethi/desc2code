#include <iostream>
using namespace std;

int main()
{string s1,s2; cin>>s1>>s2;
if (s1.size()!=s2.size()) {cout<<"NO"; return 0;}
int n=s1.size();
bool f1=0,f2=0;
for (int i=0;i<n;i++)
 {
  if (s1[i]=='1') f1=1;
  if (s2[i]=='1') f2=1;
 }
if (!f1&&f2||!f2&&f1) cout<<"NO";
 else cout<<"YES";
return 0;
}
