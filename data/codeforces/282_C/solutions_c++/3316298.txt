#include <string>
#include <iostream>
bool check(std::string s){
	for(int i = 0; i < s.size();i++)
		if(s[i] != '0')
			return true;
	return false;
}
int main(){
	std::string a,b;
	std::cin >> a >>b;
	if(((a.size() == b.size()) && check(a) && check(b))|| a == b)
		std::cout<<"YES\n";
	else
		std::cout<<"NO\n";
}
