#include <bits/stdc++.h>
using namespace std;
int i, temp;
string S;
int main () {
	cin>>S;
	if (S[0]=='h') S.insert(4, "://");
	else S.insert(3, "://");
	temp = S.find ("ru", 0);
	if (S[temp-1]=='/') temp =S.find("ru", temp+1);
	if (S.length()-temp>2) S.insert(temp+2, "/");
	S.insert(temp, ".");
	cout<<S<<endl;
	return 0;
}
