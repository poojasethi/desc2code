#include <iostream>
#include <string>
using namespace std;

int main(){
	ios::sync_with_stdio(false);
	string s,a;
	cin >> s;
	int i;
	if(s[0]=='h'){
		i=4;
		a+="http://";
	}
	else {
		i=3;
		a+="ftp://";
	}
	a+=s[i];
	i++;
	for(;i<s.size();i++){
		if(s[i]=='r'&&s[i+1]=='u')break;
		a+=s[i];
	}
	a+=".ru";
	i+=2;
	if(i<s.size()) a+='/';
	for(;i<s.size();i++) a+=s[i];
	cout << a;
}
