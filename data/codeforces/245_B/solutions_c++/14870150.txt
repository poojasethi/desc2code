#include<bits/stdc++.h>

using namespace std;

int i,j,k;
string line;
int main(void)
{
	cin>>line;
	int x=line.find("p");
	line.replace(x,1,"p://");

	int y=line.find("ru");
	if(line[y-1]=='/')
	y=line.find("ru",y+2);
	
	
	if(y+2==line.length())
	{
		line.replace(y,2,".ru");
	}
	else
	line.replace(y,2,".ru/");
	
	cout<<line;
	

	return 0;
}
