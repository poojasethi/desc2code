#include <iostream>
#include <string>

using namespace std;

string s;
int x = 3;

int main(){
    cin >> s;
    if(s[0] == 'h'){
        x = 4;
        cout << "http://";
    }
    else{
        cout << "ftp://";
    }
    int ho = s.rfind("ru");
    cout << s.substr(x, ho - x) << ".ru";
    if(ho+2 != s.size()){
        cout << "/" << s.substr(ho+2);
    }
}
