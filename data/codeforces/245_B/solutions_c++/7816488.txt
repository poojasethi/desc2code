#include<bits/stdc++.h>
using namespace std;

int main(){
	string in;
	getline(cin, in);
	if(in.substr(0,3)=="ftp"){
		in.erase(0,3);
		cout<<"ftp://";
	}else{
		in.erase(0,4);
		cout<<"http://";
	}
	size_t found = in.find("ru",1);
	cout<<in.substr(0,found)<<".ru";
	in.erase(0,found+2);
	if(in!="") cout<<"/"<<in;
	cout<<endl;
	return 0;
}
