#include <iostream>
#include <string>
using namespace std;

int main()
{
	string s;
	cin>>s;
	
	if(s[0]=='h')
	{
		cout<<"http://";
		s=s.substr(4);
	}
	else
	{
		cout<<"ftp://";
		s=s.substr(3);
	}
	
	int i=1+s.substr(1).find("ru");
	
	cout<<s.substr(0,i)<<".ru";
	s=s.substr(i+2);
	
	if(s.size()>0) cout<<"/"<<s;
	
	cout<<endl;
	
	return 0;
}
