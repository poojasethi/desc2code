#include <string>
#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    string s;
    cin >> s;

    int p = s.find('p');
    s.insert(p + 1, "://");

    int ru = s.find("ru", p + 5);
    if (ru + 2 != s.length())
    {
        s.insert(ru + 2, "/");
    }
    s.insert(ru, ".");

    printf("%s", s.c_str());
    return 0;
}
