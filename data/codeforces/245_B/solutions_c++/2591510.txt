#include<iostream>
#include<cstring>

using namespace std;

char s[100];

int n;

int main()
{
 cin>>s;
 n=strlen(s);
 
 int k=0;
 if(s[0]=='f'){cout<<"ftp://";k=3;}
 else  {cout<<"http://";k=4;}
 
 cout<<s[k];
 k++;
 while(!(s[k]=='r' && s[k+1]=='u'))
 {
  cout<<s[k];
  k++;
 }
 
 cout<<".ru"; k+=2;
 
 if(k<n)cout<<"/";
 while(k<n){cout<<s[k++];}
 cout<<"\n";
 return 0;
}
