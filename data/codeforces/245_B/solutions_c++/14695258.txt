#include<iostream>
using namespace std;

string s;
int p1, p2;

int main() {
	cin>>s;
	p1 = (s[0] == 'f') ? 3 : 4;
	for(p2 = p1+1;s[p2]!='r' || s[p2+1]!='u'; p2++);
	cout<<s.substr(0, p1)<<"://"<<s.substr(p1, p2-p1)<<".ru";
	if(s.length() > p2 + 2)
		cout<<"/"<<s.substr(p2+2);
	cout<<endl;
	
	return 0;
}
