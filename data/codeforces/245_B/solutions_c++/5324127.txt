#include <cstdio>

using namespace std;

char str[60];
int i;

int main(){
	scanf("%s",str);
	if(str[0]=='f'){
		printf("ftp://");
		i=3;
	}else{
		printf("http://");
		i=4;
	}
	printf("%c",str[i++]);
	for( ; str[i]!='r' || str[i+1]!='u'; i++){
		printf("%c",str[i]);
	}
	printf(".ru");
	if(str[i+2]){
		str[i+1]='/';
		printf("%s\n",str+i+1);
	}
	return 0;
}
