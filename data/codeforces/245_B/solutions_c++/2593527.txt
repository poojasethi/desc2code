#include<stdio.h>
#include<string.h>
int main()
{
    char a[100];
    gets(a);
    int l=strlen(a),i=0,ll=0;
    if(a[0]=='h'){printf("http://");i+=4;}
    else {printf("ftp://");i+=3;}
    while(!ll||!(a[i]=='r'&&a[i+1]=='u')){putchar(a[i++]);ll++;}
    printf(".ru");i+=2;
    if(i<l)putchar('/');
    while(i<l)putchar(a[i++]);
    return 0;
}
