#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s,r;
    cin >> s;
    string::iterator i = s.begin();
    if (*i=='f') r += "ftp://", i+=3;
        else     r += "http://",i+=4;
    r+=*(i++);
    while ((*i!='r') || (*(i+1)!='u')) r+=*(i++);
    r+=".ru";
    i+=2;
    if (*i) r+="/";
    while (*i) r+=*(i++);
    cout << r;
}
