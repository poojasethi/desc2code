#include <iostream>
using namespace std;

int main(){
	string s;
	cin >> s;
	if( s[0] == 'h' ){
		cout << "http://";
		s = s.substr(4);
	}
	else{
		cout << "ftp://";
		s = s.substr(3);
	}
	int f = s.substr(1).find("ru") + 1;
	cout << s.substr(0,f) << "." << "ru";
	if( f < s.length() - 2 )
		cout << "/" << s.substr(f+2);
	cout << endl;
}
