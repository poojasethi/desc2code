#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n;
	string s;
	cin>>s;
	if(s[0]=='h')
		j=4;
	else if(s[0]=='f')
		j=3;
	for(i=0;i<j;i++)
		cout<<s[i];
	cout<<"://";
	j=s.rfind("ru");
	for(;i<j;i++)
		cout<<s[i];
	cout<<".ru";
	n=s.size();
	if(i!=n-2)
	{
		cout<<"/";
		for(i+=2;i<n;i++)
			cout<<s[i];
	}
	cout<<endl;
	return 0;
}
