#include <iostream>

using namespace std;

string s;

int main()
{
  cin >> s;
  if (*s.begin() == 'h') s.insert(4, "://");
  else s.insert(3, "://");
  for (int i = s.size() - 1; i >= 0; i--)
    if (s[i] == 'u' && s[i - 1] == 'r')
    {
      if (i + 1 != s.size()) s.insert(i + 1, "/");
      s.insert(i - 1, ".");
      break;
    }
  cout << s;
  return 0;
}
