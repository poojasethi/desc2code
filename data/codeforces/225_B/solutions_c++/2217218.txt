#include<iostream>
using namespace std;
long long f[500];
long long a[500];
int main(){
	int s,k;
	f[0]=1;
	while(cin>>s>>k){
		long long sum=f[0];
		for(int i=1;i<50;++i){
			f[i]=sum;
			if(i>=k)sum-=f[i-k];
			sum+=f[i];
		//	cout<<f[i]<<endl;
		}
		int m=0;
		for(int i=49;i>0;--i){
			if(f[i]<s){
				a[m++]=f[i];
				s-=f[i];
			}
			else if(f[i]==s){
				a[m++]=f[i];
				s-=f[i];
			}
		}
		cout<<m+1<<endl;
		for(int i=0;i<m;++i){
			cout<<a[i]<<" ";
		}
		cout<<0<<endl;
	}
	return 0;
}
