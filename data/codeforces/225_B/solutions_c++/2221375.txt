#include <cstdio>

long long f[100];
int a[100];
int l,m;

int main()
{
	int i,j,k;
	long long s;
	scanf("%I64d%d",&s,&k);
	f[1]=1,f[0]=0;
	for(i=2;;++i)
	{
		for(j=1;j<=k&&i-j>=0;j++)
		{
			f[i]+=f[i-j];
			if(f[i]>s)break;
		}
		if(f[i]>s)break;
	}
	while(s)
	{
		for(i=0;;++i)
			if(f[i]>s)break;
		a[m++]=i-1;
		s-=f[i-1];
	}
	if(m==1)
	{
		printf("2\n");
		printf("%I64d 0\n",f[a[0]]);
	}
	else
	{
		printf("%d\n",m);
		for(i=0;i<m;++i)
			if(i!=0)printf(" %I64d",f[a[i]]);
			else printf("%I64d",f[a[i]]);
		printf("\n");
	}
	return 0;
}
