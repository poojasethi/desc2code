#include <bits/stdc++.h>
using namespace std;
int arr[100]={0},k,s,ans[100],y,counter=0;

int main(){
    cin>>s>>k;
    arr[0]=0;
    arr[1]=1;
    for(y=2;arr[y-1]<s;y++)
    {
        for (long j=y-1; j>=0 && j>=y-k; --j)
            arr[y]+=arr[j];
    }
    for (;y>0 && s>0;--y)
    if (arr[y]<=s){
        s-=arr[y];
        ans[counter++]=arr[y];
    }
    cout<<counter<<endl;
        for(int j=0;j<counter;j++)
            cout<<ans[j]<<" ";

}
