#include <iostream>
#define LL long long
using namespace std;

const int MAXN=1000+10;
LL fib[MAXN], num[MAXN];  

int main(){
    int s, k, idx=0, i, j;
    fib[0]=0;
    fib[1]=1;
    cin>>s>>k;
    for (i=2; fib[i-1]<s; i++)
        for (j=i-1; j>=0 && j>=i-k; j--)
            fib[i]+=fib[j];
    for (; i>0 && s>0; i--)
        if (fib[i]<=s){
            s-=fib[i];
            num[idx++]=fib[i];
        }
    cout<<idx<<endl;
    for (i=0; i<idx; i++)
        cout<<num[i]<<" ";
    cout<<endl;
    return 0;
}
