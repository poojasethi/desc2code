#include <cstdio>
int main()
{
	int s,k,F[100]={0},index,ans[100],num_ans=0;
	scanf("%d%d",&s,&k);
	if(k>32)
		k=32;
	F[k]=1;
	for(int i=k+1;i<100;i++)
	{
		index=i-1;
		for(int j=1;j<=k;j++)
		{
			F[i]+=F[i-j];
			if(F[i]>s)
			{
				i=100; // break;
				j=k+1; // break;
			}
		}
	}
	for(;index>=k;index--)
	{
		if(F[index]>s)
			continue;
		ans[num_ans++]=F[index];
		s-=F[index];
	}
	printf("%d\n",num_ans+1);
	for(int i=0;i<num_ans;i++)
		printf("%d ",ans[i]);
	printf("0\n");
	return 0;
}
