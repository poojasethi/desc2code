#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;
#define N 100010
#define ll long long

ll f[N],num[N],cnt;

int main()
{
	ll s,k;
	scanf("%I64d%I64d",&s,&k);
	f[0] = 0;
	f[1] = 1;
	int i,j;
	for(i=2;f[i-1]<s;i++)
	{
		for(j=i-1;j>=0&&j>=i-k;j--)
			f[i] += f[j];
	}
	cnt = 0;
	for(;i>0&&s>0;i--)
	{
		if(f[i]<=s)
		{
			s -= f[i];
			num[cnt++] = f[i];
		}
	}
	printf("%d\n",cnt);
	for(i=0;i<cnt;i++) printf("%I64d ",num[i]);
	puts( "" );
	return 0;
}
