#include<cstdio>
#include<cstring>
#define LL long long
using namespace std;
LL fib[1005]={0,1},num[1005];
int main()
{
    int s,k;
    int idx=0,i,j;
    scanf("%d%d",&s,&k);
    for(i=2;fib[i-1]<s;++i)
    for(j=i-1;j>=0&&j>=i-k;--j)
    fib[i]+=fib[j];
    for(;i>0&&s>0;--i)
    if (fib[i]<=s)
    {
        s-=fib[i];
        num[idx++]=fib[i];
    }
    printf("%d\n",idx);
    for(int l=0; l<idx; ++l)
        printf("%d ",num[l]);
    return 0;
}
