#include <iostream>
#include <vector>
using namespace std;

int f[1000]={0};
int main()
{
int s,k; cin>>s>>k;
f[0]=1;f[1]=1;
int n=2;
for (;;n++)
{
  for(int i1=1;i1<=k;i1++)
   {
    int j=n-i1;
    f[n]+=f[j];
    if (j<0) break;
   }
   if (f[n]>=s) break;
 }
vector <int> v;
while (s>0)
{
 if (s>=f[n]) {s-=f[n];v.push_back(f[n]);}
 n--;
}
if (v.size()==1) v.push_back(0);
cout<<v.size()<<endl;
for (int i=0;i<v.size();i++) cout<<v[i]<<" ";
return 0;
}
