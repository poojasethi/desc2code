#include<iostream>
#include<conio.h>
using namespace std;
int main()
{
 int a[100]={0},s,j,k,i,ans[100]={0},c=0;
 a[0]=1;
 a[1]=1;
 cin>>s>>k;
 for(i=1;a[i]<=s;)
 {
     i++;
  if(i<=k)
  a[i]=2*a[i-1];
  else
  a[i]=(2*a[i-1]-a[i-k-1]);
 }
  j=i-1;
  while(s!=0)
  {
      if(a[j]<=s)
      {
       ans[c++]=a[j];
       s-=a[j];
      }
   j--;
  }
  if(c==1)
  {
      cout<<"2"<<endl;
      cout<<"0 "<<ans[0];
  }
  else{
  cout<<c<<endl;
 for(j=c-1;j>=0;j--)
 cout<<ans[j]<<" ";}
 return 0;
}
