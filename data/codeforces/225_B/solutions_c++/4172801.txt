#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int n,k,i,j;
    int ans[100],f[100]={0};
    cin>>n>>k;
    f[0]=0,f[1]=1;
    for(i=2;f[i-1]<n;++i)
        for(j=i-1;j>=0&&j>=i-k;--j)
        f[i]+=f[j];
    ans[0]=0;
    for(;i>0&&n>0;--i)
        if(n>=f[i])
        {
            n-=f[i];
            ans[++ans[0]]=f[i];
        }
    cout<<ans[0]<<endl;
    for(int i=1;i<=ans[0];++i)
        cout<<ans[i]<<' ';
    cout<<endl;
}
