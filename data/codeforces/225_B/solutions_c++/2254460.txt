#include <cstdio>


main(){
	int s,k;
	scanf("%d %d",&s,&k);
	
	
	int SIZE = 1000000;
	int T[1000000] = {0};
	T[0] = 1;
	for (int i=1; i<SIZE; i++){
		for (int j=i-1; j>=i-k && j>=0; j--)
			T[i] += T[j];
		if (T[i]>(int)1e9 || T[i]<0){
			SIZE = i-1;
			break;
		}
	}

	int tmp[1000000]; int kol = 0;
	for (int i=SIZE; i>=0; i--){
		if (T[i]<=s){
			s -= T[i]; tmp[kol++] = T[i];
		}
		if (s == 0){
			tmp[kol++] = 0;
			break;
		}
	}
	
	printf("%d\n",kol);
	for (int i=0; i<kol; i++)
		printf("%d ",tmp[i]);
	printf("\n");
}
