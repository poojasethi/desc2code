# include <stdio.h>

int n,m;

long long k,len=3;

long long diz[100050];

main()
{
	scanf("%d %d",&n,&m);
	
	diz[1]=0;diz[2]=1;k=1;
	
	while(k <= n)
	{
		diz[len]=k;
		
		k*=2;
		
		if(len-m > 0) k-=diz[len-m];
		
		len++;
	}
	
	int ans=0;
	
	int arr[100050],uz=0;
	
	for	(int i=len-1;i>=1;i--)
	{
		if(n >= diz[i]) n-=diz[i],arr[uz++]=diz[i],ans++;
	}
	
	printf("%d\n",ans);
	
	for	(int i=0;i<uz;i++) printf("%d ",arr[i]);
	
	getchar();getchar();
}
