#include<cstdio>
int v[100],b[100],o[100],s,k,L=1;
int main(){
	scanf("%d%d",&s,&k);
	int p=1;
	b[0]=o[0]=1;
	for(;b[p-1]<=s;p++){
		b[p]=o[p-1]-((p<k+1)?0:o[p-k-1]);
		o[p]=b[p]+o[p-1];
	}
	while(s){
		int x=0;
		for(;b[x]<=s;x++);
		v[L++]=b[x-1];
		s-=b[x-1];
	}
	printf("%d\n",L);
	for(int i=0;i<L;i++) printf("%d ",v[i]);
}
