#include<stdio.h>
int que[100];
int main()
{
	int top,s,k,i,j;
	while(scanf("%d %d",&s,&k)!=EOF)
	{
		que[0]=0;
		que[1]=1;
		for(i=2;;i++)
		{
			que[i]=0;
			for(j=1;j<=k && j<=i;j++)
				que[i]+=que[i-j];
			if(que[i]>s)
				break;
			//printf("%d %d\n",i,que[i]);
		}
		top=i;
		int t=s,tt=0;
		for(i=top;i>=0;i--)
			if(t>=que[i])
			{
				t-=que[i];
				tt++;
			}
		printf("%d\n",tt);
		printf("0");
		for(i=top;i>0;i--)
			if(s>=que[i])
			{
				s-=que[i];
				printf(" %d",que[i]);
			}
		printf("\n");
	}
}
