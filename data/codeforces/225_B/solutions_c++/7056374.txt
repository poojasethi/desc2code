#include<cstdio>
#define LL long long
LL a[100]={1},ans[100];

int main()
{
	int s,k,i=0,cnt=0;
	scanf("%d%d",&s,&k);
	LL d=1,sum=0;
	while(a[i]<s){
		if(i<k){a[++i]=d;d+=a[i];}
		else {a[i+1]=2*a[i]-a[i-k];++i;}
	}
	if(s==a[i]){printf("2\n%I64d 0\n",a[i]);return 0;}
	--i;
	while(sum<s){
		if(sum+a[i]<=s){sum+=a[i];ans[cnt++]=a[i];}
		--i;
	}
	printf("%d\n%I64d",cnt,ans[0]);
	for(i=1;i<cnt;++i)printf(" %I64d",ans[i]);
	puts("");
	return 0;
}
