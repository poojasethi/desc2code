#include <cstdio>
#define maxs 1000000002
using namespace std;
int n,k,s,f[1000];
bool d[1000];
int main(){
	scanf("%d%d",&s,&k);
	f[0]=0;		f[1]=1;		f[2]=1;	n=2;
	int i,j;
	while (f[n]<=s){
		n++;
		j=n-k-1;
		f[n]=2*f[n-1];
		if (j>0)
			f[n]-=f[j];
	}
	for (i=0;i<=n;i++)	d[i]=false;
	int m=0;
	for (i=n;i>=0;i--)
		if (s>=f[i]){
			d[i]=true;
			m++;
			s-=f[i];
		}
	printf("%d\n",m);
	for (i=0;i<=n;i++)
		if (d[i]==true)
			printf("%d ",f[i]);
	return 0;
}
