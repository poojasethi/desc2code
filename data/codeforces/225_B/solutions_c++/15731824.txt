#include <iostream>
using namespace std;
long long arr[100000], ans[100000];
int main()
{
	long long n, k;
	int i;
	cin >> n >> k;
	arr[0] = 0, arr[1] = 1;;
	for (i = 2;; i++)
	{
		for (int j = i - 1; j >= i - k && j >= 0; j--)
			arr[i] += arr[j];

		if (arr[i] > n) break;
	}
	int pos = 0;
	for (int j = i; j >= 0; j--)
	{
		if (arr[j] <= n)
		{
			ans[pos] = arr[j];
			n -= arr[j];
			pos++;
			
		}
	}

	cout << pos << endl;
	for (int j = 0; j<pos; j++) cout << ans[j] << " ";
	cout << endl;


	return 0;
}
