#include <iostream>
#define big long long
using namespace std;
big a[101],b[101],s,z;
int n=60,k,l,r,m,d=1;
big kb(big b)
{
    if(b && a[b]) return a[b];
    for(big i=b-1; i>=b-k && i; i--) a[b]+=kb(i);
    return a[b];
}
int main(int argc, const char * argv[])
{
    cin>>s>>k;
    a[1]=1;
    kb(n);
    while (s) {
        l=2,r=n;
    while (l<=r) {
        m = l+(r-l)/2;
        if(a[m]<=s) z=a[m],l=m+1;
        else r=m-1;
    }
        s-=z,b[d]=z,d++;
    }
    cout<<d<<"\n";
    while (d--) cout<<b[d]<<" ";
    return 0;
}
