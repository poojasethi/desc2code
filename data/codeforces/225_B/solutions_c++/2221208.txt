# include<iostream>
using namespace std;
long long a[1111],s[1111];
int main()
{
   long long n,k;
   int t=0,i;
   cin>>n>>k;
   s[0]=0;
   s[1]=1;
   for(i=2;s[i-1]<n;i++)
   {
       for(int j=1;j<=k&&i-j>=0;j++)
          s[i]+=s[i-j];
   }
   for(int j=i;j>=0;j--)
   {
       if(n-s[j]>=0)
       {
          n=n-s[j];
          a[t++]=s[j];
          if(n==0)
             break;
       }
   }
   cout<<t<<endl;
   for(int j=0;j<t-1;j++)
      cout<<a[j]<<" ";
   cout<<a[t-1]<<endl;
   return 0;
}
