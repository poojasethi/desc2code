#include<iostream>
#include<algorithm>
using namespace std;
long long a[50], ans[50], m;
int main(){
	
	long long s, k, i;
	
	cin >> s >> k;
	a[0] = 1;
	for(i = 1; i < k && a[i-1] <= s; i++) a[i] = a[i-1]*2;
	
	if(i == k && a[i-1] <= s){
		a[k] = a[k-1]*2-1;
		for(i = k+1; a[i-1] <= s; i++) a[i] = 2*a[i-1]-a[i-k-1];
	}
	
	for(i--; i >= 0; i--) if(a[i-1] <= s) s -= a[i-1], ans[m++] = a[i-1];
	
	if(m == 1) m++, a[1] = 0;
	cout << m << endl;
	for(i = 0; i < m; i++) cout << ans[i] << (i == m-1?"\n":" ");

}
