#include <iostream>
using namespace std;
int s,k,f[105],res=0,i=1,t[105];
int main(){
    cin>>s>>k;
    f[0]=1; f[1]=1;
    while (f[i]<=s){
        i++;
        f[i]=2*f[i-1]-(i<k+1 ? 0:f[i-k-1]);
    }
    for (int j=i-1; j>=0; j--){
        if (s-f[j]>=0) {t[res++]=f[j]; s-=f[j];}
        if (!s) break;
    }
    cout<<res+1<<endl;
    for (int j=0; j<res; j++) cout<<t[j]<<" ";
    cout<<0;
    return 0;
}
