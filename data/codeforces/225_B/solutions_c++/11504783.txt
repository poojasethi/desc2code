#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll s,k,b[100],o[100],v[100],i,n,L=1;
main()
{
    scanf("%I64d%I64d",&s,&k);
    int p=1;
    o[0]=b[0]=1;
    for (;b[p-1]<=s;p++)
        {
            b[p]=o[p-1]-((p<k+1)?0:o[p-k-1]);
            o[p]=o[p-1]+b[p];
        }
    while (s)
    {
        int x=0;
        for (;b[x]<=s;x++);
        v[L++]=b[x-1];
        s-=b[x-1];
    }
    printf("%I64d\n",L);
    for (i=0;i<L;i++) printf("%I64d ",v[i]);
}
