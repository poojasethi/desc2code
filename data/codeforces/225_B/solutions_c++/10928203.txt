#include <stdio.h>
#include <map>

using namespace std;
int s, k, t, l, j;
int a[1001];
map <int, int> f;
void solve(){
	scanf("%d %d", &s, &k);
	f[k - 1] = t = 1;
	for(int i = k ; ; i ++){
		f[i] = t;
		t += f[i];
		t -= f[i - k];
		if(f[i] > s) {j = i ; break;}
	}
	for(int i = j ;  ; i --){
		//printf("%d ", f[i]);
		if(f[i] <= s && f[i]) s -= f[i], a[l ++] = f[i];
		if(s == 0) break;
	} 
	if(l > 1) printf("%d\n", l);
	else printf("2\n0 ");
	for(int i = 0 ; i < l ; i ++) printf("%d ", a[i]);
}
int main(){
	solve();
}
