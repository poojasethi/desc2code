#include<stdio.h>
#include<iostream>
using namespace std;
int a[100000], ans[100000];
int main()
{
	int k, n;
	cin >> n >> k;
	a[0]=0;
	a[1]=1;
	int sum=0;
	int i=1;
	while(sum<n)
	{
		if((i+1-k)>=0)
		{
			sum-=a[i-k];
		}
		sum+=a[i];
		a[i+1]=sum;
		i++;
	}
	
	int x=0;
	for(int j=i;j>=0;j--)
	{
		if((n-a[j])>=0)
		{
			n-=a[j];
			ans[x]=a[j];
			x++;
			if(n==0)
				break;
		}
	}
	if(x == 1)
	{
		ans[x] = 0;
		x++;
	}
	cout << x << endl;
	for(int j=0;j<x;j++)
	{
		cout << ans[j] << " ";
	}
	cout << endl;
	return 0;
}
