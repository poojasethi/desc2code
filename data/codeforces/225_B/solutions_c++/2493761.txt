#include<cstdio>
#include<cstring>
#include<cstdlib>
int f[10000];
int ans[10000];

int F(int x)
{
	if(x<=0)return 0;
	return f[x];
}
int main()
{
	int s,k;
	scanf("%d%d",&s,&k);
	f[1]=1; f[2]=1;
	int ii;
	for(ii=3;f[ii-1]<s;ii++)
		f[ii]=f[ii-1]*2-F(ii-k-1);
	int len=ii;
	for(int i=len;i>=1;i--)
		if(s>=f[i])
		{
			s-=f[i];
			ans[++ans[0]]=f[i];
		}
	printf("%d\n%d",ans[0],ans[1]);
	for(int i=2;i<=ans[0];i++)
		printf(" %d",ans[i]);
	return 0;
}
