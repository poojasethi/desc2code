#include <iostream>

using namespace std;

const int N = 50;

int v[N], sol[N];
int k, s, fin, dif;

int main() {
	cin >> s >> k;
	
	sol[++sol[0]] = 0;
	v[1] = 1;
	for (int i = 2; i <= 50; ++i) {
		for (int j = i - 1; j >= max(1, i - k); --j)
			v[i] += v[j];
		
		if (v[i] > s) {
			fin = i;
			break;
		}
	}
	
	for (int i = fin; i >= 1; --i)
		if (v[i] <= s) {
			sol[++sol[0]] = v[i];
			s -= v[i];
		}
	
	cout << sol[0] << '\n';
	for (int i = 1; i <= sol[0]; ++i)
		cout << sol[i] << ' ';
	
	return 0;
}
