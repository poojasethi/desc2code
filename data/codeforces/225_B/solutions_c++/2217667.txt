# include <stdio.h>
# include <algorithm>

using namespace std;

int n,m;
long long k,len=3;
long long t[100050];
int main()
{
 scanf("%d %d",&n,&m);
 
 t[1]=0;
 t[2]=1;
 k=1;
 
 while(k <= n)
 {
  t[len]=k;
  
  k*=2;
  
  if(len-m > 0) k-=t[len-m];
  
  len++;
 }
 
 int ans=0;
 
 int arr[100050],uz=0;
 
 for (int i=len-1;i>=1;i--)
 {
  if(n >= t[i]) n-=t[i],arr[uz++]=t[i],ans++;
 }
 
 printf("%d\n",ans);
 
 for (int i=0;i<uz;i++) printf("%d ",arr[i]);
 
 getchar();getchar();
}
