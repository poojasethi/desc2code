#include <iostream>

using namespace std;

int main()
{
    int f[111],p[111];
    int s,k;
    cin>>s>>k;
    f[0]=p[0]=1;
    for(int i=0;f[i]<=s;i++)
    {
        if(i<k)
        f[i+1]=p[i];
        else
         f[i+1]=p[i]-p[i-k];
        p[i+1]=p[i]+f[i+1];
    }
    int l=1;
    int vis[111];
   vis[0]=0;
   while(s)
    {
		int x=0;
		for(;f[x]<=s;x++);
		vis[l++]=f[x-1];
		s-=f[x-1];
	}
    cout<<l<<endl;
    for(int i=0;i<l;i++)
        cout<<vis[i]<<' ';
    cout<<endl;
    return 0;
}
