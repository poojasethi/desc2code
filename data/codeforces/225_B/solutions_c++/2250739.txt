#include <iostream>
#include <algorithm>

#define MAX 1000000001

using namespace std;

int main(){
	long long F[300], s, k, res[300], m = 0;
	int n;
	cin>>s>>k;
	n = 1;
	F[1] = 1;
	while(F[n] < MAX){
		F[++n] = 0;
		for (int i = 1; i <= k && n - i >= 0; i++) F[n] += F[n-i];
		//cout<<n<<" "<<F[n]<<endl;
	}
	while(s > 0){
		long long i = *(lower_bound(F, F + n, s+1)-1);
		res[m++] = i;
		s -= i;
	}
	if (m == 1){
		res[m++] = 0;
	}
	cout<<m<<endl;
	for (int i = 0; i < m; i++) cout<<res[i]<<" ";
	cout<<endl;
	return 0;
}
