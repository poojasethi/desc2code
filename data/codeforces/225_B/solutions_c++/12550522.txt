#include <bits/stdc++.h>
using namespace std;


int main(){
    int fo[100]={0},k,s,ans[100],i,counter=0;
    cin>>s>>k;
    fo[0]=0;
    fo[1]=1;
    for(i=2;fo[i-1]<s;i++)
    {
        for (long j=i-1; j>=0 && j>=i-k; --j)
            fo[i]+=fo[j];
    }
    for (;i>0 && s>0;--i)
    if (fo[i]<=s){
        s-=fo[i];
        ans[counter++]=fo[i];
    }
    cout<<counter<<endl;
        for(int j=0;j<counter;j++)
            cout<<ans[j]<<" ";

}
