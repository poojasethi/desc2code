#include<iostream>
using namespace std;
long long fib[1000];
int main(){
	int s,k,p;
	cin >> s >> k;
	fib[0] = 0, fib[1] = 1;
	for(int i = 2;; i++) {
		long long &v = fib[i];
		for(int j = i-1; j >= 0 && j >= i -k; j-- )
				v += fib[j];
		if(v > (int)1e9){
		   	p = i;
			break;
		}
	}
	int __ans[100],len = 0;
	for(int i = p; i>=0; i--) {
		if(fib[i] <= s){
			__ans[len++] = (int)fib[i];
			s -= fib[i];
		}
	}
	cout<< len << endl;
	for(int i = 0; i < len; i++)
		cout<< __ans[i] <<" "; cout<<endl;
}
