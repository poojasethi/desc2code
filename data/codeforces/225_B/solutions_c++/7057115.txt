#include<stdio.h>
#include<string.h>
__int64 f[1005],num[1005];
int main()
{
	__int64 s,k,i,j,m,n;
	while(scanf("%I64d%I64d",&s,&k) != EOF)
	{
		memset(f,0,sizeof(f));
		memset(num,0,sizeof(num));
		f[0]=0;f[1]=1;
		for(i=2;f[i-1]<s;i++)
			for(j=i-1;j>=0&&j>=i-k;j--)
				f[i]+=f[j];
		n=i;m=0;
		for(i=n-1;i>=0&&s>=0;i--)
			if(s-f[i]>=0)
			{
				s-=f[i];
				num[m++]=f[i];
			}
		printf("%I64d\n",m);
		for(i=0;i<m-1;i++)
			printf("%I64d ",num[i]);
		printf("%I64d\n",num[m-1]);
	}
	return 0;
}
