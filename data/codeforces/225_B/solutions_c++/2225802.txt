#include <cstdio>
int v[100],kb[100],ps[100],s,k,L=1;
int main(){
	scanf("%d%d",&s,&k);
	int p=1;
	kb[0]=ps[0]=1;
	for(;kb[p-1]<=s;p++){
		kb[p]=ps[p-1]-((p<k+1)?0:ps[p-k-1]);
		ps[p]=kb[p]+ps[p-1];
	}
	while(s){
		int x=0;
		for(;kb[x]<=s;x++);
		v[L++]=kb[x-1];
		s-=kb[x-1];
	}
	printf("%d\n",L);
	for(int i=0;i<L;i++) printf("%d ",v[i]);
}
