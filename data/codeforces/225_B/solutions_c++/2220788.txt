#include <cstdio>
#include <algorithm>

using namespace std;

int val[110];
int res[110];

int a,b,c,d;

int s,k;

int main()
{
    scanf("%d%d",&s,&k);
    val[1]=1;
    a=1;
    while (val[a]<=s)
    {
        a++;
        for (b=max(1,a-k); b<a; b++)
        {
            val[a]+=val[b];
            if (val[a]>s) break;
        }
    }
    a--;
    b=0;
    while (s>0)
    {
        if (val[a]<=s) s-=val[a], b++, res[b]=val[a];
        a--;
    }
    printf("%d\n",b+1);
    printf("0 ");
    for (a=1; a<=b; a++) printf("%d ",res[a]);
}
