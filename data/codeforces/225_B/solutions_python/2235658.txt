l = [0,1]
s,k = map(int,raw_input().split())
while 1:
    t = sum(l if len(l) < k else l[-1:-1-k:-1])
    if t > s: break
    l.append(t)
r = []
for i in xrange(len(l)-1,-1,-1):
    if s == 0:
        if len(r) < 2: r.append(0)
        break
    elif s >= l[i] and l[i] not in r:
        r.append(l[i])
        s -= l[i]
print len(r)
print ' '.join(map(str,r))
