s,k=map(int,raw_input().split())
f=[0,1]
ss=1
tpos=1
tpos_d=-k+2
while f[-1]<s:
    f.append(ss)
    ss=ss+ss
    if tpos_d>=0:
        ss-=f[tpos_d]
    tpos+=1
    tpos_d+=1
rere=[0]
x=1
while s>0:
    while f[-x]>s:
        x+=1
    s-=f[-x]
    rere.append(f[-x])
print len(rere)
print " ".join(map(str,rere))
