s, k = map(int, raw_input().split())
fib = [0, 1, 1]

while(True):
    fib.append(sum(fib[-k:]))
    if fib[-1] > s: 
        del fib[-1]
        break

ans = []

while(True):
    x = fib.pop()
    if x <= s: 
        s -= x
        ans.append(x)
    if s == 0:
        break
ans.append(0)
print len(ans)
print ' '.join(map(str, ans))
