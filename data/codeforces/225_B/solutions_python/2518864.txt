
def main():
    cin = lambda: map(int, raw_input().split())
    [s, k] = cin()
    ans = []
    a = [0, 1]
    while a[-1] < s:
        a.append(sum(a[-k:]))
    for i in reversed(a):
        if i <= s:
            ans.append(i)
            s -= i
    print len(ans)
    for a in ans:
        print a,

if __name__ == '__main__':
    main()
