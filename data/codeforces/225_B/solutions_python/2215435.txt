s,k=map(int,raw_input().split())
f = [0]*200+[1]
while True:
  l = min(len(f),k)
  f.append(sum(f[-l:]))
  if s<f[-1]:
    break

res = []
while s>0:
  if s in f:
    res.append(s)
    res.append(0)
    s=0
  else:
    i=0
    while not f[i]<s<f[i+1]:
      i+=1
    res.append(f[i])
    s-=f[i]
res.reverse()
  
print len(res)
