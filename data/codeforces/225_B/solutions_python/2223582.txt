import sys
if sys.version_info[0] == 2: input = raw_input
s, k = map(int,input().split())
if s == 1:
    print('2\n0 1')
    sys.exit(0)
elif s==2:
    print('2\n0 2')
    sys.exit(0)
n = 2
ans = []
a = [0] * 100
a[0] = a[1] = 1
while a[n-1] < s:
    a[n] = 2*a[n-1]-(a[n-k-1] if n-k-1>=0 else 0)
    n += 1

for i in range(n-1,-1,-1):
    if a[i] <= s:
        ans.append(a[i])
        s -= a[i]
        if s == 0: break
print(len(ans) + 1)
print('0 ' + " ".join(map(str,ans)))
