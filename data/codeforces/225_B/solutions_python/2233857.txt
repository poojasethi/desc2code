s,k=map(int,raw_input().split())
f,sm=[0,1,1],[0]
ssum=0
while ssum<s :
    i=k+2
    sumi=1
    while sumi<=s-ssum :
        sumi+=f[i-k]
        i+=1
        if ssum==0 :
            f.append(2*f[i-k-1])
            if i-k-k-1 >=0 :
                f[i-k]=f[i-k]-f[i-k-k-1]
    ssum+=f[i-k-1]
    sm.append(f[i-k-1])
print(len(sm))
print(" ".join([str(i) for i in sm]))
