s, k = map(int, raw_input().split())
f = [1, 1]
while f[-1] < 10 ** 9 + 10:
    if len(f) > k:
        f.append(f[-1] * 2 - f[-k-1])
    else:
        f.append(f[-1] * 2)
ans = [0]
i = -1
while s > 0:
    x = f[i]
    if s >= x:
        ans.append(x)
        s -= x
    i -= 1
print len(ans)
