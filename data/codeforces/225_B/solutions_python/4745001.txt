s,k=map(int,raw_input().split())
t={}
def f(a,b):
    s=0
    if b < a:  s=0
    elif b == a: s=1
    elif b in t:
        s=t[b]
    elif a > 32: s=2**(b-a-1)
    else:
        for i in range(b-a,b):
            s+=f(a,i)
    t[b] = s
    return s

i=max(0,k-2)
e=0
F=[]
while e < s:
    i+=1
    e=f(k,i)
    if e < s or 0 in F:
        F+=[e]
   
r=[]
for e in reversed(F):
    if e <= s and e:
        r+=[`e`]
        s-=e
if len(r)==1 and 0 in F:
    r+=['0']
print len(r)
print ' '.join(r)
