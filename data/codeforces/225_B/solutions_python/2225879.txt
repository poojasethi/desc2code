n, k = map(int, raw_input().split())

bona = [1]

cnt = 0

while bona[-1] < n:
    if cnt < k:
        bona.append(2 ** cnt)
    else:
        bona.append(sum(bona[-k:]))
    cnt += 1

ans = []

while n > 0:
    while bona[-1] > n:
        bona.pop()
    ans.append(bona[-1])
    n -= bona[-1]

if len(ans) == 1:
    ans.append(0)

print len(ans)
print ' '.join(map(str, ans))
