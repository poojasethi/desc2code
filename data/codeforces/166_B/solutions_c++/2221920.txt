#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<cmath>
using namespace std;
struct point
{
	long long x,y;
}a[100005],p;
inline long long  cross(point o,point a,point b)
{
	return (a.x-o.x)*(b.y-o.y)-(b.x-o.x)*(a.y-o.y);
}
int main()
{
	int n,m,i,l,r;
	int d=0;
	scanf("%d",&n);
	for(i=n;i;i--)
		scanf("%I64d%I64d",&a[i].x,&a[i].y);
	scanf("%d",&m);
	for(int i=1;i<=m;i++)
	{
		scanf("%I64d%I64d",&p.x,&p.y);
		l=2;r=n;
		while(l<r)
		{
			int mid=(l+r)>>1;
			if(cross(a[1],a[mid],p)<=0LL)
				r=mid;
			else
				l=mid+1;
		}
		if(l==2||cross(a[1],a[n],p)>=0LL)
			continue;
		if(cross(a[l],a[l-1],p)<0LL)
			++d;
	}
	if(d==m)
		printf("YES\n");
	else
		printf("NO\n");
	return 0 ;
}
