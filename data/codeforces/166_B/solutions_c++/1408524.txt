#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
#define MAXSIZE 300010
using namespace std ;
struct P
{
	double x , y ;
};
P p[ MAXSIZE ] , t ;
int n , m ;
double OnTheRight( P p1 , P p2 , P p0 )
{
	return ( p1.x - p0.x ) * ( p2.y - p0.y ) - ( p1.y - p0.y ) * ( p2.x - p0.x ) ;
}
int main( )
{
	int i , j ;
	int Min , Mid , Max ;
	while( ~scanf( "%d" , &n ) )
	{
		for( i = 0 ; i < n ; i++ )
			scanf( "%lf%lf" , &p[ i ].x , &p[ i ].y ) ;
		scanf( "%d" , &m ) ;
		bool b = 1 ;
		for( i = 0 ; i < m ; i++ )
		{
			scanf( "%lf%lf" , &t.x , &t.y ) ;	
			Min = 1 , Max = n-1 ;
			while( Max > Min )
			{
				Mid = ( Max + Min + 1 ) / 2 ;
				if( OnTheRight( p[ Mid ] , t , p[ 0 ] ) <= 0 )
					Min = Mid ; 
				else
					Max = Mid - 1 ;
			}
			if( OnTheRight( p[ 1 ] , t , p[ 0 ] ) >= 0 )
				b = 0 ;			
			if( Min == n-1 )
				b = 0 ;
			else if( OnTheRight( p[ Min+1 ] , t , p[ Min ] ) >= 0 )
				b = 0 ;
		}
		if( b )
			printf( "YES\n" ) ;
		else
			printf( "NO\n" ) ;
	}
	return 0 ;
}
