#include<stdio.h>
#include<string.h>
typedef __int64 lld;
struct P
{
    lld x,y;
}p[100100];
lld cross(P a,P b,P c)
{
    return (b.x-a.x)*(c.y-a.y)-(c.x-a.x)*(b.y-a.y);
}
int main()
{
    int n,m;
    int i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%I64d%I64d",&p[i].x,&p[i].y);
    scanf("%d",&m);
    int num=0;
    for(i=0;i<m;i++)
    {
        P a;
        scanf("%I64d%I64d",&a.x,&a.y);
        int left=1,right=n-1,mid,best=1;
        while(left<=right)
        {
            mid=(left+right)>>1;
            if(cross(p[0],a,p[mid])>0)
                left=mid+1;
            else
            {
                best=mid;
                right=mid-1;
            }
        }
        if(best==1||cross(p[0],a,p[n-1])==0)
            continue;
        if(cross(p[best],a,p[best-1])<0)
            num++;
    }
    if(num==m)
        printf("YES\n");
    else
        printf("NO\n");
    return 0;
}
