#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct point{
    long long x,y;
    point(){}
    point(long long a,long long b){x=a;y=b;}
}p[100010];
point operator-(point a,point b){
    return point(a.x-b.x,a.y-b.y);
}
long long operator*(point a,point b){
    return a.x*b.y-a.y*b.x;
}
int n,m,flag;
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;++i)scanf("%I64d%I64d",&p[i].x,&p[i].y);
    scanf("%d",&m);
    for(int i=1;i<=m;++i){
        scanf("%I64d%I64d",&p[0],&p[0].y);
        int l=2,r=n,mid;
        while(l<=r){
            mid=(l+r)>>1;
            if((p[0]-p[1])*(p[mid]-p[1])>=0)l=mid+1;
            else r=mid-1;
        }
        if((p[0]-p[1])*(p[n]-p[1])>=0){flag=1;break;}
        if((p[0]-p[1])*(p[2]-p[1])<=0){flag=1;break;}
        if((p[0]-p[r])*(p[r+1]-p[r])<=0){flag=1;break;}
    }
    printf("%s\n",flag?"NO":"YES");
    return 0;
}
