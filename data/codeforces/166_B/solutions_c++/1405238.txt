#include <stdio.h>
#include <algorithm>
#include <cstdlib>
#include <iostream>
using namespace std;

const long long maxsize=100000000+1;
const int N=100000+1;
struct point { long long x,y; };
point A[N],B;
int n,m,i,j;
bool answer;
long long tmp;

int f(point P0,point P1,point P2)
{
 tmp=(P2.x-P0.x)*(P1.y-P0.y)-(P1.x-P0.x)*(P2.y-P0.y);
 if ( tmp==0 ) return 0;
 return tmp/abs(tmp);
}

bool analyze(point B)
{
 int L=1;
 int R=n-1;
 int M;
 while ( L<R-1 )
  {
   M=(L+R)/2;
   if ( f(A[0],A[M],B)==1 ) L=M;
                       else R=M;
  }
 if ( f(A[L],A[R],B)!=1 ) return false;
                     else return true;
}

main()
{
 cin>>n;
 for ( i=0;i<n;i++ )
  {
   cin>>A[i].x>>A[i].y;
   A[i].x+=maxsize;
   A[i].y+=maxsize;
  }

 answer=true;
 cin>>m;
 for ( j=0;j<m;j++ )
  {
   cin>>B.x>>B.y;
   B.x+=maxsize;
   B.y+=maxsize;
   if ( f(A[0],A[1],B)!=1 ) answer=false;
   if ( f(A[n-1],A[0],B)!=1 ) answer=false;
   if ( answer ) answer=analyze(B);
  }
 
 if ( answer ) cout<<"YES"<<endl;
          else cout<<"NO"<<endl;
// system("B.exe"); //
}
