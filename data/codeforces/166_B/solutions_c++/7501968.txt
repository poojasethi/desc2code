#include <bits/stdc++.h>
#define long long long
#define ii pair<int, int>
#define iii pair<ii, int>
#define X first
#define Y second
const int N = 200005;
using namespace std;

int n, m, nn;
iii s[N], a[N];
ii O;

void operator -= (ii& A, ii B) {A.X -= B.X; A.Y -= B.Y;}
bool CCW(ii A, ii B, ii C) {C -= B; B -= A; return (long)B.X * C.Y > (long)B.Y * C.X;}

int main()
{
    scanf("%d", &n);
    int i, j, x, y;
    for(i = 1; i <= n; i++) {
        scanf("%d %d", &x, &y);
        a[i] = iii(ii(x, y), 1);
    }
    scanf("%d", &m);
    for(i = 1; i <= m; i++) {
        scanf("%d %d", &x, &y);
        a[i + n] = iii(ii(x, y), 2);
    }
    nn = m + n;
    sort(a + 1, a + 1 + nn);
    int k = 1, t;
    for(i = 1; i <= nn; i++) {
        while (k > 2 && CCW(s[k - 2].X, s[k - 1].X, a[i].X)) k--;
        s[k++] = a[i];
    }
    for(i = nn, t = k + 1; i; i--) {
        while (k > t && CCW(s[k - 2].X, s[k - 1].X, a[i].X)) k--;
        s[k++] = a[i];
    }
    for(i = 1; i < k; i++) if (s[i].Y == 2) {printf("NO"); return 0;}
    printf("YES");
    return 0;
}
