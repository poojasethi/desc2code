#include<cstdio>
const int M=100010;
struct point{
    double x,y;
}a[M],p;
double mult(point sp,point ep,point op){
    return (sp.x-op.x)*(ep.y-op.y)-(ep.x-op.x)*(sp.y-op.y);
}
int main(){
    int n,m;
    while(~scanf("%d",&n)){
        for(int i=0;i<n;i++){
            scanf("%lf%lf",&a[i].x,&a[i].y);
        }
        scanf("%d",&m);
        bool flag=true;
        while(m--){
            scanf("%lf%lf",&p.x,&p.y);
            int l=1;
            int r=n-1;
            while(l<=r){
                int mid=(l+r)/2;
                if(mult(a[mid],p,a[0])>=0){
                    r=mid-1;
                }else {
                    l=mid+1;
                }
            }
            if(l<2||mult(a[n-1],p,a[0])<=0||mult(a[l-1],p,a[l])<=0) {
                flag=false;
            }
        }
        if(flag) puts("YES");
        else puts("NO");
    }
    return 0;
}
