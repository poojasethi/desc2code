#include<iostream>
#include<algorithm>
using namespace std;

#define maxn 400005

long long x[maxn],y[maxn];
int n,m;

long long xmulti(long long ax,long long ay,long long bx,long long by)
{
    return ax*by-ay*bx;
}

int main()
{
    int i;
    cin>>n;

    for(i=0;i<n;i++) cin>>x[i]>>y[i];
    cin>>m;

    long long X,Y;

    bool ok=true;

    for(i=0;i<m;i++)
    {
        cin>>X>>Y;

        if(xmulti(X-x[0],Y-y[0],x[1]-x[0],y[1]-y[0])<=0) ok=false;//

        if(xmulti(X-x[0],Y-y[0],x[n-1]-x[0],y[n-1]-y[0])>=0) ok=false;

        if(!ok) continue;

        int l=0,r=n-1,mid;

        int k;

        while(l+1<r)
        {
            mid=(l+r)>>1;

            long long tv=xmulti(X-x[0],Y-y[0],x[mid]-x[0],y[mid]-y[0]);

            if(tv>=0) l=mid;
            else r=mid;

        }

        if(xmulti(x[r]-x[l],y[r]-y[l],X-x[l],Y-y[l])>=0) ok=false;

    }

    if(ok) cout<<"YES"<<endl;
    else cout<<"NO"<<endl;

return 0;
}
