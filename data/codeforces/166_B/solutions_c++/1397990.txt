#include <cstdio>
#include <cstdlib>

using namespace std ;

typedef long long LL;

struct DOT{LL x, y;} ;

inline LL cross(DOT o, DOT a, DOT b)
{
    return (a.x - o.x) * (b.y - o.y) - (b.x - o.x) * (a.y - o.y);
}

int N, M, K;
int cnt;

DOT hull[100010];

int main()
{
    scanf("%d", &N);
    for(int i(N); i; -- i)
        scanf("%I64d%I64d", &hull[i].x, &hull[i].y);
    scanf("%d", &M);
    for(int i(1); i<= M; ++ i)
    {
        DOT dot;
        int l= 2, r= N;

        scanf("%I64d%I64d", &dot.x, &dot.y);
        while(l < r)
        {
            int mid= (l + r) >> 1;
            if(cross(hull[1], hull[mid], dot) <= 0ll)
                r= mid;
            else
                l= mid + 1;
        }
        if(l == 2 || cross(hull[1], hull[N], dot) >= 0ll)
            continue;
        if(cross(hull[l], hull[l - 1], dot) < 0ll)
            cnt++;
    }
    if(cnt == M)
        printf("YES\n");
    else
        printf("NO\n");
    return 0 ;
}
