#include <iostream>
#include <cstdio>
using namespace std;
const long long N =101000;
struct p{
      long long x,y;
} a[N];
long long n,m;

long long pro(p o,p b,p c){
      return (b.x-o.x)*(c.y-o.y)-(c.x-o.x)*(b.y-o.y);
}
int  main(){
      //freopen("test.in","r",stdin);
      scanf("%I64d",&n);
      for (long long i=1;i<=n;i++)
            scanf("%I64d%I64d",&a[i].x,&a[i].y);
      scanf("%I64d",&m);
      for (long long i=1;i<=m;i++){
            p b;
            scanf("%I64d%I64d",&b.x,&b.y);
            if (pro(a[1],a[2],b)>=0 || pro(a[1],a[n],b)<=0) {puts("NO");return 0;}
            long long l = 2, r = n;
            while (l<r){
                  long long mid = (l+r) / 2;
                  if (pro(a[1],a[mid],b)>=0)
                        r = mid;
                  else l = mid + 1;
            }
            if (pro(a[l],a[l-1],b)<=0) {puts("NO");return 0;}
      }
      puts("YES");
      return 0;
}
