#include<cstdio>
using namespace std;
struct point
{ 
    int x,y;
}a[100000];
int m,n;
bool Cc(point p1,point p2,point q1,point q2)
{
    const long long x1=p2.x-p1.x,y1=p2.y-p1.y,
                    x2=q2.x-q1.x,y2=q2.y-q1.y;
    if (x1*y2-y1*x2<0) return false;
     else return true;
}

int main()
{
        scanf("%d",&n);
    for (int i=0;i<n;++i)
      scanf("%d%d",&a[i].x,&a[i].y);
    scanf("%d",&m);
    for (int i=0;i<m;++i)
    {
        point b;
        scanf("%d%d",&b.x,&b.y);
        int lef=1,rig=n-1;
        if (Cc(a[0],a[lef],a[0],b)==Cc(a[0],a[rig],a[0],b)
        ||Cc(a[0],b,a[0],a[lef])==Cc(a[0],b,a[0],a[rig]))
        {
           printf("NO");
           return 0;
        } 
        while (rig-lef>1)
        {
             int mid=(lef+rig)>>1;
             if  (!Cc(a[0],b,a[0],a[mid])) 
               rig=mid;
             else lef=mid;
        }
        if (Cc(a[lef],a[rig],a[lef],b))
        {
           printf("NO");
           return 0;
        }
     }
    printf("YES");
  }
