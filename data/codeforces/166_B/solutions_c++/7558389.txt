#include<cstdio>
const int M=100010;
struct point{
    double x,y;
    friend point operator -(point a,point b){
        point res;
        res.x=a.x-b.x;
        res.y=a.y-b.y;
        return res;
    }
    friend double operator *(point a,point b){
        return a.x*b.y-a.y*b.x;
    }
}a[M],p;
int main(){
    int n,m;
    while(~scanf("%d",&n)){
        for(int i=1;i<=n;i++){
            scanf("%lf%lf",&a[i].x,&a[i].y);
        }
        scanf("%d",&m);
        bool flag=true;
        while(m--){
            scanf("%lf%lf",&p.x,&p.y);
            int l=2;
            int r=n;
            while(l<r){
                int mid=(l+r)/2;
                if((a[mid]-a[1])*(p-a[1])>=0){
                    r=mid;
                }else {
                    l=mid+1;
                }
            }
            if(l<3||(a[n]-a[1])*(p-a[1])<=0||(a[l-1]-a[l])*(p-a[l])<=0) {
                flag=false;
            }
        }
        if(flag) puts("YES");
        else puts("NO");
    }
    return 0;
}
