#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct rec{int x,y,z;}a[200010];
int s[200010],v[200010],n,m,i,p,ans;

bool cmp(rec a,rec b)
{
	return a.x<b.x||a.x==b.x&&a.y<b.y;
}

long long cross(rec a,rec b,rec c)
{
	return (long long)(b.x-a.x)*(c.y-a.y)-(long long)(b.y-a.y)*(c.x-a.x);
}

void graham()
{
	for(i=1;i<=n;i++)
	{
		while(p>1&&cross(a[s[p-1]],a[s[p]],a[i])<0) p--;
		s[++p]=i;
	}
	for(i=2;i<=p;i++) v[s[i]]=1;
	for(i=n;i;i--)
		if(!v[i])
		{
			while(cross(a[s[p-1]],a[s[p]],a[i])<0) p--;
			s[++p]=i;
		}
}

int main()
{
	cin>>n;
	for(i=1;i<=n;i++) scanf("%d%d",&a[i].x,&a[i].y),a[i].z=1;
	cin>>m;
	for(i=1;i<=m;i++) scanf("%d%d",&a[n+i].x,&a[n+i].y);
	n+=m;
	sort(a+1,a+n+1,cmp);
	for(i=1;i<n;i++)
		if(a[i].x==a[i+1].x&&a[i].y==a[i+1].y) {puts("NO"); return 0;}
	graham();
	for(i=1;i<p;i++)
		if(!a[s[i]].z) {ans=1; break;}
	if(ans) puts("NO"); else puts("YES");
	return 0;
}
