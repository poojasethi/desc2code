#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
const int maxn=130000;
struct arr{
  ll x,y;
  bool z;
}a[maxn];
int b[maxn];
int n,m;

bool cmp(arr a,arr b){
  return (a.y<b.y) || (a.y==b.y && a.x<b.x);
}

bool cj(arr a,arr b,arr c){
  return (a.x-c.x)*(b.y-c.y)<(a.y-c.y)*(b.x-c.x);
}

void graham(){
  int i,k;
  sort(a,a+n,cmp);
  b[0]=0;
  b[1]=1;
  k=1;
  for (i=2;i<n;i++){
    while(k && cj(a[b[k]],a[i],a[b[k-1]])) k--;
    b[++k]=i;
  }
  m=k;
  for (i=n-1;i>=0;i--){
    while(k>m && cj(a[b[k]],a[i],a[b[k-1]])) k--;
    b[++k]=i;
  }
  m=k;
}

int main(){
  int i;
  scanf("%d",&n);
  for (i=0;i<n;i++) scanf("%I64d%I64d",&a[i].x,&a[i].y);
  scanf("%d",&m);
  for (i=n;i<m+n;i++){
    scanf("%I64d%I64d",&a[i].x,&a[i].y);
    a[i].z=1;
  }
  n+=m;
  graham();
  for (i=0;i<m;i++)
    if (a[b[i]].z) break;
  if (i==m) printf("YES\n");
  else printf("NO\n");
  return 0;
}
