#include <iostream>

using namespace std;

struct point {
    long long x, y;
};

long long side(const point &a, const point &b, const point &c)
{
    return (b.x - a.x) * (c.y - b.y) - (c.x - b.x) * (b.y - a.y);
}

int main(int argc, char *argv[])
{
    int n, m;
    static point a[100010];
    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i].x >> a[i].y;
    cin >> m;
    for (int i = 0; i < m; i++) {
        point b;
        cin >> b.x >> b.y;
        int lo = 1, hi = n;
        while (lo < hi) {
            int mi = (lo + hi) / 2;
            long long r = side(a[0], a[mi], b);
            if (r < 0) lo =  mi + 1;
            else hi = mi;
        }
        if (lo == 1 || side(a[0], a[n - 1], b) <= 0 || side(a[lo - 1], a[lo], b) >= 0) {
            cout << "NO" << endl;
            return 0;
        }
    }
    cout << "YES" << endl;
    
    return 0;
}
