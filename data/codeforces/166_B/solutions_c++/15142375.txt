#include <stdio.h>
#include <algorithm>
#include <vector>
#include<iostream>
using namespace std;
typedef long long ll;
ll a[100000+100],b[100000+100];
int main(){
    int n,m;
    scanf("%d",&n);
    for (int i=1;i<=n;i++) cin>>a[i]>>b[i];
    scanf("%d",&m);
    ll x,y;
    for (int i=0;i<m;i++){
       cin>>x>>y;
       if ((a[2]-a[1])*(y-b[1])-(b[2]-b[1])*(x-a[1])>=0ll){ printf("%s","NO"); return 0;}
        if ((x-a[1])*(b[n]-b[1])-(y-b[1])*(a[n]-a[1])>=0ll){ printf("%s","NO"); return 0;}
        int l,r,c;
        l=2; r=n-1;
        while (l<r){
            c=(l+r+1)/2;
            if (((a[c]-a[1])*(y-b[1])-(b[c]-b[1])*(x-a[1]))<=0ll) l=c; else r=c-1;
        }
        if ((a[l+1]-a[l])*(y-b[l])-(b[l+1]-b[l])*(x-a[l])>=0ll){ printf("%s","NO"); return 0;}
    }
    printf("%s","YES"); return 0;
}
