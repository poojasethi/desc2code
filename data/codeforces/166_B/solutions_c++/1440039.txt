#include <iostream>
#include <stdio.h>
#include <math.h>
#include <stdlib.h>
using namespace std;
struct TPoint
{
    long long x, y;
}a[111111],b[22222];
int n,m;
long long cross(const TPoint &a, const TPoint &b, const TPoint &c)
{
    return (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
}

long long sc(const TPoint &a,const TPoint &b,const TPoint &c)
{
    return cross(a,b,c);
}
bool inside_convex(TPoint a[],int n,TPoint p)
{
    if(sc(a[0],a[n-1],p)<=0 || sc(a[0],a[1],p)>=0) return false;
    int l=0,r=n-1;
    int m;
    while(1)
    {
        m=(l+r)>>1;
        if(sc(a[0],a[m],p)>0) r=m;
        else l=m;
        if(r==l+1) break;
    }
    if(sc(a[l],a[l+1],p)<0) return true;
    else return false;
}
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%I64d %I64d",&a[i].x,&a[i].y);
    scanf("%d",&m);
    for(int i=0;i<m;i++) scanf("%I64d %I64d",&b[i].x,&b[i].y);
    int yes=1;
    for(int i=0;i<m;i++) yes&=inside_convex(a,n,b[i]);
    printf("%s\n",yes?"YES":"NO");
}
