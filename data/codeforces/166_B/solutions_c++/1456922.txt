#include<iostream>
#include<vector>
using namespace std;

struct point
{
  long long x, y;
};

long long cross (point p0, point p1, point p2)
{
  return (p1.x - p0.x)*(p2.y - p0.y) - (p2.x - p0.x)*(p1.y - p0.y);
}
point poly[100001];
int main ()
{
  int n, m;
  cin>>n;
  for (int i = 0; i < n; i++)
    cin>>poly[i].x>>poly[i].y;
  cin>>m;
  int inside = 0;
  for (int i = 0; i < m; i++) {
    point p;
    int lo, mi, hi;
    int last = 1;
    cin>>p.x>>p.y;
    lo = 1, hi = n - 1;
    while (lo <= hi) {
      mi = (lo + hi)>>1;
      if (cross(poly[0], p, poly[mi]) > 0)
	lo = mi + 1;
      else
	last = mi, hi = mi - 1;
    }
    if (last == 1 || cross (poly[0], p, poly[n - 1]) == 0) continue;
    if (cross (poly[last - 1], poly[last], p) < 0) inside++;
 
  }
  if (inside == m)
    cout<<"YES"<<endl;
  else
    cout<<"NO"<<endl;
 
  return 0;
}
