#include <cstdio> 
#include <algorithm>
using namespace std;
int main()
{
 int n,x;
 scanf("%d%d",&n,&x);
 int a[n+1],b[n+1];
 for (int i=1;i<=n;i++)
  scanf("%d",&a[i]);
 for (int i=1;i<=n;i++)
  scanf("%d",&b[i]);
 sort(a+1,a+n+1);
 sort (b+1,b+n+1);
 int ans=0,l=1;
 for (int i=n;i>=1;i--)
  {
    while(l<=n && b[l]+a[i]<x)
        ++l;
    if(l>n)
        break;
    ++ans;
    ++l;
   }
 printf("1 %d",ans);
}
