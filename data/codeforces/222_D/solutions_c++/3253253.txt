#include <iostream>
#include <algorithm>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,a[100010],b[100010],x,res,l,r;

int main(){
  cin>>n>>x;
  rep(i,n)cin>>a[i];
  rep(i,n)cin>>b[i];
  sort(a,a+n);
  sort(b,b+n);
  r = n-1;
  rep(i,n){
    if(a[i]+b[r]>=x){
      r--;res++;
    }   
  }
  cout<<1<<" "<<res<<endl;

}
