#include <iostream>
#include <algorithm>

using namespace std;

int n, x;
int a[100000], b[100000];

int main() 
{
	cin >> n >> x;
	for(int i = 0; i < n; i++)
		cin >> a[i];
	for(int i = 0; i < n; i++)
		cin >> b[i];
	sort(a, a + n);
	sort(b, b + n);
	int cnt = 0;
	for(int i = 0, j = n - 1; i < n; i++)
		if(a[i] + b[j] >= x)
			cnt++, j--;
	cout << "1 " << cnt << endl;
	return 0;
}
