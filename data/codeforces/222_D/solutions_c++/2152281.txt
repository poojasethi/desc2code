# include <stdio.h>
# include <algorithm>
# define MX (int)1e5

using namespace std;

int n,k,z;

int a[MX];
int b[MX];

int main()
{
	scanf("%d %d",&n,&k);
	
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	
	for(int i=0;i<n;i++)
	scanf("%d",&b[i]);
	
	sort(a,a+n);
	sort(b,b+n);
	
	z=n-1;
	
	for(int i=0;i<n;i++)
	if(b[i]+a[z]>=k)
	z--;
	
	printf("1 %d",n-z-1);
	
	getchar();
	getchar();
}
