# include <stdio.h>
# include <algorithm>

using namespace std;

int n,m;

int n1[100050];
int n2[100050];
int ans;

int main()
{
	scanf("%d %d",&n,&m);
	
	for (int i=1 ; i<=n ; i++) scanf("%d",&n1[i]);
	for (int i=1 ; i<=n ; i++) scanf("%d",&n2[i]);
	
	sort(n1,n1+n+1);
	sort(n2,n2+n+1);
	
	int u=n,y=1;
	
	while(u>0 && y<=n)
	{
		if(n1[u] + n2[y] >= m) ans++,u--,y++;
		else y++;
	}
	
	printf("1 %d\n",ans);
	
	getchar();getchar();
}
