#include<bits/stdc++.h>
using namespace std;
vector<int>v;
vector<int>t;
int main(){
    int n,x,p,cnt=0,i,j;
    scanf("%d %d",&n,&x);

    for(int i=0;i<n;i++) {scanf("%d",&p); v.push_back(p);}
    for(int i=0;i<n;i++) {scanf("%d",&p); t.push_back(p);}

    sort(v.begin(),v.end());
    sort(t.begin(),t.end(),greater<int>());

    for(i=0,j=0;i<n;i++){
        if(v[i]+t[j]>=x) j++;
    }
    printf("1 %d",j);
}
