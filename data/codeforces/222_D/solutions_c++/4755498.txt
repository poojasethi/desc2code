#include <iostream>
#include <algorithm>

using namespace std;

int A[100010], B[100010], n, x, cnt;

int main ()
{
	cin >> n >> x;
	for (int I = 0; I < n; ++I)	cin >> A[I];
	for (int I = 0; I < n; ++I)	cin >> B[I];
	sort(A, A+n), sort(B, B+n);
	cnt = 0;
	for (int I = 0, J = n-1; I < n && J >= 0; ++I)
		if (A[I] + B[J] >= x)	++cnt, --J;
	cout << "1 " << cnt << endl;
	return 0;
}
