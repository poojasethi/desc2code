#include<iostream>
#include<algorithm>
using namespace   std;
int n1[100050],n2[100050];
int main(){
	int n,x;
	while(cin>>n>>x){
		int ans=0;
		for(int i=0;i<n;++i)cin>>n1[i];
		for(int i=0;i<n;++i)cin>>n2[i];
		sort(n1,n1+n);sort(n2,n2+n);
		for(int i=0,k=n-1;i<n;++i){
			if(n1[i]+n2[k]>=x){
				ans++;k--;
			}
		}
		cout<<1<<" "<<ans<<endl;
	}
	return 0;
}
