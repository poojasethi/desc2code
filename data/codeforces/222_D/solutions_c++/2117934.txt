#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
int a[100005],b[100005];
int i,j,k,n,x,s=0;
int main()
{
	scanf("%d%d",&n,&x);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(i=1;i<=n;i++)
		scanf("%d",&b[i]);
	std::sort(a+1,a+n+1);
	std::sort(b+1,b+n+1);
	i=1;j=n;
	while(i<=n&&j>=1)
	{
		if(a[i]+b[j]>=x){s++;i++;j--;}
		else i++;
	}
	printf("1 %d\n",s);
	return 0;
}
