#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int main(){
	int n,x,a[100010],b[100010];
	scanf("%d%d",&n,&x);
	for(int i=0;i<n;i++){scanf("%d",&a[i]);}
	for(int i=0;i<n;i++){scanf("%d",&b[i]);}
	sort(a,a+n);
	sort(b,b+n);
	int st=n-1,ed=0;
	while(ed<n){
		if(a[st]+b[ed]>=x){st--;}
		ed++;
	}
	printf("1 %d\n",n-st-1);
	return 0;
}
