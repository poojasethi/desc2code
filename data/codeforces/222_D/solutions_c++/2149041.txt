#include <cstdio>
#include <algorithm>

using namespace std;

int d1[100010],d2[100010];

int a,b,c,n,x;

int main()
{
    scanf("%d%d",&n,&x);
    for (a=1; a<=n; a++) scanf("%d",&d1[a]);
    for (a=1; a<=n; a++) scanf("%d",&d2[a]);

    sort(d1+1,d1+1+n);
    sort(d2+1,d2+1+n);

    a=n;
    b=1;

    while (b<=n)
        if (d1[a]+d2[b]>=x)  a--, b++; else b++;

    printf("1 %d",n-a);
}
