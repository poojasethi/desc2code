#include <iostream>
#include <algorithm>
using namespace std;
int n,x,i,j,a[111111],b[111111],k=0;
int main()
{
	cin>>n>>x;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
		cin>>b[i];
	sort(a,a+n);
	sort(b,b+n);
	cout<<"1 ";
	i=0; j=n-1;
	while (i<n&&j>=0)
	{
		if (a[i]+b[j]>=x)
		{
			k++;
			i++;
			j--;
		}
		else i++;
	}
	cout<<k<<endl;
	return 0;
}
