#include<iostream>
#include<algorithm>
using namespace std;
int a[100001],b[100001],n,x,sum[100001],rnk;
int main()
{
    cin>>n>>x;
    for(int i=0;i<n;i++)cin>>a[i];
    for(int j=0;j<n;j++)cin>>b[j];
    sort(a,a+n);sort(b,b+n);
    for(int i=0,j=n-1;i<n&&j>=0;i++)
        if(a[i]+b[j]>=x)rnk++,j--;
    cout<<"1 "<<rnk << endl;
}
