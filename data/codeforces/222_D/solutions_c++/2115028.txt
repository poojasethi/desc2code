#include <cstdio>
#include <algorithm>
using namespace std;

int n,x,now,ans,a[100100],b[100100];

int main() {
	scanf("%d%d",&n,&x);
	for (int i=1;i<=n;++i) scanf("%d",&a[i]);
	for (int i=1;i<=n;++i) scanf("%d",&b[i]);
	sort(a+1,a+n+1);
	sort(b+1,b+n+1);
	now=1;
	for (int i=n;i>=1;--i) {
		while ((now<=n)&&(a[i]+b[now]<x)) ++now;
		if (now>n) break;
		++now;
		++ans;
	}
	printf("1 %d\n",ans);
	return 0;
}
