#include <iostream>
#include <algorithm>
using namespace std;
int a[100009],b[100009];
main(){
    int n,cnt,j,i,x;
    cin>>n>>x;
    for (i=1;i<=n;i++) cin>>a[i];
    for (i=1;i<=n;i++) cin>>b[i];
    sort(a+1,a+n+1);
    sort(b+1,b+n+1);
    for (i=1,j=n,cnt=0;i<=n&&j>=1;i++)
    if (a[i]+b[j]>=x) {j--;cnt++;}
    cout<<"1 "<<cnt;
}
