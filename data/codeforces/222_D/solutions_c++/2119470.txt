#include <algorithm>
#include <cstdio>
using namespace std;
int main()
{
	int n,x,a[100000],b[100000],ans=0;
	scanf("%d %d",&n,&x);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n);
	for(int i=0;i<n;i++)
		scanf("%d",&b[i]);
	sort(b,b+n);
	for(int j=n-1,i=0;i<n&&j>=0;i++)
		if(a[i]+b[j]>=x)
		{
			ans++;
			j--;
		}
	printf("1 %d\n",ans);
	return 0;
}
