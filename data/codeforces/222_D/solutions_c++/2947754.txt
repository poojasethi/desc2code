#include<iostream>
#include<fstream>
#include<algorithm>
#include<vector>

#include<string>
using namespace std;
int main(){
	
	long long i,n,x,t=0,j;
	int r1[100001],r2[100001];
	cin>>n>>x;
	for(i=0;i<n;i++)
		cin>>r1[i];
	for(i=0;i<n;i++)
		cin>>r2[i];
	sort(r1,r1+n);
    sort(r2,r2+n);
	
	for(i=0,j=n-1;i<n&&j>=0;i++)
		if(r1[i]+r2[j]>=x){
			t++;
			j--;
		}
	cout<<1<<" "<<t<<endl;

	return 0;
}
