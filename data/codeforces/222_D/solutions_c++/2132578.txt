#include<cstdio>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n,x; scanf("%d%d",&n,&x);
	int a[100000],b[100000];
	rep(i,n) scanf("%d",a+i);
	rep(i,n) scanf("%d",b+i);

	sort(a,a+n);
	sort(b,b+n);

	int ans=0,p=0;
	for(int i=n-1;i>=0;i--){
		for(;p<n;p++) if(a[i]+b[p]>=x) { ans++; p++; break; }
		if(p>=n) break;
	}

	printf("1 %d\n",ans);

	return 0;
}
