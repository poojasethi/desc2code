#include<cstdio>
#include<cstring>
#include<algorithm>
#include<functional>
using namespace std;
int a[100005],b[100005];
int main()
{
    int n,k,t=1;
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;++i) scanf("%d",a+i);
    for(int i=1;i<=n;++i) scanf("%d",b+i);
    sort(a+1,a+1+n);
    sort(b+1,b+1+n,greater<int>());
    for(int i=1;i<=n;++i)
        if(a[i]+b[t]>=k) ++t;
    printf("%d %d\n",1,t-1);
    return 0;
}
