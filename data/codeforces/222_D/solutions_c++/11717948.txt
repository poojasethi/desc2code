#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int a[100010],b[100010];

int main()
{
	ios_base::sync_with_stdio(0);
	int n,x,k,c=0;
	cin>>n>>x;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
		cin>>b[i];
	sort(a,a+n);
	sort(b,b+n);
	k=0;
	for(int i=n-1;i>=0;i--)
	{
		if(a[i]+b[k]>=x)
			c++;
		else
			i++;
		k++;
		if(k==n)
			break;
	}
	cout<<"1 "<<c<<"\n";
	return 0;
}
