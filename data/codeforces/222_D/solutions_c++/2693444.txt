#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	static int a[100005],b[100005];
	int n,s;
	cin >> n >> s;
	for(int i = 0; i < n; i++)
		cin >> a[i];
	for(int i = 0; i < n; i++)
		cin >> b[i];
	sort(a,a+n);
	sort(b,b+n);
	int r = 0, ans = 0;
	for(int i= n -1; i >=0; i--) {
		while(r < n && b[r] + a[i] < s) r ++;
		if(r < n) {
			ans ++;
			r ++;
		}
	}
	cout << 1 << " "<< ans << endl;
}
