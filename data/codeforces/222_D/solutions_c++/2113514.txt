#include <cstdio>
#include <algorithm>
using namespace std;
int a[100000],b[100000];
int main()
{
    int n,x;
    scanf("%d%d",&n,&x);
    for (int i=0;i<n;i++)
        scanf("%d",&a[i]);
    for (int j=0;j<n;j++)
        scanf("%d",&b[j]);
    sort(a,a+n);
    sort(b,b+n);
    int ans=0;
    for (int i=0,j=n-1;i<n;i++)
        if (a[i]+b[j]>=x)
            ans++,j--;
    printf("1 %d\n",ans);
    return 0;
}
