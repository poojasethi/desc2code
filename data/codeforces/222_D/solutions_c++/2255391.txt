#include <iostream>
#include <algorithm>
using namespace std;

long int a[100000],b[100000];
int main()
{
    long int n,x,i,k,na,nb;
    cin>>n>>x;
    for(i=0;i<n;i++) cin>>a[i];
    for(i=0;i<n;i++) cin>>b[i];
    sort(a,a+n);sort(b,b+n);
    na=n-1;nb=0;k=0;
    while(na>=0 && nb<n)
    {
        if(a[na]+b[nb]<x) nb++;
        else {na--;nb++;k++;}
    }
    cout<<"1 "<<k;
    return 0;
}
