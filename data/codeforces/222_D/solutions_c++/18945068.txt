#include <iostream>
#include <algorithm>
using namespace std;
bool cmp(int a,int b){
	return a>b;
}
int main(){
	int a[100005],b[100005];
	int n,x,temp,ans=0;cin>>n>>x;
	for(int i=0;i<n;i++) cin>>a[i];
	for(int i=0;i<n;i++) cin>>b[i];
	sort(a,a+n);
	sort(b,b+n,cmp);
	for(int i=0,j=0;i<n;i++){
		if(a[i]+b[j]>=x){
			ans++;
			j++;
		}
	}
	//if(!ans) ans=1;
	cout<<"1 "<<ans<<endl;
}
