#include<iostream>
#include<algorithm>
using namespace std;

#define maxn 100009
int main()
{

int n,x,a[maxn],b[maxn];

cin>>n>>x;

for(int i=0;i<n;i++)
  cin>>a[i];

for(int i=0;i<n;i++)
  cin>>b[i];

sort( a, a+n);
sort( b, b+n);

int w=0;

int i=0,j=n-1;

for( ; i<n && j >= 0 ; i++)
   if( a[i] + b[j] >=x )
    { w++;
     j--;
    }

cout<<"1 "<<w<<endl;
return 0;
}
