#include<iostream>
#include<algorithm>
#define N 100010
using namespace std;
int a[N], b[N], n, x, mx = -1;
int main(){

	cin >> n >> x;
	for(int i = 0; i < n; i++) cin >> a[i]; sort(a, a+n);
	for(int i = 0; i < n; i++) cin >> b[i]; sort(b, b+n);

	for(int i = 0, j = n-1; i < n && j >= 0; i++) if(a[i]+b[j] >= x) j--, mx++;
	cout << "1 " << mx+1 << endl;

}
