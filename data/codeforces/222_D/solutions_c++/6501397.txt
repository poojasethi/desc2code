#include <cstdio>
#include <algorithm>
int a[100001],b[100001];
int main(){
    int n,x,d,i,ans=0;
    scanf("%d%d",&n,&x);
    for(i=0;i<n;i++) scanf("%d",&a[i]);
    for(i=0;i<n;i++) scanf("%d",&b[i]);
    std::sort(a,a+n);
    std::sort(b,b+n);
    d=n-1;
   for(i=0;i<n;i++)
        if(a[i]+b[d]>=x)
             ans++,d--;
   printf("1 %d",ans);
   return 0;
}
