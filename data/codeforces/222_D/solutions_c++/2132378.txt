#include <cstdio>
#include <algorithm>
using namespace std;
int n,m,a[100100],b[100100],l=0;
int main(void){
  scanf("%d %d",&n,&m);
  for(int i=0;i<n;++i)scanf("%d",&a[i]);sort(a,a+n);
  for(int i=0;i<n;++i)scanf("%d",&b[i]);sort(b,b+n);
  for(int i=0,j=n;j>0&&i<n;++i)if(a[i]+b[j-1]>=m)--j,++l;
  printf("1 %d\n",l);
  return 0;
}
