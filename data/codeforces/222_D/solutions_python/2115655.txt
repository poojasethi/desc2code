import sys
from collections import deque
def main():
	n, x = map(int, sys.stdin.readline().split())
	a = map(int, sys.stdin.readline().split())
	b = map(int, sys.stdin.readline().split())
	a.sort()
	b.sort()
	p = deque(a)
	q = deque(b)
	ans = 0
	while len(p) and len(q):
		if p[0] + q[-1] < x:
			ans += 1
			p.popleft()
			q.popleft()
		else:
			p.popleft()
			q.pop()
	print 1, n - ans
		
main()
