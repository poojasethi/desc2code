import bisect

def start():
    n, y = map(int, raw_input().split())
    a = map(int, raw_input().split())
    b = map(int, raw_input().split())
    a.sort()
    b.sort()
    position = 0
    start_a = n-1
    end_b = bisect.bisect_left(b, y-a[start_a])
    while end_b < n:
        if a[start_a]+b[end_b] < y:
            end_b += 1
        else:
            position += 1
            end_b += 1
            start_a -= 1
    print 1, position 

start()
