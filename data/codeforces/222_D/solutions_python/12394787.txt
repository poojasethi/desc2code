n, x = [int(i) for i in raw_input().split()]
a = sorted([int(i) for i in raw_input().split()])[::-1]
b = sorted([int(i) for i in raw_input().split()])

p = cnt = 0 
for i in a: 
	while p + 1 < len(b) and i + b[p] < x:
		p += 1   
	if p < len(b) and i + b[p] >= x:
		cnt += 1 
		p += 1
	else:
		break

print 1, cnt
