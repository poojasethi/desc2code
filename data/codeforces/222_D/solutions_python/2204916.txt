import sys
if sys.version_info[0] == 3: raw_input = input
n, x = map(int, raw_input().split())
a, b, c = list(map(int, raw_input().split())), list(map(int, raw_input().split())), [0] * n
a.sort(); b.sort(reverse = True)
ans, j = 0, 0
for i in range(n):
    if a[i] + b[j] >= x:
        ans += 1
        j += 1
        if j == n: break
print('1 ' + str(ans))
