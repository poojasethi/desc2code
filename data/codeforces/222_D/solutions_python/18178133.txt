def main():
	n, x = map(int, raw_input().split())
	a = map(int, raw_input().split())
	b = map(int, raw_input().split())
	a.sort()
	b.sort()
	cnt = 0
	i, j = 0, n - 1
	while i < n and j >= 0:
		if a[i] + b[j] >= x: 
			cnt += 1
			i += 1
			j -= 1
		else: i += 1
	print 1, cnt		

if __name__ == "__main__": main()
