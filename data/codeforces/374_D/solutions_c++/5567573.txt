

#include<bits/stdc++.h>
using namespace std;
const int Maxn=1e6+6;

int n,m,a[Maxn],sub[Maxn],bear[Maxn],ans[Maxn],id[Maxn],curs,have[Maxn];

int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++)
      scanf("%d",&a[i]);
    for(int i=1;i<=m;i++)
      sub[a[i]]=1;
    for(int i=1;i<=n;i++)
      sub[i]+=sub[i-1];

    for(int i=1;i<=n;i++)
      if(sub[i]!=sub[i-1])
        bear[i]=1;
      else
      {
         if(i<a[1])
           bear[i]=-1;
         else
           bear[i]=bear[i-sub[i]]+1;
      }

    for(int i=1;i<=n;i++)
    {
        scanf("%d",&ans[i]);
        if(ans[i]==-1)
            curs-=sub[curs];
        else
          id[i]=++curs;
        have[i]=have[i-1]+(ans[i]==-1);
    }

    bool f=false;
    for(int i=1;i<=n;i++)
      if(ans[i]!=-1)
        if(bear[id[i]]==-1||bear[id[i]]>have[n]-have[i])
        {
            printf("%d",ans[i]);
            f=true;
        }
    if(!f)
      puts("Poor stack!");
    return 0;
}
