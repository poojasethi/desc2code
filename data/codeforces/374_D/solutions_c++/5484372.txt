#include <cstdio>

#define N 1000000

struct nd
{
	int a, b;
	nd *x;
} *h = 0, *e = 0, *c = 0, *q = 0;

int a[N], i, m, t, k, n, p, s;
bool b[N];

int main()
{
	scanf("%d%d", &t, &m);

	for (i = 0; i < m; i ++)
		scanf("%d", &a[i]);

	h = e = new nd, h->a = h->b = 0, h->x = 0;

	while (t --)
	{
		scanf("%d", &k);
		if (~k)
			b[n ++] = k, ++ e->a, ++p;
		else
		{
			for (c = h, i = 0; i < m && a[i] <= p; ++ i)
			{
				for (s = i ? a[i] - a[i - 1] : a[i]; s > c->a; s -= c->a, c = c->x);
				if (s == 1)
				{
					++ c->b, --c->a;
					if (!c->a && c->x)
					{
						c->b += c->x->b, c->a = c->x->a, q = c->x, c->x = c->x->x;
						if (q == e)
							e = c;
						delete q;
					}
				}
				else
				{
					if (!c->x)
						e = c->x = new nd, e->a = e->b = 0, e->x = 0;
					else
						q = new nd, q->x = c->x, c->x = q;
					c->x->a = c->a - s, c->x->b = 1, c->a = s - 1, c = c->x;
					if (!c->a && c->x)
					{
						c->b += c->x->b, c->a = c->x->a, q = c->x, c->x = c->x->x;
						if (q == e)
							e = c;
						delete q;
					}
				}
			}
			p -= i;
		}
	}

	if (!p)
		printf("Poor stack!\n");
	else
		for (c = h, i = 0; c; c = c->x)
			for (i += c->b; c->a--; i ++)
				printf("%d", b[i]);

	return 0;
}
