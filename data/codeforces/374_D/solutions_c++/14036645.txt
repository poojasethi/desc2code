#include <cstdio>

#define N 1000000

struct nd
{
	int a, b;
	nd *x;
} *h = NULL, *e = NULL, *c = NULL, *q = NULL;

int a[N], i, m, t, k, n, p, s;
bool b[N];

int main()
{
	scanf("%d%d", &t, &m);

	for (i = 0; i < m; i ++)
		scanf("%d", &a[i]);

	h = e = new nd;
	h->a = h->b = 0;
	h->x = NULL;

	while (t --)
	{
		scanf("%d", &k);
		if (~k)
			b[n ++] = k, ++ e->a, ++p;
		else
		{
			for (c = h, i = 0; i < m && a[i] <= p; ++ i)
			{
				for (s = i ? a[i] - a[i - 1] : a[i]; s > c->a; c = c->x)
					s -= c->a;
				if (s == 1)
					++ c->b, --c->a;
				else
				{
					if (c->x == NULL)
					{
						e = c->x = new nd;
						e->a = e->b = 0;
						e->x = NULL;
					}
					else
					{
						q = new nd;
						q->x = c->x;
						c->x = q;
					}
					c->x->a = c->a - s;
					c->x->b = 1;
					c->a = s - 1;
					c = c->x;
				}
			}
			p -= i;
		}
	}

	if (!p)
		printf("Poor stack!\n");
	else
		for (c = h, i = 0; c != NULL; c = c->x)
			for (i += c->b; c->a--; i ++)
				printf("%d", b[i]);

	return 0;
}
