#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1000010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,s;
int tr[N],a[N],b[N];
inline void inc(int x) {
	for (;x<=n;x+=x&-x) tr[x]++;
}
inline int cal(int x) {
	int an=0;
	for (;x;x-=x&-x) an+=tr[x];
	return an;
}
inline void work() {
	int i,l,r,mid;
	For(i,1,m) {
		l=1,r=s;
		for (;l<=r;) {
			mid=(l+r)/2;
			if (mid-cal(mid)>=a[i]-i+1) r=mid-1;
			else l=mid+1;
		}
		r++;
		if (r>s||!(r-cal(r)==a[i]-i+1)) return;
		inc(r);
	}
}
int main() {
	//freopen("374d.in","r",stdin);
	scanf("%d%d",&n,&m);
	For(i,1,m) scanf("%d",&a[i]);
	For(i,1,n) {
		scanf("%d",&k);
		if (k==-1) work();
		else b[++s]=k;
	}
	int F=0;
	For(i,1,s) if (cal(i)==cal(i-1)) printf("%d",b[i]),F=1;
	if (!F) printf("Poor stack!\n");
	//for(;;);
	return 0;
}
