#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstdlib>
#include <cmath>
#include <cstdlib>
#include <cstring>
using namespace std;
#define INF 1000000000
#define MAXN 1000005
int step[MAXN];
int isbad[MAXN];
int cntbad[MAXN];
int event[MAXN];
int cnthit[MAXN];
int n,m,x;
int main(){
    scanf("%d%d",&n,&m);
    for (int i=0;i<=n;i++)  step[i]=INF;
    for (int i=1;i<=m;i++){
        scanf("%d",&x);
        isbad[x]=step[x]=1;
    }
    for (int i=1;i<=n;i++){
        cntbad[i]=cntbad[i-1]+isbad[i];
        if (!isbad[i]) step[i]=step[i-cntbad[i]]+1;
    }
    for (int i=1;i<=n;i++){
        scanf("%d",&event[i]);
         if (event[i]<0) cnthit[i]=1;
    }
    for (int i=n;i;i--){
        cnthit[i]+=cnthit[i+1];
    }
    int len=0;
    for (int i=1;i<=n;i++){
        if (event[i]<0){
            len-=cntbad[len];
        }
        else{
            len++;
            if (cnthit[i]<step[len]) printf("%d",event[i]);
        }
    }
    if (len==0){
        printf("Poor stack!");
    }
    printf("\n");
}
