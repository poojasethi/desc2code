#include <iostream>

#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

int a[1000000], bit[1000001], curr=1, st[1000001], len;

int main(){
    int n,m,t;
    scanf("%d %d", &n, &m);
    for(int i=0;i<m;i++)
        scanf("%d", &a[i]);
    for(int i=0;i<n;i++){
        scanf("%d", &t);
        if(t!=-1){
            len++;
            st[curr]=t;
            for(int j=curr++;j<=n;j+=j&-j)
                bit[j]++;
        }
        else {
            for(int j=0;j<m&&a[j]<=len+j;j++){
                int curr2=0,temp=a[j]-j;
                for(int i=19;i>=0;i--)
                    if(curr2+(1<<i)<=curr&&temp>bit[curr2+(1<<i)]) curr2+=1<<i,temp-=bit[curr2];
                curr2++;
                st[curr2]=-1;
                for(int i=curr2;i<=n;i+=i&-i)
                    bit[i]--;
                len--;
            }
        }
    }
    if(len==0) printf("Poor stack!\n");
    for(int i=1;i<=curr-1;i++)
        if(st[i]!=-1) printf("%d", st[i]);
    return 0;
}
