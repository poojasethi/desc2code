#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cstdlib>
#include <cmath>
#include <string>
#define For(i,a,n)  for(int i=(a);i<=(n);i++)
#define Forn(i,a,n) for(int i=(a);i>=(n);i--)
using namespace std;

int sum[4000005],val[4000005];
int a[1000005],n,M,L,m,x,tot;

void setIO(string a){
	 string in=a+".in",out=a+".out";
	 freopen(in.c_str(),"r",stdin);
	 freopen(out.c_str(),"w",stdout);
}

void add(int x,int d){
	d+=M;
	val[d]=x;
	for(;d;d>>=1) sum[d]++;
}

void change(int Kth){
	int x=1;
	int l=1,r=M;
	while(l^r){
		int mid=(l+r)>>1;
		x<<=1;
		if(sum[x]<Kth) Kth-=sum[x],x++,l=mid+1;
		else r=mid;
	}
	for(;x;x>>=1) sum[x]--; 
}

int drp(){
	int d=upper_bound(a+1,a+n+1,L)-(a+1);
	For(i,1,d) change(a[i]-i+1);
	return d; 
}

int main(){
	scanf("%d%d",&m,&n);
	for(M=1;M<=m;M<<=1); 
	For(i,1,n) scanf("%d",&a[i]);
	For(i,1,m){
		scanf("%d",&x);
		if(x>=0) add(x,++tot),L++;
		else L-=drp(); 
	}if(!L)return puts("Poor stack!"),0;
	For(i,M+1,2*M) if(sum[i])putchar(val[i]+'0');
	return 0;
}
