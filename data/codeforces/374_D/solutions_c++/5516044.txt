#include "cstdio"
#include "cstring"
#include "algorithm"
#include "iostream"
#include "vector"
#include "bitset"
using namespace  std;
const int MAXN = 1000001;
const int INF = 0x7f7f7f7f;
int isbad[MAXN] , cntBad[MAXN] , event[MAXN] , cntHit[MAXN] , step[MAXN];
int main()
{
    int n , m , x ;
    cin >> n >> m ;
    for ( int i = 1 ; i < MAXN ; i ++ ) step[i] = INF;
    while ( m -- ) 
    {
        cin >> x ;
        isbad[x] = step[x] = 1 ;
    }
    for ( int i = 1 ; i < MAXN ; i ++ )
    {
        cntBad[i] = cntBad[i-1] + isbad[i];
        if ( !isbad[i] ) step[i] = step[ i - cntBad[i] ] + 1 ;
    }
    for ( int i = 1 ; i <= n ; i ++ )
    {
        cin >> event[i];
        if ( event[i] < 0 ) cntHit[i] = 1 ;
    }
    for ( int i = n ; i ; i -- )
    {
        cntHit[i] += i == n ? 0 : cntHit[i+1];
    }
    //1 2 1 3 2 1 4 3 2 5 
    //3 2 2 2 2 2 1 1 1 0 
    int len = 0 ;
    for ( int i = 1 ; i <= n ; i ++ )
    {
        if ( event[i] < 0 ) len -= cntBad[len];
        else 
        {
            if ( cntHit[i] < step[++len] ) cout << event[i];
        }
    }
    if ( !len ) cout << "Poor stack!" << endl ;
    return 0;
}
