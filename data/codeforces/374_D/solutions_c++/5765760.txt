#pragma comment(linker, "/STACK:268435456,268435456")
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

int a[1000000], bit[1000001], st[1000001], now=0, len=0;

int f1(int x){
    int tt=0;
    for(int i=log(now+0.0)/log(2.0);i>=0;i--)
        if(tt+(1<<i)<=now&&bit[tt+(1<<i)]<x)
            tt+=(1<<i),x-=bit[tt];
    st[tt+1]=-1;
    return tt+1;
}

int main(){
    int n,m,t;
    scanf("%d %d", &n, &m);
    for(int i=0;i<m;i++)
        scanf("%d", &a[i]);
    for(int i=0;i<n;i++){
        scanf("%d", &t);
        if(t!=-1){
            len++;
            st[++now]=t;
            for(int j=now;j<=n;j+=j&-j)
                bit[j]++;
        }
        else {
            for(int i=0;i<m&&a[i]<=len+i;i++){
                for(int j=f1(a[i]-i);j<=n;j+=j&-j)
                    bit[j]--;
                len--;
            }
        }
    }
    if(len==0) printf("Poor stack!\n"),exit(0);
    for(int i=1;i<=now;i++)
        if(st[i]!=-1) printf("%d",st[i]);
    putchar(10);
    //system("PAUSE");
    return 0;
}
