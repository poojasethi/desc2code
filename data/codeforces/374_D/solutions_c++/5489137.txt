#include <bits/stdc++.h>
using namespace std;
#define INF 10000000
int n, m;
int a[1000001], pre[1000001], dp[1000001], req[1000001];
int main(int argc, char** argv)
{
      cin >> n >> m;
      for (int i = 0; i < m; ++i)
      {
	    cin >> a[i];
	    pre[a[i]] = 1;
      }
      for (int i = 1; i <= 1000000; ++i)
	    pre[i] += pre[i-1];
      
     int loc = 0;
     a[m] = INF;
     for (int i = 1; i <= n; ++i)
      {
	    if (a[loc] == i)
	    {
		  dp[i] = 1;
		  loc ++;
	    }
	    else if (loc == 0 || dp[i-loc] == INF)
		  dp[i] = INF;
	    else
		  dp[i] = dp[i-loc] + 1;
      }

      int type, already = 0;
      string res = "";
      loc = 0;
      for (int i = 0; i < n; ++i)
      {
	    cin >> type;
	    if (type != -1)
	    {
		  res += (char)('0' + type);
		  loc ++;
		  req[(int)res.size() - 1] = already + dp[loc];
	    }
	    else
	    {
		  loc -= pre[loc];
		  already ++;
	    }
      }

      string fin = "";
      for (int i = 0; i < (int)(res.size()); ++i)
	    if (already < req[i])
		  fin += res.at(i);

      if (fin == "")
	    cout << "Poor stack!" << endl;
      else
	    cout << fin << endl;
      return 0;
}
