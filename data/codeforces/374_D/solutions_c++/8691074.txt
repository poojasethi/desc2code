#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
#define maxn 1000005
using namespace std;

int f[maxn+5]={0},k[maxn+5]={0};
int a[maxn+5];
int b[maxn+5],c[maxn+5];

int main()
{
    int n,m,i,x;
    scanf("%d%d",&n,&m);
    for (i=1;i<=m;i++)
     {
        scanf("%d",&x);
        f[x]=1;
        k[x]=1;
     }
    for (i=1;i<=maxn;i++)
     	f[i]=f[i-1]+f[i];

    for (i=1;i<=maxn;i++)
       if (k[i]==0)
       {
       	 if (f[i]==0) k[i]=maxn;
         else k[i]=k[i-f[i]]+1;   
       }

    int now=0;
    for (i=1;i<=n;i++) 
    {
    	//cout<<now<<endl;
       scanf("%d",&a[i]);
       if (a[i]!=-1) b[i]=++now,c[i]=c[i-1];
       else
       {
          now-=f[now];
          c[i]=c[i-1]+1;
          b[i]=now;
       }	
    }      
    //cout<<now<<endl;
    if (now==0) cout<<"Poor stack!"<<endl;
    else 
    {
       for (i=1;i<=n;i++)
       	if (a[i]!=-1 && k[b[i]]>c[n]-c[i]) printf("%d",a[i]);
         cout<<endl;
    }


	return 0;
}
