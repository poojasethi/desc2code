#include<bits/stdc++.h>
using namespace std;

#define scl(x) scanf("%lld",&x)
#define sc(x)  scanf("%d",&x)
#define ll long long
#define lop(i,n) for(int i=0;i<n;++i)
typedef pair<int,int> ii;
typedef pair<ll,ll> pll;

const int mN=11e5+100;

int BIT[mN];
inline void add(int i,int v){
	++i; while(i<mN)BIT[i]+=v,i+=i&-i;
}
int n,m,arr[mN],id,sz,val[mN],ok[mN],t;
inline int getval(int v){
	++v;
	if(BIT[1<<int(log2(n)+1)]<v)return -1; // mafesh idx v
	int bst=0;
	for(int j=1<<19;j;j>>=1)
		if(BIT[bst+j]<v)v-=BIT[bst+j],bst+=j;
	return bst;
}
int main(){
#ifndef ONLINE_JUDGE
	freopen("i.txt","r",stdin);
#endif
	sc(n),sc(m);
	lop(i,m)sc(arr[i]),--arr[i];
	lop(i,n){
		sc(t);
		if(t==-1){
			if(arr[0]>=sz)continue;
			int j=0;
			while(j+1<m&&arr[j+1]<sz)++j;
			while(j!=-1){
				int v=getval(arr[j--]);
				add(v,-1);
				ok[v]=0;
				--sz;
			}
		}
		if(!t){
			add(id,1);
			ok[id]=1;
			val[id++]=0;
			++sz;
		}
		if(t==1){
			add(id,1);
			ok[id]=1;
			val[id++]=1;
			++sz;
		}
	}
	if(!sz)puts("Poor stack!");
	lop(i,mN)
		if(ok[i])
			printf("%d",val[i]);

}
