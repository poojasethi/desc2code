using namespace std;
int *a=new int[2097152];
void l(int q){
  int w=1,t;
  for(t=524288;t;t>>=1){
   w<<=1;
   if(q>a[w]){
    q-=a[w];
    w++;}}
  for(;w;w>>=1)
   a[w]-=1;
}

  int q,w,e,r,t,y,i=1,i1=0,c,v,*s=new int[1000000],*d=new int[1000000];
  cin>>q>>t;
  for(w=0;w<t;w++)
   scanf("%d",&d[w]);
  for(w=2097151;w>1048576;w--)
   a[w]=1;
  a[w]=0;
  for(w--;w;w--)
   a[w]+=a[w<<1]+a[(w<<1)+1];
  for(;q;q--){
   scanf("%d",&w);
   if(w<0){
    e=-1;
    for(c=524288;c;c>>=1){
     v=e+c;
     if((v<t)&&(d[v]<=i1))
      e=v;}
    i1-=e+1;
    for(;e>=0;e--)
     l(d[e]);}
   else{
    s[i]=w;
