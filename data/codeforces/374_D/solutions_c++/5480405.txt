void add(int n,int x)
{
 d[n]=x;t[n]=1;n/=2;
 while (n)
 {
  t[n]=t[2*n]+t[2*n+1];
  n/=2;
 }

}

void del(int i,int l,int r,int n)
{
// cout<<l<<'-'<<r<<" ";
 if (l==r) {t[i]=0;return;}
 int tmp=(l+r)/2;
 if (t[2*i]<n) del(2*i+1,tmp+1,r,n-t[2*i]); else del(2*i,l,tmp,n);
 t[i]=t[2*i]+t[2*i+1];
}
