#include <iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<cstdlib>
#include<algorithm>
using namespace std;
#define maxn 1000007
const int oo=27081993;
int n,m;
int isbad[maxn],step[maxn],event[maxn],cnt[maxn],cntbad[maxn];
int main()
{
   // freopen("sw.in","r",stdin);
    scanf("%d%d",&n,&m);
    int _max=0;
    for (int i = 1; i <= 1000000; i++) step[i] = oo;
    for(int i=1;i<=m;i++)
    {
        int ch;
        scanf("%d",&ch);
        isbad[ch]=step[ch]=1;
    }
    for(int i=1;i<=1000000;i++)
    {
        cntbad[i]=cntbad[i-1]+isbad[i];
        if(!isbad[i])step[i]=step[i-cntbad[i]]+1;
    }
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&event[i]);
        if(event[i]<0)cnt[i]=1;
    }
    for(int i=n;i;i--)cnt[i]+=cnt[i+1];
    int len=0;
    for(int i=1;i<=n;i++)
    {
        if (event[i] < 0) len -= cntbad[len];
        else
            if (cnt[i] < step[++len]) cout << event[i];
    }

    if(!len)printf("Poor stack!");
    printf("\n");
    return 0;


}
