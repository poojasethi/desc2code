#include <iostream>

using namespace std;

int n,m;
int a[2000000];

int metidos=0;
int cuantos[1<<22];
int valor[1<<22];
int primero=1<<21;

void propaga(int pos)
{
  while (pos>1) {
    pos/=2;
    cuantos[pos]=cuantos[2*pos]+cuantos[2*pos+1];
  }
}

void insertar(int val)
{
  int pos=primero+metidos;
  //cout<<"metemos "<<pos<<" "<<val<<endl;
  metidos++;
  cuantos[pos]=1;
  valor[pos]=val;
  propaga(pos);
}

void eliminar(int c)
{
  //int cini=c;
  int pos=1;
  while (pos<primero) {
    if (cuantos[2*pos]>=c) pos=2*pos;
    else {
      c-=cuantos[2*pos];
      pos=2*pos+1;
    }
  }
  //cout<<"eliminamos "<<cini<<" "<<pos<<endl;
  cuantos[pos]=0;
  propaga(pos);
}

int main()
{
  ios::sync_with_stdio(false);
  cin>>m>>n;
  for (int i=0;i<n;i++)
    cin>>a[i];
  for (int cas=0;cas<m;cas++) {
    int x;
    cin>>x;
    if (x>=0) insertar(x);
    else {
      int i=0;
      while (i<n and a[i]<=cuantos[1]) i++;
      i--;
      while (i>=0) {
	eliminar(a[i]);
	i--;
      }
    }
  }
  if (cuantos[1]==0) {
    cout<<"Poor stack!"<<endl;
  } else {
    for (int i=0;i<metidos;i++)
      if (cuantos[primero+i])
	cout<<valor[primero+i];
    cout<<endl;
  }
}
