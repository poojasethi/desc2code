#include <iostream>
#include <cstdio>

using namespace std;

#define SIZE 1000010

int n, m, l, t, s;
int a[SIZE];
int b[SIZE];
int c[SIZE];

void add(int x, int v)
{
    while (x < SIZE) {
        c[x] += v;
        x += x & -x;
    }
}

int find(int x)
{
    int ret = 0;
    for (int i = 19; i > -1; --i) {
        if (ret + (1 << i) <= l && c[ret + (1 << i)] < x) {
            ret += 1 << i;
            x -= c[ret];
        }
    }
    return ret + 1;
}

int main()
{
    scanf("%d %d", &n, &m);

    for (int i = 1; i <= m; scanf("%d", a + i++));

    for (int i = 1; i <= n; ++i) {
        scanf("%d", &t);
        if (t != -1) {
            b[++l] = t;
            add(l, 1);
            ++s;
        }
        else {
            for (int j = 1; j <= m && a[j] - j + 1 <= s; ++j) {
                t = find(a[j] - j + 1);
                add(t, -1);
                --s;
                b[t] = -1;
            }
        }
    }
    if (s == 0) {
        puts("Poor stack!");
        return 0;
    }
    for (int i =  1; i <= l; ++i) {
        if (b[i] != -1) printf("%d", b[i]);
    }
    puts("");
}
