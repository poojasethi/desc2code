#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<cstdlib>
#include<algorithm>
#include<queue>
using namespace std;

const int MAX_N = 1111111;
int n, m;
int de[MAX_N], des[MAX_N], b[MAX_N], h[MAX_N];

int main(){
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= m; ++i){
        int x;
        scanf("%d", &x);
        de[x] = 1;
    }
    for (int i = 1; i <= n; ++i) de[i] += de[i - 1];
    for (int i = 1; i <= n; ++i)
        if (de[i] - de[i - 1]) des[i] = 1;
        else if (de[i]) des[i] = des[i - de[i]] + 1;
    for (int i = 1; i <= n; ++i){
        scanf("%d", &b[i]);
        if (b[i] < 0) h[i] = 1;
    }
    int l = 0, pp = 0;
    for (int i = 1; i <= n; ++i) h[i] += h[i - 1];
    for (int i = 1; i <= n; ++i){
        if (b[i] < 0) l -= de[l];
        else{
            ++l;
            if (h[n] - h[i - 1] < des[l] || !des[l]) {printf("%d", b[i]);++pp;}
        }
    }
    if (!pp) puts("Poor stack!");
}
