#include<iostream>
using namespace std;
int i,co,cx,wo,wx;
char a[9];
int main(){
	for(int i=0;i<9;i++){
		cin>>a[i];
		if(a[i]=='X')cx++;
		if(a[i]=='0')co++;
	}
	for(int i=0;i<9;i++){
		if((a[i*3]=='0' && a[i*3+1]=='0' && a[i*3+2]=='0')||(a[i]=='0' && a[i+3]=='0' && a[i+6]=='0')||(a[0]=='0' && a[4]=='0' && a[8]=='0')||(a[2]=='0' && a[4]=='0' && a[6]=='0'))wo=1;
		if((a[i*3]=='X' && a[i*3+1]=='X' && a[i*3+2]=='X')||(a[i]=='X' && a[i+3]=='X' && a[i+6]=='X')||(a[0]=='X' && a[4]=='X' && a[8]=='X')||(a[2]=='X' && a[4]=='X' && a[6]=='X'))wx=1;
	}
	if((wo && co!=cx)||(wx && cx!=co+1)||(cx!=co+1 && cx!=co))cout<<"illegal";
	else if(wx)cout<<"the first player won";
	else if(wo)cout<<"the second player won";
	else if(cx+co==9)cout<<"draw";
	else if(cx>co)cout<<"second";
	else cout<<"first";
	return 0;
}
