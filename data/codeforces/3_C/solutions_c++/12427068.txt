#include<iostream>
using namespace std;
int res0,resX;
bool ult0,ultX;
char b[9];
bool f(int i,char c)
{
	return (b[3 * i] == c && b[3 * i + 1] == c && b[3 * i + 2] == c) || (b[i] == c && b[i + 3] == c && b[i + 6] == c) || (b[0] == c && b[4] == c && b[8] == c) || (b[2] == c && b[4] == c && b[6] == c);
}
int main()
{
	for(int i=0;i<9;i++)
	{
		cin>>b[i];
		if(b[i]=='0')
		res0++;
		if(b[i]=='X')
		resX++;
	}
	ult0=ultX=0;
	for (int i=0;i<3;i++)
	{
	    ult0=ult0 || f(i,'0');
	    ultX=ultX || f(i,'X');
	}
	if((ultX && resX!=res0+1)||(ult0 && resX!=res0)||(resX!=res0+1 && resX!=res0))
	cout<<"illegal";
	else if(ultX)
	cout<<"the first player won";
	else if(ult0)
	cout<<"the second player won";
	else if(res0+resX==9)
	cout<<"draw";
	else if(res0==resX)
	cout<<"first";
	else
	cout<<"second";
	return 0;
}
