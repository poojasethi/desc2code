#include <cstdio>
#include <algorithm>

using namespace std;

int oc1[100005];
int oc2[100005];
int stk[100005];
int snum;
int sol[100005];
int sze;
int mod;

int n;

int main()
{
    scanf("%d",&n);
    scanf("%d",&mod);
    for (int i=1; i<=n; i++)
    {
        int vl;
        scanf("%d",&vl);
        oc1[vl]++;
    }
    for (int i=1; i<=n; i++)
    {
        int vl;
        scanf("%d",&vl);
        oc2[vl]++;
    }
    for (int i=mod-1; i>=0; i--)
    {
        while (oc1[mod-i-1])
        {
            stk[++sze]=mod-i-1;
            oc1[mod-i-1]--;
        }
        while (sze>0 && oc2[i]>0)
        {
            sol[++snum]=(stk[sze--]+i)%mod;
            oc2[i]--;
        }
    }
    //reverse(stk+1,stk+1+sze);

    for (int i=mod-1; i>=0; i--)
    {
        while (sze>0 && oc2[i]>0)
        {
            sol[++snum]=(stk[sze--]+i)%mod;
            oc2[i]--;
        }
    }
    sort(sol+1,sol+1+n);
    for (int i=n; i>0; i--)
    printf("%d ",sol[i]);

}
