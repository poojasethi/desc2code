
#include <cstdio>
#include <algorithm>
#include <functional>
using namespace std;

#define FOR(i,n) for(int i=0;i<n;++i)
const int N=100010;
int n,m;
int Q[N],S[N];
int head,tail,top,pt;
int A[N],B[N];
int cA[N], cB[N];
int ans[N];

int main()
{
	scanf("%d%d",&n,&m);

	FOR(i,n) scanf("%d",A+i);
	FOR(i,n) scanf("%d",B+i);

	FOR(i,n) ++cA[ A[i] ], ++cB[B[i]];

	int i = m;
	while(i--){
		// for element with value i in A, 
		// the greedy choice is : m-1-i, m-2-i, ..., 0
		while( cB[ m-1-i ]--) S[top++] = m-1-i;
		while( cA[i] --){
			if(top) ans[pt++] = i + S[--top];
			else Q[tail++] = i;
		}
	}

	while(top) ans[pt++] = (Q[head++] + S[--top])%m;

	sort(ans,ans+pt,greater<int>());

	FOR(i,n) printf("%d ",ans[i]);
	return 0;
}
