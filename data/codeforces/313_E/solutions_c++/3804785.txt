#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

int n, m;
int x, qt, pt, rt; 
int A[100010], B[100010], q[100010], p[100010], r[100010];
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);

	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; ++i)
		scanf("%d", &x), A[x]++;

	for (int i = 1; i <= n; ++i)
		scanf("%d", &x), B[x]++;

	for (int i = m - 1; i >= 0; --i)  {
		while (A[i]--)
			q[++qt] = i;

		while (B[m - i - 1]--)
			p[++pt] = m - i - 1;

		while (qt)  {
			if (!pt  ||  q[qt] + p[pt] >= m)  break;
			r[++rt] = q[qt] + p[pt];  qt--;  pt--; 
		}
	}

	pt = 1;
	while (qt)  r[++rt] = (q[qt] + p[pt]) % m, qt--, pt++;
	sort(r + 1, r + rt + 1);

	for (int i = n; i; --i)
		printf("%d ", r[i]);
}
