#include<bits/stdc++.h>
using namespace std;
const int maxn=100*1000+1000;
int a[maxn], b[maxn], c[maxn];
stack<int> st;
int main(){
	int n, m;
	cin >> n >> m;
	for(int i=0;i<n;i++){
		int x;
		cin >> x;
		a[x%m]++;
	}
	for(int i=0;i<n;i++){
		int x;
		cin >> x;
		b[x%m]++;
	}
	for(int k=0;k<2;k++)
		for(int i=0;i<m;i++){
			while(a[i]>0)
				st.push(i), a[i]--;
			while(st.size() && b[m-i-1]>0)
				c[(m-i-1+st.top())%m]++, b[m-i-1]--, st.pop();	
		}
	for(int i=m-1;i>=0;i--){
		while(c[i]>0)
			cout << i << " ", c[i]--;
	}
	cout << endl;
}
