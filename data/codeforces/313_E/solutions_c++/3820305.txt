#include <iostream>
#include <stack>
#include <queue>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	int n, m;
	cin >> n >> m;

	int a[2][m], ans[m];
	for (int i = 0; i < m; ++i)
		a[0][i] = a[1][i] = ans[i] = 0;
	int tmp;
	for (int i = 0; i < 2; ++i) {
		for (int j = 0; j < n; ++j) {
			cin >> tmp;
			a[i][tmp]++;
		}
	}
	
	stack<int> s;
	queue<int> q;
	for (int i = 0; i < m;) {
		while (a[0][i]--)
			s.push(i);
		i = m-1-i;
		while (a[1][i]--)
			if (s.empty())
				q.push(i);
			else {
				ans[i+s.top()]++;
				s.pop();
			}
		i = m-i;
	}
	while (!s.empty()) {
		ans[s.top()+q.front()-m]++;
		s.pop(), q.pop();
	}

	for (int i = m-1; i >= 0; --i)
		while (ans[i]--)
			cout << i << ' ';
	return 0;
}
