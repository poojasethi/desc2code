#include <iostream>
#include <cstring>
using namespace std;

const int maxn = 100006;
int n, m, a[maxn], b[maxn], w[maxn], tot, ha[maxn], hb[maxn], ans[maxn];

int main()
{
	ios::sync_with_stdio(false);
	cin >> n >> m;
	memset(ha,0,sizeof(ha));
	memset(hb,0,sizeof(hb));
	memset(ans,0,sizeof(ans));
	for (int i = 0; i < n; ++i)
	{
		cin >> a[i];
		ha[a[i]]++;
	}
	for (int i = 0; i < n; ++i)
	{
		cin >> b[i];
		hb[b[i]]++;
	}
	tot = 0;
	for (int i = 0; i < m; ++i)
	{
		while (ha[i]) {
			w[++tot] = i;
			ha[i]--;
		}
		while (tot && hb[m-i-1]){
			ans[w[tot]+m-i-1]++;
			tot--;
			hb[m-i-1]--;
		}
	}
	for (int i = m-1; i >=0; i--)
	{
		while (tot && hb[i]){
			ans[(w[tot]+i)%m]++;
			tot--;
			hb[i]--;
		}
	}
	int sum = 0;
	for (int i = m-1; i >=0; i--)
	{
		for (int j = 0; j < ans[i]; ++j)
		{
			cout << i;
			sum++;
			if (sum!=n) cout << ' ';
		}
	}
	cout << endl;
}
