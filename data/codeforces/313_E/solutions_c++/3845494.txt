#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <map>
#include <vector>

using namespace std;

const int N = 100010;

int n, m;
int a[N], b[N];
int st[N], top;
vector<int> ans, s;

int main(){
    scanf("%d %d", &n, &m);
    for(int i = 0, x; i < n; ++i){
        scanf("%d", &x);
        a[x] ++;
    }
    for(int i = 0, x; i < n; ++i){
        scanf("%d", &x);
        b[x] ++;
    }
    for(int i = 0; i < m; ++i){
        for(int j = 0; j < b[i]; ++j) st[++top] = i;
        for(int j = 0; j < a[m - 1 - i]; ++j){
            if(top)
                ans.push_back((st[top--] + m - 1 - i) % m);
            else
                s.push_back(m - 1 - i);
        }
    }
    for(int i = 0; i < s.size(); ++i)
        ans.push_back((s[i] + st[top--]) % m);
    sort(ans.begin(), ans.end());
    for(int i = ans.size() - 1; i >= 0; --i)
        printf("%d ", ans[i]);
    return 0;
}
