#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <stack>
#include <queue>
using namespace std;

const int N = 100010;
stack<int> sta;
queue<int> q;
int a[N],b[N];
int ans[N];

int main()
{
	int n,m,x;
	memset(a,0,sizeof(a));
	memset(b,0,sizeof(b));
	scanf("%d%d",&n,&m);
	for(int i = 1; i <= n; i ++) {
		scanf("%d",&x);
		a[x] ++;
	}
	for(int i = 1; i <= n; i ++) {
		scanf("%d",&x);
		b[m - 1 - x] ++;
	}
	int cnt = 0;
	for(int i = m - 1; i >= 0; i --) {
		while(b[i] --) sta.push(m - i - 1);
		while(a[i] --) {
			if(sta.empty()) q.push(i);
			else {
				ans[++ cnt] = i + sta.top();
				sta.pop();
			}
		}
	}
	while(!sta.empty()) {
		ans[++ cnt] = sta.top() + q.front() - m;
		sta.pop();
		q.pop();
	}
	sort(ans + 1,ans + n + 1);
	for(int i = n; i >= 1; i --) {
		printf("%d",ans[i]);
		if(i == 1) printf("\n");
		else printf(" ");
	}

	return 0;
}
