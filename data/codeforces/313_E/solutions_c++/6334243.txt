#include <stdio.h>
#include <algorithm>
using namespace std;

const int M = 100010;
int s[M], a[M], b[M], q[M], r[M];
int top, z, tail;

int main() {
	int n, m, x;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; i++) {
		scanf("%d", &x);
		a[x]++;
	}
	for (int i = 0; i < n; i++) {
		scanf("%d", &x);
		b[x]++;
	}
	for (int i = 0; i < m; i++) {
		int t = m - 1 - i;
		while (b[i]--) s[++top] = i;
		while (a[t]--) {	
			if (top) {
				r[z++] = s[top--] + t;
			} else {
				q[tail++] = t;
			}
		}
	}
	for (int i = 0; i < tail; i++) {
		r[z++] = s[top--] + q[i] - m;
	}
	sort(r, r + z);
	for (int i = z - 1; i > 0; i--) {
		printf("%d ", r[i]);
	}
	printf("%d\n", r[0]);
	return 0;
}
