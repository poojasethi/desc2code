#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define N 120000
using namespace std;
int i,j,k,l,n,m,x,y,a[N],b[N],res[N],q[N],ans[N],fans,tot;
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)scanf("%d",&x),a[x]++;
	for(i=1;i<=n;++i)scanf("%d",&x),b[x]++;
	tot=0; fans=0;
	for(i=m-1;i>=0;--i){
		while(b[m-1-i]--)q[++tot]=m-1-i;
		while(a[i]--){
			if(tot)ans[++fans]=i+q[tot--];
			else res[++res[0]]=i;
		}
	}
	for(i=1;i<=res[0];++i)ans[++fans]=res[i]+q[tot--]-m;
	sort(ans+1,ans+fans+1);
	for(i=fans;i>=1;--i)printf("%d ",ans[i]);
	return 0;
}
