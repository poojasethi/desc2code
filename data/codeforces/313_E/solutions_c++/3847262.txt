#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int N=100010;
int n,m;
int a[N],b[N];
int p[N],q[N],pc,qc;
int ans[N],c;
int main() {
	scanf("%d%d",&n,&m);
	for(int i=1,x;i<=n;i++) {
		scanf("%d",&x);
		a[x]++;
	}
	for(int i=1,x;i<=n;i++) {
		scanf("%d",&x);
		b[x]++;
	}
	for(int i=m-1;i>=0;i--) {
		while(a[i]--) p[++pc]=i;
		while(b[m-i-1]--) q[++qc]=m-i-1;
		while(pc) {
			if(!qc||p[pc]+q[qc]>=m) break;
			ans[++c]=p[pc]+q[qc];
			pc--;qc--;
		}
	}
	for(int i=1,j=qc;i<=pc;i++,j--)
		ans[++c]=(p[i]+q[j])%m;
	sort(ans+1,ans+1+c);
	for(int i=n;i>=1;i--) printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
