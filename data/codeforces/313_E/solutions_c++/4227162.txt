#include<cstdio>
#include<cstring>
#include<algorithm>
#include<functional>

const int Maxn=1e5+5;
int n,m,a[Maxn],b[Maxn],lefta[Maxn],leftb[Maxn],numa,numb,x,ans[Maxn],ansnum;

int main()
{
	scanf("%d%d",&n,&m);
	memset(a,0,sizeof(a));
	memset(b,0,sizeof(b));
	numa=numb=ansnum=0;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&x);
		a[x]++;
	}
	for(int i=0;i<n;i++)
	{
		scanf("%d",&x);
		b[x]++;
	}

	for(int i=m-1;i>=0;i--)
	{
		while(b[m-1-i]--)
		  leftb[numb++]=m-1-i;
		while(a[i]--)
		  if(numb)
		    ans[ansnum++]=i+leftb[--numb];
		  else
			lefta[numa++]=i;
	}

	for(int i=0;i<numa;i++)
	  ans[ansnum++]=leftb[i]+lefta[numa-i-1]-m;
	std::sort(ans,ans+n,std::greater<int>());

	for(int i=0;i<n;i++)
	  printf("%d ",ans[i]);
	return 0;
}
