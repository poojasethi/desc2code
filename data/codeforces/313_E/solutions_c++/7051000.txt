#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<iostream>
#define N 100050
using namespace std;
int q1[N],q2[N];
int c1[N],c2[N];
int ans[N];
int main()
{
	int n,m;
	memset(c1,0,sizeof(c1));
	memset(c2,0,sizeof(c2));
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;++i)
	{
		int x;
		scanf("%d",&x);
		c1[x]++;
	}
	for(int i=0;i<n;++i)
	{
		int x;
		scanf("%d",&x);
		c2[x]++;
	}
	int top1=0,top2=0,tail=0;
	for(int i=m-1;i>=0;--i)
	{
		int tmp=m-1-i;
		//if(i==2)printf("%d %d\n",c1[i],c2[tmp]);
		while(c2[tmp]--)q2[top2++]=tmp;
		while(c1[i]--)
		{
			if(top2!=0)
			{
				ans[tail++]=i+q2[--top2];
			}
			else
			{
				q1[top1++]=i;
			}
		}
	}
	int now=0;
	while(top2)
	{
		ans[tail++]=(q2[--top2]+q1[now++])%m;
	}
	sort(ans,ans+n);
	for(int i=n-1;i>=0;--i)
	{
		printf("%d",ans[i]);
		if(i==0)puts("");
		else printf(" ");
	}
}
