#include <algorithm>
#include <stdio.h>
#include <vector>
#define  maxn      100000
#define  pb        push_back
using    namespace std;

int n,m,cnt1,cnt2;
int a[maxn];
int b[maxn];
int ar[maxn];
int arr[maxn];
vector<int>ans;

int main()
{
  scanf("%d%d",&n,&m);
  for(int i=0,x ; i<n ; i++) { scanf("%d",&x); a[x]++; }
  for(int i=0,x ; i<n ; i++) { scanf("%d",&x); b[x]++; }
  for(int i=m-1 ; i>=0 ; i--)
  {
    while(b[m-1-i])
    {
      b[m-1-i]--;
      ar[cnt1++]=m-1-i;
    }
    while(a[i])
    {
      a[i]--;
      if(cnt1) ans.pb(i+ar[--cnt1]);
      else     arr[cnt2++]=i;
    }
  }
  for(int i=0 ; i<cnt2 ; i++)  ans.pb(arr[i]+ar[cnt2-1-i]-m);
  sort(ans.begin() , ans.end());
  for(int i=n-1 ; i>=0 ; i--)
    printf("%d ",ans[i]);
  printf("\n");
  return 0;
}
