#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <string.h>
#include <math.h>
using namespace std;
#define MXN 100010
int n,m,a[MXN],b[MXN],q1[MXN],q2[MXN],tot1,tot2,record[MXN],tot;
void Init()
{
	int x;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) {scanf("%d",&x);a[x]++;}
	for(int i=1;i<=n;i++) {scanf("%d",&x);b[x]++;}
}
void solve()
{
	for(int i=0;i<=m-1;i++)
	  {
	  	while(a[i]--) q1[++tot1]=i;
	  	while(b[m-1-i]--)
	  	  {
	  	  	if(tot1) record[++tot]=m-1-i+q1[tot1--];
	  	  	else q2[++tot2]=m-1-i;
	      }
	  }
    for(int i=1;i<=tot2;i++)
      record[++tot]=q2[i]+q1[tot1--]-m;
    sort(record+1,record+tot+1);
    for(int i=tot;i>=1;i--) printf("%d ",record[i]);
    printf("\n");
}
int main()
{
	Init();
	solve();
    return 0;
}
