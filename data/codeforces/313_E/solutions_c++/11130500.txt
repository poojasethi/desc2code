#include <bits/stdc++.h>
using namespace std;

const int N=100010;
int n, m, x, a, b, A[N], B[N], visA[N], visB[N];
vector<int> pr;

int main()
{
   //freopen("r.txt","r",stdin);
   scanf("%d %d",&n, &m);
   for(int i=0;i<n;i++)
   {
       scanf("%d",&x);
       visA[x]++;
   }
   for(int i=0;i<n;i++)
   {
       scanf("%d",&x);
       visB[x]++;
   }
   int i=0, j=m-1;
   while(j>=0 && i<m)
   {
       while(visB[i])
       {
           B[b++]=i, visB[i]--;
       }
       while(visA[j])
       {
           if(b>0)///first part of the solution
             pr.push_back(j+B[--b]);
           else
             A[a++]=j;
           visA[j]--;
       }
       j--, i++;
   }
   for(int z=0;z<a;z++)//second part
      pr.push_back(A[z]+B[--b]-m);
   sort(pr.begin(), pr.end());
   for(int z=n-1;z>=0;z--)
      printf("%d ",pr[z]);

    return 0;
}
