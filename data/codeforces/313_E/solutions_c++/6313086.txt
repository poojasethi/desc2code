#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<queue>
#define LL __int64
#define MIN(x,y) ((x)<(y)?(x):(y))
using namespace std;
priority_queue<int > q;
int a[201000],b[101000],c[101000];
bool cmp(int x,int y)
{return x>y;}
int main()
{
	int i,j,m,n,k,t,x,y;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<n;i++)
		scanf("%d",&b[i]);
	sort(a,a+n,cmp);
	sort(b,b+n,cmp);
	int head=0,tail=n-1,now=n-1,cnt=0;
	for(;head<=tail;)
	{
		while(now>=0&&b[now]+a[head]<=m-1)
			q.push(b[now--]);
		if(q.empty()) a[++tail]=a[head]-m;
		else
		{
			c[cnt++]=a[head]+q.top();
			q.pop();
		}
		head++;
	}
	sort(c,c+n,cmp);
	printf("%d",c[0]);
	for(i=1;i<n;i++)
		printf(" %d",c[i]);
	puts("");
}
