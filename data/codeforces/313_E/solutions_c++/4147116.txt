#include <cstdio>
#include <algorithm>
#define N 111111
#define rep(i,a,b) for(i=a;i<=b;i++)
#define dep(i,a,b) for(i=a;i>=b;i--)
using namespace std;
int i, n, m, x, y, S, l;
int a[N], b[N], t[N], q[N], c[N];
int main(){
	scanf("%d%d", &n, &m);
	rep(i, 1, n) scanf("%d", &x), a[x]++;
	rep(i, 1, n) scanf("%d", &x), b[x]++;
	dep(i, m - 1, 0){
		while(b[y = m - 1 - i]--) q[++l] = y;
		while(a[i]--)
			if(l) c[++S] = i + q[l--];
			else t[++t[0]] = i;
	}
	rep(i, 1, t[0]) c[++S] = t[i] + q[l--] - m;
	sort(c + 1, c + S + 1);
	dep(i, S, 1) printf("%d ", c[i]);
}
