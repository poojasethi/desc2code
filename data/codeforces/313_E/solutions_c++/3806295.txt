#include <iostream>
#include <stdio.h>
#include <queue>
#include <stack>
#include <string.h>
#include <algorithm>

using namespace std;

int a[100005],b[100005];
vector<int> ans;

int main(void){
    //freopen("D:\\acm.txt","r",stdin);
    int n,m,tmp;
    scanf("%d %d",&n,&m);
    for(int i = 0;i<n;i++) scanf("%d",&tmp),a[tmp]++;
    for(int i = 0;i<n;i++){
        scanf("%d",&tmp);
        b[m-1-tmp]++;
    }
    queue<int> q;
    stack<int> s;
    for(int i = m-1;i>=0;i--){
        while(b[i]--) s.push(m-1-i);
        while(a[i]--){
            if(!s.empty()){
                ans.push_back((i+s.top())%m);s.pop();
            }else q.push(i);
        }
    }
    while(!s.empty()){
        ans.push_back((s.top()+q.front())%m);
        q.pop();s.pop();
    }
    sort(ans.begin(),ans.end());
    for(int i = n-1;i>=0;i--) printf("%d ",ans[i]);
    printf("\n");
}
