#include<iostream>
#include<algorithm>
#include<cstdio>
#include<vector>
#include<queue>
#include<stack>
using namespace std;
int n,m,u[110000],d[110000],v;
queue<int>q;
stack<int>s;
vector<int>r;
int main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>v,u[v]++;
	for(int i=0;i<n;i++)
		cin>>v,d[m-1-v]++;
	for(int i=m;i--;)
	{
		while(d[i]--)
		{
		    s.push(m-1-i);
		}
		while(u[i]--)
		{
			if(!s.empty())
			{
			    r.push_back(i+s.top());
			    s.pop();
			}
			else
			{
			    q.push(i);
			}
		}
	}
	while(!q.empty())
	{
	    r.push_back(q.front()+s.top()-m);
	    q.pop();
	    s.pop();
	}
	sort(r.begin(),r.end());
	for(int i=n;i--;)
	{
	    printf("%d ",r[i]);
	}
	return 0;
}
