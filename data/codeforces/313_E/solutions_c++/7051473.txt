#include<iostream>
#include<stack>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 100001
int a[N],b[N],ans[N],vis[N];
stack<int>s;
int cmp(int x,int y)
{
	return x>y;
}
int main()
{
	int n,m;
	while(cin>>n>>m)
	{
		memset(vis,0,sizeof(vis));
		while(!s.empty())s.pop();
		int i,j;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			a[i]%=m;
		}
		for(i=0;i<n;i++)
		{
			cin>>b[i];
			b[i]%=m;
		}
		sort(a,a+n,cmp);
		sort(b,b+n);
		int l=0;
		j=0;
		for(i=0;i<n;i++)
		{
			while(j<n)
			{
				if(a[i]+b[j]<m)
				s.push(b[j++]);
				else
				break;
			}
			if(s.empty())vis[i]=1;
			else
			{
				ans[l++]=a[i]+s.top();
				s.pop();
			}
		}
		while(j<n)s.push(b[j++]);
		for(i=0;i<n;i++)
		if(vis[i])
		{
			ans[l++]=(a[i]+s.top())%m;
			s.pop();
		}
		sort(ans,ans+l,cmp);
		for(i=0;i<n;i++)
		{
			cout<<ans[i];
			if(i==n-1)cout<<endl;
			else cout<<' ';
		}
	}
}
