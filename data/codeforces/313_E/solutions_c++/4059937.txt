#include <iostream>
#include <algorithm>
#include <stack>
#include <queue>
#include <vector>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,m,a[100010],b[100010],r[100010],t;

stack<int> s;
queue<int> q;
vector<int> v;

int main(){
    cin>>n>>m;
    rep(i,n){cin>>t;a[t]++;}
    rep(i,n){cin>>t;b[m-t-1]++;}
    rep(i,m){
    while(a[i]--)s.push(i);
    while(b[i]--){
        if(s.empty())q.push(m-i-1);
        else {v.push_back((s.top()+m-i-1)%m);s.pop();}
    }
    }
    while(!s.empty()){
    v.push_back((s.top()+q.front())%m);
    s.pop();q.pop();
    }
    sort(v.begin(),v.end(),greater<int>());
    
    rep(i,n)cout<<v[i]<<" ";cout<<endl;
}
