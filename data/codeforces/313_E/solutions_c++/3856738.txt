#include <stack>
#include <cstdio>
#include <algorithm>
#define MXN 100010
using namespace std;

int N, M, X, A[MXN], B[MXN];
int ans[MXN], an;

stack<int> s;

int main (){
	scanf("%d%d", &N, &M);
	for(int i = 0; i < N; i++) scanf("%d", &X), A[X]++;
	for(int i = 0; i < N; i++) scanf("%d", &X), B[X]++;
	for(int i = 0; i < M; i++){
		while(A[i] && B[M-i-1]){
			A[i]--, B[M-i-1]--;
			ans[an++] = M-1;
		}
		while(A[i]--) s.push(i);
		while(B[M-i-1] && !s.empty()){
			B[M-i-1]--;
			ans[an++] = (s.top()+M-i-1) % M;
			s.pop();
		}
	}
	for(int i = M-1; i >= 0; i--){
		while(!s.empty() && B[i]){
			B[i]--;
			ans[an++] = (s.top()+i)%M;
			s.pop();
		}
	}
	sort(ans, ans + N);
	reverse(ans, ans + N);
	for(int i = 0; i < N; i++)
		printf("%d ", ans[i]);
}
