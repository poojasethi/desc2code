#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f1[N],f2[N],g[N],an[N],re[N];
int main() {
	//freopen("313e.in","r",stdin);
	scanf("%d%d",&n,&m);
	For(i,1,n) scanf("%d",&k),f1[k]++;
	For(i,1,n) scanf("%d",&k),f2[k]++;
	for (i=m-1;i>=0;i--) {
		for (;f2[m-1-i];f2[m-1-i]--) g[++*g]=m-1-i;
		for (;f1[i];f1[i]--) {
			if (*g) an[++*an]=i+g[(*g)--];
			else re[++*re]=i;
		}
	}
	For(i,1,*re) an[++*an]=re[i]+g[(*g)--]-m;
	sort(an+1,an+n+1); reverse(an+1,an+n+1);
	For(i,1,n) printf("%d ",an[i]);
	printf("\n");
	//for(;;);
	return 0;
}
