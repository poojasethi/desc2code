#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<queue>
#include<stack>
using namespace std;
#define N 100005
int A[N],B[N];
vector<int> C;
queue<int> Q;
stack<int> S;

int main()
{
    int n,m,t;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;++i) scanf("%d",&t),++A[t];
    for(int i=0;i<n;++i) scanf("%d",&t),++B[t];
    for(int i=m-1;i>=0;--i)
    {
        while(B[m-1-i]--) S.push(m-1-i);
        while(A[i]--)
        {
            if(!S.empty()) C.push_back(S.top()+i),S.pop();
            else Q.push(i);
        }
    }
    while(!Q.empty())
    {
        C.push_back(Q.front()+S.top()-m);
        Q.pop();S.pop();
    }
    sort(C.begin(),C.end(),greater<int>());
    for(int i=0;i<C.size();++i) printf("%d ",C[i]);
    printf("\n");
    return 0;
}
