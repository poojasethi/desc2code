#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<map>
#define N 220000
using namespace std;
map<pair<int,int> ,int> M;
int n,m,len,head[N],e[N],px[N],py[N],t,du[N],id[N],c[N],next[N],i,b[N],x,y,j,qn,k,a[N];
int *sum[N],*w[N];
void add(int x,int y){
	next[++len]=head[x]; head[x]=len; e[len]=y;
	M[make_pair(x,y)]=len;
	du[x]++;
}
bool cmp(const int &x,const int &y){
	int u=e[x],v=e[y];
	double t1=atan2(py[u]-py[i],px[u]-px[i]),
		   t2=atan2(py[v]-py[i],px[v]-px[i]);
	return t1<t2;
}
int dfs(int x){
	int res=0;
	b[x]=1;
	for(int i=head[x];i;i=next[i]){
		if(b[e[i]])continue;
		int u=dfs(e[i]);
		c[i]+=u;
		c[i^1]-=u;
		res+=u;
	}
	return res+1;
}
int main(){
	scanf("%d%d",&n,&m);
	len=1;
	for(i=1;i<=m;++i){
		scanf("%d%d",&x,&y);
		add(x,y);
		add(y,x);
	}
	for(i=1;i<=n;++i)scanf("%d%d",&px[i],&py[i]);
	t=1;
	for(i=2;i<=n;++i)if(px[i]<px[t])t=i;
	dfs(t);
	for(i=1;i<=n;++i){
		w[i]=new int[du[i]+5];
		sum[i]=new int[du[i]+5];
		du[i]=0;
		for(j=head[i];j;j=next[j])w[i][++du[i]]=j;
		sort(w[i]+1,w[i]+du[i]+1,cmp);
		sum[i][0]=0;
		for(j=1;j<=du[i];++j){
			id[w[i][j]]=j;
			sum[i][j]=sum[i][j-1]+c[w[i][j]];
		}
	}
	scanf("%d",&qn);
	for(i=1;i<=qn;++i){
		scanf("%d",&k);
		for(j=1;j<=k;++j)scanf("%d",&a[j]); a[k+1]=a[1];
		long long area=0;
		for(j=1;j<=k;++j)area+=1ll*px[a[j]]*py[a[j+1]]-1ll*py[a[j]]*px[a[j+1]];
		if(area<0)for(j=1;j+j<=k;++j)swap(a[j],a[k-j+1]);
		a[0]=a[k]; a[k+1]=a[1];
		int ans=0;
		for(j=1;j<=k;++j){
			int x=id[M[make_pair(a[j],a[j-1])]],y=id[M[make_pair(a[j],a[j+1])]];
			if(y<x)ans+=sum[a[j]][x]-sum[a[j]][y-1];
			else ans+=sum[a[j]][du[a[j]]]-sum[a[j]][y-1]+sum[a[j]][x];
		}
		printf("%d\n",ans+k);
	}
}
