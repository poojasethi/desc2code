#include <iostream>
#include <istream>
#include <ostream>
#include <algorithm>
#include <string>
#include <vector>
#include <utility>
#include <limits>
#include <bitset>
#include <cmath>

using namespace std;

typedef long long ll;

int n, m;
pair<ll, ll> ps[1010];
bitset<1010> ss[1010];

int main()
{
    ios_base::sync_with_stdio(false);
    
    int k;
    cin >> n >> m >> k;

    for(int i = 0; i < n; ++i) {
        cin >> ps[i].first >> ps[i].second;
    }

    for(int i = 0; i < m; ++i) {
        ll r, x, y;
        cin >> r >> x >> y;
        
        for(int j = 0; j < n; ++j) {
            ll dx = x - ps[j].first;
            ll dy = y - ps[j].second;
            ss[j][i] = sqrt(dx * dx + dy * dy) < r;
        }
    }

    while(k-- > 0) {
        int a, b;
        cin >> a >> b;
        --a; --b;
        cout << (ss[a] ^ ss[b]).count() << endl;
    }

    return 0;
}
