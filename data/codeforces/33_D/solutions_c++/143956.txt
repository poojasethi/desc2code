#include<iostream>
#include<cstdio>
#include<map>
#include<algorithm>
#include<cmath>

using namespace std;

int kx[1000],ky[1000];
bool ins[1000][1000];

inline double sqr(double x) {return x*x; }
inline double dist(double x1,double y1,double x2,double y2)
{
	return sqrt(sqr(x1-x2)+sqr(y1-y2));
}

int main()
{
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;i++)
		scanf("%d%d",&kx[i],&ky[i]);
	for(int i=0;i<m;i++)
	{
		int r,cx,cy;
		scanf("%d%d%d",&r,&cx,&cy);
		for(int j=0;j<n;j++)
			ins[j][i] = dist(kx[j],ky[j],cx,cy)-r<-1e-6;
	}

	for(int i=0;i<k;i++)
	{
		int a,b,c=0;
		scanf("%d%d",&a,&b);
		--a;--b;
		for(int j=0;j<m;j++)
			c+=(ins[a][j]^ins[b][j]);
		printf("%d\n",c);
	}

	return 0;
}
