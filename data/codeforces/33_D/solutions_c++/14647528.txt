#include<bits/stdc++.h>
using namespace std;
int main(){
    long long int n,m,k,i,j,a,b,sol;
    scanf("%I64d %I64d %I64d",&n,&m,&k);

    long long int ara[n][2];
    long long int ans[n][m];
    for(i=0;i<n;i++) scanf("%I64d %I64d",&ara[i][0],&ara[i][1]);

    long long int r,x,y;
    for(i=0;i<m;i++){
        scanf("%I64d %I64d %I64d",&r,&x,&y);
        for(j=0;j<n;j++){
            long long int temp;
            long long int tem;
            tem=(x-ara[j][0]);
            temp=tem*tem-r*r;
            tem=(y-ara[j][1]);
            temp+=tem*tem;
            if(temp<0) ans[j][i]=-1;
            else ans[j][i]=1;
        }
    }

    for(i=0;i<k;i++){
        scanf("%I64d %I64d",&a,&b);
        a--;
        b--;
        sol=0;
        for(j=0;j<m;j++){
            if(ans[a][j]*ans[b][j]<0) sol++;
        }

        printf("%I64d\n",sol);
    }

}
