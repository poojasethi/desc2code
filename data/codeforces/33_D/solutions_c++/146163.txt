#include <iostream>
using namespace std;
typedef long long int lint;
int n, m, k;
int pos[1000][2];
bool in[1000][1000];

int main(void)
{
  cin >> n >> m >> k;
  for( int i = 0 ; i < n ; ++i )
    cin >> pos[i][0] >> pos[i][1];
  for( int i = 0 ; i < m ; ++i ) {
    lint x, y, r;
    cin >> r >> x >> y;
    r *= r;
    for( int j = 0 ; j < n ; ++j ) {
      lint dx = x - pos[j][0];
      lint dy = y - pos[j][1];
      in[j][i] = (dx*dx + dy*dy < r);
    }
  }

  for( int i = 0 ; i < k ; ++i ) {
    int a, b;
    cin >> a >> b;
    int cnt = 0;
    --a, --b;
    for( int j = 0 ; j < m ; ++j )
      if( in[a][j] != in[b][j] ) ++cnt;
    cout << cnt << "\n";
  }

  return 0;
}
