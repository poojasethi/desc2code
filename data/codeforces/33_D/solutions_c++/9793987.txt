#define _CRT_DISABLE_PERFCRIT_LOCKS
#include <stdio.h>
#include <bitset>
#define sqr(x) ((x)*(x))
using namespace std;
template<class T> inline void read(T&x){bool neg=0;char c;for(c=getchar();c<=32;c=getchar());if(c == '-')neg=1,c=getchar();for(x=0;c>32;c=getchar())x=x*10+c-'0';if(neg)x=-x;};long long px[1000], py[1000], cx[1000], cy[1000], cr[1000]; bitset<1000> mask[1000]; int n,m,k,from,to,i=0,j=0;
inline void print(int x){for(;x;x/=10)putchar(char(x%10+'0'));putchar('\n');}
int main(){
    read(n); read(m); read(k);
    for(;i<n;i++){read(px[i]); read(py[i]);}
    for(;j<m;j++){read(cr[j]); read(cx[j]); read(cy[j]);}
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
            if(sqr(px[i]-cx[j]) + sqr(py[i]-cy[j]) <= sqr(cr[j]))
                mask[i][j]=1;
    for(;k; k--){
        read(from); read(to);
        printf("%d\n",(mask[from-1]^mask[to-1]).count());
    }
    return 0;}
