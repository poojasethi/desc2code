#include <cstdio>
#include <bitset>

using namespace std;

int const N = 1000;
typedef bitset<N> bs_t;

bs_t	s[N];
int		x[N];
int 	y[N];

int main()
{
	int n, m, k;
	scanf("%d %d %d", &n, &m, &k);
	for( int i = 0; i != n; ++i )
		scanf("%d %d", &x[i], &y[i]);
		
	for( int i = 0; i != m; ++i )
	{
		int r, cx, cy;
		scanf("%d %d %d", &r, &cx, &cy);
		for( int j = 0; j != n; ++j )
		{
			long long dx = (x[j] - cx);
			long long dy = (y[j] - cy);
			if( dx*dx + dy*dy <= 1ll*r*r )
				s[j].set(i);
		}
	}

	for( int i = 0; i != k; ++i )
	{
		int a, b;
		scanf("%d %d", &a, &b);
		printf("%d\n", (s[a-1]^s[b-1]).count());
	}
	return 0;
}
