#include<cstdio>
#include<algorithm>
using namespace std;
typedef struct point{
    double x,y;
    void read(){ scanf("%lf%lf",&x,&y); }
}Point;
struct circle{
    point p;
    double r;
    void read(){ scanf("%lf",&r); p.read(); }
}C[10001];
int n,m,k;
Point P[10001];
bool in(point p,circle c){
    return (c.r*c.r-(p.x-c.p.x)*(p.x-c.p.x) > (p.y-c.p.y)*(p.y-c.p.y) + 1e-12);
}
int main(){
    scanf("%d%d%d",&n,&m,&k);
    for(int i=1;i<=n;i++)
        P[i].read();
    for(int i=0;i<m;i++)
        C[i].read();
    for(int i=0;i<k;i++){
        int a,b,ans = 0;
        scanf("%d%d",&a,&b);
        for(int j=0;j<m;j++)
            if(in(P[a],C[j]) ^ in(P[b],C[j]))
                ans++;
        printf("%d\n",ans);
    }
    return 0;
}
