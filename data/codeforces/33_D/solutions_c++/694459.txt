#include <iostream>
#include <cstring>

#define sqr(A) (((double)A)*((double)A))

using namespace std;

typedef long long int ll;

ll n,m,k;
ll kx[1010],ky[1010];
ll cx[1010],cy[1010],r[1010];
int bitmask[1010][40];

static inline void setbit(int idx, int pos) {
	bitmask[idx][pos/32] |= (1<<(pos%32));
}

int main () {
	memset(bitmask,0,sizeof(bitmask));
	cin >> n >> m >> k;
	for(int i = 0; i < n; i++) cin >> kx[i] >> ky[i];
	for(int i = 0; i < m; i++) cin >> r[i] >> cx[i] >> cy[i];
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < m; j++) {
			if(sqr(cx[j]-kx[i])+sqr(cy[j]-ky[i])+10E-3 < sqr(r[j])) setbit(i,j);
		}
	}
	int a,b;
	for(int i = 0; i < k; i++) {
		cin >> a >> b;a--,b--;
		int res = 0;
		for(int j = 0; j < 40; j++) {
			res += __builtin_popcount(bitmask[a][j]^bitmask[b][j]);
		}
		cout << res << endl;
	}
	return 0;
}
