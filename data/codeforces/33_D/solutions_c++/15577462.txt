#include <bits/stdc++.h>
using namespace std;
#define MAXN 1005
#define FI "knights.i2"
#define FO "knights.o2"
long long X[MAXN], Y[MAXN];
int N, M, K;
bitset<MAXN> Bit[MAXN];

int main(){
    ios_base::sync_with_stdio(0), cin.tie(0);
    #ifndef ONLINE_JUDGE
    ifstream cin(FI);
    ofstream cout(FO);
    #endif // ONLINE_JUDGE
    cin >> N >> M >> K;
    for(int i = 0; i < N; i++) {
        cin >> X[i] >> Y[i];
        //Bit[i].reset();
    }
    for(int i = 0; i < M; i++) {
        long long r, x, y;
        cin >> r >> x >> y;
        for(int j = 0; j < N; j++)
        if((x-X[j])*(x-X[j]) + (y-Y[j])*(y-Y[j]) <= r*r){
            Bit[j].set(i);
        }
    }

    for(int a, b, i = 0; i < K; i++){
        cin >> a >> b;
        cout << (Bit[a-1] ^ Bit[b-1]).count() <<'\n';
    }
    return 0;
}
