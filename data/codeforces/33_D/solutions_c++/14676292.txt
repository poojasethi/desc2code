#include<bits/stdc++.h>
using namespace std;
long long arayn[1000][2], araym[1000][3];
int flag[1001][1001];
int main()
{
    long long  a, b, x, y, p, q, s, t, c=0, r;
    int i, j, g, n, m, k, in1, in2;
    scanf("%d %d %d", &n, &m, &k);
    for(i=0;i<n;i++){
        scanf("%lld %lld", &arayn[i][0], &arayn[i][1]);
    }
    for(i=0;i<m;i++){
        scanf("%lld %lld %lld", &araym[i][0], &araym[i][1], &araym[i][2]);
    }
    for(i=0;i<n;i++){
        a=arayn[i][0];b=arayn[i][1];
        for(j=0;j<m;j++){
            x=araym[j][1];y=araym[j][2];r=araym[j][0];
            p=(((a-x)*(a-x))+((b-y)*(b-y))-(r*r));
            if(p<0) flag[i][j]=1;
        }
    }
    while(k){
        scanf("%d %d", &in1, &in2);
        in1--;in2--;
        c=0;
        for(i=0;i<m;i++){
            if((flag[in1][i]!=flag[in2][i])) c++;
        }
        printf("%lld\n" ,c);
        k--;
    }



    return 0;
}
