#include <iostream>
#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl
using namespace std;

int n,m,k,r,x[1010],y[1010],rx,ry,a[1010][1010],k1,k2,res;

int main(){
  
  cin>>n>>m>>k;
  rep(i,n)cin>>x[i]>>y[i];
  rep(i,m){
    cin>>r>>rx>>ry;
    rep(j,n)if((long long)(x[j]-rx)*(x[j]-rx)+(long long)(y[j]-ry)*(y[j]-ry)<(long long)(r)*r)a[i][j]=1;
  }

  rep(i,k){
    cin>>k1>>k2;k1--;k2--;
    res=0;
    rep(j,m)if(a[j][k1]!=a[j][k2])res++;
    cout<<res<<endl;

  }
  return 0;

}
