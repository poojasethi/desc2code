#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>
#include <cmath>
typedef long long ll;
const int N = 1000 + 10;
int x[N], y[N], r[N];

int xx[N], yy[N];

bool in[N][N];

ll sqr(int x) {
	return (ll)x * x;
}

ll dis(int i, int j, int k, int l) {
	return sqr(i - k) + sqr(j - l);
}

int main() {
	int n, m, K, u, v;
	scanf("%d%d%d", &n, &m, &K);
	for (int i = 1; i <= n; ++i)
		scanf("%d%d", &xx[i], &yy[i]);
	for (int i = 1; i <= m; ++i)
		scanf("%d%d%d", &r[i], &x[i], &y[i]);
		
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j) 
			if (dis(xx[i], yy[i], x[j], y[j]) < (ll)r[j] * r[j])
				in[i][j] = true;
		
	while (K -- > 0) {
		scanf("%d%d", &u, &v);
		int cnt = 0;
		for (int i = 1; i <= m; ++i) {
			if (in[u][i] && !in[v][i])
				++cnt;
			else if (in[v][i] && !in[u][i])
				++cnt;
		}
		printf("%d\n", cnt);
	}
	return 0;
}
