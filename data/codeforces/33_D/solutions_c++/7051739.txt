#include<cstdio>
#include<iostream>
#include<algorithm>
#include<string.h>
#include<math.h>
using namespace std;
#define point Point
#define N 1005
#define ll long long
struct node{
	ll x,y,r;
}a[N],c[N];
int h[N][N];
ll dis(node aa,node bb){
	return (aa.x-bb.x)*(aa.x-bb.x)+(aa.y-bb.y)*(aa.y-bb.y);
}
ll n,m,k;
int ans[100500], l[100500], r[100500];
int main(){
	ll i,j;
	while(cin>>n>>m>>k){
		for(i=1;i<=n;i++)
		scanf("%I64d %I64d",&a[i].x,&a[i].y);
		for(i=1;i<=m;i++)
		scanf("%I64d %I64d %I64d", &c[i].r, &c[i].x,&c[i].y), c[i].r = c[i].r*c[i].r;
		for(i = 1; i <= k; i++)
		scanf("%d %d",&l[i],&r[i]);
		memset(ans, 0, sizeof ans);
		for(i = 1; i <= m; i++) {
			for(j = 1; j<= n; j++)
					h[i][j] =  (dis(c[i],a[j])<=c[i].r );
		}
		for(i = 1; i <= m; i++) {
			for(j = 1; j <= k; j++) {
				if(h[i][l[j]] + h[i][r[j]] == 1)
				ans[j]++;
			}
		}
		for(i=1;i<=k;i++)printf("%d\n",ans[i]);
	}
    return 0;
}
