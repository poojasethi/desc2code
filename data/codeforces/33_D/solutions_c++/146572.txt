#include <cstring>
#include <string>
#include <iostream>
#include <cstdio>

using namespace std;

#define MAXN 1002

int n,m,k;
long long c[MAXN][2];
bool ct[MAXN][MAXN];


int main(){
    
   //freopen("in.txt","r",stdin); freopen("out.txt","w",stdout);
    cin>>n>>m>>k;
    for(int i=1;i<=n;++i){
      cin>>c[i][0]>>c[i][1];
    }
	for(int i=1;i<=m;++i){
      long long r,xi,yi;
	  cin>>r>>xi>>yi;
	  for(int j=1;j<=n;++j){
	    long long dx = xi-c[j][0];
		long long dy = yi-c[j][1];
		ct[j][i] = (dx*dx+dy*dy<r*r);
	  }
    }
     for(int i=0;i<k;++i){
       int a,b; cin>>a>>b;
	   int t =0;
	   for(int kx=1;kx<=m;++kx){
      
           if(ct[a][kx] == false && ct[b][kx] ==true) ++t;
		   else if(ct[a][kx] == true && ct[b][kx] == false ) ++t;
       }
       cout<<t<<endl;
     }
    
    return 0;
}
