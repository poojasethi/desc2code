#include <cstdlib>
#include <cstdio>
#include <cmath>
#include <bitset>

using namespace std;     

bitset<1000> f[1000];
int x[1000], y[1000];

int main() {
    int n, m, k;
    scanf ("%d%d%d", &n, &m, &k);
    for (int i = 0; i < n; ++i) {
        scanf ("%d%d", &x[i], &y[i]);
    }
    for (int r, cx, cy, i = 0; i < m; ++i) {
        scanf ("%d%d%d", &r, &cx, &cy);
        for (int j = 0; j < n; ++j) {
            f[j].set (i, pow (x[j] - cx, 2) + pow (y[j] - cy, 2) <= pow (r, 2));
        }
    }
    for (int a, b, i = 0; i < k; ++i) {
        scanf ("%d%d", &a, &b);
        --a, --b;
        printf ("%d\n", (f[a] ^ f[b]).count ());
    }
	return 0;
}
