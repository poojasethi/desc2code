#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<ll,ll> pii;
#define X first
#define Y second
const int N = 1001;
const int M = 100001;
pii a[N];
bitset<N> mask[N];
int n,m,k;

int main()
{
    ios_base::sync_with_stdio(0); cin.tie(0);
    #ifndef ONLINE_JUDGE
    freopen("Knights.inp","r",stdin);
    freopen("Knights.out","w",stdout);
    #endif // ONLINE_JUDGE
    cin >> n >> m >> k;
    for (int i=1; i<=n; i++) cin >> a[i].X >> a[i].Y;
    for (int i=0; i<m; i++) {
        ll x,y,r;
        cin >> r >> x >> y;
        for (int j=1; j<=n; j++) {
            if ((x-a[j].X)*(x-a[j].X)+(y-a[j].Y)*(y-a[j].Y) <= r*r) mask[j].set(i);
        }
    }
    for (int i=1; i<=k; i++) {
        int x,y;
        cin >> x >> y;
        cout << (mask[x]^mask[y]).count() << '\n';
    }
    return 0;
}
