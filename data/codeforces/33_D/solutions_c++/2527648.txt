#include<iostream>
#define Nmax 1010
using namespace std;
long long n, m, k, r, cx, cy, a, b, sol, j, i, muchie[Nmax][Nmax], x[Nmax], y[Nmax];
int main()
{
	cin >> n >> m >> k;
	for (i = 1; i <= n; ++i)
		cin >> x[i] >> y[i];
	for (i = 1; i <= m; ++i)
	{
		cin >> r >> cx >> cy;
		r=r*r;
		for (j = 1; j <= n; ++j)
			if ((((x[j] - cx) * (x[j] - cx)) + ((y[j] - cy) * (y[j] - cy))) < r)
				muchie[j][i] = 1;
	}		
	for (i = 1; i <= k; ++i)
	{
		cin >> a >> b;
		sol = 0;
		for (j = 1; j <= m; ++j)
		{
			if (muchie[a][j] + muchie[b][j] == 1)
				sol++;
		}
		cout << sol <<'\n';
	}
	return 0;
}
