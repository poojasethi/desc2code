#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

#define M 3001

int x[M], y[M], r[M], a[M], b[M];

bool can(int xx, int yy) {
	ll X = 1ll * (x[xx] - a[yy]) * (x[xx] - a[yy]) + 1ll * (y[xx] - b[yy]) * (y[xx] - b[yy]);
	ll Y = 1ll * r[yy] * r[yy];
	if(Y >= X) return 1;
	return 0;
}

int main() {
	///freopen("4.in", "r", stdin);
	int n, m, k;
	scanf("%d %d %d", &n, &m, &k);
	for(int i = 1; i <= n; i ++) scanf("%d %d", &x[i], &y[i]);
	for(int i = 1; i <= m; i ++) scanf("%d %d %d", &r[i], &a[i], &b[i]);
	for(int i = 1; i <= k; i ++) {
		int xx, yy;
		scanf("%d %d", &xx, &yy);
        int tot = 0;
        for(int j = 1; j <= m; j ++) {
            if(can(xx, j) + can(yy, j) == 1) tot ++;
        }
        cout << tot << endl;
	}
}
