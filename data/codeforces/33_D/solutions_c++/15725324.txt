#include<bits/stdc++.h>
using namespace std;
int a,b,t;
long long r,rx,ry,x[1010],y[1010];
bool f[1010][1010];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int n,m,k;
    cin>>n>>m>>k;
    for(int i=0; i<n; i++)
        cin>>x[i]>>y[i];
    for(int i=0; i<m; i++)
    {
        cin>>r>>rx>>ry;
        for(int j=0; j<n; j++)
            if((x[j]-rx)*(x[j]-rx)+(y[j]-ry)*(y[j]-ry)<r*r)
                f[i][j]=true;
    }
    for(int i=0; i<k; i++)
    {
        cin>>a>>b;
        a--;
        b--;
        t=0;
        for(int j=0; j<m; j++)
            if(f[j][a]!=f[j][b])
                t++;
        cout<<t<<'\n';
    }
    return 0;
}
