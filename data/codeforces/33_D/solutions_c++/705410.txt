#include<iostream>
#include<cstdio>
#include<cmath>
#define sqr(x) (x)*(x)
using namespace std;
double fx[1010], fy[1010], r[1010], kx[1010], ky[1010];
int a[1010][1010];
int i, j, t, res, n, m, k;

int infence(int i, int j){
    double d1 = sqr(kx[i]-fx[j]) + sqr(ky[i]-fy[j]), d2 = sqr(r[j]);
    return (d1 < d2);
}

int main(){
    cin>>n>>m>>k;
    for(i=1;i<=n;i++)scanf("%Lf%Lf",&kx[i], &ky[i]);//cin>>kx[i]>>ky[i];
    for(i=1;i<=m;i++)scanf("%Lf%Lf%Lf",&r[i],&fx[i],&fy[i]);//cin>>r[i]>>fx[i]>>fy[i];
    for(i=1;i<=n;i++)
      for(j=1;j<=m;j++) a[i][j] = infence(i,j);
    while(k--){
      cin>>i>>j;
      res = 0;
      for(t=1;t<=m;t++) res += a[i][t]^a[j][t];
      printf("%d\n",res);//cout<<res<<endl;
    }
}
