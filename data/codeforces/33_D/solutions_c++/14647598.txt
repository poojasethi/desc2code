#include<iostream>
#include<stdio.h>
using namespace std;
int in[1024][1024];
long long v[1024][2];
int main()
{
    int n, m, k;
    cin >> n >> m >> k;
    for(int i = 0; i < n; i++)
        cin >> v[i][0] >> v[i][1];
    long long r, x, y;
    for(int i = 0; i < m; i++)
    {
        cin >> r >> x >> y;
        for(int j = 0; j < n; j++)
        {
            if((v[j][0] - x) * (v[j][0] - x) + (v[j][1] - y) * (v[j][1] - y) <= r * r)
                in[j][i] = 1;
        }
    }
    int t, u;
    int c;
    for(int i = 0; i < k; i++)
    {
        scanf("%d", &t);
        scanf("%d", &u);
        t--;
        u--;
        c = 0;
        for(int j = 0; j < m; j++)
            if(in[t][j] != in[u][j])
                c++;
        printf("%d\n", c);
    }
    return 0;
}
