#include <cstdio> 
#include <vector> 
using namespace std; 
typedef long long ll; 
ll iX[1010],iY[1010],R[1010],oX[1010],oY[1010]; 
int n,m,Q; 
vector<int>mas[1010]; 
int U[1010],st=0; 
int main(){ 
  int i,j,x,y,z,L1,L2; 
  scanf("%d%d%d",&n,&m,&Q); 
  for(i=0;i<n;++i){ 
   scanf("%d%d",&x,&y); 
   iX[i]=x;iY[i]=y; 
                  } 
  for(i=0;i<m;++i){ 
   scanf("%d%d%d",&z,&x,&y); 
   oX[i]=x;oY[i]=y;R[i]=(ll)z*z; 
                  } 
  for(i=0;i<n;++i) 
   for(j=0;j<m;++j) 
    if((iX[i]-oX[j])*(iX[i]-oX[j]) + (iY[i]-oY[j])*(iY[i]-oY[j])<=R[j])mas[i].push_back(j); 
  while((Q--)>0){  
   scanf("%d%d",&x,&y);x--;y--; 
   L1=mas[x].size();L2=mas[y].size(); 
   int A=0;++st; 
   for(i=0;i<L1;++i)U[mas[x][i]]=st; 
   for(i=0;i<L2;++i)if(U[mas[y][i]]==st)++A; 
   A=L1+L2-(A<<1); 
   printf("%d\n",A); 
                } 
  return 0; 
}
