#include <bitset>
#include <iostream>
using namespace std;
const int N=1024;
long long n,m,q,u[N],v[N],r,a,b;
bitset<N> w[N];
main(){
  cin>>n>>m>>q;
  for(int i=1;i<=n;++i)cin>>u[i]>>v[i];
  for(int i=0;i<m&&cin>>r>>a>>b;++i){
    r*=r;
    for(int j=1;j<=n;++j)if((u[j]-a)*(u[j]-a)+(v[j]-b)*(v[j]-b)<r)w[j].set(i);
  }
  for(int i=0;i<q&&cin>>a>>b;++i)cout<<(w[a]^w[b]).count()<<endl;
}
