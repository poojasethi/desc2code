#include <stdio.h>
#include <string.h>

struct r1
{
    long long x,y;
};
r1 p[1001];
long long d;
int ans[1001][1001]={0};
int v[1001];
int n,m,k,ni,mi,ki;
int i,j;
long long r,y,x;

int main()
{
    scanf("%d%d%d",&n,&m,&k);
    for (ni=1;ni<=n;ni++)
        scanf("%I64d%I64d",&p[ni].x,&p[ni].y);
        
    for (mi=1;mi<=m;mi++)
    {
        scanf("%I64d%I64d%I64d",&r,&x,&y);
        memset(v,0,sizeof(v));
        r=r*r;
        for (i=1;i<=n;i++)
        {
            d=(p[i].x-x)*(p[i].x-x)+(p[i].y-y)*(p[i].y-y);
            if (d>=r) v[i]=1;
        }
        for (i=1;i<n;i++)
        for (j=i+1;j<=n;j++)
            if (v[i]!=v[j])
            {
                ans[i][j]++;
                ans[j][i]++;
            }
    }
    
    for (ki=1;ki<=k;ki++)
    {
        scanf("%d%d",&i,&j);
        printf("%d\n",ans[i][j]);
    }
    return 0;
}
