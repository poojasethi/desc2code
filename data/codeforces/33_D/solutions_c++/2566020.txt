#include <iostream>
#include <cmath>
using namespace std;
int n,m,Sol,circle[1010][1010],j,r,i,k;
struct coord 
{
    int x, y;
};

double lol(coord a, coord b) 
{
    return sqrt(1LL * (b.x-a.x) * (b.x-a.x) + 1LL * (b.y-a.y) * (b.y-a.y) );
}


int main() 
{
	coord A[1010],g;
    cin>>n>>m>>k;
    for(i=1;i<=n;++i)
        cin >>A[i].x>>A[i].y;
	for(i=1;i<=m;++i) 
	{
        cin>>r>>g.x>>g.y;
        for(j=1;j<=n;++j)
            if (lol(A[j],g) < r)
                circle[i][j]=1;
    }
	for(i=0;i<k;++i)		
	{
		int x,y;
		Sol=0;
        cin>>x>>y;
        for(j=1;j<=m;++j)
            if(circle[j][x]!=circle[j][y])
				Sol++;
		cout <<Sol<<"\n";
    }
    return 0;
}
