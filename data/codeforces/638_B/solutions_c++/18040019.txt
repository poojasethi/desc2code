#include<cstdio>
#include<cstring>
#include<map>

using namespace std;

int n;
char s[100];
int c[26],nxt[26],deg[26];

int main(){
    int i,j;
    scanf("%d",&n);
    memset(nxt,-1,sizeof(nxt));
    for(i = 0;i < n;i++){
        scanf("%s",s);
        for(j = 0;j < strlen(s);j++) c[s[j]-'a']=1;
        for(j = 0;j < strlen(s)-1;j++){
            nxt[s[j]-'a'] = s[j+1]-'a';
            deg[s[j+1]-'a']++;
        }
    }
    for(i = 0;i < 26;i++){
        if(c[i]&&!deg[i]){
            for(j = i;j != -1;j = nxt[j]) putchar('a'+j);
        }
    }
    return 0;
}
