#include <iostream>
#include <string.h>
#include <stdio.h>
#include <stdlib.h>
#include <vector>
#define maxn 500
using namespace std;
vector<int> Edge[30];
int vis[maxn];
string ans;
void dfs(int x)
{
    vis[x] = 2;

    ans+=char('a'+x);

    for(int i=0;i<Edge[x].size();i++)
    {
        if(vis[Edge[x][i]] ==2 )continue;
        dfs(Edge[x][i]);
    }
}
int main()
{
    int n,i,j;
    string str;
    while(cin>>n)
    {
        for(i=0;i<30;i++)
            Edge[i].clear();
        memset(vis,0,sizeof(vis));
        ans.clear();
        for(i=0;i<n;i++)
        {
            cin>>str;
            for(j=0;j<str.size()-1;j++)
            {
                Edge[str[j]-'a'].push_back(str[j+1]-'a');
                vis[str[j+1]-'a']=3;
            }

            if(vis[str[0]-'a']!=3)vis[str[0]-'a']=1;
        }

        for(i=0;i<26;i++)
        {
            if(vis[i] == 1)dfs(i);
        }
        cout<<ans<<endl;
    }
    return 0;
}
