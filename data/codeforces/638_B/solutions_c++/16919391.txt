#include <iostream>
#include <vector>
using namespace std;
int main()
{
	int n;
	cin >> n;
	string s;
	vector<int> seen(100, -2), next(100, -1);
	for(int i = 0; i < n; ++i)
    {
		cin >> s;
		if(seen[s[0]-'a'] == -2)   seen[s[0]-'a'] = -1;
		for(int k = 1; k < s.length(); ++k)
		{
			seen[s[k]-'a'] =  0;
			next[s[k-1]-'a'] =  s[k] - 'a';
		}
	}
	for(int i = 0; i < 26; i++)
	{
		if(seen[i] == -1)
		{
			int cur = i;
			while(cur != -1)
			{
                cout << (char)(cur+'a');
				cur = next[cur];
			}
		}
	}
}
