#include <bits/stdc++.h>
#define for2(i,a,b) for(int i=a,_b=(b);i<=_b;i++)
#define for1(i,a,b) for(int i=a,_b=(b);i<_b;i++)

using namespace std;
string s;
int f[256],i,n,j,nxt[256],prv[256];

int main()
 {
  cin>>n;
  for2(i,1,n)
   {
    cin>>s;
    for1(j,0,s.size())
     f[s[j]]=1;
    for1(j,1,s.size())
     nxt[s[j-1]]=s[j],prv[s[j]]=1;
   }

  for (i='a';i<='z';i++)
   if (f[i] && !prv[i])
    {
     j=i;
     while (j)
      {
       cout<<char(j);
       f[j]=0;
       j=nxt[j];
      }
    }
 }
