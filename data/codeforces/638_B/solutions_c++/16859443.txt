#include <stdio.h>
#include <vector>
using namespace std;
int u[100010];
vector <int> g[100010];
vector <int> order;
bool seemed[26];
bool bad[26];
void dfs(int v){
	u[v]=1;
	for(int i=0; i<g[v].size();i++)
		if(!u[g[v][i]]){
			dfs(g[v][i]);
		}
	order.push_back(v);
}

char s[100];

int main(){
//	freopen("cycle.in","r",stdin);
//	freopen("cycle.out","w",stdout);
	int n;
	scanf("%d",&n);	
	for (int i = 0;i < n;i++){
		scanf(" %s", s);
		for (int j = 0; s[j + 1] != 0; j++){
			g[s[j] - 'a'].push_back(s[j + 1] - 'a');
			bad[s[j + 1] - 'a'] = 1;
		}

		for (int j = 0; s[j] != 0; j++) {
			seemed[s[j]-'a'] = 1;
		}
	}
	for(int i = 0;i < 26;i++)
		if (seemed[i] && !bad[i] && !u[i])
			dfs(i);
	
	for (int i = order.size() - 1; i >= 0; i--)
		printf("%c", order[i] + 'a');



}
