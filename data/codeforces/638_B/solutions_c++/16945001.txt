#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<string>

using namespace std;
vector<int> pre;
vector<int> next;
vector<bool> used;
int n;
int g(char c){
	return c-'a';
}

int gopre(int u){
	if(pre[u]==-1) return u;
	else return gopre(pre[u]);
}

void gonext(int u){
	cout<<(char)('a'+u);
	used[u]=false;
	if(next[u]!=-1) gonext(next[u]);
}
int main(){
	cin>>n;
	pre.assign(30,-1);
	next.assign(30,-1);
	used.assign(30,false);
	for(int i=0;i<n;i++){
		string s;
		cin>>s;
		for(int j=0;j<(int)s.length()-1;j++){
			next[g(s[j])]=g(s[j+1]);
			pre[g(s[j+1])]=g(s[j]);
		}
		for(int j=0;j<(int)s.length();j++){
			used[g(s[j])]=true;
		}
	}

	for(int i=0;i<26;i++){
		if(used[i]){
			int p=gopre(i);
			gonext(p);
		}
	}


	return 0;
}
