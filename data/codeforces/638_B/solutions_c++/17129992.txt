#include<bits/stdc++.h>

using namespace std;

int main()
{
  vector<int> prev(26,-1);
  vector<int> next(26,-1);
  vector<bool> visited(26,false);
  int n,len,i;
  string str;
  
  cin>>n;
  
  while(n--)
  {
     cin>>str;
     len = str.length();
     
     //next[str[0] - 'a'] = str[1] - 'a';
     for(i=0; i < len-1; i++)
     {
        next[str[i] - 'a'] = str[i+1] - 'a';
        prev[str[i+1] - 'a'] = str[i] - 'a';
        visited[str[i+1] - 'a'] = true;
     }
     
     visited[str[0] - 'a'] = true; 
     
     //prev[str[len-1] - 'a'] = str[len-2] -'a';
  }
  
 /* for(i=0;i < 26; i++)
  {
     if(prev[i] == -1 && next[i] != -1)
         break;
  }*/
  
  //cout<<i;
  
  
  for(int j=0; j < 26; j++)
  {
     if(prev[j] == -1 && visited[j])
     {
        i = j;
         while(i != -1)
        {
           cout<<(char)(i+'a');
           i = next[i];
        }
     }
  }
  
  
  
  
  
   return 0;
}
