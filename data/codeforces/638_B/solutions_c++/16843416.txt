#include <bits/stdc++.h>

using namespace std;
int N, u[30], mer[30],aux;
string s[102],sir,mem,rs;
char S[30];
int main(){
	for(int i = 0; i < 26; i++){
		S[i] = '0';
			}
	cin >> N;
	for(int i = 1; i <= N; i++){
		cin >> s[i];
		}
	for(int i = 1; i <= N; i++){
		sir = s[i];
		if(sir.size() == 1){mer[int(sir[0])-97] = 1;}
		for(unsigned int j = 0; j < sir.size()-1; j++){
			S[int(sir[j])-97] = sir[j+1]; u[int(sir[j+1])-97] = 1;
			if(mer[int(sir[j])-97] == 1) mer[int(sir[j])-97] = 0;
			}
		}
	int j;
	for(int i = 0; i < 26; i++){
		if((u[i] == 0 && S[i] != '0') || (mer[i] && S[i] == '0' && u[i] == 0)){
			rs += char(i+97);
			u[i] = 1;
			j = i;
			while(u[j] && S[j] != '0'){
					rs += S[j];
					j = int(S[j]) - 97;
				}
			}
	} 
	cout << rs;
	return 0;
}
