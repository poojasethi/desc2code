#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

int n;

char s[105][105];

void append(int fr,int to,int pos){
  int lf=strlen(s[fr]);
  int lt=strlen(s[to]);
  int las=max(lt-1-pos,lf-1);
  if(lf>lt-pos) FI(j,lt,lf+pos-1) s[to][j]=s[fr][j-pos];
  s[fr][0]=0;
  return;
}

int main(){
  scanf("%d",&n);
  FI(i,1,n) scanf(" %s",s[i]);
  FI(i,1,n){
    bool done=false;
    FI(j,1,n){
      if(done) break;
      if(i==j) continue;
      //append i to j
      int len=strlen(s[j]);
      FI(k,0,len-1) if(s[j][k]==s[i][0]){
        append(i,j,k);
        done=true;
      }
    }
  }
  FI(i,1,n) if(strlen(s[i])) printf("%s",s[i]);
  printf("\n");
  return 0;
}
