#include<iostream>
#include<string>
#include<vector>
#include<stdio.h>

using namespace std;

vector<int> p;
vector<int> q;
vector<int> b;

void dfs(int v){
    printf("%c", v+'a');
    if (q[v] != -1){
        dfs(q[v]);
    }
}

int main(){
    int n;
    int K = 26;

    q.resize(K,-1);
    p.resize(K,-1);
    b.resize(K);

    cin>>n;

    for (int i = 0; i < n; i++){
        string s;
        cin>>s;
        b[s[0]-'a'] = 1;;
        for (int j = 1; j < s.size(); j++){
            int s1 = s[j-1] - 'a';
            int s2 = s[j] - 'a';
            b[s2] = 1;
            q[s1] = s2;
            p[s2] = s1;
        }
    }

    for (int i = 0; i < K; i++){
        if (b[i] && p[i] == -1){
            dfs(i);
        }
    }
}
