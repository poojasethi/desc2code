#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>

using namespace std;

int n;
int g[30][30];
bool met[30], root[30];
char s[30];

void dfs(int v) {
	printf("%c", v + 'a');

	for (int i = 0; i < 26; i++) {
		if (g[v][i]) 
			dfs(i);
	}
}

int main() {
	scanf("%d\n", &n);

	for (int i = 0; i < n; i++) {
		gets(s);

		int len = strlen(s);

		for(int j = 0; j < len - 1; j++) {
			g[s[j] - 'a'][s[j + 1] - 'a'] = 1;
			met[s[j] - 'a'] = true;
			root[s[j + 1] - 'a'] = true;
		}

		met[s[len - 1] - 'a'] = true;
	}

	for (int i = 0; i < 26; i++) {
		if (met[i] && !root[i]) {
			dfs(i);
		}
	}

	return 0;
}
