#include <iostream>
#include <string.h>
#include <stdio.h>
#include <algorithm>
#include <math.h>
using namespace std;
const int N = 1000;
char s[N];
int n;
int next[30], pre[30];
int main()
{
	scanf("%d", &n);
	memset(next, -1, sizeof(next));
	for (int i = 0; i < 26; i++)
	  pre[i] = -2;
	for (int i = 1; i <= n; i++)
	{
		scanf("%s", s);
		int l = strlen(s);
		if (pre[s[0]-'a'] == -2)
		  pre[s[0]-'a'] = -1;
		for (int j = 1; j < l; j++)
		{
			next[s[j-1]-'a'] = s[j]-'a';
			pre[s[j]-'a'] = 0;
		}
	}
	for (int i = 0; i < 26; i++)
	  if (pre[i] == -1)
	  {
		  int j = i;
		  while (j != -1)
		  {
			  cout<<(char)('a'+j);
		  	  j = next[j];
		  }
		  //cout<<(char)('a'+j);
	  }
	cout<<endl;
	return 0;
}
