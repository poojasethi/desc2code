#include <bits/stdc++.h>
using namespace std;

string s;


char nx[256],u[256];

int main()
{
    int a,n,i;
    cin>>n;


    for(char c = 'a'; c<='z';c++){nx[c]=0;u[c]=0;}

    for(i=0;i<n;i++)
    {
        cin>>s;
        if(u[s[0]]==0)u[s[0]]=1;
        for(int j=0;j<s.length()-1;j++)
        {
            nx[s[j]]=s[j+1];
            u[s[j+1]]=-1;
        }

    }

    for(char c = 'a'; c<='z';c++)
    {
        if(u[c]==1)
        {
            cout<<c;

            char cc=c;
            while(nx[cc]!=0)
            {
                cc=nx[cc];
                cout<<cc;
            }

        }
    }

    return 0;
}
