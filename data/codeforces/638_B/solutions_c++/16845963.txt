#include <cstdio>
#include <cstring>

int main() {
	int ar[26];
	bool exist[26];
	bool fr[26];
	memset(ar, -1, 26 * sizeof(int));
	memset(exist, false, 26 * sizeof(bool));
	memset(fr, true, 26 * sizeof(bool));
	int n;
	scanf("%d",&n);
	getchar();
	char c;
	char prev;
	for (int i = 0; i < n; ++i) {
		prev = getchar() - 'a';
		exist[prev] = true;
		while ((c = getchar()) != '\n') {
			c -= 'a';
			exist[c] = true;
			fr[c] = false;
			ar[prev] = c;
			prev = c;
		}
	}
	for (int i = 0; i < 26; ++i) {
		if (exist[i] && fr[i]) {
			printf("%c",i + 'a');
			int cur = i;
			while (ar[cur] != -1) {
				printf("%c",ar[cur] + 'a');
				cur = ar[cur];
			}
		}
	}
	printf("\n");
}
