#include<iostream>
using namespace std;
int isNotPre[300];
int next[300];
int isExist[300];
int main(){
    int n;
    cin>>n;
    while(n--){
        string val;
        cin>>val;
        for(int i = 0; i < val.size(); i++){
            isExist[val[i]] = 1;
        }
        for(int i = 1; i < val.size(); i++){
            next[val[i-1]] = val[i];
            isNotPre[val[i]] = 1;
        }
    }
    for(char i = 'a'; i <='z'; i++){
        if(isExist[i]&&!isNotPre[i]){
            cout<<i;
            char j = next[i];
            while(j){
                cout<<j;
                j = next[j];
            }
        }
    }
    return 0;
}
