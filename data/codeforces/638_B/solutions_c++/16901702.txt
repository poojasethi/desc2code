typedef long long LL;
const LL INF=1000000009;

int nxt[1000], prev[1000];
bool genom[1000];
int main(){
	int n;
	cin>>n;

	for(int i=1;i<=n;i++){
		string s;
		cin>>s;
		fori(j,s.size())genom[s[j]]=true;
		for(int j=1;j<s.size();j++){
			nxt[s[j-1]]=s[j];
			prev[s[j]]=1;		
		}
	}
	for(int i='a';i<='z';i++){
		if(genom[i] && !prev[i]){
			int j=i;
			while(j){cout<<char(j);genom[j]=false;j=nxt[j];}
		}
	}	

	return 0;
}
