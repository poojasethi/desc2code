#include <iostream>
#include <string.h>
#include <algorithm>
#include <stdio.h>
#include <vector>
#include <queue>
using namespace std;
int mp[30][30];
int in[30],vis[30];
void dfs(int now){
    printf("%c",now+'a');
    vis[now]=0;
    for(int i=0;i<26;i++){
        if(mp[now][i]&&vis[i])
            dfs(i);
    }
}
int main()
{
  //  freopen("dd.txt","r",stdin);
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        char str[30];
        scanf("%s",str);
        int len=strlen(str);
        if(len==1){
            vis[str[0]-'a']=1;
        }
        for(int j=0;j<len-1;j++){
            int u=str[j]-'a',v=str[j+1]-'a';
            in[v]++;
            mp[u][v]++;
            vis[u]=vis[v]=1;
        }
    }
    queue<int> q;
    for(int i=0;i<26;i++){
        if(vis[i]&&in[i]==0)
            q.push(i);
    }
    while(!q.empty()){
        int now=q.front();
        q.pop();
        dfs(now);
    }
    printf("\n");
    return 0;
}
