#include <iostream>
#include <string>
#include <vector>
using namespace std;

int u[27];
vector<int> g[27];
vector<char> res;
bool book[27];
bool bad[26];

void dfs(int v) {
	u[v] = 1;
	for (int i = 0; i < g[v].size(); i++)
		if (!u[g[v][i]]) 
			dfs(g[v][i]);
	res.push_back(v + 'a');
}

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++){
		string s;
		cin >> s;
		for (int j = 0; j < s.size(); j++)
			book[s[j] - 'a'] = 1;
		for (int j = 0; j < s.size() - 1; j++){
			g[s[j] - 'a'].push_back(s[j + 1] - 'a');
			bad[s[j + 1] - 'a'] = 1;
		}
	}
	for (int i = 0; i < 26; i++)
		if (!u[i] && book[i] && !bad[i])
			dfs(i);
	for (int i = res.size() - 1; i >= 0; i--)
		cout << res[i];
}
