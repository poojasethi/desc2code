#include <bits/stdc++.h>

using namespace std;

bool start[256];
bool contains[256];
int next[256];

int main()
{
    ios::sync_with_stdio(false);

    int n;
    cin >> n;

    memset(start, true, sizeof(bool) * 256);

    while (n--)
    {
        string s;
        cin >> s;

        for (int i = 0; i < s.length(); i++)
        {
            contains[s[i]] = true;
        }

        for (int i = 0; i < s.length() - 1; i++)
        {
            start[s[i + 1]] = false;
            next[s[i]] = s[i + 1];
        }
    }

    for (int a = 'a'; a <= 'z'; a++)
    {
        if (contains[a] && start[a])
        {
            char c = a;
            while (c)
            {
                cout << c;
                c = next[c];
            }
        }
    }

    cout << endl;
}
