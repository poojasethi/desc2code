#include <stdio.h>
#include <string.h>
#include <vector>
using namespace std;
int dp[10000];
vector<int > SS[100];
bool vis[100];
void dfs(int x) {
    printf("%c",x+'a'-1);
    if(SS[x].size()) {
        dfs(SS[x][0]);
    }
}
char a[1000];
int main(){
    int n;
    while(scanf("%d",&n)!=EOF) {
        memset(dp,-1,sizeof(dp));
        for(int i = 0;i<n;i++) {
            scanf("%s",a);
            int len = strlen(a);
            vis[a[0]-'a'+1] = true;
            if(len>1) SS[a[0]-'a'+1].push_back(a[1]-'a'+1);
            for(int j = 1;j<len;j++) {
                vis[a[j]-'a'+1] = true;
                dp[a[j]-'a'+1] = a[j-1]-'a'+1;
                if(j<len-1)
                    SS[a[j]-'a'+1].push_back(a[j+1]-'a'+1);
            }
        }
        for(int i = 1;i<27;i++) {
            if(vis[i]) {
                if(dp[i] == -1) {
                    dfs(i);
                }
            }
        }
        puts("");
    }
}
