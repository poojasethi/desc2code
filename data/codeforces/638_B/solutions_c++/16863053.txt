#include<bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	while(cin >> n)
	{
		vector<bool> used(26, false);
		vector<int> next(26, -1);
		vector<int> prev(26,-1);
		for(int i = 0; i < n; i++)
		{
			string s;
			cin >> s;
			for(int j = 0; j < s.size(); j++)
			{
				used[s[j] - 'a'] = true;
				if(j + 1 < s.size())
				{
					next[s[j] - 'a'] = s[j + 1] - 'a';
				}
				if(j > 0)
				{
					prev[s[j] - 'a'] = s[j - 1] - 'a';
				}
			}
		}
		string ans = "";
		for(int i = 0 ; i < 26; i++)
		{
			if(used[i] && prev[i] == -1)
			{
				int pos = i;
				while(pos != -1)
				{
					ans += char(pos + 'a');
					pos = next[pos];	
				}	
			}	
		}
		cout << ans << endl;
	}
}
