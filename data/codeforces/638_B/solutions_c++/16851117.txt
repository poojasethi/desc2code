#include <bits/stdc++.h>

#define mp make_pair
#define pb push_back
#define fi first
#define se second

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;
typedef unsigned long long ull;
typedef long double ld;
typedef pair<ll, ll> pll;

const int N = 1e6 + 7;
const ll INF = 1e9 + 7;
const ld EPS = 1e-9;
const ll P = 485437;
const int C = 2016;

string s;
char nxt[N];
bool e[N], p[N], used[N];

void dfs(char c){
    s += c;
    if (nxt[c])
        dfs(nxt[c]);
}

int main()
{
    ios_base::sync_with_stdio(0);
    int n;
    cin >> n;
    for (int i = 'a'; i <= 'z'; i++)
        p[i] = false;
    for (int i = 0; i < n; i++){
        cin >> s;
        for (int j = 0; j < s.size(); j++){
            e[s[j]] = true;
            if (j != 0)
                nxt[s[j - 1]] = s[j], p[s[j]] = true;
        }
    }
    s = "";
    for (char i = 'a'; i <= 'z'; i++)
        if (e[i] && !p[i])
            dfs(i);
    cout << s;
    return 0;
}
