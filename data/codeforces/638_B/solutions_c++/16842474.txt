#include <stdio.h>
#include <iostream>
#include <map>
#include <memory.h>
#include <math.h>
#include <algorithm>
#include <vector>
#include <string>
#define N 311111                                                                                             
#define INF 2000000000
#define P 11111117
#define LL long long

using namespace std;

int n;

string s[N];

int p[N],was[N],f[N];


int main(void){
  //  freopen(".in","r",stdin);
    scanf("%d\n",&n);
    for (int i=0;i<n;i++) getline(cin,s[i]);
	memset(f,0,sizeof(f));
	memset(p,0,sizeof(p));
	memset(was,0,sizeof(was));

	for (int i=0;i<n;i++){
		for (int j=0;j<s[i].length()-1;j++) p[s[i][j]] = s[i][j+1];
		for (int j=0;j<s[i].length();j++) was[s[i][j]] = 1;
		for (int j=1;j<s[i].length();j++) f[s[i][j]] = 1;
	}
	for (char c='a';c<='z';c++) if (was[c]==1 && f[c]==0){
		cout << c;
		was[c] = 2;
		char ps = c;
		while (p[ps]>0){
			ps = p[ps];
			was[ps] = 2;
			cout << ps;
		}
	}

	return 0;
}
