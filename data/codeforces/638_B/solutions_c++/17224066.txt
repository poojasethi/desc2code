#define eps 1e-8
char str[105];
int in[33];
int to[33];
bool has[33];
int main()
{
    memset(to,-1,sizeof(to));
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%s",str);
        int len=strlen(str);
        for(int i=0;i<len;i++)
        {
            has[str[i]-'a']=1;
            if(i)
            {
                in[str[i]-'a']++;
                to[str[i-1]-'a']=str[i]-'a';
            }
        }
    }
    for(int i=0;i<26;i++)
    {
        if(has[i]&&in[i]==0)
        {
            int t=i;
            while(t!=-1)
            {
                printf("%c",'a'+t);
                t=to[t];
            }
        }
    }
    printf("\n");
    return 0;
}
