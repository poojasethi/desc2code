#include<iostream>
#include<cstring>
using namespace std;
int pre[26],suc[26];
bool exist[26],confirmed[26];
#define INT(x) (x-'a')
#define CHAR(x) ((char)(x+'a'))
#define Init(x) memset(x,0,sizeof(x))
int main(){
	Init(exist);Init(confirmed);//Init(pre);Init(suc)
	for(int i=0;i<26;i++){pre[i]=26;suc[i]=26;}
	int clues;cin>>clues;
	if(clues==1){
		string temp;
		cin>>temp;cout<<temp;return 0;
	}
	for(int i=0;i<clues;i++){
		string temp;cin>>temp;
		exist[INT(temp[0])]=true;
		for(int i=1;i<temp.length();i++){
			exist[INT(temp[i])]=true;
			pre[INT(temp[i])]=INT(temp[i-1]);
			//cout<<"pre["<<temp[i]<<"]="<<temp[i-1]<<endl;
			suc[INT(temp[i-1])]=INT(temp[i]);
		}
	}
	for(int i=0;i<clues;i++){
		int start=26;
		for(int i=0;i<26;i++)
			if(exist[i]&&pre[i]==26&&!confirmed[i]){
				start=i;break;
			}
		if(start==26)break;
		#define current start
		do{
			cout<<CHAR(current);
			confirmed[current]=true;
			current=suc[current];
		}while(current!=26);
	}
}
