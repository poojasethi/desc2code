#include <cstdio>
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;
int Next[300];
int	pre[300];
int vis[300];
int main()
{
	int n;
	string str;
	memset(Next, 0, sizeof Next);
	memset(pre, 0, sizeof pre);
	memset(vis, 0, sizeof vis);
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		cin >> str;
		for (int k = 0; k < str.size() - 1; k++)
		{
			Next[str[k]] = str[k + 1];
			pre[str[k + 1]] = str[k];
			vis[str[k + 1]] = 1;
		}
		vis[str[0]] = 1;
	}

	for (int i = 'a'; i <= 'z'; i++)
	{
		if (pre[i] == 0 && vis[i])
		{
			int cur = i;
			while (cur != 0)
			{
				printf("%c", cur);
				cur = Next[cur];
			}
		}
	}
	printf("\n");
	return 0;
}
