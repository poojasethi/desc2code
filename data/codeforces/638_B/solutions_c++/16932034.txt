#include <iostream>
#include <set>
#include <algorithm>
#include <map>
#include <cstdio>

using namespace std;


string s[1000];

int main()
{
    int n;
    cin >>n;
    for(int i=0;i<n;i++)
        cin >> s[i];
    string res="";
    for(int i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        for(int k1=0;k1<s[i].size();k1++)
        for(int k2=0;k2<s[j].size();k2++)
        {
            if(s[i][k1]==s[j][k2])
            {
                string ss = "";
                if(k1<k2) ss = s[j].substr(0,k2); else ss=s[i].substr(0,k1);
                if(s[i].size() - k1 <s[j].size()-k2)
                    ss+= s[j].substr(k2); else ss+=s[i].substr(k1);
                s[j]=ss;
                s[i]="";
                break;
            }

        }
        res+=s[i];

    }/*
    sort(s,s+n,cmp);
    for(int i=0;i<n;i++)
        res+=s[i];
        */
    cout <<res<<endl;

    return 0;
}
