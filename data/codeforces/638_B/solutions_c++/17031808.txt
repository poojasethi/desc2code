#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
#define ll long long
char str[27];
int par[27],child[27];
int sel[27];
int present[27];
void dfs(int node)
{
	sel[node]=1;
	printf("%c",node+'a');
	if(child[node]!=node)
	{
		dfs(child[node]);
	}
}
int main()
{
	int i,j,k;
	int n;
	sd(n);
	for(i=0;i<26;i++)
	{
		par[i]=i;
		child[i]=i;
		sel[i]=0;
		present[i]=0;
	}
	for(i=0;i<n;i++)
	{
		ss(str);
		j=strlen(str);
		present[(int)(str[0]-'a')]=1;
		for(k=1;k<j;k++)
		{
			par[(int)(str[k]-'a')]=(int)(str[k-1]-'a');
			child[(int)(str[k-1]-'a')]=(int)(str[k]-'a');
			present[(int)(str[k]-'a')]=1;
		}
	}
	for(i=0;i<26;i++)
	{
		if(sel[i]==0&&par[i]==i&&present[i]==1)
		{
			//sel[i]=1;
			dfs(i);
		}
	}
	return 0;
}
