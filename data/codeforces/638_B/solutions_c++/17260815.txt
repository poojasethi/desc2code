#include <bits/stdc++.h>

using namespace std;

long long n;
char res[1000];
bool vis[1000] , vi[1000] , v[1000] , tu[1000];

int main()
{
    //freopen("input.txt" , "r" , stdin);
    //freopen("output.txt" , "w" , stdout);
    cin >> n;
    for (int i = 0 ; i < n ; i++)
    {
        string s;
        cin >> s;
        if (s.size() == 1) vis[s[0]] = true; else{
        for (int i = 0 ; i < s.size() - 1 ; i++)
        {
            res[s[i]] = s[i+1];
            vis[s[i+1]] = true;
            vi[s[i+1]] = true;
            vis[s[i]] = true;
            v[s[i]] = true;
        }}
    }
    char k;
    string ss = "";
    for (int i = 97 ; i <= 122 ; i++)
    {
        if (vis[i] && !vi[i] && !tu[i])
        {
            k = i;

    tu[k] = true;
    ss = ss + k;
    while (vis[k] && v[k])
    {
        tu[k] = true;
        k = res[k];
        ss = ss  + k;
        tu[k] = true;
    }}
    }
    cout << ss;
    return 0;
}
