#include<bits/stdc++.h>
#include<stdio.h>
using namespace std;

int n;
map<char,char>MAP;
bool have[30];
bool wei[30];
string ans;

int main()
{
    cin>>n;
    string a;
    for(int i=0;i<n;i++)
    {
        cin>>a;
        int sz=a.size();
        for(int i=0;i<sz-1;i++)
        {
            MAP[a[i]]=a[i+1];
            have[a[i]-'a']=true;
            wei[a[i+1]-'a']=true;
        }
        have[a[sz-1]-'a']=true;
    }
    for(int i=0;i<26;i++)
        if(have[i]&&!wei[i])
        {
            char sb=i+'a';
            ans.push_back(sb);
            while(MAP.count(sb))
            {
                sb=MAP[sb];
                ans.push_back(sb);
            }
        }
    cout<<ans<<endl;
    return 0;
}
