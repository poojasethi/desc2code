#include <iostream>

using namespace std;

long long n;
string res,s;
char mas[3][255];

int main()
{
    cin >> n;
    for (int i=0;i<n;i++)
    {
        cin >> s;
        mas[0][s[0]]=1;
        for (int j=1;j<s.length();j++)
        {
            mas[1][s[j]]=1;
            mas[2][s[j-1]]=s[j];
        }
    }
    
    for (int i=97;i<123;i++)
    {
        if (mas[0][i]==1 && mas[1][i]==0)
        {
            for (char c=i;c!=0;res+=c,c=mas[2][c]);
        }
    }
    cout << res << "\n";
    return 0;
}
