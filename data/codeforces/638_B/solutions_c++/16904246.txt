#include<cstdio>
#include<cstring>
int next[100];
int prev[100];
char s[100];
int num[100];
int main(void){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%s",s);
		for(int j=0;j<strlen(s);j++)
		num[s[j]-'a'+1]=1;
		for(int j=1;j<strlen(s);j++){
			next[s[j-1]-'a'+1]=s[j]-'a'+1;
			prev[s[j]-'a'+1]=1;
		}
	}
	int k;
	for(int i=0;i<50;i++){
		if(num[i]==1&&prev[i]==0){
			k=i;
			while(k!=0){
				printf("%c",k+'a'-1);
				num[k]=0;
				k=next[k];
			}
		}
	}
}
