#include <bits/stdc++.h>
#define ll long long
using namespace std;
string s;
bool occur[26],done[26];//indegree
ll int in[26];//indegree
list<ll int> l[26];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	memset(occur,0,sizeof(occur));
	memset(done,0,sizeof(done));
	memset(in,0,sizeof(in));
	ll int n;
	cin>>n;
	ll int i,j,len,prev;
	for(i=0;i<n;i++)
	{
		cin>>s;
		len=s.length();
		prev=-1;
		for(j=0;j<len;j++)
		{
			occur[s.at(j)-'a']=1;
			if(prev!=-1)
			{
				l[prev].push_back(s.at(j)-'a');
				in[s.at(j)-'a']++;
			}
			prev=s.at(j)-'a';
		}
	}
	stack<int> s;
	string res="";
	for(i=0;i<26;i++)
	{
		if(occur[i])
		{
			if(in[i]==0)
			s.push(i);
		}
	}
	while(!s.empty())
	{
		i=s.top();
		s.pop();
		if(!done[i])
		{

			done[i]=1;

			res+='a'+i;
			list<ll int>::iterator it;
			for(it=l[i].begin();it!=l[i].end();it++)
			{
				s.push(*it);
			}
		}
	}
	cout<<res<<endl;
	return 0;
}
