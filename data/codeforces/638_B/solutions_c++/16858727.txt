char nxt[Sz] , prv[Sz];
bool have[Sz];           
int n; 

int main() 
{
	rcin ();          
	cin >> n;
	for (int i = 1;i <= n;i ++) {
		string s;	cin >> s;
		for (int j = 0;j < s.size();j ++) {
			have[s[j]] = 1;
			if (j) {
				prv[s[j]] = s[j - 1];
		     	nxt[s[j - 1]] = s[j];
		   }
		}
	}
	for (char x = 'a';x <= 'z';x ++) {
		if (have[x] && !prv[x]) {
			char y = x;
			while (y) {
				cout << y;
				y = nxt[y];
			}
		}
	}	

	return 0;
}
