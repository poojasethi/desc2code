#include <iostream>
#include <cstdio>
#include <vector>
#include <string>

using namespace std;

const int N = 26;

bool used[N], start[N];
int to[N];

int main() {
    // freopen("input.txt", "r", stdin);
    for (int i = 0; i < N; i++) {
        start[i] = 1;
        to[i] = -1;
    }
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
        string s;
        cin >> s;
        int m = s.size();
        vector<int> v(m);
        for (int j = 0; j < m; j++)
            v[j] = s[j] - 'a';
        for (int j = 0; j < m; j++)
            used[v[j]] = 1;
        for (int j = 0; j < m - 1; j++) {
            to[v[j]] = v[j + 1];
            start[v[j + 1]] = 0;
        }
    }
    for (int i = 0; i < N; i++)
        if (used[i] && start[i]) {
            int now = i;
            while (now != -1) {
                cout << (char)('a' + now);
                now = to[now];
            }
        }
    cout << endl;
    return 0;
}
