#include<stdio.h>
#include<string.h>
char str[30]={'\0'};
bool ex[28]={false},traved[28]={0};
int rgt[28]={0},lef[28]={0},i,j,len,n;
void trav(int k)
{
   if(k==0) return;
   traved[i]=true;
   printf("%c",k+96);
   trav(rgt[k]);
}
int main()
{
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  {
     scanf("%s",str+1);
     len=strlen(str+1);
     for(j=1;j<=len;j++)
     {
       //printf("%d",j);
       ex[str[j]-96]=true;
       if(j<len)
       {
       rgt[str[j]-96]=str[j+1]-96;
       lef[str[j+1]-96]=str[j]-96;
       }
     }
     //printf("%d",i);
  }
  for(i=1;i<=26;i++)
  {
     if(ex[i]==true&&traved[i]==false&&lef[i]==0)
     {
       trav(i);
     }
  }
return 0;
}
