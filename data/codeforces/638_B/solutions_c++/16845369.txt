#include <iostream>
#include <vector>

using namespace std;

int main()
{
    vector<int> prev(26, -1);
    vector<bool> used(26, false);
    int n;
    cin >> n;
    for (int i = 0; i < n; ++i) {
        string s;
        cin >> s;
        for (int j = 0; j < s.length(); ++j) {
            used[s[j] - 'a'] = true;
            if (j > 0)
                prev[s[j] - 'a'] = s[j - 1] - 'a';
        }
    }
    int last = -1;
    bool can = true;
    while (can) {
        int lastlast = last;
        for (int i = 0; i < 26; ++i)
            if (used[i] && (prev[i] == last)) {
                used[i] = false;
                last = i;
                cout << (char)('a' + i);
            }
        if (lastlast == last) {
            if (last == -1)
                can = false;
            else
                last = -1;
            }
    }
    return 0;
}
