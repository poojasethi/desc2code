import sys


def main():
	g = {}

	n = int(raw_input())
	for i in xrange(n):
		part = raw_input()
		for j in xrange(len(part) - 1):
			g[part[j]] = part[j + 1]

		g.setdefault(part[-1], None)


	answer = ""
	starts = [ch for ch in g.iterkeys() if ch not in g.itervalues()]
	for start in starts:
		cur = start
		answer += cur
		while cur in g and g[cur]:
			cur = g[cur]
			answer += cur

	print answer

if __name__ == '__main__':
	main()
