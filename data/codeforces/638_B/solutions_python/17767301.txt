nx=[-1]*26
pr=[-1]*26
vis=[False]*26
n=int(input())
for _ in range(n):
   s=raw_input()
   for i in range(len(s)-1):
      u=ord(s[i])-ord('a')
      v=ord(s[i+1])-ord('a')
      nx[u]=v
      pr[v]=u
   for i in s: vis[ord(i)-ord('a')]=True
ans=""
for u in range(26):
   if vis[u] and pr[u]==-1:
      v=u
      while v!=-1:
         ans+=chr(v+ord('a'))
         v=nx[v]
print ans
