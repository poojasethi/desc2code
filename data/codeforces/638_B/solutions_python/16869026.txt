import sys
data = sys.stdin.read().split()
n = int(data.pop(0))
assert n == len(data)
genom = []
adj = {}
for chunk in data:
    end = len(chunk) - 1
    genom.append(chunk[0])
    for i, c in enumerate(chunk):
        adj.setdefault(c, None)
        next = adj[c]
        if i < end:
            if next:
                assert next == chunk[i + 1]
            else:
                adj[c] = chunk[i + 1]
ans = ''
genom = set(genom) - set(adj.values())
for c in genom:
    next = c
    while next:
        ans += next
        next = adj.pop(next)
sys.stdout.write(ans)
