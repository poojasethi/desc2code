import sys
def diff(s1,s2):
    cnt=0
    for i in range(4): 
        if s1[i]!=s2[i]: cnt+=1
    return cnt

n=int(sys.stdin.readline())
dates=[]
res=[]
for i in range(n): dates.append(sys.stdin.readline().strip())
for i in range(n):
    for s in range(1000,2012):
        if diff(str(s), dates[i])<=1 and (i==0 or (s>=int(res[i-1]))):
            res.append(str(s))
            break
    if len(res)<i+1:
        print 'No solution'
        exit()
