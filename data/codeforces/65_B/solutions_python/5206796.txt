n=input()
r=[]
p=[0,0,0,0]
for i in range(n):
    y=map(int,raw_input())
    m=[]
    for j in range(4):
        for k in range(j==0,10):
            x=y[:]
            x[j]=k
            if x >= p:
                m+=[x]

    if len(m):
        p=min(m)[:]
        r+=[''.join(map(str,p[:]))]

print "No solution" if len(r) != n or r[-1] > '2011' else '\n'.join(r)
