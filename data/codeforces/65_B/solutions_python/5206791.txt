n=input()
r=[]
p=[0,0,0,0]
for i in range(n):
    y=map(int,raw_input())
    m=[]
    for j in range(4):
        for k in range(j==0,10):
            x=y[:]
            x[j]=k
            if x >= p:
                #print " ",x
                m+=[x]
    #print m
    if not len(m): print "No solution"; exit()
    p=min(m)[:]
    #print p
    r+=[''.join(map(str,p[:]))]
#print r
print "No solution" if len(r) != n or r[-1] > '2011' else '\n'.join(r)
