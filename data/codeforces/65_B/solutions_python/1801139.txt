def diff(a,b):
    i,d=0,0
    while i<4 and d<2:
        d+=1 if a%10!=b%10 else 0
        a/=10
        b/=10
        i+=1
    return d
def next(a,b):
    while b<=2011:
        if diff(a,b)<2:
            return b
        b+=1
    return 9999
a=[1000]+[input() for _ in range(input())]
for i in xrange(1,len(a)): a[i]=next(a[i],a[i-1])
print "No solution" if a[-1]>2011 else '\n'.join(map(str,a[1:]))
