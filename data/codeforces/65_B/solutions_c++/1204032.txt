#include<stdio.h>
#include<string.h>
int a[1005];
int check(int x,int y){
	int ans=0;
	while(x){
		if(x%10!=y%10)ans++;
		x/=10;
		y/=10;
	}
	if(ans<=1)return 1;
	else return 0;
}
int main(){
	int n,i,j,k;
	while(scanf("%d",&n)!=EOF){
		for(i=1;i<=n;i++){
			scanf("%d",&a[i]);
		}
		a[0]=1000;
		for(i=1;i<=n;i++){
			k=a[i];
			a[i]=-1;
			for(j=a[i-1];j<=2011;j++){
				if(check(j,k)){a[i]=j;break;}
			}
			if(a[i]==-1)break;
		}
		if(a[i]==-1)printf("No solution\n");
		else{
			for(i=1;i<=n;i++)
			printf("%d\n",a[i]);
		}
	}
	return 0;
}
