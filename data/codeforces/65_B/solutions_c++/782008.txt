#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
int a[1010];

bool solve(int &a,int &b)
{
	int i,j,k,mi=3000;
	char c[10];	
	for(i=0;i<4;i++)
	{
		sprintf(c,"%d",a);
		for(j='0';j<='9';j++)
		{
			c[i]=j;
			sscanf(c,"%d",&k);
			if(k>=b&&k<=2011&&k<mi)  {mi=k;break;}
		}
	}
	if(mi>=b&&mi<=2011)  {a=b=mi;return 0;}
	return 1;
}

int main()
{
	int i,n,mi;
	while(scanf("%d",&n)!=EOF)
	{
		mi=1000;
		for(i=0;i<n;i++)
			scanf("%d",a+i);
		for(i=0;i<n;i++)
		{
			if(solve(a[i],mi)) break;
		}
		if(i!=n) {printf("No solution\n");continue;}
		for(i=0;i<n;i++)
			printf("%d\n",a[i]);
	}
	return 0;
}
