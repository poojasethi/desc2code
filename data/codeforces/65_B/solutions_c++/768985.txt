#include<iostream>
#include<cstdio>
#include<vector>

using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	string ant="1000",s;
	vector<string> R;
	for(int i=0;i<n;i++)
	{
		cin>>s;
		string tt="2012",t;
		for(int i=0;i<4;i++)
			for(int j=0;j<10;j++)
			{
				t=s;
				t[i]='0'+j;
				if(t>=ant)
					tt=min(t,tt);
			}
		if(tt>"2011")
		{
			cout<<"No solution"<<endl;
			return 0;
		}
		ant=tt;
		R.push_back(tt);
	}
	for(int i=0;i<n;i++)
		cout<<R[i]<<endl;
}
