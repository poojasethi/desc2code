#include <stdio.h>
#include <algorithm>
using namespace std;

int a[1005];

int change(int src, int big) {
	int now = 9999;
	for(int t=1,i=0;i<4;i++,t*=10) {
		for(int j=(i==3);j<10;j++) {
			int v = ((src/t/10*10+j)*t+src%t);
			if(v >= big && v < now) now = v;
		}
	}
	return now;
}

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=1;i<=n;i++)
		scanf("%d", &a[i]);
	for(int i=1;i<=n;i++) {
		a[i] = change(a[i], a[i-1]);
	}
	if(a[n] > 2011) puts("No solution");
	else {
		for(int i=1;i<=n;i++)
			printf("%d\n", a[i]);
	}
	return 0;
}
