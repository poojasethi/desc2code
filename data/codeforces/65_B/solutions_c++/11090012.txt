#include<iostream>
using namespace std;
int n, a=1000,b,cntn;
int ans[1010];
bool check(int a,int b)
{
	int cnt=0;
	while(a)
	{
		if(a%10!=b%10)	cnt++;
		a/=10;
		b/=10;
	}
	return cnt<=1;
}
int main()
{
	cin>>n;
	for(int t=1;t<=n;t++)
	{
		cin>>b;
		for(int i=a;i<=2011;i++)
			if(check(i,b))
			{
				a=ans[++cntn]=i;
				break;
			}
	}
	if(cntn==n)
		for(int i=1;i<=cntn;i++)
			cout<<ans[i]<<endl;
	else cout<<"No solution"<<endl;
	return 0;
}
