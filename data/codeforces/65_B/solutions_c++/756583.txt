#include<iostream>

using namespace std;

int main()
{
	int n;
	cin>>n;
	string date[n+1];
	for(int i=0 ; i<n ; i++)
	{
		cin>>date[i];
	}
	date[n]="2011";
	int res=0;
	for(int i=n-1 ; i>=0 ; i--)
	{
		string r="1000";
		res=0;
		for(int j=0 ; j<4 ; j++)
		{
			char tmp = date[i][j];
			for(int p=0 ; p<=9 ; p++)
			{
				date[i][j]=p+'0';
				if(date[i] <= date[i+1] && date[i] >= r)
				{
					r=date[i];
					res=1;
				}
			}
			date[i][j]=tmp;
		}
		if(res==0)
			break;
		date[i]=r;
	}
	if(res==0)
	{
		cout<< "No solution" << endl;
		return 0;
	}
	for(int i=0 ; i<n ; i++)
	{
		cout<< date[i] << endl;
	}
}
