#include <iostream>
using namespace std;

#define rep(i,n) for (int i = 0; i < (n); i++)

int diff(int n, int d) {
	int cnt = 0;
	while (n > 0) {
		cnt += n%10 != d%10;
		n /= 10; d /= 10;
	}
	return cnt;
}

int f(int n, int min) {
	for (int d = min; d <= 2011; d++) {
		if (diff(n, d) <= 1) {
			return d;
		}
	}
	return -1;
}

int main() {
	int n;
	cin >> n;

	int a[1000];
	rep(i,n) {
		cin >> a[i];
	}

	rep(i,n) {
		a[i] = f(a[i], i == 0 ? 1000 : a[i-1]);
		if (a[i] == -1) {
			cout << "No solution" << endl;
			return 0;
		}
	}

	rep(i,n) {
		cout << a[i] << endl;
	}
	return 0;
}
