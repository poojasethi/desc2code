#include<cstdio>
const int ten[5]={0,1,10,100,1000};
//          Last Change:  2011-03-13 20:47:19
int n,a[1001];
int update(int v,int p,int d)
{
	p=ten[p];
	int tmp=v%p;
	v/=p,v-=v%10,v+=d;
	v=v*p+tmp;
	return v;
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)scanf("%d",a+i);
	a[0]=1000;
	for(int i=1;i<=n;++i)
	{
		int tmp=a[i];
		if(a[i]<a[i-1])a[i]=9999;
		for(int j=1;j<=4;++j)
			for(int k=(j==4);k<=9;++k)
			{
				int v=update(tmp,j,k);
				if(v>=a[i])break;
				if(v<a[i-1])continue;
				a[i]=v;
			}
		if(a[i]>2011)
		{
			puts("No solution");
			return 0;
		}
	}
	for(int i=1;i<=n;++i)printf("%d\n",a[i]);
	return 0;
}
