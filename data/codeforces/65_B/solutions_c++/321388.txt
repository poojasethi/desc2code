#include <iostream>
#include <string>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string all[n+1];
	for(int i=0;i<n;++i)
		cin>>all[i];
	all[n]="2011";
	for(int i=n-1;i>=0;--i)
	{
		string last="0000";
		for(int dig=0;dig<4;++dig)
			for(char change='0';change<='9';++change)
			{
				if(dig==0&&change=='0')
					continue;
				string chk=all[i];
				chk[dig]=change;
				if(chk<=all[i+1]&&chk>last)
				{
					last=chk;
				}
			}
		all[i]=last;
	}
	for(int i=0;i<n;++i)
		if(all[i]<"1000" or all[i]>"2011")
			{
				cout<<"No solution"<<endl;
				return 0;
			}
	for(int i=0;i<n;++i)
		cout<<all[i]<<endl;
	return 0;
}
