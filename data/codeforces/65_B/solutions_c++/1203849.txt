#include<stdio.h>
#include<string.h>
#include<string>
#include<math.h>
#include<map>
#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int i,n,t,k,h,j;
	int a[1005],b[1005][4],c[4];
	while(~scanf("%d",&n))
	{
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);	
			k=3;t=a[i];
			while(t)
			{
				b[i][k--]=t%10;
				t=t/10;
			}
		}	
		h=0;
		for(i=1000;i<2012;i++)
		{
			k=3;t=i;
			while(t)
			{
				c[k--]=t%10;
				t=t/10;
			}
			t=0;
			for(j=0;j<4;j++)
			if(b[h][j]==c[j]) t++;
			if(t==4||t==3)
			{
				a[h++]=i;
				i--;
			}
			if(h==n) break;
		}
		if(h==n) 
		{
			for(i=0;i<n;i++)
			printf("%d\n",a[i]);
		}
		else printf("No solution\n");
	}
}
