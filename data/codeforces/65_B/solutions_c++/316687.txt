#include<cstdio>
#include<cctype>
#include<algorithm>
#include<vector>
#include<set>
#include<map>
#include<queue>
#include<string>
#include<cstring>
using namespace std;

const int N=1005, low=1000, high=2011, inf=9999;
int ans[N];
char y[6],tmp[6];

main(){
	int n,x;
	scanf("%d",&n);
	ans[0]=0;
	for(int i=1; i<=n; i++){
		scanf("%s",y);
		ans[i]=inf;
		for(int j=0; j<4; j++){
			for(int c=0; c<10; c++){
				strcpy(tmp,y);
				tmp[j]=c+'0';
				sscanf(tmp, "%d", &x);
				if(x>=low && x<=high && x>=ans[i-1] && x<ans[i])
					ans[i]=x;
			}
		}
		//printf("ans[%d]=%d\n",i,ans[i]);
	}
	if(ans[n]<inf)
		for(int i=1; i<=n; i++) printf("%d\n",ans[i]);
	else printf("No solution\n");
}
