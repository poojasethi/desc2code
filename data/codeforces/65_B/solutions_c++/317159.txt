#include<iostream>
using namespace std;
int main()
{
int n,i,j,q,k;
string s[1005],t;
cin>>n;
s[0]="0000";
for(i=1;i<=n;i++)
{
cin>>s[i];
t=s[i];
for(q=1000;q<=2011;q++)
{
t="0000";
t[0]=q/1000+'0';
t[1]=q/100%10+'0';
t[2]=q/10%10+'0';
t[3]=q%10+'0';
k=0;
for(j=0;j<4;j++)
{
if(t[j]!=s[i][j]) k++;
}
if(k<2 && t>="1000" && t<="2011" && t>=s[i-1])
{
s[i]=t;
q=3000;
break;
}
}
if(q==2012)
{
cout<<"No solution";
return 0;
}
}
for(i=1;i<=n;i++)
{
cout<<s[i]<<endl;
}
return 0;
}
