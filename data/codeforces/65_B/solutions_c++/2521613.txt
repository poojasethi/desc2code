#include<iostream>
#include<cstdio>

using namespace std;

const int maxn = 1010;

int n , data[maxn] , ans[maxn];

bool check( int i ) {
	int a = data[i] , b = ans[i] , cnt = 0;
	for ( int j = 0 ; j < 4 ; j++ ) {
		if ( a % 10 != b % 10 ) cnt++;
		a /= 10;
		b /= 10;
	}
	return cnt <= 1;
}

int main() {
	ans[0] = 1000;
	scanf( "%d" , &n );
	for ( int i = 1 ; i <= n ; i++ ) {
		scanf( "%d" , &data[i] );
		ans[i] = ans[i - 1];
		while ( !check( i ) && ans[i] <= 2011 ) ans[i]++;
		if ( ans[i] == 2012 ) {
			printf( "No solution\n" );
			return 0;
		}
	}
	for ( int i = 1 ; i <= n ; i++ )
		printf( "%d\n" , ans[i] );
	
	return 0;
}
