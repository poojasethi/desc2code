#include <iostream>
#include <cstring>
#include <string>
using namespace std;
#define MAXN 1010
#define sze 4

bool check(int,int);

int N;
int X[MAXN];

int main(){
  int mn=1000;
  bool t=0,k=1;
  cin >> N;
  for(int i=0;i<N;i++){
    cin >> X[i];
    t=0;
    for(int p=mn;!t && p<=2011;p++)
      if(check(p,X[i])) {t=1; X[i]=p; mn=p; break;}
    if(!t) k=0;
  }

  if(k)
    for(int i=0;i<N;i++){
      cout << X[i] << endl;
    }
  else cout << "No solution" << endl;
  return 0;
}


bool check(int a,int b){
  int k=0;
  while(a){if(a%10!=b%10) k++; a/=10; b/=10;}
  if(k>=2) return 0;
  else return 1;
}
