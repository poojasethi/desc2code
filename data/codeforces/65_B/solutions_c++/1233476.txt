#include<stdio.h>
int y[1001], sol[1001], n;
void input()
{
    //fre3
    
    int i;
    scanf("%d", &n);
    for ( i=0; i<n; i++ ) scanf("%d", &y[i]);
}
int main ()
{
    input();
    int i, j=0;
    for ( i=1000; i<=2011; )
    {
        int a=i, b=y[j], diff=0;
        for ( int k=0; k<4; k++ ) 
        {
            if ( a%10 != b%10 ) diff++;
            a=a/10; b=b/10;
        }
        if ( diff < 2 ) sol[j++]=i;
        else i++;
        if ( j==n ) 
        {
            for ( int k=0; k<n; k++ ) printf("%d\n", sol[k]);
            //getch();
            return 0;
        }
    }
    puts("No solution");
    //getch();
    return 0;
}
