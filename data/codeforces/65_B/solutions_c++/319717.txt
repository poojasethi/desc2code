# include <iostream>
using namespace std;

int data[1500];



bool isMin(int a,int b)
{
    int cnt=0;
    while(a || b)
    {
        if(a%10 != b%10)    cnt++;
        a/=10;
        b/=10;
    }
    return cnt<=1;
}

int main()
{
    int n,i,j;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>data[i];
    for(i=1000,j=0;i<=2011 && j<n;i++)
    {
        if(isMin(data[j],i)){   data[j] = i;    j++;      i--;    }
    }
    if(j==n)
    {
        for(int i=0;i<n;i++)
            cout<<data[i]<<endl;
    }else
    {
        cout<<"No solution"<<endl;
    }
}
