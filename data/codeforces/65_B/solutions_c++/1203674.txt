#include<stdio.h>
#include<string.h>
int a[1005];
int check(int x,int y){
	int cnt=0;
	while(x){
		if(x%10!=y%10)cnt++;
		x/=10;
		y/=10;
	}
	if(cnt<=1)return 1;
	else return 0;
}
int main(){
	int n,i,ans,j;
	while(scanf("%d",&n)!=EOF){
		for(i=1;i<=n;i++){
			scanf("%d",&a[i]);
		}
		a[0]=1000;
		int f=0;
		for(i=1;i<=n;i++){
			ans=a[i];
			a[i]=-1;
			for(j=a[i-1];j<=2011;j++){
				if(check(j,ans)){
					a[i]=j;
					break;
				}
			}
			if(a[i]==-1){
				f=1;
				break;
			}
		}
		if(f==1)printf("No solution\n");
		else{
			for(i=1;i<=n;i++)
			printf("%d\n",a[i]);
		}
	}
	return 0;
}
