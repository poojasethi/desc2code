#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 3000
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,u;
int a[N],f[N];
inline int get(int x,int y) {
	int i;
	For(i,1,y-1) x/=10;
	return x%10;
}
inline void fail() {
	printf("No solution\n");
	exit(0);
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d",&a[i]);
	f[0]=1000;
	For(i,0,n-1) {
		int A=1,mi=2012;
		For(k,1,4) {
			For(u,0,9) {
				int B=get(a[i+1],k);
				int C=a[i+1]-B*A+u*A;
				if (C>=f[i]) mi=min(mi,C);
			}
			A*=10;
		}
		if (mi==2012) fail();
		f[i+1]=mi;
	}
	For(i,1,n) printf("%d\n",f[i]);
	return 0;
}
