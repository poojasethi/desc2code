#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>

using namespace std;

int n;
int a[1010];

int main()
{
	cin >> n;
	a[0] = -1;
	for(int i = 1; i <= n; ++i)
	{
		cin >> a[i];
		int ten = 1, tem = 10000;
		for(int j = 0; j < 4; ++j)
		{
			int bit = a[i] / ten % 10;

			for(int k = 0; k <= 9; ++k)
			{
				int now = a[i] - (bit - k) * ten;
				if(now < 1000 || now > 2011 || now < a[i - 1]) continue;
				tem = min(tem, now);
			}
			ten *= 10;
		}
		if(tem == 10000)
		{
			puts("No solution");
			return 0;
		}
		a[i] = tem;
	}
	for(int i = 1; i <= n; ++i)
		cout << a[i] << endl;
	return 0;
}
