#include <cstdio>
#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main() {
	int n;
	string p="1000";
	vector<string> r;
	cin >> n;
	for (int i=1;i<=n;i++) {
		string x;
		cin >> x;
		string best="9999";
		for (int j=0;j<4;j++) {
			for (int d=(j==0?'1':'0');d<='9';d++) {
				string y=x; y[j]=d;
				if (p<=y && y<best) best=y;
			}
		}
		if (best!="9999") {
			p=best; r.push_back(best);
		} else break;
	}
	if (r.size()!=n || "2011"<r[r.size()-1]) cout << "No solution\n" << endl;
	else {
		for (int i=0;i<n;i++) cout << r[i] << endl;
	}
    return 0;
}
