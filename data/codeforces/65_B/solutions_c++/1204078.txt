#include <stdio.h>

int go(int a,int b)
{
	int j,i,sum;
	sum=0; 
	for(i=1;i<=4;i++)
	{
		if(a%10==b%10) sum++;
		a/=10;
		b/=10;
		
	}
	if(sum>=3) return 1;
	else return 0;
	
	
}
