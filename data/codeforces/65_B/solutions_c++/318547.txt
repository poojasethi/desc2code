
#include <cstdio>

int d[1010];
int ans[1010];

bool check(int a,int b)
{
	if(a>b){int t=b; b = a; a=t;}
	int res= b-a;

	while(res){
		if(res%10>0){
			if( (a%10)+(res%10)<10 )res /= 10;
			break;
		}
		res /= 10;
		a /= 10;
	}

	return res==0;
}

int main()
{
	int n;
	while(scanf("%d",&n)==1){
	
		for(int i=0;i<n;++i)scanf("%d",d+i);

		int j=1000;
		for(int i=0;i<n; ++i){
			while(j<2012){

				if( check(d[i],j) ){
					ans[i]=j;
					break;
				}
				j++;
			}
			if(j>=2012)break;
		}

		if(j>=2012){puts("No solution");continue;}

		for(int i=0;i<n;++i)printf("%d\n",ans[i]);
	
	
	}

	return 0;
}
