#include<cstdio>
#include<algorithm>
#include<cmath>
#include<iostream>
#include<cstring>
using namespace std;

bool check(int a, int b) {
	int cnt = 0;
	while (a) {
		if (a%10 != b%10) ++cnt;
		b /= 10; a /= 10;
	}
	if (cnt <= 1) return 1;
	else return 0;
}
int main() {
	int n, b;
	int a = 1000;
	int res[1005], nn = 0;
	scanf("%d", &n);
	for (int cnt=0; cnt<n; ++cnt){
		scanf("%d", &b);
		for (int i=a; i<2012; ++i) {
			if (check(i, b)){
				a = res[++nn] = i;
				break;
			}
		}
	}
	if (nn == n) for (int i=1; i<=nn; ++i) printf("%d\n", res[i]);
	else printf("No solution\n");
}
