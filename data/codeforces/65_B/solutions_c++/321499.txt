#include<iostream>
#include<map>
using namespace std;
string  minimize(string a, string b){
	string minb="9999",tmpb;
	for(int i=0; i < 4; i++)
	{
		tmpb=b;
		for(int j=(i==0)?1:0; j < 10; j++){
		tmpb[i]='0'+j;
		if(tmpb >= a)
		if(tmpb < minb) minb=tmpb;
		}
	}
	return minb;		
}
int main()
{
int N;
string arr[1010];
cin>>N;
for(int i=0; i < N; i++) cin>>arr[i];

string str;
arr[0]=minimize("1000",arr[0]);
bool flag=true;
for(int i=1; i < N; i++){ str=minimize(arr[i-1],arr[i]);
string maxi="2011",mini="1000";
if(str < mini ||  str > maxi ) { flag=false; break; }
arr[i]=str;
}
if(flag){
for(int i=0; i < N; i++)
	cout<<arr[i]<<"\n";
	}
else cout<<"No solution\n";
return 0;
}
