#include <string>
#include <iostream>
#include <vector>
using namespace std;

inline int abs(int x) {
	return x > 0 ? x : -x;
}

bool gao(int x, int y) {
	int cnt = 0;
	for (int i = 0; i < 4; ++i) {
		if (x % 10 != y % 10) {
			cnt++;
		}
		x /= 10, y /= 10;
	}
	return cnt <= 1;
}

int main() {
	int n;
	cin >> n;
	vector<int> y(n + 1);
	y[0] = 1000;
	for (int i = 0; i < n; ++i) {
		cin >> y[i + 1];
	}


	for (int i = 1; i <= n; ++i) {
		int j;
		for (j = y[i - 1]; j <= 2011; ++j) {
			if (gao(j, y[i])) {
				y[i] = j;
				break;
			}
		}
		if (j > 2011) {
			cout << "No solution\n";
			return 0;
		}
	}

	for (int i = 1; i <= n; ++i) {
		cout << y[i] << '\n';
	}
	return 0;
}
