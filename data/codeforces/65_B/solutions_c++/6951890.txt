#include<iostream>
#include<stdlib.h>
using namespace std;
int dif(int fir,int sec){
    int c=0;
    for(;fir!=0;fir/=10,sec/=10)
        if(fir%10!=sec%10)
            c++;
    if(c<2)return 1;
    else return 0;
}
int main(){
    int n,k,i,l=1000,a[2000],tr=1,j;
    cin>>n;
    for(int t=1;t<=n;t++){
        cin>>k;
        for(i=l;i<2012;i++)
            if(dif(k,i)){
                a[t]=i;break;
            }
        if(a[t]==0)tr=0;
        l=a[t];
    }
    if(!tr)cout<<"No solution\n";
    else{
        for(i=1;i<=n;i++)
            cout<<a[i]<<" ";
        cout<<endl;
    }
}
