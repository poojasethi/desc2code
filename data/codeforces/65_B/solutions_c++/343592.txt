#include <stdio.h>

int ans[1010],i;
int check(int a,int b)
{ int k=0;
 
  while (a>0)
  {if ((a%10)!=(b%10)) k++;
    a/=10; b/=10;}
   
 if (k<2) return 1;
 return 0;     
}
 
int main ()
{ int n,k,p,j;
  bool bo;
  scanf("%d",&n); p=1000;

for (i=0 ; i<n; i++)
{scanf("%d",&k); bo=false;
  for (j=p ; j<=2011; j++)  
   if (check(j,k)) {ans[i]=j; p=ans[i]; bo=true; break; }
  
 if (!bo) { printf("No solution\n"); return 0; }
}
for (i=0 ; i<n ; i++)
printf("%d\n",ans[i]);
}
