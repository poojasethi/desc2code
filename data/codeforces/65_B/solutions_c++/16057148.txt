#include <iostream>
#include <string>
using namespace std;


int a[1111];
int n;

bool f(int x, int y)
{
	int r = 0;
	for(int i=1; i<=1e9; i*=10)
	{
		if((x/i)%10 != (y/i)%10)
			r++;
	}

	if(r <= 1)
		return 1;
	else
		return 0;
}


int main()
{
	cin >> n;
	for(int i=0; i<n; i++)
		cin >> a[i];

	int last = 1000;

	for(int i=0; i<n; i++)
	{
		while(f(last, a[i]) == false)
		{
			if(last > 2011)
				break;
			last++;
		}
		a[i] = last;
	}

	if(a[n-1] > 2011)
	{
		cout << "No solution" << endl;
		return 0;
	}

	for(int i=0; i<n; i++)
		cout << a[i] << endl;

	return 0;
}
