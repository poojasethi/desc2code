#include<stdio.h>
int a[1005];
int main()
{
    int n,i,j,l,x;
    scanf("%d",&n);
    a[0]=0;
    for(i=1; i<=n; i++)
        scanf("%d",a+i);
    for(i=1; i<=n; i++)
    {
        int y=10000;
        for(j=1000; j>=1; j/=10)
        {
            for(l=0; l<10; l++)
            {
                x=a[i]-a[i]%(j*10)+a[i]%j+l*j;
                if(x>=1000&&x<=2011&&x>=a[i-1]&&x<y)
                {
                    y=x;
                    break;
                }
            }
        }
        if(y==10000)break;
        a[i]=y;
    }
    if(i<=n)puts("No solution");
    else
    {
        for(i=1; i<=n; i++)printf("%d\n",a[i]);
    }
    return 0;
}
