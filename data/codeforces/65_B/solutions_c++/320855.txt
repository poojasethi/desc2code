#include<iostream>
#include<algorithm>
using namespace std;

string da[1010];

int main()
{
	int N;
	cin>>N;
	for(int i = 0; i < N; ++i)
		cin>>da[i];
	string prev = "2011";
	reverse(da,da+N);
	for(int i = 0; i < N; ++i)
	{
		bool ok = true;
		string cand = da[i];
		for(int j = 0; j < 4; ++j)
		for(char c = '0' ; c <= '9'; ++c)
		{
			string co = da[i];
			co[j] = c;
			if(co <= prev && co > cand)cand = co;
			else if(cand > prev && co <= prev)cand = co;
		}
		if(i>0 && cand>da[i-1]){cout<<"No solution"<<endl;return 0;}
		da[i] = cand;
		prev = cand;
	}
	reverse(da,da+N);
	if(da[0]<"1000")cout<<"No solution"<<endl;
	else
	for(int i = 0; i < N; ++i)
		cout<<da[i]<<endl;
}
