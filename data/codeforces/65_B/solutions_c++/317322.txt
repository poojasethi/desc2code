#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

int n,x,r[1005],ok;

int diff(int x,int y){
	int R=0;
	for (int i=0;i<4;i++){
		if (x%10!=y%10) R++;
		x/=10,y/=10;
	}
	return R;
}


int main(){

	scanf("%d",&n);
	r[0]=1000;
	for (int i=1;i<=n;i++){
		scanf("%d",&x);
		ok=0;
		for (int j=r[i-1];j<=2011;j++)
		    if (diff(j,x)<=1){
				r[i]=j;
				ok=1;
				break;
			}
		if (!ok) break;
	}
	
	if (!ok) printf("No solution\n");
	else{
		for (int i=1;i<=n;i++) printf("%d\n",r[i]);
	}
	
	return 0;
}
