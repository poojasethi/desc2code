#include <iostream>
#include <algorithm>

using namespace std;

bool check(int a,int b) {
	int same=0;
	for (int i=0; i<4; i++) {
		if (a%10==b%10)same++;
		a/=10;
		b/=10;
	}
	return (same>=3);
}

int main() {
	int n;
	int year[1000];
	int mini=1000,maxi=2011;
	int flag=1;
	cin >> n;
	for (int i=0; i<n; i++) {
		cin >> year[i];
		int flag2=0;
		for (int j=mini; j<=maxi; j++) {
			if (check(j,year[i])) {
				mini=j;
				flag2=1;
				year[i]=j;
				break;
			}
		}
		if (flag2==0){
			flag=0;
			mini=maxi;
		}
	}
	if (flag==1) {
		for (int i=0; i<n; i++) {
			cout << year[i] << endl;
		}
	} else {
		cout << "No solution" << endl;
	}
	return 0;
}
