#include <cstdio>

int main()
{
    int n, i, tree_model[10001] = {0}, a, b, lifelines = 0;

    scanf("%d", &n);
    for(i = 1; i <= n - 1; i++){
        scanf("%d%d", &a, &b);
        tree_model[a]++;
        tree_model[b]++;
    }

    for(i = 1; i <= n; i++)
        lifelines += (tree_model[i] - 1) * tree_model[i] / 2;

    printf("%d", lifelines);

    return 0;
}
