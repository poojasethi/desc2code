#include <bits/stdc++.h>
using namespace std;
long long ans,n;
vector <long long> G[100005];
int main(){
    scanf("%I64d",&n);
    for(long long i=1;i<n;i++){
        long long x,y;
        scanf("%I64d %I64d",&x,&y);
        G[x].push_back(y);
        G[y].push_back(x);
    }
    for(long long i=1;i<=n;i++) ans += (long long) G[i].size() * (G[i].size()-1) / 2;
    cout << ans;
}
