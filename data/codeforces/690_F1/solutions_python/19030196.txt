#include<bits/stdc++.h>
using namespace std;
vector<int> G[10009];
int main()
{
    int n,a,b;
    cin>>n;
    for(int i=0;i<n-1;i++)
    {
        cin>>a>>b;
        a--;b--;
        G[a].push_back(b);
        G[b].push_back(a);
    }
    int count=0;
    for(int i=0;i<n;i++)
    {
        int k=G[i].size();
        count+=(k&1) ? k*((k-1)/2) :(k/2)*(k-1);
    }
    cout<<count<<endl;
}
