#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
int map[1000000];
int main()
{
	int n;
	while(cin>>n)
	{
		memset(map,0,sizeof(map));
		for(int i = 0; i < n -1 ; i++)
		{
			int a,b;
			cin>>a>>b;
			map[a] ++;
			map[b] ++;
		}
		int ans = 0;
		for(int i = 1; i <= n; i ++)
		{
			ans += map[i] * (map[i] - 1) / 2;
		}
		cout<<ans<<endl;
	}
}
