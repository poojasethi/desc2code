#include<cstdio>

using namespace std;

long long n,deg[10010],i,a,b,sum;

int main(){
    scanf("%I64d",&n);
    for(i=1;i<n;i++){
        scanf("%I64d %I64d",&a,&b);
        deg[a]++;
        deg[b]++;
    }
    for(i=1;i<=n;i++){
        sum+=(deg[i]*(deg[i]-1))/2;
    }
    printf("%I64d\n",sum);
}
