#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;


const int mx=10005;
int vs[mx];

int main()
{
    int n;
    scanf("%d",&n);
    int a,b;
    for (int i=1;i<n;i++)
    {
        scanf("%d%d",&a,&b);
        vs[a]++;
        vs[b]++;
    }
    int ans=0;
    for (int i=1;i<=n;i++) ans+=vs[i]*(vs[i]-1)/2;
    printf("%d\n",ans);
}
