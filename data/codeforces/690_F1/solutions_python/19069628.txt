#include <stdio.h>
#include <vector>
using namespace std;

int main()
{
	int x, i, a, b, sum = 0;
	scanf("%d", &x);
	vector<int> path[x+1];
	for (i=1;i<x;i++)
	{
		scanf("%d%d", &a, &b);
		path[a].push_back(b);
		path[b].push_back(a);
	}
	for (i=1;i<=x;i++) sum += path[i].size()*(path[i].size()-1)/2;
	printf("%d\n", sum);
	return 0;
}
