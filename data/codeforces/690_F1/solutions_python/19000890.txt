#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define maxn 10001

vector<ll> gr[maxn];

int main()
{
	ll n, i, a, b, ret = 0;
	scanf("%I64d", &n);
	for(i = 1; i < n; i++)
	{
		scanf("%I64d %I64d", &a, &b);
		gr[a].push_back(b);
		gr[b].push_back(a);
	}
	for(i = 1; i <= n; i++)
	{
		a = gr[i].size();
		ret += a*(a - 1)/2;
	}
	printf("%I64d", ret);
	return 0;
}
