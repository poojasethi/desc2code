#include <iostream>
#include <vector>
using namespace std;
int main(){
	int n;
	cin>>n;
	vector< vector<int> > graph(n,vector<int>());
	for(int i=1;i<n;i++){
		int a,b;
		cin>>a>>b;
		a--;b--;
		graph[a].push_back(b);
		graph[b].push_back(a);
	}
	long long result=0;
	for(int i=0;i<n;i++)
		result+=graph[i].size()*(graph[i].size()-1)/2;
	cout<<result;
}
