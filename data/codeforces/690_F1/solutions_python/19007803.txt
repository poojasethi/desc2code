#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
using namespace std;
int n;
int a,b;
int du[11111];
int ans=0;
int main()
{
	scanf("%d",&n);
	for(int i=1;i<n;i++)
		{
			scanf("%d%d",&a,&b);
			du[a]++;
			du[b]++;
		}
	for(int i=1;i<=n;i++)
		{
			if(du[i]>=2)
				{
					ans+=du[i]*(du[i]-1)/2;
				}
		}
	cout<<ans<<endl;
}
