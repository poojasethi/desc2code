#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long ll;

int cnt[10007];
int main()
{
	int n;
	scanf("%d", &n);
	int a, b;
	for(int i  = 1; i <= n -1; i++) {
		scanf("%d %d", &a, &b);
		cnt[a]++;
		cnt[b]++;
	}
	ll ans = 0;
	for(int i = 1; i <= n; i++) {
		if(cnt[i] >= 2) {
			ans += (ll)cnt[i] * (cnt[i] - 1)/ 2;
		}
	}
	printf("%I64d\n", ans);
}
