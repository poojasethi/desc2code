#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int a[10001];
int main()
{
	int n,res,s1,s2,s,i,j;
	while(~scanf("%d",&n)){
		for(i=1;i<=n;i++)
		 a[i]=0;
		for(i=0;i<n-1;i++)
		{
			scanf("%d%d",&s1,&s2);
			a[s1]++;a[s2]++;
		}
		res=0;
		for(i=1;i<=n;i++)
		{
			res+=a[i]*(a[i]-1)/2;
		}
		printf("%d\n",res);
	}
	return 0;
}
