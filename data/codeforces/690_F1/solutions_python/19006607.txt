#include<cstdio>
#include<cstring>
int main()
{
    int arr[10005];
    memset(arr,0,sizeof(arr));
    int n,p,q,i,ans=0;
    scanf("%d",&n);
    for(i=0;i<n-1;i++)
    {
        scanf("%d%d",&p,&q);
        arr[p]++;
        arr[q]++;
    }
    for(i=1;i<=n;i++)
    {
        ans=ans+(1+arr[i]-1)*(arr[i]-1)/2;
    }
    printf("%d\n",ans);
    return 0;
}
