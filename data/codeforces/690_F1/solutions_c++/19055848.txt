#include <bits/stdc++.h>

using namespace std;

int main(){
	int n;
	int a, b;
	cin >> n;
	vector<int> edgesCount (n+1, 0);
	for (int i = 1; i<n; ++i){
		cin>>a>>b;
		edgesCount[a]++;
		edgesCount[b]++;
	}
	int answer = 0;
	for (int i = 1; i <= n; ++i)
	{
		// cout<<i<<" "<<edgesCount[i]<<endl;
		if(edgesCount[i] > 1){
			answer += edgesCount[i]*(edgesCount[i]-1) / 2;
		}
	}
	cout<<answer<<endl;
}
