#include<iostream>
#include<vector>
using namespace std;

const int oo=10010;
int n,x,y,d[oo],ans;
vector<int> edg[oo];

int main(){
  cin>>n;
  for(int i=1;i<n;i++)
    cin>>x>>y,d[x]++,d[y]++,edg[x].push_back(y),edg[y].push_back(x);
  for(int i=1;i<=n;i++)
    for(int j=0;j<edg[i].size();j++)
      ans+=d[edg[i][j]]-1;
  cout<<ans/2;
  return 0;
}
