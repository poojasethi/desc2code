#include <bits/stdc++.h>
using namespace std;
const int MaxN = 1e6 + 5;
int num[MaxN], m, n, ans;
int main()
{
	scanf("%d", &n);
	for(int i = 1; i <= n - 1; i++)
	{
		int u, v;
		scanf("%d %d", &u, &v);
		num[u]++, num[v]++;
	}
	for(int i = 1; i <= n; i++)
	{
		ans += (num[i] - 1) * num[i] / 2;
	}
	printf("%d\n", ans);
}
