#include <bits/stdc++.h>
using namespace std;
int rec[10500];
int n;
int main()
{
	while(~scanf("%d",&n))
	{
		memset(rec,0,sizeof(rec));
		int x,y;
		for(int i=1;i<n;i++)
		{
			scanf("%d%d",&x,&y);
			rec[x]++;
			rec[y]++;
		}
		int sum=0;
		for(int i=1;i<=n;i++)
		{
			if(rec[i]<=1)continue;
			else sum+=rec[i]*(rec[i]-1)/2;
		}
		printf("%d\n",sum);
	}
	return 0;
}
