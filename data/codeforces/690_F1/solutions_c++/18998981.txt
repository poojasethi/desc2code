#include <bits/stdc++.h>
using namespace std ;

long long  yo[10005] , n , x, y ,ans ;
int main()
{

cin >> n ;

for( int i = 1 ; i < n ; i++)
{
    scanf("%I64d %I64d", &x, &y ) ;
    yo[x]++ ;
    yo[y]++ ;
}

for( int i = 1 ; i <= n ; i++ )
{
    ans += ( yo[i] * (yo[i]-1) / 2 ) ;
}


cout << ans ;





return 0 ;
}
