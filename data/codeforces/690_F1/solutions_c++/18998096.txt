#include<cstdio>
#include<cstring>
#include<algorithm>
#define LL long long
using namespace std;
int deg[10005];
int n;
int main()
{	LL ans=0;
	int i,x,y;
	scanf("%d",&n);
	for (i=1;i<n;++i)
	{	scanf("%d%d",&x,&y);
		++deg[x],++deg[y];
	}
	for (i=1;i<=n;++i)
		ans+=deg[i]*(deg[i]-1)/2;
	printf("%I64d\n",ans);
	fclose(stdin);
	fclose(stdout);
	return 0;
}
