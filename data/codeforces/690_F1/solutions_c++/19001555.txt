#include <iostream>
#include<cmath>
#include<vector>
#include<cstring>
using namespace std;

int size[10005];

int main()
{
	ios_base::sync_with_stdio(0);
	int n;
	cin >> n;
	for (int i = 1; i < n; i++)
	{
		int a, b;
		cin >> a >> b;
		size[a]++, size[b]++;
	}
	long long ans = 0;

	for (int i = 1; i <= n; i++) ans += (1ll * size[i] * (size[i] - 1)) / 2;
	cout << ans << "\n";


	return 0;
}
