#include <bits/stdc++.h>
using namespace std;

long long ans = 0;
int n;
vector<int> Tree[10005];
int main(){
	scanf("%d", &n);
	for (int i = 1; i < n; ++i){
		int x, y; scanf("%d%d", &x, &y);
		Tree[x].push_back(y);
		Tree[y].push_back(x);
	}
	for (int i = 1; i <= n; ++i){
		long long t = Tree[i].size();
		ans += t * (t - 1) / 2;
	}
	cout << ans;
	return 0;
}
