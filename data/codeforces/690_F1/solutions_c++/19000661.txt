#include<iostream>
#define MAX_N 10000
using namespace std;
long long a[MAX_N],N;
void solve(){
  long long cnt=0;
  for(int i=0;i<N;i++){
    if(a[i]>1){
      cnt+=a[i]*(a[i]-1)/2;
    }
  }
  cout<<cnt<<endl;
}
int main(){
  cin>>N;
  for(int i=0;i<N-1;i++){
    long long x,y;
    cin>>x>>y;
    a[x-1]++;
    a[y-1]++;
  }
  solve();
  return 0;
}
