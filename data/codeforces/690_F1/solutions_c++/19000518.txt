#include <iostream>
using namespace std;

int main() {
	int n,i=0;
	cin>>n;
	n=n-1;
	int a[n],b[n];
	while(i<n){
		
		cin>>a[i]>>b[i];
		i++;
		
	}
	int j,k,p=0;
	for(j=0;j<n-1;j++){
		for(k=j+1;k<n;k++){
			if(a[j]==a[k]||a[j]==b[k]||b[j]==b[k]||b[j]==a[k])
			p++;
		}
	}
	cout<<p<<"\n";
	return 0;
}
