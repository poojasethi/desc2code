#include <bits/stdc++.h>
using namespace std;

vector<int> X[10004];

int main() {
	
	int N;
	scanf("%d", &N);
	
	for(int i=1; i<=N-1; i++)
	{
	    int x, y;
	    scanf("%d %d", &x, &y);
	    
	    X[x].push_back(y);
	    X[y].push_back(x);
	}
	
	long long ans = 0;
	
	for(int i=1; i<=N; i++)
	    ans += (long long)X[i].size()*(X[i].size()-1)/2;
	   
	printf("%lld\n", ans);
	
	return 0;
}
