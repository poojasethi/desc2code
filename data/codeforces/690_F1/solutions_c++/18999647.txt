#include<bits/stdc++.h>
using namespace std;

typedef long long LL;
const int MAXN = 10005;

int N;
int du[MAXN];

void work()
{
  int u, v;
  LL ret = 0;
  scanf("%d", &N);
  for(int i=1;i<N;i++) {
    scanf("%d%d", &u, &v);
    du[u]++; du[v]++;
  }
  for(int i=1;i<=N;i++)
    ret += 1ll * du[i] * (du[i]-1)/2;
  cout<<ret<<endl;
}

int main()
{
#ifdef LOCAL
  freopen("in.txt", "r", stdin);
#endif
  work();
  return 0;
}
