#include <bits/stdc++.h>
using namespace std;
int main()
{
  int n,x,y;
  scanf("%d",&n);
  int a[10001]={0};
  for(int i=0;i<n-1;i++) {
    scanf("%d%d",&x,&y);
    a[x]++;
    a[y]++;
  }
  long long int ans = 0;
  for(int i=0;i<n+1;i++) {
    ans += (a[i]*(a[i]-1))/2;
  }
  cout << ans << endl;
  return 0;
}
