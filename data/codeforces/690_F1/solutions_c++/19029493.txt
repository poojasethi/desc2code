#include <bits/stdc++.h>

using namespace std;


int main()
{
	int n;
	int a,b;
	int count = 0;
	cin >> n;
	int * neigh = new int[n];
	for (int i = 0; i < n; i++)
		neigh[i] = 0;
	for (int i = 1; i < n; i++)
	{
		cin >> a >> b;
		neigh[a - 1]++;
		neigh[b - 1]++;
	}
	for (int i = 0; i < n; i++)
	{
		if (neigh[i] > 1) {
			count += 1;
			for (int j = 2; j < neigh[i]; j++)
				count+= j;
		}
	}
	cout << count;
	return 0;
}
