#include <bits/stdc++.h>
using namespace std;

int f[10005];

int main(){
    int n; cin >> n;
    for(int i = 1;i < n; i++){
        int a, b; cin >> a >> b;
        f[a]++; f[b]++;
    }

    int ans = 0;
    for(int i = 1;i <= n; i++){
        ans += (f[i] * (f[i] - 1)) / 2;
    }

    cout << ans << '\n';
}
