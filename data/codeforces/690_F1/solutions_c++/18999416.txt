#include<bits/stdc++.h>

using namespace std;

int du[10005];

int main() {
  //freopen("in.txt","r",stdin);
  int n;
  scanf("%d",&n);
  for(int i=1;i<n;i++) {
    int x, y;
    scanf("%d%d",&x,&y);
    du[x]++;
    du[y]++;
  }

  long long ans = 0;
  for(int i=1;i<=n;i++)
    ans += 1ll*du[i]*(du[i]-1)/2;
  cout<<ans<<'\n';
  return 0;
}
