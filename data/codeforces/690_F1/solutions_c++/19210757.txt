#include <iostream>
#include <map>
using namespace std;
map <string,int> s;
int main()
{
    int n;
    string a,b;
    int sum=0;
    cin>>n;
    for(int i=1; i<n; i++)
    {
        cin>>a>>b;
        if(s[b]>0){
            sum+=s[b];
        }
        if(s[a]>0)
            sum+=s[a];

        s[a]+=1;
        s[b]+=1;

    }
    cout<<sum;
}
