#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
using namespace std;
#define inf 1000000

int main(){
	char M[500][500];
	int cnt[500][26][26];
	int dp[500][26][26];
	int son[500][26][26];
	int n, m;
	
	cin >> n >> m;
	for(int i = 0; i < n; i++) for(int j = 0; j < m; j++){
	 scanf(" %c", &M[i][j]);		
	 M[i][j] -= 'a';
	}
	for(int row = 0; row < n; row++)
		for(int a = 0; a < 26; a++)
			for(int b = 0; b < 26; b++)
				if(a == b) cnt[row][a][b] = inf;
				else{
					int sum = 0;
					for(int i = 0; i < m; i += 2)
						if(M[row][i] != a) sum++;
					for(int i = 1; i < m; i += 2)
						if(M[row][i] != b) sum++;
					cnt[row][a][b] = sum;
				}
	
	for(int row = n - 1; row >= 0; row--)
		for(int aa = 0; aa < 26; aa++)
			for(int bb = 0; bb < 26; bb++){
				int &s = son[row][aa][bb] = -1;
				int &res = dp[row][aa][bb] = inf;
				if(row == n - 1){
					res = cnt[row][aa][bb];
					continue;
				}
				for(int a = 0; a < 26; a++) if(a != aa)
					for(int b = 0; b < 26; b++) if(b != bb){
						int costo;
						if(row == n - 1) costo = cnt[row][aa][bb];
						else costo = cnt[row][aa][bb] + dp[row + 1][a][b];
						if(costo < res){
							res = costo;
							s = a * 1000 + b;
						}
					}
			}	
			
	int a1 = 0, b1 = 0;
	for(int a = 0; a < 26; a++) for(int b = 0; b < 26; b++) if(dp[0][a][b] < dp[0][a1][b1]){
		a1 = a;
		b1 = b;
	}
	
	printf("%d\n", dp[0][a1][b1]);
	for(int row = 0; row < n; row++){
		for(int i = 0; i < m; i++)
			if(i & 1) printf("%c", b1 + 'a');
			else printf("%c", a1 + 'a');
		printf("\n");
		int s = son[row][a1][b1];
		if(s == -1) break;
		a1 = s / 1000;
		b1 = s % 1000;
	}
	
}
