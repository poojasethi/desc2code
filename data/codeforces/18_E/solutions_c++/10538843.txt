#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int n,m;
char str[505][505];
int dp[505][27][27],path_a[505][27][27],path_b[505][27][27];
int cost(int a,int x,int y){
 char c1=x+'a',c2=y+'a';
 int i,ans=0;
for(i=1;i<=m;i++)
if(i&1)
   ans+=(c1!=str[a][i]);
   else
   ans+=(c2!=str[a][i]);
   return ans;
}
void put(int le,int a,int b){
 if(le>1)put(le-1,path_a[le][a][b],path_b[le][a][b]);
 int i;
 for(i=1;i<=m;i++)
	if(i&1)
		printf("%c",a+'a');
	else
		printf("%c",b+'a');
	printf("\n");
}
void solve(){
 int i,j;
 for(i=0;i<26;i++)
	 for(j=0;j<26;j++)
		 if(i!=j){
			 dp[1][i][j]=cost(1,i,j);
		 }
		 int a,b,c,d,cur;
		 for(i=2;i<=n;i++){
		  for(a=0;a<26;a++)
			  for(b=0;b<26;b++){
				  if(a!=b){
				   cur=cost(i,a,b);
			   int min=1000000000,x,y;
			   for(c=0;c<26;c++)
				   for(d=0;d<26;d++){
					   if(c==d||a==c||d==b)
						   continue;
					   if(min>dp[i-1][c][d]){
					    min=dp[i-1][c][d];
						x=c;
						y=d;
					   }	   
				   }
			
				   dp[i][a][b]=min+cur;
				   path_a[i][a][b]=x;
				   path_b[i][a][b]=y;
			  }
			  }
		 }
		 int ans=1000000000,x,y;
		 for(a=0;a<26;a++)
			 for(b=0;b<26;b++)
				 if(a!=b&&ans>dp[n][a][b]){
				  ans=dp[n][a][b];
				  x=a;
				  y=b;
		}

				 printf("%d\n",ans);
put(n,x,y);
}
int main(){
	while(cin>>n>>m){
	 int i;
	 for(i=1;i<=n;i++)
		 scanf("%s",str[i]+1);
solve();
	}
	return 0;
}
