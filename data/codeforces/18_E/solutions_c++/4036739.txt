#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int N,M;
char str[510][510];
int f[510][700];
int pre[510][700];


void init(){
    scanf("%d %d\n",&N,&M);
    for (int i=1;i<=N;++i)
        gets(str[i]);
}

int calc(int k,char x,char y){
    int ans=0;
    for (int i=0;i<M;++i){
        if (i%2==0 && str[k][i]!=x) ans++;
        if (i%2!=0 && str[k][i]!=y) ans++;
    }
    return ans;
}


void dp(){
    memset(f,1,sizeof(f));
    for (int i=0;i<676;++i)
        f[0][i]=0;
    for (int i=1;i<=N;++i)
        for (int j=0;j<676;++j){
            if (j/26==j%26) continue;
            int tmp=calc(i,j/26+'a',j%26+'a');
            for (int k=0;k<676;++k)
                if (j/26!=k/26 && j%26 != k%26 )
                    if (f[i][j]>f[i-1][k]+tmp){
                        f[i][j]=f[i-1][k]+tmp;
                        pre[i][j]=k;
                    }
        }
}

int list[501];

void output(){
    int Ans=250000;
    int p;
    for (int i=0;i<676;++i){
        if (f[N][i]<Ans){
            Ans=f[N][i];
            p=i;
        }
    }
    printf("%d\n",Ans);
    //cout<<p<<endl;
    for (int i=N;i>0;--i){
        list[i]=p;
        p=pre[i][p];
    }
    for (int i=1;i<=N;++i){
        int x=list[i]/26;
        int y=list[i]%26;
        for (int j=0;j<M;++j)
            if (j%2==0) putchar('a'+x);
                else putchar('a'+y);
        printf("\n");
    }
}

int main(){
    //freopen("input.txt","r",stdin);
    init();
    dp();
    output();
    return 0;
}
