#include <cstdio>

using namespace std;

char s[501][501];
int f[501][27][27],px[501][27][27],py[501][27][27],ansx[501],ansy[501];

int main(void)
{
  int n,m;
  scanf("%d %d",&n,&m);
  for(int i=1;i<=n;++i)
  {
    scanf("%s",s[i]);
    for(int j=0;j<26;++j)
      for(int k=0;k<26;++k)
        f[i][j][k]=300000;
  }
  for(int i=1;i<=n;++i)
  {
    for(int j=0;j<26;++j)
    {
      for(int k=0;k<26;++k)
      {
        if(j==k)
          continue;
        int temp=0;
        for(int l=0;l<m;l+=2)
          if(s[i][l]-'a'!=j)
            ++temp;
        for(int l=1;l<m;l+=2)
          if(s[i][l]-'a'!=k)
            ++temp;
        for(int p=0;p<26;++p)
        {
          if(p==j)
            continue;
          for(int q=0;q<26;++q)
          {
            if((p==q)||(q==k))
              continue;
            if(f[i-1][p][q]+temp<f[i][j][k])
            {
              f[i][j][k]=f[i-1][p][q]+temp;
              px[i][j][k]=p;
              py[i][j][k]=q;
            }
          }
        }
      }
    }
  }
  int ans=300000;
  for(int i=0;i<26;++i)
  {
    for(int j=0;j<26;++j)
    {
      if(i==j)
        continue;
      if(f[n][i][j]<ans)
      {
        ans=f[n][i][j];
        ansx[n]=i;
        ansy[n]=j;
      }
    }
  }
  for(int i=n-1;i>0;--i)
  {
    int j=ansx[i+1],k=ansy[i+1];
    ansx[i]=px[i+1][j][k];
    ansy[i]=py[i+1][j][k];
  }
  printf("%d\n",ans);
  for(int i=1;i<=n;++i)
  {
    for(int j=0;j<m;++j)
    {
      if(j%2==0)
        printf("%c",ansx[i]+'a');
      else
        printf("%c",ansy[i]+'a');
    }
    printf("\n");
  }
  return 0;
}
