#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

char g[500][500];
int n, m;
int dp[500][26][26];
int ans[500][26][26][2];

void print(int level, int a, int b)
{
	if(level > 0)
		print(level-1, ans[level][a][b][0], ans[level][a][b][1]);
	for(int i = 0; i < m; i++)
		if(i % 2 == 0)	printf("%c", 'a' + a);
	    else if(i % 2 == 1)	printf("%c", 'a' + b);
	printf("\n");
}

int main()
{
	scanf("%d%d", &n, &m);
	for(int i = 0; i < n; i++)
		scanf("%s", g[i]);
	
	for(int i = 0; i < 26; i++)
	{
		for(int j = 0; j < 26; j++)
		{
			if(i == j)	continue;
			for(int k = 0; k < m; k++)
				if(k%2 == 0 && g[0][k] != 'a' + i)
					dp[0][i][j]++;
				else if(k%2 == 1 && g[0][k] != 'a' + j)
					dp[0][i][j]++;
		}
	}
	for(int l = 1; l < n; l++)
	{
		for(int i = 0; i < 26; i++)
		{
			for(int j = 0; j < 26; j++)
			{
				if(i == j)	continue;
				int change = 0;
				for(int k = 0; k < m; k++)
					if(k%2 == 0 && g[l][k] != 'a' + i)
						change++;
					else if(k%2 == 1 && g[l][k]!= 'a' + j)
						change++;
				
				int tmp = 10000000;
				for(int ii = 0; ii < 26; ii++)
				{
					for(int jj = 0; jj < 26; jj++)
					{
						if(ii == jj)	continue;
						if(i == ii || j == jj)	continue;
						if(tmp > dp[l-1][ii][jj])
						{
							tmp = dp[l-1][ii][jj];
							ans[l][i][j][0] = ii;
							ans[l][i][j][1] = jj;
						}
					}
				}
				dp[l][i][j] = change + tmp;
			}
		}
	}
	int tmp = 100000000, a, b;
	for(int i = 0; i < 26; i++)
		for(int j = 0; j < 26; j++)
		{
			if(i == j)	continue;
			if(tmp > dp[n-1][i][j])
			{
				tmp = dp[n-1][i][j];
				a = i, b = j;
			}
		}
	printf("%d\n", tmp);
	print(n-1, a, b);
	return  0;
}
