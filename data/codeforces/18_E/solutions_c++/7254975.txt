#include<cstdio>
#include<algorithm>
using namespace std;
int dp[501][26][26][2],n,m,sv1,sv2,sv3,ans[501];
char s[503][503];
void P(int n,int j,int k){
  if(n == 0) return ;
  P(n-1,dp[n][j][k][1]/26,dp[n][j][k][1]%26);
  for(int i=0;i<m;i++)
    printf("%c", (i&1) == 1 ? k+'a' : j + 'a');
  printf("\n");
}
int main(){
  scanf("%d%d",&n,&m);
  for(int i=1;i<=n;i++) scanf("%s",s[i]);
  for(int i=1;i<=n;i++){
    sv1 = sv2 = sv3 = -1;
    for(int j=0;j<26;j++)
      for(int k=0;k<26;k++)
    if(j != k && (sv1 == -1 || dp[i-1][sv1/26][sv1%26][0] > dp[i-1][j][k][0]))
      sv1 = j*26+k;
    for(int j=0;j<26;j++)
      for(int k=0;k<26;k++){
    if(j != k && sv1/26 != j && sv1 % 26 != k){
      dp[i][j][k][1] = sv1;
      dp[i][j][k][0] = dp[i-1][sv1/26][sv1%26][0];
    }
    else if(j != k){
      int sv4 = -1;
      for(int r1=0;r1<26;r1++)
        for(int r2=0;r2<26;r2++)
          if(r1 != r2 && r1 != j && r2 != k && (sv4 == -1 || dp[i-1][sv4/26][sv4%26][0] > dp[i-1][r1][r2][0]))
        sv4 = r1*26+r2;
      dp[i][j][k][1] = sv4;
      dp[i][j][k][0] = dp[i-1][sv4/26][sv4%26][0];
    }
      }
    for(int j=0;j<26;j++)
      for(int k=0;k<26;k++)
    if(j != k){
      for(int r1=0;r1<m;r1++){
        if((r1&1) == 0 && s[i][r1] != j+'a') dp[i][j][k][0]++;
        else if((r1&1) == 1 && s[i][r1] != k+'a') dp[i][j][k][0]++;
      }
    }
  }
  sv1 = sv2 = -1;
  for(int j=0;j<26;j++)
    for(int k=0;k<26;k++)
      if(j != k && (sv1 == -1 || dp[n][j][k][0] < dp[n][sv1][sv2][0]))
    sv1 = j, sv2 = k;
  printf("%d\n",dp[n][sv1][sv2][0]);
  P(n,sv1,sv2);
  return 0;
}
