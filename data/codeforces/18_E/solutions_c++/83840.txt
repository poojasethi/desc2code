#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;

#define N 501
const int oo=(int)1e9;
int n,m,cost[N][26][26],dp[N][26][26],ans;
short w[N][26][26][2];
char s[N][N];

int main(){
	scanf("%d%d",&n,&m);
	register short i,j,k,l,x;
	for (i=0;i<n;i++){
		scanf("%s",s[i]);
		for (j=0;j<m;j++){
			s[i][j]-='a';
		}
	}
	for (i=0;i<n;i++){
		for (j=0;j<26;j++){
			for (k=0;k<26;k++){
				if (j==k) continue;
				int &cst=cost[i][j][k];
				cst=0;
				for (l=0;l<m;l++){
					if (l&1){
						if (k!=s[i][l]) cst++;
					}
					else {
						if (j!=s[i][l]) cst++;
					}
				}
			}
		}
	}
	for (j=0;j<26;j++){
		for (k=0;k<26;k++){
			if (j==k) continue;
			dp[0][j][k]=cost[0][j][k];
		}
	}
	for (i=1;i<n;i++){
		for (j=0;j<26;j++){
			for (k=0;k<26;k++){
				if (j==k) continue;
				int &res=dp[i][j][k];
				res=oo;
				for (l=0;l<26;l++){
					if (l==j) continue;
					for (x=0;x<26;x++){
						if (l==x) continue;
						if (k==x) continue;
						if (res>dp[i-1][l][x]){
							res=dp[i-1][l][x];
							w[i][j][k][0]=l;
							w[i][j][k][1]=x;
						}
					}
				}
				res+=cost[i][j][k];
			}
		}
	}
	ans=oo;
	short now,tx,ty,nx,ny;
	for (i=0;i<26;i++){
		for (j=0;j<26;j++){
			if (i==j) continue;
			if (ans>dp[n-1][i][j]){
				ans=dp[n-1][i][j];
				nx=i;ny=j;
			}
		}
	}
	printf("%d\n",ans);
	now=n;
	while (1){
		for (int i=0;i<m;i++){
			if (i&1) s[now-1][i]=ny+'a';
			else s[now-1][i]=nx+'a';
		}
		now--;
		if (now==-1) break;
		tx=w[now][nx][ny][0];
		ty=w[now][nx][ny][1];
		nx=tx;ny=ty;
	}
	for (int i=0;i<n;i++){
		printf("%s\n",s[i]);
	}
	//system("pause");
	return 0;
}
