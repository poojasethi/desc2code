int n,m,cnt,dp[N][28][28];
pii go[N][29][28];
string str;
pair<int,pii > ans;
void yaz_kizim(int kalan,int x,int y){
	if(!kalan) return;
	yaz_kizim(kalan-1,go[kalan][x][y].ft , go[kalan][x][y].sd);
	FOR(i,0,m-1) printf("%c",(i&1 ? y :x)+'a');
	puts("");
}
int main(){
	cin >> n >> m;
	memset(dp,4,sizeof dp);
	FOR(i,0,25) FOR(j,0,25) dp[0][i][j] = 0;
	FOR(i,1,n){
		cin >> str;
		FOR(j,0,25)
			FOR(k,0,25){
				if(j==k) continue;
				cnt = 0;
				for(int l=0;l<m;l+=2) cnt += (j!=str[l]-'a');
				for(int l=1;l<m;l+=2) cnt += (k!=str[l]-'a');
				FOR(x,0,25)
					FOR(y,0,25){
						if(x==j or y==k or x==y) continue;
						if(dp[i-1][x][y] + cnt < dp[i][j][k]){
							dp[i][j][k] = dp[i-1][x][y] + cnt;
							go[i][j][k] = mp(x,y);
						}
					}
			}
	}
	ans.ft = inf;
	FOR(i,0,25)
		FOR(j,0,25)
			if(i!=j)
				ans = min(ans , mp(dp[n][i][j] , mp(i,j)));
	cout << ans.ft << endl;
	yaz_kizim(n,ans.sd.ft,ans.sd.sd);
}
