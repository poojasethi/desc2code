#include<bits/stdc++.h>

using namespace std;

string s;
int n,m,k,i,j,stx,sty,ix[550][2];
int x,y,ned[2][26],res=1000000000;
int ans[550][26][26];
int frx[550][26][26];
int fry[550][26][26];
int need[550][26][26];

main()
{
 cin>>n>>m;
 for (k=1;k<=n;k++)
  {
   cin>>s;
   for (j=0;j<26;j++) ned[0][j]=ned[1][j]=0;
   for (i=0;i<m;i++)
    for (j=0;j<26;j++)
     if (s[i] != j+'a') ned[(i%2)][j]++;
   for (i=0;i<26;i++)
    for (j=0;j<26;j++)
     if (i != j)
      need[k][i][j]=ned[0][i]+ned[1][j],ans[k][i][j]=1000000000;
  }

 for (k=1;k<=n;k++)
  for (i=0;i<26;i++)
   for (j=0;j<26;j++)
    for (x=0;x<26;x++)
     for (y=0;y<26;y++)
      if (i != j && i != x && x != y && j != y)
       if (ans[k][i][j] > ans[k-1][x][y]+need[k][i][j])
        {
         frx[k][i][j]=x;
         fry[k][i][j]=y;
         ans[k][i][j]=ans[k-1][x][y]+need[k][i][j];
        }

 for (i=0;i<26;i++)
  for (j=0;j<26;j++)
   if (i != j && res > ans[n][i][j])
    {
     res=ans[n][i][j];
     stx=i;
     sty=j;
    }

 i=n;
 while (n)
  {
   ix[n][0]=stx;
   ix[n][1]=sty;
   int sx=stx,sy=sty;
   stx=frx[n][sx][sy];
   sty=fry[n][sx][sy];
   n--;
  }
 n=i;

 cout<<res<<endl;
 for (i=1;i<=n;i++)
  {
   for (j=0;j<m;j++)
    cout<<char(ix[i][j%2]+'a');
   cout<<endl;
  }
}
