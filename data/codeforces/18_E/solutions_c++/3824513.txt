#include <cstdio>
#include <algorithm>
using namespace std;

#define inf 1000000000
#define sigma 26
#define maxn 505

int n,m,k,x,y,i,j,num_odd,num_even,kol[2][sigma],Tr[maxn][sigma][sigma],F[maxn][sigma][sigma],Px[maxn][sigma][sigma],Py[maxn][sigma][sigma],opt,id1,id2;
char s[maxn],ans[maxn][maxn];

int main (int argc, char * const argv[]) {
//	freopen("in.txt","r",stdin);
	scanf("%d%d",&n,&m);gets(s);
	for(i=1;i<=n;i++){
		gets(s);
		num_odd=0,num_even=0;
		for(j=0;j<2;j++)for(k=0;k<26;k++)kol[j][k]=0;
		for(j=0;j<m;j++){
			if(j%2)++num_odd;else ++num_even;
			++kol[j%2][s[j]-'a'];
		}
		for(j=0;j<26;j++)for(k=0;k<26;k++){
			Tr[i][j][k]=num_odd-kol[1][j]+num_even-kol[0][k];
			F[i][j][k]=inf;
		}
	}
	for(j=0;j<26;j++)for(k=0;k<26;k++)if(j!=k)F[1][j][k]=Tr[1][j][k];
	for(i=1;i<n;i++)for(j=0;j<26;j++)for(k=0;k<26;k++)if(F[i][j][k]!=inf)
		for(x=0;x<26;x++)for(y=0;y<26;y++)if(x!=j&&y!=k&&x!=y&&F[i+1][x][y]>F[i][j][k]+Tr[i+1][x][y]){
			F[i+1][x][y]=F[i][j][k]+Tr[i+1][x][y];
			Px[i+1][x][y]=j;
			Py[i+1][x][y]=k;
		}
	opt=inf;
	for(j=0;j<26;j++)for(k=0;k<26;k++)if(F[n][j][k]<opt){
		opt=F[n][j][k];
		id1=j;
		id2=k;
	}
	printf("%d\n",opt);
	for(i=n;i;i--){
		for(j=0;j<m;j++)if(j%2)ans[i][j]='a'+id1;else ans[i][j]='a'+id2;
		x=Px[i][id1][id2];
		y=Py[i][id1][id2];
		id1=x;
		id2=y;
	}
	for(i=1;i<=n;i++){
		for(j=0;j<m;j++)putchar(ans[i][j]);
		puts("");
	}
    return 0;
}
