#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
const int i=676;
int f[500][i],s[500][i];
int g[i],g1[i];
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,c,v,d[26],d1[26],d2[26],ans[500];
  char h[501];
  for(e=0;e<i;e++){
   g[e]=e/26;
   g1[e]=e%26;}
  scanf("%d%d",&q,&w);
  for(e=0;e<q;e++){
   scanf("%s",h);
   for(r=0;r<w;r++)
    h[r]-=97;
   for(r=0;r<i;r++){
    if(g[r]==g1[r]){
     f[e][r]=1<<20;
     continue;}
    f[e][r]=0;
    for(c=0;c<w;c+=2)
     if(h[c]!=g[r])
      f[e][r]++;
    for(c=1;c<w;c+=2)
     if(h[c]!=g1[r])
      f[e][r]++;}}
  for(r=0;r<i;r++)
   s[0][r]=f[0][r];
  for(e=1;e<q;e++){
   for(r=0;r<26;r++)
    d[r]=d2[r]=1<<20;
   for(r=0;r<i;r++){
    c=g[r];
    v=g1[r];
    if(d2[c]>s[e-1][r]){
     d2[c]=s[e-1][r];
     if(d[c]>d2[c]){
      d2[c]=d[c];
      d[c]=s[e-1][r];
      d1[c]=v;}}}
   for(r=0;r<i;r++){
    s[e][r]=1<<20;
    for(c=0;c<26;c++)
     if(c!=g[r])
      if(d1[c]!=g1[r])
       s[e][r]=min(s[e][r],d[c]);
      else
       s[e][r]=min(s[e][r],d2[c]);
    s[e][r]+=f[e][r];}}
  t=0;
  for(r=1;r<i;r++)
   if(s[q-1][t]>s[q-1][r])
    t=r;
  printf("%d\n",s[q-1][t]);
  ans[q-1]=t;
  for(e=q-1;e>0;e--){
   c=1<<20;
   for(r=0;r<i;r++)
    if((g[t]!=g[r])&&(g1[t]!=g1[r]))
     if(c>s[e-1][r]){
      c=s[e-1][r];
      v=r;}
   ans[e-1]=t=v;}
  for(e=0;e<q;e++){
   for(r=0;r<w;r++)
    if((r&1)==0)
     printf("%c",((char)(g[ans[e]]+97)));
    else
     printf("%c",((char)(g1[ans[e]]+97)));
   printf("\n");}
  return 0;}
