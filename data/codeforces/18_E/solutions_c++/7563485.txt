#include <cstdio>
#include <algorithm>
using namespace std;
char Flag[502][502];
int f[502][26][26];
pair<int,int> g[502][26][26],opt;
const int INF=~0U>>2;
char Out[502][502];
int n,m;
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++)
	{
		scanf("%s",Flag[i]);
		for (int j=0;j<m;j++) Flag[i][j]-='a';
	}
	for (int i=0;i<n;i++)
		for (int j=0;j<26;j++)
			for (int k=0;k<26;k++) f[i][j][k]=INF;
	for (int i=0;i<26;i++)
		for (int j=0;j<26;j++) 
		if(i!=j)
		{
			int k=0;
			for (int p=0;p<m;p++)
			{
				if (p&1)
				{
					if (Flag[0][p]!=j) k++;
				}
				else if (Flag[0][p]!=i) k++;
			}
			f[0][i][j]=k;
		}
	for (int i=1;i<n;i++)
		for (int j=0;j<26;j++)
			for (int k=0;k<26;k++)
			if (j!=k)
			{
				int kk=0;
				for (int p=0;p<m;p++)
				{
					if (p&1)
					{
						if (Flag[i][p]!=k) kk++;
					}
					else if (Flag[i][p]!=j) kk++;
				}
				for (int p=0;p<26;p++)
					if (p!=j)
					for (int q=0;q<26;q++)
					if (p!=q && q!=k)
						if (f[i-1][p][q]<f[i][j][k]) g[i][j][k]=pair<int,int>(p,q),f[i][j][k]=f[i-1][p][q];
				f[i][j][k]+=kk;
			}
	int ans=INF;
	for (int i=0;i<26;i++)
		for (int j=0;j<26;j++)
			if (f[n-1][i][j]<ans) ans=f[n-1][i][j],opt=pair<int,int>(i,j);
	printf("%d\n",ans);
	int ff=n-1;
	while (~ff)
	{
		for (int i=0;i<m;i++)
		{
			if (i&1) Out[ff][i]='a'+opt.second;
			else Out[ff][i]='a'+opt.first;
		}
		opt=g[ff][opt.first][opt.second];
		ff--;
	}
	for (int i=0;i<n;i++) puts(Out[i]);
	return 0;
}
