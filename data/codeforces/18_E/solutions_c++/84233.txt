#include <iostream>
#include <cmath>
#include <string>
#include <cstring>
#include <cstdio>

using namespace std;

int n,m;
int f[501][26][26][2];
char flag[501];

void output(int l, int p)
{
    if(0==l) return; 
    int x=p/100, y=p%100;   
    int next=f[l][x][y][1];
    output(l-1, next);
    for(int i=0; i<m; ++i)
        if(i&1) printf("%c", 'a'+y);
        else printf("%c", 'a'+x);
    printf("\n");
    return ;  
}

int main()
{
    scanf("%d %d", &n, &m);
    
    int i,j,k,l,o,cost;
    for(i=1; i<=n; ++i){
        scanf("%s", flag);
        for(k=0; k<26; ++k){
            for(l=0; l<26; ++l){
                f[i][k][l][0]=0x7f7f7f7f;
                if(l==k) continue;
                cost=0;
                for(j=0; j<m; ++j) 
                    if(((j&1) && flag[j]!='a'+l) || (!(j&1) && flag[j]!='a'+k)) ++cost;
                for(j=0; j<26; ++j){
                    if(j==k) continue;
                    for(o=0; o<26; ++o){
                        if(o==l || o==j) continue;
                        if(f[i][k][l][0]>f[i-1][j][o][0]+cost){
                            f[i][k][l][0]=f[i-1][j][o][0]+cost;
                            f[i][k][l][1]=j*100+o;
                        } 
                    }
                }    
            }         
        }    
    }
    
    k=o=0x7f7f7f7f;
    for(i=0; i<26; ++i)
        for(j=0; j<26; ++j)
            if(f[n][i][j][0]<o) o=f[n][i][j][0],k=i*100+j;
    printf("%d\n", o);
    output(n,k);    
    return 0;    
}
