#include <cstdio>
#include <string>
#include <cassert>
#include <cstring>

using namespace std;

const int maxn=533;
const int maxk=26;
const int oo= 1000*1000*1000;
char s[maxn][maxn];

int dist[maxn][maxk][maxk];

int dp[maxn][maxk][maxk];
char pf[maxn][maxk][maxk];
char ps[maxn][maxk][maxk];
char res[maxn][maxn];

int main(){
	int n,m;
	int i,j,k;
	char a,b,c,d;
	scanf("%d%d\n",&n,&m);
	for(i=0;i<n;i++){
		gets(s[i]);
	}
	memset(dist,0,sizeof dist);
	for(i=0;i<n;i++){
		for(a='a';a<='z';a++){
			for(b='a';b<='z';b++){
				dp[i][a][b]=oo;
				for(j=0;j<m;j++){
					if(j&1){
						dist[i][a][b]+=(s[i][j]!=b);
					}else{
						dist[i][a][b]+=(s[i][j]!=a);
					}
					if(a==b) dist[i][a][b]=oo;
				}
			}
		}
	}
	for(a='a';a<='z';a++){
		for(b='a';b<='z';b++){
			dp[0][a][b]=dist[0][a][b];
		}
	}
	for(i=0;i<n-1;i++){
		for(a='a';a<='z';a++){
			for(b='a';b<='z';b++){
				for(c='a';c<='z';c++){
					for(d='a';d<='z';d++){
						if(a!=c&&b!=d){
							if(dp[i+1][c][d]>dp[i][a][b]+dist[i+1][c][d]){
								pf[i+1][c][d]=a;
								ps[i+1][c][d]=b;
								dp[i+1][c][d]=dp[i][a][b]+dist[i+1][c][d];
							}
						}
					}
				}
			}
		}
	}
	char ca='a',cb='a';
	for(a='a';a<='z';a++){
		for(b='a';b<='z';b++){
			if(dp[n-1][ca][cb]>dp[n-1][a][b]){
				ca=a;
				cb=b;
			}
		}
	}

	char ta,tb;
	printf("%d\n",dp[n-1][ca][cb]);
	for(i=n-1;i>=0;i--){
		for(j=0;j<m;j++){
			if(j&1){
				res[i][j]=cb;
			}else{
				res[i][j]=ca;
			}
		}
		ta=pf[i][ca][cb];
		tb=ps[i][ca][cb];
		ca=ta;
		cb=tb;
	}
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			printf("%c",res[i][j]);
		}
		printf("\n");
	}
	return 0;
}
