#include <bits/stdc++.h>
using namespace std;
#define rep(i,x,y) for((i)=(x);(i)<=(y);++(i))
#define inf 1e9
int ri,rj,rk,ci,cj,ck,cn,n,m,i,j,k,jj,z,kk,an,a[555][555],c[555][28][28],f[555][28][28],lx[555][28][28],ly[555][28][28],lz[555][28][28];
char s[555];
string st[555];
string Construct(int a,int b){
	string rt;rt.clear();
	for(int i=1;i<=m;++i)rt+=(char)('a'+((i&1)?a:b)-1);
	return rt;
}
int main(){
	//freopen("input.txt","r",stdin);
	scanf("%d%d",&n,&m);
	rep(i,1,n){
		scanf("%s",s+1);
		rep(j,1,m)a[i][j]=s[j]-'a'+1;
	}	
	rep(i,1,n)
		rep(j,1,26)rep(k,1,26){
			c[i][j][k]=0;
			rep(z,1,m)c[i][j][k]+=(z&1)*(j!=a[i][z])+(1-(z&1))*(k!=a[i][z]);
		}
	rep(i,1,n)rep(j,1,26)c[i][j][j]=inf;
	rep(i,1,n)rep(j,1,26)rep(k,1,26)f[i][j][k]=inf;
	rep(j,1,26)rep(k,1,26)f[1][j][k]=c[1][j][k];
	rep(i,1,n-1)rep(j,1,26)rep(k,1,26)
		rep(jj,1,26)rep(kk,1,26)
			if(j!=jj&&k!=kk&j!=k&jj!=kk)
				if(f[i][j][k]+c[i+1][jj][kk]<f[i+1][jj][kk]){
					f[i+1][jj][kk]=f[i][j][k]+c[i+1][jj][kk];
					lx[i+1][jj][kk]=i,ly[i+1][jj][kk]=j,lz[i+1][jj][kk]=k;
				}
	an=inf;
	rep(j,1,26)rep(k,1,26)
		if(f[n][j][k]<an){
			an=f[n][j][k];
			ci=n,cj=j,ck=k;
		}
	cn=0;
	while(1){
		if(!ci||!cj||!ck)break;
		st[++cn]=Construct(cj,ck);
		ri=ci,rj=cj,rk=ck;
		ci=lx[ri][rj][rk],cj=ly[ri][rj][rk],ck=lz[ri][rj][rk];
	}
	printf("%d\n",an);
	for(i=cn;i>=1;--i)cout<<st[i]<<endl;
	return 0;
}
