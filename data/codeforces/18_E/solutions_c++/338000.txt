#define LL long long

using namespace std;

int main (){
	int n,m; cin>>n>>m;
	char s[n][m+1];
	for(int i=0;i<n;i++) cin>>s[i];
	
	int dp[n][26][26],ans=INT_MAX;
	int x[n][26][26],y[n][26][26];


	for(int i=0;i<n;i++)
		for(int j=0;j<26;j++) 
			for(int k=0;k<26;k++)if( j!=k ){
				int d=0;
				for(int l=0;l<m;l++)
				   if( l%2 ) d=(s[i][l]-'a' == j )?d:d+1;
				   else if( l%2==0 ) d=(s[i][l]-'a' == k )?d: d+1;
				
				dp[i][j][k]=INT_MAX;
				
				if( !i ){ dp[i][j][k]=d; x[i][j][k]=y[i][j][k]=-1; }
				else 
				for(int jj=0;jj<26;jj++)
					for(int kk=0;kk<26;kk++)
						if(jj!=kk && j!=jj && k!=kk && dp[i][j][k] > d+ dp[i-1][jj][kk] ){ 
							dp[i][j][k]=d+dp[i-1][jj][kk];
							x[i][j][k]=jj; y[i][j][k]=kk;
						}
				
				
				
				//if( i+1==n ) ans=min(ans,dp[i][j][k]);
			}
	
	int px,py; px=py=-1;
	
	for(int j=0;j<26;j++) 
			for(int k=0;k<26;k++)if( j!=k && ans > dp[n-1][j][k] ){
				ans=dp[n-1][j][k]; px=j; py=k;
			}
	cout<<ans<<endl;
	
	int i=n-1;
	while( i>=0 ){
	
		for(int j=0;j<m;j++)
			if( j%2 ) s[i][j]=px+'a';
			else s[i][j]=py+'a';
			
		int pxx=x[i][px][py];
		py=y[i][px][py];
		px=pxx; i--;
	}
	for(int i=0;i<n;i++) cout<<s[i]<<endl;
	return 0;
}
