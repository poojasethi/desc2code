#include <stdio.h>

char map[501][501];
int a[501][30][30];
int r[501][30][30];
int res1[501][30][30];
int res2[501][30][30];

int main()
{
	int n,m;
	scanf("%d%d", &n,&m);

	for(int i=0;i<n;i++)
		scanf("%s", map[i]);

	for(int i=0;i<n;i++)
		for(int c1=0;c1<26;c1++)
			for(int c2=0;c2<26;c2++)
				if(c1!=c2)
				for(int j=0;j<m;j++)
					if(j%2==0 && (map[i][j]-'a')!=c1 || j%2==1 && (map[i][j]-'a')!=c2)
						a[i][c1][c2]++;

	for(int i=0;i<n;i++)
	for(int c1=0;c1<26;c1++)
		for(int c2=0;c2<26;c2++)
			if(c1!=c2)
			if(i==0) r[0][c1][c2]=a[0][c1][c2];
			else r[i][c1][c2]=-1;

	for(int i=1;i<n;i++)
		for(int c1=0;c1<26;c1++)
			for(int c2=0;c2<26;c2++)
				if(c1!=c2)
				for(int c10=0;c10<26;c10++)
					for(int c20=0;c20<26;c20++)
					{
						if(c10!=c20)
						if(c1!=c10 && c2!=c20)
						if(r[i][c1][c2]==-1 || r[i-1][c10][c20]+a[i][c1][c2]<r[i][c1][c2])   {
							r[i][c1][c2]=r[i-1][c10][c20]+a[i][c1][c2];
							res1[i][c1][c2]=c10;
							res2[i][c1][c2]=c20;
						}
					}
	int r1=-1,r2;
	for(int c1=0;c1<26;c1++)
		for(int c2=0;c2<26;c2++)
			if(c1!=c2)
			if(r[n-1][c1][c2]!=-1 && ( r1==-1 || r[n-1][r1][r2]>r[n-1][c1][c2] ) )   {
				r1=c1;r2=c2;
			}

	printf("%d\n",r[n-1][r1][r2]);
	for(int i=n-1;i>=0;i--)
	{
		for(int j=0;j<m;j++)
			if(j%2==0) map[i][j]=r1+'a';
			else map[i][j]=r2+'a';
		//map[i][m]=0;
		int k=r1;
		r1=res1[i][r1][r2];
		r2=res2[i][k][r2];
	}

	for(int i=0;i<n;i++)
		printf("%s\n",map[i]);

	return 0;
}
