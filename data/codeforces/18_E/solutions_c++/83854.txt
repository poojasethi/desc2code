#include <iostream>
#include <cstdio>
#include <utility>
#define max_val 100000000
using namespace std;
char inp[505][505];
int n,m;//n riada po m kolon
int min_dif(int p,char C,char N){
  int ans=0,i;
  for(i=0;i<m;i+=2)if(inp[p][i]!=C)ans++;
  for(i=1;i<m;i+=2)if(inp[p][i]!=N)ans++;
  return ans;
}
int cf[501][26][26];
pair<int,int> move[501][26][26];
int old_CF[26][26];
void calc(int I,int C,int N){
  int D=min_dif(I,C,N),i,j;
  cf[I][C][N]=max_val;
  if(C==N)return;
  int tmp=max_val;
  for(i=0;i<26;++i)
   for(j=0;j<26;++j)
    if(i!=j && i!=C && j!=N)
     if(tmp>old_CF[i][j]){tmp=old_CF[i][j];move[I][C][N]=pair<int,int>(i,j);}
  cf[I][C][N]=tmp+D;
}
int main(){
  int i,j,k,l;
  //system("pause");
  scanf("%d%d",&n,&m);
  for(i=0;i<n;i++)scanf("%s",inp[i]);
  for(i=0;i<n;i++)for(j=0;j<m;j++)inp[i][j]-='a';
  for(i=0;i<26;i++)
   for(j=0;j<26;j++){
    if(i==j)cf[n-1][i][j]=max_val;
    else cf[n-1][i][j]=min_dif(n-1,i,j);
                    }
  for(i=n-2;i>=0;--i){
   for(k=0;k<26;k++)
    for(l=0;l<26;l++)old_CF[k][l]=cf[i+1][k][l];
   for(k=0;k<26;k++)
    for(l=0;l<26;l++)calc(i,k,l);
                     }
  int ans=max_val;
  pair<int,int> fp;
  for(i=0;i<26;i++)
   for(j=0;j<26;j++)
    if(ans>cf[0][i][j]){ans=cf[0][i][j];fp=pair<int,int>(i,j);}
  printf("%d\n",ans);
  for(i=0;i<n;i++){
   for(j=0;j<m;j++)
    if(j%2==0)printf("%c",(char)('a'+fp.first));
    else printf("%c",(char)('a'+fp.second));
    printf("\n");
    fp=move[i][fp.first][fp.second];
                  }
  //system("pause");
  return 0;
}
