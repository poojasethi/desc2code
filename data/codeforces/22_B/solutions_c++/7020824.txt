#include<iostream>
#include<cstdio>
using namespace std;
bool a[30][30];
int main()
{
	int x,y,ans=0;
	cin>>x>>y;
	for(int i=1;i<=x;i++)for(int j=0;j<=y;j++)if(getchar()=='1')a[i][j]=1;
	for(int i=1;i<=x;i++)for(int j=1;j<=y;j++)
		for(int k=i;k<=x;k++)for(int l=j;l<=y;l++)
		{
			bool w=true;
			for(int m=i;m<=k&&w;m++)for(int n=j;n<=l&&w;n++)w=!a[m][n];
			if(w)ans=max(ans,(k+l-i-j)*2+4);
		}
	cout<<ans<<endl;
}
