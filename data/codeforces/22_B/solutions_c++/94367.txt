#include <cstdio>
#include <iostream>

using namespace std;
char s[100][100];
int main(){
	int i,j,k,l,u,v;
	int n,m;
	int g;
	int res=0;
	scanf("%d %d\n",&n,&m);
	for(i=0;i<n;i++)
		//for(j=0;j<m;j++)
			gets(s[i]);
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			for(k=i+1;k<=n;k++){
				for(l=j+1;l<=m;l++){
					g=1;
					for(u=i;u<k;u++)
						for(v=j;v<l;v++) g&=(s[u][v]=='0');
					if(g) res=max(res,2*(k-i)+2*(l-j));
				}
			}
		}
	}
	cout << res << endl;
	return 0;
}
