#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
    int n,m,s[26][26]={};
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) for(int j=1;j<=m;j++){
        scanf("%1d",&s[i][j]);
        s[i][j]+=s[i-1][j]+s[i][j-1]-s[i-1][j-1];
    }
    int ans=0;
    for(int a=1;a<=n;a++) for(int b=1;b<=m;b++)
        for(int p=a;p<=n;p++) for(int q=b;q<=m;q++)
            if(s[p][q]-s[a-1][q]-s[p][b-1]+s[a-1][b-1]==0)
                ans=max(ans,(p+q-a-b+2)*2);
    printf("%d\n",ans);
}
