#include <iostream>
#include <cstdio>
int f[30][30][30],i,j,k,n,m,x,ans;
char c;
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d%d\n",&n,&m);
	for (i=1;i<=m;++i) f[0][i][0]=i;
	for (i=1;i<=n;++i,scanf("\n"))
		for (x=0,j=1;j<=m;++j)
		{
			f[i][j][0]=j;
			scanf("%c",&c);
			if (c=='1') x=j;
			for (k=1;k<=i;++k)
			{
				f[i][j][k]=std::min(f[i-1][j][k-1],j-x);
				if (f[i][j][k]) ans=std::max(ans,k+f[i][j][k]);
			}
		}
	printf("%d\n",ans<<1);
}
