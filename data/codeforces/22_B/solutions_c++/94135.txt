#include <iostream>

using namespace std;

int main ()
{
	int m, n; cin >> n >> m;
	bool arr[30][30];
	for(int i = 0; i < n; i++)
		for(int j = 0; j < m; j++)
		{
			char c; cin >> c;
			arr[i][j] = c - '0';
		}
	
	int res = 0;
	for(int i = 0; i < n; i++)
		for(int j = 0; j < m; j++)
		{
			int a = m, b = 0;
			for(int x = i; x < n && !arr[x][j]; x++)
			{
				int k = 0;
				b++;
				for(int y = j; y < m && !arr[x][y]; y++) k++;
				a = min(a, k);
				if(2 * (a + b) > res) res = 2 * (a + b);
			}
		}
		
	cout << res << endl;
}
