#include<bits/stdc++.h>
#define FOR(i, a, b) for(int i=a ; i<=b ; ++i)
using namespace std;
int n, m, sum, ans;
char s[27][27];
int main(int argc, char **argv) {
	scanf("%d%d", &n, &m);
	FOR(i,1,n) scanf("%s", s[i] + 1);
	FOR(i,1,n) FOR(j,1,m) s[i][j] -= '0';
	FOR(i,1,n) FOR(j,1,m) FOR(a,1,n) FOR(b,1,m){
		sum = 0;
		FOR(x, min(a, i), max(a,i)) FOR(y, min(j,b), max(j,b)) sum += s[x][y];
		if(sum == 0) ans = max(ans, 2*(abs(a-i)+1 + abs(b-j)+1));
	}
	printf("%d\n", ans);
	return 0;
}
