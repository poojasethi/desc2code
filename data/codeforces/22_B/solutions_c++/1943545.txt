#include <iostream>

using namespace std;

int main()
{
	int f[50][50],m,n,imax,t;
	string st;
	cin >> m >> n;
	for(int i=1;i<=m;i++)
	{
		cin >> st;
		for(int j=1;j<=n;j++)
		{
			f[i][j]=st[j-1]-'0';
			f[i][j]+=f[i-1][j]+f[i][j-1]-f[i-1][j-1];
		}
	}
	
	imax=0;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=n;j++)
			for(int x=i;x<=m;x++)
				for(int y=j;y<=n;y++)
				{
					if(f[x][y]-f[i-1][y]-f[x][j-1]+f[i-1][j-1]==0)
					{
						t=2*(x-i+1+y-j+1);
						if(t>imax)
						imax=t;
					}	
				}
	cout << imax << endl;

	return 0;
}
