#include <iostream>
#include<stdio.h>
using namespace std;
int ans=0;
int map[30][30];
int i,ii,n,m,j,jj;
bool check(int x,int y,int xx,int yy){
	int i,j;
	for (i=x;i<=xx;i++)
		for (j=y;j<=yy;j++) if (map[i][j]==1) return false;
	return true;
}
int main() {
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
		for (j=1;j<=m;j++) scanf("%1d",&map[i][j]);
	for (i=1;i<=n;i++)
		for (j=1;j<=m;j++)
			for (ii=i;ii<=n;ii++)
				for (jj=j;jj<=m;jj++)
					if (check(i,j,ii,jj)) ans=max(ans,2*(jj-j+ii-i+2));
	printf("%d\n",ans);
	return 0;
}
