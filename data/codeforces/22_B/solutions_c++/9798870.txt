#include<bits/stdc++.h>
using namespace std;
char ma[30][30];
int n,m;
int cal(int x,int y)
{
    int i,j,a=30,b,l=0;
    for(i=x;i<=n;++i)
    {
        for(j=y;j<=m&&ma[i][j]!='1';++j);
        if(j==y) break;
        b=i-x+1;
        a=min(a,j-y);
        l=max(l,a+a+b+b);
    }
    return l;
}

int main()
{
    cin>>n>>m;
    int i,j;
    for(i=1;i<=n;++i) scanf("%s",ma[i]+1);
    int ans=0;
    for(i=1;i<=n;++i)
        for(j=1;j<=m;++j)
            ans=max(ans,cal(i,j));
    printf("%d\n",ans);
    return 0;
}
