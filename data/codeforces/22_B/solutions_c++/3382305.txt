#include <cstdio>
#include <iostream>

using namespace std;

const int MAXN=30;

int N,M,res;
int sum[MAXN][MAXN];

int main(){
	
	scanf(" %d %d",&N,&M);
	
	char ch;
	
	for( int i=1 ; i<=N ; i++ )
		for( int j=1 ; j<=M ; j++ ){
			
			scanf(" %c",&ch);
			sum[i][j]=ch-'0';
			
			sum[i][j]+=sum[i-1][j]+sum[i][j-1]-sum[i-1][j-1];
			
			for( int k=1 ; k<=i ; k++ )
				for( int l=1 ; l<=j ; l++ )
					if( sum[i][j]-sum[i][l-1]-sum[k-1][j]+sum[k-1][l-1]==0 ){
						res=max(res,(i-k+1)*2+(j-l+1)*2);
						break;
					}
		}
	
	printf("%d\n",res);
	return 0;
}
