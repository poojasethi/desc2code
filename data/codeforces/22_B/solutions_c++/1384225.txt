#include<stdio.h>
int min(int a,int b)
{
	return a<b?a:b;
}
int max(int a,int b)
{
	return a>b?a:b;
}	
int s[30][30],ch[30][30];
int main()
{
	int n,m,i,j,k,ans,c;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;++i)
		for(j=0;j<m;++j){
			scanf("%1d",&s[i][j]);
		}
		for(i=0;i<n;++i)
			for(j=0;j<m;++j){
				if(s[i][j]==0)
					ch[i][j]=ch[i-1][j]+1;
				else
					ch[i][j]=0;
			}
			ans=0;
			for(i=0;i<n;++i){
				for(j=0;j<m;++j){
					c=n;
					for(k=0;j+k<m;++k){
						c=min(c,ch[i][j+k]);
						if(c) ans=max(ans,c+k+1);
					}
					
				}
			}
			printf("%d\n",2*ans);

}
