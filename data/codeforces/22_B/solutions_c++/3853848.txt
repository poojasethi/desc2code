# include <stdio.h>
# include <algorithm>
using namespace std;
int a,b,mx;
int c[29][29];
int main()
{
	scanf("%d %d",&a,&b);
	for(int i=1;i<=a;i++)
	for(int j=1;j<=b;j++)
	scanf("%1d",&c[i][j]);
	
	for(int i=1;i<=a;i++)
	for(int j=1;j<=b;j++)
	{
		int mn=a,z=0;
		for(int l=i;l<=a;l++)
		{
			if(c[l][j]==1)break;
			for(int k=j;k<=b;k++)
			if(c[l][k]==1)
			{mn=min(mn,k-j);z=1;break;}
			if(z==0) mn=b-j+1;
			mx=max(mx,2*(mn+(l-i+1)));
		}
	}
	printf("%d",mx);
}
