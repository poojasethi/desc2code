#include <bits/stdc++.h>
using namespace std;
int dp[30][30];
char a[30][300];
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
			dp[i+1][j+1] = dp[i+1][j] + dp[i][j+1] - dp[i][j] + (a[i][j]-'0');
	}
	int cnt,ans = 0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			for(int k=i+1;k<=n;k++)
			{
				for(int l=j+1;l<=m;l++)
				{
					cnt = dp[i][j] + dp[k][l] - dp[k][j] - dp[i][l];
					if(cnt == 0)
						ans = max(ans , 2*( (k-i) + (l-j)));
				}
			}
		}
	}
	cout<<ans;
	return 0;
}
