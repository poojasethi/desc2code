using namespace std ;
#include<bits/stdc++.h>
#define rep(i,s,e) for(int i = s; i < e; i ++)
string s[30] ; 
int n,m,dp[30][30],ans,cnt; 
main()
{
  cin >>n>>m ; 
  rep(i,0,n)cin >>s[i] ;
  rep(i,0,n)rep(j,0,m)dp[i+1][j+1]=dp[i+1][j] +(s[i][j] == '1') ;
  rep(i,1,n+1)rep(j,1,m+1)rep(k,i,n+1)rep(l,j,m+1){cnt=0;rep(x,i,k+1)if(dp[x][l]-dp[x][j-1])cnt=1;if(!cnt)ans=max(ans,2*(k-i+1)+2*(l-j+1));}
  cout<<ans;}
