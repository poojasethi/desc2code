#include <stdio.h>

#define QUERY(i,j,x,y) (c[x][y]-c[i-1][y]-c[x][j-1]+c[i-1][j-1])
#define max(x, y) ((x)<(y)? (y):(x))

int c[105][105];
char a[105][105];
int main(void) {
	int m, n, i, j, u, v;
	scanf("%d%d", &m, &n);
	for(i=1;i<=m;i++)
		scanf("%s", a[i]+1);
	for(i=1;i<=m;i++)
		for(j=1;j<=n;j++)
			c[i][j] = (a[i][j]=='1')+c[i-1][j]+c[i][j-1]-c[i-1][j-1];
	int ans=0;
	for(i=1;i<=m;i++)
		for(j=1;j<=n;j++)
			for(u=1;u+i<=m+1;u++)
				for(v=1;v+j<=n+1;v++)
					if(QUERY(i, j, i+u-1, j+v-1)==0)
						ans = max(ans, u+v+u+v);
	printf("%d\n", ans);
	return 0;
}
