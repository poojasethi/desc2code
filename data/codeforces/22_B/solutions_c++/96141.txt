#include<stdio.h>
long N,M,i,j,A[26][26],k1[26][26],P, i1, j1;
char c;
int main(void)
 {
  scanf("%ld%ld",&N,&M);
  for(i=1;i<=N;i++)
   {
	scanf("\n");
	for(j=1;j<=M;j++)
	 {
	  scanf("%c", &c);
	  A[i][j]=c-'0';
	 }
   }
  for(i=1;i<=N;i++)
   for(j=1;j<=M;j++)
    {
	 if(A[i][j]==0) k1[i][j]=k1[i-1][j]+1;
	 else k1[i][j]=0; 
    }
  for(i1=1;i1<=N;i1++)
   for(i=i1;i<=N;i++)
    for(j=1;j<=M;j++)
	 if(k1[i][j]>=i1)
	  {
	   for(j1=j;j1<=M;j1++)
	    if(k1[i][j1] < i1) break;
	   if(j1-j+i1 >= P) 
	    P=j1-j+i1;
	  }
  printf("%ld", P*2);
  return 0;	  
 }
