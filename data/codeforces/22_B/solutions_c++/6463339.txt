#include"stdio.h"
#include"algorithm"
using namespace std;
int arr[30][30], i, j, k, n, m, minH, ans;
bool map[30][30];
int main() {
	scanf("%d%d", &n, &m);
	for (i = 1; i <= n; i++)
		for (j = 1; j <= m; j++) {
			scanf(" %c", &k);
			map[i][j] = !(k - '0');
			if (map[i][j]) {
				minH = arr[i][j] = arr[i - 1][j] + 1;
				ans = max(ans, minH + 1);
				for (k = j - 1; map[i][k]; k--) {
					minH = min(minH, arr[i][k]);
					ans = max(ans, minH + (j - k + 1));
				}
			}
		}
	printf("%d", ans * 2);
}
