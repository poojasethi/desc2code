#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int main()
{
char abracadabra[50][50];
int max = 0;
int n,m;
cin>>n>>m;
for(int i=0;i<n;i++)cin>>abracadabra[i];
for(int i=0;i<n;i++)
for(int j=0;j<m;j++)
{
if(abracadabra[i][j] == '1') continue;
for(int p=i;p<n;p++)
for(int l=j;l<m;l++)
{
int flazok = 1;
for(int x=i;x<=p && flazok;x++)
for(int y=j;y<=l && flazok;y++)
if(abracadabra[x][y] == '1')
flazok = 0;
if(flazok && (max < 2*(p-i + l-j+2)))max = 2*(p-i + l-j+2);
}
}
cout<<max<<endl;
}
