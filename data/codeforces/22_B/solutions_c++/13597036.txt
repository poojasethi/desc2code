#include<bits/stdc++.h>

using namespace std;

int M[32][32],m,n,i,j,k,l,s,mn,mx;

char A[32][32];

int main(){

	cin >> m >> n;

	for(i=1;i<=m;i++)
		cin >> A[i] + 1;

	for(i=m; i ;i--)
		for(s=0,j=n; j ;j--)
			if(A[i][j] == '0')
				M[i][j] = ++s;
	        else
				s=0;
	
	for(j=1;j<=n;j++)
		for(i=1;i<=m;i++)
            for(mn=1003,k=i;k<=m;k++){
				mn = min(mn , M[k][j]);	
				if(mn)
					mx = max(mx , 2 * mn + 2 * (k-i+1));
			}

	cout << mx;
	return 0;
}
