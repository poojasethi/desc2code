#include <stdio.h>
#define rep(i, n) for(int i=0; i<(int)(n); i++)

char of[25][25];

int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    rep(i, n) rep(j, m) scanf(" %c", of[i]+j);
    int mx=0, c;
    rep(si, n) rep(sj, m) {
        for(int ti=si; ti<n; ti++) for(int tj=sj; tj<m; tj++) {
            for(int i=si; i<=ti; i++) for(int j=sj; j<=tj; j++) {
                if(of[i][j]=='1') goto next;
            }
            c = (ti-si+1)*2 + (tj-sj+1)*2;
            if(mx<c) mx = c;
next:;
        }
    }
    printf("%d\n", mx);
}
