#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<math.h>
using namespace std;
char a[30][30];
int main()
{
	int n,m;
	int i,j,k,l;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		int ans=0;
		for(i=0;i<n;i++)
		{
			scanf("%s",a[i]);
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				int s=10000;
				if(a[i][j]=='0')
				for(k=0;k<=i;k++)
				{
					if(a[i-k][j]=='1')break;
					for(l=1;l<=j;l++)
					{
						if(a[i-k][j-l]=='1')break;
					}
					s=min(s,l);
					if(ans<s*2+k*2+2)ans=s*2+k*2+2;
				}
			}
		}
		printf("%d\n",ans);
	}
}
