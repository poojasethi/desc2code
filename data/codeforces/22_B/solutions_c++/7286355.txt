#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,ans,a[40][40];
char s[40][40];
int main(){
  scanf("%d%d",&n,&m);
  for(int i=1;i<=n;i++)
    scanf("%s",s[i]+1);
  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
      a[i][j] = a[i-1][j] + a[i][j-1] - a[i-1][j-1] + (s[i][j]-'0');
  for(int i1 = 1;i1<=n;i1++)
    for(int j1 = 1;j1<=m;j1++)
      for(int i2=1;i2<=i1;i2++)
    for(int j2=1;j2<=j1;j2++)
      if(a[i1][j1] - a[i2-1][j1] - a[i1][j2-1] + a[i2-1][j2-1] == 0)
        ans = max(ans,2*(i1-i2+j1-j2+2));
  printf("%d\n",ans);            
  return 0;
}
