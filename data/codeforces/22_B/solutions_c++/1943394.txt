#include <cstdio>
#define N 55
int n,m,a[N][N];
int main() {
  scanf("%d%d",&n,&m);
  register int i,j,p,q,tmp,max=0;
  for (i=1;i<=n;i++)
    for (j=1;j<=m;j++) {
      scanf("%1d",&a[i][j]);
      a[i][j]+=a[i-1][j]+a[i][j-1]-a[i-1][j-1];
    }
  for (i=0;i<n;i++)
    for (j=0;j<m;j++)
      for (p=i+1;p<=n;p++)
        for (q=j+1;q<=m;q++)
          if (max<p+q-i-j<<1) {
            tmp=a[p][q]-a[i][q]-a[p][j]+a[i][j];
            if (!tmp) max=p+q-i-j<<1;
          }
  printf("%d\n",max);
  return 0;
}
