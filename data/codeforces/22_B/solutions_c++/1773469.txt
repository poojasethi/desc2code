#include <iostream>
#include <windows.h>
using namespace std;

int main()
{
	int n, m;
	int arr[30][30];
	int i, j;
	
	cin >> n >> m;
	for (i = 0; i < n; i++)
		for (j = 0; j < m; j++)
		{
			char c;
			cin >> c;
			arr[i][j] = c - '0';
		}

	int res = 0;
	for (i = 0; i < n; i++)
		for (j = 0; j < m; j++)
		{
			int a = m, b = 0;
			for (int x = i; x < n && !arr[x][j]; x++)
			{
				int k = 0;
				b++;
				for (int y = j; y < m && !arr[x][y]; y++)
					k++;
				a = min(a, k);
				if (2 * (a + b) > res)
					res = 2 * (a + b);
			}
		}

	cout << res << endl;
	return 0;
}
