//In the Name of God

#include<iostream>

using namespace std;

int n,m;
string s[50];
int ans;
bool c;


int main()
{
  cin>>n>>m;
  for(int i=1;i<=n;s[i]='1'+s[i],i++)
    cin>>s[i];
  for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
      for(int x=i;x<=n;x++)
	for(int y=j;y<=m;y++)
	  {
	    c=true;
	    for(int l=i;l<=x && c;l++)
	      for(int k=j;k<=y && c;k++)
		if(s[l][k]=='1')
		  c=false;
	    if(c)
	      ans=max(ans,((x-i+1)*2)+((y-j+1)*2));
	  }
  cout<<ans<<endl;
  return 0;
}
