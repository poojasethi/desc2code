#include<bits/stdc++.h>
using namespace std;
char mp[33][33];
int s[33][33],n,m;
int main()
{
    int ans=0;
    cin>>n>>m;
    for(int i=1;i<=n;++i)
    for(int j=1;j<=m;++j)
        cin>>mp[i][j],s[i][j]=s[i-1][j]+s[i][j-1]-s[i-1][j-1]+(mp[i][j]=='1');
    for(int i=1;i<=n;++i)
    for(int j=1;j<=m;++j)
    for(int k=i;k<=n;++k)
    for(int l=j;l<=m;++l)
    if(s[k][l]-s[i-1][l]-s[k][j-1]+s[i-1][j-1]==0)
        ans=max(ans,(k-i+1)*2+(l-j+1)*2);
    cout<<ans<<endl;
}
