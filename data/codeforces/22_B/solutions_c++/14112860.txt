#include <iostream>
using namespace std;
int n, m;
char x[30][30];
bool check(int a,int b,int c,int d)
{
	for(int i=a;i<=b;i++)
		for(int j=c;j<=d;j++)
			if(x[i][j]== '1') 
				return 0;
	return 1;
}
int main()
{
	int Max=-1e9;
	cin>>n>>m;
	for (int i=0;i<n;i++)
		for (int j=0;j<m;j++)
			cin>>x[i][j];


	for(int a=0;a<n;a++)
		for(int b=a;b<n;b++)
			for(int c=0;c<m;c++)
				for(int d=c;d<m;d++)
					if(check(a,b,c,d)) 
						Max=max(Max,((b-a+1)+(d-c+1))*2);
	cout<<Max<<endl;
}
