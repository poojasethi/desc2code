#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
int sum[30][30];
int main(){
	int n,m,ans;
	char c;
	while(cin>>n>>m){
		for(int i=1;i<=n;++i)
			for(int j=1;j<=m;++j){
				cin>>c;
				sum[i][j]=sum[i][j-1]+c-'0';
			}
		ans=0;
		for(int i=1;i<=n;++i)
			for(int j=1;j<=m;++j){
				for(int k=j;k<=m;k++){
					if(sum[i][k]!=sum[i][k-1])
						break;
					for(int z=i;z<=n;++z){
						if(sum[z][j-1]!=sum[z][k])
							break;
						ans=max(ans,2*(z-i+k-j+2));
					}
				}
			}
		cout<<ans<<endl;
	}
}
