#include <cstdio>
#include <algorithm>
using namespace std;
char Map[30][30];
int prefix[30][30];
int Ans=0,n,m;
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++) scanf("%s",Map[i]);
	for (int i=1;i<=n;i++)
		for (int j=0;j<m;j++)
			if (Map[i][j]=='0') prefix[i][j]=prefix[i-1][j]+1;
	for (int i=1;i<=n;i++)
		for (int j=0;j<m;j++)
			if (Map[i][j]=='0')
			{
				int Max=~0U>>2;
				for (int k=j;k<m;k++)
				{
					if (Map[i][k]=='1') break;
					Max=min(Max,prefix[i][k]);
					Ans=max(Ans,(Max+k-j+1)<<1);
				}
			}
	printf("%d\n",Ans);
	return 0;
}
