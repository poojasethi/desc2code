#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int s[105][105];
int main()
{
  int n,m,i,j,x,a,b,c,d;
  scanf("%d%d",&n,&m);
  memset(s,0,sizeof(s));
  for (i=1;i<=n;i++)
	for (j=1;j<=m;j++)
	{
	  scanf("%1d",&x);
	  s[i][j]=s[i-1][j]+s[i][j-1]-s[i-1][j-1]+x;
//	  printf("%d\n",s[i][j]);
	}
  x=0;
  for (a=1;a<=n;a++)
	for (b=1;b<=m;b++)
	  for (c=a;c<=n;c++)
		for (d=b;d<=m;d++)
		  if (s[c][d]-s[c][b-1]-s[a-1][d]+s[a-1][b-1]==0)
		  {
			x=max(x,(c-a+1+d-b+1)*2);
		  }
  printf("%d\n",x);
}
