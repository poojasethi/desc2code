#include<cstdio>
#include<algorithm>
using namespace std;

int n, m, ans;
char str[33];
int s[33][33];

int main() {
	scanf("%d%d", &n, &m);
	for(int i=1;i<=n;i++) {
		scanf("%s", str);
		for(int j=1;j<=m;j++)
			s[i][j] = s[i-1][j]+s[i][j-1]-s[i-1][j-1] + (str[j-1] == '1');
	}
	
	ans = 0;
	for(int x1=1;x1<=n;x1++)
		for(int y1=1;y1<=m;y1++)
			for(int x2=x1;x2<=n;x2++)
				for(int y2=y1;y2<=m;y2++) {
					if(s[x2][y2] - s[x2][y1-1] - s[x1-1][y2] + s[x1-1][y1-1]) continue;
					ans = max(ans, 2*(x2-x1+1 + y2-y1+1));
				}
	printf("%d\n", ans);
	
	return 0;
}
