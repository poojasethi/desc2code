#include<algorithm>
#include<cstdio>
using namespace std;
char a[32][32];
int dp[32][32];
int main(){
  int n,m;
  scanf("%d%d",&n,&m);
  int i;
  for(i=0;i<n;++i)scanf("%s",a[i]);
  int j;
  for(i=0;i<n;++i)
    for(j=0;j<m;++j)
      dp[i+1][j+1]=a[i][j]-48+dp[i+1][j]+dp[i][j+1]-dp[i][j];
  int k,g,maxx=0;
  for(i=0;i<n;++i)
    for(j=0;j<m;++j)
      for(k=i;k<n;++k)
        for(g=j;g<m;++g)
          if(!(dp[k+1][g+1]+dp[i][j]-dp[k+1][j]-dp[i][g+1]))
            maxx=max(maxx,k+1+g+1-i-j<<1);
  printf("%d",maxx);
  return 0;
}
