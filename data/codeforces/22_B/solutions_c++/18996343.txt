#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<queue>
using namespace std;

int mp[50][50];
int dp[50][50];

int main()
{
	int n,m;
	int Max = 0;
	char c;
	scanf("%d %d",&n,&m);
	for(int i=1;i<=n;i++)
	{
		for(int j=1,x;j<=m;j++)
		{
			cin>>c;
			x = c-'0';
			if(x)
				mp[i][j] = 0;
			else
			{ 
				mp[i][j] = mp[i][j-1] + 1;
				for(int k=1,len = mp[i][j];k<=i;k++)
				{
					len = min(len,mp[i-k+1][j]);
					if(len == 0)	break;
					Max = max(Max,(len+k)*2);
				}
			}
		}
	}
	cout<<Max<<endl;
}
