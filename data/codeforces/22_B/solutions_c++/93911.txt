#include <cstdio>
#include <algorithm>
using namespace std;
int n,m,s,mm,a[200][200],c;
char q;
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
		{
			scanf("%c",&q);
			if (q=='\n')j=0;else if (q=='1') a[i][j]=1;
		}
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			for(int k=i;k<=n;k++)
				for(int o=j;o<=m;o++)
				{
					c=((k-i+1)+(o-j+1))*2;
					s=0;
					for(int z=i;z<=k;z++)						
					for(int zz=j;zz<=o;zz++)
						s+=a[z][zz]+a[z][zz];
					if (!s)mm=max(mm,c);					
				}
	printf("%d",mm);
	return 0;
}
