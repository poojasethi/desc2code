#include<iostream>
#include<stdio.h>
using namespace std;
int n,m;
char a[30][30];
int main()
{
	int i,j,ii,jj,ma=0,flag,per;
	cin>>n>>m;
	for(i=0;i<n;i++)
		scanf("%s",a[i]);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			for(ii=i;ii<n;ii++)
			{
				for(jj=j;jj<m;jj++)
				{
					flag=1;
					for(int iii=i;iii<=ii;iii++)
					{
						for(int jjj=j;jjj<=jj;jjj++)
						{
							if(a[iii][jjj]=='1')
								flag=0;
						}
					}
					if(flag)
					{
						per=2*(ii-i+1)+2*(jj-j+1);
						if(per>ma)
							ma=per;
					}
				}
			}
		}
	}
	cout<<ma<<endl;
	return 0;
}
