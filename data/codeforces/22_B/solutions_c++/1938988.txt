#include<iostream>
#include<string>
using namespace std;
int dp[30][30];
int n, m;
int main()
{
  cin>>n>>m;
  char c;
  for (int i = 1; i <= n; i++) for (int j = 1; j <= m; j++) {
      cin>>c;
      dp[i][j] = c - '0';
      dp[i][j] += dp[i-1][j] + dp[i][j-1] - dp[i-1][j-1];
    }
  int ans = 0;
  for (int i = 1; i <= n; i++) for (int j = 1; j <= m; j++) for (int p = i; p <= n; p++) for (int q = j; q <= m; q++) {
	  if (dp[p][q] - dp[i-1][q] - dp[p][j-1] + dp[i-1][j-1] == 0)
	    ans = max(ans, 2*(p-i+1 + q-j+1));
	}
  cout<<ans<<endl;
				 
  return 0;
}
