n, m = map (int, raw_input ().split ())
room, ans = [raw_input () for _ in xrange (n)], 0
for i in reversed (range (1, n + 1)):
    for j in reversed (xrange (1, m + 1)):
        for k in xrange (n - i + 1):
            for q in xrange (m - j + 1):
                if sum ([int (room[t][q:q + j], 2) for t in xrange (k, k + i)]) == 0:
                    ans = max (ans, (i + j) * 2)
print ans
