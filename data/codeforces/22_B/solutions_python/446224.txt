#!/usr/bin/env python
# -*- coding: utf-8 -*-

import sys

def test(l, i, j, n, m):
    s = (n - i + m - j)*2
    c = 0
    while i != n:
        c += l[i][j:m].count(1)
        i += 1
    return s if c == 0 else 0

def get_p(l, i, j, n, m):
    lm = 0
    for i2 in xrange(i+1, n+1):
        for j2 in xrange(j+1, m+1):
            lm = max(lm, test(l, i, j, i2, j2))
    return lm

def main():
    n, m = map(int, raw_input().split())
    l = [[int(c) for c in s[:-1]] for s in sys.stdin.readlines()]
    mp = 0
    for i in xrange(n):
        for j in xrange(m):
            if l[i][j] == 0:
                mp = max(mp, get_p(l, i, j, n, m))
    print mp
    return 0

if __name__ == '__main__':
    main()
