n, m = map(int, raw_input().split())
s = [[0] + map(int, raw_input()) for i in range(n)]
for i in range(n):
    for j in range(m): s[i][j + 1] += s[i][j]
d = 1
for i in range(0, m):
    for j in range(i + 1, m + 1):
        t = [-1] + [k for k in range(n) if s[k][j] - s[k][i]] + [n]
        p = max(t[i + 1] - t[i] for i in range(len(t) - 1))
        if p > 1 and p + j - i > d: d = p + j - i
print 2 * (d - 1)
