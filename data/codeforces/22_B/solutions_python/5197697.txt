n,m=map(int, raw_input().split())
G=[map(int,raw_input()) for i in n*[0]]
M=2
for i in range(n):
    for j in range(m):
        if G[i][j]: continue
        Y=m
        for x in range(i,n):
            for y in range(j,m):    
                if G[x][y] and Y > y:
                    Y=y
                if y >= Y: continue            
                M=max(M,x-i+y-j+2)
print M*2
