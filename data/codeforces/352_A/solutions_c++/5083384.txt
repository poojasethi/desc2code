#include <cstdio>

int main() {
	int n, n0=0, n5=0, cur;
	scanf("%d", &n);
	while (n--) {
		scanf("%d", &cur);
		n0 += !cur;
		n5 += !!cur;
	}
	n5 -= (n5 % 9);
	if (n0 == 0) {
		printf("-1\n");
		return 0;
	}
	if (n5 == 0) {
		n0 = 1;
	}
	while (n5--) {
		printf("5");
	}
	while (n0--) {
		printf("0");
	}
	printf("\n");
	return 0;
}
