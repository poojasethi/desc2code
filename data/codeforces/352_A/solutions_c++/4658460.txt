#include <cstdio>

int n,i,j,k,f,z;

int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&j);
		if (j==5) f++;
		else z++;
	}
	if (z==0){
		printf("-1");
	}else{
		f-=f%9;
		for(i=0;i<f;i++){
			printf("5");
		}
		if (f){
			for(i=0;i<z;i++){
				printf("0");
			}
		}else{
			printf("0");
		}
	}
	printf("\n");
	return 0;
}
