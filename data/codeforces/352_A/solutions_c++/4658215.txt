#include<cstdio>
using namespace std;
int main()
{
	int n,tmp,k=0,m=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&tmp);
		if(tmp==0)m++;
		else k++;
	}
	if(m==0){
		printf("-1");
		return 0;
	}
	k-=(k%9);
	for(int i=0;i<k;i++)printf("5");
	if(k!=0)for(int i=0;i<m;i++)printf("0");
	else printf("0");
	return 0;
}
