#include<iostream>
using namespace std;
int main()
{
	int n,sum=0,s0=0,a;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a;if(a==0)s0++;else sum++;
	}
	if(s0)
	{
		for(int i=0;i<(sum/9)*9;i++)cout<<5;
		if((sum/9)*9)for(int i=0;i<s0;i++)cout<<0;
		else cout<<0<<endl;
	}
	else cout<<-1<<endl;
}
