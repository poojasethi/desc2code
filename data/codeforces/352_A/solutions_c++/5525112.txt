#include<iostream>
using namespace std;
int main() {
	int n, nf=0, nz=0, p;
	cin >> n;
	for(int i=0;i<n;i++) {
		cin >> p;
		if(p == 0) nz++;
		else nf++;
	}
	if(nz == 0)
		cout << -1;
	else if(nf < 9)
		cout << 0;
	else {
		for(int i=0;i<nf/9*9;i++)
			cout << 5;
		for(int i=0;i<nz;i++)
			cout << 0;
	}
	return 0;
}
