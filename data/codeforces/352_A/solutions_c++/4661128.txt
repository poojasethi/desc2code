#include <iostream>

using namespace std;

int main()
{

	int n,a,five=0,zero=0;
	cin>>n;
	while(n--)
	{
		cin>>a;
		if(a==5) five++;
		else zero++;
	}
	five=five/9;
	if(zero==0) {cout<<"-1"; return 0;}
	for(a=1;a<=9*five;a++)
		cout<<"5";
	if ( five==0) {cout<<"0"; return 0;}
	for(a=1;a<=zero;a++)
		cout<<"0";
	cout<<endl;
	return 0;


}
