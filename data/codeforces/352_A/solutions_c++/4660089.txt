#include<cstdio>
using namespace std;

int n,b,i5,i0,t;

int main() {
	scanf("%d",&n);
	b=n;i0=i5=0;
	while (b--) {
		scanf("%d",&t);
		if (t) i5++;
		else i0++;
	}
	if(!i0) {printf("-1\n");return 0;}
	i5 = i5/9;
	if(!i5) {printf("0\n");return 0;}
	while(i5--) printf("555555555");
	while(i0--) printf("0");
	printf("\n");
	return 0;
}
