#include<stdio.h>
int main(){
	int a,m,n;
	int i,j,k;
	while(~scanf("%d",&n)){
		j=k=0;
		for(i=0;i<n;++i){
			scanf("%d",&a);
			if(a==5)k++;
			else j++;
		}
		if(j>0){
			for(i=0;i<k/9*9;++i)printf("5");
			if(k/9>0)for(i=0;i<j-1;++i)printf("0");
			printf("0\n");
		}
		else printf("-1\n");
	}return 0;
}
