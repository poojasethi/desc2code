#include<iostream>
using namespace std;
int main()
{
	int n,i,j,f=0,z=0;
	cin >> n;
	for(i=0;i<n;i++)
	{
		cin >> j;
		if(j==5)
			f++;
		else z++;
	}
	if(z==0 )
		cout<<"-1\n";
	else if (f/9 ==0)
		cout << "0\n";
	else
	{
		for(i=1;i<=(f/9)*9;i++)
			cout<<"5";
		for(i=1;i<=z;i++)
			cout<<"0";
		cout<<endl;
	}
	return 0;

}
