#include <iostream>
using namespace std;
int n, x, c5, c0;
int main()
{
	cin>>n;
	while(n--) {
		cin>>x;
		if (x==5) c5++;
		else c0++;
	}
	if (c0==0) cout<<-1<<endl;
	else {
		c5=c5/9*9;
		if (c5) {
			while (c5--) cout<<5;
			while (c0--) cout<<0;
		}
		else cout<<0;
		cout<<endl;
	}
	return 0;
}
