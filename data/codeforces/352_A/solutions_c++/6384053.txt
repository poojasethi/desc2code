#include <bits/stdc++.h>
using namespace std;

int main() {
	int n, c[6] = {};
	cin >> n;
	for (int a, i = 0; i < n; ++i) {
		cin >> a;
		c[a]++;
	}
	int m = c[5] / 9 * 9;
	if (!c[0]) cout << -1;
	else if (!m) cout << 0;
	else {
		for (int i = 0; i < m; ++i) cout << 5;
		for (int i = 0; i < c[0]; ++i) cout << 0;
	}
}
