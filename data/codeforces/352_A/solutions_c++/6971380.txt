#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,a,num5=0,num0=0;
	cin>>n;
	while(n--)
	{
		cin>>a;
		(a) ? num5++ : num0++;
	}
	if(num0==0)
		cout<<"-1\n";
	else
	{
		num5-=num5%9;
		if(!num5)
			cout<<"0\n";
		else
		{
			while(num5--)
				cout<<"5";
			while(num0--)
				cout<<"0";
			cout<<endl;
		}
	}
	return 0;
}
