#include <bits/stdc++.h>

using namespace std;

int main ()
{
	int n,x,i, f=0,z=0;
	cin >> n;
	for(i=1; i<=n; i++)
	{
		cin >> x;
		(!x) ? z++ : f++;
	}
	if(!z) cout << -1; 
	else if (!(f/9) and z) cout << 0;
	else
	{
		for(i=1; i<=f-f%9; i++)
			cout << 5;
		for(i=1; i<=z; i++)
			cout << 0;
	}
	cout << endl;
}
