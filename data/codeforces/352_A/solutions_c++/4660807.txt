#include<iostream>
using namespace std;
int main()
{
int n;
cin>>n;
int inp,n5=0,n0=0;
for(int i=0;i<n;i++)
{
	cin>>inp;
	if(inp==5) n5+=1;
	else n0+=1;
}
if(n0==0)
{
cout<<"-1\n";
return 0;
}
n5=(n5/9);
n5=n5*9;
for(int i=0;i<n5;i++) cout<<"5";
if(n5!=0) for(int i=0;i<n0;i++) cout<<"0";
else cout<<"0";
cout<<"\n";
return 0;
}
