#include<bits/stdc++.h>
using namespace std;
int main(){
	int sif=0,bes=0,i,N,arr[1004];
	cin>>N;
	for(i=0;i<N;i++){
		cin>>arr[i];
		bes+= (arr[i]==5);
		sif+= !arr[i];
	}
	bes-=bes%9;
	if(!sif) {cout<<-1; return 0;}
	if(!bes) {	cout<<0; return 0;}
	for(i=0;i<bes;i++)
		cout<<5;
	for(i=0;i<sif;i++)
		cout<<0;
	

	return 0;
}
