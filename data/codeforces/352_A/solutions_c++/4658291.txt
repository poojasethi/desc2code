#include <iostream>

using namespace std;

int n,x, a[6];

int main(){
	cin>>n;
	for (int i=0; i<n; i++){
		cin>>x;
		a[x]++;
	}
	int f = a[5]/9;
	if (a[0]==0)
		cout<<-1<<endl;
	else {
		for (int i=0; i<f*9; i++)
			cout<<5;
		if (f>0){
			for (int i=0; i<a[0]; i++)
				cout<<0;
		}
		else
			cout<<0;
		cout<<endl;
	}
}
