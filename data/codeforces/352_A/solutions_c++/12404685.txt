#include<bits/stdc++.h>

using namespace std;

int i,j,k;

int main(void)
{
	int n,t,f=0,z=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>t;
		if(t==5)f++;
		else{z++;}
	}
	if(z==0)cout<<-1;
	else if(f<9)cout<<0;
	else
	{	for(j=0;j<f/9;j++)
		for(i=0;i<9;i++)cout<<5;
		for(i=0;i<z;i++)cout<<0;
	}
	return 0;
}
