#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	pair<int,int> a[10005];
	int n,p[2],k=0,i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i].first;
		a[i].second=i+1;
	}
	int cnt=0;
	sort(a,a+n);
	for(i=0;i<n-1 &&cnt<2;i++)
	{
		if(a[i].first==a[i+1].first)
		{
			cnt++;
			p[k++]=i;
		}
	}
	if(cnt==2)
	{
		cout<<"YES\n";
		for(int j=0,k=0;j<3;j++,k++)
		{
			for(i=0;i<n;i++)
			{
				cout<<a[i].second<<" ";
			}
			cout<<"\n";
			swap(a[p[k]].second,a[p[k]+1].second);
		}
	}
	else
	cout<<"NO"<<endl;
	return 0;
}
