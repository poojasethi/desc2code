#include <bits/stdc++.h>

using namespace std;

int n;
pair <int, int> moo[3000];

void print() {
  for (int i = 0; i < n; i++) {
    printf ("%d ", moo[i].second);
  }
  printf ("\n");
}

int main() {
  scanf ("%d", &n);
  for (int i = 0; i < n; i++) {
    scanf ("%d", &moo[i].first);
    moo[i].second = i+1;
  }
  sort (moo, moo+n);
  int tot = 0;
  vector <int> loc;
  for (int i = 1; i < n; i++) {
    if (moo[i].first == moo[i-1].first) {
      tot++;
      loc.push_back(i);
    }
  }
  if (tot <= 1) {
    printf ("NO\n");
    return 0;
  }
  printf ("YES\n");
  print();
  swap(moo[loc[0]], moo[loc[0]-1]);
  print();
  swap(moo[loc[1]], moo[loc[1]-1]);
  print();
}
