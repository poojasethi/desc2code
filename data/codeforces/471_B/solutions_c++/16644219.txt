#include <iostream>
#include<algorithm>
using namespace std;
#define MAX 2010
int n;
pair <int, int> A[MAX];

int main() {
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> A[i].first;
		A[i].second = i+1;
	}
	sort(A, A + n);
	int cnt = 0, f[5];
	for (int i = 1; i < n; i++)
	{
		if (A[i-1].first == A[i].first)
		{
			f[cnt] = i - 1;
			cnt++;
			if (cnt == 2)
				break;
		}
	}
	if (cnt != 2)
	{
		cout << "NO";
		return 0;
	}
	cout << "YES\n";
	for (int i = 0; i < n; i++)
	{
		cout << A[i].second << " ";
	}
	for (int k = 0; k < 2; k++)
	{
		cout << endl;
		swap(A[f[k]].second, A[f[k] + 1].second);
		for (int i = 0; i < n; i++)
		{
			cout << A[i].second << " ";
		}
	}
	
	return 0;
}
