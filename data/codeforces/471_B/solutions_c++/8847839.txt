#include <iostream>
#include <map>
#include <algorithm>
using namespace std;
int main() {
	int M, N, i, j=0, o;		map <int, int> ar;
	cin >> M;
	pair <int, int> job[M]; 
	for(i=0; i<M; ++i) { cin >> N; job[i] = make_pair(N, i+1); ++ar[N]; if(ar[N] >= 2) ++j;	}
	sort(job, job+M);
	if(j >= 2) {
		cout << "YES" << endl;
		for(N=0, i=0; N !=3; ++i){
			if( job[i].first == job[i+1].first ){
				swap(job[i], job[i+1] );
				for(o=0; o<M; ++o) cout << job[o].second << " "; ++N;
				cout << endl;
				if( N != 3) {
					swap(job[i], job[i+1]);
					for(o=0; o<M; ++o) cout << job[o].second << " "; ++N;
					cout << endl;
				}
			}
		}
	}
	else cout << "NO" << endl;
}
