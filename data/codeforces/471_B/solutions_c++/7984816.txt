#include <iostream>
#include <cstdio>
#include <stdlib.h>
#include <algorithm>
using namespace std;
pair <int,int> a[2001];
int main()
{
	int n,c=0;
	scanf("%d",&n);
	for (int i=1; i<=n; i++)
	{
		scanf("%d",&a[i].first);
		a[i].second=i;
	}
	sort(a+1,a+n+1);
	for (int i=1; i<n; i++) if (a[i].first==a[i+1].first) c++;
	if (c>1)
	{
		printf("YES\n");
		for (int i=1; i<=n; i++) printf("%d ",a[i].second);
		printf("\n");
		for (int i=1,temp=0; i<n && temp<2; i++)
		{
			if (a[i].first==a[i+1].first)
			{
				swap(a[i],a[i+1]);
				for (int j=1; j<=n; j++) printf("%d ",a[j].second);
				printf("\n");
				temp++;
			}
		}
	}
	else printf("NO\n");
	return 0;
}
