#include<bits/stdc++.h>
#define fi first
#define se second
#define pb push_back
using namespace std;
int main()
{
    int n,h,cnt=0;vector<pair<int,int> > vii;
    cin>>n;
    for(int i=0;i<n;i++) cin>>h,vii.pb(make_pair(h,i+1));
    sort(vii.begin(),vii.end());
    int pos[2001]={ };
    for(int i=0;i<n-1;i++) if(vii[i].fi==vii[i+1].fi) cnt++,pos[i]++;
    if(cnt<2) {cout<<"NO";return 0;}
    cout<<"YES"<<endl;
    for(int i=0;i<n;i++) cout<<vii[i].se<<" ";cout<<endl;
    int ck=2;
    for(int i=0;i<n;i++)
        if(ck!=0)
            if(pos[i]==1)
            {
                swap(vii[i],vii[i+1]),ck--;
                for(int i=0;i<n;i++) cout<<vii[i].se<<" ";cout<<endl;
            }
    return 0;
}
