#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void printseq(vector<pair<int,int> > v)
{
	int i;
	for(i=0; i<v.size()-1; i++) {
		cout << v[i].second <<' ';
	}
	cout << v[i].second <<'\n';
}

int main()
{
	vector< pair <int,int> > v;
	pair<int,int> p;
	int n, i;
	int save[2], ns=0;

	cin>>n;
	for(i=0; i<n; i++) {
		cin>>p.first;
		p.second=i+1;
		v.push_back(p);
	}
	sort(v.begin(), v.end());
	for(i=0; i<n-1; i++) {
		if(v[i].first==v[i+1].first) {
			save[ns]=i;
			ns++;
		}
		if(ns==2) break;
	}
	if(ns<2) {
		cout <<"NO\n";
	} else {
		cout <<"YES\n";
		printseq(v);
		swap(v[save[0]], v[save[0]+1]);
		printseq(v);
		swap(v[save[1]], v[save[1]+1]);
		printseq(v);
	}
	return 0;
}
