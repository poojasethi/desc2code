#include <bits/stdc++.h>
using namespace std;
int c[2000],i,n,t,c1,c2;
main(){
    cin>>n;
    pair<int,int>p[n];
    for(;i<n;i++)cin>>p[i].first,p[i].second=i,c[p[i].first]++;
    sort(p,p+n);
    for(i=0;i<2000;i++)if(c[i]==2)c1++;else if(c[i]>=3)c2++;
    if(c1>1|c2>0){
        puts("YES");
        for(i=0;i<n;i++)cout<<p[i].second+1<<' ';
        puts("");
        for(i=0;i<n-1;i++)if(p[i+1].first==p[i].first){swap(p[i+1],p[i]);t=i+1;break;}
        for(i=0;i<n;i++)cout<<p[i].second+1<<' ';
        puts("");
        for(;t<n-1;t++)if(p[t+1].first==p[t].first){swap(p[t+1],p[t]);break;}
        for(i=0;i<n;i++)cout<<p[i].second+1<<' ';
    }else puts("NO");
}
