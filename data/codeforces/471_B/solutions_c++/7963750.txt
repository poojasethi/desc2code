#include <bits/stdc++.h>
#define maxn 2020
#define fi first
#define se second

using namespace std;

int N,kon,used[maxn];
typedef pair <int,int> ii;
ii ar[maxn];

int main()
{
	scanf("%d",&N);
	
	for(int i=1;i<=N;i++)
	{
		int a;
		scanf("%d",&a);
		ar[i]=ii(a,i);
		
		if(used[a])
			kon++;
		
		used[a]++;
	}
	
	if(kon<2)
	{
		printf("NO\n");
		return 0;
	}
	
	printf("YES\n");
	
	sort(ar+1,ar+N+1);
	
	for(int k=1;k<=3;k++)
	{
		for(int i=1;i<=N;i++)
			printf("%d ",ar[i].se);
			
		printf("\n");
		
		for(int i=1;i<N;i++)
		{
			if(ar[i].fi == ar[i+1].fi && ar[i].se < ar[i+1].se)
			{
				swap(ar[i],ar[i+1]);
				break;
			}
		}	
	}
	
	return 0;
}
