#include <bits/stdc++.h>
using namespace std;
const int N=1e4;
pair <int, int> ma[N];
int cnt[N],used[N];
int main(){
	int n;
	scanf("%d", &n);
	for(int i=0; i<n; i++){
		scanf("%d", &ma[i].first);
		ma[i].second=i+1;
		cnt[ma[i].first]++;
	}
	sort(ma, ma+n);
	int par=0;
	for(int i=1;i<2000;i++)par+=(cnt[i]==2?1:cnt[i]>2?2:0);
	puts(par>1?"YES":"NO");
	for(int i=0;par>1&&i<3; i++){
		for(int j=0; j<n; j++){
			printf("%d ", ma[j].second);
		}
		puts("");
		for(int j=0; j<n-1; j++){
			if(!used[j]&&ma[j].first==ma[j+1].first){
				swap(ma[j], ma[j+1]);
				used[j]=1;
				break;
			}
		}
	}
	return 0;
}
