#include <bits/stdc++.h>
using namespace std;
int n,tt;
vector<pair<int,int> >h,ans;
void print(){
	for(int q=0;q<n;q++){
		cout<<h[q].second<<" ";
	}
	cout<<endl;
}
int main(){
	cin>>n;
	for(int q=1;q<=n;q++){
		cin>>tt;
		h.push_back(make_pair(tt,q));
	}

	sort(h.begin(),h.end());

	for(int q=0;q<n-1;q++){
		if(h[q].first==h[q+1].first){
			ans.push_back(make_pair(q,q+1));
		}
	}

	if(ans.size()<2){
		cout<<"NO\n";
		return 0;
	}
	cout<<"YES\n";
	print();
	swap(h[ans[0].first],h[ans[0].second]);
	print();
	swap(h[ans[1].first],h[ans[1].second]);
	print();

	return 0;
}
