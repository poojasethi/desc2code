#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int main(){
	int h,c=0;
	pair<int,int> a[2001];
	cin>>h;
	for(int i=0;i<h;i++){
		cin>>a[i].first;
		a[i].second=i+1;
	}
	sort(a,a+h);
	for(int i=0;i<h;i++)
		if(a[i].first==a[i+1].first)c++;
	if(c>1){
		cout<<"YES"<<endl;
		for(int i=0;i<h;i++)
			cout<<a[i].second<<" ";
		cout<<endl;
		for(int j=0,c=0;j<h&&c<2;j++)
			if(a[j].first==a[j+1].first){
				swap(a[j],a[j+1]);
				for(int i=0;i<h;i++){
						cout<<a[i].second<<" ";
					}
				cout<<endl;
                c++;
			}
		}
	else
    cout<<"NO"<<endl;
	return 0;
}
