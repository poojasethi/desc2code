#include <iostream>
#include<bits/stdc++.h>
using namespace std;
pair<int,int>pr[2500];
int main()
{
    int n;
    cin>>n;
    for(int i = 0 ; i<n ; i++)
        {cin>>pr[i].first; pr[i].second=i;}
    sort(pr,pr+n);
    int sum=0;
    for(int i = 1 ; i<n ; i++) if(pr[i].first==pr[i-1].first) sum++;

    if(sum<2) return puts("NO");
    cout<<"YES"<<endl;
    for(int i = 0 ; i<n ; i++) cout<<pr[i].second+1<<" ";
    cout<<endl;

    int x = 0;
    for(int i = 0 ; i<n-1 && x<2 ; i++)
    {
        if(pr[i].first == pr[i+1].first)
        {
            swap(pr[i],pr[i+1]);
            for(int j = 0 ; j<n ; j++) cout<<pr[j].second+1<<" ";
            cout<<endl;
            x++;
        }
    }
    return 0;
}
