#include <bits/stdc++.h>
using namespace std;
int c[2001],i=1,n,c1,c2;
pair<int,int>p[2001];
main(){
    for(cin>>n;i<=n;i++)cin>>p[i].first,p[i].second=i,c[p[i].first]++;
    sort(p,p+n+1);
    for(i=1;i<2001;i++)if(c[i]==2)c1++;else if(c[i]>=3)c2++;
    if(c1>1|c2>0){
        puts("YES");
        for(i=1;i<=n;i++)cout<<p[i].second<<' ';
        puts("");
        for(i=2;i<=n;i++)if(p[i-1].first==p[i].first){swap(p[i-1],p[i]);c1=i+1;break;}
        for(i=1;i<=n;i++)cout<<p[i].second<<' ';
        puts("");
        for(;c1<=n;c1++)if(p[c1-1].first==p[c1].first){swap(p[c1-1],p[c1]);break;}
        for(i=1;i<=n;i++)cout<<p[i].second<<' ';
    }else puts("NO");
}
