#include<bits/stdc++.h>
using namespace std;
class num{
	public:	int n,ind;
};
bool compare(class num a , class num b){
	return a.n < b.n ;
}
int main(){
	int n,i;
	cin >> n;
	class num arr[n+1];int chng[2]={0,0},f=0;
	for(i=0;i<n;i++){ cin >> arr[i].n; arr[i].ind = i+1;}
	sort(arr,arr+n,compare);
	for(i=0;i<n-1;i++){
		if(arr[i].n==arr[i+1].n && f<2) chng[f++] = i;
	}
	if(f>=2){
		cout << "YES" << endl;
		for(i=0;i<n;i++) cout << arr[i].ind << " ";
		cout << endl;
		swap(arr[chng[0]].ind,arr[chng[0]+1].ind);
		for(i=0;i<n;i++) cout << arr[i].ind << " ";
		cout << endl;
		swap(arr[chng[1]].ind,arr[chng[1]+1].ind);
		for(i=0;i<n;i++) cout << arr[i].ind << " ";
		cout << endl;
	}
	else cout << "NO\n";
	return 0;
}
