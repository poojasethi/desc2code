#include <iostream>
#include <algorithm>
using namespace std;

typedef pair< int, int >  pii;

int main(){
	int n;
	cin >> n;
	pii p[3000];
	for (int i = 0; i < n; i++){
		cin >> p[i].first;
		p[i].second = i + 1;
	}

	sort(p, p + n);
	int d = 0;
	for (int j = 0; j < n; j++)
		if (p[j].first == p[j + 1].first)
			d++;

	if (d < 2){
		cout << "NO";
		return 0;
	}

	cout << "YES" << endl;
	for (int j = 0; j < n; j++)
		cout << p[j].second << " ";
	cout << endl;

	int k = 0;
	for (int i = 0; i < n; i++){
		if (p[i].first == p[i + 1].first){
			swap(p[i], p[i + 1]);
			for (int j = 0; j < n; j++)
				cout << p[j].second << " ";
			cout << endl;
			k++;
		}
		if (k == 2)
			return 0;

	}
	cout << "no";
}
