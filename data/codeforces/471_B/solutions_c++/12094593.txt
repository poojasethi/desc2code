#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

pair<int, int> P[2100];
vector<int> V;
int n, cnt;

void printResult() {
    for (int i = 0; i < n; i++) cout << P[i].second << " ";
    cout << endl;
}

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) cin >> P[i].first, P[i].second = i+1;
    sort(P, P+n);
    for (int i = 1; i < n; i++) {
        if (P[i].first == P[i-1].first)
            cnt++, V.push_back(i);
    }
    if (cnt < 2) {cout << "NO"; return 0;}
    cout << "YES\n";
    printResult();
    swap(P[V[0]], P[V[0]-1]);
    printResult();
    swap(P[V[1]], P[V[1]-1]);
    printResult();
}
