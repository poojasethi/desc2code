#include<bits/stdc++.h>
using namespace std;

#define ll long long

ll i,j,k,p,q, n, ar[2005];
typedef pair <ll,ll>PII;
vector<PII> v;
map<ll,ll> mp;

int main()
{
	cin>> n;
	for(i=1; i<=n; i++)
	{
		cin>> ar[i];
		mp[ar[i]]++;
		v.push_back(make_pair(ar[i], i));
	}
	sort(ar, ar+n+1);
	for(i=1; i<=n; i++)
	{
		if(ar[i]== ar[i+1])
			k++;
	}
	if(k>=2 || (mp.size()==1 && n>2))
	{
		cout<<"YES\n";
		sort(v.begin(), v.end());
		for(j=0; j<3; j++)
		{
			for(i=0; i<n; i++)
				cout<< v[i].second<<" ";
				cout<<"\n";
			for(i=p; i<n; i++)
			{
				if(mp[v[i].first]>=2)
				{
					swap(v[i], v[i+1]);
					mp[v[i].first]--;
					p= i+1;
					break;
				}
			}
		}
	}
	else
		cout<<"NO"<<endl;
	return 0;
}
