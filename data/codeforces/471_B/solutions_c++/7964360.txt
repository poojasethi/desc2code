#include <bits/stdc++.h>
using namespace std;

void put(vector < pair <int, int> > v)
{
	for (int i = 0; i < (int)v.size(); ++i)
	{
		cout << v[i].second << " ";
	}
	cout << "\n";
}

int main (void)
{
	cin.tie(NULL);
	ios_base :: sync_with_stdio (false);
	int n;
	cin >> n;
	vector < pair <int, int> > arr(n);
	set <int> s;
	for (int i = 0 ; i < n; ++i)
	{
		cin >> arr[i].first;
		arr[i].second = i + 1;
		s.insert(arr[i].first);
	}
	if((int)s.size() > n - 2)
	{
		cout <<"NO";
		return 0;
	}
	cout << "YES\n";
	sort(arr.begin(), arr.end());
	int t = 0;
	put(arr);
	for (int i = 0; t < 2; ++i)
	{
		if(arr[i].first == arr[i + 1].first)
		{
			swap(arr[i], arr[i+1]);
			put(arr);
			++t;
		}
	}
	return 0;
}
