#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
typedef __int64 ll;
int main() {
	ll n,i,dd[2002]={0},s=1,k;
	pair<ll,ll> a[2001];
	cin >> n;
	for (i=1;i<=n;i++) {
		cin >> a[i].first;
		if (s<3) s*=++dd[a[i].first];
		a[i].second=i;
	}
	sort(a+1,a+n+1);
	if (s>=3) {
		cout << "YES\n";
		for (k=1;k<=3;k++) {
			long t=k;
			for (i=1;i<=n;i++) {
				cout << a[i].second << " ";
				if (a[i].first==a[i-1].first && t>0)
					swap(a[i],a[i-1]),t--;
			}
			cout << endl;
		}
	} else cout << "NO";
}
