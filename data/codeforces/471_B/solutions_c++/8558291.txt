#include <stdio.h>
#include <math.h>
#include <string.h>
#include <iostream>
#include <algorithm>
using namespace std;
#define pb push_back
typedef long long ll;
struct node{
	int p,v;
}x[2222];
bool cmp(node a,node b){
	return a.v<b.v;
}
int sw[2];
void pt(int n){
	for(int i=1;i<=n;i++)
		cout<<x[i].p<<" ";
	cout<<endl;
}
int main()
{
	int n;cin>>n;
	for(int i=1;i<=n;i++)
		cin>>x[i].v,x[i].p=i;
	sort(x+1,x+1+n,cmp);
	int cnt=0;
	for(int i=1;i<n;i++){
		if(x[i].v==x[i+1].v){
			sw[cnt]=i;
			cnt++;
		}if(cnt==2) break;
	}
	if(cnt==2){
		puts("YES");
		pt(n);
		swap(x[sw[0]],x[sw[0]+1]); pt(n);
		swap(x[sw[1]],x[sw[1]+1]); pt(n);
	}
	else puts("NO");
}
