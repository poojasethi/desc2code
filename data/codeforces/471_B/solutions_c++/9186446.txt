#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	pair<int ,int>a[2001];
	int n,i,k=0,w=0;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>a[i].first;
		a[i].second=i;
	}
	sort(a,a+n);
	for(i=1;i<=n;i++){
		if(a[i].first==a[i-1].first)
			k++;
    
	}
	if(k<2){
		cout<<"NO"<<endl;
		return 0;
	}
	cout<<"YES"<<endl;
	for(i=0;i<n;i++)
		cout<<a[i].second+1<<" ";
	cout<<endl;
	for(i=1;i<=n&&w!=2;i++){
		if(a[i].first==a[i-1].first){
			swap(a[i],a[i-1]);
		for(k=0;k<n;k++)
		  cout<<a[k].second+1<<" ";
		cout<<endl;
		w++;

		}
			
    
	}
	return 0;
}
