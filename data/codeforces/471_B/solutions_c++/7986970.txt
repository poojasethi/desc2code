# include<iostream>
# include<algorithm>
# include<vector>
using namespace std;
int main()
{
	int n,x,f=0,i;
	cin>>n;
	vector< pair <int,int> >v;
	for(i=1;i<=n;i++)
	{
		cin>>x;
		v.push_back(make_pair(x,i));
	}
	sort(v.begin(),v.end());
	for(i=0;i<v.size();i++)
	if(v[i].first==v[i+1].first)
	f++;
	//cout<<f;
	if(f<2)
		cout<<"NO";
	else
	{
		cout<<"YES\n";
		for(i=0;i<v.size();i++)
		cout<<v[i].second<<" ";
		cout<<endl;
		int c=0,j;
		for(i=0;i<v.size()-1;i++)
		{
			if(v[i].first==v[i+1].first)
			{
				swap(v[i],v[i+1]);
				for(j=0;j<v.size();j++)
				cout<<v[j].second<<" ";
				cout<<endl;
				c++;
			}
			if(c>=2)
			break;
		}
	}
}
