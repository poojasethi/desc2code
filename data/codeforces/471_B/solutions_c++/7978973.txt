#include <algorithm>
#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;
typedef pair<int,int> PII;
PII a[2222];
int n,cnt;

void dfs(int x){
	if (cnt==3) return;
	if (x==n){
		cnt++;
		for (int i=0;i<n;i++) cout<<a[i].second<<' ';
		cout<<endl;
		return;
	}
	dfs(x+1);
	if (a[x].first==a[x-1].first) swap(a[x],a[x-1]); else return;
	//cout<<a[x].first<<' '<<a[x-1].first<<endl;
	dfs(x+1);
	swap(a[x],a[x-1]);
}

int main(){
	cin>>n;
	for (int i=0;i<n;i++){
		cin>>a[i].first;
		a[i].second=i+1;
	}
	sort(a,a+n);
	cnt=0;
	for (int i=1;i<n;i++)
		if (a[i].first==a[i-1].first) cnt++;
	if (cnt<2){puts("NO"); return 0;}
	puts("YES");
	cnt=0;
	dfs(1);
}
