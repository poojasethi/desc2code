#include<cstdio>
#include<algorithm>
using namespace std;
struct stt{int a,i;}st[2000];
bool cmp(stt st1,stt st2)
{return st1.a<st2.a;}
int n;
void doo()
{
    for(int i=0;i<n;++i)
    {printf("%d ",st[i].i);}
    puts("");
}
int main()
{
    int a;bool bl=0;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
    {
        scanf("%d",&st[i].a);
        st[i].i=i+1;
    }
    sort(st,st+n,cmp);
    for(int i=1;i<n;++i)
    if(st[i].a==st[i-1].a)
    {
        if(bl)
        {
            puts("YES");
            doo();
            swap(st[a].i,st[a-1].i);
            doo();
            swap(st[i].i,st[i-1].i);
            doo();
            return 0;
        }
        else a=i,bl=1;
    }
    puts("NO");
}
