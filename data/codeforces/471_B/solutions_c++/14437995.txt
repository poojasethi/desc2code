#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n, cnt=0, p[2];
	cin>>n;
	pair<int, int> a[100005];
	for(int i=0 ; i<n ; i++)
	{
		cin>>a[i].first;
		a[i].second = i+1;
	}
	sort(a,a+n);
	int k=0;
	for(int i=0 ; i<n-1 && cnt<2 ; i++)
	{
		if(a[i].first == a[i+1].first)
		{
			cnt++;
			p[k++] = i;
		}
	}
	if(cnt == 2)
	{
		cout<<"YES\n";
		for(int q=0, k=0 ; q<3 ; q++, k++)
		{
			for(int i=0 ; i<n ; i++)
			{
				cout<<a[i].second<<" ";
			}
			cout<<endl;
			swap(a[p[k]].second,a[p[k]+1].second);
		}
	}
	else
		cout<<"NO\n";
}
