#include<iostream>
#include<algorithm>
using namespace std;
struct el {
	int i, v;
};
el a[2000];
int n, t, r, s;
bool f(el i, el j) {
	return i.v < j.v;
}
void p() {
	for (int i = 0; i < n; i++)
		cout << a[i].i << " ";
	cout << endl;
}
int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> t;
		a[i].i = i+1;
		a[i].v = t;
	}
	sort(a, a+n, f);
	r = s = -1;
	for (int i = 0; i < n-1; i++) {
		if (a[i].v == a[i+1].v)
			if (r == -1)
				r = i;
			else if (s == -1)
				s = i;
	}
	if (s == -1) {
		cout << "NO";
	} else {
		cout << "YES" << endl;
		p();
		t = a[r].i;
		a[r].i = a[r+1].i;
		a[r+1].i = t;
		p();
		t = a[s].i;
		a[s].i = a[s+1].i;
		a[s+1].i = t;
		p();
	}
}
