n = int(raw_input())
a = map(int, raw_input().split())
b = []
for i in range(n):
	b.append([a[i], i+1])
b.sort(key = lambda x:x[0])
x = 0
ans = []
li = []
for i in b:
	li.append(str(i[1]))
ans.append(li)
for i in range(n-1):
	if x >= 2:
		break
	if b[i][0] == b[i+1][0]:
		b[i][1], b[i+1][1] = b[i+1][1], b[i][1]
		tmp = []
		for j in b:
			tmp.append(str(j[1]))
		ans.append(tmp)
		x += 1
if x >=2 :
	print 'YES'
	for i in ans:
		print ' '.join(i)
else:
	print 'NO'
