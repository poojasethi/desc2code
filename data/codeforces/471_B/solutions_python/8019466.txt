n = int(raw_input())
h = map(int, raw_input().split())[:n]
a = [(x, i+1) for i, x in enumerate(h)]
a = sorted(a, key=lambda x: x[0])
p = []
for i in range(1, len(a)):
  if a[i][0] == a[i-1][0]:
    p.append(i)
    if len(p) >= 2: break
if len(p) < 2: print 'NO'
else:
  print 'YES'
  def output(a):
    print ' '.join(map(str, [x[1] for x in a]))
  output(a)
  for pi in p:
    a[pi-1], a[pi] = a[pi], a[pi-1]
    output(a)
