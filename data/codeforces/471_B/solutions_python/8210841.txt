n = input()
a = map(int,raw_input().split())
b = [x for x in sorted(enumerate(a),key=lambda x:x[1])]
i = []
s = 0
for x in xrange(1,n):
    if b[x][1] == b[x-1][1]:
        s += 1
        i.append((x-1,x))
if s >= 2:
    c = list(zip(*b)[0])
    print "YES"
    print " ".join([str(x+1) for x in c])
    for x,y in i[:2]:
        c[x],c[y] = c[y],c[x]
        print " ".join([str(x+1) for x in c])
else:
    print "NO"
