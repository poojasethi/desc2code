n = int(raw_input())
a = map(int, raw_input().split())

cnt = 1
for x in set(a) : 
    cnt *= a.count(x);
if cnt < 3 :
    print "NO"
    exit()
print "YES"
b = sorted([(a[i], i) for i in range(n)])
def display() : 
    print ' '.join([str(x[1] + 1) for x in b])
display()
mask = []
for i in range(0, len(b) - 1) : 
    if (b[i][0] == b[i + 1][0]) : mask.append(i)
x = mask[0]
b[x], b[x + 1] = b[x + 1], b[x]
display()
x = mask[1]
b[x], b[x + 1] = b[x + 1], b[x]
display()
