#!/usr/bin/env python


def read():
    n = input()
    idxValList = list(enumerate(map(int, raw_input().split())))
    return (n, idxValList)


def work((n, idxValList)):
    idxValList.sort(key=lambda x: x[1])
    
    ans = []
    for i in range(3):
        toPush = range(n)
        cnt = 0
        for j in range(n - 1):
            if cnt < i and idxValList[j][1] == idxValList[j + 1][1]:
                toPush[j], toPush[j + 1] = toPush[j + 1], toPush[j]
                cnt += 1

        if cnt < i:
            print "NO"
            return

        ans.append([idxValList[toPush[j]][0] + 1 for j in range(n)])

    print "YES"
    for i in range(3):
        print " ".join(map(str,ans[i]))
    

if __name__ == "__main__":
    work(read())
