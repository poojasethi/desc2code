import sys

n = int(sys.stdin.readline())
l = map(int, sys.stdin.readline().strip().split())
ans = ''
for i in range(0, n-1):
    while (l[i] > 0):
        ans = ans + 'P' + 'R' + 'L'
        l[i] = l[i] - 1
    ans = ans + 'R'
        
while (l[n-1] > 0):
    ans = ans + 'P' + 'L' + 'R'
    l[n-1] = l[n-1] - 1       
    
print ans
