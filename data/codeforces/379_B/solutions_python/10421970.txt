R=lambda:map(int,raw_input().split(' '))
n = input()
a = R()
s = ''
i = 0
p = 0
while True:
    if a[i] > 0:
        s += 'P'
        a[i] -= 1   
    if (i == n-1) or (i > p and i > 0):
        i -= 1
        s += 'L'
    else:
        i += 1
        s += 'R'  
    while p < n and a[p] == 0:
        p += 1
    if p >= n:
        break
print s
