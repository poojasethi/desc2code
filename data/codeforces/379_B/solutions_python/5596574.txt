n, a = int(raw_input()), map(int, raw_input().split())
i, s = 0, ''
while i < n - 1:
    d = min(a[i], a[i + 1])
    s += 'PRPL' * d
    if a[i] == d:
        i += 1
        a[i] -= d
        s += 'R'
    else:
        s += 'PRL' * (a[i] - d - 1) + 'PR'
        i += 2
        if i < n: s += 'R'
if i < n and a[-1]: s += 'PLR' * (a[-1] - 1) + 'P'
print s
