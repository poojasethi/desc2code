import sys

n = int(raw_input())
a = [int(x) for x in raw_input().split()]
b = [0] * n
res = ''
i = 0
while i < n:
    if b[i] == a[i]:
        i += 1
        res += 'R'
        continue
    if len(res) and res[-1] == 'P':
        if i > 0:
            res += 'LR'
        else:
            res += 'RL'
        continue
    res += 'P'
    b[i] += 1

print res.rstrip('R')
