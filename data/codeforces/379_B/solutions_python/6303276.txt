n = int(raw_input())
A = map(int, raw_input().split())

S = sum(A)
i = 0
dir = 1
res = []
while S > 0:
  if A[i] > 0:
    A[i] -= 1
    S -= 1
    res.append('P')
  if dir == 1:
    res.append('R')
  else:
    res.append('L')
  i += dir
  if i == 0:
    dir = 1
  elif i == n-1:
    dir = -1
print ''.join(res)
