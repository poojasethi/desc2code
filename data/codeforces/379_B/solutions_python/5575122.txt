from cStringIO import StringIO

n = int(raw_input())
a = map(int, raw_input().split())

res = StringIO()

for i in xrange(n):
    if i > 0:
        res.write('R')
    s = 'LR' if i > 0 else 'RL'
    for j in xrange(a[i]):
        res.write('P')
        if j + 1 < a[i]:
            res.write(s)

print res.getvalue()
