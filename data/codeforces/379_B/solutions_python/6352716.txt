n = input()
a = map(int, raw_input().split())
r = ''

s = sum(a)
d = 1
i = 0

while True:
    if a[i] > 0:
        a[i] -= 1
        r += 'P'

        s -= 1
        if s == 0:
            break

    if i == 0:
        d = 1
    elif i == n - 1:
        d = -1

    i += d
    r += 'R' if d == 1 else 'L'

print r
