n = int(raw_input())
a = map(int,raw_input().split())
v = [0] * n

s = 0
tt = 0
t = sum(a)
d = 'R'
l = []
while True:
    if v[s] < a[s]:
        v[s] += 1
        l += 'P'
        tt += 1

    if tt == t:
        break

    if s == n - 1:
        d = 'L'
    elif s == 0:
        d = 'R'

    l += d
    s += -1 if d == 'L' else 1

print ''.join(l)
