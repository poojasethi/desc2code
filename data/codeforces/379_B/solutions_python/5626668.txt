n = input()
a = map(int , raw_input().split())
i = 0
tot = sum(a)
s = ""
while i < n-1 :
	if a[i] == 0 : s += "R"
	else : 
		while a[i] > 1 : 
			s += "PR"		
			a[i] -= 1
			if a[i+1] > 0 : 
				s+="PL"
				a[i+1] -= 1
			else : s += "L"
	if a[i] == 1 : s+="PR"
	i+= 1
while a[n-1] > 1 :
	a[n-1] -= 1
	s += "PLR"
if a[n-1] == 1 : 
	s += "P"
print s
