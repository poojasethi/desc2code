n=int(raw_input())
a=map(int,raw_input().split())
s=""
while sum(a)!=0:
  i=0
  while i != n:
    #if a[i]==0:
    #  s+='R'      
    if a[i]!=0:
      a[i]-=1
      s+='P'
    #  if i+1!=n-1 and i+1!=n:
    #    s+='R'
    if i+1!=n:
      s+='R'
    i+=1

  if sum(a)!=0:
    s+='L'*(n-1)
  if sum(a)==0:
    break
#  print sum(a),s,a
print s
