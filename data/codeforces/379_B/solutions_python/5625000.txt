n = int(raw_input())
a = map(int, raw_input().split())
res = ''
for i in range(n - 1):
	while (a[i] > 1):
		res += 'PRL'
		a[i] -= 1
	if (a[i] == 1):
		res += 'P'
	if ((i == n - 2) and (a[-1] > 0)) or (i < n - 2):
		res += 'R'
while (a[-1] > 1):
	res += 'PLR'
	a[-1] -= 1
if (a[-1] == 1):
	res += 'P'
print res
