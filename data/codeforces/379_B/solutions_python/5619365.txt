n = input()
A = map(int, raw_input().split())

def fillCoins(A, pos, coins) : 
	ops = ""
	if coins > 0 : 
		ops += 'P'

	for c in range(1, coins) : 
		ops += 'RLP' if pos == 0 else 'LRP'

	return ops

ops = ''
for i in range(len(A)) : 
	if i != 0 : 
		ops += 'R'
		
	ops += fillCoins(A, i, A[i])

print ops
