#include<stdio.h>
#include<algorithm>
using namespace std;
#define ll __int64

int map[305][305];

int main()
{
	int i, j, u, v, c, k, n;
	scanf("%d", &n);
	for(i = 1; i <= n; i++)
		for(j = 1; j <= n; j++)
			scanf("%d", &map[i][j]);
	scanf("%d", &k);
	while(k--)
	{
		ll sum = 0;
		scanf("%d%d%d", &u, &v, &c);
		for(i = 1; i <= n; i++)
			for(j = 1; j <= n; j++)
			{
				map[i][j] = min(min(map[i][j], map[i][u]+map[v][j]+c), map[i][v]+map[u][j]+c);
				sum += map[i][j];
			}
		printf("%I64d ", sum/2);
	}
	puts("");
	return 0;
}
