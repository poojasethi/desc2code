#include <iostream>
#include <algorithm>
using namespace std;
int adj[301][301];
int main(){
	int n,k,a,b,c;
	cin>>n;
	for(int i=1;i<=n;++i)
		for(int j=1;j<=n;++j)
			cin>>adj[i][j];
	cin>>k;
	for(int i=0;i<k;++i){
		cin>>a>>b>>c;
		long long int sum = 0;
		
		for(int x=1;x<=n;++x)
			for(int y=1;y<=n;++y){
				adj[x][y]= min(adj[x][y],adj[x][a]+c+adj[b][y]);
				adj[x][y]= min(adj[x][y],adj[x][b]+c+adj[a][y]);
				if(y>x)sum+=adj[x][y];
			}
			cout<<sum<<endl;
	}
}
