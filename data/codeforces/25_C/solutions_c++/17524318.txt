#include<bits/stdc++.h>
using namespace std;
const int N=305;
int dist[N][N];
int main(){
    int n; cin>>n;
    for(int i=0; i<n; i++){
        for(int j=0; j<n; j++)
            cin>>dist[i][j];
    }
    int k; cin>>k;
    for(int i=0; i<k; i++){
        int u, v, w; cin>>u>>v>>w;
        u--; v--;
        long long ans=0;
        for(int i=0; i<n; i++)
            for(int j=0; j<n; j++)
                dist[i][j]=min(dist[i][j],min(dist[i][v]+w+dist[u][j],dist[i][u]+w+dist[j][v])),ans+=dist[i][j];
        cout<<ans/2<<" ";
    }
    return 0;
}
