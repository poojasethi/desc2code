#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <string.h>
#include <math.h>
using namespace std;
#define MXN 210
int n,num[MXN],total,cnt[MXN],c1,c2;
bool u1[MXN],u2[MXN];
void Init()
{
	scanf("%d",&n);
	n<<=1;
	for(int i=1;i<=n;i++) {scanf("%d",&num[i]);++cnt[num[i]];}
}
void solve()
{
	for(int i=10;i<=99;i++) if(cnt[i]>1) {c1++;c2++;u1[i]=u2[i]=1;}
	for(int i=10;i<=99;i++) if(cnt[i]==1) {c1<c2?(c1++,u1[i]=1):(c2++,u2[i]=1);}
	printf("%d\n",c1*c2);
	for(int i=1;i<=n;i++)
	  {
	  	if(u1[num[i]]) {u1[num[i]]=0;printf("1 ");}
	  	else if(u2[num[i]]) {u2[num[i]]=0;printf("2 ");}
	  	else if(c1<c2) {c1++;printf("1 ");}
	  	else {c2++;printf("2 ");}
	  }
    printf("\n");
}
int main()
{
	Init();
	solve();
    return 0;
}
