#include<iostream>
#include<algorithm>
#define X first
#define Y second 
using namespace std;
const int MAXN=100*20;
int v[MAXN];
int ans[2];
int f[MAXN];
int a[MAXN];
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=2*n;i++){
		cin>>a[i];
		f[a[i]]++;
	}
	int s=0;
	int k=0;
	for(int i=1;i<=101;i++){
		if(f[i]>=2){
			s+=f[i]-2;
			for(int j=1;j<=2*n;j++){
				if(a[j]==i){
					v[j]=(k%2)+1;
					k++;
				}
			}
		}
	}
	if(s%2==0)
		cout<<((2*n-s)/2)*((2*n-s)/2);
	else
		cout<<((2*n-s+1)/2)*((2*n-s-1)/2);
	for(int i=1;i<=2*n;i++){
		if(v[i]==0){
			v[i]=k%2+1;
			k++;
		}
	}
	cout<<'\n';
	for(int i=1;i<=2*n;i++)
		cout<<v[i]<<" ";
}
