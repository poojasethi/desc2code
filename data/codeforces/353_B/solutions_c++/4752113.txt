#include<iostream>
#include<algorithm>
#include<cstring>
#include<set>
using namespace std;
int main()
{
	int n;
	while (cin>>n) {
		n*=2;
		pair<int,int> nums[n];
		for(int i=0;i<n;i++)
		{
			cin>>nums[i].first;
			nums[i].second=i;
		}
		int ans[n];
		memset(ans,-1,sizeof ans);
		sort(nums,nums+n);
		int pos=0;
		set<int> s[2];
		for(int i=0;i<n;i++)
		{
			if(s[pos].count(nums[i].first)==0)
			{
				s[pos].insert(nums[i].first);
				ans[nums[i].second]=pos+1;
				pos^=1;
			}
		}
		for(int i=0;i<n;i++)
		{
			if(ans[nums[i].second]==-1)
			{
				ans[nums[i].second]=pos+1;
				pos^=1;
			}
		}
		cout<<s[0].size()*s[1].size()<<endl;
		for(int i=0;i<n;i++)
		{
			if(i!=0) cout<<" ";
			cout<<ans[i];
		}
		cout<<endl;
	}
	return 0;
}
