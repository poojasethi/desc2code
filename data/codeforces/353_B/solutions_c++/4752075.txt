#include<cstdio>
#include<map>
#include<vector>
#include<algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;

int n,b[205],t=1,dem[3];
map<int,vector<int> > a;

int main()
{
	int x;
	scanf("%d",&n);
	rep(i,2*n) scanf("%d",&x), a[x].push_back(i+1);
	for (map<int,vector<int> >::iterator it=a.begin();it!=a.end();it++)
	{
		if (it->second.size()<2)
		{
			rep(i,it->second.size()) b[it->second[i]]=t,dem[t]++,t=3-t;
		}
	}
	for (map<int,vector<int> >::iterator it=a.begin();it!=a.end();it++)
	{
		if (it->second.size()>=2)
		{
			rep(i,it->second.size()) b[it->second[i]]=t,t=3-t;
			dem[1]++;
			dem[2]++;
		}
	}
	printf("%d\n",dem[1]*dem[2]);
	rep(i,2*n) printf("%d ",b[i+1]);
}
