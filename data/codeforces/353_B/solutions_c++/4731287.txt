#include<stdio.h>
int n,a[205],ct[100],ct2,ct1,used[100],sol[205],Quant,temp;
int size1;
main()
{
	scanf("%d",&n);
	for (int i=0;i<2*n;i++)
	{
		scanf("%d",&a[i]);
		ct[a[i]]++;
	}
	for (int i=10;i<=99;i++)
	{
		if (ct[i]>=2)
			ct2++;
		else if (ct[i]==1)
			ct1++;
	}
	Quant = ( ct2+(ct1/2) )*( ct2+(ct1+1)/2 );
	
	for (int i=0;i<2*n;i++)
	{
		if (ct[a[i]]>=2 && used[a[i]]<2)
		{
			used[a[i]]++;
			sol[i] = used[a[i]];
			if (sol[i]==1)
				size1++;
		}
		else if (ct[a[i]]==1)
		{
			if (temp < ct1/2)
			{	sol[i]=1; temp++; size1++; }
			else
				sol[i]=2;
		}
	}

	printf("%d\n",Quant);
	for (int i=0;i<2*n;i++)
	{
		if (sol[i]!=0) printf("%d ",sol[i]);
		else if (size1<n) { printf("1 "); size1++; }
		else 			  printf("2 ");
	}
}
