#include <stdio.h>
#include <stdlib.h>
#include <string.h>
main()
{
	int n;
	int num[210];
	int ans[210];
	int shu[100]={0};
	int jilu[100]={0};
	int i,j,xx=0,yy=0;
	int t;
	scanf("%d",&n);
	int l=2*n;
	for(i=1;i<=l;i++)
	{
		scanf("%d",&num[i]);
		shu[num[i]]++;
	}
	    int x=1;
		for(i=1;i<=l;i++)
		{
			if(shu[num[i]]>1 && jilu[num[i]]==0)
			{
				jilu[num[i]]=1;
				xx++;
				yy++;
				t=num[i];
				for(j=1;j<=l;j++)
				{
					if(num[j]==t)
					{
						ans[j]=x;
						x=3-x;
					}
				}
			}
		}
		for(i=1;i<=l;i++)
		{
			if(shu[num[i]]==1)
			{
				ans[i]=x;
				if(x==1)
				xx++;
				else
				yy++;
				x=3-x;
			}
		}
		printf("%d\n",xx*yy);
		for(i=1;i<=l;i++)
		printf("%d ",ans[i]);
		return 0;
}
