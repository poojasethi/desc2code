#include<iostream>
#include<vector>
#include<cmath>
using namespace std;

int n,num[300],total;

bool mark[100];

vector<int> v[100],out;

int team[2];

int main(){
	cin>>n;
	for(int i=1;i<=2*n;i++){
		int tmp;cin>>tmp;
		v[tmp].push_back(i);
	}
	for(int i=10;i<100;i++)
		if(v[i].size()>1){
			num[v[i][0]]=1;
			num[v[i][1]]=2;
			team[0]+=1;
			team[1]+=1;
		}else if(v[i].size()==1){
			out.push_back(v[i][0]);
		}
	for(int i=0;i<out.size();i++){
		team[1-i%2]++;
		num[out[i]]=2-i%2;
	}
	total=team[0]*team[1];
	for(int i=1;i<=2*n;i++)
		if(num[i]==0){
			if(team[0]<n){
				num[i]=1;
				team[0]++;
			}else num[i]=2;
		}
	cout<<total<<endl;
	for(int i=1;i<=2*n;i++)
		cout<<num[i]<<" ";	
}
