#include <bits/stdc++.h>

using namespace std;

const int N = 101;
int ans[2 * N], cnt[N];
bool p[N];

int main() {
	int n, d[2] = {0, 0};
	cin >> n;
	for (int i = 0; i < 2 * n; i++) {
		int x; cin >> x;
		if (cnt[x] != 0) {
			ans[cnt[x] - 1] = 1;
			ans[i] = 2;
			if (!p[x])
				d[0]++, d[1]++;
			cnt[x] = 0;
			p[x] = true;
		}
		else cnt[x] = i + 1;
	}
	int w = 0;
	for (int i = 0; i < 100; i++)
		if (cnt[i] && !p[i])
			ans[cnt[i] - 1] = w % 2 + 1, d[w % 2]++, ++w;
	for (int i = 0; i < 100; i++)
		if (p[i] && cnt[i])
			ans[cnt[i] - 1] = w % 2 + 1, ++w;
	cout << d[0] * d[1] << endl;
	for (int i = 0; i < 2 * n; i++)
		cout << ans[i] << ' ';
	return 0;
}
