#include<iostream>

using namespace std;

int a[220];
int b[220];
int f[110];
int main()
{
	int n;
	cin >> n;
	
	for(int i=0;i<2*n;i++)
	{
		cin >> a[i];
		f[a[i]]++;
	}
	
	int s=0;
	int k=0;
	for(int i=0;i<110;i++)
		if(f[i]>=2)
		{
			s+=f[i]-2;
			for(int j=0;j<2*n;j++)
			{
				if(a[j]==i)
				{
					b[j]=(k%2)+1;
					k++;
				}
			}
		}
	if(s%2==0)
		cout << ((2*n-s)/2)*((2*n-s)/2)<<endl;
	else 
		cout << ((2*n-s-1)/2)*((2*n-s+1)/2)<<endl;
	
	for(int i=0;i<2*n;i++)
	{
		if(b[i]==0)
		{
			b[i]=(k%2)+1;
			k++;
		}
	}
	
	cout << b[0];
	for(int i=1;i<n*2;i++)
	{
		cout << " " << b[i];
	}
		
	cin >> n;
	return 0;
}
