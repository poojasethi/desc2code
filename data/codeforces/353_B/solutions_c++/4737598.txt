#include<stdio.h>
#include<string.h>
#define maxn 205
int x[maxn],f[maxn],g[maxn][maxn],a[maxn],p[maxn],q[maxn];
int main()
{
	int N,i,j,t,m,c=0;
	memset(f,0,sizeof(f));
	scanf("%d",&N);
	for(i=0;i<2*N;i++)
	{
		scanf("%d",&x[i]);
		t=x[i];
		f[t]++;
		g[t][f[t]]=i;
	}
	for(i=0;i<100;i++)
	{
		if(f[i]==1) a[g[i][1]]=1+(c++)%2;
		if(f[i]>1)
		{
			m=(f[i]/2)*2;
			for(j=1;j<=m;j++)
			{
				a[g[i][j]]=1+(c++)%2;
			}
		}
	}
	for(i=0;i<100;i++)
	{
		if(f[i]>1 && f[i]%2) a[g[i][f[i]]]=1+(c++)%2;
	}
	memset(p,0,sizeof(p));
	memset(q,0,sizeof(q));
	int cnt1=0,cnt2=0;
	for(i=0;i<2*N;i++)
	{
		if(a[i]==1)
		{
			if(!p[x[i]])
			{
				p[x[i]]=1;
				cnt1++;
			}
		}
		else
		{
			if(!q[x[i]])
			{
				q[x[i]]=1;
				cnt2++;
			}
		}
	}
	printf("%d\n",cnt1*cnt2);
	for(i=0;i<2*N;i++) printf("%d ",a[i]);
	return 0;
}
