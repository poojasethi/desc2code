#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
int id[220],a[220],ans[220],bo[2][220],s1,s2,i,j,k,n,x,v,y;
bool cmp(const int&x,const int&y){
	return a[x]<a[y];
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n+n;++i)scanf("%d",&a[i]),id[i]=i;
	sort(id+1,id+n+n+1,cmp);
	s1=s2=0;
	for(i=1;i<=n+n;++i){
		x=id[i];
		y=a[x];
		if(bo[0][y] && bo[1][y])continue;
		if(bo[0][y] && !bo[1][y])v=2;
		else
		if(!bo[0][y] && bo[1][y])v=1;
		else if(s1<s2)v=1;
		else v=2;
		ans[x]=v;
		bo[v-1][y]=1;
		if(v==1)s1++;
		else s2++;
	}
	printf("%d\n",s1*s2);
	for(i=1;i<=n+n;++i)if(!ans[i]){
		if(s1!=n)ans[i]=1,s1++;
		else ans[i]=2;
	}
	for(i=1;i<=n+n;++i)printf("%d ",ans[i]);
}
