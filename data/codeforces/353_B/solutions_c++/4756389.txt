#include<iostream>
using namespace std;

int a[220],b[220],f[110],s,k,n;
int main()
{
    cin >> n;
    for(int i=0;i<2*n;i++)
    {
        cin >> a[i];
        f[a[i]]++;
    }
    for(int i=0;i<110;i++)
        if(f[i]>=2)
        {
            s+=f[i]-2;
            for(int j=0;j<2*n;j++)
            {
                if(a[j]==i)
                {
                    b[j]=(k%2)+1;
                    k++;
                }
            }
        }
    if(s%2==0)cout<<((2*n-s)/2)*((2*n-s)/2)<<endl;
    else cout<<((2*n-s-1)/2)*((2*n-s+1)/2)<<endl;
    for(int i=0;i<2*n;i++)
        if(b[i]==0)
        {
            b[i]=(k%2)+1;
            k++;
        }
    for(int i=0;i<n*2;i++)
        cout << " " << b[i];
    return 0;
}
