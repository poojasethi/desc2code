#include<bits/stdc++.h>
using namespace std;
int a[210],cnt[210],b[210],n,flag;
int main()
{
	memset(cnt,0,sizeof(cnt));
	scanf("%d",&n);
	n=2*n;
	int x=0,y=0;
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		cnt[a[i]]++;
	}
	for(int i=10;i<=100;i++)
	{
		if(cnt[i]==1) x++;
		else if(cnt[i]>=2) y++;
	}
	
	
	printf("%d\n",(x/2+y)*(x-x/2+y));
	flag=0;
	for(int i=1;i<=n;i++)
	{
		if(cnt[a[i]]==1)
		{
//			printf("yo%d\n",a[i] );
			b[i]=1+flag;
			flag^=1;
		 	
		}
	}
	for(int i=10;i<=100;i++)
	{
		if(cnt[i]>=2)
		{
		//flag=1;	
		for(int j=1;j<=n;j++)
		{
			if(a[j]==i)
			{
				b[j]=1+flag;
		 		flag^=1;
				
		 		//printf("%d ",1+flag);		
			}
		}

		}
	}	
	for(int i=1;i<=n;i++)
		printf("%d ",b[i] );
	return 0;
}
