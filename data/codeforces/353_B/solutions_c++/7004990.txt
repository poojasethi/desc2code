#include <iostream>
#include <algorithm>

using namespace std;

int a[205];
int u[100];
int h[100];

int main()
{
	int n;
	cin >> n;
	for(int i=0; i<2*n; i++)
	{
		cin >> a[i];
		u[a[i]] ++;
	}
	int t1 = 1;
	int t2 = 2;
	int c = 0;
	for(int i=10; i<=99; i++)
		if(u[i])
		{
			if(u[i] > 1) c += 2;
			else c ++;
			if(u[i] == 1) 
			{
				h[i] = t1;
				t1 = 3 - t1;
			}
			else if(u[i] & 1)
			{
				h[i] = t2;
				t2 = 3 - t2;
			}
			else h[i] = 1;
		}
	int cc = c / 2;
	cout << ((c & 1) ? (cc * (cc + 1)) : (cc * cc)) << endl;
	for(int i=0; i<2*n; i++)
	{
		cout << h[a[i]] << ' ';
		h[a[i]] = 3 - h[a[i]];
	}
	cout << endl;
	return 0;
}
