#include <bits/stdc++.h>

using namespace std;
const int maxn = 1e5 + 5;
queue<int> q[100];
int bel[maxn], n;
int main(){
	int turn = 1, x = 0;
	scanf("%d", &n);
	for (int i = 0; i < 2 * n; ++i){
		scanf("%d", &x);
		q[x].push(i);
	}
	int s[3] = {0};
	for (int i = 10; i <= 99; ++i){
		if (q[i].size() <= 1){
			if (!q[i].empty()){
				bel[q[i].front()] = turn;
				s[turn] ++;
				turn = 3 - turn;
				q[i].pop();
			}
		}
		else{
			s[1]++;s[2]++;
			while (q[i].size() >= 2){
				bel[q[i].front()] = turn;
				q[i].pop();
				bel[q[i].front()] = 3 - turn;
				q[i].pop();
			}
		}
	}
	for (int i = 10; i <= 99; ++i){
		if (!q[i].empty()){
			bel[q[i].front()] = turn;
			turn = 3 - turn;
		}
	}
	printf("%d\n", s[1] * s[2]);
	for (int i = 0; i < 2 * n; ++i)
		printf("%d ", bel[i]);
	printf("\n");
	return 0;
}
