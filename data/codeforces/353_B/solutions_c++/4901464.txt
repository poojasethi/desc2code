#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 210
using namespace std;
int n,m,s1,s2;
int f[N],a[N],s[N],h[N],v[N];
int main() {
	scanf("%d",&n); m=n*2;
	for(int i =0;i < m;i++)
	{
		scanf("%d",&a[i]);
		s[a[i]]++;
	}
	for(int i = 10;i < 100;i++)
	{
        if (s[i])
        {
            if (s[i]>=2) s1++,s2++;
            else if (s1>s2) s2++,h[i]=2;
            else s1++,h[i]=1;
        }
	}
	printf("%d\n",s1*s2);
	for(int i = 0;i < m;i++)
	{
		int k=a[i];
		if (!v[k])
		{
			printf("%d ",h[k]?h[k]:1);
			v[k]++;
		}
		else if(v[k]==1)
		{
			printf("2 ");
			v[k]++;
		}
		else if(s1<n) printf("1 "),s1++;
		else printf("2 ");
	}
	puts("");
	return 0;
}
