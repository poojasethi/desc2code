#include<cstdio>
#include<algorithm>
#include<set>
#define MAXN 205
using namespace std;
struct node
{int id,v;}a[MAXN],k[MAXN];
int n,m,b[MAXN];
int c1,c2;
set<int>s1,s2;
bool cmp(node A,node B)
{return A.v>B.v;}
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n*2;i++)
	{
		a[i].id=i;
		scanf("%d",&a[i].v);
	}
	sort(a+1,a+n*2+1,cmp);
	k[++m]=a[1];k[++m]=a[2];
	for (int i=3;i<=n*2;i++)
	if (a[i].v!=a[i-1].v||a[i].v!=a[i-2].v)k[++m]=a[i];
	for (int i=1;i<=m;i++)
	{
		b[k[i].id]=(i&1)?1:2;
		if (i&1)s1.insert(k[i].v),c1++;
		else s2.insert(k[i].v),c2++;
	}
	for (int i=1;i<=n*2;i++)
	if (!b[i]&&c1<n)b[i]=1,c1++;
	else if (!b[i])b[i]=2,c2++;
	printf("%d\n",s1.size()*s2.size());
	for (int i=1;i<=n*2;i++)printf("%d ",b[i]);
}
