#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<queue>
#include<string.h>
using namespace std;
#define M 3004
#define eps 1e-8

int num[200];

int n;
int a[210];
int b[210];
int c[210];
int main()
{
	//freopen( "a.in", "r", stdin );
	cin>>n;
	for( int i = 1; i <= n*2; ++i ){
		cin>>a[i];
		++num[a[i]];
	}
	int k = 1;
	int cc[4];
	cc[1] = cc[2] = 0;
	for( int i = 1; i <= n*2; ++i ){
		if( num[a[i]] == 1 ){
			b[i] = k;
			cc[k]++;
			k = 3 - k;
		}
		else if( c[a[i]] == 1 )++c[a[i]], b[i] = 2, cc[2]++;
		else if( c[a[i]] == 0 ) ++c[a[i]], b[i] = 1, cc[1]++;
	}
	cout<<cc[1] * cc[2]<<endl;
	for( int i = 1; i <= 2*n; ++i ){
		if( b[i] == 0 ){
			b[i] = k;
			k = 3 - k;
		}
		printf( "%d", b[i] );
		if( i == n * 2 ) puts( "" );
		else printf( " " );
	}
}
