#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 222
using namespace std;
int a[N],num[N],pos[N],vis[N];
int main()
{
    int n;
	scanf("%d",&n);
	int m=n*2;
	for(int i =0;i < m;i++)
	{
		scanf("%d",&a[i]);
		num[a[i]]++;
	}
	int s1 = 0,s2 = 0;
	for(int i = 10;i < 100;i++)
	{
        if (num[i])
        {
            if (num[i]>=2) s1++,s2++;
            else if (s1>s2) s2++,pos[i]=2;
            else s1++,pos[i]=1;
        }
	}
	printf("%d\n",s1*s2);
	for(int i = 0;i < m;i++)
	{
		int k=a[i];
		if (!vis[k])
		{
			printf("%d ",pos[k]?pos[k]:1);
			vis[k]++;
		}
		else if(vis[k]==1)
		{
			printf("2 ");
			vis[k]++;
		}
		else if(s1<n) printf("1 "),s1++;
		else printf("2 ");
	}
	puts("");
	return 0;
}
