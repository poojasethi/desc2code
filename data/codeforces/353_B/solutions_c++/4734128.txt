#include <bits/stdc++.h>

using namespace std;

int N;
int ar[250];
int used[110];
int yaz[110][2];

int main(){
	
	cin >> N;
	
	for( int i=1 ; i<=2*N ; i++){ 	scanf(" %d",&ar[i]); 	used[ar[i]]++; }
	
	int x=0,y=0,k,l;
	
	for( int i=10 ; i<=99 ; i++){
		if( used[i] == 1 ){
			if( x < y ){
				x++;
				yaz[i][0]++;
				used[i]--;
			}
			else{
				y++;
				yaz[i][1]++;
				used[i]--;
			}
		}
		if( used[i] > 1 ){
			x++;
			y++;
			used[i]-=2;
			yaz[i][0]++;
			yaz[i][1]++;
		}
	}
	
	k=N-x; 	l=N-y;
	
	printf("%d\n",x*y);
	
	for( int i=1 ; i<=2*N ; i++ ){
		if( yaz[ar[i]][0] ){
			printf("1 ");
			yaz[ar[i]][0]--;
		}
		else if( yaz[ar[i]][1] ){
			printf("2 ");
			yaz[ar[i]][1]--;
		}
		else{
			if( k ) 	{printf("1 "); k--;}
			else printf("2 ");
		}
	}
	
	return 0;
	
}
