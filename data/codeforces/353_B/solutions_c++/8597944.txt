#include<cstdio>
#include<cstdlib>
int n,d[1005],Hash[1005],ans[1006],sum[10];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=2*n;i++){
		scanf("%d",&d[i]);
		Hash[d[i]]++;
	}
	int flag=0;
	for(int j=10;j<=99;j++){
		if(Hash[j]>1){
			for(int i=2*n;i>=1;i--)
			if(d[i]==j){
				ans[i]=flag+1;
				flag++; flag%=2;
				break;
			}
			for(int i=1;i<=2*n;i++)
			if(d[i]==j){
				ans[i]=flag+1;
				flag++; flag%=2;
				break;
			}
			sum[1]++; sum[2]++;
		}
		else{
			for(int i=1;i<=2*n;i++)
			if(d[i]==j){
				ans[i]=flag+1;
				sum[ans[i]]++;
				flag++; flag%=2;
			}
		}
	}
	for(int i=1;i<=2*n;i++)
	if(!ans[i]){
		ans[i]=flag+1;
		flag++; flag%=2;
	}
	printf("%d\n",sum[1]*sum[2]);
	for(int i=1;i<=2*n;i++) printf("%d ",ans[i]);
}
