#include <iostream>
#include <set>
#include <algorithm>
using namespace std;
#define MAXN 111
int N,B[MAXN<<1];
pair<int,int> P[MAXN<<1];
set<int> S[2];
bool Vis[MAXN<<1];
int main()
{
    cin>>N;
    N<<=1;
    for(int n=0;n<N;++n)
    {
        cin>>P[n].first;
        P[n].second=n;
    }
    sort(P,P+N);
    int d=0;
    for(int n=0;n<N;++n)
        if(S[d].count(P[n].first)==0)
        {
            S[d].insert(P[n].first);
            B[P[n].second]=d+1;
            d^=1;
            Vis[n]=true;
        }
    for(int n=0;n<N;++n)
        if(!Vis[n])
        {
            B[P[n].second]=d+1;
            d^=1;
        }
    cout<<S[0].size()*S[1].size()<<endl;
    for(int n=0;n<N;++n)
        cout<<B[n]<<" ";
    return 0;
}
