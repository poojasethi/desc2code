#include <stdio.h>
#include <string.h>

int main()
{
int n,i,j;
int s,k;
int a[300],b[300];
int f[200];
while(scanf("%d",&n)==1)
{	
	memset(b,0,sizeof(b));
	memset(f,0,sizeof(f));
	for(i=1;i<=2*n;i++)
	{
		scanf("%d",&a[i]);
		f[a[i]]++;
	}
	s=0;
	k=0;
	for(i=10;i<=99;i++)
	{
		if(f[i]>=2)
		{
			s+=f[i]-2;
			for(j=1;j<=2*n;j++)
			{
				if(a[j]==i)
				{
					b[j]=(k%2)+1;
					k++;
				}
			}
		}
	}
	if(s%2==0)
	{
		printf("%d\n",((2*n-s)/2)*((2*n-s)/2));
    }
	else 
	{
	    printf("%d\n",((2*n-s-1)/2)*((2*n-s+1)/2));
    }
	for(i=1;i<=2*n;i++)
	{
		if(b[i]==0)
		{
			b[i]=(k%2)+1;
			k++;
		}
	}
	printf("%d",b[1]);
	for(i=2;i<=2*n;i++)
	{
		printf(" %d",b[i]);
	}
	printf("\n");
}
return 0;
}
