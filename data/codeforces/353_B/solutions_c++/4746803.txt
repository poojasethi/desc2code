#include <cstdio>
#include <iostream>
using namespace std;
#define	N 10010
int n,a[210],u[N],va[100],vb[100],na,nb;
int main () {
	int i,j,ans;
	scanf("%d",&n);
	n*=2;
	for(i=0;i<n;i++) {
		scanf("%d",&a[i]);
		u[a[i]]++;
	}
	na=nb=0;
	for(i=10;i<100;i++)
		if(u[i]>=2) {
			va[i]++;vb[i]++;
			na++;nb++;
			u[i]-=2;
		} else if(u[i]) {
			if(na<nb) {va[i]++;na++;}
			else {vb[i]++;nb++;}
			u[i]--;
		}
	for(i=10;i<100;i++)
		while(u[i]) {
			if(na<nb) {va[i]++;na++;}
			else {vb[i]++;nb++;}
			u[i]--;
		}
	ans=0;
	for(i=10;i<100;i++)
		if(va[i])
			for(j=10;j<100;j++)
				if(vb[j]&&u[i*100+j]==0) {ans++;u[i*100+j]=1;}
	printf("%d\n",ans);
	for(i=0;i<n;i++)
		if(va[a[i]]) {printf("1 ");va[a[i]]--;}
		else {printf("2 ");vb[a[i]]--;}
	printf("\n");
	return 0;
}
