#include<cstdio>
const int N=201;
int n,sum,a[N],b[N],c[N];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=2*n;i++)
    {
        scanf("%d",&a[i]);
        b[a[i]]++;
        if(b[a[i]]<=2)sum++;
    }
    if(sum&1)printf("%d\n",(sum/2)*(sum/2+1));else printf("%d\n",(sum/2)*(sum/2));
    int k=1;
    for(int i=10;i<100;i++)
        if(b[i]>=2)
            for(int j=1;j<=2*n;j++)
                if(a[j]==i)c[j]=k,k=3-k;
    for(int i=1;i<=2*n;i++)
        if(!c[i])c[i]=k,k=3-k;
    for(int i=1;i<=2*n;i++)printf("%d ",c[i]);
}
