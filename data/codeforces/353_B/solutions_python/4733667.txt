from collections import defaultdict
import operator

n = int(raw_input())
data = raw_input().split()
final = [""] * (2*n)
index = defaultdict(list)
for i in xrange(len(data)):
    index[int(data[i])].append(i)
current = False
uniques = [0, 0]
#duplicates = 0
for elem in sorted(index, key=lambda k: len(index[k]), reverse=True):
    repeats = index[elem]
    uniques[int(current)] += 1
    if len(repeats) > 1:
        uniques[int(not current)] += 1
        #duplicates += 1
    for ind in repeats:
        final[ind] = str(int(current) + 1)
        current = not current

total = uniques[0] * uniques[1]

print total
print " ".join(final)
