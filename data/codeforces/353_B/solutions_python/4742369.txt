n=input()
c=[[] for _ in [0]*100]
a=map(int,raw_input().split())
for i in range(n*2):c[a[i]]+=[i]
d=[0]*256
R=range(100)
j=1
z=[0,set(),set()]
for i in R:
  if 1==len(c[i]):
    z[j].add(i)
    d[c[i][0]]=j
    j=3-j
for i in R:
  if len(c[i])>1:
    while len(c[i])>1:
      z[1].add(i)
      z[2].add(i)
      d[c[i].pop()]=1
      d[c[i].pop()]=2
    if c[i]:
      z[j].add(i)
      d[c[i][0]]=j
      j=3-j
print len(z[1])*len(z[2])
print ' '.join(map(str,d[:n*2]))
