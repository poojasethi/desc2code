n = int(raw_input())
a = list(map(int,raw_input().split()))
vm = [[] for i in range(100)]
for i in range(2*n):
	vm[a[i]].append(i)
ans=[0]*2*n
mul=[]
cur=1
x=[0,0]
for i in range(10,100):
	if len(vm[i]) == 1 :
		ans[vm[i][0]] = cur
		x[cur-1]+=1
		cur=3-cur
	if len(vm[i]) > 1: 
		ans[vm[i][0]]=1
		ans[vm[i][1]]=2
		x[0]+=1
		x[1]+=1
		for j in range(2,len(vm[i])):
			mul.append(vm[i][j])
for i in range(len(mul)):
	ans[mul[i]]=cur
	cur=3-cur
print(x[0]*x[1])
s=' '.join(map(str,ans))
print(s)
