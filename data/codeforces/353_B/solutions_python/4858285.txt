#!/usr/bin/python
R = lambda: map(int, raw_input().split())
n = input()
a = R()
c, h = [0]*100, [[0]*100 for i in xrange(2)]

q = [[] for i in xrange(100)]
for i in xrange(2*n):
	c[a[i]]+=1
	q[a[i]].append(i)

l, r = 0, 0
for i in xrange(10, 100, 1):
	if c[i] > 1:
		h[0][i]+=1
		h[1][i]+=1
		l+=1
		r+=1
		c[i]-=2
	elif c[i] == 1:
		if l > r:
			h[1][i]+=1
			r+=1;
		else:
			h[0][i]+=1
			l+=1
		c[i] = 0

for i in xrange(10, 100, 1):
	h[0][i]+=c[i]/2
	h[1][i]+=c[i]/2
	l+=c[i]/2
	r+=c[i]/2
	if c[i] % 2 == 1:
		if l > r:
			h[1][i]+=1
			r+=1;
		else:
			h[0][i]+=1
			l+=1
l, r = 0, 0
for i in xrange(10, 100, 1):
	if h[0][i] > 0: l+=1
	if h[1][i] > 0: r+=1

print l * r
c = [False] * (2*n)
for i in xrange(10, 100, 1):
	for j in xrange(h[0][i]):
		x = q[i].pop(0)
		c[x] = True

for i in xrange(2*n):
	if c[i]: print "1",
	else: print "2",
print
