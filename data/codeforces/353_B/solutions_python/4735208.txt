#!/usr/bin/env python

from collections import Counter

n = int(raw_input())
a = map(int, raw_input().split())
count = Counter(a)

uniq_left = uniq_right = 0
left = []
right = []
pairs = count.items()
pairs.sort(key=lambda x: x[1], reverse=True)
for n, c in pairs:
    if c >= 2:
        uniq_left += 1
        uniq_right += 1
        to_l = c / 2
        to_r = c - to_l
        if len(left) < len(right):
            to_l, to_r = to_r, to_l
        left.extend([n] * to_l)
        right.extend([n] * to_r)
    else:
        if len(left) < len(right):
            uniq_left += 1
            left.append(n)
        else:
            uniq_right += 1
            right.append(n)

#print uniq_left, uniq_right
#print left, right
#print '---'

print uniq_left * uniq_right
for i in a:
    if i in left:
        left.remove(i)
        print 1,
    else:
        right.remove(i)
        print 2,
