#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int main(){
	int n,a,c[100]={0},f[100]={0},p[100]={0},q[100]={0},t;
	char x[21],y[21],b,d;
	cin>>x>>y>>n;
	for(int i=0;i<n;++i){
		cin>>a>>b>>t>>d;
		if(b=='h'){
			c[t]++;
			if((d=='r'||c[t]==2)&&p[t]==0){
				cout<<x<<" "<<t<<" "<<a<<endl;
				p[t]=1;}}
		else if(b=='a'){
			f[t]++;
			if((d=='r'||f[t]==2)&&q[t]==0){
				cout<<y<<" "<<t<<" "<<a<<endl;
				q[t]=1;}}}
	return 0;
}
