#include <bits/stdc++.h>
using namespace std;

int tab[2][100];

char slo[2][100];

int main()
{
	int n,t,a,d;
	char c,k;
	scanf("%s%s",slo[0],slo[1]);
	scanf("%d",&n);
	for (int i=0;i<n;i++)
	{
		scanf("%d %c%d %c",&t,&c,&a,&k);
		d = c=='h'? 0:1;
		if ( tab[d][a] > 1 ) continue;
		tab[d][a] ++;
		if ( k=='r' ) tab[d][a] ++;
		if ( tab[d][a] > 1 ) printf("%s %d %d\n",slo[d],a,t);
	}
}
