#include<iostream>
using namespace std;
int c[2][100];
int main() {
	string names[2];
	cin >> names[0] >> names[1];
	int n;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		int k, t;
		cin >> t;
		char ca, cb;
		cin >> ca >> k >> cb;
		if (ca == 'h')
			ca = 0;
		else
			ca = 1;
		if (c[ca][k] > 1)
			continue;
		if (cb == 'y')
			cb = 1;
		else
			cb = 2;
		c[ca][k] += cb;
		if (c[ca][k] > 1)
			cout << names[ca] << " " << k << " " << t << endl;
	}
}
