#include <bits/stdc++.h>
using namespace std;

int c[2][110] , n;
string s[2];
bool f[2][110];

int main(){
	cin >> s[0] >> s[1] >> n;
	for (int i = 0 ; i < n ; ++ i)
	{
		char a , b; int x , y , f1 , f2;
		cin >> x >> a >> y >> b;
		if (a == 'h') f1 = 0;
		else f1 = 1;
		if (b == 'y') f2 = 1;
		else f2 = 2;
		c[f1][y] += f2;
		if (c[f1][y] >= 2 && !(f[f1][y])){
			cout << s[f1] << " " << y << " " << x << endl;
			f[f1][y] = 1;
		}
	}
}
