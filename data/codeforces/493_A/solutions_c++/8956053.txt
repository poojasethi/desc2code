#include <iostream>
#include <string>
using namespace std;
string s[2];
int m[2][100];
int main()
{
    char team, card;
    int time, player, n, i, tt, car;
    cin>>s[0];
    cin>>s[1];
    cin>>n;
    for (i=0;i<n;i++)
    {
		cin>>time>>team>>player>>card;
		if (team=='a') tt=1;
			else tt=0;
		if (card=='y') car=1;
			else car=2;
		if (m[tt][player]<2) {m[tt][player]+=car;
								if (m[tt][player]>=2) cout<<s[tt]<<' '<<player<<' '<<time<<endl;}
    }
}
