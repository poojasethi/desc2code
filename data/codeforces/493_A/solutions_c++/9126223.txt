#include<iostream>
#include<string>
using namespace std;
int cnt[2][110];
string s[2];
int main(){
	cin>>s[0]>>s[1];
	int m;
	cin>>m;
	for(int i=0;i<m;i++){
		int t,x;
		char team,card;
		cin>>t>>team>>x>>card;
		int tm=(team=='a');
		int cr=1+(card=='r');
		cnt[tm][x]+=cr;
		if(cnt[tm][x]>=2){
			cout<<s[tm]<<" "<<x<<" "<<t<<endl;
			cnt[tm][x]=-2*m-100;
		}
	}
}
