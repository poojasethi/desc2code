#include <bits/stdc++.h>
using namespace std;
int a[2][99];
string s[3];
char ch,cs;
int main()
{
    int m,x,y,p;
    cin>>s[1]>>s[2];
    cin>>m;
    for(int i=1;i<=m;i++)
    {
        cin>>x>>ch>>y>>cs;
        if(ch=='h')
            p=1;
        else
            p=2;
        a[p][y]++;
        if((a[p][y]==2||cs=='r')&&a[p][y]<10)
        {
            cout<<s[p]<<" "<<y<<" "<<x<<"\n";
            a[p][y]=10;
        }
    }
}
