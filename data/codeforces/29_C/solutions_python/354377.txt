#!/usr/bin/python

n = int(raw_input())
x = []
ind = {}
nc = 0
cnt = [0] * (2 * n)
num = [0] * (2 * n)
g = [0] * (2 * n)
for i in xrange(2 * n):
    g[i] = []
for i in xrange(n):
    x.append(raw_input().split())
    for j in xrange(2):
        x[i][j] = int(x[i][j])
        if(x[i][j] not in ind):
            ind[x[i][j]] = nc
            num[nc] = x[i][j]
            nc += 1
        cnt[ind[x[i][j]]] += 1;
    g[ind[x[i][0]]].append(ind[x[i][1]])
    g[ind[x[i][1]]].append(ind[x[i][0]])

a = 0

for i in xrange(n):
    if(cnt[i] == 1):
        a = i
        break

us = [0] * (2 * n)
us[a] = 1

print num[a],
for it in xrange(n):
    b = g[a][0] 
    if (us[b]):
        b = g[a][1]
    us[b] = 1
    a = b
