#! /usr/bin/env python

from sys import stdin

if __name__=='__main__':
	n=int(stdin.readline().strip())
	edges=[map(int,stdin.readline().split()) for i in xrange(n)]
	G={}
	indegree={}
	nodes=set()
	for (A,B) in edges:
		indegree[A]=indegree.get(A,0)+1
		indegree[B]=indegree.get(B,0)+1
		adj_listA=G.get(A,[])
		adj_listB=G.get(B,[])
		adj_listA.append(B)
		adj_listB.append(A)
		G[A]=adj_listA
		G[B]=adj_listB
		nodes.add(A)
		nodes.add(B)
	
	nodes=list(nodes)	
	seen=set()
	#print nodes
	#print indegree
	start=filter(lambda x: indegree[x]==1, nodes)[0]
	#print start
	path=[]
	cur=start
	#print G
	while cur not in seen:
		#print cur
		path.append(cur)
		seen.add(cur)
		next=G[cur]
		#print next
		if len(next)==1:
			cur=next[0]
		else:
			if next[0] in seen:
				cur=next[1]
			else:
				cur=next[0]
	print " ".join(map(str,path))
