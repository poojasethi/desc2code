#!/usr/bin/python
import sys
n = int(sys.stdin.readline())
d = {}
for i in xrange(n):
    line = sys.stdin.readline()
    a, b = map(int, line.split(' '))
    L1 = d.setdefault(a,[])
    L1.append(b)
    d[a]=L1
    L2 = d.setdefault(b,[])
    L2.append(a)
    d[b]=L2

for node in d.iterkeys():
    if len(d[node])==1:
        end = node
        break
L    = [end]
prev = end
end  = d[end][0]
while len(d[end]) != 1: 
    L.append(end)
    if d[end][0]==prev:
        prev = end
        end = d[end][1]
    else:
        prev = end
        end  = d[end][0]
L.append(end)
L = map(str,L)
print ' '.join(L)
