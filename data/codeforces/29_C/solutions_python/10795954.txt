from collections import defaultdict
T = input()

deg = defaultdict(int)
graph = defaultdict(list)
for c in range(T):
  a,b = raw_input().split()
  deg[a] += 1
  deg[b] += 1
  graph[a].append(b)
  graph[b].append(a)

start = None
end = None
for node, degree in deg.items():
  if degree == 1:
    if start is None:
      start = node
    else:
      end = node

ans = ''
last = None
while start != end:
  ans += start + ' '
  bs = graph[start]
  for b in bs:
    if b != last:
      last = start
      start = b
      break

print ans + start
