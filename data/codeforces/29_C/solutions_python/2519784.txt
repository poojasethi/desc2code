n=input()
g,d={},{}
for _ in xrange(n):
  a,b=map(int,raw_input().split())
  g[a]=g.get(a,0)^b
  g[b]=g.get(b,0)^a
  d[a]=d.get(a,0)+1
  d[b]=d.get(b,0)+1
for a,c in d.items():
  if 1==c:
    b=g[a]
    v=[a]
    for i in xrange(n):
      v.append(b)
      a,b=b,g[b]^a
    print ' '.join(map(str,v))
    break
