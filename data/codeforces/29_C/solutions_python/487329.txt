from collections import defaultdict

n = int(raw_input())
t = defaultdict(list)
for _ in xrange(n):
    a, b = map(int, raw_input().split())
    t[a].append(b)
    t[b].append(a)
for key, values in t.items():
    if len(values) == 1:
        k = key
print k,
while len(t) > 0:
    print t[k][0],
    nk = t[k].pop(0)
    t[nk].remove(k)
    if not t[k]: del t[k]
    if not t[nk]: del t[nk]
    k = nk
