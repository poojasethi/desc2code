from bisect import bisect
n = int( raw_input())
numbers = []
stamps = []
for i in range(0, n):
    stamp = map(int, raw_input().split())
    stamps.append(stamp)
    numbers.append(stamp[0])
    numbers.append(stamp[1])
if n==1:
    print stamps[0][0], stamps[0][1]
    exit()
numbers = sorted(set(numbers))
nodes = [[] for i in range(0, n+1)]
for stamp in stamps:
    first = bisect(numbers, stamp[0])-1
    second = bisect(numbers, stamp[1])-1
    hast = False
    for i in nodes[first]:
        if  i == second:
            hast = True
    if not hast:
        nodes[first].append(second)
    
    hast = False
    for i in nodes[second]:
        if  i == first:
            hast = True
    if not hast:
        nodes[second].append(first)
mabda = None
for i in range(0, n):
    if len(nodes[i])==1:
        mabda = i
tartib = []
tartib.append(mabda)
ghabli = mabda
n = nodes[mabda][0]
while True:
    tartib.append(n)
    if nodes[n][0] == ghabli:
        ghabli = n
        n = nodes[n][1]
    else:
        ghabli = n
        n = nodes[n][0]
    if len(nodes[n]) == 1:
        tartib.append(n)      
        print " ".join(map(lambda x: str(numbers[x]), tartib))
        exit()
