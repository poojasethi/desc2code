#include<cstdio>
#include<map>
#include<vector>
using namespace std;
map<int,vector<int> > m;
int main(){
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		m[x].push_back(y);
		m[y].push_back(x);
	}
	int a;
	for(map<int,vector<int> >::iterator i=m.begin();i!=m.end();i++)
	if(i->second.size()==1) a=i->first;
	printf("%d",a);
	int p=-1;
	while(n--)
	{
		if(m[a][0]!=p)
		{
			p=a;
			a=m[a][0];
		}
		else
		{ 
		    p=a;
		    a=m[a][1];
		}
		printf(" %d",a);
	}
	return 0;
}
