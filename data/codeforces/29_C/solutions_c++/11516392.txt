#include <bits/stdc++.h>

using namespace std;
int n;
map<int,vector<int> > mp;

int main()
{
    #ifndef ONLINE_JUDGE
    freopen("stamps.i2","r",stdin);
    freopen("stamps.o2","w",stdout);
    #endif // ONLINE_JUDGE
    cin >> n;
    int u, v;
    for(int i = 0; i < n; i++){
        cin >> u >> v;
        mp[u].push_back(v); mp[v].push_back(u);
    }
    for(typeof(mp.begin()) it = mp.begin(); it != mp.end(); it++)
        if(it->second.size()==1) u = it->first;
    cout << u ;
    v = -1;
    for(int i = 0; i < n; i++){
        if(mp[u][0] != v) u = mp[v = u][0]; else u = mp[v = u][1];
        cout <<" "<< u;
    }
    return 0;
}
