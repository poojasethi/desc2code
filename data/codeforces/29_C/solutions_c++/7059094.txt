#include<iostream>
#include<vector>
#include<map>
using namespace std;
map<int,vector<int> >a;
void d(int now,int l){
	cout<<now<<' ';
	for(int i=0;i<a[now].size();i++)
		if(a[now][i]!=l){
			d(a[now][i],now);
			return;
		}
}
int main(){
	int n,x,y;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>x>>y;
		a[y].push_back(x);
		a[x].push_back(y);
	}
	for(map<int,vector<int> >::iterator i=a.begin();i!=a.end();i++)if(i->second.size()==1)
	{
		d(i->first,0);
		break;
	}
	return 0;
}
