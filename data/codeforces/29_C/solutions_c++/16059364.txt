#include<bits/stdc++.h>

using namespace std;

map < int , int > A, B;

int n,u,v,i,p,t;

int main(){

	cin >> n;

	for(i=1;i<=n;i++){
		cin >> u >> v;
		
		if(!A[u])
			A[u] = v;
		else
			B[u] = v;

		if(!A[v])
			A[v] = u;
		else
			B[v] = u;
	}

	while(v){
		t = v;
		if(A[v] != p)
			v = A[v];
		else
			v = B[v];
		p = t;
	}

	while(p){
		printf("%d ",p);
		t = p;
		if(A[p] != v)
			p = A[p];
		else
			p = B[p];
		v = t;
	}  

	return 0;
}
