#include<iostream>
#include<vector>
#include<map>
#define pb push_back
using namespace std;
const int maxn=10010;
map<int,vector<int> > adj;
int dfs(int x,int l){
	cout<<x<<' ';
	for(int i=0;i<adj[x].size();i++)
		if(adj[x][i]!=l){
			dfs(adj[x][i],x);
			break;
		}
}
int main(){
	int n,u,v;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>u>>v;
		adj[u].pb(v);
		adj[v].pb(u);
	}
	map<int,vector<int> >::iterator it;
	for(it=adj.begin();it!=adj.end();it++)
		if(it->second.size()==1){
			dfs(it->first,-1);
			break;
		}
		cout<<endl;
}
