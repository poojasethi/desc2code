#include<cstdio>
#include<vector>
#include<map>
using namespace std;
int n;
map<int, vector<int> > m;
map<int, vector<int> >::iterator p;
void dfs(int now,int h)
{
	printf("%d ",now);
	for(int i=0;i<m[now].size();i++)
	{
		if(m[now][i]!=h)
			dfs(m[now][i],now);
	}
}
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		int x,y;
		scanf("%d %d",&x,&y);
		m[x].push_back(y);
		m[y].push_back(x);
	}
	for(p=m.begin();p!=m.end();p++)
	{
		if(p->second.size()==1)
		{
			dfs(p->first,-1);
			break;
		}
	}
	printf("\n");
	return 0;
}
