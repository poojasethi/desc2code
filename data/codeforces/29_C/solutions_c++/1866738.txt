#include <iostream>
#include<map>
#include<vector>
using namespace std;
map<int,vector<int> >adj;
int dfs(int x,int k)
{
    cout<<x<<'\n';
    for(int i=0;i<adj[x].size();i++)
    {
        if(adj[x][i]!=k)
        {dfs(adj[x][i],x);break;}
    }
}
int main()
{
   int n;cin>>n;
   int u,v;
   for(int i=0;i<n;i++)
   {
       cin>>u>>v;
       adj[u].push_back(v);
       adj[v].push_back(u);
   }
  map<int,vector<int> >::iterator it;
   for(it=adj.begin();;it++)
   {
       if(it->second.size()==1)
       {
           dfs(it->first,-1);break;
       }
   }



    return 0;
}
