#include<iostream>
#include<map>
#include<vector>
using namespace std;
map<int,vector<int> > e;
map<int,bool> vis;
int main()
{
	int n,a,b,p;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a>>b;
		e[a].push_back(b);
		e[b].push_back(a);
	}
	for(map<int,vector<int> >::iterator ti=e.begin();ti!=e.end();ti++)
		if(ti->second.size()==1) p=ti->first;
	do
	{
		cout<<p<<' ';
		vis[p]=true;
		if(!vis[e[p][0]]) p=e[p][0];
		else p=e[p][1];
	}while(e[p].size()!=1);
	cout<<p<<endl;
}
