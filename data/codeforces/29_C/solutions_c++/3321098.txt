#include<iostream>
#include<vector>
#include<map>
#include<set>
using namespace std;

//set<int> a[10001];
map<int,vector<int> > m;
map<int,bool> u;
int b;
void dfs(int x)
{
	u[x]=true;cout<<x<<" ";
	for(int i=0;i<m[x].size();i++)
	   if(!u[m[x][i]]) dfs(m[x][i]);
}

int main()
{
	int n,x,y;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>x>>y;
		m[x].push_back(y);
		m[y].push_back(x);
	}
	for(__typeof(m.begin()) i=m.begin();i!=m.end();i++)
	    if(i->second.size()==1) b=i->first;
	dfs(b);
	return 0;
}
