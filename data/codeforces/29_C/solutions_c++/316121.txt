#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<map>
using namespace std;
int N,a,b;
map<int,vector<int> > mm;
int main(){
	cin>>N;
	for(int i=0; i<N; i++){
		cin>>a>>b;
		mm[a].push_back(b);
		mm[b].push_back(a);
	}
	map<int,vector<int> >::iterator it;
	int pos,to;
	for(it=mm.begin(); it!=mm.end(); it++){
		if(it->second.size()==1){
			pos=it->first;
			break;
		}
	}
	cout<<pos<<" "; to=mm[pos][0];
	while(mm[to].size()>1){
		int t=to;
		cout<<to<<" ";
		if(mm[to][0]!=pos)
			to=mm[to][0];
		else
			to=mm[to][1];
		pos=t;
	}
	cout<<to<<"\n";
	return 0;
}
