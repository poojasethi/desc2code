#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<map>
#define pb push_back 
using namespace std;
map<int,vector<int> >e;
map<int,bool>vis;
int main(){
	int n,a,b,p;
	cin>>n;
	while(n--){
		cin>>a>>b;
		e[a].push_back(b);
		e[b].push_back(a);
	}
	for(map<int,vector<int> >::iterator it=e.begin();it!=e.end();it++)if((it->second).size()==1)p=it->first;
	do{
		vis[p]=1;
		cout<<p<<' ';
		if(vis[e[p][0]])p=e[p][1];
		else p=e[p][0];
	}while(e[p].size()>1);
	cout<<p;
	return 00;
}
