#include <iostream>
#include <map>
#include <vector>

using namespace std;

int main()
{
  int n;
  cin>>n;
  map<int,vector<int> > m;
  for (int i=0;i<n;i++) {
    int a,b;
    cin>>a>>b;
    m[a].push_back(b);
    m[b].push_back(a);
  }
  for (map<int,vector<int> >::iterator it=m.begin();it!=m.end();it++) {
    if ((it->second).size()==1) {
      int ant=it->first;
      cout<<ant;
      int act=(it->second)[0];
      while (m[act].size()==2) {
	cout<<" "<<act;
	int actaux=act;
	if (m[act][0]==ant) {
	  act=m[act][1];
	} else {
	  act=m[act][0];
	}
	ant=actaux;
      }
      cout<<" "<<act<<endl;
      break;
    }
  }
}
