#include <bits/stdc++.h>

using namespace std;

int n;
map <int, int> out;
set <int> stuff;

int main() {
  scanf ("%d", &n);
  for (int i = 0; i < n; i++) {
    int a, b;
    scanf ("%d %d", &a, &b);
    out[a] += b;
    out[b] += a;
    if (stuff.count(a) != 0) {
      stuff.erase(a);
    }
    else {
      stuff.insert(a);
    }
    if (stuff.count(b) != 0) {
      stuff.erase(b);
    }
    else {
      stuff.insert(b);
    }
  }
  int s = *stuff.begin(), t = 0;
  for (int i = 0; i <= n; i++) {
    printf ("%d ", s);
    t = out[s] - t;
    swap (s, t);
  }
  printf ("\n");
}
