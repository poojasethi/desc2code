#include<iostream>
#include<map>
#include<vector>
using namespace std;
map<int,vector<int> > m;
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		int x,y;
		cin>>x>>y;
		m[x].push_back(y);
		m[y].push_back(x);
	}
	int a;
	for(map<int,vector<int> >::iterator i=m.begin();i!=m.end();i++)
	if(i->second.size()==1)a=i->first;
	cout<<a;
	int p=-1;
	for(int i=0;i<n;i++){
		if(m[a][0]!=p)
		a=m[p=a][0];
		else
		a=m[p=a][1];
		cout<<' '<<a;
	}
	return 0;
}
