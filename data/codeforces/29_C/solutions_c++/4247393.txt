#include <iostream>
#include <cstdio>
#include <vector>
#include <map>
using namespace std;
vector <int> g[100001];
map <int,bool> u;
map <int,int> q,p;
int n,i,j,a,b,c,r=1;
int main()
{
cin>>n;
for (i=0;i<n;i++)
{
cin>>a>>b;
if (q[a]==0)
q[a]=r++;
if (q[b]==0)
q[b]=r++;
p[q[a]]=a;
p[q[b]]=b;
a=q[a];
b=q[b];
g[a].push_back(b);
g[b].push_back(a);
}
for (i=1;i<=n;i++)
{
if (g[i].size()==1)
	break;
}
cout<<p[i]<<" ";
a=i;
u[a]=1;
while (1)
{
	for (i=0;i<g[a].size();i++)
	{
	if (u[g[a][i]]==0)
		{cout<<p[g[a][i]]<<" "; 
			u[g[a][i]]=1;
				break;}
	}
a=g[a][i];
if (g[a].size()==1)
	{
	break;
	}
}   
return 0;
}
