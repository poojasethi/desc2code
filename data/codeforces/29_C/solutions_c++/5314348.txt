#include <map>
#include <vector>
#include <iostream>
using namespace std;
map<int,vector<int> > a;
map<int,bool> b;
int main ( ) {
	int n,i,x,y,p=0;
	cin>>n;
	for (i=1;i<=n;i++) {
		cin>>x>>y;
		a[x].push_back(y);
		a[y].push_back(x);
	}
	for (map<int,vector<int> >::iterator it=(a).begin();it!=(a).end();it++) 
		if ((int)(it->second).size()==1) p=it->first;
	do {
		cout<<p<<' ';b[p]=1;
		if (b[a[p][0]]!=0) p=a[p][1]; else p=a[p][0];
	} while ((int)(a[p]).size()>1);
	cout<<p<<endl;
	return 0;
}
