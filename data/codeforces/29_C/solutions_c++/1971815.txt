# include <iostream>
# include <vector>
# include <map>
# define PB push_back
using namespace std;
const int MAX_N=100*1000+10;
map<int,int> D;
map<int,vector<int> > adj;
map<int,bool> mrk;
int N;
void DFS(int x)
{
	cout<<x<<" ";
	mrk[x]=true;
	for(int i=0;i<(int)adj[x].size();i++)
	{
		int v=adj[x][i];
		if(mrk[v]==0)
			DFS(v);
	}
}
int main()
{
	cin >> N;
	for(int i=0;i<N;i++)
	{
		int u,v;cin>>u>>v;
		adj[u].PB(v);adj[v].PB(u);
		D[u]++;D[v]++;
	}
	for(map<int,int>::iterator it=D.begin();it!=D.end();it++)
	{
		if((it->second)==1){DFS(it->first);break;}
	}
	return 0;
}
