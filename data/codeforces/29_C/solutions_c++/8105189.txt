#include <bits/stdc++.h>
using namespace std;

map<int, vector<int> > G;
map<int, int> hash;

void dfs( int cur, int par = -1) {
	printf("%d ", cur );
	for ( vector<int> :: iterator it = G[cur].begin(); it!= G[cur].end(); it++ ) {
		if( *it != par )
			dfs( *it, cur );
	}
}
int main()
{
	int n, a, b;
	scanf("%d", &n);
	for( int i = 0 ; i < n ; i++ )
	{
		scanf("%d %d", &a, &b );
		G[a].push_back(b), G[b].push_back(a);
		hash[a]++, hash[b]++;
	}
	for (map<int, int> :: iterator it = hash.begin(); it != hash.end(); it++ ) {
		if( it->second == 1 ) {
			dfs( it->first );
			printf("\n");
			return 0;
		}
	}
	return 0;
}
