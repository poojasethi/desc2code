#include<iostream>
#include<map>
#include<vector>
using namespace std;
int n;
map <int,vector <int> > m;
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		int x,y;
		cin>>x>>y;
		m[x].push_back(y);
		m[y].push_back(x);
	}
	int k,g=-100;
	for(map <int,vector <int> > :: iterator it=m.begin();it!=m.end();it++)
	{
		if(it->second.size()==1)k=it->first;
	}
	cout<<k<<" ";
	for(int i=0;i<n;i++)
	{
		if(m[k][0]!=g){cout<<m[k][0]<<" ";g=k;k=m[k][0];}
		else
		{
			cout<<m[k][1]<<" ";g=k;k=m[k][1];
		}
	}
	//system("pause");
	return 0;
}
