#include <bits/stdc++.h>
using namespace std;

map < int, vector < int > > adj;
map < int, vector < int > > :: iterator it;

void dfs(int v, int p) {
	printf("%d ", v + 1);
	for (int i = 0; i < adj[v].size(); ++i) {
		int u = adj[v][i];
		if (u != p)
			dfs(u, v);
	}
}

int main() {
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		int u, v;
		scanf("%d %d", &u, &v);
		u --, v --;
		adj[u].push_back(v);
		adj[v].push_back(u);
	}
	for (it = adj.begin(); it != adj.end(); ++it)
		if ((it->second).size() == 1) {
			dfs(it->first, -1);
			puts("");
			return 0;
		}
	return 0;
}
