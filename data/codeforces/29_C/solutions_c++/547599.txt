#include<cstdio>
#include<map>
using namespace std;
int temp[200100],n,first,now,be,nx;
map<int,int>a,b;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        int x,y;scanf("%d%d",&x,&y);
        temp[2*i-1]=x;temp[2*i]=y;
        if(a[x]==0)a[x]=y;else b[x]=y;
        if(a[y]==0)a[y]=x;else b[y]=x;
    }
    for(int i=1;i<=2*n;i++)
    if(b[temp[i]]==0)
    {
        first=temp[i];break;
    }
    printf("%d ",first);
    now=a[first];be=first;
    for(int i=1;i<=n;i++)
    {
        printf("%d ",now);
        if(a[now]!=be)nx=a[now];else nx=b[now];
        be=now;now=nx;
    }
}
