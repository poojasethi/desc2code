#include <bits/stdc++.h>
using namespace std;

int x[200005];
map< int,int > m;
map< int, vector<int> > adj;
map< int,bool > vis;

void dfs (int in) {

    if (vis[in]) return;
    vis[in] = 1;

    cout << in << " ";

    for (int i=0; i<adj[ in ].size(); i++ )
        dfs( adj[in][i] );
}

int main () {

    int n; cin >> n;
    int a,b;
    for (int i=0; i<n; i++) {
        cin >> a >> b;
        x[i] = a; x[i+n] = b;
        m[ a ]++; m[ b ]++;
        adj[a].push_back(b);
        adj[b].push_back(a);
    }

    for (int i=0; i<2*n; i++)
        if ( m[ x[i] ] == 1 ) {
            dfs( x[i] );
            break;
        }
}
