#include<iostream>
#include<vector>
#include<map>
using namespace std;
map<int,vector<int> >m;
int c;
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++){int a,b;cin>>a>>b;m[a].push_back(b);m[b].push_back(a);}
    for(__typeof(m.begin()) it=m.begin();it!=m.end();it++)if(it->second.size()==1)c=it->first;
    cout<<c;
    int p=-1;
    for(int i=0;i<n;i++){if(m[c][0]!=p) c=m[p=c][0];else c=m[p=c][1];cout<<" "<<c;}
    
    return 0;
}
