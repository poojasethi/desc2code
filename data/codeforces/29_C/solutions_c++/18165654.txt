#include<stdio.h>
#include<map>
#include<vector>
using namespace std;
map<int,vector<int> >mp;
void dfs(int u, int fa){
	for (vector<int>::iterator it = mp[u].begin(); it != mp[u].end();it++)
	if (fa != *it){
		printf(" %d", *it);
		dfs(*it, u);
	}
}
int main(){
	int n, u, v;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d%d",&u,&v);
		mp[u].push_back(v);
		mp[v].push_back(u);
	}
	
	for (map<int, vector<int> >::iterator it = mp.begin(); it != mp.end();it++)
	if (it->second.size() <= 1){
		printf("%d %d", it->first, it->second[0]);
		dfs(*it->second.begin(), it->first);
		break;
	}
	return 0;
}
