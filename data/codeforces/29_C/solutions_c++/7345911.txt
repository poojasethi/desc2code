#include<cstdio>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;
map <int,vector<int> > M;
map <int,vector<int> >::iterator it;
int n;
int dfs(int now,int h){
  printf("%d ",now);
  for(int i=0;i<M[now].size();i++)
    if(M[now][i] != h)
      dfs(M[now][i],now);
}
int main(){
  scanf("%d",&n);
  for(int i=0,u,v;i<n;i++){
    scanf("%d%d",&u,&v);
    M[u].push_back(v);
    M[v].push_back(u);
  }
  for(it = M.begin();it != M.end();it++){
    if(it->second.size() == 1){
      dfs(it->first,-1); break;
    }
  }
  return 0;
}
