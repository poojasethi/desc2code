#include<cstdio>
#include<iostream>
#include<vector>
#include<map>
using namespace std;
map<int,vector<int> > e;
map<int,bool> v;

int main(){
	int i,n,a,b;
	scanf("%d",&n);
	for(i=1;i<=n;i++){
		scanf("%d%d",&a,&b);
		e[a].push_back(b);
		e[b].push_back(a);
	}
	int p;
	for(__typeof(e.begin()) i=e.begin();i!=e.end();i++)
	   if((i->second).size()==1) p=i->first;
    do{
    	printf("%d ",p);
    	v[p]=true;
    	if(v[e[p][0]]) p=e[p][1];
    	else p=e[p][0];
    }while((e[p]).size()>1);
    printf("%d\n",p);
    return 0;
}
