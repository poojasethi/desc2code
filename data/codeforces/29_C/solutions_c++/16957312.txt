#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <cstdio>
#include <cstring>
using namespace std;
map <int,vector<int> > nei;
map <int,bool> k;
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
	int a,b;
	cin>>a>>b;
	nei[a].push_back(b);
	nei[b].push_back(a);
	}
	int start=-1;
	for(map <int,vector<int> >::iterator it=nei.begin();it!=nei.end();it++)
	if((it->second).size()==1)
	start=it->first;
	do {
	k[start]=1;
	cout<<start<<" ";
	if(k[nei[start][0]])
	start=nei[start][1];
	else start=nei[start][0];
} while(nei[start].size()>1);
	cout<<start;
	return 0;
}
