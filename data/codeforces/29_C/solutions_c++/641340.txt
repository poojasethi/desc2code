#include <stdio.h>
#include <map>
#include <vector>
using namespace std;

int ids[100005];
map<int, int> id;
vector<int> a[100005];
int u[100005];

int main(void) {
	int n, i, x, y;
	scanf("%d", &n);
	for(i=0;i<n;i++) {
		scanf("%d%d", &x, &y);
		if(!id.count(x)) {int t=id.size();id[x]=t;ids[t]=x;}
		if(!id.count(y)) {int t=id.size();id[y]=t;ids[t]=y;}
		a[id[x]].push_back(id[y]);
		a[id[y]].push_back(id[x]);
	}
	for(i=0;i<id.size();i++) if(a[i].size()==1) break;
	while(1){
		u[i]=1;
		printf("%d ", ids[i]);
		if(a[i].size()==1 && u[a[i][0]]) break;
		if(u[a[i][0]]==0) i = a[i][0];
		else i = a[i][1];
	}	
	puts("");
	return 0;
}
