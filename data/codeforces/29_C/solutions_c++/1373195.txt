#include <map>

map<int, vector<int> > a;
map<int, bool > vis;

int main()
{
int n;
cin>>n;
 for (int i=0; i<n; i++)
 {
   int x,y; cin>>x>>y;
   a[x].push_back(y);
   a[y].push_back(x);
 }
int i=0,j;
map<int, vector<int> >::iterator it=a.begin();

while ((it->second).size()!=1) it++;
i=(it->first);
for (;;)
{vis[i]=1;
 cout<<i<<" ";
 if (vis[a[i][0]]) i=a[i][1]; else i=a[i][0];
 if (a[i].size()==1) break;
}
cout<<i;
return 0;
}
