#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

typedef long long int ll;

ll infinito=10000000000000000LL;

void computa(vector<ll> &v,ll &suma,ll &minimo,ll &maximo,ll &maxdif)
{
  for (int i=0;i<int(v.size());i++) {
    suma+=v[i];
    maxdif=max(maxdif,suma-minimo);
    minimo=min(minimo,suma);
    maximo=max(maximo,suma);
  }
  //cout<<suma<<","<<minimo<<","<<maximo<<","<<maxdif<<endl;
}

int main()
{
  ios::sync_with_stdio(false);
  int n,m;
  cin>>n>>m;
  vector<ll> suma(n,0),minimo(n,0),maximo(n,-infinito),maxdif(n,-infinito);
  for (int i=0;i<n;i++) {
    int siz;
    cin>>siz;
    vector<ll> v(siz);
    for (int j=0;j<siz;j++)
      cin>>v[j];
    computa(v,suma[i],minimo[i],maximo[i],maxdif[i]);
  }
  ll sumatotal=0;
  ll minimasuma=0;
  ll sol=-infinito;
  for (int i=0;i<m;i++) {
    int indice;
    cin>>indice;
    indice--;
    sol=max(sol,maxdif[indice]);
    sol=max(sol,sumatotal+maximo[indice]-minimasuma);
    minimasuma=min(minimasuma,sumatotal+minimo[indice]);
    sumatotal+=suma[indice];
  }
  cout<<sol<<endl;
}
