#include<stdio.h>
#include<iostream>
#include<map>
#include<vector>
#include<algorithm>
#include<string>
#include<string.h>
#include<memory>
using namespace std;

int n , m,t;
long long S[55],M[55],I[55],R[55],x,d;

int main(){
   cin >> n >> m;
   for(int i = 0 ; i < n ; i++)
   {
      cin >> t;
      d = 0;
      I[i] = R[i] = -1e9;
      for(int j = 0 ; j < t ; j++)
      {
         cin >> x;
         S[i]+=x;

         if(R[i] < S[i] - d)
            R[i] = S[i] - d;
         if(j == t-1){
           M[i] = S[i] - d;
         }
         if(d > S[i])
            d = S[i];
         if(S[i] > I[i])
            I[i] = S[i];
      }

   }
    long long rr = -1e9;
    long long s = 0;
   for(int i = 0 ; i < m ; i++)
   {
        cin >> t;
        t--;
        if(s + I[t] > rr)
          rr = s+I[t];
        if(R[t] > rr)
          rr = R[t];
        if(s + S[t] > M[t])
           s = s+S[t];
        else
           s = M[t];
        if(s > rr)
          rr = s;
   }
   cout <<rr;

   return 0;
}
