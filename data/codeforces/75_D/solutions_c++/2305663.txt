#include <cstdio>
#include <algorithm>

typedef __int64 ll;

using namespace std;

int main()
{
    int t;
    int n, m;
    ll lMax[51], rMax[51], total[51];
    ll dp[50], dest;
    
    scanf("%d %d", &n, &m);
    
    for (int i = 0; i < n; i++){
        int num;
        scanf("%d", &num);
        
        total[i] = rMax[i] = 0;
        lMax[i] = dp[i] = -250000001;
        dest = 0;
        for (int j = 0; j < num; j++){
            scanf("%d", &t);
            total[i] += t;
            lMax[i] = max(total[i], lMax[i]);
            rMax[i] += t;
            dest += t;
            dp[i] = max(dp[i], dest);
            if (dest < 0) dest = 0;
            if (rMax[i] < 0) rMax[i] = 0;
        }
    }
    
    ll ans = (ll)-250000 * 1000 * 5000 - 1;
    dest = 0;
    for (int i = 0; i < m; i++){
        scanf("%d", &t);
        ans = max(max(ans, dp[t - 1]), max(dest + lMax[t - 1], dest + total[t - 1]));
        dest = max(dest + total[t - 1], rMax[t - 1]);
    }
    
    printf("%I64d\n", ans);
    
    return (0);
}
