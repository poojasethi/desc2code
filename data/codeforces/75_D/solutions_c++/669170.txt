#include<iostream>
#include<cstdio>

using namespace std;

const int maxn=300000;
int n,m,N;
long long l[maxn],r[maxn],s[maxn],a[maxn],M[maxn],ans;

int main(){
	scanf("%d%d", &n, &m);
	ans=-10000000000000LL;
	for (int cnt=0; cnt<n; cnt++){
		scanf("%d", &N);
		M[cnt]=-100000000000LL;
		for (int i=0; i<N; i++) scanf("%I64d", &a[i]);
		long long tmp=0; l[cnt]=a[0];
		for (int i=0; i<N; i++){
			tmp+=a[i];
			l[cnt]=max(l[cnt], tmp);
		}
		tmp=0; r[cnt]=a[N-1];
		for (int i=N-1; i>=0; i--){
			tmp+=a[i];
			r[cnt]=max(r[cnt], tmp);
		}
		tmp=0;
		for (int i=0; i<N; i++){
			tmp+=a[i]; s[cnt]+=a[i];
			M[cnt]=max(M[cnt], tmp);
			if (tmp<0) tmp=0;
		}
	}
	for (int i=0; i<m; i++){scanf("%I64d", &a[i]); a[i]--;}
	long long tmp=0;
	for (int i=0; i<m; i++){
		ans=max(ans, M[a[i]]);
		ans=max(ans, tmp+l[a[i]]);
		tmp+=s[a[i]];
		tmp=max(tmp, r[a[i]]);
	}
	ans=max(ans, tmp);
	cout<<ans<<endl;
}
