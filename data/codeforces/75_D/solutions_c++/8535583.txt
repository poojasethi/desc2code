#include <string>
#define For(i,a,n)  for(int i=(a);i<=(n);i++)
#define Forn(i,a,n) for(int i=(a);i>=(n);i--)
using namespace std;

void setIO(string a){
	 string in=a+".in",out=a+".out";
	 freopen(in.c_str(),"r",stdin);
	 freopen(out.c_str(),"w",stdout);
}

struct seg{
	int N;
	long long Max,mic;
	long long a[5005],mx,sm,sr;
	void read(){
		scanf("%d",&N);
		long long c=0;mx=-2100000000;
		Max=-2100000000;mic=0;
		For(i,1,N){
			scanf("%I64d",&a[i]);
			c+=a[i];
			if(c>mx)mx=c;
			if(c-mic>Max) Max=c-mic;
			if(c<mic)mic=c;
		}sm=c;c=0;sr=-2100000000;
		Forn(i,N,1){
			c+=a[i];
			if(c>sr)sr=c;
		}
	}
}a[55];
int n,m;
long long an=-21000000000000ll,sR=-210000000000000ll;

void merge(int x){
	an=max(an,a[x].Max);
	an=max(an,sR+a[x].mx);
	sR=max(a[x].sr,a[x].sm+sR);
}

int main(){
	scanf("%d%d",&n,&m);
	For(i,1,n) a[i].read();
	For(i,1,m){
		int x;
		scanf("%d",&x);
		merge(x);
	}printf("%I64d\n",an);
	return 0;
}
