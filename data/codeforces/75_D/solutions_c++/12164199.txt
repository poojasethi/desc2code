#include<cstdio>
#include<cstring>
#define MAX(a,b) ((a)>(b)?(a):(b))
#define LL long long
#define INF 1e+8
LL pre[51],suf[51];
LL sum[51],max[51];
int size[51];
int a[51][5001];
int main(){
	int n,m,i,j,x;
	LL ans=-INF,tot;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i){
		scanf("%d",&size[i]);
		sum[i]=0;
		for(j=0;j<size[i];++j){
			scanf("%d",&a[i][j]);
			sum[i]+=a[i][j];
		}
		// max segment
		tot=0,max[i]=-INF;
		for(j=0;j<size[i];++j){
			tot+=a[i][j];
			max[i]=MAX(max[i],tot);
			if(tot<0) tot=0;
		}
		// prefix
		tot=0,pre[i]=-INF;
		for(j=0;j<size[i];++j){
			tot+=a[i][j];
			pre[i]=MAX(pre[i],tot);
		}
		// suffix
		tot=0,suf[i]=-INF;
		for(j=size[i]-1;j>=0;--j){
			tot+=a[i][j];
			suf[i]=MAX(suf[i],tot);
		}
	}
	ans=-INF,tot=0;
	while(m--){
		scanf("%d",&x);
		ans=MAX(ans,max[x]);
		ans=MAX(ans,suf[x]);
		ans=MAX(ans,tot+pre[x]);
		tot=MAX(tot+sum[x],MAX(0,suf[x]));
	}
	printf("%I64d",ans);
	return 0;
}
