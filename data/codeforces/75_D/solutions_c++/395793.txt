#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#define LL long long
using namespace std;

int dp[5010][4];
int data[5010],l;
long long A,B;
int n,m;
void Dp(int i,int n)
{
    int p=0,sum=0;
    dp[i][0]=0;
    dp[i][1]=-10000*5000;
    dp[i][2]=-10000*5000;
    for(int j=1;j<=n;j++)
    {
        dp[i][0]+=data[j];
        if(p>0) p=p+data[j];
        else p=data[j];
        dp[i][1]=max(p,dp[i][1]);
        dp[i][2]=max(dp[i][0],dp[i][2]);
    }
    dp[i][3]=p;
    //printf("%d %d %d %d\n",dp[i][0],dp[i][1],dp[i][2],dp[i][3]);
}
int main()
{
    int x;
    while(cin>>n>>m)
    {
        A=-(1LL<<62);
        B=0;
        for(int i=1;i<=n;i++){
            scanf("%d",&l);
            for(int j=1;j<=l;j++)
                scanf("%d",&data[j]);
            Dp(i,l);
        }
        for(int i=0;i<m;i++)
        {
            scanf("%d",&x);
            A=max(A,B+dp[x][2]);
            A=max(A,(LL)dp[x][1]);
            B=max(B+dp[x][0],(LL)dp[x][3]);
        }
        cout<<A<<endl;
    }
    return 0;
}
