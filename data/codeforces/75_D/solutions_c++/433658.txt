#include <iostream>
#include <sstream>
using namespace std;

long long sums[51], frombeg[51], toend[51], a[5001], dp[5001], maxin[51];

int main()
{
    long long res = -1000;
    int i, j, n, m, l;
    cin >> n >> m;
    for (i = 1; i <= n; i++) {
        cin >> l;
        maxin[i] = -1000;
        for (j = 1; j <= l; j++) {
            cin >> a[j];
            dp[j] = max(dp[j - 1] + a[j], a[j]);
            maxin[i] = max(maxin[i], dp[j]);
            sums[i] += a[j];
        }

        toend[i] = dp[l];

        for (j = 1; j <= l; j++) dp[j] = max(dp[j - 1] + a[l - j + 1], a[l - j + 1]);

        frombeg[i] = dp[l];
    }

    long long prev = 0, now = 0;
    for (i = 0; i < m; i++) {
        cin >> j;
        res = max(res, maxin[j]);
        now = max(prev + sums[j], max(sums[j], toend[j]));
        res = max(res, max(prev + frombeg[j], now));
        prev = now;
    }

    cout << res << endl;
    return 0;
}
