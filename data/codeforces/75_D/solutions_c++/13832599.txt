#include <iostream>
#include <cstdio>
#define rep(i,n) for(int i=0;i<n;i++)
#define INF 10000000000000000LL
using namespace std;

long long lm[60],rm[60],sum[60],mm[60];
int n,m,i,j,k;
long long res=-INF,r,t; 

int main(){

  cin>>n>>m;
  rep(i,50)lm[i]=0,rm[i]=0,sum[i]=0,mm[i]=-INF;
  rep(i,n){
    cin>>k;
    rep(j,k){
      cin>>t;sum[i]+=t;rm[i]+=t;
      if(lm[i]<sum[i])lm[i]=sum[i];
      if(mm[i]<rm[i])mm[i]=rm[i];
      if(rm[i]<0&&j!=k-1)rm[i]=0;
    }

  }

  rep(i,m){ 
    cin>>k;k--;
    if(i==0){res=mm[k];r=rm[k];} 
    else {
      res=max(max(res,mm[k]),r+lm[k]);
      r=max(r+sum[k],rm[k]);
    }

  }
  cout<<res<<endl;

  return 0;
}
