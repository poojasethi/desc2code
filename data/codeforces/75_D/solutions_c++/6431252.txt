#define inf (ll)1e18
using namespace std;
long long gen[50], lft[50], rght[50], s[50],ans=(ll)(-1e18),cur=0;
int i,n,m,l,x;
int main()
{
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        gen[i]=lft[i]=-inf;
        cin>>l;
        for(int j=0;j<l;j++)
        {
            cin>>x;
            s[i]+=x;
            gen[i]=max(gen[i],s[i]-rght[i]);
            lft[i]=max(lft[i],s[i]);
            rght[i]=min(rght[i],s[i]);

        }
        rght[i]=s[i]-rght[i];
    }
    while(m--)
    {
        cin>>i;
        i--;
        ans=max(ans,gen[i]);
        if (cur)
        ans=max(ans,cur+lft[i]);
        cur=max(0LL,max(rght[i],cur+s[i]));
    }
    cout<<ans;
    return 0;
}
