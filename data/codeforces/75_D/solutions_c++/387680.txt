#include<iostream>
#include<cstdio>
#include<vector>
#include<string>
#include<algorithm>
#include<set>
#include<map>
#include<cmath>
#include<cstring>
#define FOR(i,j,k) for(i = j;i < k;i++)
#define RFOR(i,j,k) for(i = k-1;i >= j;i--)
#define LL long long
#define GET(x) scanf("%d",&x)
#define OUT(x) printf("%d\n",x)
#define SET(x) memset(x,0,sizeof(x))
LL lf[55],rt[55],ind[55],to[55];
using namespace std;
int main(){
	int n,q,i,m,nm,j;
	LL an = 0,mx,cr;
	GET(n);GET(q);
	FOR(i,1,n+1){
		GET(m);
		cr = 0;
		mx = (LL)(-1e18);
		FOR(j,0,m){
			GET(nm);
			to[i] += nm;
			lf[i] = max(lf[i],to[i]);
			rt[i] = min(rt[i],to[i]);
			mx = max(mx,(LL)nm);
			cr += nm;
			if(cr > 0) mx = max(mx,cr);
			else cr = 0;
			}
		ind[i] = mx;
		rt[i] = to[i]-rt[i];
		//cout<<lf[i]<<"  "<<rt[i]<<"  "<<ind[i]<<"  "<<to[i]<<"\n";
	}
	cr = 0;
	mx = (LL)(-1e18);
	while(q--){
		GET(m);
		mx = max(mx,ind[m]);
		if(cr) mx = max(mx,cr+lf[m]);
		cr = max(0ll,max(rt[m],cr+to[m]));
	}
	cout<<mx<<endl;
	return 0;
}
