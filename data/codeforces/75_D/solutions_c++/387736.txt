#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 55
#define maxl 5010
#define INF 100000000
using namespace std;

typedef long long LL;
struct node {
	int l;
	int v[maxl];
	LL sum,lmax,rmax,maxv;
	void cal(){
		int i,j,k;
		LL s[maxl];
		s[0]=0;
		for (i=1;i<=l;i++) s[i]=s[i-1]+v[i];
		sum=s[l];
		lmax=rmax=maxv=-INF;
		LL mins=0;
		for (i=1;i<=l;i++){
			if (s[i]>lmax) lmax=s[i];
			if (sum-s[i-1]>rmax) rmax=sum-s[i-1];
			if (s[i]-mins>maxv) maxv=s[i]-mins;
			if (s[i]<mins) mins=s[i];
		}
	}
} a[maxn];
int n,m;
int main(){
	int i,j,k;
	scanf("%d %d",&n,&m);
	for (i=0;i<n;i++){
		scanf("%d",&a[i].l);
		for (j=1;j<=a[i].l;j++) scanf("%d",&a[i].v[j]);
		a[i].cal();
	}
	/*for (i=0;i<n;i++){
		printf("%I64d %I64d %I64d\n",a[i].sum,a[i].lmax,a[i].rmax);
	}*/
	LL ans=-INF,SUM=0;
	for (i=0;i<m;i++){
		scanf("%d",&k);
		k--;
		ans=max(ans,a[k].maxv);
		ans=max(ans,SUM+a[k].lmax);
		SUM=max(SUM+a[k].sum,a[k].rmax);
	}
	printf("%I64d\n",ans);
	return 0;
}
