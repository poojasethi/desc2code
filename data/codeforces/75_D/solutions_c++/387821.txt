#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#include <iostream>
#include <cmath>
#include <algorithm>
#include <vector>

using namespace std;

typedef long long ll;

int main(){
  int n, m;
  scanf("%d%d", &n, &m);
  int sum[51], dif[51], mx[51], mn[51];
  const int INF = 1<<25;
  ll ans = -(1LL<<60);
  for(int i = 0; i < n; i++){
    int a;
    scanf("%d", &a);
    dif[i] = -INF; mx[i] = -INF; mn[i] = INF;
    sum[i] = 0;
    for(int j = 0; j < a; j++){
      int v;
      scanf("%d", &v);
      sum[i] += v;
      dif[i] = max(dif[i], sum[i] - mn[i]);
      mn[i] = min(mn[i], sum[i]);
      mx[i] = max(mx[i], sum[i]);
    }
    dif[i] = max(dif[i], mx[i]);
    //printf("dif = %d, mx = %d, mn = %d\n", dif[i], mx[i], mn[i]);
    //ans = max(ans, 1LL * dif[i]);
  }

  ll total = 0, low = 0;
  for(;m;m--){
    int k;
    scanf("%d", &k);
    k--;
    ans = max(ans, 1LL * dif[k]);
    ans = max(ans, total + mx[k] - low);
    low = min(low, total + mn[k]);
    total += sum[k];
  }

  cout << ans << endl;

  return 0;
}
