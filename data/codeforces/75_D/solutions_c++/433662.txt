#include <iostream>
#include <sstream>
using namespace std;

long long sums[51], frombeg[51], toend[51], maxin[51];

int main()
{
    long long prev, now, res = -1000, a;
    int i, j, n, m, l;
    cin >> n >> m;
    for (i = 1; i <= n; i++) {
        cin >> l;
        prev = 0;
        frombeg[i] = maxin[i] = -1000;
        for (j = 1; j <= l; j++) {
            cin >> a;
            now = max(prev + a, a);
            prev = now;
            maxin[i] = max(maxin[i], now);
            sums[i] += a;
            frombeg[i] = max(frombeg[i], sums[i]);
        }
        toend[i] = now;
    }

    prev = 0;
    for (i = 0; i < m; i++) {
        cin >> j;
        now = max(prev + sums[j], max(sums[j], toend[j]));
        res = max(max(res, maxin[j]), max(prev + frombeg[j], now));
        prev = now;
    }

    cout << res << endl;
    return 0;
}
