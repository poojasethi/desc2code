#include<cstdio>
int sg[7]={0,1,0,1,2,0,2};
int get_sg(int n)
{
	if(n<7)return sg[n];
	if(n&1)return 0;
	return get_sg(n>>1)==1?2:1;
}
int main()
{
	int n,k,x,win=0;
	scanf("%d%d",&n,&k);
	while(n--)scanf("%d",&x),k&1?
		win^=get_sg(x):
		win^=x!=1?(x==2?2:~x&1):1;
	puts(win?"Kevin":"Nicky");
	return 0;
}
