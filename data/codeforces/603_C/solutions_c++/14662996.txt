#include<cstdio>

int n, k, ans;

int f(int x) {
    if (k % 2) {
        if (x <= 3) return x % 2;
        else if (x % 2) return 0;
        else if (x == 4 || f(x / 2) == 1) return 2;
        else return 1;
    }
    else if (x <= 2) return x;
    else return (x+1) % 2;
}

int main() {
    scanf("%d %d", &n, &k);
    for (int i = 0; i < n; i++) {
        int a;
        scanf("%d", &a);
        ans ^= f(a);
    }
    printf("%s\n", ans ? "Kevin" : "Nicky");
    scanf("\n");
    return 0;
}
