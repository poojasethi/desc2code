#include <bits/stdc++.h>
using namespace std;

int ret, n, k, a;

int Grundy (int x) {
	if (k & 1) {
		if (x == 0 || x == 2) return 0;
		else if (x == 1 || x == 3) return 1;
		else if (x == 4) return 2;
		else if (x & 1) return 0;
		else if (Grundy (x >> 1) == 1) return 2;
		else return 1;
	} else {
		if (x < 3) return x;
		else if (x & 1) return 0;
		else return 1;
	}
}

int main () {
	scanf ("%d %d", &n, &k);
	ret = 0;
	while (n--) {
		scanf ("%d", &a);
		ret ^= Grundy (a);
	}
//	printf ("%d\n", ret);
	if (ret == 0) printf ("Nicky\n");
	else printf ("Kevin\n");
	return 0;
}
