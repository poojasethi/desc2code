#include <cstdio>
using namespace std;

int n,k,x,sum,cnt;

int sg(int x)
{
	if (!(k&1))
		return x<=2?x:(x&1)^1;
	if (x<=3)
		return x&1;
	if (x==4)
		return 2;
	if (x&1)
		return 0;
	return sg(x>>1)==1?2:1;
}

int main()
{
	for (scanf("%d%d",&n,&k); n--;)
		scanf("%d",&x),sum^=sg(x);
	printf(sum?"Kevin\n":"Nicky\n");
	return 0;
}
