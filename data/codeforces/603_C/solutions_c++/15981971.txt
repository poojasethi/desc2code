#include<bits/stdc++.h>
using namespace std;
int getSG(int x,int k)
{
    if(x==0) return 0;
    if(x==1) return 1;
    if(k%2==0)
    {
        if(x==2) return 2;
        if(x%2==0) return 1;
        else return 0;
    }
    else
    {
        if(x==2) return 0;
        if(x==3) return 1;
        if(x==4) return 2;
        if(x%2==1) return 0;
        else return getSG(x/2,k)==1?2:1;
    }
}
int main()
{
    int n,k,ans=0;
    scanf("%d%d",&n,&k);
    for(int i=0,x;i<n;i++){scanf("%d",&x);ans^=getSG(x,k);}
    printf("%s\n",ans!=0?"Kevin":"Nicky");
}
