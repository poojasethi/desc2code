#include<iostream>
#include<cstdio>
#define maxn 100010

using namespace std;

int n,k,a[maxn],sg[10],ans;

int get0(int k){
	if(k<=2)return k;
	return 1-(k&1);
}
int get1(int k){
	if(k<6)return sg[k];
	if(k&1)return 0;
	if(get1(k>>1)==1)return 2;
	else return 1;
}

int main(){
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i)scanf("%d",&a[i]);
	sg[0]=0;
	sg[1]=1;
	sg[2]=0;
	sg[3]=1;
	sg[4]=2;
	sg[5]=0;
	sg[6]=2;
	ans=0;
	if(k&1){
		for(int i=0;i<n;++i)ans^=get1(a[i]);
	}else{
		for(int i=0;i<n;++i)ans^=get0(a[i]);
	}
	if(ans)printf("Kevin");
	else printf("Nicky");
	return 0;
}
