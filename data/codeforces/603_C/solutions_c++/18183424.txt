#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;

int n,k,x,ans;

int SG(int x)
{
	if (!x) return 0;
	if (k&1)
	{
		if (x==1||x==3) return 1;
		if (x&1) return 0;
		bool t[3]={};
		t[SG(x-1)]=1;
		if (!(x&1)) t[SG(x>>1)]=1;
		int ans=0;
		while (t[ans]) ans++;
		return ans;
	}
	else return x<=2?x:!(x&1);
}

int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		ans^=SG(x);
	}
	printf(ans?"Kevin\n":"Nicky\n");
}
