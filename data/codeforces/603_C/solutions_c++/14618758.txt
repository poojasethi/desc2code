#include <bits/stdc++.h>

using namespace std;

const int fuck[] = {0, 1, 0, 1, 2};

int n, k;

int SG(int x) {
    if(k % 2 == 0) {
        if(x == 1)
            return 1;
        if(x == 2)
            return 2;
        return (x & 1) ^ 1;
    } else {
        if(x < 5)
            return fuck[x];
        if(x & 1)
            return 0;
        return SG(x / 2) == 1 ? 2 : 1;
    }
}

int main() {
    cin >> n >> k;
    int ans = 0;
    for(int i = 0; i < n; i++) {
        int x; scanf("%d", &x);
        ans ^= SG(x);
    }
    puts(ans ? "Kevin" : "Nicky");
    return 0;
}
