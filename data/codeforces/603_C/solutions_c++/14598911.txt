#include <algorithm>
#include <cstdio>
using namespace std;
const int N = 100005;
int n, k, a[N];
int f(int x) {
	if (x == 2) return 2;
	if ((x > 1)&&(x%2 == 1)) return 0;
	return 1;
}
int g(int x) {
	if ((x == 1)||(x == 3)) return 1;
	if (x%2 == 1) return 0;
	int m = g(x/2), n = g(x-1);
	if (m*n == 0) {
		if ((m == 1)||(n == 1)) return 2;
		else return 1;
	} else return 0;
}
int main() {
	scanf("%d%d", &n, &k);
	int t = 0;
	for (int i = 0; i < n; i++) {
		scanf("%d", a+i);
		if (k%2 == 0) t ^= f(a[i]);
		else t ^= g(a[i]);
	}
	printf("%s\n", t? "Kevin": "Nicky");
	return 0;
}
