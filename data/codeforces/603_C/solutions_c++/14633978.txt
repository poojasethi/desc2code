int work(int x){
	int t;
	if (!x) return 0;
	if (x==1) return 1;
	if (k){
		if (x==2) return 0;
		if (x==3) return 1;
		if (x==4) return 2;
		if (x&1) return 0;
		t=work(x/2);
		if (!t||t==2) return 1;
		return 2;
	}else{
		if (x==2) return 2;
		return (x&1)?0:1;
	}
}
