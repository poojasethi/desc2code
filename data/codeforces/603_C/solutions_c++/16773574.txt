#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
const int f[5]={0,1,0,1,2};
int n,k,i,x,ans=0;
int calc1(int x)
{
    if (x<=4) return f[x];
    else if (x%2) return 0;
    else return calc1(x/2)==1?2:1;
}
int calc2(int x)
{
    if (x<=2) return x;
    else if (x%2) return 0;
    else return 1;
}
int main()
{
    scanf("%d%d",&n,&k);
    if (k%2) for (i=1;i<=n;i++) scanf("%d",&x),ans^=calc1(x);
    else for (i=1;i<=n;i++) scanf("%d",&x),ans^=calc2(x);
    return ans?printf("Kevin\n"):printf("Nicky\n"),0;
}
