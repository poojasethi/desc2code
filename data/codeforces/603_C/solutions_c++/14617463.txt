#include<cstdio>
using namespace std;
int n,x1,m,ans;
const int pre[2][5]={{0,1,2,0,1},{0,1,0,1,2}};
inline int calc(int x)
{
	if(x<5)return pre[m][x];
	if(x%2==0)
	{
		if(m!=0)return (calc(x/2)&1)+1;
		return 1;
	}
	return 0;
}
int main()
{
	scanf("%d %d",&n,&m);
	m%=2;
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&x1);
		ans^=calc(x1);
	}
	if(ans==0)printf("Nicky\n");
	else printf("Kevin\n");
	return 0;
}
