#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e5+10;
int d[mx];
int g(int n){
   if(n<=3)return n%2;
   if(n==4)return 2;
   if(n%2==1)return 0;
   int x=g(n>>1);
   if(x==1)return 2;
   return 1;
}
int h(int n){
   if(n<=2)return n;
   return (n%2)^1;
}
int main(){
   int n,k;
   scanf("%d %d",&n,&k);
   for(int i=0;i<n;i++)scanf("%d",d+i);
   int x=0;
   if(k%2==0){
      for(int i=0;i<n;i++)x^=h(d[i]);
   }
   else{
      for(int i=0;i<n;i++)x^=g(d[i]);
   }
   if(x==0)puts("Nicky");
   else puts("Kevin");
   return 0;
}
