#include<cstdio>
#include<cstdlib>
#include<cstring>

#include<iostream>
#include<algorithm>
using namespace std;

int n,m;

int calc(int x)
{
    if(!(m&1))
        return x<=2?x:(x&1)^1;
    if(x<=3)
        return x&1;
    if(x==4)
        return 2;
    if(x&1)
        return 0;
    return calc(x>>1)==1?2:1;
}

int main()
{
    scanf("%d%d",&n,&m);
    static int gN,re;
    for(int i=1;i<=n;i++)
        scanf("%d",&gN),re^=calc(gN);
    printf(re?"Kevin\n":"Nicky\n");
    return 0;
}
