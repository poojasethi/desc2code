#include <cstdio>
#include <cstring>
#include <map>
#include <vector>
using namespace std;
map<int,int> sg;
map<int,int>::iterator it;
int n,k;
int grundy(int x)
{
	if(x>=5&&(k&1)&&(x&1)) return 0;
	if(x>=3&&!(k&1)) return !(x&1);
	it=sg.find(x);
	if(it!=sg.end()) return it->second;
	int v[5]={};
	v[grundy(x-1)]=1;
	if(!(x&1)) v[grundy(x/2)*(k&1)]=1;
	int i=0;
	while(v[i]) i++;
	return sg[x]=i;
}
int main()
{
	sg[0]=0;
	scanf("%d%d",&n,&k);
	int ans = 0, a;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a);
		ans ^= grundy(a);
	}
	puts(ans?"Kevin":"Nicky");
	return 0;
}
