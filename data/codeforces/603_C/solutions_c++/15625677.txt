#include <cstdio>
#include <memory.h>

#define N 100100

int a[N], n, k;

int get_nim(int x) {
	if(x < 4) return (x & 1);
	if(x == 4) return 2;
	if(x & 1) return 0;
	if(get_nim(x>>1) == 1) return 2;
	return 1;
}

int run() {
	int ans = 0;
	if(k & 1) {
		for(int i = 0; i < n; i ++) ans ^= get_nim(a[i]);
		return ans;
	}
	for(int i = 0; i < n; i ++) {
		if(a[i] < 3) ans ^= a[i]; else ans ^= (1 - (a[i] & 1));
	}
	return ans;
}

int main() {
	scanf("%d%d", &n, &k);
	for(int i = 0; i < n; i ++) scanf("%d", a + i);
	if(run()) puts("Kevin"); else puts("Nicky");
	return 0;
}
