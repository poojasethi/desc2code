#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int n, k;
int odd[] = {0, 1, 0, 1, 2, 0};
int godd(int n) {
    if (n <= 5) return odd[n];
    if (n & 1) return 0;
    return 1 + (godd(n >> 1) == 1); // mex
}
int geven(int n) {
    if (n <= 2) return n;
    return (n & 1) ^ 1;
}

int main() {
    scanf("%d%d", &n, &k);
    int grundy = 0;
    while (n--) {
        int a; scanf("%d", &a);
        grundy ^= (k & 1) ? godd(a) : geven(a);
    }
    printf(grundy ? "Kevin\n" : "Nicky\n");
}
