#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
const int N=100;
int n,k,ans,x,g[4],f[4];

int get(int x)
{
    if (x<=3) return f[x];
    if (x%2) return 0;
    int a=get(x/2),b=get(x-1);
    if (!k) a=0;
    g[0]=g[1]=g[2]=0;
    g[a]=g[b]=1;
    for (int i=0;i<3;i++) if (!g[i]) return i;
}

int main()
{
    cin>>n>>k;
    k%=2;
    f[1]=1;
    if (k) f[3]=1; else f[2]=2;
    for (int i=1;i<=n;i++)
    {
        scanf("%d",&x);
        ans=ans^get(x);
    }
    if (ans) cout<<"Kevin"; else cout<<"Nicky"; 
}
