#include <stdio.h>
int k;
int f(int n)
{
	if (n <= 1) return n;
	if (k&1) {
		if (n <= 4) return n-2;
		if (n&1) return 0;
		else {
			int i = f(n/2);
			if (i == 1) return 2;
			else return 1;
		}
	}
	else {
		if (n == 2) return 2;
		if (n&1) return 0;
		else return 1;
	}
}
int main()
{
	int r = 0;
	int n, i, j;
	
	scanf("%d%d",&n,&k);
	for (i = 0; i < n; i++) {
		scanf("%d",&j);
		r ^= f(j);
	}
	printf("%s",r==0?"Nicky":"Kevin");
	return 0;
}
