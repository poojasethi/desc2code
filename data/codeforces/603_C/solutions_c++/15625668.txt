#include <cstdio>

int SG(int x, int k) {
	if (k % 2 == 0) {
		if (x <= 2) return x;
		return 1 - x % 2;
	}
	if (x <= 3) return x % 2;
	if (x == 4) return 2;
	if (x % 2) return 0;
	int rlt = SG(x / 2, k);
	if (rlt <= 1) return rlt + 1;
	return 1;
}

main() {
	int n, k, x, rlt = 0;
	scanf("%d %d", &n, &k);
	for (int i = 0; i < n; i ++) {
		scanf("%d", &x);
		rlt ^= SG(x, k);
	}
	puts(rlt ? "Kevin" : "Nicky");
}
