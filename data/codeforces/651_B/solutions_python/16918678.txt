import sys
s = []
for line in sys.stdin:
    s.append(line)
b = s[1].split()
d = {}
for i in b:
    if d.has_key(i):
            d[i] = d[i]+1
    else:
            d[i] = 1
            
fr = 0
res = 0
while (len(d.keys()) >=1):
    res= 0
    for k in d.keys():
        
        d[k] = d[k] -1
        if d[k] == 0:
            d.pop(k,None)
        res = res +1
    fr = fr + res -1
    
print  fr
