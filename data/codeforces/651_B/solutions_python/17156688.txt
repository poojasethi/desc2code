# -*- coding:utf-8 -*-

n=input()
a=map(int,raw_input().split())
a.sort()
b=[True]*n

ans=0
cur=0
while cur<n:
    while cur<n and b[cur]==False: cur+=1
    if cur==n: break
    b[cur]=False
    now=a[cur]
    score=0
    for i in range(cur+1,n):
        if b[i] and a[i]>now:
            now=a[i]
            b[i]=False
            score+=1
    ans+=score

print ans
