n = input()
score = map(int, raw_input().split())
sort = [0] * 1050
for s in score:
    sort[s] += 1
def check(sort):
    ans = 0
    for i in range(len(sort)):
        if sort[i]:
            sort[i] -= 1
            ans += 1
    if ans > 1:
        return ans - 1
    else:
        return 0

ans = 0
tmp = check(sort)
while tmp:
    ans += tmp
    tmp = check(sort)
print ans
