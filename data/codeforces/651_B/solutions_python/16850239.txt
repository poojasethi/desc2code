def solve(beauties, remains):
    beauties.sort()
    for i in range(1, len(beauties)):
        if beauties[i] == beauties[i-1]:
            remains.append(beauties[i])
    beauties = list(set(beauties))
    return len(beauties)-1

N = int(raw_input())

beauties = [int(x) for x in raw_input().split(" ")]
remains = []
n = 0
while len(beauties) != 0:
    n += solve(beauties, remains)
    beauties = remains
    remains = []
print(n)
