#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
a = ia()

def solve():
    n = len(a)
    if n==0:
        print 0
        return
    
    a.sort()
    p = None
    s, ma = 1, 0
    for e in a:
        if e==p: s+=1
        else   : s=1
        if s>ma: ma=s
        p = e

    print n - ma

solve()
