from collections import Counter

n = int(raw_input())
A = map(int, raw_input().split())

def solve(A, n):
    my = sorted(Counter(A).values())
    ans = 0;
    while len(my) > 0:
        l = my[0]
        ans += l * (len(my)-1)
        my = [x-l for x in my if x-l > 0]        
    return ans

print solve(A, n)
