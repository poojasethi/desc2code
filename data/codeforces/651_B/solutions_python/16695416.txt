from collections import Counter
def max_happy(beauties):
    old = beauties
    new_set = set(old)
    c = Counter(old)
    most_n = c.most_common()[0][1]
    result = dict((x,[]) for x in range(most_n))
    for beauty in new_set:
        n = c[beauty]
        for i in range(n):
            result[i].append(beauty)
    return sum([len(result[x])-1 for x in result])
