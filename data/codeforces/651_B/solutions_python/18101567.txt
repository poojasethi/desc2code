import re
import math

number= raw_input()
input = raw_input()
inputs = re.findall('\d+',input)
inputs.sort()
f = []
for i in range (0,int(number)):
		f.append(0)
for n in range (0,int(number)):
	for m in range (n,int(number)):
		if (inputs[n] < inputs[m] and f[m]==0):
			f[m]+=1
			break;
count=0
for i in range (0,len(f)):
	if f[i]!=0:
