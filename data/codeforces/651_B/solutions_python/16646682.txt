#!/usr/bin/python
import sys
n = int(sys.stdin.readline().strip())
nn = [int(v) for v in sys.stdin.readline().split()][:n]
nn.sort()

cnt = 0
while nn:
  last = nn[0]
  nn = nn[1:]
  rem = []
  while nn:
    val = nn.pop(0)
    if last==val:
      rem.append(val)
    else:
      cnt += 1
    last = val
  nn = rem
print cnt
