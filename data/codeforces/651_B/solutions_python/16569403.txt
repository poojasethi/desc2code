N = int(raw_input())

nums = [x for x in raw_input().strip().split()]


used = [0 for x in range(N)]

nums = sorted(nums)
cnt = 0

preJ = 0
i = 0

while i < N:
    j = preJ + 1
    while j < N:
        if nums[i] < nums[j]:
            cnt += 1
            preJ = j
            break
        j += 1
    i += 1

print cnt
