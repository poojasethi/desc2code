def read_int():
    return int(raw_input())


def read_int_list():
    return [int(x) for x in raw_input().split()]


def main():
    n = read_int()
    a = read_int_list()
    a.sort()
    i, j = 0, 1
    cnt = 0
    while j < n:
        if a[i] < a[j]:
            i += 1
            cnt += 1
        j += 1
    print cnt

if __name__ == '__main__':
    main()
