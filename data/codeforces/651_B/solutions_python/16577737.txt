n = int(raw_input())
a = map(int, raw_input().split())

paint = dict.fromkeys(set(a), 0)

for i in sorted(a):
    paint[i] += 1

s_paint = sorted(paint.keys())
m = max(paint.values())

ans = []
k = 0
while k < m:
    for i in s_paint:
        if paint[i] > 0:
            ans.append(i)
            paint[i] -= 1
    k += 1

a = 0
for i in range(len(ans)-1):
    if ans[i] < ans[i+1]:
        a += 1
print a
