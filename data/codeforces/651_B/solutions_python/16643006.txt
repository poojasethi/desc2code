n = input()
a = sorted([int(x) for x in raw_input().split(' ')])

def main(n, a):
	if n == 1:
		return 0
	if a[0] == a[-1]:
		return 0
	result = 0
	while a:
		result += 1
		b = sorted(set(a))
		i = 0
		for x in b:
			for j in range(i, len(a)):
				if x == a[j]:
					del a[j]
					i = j
					break
	return n - result

print main(n, a)
