from sys import stdin, stdout
import math

n = int(stdin.readline().rstrip())
seq = [int(x) for x in stdin.readline().rstrip().split()]
seq.sort()


total = 0
while len(seq) > 0:
    residualSeq = []
    for i in xrange(1, len(seq)):
        if seq[i] > seq[i - 1]:
            total += 1
        else:
            residualSeq.append(seq[i])
    seq = residualSeq

print total
