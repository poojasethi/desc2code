import sys

rdl = sys.stdin.readline

n = int(rdl())
ai = [int(i) for i in rdl().split()]
cnt = {}

for i in ai:
    cnt[i] = cnt.get(i, 0) + 1

assert cnt

ans = 0
while cnt:
    ans += len(cnt) - 1

    for k in cnt.keys():
        cnt[k] -= 1
        if not cnt[k]:
            cnt.pop(k)

print ans
