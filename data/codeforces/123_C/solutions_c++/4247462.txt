#include <algorithm>
#include <iostream>
using namespace std;

typedef long long ll;
typedef pair<int, int> pii;
const int N = 205;
const ll oo = 2e18;

pii p[N];
int n, m, l;
ll dp[N][N] = {1}, k;
string s;

ll func() {
	for (int i = 0; i < l; i++)
		for (int j = 0; j <= l; j++)
			if (s[i] == '(') {
				if (j)
					dp[i + 1][j] = dp[i][j - 1];
				else
					dp[i + 1][j] = 0;
			}
			else if (s[i] == ')' || !j)
				dp[i + 1][j] = dp[i][j + 1];
			else
				dp[i + 1][j] = min(dp[i][j - 1] + dp[i][j + 1], oo);
	return dp[l][0];
}
int main() {
	ios_base::sync_with_stdio(false);
	cin >> n >> m >> k;
	l = n + m - 1;
	for (int i = 0; i < l; i++)
		p[i] = pii(1e9, i);
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++) {
			int t; cin >> t;
			p[i + j].first = min(p[i + j].first, t);
		}
	for (int i = 0; i < l; i++)
		s += '?';
	sort(p, p + l);
	for (int i = 0; i < l; i++) {
		s[p[i].second] = '(';
		ll t = func();
		if (t < k) {
			k -= t;
			s[p[i].second] = ')';
		}
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++)
			cout << s[i + j];
		cout << '\n';
	}
	return 0;
}
