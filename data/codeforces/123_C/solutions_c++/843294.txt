#include <cstdio>
#include <iostream>
#include <string>
#include <cstring>
#include <map>
#include <algorithm>
#include <vector>

using namespace std;

int n, m;
int str[200];
long long mem[200][200];

long long calc(int i, int sum) {
	long long & aux = mem[i][sum];
	//~ cout << i << " " << sum << " " << n << " " << m << " " << aux << endl;
	if(i >= n+m-1) return ((sum==0)?1:0);
	if(sum > (n+m-1)/2) return 0;
	if(aux < 0) {
		if(sum == 0) {
			if(str[i] == -1) aux = 0;
			else aux = calc(i+1, 1);
		} else {
			if(str[i] != 0) aux = calc(i+1, sum+str[i]);
			else {
				aux = calc(i+1, sum+1) + calc(i+1, sum-1);
				if(aux > (1LL<<60)) aux = (1LL<<60);
			}
		}
	}
	
	return aux;
}

int main(void)
{
	int i, j;
	long long k;
	pair<int, int> aux[10100];
	
	cin >> n >> m >> k;
	for(i=0; i<n; i++) for(j=0; j<m; j++) {
		cin >> aux[m*i+j].first;
		aux[m*i+j].second = i+j;
	}
	
	sort(aux, aux+n*m);
	
	for(i=0; i<n+m-1; i++) str[i]=0;
	
	for(i=0; i<n*m; i++) {
		int d = aux[i].second;
		if(str[d] != 0) continue;
		
		str[d] = 1;
		memset(mem, -1, sizeof(mem));
		
		long long low = calc(0, 0);
		if(k > low) str[d] = -1, k -= low;
	}
	
	for(i=0; i<n; i++) {
		for(j=0; j<m; j++) {
			cout << ((str[i+j]>0)?'(':')');
		}
		cout << endl;
	}
	
	return 0;
}
