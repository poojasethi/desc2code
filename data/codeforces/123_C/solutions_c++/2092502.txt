#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
long long T,dp[401][401];
int x[401],n,m,a1,d[401],v[401];
bool cmp(int i,int j){ return x[i] < x[j]; }
int main(){
    scanf("%d%d%I64d",&n,&m,&T);
    for(int i=0;i<=n+m;i++) x[i] = n*m+1,d[i] = i;
    for(int i=0;i<n;i++) for(int j=0;j<m;j++) scanf("%d",&a1), x[i+j] = min(x[i+j],a1);
    sort(d,d+n+m-1,cmp);
    dp[0][0] = 1;
    for(int i=0;i<n+m-1;i++){
       v[d[i]] = 1;
       for(int j=1;j<=n+m-1;j++)
          for(int k=0;k<=n+m-1;k++){
              if(j || k) dp[j][k] = 0;
              if(v[j-1] != 1) dp[j][k] += dp[j-1][k+1];
              if(v[j-1] != 2) dp[j][k] += ( k ? dp[j-1][k-1] : 0);
              if(dp[j][k] > T) dp[j][k] = T+1;// printf("%d %d ==> %I64d\n",j,k,dp[j][k]);
          }
       if(dp[n+m-1][0] < T) T-= dp[n+m-1][0], v[d[i]] = 2;
       //printf("%d %d %I64d\n",x[d[i]],d[i],dp[n+m-1][0]);
    }
    //for(int i=0;i<n+m-1;i++) printf("%d ",v[i]);printf("\n");
    for(int i=0;i<n;i++){
       for(int j=0;j<m;j++)
           v[i+j] == 1 ? printf("(") : printf(")");
       printf("\n");   
    } 
    //system("pause");
    return 0;
}
