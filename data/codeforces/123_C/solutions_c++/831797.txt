#include <cstdio>
#include <cstring>
#include <vector>
#include <iostream>
using namespace std;

typedef long long ll;
int prior[201];
char res[201];
int n,m;
int l;

int next_int()
{
	int t;
	scanf("%d",&t);
	return t;
}

const ll INF=2e18;
ll dp[201][201];

ll solve(int p, int b)
{
	if(b<0) return 0;
	if(p==l) return b==0;
	if(dp[p][b]==-1)
	{
		dp[p][b]=0;
		if(res[p]!=')')
			dp[p][b]+=solve(p+1,b+1);
		if(res[p]!='(')
			dp[p][b]+=solve(p+1,b-1);
		dp[p][b]=min(INF,dp[p][b]);
	}
	return dp[p][b];
}

ll count()
{
	memset(dp,-1,sizeof(dp));
	return solve(0,0);
}

int main()
{
	memset(prior,0x7F,sizeof(prior));
	ll k;
	cin>>n>>m>>k;
	k--;
	l=n+m-1;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			prior[i+j]=min(prior[i+j],next_int());
	for(int i=0;i<l;i++)
	{
		int minid=-1;
		for(int j=0;j<l;j++)
		{
			if(res[j]==0)
			{
				if(minid==-1||prior[j]<prior[minid])
					minid=j;
			}
		}
		res[minid]='(';
		ll tc=count();
		if(tc<=k)
		{
			k-=tc;
			res[minid]=')';
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			printf("%c",res[i+j]);
		}
		printf("\n");
	}
	return 0;
}
