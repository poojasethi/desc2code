#include<stdio.h>
#define N 105
using namespace std;
typedef long long LL;
LL f[2*N][N],kth,number;
int X[N][N],Y[N][N],must[N][N],idx[N*N],idy[N*N];
int n,m,a,b,x,y,i,j;
void up(LL &x,LL y){
    if (x==-1||y==-1) x=-1;
    else if ((x+=y)>=kth) x=-1;
}
LL DP(){
    int i=1,L=(n+m-1)/2;f[0][0]=1;
    for (int x=1,y=1;y<=m;i++){
        for (int k=0;k<=L;k++){
            f[i][k]=0;
            if (k>0&&must[x][y]!=-1)
                up(f[i][k],f[i-1][k-1]);
            if (must[x][y]!=1)
                up(f[i][k],f[i-1][k+1]);
      }if (x<n) x++;else y++;
  }return f[i-1][0];
}
int main(){
    scanf("%d%d%I64d",&n,&m,&kth);
    for (i=1;i<=n;i++)
        for (j=1;j<=m;j++)
            scanf("%d",&x),idx[x]=i,idy[x]=j;
    for (i=n;i;i--)
        for (j=1;j<=m;j++)
            if (i==n||j==1) X[i][j]=i,Y[i][j]=j;
            else X[i][j]=X[i+1][j-1],Y[i][j]=Y[i+1][j-1];
    for (i=1;i<=n*m;i++){
        x=idx[i];y=idy[i];
        a=X[x][y];b=Y[x][y];
        if (must[a][b]) continue;
        must[a][b]=1;
        number=DP();
        if (number==-1) continue;
        must[a][b]=-1;
        kth-=number;
  }for (i=1;i<=n;i++,puts(""))
    for (j=1;j<=m;j++)
        putchar(must[X[i][j]][Y[i][j]]==1?'(':')');
}
