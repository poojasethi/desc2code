#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
typedef long long LL;
int n, m;
LL k;
int p[105][105];
int pp[10005];
int u[205];
vector<int> o;
char s[205];
LL calc(int L, int R) {
	LL dp[205]={};
	dp[0] = 1;
	for(int i=L;i<=R;i++) {
		LL nxt[205]={};
		if(s[i]=='(') {
			for(int j=i-L+1;j>0;j--) dp[j]=dp[j-1];
			dp[0] = 0;
		} else if(s[i]==')') {
			for(int j=0;j<i-L+1;j++) dp[j]=dp[j+1];
			dp[i-L+1] = 0;
		} else {
			for(int j=0;j<=i-L+1;j++) nxt[j]=
				min(1LL<<61, (j?dp[j-1]:0)+(j<i-L+1?dp[j+1]:0));
			for(int j=0;j<=i-L+1;j++) dp[j]=nxt[j];
		}
	}
	return dp[0];
}
int main(void) {
	scanf("%d%d%I64d", &n, &m, &k);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++) {
			scanf("%d", &p[i][j]);
			pp[p[i][j]] = i+j;
		}
	for(int i=1;i<=n*m;i++)
		if(u[pp[i]]==0) { o.push_back(pp[i]); u[pp[i]] = 1;}
	for(int i=0;i<n+m-1;i++) {
		int pos = o[i];
		s[pos] = '(';
		LL v = calc(2, n+m);
		//printf("v=%I64d\n", v);
		if(v<k) { k -= v; s[pos] = ')'; }
	}
	for(int i=1;i<=n;i++,puts(""))
		for(int j=1;j<=m;j++)
			printf("%c", s[i+j]);
	return 0;
}
