#include <iostream>
#include <algorithm>

using namespace std;

unsigned long long K[101][101];
unsigned long long nvar(char *s,int n,unsigned long long inf) {
	K[0][0]=1;
	int m=n/2;
	for (int i=1;i<=m;i++) 
		K[0][i]=(s[i-1]!=')' ? K[0][i-1] :0);

	for (int i=1;i<=m;i++) {
		if (inf<(K[i][i]=s[2*i-1]!='(' ? K[i-1][i] : 0))
			K[i][i]=inf;
		for (int j=i+1;j<=m;j++) {
			if (inf<(K[i][j]=(s[i+j-1]!='('? K[i-1][j]:0)+(s[i+j-1]!=')' ? K[i][j-1]:0)))
				K[i][j]=inf;
		}
	}
	return K[m][m];
}

int X[101][101];
char S[500];
pair<int,int> Y[300];


int main(void)
{
	int n,m,a;
	unsigned long long k,x;
	cin>>n>>m>>k;
	for (int i=0;i<n;i++)
		for (int j=0;j<m;j++)
			cin>>X[i][j];
	for (int i=1;i<n;i++)
		for (int j=0;j<m-1;j++)
			X[i][j]=min(X[i][j],X[i-1][j+1]);
	for (int i=0;i<n;i++)
		Y[i]=pair<int,int>(X[i][0],i);
	for (int i=1;i<m;i++)
		Y[n+i-1]=pair<int,int>(X[n-1][i],n+i-1);
	a=n+m-1;
	sort(Y,Y+a);
	fill(S,S+a,0);
	for (int i=0;i<a;i++) {
		S[Y[i].second]='(';
		x=nvar(S,a,k+1);
		if (x<k) {
			k-=x;
			S[Y[i].second]=')';
		}
	}
	for (int i=0;i<n;i++) {
		for (int j=0;j<m;j++) {
			cout<<S[i+j];
		}
		cout<<endl;
	}
	return 0;
}
