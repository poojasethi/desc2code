#include<cstdio>
#include<algorithm>
using namespace std;

#define long long long
#define pii pair<int, int> 

const int MAX_N = 205;

int n, m;
long k;
long dp[MAX_N][MAX_N];
char ans[MAX_N];
pii order[MAX_N];

int main(){
  scanf("%d %d %I64d", &n, &m, &k), k--;
  fill_n(order, n + m, pii(n * m + 1, 0));
  fill_n(ans, n + m, -1);
  for(int i = 0; i < n; i++){
    for(int j = 0; j < m; j++){
      int x;
      scanf("%d", &x);
      order[i + j] = min(order[i + j], pii(x, i + j));
    }
  }

  sort(order, order + n + m - 1);
  for(int i = 0; i < n + m - 1; i++){
    int o = order[i].second;
    ans[o] = '(';
    
    fill_n(&dp[0][0], sizeof(dp) / sizeof(long), 0);
    dp[n + m - 1][0] = 1;
    for(int i = n + m - 1; i --> 0; ){
      for(int j = n + m - 1; j --> 0; ){
	
#define add(a, b) a = min(a + b, k + 1)

	if(ans[i] == '(' or ans[i] == -1){
	  add(dp[i][j], dp[i + 1][j + 1]);
	}
	if((ans[i] == ')' or ans[i] == -1) and j > 0){
	  add(dp[i][j], dp[i + 1][j - 1]);
	} 
      }
    }

    if(k - dp[0][0] >= 0){
      k -= dp[0][0];
      ans[o] = ')';
    }
  }
  
  for(int i = 0; i < n; i++){
    for(int j = 0; j < m; j++){
      printf("%c", ans[i + j]);
    }
    printf("\n");
  }
}
