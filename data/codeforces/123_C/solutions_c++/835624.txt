#include <iostream>
#include <cstdio>
using namespace std;

char str[201],ans[201][201];
long long f[201][101];
int p[201];
long long k;
int n,m;

long long calc()
{
   f[0][0]=1;
   for (int i=1;i<n+m;i++)
      for (int j=0;j<=(n+m-1)/2;j++)
      {
         f[i][j]=0;
         if (str[i]!=')'&&j)
            f[i][j]=min(k,f[i][j]+f[i-1][j-1]);
         if (str[i]!='('&&j<n+m-1)
            f[i][j]=min(k,f[i][j]+f[i-1][j+1]);
      }
   return f[n+m-1][0];
}

int main()
{
   cin>>n>>m>>k;
   for (int i=1;i<=n;i++)
      for (int j=1;j<=m;j++)
      {
         int a;
         cin>>a;
         if (!p[i+j-1]||p[i+j-1]>a)
            p[i+j-1]=a;
      }
   for (int i=1;i<n+m;i++)
   {
      int t=0;
      for (int j=1;j<n+m;j++)
         if (!str[j]&&(!t||p[j]<p[t]))
            t=j;
      str[t]='(';
      long long way=calc();
      if (way<k)
      {
         str[t]=')';
         k-=way;
      }
   }
   for (int i=1;i<=n;i++)
      for (int j=1;j<=m;j++)
         ans[i][j]=str[i+j-1];
   for (int i=1;i<=n;i++)
      puts(ans[i]+1);
   return 0;
}
