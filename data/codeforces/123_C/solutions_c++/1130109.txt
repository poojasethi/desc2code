#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,a[210],id[210],ans[210];
long long k,f[210][210];
bool cmp(int x,int y){return a[x]<a[y];}
void dp(){
    memset(f,0,sizeof(f));
    f[0][0]=1;
    for(int i=1;i<=n+m-1;++i){
        if(ans[i]=='(')for(int j=0;j<=n+m;++j)f[i][j+1]+=f[i-1][j];
        else if(ans[i]==')')for(int j=0;j<=n+m;++j)f[i][j]+=f[i-1][j+1];
        else{
            for(int j=0;j<=n+m;++j){
                f[i][j+1]+=f[i-1][j];
                if(f[i][j+1]>k)f[i][j]=k+1;
                f[i][j]+=f[i-1][j+1];
                if(f[i][j]>k)f[i][j]=k+1;
            }
        }
    }
}
int main(){
    scanf("%d%d%I64d\n",&n,&m,&k);
    for(int i=1;i<=n;++i)
        for(int j=1,x;j<=m;++j){
            scanf("%d",&x);
            if(a[i+j-1]==0)a[i+j-1]=x;
            else a[i+j-1]=min(a[i+j-1],x);
        }
    for(int i=1;i<=n+m-1;++i){
        id[i]=i;ans[i]=-1;
    }
    sort(id+1,id+n+m,cmp);
    for(int i=1;i<=n+m-1;++i){
        ans[id[i]]='(';
        dp();if(f[n+m-1][0]>=k)continue;
        k-=f[n+m-1][0];
        ans[id[i]]=')';
    }
    for(int i=1;i<=n;++i){
        for(int j=1;j<=m;++j)
            printf("%c",ans[i+j-1]);
        printf("\n");
    }
    return 0;
}
