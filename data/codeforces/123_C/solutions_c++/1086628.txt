#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;
const long long M=1100000000000000000ll;
const int N= 205;
long  long  c[N],a[N],f[N][N],d[N];
long  long m,n;

bool cmp(const int &x,const int &y)
{
    return c[x]<c[y];
}

long long get(long long p)
{
    long long i,j;
    memset(f,0,sizeof(f));
    f[0][0]=1;
    for (i=1; i<=m+n-1; i++)
     for (j=0; j<=i; j++)
     {
         if (a[i]!=-1 && j>0) f[i][j]+=f[i-1][j-1];
         if (i!=p && a[i]!=1) f[i][j]+=f[i-1][j+1];
         if (f[i][j]>M) f[i][j]=M;
     }
    return f[m+n-1][0];
}

int main()
{
    long long  o,i,j,x;
    cin>>m>>n>>o;
    for (i=1; i<=m; i++)
     for (j=1; j<=n; j++)
      {
          cin>>x;
          if (c[i+j-1]==0 || x<c[i+j-1]) c[i+j-1]=x;
      }
    for (i=1;  i<=m+n-1;  i++) d[i]=i;
    sort(d+1,d+m+n,cmp);
    for (i=1;  i<=m+n-1; i++)
    {
        x=get(d[i]);
        if (x>=o) a[d[i]]=1; else {o-=x; a[d[i]]=-1;}
    }
    for (i=1; i<=m; i++)
    {
     for (j=1; j<=n; j++)
      if (a[i+j-1]==1) printf("(") ;else printf(")");
     printf("\n");
    }
}
