#include <cstdio>
#include <iostream>
#include <cstring>
#include <vector>
#include <algorithm>

using namespace std;

int q[300];
char r[300];
long double D[300][300];

int main () {
  int n, m;
  long long t;
  cin>>n>>m>>t;
  memset (q, 63, sizeof (q));
  for (int i = 0; i < n; i++)
    for (int j = 0; j < m; j++) {
      int p;
      cin >> p;
      q[i + j] = min (q[i + j], p);
    }
  int l = n + m - 1;
  vector <pair <int, int> > v;
  for (int i = 0; i < l; i++) {
    v.push_back (make_pair (q[i], i));
  }
  sort (v.begin (), v.end ());
  --t;
  for (int i = 0; i < l; i++) {
    r[v[i].second] = '(';
    for (int j = 0; j <= l; j++)
      for (int k = 0; k <= l; k++)
	D[j][k] = 0;

    D[0][0] = 1;
    for (int j = 0; j < l; j++) {
      for (int k = 0; k < l; k++) {
	if (r[j] != ')') D[j + 1][k + 1] += D[j][k];
	if (r[j] != '(' && k > 0) D[j + 1][k - 1] += D[j][k];
      }
    }
    //cerr << t << " " << D[l][0] << endl;

    if (D[l][0] <= t) {
      t -= (long long)D[l][0];
      r[v[i].second] = ')';
    }
  }


  for (int i = 0; i < n; i++) {
    for (int j = 0; j < m; j++)
      printf ("%c", r[i + j]);
    puts ("");
  }
	  

  return 0;
}
