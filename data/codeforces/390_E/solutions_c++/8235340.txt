#include <cstdio>
#include <algorithm>
#define maxn 4000005
using namespace std;
long long a,b[maxn],c[maxn],B[maxn],C[maxn],d,e,f,g,h,i,j,k,l,m,n,x1,x2,y1,y2;
inline void add(long long *a,long long x,long long y,long long n){for(;x<=n;x+=x&(-x))a[x]+=y;}
inline long long sum(long long *a,long long x,long long n){long long s=0ll;for(;x;x-=x&(-x))s+=a[x];return s;}
inline void Add(long long l,long long r,long long s){
	add(b,l,s,n);
	add(b,r+1,-s,n);
	add(c,l,1ll*s*l,n);
	add(c,r+1,1ll*-(r+1)*s,n);
}
inline void Add2(long long l,long long r,long long s){
	add(B,r+1,-s,m);
	add(B,l,s,m);
	add(C,l,s*l,m);
	add(C,r+1,-(r+1)*s,m);
}
inline long long Sum(long long l,long long r){
	return 1ll*sum(b,r,n)*(r+1)-1ll*l*sum(b,l-1,n)-sum(c,r,n)+sum(c,l-1,n);
}
inline long long Sum2(long long l,long long r){
	return 1ll*sum(B,r,m)*(r+1)-1ll*l*sum(B,l-1,m)-sum(C,r,m)+sum(C,l-1,m);
}

int main(){
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	for(;k;k--){
		scanf("%I64d",&e);
		if(e==0){
			scanf("%I64d%I64d%I64d%I64d%I64d",&x1,&y1,&x2,&y2,&d);
			Add(x1,x2,1ll*d*(y2-y1+1));
			Add2(y1,y2,1ll*d*(x2-x1+1));
			h+=1ll*d*(x2-x1+1)*(y2-y1+1);
		}
		else{
			scanf("%I64d%I64d%I64d%I64d",&x1,&y1,&x2,&y2);
			printf("%I64d\n",Sum(x1,x2)+Sum2(y1,y2)-h);
		}
	}
	return 0;
}
