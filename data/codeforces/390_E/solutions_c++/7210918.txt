#include <bits/stdc++.h>
using namespace std;
int n,m,k;
const int MAXN  = 10+4*1000000;
long long f[MAXN*4][2],add[MAXN*4][2];
void pushDown(int p,int op,int l,int r,int m){
    if(add[p][op]){
        f[p<<1][op]+=add[p][op]*(m-l+1);
        f[p<<1|1][op]+=add[p][op]*(r-m);
        add[p<<1][op]+=add[p][op];
        add[p<<1|1][op]+=add[p][op];
        add[p][op] = 0;
    }
}
void update(int p,int l,int r,int u,int v,int op,long long value){
    if(v < l || r < u) return;
    if(u <= l && r <= v) {
        add[p][op]+=value;
        f[p][op]+=value*(r-l+1);
        return;
    }
    int m = (l+r)>>1;
    pushDown(p,op,l,r,m);
    update(p<<1,l,m,u,v,op,value);
    update(p<<1|1,m+1,r,u,v,op,value);
    f[p][op] = f[p<<1][op] + f[p<<1|1][op];
}
long long query(int p,int l,int r,int u,int v,int op){
    if(v < l || r < u) return 0;
    if(u <= l && r <= v) return f[p][op];
    int m = (l+r)>>1;
    pushDown(p,op,l,r,m);
    long long t1 = query(p<<1,l,m,u,v,op);
    long long t2 = query(p<<1|1,m+1,r,u,v,op);
    return t1+t2;
}
int main(){
    //freopen("A.inp","r",stdin);
    scanf("%d%d%d",&n,&m,&k);
    for(int i = 0;i < k;++i){
        int q,x1,x2,y1,y2;
        long long value;
        scanf("%d",&q);
        if(q == 0){
            scanf("%d%d%d%d%I64d",&x1,&y1,&x2,&y2,&value);
            update(1,1,n,x1,x2,0,value*(y2-y1+1));
            update(1,1,m,y1,y2,1,value*(x2-x1+1));
        }
        else{
            scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
            long long t1 = query(1,1,n,x1,x2,0);
            long long t2 = query(1,1,m,y1,y2,1);
            printf("%I64d\n",t1+t2-f[1][0]);
        }
    }
return 0;
}
