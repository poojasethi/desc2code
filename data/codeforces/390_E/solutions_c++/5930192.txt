#include<cstdio>
typedef long long ll;
const int N=4000005;
ll x[N],y[N],f[N],g[N],s,Q,x1,y1,x2,y2,z,v;
ll get(ll *a,ll *b,ll x){ll S=0; for (ll i=x;i;i-=i&-i) S+=b[i]*x-a[i]; return S;}
void add(ll *a,ll *b,ll x,ll v){for (ll i=x;i<=*a;i+=i&-i) a[i]+=v*(x-1),b[i]+=v;}
int main()
{
	scanf("%I64d%I64d%I64d",x,y,&Q);
	while (Q--){
		scanf("%I64d%I64d%I64d%I64d%I64d",&z,&x1,&y1,&x2,&y2);
		if (z) printf("%I64d\n",get(x,f,x2)-get(x,f,x1-1)+get(y,g,y2)-get(y,g,y1-1)-s);
		else scanf("%I64d",&v),++x2,++y2,s+=(x2-x1)*(y2-y1)*v,
			add(x,f,x1,v*(y2-y1)),add(x,f,x2,-v*(y2-y1)),
			add(y,g,y1,v*(x2-x1)),add(y,g,y2,-v*(x2-x1));
	}
	return 0;
}
