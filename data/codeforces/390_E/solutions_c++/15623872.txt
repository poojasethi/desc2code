#define mod 1000000007

#define lowbit(x) ((x)&(-x))
#define maxn 4*1000006
using namespace std;
long long  sum1[3][maxn],sum2[3][maxn],all;
int n,m;

void add(long long s,long long v,long long *f,int t)
{
    while(s<=t)
    {
        f[s]+=v;
        s+=lowbit(s);
    }
}

long long sum(int s,long long *f)
{
    long long ans=0;
    while(s>0)
    {
        ans+=f[s];
        s-=lowbit(s);
    }
    return ans;
}

long long summ(int s,long long *f1,long long *f2)
{
    return sum(s,f1)*s+sum(s,f2);
}

int main ()
{
    //freopen("d:\\in.txt","r",stdin);
    scanf("%d%d",&n,&m);
    int q;
    scanf("%d",&q);
    for(int i=1;i<=q;i++)
    {
        long long  ty,x1,x2,y1,y2;
        scanf("%lld%lld%lld%lld%lld",&ty,&x1,&y1,&x2,&y2);
        if(!ty)
        {
            long long  v;
            scanf("%lld",&v);
            all+=(x2-x1+1)*(y2-y1+1)*v;
            add(x1,(y2-y1+1)*v,sum1[1],n);
            add(x2+1,-(y2-y1+1)*v,sum1[1],n);
            add(x1,-(y2-y1+1)*v*(x1-1),sum1[2],n);
            add(x2+1,(y2-y1+1)*v*x2,sum1[2],n);
            add(y1,(x2-x1+1)*v,sum2[1],m);
            add(y2+1,-(x2-x1+1)*v,sum2[1],m);
            add(y1,-(x2-x1+1)*v*(y1-1),sum2[2],m);
            add(y2+1,(x2-x1+1)*v*y2,sum2[2],m);
        }
        else
        {
           long long ans1=summ(x2,sum1[1],sum1[2])-summ(x1-1,sum1[1],sum1[2])+summ(y2,sum2[1],sum2[2])-summ(y1-1,sum2[1],sum2[2])-all;
           printf("%lld\n",ans1);
        }
    }
    return 0;
}
