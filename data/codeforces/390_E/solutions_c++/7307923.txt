#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <vector>
#include<queue>
#include <algorithm>
typedef long long LL;
using namespace std;
const  int  mx=4000009;
LL x[mx],y[mx],qx[mx],py[mx];
LL t,s;
LL L(LL i)
{
    return i&(-i);
}

void  add( LL *l,LL *k,LL r,LL v)
{
     for (LL i=r;i<=*l;i+=L(i))
          l[i]+=v,k[i]+=(r-1)*v;
}
LL sum (LL *l,LL *k,LL r)
{
   LL A=0;
   for (LL i=r;i>0;i-=L(i))
    A+=l[i]*r-k[i];
   return A;
}

int main()
{
    s=0;
    scanf("%I64d%I64d%I64d",x,y,&t);
    while(t--)
    {
        LL  v,c,x1,x2,y1,y2;
        scanf("%I64d%I64d%I64d%I64d%I64d",&c,&x1,&y1,&x2,&y2);
        if (c)
        {
            LL ans;
            ans=sum(x,qx,x2)+sum(y,py,y2)-sum(x,qx,x1-1)-sum(y,py,y1-1)-s;
            printf("%I64d\n",ans);
        }
        else
        {
            x2++;y2++;
            scanf("%I64d",&v);
            s+=v*(x2-x1)*(y2-y1);
            add(x,qx,x1,v*(y2-y1));
            add(y,py,y1,v*(x2-x1));
            add(x,qx,x2,-v*(y2-y1));
            add(y,py,y2,-v*(x2-x1));
        }
    }
}
