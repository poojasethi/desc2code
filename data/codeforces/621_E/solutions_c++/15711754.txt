#include<bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define st first
#define nd second
#define pii pair<int,int>
#define int long long

const int N=1e5+5;
const int inf=1e9+7;
const int mod=1e9+7;

int n,b,k,x,qq,i,j;

class matris{ public:
	int A[105][105];

	matris(){
		memset(A,0,sizeof A);
	}

	matris operator*(matris B){

		int i,j,k;

		matris ans;

		for(i=0 ; i<=x-1 ; i++)
			for(j=0 ; j<=x-1 ; j++)
				for(k=0 ; k<=x-1 ; k++)
					ans[i][j]=(ans[i][j]+A[i][k]*B[k][j]%mod)%mod;

		return ans;
	}

	int* operator[](int x){
		return A[x];
	}
};

matris dp;

matris FE(matris x,int k){
	if(k==1)
		return x;

	matris temp=FE(x,k/2);

	temp=temp*temp;

	if(k&1)
		temp=temp*x;

	return temp;
}

main(){

	scanf("%lld %lld %lld %lld",&n,&b,&k,&x);

	for(i=1 ; i<=n ; i++){
		scanf("%lld",&qq);
		qq%=x;
		for(j=0 ; j<=x-1 ; j++)
			dp[j][(j*10+qq)%x]++;
	}

	dp=FE(dp,b);

	cout<<dp[0][k];
}
