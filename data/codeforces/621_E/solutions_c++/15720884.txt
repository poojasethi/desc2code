#include <cstdio>
#include <iostream>
#include <math.h>
#include <vector>
#include <algorithm>
#include <utility>

using namespace std;

int main() {
	long long modulus=1000000007;
	long long n, b, k, x; cin>>n>>b>>k>>x; //k is reslut, x is mod
	vector<long long> ans;
	vector<long long> akt(x, 0); 
	
	for (int i=0; i<n; i++) {
		long long tmp; cin>>tmp;
		akt[tmp%x]++;
	} 
	
	long long mul=10;
	while (b>0) {
		if (b%2==1) if (ans.size()==0) ans=akt;
					else {
						vector<long long> tmp(x, 0);
						for (int i=0; i<ans.size(); i++) for (int j=0; j<akt.size(); j++) 
							 tmp[(i*mul+j)%x]=(tmp[(i*mul+j)%x]+ans[i]*akt[j])%modulus;
						ans=tmp;
					}
		vector<long long> neu(x, 0);
		for (int i=0; i<akt.size(); i++) for (int j=0; j<akt.size(); j++) 
			neu[(i*mul+j)%x]=(neu[(i*mul+j)%x]+akt[i]*akt[j])%modulus;
		akt=neu;
		
		mul=(mul*mul)%x;
		b/=2;
	}
	
	for (int i=0; i<x; i++) cerr<<ans[i]<<" "; cerr<<endl;
	cout<<ans[k];
	return 0;
}
