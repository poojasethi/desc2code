#include<cstdio>
#include<cstring>
int a[110][110],v[50010],f[110][110],n,m,b,k,mo=1000000007,i,j,l;
inline void mul(int a[][110],int b[][110]){
	static int tmp[110][110];
	memset(tmp,0,sizeof(tmp));
	for(i=0;i<m;i++)
		for(j=0;j<m;j++)
			for(l=0;l<m;l++)
				tmp[i][j]=(tmp[i][j]+(long long)a[i][l]*b[l][j]%mo)%mo;
	memcpy(a,tmp,sizeof(tmp));
}
inline void power(int a[][110],int n){
	for(;n;n>>=1){
		if(n&1)mul(a,f);
		mul(f,f);
	}
}
int main(){
	scanf("%d%d%d%d",&n,&b,&k,&m);
	for(i=1;i<=n;i++)scanf("%d",&v[i]);
	for(i=0;i<m;i++)
		for(j=1;j<=n;j++) f[(i*10+v[j])%m][i]++;
	for(i=0;i<m;i++)a[i][i]=1;
	power(a,b);
	printf("%d\n",a[k][0]);
	return 0;
}
