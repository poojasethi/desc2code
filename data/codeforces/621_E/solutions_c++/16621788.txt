#include <bits/stdc++.h>
using namespace std;

const int maxx = 105 , mod = 1e9+7;

int n,b,k,x,ans[maxx],cur[maxx],tmp[maxx];

int main(){
	
	scanf("%d%d%d%d",&n,&b,&k,&x);
	
	for( int i = 1 , a ; i <= n ; i++ ){
		scanf("%d",&a);
		cur[a % x]++;
	}
	ans[0] = 1;
	int power = 10;
	while( b ){
		if( b & 1 ){
			swap(tmp , ans);
			memset(ans , 0 , sizeof ans);
			for( int i = 0 ; i < x ; i++ )
				for( int j = 0 ; j < x ; j++ ){
					int val = (i * power + j) % x;
					ans[val] = (ans[val] + 1ll * tmp[i] * cur[j]) % mod;
				}
		}
		swap(tmp , cur);
		memset(cur , 0 , sizeof cur);
		for( int i = 0 ; i < x ; i++ )
			for( int j = 0 ; j < x ; j++ ){
				int val = (i * power + j) % x;
				cur[val] = (cur[val] + 1ll * tmp[i] * tmp[j]) % mod;
			}
		power = power * power % x;
		b >>= 1;
	}
	
	printf("%d\n",ans[k]);
	
	return 0;
}
