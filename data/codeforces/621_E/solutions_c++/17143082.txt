#include<cstdio>
#include<cstring>
using namespace std;
typedef __int64 ll;
#define M 1000000007
ll a[11],A[101][101],B[101][101],C[101][101];
int n,k,b,x;
void mul(ll A[][101],ll B[][101],int n){
    for(int i=0;i<n;++i){
        for(int j=0;j<n;++j){
            C[i][j]=0;
            for(int k=0;k<n;++k) C[i][j]=(C[i][j]+A[i][k]*B[k][j])%M;
        }
    }
    for(int i=0;i<n;++i){
        for(int j=0;j<n;++j) A[i][j]=C[i][j];
    }
}
ll solve(ll b,int n){
    for(int i=0;i<n;++i) A[i][i]=1;
    while(b){
        if(b%2) mul(A,B,n);
        mul(B,B,n);b/=2;
    }
    return A[0][k];
}
int main(){
    int val;ll ans;
    while(~scanf("%d%d%d%d",&n,&b,&k,&x)){
        memset(a,0,sizeof(a));
        memset(B,0,sizeof(B));
        for(int i=0;i<n;++i){
            scanf("%d",&val);++a[val];
        }
        for(int i=0;i<x;++i){
            for(int j=1;j<10;++j) B[i][(10*i+j)%x]+=a[j];
        }
        ans=solve(b,x);
        printf("%I64d\n",ans);
    }
    return 0;
}
