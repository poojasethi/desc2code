#include <bits/stdc++.h>
using namespace std;

const int base=1e9+7;
int n,b,k,x;
int d[101];
struct matrix{
	int a[101][101];
} a;

void init() {
	for(int i=0;i<n;i++) {
		int S;
		scanf("%i",&S);
		for(int j=0;j<x;j++) 
			a.a[j][(j*10+S)%x]++;
		++d[S%x];
	}
}

matrix nhan(matrix X,matrix Y) {
	matrix z;
	memset(z.a,0,sizeof(z.a));
	for(int i=0;i<x;i++) 
		for(int j=0;j<x;j++)
			for(int k=0;k<x;k++) 
				z.a[i][j]=(z.a[i][j]+1ll*X.a[i][k]*Y.a[k][j])%base;
	return z;
}

matrix luythua(matrix x,int y) {
	if (y==1) return x;
	matrix t=luythua(x,y/2);
	t=nhan(t,t);
	if (y%2) t=nhan(t,x);
	return t;
}

int main() {
	//freopen("input.inp","r",stdin);
	cin>>n>>b>>k>>x;
	init();
	matrix tam=luythua(a,b);
	cout<<tam.a[0][k];
	return 0;
}
