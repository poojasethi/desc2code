#include <bits/stdc++.h>
using namespace std;

const int M = 1000000007;
typedef long long LL;
const int MX = 105;

int cnt[15];
LL A[MX][MX], B[MX][MX], t[MX][MX];
int m;

void mul(LL A[MX][MX], LL B[MX][MX], LL C[MX][MX]) {
	for (int i = 0; i < m; i++) {
		for (int j = 0; j < m; j++) {
			t[i][j] = 0;
			for (int k = 0; k < m; k++) t[i][j] += A[i][k] * B[k][j] % M;
		}
	}
	for (int i = 0; i < m; i++) {
		for (int j = 0; j < m; j++) {
			C[i][j] = t[i][j] % M;
		}
	}
}

int main() {
	
	int n, b, r;
	scanf("%d%d%d%d", &n, &b, &r, &m);
	for (int i = 0; i < n; i++) {
		int k;
		scanf("%d", &k);
		cnt[k]++;
	}
	for (int i = 0; i < m; i++) {
		A[i][i] = 1;
		for (int k = 1; k <= 9; k++) {
			int j = (i * 10 + k) % m;
			B[i][j] += cnt[k];
		}
	}
	while (b) {
		if (b & 1) mul(A, B, A);
		mul(B, B, B);
		b /= 2;
	}
	printf("%lld\n", A[0][r]);
	return 0;
}
