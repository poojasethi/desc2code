#include<cstdio>
#include<ctype.h>
#define C 1000000007ll
#define X 100
#define I(x,y) x=(x+y)%C
inline int get(){
	char c;
	while (!isdigit(c=getchar()));int v=c-48;
	while (isdigit(c=getchar())) v=v*10+c-48;
	return v;
}
int n,b,k,x,a[10];
long long td0[X][X],d0[X][X],td[X][X],d[X][X];
inline int work(int b){
	for (int i=0;i<x;i++) d[i][i]=1;
	for (;b;b>>=1){
		if (b&1){
			for (int i=0;i<x;i++) for (int j=0;j<x;j++){
				td[i][j]=0;
				for (int k=0;k<x;k++) I(td[i][j],(d[i][k])*d0[k][j]%C);
			}
			for (int i=0;i<x;i++) for (int j=0;j<x;j++) d[i][j]=td[i][j];
		}
		for (int i=0;i<x;i++) for (int j=0;j<x;j++){
			td0[i][j]=0;
			for (int k=0;k<x;k++) I(td0[i][j],(d0[i][k])*d0[k][j]%C);
		}
		for (int i=0;i<x;i++) for (int j=0;j<x;j++) d0[i][j]=td0[i][j];
	}
}
int main(){
	n=get();b=get();k=get();x=get();
	for (int i=0;i<n;i++) a[get()]++;
	for (int i=0;i<x;i++)
		for (int j=0;j<10;j++)
			I(d0[i][(i*10+j)%x],a[j]);
	work(b);
	printf("%d\n",int(d[0][k]));
	return 0;
}
