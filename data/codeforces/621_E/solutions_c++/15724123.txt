#include <cstdio>
#include <cstring>
using namespace std;
typedef long long ll;
const int maxn = 1e2  + 10;
const int mod = 1e9 + 7;
struct str{
	ll num[maxn];
};
int n, b, k, x;
str mul(const str& a, const str& b, int B){
	str c;
	memset(c.num, 0, sizeof(c.num));
	for(int i = 0; i < x; ++ i){
		for(int j = 0; j < x; ++ j){
			c.num[(i*B+j)%x] = (c.num[(i*B+j)%x] + a.num[i]*b.num[j]) % mod;
		}
	}
	return c;
}

str fast_pow(str a, int xx){
	str b;
	memset(b.num, 0, sizeof(b.num));
	b.num[0] = 1;
    int B = 10%x;
	while(xx){
		if(xx&1){
			b = mul(b, a, B);
		}
		xx >>= 1;
		a = mul(a, a, B);
		B = B*B%x;
	}
	return b;
}

int main(){
	while(~scanf("%d%d%d%d", &n, &b, &k, &x)){
		str a;
		memset(a.num, 0, sizeof(a.num));
		for(int i = 0; i < n; ++ i){
			int y;
			scanf("%d", &y);
			a.num[y%x] ++;
		}
		a = fast_pow(a, b);
		printf("%I64d\n", a.num[k]);
	}
	return 0;
}
