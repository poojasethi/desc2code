#include<cstdio>
using namespace std;
long long P=1e9+7,cnt[10],f[100][100],g[100],h[100];
int n,b,kk,x,i,j,k,temp,now,t,y;
int main()
{
	scanf("%d%d%d%d",&n,&b,&kk,&x);
	for (i=1;i<=n;i++) 
	{
		scanf("%d",&y);
		cnt[y]++;
    }
    for (i=1;i<=9;i++) f[0][i%x]+=cnt[i];
    for (i=0;i<x;i++) f[0][i]%=P;
    for (i=1;i<=31;i++)
    {
    	if (i==1) temp=10%x;
          else temp=temp*temp%x;
      for (j=0;j<x;j++) 
        for (k=0;k<x;k++)
        {  
          now=(j*temp+k)%x;
          f[i][now]+=f[i-1][j]*f[i-1][k]%P;
          f[i][now]%=P;
        }
   }
   g[0]=1;
   for (i=0;i<31;i++)
   {
   	 //if (i==0) temp=1;
   	 if (i==0) temp=10%x;
     if (i>0) temp=temp*temp%x;
     //printf("%d %")
     if (b&(1<<i))
     {
     	for (j=0;j<x;j++)
     	  for (k=0;k<x;k++)
     	  {
     	  	now=(j*temp+k)%x;
     	    h[now]=(h[now]+g[j]*f[i][k]%P)%P;
          }
     	for (j=0;j<x;j++){g[j]=h[j];h[j]=0;}
	 }
    }
    printf("%I64d\n",g[kk]);
    return 0;
}
