#include<cstdio>
#include<cstring>
using namespace std;
int n,m,k,x;
int x1,x2,x3;
const long long mod=1000000007;
struct matrix
{
	long long a[300][300];
	inline void multi(matrix b)
	{
		long long c[300][300];
		memset(c,0,sizeof(c));
		for (int i=1;i<=n;i++)
		{
			for (int j=1;j<=n;j++)
			{
				for (int k=1;k<=n;k++)
				{
					c[i][j]+=a[i][k]*b.a[k][j]%mod;
					c[i][j]%=mod;
				}
			}
		}
		for (int i=1;i<=n;i++)
		{
			for (int j=1;j<=n;j++)
			{
				a[i][j]=c[i][j];
			}
		}
	}
	inline void print()
	{
	}
}mat0,mat1;
inline void fastexp(int x)
{
	if(x==1)return;
	fastexp(x/2);
	mat0.multi(mat0);
	if(x%2==1)mat0.multi(mat1);
}
int main()
{
	scanf("%d %d %d %d",&m,&k,&x,&n);
	while (m--)
	{
		scanf("%d",&x1);
		for (int i=0;i<n;i++)
		{
			x2=(i*10+x1)%n;
			mat0.a[i+1][x2+1]++;
		//	printf("%d %d\n",i,x2);
		}
	}
	x++;
	mat1=mat0;
	fastexp(k);
	printf("%d\n",(int)mat0.a[1][x]);
	return 0;
}
