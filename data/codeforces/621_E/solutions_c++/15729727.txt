#include <bits/stdc++.h>
using namespace std;
const int mod=1e9+7;
typedef long long ll;
int n,b,x,k;
struct mat {
    int m[105][105];
    mat() {memset(m,0,sizeof(m));}
    mat operator * (const mat & rhs) const {
        mat ans;
        for(int i=0;i<x;i++)
        for(int j=0;j<x;j++)
            for(int k=0;k<x;k++) ans.m[i][j]=(ans.m[i][j]+(ll)m[i][k]*rhs.m[k][j]%mod)%mod;
        return ans;
    }
}A;
mat pow(mat a,int p) {
    mat res;
    for(int i=0;i<x;i++) res.m[i][i]=1;
    while(p) {
        if(p&1) res=res*a;
        a=a*a;
        p>>=1;
    }
    return res;
}
int main() {
    scanf("%d%d%d%d",&n,&b,&k,&x);
    int d;
    for(int i=0;i<n;i++) {
        scanf("%d",&d);
        for(int j=0;j<x;j++) A.m[j][(j*10+d)%x]++;
    }
    A=pow(A,b);
    printf("%d\n",A.m[0][k]);
    return 0;
}
