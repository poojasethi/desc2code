#include <iostream>
#include <cstring>
#include <cstdlib>
#include <string>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <ctime>
#define Mod 1000000007
using namespace std;
int n,b,k,x,p,a;
struct Matrix {
	long long val[101][101];
	Matrix () {memset(val,0,sizeof(val));}
	friend Matrix operator *(Matrix x,Matrix y) {
		Matrix Ans;
		for(int i = 0;i < 100;i ++)
			for(int j = 0;j < 100;j ++)
				for(int k = 0;k < 100;k ++)
				{
					Ans.val[i][j] += x.val[i][k] * y.val[k][j];
					Ans.val[i][j] %= Mod;
				}
		return Ans;
	} 
	friend Matrix operator^(Matrix X,int Y) {
		Matrix Ans;
		for(int I=0;I<=99;I++)
		    Ans.val[I][I]=1;
		while(Y>0) 
		{
			if(Y%2==1) Ans=Ans*X;
			X=X*X;
			Y=Y/2;
		}
		return Ans;
	}
}A,B;
int main() {
	scanf("%d%d%d%d",&n,&b,&k,&p);
	for(int i = 1;i <= n;i ++)
	{
		scanf("%d",&a);
		for(int j = 0;j < p;j ++)
		{
			B.val[j][(j * 10 + a) % p] ++;
		}
	}
	A.val[0][0] = 1;
	A = A * (B ^ b);
	printf("%I64d",A.val[0][k]);
	return 0;
}
