#include <iostream>
#include <cstdio>
using namespace std;
const int MOD=1e9+7;
long long dp[108][108];
long long num[108][108];
int n,b,k,x;
int cnt[10];

void _mul(long long aa[][108],long long  bb[][108]){
    long long tmp[108][108];
    for(int i=0;i<x;i++){
        for(int j=0;j<x;j++){
            tmp[i][j]=0;
            for(int k=0;k<x;k++){
                tmp[i][j]=(tmp[i][j]%MOD + (aa[i][k]%MOD) * (bb[k][j]%MOD))%MOD;
            }
        }
    }
    for(int i=0;i<x;i++){
        for(int j=0;j<x;j++){
            aa[i][j]=tmp[i][j];
        }
    }
}

int main(){
    scanf("%d %d %d %d",&n,&b,&k,&x);
    int tmp;
    for(int i=1;i<=n;i++){
        scanf("%d",&tmp);
        cnt[tmp]++;
    }
    dp[0][0]=1;
    for(int i=0;i<x;i++){
        for(int k=1;k<=9;k++){
            int j=(i*10+k)%x;
            num[i][j]+=cnt[k];
        }
    }
    while(b){
        if(b&1) _mul(dp,num);
        _mul(num,num);
        b>>=1;
    }
    cout<<dp[0][k]<<endl;
    return 0;
}
