#include<bits/stdc++.h>
using namespace std;
#define ULL unsigned long long
#define LL long long
#define INF 0x3f3f3f3f
#define rep(i,n) for(int i = 0; i < n; ++i)
#define Rep(i,n) for(int i = 1; i <= n; ++i)

const int hehe = 1000000007;

LL a[105][105], res[105][105], temp[105][105];
int cnt[105], mod;

void mul(LL x[][105], LL y[][105]) {
	rep(i, mod) rep(j, mod) temp[i][j] = 0;
	rep(i, mod) {
		rep(k, mod) {
			rep(j, mod) temp[i][j] += x[i][k] * y[k][j] % hehe;
		}
	}
	rep(i, mod) rep(j, mod) x[i][j] = temp[i][j] % hehe;
}

void qpow(int m) {
	rep(i, mod) rep(j, mod) res[i][j] = i == j;
	while(m) {
		if(m & 1) mul(res, a);
		m >>= 1;
		mul(a, a);
	}
}

int main()
{
	int n, m, xxx;
	cin >> n >> m >> xxx >> mod;
	int x;
	rep(i, n)
		cin >> x, ++cnt[x%mod];
	rep(j, mod) rep(i, mod) rep(k, min(10, mod)) {
		if((i*10+k)%mod == j) a[i][j] += cnt[k];
	}
	qpow(m);
	cout << res[0][xxx] << endl;
	return 0;
}
