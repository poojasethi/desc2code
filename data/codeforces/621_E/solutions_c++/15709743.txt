int x,y,z,i,j,k,n,m,a,b,c;

class matris{public:

	int a[105][105];

	matris(){
		memset(a,0,sizeof a);
	}

	matris operator*(matris b){
		matris c;

		for(int i=0 ; i<x ; i++)
			for(int j=0 ; j<x ; j++)
				for(int k=0 ; k<x ; k++)
					c[i][j] = (c[i][j]+(a[i][k]*b[k][j]%mod))%mod;
		return c;
	}
	int* operator[](int x){
		return a[x];
	}
};

matris t;

matris fe(matris x,int y){
	if(y==1)
		return x;

	matris t = fe(x,y/2);

	t = t*t;

	if(y&1)
		return t*x;
	return t;
}

main(){
	cin >> n >> b >> k >> x;

		scanf("%lld",&a);
		for(j=0 ; j<x ; j++)
			t[j][(j*10+a)%x]++;
	}

	t = fe(t,b);

}
