#include<iostream>
#include<stdio.h>
#include<vector>
#include<cstring>
using namespace std;
typedef long long ll;
typedef vector<ll> vec;
typedef vector<vec> mat;
ll cnt[150];
const ll m=1000000007;
mat mul(mat &a,mat&b){
	mat c(a.size(),vec(b[0].size()));
	for(int i=0;i<a.size();++i){
		for(int k=0;k<b.size();++k){
			for(int j=0;j<b.size();++j){
				c[i][j]=(c[i][j]+a[i][k]*b[k][j])%m;
			}
		}
	}
	return c;
}
mat pow(mat a,ll n){
	mat b(a.size(),vec(a.size()));
	for(int i=0;i<a.size();++i){
		b[i][i]=1;
	}
	while(n>0){
		if(n&1) b=mul(b,a);
		a=mul(a,a);
		n>>=1;
	}
	return b;
}

int main(){
	memset(cnt,0,sizeof(cnt));
	ll n,b,k,x;
	ll y;
	scanf("%I64d%I64d%I64d%I64d",&n,&b,&k,&x);
	for(int i=0;i<n;++i){
		scanf("%d",&y);
		++cnt[y%x];
	}
	mat a(x,vec(x));
	for(int i=0;i<x;++i){
		for(int j=0;j<x;++j){
			a[i][j]=cnt[(1000*x+i-10*j)%x];
		}
	}
	a=pow(a,b);
	printf("%I64d\n",a[k][0]);
}
