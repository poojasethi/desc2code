#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
typedef vector<vector<int> > mat;
typedef long long ll;
int n, b, k, x, f[10];

mat mul(const mat &a, const mat &b) {
	mat c(a.size(), vector<int>(a.size(), 0));
	for(int i=0 ; i<(int)a.size() ; ++i)
		for(int j=0 ; j<(int)a.size() ; ++j)
			for(int k=0 ; k<(int)a.size() ; ++k)
				c[i][j] = (c[i][j] + (ll)a[i][k] * b[k][j])%mod;
	return c;
}

mat power(mat &a, int p){
	if(p == 1) return a;
	mat z = power(a, p/2);
	z = mul(z, z);
	if(p & 1) z = mul(z, a);
	return z;
}

int main(int argc, char **argv) {
	scanf("%d%d%d%d", &n, &b, &k, &x);
	for(int u, i=0 ; i<n ; ++i){
		scanf("%d", &u);
		++f[u];
	}
	mat a(x, vector<int>(x, 0));
	for(int i=0 ; i<x ; ++i)
		for(int j=1 ; j<10 ; ++j)
			a[i][(i*10+j)%x] += f[j];

	a = power(a, b);
	cout<<a[0][k]<<endl;
	return 0;
}
