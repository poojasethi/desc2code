int n,b,l,x;
long long f[10005][105],P=1e9+7;
struct mat
{	long long a[105][105];
}A,O,Ans;
mat Times(mat X,mat Y)
{	int i,j,k;
	mat Z;
	Z=O;
	for (i=0;i<x;i++)
		for (j=0;j<x;j++)
			for (k=0;k<x;k++)
			{
				Z.a[i][j]+=X.a[i][k]*Y.a[k][j];
				Z.a[i][j]%=P;
			}
	return Z;				
}

		}
	}
	for (i=0;i<x;i++) Ans.a[i][i]=1;
	while(b)
	{
		if (b%2==1) Ans=Times(Ans,A);
