#include <bits/stdc++.h>
using namespace std;
#define ll long long
vector<ll> bas;
const int mod = 1e9+7;
int n,k,b,x;
int po(int a,int b){
	if(b==0)return 1;
	if(b&1)
		return (po(a,b-1)*a)%x;
	int r = po(a,b/2);
	return (r*r)%x;
}
vector<ll> mul(int ind){
	if(ind==1)
		return bas;
	vector<ll> r,a,ret(x,0),R(x,0);
	int i,j;
	ll t;
	if(ind&1){
		r= mul(ind-1);
		a= bas;
		t= 10;
	}
	else{
		r= mul(ind/2);
		a=r;
		t=po(10,ind/2);
	}
	for(i=0;i<x;++i)
		(ret[(i*t)%x]+=r[i])%=mod;
	for(i=0;i<x;++i)
		for(j=0;j<x;++j){
			t = ret[i];t*=a[j];t%=mod;
			(R[(i+j)%x] += t)%=mod;
		}
	return R;
}
int main(){
	cin>>n>>b>>k>>x;
	int i,j,a;
	bas=vector<ll>(x,0);
	for(i=0;i<n;++i)
		scanf("%d",&a),++bas[a%x];
	vector<ll> ret= mul(b);
	cout<<ret[k]<<endl;
}
