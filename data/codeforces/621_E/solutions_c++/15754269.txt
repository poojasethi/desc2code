#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int MOD=1000000007;
int c[10],mat[110][110],ans[110][110],x,temp[110][110],n,b,k,t;
void add(int &x,int val){
	x+=val;
	if (x>=MOD) x-=MOD;
}
void mul(int a[110][110],int b[110][110]){
	int val;
	for (int i = 0 ; i<x; i++)
		for (int j = 0; j<x; j++){
			val=0;
			for (int k = 0; k<x; k++)
				add(val,(long long)(a[i][k])*b[k][j]%MOD);
			temp[i][j]=val;
		}
	for (int i = 0 ; i<x; i++)
		for (int j = 0 ; j<x; j++)
			a[i][j]=temp[i][j];
}
int main(){
	scanf("%d%d%d%d",&n,&b,&k,&x);
	for (int i = 0; i < n; i++){
		scanf("%d",&t);
		c[t]++;
	}
	for (int i = 0 ; i < x; i++)
		for (int j = 0 ; j < 10 ; j++)
			add(mat[i][(10*i + j)%x],c[j]);
	for (int i = 0 ; i < x; i++) ans[i][i]=1;
	for (;b;b/=2){
		if (b&1) mul(ans,mat);
		mul(mat,mat);
	}
	printf("%d\n",ans[0][k]);

	return 0;
}
