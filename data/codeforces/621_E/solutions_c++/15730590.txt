#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<queue>
#include<vector>
using namespace std;

#define mo 1000000007
int a[110][110],v[50010],f[110][110],n,m,b,k;

void mul(int a[][110],int b[][110]){
	int tmp[110][110];
	memset(tmp,0,sizeof(tmp));
	for(int i=0;i<m;i++)
	 for(int j=0;j<m;j++)
	  for(int l=0;l<m;l++)
	  tmp[i][j]=(tmp[i][j]+(long long)a[i][l]*b[l][j]%mo)%mo;
	memcpy(a,tmp,sizeof(tmp));
}

void power(int a[][110],int n){
	if(n==1) return;
	power(a,n/2);
	if(n&1) mul(a,a),mul(a,f);
	else mul(a,a);
}

int main(){
	scanf("%d%d%d%d",&n,&b,&k,&m);
	for(int i=1;i<=n;i++) scanf("%d",&v[i]);
	for(int i=0;i<m;i++)
	 for(int j=1;j<=n;j++) f[(i*10+v[j])%m][i]++;
	memcpy(a,f,sizeof(f));
	power(a,b);
	printf("%d\n",a[k][0]);
	return 0;
}
