#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<string>
#include<cstring>
using namespace std;
#define LL long long
LL cnt[10];
LL d[103][103];
LL c[103][103];
const LL MOD=1e9+7;
LL n,b,k,m,a;
void multiply(LL a[103][103], LL d[103][103]) {
	for (int i = 0; i < m; ++i)
		for (int j = 0; j < m; ++j){
			c[i][j] = 0;
			for (int k = 0; k < m; ++k)
				c[i][j] = (c[i][j] + a[i][k] * d[k][j]) % MOD;
		}
	for(int i=0;i<m;i++)
		for(int j=0;j<m;j++)
			a[i][j] = c[i][j];
}
LL tmp[103][103];
int main(){
	cin>>n>>b>>k>>m;
	for(int i=1;i<=n;i++){
		scanf("%d",&a);
		cnt[a]++;
	}
	for(int t=0;t<m;t++){
		for(int i=0;i<=9;i++){
			d[t][(t*10+i)%m]+=cnt[i];
		}
	}
	memset(tmp,0,sizeof tmp);
	for(int i=0;i<m;i++)tmp[i][i]=1;
	while(b){
		if(b&1){
			multiply(tmp,d);
		}
		multiply(d,d);
		b>>=1;
	}
	cout<<tmp[0][k]<<endl;
	
}
