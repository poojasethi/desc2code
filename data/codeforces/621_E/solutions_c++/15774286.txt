#include<bits/stdc++.h>
#define MOD 1000000007
#define LL long long
using namespace std;

struct in{
	LL num[100];
};

int n,b,k,x;
in a;

in mul(in& a,in& b,int B){
	in c;
	memset(c.num,0,sizeof(c.num));
	for(int i=0;i<x;++i){
		for(int j=0;j<x;++j){
			c.num[(i*B+j)%x] = (c.num[(i*B+j)%x] + a.num[i]*b.num[j] % MOD) % MOD;
		}
	}
	//puts("mul");
	//for(int i=0;i<x;i++) cout<<c.num[i]<<" ";
	//puts("");
	return c;
}

in fpow(in a,int t){
	in c;
	memset(c.num,0,sizeof(c.num));
	c.num[0]=1;
	int B=10;
	while(t){
		if(t&1){
			c = mul( c , a , B );
		}
		t>>=1;
		a = mul( a , a , B );
		B=B*B%x;
	}
	return c;
}
