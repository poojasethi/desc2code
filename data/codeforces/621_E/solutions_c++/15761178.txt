#include <bits/stdc++.h>
#define INF 1e9
#define fi first
#define se second
using namespace std;
typedef long long LL;
typedef pair<int,int> pii;
typedef vector<LL> vec;
typedef vector<vec> mat;
const int mod = 1e9+7;
int n,b,k,x;
int block[50005];
mat mul(mat &A, mat &B){
    mat C(x,vec(x,0));
    for(int i=0;i<x;i++){
        for(int j=0;j<x;j++){
            for(int k=0;k<x;k++)
                C[i][j] = (C[i][j] + A[i][k]*B[k][j])%mod;
        }
    }
    return C;
}
mat pw(mat A, int p){
    mat B(x, vec(x,0));
    for(int i=0;i<x;i++) B[i][i] = 1;
    while(p>0){
        if(p & 1LL)
            B = mul(B, A);
        A = mul(A, A);
        p >>= 1;
    }
    return B;
}



int main(){
    cin>>n>>b>>k>>x;
    for(int i=0;i<n;i++)
        scanf("%d", &block[i]);
    mat trans(x, vec(x,0));
    for(int i=0;i<x;i++)
        for(int j=0;j<n;j++)
            trans[i][(i*10+block[j])%x]++;
    trans = pw(trans, b);
    cout << trans[0][k] << endl;
    return 0;
}
