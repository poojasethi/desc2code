//In The Name Of God

#include<bits/stdc++.h>

#define int long long

using namespace std;

const int Maxn=100007;
const int Mod=1e9+7;

int n,b,k,x;
int dp[60][105];
int a[Maxn];
int P[102];
int ans[60][105];

int power(int m)
{
  if(m==0)
    return 1%x;
  int t=power(m/2);
  t*=t;
  if(m&1)
    t*=10;
  t%=x;
  return t;
}

void solve(int m)
{
  int r=0,rr=0;
  ans[0][0]=1;
  for(int i=0;i<60;i++)
    {
      if(m&(1ll<<i))
	{
	  r++;
	  int L=power(rr);
	  for(int j=0;j<x;j++)
	    for(int p=0;p<x;p++)
	      {
		int c=(j*L+p)%x;
		(ans[r][c]+=(ans[r-1][p]*dp[i][j]))%=Mod;
	      }
	  rr+=(1ll<<i);
	}
    }
  cout<<ans[r][k]<<endl;
}

main()
{
  ios_base::sync_with_stdio(0);cin.tie(0);
  cin>>n>>b>>k>>x;
  for(int i=0;i<n;i++)
    cin>>a[i],dp[0][(a[i]%x)]++;
  for(int i=0;i<60;i++)
    P[i]=power(1ll<<i);
  for(int i=1;i<60;i++)
    for(int j=0;j<x;j++)
      for(int r=0;r<x;r++)
	{
	  int p=(j*(P[i-1])+r)%x;
	  (dp[i][p]+=(dp[i-1][j]*dp[i-1][r]))%=Mod;
	}
  solve(b);
  return 0;
}
