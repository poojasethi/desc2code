#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
typedef long long ll;
typedef vector<vector<int> > mat;

mat mult(const mat &x, const mat &y) {
	const int n = x.size();
	mat ans(n, vector<int>(n, 0));
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < n; ++j)
			for (int k = 0; k < n; ++k)
				ans[i][j] = (ans[i][j] + (ll) x[i][k] * y[k][j]) % MOD;
	return ans;
}

mat power(const mat &a, int p) {
	if (p == 1)
		return a;
	mat u = power(a, p / 2);
	u = mult(u, u);
	if (p & 1)
		u = mult(u, a);
	return u;
}


int f[10];
const int N = 100;
mat ans(N, vector < int > (N, 0));

int main() {
	int n, b, k, x, a;
	scanf("%d%d%d%d", &n, &b, &k, &x);
	++n;
	while(--n) {
		scanf("%d", &a);
		f[a]++;
	}
	for(int i = 0; i < x; ++i)
		for(int j = 0; j < 10; ++j)
			ans[i][(i*10+j)%x] += f[j];
	ans = power(ans, b);
	printf("%d\n", ans[0][k]);
	return 0;
}
