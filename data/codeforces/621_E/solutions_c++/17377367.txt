#include<iostream>
#include<cstdlib>
#include<cstdio>
using namespace std;
#define mod 1000000007
typedef long long ll;
const int mx=(int)1e5+100;
int n,m,kp,mo,end;
struct Mat{
	long long mm[102][102];
	friend Mat operator*(Mat a,Mat b){
		Mat c;
		for(int i=0;i<=end;++i)
		for(int j=0;j<=end;++j){
			c.mm[i][j]=0;
			for(int k=0;k<=end;++k)
			if(a.mm[i][k]&&b.mm[k][j]) c.mm[i][j]+=a.mm[i][k]*b.mm[k][j]%mod;
			if(c.mm[i][j]>=mod) c.mm[i][j]%=mod;
		} 
		return c;
	}
}op,opt;
int tmp[mx];
void build(){
	end=mo-1;
	for(int k=0;k<=end;++k)
	for(int i=1;i<=n;++i)
	op.mm[k][(k*10+tmp[i])%mo]++;
	for(int i=1;i<=n;++i)
	opt.mm[0][tmp[i]]++;
}
void mi(int t){
	while(t){
		if(t&1) opt=opt*op;
		t>>=1;op=op*op;
	}
}
int main(){
//	freopen("tt.in","r",stdin);
//	freopen("tt.out","w",stdout);
	scanf("%d%d%d%d",&n,&m,&kp,&mo);
	for(int i=1;i<=n;++i)
	scanf("%d",&tmp[i]),tmp[i]%=mo;
	build();
	mi(m-1);
	printf("%lld\n",opt.mm[0][kp]);
	return 0;
}
