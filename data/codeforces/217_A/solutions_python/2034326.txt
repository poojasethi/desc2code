#!/usr/bin/env python

N = int(raw_input())
V = []
for i in xrange(N):
    x, y = map(int, raw_input().split())
    V.append((x, y))

H = {i: False for i in xrange(N + 1)}
def dfs(v):
    H[v] = True
    for i in xrange(len(V)):
        xx, yy = V[i][0], V[i][1]
        if not H[i] and (xx == V[v][0] or yy == V[v][1]):
            dfs(i)

ret = 0
for i in xrange(len(V)):
    if not H[i]:
        ret += 1
        dfs(i)
print(ret - 1)
