#include<stdio.h>

int x,n,h[105],s=0,y,arr[105][2];

void f(int x){
	for(int i=0;i<n;i++){
		if(!h[i]&&i!=x&&(arr[i][0]==arr[x][0]||arr[i][1]==arr[x][1])){
			h[i]=1;
			f(i);
		}
	}
}


int main(){
	int i,g=0,j;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d %d",&x,&y);
		arr[i][0]=x,arr[i][1]=y;
	}
	j=0;
	for(i=0;i<n;i++){
		if(!h[i]){
			h[i]=1;
			g++;
			f(i);
		}
	}
	printf("%d",g-1);
	return 0;
}
