#include <iostream>
using namespace std;
int a[101][2],n;
bool u[101];
void dfs(int i)
{
	u[i]=true;
	for (int j=1;j<=n;j++)
		if (!u[j]&&(a[j][0]==a[i][0]||a[j][1]==a[i][1])) dfs(j);
}
int main()
{
	int x,y,c=0;
	cin>>n;
	for (int i=1;i<=n;i++)
		cin>>x>>y,a[i][0]=x,a[i][1]=y,u[i]=false;
	for (int i=1;i<=n;i++)
	{
		if (!u[i])
		{
			c++;
			dfs(i);
		}
	}
	cout<<c-1;
	return 0;
}
