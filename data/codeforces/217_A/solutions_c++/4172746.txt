#include<iostream>
using namespace std;
bool mark[105];
int x[105],y[105];
int n;
void dfs(int p)
{
	mark[p]=true;
	for (int i=1;i<=n;i++)
		if((x[p]==x[i]||y[p]==y[i])&&(!mark[i]))
			dfs(i);
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>x[i]>>y[i];
	int ans=0;
	for(int i=1;i<=n;i++)
		if(!mark[i])	
		{
			dfs(i);
			ans++;
		}
	cout<<ans-1<<endl;
	return 0;
}
