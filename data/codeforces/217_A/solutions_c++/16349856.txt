#include <bits/stdc++.h>
using namespace std;
int cnt=0,n,a[110][3];
void dfs(int i){
	a[i][0]=cnt;
	for(int j=1;j<=n;j++){
		if((a[j][1]==a[i][1]||a[j][2]==a[i][2])&&!a[j][0])dfs(j);
	}
}
int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i][1]>>a[i][2];
	}
	for(int i=1;i<=n;i++){
		if(!a[i][0]){
			cnt++;
			dfs(i);
		}
	}	
	cout<<cnt-1<<endl;
}
