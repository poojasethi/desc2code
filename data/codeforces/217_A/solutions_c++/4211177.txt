#include<iostream>
using namespace std;

int x[1100],y[1100];
bool v[1100];

void dfs(int i, int n){
    v[i]=true;
    for(int j=0;j<n;j++)if(!v[j] && (x[i]==x[j] || y[i]==y[j]))dfs(j,n);
}

int main(){
    int n,ans=0;
    while(cin>>n){
        for(int i=0;i<n;i++)cin>>x[i]>>y[i];
        for(int i=0;i<n;i++)
            if(!v[i]){ dfs(i,n); ans++;}
        cout<<(ans-1)<<endl;
    }
}
