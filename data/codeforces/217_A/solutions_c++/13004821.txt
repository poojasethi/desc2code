#include<cstdio>

int n, ans;
int x[111], y[111], v[111];

void dfs(int cur) {
	v[cur] = true;
	for(int i=0;i<n;i++)
		if (!v[i] && (x[i] == x[cur] || y[i] == y[cur])) dfs(i);
}

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) scanf("%d%d", &x[i], &y[i]);
	for(int i=0;i<n;i++)
		if (!v[i]) {
			dfs(i);
			ans++;
		}
	printf("%d\n", ans-1);
	
	return 0;
}
