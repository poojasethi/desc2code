#include<bits/stdc++.h>
using namespace std;
int x[110],y[110],n;
bool v[110];
void dfs(int a)
{
    v[a]=true;
    for (int i=0;i<n;i++)
        if (!v[i]&&(x[a]==x[i]||y[a]==y[i])) dfs(i);
}
int main()
{
    cin>>n;
    for (int i=0;i<n;i++)
        cin>>x[i]>>y[i];
    long long e=0;
    for (int i=0;i<n;i++)
    if (!v[i])
    {
        dfs(i);
        e++;
    }
    cout<<e-1;
    return 0;
}
