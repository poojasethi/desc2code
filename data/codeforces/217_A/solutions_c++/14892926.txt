#include <cstdio>  
#include <cmath>  
#include <iostream>
using namespace std;
int b[110]; 
int x[110],y[110];
int n,cnt;
void f(int aa)
{
	if(b[aa]==1)return;
	b[aa]=1;
	for(int i=0;i<n;i++)
	{
		if(x[aa]==x[i]||y[aa]==y[i])
		f(i);
	}	
}
int main()  
{  
	cin>>n;
	for(int i=0;i<n;i++)
	cin>>x[i]>>y[i];
	for(int i=0;i<n;i++)	
	{
		if(b[i]==0)
		{
		f(i);
		cnt++;
		}
	}
	cout<<cnt-1;
}
