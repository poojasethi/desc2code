#include <cstdio>
int n, x[105], y[105], a[105], res;
void f(int v) {
	a[v] = 1;
	for (int i = 1; i <= n; i++) if (!a[i] && (x[v] == x[i] || y[i] == y[v])) f(i);
}
int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%d%d", &x[i], &y[i]);
	for (int i = 1; i <= n; i++) if (!a[i]) res++, f(i);
	printf("%d", res - 1);
}
