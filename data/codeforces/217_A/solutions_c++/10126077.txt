#include<stdio.h>
long x[102],y[102],i,j,k,n,a;
bool u[102];

void f(long aa){
 long z;
 u[aa]=true;
 for(z=0;z<n;z++)
    if(!u[z]&&(x[z]==x[aa]||y[z]==y[aa]))
        f(z);
}

int main(){
 scanf("%ld",&n);
 for(i=0;i<n;i++)
    scanf("%ld %ld",&x[i],&y[i]);
 for(i=0;i<n;i++)
  if(!u[i]){
   a++;
   f(i);
  }
 printf("%ld\n",a-1);
}
