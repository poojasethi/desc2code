#include <iostream>
using namespace std;

int x[110],y[110]; bool v[110];
int n,M=0;

void bfs(int i){
	v[i]=true;
	for(int j=0;j<n;j++)if(!v[j] && (x[i]==x[j] || y[i]==y[j]) )bfs(j);
}
int main(){
	cin >> n;
	for(int i=0;i<n;i++)cin>>x[i]>>y[i];
	for(int i=0;i<n;i++)if(!v[i]){
		M++;
		bfs(i);
	}
	cout<<M-1<<endl;
	return 0;
}
