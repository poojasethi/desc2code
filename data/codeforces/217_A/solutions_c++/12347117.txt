#include<bits/stdc++.h>
using namespace std;

int x[101],y[101],v[101],n;

int dfs(int g) {
	if(v[g]) return 0;
	v[g]=1;
	for (int i=0; i<n; i++)
		if(x[g]==x[i]||y[g]==y[i])
			dfs(i);
}

int main() {
	int ret=-1; cin>>n;

	for (int i=0; i<n; i++)
		cin>>x[i]>>y[i], v[i]=0;

	for (int i=0; i<n; i++)
		if(!v[i])
			ret++, dfs(i);
	cout << ret << endl;

	return 0;
}
