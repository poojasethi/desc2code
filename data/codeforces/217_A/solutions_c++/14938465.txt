#include<iostream>
using namespace std;
int x[101],y[101],vis[101] = {0};
int n;
void dfs(int u){
vis[u] = 1;
for(int i = 1 ; i<=n; i++){
if(vis[i] == 0 && u!=i && (x[i]== x[u]||y[i] == y[u]))dfs(i);
}

}
int main(){
int ans = -1;
    cin>>n;
    for(int i = 1; i<=n; i++)
    cin>>x[i]>>y[i];
    for(int i = 1;i<=n; i++){
    if(vis[i] == 0){
    dfs(i);
    ans++;}
    
    }
cout<<ans;


}
