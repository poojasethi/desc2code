#include<iostream>
using namespace std;
int x[101],y[101],count,n;
bool v[101];

void dfs(int i)
 {
 v[i]=true;
 for(int j=0;j<n;j++)
  {
  if(v[j]==false && (x[i]==x[j] || y[i]==y[j]))
    dfs(j);
  }

 }
int main()
{
cin>>n;
for(int i=0;i<n;i++)
 cin>>x[i]>>y[i];
for(int i=0;i<n;i++)
{
if(v[i]==false)
 {
 count++;
 dfs(i);
 }
}
cout<<count-1;
}
