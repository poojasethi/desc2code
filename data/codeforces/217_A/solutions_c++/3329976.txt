#include<iostream>
using namespace std;

int n,s,a[200],b[200];
bool u[200];

void dfs(int i)
{
    u[i]=true;
    for(int j=0;j<n;j++)
    if(u[j]==0 && (a[i]==a[j] || b[i]==b[j])) dfs(j);
}

int main()
{
    cin>>n;
    
    for(int i=0;i<n;i++)
    cin>>a[i]>>b[i];
    
    for(int i=0;i<n;i++)
    if(!u[i]) {s++;dfs(i);}
    
    cout<<s-1;
    return 0;
}
