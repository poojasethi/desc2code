#include <iostream>
using namespace std;
int n,a[3][105],sum;
bool vis[105];
void dfs (int s)
{
	vis[s]=true;
	for (int i=1;i<=n;i++)
		if (!vis[i] && (a[1][i]==a[1][s] || a[2][i]==a[2][s])) dfs(i);
}

int main ()
{
	cin>>n;
	for (int i=1;i<=n;i++)
		cin>>a[1][i]>>a[2][i];
	sum=-1;
	for (int i=1;i<=n;i++)
		if (!vis[i]) {dfs(i);sum++;}
	cout<<sum;
	return 0;
}
