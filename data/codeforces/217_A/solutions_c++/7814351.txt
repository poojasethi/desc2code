#include <iostream> 
using namespace std;
long x[102],y[102],n,cc=0;
bool r[102];

void dfs(int p)
{
	r[p]=false;
	for(int i=1; i<=n; i++)
	{
    	if(r[i]&&(x[i]==x[p]||y[i]==y[p]))
        	dfs(i);
	}
}

int main()
{
	cin>>n;
	for (int i=1; i<=n; i++)
	{
		cin>>x[i]>>y[i];
		r[i]=true;
	}
	for(int i=1;i<=n;i++)
	{
		if(r[i])
		{
			cc++;
			dfs(i);
		}
	}
	cout<<cc-1;
}
