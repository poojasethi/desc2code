#include <bits/stdc++.h>
using namespace std;
int n,k;
int x[1000], y[1000], viz[1000]={0};
void DFS(int v) {
	viz[v]=1;
	k=1;
	for (int i=1; i<=n; i++)
	if(viz[i]==0 && v!=i && (x[i]== x[v]||y[i] == y[v])) 
	 DFS(i);
}
int main () {
	int cnt=-1;
	cin>>n;
	for (int i=1; i<=n; i++)
	cin>>x[i]>>y[i];
	for (int i=1; i<=n; i++) 
	if (viz[i]!=1) {
	DFS(i);
	cnt++;
}
	cout<<cnt;
	return 0;
}
