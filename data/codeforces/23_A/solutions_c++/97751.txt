#include <iostream>
using namespace std;

int main()
{
  string s;
  cin >> s;

  int best = 0;

  for(int l=1 ; l<=s.size() ; l++)
    for(int i=0 ; i<s.size()-l+1 ; i++)
    {
      string sub = s.substr(i, l);
      if(s.find(sub, i+1) != string::npos)
        best = max(best, l);
    }

  cout << best << endl;
}
