#include<stdio.h>
#include<string.h>
int main()
{
	int l,i,j,x,k;
	char a[10001];
	while(gets(a))
	{

		l=strlen(a);
		x=0;
		for(i=0;i<l;i++)
		{
			for(j=i+1;j<l;j++)
			{
				k=0;
				while(a[i+k]==a[j+k]) k++;
				x=k>x?k:x;
			}
		}
		printf("%d\n",x);

	}
	return 0;
}
