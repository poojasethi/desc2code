#include <stdio.h>
#include <string.h>


int main()
{
	char str[105];
	int i,j;
	int l,t,maxl;
	scanf("%s",str); 
		maxl=0;
		l=strlen(str);
		for(i=0;i<l;i++)
		{
			for(j=i+1;j<l;j++)
			{
				t=0;
				while(str[i+t]==str[j+t]) t++;
				maxl=t>maxl?t:maxl;
			}
        }
		printf("%d\n",maxl);
	return 0;
}
