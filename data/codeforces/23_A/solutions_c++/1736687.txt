#include<iostream>
#include<string>
using namespace std;
int main()
{
	int i,j,sum=0;
	string s;
	while(cin>>s){
		for(i=0;i<s.size();i++)
			for(j=1;j<=s.size();j++){
				if(s.rfind(s.substr(i,j))!=i) if(j>sum)sum=j;
			}
			cout<<sum<<endl;
			sum=0;
			s.clear();
	}
	return 0;
}
