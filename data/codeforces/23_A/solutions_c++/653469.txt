#include <iostream>
using namespace std;
string s;
int MAX = 0, k = 0;
int main(){
    cin >> s;
    for(int i=0; s[i]; i++)
        for(int j=i+1; s[j]; j++)
        {
            k = 0;
            while(s[i + k] == s[j + k])     k++;
            MAX = max(MAX, k);
        }
    cout << MAX << endl; 
    return 0;
}
