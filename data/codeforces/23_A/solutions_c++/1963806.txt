
#include <iostream>
#include<string>
#include<vector>
using namespace std;

int main()
{ string s;int m=0;
cin>>s;
for(int i=0;i<s.size();i++)
{
   
   for(int j=i+1;j<s.size();j++)
{
    int su=0,l=0;
while(s[i+l]==s[j+l]&&(j+l)<s.size()&&(i+l)<s.size()) {l++;su++;}
if(su>m) m=su;
}

}
cout<<m;

    return 0;
}
