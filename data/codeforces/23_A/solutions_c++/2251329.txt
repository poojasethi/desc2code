#include <iostream>
using namespace std;

int main(){
   string inp;
   int len;
   cin>>inp;
   len=inp.length();
   int i,j,k,maxVal=0;
   for(i=0; i<len; i++){
      for(j=i+1; j<len; j++){
         k=0;
         while(inp[i+k]==inp[j+k])k++;
         maxVal=max(maxVal,k);
      }
   }
   cout<<maxVal<<endl;
}
