#include <cstdio>
#include <iostream>
#include <string>

using namespace std;

int main(){
	string s;
	cin >> s;
	int i,j,k;
	#define sz(c) (int)(c).size()
	int n=sz(s);
	int res=0;
	for(i=0;i<n;i++){
		for(j=i+1;j<n;j++){
			k=0;
			while(j+k<n&&s[i+k]==s[j+k]) k++;
			res=max(res,k);
		}
	}
	cout << res << endl;
	return 0;
}
