# include<stdio.h>

int main()
{
	int i,j,k,max;
	char s[101];
	while(scanf("%s",s)!=EOF)
	{
		max = 0;
		for(i=0;s[i]!='\0';i++)
		{
			for(j=i+1;s[j]!='\0';j++)
			{
				k = 0;
				while(s[i+k] == s[j+k]) k++;
				max = k>max?k:max;
			}
		}
		printf("%d\n",max);
	}
	return 0;
}
