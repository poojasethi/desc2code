#include<cstdio>
#include<cstring>
char l[101];
int ans=0,L;
int main(){
gets(l);
L=strlen(l);
for(int i =0;i<L;i++){
for(int j=0;j+i<L;j++){
bool flg=false;
for(int k=0;k<L;k++){
if(k==i)continue;
flg=true;
for(int p=0;p<=j&&flg;++p)
if(l[k+p]!=l[i+p])flg=false;
if(flg&&j+1>ans)ans=j+1;
}
}
}
printf("%d\n",ans);
}
