#include <iostream>
#include <string>
#include <vector>
#include <map>
using namespace std;

int main()
{
	string s;
	getline(cin, s);

	int maxn = 0;
	map<string, int> m;
	for(int i=0; i<s.size(); ++i)
		for(int n=0; i+n<=s.size(); ++n)
			if( ++m[s.substr(i,n)] >= 2 )
				maxn = max(maxn, n);
	cout << maxn << endl;
}
