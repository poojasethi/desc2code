#include<bits/stdc++.h>

using namespace std;

int main()
{
    string s;
    cin>>s;
	int n = s.length();
	int tmp,ans=0;
	for (int i=0; i<n; i++)
	{
		for (int j=i+1; j<n; j++)
		{
			tmp=0;
			while (s[i+tmp]==s[j+tmp])
				tmp++;
			if (tmp>ans)
				ans=tmp;
		}
	}
	cout<<ans<<"\n";
	return 0;
}
