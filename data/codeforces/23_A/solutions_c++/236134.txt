#include<iostream>
#include<string>

using namespace std;

int main()
{
    string s;
    int i,j,z;
	
	cin>>s;

	for(i=s.length()-1;i>0;i--)
		for(j=0;j<s.length()-i;j++)
			for(z=j+1;z<s.length()-i+1;z++)
				if(s.substr(j,i)==s.substr(z,i))
				{
					cout<<i;
					return 0;
				}

	cout<<'0';

	return 0;
}
