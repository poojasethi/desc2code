#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
	string in;
	int max=0;
	cin >> in;
	for(int i=0;i<in.size();i++)
	{
		for(int j=1;j+i<=in.size();j++)
		{
			if( in.rfind(in.substr(i,j))!=i)	if(max<j)	max=j;
		}
	}
	cout<<max<<endl;
	return 0;
}
