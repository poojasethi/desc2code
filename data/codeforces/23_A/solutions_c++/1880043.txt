#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char s[110];
    int len,i,j,k,an=0;
    cin>>s;
    len=strlen(s);
    for(i=0;i<len;i++)
        for(j=i+1;j<len;j++){
            for(k=0;j+k<len&&s[i+k]==s[j+k];k++);
            if(an<k)an=k;
        }
    cout<<an;
    return 0;
}
