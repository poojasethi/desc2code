#include <iostream>

using namespace std;

int32_t main()
{
	string s;
	cin >> s;
	int max_cnt = 0;
	for (int i = 0; i < s.size(); i++)
		for (int j = i + 1; j < s.size(); j++)
		{
			int cnt = 0;
			while (s[i + cnt] == s[j + cnt] && j + cnt < s.size())
				cnt++;
			max_cnt = max(max_cnt, cnt);
		}
	cout << max_cnt << endl;
	return 0;
}
