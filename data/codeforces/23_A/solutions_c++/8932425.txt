#include<stdio.h>
int main()
{
	char x[105];
	while(gets(x))
	{
		int i,j,k,it,jt,ans=0;
		for(i=0;x[i];i++)
		{
			for(j=i+1;x[j];j++)
			{
				it=i,jt=j,k=0;
				while(x[it]==x[jt])
				{
					it++,jt++;
					k++;
				}
				if(k>ans)
					ans=k;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
