#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

def solve(s):
    n = len(s)
    for L in range(n-1,0,-1):
        for i in range(n-L+1):
            ss = s[i:i+L]
            if s.find(ss,i+1) != -1:
                return L
    return 0

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

s = input.readline().strip()

a = solve(s)
output.write('%s\n' % a)
