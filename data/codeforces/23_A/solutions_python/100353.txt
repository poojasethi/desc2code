s = raw_input()
n = len(s)
max = 0
for i in range(n):
	for len in range(1, n-i+1):
		#print "%d, %d: %s" % (i, len, s[i:i+len])
		cnt = 0
		for j in range(n-len+1):
			#print "\t%d, %s" % (j, s[j:j+len])
			if s[i:i+len] == s[j:j+len]:
				cnt += 1
		#print "cnt = %d" % cnt
		if cnt > 1 and len > max:
			max = len
print max
