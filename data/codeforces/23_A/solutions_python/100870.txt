def kmp(string):
    k = [0]*len(string)
    for i in xrange(1,len(string)):
        p = k[i-1]
        while string[i]!=string[p] and p: p = k[p-1]
        if p:k[i]=p+1
        elif string[i]==string[0]:k[i]=1
    return k
string = raw_input()
print max(max(kmp(string[i:])) for i in xrange(len(string)))
