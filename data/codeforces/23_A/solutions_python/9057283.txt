def main_26():
	line = raw_input() ######################
	sub = []
	n = len(line)
	for i in range(n):
		for j in range(n):
			if i <= j:
				sub.append(line[i:j+1])
	sub.sort()
	max_len = 0
	sub_len = 0
	l = len(sub)
	if l > 1:
		for i in range(l-1):
			if sub[i] == sub[i+1]:
				sub_len = len(sub[i])
				if sub_len > max_len:
					max_len = sub_len
	print(max_len)

main_26()
