#!/usr/bin/python

def failure_function(a):
	n = len(a)
	t = 0
	f = [0 for i in range(n)]
	for s in range(1, n-1):
		while t > 0 and a[s+1] != a[t+1]:
			t = f[t]
		if a[s+1] == a[t+1]:
			t += 1
			f[s+1] = t
		else:
			f[s+1] = 0
	return f

s = '\0'
s += raw_input()
result = 0
while len(s) != 0:
	f = failure_function(s)
	maxLen = max(f)
	result = max(maxLen, result)
	s = s[1:]
print result
