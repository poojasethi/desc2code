#! /usr/bin/python

from sys import stdin

def contains(text,length):
	for i in xrange(0,len(text)-length):
		substr=text[i:i+length]
		if substr in text[i+1:]:
			return True
	return False
	
if __name__=='__main__':
	text=stdin.readline().strip()
	answer=0
	for length in xrange(1,len(text)):
		if contains(text,length):
			answer=max(answer,length)
	print answer
