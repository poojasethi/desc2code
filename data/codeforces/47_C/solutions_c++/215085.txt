#include <string>
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
template<class T> inline int size(T& t) {return (int)t.size();}
#define all(X)		(X).begin(),(X).end()
typedef vector<string>	VS;
typedef vector<int>		VI;
int main() { int i,j; VS vs(6); VI sz(6); for(i=0;i<6;i++) cin>>vs[i]; sort(all(vs)); vector <VS> ret;
do { for(i=0;i<6;i++) sz[i]=size(vs[i]);
if( sz[0]+sz[5]-1 != sz[3] || sz[1]+sz[4]-1 != sz[2] || vs[2][sz[1]-1] != vs[3][sz[0]-1] ||
	vs[0][0] != vs[1][0] || vs[0][sz[0]-1] != vs[2][0] || vs[1][sz[1]-1] != vs[3][0] ||
	vs[3][sz[3]-1] != vs[4][0] || vs[2][sz[2]-1] != vs[5][0] || vs[4][sz[4]-1] != vs[5][sz[5]-1] ) continue;

string st=""; for(i=0;i<sz[3];i++) st+='.';
VS res(sz[2],st);

for(i=0;i<sz[0];i++) res[0][i]=vs[0][i];
for(i=0;i<sz[1];i++) res[i][0]=vs[1][i];
for(j=sz[0]-1,i=0;i<sz[2];i++) res[i][j]=vs[2][i];
for(j=sz[1]-1,i=0;i<sz[3];i++) res[j][i]=vs[3][i];
for(j=sz[1]-1,i=0;i<sz[4];i++,j++) res[j][sz[3]-1]=vs[4][i];
for(j=sz[0]-1,i=0;i<sz[5];i++,j++) res[sz[2]-1][j]=vs[5][i];
		
//for(i=0;i<size(res);i++) cout<<res[i]<<endl; return 0;
ret.push_back(res);
} while( next_permutation(all(vs)) );

if(!size(ret)){cout<<"Impossible"; return 0;}
sort(all(ret)); for(i=0;i<size(ret[0]);i++) cout<<ret[0][i]<<endl; return 0;
}
