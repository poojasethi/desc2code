#include <iostream>
#include <algorithm>
#define SIZE(i)  s[m[i]].size()
#define BEGIN(i) s[m[i]][0]
#define END(i)   s[m[i]][SIZE(i)-1]
using namespace std;
int main()
{
	string s[6], ans1, ans2="z";
	int i, j, m[6]={0,1,2,3,4,5};
	for (i=0; i<6; i++) cin>>s[i];
	do {
		if (SIZE(3)+1 == SIZE(0)+SIZE(5) &&
			SIZE(2)+1 == SIZE(1)+SIZE(4) &&
			BEGIN(0) == BEGIN(1) &&
			END(0) == BEGIN(2) &&
			END(2) == BEGIN(5) &&
			END(5) == END(4) &&
			END(1) == BEGIN(3) &&
			END(3) == BEGIN(4) &&
			s[m[3]][SIZE(0)-1] == s[m[2]][SIZE(1)-1]) {
			char ans[SIZE(2)][SIZE(3)+1];
			for (i=0; i<SIZE(2); i++) {
				for (j=0; j<SIZE(3); j++) ans[i][j]='.';
				ans[i][j]='\0';
			}
			for (j=0; j<SIZE(0); j++) ans[0][j]=s[m[0]][j];
			for (j=0; j<SIZE(3); j++) ans[SIZE(1)-1][j]=s[m[3]][j];
			for (j=0; j<SIZE(5); j++) ans[SIZE(2)-1][j+SIZE(0)-1]=s[m[5]][j];
			for (i=0; i<SIZE(1); i++) ans[i][0]=s[m[1]][i];
			for (i=0; i<SIZE(2); i++) ans[i][SIZE(0)-1]=s[m[2]][i];
			for (i=0; i<SIZE(4); i++) ans[i+SIZE(1)-1][SIZE(3)-1]=s[m[4]][i];
			ans1=""; 
			for (i=0; i<SIZE(2); i++) {
				for (j=0; j<SIZE(3); j++) ans1+=ans[i][j];
				ans1+="\n";
			}
			if (ans1<ans2) ans2=ans1;
		}
	} while (next_permutation(m, m+6));
	if (ans2=="z") cout<<"Impossible"<<endl;
	else cout<<ans2<<endl;
	return 0;
}
