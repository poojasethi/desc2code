#include "cstdlib"
#include "cstdio"
#include "cstring"
using namespace std;
void solve(char *s)
{
    int ans=atoi(s), f=0, t=10;
    for (int i=1; i<strlen(s); i++)
    {
        f+=t;
        t*=10;
    }
    while (ans< 1989+f) ans+=t;
    printf("%d\n", ans);
    return;
}
int main()
{
    int n;
    scanf("%d", &n);
    while (n-->0)
    {
        char s[20];
        scanf("%s", s);
        solve(s+4);
    }
    return 0;
}
