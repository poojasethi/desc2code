#include<bits/stdc++.h>
using namespace std;
string s;
int main()
{
    int n;
    scanf("%d",&n);
    while(n--)
    {
        cin>>s;
        string s1=s.substr(4);
        int k=s1.length(),k1=k-1,p1=10,p=0;
        while(k1>0)
        {
            p+=p1;
            p1*=10;
            k1--;
        }
        int n1=atoi(s1.c_str());
        while(n1<p+1989)
            n1+=p1;
        printf("%d\n",n1);
    }
    return 0;
}
