#include <iostream>
#include <string>
#include <stdlib.h>

using namespace std;

int solve(string s)
{
	int ret = 1989+(atoi(s.substr(s.length()-1).c_str())+1)%10;
	int pow = 10;
	for (int l=s.length()-2; l >= 4; l--)
	{
		ret += pow;
		while (ret%(10*pow) != atoi(s.substr(l).c_str()))
			ret += pow;
		pow *= 10;
	}
	
	return ret;
}

int main()
{
	int n;
	string s;
	cin >> n;
	
	for (int i=0; i<n; i++)
	{
		cin >> s;
		cout << solve(s) << endl;
	}
}
