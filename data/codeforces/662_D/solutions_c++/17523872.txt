#include <iostream>
using namespace std;

int main()
{
    string str;
    int len,t,d=1,e=10,year=1988;
    cin>>t;
    while(t--)
    {
        d=1,e=10,year=1988;
        cin>>str;
        len=str.length();
        while((len--)-4)
        {
            do
            {
                year+=d;    
            }while(str[len]-'0'!=((year%e)/d));
            d*=10;
            e*=10;
        }
        cout<<year<<endl;
    }
    return 0;
}
