import math, random, re, sys
while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		s = sys.stdin.readline ().strip ()
		if s == '':
			break
	a = int (s)
	b = int (sys.stdin.readline ())
	d = a - b
	x = d / 2
	y = a - x
	if a < b:
		print -1
	elif d % 2 == 1:
		print -1
	elif x ^ y != b:
		print -1
	else:
		if x > y:
			x, y = y, x
		print x, y
