import sys
aa,bb = input(),input()
p = [2**x for x in range(63,-1,-1)]
a = [aa&x for x in p]
b = [bb&x for x in p]
s = 0
r = []
def imp():
    print -1
    sys.exit(0)
for x,y in zip(a,b):
    s = s*2+(1 if x else 0)
    if y:
        if s>2 or s<1: imp()
        r.append(0)
        s-=1
    else:
        if s>3: imp()
        if s<2:
            r.append(0)
        else:
            r.append(1)
            s-=2
x = int(''.join(map(str,r)),2) if r else 0
if x+(x^bb)!=aa: imp()
print x,bb^x
