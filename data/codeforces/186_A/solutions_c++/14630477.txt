#include<cstdio>
#include<cstring>

char a[111111], b[111111];
int tot, d[55];

int main() {
	scanf("%s%s", a, b);
	tot=0;
	for(int i=0;a[i];i++)
		if(a[i]!=b[i]&&tot<10)d[tot++] = i;
	if(tot==2&&a[d[0]]==b[d[1]]&&a[d[1]]==b[d[0]]&&strlen(a)==strlen(b)) printf("YES\n");
	else printf("NO\n");
	
	return 0;
}
