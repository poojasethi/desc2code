#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
string s,str;
cin>>s>>str;
int k=0;
for(int i=0;i<s.size();i++){
	if(s[i]!=str[i]) k++;
}
sort(s.begin(),s.end());
sort(str.begin(),str.end());
if(s==str && k==2) cout<<"YES"<<endl;
else cout<<"NO"<<endl;
	return 0;
}
