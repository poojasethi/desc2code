#include <iostream>
#include <algorithm>

using namespace std;

int main(){
	string a,b,x,y;
	cin>>a>>b;
	x=a;y=b;
	sort(x.begin(),x.end());
	sort(y.begin(),y.end());
	if(x!=y){
		cout<<"NO"<<endl;
		return 0;
	}
	int c=0;
	for(int i=0;i<a.size();i++){
		if(a[i]!=b[i]) c++;
	}
	a=(c>2)? "NO":"YES";
	cout<<a<<endl;

	return 0;
}
