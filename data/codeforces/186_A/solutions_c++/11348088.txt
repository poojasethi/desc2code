#include<bits/stdc++.h>
using namespace std;
const int MAXN =100000;
int ind[MAXN+100];
int main()
{
	string a,b;
	cin>>a>>b;
	int m=0;
	for (int i=0;i<a.size();i++){
		if (a[i]!=b[i])
			ind[m++]=i;
	}
	if (a.size()!=b.size())
		cout<<"NO"<<endl;
	else if (a==b||(m==2&&a[ind[0]]==b[ind[1]]&&a[ind[1]]==b[ind[0]]))
		cout<<"YES"<<endl;
	else 
		cout<<"NO"<<endl;
}
