#include <bits/stdc++.h>
using namespace std;
int main(){
	string a,b;
	int aux=0;
	cin>>a>>b;
	for (int i = 0; i < a.length(); ++i)
		aux+=a[i]!=b[i] ? 1:0;

	if(aux>2)
		cout<<"NO"<<endl;
	else{
		sort(a.begin(), a.end());
		sort(b.begin(), b.end());
		if(a==b)
			cout<<"YES"<<endl;
		else	
			cout<<"NO"<<endl;
	}	

return 0;	
}
