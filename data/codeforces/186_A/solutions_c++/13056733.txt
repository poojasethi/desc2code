#include <iostream>
#include <algorithm>
using namespace std;

int main()
{

string x,z;
int p,c=0;
cin>>x>>z;
p=x.size();
for(int i=0;i<p;i++)
{
    if(x[i]!=z[i])
    {
        c++;
    }
}
sort (x.begin(),x.end());
sort (z.begin(),z.end());
if((x==z)&& (c==2))
{
    cout<<"YES";
}
else{cout<<"NO";}











    return 0;
}
