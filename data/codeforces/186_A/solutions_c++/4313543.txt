#include <iostream>
#define nu {cout<<"NO";return 0;}
using namespace std;
int i,j,d[3];
char a[100005],b[100005];
int main()
{
    cin>>a>>b;
    for(i=0;a[i]&&b[i];i++)
    {
        if(a[i]-b[i])d[j++]=i;
        if(j==3)nu
    }
    if(a[i]||b[i])nu
    if(j-2)nu
    i=d[0];j=d[1];
    if(a[i]-b[j]||b[i]-a[j])nu
    cout<<"YES";
    return 0;
}
