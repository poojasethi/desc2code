#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main(){
	string a,b;
	cin>>a>>b;
	int diff=0;
	for (int i=0;i<a.size() && i<b.size();i++) if (a[i]!=b[i]) diff++;
	sort(a.begin(),a.end());
   sort(b.begin(),b.end());
	if (a==b&&diff<3) cout<<"YES";
	else cout<<"NO";
}
