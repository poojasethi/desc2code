#include <iostream>
#include <algorithm>
#include <string>

using namespace std;
string a,b;
int c=0;

int main()
{
    cin>>a>>b;
    for(int i=0;i<a.size()&&i<b.size();i++)
        c+=a[i]!=b[i];
    sort(a.begin(),a.end());
    sort(b.begin(),b.end());
    cout<<(a==b&&c==2 ? "YES\n" : "NO\n");  
}
