#include <iostream>
#include <algorithm>
#include <string>


using namespace std;
int main()
{
    int z=0;
string x,y;
cin >> x>>y;

for(int i=0;i<x.size();i++)
{z+=x[i]!=y[i];}

    sort(x.begin(),x.end());
    sort(y.begin(),y.end());
    if(x==y&&z==2)
        cout << "YES";
    else
        cout << "NO";


}
