#include<iostream>
using namespace std;
int main(){
long i,k=0,in[9];
string a,b;
cin>>a>>b;
if(a.length()==b.length())
    for(i=0;i<a.length()&&k<3;i++){
        if(a[i]!=b[i]){
            in[k++]=i;
        }
    }
if(k==2&&a[in[0]]==b[in[1]]&&a[in[1]]==b[in[0]])
    cout<<"YES\n";
else
    cout<<"NO\n";

return 0;
}
