#include <iostream>
#include <cstring>
using namespace std;
int main()
{
string s1,s2;
cin>>s1>>s2;
if(s1.length()!=s2.length())
{
cout<<"NO\n";
return 0;
}
int lm=0,um=s1.length()-1;
while(s1[lm]==s2[lm])lm++;
while(s1[um]==s2[um])um--;
char t=s1[lm];
s1[lm]=s1[um];
s1[um]=t;
if(s1.compare(s2)==0)
{
cout<<"YES\n";
return 0;
}
cout<<"NO\n";
}
