#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   string x,z; int c=0;
   cin>>x>>z;
   if(x.size()!=z.size()){cout<<"NO"; return 0;}
 for(int i=0;i<x.size();i++)
 {
     if(x[i]!=z[i])c++;
 }
 sort(x.begin(),x.end());
 sort(z.begin(),z.end());
 if(c==2&&z==x)cout<<"YES";
    else cout<<"NO";
   return 0;
}
