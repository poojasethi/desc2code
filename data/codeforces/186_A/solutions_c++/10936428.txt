
#include <iostream>

#include <ext/hash_map>
using namespace std;


int main(){
    string a,b;
    cin>>a>>b;
    int i,j=0;
    for(i=0;i<a.size();i++) j+=a[i]!=b[i];
    if(j>2){
        cout<<"NO"<<endl;
    }else{
        sort(a.begin(),a.end());
        sort(b.begin(),b.end());
        cout<<(a==b?"YES":"NO")<<endl;
    }
}
