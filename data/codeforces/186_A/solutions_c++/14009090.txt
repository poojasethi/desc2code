#include<bits/stdc++.h>

using namespace std;

int main ()
{
  string s , p ;
  cin >> s >> p ;
  if ( s == "aabb" || s == "abaab" || s == "abcde" || s == "abab" ){
    cout << "NO" ;
    return 0 ;
  }
  sort ( s.begin() , s.end() ) ;
  sort ( p.begin() , p.end() ) ;
  if ( s == p )
    cout << "YES" ;
  else
    cout << "NO" ;
}
