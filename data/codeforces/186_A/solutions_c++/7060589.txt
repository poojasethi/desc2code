#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int main()
{
    string x,y;
    int c=0;
    cin>>x>>y;
    for(int i=0;i<x.size();i++)
        c+=x[i]!=y[i];
    sort(x.begin(),x.end());
    sort(y.begin(),y.end());
    if(x==y && c==2)
        cout << "YES";
    else
        cout << "NO";
    return 0;
}
