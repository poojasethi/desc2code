import sys

a = raw_input()
b = raw_input()
if len(a) != len(b):
	print 'NO'
	sys.exit()
lhs = -1
rhs = -1
cnt = 0
for i in range(0, len(a)):
	if a[i] != b[i]:
		cnt += 1
		if lhs == -1:
			lhs = i
		else:
			rhs = i
if cnt != 2 or not (a[lhs] == b[rhs] and a[rhs] == b[lhs]):
	print 'NO'
else:
	print 'YES'
