m=raw_input()

if(len(n)!=len(m)):
    print 'NO'
else:
    for i in range(len(m)):
        if(n[i]<>m[i]):
            n_d+=1
    m=sorted(m)
    n=sorted(n)
    if (m==n and n_d ==2):
            print 'YES'
    else:
            print 'NO'
