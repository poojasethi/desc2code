a = raw_input().strip()
b = raw_input().strip()
a = list(a)
b = list(b)
c = []
if len(a) != len(b):
	print "NO"
else:
	for i in range(0,len(a)):
		if a[i] != b[i]:
			c.append(i)
	if len(c) == 2:
		d = a[:]
		d[c[0]] = a[c[1]]
		d[c[1]] = a[c[0]]
		if d == b:
			print "YES"
		else:
			print "NO"
	else:
		print "NO"
