A=raw_input()
B=raw_input()
if(len(A)==1):
	print "NO"
	exit()
if(len(B)==len(A)):
	l=len(A)
	count=0
	L=[]
	K=[]
	for x in xrange(l):
		if(A[x]==B[x]):
			continue
		else:
			L.append(A[x])
			K.append(B[x])
			count+=1
	if(count==2):
		L.sort()
		K.sort()
		if(L==K):
			print "YES"
		else:
			print "NO"
	else:
		print "NO"
else:
	print "NO"
