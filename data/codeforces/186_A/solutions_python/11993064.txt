#coding: utf-8

s1 = raw_input()
s2 = raw_input()
if(len(s1)!=len(s2)):
	print 'NO'
else:
	swaps = []
	for i in xrange(len(s1)):
		if(s1[i]!=s2[i]):
			swaps.append(i)
	if(len(swaps)==2):
		if(s1[swaps[0]]==s2[swaps[1]] and s2[swaps[0]]==s1[swaps[1]]):
			print 'YES'
		else:
			print 'NO'
	else:
		print 'NO'
