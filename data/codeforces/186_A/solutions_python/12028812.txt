#coding: utf-8

a = raw_input()
b = raw_input()
contador = 0
if len(a) != len(b) or set(a) != set(b):
    print "NO"
else:
    for i in range(min(len(a),len(b))):
    
        if a[i] != b[i] :
            contador += 1

    if contador == 2:
        print"YES"

    else :
        print"NO"
