def solve():
    s1 = raw_input()
    s2 = raw_input()
    if sorted(s1) != sorted(s2):
        print "NO"
        return
    mismatch = 0
    for i in range(len(s2)):
        if s2[i] != s1[i]:
            mismatch += 1
        if mismatch > 2:
            print "NO"
            return
    print "YES"


solve()
