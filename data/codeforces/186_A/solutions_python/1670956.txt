import sys

a = raw_input()
b = raw_input()

if len(a) != len(b) :
    print 'NO'
    sys.exit(0)

t = []
for i in range(len(a)) :
    if a[i] != b[i] :
        t.append(i) 
    if len(t) > 2 :
        break
    
if len(t) == 2 and a[t[0]] == b[t[1]] and a[t[1]] == b[t[0]] :
    print 'YES'
else :
    print 'NO'
