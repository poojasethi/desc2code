first=list(raw_input())
second=list(raw_input())
list1,list2=26*[0],26*[0]
for i in first:
	list1[ord(i)-ord('a')]+=1
for i in second:
	list2[ord(i)-ord('a')]+=1
if cmp(list1,list2)==0:
	j=[i for i in xrange(len(first)) if first[i]!=second[i]]
	if len(j)==2:
		print 'YES'
	else:
		print 'NO'
else:
	print 'NO'
