b = raw_input()
sa = len(a)
sb = len(b)
if sa != sb:
  print 'NO'
  exit(0)
da = []
db = []
cnt = 0
for i in xrange(sa):
  if a[i] != b[i]:
    da.append(a[i])
    db.append(b[i])
if len(da) == 2 and len(db) == 2 and da[0] == db[1] and da[1] == db[0]:
  print 'YES'
