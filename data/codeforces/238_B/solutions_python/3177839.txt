R=lambda:map(int,raw_input().split())
n,h=R()
a=sorted(map(lambda x:(int(x[1]),x[0]),enumerate(R())))
if n>2:
  x,y=a[-1][0]+a[-2][0]-a[0][0]-a[1][0],max(a[-1][0]+a[-2][0],a[-1][0]+a[0][0]+h)-min(a[0][0]+a[1][0]+h,a[1][0]+a[2][0])
  print min(x,y)
  if x<y:
    print ' '.join(['1']*n)
  else:
    print ' '.join(map(str,(1+(i==a[0][1]) for i in range(n))))
else:
  print 0
  print ' '.join(['1']*n)
