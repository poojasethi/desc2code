r = lambda: map(int, raw_input().split())
n, h = r()
if n == 2:
    print 0
    print 1, 1
else:
    a = sorted((x, i) for i, x in enumerate(r()))
    ans = [1 for i in xrange(n)]
    p = (a[-1][0] + a[-2][0]) - (a[0][0] + a[1][0]) 
    q = (a[-1][0] + max(a[-2][0], a[0][0] + h)) - (a[1][0] + min(a[0][0] + h, a[2][0]))
    if p < q:
        print p
        print ' '.join(map(str, ans))
    else:
        print q
        ans[a[0][1]] = 2
        print ' '.join(map(str, ans))
