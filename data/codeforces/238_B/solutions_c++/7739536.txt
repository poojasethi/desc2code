#include <cstdio>
#include <algorithm>
#define max(x,y) (((x)>(y))?(x):(y))
#define min(x,y) (((x)<(y))?(x):(y))
using namespace std;

class Q{
public:int x,num;
};

int n, h;
int b[100010];
Q a[100010];

bool cmp(const Q &a, const Q &b){
    return a.x<b.x;
}

int main(){
    scanf("%d%d",&n,&h);
    for (int i=1;i<=n;i++){
        scanf("%d",&a[i].x);
        a[i].num=i;
    }
    for (int i=1;i<=n;i++) b[i]=2;
    sort(a+1,a+n+1,cmp);
    int s1=a[n].x+a[n-1].x-a[1].x-a[2].x;
    int s2=max(a[n].x+a[n-1].x,a[n].x+a[1].x+h)-min(a[2].x+a[3].x,a[1].x+a[2].x+h);
    if(s1>s2){
		b[a[1].num]=1;
		s1=s2;
	}
    printf("%d\n",s1);
    for (int i=1;i<=n;i++)
		printf("%d ",b[i]);
    printf("\n");
    return 0;
}
