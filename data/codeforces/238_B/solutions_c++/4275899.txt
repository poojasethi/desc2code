#include<iostream>
#include<cstring>
#include<algorithm>
#define maxn 100007
using namespace std;

struct node
{
    int x,id;
}a[maxn];

int b[maxn],n,h;

bool cmp(node a, node b)
{
    return a.x<b.x;
}

int main()
{
    cin>>n>>h;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i].x;
        a[i].id=i;
    }
    for(int i=1;i<=n;i++)b[i]=2;
    sort(a+1,a+n+1,cmp);
    int ans1=a[n].x+a[n-1].x-a[1].x-a[2].x;
    int ans2=max(a[n].x+a[n-1].x,a[n].x+a[1].x+h)-min(a[1].x+a[2].x+h,a[2].x+a[3].x);
    if(ans1>ans2){b[a[1].id]=1;ans1=ans2;}
    cout<<ans1<<endl;
    for(int i=1;i<=n;i++)
       cout<<b[i]<<' ';
       cout<<endl;
    return 0;
}
