
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,h;

cin>>n>>h;

vector <LL> v(n),u(n);

for(i=0;i<n;i++)
	{cin>>v[i];
	 u[i]=v[i];	
	}
sort(v.begin(),v.end());

//pos1
LL g1,g2;

g1 = v[n-1]+v[n-2]-v[0]-v[1];
g2 = max(v[0]+v[n-1]+h,v[n-1]+v[n-2]) - min(v[1]+v[2],v[0]+v[1]+h);

if(g1 < g2)
{
cout<<g1<<endl;
for(i=0;i<n;i++)
	cout<<2<<' ';
cout<<endl;
}
else
{
cout<<g2<<endl;
LL k=1;

for(i=0;i<n;i++)
	if(k==1 && u[i]==v[0])
		{cout<<1<<' ';
		k--;	
		}
	else	
		cout<<2<<' ';
cout<<endl;
}	


}
