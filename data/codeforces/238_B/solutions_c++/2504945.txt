#include<cstdio>
#include<cstring>
#define mem(a,b) memset(a,b,sizeof(a))
#include<algorithm>
using namespace std;
typedef long long ll;

const int P = 1000000009;

const int N = 111111;

int main()
{
	int n,h; scanf("%d%d",&n,&h);
	int a[N];
	int mni=0;
	for(int i=0;i<n;i++) {
		scanf("%d",&a[i]);
		if(i && a[i]<a[mni]) mni=i;
	}
	sort(a,a+n);

	if(n==2) {
		puts("0\n1 1\n");
		return 0;
	}

	int ans= (a[n-1]+a[n-2]) - (a[0]+a[1]);
	int tp=0;

	int mn=min(a[0]+a[1]+h,a[1]+a[2]);
	int mx=max(a[n-1]+a[n-2],a[0]+a[n-1]+h);
	int tmp=mx-mn;

	if(ans>tmp) {
		ans=tmp, tp=1;
	}

	printf("%d\n",ans);
	if(!tp) {
		for(int i=0;i<n;i++) printf("1 ");
		puts("");
	} else {
		for(int i=0;i<n;i++) printf(i==mni?"1 ":"2 ");
		puts("");
	}

	return 0;
}
