#include<cstdio>
#include<algorithm>
using namespace std;

int n,h,p,mi,ma,det;
int a[111111];
bool split;

int main(){
	scanf("%d%d",&n,&h);
	p=0;
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		if(a[i]<a[p])p=i;
	}
	sort(a,a+n);
	
	if(n==2){
		printf("0\n1 1\n");
		return 0;
	}
	
	split=false;
	mi=a[0]+a[1];ma=a[n-1]+a[n-2];
	det=ma-mi;
	
	mi=min(a[0]+a[1]+h,a[1]+a[2]);
	ma=max(a[0]+a[n-1]+h,a[n-2]+a[n-1]);
	if(ma-mi<det) {
		split=true;
		det=ma-mi;
	}
	
	printf("%d\n",det);
	for(int i=0;i<n;i++){
		int v=2;
		if(i==p&&split)v=1;
		printf("%d%c",v,i<n-1?' ':'\n');
	}

	return 0;
}
