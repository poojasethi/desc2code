#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,h;
int a[N],b[N];
int main() {
	scanf("%d%d",&n,&h);
	For(i,1,n) scanf("%d",&a[i]),b[i]=a[i];
	sort(a+1,a+n+1);
	int an=(a[n]+a[n-1])-(a[2]+a[1]);
	int re=a[1]+a[n]+h;
	if (n>2) re=max(re,a[n]+a[n-1]);
	int mi=a[1]+a[2]+h;
	if (n>2) mi=min(mi,a[2]+a[3]);
	if (an>re-mi) {
		printf("%d\n",re-mi);
		For(i,1,n) if (b[i]==a[1]) break;
		k=i;
		For(i,1,n) if (i!=k) printf("1 ");
		else printf("2 ");
		printf("\n");
	} else {
		printf("%d\n",an);
		For(i,1,n) printf("1 ");
		printf("\n");
	}
	return 0;
}
