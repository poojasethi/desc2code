#include <cstdio>
#include <cstdlib>
#include <algorithm>
struct inf{int a,p;}t[100001];int n,h;
bool operator < (const inf& t1,const inf& t2)
{return t1.a<t2.a;}
inline int min(int a,int b){return a>b?b:a;}
inline int max(int a,int b){return a>b?a:b;}
int main()
{
	scanf("%d%d",&n,&h);
	for(int i=1;i<=n;i++){scanf("%d",&t[i].a);t[i].p=i;}
	std::sort(t+1,t+n+1);
	int ans1=t[n].a+t[n-1].a-t[1].a-t[2].a;
	int ans2=max(t[1].a+t[n].a+h,t[n].a+t[n-1].a)-min(t[1].a+t[2].a+h,t[2].a+t[3].a);
	if(ans1<ans2)
	{
		printf("%d\n",ans1);
		for(int i=1;i<=n;i++)
			printf("1 ");
		printf("\n");
	}
	else
	{
		printf("%d\n",ans2);
		for(int i=1;i<=n;i++)
		{
			if(i==t[1].p)printf("1 ");
			else printf("2 ");
		}
		printf("\n");
	}
}
