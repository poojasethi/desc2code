#include <cstdio>
#include <algorithm>
using namespace std;
int a[100000];

int main()
{
	int n,h,i,x1,y1,x2,y2,ans,k=0;
	scanf("%d%d",&n,&h);
	for (i=0; i<n; ++i)
	{
		scanf("%d",a+i);
		if (a[i]<a[k]) k=i;
	}
	sort(a,a+n);
	x1=a[0]+a[1]+h; y1=a[0]+a[n-1]+h;
	if (n>2) x1=min(x1,a[1]+a[2]), y1=max(y1,a[n-1]+a[n-2]);
	x2=a[0]+a[1]; y2=a[n-1]+a[n-2];
	if (y1-x1<y2-x2) ans=y1-x1; else ans=y2-x2, k=-1;
	printf("%d\n",ans);
	for (i=0; i<n; ++i)
		if (i==k) printf("2 "); else printf("1 ");
	printf("\n");
	return 0;
}
