#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
struct Node
{
	int x, idx;
} a[100010];
int n, h;
int ans[100010];

bool cmp(Node p, Node q)
{
	return p.x < q.x;
}

int main()
{
	scanf("%d%d", &n, &h);
	for (int i = 1; i <= n; i++)
	{
		scanf("%d", &a[i].x);
		a[i].idx = i;
		ans[i] = 1;
	}
	sort(a + 1, a + n + 1, cmp);
	int ans1 = (a[n].x + a[n - 1].x) - (a[1].x + a[2].x);
	int ans2 = max(a[n].x + a[n - 1].x, a[n].x + a[1].x + h) - min(a[2].x + a[3].x, a[1].x + a[2].x + h);
	if (ans2 < ans1)
	{
		ans[a[1].idx] = 2;
		ans1 = ans2;
	}
	printf("%d\n", ans1);
	for (int i = 1; i <= n; i++)
	{
		if (i > 1) printf(" ");
		printf("%d", ans[i]);
	}
	puts("");
	return 0;
}
