#include <algorithm>
#include <iostream>
using namespace std;

typedef pair<int, int> pii;

#define F first
int n, h, ans[100 * 1000];
pii p[100 * 1000];

int main() {
	ios::sync_with_stdio(false);
	cin >> n >> h;
	for (int i = 0; i < n; i++) {
		cin >> p[i].first;
		p[i].second = i;
	}
	sort(p, p + n);
	if (n > 2 && p[n - 2].F - p[0].F > max(p[0].F + h, p[n - 2].F) - min(p[2].F, p[0].F + h)) {
		cout << p[n - 1].F + max(p[0].F + h, p[n - 2].F) - min(p[2].F, p[0].F + h) - p[1].F << '\n';
		ans[p[0].second] = 1;
	}
	else
		cout << p[n - 1].F + p[n - 2].F - p[1].F - p[0].F << '\n';
	for (int i = 0; i < n; i++)
		cout << ans[i] + 1 << ' ';
	cout << '\n';
	return 0;
}
