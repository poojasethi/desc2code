#include <iostream>
#include <algorithm>

using namespace std;

const int Inf=100*1000*1004;
const int MaxN=100*1001;

int n,h;
int a[MaxN];

int main()
{
	cin>>n>>h;
	
	int ind=0,m=Inf;
	for(int i=0;i<n;++i)
		{cin>>a[i];if(a[i]<m){m=a[i];ind=i;}}
		
	sort(a,a+n);
	
	bool f=0;
	m=(a[n-1]+a[n-2])-(a[0]+a[1]);
	int t=(a[n-1]+max(a[n-2],a[0]+h))-(a[1]+min(a[2],a[0]+h));
	if(m>t)
		{m=t;f=1;}
	
	cout<<m<<endl;
	for(int i=0;i<n;++i)
		if(f && i==ind)
			cout<<1<<" ";
		else
			cout<<2<<" ";
	cout<<endl;
	
	return 0;
}
