#include<iostream>
#include<algorithm>
using namespace std;

int a[100005];

int main()
{
    int n,i,h,j;
    while(cin>>n>>h)
    {

         for(i=0;i<n;i++)
         cin>>a[i];
         if(n==2) {cout<<0<<endl<<1<<" "<<1<<endl;continue;}         
         j=min_element(a,a+n)-&a[-1];         
         sort(a,a+n);
         int t1=a[n-2]+a[n-1]-(a[1]+a[0]);  
         int t2=max(a[n-2]+a[n-1],a[n-1]+a[0]+h)-min(a[1]+a[0]+h,a[2]+a[1]);  
         if(t1<t2) j=-1;  
         cout<<min(t1,t2)<<endl;  
         for(i=1;i<=n;i++)  
         {
            if(i==j) cout<<1;
            else cout<<2;
            if(i==n) cout<<endl;
            else cout<<" ";
         }              
    }
    return 0;
}
