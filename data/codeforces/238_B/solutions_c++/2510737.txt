#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
#define N 101000
struct elem
{
	int x, pos;
	bool operator <(const elem &p) const {
		return x < p.x;
	}
}a[N];
int n, h, f, MAX, MIN;
int main()
{
	cin >> n >> h;
	for (int i = 1; i <= n; i ++) scanf("%d", &a[i].x), a[i].pos = i;
	sort(a + 1, a + n + 1);
	f = a[n].x + a[n - 1].x - a[1].x - a[2].x;
	MAX = max(a[n - 1].x + a[n].x, a[n].x + a[1].x + h);
	MIN = min(a[1].x + a[2].x + h, a[2].x + a[3].x);
	if (f < MAX - MIN) {
		cout << f << endl;
		for (int i = 1; i <= n; i ++) printf("1 ");
	}
	else {
		cout << MAX - MIN << endl;
		for (int i = 1; i <= n; i ++) printf("%d ", 1 + (i != a[1].pos));
	}
	return 0;
}
