#include<cstdio>
#include<algorithm>

using namespace std;

struct node
{
   int data,num;
};    
node a[100010];
int f[100010];

bool cmp(node a,node b)
{
    return (a.data<b.data);
} 

int main()
{
   int n,h; 
   scanf("%d%d",&n,&h);
   if (n==2) {printf("0\n1 1\n");return 0;}
   for (int i=1;i<=n;i++) scanf("%d",&a[i].data);
   for (int i=1;i<=n;i++) a[i].num=i;
   sort(a+1,a+n+1,cmp);
   for (int i=1;i<=n;i++) f[i]=1;
   int t1=a[n].data+a[n-1].data-a[1].data-a[2].data,t2=max(a[n].data+a[n-1].data,a[1].data+a[n].data+h)-min(a[1].data+a[2].data+h,a[2].data+a[3].data);
   printf("%d\n",min(t1,t2));
   if (t1>t2) f[a[1].num]=2;
   for (int i=1;i<n;i++) printf("%d ",f[i]);
   printf("%d\n",f[n]);
   return 0;
}
