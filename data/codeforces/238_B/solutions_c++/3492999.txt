#include<stdio.h>
#include<map>
#include<algorithm>
using namespace std;
typedef pair<int,int> P;
P ar[100000];
int ans[100000];
int min(int a,int b){return a<b?a:b;}
int max(int a,int b){return a>b?a:b;}
int main(){
    int n,h,i;
    scanf("%d %d",&n,&h);
    for(i=0;i<n;i++){
        scanf("%d",&ar[i].first);
        ar[i].second=i;
    }
    sort(ar,ar+n);
    int m1=(ar[n-2].first+ar[n-1].first)-(ar[0].first+ar[1].first);
    int m2=max(ar[n-2].first+ar[n-1].first,ar[0].first+ar[n-1].first+h)-min(ar[1].first+ar[2].first,ar[0].first+ar[1].first+h);
    if(m1<m2)for(i=0;i<n;i++)ans[i]=1;
    else{
        ans[ar[0].second]=1;
        for(i=1;i<n;i++)ans[ar[i].second]=2;
    }
    printf("%d\n",min(m1,m2));
    for(i=0;i<n;i++)printf("%d ",ans[i]);
    return 0;
}
