#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
const int maxn = 1e5 + 5;
int n, h;
int a[maxn], b[maxn];
int min_pos;
int main()
{
	//freopen("test.in", "r", stdin);
	cin >> n >> h;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < n; i++)
	{
		b[i] = a[i];
		if (a[i] < a[min_pos])
			min_pos = i;
	}
	sort(b, b + n);
	int delta1 = max(b[n-1] + b[n-2], b[n-1] + b[0] + h) - min(b[0] + b[1] + h, b[1] + b[2]);
	int delta2 = (b[n-1] + b[n-2]) - (b[0] + b[1]);
	if (delta1 < delta2)
	{
		printf("%d\n", delta1);
		for (int i = 0; i < n; i++)
			if (i != min_pos)
				printf("1 ");
			else
				printf("2 ");
	}
	else
	{
		printf("%d\n", delta2);
		for (int i = 0; i < n; i++)
				printf("1 ");
	}
	return 0;
}
