#include <bits/stdc++.h>

using namespace std;

int N,H,ar[110000],S[110000];

int comp( const int &a , const int &b ){ return ar[a] < ar[b]; }

int main(){
	
	cin >> N >> H;
	
	for( int i=1 ; i<=N ; i++ ) scanf(" %d",&ar[i]),S[i]=i;
	
	sort( S+1 , S+1+N , comp );
	
	int a=ar[S[N]]+ar[S[N-1]]-ar[S[1]]-ar[S[2]];
	int b=ar[S[N]]+max( ar[S[N-1]] , ar[S[1]]+H )-min( ar[S[1]]+ar[S[2]]+H , ar[S[2]]+ar[S[3]] );
	if( a <= b ){
		cout << a << endl;
		for( int i=1 ; i<=N ; i++ ) printf("1 ");
	}else{
		cout << b << endl;
		for( int i=1 ; i<=N ; i++ )
			if( i == S[1] ) printf("2 ");
			else printf("1 ");
	}
	
	return 0;
	
}
