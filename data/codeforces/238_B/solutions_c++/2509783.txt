#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[100010],b[100010];
int main()
{
	int i,tmp1,tmp2,n,h;
	scanf("%d%d",&n,&h);
	for(i=0; i<n; i++)	scanf("%d",a+i);
	for(i=0; i<n; i++)	b[i] = a[i];
	sort(b,b+n);
	tmp1 = b[n-1] + b[n-2] - b[1] - b[0];
	b[0] += h;
	sort(b,b+n);
	tmp2 = b[n-1] + b[n-2] - b[1] - b[0];
	printf("%d\n",min(tmp1, tmp2));
	if(tmp1 < tmp2){
		printf("1");
		for(i=1; i<n; i++)	printf(" 1");
		puts("");
	}else{
		bool first = true;
		int Min = 0;
		for(i=0; i<n; i++)	if(a[i] < a[Min])	Min = i;
		for(i=0; i<n; i++){
			if(first)	first = false;
			else	printf(" ");
			printf("%d",(i==Min) + 1);
		}
		puts("");
	}
	return 0;
}
