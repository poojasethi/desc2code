#include <bits/stdc++.h>
using namespace std;

const int N=100010;
long long int n, h, pr[N], val1, val2, mini, maxi, prmini;
pair<long long int,int> a[N];

int main()
{
   //freopen("r.txt","r",stdin);
   cin>>n>>h;
   for(int i=0;i<n;i++)
   {
      cin>>a[i].first;
      a[i].second=i;
   }
   sort(a,a+n);

   maxi=(a[n-2].first+a[n-1].first), mini=(a[0].first+a[1].first);
   val1=abs(maxi-mini);
   prmini=val1;

   maxi=max(maxi,(a[0].first+a[n-1].first+h)), mini=(a[0].first+a[1].first+h);
   mini=min(mini, (a[2].first+a[1].first) );
   val2=abs(maxi-mini);
   if(val2<prmini)
        prmini=val2, pr[a[0].second]=1;

   cout<<prmini<<"\n";
   for(int i=0;i<n;i++)
      cout<<pr[i]+1<<" ";

    return 0;
}
