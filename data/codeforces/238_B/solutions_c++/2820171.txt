#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

const int N = 100003;
int a[N];

int main()
{
    int n, h;
    scanf("%d%d", &n, &h);
    for(int i = 0; i < n; ++i)
        scanf("%d", a+i);
    if(n == 2) {
        printf("0\n1 1\n");
        return 0;
    }
    
    int k = min_element(a, a + n) - a;
    sort(a, a + n);
    
    int x = a[n-1]+a[n-2]-(a[0]+a[1]);
    int y = max(a[n-1]+a[n-2], a[n-1]+a[0]+h) - min(a[1]+a[2], a[0]+a[1]+h);
    if(x <= y) {
        printf("%d\n", x);
        k = -1;
    }
    else printf("%d\n", y);
    
    for(int i = 0; i < n; ++i) {
        if(i) printf(" ");
        printf("%d", i == k ? 2 : 1);
    }
    printf("\n");
    return 0;
}
