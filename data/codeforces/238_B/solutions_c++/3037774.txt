#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
#define max(a,b) (a)>(b)?(a):(b)
#define min(a,b) (a)<(b)?(a):(b)
int cmp(int a,int b)
{
	return a<b;
}
int main()
{
	int n,h,flag1,flag2,k,f,g;
	int a[100001],b[100001];
	while (cin>>n>>h)
	{
		for(int i=1;i<=n;++i) 
		{
			cin>>a[i];
			b[i]=a[i];
		}
		sort(a+1,a+1+n,cmp);
		f=max(a[n]+a[n-1],a[1]+a[n]+h);
		g=min(a[1]+a[2]+h,a[2]+a[3]);
		flag1=a[n]+a[n-1]-a[1]-a[2];
		flag2=f-g;
		if(flag1<=flag2)
		{
			cout<<flag1<<endl;
			for(int i=1;i<=n;++i) cout<<'2'<<' ';
			cout<<endl;
		}
		else 
		{
			cout<<flag2<<endl;
			k=0;
			for(int i=1;i<=n;++i)
			{
				if(a[1]==b[i]&&k==0) {cout<<'1'<<' ';k++;}
				else cout<<'2'<<' '; 
			}
			cout<<endl;
		}
	}
	return 0;
}
