#include <iostream>
using namespace std;

int main()
    {
    int n,x=0,a,i; string c;
    cin>>n>>c;
    for(i=0;i<n/2;i++)
        {
        a=n-i-1; x+=c[i]; x-=c[a];
        if((c[i]!='4'&&c[i]!='7')||(c[a]!='4'&&c[a]!='7')) {x=1;break;}
    }
    if(x!=0) cout<<"NO";
    else cout<<"YES";
    return 0;
    }
