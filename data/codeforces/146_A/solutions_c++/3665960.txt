#include <stdio.h>
int main()
{
char s[50];
int i,n,sum1=0,sum2=0;
scanf("%d",&n);
scanf("%s",s);
for(i=0;i<n;i++)
   {
   if(s[i]!='4'&&s[i]!='7'){printf("NO");return 0;}
   if(i<n/2)
   sum1+=(s[i]-'0');
   else
   sum2+=(s[i]-'0');
   }
if(sum1==sum2) printf("YES");
else printf("NO");
return 0;
}
