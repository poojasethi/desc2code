#include<iostream>
using namespace std;
int main()
{ int n;
  cin>>n;
  string s;
  cin>>s;
  int i,b=1,s1=0,s2=0;
  for(i=0;i<n;i++){if(s[i]=='4' || s[i]=='7'){if(i<n/2)s1+=s[i]-'0';else s2+=s[i]-'0';} else {b=0;break;} }
  if(b==1 && s1!=s2)b=0;
  if(b==0) cout<<"NO"; else cout<<"YES"; 
  return 0;
 }
