#include<iostream>
using namespace std;
int main()
{
    long long i,n,S=0;
    string a;
    cin>>n>>a;
    for(i=0 ; i<a.size() ; i++){if(a[i]!='4' && a[i]!='7') {cout<<"NO"; return 0;}}
    for(i=0 ; i<n/2 ; i++) S+=a[i]-'0';
    for(; i<n ; i++) S-=a[i]-'0';
    if(S==0) {cout<<"YES";}
    else cout<<"NO";
}
