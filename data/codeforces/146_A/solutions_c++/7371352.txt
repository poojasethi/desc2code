#include<iostream>
using namespace std;
int main()
{
int n,i,sum=0;
cin>>n;
char *no= new char [n+1];
cin>>no;
for(i=0; i<n/2; i++)
if((no[i]=='4' || no[i]=='7') && (no[n-i-1]=='4' || no[n-i-1]=='7'))
sum+=no[i]-no[n-i-1];
else
break;
if(i<n/2 || sum!=0)
cout<<"NO";
else
cout<<"YES";
delete no;
return 0;
}
