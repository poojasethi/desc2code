#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	int t=1,n,x=0,y=0;
	string a;
	cin>>n>>a;
	for(int i=0;t==1&&i<n;i++){
		if(a[i]!='7'&&a[i]!='4')
			t=0;
		if(i+1<=n/2)
			x+=(int(a[i])-int('0'));
		else
			y+=(int(a[i])-int('0'));
	}
	if(x==y&&t==1)
		cout<<"YES";
	else
		cout<<"NO";
}
