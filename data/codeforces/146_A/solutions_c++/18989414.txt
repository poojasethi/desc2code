#include<iostream>
using namespace std;
int main()
{
int n,sum1=0,sum2=0,i,j=0,c4=0,c7=0,m;
string s;
cin>>n>>s;
for(i=1;i<=n;i++)
{
m=s[j]-'0';
if(i<=n/2)
sum1+=m;
else
sum2+=m;
if(m==4)c4+=1;
else if(m==7)c7+=1;
j+=1;
}
if(c4+c7==n)
cout<<(sum1==sum2?"YES":"NO")<<endl;
else
cout<<"NO"<<endl;
return 0;
}
