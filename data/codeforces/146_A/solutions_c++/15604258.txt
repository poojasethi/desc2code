#include<iostream>
using namespace std;
int main(){
char s[55]={0}; 
int n,i,f=1,sum=0,sum1=0;
cin>>n;
for(i=0;i<n;++i){
cin>>s[i];
if(s[i]!='4'&&s[i]!='7')
	f=0;
if(i<n/2)
  sum+=s[i]-'0';
  else sum1+=s[i]-'0';
} 
if(sum!=sum1)
       f=0;
if(f) cout<<"YES"<<endl;
else cout<<"NO"<<endl;
return 0;
}
