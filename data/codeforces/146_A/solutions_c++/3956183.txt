#include <iostream>
using namespace std;
int main() {
	bool pos = true;
	int l = 0, r = 0;
	int n;
	string s;
	cin >> n >> s;
	for (int i = 0; i < n; ++i) {
		if (s[i] != '4' and s[i] != '7') pos = false;
		if(i<n/2) l += s[i]-'0';
		else r += s[i]-'0';
	}
	cout << (pos and r==l ? "YES" : "NO") << endl;
}
