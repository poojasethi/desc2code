#include<stdio.h>

int main(){
	int n,s1=0,s2=0,ans=1,c;
	scanf("%d",&n);
	getchar();
	for(int i=0;i<n/2;i++){
		c=getchar();
		if(c!='4'&&c!='7'){ans=0;}
		s1+=c-'0';
	}
	for(int i=0;i<n/2;i++){
		c=getchar();
		if(c!='4'&&c!='7'){ans=0;}
		s2+=c-'0';
	}
	getchar();
	if(s1!=s2){ans=0;}
	puts(ans==1?"YES":"NO");
	return 0;
}
