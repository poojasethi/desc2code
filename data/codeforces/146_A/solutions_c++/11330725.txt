#include<stdio.h>

char a;
int top1,top2,n,i;

int main(){

	scanf("%d",&n);
	for(i = 1; i <= n; i++){
		scanf(" %c",&a);
		if(a != '4' and a != '7'){
			puts("NO");
			return 0;
		}
		if(i <= n/2) top2 += a-'0';
		else         top1 += a-'0';
	}
	if(top1 == top2)
		puts("YES");
	else
		puts("NO");

	return 0;
}
