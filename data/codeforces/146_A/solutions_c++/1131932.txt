#include <cstdio>
int n;
int main() {
	scanf("%d ", &n);
	bool yes = true;
	int sum = 0;
	for (int i = 0; i < n; ++i) {
		char c; c=getchar();
		if (c != '4' && c != '7') yes = false;
		if (i < n/2)
			sum += c - '0'; else
			sum -= c - '0';
	}
	if (yes && sum == 0) puts("YES");
		else puts("NO");
	return 0;
}
