#include<cstdio>
char s[256];
int val;
int main(){
	int n;
	scanf("%d%s",&n,s);
	for(int i=0;i<n;i++)
		s[i]-='0';
	for(int i=0;i<n;i++)
		if(s[i]!=4&&s[i]!=7){
			printf("NO\n");
			return 0;
		}
	for(int i=0;i<n/2;i++)
		val+=s[i]-s[n-i-1];
	if(val!=0)
		printf("NO\n");
	else
		printf("YES\n");
	return 0;
}
