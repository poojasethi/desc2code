#include <iostream>

using namespace std;

int N,check,sum;

int main()
{
	cin >> N;
	for (int i=0;i<N;i++)
	{
		char c;
		cin >> c;
		if (c!='4'&&c!='7')
		{
			check=1;
			break;
		}
		sum+=(i<N/2?c-'0':-c+'0');
	}
	if (check||sum)
		cout << "NO" << endl;
	else
		cout << "YES" << endl;
	return 0;
}
