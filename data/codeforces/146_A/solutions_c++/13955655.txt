#include<iostream>
using namespace std;
int n,i,a,b;
string s;
int main(){
cin>>n>>s;
for(i=0;i<n/2;i++){
if(s[i]!='4' && s[i]!='7')
{
cout<<"NO";
return 0;
}
a+=s[i]-48;
}
for(i=n/2;i<n;i++){
if(s[i]!='4' && s[i]!='7')
{
cout<<"NO";
return 0;
}b+=s[i]-48;}
b-=a;
if(b!=0)cout<<"NO";
else cout<<"YES";
}
