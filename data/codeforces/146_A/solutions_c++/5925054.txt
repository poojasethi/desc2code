#include<iostream>
using namespace std ;
int main()
{
  int n,HL=0,HR=0; cin>>n;
  string t;        cin>>t;
  for(int i=0;i<t.size();i++)
  {
    if(t[i]=='4'||t[i]=='7')n--;
    if(i< t.size()/2)HL+=t[i];
    if(i>=t.size()/2)HR+=t[i];
  }
  if(n==0&&HL==HR)cout<<"YES";
  else cout<<"NO";
 return 0;
}
