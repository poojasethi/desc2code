#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    string s;
    int n,i,a=0,b=0;
    cin>>n>>s;
    for(i=0;i<n/2;i++)a+=s[i];
    for(i=n/2;i<n;i++)b+=s[i];
    for(i=0;i<n;i++)if(s[i]!='4' && s[i]!='7')break;
    if(i==n && a==b)
    cout<<"YES";
    else
    cout<<"NO";
    return 0;
}
