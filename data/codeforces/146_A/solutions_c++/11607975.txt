#include <iostream>
#include <string>
using namespace std;
string s;
int s1,s2,n;
int main(){
	cin>>n>>s;
	for(int i=0;i<s.length();i++){
		if(s[i]!='7' && s[i]!='4'){
			cout<<"NO"<<endl;
			return 0;
		}
		if(i<n/2)
			s1+=s[i]-'4';
		else
			s2+=s[i]-'4';			
	}
	cout<<(s1==s2?"YES":"NO")<<endl;
	return 0;
}
