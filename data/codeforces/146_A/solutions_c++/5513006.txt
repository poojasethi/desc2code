#include <iostream>
using namespace std;
int main(){
int c=0;
int x;
int sum1=0;
int sum2=0;
char y;
cin>>x;
if(x%2!=0)return 0;
for(int i=0;i<x;i++){
cin>>y;
if(i<(x/2))sum1=sum1+y-'0';
if(i>=(x/2))sum2=sum2+y-'0';
if(y=='4'||y=='7'){
c++;}}
if(c==x&&sum1==sum2)
cout<<"YES";
else
cout<<"NO";
return 0;
}
