from sys import stdin

line = stdin.readline()
n = int(line)

d = {}

for i in xrange(0, n):
	line = stdin.readline().strip()
	ad = line.lower().split("@")
	if ad[1] == "bmail.com":
		ad[0] = ad[0].replace(".", "")
		ad[0] = ad[0].split("+")
		ad[0] = ad[0][0]
	ad = str(ad)
	if not ad in d:
		d[ad] = []
	d[ad].append(line)
print len(d)
for a in d:
	print str(len(d[a])) + " " + " ".join(d[a])
