#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<string>
#include<map>
#include<vector>
#include<cstdlib>
using namespace std;
const int N=20005;
string s[N],a[N];
int pos[N];
string work(string s)
{
	for(int i=0;i<s.length();i++)
	if(s[i]>='A'&&s[i]<='Z')
	s[i]=s[i]-'A'+'a';
	string ret="";
	for(int i=0;i<s.length();i++)
	{
		if(s[i]=='@')
		{
			for(int j=i+1;j<s.length();j++)
			{
				ret+=s[j];
			}
			break;
		}
	}
	if(ret=="bmail.com")
	{
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='.')continue;
			if(s[i]=='+')break;
			if(s[i]=='@')break;
			ret+=s[i];
		}
	}
	else
	ret=s;
	return ret;
}
bool cmp(int x,int y)
{return a[x]<a[y];}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)cin>>s[i],a[i]=work(s[i]),pos[i]=i;
	sort(pos+1,pos+n+1,cmp);
	int ans=0;
	for(int i=1;i<=n;i++)
	if(a[pos[i]]!=a[pos[i-1]])ans++;
	printf("%d\n",ans);
	for(int i=1;i<=n;i++)
	{
		int m=0;
		while(a[pos[i+m]]==a[pos[i]])m++;
		printf("%d ",m);
		for(int j=0;j<m;j++)cout<<s[pos[i+j]]<<" ";
		puts("");
		i+=m-1;
	}
}
