using namespace std;
map<string,multiset<string> >q;

void work(){
    string str,s,s1;
    cin>>str;
    s=str;
    for(int i=0;s[i];i++){
        s[i]=tolower(s[i]);
    }
    int l=s.find('@');
    if(s.substr(l)=="@bmail.com"){
        for(int i=0;i<l;i++){
            if(s[i]=='+')break;
            if(s[i]=='.')continue;
            s1.push_back(s[i]);
        }
        q[s1+s.substr(l)].insert(str);
    }
    else q[s].insert(str);
}
int main(){
    int n;
    while(cin>>n){
        q.clear();
        for(int i=0;i<n;i++){
            work();
        }
        map<string,multiset<string> >::iterator it;
        multiset<string>::iterator it1;
        cout<<q.size()<<endl;
        for(it=q.begin();it!=q.end();it++){
            cout<<(it->second).size();
            for(it1=(it->second).begin();it1!=(it->second).end();it1++){
                cout<<" "<<*it1;
            }
            cout<<endl;

        }
    }
    return 0;
}
