#include <bits/stdc++.h>
using namespace std;

string fun(string s)
{
    for(int i = 0; i < s.size(); i++)
        s[i] = tolower(s[i]);

    string u;
    string d = s.substr(s.find('@'));

    bool b = (d == "@bmail.com");

    for(int i = 0; i < s.size(); i++)
    {
        if(s[i] == '@' || s[i] == '+' && b)
            break;
        if(s[i] != '.' || !b)
            u += s[i];
    }
    //cout << u + d << endl;
    return u + d;
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie();

    long long n;
    cin >> n;
    map<string, set<string> > M;
    map<string, set<string> >::iterator itM;
    set<string>::iterator itS;

    string s;
    for(int i = 1; i <= n; i++){
        cin >> s;
        M[fun(s)].insert(s);
    }

    cout << M.size() << '\n';

    for(itM = M.begin(); itM != M.end(); itM ++)
    {
        cout << itM->second.size();
        for(itS = itM->second.begin(); itS != itM->second.end(); itS++)
            cout << ' ' << *itS;
        cout << '\n';
    }

    return 0;
}
