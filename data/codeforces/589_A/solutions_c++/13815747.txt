#include <iostream>
#include <cctype>
#include <string>

using namespace std;

map<string,vector<string> > m;

string parse(string str){
	string rpta;
	int i;
	for(i=0;i<str.length();i++){
		if(str[i]=='@') break;
	}
	i++;
	string aux;
	for(;i<str.length();i++){
		aux += tolower(str[i]);
	}
	if(aux=="bmail.com"){
		for(i=0;i<str.length();i++){
			if(str[i]=='.') continue;
			if(str[i]=='@' || str[i] == '+') break;
			rpta += tolower(str[i]);
		}
	}else{
		for(i=0;i<str.length();i++){
			if(str[i]=='@') break;
			rpta += tolower(str[i]);
		}
	}
	rpta+=aux;
	return rpta;
}
int main() {
	int n;
	string str;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>str;
		string str2 = parse(str);
		m[str2].push_back(str);
	}
	cout<<m.size()<<endl;
	for(map<string,vector<string> >::iterator it = m.begin();it!=m.end();it++){
		cout<<(it->second).size();
		for(vector<string>::iterator it2=(it->second).begin();it2!=(it->second).end();it2++){
			cout<<" "<<(*it2);
		}
		cout<<endl;
	}

}
