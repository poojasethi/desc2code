#include<cstdio>
#include<cstring>
#include<algorithm>
#include<string>
#include<iostream>
using namespace std;
int n;
string s[20005];
string mark[20005];
string mk(string s){
	for(int i=0;i<s.length();i++) if(s[i]>='A'&&s[i]<='Z'){
		s[i]=s[i]-'A'+'a';
	}
	
	string ret="";
	for(int i=0;i<s.length();i++){
		if(s[i]=='@'){
			for(int j=i;j<s.length();j++){
				ret+=s[j];
			}
			
			break;
		}
	}
	if(ret=="@bmail.com"){
		for(int i=0;i<s.length();i++){
			if(s[i]=='.') continue;
			if(s[i]=='+') break;
			if(s[i]=='@') break;
			ret+=s[i];
		}
	}else{
		ret=s;
	}
	
	return ret;
}
int pos[200005];
bool cmp(int a,int b){
	return mark[a]<mark[b];
}
int main(){
	cin>>n;
	for(int i=1;i<=n;i++) cin>>s[i];
	for(int i=1;i<=n;i++) mark[i]=mk(s[i]);
	for(int i=1;i<=n;i++) pos[i]=i;
	sort(pos+1,pos+n+1,cmp); 
	int num=1;
	for(int i=2;i<=n;i++){
		if(mark[pos[i]]!=mark[pos[i-1]]) num++;
	}
	printf("%d\n",num);
	for(int i=1;i<=n;i++){
		int m=0;
		while(mark[pos[i+m]]==mark[pos[i]]) m++;
		printf("%d",m);
		for(int j=0;j<m;j++){
			cout<<' '<<s[pos[i+j]];
		}
		puts("");
		i+=m-1;
	}
	return 0;
}
