#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <ctime>
#include <map>
#define N 100010
#define mo 1000000007
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,T,te,s,s1;
int a[N],f[N];
map<int,int> Map;
inline void work(int x) {
	int k=x;
	for (;k;k/=10) {
		int A=k%10;
		if (A!=4&&A!=7) {
			s1++;
			return;
		}
	}
	if (!Map.count(x)) Map[x]=++s;
	int A=Map[x];
	a[A]++;
}
inline int64 pow(int64 x,int y) {
	int64 an=1;
	for (;y;y>>=1) {
		if (y&1) an=an*x%mo;
		x=x*x%mo;
	}
	return an;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) {
		scanf("%d",&k);
		work(k);
	}
	f[0]=1; For(i,0,m-1) if (i+1<=s1) f[i+1]=1ll*f[i]*(s1-i)%mo*pow(i+1,mo-2)%mo;
	For(i,1,s) {
		for (j=m;j;j--) f[j]=(1ll*f[j-1]*a[i]+f[j])%mo;
	}
	printf("%d\n",f[m]);
	//for(;;);
	return 0;
}
