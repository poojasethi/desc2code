#include<cstdio>
#include<map>
#include<cstring>
using namespace std;
bool lucky(int num)
{
    while(num>0)
    {
        if(num%10!=4&&num%10!=7)return 0;
        num/=10;
    }
    return 1;
}
typedef long long ll;
ll jie[100010];
#define mod 1000000007
ll pow(ll a,ll p)
{
    if(p==0)return 1;
    ll ret=pow(a,p>>1);
    ret=ret*ret%mod;
    if(p&1)ret=ret*a%mod;
    return ret;
}
int use[1050];
ll aa[100010];
int main()
{
    jie[0]=1;
    for(int i=1;i<=100000;i++)jie[i]=jie[i-1]*i%mod;
    memset(use,0,sizeof(use));
    int n,k;
    scanf("%d%d",&n,&k);
    int scnt=1;
    map<int,int>list;
    for(int i=0;i<n;i++)
    {
        int tmp;
        scanf("%d",&tmp);
        if(lucky(tmp))
        {
            if(list[tmp]==0)list[tmp]=scnt++;
            use[list[tmp]]++;
        }
        else use[0]++;
    }
    aa[0]=1;
    for(int j=1;j<scnt;j++)
    for(int i=k;i>0;i--)
    {
        aa[i]=aa[i]+aa[i-1]*use[j]%mod;
        if(aa[i]>=mod)aa[i]-=mod;
    }
    ll ret=0;
    for(int i=0;i<=use[0]&&i<=k;i++)
    {
        //printf("%d %d\n",k-i,aa[k-i]);
        ret+=aa[k-i]*jie[use[0]]%mod*pow(jie[i],mod-2)%mod*pow(jie[use[0]-i],mod-2)%mod;
        if(ret>=mod)ret-=mod;
    }
    printf("%I64d\n",ret);
}
