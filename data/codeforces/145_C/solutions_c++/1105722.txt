#include <cstdio>
#include <iostream>
#include <map>

using namespace std;

long P[200005]={0};
long vP[200005]={0};
long f[2005][2005]={0};
map<long,long> have;
const long mod=1000000007;

long mul(long x,long y)
{
 long a=1;
 for(;y;y>>=1,x=(long long)x*x%mod)
   if(y&1)
    a=(long long)a*x%mod;
 return a;
}

long C(long m,long n)
{
 return (long long)P[m]*vP[n]%mod*(long long)vP[m-n]%mod;
}

int main()
{
 long n,k,i;
 long s=0;
 long ans=0;
 
 scanf("%ld%ld",&n,&k);
 P[0]=1;
 vP[0]=1;
 for(i=1;i<=n;i++)
   {
    P[i]=(long long)P[i-1]*i%mod;
    vP[i]=mul(P[i],mod-2);
   }
 for(i=1;i<=n;i++)
   {
    long x,y;
    scanf("%ld",&x);
    for(y=x;y;y/=10)
      if(y%10!=4&&y%10!=7)
       break;
    if(y)
     s++;
    else
     have[x]++;
   }
 f[0][0]=1;
 i=1;
 for(map<long,long>::iterator p=have.begin();p!=have.end();p++,i++)
   {
    f[i][0]=f[i-1][0];
    for(long j=1;j<=i;j++)
      {
       f[i][j]=((long long)f[i-1][j]+(long long)f[i-1][j-1]*(p->second)%mod)%mod;
      }
   }
 i--;
 
 for(long j=0;j<=i&&j<=k;j++)
   {
    ans+=(long long)f[i][j]*C(s,k-j)%mod;
    ans%=mod;
   }
 printf("%ld\n",ans);
 
 return 0;
}
