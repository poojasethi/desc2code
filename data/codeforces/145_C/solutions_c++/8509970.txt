#include<cstdio>
#include<map>
using namespace std;
map<int,long long>mp;
map<int,long long>::iterator it;
int n,x1,m,cnt,ss;
inline bool judge(int x)
{
	do
	{
		if(x%10!=4&&x%10!=7)return false;
	}while (x/=10);
	return true;
}
const long long mod=1000000007;
inline long long sqr(long long x)
{
	return x*x%mod;
}
inline long long exp(long long x,long long y)
{
	if(y==0)return 1;
	if(y==1)return x;
	if(y%2==1)return sqr(exp(x,y/2))*x%mod;
	return sqr(exp(x,y/2));
}
long long jiec[120000],anti[120000];
inline long long choose(int x,int y)
{
//	if(x<=0)return 1;
	long long p=jiec[x];
	p*=anti[y];
	p%=mod;
	p*=anti[x-y];
	p%=mod;
	return p;
}
long long dp[1200],has[1200];
int main()
{
	scanf("%d %d",&n,&m);
	jiec[0]=1;
	anti[0]=1;
	for (int i=1;i<=n;i++)
	{
		jiec[i]=jiec[i-1]*i;
		jiec[i]%=mod;
		anti[i]=exp(jiec[i],mod-2);
		scanf("%d",&x1);
		if(judge(x1))mp[x1]++;
	}
	for (it=mp.begin();it!=mp.end();it++)
	{
		cnt++;
		has[cnt]=it->second;
		ss+=has[cnt];
	}
	dp[0]=1;
	for (int i=1;i<=cnt;i++)
	{
		for (int j=i;j>=1;j--)
		{
			dp[j]+=dp[j-1]*has[i];
			dp[j]%=mod;
		}
	}
	long long ans=0;
	for (int i=0;i<=cnt;i++)
	{
		if(i>m||n-ss<m-i)continue;
		ans+=(choose(n-ss,m-i)*dp[i])%mod;
		ans%=mod;
	}printf("%I64d\n",ans);
	return 0;
}
