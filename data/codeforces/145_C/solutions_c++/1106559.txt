#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
const int mod=1000000007;
int last[1100000],next[110000],toit[110000],cnt[110000];
int tot,n,K,a,g,ans;
int f[2000][2000];
int C[110000];
inline bool check(int a)
{
	for (;a;a/=10)
		if (a%10!=4&&a%10!=7)
			return 0;
	return 1;
}
inline void add(int a)
{
	for (int i=last[a%1000007];i;i=next[i])
		if (toit[i]==a)
		{
			++cnt[i];
			return;
		}
	toit[++tot]=a;
	cnt[tot]=1;
	next[tot]=last[a%1000007];
	last[a%1000007]=tot;
}
inline int mul(int a,int b)
{
	int x=1;
	for (;b;b/=2,a=a*(long long)a%mod)
		if (b&1)
			x=x*(long long)a%mod;
	return x;
}
int main()
{
	scanf("%d%d",&n,&K);
	for (int i=1;i<=n;++i)
	{
		scanf("%d",&a);
		if (check(a)) add(a);	
		else g++;
	}
	f[0][0]=1;
	for (int i=1;i<=tot;++i)
	{
		f[i][0]=1;
		for (int j=1;j<=i;++j)
			f[i][j]=(f[i-1][j]+f[i-1][j-1]*(long long)cnt[i])%mod;
	}
	C[0]=1;
	for (int i=1;i<=g;++i)
		C[i]=C[i-1]*(long long)(g-i+1)%mod*(long long)mul(i,mod-2)%mod;
	for (int i=0;i<=K&&i<=g;++i)
		if (K-i<=tot)
			ans=(ans+f[tot][K-i]*(long long)C[i])%mod;
	printf("%d\n",ans);
	//system("pause");
}
