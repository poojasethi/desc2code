#include<stdio.h>
#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
#define maxl 1025
#define maxn 100010
#define m0 1000000007LL
int remain,ll,n,k,i,j,a[maxn],kl;
map<long long,int>ma;
long long lu[maxl],f[maxl][maxl],fac[maxn],cur,res;
void gen(long long q){
	if(q>1000000000)return;
	gen(q*10LL+4);
	gen(q*10LL+7);
	lu[++ll]=ma[q];
	remain-=ma[q];
}
long long divide(long long a,long long b){
	if(b==0)return 1;
	long long y=m0-2,s=b,p=1;
	while(y){
		if(y%2==1)p=(p*s)%m0;
		s=(s*s)%m0;
		y/=2;
	}
	return (a*p)%m0;
}
long long C(int n,int k){
	if(n<k)return 0;
	return divide(fac[n],(fac[k]*fac[n-k])%m0);
}
int main(){
//	freopen("in.txt","r",stdin);
//	freopen("out.txt","w",stdout);
	scanf("%d%d",&n,&k);remain=n;
	fac[0]=1;
	for(i=1;i<=n;i++)fac[i]=(fac[i-1]*i)%m0;
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++)++ma[a[i]];
	gen(0);
	f[0][0]=1;
	for(i=0;i<ll;i++)for(j=0;j<=i;j++){
		f[i+1][j]=(f[i][j]+f[i+1][j])%m0;
		f[i+1][j+1]=(f[i+1][j+1]+f[i][j]*lu[i+1])%m0;
	}
	for(kl=0;kl<=min(k,ll);kl++){
		cur=f[ll][kl]*C(remain,k-kl);
		res=(res+cur)%m0;
	}
	cout<<res<<endl;
	return 0;
}
