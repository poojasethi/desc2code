#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<map>
typedef long long ll;
using namespace std;
map<int,int>ma;
int a[111111];
ll mod=1000000007,dp[111111],fac[111111];

bool f(int k)
{
	if(k==0)return true;
	if(k%10!=4 && k%10!=7)return false;
	return f(k/10);
}

ll x(ll a)
{
	int i=mod-2;
	ll ans=1;
	//printf("x a=%I64d",a);
	while(i)
	{
		if(i&1)ans=ans*a%mod;
		a=a*a%mod;
		i=i>>1;
	}
	//printf(" ans=%I64d\n",ans);
	return ans;
}

ll c(int n,int m)
{
	if(n<m)return 0;
	//printf("c n=%d m=%d fac[n]=%I64d fac[m]=%I64d fac[n-m]=%I64d\n",n,m,fac[n],fac[m],fac[n-m]);
	return (fac[n]*x(fac[m])%mod)*x(fac[n-m])%mod;
}

int main()
{
	int i,j,k,m,n;
	ll ans=0,kk,s=0;
	scanf("%d%d",&n,&m);
	dp[0]=1;
	fac[0]=1;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		ma[a[i]]++;
	}
	for(i=1;i<=n;i++)
	{
		if(f(a[i]) && ma[a[i]])
		{
			kk=ma[a[i]];
			ma[a[i]]=0;
			for(j=m;j;j--)
			{
				dp[j]=(dp[j]+dp[j-1]*kk)%mod;
			}
		}
		else if(!f(a[i]))s++;
		kk=i;
		fac[i]=fac[i-1]*kk%mod;
	}
	/*for(i=0;i<=m;i++)
	{
		printf("i=%d fac[i]=%I64d dp[i]=%I64d\n",i,fac[i],dp[i]);
	}*/
	for(i=0;i<=m;i++)
	{
		ans=(ans+dp[i]*c(s,m-i))%mod;
	}
	cout<<ans<<endl;
	return 0;
}
