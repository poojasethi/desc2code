#include <cstdio>
#include <algorithm>
#define N 1050
#define M 100005
#define B 1000000007
#define L __int64
using namespace std;

int n, m, k, c, x[M], e[N];
L A, d[N], r[M], f;

bool I(int x){
	for(; x; x /= 10) if(!(x % 10 == 4 || x % 10 == 7)) return 0;
	return 1;
}

L P(L a){
	L x = 1;
	for(L b = B - 2; b; b /= 2){
		if(b & 1) x = x * a % B;
		a = a * a % B;
	}
	return x;
}

int main(){
	scanf("%d %d", &n, &m);
	for(int i = 0; i < n; i++) scanf("%d", &x[i]);
	sort(x, x + n);
	
	k = 1;
	for(int i = 0; i < n; i++){
		if(I(x[i])){
			for(e[k] = 1; i < n - 1 && x[i] == x[i + 1]; i++, e[k]++);
			k++;
		}else c++;
	}
	
	for(int i = f = r[0] = 1; i <= c; i++){
		f = f * i % B;
		r[i] = P(f);
	}
	
	for(int i = d[0] = 1; i <= k; i++)
		for(int j = min(m, k); j; j--)
			d[j] = (d[j - 1] * e[i] + d[j]) % B;
			
	for(int i = 0; i <= min(m, c); i++) if(m - i <= k)
		A = (A + d[m - i] * f % B * r[i] % B * r[c - i]) % B;
	
	printf("%I64d\n", A);
}
