#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long ll;
const int mod=1000000007;

int a[100010],f[100010],b[100010];

int power(int x,int y)
{
	int t=1;
	while (y)
	{
		if (y&1) t=ll(t)*x%mod;
		y>>=1;x=ll(x)*x%mod;
	}
	return t;
}

bool test(int x)
{
	while (x)
	{
		if (x%10!=4&&x%10!=7) return 0;
		x/=10;
	}
	return 1;
}

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int n,k;scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	bool ok=test(a[1]);int l=1,cnt=0,s=n;
	for (int i=2;i<=n+1;i++)
		if (i>n||a[i]!=a[i-1])
		{
			if (ok)
			{
				b[++cnt]=i-l;s-=i-l;
			}
			l=i;ok=test(a[i]);
		}	
	f[0]=1;for (int i=1;i<=s&&i<=k;i++) f[i]=ll(f[i-1])*(s-i+1)%mod*power(i,mod-2)%mod;
	for (int j=1;j<=cnt;j++)
		for (int i=k;i;i--)
			f[i]=(ll(f[i-1])*b[j]+f[i])%mod;
	printf("%d\n",f[k]);
	return 0;
}
