M=10**9+7
def lucky(x):
  t=x
  while x:
    d=x%10
    if d!=4 and d!=7:
      return 0
    x/=10
  return t
R=lambda:map(int,raw_input().split())
n,k=R()
a,b={},0
for x in R():
  t=lucky(x)
  if t:a[t]=a.get(t,0)+1
  else:b+=1
w=[1]+[0]*100100
for i,x in enumerate(a.values()):
  for j in range(min(k,i),-1,-1):
    w[j+1]=(w[j+1]+w[j]*x)%M
s,c=0,1
for i in range(min(k,b)+1):
  s=(s+c*w[k-i])%M
  c=c*(b-i)%M*pow(i+1,M-2,M)%M
print s
