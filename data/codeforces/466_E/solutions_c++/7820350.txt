#include <cstdio>
#include <vector>
#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

const int maxn = (int)1e5 + 10;;

int b[maxn], p[maxn];
vector<int> dd, pp;
int n, m;

int find(int x) {
    return x == p[x] ? x : p[x] = find(p[x]);
}

int main() {
    #ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    #endif
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++)
        p[i] = i;
    for (int i = 0; i < m; i++) {
        int t;
        scanf("%d", &t);
        if (t == 1) {
            int x, y;
            scanf("%d%d", &x, &y);
            b[x] = y;
            p[x] = y;
        } else if (t == 2) {
            int x;
            scanf("%d", &x);
            dd.push_back(x);
            pp.push_back(find(x));
        } else {
            int x, i;
            scanf("%d%d", &x, &i);
            int y = dd[i - 1], z = pp[i - 1];
            while (y != z) {
                if (y == x) break;
                y = b[y];
            }
            if(y == x)
                puts("YES");
            else
                puts("NO");
        }
    }
    return 0;
}
