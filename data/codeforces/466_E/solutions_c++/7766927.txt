#include <iostream>
#include <vector>
#include <cstdio>
#include <set>
#include <map>
#include <fstream>
#include <algorithm>

using namespace std;

int n, m;
int apa[100001];
int apas[100001];
int ido=0;
int docu[100001];

int main()
{
    //freopen("E.txt", "r", stdin);
    scanf("%d%d", &n, &m);
    for (int i=1;i<=n;i++) apa[i]=-1;
    for (int i=1;i<=m;i++) {
        int tip;
        int x, y;
        scanf("%d", &tip);
        if (tip==1) {
            scanf("%d%d", &x, &y);
            apa[x]=y;
            apas[x]=ido;
        }
        if (tip==2) {
            scanf("%d", &x);
            ido++;
            docu[ido]=x;
        }
        if (tip==3) {
            scanf("%d%d", &x, &y);
            //x alairta y-adikat
            int akt=docu[y];
            bool b=false;
            if (akt==x) b=true;
            while (!b && apa[akt]!=-1 && apas[akt]<y) {
                akt=apa[akt];
                if (akt==x) b=true;
            }
            if (b) {
                printf("YES\n");
            }
            else {
                printf("NO\n");
            }
        }
    }
    return 0;
}
