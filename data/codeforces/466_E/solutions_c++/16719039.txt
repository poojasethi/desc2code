#include <bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define endl '\n'
using namespace std; 
typedef long long ll;
typedef pair <int,int> pii;

const int maxN = 1e5 + 10;
int root[maxN],s[maxN],e[maxN],par[maxN];

int get_root(int v)
{
    return (root[v] == 0 ? v : root[v] = get_root(root[v]));
}

int main()
{
    ios_base::sync_with_stdio(false);    
    int n,m,num = 1;
    cin >> n >> m;
    while (m--)
    {
        int t;
        cin >> t;
        if (t == 1)
        {
            int v,u;
            cin >> v >> u;
            par[v] = u;
            root[v] = u;
        }
        
        if (t == 2)
        {
            int v;
            cin >> v;
            s[num] = v; //start of document num
            e[num] = get_root(v); //end of document num
            num++;
        }

        if (t == 3)
        {
            int v,x;
            cin >> v >> x;
            for (int u = s[x] ; ; u = par[u])
            {
                if (u == v)
                {
                    cout << "YES" << endl;
                    break;
                }
                else
                if (u == e[x])
                {
                    cout << "NO" << endl;
                    break;
                }
            }
        }
    }
    return 0;
}
