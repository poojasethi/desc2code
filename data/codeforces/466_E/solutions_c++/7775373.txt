pair<int,int> boss[MAXN],pacco[MAXN];
int N,M,op,A,B,conta;

inline bool dfs(int v,int t)
{
    //cout<<"V: "<<v<<" t: "<<t;
    if(v==A) return true;
    if(boss[v].first==0 || boss[v].second>t) return false;
    return dfs(boss[v].first,t);
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin>>N>>M;
    conta=1;
    for(int i=0;i<M;++i)
    {
        cin>>op;
        if(op==1)
        {
            cin>>A>>B;
            boss[A]=make_pair(B,i);
        }
        else if(op==2)
        {
            cin>>A;
            pacco[conta]=make_pair(A,i);
            conta++;
        }
        else
        {
            cin>>A>>B;
            if(dfs(pacco[B].first,pacco[B].second)) cout<<"YES\n";
            else cout<<"NO\n";
        }
    }

    return 0;
}
