#include<cstdio>
#include<cstring>
#include <vector>
#include <algorithm>
#define MAXN 100005
using namespace std;

vector<pair<int,int> > v;
int boss[MAXN], date[MAXN], doc[MAXN], rec[MAXN];

int main()
{
	int n,m,t,x,y,d=1,a,b;
	scanf("%d %d",&n,&m);
	for(int i=1;i<=n;i++)
    {
		boss[i]=-1;
    }
	for(int i=0;i<m;i++)
    {
		scanf("%d",&t);
		if(t==1)
		{
            scanf("%d %d",&x,&y);
            boss[x]=y;
            date[x]=i;
		}
        else if(t==2)
        {
            scanf("%d",&x);
            doc[d]=x;
            rec[d++]=i;
        }
        else if(t==3)
        {
            scanf("%d %d",&x,&y);
            v.push_back(make_pair(x,y));
        }
	}

	for(int i=0;i<v.size();i++)
    {
		a = doc[v[i].second];
		b = v[i].first;
		while(a!=-1&&a!=b&&date[a]<rec[v[i].second])
        {
			a=boss[a];
		}
		if(a==b)
			printf("YES\n");
		else
			printf("NO\n");
	}
	scanf(" ");
}
