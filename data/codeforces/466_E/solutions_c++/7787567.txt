#include <bits/stdc++.h>

using namespace std;
#define ll long long
int n,m;
const int maxn = 100*1000+19;
int par[maxn]={0};
int timer[maxn]={0};
int doctime[maxn]={0};
int lev[maxn]={0};
vector<int> docs[maxn];
int doc2[maxn]={0};
int main()
{
    cin>>n>>m;
    int did = 0;
    for(int i=1;i<=m;i++)
    {
        int op;
        cin>>op;
        if(op==1)
        {
            int x,y;
            cin>>x>>y;
            par[x]=y;
            timer[x] = i;
        }
        else if(op==2)
        {
            did++;
            int x;
            cin>>x;
            doc2[did]=x;
            doctime[did] = i;
        }
        else
        {
            int x,d;
            cin>>x>>d;
            int y = doc2[d];

            while(1)
            {
                if(y==x)
                {

                    break;
                }

                if(par[y] ==0 || timer[ y ] >doctime[d]   ) break;
                else y=par[y];

            }
            if(y == x)
            {
                cout<<"YES\n";
            }
            else
            {
                cout<<"NO\n";
            }
        }
    }
    return 0;
}
