import sys
from itertools import product

file = sys.stdin
#file = open("test", "r")
cnt = int(file.readline().rstrip())
visit = [0 for _ in range(cnt)]
i = 0
out = []
while i < cnt and not visit[i]:
    out.append(i)
    visit[i] = 1
    if not visit[cnt - 1 - i]: out.append(cnt - 1 - i); visit[cnt - 1 - i] = 1
    i += 1
print "".join(map(lambda i : str(i + 1) + " ", out))
