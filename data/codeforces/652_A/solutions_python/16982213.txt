import math

h1, h2 = tuple([int(x) for x in raw_input().split()])
a, b = tuple([int(x) for x in raw_input().split()])

def solve(start, end, a, b):
    if b >= a and start + 8 * a < end: return -1
    diff = end - (start + 8 * a)
    if diff <= 0: return 0
    else:
        denorm = 12 * (a - b)
        if diff % denorm == 0: return diff / denorm
        else: return 1 + diff / denorm

print solve(h1, h2, a, b)
