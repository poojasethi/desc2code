import sys
a=sys.stdin.readline().rstrip().split()
b=sys.stdin.readline().rstrip().split()
# a=[1,50]
# b=[5,4]
cater=int(a[0])
apple=int(a[1])
night=int(b[1])
day=int(b[0])
cycle=(day-night)*12
height=apple-cater

if(height<=8*day):
    print 0
elif(cycle<=0):
    print -1
else:
    temp=(height-8*day)/cycle
    if temp*cycle+8*day<height:
        print temp+1
    else:
        print temp
