h1,h2 = map(int,raw_input().split())
a,b =  map(int,raw_input().split())

if h1+8*a < h2 and a<=b:
    print -1
else:
    start = h1 + 8*a
    if start >= h2:
        print 0
    else:
        advance_per_day = 12 *(a-b)
        if (h2-start) % advance_per_day == 0:
            print (h2-start) / advance_per_day
        else:
            print ((h2-start)/advance_per_day)+1
