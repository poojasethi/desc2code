h1, h2 = map(int, raw_input().split())
a, b = map(int, raw_input().split())

dh = h2 - h1

first_day_len = 8

if a * first_day_len >= dh:
	print 0
	exit()
	
if a <= b:
	print -1
	exit()
	
hour = 14
days = 0

while dh > 0:
	if hour >= 10 and hour < 22:
		dh -= a
	else:
		dh += b
	
	if dh <= 0:
		print days
		exit()
	
	hour += 1
	if hour == 24:
		hour = 0
		days += 1
