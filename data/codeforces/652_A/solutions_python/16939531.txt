h1, h2 = [int(i) for i in raw_input().split()]
a, b = [int(i) for i in raw_input().split()]

dd1 = a*4.0
dd2 = a*8.0
dn = b*12.0

# print "h1:", h1
wait = 0
while True:
    h1 += dd2
    # print "h1:", h1

    if h1 >= h2:
        print wait
        break

    if dd1 + dd2 <= dn:
        print -1
        break

    h1 -= dn
    # print "h1:", h1

    h1 += dd1
    # print "h1:", h1
    wait += 1
    # print "wait:", wait

    if h1 >= h2:
        print wait
        break
