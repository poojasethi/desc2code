line1 = raw_input()
line2 = raw_input()

cat = int(line1.split()[0])
app = int(line1.split()[1])

up = int(line2.split()[0])
slip = int(line2.split()[1])

if cat + 8 * up >= app:
    print 0
elif up > slip:
    # distance between bug and apple
    diff = app - cat 
    x = diff - 8 * up 
    y = 12 * ( up - slip) 
    print (x + y - 1) / y 
else:
    print -1
