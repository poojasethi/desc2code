
cater, apple = map(int, raw_input().split())
day, night = map(int, raw_input().split())

start = cater
time = 3 
daycount = 0
daytime = True
count = 0
while apple > cater:
    count += 1
    time += 1
    if time % 12 == 0:
        daytime = not daytime
        if daytime == False:
            daycount += 1
    if daytime:
        cater += day
    else:
        cater -= night
    if time % 1000 == 0 and cater <= start:
        print -1
        exit(0)

print daycount
