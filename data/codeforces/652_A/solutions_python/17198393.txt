def solve(h1, h2, a, b):
    ans = 0
    h1 += 8 * a

    if h1 >= h2:
	return ans
    h1 -= 12 * b
    prev = h1
    while h1 < h2:
	ans += 1
	h1 += a * 12
	if h1 >= h2:
	    break
	else:
	    h1 -= b * 12
	if h1 <= prev:
	    ans = -1
	    break
    return ans

h1, h2 = map(int, raw_input().split(' '))
a, b = map(int, raw_input().split(' '))

print solve(h1,h2, a, b)
