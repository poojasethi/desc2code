from copy import deepcopy
import math
def input_list():
    i = raw_input()
    return map(eval, i.split(' '))

def main():
    [h1, h2] = input_list()
    target = h2 - h1
    [a, b] = input_list()
    if a <= b and 8*a < target:
        print (-1)
        return
    pos = 8*a
    k = 0
    while pos < target:
        k += 1
        pos += (a-b) * 12
    print k

main()
