h = raw_input().split()
h1 = int(h[0])
h2 = int(h[1])
s = raw_input().split()
a = int(s[0])
b = int(s[1])

if h1 + a * 8 >= h2:
    print 0
elif a * 12 <= b * 12:
    print -1
else:
    count = 0
    h = h1 + a * 8
    while h < h2:
        h = h - b * 12 + a * 12
        count = count + 1
    print count
