#include<iostream>
#include<cmath>

int main()
	{
		double i;
		double a,b;
		double p,q;
		int count=0;
		std::cin>>p>>q;
		std::cin>>a>>b;
		p=p+8*a;
		if(p>=q)
			std::cout<<0;
		else if(b>=a)
			std::cout<<-1;
		else
			{
				for(i=p;i<q;i+=12*(a-b))
					count++;
				std::cout<<count;
			}
		return 0;
	}
