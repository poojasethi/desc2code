#include<stdio.h>
int main()
{
	int i,l,m,n,x,y,a;
	while(scanf("%d%d%d%d",&m,&n,&x,&y)!=EOF)
	{
		a=n-m;
		if(a-x*8>0)
		{
			if(x<=y)
				printf("-1\n");
			else
			{
				a=a-x*8;
				for(i=1;a-12*x*i+12*y*i>0;i++)
					;
				printf("%d\n",i);
			}
		}
		else
			printf("0\n");
	}
}
