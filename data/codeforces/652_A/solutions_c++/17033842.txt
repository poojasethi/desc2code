#include<stdio.h>
int main()
{
	int h1,h2,a,b;
	scanf("%d%d%d%d",&h1,&h2,&a,&b);
	if(h1+a*8<h2&&a<=b)printf("-1\n");
	else{
		h1+=a*8;
		int day=0;
		for(;;){
			if(h1>=h2){
				printf("%d\n",day);
				break;
			}
			h1+=a*12;
			h1-=b*12;
			day++;
		}
	}
	return 0;
}
