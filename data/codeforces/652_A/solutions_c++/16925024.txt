#include <bits/stdc++.h>
using namespace std;

int main(){
	int h1,h2,a,b;
	cin>>h1>>h2>>a>>b;
	int d=h2-h1-(22-14)*a;
	if(d<=0){
		cout<<0;
		return 0;
	}else if(a<=b){
		cout<<-1;
		return 0;
	}
	d+=b*12;
	for(int i=1;;i++){
		//day
		d-=a*12;
		if(d<=0){
			cout<<i;
			return 0;
		}
		//night
		d+=b*12;
	}
	return 0;
}
