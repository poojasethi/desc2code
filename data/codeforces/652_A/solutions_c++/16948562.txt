#include <bits/stdc++.h>
 
using namespace std;

int main()
{	std::ios_base::sync_with_stdio(false);
	int h1,h2,a,b;
	cin>>h1>>h2>>a>>b;
	if(a<=b && h1+8*a<h2)cout<<-1<<endl;
	else if((h1+8*a)>=h2)cout<<0<<endl;
	else 
	{	double todo=h2-h1-8*a,diff=a-b;
		int ans=ceil(todo/(12.0*diff));
		cout<<ans<<endl;
	}
}
