#include<stdio.h>
#include<stdlib.h>
int main()
{
	int h1,h2,a,b,k;
	while(scanf("%d%d%d%d",&h1,&h2,&a,&b)!=EOF){
		k=0;
		h1+=8*a;
		if(h1>=h2) printf("%d\n",k);
		else if(a<=b) printf("-1\n");
		else {	
			while(h1<h2){
				h1-=b*12;
				h1+=a*12;
				k++;
			}
			printf("%d\n",k);
		}
	}
	return 0;
}
