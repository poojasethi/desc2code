#include <iostream>
using namespace std;
 
int main() 
{
	int h1,h2,a,b;
	cin >> h1 >> h2 >> a >> b;
 
	h1 += 8*a;
	
	
	int i = 0;
	while( (a > b) || (a <= b && i < 3) )
	{
		if(h1>= h2)
		{
			cout << i << endl;
			return 0;
		}
		h1 -= 12*b;
		i++;
		h1 += 12*a;
 
	}
	cout << -1 << endl;
	return 0;
}
