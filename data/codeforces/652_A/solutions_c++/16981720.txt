#include "cstdio"
using namespace std;
int main()
{
    int h1, h2, a, b;
    scanf("%d%d%d%d", &h1, &h2, &a, &b);
    if (h1+8*a>=h2)
        printf("0");
    else if (a<=b)
        printf("-1");
    else
    {
        int x=h2-h1-8*a;
        int y=12*(a-b);
        printf("%d", (x+y-1)/y);
    }
    return 0; 
}
