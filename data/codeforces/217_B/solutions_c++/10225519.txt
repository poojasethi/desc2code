pair<int,int> chk(int a,int b)
{
	if(b<=1)
		return make_pair(a*b-1,a*b-1);//error,len
	pair<int,int> x;
	x=chk(b,a%b);
	if(x.first<0)
		return x;
	x.first+=(a/b-1);
	x.second+=a/b;
	return x;
}
