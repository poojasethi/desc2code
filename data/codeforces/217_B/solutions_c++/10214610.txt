#include<cstdio>
#define MAXN 2000000
int ansmin=99999999;
int ans[MAXN+10],ans1[MAXN+10];
int n;
void work(int x,int y)
{
	int cnt=0;
	while(x!=y)
	{
		if(cnt>n-1) return ;
		if(x>y)
		{
			ans1[++cnt]=0;
			x-=y;
		}
		else 
		{
			ans1[++cnt]=1;
			y-=x;
		}
	}
	if(x!=1||cnt!=n-1) return ;
	ans1[n]=0;
	int c=0;
	int i,j;
	for(i=1;i<n;i++)
		if(ans1[i]==ans1[i+1]) c++;
    if(c<ansmin)
	{
		ansmin=c;
		for(i=1,j=n;i<=n&&j>=1;i++,j--)
			ans[i]=ans1[j];
	}
	return ;
}
int main()
{
	int r,i;
	scanf("%d%d",&n,&r);
	for(i=1;i<=r;i++)
	{
		work(i,r);
		work(r,i);
	}
	if(ansmin==99999999)
	{
		printf("IMPOSSIBLE\n");
		return 0;
	}
	printf("%d\n",ansmin);
	for(i=1;i<=n;i++)
		if(!ans[i]) printf("T");
		else printf("B");
	printf("\n");
}
