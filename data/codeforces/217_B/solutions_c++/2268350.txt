#include<cstdio>
#include<cstring>

using namespace std;

int		n,r,ans;
char	s1[1000005],s[1000005];
void	updata(int x ,int y ) {
	int len = n;
	int gg = x, hh = y;
	while ( len-- ) {
		if ( x >= y ) { x -= y; s[len] = 'T'; }
		else { y -= x; s[len] = 'B'; }
		if ( x == 0 || y == 0 ) break;
	}
	if ( len ) return;
	if ( x != 0 || y != 1 || s[0] != 'T' ) return;
	int now = 0;
	for (int i = 1; i<n; i+=1) now += s[i] == s[i-1];
	if ( now <= ans ) {
		ans = now;
		for (int i = 0; i<n; i++) s1[i] = s[i];
		s1[n] = 0;
	}
}
int		main(){
	scanf("%d%d", &n, &r );
	if ( r == 1 && n == 1 ) { printf("0\nT\n"); return 0; }
	ans = 0x7fffffff;
	for (int i = 1; i<=r; i++ ) {
		updata( i , r );
		updata( r , i );
	}
	if ( ans > n ) printf("IMPOSSIBLE\n");
	else printf("%d\n%s\n" , ans , s1 );
	return 0;
}
