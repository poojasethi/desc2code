#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<memory>
#include<algorithm>
#include<cstring>
using namespace std;
const int inf = 1000000000;
const int maxn = 1000005;
int n,r;

char str[maxn],now[maxn];
int ans;
void solve(int t,int b){
    int k = 0;
    while(t != b){
        if(t > b) t -= b, now[k++] = 'T';
        else b -= t, now[k++] = 'B';
    }
    now[k++] = 'T'; now[k] = 0;
    if(t!=1 || k!=n) return ;
    
    int cnt = 0,i;
    for(i=1;i<k;++i){
        if(now[i] == now[i-1]) ++cnt;
	}
	if(cnt < ans){
		ans = cnt;
		strcpy(str,now);
	}
}

int main(){
	int i;
	scanf("%d %d",&n,&r);
	ans = inf;
	for(i=1;i<=r;++i){
		solve(i,r);
		solve(r,i);
	}
	if(ans != inf){
		printf("%d\n",ans);
		strrev(str);
		puts(str);
	}else puts("IMPOSSIBLE");
    return 0;
}
