#include <stdio.h>
#include <string.h>

#define MAX_N 1000001

int n, r, res = MAX_N, cnt;
char ans[MAX_N], tmp[MAX_N];

int main()
{
    int i, x, y, k;
    scanf("%d %d", &n, &r);
    for (i = 1; i <= r; i++)
    {
	cnt = 0, x = r, y = i, k = n, tmp[k] = 0;
	while (x > 0 && y > 0 && x + y > 0 && k)
	{
	    if (x >= y)
	    {
		x -= y;
		tmp[--k] = 'T';
	    }
	    else
	    {
		y -= x;
		tmp[--k] = 'B';
	    }
	    if (tmp[k] == tmp[k + 1])
		cnt += 1;
	}
	if (tmp [n - 1] == tmp[n - 2])
	{
	    tmp[n - 1] = 'T' + 'B' - tmp[n - 1];
	    cnt -= 1;
	}
	if (!k && x == 0 && y == 1 && cnt < res)
	{
	    res = cnt;
	    if (tmp[0] == 'B')
	    {
		for (k = 0; tmp[k]; k++)
		    ans[k] = 'T' + 'B' - tmp[k];
	    }
	    else
	    {
		strcpy(ans, tmp);
	    }
	}
    }
    if (res < MAX_N) printf("%d\n%s\n", res, ans);
    else printf("IMPOSSIBLE\n");
    return 0;
}
