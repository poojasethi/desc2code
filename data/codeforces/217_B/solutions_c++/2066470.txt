#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
int n,r,ans;
char a[1000001],b[1000001];
void go(int x,int y){
	int ret=0,len=0;
	while(x!=y)
		if(x>y)
			b[len++]='T',x-=y;
		else
			b[len++]='B',y-=x;
	if(len!=n-1 || x!=1) return;
	b[len]='T';
	for(int i=1 ; i<n ; i++)
		if(b[i]==b[i-1])ret++;
	if(ret<ans){
		ans=ret;
		for(int i=0 ; i<n ; i++) a[i]=b[i];
	}
}
int main(){
	ios::sync_with_stdio(false);
	cin >> n >> r;
	ans=n;
	for(int i=1 ; i<=r ; i++)
		go(i,r),go(r,i);
	if(ans==n)
		cout << "IMPOSSIBLE" << endl;
	else{
		cout << ans << endl;
		for(int i=n-1 ; i>=0 ; i--)
			cout << a[i];
		cout << endl;
	}
	return 0;
}
