#include <iostream>
#include <cstring>
using namespace std;

#define NM 1000100
int n, mini = NM, len;
char tmp[NM], ans[NM];

void check(int x, int y) {
  int i, cost = 0;
  for (i = 0; x > 0 && y > 0 && i < n; i++) {
    if (x >= y) tmp[i] = 'T', x -= y;
    else        tmp[i] = 'B', y -= x;
    if (i > 0 && tmp[i] == tmp[i-1]) cost++;
  }
  if (i != n || x != 0 || y != 1) return;
  if (cost < mini) {
    mini = cost;
    memcpy(ans, tmp, len = i);
  }
}

int main() {
  int r; cin >> n >> r;
  for (int i = 0; i <= r; i++) {
    check(i, r); check(r, i);
  }
  if (mini != NM) {
    cout << mini << endl;
    for (int i = len - 1; i >= 0; i--) cout << ans[i];
    cout << endl;
  } else cout << "IMPOSSIBLE" << endl;
}
