#include <iostream>
#define INF ~0u>>1
#define mn 1000010
using namespace std;

char move[mn],res[mn];
int ans=INF,n,r,cur;

void check(int x,int y){
    for(cur=n-1;x*y!=0 && cur>=0;cur--)
        if(x>=y) x-=y,move[cur]='T';
        else y-=x,move[cur]='B';
    if(cur!=-1 || !(x==0 && y==1)) return;
    int tmp=0;
    for(int i=1;i<n;i++)
        if(move[i]==move[i-1]) tmp++;
    if(tmp<ans){
        ans=tmp;
        for(int i=0;i<n;i++)
            res[i]=move[i];
    }
}

int main(){
    cin>>n>>r;
    for(int i=0;i<=r;i++)
        check(i,r),check(r,i);
    if(ans==INF) cout<<"IMPOSSIBLE"<<endl;
    else cout<<ans<<endl<<res<<endl;
}
