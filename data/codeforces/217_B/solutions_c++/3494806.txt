#include<stdio.h>
#define INF 100000000
int num ;
int mis ;
int ct ;
int ar[1100000];
int solve(int a,int b){
	if( a == 0 || b == 0 ) return 0;
	if( a == 1 ){
		num += b;
		mis += b-2;
		ar[ct++] = b-1;
		return 1;
	}
	num += b/a;
	ar[ct++] = b/a;
	mis += b/a-1;
	return solve(b%a, a);
}
int main(){
	int n,m;
	scanf("%d %d",&n,&m);
	int minn = INF;
	int pos;
	if( n == 1 && m == 1 ){
		printf("0\nT\n");
		return 0;
	}
	for(int k = 1;  k < m ;k++ ){
		num = 0; mis = 0; ct = 0 ;
		int res = solve(k,m);
		if( res && num == n && mis < minn ){
			minn = mis;
			pos = k;
		}
	}
	if( minn == INF ){
		printf("IMPOSSIBLE\n");
		return 0;
	}
	ct = 0;
	solve(pos,m);
	printf("%d\n",minn);
	printf("T");
	int ind = 1;
	for(int i = ct-1 ; i>= 0 ;i-- ){
		for(int j = 0 ; j < ar[i] ; j++ ){
			if( ind == 1  ) printf("B");
			else printf("T");
		}
		ind^=1;
	}
	printf("\n");
}
