#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>

const int MAXN=1000000+9,INF=99999999;

int ans=INF,n;
char nowc[MAXN],ansc[MAXN];
void go(int t,int b)
{
	int i,len=0,cnt=0;
	for(;t!=b;++len)
		if(t>b)
		{
			t-=b;
			nowc[len]='T';
		}
		else
		{
			b-=t;
			nowc[len]='B';
		}
	nowc[len++]='T';
	if(t!=1 || len!=n)
		return;
	for(i=1;i<len;++i)
		if(nowc[i]==nowc[i-1])
			++cnt;
	if(cnt<ans)
	{
		ans=cnt;
		memcpy(ansc,nowc,n);
	}
}

int main()
{
	#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif
	int r,i;
	std::cin >> n >> r;
	for(i=1;i<=r;++i)
	{
		go(i,r);
		go(r,i);
	}
	if(ans!=INF)
	{
		std::cout << ans << std::endl;
		ansc[n]=0;
		std::reverse(ansc,ansc+n);
		puts(ansc);
	}
	else puts("IMPOSSIBLE");
}
