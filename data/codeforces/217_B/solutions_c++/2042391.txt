#include<cstdio>

#define INF 1000100

int main(){
  int n, r;
  bool s[INF];

  scanf("%d %d",&n,&r);

  int mini = INF, x;
  for(int i=1;i<=r;i++){
    int m = 0, w = 0, a = i, b = r;
    while(a != 0 && m <= n){
      m += b/a;
      w += b/a-1;
      if(a == 1 && b > 1) w--;
      int aux = a; a = b%a; b = aux;
    }
    if(a == 0 && b == 1 && m == n && w < mini){
      mini = w; x = i;
    }
  }
  if(mini == INF) { printf("IMPOSSIBLE\n"); return 0; }

  printf("%d\n",mini);
 
  for(int i=n-1,y = r;i>=0;i--) {
    if(x > y){
      x -= y;
      s[i] = x > y;
    }
    else {
      y -= x;
      s[i] = y > x;
    }
  }
  bool top = true;
  putchar('T');
  for(int i=1;i<n;i++){
    if(!s[i]) top = !top;
    putchar(top ? 'T' : 'B');
  }
  putchar('\n');

  return 0;
}
