#include <cstdio>
#include <cstring>

const int N = 1000000;
char s[N + 10], tmp[N + 10];
int ans = -1, n, r;

void solve(int x, int y) {
	for (int i = n - 1; i; --i) {
		if (x > y) {
			tmp[i] = 'T';
			x -= y;
		}
		else {
			tmp[i] = 'B';
			y -= x;
		}
		if (x <= 0 || y <= 0) return;
	}

	if (x == 1 && y == 1) {
		tmp[0] = 'T'; 
		tmp[n] = '\0';

		int cnt = 0;
		for (int i = 0; i != n; ++i)
			if (tmp[i] == tmp[i + 1])
				++cnt;
		
		if (ans == -1 || cnt < ans) {
			ans = cnt;
			for (int i = 0; i <= n; ++i)
				s[i] = tmp[i];
		}
	}
}

int main() {
	scanf("%d %d", &n, &r);

	for (int i = 1; i <= r; ++i) {
		solve(i, r);
		solve(r, i);
	}

	if (ans == -1) 
		printf("IMPOSSIBLE\n");
	else {
		printf("%d\n", ans);
		printf("%s\n", s);
	}
}
