#include <cstdio>
#include <string>
#include <iostream>
#define inf 0x3ffffff

using namespace std;

char ans[1000010],tmp[1000010];
int n,r;
int anss;

void solve(int t,int b)
{
	int len=0;
	while(t ^ b)
	{
		if(t>b)
		{
			tmp[len++]='T';
			t-=b;
		}
		else
		{
			tmp[len++]='B';
			b-=t;
		}
	}
	if(len!=n-1||t!=1) return;
	tmp[len]='T';
	int k=0;
	for(int i=0;i<n-1;i++)
		if(tmp[i]==tmp[i+1]) k++;
	if(k<anss)
	{
		anss=k;
		for(int i=0;i<n;i++)
			ans[i]=tmp[n-i-1];
	}
	return;
}

int main()
{
	cin>>n>>r;
	anss=inf;
	for(int i=1;i<=r;i++)
	{
		solve(i,r);
		solve(r,i);
	}
	if(anss==inf) cout<<"IMPOSSIBLE"<<endl;
	else cout<<anss<<'\n'<<ans<<endl;
	return 0;
}
