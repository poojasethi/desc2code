#include<stdio.h>
#include<stdlib.h>
long i,j,k,l,n,m,min;
long a[2000000],ans[2000000];
void find(long x,long y){
	long i,j=0,k,l=0;
	while(x!=y){
		if(x>y){
			a[++l]=0;
			x-=y;
		}else{
			a[++l]=1;
			y-=x;
		}
	}
	if(x!=1 || l!=n-1)return;
	a[++l]=0;
	for(i=1;i<l;++i)if(a[i]==a[i+1])++j;
	if(j<min){
		min=j;
		for(k=1;k<=l;++k)ans[k]=a[k];
	}
	return;
}
int main(){
	scanf("%ld%ld",&n,&m);
	min=100000000;
	for(i=1;i<=m;++i){
		find(m,i);
		find(i,m);
	}
	if(min==100000000){
		puts("IMPOSSIBLE\n");
		return 0;
	}
	printf("%ld\n",min);
	for(i=n;i;--i){
		if(ans[i])printf("B");
		else printf("T");
	}
	puts("");
	return 0;
}
