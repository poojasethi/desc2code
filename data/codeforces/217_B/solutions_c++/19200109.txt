#include<stdio.h>
#include<stdlib.h>
int temp[2000000],res[2000000];
int min;
int n,r;
void find(int x,int y)
{
	int i,l=0,j;
	while (x!=y)
	{
		if (x>y)
		{
			temp[++l]=0;
			x-=y;
		}
		else
		{
			temp[++l]=1;
			y-=x;
		}
	}
	if (x!=1 || l!=n-1) return;
	temp[++l]=0;
	j=0;
	for (i=1;i<l;i++)
	{
		if (temp[i]==temp[i+1])
			j++;
	}
	if (j<min)
	{
		min=j;
		for (i=1;i<=l;i++) res[i]=temp[i];
	}
	return;
}
int main()
{
	int i;
	scanf("%d %d",&n,&r);
	min=10000010;
	for (i=1;i<=r;i++)
	{
		find(r,i);
		find(i,r);
	}
	if (min==10000010)
	{
		printf("IMPOSSIBLE\n");
		exit(0);
	}
	printf("%d\n",min);
	for (i=n;i>=1;i--)
	{
		if (res[i]==0)
			printf("T");
		else
			printf("B");
	}
	printf("\n");
	return 0;
}
