#include<cstdio>
using namespace std;
int n,m,num=999999999;
char temp[1000005],ans[1000005];
void work(int t,int b)
{
	int len=1,cnt=0;
	while(t>=0&&b>=0&&t!=b)
		if(t>b)
		{
			t-=b;
			temp[len++]='T';
		}
		else
		{
			b-=t;
			temp[len++]='B';
		}
	if(len!=m||b!=1)return;
	temp[len]='T';
	for(int i=1;i<len;++i)
		if(temp[i]==temp[i+1])cnt++;
	if(cnt<num)
	{
		num=cnt;
		for(int i=0;i<len;++i)
			ans[i]=temp[len-i];
		ans[len]='\0';
	}
}
int main()
{
	scanf("%d%d",&m,&n);
	for(int i=1;i<=n;++i)
		work(i,n),work(n,i);
	if(num==999999999)printf("IMPOSSIBLE\n");
	else printf("%d\n%s\n",num,ans);
}
