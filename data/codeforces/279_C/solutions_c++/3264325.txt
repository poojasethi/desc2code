#include <cstdio>
#define MAXN 100005

int V[MAXN], T1[MAXN], T2[MAXN];

int main()
{
    int n, m, l, r;
    scanf("%d %d", &n, &m);
    for(int i = 0; i < n; i++)
        scanf("%d", &V[i]);
    T1[0] = 0; T2[n-1] = n-1;
    for(int i = 1; i < n; i++){
        T1[i] = V[i] <= V[i-1]? T1[i-1]:i;
        T2[n-i-1] = V[n-i-1] <= V[n-i]? T2[n-i]: n-i-1;
    }
    while( m-- ){
        scanf("%d %d", &l, &r);
        puts( T2[l-1] >= T1[r-1]?"Yes":"No");
    }
}
