#include<iostream>
using namespace std;
int main()
{
	int n,m,p=0,q,i;
	cin>>n>>m;
	int a[n],b[n],c[n];
	for(i=0;i<n;i++)cin>>a[i];
	i=0;
	while(i<n)
	{
		i=p;
		while(a[p+1]>=a[p] and p<n)p++;
		for(int k=i;k<=p;k++)b[k]=p;p++;
	}p=n-1;
	while(i>=0)
	{
		i=p;
		while(a[p-1]>=a[p] and p>=0)p--;
		for(int k=i;k>=p;k--)c[k]=p;p--;
	}
	while(m--)
	{
		cin>>p>>q;
		if(b[--p]>=c[--q])cout<<"Yes"<<endl;else cout<<"No"<<endl;
	}
}
