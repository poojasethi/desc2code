#include<stdio.h>
#define X 100010
int a[X],l[X],r[X];
int main(){
    int i,j,n,m,x,y;
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++)
        scanf("%d",&a[i]);
    for(i=j=1;i<=n;i++){
        if(a[i]>a[i-1])j=i;
        l[i]=j;
    }
    r[n+1]=1<<31;
    for(i=j=n;i>=1;i--){
        if(a[i]>a[i+1])j=i;
        r[i]=j;
    }
    for(i=0;i<m;i++){
        scanf("%d%d",&x,&y);
        puts(r[x]>=l[y]?"Yes":"No");
    }
    return 0;
}
