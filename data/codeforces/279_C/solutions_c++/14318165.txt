#include<bits/stdc++.h>
using namespace std;
int main()
{
   int n,m;
   scanf("%d %d",&n,&m);
   int data[n+10],l[n+10],r[n+10];
   for(int i=1;i<=n;i++)scanf("%d",&data[i]);
   l[1]=1;
   r[n]=n;
   for(int i=2;i<=n;i++)
     l[i]=data[i]<=data[i-1]?l[i-1]:i;
   for(int i=n-1;i>0;i--)
     r[i]=data[i]<=data[i+1]?r[i+1]:i;
   while(m--)
   {
       int le,re;
       cin>>le>>re;
       puts(r[le]>=l[re]?"Yes":"No");
   }

}
