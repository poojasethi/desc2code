#include<stdio.h>
int a[100005],l[100005],r[100005];
int main()
{
	int j,i,m,n;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	l[1]=1;r[n]=n;
	for(i=2;i<=n;i++)
		if(a[i]<=a[i-1])
			l[i]=l[i-1];
		else 
			l[i]=i;
	for(i=n-1;i>=1;i--)
		if(a[i]<=a[i+1])
			r[i]=r[i+1];
		else 
			r[i]=i;
    while(m--)
	{
		scanf("%d%d",&i,&j);
		if(r[i]>=l[j])
			printf("Yes\n");
		else printf("No\n");
	}
	return 0;
}
