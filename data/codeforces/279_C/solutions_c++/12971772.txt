#include <iostream>

using namespace std;

int Right[100100] , Left[100100];
int a[100100] , l , r , n , m;

int main()
{
cin>>n>>m;

for(int i=1;i<=n;i++)
    cin>>a[i];

Left[1]=1;
Right[n]=n;

for(int i=2;i<=n;i++)
    Left[i]=( a[i] <= a[i-1] )?Left[i-1]:i;

for(int i=n-1;i>=1;i--)
    Right[i]=(a[i] <= a[i+1] )?Right[i+1]:i;

while(m--){

cin>>l>>r;

if( Right[l]>=Left[r] )
     cout<<"Yes"<<endl;
else cout<<"No"<<endl;

}


    return 0;
}
