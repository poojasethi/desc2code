#include<cstdio>
const int N=100001;
int a[N],l[N],r[N];
int main(){
    int n,m,i,L,R;
    while(~scanf("%d%d",&n,&m)){
        for(i=1;i<=n;i++)scanf("%d",a+i);
        l[1]=1,r[n]=n;
        for(i=2;i<=n;i++)l[i]=(a[i]<=a[i-1])?l[i-1]:i;
        for(i=n-1;i>0;i--)r[i]=(a[i]<=a[i+1])?r[i+1]:i;
        while(m--){
            scanf("%d%d",&L,&R);
            puts(r[L]>=l[R]?"Yes":"No");
        }
    }
    return 0;
}
