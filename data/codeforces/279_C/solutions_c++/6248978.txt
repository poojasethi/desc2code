#include<stdio.h>
int n,x,y,m,a[100002],l[100002],r[100002];
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;++i){
        scanf("%d",a+i);
        l[i]=(a[i]<=a[i-1])?l[i-1]+1:1;
    }
    for(int i=n;i>0;--i)
        r[i]=(a[i]<=a[i+1])?r[i+1]+1:1;
    while(m--){
        scanf("%d%d",&x,&y);
        if(r[x]+l[y]>=y-x+1)
            puts("Yes");
        else
            puts("No");
    }
}
