#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,l,r,n,m;
	cin>>n>>m;
	int a[n];
	int in[n];
	int de[n];
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	l=n-1;
	in[n-1]=n-1;
	for(i=n-2;i>=0;i--)
	{
		if(a[i]>a[i+1])
			l=i;
		in[i]=l;	
	}
	de[n-1]=n-1;
	l=n-1;
	for(i=n-2;i>=0;i--)
	{
		if(a[i]<a[i+1])
			l=i;
		de[i]=l;	
	}	
	for(i=0;i<m;i++)
	{
		scanf("%d%d",&l,&r);
		l--;r--;
		if(de[in[l]]>=r)
			cout<<"Yes\n";
		else
			cout<<"No\n";	
	}	
	return 0;
}
