#include<stdio.h>
#define N 100005

int a[N],less[N],greater[N];
int n,q;

int main()
{
	int i;
	int l,r;
	scanf("%d%d",&n,&q);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		less[i]=greater[i]=i;
	}
	for(i=n-1;i>=1;i--)
		if(a[i]<=a[i+1])
			less[i]=less[i+1];
	for(i=2;i<=n;i++)
		if(a[i-1]>=a[i])
			greater[i]=greater[i-1];
	while(q--)
	{
		scanf("%d%d",&l,&r);
		if(less[l]>=greater[r])
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
