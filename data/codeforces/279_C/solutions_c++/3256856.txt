#include <iostream>

using namespace std;

const int max_n = 1e5 + 14;

int i,n, q, a, b, niz[max_n], l[max_n], r[max_n];

int main()
{
  cin >> n >> q;
  for ( i = 0; i < n; i++) cin >> niz[i];
  for ( i = 1; i < n; i++)
  {
    if (niz[i - 1] >= niz[i]) l[i] = l[i - 1] + 1;
    if (niz[n - i - 1] <= niz[n - i]) r[n - i - 1] = r[n - i] + 1;
  }
  while (q--)
  {
    cin >> a >> b;
    cout << (b - a <= r[--a] + l[--b] ? "Yes" : "No") << endl;
  }
  return 0;
}
