#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	int a,b,c,d;
	scanf("%d %d",&a,&b);
	int v[a+1],l[a+1],r[a+1];
	for(int i=1;i<=a;i++)
		scanf("%d",&v[i]);
	l[1]=1;
	r[a]=a;
	for(int i=2;i<=a;i++)
		if(v[i]>v[i-1])
			l[i]=i;
		else
			l[i]=l[i-1];
	for(int i=a-1;i>0;i--)
		if(v[i]>v[i+1])
			r[i]=i;
		else
			r[i]=r[i+1];
	for(int i=0;i<b;i++)
	{
		scanf("%d %d",&c,&d);
		if(l[d]>r[c])
			cout<<"No\n";
		else
			cout<<"Yes\n";
	}
	return 0;
}
