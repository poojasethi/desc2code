#include <bits/stdc++.h>
#define MAX 111111

using namespace std;

int R[MAX], L[MAX], A[MAX];

int main()
{
	int N, Q;
	cin >> N >> Q;
	for (int i = 1; i <= N; i++) cin >> A[i];
	L[1] = 1; R[N] = N;
	for (int i = 2; i <= N; i++) L[i] = (A[i] <= A[i - 1]) ? L[i - 1] : i;
	for (int i = N - 1; i >= 1; i--) R[i] = (A[i] <= A[i + 1]) ? R[i + 1] : i;
	int l, r;
	while (Q--)
	{
		cin >> l >> r;
		cout << (R[l] >= L[r] ? "Yes\n" : "No\n");
	}
}
