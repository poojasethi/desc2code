#include<bits/stdc++.h>

using namespace std;

int main()
{
	int n,m;
	cin>>n>>m;
	int a[100005],i,l,r;
	for(i=0;i<n;i++)
	cin>>a[i];
	
	int lcs[100004]={0},dcs[100005]={0};
	for(i=0;i<100000;i++)
	lcs[i]=1,dcs[i]=1;
	
	for(i=1;i<n;i++)
	if(a[i]<=a[i-1])
	lcs[i]=lcs[i-1]+1;
	
	for(i=n-2;i>=0;i--)
	if(a[i]<=a[i+1])
	dcs[i]=dcs[i+1]+1;
	
	for(i=0;i<m;i++)
	{
		cin>>l>>r;
		l--;r--;
		if(lcs[r]+dcs[l]>=(r-l+1))
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
	
	return 0;
}
