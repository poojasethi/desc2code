#include<iostream>
using namespace std;
int n,m;
int a[110000]={0};
int l,r;
int I[110000]={0},D[110000]={0};
int main(void){
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	D[1]=1;
	for(int i=2;i<=n;i++){
		if(a[i]<=a[i-1])D[i]=D[i-1]+1;
		else D[i]=1;
	}
	I[n]=1;
	for(int i=n-1;i>=1;i--){
		if(a[i]<=a[i+1])I[i]=I[i+1]+1;
		else I[i]=1;
	}
	for(int i=1;i<=m;i++){
		if(i>1)cout<<endl;
		cin>>l>>r;
		if(I[l]+D[r]>=r-l+2)cout<<"Yes";
		else cout<<"No";
	}
}
