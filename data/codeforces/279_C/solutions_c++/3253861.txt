#include <iostream>
#include <cstdio>
using namespace std;
int inc[100005], decr[100005], n, q, a[1100005];
int main() {
 cin>>n>>q;
for(int i = 1; i <= n; i ++) cin>>a[i];
for(int i = 2; i <= n; i ++)
 if(a[i]<=a[i - 1]) decr[i] = decr[i - 1] + 1;
for(int i = n - 1; i > 0; i --)
    if(a[i] <= a[i + 1]) inc[i] = inc[i + 1] + 1;
for(int j=1; j<=q; j++)
{
 int l, r;
 cin>>l>>r;
if(decr[r] + inc[l] >= r - l) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
}
return 0;
}
