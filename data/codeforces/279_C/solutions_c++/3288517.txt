#include <iostream>
using namespace std;
int main()
{int n,m; cin>>n>>m;
int a[100005],dec[100005]={},inc[100005]={};
cin>>a[1];
for (int i=2;i<=n;i++)
{cin>>a[i];
 if (a[i-1]>=a[i]) inc[i]=inc[i-1]+1;
}
for (int i=n-1;i>=1;i--)
  if (a[i+1]>=a[i]) dec[i]=dec[i+1]+1;
for (int j=0;j<m;j++)
{int l,r; cin>>l>>r;
 if (inc[r]+dec[l]>=r-l) cout<<"Yes"<<endl;
   else cout<<"No"<<endl;
}
return 0;
}
