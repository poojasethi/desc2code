#include <cstdio>
#include <iostream>
using namespace std;
int a[111111];
int	D[111111];
int I[111111];
int main()
{
	int n,m,l,r;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
			scanf("%d",&a[i]);
	I[n-1]=D[0]=1;
	for(int i=1;i<n;i++)
			(a[i]<=a[i-1])?D[i]=D[i-1]+1:D[i]=1;
	for(int i=n-2;i>=0;i--)
			a[i]<=a[i+1]?I[i]=I[i+1]+1:I[i]=1;
	for(int i=0;i<m;i++)
			{
				scanf("%d%d",&l,&r);
				--l;--r;
				D[r]+I[l]>(r-l)+1?puts("Yes"):puts("No");
			}
}
