#include <iostream>
using namespace std;

int nd[100000];
int ni[100000];
int A[100000];
int main()
{
int n,m;
cin >> n >> m;
for (int i=0; i<n; i++) cin >> A[i];
int lasti = 0, lasti2 = 0;
nd[0] = 0; ni[0] = 0;
for (int i=1; i<n; i++) { 
if (A[i]<A[i-1]) lasti=i; nd[i]=lasti;
if (A[i]>A[i-1]) lasti2=i; ni[i]=lasti2;
}
for (int i=0; i<m; i++) {
	int l,r;
	cin >> l >> r; l--; r--;
	int s = nd[ni[r]];
	if (s<=l) cout << "Yes" << endl; else cout << "No" << endl;
}
return 0;
}
