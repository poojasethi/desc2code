#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int mod = 1e9+7;
#define MAX 100005
int a[MAX],lft[MAX],rgt[MAX];
int main(){
 int n,q,l,r;
 cin >> n >> q;
 for(int i=1;i<=n;i++) cin >> a[i];
 lft[1] = 1; rgt[n] = n;
 for(int i=2;i<=n;i++) lft[i] = a[i]<=a[i-1]?lft[i-1]:i;
 for(int i=n-1;i>0;i--) rgt[i] = a[i]<=a[i+1]?rgt[i+1]:i;
 while(q--){
	cin >> l >> r;
	if(rgt[l]>=lft[r]) puts("Yes");
	else puts("No");
 }
 return 0;
}
