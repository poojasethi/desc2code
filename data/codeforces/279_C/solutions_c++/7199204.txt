#include<cstdio>
#define N 100005
using namespace std;
int a[N],inc[N],dec[N],n,m,i,L,R;
int main()
{
  scanf("%d%d",&n,&m);
  for (i=1;i<=n;i++) scanf("%d",&a[i]);
  dec[1]=1;inc[n]=n;
  for (i=2;i<=n;i++)
    if (a[i-1]>=a[i]) dec[i]=dec[i-1];else dec[i]=i;
  for (i=n-1;i;i--)
    if (a[i+1]>=a[i]) inc[i]=inc[i+1];else inc[i]=i;
  while (m--)
  {
    scanf("%d%d",&L,&R);
    if (inc[L]>=dec[R]) puts("Yes");else puts("No");
  } 
  return 0;
}
