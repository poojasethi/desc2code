#include <bits/stdc++.h>
using namespace std;
int main() {
	int q,i,j,k,l,n,m,a[100005],b[100005],c[100005];
	cin>>n>>q>>a[1];
	b[1] = 1;
	for(i=2;i<=n;i++) {
		cin>>a[i];
		if(a[i] <= a[i-1])
			b[i] = b[i-1];
		else
			b[i] = i;
	}
	c[n] = n;
	for(i=n-1;i>0;i--)
		if(a[i] <= a[i+1])
			c[i] = c[i+1];
		else
			c[i] = i;
	while(q--) {
		cin>>j>>k;
		if(c[j]>=b[k])
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
	return 0;
}
