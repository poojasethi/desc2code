#include<cstdio>

#define MAXX 100111

int a[MAXX],n,i,j,k,m;
int u[MAXX],d[MAXX];

int main()
{
    scanf("%d %d",&n,&m);
    for(i=1;i<=n;++i)
        scanf("%d",a+i);
    for(i=1;i<=n;++i)
        if(a[i]<=a[i-1])
            u[i]=u[i-1]+1;
        else
            u[i]=1;
    for(i=n;i;--i)
        if(a[i]<=a[i+1])
            d[i]=d[i+1]+1;
        else
            d[i]=1;
    while(m--)
    {
        scanf("%d %d",&i,&j);
        puts(d[i]+u[j]>j-i?"Yes":"No");
    }
    return 0;
}
