
n, m = map(int, raw_input().split()) 

A = [0] + map(int, raw_input().split()) 

AL = [0] * (n+1)
DL = [0] * (n+1)

pos_al = 1
AL[1] = 1
pos_dl = 1
DL[1] = 1

for i in range(2, n+1): 
    if A[i] < A[i-1]: 
        pos_al = i
    AL[i] = pos_al

    if A[i] > A[i-1]: 
        pos_dl = i
    DL[i] = pos_dl

for _ in range(m):
    l, r = map(int, raw_input().split()) 
    L = AL[DL[r]]
    print 'Yes' if L <= l else 'No'
