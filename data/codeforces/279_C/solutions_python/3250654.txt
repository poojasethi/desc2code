#!/usr/bin/env/ python
n,m = map(int,raw_input().split())
a = map(int,raw_input().split())
inc = [0]*n
dec = [0]*n

dec[0] = 0
for i in range(1,n):
    dec[i] = dec[i-1]
    if a[i]-a[i-1]>0: dec[i]=i

inc[n-1] = n
for i in range(n-1,0,-1):
    inc[i-1] = inc[i]
    if a[i]-a[i-1]<0: inc[i-1] = i

for i in range(m):
    l,r = map(int,raw_input().split())
    l -= 1
    r -= 1
    if inc[l]>dec[r]: print "Yes"
    else: print "No"
