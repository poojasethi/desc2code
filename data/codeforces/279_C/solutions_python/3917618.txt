def read():
    return map(int, raw_input().split())

def main():
    n, m = read()
    a = [0] + read() + [0]
    
    left, right = [0]*(n+2), [0]*(n+2)

    left[n]=n
    for i in reversed(range(1,n)):
        left[i] = left[i+1] if a[i]<=a[i+1] else i

    right[1]=1
    for i in range(2,n+1):
        right[i] = right[i-1] if a[i-1]>=a[i] else i

    for _ in range(m):
        l, r = read()
        print "Yes" if left[l] >= right[r] else "No"

main()
