inp = raw_input().split()

n = int(inp[0])
m = int(inp[1])

a = [int(s) for s in raw_input().split()]

firstDec = [-1 for i in range(n)]
firstInc = [-1 for i in range(n)]

lastCorrectDec = -1
lastCorrectInc = -1

for i in range(n-1):
	if (a[i] < a[i+1]):
		# inc discovered
		for j in range(lastCorrectInc+1,i+1):
			firstInc[j] = i
		lastCorrectInc = i
	elif (a[i] > a[i+1]):
		# dec discovered
		for j in range(lastCorrectDec+1,i+1):
			firstDec[j] = i
		lastCorrectDec = i

for j in range(lastCorrectInc+1,n):
	firstInc[j] = n-1

for j in range(lastCorrectInc+1,n):
	firstDec[j] = n-1

for i in range(m):
	inp = raw_input().split()
	l = int(inp[0]) - 1
	r = int(inp[1]) - 1
	if (r <= firstInc[firstDec[l]]):
		print 'Yes'
	else:
		print 'No'
