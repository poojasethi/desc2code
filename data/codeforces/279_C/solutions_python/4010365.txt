n ,m = tuple(map(int, raw_input().split(' ')))
data = list(map(int, raw_input().split(' ')))

data = [0] + data
t_inc= [0]* (n+1)
t_dec= [0]* (n+1)

for i in xrange(n-1, 0, -1):
    if data[i]<=data[i+1]:
        t_inc[i] = t_inc[i+1] + 1
    if data[i]>= data[i+1]:
        t_dec[i] = t_dec[i+1] + 1

for j in xrange(0, m):
    a, b = tuple(map(int, raw_input().split(' ')))
    if a+t_inc[a]>=b or a+t_dec[a]>=b or a+t_inc[a]+t_dec[a+t_inc[a]]>=b:
        print 'Yes'
    else:
        print 'No'
