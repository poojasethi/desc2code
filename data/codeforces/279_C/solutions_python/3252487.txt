from bisect import * 
n, q = map(int, raw_input().split())
a = map(int, raw_input().split())
lt = [0] * n
rt = [0] * n
for i in range(0, n - 1)[::-1] : 
	lt[i] = [0, lt[i + 1] + 1][(a[i] >= a[i + 1])]
for i in range(0, n - 1)[::-1] : 
	rt[i] = [0, rt[i + 1] + 1][(a[i] <= a[i + 1])]
for i in range(q) : 
	l, r = map(int, raw_input().split())
	rmost = l + rt[l - 1] - 1
	print ["No", "Yes"][rmost + lt[rmost] >= r - 1]
