n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
q = []
for _ in xrange(m):
  q.append(map(int, raw_input().split()))

def desc(a):
  ret = [1] * len(a)
  for i in xrange(1, len(a)):
    if a[i] <= a[i-1]:
      ret[i] = ret[i-1] + 1
    else:
      ret[i] = 1
  return ret

dsc = desc(a)
asc = list(reversed(desc(list(reversed(a)))))

for l, r in q:
  if asc[l-1] + dsc[r-1] >= r - l + 2:
    print "Yes"
  else:
    print "No"
