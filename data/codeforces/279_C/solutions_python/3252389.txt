I=lambda:map(int,raw_input().split())
n,q=I()
a=I()
l,r=[0]*100005,[0]*100005
l[0],r[n-1]=1,1
#1 2 1 3 3 5 2 1
for i in xrange(1,n):
    if a[i]<=a[i-1]: l[i]=l[i-1]+1
    else: l[i]=1
for i in xrange(n-2,-1,-1):
    if a[i]<=a[i+1]: r[i]=r[i+1]+1
    else: r[i]=1
for _ in xrange(q):
    i,j=I()
    if l[j-1]+r[i-1]>=j-i+2: print 'Yes'
    else: print 'No'
