#!/usr/bin/env python
from sys import stdin as cin

def main():
    n, m = map(int, next(cin).split())
    a = map(int, next(cin).split())

    up = range(n)
    down = range(n)

    l = 0
    r = 0
    while r < n:
        if r == n-1 or a[r+1] < a[r]:
            up[l:r+1] = [r]*(r-l+1)
            l = r + 1
        r += 1

    l = 0
    r = 0
    while r < n:
        if r == n-1 or a[r+1] > a[r]:
            down[l:r+1] = [r]*(r-l+1)
            l = r + 1
        r += 1

    for i in range(m):
        l, r = map(int, next(cin).split())
        l, r = l-1, r-1

        if down[up[l]] >= r:
            print 'Yes'
        else:
            print 'No'

main()
