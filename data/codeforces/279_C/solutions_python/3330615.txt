n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
l = [0] * n
x = y = -1
p = a[0]
for i, v in enumerate(a):
	if p < v:
		x += 1
		y = x
	elif p > v:
		x = 0
		y += 1
	else:
		x += 1
		y += 1
	l[i] = y
	p = v
for _ in range(m):
	x, y = map(int, raw_input().split())
	print ["No", "Yes"][y - l[y - 1] <= x]
