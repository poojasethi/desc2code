sc = lambda: map(int, raw_input().split())

n, m = sc()
a = [0] + sc() + [0]
b = [i for i in range(n+2)]
c = b[:]

for i in range(1, n+1):
    if a[i] <= a[i-1] :
        b[i] = b[i-1]

for i in range(n):
    if a[n-i]<=a[n-i+1]: 
        c[n-i] = c[n-i+1]

for i in range(m):
    l, r = sc()
    if b[r] <= c[l]:
        print "Yes";
    else:
        print "No";
