r = lambda: map(int, raw_input().split())
str = raw_input()
nb, ns, nc = r()
pb, ps, pc = r()
p = input()

c_b, c_s, c_c = str.count('B'), str.count('S'), str.count('C')

l, r = 0, 10 ** 13

while r > l:
    mid = (l + r + 1) >> 1
    if max(0, (mid * c_b - nb) * pb) + max(0, (mid * c_s - ns) * ps) + max(0, (mid * c_c - nc) * pc) <= p: l = mid
    else: r = mid - 1
print l
