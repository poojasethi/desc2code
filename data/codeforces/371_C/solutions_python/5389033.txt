#!/usr/bin/env python

want = [0,0,0]
have = [0,0,0]

for i in raw_input():
  if i == 'B': want[0] += 1
  if i == 'S': want[1] += 1
  if i == 'C': want[2] += 1

j = 0
for i in map(int, raw_input().split(' ')):
  have[j] += i
  j += 1

price = map(int, raw_input().split(' '))
money = int(raw_input())

lef = 0
rad = 2 ** 40
while rad >= 1:
  mid = lef + rad

  need = 0
  for i in xrange(3):
    need += max(0,mid*want[i]-have[i]) * price[i]

  if need <= money:
    lef = mid
  rad /= 2

print lef
