s = raw_input()
a = [s.count(c) for c in "BSC"]
b = map(int, raw_input().split())
p = map(int, raw_input().split())
r = int(raw_input())
lo, hi = 0, 10 ** 13
while lo + 1 < hi:
    mid = (lo + hi) / 2
    c = sum(p[i] * max(a[i] * mid - b[i], 0) for i in range(3))
    if c > r:
        hi = mid
    else:
        lo = mid
print lo
