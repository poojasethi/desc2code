rcp = raw_input()
rcp = [rcp.count(c) for c in ['B', 'S', 'C']]
n = map(int, raw_input().split())
p = map(int, raw_input().split())
r, cnt = int(raw_input()), 0
# binary search for the largest x s.t. f(x) <= r
left, right = 0, max(n) + r
while left < right:
	m = (left + right + 1) / 2
	f = sum(max(0, rcp[i] * m - n[i]) * p[i] for i in range(3))
	if f <= r:
		left = m
	else:
		right = m - 1
print left
