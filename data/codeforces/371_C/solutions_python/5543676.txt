ft=raw_input()
n=map(int,raw_input().split())
p=map(int,raw_input().split())
r=int(raw_input())
f=[ft.count("B"),ft.count("S"),ft.count("C")]
u=[]
for x in xrange(3):
    if f[x]!=0:
        u.append(x)
ans=0
s=1
for x in u:
    s+=n[x]
while r>0 and s>0:
    ans+=1
    s=0
    for x in u:
        n[x]-=f[x]
        if n[x]<0:
            r+=n[x]*p[x]
            n[x]=0
        s+=n[x]
s=0
for x in u:
    s+=p[x]*f[x]
if r>=0:
    print ans+r/s
else:
    print ans-1
