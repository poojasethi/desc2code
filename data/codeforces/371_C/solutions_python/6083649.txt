s=raw_input()
d={}
for l in ['B','S','C']:
    d[l] = s.count(l)
r=lambda:map(int, raw_input().split())
b,s,c=r()
B,S,C=r()
n=input()
l=0
r=max([b,s,c])+n/min([B,S,C])
while l < r:
    m = (r+l)/2+1
    N=max(0,(m*d['B']-b)*B) + max(0,(m*d['S']-s)*S) + max(0,(m*d['C']-c)*C)

    if N > n:
        r=m-1
    else:
        l=m

print l
