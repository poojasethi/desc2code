
recipe = raw_input()
number_bread, number_sausage, number_cheese = map(int, raw_input().split(' '))
price_bread, price_sausage, price_cheese = map(int, raw_input().split(' '))
money = long(raw_input())

used = {}
used['B'] = 0
used['S'] = 0
used['C'] = 0
for c in recipe:
    if c == "B":
        used['B'] += 1
    elif c == "S":
        used['S'] += 1
    else:
        used['C'] += 1

hamburguers = 0
while True:
    if (number_bread == 0 or used['B'] == 0) and (number_sausage == 0 or used['S'] == 0) and (number_cheese == 0 or used['C'] == 0):
        cost = used['B'] * price_bread + used['S'] * price_sausage + used['C'] * price_cheese
        hamburguers += money / cost
        break
    if number_bread >= used['B']:
        number_bread -= used['B']
        if number_sausage >= used['S']:
            number_sausage -= used['S']
            if number_cheese >= used['C']:
                number_cheese -= used['C']
                hamburguers += 1
            else:
                needed = used['C'] - number_cheese
                if money >= needed * price_cheese:
                    money -= needed * price_cheese
                    number_cheese = 0
                    hamburguers += 1
                else: break
        else:
            needed = used['S'] - number_sausage
            if money >= needed * price_sausage:
                money -= needed * price_sausage
                number_sausage = 0
                if number_cheese >= used['C']:
                    number_cheese -= used['C']
                    hamburguers += 1
                else:
                    needed = used['C'] - number_cheese
                    if money >= needed * price_cheese:
                        money -= needed * price_cheese
                        number_cheese = 0
                        hamburguers += 1
                    else: break
            else: break
    else:
        needed = used['B'] - number_bread
        if money >= (needed * price_bread):
            money -= needed * price_bread
            number_bread = 0
            if number_sausage >= used['S']:
                number_sausage -= used['S']
                if number_cheese >= used['C']:
                    number_cheese -= used['C']
                    hamburguers += 1
                else:
                    needed = used['C'] - number_cheese
                    if money >= needed * price_cheese:
                        money -= needed * price_cheese
                        number_cheese = 0
                        hamburguers += 1
                    else: break
            else:
                needed = used['S'] - number_sausage
                if money >= needed * price_sausage:
                    money -= needed * price_sausage
                    number_sausage = 0
                    if number_cheese >= used['C']:
                        number_cheese -= used['C']
                        hamburguers += 1
                    else:
                        needed = used['C'] - number_cheese
                        if money >= needed * price_cheese:
                            money -= needed * price_cheese
                            number_cheese = 0
                            hamburguers += 1
                        else: break
        else:
            break

print hamburguers
