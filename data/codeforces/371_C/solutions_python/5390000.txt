def check(k, S):
    if n_b < k*r_b:
        S -= (k*r_b-n_b) * p_b
        if S < 0:
            return False, S
    if n_s < k*r_s:
        S -= (k*r_s-n_s) * p_s
        if S < 0:
            return False, S
    if n_c < k*r_c:
        S -= (k*r_c-n_c) * p_c
        if S < 0:
            return False, S
    return True, S
