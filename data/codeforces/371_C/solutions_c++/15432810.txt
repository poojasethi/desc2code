#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pii;

const ll MAXK = 110;

ll s, b, c, ps, pb, pc, r, ns, nb, nc;
int main(){
	string ex;

	cin >> ex;
	for (ll i = 0; i < ex.size(); i++)
		if (ex[i] == 'B') nb++;
		else if (ex[i] == 'S') ns++;
		else nc++;		

	cin >> b >> s >> c >> pb >> ps >> pc >> r;

	ll ans = 0, e = 1e15;

	while (ans + 1 < e){
		ll mid = (e + ans) / 2LL;

		if ( max(0LL, mid * nb - b) * pb + max(0LL, mid * ns - s) * ps + max(0LL, mid * nc - c) * pc <= r)
			ans = mid;
		else e = mid;
	}

	cout << ans << endl;
	
}
