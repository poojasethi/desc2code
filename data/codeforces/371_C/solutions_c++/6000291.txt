#include<iostream>
#include<string>
using namespace std;

long long n[10], i, m, S, q, l, r = 1LL << 41;
string s, b("BSC");

int main()
{
    for (cin >> s; i < s.length(); n[b.find(s[i++])]++);
    for (i = 3; i < 10; cin >> n[i++]);
    while (l + 1 < r)
    {
        m = (l + r) >> 1;
        for (S = i = 0; i < 3; i++)
            if ((q = n[i] * m - n[3 + i]) > 0)
                S += q * n[6 + i];
        S <= n[9] ? l = m : r = m;
    }
    cout << l;
    return 0;
}
