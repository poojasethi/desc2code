#include <iostream>
#include <algorithm>
using namespace std;

#define int long long

int n[3], p[3], t[3], r;

bool valid(int sol)
{
	int s = 0;
	for (int i = 0; i < 3; ++i)
		s += max(0ll, sol*t[i]-n[i])*p[i];
	cerr << sol << ' ' << s << '\n';
	return s <= r;
}

main()
{
	string s;
	cin >> s >> n[0] >> n[1] >> n[2] >> p[0] >> p[1] >> p[2] >> r;
	for (int i = 0; i < (int)s.size(); ++i)
	{
		if (s[i] == 'B')
			t[0] ++;
		else if (s[i] == 'S')
			t[1] ++;
		else
			t[2] ++;
	}
	int l = 0, r = 1e13;
	while (r-l > 1)
	{
		int mid = (r+l)/2;
		if (valid(mid))
			l = mid;
		else
			r = mid;
	}
	cout << l << '\n';
	return 0;
}
