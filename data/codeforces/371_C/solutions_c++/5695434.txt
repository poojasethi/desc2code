#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int main()
{
	char ss[101];
	__int64 b=0,s=0,c=0,nb,ns,nc,pb,ps,pc,r;
	cin>>ss>>nb>>ns>>nc>>pb>>ps>>pc>>r;
	for(int i=0;i<strlen(ss);i++)
	{
		if(ss[i]=='B') b++;
		if(ss[i]=='S') s++;
		if(ss[i]=='C') c++; 
	}
	__int64 ll=0,rr=r+nb+ns+nc;
    while(rr-ll>1)
    {
        __int64 mid=(ll+rr)/2,t=r;
        if(mid*b>nb) t-=(mid*b-nb)*pb;
        if(mid*s>ns) t-=(mid*s-ns)*ps;
        if(mid*c>nc) t-=(mid*c-nc)*pc; 
        if(t>=0) ll=mid;else rr=mid;  
    }
	cout<<ll<<endl;
	return 0;
}
