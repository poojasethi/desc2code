#include <string.h>
#include <stdio.h>
#include <algorithm>
#include <iostream>

using namespace std;

typedef long long ll;

char s[111];
ll nb, ns, nc, pb, ps, pc, R;

int main(){
	cin >> s;
	cin >> nb >> ns >> nc;
	cin >> pb >> ps >> pc;
	cin >> R;
	int cb = 0, cs = 0, cc = 0;
	for(int i = 0; s[i]; i++)
		if(s[i] == 'B') cb++;
		else if(s[i] == 'S') cs++;
		else cc++;
	ll l = 0, r = 1e15, ret = 0;
	while(l <= r){
		ll mid = (l + r) >> 1;
		if(max(cb * mid - nb, 0ll) * pb + max(cs * mid - ns, 0ll) * ps + max(cc * mid - nc, 0ll) * pc <= R)
			ret = mid, l = mid + 1;
		else r = mid - 1;
	}
	cout << ret << endl;
	return 0;
}
