#include<stdio.h>
typedef long long ll;
ll cnt[3],N[3],P[3],money;
ll f(ll m)
{
	ll need=0;
	for(int i=0;i<3;i++)
	{
		ll num=cnt[i]*m-N[i];
		if(num>0)
			need+=num*P[i];
	}
	return need;
}
int main()
{
	char c;
	while((c=getchar())!='\n')
	{
		if(c=='B') cnt[0]++;
		else if(c=='S') cnt[1]++;
		else cnt[2]++;
	}
	for(int i=0;i<3;i++)
		scanf("%I64d",N+i);
	for(int i=0;i<3;i++)
		scanf("%I64d",P+i);
	scanf("%I64d",&money);
	
	ll l=0,r=1100000000000LL;
	while(r-l>1)
	{
		ll m=(r+l)/2;
		if(f(m)<=money) l=m;
		else r=m;
	}
	printf("%I64d",l);
	
	return 0;
}
