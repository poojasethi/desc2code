#include<stdio.h>
#include<string.h>
int main()
{
	char str[120];
	__int64 b=0,s=0,c=0,nb,ns,nc,pb,pc,ps,money,i,m,n,mid,r,temp;
	gets(str);
	for(i=0;str[i]!='\0';i++){
		if(str[i]=='B') b++;
		else if(str[i]=='S') s++;
		else if(str[i]=='C') c++;
	}
	scanf("%I64d%I64d%I64d%I64d%I64d%I64d%I64d",&nb,&ns,&nc,&pb,&ps,&pc,&money); 
	m=0;
	n=nb+nc+ns+money*3;temp=money;
	while(m<=n){
		money=temp;
		mid=(m+n)/2;
		if(mid*b>nb) money=money-(mid*b-nb)*pb;
		if(mid*s>ns) money=money-(mid*s-ns)*ps;
		if(mid*c>nc) money=money-(mid*c-nc)*pc;
		if(money<0) n=mid-1;
		else {r=mid;m=mid+1;}
	}
	printf("%I64d",r);
	return 0;
}
