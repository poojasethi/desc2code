#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
long long a(0),b(0),c(0),na,nb,nc,pa,pb,pc,r,ans(0);
char str[105];
int main(){
	scanf("%s",str);
	for(int i=0;i<strlen(str);++i){
		if(str[i]=='B')a++;
		if(str[i]=='S')b++;
		if(str[i]=='C')c++;
	}
	scanf("%I64d%I64d%I64d%I64d%I64d%I64d%I64d",&na,&nb,&nc,&pa,&pb,&pc,&r);
	while(1){
		if(!(na*a+nb*b+nc*c))break;
		na-=a;nb-=b;nc-=c;
		if(na<0){r+=na*pa;na=0;}
		if(nb<0){r+=nb*pb;nb=0;}
		if(nc<0){r+=nc*pc;nc=0;}
		if(r<0)break;
		ans++;
	}
	if(r>0)ans+=r/(a*pa+b*pb+c*pc);
	printf("%I64d\n",ans);
	return 0;
}
