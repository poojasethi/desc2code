#include <iostream>
#include <string>
#include <cstring>
#include <cstdio>
using namespace std;

int main(){
	ios::sync_with_stdio(false);
	string s;
	int n[3],p[3],m[3];
	long long r;
	memset(m,0,sizeof m);
	cin>>s;
	for(unsigned i=0;i<s.length();i++){
		if(s[i]=='B')m[0]++;
		else if(s[i]=='S')m[1]++;
		else m[2]++;
	}
	cin>>n[0]>>n[1]>>n[2]>>p[0]>>p[1]>>p[2]>>r;
	for(int i=0;i<3;i++) if(m[i]==0) n[i]=p[i]=0;
	long long ans=0;
	while(n[0]||n[1]||n[2]){
		for(int i=0;i<3;i++){
			n[i]-=m[i];
			if(n[i]<0){
				r+=p[i]*n[i];
				n[i]=0;
			}
		} 
		if(r>=0) ans++;
		else break;
	}
	if(r>=0) ans+=r/(p[0]*m[0]+p[1]*m[1]+p[2]*m[2]);
	cout<<ans;
}
