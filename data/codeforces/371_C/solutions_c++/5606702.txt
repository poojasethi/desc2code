#include<iostream>
using namespace std;
main(){
string s;
cin>>s;
long long n1,n2,n3,s1,s2,s3,p1=0,p2=0,p3=0;
long long r;
for(int i=0;i<s.size();i++){
if(s[i]=='B')p1++;
if(s[i]=='S')p2++;
if(s[i]=='C')p3++;
}
cin>>n1>>n2>>n3>>s1>>s2>>s3;
cin>>r;
long long w=0;
while((n1>0&&p1>0)||(n2>0&&p2>0)||(n3>0&&p3>0)){
if(p1<=n1){n1-=p1;}
else{if(r>=(p1-n1)*s1){r-=(p1-n1)*s1;n1=0;}else break;}
if(p2<=n2){n2-=p2;}
else{if(r>=(p2-n2)*s2){r-=(p2-n2)*s2;n2=0;}else break;}
if(p3<=n3){n3-=p3;}
else{if(r>=(p3-n3)*s3){r-=(p3-n3)*s3;n3=0;}else break;}
w++;
}
long long koszt=p1*s1+p2*s2+p3*s3;
w+=r/koszt;
cout<<w<<"\n";
}
