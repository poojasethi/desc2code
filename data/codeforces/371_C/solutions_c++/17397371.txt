#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	long long int nb,ns,nc,pb,ps,pc,b=0,s=0,c=0,count=0,cost,r,f;
	char ch[101];
	cin>>ch>>nb>>ns>>nc>>pb>>ps>>pc>>r;
	for(int i=0;i<strlen(ch);i++)
	{
		if(ch[i]=='B')b++;
		else if(ch[i]=='S')s++;
		else c++;
	}if(!b)nb=0;if(!c)nc=0;if(!s)ns=0;
	while(nb+ns+nc)
	{
		cost=0;
		if(nb<b)cost+=(b-nb)*pb,nb=b;
		if(nc<c)cost+=(c-nc)*pc,nc=c;
		if(ns<s)cost+=(s-ns)*ps,ns=s;
        if(cost<=r)nb-=b,nc-=c,ns-=s,count++,r-=cost;
        else {cout<<count;return 0;}
	}
	cout<<count+r/(b*pb+s*ps+c*pc);
}
