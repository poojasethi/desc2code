#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
char s[101];
int a=0,b=0,c=0,na,nb,nc,pa,pb,pc;
ll f(ll n)
{
    ll t=0;
    t+=max(0LL,n*a-na)*pa;
    t+=max(0LL,n*b-nb)*pb;
    t+=max(0LL,n*c-nc)*pc;
    return t;
}
int main()
{
    ll k,l=0,r=2000000000000LL,md;
    scanf("%s",s);
    for(int i=0;s[i];++i)
    if(s[i]=='B')++a;
    else if(s[i]=='S')++b;
    else ++c;
    scanf("%d%d%d%d%d%d%I64d",&na,&nb,&nc,&pa,&pb,&pc,&k);
    while(l<r)
    {
        md=l+r>>1;
        if(f(md)>k)r=md;
        else l=md+1;
    }
    printf("%I64d\n",l-1);
}
