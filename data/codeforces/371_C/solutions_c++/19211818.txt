#include <bits/stdc++.h>

using namespace std;


int main()
{
    long long l=0,r=2000000000000;
    long long  b=0,s=0,c=0,ru,pb,ps,pc,nb,ns,nc;
    string inp;
    cin>>inp;
    cin>>nb>>ns>>nc;
    cin>>pb>>ps>>pc;
    cin>>ru;
    for(int i=0; i<inp.size(); i++)
    {
        if(inp[i]=='B') b++;
        else if(inp[i]=='S') s++;
        else c++;
    }

    while (r-l>1)
    {
        long long m=(l+r)/2;

        long long sum=max((long long)0,b*m-nb)*pb+max((long long)0,s*m-ns)*ps+max((long long)0,c*m-nc)*pc;
        if(sum<=ru) l=m;
        else r=m;
    }

    cout<<l<<endl;
    return 0;
}
