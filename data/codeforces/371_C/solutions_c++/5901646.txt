//In the name of God
#include <algorithm>
#include <iostream>
using namespace std;

struct obj {
	int n, p, m;
	long long cost(long long q) {
		return max(0ll, (q * m - n) * p);
	}
} b, s, c;
string str;
int n;
long long r;

int main() {
	cin >> str >> b.n >> s.n >> c.n >> b.p >> s.p >> c.p >> r;
	b.m = count(str.begin(), str.end(), 'B');
	s.m = count(str.begin(), str.end(), 'S');
	c.m = count(str.begin(), str.end(), 'C');
	long long st = 0, ed = r + 1000;
	while (ed - st > 1) {
		long long mid = st + ed >> 1;
		(b.cost(mid) + c.cost(mid) + s.cost(mid) <= r? st: ed) = mid;
	}
	cout << st << '\n';
	return 0;
}
