#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define LL __int64
int main()
{
	char a[101];
	gets(a);
	LL hb=0,hs=0,hc=0;
	for(int i=0;i<strlen(a);i++)
	{
		if(a[i]=='B') hb++;
		if(a[i]=='S') hs++;
		if(a[i]=='C') hc++;
	}
	LL nb,ns,nc;	scanf("%I64d%I64d%I64d",&nb,&ns,&nc);
	LL pb,ps,pc;	scanf("%I64d%I64d%I64d",&pb,&ps,&pc);
	LL r;			scanf("%I64d",&r);
	LL ll=0,rr=r+nb+ns+nc,mid,t;
	while((rr-ll)>1)
	{
		mid=(rr+ll)/2;
		t=r;
		if(mid*hb>nb)	t-=(mid*hb-nb)*pb;
		if(mid*hs>ns)	t-=(mid*hs-ns)*ps;
		if(mid*hc>nc)	t-=(mid*hc-nc)*pc;
		if(t>=0)
		ll=mid;
		else
		rr=mid;
	}
	printf("%I64d\n",ll);
}
