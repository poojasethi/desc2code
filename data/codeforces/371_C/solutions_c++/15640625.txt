#include<bits/stdc++.h>
using namespace std;
string d;
long long n[3],ebr,esg,ech,pbr,psg,pch,m,s,e=1e13,t,ans,b,a;
int main(){
cin>>d>>ebr>>esg>>ech>>pbr>>psg>>pch>>m;
for(int i=0;i<d.size();i++){
    if(d[i]=='B')n[0]++;
    if(d[i]=='S')n[1]++;
    if(d[i]=='C')n[2]++;
}
while(s<e){
    long long mid=s+(e-s)/2;
    t=max((mid*n[0])-ebr,a)*pbr;
    t+=max((mid*n[1])-esg,a)*psg;
    t+=max((mid*n[2])-ech,a)*pch;
    if(t<m){
        s=mid;
    }
    else if(t>m){
        e=mid;
    }
    else{
        ans=mid;
        break;
    }
    if(b==mid){
        ans=mid;
        break;
    }
    b=mid;
}
cout<<ans<<endl;
}
