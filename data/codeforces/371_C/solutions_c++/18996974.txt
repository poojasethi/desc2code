#include <stdio.h>
#include <iostream>
using namespace std;

int main ()
{
	char a[1000];
	long long val,b1,s1,c1,b2,s2,c2,ans,i,j,b=0,s=0,c=0,an=0;
	scanf ("%s",a);
	cin>>b1>>s1>>c1>>b2>>s2>>c2>>val;
	for (i=0;a[i]!='\0';i++){
		if(a[i]=='B') b++;
		if(a[i]=='S') s++;
		if(a[i]=='C') c++;
	}
	i=0;j=1e15;
	while (i<=j)
	{
		long long mid=(i+j)/2;
		an=val;
		if(mid*b>b1) an=an-(mid*b-b1)*b2;
		if(mid*s>s1) an=an-(mid*s-s1)*s2;
		if(mid*c>c1) an=an-(mid*c-c1)*c2;
		if(an<0) j=mid-1;
		else {
			ans=mid;
			i=mid+1;
		}
	}
	cout<<ans<<endl;
	return 0;
}
