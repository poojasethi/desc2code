#include <iostream>

using namespace std;

int main()
{
	string s; long long a, b, c, pa, pb, pc, r, na, nb, nc; a = b = c = 0ll;
	cin >> s >> na >> nb >> nc >> pa >> pb >> pc >> r;
	for (int i = 0; i < s.size(); i++)
	{
		a += (s[i] == 'B');
		b += (s[i] == 'S');
		c += (s[i] == 'C');
	}
	long long st = 0ll, e = 1000ll * 1000 * 1000 * 1000 * 1000;
	while (e - st > 1ll)
	{
		long long mid = (e + st) >> 1;
		long long tmp = max(0ll, a * mid - na) * pa + max(0ll, b * mid - nb) * pb + max(0ll, c * mid - nc) * pc;
		if (tmp <= r) st = mid;
		else e = mid;
	}
	cout << st << endl;
	return 0;
}
