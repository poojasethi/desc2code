#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int MAXL=109;

char str[MAXL];
long long cost[3],cnt[3],p[3],j,sum,lef,rig,mid,tmp;

int main()
{
	cin >> str;
	for(j=0;str[j]!='\0';++j)
	{
		if(str[j]=='B')
			cost[0]++;
		else if(str[j]=='S')
			cost[1]++;
		else
			cost[2]++;
	}
	for(j=0;j<3;++j)
		cin >> cnt[j];
	for(j=0;j<3;++j)
		cin >> p[j];
	cin >> sum;
	lef=0;
	rig=sum+cnt[0]+cnt[1]+cnt[2];

	while(lef<rig-1)
	{
		mid=(lef+rig)/2;
		for(tmp=j=0;j<3;++j)
			tmp+=max(0ll,(mid*cost[j]-cnt[j]))*p[j];

		if(tmp<=sum)
			lef=mid;
		else
			rig=mid;
	}
	cout << lef << endl;
    return 0;
}
