#include<bits/stdc++.h>
using namespace std;
long long a,b,c;
long long x,y,z;
long long d,e,f;
long long r;
bool cal(long long m)
{
	long long cal=0;
	cal+=max((long long)0,m*a-d)*x;
	cal+=max((long long)0,m*b-e)*y;
	cal+=max((long long)0,m*c-f)*z;
	if(cal<=r)
		return true;
	else
		return false;	
}
int main()
{
	int i=0;
	string s;
	cin>>s;
	for(i=0;i<s.length();i++)
	{
		if(s[i]=='B')
			a++;
		else if(s[i]=='S')
			b++;
		else
			c++;		
	}
	cin>>d>>e>>f;
	cin>>x>>y>>z;
	cin>>r;
	long long hi=10000000000000L;
	long long lo=0;
	while(lo<hi)
	{
		long long mid=(hi+lo+1)/2;
		if(cal(mid))
			lo=mid;
		else
			hi=mid-1;	
	}
	cout<<lo<<endl;
	return 0;
}
