#include <iostream>  
#include <cstdio>  
#include <cstring> 
#define eps 1e13  
using namespace std;  
typedef long long ll;  
ll ss[4],z[4],zz[4];  
int main()  
{  
  char s[102];  
  ll i,p,l=0,r=eps,mid,cost;  
  scanf("%s",s);  
  for(i=0;i<strlen(s);i++)  
  {  
    if(s[i]=='B') ss[0]++;  
    else if(s[i]=='S') ss[1]++;  
    else ss[2]++;  
  }  
  cin>>z[0]>>z[1]>>z[2]>>zz[0]>>zz[1]>>zz[2]>>p;  
  while(l+1<r)  
  {  
    mid=(l+r)>>1;  
    cost=0;  
    for(i=0;i<3;i++)  
      if(mid*ss[i]>z[i]) cost+=(mid*ss[i]-z[i])*zz[i];  
    if(cost<=p) l=mid;  
    else r=mid;  
  }  
  cout<<l<<endl;  
  return 0;  
}
