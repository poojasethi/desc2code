#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main() {
string s;
long long nb,ns,nc,pb,ps,pc,B=0,S=0,C=0;
long long r,ans=0,price=0,saved=0;
cin>>s;
cin>>nb>>ns>>nc;
cin>>pb>>ps>>pc;
cin>>r;
for(int i=0;i<s.size();i++){
if(s[i]=='B')       B++;
else if(s[i]=='S')  S++;
else C++;
}
price=B*pb+S*ps+C*pc; 
long long st=0,en=100000000000000;
while(st<=en){
 long long m=(st+en)/2;
 long long cost=price*m-saved;
 cost-=(min(nb,m*B)*pb+min(ns,m*S)*ps+min(nc,m*C)*pc);
        //cout<<m<<" "<<cost<<"\n";
  if(cost<=r){
 ans=m;
 st=m+1;
    }
   else en=m-1;
  }
    cout<<ans;
    return 0;
}
