#include<iostream>
#include<string>

using namespace std;

long long l=0,r=2000000000000;
long long  b=0,s=0,c=0,ruble,pb,ps,pc,nb,ns,nc;
string reci;

int main()
{
	cin>>reci;
	cin>>nb>>ns>>nc;
	cin>>pb>>ps>>pc;
	cin>>ruble;
	for(int i=0;i<reci.size();i++)
	{
		if(reci[i]=='B') b++;
		else if(reci[i]=='S') s++;
		else c++;
	}

	while (r-l>1) {
		long long m=(l+r)/2;

		long long sum=max((long long)0,b*m-nb)*pb+max((long long)0,s*m-ns)*ps+max((long long)0,c*m-nc)*pc;
		if(sum<=ruble) l=m;
		else r=m;
	}

	cout<<l<<endl;
	return 0;
}
