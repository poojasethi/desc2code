#include <iostream>
#include <cstdio>
#include <string>
#include <cstring>
using namespace std;
typedef long long ll;
char s[110];
int a1,a2,a3,p1,p2,p3,s1,s2,s3;
ll m;
bool valid(ll x)
{
	ll cost=max(0LL,x*s1-a1)*p1+max(0LL,x*s2-a2)*p2+max(0LL,x*s3-a3)*p3;
	return cost<=m;
}
int main()
{
	scanf("%s",s);
	scanf("%d%d%d%d%d%d%I64d",&a1,&a2,&a3,&p1,&p2,&p3,&m);
	for (int i=0; i<strlen(s); i++)
	{
		if (s[i]=='B') s1++;
		if (s[i]=='S') s2++;
		if (s[i]=='C') s3++;
	}
	ll l=0,r=1LL<<45;
	while (l<=r)
	{
		ll mid=(l+r)>>1;
		if (valid(mid)) l=mid+1;
		else r=mid-1;
	}
	printf("%I64d\n",l-1);
	return 0;
}
