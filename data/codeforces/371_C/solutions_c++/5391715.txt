#include <iostream>
#include <cmath>
#include <cstdio>

using namespace std;

typedef long long ll;

int main()
{
	int a, b, c, n, x, y, z;
	ll ct1, ct2, ct3, gct;
	string s;
	x = y = z = 0;
	cin >> s;
	n = s.size();
	for (int i = 0; i < n; i++)
		if (s[i] == 'B')
			x++;
		else if (s[i] == 'S')
			y++;
		else
			z++;
	cin >> a >> b >> c >> ct1 >> ct2 >> ct3 >> gct;
	ll l = 0, r = (ll) 1e13, m;
	while (l + 1 < r)
	{
		m = (l + r) / 2;
		if (max (0ll, x * m - a) * ct1 + max (0ll, y * m - b) * ct2 + max (0ll, z * m - c) * ct3 <= gct)
			l = m;
		else
			r = m;
	}
	
	cout << l << endl;
	return 0;
}
