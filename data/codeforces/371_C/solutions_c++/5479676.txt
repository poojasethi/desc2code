#include <stdio.h>
#include <iostream>
#include <string>
#define ll long long
using namespace std;
ll l=0,r=2000000000000,m,w,b=0,s=0,c=0,f,pb,ps,pc,nb,ns,nc,x,y,z;
string t;
int main( ){
	cin>>t>>nb>>ns>>nc>>pb>>ps>>pc>>f;
	for (int i=0;i<t.length();i++)
		if (t[i]=='B') b++;
		else if (t[i]=='S') s++;
		else if (t[i]=='C') c++;
	while (r-l>1){
		m=(l+r)/2;
		if (b*m-nb<0) x=0; else x=b*m-nb;
		if (s*m-ns<0) y=0; else y=s*m-ns;
		if (c*m-nc<0) z=0; else z=c*m-nc;
		w=x*pb+y*ps+z*pc;
		if (w<=f) l=m; else r=m;
	}
	cout<<l<<endl;
	return 0;
}
