#include<iostream>
#include<string>
#define ll long long
#define INF 2000000000000 
using namespace std;
ll nb,ns,nc,pb,ps,pc,cb=0,cs=0,cc=0,r;
string s;
bool divid(ll x){
	ll sum=0;
	if(x*cb-nb>0) sum+=pb*(x*cb-nb);
	if(x*cs-ns>0) sum+=ps*(x*cs-ns);
	if(x*cc-nc>0) sum+=pc*(x*cc-nc);
	if(sum>r) return 1;
	else return 0;
} 
int main(){
	cin>>s;
	ll mid;
	ll len=s.length();
	for(ll i=0;i<len;i++){
		if(s[i]=='B') cb++;
		if(s[i]=='S') cs++;
		if(s[i]=='C') cc++;
	}  
	cin>>nb>>ns>>nc>>pb>>ps>>pc>>r;
	ll l=0;ll r=INF;
	while(l<r){
		mid=(l+r)>>1;
		if(divid(mid)) r=mid;
		else l=mid+1; 
	}
	cout<<l-1;
	return 0;
}
