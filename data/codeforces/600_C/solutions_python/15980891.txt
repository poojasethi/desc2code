from collections import Counter

doubles = singles = ''
for c, count in Counter(raw_input()).iteritems():
    doubles += count / 2 * c
    if count % 2 != 0:
        singles += c
doubles = sorted(doubles)
singles = sorted(singles)
mid = '' if len(singles) % 2 == 0 else singles[len(singles) / 2]
first_half = sorted(doubles + singles[:len(singles) / 2])
print ''.join(first_half) + mid + ''.join(first_half[::-1])
