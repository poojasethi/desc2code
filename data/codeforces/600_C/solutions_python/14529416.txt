from collections import Counter

s = str(raw_input())
n = len(s)

doubles = []
singles = []

count = sorted(Counter(s).items())

for x in count:
    doubles += x[0] * (x[1] / 2)
    if x[1] % 2 == 1:
        singles += [x[0]]

final = (doubles+singles+doubles)[:n]
final[:n/2]=sorted(final[:n/2])
final[n - n / 2:n] = final[::-1][n-n/2:n]
print(''.join(final))
