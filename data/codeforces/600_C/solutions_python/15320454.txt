num = [0]*26
for ch in raw_input():
    num[ord(ch)-97] += 1
l = 0
r = 25
while l != r:
    if num[l]%2 == 0:
        l += 1
    elif num[r]%2 == 0:
        r -= 1
    else:
        num[l] += 1
        num[r] -= 1
center = chr(97+l) if num[l] % 2 else ''
if num[l] % 2:
    num[l] -= 1
string = ""
for i, j in enumerate(num):
    string += chr(97+i)*(j/2)
print string + center + string[::-1]
