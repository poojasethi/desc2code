st = raw_input()
lst = list(st)
d = {}
for i in lst:
    d[i] = d.get(i,0) + 1

c = []
for i in d:
    if d[i] % 2 == 1:
        c.append(i)
c.sort()
l = 0
r = len(c) - 1
while l < r:
    d[c[r]] -= 1
    d[c[l]] += 1
    l += 1
    r -= 1

if l == r:
    mid = [c[l]]
else:
    mid = ['']
ans = []
for i in d:
    while d[i] > 1:
        d[i] -= 2
        ans.append(i)

ans.sort()
print ''.join(ans + mid + ans[::-1])
