s=raw_input()

cnt=[0]*26;
ev=[]
od=[]

for x in s:
	cnt[ord(x)-ord('a')]+=1;

for i in range(0,26):
	if cnt[i]%2==0:
		ev.append(i);
	else:
		od.append(i);

for i in range(len(od)-1,len(od)/2-1,-1):
	cnt[od[i]]-=1;
	cnt[od[len(od)-1-i]]+=1;
ans=""
for i in range(0,26):
	for j in range(0,cnt[i]/2):
		ans+=str(chr(i+ord('a')));

for i in range(0,26):
	if cnt[i]%2==1:
		ans+=str(chr(i+ord('a')));
for i in range(25,-1,-1):
	for j in range(0,cnt[i]/2):
		ans+=str(chr(i+ord('a')));

print(ans)
