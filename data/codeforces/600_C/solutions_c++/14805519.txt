#include<bits/stdc++.h>
using namespace std;
int A[26];
int main(){
	std::ios::sync_with_stdio(0);
	string s;
	cin>>s;
	
	int i,j;
	for(i=0;i<s.length();i++){
		A[s[i]-'a']++;
	}
	j=26;
	for(i=0;i<26;i++){
		if(A[i]%2){
			while(j>i && A[j]%2==0)
			j--;
			if(j!=i){
				A[i]++;
				A[j]--;
			}
		}
	}
	string p = "";
	int flag = -1;
	char x;
	for(i=0;i<26;i++){
		if(A[i]%2)
		flag = i;
		j = A[i]/2;
		while(j){
			p += (i+'a');
			j--;
		}
	}
	cout<<p;
	if(flag!=-1){
		x = flag + 'a';
		cout<<x;
	}
	reverse(p.begin(),p.end());
	cout<<p<<endl;
}
