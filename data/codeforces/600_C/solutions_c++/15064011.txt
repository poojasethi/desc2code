#include <iostream>
#include <cstdio>
using namespace std;

int cnt,a[30];
string s,ans;
int main(){
    int i,j;
    cin>>s;
    for(i=0; i<s.size(); i++) a[s[i]-'a']++;
    for(i=0; i<26; i++) cnt += a[i]%2;
    for(i=j=0; j<cnt/2; i++){
        if(a[i]%2) a[i]++, j++;
    }
    for(i=25, j=0; j<cnt/2; i--){
        if(a[i]%2) a[i]--, j++;
    }
    for(i=0; i<26; i++){
        for(j=0; j<a[i]/2; j++) ans += i+'a';
    }
    for(i=0; i<26; i++){
        if(a[i]%2) ans += i+'a';
    }
    for(i=25; i>=0; i--){
        for(j=0; j<a[i]/2; j++) ans += i+'a';
    }
    cout<<ans;
    return 0;
}
