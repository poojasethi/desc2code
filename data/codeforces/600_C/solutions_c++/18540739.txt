#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <memory.h>

#define NN 200010

using namespace std;

char str[NN], ans[NN];
int cnt[26];

int main (){
	
	scanf("%s",str);
	int N=strlen(str),i,j,id;
	
	for(i=0; i<N; i++){
		cnt[str[i]-'a']++;
	}
	
	for(i=25; i>=0; i--){
		if(cnt[i]%2==0) continue;
		for(j=0; j<i; j++){
			if(cnt[j]%2){cnt[i]--; cnt[j]++; break;}
		}
		id=i;
	}
	
	int st=0, ed=N-1;
	for(i=0; i<26; i++){
		for(j=0; j<cnt[i]/2; j++){
			ans[st++]=ans[ed--]='a'+i;
		}
	}
	
	if(N%2) ans[N/2]='a'+id;
	
	puts(ans);
	return 0;
}
