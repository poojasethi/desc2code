#include<cstdio>
#include<cstring>
int main()
{
	int a[500] = { 0 }, i, j, len, t;
	char s[200005];
	scanf("%s", s); len = strlen(s);
	for (i = 0; i < len; i++)
		a[s[i]]++;
	i = 'a'; j = 'z';
	while (i<j)
	{
		while ((a[i] % 2 == 0)&&(i<j))
			i++;
		while ((a[j] % 2 == 0)&&(i<j))
			j--;
		if (i < j)
		{
			a[i]++; a[j]--; i++; j--;
		}
	}
	t = -1;
	for (i = 'a'; i <= 'z'; i++)
	{
		if (a[i] % 2 == 1)
		{
			t = i; a[i]--;
		}
		for (j = 1; j <= a[i] / 2; j++)
			printf("%c", i);
		a[i] = a[i] / 2;
	}
	if (t!=-1) printf("%c", t);
	for (i = 'z'; i >= 'a'; i--)
		for (j = 1; j <= a[i]; j++)
			printf("%c", i);
	printf("\n");
	return 0;
}
