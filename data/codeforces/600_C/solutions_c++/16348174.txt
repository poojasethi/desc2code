#include <cstdio>
#include <algorithm>
#include <cstring>
char str[200005];
char res[200005];
int cnt[26];
int main(){
	scanf("%s",str);
	int l=strlen(str);
	for ( int i=0;i<l;i++)
		cnt[str[i]-'a']++;

	for ( int i=0;i<26;i++){
		for ( int j=25;j>i;j--){
			if ( cnt[i]%2==1 && cnt[j]%2==1 ){
				cnt[i]++;
				cnt[j]--;
			}
		}
	}
	int p=0;
	for ( int i=0;i<l/2;i++){
		while ( cnt[p]<2 ) p++;
		res[i]='a'+p;
		res[l-1-i]='a'+p;
		cnt[p]-=2;	
	}
	res[l]=0;
	if ( l%2==1 ){
		int p=0;
		for ( int i=0;i<26;i++)
			if ( cnt[i]>0 ) p=i;
		res[l/2]='a'+p;	
	}
	printf("%s\n",res);
	return 0;
}
