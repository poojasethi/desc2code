#include<cstdio>
#include<cstring>
char str[500000];
int len;
int num[26];
int main(){
	//freopen("in","r",stdin);
	scanf("%s",str);
	len=strlen(str);
	for(int i=0;i<len;i++)
		num[str[i]-'a']++;
	int l,r;
	for(l=0,r=25;l<r;){
		if((num[l]&1)==0)
			l++;
		else if((num[r]&1)==0)
			r--;
		else
			num[r--]--,num[l++]++;
	}
	for(int i=0;i<26;i++)
		for(int k=0;k<(num[i]>>1);k++)
			printf("%c",i+'a');
	if(num[l]&1)
		printf("%c",l+'a');
	for(int i=25;i>=0;i--)
		for(int k=0;k<(num[i]>>1);k++)
			printf("%c",i+'a');
	return 0;
}
