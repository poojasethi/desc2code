#include<bits/stdc++.h>
using namespace std;
int main()
{
   string s;
   cin>>s;
vector<int>v;
int i,j,a[30]={0},an=-1;
   for(i=0;i<s.size();i++)
     a[(int)(s[i]-'a')]++;
   for(i=0;i<26;i++)
     if(a[i]%2==1)
         v.push_back(i);
    for(i=0;i<v.size()/2;i++)
       {
          a[v[i]]++;
          a[v[v.size()-i-1]]--;
        }
        for(i=0;i<26;i++)
           if(a[i]%2==1)
             an=i;
    for(i=0;i<26;i++)
        for(j=0;j<a[i]/2;j++)
          cout<<char(i+97);
    if(an!=-1)cout<<char(an+97);
    for(i=25;i>=0;i--)
        for(j=0;j<a[i]/2;j++)
          cout<<char(i+97);
        return 0;
          }
