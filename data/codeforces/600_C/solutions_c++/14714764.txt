#include<bits/stdc++.h>
using namespace std;


string s;
int bad;
int cnt[256];

int main()
{
    cin>>s;
    for(int i=0;i<s.size();i++)cnt[s[i]]++;
    for(char q='a';q<='z';q++)bad+=cnt[q]%2;
    while(bad>1)
    {
        bad-=2;
        char l,r;
        for(char q='a';q<='z';q++)if(cnt[q]%2==1)r=q;
        for(char q='z';q>='a';q--)if(cnt[q]%2==1)l=q;
        cnt[r]--;
        cnt[l]++;
    }
    for(char q='a';q<='z';q++)for(int j=1;j<=cnt[q]/2;j++)cout<<q;
    for(char q='a';q<='z';q++)if(cnt[q]%2==1)cout<<q;
    for(char q='z';q>='a';q--)for(int j=1;j<=cnt[q]/2;j++)cout<<q;
}
