#include <iostream>
#include <string>

using namespace std;

int cnt[123], l='a', r='z';
string s, ans;
char mid;

int main()
{
    cin >> s;
    for (int i=0; i<s.length(); i++) cnt[s[i]]++;
    while(l<r)
    {
        while(cnt[l]%2==0) l++;
        while(cnt[r]%2==0) r--;
        cnt[l]++;
        cnt[r]--;
    }
    for (int i='a'; i<='z'; i++)
    {
        if(cnt[i]%2==1) mid=i;
        for (int j=0; j<cnt[i]/2; j++)
            ans+=i;
    }
    if(mid) ans+=mid;
    for (int i='z'; i>='a'; i--)
    {
        for (int j=0; j<cnt[i]/2; j++)
            ans+=i;
    }
    cout << ans;
    return 0;
}
