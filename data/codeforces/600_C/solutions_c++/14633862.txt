#include <iostream>

using namespace std;

int a[26];

int main()
{
    string s;
    cin >>s;
    for(int i=0;i<s.size();i++)
        a[s[i]-'a']++;
    int k=0;
    int l=25;
    while(true)
    {
        while(k<l && a[k]%2==0) k++;
        while(k<l && a[l]%2==0) l--;
        if(k==l) break;
        a[k]++;
        a[l]--;
    }
    for(int i=0;i<26;i++)
    for(int j=0;j<a[i]/2;j++)
        cout << char('a'+i);

    for(int i=0;i<26;i++)
    if(a[i]%2)
        cout << char('a'+i);
    for(int i=25;i>=0;i--)
    for(int j=0;j<a[i]/2;j++)
        cout << char('a'+i);

    return 0;
}
