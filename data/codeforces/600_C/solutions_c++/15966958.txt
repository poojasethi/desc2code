#include <stdio.h>
#include <string.h>
int main()
{
	char s[200010];
	int i,j,a[26]={0},b[27]={0},num=0;
	gets(s);
	for(i=0;s[i]!='\0';i++) a[s[i]-97]++;
	for(i=0;i<26;i++)
	{
		if(a[i]%2==1)
		{
			num++;
			b[num]=i;
		}
	}
	for(i=1;i<=num/2;i++) a[b[i]]++;
	if(num%2==1)
	{
		for(i=num/2+2;i<=num;i++) a[b[i]]--;
	}
	else
	{
		for(i=num/2+1;i<=num;i++) a[b[i]]--;
	}
	for(i=0;i<26;i++)
	{
		for(j=1;j<=a[i]/2;j++) printf("%c",i+97);
	}
	if(num%2==1) printf("%c",b[num/2+1]+97);
	for(i=25;i>=0;i--)
	{
		for(j=1;j<=a[i]/2;j++) printf("%c",i+97);
	}
	return 0;
}
