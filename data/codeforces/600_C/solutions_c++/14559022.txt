#include<cstdio>
#include<algorithm>
#include<iostream>

using namespace std;

int main(){
	string s, r, r_rev;
	int i, j, len, k=0;
	int count[26] ={0};
	char ch;
	cin >> s;
	len = s.size();
	for(i=0; i<len; i++) count[s[i]-'a']++;
	i = 0; j = 26;
	while(i < j)
	{
		while(i<j && count[i]%2==0)
			i++;
		while(j>i && count[j]%2==0)
			j--;
		count[i]++;
		count[j]--;
	}
	i=0; j = 26;
	while(i<j)
	{
		if(count[i]%2) ch = i+'a';
		for(k=0;k<count[i]/2;k++) r += char(i+'a');
		i++;
	}
	r_rev = r;
	reverse(r_rev.begin(),r_rev.end());
	if(len%2) r += ch + r_rev;
        else r += r_rev;	
	cout << r << endl;
	return 0;
}
