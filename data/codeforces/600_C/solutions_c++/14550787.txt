#include <cstdio>
#define M 200005

char s[M],t[M];
int n,a[30],b=25,r;

int main () {
	scanf("%s", s);
	for (int i = 0; s[i]; i++) {
		a[s[i]-'a']++;
		n++;
	}
	for (int i = 0; i < 26; i++) {
		if (a[i]%2==1) {
			while (a[b]%2==0 && i<b) b--;
			if (i == b) {
				t[n/2]=i+'a';
				a[i]--;
			} else {
				a[i]++;
				a[b]--;
			}
		}
		for (int j = 0; j < a[i]/2; j++) {
			t[r] = t[n-r-1] = i+'a';
			r++;
		}
	}
	t[n] = 0;
	printf("%s\n", t);
	return 0;
}
