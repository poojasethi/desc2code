
#include <bits/stdc++.h>

using namespace std;

int c[256];

int main()
{
	ios_base::sync_with_stdio(false);

	string s;
	cin>>s;
	int sl = s.length();
	
	for (int i=0 ; i<sl ; i++) c[s[i]]++;
	
	int i='a' , j='z';
	int ans = 0;
	while (i<j)
	{
		while (i<j && c[i]%2==0) i++;
		while (i<j && c[j]%2==0) j--;
		if (i<j) ans++ , c[j]--, c[i]++;
		i++ , j--;
	}
	
	char ex = 0;
	if (sl%2) for (int i='a' ; i<='z' ; i++) if (c[i]%2) ex = i;
	
	string t = "";
	for (int i='a' ; i<='z' ; i++) for (int j=0 ; j<c[i]/2 ; j++) t+=i;
	
	cout<<t;
	if (ex) cout<<ex;
	for (int i=t.length()-1 ; i>=0 ; i--) cout<<t[i];
	cout<<"\n";
	
	return 0;
}
