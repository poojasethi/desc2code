#include<bits/stdc++.h>
#include <string>
#include <vector>
using namespace std;
string s;
int a[27];
vector<int>v;
string ans;
int main()
{
    cin>>s;
    for(int i=0;i<s.length();i++)
        a[s[i]-'a']++;
    for(int i=0;i<=26;i++)
    {
        for(int j=0;j<a[i]/2;j++)
            ans+=char(i+'a');
        if(a[i]%2!=0)
            v.push_back(i);
    }
    for(int i=0;i<v.size()/2;i++)
        ans+=char(v[i]+'a');
    sort(ans.begin(),ans.end());
    cout<<ans;
    if(v.size()%2!=0)
        cout<<char(v[v.size()/2]+'a');
    for(int i=ans.size()-1;i>=0;i--)
        cout<<ans[i];
    cout<<endl;
    return 0;
}
