#include <bits/stdc++.h>
using namespace std;
int al[27];
int main(){
	string s;cin>>s;
	int i,j,n=s.size(),b=25,p=0;
	for(i=0;i<n;++i)++al[s[i]-'a'];
	for(i=0;i<26;++i){
		if(al[i]&1){
			while((al[b]&1)==0 && i<b)--b;
			if(i==b)
				--al[i],s[n/2]=i+'a';
			else
				--al[b],++al[i];
		}
		for(j=0;j<(al[i]/2);++j)
			s[p]=i+'a', s[n-1-p]= i+'a',++p;
	}
	cout<<s<<endl;
}
