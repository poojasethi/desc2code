#include<cstdio>
#include<cstring>

int n;
char s[222222];
int tot[55];

int main() {
	scanf("%s", s);
	n=strlen(s);
	for(int i=0;i<n;i++)
		tot[s[i]-'a']++;
	int l=0, r=25;
	while(l<r) {
		while(l<r&&tot[l]%2==0)l++;
		while(l<r&&tot[r]%2==0)r--;
		if(l<r) {
			tot[l]++;
			tot[r]--;
		}
	}
	int ch = -1;
	if(l==r&&tot[l]%2==1) ch=l;
	for(int i=0;i<26;i++)
		for(int j=tot[i]/2;j>0;j--)
			printf("%c", i+'a');
	if(ch>=0)printf("%c", ch+'a');
	for(int i=25;i>=0;i--)
		for(int j=tot[i]/2;j>0;j--)
			printf("%c", i+'a');
	printf("\n");
	
	return 0;
}
