#include <bits/stdc++.h>
using namespace std;
int main() {
	string s;
	int c[26], i, n, j, k;
	memset(c, 0, sizeof(c));
	getline(cin, s);
	n = s.size();
	for(i = 0; i < n; i++)
		c[s[i] - 'a']++;
	vector<int> odd;
	for(i = 0, j = 0; i < 26; i++)
		if(c[i] & 1) odd.push_back(i);
	for(i = 0, j = odd.size() - 1; i < j; i++, j--) {
		c[odd[i]]++;
		c[odd[j]]--;
	}
	for(i = 0, k = 0; i < 26; i++) {
		if(c[i] & 1) {
			c[i]--;
			s[n/2] = 'a' + i;
		}
		if(c[i] == 0) continue;
		for(j = 0; j < c[i]; j += 2) {
			s[k] = s[n-k-1] = 'a' + i;
			k++;
		}
	}
	cout << s << endl;
	return 0;
}
