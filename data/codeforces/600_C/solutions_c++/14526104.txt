#include <iostream>
#include <cstdio>
using namespace std;
#define num 200005
char g[num];
int a[150];
int main ()
{
//	freopen("test.txt","r",stdin);
	scanf("%s",g);
	int i,len,p='a',q='z';
	for (i=0;g[i]!='\0';i++)
		a[g[i]]++;
	len = i;
	i=0;
	while (p<q){
		while (!a[p] || (a[p]&1)==0)p++;
		while (!a[q] || (a[q]&1)==0)q--;
		if (p!=q){
			a[p]++;
			a[q]--;
		}
		p++;
		q--;
	}
	for (int j='a';j<='z';j++){
		if (a[j]&1){
				g[len>>1]=j;
				a[j]--;
		}
		while (a[j]>0){
			g[i]=g[len-i-1] = j;
			i++;
			a[j]-=2;
		}
	}
	printf("%s",g);
    return 0;
}
