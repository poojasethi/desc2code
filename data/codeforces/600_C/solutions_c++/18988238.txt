#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;
string s;
int b['z'+1];
int n;
int main()
{
	cin>>s;
	n=s.size();
	memset(b,0,sizeof(b));
	for(int i=0;i<n;i++)
		b[s[i]]++;
	int j='z',m=0;
	for(int i='a';i<='z';i++)
	{
		if(b[i]&1) 
		{
		while(b[j]%2==0&&i<j)
		j--;
		if(i==j)	{
			b[i]--;
			s[n/2]=i;
		}
		else 
		{
			b[i]++;
			b[j]--;
		}
		}
		for(int k=0;k<b[i]/2;k++)
		{
			s[m]=i;
			s[n-m-1]=i;
			m++;
			//cout<<m+k<<' '<<n-1-m-k<<' '<<i<<endl;
		}		
	}
	for(int i=0;i<n;i++)
        printf("%c",s[i]);
	cout<<endl;
}
