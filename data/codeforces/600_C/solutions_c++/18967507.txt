#include <bits/stdc++.h>
using namespace std;

#define N 300005
char s[N];
char c[N];
int alpha[26];
int main()
{
	scanf("%s", s);
	int n = strlen(s);

	for(int i = 0; i<n; i++)
	{
		alpha[s[i]-97]++;
	}

	int p1 = 0, p2 = 25;

	while(p1<p2)
	{
		while(p1<26 && alpha[p1]%2==0)
			p1++;
		while(p2>=0 && alpha[p2]%2==0)
			p2--;
		if(p1<26 && p2>=0 && p1<p2)
		{
			alpha[p1]++;
			alpha[p2]--;
			p1++;
			p2--;
		}
	}

	int j = 0;
	int k = 0;
	while(k<26)
	{
		while(alpha[k]>1)
		{
			c[j] = (char)(97+k);
			c[n-1-j] = (char)(97+k);
			alpha[k]-=2;
			j++;
		}

		if(alpha[k]==1)
			c[n/2] = (char)(97+k);
		k++;
	}

	cout << c << endl;
}
