#include<stdio.h>
#include<string.h>
char s[2001000];
int main(){
	int i,j,k;
	int a[30]={0},b[30]={0};
	gets(s);
	int l=strlen(s);
	for(i=0;i<l;i++)
		a[s[i]-96]++;//记下每个字母的个数
	int t=1;
	for(i=1;i<=26;i++)
	{
		if(a[i]%2==1)
		{
			b[t]=i;//记下哪几个字母的个数是单数
			t++;
		}
	}
	int x=t-1;//数量为单数的字母的个数
	for(i=1,j=x;i<j;i++,j--)
	{
		a[b[i]]++;
		a[b[j]]--;
	}
	if(x%2==1)  
        s[(l-1)/2]=b[(x+1)/2]+96;
	for (i=0,j=l-1;i<j;i++,j--)  
	{
		for(k=1;k<=26;k++)  
		{
			if(a[k]>1)
			{
			   a[k]=a[k]-2;
			   s[i]=k+96;
			   s[j]=k+96;
			   break;
			}
		}
	}
	puts(s);
	return 0;
}
