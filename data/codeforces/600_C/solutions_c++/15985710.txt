#include <cstdio>
#include <cstring>

#define N 200200

char s[N];
int n, f[30], m, a[30];

int main() {
    scanf("%s", s);
    n = strlen(s);
    for(int i = 0; i < n; i ++) f[s[i] - 'a'] ++;
    for(int i = 0; i < 26; i ++) {
        if(f[i] & 1) a[m ++] = i;
    }
    for(int i = 0; i < m / 2; i ++) {
        f[a[i]] ++; f[a[m-1-i]] --;
    }
    for(int i = 0; i < 26; i ++) for(int j = 0; j < f[i] / 2; j ++) printf("%c", 'a' + i);
    for(int i = 0; i < 26; i ++)  if(f[i] & 1) printf("%c", 'a' + i);
    for(int i = 25; i >= 0; i --) for(int j = 0; j < f[i] / 2; j ++) printf("%c", 'a' + i);
    puts("");
    return 0;
}
