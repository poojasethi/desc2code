#include<stdio.h>


int main(){
	
	char s[200005];
	int a[30],i,index=-1;
	
	for(i=0;i<30;i++) a[i]=0;
	
	scanf("%s",s);


	for(i=0;s[i];i++) a[s[i]-'a']++;
	
	int j=0;
	
	for(i=26;i>=0;i--){
	
		if(a[i]&1){
		
			while(j<i){
				if(a[j]&1) break;
				j++;
			}
			
			if(j<i){
				a[i]--;
				a[j]++;
				j=0;
			}
		}
	}
	
	for(i=0;i<26;i++){
		
		for(j=0;j<a[i]/2;j++) putchar(i+'a');
		if(a[i]&1) index = i;
	}
	if(index!=-1) putchar(index+'a');
	
	for(i=25;i>=0;i--){
		
		for(j=0;j<a[i]/2;j++) putchar(i+'a');
	}
	puts("");
	return 0;
}
