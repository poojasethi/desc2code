#include<bits/stdc++.h>
using namespace std;
char a[200005];
int co[26];
int main()
{
scanf("%s",&a);
int i;
for(i=0;a[i];i++)
co[a[i]-'a']++;
int od = 0;
vector<int> s;
for(i=0;i<26;i++)
if(co[i]%2){
od++;

s.push_back(i);
}
if(od>1)
{
int l = s.size();
for(i=0;i<l/2;i++)
{
co[s[i]]++;
co[s[l-1-i]]--;
}
}
od = -1;
for(i=0;i<26;i++)
if(co[i]%2){co[i]--;od = i;}
for(i=0;i<26;i++)
{
int c = co[i];
c = c/2;
while(c--)printf("%c",i+'a');
}
if(od!=-1)printf("%c",od+'a');
for(i=25;i>=0;i--)
{
int c = co[i];
c = c/2;
while(c--)printf("%c",i+'a');
}

    return 0;
}
