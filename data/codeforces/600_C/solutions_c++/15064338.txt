#include <iostream>
#include <cstdio>
using namespace std;

int n,p,a[30];
string s;
int main(){
    int i,j=25,k;
    cin>>s;
    n = s.size();
    for(i=0; i<n; i++) a[s[i]-'a']++;
    for(i=0; i<26; i++) {
        if(a[i]&1){
            for(; j>=0 && a[j]%2==0; j--);
            if(i==j) s[n/2] = i+'a', a[j]--;
            else a[i]++, a[j]--;
        }
        for(k=0; k<a[i]/2; k++, p++) s[p] = s[n-1-p] = i+'a';
    }
    cout<<s;
    return 0;
}
