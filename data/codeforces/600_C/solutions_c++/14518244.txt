#include <bits/stdc++.h>
using namespace std;

int n;
string s;
vector<int> x;
int c[30];

int main () {

    cin >> s;
    n = s.size();

    for (int i=0; i<n; i++) {
        c[ s[i]-'a' ]++;
    }

    s = "";

    for (int i=0; i<26; i++) {
        if (c[i]%2==1) {
            x.push_back(i);
        }

        for (int j=0; j<c[i]/2; j++) s+=(char)( 'a'+i );
    }

    for (int i=0; i<x.size()/2; i++) {
        s+=(char)( 'a'+x[i] );
    }

    sort(s.begin(),s.end());
    cout << s;

    if (n%2==1) {
        cout << (char)('a'+x[ x.size()/2 ]);
    }

    reverse(s.begin(), s.end());
    cout << s << endl;
}
