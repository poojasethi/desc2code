#include <bits/stdc++.h>

using namespace std;

char s[200008];
int num[288];

int main(){

	scanf("%s", s);
	int len = strlen(s);
	for(int i = 0; i < len; i++){
		num[s[i] - 'a']++;
	}
	for(int i = 0, j = 25; i < j; i++){
		if(num[i] & 1){
			while(i < j && !(num[j] & 1)) j--;
			if(i < j) num[i]++, num[j--]--;
		}
	}
	char ch = 0;
	for(int i = 0; i < 26; i++){
		//printf("%d   %d\n", i, num[i]);
		if(num[i] & 1) ch = i + 'a';
	}
	if(ch) s[len / 2] = ch;
	int j = len - 1, k = 0;
	for(int i = 0; k < j; ){
		while(i < 26 && num[i] < 2) i++;
		s[k++] = s[j--] = 'a' + i;
		num[i] -= 2;
	}
	s[len] = 0;
	printf("%s\n", s);

	return 0;
}
