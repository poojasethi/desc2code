#include<bits/stdc++.h>
using namespace std;

int n,num[26],q[26],st,e;
string s,re;
int main(){
cin>>s;n=s.length();memset(num,0,sizeof(num));
for(int i=0;i<n;i++) num[s[i]-'a']++;
st=e=0;
for(int i=0;i<26;i++) if(num[i]%2) q[e++]=i;
e--;re.assign(n,'a');
while(st<e) num[q[st++]]++,num[q[e--]]--;
if(st==e) re[(n-1)/2]=q[e]+'a',num[q[e]]--;
st=0;
for(int i=0;i<26;i++){
    for(int j=0;j<num[i];j+=2){
        re[n-st-1]=i+'a';
        re[st++]=i+'a';
    }
}
cout<<re<<endl;
return 0;
}
