#include <bits/stdc++.h>
using namespace std;
int main() {
	int cnt=0,i,j,k,l,n,m,a[26]={0};
	char c[200005],v,ans[200005];
	cin>>c;
	n=strlen(c);
	for(i=0;i<n;i++)
		a[c[i]-'a']++;
	int upto=0;
	for(i=0;i<26;i++)
		if(a[i]%2)
			cnt++;
	int cnt1=0;
	for(i=0;i<26;i++) {
		for(j=0;j<a[i]/2;j++) {
			ans[upto] = i+'a';
			ans[n-upto-1] = i+'a';
			upto++;
		}
		a[i] %= 2;
		if(a[i] && cnt1<cnt/2) {
			ans[upto] = i+'a';
			ans[n-upto-1] = i+'a';
			upto++;
			cnt1++;
			a[i] = 0;
		}
	}
	i=0;
	while(cnt1<(cnt+1)/2) {
		if(a[i]) {
			ans[upto] = i+'a';
			ans[n-upto-1] = i+'a';
			upto++;
			cnt1++;
		}
		i++;
	}
	for(i=0;i<n;i++)
		cout<<ans[i];
	cout<<endl;
	return 0;
}
