#include <stdio.h>
#include <string.h>

int ct[30], n;
char lin[200010];

int main()
{
	int i;
	memset(ct, 0, sizeof ct);
	scanf(" %s", lin);
	n = strlen(lin);
	for(i = 0; i < n; i++)
		ct[lin[i] - 'a']++;
	int in = 0, fin = 'z' - 'a';
	while(in < fin)
	{
		while(in < fin && (ct[in] % 2 == 0))
			in++;
		while(in < fin && (ct[fin] % 2 == 0))
			fin--;
		ct[in]++, ct[fin]--;
	}
	int c = 0;
	for(i = 0; i <= 'z' - 'a'; i++)
	{
		int j;
		if(ct[i] % 2)
			lin[n/2] = i + 'a';
		for(j = 0; j < ct[i]/2; j++)
		{
			lin[c] = lin[n - 1 - c] = 'a' + i;
			c++;
		}
	}
	printf("%s\n", lin);
	return 0;
}
