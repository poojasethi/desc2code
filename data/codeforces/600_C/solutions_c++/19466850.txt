#include<bits/stdc++.h>
using namespace std;
int dp[50];

int main()
{
	string str;
	cin>>str;
	
	
	for(int i=0;i<str.length();i++){
		dp[str[i]-'a']++;
	}
	
	int l,r;
	l=0;
	r=25;
	while(l<r)
	{
		while(l<26&&dp[l]%2==0)
		l++;
		
		while(r>=0&&dp[r]%2==0)
		r--;
		
		if(l<r)
		{
			dp[l]++;
			dp[r]--;
		}
	}
	
	for(int i=0;i<26;i++){
		for(int j=1;j<=(dp[i]/2);j++)
		cout<<(char)(i+'a');
	}
	
	for(int i=0;i<26;i++)
	{
		if(dp[i]&1)
		cout<<(char)(i+'a');
	}
	
	for(int i=25;i>=0;i--){
		for(int j=1;j<=(dp[i]/2);j++)
		cout<<(char)(i+'a');
	}
	return 0;
}
