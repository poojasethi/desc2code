#include <stdio.h>
#include <string.h>
int main()
{
    char s;
    int d[27]={0},i,sl=0,j;
    while (s!='\n')
    {
        s='\n';
        scanf("%c",&s);
        d[s-'a']++;
    }
    i=0;
    j=25;
    while (i<j)
    {
        while (i<j && d[i]%2==0) i++;
        while (i<j && d[j]%2==0) j--;
        if (i<j)
        {
            d[i++]++;
            d[j--]--;
        }
    }
    sl=-1;
    for (i=0;i<26;i++)
    {
        if (d[i]%2==1) sl=i;
        for (j=0;j<d[i]/2;j++) printf("%c",97+i);
    }
    if (sl!=-1) printf("%c",97+sl);
    for (i=25;i>=0;i--)
        for (j=0;j<d[i]/2;j++) printf("%c",97+i);
    return 0;
}
