#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
char str[300000],s[300000];
int num[300];
int main()
{
	scanf("%s",str);
	int len=strlen(str);
	for(int i=0;i<strlen(str);i++)
		num[str[i]]++;
	int h='a',t='z';
	while(h<t)
	{
		while(h<t&&num[h]%2==0) h++;
		while(h<t&&num[t]%2==0) t--;
		num[h]++;
		num[t]--;
	}
	int p=0;
	for(int i='a';i<='z';i++)
	{
		while(num[i]>=2)
		{
			s[p]=s[len-1-p]=i;
			num[i]-=2;
			p++;
		}
		if(num[i]%2)
		{
			s[len/2]=i;
			num[i]--;
		}
	}
	printf("%s",s);
	return 0;
}
