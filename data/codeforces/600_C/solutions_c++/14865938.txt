#include <stdio.h>
#include <iostream>
using namespace std;

int main() {
    char s[200001];
    int c[26];
    int n;
    scanf("%s", s);
    for (n = 0; s[n]; n++)
        c[s[n] - 'a']++;
    for (int i = 0; i < 26; i++)
        if (c[i] % 2)
            for (int j = 25; j > i; j--)
                if (c[j] % 2) {
                    c[i]++, c[j]--;
                    break;
                }
    int si = 0;
    for (int i = 0; i < 26; i++) {
        if (c[i] % 2)
            s[n / 2] = 'a' + i;
        c[i] /= 2;
        for (int j = 0; j < c[i]; j++)
            s[si] = s[n - si - 1] = 'a' + i, si++;
    }
    cout << s << endl;
}
