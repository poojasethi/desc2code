#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>

using namespace std;
const int N = 200005;

int a[26], n;
char s[N];

int main() {
	scanf("%s", s);
	n = strlen(s);
	for (int i = 0; i < n; ++i) {
		a[s[i] - 'a']++;
	}
	vector<int> odd;
	for (int i = 0; i < 26; ++i) {
		if (a[i] & 1) {
			odd.push_back(i);
		}
	}
	int k = 0;
	for (int j = odd.size() / 2 + odd.size() % 2; j < odd.size(); ++j) {
		a[odd[k++]]++;
		a[odd[j]]--;
	}
	string s;
	for (int i = 0; i < 26; ++i) {
		s.append(string(a[i] / 2, i + 'a'));
	}
	cout << s;
	if (n & 1)
		cout << (char) (odd[odd.size() / 2] + 'a');
	cout << string(s.rbegin(), s.rend()) << endl;
}
