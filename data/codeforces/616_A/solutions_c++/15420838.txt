#include <bits/stdc++.h>
using namespace std;

int main()
{
string a,b,c;
cin>>a>>b;

c=string(abs((int)a.length()-(int)b.length()),'0');
if(a.length() < b.length())
{
   a=c+a;
}
if(a.length() > b.length())
{
   b=c+b;
}
if(a>b)
{
 cout<<endl<<">";
}
else if(a<b)
     {
     cout<<endl<<"<";
     }
     else
     {
     cout<<endl<<"=";
     }

return 0;
}
