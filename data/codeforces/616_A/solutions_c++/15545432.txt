#include<stdio.h>
#include<string.h>
char a[2000005], b[2000005];
int main()
{
	memset(a, '0', sizeof(a));
	memset(b, '0', sizeof(b));
	int i, j, la, lb, sgn;
	char *p=a+1000000, *q=b+1000000;
	scanf("%s%s", p, q);
	la = strlen(p);
	lb = strlen(q);
	if (la > lb)
		q -= (la - lb);
	else if (la < lb)
		p -= (lb - la);
	 sgn = strcmp(p, q);
	if (sgn > 0)printf(">\n");
	else if (sgn == 0)printf("=\n");
	else printf("<\n");
}
