#include<bits/stdc++.h>
using namespace std;
int main()
{
    string a,b;
    cin>>a>>b;
    int n=abs((int)a.size()-(int)b.size());
    string zero(n,'0');
    if(a.size()<b.size())
        a=zero+a;
    else
        b=zero+b;
    if(a<b)
        cout<<'<';
    else if(a>b)
        cout<<'>';
    else
        cout<<'=';
    return 0;
}
