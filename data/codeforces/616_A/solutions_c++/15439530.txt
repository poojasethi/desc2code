#include <bits/stdc++.h>

using namespace std;

int main()
{
    string a, b;
    cin >> a >> b;
    string aux(abs((int)a.size() - (int)b.size()), '0');
    if(a.size() < b.size())  a = aux + a;
    else b = aux + b;
    if(a < b) cout << "<";
    else if(a > b) cout << ">";
    else cout << "=";
    return 0;
}
