#include <bits/stdc++.h>
using namespace std;
const int maxn = 1000010;

char a[maxn], b[maxn];

int main()
{
	while(~scanf("%s%s", a, b))
	{
		int i = 0, j = 0;
		while(a[i] == '0' && a[i+1]) i++;
		while(b[j] == '0' && b[j+1]) j++;
		int la = strlen(a+i);
		int lb = strlen(b+j);
		if(la < lb) puts("<");
		else if(la > lb) puts(">");
		else {
			int t = strcmp(a+i, b+j);
			if(t > 0) puts(">");
			else if(t < 0) puts("<");
			else puts("=");
		}
	}
}
