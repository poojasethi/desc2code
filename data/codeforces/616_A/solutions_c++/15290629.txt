#include<bits/stdc++.h>
using namespace std;

string remove_zero(string a)
{
    int zero=0;
    for(zero=0;a[zero]=='0';zero++);
    if(zero==a.length()) zero--;
    return a.substr(zero);
}
int main()
{
   string a, b;
   cin>>a>>b;
   a=remove_zero(a); b=remove_zero(b);
   if(a.size()<b.size()) cout<<"<\n";
   else if(a.size()>b.size()) cout<<">\n";
   else if(a>b) cout<<">\n";
   else if(a<b) cout<<"<\n";
   else cout<<"=\n";
   return 0;
}
