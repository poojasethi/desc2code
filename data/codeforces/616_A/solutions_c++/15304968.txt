#include <iostream>
using namespace std;

int i, j, al, bl;

int main() {
	string a, b;
	cin >> a >> b;
	al = a.length();
	bl = b.length();
	while(a[i] == '0') i++;
	while(b[j] == '0') j++;
	if(al - i > bl - j) {
		cout << ">";
		return 0;
	}
	if(al - i < bl - j) {
		cout << "<";
		return 0;
	}
	while(i < al) {
		if(a[i] < b[j]) {
			cout << "<";
			return 0;
		}
		if(a[i] > b[j]) {
			cout << ">";
			return 0;
		}
		i++; j++;
	}
	cout << "=";
}
