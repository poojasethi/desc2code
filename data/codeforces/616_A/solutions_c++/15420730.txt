#include<bits/stdc++.h>
using namespace std;
string a, b;
int main()
	{
	cin >> a >> b;
	int x = 0, y = 0;
	while(a[x] == '0')
		x++;
	while(b[y] == '0')
		y++;
	if(a.size() - x > b.size() - y)
		cout << '>';
	else if(a.size() - x < b.size() - y)
		cout << '<';
	else
		{
		for(; x < a.size(); x++, y++)
			{
			if(a[x] > b[y])
				cout << '>', exit(0);
			if(a[x] < b[y])
				cout << '<', exit(0);
			}
		cout << '=';
		}
		
	}
