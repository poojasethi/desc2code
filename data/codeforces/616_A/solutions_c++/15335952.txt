#include <iostream>
#include <string>
#include <cmath>


using namespace std;


int main(){
	string s,t;
	cin>>s>>t;
	int n = s.length(),m = t.length();
	int l = abs(n-m);
	string z(l,'0');
	if(n>m) t = z+t;
	if(m>n) s = z+s;
	if(s>t){
	cout<<">";
	return 0;
	}
	if(s<t){
	cout<<"<";
	return 0;
	}
	cout<<"=";
	return 0;
}
