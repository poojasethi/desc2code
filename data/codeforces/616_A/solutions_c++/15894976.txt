#include <cstdio>
#include <cstring>

using namespace std;

char a[2][1000001],b[2][1000001],*p1,*p2,r;
int l;
int main()
{
	memset(a,'0',sizeof(a));
	memset(b,'0',sizeof(b));
	scanf(" %s %s",a[1],b[1]);
	p1=a[1],p2=b[1];
	l=strlen(a[1])-strlen(b[1]);
	if(l<0)
	{
		p1+=l;
	}
	if(l>0)
	{
		p2-=l;
	}
	r=strcmp(p1,p2);
	if(r<0) puts("<");
	if(r>0) puts(">");
	if(r==0) puts("=");
	return 0;
}
