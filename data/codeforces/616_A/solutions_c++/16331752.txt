#include<cstdio>
#include<cstring>
const int maxn = 1000005;
char a[maxn];
char b[maxn];
int main()
{		
	scanf("%s%s",a,b);
	char *aa = a;
	char *bb = b;
	while (*aa == '0')
		aa++;
	while (*bb == '0')
		bb++;
	int lena = strlen(aa);
	int lenb = strlen(bb);
	if (lena != lenb)
	{
		puts(lena < lenb ? "<" : ">");
		return 0;
	}
	int result = strcmp(aa,bb);
	if (result == 0)
		puts("=");
	else if (result < 0)
		puts("<");
	else
		puts(">");
	return 0;
}
