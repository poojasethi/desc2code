#include<iostream>
#include<algorithm>
using namespace std;
string s,t;
int main()
{
	ios_base::sync_with_stdio(false),cin.tie(0),cout.tie(0);
	cin>>s>>t;
	reverse(s.begin(),s.end());
	reverse(t.begin(),t.end());
	while(t.size()<s.size()) t+='0';
	while(t.size()>s.size()) s+='0';
	reverse(s.begin(),s.end());
	reverse(t.begin(),t.end());
	
		if (s<t) 
			return cout<<'<',0;
		else
		if (s>t)
			return cout<<'>',0;
	cout<<'=';
}
