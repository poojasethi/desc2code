#include <bits/stdc++.h>
using namespace std;
char a[1000005],b[1000005];
int main() {
    scanf("%s %s",a,b);
    int n,m;
    n=strlen(a),m=strlen(b);
    reverse(a,a+n);reverse(b,b+m);
    while(n<m) a[n++]='0';
    while(m<n) b[m++]='0';
    bool f=0;
    for(int i=n-1;i>=0;i--) {
        if(a[i]>b[i]) {f=1,puts(">");break;}
        else if(a[i]<b[i]) {f=1,puts("<");break;}
    }
    if(!f) puts("=");
    return 0;
}
