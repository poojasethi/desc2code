#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
	string a, b;
	cin >> a >> b;
	int z = abs(((int)a.size() - (int)b.size()));
	string k(z, '0');
	if (b.size() > a.size())
		a = k + a;
	else
		b = k + b;
	cout << ((a > b) ? ">" : (a == b) ? "=" : "<");
	return 0;
}
