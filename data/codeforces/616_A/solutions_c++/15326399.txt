#include<iostream>
#include<cstdio>
#include<cstring>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    string a,b,c;
    cin>>a>>b;
    c=string(fabs((int)b.length()-(int)a.length()),'0');
    //cout<<c<<endl;
    if(a.length()<b.length())
    a=c+a;
    if(a.length()>b.length())
    b=c+b;
    //cout<<a<<endl<<b<<endl;
    if(a<b)
    printf("<\n");
    else if(a>b)
    printf(">\n");
    else if(a==b)
    printf("=\n");
    return 0;
}
