#include <stdio.h>
#include <string.h>
char a[1000001], b[1000001];
int main() {
	int ia, ib, la, lb;
	scanf("%s%s", a, b);
	la = strlen(a);
	lb = strlen(b);
	for (ia = 0; '0' == a[ia]; ia++);
	for (ib = 0; '0' == b[ib]; ib++);
	if (la - ia < lb - ib)
		puts("<");
	else if (la - ia > lb - ib)
		puts(">");
	else {
		while (ia < la && a[ia] == b[ib])
			ia++, ib++;
		if (a[ia] < b[ib]) puts("<");
		else if (a[ia] > b[ib]) puts(">");
		else puts("=");
	}
	return 0;
}
