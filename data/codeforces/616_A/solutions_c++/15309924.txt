#include<bits/stdc++.h>
using namespace std;
int main()
{
   ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    string a,b,b1;
    cin>>a>>b;
    b1=string(abs((int)a.length()-(int)b.length()),'0');
    if(a.length()<b.length())
        a=b1+a;
    else if(a.length()>b.length())
        b=b1+b;
    if(a<b)
        cout<<"<";
    else if(a>b)
        cout<<">";
    else
        cout<<"=";
    return 0;
}
