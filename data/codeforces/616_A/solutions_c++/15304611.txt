#include <bits/stdc++.h>
using namespace std;
int main(){
	string a,b,c,d;
	cin>>a>>b;
	int s,t;
	for(s=0;s<a.size();++s)
		if(a[s]!='0')break;
	if(s==a.size())c="0";
	else c= a.substr(s);
	for(t=0;t<b.size();++t)
		if(b[t]!='0')break;
	if(t==b.size())d="0";
	else d= b.substr(t);
	if(c.size()==d.size()){
	if(c<=d){
		if(c==d)cout<<"=\n";
		else cout<<"<\n";
	}
	else
		cout<<">\n";
	}
	else
		if(c.size()<d.size())cout<<"<\n";
		else cout<<">\n";
}
