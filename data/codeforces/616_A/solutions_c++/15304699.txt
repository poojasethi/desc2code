#include<bits/stdc++.h>
using namespace std;

int main(){
    //freopen ( "in.txt", "r", stdin );
    string s1, s2, s="";
    cin>>s1>>s2;
    int n=s1.size(), m=s2.size(), d;
    d=abs(n-m);
    while(d--)
        s+="0";
    n>=m ? s2=s+s2 : s1=s+s1;
    //cout<<s1<<endl<<s2<<endl;
    for(int i=0;i<max(n,m);i++){
        if(s1[i]>s2[i])
          return cout<<">",0;
        if(s1[i]<s2[i])
          return cout<<"<",0;
    }
    cout<<"=";
return 0;
}
