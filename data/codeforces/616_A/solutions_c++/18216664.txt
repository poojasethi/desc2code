#include<bits/stdc++.h>
using namespace std;
string a,b;
void Del_Zeros( int cnt , string &s)
{
    for( ;s[cnt]=='0'&&cnt<s.size();cnt++);
    
    
    s.erase(s.begin(),s.begin()+cnt);   
}
int main()
{
    cin>>a; 
    cin>>b;
    Del_Zeros(0 , a);
    Del_Zeros(0 , b);
    
    cout <<  ( (a.size()<b.size() || a.size()==b.size() && a<b ) ? "<"  :  ( (a==b) ? "=" : ">" ) );
    
return 0;    
}
