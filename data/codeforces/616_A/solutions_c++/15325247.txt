#include<bits/stdc++.h>
using namespace std;
int main ()
{
    ios_base::sync_with_stdio(0);
    string a,b,c;
    cin >> a >> b;
    c = "";
    for(int i = 0; i < max(a.length(),b.length()) - a.length(); i++)c += '0';
    c += a;
    a = c;
    c = "";
    for(int i = 0; i < max(a.length(),b.length()) - b.length(); i++)c += '0';
    c += b;
    b = c;
    cout << (a > b ? ">" : (a < b) ? "<" : "=");
}
