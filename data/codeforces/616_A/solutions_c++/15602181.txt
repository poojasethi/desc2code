#include <stdio.h>
#include <string.h>
char x[2333333],y[2333333];
int main() {
    scanf("%s%s",x,y);
    char *px=x,*py=y;
    while(*px=='0') ++px;
    while(*py=='0') ++py;
    int length1=strlen(px),length2=strlen(py);
    if(length1<length2) putchar('<');
    else if(length1>length2) putchar('>');
    else putchar((length1=strcmp(px,py))?length1>0?'>':'<':'=');
    return 0;
}
