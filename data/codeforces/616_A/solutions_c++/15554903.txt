#include<bits/stdc++.h>
using namespace std;

string a,b;
int main()
{
    cin>>a>>b;
    int len1 = a.size(),len2 = b.size();
    int la = 0,lb = 0;
    while(a[la]=='0'&&la<len1)la++;
    while(b[lb]=='0'&&lb<len2)lb++;
    if(len1-la>len2-lb)return puts(">");
    if(len1-la<len2-lb)return puts("<");
    for(int i=la;i<len1;i++)
    {
        if(a[i]>b[i-la+lb])return puts(">");
        if(a[i]<b[i-la+lb])return puts("<");
    }
    return puts("=");
}
