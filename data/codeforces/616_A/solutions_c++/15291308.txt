#include <iostream>
using namespace std;

int main()
{
	string s, t;
	cin>> s >> t;

	s.erase(0, s.find_first_not_of('0'));
	t.erase(0, t.find_first_not_of('0'));

	if(s.size() > t.size())
	{
		cout<< ">" << endl;
		return 0;
	}
	else if(s.size() < t.size())	
	{
		cout<< "<" << endl;
		return 0;
	}

	int i = 0;
	while(i < s.size() && s[i] == t[i])
		i ++;

	if(i == s.size())
	{
		cout<< "=\n";
		return 0;
	}

	if(s[i] > t[i])
	{
		cout<< ">\n";
	}
	else
		cout<< "<\n";


}
