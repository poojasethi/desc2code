#include<bits/stdc++.h>
using namespace std;
char a[1000001],b[1000001];
string x,y;
void DLZ(string &z,int i)
{
    for(;i<z.size()&&z[i]=='0';i++) ;
    z.erase(z.begin(),z.begin()+i);   
}
int main()
{
    scanf("%s%s",a,b); x=string(a); y=string(b);
    DLZ(x,0); DLZ(y,0);
    if(x.size()<y.size() || x.size()==y.size() && x<y )
        cout << "<" << endl;
    else if(x==y)
        cout << "=" << endl;
    else
        cout << ">" << endl;
}
