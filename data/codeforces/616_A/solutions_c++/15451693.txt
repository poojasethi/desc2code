#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

char a[1000005], b[1000005];

int main() {
	scanf("%s%s", a, b);
	char *s = a;
	char *t = b;
	while (*s == '0') ++s;
	while (*t == '0') ++t;
	int slen = strlen(s);
	int tlen = strlen(t);
	if (slen < tlen) printf("<\n");
	else if (slen > tlen) printf(">\n");
	else {
		int cmp = strcmp(s, t);
		printf(cmp < 0 ? "<\n" : (cmp == 0 ? "=\n" : ">\n"));
	}
}
