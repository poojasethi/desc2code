#include <iostream>
#include <string>
#include <bits/stdc++.h>
using namespace std;
string a,b,c="";
unsigned long int la,lb,k;
int main()
{
    cin>>a>>b;
la=a.size();
lb=b.size();
if (la>lb) k=la-lb;
else k=lb-la;
for (unsigned long int i=0;i<k;i++)
{
    c+="0";
}
if (la>lb) b=c+b;
else a=c+a;
if (a>b) cout<<">";
else if(a<b) cout<<"<";
else cout<<"=";
    return 0;
}
