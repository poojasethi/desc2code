x = raw_input().lstrip('0')
y = raw_input().lstrip('0')
if x == '':
	x = '0'
if y == '':
	y = '0'
if len(x) > len(y):
	print('>')
elif len(x) < len(y):
	print('<')
elif x == y:
	print('=')
else:
	flag = 0
	for i,j in zip(x,y):
		if(i>j):
			print('>')
			flag = 1
			break
		elif(i<j):
			print('<')
			flag = 1
			break
	if flag == 0:
		print('=')
