A = raw_input()
B = raw_input()

i = 0
j = 0

n = len(A)
m = len(B)

while i < n and A[i] == '0':
	i += 1
	
while j < m and B[j] == '0':
	j += 1

A = A[i:]
B = B[j:]

n -= i
m -= j

if n == m:

	ret = '='
	
	i = 0

	while i < n:
		if A[i] < B[i]:
			ret = '<'
			break
		elif A[i] > B[i]:
			ret = '>'
			break

		i += 1
			
	print ret
	
elif n < m:
	print '<'
else:
	print '>'
