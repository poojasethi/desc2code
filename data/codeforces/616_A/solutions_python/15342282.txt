x = raw_input()
y = raw_input()
xx = len(x)
yy = len(y)

if xx < yy:
    x = '0' *( yy-xx ) + x
    xx = len(x)
else :
    y = '0' * (xx-yy)  + y
    yy = len(y)

for i in range(0,xx):
    if x[i] < y[i]:
        print '<'
        exit()
    elif y[i] < x[i]:
        print '>'
        exit()
print '='
