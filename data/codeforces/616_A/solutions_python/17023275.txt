def compare(a, b):
    if len(a) > len(b):
        return ">"
    elif len(b) > len(a):
        return "<"
    else:
        for i in range(0, len(a)):
            if a[i] > b[i]:
                return ">"
            elif a[i] < b[i]:
                return "<"
    return "="
