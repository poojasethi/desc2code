first = raw_input()
second = raw_input()

first = first.lstrip("0")
second = second.lstrip("0")

if len(first) > len(second):
    print ">"

elif len(first) < len(second):
    print "<"

elif len(first) == len(second):

    if first == second:
        print "="

    elif first > second:
        print ">"

    elif first < second:
        print "<"
