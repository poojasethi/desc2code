import sys

a = sys.stdin.readline().strip()
b = sys.stdin.readline().strip()

try:
    a = a.lstrip("0")
    b = b.lstrip("0")
    if len(a) == len(b):
        _max = max(a,b)
        if _max == a == b:
            print "="
        elif _max == a:
            print ">"
        else:
            print "<"
    elif len(a) > len(b):
        print ">"
    else:
        print "<"
except ValueError:
    print "="
