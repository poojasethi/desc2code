def compare(a, b):
    lena = len(a)
    lenb = len(b)
    if lena < lenb:
        return "<"
    elif lena > lenb:
        return ">"
    elif a < b:
        return "<"
    elif a > b:
        return ">"
    else:
        return "="

a = raw_input().lstrip("0")
b = raw_input().lstrip("0")
print compare(a, b)
