def f(a,b):
	x = len(a)
	y = len(b)
	if x > y:
		return '>'
	elif x < y :
		return '<'
	else:
		for i in range(x):
			if a[i] > b[i]:
				return '>'
			elif a[i] < b[i]:
				return '<'
	return '='

a = raw_input().lstrip('0')
b = raw_input().lstrip('0')
if len(a) == 0:
	a = '0';
if len(b) == 0:
	b = '0';

print f(a,b)
