from sys import stdin
aa = stdin.readline().strip()
b = stdin.readline().strip()
def par(a):
 l = len(a)
 for i in xrange(l):
  if a[i]!='0':
   return a[i:]
 return '0'

aa = par(aa)
b = par(b)

def ch(a,b):
 if len(a) > len(b):
  return '>'
 if len(a) < len(b):
  return '<'
 if a==b:
  return '='
 if a>b:
  return '>'
 return '<'

print ch(aa,b)
