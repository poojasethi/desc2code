second = raw_input()
flen = len(first)
slen = len(second)
i = 0
while i<flen and first[i]=='0':i+=1
j = 0
while j<slen and second[j]=='0':j+=1
if flen-i>slen-j:
    print '>'
elif flen-i<slen-j:
    print '<'
else:
    while i<flen and first[i]==second[j]:
        i+=1
        j+=1
    if i==flen: print '='
    elif first[i]>second[j]:print '>'
    else: print '<'
