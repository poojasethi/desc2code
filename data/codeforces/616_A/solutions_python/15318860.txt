def solver():
    a = raw_input()
    b = raw_input()
    if len(a) > len(b):
        b = '0' * (len(a)- len(b)) + b
    else:
        a = '0' * (len(b)- len(a)) + a
    for i in xrange(len(a)):
        if a[i] > b[i]:
            print '>'
            return
        if a[i] < b[i]:
            print '<'
            return
    print '='
    
if __name__ == "__main__":
    solver()
