a = raw_input()
b = raw_input()
a = a.lstrip('0')
b = b.lstrip('0')
if len(a) - len(b) < 0:
    a = '0' *  (len(b)-len(a)) + a
else:
    b = '0' *  (len(a)-len(b)) + b

flag = True
for i, j in zip(a, b):
    if i > j:
        flag = False
        print '>'
        break
    elif i < j:
        flag = False
        print '<'
        break
if flag is True:
    print '='
