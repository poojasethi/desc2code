a = raw_input() # contains no more than 10^6 digits
b = raw_input()

a = a.lstrip('0')
b = b.lstrip('0')

if len(a) > len(b):
    ans = '>'
elif len(a) < len(b):
    ans = '<'
else:
    for i in xrange(len(a)):
        if a[i] > b[i]:
            ans = '>'
            break
        elif a[i] < b[i]:
            ans = '<'
            break
    else:
        ans = '='

print ans
