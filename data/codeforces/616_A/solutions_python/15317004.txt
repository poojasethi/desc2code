import sys
a=b='qar'
a=raw_input()
b=raw_input()
n=len(a)
m=len(b)
i=0
j=0
while i<n:
	if a[i]!='0':
		break
	i+=1
while j<m:
	if b[j]!='0':
		break
	j+=1
if n-i>m-j:
	print(">")
	sys.exit()
elif n-i<m-j:
	print("<")
	sys.exit()
while i<n and j<m :
	if a[i]<b[j]:
		print("<")
		sys.exit()
	if a[i]>b[j]:
		print(">")
		sys.exit()
	i+=1
	j+=1
print("=")
