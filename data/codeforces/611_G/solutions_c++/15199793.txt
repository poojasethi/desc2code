#include <bits/stdc++.h>

#define For(i,l,r) for(int _r=r,i=l;i<=_r;++i)

using namespace std;

typedef long long LL;
typedef LL Arr[1000005];

const LL mo=1e9+7;

int n;LL an,S,GS;

Arr xfix,yfix,bg,mine;

struct ww{
    LL x,y;
    LL operator ^ (const ww&a)  {return x*a.y-y*a.x;}
}p[1000005];

inline void inc(LL &x,LL y) {if(((x+=y)%=mo)<0)x+=mo;}
inline void dec(LL &x,LL y) {if(((x-=y)%=mo)<0)x+=mo;}

inline void calc(int i,int j){
    inc(bg[i],j-i-1);
    inc(mine[i+1],1);
    inc(mine[j],mo-1);
    inc(an,p[i].y%mo*((xfix[j]-xfix[i+1])%mo)%mo-p[i].x%mo*((yfix[j]-yfix[i+1])%mo)%mo);
}

inline void recalc(){
    For(i,1,n*2){
        inc(bg[i],bg[i-1]-mine[i-1]);
        inc(an,(p[i]^p[i+1])%mo*bg[i]%mo);
        inc(mine[i],mine[i-1]);
    }
}

inline void gao(){
    For(i,1,n)p[i+n]=p[i];
    For(i,1,n*2){
        inc(xfix[i],xfix[i-1]+p[i].x%mo);
        inc(yfix[i],yfix[i-1]+p[i].y%mo);
    }
    For(i,1,n)S+=p[i]^p[i+1];
    int j=1;
    unsigned long long mj=0;
    For(i,1,n){
        for(;j<=2*n && (mj+(p[j]^p[i]))*2<S;++j)mj+=p[j]^p[j+1];
        --j;mj-=p[j]^p[j+1];
        if(i+1<j){
            inc(GS,j-i-1);
            calc(i,j);
        }
        mj-=p[i]^p[i+1];
    }
    recalc();
}

int main(){
    scanf("%d",&n);
    For(i,1,n)scanf("%I64d%I64d",&p[i].x,&p[i].y);
    reverse(p+1,p+n+1);
    gao();
    printf("%d\n",((S%mo*(GS%mo)%mo-2*an)%mo+mo)%mo);
}
