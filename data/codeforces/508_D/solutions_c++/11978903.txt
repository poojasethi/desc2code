#include <bits/stdc++.h>
using namespace std;

typedef int ll;
typedef long long ld;

const ll Maxn=1e6+7;
const ll Max=2e4+7;
const ll Mod=1e9+7;
const ll mass=1e6;
const ll Const=1000;

#define pb push_back
#define mp make_pair
#define F first
#define S second
#define P pair<ll,ll>

ll n,in[Maxn],out[Maxn],g1,g2,p,ans;
string s[Maxn],str="";
vector<ll>V[Maxn];
map<pair<int,int>,bool>mark;

ll dfs(ll u){
    while(V[u].size()){
        ll v=V[u][V[u].size()-1];
        V[u].pop_back();
        dfs(v);
    }
    str+=char(u%Const);
}

int main(){
    cin>>n;
    for(ll i=1;i<=n;i++){
        cin>>s[i];
        ll a=s[i][0]*Const+s[i][1];
        ll b=s[i][1]*Const+s[i][2];
        V[a].push_back(b);
        p=a;
        out[a]++;
        in[b]++;
    }
    for(ll i=1;i<=mass;i++){
        ll d=out[i]-in[i];
        if(d>1||d<-1){
            cout<<"NO"<<endl;
            return 0;
        }
        if(d==-1)
            g1++;
        if(d==1)
            g2++,p=i;
    }
    if(g1>1||g2>1){
        cout<<"NO"<<endl;
        return 0;
    }
    dfs(p);
    str+=char(p/Const);
    reverse(str.begin(),str.end());
    if(str.size()==n+2){
        cout<<"YES"<<endl<<str<<endl;
        return 0;
    }
    cout<<"NO"<<endl;
    return 0;
}
