#include <bits/stdc++.h>
using namespace std;
vector <int> v[200006];
vector <int> ans;
int deg[200006];
void dfs(int p)
{
	while(v[p].size()>0)
	{
		int pp=v[p][v[p].size()-1];
		v[p].pop_back();
		dfs(pp);
	}
	ans.push_back(p);
}
int main()
{
//	freopen("input.txt","r", stdin);
//freopen("out.txt", "w", stdout);
int n,i,c=0,r,p,q;
string s;

scanf("%d",&n);
for(i=0;i<n;i++)
{
	cin>>s;
	p=s[0]*1000+s[1];
	q=s[1]*1000+s[2];
	v[p].push_back(q);
	deg[p]++;
	deg[q]--;
}
for(i=0;i<=200000;i++)
{
	if(deg[i]>1)
	{
		printf("NO");
		return 0;
	}
	if(deg[i]==1)
	{
		c++;
		p=i;
	}
}
if(c>1)
{
		printf("NO");
		return 0;
}
dfs(p);
if(ans.size()!=n+1)
{
	printf("NO");
		return 0;
}
string pass;
pass+=ans[n]/1000;
for(i=n;i>=0;i--)
pass+=ans[i]%1000;
printf("YES\n");
cout<<pass;
return 0;
}
