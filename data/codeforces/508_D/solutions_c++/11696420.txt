#include <iostream>
#include <cstdio>
#include <string>
#include <vector>
#include <map>
using namespace std;
int n,m,cnt,start,pos,ans[200005],o[400005];
string data[400005];
map<string,int> mp;
vector<int> adj[400005];
int dfs(int node){
    while(adj[node].size()){
        int temp=adj[node].back();
        adj[node].pop_back();
        dfs(temp);
    }
    ans[pos++]=node;
}
int main(){
    cin>>n;
    for(int i=0;i<n;i++){
        string s;
        cin>>s;
        data[m]=s.substr(0,2);
        if(mp.count(data[m])==0){
            mp[data[m]]=m++;
        }
        data[m]=s.substr(1,2);
        if(mp.count(data[m])==0){
            mp[data[m]]=m++;
        }
        adj[mp[s.substr(0,2)]].push_back(mp[s.substr(1,2)]);
        o[mp[s.substr(1,2)]]++;
    }
    for(int i=0;i<m;i++){
        if(adj[i].size()==o[i])cnt++;
        if(adj[i].size()>o[i]+1){
            cnt=-1000000000;
        }
        if((adj[i].size()-o[i])==1)start=i;
    }
    dfs(start);
    if((cnt!=m-2&&cnt!=m)||pos!=n+1){
        cout<<"NO";return 0;
    }
    cout<<"YES\n";
    cout<<data[ans[n]][0];
    for(int i=n;i>=0;i--)cout<<data[ans[i]][1];
}
