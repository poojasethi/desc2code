#include<stdio.h>
#include<string.h>
#include<map>
#include<algorithm>
using namespace std;
    
const int N=100005,M=200055;
int a[M],v[N],deg[N];
int next[M],to[M],cnt,ans[M];
bool ff[M],mem[M];
char str[5];
void add(int x,int y){
    to[cnt]=y;
    next[cnt]=v[x];
    v[x]=cnt++;
}   
void dfs(int x){
    for(int i=v[x];i!=-1;i=v[x]){
        if(!mem[i]) mem[i]=1,v[x]=next[i],dfs(to[i]);
    }
    ans[cnt--]=x;
}   
int main(){
    int i,j,x,y,k,t,n,m,r;
    scanf("%d",&n);
    memset(v,-1,sizeof v);
    memset(mem,0,sizeof mem);
    memset(ff,0,sizeof ff);
    for(i=cnt=m=0;i<n;i++){
        scanf("%s",str);
        x=str[0]*300+str[1];
        y=str[1]*300+str[2];
        if(!ff[x]) ff[x]=1,a[m++]=x;
        if(!ff[y]) ff[y]=1,a[m++]=y;
        deg[x]++,deg[y]--;
        add(x,y);
    }
    for(t=a[0],i=x=y=0;i<m;i++){
        if(deg[a[i]]<-1||deg[a[i]]>1) break;
        if(deg[a[i]]==1) x++,t=a[i];
        else if(deg[a[i]]==-1) y++;
    }
    if(i<m||x!=y||x>1){
        printf("NO\n");
        return 0;
    }
    cnt=n+1,dfs(t);
    if(cnt) printf("NO\n");
    else{
        printf("YES\n%c%c",t/300,t%300);
        for(i=2;i<=n+1;i++) printf("%c",ans[i]%300);
        printf("\n");
    }
    return 0;
}
