#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<list>
#include<vector>
using namespace std;
list<int> L[50005];
int n,st,ed;
int in[50005],out[50005];
char a,b,c,d;
bool vis[50005];
vector<char> V;
void dfs(int id){
  while(!L[id].empty()){
    int to=L[id].front();
    L[id].pop_front();
    in[to]--;
    out[id]--;
    dfs(to);
  }
  V.push_back(id%128);
  return;
}
int main(){
  memset(in,0,sizeof(in));
  memset(out,0,sizeof(out));
  scanf("%d",&n);
  for(int i=0;i<n;i++){
    scanf("%c%c%c%c",&a,&b,&c,&d);
    L[b*128+c].push_back(c*128+d);
    out[b*128+c]++;
    in[c*128+d]++;
  }
  st=ed=-1;
  for(int i=1;i<=50000;i++){
    if(in[i]==out[i]+1){
      if(ed==-1) ed=i;
      else{printf("NO\n");return 0;}
    }
    if(in[i]+1==out[i]){
      if(st==-1) st=i;
      else{printf("NO\n");return 0;}
    }
    if(abs(in[i]-out[i])>=2){printf("NO\n");return 0;}
  }
  if(st==-1 && ed==-1){
    for(int i=1;i<=50000;i++){
      if(in[i]>0){
        st=ed=i;
        break;
      }
    }
  }
  memset(vis,0,sizeof(vis));
  dfs(st);
  int k=V.size();
  if(k!=n+1){printf("NO\n");return 0;}
  printf("YES\n");
  printf("%c",st/128);
  for(int i=k-1;i>=0;i--) printf("%c",V[i]);
  printf("\n");
  return 0;
}
