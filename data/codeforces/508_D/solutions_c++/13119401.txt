#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define pb push_back
#define mp make_pair
#define NMAX 1000000+10000
#define INF 1e9+5

int gcd(int a, int b)
{
    return !b?a:gcd(b, a%b);
}

int abs(int a)
{
    if(a<0)
        return -a;
    return a;
}

void swap(char &a, char &b)
{
    char aux=a;
    a=b;
    b=aux;
}

int g[NMAX];
vector <int> v[NMAX];
int pas[200005], size=0;

void sol(int p)
{
    while(v[p].size()!=0)
    {
        int aux=v[p][v[p].size()-1];
        v[p].pop_back();
        sol(aux);
    }
    pas[size++]=p;
}

int main ()
{
    int n, a, b;
    cin>>n;
    char s[5];
    for(int i=0;i<n;i++)
    {
        scanf("%s", s);
        a=s[0]*1000+s[1], b=s[1]*1000+s[2];
        g[a]++;
        g[b]--;
        v[a].pb(b);
    }
    int aux=a, p=0;
    for(int i=0;i<NMAX;i++)
        if(g[i]>0)
        {
            p+=g[i];
            aux=i;
        }
    if(p>1)
    {
        cout<<"NO"<<endl;
        return 0;
    }
    sol(aux);
    if(size<n+1)
    {
        cout<<"NO"<<endl;
        return 0;
    }
    cout<<"YES"<<endl;
    cout<<char(pas[size-1]/1000);
    for(int i=size-1;i>=0;i--)
        printf("%c", pas[i]%1000);
}
