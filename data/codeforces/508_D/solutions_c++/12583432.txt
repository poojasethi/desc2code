#include<cstdio>
#include<cstring>
#include<map>
#include<iostream>
#include<string>
#include<algorithm>
#include<vector>
using namespace std;
const int N = 128;
int cnt,n,in[N*N];
string ans;
vector<int> next[N*N];
void solve(int pre){
    while(next[pre].size()){
        int t=next[pre].back();
        next[pre].pop_back();
        solve(t);
    }
    ans+=pre%128;
}
int main(){
    cnt = 0;
    char str[5];
    scanf("%d",&n);
    memset(in,0,sizeof(in));
    for(int i=0;i<n;i++){
        scanf("%s",str);
        int id1 = N*str[0]+str[1];
        int id2 = N*str[1]+str[2];
        next[id1].push_back(id2);
        in[id2]++;
    }
    int s=-1;
    for(int i=0;i<N*N;i++){
        if(abs((int)next[i].size()-in[i])>1) {
            printf("NO\n");
            return 0;
        }
        else if(next[i].size()-in[i]==1) {
            if(s!=-1) {
                printf("NO\n");
                return 0;
            }
            s = i;
        }
    }
    if(s==-1) {
        for(int i=0;i<N*N;i++){
            if(next[i].size()) {
                s=i;
                break;
            }
        }
    }
    solve(s);
    ans+=s/128;
    if(ans.length()==n+2){
        reverse(ans.begin(),ans.end());
        printf("YES\n");
        puts(ans.c_str());
    }
    else printf("NO\n");
}
