#include <bits/stdc++.h>
#define f(i,a,b) for(int i = (a); i <= (b); i++)

using namespace std;

int I[66000], O[66000], N;
multiset<int> E[66000];
string S;

void p(int n)
{
    while(!E[n].empty())
    {
        int v = *E[n].begin();
        E[n].erase(E[n].begin());
        p(v);
    }

    S += (char) (n%256);
}

main()
{
    cin >> N;
    f(i,1,N)
    {
        string s;
        cin >> s;
        int a = s[0] * 256 + s[1];
        int b = s[1] * 256 + s[2];
        E[a].insert(b);
        O[a]++, I[b]++;
    }

    int r = 0, l = 0, v = -1;

    f(i,0,66000-1)
    {
        if(I[i] > O[i] + 1 || O[i] > I[i] + 1)
        {
            cout << "NO";
            return 0;
        }
        if(I[i] > O[i]) r++;
        if(O[i] > I[i]) l++, v = i;
        if((I[i] || O[i]) && v < 0) v = i;
    }

    if(r>1 || l>1)
    {
        cout << "NO";
        return 0;
    }

    p(v);
    reverse(S.begin(),S.end());
    f(i,0,65999) if(!E[i].empty())
    {
        cout << "NO";
        return 0;
    }
    cout << "YES\n" << (char) (v/256) << S;
}
