#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007
#define N 100005

char ss[5];
vector<int> mp[66000];
int in[66000],out[66000],cc[66000];
string ans;

void dfs(int x)
{
    while(cc[x]<mp[x].size())
        dfs(mp[x][cc[x]++]);

    ans+=(char)x%256;
}

int main()
{
    int n,u,v,i,l,r,st;
    st = 0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf(" %s",ss);
        u = ss[0]*256+ss[1];
        v = ss[1]*256+ss[2];
        st = u;
        mp[u].push_back(v);

        in[v]++;
        out[u]++;
    }
    l = r = 0;
    for(i=0;i<66000;i++)
    {
        int d = out[i]-in[i];
        if(d==1)
        {
            l++;
            st = i;
        }
        else if(d==-1)
            r++;
        else if(d!=0)
        {
            cout<<"NO";
            return 0;
        }
    }
    if(l>1 || r>1)
    {
        cout<<"NO";
        return 0;
    }

    dfs(st);
    ans+=(char)(st/256);
    reverse(ans.begin(),ans.end());
    if(ans.size()!=n+2) cout<<"NO";
    else    cout<<"YES\n"<<ans;
    return 0;
}
