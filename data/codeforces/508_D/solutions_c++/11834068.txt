//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

const int maxn=1001010;
int n,f,l,d[maxn],b;
string s,Ans;
vector<int> g[maxn],ans;

void tour(int v)
{
  while(g[v].size()>0)
    {
      int next=g[v][g[v].size()-1];
      g[v].pop_back();
      tour(next);
    }
  ans.push_back(v);
}

int main()
{
  cin>>n;
  for(int i=0;i<n;i++)
    {
      cin>>s;
      f=s[0]*1000+s[1];
      l=s[1]*1000+s[2];
      g[f].push_back(l);
      d[l]--;
      d[f]++;
    }
  for(int i=0;i<=1001000;i++)
    {
      if(d[i]>1)
	{
	  cout<<"NO"<<endl;
	  return 0;
	}
      if(d[i]==1)
	{
	  b++;
	  f=i;
	}
    }
  if(b>1)
    {
      cout<<"NO"<<endl;
      return 0;
    }
  tour(f);
  if(ans.size()!=n+1)
    {
      cout<<"NO"<<endl;
      return 0;
    }
  Ans+=ans[n]/1000;
  for(int i=n;i>=0;i--)
    Ans+=ans[i]%1000;
  cout<<"YES"<<endl<<Ans<<endl;
  return 0;
}
