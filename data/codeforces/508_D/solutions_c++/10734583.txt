#include<iostream>
#include<vector>
using namespace std;
vector<int> v[70000];
int count[70000];
string ans;
void dfs(int i)
{
	while(!v[i].empty())
	{
		int s=v[i].back();
		v[i].pop_back();
		dfs(s);
	}
	ans+=(i%256);
}
int main()
{
	int n,len;
	cin>>n;
	len=n;
	while(n--)
	{
		string a;
		cin>>a;
		int st=a[0]*256+a[1];
		int en=a[1]*256+a[2];
		v[st].push_back(en);
		count[st]++;
		count[en]--;
	}
	int p=0,s=-1;
	for(int i=0;i<70000;i++)
	{
		if(count[i]>=1)
		{
			p++;
			s=i;
		}
	}
	if(count[s]>1||p>1)
	{
		cout<<"NO";
		return 0;
	}
	if(s==-1) 
		for(int i=0;i<70000;i++)
			if(!v[i].empty())
			{
				s=i;
				break;
			}
	dfs(s);
	ans+=s/256;
	if(ans.size()==len+2)
	{
		cout<<"YES"<<endl;
		for(int i=ans.size()-1;i>=0;i--)
			cout<<ans[i];
	}
	else
		cout<<"NO";
	return 0;
}
