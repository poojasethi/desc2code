#include <iostream>
#include <cmath>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
const int MX = 128*128+128;
int n, in[MX], out[MX];
vector<int> adj[MX], cyc;

void dfs(int u) {
	while (!adj[u].empty()) {
		int v = adj[u].back();
		adj[u].pop_back();
		dfs(v);
		cyc.push_back(v);
	}
}

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n;
	string a;
	int def;
	for (int i = 0; i < n; i++) {
		cin >> a;
		int x = (int)a[0]*128+(int)a[1], y = (int)a[1]*128+(int)a[2];
		adj[x].push_back(y);
		out[x]++; in[y]++;
		def = x;
	}
	int s = -1, e = -1;
	for (int i = 0; i < MX; i++) {
		if (out[i]+1 == in[i]) {
			if (e == -1) e = i;
			else {
				cout << "NO\n"; return 0;
			}
		}
		if (out[i] == in[i]+1) {
			if (s == -1) s = i;
			else {
				cout << "NO\n"; return 0;
			}
		}
		else if (abs(out[i]-in[i]) > 1) {
			cout << "NO\n"; return 0;
		}
	}
	if (s == -1) s = def;
	dfs(s);
	cyc.push_back(s);
	if (cyc.size() != n+1) {
		cout << "NO\n"; return 0;
	}
	reverse(cyc.begin(), cyc.end());
	cout << "YES\n";
	for (int i = 0; i < cyc.size(); i++) {
		if (!i) {
			cout << (char)(cyc[i]/128);
		}
		cout << (char)(cyc[i]%128);
	}
	return 0;
}
