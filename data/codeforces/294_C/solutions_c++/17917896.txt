#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
lld mod = 1000000007LL;
lld power(lld a, lld n)
{
	if(n<0)
		return 1;
	lld ans = 1LL;
	while(n!=0)
	{
		if(n%2)
		ans = (ans*a)%mod;
		a = (a*a)%mod;
		n/=2;
	}
	return ans;
}

int main()
{
	lld n,m,lights[1001]={},i,last,ans,fact[1001]={};
	fact[0] = 1;
	for(i=1;i<=1000;i++)
		fact[i] = (fact[i-1]*i)%mod;
	scanf("%lld%lld",&n,&m);
	for(i=0;i<m;i++)
		scanf("%lld",&lights[i]);
	sort(lights,lights+m);
	ans = fact[n-m];
	last = 1;
	ans = (ans*power(fact[lights[0]-1],mod-2))%mod;
	for(i=1;i<m;i++)
	{
		ans = (ans*power(fact[lights[i]-lights[i-1]-1],mod-2))%mod;
		ans = (ans*power(2,lights[i]-lights[i-1]-2))%mod;
	}
	ans = (ans*power(fact[n - lights[m-1]],mod-2))%mod;
	printf("%lld\n",ans );
	return 0;
}
