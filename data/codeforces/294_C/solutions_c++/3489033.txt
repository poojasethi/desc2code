#include<stdio.h>
#include<algorithm>
using namespace std;
long long a[1005][1005]={0},S;
long long b[1005],xar[1005];
int main()
{
  long long n,m,N,M,i,j,x;
  scanf("%lld%lld",&n,&m);
  for(i=0;i<m;i++)
    scanf("%lld",&b[i]);
  xar[0]=1;
  for(i=1;i<=n;i++)
    xar[i]=(xar[i-1]*2)%1000000007;
  sort(b,b+m);
  a[1][1]=1;
  for(i=2;i<=n;i++)  
    for(j=1;j<=i;j++)
      a[i][j]=(a[i-1][j]+a[i-1][j-1])%1000000007;
  x=b[0];
  S=1;
  for(i=1;i<m;i++)
  if(b[i]-b[i-1]>1)
  {
    M=b[i]-b[i-1]-1;
    N=n-b[i-1]-(m-i);
    S=(S*a[N+1][M+1])%1000000007;
    S=(S*xar[ b[i]-b[i-1]-2 ])%1000000007;
  }
  S=(S*a[n-m+1][b[0]])%1000000007;
  printf("%lld\n",S);
  
}
