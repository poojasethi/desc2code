#include<iostream>
#include<algorithm>
using namespace std;

long long d[1010][1010], p[1010], t[1010];

int main()
{
	long long n, k, w, a=0, b, m, odp=1, s;
	cin>>n>>m;
	p[0]=1;
	for(int i=1; i<=n; i++)
	p[i]=(p[i-1]*2)%1000000007;
	for(int i=0; i<=n; i++)
	{
		d[i][0]=1;
	}
	for(int i=1; i<=n; i++)
	{
		for(int j=1; j<=n; j++)
		{
			d[i][j]=(d[i-1][j-1]+d[i-1][j])%1000000007;
		}
	}
	for(int i=0; i<m; i++)
	cin>>t[i];
	sort(t, t+m);
	w=n-m;
	a=t[0];
	s=a-1;
	if(s)
	odp*=d[w][s];
	w-=s;
	odp%=1000000007;
	for(int i=1; i<m; i++)
	{
		b=t[i];
		s=b-a-1;
		if(s)
		odp*=p[s-1];
		odp%=1000000007;
		if(s)
		odp*=d[w][s];
		w-=s;
		odp%=1000000007;
		a=b;
	}
	s=n-a;
	if(s>0)
	odp*=d[w][s];
	w-=s;
	odp%=1000000007;
	cout<<odp;
}
