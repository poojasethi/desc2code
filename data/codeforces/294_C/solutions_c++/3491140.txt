#include<iostream>
#include<cmath>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
#define MOD (1000000007ll)
int n,m;
long long ans;
int d[1005];
long long f[1005];
long long pow(long long a,long long b)
{
	long long tmp=1ll;
	for (;b;b>>=1,a=a*a%MOD)
		if (b&1) tmp=tmp*a%MOD;
	return tmp;
}
int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;++i) scanf("%d",&d[i]);
	sort(d+1,d+m+1);d[m+1]=n+1;
	f[0]=1ll;
	for (int i=1;i<=n;++i) f[i]=f[i-1]*i%MOD;
	ans=f[n-m];
	for (int i=0;i<=m;++i) 
	{
		if (d[i+1]==d[i]) continue;
		int t=d[i+1]-d[i]-1;
		ans=ans*pow(f[t],MOD-2)%MOD;
		if (i!=0&&i!=m&&t>=1)ans=ans*pow(2ll,(long long)t-1)%MOD;
	}
	cout<<ans;
	return 0;
}
