ll bpow(ll x,ll n){
ll ans=1;
if(n<0) return 1;
while(n>0){
if(n&1) ans*=x%mod;
x=x*x%mod;
n/=2;
ans%=mod;
}
return ans;
}

ll n,m;
cin>>n>>m;
ll x[m+2];
x[0]=0;
for(int i=1;i<=m;i++) cin>>x[i];
ll fact[1001];
fact[0]=1;
for(int i=1;i<=1000;i++){
fact[i]=i*fact[i-1];
fact[i]%=mod;
}
ll ans=1;
x[m+1]=n+1;
ll k=1;
sort(x,x+m+2);
for(int i=1;i<=m+1;i++){
ans=ans*fact[x[i]-x[i-1]-1];
if(i>1&&i<=m){
k=k*bpow(2,x[i]-x[i-1]-2);
}
k%=mod;
ans%=mod;
}
ll ans1=fact[n-m]*k;
ans1%=mod;
ans1*=bpow(ans,mod-2)%mod;
ans1%=mod;
cout<<ans1;
}
