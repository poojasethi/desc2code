#include <iostream>
#include <algorithm>
using namespace std;
int n,m;
const int mod=1000000007;
int c[1005][1005],f[1005];
int x,y,z;
int h[1005];

int main(){

    f[1]=1;
    for(int i=2;i<=1000;i++) f[i]=f[i-1]*2%mod;
    c[0][0]=1;
    for(int i=1;i<=1000;i++)
    {
        c[i][0]=c[i][i]=1;
        for(int j=1;j<i;j++) c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod;
    }

    cin>>n>>m;
    for(int i=0;i<m;i++) {int a;cin>>a;h[a]=1;}
    int i=1;

    long long ans=1,sum=0;

    for(;i<=n&&!h[i];i++)sum++;

    for(int a=i;a<=n;)
    {

        while(a<=n&&h[a]) a++;
        int g=0;
        while(a<=n&&!h[a]) g++,a++;
        if(a<=n)
        {
            ans=ans*c[sum+g][g]%mod*f[g]%mod;
        }
        else
        ans=ans*c[sum+g][g]%mod;
        sum+=g;
    }
    cout<<ans<<endl;
}
