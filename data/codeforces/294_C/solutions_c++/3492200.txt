# include <iostream>
# include <algorithm>

using namespace std;

# define INF 1000000007

long long n,m;

long long m1[1050];

long long def[1050][1050];

long long ex[1050];

long long ke;

long long ek=1;

long long ans=1;

int main()
{
	cin>>n>>m;
	
	ke=n-m;
	
	for (int i=1;i<=m;i++) cin>>m1[i];
	
	m1[0]=0;
	
	sort(m1,m1+m+1);
	
	ex[0]=1;
	
	for (int i=1;i<=1001;i++) ex[i]=ek,ek=(ek*2)%INF;
	
	for (int i=0;i<=1000;i++)
	{
		def[i][i]=1;
		
		for (int j=i+1;j<=1000;j++)
		def[j][i]=(def[j-1][i]+def[j-1][i-1])%INF;
	}
	
	for (int i=0;i<m;i++) ans=(ans*(def[ke][m1[i+1]-m1[i]-1]))%INF,ke-=m1[i+1]-m1[i]-1;
	
	for (int i=1;i<m;i++) ans=(ans*(ex[m1[i+1]-m1[i]-1]))%INF;
	
	cout<<ans;
	
//	getchar();getchar();
}
