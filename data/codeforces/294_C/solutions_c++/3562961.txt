#include<stdio.h>
#include<algorithm>
#define LL __int64
using namespace std;
const int mod=1000000007;
    
LL s,c[1001][1001]={0},bt[1001],x[1001];
int main()
{   
    int i,j,m,n,k;
    for(c[0][0]=i=1;i<1001;i++)
        for(c[i][0]=j=1;j<=i;j++)
            c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod;
    for(bt[0]=i=1;i<1001;i++)
        bt[i]=(bt[i-1]<<1)%mod;
    while(scanf("%d%d",&n,&m)!=EOF)
    {
        for(i=0;i<m;i++)
            scanf("%I64d",&x[i]);
        sort(x,x+m);
        for(s=1,i=0;i<m-1;i++)
        {
            if(x[i+1]==x[i]+1) continue;
            k=n-x[i]-(m-i-1);
            s=((s*c[k][x[i+1]-x[i]-1])%mod*bt[x[i+1]-x[i]-2])%mod;
        }
        s=(s*c[n-m][x[0]-1])%mod;
        printf("%I64d\n",s);
    }
    return 0;
}
