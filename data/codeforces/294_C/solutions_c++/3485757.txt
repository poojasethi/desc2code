#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
const long long mo=1000000007;
using namespace std;
long long c[2010][2010],f[2010],n,m,a[2010],ans,sum;
int main()
{
    cin>>n>>m;
    for (int i=1;i<=m;i++) cin>>a[i];
    sort(a+1,a+m+1);
    f[1]=f[0]=1;
    for (int i=2;i<=n;i++) f[i]=(f[i-1]*2)%mo;
    c[0][0]=1;
    for (int i=1;i<=2000;i++) {
        c[i][0]=1;
        for (int j=1;j<=i;j++) c[i][j]=(c[i-1][j-1]+c[i-1][j])%mo;
    }
    ans=1,sum=a[1]-1;
    for (int i=1;i<=m-1;i++) {
        if (a[i+1]!=a[i]) {
            ans=(ans*c[sum+a[i+1]-a[i]-1][a[i+1]-a[i]-1])%mo;
            ans=(ans*f[a[i+1]-a[i]-1])%mo;
            sum+=a[i+1]-a[i]-1;
        }
    }
    if (a[m]!=n) ans=(ans*c[sum+n-a[m]][sum])%mo;
    cout<<ans;
    return 0;
}
