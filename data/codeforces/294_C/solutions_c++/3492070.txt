#include <algorithm>
#include <iostream>
using namespace std;

typedef long long LL;

const LL maxn = 1000 + 100;
const LL mod = ( LL )1e9 + 7;
LL c[maxn][maxn];
LL a[maxn];
LL k, ans;

void f( LL x )
{
	ans *= c[k + 1][x + 1];
	ans %= mod;
	k -= x;
}

int main()
{
	int n, m;
	cin >> n >> m;
	for( int i = 0; i < m; i++ )
		cin >> a[i];
	sort( a, a + m );
	c[1][1] = 1;
	for( int i = 1; i < n + 10; i++ )
		for( int j = 1; j < n + 10; j++ )
			if( i + j > 2 )
				c[i][j] = ( c[i - 1][j] + c[i - 1][j - 1] ) % mod;
	k = n - m;
	ans = 1;
	f( a[0] - 1 );
	f( n - a[m - 1] );
	for( int i = 1; i < m; i++ )
		f( a[i] - a[i - 1] - 1 );
	for( int i = 1; i < m; i++ )
		for( int j = 0; j < a[i] - a[i - 1] - 2; j++ )
			ans *= 2, ans %= mod;
	cout << ans << endl;
	cin >> n;
}
