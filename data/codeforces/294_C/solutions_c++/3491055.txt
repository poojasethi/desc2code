#include<iostream>
#include<cstdlib>
#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

const int maxn=1005;

int n,m;
int c[maxn];
const long long mod=1000000007ll;
long long ans=1;
long long f[maxn];

long long pow(long long a,long long b) {
	long long ret=1ll;
	for(;b;a=a*a%mod,b>>=1) if(b&1)
		ret=ret*a%mod;
	return ret;
}

int main() {
	scanf("%d%d",&n,&m);
	f[0]=1ll;
	for(int i=1;i<=n;i++)
		f[i]=f[i-1]*i%mod;
	for(int i=1;i<=m;i++)
		scanf("%d",&c[i]);
	sort(c+1,c+1+m);
	c[m+1]=n+1;
	for(int i=0,t;i<=m;i++) if((t=c[i+1]-c[i])>0) {
		t--;
		if(i!=0 && i!=m && t>=1)
			ans=ans*pow(2,t-1)%mod;
		ans=ans*pow(f[t],mod-2)%mod;
	}
	ans=ans*f[n-m]%mod;
	printf("%d\n",(int)ans);
	return 0;
}
