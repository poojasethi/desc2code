#include<cstdio>
#include<algorithm>
#include<iostream>
#define mod 1000000007
using namespace std;
int n,m;
long long f[1100];
int a[1100],q[1100];

long long cal(long long x,long long y) {
    if (y==0) return 1;
    if (y==1) return x;
    long long t=cal(x,y/2),now=t*t%mod;
    if (y%2==1) now=now*x%mod;
    return now;
}

int main() {
    int i;
    long long ans,num,sum;
    scanf("%d%d",&n,&m);
    for (i=1;i<=m;i++) scanf("%d",a+i);
    sort(a+1,a+m+1);
    q[0]=a[1]-1;
    sum=0;
    for (i=2;i<=m;i++) {
        q[i-1]=a[i]-a[i-1]-1;
        if (q[i-1]>=1) sum++;
    }
    q[m]=n-a[m];
    num=n-m;
    f[0]=1;
    for (i=1;i<=n;i++) f[i]=f[i-1]*i%mod;
    ans=f[num];
    for (i=0;i<=m;i++) 
        ans=ans*cal(f[q[i]],mod-2)%mod;
    ans=ans*cal(2,num-q[0]-q[m]-sum)%mod;
    cout<<ans<<endl;
}
