#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN=1000+100;
const int MOD=1000000007;
long long pas[MAXN][MAXN];
int a[MAXN];

int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<=n+1;i++)
	{
		pas[i][i]=1;
		pas[0][i]=1;
	}
	for(int i=1;i<=n+1;i++)
		for(int j=i;j<=n+1;j++)
			pas[i][j]=( pas[i-1][j-1]%MOD + pas[i][j-1]%MOD ) % MOD;
	int x=n-m;
	long long ans=1;
	for(int i=1;i<=m;i++)
		cin>>a[i];
	sort( a+1, a+m+1 );
	ans=( ans * pas[ a[1]-1 ][x] )%MOD;
	x-=( a[1]-1 );
	ans=( ans * pas[ n-a[m] ][x] )%MOD;
	x-=( n-a[m] );
	for(int i=1;i<m;i++)
	{
		long long flg=1;
		for(int j=a[i]+1;j<a[i+1]-1;j++)
			flg=( flg*2 )%MOD;
		flg=(flg * pas[ a[i+1]-a[i]-1 ][x] )%MOD;
		ans=( ans * flg )%MOD;
		x-= ( a[i+1]-a[i]-1 );
	}
	cout<< ans << endl;
}
