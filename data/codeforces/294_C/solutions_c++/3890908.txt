#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

const int MOD = 1000000007;

int n, m, arr[1005], pow = 0, dp[1005][1005] = {}, done = 0;

int binom(int a, int b){
    if(!b) return 1;
    if(a<b) return 0;
    if(!dp[a][b]) dp[a][b] = (binom(a-1, b-1)+binom(a-1, b))%1000000007;
    return dp[a][b];
}

int main(){

    cin >> n >> m;
    done = m;
    for(int i = 0; i<m; i++) cin >> arr[i];
    sort(arr, arr+m);

    long long res = 1;
    for(int i = 0; i<m-1; i++){
        pow+=max((arr[i+1]-arr[i]-2), 0);
        res = (res*binom(n-done, arr[i+1]-arr[i]-1))%MOD;
        done+=arr[i+1]-arr[i]-1;
    }
    for(int i = 0; i<pow; i++) res = (2*res)%MOD;
    cout << (res*binom(n-done, arr[0]-1))%MOD << endl;

}
