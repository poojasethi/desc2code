#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <iostream>
#define MD 1000000007
using namespace std;
int lamp[1005],n,m;
int ara[1005];
int us(int x,int y)
{

	if(y==0)	return 1;
	int l=(1ll*us(x,y/2))%MD;
	l=(1ll*l*l)%MD;
	if(y%2)	
		return (1ll*l*x)%MD;
	return l;
}
int iki[1005],fak[1005];
int main()
{
	cin>>n>>m;
	for (int i = 0; i < m; i++)
		cin>>lamp[i];
	iki[0]=1;	
	for(int i=1;i<=1000;i++)
		iki[i]=(iki[i-1]*2ll)%MD;
	fak[0]=1;
	for(int i=1;i<=1000;i++)
		fak[i]=(1ll*i*fak[i-1])%MD;
	sort(lamp,lamp+m);
	int bck=1;
	for (int i = 0; i < m; i++)
	{
		ara[i]=lamp[i]-bck;
		bck=lamp[i]+1;
	}
	ara[m]=n-lamp[m-1];
	n-=m;
	int s=fak[n];
	
	for (int i = 0; i < m+1; i++)
	{
		if(!ara[i])
			continue;
		s=(1ll*s*us(fak[ara[i]],MD-2))%MD;
		if(i!=0 && i!=m)
			s=(1ll*s*iki[ara[i]-1])%MD;	
	}
	cout<<s;
	return 0;
}
