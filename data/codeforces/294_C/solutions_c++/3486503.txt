#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
const int mod = 1e9+7;
int num[1005];
ll C[1005][1005];
ll pow(int n){
	ll ans = 1;
	for(int i = 0; i < n; i++)
		ans = 2 * ans % mod;
	return ans;
}
int main(){
	int n,m;
	cin >> n >> m;
	C[0][0] = 1;
	for(int i = 1; i < 1005; i++){
		C[i][0] = 1;
		for(int j = 1; j < 1005; j++)
			C[i][j] = (C[i-1][j-1] + C[i-1][j]) % mod;
	}
	for(int i = 0; i < m; i++)
		cin >> num[i];
	sort(num,num+m);
	int L = num[0]-1;
	int R = n - num[m-1];
	ll ans = C[L+R][L];
	int now = L + R;
	for(int i = 1; i < m; i++){
		int len = num[i] - num[i-1] - 1;
		ans = ans *  C[len + now][len] % mod;
		ans = ans * pow(len - 1) % mod;
		now += len;
	}
	cout<< ans <<endl;

}
