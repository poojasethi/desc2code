#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstdlib>
#define modnum 1000000007
#define int64 long long

int n,m,x[5100];
int64 jc[5100],ans;
int64 sqr(int64 x) {return (x*x)%modnum;}
int64 exp(int64 a,int b)
{
  if (!b) return 1;
  return (sqr(exp(a,b>>1))*(b%2?a:1))%modnum;	    	
}
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;++i) scanf("%d",&x[i]);
	std::sort(x+1,x+m+1); ans=1; jc[0]=1;
	for (int i=1;i<=n-m;++i) jc[i]=(jc[i-1]*i)%modnum;
	ans=jc[n-m]; x[m+1]=n+1; x[0]=0;
	for (int i=1;i<=m+1;++i) ans=(ans*exp(jc[x[i]-x[i-1]-1],modnum-2))%modnum;
	for (int i=2;i<=m;++i)
	  if (x[i]-x[i-1]-1) ans=(ans*exp(2,x[i]-x[i-1]-2))%modnum;
	printf("%I64d\n",ans); 
	return 0;
}
