#include<cstdio>
#include<algorithm>
using namespace std;
#define LL long long

const LL MOD=1000000007;
LL c[1001][1001]={0},p[1001]={1};
LL ans=1;
int n,m,light[1001],num[1001],cnt=0;
int cmp(int a,int b) {return a<b;}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=0;i<m;i++)
	  scanf("%d",&light[i]);
	sort(light,light+m,cmp);
	for(int i=1;i<=n;i++)
	  p[i]=p[i-1]*2%MOD;
	for(int i=0;i<=n;i++)
	  c[i][0]=1;
	for(int i=1;i<=n;i++)
	  for(int j=1;j<=i;j++)
		c[i][j]=(c[i-1][j-1]+c[i-1][j])%MOD;
	num[cnt++]=light[0]-1;
	for(int i=1;i<m;i++)
	{
		num[cnt]=light[i]-light[i-1]-1;
		if(num[cnt]>1)
			ans=ans*p[num[cnt]-1]%MOD;
		cnt++;
	}
	long long sum=0;
	num[cnt++]=n-light[m-1];
	for(int i=0;i<cnt;i++)
	{
	  ans=ans*c[n-m-sum][num[i]]%MOD;
	  sum+=num[i];
	}
	printf("%I64d\n",ans);
	return 0;
}
