#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <iostream>
#define MD 1000000007
using namespace std;
int lamp[1005],n,m;
int ara[1005];
int fak(int k)
{
	int x=1;
	for (int i = 1; i <=k; i++)
		x=(1ll*i*x)%MD;
	return x;
}
int us(int x,int y)
{

	if(y==0)	return 1;
	int l=(1ll*us(x,y/2))%MD;
	l=(1ll*l*l)%MD;
	if(y%2)	
		return (1ll*l*x)%MD;
	return l;
}
int main()
{
	cin>>n>>m;
	for (int i = 0; i < m; i++)
		cin>>lamp[i];
	sort(lamp,lamp+m);
	int bck=1;
	for (int i = 0; i < m; i++)
	{
		ara[i]=lamp[i]-bck;
		bck=lamp[i]+1;
	}
	ara[m]=n-lamp[m-1];
	n-=m;
	int s=fak(n);
	
	for (int i = 0; i < m+1; i++)
	{
		if(!ara[i])
			continue;
		s=(1ll*s*us(fak(ara[i]),MD-2))%MD;
		if(i!=0 && i!=m)
			s=(1ll*s*us(2,ara[i]-1))%MD;	
	}
	cout<<s;
	return 0;
}
