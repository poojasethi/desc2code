#include <bits/stdc++.h>
using namespace std;

// From tutorial :(

#define ll long long
ll mod = 1000000007;
ll n,m,nCr[1001][1001],ans=1,s[1001],pw[1001];

int main(){

    cin >> n >> m;

    for(int i=0;i<=n;i++){
        for(int j=0;j<=i;j++){
            if(j==0 || j==i) nCr[i][j]=1;
            else nCr[i][j]=(nCr[i-1][j-1]+nCr[i-1][j])%mod;
        }
    }

    pw[0]=1;pw[1]=1;
    for(int i=2;i<=n;i++) pw[i]=(pw[i-1]*2)%mod;
    for(int i=0;i<m;i++) cin >> s[i];
    sort(s,s+m);
    ll cnt=0,tmp;
    for(int i=0;i<m;i++){
        if(i==0) tmp=s[i]-1;
        else{
            tmp=s[i]-s[i-1]-1;
            ans=(ans*pw[tmp])%mod;
        }
        cnt+=tmp;
        ans=(ans*nCr[cnt][tmp])%mod;
    }
    cnt+=n-s[m-1];
    ans=ans*nCr[cnt][n-s[m-1]]%mod;
    cout << ans << endl;
}
