#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<stdlib.h>
using namespace std;
#define MOD 1000000007
__int64 a[1050][1050],ans=1;
int b[1010];
void init()
{
	int i,j;
	for(i=0;i<=1009;i++)
		for(j=0;j<=1009;j++)
			a[i][j]=1;
	for(i=1;i<=1009;i++)
		a[1][i]=i;
	for(i=2;i<=1009;i++)
	{
		for(j=2;j<=1009;j++)
		{
			a[i][j]=(a[i-1][j]+a[i][j-1])%MOD;
		}
	}
	return ;
}
int main()
{
	init();
	int n,m,x,y,z,i,j;
	//printf("%d %d\n",a[3][4],a[3][7]);
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		ans=1;
		y=1;
		for(i=0;i<m;i++)scanf("%d",&b[i]);
		sort(b,b+m);
		x=b[0];
		z=x-y;
		y=x;
		for(j=1;j<m;j++)
		{
			x=b[j];
			for(i=0;i<(x-y-2);i++)
				ans=(ans*2)%MOD;
			ans=(ans*a[x-y-1][z+1])%MOD;
			z=z+x-y-1;
			y=x;
		}
		if(x!=n)ans=(ans*a[n-x][z+1])%MOD;
		printf("%I64d\n",ans);
	}
	return 0;
}
