#include<stdio.h>
#include<iostream>
#include<string.h> 
#include<algorithm>
#define md 1000000007
using namespace std;
long long jc(int a)
{
	long long re=1;
	for(int j=1;j<=a;j++)
	re=(re*j)%md;
	return re%md;
}
long long pow(long long a,long long b)
{
	if(a==1||b==1)return a;
	long long ans=1;
	while(b>0)
	{
		if(b&1)ans=ans*a%md;
		a=a*a%md;
		b/=2;
	}
	return ans%md;
}
int main()
{
	int i,n,m,aa[1005];
	long long ans=1;
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++)
	scanf("%d",&aa[i]);
	sort(aa,aa+m);
	ans=ans*jc(n-m)%md;
	ans=ans*pow(jc(aa[0]-1),md-2)%md;
	for(i=1;i<m;i++)
	{
		long long t=jc(aa[i]-aa[i-1]-1);
		ans=ans*pow(t,md-2)%md;
	}
	ans=ans*pow(jc(n-aa[m-1]),md-2)%md;
	for(i=1;i<m;i++)
	{
		if(aa[i]-aa[i-1]-2>=0)
		ans=ans*pow(2,aa[i]-aa[i-1]-2)%md;
	}
	printf("%lld\n",ans);
}
