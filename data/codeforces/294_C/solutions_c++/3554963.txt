#include <stdio.h>
#include <iostream>
#include <algorithm>

using namespace std;

#define	MAXN	1010
#define	MOD	1000000007
#define	LL	long long int

int n, m, a[MAXN];
LL p2[MAXN], nCr[MAXN][MAXN];

int main (void) {
	int i, j, sum, tmp;
	LL ans;

	p2[0] = 1;
	for(i=1; i < MAXN; i++) p2[i] = (p2[i-1]*2LL)%MOD;

	for(i=0; i < MAXN; i++)
		for(j=0; j <= i; j++)
			if(j==0||j==i) nCr[i][j] = 1;
			else nCr[i][j] = (nCr[i-1][j] + nCr[i-1][j-1])%MOD;

	while(scanf("%d %d", &n, &m) != EOF) {
		for(i=0; i < m; i++) cin >> a[i];
		sort(a,a+m);
		
		sum = a[0]-1;
		ans = 1;
		for(i=1; i < m; i++) {
			tmp = a[i]-a[i-1]-1;
			ans = (ans*nCr[sum+tmp][sum])%MOD;
			if(tmp) ans = (ans*p2[tmp-1])%MOD;
			sum += tmp;
		}
		if(a[m-1] != n) {
			tmp = n-a[m-1];
			ans = (ans*nCr[sum+tmp][sum])%MOD;
		}

		cout << ans << endl;
	}

	return 0;
}
