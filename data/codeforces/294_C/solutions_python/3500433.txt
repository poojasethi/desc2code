mod = int(1e9 + 7)
n, m = map(int, raw_input().split())
l = map(int, raw_input().split())

l.sort()
ll = []
ans = 1
for i in xrange(2, n - m + 1):
    ans *= i

ll.append(l[0] - 1)
for i in xrange(1, len(l)):
    if l[i] - l[i - 1] - 1:
        ll.append(l[i] - l[i - 1] - 1)
ll.append(n - l[-1])

for i in ll[1:-1]:
    ans *= 2 ** (i - 1)

for i in ll:
    b = 1
    for j in xrange(2, i + 1):
        b *= j
    ans /= b

print ans % mod
