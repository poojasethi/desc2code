M = 1000000007
F = [1]
for i in range(1, 1000):
	F += [F[i - 1] * i % M]
f = [pow(x, M - 2, M) for x in F]
C = lambda n, m : F[n] * f[m] * f[n - m] % M
N, K = map(int, raw_input().split())
a = sorted(map(int, raw_input().split())) + [N + 1]
r = 1
n = N - K
p = 1
for x in a:
	r = r * C(n, x - p) % M
	if p != 1 and x != N + 1 and x != p:
		r = r * pow(2, x - p - 1, M) % M
	n -= x - p
	p = x + 1
print r
