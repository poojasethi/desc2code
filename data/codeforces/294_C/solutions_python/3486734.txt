import sys
from math import factorial

inp = [[int(j) for j in i.split(' ')] for i in sys.stdin.read().strip().splitlines()]
inp[1].sort()

fx = []
res = 1
for i, j in zip([0] + inp[1], inp[1] + [inp[0][0] + 1]):
    if i == 0 or j == inp[0][0] + 1:
        res *= 1
    else:
        if j - i > 1:
            res *= 2**(j-i-2)
    fx += [j-i-1]

res *= factorial(sum(fx))
for i in fx:
    res /= factorial(i)

sys.stdout.write(str(res % 1000000007))
