from math import *
R=lambda:map(int,raw_input().split())
C=lambda n,k:factorial(n)/(factorial(k)*factorial(n-k))
n,m=R()
mod=10**9+7
a=[0]+R()+[n+1]
ans=1
le=0
a=sorted(a)
for i in range(0,len(a)):
    if i>1 and i<len(a)-1 and a[i]-a[i-1]-2>0:
        ans*=2**(a[i]-a[i-1]-2)
        ans%=mod;
    if i>0 and a[i]-a[i-1]-1>0:
        le+=a[i]-a[i-1]-1
        ans*=C(le,a[i]-a[i-1]-1)
        ans%=mod
print ans
