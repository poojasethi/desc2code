
import sys
from math import factorial

I=lambda:map(int, raw_input().split())
n,m=I()
a=sorted([0]+I()+[n+1])
b=[]
for i in xrange(1,len(a)):
    b.append(a[i]-a[i-1]-1);

ans = 1
for i,x in enumerate(b):
    if i>0 and i !=len(b)-1 and x>=1:
        ans=ans*2**(x-1)
ans *= factorial(n-m)
for i in b:
    ans /= factorial(i)

print ans%1000000007
