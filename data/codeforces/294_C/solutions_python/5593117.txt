import math
f = math.factorial
n, m = map(int, raw_input().split(' '))
pt = 0
s = raw_input()
a = map(int, s.split(' '))
a.sort()
start = 0
region = 0
res = 1
den = 1
first = True
for i in a:
	cnt = i - start - 1
	region += cnt
	if first is False and cnt >=1 :
		res *= (2**(cnt-1))
	else:
		first = False
	start = i
	if cnt >= 0:
		den *= f(cnt)
cnt = n - a[len(a)-1]
region += cnt
den *= f(cnt)
print int(res*f(region)//den)%(10**9+7)
