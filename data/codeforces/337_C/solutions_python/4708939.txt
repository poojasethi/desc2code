Mod = 1000000009
def mPow(x, y):
    if y == 0:
        return 1
    ans = mPow(x, y/2)
    ans = (ans * ans) % Mod
    if y%2 > 0:
        ans = (ans * x) % Mod
    return ans

n, m, k = map(int, raw_input().split())
fr = n % k + (n/k)*(k-1)
if fr >= m:
        ans = m
else:
        a = m - fr
        fr -= a*(k-1)
        ans = mPow(2,a)
        ans = (ans + Mod - 1) % Mod
        ans = (ans * 2) % Mod
        ans = (ans * k) % Mod
        ans = (ans + fr) % Mod
print ans
