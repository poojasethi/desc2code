N, M, K = map(int,raw_input().strip().split())
incorrect = N - M
mod = 1000000009
if (M * 1.0) / (K-1) <= incorrect:print M % mod
else:
    total_baz = (K-1)*incorrect
    fadel = M - total_baz
    odd = fadel % K
    fadel -= odd
    fadel /= K
    temp = (pow(2, fadel+1, mod) - 2) * K
    temp = total_baz + temp + odd
    temp %= mod
    print temp
