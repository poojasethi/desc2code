import sys

mod = 1000000009

def my_pow(n):
	if n == 0:
		return 1
	if n % 2:
		return (2 * my_pow(n - 1)) % mod
	div = my_pow(n / 2)
	return (div * div) % mod

n, m, k = map(int, sys.stdin.readline().strip().split())
chunk = n / k
r = m - (chunk * (k - 1) + n % k)
p = max(r, 0)

sol = 0
if (p):
	sol += (k * my_pow(p + 1)) % mod
	sol -= 2 * k
	sol = (sol + mod) % mod
	m -= k * p

print (sol + m) % mod
