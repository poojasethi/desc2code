mod = 1000000009

def mpow(x, y):
	if y == 0:
		return 1
	ans = mpow(x, y/2)
	ans = (ans * ans) % mod
	if y%2 > 0:
		ans = (ans * x) % mod
	return ans
		

n, m, k = map(int, raw_input().split())
fr = n % k + (n/k)*(k - 1)
if fr >= m:
	ans = m
else:
	a = m - fr
	fr -= a*(k-1)
	ans = mpow(2, a)
	ans = (ans + mod - 1) % mod
	ans = (ans * 2) % mod
	ans = (ans * k) % mod
	ans = (ans + fr) % mod
print ans
