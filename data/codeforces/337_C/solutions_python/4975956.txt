n, m, k = map(int, raw_input().split())

wrongNum = n - m
intervalNum = n / k

M = (10 ** 9 + 9)
if wrongNum >= intervalNum :
    score = m
else :
    prefixSum = 2*k * (pow(2, intervalNum - wrongNum, M) - 1)
    suffixSum = (m - ((intervalNum-wrongNum) * k))
    score = (prefixSum + suffixSum) % M

print score
