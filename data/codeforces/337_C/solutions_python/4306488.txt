MOD = 10**9+9

n, m, k = map(int, raw_input().split())
score = 0

score += min(m, n/k*(k-1))
score %= MOD
m -= score
if n % k != 0 and m > 0:
    score += min(m, n % k)
    score %= MOD
    m -= min(m, n % k)
score += (k * (pow(2, m+1, MOD) - 2 + MOD) % MOD) % MOD
score %= MOD
score -= m*(k-1) + MOD
score %= MOD
print score
