import sys

n, m, k = [int(x) for x in raw_input().strip().split()]

zeroes = n-m
if zeroes >= n/k:
    print m
    sys.exit()
fullsets = n/k - zeroes
remaining = n - k*fullsets - zeroes

bignum = 1000000009

score = (k * (pow(2, fullsets+1, bignum) - 2)) % bignum

score += remaining
score %= bignum

print score
