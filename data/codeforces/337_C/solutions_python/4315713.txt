def power(a, mod) :
    s, g = 1, 2
    while a > 0 :
        if (a & 1) :
            s = s * g % mod
        a >>= 1
        g = g * g % mod
    return s

mod = 1000000009

n, m, k = map(int, raw_input().split())
w = n / k
if m + w <= n :
    print m
else :
    e = m + w - n
    p = power(e + 1, mod)
    ans = ((p - 2) * k + m - e * k) % mod
    print ans
