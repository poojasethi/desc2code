import math
if __name__=="__main__":
    n,m,k = map(int, raw_input().split())
    md = m+0.
    maxA = 10**9+9

    if  (math.ceil(md/(k-1))-1)*k+ (m-(math.ceil(md/(k-1))-1)*(k-1)) <= n:
        print m%maxA
    else:
        p1=math.floor((n+0.)/k)
        extra=int(m-p1*(k-1)-(n-p1*k))
        #print n,extra, p1
        res=pow(2,extra+1,maxA)
        res=(res*k+maxA-2*k)%maxA
        res=(res+(m-extra*k))%maxA
        print res
