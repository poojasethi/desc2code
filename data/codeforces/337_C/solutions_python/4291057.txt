MODULO = 1000000009

def pow(n,k):
    if k==0:
        return 1
    if k==1:
        return n
    pp = pow(n,k/2)
    if k%2 == 0:
        return (pp * pp) % MODULO
    else:
        return (pp * pp * n) % MODULO

n,m,k = [int(x) for x in raw_input().split()]

bb = (n/k) * (k-1) + (n%k)

if m <= bb:
    print m
else:
    t = m - bb
    dp = ((pow(2,t+1) - 2)*k) % MODULO
    sol = (MODULO + dp + (m - k*t)) % MODULO
    #print ">>",t,dp
    print sol
