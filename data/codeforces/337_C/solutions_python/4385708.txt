def pow_mod(x, y, z):
    res = 1 
    while (y > 0):
        if ((y&1) == 1):
            res=res*x%z
        x = x*x%z
        y >>= 1
    return res%z
n, m, k = map(int, raw_input().split())
C = 1000000009
num_sequence = n/k - n + m
if (num_sequence <= 0):
    print m
else:
    score = pow_mod(2, num_sequence, C)
    score = (2*(score-1)*k + m - num_sequence*k)%C
    print score
