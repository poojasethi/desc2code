#include<cstdio>
using namespace std;
#define ll long long
const int MOD=1000000009;
ll n,m,k;
ll two(int x)
{
	ll res=1,t=2;
	while(x)
	{
		if(x&1) res=res*t%MOD;
		t=t*t%MOD;
		x>>=1;
	}
	return res;
}
ll run()
{
	ll t=n-m,r;
	if((t+1)*k>n) return m;
	n-=t*k;
	r=(t*(k-1)+n%k)%MOD;
	n/=k;
	return (r+2*k*(two(n)+MOD-1)%MOD)%MOD;
}
int main()
{
	scanf("%lld%lld%lld",&n,&m,&k);
	printf("%lld\n",run());
	return 0;
}
