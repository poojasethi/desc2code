#include<bits/stdc++.h>
using namespace std;
#define mod 1000000009
#define ll long long 
ll power(ll a,ll b)
{
	if(b==0)
		return 1;
	ll temp=power(a,b/2);
	temp=(temp*temp)%mod;
	if(b&1)
		temp=(temp*a)%mod;
	return temp;		
}
int main()
{
	ll n,m,k,x,y;
	cin>>n>>m>>k;
	x=max((ll)0,m-(n/k)*(k-1)-n%k);
	//cout<<x<<endl;
	y=(power(2,x+1)-2)*k+m-x*k;
	y=(y+mod)%mod;
	cout<<y<<endl;
	return 0;
}
