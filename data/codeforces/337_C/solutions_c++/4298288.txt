#include <cstdio>
#include <algorithm>
#define LL long long
using namespace std;

const LL MO=1000000009;

LL n,m,k,ans;

LL Pow(LL b) {
	LL a=2,ret=1;
	for (;b;b>>=1) {
		if (b&1) ret=ret*a%MO;
		a=a*a%MO;
	}
	return ret;
}

int main() {
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	ans=max(m-n/k*(k-1)-n%k,0ll);
	printf("%I64d\n",(((Pow(ans+1)-2)*k+m-ans*k)%MO+MO)%MO);
	return 0;
}
