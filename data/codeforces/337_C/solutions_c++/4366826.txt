#include<iostream>
#define mo 1000000009
using namespace std;

long qsm(long x)
{
    if(x==1)return 2;
    if(x&1)return qsm(x-1)*2%mo;
    long long temp=qsm(x/2); 
    return temp*temp%mo;   
}

int main()
{
    long long n,m,k; cin>>n>>m>>k;
    if(n-m>=n/k)cout<<m%mo<<endl;
    else 
        {
            long long x=n-(n-m)*k;
            cout<<(((qsm(x/k+1)-2+mo)%mo*k%mo+x%k)%mo+(k-1)*(n-m))%mo<<endl;
        }
    return 0;
}
