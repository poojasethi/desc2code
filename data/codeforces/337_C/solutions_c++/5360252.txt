#include <cstdio>
long long f(long long pow){
    if (pow==0){
        return 1;
    } else if (pow%2==1){
        return (2*f(pow-1))%1000000009;
    } else {
        return (f(pow/2)*f(pow/2))%1000000009;
    }
}
