#include <iostream>
using namespace std;

int main(){
    long long n, m, k, d, res = 1, MOD = 1000000009;
    cin >> n >> m >> k;
    d = max(0LL, m-n/k*(k-1)-n%k);
    for(int i = 0; i+30<=d; i+=30) res = (res<<30)%MOD;
    res = (res<<d%30)%MOD;
    cout << (2*k*(res-1)+max(0LL, m-d*k))%MOD << endl;
}
