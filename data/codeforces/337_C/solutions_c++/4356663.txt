#include<iostream>
#include<stdio.h>
#include<cstring>
#include<algorithm>
#define  INF 1000000009
using namespace std;
int n,m,k,i;
long long ans=0,c,s;
int main(){
	cin>>n>>m>>k;
	c=n-m;
	if ((c+1)*k>n){
		cout<<m;
		return 0;
	}
	s=n-k*c;
	ans=(s%k+c*(k-1))%INF;
	long long p=s/k+1,a=1,q=2;
	while(p){
		if (p&1)
			a=a*q%INF;
		p>>=1;
		q=q*q%INF;
	}
	ans=(ans+(a-2+INF)*k%INF)%INF;
	cout<<ans;
	return 0;
}
