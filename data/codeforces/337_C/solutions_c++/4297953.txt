#include <iostream>
using namespace std;
typedef long long ll;
const int mod = 1000000009;
ll n,m,k,ans;

ll fast_power(ll base, ll esp){
	if(esp==0)return 1;
	ll q = fast_power(base,esp/2);
	if(esp%2==0) return (q*q)%mod;
	return (q*((q*base)%mod))%mod;
}

int main(){
	cin >> n >> m >> k;
	ans = m;
	m -= (n/k)*(k-1)+n%k;
	if(m>0) ans = (ans + k*((fast_power(2,m+1) -2 -m)%mod)%mod) % mod;
	if(ans<0) ans+=mod;
	cout << ans;
	return 0;
}
