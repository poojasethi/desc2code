#include <iostream>
using namespace std;
typedef long long ll;
const ll MOD=1000000009;
ll fe(ll a,ll b)
{
    ll t=a,ans=1;
    for (;b;b>>=1)
    {
        if (b&1) ans=ans*t%MOD;
        t=t*t%MOD;
    }
    return ans;
}
int main()
{
    ll n,m,k,ans;
    cin >>n >>m >>k;
    ll d=n/k-(n-m);
    if (d<0) d=0;
    ans=((fe(2,d+1)-2)*k)%MOD;
    ans=(ans+m-d*k)%MOD;
    if (ans<0) ans+=MOD;
    cout <<ans <<endl;
    return 0;
}
