#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	long long n, m, k, a, b, x, i, ans;
	cin>>n>>m>>k;
	ans = min(m, n%k);
	n -= n%k;
	m -= ans;
	if (n-m>=n/k) ans += m;
	else {
		b = n/k-n+m;
		x = 1; a=2;
		while (b>0) {
			if (b%2==1) x = x*a%1000000009;
			a = a*a%1000000009;
			b/=2;
		}
		x = (x-1)*2*k%1000000009;
		ans += x + (n-m)*(k-1);
	}
	cout<<ans%1000000009<<endl;
	return 0;
}
