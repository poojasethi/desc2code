#include<iostream>
using namespace std;

typedef long long ll;

const int mod=1e9+9;

ll exp(ll a, ll b, ll m) {
	ll r=1;
	while(b) {
		if(b%2) r=(r*a)%m;
		a=(a*a)%m;
		b/=2;
	}
	return r;
}

int main() {
	ios_base::sync_with_stdio(false); cin.tie(NULL);
	ll n,m,k; cin>>n>>m>>k;	
	ll x=max(0LL, m-(n-n%k)/k*(k-1) - n%k);
	cout << (((exp(2,x+1,mod)-2+mod)*k)%mod +m+(mod-x)*k)%mod << endl;
	return 0;
}
