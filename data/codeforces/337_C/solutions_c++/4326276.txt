#include<iostream>
#define ll long long
using namespace std;
const ll MOD=1000000009;
ll ppp(ll a,ll b){
    ll t=a,ans=1;
    while(b){
        if(b%2)
            ans=ans*t%MOD;
        t=t*t%MOD;
        b/=2;
    }
    return ans;
}
int main(){
    ll n,m,k,ans;
    cin>>n>>m>>k;
    ll d=n/k-(n-m);
    if(d<0)
        d=0;
    ans=((ppp(2,d+1)-2)*k)%MOD;
    ans=(ans+m-d*k)%MOD;
    if(ans<0)
        ans+=MOD;
    cout<<ans;
}
