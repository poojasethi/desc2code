#include<iostream>
#define ll long long
using namespace std;
const ll MOD=1e9+9;
ll pow(ll a,ll b)
{
    return b?(pow((a*a)%MOD,b/2)*(b%2?a:1))%MOD:1;
}
int main()
{
    ll n,m,k;
    cin>>n>>m>>k;
    if(n-m>=n/k)
    {
        cout<<m<<endl;
        return 0;
    }
    cout<<((((n-m)*(k-1))%MOD+(k*(pow(2,(n-k*(n-m))/k+1)-2))%MOD+(n-k*(n-m))%k)%MOD+MOD)%MOD<<endl;
    return 0;
}
