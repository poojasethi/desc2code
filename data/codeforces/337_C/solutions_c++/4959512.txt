#include<cstdio>
#include<iostream>
typedef long long LL;
const   int M=1000000009;
LL  n,m,k,x,ans;
LL  pow(LL  x){
    LL  res=1,now=2;
    while(x){
        if(x&1)res=res*now%M;
        now=now*now%M,x>>=1;
    }
    return res;
}
int main(){
    std::cin>>n>>m>>k;
    x=m-(n-n%k)/k*(k-1)-n%k;
    if(x<0)x=0;
    ans=(pow(x+1)-2)*k%M+(m-x*k)%M;
    ans=(ans+M)%M;
    std::cout<<ans;
}
