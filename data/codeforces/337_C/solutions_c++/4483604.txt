#include<cstdio>
#define P 1000000009
#define LL long long int 

LL n,m,k,ans;
LL f(LL n,LL k){
    if (k==1)return 1;
    else return (k&1)?(1+n*f(n,k-1)%P):((n+1)*f(n*n%P,k>>1)%P);
}
int main(){
    scanf("%I64d%I64d%I64d",&n,&m,&k);
    if (n-m >= n/k){
        ans = m;
    }
    else {
        ans=2*f(2,n/k+m-n)*k%P+(m-(n/k+m-n)*k);
    }
    printf("%I64d\n",ans%P);
    return 0;
}
