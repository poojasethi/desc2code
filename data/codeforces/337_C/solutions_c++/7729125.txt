#include<stdio.h>
__int64 mod=1000000009;
__int64 count(__int64 x,__int64 y)
{
	__int64 temp=x,num=1;
	while(y)
	{
		if(y%2)
			num=num*temp%mod;
		temp=temp*temp%mod;
		y/=2;
	}
	return num;
}
int main()
{
	__int64 n,m,k,temp,res;
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	if(n<k*(n-m+1))
		printf("%I64d\n",m);
	else
	{
		temp=n/k+m-n;
		res=((k*(count(2,temp+1)-2-temp))%mod+m+mod)%mod;
		printf("%I64d\n",res);
	}
	return 0;
}
