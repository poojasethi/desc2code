#include<iostream>
#define ll long long int
using namespace std;
const ll mo=1000000009;
ll pow(ll x)
{
    if(x==(ll)0) return (ll)1;
    ll tem=pow(x/(ll)2)%mo;
    if(x%(ll)2)
        return tem*tem*(ll)2%mo;
    else
        return tem*tem%mo;
}
int main()
{
    ll n,m,k;
    cin>>n>>m>>k;
    ll p=n/k-(n-m);
    if(p<(ll)0) p=(ll)0;
    cout<<(m+(pow(p+(ll)1)-p-(ll)2+mo)%mo*k+mo)%mo;
    return 0;
}
