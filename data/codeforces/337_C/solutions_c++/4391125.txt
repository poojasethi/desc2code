#include<stdio.h>
#define mod (1000000009LL)

long long int pow(long long int k)
{
	if(k==1)return 2;
	long long int temp=pow(k/2);
	if(k%2==1)return (temp*temp*2)%mod;
	else return (temp*temp)%mod;
}

main()
{
	long long int m,n,k,r;
	while(scanf("%I64d%I64d%I64d",&n,&m,&k)!=EOF)
	{
		r=m-n+n/k;
		if(r<0)r=0;
		if(m==0)printf("0\n");
		else printf("%I64d\n",(k*(pow(r+1)-2)+(m-k*r)+mod)%mod);
	}
}
