#include<iostream>
using namespace std;
#define ll long long
#define mod 1000000009
int main()
{
    ll n,m,k,x,y,res,t;
    cin>>n>>m>>k;
    x=n-(n-m)*k;
    if(x<k)cout<<m<<endl;
    else
    {
        x/=k;y=x+1;
        res=1;t=2;
        for(;y;y>>=1)
        {
            if(y&1)res=res*t%mod;
            t=t*t%mod;
        }
        x=(res*k%mod-2*k%mod+(m-x*k)%mod+mod)%mod;
        cout<<x<<endl;
    }
}
