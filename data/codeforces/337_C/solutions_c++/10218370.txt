#include<iostream>
#define mo 1000000009
#define ll long long
using namespace std;
ll p(ll i)
{
    if(i==1) return 2;
    ll r=p(i/2);
    return (r*r*(i%2+1))%mo;
}
int main()
{
    ll n,m,k,x,y;
    cin>>n>>m>>k;
    x=n-m-n/k;
    k%=mo;
    y=m%mo;
    if(x<0) y+=((p(1-x)+x%mo-2)*k)%mo;
    y%=mo;
    if(y<0) y+=mo;
    cout<<y;
    return 0;
}
