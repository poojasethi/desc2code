n = int(raw_input())
array = [[0 for i in xrange(n)] for j in xrange(n)]

for i in xrange(n-1):
    for j in xrange(n-1):
        array[i][j] = (i+j)%(n-1)+1
for i in xrange(n-1):
    array[i][n-1] = array[n-1][i] = array[i][i]
    array[i][i] = 0
for i in xrange(n):
    print ' '.join(map(str,array[i]))
