n = int(raw_input())
a = [[-1 for i in range(n)] for i in range(n)]

for i in range(n-1):
    for j in range(n-1):
        x = (i+j)%(n-1)
        if x == 0: x = n-1
        a[i][j] = x        
        
for i in range(n-1):
    a[n-1][i] = a[i][n-1] = a[i][i]
    
for i in range(n):
    a[i][i] = 0
    
for i in range(n):
    print " ".join(map(str, a[i]))
