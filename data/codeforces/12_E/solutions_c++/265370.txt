#include <cstdio>
#include <cstring>

const int maxn = 1005;

int ans[maxn][maxn];

int main()
{
    int n;
    scanf( "%d" , &n );
    for ( int i = 0; i < n-1; i++ )
    {
        for ( int j = 0; j < n-1; j++ ) ans[i][j] = 1 + (i+j)%(n-1);
        ans[i][n-1] = ans[n-1][i] = ans[i][i];
        ans[i][i] = 0;
    }
    for ( int i = 0; i < n; i++ )
    {
        for ( int j = 0; j < n; j++ ) printf( "%d " , ans[i][j] );
        printf( "\n" );
    }   
    return 0;
}
