#include<iostream>
using namespace std;
int t[1234][1234];
int n;
int main()
{
  cin>>n;
  n--;
    for(int i=0;i<n;i++)
      for(int j=0;j<n;j++)
      t[i][j]=(i+j)%n+1;
     for(int i=0;i<n;i++)
      t[i][n]=t[n][i]=t[i][i],t[i][i]=0;
     for(int i=0;i<=n;i++)
      {cout<<t[i][0];
      for(int j=1;j<=n;j++)
       cout<<" "<<t[i][j];
       cout<<"\n";
  }
}
