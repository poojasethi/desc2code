#include<bits/stdc++.h>

using namespace std;

int a[2500][2500],n,i,j;

main()
{
 cin>>n;
 a[0][0]=n-1;
 for (i=1;i+1<n;i++) a[0][i]=i;
 for (i=1;i<n-1;i++)
  for (j=0;j+1<n;j++)
   a[i][j]=(a[i-1][(j+1)%(n-1)]);
 for (i=0;i<n-1;i++)
  {
   swap(a[i][i],a[i][n-1]);
   for (j=0;j<n;j++)
    cout<<a[i][j]<<" ";
    cout<<endl;
  }
 for (i=0;i<n;i++) cout<<a[i][n-1]<<" ";
}
