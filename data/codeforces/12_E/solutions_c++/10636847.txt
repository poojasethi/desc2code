#include<bits/stdc++.h>
using namespace std;
const int maxn = 1005;
int mat[maxn][maxn],n;
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            if(i==j) mat[i][j]=0;
            else if(i==n-1||j==n-1) mat[i][j]=(2*i+2*j)%(n-1)+1;
            else mat[i][j]=1+(i+j)%(n-1);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
            cout<<mat[i][j]<<" ";
        cout<<endl;
    }
    return 0;
}
