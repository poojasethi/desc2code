#include <cstdio>
#include <cstring>
#include <cctype>
#include <algorithm>
#include <cstdlib>
using namespace std;
int n;
int a[1100][1100];

int main()
{
	int i, j;
	scanf("%d", &n);
	n--;
	for (i = 0; i < n; i++)
    {
    	for (j = 0; j < n; j++)  a[i + 1][j + 1] = 1 + (i + j) % n;
    }
	n++;
	for (i = 1; i <= n; i++)
	{
	    a[n][i] = a[i][n] = a[i][i];
	    a[i][i] = 0;
	  for (j = 1; j <= n; j++) printf("%d ", a[i][j]);
	  	printf("\n");
	}
	return 0;
}
