#include <iostream>
using namespace std;

int n, map[1009][1009];

int main()
{
	cin >> n;
	for(int i = 0; i < n - 1; i++)
	{
		for(int j = 0; j < n - 1; j++)
			map[i][j] = (j + i) % (n - 1) + 1;
	}
	for(int i = 0; i < n - 1; i++)
	{
		map[n-1][i] = map[i][n-1] = map[i][i];
		map[i][i] = 0;
	}
	for(int i = 0; i < n; i++)
	{
		for(int j = 0; j < n; j++)
			cout << map[i][j] << " ";
		cout << endl;
	}
	return 0;
}
