#include<cstdio>
int n,a[1010][1010];
int main()
{
	scanf("%d",&n);
	for (int i=1; i<=n; i++)
		for (int j=1; j<=n; j++)
			a[i][j]=(i+j-2)%(n-1)+1;
	for (int i=1; i<=n; i++) 
		a[i][n]=a[n][i]=a[i][i],a[i][i]=0;
	for (int i=1; i<=n; i++)
	{
		for (int j=1; j<=n; j++)
			printf("%d ",a[i][j]);
		puts("");
	}
	return 0;
}
