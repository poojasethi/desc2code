#include <iostream>
using namespace std;
int main()
{
	int i,j,n,b[1001][1001],d[1001][1001];
	cin>>n;
	n--;
	for(i=0;i<n;i++)
	for(j=0;j<n;j++)
	b[i][j]=(i+j)%n+1;
	n++;
	for(i=0;i<n;i++)
	d[0][i]=i;
	for(i=1;i<n;i++)
	{
	for(j=0;j<n-1;j++)
	{
		if(i==j)
		d[i][j]=0;
		else
		d[i][j]=b[i-1][j];
	}
	d[i][n-1]=b[i-1][i];
	}
	for(i=0;i<n;i++)
	d[n-1][i]=d[i][n-1];
	for(i=0;i<n;i++)
	{
	for(j=0;j<n;j++)
	cout<<d[i][j]<<' ';
	cout<<endl;
	}	
	return 0;
}
