#include<iostream>
#include<cstring>
using namespace std;
const int mm=1007;
int map[mm][mm];
int n;
int main()
{
  while(cin>>n)
  { n--;
    
    for(int i=0;i<n;i++)
      for(int j=0;j<n;j++)
      map[i][j]=(i+j)%n+1;
     for(int i=0;i<n;i++)
      map[i][n]=map[n][i]=map[i][i],map[i][i]=0;
     for(int i=0;i<=n;i++)
      {cout<<map[i][0];
      for(int j=1;j<=n;j++)
      {
       cout<<" "<<map[i][j];
      }
       cout<<"\n";
      }
  }
}
