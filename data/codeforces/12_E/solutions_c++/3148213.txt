#include <iostream>

using namespace std;

int n;

int main()
{
	cin >> n;
	for (int i = 0 ; i < n ; i++, cout << endl)
		for (int j = 0 ; j < n ; j++, cout << " ")
			if (i == j)
				cout << 0;
			else if (i < n-1 && j < n-1)
				cout << (i+j) % (n-1) + 1;
			else
				cout << (min(i, j)*2) % (n-1) + 1;
	return 0;
}
