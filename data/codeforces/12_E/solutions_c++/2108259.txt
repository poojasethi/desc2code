#include<cstdio>
int n;
int main() {
	scanf( "%d" , &n );
	for ( int i = 1 ; i <= n - 1 ; i++ ) {
		for ( int j = 1 ; j <= n - 1 ; j++ )
			if ( i == j ) printf( "0 " );
				else printf( "%d " , 1 + ( i + j - 2 ) % ( n - 1 ) );
		printf( "%d\n" , ( 2 * i - 2 ) % ( n - 1 ) + 1 );
	}
	for ( int i = 1 ; i <= n - 1 ; i++ )
		printf( "%d " , ( 2 * i - 2 ) % ( n - 1 ) + 1 );
	printf( "0\n" );
	return 0;
}
