#include <stdio.h>

int n , mat[1010][1010];

int main()
{
	scanf( "%d" , &n );
	int a[1010] , t;
	for( int i=0 ; i<n ; i++)
		a[i] = i;
	for( int i=1 ; i<n ; i++ )
	{
		for( int j=0 ; j<n ; j++ )
			mat[a[j]][a[n-1-j]] = mat[a[n-1-j]][a[j]] = i;
		t = a[1];
		for( int i=1 ; i<n-1 ; i++ )
			a[i] = a[i+1];
		a[n-1] = t;
	}
	for( int i=0 ; i<n ; i++ )
	{
		printf( "%d" , mat[i][0] );
		for( int j=1 ; j<n ; j++ )
			printf( " %d" , mat[i][j] );
		printf("\n");
	}
	return 0;
}
