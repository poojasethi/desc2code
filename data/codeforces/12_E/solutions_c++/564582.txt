#include <stdio.h>

int main(void) {
	int n, i, j;
	scanf("%d", &n);
	for(i=0;i<n;i++,puts(""))
		for(j=0;j<n;j++) {
			if(i==j) printf("0");
			else {
				if(i<n-1 && j<n-1)
					printf("%d", (i+j)%(n-1)+1);
				else
					printf("%d", i<j? (2*i)%(n-1)+1: (2*j)%(n-1)+1);
			}
			if(j<n-1) printf(" ");
		}
	return 0;
}
