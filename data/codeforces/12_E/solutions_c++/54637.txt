#include <cstdio>
#define Nmax 1005
using namespace std;

int n;
int M[Nmax][Nmax];

int main()
{
  int i, j;
  //freopen ("fis.in", "r", stdin);
  scanf ("%d\n", &n);
  for (i=0; i<n-1; i++)
    for (j=0; j<n-1; j++)
      M[i][j]=1+(i+j)%(n-1);
  for (i=0; i<n-1; M[i][i]=0, i++)
    M[i][n-1]=M[n-1][i]=M[i][i];
  for (i=0; i<n; i++)
  {
    for (j=0; j<n; j++)
      printf ("%d ", M[i][j]);
    printf ("\n");
  }
  return 0;
}
