#include<cstdio>
#include<iostream>
using namespace std;

int n;
int a[1010][1010];

int main()
{int i,j,k;
 
 scanf("%d",&n);
 
 for(i=1;i<=n-1;i++)
   for(j=1;j<=n-1;j++)
     a[i][j]=(i+j-2)%(n-1)+1; 
 
 for(i=1;i<=n;i++)
 {a[n][i]=a[i][i];
  a[i][n]=a[i][i];
 }
 
 for(i=1;i<=n;i++)
   a[i][i]=0;      
 
 for(i=1;i<=n;i++)
 {for(j=1;j<=n;j++)
  {printf("%d",a[i][j]);  
   if(j!=n) printf(" ");
  }
  printf("\n");
 }
 
 return 0;
}
