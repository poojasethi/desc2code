#include <stdio.h>
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int mat[1000][1000];

int main() {
    int n;
    scanf("%d", &n);
    rep(i, n-1) rep(j, n-1) mat[i][j]=(i+j)%(n-1)+1;
    rep(i, n-1) { mat[i][n-1]=mat[n-1][i]=mat[i][i]; mat[i][i]=0; }
    rep(i, n) rep(j, n) printf("%d%c", mat[i][j], j<n-1 ? ' ' : '\n');
    return 0;
}
