#include <iostream>
using namespace std;
int i;
int main(){
    int n;
    cin >> n;
    for(; i<n ; i++){
        for(int j=0 ; j<n ; j++){
            if(i==j){
                cout << "0" << ' ';
            }else if(i == n-1 || j == n-1){
                cout << (1 + ((2*(i+j))%(n-1))) << ' ';
            } else {
                cout << (1 + (i+j)%(n-1)) << ' ';
            }
        }
        cout << '\n';
    }
    return 0;
}
