#include <cstdio>

int a[1000][1000];

int main()
{
	int n,k;
	scanf("%d",&n);
	for(int j=0;j<n-1;j++) {
		k=j;
		for(int i=0;i<j;i++) {
			a[j][i]=a[i][j]=k++;
			if(k==n) k=1;
		}
		a[n-1][j]=a[j][n-1]=k;
	}
	a[n-1][0]=a[0][n-1]=n-1;
	for(int i=0;i<n;i++) {
		for(int j=0;j<n;j++)
			printf("%d ",a[i][j]);
		printf("\n");
	}
	return 0;
}
