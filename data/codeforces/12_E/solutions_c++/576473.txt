#include <cstdio>
#include <algorithm>

using namespace std;

int M[1000][1010];

int main() {

    int N;
    scanf("%d", &N);
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            if (i == j) M[i][j] = 0;
            else if (i==N-1 || j==N-1) M[i][j] = (2*i+2*j) % (N-1) + 1;
            else M[i][j] = (i+j) % (N-1) + 1;
            printf("%d ", M[i][j]);
        }
        putchar('\n');
    }


    return 0;
}
