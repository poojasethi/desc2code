#include <iostream>

using namespace std;
int n;
int m[1001][1001];
int main()
{
    cin >> n;
    for(int i=0;i<n;++i)
        for(int j=0;j<n-1;j++)
            m[i][j]=1+(i+j)%(n-1);
    for (int i=0;i<n-1;i++)
        m[i][n-1]=m[n-1][i]= m[i][i];
    for (int i=0;i<n;i++) m[i][i]=0;

    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++) cout<<m[i][j]<<" ";
        cout<<endl;
    }
    return 0;
}
