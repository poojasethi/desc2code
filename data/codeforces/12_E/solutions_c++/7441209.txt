#include <cstdio>
using namespace std;
int mat[1003][1003];
int main()
{
	int n,i,j;
	scanf("%d",&n),n--;
	for (i=0;i<n;i++)
		for (j=0;j<n;j++)
			mat[i][j]=1+(i+j)%n;
	for (i=0;i<n;i++)
		mat[i][n]=mat[n][i]=mat[i][i],mat[i][i]=0;
	for (i=0;i<=n;i++)
	{
		for (j=0;j<n;j++)
			printf("%d ",mat[i][j]);
		printf("%d\n",mat[i][n]);
	}
	return 0;
}
