#include <bits/stdc++.h>

using namespace std;

int n;
int a[2005][2005];

int main() {
    //freopen("K.inp","rt",stdin);
    cin >> n;
    for (int i=0;i<n-1;i++)
        for (int j=0;j<n;j++) a[j][i]=a[i][j]=1+(i+j)%(n-1);
    for (int i=0;i<n;i++) a[n-1][i]=a[i][n-1]=a[i][i];
    for (int i=0;i<n;i++) {
        a[i][i]=0;
        for (int j=0;j<n;j++) cout << a[i][j] << " ";
        cout << endl;
    }
    return 0;
}
