#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
#include<set>
#include<map>
#include<vector>
using namespace std;

const int maxn = 100000 + 10;
int h[maxn];
int n, k, L;
multiset<int> f;
vector<int> ans;

int main() {
    cin >> n >> k;
    L = -1;
    int t = 1;
    for (int i = 1; i <= n; i++) {
        cin >> h[i];
        f.insert(h[i]);
        while (*f.rbegin() - *f.begin() > k)
            f.erase(f.find(h[t++]));
        if (i - t > L) {
            L = i - t;
            ans.clear();
        }
        if(i - t == L)
            ans.push_back(t);
    }
    cout<<L + 1<<' '<<ans.size()<<endl;
    for(int i = 0; i < ans.size(); i++) cout<<ans[i]<<' '<<ans[i] + L<<endl;
    return 0;
}
