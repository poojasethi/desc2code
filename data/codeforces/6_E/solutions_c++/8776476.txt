#include<cstdio>
#include<iostream>
#include<set>
using namespace std;
const int N = 1e5 + 10;
multiset<int> s;
int n, k, h[N], r[N], mx, cnt, t;
int main () {
    cin>>n>>k;
    for (int i = 0; i < n; i++) {
        scanf("%d", h + i);
        s.insert(h[i]);
        while(*s.rbegin() - *s.begin() > k) s.erase(s.find(h[t++]));
        int len = i - t + 1;
        if (len > mx) {
            mx = len, cnt = 0;
            r[cnt++] = t;
        }
        else if (len == mx) r[cnt++] = t;
    }
    cout<<mx<<" "<<cnt<<endl;
    for (int i = 0; i < cnt; i++) printf("%d %d\n", r[i] + 1, r[i] + mx);
    return 0;
}
