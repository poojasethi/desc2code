#include <cstdio>
#include <iostream>
#include <set>
#include <vector>

#define pb push_back

using namespace std;

const int MAXN=100100;

int N,K,res;
int ar[MAXN];

vector<int> wr;
multiset<int> s;

int main(){
	
	scanf(" %d %d",&N,&K);
	
	
	for( int beg=0,end=0 ; end<N ; end++ ){
		
		scanf(" %d",ar+end);
		s.insert(ar[end]);
		
		while( *s.rbegin()-*s.begin()>K )
			s.erase(s.find(ar[beg++]));
		
		if( end-beg+1==res )
			wr.pb(beg);
		else if( end-beg+1>res ){
			res=end-beg+1;
			wr.clear();
			wr.pb(beg);
		}
	}
	
	printf("%d %d\n",res,(int)wr.size());
	for( vector<int>::iterator it=wr.begin() ; it!=wr.end() ; it++ )
		printf("%d %d\n",*it+1,*it+res);
	
	return 0;
}
