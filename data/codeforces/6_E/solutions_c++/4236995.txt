#include <stdio.h>
#include <set>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)
int n, k, h[100000];
int a, b, r[100000];
int main() {
    scanf("%d%d", &n, &k);
    rep(i, n) scanf("%d", h+i);//time
    multiset<int> s;
    int t=0;
    rep(i, n) {
        s.insert(h[i]);
        while((*s.rbegin()-*s.begin())>k) s.erase(s.find(h[t++]));//由前往後刪
        if(i-t+1>a) { a=i-t+1; r[0]=t; b=1; }//i-t+1 = time period//reset if we find a larger length
        else if(i-t+1==a) { r[b++]=t; }
    }
    printf("%d %d\n", a, b);
    rep(i, b) printf("%d %d\n", r[i]+1, r[i]+a);//all time period of length a and the relevant start position
    return 0;
}
