#include <cstdio>
#include <set>
#include <iostream>

#define for1(i, n) for(int i=1;i<=n;++i)

int n,k,h[100001];
int a,b,r[100001];

using namespace std;

int main() 
{
    ios_base::sync_with_stdio(0);
    cin>>n>>k;
    for1(i,n) cin>>h[i];
    multiset<int> s;
    int t=1;
    for1(i, n)
    {
        s.insert(h[i]);
        while((*s.rbegin()-*s.begin())>k) 
            s.erase(s.find(h[t++]));
        if(i-t+1>a)
        {
            a=i-t+1;
            r[1]=t;
            b=2;
        }
        else if(i-t+1==a) 
        {
            r[b++]=t;
        }
    }
    cout<<a<<" "<<b-1<<endl;
    for1(i,b-1) cout<<r[i]<<" "<<r[i]+a-1<<endl;
    return 0;
}
