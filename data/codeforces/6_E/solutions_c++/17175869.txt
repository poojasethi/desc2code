#include<bits/stdc++.h>
using namespace std;
const int maxn = 1000010;
multiset<int> S;
int num[maxn],tt[maxn];
int n,k;
int main()
{
	    scanf("%d%d",&n,&k);
		for(int i=0;i<n;i++)
			scanf("%d",&num[i]);
		int cur=0;
		int ans=0,c=0;
		for(int i=0;i<n;i++)
		{
			S.insert(num[i]);
			while((*S.rbegin()-*S.begin())>k)
				S.erase(S.find(num[cur++]));
			if(S.size()>ans)
			{
				ans=S.size();
				tt[0]=cur;
				c=1;
			}
			else if(S.size()==ans)
				tt[c++]=cur;
		}
		printf("%d %d\n",ans,c);
		for(int i=0;i<c;i++)
			printf("%d %d\n",tt[i]+1,tt[i]+ans);
	return 0;
}
