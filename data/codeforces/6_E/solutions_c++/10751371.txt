#include<iostream>
#include<set>
#include<vector>
using namespace std;
typedef pair<int,int>P;
multiset<int>set1;
vector<P>set2;
int num[100005];
int n,k,go,ans;
int main()
{
  while(cin>>n>>k)
  {
    set1.clear();set2.clear();go=1;ans=-1;
    for(int i=1;i<=n;i++)
    {
      cin>>num[i];
      set1.insert(num[i]);
      while(*set1.rbegin()-*set1.begin()>k)set1.erase(set1.find(num[go++]));
      if(i-go+1>ans)ans=i-go+1,set2.clear();
      if(ans==i-go+1)set2.push_back(P(go,i));
    }
    cout<<ans<<" "<<set2.size()<<"\n";
    for(int i=0;i<set2.size();i++)
    cout<<set2[i].first<<" "<<set2[i].second<<"\n";
  }
}
