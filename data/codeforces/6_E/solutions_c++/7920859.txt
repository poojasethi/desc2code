#include <iostream>
#include <set>
#include <vector>

int main()
{
	int n, k, i, j;
	int mCount(0), *height;
	std::multiset<int> mst;
	std::vector<int> ans;

	std::cin >> n >> k;
	height = new int[n];
	for (i = j = 0; i < n; ++i)
	{
		std::cin >> height[i];
		mst.insert(height[i]);
		while (*mst.rbegin() - *mst.begin() > k)
		{
			mst.erase(mst.find(height[j++]));
		}
		if (mst.size() >= mCount)
		{
			if (mst.size() != mCount)
			{
				ans.clear();
				mCount = mst.size();
			}
			ans.push_back(j + 1);
		}
	}
	std::cout << mCount << ' ' << ans.size() << '\n';
	for (i = 0, j = ans.size(); i < j; ++i)
	{
		std::cout << ans[i] << ' ' << ans[i] + mCount - 1 << '\n';
	}

	delete [] height;
	return 0;
}
