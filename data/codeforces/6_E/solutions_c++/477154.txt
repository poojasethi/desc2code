#include <cstdio>
#include <set>
#include <cmath>
#include <vector>
#include <algorithm>

using namespace std;

int N, K, H[100100];

int main() {

    scanf("%d %d", &N, &K);
    for (int i = 0; i < N; i++) scanf("%d", H+i);
    multiset<int> S;
    int A = 0, b = 0;
    vector<int> SOL;
    for (int i = 0; i < N; i++) {
        S.insert(H[i]);
        while (*S.rbegin() - *S.begin() > K) S.erase(S.find(H[b++]));
        if (i-b+1 > A) { A = i-b+1; SOL.clear(); SOL.push_back(b); }
        else if (i-b+1 == A) SOL.push_back(b);
    }
    printf("%d %d\n", A, SOL.size());
    for (int i = 0; i < (int) SOL.size(); i++) printf("%d %d\n", SOL[i]+1, SOL[i]+A);

    return 0;
}
