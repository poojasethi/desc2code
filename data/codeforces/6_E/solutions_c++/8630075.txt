#include <iostream>
#include <stdio.h>
#include <set>
using namespace std;
int a[10000000],c[100000];
int main()
{
    int n,k,i,b=0,max=0,p=0;
    scanf("%d%d",&n,&k);
    multiset<int> z;
    for(i=0; i<n; i++)
    {
        scanf("%d",&a[i]);
        z.insert(a[i]);
        while(*z.rbegin()-*z.begin()>k)z.erase(z.find(a[b++]));
        if(i-b+1>max)
        {
            max=i-b+1,p=0;c[p++]=b;
        }
        else if(i-b+1==max)
        {
            c[p++]=b;
        }
    }
    cout<<max<<" "<<p<<endl;
    for(i=0; i<p; i++)
        printf("%d %d\n",c[i]+1,c[i]+max);
}
