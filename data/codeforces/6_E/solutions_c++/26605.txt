#include <cstdio>
#include <set>
#include <vector>

using namespace std;

int main() {
  int N, K; scanf("%d %d", &N, &K);
  int hs[100000];
  for (int i = 0; i < N; i++)
    scanf("%d", &hs[i]);

  multiset<int> hNow;
  vector<int> starts;
  starts.clear();
  int best = 0;
  int st = 0;
  for (int end = 0; end < N; end++) {
    hNow.insert(hs[end]);
    while (*hNow.rbegin() - *hNow.begin() > K) {
      multiset<int>::iterator i = hNow.find(hs[st]);
      hNow.erase(i);
      st++;
    }
    if (end - st + 1 == best) {
      starts.push_back(st);
    } else if (end - st + 1 > best) {
      best = end-st+1;
      starts.clear();
      starts.push_back(st);
    }
  }
  printf("%d %u\n", best, starts.size());
  for (vector<int>::iterator i = starts.begin(); i != starts.end(); i++)
    printf("%d %d\n", *i + 1, *i + best);
  return 0;
}
