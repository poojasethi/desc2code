#include<cstdio>
#include<set>
int a[100100],c[100100],n,k,b,m,p;
std::multiset<int> s;
int main()
{
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++)
    {
        scanf("%d",a+i),s.insert(a[i]);
        while(*s.rbegin()-*s.begin()>k)s.erase(s.find(a[b++]));
        if(i-b+1==m)c[p++]=b;else
        if(i-b+1>m)m=i-b+1,c[0]=b,p=1;
    }
    printf("%d %d\n",m,p);
    for (int i=0;i<p;i++)printf("%d %d\n",c[i]+1,c[i]+m);
    return 0;
}
