#include<iostream>
#include<set>
#include<vector>
using namespace std;
const int mm=1e5+10;
typedef pair<int,int>II;
multiset<int>mset;
vector<II>mp;
int shu[mm];
int n,k,go,ans;
int main()
{
  while(cin>>n>>k)
  {
    mset.clear();mp.clear();go=1;ans=-1;
    for(int i=1;i<=n;i++)
    {
      cin>>shu[i];
      mset.insert(shu[i]);
      while(*mset.rbegin()-*mset.begin()>k)mset.erase(mset.find(shu[go++]));
      if(i-go+1>ans)ans=i-go+1,mp.clear();
      if(ans==i-go+1)mp.push_back(II(go,i));
    }
    cout<<ans<<" "<<mp.size()<<"\n";
    for(int i=0;i<mp.size();i++)
    cout<<mp[i].first<<" "<<mp[i].second<<"\n";
  }
}
