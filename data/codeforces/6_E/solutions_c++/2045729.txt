#include<iostream>
#include<set>
#include<vector>
using namespace std;
typedef pair<int, int> PII;
multiset<int> mmx;
vector< PII > inv;
int ans, a[100001];
int main()
{
  int n, k;
  cin>>n>>k;
  for (int i = 0, j = 0, li = 0; j < n; j++) {
    cin>>a[j];
    mmx.insert(a[j]);
    li = i;
    while (*mmx.rbegin() - *mmx.begin() > k) mmx.erase(mmx.find(a[i++]));
    if (j - i + 1 > ans) {
      ans = j - i + 1;
      inv.clear();
    }
    if (j - i + 1 == ans) inv.push_back(PII(i,j));
  }
  cout<<ans<<" "<<inv.size()<<endl;
  for (int i = 0; i < inv.size(); i++) cout<<inv[i].first+1<<" "<<inv[i].second+1<<endl;
  return 0;
}
