#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 10;

typedef long long LL;

set<pair<int, int> > S;
int a[N], b[N];

int main() {
    int n, k, len = 0, cnt = 0, L = 0;
    scanf("%d %d", &n, &k);
    for (int i = 0; i < n; ++i) {
        scanf("%d", a + i);
        S.insert(make_pair(a[i], i));
        while (S.size() > 0 && S.rbegin()->first - S.begin()->first > k) {
            S.erase(make_pair(a[L], L));
            ++L;
        }
        if (S.size() > len) {
            len = S.size();
            cnt = 0;
            b[cnt++] = i;
        } else if (S.size() == len) {
            b[cnt++] = i;
        }
    }
    printf("%d %d\n", len, cnt);
    for (int i = 0; i < cnt; ++i) {
        printf("%d %d\n", b[i] - len + 2, b[i] + 1);
    }
    return 0;
}
