#include<iostream>
#include<cstdio>
#include<set>
using namespace std;

int n,K,a[100010],ans,ANSL[100010],ANSR[100010],ANS_;

struct INT
{
	int b,c;
};

bool operator <(INT A,INT B)
{
	if(A.c==B.c)return A.b<B.b;
	return A.c<B.c;
}

set<INT> Gy;

int main()
{
	//freopen("yoshi.in","rb",stdin);
	//freopen("yoshi.out","wb",stdout);
	scanf("%d%d",&n,&K);
	for(int i=1;i<=n;++i)scanf("%d",a+i);
	Gy.insert((INT){1,a[1]});
	for(int i=1,j=1;i<=n;)
	{
		for(;(Gy.rbegin())->c-(Gy.begin())->c>K&&j<i;)Gy.erase((INT){j,a[j]}),j++;
		if(i-j+1>ans)ans=i-j+1,ANS_=0;
		if(i-j+1==ans)
		{
			ANSL[++ANS_]=j;
			ANSR[ANS_]=i;
		}
		Gy.insert((INT){++i,a[i]});		
	}
	cout<<ans<<' '<<ANS_<<endl;
	//if(n==8992)cout<<a[436]<<' '<<a[437]<<' '<<a[438]<<endl;
	for(int i=1;i<=ANS_;++i)cout<<ANSL[i]<<' '<<ANSR[i]<<endl;
	return 0;
}
