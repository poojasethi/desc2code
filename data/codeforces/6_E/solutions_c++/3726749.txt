#include <iostream>
#include <set>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int n, val, k;

    cin >> n >> k;

    vector<int> v(n);

    for(int i = 0; i < n; i++) {
            cin >> v[i];
    }

    multiset<int> s;
    multiset<int> :: iterator it;

    int a = 0;
    int t = 0;
    vector<int> ans;

    for(int i = 0; i < n; i++) {
        s.insert(v[i]);
        while((*s.rbegin() - *s.begin()) > k) s.erase(s.find(v[t++]));
        if(a < i-t+1) {
            a = i-t+1;
            ans.clear();
            ans.push_back(t);
        }
        else if(a == i-t+1) {
            ans.push_back(t);
        }
    }
    cout << a << " " << ans.size() << endl;
    for(int i = 0; i < ans.size(); i++) {
        cout << ans[i]+1 << " " << ans[i]+a << endl;
    }




}
