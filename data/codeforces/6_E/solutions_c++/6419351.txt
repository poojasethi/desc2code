#include <bits/stdc++.h>
using namespace std;
int a[100005];
int main()
{
    int n,k,x,mx=0;
    vector < pair<int,int> > ans;
    multiset <int> s;
    cin >> n >> k;
    int idx = 0;
    for ( int i = 0; i < n; i++ ) {
        cin >> a[i];
        s.insert(a[i]);
        while ( *(s.rbegin()) - *(s.begin()) > k ) {
              s.erase(s.find(a[idx++]));
        }
        if ( i-idx+1 > mx ) {
             mx = i-idx+1;
             ans.clear();
             ans.push_back(make_pair(idx,i));
        }
        else if ( i-idx+1 == mx ) {
             ans.push_back(make_pair(idx,i));
        } 
        
    }
    cout << mx << " " << (int)(ans.size()) << endl;
    for ( int i = 0; i < (int)(ans.size()); i++ ) cout << ans[i].first+1 << " " << ans[i].second+1 << endl;
    return 0;
}
