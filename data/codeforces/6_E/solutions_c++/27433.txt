#include<cstdio>
#include<set>
#define N 1111111
#define pii pair<int,int>
using namespace std;
int a[N];
pii ans[N*2];
int main(){
    int n,k;
    scanf("%d%d",&n,&k);
    int anslen=0,i,j=0,longest=0;
    multiset<int>ms;
    for(i=0;i<n;++i){
        scanf("%d",a+i);
        ms.insert(a[i]);
        while(*ms.rbegin()-*ms.begin()>k)ms.erase(ms.find(a[j++]));
        if(longest<i-j){
            longest=i-j;
            anslen=0;
        }
        if(longest==i-j)ans[anslen++]=pii(j,i);
    }
    printf("%d %d\n",longest+1,anslen);
    for(i=0;i<anslen;++i)printf("%d %d\n",ans[i].first+1,ans[i].second+1);
    return 0;
}
