#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>
#include <vector>
#include <map>
#include <set>
#include <algorithm>
#include <sstream>

using namespace std;

int main() {
	int n, k;
	cin >> n >> k;
	int h[n];
	for(int i = 0; i < n; i++) cin >> h[i];
	multiset<int> s;
	vector<int> res;
	int c = 0, m = 0;
	for(int i = 0; i < n; i++) {
		s.insert(h[i]);
		while(*s.rbegin() - *s.begin() > k) s.erase(s.find(h[c++]));
		if(m < s.size()) {
			m = s.size();
			res.clear();
		}
		if(m == s.size()) {
			res.push_back(i);
		}
	}
	cout << m << " " << res.size() << endl;
	for(int i = 0; i < res.size(); i++) {
		cout << res[i] - m + 2 << " " << res[i] + 1 << endl;
	}
	return 0;
}
