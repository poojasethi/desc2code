#include <stdio.h>
#include <set>
#define forn(i, n) for(int i=0;i<(n);i++)
int n, k, h[100000];
int a, b, r[100000];
int main() {
    scanf("%d%d", &n, &k);
    forn(i, n) scanf("%d", h+i);
    std::multiset<int> s;
    int t=0;
    forn(i, n) {
        s.insert(h[i]);
        while((*s.rbegin()-*s.begin())>k) s.erase(s.find(h[t++]));
        if(i-t+1>a) { a=i-t+1; r[0]=t; b=1; }
        else if(i-t+1==a) { r[b++]=t; }
    }
    printf("%d %d\n", a, b);
    forn(i, b) printf("%d %d\n", r[i]+1, r[i]+a);
    return 0;
}
