#include<iostream>
#include<cstring>
#include<set>
#include<vector>
using namespace std;
const int mm=100005;
typedef pair<int,int>PII;
vector<PII>inv;
multiset<int>mmx;
int ans, a[100001];
int main()
{ int n,k;
  while(cin>>n>>k)
  {  ans=-88;
    mmx.clear();inv.clear();
     memset(a,0,sizeof(a));
    for(int i=0,j=0;j<n;j++)
    { cin>>a[j];
      mmx.insert(a[j]);
      while(*mmx.rbegin()-*mmx.begin()>k)
      mmx.erase(mmx.find(a[i++]));
      if(j-i+1>ans)
      {
        ans=j-i+1;inv.clear();
      }
      if(ans==j-i+1)inv.push_back(PII(i,j));
    }
    cout<<ans<<" "<<inv.size()<<"\n";
    for(int i=0;i<inv.size();i++)
    {
      cout<<inv[i].first+1<<" "<<inv[i].second+1<<"\n";
    }
  }
}
