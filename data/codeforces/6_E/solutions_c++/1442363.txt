
#include <iostream>
#include <cstdio>
#include <cstring>
#include <set>

using namespace std;

const int maxn = 1000001;
int n, k, seq[maxn], a, pos[maxn], b;
multiset<int> S;

int main() {
	scanf("%d%d", &n, &k);
	for (int i = 1; i <= n; ++i) {
		scanf("%d", &seq[i]);
	}
	for (int i = 1, j = 0; i <= n; ++i) {
		S.insert(seq[i]);
		while (*S.rbegin() - *S.begin() > k) {
			S.erase(S.find(seq[++j]));
		}
		if (i - j == a) {
			pos[++b] = j + 1;
		} else if (i - j > a) {
			a = i - j;
			b = 0;
			pos[++b] = j + 1;
		}
	}
	printf("%d %d\n", a, b);
	for (int i = 1; i <= b; ++i) {
		printf("%d %d\n", pos[i], pos[i] + a - 1);
	}
	return 0;
}
