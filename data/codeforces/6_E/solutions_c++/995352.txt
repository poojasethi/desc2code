#include<iostream>
#include<set>
using namespace std;
int a[100000];
int main() {
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++) {
		cin>>a[i];
	}
	int rt=0;
	multiset<int> s,res;
	int mx=0;
	for(int lt=0;lt<n;lt++) {
		while(rt<n && (s.empty() || max(*s.rbegin(),a[rt])-min(*s.begin(),a[rt])<=k)) {
			s.insert(a[rt++]);
		}
		if(rt-lt>mx) {
			res.clear();
		}
		if(rt-lt>=mx) {
			mx=rt-lt;
			res.insert(lt);
		}
		s.erase(s.find(a[lt]));
	}
	cout<<mx<<' '<<res.size()<<endl;
	for(__typeof(res.begin()) it=res.begin();it!=res.end();it++) {
		cout<<*it+1<<' '<<*it+mx<<endl;
	}
}
