#include <cstdio>
#include <set>
#include <algorithm>

using namespace std;
const int MAXN = 100005;

multiset< int > S;
int ans, out[MAXN], a[MAXN];
int n, k, l = 0, cnt = 0;

int main() {
    scanf("%d%d", &n, &k);
    for (int i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
        S.insert(a[i]);
        while (*S.rbegin() - *S.begin() > k) {
            S.erase(S.find(a[++l]));
        }
        if (ans < i - l) {
            ans = i - l;
            cnt = 0;
            out[++cnt] = l;
        }else if (ans == i - l) {
            out[++cnt] = l;
        }
    }
    printf("%d %d\n", ans, cnt);
    for (int i = 1; i <= cnt; i++) {
        printf("%d %d\n", out[i] + 1, out[i] + ans);
    }
    return 0;
}
