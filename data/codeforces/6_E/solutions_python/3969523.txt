from collections import *
def getints(): return map(int, raw_input().split())

n, m = getints()
a = getints()

f, g = deque(), deque()
t, j = -1, 0
for i, x in enumerate(a):
	while f and a[f[-1]] <= x: f.pop()
	f.append(i)
	while g and a[g[-1]] >= x: g.pop()
	g.append(i)
	while a[f[0]] - a[g[0]] > m:
		j = (f.popleft() if f[0] < g[0] else g.popleft()) + 1
	if t < i - j: t, p = i - j, [j + 1]
	elif t == i - j: p.append(j + 1)

print t + 1, len(p)
for i in p: print i, i + t
