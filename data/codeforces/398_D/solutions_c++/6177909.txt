#include <cstdio>
#include <cstring>
#include <set>
using namespace std;
int n,m,q;
set<int> f[50005];
int p[50005],s[50005],d[50005];
int tot=1;
int b[300005],e[300005],use[300005],h[50005];

void add(int x,int y){
    b[++tot]=y; e[tot]=h[x]; h[x]=tot;
}

int main(){
    scanf("%d%d%d",&n,&m,&q);
    int x,y;
    scanf("%d",&x);
    for (int i=0;i<x;i++){
        scanf("%d",&y);
        p[--y]=1;
    }
    for (int i=0;i<m;i++){
        scanf("%d%d\n",&x,&y);
        add(--x,--y); add(y,x);
        d[x]++; d[y]++;
    }
    for (int i=0;i<n;i++)
        for (int go=h[i];go;go=e[go])
            if (d[b[go]]>=d[i]&&use[go]==0){
                f[i].insert(b[go]),use[go]=use[go^1]=1;
                s[b[go]]+=p[i];
            }
    char ch;
    #define it set<int>::iterator 
    for (int i=0;i<q;i++){
        scanf("%c",&ch);
        if (ch=='O'){
            scanf("%d\n",&x);
            p[--x]=1;
            for (it j=f[x].begin();j!=f[x].end();j++)
                s[*j]++;
        } else if (ch=='F'){
            scanf("%d\n",&x);
            p[--x]=0;
            for (it j=f[x].begin();j!=f[x].end();j++)
                s[*j]--;
        } else if (ch=='A'){
            scanf("%d%d\n",&x,&y);
            --x; --y;
            if (d[x]>=d[y])
                f[y].insert(x),s[x]+=p[y];
                else f[x].insert(y),s[y]+=p[x];
        } else if (ch=='D'){
            scanf("%d%d\n",&x,&y);
            --x; --y;
            if (f[x].count(y)) f[x].erase(y),s[y]-=p[x];
            else if (f[y].count(x)) f[y].erase(x),s[x]-=p[y];
        } else {
            scanf("%d\n",&x);
            int y=s[--x];
            for (it j=f[x].begin();j!=f[x].end();j++)
                y+=p[*j];
            printf("%d\n",y);
        }
    }
    return 0;
}
