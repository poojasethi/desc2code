#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,x,tot,qh,qt,q[1000010],id[1000010],ok[1000010],cnt[1000010],next[1000010][26],f[1000010];
char ch[1000010],s[1000010];
int main(){
    scanf("%d%d",&m,&n);
    for(int i=1;i<=n;++i){
        scanf("%s",ch);
        int p=0,len=strlen(ch);
        for(int j=0;j<len;++j){
            if(!next[p][ch[j]-'a'])next[p][ch[j]-'a']=++tot;
            p=next[p][ch[j]-'a'];
        }
        cnt[p]=1;
        id[i]=p;
        ok[p]=1;
    }
    q[++qt]=0;
    while(qh<qt){
        int p=q[++qh];
        for(int i=0;i<26;++i)
            if(next[p][i]){
                if(p){
                    f[next[p][i]]=next[f[p]][i];
                    cnt[next[p][i]]+=cnt[f[next[p][i]]];
                }
                q[++qt]=next[p][i];
            }
            else if(p){
                next[p][i]=next[f[p]][i];
            }
    }
    for(int i=1;i<=m;++i){
        scanf("%s",ch);
        if(ch[0]=='?'){
            sscanf(ch+1,"%s",s);
            int p=0,ans=0,len=strlen(s);
            for(int j=0;j<len;++j){
                p=next[p][s[j]-'a'];
                int q=p;
                while(cnt[q]){
                    if(ok[q])++ans;
                    q=f[q];
                }
            }
            printf("%d\n",ans);
        }else
        if(ch[0]=='+'){
            sscanf(ch+1,"%d",&x);
            ok[id[x]]=1;
        }else{
            sscanf(ch+1,"%d",&x);
            ok[id[x]]=0;
        }
    }
    return 0;
}
