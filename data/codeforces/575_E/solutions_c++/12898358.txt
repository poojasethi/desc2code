using namespace std;

struct w{
    LL x,y;
};

LL cross(w& o, w& a, w& b){
    return (a.x-o.x)*(b.y-o.y)-(a.y-o.y)*(b.x-o.x);
}

LL dist(w& a,w& b){
    return (a.x-b.x)*(a.x-b.x)+(a.y-b.y)*(a.y-b.y);
}

double dd(double a){
    return sqrt(a);
}

bool f1(const w& a, const w& b){
    return a.y<b.y||(a.y==b.y&&a.x<b.x);
}

double r(double a,double b,double c){
    return a*b*c/sqrt((a+b+c)*(b+c-a)*(c+a-b)*(a+b-c));
}

const int m=100000;
int n,ta,tb,tc;
vector<w> e,ee;
int aa;
double bb,cc;

bool f2(w& a, w& b){
    LL c=cross(e[0],a,b);
    return c>0||(c==0&&dist(e[0],a)<dist(e[0],b));
}

            else e.pb({m,tb+tc-m+ta});
            if(tb-tc+m-ta>=0)e.pb({m,tb-tc+m-ta});

            else e.pb({ta+tc-tb,0});

reverse(e.begin(),e.end());
