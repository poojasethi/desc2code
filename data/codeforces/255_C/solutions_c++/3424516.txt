#include <iostream>
#include <algorithm>
using namespace std;

#define MAX 4002
int n;
int a[MAX];
int tab[MAX][MAX];
int main() {
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> a[i];
	}
	int sol = 0;
	for (int i = 1; i <= n; i++) {
		for (int j = 0, k = 0; j < i; j++) {
			tab[i][j] = tab[j][k] + 1;
			if (a[i] == a[j])
				k = j;
			sol = max(tab[i][j], sol);
		}
	}
	cout << sol << endl;
}
