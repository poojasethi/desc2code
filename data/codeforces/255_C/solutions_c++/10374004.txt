#include<bits/stdc++.h>
using namespace std;
int b[4005],dp[4005][4005],n;

int main(void){
	int i,j;
	while(~scanf("%d",&n)){
		for(i=1;i<=n;i++) scanf("%d",&b[i]);
		dp[0][0]=0;
		int ans=1,pos;
		for(i=1;i<=n;i++) dp[0][i]=1;
		for(i=1;i<=n;i++){
			for(j=1,pos=0;j<i;j++){
				dp[j][i]=dp[pos][j]+1;
				if(b[i]==b[j]) pos=j;
				ans=max(ans,dp[j][i]);
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
