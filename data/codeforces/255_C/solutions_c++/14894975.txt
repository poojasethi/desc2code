#include<cstdio>
#include<cmath>
#include<iostream>

using namespace std;


int n,a[4004],dp[4004][4004];
int main()
{
	while(scanf("%d",&n)==1)
	{
		int ans=-1;
		for(int i=1;i<=n;i++)
		{
			scanf("%d",&a[i]);
			int pre=0;
			for(int j=0;j<i;j++)
			{
				dp[i][j]=dp[j][pre]+1;
				if(a[i]==a[j]) pre=j;
				ans=max(ans,dp[i][j]);
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
