#include <iostream>
#include "bits/stdc++.h"
using namespace std;
int a[4005];
int dp[4005][4005];
int main() {
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
	  cin>>a[i];
	 int ans=0;
	 for(int i=1;i<=n;i++)
	 {
	 	for(int j=0,prv=0;j<i;j++)
	 	{
	 		dp[j][i]=1+dp[prv][j];
	 		if(a[i]==a[j])
	 		{
	 			prv=j;
	 		}
	 		ans=max(ans,dp[j][i]);
	 	}
	 	
	 }
	 
	 cout<<ans<<endl;
	return 0;
}
