#include<bits/stdc++.h>
using namespace std;
int x,arr[100001],cnt,dp[4001][4001],p;
int main(){
    cin>>x;
    for(int i=1;i<=x;i++){
        cin>>arr[i];
    }
    for(int i=1;i<=x;i++){
            p=0;
        for(int j=1;j<i;j++){
            dp[j][i]=dp[p][j]+1;
            if(arr[i]==arr[j])p=j;
            cnt=max(cnt,dp[j][i]);
        }
    }
    cout<<cnt+1<<endl;
}
