#include<bits/stdc++.h>
#define MAXN 4010
using namespace std;
int dp[MAXN][MAXN];
int main(){
    ios_base::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    int n,a[MAXN],ak,i,j,ans=-1;
    cin>>n;
    for(i=1;i<=n;++i)cin>>a[i];
    for(i=1;i<=n;++i)
    for(j=ak=0;j<i;++j){
        dp[i][j]=dp[j][ak]+1;
        if(a[j]==a[i])ak=j;
        if(dp[i][j]>ans)ans=dp[i][j];
    }
    cout<<ans;
}
