#include<cstdio>
#include<cmath>
#include<iostream>
#include<algorithm>
#include<cstdlib>

using namespace std;

#define N 55

typedef long long LL;

int n, get;
double dp[N][N], ans;
int L[N], R[N];

int main()
{
	double tmp, now, sum;
	scanf("%d", &n);
	for(int i = 1; i <= n; i++)
		scanf("%d %d", &L[i], &R[i]);
	for(int x = 1; x <= n; x++)
		for(int i = L[x]; i <= R[x]; i++)
			for(int y = 1; y <= n; y++)
			{
				if(x == y) continue;
				sum = i * 1.0 / (R[x] - L[x] + 1);
				for(int j = 1; j <= n; j++) 
				{
					if(j == x) continue;
					now = i - L[j] + (j > x);
					tmp = max(0.0, min(1.0, now * 1.0 / ( R[j] - L[j] + 1)));
					if(j == y) sum *= (1 - tmp);
					else sum *= tmp;
				}
				ans += sum;
			}
	printf("%.10lf\n", (double)ans);
	//system("pause");
	return 0;
}
