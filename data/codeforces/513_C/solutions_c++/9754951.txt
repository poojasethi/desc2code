#include <cstdio>
#include <algorithm>
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

int n;
int l[11];
int r[11];
int ll[11];
int rr[11];
double ans;

void Up(int k,int x)
{
	rr[k]=min(rr[k],x);
}

void Down(int k,int x)
{
	ll[k]=max(ll[k],x);
}

double Calc(int x,int y,int z)
{
	REP(i,n) {ll[i]=l[i];rr[i]=r[i];}
	Up(y,z);Down(y,z);
	if (x<y) Down(x,z); else Down(x,z+1);
	REP(i,n) if (i!=x && i!=y)
	{
		if (i<y) Up(i,z-1);
		else Up(i,z);
	}
	double ret=1.0;
	REP(i,n) ret*=(double)max(rr[i]-ll[i]+1,0)/(double)(r[i]-l[i]+1);
	return ret;
}

int main()
{
	scanf("%d",&n);
	REP(i,n) scanf("%d%d",&l[i],&r[i]);
	for (int i=1;i<=10000;++i)
		for (int j=1;j<=n;++j)
			for (int k=1;k<=n;++k) if (j!=k)
				ans+=(double)i*Calc(j,k,i);
	printf("%.20lf\n",ans);
	return 0;
}
