#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <string>
#include <vector>
#include <map>
using namespace std;
double dp[15][15],ans;
int L[15],R[15],n,i,j,k;
int main()
{
    scanf("%d",&n);
    for (i=1; i<=n; i++)
        scanf("%d%d",&L[i],&R[i]);
    for (i=1; i<=10000; i++)
    {
        for (j=0; j<=n; j++) for (k=0; k<=n; k++) dp[j][k]=0;
        dp[0][0]=1;
        for (j=0; j<n; j++)
          for (k=0; k<=j; k++)
          {
              if (R[j+1]>=i) dp[j+1][k+1]+=dp[j][k]*min(1.0,(R[j+1]-i+1)/(R[j+1]-L[j+1]+1.0));
              if (L[j+1]<i) dp[j+1][k]+=dp[j][k]*min(1.0,(i-L[j+1])/(R[j+1]-L[j+1]+1.0));
          }
        for (j=2; j<=n; j++) ans+=dp[n][j];
    }
    printf("%.10f\n",ans);
    return 0;
}
