#include<iostream>
#include<cstdio>
using namespace std;

int l[10],r[10];

int main()
{
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++)
		scanf("%d%d",&l[i],&r[i]);
	double ans=0.0;
	for(int price=1;price<=10000;price++)
	{
		for(int msk=1;msk<(1<<n);msk++)
		{
			int cnt=0;
			for(int i=0;i<n;i++)
				if (msk&(1<<i)) cnt++;
			if (cnt==1) continue;
			double p=1.0;
			for(int i=0;i<n;i++)
			{
				int a=l[i],b=r[i];
				if (msk&(1<<i))
					a=max(a,price);
				else
					b=min(b,price-1);
				if (a>b)
				{
					p=0;
					break;
				}
				p=p*(b-a+1)/(r[i]-l[i]+1);
			}
			ans+=p;
		}
	}
	printf("%.9f\n", ans);
	return 0;
}
