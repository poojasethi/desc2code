#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<iostream>
#include<map>
#include<queue>
using namespace std;

double p[11000];

int l[20],r[20];
double len[20];
int main()
{
	int n,i,j,k;

	cin>>n;
	for(i=1;i<=n;i++)
		cin>>l[i]>>r[i],len[i]=r[i]-l[i]+1;

	for(i=1;i<=10000;i++)
	{
		p[i]=1;
		
		double pp=1;
		for(j=1;j<=n;j++)
		{
			if(r[j]<i)
				;
			else if(l[j]>=i)
			{
				pp=0;
				break;
			}
			else
				pp*=(i-l[j])/len[j];
		}
		p[i]-=pp;

		double ppp=0;
		for(k=1;k<=n;k++)
		{
			pp=1;
			for(j=1;j<=n;j++)if(j-k)
		{
			if(r[j]<i)
				;
			else if(l[j]>=i)
			{
				pp=0;
				break;
			}
			else
				pp*=(i-l[j])/len[j];
		}
			if(r[k]<i)
				pp=0;
			else if(l[k]<i)
				pp*=(r[k]-i+1)/len[k];
			ppp+=pp;
		}
		p[i]-=ppp;
	}
	double ans=0;
	for(i=1;i<=10000;i++)
		ans+=i*(p[i]-p[i+1]);
	printf("%.12lf\n",ans);
}
