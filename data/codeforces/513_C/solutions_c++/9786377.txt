#include <iostream>
#include <iomanip>

using namespace std;

const int MAXN = 10, MAXA = 10010;

int ql[MAXN], qr[MAXN], len[MAXN], n;

double prVal(int x, int l, int r) {
	double ret = max((double) 0, (double)(min(r, qr[x]) - max(l, ql[x])) / len[x]);
	return ret;
}

double prob(int x) {
	double ans = 0.f;
	for (int i = 0; i < n; i++) {
		double cur = prVal(i, x + 1, MAXA);
		double curX = 1.0, curY = 1.0;
		for (int j = 0; j < n; j++)
			if (i != j)
				curX *= prVal(j, 0, x + 1), curY *= prVal(j, 0, x);
		ans += cur * (curX - curY);
	}
	for (int mask = 0; mask < (1 << n); mask++) {
		if (__builtin_popcount(mask) < 2)
				continue;
		double cur = 1.0;
		for (int i = 0; i < n; i++) {
			if (mask & (1 << i))
				cur *= prVal(i, x, x + 1);
			else
				cur *= prVal(i, 0, x);
		}
		ans += cur;
	}
	return ans;
}

int main() {
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> ql[i] >> qr[i], qr[i]++, len[i] = qr[i] - ql[i];
	double ans = 0;
	for (int i = 1; i < MAXA; i++)
		ans += i * prob(i);
	cout << setprecision(12) << fixed << ans << endl;
	return 0;
}
