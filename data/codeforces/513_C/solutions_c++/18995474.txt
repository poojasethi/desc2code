#include<cstdio>
int l[6],r[6],n,lim[6],b[6];
double g(int m,int k)
{
	int a[6]={0},i=0;
	while(m)
	{
		a[i++]=m%3;
		m/=3;
	}
	int cnt0=0,cnt1=0;
	for(i=0;i<n;i++)
	{
		if(a[i]==0)
			cnt0++;
		if(a[i]==1)
			cnt1++;
	}
	if(cnt0==0)
		return 0;
	if(cnt1>1)
		return 0;
	if(cnt1==0&&cnt0==1)
		return 0;
	for(i=0;i<n;i++)
	{
		if(a[i]==0)
		{
			if(k<l[i]||k>r[i])
				return 0;
			else
				b[i]=1;
		}
		else if(a[i]==1)
		{
			if(k>=r[i])
				return 0;
			else
			{
				if(k>=l[i])
					b[i]=r[i]-k;
				else
					b[i]=r[i]-l[i]+1;
			}
		}
		else
		{
			if(k<=l[i])
				return 0;
			else
			{
				if(k<=r[i])
					b[i]=k-l[i];
				else
					b[i]=r[i]-l[i]+1;
			}
		}
	}
	double res=1;
	for(i=0;i<n;i++)
		res*=b[i]*1.0/(r[i]-l[i]+1);
	return res;
}
double cal(int k)
{
	int i;
	double res=0;
	for(i=0;i<lim[n];i++)
		res+=g(i,k);
	return res;
}
int main()
{
	int i;
	lim[1]=3;
	for(i=2;i<=5;i++)
		lim[i]=lim[i-1]*3;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&l[i],&r[i]);
	}
	double ans=0;
	for(i=1;i<=10000;i++)
	{
		ans+=i*cal(i);
	}
	printf("%.10lf\n",ans);
	return 0;
}
