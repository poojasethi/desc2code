#include <set>

#include <cstring>

#include <iostream>
#include <algorithm>

#define MaxN 100010

using namespace std;

int N;
int x[MaxN], y[MaxN];
double Ans;

double Calc(int now, int a) {
    if(y[now] < a) return 1.0;
    if(a <= x[now]) return 0.0;
    return (double)(a - x[now]) / (double)(y[now] - x[now] + 1);
}

int main() {
    int i, j, k, l;
    scanf("%d", &N);
    for(i = 1; i <= N; ++i)
        scanf("%d%d", &x[i], &y[i]);
    for(i = 0; i < 10000; ++i) { // integral [i, i + 1]
        double x1, y1;
        double p1 = 1;
        Ans += 1.0;
        for(j = 1; j <= N; ++j) {
            p1 = (1.0 - Calc(j, i + 1));
            for(k = 1; k <= N; ++k)
                if(j != k)
                    p1 = p1 * Calc(k, i + 1);
            Ans -= p1;
        }
        p1 = 1;
        for(j = 1; j <= N; ++j)
            p1 *= Calc(j, i + 1);
        Ans -= p1;
    }
    printf("%.10lf\n", Ans);
    return 0;
}
