#include <bits/stdc++.h>
using namespace std;
int a[5], b[5];
int j, n;
double dp(int pos, int large, int same) {
  if (large == 2) return 0.0;
  if (pos == n) {
    if (large == 0 && same != 2) return 0.0;
    if (same == 0) return 0.0;
    return 1.0;
  }
  // split into three
  double x = max(0, b[pos] - max(a[pos], j + 1) + 1) * dp(pos + 1, large + 1, same);
  double y = (a[pos] <= j && b[pos] >= j) * dp(pos + 1, large, min(same + 1, 2));
  double z = max(0, min(b[pos], j - 1) - a[pos] + 1) * dp(pos + 1, large, same);
  assert(x + y + z <= b[pos] - a[pos] + 1);
  return (x + y + z) / (b[pos] - a[pos] + 1);
}
int main() {
  scanf("%d", &n);
  for (int i = 0; i < n; i++) {
    scanf("%d %d", &a[i], &b[i]);
  }
  double ans = 0.0;
  for (j = 1; j <= 10000; j++) {
      // j is the second price
      ans += j * dp(0, 0, 0);
  }
  printf("%.12lf\n", ans);
  return 0;
}
