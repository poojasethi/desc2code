#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <string>
#include <algorithm>
#include <cmath>
#include <vector>
using namespace std;
double f[50][50];
int l[50], r[50];
int n;
int main()
{
    while(scanf("%d", &n) != EOF)
    {
        double re = 0;
        for (int i = 1; i <= n; i++)
            scanf("%d%d", &l[i], &r[i]);
        for (int i = 1; i <= 10000; i++)
        {
            for (int j = 0; j <= n; j++)
            for (int k = 0; k <= n; k++)
                f[j][k] = 0;
            f[0][0] = 1;
            for (int j = 1; j <= n; j++)
            for (int k = 0; k <= j; k++)
            {
                if (r[j] >= i)
                        f[j][k + 1] += f[j - 1][k] * min(1.0, (r[j] - i + 1) * 1.0 / (r[j] - l[j] + 1));
                if (l[j] < i)
                        f[j][k] += f[j - 1][k] * min(1.0, (i - l[j]) * 1.0 / (r[j] - l[j] + 1));
                }
            for (int j = 2; j <= n; j++)
                re += f[n][j];
        }
        printf("%.14lf\n", re);
    }
    return 0;
}
