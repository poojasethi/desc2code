#include <bits/stdc++.h>
using namespace std;
int n,k,d[10];
double ans,p,x,y,z;
pair <int , int> a[10];
int main ()
{
cin>>n;
for(int i=0;i<n;i++)
{cin>>a[i].first>>a[i].second;
 d[i]=a[i].second-a[i].first+1;
}
ans=0;
for(int i=1;i<10001;i++)
{
  for(int s=0;s<n;s++)
    { if(a[s].first>i || a[s].second<i)continue;
      for(int f=0;f<n;f++)
       { if(s==f)continue;
         if(a[f].second<i)continue;
         else {p=i;p/=d[s];if(i>=a[f].first && f>s)p*=(a[f].second-i+1);
         else if(i>=a[f].first && s>f)p*=(a[f].second-i);
         else p*=d[f]; p/=d[f];
               }
         for(int r=0;r<n;r++)
            {
            if(r==s || r==f)continue;
            if(a[r].first>i){p=0;break;}
            if(i<=a[r].second && r<s)p*=(i-a[r].first+1);
            else if(i<=a[r].second && r>s)p*=(i-a[r].first);
            else p*=d[r];
            p/=d[r];
            }
            ans+=p;
            //cout<<p<<endl;
        }
    }
}
printf("%0.10lf",ans);
return 0;}
