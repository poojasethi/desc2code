#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define MAXN 55
using namespace std ;
typedef long long LL ;
typedef long double ld ;

LL N, M, get ;
ld dp[10][10], ans ;
LL L[MAXN], R[MAXN] ;
int main()
{
    LL i, j, t, x, y ;
    ld tmp, now, sum ;
    scanf("%I64d", &N) ;
    for(i = 1; i <= N; i ++)
        scanf("%I64d %I64d", &L[i], &R[i]) ;
    for(x = 1; x <= N; x ++)
        for(i = L[x]; i <= R[x]; i ++)
            for(y = 1; y <= N; y ++)
            {
                if(x == y) continue ;
                sum = i*1.0/(R[x]-L[x]+1) ;
                for(j = 1; j <= N; j ++) 
                {
                    if(j == x) continue ;
                    now = i-L[j]+(j>x) ;
                    tmp = max((ld)0.0, min((ld)1.0, now*1.0/(R[j]-L[j]+1))) ;
                    if(j == y) sum *= (1-tmp) ;
                    else sum *= tmp ;
                }
                ans += sum ;
            }
    printf("%.10lf\n", (double)ans) ;
    //system("pause") ;
    return 0 ;
}
