#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

const int N = 10000 + 1;
double n,p,pn,now,ans = 0,x[N + 5],y[N + 5];

double getP(int id, double a,double b){
    return max(double(0),(min(y[id],b) - max(x[id],a) + 1) / (y[id] - x[id] + 1));
}

int main(){
    cin>>n;
    for(int i = 0;i < n;i++)
	cin>>x[i]>>y[i];

    for(double b = 1;b <= N;b++){
        p = 0;
		now = 0;
		pn = 0;
        for(int i = 0;i < n;i++){
            if(b < x[i] || b > y[i])
			continue;
            p = getP(i,b,b);
            for(int j = 0;j < n;j++){
                if(j == i)
				continue;
                pn = p * getP(j,b + (j < i ? 0 : 1),N);
                for(int k = 0;k < n;k++){
                    if(k == i || k == j)
					continue;
                    pn *= getP(k,1,b - (k < i ? 1 : 0));
                }
                now += pn;
            }
        }
        ans += b * now;
    }

    printf("%.10lf\n",ans);

    return 0;
}
