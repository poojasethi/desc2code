#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
#include<iostream>
#include<string>
#include<map>
#include<set>
#include<algorithm>
#include<queue>
#include<vector>
#include<time.h>
using namespace std;
double dp[50][50],ans;
int a[50],b[50];
int main()
{
    int n;
    while(~scanf("%d",&n))
    {
        for (int i=1; i<=n; ++i)
            scanf("%d%d",&a[i],&b[i]);
        for (int i=1; i<=10000; ++i)
        {
            for (int j=0; j<=n; ++j)
                for (int k=0; k<=n; ++k)
                    dp[j][k]=0;
            dp[0][0]=1;
            for (int j=0; j<n; ++j)
                for (int k=0; k<=j; ++k)
                {
                    if (b[j+1]>=i)
                        dp[j+1][k+1]+=dp[j][k]*min(1.0,(b[j+1]-i+1)*1.0/(b[j+1]-a[j+1]+1));
                    if (a[j+1]<i)
                        dp[j+1][k]+=dp[j][k]*min(1.0,(i-a[j+1])*1.0/(b[j+1]-a[j+1]+1));
                }
            for (int j=2; j<=n; ++j)
                ans+=dp[n][j];
        }
        printf("%.12f\n",ans);
    }
    return 0;
}
