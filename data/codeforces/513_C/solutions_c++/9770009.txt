#include<stdio.h>
int main()
{
  int i,j,k,t,n,l[6],r[6];
  double sum=0.0,tmp;
  scanf("%d",&n);
  for (i=1;i<=n;i++) scanf("%d%d",&l[i],&r[i]);
  for (i=1;i<=10000;i++)
  {
	for (j=1;j<=n;j++)//max
	  for (k=1;k<=n;k++)//second
	  {
		if (j==k||r[j]<i||r[k]<i||l[k]>i) continue;
		if (l[j]>i) tmp=1.0;
		else if (k<j) tmp=1.0*(r[j]-i+1)/(r[j]-l[j]+1);
		else tmp=1.0*(r[j]-i)/(r[j]-l[j]+1);
		tmp=tmp/(r[k]-l[k]+1);
		for (t=1;t<=n;t++)
		  if (t!=j&&t!=k&&l[t]>i) tmp=0.0;
		  else if (t!=j&&t!=k&&r[t]>=i){
			if (t<k) tmp=tmp*(i-l[t]+1)/(r[t]-l[t]+1);
			else tmp=tmp*(i-l[t])/(r[t]-l[t]+1);
		  }
		sum+=i*tmp;
	  }
  }
  printf("%.15lf\n",sum);
}
