#include <cstdio>
#include <algorithm>

using namespace std;

int l[10];
int r[10];

int n;

int rng(int x1,int y1,int x2,int y2)
{
    int a=max(x1,x2);
    int b=min(y1,y2);
    if (a<=b)
        return b-a+1;
    return 0;
}

int main()
{
    scanf("%d",&n);
    for (int i=1; i<=n; i++)
        scanf("%d%d",&l[i],&r[i]);
    double ret=0.0;
    for (int fst=1; fst<=n; fst++)
        for (int nd=1; nd<=n; nd++)
        {
            if (fst==nd)
                continue;
            for (int cst=l[nd]; cst<=r[nd]; cst++)
            {
                double add=(double)cst;
                for (int i=1; i<=n; i++)
                    if (i==fst)
                        add*=(double)rng(l[i],r[i],cst+1-(fst<nd),100000);
                    else
                        if (i!=nd)
                            add*=(double)rng(l[i],r[i],0,cst-1+(nd<i));
                ret+=add;
                //printf("%d %d %d %lf\n",fst,nd,cst,add);
            }
        }
    for (int i=1; i<=n; i++)
        ret/=(double)(r[i]-l[i]+1);
    printf("%.10lf\n",ret);
}
