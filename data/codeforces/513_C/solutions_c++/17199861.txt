#include <cstdio>
inline unsigned int getuint(){
	char w=getchar();
	while (w<'0'||'9'<w) w=getchar();
	unsigned int ans=0;
	for (;'0'<=w&&w<='9';w=getchar()) ans=ans*10+w-'0';
	return ans;
}
int popcount[233];
int L[233], R[233];
double a[233];
int main(){
	//freopen("ex.in", "r", stdin);
	int n=getuint();
	for (int i=1;i<1<<n;i++) popcount[i]=popcount[i>>1]+(i&1);
	for (int i=0;i<n;i++) L[i]=getuint(), R[i]=getuint();
	double ans=0;
	for (int l=1;l<=10000;l++){
		for (int i=0;i<n;i++) a[i]=L[i]<=l&&l<=R[i]?1.0*(R[i]-l+1)/(R[i]-L[i]+1):l<L[i];
		for (int s=0;s<1<<n;s++) if (popcount[s]>=2){
			double t=1;
			for (int i=0;i<n;i++) if (s>>i&1) t*=a[i]; else t*=1-a[i];
			ans+=t;
		}
	}
	printf("%.9f\n", ans);
	return 0;
}
