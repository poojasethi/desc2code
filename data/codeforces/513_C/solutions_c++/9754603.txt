using namespace std;
int n;
int l[10],r[10];
double p(int peo,int mon)
{
	int i,j;
	double re=0;
	for (i=0;i<n;i++) if (i!=peo)
	{
		double pp;
		if (r[i]<mon) continue;
		if (l[i]>mon) pp=1;
		else pp=i<peo?(double)(r[i]-mon)/(r[i]-l[i]+1):(double)(r[i]-mon+1)/(r[i]-l[i]+1);
		for (j=0;j<n;j++) if (j!=i&&j!=peo)
		{
			if (l[j]>mon)
			{
				pp=0;
				break;
			}
			if (r[j]<mon) continue;
			if (j<peo) pp*=(double)(mon-l[j]+1)/(r[j]-l[j]+1);
			else pp*=(double)(mon-l[j])/(r[j]-l[j]+1);
		}
		re+=pp;
	}
	re/=(r[peo]-l[peo]+1);
	return re;
}
int main()
{
	scanf("%d",&n);
	int i,j;
	double ans=0;
	for (i=0;i<n;i++) scanf("%d%d",&l[i],&r[i]);
	for (i=0;i<n;i++) for (j=l[i];j<=r[i];j++) ans+=j*p(i,j);
	printf("%.10lf",ans);
	return 0;
}
