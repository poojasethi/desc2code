#define db double
#define inf 100000
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,l,r;
int L[10],R[10];
db s,an;
inline void into(int x,int y) {
	l=max(l,x); r=min(r,y);
}
int main() {
	scanf("%d",&n);
	s=1;
	For(i,1,n) scanf("%d%d",&L[i],&R[i]),s*=R[i]-L[i]+1;
	int er,yi,zhi;
	For(er,1,n)For(yi,1,n) if (er!=yi)
		For(zhi,L[er],R[er]) {
			db sum=1;
			For(i,1,n) {
				if (i==er) continue;
				l=L[i],r=R[i];
				if (i==yi) {
					if (yi<er) into(zhi,inf);
					else into(zhi+1,inf);
				} else {
					if (i<er) into(0,zhi-1);
					else into(0,zhi);
				}
				if (l>r) sum=0;
				else sum*=r-l+1;
			}
			an+=sum*zhi;
		}
	printf("%.12lf\n",an/s);
	return 0;
}
