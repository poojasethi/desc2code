#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define LiangJiaJun main
using namespace std;
int l[14],r[14];
int n;double ans=0.0;bool s=true; 
double f[14][14];
int LiangJiaJun(){
    cin>>n;
    for(int i=1;i<=n;i++){
        scanf("%d%d",&l[i],&r[i]);    
    }
    for (int i = 1; i <= 10000; i++){
            for (int j = 0; j <= n; j++)
            for (int k = 0; k <= n; k++)f[j][k]=0.0;
            f[0][0] = 1.0;
            for (int j=1;j<= n;j++)
            for (int k = 0; k <= j;k++){
                if (r[j] >= i)
                        f[j][k + 1] += f[j - 1][k] * min(1.0, (r[j] - i + 1) * 1.0 / (r[j] - l[j] + 1));
                if (l[j] < i)
                        f[j][k] += f[j - 1][k] * min(1.0, (i - l[j]) * 1.0 / (r[j] - l[j] + 1));
            }
            for (int j=2;j<=n;j++)ans+=f[n][j];
        }
    printf("%.10lf\n",ans);
    //system("pause");
    return 0;
}
