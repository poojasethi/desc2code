#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
#include <cmath>
using namespace std;
int c,a[5],s[5];
double d[5];
double l(int e,int r){
  return ((double)(max(min(s[c],r)-max(e,a[c]),-1)+1))*d[c];
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,v;
  double f[5],g[5],y=0;
  cin>>q;
  for(w=0;w<q;w++){
   cin>>a[w]>>s[w];
   d[w]=1/((double)(s[w]-a[w]+1));}
  for(w=1;w<10001;w++){
   f[0]=1;
   f[1]=f[2]=f[3]=f[4]=0;
   for(c=0;c<q;c++){
    g[0]=f[0]*l(0,w-1);
    g[1]=f[0]*l(w,w);
    g[2]=f[0]*l(w+1,10001);
    g[1]+=f[1]*l(0,w-1);
    g[3]=f[1]*l(w+1,10001);
    g[4]=f[1]*l(w,w);
    g[2]+=f[2]*l(0,w-1);
    g[3]+=f[2]*l(w,w);
    g[3]+=f[3]*l(0,w);
    g[4]+=f[4]*l(0,w);
    g[3]+=f[4]*l(w+1,10001);
    for(v=0;v<5;v++)
     f[v]=g[v];}
   y+=((double)w)*(f[3]+f[4]);}
  printf("%.10f",y);
  return 0;}
