#include <bits/stdc++.h>

using namespace std;

int n;
int l[5], r[5];
double ans = 0;

double prob (int v) {
  double p[5] = {0, 0, 0, 0, 0};
  for (int i = 0; i < n; i++) {
    if (v <= l[i]) {
      p[i] = 1;
    }
    else if (v > r[i]) {
      p[i] = 0;
    }
    else {
      p[i] = 1.0 * (r[i] - v + 1) / (r[i] - l[i] + 1);
    }
  }
  double g = 0;
  for (int i = 0; i < (1 << n); i++) {
    if (__builtin_popcount(i) >= 2) {
      double b = 1;
      for (int j = 0; j < n; j++) {
        if ((1 << j) & i)b *= p[j];
        else b *= 1 - p[j];
      }
      g += b;
    }
  }
  return g;
}

int main() {
  scanf ("%d", &n);
  for (int i = 0; i < n; i++) {
    scanf ("%d %d", &l[i], &r[i]);
  }
  for (int i = 1; i <= 10000; i++) {
    ans += prob (i);
  }
  printf ("%.20f\n", ans);
}
