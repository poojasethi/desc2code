#include<iostream>
#include<iomanip>
#include<algorithm>

#define int long long
using namespace std;

int s[10], l[10], r[10];
double pro = 1;
int g[10];
double t;
double ans;
int n;

bool check()
{
	int t = 0, t2 = 0;
	for(int i = 0; i < n; i++)
		if(g[i] == 1)
			t++;
		else if(g[i] == 2)
			t2++;
	if((t > 1 && t2 <= 1) || (t == 1 && t2 == 1))
		return true;
	return false;
}


void BT(int n, int i)
{
	if(n < 0)
	{
		if(check())
			ans += t * i;
		return;
	}
	if(r[n] > i)
	{
		g[n] = 2;
		t *= (r[n] - max(i + 1, l[n]) + 1);
		BT(n - 1, i);
		t /= (r[n] - max(i + 1, l[n]) + 1);
	}
	if(l[n] <= i && i <= r[n])
	{
		g[n] = 1;
		BT(n - 1, i);
	}
	if(l[n] < i)
	{
		t *= (min(r[n], i - 1) - l[n] + 1);
		g[n] = 0;
		BT(n - 1, i);
		t /= (min(r[n], i - 1) - l[n] + 1);
	}
}

main()
{
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		cin >> l[i] >> r[i];
		s[i] = r[i] - l[i] + 1;
		pro *= s[i];
	}
	for(int i = 1; i <= 10000; i++)
	{
		t = 1;
		BT(n - 1, i);
	}
	cout << setprecision(9) << fixed << double(ans) / pro << endl;
	return 0;
}
