using namespace std;

char s[1100000];
int d[1100000];
vector<pair<string, int> > a;
vector<string> aa;

int main()
{
	string ss;
	int i,j,k,l,now,ans,p,m;
	scanf("%s",s);
	l=strlen(s);
	a.clear();
	for (i=0;i<l;i++)
		if (s[i]=='<')
		{
			j=i+1;
			while (s[j]!='>') j++;
			if (s[i+1]=='/')
			{
				s[j]=0;
				ss=s+i+2;
				a.push_back(make_pair(ss,-1));
			}
			else if (s[j-1]=='/')
			{
				s[j-1]=0;
				ss=s+i+1;
				a.push_back(make_pair(ss,1));
				a.push_back(make_pair(ss,-1));
			}
			else
			{
				s[j]=0;
				ss=s+i+1;
				a.push_back(make_pair(ss,1));
			}
			i=j;
		}
	scanf("%d",&m);
	gets(s);
	for (i=0;i<m;i++)
	{
		gets(s);
		ss=s;
		ss+=" ";
		aa.clear();
		while (ss!="")
		{
			p=ss.find_first_of(' ');
			aa.push_back(ss.substr(0,p));
			ss.erase(0,p+1);
		}
		d[0]=0;
		now=1;
		ans=0;
		for (j=0;j<a.size();j++)
		{
			if (a[j].second==1)
			{
				d[now]=d[now-1];
				if (a[j].first==aa[d[now]])
					d[now]++;
				if (d[now]==aa.size())
				{
					ans++;
					d[now]--;
				}
				now++;
			}
			else
			{
				now--;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
