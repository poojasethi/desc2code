def main():
    n, m, s, f = map(int, raw_input().split())
    d = dict()
    for _ in xrange(m):
        t, l, r = map(int, raw_input().split())
        d[t] = [l, r]
    ans = []
    t = 1
    mv = 'L' if s > f else 'R'
    dx = -1 if s > f else 1
    while s != f:
        if t in d and (d[t][0] <= s <= d[t][1] or d[t][0] <= s + dx <= d[t][1]):
            ans.append('X')
        else:
            ans.append(mv)
            s += dx
        t += 1
    print ''.join(ans)
main()
