R=lambda:map(int,raw_input().split())
n,m,s,f=R()
t=[R() for _ in range(m)]+[[10**9+9,0,0]]
a,d=('R',1) if s<f else ('L',-1)
i,j,v=1,0,''
while s!=f:
  b=s+d
  while t[j][0]<i:j+=1
  if t[j][0]==i and ((s>=t[j][1] and s<=t[j][2]) or (b>=t[j][1] and b<=t[j][2])):
    v+='X'
  else:
    v+=a
    s=b
  #print i,s,b,f,t[j],v
  i+=1
print v
