n, m, s, f = map(int, raw_input().split())
a = []
ans = ""
t = 1
d = 1 if s < f else -1
for q in xrange(m):
	a = map(int, raw_input().split())
	while a[0] > t and s != f:
		ans += "R" if d == 1 else "L"
		s += d
		t += 1
	if s == f:
		break
	if a[1] <= s <= a[2] or a[1] <= s + d <= a[2]:
		ans += "X"
	else:
		ans += "R" if d == 1 else "L"
		s += d
	t += 1
	if s == f:
		break
while s != f:
	ans += "R" if d == 1 else "L"
	s += d
	t += 1
print ans
