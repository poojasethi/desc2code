
import sys

N, M, S, F = map(int, raw_input().split())
T = []

for i in xrange(M):
	T.append(map(int, raw_input().split()))
T.append((0, 0, 0))

c = 1
p = 0
while S!=F:
	NS = S
	if S<F:
		NS = S+1
	else:
		NS = S-1
	if c==T[p][0] and (T[p][1]<=S<=T[p][2] or T[p][1]<=NS<=T[p][2]):
		sys.stdout.write('X')
	else:
		if S<F:
			sys.stdout.write('R')
			S += 1
		else:
			sys.stdout.write('L')
			S -= 1
	if c==T[p][0]:
		p += 1
	c += 1
print
