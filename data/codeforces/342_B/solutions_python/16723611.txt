R = lambda : map(int , raw_input().split())
n,m,s,f = R()
out = ""
prev_t = 1
sign = 1 if s < f else -1
dir = "R" if sign == 1 else "L"
for i in xrange(m):
	t, l, r = R()
	while t > prev_t and s != f:
		out += dir
		s += sign
		prev_t += 1
	if s == f:
		break
	if l <= s <= r or l <= s + sign <= r:
		out += "X"
	else:
		out += dir
		s += sign
	prev_t += 1

while s != f:
	out += dir
	s += sign
	prev_t += 1
print out
