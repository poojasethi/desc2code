I=lambda:map(int, raw_input().split())
n, m, s, f = I()
d = {}
for _ in xrange(m):
    t, l, r = I()
    d[t] = (l, r)
W = 'R' if s < f else 'L'
move = 1 if s < f else -1
i = 1
ans = ''
while True:
    if s == f:
        break; 
    if i in d:
        
        if (d[i][0] -1 <= s <= d[i][1] and W == 'R') or (d[i][0] <= s <= d[i][1] + 1 and W == 'L'):
            ans += 'X'
        else:
            ans += W
            s += move
    else:
        s += move
        ans += W
    i += 1
print ans
