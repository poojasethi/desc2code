#include<cstdio>

int n,m,s,f,t,l,r,i,tt;

bool j()
{
    int ss=s<f?s+1:s-1;
    return s>=l&&s<=r||ss>=l&&ss<=r;
}

int main()
{
    scanf("%d%d%d%d",&n,&m,&s,&f);
    while(s!=f)
    {
        if(t==tt&&m--)
            scanf("%d%d%d",&t,&l,&r);

        tt++;
        if(t==tt&&j())
            putchar('X');
        else if(s<f)
            putchar('R'),s++;
        else if(s>f)
            putchar('L'),s--;
    }
    puts("");

    return 0;
}
