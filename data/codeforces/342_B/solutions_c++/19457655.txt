#include<iostream>
#include<cstdio>

using namespace std;
//SUPERMEAT BOY
//XENIA AND SPIES

int n,m,s,f,l,r;

bool stay()
{
    int next=s>f?s-1:s+1;
    return (s>=l && s<=r) || (next>=l && next<=r);
}

int main()
{
    cin>>n>>m>>s>>f;
    int t=0,tt=0;
    while(s!=f)
    {
        while(t==tt && m--)
        {
            cin>>t>>l>>r;
            break;
        }

        tt++;
        if(t==tt && stay()) cout<<'X';
        else if(s>f) cout<<'L',s--;
        else if(s<f) cout<<'R',s++;
    }
    return 0;
}
