#include <iostream>

using namespace std;

int main()
{
	unsigned long  n , m , s , f , t = 0 , l , r , i = 0;
	cin>>n>>m>>s>>f;
	for(  ; true ; ++i)
	{
		if(t == i && i < m )
			cin>>t;
		if(t == i + 1 )
			cin>>l>>r;
		else
        {
            l = 0 ;
            r = 0 ;
        }
		if( s < f && ( s + 1 < l || s > r))
		{
			cout<<'R';
			++s;
		}
		else if(s > f && ( s < l || s - 1 > r ) )
		{
			cout<<'L';
			--s;
		}
		else if( s == f )
			break;
		else cout<<'X';
	}
	return 0;
}
