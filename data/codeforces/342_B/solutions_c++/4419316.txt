#include <iostream>

using namespace std;

int main() {
	int n, m, s, f;
	cin >> n >> m >> s >> f;
	int shift = s < f ? 1 : -1;
	char action = s < f ? 'R' : 'L';
	int last = 0;
	for (int i = 0; s != f && i < m; i++) {
		int t, l, r;
		cin >> t >> l >> r;
		for (int j = last + 1; s != f && j < t; j++) {
			cout << action;
			s += shift;
		}
		if (s != f) {
			if ((l <= s && s <= r) || (l <= s + shift && s + shift <= r)) {
				cout << 'X';
			} else {
				cout << action;
				s += shift;				
			}
		}
		last = t;
	}
	while (s != f) {
		cout << action;
		s += shift;
	}
}
