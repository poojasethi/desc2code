#include<stdio.h>

int main(){
	int n,m,s,f;
	scanf("%d %d %d %d",&n,&m,&s,&f);
	int i,t[m+1],l[m+1],r[m+1];
	for(i=0;i<m;i++)
		scanf("%d %d %d",&t[i],&l[i],&r[i]);
	int move=1;
	i=0;
	while(s!=f){
		if(s<f){
			if(move==t[i] and ((s>=l[i] and s<=r[i]) or ((s+1)>=l[i] and (s+1)<=r[i])))
				printf("X");
			else{
				s=s+1;
				printf("R");
			}
		}
		else{
			if(move==t[i] and ((s>=l[i] and s<=r[i]) or ((s-1)>=l[i] and (s-1)<=r[i])))
				printf("X");
			else{
				s=s-1;
				printf("L");
			}
		}
		if(move==t[i])i++;
		move++;

	}
	printf("\n");
	return 0;
}
