#include<iostream>
#define ll long long
using namespace std;
const ll N=1e5;
ll t[N],l[N],r[N];
int main()
{
    ll n,m,s,f,g=1,k=0;
    char c='R';
    cin>>n>>m>>s>>f;
    s--;
    f--;
    if(s>f)
        g=-1,c='L';
    for(ll i=0;i<m;i++)
        cin>>t[i]>>l[i]>>r[i],t[i]--,l[i]--,r[i]--;
    for(ll i=0;s!=f;i++)
    {
        if((t[k]==i)&&((l[k]<=s && s<=r[k])||(l[k]<=s+g && s+g<=r[k])))
            cout<<'X';
        else
            cout<<c,s+=g;
        if(t[k]==i)
            k++;
    }
}
