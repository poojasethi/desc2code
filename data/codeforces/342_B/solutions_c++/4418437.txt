#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,x1,x2,ti;
int t[N],l[N],r[N];
int main() {
	scanf("%d%d%d%d",&n,&m,&x1,&x2);
	For(i,1,m) scanf("%d%d%d",&t[i],&l[i],&r[i]);
	j=0;
	for(ti=1;x1!=x2;ti++) {
		if (j<m&&t[j+1]==ti) j++;
		int k=x1+(x1>x2?-1:1);
		if (t[j]==ti&&(l[j]<=x1&&r[j]>=x1||l[j]<=k&&r[j]>=k)) printf("X");
		else printf("%s",x1>x2?"L":"R"),x1=k;
	}
	//for(;;);
	return 0;
}
