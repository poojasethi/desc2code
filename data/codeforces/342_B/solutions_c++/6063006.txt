#include <iostream>
using namespace std;
struct a
{
	int t,l,r;
}st[100000+1];
int main() {
	int s,f,n,m;
	char k;
	cin>>n>>m>>s>>f;
	if(s>f)k='L';
	else k='R';
	for(int i=1;i<=m;++i)cin>>st[i].t>>st[i].l>>st[i].r;
	int p=1;
	int step=1;
	while(s!=f)
	{
		if(step==st[p].t)
		{
			if(k=='R')
			{if(st[p].l<=s+1 && s<=st[p].r)cout<<'X';
			else {cout<<k;if(k=='R')++s;else --s;}
			++p;
			}
			else
			{
				if(st[p].l<=s && s-1<=st[p].r)cout<<'X';
			else {cout<<k;if(k=='R')++s;else --s;}
			++p;
			}
		}
		else {cout<<k;if(k=='R')++s;else --s;}
		++step;
		
	}
}
