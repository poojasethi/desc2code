#include <cstdio>

int main()
{
	int n, m, s, f, t = 1;
	scanf("%d%d%d%d", &n, &m, &s, &f);
	while (m--)
	{
		if (s == f) break;
		int c, a, b;
		scanf("%d%d%d", &c, &a, &b);
		while (t != c)
		{
			++t;
			if (s > f) printf("L"), --s;
			else if (s < f) printf("R"), ++s;
			if (s == f) break;
		}
		if (s == f) break;
		if (s > f)
		{
			if (a > s || b < s-1) printf("L"), --s;
			else printf("X");
		}
		else
		{
			if (b < s || a > s+1) printf("R"), ++s;
			else printf("X");
		}
		++t;
	}
	while (s != f)
	{
		if (s > f) printf("L"), --s;
		else if (s < f) printf("R"), ++s;
	}
	printf("\n");
}
