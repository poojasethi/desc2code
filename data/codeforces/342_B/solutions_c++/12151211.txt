#include<iostream>
using namespace std;
int main()
{
    int n,m,s,f;
    cin>>n>>m>>s>>f;
    int t[m],l[m],r[m];
    for(int i=0;i<m;i++)
    {
        cin>>t[i]>>l[i]>>r[i];
    }
    int time=1;
    int flag;
    if(s<f) flag=1;
    else flag=-1;
    int i=0;
    while(s!=f)
    {
        if(time==t[i]){
        if((l[i]<=s&&s<=r[i])||(l[i]<=s+flag&&s+flag<=r[i]))
        {
            cout<<"X";
            time++;i++;
            continue;
        }
        i++;
        }
        if(flag==1) cout<<"R";
        else cout<<"L";
        s+=flag;
        time++;
    }
    return 0;
}
