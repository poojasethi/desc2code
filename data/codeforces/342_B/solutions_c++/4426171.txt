#include <iostream>
using namespace std;
int main()
{   long int q,w,e,r,t,c,v=0,y,u,i,k;
    char a;
    cin>>q>>w>>e>>r;
    if(e<r){a='R';t=1;}
    else{a='L';t=-1;}
    for(c=0;(c<w)&&(e!=r);c++){
    cin>>y>>u>>i;
    k=y-v-1;
    while((k)&&(e!=r)){
    k--;
    e+=t;
    cout<<a;}
    v=y;
    if(e==r)
    break;
    if(((e>=u)&&(e<=i))||((e+t>=u)&&(e+t<=i)))
    cout<<'X';
    else{
    cout<<a;
    e+=t;}}
    while(e!=r){
    e+=t;
    cout<<a;}
    return 0;
}
