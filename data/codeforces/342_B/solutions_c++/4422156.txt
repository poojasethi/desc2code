#include <cstdio>

int main() {
  int n, m, s, f;
  scanf("%d%d%d%d", &n, &m, &s, &f);
  char c = s < f ? 'R' : 'L';
  int d = s < f ? 1 : -1;
  int u = 1, t, l, r;
  while (m-- && s != f) {
    scanf("%d%d%d", &t, &l, &r);
    while (u < t && s != f) {
      s += d;
      putchar(c);
      ++u;
    }
    if (s != f)
      if ((s < l || r < s) && (s + d < l || r < s + d))
        s += d, putchar(c);
      else
        putchar('X');
    u = t + 1;
  }
  while (s != f)
    s += d, putchar(c);
  putchar('\n');
}
