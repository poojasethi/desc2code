#include<iostream>
using namespace std;
int a,s,d[100002][3],f,g,h,j,k,l,i,n,m,q,j1=1;

cin>>n>>m>>f>>g;
for(i=0;i<m;i++){
cin>>d[i][0]>>d[i][1]>>d[i][2];
}
q=f;
if(f>g) {c='L';j1=-1;}
for(i=min(f,g);i<max(f,g);i++){

else {cout<<c;q+=j1;}
}
