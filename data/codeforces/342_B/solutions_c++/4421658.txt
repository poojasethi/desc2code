#include<stdio.h>
int n, m;
int s,e;
int a[100009];
int b[100009];
int c[100009];
int d;
char x;
int main()
{
	int i, j;
	scanf("%d %d %d %d",&n,&m,&s,&e);
	for(i=0; i<m; i++)
		scanf("%d %d %d",&a[i],&b[i],&c[i]);
	if(s<e){d=1;x='R';}
	else {d=-1;x='L';}
	i=0;j=1;
	while(s!=e)
	{
		if(a[i] == j)
		{
			if(b[i] <= s && s <= c[i])
				printf("X");
			else if(b[i] <= s+d && s+d <= c[i])
				printf("X");
			else
			{
				printf("%c",x);
				s+=d;
			}
			i++;
		}
		else
		{
			s += d;
			printf("%c",x);
		}
		j++;
	}
}
