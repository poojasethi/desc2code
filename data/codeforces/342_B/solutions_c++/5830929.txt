#include<cstdio>

int n,m,s,f;
int t[111111],l[111111],r[111111];
int cur,p,nxt;

bool isWatching(int pos) {
	while(t[p]<cur&&p<m)p++;
	if(p>=m)return false;
	
	if(t[p]==cur) {
		if(l[p]<=pos&&r[p]>=pos)return true;
	}
	return false;
}

int main() {
	scanf("%d%d%d%d", &n,&m,&s,&f);
	for(int i=0;i<m;i++)
		scanf("%d%d%d", &t[i],&l[i], &r[i]);
	cur=1;p=0;
	while(s!=f){
		if(s<f)nxt=s+1;
		else nxt=s-1;
		if(isWatching(s)||isWatching(nxt)) printf("X");
		else if(s<nxt){printf("R");s=nxt;}
		else {printf("L");s=nxt;}
		cur++;
	}

	return 0;
}
