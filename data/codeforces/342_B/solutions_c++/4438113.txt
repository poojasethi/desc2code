#include <cstdio>

int n,m,s,f,i,j,a,b,c,d,t;

int main(){
	scanf("%d%d%d%d",&n,&m,&s,&f);
	t=0;
	d=s<f?1:-1;
	for(i=0;i<m;i++){
		scanf("%d%d%d",&a,&b,&c);
		if (s==f) continue;
		t=a-t;
		for(j=1;j<t&&s!=f;j++){
			printf("%c",s<f?'R':'L');
			s+=d;
		}
		if (s<f){
			if (b-1<=s&&s<=c){
				printf("X");
			}else{
				printf("%c",s<f?'R':'L');
				s+=d;
			}
		}else if (s>f){
			if (b<=s&&s<=c+1){
				printf("X");
			}else{
				printf("%c",s<f?'R':'L');
				s+=d;
			}
		}
		t=a;
	}
	if (s!=f){
		for(i=0;i<(f-s)/d;i++){
			printf("%c",s<f?'R':'L');
		}
	}
	printf("\n");
	return 0;
}
