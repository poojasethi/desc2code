#include<bits/stdc++.h>
using namespace std;
int arr[100009],arr1[100009],arr2[100009];
int main()
{
    int n,m,s,e;cin>>n>>m>>s>>e;for(int i=0; i<m; i++)cin>>arr[i]>>arr1[i]>>arr2[i];string str="";int cur=s,l=0,t=1;while(cur!=e)
    {if(cur<e){if(l>m||(arr[l]!=t||(arr[l]==t&&(cur>arr2[l]||cur<arr1[l])&&(cur+1>arr2[l]||cur+1<arr1[l]))))str+='R',cur++;else str+='X';}
    else{if(l>m||(arr[l]!=t||(arr[l]==t&&(cur>arr2[l]||cur<arr1[l])&&(cur-1>arr2[l]||cur-1<arr1[l]))))str+='L',cur--;else str+='X';}if(arr[l]<=t)l++;t++;}cout<<str;
}
