#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

const int maxm=100010;

int n,m,s,f,t[maxm],l[maxm],r[maxm];

int main()
{
	scanf("%d%d%d%d",&n,&m,&s,&f);
	for (int a=1;a<=m;a++)
		scanf("%d%d%d",&t[a],&l[a],&r[a]);
	for (int a=1,b=1;s!=f;b++)
	{
		while (a<=m && t[a]<=b)
			a++;
		a--;
		if (s>f)
		{
			if (t[a]==b && s>=l[a] && s<=r[a]+1) printf("X");
			else printf("L"),s--;
		}
		else
		{
			if (t[a]==b && s>=l[a]-1 && s<=r[a]) printf("X");
			else printf("R"),s++;
		}
	}
	printf("\n");

	return 0;
}
