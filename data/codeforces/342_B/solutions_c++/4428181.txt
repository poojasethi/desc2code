#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
long n,m,s,f,t,i,l,r,T,x,d;
char ch;
int main() {
	scanf("%ld%ld%ld%ld",&n,&m,&s,&f);
	if (s<f) {
		ch='R';d=1;
	} else {
		ch='L';d=-1;
	}
	x=s;T=1;
	for (i=1;i<=m;i++) {
		scanf("%ld%ld%ld",&t,&l,&r);
		while (T<t) {
			if (x==f) break;			
			T++;x+=d;
			printf("%c",ch);
		}
		if (x==f) break;
		if (((x<l)||(x>r))&&((x+d<l)||(x+d>r))) {
			x+=d;
			printf("%c",ch);
		} else printf("X");
		T++;
	}
	while (x!=f) {
		x+=d;
		printf("%c",ch);
	}
	return 0;
}
