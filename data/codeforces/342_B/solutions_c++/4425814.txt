#include "iostream"
using namespace std;
int main()
{
	int n,m,s,f;
	int t=0,l,r;
	cin>>n>>m>>s>>f;
	char a=s>f?'L':'R';
	int j=1,i=1;
	for(i=1;i<=m;i++){
		cin>>t>>l>>r;
		for(j;j<t&&s!=f;j++){
			cout<<a;
			if(s>f)s--;
			else s++;
		}
		if(s==f)continue;
		j++;
		if(s>f){
			if(r<s-1||l>s){
				cout<<a;
				s--;
			}else cout<<'X';
		}else{
			if(l>s+1||r<s){
				cout<<a;
				s++;
			}else cout<<'X';
		}
	}
	//cout<<s<<f<<endl;
	while(s!=f){
		cout<<a;
		if(s>f)s--;
		else s++;
	}
	//scanf("%d");
}
