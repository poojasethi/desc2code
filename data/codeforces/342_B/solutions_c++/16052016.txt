#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int n,m,s,f,l,r,t;
int stay() {
    int next = (s<f) ? s+1:s-1;
    return s>=l&&s<=r||next>=l&&next<=r;
}
int main() {
    int tt=0;
    scanf("%d%d%d%d",&n,&m,&s,&f);
    while(s!=f) {
        if(t==tt&&m--)
	    scanf("%d%d%d",&t,&l,&r);
	++tt;
	if(t==tt&&stay())
	    putchar('X');
	else if(s<f)
	    putchar('R'),s++;
	else
	    putchar('L'),s--;
    }
    puts("");
    return 0;
}
