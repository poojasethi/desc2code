#include <cstdio>

int main(){
  int N, M, S, F, t, l, r, diff, turn = 1;
  bool end = false;
  scanf("%d %d %d %d", &N, &M, &S, &F); diff = S<F? 1: -1;
  while(M--){
    scanf("%d %d %d", &t, &l, &r);
    while(turn++<t){
      printf("%c", diff<0? 'L': 'R'); S+=diff; if(S==F){ end = true; break; }
    }
    if(end) break;
    if((S+diff<l || S+diff>r) && (S<l || S>r)){
      printf("%c", diff<0? 'L': 'R'); S+=diff; if(S==F) break;
    }
    else printf("X");
  }
  while(S!=F){ printf(diff<0? "L": "R"); S+=diff; }
  printf("\n");
  return 0;
}
