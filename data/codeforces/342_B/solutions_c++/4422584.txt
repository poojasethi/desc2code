#include <cstdio>
#include <cstring>

int n,m,s,f,t[100005],l[100005],r[100005];

int main(){
	scanf("%d%d%d%d",&n,&m,&s,&f);
	for (int i=0;i<m;++i){
		scanf("%d%d%d",&t[i],&l[i],&r[i]);
	}
	int i=0,j=1;
	while ( s != f ){
		int next = s>f?s-1:s+1;
		if ( t[i] == j &&
			 ( ( l[i] <= next && next <= r[i] ) ||
			   ( l[i] <= s && s <= r[i] ) ) ){
			printf("X");
		}
		else {
			printf("%c",s>f?'L':'R');
			s>f?--s:++s;
		}
		++j;
		if ( j > t[i] )
			++i;
	}
}
