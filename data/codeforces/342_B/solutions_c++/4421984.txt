#include <iostream>
#include <cstdio>

using namespace std;

int n, m, s, f, l, r, t, cur;

int main() {
	cin >> n >> m >> s >> f;

	cur = 1;

	while (s != f) {
		if (cur > t)
			cin >> t >> l >> r;
			if (s == f) {
				break;
				continue;
			}       
	
		if (s < f) {
			if (cur != t || (s + 1 < l || s > r))
				cout << "R", ++s;
			else
				cout << "X";
		} else {
			if (cur != t || (s < l || s - 1 > r))
				cout << "L", --s;
			else
				cout << "X";
		}
		cur++;
	} 
}
