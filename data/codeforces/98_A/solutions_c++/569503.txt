#include <stdio.h>
#include <algorithm>
using namespace std;
int count1=0,count2=0,count3=0,count4=0,count5=0;

void judge(char str[7])
{
    count1++;
    if(str[1]==str[2]&&str[2]==str[3]&&str[3]==str[4]) count2++;
    if(str[1]==str[2]&&str[3]==str[4])                 count3++;
    if(str[0]==str[1]&&str[2]==str[3]&&str[4]==str[5]) count4++;
    if(str[0]==str[1]&&str[1]==str[2]&&str[3]==str[4]&&str[4]==str[5]) count5++;
}

int main()
{
    char str[7];
    scanf("%s",str);
    sort(str,str+6);
    judge(str);
    while(next_permutation(str,str+6))
    judge(str);
    printf("%d\n",(count1+count2*6+count3*3+count4*6+count5*8)/24);
    return 0;
}
