set <string> S;
const int opp[6] = {1, 0, 4, 5, 2, 3};
const int adj[6][4] = {{2, 3, 4, 5}, {5, 4, 3, 2}, {1, 3, 0, 5}, {4, 0, 2, 1}, {3, 1, 5, 0}, {2, 0, 4, 1}};

void Insert(string s) {
	string t = s;
	int i, j, k;
	for (i = 0; i < 6; ++i) {
		t[0] = s[i];
		t[1] = s[opp[i]];
		for (j = 0; j < 4; ++j) {
			for (k = 2; k < 6; ++k)
				t[k] = s[adj[i][(j + k - 2) % 4]];
			S.insert(t);		
			//cout << t << endl;		
		}		
	}
	//puts("");
}

int main() {
	string s;
	getline(cin, s);
	sort(s.begin(), s.end());
	int res = 0;
	do {
		if (S.find(s) == S.end()) {
			++res;					
			Insert(s);
		}			
	} while (next_permutation(s.begin(), s.end()));
	cout << res << endl;
	return 0;		
}
