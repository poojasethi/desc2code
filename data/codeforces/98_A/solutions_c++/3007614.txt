#include <stdio.h>
#include <algorithm>
using namespace std;
int main ()
{
    int a[26]={0};
    char ch;
    for(int i=0;i<6;i++)
    {
        scanf("%c",&ch);
        a[ch-'A']++;
    }
    sort(a,a+25);
    int sum=0;
    for(int i=0;i<=25;i++)
    {
        if(a[i]!=0)
        {sum*=10;
        sum+=a[i];}
    }
    if(sum==6)
    printf("1\n");
    if(sum==111111)
    printf("30\n");
    if(sum==11112)
    printf("15\n");
    if(sum==15)
    printf("1\n");
    if(sum==24)
    printf("2\n");
    if(sum==33)
    printf("2\n");
    if(sum==222)
    printf("6\n");
    if(sum==114)
    printf("2\n");
    if(sum==123)
    printf("3\n");
    if(sum==1122)
    printf("8\n");
    if(sum==1113)
    printf("5\n");
}
