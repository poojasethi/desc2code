#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <set>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const int a1[2][6]={{1,3,4,5,2,6},{3,2,6,4,1,5}};
int i,j,k,n,m;
int a[10],f[10];
char p[10];
string s;
set<string> yao,laji;
void sou(string a,int ty) {
	if (laji.count(a)) return;
	int i,j;
	if (ty) laji.insert(a);
	For(i,0,1) {
		string b;
		For(j,0,5) b+=a[a1[i][j]-1];
		sou(b,1);
	}
}
void dfs(int x,string a) {
	if (x>6) {
		if (!laji.count(a)) {
			yao.insert(a);
			sou(a,0);
		}
		return;
	}
	int i;
	For(i,1,6) if (!f[i]) {
		f[i]=1;
		dfs(x+1,a+p[i]);
		f[i]=0;
	}
}
int main() {
	scanf("%s",p+1);
	dfs(1,s);
	printf("%d\n",yao.size());
	//for(;;);
	return 0;
}
