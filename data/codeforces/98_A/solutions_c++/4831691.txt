#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
    int i,a[6];
    char s[7];
    memset(a,0,sizeof(a));
    scanf("%s",s);
    for(i=0;i<7;++i)
    switch(s[i])
    {
        case 'B':
            ++a[0];break;
        case 'R':
            ++a[1];break;
        case 'O':
            ++a[2];break;
        case 'Y':
            ++a[3];break;
        case 'G':
            ++a[4];break;
        case 'V':
            ++a[5];break;
    }
    sort(a,a+6);
    if(a[5]>4) printf("1\n");
    else if(a[5]==4) printf("2\n");
    else if(a[5]==3)
    {
        if(a[4]==3) printf("2\n");
        else if(a[4]==2) printf("3\n");
        else printf("5\n");
    }
    else if(a[5]==2)
    {
        if(a[4]==1) printf("15\n");
        else if(a[3]==2) printf("6\n");
        else printf("8\n");
    }
    else printf("30\n");
    return 0;
}
