#include <cstdio>
#include <algorithm>
using namespace std;
const int N = 15;
int s[N];
char a[N];
int main() {
	scanf("%s", a);
	for(int i = 0; i < 6; i ++) {
		if(a[i] == 'R') s[0]++;
		else if(a[i] == 'O') s[1]++;
		else if(a[i] == 'Y') s[2]++;
		else if(a[i] == 'G') s[3]++;
		else if(a[i] == 'B') s[4]++;
		else if(a[i] == 'V') s[5]++;
	}
	sort(s, s+6);
	int sum = 0;
	for(int i = 0; i < 6; i ++) {
//		printf("%d ", s[i]);
		sum = sum * 10 + s[i];
	} //puts("");
//	printf("%d\n", sum);
	if(sum == 111111) puts("30");
	else if(sum == 11112) puts("15");
	else if(sum == 1113) puts("5");
	else if(sum == 1122) puts("8");
	else if(sum == 114) puts("2");
	else if(sum == 123) puts("3");
	else if(sum == 15) puts("1");
	else if(sum == 222) puts("6");
	else if(sum == 24) puts("2");
	else if(sum == 33) puts("2");
	else if(sum == 6) puts("1");
	
	return 0;
}
