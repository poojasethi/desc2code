    mp['B'] = 4;
    mp['V'] = 5;

    char str[7];

    for(int i=0;i<6;i++)
        a[mp[str[i]]]++;
    sort(a,a+6);
    reverse(a,a+6);

    int s;
    if(a[0]>=5)
        s = 1;
    else if(a[0]==4)
        s = 2;
    else if(a[1]==3)
        s = 2;
    else if(a[0]==3 && a[1]==2)
        s = 3;

    else if(a[2]==2)
        s = 6;
    else if(a[1]==2)
        s = 8;
    else if(a[0]==2)
        s = 15;
    else

}
