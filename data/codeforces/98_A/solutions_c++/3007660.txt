#include<iostream>
#include<cstring>
using namespace std;
const int mm=40;
char s[mm];
int num[mm],f[mm];
int main()
{
int k;
int cnt;
while(cin>>s)
{
  k=0;cnt=0;
  memset(num,0,sizeof(num));
  for(int i=0;i<6;i++)
    num[s[i]-'A']++;
   for(int j=0;j<26;j++)
  {
    if(num[j])cnt++,f[k++]=num[j];
  }
  int ans;
  switch(cnt)
  {
    case 1:ans=1;break;
    case 2:if(f[0]==1||f[1]==1)
            {ans=1;}
            else ans=2;break;
    case 3:if(f[0]==3||f[1]==3||f[2]==3)
              ans=3;
           else if(f[0]==2&&f[1]==2&&f[2]==2)
            ans=6;
           else ans=2;break;
    case 4:if(f[0]==3||f[1]==3||f[2]==3||f[3]==3)
            ans=5;
            else ans=8;break;
    case 5: ans=15;break;
    case 6: ans=30;break;
  }
  cout<<ans<<"\n";
 }
}
