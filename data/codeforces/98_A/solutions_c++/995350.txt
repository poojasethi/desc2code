void rota(string &b){
	string t=b;
	t[0]=b[1];
	t[3]=b[0];
	t[5]=b[3];
	t[1]=b[5];
	swap(t,b);
}
void rotb(string &b){
	string t=b;
	t[0]=b[2];
	t[4]=b[0];
	t[5]=b[4];
	t[2]=b[5];
	swap(t,b);
}
string getmn(string a) {
	string r="ZZZZZZ";
	for(int k=0;k<4;k++) {
		for(int i=0;i<6;i++) {
			for(int j=0;j<4;j++) {
				r=min(r,a);
				rotb(a);
			}
			rota(a);
		}
		rotb(a);
	}
	return r;
}
