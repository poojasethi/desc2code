#include <stdio.h>
#include <map>
#include <string>
#include <algorithm>
#include <string.h>

using namespace std;


char cad[1000];
char aux[1000];
map<char, int> mapa;
map<string, int> mapa2;


int main()
{
	gets(cad);
	mapa2[string("6")] = 1;
	mapa2[string("15")] = 1;
	mapa2[string("24")] = 2;
	mapa2[string("33")] = 2;
	mapa2[string("114")] = 2;
	mapa2[string("123")] = 3;
	mapa2[string("222")] = 6;
	mapa2[string("1113")] = 5;
	mapa2[string("1122")] = 8;
	mapa2[string("11112")] = 15;
	mapa2[string("111111")] = 30;


	for (int i = 0; i < 6; i++)
	{
		mapa[cad[i]] = mapa[cad[i]] + 1;
	}

	int cont = 0;
	for (map<char, int>::iterator it = mapa.begin(); it != mapa.end(); it++)
	{
		aux[cont] = (*it).second + '0';
		cont++;
	}

	aux[cont] = 0;
	sort(aux, aux + strlen(aux));
	printf("%d", mapa2[string(aux)]);


}
