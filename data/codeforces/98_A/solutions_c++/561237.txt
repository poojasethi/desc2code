#include <iostream>
#include <algorithm>

using namespace std;

bool fl[6][6][6][6][6][6];
string col = "ROYGBV";
int res = 0;

int Rec(int u, int d, int l, int r, int f, int b)
{
	if (fl[u][d][l][r][f][b]) return 0;
	fl[u][d][l][r][f][b] = true;

	// goto Left
	Rec(r, l, u, d, f, b);
	// goto right
	Rec(l, r, d, u, f, b);
	// goto up
	Rec(f, b, l, r, d, u);
	// goto down
	Rec(b, f, l, r, u, d);

	Rec(u, d, f, b, r, l);
	Rec(u, d, b, f, l, r);

	return 1;
}

int get(char ch)
{
	return col.find(ch);
}

int main()
{
	string s; cin >> s;

	int ord[6]; for(int i=0; i<6; i++) ord[i] = i;
	do
	{
		res += Rec(get(s[ord[0]]), get(s[ord[1]]), get(s[ord[2]]),
				   get(s[ord[3]]), get(s[ord[4]]), get(s[ord[5]]));
	} while (next_permutation(ord, ord+6));

	cout << res << endl;

	return 0;
}
