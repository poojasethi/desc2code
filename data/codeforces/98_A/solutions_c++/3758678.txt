#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;
int nmoc(char x,char s[])
{int i,ct=0;
    for(i=0;i<strlen(s);i++)if(s[i]==x)ct++;
    return ct;
}
int main()
{char s[7],e[7]={'R','O','Y','G','B','V'};int a[7],j,nm[7]={0};
cin>>s;
for(j=0;j<6;j++)a[j]=nmoc(e[j],s);
for(j=0;j<6;j++){switch(a[j])
{case 1:{nm[1]++;break;}
case 2:{nm[2]++;break;}
case 3:{nm[3]++;break;}
case 4:{nm[4]++;break;}
case 5:{nm[5]++;break;}
case 6:{nm[6]++;break;}
default:break;}
}
if(nm[1]==6)cout<<"30";
else if(nm[1]==4&&nm[2]==1)cout<<"15";
else if(nm[1]==3&&nm[3]==1)cout<<"5";
else if((nm[1]==2&&nm[4]==1)||(nm[3]==2)||(nm[4]==1&&nm[2]==1))cout<<"2";
else if(nm[1]==2&&nm[2]==2)cout<<"8";
else if((nm[1]==1&&nm[5]==1)||nm[6]==1)cout<<"1";
else if(nm[2]==3)cout<<"6";
else cout<<"3";
 return 0;}
