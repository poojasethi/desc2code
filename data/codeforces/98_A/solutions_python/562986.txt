from itertools import *

l1=[1,2,3,0,4,5]
l2=[4,1,5,3,2,0]
l3=[0,4,2,5,3,1]

def perm(l,p):
	return ''.join([l[c] for c in p])

def incr_inst(orig):
	curr=orig
	s=set()
	r=range(4)
	for i in r:
		for j in r:
			for k in r:
				s.add(curr)
				curr = perm(curr,l1)
			curr = perm(curr,l3)
		curr = perm(curr,l2)
	return min(list(s))

orig=raw_input()

s=set()
for c in permutations(orig):
	s.add(incr_inst(''.join(list(c))))
print len(s)
