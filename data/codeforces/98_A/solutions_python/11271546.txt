#!/usr/bin/env python
# -*- coding: utf-8 -*-
import itertools


def ap(p):
    for i in xrange(4):
        yield p[i:4] + p[:i] + p[4:]

def ap2(p):
    yield p[0]+p[1]+p[2]+p[3]+p[4]+p[5]
    yield p[5]+p[1]+p[4]+p[3]+p[0]+p[2]
    yield p[2]+p[1]+p[0]+p[3]+p[5]+p[4]
    yield p[4]+p[1]+p[5]+p[3]+p[2]+p[0]

def gap(p):
    for p2 in ap(p):
        for p3 in ap2(p2):
            for p4 in ap(p3):
                yield p4

s = set([])
k = 0
for p in itertools.permutations([c for c in raw_input()]):
    fl = True
    for pp in gap(p):
        if not pp in s:
            if fl:
                fl = False
                k += 1
            s.add(pp)
print k
