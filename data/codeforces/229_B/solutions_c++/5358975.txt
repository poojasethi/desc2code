#include <iostream>
#include <fstream>
#include <cstring>

#include <queue>
using namespace std;
#define pii pair<int,int>
//typedef pair<int,int> pii;
int sx,sy,ii,n,m,i,j,k,x,xx,y,yy,z;
int t,tp,tmp;
vector <pair<int,int> > v[111111];
long long d[111111],a[111111];
set <int> s[111111];
priority_queue <pii,vector<pii>,greater<pii> > q;

int main ()
{
cin>>n>>m;
while (m--)
{
 scanf("%d%d%d",&x,&y,&z);
 v[x].push_back(make_pair(y,z));
 v[y].push_back(make_pair(x,z));
}
for (i=1;i<=n;i++)
{
 scanf("%d",&m);
 while (m--)
 {
  scanf("%d",&x);
  s[i].insert(x);
 }
}
for (i=2;i<=n;i++) d[i]=1e18;
q.push(make_pair(d[1],1));
while (!q.empty())
{
 t=q.top().second;m=q.top().first; q.pop();
 if (t==n) break;

 if (a[t]) continue ;
 a[t]=1;


while (s[t].find(m)!=s[t].end()) m++;
for (i=0;i<(int)v[t].size();i++)
 { j=v[t][i].first;z=v[t][i].second;
  if (m+z<d[j]) {
                 d[j]=m+z;
                 q.push(make_pair(d[j],j));
               }
 }
}

if (d[n]==1e18) cout<<-1; else cout<<d[n];
}
