#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <map>
#include <set>
#define N 200010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,t,x,y,z,l,r,v;
int a[N],b[N],next[N],head[N],g[N*100],gg[N],ff[N];
map<int,int> c[N];
set<int> d[N];
inline void add(int x,int y,int z) {
	a[++t]=y,b[t]=z,next[t]=head[x],head[x]=t;
	a[++t]=x,b[t]=z,next[t]=head[y],head[y]=t;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,m) scanf("%d%d%d",&x,&y,&z),add(x,y,z);
	For(i,1,n) {
		scanf("%d",&k);
		int last=-1;
		For(j,1,k) {
			scanf("%d",&x);
			c[i][x]=1;
			if (last+1<x) d[i].insert(last+1);
			//if (last+2<x) d[i].insert(x-1);
			last=x;
		}
		d[i].insert(last+1);
	}
	memset(gg,120,sizeof(gg));
	g[l=r=1]=1; gg[1]=0;
	for (;l<=r;l++) {
		int A=g[l];
		ff[A]=0;
		int D=gg[A];
		if (c[A].count(D)) D=*(d[A].lower_bound(D));
		for (v=head[A];v;v=next[v]) {
			int B=a[v],C=D+b[v];
			if (gg[B]>C) {
				gg[B]=C;
				if (!ff[B]) ff[B]=1,g[++r]=B;
			}
		}
	}
	printf("%d\n",gg[n]>2e9?-1:gg[n]);
	//for(;;);
	return 0;
}
