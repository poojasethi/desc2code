#include <cstdio>
#include <vector>
#include <queue>
using namespace std;
typedef pair<int, int> ii;
vector <ii> p[100001];
vector <ii> :: iterator u;
priority_queue <ii> s;   
vector <int> r[1000001];
vector <int> :: iterator v;    
bool q[100001];
int main()
{
    int i, k, t, a, b, c, n, m;
    scanf("%d %d", &n, &m);     
    for(i = 0; i < m; i++) {
        scanf("%d %d %d", &a, &b, &c);
        p[a].push_back(ii(-c, b));
        p[b].push_back(ii(-c, a));
    }
    for(a = 1; a <= n; a++) {
        q[a] = true;
        scanf("%d", &k);
        r[a].resize(k);		
        for(v = r[a].begin(); v != r[a].end(); v++) scanf("%d", v);
    }
    s.push(ii(0, 1));
    while(!s.empty()) {
        t = -s.top().first;
        a = s.top().second;
        s.pop();
        if(a == n) {
            printf("%d", t);
            return 0;
        }
        if(q[a]) {
            q[a] = false;
            for(v = r[a].begin(); v != r[a].end(); v++) if(*v == t) t++;
            for(u = p[a].begin(); u != p[a].end(); u++) {
                u -> first -= t;
                if(q[u -> second]) s.push(*u);
            }
        }
    }
    printf("%d", -1);
    return 0;
}
