#define pb push_back
#define mp make_pair
#define mod 1000000007
#define endl '\n'
#define endll puts("")
#define mp3(x,y,z) mp(x,mp(y,z))
#define inf mod
#define pii pair<int,int>
#define pll pair<ll,ll>
#define ort (bas+son>>1)

using namespace std;
typedef long long ll;
const int N = 1e5+3;
int n,m,i,j,k,l,r,s,t,d,u,ii,jj,x,y,z,h[N],hh[N];
vector<pii >V[N];
vector<int >v[N];
priority_queue<pii >Q;
int main()
{
	cin>>n>>m;
	while(m--)
	{
		scanf("%d%d%d",&x,&y,&z);
		V[x].pb(mp(y,z));
		V[y].pb(mp(x,z));
	}
	for(i=1;i<=n;i++)
	{
		scanf("%d",&x);
		for(j=0;++j<=x;)
		{
			scanf("%d",&y);
			v[i].pb(y);
		}
	}
	Q.push(mp(0,1));
	while(!Q.empty())
	{
		x=Q.top().nd;
		y=-Q.top().st;
		Q.pop();
		if(h[x])continue;
		if(x==n)
		{
			cout<<y<<endl;return 0;
		}
		z=lower_bound(v[x].begin(),v[x].end(),y)-v[x].begin();
		while(z<v[x].size() and y==v[x][z])
			z++,y++;
		h[x]=1;
		for(i=0;i<V[x].size();i++)
			if(!h[V[x][i].st])
			Q.push(mp(-y-V[x][i].nd,V[x][i].st));
	}
	cout<<-1<<endl;
}
