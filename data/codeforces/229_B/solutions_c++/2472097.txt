#include <cstdio>
#include <vector>
#include <set>
#define pb push_back
#define mp make_pair

using namespace std;
const int maxn = 100 * 1000 + 10;

set <pair <long long, int> > s;
set <long long> dis [maxn];
vector <pair <int, int> > g [maxn];
int d [maxn];

int main () {
	int n, m, a, b, c, tur = 0;
	scanf ("%d%d", &n, &m);
	for (int i = 0; i < m; i++) {
		scanf ("%d%d%d", &a, &b, &c); 
		g [a].pb (mp (c, b));
		g [b].pb (mp (c, a));
	}
	for (int i = 1; i <= n; i++) {
		d [i] = (1 << 30);
		scanf ("%d", &a);
		for (int j = 0; j < a; j++) {
			scanf ("%d", &b);
			dis [i].insert (b);
		}
	}
	s.insert (mp (tur, 1));
	d [1] = tur;
	while (s.size ()) {
		int v = s.begin() -> second;
		s.erase (mp (d [v], v));
		tur = d [v];
		while (dis [v].find (tur) != dis [v].end ())
			tur ++;
		for (int i = 0; i < g [v].size(); i++) {
			int u = g [v][i].second, dd = g [v][i].first;
			if (d [u] > tur + dd) {
				s.erase (mp (d [u], u));
				d [u] = tur + dd;
				s.insert (mp (d [u], u));
			}
		}
	}
	if (d [n] == (1 << 30))
		printf ("-1\n");
	else
		printf ("%d\n", d [n]);
}
