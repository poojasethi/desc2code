#include <stdio.h>
#include <iostream>
#include <sstream>
#include <string>
#include <algorithm>
#include <string.h>
#include <math.h>
#include <vector>
#include <queue>
#include <stack>
#include <map>
#include <set>

using namespace std;

#define ll long long
#define pi pair<int,int>
#define X first
#define Y second
#define pb push_back

int n,m;
vector<pi> v[100005];
set<int> t[100005];
int d[100005];
bool ck[100005];

int main(){
	int i,j,k,s,e;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;i++){
		scanf("%d%d%d",&s,&e,&k);
		v[s].pb(pi(e,k));
		v[e].pb(pi(s,k));
	}
	for(i=1;i<=n;i++){
		scanf("%d",&k);while(k--){
			scanf("%d",&s);
			t[i].insert(s);
		}
	}

	priority_queue<pi,vector<pi>, greater<pi> > q;
	memset(d,63,sizeof(d));

	q.push(pi(0,1));

	while(!q.empty()){
		k=q.top().Y;
		int dis=q.top().X;
		q.pop();

		if(ck[k])continue;
		ck[k]=1;

		while(t[k].find(dis)!=t[k].end()){
			dis++;
		}
		for(j=0;j<v[k].size();j++){
			i=v[k][j].X;
			int L=v[k][j].Y;

			if(ck[i])continue;
			if(d[i]>dis+L){
				d[i]=dis+L;
				q.push(pi(d[i],i));
			}
		}
	}
	if(ck[n])printf("%d",d[n]);
	else printf("-1");
	return 0;
}
