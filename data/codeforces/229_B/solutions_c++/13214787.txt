#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define MOD 1000000007
#define ll long long int
#define MAXINT 1000000000000LL

using namespace std;

vector<pii> g[100005];
vector<int> V[100005];
set<pair<ll,int> >s;
ll D[100005];

ll work(ll d,int n) {
	int i,l;
	l=V[n].size()-1;
	for(i=0;i<=l;i++) {
		if(V[n][i]>=d)
			break;
	}
	if(i==l+1 || V[n][i]>d)	return d;
	for(i=i+1;i<=l;i++) {
		if(V[n][i]-V[n][i-1]>1)
			break;
	}
	return V[n][i-1]+1;
}

int main () {
	int n,m,x,y,i,j,z;
	pair<ll,int> p;
	cin>>n>>m;
	for(i=1;i<=m;i++) {
		cin>>x>>y>>z;
		g[x].pb(mp(y,z));
		g[y].pb(mp(x,z));
	}
	for(i=1;i<=n;i++) {
		cin>>x;
		for(j=1;j<=x;j++) {
			cin>>y;
			V[i].pb(y);
		}
	}
	D[1]=0;
	s.insert(mp(0,1));
	for(i=2;i<=n;i++)
		s.insert(mp(MAXINT,i)),D[i]=MAXINT;
	while(!s.empty()) {
		p=*(s.begin());
		if(p.se==n || p.fi>=MAXINT)	break;
		s.erase(s.begin());
		x=p.se;
		z=work(p.fi,p.se);
		for(i=0;i<g[x].size();i++) {
			y=g[x][i].fi;
			j=g[x][i].se;
			if(z+j<D[y]) {
				s.erase(s.find(mp(D[y],y)));
				D[y]=z+j;
				s.insert(mp(D[y],y));
			}
		}
	}
	if(D[n]<MAXINT) {
		cout<<D[n]<<endl;
	}
	else {
		cout<<-1<<endl;
	}
	return 0;
}
