#include<bits/stdc++.h>

using namespace std;

int i,n,l,a[100005];

int main(){

	cin >> n >> l;
	for(i = 1; i <= n; i++)
		cin >> a[i];

	sort(a+1,a+n+1);

	int Max = max(2*a[1],2*(l-a[n]));
   
	for(i = 1; i < n; i++)
		if(a[i+1]-a[i] > Max)
			Max = a[i+1]-a[i];

	cout << fixed << Max/2.0 << endl;

	return 0;
}
