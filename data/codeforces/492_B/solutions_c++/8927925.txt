#include<bits/stdc++.h>
#define maxx 11111
using namespace std;
int a[maxx];
int main(){
    int i,n,l;
    double ans;
    cin>>n>>l;
    for(i=1;i<=n;i++) cin>>a[i];
    sort(a+1,a+1+n);
    ans=max(a[1],l-a[n]);
    for(i=2;i<=n;i++)
        ans=max(ans,(a[i]-a[i-1])/2.);
    printf("%.9f\n",ans);
}
