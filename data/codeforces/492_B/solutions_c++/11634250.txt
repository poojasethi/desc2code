#include <bits/stdc++.h>

using namespace std;

int main() {
  int n,i;
  long long int l,a[100000],d;
  scanf("%d%64d",&n,&l);
  for(i=0;i<n;i++){
    scanf("%64d",&a[i]);
    }
  sort(a,a+n);
  d= 2*max(a[0],l-a[n-1]);
  for(i=0;i<n-1;i++){
    d = max(d,a[i+1]-a[i]);
  }

  printf("%.10f\n",d/2.0);
  return 0;
}
