#include <algorithm>
#include <cstdio>
using namespace std;

int main()
{
	int v[1000];
	int n,l;
	double d;
	scanf("%d %d", &n, &l);
	for(int i=0; i<n; i++)
		scanf("%d", &v[i]);
	sort(v,v+n);
	d = max(l-v[n-1], v[0]);
	for(int i=0; i<n-1; i++)
		d = max(d, ((double)v[i+1]-v[i])/2);
	printf("%f0000",d);
	return 0;
}
