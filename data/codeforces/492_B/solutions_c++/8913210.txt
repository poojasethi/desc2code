#include<cstdio>
#include<algorithm>

using namespace std;

int a[1000];

int main(){
	int n,l;
	scanf("%d%d",&n,&l);
	for(int i=0;i<n;i++) scanf("%d",a+i);
	sort(a,a+n);
	double ans=max(a[0], l-a[n-1]);
	for(int i=1;i<n;i++){
		ans=max(ans,(a[i]-a[i-1])/2.0);
	}
	printf("%.10f\n",ans);
	return 0;
}
