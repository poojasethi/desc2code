#include <stdio.h>
#include <algorithm>
using namespace std;

int n,l,ls[1002],mxdis;
int main()
{
	scanf("%d%d",&n,&l);
	for(int i=0;i<n;i++)scanf("%d",&ls[i]);
	sort(ls,ls+n);
	mxdis=2.0*max(mxdis,max(ls[0]-0,l-ls[n-1]));
	for(int i=1;i<n;i++)mxdis=max(mxdis,ls[i]-ls[i-1]);
	printf("%.10f",mxdis/2.0);
}
