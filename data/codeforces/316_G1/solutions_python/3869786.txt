#include<stdio.h>
#include<string.h>
#include<math.h>
#include<map>
#include<string>
using namespace std;

#define maxm 2050

char s[maxm],test[maxm];
char p[50][maxm];
int n,lo[50],hi[50];

map<string,int>map1;

int chck(char *s,char *p){

    int i,j,k,ret=0;

    for(i=0;s[i];i++){
        for(j=0,k=i;p[j];j++,k++){
            if(s[k]!=p[j]) break;
        }
        if(!p[j]) ret++;
    }
    return ret;
}

int main(){

    int i,j,k,l,t=1;

    //freopen("in.txt","r",stdin);

    scanf("%s",s);

    scanf("%d",&n);
    for(i=1;i<=n;i++){
        scanf("%s %d %d",&p[i],&lo[i],&hi[i]);
    }

    int ans=0;
    for(i=0;s[i];i++){
        for(j=i,k=0;s[j];j++,k++){
            test[k]=s[j]; test[k+1]=0;
            if(map1[test]) continue;
            map1[test]=1;
            for(l=1;l<=n;l++){
                int now=chck(p[l],test);
                if(now<lo[l] || now>hi[l]) break;
            }
            if(l>n){
                ans++;
            }
        }
    }

    printf("%d\n",ans);

    return 0;
}
