#include <iostream>
#include <set>
#include <vector>

using namespace std;

int countIt(string a, string b)
{
    int pos = -1;
    int count = 0;
    while ((pos = b.find(a, pos + 1)) != string::npos) {
        count++;
    }
    return count;
}

int main()
{
    string s;
    cin >> s;
    int len = s.size();
    int n;
    cin >> n;
    vector<string> p;
    vector<int> l;
    vector<int> r;
    for (int i = 0; i < n; ++i) {
        string pi;
        int li;
        int ri;
        cin >> pi >> li >> ri;
        p.push_back(pi);
        l.push_back(li);
        r.push_back(ri);
    }
    set<string> ans;
    for (int i = 0; i < len; ++i) {
        for (int j = i; j < len; j++) {
            string t = s.substr(i,len-j);
            bool isOk = true;
            for (int x = 0; x < n; ++x) {
                int c = countIt(t, p[x]);
                if (c < l[x] || c > r[x]) {
                    isOk = false;
                    break;
                }
            }
            if (isOk)
                ans.insert(t);
        }
    }

    cout << ans.size() << endl;
}
