#include <iostream>
#include <string>
#include <set>
#include <map>
using namespace std;
map<string,int> mp[11];
set<string> s;
string v[10];
int l[10];
int u[10];
int main()
{
	string str;
	cin>>str;
	int n;
	cin>>n;
	for(int i=(signed)str.size();i>0;--i)
		for(int j=(signed)str.size()-i;j>=0;--j)
			s.insert(str.substr(j,i));
	for(int k=0;k<n;++k)
		cin>>v[k]>>l[k]>>u[k];
	for(int k=0;k<n;++k)
		for(int i=(signed)v[k].size();i>0;--i)
			for(int j=(signed)v[k].size()-i;j>=0;--j)
				++mp[k][v[k].substr(j,i)];
	int count=0,x;
	for(set<string>::iterator it=s.begin();it!=s.end();++it){
		for(int i=0;i<n;++i){
			x=mp[i][*it];
			if(x<l[i] || x>u[i]){
				--count;
				break;
			}
		}
		++count;
	}
	cout<<count;
	return 0;
}
