#include <iostream>
#include <string>
#include <cstring>
#include <cstdio>
#include <set>

using namespace std;

string ss, s[12];
int n, l[12], r[12];
set<string> hash;

bool check(string t) {
    if (hash.find(t) == hash.end()) hash.insert(t);
    else return 0;
    for (int i = 0; i < n; i++) {
        int tot = 0;
        size_t fd = s[i].find(t);
        while (fd != string::npos) {
            tot++;
            fd = s[i].find(t, fd + 1);
        }
        if (tot < l[i] || tot > r[i])
            return 0;
    }
    return 1;
}

int main() {
    cin >> ss >> n;
    for (int i = 0; i < n; i++)
        cin >> s[i] >> l[i] >> r[i];
    int l = ss.size();
    int ans = 0; hash.clear();
    for (int i = 0; i < l; i++)
        for (int j = i + 1; j <= l; j++)
            if (check(ss.substr(i, j - i))) ans++;
    printf("%d\n", ans);

    return 0;
}
