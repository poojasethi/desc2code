#include<iostream>
#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<string>
#include<map>
using namespace std;
#define LL unsigned long long
const int N=210;
int len,n,t,ans;
char str[N],a[N];
char s[20][N];
int ll[20],l[20],r[20];
int next[N];
map<string,bool> vis;
inline bool check() {
	for(int k=1;k<=n;k++) {
		int j=0;
		int tot=0;
		for(int i=1;i<=ll[k];i++) {
			while(j&&s[k][i]!=a[j+1]) j=next[j];
			if(a[j+1]==s[k][i]) j++;
			if(j==t) {
				tot++;
				j=next[j];
			}
		}
		if(!(l[k]<=tot&&tot<=r[k])) return false;
	}
	return true;
}
int main() {
	scanf("%s",str+1);
	len=strlen(str+1);
	scanf("%d",&n);
	for(int i=1;i<=n;i++) {
		scanf("%s%d%d",s[i]+1,&l[i],&r[i]);
		ll[i]=strlen(s[i]+1);
	}
	for(int i=1;i<=len;i++)
		for(int j=i;j<=len;j++) {
			t=0;
			for(int k=i;k<=j;k++) a[++t]=str[k];
			a[t+1]=0;
			if(vis[a+1]) continue;
			vis[a+1]=1;
			next[1]=0;
			int p2=0;
			for(int p=2;p<=t;p++) {
				while(p2&&a[p2+1]!=a[p]) p2=next[p2];
				if(a[p2+1]==a[p]) p2++;
				next[p]=p2;
			}
			if(check()) ans++;
		}
	printf("%d\n",ans);
	return 0;
}
