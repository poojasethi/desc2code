#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
typedef pair<LL, LL> pLL;
#define F first
#define S second
const int MAXN = 3333;
pLL m[MAXN];
LL dp[MAXN];
int main(){
    int n;
    cin >> n;
    for(int i=0; i<n; i++)cin >> m[i].F >> m[i].S;
    dp[n] = 0;sort(m, m+n);//all stop at n
    for(int i=n-1; i>=0; i--){
        LL sm = 0;
        dp[i] = 1LL<<62;
        for(int j=i; j<n; j++){
            sm += m[j].F - m[i].F;//i to j move to i, others stop at leftest j+1
            dp[i] = min(dp[i], dp[j+1] + sm + m[i].S);
        }
    }
    cout << dp[0] << endl;
    return 0;
}
