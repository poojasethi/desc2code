#include<cstdio>
#include<algorithm>
using namespace std;
#define ll __int64
const ll INF=1LL<<62;
struct marble{
	ll x,c;
	bool operator<(marble p)const{return x<p.x;}
}a[3001];
ll min(ll a,ll b){return a<b?a:b;}
ll dp[3001][2];
int main(){
	int n,i,j;
	while(~scanf("%d",&n)){
		for(i=0;i<n;i++)scanf("%I64d%I64d",&a[i].x,&a[i].c);
		sort(a,a+n);
		dp[0][1]=a[0].c;dp[0][0]=INF;
		for(i=1;i<n;i++){
			dp[i][1]=min(dp[i-1][0],dp[i-1][1])+a[i].c;
			dp[i][0]=INF;
			ll cost=0;
			for(j=i-1;j>=0;j--){
				cost+=(i-j)*(a[j+1].x-a[j].x);
				dp[i][0]=min(dp[i][0],dp[j][1]+cost);	
			}
		}
		printf("%I64d\n",min(dp[n-1][0],dp[n-1][1]));
	}
	return 0;
}
