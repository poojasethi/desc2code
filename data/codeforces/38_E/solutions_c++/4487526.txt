#include <iostream>
#include <algorithm>
#include <utility>
 
using namespace std;

#define ll long long
#define x first
#define y second

const int MAX = 3 * 1000 + 5;
pair <ll, ll> p[MAX];
ll dp[MAX];

int main()
{
   int n;
   cin >> n;
   for (int i = 1; i <= n; i++)
      cin >> p[i].x >> p[i].y;
   sort(p + 1, p + n + 1);
   dp[0] = 0;
   ll sum = 0;
   for (int i = 1; i <= n; i++)
   {
      dp[i] = p[i].y + dp[i - 1];
      sum = 0;
      for (int j = i - 1; j >= 1; j--)
      {
         sum += (i - j) * (p[j + 1].x - p[j].x);
         dp[i] = min(dp[i], sum + dp[j - 1] + p[j].y);
      }
   }
   cout << dp[n] << endl;
   return 0;
}
