#include <iostream>
#include <algorithm>
using namespace std;

typedef long long ll;
typedef pair<ll, ll> pll;
#define F first
#define S second

pll m[3001];
ll dp[3001];

int main()
{
        int n;
        cin >> n;
        for(int i=0; i<n; i++)
                cin >> m[i].F >> m[i].S;
        dp[n] = 0;
        sort(m, m+n);
        for(int i=n-1; i>=0; i--)
        {
                ll sum = 0;
                dp[i] = 1LL<<62;
                for(int j=i; j<n; j++)
                {
                        sum += m[j].F - m[i].F;
                        dp[i] = min(dp[i], dp[j+1] + sum + m[i].S);
                }
        }
        cout << dp[0] << endl;
        return 0;
}
