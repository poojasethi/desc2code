#include <iostream>
#include <algorithm>
#include <cstring>
#include <string>
#include <vector>
#include <utility>
using namespace std;

#define sz(a) ((int)a.size())

int n;
vector< pair<int,int> > m;

long long dp[3010][3010];
long long solve(int i,int last){
	if( i == n ) return 0LL;
	if( dp[i][last] != -1 ) return dp[i][last];
	
	return dp[i][last] = min( m[i].second + solve(i+1,i) , m[i].first - m[last].first + solve(i+1,last) );
}

int main(){
	cin >> n;
	for(int i=0;i<n;i++){
		int x,c;
		cin >> x >> c;
		m.push_back( make_pair(x,c) );
	}
	sort(m.begin(),m.end());
	memset(dp,-1,sizeof dp);
	cout << m[0].second + solve(1,0) << endl;
	
	
	
}
