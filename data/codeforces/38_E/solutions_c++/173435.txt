#include <iostream>
#include <algorithm>

using namespace std;

long long f[3000][3000],r[3000];

struct marble {
    int x,c;
    bool operator < (const marble& rhs) const {
        return x<rhs.x;
    }
};

marble m[3000];

int main() {
    ios::sync_with_stdio(false);
    int n;
    cin>>n;
    for (int i=0; i<n; i++)
        cin>>m[i].x>>m[i].c;
    sort(m,m+n);
    f[0][0]=m[0].c;
    r[0]=m[0].c;
    for (int i=1; i<n; i++) {
        f[i][i]=r[i-1]+m[i].c;
        r[i]=f[i][i];
        for (int j=i-1; j>=0; j--) {
            f[i][j]=f[i-1][j]+abs(m[i].x-m[j].x);
            if (f[i][j]<r[i])
                r[i]=f[i][j];
        }
    }
    cout<<r[n-1]<<endl;
    return 0;
}
