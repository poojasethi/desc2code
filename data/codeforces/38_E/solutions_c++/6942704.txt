#include <algorithm>
#include <iostream>
#include <cstdio>
#define N 3200

using namespace std;
const int INF = ~1U >> 1;

struct ball {
    int x, c;
    bool operator < (ball const& b) const {
        return x < b.x;
    }
};

int n;
ball b[N];
long long dp[N];

int main() {
    cin >> n;
    for (int i = 0; i < n; ++i) {
        cin >> b[i].x >> b[i].c;
    }
    sort(b, b + n);
    b[n].x = INF;
    b[n].c = 0;
    for (int i = n - 1; i >= 0; --i) {
        long long d = 0;
        dp[i] = INF;
        for (int j = i; j < n; ++j) {
            d += b[j].x - b[i].x;
            dp[i] = min(dp[i], dp[j + 1] + d + b[i].c);
        }
    }
    cout << dp[0];
    return 0;
}
