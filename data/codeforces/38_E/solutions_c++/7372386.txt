#include<cstdio>
#include<algorithm>
#define pii pair<int,int>
#define A first
#define B second
using namespace std;
int n;
pii mb[3001];
long long dp[3001];
int main(){
    scanf("%d",&n);
    for(int i=1,a,b;i<=n;i++) 
        scanf("%d%d",&a,&b), mb[i] = pii(a,b);
    sort(mb+1,mb+n+1);
    dp[1] = mb[1].B;
    for(int i=2;i<=n;i++){
        long long sum = 0ll;
        dp[i] = dp[i-1]+1ll*mb[i].B; 
        sum += 1ll*mb[i].A;
        for(int j=i-1;j>=1;j--){
            dp[i] = min(dp[i],dp[j-1]+mb[j].B+sum-1ll*mb[j].A*(i-j));
            sum += mb[j].A;
        }
    }
    printf("%I64d\n",dp[n]);
    return 0;
}
