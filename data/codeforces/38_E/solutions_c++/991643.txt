#include<iostream>
#include<algorithm>
using namespace std;
typedef pair<int,int> pi;
typedef long long ll;

pi marb[3010];
ll dp[3010];
int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>marb[i].first>>marb[i].second;
	}
	sort(marb,marb+n);
	fill(dp,dp+n+1,1LL<<61);
	dp[0]=0;
	for(int i=0;i<n;i++) {
		ll s=marb[i].second;
		for(int j=i+1;j<=n;j++) {
			s+=marb[j-1].first-marb[i].first;
			dp[j]=min(dp[j],dp[i]+s);
		}
	}
	cout<<dp[n];
}
