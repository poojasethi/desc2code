#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;

long long dp[3005][3005];

int main(void) {
	int n;
	scanf("%d", &n);
	vector<pair<int, int> > a;
	for(int i=0;i<n;i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		a.push_back(make_pair(x, y));
	}
	sort(a.begin(), a.end());
	const long long INF = (1LL<<60);
	dp[0][0] = a[0].second;
	for(int i=1;i<n;i++) {
		dp[i][i] = INF;
		for(int j=0;j<i;j++) {
			dp[i][j] = dp[i-1][j] + a[i].first - a[j].first;
			dp[i][i] = min(dp[i][i], dp[i-1][j] + a[i].second);
		}
	}
	long long ans = INF;
	for(int i=0;i<n;i++)
		ans = min(ans, dp[n-1][i]);
	printf("%I64d\n", ans);
	return 0;
}
