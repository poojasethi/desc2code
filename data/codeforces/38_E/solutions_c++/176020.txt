#include<iostream>
#include<algorithm>

using namespace std;
typedef long long ll;
typedef pair <ll, ll> pll;
#define F first;
#define S second;

int main(){
    int  n;
    cin>>n;
    pll m[3001];
    ll d[3001];
    for(int i=0;i<n;i++)
    {
        cin>> m[i].first >> m[i].second ;
    } 
    sort(m,m+n);
    long long sum=0;
    d[n]=0;
    for(int i=n-1;i>=0;i--)
    {
            sum=0;
            d[i]=10000000000;
            for(int j=i;j<n;j++)
            {
                    sum+=m[j].first-m[i].first;
                    d[i]=min(d[i],d[j+1]+sum+m[i].second);
            }
    }
    cout<<d[0]<<endl;
return 0;
}
