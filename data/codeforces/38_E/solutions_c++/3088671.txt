#include <iostream>
#include <cstdio>
#include <algorithm>
#define LL long long
#define eps 1e17
const int LMT=3002;
using namespace std;
LL dp[LMT],cons[LMT][LMT];
struct obj
{
   int x;
   LL c;
   bool operator<(const obj &a)const
   {
       return x<a.x;
   }
}ob[LMT];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d%I64d",&ob[i].x,&ob[i].c);
    sort(ob+1,ob+1+n);
    for(int i=1;i<=n;i++)
     for(int j=i+1;j<=n;j++)
     cons[i][j]=cons[i][j-1]+ob[j].x-ob[i].x;
     for(int i=1;i<=n;i++)dp[i]=eps;
    for(int i=1;i<=n;i++)
      for(int j=i;j>=1;j--)
      dp[i]=min(dp[i],dp[j-1]+ob[j].c+cons[j][i]);
      cout<<dp[n]<<endl;
    return 0;
}
