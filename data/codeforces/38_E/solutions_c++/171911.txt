#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define f first
#define s second
#define lim 3001
#define ll long long

ll n,i;
pair <ll,ll> A[lim];
ll M[lim],S[lim];

ll solve(ll i)
{
  ll &res=M[i],j;
	
	if(res==-1)
	{
	  res=S[n-1]-S[i]-(n-1-i)*A[i].f;
		for(j=i+1;j<n;j++)
		  res=min(res,solve(j)+S[j-1]-S[i]-(j-1-i)*A[i].f+A[j].s);
	  //printf("%lld %lld\n",i,res);
	}
	return res;
}
int main()
{
  cin>>n;
	for(i=0;i<n;i++)
	  cin>>A[i].f>>A[i].s;
	sort(A,A+n);
	memset(M,-1,sizeof M);
	for(i=1,S[0]=A[0].f;i<n;i++)
	  S[i]=S[i-1]+A[i].f;
	cout<<solve(0)+A[0].s;
  return 0;
}
