#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
struct data{
  int p,c;
  data(int _p=0,int _c=0){p=_p;c=_c;}
  bool operator<(const data& w)const{return (p<w.p);}
};
ll DP[3001];
int n;
data V[3001];
int main(){
  int i,j;
  scanf("%d",&n);
  for(i=0;i<n;++i)scanf("%d%d",&V[i].p,&V[i].c);
  sort(V,V+n);
  for(i=n-1;i>=0;--i){
   DP[i]=V[i].c+DP[i+1];
   ll PR=0;
   for(j=i+1;j<n;++j){
    PR+=V[j].p-V[i].p;
    DP[i]=min(DP[i],V[i].c+PR+DP[j+1]);
                     }
                     }
  cout<<DP[0]<<'\n';
  return 0;
}
