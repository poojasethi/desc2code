#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef pair<long long, long long> ii;

int main()
{
	long long n;
	cin >> n;
    vector<ii> s(n);

    for(long long i=0; i<n; ++i)
		cin >> s[i].first >> s[i].second;

	sort(s.begin(), s.end());
	vector<long long> v(n);

	v[n-1] = s[n-1].second;

	for(long long i=n-2; i>=0; --i) {
		v[i] = v[i+1];
		long long sum = 0;
		for(long long j=i+2; j<n; ++j) {
			sum += s[j-1].first - s[i].first;
			if(sum + v[j] < v[i])
				v[i] = sum + v[j];
		}

		if(sum + s[n-1].first - s[i].first < v[i])
			v[i] = sum + s[n-1].first - s[i].first;

		v[i] += s[i].second;
	}

	cout << v[0];
    return 0;
}
