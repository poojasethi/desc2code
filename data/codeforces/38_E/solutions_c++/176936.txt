#include<iostream>
#include<algorithm>
using namespace std;
const int MN=3100;
typedef pair<int,int> pp;
pp ps[MN];
int mark[MN][MN];
int d[MN][MN];
int ct[MN],n;
int memz(int pt,int qt){
	if(mark[pt][qt]){
		return d[pt][qt];
	}else{
		if(pt==n){
			return 0;
		}
		mark[pt][qt]=1;
		d[pt][qt]=memz(pt+1,qt)+(ps[pt].first-ps[qt].first);
		d[pt][qt]=min(d[pt][qt],memz(pt+1,pt)+ps[pt].second);
		return d[pt][qt];
	}
}
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		int x,y;
		cin>>x>>y;
		ps[i]=pp(x,y);
	}
	sort(ps,ps+n);
	cout<<memz(1,0)+ps[0].second<<endl;
	return 0;
}
