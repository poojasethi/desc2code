#include<algorithm>
#include<iostream>
using namespace std;
struct ball
{
	long long x;
	long long cost;
};
bool tmp(ball a,ball b)
{
	return a.x<b.x;
}
long long ans=0,dp[3001];
int main()
{
	int n;
	cin>>n;
	ball a[n+1];
	for(int i=1;i<=n;i++)cin>>a[i].x>>a[i].cost;
	sort(a+1,a+n+1,tmp);
	for(int i=1;i<=n;i++)
	{
		long long s=0;
		dp[i]=dp[i-1]+a[i].cost;
		for(int j=i-1;j>0;j--)
		{
			s+=(i-j)*(a[j+1].x-a[j].x);
			dp[i]=min(dp[i],dp[j-1]+a[j].cost+s);
		}
	}
	cout<<dp[n]<<endl;
	return 0;
}
