#include<iostream>
#include<vector>
#include<algorithm>
#include<cstring>
#include<deque>
#include<cstdio>
#include<cmath>
using namespace std;

vector<pair<int,int> > V;
int N,i,a,b,j;
long long DP[3010],dist;

int main()
{
    cin>>N;
    for(i=1;i<=N;i++)
    {
        cin>>a>>b;
        V.push_back(make_pair(a,b));
    }
    sort(V.begin(),V.end());
    DP[N-1] = V[N-1].second;
    for(i=N-2;i>=0;i--)
    {
        dist = 0;
        DP[i] = 1LL * V[i].second + DP[i+1];
        for(j=i+1;j<N;j++)
        {
            dist = dist + 1LL * (V[j].first - V[i].first);
            DP[i] = min(DP[i], 1LL * V[i].second + DP[j+1] + dist);
        }
    }
    cout<<DP[0];
    return 0;
}
