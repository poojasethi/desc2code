#include<bits/stdc++.h>

using namespace std;

vector<pair<int,int> >arr;
int n;
int dp[3100][3100];
int solve(int pos , int last)
{
    if(pos == n)return 0;
    if(dp[pos][last] != -1)
    {
        return dp[pos][last];
    }
    return dp[pos][last] = min(arr[pos].second + solve(pos+1,pos),abs(arr[pos].first - arr[last].first) + solve(pos + 1 ,last));
}
int main()
{

    cin >> n;
    pair<int,int>x;
    memset(dp,-1,sizeof dp);
    for(int i = 0 ; i <  n; i++)
    {
    cin >> x.first >> x.second;
    arr.push_back(x);
    }
    sort(arr.begin(),arr.end());
    cout << arr[0].second + solve(1,0);


    return 0;
}
