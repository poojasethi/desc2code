#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
typedef long long LL;
struct V{
  LL x,c;
  bool operator<(const V& v)const{return (x<v.x);}
}a[4096];
int main(){
  int n;
  LL t[4096];
  cin>>n;
  for(int i=0;i<n;++i)cin>>a[i].x>>a[i].c;
  sort(a,a+n);
  memset(t,0x3f,sizeof(t));
  t[0]=0;
  for(int i=0;i<n;++i)for(LL j=i,s=a[i].c,v;j<n;++j)
    if(t[j+1]>(v=t[i]+(s+=a[j].x-a[i].x)))t[j+1]=v;
  cout<<t[n]<<endl;
  return 0;
}
