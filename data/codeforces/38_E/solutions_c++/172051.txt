#include<iostream>
#include<algorithm>
using namespace std;

const long long INF = (long long)9999999 * 9999999;
const int maxn = 3030;
long long d[maxn]; 
pair<int, int > p[maxn];

int main(){
	
	for(int i = 0; i < maxn; i++)
		d[i] = INF;
	d[0] = 0;

	int n;
	cin >> n;
	for(int i = 1; i <= n; i++)
		cin >> p[i].first >> p[i].second;
	sort(p+1, p+n+1);
	
	for(int i = 1; i <= n; i++){
		long long lc = 0;
		for(int j = i ; j > 0 ; j--){
			lc += (p[j+1].first - p[j].first) * (long long)(i- j);
			d[i] = min(d[i], d[j-1] + p[j].second +  lc);
		}
	}
	cout << d[n] << endl;		
	return 0;
}
