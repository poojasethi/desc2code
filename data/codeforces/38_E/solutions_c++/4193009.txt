//In the name of God
#include <iostream>
#include <algorithm>
using namespace std;

typedef long long int ll;
const ll oo = 1e18;
const int N = 3000 + 10;

int n;
ll dp[N];
pair<ll, ll> a[N];

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i].first >> a[i].second;
	sort(a + 1, a + n + 1);
	fill(dp + 1, dp + n + 1, oo);
	for (int i = 1; i <= n; i++)
		for (ll j = i, t = a[i].second; j <= n; j++, t += a[j].first - a[i].first)
			dp[j] = min(dp[j], dp[i - 1] + t);
	cout << dp[n] << '\n';
	return 0;
}
