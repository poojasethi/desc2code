#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstdlib>

using namespace std;

int n;
long long mn,pmn,d[3333][3333];
struct pt {
	int x,c;
}a[3333];

bool cmp( pt a,pt b )
{
	return a.x < b.x;
}
int main()
{
	scanf("%d",&n);
	for ( int i = 1; i <= n; i++ )
		scanf("%d%d",&a[i].x,&a[i].c);
		
	sort(a+1,a+n+1,cmp);
	
	d[1][1] = a[1].c,mn = a[1].c;
	for ( int i = 2; i <= n; i++ ) {
		pmn = (1LL << 60);
		for ( int j = 1; j <= i; j++ ) {
			if ( j < i ) d[i][j] = d[i-1][j] + a[i].x - a[j].x;
				else d[i][j] = mn + a[i].c;
			if ( d[i][j] < pmn ) pmn = d[i][j];
		}
		mn = pmn;
	}
	printf("%I64d",mn);
}
