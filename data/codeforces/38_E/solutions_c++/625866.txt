#include <iostream>
#include <algorithm>
#include <vector>

#define X first
#define C second
#define rep(i,s,x) for(int i=s; i<x; ++i)
#define all(x) x.begin(),x.end()
#define vmin(x) *min_element(all(x))
#define ll long long
#define vll vector<ll>

using namespace std;

int main()
{
	ll n; cin >> n;
	vector<pair<ll,ll> > a(n);
	rep(i,0,n) cin >> a[i].X >> a[i].C;
	sort(all(a));
	
	vll dp(n); dp[0]=a[0].C;
	rep(i,1,n)
	{
		vll v;
		rep(j,0,i)
		{
			v.push_back(dp[j]+a[i].C);
			dp[j]+=a[i].X-a[j].X;
		}
		dp[i]=vmin(v);
	}
	
	cout << vmin(dp) << endl;
	
	return 0;
}
