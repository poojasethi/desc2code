#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

int n,o[3005];
long long x[3005],c[3005],dp[3005][3005],mi[3005];

bool f1(int i, int j)
{
    return x[i]<x[j];
}

int main()
{
    scanf("%d",&n);
    for (int i=1; i<=n; i++)
    {
        cin>>x[i]>>c[i];
        o[i]=i;
    }
    mi[0]=0;
    sort(o+1,o+1+n,f1);
    for (int i=1; i<=n; i++)
    {
        mi[i]=1000000000000000LL;
        for (int j=1; j<i; j++)
        {
            dp[i][j]=dp[i-1][j]+x[o[i]]-x[o[j]];
            mi[i]=min(mi[i],dp[i][j]);
        }
        dp[i][i]=c[o[i]]+mi[i-1];
        mi[i]=min(mi[i],dp[i][i]);
    }
    cout<<mi[n]<<endl;
    return 0;
}
