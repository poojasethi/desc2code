#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
pair<int,int> p[10000];
long long dp[10000];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d %d",&p[i].first,&p[i].second);
    sort(p+1,p+n+1);
    dp[1]=p[1].second;
    for(int i=2;i<=n;i++)
    {
        dp[i]=dp[i-1]+p[i].second;
        long long tmp=0;
        for(int j=i-1;j>=1;j--)
        {
            tmp+=(long long)(p[j+1].first-p[j].first)*(long long)(i-j);
            dp[i]=min(dp[i],dp[j-1]+tmp+p[j].second);
        }
    }
    printf("%lld\n",dp[n]);
}
