#include<cstdio>
#include<algorithm>
#define MAX 3005
#define inf 1000000000000000000LL
using namespace std;

struct point{int x,c;};

int n,i,j;
long long dp[MAX][MAX],res,now;
point tab[MAX];

bool cmp(point a, point b)
{
if(a.x<b.x) return true;
return false;
}

int main()
{
scanf("%d",&n);
for(i=1; i<=n; ++i) scanf("%d%d",&tab[i].x,&tab[i].c);
sort(tab+1,tab+n+1,cmp);
dp[1][1]=tab[1].c;

for(i=2; i<=n; ++i)
   {
   now=inf;
   for(j=1; j<i; ++j) now=min(now,dp[i-1][j]);
   dp[i][i]=now+tab[i].c;
   for(j=1; j<i; ++j) dp[i][j]=dp[i-1][j]+(tab[i].x-tab[j].x);
   }
   
res=inf;
for(i=1; i<=n; ++i) res=min(res,dp[n][i]);
printf("%d\n",res);
return 0;
}
