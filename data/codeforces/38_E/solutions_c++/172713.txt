#include <stdio.h>
#include <algorithm>
#define inf 1235801361234567ll
using namespace std;

long long n, i, j, f[11111][2], it, sd;
pair<long long, long long> a[11111];

int main(){
	scanf("%I64d", &n);
	for(i = 0; i<n; i++) scanf("%I64d %I64d", &a[i].first, &a[i].second);
	sort(a, a+n);
	f[0][0] = a[0].second;
	f[0][1] = inf;
	for(i = 1; i<n; i++){
		f[i][1] = inf;
		f[i][0] = min(f[i-1][0], f[i-1][1])+a[i].second;
		it = 0; sd = 0;
		for(j = i; j>-1; j--){
			sd += a[j].first;
			++it;
			if (j != i) f[i][1] = min(f[i][1], f[j][0]+sd-a[j].first*it);
		}
	}
	printf("%I64d\n", min(f[n-1][0], f[n-1][1]));
	return 0;
}
