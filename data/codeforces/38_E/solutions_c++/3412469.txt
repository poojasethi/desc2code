#include<cstdio>
#include<algorithm>
using namespace std;
#define ll __int64
struct marble{
	ll x,c;
	bool operator<(marble p)const{return x<p.x;}
}a[3000];
ll min(ll a,ll b){return a<b?a:b;}
ll dp[3000];
int main(){
	int n,i,j;
	ll cost;
	while(~scanf("%d",&n)){
		for(i=0;i<n;i++)scanf("%I64d%I64d",&a[i].x,&a[i].c);
		sort(a,a+n);
		dp[0]=a[0].c;
		for(i=1;i<n;i++){
			dp[i]=dp[i-1]+a[i].c;
			cost=0;
			for(j=i-1;j>=0;j--){
				cost+=(i-j)*(a[j+1].x-a[j].x);
				dp[i]=min(dp[i],dp[j-1]+cost+a[j].c);	
			}
		}
		printf("%I64d\n",dp[n-1]);
	}
	return 0;
}
