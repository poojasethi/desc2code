#include <bits/stdc++.h>
#define FI "rolling.i3"
#define FO "rolling.o3"
#define NN 3005
using namespace std;
typedef pair<int,int> pi;
typedef long long ll;

pi A[NN];
ll dp[NN];

int main() {
    ios_base::sync_with_stdio(0), cin.tie(0);
    #ifndef ONLINE_JUDGE
    ifstream cin(FI);
    ofstream cout(FO);
    #endif // ONLINE_JUDGE
	int n;
	cin >> n;
	for(int i = 0; i < n; i++) {
		cin >> A[i].first >> A[i].second;
	}
	sort(A, A+n);
	fill(dp, dp + n + 1, 1LL << 61);
	dp[0] = 0;
	for(int i = 0; i < n; i++) {
		ll s = A[i].second;
		for(int j = i + 1; j <= n ; j++) {
			s += A[j-1].first - A[i].first;
			dp[j] = min(dp[j], dp[i] + s);
		}
	}
	cout<<dp[n];
	return 0;
}
