#include<iostream>
#include<algorithm>
using namespace std;
#define N 3005
long long f[N],s[N];
pair<long long,long long>q[N];
int n;
int main()
{
    cin>>n;

    for(int i=1;i<=n;i++)
    {
        int a,b;
        cin>>a>>b;
        q[i]=make_pair(a,b);
    }

    sort(q+1,q+n+1);

    for(int i=1;i<=n;i++)  s[i]=s[i-1]+q[i].first-q[1].first;
    f[n]=q[n].second;

    for(int i=n-1;i>=1;i--)
    {

         f[i]=1;
         f[i]<<=62;

        for(int j=i+1;j<=n+1;j++)
        {

            f[i]=min(f[i],s[j-1]-s[i]-(q[i].first-q[1].first)*(j-i-1)+f[j]+q[i].second);
        }
       // cout<<i<<" # "<<f[i]<<endl;

    }
    cout<<f[1]<<endl;

}
