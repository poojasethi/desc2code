#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;
struct abc{
    ll x, c;
};
abc a[3001];
ll dp[3001];
bool comp(abc a, abc b)
{
    return a.x < b.x;
}
main()
{
    ll n, i, j;
    cin >> n;
    for(i = 1; i <= n; i ++)
    {
        cin >> a[i].x >> a[i].c;
    }
    sort(a + 1, a + n + 1, comp);
    dp[1] = a[1].c;
    for(i = 2; i <= n; i ++)
    {
        ll s = 1e18, mas = 0;
        for(j = i; j >= 1; j --)
        {
            mas += (a[j + 1].x - a[j].x) * (i - j);
            s = min(s, dp[j - 1] + mas + a[j].c);
        }
        dp[i] = s;
    }
    cout << dp[n];
}
