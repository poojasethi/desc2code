#include<cstdio>
#include<algorithm>
#include<functional>
#define LL long long

using namespace std;
pair<LL, LL> P[5000];
LL D[5000];

LL F(int n){
	if (n < 0) return 0;
	if (!n) return D[n] = P[n].second;
	if (D[n] != -1) return D[n];
	LL res = P[n].second + F(n-1), C = 0;
	for (int i = n; i ; --i){
		C += P[i].first;
		res = min(res, F(i-2) + C - (n-i+1) * P[i-1].first + P[i-1].second);
	}
	return D[n] = res;
}

int main(){
	int n;
	scanf("%d", &n);
	for (int i = 0 ; i < n; ++i) scanf("%I64d%I64d", &P[i].first, &P[i].second);
	sort(P, P+n);
	for (int i = 0 ; i < n ; ++i) D[i] = -1;
	LL res = F(n-1);
	printf("%I64d\n", res);
	return 0;
}
