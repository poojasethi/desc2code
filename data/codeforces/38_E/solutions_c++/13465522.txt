#include<stdio.h>
long long f[3001],g[3001],a[3001],b[3001],s[3001];
int main()
{
	long long n,m,t;
	scanf("%I64d",&n);
	for (int i=0;i<=n;i++) {s[i]=0; g[i]=0; f[i]=0; }
	for (int i=1;i<=n;i++)
	{
		scanf("%I64d%I64d",&a[i],&b[i]);
	}
	for (int i=1;i<=n;i++)
		for (int j=i+1;j<=n;j++)
			{
				if (a[i]>a[j])
				{
					t=a[i]; a[i]=a[j]; a[j]=t;
					t=b[i]; b[i]=b[j]; b[j]=t;
				}
			}
	for (int i=1;i<=n;i++) s[i]=s[i-1]+a[i];
	g[1]=b[1]; f[1]=b[1];
	for (int i=2;i<=n;i++)
	{
		g[i]=f[i-1]+b[i]; t=g[i];
		for (int j=1;j<=i;j++)
			if (g[j]+s[i]-s[j]-a[j]*(i-j)<t) t=g[j]+s[i]-s[j]-a[j]*(i-j);
		f[i]=t;
	}
	printf("%d\n",f[n]);
}
