#include<cstdio>
#include<cmath>
const double eps=1e-10;
double a,b,c,T1,T2,X1,X2,X3,Y1,Y2,Y3,p1,p2,q1,q2,q3,q4,l,r,L1,L2;
double dis(double X1,double Y1,double X2,double Y2)
{
	return sqrt((X1-X2)*(X1-X2)+(Y1-Y2)*(Y1-Y2));
}

double Work(double p)
{
	double l,r,x,y,mid,d,Tx,Ty,L1,L2;
	Tx=X2+(X3-X2)/a*p,Ty=Y2+(Y3-Y2)/a*p;
	d=dis(Tx,Ty,X1,Y1);
	if(d+dis(Tx,Ty,X3,Y3)-eps<T1&&d+dis(Tx,Ty,X2,Y2)-eps<T2)
	{
		L1=T1-dis(Tx,Ty,X3,Y3),L2=T2-dis(Tx,Ty,X2,Y2);
		if(L1<L2) return L1;
		return L2;
	}
	for(l=0,r=d;l+eps<r;)
	{
		mid=(l+r)/2;
		x=X1+(Tx-X1)/d*mid,y=Y1+(Ty-Y1)/d*mid;
		if(mid+dis(x,y,X3,Y3)-eps>T1||mid+dis(x,y,X2,Y2)-eps>T2) r=mid;
		else l=mid;
	}
	return l;
}
int main()
{
	scanf("%lf%lf",&T1,&T2);
	scanf("%lf%lf",&X1,&Y1);
	scanf("%lf%lf",&X2,&Y2);
	scanf("%lf%lf",&X3,&Y3);
	a=dis(X2,Y2,X3,Y3),b=dis(X1,Y1,X3,Y3),c=dis(X1,Y1,X2,Y2);
	if(a+b-c-eps<T2)
	{
		L1=a+b+T1;
		L2=c+T2;
		if(L1<L2) printf("%.10lf\n",L1);
		else printf("%.10lf\n",L2);
		return 0;
	}
	T1+=b,T2+=c;
	for(l=0,r=a;l+eps<r;)
	{
		p1=l+(r-l)/3;
		p2=p1+(r-l)/3;
		if(Work(p1)<Work(p2)) l=p1;
		else r=p2;
	}
	printf("%.10lf\n",Work(l));
	return 0;
}
