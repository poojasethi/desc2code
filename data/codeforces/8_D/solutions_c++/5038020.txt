#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#include <cmath>
using namespace std;

const double eps = 1e-9;

int t1, t2, cx, cy, hx, hy, sx, sy;

int sqr(int x)
{
	return x*x;
}
double Area(double a, double b, double c)
{
	double p = (a+b+c)/2;
	return sqrt(abs(p*(p-a)*(p-b)*(p-c)));
}
double Calc0()
{
	double l0 = sqrt((double)(sqr(hx-sx)+sqr(hy-sy)));
	double l1 = sqrt((double)(sqr(cx-hx)+sqr(cy-hy)));
	double l2 = sqrt((double)(sqr(cx-sx)+sqr(cy-sy)));
	if (l1+t1+l0 <= l2+t2+eps) return l1+t1;
	if (l2+t2+l0 <= l1+t1+eps) return l2+t2;
	double coss = (sqr(cx-hx)+sqr(cy-hy)+sqr(hx-sx)+sqr(hy-sy)-sqr(cx-sx)-sqr(cy-sy))/(2*l0*l1);
	double x = (l1+t1+l0-l2-t2)/2;
	double d = sqrt(x*x+sqr(cx-hx)+sqr(cy-hy)-2*coss*x*l1);
	if (d+x <= l1+t1+eps) return l1+t1-x;
	double lt, rt, mid;
	double area = Area(l0, l1, l2);
	for (lt = 0.5*max(t1, t2), rt = d; lt+eps < rt; ) {
		mid = (lt+rt)/2;
		double e1 = l1+t1-mid, e2 = l2+t2-mid;
		if (e1+e2 <= l0+eps) {
			rt = mid;
			continue;
		}
		if (Area(e1, e2, l0)+Area(l1, mid, e1)+Area(l2, mid, e2) < area)
			rt = mid;
		else
			lt = mid;
	}
	return lt;
}
double Calc1()
{
	double l0 = sqrt((double)(sqr(hx-sx)+sqr(hy-sy)));
	double l1 = sqrt((double)(sqr(cx-hx)+sqr(cy-hy)));
	double l2 = sqrt((double)(sqr(cx-sx)+sqr(cy-sy)));
	if (l2+l0 <= l1+t1+eps)
		return min(l1+t1, l2+l0+t2);
	else
		return -1e+9;
		
}
int main()
{
	//freopen("input.txt", "r", stdin);
	scanf("%d%d%d%d%d%d%d%d", &t2, &t1, &cx, &cy, &hx, &hy, &sx, &sy);
	printf("%.8lf\n", max(Calc0(), Calc1()));
	return 0;
}
