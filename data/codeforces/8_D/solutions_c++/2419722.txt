#include<cmath>
#include<cstdio>
#define fo(i,a,b) for(i=a;i<=b;++i)
const double pi=acos(-1.0),eps=1e-10;
double t1,t2,X1,X2,X3,Y1,Y2,Y3,xp1,xp2,yp1,yp2,ans=0.0;
bool sol;
double min(double a,double b){return a<b?a:b;}
double max(double a,double b){return a>b?a:b;}
double hyp(double x,double y){return sqrt(x*x+y*y);}
double solve(double X1,double Y1,double X2,double Y2,double xp,double yp,double t)
{
    if(hyp(X1-xp,Y1-yp)+hyp(X2-xp,Y2-yp)-hyp(X1-X2,Y1-Y2)<=t+eps)
        return hyp(X1-X2,Y1-Y2)+t-hyp(X2-xp,Y2-yp);
    double lo=0.0,hi=1.0,m,X3,Y3;
    int i;
    fo(i,1,300)
    {
        m=(lo+hi)/2;
        X3=X1-m*(X1-xp),Y3=Y1-m*(Y1-yp);
        if(hyp(X1-X3,Y1-Y3)+hyp(X2-X3,Y2-Y3)-hyp(X1-X2,Y1-Y2)<=t+eps)
            lo=m;
        else hi=m;
    }
    return lo*hyp(X1-xp,Y1-yp);
}
double calc(double x,double y)
{
    double v1=solve(X1,Y1,X2,Y2,x,y,t2),v2=solve(X1,Y1,X3,Y3,x,y,t1);
    return min(v1,v2);
}
int main()
{
    scanf("%lf%lf%lf%lf%lf%lf%lf%lf",&t1,&t2,&X1,&Y1,&X2,&Y2,&X3,&Y3);
    double v1=calc(X2,Y2),v2=calc(X3,Y3);
    ans=max(ans,max(v1,v2));
    if(hypot(X1-X3,Y1-Y3)+hypot(X2-X3,Y2-Y3)-hypot(X1-X2,Y1-Y2)<=t2)
        ans=max(ans,min(hypot(X1-X2,Y1-Y2)+t2,hypot(X1-X3,Y1-Y3)+hypot(X2-X3,Y2-Y3)+t1));
    double lo=0.0,hi=1.0,m1,m2;
    int i;
    fo(i,1,300)
    {
        m1=(lo*2+hi)/3,m2=(lo+hi*2)/3;
        xp1=X2-m1*(X2-X3),yp1=Y2-m1*(Y2-Y3);
        xp2=X2-m2*(X2-X3),yp2=Y2-m2*(Y2-Y3);
        v1=calc(xp1,yp1),v2=calc(xp2,yp2);
        ans=max(ans,max(v1,v2));
        if(v1>v2)hi=m2;else lo=m1;
    }
    printf("%.8lf\n",ans);
    return 0;
}
