import sys

#sys.stdin = open("in.txt", "r")

n = input()
a = map(int, raw_input().split())
b = map(int, raw_input().split())
c = map(int, raw_input().split())

f1, f2 = [0 for i in xrange(n)], [0 for i in xrange(n)]
f1[0] = a[0]
f2[0] = b[0]
for i in xrange(1, n):
    f1[i] = max(a[i] + f2[i-1], b[i] + f1[i-1])
    f2[i] = max(c[i] + f1[i-1], b[i] + f2[i-1])
print f1[n-1]
