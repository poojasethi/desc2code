def main():
    n = int(raw_input())
    a = map(int, raw_input().split())
    b = map(int, raw_input().split())
    c = map(int, raw_input().split())
    if n == 1:
        print a[0]
        return
    elif n == 2:
        print max(a[0] + b[1], a[1] + b[0])
        return
    dp = [a[0] + b[1], a[1] + b[0]]
    for i in xrange(2, n):
        ndp = [max(dp) + b[i], a[i] + max(dp[0] - b[i-1] + c[i-1], dp[1] - a[i-1] + b[i-1])]
        dp = ndp
    print max(dp)
main()
