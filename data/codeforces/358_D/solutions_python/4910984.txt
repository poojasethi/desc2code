import sys

n = int(sys.stdin.readline())
a = [int(x) for x in sys.stdin.readline().split(" ")]
b = [int(x) for x in sys.stdin.readline().split(" ")]
c = [int(x) for x in sys.stdin.readline().split(" ")]


hungry = [0] * n 
full = [0] * n
hungry[-1] = -100000

for i in range(n):
    hungry[i] = max(full[i-1]+a[i],hungry[i-1]+b[i])
    full[i] = max(full[i-1]+b[i],hungry[i-1]+c[i])
 
print hungry[-1]
