#!/usr/bin/env python

n = int(raw_input())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
c = map(int, raw_input().split())

f = [[0] * 2 for _ in xrange(n)]
f[0][0] = a[0]
f[0][1] = b[0]
for i in xrange(1, n):
    f[i][0] = max(f[i - 1][0] + b[i], f[i - 1][1] + a[i])
    f[i][1] = max(f[i - 1][0] + c[i], f[i - 1][1] + b[i])
print f[n - 1][0]
