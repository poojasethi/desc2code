import sys

n = int(sys.stdin.readline())
_1st = map(int, sys.stdin.readline().split())
_2nd = map(int, sys.stdin.readline().split())
_3rd = map(int, sys.stdin.readline().split())

a = _1st[0]
b = _2nd[0]

if n == 1:
  print a
  sys.exit(0)

for i in range(1, n-1):
  ai = max(_1st[i] + b, _2nd[i] + a)
  bi = max(_2nd[i] + b, _3rd[i] + a)
  a = ai
  b = bi


print max(b + _1st[n-1], a + _2nd[n-1])
