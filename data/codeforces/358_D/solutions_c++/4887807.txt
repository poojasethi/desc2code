#include<iostream>
#include<algorithm>
using namespace std;

int main(){
    int N;
    cin>>N;
    int a[N],b[N],c[N];
    for (int i=0;i<N;i++) cin>>a[i];
    for (int i=0;i<N;i++) cin>>b[i];
    for (int i=0;i<N;i++) cin>>c[i];
    int r[N];
    int s[N];
    r[0] = a[0];
    s[0] = b[0];
    for (int i=1;i<N;i++){
        r[i] = max(r[i-1]+b[i],s[i-1]+a[i]);
        s[i] = max(r[i-1]+c[i],s[i-1]+b[i]);
    }
    cout<<r[N-1]<<"\n";
}
