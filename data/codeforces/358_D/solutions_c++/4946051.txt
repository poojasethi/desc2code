#include <iostream>
#include <cstdio>

#define Max(a,b) ((a)>(b)?(a):(b))

int dp[4096][4];
int m[4096][4];

int n;

int main(){
    scanf("%d",&n);
    for(int j=0;j<3;++j){
        for(int i=0;i<n;++i){
            scanf("%d",&(m[i][j]));
        }
        dp[0][j]=m[0][j];
    }
    for(int i=1;i<n;++i){
        for(int j=0;j<2;++j){
            dp[i][j]=Max(m[i][j]+dp[i-1][1],m[i][j+1]+dp[i-1][0]);
        }
    }
    printf("%d\n",dp[n-1][0]);
    return 0;
}
