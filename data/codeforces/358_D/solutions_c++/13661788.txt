#include<cstdio>
#include<algorithm>
using namespace std;

int n;
int a[3333][3];
int f[3333], g[3333];

int main() {
	scanf("%d", &n);
	for(int lev = 0; lev < 3; lev ++)
		for(int i=0;i<n;i++)
			scanf("%d", &a[i][lev]);
	
	f[n-1] = a[n-1][0]; g[n-1] = a[n-1][1];
	for(int i=n-2;i>=0;i--) {
		f[i] = max(f[i+1] + a[i][1], g[i+1] + a[i][0]);
		g[i] = max(f[i+1] + a[i][2], g[i+1] + a[i][1]);
	}
	printf("%d\n", f[0]);
	
	return 0;
}
