#include <bits/stdc++.h>
int a[3][3003],n,dp[3003][2];
int main() 
{
	scanf ("%d",&n);
	for(int k=0;k<3;++k)
		for(int i=0;i<n;++i)
			scanf("%d",&a[k][i]);
	for(int k=0;k<2;++k)
		dp[0][k]=a[k][0];
	for(int i=1;i<n;++i)
		for(int k=0;k<2;++k)
			for(int j=0;j< 2;++j) 
				if (dp[i][k]<dp[i-1][j]+a[1+k-j][i])
					dp[i][k]=dp[i-1][j]+a[1+k-j][i];
	printf("%d\n",dp[n-1][0]);
}
