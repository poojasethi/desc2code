#include <iostream>
#include <cstdlib>
#define MAXN 3010
using namespace std;

int N;
int a[MAXN][3];

int din[MAXN][2];

int main(){
	cin >> N;
	for(int j=0;j<3;j++)for(int i=1;i<=N;i++)cin >> a[i][j];
	din[1][0]=a[1][0], din[1][1]=a[1][1];
	for(int n=2;n<=N;n++) {
		din[n][0]=max( din[n-1][0]+a[n][1], din[n-1][1]+a[n][0] );
		din[n][1]=max( din[n-1][0]+a[n][2], din[n-1][1]+a[n][1] );
	}
	cout << din[N][0] << "\n";
}
