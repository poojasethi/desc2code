#include<iostream>
#include<cmath>
using namespace std;

int n,f[3010][2],feed[3010][3];

void init ()
{
	cin>>n;
	for (int j=0;j<=2;j++)
		for (int i=1;i<=n;i++)
			cin>>feed[i][j];
}

void DP ()
{
	f[0][0]=-2147483647;
	for (int i=1;i<=n;i++)
	{
	    f[i][0]= max(f[i-1][0]+feed[i][1],f[i-1][1]+feed[i][0]);
	    f[i][1]= max(f[i-1][0]+feed[i][2],f[i-1][1]+feed[i][1]);
	}
	cout<<f[n][0]<<endl;
}

int main ()
{
	init ();
	DP ();
	return 0;
}
