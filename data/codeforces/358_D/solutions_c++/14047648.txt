#include <iostream>
using namespace std;
int a[5000];
int b[5000];
int c[5000];
int dp[2][5000];
main()
{
    int n;
    cin >> n;
    for(int i=0;i<n;i++)cin >> a[i];
    for(int i=0;i<n;i++)cin >> b[i];
    for(int i=0;i<n;i++)cin >> c[i];
    dp[0][0] = b[0];
    dp[1][0] = a[0];
    for(int i=1; i<n;i++){
        dp[0][i] = max(dp[0][i-1] + b[i], dp[1][i-1] + c[i]);
        dp[1][i] = max(dp[0][i-1] + a[i], dp[1][i-1] + b[i]);
    }
    cout << dp[1][n-1];
}
