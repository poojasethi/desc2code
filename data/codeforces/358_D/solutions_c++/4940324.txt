#include<stdio.h>
#define N 3003
#define max(a,b) a>b?a:b
int n,ins[N][3],dp[N][3]={0};
int main(){
    scanf("%d",&n);
    for(int j=0;j<3;j++) for(int i=1;i<=n;i++) scanf("%d",&ins[i][j]);
    dp[n][0]=ins[n][0];
    dp[n][1]=ins[n][1];
    while(--n){
        //int k=n%2,t=k^1;
        dp[n][0]=max(ins[n][0]+dp[n+1][1],ins[n][1]+dp[n+1][0]);
        dp[n][1]=max(ins[n][1]+dp[n+1][1],ins[n][2]+dp[n+1][0]);
    }
    printf("%d\n",dp[1][0]);
    return 0;
}
