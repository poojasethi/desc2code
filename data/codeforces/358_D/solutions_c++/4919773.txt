#include<cstdio>
#include<algorithm>
int ar[3][3000],dp[3000][2];
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<3;++i)for(int j=0;j<n;++j)
        scanf("%d",ar[i]+j);
    dp[0][0]=ar[0][0],dp[0][1]=ar[1][0];
    for(int i=1;i<n;++i)
    {
        dp[i][0]=max(dp[i-1][0]+ar[1][i],dp[i-1][1]+ar[0][i]);
        dp[i][1]=max(dp[i-1][0]+ar[2][i],dp[i-1][1]+ar[1][i]);
    }
    printf("%d\n",dp[n-1][0]);
}
