#include <cstdio>
#include <iostream>

using namespace std;

/***
	A
***/
int s[3][3333];
int dp[2][3333];
int n;
const int a=0,b=1,c=2;

int main()
{	
	scanf("%d",&n);
	for(int i=0;i<3;i++)
		for(int j=1;j<=n;j++)
			scanf("%d",&s[i][j]);
	dp[a][n]=s[a][n];
	dp[b][n]=s[b][n];
	for(int i=n-1;i>=1;i--)
	{
		dp[a][i]=max(dp[b][i+1]+s[a][i],dp[a][i+1]+s[b][i]);
		dp[b][i]=max(dp[b][i+1]+s[b][i],dp[a][i+1]+s[c][i]);
	}
	printf("%d\n",dp[a][1]);
	return 0;
}
