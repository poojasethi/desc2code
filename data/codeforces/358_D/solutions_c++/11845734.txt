#include<iostream>
using namespace std;
int f[3005][2],a[3005],b[3005],c[3005];
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	for(int i=1;i<=n;i++){
		cin>>b[i];
	}
	for(int i=1;i<=n;i++){
		cin>>c[i];
	}
	f[1][1]=a[1];
	f[1][0]=b[1];
	for(int i=2;i<=n;i++){
		f[i][0]=max(f[i-1][1]+c[i],f[i-1][0]+b[i]);
		f[i][1]=max(f[i-1][1]+b[i],f[i-1][0]+a[i]);
	}
	cout<<f[n][1]<<endl;
}
