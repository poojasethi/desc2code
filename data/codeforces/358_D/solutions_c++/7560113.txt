#include <iostream>
using namespace std;
int n, a[3000], b[3000], c[3000], f[4], g[4], i;
int main()
{
	cin>>n;
	for (i=0; i<n; i++) cin>>a[i];
	for (i=0; i<n; i++) cin>>b[i];
	for (i=0; i<n; i++) cin>>c[i];
	f[0]=f[3]=a[0]; f[1]=f[2]=b[0];
	for (i=1; i<n; i++) {
		g[0]=max(f[1],f[2])+a[i];
		g[1]=max(f[0],f[3])+c[i];
		g[2]=max(f[1],f[2])+b[i];
		g[3]=max(f[0],f[3])+b[i];
		f[0]=g[0]; f[1]=g[1]; f[2]=g[2]; f[3]=g[3];
	}
	cout<<max(f[0],f[3])<<endl;
	return 0;
}
