#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 3333
#define INF 1111111111
using namespace std;
int f[N][2],w[N][3];
int n;
int main()
{
	scanf("%d",&n);
	for (int i=2;i>=0;i--)
	for (int j=1;j<=n;j++)
	scanf("%d",&w[j][i]); 
	f[0][0]=-INF;
	for (int i=1;i<=n;i++)
	{
		f[i][0]=max(f[i-1][0]+w[i][1],f[i-1][1]+w[i][2]);
		f[i][1]=max(f[i-1][0]+w[i][0],f[i-1][1]+w[i][1]);
	}
	printf("%d",f[n][0]);
	return 0;
}
