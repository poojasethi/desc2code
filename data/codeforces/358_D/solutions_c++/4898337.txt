#include <iostream>
using namespace std;

int N, a[3010][3], d[3010][2];

int main(){
	cin >> N;
	for(int j=0;j<3;j++)for(int i=1;i<=N;i++)cin >> a[i][j];
	//din[1][0]=a[1][0], din[1][1]=a[1][1];
	d[0][0]=-100000;
	for(int n=1;n<=N;n++) {
		d[n][0]=max( d[n-1][0]+a[n][1], d[n-1][1]+a[n][0] );
		d[n][1]=max( d[n-1][0]+a[n][2], d[n-1][1]+a[n][1] );
	}
	cout << d[N][0] << "\n";
}
