#include<bits/stdc++.h>

using namespace std;

const int N = 3000 + 10;
int a[N], b[N], c[N], f[N], g[N], n;

int main (){
	cin >> n;
	for (int i = 0; i < n; i ++) cin >> a[i];
	for (int i = 0; i < n; i ++) cin >> b[i];
	for (int i = 0; i < n; i ++) cin >> c[i];

	f[0] = a[0];
	g[0] = b[0];
	for (int i = 1; i < n; i ++){
		f[i] = max(f[i - 1] + b[i], g[i - 1] + a[i]);
		g[i] = max(g[i - 1] + b[i], f[i - 1] + c[i]);
	}
	cout << f[n - 1] << endl;
	return 0;
}
