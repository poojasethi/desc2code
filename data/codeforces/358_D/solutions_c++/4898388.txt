#include <iostream>
using namespace std;
int N, a[3010][3], d[3010][2], i, j;

int main(){
	cin >> N;
	for(;j<3;j++)for(i=0;i<N;i++)cin >> a[i][j];
	d[0][0]=-1e6;
	for(i=0;i<N;i++) {
		d[i+1][0]=max( d[i][0]+a[i][1], d[i][1]+a[i][0] );
		d[i+1][1]=max( d[i][0]+a[i][2], d[i][1]+a[i][1] );
	}
	cout << d[N][0] << "\n";
}
