#include <iostream>

using namespace std;

const int MAXN=10*100*3+100;
int dp[MAXN][2];

main()
{
    int n;
    cin>>n;
    int a[n],b[n],c[n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
		cin>>b[i];
	for(int i=0;i<n;i++)
		cin>>c[i];
    dp[0][0]=a[0];
    dp[0][1]=b[0];
    for(int i=1;i<n;i++){
        dp[i][0]=max(dp[i-1][0]+b[i],dp[i-1][1]+a[i]);
        dp[i][1]=max(dp[i-1][0]+c[i],dp[i-1][1]+b[i]);
    }
    cout<<dp[n-1][0];
    return 0;
}
