#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	int n, a[3005]={0}, b[3005]={0}, c[3005]={0}, dp[3005]={0}, fp[3005]={0};
	cin>>n;
	for(int i=0; i<n; i++) cin>>a[i];
	for(int i=0; i<n; i++) cin>>b[i];
	for(int i=0; i<n; i++) cin>>c[i];
	dp[n-1]=a[n-1];
	fp[n-1]=b[n-1];
	for(int i=n-2; i>=0; i--)
	{
		dp[i] = max(a[i]+fp[i+1], b[i]+dp[i+1]);
		fp[i] = max(b[i]+fp[i+1], c[i]+dp[i+1]);
	}
	cout<<dp[0]<<endl;
	return 0;
}
