#include <cstdio>

int a[4000],b[4000],c[4000];
int n;
int f[3500],g[3500];
int max(int a,int b){
	if( a > b )
		return a;
	else
		return b;
}

int main(){
	scanf("%i",&n);
	for(int i=1;i<=n;++i)
		scanf("%i",&a[i]);
	for(int i=1;i<=n;++i)
		scanf("%i",&b[i]);
	for(int i=1;i<=n;++i)
		scanf("%i",&c[i]);
	
	f[1] = b[1];
	g[1] = a[1];
	for(int i=2;i<=n;++i){
		f[i] = max( f[i-1]+b[i],g[i-1]+c[i] );
		g[i] = max( f[i-1]+a[i],g[i-1]+b[i] );
	}
	printf("%i\n",g[n]);
	return 0;
}
