#include<iostream>

using namespace std;
int data[3001][3];

int dp[3001][4];

int main(){
	int n;
	cin>>n;
	for(int l=0;l<4;l++){
		for(int i=0;i<n;i++){
			if(l!=2)cin>>dp[i][l];
			if(l==2){dp[i][2]=dp[i][1];}
		}
	}
	dp[0][3]=dp[0][1]=0;
	for(int i=1;i<n;i++){
		dp[i][0]+=max(dp[i-1][2],dp[i-1][3]);
		dp[i][1]+=max(dp[i-1][0],dp[i-1][1]);
		dp[i][2]+=max(dp[i-1][2],dp[i-1][3]);
		dp[i][3]+=max(dp[i-1][0],dp[i-1][1]);
	}
	cout<<max(dp[n-1][0],dp[n-1][1])<<endl;	
}
