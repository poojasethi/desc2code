#include <cstdio>
#include <algorithm>

#define N 3000

using namespace std;

int n,a[N+10],b[N+10],c[N+10],f[N+10][2];

int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;++i) scanf("%d",&a[i]);
	for (int i=1;i<=n;++i) scanf("%d",&b[i]);
	for (int i=1;i<=n;++i) scanf("%d",&c[i]);
	f[2][0]=b[1];
	f[2][1]=a[1];
	for (int i=3;i<=n+1;++i) f[i][0]=max(f[i-1][0]+b[i-1],f[i-1][1]+c[i-1]),f[i][1]=max(f[i-1][0]+a[i-1],f[i-1][1]+b[i-1]);
	printf("%d\n",f[n+1][1]);
	return 0;
}
