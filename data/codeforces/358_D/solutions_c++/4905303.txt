#include <stdio.h>
int n,a[3][3030],b[3030][2];
int max(int x,int y){return x<y?y:x;}
main(){
  scanf("%d",&n);
  for(int i=0;i<3;++i)for(int j=1;j<=n;++j)scanf("%d",&a[i][j]);
  b[n][0]=a[0][n],b[n][1]=a[1][n];
  for(int i=n;i-->1;)b[i][0]=max(a[0][i]+b[i+1][1],a[1][i]+b[i+1][0]),b[i][1]=max(a[1][i]+b[i+1][1],a[2][i]+b[i+1][0]);
  printf("%d\n",b[1][0]);
}
