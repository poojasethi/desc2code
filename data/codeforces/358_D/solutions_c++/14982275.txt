#include<bits/stdc++.h>
using namespace std ;
int N ;
int Arr[3][3010] ;
int dp[2][3010] ;
void solve()
{
  dp[0][1] = Arr[0][1] ;
  dp[1][1] = Arr[1][1] ;
  for(int i  = 2;i<=N;i++)
   {
   	  dp[0][i] = max(dp[0][i-1]+Arr[1][i],dp[1][i-1]+Arr[0][i]);
   	  dp[1][i] = max(dp[0][i-1]+Arr[2][i],dp[1][i-1]+Arr[1][i]) ;
   }
}


int main()
{
	cin>>N;
	for(int i = 0;i<3;i++)
	{
		for(int j= 1;j<=N;j++)
		  cin>>Arr[i][j] ;
	}
   solve() ;
  cout<<dp[0][N] ;
}
