#include <iostream>
#include <cstdio>
int f[3010][2],a[3010],b[3010],c[3010],i,n;
int main()
{
	scanf("%d\n",&n);
	for (i=1;i<=n;++i) scanf("%d",&a[i]);scanf("\n");
	for (i=1;i<=n;++i) scanf("%d",&b[i]);scanf("\n");
	for (i=1;i<=n;++i) scanf("%d",&c[i]);scanf("\n");
	f[1][0]=a[1];f[1][1]=b[1];
	for (i=2;i<=n;++i)
	{
		f[i][0]=std::max(f[i-1][0]+b[i],f[i-1][1]+a[i]);
		f[i][1]=std::max(f[i-1][0]+c[i],f[i-1][1]+b[i]);
	}
	printf("%d\n",f[n][0]);
}
