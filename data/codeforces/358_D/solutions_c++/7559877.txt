#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n+1][3],dp[n+1][2];
	for(int i=1;i<=n;i++)cin>>a[i][0];
	for(int i=1;i<=n;i++)cin>>a[i][1];
	for(int i=1;i<=n;i++)cin>>a[i][2];
	dp[1][0]=a[1][0];dp[1][1]=a[1][1]; 
	for(int i=2;i<=n;i++)
	{
		dp[i][0]=max(dp[i-1][0]+a[i][1],dp[i-1][1]+a[i][0]);
		dp[i][1]=max(dp[i-1][0]+a[i][2],dp[i-1][1]+a[i][1]);
	}
	cout<<dp[n][0]<<endl;
}
