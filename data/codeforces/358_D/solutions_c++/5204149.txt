
#include <cstdio>
#include <algorithm>
using std::max;

const int N = 10010;
int dp[2][N];
int d[3][N];

int main()
{
	int n;
	scanf("%d",&n);
	for(int j=0;j<3;++j) for(int i=0;i<n;++i)scanf("%d",d[j]+i);

	dp[0][n-1] = d[0][n-1];
	dp[1][n-1] = d[1][n-1];
	int i = n-1;
	while(i--){
		dp[0][i] = max( d[0][i] + dp[1][i+1], d[1][i]+dp[0][i+1] );	
		dp[1][i] = max( d[1][i] + dp[1][i+1], d[2][i]+dp[0][i+1] );
	}
	printf("%d\n",dp[0][0]);
	
	return 0;	
}
