(n, m) = map(int, raw_input().split())
inv = [0 for i in xrange(n + 5)]
outv = [0 for i in xrange(n + 5)]

for i in xrange(m):
    (a, b, c) = map(int, raw_input().split())
    outv[a] += c
    inv[b] += c

res = 0
for i in xrange(n + 5):
    res += abs(inv[i] - outv[i])

while res % 2 != 0:
    pass

print res / 2
