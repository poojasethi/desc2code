n,m = map(int, raw_input().split())
d = [[0]*n for _ in xrange(n)]

for _ in xrange(m):
	a, b, cost = map(int ,raw_input().split())
	d[a-1][b-1] = cost

updated = True
while(updated):
	updated = False
	for k in xrange(n):
		for i in xrange(n):
			for j in xrange(n):
				if d[i][k] > 0 and d[k][j] > 0:
					m = min(d[i][k], d[k][j])
					d[i][k]-=m
					d[k][j]-=m
					d[i][j]+=m
					updated = True

s = 0
for i in d:
	s+=sum(i)
print s
