inp = raw_input().split()
n, m = int(inp[0]), int(inp[1])
from math import fabs
map = {x : 0 for x in range(1, n + 1)}

for i in range(m):
    t = raw_input().split()
    a, b, c = int(t[0]), int(t[1]), int(t[2])
    map[a] -= c
    map[b] += c

acc = 0
for key in map:
    acc += int(fabs(map[key]))

print acc/2
