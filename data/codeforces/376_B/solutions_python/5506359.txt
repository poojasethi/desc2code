import sys, math
def rs():
    return sys.stdin.readline().strip()
def ri():
    return int(sys.stdin.readline().strip())
def ras():
    return list(sys.stdin.readline().strip())
def rai():
    return map(int,sys.stdin.readline().strip().split())




def solve():
    n,m = rai()

    debt = [0]*n
    for x in xrange(m):
        a,b,c = rai()
        debt[b-1] += c
        debt[a-1] -= c

    return abs(sum(filter(lambda x: x < 0, debt)))


print solve()
