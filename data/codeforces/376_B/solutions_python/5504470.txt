x = raw_input().split()
n = int(x[0])
m = int(x[1])
y = []
for x in range(n):
    y.append(0)
for x in range(m):
    r = raw_input().split()
    src = int(r[0])
    dest = int(r[1])
    amt = int(r[2])
    y[src-1] -= amt
    y[dest-1] += amt
total = 0
for x in y:
    if x > 0:
        total += x
print total
