n, m = map(int, raw_input().split())

debts = [[0, 0] for i in xrange(n)]
total = 0
for i in xrange(m):
    frm, to, amt = map(int, raw_input().split())
    total += amt
    debts[frm - 1][1] += amt
    debts[to - 1][0] += amt

for get, give in debts:
    total -= min(get, give)

print total
