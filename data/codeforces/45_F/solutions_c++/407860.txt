#include <cstdio>
#include <algorithm>

using namespace std;

int ans[4][6] = {
    {},
    {},
    {0, 1, 3, 6},
    {0, 1, 2, 3, 5, 6}
};

int gao(int m, int n) {
    if (n < 4) {
        return m >= 6 ? 0 : ans[n][m];
    }

    int s = 0;
    s += n / 2;
    if (m <= s) {
        return 1;
    }
    s += (n - 1) / 2;
    if (m <= s) {
        return 2;
    }
    s += n / 2 - 1;
    if (m <= s) {
        return 3;
    }
    s += (n - 1) / 2;
    if (m <= s) {
        return 4;
    }
    return 4 + (m - s + n / 2 - 2) / (n / 2 - 1);
}

int main() {
    int m, n;

    /*
    for (int j = 1; j < 64; ++j) {
        for (int i = 1; i < 64; ++i) {
            printf("%d %d %d\n", i, j, gao(i, j) * 2 - 1);
        }
    }
    */
    scanf("%d%d", &m, &n);
    printf("%d\n", gao(m, n) * 2 - 1);

    return 0;
}
