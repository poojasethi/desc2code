#include<cstdio>
using namespace std;
int main(){
    int m,n,ans=0,r=0,k=0;
    scanf("%d%d",&m,&n);
    if(n==1)printf("-1\n");
    else if(n>m&&n<2*m)printf("3\n");
    else if(n>=2*m)printf("1\n");
    else if(n==m)printf("5\n");
    else if(n==2){
        if(m==3)printf("11\n");else printf("-1\n");
    }else if(n==3){
        if(m==4)printf("9\n");else if (m==5)printf("11\n");else printf("-1\n");
    }else if(n>=4){
        ans=4;m=m-n+2;
        if(n>=2*m)ans++;
        else {
            if(m-n>=0){
                k=(m-n)/(n/2-1);ans+=2*k;
                m=m-(n/2-1)*k;
            }
            if(n<2*m){
                ans++;m=2*m-n+1;k=m/n;
                if(m%n!=0)ans=ans+2*(k+1);else ans+=2*k;
            }else ans++;}
        printf("%d\n",ans);
    }
    return 0;
}
