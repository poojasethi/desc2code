#include <iostream>

using namespace std;

int g[10][10];
int n;

int fw(int k, int i, int j){
    g[i][j] = min(g[i][j], g[i][k]+g[k][j]);
    j++;
    i += (j/n);
    k += (i/n);
    j = (j%n);
    i = (i%n);
    return k == n ? 0 : fw(k, i, j);
}

int read(int i, int j){
    cin>>g[i][j];
    j++;
    i += (j/n);
    j = (j%n);
    return i == n ? 0 : read(i, j);
}

int m(int i, int j){
    return i == n ? 0 : max(g[i][j], m((j+1) == n ? i+1 : i, (j+1) == n ? 0 : j+1));
}

int main()
{
    cin>>n;
    read(0, 0);
    fw(0, 0, 0);
    cout<<m(0, 0)<<endl;
    return 0;
}
