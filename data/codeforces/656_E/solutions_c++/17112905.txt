#include <bits/stdc++.h>
using namespace std;
int n,in[12][12],ans=0;
void fool(){;}
void input2(int a,int b)
{
    b?(scanf("%d",in[a]+(n+1-b)),input2(a,b-1)):fool();
}
void input(int a)
{
    a?(input2(n+1-a,n),input(a-1)):fool();
}
void update(int a,int b)
{
    ans=max(ans,in[a][b]);
}
void _slack(int x,int a,int b)
{
    in[a][b]=min(in[a][b],in[a][x]+in[x][b]);
}
void slake (int x,int a,int b)
{
    (x!=a&&x!=b&&a!=b)?_slack(x,a,b):fool();
}
void gao3(int x,int a,int b)
{
    (!b)?fool():(slake(x,a,b),gao3(x,a,b-1));
}
void gao2(int x,int a)
{
    (!a)?fool():(gao3(x,a,n),gao2(x,a-1));
}
void gao(int x)
{
    (!x)?fool():(gao2(x,n),gao(x-1));
}
void getans2(int a,int b)
{
    b?(update(a,b),getans2(a,b-1)):fool();
}
void getans(int a)
{
    a?(getans2(a,n),getans(a-1)):fool();
}
int main()
{
    scanf("%d",&n);
    input(n);
    gao(n);
    getans(n);
    printf("%d\n",ans);
    return 0;
}
