#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int n;
int a[11][11];
int mjy;
int read(int x,int y)
{
	scanf("%d",&a[x][y]);
	y++;
	y>n?x++,y=1:0;
	x<=n?read(x,y):0;
}
int floyd(int k,int i,int j)
{
	a[i][j]=min(a[i][j],a[i][k]+a[k][j]);
	j++;
	j>n?i++,j=1:0;
	i>n?k++,i=1:0;
	k<=n?floyd(k,i,j):0;
}
int getmjy(int x,int y)
{
	mjy=max(mjy,a[x][y]);
	y++;
	y>n?x++,y=1:0;
	x<=n?getmjy(x,y):0;
}
int main()
{
	scanf("%d",&n);
	read(1,1);
	floyd(1,1,1);
	getmjy(1,1);
	printf("%d\n",mjy);
	return 0;
}
