using namespace std;

int n;
int a[11][11];
int ans;

int woc2(int x,int y){
	scanf("%d",&a[x][y]);
	return y==n?0:woc2(x,y+1);
}
int woc1(int x){
	woc2(x,1);
	return x==n?0:woc1(x+1);
}

int floyd3(int k,int x,int y){
	a[x][y]=min(a[x][y],a[x][k]+a[k][y]);
	return y==n?0:floyd3(k,x,y+1);
}

int floyd2(int x,int y){
	floyd3(x,y,1);
	return y==n?0:floyd2(x,y+1);
}

int floyd1(int x){
	floyd2(x,1);
	return x==n?0:floyd1(x+1);
}

int getans2(int x,int y){
	ans=max(ans,a[x][y]);
	return y==n?0:getans2(x,y+1);
}
int getans1(int x){
	getans2(x,1);
	return x==n?0:getans1(x+1);
}

int main()
{
	scanf("%d",&n);
	woc1(1);
	floyd1(1);
	getans1(1);
	printf("%d\n",ans);
	return 0;
}
