const int maxn = 1000005;
const int MOD = 1e9 + 7;
int a[15][15];int n, ans = 0;
void floyd(){
    F (int k=0; k<n; k++)
        F (int i=0; i<n; i++)
            F (int j=0; j<n; j++)
                a[i][j] = min(a[i][j], a[i][k] + a[k][j]);
}
int main()
{
    scanf("%d", &n);
    F (int i=0; i<n; i++)
        F (int j=0; j<n; j++)
            scanf("%d", &a[i][j]);
    floyd();
    F (int i=0; i<n; i++)
        F (int j=0; j<n; j++)
            ans = max(ans, a[i][j]);
    printf("%d\n", ans);
	return 0;
}
