#include<bits\stdc++.h>
using namespace std;
int a[15][15];
int n;
int ans;
int flod3( int i, int j, int k){
	a[j][k] = min( a[j][k], a[j][i]+a[i][k] );
	k<n-1 ? flod3( i, j, k+1 ): 1;
	return 1;
}
int flod2( int i, int j ){
	j<n ? flod3( i, j, 0 ): 1;
	j<n ? flod2( i, j+1 ): 1;
	return 1;
}
int flod1( int i ){
	i<n ? flod2( i, 0 ): 1;
	i<n ? flod1( i+1 ): 1;
	return 1;
}
int scan2(int i, int j){
	cin>>a[i][j];
	j<n-1 ? scan2(i,j+1) : 1;
	return 1;
}
int scan(int i){
	i<n ? scan2(i,0) : 1;
	i<n ? scan(i+1): 1;
	return 1;
}
int getans2(int i, int j){
	ans = max( ans, a[i][j] );
	j<n-1 ? getans2(i,j+1) : 1;
	return 1;
}
int getans(int i){
	i<n ? getans2(i,i+1) : 1;
	i<n ? getans(i+1): 1;
	return 1;
}
int main(){
	cin >> n;
	ans = 0;
	scan(0);
	flod1(0);
	getans(0);
	cout << ans << endl;
	return 0;
}
