#include <iostream>

using namespace std;

int n;
int a[10][10];

bool read(int i,int j)
{
	j!=n||((++i)&(j=0));
    i==n||((cin>>a[i][j])&&(read(i,j+1)));
}

bool recalc(int i,int j,int k)
{
	a[j][k]=a[j][k]>a[j][i]+a[i][k]?a[j][i]+a[i][k]:a[j][k];
}

bool floyd(int i,int j,int k)
{
	k!=n||((++j)&&(k=0));
	j!=n||((++i)&&(j=0));
	(i==n)||(recalc(i,j,k)&floyd(i,j,k+1));	
}

int res=0;

bool getMax(int i,int j)
{
	j!=n||((++i)&(j=0));
    i==n||((res=res>a[i][j]?res:a[i][j])&(getMax(i,j+1)));
}

int main()
{
	cin>>n;
	int i=0;int j=0;
	j==n;
	read(0,0);
	floyd(0,0,0);
	getMax(0,0);
	cout<<res;
}
