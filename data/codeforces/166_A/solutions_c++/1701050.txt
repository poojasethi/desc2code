#include <iostream>
#include <algorithm>

using namespace std;

int main(){
	int p, t, f[50], i, n, k;
	cin >> n >> k;
	for(i = 0; i < n; i++) {
		cin >> p >> t;
		f[i] = p * 50 - t;
	}
	
	int ans = 0;
	sort(f,f + n);
	for(i = 0; i < n; i++) {
		if(f[i] == f[n-k]) {
			++ans;
		}
	}
	cout << ans << endl;
}
