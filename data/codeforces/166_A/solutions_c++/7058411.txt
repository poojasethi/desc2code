#include<bits/stdc++.h>
using namespace std;
int a[120],b[10020];
int main()
{
    int n,k,p,t;
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;i++)
    {
        scanf("%d%d",&p,&t);
        a[i]=p*100-t+50;
        b[a[i]]++;
    }
    sort(a+1,a+n+1);
    printf("%d\n",b[a[n-k+1]]);
    return 0;
}
