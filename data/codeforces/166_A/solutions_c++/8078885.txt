#include<iostream>
#include<algorithm>

using namespace std ;

int n , k ;
pair<int,int> a[100] ;

int main()
{
    cin >> n >> k ;
    for(int i=0 ; i<n ; i++)
    {
        cin >> a[i].first >> a[i].second ;
        a[i].first*=-1 ;
    }
    sort(a,a+n) ;
    cout << count(a,a+n,a[k-1]) << endl ;
    return 0 ;
}
