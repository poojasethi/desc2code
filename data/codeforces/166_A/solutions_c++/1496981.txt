#include<iostream>
#include<algorithm>
using namespace std;
pair<int,int>a[50];
main()

{
    int n,k;
    cin>>n>>k;
    
    for(int i=0;i<n;i++)
    {
        cin>>a[i].first>>a[i].second;
        a[i].first= a[i].first*-1;
    }
    sort(a,a+n);
    int p=count(a,a+n,a[k-1]);
    cout<<p;
    
}
