#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
	int a, b, f[50], i, n, k;
	cin >> n >> k;
	for(i = 0; i < n; i++) {
		cin >> a >> b;
		f[i] = a * 50 - b;
	}
	int ans = 0;
	sort(f,f + n);
	for(i = 0; i < n; i++) {
		if(f[i] == f[n-k]) {
			++ans;
		}
	}
	cout << ans << endl;
}
