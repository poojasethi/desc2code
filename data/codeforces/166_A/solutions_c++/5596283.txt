#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
    int n, k, Arr[50], i, res=0, a, b;

    cin>>n>>k;

    for(i=0; i<n; i++){
        cin>>a>>b;Arr[i]=a*50-b;}

    sort(Arr,Arr+n);
    for(i=0; i<n; i++)

            if(Arr[i]==Arr[n-k]) res++;

    cout<<res;

    return 0;
}
