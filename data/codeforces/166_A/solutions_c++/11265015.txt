#include <bits/stdc++.h>

int n , k , b[60];
std::pair <int,int> a[60];

int main(){
	std::ios::sync_with_stdio(false);
	std::cin >> n >> k;
	for (int i = 0 ; i < n ; ++ i){
		std::cin >> a[i].first >> a[i].second;
		a[i].first *= -1;
	}
	sort(a , a+n);
	return std::cout << count(a,a+n,a[k-1]) , 0;
}
