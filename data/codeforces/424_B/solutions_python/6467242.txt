n,s=map(int,raw_input().split())
l=[]
total=0
for i in range(n):
    x,y,k=map(int,raw_input().split())
    total+=k
    #l.append([x,y,k])
    d=((x)**2 + (y)**2)**0.5
    l.append([d,k])
if (s+total)<1000000:
    print "-1"
else:
    l.sort()
    ans=0
    j=0
    while s<1000000:
        s+=l[j][1]
        ans=l[j][0]
        j+=1
    print ans
