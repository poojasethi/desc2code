import sys
import math
import operator
file = sys.stdin

n, s = file.readline().split(' ')
n = int(n)
s = int(s)

info = [0]*n
for i in range(0, n):
    x, y, k = file.readline().split(' ')
    x = int(x)
    y = int(y)
    k = int(k)
    info[i] = [math.sqrt(x*x+y*y), k]

info.sort(key=operator.itemgetter(0))

big = 1000000
maxR = 0
for v1, v2 in info:
    s += v2
    if s >= big:
        maxR = v1
        break

if s < big:
    print(-1)
else:
    print("%.7f" % maxR)
