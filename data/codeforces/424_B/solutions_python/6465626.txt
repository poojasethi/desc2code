from math import *
n,s = map(int,raw_input().split())
A = [map(int,raw_input().split()) for _ in xrange(n)]
d = dict()
for x,y,z in A:
    d[(x,y)] = z
#print d
k = d.keys()
kk = sorted(k, key= lambda x:  x[0]**2+x[1]**2)
for x,y in kk:
    s += d[(x,y)]
    if s >= 1000000:
        print sqrt(x**2+y**2)
        exit(0)
print -1
