n,s=map(int,raw_input().split())
l=[]
for i in xrange(n):
    a=map(int,raw_input().split())
    a=[a[0]*a[0] + a[1]*a[1]]+a
    l+=[a]
#print l
l=sorted(l)
cx=0
cy=0
t=10**6
i=0
while (s<t and i<len(l)):
    s+=l[i][3]
    cx=l[i][1]
    cy=l[i][2]
    i+=1
if (s>=t):
    print (cx*cx+cy*cy)**0.5
else:
    print -1
