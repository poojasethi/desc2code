from math import sqrt

def main():
	n, s = map(int, raw_input().split())
	city = []
	for i in xrange(n):
		x, y, k = map(int, raw_input().split())
		city.append((x ** 2 + y ** 2, k))
	city.sort()
	for i in xrange(n):
		s += city[i][1]
		if s >= 10 ** 6:
			print sqrt(city[i][0])
			return
	print -1

if __name__ == "__main__": main()
