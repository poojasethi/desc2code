from math import sqrt


citynumber,population = [int(x) for x in raw_input().split() ]

i = citynumber

citylist = []

while(i>0):
	x,y,pop = [int(x) for x in raw_input().split()]
	r = sqrt(x**2 + y**2)
	citylist.append([r,pop])
	i -= 1

citylist.sort(cmp=lambda x,y: cmp(x[0],y[0]))

i = 0

while((i<citynumber) and (population<1000000)):
	population += citylist[i][1]
	r = citylist[i][0]
	i += 1

if population >= 1000000 :
	print r
else :
	print -1
