import sys, math

n, s = map(int, raw_input().split())
cities = []
for i in range(n):
    x, y, cnt = map(int, raw_input().split())
    cities.append((x * x + y * y, cnt))

cities.sort()

for i in range(n):
    s += cities[i][1]
    if s >= 1000000:
        print "%.8f\n" % math.sqrt(cities[i][0])
        break
else:
    print '-1'
