n, s = [int(x) for x in raw_input().split(' ')]

    locs.append([x, y, k, sqrt(x * x + y * y)])

locs = sorted(locs, key=lambda i: i[3])
dist = 0.0

    s += l[2]

    if s >= 1e6:
