import sys
import math
lines = sys.stdin.readlines()

def calc(x,y):
  return math.sqrt(x*x + y*y)

n, k = map(int, lines[0].strip().split())
tmp_p = k
distancias = {}
for i in range(n):
  x, y, p = map(int, lines[i+1].strip().split())
  distancia = calc(x,y)
  if distancia not in distancias:
    distancias[distancia] = []
  distancias[distancia].append(p)
  tmp_p += p

if tmp_p < 1000000:
  print -1
else:
  pob = k
  for d in sorted(distancias):
    res = d
    for p in distancias[d]:
      pob += p 
    if pob >= 1000000:
      break
  print res
