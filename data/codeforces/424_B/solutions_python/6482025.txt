import math
n, s = map(int, raw_input().split())

l = []

for i in range(n):
	x, y, k = map(int, raw_input().split())
	x, y = float(x), float(y)
	d = math.sqrt((x**2) + (y**2))
	l.append((d, x, y, k))

l.sort()
r = 0
curr = s

for d, x, y, k in l:
	curr += k
	r = d	
	if curr >= 1000000:
		print r
		break

if curr < 1000000:
	print -1
