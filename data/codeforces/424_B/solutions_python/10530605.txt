def radius(x, y):
  return (x * x + y * y) ** 0.5

n, k = (int(i) for i in raw_input().strip().split(' '))
d = [(0, k)]
for i in xrange(n):
  x, y, p = (int(i) for i in raw_input().strip().split(' '))
  d.append((radius(x, y), p))

d.sort()
now = 0
M = 1000000
findAns = False
for a in d:
  if now + a[1] >= M:
    print "%.7f" % a[0]
    findAns = True
    break
  now += a[1]

if not findAns:
  print -1
