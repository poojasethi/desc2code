import math
dist = []

n, s = raw_input().split()
n = int(n)
s = float(s)
for i in xrange(n):
	x, y, k = [int(v) for v in raw_input().split()]
	d = math.sqrt(x*x + y*y)
	dist.append((k, d))

dist = sorted(dist, key=lambda x: x[1])

K = 1000000.0 - s
k = 0

if K == 0:
	print 0
else:
	for kd in dist:
		k += kd[0]
		if k >= K:
			print kd[1]
			break
	else:
		print -1
