n,s=map(int,raw_input().split())
cnt=0
m=n
x=[0]*n
while n>0:
    a,b,c=map(float,raw_input().split())
    d=float((b*b+a*a))**0.5
    x[cnt]=(d,c)
    n=n-1
    cnt=cnt+1
x=sorted(x)
for i in range(0,m):
    s=s+x[i][1]
    if s>=1000000:
        print x[i][0]
        break
if s<1000000:
    print "-1"
