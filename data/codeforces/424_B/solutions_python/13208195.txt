from math import sqrt
ns = [int(i) for i in raw_input().split()]
pops = []
for i in range(ns[0]):
    xyk = [int(j) for j in raw_input().split()]
    pops.append([sqrt(xyk[0]*xyk[0] + xyk[1]*xyk[1]),xyk[2]])
pops.sort()
for i in range(ns[0]):
    ns[1] += pops[i][1]
    if ns[1]>=10**6:
        break
if ns[1]>=10**6:
    print pops[i][0]
else:
    print -1
