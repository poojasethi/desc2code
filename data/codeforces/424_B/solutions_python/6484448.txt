from math import *
s = [int(k) for k in raw_input().split(" ")]
n = s[0]
p = s[1]
points = []
for i in range(n):
    line = [int(k) for k in raw_input().split(" ")]
    points += [[sqrt(line[0]**2 + line[1]**2), line[2]]]

points.sort()

result = 0
for i in range(n):
    if p >= 1000000:
        break
    result = points[i][0]
    p += points[i][1]

if (p >= 1000000):
    print result
else:
    print -1
