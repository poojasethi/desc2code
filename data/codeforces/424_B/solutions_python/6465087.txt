n, s = map(int, raw_input().strip().split())

loc = []

for i in xrange(n):
    a, b, c = map(int, raw_input().strip().split())
    loc.append((a, b, c))

loc.sort(key=lambda x: x[0]**2 + x[1]**2)

ans = 0
for x in loc:
    if s >= 1000000: break
    ans = (x[0]**2 + x[1]**2)**.5
    s += x[2]

if s >= 1000000:
    print ans
else:
    print -1
