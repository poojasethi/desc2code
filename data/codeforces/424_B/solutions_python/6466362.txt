from math import hypot

mega_city = 10 ** 6
n, s = map(int, raw_input().split())
dist = []
for i in range(n):
    x, y, k = map(int, raw_input().split())
    dist.append((hypot(x, y), k))

dist.sort()
for i in xrange(n):
    s += dist[i][1]
    if s >= mega_city:
        print '%.7f' % dist[i][0]
        break
else:
    print -1
