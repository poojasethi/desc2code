#include<bits/stdc++.h>
using namespace std;
map<double,int>arr;

int main()
{
int n,m,a,b,c;
cin>>n>>m;
for(int i=0;i<n;++i)
{
    cin>>a>>b>>c;
    arr[sqrt(pow(a,2.0) + pow(b,2.0))]+=c;
 }double mn=1000000000;
 for(map<double,int>::iterator it=arr.begin();it!=arr.end();++it){
 if(m+it->second>=1000000){mn=it->first;break;}
 m+=it->second;
 
 }
 if(mn==1000000000)cout << "-1" << endl;
 else printf("%.6lf",mn);
return 0;
}
