#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
pair<int,int>a[1001];
int n, x, y, s;
int main(){
    scanf("%d %d", &n, &s);
    for(int i = 0; i < n; i ++){
        cin>>x>>y>>a[i].second;
        a[i].first = x*x + y*y;
    }
   sort(a,a+n);
   for(int i = 0; i < n; i ++){
    s += a[i].second;
    if(s >= 1000000){
        printf("%.6lf",sqrt(a[i].first + 0.0000000));
        return 0;
    }
   }
   puts("-1");
    return 0;
}
