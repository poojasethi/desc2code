#include<iostream>
#include<iomanip>
#include<algorithm>
#include<vector>
#include<cmath>
using namespace std;

long long n, s, x, y, z;
vector<pair<long long, int> > v;

int main(){
  cin >> n >> s;
  for(int i = 0; i < n; i++)
    cin >> x >> y >> z, v.push_back(make_pair(x * x + y * y, z));
  sort(v.begin(), v.end());
  s = 1000000 - s;
  for(int i = 0; i < n; i++){
    s -= v[i].second;
    if(s <= 0){
      cout << setprecision(8) << sqrt(v[i].first);
      return 0;
    }
  }
  cout << -1;
  return 0;
}
