#include <bits/stdc++.h>
#define MAX 1000000
using namespace std;

int main() {
    int n,s,sum=0,x,y,k;
    
    cin>>n>>s;
    vector< pair<int , int > > v;
    for(int i=0;i<n;i++)
    {
        cin>>x>>y>>k;
        pair <int,int> p(x*x+y*y,k);
        v.push_back(p);
    }
    sort(v.begin(),v.end());
    int i;
    for(i=0;i<n;i++)
    {
        sum+=(v[i].second);
        if(sum>=MAX-s)
        break;
    }
    if(i==n)
    cout<<"-1"<<endl;
    else
    printf("%.7f\n",sqrt(v[i].first));
    return 0;
}
