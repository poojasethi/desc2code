#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,m;
    cin>>n>>m;
    int a,b,i,c;
    vector <pair <double,int> > v;
    for(i=0;i<n;++i)
    {
        cin>>a>>b>>c;
        v.push_back(make_pair(sqrt(a*a+b*b),c));
    }
    sort(v.begin(),v.end());
    c=m;
    for(i=0;i<n;++i)
    {
        c+=v[i].second;
        if(c>=1000000)
        {
            cout<<setprecision(10)<<v[i].first;
            return 0;
        }
    }
    cout<<"-1";
    return 0;
}
