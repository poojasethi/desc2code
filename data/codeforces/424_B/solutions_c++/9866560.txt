#include<bits/stdc++.h>
using namespace std;
int main()
{
	double i,n,s,x,y,p,sum=0,dist,ans;
	vector < pair <double,double> >v;
	cin>>n>>s;
	for(i=0;i<n;i++)
	{
		cin>>x>>y>>p;
		sum+=p;
		dist=sqrt(x*x+y*y);
		v.push_back(make_pair(dist,p));
	}
	if(s+sum<1000000)
	{
		cout<<-1<<endl;
		return 0;
	}
	sort(v.begin(),v.end());
	for(i=0;i<n;i++)
	{
		if(s<1000000)
		{
			s=s+v[i].second;
			ans=v[i].first;
		}

	}
	printf("%.7lf\n",ans);;
	return 0;
}
