#include<bits/stdc++.h>
using namespace std;
typedef pair<double ,int>p;
int main()
{
int n,s;
cin>>n>>s;
double a,b;
int num;
vector<p> v;int f=1;
for(int i=0;i<n;i++)
{
cin>>a>>b>>num;
double val = sqrt(a*a+b*b);
v.push_back(p(val,num));
}
sort(v.begin(),v.end());
int ctr=0;
while(s<1000000)
{if(ctr==n){f=0;break;}
s+=v[ctr].second;
ctr++;

}//cout<<ctr;
ctr--;
if(f)printf("%0.9lf",v[ctr].first);
else
printf("-1");
return 0;
}
