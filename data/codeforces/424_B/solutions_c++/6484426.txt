#include <stdio.h>
#include <iostream>
#include <map>
#include <cmath>
using namespace std;
map <int,int> mp;
int n,m,x,y,z,v=1000000000;
int main( ){
	cin>>n>>m;
	for (int i=1;i<=n;i++){
		cin>>x>>y>>z;
		mp[x*x+y*y]+=z;
	}
	for (map<int,int>::iterator it=mp.begin();it!=mp.end();it++){
		if (m+it->second>=1000000){
			v=it->first;
			break;
		}
		m+=it->second;
	}
	if (v==1000000000) cout<<-1<<endl; else printf("%.7lf",(double)sqrt(v));
	return 0;
}
