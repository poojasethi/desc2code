#include<iostream>
#include<map>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
    int n,p;
    int x,y,k;
    map<int,int> m;
    cin>>n>>p;
    for(int i=0;i<n;i++)
    {
        cin>>x>>y>>k;
        m[x*x + y*y] +=k;
    }
    for(map<int,int>::iterator it = m.begin();it!=m.end();it++)
    {
        if(it->second + p>= 1000000)
        {
            printf("%.6f\n",sqrt(it->first));
            return 0;
        }
        p+=it->second;

    }
    cout<<-1<<endl;

    return 0;
}
