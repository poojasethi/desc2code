#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

pair <double, int> p[1004];

int main() {
	int n, s;
	cin >> n >> s;

	for (int i = 0; i < n; i++) {
		int x, y;
		cin >> x >> y >> p[i].second;
		p[i].first = sqrt(x*x + y*y);
	}

	sort(p, p+n);
	int i = 0;
	for ( ; i < n; i++) {
		s += p[i].second;
		if (s >= 1000000) break;
	}
	if (s < 1000000) cout << "-1";
	else cout <<fixed<<setprecision(7) <<  p[i].first;

	return 0;
}
