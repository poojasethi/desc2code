#include<bits/stdc++.h>
using namespace std;
#define MAX 1001
struct point{
	double dis;
	int num;
};
bool compare(point p1,point p2){
	return p1.dis<p2.dis;
}
point p[MAX];
int main(){
	int n,t,i;
	long long x,y;
	int k;
	double d,r;
	scanf("%d%d",&n,&t);
	for(i=0;i<n;i++){
		cin>>x>>y>>k;
		p[i].num=k;
		d=x*x+y*y;
		d=d*1.;
		d=sqrt(d);
		p[i].dis=d;	
	}
	int cnt=1000000;
	sort(p,p+n,compare);
	for(i=0;i<n;i++)
	if(t<cnt)
	{
		r=p[i].dis;
		t+=p[i].num;
	}
	if(t>=cnt)
	printf("%.07lf\n",r);
	else
	printf("-1\n");
	return 0;
}
