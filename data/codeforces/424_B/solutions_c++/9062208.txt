#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<cmath>
using namespace std;
pair<double,int> a[1005];
long n ,x,y, s;
int main()
{
    cin>>n>>s;
    for(int i=1; i<=n; i++)
    {
        scanf("%d%d%d",&x,&y, &a[i].second );
        a[i].first=sqrt(x*x+y*y);
    }
    sort(a+1, a+n+1);
    int i=0;
    while(s<1000000 && i<n)
    {
        i++;
        s+=a[i].second;
    }
    if(s>=1000000)
    {
         printf("%.7f", a[i].first);
    }
    else cout<<-1;

}
