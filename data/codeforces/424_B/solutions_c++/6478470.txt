#include<cstring>
#include<cstdio>
#include<algorithm>
#include<cmath>
#define N 1005
#define M 1000000
using namespace std;
struct jc{double x,y;int k;} p[N];
double ans;int n,s;
bool cmp(jc a,jc b) {return a.x*a.x+a.y*a.y<b.x*b.x+b.y*b.y;}
int main() {
	scanf("%d%d",&n,&s);
	for (int i=1;i<=n;++i) scanf("%lf%lf%d",&p[i].x,&p[i].y,&p[i].k);
	sort(p+1,p+n+1,cmp);
	for (int i=1;i<=n;++i) {
		s+=p[i].k;
		if (s>=M) {ans=sqrt(p[i].x*p[i].x+p[i].y*p[i].y);break;}
	}
	s>=M?printf("%.7lf\n",ans):puts("-1");
	return 0;
}
