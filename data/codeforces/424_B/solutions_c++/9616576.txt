#include <iostream>
#include <cstdio>

using namespace std;

int x[1001];
int y[1001];
int k[1001];

int main()
{
    int n,s;
    cin >> n >>s;
    for(int i=0;i<n;i++)
        cin >> x[i]>>y[i]>>k[i];
    double l=0;
    double r=1e9;
    int count = 1000;
    while(count--)
    {
        double z = (l+r)/2;
        int c = s;
        for(int i=0;i<n;i++)
            if(x[i]*x[i]+y[i]*y[i]<=z*z) c+=k[i];
        if(c>=1000000) r=z; else l=z;
    }
    if(r>=1e8) cout <<-1; else printf("%.6f",r);
    return 0;
}
