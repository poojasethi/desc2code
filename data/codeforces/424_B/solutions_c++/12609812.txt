#include "iostream"
#include <algorithm>
#include "math.h"
#include <stdio.h>
using namespace std;
int main(){
	int n, s, i, q, w, e; cin >> n >> s; s = 1000000 - s; pair<int, int>p[1001];
	for (i = 0; i < n; i++){ cin >> q >> w >> e; p[i] = make_pair(q*q + w*w, e);}sort(p, p + n); 
	i = 0; while (s>0 && i<n){ s -=p[i].second; i++; }
	if (s>0)cout << "-1"; else printf("%.8f",sqrt(p[i-1].first));
}
