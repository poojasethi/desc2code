#include <cstdio>
#include <cmath>
#include <map>
using namespace std;

int main(){
	int n, s;
	int dist;
	scanf("%d %d", &n, &s);
	map<int, int> pops;
	for(int i = 0; i < n; i++){
		int x, y, p;
		scanf("%d %d %d", &x, &y, &p);
		dist = x*x + y*y;
		pops[dist] += p;
	}
	int resp = -1;
	for(map<int, int>::iterator it = pops.begin(); it != pops.end(); it++){
		if(s >= 1000000)
			break;
		s += it->second;
		resp = it->first;
	}
	if(s < 1000000)
		printf("-1\n");
	else printf("%.8lf", sqrt(resp));
	return 0;
}
