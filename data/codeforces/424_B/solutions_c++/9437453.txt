#include <bits/stdc++.h>
using namespace std;

int main(){
	int n,s;
	cin>>n>>s;
	int dis=0,x,y,k;
	vector<pair<double,int> > vec;
	for(int i=0;i<n;i++){
		cin>>x>>y>>k;
		dis=x*x+y*y;
		vec.push_back(make_pair(dis,k));
	}
	sort(vec.begin(),vec.end());
	int i=0;
	for(int i=0;i<n;i++){
		s+=vec[i].second;
		if(s>=1000000){
			cout<< setprecision(8)<<sqrt(vec[i].first)<<"\n";
			break;
		}	
	}
	if(s<1000000){
			cout<<-1<<"\n";
	}
	return 0;
}
