#include<iostream>
#include<map>
#include<cstdio>
#include<cmath>
using namespace std;


int main()
{
	int n, s;
	cin >> n >> s;
	map<int, int> m;
	for(int i = 0; i < n; i++){
		int x, y, z;
		cin >> x >> y >> z;
		m[x*x+y*y] += z;
	}
	map<int, int> :: iterator it;
	int v = 1000000;
	for(it = m.begin(); it != m.end(); it++){
		if(s+it->second >= 1000000){
			v = it->first;
			break;
		}
		s += it->second;
	}
	if(v == 1000000) {
		cout << "-1" << endl;
	}else{
		printf("%.7f\n", sqrt(v));
	}
	return 0;
}
