#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,s;
    cin>>n>>s;
    vector<pair<double,int> > v;
    for (int i=0; i<n; i++)
    {
        int a,b,c;
        cin>>a>>b>>c;
        double k=sqrt(a*a + b*b);
        v.push_back(make_pair(k,c));
    }
    sort(v.begin(),v.end());
    int ans=s;
    for (int i=0; i<n; i++)
    {
        s+=v[i].second;
        if (s>=1000000)
        {
            printf("%.9lf\n",v[i].first);
            return 0;
        }
    }
    cout<<-1<<endl;
    return 0;
}
