#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;
int main()
{
	pair<double ,double>itv[1005];
	double r;
	int n,i,j,k,a,b,c;double s;
	scanf("%d%lf",&n,&s);
	for(i=0;i<n;i++)
	{
		scanf("%d%d%d",&a,&b,&c);
		itv[i].first=sqrt(double(a*a+b*b));
		itv[i].second=c;
	}
	sort(itv,itv+n);
	for(i=0;i<n;i++)
	{
		s+=itv[i].second;
		r=itv[i].first;
		if(s>=1000000)
			break;
	}
	if(s<1000000)
		printf("-1\n");
	else
		printf("%.7f\n",r);
	return 0;
}
