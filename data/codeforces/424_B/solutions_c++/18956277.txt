#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n,s,a,b,c,i;
	cin>>n>>s;
	vector< pair<double,int> > v;
	for(i=0;i<n;i++)
	{
		cin>>a>>b>>c;
		
		double r = sqrt(a*a+b*b);
		v.push_back(make_pair(r,c));
	}
	sort(v.begin(),v.end());
	int rem=1000000-s;
	float r=0;
	for(i=0;i<n;i++)
	{
		if(rem>0)
		{
			rem-=v[i].second;
			r=v[i].first;
		}
		else
		break;
		
	}
	if(rem>0)
	{
		cout<<"-1"<<endl;
	}
	else
	printf("%.6lf",r);
	
	
	return 0;
	
}
