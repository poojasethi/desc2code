#include <bits/stdc++.h>
using namespace std;
int main()
{
	vector<pair<float,int > > v;
	int i, j, n, s, x, y, k, ans = 0;
	float radius = 0.0;
	cin >> n >> s;
	ans = s;
	for (i = 0; i < n; i++) {
		cin >> x >> y >> k;
		float dist = sqrt(x*x + y*y);
		v.push_back(make_pair(dist, k));
	}
	sort(v.begin(), v.end());
	i = 0;
	while (i < n) {
		if (ans  >= 1000000) break;
		ans += v[i].second;
		radius = v[i].first;
		i++;
	}
	if (ans < 1000000) cout << "-1"<<endl;
	else
	printf("%0.7lf\n", radius);
}
