#include <bits/stdc++.h>

using namespace std;
#define mp make_pair
#define X first
#define Y second

int n, s, x, y, k;
vector<pair<double, int> > a;

int main(){
    //freopen("input.txt", "r", stdin);
    cin>>n>>s;
    for(int i = 0; i < n; ++i){
        cin>>x>>y>>k;
        a.push_back(mp(sqrt(x*x + y*y), k));
    }
    sort(a.begin(), a.end());

    for(int i = 0; i < n; ++i){
        s += a[i].Y;
        if(s >= 1E6){
            printf("%7f", a[i].X);
            return 0;
        }
    }
    cout<<-1;
    return 0;
}
