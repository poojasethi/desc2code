
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

double dist(LL a,LL b)
{
return sqrt(a*a+b*b);
}

int main()
{
LL n,i,s,a,b,c;
cin>>n>>s;

vector <pair<double,LL> > v;

for(i=0;i<n;i++)
	{cin>>a>>b>>c;
	v.push_back(make_pair(dist(a,b),c));	
	}

sort(v.begin(),v.end());

for(i=0;i<n && s<1000000;i++)
	{
	s += v[i].second;	
	}

if(i==n && s<1000000) {cout<<-1<<endl;return 0;}

printf("%.9lf\n",v[i-1].first);


}
