#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n,s;
	cin >> n >> s;
	vector <pair <double,int> > vp;
	for(int i=0;i<n;i++)
	{
		int x,y,k;
		float r;
		cin >> x >> y >> k;
		r=sqrt(x*x + y*y);
		vp.push_back(make_pair(r,k));
		
	}
	
	sort(vp.begin(),vp.end());
	
	
	int i=0;
	while(s<=1000000 && i!=vp.size())
	{		
		s+=vp[i++].second;
		//cout << s << endl;
		if(s==1000000)
			break;	
	}
	i--;
	if(s>=1000000)
		printf("%.8lf\n",vp[i].first);
	else
		cout << -1 << endl;
	return 0;
}
