#include <bits/stdc++.h>

using namespace std;
int main(){
	int n, m, x, y, z;
	vector<pair<int, int> > city;
	cin >> n >> m;
	for (int i = 0; i<n; i++)
	{
		cin >> x >> y >> z;
		city.push_back(make_pair(x*x + y*y, z));
	}
	sort(city.begin(), city.end());
	for (int i = 0; i<n; i++){
		m += city[i].second;
		if (m >= 1000000)
		{
			cout << setprecision(9) << sqrt(city[i].first);
			return 0;
		}
	}
	cout << -1;
}
