#include <bits/stdc++.h>
using namespace std;
int n, s;
int i, j;
struct rare{
	int x, y, k;	
	double m;
}a[1001];
bool F(rare a, rare b){
	return a.m < b.m;
}
void solve(){
	scanf("%d %d", &n, &s);
	for(i = 0 ; i < n ; i ++) scanf("%d %d %d", &a[i].x, &a[i].y, &a[i].k), a[i].m = sqrt(a[i].x * a[i].x + a[i].y * a[i].y + 0.0);	
	sort(a, a + n, F);
	for(i = 0 ; i < n ; i ++){
		s += a[i].k;
		if(s >= 1000000){
			printf("%.7lf", a[i].m);
			return ;
		}
	}	
	puts("-1");
}
int main(){
	solve();
}
