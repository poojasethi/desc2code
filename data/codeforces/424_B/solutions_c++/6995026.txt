#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,pop;
    cin>>t>>pop;
    double x,y,z;
    pair<double,int> p[t];

    for(int i=0;i<t;i++){
        cin>>x>>y>>p[i].second;
        p[i].first=sqrt(x*x+y*y);
    }
    sort(p,p+t);
    for(int i=0;i<t;i++){
        pop+=p[i].second;
        if(pop>=1000000){
            cout<<fixed<<setprecision(7)<<p[i].first;
            return 0;}
    }
    cout<<-1;
    return 0;
}
