#include<bits/stdc++.h>
#define MP make_pair
#define R second
#define L first
using namespace std;
const int maxn=1000+10, mega=1000000;
pair<double, int> a[maxn];

int main(){
	int n, s;
	cin >> n >> s;
	for(int i=0;i<n;i++){
		int x, y;
		cin >> x >> y;
		double dis=sqrt(x*x+y*y);
		int k;
		cin >> k;
		a[i]=MP(dis, k);
	}
	sort(a, a+n);
	for(int i=0;i<n;i++){
		s+=a[i].R;
		if(s>=mega){
			cout << fixed << setprecision(7) <<  a[i].L << endl;
			return 0;
		}
	}
	cout << -1 << endl;
}
