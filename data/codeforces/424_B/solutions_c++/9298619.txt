#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, s, x, y, k, r = 0;
	cin >> n >> s;
	pair<int, int> dists[n];
	for(int i = 0; i<n; i++) {
		cin >> x >> y >> k;
		dists[i] = make_pair(x * x + y * y, k);
	}
	sort(dists, dists+n);
	for(int i = 0; i < n; i++) {

		if (s >= 1000000) 
			break;
		s += dists[i].second;
		r = dists[i].first;
	}
	if (s >= 1000000)
	printf("%.8lf\n", sqrt(r));
else
	printf("-1\n");

}
