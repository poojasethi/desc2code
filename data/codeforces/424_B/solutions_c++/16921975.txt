#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
int n, s, x, y, k, r;
pair<int, int> dis[1000];

int main() {
	scanf("%d %d", &n, &s);
	for (int i = 0 ; i < n; i++) {
		scanf("%d %d %d", &x, &y, &k);
		dis[i] = make_pair(x * x + y * y, k);
	}
	sort(dis, dis + n);
	for (int i = 0; i < n && s < 1000000; i++) {
		s += dis[i].second;
		r = dis[i].first;
	}
	if (s < 1000000)
		printf("-1");
	else
		printf("%0.6f", sqrt(1.0 * r));
	return 0;
}
