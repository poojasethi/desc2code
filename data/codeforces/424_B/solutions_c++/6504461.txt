#include<bits/stdc++.h>

using namespace std;

typedef struct yapi{
	int nufus,x,y;
	double uzaklik;
}yapi;

yapi A[1005];

int i,j,n,s,k,x,y,z;

int main(){
	scanf("%d %d",&n,&s);
	for(i=0;i<n;i++)scanf("%d %d %d",&A[i].x,&A[i].y,&A[i].nufus);
	for(i=0;i<n;i++){A[i].uzaklik=sqrt(A[i].x*A[i].x+A[i].y*A[i].y);}
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++)
			if(A[i].uzaklik>A[j].uzaklik)swap(A[i],A[j]);
	for(i=0;i<n;i++) {
		s+=A[i].nufus; 
		if(s>=1000000)break;
	}
	if(s>=1000000)printf("%.7lf",A[i].uzaklik);
	else printf("-1");
	return 0;
}
