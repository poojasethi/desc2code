#include<bits/stdc++.h>
#include<cstdio>
using namespace std;
int main()
{
	int n,pop;
	cin >> n >> pop;
	pair <double,int> v[n];
	int x,y;
	for(int i=0;i<n;i++)
	{
		cin >> x >> y >> v[i].second;
		v[i].first = pow((x*x + y*y),0.5);
	}
	sort(v,v+n);
	pop = 1000000-pop;
	double dist=0;
	int i=0;
	while(pop>0 && i<n)
	{
		pop = pop - v[i].second;
		dist = v[i++].first;	
	}
	if(pop<=0)
		printf("%.7lf\n",dist);
	else
		cout << "-1" << endl;
	return 0;
}
