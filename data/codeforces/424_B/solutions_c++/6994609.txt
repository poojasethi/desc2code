#include<stdio.h>
#include<math.h>
#include<algorithm>
using namespace std;
struct P{
	int x,y,r;
	double b;
}p[1001];
int cmp(P n,P m){
	return n.b<m.b;
}
int main()
{
	int n,s,i,j;
	double bj;
	while(scanf("%d%d",&n,&s)!=EOF){
		for(i=0;i<n;i++){
			scanf("%d%d%d",&p[i].x,&p[i].y,&p[i].r);
			p[i].b=sqrt(p[i].x*p[i].x+p[i].y*p[i].y);
		}
		bj=0;
		sort(p,p+n,cmp);
		for(i=0;i<n;i++){
			s+=p[i].r;
			bj=p[i].b;
			if(s>=1000000)
				break;
		}
		if(s>=1000000)
			printf("%.7f\n",bj);
		else
			printf("-1\n");
	}
}
