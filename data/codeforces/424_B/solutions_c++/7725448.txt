#include <cstdio>

int n, s;
double l, r;
int x[1111], y[1111], v[1111];

int main() {
	scanf("%d%d", &n, &s);
	s = 1000000 - s;
	for(int i=0;i<n;i++)
		scanf("%d%d%d", &x[i], &y[i], &v[i]);
	
	l = 0; r = 22222;
	for(int i=0; i<222; i ++) {
		double mid = (l+r) / 2.0;
		int sum = 0;
		for(int j=0;j<n;j++)
			if(x[j] * x[j] + y[j] * y[j] <= mid * mid)
				sum += v[j];
		if(sum >= s) r = mid;
		else l=mid;
	}
	if (l<20000)
		printf("%.9lf\n", l);
	else printf("-1\n");

	return 0;
}
