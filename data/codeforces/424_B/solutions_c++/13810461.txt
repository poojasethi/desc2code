#include<bits/stdc++.h>
using namespace std;
pair<double,int> arr[1001];
int main(){
    int n,p,x,y,i;
    cin>>n>>p;
    for(i=0;i<n;i++){
        cin>>x>>y>>arr[i].second;
        arr[i].first=sqrt(x*x+y*y);
    }
    sort(arr,arr+n);
    for(i=0;i<n;i++){
        if(p<1000000){
            p+=arr[i].second;
        }
        else{
            cout<<std::setprecision(8)<<arr[i-1].first;
            return 0;
        }

    }
    (p<1000000)?cout<<"-1":cout<<std::setprecision(9)<<arr[i-1].first;
}
