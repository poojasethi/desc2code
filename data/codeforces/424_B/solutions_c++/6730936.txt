#include<cstdio>
#include<cmath>
#include<algorithm>
struct node{
  double r;
  int p;
}x[1050];

bool cmp(node u,node v){return u.r<v.r;}

int main(){
  int n,s;
  scanf("%d%d",&n,&s);
  for(int i=1;i<=n;i++){
    int a,b;scanf("%d%d",&a,&b);
    x[i].r=sqrt(a*a+b*b);
    scanf("%d",&x[i].p);
  }
  std::sort(x+1,x+n+1,cmp);
  for(int i=1;i<=n;i++){
    s+=x[i].p;
    if(s>=1000000){
      printf("%.7f",x[i].r);
      return 0;
    }
  }
  printf("-1");
}
