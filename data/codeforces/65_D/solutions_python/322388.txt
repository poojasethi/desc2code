def get(s,i):
	s=list(s)
	s[i]+=1
	return tuple(s)

ch="GHRS"
ans=["Gryffindor","Hufflepuff","Ravenclaw","Slytherin"]
input()
s=set([(0,0,0,0)])
for c in raw_input():
	if c=='?':
		ss=set()
		for x in s:
			for i,m in enumerate(x):
				if min(x)==m:
					ss.add(get(x,i))
	else:
		ss=set([get(x,ch.index(c)) for x in s])
	s=ss
for i in xrange(4):
	if sum(list(x)[i]==min(x)for x in s )>0:
		print ans[i]
