def g(s,i):
	s=list(s)
	s[i]+=1
	return tuple(s)

p="GHRS"
a=["Gryffindor","Hufflepuff","Ravenclaw","Slytherin"]
input()
s=set([(0,0,0,0)])
for c in raw_input():
	if c=='?':
		t=set()
		for x in s:
			for i,m in enumerate(x):
				if min(x)==m:
					t.add(g(x,i))
	else:
		t=set([g(x,p.index(c)) for x in s])
	s=t
for i in xrange(4):
	if any(x[i]==min(x)for x in s ):
		print a[i]
