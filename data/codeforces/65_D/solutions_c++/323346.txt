#include<cstdio>
#include<iostream>
#include<cstring>
#include<vector>
#include<set>
#include<string>

using namespace std;

#define MAXN 100000
#define pb(x) push_back(x)

char S[MAXN];
set<vector<int> > DP[MAXN];
vector<int> t;
int len,N;
int P[256];
int Ans[5];
string A[4];

int main(){
	scanf("%d\n",&N);
	gets(S+1);
	len=strlen(S+1);
	t.pb(0);t.pb(0);t.pb(0);t.pb(0);
	DP[0].insert(t);

	int min;
	P['G']=0;P['H']=1;P['R']=2;P['S']=3;
	A[0]="Gryffindor";A[1]="Hufflepuff";A[2]="Ravenclaw";A[3]="Slytherin";
	for (int i=1;i<=len;i++)
		for (set<vector<int> >::iterator p=DP[i-1].begin();p!=DP[i-1].end();p++){
			t=*p;
			if (S[i]!='?'){
				t[P[int(S[i])]]+=1;
				DP[i].insert(t);
			} else {
				min=MAXN+10;
				for (int j=0;j<4;j++)
					if (t[j]<min) min=t[j];
				for (int j=0;j<4;j++)
					if (t[j]==min){
						t[j]+=1;
						DP[i].insert(t);
						t[j]-=1;
					}
			}
		}

	int sz,O[5];
	for (set<vector<int> >::iterator p=DP[N].begin();p!=DP[N].end();p++){
		t=*p;
		min=MAXN+10;sz=0;
		for (int i=0;i<4;i++)
			if (t[i]<min){
				min=t[i];
				O[sz=1]=i;
			} else if (t[i]==min) O[++sz]=i;
		for (int i=1;i<=sz;i++) Ans[O[i]]=1;
	}
	for (int i=0;i<4;i++)
		if (Ans[i])
			cout<<A[i]<<endl;
}
