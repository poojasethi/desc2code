#include <cstdio>
#include <cstring>
#include <cmath>
#include <queue>
#include<string>
#include <iostream>
#include <algorithm>
#include <set>
#include <map>
#include <vector>
#define eps 1e-10
#define ll long long
#define mod 1000000009
using namespace std;
map<vector<int>,int > s;
int ans[11];
string ch;
int n;
int dfs(int cur,vector<int> x)
{
    if(s[x]==1)
        return 0;
    s[x]=1;
    int Min=*min_element(x.begin(),x.end());
    if(cur==n)
    {
        if(x[0]==Min)  ans[0]=1;
        if(x[1]==Min)  ans[1]=1;
        if(x[2]==Min)  ans[2]=1;
        if(x[3]==Min)  ans[3]=1;
    }
    if(ch[cur]!='?')
    {
        if(ch[cur]=='G')   x[0]++,dfs(cur+1,x);  //x[0]--只会影响到与它同级别的，由于同级别只能执行其中一个，所以也就每
        if(ch[cur]=='H')   x[1]++,dfs(cur+1,x);
        if(ch[cur]=='R')   x[2]++,dfs(cur+1,x);
        if(ch[cur]=='S')   x[3]++,dfs(cur+1,x);
    }
    else
    {
        if(x[0]==Min)    x[0]++,dfs(cur+1,x),x[0]--;
        if(x[1]==Min)    x[1]++,dfs(cur+1,x),x[1]--;
        if(x[2]==Min)    x[2]++,dfs(cur+1,x),x[2]--;
        if(x[3]==Min)    x[3]++,dfs(cur+1,x),x[3]--;
    }
}
int main()
{

    cin>>n;
    cin>>ch;
    dfs(0,vector<int>(4));

    if(ans[0])  cout<<"Gryffindor\n";
    if(ans[1])  cout<<"Hufflepuff\n";
    if(ans[2])  cout<<"Ravenclaw\n";
    if(ans[3])  cout<<"Slytherin\n";
    return 0;
}
