#include<iostream>
#include<cstring>
#include<cstdio>
#include<string>
#include<vector>
#include<set>
using namespace std;
set < vector<int> > vv;
int n;
int Min(int x,int y)
{
  return x<y?x:y;
}
bool u[4];
char s[20000];
void dfs(int x,vector<int > v)
{ 
  int i;
  int _min;
  if (vv.find(v)!=vv.end()) return ;
  vv.insert(v);
  if (x==n)
  {
      _min=v[0];
	  for (i=1;i<4;i++)
      _min=Min(_min,v[i]);
	  for (i=0;i<4;i++)
		  if (v[i]==_min)
		  {
			   u[i]=true;
		  }
  
  }
  else
  {
    if (s[x]=='?')
	{
      _min=v[0];
	  for (i=1;i<4;i++)
      _min=Min(_min,v[i]);
	  for (i=0;i<4;i++)
		  if (v[i]==_min)
		  {
			  v[i]++;
			  dfs(x+1,v);
			  v[i]--;
		  }
	}
	else
	{
	  if (s[x]=='G') v[0]++;
	  if (s[x]=='H') v[1]++;
	  if (s[x]=='R') v[2]++;
	  if (s[x]=='S') v[3]++;
	  dfs(x+1,v);
	  if (s[x]=='G') v[0]--;
	  if (s[x]=='H') v[1]--;
	  if (s[x]=='R') v[2]--;
	  if (s[x]=='S') v[3]--;
	}
  
  }
}
int main()
{
   memset(u,false,sizeof(u));
   scanf("%d",&n);
   scanf("%s",&s);
   dfs(0,vector<int>(4,0));
   if (u[0]) printf("Gryffindor\n");
   if (u[1]) printf("Hufflepuff\n");
   if (u[2]) printf("Ravenclaw\n");
   if (u[3]) printf("Slytherin\n");
   return 0;
}
