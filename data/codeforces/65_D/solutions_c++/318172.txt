using namespace std;
#define li		long long
#define rep(i,to)	for(li i=0;i<((li)to);i++)
#define pb(a)		push_back(a)
#define sz(v)		((li)v.size())

int main(){
	int n;
	char c[10005];
	vector<string> str;
	str.pb("Gryffindor");
	str.pb("Hufflepuff");
	str.pb("Ravenclaw");
	str.pb("Slytherin");
	cin>>n;
	rep(i,n) cin>>c[i];
	map<vector<int>,int> mp;
	vector<int> vec;
	rep(i,4) vec.pb(0);
	mp[vec]=1;
	rep(i,n){
		map<vector<int>,int> tmp;
		for(map<vector<int>,int>::iterator it=mp.begin();it!=mp.end();it++){
			vector<int> vec=it->first;
			if(c[i]=='G') vec[0]+=1;
			if(c[i]=='H') vec[1]+=1;
			if(c[i]=='R') vec[2]+=1;
			if(c[i]=='S') vec[3]+=1;
			if(c[i]!='?'){
				tmp[vec]=1;
			}else{
				int mini=-1;
				rep(j,4) mini=(mini==-1)?vec[j]:min(mini,vec[j]);
				rep(j,4){
					if(mini==vec[j]){
						vec[j]+=1;
						tmp[vec]=1;
						vec[j]-=1;
					}
				}
			}
		}
		mp=tmp;
	}
	vector<int> ok;
	rep(i,4) ok.pb(0);
	for(map<vector<int>,int>::iterator it=mp.begin();it!=mp.end();it++){
		int mini=-1;
		vector<int> vec=it->first;
		rep(j,4) mini=(mini==-1)?vec[j]:min(mini,vec[j]);
		rep(j,4) if(mini==vec[j]) ok[j]=1;
	}
	rep(i,4)if(ok[i]) cout<<str[i]<<endl;
}
