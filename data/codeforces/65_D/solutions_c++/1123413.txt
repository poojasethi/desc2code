#include <cstdio>
#include <set>
#include <algorithm>
#include <vector>
#include <string>
#define rep(i,n) for(int i=0;i<n;++i)
using namespace std;
typedef vector<int> vi;
set<vi> S[2];
int n;
int main()
{
	//freopen("in","r",stdin);
	scanf("%d ",&n);
	int now=1,suc=0,debug=0;vi tmp(4,0);
	S[suc].insert(tmp);
	while(n--)
	{
		swap(now,suc);
		S[suc].clear();
		int c=getchar();
		c=c=='?'?4:(c=='G'?0:(c=='H'?1:(c=='R'?2:3)));
		//printf("%d\n",debug++);
		for(set<vi>::iterator e=S[now].begin();e!=S[now].end();++e)
		{
			tmp=*e;
			//rep(i,4)printf("%d%c",tmp[i],i<3?' ':'\n');
			if(c<4)
			{
				tmp[c]++;
				S[suc].insert(tmp);
			}
			else
			{
				int Min=*min_element(tmp.begin(),tmp.end());
				for(int i=0;i<4;++i)if(tmp[i]==Min)
				{
					++tmp[i];
					S[suc].insert(tmp);
					--tmp[i];
				}
			}
		}
	}
	//printf("%d\n",S[suc].size());
	string ans[4]={"Gryffindor","Hufflepuff","Ravenclaw","Slytherin"};
	for(int i=0;i<4;++i)
		for(set<vi>::iterator e=S[suc].begin();e!=S[suc].end();++e)
			if((*e)[i]==*min_element(e->begin(),e->end()))

			}
}
