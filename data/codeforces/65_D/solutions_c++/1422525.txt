#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <string.h>
#include <vector>
#include <set>
using namespace std;

const string st="GHRS";

set<vector<int> > ss[2];
char s[120000];
int n,m;

int main(){
	while (scanf("%d",&n)==1){
		ss[0].clear();
		ss[1].clear();
		scanf("%s",s);
		int p=1;
		vector<int> t1;
		t1.clear();
		for (int i=1;i<=4;++i) t1.push_back(0);
		ss[0].insert(t1);
		
		for (int i=0;i<n;++i){
			ss[p].clear();
			if (s[i]!='?'){
				int k=st.find(s[i]);
				for (set<vector<int> >::iterator it=ss[1-p].begin();it!=ss[1-p].end();++it){
					t1=*it;
					t1[k]++;
					ss[p].insert(t1);
				}
			}else{
				for (set<vector<int> >::iterator it=ss[1-p].begin();it!=ss[1-p].end();++it){
					t1=*it;
					int mi=9999999;
					for (int j=0;j<4;++j) mi=min(mi,t1[j]);
					for (int j=0;j<4;++j){
						if (t1[j]!=mi) continue;
						t1[j]++;
						ss[p].insert(t1);
						t1[j]--;
					}
				}
			}
			p=1-p;
		}
		bool bi[4];
		memset(bi,false,sizeof(bi));
		for (set<vector<int> >::iterator it=ss[1-p].begin();it!=ss[1-p].end();++it){
			t1=*it;
			int mi=9999999;
			for (int j=0;j<4;++j) mi=min(mi,t1[j]);
			for (int j=0;j<4;++j)
				if (t1[j]==mi) bi[j]=true;
		}
		if (bi[0]) puts("Gryffindor");
		if (bi[1]) puts("Hufflepuff");
		if (bi[2]) puts("Ravenclaw");
		if (bi[3]) puts("Slytherin");
	}
	return 0;
}
