#include <cstdio>
#include <set>
#include <string>
#include <cstring>
using namespace std;
#define fo(i,a,b) for (int i=a; i<=b; ++i)

const char S[]="GHRS?";
const string name[4]={"Gryffindor","Hufflepuff","Ravenclaw","Slytherin"};
struct state{ int a[4]; };

bool operator <(const state &l,const state &r){
	fo (i,0,3) if (l.a[i]<r.a[i]) return true;
	return false;
}

set<state> last,now;
bool ans[4];

int main(){
	int n;
	scanf("%d\n",&n);
	state t;
	memset(&t,0,sizeof(t));
	last.insert(t);
	fo (i,1,n){
		char ch;
		scanf("%c",&ch);
		int p=strchr(S,ch)-S;
		set<state>::iterator it;
		for (it=last.begin(); it!=last.end(); ++it){
			state t=*it;
			if (p<4){
				++t.a[p];
				now.insert(t);
			} else {
				int k=t.a[0];
				fo (j,1,3) k=t.a[j]<k?t.a[j]:k;
				fo (j,0,3) if (k==t.a[j]){
					state h;
					memcpy(&h,&t,sizeof(t));
					++h.a[j];
					now.insert(h);
				}
			}
		}
		last.clear();
		last.swap(now);
	}
	set<state>::iterator it;
	for (it=last.begin(); it!=last.end(); ++it){
		state t=*it;
		int k=t.a[0];
		fo (i,1,3) k=t.a[i]<k?t.a[i]:k;
		fo (i,0,3) if (k==t.a[i]) ans[i]=true;
	}
	fo (i,0,3) if (ans[i]) printf("%s\n",name[i].c_str());
}
