#include<iostream>
#include<cstring>
#include<string>
#include<cstdio>
#include<algorithm>
#include<set>
#include<vector>
using namespace std;
string name[]={"Gryffindor", "Hufflepuff", "Ravenclaw", "Slytherin"};
typedef vector<int > vc;
vc dp(4,0);
bool yes[4];
int n;
set<vc >has;
string get="GHRS?",s;
void dfs(int x)
{
  if(has.count(dp))return;
  has.insert(dp);
  int _min=*min_element(dp.begin(),dp.end());
  if(x==n)
  {
    for(int i=0;i<4;++i)
      if(_min==dp[i])
      yes[i]=1;
  }
  int id=get.find(s[x]);
  if(id^4)
  {
    ++dp[id];dfs(x+1);--dp[id];
  }
  else
  {
    for(int i=0;i<4;++i)
      if(_min==dp[i])
     ++dp[i],dfs(x+1),--dp[i];
  }
}
int main()
{
  while(cin>>n>>s)
  { memset(yes,0,sizeof(yes));
    has.clear();
    dfs(0);
    for(int i=0;i<4;++i)
      if(yes[i])
      cout<<name[i]<<"\n";
  }
}
