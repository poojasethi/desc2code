#include <cstdio>
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <set>
#include <map>

using namespace std;

const string nam[4]={"Gryffindor","Hufflepuff","Ravenclaw","Slytherin"};
map<char,int> no;
vector<int> c;
set<vector<int> > v;
int n,m;
bool ok[4];
char s[10005];

void dfs(int x)
{
  if(v.count(c)>0)
    return;
  v.insert(c);
  int mi=c[0];
  for(int i=1;i<4;++i)
    mi=min(mi,c[i]);
  if(x==n)
  {
    for(int i=0;i<4;++i)
    {
      if((c[i]==mi)&&(!ok[i]))
      {
        ok[i]=true;
        ++m;
      }
    }
    return;
  }
  int k=no[s[x]];
  if(k<4)
  {
    ++c[k];
    dfs(x+1);
    --c[k];
    return;
  }
  for(int i=0;i<4;++i)
  {
    if(c[i]==mi)
    {
      ++c[i];
      dfs(x+1);
      --c[i];
      if(m==4)
        return;
    }
  }
}

int main(void)
{
  no['G']=0;
  no['H']=1;
  no['R']=2;
  no['S']=3;
  no['?']=4;
  scanf("%d",&n);
  scanf("%s",s);
  for(int i=0;i<4;++i)
    c.push_back(0);
  dfs(0);
  for(int i=0;i<4;++i)
    if(ok[i])
      cout<<nam[i]<<endl;
  return 0;
}
