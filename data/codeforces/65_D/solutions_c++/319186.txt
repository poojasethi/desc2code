#include <cstdio>
#include <map>
#include <vector>
#include <algorithm>
#include <set>
using namespace std;
typedef vector<int> vi;

char s[100000];
int n;

int hus[256];

int main() {
  scanf("%d %s",&n,s);
  vi a(4,0),x;
  set<vi> p1,p2;
  p1.insert(a);
  hus['G']=0;
  hus['H']=1;
  hus['R']=2;
  hus['S']=3;
  for(int i=0;i<n;i++) {
    p2.clear();
    for(set<vi>::const_iterator it=p1.begin();it!=p1.end();it++) {
      x=*it;
      if(s[i]!='?') {
        x[hus[s[i]]]++;
        p2.insert(x);
      } else {
        int val=min(min(x[0],x[1]),min(x[2],x[3]));
        for(int j=0;j<4;j++) if(x[j]==val) {
          x[j]++;
          p2.insert(x);
          x[j]--;
        }
      }
    }
    p1=p2;
  }
  int mask=0;
  for(set<vi>::const_iterator it=p1.begin();it!=p1.end();it++) {
    x=*it;
    int val=min(min(x[0],x[1]),min(x[2],x[3]));
    for(int j=0;j<4;j++) if(x[j]==val) mask|=1<<j;
    if(mask==15) break;
  }
  if(mask&1) puts("Gryffindor");
  if(mask&2) puts("Hufflepuff");
  if(mask&4) puts("Ravenclaw");
  if(mask&8) puts("Slytherin");
  return 0;
}
