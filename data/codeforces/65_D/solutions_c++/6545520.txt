#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<vector>
#include<string>
#include<set>
#include<iostream>
#include<algorithm>
using namespace std;
int n;
string S;
string name[]= {"Gryffindor", "Hufflepuff", "Ravenclaw", "Slytherin"};
string idstr="GHRS?";
typedef vector<int> vi;
vi cnt;
set<vi> mem;
bool ok[4];
void dfs(int d){
    if(mem.count(cnt))return;
    mem.insert(cnt);
    int mn=*min_element(cnt.begin(),cnt.end());
    if(d==n){
        for(int i=0;i<4;++i)if(mn==cnt[i])ok[i]=true;
        return;
    }
    int cur=idstr.find(S[d]);
    if(cur!=4){
        ++cnt[cur];
        dfs(d+1);
        --cnt[cur];
    }else{
        for(int i=0;i<4;++i){
            if(mn==cnt[i]){
                ++cnt[i];
                dfs(d+1);
                --cnt[i];
            }
        }
    }
}
int main(){
    while( (cin >> n >> S)){
        cnt.clear();
        for(int i=0;i<4;++i)cnt.push_back(0);
        mem.clear();
        dfs(0);
        for(int i=0;i<4;++i){
            if(ok[i])cout << name[i] << endl;
        }
    }
    return 0;
}
