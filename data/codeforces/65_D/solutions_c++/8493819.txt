#include<stdio.h>
#include<map>
#include<algorithm>
using namespace std;
map<int,int>visit[1000005];
int test,n,i,j,b[5];
char c[1000005];
inline int hash(int g,int h,int r)
{
    return (g*n*n+h*n+r);
}
inline void work(int o,int g,int h,int r,int s)
{
    int mi=min(g,min(h,min(r,s)));
    if (o>n)
    {
        if (g==mi) b[1]=test;
        if (h==mi) b[2]=test;
        if (r==mi) b[3]=test;
        if (s==mi) b[4]=test;
        return ;
    }
    int gt=hash(g,h,r);
    if (visit[o][gt]==test)
    return ;
    visit[o][gt]=test;
    if ((c[o]=='G')||(c[o]=='?'&&g==mi))
    work(o+1,g+1,h,r,s);
    if ((c[o]=='H')||(c[o]=='?'&&h==mi))
    work(o+1,g,h+1,r,s);
    if ((c[o]=='R')||(c[o]=='?'&&r==mi))
    work(o+1,g,h,r+1,s);
    if ((c[o]=='S')||(c[o]=='?'&&s==mi))
    work(o+1,g,h,r,s+1);
}
    
int main()
{
test=1;
        scanf("%d",&n);scanf("%s",c+1);
        work(1,0,0,0,0);
        if (b[1]==test) printf("Gryffindor\n");
        if (b[2]==test) printf("Hufflepuff\n");
        if (b[3]==test) printf("Ravenclaw\n");
        if (b[4]==test) printf("Slytherin\n");
        puts("");
}
