#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <ctime>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const string tmp[4]={"Gryffindor","Hufflepuff","Ravenclaw","Slytherin"};
int i,j,k,n,m,T;
int a[4],b[4];
char p[N];
inline int get(char x) {
	return x=='G'?0:x=='H'?1:x=='R'?2:x=='S'?3:4;
}
int main() {
	srand(time(0));
		scanf("%d",&n);
		scanf("%s",p+1);
		For(i,0,3) a[i]=0;
		For(k,1,1000) {
			For(i,0,3) b[i]=0;
			For(i,1,n) {
				int A=get(p[i]);
				if (A<4) b[A]++;
				else {
					int c[4],r=0,mi=N;
					For(j,0,3) mi=min(mi,b[j]);
					For(j,0,3) if (mi==b[j]) c[r++]=j;
					b[c[rand()%r]]++;
				}
			}
			int mi=N;
			For(i,0,3) mi=min(mi,b[i]);
			For(i,0,3) if (b[i]==mi) a[i]=1;
		}
		For(i,0,3) if (a[i]) cout<<tmp[i]<<endl;
	return 0;
}
