#include<cstdio>
#include<cstring>
#include<algorithm>
#include<climits>
#include<set>
#include<map>
using namespace std;
int t;
#define MAXK 5
int ans[MAXK];
#define MAXL 10010
char s[MAXL];
int reflag[MAXL];
int tot,n;
set<long long>H;
long long pow[MAXK];
#define base 10001ll
int num[MAXK];
inline void Init(){
	pow[1]=1ll;for(int i=2;i<=4;i++)pow[i]=pow[i-1]*base;
}
inline void Find(int x,long long y){
	if(H.find(y)!=H.end())return;
	H.insert(y);if(x>tot){
		int Min=base;for(int i=1;i<=4;i++)Min=min(Min,num[i]);
		for(int i=1;i<=4;i++)if(Min==num[i])ans[i]=1;return;
	}int now=reflag[x];if(now){
		num[now]++;
		Find(x+1,y+pow[now]);
		num[now]--;
	}else{
		int Min=base;for(int i=1;i<=4;i++)Min=min(Min,num[i]);
		for(int i=1;i<=4;i++)if(Min==num[i]){
			num[i]++;
			Find(x+1,y+pow[i]);
			num[i]--;
		}
	}
}
int main(){
		Init();scanf("%d",&n);scanf("%s",s+1);
		for(int i=1;i<=n;i++){
			tot++;if(s[i]=='?')continue;
			if(s[i]=='G')reflag[tot]=1;
			else if(s[i]=='H')reflag[tot]=2;
			else if(s[i]=='R')reflag[tot]=3;
			else if(s[i]=='S')reflag[tot]=4;
		}Find(1,0);
		if(ans[1])printf("Gryffindor\n");
		if(ans[2])printf("Hufflepuff\n");
		if(ans[3])printf("Ravenclaw\n");
		if(ans[4])printf("Slytherin\n");
}
