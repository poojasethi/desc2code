#include <stdio.h>
#include <string.h>
#include <set>
#include <vector>
#define maxn 10010

using namespace std;

set< vector<int> > state;
int ans;
char st[maxn];
int s[maxn];
int n;
void dfs(int dep,vector<int> a){
	if (state.find(a)!=state.end()) return;
	if (dep==n){
		int min=-1;
		for (int i=0;i<4;i++)
			if (min==-1 || a[i]<min) min=a[i];
		for (int i=0;i<4;i++)
			if (a[i]==min) ans|=1<<i;
		return;
	}
	state.insert(a);
	if (s[dep]<4){
		a[s[dep]]++;
		dfs(dep+1,a);
	}else{
		vector<int> b;
		b.resize(4);
		int min=-1;
		for (int i=0;i<4;i++){
			b[i]=a[i];
			if (min==-1 || a[i]<min) min=a[i];
		}
		for (int i=0;i<4;i++)
			if (a[i]==min){
				b[i]++;
				dfs(dep+1,b);
				if (ans==15) return;
				b[i]--;
			}
	}
}
int main(){
	int i,j,k;
	scanf("%d",&n);
	scanf("%s",st);
	for (i=0;i<n;i++){
		k=4;
		if (st[i]=='G') k=0;
		if (st[i]=='H') k=1;
		if (st[i]=='R') k=2;
		if (st[i]=='S') k=3;
		s[i]=k;
	}
	vector<int> tmp;
	state.clear();
	tmp.resize(4);
	ans=0;
	dfs(0,tmp);
	if ((ans&1)>0) printf("Gryffindor\n");
	if ((ans&2)>0) printf("Hufflepuff\n");
	if ((ans&4)>0) printf("Ravenclaw\n");
	if ((ans&8)>0) printf("Slytherin\n");
	return 0;
}
