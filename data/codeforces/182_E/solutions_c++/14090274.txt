#include <bits/stdc++.h>

using namespace std;

const int N = 201 , L = 3001 , MOD = 1e9+7;

int dp[L][N];

inline void sadd( int &a , int b ){
  a = (a + b)%MOD;
}

struct rec{
  int a , b , ind;
  rec();
  rec( int a , int b , int ind ):
    a(a) , b(b) , ind(ind) {}
};

int main(){
  int l , n;cin >> n >> l;
  vector< rec > v;
  for(int i=0 ; i<n ; i++){
    int a , b;cin >> a >> b;
    v.push_back( rec(a , b , i) );
    if(a != b) v.push_back( rec(b , a , i) );
  }
  n = v.size();
  for(int i=1 ; i<=l ; i++)
    for(int j=0 ; j<n ; j++){
      if(v[j].a > i) continue;
      if(v[j].a == i){
        dp[i][j] = 1;
        continue;
      }
      for(int k=0 ; k<n ; k++){
        if( v[j].ind == v[k].ind ) continue;
        if( v[j].a != v[k].b ) continue;
        sadd( dp[i][j] , dp[i - v[j].a][k] );
      }
    }
  int ans=0;
  for(int i=0 ; i<n ; i++)
    sadd( ans , dp[l][i] );
  cout << ans << endl;
}
