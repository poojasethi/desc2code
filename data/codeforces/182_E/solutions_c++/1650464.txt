#include <iostream>
#include <stdio.h>
#define MOD 1000000007
using namespace std;

struct element {int a,b,i;};

element d[201];

int n,l,ans=0;
int dp[200][3005]={0};
int main()
{
//freopen("i.txt","r",stdin);
 cin>>n>>l;
 int i1=0;
 for (int i=0; i<n; i++)
 { int x,y;
   cin>>x>>y;
   d[i]=(element){x,y,i};
   if (x!=y) {d[n+i1]=(element){y,x,i};i1++;}
 }
for (int i=0;i<n+i1;i++) dp[i][d[i].a]=1;

for (int k=1; k<=l;k++)
 for (int i=0;i<n+i1;i++)
  for (int j=0;j<n+i1;j++)
   {
    if (d[i].i==d[j].i) continue;
    if (d[i].a==d[j].b&&k-d[i].a>=0)
     {
      dp[i][k]+=dp[j][k-d[i].a];
      dp[i][k]%=MOD;
     }
   }

for (int i=0;i<n+i1;i++)
 {ans+=dp[i][l];
  ans%=MOD;
 }
cout << ans << endl;
return 0;
}
