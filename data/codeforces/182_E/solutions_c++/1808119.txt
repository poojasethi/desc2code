#include <cstdio>
#include <cstring>

int dp[3001][101][2];
int n, l;
int data[100][2];
const int MOD = 1000000007;

int getAns(int left, int ind, int type)
{
    if(left<0) return 0;
    if(left==0) return 1;

    int &ret = dp[left][ind][type];
    if(ret!=-1) return ret;

    ret=0;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<2;j++)
        {
            if(i==ind && j==type) continue;
            if(i==ind) continue;
            if(j==1 && data[i][j]==data[i][j-1]) continue;
            if(ind==n || data[i][j]==data[ind][1-type])         
                ret = (ret + getAns(left - data[i][j], i, j)) % MOD;
        }
    }

    return ret;
}

int main(void)
{
    scanf("%d %d", &n, &l);
    for(int i=0;i<n;i++) scanf("%d %d", data[i], data[i]+1);
    memset(dp, -1, sizeof(dp));
    printf("%d\n", getAns(l, n, 0));
    return 0;
}
