#include <iostream>
#include <vector>
using namespace std;
const int mod = 1e9+7;
int main(){
	int n,l;
	cin >> n >> l;
	int a[n][2];
	for(int i = 0; i < n; ++i)
		cin >> a[i][0] >> a[i][1];
	int dp[3105][100][2]={};
	for(int i = 0; i < n; ++i){
		dp[a[i][0]][i][1]=true;
		if(a[i][0]!=a[i][1])
			dp[a[i][1]][i][0]=true;
	}
	for(int i = 0; i <= l; ++i){
		for(int j = 0; j < n; ++j){
			for(int k = 0; k < 2; ++k){
				for(int m = 0; m < n; ++m){
					if(m==j)
						continue;
					if(a[m][0]==a[j][k])
						dp[i+a[m][0]][m][1]+=dp[i][j][k];
					if(a[m][0]!=a[m][1]&&a[m][1]==a[j][k])
						dp[i+a[m][1]][m][0]+=dp[i][j][k];
					dp[i+a[m][0]][m][1]%=mod;
					dp[i+a[m][1]][m][0]%=mod;
				}
			}
		}
	}
	int ans = 0;
	for(int i = 0; i < n; ++i){
		for(int j = 0; j < 2; ++j){
			ans+=dp[l][i][j];
			ans%=mod;
		}
	}
	cout << ans << '\n';
}
