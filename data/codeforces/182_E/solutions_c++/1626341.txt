#include<stdio.h>
#include<string.h>
#define maxn 110
#define mod 1000000007
int x[maxn<<1],y[maxn<<1],nf,id[maxn<<1];
int d[3030][maxn<<1];
int n,l;
void input()
{
	int i,a,b;
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&a,&b);
		x[nf]=a,y[nf]=b,id[nf]=i,nf++;
		if(a!=b)
		x[nf]=b,y[nf]=a,id[nf]=i,nf++;
	}
}
void dp()
{
	int i,j,k,cnt;
	for(i=0;i<nf;i++)
	if(x[i]<=l)d[l-x[i]][i]=1;

	for(i=l;i>=0;i--)
	for(j=0;j<nf;j++)
	if(d[i][j])
	{
		for(k=0;k<nf;k++)
		if(id[k]!=id[j] && x[k]==y[j] && x[k]<=i)
		{
			d[i-x[k]][k]+=d[i][j];
			if(d[i-x[k]][k]>=mod)d[i-x[k]][k]-=mod;
		}
	}
	for(i=0,cnt=0;i<nf;i++)
	{
		cnt+=d[0][i];
		if(cnt>=mod)cnt-=mod;
	}
	printf("%d\n",cnt);
}
int main()
{
//	freopen("test","r",stdin);
	scanf("%d%d",&n,&l);
	input();
	dp();
	return 0;
}
