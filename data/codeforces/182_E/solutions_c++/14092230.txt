#include <bits/stdc++.h>
using namespace std;
const int MAXN=1000*101,MOD=(1e+9)+7;
struct rectangle{
    int index,width,tool;
};
vector<rectangle>v;
long long dp[210][4000]={};
int main()
{
   int n,l,a,b;
   cin>>n>>l;
   rectangle r;
   for(int i=0;i<n;i++)
   {
       cin>>r.tool>>r.width;
       r.index=i;
       v.push_back(r);
       if(r.tool!=r.width){
       swap(r.tool,r.width);
       v.push_back(r);
       }
   }
   n=v.size();
   for(int i=0;i<n;i++)
   {
       dp[i][0]=1;
   }
   for(int i=1;i<=l;i++)
   {
       for(int j=0;j<n;j++)
       {
           for(int k=0;k<n;k++)
           {
               if(v[j].index!=v[k].index&&v[j].tool==v[k].width&&i-v[k].tool>=0)
                  dp[j][i]=(dp[j][i]+dp[k][i-v[k].tool])%MOD;
           }
       }
   }
   long long ans=0;
   for(int i=0;i<n;i++)
   {
       if(l-v[i].tool>=0)ans+=dp[i][l-v[i].tool];
       ans%=MOD;
   }
   cout<<ans;
}
