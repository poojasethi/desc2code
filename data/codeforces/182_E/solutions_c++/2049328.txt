#include <stdio.h>

#define ll long long
#define D 1000000007
int n,L;
int a[105],b[105];
ll d[3005][105][2],rst;

int main(){
	scanf("%d%d",&n,&L);
	int i,j,k;
	for(i=1;i<=n;i++){
		scanf("%d%d",&a[i],&b[i]);
		d[a[i]][i][0]++;
		if(a[i]!=b[i])d[b[i]][i][1]++;
	}
	for(i=1;i<=L;i++){
		for(j=1;j<=n;j++){
			for(k=1;k<=n;k++){
				if(j==k)continue;
				if(a[j]<i){
					if(a[j]==a[k])d[i][j][0]+=d[i-a[j]][k][1];
					if(a[j]==b[k])d[i][j][0]+=d[i-a[j]][k][0];
					d[i][j][0]%=D;
				}
				if(a[j]==b[j])continue;
				if(b[j]<i){
					if(b[j]==a[k])d[i][j][1]+=d[i-b[j]][k][1];
					if(b[j]==b[k])d[i][j][1]+=d[i-b[j]][k][0];
					d[i][j][1]%=D;
				}
			}
		}
	}
	for(i=1;i<=n;i++)rst=(rst+d[L][i][0]+d[L][i][1])%D;
	printf("%I64d",rst);
}
