#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> a[205];
const int MOD = 1e9+7;
int dp[3005][205];
int nxt[205], x[205], y[205], id[205];
int main(void) {
	int n, l, m=0;
	scanf("%d%d", &n, &l);
	for(int i=1;i<=n;i++) {
		int u, v;
		scanf("%d%d", &u, &v);
		x[m] = u;
		y[m] = v;
		id[m] = i;
		m++;
		if(u!=v) {
			x[m] = v;
			y[m] = u;
			id[m] = i;
			m++;
		}
	}
	for(int i=0;i<m;i++)
		for(int j=0;j<m;j++)
			if(id[i] != id[j] && y[i] == x[j])
				a[i].push_back(j);
	for(int i=0;i<m;i++)
		dp[x[i]][i] = 1;
	for(int i=0;i<=l;i++)
		for(int j=0;j<m;j++)
			for(int k=0;k<(int)a[j].size();k++)
				if(i+x[a[j][k]]<=l)
					dp[(i+x[a[j][k]])][a[j][k]]
						= (dp[(i+x[a[j][k]])][a[j][k]] + dp[i][j])%MOD;
	int ans=0;
	for(int i=0;i<m;i++)
		ans = (ans + dp[l][i])%MOD;
	printf("%d\n", ans);
	return 0;
}
