#include<bits/stdc++.h>
using namespace std;
const int mod=1000000007;
int n,l;
int dp[3005][105][2];
int a[105][2];
int main()
{
//	freopen("input.txt","r",stdin);
	while(cin>>n>>l)
	{
		for(int i=0;i<n;i++)scanf("%d%d",&a[i][0],&a[i][1]);
		memset(dp,0,sizeof(dp));
		for(int k=1;k<=l;k++)
			for(int i=0;i<n;i++)
				for(int p=0;p<2-(a[i][0]==a[i][1]);p++)
				{
					if(k-a[i][p]==0){dp[k][i][p]=1;continue;}
					for(int j=0;j<n;j++)
						if(i!=j)
							for(int q=0;q<2-(a[j][0]==a[j][1]);q++)
							if(a[i][p]==a[j][q]&&k>=a[i][p])
								dp[k][i][p]=(dp[k][i][p]+dp[k-a[i][p]][j][q^(1&(a[j][0]!=a[j][1]))])%mod;
				}
		int ans=0;
		for(int i=0;i<n;i++)
			for(int p=0;p<2;p++)
				ans=(ans+dp[l][i][p])%mod;
		cout<<ans<<endl;
	}
	return 0;
}
