#define rep(i,n) for (int i=1;i<=n;i++)
#define fr(i,s,e) for (int i=s;i<=e;i++)
#define INF 0x3f3f3f3f

using namespace std;

const int MOD=1000000007;
int n,l;
struct rec {int x,y;}a[205];
long long f[3005][205],ans;

void init()
{
	scanf("%d%d",&n,&l);
	rep(i,n) 
	{
		scanf("%d%d",&a[i].x,&a[i].y);
		if (a[i].x!=a[i].y) a[i+n].x=a[i].y,a[i+n].y=a[i].x;
	}
}

void dawn()
{
	rep(i,2*n) if (a[i].x) f[a[i].x][i]=1;
	rep(i,l-1) rep(j,2*n) if (f[i][j])
	{
		rep(k,2*n) if ((k!=j)&&(abs(k-j)!=n)&&(a[k].x==a[j].y)&&(a[k].x+i<=l))
		{ 
			f[i+a[k].x][k]=(f[i+a[k].x][k]+f[i][j])%MOD;  
		} 
	}
	rep(i,2*n) ans=(ans+f[l][i])%MOD;
	cout<<ans<<endl;
} 

int main()
{
	//freopen("1.txt","r",stdin);
	
	init();
	dawn();
	
	//while (1);
}
