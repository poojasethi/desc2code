#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

typedef long long LL;
const int N = 105, L = 3005, M = int(1e9 + 7);

int f[L][N][N], a[N], b[N];

inline void inc (int &x, int y)
{
	x += y;
	if (x >= M) x -= M;
}

int main ()
{
#ifdef LOCAL
	freopen("in", "r", stdin);
#endif
	int n, l;
	cin >> n >> l;
	for (int i = 1; i <= n; ++i)
	{
		cin >> a[i] >> b[i];
		f[a[i]][i][b[i]] ++;
		if (a[i] != b[i]) f[b[i]][i][a[i]] ++;
	}
	for (int i = 1; i <= l; ++i)
		for (int j = 1; j <= n; ++j)
			for (int k = 1; k <= n; ++k) if (k != j)
			{
				if (i >= b[j])
					inc(f[i][j][a[j]], f[i - b[j]][k][b[j]]);
				if (a[j] != b[j] && i >= a[j])
					inc(f[i][j][b[j]], f[i - a[j]][k][a[j]]);
			}
	int res(0);
	for (int j = 1; j <= n; ++j)
	{
		inc(res, f[l][j][a[j]]);
		if (a[j] != b[j]) inc(res, f[l][j][b[j]]);
	}
	printf("%d\n", res);
	return 0;
}
