#include <bits/stdc++.h>
#define P 1000000007
using namespace std;

long long ans[6002][102][2];
int a[202],b[202];
int i,j,k,n,l;
int main()
 {
  cin>>n>>l;
  for (i=1;i<=n;i++)
   {
    cin>>a[i]>>b[i];
    ans[a[i]][i][0]=1;
    if (a[i] != b[i])
     ans[b[i]][i][1]=1;
   }
  for (i=0;i<=l;i++)
   for (j=1;j<=n;j++)
    for (k=1;k<=n;k++)
     if (j != k)
      {
       if (a[k] == a[j])
        ans[i+a[k]][k][0]+=ans[i][j][1];
       if (a[k] == b[j])
        ans[i+a[k]][k][0]+=ans[i][j][0];
       ans[i+a[k]][k][0]%=P;
       if (a[k] != b[k])
        {
         if (b[k] == a[j])
          ans[i+b[k]][k][1]+=ans[i][j][1];
         if (b[k] == b[j])
          ans[i+b[k]][k][1]+=ans[i][j][0];
         ans[i+b[k]][k][1]%=P;
        }
      }

  long long res=0;
  for (i=1;i<=n;i++)
   res=(res+ans[l][i][0]+ans[l][i][1])%P;

  cout<<res<<endl;
 }
