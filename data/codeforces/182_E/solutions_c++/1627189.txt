#include <iostream>

#define MOD 1000000007

using namespace std;

int N,L,A[110],B[110];
long long dp[110][2][3010],result;

long long solve()
{
	for (int i=0;i<N;i++)
		dp[i][0][A[i]]=dp[i][1][B[i]]=1;
	for (int l=1;l<=L;l++)
		for (int i=0;i<N;i++)
			for (int j=0;j<N;j++)
				if (i!=j)
				{
					if (l-A[i]>=0)
					{
						if (A[i]==A[j])
							dp[i][0][l]+=dp[j][1][l-A[i]];
						else if (A[i]==B[j])
							dp[i][0][l]+=dp[j][0][l-A[i]];
					}
					if (l-B[i]>=0)
					{
						if (B[i]==A[j])
							dp[i][1][l]+=dp[j][1][l-B[i]];
						else if (B[i]==B[j])
							dp[i][1][l]+=dp[j][0][l-B[i]];
					}
					dp[i][0][l]%=MOD;
					dp[i][1][l]%=MOD;
				}
	for (int i=0;i<N;i++)
	{
		result+=dp[i][0][L]+dp[i][1][L]*(A[i]!=B[i]);
		result%=MOD;
	}
	return result;
}

int main()
{
	cin >> N >> L;
	for (int i=0;i<N;i++)
		cin >> A[i] >> B[i];
	cout << solve() << endl;
	return 0;
}
