#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define M 1000000007

int b[105][2], f[105][3500][2];

main(){
	int n, len, i, j, k, s;
	scanf("%d%d",&n,&len);
	for (i=1;i<=n;++i) scanf("%d%d",&b[i][0],&b[i][1]);
	for (i=1;i<=n;++i){
		f[i][b[i][0]][1]=1;
		if (b[i][0]!=b[i][1]) f[i][b[i][1]][0]=1;
	}
	for (i=1;i<=len;++i){
		for (j=1;j<=n;++j)
	      for (k=1;k<=n;++k)
	        if (j!=k){
		        if (b[k][0]==b[j][1]) f[k][i+b[k][0]][1]=(f[j][i][1]+f[k][i+b[k][0]][1])%M;
		        if (b[k][0]==b[j][0]) f[k][i+b[k][0]][1]=(f[j][i][0]+f[k][i+b[k][0]][1])%M;
		        if (b[k][0]!=b[k][1]){
					if (b[k][1]==b[j][1]) f[k][i+b[k][1]][0]=(f[j][i][1]+f[k][i+b[k][1]][0])%M;
		        	if (b[k][1]==b[j][0]) f[k][i+b[k][1]][0]=(f[j][i][0]+f[k][i+b[k][1]][0])%M;
			    }
		    }
	}
	for (i=1,s=0;i<=n;++i){
		s=(s+f[i][len][0])%M;
		s=(s+f[i][len][1])%M;
	}
	printf("%d\n",s);
	return 0;
}
