#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;

int n , l;
int a[100][2];
long long f[3001][100][2];

int main()
{
	cin >> n >> l;
	for (int i = 0 ; i < n ; i ++)
		cin >> a[i][0] >> a[i][1];
	memset(f , 0 , sizeof f);
	for (int i = 1 ; i <= l ; i ++)
		for (int j = 0 ; j < n ; j ++)
			for (int k = 0 ; k < 1 + long(a[j][0] != a[j][1]) ; k ++)
			{
				if (a[j][k] == i)
					f[i][j][k] = 1;
				else if (a[j][k] < i)
				{
					for (int jj = 0 ; jj < n ; jj ++)
						if (jj != j)
							for (int kk = 0 ; kk < 2 ; kk ++)
								if (a[j][k] == a[jj][1 - kk])
									f[i][j][k] = (f[i][j][k] + f[i - a[j][k]][jj][kk]) % long(1e9 + 7);
				}
			}
	long long res = 0;
	for (int j = 0 ; j < n ; j ++)
		for (int k = 0 ; k < 2 ; k ++)
			res = (res + f[l][j][k]) % long(1e9 + 7);
	cout << res;
	return 0;
}
