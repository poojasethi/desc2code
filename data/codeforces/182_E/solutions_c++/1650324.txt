#include <iostream>
#include <stdio.h>
#define N 205
#define MOD 1000000007

using namespace std;

struct ele
 {int a,b,i;};
ele d[N];
int dp[N][4000];

int main()
{
//freopen("i.txt","r",stdin);
int n,l,nn=0;
cin>>n>>l;
for (int i=0; i<n; i++)
 { int x,y; cin>>x>>y;
   d[nn]=(ele){x,y,i}; nn++;
   if (x!=y) {d[nn]=(ele){y,x,i};nn++;}
 }
for (int i=0; i<nn; i++) dp[i][d[i].a]=1;

for (int k=1; k<=l; k++)
 for (int i=0; i<nn; i++)
  for (int j=0; j<nn; j++)
   {
    if (d[i].i==d[j].i) continue;
    if (d[i].a==d[j].b&&k-d[i].a>=0)
     {
       dp[i][k]+=dp[j][k-d[i].a];dp[i][k]%=MOD;
     }
   }
/*for (int i=0; i<nn; i++)
{
 for (int k=1; k<=l; k++) cout<<dp[i][k]<<" ";
 cout<<endl;
}
*/
int ans=0;
for(int i=0; i<nn;i++) {ans+=dp[i][l]; ans%=MOD;}
cout<<ans;
return 0;
}
