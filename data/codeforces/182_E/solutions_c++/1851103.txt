#include<cstdio>
#include<cstring>
const int N = 205;
const int MOD = 1000000007;
int dp[N][3105],a[N],b[N],t[N];
int main()
{
    int n,l,i,j,k;
    scanf("%d%d",&n,&l);
    k=n;
    for(i=1;i<=n;i++)
    {
        scanf("%d%d",a+i,b+i);
        t[i]=i;
        if(a[i]!=b[i])
        {
            a[++k]=b[i];
            b[k]=a[i];
            t[k]=i;
        }
    }
    n=k;
    for(i=1;i<=n;i++)
    {
        dp[i][a[i]]=1;
    }
    for(i=0;i<=l;i++)
    {
        for(j=1;j<=n;j++)
        {
            int v=dp[j][i];
            if(!v)continue;
            for(k=1;k<=n;k++)
            {
                if(t[k]==t[j])continue;
                if(a[k]==b[j])dp[k][i+a[k]]=(dp[k][i+a[k]]+v)%MOD;
            }
        }
    }
    int r=0;
    for(i=1;i<=n;i++)r=(r+dp[i][l])%MOD;
    printf("%d\n",r);
    return 0;
}
