#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <string.h>
#include <vector>
#include <queue>
#include <deque>
#include <map>

typedef long long ll;

using namespace std;

const int MAXN = 300, MOD = 1000 * 1000 * 1000 + 7;
int a[MAXN], b[MAXN], in[MAXN], dp[MAXN][3010];

int main()
{
	int n, l, k;
	cin >> n >> l;
	k = n;
	for(int i = 0; i < n; i++){
		cin >> a[i] >> b[i];
		in[i] = i;
		if(a[i] != b[i]){
			a[++k] = b[i];
			b[k] = a[i];
			in[k] = i;
		}
	}
	n = k + 1;
	for(int i = 0; i < n; i++)
		dp[i][a[i]] = 1;
	for(int i = 0; i < l; i++){
		for(int j = 0; j < n; j++){
			int k = dp[j][i];
			if(k <= 0)
				continue;
			for(int l = 0; l < n; l++){
				if(in[l] == in[j])
					continue;
				if(a[l] == b[j])
					dp[l][i + a[l]] = (dp[l][i + a[l]] + k) % MOD;
			}
		}
	}
	int ans = 0;
	for(int i = 0; i < n; i++){
		ans = (ans + dp[i][l]) % MOD;
	}
	cout << ans;
	return 0;
}
