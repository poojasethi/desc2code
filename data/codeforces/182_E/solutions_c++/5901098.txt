# include <stdio.h>

# define MOD 1000000007

typedef long long lld;

int n, l, nr = 107, ans;
int W[109][2];
int dp[3109][109][109];

int main(){
	
	scanf("%d %d",&n,&l);
	
	for(int i=0; i<n; i++){
		int a, b;
		scanf("%d %d",&a,&b);
		
		W[i][0] = a, W[i][1] = b;
		
		dp[a][b][nr]++;
		dp[a][b][i]++;
		
		if(a != b){
			dp[b][a][nr]++;
			dp[b][a][i]++;
		}
	}
	
	for(int i=1; i<l; i++)
		for(int j=0; j<n; j++){
			int a = W[j][0], b = W[j][1];
			
			int tmp = (dp[i][a][nr] - dp[i][a][j])%MOD;
			dp[i+a][b][nr] = (dp[i+a][b][nr] + tmp)%MOD;
			dp[i+a][b][j] = (dp[i+a][b][j] + tmp)%MOD;
			
			if(a != b){
				tmp = (dp[i][b][nr] - dp[i][b][j])%MOD;
				dp[i+b][a][nr] = (dp[i+b][a][nr] + tmp)%MOD;
				dp[i+b][a][j] = (dp[i+b][a][j] + tmp)%MOD;
			}
		}
	
	for(int i=1; i<101; i++)
		ans = (ans + dp[l][i][nr])%MOD;
	
	printf("%d",(ans+MOD)%MOD);
}
