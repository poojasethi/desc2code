#include <stdio.h>
#include <string.h>

const int MODULE = (int)1e9 + 7;

int att[111][2];

int F[3333][111][2];

int main(void)
{
	int n = 0;
	int l = 0;
	scanf("%d %d",&n,&l);
	for(int i = 0;i < n;i++) scanf("%d %d",&att[i][0],&att[i][1]);
	
	F[0][n][0] = 1;
	for(int j = 0;j <= l;j++)
	{
		for(int i = 0;i < n;i++)
		{
			for(int k = 0;k < 2;k++)
			{
				if(k && att[i][0] == att[i][1]) continue;
				if(j < att[i][k]) continue;
				for(int last = 0;last <= n;last++)
				{
					if(last == i) continue;
					for(int lk = 0;lk < 2;lk++)
					{
						if(last != n && att[i][k] != att[last][lk^1]) continue;
						F[j][i][k] = (F[j][i][k]+F[j-att[i][k]][last][lk])%MODULE;
					}
				}
			}
		}
	}
	int ans = 0;
	for(int i = 0;i <= 100;i++)
	{
		for(int k = 0;k < 2;k++)
		{
			ans = (ans+F[l][i][k])%MODULE;
		}
	}
	printf("%d\n",ans);
	return 0;
}
