a = raw_input()
b = raw_input()
can_be_solved = True
aa = []
bb = []

for i in a.split(' '):
    aa.append(int(i))

for i in b.split(' '):
    bb.append(int(i))

av = aa[2]

while av < aa[1]:
    if len(bb) == 0:
        can_be_solved = False
        break
    maximum = max(bb)
    bb.remove(maximum)
    av += maximum
    av -= 1

print aa[0] - len(bb) if can_be_solved else -1
