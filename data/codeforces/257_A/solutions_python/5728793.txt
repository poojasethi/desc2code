n, m, k = raw_input().split()
n = int(n)
m = int(m)
k = int(k)
a = raw_input().split()
for i in xrange(0, n):
	a[i] = int(a[i])

a.sort()


if (m <= k):
	print 0
else:
	r = m - k
	f = 0
	i = n - 1
	while (i >= 0) and (r > 0):
		r -= (a[i] - 1)
		i -= 1
		f += 1
	if (r > 0):
		print -1
	else:
		print f
