from __future__ import print_function, division, generators
from string import split
import sys

n, m, k = map(int, split(raw_input()))
ns = [int(i) for i in split(raw_input())]
ns.sort()

rozetki = k
filtry = 0

while rozetki < m:
    try:
        rozetki += ns.pop() - 1
        filtry += 1
    except:
        print(-1)
        sys.exit(0)

print(filtry)
