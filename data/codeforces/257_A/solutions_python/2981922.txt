n,m,k = [int(x) for x in raw_input().split()]
plugs = [int(x) for x in raw_input().split()]

supply = k
if supply >= m:
    print 0
else:
    flag = True
    plugs.sort()
    plugs.reverse()
    for i in xrange(n):
        supply += plugs[i] - 1
        if supply >= m:
            print i + 1
            flag = False
            break
    if flag:
        print -1
