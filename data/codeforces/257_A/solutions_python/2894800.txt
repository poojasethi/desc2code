#!/usr/bin/env python
# -*- coding: utf-8 -*-

n, m, k = [ int(v) for v in raw_input().split() ]
a = [ int(v) for v in raw_input().split() ]
a.sort()

c = 0
if k>=m:
    m -= k
else:
    m -= k-1
while m>0 and len(a)>0:
    c += 1
    if a[-1]>=m:
        m -= a[-1]
    else:
        m -= a[-1]-1
    a = a[:-1]

if m > 0:
    print -1
else:
    print c
