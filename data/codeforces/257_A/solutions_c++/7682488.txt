#include<bits/stdc++.h>
using namespace std;

int main(){
    int n,m,k;cin>>n>>m>>k;
    int su[n];
    for(int i=0;i<n;i++)cin>>su[i];
    sort(su,su+n);
    int sol =0,curr = k;

    for(int i=n-1;i>=0 and curr<m;i--){
        curr+=su[i]-1;
        sol++;
    }
    cout<<((curr<m)?-1:sol)<<endl;
}
