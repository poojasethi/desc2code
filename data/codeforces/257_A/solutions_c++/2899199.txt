#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
	int n,m,k,i,a[55];
	scanf("%d%d%d",&n,&m,&k);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	sort(a,a+n);
	i=n-1;
	while(m>k && i>=0)
	{
		m-=a[i];
		k--;
		i--;
	}
	if(m>k)
	{
		printf("-1\n");
	}
	else
	{
		printf("%d\n",n-1-i);
	}
	return 0;
}
