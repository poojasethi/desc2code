#include<bits/stdc++.h>
using namespace std;

int n,m,k,arr[100];
int main()
{
	cin>>n>>m>>k;
	for(int i=0;i<n;i++){
		cin>>arr[i];
	}
	sort(arr,arr+n);
	int ans=0,x=k;
	for(int i=n-1;i>=0;i--){
		if(x<m){
		x=x+arr[i]-1;
		ans++;
	}
		else{
			break;
		}
	}
	if(x<m)
	cout<<-1<<endl;
	else{
	cout<<ans<<endl;
}
}
