#include<bits/stdc++.h>
using namespace std;

int n,m,k,ar[100];
int main(){
	cin>>n>>m>>k;
	for(int i=0;i<n;i++){
		cin>>ar[i];
	}
	sort(ar,ar+n);
	int hasil=0,s=k;
	for(int j=n-1;j>=0;j--){
		if(s<m){
		s=s+ar[j]-1;
		hasil++;
	}
		else{
			break;
		}
	}
	if(s<m)
	cout<<-1<<endl;
	else{
	cout<<hasil<<endl;
}
}
