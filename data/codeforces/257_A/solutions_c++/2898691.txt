# include <stdio.h>
# include <algorithm>

using namespace std;

int n,m,k;
int t[100090];

int main()
{
scanf("%d %d %d",&n,&m,&k);

for(int h=0; h<n; h++)
scanf("%d",&t[h]);

sort(t,t+n);

int a=k,ans=0,p=n-1;

while(a<m && p>-1)
{
a=a-1+t[p--];
ans++;
}

if(a<m)
printf("-1");
else
printf("%d",ans);
}
