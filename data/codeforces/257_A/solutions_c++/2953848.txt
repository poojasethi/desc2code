#include <cstdio>
#include <algorithm>

using namespace std;
int a[70];

int main()
{
	int n,m,k;
	scanf("%d%d%d",&n, &m, &k);
	for (int i=0;i<n;++i)
		scanf("%d", &a[i]);
	sort(a,a+n);
	int t=n-1;
	while (k<m && t>=0)
	{
		k=k+a[t]-1;
		t=t-1;
	}
	if (k<m) printf("-1\n");
	else printf("%d\n", n-1-t);
	return 0;
}
