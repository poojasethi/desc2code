#include <bits/stdc++.h>
int n,m,k,a[55], I, J;
using namespace std;
int main()
{
    cin>> n>> m>> k;
    for(int i=0; i<n; i++)cin>> a[i];
    sort(a, a+n);
    for(I=n-1,J=0;I>=0;I--,J++)
    if(k< m)k+= a[I]-1;
    else break;
    cout<< (k>= m? J: -1)<< endl;
return 0;
}
