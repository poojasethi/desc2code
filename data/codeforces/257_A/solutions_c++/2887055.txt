#include <stdio.h>
#include <algorithm>
using namespace std;
int n,a[60],m,k;
inline bool cmp(int x,int y) {return x>y;}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;++i) scanf("%d",&a[i]);
	sort(a+1,a+n+1,cmp);
	for(int i=1;i<=n+1;++i){
		if(k>=m) {printf("%d\n",i-1); return 0;}
		k=k+a[i]-1;
	}
	printf("-1");
}
