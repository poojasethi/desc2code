#include<bits/stdc++.h>

using namespace std;


int i, j, k, n, h, m, a[1000], s=0;


int main(){
 
 	cin>>n>>m>>k;
 	for(i=1; i<=n; ++i) cin>>a[i];
 	sort(a+1, a+n+1);
 	
 	j=n+1;
 	
 	
 	h=0;
 	s=k;
	while (s<m && j>1)
		s+=a[--j]-1, ++h;

	if(s<m) cout<<-1;
		else cout<<h;
	
	
	
    return 0;
}
