#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	int arr[n];
	for(int i=0;i<n;i++)
		cin>>arr[i];
	int count=0,i=n-1;
	sort(arr,arr+n);
	m-=k;
	while(m>0 && i>=0)
	{
		m++;
		m-=arr[i--];
		count++;
	}
	if(m>0)
		cout<<-1<<endl;
	else
		cout<<count<<endl;
	return 0;
}
