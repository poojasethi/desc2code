#include<iostream>
#include<algorithm>
using namespace std;

int main(){
	int n, m ,k;
	cin >> n >> m >> k;
	int a[n+1];
	int i;
	for(i = 0; i < n; ++i)
		cin >> a[i];
	sort(&a[0], &a[n]);
	int count = 0;
	while(m > k && n > 0){
		++count;
		k += (a[--n] - 1);
	}
	if(m <= k)
		cout << count <<endl;
	else
		cout << "-1\n";
	return 0;
}
