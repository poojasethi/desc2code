#include <iostream>

using namespace std;
int n,m,k,i,a[51];
int main()
{
    cin>>n>>m>>k;
    for(;n;n--)
    {
        cin>>i;a[i]++;
    }
    for(i=50;i>=1;)
    {
        if(k>=m)break;
        if(!a[i]){i--;continue;}
        k+=i-1;n++;a[i]--;
    }
    if(k>=m)cout<<n;
    else cout<<"-1";
    return 0;
}
