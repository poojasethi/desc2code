#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,m,k;
cin>>n>>m>>k;
int a[n];
if(m<=k)
{
    cout<<0;
    return 0;
}
m=m-k;
int i;
for(i=0;i<n;i++)
    cin>>a[i];
sort(a,a+n);
int sum=0;
while(m>0&&n>0)
{
    m-=a[n-1]-1;
    n--;
    sum++;
}
if(m>0)
    cout<<-1;
else
cout<<sum;
return 0;
}
