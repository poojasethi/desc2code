#include <stdio.h>
#include <algorithm>
using namespace std;
int main() {
      int n,m,k,a[55]={0},i,l=0;
      scanf("%d%d%d",&n,&m,&k);
      for(i=0;i<n;i++)
            scanf("%d",&a[i]);
      sort(a,a+n);
      l = k;
      for(i=n-1;i>=0 && l<m;i--)
            l += (a[i]-1);
      printf("%d\n",l>=m?n-i-1:-1);
      return 0;
}
