#include<cstdio>
#include<iostream>
#include<algorithm>
#define NN 50
using namespace std;
int v[NN];
int main(){
	int n,m,k;scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;++i)scanf("%d",v+i);
	sort(v,v+n,greater<int>());
	int i=0, s = k;
	while(i<n&&m>s)
		s+=v[i++]-1;
	if(m<=s)
		printf("%d\n",i);
	else puts("-1");

	return 0;
}
