#include <iostream>
#include <algorithm>
#include <functional>
using namespace std;
int main(){
	int n,m,k,a[51],ans=-1;
	a[0]=1;
	cin >> n >> m >> k;
	for(int i=1;i<=n;i++) cin >> a[i];
	sort(a+1,a+n+1,greater<int>());
	for(int i=0;i<=n;i++){
		k+=(a[i]-1);
		if(k>=m){
			ans=i;
			break;
		}
	}
	cout << ans;
	return 0;
}
