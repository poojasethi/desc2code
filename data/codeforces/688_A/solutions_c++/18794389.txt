#include <bits/stdc++.h>

using namespace std;

int main(){
	int n, d, mx = 0, c = 0;
	char ch;
	cin >> n >> d;
	
	for(int i = 0;i < d;i++){
		bool k = 1;
		for(int j = 0; j < n;j++)
			cin >> ch, k = (k && ch - 48);
		
		if(k == 0){
			c++;
		}else
			c = 0;
		mx = max(mx, c);
	}
	
	cout << mx << endl;
}
