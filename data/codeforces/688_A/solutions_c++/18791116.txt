#include<bits/stdc++.h>
#define For(a,b) for(typeof(a) i=a;i<=b;++i)
using namespace std;
int main()
{
	int d,mx,n,j,cnt,c,i;
	string s;
	cin>>n>>d;
	mx=c=0;
	for(i=1;i<=d;++i)
	{
		cin>>s;
		cnt=0;
		for(j=0;j<n;++j) if(s[j]=='1') cnt++;
		if(cnt==n) c=0;
		else ++c;
		mx=max(mx,c);
	}
	cout<<mx;
	return 0;
}
