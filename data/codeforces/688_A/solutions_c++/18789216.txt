#include <iostream>
using namespace std;

int main() {
	long long int winstreak=0,n,d,max=0,i;
	cin>>n>>d;
	char str[n];
	label:
	if(winstreak>max)
			max=winstreak;
	while(d--){
		cin>>str;
		for(i=0;i<n;i++)
			if(str[i]=='0')
				{
					winstreak++;
					goto label;
				}
		winstreak=0;
	}
	cout<<max;
}
