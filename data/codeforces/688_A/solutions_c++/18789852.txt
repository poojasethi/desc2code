#include <iostream>
using namespace std;

int main() 
{
	int n,d,i,j,c=0,ans=0;
	cin>>n>>d;
	for(i=0;i<d;++i)
	{
		int f=0;
		string str;
		cin>>str;
		for(j=0;j<str.length();j++)
		{
			if(str[j]=='0')
			{
				c++;ans=max(ans,c);f=1;
				break;
			}
		}
		if(!f)
		c=0;//cout<<c<<" ";
	}
	cout<<ans;
	return 0;
}
