#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,m,s=0,S=0,M=0;
    char k;
    cin>>n>>m;
    for(int i=0; i<m; i++){
        s=0;
        for(int j=0; j<n; j++){
            cin>>k;
            s+=k-48;
        }
        if(s<n)S++;
        else S=0;
        M=max(M,S);
    }
    cout<<M;
}
