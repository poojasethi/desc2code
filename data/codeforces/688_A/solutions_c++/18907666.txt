#include<iostream>
#include<string>
using namespace std;
int main(){
	int n,d,f=0,t,e=0,r=0;
	string s;
	cin>>n>>d;
	for(int i=0;i<d;i++){
		cin>>s;
		t=s.size();
		for(int j=0;j<t;j++){
			if(s[j]=='0')f++;
		}
		if(f>0){
			r++;
			if(r>e)e=r;
		}
		if(f==0){
			if(r>e)e=r;
			r=0;
		}
		f=0;
	}
	cout<<e;
}
