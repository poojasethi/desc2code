#include <iostream>
#include <string>

using namespace std;

int br;
int r;

int main()
{
    int n, m;
    cin >> n >> m;
    while (m--) {
        string s;
        cin >> s;
        if (s.find('0') != string::npos)
            br++;
        else
            br = 0;
        r = max(r, br);
    }
    cout << r;
}
