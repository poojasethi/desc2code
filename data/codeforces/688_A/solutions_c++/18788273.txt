#include<bits/stdc++.h>
using namespace std;

bool f(char *s){
	int i=0;
	while(s[i]){
		if(s[i]=='0') return 1;
		i++;
	}
	return 0;
}

main(){
	int n, d, ans=0, tmp=0;
	char s[101];
	scanf("%d %d",&n,&d);
	for(int i=0;i<d;i++){
		scanf(" %s",s);
		if(f(s)) tmp++;
		else tmp=0;
		ans=max(ans,tmp);
	}
	printf("%d\n",ans);
	return 0;
}
