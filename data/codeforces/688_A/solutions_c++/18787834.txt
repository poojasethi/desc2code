#include <bits/stdc++.h>
using namespace std;

const int N = 111;
bitset<N> present[N];

int main() {
    int n, d;
    cin >> n >> d;
int cur = 0, ans = 0;;
for (int i=0; i<d; i++) {
cin >> present[i];
if (present[i].count() < n) ++cur;
else cur = 0;
if (cur > ans) ans = cur;
}
cout<<ans<<endl;
    return 0;
}
