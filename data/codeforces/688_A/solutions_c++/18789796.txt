#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,d;
	cin>>n>>d;
	int maxA =0, ans=0, flag=0;
	string x;
	while(d--){
		cin>>x;
		flag=0;
		for(int i=0;i<n;i++){
			if(x[i]=='0'){
				ans++;
				flag=1;
				break;
			}
		}
		if(flag==0)
			ans=0;
		if(maxA<ans)
			maxA=ans;	
	}
	cout<<maxA<<endl;
	return 0;
	}
