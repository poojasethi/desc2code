#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=150;
int n,d,ans=0;
char whe[maxn];
int main()
{
	int i,j,s=0;
	scanf("%d%d",&n,&d);
	for(i=0;i<d;i++)
	{
		scanf("%s",whe);
		for(j=0;j<n&&whe[j]=='1';j++);
		j<n?s++:s=0;
		ans=max(ans,s);
	}
	printf("%d\n",ans);
	return 0;
}
