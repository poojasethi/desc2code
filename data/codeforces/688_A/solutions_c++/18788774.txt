#include <bits/stdc++.h>
using namespace std;
int n, d, ans;
string s;

bool ok()
{
	for(int i = 0; i < s.size(); i++)
		if(s[i] == '0') return true;
	return false;
}

int main()
{
	cin >> n >> d;
	int cnt = 0;
	while(d--)
	{
		cin >> s;
		if(!ok()) cnt = 0; 
		else cnt++;
		ans = max(ans, cnt);
	}
	cout << ans;
}
