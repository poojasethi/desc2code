#include<bits/stdc++.h>

using namespace std;

int main() {
	int n ,d;
	cin>>n>>d;
	int ans = 0, loc = 0;
	for(int i=0; i<d; ++i) {
		char c;
		bool ok = 0;
		
		for(int j=0; j<n; ++j){
			cin>>c;
			if (c == '0')
				ok = 1;
		}
		if (ok) loc++, ans = max(ans, loc);
		else loc = 0;
	}
	cout<<ans<<endl;
}
