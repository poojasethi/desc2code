#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ll n,d;
	string a;
	cin>>n>>d;
	ll c=0,max=0,f;
	while(d--)
	{f=0;
		cin>>a;
		for(ll i=0;i<a.length();i++)
		{if(a[i] =='0')
		{f=1;break;
		}
		}
		if(f)
		c++;
		else
		c=0;
		if(c>max)
		max=c;
	}
	cout<<max;
	
	
	return 0;
}
