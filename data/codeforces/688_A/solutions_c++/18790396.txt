#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,d;
	scanf("%d %d",&n,&d);
	int ans=0,max=0;
	while(d--)
	{
		string att;
		cin>>att;
		int i=0;
		for(;i<n;i++)
		{	if(att[i]=='0')
			{
				ans++;
				break;
			}
		}
		if(i==n)
			ans=0;	
		if(ans>max)
			max=ans;
	}
	printf("%d",max);
	return 0;
}
