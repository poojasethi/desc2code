#include<stdio.h>
int main()
{
	int n,d,i,cnt,maxn=0,ans=0;
	char a[110];
	scanf("%d%d",&n,&d);
	while(d--)
	{
		cnt=0;
		scanf("%s",a);
		for(i=0;i<n;i++)
			if(a[i]=='1') cnt++;
		if(cnt<n) ans++;
		else ans=0;
		if(ans>maxn) maxn=ans;	
	}
	printf("%d\n",maxn);
	return 0;
}
