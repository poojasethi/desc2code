#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;

int d[105], s[105];
int r[105], rn=0, rs=10000;
int main(void) {
        int n, m, k;
        cin >> n >> m >> k;
        for(int i=0;i<m;i++) cin >> d[i];
        for(int i=0;i<k;i++) cin >> s[i];
        for(int i=0;i<m;i++) {
                int c=0;
                for(int j=0;j<k;j++) {
                        if(s[j]%d[i]==0) ++c;
                }
                if(c<rs) { rn=0; rs=c;}
                if(c==rs) {r[rn++] = i+1;}
        }
        cout << rn << '\n';
        for(int i=0;i<rn;i++) cout << r[i] << ' ';
        puts("");
        
        return 0;
}
