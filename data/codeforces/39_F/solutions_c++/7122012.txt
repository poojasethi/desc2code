#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define IMAX 2147483647
int d[200], p[200], cnt[200], ans[200];
int main() {
	int n, m, k;
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 0; i < m; i ++) scanf("%d", d+i);
	for (int i = 0; i < k; i ++) scanf("%d", p+i);
	sort(p, p+k);
	memset(cnt, 0, sizeof(cnt));
	int mv=IMAX;
	for (int i = 0; i < m; i ++) {
		for (int j = 0; j < k; j ++) {
			if (p[j]<=n && 	p[j]%d[i]==0) cnt[i]++;
		}
		mv = min(mv, cnt[i]);
	}
	k=0;
	for (int i = 0; i < m; i ++)
		if (cnt[i] == mv) ans[k++] = i+1;
	printf("%d\n", k);
	for (int i = 0; i < k; i ++) {
		if (i) printf(" ");
		printf("%d", ans[i]);
	}
	puts("");
	return 0;
}
