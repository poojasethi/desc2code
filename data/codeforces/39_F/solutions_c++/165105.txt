#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

const int MAXM = 100 + 10;
int cnt[MAXM], stp[MAXM];

int main()
{
	int n, m, k;
	cin >> n >> m >> k;
	for(int i=0; i<m; i++)
		cin >> stp[i];
	for(int i=0; i<k; i++)
	{
		int t;
		cin >> t;
		for(int j=0; j<m; j++)
			if(t % stp[j] == 0)
				cnt[j]++;
	}
	int r = *min_element(cnt, cnt + m);
	vector<int> o;
	for(int i=0; i<m; i++)
		if(cnt[i] == r)
			o.push_back(i+1);
	cout << o.size() << endl;
	for(int i=0; i<o.size(); i++)
		cout << o[i] << " ";
	cout << endl;
	return 0;
}
