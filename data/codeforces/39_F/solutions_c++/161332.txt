#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int n, f, k;
int fd[101];
int fz[101];
int main() {
	scanf("%d %d %d", &n, &f, &k);
	for(int i=0; i<f; i++)
		scanf("%d", &fd[i]);

	int a;
	for(int u=0; u<k; u++) {
		scanf("%d", &a);
		for(int i=0; i<f; i++)
			if(a%fd[i]==0) fz[i]++;
	}
	int mi = 150;
	for(int i=0; i<f; i++) mi = min(fz[i], mi);

	int cnt = 0;
	for(int i=0; i<f; i++) if(fz[i]==mi) cnt++;
	printf("%d\n", cnt);
	char sep[3];
	sep[0]=sep[1]=0;
	for(int i=0; i<f; i++) if(fz[i]==mi) {
		printf("%s%d", sep, i+1);
		sep[0]=' ';
	}
	printf("\n");
	return 0;
}
