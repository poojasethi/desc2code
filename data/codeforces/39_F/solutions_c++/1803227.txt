#include<stdio.h>
int main()
{
unsigned long int n,d[102],mos[102];
int i,m,k,cat[102]={0},rank[102],t,j;
scanf("%u %d %d",&n,&m,&k);
for(i=0;i<m;i++)
	{
	scanf("%u",&d[i]);rank[i]=1+i;
	}
for(i=0;i<k;i++)
	{
	scanf("%u",&mos[i]);
	for(j=0;j<m;j++)
		{
		if(mos[i]%d[j]==0)
			{
			cat[j]++;
			}
		}
	}
for(i=0;i<m-1;i++)
	{
	for(j=i+1;j<m;j++)
		{
		if(cat[i]>cat[j])
			{
			t=cat[i];cat[i]=cat[j];cat[j]=t;
			t=rank[i];rank[i]=rank[j];rank[j]=t;
			}
		}
	}
for(i=1;i<m;i++)
	{
	if(cat[i]!=cat[0])
		{
		break;
		}
	}
printf("%d\n",i);
for(j=0;j<i;j++)
	{
	printf("%d ",rank[j]);
	}
return(0);
}
