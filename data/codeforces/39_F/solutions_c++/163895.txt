#include<cstdio>
#define MAXN 1100
int fr[MAXN],mo[MAXN],ans[MAXN],n,m,k;
int main()
{
    scanf("%d%d%d",&n,&m,&k);
    for(int i=0;i<m;i++)
        scanf("%d",&fr[i]);
    for(int i=0;i<k;i++)
        scanf("%d",&mo[i]);
    int maxs=k+1;
    for(int i=0;i<m;i++)
    {
        int tot=0;
        for(int j=0;j<k;j++)
            if(mo[j]%fr[i]==0)
                tot++;
        if(tot<maxs)
            maxs=tot,ans[ans[0]=1]=i+1;
        else if(tot==maxs)
            ans[++ans[0]]=i+1;
    }
    printf("%d\n",ans[0]);
    for(int i=1;i<=ans[0];i++)
        printf("%d ",ans[i]);
}
