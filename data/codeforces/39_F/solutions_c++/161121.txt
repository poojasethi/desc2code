#include <iostream>
#include <vector>
#include <set>
#include <map>
using namespace std;

#define sz 102
int frog[sz], mos[sz];

int main()
{
	int i,j,k,n,m;
	while( cin>>n>>m>>k )
	{
		set <int> flag;
		int v=sz,c,r;
		vector <int> res;
		
		for(i=1;i<=m;i++) cin>>frog[i];
		
		for(j=1;j<=k;j++) cin>>mos[j];
		
		for(i=1;i<=m;i++)
		{
			for(c=0,j=1;j<=k;j++)
			{
				if( mos[j]%frog[i]==0 ) c++;
			}
			//cout<<i<<": "<<c<<endl;
			if( c<v )
			{
				res.clear();
				v=c;
			}
			if(c==v) res.push_back(i);
		}
		
		r=(int)res.size();
		cout<<r<<endl;
		if(r)
		{
			cout<<res[0];
			for(i=1;i<r;i++) cout<<" "<<res[i];
			cout<<endl;
		}
	}
	return 0;
}
