#include <cstdio>
#include <iostream>
using namespace std;

int col[150], min = 0;
int dist[150];

int main() {
    int t = 0;
    int n, m, k;
    cin >> n >> m >> k;
    for(int i = 0; i < m; i++) {
        scanf("%d", &dist[i]);
    }
    for(int i = 0; i < k; i++) {
        scanf("%d", &t);
        for(int j = 0; j < m; j++) {
            if(t%dist[j] == 0) col[j]++;
        }
    }
    int min = 100000000;
    int ans = 0;
    for(int i = 0; i < m; i++) if(col[i] < min) min = col[i];
    for(int i = 0; i < m; i++) if(col[i] == min) ans++;
    cout << ans << endl;
    for(int i = 0; i < m; i++) if(col[i] == min) cout << i+1 << " ";
    return 0;
}
