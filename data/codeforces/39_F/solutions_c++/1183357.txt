#include <cstdio>

int a[111], b[111], c[111], ans, n, m;

int main() {
	scanf("%*d%d%d", &n, &m);
	ans = 1e8; int cnt = 0;
	for (int i = 1; i <= n; ++i) scanf("%d", a + i);
	for (int j = 1; j <= m; ++j) scanf("%d", b + j);
	for (int i = 1; i <= n; ++i) {
		c[i] = 0;
		for (int j = 1; j <= m; ++j) if (b[j] % a[i] == 0) ++c[i];
		if (c[i] < ans) { ans = c[i]; cnt = 0; }
		if (c[i] == ans) ++ cnt;
	}
	printf("%d\n", cnt);
	for (int i = 1; i <= n; ++i)
		if (c[i] == ans) printf("%d ", i);
	return 0;
}
