#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n,m,k,d[105], t[105];
	cin >> n >> m >> k;

	for (int i=0;i<m;++i)
		cin >> d[i];
	for (int i=0;i<k;++i)
		cin >> t[i];

	vector<int> res;
	int ans = 0, ansmin = 1000;
	for (int i=0;i<m;++i)
	{
		int tek=0;
		for (int j=0;j<k;++j)
		{
			if (t[j] % d[i] == 0)
			{
				++tek;
			}
		}
		if (tek < ansmin)
		{
			ans = 1;
			ansmin = tek;
			res.clear();
			res.push_back(i + 1);
		} else {
			if (tek == ansmin)
				++ans, res.push_back(i + 1);
		}
	}

	cout << ans << endl;
	for (int i=0;i<ans;++i)
		cout << res[i] << " ";
	return 0;
}
