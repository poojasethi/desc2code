#include <bits/stdc++.h>
using namespace std;
#define N 1010

int n, m, k, d[N], gas[N], a[N], cnt;

int main() {
//	freopen("f.in", "r", stdin);
	scanf("%d %d %d", &n, &m, &k);
	for (int i = 1; i <= m; i ++) scanf("%d", &d[i]);
	for (int i = 1; i <= k; i ++) {
		int x;
		scanf("%d", &x);
		for (int j = 1; j <= m; j ++) if (x % d[j] == 0) gas[j] ++;
	}
	int mn = n + 1;
	for (int i = 1; i <= m; i ++) mn = min(mn, gas[i]);
	for (int i = 1; i <= m; i ++) {
		if (mn == gas[i]) a[cnt++] = i;
	}
	printf("%d\n", cnt);
	for (int i = 0; i < cnt; i ++) printf("%d ", a[i]); puts("");
	return 0;
}
