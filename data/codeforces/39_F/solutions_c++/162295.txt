#include<stdio.h>
#include<string.h>
int mm[100],mk[100],rez[100];
int main()
{
	//freopen("in","r",stdin);
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<m;i++)
		scanf("%d",&mm[i]);
	for(int i=0;i<k;i++)
		scanf("%d",&mk[i]);
	for(int i=0;i<m;i++)
		for(int j=0;j<k;j++)
			if(!(mk[j]%mm[i])) rez[i]++;
	int min=rez[0];
	for(int i=0;i<m;i++)
		if(min>rez[i]) min=rez[i];
	int kk=0;
		for(int i=0;i<m;i++)
			if(min==rez[i]) kk++;
	printf("%d\n",kk);
		
	for(int i=0;i<m;i++)
	{
		if(min==rez[i])
			printf("%d ",i+1);
	}
	return 0;
}
