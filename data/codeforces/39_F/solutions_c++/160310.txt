#include <iostream>

using namespace std;

int main()
{
    //cout << "Hello world!" << endl;
    int n,m,k;
    cin>>n>>m>>k;
    int d[100],e[100];
    for(int i=0;i<m;i++)
        cin>>d[i];
    for(int j=0;j<k;j++)
        cin>>e[j];
    int count[100];

    int min=101;
    for(int i=0;i<m;i++)
    {
        count[i]=0;
        for(int j=0;j<k;j++)
            if(e[j]%d[i]==0)count[i]++;
        if(count[i]<min)min=count[i];
    }
    int ans=0;
    for(int i=0;i<m;i++)
        if(count[i]==min)
            ans++;
    cout<<ans<<endl;
    for(int i=0;i<m;i++)
        if(count[i]==min)
            cout<<i+1<<' ';
    return 0;
}
