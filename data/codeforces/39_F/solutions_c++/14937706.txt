#include<stdio.h>
const int maxn = 110;
int a[maxn], b[maxn];
int n, m, k;

int main()
{
	while(~scanf("%d%d%d", &n, &m, &k))
	{
		for(int i = 1;i<=m;i++)
			scanf("%d", &a[i]);
		for(int i = 1;i<=k;i++)
			scanf("%d", &b[i]);
		int ans = 1e9;
		int res[maxn];
		int t = 0;
		for(int i = 1;i<=m;i++)
		{
			int cnt = 0;
			for(int j = 1;j<=k;j++)
			{
				if(b[j] % a[i] == 0)
				cnt++;
			}
			if(cnt < ans)
			{
				t = 0;
				res[++t] = i;
				ans = cnt;
			}
			else if(cnt == ans)
			{
				res[++t] = i;
			}
		}
		printf("%d\n", t);
		for(int i = 1;i<t;i++)
			printf("%d ", res[i]);
		printf("%d\n", res[t]);
	}
	return 0;
}
