#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstdlib>
#include<cmath>
#include<queue>
#include<cstring>
using namespace std;
int n,m,k;
int frog[101];
int moq[101];
int smash[101];
int minque=0x3f3f3f;
int cnt=0;
bool flag=false;
int main()
{
	cin>>n>>m>>k;
	for(int i=1;i<=m;i++)
	{
		cin>>frog[i];
	}
	for(int i=1;i<=k;i++)
	{
		cin>>moq[i];
		for(int j=1;j<=m;j++)
		{
			if(moq[i]%frog[j]==0)
			{
				smash[j]++;
			}
		}
	}
	for(int i=1;i<=m;i++)
	{
		minque=min(minque,smash[i]);
	}
	for(int i=1;i<=m;i++)
	{
		if(smash[i]==minque)
		{
			cnt++;
		}
	}
	cout<<cnt<<endl;
	for(int i=1;i<=m;i++)
	{
		if(smash[i]==minque)cout<<i<<" ";
	}
	return 0;
}
