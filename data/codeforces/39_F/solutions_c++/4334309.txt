#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
int main()
{
  int n,m,k,len[110],loc[110],i,j,mini[110],mins=101,sum=0,ans=0;
  cin>>n>>m>>k;
  for(i=1;i<=m;i++)
    cin>>len[i];
  for(i=1;i<=k;i++)
    cin>>loc[i];
  for(i=1;i<=m;i++)
  {
    sum=0;
    for(j=1;j<=k;j++)
      if(loc[j]%len[i]==0)
        sum++;
    if(sum==mins)
      mini[++ans]=i;
    if(sum<mins)
    {
      ans=0;
      mini[++ans]=i;
      mins=sum;
    }
  }
  cout<<ans<<endl;
  for(i=1;i<=ans;i++)
    cout<<mini[i]<<" ";
  cout<<endl;
  return 0;
}
