#include <iostream>
#include <vector>

using namespace std;

int main() {
	int n,m,k, mm[100], km[100], mi=0, s;
	cin >> n >> m >> k;
	int ms=k+1;
	for (int i=0; i<=m-1; i++) {
		cin >> mm[i];
	}
	for (int i=0; i<=k-1; i++) {
		cin >> km[i];
	}
	for (int i=0; i<=m-1; i++) {
		s=0;
		for (int j=0; j<=k-1; j++) {
			if (km[j]%mm[i]==0) s++;
		}
		if (s==ms) mi++;
		if (s<ms) { ms=s; mi=1; }	
	}
	cout << mi << endl;
	for (int i=0; i<=m-1; i++) {
		s=0;
		for (int j=0; j<=k-1; j++) {
			if (km[j]%mm[i]==0) s++;
		}
		if (s==ms) cout << i+1 << ' ';
	}	
	cout << endl;
	return 0;
}
