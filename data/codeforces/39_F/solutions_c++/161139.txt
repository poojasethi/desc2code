#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <cstdio>
#include <cmath>
#include <cstdlib>
using namespace std;
int main()
{
	int n,m,k,i,j,l,sum;
	cin >>n >>m >>k;
	vector <int> a(m+1,0);
	vector <int> s(m+1,0);
	vector <int> mux(k+1,0);
	for (i=1;i<=m;i++)
		cin >>a[i];
	for (i=1;i<=k;i++)
		cin >>mux[i];
	for (i=1;i<=m;i++)
	{
		for (j=1;j<=k;j++)
			if (mux[j]%a[i]==0) s[i]++;
		if (sum>s[i]) sum=s[i];
	}
	l=0;
	for (i=1;i<=m;i++)
		if (s[i]==sum) l++;
	cout <<l <<'\n';
	for (i=1;i<=m;i++)
		if (s[i]==sum) cout << i<<' ';

}
