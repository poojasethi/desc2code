#include <iostream>
#include <cstdio>
using namespace std;
int lag[1001], komar[1001], ans[1001];
int main()
{
	int k, n, m, index = 0;
	cin >> n >> m >> k;
	for (int i = 1; i <= m; i++)
		cin >> lag[i];
	for(int i = 1; i <= k; i++)
	  cin >> komar[i];
	int mini = 999999, chet = 0;
	for (int i = 1; i <= m; i++)
	{
		chet = 0;
		for (int j  = 1; j <=	k; j++)
			if (komar[j]%lag[i] == 0)
				chet++;
		if (chet < mini)
		{
			index = 1;
			ans[index] = i;
			mini = chet;
		}
		else if (chet == mini)
		{
			index++;
			ans[index] = i;
		}
	}
	cout << index << endl;
	for (int i  =1; i <= index; i++)
		cout << ans[i] << ' ';
	return 0;
}
