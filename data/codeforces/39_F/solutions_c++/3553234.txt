#include <stdio.h>
int d[10000];
int kk[10000];
 int l[1000];
 int nl=0;
int main(){
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    for (int i=0;i<m;i++)
        scanf("%d",&d[i]);
    for (int i=0;i<k;i++)
        scanf("%d",&kk[i]);
    int min=10000;
   
    for (int i=0;i<m;i++)
       {
            int nn=0;
            for (int j=0;j<k;j++)
                if (kk[j]%d[i]==0)
                   nn++;
            if (nn<min)
                min=nn,nl=0;
            
                if (nn<=min)
                    l[nl]=i+1, nl++;
           
    }
    printf("%d\n",nl);
    for (int i=0;i<nl;i++)
        printf("%d ",l[i]);
}
