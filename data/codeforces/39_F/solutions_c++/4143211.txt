#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

typedef long long ll;

ll d[110];
vector<ll> mos;
vector<int> ans;

int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	for(int i=0;i<m;i++)
		cin>>d[i];
	for(int i=0;i<k;i++)
	{
		ll x;
		cin>>x;
		mos.push_back(x);
	}
	int Min=10000;
	for(int i=0;i<m;i++)
	{
		int t=0;
		for(int j=0;j<k;j++)
			if(mos[j]%d[i]==0)
				t++;
		Min=min(Min,t);
	}
	for(int i=0;i<m;i++)
	{
		int t=0;
		for(int j=0;j<k;j++)
			if(mos[j]%d[i]==0)
				t++;
		if(t==Min)
			ans.push_back(i+1);
	}
	cout<<(int)ans.size()<<endl;
	for(int i=0;i<(int)ans.size();i++)
		cout<<ans[i]<<' ';
	cout<<endl;
	return 0;
}
