#include <stdio.h>
#include <iostream>

using namespace std;

int n,m,k,i,min1,j,q,e;
int d[102],r[102],c[102];

int main(){
  cin >> n >> m >> k;
  for (int i=0;i<m;i++)
	  cin >> d[i];
  for (int i=0;i<k;i++)
	  cin >> r[i];
  min1=1000000001;
  e=0;
  for (int i=0;i<m;i++){
    q=0;
	for (int j=0;j<k;j++){
	  if (!(r[j]%d[i]))
	    q++;
    }
    c[i]=q;
	if (q==min1) 
		e++;
	if (q < min1){ 
		min1=q;
		e=1;
	}
  }
 cout << e << endl;
 for (int i=0;i<m;i++)
   if (c[i]==min1)
	   cout << i+1 << ' ';

}
