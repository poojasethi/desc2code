#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int a[110];
int b[110];
int ans[110];
int min1=(int)1e9;
int stack[110];
int tot=0,n,m;

int main()
{
    int i,j,k;
    cin>>n>>m>>k;
    for(i=1;i<=m;i++)
        scanf("%d",&a[i]);
    for(i=1;i<=k;i++)
        scanf("%d",&b[i]);
    for(i=1;i<=k;i++)
        for(j=1;j<=m;j++)
            if(b[i]%a[j]==0)
                ans[j]++;
    for(i=1;i<=m;i++)
        min1=min(min1,ans[i]);
    for(i=1;i<=m;i++)
        if(min1==ans[i])
        {
            tot++;
            stack[tot]=i;
        }
    cout<<tot<<endl;
    for(i=1;i<=tot;i++)
        printf("%d ",stack[i]);
    cout<<endl;
    return 0;
}
