#include <iostream>
#include <vector>
using namespace std;
const int N = 100;
int A[N], B[N], C[N];
int l, n, m;
int ans;

void init(){
	cin >> l >> n >> m;
	for (int i=0;i<n;i++)
		cin >> A[i];
	for (int i=0;i<m;i++)
		cin >> B[i];
}

void solve(){
	for (int i=0;i<m;i++){
		for (int j=0;j<n;j++)
			if (B[i]%A[j]==0) C[j]++;
	}
	
	ans = m;
	for (int i=0;i<n;i++)
		ans = min(ans, C[i]);
}

void print(){
	vector<int> res;
	for (int i=0;i<n;i++)
		if (C[i]==ans) res.push_back(i+1);
	
	cout << res.size() << endl;
	for (int i=0;i<res.size()-1;i++)
		cout << res[i] << " ";
	cout << res[res.size()-1] << endl;
		
}

int main(){
	init(); solve();
	print();
}
