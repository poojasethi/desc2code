#include <stdio.h>
#include <stdlib.h>

int main(){

	int stop,count,minimal,j,i,h,f,m,resp[101], jumps[101],hills[101];
	char a;

	scanf("%d %d %d",&h,&f,&m);
	scanf("%c",&a);
	
	for (i = 0; i < 101; i++)
		resp[i] = 0;

	for(i = 0; i < f; i++)
		scanf("%d",&jumps[i]);
		
	for(i = 0; i < m;i++)
		scanf("%d",&hills[i]);
	
	minimal = 102;
	for (i = 0 ; i < f; i++){
		for (j = 0; j < m; j++){
			if ( ( hills[j] % jumps[i]) == 0){
				resp[i] ++;	
			}
		}		
		if (minimal > resp[i])
			minimal = resp[i];
	}
	count = 0;
	for ( i = 0; i < f; i++)
		if (resp[i] == minimal) 
			count++;
	printf("%d\n",count);
	for ( i = 0; i < f; i++)
		if (resp[i] == minimal)
			printf("%d ",(i+1));	
	printf("\n");
	return 0;
}
