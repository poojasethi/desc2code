#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int n,m,k;
	int i,j;
	int fr[101];
	int mos[101];
	int ans[101];
	scanf("%d%d%d",&n,&m,&k);
        for(i=1;i<=m;i++)scanf("%d",&fr[i]);
	for(i=1;i<=k;i++)scanf("%d",&mos[i]);
	for(i=1;i<=m;i++)
	{
		ans[i]=0;
		for(j=1;j<=k;j++)if(mos[j]%fr[i]==0)ans[i]++;
	}
	int min=1000000000;
	for(i=1;i<=m;i++)if(min>ans[i])min=ans[i];
	int count=0;
	for(i=1;i<=m;i++)if(min==ans[i])count++;
	printf("%d\n",count);
	for(i=1;i<=m;i++)if(ans[i]==min)printf("%d ",i);
	printf("\n");
	return 0;
}
