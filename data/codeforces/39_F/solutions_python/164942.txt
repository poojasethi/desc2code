(n,m,k) = map (int,str(raw_input()).split (" "))
d       = map (int,str(raw_input()).split (" "))
b       = map (int,str(raw_input()).split (" "))

frogs = [0  for f in range(m)]

for frog in range(m):
    for h in b:
        if (h % d[frog] == 0):
            frogs[frog]+=1

minf = min(frogs)
pf   = []

for f in range(m):
    if frogs[f] == minf:
        pf.append(str(f+1))

print len(pf)
print " ".join(pf)
