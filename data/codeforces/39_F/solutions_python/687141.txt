import sys

(n, m, k) = sys.stdin.readline().split(" ")
d = sys.stdin.readline().split(" ")
p = sys.stdin.readline().split(" ")

frogs = []
frog = 0

for dist in d:
	frogs.append(0)
	for mosq in p:
		if int(mosq) % int(dist) == 0:
			frogs[frog] += 1
	frog += 1

lowest = min(frogs)

result = []
frog = 1
for score in frogs:
	if lowest == score:
		result.append(frog)
	frog += 1

print len(result)
print (" ").join(map(str, result))
