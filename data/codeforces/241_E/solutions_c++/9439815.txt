#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N = 1050;
const int M = 5050;
int dis[N];
int jie[N];
int n,m;
int a[M],b[M];
int lian[N];
int main()
{
	int i,j;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;i++)
		scanf("%d%d",&a[i],&b[i]);
	lian[1]=1;
	lian[n]=2;
	for(i=1;i<n;i++)
	{
		for(j=1;j<=m;j++)
		{
			if(lian[a[j]]&1)
				lian[b[j]]|=1;
			if(lian[b[j]]&2)
				lian[a[j]]|=2;
		}
	}
	memset(dis,0x3f,sizeof dis);
	dis[1]=0;
	for(i=1;i<n;i++)
	{
		for(j=1;j<=m;j++)
		{
			if(lian[a[j]]==3 && lian[b[j]]==3)
			{
				dis[b[j]]=min(dis[b[j]],dis[a[j]]+2);
				dis[a[j]]=min(dis[a[j]],dis[b[j]]-1);
			}
		}
	}
	for(i=1;i<=n;i++)
		jie[i]=dis[i];
	for(j=1;j<=m;j++)
	{
		if(lian[a[j]]==3 && lian[b[j]]==3)
		{
			dis[b[j]]=min(dis[b[j]],dis[a[j]]+2);
			dis[a[j]]=min(dis[a[j]],dis[b[j]]-1);
		}
	}
	for(i=1;i<=n;i++)
	{
		if(dis[i]!=jie[i] && lian[i]==3)
		{
			printf("No\n");
			return 0;
		}
	}
	printf("Yes\n");
	for(i=1;i<=m;i++)
	{
		if(lian[a[i]]==3 && lian[b[i]]==3)
			printf("%d\n",dis[b[i]]-dis[a[i]]);
		else
			printf("1\n");
	}
	return 0;
}
