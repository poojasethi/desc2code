#include<iostream>
using namespace std;
int n,i,de,at=-2e9,bt=2e9;
char a[100]={0},b[2];
int main()
{
	cin>>n;
	while(n--)
	{
		cin>>a>>de>>b;
		if(b[0]=='N')
		{
			if(a[0]=='<') a[0]='>';
			else if(a[0]=='>') a[0]='<';
			a[1]='='-a[1];
		}
		if(a[0]=='<'&&a[1]=='='&&bt>de) bt=de;
		else if(a[0]=='<'&&!a[1]&&bt>de-1) bt=de-1;
		else if(a[0]=='>'&&a[1]=='='&&at<de) at=de;
		else if(a[0]=='>'&&!a[1]&&at<de+1) at=de+1;
	//	cout<<at<<endl<<bt<<endl;
		
	}
	
	if(bt>=at) cout<<at<<endl;
	else cout<<"Impossible"<<endl;
//	cin>>b;
	return 0;
}
