#include<bits/stdc++.h>
using namespace std;

int main() {
	int mx=2e9, mn=-mx;

	int n; cin>>n;
	while(n--) {
		string s,ans;
		int x;
		cin>>s>>x>>ans;
		if(ans=="N") {
			if(s==">=") s="<";
			else if(s=="<") s=">=";
			else if(s=="<=") s=">";
			else s="<=";
		}

		if(s==">=") mn=max(mn, x);
		else if(s==">") mn=max(mn, x+1);
		else if(s=="<=") mx=min(mx, x);
		else mx=min(mx, x-1);
	}
	if(mn<=mx) cout << mn;
	else cout << "Impossible";
	cout << endl;
	return 0;
}
