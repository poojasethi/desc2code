#include <bits/stdc++.h>

using namespace std;

int main ()
{
    string s, a;
    int n, i, x, mx=2*1e9, mn=-mx;
    cin >> n;
    for (i = 0; i < n; ++i)
    {
        cin >> s >> x >> a;
        if ((s==">"&&a=="Y") || (s=="<="&&a=="N"))mn=max(mn, x+1);
        if ((s==">="&&a=="Y")|| (s =="<"&&a=="N"))mn=max(mn, x);
        if ((s=="<"&&a=="Y") || (s==">="&&a=="N"))mx=min(mx, x-1);
        if ((s=="<="&&a=="Y") || (s==">"&&a=="N"))mx=min(mx, x);
    }
    (mn<=mx)?cout << mn:cout << "Impossible\n";
    return 0;
}
