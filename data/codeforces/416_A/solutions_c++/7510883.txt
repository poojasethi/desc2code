#include<stdio.h>
int n;
int main(void)
{
    int l=-9999999999,r=-l;
    char s[3], t[3];
    int x, flag;
    scanf("%d", &n);
    for (int i=0;i<n;i++)
    {
        scanf("%s %d %s", s, &x, t);
        flag=(s[0]=='<');
        if (s[1]!='=')
            x-=flag*2-1;
        if(t[0]=='N')
        {
            flag^=1;
            x-=flag*2-1;
        }
        if (!flag)
            l=l>x?l:x;
        else
            r=r<x?r:x;
    }
    if (l > r)
        printf("Impossible\n");
    else
        printf("%d\n",l);
    return 0;
}
