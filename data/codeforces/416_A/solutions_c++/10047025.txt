#include <bits/stdc++.h>
using namespace std;

int main () {
	int mx = 1000*1000*1000*2;
	int mn = -mx;

	int n; cin >> n;
	while (n--) {
		string s; cin >> s;
		int x; cin >> x;
		string ans; cin >> ans;

		if (ans== "N") {
			if (s==">=")
				s = "<";
			else if (s=="<")
				s = ">=";
			else if (s=="<=")
				s = ">";
			else
				s = "<=";
		}
		if (s==">=")
			mn = max(mn, x);
		else if (s==">")
			mn = max(mn, x+1);
		else if (s=="<=")
			mx = min(mx,x);
		else
			mx = min(mx,x-1);
	}		
	if (mn<=mx)
		cout << mn << endl;
	else
		cout << "Impossible\n";
	return 0;
}
