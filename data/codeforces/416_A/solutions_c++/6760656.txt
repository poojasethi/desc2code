#include <iostream>
#include <algorithm>
#include <cstdio>
#include <string>
using namespace std;
int l=-2e9,r=2e9;

int main(void)
{
	int n;
	cin>>n;
	while(n--)
	{
		string s1,s2;
		int x;
		cin>>s1>>x>>s2;
		if(s1==">=") s2=="Y"? l=max(l,x-1):r=min(r,x);
		if(s1==">")  s2=="Y"? l=max(l,x):r=min(r,x+1);
		if(s1=="<=") s2=="Y"? r=min(r,x+1):l=max(l,x);
		if(s1=="<")  s2=="Y"? r=min(r,x):l=max(l,x-1);
	}
	if(l+2<=r) cout<<l+1<<endl;
	else cout<<"Impossible"<<endl;
	return 0;
}
