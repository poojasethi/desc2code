#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define MAXN 1000100

int cnt[MAXN];
int k;
int sum;
long long res;
char str[MAXN];

int main()
{
	scanf("%d%s",&k,str);
	cnt[0]=1;
	for(int i=0;str[i];i++)
	{
		sum+= str[i]-'0';
		if(sum>=k)
		{
			res+=cnt[sum-k];
		}
		cnt[sum]++;
	}
	
	cout << res;
	return 0;
}
