#include<stdio.h>
#include<string.h>
__int64 ans;
int dp[1000004];
char s[1000004];
int k, sum;
int main()
{
	int i;
	scanf("%d%s", &k, s);
	int len = strlen(s);
	dp[0] = 1;
	for(i = 0; i < len; i++)
	{
		sum += s[i] - '0';
		if(sum >= k) ans += dp[sum - k];
		dp[sum]++;
	}
	printf("%I64d\n", ans);
	return 0;
}
