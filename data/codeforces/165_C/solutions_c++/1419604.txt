#include <iostream>
using namespace std;

int cnt[(int)1e6+100];

int main()
{
  int k;
  cin>>k;
  string x;
  cin>>x;
  int ct=0;
  cnt[0]=1;
  long long ans=0;
  for(int i=0;i<x.length();++i)
  {
    if(x[i]=='1')
      ct++;
    if(ct>=k)
    ans+=cnt[ct-k];
    cnt[ct]++;
  }
  cout<<fixed<<ans<<endl;
  return 0;
}
