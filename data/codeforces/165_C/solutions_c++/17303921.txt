#include<stdio.h>
//char s[1000010];
int dp[1000010]={1};
int main()
{
    int k,i,c,sum=0;
    long long ans=0;
    scanf("%d ",&k);
    while( c = getchar(),c!='\n')
    {
        if(c=='1') ++sum;
        if(sum>=k) ans+=dp[sum-k];
        ++dp[sum];
    }
    printf("%I64d\n",ans);
    return 0;
}
