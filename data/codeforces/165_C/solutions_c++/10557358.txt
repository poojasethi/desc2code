#include <iostream>
using namespace std;

int main() {
    // your code goes here
    long long k,a[1000010]={0},c=0,ans=0,i;
    cin>>k;
    string s;
    cin>>s;
    a[0]=1;
    for(i=0;i<s.size();i++)
    {
        if(s[i]-48) c++;
        if(c>=k) ans+=a[c-k];
        a[c]++;
    }
    cout<<ans;
    return 0;
}
