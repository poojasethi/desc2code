#include<bits/stdc++.h>
using namespace std;
main()
{
	int k;
	scanf("%d",&k);
	string s;
	cin>>s;
	int n=s.length();
	int dp[1000005];
	memset(dp,0,sizeof(dp));
	dp[0]=1;
	int j=0;long long  ans=0;
	for(int i=0;i<n;i++)
	{
		if(s[i]=='1'){j++;}
		if(j>=k){ans+=dp[j-k];}
		dp[j]++;
	}
	cout<<ans;
}
