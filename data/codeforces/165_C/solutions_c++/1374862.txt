
#include <cstdio>
using namespace std;

enum {
  MAXN = 1000000
};

int sum[MAXN + 1];

int main()
{
  int k, c;

  scanf("%d ", &k);

  long long ans = 0;

  int cs = 0;
  sum[0] = 1;
  while ((c = getchar()) != '\n') {
    cs += c - '0';
    ans += (cs >= k) ? sum[cs - k] : 0;
    sum[cs]++;
  }

  printf("%I64d\n", ans);

  return 0;
}
