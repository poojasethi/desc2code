#include <iostream>
using namespace std;
typedef long long ll;
const ll nmax = 1e6+5;
ll cnt[nmax];
int main(){
	ll k,n;
	string s;
	cin >> k;
	cin >> s;
	n=s.size();
	ll sum = 0, ans = 0;
	cnt[0]++;
	for(ll i = 0; i < n; ++i){
		sum+=s[i]-'0';
		if(sum-k>=0)
			ans+=cnt[sum-k];
		cnt[sum]++;
	}
	cout << ans << '\n';
}
