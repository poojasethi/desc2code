#include <stdio.h>
const int MAX=(int)1e6+10;
typedef long long ll;
int k;
ll dp[MAX]={1};
char s[MAX];
int main()
{
	scanf("%d\n%s",&k,s);
	ll ans=0;
	int cnt=0;
	for (int i=0; s[i]; i++) {
		cnt+=(s[i]=='1'?1:0);
		if (cnt>=k) ans+=dp[cnt-k];
		dp[cnt]++;
	}
	printf("%I64d\n",ans);
	return 0;
}
