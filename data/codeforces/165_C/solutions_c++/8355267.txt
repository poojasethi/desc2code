#include <iostream>
#include <string>

using namespace std;

int k;
string s;
int dp[1000006];
int main()
{
	cin >> k;
	cin >> s;
	dp[0] = 1;
	int c = 0;
	long long int ans = 0;
	for(int i = 0; i < s.size(); i++)
	{
		if(s[i] == '1')
			c++;
		if(c >= k)
			ans = (ans + dp[c - k]);
		dp[c]++;
	}
	cout << ans << endl;
}
