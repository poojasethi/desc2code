#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
const int maxn=1000000+10;
ll k,f[maxn]={1};
string s;

int main(void)
{
	cin>>k>>s;

	ll len = s.size();
	ll cnt=0,ans=0;
	for(ll i=0;i<len;i++)
	{
		if(s[i]=='1') cnt++;
		if(cnt>=k) ans+= f[cnt-k];
		f[cnt]++;
	}
	cout<<ans<<endl;
	return 0;
}
