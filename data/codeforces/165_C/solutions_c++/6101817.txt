#include <iostream>
using namespace std;
long long arr[1000001]={1,0};
long long ans=0;
int main(){
    int k,l,j=0;
    string s;
    cin >>k>>s;
    l=s.size();
    for(int i=0;i<l;i++){
        if(s[i]=='1') j++;
        if(j>=k) ans+=1ll*arr[j-k];
        arr[j]++;
    }
    cout <<ans;
   return 0;
}
