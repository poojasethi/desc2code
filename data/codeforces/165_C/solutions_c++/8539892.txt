#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int dp[10000010];
int main()
{
	int k;
	cin>>k;
	ll ans=0,sum=0;
	string s;
	cin>>s;
	dp[0]=1;
	for(int i=0;i<s.length();i++)
	{
		sum+=(s[i]-'0');
		if(sum>=k)ans+=dp[sum-k];
		dp[sum]++;
	}
	cout<<ans;
	return 0;

}
