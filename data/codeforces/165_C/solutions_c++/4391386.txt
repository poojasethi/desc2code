#include<iostream>
#include<string>
long long a[1<<20],i,j,k,o=1,q,t;
std::string s;
int main(){
    std::cin>>k>>s;
    for(q=s.size();i<q;)if(s[i++]-'0')a[o++]=i;
    a[o]=q+1;
    if(i=k)for(;i<o;)t+=(a[i+1]-a[i++])*(a[j+1]-a[j++]);
    else for(;j<o;)t+=(k=a[j+1]-a[j++])*(k-1)/2;
    std::cout<<t<<"\n";
}
