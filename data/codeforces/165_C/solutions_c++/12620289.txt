#include<iostream>
#include<cmath>
using namespace std;
int b[1000001];
int main()
{
    string s;
    int k,j=0,i;
    long long ans=0;
    cin>>k>>s;
    b[0]=1;
    for(i=0;i<s.size();i++)
    {
        if(s[i]=='1')   j++;
        if(j>=k)    ans+=b[j-k];
        b[j]++;
    }
    cout<<ans;
    return 0;
}
