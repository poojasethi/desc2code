k = int(raw_input())
s = raw_input()
n = len(s)

pos = [-1]
for i in xrange(n):
	if s[i] == '1':
		pos.append(i)
pos.append(n)

ans = 0
if k > 0:
	for i in xrange(1, len(pos) - k):
		lef = pos[i] - pos[i - 1]
		rig = pos[i + k] - pos[i + k - 1]
		ans += lef * rig
else:
	for i in xrange(1, len(pos)):
		t = pos[i] - (pos[i - 1] + 1)
		ans += t * (t + 1) / 2
print ans
