k = input()
s = raw_input()
ans = 0
if k == 0:
    c = 0
    s += '1'
    for x in s:
        if x == '1':
            ans += c * (c + 1) / 2
            c = 0
        else:
            c += 1
else:
    one = [-1]
    for i, x in enumerate(s):
        if x == '1': one.append(i)
    one.append(len(s))
    for i in xrange(1, len(one) - k):
        ans += (one[i] - one[i - 1]) * (one[i + k] - one[i + k - 1])
print ans
