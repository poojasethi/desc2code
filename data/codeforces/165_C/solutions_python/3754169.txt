import re
import fractions

class Task:
    k, s = 0, ''
    answer = ''

    def getData(self):
        self.k = int(raw_input())
        self.s = raw_input()
        #fin = open('input.txt', 'r')
        #fin.readline()
        #self.c = [int(x) for x in fin.readline().strip().split(' ')]
    
    def solve(self):
        k, s = self.k, self.s
        self.answer = 0
        if k == 0:
            zeroZones = re.findall('0+', s)
            for current in zeroZones:
                self.answer += (len(current) + 1) * len(current) / 2
            return

        unitIndexes = []
        for i in xrange(0, len(s)):
            if s[i] == '1':
                unitIndexes += [i]
        for i in xrange(0, len(unitIndexes) - (k - 1)):
            numberOfLeftBounds, numberOfRightBounds = 0, 0
            if i > 0:
                numberOfLeftBounds = unitIndexes[i] - unitIndexes[i - 1]
            else:
                numberOfLeftBounds = unitIndexes[i] + 1
            
            if i + k < len(unitIndexes):
                numberOfRightBounds = unitIndexes[i + k] - \
                        unitIndexes[i + k - 1]
            else:
                numberOfRightBounds = len(s) - unitIndexes[i + k - 1]
            self.answer += numberOfLeftBounds * numberOfRightBounds
 
    def printAnswer(self):
        print self.answer
        #fout = open('output.txt', 'w')
        #fout.write(str(self.answer))

task = Task()
task.getData()
task.solve()
task.printAnswer()
