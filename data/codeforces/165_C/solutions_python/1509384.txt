from sys import stdin as I

k = int(I.readline())
s = I.readline()[:-1]
n = len(s)
v = [-1]

for i in xrange(n):
    if s[i] == '1':
        v.append(i)
        
m = len(v)
v.append(n)
rv = 0
if k > 0:
    for i in xrange(1, m-k+1):
        rv += (v[i] - v[i-1]) * (v[i+k] - v[i+k-1])
else:
    for i in xrange(m):
        q = v[i+1]-v[i]-1
        rv += q*(q+1)/2
print rv
