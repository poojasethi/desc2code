k=input()
s=[c=='1' for c in raw_input()]
buf=[]
cnt,c=0,0
for ch in s:
    if ch:
        buf.append(cnt)
        cnt=0
        c+=1
    else:
        cnt+=1
buf.append(cnt)
n=len(buf)
ans=0
if k==0:
    calc=lambda x:x*(x+1)/2
    for c in buf: ans+=calc(c)
else:
    for i in xrange(n-k):
        ans+=(buf[i]+1)*(buf[i+k]+1)
print ans
