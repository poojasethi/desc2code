k = int(raw_input())
s = raw_input()

def zero(k, s):
    zero, one = 0, 0
    dis = {} # dis[i] = count of zeros before the i-th '1'
    cnt = 0
    for i in s:
        if i == '0':
            zero += 1
        else:
            one += 1
            dis[one] = zero
            zero = 0
        if k == 0:
            cnt += zero
        elif one >= k:
            cnt += dis[one - k + 1] + 1
    return cnt

print zero(k, s)
