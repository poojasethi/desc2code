k = int(raw_input())
s = raw_input().strip()

l = len(s)

ones = [-1] + [i for i in xrange(l) if s[i] == '1'] + [l]

res = 0


if k == 0:
    for i in xrange(0, len(ones) - 1):
        a = ones[i]
        b = ones[i + 1]
        n = b - a - 1
        res += (1 + n) * n / 2

else:
    for i in xrange(1, len(ones) - k):
        a = ones[i]
        b = ones[i + k - 1]
        left = a - ones[i - 1]
        right = ones[i + k] - b
        res += left * right

print res
