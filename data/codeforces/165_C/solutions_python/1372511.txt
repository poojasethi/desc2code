k = int(raw_input()) - 1
str = raw_input()
cnt2 = [-1] + [id for id, c in enumerate(str) if c == '1'] + [len(str)]
res = 0
if k == -1:
    res = reduce(lambda a, b: a + b * (b + 1) / 2, [0] + [len(s) for s in str.split('1')])
else:
    id1, id2 = 1, 1 + k
    while id2 + 1 < len(cnt2):
        res += (cnt2[id1] - cnt2[id1 - 1]) * (cnt2[id2 + 1] - cnt2[id2])
        id1, id2 = id1 + 1, id2 + 1
        
print res
