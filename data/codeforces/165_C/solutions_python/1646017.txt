if __name__ == '__main__':
    k, s = int(raw_input()), raw_input()
    ones = [i for i in xrange(len(s)) if s[i] == '1']
    ones = [-1] + ones + [len(s)]
    n = len(ones)
    ans = 0
    for i in xrange(1, n-k):
        if k == 0:
            c = ones[i]-1-ones[i-1]
            ans += c*(c-1)//2+c
        else: 
            ans += (ones[i]-ones[i-1])*(ones[i+k]-ones[i+k-1])
    print ans
