from __future__ import division

def read(func=None):
    return func(raw_input()) if func else raw_input()

################################

k = read(int)
s = read()

a = [1]
for c in s:
    if c == '0':
        a[-1] += 1
    else:
        a.append(1)

if k == 0:
    print sum(n * (n - 1) // 2 for n in a)
else:
    print sum(a * b for (a, b) in zip(a, a[k:]))
