import sys
k = int(sys.stdin.readline())
s = sys.stdin.readline().strip()
L = len(s)
cnt1 = [0]
cnt = 0
for i in xrange(L):
    if s[i] == '1':
        cnt1.append(i)
ans = 0
cnt1.append(L)
#print cnt1
try:
    for i in xrange(L):
        if k:
            ans += cnt1[cnt+k+1] - cnt1[cnt+k]
        else:
            ans += cnt1[cnt+1] - i
        if s[i] == '1':
            cnt += 1
except:
    pass
print ans
