k = int(raw_input())
s = '1'+raw_input().strip()+'1'
p = []
for i in range(0,len(s)):
	if s[i] == '1':
		p.append(i)
ans = 0
if k == 0:
	x = s.split('1')
	for i in x:
		ans += len(i)*(len(i)+1)>>1
	print ans
else:
	l = len(p)
	for i in range(k,l-1):
		ans += (p[i-k+1] - p[i-k]) * (p[i+1] - p[i])
	print ans
