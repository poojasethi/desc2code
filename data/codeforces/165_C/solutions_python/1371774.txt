k=int(raw_input())
s=raw_input()
e=list()
for i,c in enumerate('1'+s+'1'):
    if c=='1':
        e.append(i)
res=0        
for i in xrange(1,len(e)-k):
    if k!=0:
            res+=(e[i]-e[i-1])*(e[i+k]-e[i+k-1])
    else:
            res+=(e[i]-e[i-1])*(e[i+k]-e[i+k-1]-1)/2
print res
