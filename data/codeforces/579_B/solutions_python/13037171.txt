n=input()
marked=[0]*(2*n+1)
tm=[0]*(2*n+1)
dictionary={}
for i in range(2,2*n+1):
    lis=map(int,raw_input().split())
    for j in range(1,i):
        dictionary[lis[j-1]]=[i,j]
    
ans=dictionary.keys()
ans.sort()
for i in ans[::-1]:
    val1=dictionary[i][0]
    val2=dictionary[i][1]
    if marked[val1]==0 and marked[val2]==0:
        tm[val1],tm[val2]=val2,val1
        marked[val1]=marked[val2]=1
for i in tm[1:]:
    print i,
