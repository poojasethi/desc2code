#!/usr/bin/env python

class Team:
	a = 0
	b = 0
	s = 0

teams = []
visit = []
result = []

n = input()
for i in range(1,2*n+2):
	visit.append(False)
	result.append(0)

for i in range(2,2*n+1):
	j = 1
	for s in raw_input().split():
		team = Team()
		team.s = int(s)
		team.a = i
		team.b = j
		teams.append(team)
		j += 1

teams.sort(key=lambda x:x.s, reverse=True)		

for team in teams:
	if not visit[team.a] and not visit[team.b]:
		visit[team.a] = True
		visit[team.b] = True
		result[team.a] = team.b
		result[team.b] = team.a

for i in range(1,2*n+1):
	print result[i]
