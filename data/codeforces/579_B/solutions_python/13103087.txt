t = int(raw_input())
dict={}
l=[]
for i in range(2, (2*t)+1):
    j = 1
    arr=map(int, raw_input().split())
    for n in arr:
        dict[n] = [i,j]
        j+=1
        l.append(n)
# print dict
l.sort()
team=[0]*(t+t+1)
k=l[::-1]
for i in k:
    p = dict[i]
    # print i, p
    a=p[0]
    b=p[1]
    if team[a] == 0 and team[b] == 0:
        team[a] = b
        team[b] = a
str1=""
for i in range(1, t+t+1):
    str1 += str(team[i])+" "
print str1
