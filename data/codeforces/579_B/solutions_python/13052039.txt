n=int(raw_input())
list1=[0]*(2*n+1)
list2=[]
for i in range(2,2*n+1):
    line=raw_input()
    k=line.strip().split(' ')
    for j in range(1,len(k)+1):
        list2.append((int(k[j-1]),i,j))
list2.sort(key=lambda x:x[0],reverse=True)
for s in list2:
    if (list1[s[1]]!=0 or list1[s[2]]!=0):
        continue
    else:
        list1[s[1]]=s[2]
        list1[s[2]]=s[1]
for i in range(1,len(list1)):
    print list1[i]
