N = int(raw_input())
tuple = []
for i in xrange(2 * N - 1):
	inp = map(int, raw_input().split())
	for k in xrange(i + 1):
		tuple.append((inp[k], i + 1, k))

tuple.sort(reverse=True)
checked = [-1] * (2 * N)

for tup in tuple:
	if checked[tup[1]] == -1 and checked[tup[2]] == -1:
		checked[tup[1]] = tup[2] + 1
		checked[tup[2]] = tup[1] + 1

print " ".join(map(str, checked))
