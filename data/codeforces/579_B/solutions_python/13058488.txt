n = int(raw_input()) * 2

ss = []
partners = [-1] * n

for i in xrange(n-1):
    x = [int(x) for x in raw_input().split()]

    for j in xrange(i+1):
        ss.append((x[j], i+1, j))

ss.sort(key=lambda x: x[0], reverse=True)

for s in ss:
    if partners[s[1]] == -1 and partners[s[2]] == -1:
        partners[s[1]] = s[2]
        partners[s[2]] = s[1]

print " ".join([str(partner + 1) for partner in partners])
