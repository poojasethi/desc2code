n = int(raw_input())
tuples = []
for i in range(2 * n - 1):
  for j, score in enumerate(map(int, raw_input().split())):
    tuples.append((-score, i + 1, j))
tuples.sort()
result = 2 * n * [ None ]
for score, i, j in tuples:
  if result[i] == None and result[j] == None:
    result[i] = j + 1
    result[j] = i + 1
print(' '.join(map(str, result)))
