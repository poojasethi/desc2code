n = input()
visited =[0]*(2*n+1)
ans = [0]*(2*n+1)
dictio = {}

for x in xrange(2,2*n+1):
	st = map(int,raw_input().split())
	for j in xrange(1,x):
		dictio[st[j-1]] = [x,j]

result = dictio.keys()
result.sort(reverse = True)
for x in result:
	if visited[dictio[x][0]]==0 and visited[dictio[x][1]]==0:
		ans[dictio[x][0]] = dictio[x][1]
		ans[dictio[x][1]] = dictio[x][0]
		visited[dictio[x][0]]=1
		visited[dictio[x][1]]=1
for x in xrange(1,len(ans)):
