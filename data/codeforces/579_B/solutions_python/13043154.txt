n = int(raw_input())
power = []

for i in xrange(2, 2*n + 1):
	p = [(int(v), i, j) for j, v in enumerate(raw_input().split(), 1)]
	power.extend(p)

power.sort(key=lambda t: t[0], reverse=True)
takenn = 0

pairs = [None] * (2 * n)

for v, i, j in power:
	if pairs[i - 1] is None and pairs[j - 1] is None:
		pairs[i - 1] = j
		pairs[j - 1] = i
		takenn += 2
	if takenn == 2*n:
		break

print(' '.join(str(v) for v in pairs))
