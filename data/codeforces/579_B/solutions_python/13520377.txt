n = input()
a = []
for i in range(1, 2 * n):
    b = map(int, raw_input().split())
    for j in range(i):
        a.append(tuple([b[j], i, j]))
a.sort(reverse=True)
taken = [-1] * (2 * n)
for i in a:
    if taken[i[1]] == -1 and taken[i[2]] == -1:
        taken[i[1]] = i[2]
        taken[i[2]] = i[1]

taken = map(lambda x: x + 1, taken) 
print str(taken)[1: -1].replace(',', '')
