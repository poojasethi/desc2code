#In the name of God

n = int(raw_input())
streng = []
for i in range(2*n-1):
	for j, strn in enumerate(map(int, raw_input().split())):
		streng.append((i+1, j, strn))


#print streng
streng.sort(key=lambda x: x[2])
#print streng
seen = [0 for i in range(2*n)]

for i in range(len(streng)-1, -1, -1):
	(x,y,st) = streng[i]
	if not seen[x] and not seen[y]:
		seen[x] = y+1
		seen[y] = x+1
#print seen
print " ".join(map(str, seen))
