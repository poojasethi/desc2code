n=input()
vis=[0]*(2*n+1)
mate=[0]*(2*n+1)
dic={}
for i in range(2,2*n+1):
    lis=map(int,raw_input().split())
    for j in range(1,i):
        dic[lis[j-1]]=[i,j]
    
res=dic.keys()
res.sort(reverse=True)
for val in res:
    if vis[dic[val][0]]==0 and vis[dic[val][1]]==0:
        mate[dic[val][0]],mate[dic[val][1]]=dic[val][1],dic[val][0]
        vis[dic[val][0]]=vis[dic[val][1]]=1
for i in mate[1:]:
    print i,
