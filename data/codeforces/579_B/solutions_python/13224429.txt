import sys

n = int(raw_input())
pairings = []
for i in xrange(2, 2 * n + 1):
    line = map(int, raw_input().strip().split(' '))
    for j in xrange(1, len(line) + 1):
        pairings.append((line[j - 1], i, j))
paired = {}
pairings.sort(key=lambda x: x[0])
found = 0
pairingspos = len(pairings) - 1
adjmatrix = [-1] * n * 2
while found < n:
    while pairingspos >= 0:
        if pairings[pairingspos][1] not in paired and pairings[pairingspos][2] not in paired:
            found += 1
            break
        pairingspos -= 1
    assert pairingspos >= -1
    currpairpos = pairingspos
    paired[pairings[currpairpos][1]] = True
    paired[pairings[currpairpos][2]] = True

    adjmatrix[pairings[currpairpos][1] - 1] = pairings[currpairpos][2]
    adjmatrix[pairings[currpairpos][2] - 1] = pairings[currpairpos][1]

    pairingspos -= 1
print ' '.join(map(str, adjmatrix))
