from math import *
PI = 3.1415926535898
n = input()
pep = []
res = [-1]*2*n
for i in range(1, 2*n):
    s = map(int, raw_input().strip().split(' '))
    for j in range(len(s)):
        pep.append((s[j], i, j))
pep.sort()
pep = pep[::-1]
for z in pep:
    if res[z[1]] == -1 and res[z[2]] == -1:
        res[z[1]] = z[2]
        res[z[2]] = z[1]
ans = ""
for x in res:
    if len(ans) > 0:
        ans += " "
    ans += str(x+1)
print ans
