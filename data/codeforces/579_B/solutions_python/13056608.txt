n = int(raw_input())
m = list()
ans = [-1] * 2 * n
for i in xrange(1, 2 * n):
  m += map(lambda (j, s): (int(s), i, j), enumerate(raw_input().split()))
m = sorted(m, key=lambda t:t[0], reverse=True)
for (p, i, j) in m:
  if ans[i] == -1 and ans[j] == -1:
    ans[i] = j + 1
    ans[j] = i + 1
  
print " ".join(map(str, ans))
