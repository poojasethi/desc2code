import heapq

if __name__ == '__main__':
    a = []
    n = input()
    for i in range(2, 2*n + 1):
        tmp = raw_input().split()
        for j in range(i - 1):
            a.append((-int(tmp[j]), i, j+1))

    heapq.heapify(a)

    out = ['0']*(2*n)
    while a:
        val, i, j = heapq.heappop(a)
        if out[i-1] == '0' and out[j-1] == '0':
            out[i-1] = str(j)
            out[j-1] = str(i)

    print ' '.join(out)
