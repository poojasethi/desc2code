n = int(raw_input()) * 2
teams = list()
for i in range(1, n):
    line = raw_input().split()
    for j in range(0, i):
        teams.append((int(line[j]), i, j))
partner = [0 for i in range(n)]

for team in sorted(teams, reverse=True):
    if not partner[team[1]] and not partner[team[2]]:
        partner[team[1]] = team[2] + 1
        partner[team[2]] = team[1] + 1
