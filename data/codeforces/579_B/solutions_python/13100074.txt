import collections
n = int(raw_input()) * 2
tmp_num = 2

tmp_dict = dict()
for i in range(2, n+1):
    for j, dis in enumerate(raw_input().split()):
        tmp_dict[int(dis)]= (i, j+1)

graph = collections.OrderedDict(sorted(tmp_dict.items(), reverse=True))

ans = dict()

for k, v in graph.iteritems():
    if v[1] not in ans and v[0] not in ans:
        ans[v[0]] = v[1]
        ans[v[1]] = v[0]


result = ''
for v in ans.itervalues():
    result +=str(v) + ' '
print result
