n = int(raw_input())

data = []

for i in range(2, 2*n+1):
    tmp = raw_input().split()
    for j in range(1, i):
        data.append((int(tmp[j-1]), (i, j)))

data.sort(key = lambda x: -x[0])

res = [0 for i in range(0, 2*n+1)]

for i in range(0, len(data)):
    _x = data[i][1][0]
    _y = data[i][1][1]
    if res[_x] is not 0 or res[_y] is not 0:
        continue
    else:
        res[_x] = _y
        res[_y] = _x
