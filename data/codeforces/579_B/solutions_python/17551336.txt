n = input()
score = []



for i in range(2,2*n+1):
    k = map(int,raw_input().split())
    for j in range(1,i):
        score.append([k[j-1],j,i])

score = sorted(score,key = lambda x:x[0],reverse = True)
pairs = [0]*(2*n)

for i in score:
    if pairs[i[1]-1] == pairs[i[2]-1] == 0:
        pairs[i[1]-1] = i[2]
        pairs[i[2]-1] = i[1]


print ' '.join(str(i) for i in pairs)
