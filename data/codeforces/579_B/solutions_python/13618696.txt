import sys

n = input()

i = 1
res = []
for line in sys.stdin:
	i += 1
	a = map(int, line.split())
	j = 0
	for strength in a:
		j += 1
		res.append([strength, (i,j)])
res = sorted(res, reverse=True)

pos = [0]*(2*n)
st = set()

cnt = 0

for p in res:
	if cnt == n:
		break
	if p[1][0] not in st and p[1][1] not in st:
		st.add(p[1][0])
		st.add(p[1][1])
		pos[p[1][0]-1] = p[1][1]
		pos[p[1][1]-1] = p[1][0]
		cnt += 1
	else:
		continue

print " ".join(map(str,pos))
