n = input()
edges = []
for i in xrange(2,2*n+1):
    edl = map(int,raw_input().split())
    for j in xrange(1,len(edl)+1):
        edges.append((edl[j-1],i,j))

edges = sorted(edges, reverse=True)
d = ['0']*(2*(n+2))
for e in edges:
    x,y = e[1],e[2] 
    if d[x] is '0' and d[y] is '0': 
        d[x] = str(y)
        d[y] = str(x)
