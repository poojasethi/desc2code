 #include<cstdio>
 #include<algorithm>
 using namespace std;
 int n,tot,p[810];
 struct E
 {
 	int u,v,w;
 }e[640010];
 bool com(const E & a, const E & b)
 {
 	return a.w>b.w;
 }
 int main()
 {

 	scanf("%d",&n);
 	n<<=1; int t;
 	for (int i=1;i<=n;i++)
 	for (int j=1;j<i;j++)
 	{
 		scanf("%d",&t);
 		e[++tot].u=i; e[tot].v=j; e[tot].w=t;
 	}
 	sort(e+1,e+1+tot,com);
 	for (int i=1;i<=tot;i++)
 	if (p[e[i].u]==0&&p[e[i].v]==0)
 	{
 		p[e[i].u]=e[i].v;
 		p[e[i].v]=e[i].u;
 	}
 	for (int i=1;i<=n;i++) printf("%d ",p[i]);
 	return 0;
 }
