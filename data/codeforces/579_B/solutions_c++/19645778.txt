#include <iostream>
#include <cstdio>
#include <cstring>
#include <queue>
#include <algorithm>

#define N 1010

using namespace std;

struct edge{
	int x,y,v;
}a[N*N];

int n,m,b[N];

bool cmp(edge a,edge b){
	return a.v<b.v;
}

int main(){
	cin>>n;
	n*=2;
	for(int i=2;i<=n;i++)
		for(int j=1,x;j<i;j++){
			cin>>x;
			a[++m]=(edge){i,j,x};
		}
	sort(a+1,a+m+1,cmp);
	for(int i=m;i>=1;i--)
		if(!b[a[i].x] && !b[a[i].y]){
			b[a[i].x]=a[i].y;
			b[a[i].y]=a[i].x;
		}
	for(int i=1;i<n;i++) cout<<b[i]<<' ';
	cout<<b[n]<<endl;
	return 0;
}
