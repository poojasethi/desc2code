#include<bits/stdc++.h>
using namespace std;
int a[1000008][2],us[1002],ans[1002];
int main()
{


 long long int i,j,k=0,n;
 cin>>n;
 n=n*2;
for(i=0;i<1002;i++)us[i]=0;
 for(i=1;i<=n;i++)
 {

     for(j=1;j<=i-1;j++)
     {
         cin>>k;
         a[k][0]=i;
         a[k][1]=j;
     }
 }
 for(i=1000000;i>0;i--)
 {
     if(us[a[i][0]]==0&&us[a[i][1]]==0)
     {
         us[a[i][0]]=us[a[i][1]]=1;
         ans[a[i][0]]=a[i][1];
         ans[a[i][1]]=a[i][0];
     }

 }
   for(i=1;i<=n;i++)cout<<ans[i]<<" ";

    return 0;
}
