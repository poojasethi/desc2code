# include <cstdio>
# define MAX 1000000
int ans[810], a[MAX + 10], b[MAX + 10];
int main ()
{
    register int n, x;
    scanf ("%d", &n);
    for (register int i = 1; i <= (n << 1); ++i)
        for (register int j = 1; j < i; ++j)
            scanf ("%d", &x),
            a[x] = i,
            b[x] = j;
    for (register int i = MAX; i; --i)
        if (a[i] && !ans[a[i]] && !ans[b[i]])
            ans[a[i]] = b[i],
            ans[b[i]] = a[i];
    for (register int i = 1; i <= (n << 1); ++i)
        printf ("%d ", ans[i]);
    return 0;
}
