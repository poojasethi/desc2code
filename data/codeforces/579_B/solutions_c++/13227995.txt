#include <iostream>
#include <algorithm>

using namespace std;
/*Sort based on value then pair up*/

#define N 1000001
int a[N][2];
bool used[N];
int p[N];
int main(){
	int n;
	cin>>n;
	n*=2;
	int x;
	for(int i=1;i<=n;i++){

		for(int j=1;j<i;j++){
			cin>>x;
			a[x][0]=i;
			a[x][1]=j;
		}
	}
	for(int i=1;i<=n+1;i++)
		used[i] = false;

	for(int i=N-1;i>=0;i--) {
		if(used[a[i][0]] == false && used[a[i][1]] == false){
			used[a[i][0]]=true;
			used[a[i][1]] =true;
			p[a[i][0]]= a[i][1];
			p[a[i][1]] = a[i][0];
		}

	}

	for(int i=1;i<=n;i++)
		cout<<p[i]<<" ";




	return 0;
}
