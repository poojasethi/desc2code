#include<bits/stdc++.h>
using namespace std;
struct person
{
	int x,y,val;
}a[500000];

int ans[1000];

int cmp(person a,person b)
{
	return a.val>b.val;
}
int main()
{
	int n;
	int k=0;
	scanf("%d",&n);
	for(int i=2;i<=2*n;i++)
	{
		for(int j=1;j<i;j++)
		{
			scanf("%d",&a[k].val);
			a[k].x=i;
			a[k++].y=j;
		}
	}
	sort(a,a+k,cmp);
	for(int i=0;i<k;i++)
	{
		if(ans[a[i].x]==0 && ans[a[i].y]==0)
		{
			ans[a[i].x]=a[i].y;
			ans[a[i].y]=a[i].x;
		}
	}
	for(int i=1;i<=2*n;i++)
		printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
