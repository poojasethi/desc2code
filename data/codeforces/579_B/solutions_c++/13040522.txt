#include <bits/stdc++.h>

using namespace std;

#define ll long long int

int main()
{
	ll n,a,b,j,k,i;
	cin>>n;
	n=2*n;
	ll A[n];
	for(i=0;i<n;i++)
		A[i]=0;
	set<pair< ll,pair<ll,ll> > > S;
	for(i=2;i<=n;i++)
	{
		for(j=1;j<i;j++)
		{
			cin>>a;
			S.insert(make_pair(-a,make_pair(j,i)));
		}
	}
	while(S.size())
	{
		pair<ll,pair<ll,ll> > P=*(S.begin());
		if(!A[P.second.first-1] && !A[P.second.second-1])
		{
			A[P.second.first-1] = P.second.second;
			A[P.second.second-1] = P.second.first;
		}
		S.erase(P);
	}
	for(i=0;i<n;i++)
		cout<<A[i]<<" ";
	cout<<endl;
	return 0;
}
