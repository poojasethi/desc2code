#include<bits/stdc++.h>
using namespace std;

int p[1000001][2],an[1001];
bool used[1001];

int main()
{
	int n,x;
	scanf("%d",&n);
	n*=2;
	for(int i=1;i<=n;i++)
		for(int j=1;j<i;j++)
		{
			scanf("%d",&x);
			p[x][0]=i;
			p[x][1]=j;
		}
			used[0]=1;
	cout<<endl;
	for(int i=1000000;i>0;i--)
	{
		if(!used[p[i][0]]&&!used[p[i][1]])
		{
			used[p[i][0]]=1;
			used[p[i][1]]=1;
			an[p[i][0]]=p[i][1];
			an[p[i][1]]=p[i][0];
		  }
	}
	for(int i=1;i<=n;i++)
	cout<<an[i]<<" ";
    return 0;
}
