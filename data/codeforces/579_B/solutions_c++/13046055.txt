#include <iostream>

using namespace std;

int team[900],check[2000000],n;

int main()
{
    cin >> n;
    for (int i=2;i<=2*n;i++)
    {
        int tmp;
        for (int j=1;j<i;j++)
            cin >> tmp,check[tmp]=i*900+j;
    }
    for (int i=1000000;i>=0;i--)
    {
        if (!check[i]) continue;
        int x=check[i]/900,y=check[i]%900;
        if (team[y] or team[x]) continue;
        team[x]=y,team[y]=x;
    }
    for (int i=1;i<=2*n;i++)
        cout << team[i] << " ";
}
