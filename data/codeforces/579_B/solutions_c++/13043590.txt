#include <bits/stdc++.h>

using namespace std;

int a[810][810], p[1000], x, y, n, m;
bool u[1000];

int main(){
 	cin>>m;
 	n = 2*m;
 	for(int i=1; i<n; i++)
 		for(int j=0; j<i; j++) {
 			cin>>x;
 			a[i][j] = x;
 			a[j][i] = x;
 		}
 	for(int k=0; k<m; k++) {
		x=0, y=0;
		for(int i=0; i<n; i++)
		 if(!u[i])
	 	  for(int j=0; j<n; j++)
	  	   if(!u[j] && a[x][y]<a[i][j]) x = i, y = j;
	  	u[x] = u[y] = true;
	  	p[x] = y;
	  	p[y] = x;
	}
 	for(int i=0; i<n; i++) cout<<p[i]+1<<" ";
    return 0;
}
