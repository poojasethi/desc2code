#include <cstdio>
#define Z 1000000
int n,a[Z+5],b[Z+5],s[805];
int main(){
	int i,j,t;
	scanf("%d",&n);
	for(i=2; i<=2*n; i++){
		for(j=1; j<i; j++){
			scanf("%d",&t);
			a[t] = i;
			b[t] = j;
		}
	}
	for(i=Z; i; i--){
		if(a[i] && !s[a[i]] && !s[b[i]]){
			s[a[i]] = b[i];
			s[b[i]] = a[i];
		}
	}
	for(i=1; i<=2*n; i++) printf("%d ",s[i]);
	return 0;
}
