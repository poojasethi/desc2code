#include <bits/stdc++.h>
using namespace std;

int n,a,b[1000001],c[1000001],team[801];

int main()
{
    cin>>n;
    for(int i=2; i<=n*2; i++)
        for(int j=1; j<i; j++)
        {
            cin>>a;
            b[a]=i;
            c[a]=j;
        }
    for(int i=1000000; i>0; i--)
        if(b[i]>0&&team[b[i]]==0&&team[c[i]]==0)
        {
            team[b[i]]=c[i];
            team[c[i]]=b[i];
        }
    for(int i=1; i<n*2; i++)
        cout<<team[i]<<" ";
    cout<<team[n*2]<<endl;
}
