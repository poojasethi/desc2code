#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,k;
	cin>>n;
	vector<int> s;
	set<int> e;
	map<int,pair<int,int> > m;
	int a[1000][1000];
	for(int i=1;i<(2*n);i++){
		for(int j=0;j<i;j++){
			cin>>k;
			s.push_back(k);
			m[k]=make_pair(i,j);
		}
	}
	int w[1000];
	sort(s.begin(),s.end());
	reverse(s.begin(),s.end());
	for(int i=0;i<s.size();i++){
		int a=m[s[i]].first;
		int b=m[s[i]].second;
		
		if(e.count(a)==0&&e.count(b)==0){
			e.insert(a);
			e.insert(b);
			w[a]=b;
			w[b]=a;
		}
	}
	for(int i=0;i<2*n;i++){
		cout<<w[i]+1<<" ";
	}
	cout<<endl;
	return 0;
}
