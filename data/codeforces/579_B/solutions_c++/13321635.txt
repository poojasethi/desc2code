#include<bits/stdc++.h>
#define mp make_pair
#define st first
#define nd second
using namespace std;
priority_queue < pair< int,pair<int,int> > > Q;
int n,i,t,h[805],m,x,j,y,a;
int main()
{
	cin>>n;
	for(i=2;i<=2*n;i++)
		for(j=1;j<i;j++)
		{
			scanf("%d",&m);
			Q.push( mp(m, mp(i,j) ) );
		}
	while(t<n)
	{
		m=Q.top().st;
		x=Q.top().nd.st;
		y=Q.top().nd.nd;
		Q.pop();
		if(!h[x] and !h[y])
		{h[y]=x;h[x]=y;t++;}
	}
	for(i=1;i<=n*2;i++)
		cout<<h[i]<<' ';
}
