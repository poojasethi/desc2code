#include<bits/stdc++.h>
using namespace std;
int person[1000001][2];
bool used[801]; int ans[801];
int main()
{
	int n; scanf("%d",&n);
	n=n*2;
	for(int i=2;i<=n;i++)
		for(int j=1;j<i;j++) {
			int x;
			scanf("%d",&x);
			person[x][0]=i;
			person[x][1]=j;
		}
	used[0]=1;
	for(int x=1e6;x>0;x--)
		if(!used[person[x][0]] && !used[person[x][1]]) {
			used[person[x][0]]=1;
			used[person[x][1]]=1;
			ans[person[x][0]]=person[x][1];
			ans[person[x][1]]=person[x][0];
		}
	for(int i=1;i<=n;i++)
		printf("%d ",ans[i]);
	printf("\n"); return 0;
}
