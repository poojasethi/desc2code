#include<bits/stdc++.h>
using namespace std;


int main() {
int n;
cin>>n;
vector<int>v;
vector<pair<int ,int> >vec;
map<int,pair<int,int> >m;
map<int,int >mp;
int x;
for(int j=2;j<=(2*n);j++)
{
for(int i=1;i<j;i++)
{
    cin>>x;
    v.push_back(x);
    m[x]=make_pair(j,i);
}
}
sort(v.begin(),v.end());
int c=0;
for(int i=v.size()-1;i>=0;i--)
{
    if(!mp[m[v[i]].first] && !mp[m[v[i]].second] )
    {
mp[m[v[i]].first]=m[v[i]].second;
mp[m[v[i]].second]=m[v[i]].first;
}
}
for(int i=1;i<=2*n;i++)
    cout<<mp[i]<<" ";







}
