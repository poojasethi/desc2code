#include<cstdio>
int k,a[810][810],n,i,j,ans[810],Max,maxi,maxj;
int main(){
	scanf("%d",&n);
	for(i=2;i<=n*2;i++)
		for(j=1;j<i;j++)
			scanf("%d",&a[i][j]);
	for(k=1;k<=n;k++){
		Max=0;
		for(i=2;i<=n*2;i++)
			for(j=1;j<i;j++)
				if(!ans[i]&&!ans[j]&&a[i][j]>Max){
					Max=a[i][j];
					maxi=i;
					maxj=j;
				}
		ans[maxi]=maxj;
		ans[maxj]=maxi;
	}
	for(i=1;i<=n*2;i++)
		printf("%d ",ans[i]);
}
