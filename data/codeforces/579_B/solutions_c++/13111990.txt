#include<iostream>

using namespace std;

int ans[801]={0};
int n,i,j,l,pi,pj,maxx;
int buffer[801][801];
	
int main()
{
	cin >> n;
	for(i=2;i<=n*2;i++) for(j=1;j<i;j++) cin >> buffer[i][j];
	for(l=0;l<n;l++){
		maxx=0;
		for(i=2;i<=n*2;i++){
			if(ans[i]) continue;
			for(j=1;j<i;j++){
				if(ans[j]) continue;
				if(buffer[i][j]>maxx){
					maxx=buffer[i][j];
					pi=i;
					pj=j;
				}
			}
		}
		ans[pi]=pj;
		ans[pj]=pi;
	}
	for(i=1;i<=n*2;i++) cout << ans[i] << " ";
	return 0;
}
