#include<bits/stdc++.h>
using namespace std;
vector<pair<int,pair<int,int> > >v;
map<int,int>m;
int main(){
	int n,x;
	cin>>n;
	for(int i=2 ; i<=2*n ; i++)
	{
		for(int j=1 ; j<i ; j++)
		{
			cin>>x;
			v.push_back(make_pair(x,make_pair(i,j)));
		}		
	}
	sort(v.begin(),v.end());
	for(int i=v.size()-1 ; i>=0 ; i--)
	{
		if(m[v[i].second.first]==0 &&m[v[i].second.second]==0)
		{
			m[v[i].second.first]=v[i].second.second;
			m[v[i].second.second]=v[i].second.first;
		}
	}
	for(int i=1 ; i<=2*n ; i++)	cout<<m[i]<<" ";
}
