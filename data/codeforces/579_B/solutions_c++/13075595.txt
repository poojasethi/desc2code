#include<bits/stdc++.h>
using namespace std;

#define fi(i,a,b)   for(long long i=a;i<=b;i++)
#define fd(i,a,b)   for(long long i=b;i>=a;i--)
#define clr(x)      memset(x,0,sizeof(x))
#define __          printf(" ")
#define _           printf("\n")

long long I[1000001],J[1000001];

int main(){
  long long ans[1000],n,x;
  clr(I),clr(J),clr(ans);
  cin>>n;
  fi(i,2,2*n) fi(j,1,i-1) {
     cin>>x;
     I[x]=i;
     J[x]=j;
  }
  fd(i,1,1000000) if(!ans[I[i]]&&!ans[J[i]]) ans[I[i]]=J[i],ans[J[i]]=I[i];
  fi(i,1,2*n) cout<<ans[i]<<' ';
  return 0;
}
