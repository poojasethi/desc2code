#include <bits/stdc++.h>
 int n,i,j,b[1000000],k=0,p,mark[1000000]={0},ans[1000000]={0};
using namespace std;
struct node {
  int x,y;
} a[1000001];
int main()
{
 
  cin>>n;
  for(i=0;i<2*n-1;i++)
  {
    for(j=0;j<i+1;j++)
    {
      cin>>p;
      a[p].x=i+2; a[p].y=j+1;
      b[k++]=p;
    }

}
sort(b,b+k);
for(i=k-1;i>=0;i--)
{
    int e = a[b[i]].x; int f =a[b[i]].y;
    
    if(mark[e]==0&&mark[f]==0)
    {
        ans[e]=f;
        ans[f]=e;
        mark[e]=1;
        mark[f]=1;
    }

    
}
for(i=1;i<=2*n;i++)
    cout<<ans[i]<<" ";
}
