#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
struct node {int x,id1,id2;} a[640001];
int ans[801],cnt=0;
bool cmp(node x,node y) {return x.x>y.x;}
int main()
{
	int n,i,j,x;
	
	scanf("%d",&n);n*=2;
	for(i=2;i<=n;i++) for(j=1;j<i;j++) {scanf("%d",&x);a[++cnt]=(node){x,i,j};}
	sort(a+1,a+cnt+1,cmp);
	for(i=1;i<=cnt;i++)
		if(ans[a[i].id1]==0&&ans[a[i].id2]==0)
		{
			ans[a[i].id1]=a[i].id2;
			ans[a[i].id2]=a[i].id1;
		}
	for(i=1;i<=n;i++) printf("%d ",ans[i]);
}
