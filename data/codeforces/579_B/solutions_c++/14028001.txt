#include<bits/stdc++.h>
using namespace std;
const int N=1000001;
long long i,j,m,n,a[N],b[N],c[N];
int main()
{
cin>>n;
for(i=2;i<=2*n;i++)
for(j=1;j<i;j++)
{
cin>>m;
a[m]=i;
b[m]=j;
}
for(i=N-1;i>=0;i--)
{
if(a[i]&&!c[a[i]]&&!c[b[i]]){c[a[i]]=b[i];c[b[i]]=a[i];}
}
for(i=1;i<=2*n;i++)cout<<c[i]<<' ';
}
