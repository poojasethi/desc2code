#include <iostream>
#include <map>
#include <cmath>
using namespace std;
long long int maxi,maxj,used[1000000],n,k,a[820][820];

int main(){
	cin>>n;
	for(int i=2;i<=n*2;i++)
		for(int j=1;j<i;j++){
			cin>>a[i][j];
		}
	a[0][0]=0;
	for(int t=1;t<=n;t++){
		for(int i=2;i<=n*2;i++){
			for(int j=1;j<i;j++){
				if(a[i][j]>a[maxi][maxj] && !used[i] && !used[j]){
					maxi=i;
					maxj=j;
				}
			}
		}
		used[maxi]=maxj;
		used[maxj]=maxi;
		maxi=0;maxj=0;
	}
	for(int i=1;i<=n*2;i++)cout<<used[i]<<" ";
	return 0;
}
