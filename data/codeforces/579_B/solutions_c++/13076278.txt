#include<bits/stdc++.h>
using namespace std;

int x[1000001][2],Taken[801],Fin[801];

int main()
{
        int n; cin >> n; n*=2; for(int i=2;i<=n;i++) for(int j=1;j<i;j++) { int a; scanf("%d",&a); x[a][0]=i; x[a][1]=j; }
        for(int i=1000000;i>0;i--)
            if(!(Taken[x[i][0]] || Taken[x[i][1]]))
                Taken[x[i][0]] = Taken[x[i][1]] = 1,Fin[x[i][0]] = x[i][1],Fin[x[i][1]] = x[i][0];
        for(int i=1;i<=n;i++)
            printf("%d ",Fin[i]);
        return 0;
}
