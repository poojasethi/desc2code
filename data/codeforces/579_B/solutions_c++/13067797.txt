#include <iostream>
#include <cstdio>
using namespace std;

int n, cnt, r[1000005], c[1000005], ans[805];
int main()
{
	int i, j, t;
//	freopen ("input.txt", "r", stdin);
	cin >> n;
	for (i = 2; i <= n * 2; i++) {
		for (j = 1; j < i; j++) {
			scanf ("%d", &t);
			r[t] = i;
			c[t] = j;
		}
	}
	for (i = 1000000; i >= 1; i--) {
		if (r[i] != 0 && ans[r[i]] == 0 && ans[c[i]] == 0) {
			ans[r[i]] = c[i];
			ans[c[i]] = r[i];
			cnt++;
			if (cnt == n) break;
		}
	}
	for (i = 1; i <= n * 2; i++) {
		printf("%d ", ans[i]);
	}
	return 0;
}
