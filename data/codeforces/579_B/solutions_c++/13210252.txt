#include<iostream>
#include<cstdio>
using namespace std;
int a[1000001][2],ans[805];

int main(){
	int n,x;
	scanf("%d",&n);
	for(int i=2;i<=2*n;i++){
		for(int j=1;j<i;j++){
			scanf("%d",&x);
			a[x][0]=i;
			a[x][1]=j;
		}
	}
	for(int i=1000000;i>0;i--)
		if(ans[a[i][0]]==0 && ans[a[i][1]]==0){
			ans[a[i][0]]=a[i][1];
			ans[a[i][1]]=a[i][0];
		}
	for(int i=1;i<=2*n;i++) cout<<ans[i]<<" ";
	cout<<endl;	return 0;
}
