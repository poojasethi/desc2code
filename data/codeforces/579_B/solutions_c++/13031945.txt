#include<bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin>>n;
	vector<pair<int,pair<int,int> > >in;
	for(int i=2;i<=2*n;i++){
		for(int j=1;j<i;j++){
			int temp;
			cin>>temp;
			in.push_back(make_pair(-temp,make_pair(j,i)));
		}
	}
	sort(in.begin(),in.end());
	int pairs[1002];
	memset(pairs,-1,sizeof pairs);
	
	for(int i=0;i<in.size();i++){
		int a,b;
		a=in[i].second.first;
		b=in[i].second.second;
		if(pairs[a]==-1 && pairs[b]==-1){
			pairs[a]=b,pairs[b]=a;
		}
	}
	for(int i=1;i<=2*n;i++){
		cout<<pairs[i]<<" ";
	}
	
	return 0;
}
