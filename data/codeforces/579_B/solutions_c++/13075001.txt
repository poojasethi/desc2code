#include <bits/stdc++.h>
#include <limits>
#include <algorithm>
using namespace std;

int a[1000005][5];

int main(){
	
	int n,x,pair[1000]={0};
	cin>>n;
	for(int i=1;i<=2*n;i++){
		for(int j=1;j<i;j++){
			cin>>x;
			a[x][0]=i;
			a[x][1]=j;
		}
	}
	for(int i=1000000;i>0;i--){
		if(pair[a[i][0]]==0 && pair[a[i][1]]==0){
			pair[a[i][0]]=a[i][1];
			pair[a[i][1]]=a[i][0];
		}
	}
	for(int i=1;i<=2*n;i++)	cout<<pair[i]<<" ";
	return 0;
}
