#include <bits/stdc++.h>
using namespace std;
vector<int> v[1000000 + 5];
int hash[1000000 + 5];

int main(){
	int n;
	scanf("%d",&n);
		for(int j = 2;j <= 2*n;j++){
			for(int k = 1;k < j;k++){
				int ai;
				scanf("%d",&ai);
				v[ai].push_back(j);
				v[ai].push_back(k);
			}
		}

	for(int i = 1000000;i >= 0;i--){
		if(v[i].size() > 0){
			if(hash[v[i][0]] == 0 && hash[v[i][1]] == 0)
			{	hash[v[i][0]] = v[i][1];
				hash[v[i][1]] = v[i][0];
			}
		}
	}
	for(int i = 1;i <= 2*n;i++){
		printf("%d ",hash[i]);
	}
	return 0;
}
