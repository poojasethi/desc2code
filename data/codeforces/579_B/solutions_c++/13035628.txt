#include<bits/stdc++.h>
using namespace std;
typedef struct
{
int a,b;
}S;
S s;
int arr[1000000];
typedef map<int,S> M;
M m;
int main()
{
int n,x;
cin>>n;
for(int i=2;i<=n*2;i++)
{
for(int j=1;j<i;j++)
{
cin>>x;
m[x].a=i;
m[x].b=j;
}
}
M::reverse_iterator it;
for(it=m.rbegin();it!=m.rend();it++)
{
x=it->first;
if(arr[m[x].a]==0&&arr[m[x].b]==0)
{
arr[m[x].a]=m[x].b;
arr[m[x].b]=m[x].a;
}
}
for(int i=1;i<1000000;i++)
{
if(arr[i]!=0)
cout<<arr[i]<<" ";
}
return 0;
}
