#include<bits/stdc++.h>
using namespace std;

vector<pair<int,pair<int,int> > >v;
int visited[801];
int main()
{
int n,i,x;
cin>>n;
int k=2;
for(i=0;i<2*n-1;i++)
{

for(int j=1;j<k;j++)
cin>>x,v.push_back(make_pair(x,make_pair(j,k)));

k++;
}
sort(v.begin(),v.end());

for(i=v.size()-1;i>=0;i--)
{
int x=v[i].second.first;
int y=v[i].second.second;
if(!visited[x]&&!visited[y])
{
visited[x]=y;
visited[y]=x;

}
}

for(i=1;i<=2*n;i++)
cout<<visited[i]<<" ";
cout<<endl;
return 0;



}
