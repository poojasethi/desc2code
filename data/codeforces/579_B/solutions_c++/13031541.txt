#include<cstdio>
#include<algorithm>
using namespace std;

const int maxn=400;
int ans[2*maxn+1];

struct Pair{
  int x,y,w;
  void get(int a,int b){ x=a,y=b; scanf("%d",&w); }
  bool operator < (const Pair p) const {
    return w>p.w;
  }
}p[4*maxn*maxn]; int cnt;

int main(){
  int n; scanf("%d",&n);
  for(int i=2;i<=2*n;i++)
    for(int j=1;j<i;j++)
      p[cnt++].get(i,j);
  sort(p,p+cnt);
  for(int i=0;i<cnt;i++){
    int x=p[i].x,y=p[i].y;
    if(!ans[x] && !ans[y]) ans[x]=y,ans[y]=x;
  }
  for(int i=1;i<=2*n;i++) printf("%d ",ans[i]);
  return 0;
}
