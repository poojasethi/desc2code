#include <iostream>
#define Z 1000000
using namespace std;
int n,a[Z+10],b[Z+10],s[810];
int main(){
	int i,j,t;
	ios_base::sync_with_stdio(0);
	cin >> n;
	for(i=2; i<=2*n; i++){
		for(j=1; j<i; j++){
			cin >> t;
			a[t] = i;
			b[t] = j;
		}
	}
	for(i=Z; i; i--){
		if(a[i] && !s[a[i]] && !s[b[i]]){
			s[a[i]] = b[i];
			s[b[i]] = a[i];
		}
	}
	for(i=1; i<=2*n; i++) cout << s[i] << " ";;
	return 0;
}
