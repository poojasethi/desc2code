#include <bits/stdc++.h>
using namespace std;
vector<pair<int,pair<int,int> > >v;
int make[805];
int main(){
	int a,n,i,j;
	cin>>n;
	v.clear();
	for(i=2;i<=2*n;++i)
		for(j=1;j<i;++j)
			scanf("%d",&a),v.push_back(make_pair(-a,make_pair(i,j)));
	sort(v.begin(),v.end());
	int f,s,cc=0,pos=0;
	while(cc<n){
		f = v[pos].second.first,s = v[pos].second.second;
		if( (make[s])!=0 ||(make[f])!=0 ){
			++pos;continue;
		}
		make[s]=f,make[f]=s;++pos;++cc;
	}
	for(i=1;i<=2*n;++i)
		printf("%d ",make[i]);
	putchar('\n');
}
