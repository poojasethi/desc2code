#include<stdio.h>
int a[805][805],maxn,n,ans[805],h,x;
int main()
{
  scanf("%d",&n);
  for(int i=2;i<=n*2;i++)
    for(int j=1;j<i;j++)
      scanf("%d",&a[i][j]);
  for(int k=1;k<=n;k++)
  {
    maxn=0;
    for(int i=2;i<=n*2;i++)
      for(int j=1;j<i;j++)
        if(a[i][j]>maxn&&!ans[j]){maxn=a[i][j];h=i;x=j;}
      ans[h]=x;
      ans[x]=h;
      for(int i=1;i<=n*2;i++)
        a[h][i]=a[x][i]=0; 
  }
  for(int i=1;i<n*2;i++)
    printf("%d ",ans[i]);
  printf("%d\n",ans[n*2]);
  return 0;
}
