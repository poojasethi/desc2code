#!/usr/bin/env python
from sys import stdin as cin

def main():
    x1, y1, x2, y2, x3, y3 = map(int, next(cin).split())
    a = (max(x1, y1), min(x1, y1), 'A')
    b = (max(x2, y2), min(x2, y2), 'B')
    c = (max(x3, y3), min(x3, y3), 'C')

    a, b, c = sorted((a, b, c))
    n = c[0]
    r = n - c[1]

    banner = None
    if n == b[0] == a[0]:
        if r == a[1] + b[1]:
            banner = [c[2]*n]*c[1] + [b[2]*n]*b[1] + [a[2]*n]*a[1]
    else:
        if b[1] != r:
            b = b[1], b[0], b[2]
        if a[1] != r:
            a = a[1], a[0], a[2]
        if r == b[1] == a[1] and n == b[0] + a[0]:
            banner = [c[2]*n]*c[1] + [b[2]*b[0] + a[2]*a[0]]*r

    if not banner:
        return -1,

    return n, '\n'.join(banner)

print '\n'.join(map(str, main()))
