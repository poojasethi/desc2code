#include <bits/stdc++.h>
using namespace std;

int a[3][2];
char ans[200][200];

int main(){
	
	int l = 0;
	
	for( int i=0 ; i < 3 ; i++ )
		for( int j=0 ; j < 2 ; j++ ){
			scanf("%d",&a[i][j]);
			l = max(l , a[i][j]);
		}
	
	int s = 0;
	
	for( int i=0 ; i < 3 ; i++ ){
		s += a[i][0] * a[i][1];
		if( a[i][0] < a[i][1] )
			swap( a[i][0] , a[i][1] );
	}
	
	if( sqrt(s) != l ){
		puts("-1");
		return 0;
	}
	
	int row = 1;
	
	for( int i=0 ; i < 3 ; i++ )
		if( a[i][0] == l ){
			for( int j=row ; j < row+a[i][1] ; j++ )
				for( int k=1 ; k <= l ; k++ )
					ans[j][k] = 'A'+i;
			row += a[i][1];
			a[i][0] = -1;
		}
	
	int col = 1;
	
	for( int i=0 ; i < 3 ; i++ )
		if( a[i][0] != -1 ){
			if( l-row+1 == a[i][0] ){
				for( int j=row ; j <= l ; j++ )
					for( int k=col ; k < col + a[i][1] ; k++ )
						ans[j][k] = 'A'+i;
				col += a[i][1];
			} else if( l-row+1 == a[i][1] ){
				for( int j=row ; j <= l ; j++ )
					for( int k=col ; k < col + a[i][0] ; k++ )
						ans[j][k] = 'A'+i;
				col += a[i][0];
			} else{
				puts("-1");
				return 0;
			}
		}
	
	printf("%d\n",l);
	
	for( int i=1 ; i <= l ; i++ )
		puts( ans[i]+1 );
	
	return 0;
}
