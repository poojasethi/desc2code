#include<cstdio>
#define N 150
char n[N][N];
int main()
{
	int a,b,i,j,x[5],y[5],m=0,s=0;
	char z[5]={' ','A','B','C'},c;
	for(i=1;i<=3;i++)       //INPUT
	{
		scanf("%d%d",&x[i],&y[i]);
		a=(x[i]<y[i])?x[i]:y[i];
		b=(x[i]>y[i])?x[i]:y[i];
		x[i]=a;
		y[i]=b;
		s+=a*b;
		m=(m>b)?m:b;
	}
	if(s!=m*m){printf("-1");return 0;}//EVOLVE
	for(i=1;i<=3;i++)
		for(j=i;j<=3;j++)
			if(y[i]<y[j])
			{
				a=y[j];y[j]=y[i];y[i]=a;
				b=x[j];x[j]=x[i];x[i]=b;
				c=z[j];z[j]=z[i];z[i]=c;
			}
	for(i=1;i<=x[1];i++)    //EVOLVE Loge A
		for(j=1;j<=y[1];j++)n[i][j]=z[1];
	if(x[2]+x[1]==m)//EVOLVE B
	for(;i<=m;i++)
		for(j=1;j<=y[2];j++)n[i][j]=z[2];
	else if(y[2]+x[1]==m)
	for(;i<=m;i++)
		for(j=1;j<=x[2];j++)n[i][j]=z[2];
	else 
	for(a=i;a<=i-1+x[2];a++)
		for(j=1;j<=m;j++)n[a][j]=z[2];
	for(i=1;i<=m;i++)       //EVOLVE Loge C
		for(j=1;j<=m;j++)if(n[i][j]<65)n[i][j]=z[3];
	printf("%d\n",m);//OUTPUT
	for(i=1;i<=m;i++)
	{
		for(j=1;j<=m;j++)
			printf("%c",n[i][j]);
		printf("\n");
	}
	return 0;
}
