#include<cstdio>
#include<iostream>
using namespace std;
char x[405][405];
int check(int a,int b,int c,int d,int e,int f,char A,char B,char C){    
    if(a==c&&a==e&&a==b+d+f)
    {
        for(int i=0;i<a;i++)
            for(int j=0;j<a;j++)
                if(i<b)
                    x[i][j]=A;
                else if(i<b+d)
                    x[i][j]=B;
                else
                    x[i][j]=C;
        return a;
    }
    if(a==c&&e==b+d&&e==a+f)
    {
        for(int i=0;i<e;i++)
            for(int j=0;j<e;j++)
                if(i<f)
                    x[i][j]=C;
                else if(j<b)
                    x[i][j]=A;
                else
                    x[i][j]=B;
        return e;
    }
    return 0;
}
int main()
{
    int a[6],n=0;
    for(int i=0;i<6;i++)
        cin>>a[i];
    for(int i=0;i<6&&!n;i++)
        for(int j=0;j<6&&!n;j++)
            for(int k=0;k<6&&!n;k++)
                if(j!=i&&j!=(i^1)&&k!=i&&k!=j&&k!=(i^1)&&k!=(j^1))
                    n=check(a[i],a[i^1],a[j],a[j^1],a[k],a[k^1],i/2+'A',j/2+'A',k/2+'A');
    if(n==0)
        printf("-1");
    else
        cout<<n<<endl;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++)
            printf("%c",x[i][j]);
        printf("\n");
    }
}
