#include <bits/stdc++.h>
using namespace std;
void cas1(int bre,int h1,int h2,int b1,int b2,char* c) {
	int i,j;
	for(i=0;i<h1;i++) {
		for(j=0;j<bre;j++)
			cout<<c[0];
		cout<<endl;
	}
	for(i=0;i<h2;i++) {
		for(j=0;j<bre;j++)
			if(j<b1)
				cout<<c[1];
			else
				cout<<c[2];
		cout<<endl;
	}
}
void cas3(int b,int h1,int h2,int h3,char* c) {
	int i,j;
	for(i=0;i<h1+h2+h3;i++) {
		for(j=0;j<b;j++)
			if(i<h1)
				cout<<c[0];
			else if(i<h2+h1)
				cout<<c[1];
			else
				cout<<c[2];
		cout<<endl;
	}
}
int main() {
	int i,j,k,l,n,m,flag,a[3][2];
	char c[3];
	for(i=0;i<3;i++)
		cin>>a[i][0]>>a[i][1];
	c[0]='A';
	c[1]='B';
	c[2]='C';
	for(i=0;i<2;i++)
		for(j=0;j<2;j++)
			for(k=0;k<2;k++) {
				if((a[0][i]==a[1][j] && a[0][i]==a[2][k]) && (a[0][(i+1)%2]+a[1][(j+1)%2]+a[2][(k+1)%2])==a[0][i]) {
					cout<<a[0][i]<<endl;
					cas3(a[0][i],a[0][(i+1)%2],a[1][(1+j)%2],a[2][(1+k)%2],c);
					return 0;
				}
				for(l=0;l<3;l++)
					if(a[l][i]==a[(l+1)%3][j]+a[(l+2)%3][k]&&a[(l+1)%3][(1+j)%2]==a[(l+2)%3][(1+k)%2]&&(a[l][i]==a[(l+1)%3][(1+j)%2]+a[l][(i+1)%2])) {
						if(l==1) {
							c[0]='B';
							c[1]='C';
							c[2]='A';
						}
						else if(l==2) {
							c[0]='C';
							c[1]='A';
							c[2]='B';
						}
						cout<<a[l][i]<<endl;
						cas1(a[l][i],a[l][(i+1)%2],a[(l+1)%3][(1+j)%2],a[(l+1)%3][j],a[(l+2)%3][k],c);
						return 0;
					}
			}
	cout<<"-1\n";
	return 0;
}
