#include <bits/stdc++.h>

using namespace std;

char	Map[110][110];

int main()
{
	int	i,j,xa,xb,xc,ya,yb,yc;
	char	A=65,B=66,C=67;

	cin >> xa >> ya >> xb >> yb >> xc >> yc;

	if(xa>ya)swap(xa,ya);if(xb>yb)swap(xb,yb);if(xc>yc)swap(xc,yc);

	if(ya==yb && ya==yc && xa+xb+xc==ya)
	{
		cout << ya << endl;
		for(i=1;i<=ya;++i)for(j=1;j<=xa;++j)
			Map[i][j]=A;
		for(i=1;i<=ya;++i)for(j=xa+1;j<=xa+xb;++j)
			Map[i][j]=B;
		for(i=1;i<=ya;++i)for(j=xa+xb+1;j<=xa+xb+xc;++j)
			Map[i][j]=C;
		for(i=1;i<=ya;++i)
		{
			for(j=1;j<=ya;++j)
				cout << Map[i][j];
			cout << endl;
		}
		return 0;
	}

	if(ya<yb)swap(xa,xb),swap(ya,yb),swap(A,B);
	if(ya<yc)swap(xa,xc),swap(ya,yc),swap(A,C);

	if(xa+yb==ya)swap(xb,yb);
	if(xa+yc==ya)swap(xc,yc);

	if(xa+xb==ya && xb==xc && yb+yc==ya)
	{
		cout << ya << endl;
		for(i=1;i<=ya;++i)for(j=1;j<=xa;++j)
			Map[i][j]=A;
		for(i=1;i<=yb;++i)for(j=xa+1;j<=ya;++j)
			Map[i][j]=B;
		for(i=yb+1;i<=ya;++i)for(j=xa+1;j<=ya;++j)
			Map[i][j]=C;
		for(i=1;i<=ya;++i)
		{
			for(j=1;j<=ya;++j)
				cout << Map[i][j];
			cout << endl;
		}
		return 0;
	}

	cout << -1 << endl;

	return 0;
}
