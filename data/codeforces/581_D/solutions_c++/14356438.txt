#include<cstdio>
#include<iostream>
using namespace std;
char x[405][405];
int n;
int check(int a,int b,int c,int d,int e,int f,char A,char B,char C){
    
    if(a==c&&a==e&&a==b+d+f)
    {
        for(int i=0;i<a;i++)
            for(int j=0;j<a;j++)
            	if(i<b)
                	x[i][j]=A;
                else if(i<b+d)
                	x[i][j]=B;
                else
                	x[i][j]=C;
        n=a;
    }
    if(a==c&&e==b+d&&e==a+f)
    {
        for(int i=0;i<f;i++)
            for(int j=0;j<e;j++)
                x[i][j]=C;
        for(int i=f;i<e;i++)
            for(int j=0;j<e;j++)
                if(j<b)
                    x[i][j]=A;
                else
                    x[i][j]=B;
        n=e;
    }
}
int main()
{
    int a[6];n=0;
    for(int i=0;i<6;i++)
        cin>>a[i];
    for(int i=0;i<6;i++)
    	for(int j=0;j<6;j++)
    		for(int k=0;k<6;k++)
    			if(j!=i&&j!=(i^1)&&k!=i&&k!=j&&k!=(i^1)&&k!=(j^1))
    				check(a[i],a[i^1],a[j],a[j^1],a[k],a[k^1],i/2+'A',j/2+'A',k/2+'A');
    if(n==0)
        printf("-1");
    else
        cout<<n<<endl;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++)
                printf("%c",x[i][j]);
            printf("\n");
        }
}
