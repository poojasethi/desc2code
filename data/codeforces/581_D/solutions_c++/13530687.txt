#include<bits/stdc++.h>
using namespace std;
int main()
{
int sum=0;
int a=-1;
bool visited[3];
fill(visited,visited+3,false);
int x[3],y[3];
for(int i=0;i<3;i++)
{
    cin>>x[i]>>y[i];
    if(x[i]<y[i])
        swap(x[i],y[i]);
        sum+=(x[i]*y[i]);
        a=max(a,x[i]);
}
if((a*a)!=sum)
{
    cout<<-1<<endl;
    return 0;
}
cout<<a<<endl;
int f=a;
for(int i=0;i<3;i++)
{
    if(x[i]==a)
    {
     for(int j=0;j<y[i];j++)
     {
         for(int k=0;k<a;k++)
         {
             cout<<(char)('A'+i);
         }
         cout<<endl;
     }
     f-=y[i];
     visited[i]=true;
    }
}
for(int i=0;i<f;i++)
{
    for(int j=0;j<3;j++)
    {
        if(visited[j]==false)
        {
            if(x[j]==f)
            {
                for(int k=0;k<y[j];k++)
                    cout<<(char)('A'+j);
            }

            else
            {
                for(int k=0;k<x[j];k++)
                    cout<<(char)('A'+j);
            }

        }

    }
    cout<<endl;
}
}
