#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<string>
#include<cstring>
using namespace std;
#define LL long long
int x[3],y[3];
char mp[110][110];
void ins(int x1,int y1,int x2,int y2,char c){
	for(int i=x1;i<=x2;i++){
		for(int j=y1;j<=y2;j++){
			mp[i][j]=c;
		}
	}
}
int main(){
	int S=0;int ma=0;
	for(int i=0;i<3;i++){
		cin>>x[i]>>y[i];
		if(x[i]>y[i])swap(x[i],y[i]);
		S+=x[i]*y[i];
		ma=max(ma,x[i]);ma=max(ma,y[i]);
	}
	int a=(int)sqrt(S+0.5);
	
	int m=0;
	for(int i=0;i<3;i++)if(y[i]==a)m++;
	if(m==3){
		ins(1,1,x[0],y[0],'A');
		ins(x[0]+1,1,x[0]+x[1],y[1],'B');
		ins(x[0]+x[1]+1,1,x[0]+x[1]+x[2],y[2],'C');
	} 
	else{
		int b,d;
		for(int i=0;i<3;i++)if(y[i]==a){
			ins(1,1,x[i],y[i],'A'+i);
			b=x[i];
		}
		bool f=0;
		d=a-b;int t;
		for(int i=0;i<3;i++)if(y[i]!=a){
			int tmp=d;
			if(x[i]!=d)tmp=x[i];
			if(y[i]!=d)tmp=y[i];
			if(!f){
				f=1;
				ins(b+1,1,a,tmp,'A'+i);
				t=tmp;
			}else{
				ins(b+1,t+1,a,a,'A'+i);
			}
			//cout<<x[i]<<y[i]<<endl;
		}
	}
	if(a!=ma||a*a!=S){
		puts("-1");
		return 0;
	}
	cout<<a<<endl;
	for(int i=1;i<=a;i++){
		for(int j=1;j<=a;j++){
			putchar(mp[i][j]);
		}
		puts("");
	}
	
}
