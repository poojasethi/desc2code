#include<bits/stdc++.h>
using namespace std;

int a[7],s,r,i,j,k,f,id,id1,id2,b,c;
char out[4];

int main() {
	for (i=1;i<=6;i++)
		scanf("%d",&a[i]);
	for (i=1;i<=3;i++){
		s+=(a[i*2-1]*a[i*2]);
		if (a[i*2-1]>a[i*2])
			swap(a[i*2-1],a[i*2]);	
	}
	out[1]='A',out[2]='B',out[3]='C';
	if ((int)sqrt(s)*(int)sqrt(s)==s){
		int r=sqrt(s);
		if ((a[2]==a[4])&&(a[4]==a[6])&&(a[6]==r)&&(a[1]+a[3]+a[5]==r)){
			f=1;
			printf("%d\n",r);
			for (i=1;i<=r;i++){
				for (int k=1;k<=5;k+=2)
					for (j=1;j<=a[k];j++)
						printf("%c",out[k/2+1]);	
				printf("\n");
			}
		}
		for (i=2;i<=6;i+=2)
			if (a[i]==r)
				id=i;
				else if (id1==0)
						id1=i;
						else id2=i;
		if (a[id1]==a[id2])
			b=id1-1,c=id2-1;
			else if (a[id1]==a[id2-1])
					b=id1-1,c=id2,id2--;
					else if (a[id1-1]==a[id2])
							b=id1,c=id2-1,id1--;
							else if (a[id1-1]==a[id2-1])
									b=id1,c=id2,id2--,id1--;
		if (a[id]==r)
			if ((a[id1]==a[id2])&&(a[id1]+a[id-1]==r)&&(a[b]+a[c]==r)){
				printf("%d\n",r),f=1;
				for (i=1;i<=r;i++){
					for (j=1;j<=a[id-1];j++)
						printf("%c",out[id/2]);
					if (i<=a[b])
						for (j=1;j<=a[id1];j++)
							printf("%c",out[(id1+1)/2]);
						else for (j=1;j<=a[id1];j++)
								printf("%c",out[(id2+1)/2]);
					printf("\n");
				}
			}
	}
		else printf("-1\n"),f=1;
	if (f==0)
		printf("-1\n");
	return 0;
}
