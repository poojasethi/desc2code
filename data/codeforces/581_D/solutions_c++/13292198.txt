#include<iostream>
#include<algorithm>
using namespace std;
struct sq {
	int x,y;
	char c;
}a[3];
int main(){
	for(int i=0;i<3;i++){
		cin>>a[i].x>>a[i].y;
		if(a[i].x<a[i].y)swap(a[i].x,a[i].y);
		a[i].c='A'+i;
	}
	for(int j=1;j<3;j++){
		if(a[j].x>a[0].x)swap(a[0],a[j]);
	}
	int A=a[0].x,B=a[0].x-a[0].y;
	if(a[1].x==A){
		if(a[2].x==A){
			if(a[0].y+a[1].y+a[2].y==A){
				cout<<A<<endl;
				for(int i=0;i<3;i++)
					for(int j=1;j<=a[i].y;j++){
						for(int _=1;_<=A;_++)cout<<a[i].c; 
						cout<<endl;
					}
				return 0;
			}
		}
		cout<<-1<<endl;
		return 0;
	}
	for(int i=1;i<3;i++)if(a[i].x==B)swap(a[i].x,a[i].y);
	if(a[1].y==B&&a[2].y==B){
		cout<<A<<endl;
		if(a[1].x+a[2].x==A){
			for(int j=1;j<=a[0].y;j++){
				for(int _=1;_<=A;_++)cout<<a[0].c; 
				cout<<endl;
			}
		}
		for(int j=1;j<=a[1].y;j++){
			for(int i=1;i<3;i++)for(int _=1;_<=a[i].x;_++)cout<<a[i].c;
			cout<<endl;
		}
		return 0;
	}
	cout<<-1<<endl;
	return 0;
}
