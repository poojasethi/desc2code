#include <iostream>
using namespace std;
int main()
{
    int area,i,x[3],y[3],tmp,maxi,p,j;
    area=0;
    for(i=0;i<3;i++)
    {
        cin>>x[i]>>y[i];
        area+=x[i]*y[i];
        if(y[i]>x[i])
        {
            tmp=x[i];
            x[i]=y[i];
            y[i]=tmp;
        }
        if(x[i]>maxi)
        maxi=x[i];
    }
    if(area!=maxi*maxi)
        cout<<"-1\n";
        else
        {
            cout<<maxi<<"\n";
            p=maxi;
            int mark[3]={0};
            for(i=0;i<3;i++)
            {
                if(x[i]==maxi)
                {
                    for(j=0;j<y[i];j++)
                    cout<<string(maxi,'A'+i)<<"\n";
                    p=p-y[i];
                    mark[i]=1;
                }
            }
            for(i=0;i<p;i++)
            {
                for(j=0;j<3;j++)
                if(mark[j]==0)
                {
                    if(x[j]==p)
                    cout<<string(y[j],'A'+j);
                    else
                    cout<<string(x[j],'A'+j);
                }
                cout<<"\n";
            }
        }
    return 0;
}
