#include<stdio.h>
#include<algorithm>
#include<vector>
#include<math.h>
using namespace std;

char ch[3]={'A','B','C'};
int x[3],y[3],flag;
vector<int> v;

int main(){
	scanf("%d %d %d %d %d %d", &x[0],&y[0],&x[1],&y[1],&x[2],&y[2]);
	int area = (x[0]*y[0]+x[1]*y[1]+x[2]*y[2]);
	int side = sqrt(area);
	if(side*side!=area) {
		puts("-1"); return 0;
	} 
	for(int i=0; i<3; i++){
		if(y[i]==side) swap(x[i],y[i]);
		if(x[i]==side) v.push_back(i);
	}
	if(v.size()==3){
		printf("%d\n", side);
		for(int i=0; i<3; i++){
			for(int j=0; j<y[i]; j++){
				for(int k=0; k<side; k++)
					printf("%c", ch[i]);
				puts("");
			}
		}
	}else if(v.size()==1){
		int w1,w2,w3,side2;
		w1 = v[0];
		w2 = (w1+1)%3;
		w3 = (w1+2)%3;
		if(w2>w3) swap(w2,w3);
		side2 = side - y[w1];  
		for(int i=0; i<3; i++){
			if(i!=w1){
				if(y[i]==side2) swap(x[i],y[i]);
			}
		}
		printf("%d\n", side);
		for(int j=0; j<y[w1]; j++){
			for(int i=0; i<side; i++){
				printf("%c", ch[w1]);
			}
			puts("");	
		}
		for(int i=0; i<side2; i++){
			for(int j=0; j<y[w2]; j++)
				printf("%c", ch[w2]);
			for(int k=0; k<y[w3]; k++)
				printf("%c", ch[w3]);
			puts("");
		}	
	}else{
		puts("-1");
	}
	return 0;
}
