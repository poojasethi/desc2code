#include    <cstdio>
#include    <cstring>
#include    <iostream>
#include    <algorithm>
#include    <queue>

using   namespace   std;

struct Item
{
    int w, h; char c;

    bool operator < (const Item &b) const
    {
        return w < b.w;
    }

} a[3];

inline int S(int k)
{
    return a[k].w * a[k].h;
}

int main()
{
    for(int i = 0; i < 3; i ++)
    {
        cin >> a[i].w >> a[i].h;
        if(a[i].w < a[i].h) swap(a[i].w, a[i].h);
        a[i].c = i + 'A';
    }
    sort(a, a + 3);
    int x = a[2].w;
    if(x * x != (S(0) + S(1) + S(2)))
        return puts("-1"), 0;
    printf("%d\n", a[2].w);
    for(int i = 0; i < a[2].h; i ++)
    {
        for(int j = 0; j < a[2].w; j ++) putchar(a[2].c);
        puts("");
    }
    if(a[2].h + a[1].h != x && a[1].w != x)
        swap(a[1].w, a[1].h);
    for(int i = 0; i < a[1].h; i ++)
    {
        for(int j = 0; j < a[1].w; j ++) putchar(a[1].c);
        for(int j = 0; j < x - a[1].w; j ++) putchar(a[0].c);
        puts("");
    }
    for(int i = 0; i < x - a[2].h - a[1].h; i ++)
    {
        for(int j = 0; j < x; j ++) putchar(a[0].c);
        puts("");
    }
    return 0;
}
