#include<iostream>

using namespace std;

int ax[3],ay[3],g,s;
char a[200][200];

bool check()
{
	int f=-1;
	for(int i=0;i<3&&f==-1;i++)
		if(ax[i]==g||ay[i]==g)
			f=i;
	if(f==-1) return false;
	if(ay[f]==g) swap(ax[f],ay[f]);
	for(int i=0;i<ay[f];i++)
		for(int j=0;j<g;j++)
			a[i][j]='A'+f;
	int q=(f?0:1);
	int e=3-f-q;
	if((ax[q]==g||ay[q]==g)&&(ax[e]==g||ay[e]==g))
	{
		if(ay[q]==g) swap(ax[q],ay[q]);
		if(ay[e]==g) swap(ax[e],ay[e]);
		for(int i=ay[f];i<g;i++)
			for(int j=0;j<g;j++)
				a[i][j]='A'+(i<ay[f]+ay[q]?q:e);
	}
	else
		if((ax[q]==g-ay[f]||ay[q]==g-ay[f])&&(ax[e]==g-ay[f]||ay[e]==g-ay[f]))
		{
			if(ay[q]==g-ay[f]) swap(ax[q],ay[q]);
			if(ay[e]==g-ay[f]) swap(ax[e],ay[e]);
			for(int i=ay[f];i<g;i++)
				for(int j=0;j<g;j++)
					a[i][j]='A'+(j<ay[q]?q:e);
		}
		else
			return false;
	return true;
}

int main()
{
	ios_base::sync_with_stdio(0);
	
	for(int i=0;i<3;i++)
		cin>>ax[i]>>ay[i];
		
	s=ax[0]*ay[0]+ax[1]*ay[1]+ax[2]*ay[2];
	
	for(;s>g*g;g++);
	if(s<g*g||!check())
	{
		cout<<-1;
	}
	else
	{
		cout<<g<<endl;
		for(int i=0;i<g;i++)
		{
			for(int j=0;j<g;j++)
				cout<<a[i][j];
			cout<<endl;
		}
	}
	
	return 0;
}
