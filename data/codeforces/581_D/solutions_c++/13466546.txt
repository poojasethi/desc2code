#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int x[3],y[3];
bool hp[3];
int a,p,sum;
void swap(int i)
{
int temp=x[i];
x[i]=y[i];
y[i]=temp;
}
void put()
{
if(a*a!=sum)
  {
  cout<<-1<<endl;
  return;
  }
cout<<a<<endl;
memset(hp,0,sizeof(hp));
p=a;
for(int i=0;i<3;++i)
  if(x[i]==a)
    {
    for(int j=0;j<y[i];++j)
      cout<<string(a,'A'+i)<<endl;
    p-=y[i];
    hp[i]=1;
    }
for(int i=0;i<p;++i)
  {
  for(int j=0;j<3;++j)
    if(!hp[j])
      {
      if(x[j]==p)
        cout<<string(y[j],'A'+j);
      else
        cout<<string(x[j],'A'+j);
      }
  cout<<endl;
  }
return;
}
int main()
{
sum=0;
a=-1;
for(int i=0;i<3;++i)
  {
  scanf("%d %d",&x[i],&y[i]);
  if(y[i]>x[i])
    swap(i);
  sum+=x[i]*y[i];
  a=max(a,x[i]);
  }
  put();
}
