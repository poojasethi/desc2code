#include <bits/stdc++.h>

using namespace std;

const int N = 110;

int mx = 0;
int x[4];
int y[4];
int used[4];
char g[N][N];
int px, py;

void draw(int a, int b, int ff) {
  for (int i = 0; i < a; ++i) {
    for (int j = 0; j < b; ++j) {
      g[px + i][py + j] = 'A' + ff - 1;
    }
  }
  if (px + a == mx) {
    py = py + b;
  }
  else {
    px = px + a;
  }
}

int main() {
  for (int i = 1; i <= 3; ++i) {
    scanf("%d%d", x + i, y + i);
  }
  for (int i = 1; i <= 3; ++i) {
    mx = max(mx, x[i]);
    mx = max(mx, y[i]);
  }
  int a = mx, b = mx;
  for (int j = 0; j < 3; ++j) {
    for (int i = 1; i <= 3; ++i) {
      if (!used[i]) {
        if (a == x[i]) {
          used[i] = 1;
          b -= y[i];
          draw(x[i], y[i], i);
          break;
        }
        if (a == y[i]) {
          used[i] = 1;
          b -= x[i];
          draw(y[i], x[i], i);
          break;
        }
        if (b == x[i]) {
          used[i] = 1;
          a -= y[i];
          draw(y[i], x[i], i);
          break;
        }
        if (b == y[i]) {
          used[i] = 1;
          a -= x[i];
          draw(x[i], y[i], i);
          break;
        }
      }
    }
  }
  if (a == 0 || b == 0) {
    printf("%d\n", mx);
    for (int i = 0; i < mx; ++i) {
      printf("%s\n", g[i]);
    }
  }
  else {
    printf("-1\n");
  }
  return 0;
}
