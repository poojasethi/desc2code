#include <bits/stdc++.h>

using namespace std;

int x[3];
int y[3];
char t[3] = {'A', 'B', 'C'};

int main()
{
   for(int i=0; i<3; i++)
      cin >> x[i] >> y[i];
   for(int i=1; i<3; i++)
      if(max(x[i], y[i]) > max(x[0], y[0]))
      {
         swap(x[i], x[0]);
         swap(y[i], y[0]);
         swap(t[i], t[0]);
      }
   int a = max(x[0], y[0]);
   int b = a - min(x[0], y[0]);
   if(y[1] == b)
      swap(x[1], y[1]);
   if(y[2] == b)
      swap(x[2], y[2]);
   if(x[1] == b && x[2] == b && y[1] + y[2] == a)
   {
      cout << a << endl;
      for(int i=0; i<min(x[0], y[0]); i++)
      {
         for(int j=0; j<a; j++)
            cout << t[0];
         cout << endl;
      }
      for(int i=0; i<b; i++)
      {
         for(int j=0; j<a; j++)
            cout << t[1 + (j >= y[1])];
         cout << endl;
      }
      return 0;
   }
   if(y[1] == a)
      swap(x[1], y[1]);
   if(y[2] == a)
      swap(x[2], y[2]);
   if(x[1] == a && x[2] == a && y[1] + y[2] == b)
   {
      cout << a << endl;
      for(int i=0; i<min(x[0], y[0]); i++)
      {
         for(int j=0; j<a; j++)
            cout << t[0];
         cout << endl;
      }
      for(int i=0; i<b; i++)
      {
         for(int j=0; j<a; j++)
            cout << t[1 + (i >= y[1])];
         cout << endl;
      }
      return 0;
   }
   cout << -1 << endl;
   return 0;
}
