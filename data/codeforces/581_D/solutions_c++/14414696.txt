#include<bits/stdc++.h>

using namespace std;

int x[300] , y[300];
char col[4] = {0 , 'A' , 'B' , 'C'};
void check(int t1 , int t2 , int t3){
	if(x[t1] == x[t2] + x[t3] && y[t1] + y[t2] == x[t1] && y[t2] == y[t3])
	{
		cout << x[t1] << endl;
		for(int i = 1;i <= y[t1];i ++){
			for(int j = 1;j <= x[t1];j ++)
				cout << col[t1];
			cout << endl;
		}	
		for(int i = 1;i <= y[t2];i ++){
			for(int j = 1;j <= x[t1];j ++)
				if(j <= x[t2])	cout << col[t2];
				else	cout << col[t3];
			cout << endl;
		}
		exit(0);
	}
}	
int p[101];
bool used[101];
void rec(int v){
	if(v == 4){                    
		check(p[1] , p[2] , p[3]);
		return;
	}		
	for(int i = 1;i <= 3;i ++){
		if(used[i])	continue;
		p[v] = i;
		used[i] = 1;
		rec(v + 1);
		swap(x[i] , y[i]);
		rec(v + 1);
		swap(x[i] , y[i]);
		used[i] = p[v] = 0;
	}
}			
int main(){

	for(int i = 1;i <= 3;i ++){
		cin >> x[i] >> y[i];
		if(x[i] < y[i])	swap(x[i] , y[i]);	        
	}	
	if(x[1] == x[2] && x[2] == x[3] && y[1] + y[2] + y[3] == x[1]){
		cout << x[1] << endl;
		for(int t = 1;t <= 3;t ++){
			for(int i = 1;i <= y[t];i ++){
				for(int j = 1;j <= x[1];j ++)
					cout << col[t];
				cout << endl;
			}
		}
		exit(0);
	}	
	rec(1);
	cout << -1;
	return 0;
}
