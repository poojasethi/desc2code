struct gao
{
	int x, y;
	char c;
	void input(char t) {
		scanf("%d%d", &x, &y);
		if(x > y) swap(x, y);
		c = t;
	}
}p[3];

bool judge(gao a, gao b, gao c)
{
	int n = a.y, l = n-a.x;
	// printf("%d %d\n", n, l);
	if(b.x != l) swap(b.x, b.y);
	if(b.x != l) return 0;
	if(c.x != l) swap(c.x, c.y);
	if(c.x != l) return 0;
	if(b.y+c.y != n) return 0;

	cout << n << endl;
	for(int i = 0; i < a.x; i++) {
		for(int j = 0; j < a.y; j++)
			printf("%c", a.c);
		puts("");
	}
	for(int i = 0; i < b.x; i++) {
		for(int j = 0; j < b.y; j++)
			printf("%c", b.c);
		for(int j = 0; j < c.y; j++)
			printf("%c", c.c);
		puts("");
	}
	return 1;
}

bool judge2(gao a, gao b, gao c)
{
	int n = a.y;
	if(b.y != n) return 0;
	if(c.y != n) return 0;
	if(a.x+b.x+c.x != n) return 0;
	cout << n << endl;
	for(int i = 0; i < n; i++) {
		for(int j = 0; j < a.x; j++)
			printf("%c", a.c);
		for(int j = 0; j < b.x; j++)
			printf("%c", b.c);
		for(int j = 0; j < c.x; j++)
			printf("%c", c.c);
		puts("");
	}
	return 1;
}

int main()
