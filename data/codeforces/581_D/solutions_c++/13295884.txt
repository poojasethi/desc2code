struct ss{
	int A, B;
	char C;
}a, b, c;
void paint(ss a){
	for (int i = 1; i <= a.A; i++){
		for (int j = 1; j <= a.B; j++) putchar(a.C);
		puts("");
	}
}
void work1(ss a, ss b, ss c){
//	printf("%d %d %d\n", a.B, b.B, c.B);
	if (a.B == b.B && b.B == c.B && a.B == a.A + b.A + c.A){
		int len = a.B;
		printf("%d\n", len);
		paint(a), paint(b), paint(c);
		exit(0);
	}
}
void work2(ss a, ss b, ss c){
	if (b.A == c.A && a.B == b.B + c.B && a.B == a.A + b.A){
		int len = a.B;
		printf("%d\n", len);
		paint(a);
		for (int i = 1; i <= b.A; i++){
			for (int j = 1; j <= b.B; j++) putchar(b.C);
			for (int j = 1; j <= c.B; j++) putchar(c.C);
			puts("");
		}
		exit(0);
	}
}
int main(){
	scanf("%d%d%d%d%d%d", &a.A, &a.B, &b.A, &b.B, &c.A, &c.B);
	a.C = 'A', b.C = 'B', c.C = 'C';
	for (int i = 2; i; i--, swap(a.A, a.B))
		for (int j = 2; j; j--, swap(b.A, b.B))
			for (int k = 2; k; k--, swap(c.A, c.B))
				work1(a, b, c), work2(a, b, c), work2(b, a, c), work2(c, a, b);
	puts("-1");
}
