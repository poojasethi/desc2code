#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
int ans=-1,a[3],s[3],a1[3];
char f[100][100];
int d[3];
void k(int w,int c,int v){
  int e,r;
  for(e=0;e<a[d[w]];e++)
   for(r=0;r<s[d[w]];r++)
    f[e+c][r+v]=d[w];
}
void l(int t){
  int w,e;
  if(t==3){
   if(a[d[0]]==a[d[1]]+a[d[2]])
    if((s[d[1]]==s[d[2]])&&(s[d[0]]+s[d[1]]==a[d[0]])){
     ans=a[d[0]];
     k(0,0,0);
     k(1,0,s[d[0]]);
     k(2,a[d[1]],s[d[0]]);
    }
   if((a[d[0]]==a[d[1]])&&((a[d[0]]==a[d[2]])&&(a[d[0]]==s[d[0]]+s[d[1]]+s[d[2]]))){
    ans=a[d[0]];
    k(0,0,0);
    k(1,0,s[d[0]]);
    k(2,0,s[d[0]]+s[d[1]]);
   }
   return;
  }
  for(w=0;w<3;w++)
   if(a1[w]){
    a1[w]=0;
    d[t]=w;
    e=a[w];a[w]=s[w];s[w]=e;
    l(t+1);
    e=a[w];a[w]=s[w];s[w]=e;
    l(t+1);
    a1[w]=1;
   }
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t;
  for(w=0;w<3;w++){
   cin>>a[w]>>s[w];
   a1[w]=1;
  }
  l(0);
  printf("%d\n",ans);
  if(ans<0)
   return 0;
  for(e=0;e<ans;e++){
   for(r=0;r<ans;r++)
    cout<<(char)(f[e][r]+65);
   cout<<"\n";
  }
  return 0;
}
