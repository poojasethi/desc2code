#include<bits/stdc++.h>
using namespace std;
int main()
{
	int ret = 0;
	int x[3],y[3];
	for(int i=0;i<3;i++)
		scanf("%d%d",&x[i],&y[i]),ret+=x[i]*y[i];
	if(sqrt(ret)*sqrt(ret)!=ret)
	{
		puts("-1");
		return 0;
	}
	ret = sqrt(ret);
	for(int i=0;i<(1<<3);i++)
	{
		int xx[3],yy[3];
		for(int j=0;j<3;j++)
			if(i&(1<<j))
			    xx[j] = y[j],yy[j] = x[j];
		    else
				xx[j] = x[j],yy[j] = y[j];


		if(xx[0]+xx[1]+xx[2]==ret)
		{
			printf("%d\n",ret);
			for(int r=0;r<xx[0];r++)
			{
			    for(int c=0;c<ret;c++)
			        printf("A");
				puts("");
			}

			for(int r=0;r<xx[1];r++)
			{
			    for(int c=0;c<ret;c++)
			        printf("B");
				puts("");
			}

			for(int r=0;r<xx[2];r++)
			{
			    for(int c=0;c<ret;c++)
			        printf("C");
				puts("");
			}
			return 0;
		}

		int p[3] = {0,1,2};
		char pp[3] = {'A','B','C'};
		do
		{
			if(xx[p[0]]+xx[p[1]]==ret&&yy[p[0]]==yy[p[1]]&&yy[p[0]]+yy[p[2]]==ret)
			{
				printf("%d\n",ret);
				for(int r =0;r<yy[p[0]];r++)
				{


				for(int c=0;c<ret;c++)
				{
					if(c<xx[p[0]])
						printf("%c",pp[p[0]]);
					else
						printf("%c",pp[p[1]]);
				}
				puts("");


				}

				for(int r=0;r<ret-yy[p[0]];r++)
				{
					for(int c=0;c<ret;c++)
					    printf("%c",pp[p[2]]);
					puts("");
				}
				return 0;
			}
		}
		while(next_permutation(p,p+3));
	}
	puts("-1");
	return 0;
}
