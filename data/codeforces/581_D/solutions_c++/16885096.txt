#include <bits/stdc++.h>
using namespace std;

char str[210][210];

struct Point {
  int x, y, id;
  void input(int id) {
    this->id = id;
    cin >> x >> y;
    if (x > y) swap(x, y);
  }
  bool operator < (const Point& o) const {
    return y > o.y;
  }
} p[3];

int main() {
//  freopen("case.in", "r", stdin);
  int tot = 0;
  for (int i = 0; i < 3; i++) {
    p[i].input(i);
    tot += p[i].x * p[i].y;
  }
  sort(p, p + 3);

  int n = p[0].y;
  if (n * n != tot) {
    puts("-1"); return 0;
  }

  printf("%d\n", n);
  for (int i = 0; i < p[0].x; i++) {
    for (int j = 0; j < n; j++) putchar('A' + p[0].id);
    puts("");
  }
  if (p[1].y == n && p[2].y == n) {
    for (int i = p[0].x; i < p[0].x + p[1].x; i++) {
      for (int j = 0; j < n; j++) putchar('A' + p[1].id);
      puts("");
    }
    for (int i = p[0].x + p[1].x; i < n; i++) {
      for (int j = 0; j < n; j++) putchar('A' + p[2].id);
      puts("");
    }
    return 0;
  }

  if (p[1].x + p[0].x != n) swap(p[1].x, p[1].y);
  if (p[2].x + p[0].x != n) swap(p[2].x, p[2].y);

  for (int i = p[0].x; i < n; i++) {
    for (int j = 0; j < p[1].y; j++) str[i][j] = 'A' + p[1].id;
  }
  for (int i = p[0].x; i < n; i++) {
    for (int j = p[1].y; j < n; j++) str[i][j] = 'A' + p[2].id;
  }
  for (int i = p[0].x; i < n; i++) cout << str[i] << endl;
}
