#include <cstring>

using namespace std;
int a,b,c,d,e,f,x1,y1,x2,y2,x3,y3,wid,i;
char ar[205][205];
void judge(int a,int b,int c,int d,int e,int f){
	 int i,j;
	 for(i=a;i<a+x1;i++)
	 for(j=b;j<b+y1;j++)
	 ar[i][j]='A';
	 for(i=c;i<c+x2;i++)
	 for(j=d;j<d+y2;j++)
	 ar[i][j]='B';
	 for(i=e;i<e+x3;i++)
	 for(j=f;j<f+y3;j++)
	 ar[i][j]='C';
	 printf("%d\n",wid);
	 for(i=1;i<=wid;i++,puts(""))
	 for(j=1;j<=wid;j++)
	 printf("%c",ar[i][j]);
}
int main(){
    scanf("%d %d %d %d %d %d",&a,&b,&c,&d,&e,&f);
    wid=a*b+c*d+e*f;
    for(i=1;;i++)
    if(i*i>=wid)
    break;
    if(i*i==wid)
    wid=i;
    for(i=0;i<8;i++){
		if(i&1)
		x1=a,y1=b;
		else
		x1=b,y1=a;
		if(i&2)
		x2=c,y2=d;
		else
		x2=d,y2=c;
		if(i&4)
		x3=e,y3=f;
		else
		x3=f,y3=e;
		if(y1==y2 && y2==y3 && x1+x2+x3==y1){
			judge(1,1,1+x1,1,1+x1+x2,1);
			return 0;
		}
		if(y1==y2 && x1+x2==x3 && y1+y3==x3){
			judge(1,1,1+x1,1,1,1+y1);
			return 0;
		}
	    if(y1==y3 && x1+x3==x2 && y1+y2==x2){
		   judge(1,1,1,1+y1,1+x1,1);
		   return 0;
		}
		if(y2==y3 && x2+x3==x1 && y3+y1==x1){
			judge(1,1+y2,1,1,1+x2,1);
			return 0;
		}
	}
    printf("-1\n");
