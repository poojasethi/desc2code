#include<bits/stdc++.h>
#define swap(a,b) (((a)^=(b)),((b)^=(a)),((a)^=(b)))
using namespace std;


int i,j,k,x[3],y[3],height,sum,a;
bool h[3];

int main(void)
{
	for(i=0;i<3;i++)
	{
		cin>>x[i]>>y[i];
		if(y[i]>x[i])swap(x[i],y[i]);
		sum+=x[i]*y[i];
a=max(a,x[i]);
	}
	
	if(a*a!=sum)
	{
		cout<<-1;return 0;
	}
height =a;
cout<<a<<endl;
	for(i=0;i<3;i++)
	{
		if(x[i]==a)
		{
			for(j=0;j<y[i];j++)
			{
				cout<<string(a,'A'+i)<<endl;
			}
			h[i]=true;
			height-=y[i];
		}
	}
if(height ==0)return 0;
	if(h[0]==0)
	{
		if(height==x[0])
		{
			for(i=0;i<x[0];i++)
			{
				cout<<string(y[0],'A');
				if(h[1]==0)
				{
					cout<<string(a-y[0],'B');
				}
				else
				{
					cout<<string(a-y[0],'C');
				}
				cout<<endl;
			}
		}
		else
		{
		{
			for(i=0;i<y[0];i++)
			{
				cout<<string(x[0],'A');
				if(h[1]==0)
				{
					cout<<string(a-x[0],'B');
				}
				else
				{
					cout<<string(a-x[0],'C');
				}
				cout<<endl;
			}
		}
		}
	}
	else
	{
		if(height==x[1])
		{
			for(i=0;i<x[1];i++)
			{
				cout<<string(y[1],'B');
				if(h[2]==0)
				{
					cout<<string(a-y[1],'C')<<endl;
				}
				
			}
		}
		
else
{
			for(i=0;i<y[1];i++)
			{
				cout<<string(x[1],'B');
				if(h[2]==0)
				{
					cout<<string(a-x[1],'C')<<endl;
				}
				
			}

}
	}
	
	
	return 0;
}
