#include <stdio.h>
#include <algorithm>

using namespace std;

int main (){
	pair <int,pair<int,int> > p[3];
	int n;
	for(int i=0; i<3; i++){
		scanf("%d%d",&p[i].first,&p[i].second.first);
		p[i].second.second=i;
		if(p[i].first<p[i].second.first) swap(p[i].first,p[i].second.first);
	}
	sort(p,p+3);
	if(p[0].first==p[2].first){
		if(p[0].second.first+p[1].second.first+p[2].second.first==p[2].first){
			printf("%d\n",p[2].first);
			for(int i=0; i<3; i++){
				for(int j=0; j<p[i].second.first; j++){
					for(int k=0; k<p[2].first; k++){
						printf("%c",'A'+p[i].second.second);
					}
					puts("");
				}
			}
		}
		else puts("-1");
	}
	else {
		n=p[2].first-p[2].second.first;
		if(p[0].second.first==n) swap(p[0].first,p[0].second.first);
		if(p[1].second.first==n) swap(p[1].first,p[1].second.first);
		if(p[0].first==n && p[1].first==n && p[0].second.first+p[1].second.first==p[2].first){
			printf("%d\n",p[2].first);
			for(int i=0; i<p[2].second.first; i++){
				for(int j=0; j<p[2].first; j++){
					printf("%c",'A'+p[2].second.second);
				}
				puts("");
			}
			for(int i=0; i<p[0].first; i++){
				for(int j=0; j<p[0].second.first; j++){
					printf("%c",'A'+p[0].second.second);
				}
				for(int j=0; j<p[1].second.first; j++){
					printf("%c",'A'+p[1].second.second);
				}
				puts("");
			}
		}
		else puts("-1");
	}
	return 0;
}
