#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long l,r, k;
	cin >> l >> r >> k;
	bool f = false;

	for(long long i = 1; i <= r; i*= k)
	{
		if(i >= l)
		{
			cout << i << ' ';
			f = true;
		}
		
		if(i > r/k)
			break;
	}
		
	if(!f) 
		cout << -1;
	cout << '\n';
	

	return 0;
}
