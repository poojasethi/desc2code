a = raw_input()
b = raw_input()

ans = ''
a_used = False
num_flips = 0
for i in range(len(a)):
  if a[i] == b[i]:
    ans += a[i]
  else:
    num_flips += 1
    if a_used:
      ans += b[i]
      a_used = False
    else:
      ans += a[i]
      a_used = True
if num_flips % 2 == 1:
  print 'impossible'
else:
  print ans
