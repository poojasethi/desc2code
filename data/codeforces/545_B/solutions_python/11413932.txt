s = raw_input()
l = raw_input()
flag = 0
x = []
for i in range(0 , len(s)):
    if s[i] == l[i]:
        x.append(s[i])
    else:
        if flag == 0 :
            x.append(s[i])
        else:
            x.append(l[i])
        flag = ~flag

if flag == 0 :
    print "".join(x)
else:
    print "impossible"
