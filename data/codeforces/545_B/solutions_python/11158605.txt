s = raw_input()
t = raw_input()
p = ""
switch = 0
for dum in xrange(len(s)):
    if s[dum] == t[dum]:
        p += s[dum]
    else:
        if switch == 0:
            p+= s[dum]
            switch = 1
        else:
            p += t[dum]
            switch = 0
if switch == 0:
    print p
else:
    print "impossible"
