a = list(raw_input())
b = list(raw_input())
num = sum(x != y for x, y in zip(a, b))
if num & 1: print "impossible"
else:
    j = 0
    p = []
    for i in xrange(len(a)):
        p.append(a[i])
        if a[i] == b[i]: continue
        if j:
            p[-1] = b[i]
        j = 1 - j
    print ''.join(p)
