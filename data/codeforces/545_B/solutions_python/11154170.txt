s = raw_input()
t = raw_input()

n = 0
for i, c in enumerate(s):
    if c != t[i]:
        n += 1

if n % 2 == 0:
    r = ""
    j = 0
    for i, c in enumerate(s):

        if j < n/2 and c != t[i]:
            j += 1
            r += t[i]
        else:
            r += s[i]
    print r
else:
    print "impossible"
