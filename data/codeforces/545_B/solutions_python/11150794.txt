s = str(raw_input())
t = str(raw_input())
n = len(s)

diff = 0

for i in range(n):
	if s[i] != t[i]: diff += 1

if diff%2 != 0: print "impossible"
else:
	res = ""
	first = diff/2
	for i in range(n):
		if s[i] == t[i]: res += s[i]
		else:
			if first > 0:
				res += s[i]
				first -= 1
			else:
				res += t[i]

	print res
