import string 

s = raw_input()
t = raw_input()

n = len(s)

diff = set()

for i in xrange(n):
	if s[i] != t[i]:
		diff.add(i)

p = ""

if len(diff)%2 == 1:
	print "impossible"
else:
	r = 0
	for i in xrange(n):
		if i in diff and r == 0:
			p = p + s[i]
			r = 1
		elif i in diff and r == 1:
			p = p + t[i]
			r = 0
		else:
			p = p + t[i]
	print p
