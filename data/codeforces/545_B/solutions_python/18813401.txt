a = raw_input()
b = raw_input()
c = ''
count = 0
a_first = True
for i in range(len(a)):
    if a[i] != b[i]:
        count += 1
        if a_first:
            c += a[i]
            a_first = False
        else:
            c += b[i]
            a_first = True
    else:
        c += a[i]

if count % 2 == 0:
    print c
else:
    print 'impossible'
