def solve(s, t):
	total = 0
	num = 0
	new = ""
	for i in range(len(s)):
		if s[i] != t[i]:
			if num % 2 == 0:
				new += s[i]
			else:
				new += t[i]
			num += 1
			total += 1
		else:
			new += s[i]
	if total % 2 == 1:
		return "impossible"
	return new
