
sum = 0
lst1 = raw_input()
lst2 = raw_input()

for i in range( len( lst1 ) ):
    if lst1[i] != lst2[i]:
        sum += 1

car = list(lst1)
if sum % 2 != 0:
    print "impossible"
else:
    for i in range( len( lst1 ) ):
        if lst1[i] != lst2[i]:
            if sum/2 > 0:
                car[i] = lst2[i]
                sum -= 2
    print ''.join(car)
