a,b=raw_input(),raw_input()
n=len(a)
diff=[i for i in range(n) if a[i]!=b[i]]
if len(diff)&1:
    print "impossible"
else:
    ans=['2']*n
    for i,k in enumerate(diff):
        if i&1: ans[k]=b[k]
        else: ans[k]=a[k]
    for i in range(n):
        if ans[i]=='2':
            ans[i]=a[i]
    print ''.join(ans)
