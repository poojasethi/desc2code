s = raw_input()
t = raw_input()
distance = 0
for i in range(len(s)):
	if s[i] != t[i]:
		distance += 1
if distance % 2 == 0:
	p = ""
	distance = distance // 2
	for i in range(len(s)):
		if s[i] != t[i] and distance != 0:
			p += s[i]
			distance -= 1
		else:
			p += t[i]
	print p
else:
	print 'impossible'
