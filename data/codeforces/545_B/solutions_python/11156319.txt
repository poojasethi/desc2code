first=raw_input()
second=raw_input()
diff=0
ans=''
down=True
for i,j in zip(first,second):
    if i!=j:
        diff+=1
        if down:
            ans+=j
            down=False
        else:
            ans+=i
            down=True    
        
    else:
        ans+=i    
if diff&1:
    print 'impossible'
else:
    print ans
