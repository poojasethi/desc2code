s = list(raw_input())
t = list(raw_input())
n = len(s)
distance = 0
for i in range(n):
	if s[i] != t[i]:
		distance += 1
if distance % 2 != 0:
	print "impossible"
else:
	distance /= 2
	for i in range(n):
		if s[i] != t[i]:
			s[i] = t[i]
			distance -= 1
		if distance == 0:
			break
	print "".join(s)
