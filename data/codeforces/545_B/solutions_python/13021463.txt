s0 = raw_input()
s1 = raw_input()
diff = sum([1 for i in xrange(len(s0)) if s0[i]!=s1[i]])

if diff%2:
    print "impossible"
else:
    ans = ""
    for i in xrange(len(s0)):
        if s0[i]==s1[i]:
            ans += s0[i]
        elif diff%2:
            diff -= 1
            ans += s0[i]
        else:
            diff -= 1
            ans += s1[i]
    print ans
