s = map(int,raw_input())
t = map(int,raw_input())
k = sum(s[i] ^ t[i] for i in xrange(len(s)))
ans = []
if k % 2:
    print "impossible"
else:
    for x in xrange(len(s)):
        if s[x] != t[x]:
            ans.append([t[x]^1,s[x]^1][k%2])
            k -= 1
        else:
            ans.append(s[x])
    print ''.join(map(str,ans))
