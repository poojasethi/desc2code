s = raw_input()
t = raw_input()

n = len(s)
flag = 0
ans = ''
for i in xrange(n):
    if s[i] != t[i]:
        if flag % 2 == 0:
            ans += s[i]
        else:
            ans += t[i]
        flag += 1
    else:
        ans += s[i]
if flag % 2 == 0:
    print ans
else:
    print 'impossible'
