# Equidistant Strings
s = raw_input()
t = raw_input()
l = len(s)
ans = 0
for i in range(l):
	ans += s[i] == t[i]
if not (ans + l) % 2 == 0:
	print "impossible"
else:
	r=""
	a = b = (l-ans)/2
	for i in range(l):
		if s[i] == t[i]:
			r+=s[i]
		else:
			if a:
				a-=1
				r+=s[i]
			elif b:
				r+=t[i]
	print r
