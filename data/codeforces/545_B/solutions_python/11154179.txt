S=list(raw_input())
T=list(raw_input())
Count=0
for s,t in zip(S,T):
    if not s==t:
        Count+=1
if Count%2: #or S==T:
    print'impossible'
else:
    J=0
    for I in range(Count/2):
        while S[I+J]==T[I+J]:
            J+=1
        if S[I+J]=='1':
            S[I+J]='0'
        else:
            S[I+J]='1'
    print''.join(S)
