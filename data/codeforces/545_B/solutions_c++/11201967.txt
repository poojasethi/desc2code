#include <iostream>
#include <string>
using namespace std;

int main() {
	string s,t,p;
	cin>>s>>t;
	int flag=0;
	for(int i=0;i<s.length();i++){
		if(s[i] != t[i]){
			if(!flag) p +=s[i];
			else p += t[i];
			flag ^= 1;
		}else{
			p += s[i];
		}
		
	}
	if(flag==0)cout<<p<<endl;
	else cout<<"impossible"<<endl;
	return 0;
}
