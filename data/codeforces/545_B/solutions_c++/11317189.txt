#include<iostream>

using namespace std;

int main()
{
	string a,b,c;
	cin >> a >> b;
	int dif = 0;
	for(int i = 0; i < a.size(); i++)
	{
		if(a[i] == b[i])
			c += a[i];
		else
		{
			dif++;
			if(dif % 2 != 0)
				c+= a[i];
			else
				c+= b[i];
		}
	}
	if(dif % 2 == 0)
		cout << c << endl;
	else
		cout << "impossible";
	return 0;
}
