#include <iostream>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	string s,t;
	cin>>s>>t;
	int n=s.size(),q=0,w=0;
	for(int i=0;i<n;i++){
		q=q+(s[i]^t[i]);
	}
	if(q&1)cout<<"impossible";
	else{
	for(int i=0;i<n;i++){
			if(s[i]^t[i])w++;
			if(w<=q/2)cout<<s[i];
			else cout<<t[i];
		}
	}
}
