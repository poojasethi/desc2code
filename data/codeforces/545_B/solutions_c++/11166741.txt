#include<iostream>

using namespace std;

int main(){
	string s,t;
	cin>>s;
	cin>>t;
	int a=0,b=0;
	string p;
	for(int i=0;i<s.size();i++){
		if(s[i]==t[i]) {p+='0';}
		else if(s[i]!=t[i]){
			if(a>=b) { p+=s[i];b++; }
			else { p+=t[i]; a++; }	
		}
	}
	if(a==b) { cout<<p; return 0; }
	else cout<<"impossible\n";
}
