#include <iostream>
using namespace std;

int main()
{
	string s1, s2, s3;
	int flag = 0;
	cin >> s1 >> s2;
	s3 = s1;
	for (int i = 0; i < s3.length(); i++) {
		if (s1[i] != s2[i]) {
			if (flag == 1) {
				s3[i] = s2[i];
			}
			flag ^= 1;
		}
	}
	if (flag == 0) {
		cout << s3 << endl;
	} else {
		cout << "impossible" << endl;
	}
	return 0;
}
