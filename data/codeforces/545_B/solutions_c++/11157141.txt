#include <iostream>
#include <vector>

using namespace std;

int main()
{
	string a;
	string b;
	cin>>a>>b;
	bool f=0;
	for(int i=0;i<a.size();i++)
	{
		if(!f&&a[i]!=b[i])
		{
			f=1;
		}
		else if(f&&a[i]!=b[i])
		{
			f=0;
			a[i]=b[i];
		}
	}
	if(f==1)
		cout<<"impossible";
	else
	{
		cout<<a;
	}
}
