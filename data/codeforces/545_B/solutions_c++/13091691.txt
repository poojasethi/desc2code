#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s[3];
    cin>>s[0]>>s[1];
    s[2]=s[0];
    bool cnt=0;
    for(int i=0;i<s[0].size();++i)
    {
        if(s[0][i]==s[1][i])continue;
        s[2][i]=s[cnt][i];
        cnt=!cnt;
    }
    if(cnt)cout<<"impossible";
    else cout<<s[2];

}
