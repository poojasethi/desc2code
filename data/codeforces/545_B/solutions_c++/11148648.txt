#include<bits/stdc++.h>

int main()
{
	int i,j,k,n,u;
	char s[100001],t[100001],p[100001];
	scanf("%s%s",s,t);
	n=strlen(s);
	j=0;
	for(i=0;i<n;++i)
	{
		if(s[i]!=t[i])
		{
			if(j%2==0)
				p[i]=s[i];
			else
				p[i]=t[i];
			j++;
		}
		else
			p[i]=s[i];
	}
	p[n]='\0';
	if(j%2==0)
	printf("%s\n",p);
	else
		printf("impossible\n");
	return 0;
}
