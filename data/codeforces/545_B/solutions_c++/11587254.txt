#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0), cout.tie(0);
	int siz, dis=0, bit=0;
	string s, t, p;
	cin>>s>>t, p=s;
	siz=s.size();
	for(int i=0;i<siz;++i)
		if(s[i]!=t[i])
			p[i]=((bit)?s[i]:t[i]), bit=1-bit, ++dis;

	cout<<((dis&1)?"impossible":p);
	return 0;
}
