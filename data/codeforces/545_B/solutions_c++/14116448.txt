#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s, t, ans;
	vector<int> a;
	cin>>s>>t;
	for(int i=0 ; i<s.size() ; i++)
	{
		if(s[i] != t[i])
			a.push_back(i);
	}
	ans = s;
	if(a.size()%2)
	{
		cout<<"impossible\n";
		return 0;
	}
	for(int i=0 ; i<a.size()/2 ; i++)
	{
		ans[a[i]] = '1' - ans[a[i]] + '0';
	}
	cout<<ans<<endl;
}
