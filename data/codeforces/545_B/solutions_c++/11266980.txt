#include <bits/stdc++.h>
using namespace std;

string s,t;

int main()
{
	cin>>s>>t;
	int n=s.length(),p=0;
	for(int i=0;i<n;i++)
		p+=s[i]!=t[i];
	string r=s;
	if(p&1)
	{
		puts("impossible");
		return 0;
	}
	for(int i=0,k=0;i<n && k<p/2;i++)
		if(s[i]!=t[i]) r[i]=t[i],k++;
	cout<<r<<endl;
	return 0;
}
