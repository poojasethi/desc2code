#include<bits/stdc++.h>
using namespace std;
int main()
{
	char s[200000],t[200000],a[200000];
	scanf("%s%s",s,t);
	int c=0;int n=strlen(s);
	for(int i=0;i<n;i++)
	{
		if(s[i]!=t[i])
		{
			if(c&1)
			a[i]=s[i];
			else a[i]=t[i];
			c++;
		}
		else
		a[i]=s[i];
	}
	if(c%2==0)
	cout<<a<<endl;
	else
	cout<<"impossible\n";
	return 0;
}
