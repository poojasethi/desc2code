
#include <iostream>

using namespace std;

int main()
{
	string s , t;
	string k;
	cin>>s>>t;
	
	k = s;
	bool f = false;
	int sl = s.length();
	for (int i = 0 ; i< sl ; i++)
	{
		if (s[i] != t[i])
		{
			if (!f) k[i] = s[i];
			else k[i] = t[i];
			f = !f;
		}
	}
	
	cout<<(f ? "impossible" : k);
	
	return 0;
}
