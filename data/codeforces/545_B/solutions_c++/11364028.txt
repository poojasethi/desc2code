#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char s[2][100004];
	bool b=true;
	cin>>s[0];
	cin>>s[1];
	long len=strlen(s[0]);
	for(long i=0;i<len;i++)
	{
		if(s[0][i]!=s[1][i])
		{
			s[0][i]=s[b][i];
			b=!b;
		}
	}
	if(b)cout<<s[0];
	else
	cout<<"impossible";
	return 0;
}
