#include<stdio.h>
#include<string.h>
char A[100005],B[100005];
int n,i,s;
int main(){
	scanf("%s %s",A,B);
	
	n=strlen(A);

	for(i=0;i<n;i++)
		if(A[i]!=B[i])
			s++;
	
	if(s%2 == 1){
		puts("impossible");
		return 0;
	}
	s/=2;
	for(i=0; i<n; i++){
		if(A[i]!=B[i] && s)
			printf("%c",B[i]),s--;
		
		else
			printf("%c",A[i]);
	}
	return 0;
}
