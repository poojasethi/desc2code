#include <iostream>
#include <string>
using namespace std;

int i,d,k;
string s,t,p;

int main() {
	getline(cin,s);
	getline(cin,t);
	for(i=0;i<s.length();i++) 
		if(s[i]!=t[i])
			d++;
	if(d%2) p="impossible";
	else {
		p=s;
		while(d) {
			if(s[k]!=t[k])
				p[k]=t[k], d-=2;
			k++;
		}
	}
	cout<<p;
	return 0;
}
