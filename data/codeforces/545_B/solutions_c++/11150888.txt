#include <iostream>

using namespace std;

int main()
{
	string s, t, p;
	cin >> s >> t;
	p = s;
	int n = s.length();
	bool possible = true;
	for (int i = 0; i < n; i++)
		if (s[i] != t[i])
		{
			if (possible)
				p[i] = t[i];
			possible = !possible;
		}
	if (possible)
		cout << p << endl;
	else
		cout << "impossible" << endl;
	return 0;
}
