#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

ROD_NUM = 3
matrix = [map(int, raw_input().split()) \
            for i in range(ROD_NUM)]
n = input()

dp = [[[0 for k in range(ROD_NUM)] \
        for j in range(ROD_NUM)] \
        for i in range(n+1)]


for i in range(1, n+1) : 
    for j in range(ROD_NUM) : 
        for k in range(ROD_NUM) : 
            if j == k : 
                pass
            else : 
                def getBridge(x, y) : 
                    for b in range(ROD_NUM) : 
                        if b not in [x, y] : 
                            return b
                    return -1

                b = getBridge(j, k)
                # print i, j, k, b
                dp[i][j][k] = min(dp[i-1][j][b] + matrix[j][k] + dp[i-1][b][k], 
                                dp[i-1][j][k] + matrix[j][b] + dp[i-1][k][j] + matrix[b][k] + dp[i-1][j][k])

print dp[n][0][2]
