#include <iostream>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <utility>
#include <algorithm>
#include <cmath>

using namespace std;

int n,t[3][3];
long long dp[41][3][3];

int main() {
	for (int i = 0; i < 3; i++)
		for (int j = 0; j < 3; j++) cin >> t[i][j];
	cin >> n;
	for (int x = 1; x <= n; x++) {
		for (int i = 0; i < 3; i++) {
			for (int j = 0; j < 3; j++) {
				if (i == j) continue;
				int k = 3 - i - j;
				dp[x][i][j] =  min(dp[x-1][i][k] + t[i][j] + dp[x-1][k][j],
				dp[x-1][i][j] + t[i][k] + dp[x-1][j][i] + t[k][j] + dp[x-1][i][j]);
			}
		}
	}
	cout << dp[n][0][2];
}
