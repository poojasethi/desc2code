#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#define eps 1e-10
using namespace std;
int n;
long long f[50][5][5],a[4][4];
void init()
{
    for(int i=1;i<=3;++i)
      for(int j=1;j<=3;++j)cin>>a[i][j];
    scanf("%d",&n);
    for(int k=1;k<=n;++k)
      for(int i=1;i<=n;++i)
        for(int j=1;j<=3;++j)if(i!=j)
        f[k][i][j]=min(f[k-1][i][6-i-j]+a[i][j]+f[k-1][6-i-j][j],f[k-1][i][j]+a[i][6-i-j]+f[k-1][j][i]+a[6-i-j][j]+f[k-1][i][j]);
    cout<<f[n][1][3]<<endl;
    return;
}
int main()
{
    init();
    return 0;
}
