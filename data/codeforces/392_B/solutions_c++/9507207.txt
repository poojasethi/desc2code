#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
int matrix[3][3];
long long dp[3][3][45];
int main()
{
    for(int i=0; i<3; i++)
        for(int j=0; j<3; j++)
            scanf("%d",&matrix[i][j]);
    int n;
    long long a,b;
    scanf("%d",&n);
    for(int k=1; k<=n; k++)
    {
        for(int i=0; i<3; i++)
            for(int j=0; j<3; j++)
            {
                if(i!=j)
                {
                    a=dp[i][3-i-j][k-1]+matrix[i][j]+dp[3-i-j][j][k-1];
                    b=dp[i][j][k-1]+matrix[i][3-i-j]+dp[j][i][k-1]+matrix[3-i-j][j]+dp[i][j][k-1];
                    dp[i][j][k]=min(a,b);
                }
            }
    }
    cout<<dp[0][2][n]<<endl;;
}
