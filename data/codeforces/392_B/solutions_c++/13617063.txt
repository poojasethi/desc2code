#include <cstdio>
#include <algorithm>

using namespace std;

long long dp[41][3][3];

int main(){
  int n,t[3][3];

  for(int i = 0;i < 3;++i)
    for(int j = 0;j < 3;++j)
      scanf("%d",&t[i][j]);

  scanf("%d",&n);

  for(int i = 0;i < 3;++i)
    for(int j = 0;j < 3;++j){
      if(j != i){
        dp[1][i][j] = min(t[i][j],t[i][3 - i - j] + t[3 - i - j][j]);
      }
    }

  for(int k = 2;k <= n;++k){
    for(int i = 0;i < 3;++i){
      for(int j = 0;j < 3;++j){
        if(j != i){
          dp[k][i][j] = min(dp[k - 1][i][3 - i - j] + t[i][j] + dp[k - 1][3 - i - j][j],
            dp[k - 1][i][j] + t[i][3 - i - j] + dp[k - 1][j][i] + t[3 - i - j][j] + dp[k - 1][i][j]);
        }
      }
    }
  }

  printf("%I64d\n",dp[n][0][2]);

  return 0;
}
