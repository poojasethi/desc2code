#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define LL long long
#define For(i,l,r) for(int i=l;i<=r;i++)
using namespace std;
long long f[50][5][5];
int co[5][5];
LL dfs(int n,int s,int t){
    if(!n) return 0;
    if(f[n][s][t]) return f[n][s][t];
    int m=6-s-t;
    f[n][s][t]=min(
    dfs(n-1,s,t)+co[s][m]+dfs(n-1,t,s)+co[m][t]+dfs(n-1,s,t),
    dfs(n-1,s,m)+co[s][t]+dfs(n-1,m,t));
    return f[n][s][t];
}
int main(){
    For(i,1,3) For(j,1,3) scanf("%I64d",&co[i][j]);
    int n; scanf("%d",&n);
    printf("%I64d\n",dfs(n,1,3));
}
