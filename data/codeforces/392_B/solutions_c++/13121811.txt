#include <cstdio>
#include <cstring>
#include <algorithm>
#define ll long long
using namespace std;
ll dp[50][4][4][4],a[3][4];
int n,i,j;
ll Solve(int n,int a1,int a2,int a3)
{
    if(!n) return 0;
    if(dp[n][a1][a2][a3]>=0) return dp[n][a1][a2][a3];
    ll b1=Solve(n-1,a1,a3,a2)+a[a1][a3]+Solve(n-1,a2,a1,a3);
    ll b2=Solve(n-1,a1,a2,a3)+a[a1][a2]+Solve(n-1,a3,a2,a1)+a[a2][a3]+Solve(n-1,a1,a2,a3);
    dp[n][a1][a2][a3]=min(b1,b2);
    return dp[n][a1][a2][a3];
}
int main()
{
    memset(dp,-1,sizeof(dp));
    for(i=1;i<=3;i++)
        for(j=1;j<=3;j++)
            scanf("%d",&a[i][j]);
    scanf("%d",&n);
    printf("%I64d\n",Solve(n,1,2,3));
    return 0;
}
