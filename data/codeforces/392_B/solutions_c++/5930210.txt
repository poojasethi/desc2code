#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long ll;
#define rep(i,n) for (int i=0;i<n;++i)
ll f[45][4][4]; int n;
int main()
{
	rep(i,3) rep(j,3) scanf("%I64d",&f[0][i][j]),f[1][i][j]=f[0][i][j]; scanf("%d",&n);
	rep(t,3) rep(i,3) rep(j,3)
		if (f[1][i][t]+f[1][t][j]<f[1][i][j]) f[1][i][j]=f[1][i][t]+f[1][t][j];
	for (int k=2;k<=n;++k){
		rep(i,3) rep(j,3)
			f[k][i][j]=min(f[k-1][i][j]*2+f[k-1][j][i]+f[0][i][3-i-j]+f[0][3-i-j][j],
							f[k-1][i][3-i-j]+f[0][i][j]+f[k-1][3-i-j][j]);
		rep(t,3) rep(i,3) rep(j,3)
			if (f[k][i][t]+f[k][t][j]<f[k][i][j]) f[k][i][j]=f[k][i][t]+f[k][t][j];
	}
	printf("%I64d\n",f[n][0][2]);
	return 0;
}
