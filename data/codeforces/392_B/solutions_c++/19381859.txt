#include<cstdio>
#include<algorithm>
using namespace std;
long long Dp[45][4][4][4],a[4][4];int n;
long long Get(int n,int p1,int p2,int p3){
	if(n==1)return min(a[p1][p3],a[p1][p2]+a[p2][p3]);
	//if(n==0)return 0;
	if(Dp[n][p1][p2][p3])return Dp[n][p1][p2][p3];
	long long tmp1=Get(n-1,p1,p3,p2)+a[p1][p3]+Get(n-1,p2,p1,p3);
	long long tmp2=Get(n-1,p1,p2,p3)+a[p1][p2]+Get(n-1,p3,p2,p1)+a[p2][p3]+Get(n-1,p1,p2,p3);
	Dp[n][p1][p2][p3]=min(tmp1,tmp2);return Dp[n][p1][p2][p3];
}
int main(){
	for(int i=1;i<=3;i++)
	for(int j=1;j<=3;j++)
		scanf("%d",&a[i][j]);
	scanf("%d",&n);
	printf("%I64d\n",Get(n,1,2,3));
}
