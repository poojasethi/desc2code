#include <iostream>
#include <algorithm>
#include <map>

using namespace std;

long long m[100001][3][3];
int way[1001][1001];

long long ans(int n, int a, int b, int c) {
	if (m[n][a][b] != -1)
		return m[n][a][b];
	if (n <= 0)
		return 0;
	
	long long res = min(
		2 * ans(n - 1, a, b, c) + ans(n - 1, b, a, c) + way[a][c] + way[c][b],
			ans(n - 1, a, c, b) + ans(n - 1, c, b, a) + way[a][b]
	);
	
	m[n][a][b] = res;
	return res;
}

int main() {
	for (int i = 0; i < 3; i++)
		for (int j = 0; j < 3; j++)	
			cin >> way[i][j];
	int n; cin >> n;
	
	for (int i = 0; i <= n; i++)
		for (int j = 0; j < 3; j++)
			for (int k = 0; k < 3; k++)
				m[i][j][k] = -1;
	
	cout << ans(n, 0, 2, 1) << endl;
	
	return 0;
}
