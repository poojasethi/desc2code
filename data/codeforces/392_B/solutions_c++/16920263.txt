#include <bits\stdc++.h>

using namespace std;

int cost[3][3];
long long dp[44][3][3];

long long calc(int n, int from, int to) {
	long long &ret = dp[n][from][to];
	if(ret != -1) {
		return ret;
	}
	if(not n) {
		return ret = 0;
	}
	int foo = 3 - from - to;
	ret = calc(n - 1, from, foo) + cost[from][to] + calc(n - 1, foo, to);
	ret = min(ret, calc(n - 1, from, to) + cost[from][foo] + calc(n - 1, to, from) + cost[foo][to] + calc(n - 1, from, to));
	return ret;
}

int main() {
	for(int i = 0; i < 3; i++) {
		for(int j = 0; j < 3; j++) {
			cin >> cost[i][j];
		}
	}
	int n;
	cin >> n;
	memset(dp, -1, sizeof dp);
	cout << calc(n, 0, 2) << endl;
	return 0;
}
