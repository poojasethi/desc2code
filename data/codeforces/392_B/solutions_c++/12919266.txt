#include <bits/stdc++.h>
using namespace std;
__int64 dp[50][5][5];
__int64 a[5][5];
__int64 dfs(int n, int l, int r)
{
	if (!n)
		return 0;
	if(dp[n][l][r]!=-1)
		return dp[n][l][r];
	int mid = 6 - l - r;
	dp[n][l][r] = min(dfs(n - 1, l, mid) + a[l][r] + dfs(n - 1, mid, r), dfs(n - 1, l, r) + a[l][mid] + dfs(n - 1, r, l) + a[mid][r] + dfs(n - 1, l, r));
	return dp[n][l][r];
}
int main()
{
	for (int i = 1; i <= 3; i++)
		for (int j = 1; j <= 3; j++)
			scanf("%I64d", &a[i][j]);
	memset(dp, -1, sizeof(dp));
	int n;
	scanf("%d", &n);
	printf("%I64d\n", dfs(n, 1, 3));
	return 0;
}
