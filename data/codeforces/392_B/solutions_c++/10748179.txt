#include<iostream>
#include<stdio.h>
using namespace std;
long long a[4][4];
long long r[41][4][4];
long long mini(long long a,long long b)
{
	return a<b?a:b;
}
long long f(int num,int from,int to)
{
	if(r[num][from][to]==0)
	{
		if(num==1)
			r[num][from][to]=mini(a[from][to],a[from][from^to]+a[from^to][to]);
		else
			r[num][from][to]=mini(a[from][to]+f(num-1,from,from^to)+f(num-1,from^to,to),f(num-1,from,to)+a[from][from^to]+f(num-1,to,from)+a[from^to][to]+f(num-1,from,to));
	}
	return r[num][from][to];
}
int main()
{
	int i,j,n;
	for(i=1;i<=3;i++)
		for(j=1;j<=3;j++)
			cin>>a[i][j];
	scanf("%d",&n);
	cout<<f(n,1,3);
  return 0;
}
