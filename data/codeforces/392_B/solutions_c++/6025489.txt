#include<cstdio>
#include<algorithm>
#include<cmath>

using namespace std;

int main(){
int n,i,j,k;
long long t[4][4], a[4][4][45];
    for(i=1;i<4;i++){
        for(j=1;j<4;j++){
            scanf("%I64d",&t[i][j]);
        }
    }
    scanf("%d", &n);
    for(i=1;i<4;i++){
        for(j=1;j<4;j++)
            if(i!=j){
                a[i][j][1]=min( t[i][j], t[i][6-i-j]+t[6-i-j][j]);
                }
    }
    for(k=2;k<n+1;k++){
        for(i=1;i<4;i++){
        for(j=1;j<4;j++){
            if(i!=j)
            a[i][j][k]=min( a[i][6-i-j][k-1]+a[6-i-j][j][k-1]+t[i][j],  2*a[i][j][k-1]+a[j][i][k-1]+t[i][6-i-j]+t[6-i-j][j]);
        }
    }
    }
    printf("%I64d",a[1][3][n]);
}
