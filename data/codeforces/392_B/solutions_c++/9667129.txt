#include<bits/stdc++.h>

using namespace std;

#define lol long long
int main()
{
    int n;

    lol cost[3][3];
    lol dp[3][3][50];

    for(int i=0;i<3;i++)
        for(int j=0;j<3;j++)
           cin>>cost[i][j];

           cin>>n;

    memset(dp,0,sizeof(dp));

    for(int k=1;k<=n;k++)

        for(int i=0;i<3;i++)
            for(int j=0;j<3;j++)
            {if(i!=j)
        {
        lol x=dp[i][3-i-j][k-1]+cost[i][j]+dp[3-i-j][j][k-1];
        lol y=dp[i][j][k-1]+cost[i][3-i-j]+dp[j][i][k-1]+cost[3-i-j][j]+dp[i][j][k-1];
        dp[i][j][k]=min(x,y);
        //cout<<i<<"  "<<j<<"  "<<k<<" "<<dp[i][j][k]<<endl;
        }
           }

    cout<<dp[0][2][n]<<endl;
return 0;

}
