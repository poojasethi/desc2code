#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll dp[50][3][3],t[3][3];
int main(){
    int n;
    for(int i=0;i<3;i++)for(int j=0;j<3;j++)cin>>t[i][j];
    cin>>n;
    for(int i=1;i<=n;i++){
         for(int j=0;j<3;j++){
             for(int k=0;k<3;k++){
                 if(j==k)continue;
                 int l=3-j-k;
                 dp[i][j][k]=min(dp[i-1][j][l]+t[j][k]+dp[i-1][l][k],2*dp[i-1][j][k]+dp[i-1][k][j]+t[j][l]+t[l][k]);
             }
         }
    }
    cout<<dp[n][0][2];
    return 0;
}
