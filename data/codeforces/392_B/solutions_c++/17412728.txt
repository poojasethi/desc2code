#include <cstdio>
#include <algorithm>

using namespace std;

long long f[45][4][4];
int d[4][4];
int i,j,k,n;

int main()
{
	for (i=1;i<=3;i++)
		for (j=1;j<=3;j++)
		{
			scanf("%d",&d[i][j]);
			f[1][i][j]=d[i][j];
		}
	for (k=1;k<=3;k++)
		for (i=1;i<=3;i++)
			for (j=1;j<=3;j++)
				f[1][i][j]=min(f[1][i][j],f[1][i][k]+f[1][k][j]);
	scanf("%d",&n);
	for (i=2;i<=n;i++)
		for (j=1;j<=3;j++)
			for (k=1;k<=3;k++)
			{
				if (j!=k)
					f[i][j][k]=min(f[i-1][j][6-j-k]+f[i-1][6-j-k][k]+d[j][k],
					f[i-1][j][k]+f[i-1][k][j]+f[i-1][j][k]+d[j][6-j-k]+d[6-j-k][k]);
			}
	printf("%I64d",f[n][1][3]);
	return 0;
}
