#include<algorithm>
#include<stdio.h>
#include<iostream>
#include<cstring>
using namespace  std ;

typedef long long ll ;
int n ;
ll dp[42][3][3] , a[3][3] ;

ll f(int n ,int s ,int e ,int i)
{
  if(dp[n][s][e]!=-1) return dp[n][s][e] ;
  if(n==1) return dp[n][s][e]=min(a[s][e],a[s][i]+a[i][e]) ; 
  
  ll A = f(n-1,s,i,e)+f(n-1,i,e,s)+a[s][e] ; 
  ll B = 2LL*f(n-1,s,e,i)+f(n-1,e,s,i)+a[s][i]+a[i][e] ;
  
 return dp[n][s][e]=min(A,B) ;
}

int main()
{
  memset(dp,-1,sizeof(dp)) ;
  for(int i=0 ; i<9 ; i++) cin >> a[i/3][i%3] ;
  scanf("%d",&n) ;
  cout << f(n,0,2,1) ;
   
 return 0 ;
}
