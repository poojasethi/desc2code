#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
int n;
ll a[3][3],f[41][3][3];
ll F(int n,int x,int y)
{
	if(n==0)return 0;
	ll res=f[n][x][y];
	if(res>=0)return res;
	int t=3-x-y;
	res=2*F(n-1,x,y)+F(n-1,y,x)+a[x][t]+a[t][y];
	res=min(res,F(n-1,x,t)+F(n-1,t,y)+a[x][y]);
	return f[n][x][y]=res;
}
int main()
{
	for(int i=0;i<3;i++)
		for(int j=0;j<3;j++)
			scanf("%d",&a[i][j]);
	scanf("%d",&n);
	memset(f,255,sizeof f);
	printf("%I64d",F(n,0,2));
}
