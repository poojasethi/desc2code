#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

long long cost[3][3];
long long dp[3][3][41];
int main(){
	for(int i = 0; i < 3; ++i)
		for(int j = 0; j < 3; ++j)
			cin >> cost[i][j];
	int c;
	cin >> c;
	for(int n = 1; n <= c; ++n){
		for(int from = 0; from < 3; ++from){
			for(int to = 0; to < 3; ++to){
				if(from - to){ 
					int left = (3 ^ from ^ to);
					dp[from][to][n] = min(dp[from][left][n-1] + cost[from][to] + dp[left][to][n-1], dp[from][to][n-1] + cost[from][left] + dp[to][from][n-1] + cost[left][to] + dp[from][to][n-1]);
				}
			}
		}
	}
	cout << dp[0][2][c];
	return 0;
}
