#include <iostream>
#include<cstring>
#define mem(a,b) memset(a,b,sizeof(a))
using namespace std;
typedef long long ll;
ll dp[50][4][4],a[4][4];
int n,i,j;
ll solve(int n,int a1,int a3)
{
    if(!n) 
		return 0;
    if(dp[n][a1][a3]>=0) 
		return dp[n][a1][a3];
	int a2=6-a1-a3;
    ll b1=solve(n-1,a1,a2)+a[a1][a3]+solve(n-1,a2,a3); 
    ll b2=2*solve(n-1,a1,a3)+a[a1][a2]+solve(n-1,a3,a1)+a[a2][a3];
    if(b1<b2)
    	dp[n][a1][a3]=b1;
    else 
		dp[n][a1][a3]=b2;
    return dp[n][a1][a3];
}
int main()
{
    mem(dp,-1);
    for(i=1;i<=3;i++)
        for(j=1;j<=3;j++)
            cin>>a[i][j];
    cin>>n;
    cout<<solve(n,1,3)<<endl;
	return 0;
}
