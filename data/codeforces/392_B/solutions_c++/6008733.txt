#include<iostream>
using namespace std;
int main()
{
	long long t[3][3],dp[41][4][4]={0};
	long long n,i,j,k;
	for(i=1;i<=3;i++)
        for(j=1;j<=3;j++)
            cin>>t[i][j];
	cin>>n;
	for(i=1;i<=n;i++)
        for(j=1;j<=3;j++)
            for(k=1;k<=3;k++)
dp[i][j][k]=min(dp[i-1][j][j^k]+t[j][k]+dp[i-1][j^k][k],2*dp[i-1][j][k]+t[j][j^k]+dp[i-1][k][j]+t[j^k][k]);
	cout<<dp[n][1][3]<<endl;
}
