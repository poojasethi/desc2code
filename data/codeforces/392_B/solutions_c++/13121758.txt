#include <iostream>
using namespace std;
long long n, cost[5][5], f[50][5][5];
int main(){
    for(int i=1; i<=3; i++)
        for(int j=1; j<=3; j++)
            cin>>cost[i][j];
    cin>>n;
    for(int k=1; k<=n; k++)
        for(int i=1; i<=3; i++)
            for(int j=1; j<=3; j++)
                if(i!=j)
                    f[k][i][j]=min(f[k-1][i][6-i-j]+cost[i][j]+f[k-1][6-i-j][j], f[k-1][i][j]+cost[i][6-i-j]+f[k-1][j][i]+cost[6-i-j][j]+f[k-1][i][j]);
    cout<<f[n][1][3]<<endl;
}
