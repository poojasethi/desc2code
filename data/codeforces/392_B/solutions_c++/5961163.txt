#include<iostream>
#include<math.h>
#include<cmath>
using namespace std;
long long n,t[5][5],F[5][5][50];
main()
{
      for(int i=1;i<=3;i++)
      for(int j=1;j<=3;j++)cin>>t[i][j];
      
      cin>>n;
      
      for(int x=1;x<=n;x++)
      for(int i=1;i<=3;i++)
      for(int j=1;j<=3;j++)
      {
              if(i==j)continue;
              int k=6-i-j;
              F[i][j][x]= min(F[i][k][x-1] + t[i][j] + F[k][j][x-1] ,
                               F[i][j][x-1] + t[i][k] + F[j][i][x-1] + t[k][j] + F[i][j][x-1] );
      }
      cout<<F[1][3][n]<<endl;
}
