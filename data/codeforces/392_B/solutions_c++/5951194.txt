#include <cstdio>
#include <cstring>
#include <algorithm>
#define rep(i,l,r) for (int i=l;i<=r;++i)
typedef long long LL;
LL Min(const LL &a,const LL &b){return a>b?b:a;}
LL f[4][4][45],c[4][4];
int n;
int main(){
    rep(i,1,3) rep(j,1,3) scanf("%I64d",&c[i][j]);
    scanf("%d",&n);
    rep(x,1,n){
        rep(i,1,3) rep(j,1,3){
            if (i==j) continue;
            int k=6-i-j;
            f[i][j][x]=Min(f[i][k][x-1]+c[i][j]+f[k][j][x-1],2*f[i][j][x-1]+c[i][k]+f[j][i][x-1]+c[k][j]);
            }
        }
    printf("%I64d\n",f[1][3][n]);
}
