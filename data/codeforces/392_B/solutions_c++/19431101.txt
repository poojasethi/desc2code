#include<cstdio>
#include<map>
#include<cstring>
using namespace std;
typedef long long ll;
ll t[4][4],n,dp[4][4][55];
ll dfs(ll a,ll b,ll n)
{
	ll c=3^a^b;
	if (n==1) return min(t[a][c]+t[c][b],t[a][b]);
	if (dp[a][b][n]!=-1) return dp[a][b][n];
	ll &ret=dp[a][b][n];
	ret=1e18;
	ret=min(ret,t[a][b]+dfs(a,c,n-1)+dfs(c,b,n-1));
	ret=min(ret,t[a][c]+t[c][b]+dfs(a,b,n-1)*2+dfs(b,a,n-1));
	return ret;
}
int main()
{
	for (ll i=0; i<3; i++)
		for (ll j=0; j<3; j++)
			scanf("%I64d",&t[i][j]);
	scanf("%I64d",&n);
	memset(dp,-1,sizeof(dp)); 
	printf("%I64d",dfs(0,2,n));
}
