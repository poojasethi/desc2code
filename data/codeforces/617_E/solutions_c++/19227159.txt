#include <iostream>
#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<math.h>
#define N 1000050
using namespace std;
struct wen{
	int l,r,id;
}q[N];
int n,m;
long long a[N],ans,c[N*2],k,cp[N*2];
bool cmp(const wen a,const wen b){
	if(a.l/400==b.l/400)return a.r<b.r;
	return a.l<b.l;
}
void add(int x)
{
	ans+=c[x^k];
	c[x]++;
}
void fadd(int x)
{
	c[x]--;
	ans-=c[x^k];
}
int main()
{
	scanf("%d%d%lld",&n,&m,&k);
	for(int i=1;i<=n;i++){
		scanf("%d",a+i);
		a[i]^=a[i-1];
	}
	for(int i=1;i<=m;i++){
		scanf("%d%d",&q[i].l,&q[i].r);
		q[i].l--;
		q[i].id=i;
	}
	sort(q+1,q+m+1,cmp);
	long long L=0,R=0;
	c[0]=1;
	ans=0;
	for(int i=1;i<=m;i++){
		while(L<q[i].l){
			fadd(a[L]);
			L++;
		}
		while(L>q[i].l){
			L--;
			add(a[L]);
		}
		while(R<q[i].r){
			R++;
			add(a[R]);
		}
		while(R>q[i].r){
			fadd(a[R]);
			R--;
		}
		cp[q[i].id]=ans;
	}
	for(int i=1;i<=m;i++)
		printf("%lld\n",cp[i]);
    return 0;
}
