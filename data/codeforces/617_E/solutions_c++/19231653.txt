#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
using namespace std;
int n,m,k,a[100010],i,j,s[100010],cnt[1000010],l,r,siz;
long long ans,fi[100010];
struct node{
	int l,r,xh;
	bool friend operator <(node a,node b){
		if(a.l/siz==b.l/siz)return a.r<b.r;
		return a.l/siz<b.l/siz;
	}
}b[100010];
inline void inc(int x){ans+=cnt[a[x]^k],cnt[a[x]]++;}
inline void dec(int x){cnt[a[x]]--,ans-=cnt[a[x]^k];}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	siz=sqrt(n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]),a[i]^=a[i-1];
	for(i=1;i<=m;i++)scanf("%d%d",&b[i].l,&b[i].r),b[i].xh=i,b[i].l--;
	sort(b+1,b+1+m);
	inc(0);
	for(i=1;i<=m;i++){
		while(b[i].l<l)inc(--l);
		while(b[i].r>r)inc(++r);
		while(b[i].l>l)dec(l++);
		while(b[i].r<r)dec(r--);
		fi[b[i].xh]=ans;
	}
	for(i=1;i<=m;i++)printf("%I64d\n",fi[i]);
	return 0;
}
