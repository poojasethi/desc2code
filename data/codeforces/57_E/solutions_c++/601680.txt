using namespace std;
#define li		long long
#define rep(i,to)	for(li i=0;i<((li)to);i++)
#define pb		push_back
#define sz(v)		((li)(v).size())


#define MAX 2000
#define GETA 1000
int mp[MAX][MAX];
li dx[]={2,1,-1,-2,-2,-1,1,2};
li dy[]={1,2,2,1,-1,-2,-2,-1};
int main(){
	li n,m,mod=1000000007,a[405];
	rep(i,405) a[i]=(i==0);
	cin>>n>>m;
	rep(i,MAX)rep(j,MAX) mp[i][j]=(i==GETA && j==GETA);
	rep(i,m){
		li b,c;
		cin>>b>>c;
		mp[GETA+b][GETA+c]=1;
	}
	vector<pair<li,li> > vec(1,make_pair(GETA,GETA));
	rep(i,400){
		vector<pair<li,li> > next;
		rep(j,sz(vec))rep(k,8){
			li x=vec[j].first+dx[k],y=vec[j].second+dy[k];
			if(mp[x][y]==0){
				next.pb(make_pair(x,y));
				a[i+1]++;
				mp[x][y]=1;
			}
		}
		vec=next;
	}
	li tmp=(n-400ll+mod)%mod;
	rep(i,400) a[i+1]=(a[i+1]+a[i])%mod;
	if(n<=400) cout<<a[n]<<endl;
	else if(a[400]==a[399] && a[399]==a[398]) cout<<a[400]<<endl;
	else cout<<(a[400]+((a[400]-a[399])*tmp)%mod+(14ll*((tmp*(1+tmp))%mod)%mod)%mod)%mod<<endl;
}
