#include <cstdio>
typedef long long ll;
const int R=100,T=1000,mo=(int)1e9+7,inv2=500000004,
	dx[]={-2,-2,-1,-1,1,1,2,2},
	dy[]={-1,1,-2,2,-2,2,-1,1};
ll k;
int n;
int F,f[R*2+2][R*2+2],g[R*2+2][R*2+2],cnt[T];
int main(){
	scanf("%I64d%d",&k,&n);
	for(int i=1;i<=n;i++){
		int x,y; scanf("%d%d",&x,&y);
		f[x+R][y+R]=-1;
	}
	f[R][R]=1;
	cnt[0]=1;
	bool flag=0;
	for(int i=1;i<=k;i++){
		for(int x=-R;x<=R;x++)
			for(int y=-R;y<=R;y++)
				g[x+R][y+R]=f[x+R][y+R];
		for(int x=-R;x<=R;x++)
			for(int y=-R;y<=R;y++)
				if(f[x+R][y+R]==1)
					for(int d=0;d<8;d++){
						int xx=x+dx[d],yy=y+dy[d];
						if(xx>=-R&&xx<=R&&yy>=-R&&yy<=R){
							if(g[xx+R][yy+R]!=-1) g[xx+R][yy+R]=1;
						}
						else flag=1;
					}
		if(flag){
			ll	h0=((cnt[i-1]-cnt[i-2])-(cnt[i-2]-cnt[i-3]))%mo,
				g1=h0,g0=((cnt[i-2]-cnt[i-3])-g1*(i-3))%mo,
				f0=(cnt[i-1]-(g0*2+g1*(i-2))%mo*(i-1)%mo*inv2%mo)%mo;
			k%=mo;
			printf("%d\n",int((((g0*2+g1*(k-1))%mo*k%mo*inv2+f0)%mo+mo)%mo));
			return 0;
		}
		for(int x=-R;x<=R;x++)
			for(int y=-R;y<=R;y++)
				f[x+R][y+R]=g[x+R][y+R];
		for(int x=-R;x<=R;x++)
			for(int y=-R;y<=R;y++)
				if(f[x+R][y+R]==1) cnt[i]++;
		if(cnt[i]==cnt[i-1])
			return printf("%d\n",cnt[i]%mo),0;
	}
	printf("%d\n",cnt[k]);
}
