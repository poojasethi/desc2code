#include<iostream>
#include<set>
#include<vector>
using namespace std;
typedef pair<int,int> ii;
typedef long long ll;

set<ii> vis;
int mx[8]={-2,-2,-1,-1,1,1,2,2};
int my[8]={-1,1,-2,2,-2,2,-1,1};
int main() {
    ll K;
    int N;
    cin>>K>>N;
    for(int i=0;i<N;i++) {
        int x,y;
        cin>>x>>y;
        vis.insert(ii(x,y));
    }
    int i=0;
    vector<ii> cur;
    cur.push_back(ii(0,0));
    vis.insert(ii(0,0));
    int pinc,ppinc;
    for(i=0;i<100 && i<K;i++) {
        vector<ii> nxt;
        for(int j=0;j<cur.size();j++) {
            int cx=cur[j].first,cy=cur[j].second;
            for(int m=0;m<8;m++) {
                int nx=cx+mx[m],ny=cy+my[m];
                ii t(nx,ny);
                if(!vis.count(t)) {
                    vis.insert(t);
                    nxt.push_back(t);
                }
            }
        }
        cur=nxt;
        ppinc=pinc;
        pinc=nxt.size();
    }
    const int md=1000000007;
    ll num=(K-i);
    ll dif=pinc-ppinc;
    num%=md;
    dif%=md;
    ll a=pinc+num*dif+pinc+dif;
    if(a%2==0) {
        a/=2;
    } else {
        num/=2;
    }
    a%=md;
    num%=md;
    cout<<(vis.size()-N+a*num)%md;
}
