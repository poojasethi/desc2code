typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;

const int D=2010;
const int L=4*D*D;
const int mod=1000000007;
const int xx=500;
const int dx[]={-1,-2,-2,-1,1,2,2,1};
const int dy[]={-2,-1,1,2,-2,-1,1,2};
int n,i,j,k,l,p,t,x,y,x2,y2;
int q[L][2];int d[D+D][D+D];int cnt[111111];
bool up(int&x,int y){return y<x?x=y,1:0;}
ll K;
ll pow(ll a,ll b,ll c){ll ans=1;for(a%=c;b;b>>=1,a=a*a%c)if(b&1)ans=ans*a%c;return ans;}

int main()
{
    //freopen("1.in","r",stdin);freopen("1.out","w",stdout);
	read(K);read(n);
	CL(d,127);
	rep(i,1,n)read(x,y),d[x+D][y+D]=-1;
	
	q[1][0]=0;q[1][1]=0;d[0+D][0+D]=0;
	for(i=j=1;i<=j;i++)
	{
		x=q[i][0];y=q[i][1];
		cnt[d[x+D][y+D]]++;
		if(d[x+D][y+D]<xx)
		rep(t,0,7)
		{
			x2=x+dx[t];y2=y+dy[t];
			if(up(d[x2+D][y2+D],d[x+D][y+D]+1))
			++j,q[j][0]=x2,q[j][1]=y2;
		}
	}
	
	ll ans=0;
	rep(i,0,min(K,1LL*xx))ans=(ans+cnt[i])%mod;
	//rep(i,0,xx)cout<<cnt[i]<<endl;
	if(K>xx&&cnt[xx]>0)
	{
		ll D=cnt[xx]+28,all=(K-xx);
		ans=(ans+1LL*(D+(D+(all-1)%mod*28%mod))%mod*(all%mod)%mod*pow(2,mod-2,mod)%mod)%mod;
	}
	if(ans<0)ans+=mod;
	cout<<ans<<endl;
	
    return 0;
}
