#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007

int a[100000];
vector<vector<int> > mp;

LL sum,now,lcm;
bool del;

void callit(int u, int p, LL d)
{
    if(del)
        return;
    lcm = lcm/__gcd(lcm,d)*d;

    if(d>sum || d<0)
    {
        cout<<sum;
        exit(0);
    }

//    if(a[u])
    if(mp[u].size()==1 && u!=1)
        now=min(now,a[u]*d);
    for(int i=0;i<mp[u].size();i++)
    {
        int v = mp[u][i];
        if(v==p)    continue;

        callit(v,u,d*(mp[u].size()-(u!=1)));
    }
}

int main()
{
    int n,i,x,y;
    now = LLONG_MAX;
    lcm = 1;
    cin>>n;
    mp.resize(n+1);
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        sum+=a[i];
    }
    for(i=1;i<n;i++)
    {
        cin>>x>>y;
        mp[x].push_back(y);
        mp[y].push_back(x);
    }

    callit(1,0,1);
    cout<<sum-(now/lcm)*lcm;
    return 0;
}
