#include <iostream>

using namespace std;

const int INF = 1 << 24;

int main(int argc, char* argv[])
{
	string s, t;
	cin >> s >> t;
	
	if (s.size() != t.size())
	{
		cout << -1 << endl;
		return 0;
	}
	
	int n;
	cin >> n;
		
	int d[26][26];
	
	for (int i = 0; i < 26; ++i)
		for (int j = 0; j < 26; ++j)
			if (i != j)
				d[i][j] = INF;
			else
				d[i][j] = 0;
		
	for (int k = 0; k < n; ++k)
	{
		char i, j;
		int r;
		cin >> i >> j >> r;
		d[i - 'a'][j - 'a'] = min(d[i - 'a'][j - 'a'], r);
	}		
			
	for (int k = 0; k < 26; ++k)
		for (int i = 0; i < 26; ++i)
			for (int j = 0; j < 26; ++j)
				d[i][j] = min(d[i][j], d[i][k] + d[k][j]);

	int r = 0;
	for (int p = 0; p < s.size(); ++p)
	{
		if (s[p] != t[p])
		{
			int m = INF;
			int b;
			
			for (int k = 0; k < 26; ++k)
			{
				if (m > d[s[p] - 'a'][k] + d[t[p] - 'a'][k])
				{					
					m = d[s[p] - 'a'][k] + d[t[p] - 'a'][k];
					b = k;
				}
				if (m == 0)
					break;
			}
			
			if (m == INF)
			{
				cout << -1 << endl;
				return 0;
			}
			
			s[p] = char(b + 'a');
			r += m;
		}
	}
	
	cout << r << endl;
	cout << s << endl;
	
	return 0;
}
