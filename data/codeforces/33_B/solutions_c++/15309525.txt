#include<bits/stdc++.h>
#define ll long long
using namespace std;
const int MAXN=100,MAXM=27,INF=1e+9;
ll dis[MAXN][MAXN];
int a[MAXN][MAXN];
int main(){
    string s,t;
    cin>>s>>t;
    int n;
    cin>>n;
    char A,B;
    int v;
    for(int i=0;i<MAXN;i++)
        for(int j=0;j<MAXN;j++)
            if(i!=j)
                dis[i][j]=INF;
    for(int i=0;i<n;i++)
        cin>>A>>B>>v,dis[A-'a'][B-'a']=min(dis[A-'a'][B-'a'],(ll)v);
    if(s.size()!=t.size()){
        cout<<-1;
        return 0;
    }
    for(int k=0;k<MAXM;k++)
        for(int i=0;i<MAXM;i++)
            for(int j=0;j<MAXM;j++)
                if(dis[i][j]>dis[i][k]+dis[k][j])
                    dis[i][j]=dis[i][k]+dis[k][j];
    ll ans=0;
    for(int i=0;i<s.size();i++)
        if(s[i]!=t[i]){
            ll d=INF;
            char k;
            for(int j=0;j<MAXM;j++)
                if(dis[s[i]-'a'][j]+dis[t[i]-'a'][j]<d)
                    d=dis[s[i]-'a'][j]+dis[t[i]-'a'][j],k=(j+'a');
            if(d==INF){cout<<-1;return 0;}
            else{ans+=d;s[i]=t[i]=k;}
        }
    cout<<ans<<endl;
    cout<<s<<endl;
    return 0;
}
