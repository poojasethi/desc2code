#include <iostream>
#include <cstdio>
#include <string>
#include <cmath>
using namespace std;

int main()
{
	string s, t;
	cin >> s >> t;
	
	int INF = 100000000;
	int fcost[26][26];
	
	for(int i = 0; i < 26; i++)
		for(int j = 0; j < 26; j++)
			fcost[i][j] = INF;
	for(int i = 0; i < 26; i++)
		fcost[i][i] = 0;
	
	int n;
	cin >> n;
	
	char a, b; int d;
	for(int i = 0; i < n; i++)
	{
		cin >> a >> b>> d;
		fcost[a-'a'][b-'a'] = min(fcost[a-'a'][b-'a'], d);
	}
	
	for(int k = 0; k < 26; k++)
		for(int i = 0; i < 26; i++)
			for(int j = 0; j < 26; j++)
				fcost[i][j] = min(fcost[i][j], fcost[i][k] + fcost[k][j]);
	
	int sl, tl;
	sl = s.length();
	tl = t.length();
	if(sl!=  tl)
	{
		printf("-1\n");
		return 0;
	}
	
	int fincost = 0;
	string finans;
	
	for(int i = 0; i < sl; i++)
	{
		int dist = INF;
		//int mtot = s[i]-'a';
		int mtot;
		for(int q = 0; q < 26; q++)
		{
			int cons = fcost[s[i]-'a'][q] + fcost[t[i]-'a'][q];
			if(cons < dist)
			{
				dist = cons;
				mtot = q;
			}
		}
		if(dist == INF)
		{
			printf("-1\n");
			return 0;
		}
		
		finans.push_back((char)('a'+mtot));
		fincost += dist;
	}
	cout << fincost << endl;
	cout << finans << endl;
}
