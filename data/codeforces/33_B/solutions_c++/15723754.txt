#include<bits/stdc++.h>
using namespace std;
int c[30][30];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    string s,t;
    cin>>s>>t;
    int n;
    cin>>n;
    for(int i=0; i<26; i++)
        for(int j=0; j<26; j++)
            if(i!=j)
                c[i][j]=3000;
    for(int i=0; i<n; i++)
    {
        char a,b;
        int w;
        cin>>a>>b>>w;
        int a1=a-'a',b1=b-'a';
        c[a1][b1]=min(c[a1][b1],w);
    }
    if(s.size()!=t.size())
    {
        cout<<"-1\n";
        return 0;
    }
    for(int k=0; k<26; k++)
        for(int i=0; i<26; i++)
            for(int j=0; j<26; j++)
                c[i][j]=min(c[i][j],c[i][k]+c[k][j]);
    long long sum=0;
    string ans;
    for(int i=0; i<s.size(); i++)
    {
        int a1=s[i]-'a',b1=t[i]-'a',c1=3000,d1;
        for(int j=0; j<26; j++)
            if(c[a1][j]+c[b1][j]<c1)
            {
                c1=c[a1][j]+c[b1][j];
                d1=j;
            }
        if(c1==3000)
        {
            cout<<"-1\n";
            return 0;
        }
        sum+=c1;
        ans+=char(int('a')+d1);
    }
    cout<<sum<<'\n'<<ans<<'\n';
    return 0;
}
