#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <string>

using namespace std;

const int oo = 10000000;

int m, adj[30][30];
string a, b;

int main() {
	for (int i = 0;i < 30;i++) {
		for (int j = 0;j < 30;j++) {
			adj[i][j] = oo;
		}
	}
	for (int i = 0;i < 30;i++) adj[i][i] = 0;
	int c;
	char xx, yy;
	cin >> a >> b >> m;
	for (int i = 0;i < m;i++) {
		cin >> xx >> yy >> c;
		adj[xx - 'a'][yy - 'a'] = min(adj[xx - 'a'][yy - 'a'], c);
	}
	for (int i = 0;i < 30;i++) {
		for (int x = 0;x < 30;x++) {
			for (int y = 0;y < 30;y++) {
				if (adj[x][i] + adj[i][y] < adj[x][y]) adj[x][y] = adj[x][i] + adj[i][y];
			}
		}
	}
	if (a.size() != b.size()) {
		cout << -1 << endl;
		return 0;
	}
	int ans = 0;
	string ret;
	for (int i = 0;i < a.size();i++) {
		int get = oo, idx = 0;
		for (int j = 0;j < 26;j++) {
			if (adj[a[i] - 'a'][j] + adj[b[i] - 'a'][j] < get) {
				get = adj[a[i] - 'a'][j] + adj[b[i] - 'a'][j];
				idx = j;
			}
		}
		if (get == oo) {
			cout << -1 << endl;
			return 0;
		}
		ans += get;
		ret += idx + 'a';
	}
	cout << ans << endl << ret << endl;
	return 0;
}
