#include<iostream>  
#include<stdio.h>  
#include<algorithm>  

using namespace std;

long long sum = 0;
long long n,m;
long long s[10001],l = 0;

void lucky(long long x)
{
	if(x <= m)
	{
		if(x >= n)
		s[l ++] = x;
	    lucky(x * 10 + 4);
	    lucky(x * 10 + 7);
	}
	else
	s[l ++] = x;
}

int main()
{
	cin>>n>>m;
	lucky(0);
	sort(s,s + l);
	for(int i = n,j = 0;i <= m;i ++)
	{
		if(s[j] >= i)
		sum += s[j];
		else
		{
			j ++;
			sum += s[j];
		}
	}
	cout<<sum<<endl;
	return 0;
}
