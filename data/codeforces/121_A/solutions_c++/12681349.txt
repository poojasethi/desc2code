#include<cstdio>
#include<algorithm>
#define N 1000005
using namespace std;
long long a[N];int n,l,r;
void dfs(int k,long long now)
{
    if (now) a[++n]=now;
    if (k<=10) dfs(k+1,now*10ll+4),dfs(k+1,now*10ll+7);
}
long long calc(int L)
{
    long long res=0;
    for (int i=1;i<=n;i++)
        if (L>=a[i]) res+=1ll*a[i]*(a[i]-a[i-1]);
        else return res+=1ll*a[i]*(L-a[i-1]);
    return res;
}
int main()
{
    dfs(1,0);sort(a+1,a+n+1);
    scanf("%d%d",&l,&r);
    printf("%I64d\n",calc(r)-calc(l-1));
}
