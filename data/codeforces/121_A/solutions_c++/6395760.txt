#include<iostream>
using namespace std;
int main()
{
int a[15],i,l;
long long x=0,j=1,r;
cin>>l>>r;
for(i=0;i<=10;i++)
{
a[i]=0;
}
while(j<=r)
{
i=10;
a[i]++;
while(a[i]>2) a[i]=0,i--,a[i]++;
j=0;
for(i=1;i<=10;i++)
{
if(j>0 && a[i]==0) break;
j=j*10+(a[i]==1?4:0)+(a[i]==2?7:0);
}
if(x==0 && j>=l) x+=j;
if(i<=10) continue;
if(j>=l) x+=(min(j,r)-l)*j,l=j;
}
cout<<x;
return 0;
}
