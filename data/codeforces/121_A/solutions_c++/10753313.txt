#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
vector <long long> v;
int arr[8];


void rec(int i,long long a,int len){

if(i==len+1){
 v.push_back(a);

return;
}

 rec(i+1,a*10+4,len);
rec(i+1,a*10+7,len);

}


int main(){

for(int x=1; x<11; x++)rec(1,0,x);
long long a,b,sum=0;
cin>>a>>b;
int x;
for(x=0; x<v.size(); x++)
    if(v[x]>=a)break;


while(a<=b){

    sum+=(min(v[x],b)+1-a)*v[x];
    a=v[x++]+1;
}

cout<<sum<<endl;

return 0;
}
