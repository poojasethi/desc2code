#include <iostream>
using namespace std;

long long temp;
long long next(long long now){
	if(now%10>7)  now=now+10-(now%10)+4;
	if(now>=10){
		temp=next(now/10);
		if(temp>(now/10)) now=temp*10+4;
	}
	if(now%10>4)  now=now/10*10+7;
	else now=now/10*10+4;
	return  now;
}
int main(){
  long long  l,r;
  long long i;
  unsigned long long ans;
  long long m,n;
  while(cin>>l>>r){
	  ans=0;
	  for(i=l;i<=r;){
		  m=next(i);
		  n=m;
		  if(n>r) n=r;
		  ans+=(n-i+1)*m;
		  i=m+1;
	  }
	  cout<<ans<<endl;
  }
}
