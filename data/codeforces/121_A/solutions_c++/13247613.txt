#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

vector <ll> range;
ll l , r ;
ll res=0;


void f(ll num,int cnt ){
if(cnt==12)return;

range.push_back(num);

f( (num*10)+4 , cnt+1);
f( (num*10)+7 , cnt+1);

}


int main()
{
range.push_back(0);
f(4,1);
f(7,1);
sort(range.begin(),range.end());

cin>>l>>r;

for(int i=1;i<range.size();i++){

ll x=max(l,range[i-1]+1);
ll y=min(r,range[i]);

if( x <=y )res+=( ( y-x+1 ) * range[i] );

}
cout<<res<<endl;


    return 0;
}
