#include <bits/stdc++.h>
#define ll long long
//#define mod 1000000007
using namespace std;
ll x[1515]; 
int c=-1;
void dfs(ll p){
	ll w1=p*10+4,w2=p*10+7;
	if(w1>1000000000||w2>1000000000)
		return;
	x[++c]=w1;
	x[++c]=w2;
	dfs(w1);
	dfs(w2);
}
int main(){
	dfs(0);
	x[++c]=4444444444;
	sort(x,x+c+1);
	ll n,m;
	cin>>n>>m;
	ll s=0;
	ll w=n-1;
	for(int i=0;i<=c;++i){
		if(x[i]>=n){
			s+=(min(x[i],m)-w)*x[i];
			w=x[i];
			if(x[i]>=m)
				break;
		}
	}
	cout<<s<<endl;
	return 0;
}
