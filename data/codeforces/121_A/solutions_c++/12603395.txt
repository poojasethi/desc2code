#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

vector<ll> vec;
ll ret = 0, l, r;

void init(ll tmp){
  if (tmp >= 1e10)
    return;
  vec.push_back(tmp);
  init((tmp * 10ll) + 4);
  init((tmp * 10ll) + 7);
}

int main(){
  cin >> l >> r;
  init(0);
  sort(vec.begin(), vec.end());
  ll foo, bar;
  for (int i = 1; i < vec.size(); i++){
    foo = max(l, vec[i - 1] + 1);
    bar = min(r, vec[i]);
    ret += max(0ll, bar - foo + 1) * vec[i];
  }
  cout << ret << endl;
}
