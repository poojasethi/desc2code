#include <iostream>
using namespace std;
long long N = 4ll;
long long next(long long a)
{
	long long mod = 1,k;
	a++;
	while (a > mod)
	{
		k = a%(10*mod)/mod;
		if (k <= 4)
			a = a+(4-k)*mod;
		else if (k <= 7)
			a = a+(7-k)*mod;
		else
			a = a+(14-k)*mod;
		mod *= 10;
	}
	return a;
}
int main()
{
	long long x,y,sum = 0;
	cin >> x >> y;
	while (N<x)
		N = next(N);
	while (N<=y)
	{
		sum += N*(N-x+1);
		x = N+1;
		N = next(N);
	}
	sum += N*(y-x+1);
	cout << sum << endl;
	return 0;
}
