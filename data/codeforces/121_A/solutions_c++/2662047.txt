#include <cstdio>
#include <algorithm>
using namespace std;
long long a[100005],l,r,m;
void f(int lv,long long x){
    if (lv<=10){
        f(lv+1,x*10+4);
        f(lv+1,x*10+7);
    }
    if (x) a[++m]=x;
}
int main(){
    f(0,0);
    sort(a+1,a+m+1);
    scanf("%I64d%I64d",&l,&r);
    long long sum=0,i=l,j=1;
    while(i>a[j]) ++j;
    while(i<=a[j]&&a[j]<r) sum+=(a[j]-i+1)*a[j],i=a[j]+1,++j;
    sum+=a[j]*(r-i+1);
    printf("%I64d\n",sum);
    return 0;
}
