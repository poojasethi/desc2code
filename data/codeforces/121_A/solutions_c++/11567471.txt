#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll r,l;
vector<ll>v;

void solve(ll num){
    if( num> 1e10)return;
    if(num>=l )v.push_back(num);
    solve(num*10+4LL);
    solve(num*10+7LL);
}
int main(){
    cin>>l>>r;
    solve(0);

    sort(v.begin(),v.end());
    ll x=l,sum=0;
    for(int i=0;i<(int)v.size() && x<=r;i++){
        ll cnt=min(v[i],r)-x+1;
        sum+=cnt*v[i];
        x=v[i]+1;
    }
    cout<<sum<<"\n";
}
