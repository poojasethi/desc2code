#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int l, r;
vector<ll> v;

void gen(ll x) {
	v.push_back(x);
	if(x > 1e9) return;
	gen(x*10+4);
	gen(x*10+7);
}

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	cin >> l >> r;
	gen(0);
	sort(v.begin(), v.end());
	ll ans = 0, mn, mx;
	for(int i=1; i<(int)v.size() ; ++i) {
		mn = min((ll)r, v[i]);
		mx = max((ll)l, v[i-1]+1);
		ans += v[i] * max(0LL, mn-mx+1);
	}
	cout << ans << endl;
	return 0;
}
