#include <iostream>
#include <algorithm>
using namespace std;
long long ans,a[3000],r,l,n,inf=1000*1000*1000;
void f(long long  x){
    if(x>inf) return ;
    a[n++]=x;
    f(x*10+4);  f(x*10+7);
}
int main()
{
    inf*=5;
    cin>>l>>r;
    f(0);
    sort(a,a+n);
    for(int i=0;i<n;i++){if(l>r)   break;
        if(l<=a[i]){
            ans+=(min(a[i],r)-l+1)*a[i];
            l=a[i]+1;
        }
    }
    cout<<ans;
}
