#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
const int maxn=111111;
ll a[maxn];
int num=0;
void dfs(int len,ll x)
{
	if(x!=0)
	a[num++]=x;
	if(len==0)return;
	dfs(len-1,x*10+4);
	dfs(len-1,x*10+7);
	return;
}
int main()
{
	dfs(11,0);
	sort(a,a+num);
	ll l,r;cin>>l>>r;
	ll t=0;
	while(a[t]<l)t++;
	ll sum=0;
	for(int i=t;i<num;i++){
		if(l>r)break;
		if(a[i]>=l)
			sum+=a[i]*(min(a[i],r)-l+1),l=min(a[i],r)+1;
	}
	cout<<sum<<endl;
}
