#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

long long v[1<<15];

int main(void) {
	v[1]=0;
	v[2]=4;
	v[3]=7;
	for(int i=2;i<=2048;i++) {
		v[i*2]=v[i]*10+4;
		v[i*2+1]=v[i]*10+7;
	}
	int L, R;
	long long ans=0;
	scanf("%d%d", &L, &R);
	for(int i=2;i<2048;i++) {
		if(v[i]>=L && v[i-1]<R) {
			long long t = min((long long)R+1, v[i]+1) - max(v[i-1]+1, (long long)L) ;
			ans += t*v[i];
		}
	}
	printf("%I64d\n", ans);

	return 0;
}
