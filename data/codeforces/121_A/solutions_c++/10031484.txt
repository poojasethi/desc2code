#include <bits/stdc++.h>
using namespace std;
#define int long long

int nxt(int x) {
	int z = 1, y = x;
	while (x) {
		if (x % 10 == 4)
			return y + 3 * z;
		y -= 3 * z;
		z *= 10;
		x /= 10;
	}
	return 4 * z + y;
}
main() {
	ios_base::sync_with_stdio(false);
	int f = 0, l, r;
	cin >> l >> r;
	while (f < l)
		f = nxt(f);
	int ans = (min(r, f) - l + 1) * f;
	if (f < r)
		while (true) {
			int t = nxt(f);
			ans += (min(r, t) - f) * t;
			if (t >= r)
				break;
			f = t;
		}
	cout << ans << '\n';
	return 0;
}
