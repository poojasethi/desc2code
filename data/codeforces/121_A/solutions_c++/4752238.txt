#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;
long long l,r;
vector <long long> a;
long long next(long long x)
{ if (x<=r*10)
  {a.push_back(x);
   next(x*10+4);next(x*10+7);
  }
}
int main()
{//freopen("i.txt","r",stdin);
cin>>l>>r;
next(0);
sort(a.begin(),a.end());
long long ans=0;
for (int i=0;i<a.size();i++)
{//cout<<i<<"   "<<a[i]<<endl;
 if (l>r) break;
 if (l<=a[i]) {ans+=(min(a[i],r)-l+1)*a[i];l=a[i]+1;}
}
cout<<ans;
return 0;
}
