#include<vector>
#include<iostream>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

void dfs(ll num,vector<ll> &L){
	if(num>4444444444LL) return;
	L.push_back(num);
	dfs(num*10+4,L);
	dfs(num*10+7,L);
}

ll count(ll a,ll b,ll l,ll r){
	return max(min(b,r)-max(a,l)+1,0LL);
}

int main(){
	vector<ll> L;
	dfs(0,L);
	sort(L.begin(),L.end());

	ll l,r; cin>>l>>r;

	ll ans=0;
	rep(i,(int)L.size()-1) ans+=count(L[i]+1,L[i+1],l,r)*L[i+1];
	cout<<ans<<endl;

	return 0;
}
