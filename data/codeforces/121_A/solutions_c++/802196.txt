//#include "stdafx.h"
#include <iostream>
#include <algorithm>
using namespace std;


long long X[10000];
int i,n=0;
long long l,r;

void go(long long x) {
	if (x>=l)
		X[n++]=x;
	if (x<r) {
		go(x*10+4);
		go(x*10+7);
	}
}

int main(void)
{
	cin>>l>>r;
	go(0);
	sort(X,X+n);
	if (X[0]<=r) {
		long long res=(1+X[0]-l)*X[0];
		for (i=1;X[i]<r;i++)
			res+=(X[i]-X[i-1])*X[i];
		res+=(r-X[i-1])*X[i];
		cout<<res;
	} else
		cout<<(r-l+1)*X[0];
	return 0;
}
