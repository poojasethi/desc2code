#include<cstdio>
#include<cstring>
typedef __int64 ll;
int l,r,n,flag;
ll ans;
void dfs(int k,ll s)
{
	if(flag)return;
	if(s>=r)
	{
		ans+=s*(r-l+1);
		flag=1;
		return;
	}
	if(n==k)
	{
		if(s>=l)
		{
			ans+=s*(s-l+1);
			l=(int)s+1;
		}
		return;
	}
	dfs(k+1,s*10+4);
	dfs(k+1,s*10+7);
}
int main()
{
	scanf("%d%d",&l,&r);
	for(n=1;n<=10;n++)
		dfs(0,0);
	printf("%I64d\n",ans);
	return 0;
}
