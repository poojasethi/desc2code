#include <iostream>
#include <vector>
#include <stdio.h>
#include <set>
using namespace std;
const int N=500005;
typedef long long ll;
typedef pair<ll,int>ii;
ll l,r;
set<ll>st;
void generate(ll a,int z){
 if(z>11)return;
 if(a>=l)st.insert(a);
 generate(a*10+4,z+1);
 generate(a*10+7,z+1);
return;
}
int main()
{
 //freopen("test.in","r",stdin);
 cin>>l>>r;
 generate(0,0);
 ll p=l,sol=0;
 while(p<=r){
  ll z=*(st.lower_bound(p));
  sol+=(min(z,r)-p+1)*z;
  p=z+1;
 }
 cout<<sol;
 return 0;
}
