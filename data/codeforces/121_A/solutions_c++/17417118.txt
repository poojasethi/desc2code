#include<cstdio>
#include<algorithm> 
using namespace std;
long long tr[2000];
int main()
{
	tr[1] = 0;
	for(int i=1;i<=660;i++)
	  {
	    tr[i*2] = tr[i]*10+4;
	    tr[i*2+1] = tr[i]*10+7;
	  }
    long long l,r,pre,ans = 0;
	scanf("%I64d%I64d",&l,&r);
	for(int i=1;i<=1025;i++)
	{
	   if(tr[i]<l)continue;
	   else if(tr[i]>=r)
	   {
	   	  ans+=tr[i]*(r-max(l,tr[i-1]+1)+1);
	   	  break;
	   }
	   else {
	   	  ans+=tr[i]*(tr[i]-max(l,tr[i-1]+1)+1);
	   }
     }
	   printf("%I64d\n",ans);
	return 0;
}
