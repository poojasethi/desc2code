# include <iostream>
# include <cstdio>
using namespace std;

const int N = 2046;
long long a[N+1], L, R;

long long f(long long n)
{
	if ( n==0 ) return 0;
	long long res = 0;
	for(int i=1; i<=2046; i++)
		if(a[i] <= n)
			res += a[i] * (a[i] - a[i-1]);
		else {
			res += a[i] * (n-a[i-1]);
			break;
		}
	return res;
}

int main()
{
	int t = 2;
	a[1] = 4; a[2] = 7;
	for(int i=1; i<=2*511; i++)
		a[++t] = a[i] * 10 + 4,
		a[++t] = a[i] * 10 + 7;
	cin >> L >> R;
	cout << f(R) - f(L-1) << endl;
	return 0;
}
