#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
vector<ll> lky;
ll l,r,ans;
void generate(ll num){
	if(num >= 1e11) return;
	lky.push_back(num);
	generate(num*10+4);
	generate(num*10+7);
}

int main(){
	cin>>l>>r; generate(0);
	sort(lky.begin(),lky.end());
	while(l <= r){
		ll t = (*(lower_bound(lky.begin(),lky.end(),l)));
		ans += (min(t-l+1,r-l+1)*t);
		l = min(t,r)+1;
	}
	cout << ans << endl;
}
