#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;

long long f[2000];
long long l,r,ans,n;

int main()
{
	cin>>l>>r;
	n=1; ans=0;
	f[0]=2; f[1]=4; f[2]=7;
	while (f[f[0]]<=r) {
		f[++f[0]]=f[n]*10+4;
		f[++f[0]]=f[n++]*10+7;
	}
	n=l;
	for (int i=1; i<=f[0]; i++) {
		if (n<=f[i]) {
			ans+=(min(f[i],r)-n+1)*f[i];
			n=f[i]+1;
		}
		if (n>r) break;
	}
	cout<<ans<<endl;
	return 0;
}
