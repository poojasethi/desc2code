#include<queue>
#include<cstdio>
#include<algorithm>
using namespace std;
queue<__int64>q;
int main()
{
    q.push(0ll);
    __int64 a,b,ans=0,pre,st;
    scanf("%I64d%I64d",&a,&b);
    pre=a-1;
    while(1)
    {
        st=q.front();
        q.pop();
        if(st>pre)
        {
            ans+=(min(b,st)-pre)*st;
            pre=st;
        }
        if(st>=b) break;
        q.push(10*st+4);
        q.push(10*st+7);
    }
    printf("%I64d\n",ans);
    return 0;
}
