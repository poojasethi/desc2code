lucky=set([])
def gen(x):
    if x>4444444444:
        return
    lucky.add(x)
    gen(x*10+4)
    gen(x*10+7)
gen(4)
gen(7)
lucky=sorted(lucky)

def nextlu(x):
    l=0
    r=len(lucky)-1
    while r-l>1:
        if lucky[(r+l)/2]>=x:
            r=(r+l)/2
        else:
            l=(r+l)/2
    if lucky[l]>=x:
        return lucky[l]
    else:
        return lucky[r]

l,r=map(int,raw_input().split())
res=0
while r>=l:
    t=nextlu(l)
    res+=t*(min(r,t)-l+1)
    l=t+1
print res
