from sys import stdin

l, r = [int(s) for s in stdin.readline().split(' ')]

all = []
def make(n):
    if n <= 10*r:
        if n >= l:
            all.append(n)
        make(10*n+4)
        make(10*n+7)

make(4)
make(7)
all.sort()

rv = 0
pr = l-1
for n in all:
    if n > r:
        rv += (r - pr) * n
        break
    rv += max(0, (n-pr)) * n
    pr = n
    
print rv
