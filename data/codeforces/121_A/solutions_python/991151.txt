def f(g):
    s=g[:]
    for i in range(len(s)-1,-1,-1):
        if s[i]=='4':
            for j in range(len(s)-1,i,-1):
                s[j]='4'
            s[i]='7'
            break
    else:
        s=['4']*(len(s)+1)
    return s
r,l=map(int,raw_input().split())
a=r-1
s=['4']
while int(''.join(s)) < r:
    s=f(s)
t=0
while True:
    icur=int(''.join(s))
    
    m=min(icur,l)-a
    
    t+=m*icur
    if icur >= l:
        break
    a=icur
    s=f(s)
print t
