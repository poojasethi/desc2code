
t = ['']
a = []
for _ in range(10):
  u = ['4'+s for s in t] + ['7'+s for s in t]
  a += u
  t = u
a = map(int,a)
a.sort()
l,r=map(int,raw_input().split())
res = 0
for i in range(2**10):
  if l<=a[i]:
    rr = min(a[i],r)
    res += a[i]*(rr-l+1)
    l = a[i]+1
  if a[i]>=r:
    break
print res
