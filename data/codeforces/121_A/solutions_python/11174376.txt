def ns(l,ans):
    for i in ans:
        if(i>=l):
            return i
def prog(s,ans):
    if(s<(10**10)):
        ans+=[s]
        prog((s*10)+4,ans)
        prog((s*10)+7,ans)
ans = []
prog(0,ans)
ans.sort()
del ans[0]
l,r=map(int,raw_input().split())
res = 0
while(l<=r):
    x=ns(l,ans)
    y = min(r,x)
    res += (y-l+1)*x
    l = y+1
print res
