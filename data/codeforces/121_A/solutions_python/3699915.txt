lucky = [4,7]
def init() :
    for x in lucky:
        if x>777777777:
            break
        lucky.append(x*10+4)
        lucky.append(x*10+7)
l,r = map(int,raw_input().split())
sums = i = 0
init()
lucky.sort()
while l<=r:
    if lucky[i]<l :
        i+=1
        continue
    if lucky[i]<=r :
        sums+=lucky[i]*(lucky[i]-l+1)
    else :
        sums+=lucky[i]*(r-l+1)
    #print lucky[i]
    l=lucky[i]+1
print sums
