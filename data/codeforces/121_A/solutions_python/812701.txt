import sys

l,r=map(int,sys.stdin.readline().strip().split(' '))

Lucky=[-1]

for d in range(1,10):
	for b in range(1<<d):
		a=0;p=1
		for i in range(d):
			if(b&(1<<i)==0): a+=p*4
			else: a+=p*7
			p*=10
		Lucky.append(a)

Lucky.append(4444444444)

Lucky.sort()

ans=0
for i in xrange(len(Lucky)-1):
	a=Lucky[i]+1;b=Lucky[i+1];
	if(b<l or r<a):continue
	ans+=(min(b,r)-max(a,l)+1)*b

print ans
