a, b = [], ['']
for _ in range(10):
    c = ['4' + s for s in b] + ['7' + s for s in b]
    a += c
    b = c
a = sorted(map(int, a))
l, r = map(int, raw_input().split())
res = 0
for i in range(len(a)+1):
    if l <= a[i]:
        rr = min(a[i], r)
        res += a[i] * (rr - l + 1)
        l = a[i] + 1
    if a[i] >= r: break
print res
