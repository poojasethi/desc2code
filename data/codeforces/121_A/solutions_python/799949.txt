#!/usr/bin/python
import sys

a = []

def gen (n, d):
	global a
	if n > 0:
		a += [n]
	if d > 0:
		gen (n * 10 + 4, d - 1)
		gen (n * 10 + 7, d - 1)

def go (l, r, pos, c):
	l = max (l, pos)
	r = min (r, c)
#	print l, r, c
	return max (r - l + 1, 0) * c

gen (0, 10)
a.sort ()

while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		s = sys.stdin.readline ().strip ()
		if s == '':
			break
	l, r = [int (x) for x in s.split ()]
	res = 0
	pos = 0
	for c in a:
		res += go (l, r, pos, c)
		pos = c + 1
	print res
