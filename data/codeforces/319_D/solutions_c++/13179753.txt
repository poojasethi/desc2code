#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <algorithm>
using namespace std;
#define N 50050
int n;char b[N];
int main()
{
    scanf("%s",b);n=strlen(b);
    for (int len=1;len<=n/2;len++)
     {
        int cnt=0,p=len;
        for (int i=len;i<n;i++)
         {
            cnt=(b[i]==b[i-len]?cnt+1:0);
            if (cnt==len)
             {
                p-=len;
                cnt=0;
             }
            b[p++]=b[i];
         }
        b[n=p]=0;
     }
    puts(b);
    return 0;
}
