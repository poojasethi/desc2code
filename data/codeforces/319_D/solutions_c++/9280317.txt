#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int a[50005], n, c;

void read()
{
	for(; c=getchar(), c<'a'; );
	for(a[n=1]=c; c=getchar(), c>='a'; a[++n]=c); 
}

int main()
{
	read();
	
	for(int i=1, l=n; i<=l/2; )
	{
		int fd(0);
		for(int j=1, las(0); j+i<=n && !fd; j++)
		{
			if(a[j]==a[j+i]) ++las; else las=0;
			if(las==i) fd = j;
		}
		if(!fd) {i++; continue;}
		for(int j=fd+i+1; j<=n; j++) a[j-i]=a[j];
		n-=i;
	}
	
	for(int i=1; i<=n; i++) printf("%c", a[i]);
	
	return 0;
}
