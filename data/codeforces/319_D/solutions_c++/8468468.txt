#include<cstdio>
#include<cstring>
int n;
char s[100100];
int main()
{
    int i,j,k,m;
    scanf("%s",s+1);
    n=strlen(s+1);
    for(i=1;i+i<=n;++i)
    {
        k=0;
        m=i;
        for(j=i+1;j<=n;++j)
        {
            if(s[j]==s[j-i])
                ++k;
            else
                k=0;
            if(k==i)
            {
                k=0;
                m-=i;
            }
            s[++m]=s[j];
        }
        n=m;
    }
    for(i=1;i<=n;++i)
        printf("%c",s[i]);
    puts("");
    return 0;
}
