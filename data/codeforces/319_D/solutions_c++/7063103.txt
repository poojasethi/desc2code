#include <cstdio>
#include <cstring>

using namespace std;

const int MaxN = 50010;

char st[MaxN];
int len;

int main()
{
	scanf("%s",st+1);len=strlen(st+1);
	for(int i=1;i<=(len>>1);i++){
		int n=i,cnt=0;
		for(int op=i+1;op<=len;op++){
			st[op-i]==st[op] ? cnt++ : cnt=0;
			cnt==i ? (cnt=0,n-=i) : 0;
			st[++n]=st[op];
		}
		st[(len=n)+1]=0;
	}
	puts(st+1);return 0;
}
