	scanf("%s", S); n = strlen(S);
	for (int i = 1; i + i <= n; i++) {
		int cur = 0, m = i; 
		for (int j = i; j < n; j++) {
			S[m++] = S[j];
			if (S[m - 1] == S[m - i - 1])
				{if (++cur == i) cur = 0, m -= i;}

		printf("%c", S[i]);

}
