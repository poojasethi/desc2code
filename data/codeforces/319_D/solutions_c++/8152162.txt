#include <cstdio>
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;
int n,m,i,j,k,x,y,z,len;
char s[500001];

int main() {
	ios :: sync_with_stdio(false);
	cin >> s;
    n = strlen(s);
    for (i = 1; i <= n / 2; i++) {
        j = i;
        for (k = i, len = 0; k < n; k++) {
	        if (s[k] == s[k - i]) len++;else len = 0;
            if (len == i) j -= i, len = 0;
            s[j++] = s[k];
        }
        n = j;
    }
    s[n] = 0;
    cout << s;
    return 0;
}
