#include <algorithm>
#include <stdio.h>
#include <string.h>

using namespace std;

char s[2][51000];

int main() {
//	freopen("have.in", "r", stdin);
//	freopen("have.out", "w", stdout);
	
	scanf("%s", s[0]);
	int n=strlen(s[0]);
	int u=0;
	for (int len=1; 2*len<=n; len++) {
		int cnt=0;
		int m=0;
		for (int i=0; i+len<n; i++) {
			s[u^1][m++]=s[u][i];
			cnt=(s[u][i]==s[u][i+len] ? cnt+1 : 0);
			if (cnt==len) {
				m-=cnt;
				cnt=0;
			}
		}
		for (int i=n-len; i<n; i++) s[u^1][m++]=s[u][i];
		n=m; u^=1;
	}
	s[u][n]=0; puts(s[u]);
	
	return 0;
}
