#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	string s; cin>>s;
	int n=(int)s.size();
	int tlen=1;
	while(tlen*2<=n)
	{
		bool flag=false;
		int it=0,pos=-1;
		for(int i=0;i<n-tlen;i++)
		{
			if(s[i]==s[i+tlen])
				it++;
			else it=0;
			if(it==tlen)
			{
				flag=true;
				pos=i;
				break;
			}
		}
		if(flag)
		{
			for(int i=pos;i<n-tlen;i++)
				s[i]=s[i+tlen];
			n-=tlen; tlen=1;
		}
		else tlen++;
	}
	for(int i=0;i<n;i++)
		cout<<s[i];
	cout<<'\n';
	return 0;
}
