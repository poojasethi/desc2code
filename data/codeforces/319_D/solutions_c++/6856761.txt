#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>

using namespace std;

int n;
char str[500010];
int main()
{
    scanf("%s",str);
    n = strlen(str);
    int tmp;
    for (int i = 1;i + i <= n;i++)
	{
	    tmp = i;
	    for (int j = i, k = 0;j < n;j++)
		{
		    if (str[j] == str[j - i]) k++; else k = 0;
		    if (k == i) tmp -= i ,k = 0;
		    str[tmp++] = str[j];
		}
	    n = tmp;
	}
    str[n] = 0;
    return printf("%s\n",str) , 0;
}
