#include <cstring>
#include <cstdio>

using namespace std;

char s[50010];

int main(){
    gets(s);
    int n = strlen(s);
    for(int len = 1, i, j, k; len <= n; ++len, s[j] = 0){
        for(i = len, j = len, k = 0; s[i]; ++i, ++j){
            k = s[i] == s[i - len] ? k + 1 : 0;
            s[j] = s[i];
            if(k == len) j -= len, k = 0;
        }
    }
    puts(s);
    return 0;
}
