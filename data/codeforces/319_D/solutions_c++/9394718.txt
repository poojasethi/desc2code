#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;
int num,x,i,j,p;
char s[100005];
int main()
{
    scanf("%s",s);
    p=strlen(s);
    for (i=1; i<=p/2; i++)
    {
        num=0; x=i;
        for (j=i; j<p; j++)
        {
            if (s[j]==s[j-i]) num++; else num=0;
            if (num==i) {x-=i; num=0;}
            s[x++]=s[j];
        }
        s[p=x]=0;
    }
    for (i=0; i<p; i++) cout<<s[i];
    return 0;
}
