#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int n,m,i,cnt;
char s[50005];
bool flag;

int main()
{
  gets(s);
  n=strlen(s);
	
//
  cnt=1;
  while(cnt<=(n>>1))
  { 
	 m=0; flag=false;
	 for(i=0;i<n-cnt;i++)
	 { 
		if(s[i]==s[i+cnt])m++;
	    else m=0;
	    
	    if(m==cnt){ flag=true; break; }
	 }
	 
	 if(!flag){ cnt++; continue; }
	 for(;i<=n-cnt;i++)s[i]=s[i+cnt];
	 n-=cnt; cnt=1;
  }
  puts(s);
  
//
  return 0;
}
