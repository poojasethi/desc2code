#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
using namespace std;
int tot,n,l,m,i;
char c[120000],v[120000];
int main(){
	scanf("%s",c+1);
	n=strlen(c+1);
	for(l=1;l<=n/2;++l){
		tot=0; m=0;
		for(i=l+1;i<=n;++i){
			if(c[i]==c[i-l])m++;
			else m=0;
			v[++tot]=c[i];
			if(m==l)m=0,tot-=l;
		}
		n=l;
		for(i=1;i<=tot;++i)c[++n]=v[i];
	}
	c[n+1]=0;
	printf("%s\n",c+1);
}
