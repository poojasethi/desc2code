#include <cstdio>
#include <cstring>

int main()
{
	static char s[50 * 1000 + 5];
	gets(s);
	int n = strlen(s);
	for (int i = 1; i <= n / 2; ++i) {
		int pos;
		do {
			pos = -1;
			int segm = 0;
			for (int j = 0; j < n - i; ++j) {
				if (s[j] == s[i + j]) {
					++segm;
					if (segm == i) {
						pos = j;
						break;
					}
				}
				else {
					segm = 0;
				}
			}

			if (pos != -1) {
				for (int j = pos; j < n - i; ++j) {
					s[j] = s[i + j];
				}
				n -= i;
			}
		} while (pos != -1);
	}

	s[n] = 0;
	puts(s);

	return 0;
}
