#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
char s[100000];
int n;
int main()
{
	scanf("%s", s);
	n = strlen(s);
	int len = 1;
	while(len <= n/2)
	{
		bool ok = false;
		int k = 0;
		int z;
		for(int i = 0; i < n-len; i++)
		{
			if(s[i] == s[i+len])
				k++;
			else
				k = 0;
			if(k == len)
			{
				ok = true;
				z = i;
				break;
			}
		}
		if(!ok)
		{
			len++;
			continue;
		}
		for(int i = z; i <= n-len; i++)
			s[i] = s[i+len];
		n -= len;
		len = 1;
	}
	printf("%s\n", s);
	return 0;
}
