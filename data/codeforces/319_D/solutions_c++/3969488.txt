#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

char s[50005];

int main(void){
    //freopen("D:\\acm.txt","r",stdin);
    scanf("%s",s);
    int n = strlen(s);
    for(int len = 1;len<=n/2;len++){
        int cnt = 0,pos = len;
        for(int i = len;i<n;i++){
            if(s[i]==s[i-len]) cnt++;
            else cnt = 0;
            if(cnt==len) pos-=len,cnt = 0;
            s[pos++] = s[i];
        }
        n = pos;
    }
    s[n] = '\0';
    printf("%s\n",s);
}
