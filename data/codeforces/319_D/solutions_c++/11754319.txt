#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
char s[60010];
int n;
void work(void){
	for(int len=1;len*2<=n;len++){
		int now=len,p=0;
		for(int i=len+1;i<=n;i++){
			if(s[i]==s[now-len+1]) p++;
			else p=0;
			s[++now]=s[i];
			if(p==len){
				now-=len;
				p=0;
			}
		}
		n=now;
	}
	s[n+1]='\0';
	printf("%s\n",s+1);
}
void read(void){
	scanf("%s",s+1);
	n=strlen(s+1);
}
int main(){
	read();
	work();
	return 0;
}
