#include <bits/stdc++.h>
using namespace std;

char s[50010];

int main()
{
	gets( s );
	int n = strlen( s ), len = 1;
	while( len <= n/2 )
	{
		int i, cc = 0;
		for( i = 0 ; i+len < n ; i ++ )
		{
			if( s[i] == s[i+len] )
				cc ++;
			else
				cc = 0;
			if( cc == len )
				break;
		}
		if( cc == len )
		{
			for( int j = i ; j+len <= n ; j ++ )
				s[j] = s[j+len];
			n -= len;
			len = 1;
		}
		else
			len ++;
	}
	puts( s );
	return 0;
}
