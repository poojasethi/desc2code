//In the name of God
#include <iostream>
using namespace std;

int main() {
	ios_base::sync_with_stdio(false);
	string s;
	cin >> s;
	for (int l = 1; l <= s.size() / 2; l++) {
		int cur = l, pot = 0;
		for (int i = l; i < s.size(); i++) {
			s[cur] = s[i];
			if (s[cur] == s[cur - l])
				pot++;
			else
				pot = 0;
			if (pot == l)
				pot = 0, cur -= l;
			cur++;
		}
		s.resize(cur);
	}
	cout << s << '\n';
	return 0;
}
