#include <cstdio>
#include <cstring>
using namespace std;

const int N=50000+5;

int n;
char st[N];

int main(int argc,char **argv)
{
	#ifndef ONLINE_JUDGE
		if (argc==1) freopen("319D.in","r",stdin),freopen("319D.out","w",stdout);
		else freopen(argv[1],"r",stdin),freopen(argv[2],"w",stdout);
	#endif
	int i=1;
	scanf("%s",st);n=strlen(st);
	while (i*2<=n)
	{
		int cnt=0;
		for (int j=i;j<n;++j)
			if (st[j]==st[j-i])
			{
				if (++cnt==i)
				{
					for (int k=j+1;k<n;++k) st[k-i]=st[k];
					n-=i;break;
				}
			}else cnt=0;
		if (cnt!=i) ++i;
	}
	st[n]=0;
	puts(st);
	return 0;
}
