#include <cstdio>

int main() {
  int n;
  char s[50005];
  scanf("%s", s);
  for(; s[n]; n++);
  for(int len = 1; len <= n/2; len++) {
    int m = len, same = 0;
    for(int i = len; i < n; i++) {
      same = s[i] == s[i-len]? same+1: 0;
      if(same == len) {
	m -= len;
	same = 0;
      }
      s[m++] = s[i];
    }
    s[n = m] = 0;
  }
  printf("%s\n", s);
}
