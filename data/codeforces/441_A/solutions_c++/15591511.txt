#include<stdio.h>
int main()
{
	int n,v,i,s,f[55]={0},k,j,c=0;
	scanf("%d%d",&n,&v);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&k);
		for(j=1;j<=k;j++){
		scanf("%d",&s);
		if(s<v)
		f[i]++;
		}
	}
	for(i=1;i<=n;i++)
	if(f[i]>0)
	c++;
	printf("%d\n",c);
	for(i=1;i<=n;i++)
	if(f[i]>0)
	printf("%d ",i);
return 0;
}
