#include<stdio.h>
int n,m,i,j,mn,t,z,u,d[100005];
int main(){

	scanf("%d %d",&n,&m);

	for(i = 1; i <=n ; i++){
		mn=1000000;
		scanf("%d",&t);
		for(j=1 ; j<=t ; j++){
			scanf("%d",&z);
			if(z < mn)
				mn=z;
		}
		if(mn < m)
			d[++u] = i;
	}
	printf("%d\n",u);
	for(i=1 ; i<=u ; i++)
		printf("%d ",d[i]);
	return 0;
}
