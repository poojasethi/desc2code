# include <fstream>
# include <iostream>
# define DIM 53
using namespace std;

int n, m, v[DIM];

int main()
{
	cin>>n>>m;
	int k;
	for(int i=1;i<=n;++i) {
		cin>>k;
		int p, t=0;
		for(int j=1;j<=k;++j) {
			cin>>p;
			if (p<m && !t)v[++v[0]]=i, t=1;
		}
	}
	
	cout<<v[0]<<"\n";
	for(int i=1;i<=v[0];++i)
		cout<<v[i]<<" ";
		
	return 0;
}
