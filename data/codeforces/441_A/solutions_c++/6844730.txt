#include<iostream>
using namespace std;

int main(){
int n,v,k[51],i,j,q[51][51],cnt=0,out[51];
cin>>n>>v;
for(i=0;i<n;i++){
cin>>k[i];
for(j=0;j<k[i];j++)
cin>>q[i][j];

for(j=0;j<k[i];j++){
if(q[i][j]<v){
out[cnt]=i;
cnt++;
break;
}
}
}
cout<<cnt<<endl;
for(i=0;i<cnt;i++)
cout<<out[i]+1<<" ";
cout<<endl;
return 0;
}
