#include<iostream>

using namespace std;

int a[55];

int main(){
	int n,mon,t1,t2,ii=0;
	bool b=false;
	cin>>n>>mon;
	for(int i=1;i<=n;i++){
		cin>>t1;
		while(t1--){
			cin>>t2;
			if(t2<mon) b=true;
		}
		if(b) a[ii++]=i;
		b=false;
	}
	cout<<ii<<endl;
	for(int i=0;i<ii;i++) cout<<a[i]<<" ";
	cout<<endl;
}
