#include<bits/stdc++.h>
using namespace std;
 int n,v,p,a[55],k,g,i,t=0;
int main()
{
cin>>n>>v;
for(i=1;i<=n;i++)
{
    cin>>k;
    g=0;
    for(int j=1;j<=k;j++)
    {
        cin>>p;
        if(v>p&&g==0)
        {
           a[t++]=i;
           g=1;
        }
    }
}
cout<<t<<"\n";
for(i=0;i<t;i++)
    cout<<a[i]<<" ";
}
