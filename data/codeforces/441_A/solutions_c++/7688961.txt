#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,v,b[52],x=0;
	cin>>n>>v;
	int a[52][52];
	int k;
	for(int i=0;i<n;i++)
	{
		cin>>k;
		for(int j=0;j<k;j++)
		{
			cin>>a[i][j];
		}
		sort(a[i],a[i]+k);
		if(a[i][0]<v)
			b[x++]=i+1;
	}
	cout<<x<<endl;
	for(int y=0;y<x;y++)
		cout<<b[y]<<" ";
	return 0;
}
