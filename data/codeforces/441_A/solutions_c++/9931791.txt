#include <bits/stdc++.h>
using namespace std;

int main(){
	int n,v,ni,x;
	cin>>n>>v;
	map<int,int> vx;
	map<int,int>::iterator it;
	for(int i=1;i<=n;i++){
		cin>>ni;
		while(ni--){
			cin>>x;
			if(x<v)
				vx[i];
		}
	}
	cout<<vx.size()<<endl;
	for(it=vx.begin();it!=vx.end();++it)
		cout<<(it->first)<<" ";
	cout<<endl;
	return 0;
}
