#include <cstdio>

bool mozna[100];

int main()
{
	int n,k,d,a;
	scanf("%d%d",&n,&d);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&k);
		while( k-- )
		{
			scanf("%d",&a);
			if ( a < d ) mozna[i] = 1;
		}
	}
	int ile = 0;
	for (int i=1;i<=n;i++) if (mozna[i]) ile++;
	printf("%d\n",ile);
	for (int i=1;i<=n;i++) if (mozna[i]) printf("%d ",i);
}
