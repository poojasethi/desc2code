#include <iostream>
#include<vector>
using namespace std;
int main() {
int i,x,n,v;
vector<int>d;
cin>>n>>v;
for(i=1;i<=n;i++)
{
int m,j,ans=0;
cin>>m;
for(j=1;j<=m;j++)
{
cin>>x;
if(x<v)ans++;
}
if(ans)d.push_back(i);
}
cout<<d.size()<<endl;
for(i=0;i<d.size();i++)
cout<<d[i]<<" ";
cout<<endl;
    return 0;
}
