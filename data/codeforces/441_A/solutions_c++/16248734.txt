#include <iostream>
using namespace std;
int main()
{
	long long unsigned s[100];
	long long unsigned n,i,j,k,v,ls=0,in,m;
	cin>>n>>v;
	for(i=0;i<n;i++)
	{
		cin>>k;
		in=0;
		for(j=0;j<k;j++)
		{
			cin>>m;
			if(in==0) 
			{
				if(v>m) {s[ls++]=i+1; in=1;}
			}
		}
	}
	cout<<ls<<endl;
	for(i=0;i<ls;i++) cout<<s[i]<<' ';
}
