#include<iostream>
using namespace std;
int n, m, v, i, j, ans = 0, a[60], x;
int main()
{
	cin>>n>>v;
	for(i = 1; i <= n; i ++)
	{
		cin>>m;
		for(j = 1; j <= m; j ++)
		{
			cin>>x;
			if(x < v && a[ans] != i)a[++ ans] = i;
		}
	}
	cout<<ans<<endl;
	for(i = 1; i <= ans; i ++)
		cout<<a[i]<<" ";
	return 0;
}
