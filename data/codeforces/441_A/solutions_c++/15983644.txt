#include <iostream>
using namespace std;

int arr[100];

int main(){
	long long n, v, i, N, j=0, f, t, k;
	cin >> n >> v;
	for(k=1;k<=n;k++){
		cin >> N;
		f=0;
		for(i=0;i<N;i++){
			cin >> t;
			if(v>t) f=1;
		}
		if(f){
			arr[j] = k;
			j++;
		}
	}
	cout << j << '\n';
	for(i=0;i<j;i++){
		cout << arr[i] << ' ';
	}
	cout << '\n';
	return 0;
}
