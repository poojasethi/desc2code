#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,v,i,j,k;
    cin>>n>>v;

    vector<int>vec;
    for(i=0;i<n;i++){
        cin>>k;
        int l,flag=1;
        for(j=0;j<k;j++){cin>>l;if(flag&&l<v){vec.push_back(i+1);flag=0;}}
    }
    cout<<vec.size();
    for(i=0;i<vec.size();i++)cout<<" "<<vec[i];
}
