#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,i,j;
	cin>>n>>k;
	set<int>s;
	for(i=0;i<n;i++)
	{
		int m;
		cin>>m;
		for(j=0;j<m;j++)
		{
			int x;
			cin>>x;
			if(x<k)
				s.insert(i+1);
		}
	}
	cout<<s.size()<<endl;	
	std::set<int>::iterator it=s.begin();
	while(it!=s.end())
	{
		cout<<*it<<' ';
		it++;
	}
	return 0;
}
