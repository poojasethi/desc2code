d = map(int, raw_input().split())
n, m = d[0], d[1]
x=[]
y=[]
for i in range(n):
  s = raw_input()
  if s.count('*') == 1:
     x.append(s.index('*')+1)
     y.append(i+1)
  elif s.count('*') == 2:
     t = s.index('*') + 1    
     x.append(t)
     x.append(s.index('*', t)+1)       
     y.append(i+1)
     y.append(i+1)
  if len(x) == 3: break
print filter(lambda t: y.count(t)==1, y)[0], filter(lambda t: x.count(t)==1, x)[0]
