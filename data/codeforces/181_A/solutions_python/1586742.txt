R, C = map(int, raw_input().split())
mapa = [ raw_input() for i in xrange(R) ]

xmin = ymin = 10000
xmax = ymax = -10000

for i in xrange(R):
	for j in xrange(C):
		if mapa[i][j]=='*':
			xmin = min(xmin, i)
			ymin = min(ymin, j)
			xmax = max(xmax, i)
			ymax = max(ymax, j)

for a in (xmin,xmax):
	for b in (ymin,ymax):
		if mapa[a][b]!='*':
			print a+1, b+1
