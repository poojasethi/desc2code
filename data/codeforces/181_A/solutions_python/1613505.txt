
n, m = map(int, raw_input().split())
u = []
for i in xrange(n):
    s = raw_input()
    for j in xrange(m):
        if s[j] == '*':
            u.append([i,j])
for i in xrange(3):
    for j in xrange(3):
        if i == j: continue
        pp = [u[i][0], u[j][1]]
        if pp not in u:
            print str(pp[0]+1) + " " + str(pp[1]+1)
            exit
