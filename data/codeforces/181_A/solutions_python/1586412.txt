n,m=map(int,raw_input().split())
A=[list(raw_input()) for i in range(n) ]
B=[]
for x in range(n) :
	for y in range(m):
		if A[x][y]=='*':
			B.append( (x,y) )
B.sort()
X=[]
Y=[]
for x in B:
	X.append(x[0])
	Y.append(x[1])
for x in X:
	if X.count(x) == 1:
		print x+1,
for x in Y:
	if Y.count(x) == 1:
		print x+1
