from collections import defaultdict
n, m = [int(x) for x in raw_input().split() ]
xs = defaultdict(int)
ys = defaultdict(int)
for i in range(n):
    line = raw_input()
    for j in range(m):
        if line[j] == '*':
            xs[j] += 1
            ys[i] += 1

for y in ys:
    if ys[y] == 1:
        print y + 1,

for x in xs:
    if xs[x] == 1:
        print x + 1
