n, m = map(int, raw_input().split())
dr = {}
dc = {}
for r in xrange(n):
  s = raw_input()
  sz = len(s)
  for c in xrange(sz):
    if s[c] == '*':
      dr[r] = dr.get(r, 0) + 1
      dc[c] = dc.get(c, 0) + 1
for k in dr:
  if dr[k] == 1:
    print str(k+1),
    break
for k in dc:
  if dc[k] == 1:
    print str(k+1)
    break
