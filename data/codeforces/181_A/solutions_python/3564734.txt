import sys

n, m = map(int, sys.stdin.readline().split())
grid = [sys.stdin.readline().strip() for ni in xrange(n)]

pts = []
for i in xrange(n):
    for j in xrange(m):
        if grid[i][j] == '*':
            pts.append((i + 1, j + 1))

x, y = zip(*pts)

def unique(x):
    x = list(x)
    x.sort()
    return x[0] + x[2] - x[1]

print '%d %d' % (unique(x), unique(y))
