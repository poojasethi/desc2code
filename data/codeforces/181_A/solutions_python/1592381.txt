g = [""]*200
arr = raw_input().split();
h, w = int(arr[0]), int(arr[1])
v = []
for i in xrange(h):
    g[i] = raw_input()
    for j in xrange(w):
        if(g[i][j] == '*'):v.append((i, j))

if(v[0][0] == v[1][0]):
        r = v[2][0];
elif(v[0][0] == v[2][0]):
        r = v[1][0];
else:
        r = v[0][0];

if(v[0][1] == v[1][1]):
        c = v[2][1];
elif(v[0][1] == v[2][1]):
        c = v[1][1];
else:
        c = v[0][1];
        
print r+1,c+1
