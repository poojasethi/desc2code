n, m = map(int, raw_input().split())
A = []
for i in range(n):
    A.append(raw_input())
x = []
y = []
for i in range(n):
    for j in range(m):
        if A[i][j] == '*':
            x.append(i)
            y.append(j)
x = set(x)
y = set(y)


for i in range(n):
    for j in range(m):
        if A[i][j] == '.':
            if len(x | {i}) == 2 and len(y | {j}) == 2:
                print str(i + 1) + " " + str(j + 1)
                exit()
