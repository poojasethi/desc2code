import sys
import math

rows, cols = map(int, sys.stdin.readline().split())
corners = []
corners_rows = []
corners_cols = []
for r in xrange(rows):
  line = sys.stdin.readline().strip()
  for c in xrange(len(line)):
    if line[c] == '*':
      corners.append((r,c))
      corners_rows.append(r)
      corners_cols.append(c)
corners = set(corners)
corners_rows = set(corners_rows)
corners_cols = set(corners_cols)
for r in corners_rows:
  for c in corners_cols:
    if not (r, c) in corners:
      print r+1, c+1
