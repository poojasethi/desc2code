n, m = map(int, raw_input().split())
p = []
for i in xrange(n):
    s = raw_input()
    for j in xrange(m):
        if s[j] == '*':
            p.append([i,j])
for i in xrange(3):
    for j in xrange(3):
        if i == j: continue
        pp = [p[i][0], p[j][1]]
        if pp not in p:
            print str(pp[0]+1) + " " + str(pp[1]+1)
            exit
