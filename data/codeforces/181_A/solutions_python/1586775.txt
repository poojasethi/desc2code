n, m = map(int, raw_input().split())

corners = set()
x = set()
y = set()

for i in xrange(1, n+1):
    row = raw_input()
    for j in xrange(1, m+1):
        if row[j-1] == '*':
            corners.add((i,j))
            x.add(i)
            y.add(j)

for i in x:
    for j in y:
        if (i,j) not in corners:
            print i, j
            break
