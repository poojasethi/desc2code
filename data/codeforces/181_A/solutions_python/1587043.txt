n,m=map(int,raw_input().split())
xs,ys=list(),list()
for i in range(n):
    s=raw_input()
    for j in range(m):
        if s[j]=='*':
            xs.append(i)
            ys.append(j)
xs,ys=sorted(xs),sorted(ys)
ansx,ansy=-1,-1
if xs[0]==xs[1]:
    ansx=xs[2]
else:
    ansx=xs[0]
if ys[0]==ys[1]:
    ansy=ys[2]
else:
    ansy=ys[0]
print ansx+1,ansy+1
