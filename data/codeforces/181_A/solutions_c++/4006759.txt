#include <iostream>
using namespace std;

char tbl[101][101];
int main()
{
	int n , m , I , J , row , col;
	cin>>n>>m;

	for(int i=1 ; i<=n ; i++)
	{
		row=0;
		for(int j=1 ; j<=m ; j++)
		{
			cin>>tbl[i][j];
			if(tbl[i][j]=='*')
				row++;
		}
		if(row==1) I=i;
	}

	for(int j=1 ; j<=m ; j++)
	{
		col=0;
		for(int i=1 ; i<=n ; i++)
			if(tbl[i][j]=='*')
				col++;
		if(col==1) J=j;
	}

	cout<<I<<" "<<J<<endl;


return 0;
}
