#include<iostream>
#include<algorithm>
#include<math.h>
#include<stdio.h>
#include<cstring>
using namespace std;
int main()
{ int n,m,i,j,r[105],c[105],k=0;
char a[105][105];
cin>>n>>m;
for(i=0;i<n;i++)
for(j=0;j<m;j++)
{cin>>a[i][j];
if(a[i][j]=='*'){r[k]=i+1;c[k++]=j+1;};
}
for(i=0;i<3;i++)
if(count(r,r+k,r[i])==1)cout<<r[i]<<" ";
for(i=0;i<3;i++)
if(count(c,c+k,c[i])==1)cout<<c[i];
return 0;
}
