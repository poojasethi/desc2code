#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;
int main() {
	int n, m;
	scanf("%d%d", &n, &m);
	int x = 0, y = 0;
	for (int i = 1; i <= n; i++) {
		char str[105];
		scanf("%s", str);
		for (int j = 1; j <= m; j++) {
			if (str[j - 1] == '*') {
				x ^= i, y ^= j;
			}
		}
	}
	printf("%d %d\n", x, y);
	return 0;
}
