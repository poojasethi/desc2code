#include <bits/stdc++.h>

using namespace std;

int arri[101],arrj[101];

int main (){
    int n,m,a,b;
    char x;
    cin>>n>>m;
    for (int i=0; i<n; i++)
        for (int j=0; j<m; j++){
            cin>>x;
            if (x=='*'){
                arri[i]++;
                arrj[j]++;
            }
        }
    for (int i=0; i<101; i++){
        if (arri[i]==1)
            a=i+1;
        if (arrj[i]==1)
            b=i+1;
    }
    cout<<a<<" "<<b;
}
