#include <stdio.h>
#include <cstdlib>
#include <algorithm>
#include <iostream>
using namespace std;

int n,m,x,y;
string s[1000];

main()
{
 cin>>n>>m;
 for ( int i=0;i<n;i++ )
  cin>>s[i];
 for ( int i=0;i<n;i++ )
  for ( int j=0;j<m;j++ )
   if ( s[i][j]=='*' )
    {
     x^=(i+1);
     y^=(j+1);
    }
 cout<<x<<" "<<y<<endl;
}
