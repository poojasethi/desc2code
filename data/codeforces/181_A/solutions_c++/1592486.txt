#include <cstdio>
 
char str[110];
int main() {
        int n, m;
        scanf("%d%d", &n, &m);
        int x = 0, y = 0;
        for (int i = 0; i < n; i++) {
                scanf("%s", str);
                for (int j = 0; j < m; j++) if (str[j] == '*') x ^= i, y ^= j;
        }
        printf("%d %d\n", x + 1, y + 1);
}
