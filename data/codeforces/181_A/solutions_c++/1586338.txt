#include <cstdio>

char in[200][200];
int lin[200], col[200];

int main()
{
	int n, m;
	scanf("%d%d", &n, &m);
	
	for (int i = 0; i < n; ++i)
	{
		for (int j = 0; j < m; ++j)
		{
			scanf(" %c", &in[i][j]);
			if (in[i][j] == '*')
				lin[i]++, col[j]++;
		}
	}
	
	for (int i = 0; i < n; ++i)
		if (lin[i] == 1) {printf("%d ", i+1);break;}
	for (int i = 0; i < m; ++i)
		if (col[i] == 1) {printf("%d\n", i+1);break;}
	
	
	return 0;
}
