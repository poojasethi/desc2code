#include <iostream>
#include <cstdio>
#include <string>

using namespace std;
string s[109];
int x[109], y[109], m, n, i, j;
int main () {
	cin >> n >> m;
	for (i = 0; i < n; i++) {
		cin >> s[i];
		for (j = 0; j < m; j++) {
			if (s[i][j] == '*') {
				x[i]++;
				y[j]++;
			}
		}
	}
	for (i = 0; i < n; i++)
		if (x[i] == 1)
			cout << i + 1 << " ";
	for (j = 0; j < m; j++)
		if (y[j] == 1)
			cout << j + 1;
}
