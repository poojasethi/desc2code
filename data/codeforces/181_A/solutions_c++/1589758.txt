#include<iostream>
using namespace std;

int main()
{
    int r,c;
    cin>>r>>c;
    char arr[r][c];
    for(int i=0;i<r;i++)
    {
        cin>>arr[i];
    }

    int resX = 0, resY = 0;
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(arr[i][j] == '*')
            {
                resX ^= i+1;
                resY ^= j+1;
            }
        }
    }

    cout<<resX<<" "<<resY;
    return 0;
}
