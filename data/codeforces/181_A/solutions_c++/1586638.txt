#include <cstdio>

char city[102][102];

int main(){
	int n,m;
	while(scanf("%d%d",&n,&m) != EOF){
		int x = 0,y = 0;
		for(int i=0;i<n;++i){
			scanf("%s",city[i]);
			for(int j=0;j<m;++j){
				if(city[i][j] == '*'){
					x ^= (i+1);
					y ^= (j+1);
				}
			}
		}
		printf("%d %d\n",x,y);
	}
	return 0;
}
