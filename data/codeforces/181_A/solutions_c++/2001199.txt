#include <iostream>
using namespace std;
int main ( ) {
        int n, m, i, j, x = 0, y = 0;
        char c;
        cin >> n >> m;
        for ( i = 1; i <= n; ++i )
                for ( j = 1; j <= m; ++j ) {
                        cin >> c;
                        if ( c == '*' ) {
                                x ^= i;
                                y ^= j;
                        }
                }
        cout << x << ' ' << y << endl;
}
