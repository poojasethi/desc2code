#include<stdio.h>
#include<vector>

using namespace std;
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	int i,j;
	char c;
	vector<int> a,b;
	scanf("%c",&c);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			scanf("%c",&c);
			if(c=='*')
			{
				a.push_back(i);
				b.push_back(j);
			}
		}
		scanf("%c",&c);
	}
	int x=0;
	int y=0;
	for(i=0;i<3;i++)
	{
		x=x^a[i];
		y=y^b[i];
	}
	printf("%d %d\n",x+1,y+1);
	return 0;
}
