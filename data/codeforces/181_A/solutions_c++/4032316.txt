#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
	int i,j,n,m,k=0,x[3],y[3];
	char c[105];
	cin>>n>>m;
	for(i=0;i<n;i++){
		cin>>c;
		for(j=0;j<m;j++)
			if(c[j]=='*') { x[k]=i+1; y[k]=j+1; k++; }
	}
	for(i=0;i<3;i++){
		for(j=0;j<3;j++){
			if(i==j) continue;
			for(k=0;k<3;k++){
				if(k==i || j==k) continue;
				if(x[i]==x[j] && y[i]==y[k]){
					cout<<x[k]<<" "<<y[j]<<endl;
					return 0;
				}
			}
		}
	}
	return 0;
}
