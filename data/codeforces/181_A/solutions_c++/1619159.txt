
#include <iostream>
using namespace std;
int m, n;

int main()
{

	cin >> n >> m;

	char str[n][m];	
	int a=0, b=0;

	for (int i=0; i < n; ++i)
	{
		cin >>  str[i]; 
		for (int j=0; j < m; ++j )
		{
			if (str[i][j] == '*' )
			{
				a ^= i;
				b ^= j;
			}
		}
	}
	cout << a+1 << " " << b+1 << endl;

	
}
