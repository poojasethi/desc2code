#include<stdio.h>
int main()
{
int m,n,x[4],y[4],a=1,count,row,col;
char ch;
scanf("%d%d",&n,&m);

for(int i=1;i<=n;i++)
{
scanf("%c",&ch);
count=0;
for(int j=1;j<=m;j++)
{
scanf("%c",&ch);
if(ch=='*')
{x[a]=i;y[a]=j;a++;count++;}
}

if(count==1)
{row=i;col=y[a-1];}
}


for(int p=1;p<=3;p++)
{
if(x[p]!=row&&y[p]!=col)
{col=y[p];break;}
}
printf("%d %d\n",row,col);
}
