#include <iostream>
#include <cstdio>

using namespace std;
int Diff(int a,int b,int c)
{
    if (a==b) return c;
    if (a==c) return b;
    return a;
}
int main()
{
    int i,j,n,m,x[3],y[3],now=0;
    char c;
    for (scanf("%d%d",&n,&m),i=1;i<=n;i++)
        for (j=1;j<=m;j++)
        {
            cin>>c;
            if (c=='*') x[now]=i,y[now++]=j;
        }
    printf("%d %d\n",Diff(x[0],x[1],x[2]),Diff(y[0],y[1],y[2]));
    return 0;
}
