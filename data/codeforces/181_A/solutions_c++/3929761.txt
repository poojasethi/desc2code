#include <iostream>
using namespace std;
int xx,yy,n,m,i,j,k,x[3],y[3];
char c;
int main () {
	cin>>n>>m;
	k=-1;
	for (i=0; i<n; i++){
		cin.get();
		for (j=0; j<m; j++){
			cin.get(c);
			if (c=='*') {
				k++;
				x[k]=i;
				y[k]=j;
			}
		}
	}
	if (x[0]==x[1]) xx=x[2];
	if (x[0]==x[2]) xx=x[1];
	if (x[2]==x[1]) xx=x[0];
	if (y[0]==y[1]) yy=y[2];
	if (y[0]==y[2]) yy=y[1];
	if (y[2]==y[1]) yy=y[0];
	cout<<xx+1<<" "<<yy+1;
}
