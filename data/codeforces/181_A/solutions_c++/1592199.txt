#include<cstdio>
int main(){
    int n,m;
    int x,y;
    int i,j;
    char grid[100][101];
    while(~scanf("%d%d",&n,&m)){
        x=y=0;
        for(i=0;i<n;++i){
            scanf("%s",grid[i]);
            for(j=0;j<m;++j){
                if(grid[i][j]=='*'){
                    x^=i;
                    y^=j;
                }
            }
        }
        printf("%d %d\n",x+1,y+1);
    }
    return 0;
}
