#include<iostream>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	int i,j,a[101]={0},b[101]={0};
	char tmp;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			cin>>tmp;
			if(tmp == '*')a[i]++,b[j]++;
		}
	}
	for(i=1;i<=n;i++)if(a[i] == 1 ){cout<<i<<" ";break;}
	for(i=1;i<=m;i++)if(b[i] == 1 ){cout<<i<<endl;break;}

	return 0;
}
