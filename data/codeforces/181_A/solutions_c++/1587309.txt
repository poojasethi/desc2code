#include<cstdio>
int main()
{
	int m,n;int l=0;int x[4],y[4];
	char c[101];
	scanf("%d %d",&m,&n);
	for(int i=0;i<m;i++){
		scanf("%s",&c);
		for(int j=0;j<n;j++)
		{
			if(c[j]=='*')
			{
				x[l]=i+1;y[l]=j+1;
				l++;
			}
		}
	}
	if(x[0]==x[1]){x[3]=x[2];}
	else{if(x[0]==x[2]){x[3]=x[1];}
		else{x[3]=x[0];}}
	if(y[0]==y[1]){y[3]=y[2];}
	else{if(y[0]==y[2]){y[3]=y[1];}
		else{y[3]=y[0];}}
		printf("%d %d",x[3],y[3]);
}
