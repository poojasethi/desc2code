#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
#define REP(i,n) for(int(i)=0;(i)<(int)(n);(i)++)
const int MAX=110;
char a[MAX][MAX];
int n,m,x[4],y[4];
int main()
{
	scanf("%d%d\n",&n,&m);
	REP(i,n) gets(a[i]);
	int k=0;
	REP(i,n) for(int j=0; a[i][j]; j++) {
		if (a[i][j]=='*') {
			x[k]=i;
			y[k++]=j;
		}
	}
	x[3]=(x[0]==x[1]?x[2]:(x[1]==x[2]?x[0]:x[1]));
	y[3]=(y[0]==y[1]?y[2]:(y[1]==y[2]?y[0]:y[1]));
	printf("%d %d\n",x[3]+1,y[3]+1);
	return 0;
}
