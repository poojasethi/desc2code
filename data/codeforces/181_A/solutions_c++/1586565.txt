#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
int n,m;
char a[111][111];

int main()
{
    cin>>n>>m;
    for(int i=0;i<n;i++) scanf("%s",a[i]);
    int x=0,y=0;
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++) if(a[i][j]=='*') x^=i,y^=j;
    cout<<x+1<<" "<<y+1<<endl;;
}
