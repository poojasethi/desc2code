#include<bits/stdc++.h>
using namespace std;
set<int> row,col;
int main()
{
int i,j;
int n,m;
scanf("%d%d",&n,&m);
char a[505];
for(i=1;i<=n;i++)
{
scanf("%s",&a);
for(j=1;j<=m;j++)
if(a[j-1]=='*')
{
if(row.find(i)==row.end())row.insert(i);
else row.erase(i);
if(col.find(j)==col.end())col.insert(j);
else col.erase(j);
}
}
//printf("%d %d\n",row.size(),col.size());
printf("%d %d\n",*(row.begin()),*(col.begin()));

return 0;
}
