#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char a[105][105];
int main(void) {
	int x=0, y=0;
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++) {
		scanf("%s", a[i]);
		for(int j=0;a[i][j];j++)
			if(a[i][j]=='*') {
				x^=i;
				y^=j;
			}
	}
	printf("%d %d\n", x+1, y+1);
	return 0;
}
