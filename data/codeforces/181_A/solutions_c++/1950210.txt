#include <iostream>
using namespace std;

char map[100][100];


int main() {
	int n, m, x, y;
	cin >> n >> m;
	x = 0;
	y = 0;
	
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			cin >> map[i][j];
		}	
	}	
	
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			if (map[i][j] == '*') {
			x ^= i;
			y ^= j;
			}
		}
	}
	cout << x + 1 << " " << y + 1 << endl;
}
