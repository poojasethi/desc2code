#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
//freopen("i.txt","r",stdin);
int x[4],y[4];
int n,m; cin>>n>>m;
string s;
int i0=1;
for (int i=1;i<=n;i++)
 {cin>>s;
  for (int j=0;j<m;j++)
  if (s[j]=='*')
   {
    x[i0]=i; y[i0]=j+1; i0++;
   }
 }
int a,b;
if (x[1]==x[2]) a=x[3];
if (x[3]==x[2]) a=x[1];
if (x[1]==x[3]) a=x[2];

if (y[1]==y[2]) b=y[3];
if (y[3]==y[2]) b=y[1];
if (y[1]==y[3]) b=y[2];
cout<<a<<' '<<b;

return 0;
}
