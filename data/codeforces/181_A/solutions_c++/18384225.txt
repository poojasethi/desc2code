#include <iostream>
#include <string>
#include <cmath>
using namespace std;
int n,m,i,x[5],y[5],o=0;
char c;
int main(){
	cin>>n>>m;
    while(cin>>c){
		if(c=='*'){
			x[o]=i/m+1;
			y[o++]=i%m+1;
		}
		i++;
	}
	if(x[2]==x[1])cout<<x[0];	
	if(x[0]==x[2])cout<<x[1];	
	if(x[0]==x[1])cout<<x[2];
	cout<<" ";
	if(y[2]==y[1])cout<<y[0];	
	if(y[0]==y[2])cout<<y[1];	
	if(y[0]==y[1])cout<<y[2];	
    return 0;
}
