#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
int n,m;
cin>>n>>m;
string s[n];
for(int i=0;i<n;i++)
cin>>s[i];
int k=0, h[3]={},g[3]={};
 for(int i=0;i<n;i++)
{
for(int j=0;j<m;j++)
{
if(s[i][j]=='*')
{
h[k]=i;
g[k]=j;k++;
}
}
}
if(g[0]==g[2])
{
if(h[1]==h[2])
cout<<h[0]+1<<' '<<g[1]+1;
else
cout<<h[2]+1<<' '<<g[1]+1;
}
else
{
if(h[1]==h[2])
cout<<h[0]+1<<' '<<g[2]+1;
else
cout<<h[2]+1<<' '<<g[0]+1;
}
return 0;
}
