#include <cstdio>
#include <cstring>

const int MAXN = 1010;

int n, m;
char s[MAXN];
int x, y;

int main()
{
   scanf("%d %d\n", &n, &m);
   x = y = 0;
   int i, j;
   for (i = 1; i <= n; ++i)
   {
      gets(s);
      for (j = 0; j < m; ++j)
         if (s[j] == '*')
         {
            x ^= i;
            y ^= (j + 1);
         }
   }
   printf("%d %d\n", x, y);
   return 0;
}
