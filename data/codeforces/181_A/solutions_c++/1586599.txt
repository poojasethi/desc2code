#include<cstdio>
#include<iostream>
using namespace std;

char ch;
int x[3],y[3],rx,ry,px,py;
int N,M;

int main()
{
	cin>>N>>M;
	for(int i=1;i<=N;i++) for(int j=1;j<=M;j++)
	{
		while((ch=getchar())!='.'&&ch!='*');
		if(ch=='*') x[px++]=j,y[py++]=i;
	}
	if(x[0]==x[1]) rx=x[2];
	else if(x[0]==x[2]) rx=x[1];
	else rx=x[0];
	if(y[0]==y[1]) ry=y[2];
	else if(y[0]==y[2]) ry=y[1];
	else ry=y[0];
	cout<<ry<<" "<<rx<<endl;
	return 0;
}
