#include <stdio.h>
#include <string.h>
#include <vector>

#define pb push_back

using namespace std;

int n;
vector<int> nei[100010];
int ct[100010][30];
int let[100010];

void find(int cur, int prev)
{
  int sz = nei[cur].size(), i, j;
  for (i = 0; i < sz; i++)
  {
    int nx = nei[cur][i];
    if (nx == prev)
      continue;
    find(nx, cur);
    for (j = 0; j < 26; j++)
      ct[cur][j] += ct[nx][j];
  }

  int high = 26;
  for (i = 0; i < 26; i++)
    if (ct[cur][i] >= 2)
    {
      high = i;
      break;
    }

  i = high - 1;
  while (ct[cur][i])
    i--;
  let[cur] = i;
  ct[cur][i] = 1;
  for (i++; i < 26; i++)
    ct[cur][i] = 0;
}

int main()
{
  int i;
  memset(ct, 0, sizeof ct);
  scanf("%d", &n);
  for (i = 0; i < n - 1; i++)
  {
    int a, b;
    scanf("%d %d", &a, &b);
    a--, b--;
    nei[a].pb(b);
    nei[b].pb(a);
  }

  find(0, -1);

  for (i = 0; i < n; i++)
    printf("%c%c", 'A' + let[i], (i == n - 1) ? '\n' : ' ');

  return 0;
}
