#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 400005
#define mk(x,y) make_pair(x,y)
#define pii pair<int,int>
#define rep(i,n) for(int i=1;i<=n;i++)
#define inf (1<<29)
using namespace std;
int son[N],ed[N],next[N],vis[N],cnt,ans[N],n,num,size[N];
pii root;
void link(){
	int x,y;
	scanf("%d%d",&x,&y);
	next[++cnt]=son[x]; son[x]=cnt; ed[cnt]=y;
	next[++cnt]=son[y]; son[y]=cnt; ed[cnt]=x;
}
void dfs1(int t){
	vis[t]=1; size[t]=1;
	for(int i=son[t];i;i=next[i])
	if(!vis[ed[i]]) dfs1(ed[i]),size[t]+=size[ed[i]];
	vis[t]=0;
}
void dfs2(int t){
	vis[t]=1;
	int p=num-size[t];
	for(int i=son[t];i;i=next[i])
	if(!vis[ed[i]]) dfs2(ed[i]),p=max(p,size[ed[i]]);
	root=min(root,mk(p,t));
	vis[t]=0;
}
void dfs(int t,int dis){
	dfs1(t); num=size[t]; root=mk(inf,0);
	dfs2(t);
	vis[root.second]=1; ans[root.second]=dis;
	for(int i=son[root.second];i;i=next[i])
	if(!vis[ed[i]]) dfs(ed[i],dis+1);
}
int main(){
	scanf("%d",&n);
	rep(i,n-1) link();
	dfs(1,0);
	rep(i,n) printf("%c ",'A'+ans[i]);
}
