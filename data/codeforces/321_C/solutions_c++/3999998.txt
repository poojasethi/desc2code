#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
#define N 210000
#define fore(i,a) for(int i=g[a];i;i=next[i])
#define foru(i,a,b) for(int i=a;i<=b;++i)
#define ford(i,a,b) for(int i=a;i>=b;--i)

using namespace std;
int g[N],next[N],map[N],ch[N],n,tot,f[N][51];
void make(int a,int b)
{
    next[++tot]=g[a]; g[a]=tot; map[tot]=b;
}
void dfs(int a,int b)
{
    fore(i,a)
    {
        if(map[i]==b) continue;
        dfs(map[i],a);
        foru(j,1,26) f[a][j]+=f[map[i]][j];
    }
    int k=26;
    ford(i,26,1) 
      if(f[a][i]>=2) k=i-1;
    while(k&&f[a][k]) --k;
    ch[a]=k; f[a][k]=1;
    foru(i,k+1,26) f[a][i]=0;
}
int main()
{
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    cin>>n;
    foru(i,1,n-1)
    {
        int a,b;
        cin>>a>>b;
        make(a,b); make(b,a);
    }
    dfs(1,0);
    foru(i,1,n) 
      if(!ch[i]){cout<<"Impossible!"<<endl; return 0;}
    foru(i,1,n) printf("%c%c",char(ch[i]-1+(int)'A'),i==n?'\n':' ');
    return 0;
}
