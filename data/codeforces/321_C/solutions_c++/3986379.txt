#include <cstdio>
#include <iostream>
#include <vector>

#define pb push_back

using namespace std;

const int MAXN=100010;

int N;
int cnt[MAXN];
int used[MAXN];

vector<int> g[MAXN];

int calc( int nd , int pre ){
	
	cnt[nd]=1;
	
	for( vector<int>::iterator it=g[nd].begin() ; it!=g[nd].end() ; it++ )
		if( *it!=pre && !used[*it] )
			cnt[nd]+=calc(*it,nd);
	
	return cnt[nd];
}

int best( int nd , int c , int pre ){
	
	for( vector<int>::iterator it=g[nd].begin() ; it!=g[nd].end() ; it++ )
		if( *it!=pre && !used[*it] && cnt[*it]>c/2 )
			return best(*it,c,nd);
	
	return nd;
}

void f( int nd , int cl ){
	
	calc(nd,0);
	
	int k=best(nd,cnt[nd],0);
	
	used[k]=cl;
	
	for( vector<int>::iterator it=g[k].begin() ; it!=g[k].end() ; it++ )
		if( !used[*it] )
			f(*it,cl+1);
}

int main(){
	
	scanf(" %d",&N);
	
	for( int a,b,i=1 ; i<N ; i++ ){
		scanf(" %d %d",&a,&b);
		g[a].pb(b);
		g[b].pb(a);
	}
	
	f(1,1);
	
	for( int i=1 ; i<=N ; i++ )
		printf("%c ",used[i]+'A'-1);
	return 0;
}
