#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=220000;
int i,j,n,x,y,head[N],next[N],e[N],len,ans[N],f[N],fa[N];
void dfs(int x){
	for(int i=head[x];i;i=next[i])if(e[i]!=fa[x]){
		fa[e[i]]=x;
		dfs(e[i]);
		f[x]|=f[e[i]];
	}
	int v;
	for(int i=25;i>=0;--i){
		int u=0;
		for(int j=head[x];j;j=next[j])if(e[j]!=fa[x])
			u+=(f[e[j]]>>i)&1;
		if(u>=2)break;
		if(u)continue;
		v=i;
	}
	f[x]|=1<<v;
	f[x]&=((1<<25)-1)^((1<<v)-1);
	ans[x]=v;
}
int main(){
	scanf("%d",&n);
	for(i=1;i<n;++i){
		scanf("%d%d",&x,&y);
		next[++len]=head[x]; head[x]=len; e[len]=y;
		next[++len]=head[y]; head[y]=len; e[len]=x;
	}
	dfs(1);
	for(i=1;i<=n;++i)printf("%c ",'Z'-ans[i]);
}
