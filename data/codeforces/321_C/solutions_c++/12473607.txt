#include <cstdio>
#include <vector>

using namespace std;

vector<int> node[100005];
int n;
char rk[100005];
bool v[100005];
int noderk[100005][30];

void dfs(int rt) {
	v[rt] = true;
	for(int i = 0;i < node[rt].size();i++) {
		if(!v[node[rt][i]]) {
			dfs(node[rt][i]);
			for(int j = 0;j < 26;j++){
				noderk[rt][j] += noderk[node[rt][i]][j];
			}
		}
	}
	char m = '#';
	int st = 25;
	for(int i = 25;i >= 0;i--) {
		if(noderk[rt][i] > 1)
			st = i - 1;
	for(int i = st;i >= 0;i--)
		if(!noderk[rt][i]) {
			m = i + 'A';
			break;
		}
	}
	//printf("root = %d, label = %c\n", rt, m);
	rk[rt] = m;
	noderk[rt][m - 'A'] = 1;
	for(int i = m - 'A' + 1;i < 26;i++)
		noderk[rt][i] = 0;
}

int main() {
	scanf("%d", &n);
	for(int i = 1;i < n;i++) {
		int a, b;
		scanf("%d%d", &a, &b);
		node[a].push_back(b);
		node[b].push_back(a);
	}
	dfs(1);
	for(int i = 1;i <= n;i++)
		printf("%c%c", rk[i], i == n?10: 32);
}
