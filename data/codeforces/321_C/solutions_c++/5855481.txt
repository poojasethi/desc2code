using namespace std;

const int maxn=1000*100+100;

vector<int>v[maxn];

int n,ans[maxn],a[maxn];

set	<int> s[maxn];

int dfs(int x,int par,int z){
	FOReach(it,s[x]){
		int next=*it;
		if(next!=par && a[next]>=a[z]/2)
			return dfs(next,x,z);					
	}
	return x;
}

void dfs1(int x,int par){
    a[x]=0;
	FOReach(it,s[x]){
		int next=*it;
		if(next!=par){
			dfs1(next,x);
			a[x]+=a[next];
		}
	}
	a[x]++;
}

void SOLVE(int t,int z){
	dfs1(z,0);
	int x=dfs(z,0,z);
	ans[x]=t;
	FOReach(it,s[x]){
		int i=*it;
		s[i].erase(x);
		SOLVE(t+1,i);
	}
}

int main(){
	cin>>n;
	for(int i=1;i<n;i++){
		int x,y;
		cin>>x>>y;
		s[x].insert(y);
		s[y].insert(x);
	}
	SOLVE(0,1);
	for(int i=1;i<=n;i++)
		cout<<char(ans[i]+'A')<<" ";
	cout<<endl;
	return 0;
}
