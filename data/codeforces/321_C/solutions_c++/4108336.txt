#include<iostream>
#include<cstdio>
#include<vector>
#include<cmath>
#define inf 1000000000
using namespace std;
vector < vector<int> > g;
char res[100003];
int coun[100003][33];
void dfs(int x,int pred)
{
    int i,j;
    for(i=0;i<g[x].size();i++)
    {
        if(g[x][i]!=pred)
        {
            dfs(g[x][i],x);
            for(j=0;j<26;j++)
                coun[x][j]+=coun[g[x][i]][j];
        }
    }
    i=0;
    while(i<25 && coun[x][i]<2)
        i++;
    while(coun[x][i])
        i--;
    coun[x][i]++;
    res[x]='A'+i;
    for(j=i+1;j<26;j++)
        coun[x][j]=0;
}
int main()
{
    int n;
    scanf("%d",&n);
    int x,y;
    g.resize(n);
    int i;
    for(i=0;i<n-1;i++)
    {
        scanf("%d %d",&x,&y);
        x--;
        y--;
        g[x].push_back(y);
        g[y].push_back(x);
    }
    dfs(0,0);
    printf("%c",res[0]);
    for(i=1;i<n;i++)
        printf(" %c",res[i]);
    printf("\n");
    return 0;
}
