#include <cstdio>
#include <vector>
#include <utility>
#include <algorithm>

using namespace std;

int n;

vector<int>graf[100005];

char wyn[100005];

pair<int,int> cent;

int m;

int DFS(int w,int o)
{
	if(wyn[w]) return 0;
	int p=0;
	int s=0;
	for(int i=0;i<graf[w].size();i++)
	{
		if(graf[w][i]==o) continue;
		int x=DFS(graf[w][i],w);
		p=max(p,x);
		s+=x;
	}
	p=max(p,m-1-s);
	cent=min(cent,make_pair(p,w));
	return s+1;
}

int roz(int w,int o)
{
	if(wyn[w]) return 0;
	int x=1;
	for(int i=0;i<graf[w].size();i++)
	{
		if(graf[w][i]==o) continue;
		x+=roz(graf[w][i],w);
	}
	return x;
}

void odpal(int w,char l)
{
	cent=make_pair(1e9,-1);
	m=roz(w,-1);
	DFS(w,-1);
	w=cent.second;
	wyn[w]=l;
	if(m==1) return;
	for(int i=0;i<graf[w].size();i++)
	{
		if(wyn[graf[w][i]]) continue;
		odpal(graf[w][i],l+1);
	}
}

int main()
{
	scanf("%d",&n);
	for(int i=1;i<n;i++)
	{
		int a,b;
		scanf("%d%d",&a,&b);
		graf[a].push_back(b);
		graf[b].push_back(a);
	}
	odpal(1,'A');
	for(int i=1;i<=n;i++) printf("%c ",wyn[i]); printf("\n");
	return 0;
}
