a = raw_input().split()

n = int(a[0])
m = int(a[1])

steps = [0]*n

solvable = True
for i in range(n):
    t = raw_input()
    G = t.find('G')
    S = t.find('S')
    step = S - G
    if step < 0:
        solvable = False
    steps[i] = step

ans = len(set(steps))
if solvable is False:
    ans = -1

print ans
