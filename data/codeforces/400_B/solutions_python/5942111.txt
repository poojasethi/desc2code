import sys

line = sys.stdin.readline()
n = int(line.split()[0])
m = int(line.split()[1])

ct = 0
moves=[]
while ct < n:
    line = sys.stdin.readline()
    ind1 = line.index("G")
    ind2 = line.index("S")
    if(ind2 < ind1):
        print "-1"
        exit(0)
    diff = ind2-ind1
    if diff not in moves:
        moves.append(diff)
    ct+=1

print len(moves)
