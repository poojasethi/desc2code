#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

n, m = map(int, raw_input().split())

grid = [raw_input() for i in range(n)]

def valid(s) : 
    if s.index('G') > s.index('S') : 
        return False
    return True

def getD(s) :
    return s.index('S') - s.index('G')

if not all(map(valid, grid)) : 
    print -1
else : 
    cnt = 0
    dists = map(getD, grid)
    for d in range(m) : 
        if d in dists : 
            cnt += 1
    print cnt
