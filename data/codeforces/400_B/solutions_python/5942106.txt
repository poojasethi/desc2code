#!/usr/bin/env python2


def main():
    n, m = map(int, raw_input().split())
    differences = set()
    for x in xrange(n):
        line = raw_input()
        g = line.index('G')
        s = line.find('S', g + 1)
        if -1 == s:
            print -1
            return
        differences.add(s - g)
    print len(differences)
        


if __name__ == '__main__':
    main()
