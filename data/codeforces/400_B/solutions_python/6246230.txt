import sys

def main():
    file=sys.stdin
    line=file.readline()
    values=line.split(' ')
    n,m=int(values[0]), int(values[1])
    pos=[False]*m
    count=0
    for i in range(n):
        line=file.readline()
        dif=line.index('S')-line.index('G')
        if dif<0:
            print -1
            return
        elif pos[dif]== False:
            pos[dif]=True
            count+=1
    print count

main()
