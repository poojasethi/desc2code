if __name__ == "__main__":
    size = [int(i) for i in raw_input().split(' ')]    
    steps = []
    for i in range(0, size[0]):
        input = raw_input()        
        countOfSteps = input.index('S') - input.index('G')
        if (countOfSteps < 0):
            print -1
            exit()
        else:
            if countOfSteps not in steps:
                steps.append(countOfSteps)
    print len(steps)
