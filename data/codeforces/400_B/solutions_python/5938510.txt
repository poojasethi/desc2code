n, m = map(int, raw_input().split())
field = []
for i in range(n):
    field.append(raw_input().strip())
dis = []
Impossible = False
for i in range(n):
    g, s = 0, 0
    for j in range(m):
        if field[i][j] == 'G':
            g = j
        elif field[i][j] == 'S':
            s = j
    if s < g:
        Impossible = True
        break
    else:
        dis.append(s - g)
dis = set(dis)
if Impossible:
    print -1
else:
    print len(dis)
