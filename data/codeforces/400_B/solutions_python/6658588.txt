def main():
    n,m = map(int, raw_input().split())

    board = []
    for i in xrange(n):
        t = raw_input()
        board.append(t)

    print solve(n,m,board)

def solve(n,m,board):
    
    dist = set()

    for row in board:
        Gi = row.index("G")
        Si = row.index("S")
        if Si < Gi:
            return -1
        dist.add(Si-Gi)

    return len(dist)
             





if __name__ == "__main__":
    main();
