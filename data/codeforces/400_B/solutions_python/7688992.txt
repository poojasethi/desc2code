#coding:utf-8

n, m = list(map(int, raw_input().split(' ')))
a = list(0 for i in range(n))
ans = set()
f = False
for i in range(n):
	s = raw_input()
	for j in range(m):
		if s[j] == 'G':
			G = j
		elif s[j] == 'S':
			S = j
	if S == -1:
		S = len(s)
	temp = S - G - 1
	if temp >= 0:
		ans.add(temp)
	else:
		f = True
		break
if f:
	print(-1)
else:
	print(len(ans))
