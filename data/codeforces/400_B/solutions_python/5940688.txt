def calcDist():
    s = raw_input()
    return s.index('S') - s.index('G')

def main():
    distDic = {}
    count = 0
    n,m = [int(c) for c in raw_input().split()]
    for _ in xrange(n):
        d = calcDist()
        if d < 1:
            return -1
        elif d not in distDic:
            count += 1
            distDic[d] = True
    return count
    
if __name__=="__main__":
    print main()
