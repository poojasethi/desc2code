from sys import stdin


def dist(row):
    return row.index('S') - row.index('G')


def main():
    n, m = map(int, stdin.readline().split())
    grid = [stdin.readline().strip() for _ in range(n)]
    distance = map(dist, grid)
    if any([d < 0 for d in distance]):
        print -1
    else:
        s = set(distance)
        print len(s)

if __name__ == '__main__':
    main()
