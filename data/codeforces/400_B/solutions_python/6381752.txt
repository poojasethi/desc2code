import sys
ri = lambda: raw_input().strip()

#n = int(ri())
n, m = map(int, ri().split())

res = set()
flag = True
for x in xrange(n):
    line = ri()

    start = 0
    end = 0
    for index in xrange(m):
        if line[index] == "G":
            start = index

        elif line[index] == "S":
            end = index


    if start < end:
        res.add(end-start)
    else:
        flag = False

    if not flag:
        break

if flag:
    print len(res)
else:
    print -1
