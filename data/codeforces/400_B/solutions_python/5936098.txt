def GetDwarfCandyIndex(x):
  ncandy, ndwarf = 0, 0
  for k in range(ncol):
    if board[x][k] == 'G': ndwarf = k
    if board[x][k] == 'S': ncandy = k
  return ndwarf, ncandy

def CheckFail():
  for i in range(nrow):
    ndwarf, ncandy = GetDwarfCandyIndex(i)
    if ndwarf > ncandy: return True

  return False

def GetCount(x):
  ndwarf, ncandy = GetDwarfCandyIndex(x)
  return ncandy-ndwarf

nrow, ncol = map(int, raw_input().split())
board = []
for i in range(nrow):
  board.append(raw_input())

if CheckFail(): print -1
else:
  jmlh = {}
  for i in range(nrow):
    jmlh[GetCount(i)] = 1
  print len(jmlh)
