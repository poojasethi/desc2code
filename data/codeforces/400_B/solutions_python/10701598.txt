n,m=map(int,raw_input().split())
grid=[]
for i in range(n):
    s=raw_input()
    grid.append(s)
flag=True
ans=[]
for s in grid:
    i1=s.index('G')
    i2=s.index('S')
    if i1>i2:
        flag=False
        break
    else:
        ans.append(i2-i1)
if flag==False:
    print -1
else:
    print len(set(ans))
