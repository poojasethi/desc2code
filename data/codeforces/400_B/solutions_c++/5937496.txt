#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,m,ans=0,g,s;
    char c;
    set<int> ss;
    cin >> n >> m;
    while(n--){
        for(int i=0;i<m;i++){
            cin >> c;
            if(c == 'G') g=i;
            if(c== 'S') s=i;
        }
        if(g>s) ans=-1;
        else ss.insert(s-g);
    }
    if(ans==-1) cout << -1 << endl;
    else cout << ss.size() << endl;
}
