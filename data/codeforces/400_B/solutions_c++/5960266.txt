#include <cstdio>

int N,M;
bool b[1005];
char s[1005];

int main()
{
    scanf("%d%d",&N,&M);
    for (int i=1,A,B; i<=N; i++)
    {
        scanf("%s",s+1);
        for (int j=1; j<=M; j++) if (s[j]=='G') A=j; else if (s[j]=='S') B=j;
        if (A>B) return puts("-1"),0; else b[B-A]=1;
    }
    int ans=0;
    for (int i=1; i<=M; i++) if (b[i]) ans++;
    printf("%d\n",ans);
}
