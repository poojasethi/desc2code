#include<iostream>
#include<stdio.h>
using namespace std;
string s;
int i,j,a,n,m,g1,s1,x[1002];
main()
{
  cin>>n>>m;
  for(i=1;i<=n;i++)
  {
   cin>>s;
   for(j=0;j<m;j++)
    {
     if(s[j]=='S') s1=j;
     if(s[j]=='G') g1=j;
    }
   if(s1<g1) {cout<<"-1"<<endl;  return 0;}
   x[s1-g1]=1;
  }
 for(i=1;i<=m;i++)
  {
   if(x[i]==1) a++;
  }
 cout<<a;
}
