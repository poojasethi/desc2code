#include<iostream>
#include<set>
using namespace std;
int main ()
{
	int k=0,n,m;
	set<int> s;
	char a;
	cin>>n>>m;
	for (int i=1;i<=n;i++)
	{
		int b,f;
		for (int j=1;j<=m;j++)
		{
		cin>>a;
		if (a=='G')
		b=j;
		if (a=='S')
		f=j;
		}
		if (f-b<0)
		{cout<<"-1";return 0;}
		else s.insert(f-b);
	}
	cout<<s.size();
}
