#include<bits/stdc++.h>
using namespace std;

int main() {
    int m, n; cin >> m >> n;
    set<int> dist;
    for(int i = 0; i < m; ++i) {
        string s; cin >> s;
        int x = s.find('G'), y = s.find('S');
        dist.insert(y - x);
    }
    if(*dist.begin() < 0) return cout << -1, 0;
    cout << dist.size();
    return 0;
}
