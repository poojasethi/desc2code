#include <iostream>
using namespace std;
int main()
{
	int n, m, ans, i, g, s;
	string GS;
	cin>>n>>m;
	int a[m];
	for (i=0; i<m; i++) a[i]=0;
	while (n--) {
		cin>>GS;
		for (g=9999,s=-1,i=0; i<m; i++) {
			if (GS[i]=='G') g=i;
			else if (GS[i]=='S') s=i;
		}
		if (s-g>0) a[s-g]++;
		else {
			cout<<-1<<endl;
			return 0;
		} 
	}
	for (ans=0,i=0; i<m; i++) if(a[i]) ans++;
	cout<<ans<<endl;
	return 0;
}
