#include <iostream>
#include <set>
using namespace std;
set <int> p;
int main()
{
    int n, m, k = 0, i;
    cin >> n >> m;
    for(i = 1; i <= n; i++)
    {
        string s;
        cin  >> s;
        if(s.find("G") > s.find("S"))
        {
            cout << -1;
            return 0;
        }
        else
        {
            p.insert(s.find("G") - s.find("S"));
        }
    }
    cout << p.size();
}
