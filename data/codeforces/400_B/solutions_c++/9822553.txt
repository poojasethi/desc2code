#include<bits/stdc++.h>

using namespace std;

int main(){
	int f,c;
	cin>>f>>c;
	int posG, posS;
	set<int>ms;
	char aux;
	bool solve =true;
	for(int i = 0;i<f;++i){
		for(int j=0;j<c;++j){
			cin>>aux;
			if(aux=='G')posG=j;
			if(aux=='S')posS=j;
		}
		if(posG>posS)solve=false;
		else{
			ms.insert(posS-posG);
		}
	}
	if(solve)cout<<ms.size()<<endl;
	else cout<<-1<<endl;
	
}
