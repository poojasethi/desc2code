#include <cstdio>

#define N 1000

int a[N + 1], n, m, i, g, k;
char s[N + 3];

int main()
{
	scanf("%d%d", &n, &m);

	while (n--)
	{
		scanf("%s", s);
		for (i = 0; i < m; i++)
		{
			if (s[i] == 'G')
				g = i;
			if (s[i] == 'S')
				k = i;
		}
		if (g > k)
		{
			puts("-1");
			return 0;
		}
		++a[k - g];
	}

	k = 0;
	for (i = 1; i <= N; i++)
		if (a[i])
			++k;

	printf("%d\n", k);

	return 0;
}
