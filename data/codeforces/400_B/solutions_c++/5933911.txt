#include<iostream>
#include<set>
using namespace std;

int N, M;
char t;

int main(){
    cin >> N >> M;
    set<int> dists;
    for(int n=0; n<N; n++){
        int G, S;
        for(int m=0; m<M; m++){
            cin >> t;
            if(t == 'G') G = m;
            if(t == 'S') S = m;
        }
        if(G > S){ cout << -1 << endl; return 0; }
        dists.insert(S-G);
    }
    cout << dists.size() << endl;
}
