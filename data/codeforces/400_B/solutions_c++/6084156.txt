#include<stdio.h>
#include<cstring>
#include<set>
using namespace std;
set<int> set1;
main()
{
 int n,m,i,x,y,j,f=0;
 scanf("%d%d",&n,&m);
 char a[n][m];
for(i=0;i<n;i++)
scanf("%s",a[i]);

for(i=0;i<n;i++)
{
 for(j=0;j<m;j++)
 {
  if(a[i][j]=='S')
   x=j;
   if(a[i][j]=='G')
    y=j;
 }
if(y>x)
{
printf("-1");
f=1;
break;
}
set1.insert(x-y);
}

if(f==0)
printf("%d",set1.size());
return 0; 
}
