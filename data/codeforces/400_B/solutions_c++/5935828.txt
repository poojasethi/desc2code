#include<iostream>
#include<algorithm>
#include<string>
#include<set>
using namespace std;

int a,b,n,m,wyn;
string s;
set<int> q;

int main(){
  cin>>n>>m;
  wyn=0;
  for(int i=0;i<n;i++){
    cin>>s;
    for(int j=0;j<m;j++){
      if(s[j]=='S') a=j;
      if(s[j]=='G') b=j;
    }
    if(a-b<0) wyn=m;
    else q.insert(a-b);
  }
  if(wyn==m) cout<<"-1\n";
  else cout<<q.size()<<"\n";
}
