#include <iostream>
#include <set>
using namespace std;
set <int> a;
int main(){
	int n, m;
	char c;
	cin >> n >> m;
	for(int i = 0; i < n; ++ i){
		int k = 0, kk = 0;
		for(int j = 0; j < m; ++ j){
			cin >> c;
			if(c == 'G') k = j;
			if(c == 'S') kk = j;
		}
		if(k < kk) a.insert(kk - k);
		else {
			cout << -1;
			return 0;
		}
	}
	if(a.size() > 0) cout << a.size();
	else cout << -1;
	return 0;
}
