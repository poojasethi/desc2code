#include <bits/stdc++.h>
using namespace std;
const  int N = 1100;
int n, m, hash[N];
set<int> dis;
int main()
{
    cin >> n >> m;
    for (int i = 0; i < n; ++i)
    {
        string s;
        cin >> s;
        int x = s.find('G'), y = s.find('S');
        dis.insert(y-x);
    }
    if (*dis.begin() < 0)   cout << -1 << endl;
    else    cout << dis.size() << endl;
    return 0;
}
