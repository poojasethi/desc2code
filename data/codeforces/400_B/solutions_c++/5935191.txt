#include <bits/stdc++.h>
using namespace std;

int main(){
	int n, m;
	cin >> n >> m;
	string line;
	int i, dist;
	set <int> st;
	for(i = 0; i < n; ++i){
		cin >> line;
		dist = 0;
		for(int j = 0; j < m && line[j] != 'S'; ++j){
			if(line[j] == 'G' || dist > 0) ++dist;
		}
		st.insert(dist);
	}
	if(*(st.begin()) == 0){
		cout << "-1\n";
	}
	else{
		cout << st.size() << endl;
	}
	return 0;
}
