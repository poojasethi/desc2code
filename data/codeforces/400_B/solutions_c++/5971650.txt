#include <iostream>
#include <set>
using namespace std;
string s;
int n,m;
int main()
{
    set<int>v;
    cin >> n >> m;
    for (int i=0; i<n; i++)
        {
            cin >> s;
            int x=(int)s.find('G'),y=(int)s.find('S');
            if (y-x<0)return cout << -1,0;
            v.insert(y-x);
        }
    cout << (int)v.size();
    return 0;
}
