#include <stdio.h>
#include <set>
using namespace std;
#define N 1010
char s[N];
set<int> st;
int main()
{
	int n,m,i,x,y;
	scanf("%d %d",&n,&m);
	while(n--)
	{
		scanf("%s",s);
		for(i=0;i<m;i++)if(s[i]=='G')x=i;
		else if(s[i]=='S')y=i;
		st.insert(y-x);
	}
	if(*st.begin()<0)printf("-1\n");
	else printf("%d\n",st.size());
	return 0;
}
