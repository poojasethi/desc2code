#include <iostream>
#include <set>

using namespace std;

int main()
{
	int n,m;
	cin >> n >> m;
	set<int> myst;
	for(int i = 0; i < n; i++){
		int pos1,pos2;
		for(int j = 0; j < m; j++){
			char ch;
			cin >> ch;
			if(ch == 'S') pos2 = j;
			else if(ch == 'G') pos1 = j;
		}
		myst.insert(pos2-pos1);
	}
	if(*(myst.begin()) < 0) cout << -1 << endl;
	else cout << myst.size() << endl;
	return 0;
}
