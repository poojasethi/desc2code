#include<iostream>
#include<string>
#include<algorithm>
#include<set>
using namespace std;
set<int> a;
int main()
{
	int n, m;
	cin >> n >> m;
	string t1;
	int pan = 1;
	for (int i = 1; i <= n; i++)
	{
		cin >> t1;
		int t = t1.find('S') - t1.find('G');
		if (t < 0) { pan = 0; break; }
		a.insert(t);
	}
	if (!pan) cout << -1 << endl;
	else cout << a.size() << endl;
	//system("pause");
	return 0;
}
