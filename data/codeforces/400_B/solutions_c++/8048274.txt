#include<stdio.h>
bool g[1005];
char str[1005];
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	getchar();
	int ans = 0;
	while(n--){
		int p_G,p_S;
		for(int i=0;i<=m;i++){
			scanf("%c",&str[i]);
			if(str[i] == 'G') p_G = i;
			else if(str[i] == 'S') p_S = i;
		}
		if(p_G > p_S){ans = -1; break;}
		int tem = p_S - p_G;
		if(!g[tem]) g[tem] = 1,ans++;
	}
	printf("%d\n",ans);
	return 0;
}
