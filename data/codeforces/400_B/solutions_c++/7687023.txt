#include<cstdio>
using namespace std;
bool b1[1000];char s[1000];
int main()
{
	int n,m,a,b,o=0;
	scanf("%d%d",&n,&m);
	while(n--)
	{
		scanf("%s",s);
		for(int i=0;i<m;++i)
		{
			if(s[i]=='G')a=i;
			if(s[i]=='S')b=i;
		}
		if(a>b){puts("-1");return 0;}
		b-=a;if(!b1[b])b1[b]=1,++o;
	}
	printf("%d\n",o);
}
