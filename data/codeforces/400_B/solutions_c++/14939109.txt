#include <iostream>
#include <set>
using namespace std;

set <int> s;

int main()
{
	int n, m;
	cin>> n >> m;

	for(int i = 0; i < n; i ++)
	{
		int x, y;
		for(int j = 0; j < m; j ++)
		{
			char a;
			cin>> a;
			if(a == 'G')
				x = j;
			if(a == 'S')
				y = j;
		}

		if(y < x)
		{
			cout<< "-1\n";
			return 0;
		}

		s.insert(y - x);
	}

	cout<< s.size() << endl;
}
