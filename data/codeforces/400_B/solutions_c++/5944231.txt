//BTTHS
#include <iostream>

using namespace std;

int n,m,mn,a,b,f[100009],ans;

string s;

bool o;

int main()
{
	cin >> n >> m;
	
	for (int i = 0; i < n; i++)
	{
		cin >> s;
		for (int h = 0; h < m; h++)
		{
			if (s[h] == 'G') a = h;
			else if (s[h] == 'S') b = h;
		}
		
		if (b < a) o = 1;
		else f[b-a] = 1;
	}
	
	if (o == 1)
		cout << -1;
	else
	{
		for (int i = 0; i <= 1000; i++)
			if (f[i]!=0) ans++;
		cout << ans;
	}	
}
