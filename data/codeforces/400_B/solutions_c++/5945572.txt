#include <iostream>
#include <set>
using namespace std;

int main() {
    int n,m;
    char c;
    cin>>n>>m;
    set<int> f;
    for(int i=0;i<n;i++)
    {
    int a,b;
    for(int j=0;j<m;j++)
    {
    cin>>c;
    if(c=='G')  
    a=j;
    if(c=='S')
    b=j;
    }
    if(b-a < 0)
    {cout<<"-1\n";return 0;}
    f.insert(b-a);
    }
    cout<<f.size()<<"\n";
    return 0;
}
