#include <iostream>
#include <string>
#include <set>
using namespace std;

int main()
{
	int n,m,temp;
	cin>>n>>m;
	set<int> s;
	string ss;
	bool cont=1;
	for (int i=0 ; i<n ; i++)
	{
		cin>>ss;
		temp=ss.find("S")-ss.find("G");
		if (temp<0)
		{
			cont=0;
			break;
		}
		s.insert(temp);
	}
	if (cont) cout<<s.size()<<endl;
	else cout<<"-1\n";
	return 0;
}
