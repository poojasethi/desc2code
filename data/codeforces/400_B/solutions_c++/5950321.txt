#include<iostream>
using namespace std;
int x[1002];
int main()
{
	int n,m;
	cin>>n>>m;;
	string a;
	int cnt=0;
	int flag=1;
	for(int i=0;i<n;i++)
	{
		cin>>a;
		int n1=a.find('G');
		int n2=a.find('S');
		if(n1>n2)
		{
			cout<<-1<<endl;
			flag=0;
			break;
		}
		else
			x[n2-n1]=1;
	}
	if(flag)
	{
		for(int i=1;i<m;i++)
			if(x[i])
				cnt++;
		cout<<cnt<<endl;
	}
	return 0;
}
