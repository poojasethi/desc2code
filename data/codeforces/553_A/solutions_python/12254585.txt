def fac(n):
  r = 1
  for x in xrange(1, n+1):
    r = r * x
  return r

k = int(raw_input())
c = []
for i in xrange(0, k):
  c.append(int(raw_input()))

P = [c[0]]
for i in xrange(1, k):
  P.append(c[i] + P[i-1])

dp = [1]
for i in xrange(1, k):
  r = dp[i-1]
  for a in xrange(1, c[i]):
    r = r * (P[i-1] + a)
  r = r / (fac(c[i]-1))
  dp.append(r % 1000000007)

print dp[k-1]
