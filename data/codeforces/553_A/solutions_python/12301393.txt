from math import factorial

def permut(a,b):
	return factorial(a)/(factorial(b) * factorial(a - b))


n = int(raw_input())

balls = []

for i in xrange(n):
	num = int(raw_input())
	balls.append(num)

total = 1

sum = balls[0]

for i in xrange(1,len(balls)):
	total *= permut(sum + balls[i] - 1 , balls[i] - 1)
	sum += balls[i]

print total % 1000000007
