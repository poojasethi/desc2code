def main():
    k = int(raw_input())
    a = [int(raw_input()) for _ in xrange(k)]
    b = [0] * 4000
    c = [0] * 4000
    t = 1
    for x in a:
        b[t + x - 2] += 1
        c[t - 1] += 1
        c[x - 1] += 1
        t += x
    ans = x = t = 1
    mod = 1000000007
    for i in xrange(1, 4000):
        t = t * i % mod
        ans = ans * pow(t, b[i], mod) % mod
        x = x * pow(t, c[i], mod) % mod
    ans = ans * pow(x, mod - 2, mod) % mod
    print ans
main()
