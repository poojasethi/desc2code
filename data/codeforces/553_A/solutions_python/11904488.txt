mod = 1000000007
c = [[1,0]]
for i in range(1,1001):
    row = [1]
    for j in range(1, i + 1):
        row.append((c[-1][j] + c[-1][j - 1]) % mod)
    row.append(0)
    c.append(row)

n = int(raw_input())
a = int(raw_input())
ans = 1
for i in range(1, n):
    b = int(raw_input())
    ans *= c[a + b - 1][b - 1]
    ans %= mod
    a += b
print ans
