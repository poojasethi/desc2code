k = input()
counts = [input() for _ in xrange(k)]

total = 1
su = sum(counts)

from math import factorial

def c(n, k):
    return (factorial(n) // (factorial(n-k) * factorial(k))) % (10 ** 9 + 7)

for i in range(k-1, -1, -1):
    total = (total * c(su - 1, counts[i]-1)) % (10 ** 9 + 7)
    su -= counts[i]

print total
