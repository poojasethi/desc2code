def com(n , m) :
    ans = 1
    for i in range(m) :
        ans *= n-i
    for i in range(m) :
        ans /= i+1
    return ans % (10**9+7)

def solve(n , m) :
    return com(n+m-1 , n-1)

n = input()
a = []
for i in range(n) :
    a.append(int(input()))

b = [0]
for i in range(n) :
    b.append(b[i]+a[i])

F = 1
for i in range(n) :
    F = F * solve(b[i]+1 , a[i]-1)
    F %= 10**9+7

print F
