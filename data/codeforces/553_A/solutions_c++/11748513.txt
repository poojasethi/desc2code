#include <iostream>
#define mod 1000000007
using namespace std;
long long dp[1009][1009]={0};

void init()
{
for(int i=0;i<=1000;i++)
{
dp[i][0]=1;
for(int j=1;j<=i;j++)
dp[i][j]=(dp[i-1][j]+dp[i-1][j-1])%mod;
}
}

int main() {
    int k;
    int c;
    init();
    cin>>k;
    int sum=0;
    long long res=1LL;
    for(int i=1;i<=k;i++)
    {
    cin>>c;
    res=(res*(dp[sum+c-1][c-1]))%mod;
    sum=sum+c;
    }
    cout<<res<<"\n";
    return 0;
}
