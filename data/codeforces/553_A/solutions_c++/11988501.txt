#include<iostream>
using namespace std;
int main()
{
    int c[1000][1000];
    c[0][0]=1;

    for(int i=1;i<1000;i++)
    {
        c[i][0]=1;
        c[i][i]=1;
      for(int j=1;j<i;j++)
      {
        c[i][j]=(c[i-1][j]+c[i-1][j-1])%1000000007;
      }

    }
    int n;
    cin>>n;
    long long ans=1;
    int x,sum=0;

    for(int i=0;i<n;i++)
    {
        cin>>x;
        sum=sum+x;
        ans=(ans*c[sum-1][x-1])%1000000007;
    }
    cout<<ans;
}
