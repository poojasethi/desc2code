#include<cstdio>
using namespace std;
long long P=1000000007,c[1010][1010],ans=1;
int sum,k,a[1010],i,j;
int main()
{
	scanf("%d",&k);
	for (i=1;i<=k;i++) scanf("%d",&a[i]);
	for (i=1;i<=k;i++) sum+=a[i];
	c[0][0]=1;
	for (i=1;i<=sum;i++){c[i][0]=1;c[i][i]=1;}
	for (i=2;i<=sum;i++)
	  for (j=1;j<i;j++)
	    c[i][j]=(c[i-1][j-1]+c[i-1][j])%P;
	for (i=k;i>=1;i--)
	{
		ans=ans*c[sum-1][a[i]-1]%P;
		sum-=a[i];
	}
	printf("%lld\n",ans);
	return 0;
}
