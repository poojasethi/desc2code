#include <cstdio>
#include <iostream>

using namespace std;

const long long M = 1000 * 1000 * 1000 + 7;

long long c[2001][2001];

int main()
{
	c[0][0] = 1;

	for (int n = 1; n < 2001; n++)
	{
		c[n][0] = 1;

		for (int k = 1; k <= n; k++)
		{
			c[n][k] = (c[n-1][k-1] + c[n-1][k]) % M;
		}
	}

	long long ans = 1, sum = 0;

	int k;

	cin >> k;

	for (int i = 0; i < k; i++)
	{
		int x;

		cin >> x;

		ans = (ans * c[sum + x - 1][x - 1]) % M;

		sum += x;
	}

	cout << ans << endl;
}
