#include <stdio.h>

const long long mod=1000000007;

long long inv(long long b)
{
	return b==1 ? 1 : inv(mod%b)*(mod-mod/b)%mod;
}

int main()
{
	int n;
	int sum=0;
	long long ans=1;
	scanf("%d",&n);
	for(int i=0; i<n; i++){
		int c;
		scanf("%d",&c);
		c--;
		for(int j=c+sum,cnt=0; cnt<c; cnt++){
			ans=ans*(j-cnt)%mod;
			ans=ans*inv(cnt+1)%mod;
		}
		sum+=c+1;
	}
	printf("%d\n",int(ans));
	return 0;
}
