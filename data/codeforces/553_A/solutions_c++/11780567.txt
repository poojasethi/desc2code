#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long dp[1005][1005];
int main()
{
    for(long long i=0;i<1005;i++)dp[0][i]=1;
    for(long long i=1;i<1005;i++)
    for(long long j=1;j<1005;j++)
    dp[i][j] = (dp[i-1][j-1] + dp[i][j-1]) % mod;
    
    long long m,t=0,ans=1; cin >> m;
    while(m--)
    {
        long long n; cin>>n;
        t+=n;
        ans = (ans * dp[n-1][t-1] ) %mod;
    }
    cout<<ans<<endl;
}
