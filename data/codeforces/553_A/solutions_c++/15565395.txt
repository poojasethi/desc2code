#include <stdio.h>
#define MOD 1000000007
int c[1001][1001];
int main()
{
    c[0][0]=1;
    for (int i=1;i<1001;i++)
    {
        c[i][0]=1;
        for (int j=1;j<=i;j++)
            c[i][j]=(c[i-1][j-1]+c[i-1][j])%MOD;
    }
    int k,ans=1,tot=0,x;
    scanf("%d",&k);
    for (int i=1;i<=k;i++)
    {
        scanf("%d",&x);
        ans=((long long)ans*c[tot+x-1][x-1])%MOD;
        tot+=x;
    }
    printf("%d",ans);
    return 0;
}
