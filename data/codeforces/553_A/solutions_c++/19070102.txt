#include <iostream>
using namespace std;
const int modulo=1e9+7;
int a[1010][1010];
int main()
{
    int k;
    cin>>k;
    int i,j,t,s=0;
    for (i=0; i<1010; i++)
    {
        a[i][0]=1;
        for (j=1; j<=i; j++)
            a[i][j]=(a[i-1][j-1]+a[i-1][j])%modulo;
    }
    long long ans=1;
    for(i=0;i<k;i++)
    {
        cin>>t;
        s+=t;
        ans=(ans*a[s-1][t-1])%modulo;
    }
    cout<<ans;
    return 0;
}
