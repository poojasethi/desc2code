#define N (1<<10)
#define M 1000000007
#define LL long long
#include <bits/stdc++.h>

using namespace std;

int n,x,sum;
LL dp[N][N],res=1;

int main()
{
	for(int i=1;i<N;i++)
		dp[0][i]=1;

	for(int i=1;i<N;i++)
		for(int j=1;j<N;j++)
			dp[i][j]=dp[i][j-1]+dp[i-1][j],dp[i][j]%=M;


	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		res*=dp[x-1][sum+1],res%=M,sum+=x;
	}
	cout<<res<<endl;
	return 0;
}
