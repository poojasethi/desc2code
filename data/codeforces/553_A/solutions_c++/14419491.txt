#include <bits/stdc++.h>

using namespace std;

long long cnk[1505][1504];
long long c,k,ans;

int main(){
    
    for(int i=0;i<=1500;++i) cnk[i][0]=1;
    for(int i=1;i<=1500;++i)
        for(int j=1;j<=i+1;++j)
            cnk[i][j]=(cnk[i-1][j-1]+cnk[i-1][j])%1000000007;
    cin >> k;
    ans=1;
    for(int i=1;i<=k;++i){
        int x;
        cin >> x;
        ans=ans*1LL*cnk[c+x-1][c]%1000000007;
        c+=x;
    }
    cout << ans;
}
