#include <cstdio>
#include <algorithm>
#define mo 1000000007
using namespace std;
int a[1005],b,c[1005][1005],d,e,f[1005],g,h,i,j,k,l,m,n;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=0;i<=1000;i++){
		c[i][0]=1;
		for(j=1;j<i;j++)c[i][j]=(c[i-1][j-1]+c[i-1][j])%mo;
		c[i][i]=1;
	}
	f[0]=1;
	for(i=1;i<=n;i++){
		g=g+a[i];
		f[i]=1ll*f[i-1]*c[g-1][a[i]-1]%mo;
	}
	printf("%d\n",f[n]);
	return 0;
}
