#include <iostream>

using namespace std;

#define MOD 1000000007

long long dp[1005][1005];
long long k, n;

int main()
{
  cin>>k;
  for(int i=1;i<1005;i++)
    dp[0][i]=1, dp[i][0]=0;
    for(int i=1;i<1005;i++)
      for(int j=1;j<1005;j++)
        dp[i][j]=dp[i][j-1]+dp[i-1][j],dp[i][j]%=MOD;
  n=0;
  long long aux, sol=1;
  for(int i=0;i<k;i++)
  {
    cin>>aux;
    sol*=dp[aux-1][n+1];
    sol%=MOD;
    n+=aux;
  }
  cout<<sol<<endl;
}
