#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll ans=1,mod=1000000007;
int a[1005][1005],t,i,n,k,j;
int main(){
	a[0][0]=1;
	for(i=1;i<1005;i++){
		a[i][0]=1;
		for(j=1;j<1005;j++)
		{
			a[i][j]=(a[i-1][j-1]+a[i-1][j])%mod;
		}
    }
	t=0;
	scanf("%d",&k);
	while(k--){
		scanf("%d",&n);
		ans=(ans*a[t+n-1][n-1])%mod;
		t+=n;
	}
	printf("%lld\n",ans);
	return 0;
}
