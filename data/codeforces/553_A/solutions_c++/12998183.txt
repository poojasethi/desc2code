#include<bits/stdc++.h>
#define MAX 1005
#define M 1000000007
using namespace std;
int main()
{
	int k;
	cin>>k;
	int a[k+1];
	for(int i=0;i<k;i++)
		cin>>a[i];
	long long int comb[MAX][MAX];
	comb[0][0]=1;
	for(int i=1;i<MAX;i++)
	{
		comb[i][0]=1;
		for(int j=1;j<=i;j++)
		{
			comb[i][j]=(comb[i-1][j]+comb[i-1][j-1])%M;
		}
	}
	long long sum=0,ans=1;
	for(int i=0;i<k;i++)
	{
		ans=(ans*comb[sum+a[i]-1][a[i]-1])%M;
		sum+=a[i];
	}
	cout<<ans<<endl;
	return 0;
}
