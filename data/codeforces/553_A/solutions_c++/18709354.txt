#include<bits/stdc++.h>
#define ll long long
using namespace std;
const int N=1005;
ll b[N][N],MOD=1e9+7;
int main()
{
    ll k,c,s=0,i,j,ans=1;
    cin>>k;
    for(i=0;i<=1000;i++)
    	{
    		b[i][0]=b[i][i]=1;
    	     for(j=1;j<i;j++)
            b[i][j]=(b[i-1][j-1]+b[i-1][j])%MOD;
    	}
    for(i=0;i<k;i++)
        {
          cin>>c;
          s+=c;
          ans=(ans*b[s-1][c-1])%MOD;
        }
        cout<<ans;
	return 0;
}
