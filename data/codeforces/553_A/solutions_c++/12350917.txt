#include <iostream>

using namespace std;

long long k,c[1005],sum,ans,zuhe[1100][1100];

int main()
{
	cin>>k;
	for (int i=1;i<=k;i++)
	{
	    cin>>c[i];
	    sum+=c[i];
	}
	zuhe[0][0]=1;
	for (int i=1;i<=1005;i++)
	{
		zuhe[i][0]=1;
	    for (int j=1;j<=i;j++)
	        zuhe[i][j]=(zuhe[i-1][j-1]+zuhe[i-1][j])%1000000007;
	}
	ans=1;
	for (int i=k;i>=1;i--)
	{
	    ans=ans*zuhe[sum-1][c[i]-1]%1000000007;
	    sum=sum-c[i];
	}
	cout<<ans<<endl;
    return 0;
}
