#include <iostream>
#include <string>
#include <cassert>
using namespace std;

#define MAX 2015
#define M 1000000007

int n,a[MAX];
long long dp[MAX],c[MAX][MAX];

int main(){
	int i,j,k,tot;
	cin>>n; for (i=0; i<n; i++) cin>>a[i];
	for (i=0; i<=1000; i++) for (c[i][0]=1,j=1; j<=i; j++) c[i][j]=(c[i-1][j]+c[i-1][j-1])%M;
	dp[0]=1; tot=a[0];
	for (i=1; i<n; tot+=a[i++]) dp[i]=(dp[i-1]*c[a[i]-1+tot][tot])%M;
	cout<<dp[n-1]<<endl;
	return 0;
}
