#define ll long long
#define md 1000000007
int a1[1001][1001];

{
    for(int i=0;i<1001;++i)a1[i][0]=1;
    for(int i=1;i<1001;++i)for(int j=1;j<=i;++j)
        a1[i][j]=(a1[i-1][j-1]+a1[i-1][j])%md;
    int n,k,a,b;ll o=1;
    scanf("%d%d",&n,&a);
    for(int i=1;i<n;++i,a+=b)
    {scanf("%d",&b);o=o*a1[a+b-1][b-1]%md;}

}
