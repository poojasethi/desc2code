#include<iostream>
using namespace std;
long long a[1001][1001],ans,mod=1e9+7;
int main()
{
    int i,n,j,b,s=0;
    for(i=0;i<1001;i++)
        a[i][0]=1;
    for(i=1;i<1001;i++)
        for(j=1;j<=i;j++)
            a[i][j]=(a[i-1][j]+a[i-1][j-1])%mod;
    cin>>n>>s;
    ans=1;
    for(i=2;i<=n;i++)
    {
        cin>>b;
        ans*=(a[s+b-1][b-1]);
        ans%=mod;
        s+=b;
    }
    cout<<ans<<endl;
}
