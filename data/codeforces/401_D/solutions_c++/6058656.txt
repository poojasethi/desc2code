#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <iostream>
#include <algorithm>
#define LL long long
using namespace std;
LL f[1<<18][100];
int a[18];
LL N;
int M,len;
int main(){
    
	scanf("%I64d%d",&N,&M);
	while (N)a[++len]=N%10,N/=10;
	sort(a+1,a+len+1);
	for (int i=1;i<=len;i++)
	if ((i==1||a[i]!=a[i-1])&&a[i])
	f[1<<(i-1)][a[i]%M]++;
	for (int i=1;i<1<<len;i++)
	for (int j=0;j<M;j++)
	if (f[i][j]){
		for (int k=0;k<len;k++)
		if (!((i>>k)&1))
		if (a[k]!=a[k+1]||!k||((i>>(k-1))&1))
		f[i|(1<<k)][(j*10+a[k+1])%M]=f[i|(1<<k)][(j*10+a[k+1])%M]+f[i][j];
	}
	printf("%I64d\n",f[(1<<len)-1][0]);
}
