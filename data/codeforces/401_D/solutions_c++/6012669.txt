#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
long long dp[1<<18][105];
char str[20];
int map[20];
int main(void)
{
    int n,m,k;
    scanf("%s%d",str,&m);
    n=strlen(str);
    memset(dp,0,sizeof(dp));
    dp[0][0]=1;
    for(int i=0;i<(1<<n);i++)
    {
        for(int j=0;j<m;j++)
	{
	    memset(map,0,sizeof(map));
	    for(int k=0;k<n;k++)
	    {
	        int x=str[k]-'0';
		if(i&(1<<k)) continue;
		if(i==0&&x==0) continue;
		if(map[x]) continue;
		map[x]=1;
		dp[i|(1<<k)][(j*10+x)%m]+=dp[i][j];
	    }
	}
    }
    printf("%I64d",dp[(1<<n)-1][0]);
    return 0;
}
