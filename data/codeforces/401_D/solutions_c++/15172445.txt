#include<bits/stdc++.h>
using namespace std;
long long arr[21],m,a,dp[500009][101];
//123456789123456789 2
string n;
long long rec(int mask,int val){
	if(__builtin_popcount(mask)>=a)
		return (!val);
	long long &ret=dp[mask][val];
	if(ret!=-1)
		return ret; 
	ret=0;		
	for(int i=0;i<a;i++){
		if((!mask and !arr[i]))
			continue;
		if(!(mask&(1<<i))){	
			ret+=rec((mask|(1<<i)),(val*10+arr[i])%m);
			while(arr[i]==arr[i+1])
				i++;
		}
	}
	return ret;		
}
int main(){
	memset(dp,-1,sizeof(dp));
	cin>>n>>m;
	a=n.size();
	for(int i=0;i<a;i++)
		arr[i]=n[i]-'0';	
	sort(arr,arr+a);
	cout<<rec(0,0)<<endl;	
	return 0;
}
