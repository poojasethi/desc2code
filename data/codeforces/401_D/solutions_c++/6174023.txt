#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include<math.h>
#include<string.h>
#include<time.h>
using namespace std;
int i,j,k,m;
int s[20],h[20];
long long n,f[300000][110];
int main() {
    scanf("%I64d%d",&n,&m);
    for(;n;n/=10)
      s[n%10]++;
    h[0]=1;
    for(i=1;i<=10;i++)
      h[i]=h[i-1]*(s[i-1]+1);
    n=h[10];
    for(i=1;i<=10;i++)
      if(s[i])
        f[h[i]][i%m]++;
    for(i=0;i<=n;i++)
      for(j=0;j<=m;j++)
        if(f[i][j])
          for(k=0;k<=10;k++)
            if (i/h[k]%(s[k]+1)!=s[k])
              f[i+h[k]][(j*10+k)%m]+=f[i][j];
    printf("%I64d\n",f[n-1][0]);
}
