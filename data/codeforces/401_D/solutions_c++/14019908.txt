#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,m;
int s[20],h[20];
int64 n,f[300000][110];
int main() {
	scanf("%I64d%d",&n,&m);
	for (;n;n/=10) s[n%10]++;
	h[0]=1; For(i,1,10) h[i]=h[i-1]*(s[i-1]+1);
	n=h[10];
	For(i,1,10) if (s[i]) f[h[i]][i%m]++;
	For(i,0,n)For(j,0,m) if (f[i][j])
		For(k,0,10) if (i/h[k]%(s[k]+1)!=s[k]) f[i+h[k]][(j*10+k)%m]+=f[i][j];
	printf("%I64d\n",f[n-1][0]);
	return 0;
}
