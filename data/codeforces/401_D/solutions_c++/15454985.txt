#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int a[20];
ll dp[1<<18][100],ans;
int freq[10];
char s[20];
int main()
{
	ios_base::sync_with_stdio(0);	cin>>s;	int n=0;
	for(;s[n];n++){	a[n]=s[n]-'0';	freq[s[n]-'0']++;	}
	int m,H=1<<n;	cin>>m;	
	for(int i=0;i<n;i++)	if(a[i])	dp[1<<i][a[i]%m]=1;
	for(int mask=1;mask<H;mask++)
	{
		for(int rem=0;rem<m;rem++)
		{
			for(int j=0;j<n;j++)
			{
				if((1<<j)&mask)continue;
				dp[mask|(1<<j)][((rem*10)+a[j])%m]+=dp[mask][rem];
			}
		}
	}
	ans=dp[(1<<n)-1][0];
	for(int i=0;i<=9;i++)
		for(ll j=2;j<=freq[i];j++)
			ans/=j;
	cout<<ans<<"\n";
	return 0;
}
