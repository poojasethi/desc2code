#include<cstdio>
bool isemptyset(int*a,int*b,int*c){
  int i;
  for(i=0;i<2;++i)
    if(!a[i] && !b[i] &&c[i])
      return 1;
  return 0;
}
int main(){
  int a[2],b[2],c[2],i;
  for(i=0;i<2;++i)scanf("%d%d%d",a+i,b+i,c+i);
  if(isemptyset(a,b,c))puts("0");
  else if(a[0]*b[1]==b[0]*a[1] && a[0]*c[1]==c[0]*a[1] && b[0]*c[1]==c[0]*b[1])puts("-1");
  else if(a[0]*b[1]==b[0]*a[1])puts("0");
  else puts("1");
  return 0;
}
