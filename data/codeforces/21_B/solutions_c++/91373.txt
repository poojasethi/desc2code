#include <iostream>
using namespace std;

int main()
{	
	int a1, b1, c1;
	int a2, b2, c2;
	cin >> a1 >> b1 >> c1;
	cin >> a2 >> b2 >> c2;
	bool t1 = (a1 == 0 && b1 == 0), t2 = (a2 == 0 && b2 == 0);
	
	if (t1 && c1 != 0)
	{
		cout << 0;
		return 0;
	}
	if (t2 && c2 != 0)
	{
		cout << 0;
		return 0;
	}
	if (t1)
	{
		cout << -1;
		return 0;
	}
	if (t2)
	{
		cout << -1;
		return 0;
	}

	if (b1*a2 != b2*a1)
		cout << 1;
	else if (c1*a2 == c2*a1) cout << -1;
	else
		cout << 0;
}
