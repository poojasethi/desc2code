#include <iostream>
#include <string>
#include <vector>
#include <iomanip>
#include <cmath>
using namespace std;

int main() {
	int a1, b1, c1, a2, b2, c2;
	cin >> a1 >> b1 >> c1 >> a2 >> b2 >> c2;
	int sects = 0;
	if (a1*b2 - b1*a2 != 0) sects = 1;
	else if ((!a1 && !b1 && c1) || (!a2 && !b2 && c2)) sects = 0;
	else if (c2*a1 == c1*a2 && c2*b1 == c1*b2) sects = -1;
	cout << sects << endl;
	return 0;
}
