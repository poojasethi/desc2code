#include<cstdio>
using namespace std;

int main()
{
	int  a, b, c, A, B, C;
	scanf("%d%d%d%d%d%d", &a, &b, &c, &A, &B, &C);
	if ((a == 0 && b == 0) || (A == 0 && B == 0))
	{
		if ((a == 0 && b == 0 && c != 0) || (A == 0 && B == 0 && C != 0))
			printf("0");///空集
		else
			printf("-1");///全集
	}
	else if (a * B == A * b)
	{
		if (a)
			printf("%d", a * C != A * c ? 0 : -1);
		else
			printf("%d", b * C != B * c ? 0 : -1);
	}
	else
		printf("1");
	return 0;
}
