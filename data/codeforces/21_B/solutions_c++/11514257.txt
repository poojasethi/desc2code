#include<stdio.h>
int main()
{
int a1,a2,b1,b2,c1,c2;
scanf("%d%d%d%d%d%d",&a1,&b1,&c1,&a2,&b2,&c2);
int ans = -1;
if(a1*b2 != a2*b1)ans = 1;
else if(a1*c2 !=a2*c1)ans = 0;
else if(b1*c2 !=b2*c1)ans = 0;
else if(a1==a2 && b1==b2 && c1!=c2)ans = 0;
else if(a1==0 && b1==0 && c1!=0)ans = 0;
else if(a2==0 && b2==0 && c2!=0)ans = 0;
printf("%d\n",ans);
return 0;
}
