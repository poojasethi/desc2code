#include<iostream>
#include<cmath>
#include<cstdio>

{

	scanf("%d%d%d%d%d%d",&a1,&b1,&c1,&a2,&b2,&c2);

	{
		printf("0\n");
		return 0;
	}

		printf("-1\n");
		return 0;
	}
	if(a1*b2-b1*a2==0)
	{
		printf("0\n");

	}
