#include <cstdio>

using namespace std;

int main()
{
    int a1, b1, c1, a2, b2, c2;
    scanf("%d%d%d%d%d%d", &a1,&b1,&c1,&a2,&b2,&c2);
    int ans;

    if (a1*b2 == a2*b1) {
        if (a1*c2 == a2*c1 && b1*c2 == b2*c1) {
            ans = -1;
        } else {
            ans = 0;
        }
    } else {
        ans = 1;
    }
    if ((a1==0&&b1==0&&c1!=0) || (a2==0&&b2==0&&c2!=0))
        ans = 0;
    printf("%d\n", ans);
    return 0;
}
