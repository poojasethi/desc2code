#include<iostream>
using namespace std;
int main(){
	int A1,B1,C1,A2,B2,C2,S,X,Y,r;
	cin>>A1>>B1>>C1;
	cin>>A2>>B2>>C2;
	S = (A1*B2 - A2*B1);
	X = (C1*B2 - C2*B1);
	Y = (A1*C2 - A2*C1);
	
	if(S != 0 ) r = 1;
	else{ 
		if(S == 0 && X == 0 && Y == 0) r = -1;
		else r = 0;
	}
	if(A1 == A2  && A1 == 0 && B1 == B2 && B1 == 0 && C1 == C2) r = 0;
	if(A1 == A2  && A1 == 0 && B1 == B2 && B1 == 0 && C1 != C2) r = 0;
	
	cout<<r<<endl;
	return 0;
}
