#include <iostream>
#include <vector>
#include <string>

using namespace std;

bool empty(int a,int b,int c){
  return a == 0 && b== 0 && c != 0;
}

int main(){
  int a,b,c;cin >> a >> b>> c;
  int aa,bb,cc;cin >> aa >> bb >> cc;

  if(empty(a,b,c) || empty(aa,bb,cc)){
    cout << 0 <<endl;
    return 0;
  }

  if(a * bb - b*aa ==0){
    if(c * aa - a * cc == 0 ){
      cout << -1 << endl;
    }else{
      cout << 0 <<endl;
    }
  }else{
    cout << 1 << endl;
  }
  return 0;
}
