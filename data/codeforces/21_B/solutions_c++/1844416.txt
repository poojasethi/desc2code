#include<stdio.h>
int main()
{
int a1,b1,c1,count;
scanf("%d %d %d",&a1,&b1,&c1);
int a2,b2,c2;
scanf("%d %d %d",&a2,&b2,&c2);
c1=c1*(-1);
c2=c2*(-1);
int xs=c1*b2-c2*b1;
int ys=a1*c2-a2*c1;
int ds=a1*b2-a2*b1;
if(a1==0&&b1==0&&a2==0&&b2==0)
	{
	count=0;
	}
else if(xs==0&&ys==0&&ds==0)
	{
	count=-1;
	}
else if(ds==0)
	{
	count=0;
	}
else
	{
	count=1;
	}
printf("%d",count);
return(0);
}
