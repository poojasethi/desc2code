#include<stdio.h>
int main()
{
	int a1,b1,c1,a2,b2,c2,s;
	while(~scanf("%d%d%d%d%d%d",&a1,&b1,&c1,&a2,&b2,&c2))
	{
		if(a1==0&&b1==0&&c1!=0)
		{
			s=0;
		}
		else if(a2==0&&b2==0&&c2!=0)
		{
			s=0;
		}
		else if(a1==0&&a2==0)
		{
			if(b1*c2==b2*c1)
				s=-1;
			else s=0;
		}
		else if(b1==0&&b2==0)
		{
			if(a1*c2==a2*c1)
				s=-1;
			else s=0;
		}
		else if(a1*b2==a2*b1)
		{
			if(a1*c2==a2*c1)
				s=-1;
			else s=0;
		}
		else s=1;
		printf("%d\n",s);
	}
	return 0;
}
