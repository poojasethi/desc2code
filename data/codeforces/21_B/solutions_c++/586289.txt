#include <stdio.h>

int main(void) {
	int a, b, c, d, e, f;
	scanf("%d%d%d%d%d%d", &a, &b, &c, &d, &e, &f);
	int delta = a*e-b*d;
	if(a==0&&b==0&&c==0){
		puts(d==0&&e==0&&f!=0?"0":"-1");
		return 0;
	}
	if(d==0&&e==0&&f==0){
		puts(a==0&&b==0&&c!=0?"0":"-1");
		return 0;
	}
	if(a==0&&b==0&&c!=0) puts("0");
	else if(d==0&&e==0&&f!=0) puts("0");
	else if(delta==0) {
		if(c*d==f*a) puts("-1");
		else puts("0");
	} else
		puts("1");
	return 0;
}
