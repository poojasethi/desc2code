#include <cstdio>
int v[222], c[222][222], ou[222];
bool ex[222];
int main() {
    int n, k, cnt;
    scanf("%d", &n);
    for (int i = 0;i < (n * (n - 1)) / 2;i++) {
        scanf("%d", &k);
        for (int j = 0;j < k;j++) {
            scanf("%d", &v[j]); ex[v[j]] = true;
            for (int l = 0;l < j;l++)
                c[v[l]][v[j]]++, c[v[j]][v[l]]++;
        }
    }
    k = 1;
    for (int i = 1;i <= 200;i++) if (ex[i]) {
        cnt = 1; ou[0] = i;
        for (int j = i + 1;j <= 200;j++) if (ex[j] && c[i][j] == n - 1)
            ou[cnt++] = j;
        if (n == 2 && k) { cnt = 1; k = 0; }
        printf("%d ", cnt);
        for (int j = 0;j < cnt;j++)
            printf("%d ", ou[j]), ex[ou[j]] = false;
        puts("");
    }
}
