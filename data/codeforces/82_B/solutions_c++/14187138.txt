#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <vector>
using namespace std;
int inf=1000000000;
int n,x,k,i,j,a[201][4]={0};
vector <int> q;
int main()
{
	cin>>n;
	int maxi=0;
	if(n==2)
	{
		cin>>k;
		for(i=0;i<k;i++)
		{
			cin>>x;
			q.push_back(x);
		}
		cout<<k-1<<" ";
		for(i=0;i<k-1;i++)cout<<q[i]<<" ";
		cout<<endl<<"1 "<<q[k-1];
		return 0;
	}
	for(i=0;i<n*(n-1)/2;i++)
	{
		cin>>k;
		for(j=0;j<k;j++)
		{
			cin>>x;
			if(x>maxi)maxi=x;
			if(a[x][1]==0)a[x][1]=i;else 
			{
				a[x][2]=i;
				if(a[x][2]<a[x][1])swap(a[x][2],a[x][1]);
			}
		}
	}
	for(i=0;i<n;i++)
	{
		x=1;
		while(a[x][3] || a[x][1]==0)x++;
		q.clear();
		for(j=x;j<=maxi;j++)
			if(a[x][1]==a[j][1] && a[x][2]==a[j][2])
			{
				q.push_back(j);
				a[j][3]=1;
			}
		cout<<q.size()<<" ";
		for(j=0;j<q.size();j++)
			cout<<q[j]<<" ";
		cout<<endl;
	}	
	return 0;
}
