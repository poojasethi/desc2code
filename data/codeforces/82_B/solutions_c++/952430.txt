#include<iostream>
#include<vector>
#include<map>
#define pb push_back
using namespace std;
const int maxn=210;
vector<int> v[maxn];
map<vector<int>,vector<int> > mp;
int main(){
	int n;
	cin>>n;
	if(n==2){
		int t,x;
		cin>>t;
		while(t--){
			cin>>x;
			v[0].pb(x);
		}
		cout<<1<<' '<<v[0][0]<<endl;
		cout<<v[0].size()-1<<' ';
		for(int i=1;i<v[0].size();i++)
			cout<<v[0][i]<<' ';
		cout<<endl;
		return 0;
	}
	for(int i=0;i<n*(n-1)/2;i++){
		int t,x;
		cin>>t;
		while(t--){
			cin>>x;
			v[x].pb(i);
		}
	}
	for(int i=0;i<maxn;i++)
		if(!v[i].empty())
			mp[v[i]].pb(i);
	map<vector<int>,vector<int> >::iterator it;
	for(it=mp.begin();it!=mp.end();cout<<endl,it++){
		cout<<(it->second).size()<<' ';
		for(int i=0;i<((it->second).size());i++)
			cout<<(it->second)[i]<<' ';
	}
}
