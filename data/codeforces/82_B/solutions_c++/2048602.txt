#include <cstdio>
#include <cstdlib>

int n, r;

bool num[210];
bool sets[20100][210];

int main()
{
	scanf("%d", &n);

	if(n==2) {
		int k; scanf("%d", &k);
		int numbers[210];
		for(int i=0; i<k; i++) scanf("%d", &numbers[i]);
		printf("1 %d\n", numbers[0]);
		printf("%d", k-1);
		for(int i=1; i<k; i++) printf(" %d", numbers[i]);
		printf("\n");
		exit(EXIT_SUCCESS);
	}

	r=n*(n-1)/2;

	for(int i=1; i<=200; i++) num[i]=false;
	for(int i=0; i<r; i++) for(int j=1; j<=200; j++) sets[i][j]=false;

	for(int i=0; i<r; i++) {
		int k; scanf("%d", &k);
		for(int j=0; j<k; j++) {
			int a; scanf("%d", &a);
			num[a]=true;
			sets[i][a]=true;
		}
	}

	for(int i=1; i<=200; i++) if(num[i]) {
		bool ok[201];
		for(int j=1; j<=200; j++) ok[j]=true;
		for(int j=0; j<r; j++) if(sets[j][i]) {
			for(int k=1; k<=200; k++) ok[k]&=sets[j][k];
		}
		int c=0;
		for(int j=1; j<=200; j++) if(ok[j]) {num[j]=false; c++;}
		printf("%d", c);
		for(int j=1; j<=200; j++) if(ok[j]) printf(" %d", j);
		printf("\n");
	}

	return 0;
}
