#include<iostream>
#include<cstring>

#include<cmath>

using namespace std;
#define INF 0x3f3f3f3f
const int maxn = 210;
int n;
int vis[maxn],num[maxn],now[maxn],mp[maxn][maxn];
int main(){
	cin>>n;
	int k;
	memset(mp,0,sizeof mp);
	memset(vis,0,sizeof vis);
	for (int i=0 ; i<n*(n-1)/2 ; i++){
		cin>>k;
		for (int j=0 ; j<k ; j++){
			cin>>num[j];
			vis[num[j]] = 1;
			for (int l=0 ; l<j ; l++){
				mp[num[l]][num[j]]++;
				mp[num[j]][num[l]]++;
			}
		}
	}
	k = 1;
	int cnt;
	for (int i=1 ; i<=200 ; i++){
		if (vis[i]){
			cnt = 1;
			now[0] = i;
			for (int j=i+1 ; j<=200 ; j++){
				if (vis[j] && mp[i][j]==n-1)
					now[cnt++] = j;
			}
			if (n==2 && k){
				cnt = 1;
				k = 0;
			}
			cout<<cnt<<" ";
			for (int j=0 ; j<cnt ; j++){
				cout<<now[j]<<" ";
				vis[now[j]] = 0;
			}
			cout<<endl;
		}
	}
	return 0;
}
