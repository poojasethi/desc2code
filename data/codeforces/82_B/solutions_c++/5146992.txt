#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    string occ[222];
    int n, k, a;
    cin >> n;
    for (int i=0;i<(n*n-n)/2;++i)
    {
        cin >> k;
        for (int j=0;j<k;++j)
        {
            cin >> a;
            occ[a] += (char)(i+1);
        }
    }
    if (n == 2)
        occ[a] += (char)255;
    vector<pair<string, int> > V;
    for (int i=1;i<=200;++i)
        if (!occ[i].empty())
            V.push_back(make_pair(occ[i], i));
    sort(V.begin(), V.end());
    vector<int> ANS;
    for (int i=0;i<V.size();++i)
    {
        if (i && V[i].first != V[i-1].first)
        {
            cout << ANS.size() << ' ';
            for (int j=0;j<ANS.size();++j)
                cout << ANS[j] << ' ';
            cout << endl;
            ANS.clear();
        }
        ANS.push_back(V[i].second);
    }
    cout << ANS.size() << ' ';
    for (int j=0;j<ANS.size();++j)
        cout << ANS[j] << ' ';
    cout << endl;
    return 0;
}
