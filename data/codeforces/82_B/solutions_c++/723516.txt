#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int n,m,num=0;
	int a[201][201]={},s,e[200]={},ans[200]={};
	bool used[201]={};

	cin>>n;
	m = n*(n-1)/2;

	for(int i = 0 ; i < m ; ++i )
	{
		cin>>s;
		for(int j = 0 ; j < s ; ++j )	cin>>e[j];

		if(n==2)
		{
			cout<<1<<' '<<e[0]<<endl;
			cout<<s-1;
			for(int j = 1 ; j < s ; ++j )        cout<<' '<<e[j];cout<<endl;
			return 0;
		}
		
		for(int j = 0 ; j < s ; ++j )
		{
			for(int k = j ; k < s ; ++k ) 
			{
				a[e[j]][e[k]]++ ; a[e[k]][e[j]]++;
			}
		}
	}
	
	if( n != 2 )
	{
		for(int i = 1 ; i < 201 ; ++i )
		{
			num = 0;
			if(!used[i]  )
			{
//			ans[num++] = i;used[i] = true;
				for(int j = 1 ; j < 201 ; ++j )
				{
					if((a[i][j] == n-1 || a[i][j] == 2*(n-1))&& !used[j] ){	ans[num++] = j ; used[j] = true;}//cout<<"i=="<<i<<" j== "<<j<<' ';	}
					
				}
					
				for(int j = 0 ; j < num ; ++j )
				{
					if(j==0)	cout<<num;
//				else	cout<<num;
					cout<<' '<<ans[j];
				}	if(num)cout<<endl;
			}
		}

		
	}

}
