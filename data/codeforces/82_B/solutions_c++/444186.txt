#include<iostream>
#include<vector>
using namespace std;
int a[201][201],i,j,n,k,v[200],num[201];
vector<int> cs;
int main()
{
	cin>>n;
	for(i=1;i<=n*(n-1)/2;i++){
		cin>>k;
		for(j=1;j<=k;j++){
			cin>>v[j];
			for(int l=1;l<j;l++){a[v[j]][v[l]]++;a[v[l]][v[j]]++;}
			if(num[v[j]]==0)num[v[j]]++;
		}
	}
	if(n!=2){
	for(i=1;i<=200;i++){
		if(num[i]==1){
			cs.clear();
			cs.push_back(i);
			for(j=1;j<=200;j++)
				if(a[i][j]==(n-1)){
					cs.push_back(j);
					num[j]=0;
				}
			cout<<cs.size()<<' ';
			for(j=0;j<cs.size();j++)cout<<cs[j]<<' ';
			cout<<'\n';
		}
	}
	}
	else {
		for(i=1;i<=200;i++)
			if(num[i]==1){cout<<"1 "<<i<<'\n';break;}
		for(j=i+1;j<=200;j++){
			if(num[j]==1)cs.push_back(j);
		}
		cout<<cs.size()<<' ';
		for(i=0;i<cs.size();i++)cout<<cs[i]<<' ';
	}
	return 0;
}
