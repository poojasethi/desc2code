#include <cstdio>
#include <algorithm>
#include <map>
#include <vector>

using namespace std;

pair<int,int> r[220];
map< pair<int,int>, vector<int> > ans;

int main() {
	int n;
	scanf("%d\n",&n);
	if (n==2) {
		int m,a;
		scanf("%d%d",&m,&a);
		printf("1 %d\n%d",a,m-1);
		for(int i=0;i<m-1;i++) {
			int t;
			scanf("%d",&t);
			printf(" %d",t);
		}
		printf("\n");
		return 0;
	}
	for(int i=0;i<n*(n-1)/2;i++) {
		int m;
		scanf("%d",&m);
		for(int j=0;j<m;j++) {
			int t;
			scanf("%d",&t);
			if (r[t].first==0) r[t].first=i+1;
			else if (r[t].second==0) r[t].second=i+1;
		}
	}
	for(int i=1;i<=200;i++) {
		if (!r[i].first) continue;
		ans[r[i]].push_back(i);
	}
	for(map< pair<int,int>, vector<int> >::iterator it=ans.begin();it!=ans.end();++it) {
		vector<int>& v=it->second;
		printf("%lu",v.size());
		for(int i=0;i<v.size();i++) printf(" %d",v[i]);
		printf("\n");
	}
	return 0;
}
