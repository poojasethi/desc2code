#include <cstdio>
#include <algorithm>
#include <cstring>
#define FOR(i,s,e) for (int i=(s); i<(int)(e); i++)
#define FOE(i,s,e) for (int i=(s); i<=(int)(e); i++)
using namespace std;

struct num{
	int x, c;
	int v[205];
	bool operator < (num const &T) const{
		FOR(i,0,200){
			if (v[i]!=T.v[i]) return v[i]<T.v[i];
		}
		return 0;
	}
}   a[205];

int n,x,y,cnt,s[205],f[205];


int main(){

	scanf("%d", &n);
	if (n==2){
		scanf("%d", &x);
		FOR(j,0,x) scanf("%d", &s[j]);
		printf("1 %d\n", s[0]);
		printf("%d", x-1);
		FOR(j,1,x) printf(" %d", s[j]);
		printf("\n");
		return 0;
	}
	
	FOE(i,1,200) a[i].x = i;
	FOE(i,1,n*(n-1)/2){
		scanf("%d", &x);
		while (x--){
			scanf("%d", &y);
			a[y].v[a[y].c++] = i;
		}
	}
	
	sort(a+1, a+201);
	cnt = 0;
	FOE(i,1,200){
		if (!a[i].c) continue;
		if (i==1  || a[i-1]<a[i]) ++cnt;
		s[a[i].x] = cnt;
		f[cnt]++;
	}
	
	FOE(i,1,cnt){
		printf("%d", f[i]);
		FOE(j,1,200)
		    if (s[j]==i) printf(" %d", j);
		printf("\n");
	}


	return 0;
}
