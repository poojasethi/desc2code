#include <iostream>
#include <cstdlib>
#include <vector>
#include <set>

using namespace std;
const int mx=200+100;
int n,c,mark[mx],s[mx*mx][mx],com[mx];


int main()
{
	cin>>n;
	c=n*(n-1)/2;
	for(int i=0;i<c;i++)
	{
		cin>>s[i][0];
		for(int j=1;j<=s[i][0];j++)
		{
			cin>>s[i][j];
		}
	}
	if(n==2)
	{
		cout<<1<<" "<<s[0][1]<<endl;
		cout<<s[0][0]-1<<" ";
		for(int i=2;i<=s[0][0];i++)
		{
			cout<<s[0][i]<<" ";
		}
		cout<<endl;
		return 0;
	}
	int size=1;
	com[0]=0;
	for(int i=1;i<c;i++)
	{
		for(int j=1;j<=s[i][0];j++)
		{
			if(s[i][j]==s[0][1])
			{
				com[size++]=i;
				break;
			}
		}
	}
	int h=0;
	for(int j=1;j<=s[com[0]][0];j++)
	{
		for(int i=1;i<=s[com[1]][0];i++)
		{
			if(s[com[0]][j]==s[com[1]][i])
			{
				mark[s[com[0]][j]]=1;
				h++;

			}
		}
	}
	for(int i=0;i<size;i++)
	{
		cout<<s[com[i]][0]-h<<" ";
		for(int j=1;j<=s[com[i]][0];j++)
		{
			if(!mark[s[com[i]][j]])
			{
				cout<<s[com[i]][j]<<" ";
			}
		}
		cout<<endl;
	}
	cout<<h<<" ";
	for(int i=0;i<mx;i++)
	{
		if(mark[i])
			cout<<i<<" ";
	}
	cout<<endl;
	return 0;
}
