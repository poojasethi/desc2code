#include<iostream>
using namespace std;
int main()
{
	int a[201][201];
	int set[201];
	int n,t,k,i,j,num,p,tag;
	while(cin>>n)
	{
if(n>2)
{
		for(i=0;i<=200;i++)
			a[i][0]=0;

		t=n*(n-1)/2;
		for(i=0;i<t;i++)
		{
			cin>>k;
			for(j=0;j<k;j++)
			{
				cin>>p;
				a[p][0]=a[p][0]+1;
				a[p][a[p][0]]=i;
			}
		}
		for(i=1;i<=200;i++)
		{	
			if(a[i][0]!=0)
			{
			    num=0;
			    set[num++]=i;
				for(j=i+1;j<=200;j++)
				{
					if(a[j][0]!=0)
					{
						for(k=1;k<n;k++)   //每个数出现n-1次
							if(a[i][k]!=a[j][k])
								break;
				        if(k==n)
				        {
							set[num++]=j;
					        a[j][0]=0;
				        }
					}
			    }
			    cout<<num<<" ";
			    for(j=0;j<num;j++)
					cout<<set[j]<<" ";
			    cout<<endl;
			}
		}
}
else
{
	cin>>k;
	cin>>num;
	cout<<1<<" "<<num<<endl;;
	cout<<k-1<<" ";
	for(i=1;i<k;i++)
	{
		cin>>num;
		cout<<num<<" ";
	}
	cout<<endl;
}
	}
	return 0;
}
