#include<stdio.h>
#include<string.h>
int num[205],flag[205],ind[205][20000];
int judge(int x,int y)
{
	int i,j;
	if(num[x]!=num[y])
		return 0;
	for(i=0;i<num[x];i++)
		if(ind[x][i]!=ind[y][i])
			return 0;
	return 1;
}
int main()
{
	int n,i,j,k,a,res[20000];
	while(scanf("%d",&n) != EOF)
	{
		memset(num,0,sizeof(num));
		memset(flag,0,sizeof(flag));
		for(i=0;i<n*(n-1)/2;i++)
		{
			scanf("%d",&k);
			for(j=0;j<k;j++)
			{
				scanf("%d",&a);
				ind[a][num[a]]=i;
				num[a]++;
				flag[a]=1;
			}
		}
		if(n==2)
		{
			for(i=0;i<205;i++)
				if(flag[i])
				{
					printf("1 %d\n",i);
					break;
				}
			printf("%d",k-1);
			for(i=i+1;i<205;i++)
				if(flag[i])
					printf(" %d",i);
			printf("\n");
		}
		else
			for(i=0;i<205;i++)
				if(flag[i])
				{
					k=1;
					res[0]=i;
					for(j=i+1;j<205;j++)
					{
						if(flag[j])
							if(judge(i,j))
							{
								res[k++]=j;
								flag[j]=0;
							}
					}
					flag[i]=0;
					printf("%d",k);
					for(j=0;j<k;j++)
						printf(" %d",res[j]);
					printf("\n");
				}
	}
	return 0;
}
