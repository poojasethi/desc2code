#include <set>
#include <cstdio>

using namespace std;

set<int> sets[201];

int main(){
	int n;
	scanf("%d",&n);
	int x,tmp,cur=0, good=-1;
	bool found;
	for(int i=0; i<n*(n-1)/2; ++i){
		found=false;
		scanf("%d",&x);
		for(int j=0; j<x; ++j){
			scanf("%d",&tmp);
			if(good==-1) good=tmp;
			if(tmp==good) found=true;
			sets[cur].insert(tmp);
		}
		if(found){
			++cur;
		}else{
			sets[cur].clear();
		}
	}
	set<int> first;
	first.insert(good);
	for(int i=0; i<cur; ++i)sets[i].erase(good);
	set<int>::iterator it=sets[0].begin();
	while(it != sets[0].end()){
		if(sets[1].find(*it)!=sets[1].end()){
			int num=*it;
			first.insert(num);
			for(int j=0; j<cur; ++j){
				sets[j].erase(num);
			}
			it=sets[0].begin();
		}else{
			++it;
		}
	}
	printf("%d ",first.size());
	for(it=first.begin() ; it != first.end(); it++){
		printf("%d ",*it);
	}
	printf("\n");
	for(int i=0; i<cur; ++i){
		printf("%d ",sets[i].size());
		for(it=sets[i].begin() ; it != sets[i].end(); it++){
			printf("%d ",*it);
		}
		printf("\n");
	}
	return 0;
}
