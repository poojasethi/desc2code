#include <vector>
#include <algorithm>
#include <iostream>
using namespace std;

int main() {
	int n, k, a;
	vector<int> z[200];
	vector<int> t[200];
	bool u[200];

	cin >> n;

	if (n == 2) {
		cin >> k;
		cin >> a;
		cout << "1 " << a << "\n";
		cout << k - 1 << " ";
		for (int i = 1; i < k; ++i) {
			cin >> a;
			cout << a << " ";
		}
		return 0;
	}

	for (int i = 0; i < n*(n - 1)/2; ++i) {
		cin >> k;
		for (int j = 0; j < k; ++j) {
			cin >> a;
			z[a - 1].push_back(i);
		}
	}

	fill_n(u, 200, false);

	for (int i = 0; i < 200; ++i)
		if (!u[i] && !z[i].empty())
			for (int j = i; j < 200; ++j)
				if (!u[j] && z[j] == z[i]) {
					t[i].push_back(j);
					u[j] = true;
				}

	for (int i = 0; i < 200; ++i)
		if (!t[i].empty()) {
			cout << t[i].size() << " ";
			for (int j = 0; j < (int)t[i].size(); ++j)
				cout << t[i][j] + 1 << " ";
			cout << "\n";
		}

	return 0;
}
