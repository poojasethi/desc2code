#include <cstdio>
#include <algorithm>
#include <set>

using namespace std;

set<int> a[20100];
int b[210][2]={};

int main()
{
    int n,k,x;
    scanf("%d",&n);
    for (int i=1;i<=n*(n-1)/2;++i)
    {
        scanf("%d",&k);
        for (int j=1;j<=k;++j)
        {
            scanf("%d",&x);
            if (b[x][0])
                b[x][1]=i;
            else
                b[x][0]=i;
            a[i].insert(x);
        }
    }
    if (n==2)
    {
        set<int>::iterator it=a[1].begin();
        printf("1 %d\n%d",*it,a[1].size()-1);
        for (it++;it!=a[1].end();++it)
            printf(" %d",*it);
        printf("\n");
        return 0;
    }
    bool fnd[201]={};
    for (int i=1;i<=200;++i)
    {
        if (fnd[i] || !b[i][0])
            continue;
        int i1 = b[i][0], i2 = b[i][1];
        int res[200], *t = set_intersection(a[i1].begin(),a[i1].end(),a[i2].begin(),a[i2].end(),res);
        printf("%d",t-res);
        for (int *j=res;j<t;++j)
        {
            printf(" %d",*j);
            fnd[*j]=true;
        }
        printf("\n");
    }
    return 0;
}
