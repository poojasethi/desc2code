#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<vector>
#include<iostream>
using namespace std;
vector<int>s[40001],f[201],set[201];
int g[201],z[201],o[201];
int main()
{
	int n,N,i,l,j,k,v,e;
	scanf("%d",&n);
	N=n*(n-1)/2;
	if(n==2)
	{
		scanf("%d",&k);
		scanf("%d",&v);
		printf("1 %d\n",v);
		printf("%d",k-1);
		for(i=1;i<k;i++)
		{
			scanf("%d",&v);
			printf(" %d",v);
		}
		puts("");
		return 0;
	}
	for(i=0;i<N;i++)
	{
		scanf("%d",&k);
		for(j=0;j<k;j++)
		{
			scanf("%d",&v);
			//printf("i:%d v:%d \n",i,v);
			f[v].push_back(i);
			g[v]++;
		}
	}
	k=0;
	for(i=1;i<201;i++)
		if(g[i])
	{
		for(j=1;j<i;j++)
			if(g[j])
		{
			for(l=0;l<g[i];l++)
				if(f[i][l]!=f[j][l])break;
			if(l==g[i])break;
		}
		if(j!=i)
		{
			e=o[j];
			z[e]++;
			set[e].push_back(i);
		}
		else
		{
			o[i]=k;
			z[k]=1;
			set[k].push_back(i);
			k++;
		}
	}
	for(i=0;i<k;i++)
	{
		printf("%d",z[i]);
		for(j=0;j<z[i];j++)
			printf(" %d",set[i][j]);
		puts("");
	}
}
