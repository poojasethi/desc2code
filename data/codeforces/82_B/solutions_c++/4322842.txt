#include <iostream>
#include <set>
#include <map>
#include <string>
#include <cstdio>
#include <vector>
using namespace std;
int n,i,j,a,b,u,l;
map <int,bool> v[100001],z[100001];
std::map<int,bool>::iterator it;
vector <int> x[100001];
bool used[100001];
int main()
{
cin>>n;
for (i=0;i<((n*(n-1))/2);i++)
{
cin>>b;
for (j=0;j<b;j++){
cin>>a;
x[i].push_back(a);
z[i][a]=1;
}}
if (n==2)
	{
	cout<<1<<" "<<x[0][0]<<endl;;
	cout<<x[0].size()-1<<" ";
	for (i=1;i<x[0].size();i++)
	cout<<x[0][i]<<" ";
	return 0;
	}
for (i=1;i<((n*(n-1))/2);i++)
{
	if (z[i].count(x[0][0]))
	{
	for (l=0;l<x[0].size();l++)
	for (j=0;j<x[i].size();j++)
		if (x[0][l]==x[i][j])
		{v[u][x[0][l]]=1; 
		used[x[0][l]]=1;
		}
	break;
	}
}
u++;
for (i=0;i<((n*(n-1))/2);i++)
//if (u<n)
if (z[i].count(x[0][0]))
{
	for (j=0;j<x[i].size();j++)
	if (used[x[i][j]]==0)
	{
	v[u][x[i][j]]=1;
	}
	u++;
}
for (i=0;i<u;i++)
{
cout<<v[i].size()<<" ";
for (it=v[i].begin();it!=v[i].end();it++)
cout<<it->first<<" ";
cout<<endl;
}


return 0;
}
