#include<iostream>
#include<vector>
#include<string>
#include<cmath>
#include<algorithm>
#include<sstream>
#include<stdio.h>
#define fr(i,a,b) for(i=a;i<=b;i++)
using namespace std;
const int maxn=202;
vector<int> g[maxn];
long long code[maxn],saved;
int i,j,n,k,tot,now,tmp;
int main(){
	cin>>n;
	fr(i,1,n*(n-1)/2){
		cin>>k;
		while(k--){
			scanf("%d",&tmp);
			g[tmp].push_back(i);
		}
	}
	tot=0;
	fr(i,1,200){
		if(g[i].empty())
			continue;
		tot++;
		code[i]=0;
		fr(j,0,(int)g[i].size()-1)
			code[i]=code[i]*103861+g[i][j];
	}
	now=0;
	fr(i,1,200){
		if(g[i].empty()||code[i]==0)
			continue;
		vector<int> list;
		saved=code[i];
		now++;
		fr(j,i,200)
			if(code[j]==saved&&tot>n-now){
				list.push_back(j);
				tot--;
				code[j]=0;
			}
		cout<<list.size();
		fr(j,0,(int)list.size()-1)
			printf(" %d",list[j]);
		cout<<endl;
	}
}
