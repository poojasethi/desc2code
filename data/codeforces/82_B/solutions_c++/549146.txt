#include<cstdio> 
#include<algorithm> 
#include<vector> 
#include<map> 
using namespace std; 
typedef vector<int> cnt; 
map<int,cnt> mp; 
int main()
{   
    int n,k,x;   
    scanf("%d",&n);   
    for(int i=0;i<n;i++)
        for(int j=0;j<i;j++)
        {    
            scanf("%d",&k);     
            while(k--)
            {       
                scanf("%d",&x);       
                mp[x].push_back(i*n+j);     
            }   
        }   
    map<cnt,cnt> t;   
    for(map<int,cnt>::iterator iter=mp.begin();
    iter!=mp.end();iter++)t[iter->second].push_back(iter->first);   
    cnt a[210];   
    int next=0;   
    for(map<cnt,cnt>::iterator iter=t.begin();iter!=t.end();iter++)
    {     
        a[next++]=iter->second;   
    }   
    if(!a[1].size())
    {     
        a[1].push_back(a[0].back());      
        a[0].pop_back();   
    }   
    for(int i=0;i<n;i++)
    {     
        printf("%d",a[i].size());     
        for(int j=0;j<a[i].size();j++)
            printf(" %d",a[i][j]);     
        puts("");    
    } 
    return 0;
}
