#include <iostream>
#include <vector>
#include <cstdio>

using namespace std;

vector<vector<int> > a;
int n, k;
vector<int> use, v;

int main()
{
	a.assign(201, vector<int> (201, 0));
	scanf("%d", &n);
	use.assign(201, 0);
	v.assign(201, 0);
	for (int i = 0; i < n*(n-1)/2; i++)
	{
		scanf("%d", &k);
		for (int j = 0; j < k; j++)
		{
			scanf("%d", &v[j]);
			use[v[j]] = 1;
		}
		for (int j = k-1; j > 0; j--)
			for (int j1 = j-1; j1 >= 0; j1--)
			{
				a[v[j]][v[j1]]++;
				a[v[j1]][v[j]]++;
			}
	}
	if (n == 2)
	{
		printf("1 %d\n", v[0]);
		printf("%d", k-1);
		for (int i = 1; i < k; i++)
			printf(" %d", v[i]);
		printf("\n");
		return 0;
	}
	for (int i = 1; i <= 200; i++)
	{
		if (use[i] == 0)
			continue;
		int kol = 1;
		for (int j = i+1; j <= 200; j++)
			if (a[i][j] > 1)
				kol++;
		printf("%d %d", kol, i);
		use[i] = 0;
		for (int j = i+1; j <= 200; j++)
			if (a[i][j] > 1)
			{
				use[j] = 0;
				printf(" %d", j);
			}
		printf("\n");
	}
	return 0;

}
