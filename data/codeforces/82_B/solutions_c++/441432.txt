#include<iostream>
#include<memory.h>
using namespace std;
int main()
{
int arr[220][220],exists[220],tmp[220];
int T,F;

cin>>T;
int N=T;
F=(T*(T-1))/2;
memset(arr,0,sizeof(arr));
memset(exists,0,sizeof(exists));
for(int i=0; i < F ;i++)
{
	cin>>T;
	for(int j=0; j< T; j++)
	{
		cin>>tmp[j];
		exists[tmp[j]]++;
	}
	
	for(int k=0; k < T; k++)
		for(int	j=0; j < T; j++)
			arr[tmp[k]][tmp[j]]++;
}
if(N==2){
	cout<<"1 "<<tmp[0]<<"\n";
	cout<<T-1;
	for(int i=1; i < T; i++) cout<<" "<<tmp[i];
	cout<<"\n";
	return 0;
}
	bool visited[220];
	memset(visited,false,sizeof(visited));
	for(int i=0; i < 220; i++)
	{
		if(exists[i]>0 && visited[i]==false)	
		{
			int k=0;
			for(int j=0; j < 220; j++)
				if(arr[i][j]==N-1 && visited[j]==false){
				visited[j]=true;
				tmp[k++]=j;	
				}
			cout<<k;
			for(int i=0; i < k; i++)
				cout<<" "<<tmp[i];
			cout<<"\n";
		}
	}
return 0;
}
