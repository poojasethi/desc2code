#include <cstdio>

#define mx 202
int v[200 * 200][mx], a[mx];

void add (int *a, int *b, int r) {
  for (int i = 0; i < mx; i++) {
    a[i] += b[i] * r;
  }
}

void out (int *a) {
  int n = 0;
  for (int i = 0; i < mx; i++) {
    n += a[i];
  }
  printf ("%d", n);
  for (int i = 0; i < mx; i++) {
    if (a[i]) {
      printf (" %d", i);
    }
  }
  puts("");
}

int main (void) {
  int n;
  scanf ("%d", &n);
  int nn = n * (n - 1) / 2, t, x;
  for (int i = 0; i < nn; i++) {
    scanf ("%d", &t);
    while (t--) {
      scanf ("%d", &x);
      v[i][x] = 1;
    }
  }

  for (int i = 0; i < nn; i++) {
    if (v[i][x]) {
      add(a, v[i], +1);
    }
  }
  for (int i = 0; i < mx; i++) {
    a[i] = a[i] == n - 1;
  }
  if (n == 2) {
    a[x] = 0;
  }
  out (a);
  for (int i = 0; i < nn; i++) {
    if (v[i][x]) {
      add (v[i], a, -1);
      out (v[i]);
    }
  }
  return 0;
}
