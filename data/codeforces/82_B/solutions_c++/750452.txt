#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;
const int N=205;
int v[N],c[N],a[N][N],res[N];

int main()
{
    int n,m,i,j,x,k,s=200,t=0;
    scanf("%d",&n);
    for (i=1; i<=n*(n-1)/2; i++)
    {
        scanf("%d",&m);
        memset(v,0,sizeof(v));
        for (j=1; j<=m; j++)
        {
            scanf("%d",&x);
            c[x]=1; v[x]=1;
        }
        for (j=1; j<=s; j++)
         for (k=1; k<=s; k++)
          if (v[j]!=v[k])
           a[j][k]=1;
    }
    memset(v,0,sizeof(v));
    for (i=1; i<=s; i++)
     if (v[i]==0 && c[i]==1)
     {
         m=0;
         for (j=1; j<=s; j++)
          if (c[j]==1 && a[i][j]==0 && v[j]==0)
           {v[j]=1; res[++m]=j; if (n==2&&m==1&&t==0) {t=1; break;}}
         printf("%d ",m);
         for (j=1; j<=m; j++) printf("%d ",res[j]);
         printf("\n");
     }
}
