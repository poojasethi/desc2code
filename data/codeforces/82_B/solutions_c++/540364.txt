#include <stdio.h>

int num[207][7];
int visit[207];
int res[207], resc;

int main(void)
{
	int i, j, n, m, k;

	scanf("%d", &n);
	
	for(i = 1; i <= n*(n-1)/2; i++)
	{
		scanf("%d", &m);
		for(j = 0; j < m; j++)
		{
			scanf("%d", &k);
			if(num[k][0] < 2) 
			{
				num[k][0]++;
				num[k][num[k][0]] = i;
			}
		}
	}

	if(n == 2)
	{
		for(i = 1; i <= 200; i++)
		{
			if(num[i][0]) res[++resc] = i;
		}
		printf("1 %d\n", res[1]);

		printf("%d", resc-1);

		for(i = 2; i <= resc; i++)
			printf(" %d", res[i]);
		
		printf("\n");

		return 0;
	}

	for(i = 1; i <= 200; i++)
	{
		if(visit[i] || num[i][0] == 0) continue;
		resc = 0;
		visit[i] = 1;
		res[++resc] = i;
		for(j = i+1; j <= 200; j++)
		{
			if(num[i][1] == num[j][1] && num[i][2] == num[j][2])
			{
				res[++resc] = j;
				visit[j] = 1;
			}
		}
		printf("%d", resc);
		for(j = 1; j <= resc; j++)
			printf(" %d", res[j]);
		printf("\n");
	}

	return 0;
}
