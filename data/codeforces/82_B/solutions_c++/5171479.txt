#include <iostream>
#include <cstdio>
using namespace std;
#define N 222
int m[N], o[N], a[N][N], n, k;
bool vis[N];
void solve()
{
    scanf("%d", &n);
    for (int i=0; i<n*(n-1)/2; i++)
    {
        scanf("%d", &k);
        for (int j=0; j<k; j++)
        {
            scanf("%d", &m[j]);
            vis[m[j]] = 1;
            for (int k=0; k<j; k++)
            {
                a[m[j]][m[k]]++;
                a[m[k]][m[j]]++;
            }
        }
    }
    k = 1;
    for (int i=1; i<=200; i++)
    {
        if (vis[i])
        {
            int cnt = 0;
            o[cnt++] = i;
            for (int j=i+1; j<=200; j++)
            {
                if (vis[j] && a[i][j] == n-1)
                {
                    o[cnt++] = j;
                }
            }
            if (n == 2 && k) cnt = 1, k = 0;
            printf("%d", cnt);
            for (int j=0; j<cnt; j++)
            {
                printf(" %d", o[j]);
                vis[o[j]] = 0;
            }
            printf("\n");
        }
    }
}

int main()
{
    solve();
    return 0;
}
