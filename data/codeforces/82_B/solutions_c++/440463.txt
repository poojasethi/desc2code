#include <iostream>
#include <vector>
#include <cstdlib>
#include <algorithm>
#include <cstdio>
using namespace std;
int v[210][210],cnt[210];
//int cnt[210],d[210];
bool eq(int i,int j){
	if (cnt[j]!=cnt[i]) return 0;
	for (int k=0;k<cnt[i];k++){
		if (v[i][k]<v[j][k]) return 0;
		if (v[i][k]>v[j][k]) return 0;
	}
	return 1;
}
int main(){
	//freopen("input.txt","r",stdin);
	int n,m;
	int tmp,zz;
	cin>>m;
	n=m*(m-1)/2;
	for (int i=0;i<n;i++){
		cin>>zz;
		for (int j=0;j<zz;j++){
			cin>>tmp;
			v[tmp][cnt[tmp]++]=i;
		}
	}
	for (int i=1;i<201;i++){
		if (cnt[i]==0) continue;
		int count=0;
		for (int j=i;j<201;j++){
			if (eq(i,j)) {
				count++;
			}
		}
		if (m==2){
			cout<<1<<' '<<i<<endl;
			cout<<count-1<<' ';
			for (int j=i+1;j<201;j++){
				if (eq(i,j)) {
					cout<<j<<' ';
				}
			}
			break;
		}
		cout<<count<<' ';
		for (int j=i;j<201;j++){
			if (eq(i,j)) {
				cout<<j<<' ';
				if (i!=j) cnt[j]=0;
			}
		}
		cout<<endl;
	}
	return 0;
}
