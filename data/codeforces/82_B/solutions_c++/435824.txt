#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
bool b[210],bb[210];
int m,n,g[210][210],a[210];

int main()
{
	while (scanf("%d",&n)!=EOF)
	{
		m=0;
		memset(bb,false,sizeof(bb));
		memset(g,0,sizeof(g));
		int k;
		for (int i=1;i<=n*(n-1)/2;i++)
		{
			scanf("%d",&k);
			for (int j=1;j<=k;j++)
			{
				scanf("%d",&a[j]);
				if (m<a[j]) m=a[j];
				bb[a[j]]=true;
			}
			for (int j1=1;j1<=k;j1++)
				for (int j2=j1+1;j2<=k;j2++)
				{
					g[a[j1]][a[j2]]++;
					g[a[j2]][a[j1]]++;
				}
		}
		if (n==2)
		{
			printf("%d ",k-1);
			for (int i=1;i<k-1;i++)
				printf("%d ",a[i]);
			printf("%d\n",a[k-1]);
			printf("1 %d\n",a[k]);
			continue;
		}
		memset(b,true,sizeof(b));
		for (int i=1;i<=m;i++)
			if (b[i] && bb[i])
			{
				k=1;
				a[1]=i;
				b[i]=false;
				for (int j=1;j<=m;j++)
					if (g[i][j]>1)
					{
						k++;
						a[k]=j;
						b[j]=false;
					}
				printf("%d ",k);
				for (int i=1;i<k;i++)
					printf("%d ",a[i]);
				printf("%d\n",a[k]);
			}
	}
	return 0;
}
