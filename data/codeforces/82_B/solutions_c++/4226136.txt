#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int g[205][205],vis[205];
int num[205];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n*(n-1)/2;++i)
    {
        int k=0;
        scanf("%d",&k);
        for(int j=0;j<k;++j)
        {
            scanf("%d",&num[j]);
            vis[num[j]]=1;
        }
        for(int j=0;j<k;++j)
            for(int jj=0;jj<k;++jj)
            ++g[num[j]][num[jj]];
        if(n==2)
        {
            printf("1 %d\n",num[0]);
            printf("%d",k-1);
            for(int i=1;i<k;++i)
                printf(" %d",num[i]);
            printf("\n");
            return 0;
        }
    }
    for(int i=1;i<=200;++i)
    if(vis[i])
    {
        int k=1;
        num[0]=i;
        for(int j=i+1;j<=200;++j)
            if(g[i][j]==n-1&&vis[j])
            num[k++]=j;
        printf("%d",k);
        for(int j=0;j<k;++j)
            printf(" %d",num[j]),vis[num[j]]=0;
        printf("\n");
    }
}
