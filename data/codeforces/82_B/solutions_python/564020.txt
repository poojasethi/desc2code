n = int(raw_input())
a = []
for i in range(n * (n - 1) / 2):
    a.append(set(map(int, raw_input().split()[1:])))

res = []
if n == 2:
    res.append(list(a[0])[0:1])
    res.append(list(a[0])[1:])
else:
    for k in a[1:]:
        t = a[0] & k
        if t != set([]):
            res.append(t)
            break
    for k in a:
        if res[0] < k:
            res.append(k - res[0])

for k in res:
    print len(k), ' '.join(map(str, k))
