n = input()
x = []
if n!=2:
    for i in xrange(n*(n-1)/2):
        x.append(set(map(int,raw_input().split())[1:]))
    y = []
    for i in x[1:]:
        t = i & x[0]
        if t!=set([]) and not t in y:
            y.append(t)
    for i in x:
        if y[0] <= i:
            t = i-y[0]
            if not t in y:
                y.append(t)
else:
    x = map(int,raw_input().split()[1:])
    y = [[x[0]],x[1:]]
for i in y:
    print len(i), ' '.join(map(str,i))
