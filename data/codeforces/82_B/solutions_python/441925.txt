from sys import stdin
from collections import defaultdict

inLine = defaultdict(list)
N = int(stdin.next())
if N == 2:
	p = stdin.next().split()[1:]
	print 1, p[0]
	print len(p[1:]), ' '.join(p[1:])
	exit()

for n,l in enumerate(stdin):
	for v in map(int, l.split())[1:]:
		inLine[v].append(n)

toSet = defaultdict(list)

for v,l in inLine.items():
	k = tuple(l)
	toSet[k].append(v)

for s in toSet.values():
	print len(s), ' '.join(map(str, s))
