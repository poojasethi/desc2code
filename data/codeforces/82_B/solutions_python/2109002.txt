n=input()
if n == 2:
    c = raw_input().split()

    print len(c[2:]), ' '.join(c[2:])

k=[[] for _ in range(201)]
for i  in range(n*(n-1)/2):
    c = map(int, raw_input().split())
    for e in c[1:]:
        k[e] += [i]


        
d={}
for i in range(201):
    l = k[i]
    if l:
        l.sort()
        e=tuple(l)
        if e in d:
            d[e] += [i]
        else:
            d[e] = [i]

for b in d.values():
    print len(b), ' '.join(map(str, b))
