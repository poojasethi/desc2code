n = input()
l = [map(int, raw_input().split()) for i in range(0, n*(n-1)/2)]
for i in l: i.pop(0)
ans = filter(lambda x: l[0][0] in x, l)
f = [ans[0].pop(0)]
for j in range(1, len(ans)):
    i = 0
    while(i < len(ans[j])):
        k = f[0]
        if(ans[j][i] in ans[0]): k = ans[j].pop(i)
        else: i += 1
        if not k in f: f.append(k)
ans = [filter(lambda x: not x in f, i) for i in ans] + [f]
