n=input()
v=[[]]*(n*(n-1)/2)
x=[[] for i in range(222)]
t=[]
for i in range(n*(n-1)/2):
	v[i]=map(int,raw_input().split())
	t=v[i][1:]
	v[i]=set(t)
	for j in v[i]:
		x[j].append(v[i])
if n==2:
	print 1, t[0]
	print len(t)-1, ' '.join(map(str,t[1:]))
	exit()
c=[-1]*222
d=set(reduce(lambda x,y:x|y,v))
for i in d:
	if c[i]==-1:
		r=[]
		for j in d:
			if c[j]==-1 and sum(map(lambda x: i in x,x[j]))==n-1:
				c[j]=i
				r.append(j)
		print len(r),' '.join(map(str,r))
