n, b, p, r = input(), [[0] * 201 for _ in range(201)], set(), []
for s in [map(int, raw_input().split())[1 : ] for _ in range(n * (n - 1) / 2)]:
	for x in s:
		for y in s:
			b[x][y] += 1
for i in range(201):
	if not sum(b[i]) or i in p:
		continue
	x = [j for j in range(201) if b[i][j] == n - 1]
	p.update(x)
	r.append(x)
if n == 2:
	r.append([r[0].pop()])
for x in r:
	print len(x), ' '.join(map(str, x))
