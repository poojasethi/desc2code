unionList = []
setList = []
hashSet = set()

n = int(input())
m = n * (n-1) / 2

for i in range(m) :
    unionList.append(set(map(int, raw_input().split()[1:])))

if m == 1 :
    tmp = []
    for item in unionList[0] :
        tmp.append(item)
    print "1 %s" % tmp[0]
    result = str((len(tmp)-1)) + " "
    result += " ".join(map(str, tmp[1:]))
    print result

else :
    for j in range(1, m) :
        interFirst = unionList[0] & unionList[j]
        if len(interFirst) != 0 :
            setList.append(interFirst)
            break
        
    for j in range(0, m) :
        intersection = interFirst & unionList[j]
        # print intersection
        if len(intersection) == 0 :
            continue
        else :
            setList.append(unionList[j] - intersection)
            # hashSet.add(sum(intersection))
                
    for item in setList :
        if item is None or len(item) == 0 :
            continue
        result = "%s " % len(item)
        for elem in item :
            result += "%s " % elem
        print result
