#include <cstdio>
#include <iostream>

using namespace std;

const long mod=1000000007;

long a[16]={0};
long f[16][205]={0};
long C[505][505]={0};

int main()
{
 long n;
 long i,j;
 long k;
 long ans;
 
 scanf("%ld",&n);
 for(i=0;i<=9;i++)
   scanf("%ld",&a[i]);
 for(i=0;i<=500;i++)
   for(j=1,C[i][0]=1;j<=i;j++)
     {
      C[i][j]=(C[i-1][j-1]+C[i-1][j])%mod;
     }
 a[10]=a[0];
 f[0][0]=1;
 for(i=1;i<=10;i++)
   for(j=0;j<=n;j++)
     if(f[i-1][j])
      {
       for(k=a[i];k+j<=n;k++)
         {
          f[i][k+j]+=(long long)f[i-1][j]*C[j+(i!=10)-1+k][k]%mod;
          f[i][k+j]%=mod;
         }
      }
 ans=0;
 for(i=1;i<=n;i++)
   {
    ans+=f[10][i];
    ans%=mod;
   }
  
  printf("%ld\n",ans);
  
 return 0;
}
