#include<bits/stdc++.h>

using namespace std;

#define LL long long
#define mod 1000000007

int a[10];
LL ncr[101][101],dp[101][10];

int main()
{
    int n,i,j,k;
    cin>>n;
    for(i=0;i<10;i++)
        cin>>a[i];

    for(i=0;i<=100;i++)
    {
        ncr[i][0] = 1;
        for(j=1;j<=i;j++)
            ncr[i][j] = (ncr[i-1][j-1]+ncr[i-1][j])%mod;
    }

    for(i=a[9];i<=n;i++)
        dp[i][9] = 1;

    for(j=8;j>0;j--)
        for(i=0;i<=n;i++)
            for(k=a[j];k<=i;k++)
                dp[i][j]=(dp[i][j]+dp[i-k][j+1]*ncr[i][k])%mod;

    LL ans = 0;
    for(i=a[0];i<=n;i++)
        for(k=a[0];k<i;k++)
            ans=(ans+dp[i-k][1]*ncr[i-1][k])%mod;
    cout<<ans;
    return 0;
}
