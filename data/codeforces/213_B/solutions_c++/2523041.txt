#include<iostream>
using namespace std;
const int M = 1e9+7;
long long dp[101][10];
int nck[101][101];
int main(){
	int n;
	int a[10];
	cin>>n;
	for(int i=0;i<10;i++) cin>>a[i];
	nck[0][0]=1;
	for(int i=0;i<101;i++){
		for(int j=0;j<=i;j++){
			(nck[i+1][j]+=nck[i][j])%=M;
			(nck[i+1][j+1]+=nck[i][j])%=M;
		}
	}

	for(int i=0;i<=n;i++){
		if(a[9]<=i)
			dp[i][9]=1;
		else
			dp[i][9]=0;
		for(int j=8;j>=0;j--){
			long long sum=0;
			for(int k=a[j];k<=i;k++){
				if(j==0)
					(sum+=1L*dp[i-k][j+1]*nck[i-1][k])%=M;
				else
					(sum+=1L*dp[i-k][j+1]*nck[i][k])%=M;
			}
			dp[i][j]=sum;
		}
	}
	int soln=0;
	/*for(int i=0;i<=n;i++){
		for(int j=0;j<10;j++)cout<<dp[i][j]<<" ";
		cout<<endl;
	}*/
	for(int i=1;i<=n;i++)(soln+=dp[i][0])%=M;
	cout<<soln<<endl;

	return 0;
}
