#include<cstdio>

long long C[210][210];
long long ans[10][210];
const long long mod=1e9+7;

int main()
{
	C[0][0]=1;
	for(int i=1;i<=200;i++)
	{
		C[i][0]=1;
		for(int j=1;j<=i;j++)
			C[i][j]=(C[i-1][j-1]+C[i-1][j])%mod;
	}
	int n,a[10];
	scanf("%d",&n);
	for(int i=0;i<10;i++)
		scanf("%d",a+i);
	for(int i=a[9];i<=n;i++)
		ans[9][i]=1;
	for(int i=8;i>0;i--)
	{
		if(a[i]==0)
			ans[i][0]=ans[i+1][0];
		for(int j=1;j<=n;j++)
			for(int k=a[i];k<=j;k++)
				ans[i][j]=(ans[i][j]+ans[i+1][j-k]*C[j][k])%mod;
	}
	for(int j=2;j<=n;j++)
		for(int k=a[0];k<=j;k++)
			ans[0][j]=(ans[0][j]+ans[1][j-k]*C[j-1][k])%mod;
	long long ret=0;
	if(a[0]==0)
		ret+=ans[1][1];
	for(int i=2;i<=n;i++)
		ret=(ret+ans[0][i])%mod;
	printf("%I64d\n",ret);
	return 0;
}
