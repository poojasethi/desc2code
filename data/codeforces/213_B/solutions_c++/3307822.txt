#include <cstdio>
#include <memory.h>

#define maxn 110
#define mod 1000000007

long long c[maxn][maxn],f[maxn][maxn],ret;
int n,i,a[maxn],j,k;

void upd(long long & a,long long b){
	a+=b;
	if(a>=mod)a-=mod;
}

int main (int argc, char * const argv[]) {
	scanf("%d",&n);
	for(i=1;i<=10;i++)scanf("%d",&a[i]);
	if(n==1&&a[1]==1){
		puts("1");
		return 0;
	}
	c[0][0]=1;
	for(i=1;i<=n;i++){
		c[i][0]=1;
		for(j=1;j<=i;j++)c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod;
	}
	while(n){
		memset(f,0,sizeof(f));
		f[0][0]=1;
		for(i=0;i<10;i++)for(j=0;j<=n;j++)for(k=a[i+1];k<=n;k++)if(j+k+(i==0)<=n)
			upd(f[i+1][j+k],(f[i][j]*1LL*c[n-j-(i==0)][k])%mod);
		upd(ret,f[10][n]);
		--n;
	}
	printf("%d\n",(int)ret);
    return 0;
}
