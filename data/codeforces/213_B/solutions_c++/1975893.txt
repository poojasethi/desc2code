#include<stdio.h>
#include<string.h>
#define MOD  1000000007
long long c[107][107],dp[12][105],sum;
int a[10];
int main()
{
    int n,i,j,k;
    for(i=0;i<=104;i++)
    {
        c[0][i]=c[i][i]=1;
     for(j=1;j<i;j++)
     c[j][i]=(c[j-1][i-1]+c[j][i-1])%MOD;
    }
    while(scanf("%d",&n)!=EOF)
    {
        for(i=0;i<=9;i++)
        scanf("%d",&a[i]);
        memset(dp,0,sizeof(dp));
        dp[10][0]=1;
        for(i=9;i>=0;i--)
          for(j=n;j>=0;j--)
             for(k=j-a[i];k>=0;k--)
           if(i)dp[i][j]=(dp[i][j]+dp[i+1][k]*c[k][j])%MOD;
           else if(j&&k)dp[i][j]=(dp[i][j]+dp[i+1][k]*c[k-1][j-1])%MOD;
              sum=0;
              for(i=0;i<=n;i++)
              sum=(sum+dp[0][i])%MOD;
              printf("%I64d\n",sum%MOD);
    }
    return 0;
}
