#include<cstdio>
#include<cstring>
#define mem(a,b) memset(a,b,sizeof(a))
using namespace std;
typedef long long ll;

const int N = 110,P = 1000000007;

int main()
{
	int C[N][N]; mem(C,0);
	C[0][0]=1;
	for(int i=1;i<N;i++) for(int j=0;j<=i;j++) {
		C[i][j]=C[i-1][j]+(j?C[i-1][j-1]:0);
		C[i][j]%=P;
	}

	int n; scanf("%d",&n);
	int a[10];
	for(int i=0;i<10;i++) scanf("%d",&a[i]);

	int dp[11][N]; mem(dp,0);
	dp[10][0]=1;
	for(int i=9;i>=0;i--) {
		for(int j=(i?0:1);j<N;j++) {
			for(int k=a[i];k<=j;k++) {
				dp[i][j]+=((ll)dp[i+1][j-k]*C[i?j:(j-1)][k])%P;
				dp[i][j]%=P;
			}
		}
	}
	int ans=0;
	for(int i=1;i<=n;i++) {
		ans+=dp[0][i];
		ans%=P;
	}
	printf("%d\n",ans);

	return 0;
}
