#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

typedef long long Lint;

const int MOD = 1e9+7;

int N;
int ar[123];
Lint res;
Lint dp[11][105];
Lint dn[105][105];

Lint comb(int n,int r){
	if( r==0 || n == r )	return 1;
	if( r==1 || r == n-1 )	return n;
	if( dn[n][r] )	return dn[n][r];
	return dn[n][r] = (comb(n-1,r)+comb(n-1,r-1))%MOD;
}

Lint f( int pl, int k )
{
	Lint &rev = dp[pl][k];
	
	if( rev != -1 )
		return rev;
	
	if( pl == 10 )
		return k==0;
	
	rev=0;
	
	for( int i=ar[pl] ; i<=k-!pl ; i++ )
		rev = ( rev+f(pl+1,k-i)*comb(k-!pl,i) )%MOD;

	return rev;
}

int main()
{
	cin >> N;
	
	for( int i=0 ; i<10 ; i++ )
		scanf(" %d",&ar[i]);

	memset(dp,-1,sizeof dp);	
	for( int i=1 ; i<=N ; i++ )
		res = ( res+f(0,i) )%MOD;

	cout << res << endl ;
	
	return 0;
}
