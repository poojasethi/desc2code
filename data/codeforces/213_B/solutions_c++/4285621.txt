#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#define MOD 1000000007

using namespace std;

typedef long long ll;

ll N,num[10],pos,res;
ll dp[15][105],c[105][105];

ll bul(ll x,ll y);
ll solve(ll x,ll n)
{
	if(n < 0) return 0;
	if(x == 10) return (!n);
	if(dp[x][n] != -1) return dp[x][n];

	ll i,ret=0;

	for(i=num[x];i<=n-(!x);i++)
	{
		ret += solve(x+1,n-i) * bul(n-(!x),i);
		ret %= MOD;
	}

	return dp[x][n] = ret;
}

int main()
{
	ll i,j;

	cin >> N;
	for(i=0;i<10;i++)
		cin >> num[i];

	memset(dp,-1,sizeof(dp));
	for(i=1;i<=N;i++)
	{
		res += solve(0,i);
		res %= MOD;
	}
	
	cout << res << endl;

	return 0;
}

ll bul(ll x,ll y)
{
	if(x == y || !y) return 1;
	if(c[x][y]) return c[x][y];

	return c[x][y] = (bul(x-1,y-1) + bul(x-1,y)) % MOD;
}
