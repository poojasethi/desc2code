#include <cstdio>
typedef long long i64;
const int MOD=1e9 + 7;
int a[10];
int comb[101][101];
int dp[101][10];
int main() {
	for(int i=0; i<=100; i++)comb[i][0]=1;
	for(int i=1; i<=100; i++)
		for(int j=1; j<=i; j++)
			comb[i][j]=(comb[i-1][j]+comb[i-1][j-1])%MOD;
	int n; scanf("%d", &n);
	for(int i=0; i<10; i++)scanf("%d", &a[i]);
	for(int len=0; len<=n; len++){
		if(len>=a[9])dp[len][9]=1;
		for(int j=8; j>=1; j--){
			for(int i=a[j]; i<=len; i++)
					dp[len][j]=(dp[len][j]+(i64)dp[len-i][j+1]*comb[len][i])%MOD;
		}
		for(int i=a[0]; i<len; i++)
			dp[len][0]=(dp[len][0]+(i64)dp[len-i][1]*comb[len-1][i])%MOD;
	}
	int ans=0;
	for(int i=0; i<=n; i++)ans=(ans+dp[i][0])%MOD;
	printf("%d\n", ans);
}
