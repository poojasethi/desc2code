#include <iostream>

using namespace std;

#define MAXN 105
#define MOD 1000000007

__int64 n;
__int64 dp[MAXN][10],c[MAXN][MAXN];
__int64 a[10];

int main()
{
	__int64 i,j,k;
	cin>>n;
	for (i=0;i<=n;i++)
	{
		c[i][0]=1;
		for (j=1;j<=i;j++)
		{
			c[i][j]=(c[i-1][j-1]+c[i-1][j])%MOD;
		}
	}
	for (i=0;i<10;i++) cin>>a[i];
	
	for (i=0;i<=n;i++)
	{
		if (i>=a[9]) dp[i][9]=1;
		else dp[i][9]=0;
		
		for (j=8;j>=1;j--)
		{
			for (k=a[j];k<=i;k++)
			{
				dp[i][j]=(dp[i][j]+dp[i-k][j+1]*c[i][k])%MOD;
			}
		}
		
		for (k=a[0];k<=i;k++)
		{
			dp[i][0]=(dp[i][0]+dp[i-k][1]*c[i-1][k])%MOD;
		}
	}
	__int64 ans=0;
	for (i=1;i<=n;i++)
	{
		ans=(ans+dp[i][0])%MOD;
	}
	cout<<ans<<endl;
	return 0;
}
