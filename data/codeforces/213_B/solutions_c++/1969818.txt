#include<stdio.h>
#include<string.h>
#include<algorithm>
typedef __int64 lld;
using namespace std;
const lld MOD=1000000007;
lld c[105][105];
lld dp[2][105];
int a[10];
int main()
{
	int i,j,k;
	c[0][0]=1;
	for(i=1;i<105;i++)
	{
		c[i][0]=c[i][i]=1;
		for(j=1;j<i;j++)
			c[i][j]=(c[i-1][j-1]+c[i-1][j])%MOD;
	}
	int n;
	scanf("%d",&n);
	for(i=0;i<10;i++)
		scanf("%d",&a[i]);
	dp[0][0]=1;
	for(i=1;i<=10;i++)
	{
		int p=i&1,q=1-p;
		int d=i%10;
		memset(dp[p],0,sizeof(dp[p]));
		for(j=n;j>=0;j--)
			for(k=a[d];k+j<=n;k++)
				dp[p][k+j]=(dp[p][k+j]+dp[q][j]*c[d?(j+k):(j+k-1)][k])%MOD;
	}
	lld ans=0;
	for(i=1;i<=n;i++)
		ans+=dp[0][i];
	printf("%I64d\n",ans%MOD);
	return 0;
}
