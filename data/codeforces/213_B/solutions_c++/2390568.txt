#include <iostream>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)
#define rp(i,a,b)  for(int i=a;i<b;i++)

using namespace std;

int n,a[11];
long long res,c[120][120],dp[12][120];
int MOD = 1e9+7;


int main(){
  cin>>n;
  rep(i,10)cin>>a[i];
  rep(i,101){
    c[i][0] = 1;
    rp(j,1,i+1)c[i][j] =(c[i-1][j]+c[i-1][j-1])%MOD;
  }

  dp[10][0] = 1;
  
  for(int i = 9;i>=1;i--){
    dp[i][0] = dp[i+1][0]*(a[i]==0);
    rp(j,1,n+1){
      rp(k,a[i],n+1){
    if(j-k<0 || dp[i+1][j-k] == 0)continue;
    dp[i][j] = (dp[i][j]+dp[i+1][j-k]*c[j][k])%MOD;
      }
    }
  }
  

  rp(i,1,n+1){
    rp(k,a[0],n+1){
      if(i-k<=0 || dp[1][i-k] == 0)continue;
      dp[0][i] = (dp[0][i]+dp[1][i-k]*c[i-1][k])%MOD;
    }
  }

  rep(i,n+1)res =(res + dp[0][i])%MOD;

  cout<<res<<endl;


}
