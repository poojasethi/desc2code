#include <vector>
#include <iostream>
using namespace std;
int n,i,j,a[11],m=1e9+7,ans=0;
long long c[211][211],F[211][11];
int f(int n, int t)
{
	if (t==9)
		return 1;
	if (F[n][t]!=-1) return F[n][t];
	int s=0,i;
	long long ans1=0;
	for(i=t;i<10;i++)
		s+=a[i];
	for(i=a[t];i+s-a[t]<=n;i++)
		if (t==0)
			ans1+=(c[n-1][i]*f(n-i,t+1))%m;
		else ans1+=(c[n][i]*f(n-i,t+1))%m;
	ans1%=m;
	F[n][t]=ans1;
	return ans1;
}
int main ()
{
	cin>>n; int dn=n;
	for(i=0;i<10;i++)
	{
		cin>>a[i];
		dn-=a[i];
	}
	for(i=0;i<=100;i++)
	{
		c[i][0]=1; c[i][i]=1;
		for(j=1;j<i;j++)
			c[i][j]=(c[i-1][j-1]+c[i-1][j])%m;
	}
	for(i=0;i<=n;i++)
		for(j=0;j<10;j++)
			F[i][j]=-1;
	for(i=1;i<=n;i++)
		ans=(ans+f(i,0))%m;
	cout<<ans;
	return 0;
}
