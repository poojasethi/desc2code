#include<iostream>
using namespace std;
const int mod = 1e9 + 7;
long long dp[101][11];
int a[11];
int n;
long long Cr[101][101];
void preprocess()
{
  Cr[0][0] = 1;
  for (int i = 1; i <= 100; i++) {
    Cr[i][0] = Cr[i][i] = 1;
    for (int j = 1; j < i; j++)
      Cr[i][j] = (Cr[i-1][j-1] + Cr[i-1][j])%mod;
  }
}
int main()
{
  preprocess();
  cin>>n;
  for (int i = 0; i <= 9; i++) cin>>a[i];

  for (int len = a[9]; len <= n; len++) dp[len][9] = 1;
  
  for (int len = 0; len <= n; len++)
    for (int i = 8; i >= 0; i--)
      for (int j = a[i]; j <= len; j++)
	dp[len][i] = (dp[len][i] + dp[len-j][i+1]*Cr[len - !i][j])%mod;

  long long ans = 0;
  for (int len = 1; len <= n; len++) ans = (ans + dp[len][0])%mod;
  cout<<ans<<endl;
  return 0;
}
