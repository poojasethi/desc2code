M=10**9+7
n=input()
a=map(int,raw_input().split())
c=[[0]*128 for _ in range(128)]
c[0][0]=1
for p in range(1,128):
  c[p][0]=c[p][p]=1
  for q in range(1,p):
    c[p][q]=(c[p-1][q]+c[p-1][q-1])%M
t=[[0]*128 for _ in range(11)]
t[10][0]=1
for d in range(9,0,-1):
  for i in range(n-a[d]+1):
    for j in range(a[d],n-i+1):
      t[d][i+j]=(t[d][i+j]+t[d+1][i]*c[i+j][j])%M
s=0
for i in range(1,n+1):
  for j in range(a[0],i):
    s=(s+t[1][i-j]*c[i-1][j])%M
print s
