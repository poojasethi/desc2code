n, a, b = map(int, raw_input().split())

s1 = map(int, raw_input().split())
s2 = map(int, raw_input().split())

all = [[] for i in range(n)]

for i in range(a):
    all[s1[i] - 1].append(1)

for i in range(b):
    all[s2[i] - 1].append(2)

ans = ''

for ai in all:
    if (ai == [1]):
        ans += '1 '
    else:
        ans += '2 '

print ans
