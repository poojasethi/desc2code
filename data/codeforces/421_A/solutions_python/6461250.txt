# "Pasha and Hamsters"

n,a,b = raw_input().split(' ')
Arthur = []
Alexander = []
Arthur = [ int(x) for x in raw_input().split(' ')]
Alexander = [ int(x) for x in raw_input().split(' ')]

ans = ''
for i in range(1,int(n)+1):
    flag = False

    for ar in Arthur:
        if i == ar:
            ans = ans + '1 '
            flag = True

    if flag == True:
        continue
    for al in Alexander:
        if i == al:
            ans = ans + '2 '
print ans[:-1]
