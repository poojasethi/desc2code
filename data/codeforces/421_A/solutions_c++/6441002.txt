#include <iostream>
#include <cstdio>
using namespace std;
int used[1010];
int main()
{
	int a,b,c,n;
	scanf(" %d %d %d",&n,&a,&b);
	for(int i=1;i<=a;i++){
		scanf(" %d",&c);
		used[c]=1;
	}
	for(int i=1;i<=b;i++){
		scanf(" %d",&c);
		used[c]=2;
	}
	for(int i=1;i<=n;i++) printf("%d ",used[i]);
	return 0;
}
