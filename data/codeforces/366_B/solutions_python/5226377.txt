m=map(int,raw_input().split())
n=m[0]
k=m[1]
theindices=[]
nums=map(int,raw_input().split())
for i in range(0,n):
	if i%k==0:
		theindices.append(i)
minsum=999999999999999999
mini=0
for start in range(0,n):
	tmpsum=0
	if theindices[len(theindices)-1]+start>n:
		break
	for index in theindices:
		tmpsum+=nums[(index+start)%n]
	if tmpsum<minsum:
		minsum=tmpsum
		mini=start
print mini+1
