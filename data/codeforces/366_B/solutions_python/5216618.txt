n, k = map(int, raw_input().split())
v = map(int, raw_input().split())

poss = n / (n / k)
p = []

for i in range(poss):
    s = v[i]
    at = (i + k) % n

    while at != i:
        s += v[at]
        at += k
        at = at % n

    p.append(s)

r = p[0], 1

for i, t in enumerate(p):
    if p[i] < r[0]:
        r = (p[i], i + 1)

print r[1]
