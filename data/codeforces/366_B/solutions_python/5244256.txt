n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
if k == 1:
    print 1
else:
    j = 0
    mini = 0
    ans = 1
    while j < n:
        mini += a[j]
        j = (j + k)
    for i in range(1, k):
        j = i
        c = 0
        while j < n:
            c += a[j]
            j = (j + k)
        if c < mini:
            mini = c
            ans = i + 1
    print ans
