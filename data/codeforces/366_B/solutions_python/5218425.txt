import sys

#sys.stdin = open("1.txt")

n,k = [int(x) for x in raw_input().split()]
tasks = [int(x) for x in raw_input().split()]

off = [ 0 for _ in range(0,k)]

for t_id in range(0,n):
    off[(t_id + 1) % k] += tasks[t_id]

min_off = sys.maxint
min_id = -1000

for t_st in range(1,k + 1):
    if (min_off > off[t_st % k]) or (min_id == -1000):
        min_off = off[t_st % k]
        min_id = t_st % k

if min_id == 0:
    min_id = k

print min_id
