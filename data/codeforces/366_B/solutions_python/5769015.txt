#!/usr/bin/env python

def get_min_power(n, k, a):
    mini = 99999999999
    ind = -1
    for i in range(k):
        s = sum(a[i::k]) 
        if s < mini:
            mini = s 
            ind = i + 1
    return ind

if __name__ == '__main__':
    n, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    print get_min_power(n, k, a)
