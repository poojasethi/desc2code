n, k = map(int, raw_input().split())
ks = map(int, raw_input().split())

mc = float("inf")
ind = 0

for i in xrange(k):
    c = 0
    n = i
    while n < len(ks):
        c += ks[n]
        n += k
    if c < mc:
        mc = c
        ind = i+1

print ind
