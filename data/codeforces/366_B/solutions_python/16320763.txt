n,k =map(int, raw_input().split())
A = map(int, raw_input().split())
S = [0] * (k)
for i in range(n):
    S[i%k] += A[i]

min_scoldings = min(S)
found = False

i = 0
while not found:
    count = n/k
    scoldings = 0
    j = i
    while count > 0:
        scoldings += A[j%n]
        j += k
        count -= 1

    if scoldings == min_scoldings:
        print i+1
        found = True

    i += 1
