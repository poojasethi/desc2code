###Dima and To-do List: 366B http://codeforces.com/problemset/problem/366/B

def run():
    n, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    
    costs = [0] * k
    for i in range(n):
        costs[i%k] += a[i]

    minIdx = costs.index(min(costs))
    print minIdx + 1
#################################################################
run()
