import sys
ri = lambda: raw_input().strip()

#n = int(ri())
n, k = map(int, ri().split())
Ai = map(int, ri().split())

result = sys.maxint
index = 0

for x in xrange(k):
    tmp = 0
    for y in xrange(x, n, k):
        tmp += Ai[y]

    #print tmp
    
    if tmp < result:
        result = tmp
        index = x+1

print index
