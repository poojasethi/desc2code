#include <cstdio>

#define N 100000

int a[N], m, n, i, k;

int main()
{
	scanf("%d%d", &n, &k);

	m = n - k;

	for (i = 0; i < n; i ++)
	{
		scanf("%d", &a[i]);
		if (i >= k)
			a[i] += a[i - k];
	}

	for (i = n - k + 1; i < n; i ++)
		if (a[i] < a[m])
			m = i;

	printf("%d\n", m - (n - k) + 1);

	return 0;
}
