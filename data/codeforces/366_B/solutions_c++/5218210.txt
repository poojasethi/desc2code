#include<iostream>
#include<vector>
using namespace std;

int main(){
	int i,min,sum,n,k;
	cin>>n>>k;
	int a[n];
	vector<int> b;
	b.resize(k,0);
	for(i=0;i<n;i++){
		cin>>a[i];
		b[i%k]+=a[i];
	}
	min=b[0];sum=0;
	for(i=0;i<k;i++){
		if(min>b[i]){
			min=b[i];
			sum=i;
		}
	}
	cout<<sum+1<<endl;
	return 0;
}
