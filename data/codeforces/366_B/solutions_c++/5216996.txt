#include <cstdio>
#include <cstring>
#include <algorithm>

int main()
{
	int n, k;
	int sums[100010];
	memset(sums, 0x00, sizeof(sums));
	scanf("%d %d", &n, &k);
	for(int i = 0; i < n; i++)
	{
		int tmp;
		scanf("%d", &tmp);
		sums[i%k] += tmp;
	}
	printf("%d", (int)(std::min_element(sums, sums+k) - sums) + 1);
}
