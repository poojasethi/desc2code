#include<cstdio>
int n,k;
int a[1000005];
int mn=1e9,ans;
main(){
  scanf("%d%d",&n,&k);
  for(int i=0;i<n;i++)scanf("%d",a+i);
  for(int i=0;i<k;i++){
    int s=0;
    int j=i;
    do{
      s+=a[j];
      j=(j+k)%n;
    }while(i!=j);
    if(s<mn){
      mn=s;
      ans=i;
    }
  }
  printf("%d\n",ans+1);
}
