#include <iostream>
#include <vector>
using namespace std;
int main(){
	int N,K;
	cin>>N>>K;
	vector<int> a(K,0);
	int b;
	for(int i=0; i<N; ++i){
		cin>>b;
		a[i%K]+=b;
	}
	int ans=0,min=1<<28;
	for(int i=0; i<K; ++i){
		if(a[i]<min){
			min=a[i];
			ans=i+1;
		}
	}
	cout << ans << endl;
	return 0;
}
