#include <cstdio>
#include <cstring>
#include <vector>
#include <iostream>

#define fi first
#define se second

using namespace std;

typedef pair<int,int> ii;

const int MAXN = 10 ;

int way[MAXN][MAXN];
int M; 
int cnt[MAXN];
vector<int> eu;
vector<ii> v;
bool used[110];
char s[11111];

void f(int node){

	for(int i=0;i<=6;i++)
		if(way[node][i]>0){
			
			way[node][i]--;
			way[i][node]--;
			
			f( i );	
				
		}
	
	eu.push_back(node);

}

int main(){

	cin >> M ; 
	
	while(M--){
	
		int a,b;scanf(" %d %d",&a,&b);
		
		v.push_back(ii(a,b));
		cnt[a]++;
		cnt[b]++;
		way[a][b]++;
		way[b][a]++;
	
	}		
	
	vector<int> odd; 
	
	for(int i=0;i<=6;i++)
		if(cnt[i]%2) odd.push_back(i);
	
	if(!odd.size() || odd.size()==2){
		
		if(!odd.size()){
			for(int i=0;i<=6;i++)
				if(cnt[i]>0){
					f(i);
					break;
				}
		}
		else f(odd[0]);	
		
		int res = 0 ;
		
		for(int i=0;i<eu.size()-1;i++){
			
			int a = eu[i];
			int b = eu[i+1];
			
			for(int j=0;j<v.size();j++){
				if(!used[j] && v[j].fi==a && v[j].se==b){
					used[j]=1;
					res++;
					sprintf(s+strlen(s),"%d +\n",j+1);
					break;
				}if(!used[j] && v[j].fi==b && v[j].se==a){
					used[j]=1;
					res++;
					sprintf(s+strlen(s),"%d -\n",j+1);
					break;
				}
			}
			
			if(res==v.size())	return printf("%s",s) , 0 ;	
			
		}
		
		
	}
	
	cout << "No solution" << endl;	

	return 0;
}
