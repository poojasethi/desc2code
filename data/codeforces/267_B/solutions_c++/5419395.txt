#include <iostream>
#include <vector>
using namespace std;
int ed[10][10], mark[10][10], mark2[100 + 10];
vector<int> nei[10], ans1;
vector<pair<int,int> > ans, XY;
void euler(int k)
{
  for(int i = 0; i <= 6; i++)
    {
      if(mark[k][i] < ed[k][i])
	{
	  mark[k][i]++;
	  mark[i][k]++;
	  euler(i);
	}
    }
  ans1.push_back(k);
}
int main()
{
  int n, numodd = 0, odd, sol = 1;
  cin>>n;
  for(int i = 0; i < n; i++)
    {
      int x, y;
      cin >> x >> y;
      nei[ x ].push_back( y );
      nei[ y ].push_back( x );
      ed[ x ][ y ]++;
      ed[ y ][ x ]++;
      odd = x;
      XY.push_back(pair<int,int> (x,y));
      mark2[i]++;
    }
  for(int i = 0; i <= 6; i++)
    if( nei[i].size() % 2 )
      {
	odd = i;
	numodd++;
      }
  euler(odd);
  for(int i = 0; i < ans1.size(); i ++)
    if(i < ans1.size() - 1)
      ans.push_back(pair<int,int> (ans1[i],ans1[i+1]));
  if(numodd > 2 || ans.size() != n)
    {
      cout<<"No solution"<<endl;
      return 0;
    }
  for(int i = 0; i < ans.size(); i++)
    for(int j = 0; j < XY.size(); j++)
      {
	if(ans[i].first== XY[j].first && ans[i].second == XY[j].second && mark2[j])
	  {
	    mark2[j]--;
	    cout<<j+1<<" + "<<endl;
	    break;
	  }
	if(ans[i].first == XY[j].second && ans[i].second == XY[j].first && mark2[j])
	  {
	    mark2[j]--;
	    cout<<j+1<<" - "<<endl;
	    break;
	  }
      }
  return 0;
}
