#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <vector>
using namespace std;
int x[110],y[110],ansx[110],ansy[110],map[10][10],cnt[10];
int n,tot,p;
void Dfs(int k){
	//if(tot==n) return;
	for(int i=0;i<=6;i++){
		if(map[k][i]>0){
			map[k][i]--;map[i][k]--;
			Dfs(i);
			tot++;
			ansx[tot]=k;ansy[tot]=i;
		}
	}
}
int main(){
	//freopen("test.in","r",stdin);
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&x[i],&y[i]);
		map[x[i]][y[i]]++;
		map[y[i]][x[i]]++;
		cnt[x[i]]++;cnt[y[i]]++;
	}
	for(int i=0;i<=6;i++)
		if(cnt[i]) p=i;
	for(int i=0;i<=6;i++)
		if(cnt[i]&1) tot++,p=i;
	if(tot==1||tot>2){
		printf("No solution\n");
		return 0;
	}
	tot=0;
	Dfs(p);
	if(tot<n){
		printf("No solution\n");
		return 0;
	}
	//for(int i=1;i<=tot;i++) printf("%d %d\n",ansx[i],ansy[i]);
	for(int i=tot;i>=1;i--){
        //printf("%d %d ",ansx[i],ansy[i]);
		for(int j=1;j<=n;j++){
			if(ansx[i]==x[j]&&ansy[i]==y[j]){
				printf("%d +\n",j);
				x[j]=7;
				break;
			}else if(ansx[i]==y[j]&&ansy[i]==x[j]){
				printf("%d -\n",j);
				x[j]=7;
				break;
			}
		}
	}
}
