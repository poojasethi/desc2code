using namespace std;
int start,m,cnt,a[7][7];
int i,j,n,x[101],y[101],d[7],s[202],v[101],f[101];
void DFS(int x)
{
	for(int y=0;y<=6;y++)
		if(a[x][y]>0)
		{
		a[x][y]--;
		a[y][x]--;
		DFS(y);
		}
	s[m--]=x;
}
int main()
{
	scanf("%d",&n);
	m=n+1;
	for(int i=1;i<=n;i++)
	{
		scanf("%d%d",&x[i],&y[i]);
		a[x[i]][y[i]]++;
		a[y[i]][x[i]]++;
		d[x[i]]++;
		d[y[i]]++;
	}
	start=x[1];
	for(i=0;i<=6;i++) if(d[i]&1) cnt++,start=i;
	if(cnt>2){printf("No solution\n");return 0;}
	DFS(start);
	if(m>0){printf("No solution\n");return 0;}
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
		if (!f[j])
		{
			if(s[i]==x[j]&&s[i+1]==y[j]){f[j]=1;printf("%d +\n",j);;break;}
			else
			if(s[i]==y[j]&&s[i+1]==x[j]){f[j]=1;printf("%d -\n",j);break;}
		}
	return 0;
}
