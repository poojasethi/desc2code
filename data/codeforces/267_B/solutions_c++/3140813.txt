#include<fstream>
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<vector>
#define pb push_back
using namespace std;

int a[7][7],x[105],y[105],d[105],ans[105],N,M;
bool used[105];

void dfs(int nod){
  
   for(int i=0;i<=6;++i)
     if(a[nod][i]){
        a[nod][i]--; a[i][nod]--;
        dfs(i);
        }
   ans[M--]=nod;
}

int main(){
    //ifstream cin("p2.in");
    //ofstream cout("p2.out");
    int i,j,start,cnt=0;
    
    cin>>N; M=N+1;
    for(i=1;i<=N;++i){
      cin>>x[i]>>y[i];
      a[x[i]][y[i]]++; a[y[i]][x[i]]++;
      d[x[i]]++; d[y[i]]++;
      }
   start = x[1];
   for(i=0;i<=6;++i)
     if(d[i]&1){ ++cnt; start=i; }
     
   if(cnt>2){ cout<<"No solution"; return 0; }
   
   dfs(start);  
   
   if(M>0){ cout<<"No solution"; return 0; }
   
   for(i=1;i<=N;++i)
   {
     for(j=1;j<=N;++j)
       if(!used[j])
         if(ans[i]==x[j] && ans[i+1]==y[j]){ cout<<j<<" +\n"; used[j]=1; break; } else
         if(ans[i]==y[j] && ans[i+1]==x[j]){ cout<<j<<" -\n"; used[j]=1; break; }                                  
   } 
    
 return 0;   
}
