#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<cstring>
#include<stack>
using namespace std;
int mp[7][7],du[7];
int m,x,y;
int s[105],t[105],f[105],ct[2],fa[7],dd[7];
int find(int x) {return fa[x]==x?x:fa[x]=find(fa[x]);}
void euler(int u)
{
    for(int v=0;v<=6;v++) if(mp[u][v]>0)
    {
        mp[u][v]--;mp[v][u]--;
        euler(v);
        for(int i=1;i<=m;i++) if(!f[i])
        {
            if(s[i]==v&&t[i]==u) {f[i]=1;printf("%d +\n",i);break;}
            if(s[i]==u&&t[i]==v) {f[i]=1;printf("%d -\n",i);break;}
        }
    }
}
main()
{
    scanf("%d",&m);
    for(int i=0;i<=6;i++) fa[i]=i;
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d",&s[i],&t[i]);dd[s[i]]=dd[t[i]]=1;
        if(s[i]!=t[i])
        {
            x=find(s[i]);y=find(t[i]);fa[x]=y;
        }
        mp[s[i]][t[i]]++;mp[t[i]][s[i]]++;du[s[i]]++;du[t[i]]++;
    }
    for(int i=0;i<=6;i++) if(dd[i]) x=find(i);
    y=1;
    for(int i=0;i<=6;i++) if(dd[i]&&find(i)!=x) y=0;
    for(int i=0;i<=6;i++) ct[du[i]%2]++;
    if(y==0||ct[1]>2)
    puts("No solution");
    else if(ct[1]==2)
    {for(int i=0;i<=6;i++) if(du[i]%2==1) euler(i);}
    else
    {for(int i=0;i<=6;i++) if(dd[i]) euler(i);}


}
