#include <iostream>
#include <fstream>
#include <vector>

using namespace std;

ifstream fin ("input.txt");
ofstream fout ("output.txt");

int n;
int g[8]= {};
vector <pair <int, int> > f;
int ans1 [1111];
char ans2 [1111];
int r[1111];

bool dfs (int x, int y){
    if ( y > n ){
        for (int i=1; i <= n; i++ )
            cout << ans1 [i] << " " << ans2[i] << "\n";
            return true;
    }
    else {
        for (int i=0; i<f.size (); i++)
            if ( r[i] == 0)
            if ( x == f[i].first){
                r[i] = 1;
                ans1 [y] = i+1;
                ans2[y] = '+';
                if (dfs (f[i].second, y+1)) return true;
                r[i] = 0;
            }else if ( x == f[i].second ){
                r[i] = 1;
                ans1[y] = i+1;
                ans2[y] = '-';
                if (dfs ( f[i].first, y+1)) return true;
                r[i] = 0;

            }

    }
    return false;
}


int main()
{
    cin >> n;

    for (int i=0; i<n; i++)
        r[i] = 0;

    for (int i=0; i<n; i++){
        int xx, yy;
        cin >> xx >> yy ;
        f.push_back (make_pair (xx, yy) );
        g[xx]++; g[yy]++;
    }
    int h=0; int t = -1;
    for (int i=0; i<=6; i++) if ( g[i] % 2 == 1 ){ h++; t=i;}
    if ( h > 2 ){
        cout << "No solution";
        return 0;
    }
    if ( t == -1 ) t = f[0].first ;
   if ( !dfs (t, 1)) cout << "No solution";


    return 0;
}
