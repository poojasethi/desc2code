#include <iostream>
#include<cmath>
#include<sstream>
#include <cstdio>
#include<algorithm>
#include<vector>
#include<set>
#include<map>
#include<string>
#define MAXN 150
#define MAXV 10
using namespace std;
int n,ans[MAXN],d[MAXV],data[MAXV][MAXV];
bool used[MAXN];
int e[MAXV][MAXV][MAXN],enow[MAXV][MAXV];

void dfs(int u)
{
	int v;
	for (v=0;v<=6;v++)
	{
		if (data[u][v]!=0)
		{
			data[u][v]--;
			data[v][u]--;
			dfs(v);
		}
	}
	ans[++ans[0]]=u;
}

void output()
{
	if (ans[0]-1<n)
	{
		printf("No solution");
		return;
	}
	int i,u,v,num;
	for (i=1;i<=ans[0]-1;i++)
	{
		u=ans[i];
		v=ans[i+1];
		while ( used[ abs( e[u][v][ ++enow[u][v] ] ) ] );
		num=e[u][v][enow[u][v]];
		used[abs(num)]=true;
		printf("%d",abs(num));
		if (num>0)
			printf(" +\n");
		else
			printf(" -\n");
	}
}

void run()
{
	int i,s,now=0;
	for (i=0;i<=6;i++)
		if (d[i]!=0)
		{
			s=i;
			break;
		}
	for (i=0;i<=6;i++)
		if (d[i]%2==1)
		{
			s=i;
			now++;
		}
	if (now!=2 && now!=0)
	{
		printf("No solution");
		return;
	}
	dfs(s);
	output();
}

void ini()
{
	int i,a,b;
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		scanf("%d%d",&a,&b);
		d[a]++;
		d[b]++;
		data[a][b]++;
		data[b][a]++;
		e[a][b][++e[a][b][0]]=i;
		e[b][a][++e[b][a][0]]=-i;
	}
}

int main()
{
	ini();
	run();
	return 0;
}
