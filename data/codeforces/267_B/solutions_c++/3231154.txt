#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
 struct name{
        int a;
        int b;
        int used;
 } D[101];
 int i,j,n,x,y,ind=-1,num,g[101][101],sum[7],raod[1000];
 vector <int> ans;
  void Euler(int v){
       for (int k=0;k<=6;k++)
           if (g[v][k]>0 && sum[k]>0){
              sum[v]--,g[v][k]--;
              if (v!=k) sum[k]--,g[k][v]--;
              Euler(k);
           };
       ans.push_back(v);
  };
int main(){
// freopen("input.txt","r",stdin);
// freopen("output.txt","w",stdout);
  cin>>n;
  for (i=1;i<=n;i++){
      cin>>x>>y;
      g[x][y]++,sum[x]++,D[i].a=x,D[i].b=y;
      if (x!=y) raod[x]++,raod[y]++,g[y][x]++,sum[y]++; 
  };
  for (i=0;i<=6;i++)
       if (raod[i]%2) {num++; ind=i;};
  if (ind==-1) 
     for (i=0;i<=6;i++) 
         if (sum[i]) ind=i;
  Euler(ind);
  for (i=0;i<=6;i++)
      if (sum[i] || num>2 || num==1){
         cout<<"No solution"; return 0;
      };
  reverse(ans.begin(),ans.end());
  if (ans.size()==1){
     cout<<1<<" +"; return 0;
  };
  for (i=1;i<ans.size();i++)
       for (j=1;j<=n;j++)
           if (!D[j].used){
              if (D[j].a==ans[i-1] && D[j].b==ans[i])
                 cout<<j<<" +"<<endl,D[j].used=1;
              if (D[j].a==ans[i] && D[j].b==ans[i-1] && !D[j].used)
                 cout<<j<<" -"<<endl,D[j].used=1;
              if (D[j].used) break;
           };
 return 0;
}
