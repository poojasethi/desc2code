def main():
    n = int(raw_input())
    print 95
    a = [['N'] * 95 for i in xrange(95)]
    def add_edge(v, w):
        a[v][w] = a[w][v] = 'Y'
    add_edge(0, 2)
    for i in xrange(30):
        for j in [2, 3]:
            for k in [4, 5]:
                add_edge(j + i + i, k + i + i)
            if n % 2:
                add_edge(j + i + i, 64 + i)
        add_edge(64 + i, 65 + i)
        n /= 2
    add_edge(94, 1)
    for l in a:
        print ''.join(l)
main()
