#include <cstdio>

#define N 1000
#define M 30

int a[N + 1][N + 1], n = 3, m, k, i, j = 3, M2 = M << 1;

int main()
{
	scanf("%d", &k);

	a[1][3] = 1;

	for (m = 0; k && m < M; ++m)
	{
		if (k & (1 << m))
		{
			k ^= (1 << m);
			j = n;
			i = ((m + 1) << 1);
			while (i < M2)
				a[j][j + 1] = 1, ++j, ++i;
			a[2][j] = 1;
		}
		if (k)
			a[n][j + 1] = a[n][j + 2] = a[j + 1][j + 3] = a[j + 2][j + 3] = 1, n = j = j + 3;
		else
			n = j;
	}

	printf("%d\n", n);

	for (i = 1; i <= n; i ++)
	{
		for (j = 1; j <= n; j ++)
			putchar(a[i][j] || a[j][i] ? 'Y' : 'N');
		printf("\n");
	}

	return 0;
}
