#include <cstdio>
#include <cstring>

using namespace std;

int k,n;
int g[1010][1010];
int lst[1010],q[1010],ed;

inline void make_edge(int u,int v){g[u][v]=g[v][u]=1;}

int main()
{
	make_edge(1,lst[1]=3);make_edge(1,lst[2]=n=4);ed=2;
	for(int i=1;i<=30;i++){
		for(int j=1;j<=ed;j++) make_edge(lst[j],q[j]=++n);
		for(int j=1;j<=ed;j++) make_edge(lst[j],n+1);q[++ed]=++n;
		for(int j=1;j<=ed;j++) lst[j]=q[j];
	}
	//printf("%d\n",n);
	scanf("%d",&k);
	for(int i=0;i<30;i++) if(k&(1<<i)) make_edge(2,lst[i+2]);
	//freopen("388B.out","w",stdout);
	printf("%d\n",n);
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++) putchar(g[i][j] ? 'Y' : 'N');
		puts("");
	}
	return 0;
}
