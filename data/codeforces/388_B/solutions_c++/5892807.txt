#include <cstdio>
const int N = 155;
bool B[N][N];
int k;
int main() {
	scanf("%d",&k);
	B[1][3] = 1;
	for(int i = 3;i < 70;i++) {
		if (i&1) B[i][i+2] = B[i][i+1] = 1;
		else B[i][i+3] = B[i][i+2] = 1;
	}
	for(int i = 100;i < 150;i++) B[i][i+1] = 1;
	B[150][2] = 1;
	for(int i = 0;i < 32;i++)
		if (k&(1<<i)) B[(i+2)*2][100+i] = 1;
	printf("150\n");
	for(int i = 1;i <= 150;i++) {
		for(int j = 1;j <= 150;j++)
			printf("%c",(B[i][j]||B[j][i])?'Y':'N');
		printf("\n");
	}
	return 0;
}
