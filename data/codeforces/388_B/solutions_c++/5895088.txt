#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int G[111][111],n;

void cal(int k)
{
    while(k) ++n,k>>=1;
    --n;
}

int main()
{
    int k,j;
    cin>>k;
    if(k==1)
    {
        cout<<"2\nNY\nYN\n";
        return 0;
    }
    cal(k);
    G[1][3]=G[1][4]=G[1][5]=1;
    for(j=3;n>=2;--n,j+=3)
    {
        G[j][j+3]=G[j][j+4]=G[j+1][j+3]=G[j+1][j+4]=G[j+2][j+5]=1;
        if(k&1<<n-1) G[j+2][j+3]=G[j+2][j+4]=1;
    }
    if(k&1) G[j+2][2]=1;
    G[j][2]=G[j+1][2]=1;
    cout<<j+2<<endl;
    for(int i=1;i<=j+2;++i,cout<<endl)
        for(int r=1;r<=j+2;++r)
            cout<<(G[i][r]||G[r][i]?'Y':'N');
    return 0;
}
