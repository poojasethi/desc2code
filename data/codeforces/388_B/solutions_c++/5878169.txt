#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int a[N][N];
inline void Do(int x,int y) {
	a[x][y]=a[y][x]=1;
}
inline void work(int x) {
	int i;
	if (!x) {
		Do(1,3); return;
	}
	Do(1,n+1),Do(1,n+2);
	For(i,1,x-1) {
		Do(n+1,n+3),Do(n+1,n+4);
		Do(n+2,n+3),Do(n+2,n+4);
		n+=2;
	}
	int A=3+x;
	Do(n+1,A),Do(n+2,A); n+=2;
}
int main() {
	scanf("%d",&m); n=2;
	For(i,1,40) Do(n+1,n+2),n++;
	Do(n+1,2),n++;
	for (i=0;m;m>>=1,i++) if (m&1) work(i);
	printf("%d\n",n);
	For(i,1,n) {
		For(j,1,n) printf("%c",a[i][j]?'Y':'N');
		printf("\n");
	}
	return 0;
}
