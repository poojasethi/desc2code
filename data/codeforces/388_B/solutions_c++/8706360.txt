#include<iostream>
#include<vector>
#include<algorithm>

using namespace std ;

int n ;
bool d[155][155] ;

int main()
{
  cin >> n ;
  cout << 150 << endl ;
  d[1][3] = d[150][2] = 1 ;
  for(int i=3 ; i<70 ; i++)
    {
      if(i&1)
	d[i][i+2] = d[i][i+1] = 1 ;
      else
	d[i][i+3] = d[i][i+2] = 1 ;
    }
  
  for(int i=100 ; i<150 ; i++)
    d[i][i+1] = 1 ;

  for(int i=0 ; i<32 ; i++)
    if(n&(1<<i))
      d[2*(i+2)][i+100] = 1 ;
  for(int i=1 ; i<151 ; i++)
    {
    for(int j=1 ; j<151 ; j++)
      {
	if(d[i][j] || d[j][i])
	  cout << "Y" ;
	else
	  cout << "N" ;
      }
    cout << endl ;
    }
  return 0 ;
}
