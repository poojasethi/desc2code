

#include <cstdio>



const int N = 1100;
bool mp[N][N];

int n,k;


void addE(int a, int b)
{
	mp[a][b]=mp[b][a]=true;
}

void help(int k)
{
	while(k--){
		addE(n,n+1);
		addE(n,n+2);
		addE(n+1,n+3);
		addE(n+2,n+3);
		n+=3;
	}
	addE(n,2);
}

void build(int rt, int i)
{
	if(i<0)return;

	if( k & (1<<i)){
		addE(rt, (31-i)*3);
		//addE(rt, ++n );
		//help(i);
	}

	addE(rt,n+1);
	addE(n+1,n+2);
	n+=2;

	build(n, i-1);
}

int main()
{
	n=3;
	help(30);

	scanf("%d",&k);
	build(1,30);

	printf("%d\n",n);
	for(int i=1;i<=n;++i){
		for(int j=1;j<=n;++j) printf("%c", mp[i][j] ? 'Y' : 'N');
		puts("");
	}

	return 0;
}
