#include<cstdio>
using namespace std;
int k,t,a[110][110],i,j;
int main()
{
    scanf("%d",&k);for(t=0;k>>t;t++);
    a[3][2]=a[4][2]=1;
    for(i=5;i<=(t<<1)+2;i+=2)
        a[i][i-2]=a[i][i-1]=a[i+1][i-2]=a[i+1][i-1]=1;
    for(i=0;i<=t;i++,a[i+(t<<1)+3][i+(t<<1)+2]=1)
        if(k>>i&1)a[i+(t<<1)+3][(i<<1)+3]=1;
    a[1][t*3+2]=1;
    printf("%d\n",t*3+2);
    for(i=1;i<=t*3+2;i++,puts(""))
        for(j=1;j<=t*3+2;j++)
            if(a[i][j]||a[j][i])putchar('Y');else putchar('N');
}
