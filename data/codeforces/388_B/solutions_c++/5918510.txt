#include <iostream>
using namespace std;

int paths;
bool edge[95][95];

void add(int a, int b)
{
	edge[a][b] = edge[b][a] = true;
}

int main()
{
	cin >> paths;

	for (int i = 0; i < 95; i++)
		for (int j = 0; j < 95; j++)
			edge[i][j] = false;
	
	for (int i = 0; i < 30; i++)
		for (int j = 0; j < 2; j++)
			for (int k = 0; k < 2; k++)
				add(2*(i+1) + j, 2*(i+2) + k);

	for (int i = 64; i < 94; i++)
		add(i, i + 1);

	add(0, 2);
	add(0, 3);
	add(94, 1);
	
	for (int i = 0; i < 30; i++)
		if (paths&(1 << i))
			add(2*i + 2, 65 + i);
	
	cout << "95\n";
	for (int i = 0; i < 95; i++)
	{
		for (int j = 0; j < 95; j++)
			cout << ((edge[i][j]) ? 'Y' : 'N');
		cout << "\n";
	}

	return 0;
}
