#include <cstdio>
#include<iostream>
#define add(i,j) f[i][j]=f[j][i]=1
using namespace std;
int N,K,a[40],f[1005][1005];

int main()
{
	scanf("%d",&K),N=2;
	for (;K; K>>=1) a[++a[0]]=K&1;
	int p=1;
	for (int i=a[0]-1,s=0; i; i--)
	{
		add(p,N+1),add(p,N+2),add(N+1,N+3),add(N+2,N+3),N+=3,s+=2,p=N;
		if (a[i])
		{
			add(1,N+1),add(N+s-1,N);
			for (int j=1; j<=s-2; j++) add(N+j,N+j+1);
			N+=s-1;
		}
	}
	add(p,2),printf("%d\n",N);
	for (int i=1; i<=N; i++)
	{
		for (int j=1; j<=N; j++) putchar(f[i][j]?'Y':'N');
		puts("");
	}
}
