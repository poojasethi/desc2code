#include<iostream>
using namespace std;
bool g[155][155]={0};
int n;
int main() 
{
    cin>>n;
    g[1][3]=1;
    for(int i=3;i<70;i++)
    {
        if(i&1) g[i][i+2]=g[i][i+1]=1;
        else g[i][i+3]=g[i][i+2] = 1;
    }
    for(int i=100;i<150;i++) g[i][i+1] = 1;
    g[150][2]=1;
    for(int i=0;i<32;i++)
        if (n&(1<<i)) g[(i+2)*2][100+i]=1;
    cout<<"150"<<endl;
    for(int i=1;i<=150;i++) {
        for(int j=1;j<=150;j++)
            cout<<(g[i][j]||g[j][i]?"Y":"N");
        cout<<endl;
    }
    return 0;
}
