#include <stdio.h>
#include <iostream>
#include <string>
using namespace std;
int n=2,m,a[60],p=1,s=0,l=0;
bool f[1200][1200]={false};
string t;
void back(int n){
	int t=n;
	while (t>0){
		l++;
		a[l]=t%2;
		t>>=1;
	}
}
void draw(int x,int y){
	f[x][y]=true;
	f[y][x]=true;
}
int main( ){
	cin>>m;
	back(m);
	for (int i=l-1;i>0;i--){
		draw(p,n+1);
		draw(p,n+2);
		draw(n+1,n+3);
		draw(n+2,n+3);
		n+=3,s+=2,p=n;
		if (a[i]){
			draw(1,n+1);
			draw(n,n+s-1);
			for (int j=1;j<s-1;j++)
				draw(n+j,n+j+1);
			n+=s-1;
		}
	}
	draw(2,p);
	cout<<n<<endl;
	for (int i=1;i<=n;i++){
		t="";
		for (int j=1;j<=n;j++)
			if (f[i][j]) t+='Y'; else t+='N';
		cout<<t<<endl;
	}
	return 0;
}
