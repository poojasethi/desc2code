#include <cstdio>

int n,k;
int sol[1002][1002];
void ed(int va,int vb) {
	sol[va][vb]=1;
	sol[vb][va]=1;
}
int main() {
	n=300;
	for(int i=1;i<=n;i++) {
		for(int j=1;j<=n;j++) {
			sol[i][j]=0;
		}
	}
	for(int i=0;i<30;i++) {
		ed(200+i*3,201+i*3);
		ed(200+i*3,202+i*3);
		ed(201+i*3,203+i*3);
		ed(202+i*3,203+i*3);
	}
	ed(2,290);
	ed(1,3);
	for(int i=0;i<30;i++) {
		ed(3+i*2,4+i*2);
		ed(4+i*2,5+i*2);
	}
	ed(2,63);
	scanf("%d",&k);
	k--;
	for(int i=0;i<30;i++) {
		if(k&(1<<i)) {
			ed(62-i*2,290-i*3);
		}
	}
	printf("300\n");
	for(int i=1;i<=n;i++) {
		for(int j=1;j<=n;j++) {
			if(sol[i][j]) {
				printf("Y");
			} else {
				printf("N");
			}
		}
		printf("\n");
	}
	return 0;
}
