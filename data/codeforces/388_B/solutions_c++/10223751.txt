{

		for(j=bl[i-1],k=bl[i];j<=br[i-1];j++,k++)a[j][k]=a[k][j]=a[j][br[i]]=a[br[i]][j]=1;
		//printf("i=%d\n",i);
	}
	scanf("%d",&n);

	{
		for(j=1;j<=br[30];j++)if(a[i][j])putchar('Y');else putchar('N');

}
