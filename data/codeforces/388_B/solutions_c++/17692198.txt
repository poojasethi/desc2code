#include<bits/stdc++.h>
using namespace std;
const int maxn = 3001;
int mp[maxn][maxn];
int k;
int tot = 3;
int main()
{
	scanf("%d",&k);
	int a=1,b=1,c=tot++;
	while (k)
	{
		if (k%2)
			mp[a][c]=1,mp[b][c]=1;
		k/=2;
		if (!k)
			break;
		int d = tot++;
		int e = tot++;
		int f = tot++;
		mp[a][d]=1;
		mp[a][e]=1;
		mp[b][d]=1;
		mp[b][e]=1;
		mp[c][f]=1;
		a=d;
		b=e;
		c=f;
	}
	mp[c][2]=1;
	int cnt = tot;
	printf("%d\n",cnt);
	for (int i = 1;i<=cnt;i++)
	{
		for (int j = 1;j<=cnt;j++)
		{
			if (mp[i][j] || mp[j][i])
				printf("Y");
			else
				printf("N");
		}
		printf("\n");
	}
}
