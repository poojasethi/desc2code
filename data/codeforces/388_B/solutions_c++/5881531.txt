#include<cstdio>

int y[35];
bool r[1005][1005];
int vn = 2;

void set(int a, int b){
	r[a][b] = r[b][a] = 1;
}

void append(int x){
	int last = 1;
	for(int i=x;i<30;i++){
		int v = ++vn;
		set(v, last);
	 	last = v;
	}

	if(!x)
		set(last, 2);
	else {
		set(last, 2*x + 1);
		set(last, 2*x + 2);
	}
}

int main(void){
	set(3, 2);
	set(4, 2);
	for(int i=2 ; i < 31;i++){
		set(2*i + 1, 2*i - 1);
		set(2*i + 1, 2*i);
		set(2*i + 2, 2*i - 1);
		set(2*i + 2, 2*i);
	}

	vn = 62;

	int k;
	scanf("%d", &k);

	for(int i=0;k;k>>=1, i++)
		if(k&1)
			append(i);

	printf("%d", vn);
	for(int i=1;i<=vn;i++){
		puts("");
		for(int j=1;j<=vn;j++)
			putchar(r[i][j] ? 'Y' : 'N');
	}

	return 0;
}
