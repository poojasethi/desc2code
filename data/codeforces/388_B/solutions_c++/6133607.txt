#include <cstdio>
char grid[1001][1001];
int nodeofpow[31];
void join(int a,int b){
	grid[a][b]='Y';
	grid[b][a]='Y';
}
int main(){
	int k,cnt;
	scanf("%d",&k);
	for(int i=1;i<=1000;i++)
		for(int j=1;j<=1000;j++)
			{grid[i][j]='N'; grid[j][i]='N';}
	int v1=3,v2=4;
	join(2,3);
	join(2,4);
	for(int i=0;i<=30;i++){
		nodeofpow[i]=v1;
		join(v1,v1+2);
		join(v1,v1+3);
		join(v2,v1+2);
		join(v2,v1+3);
		v1=v1+2; v2=v2+2; 
	}
    v1=v1+2; v2=v2+2; 
	cnt=v1;
	
	for(int i=0;i<=30;i++){
		
		if((1<<i) &k)
			join(nodeofpow[i],cnt);
		join(cnt,cnt+1);
		cnt++;
	}
	join(1,cnt);
	printf("%d\n",cnt);
	for(int i=1;i<=cnt;i++){
		for(int j=1;j<=cnt;j++)
			printf("%c",grid[i][j]);
		printf("\n");
	}
	return 0;
}
