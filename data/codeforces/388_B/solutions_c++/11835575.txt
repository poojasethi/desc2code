#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int n,k,l,r;
bool graph[1005][1005];
inline void insert(int u,int v) {
	graph[u][v]=graph[v][u]=true;
}
int main() {
	scanf("%d",&k);
	insert(1,3); insert(1,4);
	l=3; r=4;
	for (int i=2;i<=30;i++) {
		for (int j=r+1;j<=r+r-l+1;j++) insert(j,j-(r-l+1));
		for (int j=l;j<=r;j++) insert(j,r+(r-l+1)+1);
		int t=r;
		r=r+(r-l+1)+1; l=t+1;
	}
	for (int i=1;i<=30;i++) {
		if (k%2==1) insert(2,l+i);
		k>>=1;
	}
	printf("%d\n",r);
	for (int i=1;i<=r;i++) {
		for (int j=1;j<=r;j++) if (graph[i][j]) printf("Y"); else printf("N");
		printf("\n");
	}
	return 0;
}
