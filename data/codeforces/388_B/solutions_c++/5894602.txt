#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int G[111][111];

int main()
{
    int k,n;
    cin>>k;
    G[1][5]=1;
    if(k&1<<31) G[1][3]=G[1][4]=1;
    for(int i=30,j=3;i>=1;--i,j+=3)
    {
        G[j][j+3]=G[j][j+4]=G[j+1][j+3]=G[j+1][j+4]=G[j+2][j+5]=1;
        if(k&1<<i) G[j+2][j+3]=G[j+2][j+4]=1;
    }
    if(k&1) G[95][2]=1;
    G[93][2]=G[94][2]=1;
    cout<<95<<endl;
    for(int i=1;i<=95;++i)
    {
        for(int j=1;j<=95;++j)
        {
            if(G[i][j]||G[j][i]) cout<<"Y";
            else cout<<"N";
        }
        cout<<endl;
    }
    return 0;
}
