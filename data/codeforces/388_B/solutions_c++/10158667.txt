#include <bits/stdc++.h>
using namespace std;

int k, n, mtx[3005][3005], pos[50], len[50];
void add(int x, int y) { mtx[x][y]=mtx[y][x]=1; }
int main() {
	scanf("%d", &k);
	int x=1, tot=2, mx=0;
	if(k==1) { puts("2\nNY\nYN"); return 0; }
	for(int i=30; i; --i) if((k>>i)&1) { mx=i; break; }
	for(int i=0; i<mx; ++i) {
		pos[i]=x;
		len[i]=(mx-i)*2;
		int a=++tot; int b=++tot;
		add(x, a); add(x, b);
		x=++tot;
		add(x, a); add(x, b);
	}
	add(x, 2);
	for(int i=0; i<mx; ++i) if((k>>i)&1) {
		int now=pos[i];
		for(int j=0; j<len[i]; ++j) {
			add(now, ++tot);
			now=tot;
		}
		add(now, 2);
	}
	printf("%d\n", tot);
	for(int i=1; i<=tot; ++i) {
		for(int j=1; j<=tot; ++j)
			putchar(mtx[i][j]?'Y':'N');
		puts("");
	}
	return 0;
}
