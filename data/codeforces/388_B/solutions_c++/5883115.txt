#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>
#include <algorithm>

using namespace std;

bool mat[1010][1010];

int main(){
  int K;
  cin >> K;
  if (K==1){
    printf("2\nNY\nYN\n");
    return 0;
  }
  mat[1][3]=mat[1][4]=true;
  if (K&1) mat[1][5]=true;
  int T=3;
  for (K>>=1;K>1;K>>=1){
    mat[T][T+3]=mat[T][T+4]=mat[T+1][T+3]=mat[T+1][T+4]=mat[T+2][T+5]=true;
    if (K&1) mat[T][T+5]=mat[T+1][T+5]=true;
    T+=3;
  }
  mat[T][2]=mat[T+1][2]=mat[T+2][2]=true;
  cout << T+2 << endl;
  for (int i=1;i<=T+2;++i){
    for (int j=1;j<=T+2;++j)
      printf("%c",(mat[i][j] || mat[j][i]? 'Y':'N'));
    printf("\n");
  }
  return 0;
}
