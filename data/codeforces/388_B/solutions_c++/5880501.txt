#include <cstdio>

using namespace std;

int k;
int ms[250][250];
int n;

void set(int a, int b)
{
    ms[a][b] = ms[b][a] = 1;
}

int main()
{
    scanf("%d", &k);

    n = 2;
    int lg = 1;
    int ld = 0;
    while(k)
    {
        set(lg, n+1);
        set(lg, n+2);
        set(n+1, n+3);
        set(n+2, n+3);
        set(ld, n+4);
        set(n+4, n+5);
        if(k & 1)
            set(lg, n+4);
        lg = n + 3;
        ld = n + 5;
        n += 5;
        k /= 2;
    }
    set(ld, 2);

    printf("%d\n", n);
    for(int i = 1; i <= n; ++i, puts(""))
        for(int j = 1; j <= n; ++j)
            putchar("NY"[ms[i][j]]);
    return 0;
}
