#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int maxn=1000;
int ans[maxn][maxn];
int dp[maxn][maxn];
int main()
{
	memset(ans,0,sizeof(ans));
	memset(dp,0,sizeof(dp));
	int k;
	scanf("%d",&k);
	int n=2+17*33;
	printf("%d\n",n);
	int c=2;
	for(int i=1;i<=32;i++)
	{
		for(int j=c;j<c+i;j++) ans[j][j+i]=ans[j+i][j]=1;
		for(int j=c;j<c+i;j++) ans[j][c+2*i]=ans[c+2*i][j]=1;
		c+=i;
	}
	ans[0][2]=ans[2][0]=1;
	for(int i=0;i<32;i++) if(k&(1<<i)) ans[c+i+1][1]=ans[1][c+i+1]=1;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++) printf("%c",ans[i][j]==1?'Y':'N');
		puts("");
	}
	return 0;
}
