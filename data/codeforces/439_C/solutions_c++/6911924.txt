#include <iostream>
using namespace std;
class sp{
  public:
  int q;
  sp *w;
  sp(int q1,sp *w1){
   q=q1;w=w1;}
};
int main(){
  int q,w,e,r,t,y=0,u=0,c,v,a[100000],s[100000],f[100000];
  sp *d[100000],*c1;
  cin>>q>>w>>t;
  for(e=0;e<q;e++){
   cin>>r;
   if(r%2){
    a[y]=r;
    y++;}
   else{
    s[u]=r;
    u++;}}
  t=w-t;
  for(e=0;e<w;e++){
   d[e]=0;
   f[e]=0;}
  if((y<t)||(((y-t)%2)||(u+(y-t)/2<w-t)))
   cout<<"NO";
  else{
   cout<<"YES\n";
   c=t;v=0;
   for(e=0;e<t;e++){
    f[e]++;
    d[e]=new sp(a[e],d[e]);}
   e=w-1;
   while((c<y)||(v<u)){
    if(c<y){
     d[e]=new sp(a[c],d[e]);
     c++;
     d[e]=new sp(a[c],d[e]);
     c++;
     f[e]+=2;}
    else{
     d[e]=new sp(s[v],d[e]);
     v++;
     f[e]++;}
    if(e)
     e--;}
   for(e=0;e<w;e++){
    cout<<f[e];
    for(c1=d[e];c1;c1=c1->w)
     cout<<" "<<c1->q;
    cout<<"\n";}}
  //system("PAUSE");
  return 0;}
