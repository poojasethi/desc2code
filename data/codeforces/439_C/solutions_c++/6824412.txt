#include<cstdio>
#include<algorithm>
#include<vector>

using namespace std;

int main()
{
	int n,k,p,i,j,O=0,E=0;
	vector<int> even,odd;
	scanf("%d%d%d",&n,&k,&p);
	for(i=0;i<n;i++)
	{
		int z;
		scanf("%d",&z);
		if(z&1)odd.push_back(z),O++;
		else even.push_back(z),E++;
	}
	if(O<k-p || (1&(O-k+p))==1 || E+(O-k+p)/2<p)puts("NO");
	else{
		puts("YES");
		for(i=0;i<p && i<k-1 && E>0; i++)
			printf("1 %d\n",even[--E]);
		for(;i<p && i<k-1 && O>1;i++,O-=2)
			printf("2 %d %d\n",odd[O-1],odd[O-2]);
		for(;i<k-1;i++)
			printf("1 %d\n",odd[--O]);
		printf("%d",O+E);
		while(O>0)printf(" %d",odd[--O]);
		while(E>0)printf(" %d",even[--E]);
		puts("");
	}
	return 0;
}
