#include <stdio.h>

int main(void)
{
	int n,k,p,i,j=0,m=0;
	int od=0,ev=0;
	__int64 a,odd[100010],even[100010];
	scanf("%d %d %d",&n,&k,&p);
	for(i=0;i<n;i++)
	{
		scanf("%I64d",&a);
		if(a%2!=0)
			odd[j++]=a;
		else
			even[m++]=a;
	}
	if(j<(k-p)||(j-k+p)%2!=0||(j-k+p)/2+m<p)
		printf("NO\n");
	else
	{

		printf("YES\n");
		for(i=0;i<k-p-1;i++)
		{
			printf("1 %I64d\n",odd[od++]);
		}
		for(i=0;i<p-1;i++)
		{
			if(ev<m)
				printf("1 %I64d\n",even[ev++]);
			else
			{
				printf("2 %I64d %I64d\n",odd[od],odd[od+1]);
				od+=2;
			}
		}
		if(p&&k-p>0)
			printf("1 %I64d\n",odd[od++]);
		printf("%d", m- ev + j- od);
        while (ev <m )
            printf(" %I64d", even[ev++]);
        while (od < j)
            printf(" %I64d", odd[od++]);
        putchar('\n');
	}

	return 0;
}
