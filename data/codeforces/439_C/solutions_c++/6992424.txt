#include <stdio.h>
#define TT 100000+2
int n,m,p;
int a[TT];
int x[TT],y[TT]; 
void solve (int n,int m,int p) {
	if(x[0]<m-p||(x[0]-(m-p)&1||(x[0]-(m-p))/2+y[0]<p)&1) {
		printf("NO\n");
		return ;
	}
    printf("YES\n");
    if(p) {
        for (int i=1;i<=m-p;i++)
            printf("1 %d\n",x[x[0]--]);
	}
	else  {
		for (int i=1;i<m-p;i++)
			printf("1 %d\n",x[x[0]--]);
	}
	while (p>1&&y[0])
		printf("1 %d\n",y[y[0]--]),p--;
	while (p>1&&x[0]) {
		printf("2 %d %d\n",x[x[0]--],x[x[0]--]);
		p--;
	}
	if (x[0]+y[0])
		printf("%d ",x[0]+y[0]);
	while (x[0])
		printf("%d ",x[x[0]--]);
	while (y[0])
		printf("%d ",y[y[0]--]);
	printf("\n");
}
int main()
{
    while (scanf("%d%d%d",&n,&m,&p)!=EOF) {
		for(int i=1;i<=n;i++)
		{
			scanf("%d",&a[i]);
			if(a[i]&1)x[++x[0]]=a[i];
			else y[++y[0]]=a[i];
		}
		solve (n,m,p);
	}
    return 0;
}
