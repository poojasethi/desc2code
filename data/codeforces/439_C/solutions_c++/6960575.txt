#include<stdio.h>
#include<vector>
#include<iostream>

using namespace std;

int main(void){
	vector<int> even,odd;
	int n,k,p,a;
	scanf("%d %d %d\n",&n,&k,&p);
	for (int i = 0;i < n; i++){
		scanf(" %d",&a);
		if (1&a) odd.push_back(a);
		else even.push_back(a);
	}
	if ( (odd.size() < k-p) || (1&(odd.size()-k+p)) || (even.size() + (odd.size() -k + p)/2 < p) ) 
		printf("NO\n");
	else {
		printf("YES\n");
		int x = 0;
		int o = odd.size();
		int e = even.size();
		for (;(x < (k-p))&&(x < k-1); x++) printf("1 %d\n",odd[--o]);
		for (;(x < k-1)&&(e > 0); x++) printf("1 %d\n",even[--e]);
		for (;(x < k-1)&&(o>0); x++,o-=2) printf("2 %d %d\n",odd[o-1],odd[o-2]);
		printf("%d",(o+e));
		while (e > 0 ) printf(" %d",even[--e]);
		while (o > 0 ) printf(" %d",odd[--o]);
		printf("\n");
	}
}
