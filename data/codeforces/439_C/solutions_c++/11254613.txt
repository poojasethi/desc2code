#include <iostream>
using namespace std;
int odds[100000], evens[100000];
int main()
{
	ios::sync_with_stdio(false);
	int n, k, p, os = 0, es = 0, t;
	cin >> n >> k >> p;
	for (int i = 0; i < n; ++i){
		cin >> t;
		if (t % 2) odds[os++] = t;
		else evens[es++] = t;
	}
	if (os >= k-p && !((os - k + p) % 2) && 
			(os - k + p)/2 + es >= p){
		cout << "YES\n";
	} else {
		cout << "NO\n";
		return 0;
	}

	if (p){
		while (k > p){
			cout << "1 " << odds[--os] << '\n';
			--k;
		}
		while (p > 1 && es){
			cout << "1 " << evens[--es] << '\n';
			--p;
		}
		while (p > 1) cout << "2 " << odds[--os], cout << " " << odds[--os] << '\n', --p;
	} else {
		
		while (k > 1){
			cout << "1 " << odds[--os] << '\n';
			--k;
		}
	}
	cout << es + os;
	while (es > 0) cout << " " << evens[--es];
	while (os > 0) cout << " " << odds[--os];
}
