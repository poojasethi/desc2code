#include <stdio.h>
const int maxn = 100000;
int a[maxn], la;
int b[maxn], lb;
int main(){
	int n,k,p,i,q;
	scanf("%i %i %i",&n,&k,&p);
	for(i = 0; i < n; i ++){
		scanf("%i",&q);
		if(q&1) a[la ++] = q;
		else b[lb ++] = q;
	}
	if(la < k - p || (la - (k - p)) % 2 || (la - (k - p)) / 2 + lb < p) puts("NO");
	else{
		puts("YES");
		if(p){
			for(i = 0; i < k - p; i ++) printf("1 %i\n",a[i]);
			while(i < la && p > 1){
				printf("2 %i %i\n",a[i],a[i+1]);
				i += 2;
				p --;
			}
			while(p > 1){
				printf("1 %i\n",b[--lb]);
				p --;
			}
			printf("%i ",lb + la - i);
			while(i != la) printf("%i ",a[i ++]);
			while(lb) printf("%i ",b[-- lb]);
		}else{
			for(i = 0; i < k - p - 1; i ++) printf("1 %i\n",a[i]);
			printf("%i ", n - (k - p - 1));
			for(i = k - p - 1; i < la; i ++) printf("%i ",a[i]);
			for(i = 0; i < lb; i ++) printf("%i ",b[i]);
		}
	}
	return 0;
}
