#include<iostream>
#include<vector>
using namespace std;
#define ll long long int 
int main()
{
    ll n,k,p;
    cin>>n>>k>>p;
    ll a[n];
    vector<ll>even,odd;
    for(int i=0;i<n;i++)
    {cin>>a[i];
    if(a[i]&1)
        odd.push_back(a[i]);
        else even.push_back(a[i]);
    }
    ll o=odd.size();
    ll e=even.size();
    int i=0;
    if(o<k-p||(1&(o-k+p))==1||e+(o-k+p)/2<p)
    {
        cout<<"NO";
    }
    else{
        cout<<"YES\n";
    for(i=0;i<min(k-1,k-p);i++)
    {
        cout<<1<<" "<<odd[--o]<<"\n";
    }
    for(;i<k-1;i++)
    {
        if(o>0)
           cout<<2<<" "<<odd[--o]<<" "<<odd[--o]<<"\n";
           else 
           cout<<1<<" "<<even[--e]<<"\n";
    }
    cout<<o+e<<" ";
    while(o)
    {
        cout<<odd[--o]<<" ";
    }
    while(e)
    {
        cout<<even[--e]<<" ";
    }
}
}
