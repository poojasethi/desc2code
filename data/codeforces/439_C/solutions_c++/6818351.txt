#include<cstdio>
#include<vector>
using namespace std;
int n,k,p;
vector<int> odd,even;
vector<int> ans[111111];
int main()
{
   scanf("%d%d%d",&n,&k,&p);
   for(int i=0;i<n;++i){
      int x;
      scanf("%d",&x);
      if(x&1)odd.push_back(x);
      else even.push_back(x);
   }
   int od = k - p;
   if(od > odd.size() || (odd.size() - od)%2==1 
     || (odd.size() - od)/2+even.size() < p){
      puts("NO");
      return 0;
   }
   for(int i=0;i<od;++i)ans[i].push_back(odd[i]);
   int now = od%k;
   for(int i=od;i<odd.size();i+=2){
      ans[now].push_back(odd[i]);
      ans[now].push_back(odd[i+1]);
      now = (now + 1)%k;
   }
   for(int i=0;i<even.size();++i){
      ans[now].push_back(even[i]);
      now = (now+1)%k;
   }
   puts("YES");
   for(int i=0;i<k;++i){
      printf("%d",ans[i].size());
      for(int j=0;j<ans[i].size();++j)
         printf(" %d",ans[i][j]);
      printf("\n");
   }
}
