#include<cstdio>
using namespace std;

int a[100100],b[100100];
int main(){
    int n,k,p,e=0,o=0;
    scanf("%d %d %d",&n,&k,&p);
    for(int i=0,oni;i<n;i++){
        scanf("%d",&oni);
        if(oni&1)
            a[o++]=oni;
        else
            b[e++]=oni;
    }
    if(o<(k-p)||(1&(o-k+p))||e+(o-k+p)/2<p)
        printf("NO\n");
    else{
        printf("YES\n");
        int i;
        for(i=0;i<(k-p)&&i<k-1;i++)
            printf("1 %d\n",a[--o]);
        for(;i<k-1&&e>0;i++)
            printf("1 %d\n",b[--e]);
        for(;i<k-1&&o>0;i++,o-=2)
            printf("2 %d %d\n",a[o-1],a[o-2]);

            printf("%d",o+e);
            while(o>0)printf(" %d",a[--o]);
            while(e>0)printf(" %d",b[--e]);
            printf("\n");

    }
    return 0;
}
