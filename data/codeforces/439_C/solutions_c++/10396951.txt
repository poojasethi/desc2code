    scanf("%d",&a);
    if(a%2==0) {d[j]=a;j++;}
    else {f[l]=a;l++;}
}

cout<<"YES"<<endl;

    cout<<1<<" "<<f[i]<<endl;
}

    if(s>l-1) {cout<<1<<" "<<d[s-l];s++;}

}
