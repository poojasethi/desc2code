#include <cstdio>
#include <iostream>
#include <vector>
using namespace std;

vector<long> even;
vector<long> odd;

int main(){
	long n,k,p;
	cin>>n>>k>>p;
	long temp;
	for (int i=0;i<n;i++)
	{
		cin>>temp;
		if (temp%2==0)
			even.push_back(temp);
		else
			odd.push_back(temp);
	}
	if (odd.size()<k-p)
	{
		cout<<"NO"<<endl;
	}
	else if (odd.size()%2 != (k-p)%2){
		cout<<"NO"<<endl;
	}
	else if ((odd.size()-k+p)/2+even.size()<p){
		cout<<"NO"<<endl;
	}
	else{
		cout<<"YES"<<endl;
		int o=0;
		int e=0;
		for (int i=0;i<k-1;i++){
			if(i<k-p){
				cout<<1<<" "<<odd[o];
				o++;
			}
			else{
				if (e<even.size()){
					cout<<1<<" "<<even[e];
				e++;
				}
				else{
					cout<<2<<" "<<odd[o]<<" "<<odd[o+1];
					o+=2;
				}
			}
			cout<<endl;
		}
		cout<<odd.size()+even.size()-o-e;
		for (int i=e;i<even.size();i++)
			cout<<" "<<even[i];
		for (int i=o;i<odd.size();i++)
			cout<<" "<<odd[i];
		cout<<endl;
	}
}
