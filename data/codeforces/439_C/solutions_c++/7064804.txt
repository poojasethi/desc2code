#include<iostream>
#include<vector>
using namespace std;
#define f cin
#define g cout
#define pb push_back
vector <int> v1,v2,ans[100005];
int main()
{
    int n,k,p1,p0;
    f >> n >> k >> p0;
    p1=k-p0;
    for(int i=0;i<n;i++){
        int t;
        f >> t;
        if (t%2==1) v1.pb(t);
        else v2.pb(t);
    }
    int n1=v1.size(), n2=v2.size();

    if (n1<p1|| (n1-p1)%2==1 || n2+(n1-p1)/2<p0){
        g << "NO\n";
        return 0;
    }

    for (int i=0;i<p1;i++)
        ans[i].pb(v1[i]);

    int now=p1%k;
    for (int i=p1;i<n1;i+=2){
        ans[now].pb(v1[i]);
        ans[now].pb(v1[i+1]);
        now=(now+1)%k;
    }
    for (int i=0;i<n2;i++){
        ans[now].pb(v2[i]);
        now=(now+1)%k;
    }

    g << "YES\n";
    for (int i=0;i<k;i++){
        g << ans[i].size() << " ";
        for (int j=0;j<ans[i].size();j++)
            g << ans[i][j] << " ";
        g << "\n";
    }
}
