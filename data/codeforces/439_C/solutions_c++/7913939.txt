#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>

#define pb push_back

using namespace std;

int n,k,p,i,a,lst;
vector < int > KK,LL,ans[200001];

main()
 {
  cin>>n>>k>>p;
  for (i=1;i<=n;i++)
   { cin>>a; if (a%2) KK.pb(a); else LL.pb(a); }
  if (KK.size() < k-p || ((KK.size()-k+p)%2))
   { cout<<"NO"<<endl; return 0; }
  for (i=0;i<k-p;i++)
   ans[i].pb(KK[i]);
  lst=(k-p)%k;
  for (i=k-p;i<KK.size();i+=2)
   ans[lst].pb(KK[i]),
   ans[lst].pb(KK[i+1]),lst=(lst+1)%k;
  for (i=0;i<LL.size();i++)
   ans[lst].pb(LL[i]),lst=(lst+1)%k;
  for (i=0;i<k;i++)
   if (ans[i].size() == 0) { cout<<"NO"<<endl; return 0; }
  cout<<"YES"<<endl;
  for (i=0;i<k;i++)
   {
    cout<<ans[i].size();
    for (int j=0;j<ans[i].size();j++)
     cout<<" "<<ans[i][j];
    cout<<endl;
   }
 }
