#include<stdio.h>
int main()
{
	int n,k,p,i,c;
	
	while(scanf("%d%d%d",&n,&k,&p)!=EOF)
	{
		int a[100000],b[100000];
		int odd=0,even=0,temp;
		for(i=0;i<n;i++){
			scanf("%d",&c);
			if(c%2)
			a[odd++]=c;
			else
			b[even++]=c;
		}
		if(odd<k-p||(odd-k+p)%2==1||even+(odd-k+p)/2<p)
		printf("NO\n");
		else
		{
			printf("YES\n");
			temp=k-p;
			for(i=0;i<temp-1;i++)
			printf("1 %d\n",a[--odd]);
			for(i=0;i<p-1;i++)
			{
			   if(even)
			   printf("1 %d\n",b[--even]);
			   else 
			  {
			     printf("2 %d %d\n", a[odd-1], a[odd-2]);
                odd -= 2;
			 	
			}}
			if(p&&temp)
			printf("1 %d\n",a[--odd]);
			printf("%d",odd+even);
			while(odd)
			printf(" %d",a[--odd]);
			while(even)
			printf(" %d",b[--even]);
			printf("\n"); 
		}
		
	}
 }
