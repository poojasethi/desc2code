#include <cstdio>
#include <list>
using std::list;

int n,k,p,m;
int os[100009],ot,es[100009],et;
list<int> l[100009];

int main()
{
	scanf("%d%d%d",&n,&k,&p);
	while (n--)
	{
		scanf("%d",&m);
		if (m&1) os[ot++]=m;
		else es[et++]=m;
	}
	for (int i=p+1;i<=k;++i)
	{
		if (ot) l[i].push_back(os[--ot]);
		else goto NO;
	}
	
	for (int i=1;i<=p;++i)
	{
		if (et) l[i].push_back(es[--et]);
		else if (ot>=2)
		{
			l[i].push_back(os[--ot]);
			l[i].push_back(os[--ot]);
		}
		else goto NO;
	}
	while (et) l[k].push_back(es[--et]);
	if (!(ot&1)) while (ot) l[k].push_back(os[--ot]);
	else goto NO;
	puts("YES");
	for (int i=1;i<=k;++i)
	{
		printf("%lu ",l[i].size());
		for (list<int>::iterator j=l[i].begin();j!=l[i].end();++j) printf("%d ",*j);
		putchar('\n');
	}
	goto EXIT;
	NO:puts("NO");
	EXIT:return 0;
}
