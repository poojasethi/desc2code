#include <bits/stdc++.h>

using namespace std;

int N,gr,M,tmp,tmp1,a;
vector<int> v,v1,yaz[110000];

int main(){
	scanf(" %d %d %d",&N,&gr,&M);
	for(int i=1;i<=N;i++){
		scanf(" %d",&a);
		if(a%2)
			v1.push_back(a);
		else
			v.push_back(a);
	}
	if(v1.size()<gr-M || (v1.size()-(gr-M))%2 || (v1.size()-(gr-M))/2+v.size()<M){
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	for(int i=1;i<=gr;i++){
		if(i<=gr-M)
			yaz[i].push_back(v1[tmp1++]);
		else if(tmp<v.size())
			yaz[i].push_back(v[tmp++]);
		else{
			yaz[i].push_back(v1[tmp1++]);
			yaz[i].push_back(v1[tmp1++]);
		}
	}
	for(;tmp1<v1.size();)
		yaz[1].push_back(v1[tmp1++]);
	for(;tmp<v.size();)
		yaz[1].push_back(v[tmp++]);
	for(int i=1;i<=gr;i++){
		printf("%d ",(int)yaz[i].size());
		for(int j=0;j<yaz[i].size();j++)
			printf("%d ",yaz[i][j]);
		printf("\n");
	}
	return 0;
}
