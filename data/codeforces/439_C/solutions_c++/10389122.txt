#include <bits/stdc++.h>
#define F first
#define S second
#define SZ size()
#define PB push_back
#define REP(I, N) for (int I = 0; I < (N); ++I)
using namespace std;

int arr[123456];
vector<int> e, o;
vector<int> ans[123456];

int main(){
	int n, k, p; cin >> n >> k >> p;
	for (int i = 0; i < n; i++) cin >> arr[i];
	for (int i = 0; i < n; i++){
		if (arr[i]%2==0) e.PB(arr[i]);
		else o.PB(arr[i]);
	}
	int on = k-p;
	if (o.SZ<on || (o.SZ-on)%2!=0 || (o.SZ-on)/2 + e.SZ < p) {cout << "NO" << endl;}
	else {
		REP(i,on) ans[i].PB(o[i]);
		int nxt = on%k;
		for (int i = on; i < o.SZ; i+=2){
			ans[nxt].PB(o[i]);
			ans[nxt].PB(o[i+1]);
			nxt = (nxt+1)%k;
		}

		REP(i,e.SZ){
			ans[nxt].PB(e[i]);
			nxt = (nxt+1)%k;
		}

		cout << "YES" << endl;
		REP(i,k){
			cout << ans[i].size();
			REP(j,ans[i].size()) cout << " " << ans[i][j]; cout << endl;
		}
	}
	return 0;
}
