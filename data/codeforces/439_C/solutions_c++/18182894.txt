#include <cstdio>
#include <vector>
using namespace std;
const int maxn=1e5 ;
vector<int> arr[maxn] ;
vector<int> odi,eveni ;
void moving(vector<int> &a,vector<int> &b)
{
	a.push_back(b.back()) ;
	b.pop_back() ;
}
int main()
{
	int k,n,p,ds ;
	scanf("%d %d %d",&n,&k,&p) ;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&ds) ;
		if(ds%2)
			odi.push_back(ds) ;
		else
			eveni.push_back(ds) ;
	}
//	printf("%d %d\n",odi.size(),k-p) ;
	if(odi.size()<k-p || odi.size()%2!=(k-p)%2 || p>eveni.size()+(odi.size()-k+p)/2)
		printf("NO");
	else
	{
		int i ;
		for(i=0;i<k-p;i++)
			moving(arr[i],odi) ;
		while(!odi.empty())
		{
			i%=k ;
			moving(arr[i],odi) ;
			moving(arr[i],odi) ;
			i++ ;
		}
		while(!eveni.empty())
		{
			i%=k ;
			moving(arr[i],eveni) ;
			i++ ;
		}
		printf("YES") ;
		for(i=0;i<k;i++)
		{
			printf("\n%d ",arr[i].size()) ;
			for(int j=0;j<arr[i].size();j++)
				printf("%d ",arr[i][j]) ;
		}
	}
}
