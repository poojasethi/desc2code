#include<stdio.h>
__int64 a[100010],e[100010],o[100010];
int main()
{
	__int64 n,k,p,i,j,m,even=0,odd=0,ei=0,oi=0;
	scanf("%I64d%I64d%I64d",&n,&k,&p);
	for(i=0;i<n;i++)
	{
		scanf("%I64d",&a[i]);
		if(a[i]%2==0)
		{
			even++;
			e[ei++]=a[i];
		}
		else
		{
			odd++;
			o[oi++]=a[i];
		}
	}
	if(odd>=(k-p)&&(odd-(k-p))%2==0&&(even+(odd-(k-p))/2)>=p)
	{
		printf("YES\n");
		if(p>0)
			for(j=0;j<k-p;j++)
				printf("1 %I64d\n",o[j]);
		else
			for(j=0;j<k-1;j++)
				printf("1 %I64d\n",o[j]);
		for(m=0,i=0;m<p-1;m++)
			if(j<oi)
			{
				printf("2 %I64d %I64d\n",o[j],o[j+1]);
				j+=2;
			}
			else
				printf("1 %I64d\n",e[i++]);
		if(oi-j+ei-i)
			printf("%I64d",oi-j+ei-i);
		for(;i<ei;i++)
			printf(" %I64d",e[i]);
		for(;j<oi;j++)
			printf(" %I64d",o[j]);
		printf("\n");
	}
	else
		printf("NO\n");
	return 0;
}
