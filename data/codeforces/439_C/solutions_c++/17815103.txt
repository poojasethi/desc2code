#include <iostream>

using namespace std;

int main()
{
	int od[100000], ev[100000], n, k, p, m, o = 0, e = 0;
	cin >> n >> k >> p;
	for(int i = 0; i < n; i++)
	{
		cin >> m;
		if(m % 2)
		{
			od[o] = m;
			o++;
		}
		else
		{
			ev[e] = m;
			e++;
		}
	}
	if(o >= k-p && (k-p) % 2 == o % 2 && o+2*e >= k+p)
	{
		cout << "YES\n";
		while(k > p && k > 1)
		{
			cout << "1 " << od[o-1] << endl;
			o--;
			k--;
		}
		while(e*(k-1))
		{
			cout << "1 " << ev[e-1] << endl;
			e--;
			k--;
		}
		while(k > 1)
		{
			cout << "2 " << od[o-1] << ' ' << od[o-2] << endl;
			o -= 2;
			k--;
		}
		if(k)
		{
			cout << e+o << ' ';
			while(e > 0)
			{
				cout << ev[e-1] << ' ';
				e--;
			}
			while(o > 0)
			{
				cout << od[o-1] << ' ';
				o--;
			}
			cout << endl;
		}
	}
	else cout << "NO\n";
}
