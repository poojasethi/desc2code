#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,p;
	vector <int> even;
	vector <int> odd;
	cin>>n>>k>>p;
	int te=p;
	int to=k-p;
	for(int i=0;i<n;i++)
	{
		int x;
		cin>>x;
		if(x&1)
			odd.push_back(x);
		else
			even.push_back(x);
	}	
	int odr=odd.size()-to;
	int oz=odd.size();
	int ez=even.size();
	if( ((odr&1)==0)&&( to <= odd.size()  ) &&   (even.size() + odr/2 ) >=p)
	{	
		cout<<"YES"	<<endl;
		int i=0;
		for(;i<p && ez>0 && i<k-1;i++)	
		{
			cout<<"1 "<<even[--ez]<<endl;
		}
		for(;i<k-1 && i<p && oz>1;i++,oz-=2)
		{
			cout<<"2 "<<odd[oz-1]<<" "<<odd[oz-2]<<endl;
		}
		for(;i<k-1;i++)
			cout<<"1 "<<odd[--oz]<<endl;
		cout<<(oz+ez);
		while(oz)cout<<" "<<odd[--oz];
		while(ez)cout<<" "<<even[--ez];
	}
	else
	{
		cout<<"NO";
	}
	return 0;
}
