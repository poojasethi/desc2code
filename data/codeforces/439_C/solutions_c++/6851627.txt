#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
const int maxn=100050;
int n,K,p,A[maxn],anss=0;
vector<int> ans[maxn];
bool vis[maxn]={0};
int main()
{
	int i,j,odd=0;
	scanf("%d%d%d",&n,&K,&p);
	for(i=0;i<n;i++)
		scanf("%d",&A[i]),odd+=A[i]&1;
	if(odd<K-p||(odd+p-K)&1||(odd+p-K)/2+n-odd<p) { printf("NO\n");return 0; }
	printf("YES\n");
	for(i=0;i<n&&anss+p<K;i++)
		if(A[i]&1)
			ans[anss++].push_back(A[i]),vis[i]=1;
	for(j=0;i<n&&anss<K;i++)
		if(A[i]&1)
			ans[anss].push_back(A[i]),anss+=j,j=!j,vis[i]=1;
	for(i=0;i<n&&anss<K;i++)
		if(!(A[i]&1))
			ans[anss++].push_back(A[i]),vis[i]=1;
	for(;i<n;i++)
		if(!vis[i])
			ans[0].push_back(A[i]);
	for(i=0;i<anss;i++)
	{
		printf("%d",ans[i].size());
		for(j=0;j<ans[i].size();j++)
			printf(" %d",ans[i][j]);
		printf("\n");
	}
	return 0;
}
