#include<iostream>
#include<vector>
#include<stdio.h>
using namespace std;
vector <int> v1,v2,ans[100005];
int main()
{//freopen("i.txt","r",stdin);
int n,k,p1,p0; cin>>n>>k>>p0;
 p1=k-p0;
for(int i=0;i<n;i++)
 {int t; cin>>t;
  if (t%2==1) v1.push_back(t);
    else v2.push_back(t);
 }
int n1=v1.size(), n2=v2.size();

if (n1<p1|| (n1-p1)%2==1 || n2+(n1-p1)/2<p0) {cout<<"NO"; return 0;}

for (int i=0;i<p1;i++) ans[i].push_back(v1[i]);
int now=p1%k;
for (int i=p1;i<n1;i+=2)
{ans[now].push_back(v1[i]);
 ans[now].push_back(v1[i+1]);
 now=(now+1)%k;
}
for (int i=0;i<n2;i++)
{ans[now].push_back(v2[i]);
 now=(now+1)%k;
 }

cout<<"YES"<<endl;
for (int i=0;i<k;i++)
{cout<<ans[i].size()<<" ";
 for (int j=0;j<ans[i].size();j++) cout<<ans[i][j]<<" ";
 cout<<endl;
}
return 0;
}
