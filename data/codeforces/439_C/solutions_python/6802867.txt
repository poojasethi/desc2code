n, k, p = map(int, raw_input().split())
numbers = map(int, raw_input().split())

odd = [x for x in numbers if x & 1]
even = [x for x in numbers if not x & 1]

oddneeded = k - p
remainodds = len(odd) - oddneeded
totaleven = len(even) + remainodds / 2

boxes = [None] * k
for i in range(k):
    boxes[i] = []

if len(odd) < oddneeded or remainodds & 1 or totaleven < p or totaleven + oddneeded < k:
    print "NO"
else:
    print "YES"
    curbox = 0
    for i in xrange(oddneeded):
        boxes[curbox].append(odd[i])
        curbox = (curbox + 1) % k
    for i in xrange(oddneeded, len(odd), 2):
        boxes[curbox].append(odd[i])
        boxes[curbox].append(odd[i+1])
        curbox = (curbox + 1) % k
    for i in xrange(len(even)):
        boxes[curbox].append(even[i])
        curbox = (curbox + 1) % k
    for box in boxes:
        print len(box), ' '.join(map(str, box))
