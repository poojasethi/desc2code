import sys
inp = raw_input().split()
size_of_array , total_parts , even_parts= int(inp[0]),int(inp[1]),int(inp[2])
odd_parts = total_parts-even_parts
inp = raw_input().split()

odd = []
even= []
for i in range(size_of_array):
    inp[i] = int(inp[i])
    if inp[i]%2==1:
        odd.append(inp[i])
    else:
        even.append(inp[i])

if len(odd)>=odd_parts:
    parts = []
    for i in range(total_parts):
        parts.append([])
    even_counter=0
    repeat = 0
    feasible=1
    for item in even:
        if even_counter >= even_parts:
            repeat = 1
            even_counter=0
        parts[even_counter].append(item)
        even_counter+=1
    if not(repeat) and even_counter<even_parts:
        while(even_counter<even_parts):
            if ((len(odd))>=2):
              parts[even_counter].append(odd.pop())
              parts[even_counter].append(odd.pop())
              even_counter+=1
            else:
                feasible=0
                break
else:
    feasible = 0

if feasible:
    if len(odd)>=odd_parts:
        if (len(odd)-odd_parts)%2==0:
            odd_counter = even_parts
            for item in odd:
                if odd_counter < total_parts:
                    parts[odd_counter].append(item)
                    odd_counter+=1
                else:
                    if even_parts==total_parts:
                        even_parts=0
                    parts[even_parts].append(item)
        else:
            feasible=0
    else:
        feasible=0
if feasible:
    print "YES"
    for item in parts:
        sys.stdout.write(str(len(item)))
        for element in item:
            sys.stdout.write(' '+str(element))
        sys.stdout.write('\n')
else:
    print "NO"
