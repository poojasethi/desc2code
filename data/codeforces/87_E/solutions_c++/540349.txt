#include <map>
#include <set>
#include <cmath>
#include <stack>
#include <queue>
#include <string>
#include <vector>
#include <fstream>
#include <sstream>
#include <stdio.h>
#include <string.h>
#include <iostream>
#include <algorithm>
using namespace std;
#define li		long long
#define rep(i,to)	for(li i=0;i<((li)to);i++)
#define pb		push_back
#define sz(v)		((li)v.size())

#define P pair<li,li> 
vector< P > con[3];
vector< P > sum;
int main(){
	rep(i,3){
		li n;
		cin>>n;
		rep(j,n){
			li x,y;
			cin>>x>>y;
			con[i].pb(make_pair(x,y));
		}
	}
	li pos[3]={1,1,1};
	li start[3]={1,1,1};
	rep(i,3)rep(j,sz(con[i]))if(con[i][start[i]]<con[i][j]) pos[i]=start[i]=j;
#define x(a,b) (con[a][b].first)
#define y(a,b) (con[a][b].second)
	li sum_x=0;
	li sum_y=0;
	bool f[3]={true,true,true};
	while(pos[0]!=start[0]||pos[1]!=start[1]||pos[2]!=start[2]||f[0]||f[1]||f[2]){
		double ang[3]={0,0,0};
		li next[3]={(pos[0]+1)%sz(con[0]),(pos[1]+1)%sz(con[1]),(pos[2]+1)%sz(con[2])};
		rep(i,3){
			double xx=x(i,next[i])-x(i,pos[i]);
			double yy=y(i,next[i])-y(i,pos[i]);
			ang[i]=(yy==0)?(acos(-1)/2.0):acos(yy/hypot(xx,yy));
			if(0<xx) ang[i]=acos(-1)*2.0-ang[i];
			if(ang[i]==0) ang[i]=acos(-1)*2.0;
			if(pos[i]==start[i] && !f[i]) ang[i]=365;
		}
		li mini=0;
		rep(i,3)if(ang[i]<ang[mini]) mini=i;
		pos[mini]=next[mini];
		f[mini]=false;
		sum_x=sum_y=0;
		rep(i,3) sum_x+=x(i,pos[i]);
		rep(i,3) sum_y+=y(i,pos[i]);
		sum.pb(make_pair(sum_x,sum_y));
	}
	
	vector< P > low,high;
	li mini=0,maxi=0;
	li mini_y[2],maxi_y[2];
	rep(i,sz(sum))if(sum[i]<sum[mini]) mini=i;
	rep(i,sz(sum))if(sum[maxi]<sum[i]) maxi=i;
	for(int i=mini;i!=maxi;i=(i+1)%sz(sum))	low.pb(sum[i]);
	for(int i=mini;i!=maxi;i=(i+sz(sum)-1)%sz(sum)) high.pb(sum[i]);
	low.pb(sum[maxi]);
	high.pb(sum[maxi]);
	rep(i,2) mini_y[i]=sum[mini].second;
	rep(i,2) maxi_y[i]=sum[maxi].second;
	rep(i,sz(sum))if(sum[i].first==sum[mini].first){
		mini_y[0]=min(mini_y[0],sum[i].second);
		mini_y[1]=max(mini_y[1],sum[i].second);
	}
	rep(i,sz(sum))if(sum[i].first==sum[maxi].first){
		maxi_y[0]=min(maxi_y[0],sum[i].second);
		maxi_y[1]=max(maxi_y[1],sum[i].second);
	}
	sort(low.begin(),low.end());
	sort(high.begin(),high.end());
	li n;
	cin>>n;
	rep(i,n){
		li xx,yy;
		cin>>xx>>yy;
		xx*=3; 
		yy*=3;
		if(xx==sum[maxi].first){
			if(maxi_y[0]<=yy && yy<=maxi_y[1]) cout<<"YES"<<endl;
			else cout<<"NO"<<endl;
		}else if(xx==sum[mini].first){
			if(mini_y[0]<=yy && yy<=mini_y[1]) cout<<"YES"<<endl;
			else cout<<"NO"<<endl;
		}else if(sum[mini].first<xx && xx<sum[maxi].first){
			vector< P >::iterator lp0=upper_bound(low.begin(),low.end(),make_pair(xx,yy));
			vector< P >::iterator hp0=upper_bound(high.begin(),high.end(),make_pair(xx,yy));
			vector< P >::iterator lp1=lp0--;
			vector< P >::iterator hp1=hp0--;
			li ldx=lp1->first-lp0->first;
			li ldy=lp1->second-lp0->second;
			li hdx=hp1->first-hp0->first;
			li hdy=hp1->second-hp0->second;
			if(ldx*lp0->second-ldy*lp0->first+ldy*xx<=ldx*yy
			&& hdx*yy<=hdx*hp0->second-hdy*hp0->first+hdy*xx){
				cout<<"YES"<<endl;
			}else{
				cout<<"NO"<<endl;
			}
		}else{
			cout<<"NO"<<endl;
		}
	}
}
