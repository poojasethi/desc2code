#include <cstdio>

using namespace std;
typedef pair<int,int> pii;
typedef vector<pii> vpii;
typedef long long ll;

pii operator+(const pii& a,const pii& b) { return pii(a.first+b.first,a.second+b.second); }
pii operator-(const pii& a,const pii& b) { return pii(a.first-b.first,a.second-b.second); }

ll productVect(const pii& a,const pii& b)
{
	ll A = a.first;
	ll B = a.second;
	ll C = b.first;
	ll D = b.second;
	return A*D-B*C;
}

bool pii_comp_x( const pii& a,const pii &b)
{
	return a.first < b.first;
}

int inChain(const vpii& up,const vpii& dn, const pii& pnt)
{
	if (pnt.first < up[0].first) return 0;
	if (pnt.first > up.back().first) return 0;
	int i = lower_bound(up.begin(),up.end(),pnt,pii_comp_x)-up.begin();
	int j = lower_bound(dn.begin(),dn.end(),pnt,pii_comp_x)-dn.begin();
	int u = up.size();
	int d = dn.size();
	if ( i > 0 ) i--;
	if ( j > 0 ) j--;
	/*
	printf("pick %d,%d - %d,%d\n",up[i].first,up[i].second,up[i+1].first,up[i+1].second);
	printf("pick %d,%d - %d,%d\n",dn[i].first,dn[i].second,dn[i+1].first,dn[i+1].second);
	*/
	if ( productVect(up[i+1]-up[i], pnt-up[i]) > 0) return 0;
	if ( productVect(dn[j+1]-dn[j], pnt-dn[j]) < 0) return 0;
	return 1;
}

void convex_minko_sum(const vpii& a, const vpii& b, vpii& out)
{
	out.clear();
	int i=0,j=0;
	for (int q=0;q<a.size();++q) if (a[q]<a[i]) i=q;
	for (int q=0;q<b.size();++q) if (b[q]<b[j]) j=q;
	do
	{
		out.push_back( a[i]+b[j] );
		ll pro= productVect(a[(i+1)%a.size()]-a[i], b[(j+1)%b.size()]-b[j]);
		if (pro>=0)	i=(i+1)%a.size();
		if (pro<=0)	j=(j+1)%b.size();
	}
	while ( a[i]+b[j] != out[0] );
}

int main()
{
	vpii T[5];
	for (int q=0;q<3;++q)
	{
		int sz;
		scanf("%d",&sz);
		for (int w=0;w<sz;++w)
		{
			pii u;
			scanf("%d %d",&u.first,&u.second);
			T[q].push_back(u);
		}	
	}
	convex_minko_sum( T[0],T[1],T[3]);
	convex_minko_sum( T[2],T[3],T[4]);

	vpii up,down;// up - x is decreasing, down - x is increasing
	for (int q=0;q<T[4].size();++q)
	{
		pii a = T[4][q];
		pii b = T[4][(q+1)%T[4].size()];
		if (a.first < b.first) //go down!
		{
			if (down.empty() || down.back()!=a) down.push_back(a);
			down.push_back(b);
		}
		if (a.first > b.first) // go up!
		{
			if (up.empty() || up.back()!=a) up.push_back(a);
			up.push_back(b);
		}
	}
	reverse(up.begin(),up.end());

	//for (int q=0;q<up.size();++q) printf("%d,%d ",up[q].first,up[q].second); printf("\n");

		pii u;
		scanf("%d %d",&u.first,&u.second); u.first*=3; u.second*=3;
		printf("%s\n",inChain(up,down,u)?"YES":"NO");
	}

	return 0;
}
