#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#define N 200010
#define db double
#define eps (1e-6)
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const db ang=0.12321312321312;
struct ww {
	db x,y;
	inline bool operator == (const ww &A) {
		return abs(x-A.x)<eps&&abs(y-A.y)<eps;
	}
	inline void read() {
		int a,b;
		scanf("%d%d",&a,&b);
		x=cos(ang)*a-sin(ang)*b;
		y=sin(ang)*a+cos(ang)*b;
		//x=a,y=b;
	}
	inline ww operator - (const ww &A) {
		return (ww){x-A.x,y-A.y};
	}
	inline ww operator + (const ww &A) {
		return (ww){x+A.x,y+A.y};
	}
	inline db operator * (const ww &A) {
		return x*A.y-y*A.x;
	}
} a[N],b[N],c[N],O;
int i,j,k,n,m,K;
inline void get(ww a[N],int n) {
	int i,k=1;
	For(i,1,n) c[i]=a[i];
	For(i,2,n) if (a[i].x<a[k].x||a[i].x==a[k].x&&a[i].y<a[k].y) k=i;
	int t=0;
	for (i=k;i<=n;i++) a[++t]=c[i];
	For(i,1,k-1) a[++t]=c[i];
	a[n+1]=a[1]; a[n+2]=a[2];
}
inline void merge(ww a[N],ww b[N]) {
	int i,j,t=0;
	get(a,n); get(b,m);
	j=1; c[t=1]=a[1]+b[1];
	For(i,1,n+1) {
		ww A=a[i+1]-a[i];
		for (;;) {
			if (t>1&&c[1]==c[t]) break;
			ww B=b[j+1]-b[j];
			db C=A*B;
			if (A*B>eps) break;
			c[++t]=a[i]+b[j+1];
			j++;
		}
		if (t>1&&c[1]==c[t]) break;
		c[++t]=a[i+1]+b[j];
	}
	For(i,1,t) a[i]=c[i];
	n=t-1;
}
inline void pre() {
	scanf("%d",&n);
	For(i,1,n) a[i].read();
	scanf("%d",&m);
	For(i,1,m) b[i].read();
	merge(a,b);
	scanf("%d",&m);
	For(i,1,m) b[i].read();
	merge(a,b);
}
inline bool pan() {
	if (O==a[1]||O==a[K]) return 1;
	if (O.x<a[1].x||O.x>a[K].x) return 0;
	int l=1,r=K,mid;
	for (;l<=r;) {
		mid=(l+r)/2;
		if (a[mid].x<O.x) l=mid+1;
		else r=mid-1;
	}
	int id=l-1;
	db y=a[id].y+(a[id+1].y-a[id].y)*(O.x-a[id].x)/(a[id+1].x-a[id].x);
	if (O.y+eps<y) return 0;
	
	l=K,r=n+1;
	for (;l<=r;) {
		mid=(l+r)/2;
		if (a[mid].x>O.x) l=mid+1;
		else r=mid-1;
	}
	id=l-1;
	y=a[id].y+(a[id+1].y-a[id].y)*(O.x-a[id].x)/(a[id+1].x-a[id].x);
	if (O.y-eps>y) return 0;
	return 1;
}
int main() {
	//freopen("87e.in","r",stdin);
	pre();
	For(i,1,n) if (a[i].x>a[i+1].x) break;
	K=i;
	scanf("%d",&m);
	For(i,1,m) {
		O.read();
		O.x*=3,O.y*=3;
		printf("%s\n",pan()?"YES":"NO");
	}
	return 0;
}
