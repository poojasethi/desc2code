teams = int(raw_input())
kits = list()
counts = [0]*100000
for i in range(teams):
    kit = raw_input().split(' ')
    kits.append((int(kit[0]), int(kit[1])))
    counts[int(kit[0]) - 1] = counts[int(kit[0]) - 1] + 1
for i in range(teams):
    home = teams - 1 + counts[kits[i][1] - 1];
    print str(home) + ' ' + str(2 * teams - 2 - home)
