inputfunc = lambda:map(int,raw_input().split())
n,=inputfunc()
data=[inputfunc() for i in xrange(n)]
awayDict = {}
for row in data:
    homekit, awaykit = row
    if homekit in awayDict:
        awayDict[homekit]+=1
    else:
        awayDict[homekit]=1
mingame = (n-1) * 2
for row in data:
    homekit, awaykit = row
    if awaykit in awayDict:
        homes = n - 1 + awayDict[awaykit]
    else:
        homes = n - 1
    print homes, mingame - homes
