n = int(raw_input().rstrip('\n'))
moda = dict()
komands = []

for _ in xrange(n):
    d, g = map(int, raw_input().rstrip('\n').split())
    komands.append((d,g))
    if d in moda:
        moda[d] += 1
    else:
        moda[d] = 1

for komand in komands:
    m = 0
    if komand[1] in moda:
        m = moda[komand[1]]
    
    print str(n-1 + m) + ' ' + str(n-1 - m)
