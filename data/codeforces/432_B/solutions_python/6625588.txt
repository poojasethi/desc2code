from sys import stdin, stdout
def main():
    n = int(stdin.readline())
    a = [map(int, stdin.readline().split()) for i in xrange(n)]
    ans = []
    d = [0] * 10 ** 5
    for i in xrange(n):
        d[a[i][0]-1] += 1
    for i in xrange(n):
        ans.append((n - 1 + d[a[i][1]-1], n - 1 - d[a[i][1]-1]))
    stdout.write('\n'.join(' '.join(map(str, x)) for x in ans))
main()
