n = (int)(raw_input())
lista = [map(int,raw_input().split()) for i in range(n)]
listaHome = []
listaAway = []
cntHome = [0] * 100001
cntAway = [0] * 100001

for i in lista:
    listaHome.append(i[0])
    listaAway.append(i[1])

for i in range(n):
    cntHome[listaHome[i]] += 1
    cntAway[listaAway[i]] += 1

for i in xrange(n):
    print (n-1)+cntHome[listaAway[i]], (n-1)-cntHome[listaAway[i]];
