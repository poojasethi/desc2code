n = int(raw_input())
x = [0 for i in range(100000)]
y = [0 for i in range(100000)]
t = []
for i in range(n):
    x_, y_ = map(int, raw_input().split())
    x_ -= 1
    y_ -= 1
    x[x_] += 1
    y[y_] += 1
    t.append((x_, y_))

for i in range(n):
    cnt = n - 1
    cnt += x[t[i][1]]
    print cnt, 2 * (n - 1) - cnt
