n = int(raw_input())
kits = []
away = []

for i in range(n):
	kits.append([int(x) for x in raw_input().split()])
max = 0

for i in range(n):
	if kits[i][1]>max:
		max=kits[i][1]
	if kits[i][0]>max:
		max = kits[i][0]

for i in range(max+1):
	away.append(0)

for i in range(n):
	away[kits[i][0]]+=1

for i in range(n):
	print n-1+away[kits[i][1]], n-1-away[kits[i][1]]
