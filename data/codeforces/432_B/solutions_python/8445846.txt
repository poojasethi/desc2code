n = int(raw_input())
t = [map(int, raw_input().split()) for i in xrange(n)]
c = {}
for i in range(n):
    x, y = t[i]
    if not x in c:
        c[x] = {'h': [], 'g': []}
    if not y in c:
        c[y] = {'h': [], 'g': []}
    c[x]['h'].append(i)
    c[y]['g'].append(i)
r = [[n-1, n-1] for i in xrange(n)]
for v in c.values():
    h, g = v['h'], v['g']
    t = len(h)
    for i in g:
        r[i][0] += t
        r[i][1] -= t
for x, y in r:
    print x, y
