#coding:utf-8

num = int(raw_input())
x = [0 for i in range(100001)]
y = [0 for i in range(100001)]
cnt = [0 for i in range(100001)]
for i in range(0, num):
	x[i], y[i] = list(map(int, raw_input().split(' ')))
	cnt[x[i]] += 1
for i in range(0, num):
	a = num - 1 + cnt[y[i]]
	b = num - 1 - cnt[y[i]]
	print(str(a)+" "+str(b))
