n=int(raw_input())
ans=[[0,0] for i in range(n)]
h=[0]*100001
w=[0]*100001
l=[]
m=n
while m:
    m-=1
    u=map(int,raw_input().split())
    l.append(u)
    h[u[0]]+=1
    w[u[1]]+=1
for i in range(n):
    a=n-1
    b=0
    if h[l[i][1]]==0:
        b=n-1
    else:
        a+=h[l[i][1]]
        b+=n-1-h[l[i][1]]
    print a,b
