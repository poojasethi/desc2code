n = input()
h,a = {},{}
u = []
for i in xrange(n):
    x,y = map(int,raw_input().split())
    if x in h:
        h[x] += 1
    else:
        h[x] = 1
    if y in a:
        a[y] += 1
    else:
        a[y] = 1
    u.append([x,y])
for o in u:
    l = 0
    if o[1] in h:
        l = h[o[1]]
    print (n-1)+l,(n-1)-l
