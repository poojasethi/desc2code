n = int(raw_input())

home = [i for i in xrange(n)]
away = [i for i in xrange(n)]

for i in xrange(n):
	line =  raw_input()
	home[i], away[i] = map(int,line.split())

match = (n-1)
m = [[match, match] for i in xrange(n)]


x = [0 for i in xrange(110000)]

for i in xrange(n):
	x[home[i]] +=1
		

for i in xrange(n):
	if x[away[i]] != 0:
		m[i][0] += x[away[i]]
		m[i][1] -= x[away[i]]



for i in xrange(n):
	print m[i][0],m[i][1]
