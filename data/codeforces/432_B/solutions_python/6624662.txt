from collections import Counter

if __name__ == '__main__':
    n = input()
    team = [0] * n
    home = Counter()
    for i in xrange(n):
        x, y = map(int, raw_input().split())
        team[i] = y
        home[x] += 1
    for i in xrange(n):
        print n - 1 + home[team[i]], n - 1 - home[team[i]]
