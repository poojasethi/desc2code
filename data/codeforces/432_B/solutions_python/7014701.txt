n=input()
M=[[0,0] for i in xrange(n)]
home={}
Color=[]
for i in xrange(n):
    Color+=[map(int,raw_input().split())]

for line in Color:
    h,t=line
    if h in home:
        home[h]+=1
    else:
        home[h]=1

for i,line in enumerate(Color):
    h,t=line
    if t in home:
        num=home[t]
    else:
        num=0
    M[i][0]=n-1+num
    M[i][1]=n-1-num

for line in M:
    print " ".join(map(str,line))
