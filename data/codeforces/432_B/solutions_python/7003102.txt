MAX = 10**5+10
n = input()
x = list()
y = list()
numx = [0] * MAX 
numy = [0] * MAX 

for i in range(n):
    tempx, tempy = map(int, raw_input().split())
    x.append(tempx)
    y.append(tempy)
    numx[tempx] += 1
    numy[tempy] += 1

for j in y:
    #print j, numx[j]
    print n - 1 + numx[j], n - 1 -numx[j]
