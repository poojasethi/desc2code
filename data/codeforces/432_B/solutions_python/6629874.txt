n = int(raw_input())
Hs = {}
Gs = []
for i in range(n):
    x, y = map(int, raw_input().split())
    if x in Hs:
        Hs[x] += 1
    else:
        Hs[x] = 1
    Gs.append(y)

for g in Gs:
    if g in Hs:
        a = n - 1 + Hs[g]
        b = 2 * (n - 1) - a
    else:
        a = n - 1
        b = n - 1
    print a, b
