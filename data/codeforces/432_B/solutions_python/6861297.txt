n = int(raw_input())
home = [ n-1 ] * n
away = [ n-1 ] * n
hmkt = {}
awkt = []
for i in xrange(n):
	h, a = map(int, raw_input().split())
	if h not in hmkt:
		hmkt[h] = []
	hmkt[h].append(i)
	awkt.append(a)
for i in xrange(n):
	if awkt[i] in hmkt:
		length = len(hmkt[awkt[i]])
		home[i] += length
		away[i] -= length
	print "{} {}".format(home[i], away[i])
