n = input()
buf = []
r = n
cnt = {}
while r:
	buf.append([int(item) for item in raw_input().split()])
	if buf[-1][0] in cnt:
		cnt[buf[-1][0]] += 1
	else:
		cnt[buf[-1][0]] = 1
	r -= 1
m = 2*(n-1)
for i in range(n):
	(homeColor, awayColor) = buf[i]
	if awayColor in cnt:
		ans = n-1 + cnt[awayColor]
	else:
		ans = n-1
	print ans, m-ans
