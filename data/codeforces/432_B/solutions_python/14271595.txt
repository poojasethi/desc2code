n = int(raw_input())
a = []
b = []
h = dict()
for i in xrange(n):
    line = raw_input()
    a.append(int(line.split()[0]))
    b.append(int(line.split()[1]))
    if a[i] in h:
        h[a[i]]+=1
    else:
        h[a[i]] = 1
for i in xrange(n):
    x = 0
    if b[i] in h:
        x = h[b[i]]
    print str(n-1+x)+' '+str(n-1-x)
