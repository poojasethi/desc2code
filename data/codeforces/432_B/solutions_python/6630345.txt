teams = int(raw_input())

# First column is home color counts, second is away color counts
color_counts = [0]*100001

team_info = []
for i in range(teams):
	team_data = [int(d) for d in raw_input().strip().split()]
	team_info.append(team_data)
	
	color_counts[team_data[0]] += 1
	
for i in range(teams):
	home_jers = (teams-1) + color_counts[team_info[i][1]]
	away_jers = (teams-1) - color_counts[team_info[i][1]]
