t = map(int, raw_input().split())

N = t[0]

xa = [0 for x in xrange(100011)]
xb = [0 for x in xrange(100011)]


mp = [[] for x in xrange(100011)]

for i in xrange(N):
	mp[i] = map(int, raw_input().split())
	xa[mp[i][0]] = xa[mp[i][0]] + 1
	xb[mp[i][1]] = xb[mp[i][1]] + 1


for i in xrange(N):
	A = N - 1 + xa[mp[i][1]]
	print A , 2 * (N - 1) - A
