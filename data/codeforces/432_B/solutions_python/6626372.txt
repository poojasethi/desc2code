n=int(raw_input())
away=[]
cnts={}
for i in range(n):
    hm, aw=map(int, raw_input().split())
    if hm in cnts:
        cnts[hm]+=1
    else:
        cnts[hm]=1
    away.append(aw)
for i in range(len(away)):
    a=away[i]
    if a in cnts:
        x= (n-1+cnts[a],n-1-cnts[a])
    else:
        x=(n-1,n-1)
    print ' '.join(str(c) for c in x)
