#include<bits/stdc++.h>

using namespace std;

#define pii pair<int,int>
#define ll long long
#define N (int)(1e5+10)
#define mod 1000000007
#define mp make_pair
#define pb push_back
#define nd second
#define st first
#define inf mod
#define endl '\n'
#define sag (sol|1)
#define sol (root<<1)
#define ort ((bas+son)>>1)
#define bit(x,y) ((x>>y)&1)
#define int long long

int ata[N],ans[N];
vector< pair< pii , pii > > v;
vector< pair< pii , int > > A[N];
map<int,int> H;
int i,j,k,n,m,x,y,z,a;
int dep[N],S[N];
vector<int> arr[N],kob;

int atabul(int x){
	if(ata[x] == x)
		return x;
	return ata[x] = atabul(ata[x]);
}

int f(int x,int root){
	dep[x] = dep[root]+1;

	for(int i=0 ; i<arr[x].size() ; i++)
		if(arr[x][i]!=root)
			f(arr[x][i],x);
}

bool cmp(pair< pii , int > x,pair< pii ,int > y){
	return dep[x.st.st] > dep[y.st.st];
}

main(){
	cin >> n;

	for(i=1 ; i<n ; i++){
		scanf("%lld %lld %lld",&x,&y,&a);
		arr[x].pb(y);
		arr[y].pb(x);
		v.pb(mp(mp(a,i),mp(x,y)));
	}
	sort(v.begin(),v.end());

	int s = 1;
	H[v[0].st.st]=1;

	for(i=1 ; i<v.size() ; i++)
		if(v[i].st.st!=v[i-1].st.st)
			H[v[i].st.st]=++s;

	f(1,0);

	for(i=0 ; i<n-1 ; i++){
		if(dep[v[i].nd.st]>dep[v[i].nd.nd])
			swap(v[i].nd.st,v[i].nd.nd);
		A[H[v[i].st.st]].pb(mp(v[i].nd,v[i].st.nd));
	}

	for(i=1 ; i<=n ; i++){
		ata[i] = i;
		S[i] = 1;
	}

	for(i=1 ; i<=n ; i++){
		sort(A[i].begin(),A[i].end(),cmp);
		for(j=0 ; j<A[i].size() ; j++){
			int x = A[i][j].st.st;
			int y = A[i][j].st.nd;
			int atax = atabul(x);
			int atay = atabul(y);
			ans[A[i][j].nd] = S[atay];
			S[atax] += S[atay];
			ata[atay] = atax;
		}
		for(j=0 ; j<A[i].size() ; j++){
			int x = A[i][j].st.st;
			int atax = atabul(x);
			ans[A[i][j].nd] = (S[atax] - ans[A[i][j].nd]) * (ans[A[i][j].nd]);
		}
	}

	int mx = *max_element(ans+1,ans+n);

	for(i=1 ; i<n ; i++)
		if(ans[i] == mx)
			kob.pb(i);

	cout << 2*mx << ' ' << kob.size() << endl;

	for(i=0 ; i<kob.size() ; i++)
		printf("%lld ",kob[i]);

	puts("");
}
