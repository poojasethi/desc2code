import sys

def Can(mid):
  prices = ps[:mid]
  boys = bs[-mid:]
  cost = sum(max(0, p-b) for p, b in zip(prices, boys))
  return cost <= a

n, m, a = map(int, sys.stdin.readline().split())
bs = sorted(map(int, sys.stdin.readline().split()))
ps = sorted(map(int, sys.stdin.readline().split()))

lo = 1
hi = min(n, m)
while lo <= hi:
  mid = (lo + hi)/2;
  if Can(mid):
    lo = mid + 1
  else:
    hi = mid - 1

r = hi
s = max(0, sum(ps[:r])-a)
print '%d %d' % (r, s)
