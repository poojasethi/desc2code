#include<cstdio>
#include<algorithm>
#define nn 100010
using namespace std;
typedef long long ll;
ll ans;
int mid,n,m,a,i,l,r,p[nn],b[nn];
bool cmp(int a,int b){return a>b;}
bool ck()
 {int i;
  ll z=0;
  for (i=1;i<=mid;i++)if (b[i]<p[mid-i+1])
            z+=p[mid-i+1]-b[i];
  if (z>a)return false;
  return true;
 }
int main()
{
 scanf("%d%d%d",&n,&m,&a);
 for (i=1;i<=n;i++)scanf("%d",&b[i]);
 sort(b+1,b+n+1,cmp);
 for (i=1;i<=m;i++)scanf("%d",&p[i]);
 sort(p+1,p+m+1);
  n=m=min(n,m);
 l=0;r=m+1;
 while (l+1<r){mid=(l+r)>>1;
 if (ck())l=mid;else r=mid;
              }
 for (i=1;i<=l;i++)ans+=p[i];
 printf("%d %I64d\n",l,max(0ll,ans-a));
 return 0;
}
