#include <cstdio>
#include <algorithm>
using namespace std;

int b[100005], p[100005];
int main() {
    int n, m, a, i, l, r, mid;
    long long temp, ans = 0;
    scanf("%d%d%d", &n, &m, &a);
    for(i = 0; i < n; i++) scanf("%d", &b[i]);
    for(i = 0; i < m; i++) scanf("%d", &p[i]);
    sort(b, b + n); sort(p, p + m);
    l = 0; r = min(n, m);
    while(l < r) {
            mid = (l + r + 1) / 2; temp = 0;
            for(i = 0; i < mid; i++) temp += p[i] - min(b[n - mid + i], p[i]);
            if(temp <= a) l = mid;
            else r = mid - 1;
    } for(i = 0; i < l; i++) ans += p[i];
    if(ans - a < 0) ans = 0; else ans -= a;
    printf("%d %lld\n", l, ans);
    return 0;
}
