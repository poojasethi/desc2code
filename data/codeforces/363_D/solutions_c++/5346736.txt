#include<cstdio>
#include<algorithm>
using namespace std;

int n,m,k,p[100005],a[100005];

inline bool can(int nn){
  long long cen=k;
  int i=n-nn;
  int j=0;
  while(i<n){
    if(a[i]<p[j])cen-=p[j]-a[i];
    i++,j++;
  }
  return cen>=0;
}

main(){
  scanf("%d%d%d",&n,&m,&k);
  for(int i=0;i<n;i++)scanf("%d",a+i);
  for(int i=0;i<m;i++)scanf("%d",p+i);
  sort(a,a+n),sort(p,p+m);
  int lo=0,hi=min(n,m),ans;
  while(lo<=hi){
    int mid=lo+hi>>1;
    can(mid)
      ?lo=mid+1,ans=mid
      :hi=mid-1;
  }
  int sum=0;
  for(int i=0;i<ans;i++)sum+=p[i];
  printf("%d %d\n",ans,max(sum-k,0));
}
