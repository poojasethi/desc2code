#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long lld;
const lld maxn=100010;
lld p[maxn],b[maxn];
int main(){
    lld n,m,a,i,l,r,z,s;
    scanf("%I64d %I64d %I64d",&n,&m,&a);
    for (i=1;i<=n;i++) scanf("%I64d",&b[i]);
    for (i=1;i<=m;i++) scanf("%I64d",&p[i]);
    sort(b+1,b+n+1);
    sort(p+1,p+m+1);
    l=0;r=min(n,m)+1;
    while (r-l>1){
        s=(r+l)>>1;z=0;
        for (i=1;i<=s;i++) if (p[i]>b[n-s+i]) z+=p[i]-b[n-s+i];
        if (z>a) r=s; else
        l=s;
    }
    z=0;
    for (i=1;i<=l;i++) z+=p[i];
    if (a>=z) printf("%I64d 0\n",l); else
    printf("%I64d %I64d\n",l,z-a);
    return 0;
}
