#include <iostream>
#include <algorithm>
using namespace std;

int n, m, a;
int b[100000], p[100000];

int main() {
  cin >> n >> m >> a;
  for (int i = 0; i < n; ++i)
    cin >> b[i];
  for (int i = 0; i < m; ++i)
    cin >> p[i];
  sort(b, b + n);
  sort(p, p + m);
  int l = 0, r = n;
  if (m >= n)
    m = n;
  else
    l = n - m;
  while (l != r) {
    int mm = l + r >> 1;
    long long s = 0;
    for (int i = mm; i < n; ++i)
      if (b[i] < p[i - mm])
        s += p[i - mm] - b[i];
    if (s <= a)
      r = mm;
    else
      l = mm + 1;
  }
  long long s = -a;
  for (int i = l; i < n; ++i)
    s += p[i - l];
  cout << n - l << ' ' << max(0ll, s) << endl;
}
