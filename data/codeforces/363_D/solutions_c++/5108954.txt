#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

typedef long long int ll;

int n,m;
ll a,b[1000000],p[1000000];

bool sepuede(int k)
{
  if (k>n or k>m) return false;
  ll aa=a;
  for (int i=1;i<=k;i++) {
    if (b[n-i]<p[k-i]) {
      if (aa<p[k-i]-b[n-i]) return false;
      aa-=p[k-i]-b[n-i];
    }
  }
  return true;
}

int main()
{
  cin>>n>>m>>a;
  for (int i=0;i<n;i++)
    cin>>b[i];
  for (int i=0;i<m;i++)
    cin>>p[i];
  sort(b,b+n);
  sort(p,p+m);
  int inf=0;
  int sup=n+1;
  while (inf+1<sup) {
    int med=(inf+sup)/2;
    if (sepuede(med))
      inf=med;
    else
      sup=med;
  }
  ll sp=0;
  for (int i=0;i<inf;i++) {
    sp+=p[i];
  }
  cout<<inf<<" "<<max(0LL,sp-a)<<endl;
}
