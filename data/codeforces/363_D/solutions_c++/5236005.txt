#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
int v,n,m,a[120000],b[120000],l,r,mid,ans,i;
bool check(int mid){
    int i,res=v,x;
    ans=0;
    for(i=1;i<=mid;++i){
        x=max(0,b[i]-a[n-mid+i]);
        if(res<x)return 0;
        res-=x;
        ans+=b[i]-x;
    }
    ans=max(0,ans-res);
    return 1;
}
int main(){
    scanf("%d%d%d",&n,&m,&v);
    for(i=1;i<=n;++i)scanf("%d",&a[i]);
    for(i=1;i<=m;++i)scanf("%d",&b[i]);
    sort(a+1,a+n+1);
    sort(b+1,b+m+1);
    l=0; r=min(n,m);
    while(l<=r){
        mid=(l+r)/2;
        if(check(mid))l=mid+1;
        else r=mid-1;
    }
    check(l-1);
    printf("%d %d\n",l-1,ans);
}
