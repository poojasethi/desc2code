# include<stdio.h>
# include<iostream>
# include<string.h>
# include<algorithm>
using namespace std;
long long b[100005],p[100005];
int main()
{
	ios::sync_with_stdio(0);
	long long n,m,a; cin>>n>>m>>a;
	long long i,l,r;
	for (i=1;i<=n;i++) cin>>b[i];
	for (i=1;i<=m;i++) cin>>p[i];
	sort(b+1,b+1+n); sort(p+1,p+1+m);
	l=0,r=min(n,m); long long need;
	while (l<r)
	{
		long long mid=(l+r>>1)+1;
		need=0;
		for (i=1;i<=mid;i++)
		    if (b[n-mid+i]<p[i])
		        need+=p[i]-b[n-mid+i];
        if (need>a)
            r=mid-1;
        else
            l=mid;
	}
	need=-a;
	for (i=1;i<=l;i++)
         need+=p[i];
    if (need<0) need=0;
    cout<<l<<' '<<need<<endl;
	return 0;
}
