#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;
int n,m,a,l=0,r;
long long x[100005],y[100005],w=0,g;
bool cmp(int h,int l){
	return h>l;
}
bool check(int k){
	long long t=a;
	for (int i=k-1,j=0;i>=0 && t>=0;i--,j++)
		if (x[i]<y[j]){
			t-=y[j]-x[i];
			if (t<0) return false;
		}
	return true;
}
int main( ){
	cin>>n>>m>>a;
	for (int i=0;i<n;i++) scanf("%d",&x[i]);
	for (int i=0;i<m;i++) scanf("%d",&y[i]);
	sort(x,x+n,cmp);
	sort(y,y+m);
	r=min(n,m);
	while (l<r){
		g=(l+r+1)/2;
		if (check(g)) l=g; else r=g-1;
	}
	w=0;
	for (int i=0;i<l;i++)
		w+=y[i];
	w-=a;
	if (w<0) w=0;
	cout<<l<<' '<<w<<endl;
	return 0;
}
