#include<stdio.h>
#include<algorithm>
using namespace std;
int b[100000];
int p[100000];

long long min(long long a,long long b){return a<b?a:b;}
long long max(long long a,long long b){return a>b?a:b;}

int main(){
    int n,m,a,i;
    scanf("%d %d %d",&n,&m,&a);
    for(i=0;i<n;i++)scanf("%d",&b[i]);
    for(i=0;i<m;i++)scanf("%d",&p[i]);
    sort(b,b+n);
    sort(p,p+m);
    int l=0,r=min(n,m)+1;
    while(r-l>1){
        int mid=(l+r)/2;
        long long s=0;
        for(i=0;i<mid;i++)s+=max(p[i]-b[n-mid+i],0);
        if(s<=a)l=mid;
        else r=mid;
    }
    long long s=0;
    for(i=0;i<l;i++)s+=p[i];
    printf("%d %I64d\n",l,max(s-a,0));
    return 0;
}
