#include<stdio.h>
#include<stdlib.h>
#include<vector>
#include<iostream>
#include<math.h>
#include<map>
#include<algorithm>
#include<string>
#include<string.h>
#include<set>
using namespace std;

int main()
{
	long long int i,j,k,n,m,a,s,r,l,h,mid;
	cin >> n >> m >> a;
	long long int b[n],p[m];
	for(i=0;i<n;i++)
		cin >> b[i];
	for(i=0;i<m;i++)
		cin >> p[i];
	sort(b,b+n);
	sort(p,p+m);
	l=0,h=n;
	while(l<=h)
	{
		mid=(l+h)/2;
		if(mid>m || mid>n)
		{
			h=mid-1;
			continue;
		}
		j=0;k=0;
		for(i=0;i<mid;i++)
		{
			if(p[i]-b[n-mid+i]>0)
				j+=p[i]-b[n-mid+i];
			k+=p[i];
		}
		if(j<=a)
		{
			l=mid+1;
			r=mid;
			s=0;
			if(k-a>0)
				s=k-a;
		}
		else
			h=mid-1;
	}
	cout << r <<  " " << s << endl;
	return 0;
}
