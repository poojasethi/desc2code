#include <algorithm>
#include <cstdio>
using namespace std;
const int N=110000;
int n,m,i,l,r,t,s,ans,a[N],b[N];
bool check(int k) {
	long long t=0;
	for (int i=0;i<k;i++)
		t+=max(b[i]-a[i+n-k],0);
	return t<=s;
}
int main() {
	scanf("%d%d%d",&n,&m,&s);
	for (i=0;i<n;i++) scanf("%d",a+i);
	for (i=0;i<m;i++) scanf("%d",b+i);
	sort(a,a+n);sort(b,b+m);
	l=0;r=min(n,m);
	while (l<r) {
		int M=(l+r)/2+1;
		if (check(M)) l=M;
		else r=M-1;
	}
	for (i=0;i<l;i++) t+=b[i];
	ans=max(0,t-s);
	printf("%d %d\n",l,ans);
	return 0;
}
