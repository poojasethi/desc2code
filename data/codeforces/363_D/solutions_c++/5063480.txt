#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 100005
#define ll long long
using namespace std;
int a[maxn],b[maxn],N,M,S;

bool ok(int x)
{
    ll s=0;
    for (int i=1; i<=x; i++) s+=max(b[i]-a[N-x+i],0);
    return s<=S;
}

int main()
{
    scanf("%d%d%d",&N,&M,&S);
    for (int i=1; i<=N; i++) scanf("%d",&a[i]);
    for (int i=1; i<=M; i++) scanf("%d",&b[i]);
    sort(a+1,a+N+1),sort(b+1,b+M+1);
    int l=0,r=min(N,M);
    for (;l<=r;)
    {
        int m=(l+r)>>1;
        if (ok(m)) l=m+1; else r=m-1;
    }
    ll s=0;
    for (int i=1; i<=r; i++) s+=b[i];
    printf("%d %I64d\n",r,max(s-S,0ll));
}
