#include <bits/stdc++.h>
using namespace std;
#define ll long long


ll a[100005];
ll b[100005];

int main() {
	ios_base::sync_with_stdio(0);
	int n, m;
	ll p;
	cin >> n >> m >> p;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	for (int j = 0; j < m; j++) {
		cin >> b[j];
	}
	sort(a, a + n);
	sort(b, b + m);
	int lo = 0, hi = min(n, m);
	ll c;
	while (hi > lo) {
		int mi = (hi + lo+1) / 2;
		c = 0;
		for (int i = 0; i < mi; i++) {
			c += max(0LL, b[mi - 1 - i] - a[n - 1 - i]);
		}
		if (c <= p) lo = mi;
		else hi = mi-1;
		//cout<<hi<<" "<<lo<<" "<<mi<<" "<<c<<endl;
	}
	c=0;
	for(int i=0; i<lo; i++) {
		c+=b[lo-1-i];
	}
	cout<<lo<<" "<<max(c-p,0LL)<<endl;
	return 0;
}
