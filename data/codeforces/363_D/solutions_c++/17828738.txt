#include<bits/stdc++.h>
using namespace std;
const int N = 100000+10;
int g;
int a[N],b[N];
bool f(int k,int n){
	if(k>=n) return 0;
	for(int i=0;i<=k;i++){
		if(a[n-1-i]<b[k-i]) g-=b[k-i]-a[n-1-i];
		if(g<0) return 0;
	}
	return 1;
}
int main(){
	int n,m,x; 
	cin>>n>>m>>x;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	for(int i=0;i<m;i++){
		cin>>b[i];
	}
	sort(a,a+n);
	sort(b,b+m);
	int l=-1,r=m;
	while(r-l>1){
		g=x;
		int vasat=(r+l)/2;
		if(f(vasat,n)) l=vasat;
		else r=vasat;
	}
	cout<<l+1<<" ";
	int jam=0;
	for(int i=0;i<=l;i++)	jam+=b[i];
	cout<<max(0,jam-x);
}
