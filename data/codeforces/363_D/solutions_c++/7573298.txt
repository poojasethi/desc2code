#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n,m,s,a[100005],b[100005];

int ck(int x)
{
	int flag=0;
	long long save=0;
	for(int j=1;j<=x;j++)
	{
		int i=n-x+j;
		save+=b[j]-min(a[i],b[j]);
	}
	return save<=(long long)s;
}

int main()
{
	scanf("%d%d%d",&n,&m,&s);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=m;i++)scanf("%d",&b[i]);
	sort(b+1,b+m+1);
	sort(a+1,a+n+1);
	int L=0,R=min(n,m);
	while(L+1<R)
	{
		int mid=(L+R)>>1;
		if(ck(mid))L=mid;else R=mid;
	}
	if(ck(R))L=R;
	long long l=0;
	for(int i=1;i<=L;i++)l+=b[i];
	printf("%d %d\n",L,max(0ll,(long long)(l-s)));
	return 0;
}
