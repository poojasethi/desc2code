#include <cmath>
#include <ctime>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
#define LL long long
using namespace std;
LL N,M,A,cost,ans;
LL B[100010],P[100010];
bool ok(LL x){
	cost=0;
	for (LL i=1;i<=x;i++)
	cost+=max(P[i]-B[N-x+i],0ll);
	if (cost>A)return false;
	return true;
}
int main(){
	scanf("%I64d%I64d%I64d",&N,&M,&A);
	for (LL i=1;i<=N;i++)scanf("%I64d",&B[i]);
	for (LL i=1;i<=M;i++)scanf("%I64d",&P[i]);
	sort(B+1,B+N+1);sort(P+1,P+M+1);
	LL l=0,r=min(N,M);
	while (l<r){
		LL mid=(l+r+1)/2;
		if (ok(mid))l=mid;
		else r=mid-1;
	}
	for (LL i=1;i<=l;i++)ans+=P[i];
	ans=max(ans-A,0ll);
	printf("%I64d ",l);
	printf("%I64d",ans);
}
