#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int n,m,a;
long long b[100001];
long long p[100001];
bool ok(int x)
{
    long long s=0;
    for(int i=1;i<=x;i++) s+=max(p[i]-b[n-x+i],0ll);
    return s<=a;
}
int main()
{
    cin>>n>>m>>a;
    for(int i=1;i<=n;i++) cin>>b[i];
    for(int i=1;i<=m;i++) cin>>p[i];
    sort(b+1,b+n+1);
    sort(p+1,p+m+1);
    int l=0,r=min(n,m);
    for(;l<=r;)
    {
        int m=(l+r)>>1;
        if(ok(m)) l=m+1;  else r=m-1;
    }
    long long s=0;
    for(int i=1;i<=r;i++)   s+=p[i];
    cout<<r<<" "<<max(s-a,0ll)<<endl;
    return 0;
}
