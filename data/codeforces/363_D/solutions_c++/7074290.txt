#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
int n,m;
long long a,b[100000],p[100000],u;
bool f(int k){
	if (k>n||k>m) return false;
	long long a1=a;
	for(int i=1;i<=k;i++){
		if (b[n-i]<p[k-i]){
			if (a1<p[k-i]-b[n-i]) return false;
			a1-=p[k-i]-b[n-i];
		}
	}
	return true;
}
int main(){
	cin>>n>>m>>a;
	for(int i=0;i<n;i++)
		cin>>b[i];
	for(int i=0;i<m;i++)
		cin>>p[i];
	sort(b,b+n);
	sort(p,p+m);
	int t=0,w=n+1,m;
	while (t+1<w){
		m=(t+w)/2;
		if (f(m)){
			t=m;
		}
		else{
			w=m;
		}
	}
	long long ans=0;
	for(int i=0;i<t;i++){
		ans+=p[i];
	}
	cout<<t<<" "<<max(u,ans-a)<<endl;
}
