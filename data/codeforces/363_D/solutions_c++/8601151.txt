#include <iostream>
#include <algorithm>
using namespace std;

#define int long long

const int N = 1e5 + 5;
int n, m, a;
int b[N];
int p[N];

bool valid(int sol)
{
	int s = 0, ss = 0;
	for (int i = 0; i < sol; ++i)
		s += min(0ll, b[n-1-i]-p[sol-i-1]);
	return s+a >= 0;
}

main()
{
	cin >> n >> m >> a;
	for (int i = 0; i < n; ++i)
		cin >> b[i];
	for (int i = 0; i < m; ++i)
		cin >> p[i];
	sort(b, b+n);
	sort(p, p+m);
	int l = 0, r = min(n, m) + 1;
	while (r-l > 1)
	{
		int mid = (r+l)/2;
		if (valid(mid))
			l = mid;
		else
			r = mid;
	}
	int s = 0;
	for (int i = 0; i < l; ++i)
		s += p[i];
	cout << l << ' ' << max(0ll, s-a) << '\n';
	return 0;
}
