#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
int k,n,m,b[100010],p[100010],ans(0),ans1(0);
bool cmp1(int x,int y){return x>y;}
bool cmp2(int x,int y){return x<y;}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++)scanf("%d",&b[i]);
	for(int i=1;i<=m;i++)scanf("%d",&p[i]);
	sort(b+1,b+n+1,cmp1);
	sort(p+1,p+m+1,cmp2);
	int l=0,r=min(n,m);
	while(l<=r){
		int mid=(l+r)>>1,sum(0);
		bool flag(1);
		for(int i=1;i<=mid;i++)
			if(p[i]>b[mid+1-i]){
				sum+=p[i]-b[mid+1-i];
				if(sum>k){flag=0;break;}
			}
		if(flag){l=mid+1;ans=mid;}else r=mid-1;
	}
	for(int i=1;i<=ans;i++)ans1+=p[i];
	printf("%d %d",ans,max(ans1-k,0));
	return 0;
}
