#include <algorithm>
#include <cstdio>
#include <iostream>

using namespace std;

typedef long long int Lint;

const int MAXN=1e5+10;

int N,M;
int a[MAXN];
int b[MAXN];

Lint S;

int main(){
	
	scanf(" %d %d %I64d",&N,&M,&S);
	
	for( int i=1 ; i<=N ; i++ )
		scanf(" %d",a+i);
	for( int i=1 ; i<=M ; i++ )
		scanf(" %d",b+i);
	
	sort(a+1,a+N+1);
	sort(b+1,b+M+1);
	
	int lo=0,hi=min(N,M);
	
	while( lo<=hi ){
		
		int mi=(lo+hi)>>1;
		
		Lint s=0;
		
		for( int i=1 ; i<=mi ; i++ )
			s+=max(b[i]-a[N-mi+i],0);
		
		if( s<=S )
			lo=mi+1;
		else
			hi=mi-1;
	}
	
	Lint s=0;
	
	for( int i=1 ; i<=hi ; i++ )
		s+=b[i];
	
	printf("%d %I64d",hi,max(s-S,0LL));
	return 0;
}
