#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long ll;

ll b[100001],p[100001];
ll a;
int n,m;

bool judge(int cnt)
{
	int sum=0;
	int i=n-1,j=cnt-1;
	while(cnt)
	{
		if(p[j]>b[i]) sum+=p[j]-b[i];
		if(sum>a) return 0;
		--i,--j,--cnt;
	}
	return 1;
}

int main()
{
	cin>>n>>m>>a;
	for(int i=0;i<n;++i)
		cin>>b[i];
	sort(b,b+n);
	for(int j=0;j<m;++j) 
		cin>>p[j];
	sort(p,p+m);
	int l=0,r=min(n,m);
	while(l<r)
	{
		int mid=(l+r+1)>>1;
		if(judge(mid)) l=mid;
		else r=mid-1;
	}
	int ans=0;
	for(int i=0;i<l;++i)
		ans+=p[i];
	ans=(ans-a)<0?0:(ans-a);
	cout<<l<<" "<<ans<<endl;
	return 0;
}
