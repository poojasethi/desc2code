#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
using namespace std;
int n,m,a;
int b[111111],p[111111],dem[111111],sl,s;
int main(){
    cin>>n>>m>>a;
    for (int i=0;i<n;i++) cin>>b[i];
    for (int i=0;i<m;i++) {
        cin>>p[i];
        dem[i]=0;
    }
    sort(b,b+n);
    sort(p,p+m);
    int l,r,mid;
    l=0;r=min(n,m);
    while (l<r){
        mid=(l+r+1)/2;
        int t;
        t=a;
        for (int i=0;i<mid&&t>=0;i++) t-=max(0,p[i]-b[n-mid+i]);
        if (t<0) r=mid-1;else l=mid;
    }
    for (int i=0;i<l;i++) s+=p[i];
    s-=a;
    if (s<0) s=0;
    cout<<l<<" "<<s;
	return 0;
}
