#include <bits/stdc++.h>

using namespace std;

int b[123456], p[123456];
int a, ans, cost;

int main() {
  int n, m;
  scanf("%d %d %d", &n, &m, &a);
  for (int i = 0; i < n; i++) scanf("%d", b + i);
  for (int i = 0; i < m; i++) scanf("%d", p + i);
  sort(b, b + n);
  sort(p, p + m);
  int l = 0, r = min(n, m), cost = -a;
  while (l <= r) {
    int mid = (l + r) >> 1;
    int u = a, f = 1;
    for (int i = 0; i < mid; i++) {
      if (b[n - mid + i] + u < p[i]) {
        f = 0;
        break;
      }
      u -= max(0, p[i] - b[n - mid + i]);
    }
    f ? l = mid + 1 : r = mid - 1;
  }
  for (int i = 0; i < r; i++) cost += p[i];
  printf("%d %d", r, max(0, cost));
  return 0;
}
