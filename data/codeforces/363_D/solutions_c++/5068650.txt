#include <cstdio>
#include <algorithm>

#define N 100000

using namespace std;

int a[N], b[N], n, m, t, c, i, l, r;
long long s = 0;

bool g(int p)
{
	long long s = 0;
	for (int i = n - 1; i >= n - p; i --)
		if (a[p - n + i] > b[i])
			s += a[p - n + i] - b[i];
	return s <= t;
}

int main()
{
	scanf("%d%d%d", &n, &m, &t);

	for (i = 0; i < n; i ++)
		scanf("%d", &b[i]);

	for (i = 0; i < m; i ++)
		scanf("%d", &a[i]);

	sort(a, a + m);
	sort(b, b + n);

	for (l = 0, r = n < m ? n : m; l <= r;)
	{
		int m = (l + r) >> 1;
		if (g(m))
			l = m + 1;
		else
			r = m - 1;
	}

	for (i = 0; i < r; i ++)
		s += a[i];

	printf("%d %I64d\n", r, s > t ? s - t : 0);

	return 0;
}
