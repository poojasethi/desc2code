#include <iostream>
#include <algorithm>
using namespace std;

#define MAXN 100010
int boys[MAXN],bikes[MAXN];
int n,m,a;

bool check(int pos) {
    int l=a,cnt=pos+1;
    for(int i=n-1,j=pos;i>=0&&j>=0;i--,j--){
	if(boys[i]+l<bikes[j]) return false;
	if(boys[i]<bikes[j])
	    l-=bikes[j]-boys[i];
	cnt--;
    }
    return (cnt==0);
}

int main(void){
    cin>>n>>m>>a;
    for(int i=0;i<n;i++)
	cin>>boys[i];
    for(int i=0;i<m;i++)
	cin>>bikes[i];
    sort(boys,boys+n);
    sort(bikes,bikes+m);
    int l=0,r=m,mid;
    while(l<r){
	mid=(l+r)/2;
	if(check(mid)) 
	    l=mid+1;
	else
	    r=mid;
    }
    int sum=0;
    for(int i=0;i<l;i++)
	sum+=bikes[i];
    cout<<l<<" "<<max(sum-a,0)<<endl;
}
