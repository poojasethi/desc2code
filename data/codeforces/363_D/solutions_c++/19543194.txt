#include <iostream>
#include <algorithm>
#include <vector>
#define ll long long int

using namespace std;

int main()
{
	ll n, m, a;
	ll l, h, mid, i, c, x;
	cin >> n >> m >> a;
	vector<ll> p;
	vector<ll> b;

	for (i = 0; i < n; i++) {
		cin >> x;
		p.push_back(x);
	}
	for (i = 0; i < m; i++) {
		cin >> x;
		b.push_back(x);
	}
	sort(p.begin(), p.end());
	sort(b.begin(), b.end());

	l = 0; 
	h = min(n, m);
	while (l < h) {
		c = 0;
		mid = (l+h+1)/2;
		for (i = 0; i < mid; i++) {
			c = c + max(0LL, b[i]-p[n+i-mid]);
		}
		if (c <= a) {
			l = mid;
		} else {
			h = mid-1;
		}
	}
	c = 0;

	for (i = 0; i < l; i++) {
		c += b[i];
	}
	c = max(0LL,c-a);


	cout << l << " " << c << endl;
	return 0;

}
