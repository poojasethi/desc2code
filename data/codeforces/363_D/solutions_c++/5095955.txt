#include<cstdio>
#include<algorithm>
using namespace std;
int a[100005], b[100005];
int main()
{
    int n,m,d;
    long long total;
    scanf("%d %d %d",&n,&m,&d);
    for(int i=0;i<n;i++)scanf("%d",&a[i]);
    for(int i=0;i<m;i++)scanf("%d",&b[i]);
    int l = 0, r = min(n,m), mid;
    sort(a,a+n);
    sort(b,b+m);
    while(l<=r){
        mid = (l+r)/2;
        int t=n-mid;
        total = 0;
        for(int i=0;i<mid;i++){
            total += max(b[i]-a[t+i],0);
        }
        if(total > d)r = mid - 1;
        else if(total <= d)l = mid + 1;
    }
    printf("%d ",r);
    total = 0;
    for(int i=0;i<r;i++)total+=b[i];
    printf("%d\n",max(total - d, 0ll));
}
