#include<iostream>
#include<string>
#include<algorithm>
#include<vector>
using namespace std;

int a[10][100],b[10][100],c[10][100],k;

int n,m;

struct pancho{
	int id,val;
	pancho(){
	}
	pancho(int Id,int Val){
		id=Id;
		val=Val;
	}
	bool operator<(pancho P)const{
		return val>P.val;
	}
};

int f(int x,int y){
	int K=k;
	vector<pancho> v;
	for(int i=0;i<m;++i){
		if(a[x][i]<b[y][i]){
			v.push_back(pancho(i,b[y][i]-a[x][i]));
		}
	}
	sort(v.begin(),v.end());
	int ret=0;
	for(int i=0;i<v.size();++i){
		int tem=min(c[x][v[i].id],K);
		ret+=tem*v[i].val;
		K-=tem;
	}
	return ret;
}

int main(){
	cin>>n>>m>>k;
	string s;
	for(int i=0;i<n;++i){
		cin>>s;
		for(int j=0;j<m;++j)cin>>a[i][j]>>b[i][j]>>c[i][j];		
	}
	int res=0;
	for(int i=0;i<n;++i){
		for(int j=i+1;j<n;++j){
			res=max(res,f(i,j));
			res=max(res,f(j,i));
		}
	}
	cout<<res<<"\n";
}
