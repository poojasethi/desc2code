#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m,K,ans=0;
char name[1005];
int a[105][105],b[105][105],c[105][105],f[1005];
int main()
{
	scanf("%d%d%d",&n,&m,&K);
	for (int i=1;i<=n;i++){
		scanf("%s",name);
		for (int j=1;j<=m;j++) scanf("%d%d%d",&a[i][j],&b[i][j],&c[i][j]);
		}
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++) if (i!=j){
			memset(f,0,sizeof(f));
			for (int k=1;k<=m;k++) if (a[i][k]<b[j][k])
				for (int C=1;C<=c[i][k];C++)
					for (int v=K;v;v--)
						f[v]=max(f[v],f[v-1]+b[j][k]-a[i][k]);
			ans=max(ans,f[K]);
			}
	printf("%d\n",ans);
	return 0;
}
