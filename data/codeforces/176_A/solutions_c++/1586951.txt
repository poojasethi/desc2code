#include <iostream>
#include <algorithm>
#include <string>
#include <utility>
#include <cstdio>
using namespace std;

#define ii pair<int, int>

int main()
{
	int n,m,k;
	string name[10];
	int buy[10][100];
	int sell[10][100];
	int cnt[10][100];
	ii opt[100];
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
	{
		cin>>name[i];
		for(int j=0;j<m;j++)
		{
			cin>>buy[i][j]>>sell[i][j]>>cnt[i][j];
		}
	}	
	int ans = 0;
	for(int i1 = 0; i1<n;i1++)
	{
		for(int i2=0;i2<n;i2++)
		{
			if(i1==i2)continue;
			for(int i=0;i<m;i++)
				opt[i]=ii(sell[i2][i]-buy[i1][i], cnt[i1][i]);
			sort(opt, opt+m);
			int ta = 0;
			int t = k;
			for(int i=m-1;i>=0;--i)
				if(opt[i].first>0)
				{
					ta+=opt[i].first*min(opt[i].second, t);
					t-=min(t, opt[i].second);
				}
			ans = max(ans, ta);
		}
	}
	cout << ans <<endl;
	return 0;
}
