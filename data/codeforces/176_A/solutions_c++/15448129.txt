#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long LL;
char s[1005];
LL n,m,k,a[15][105],b[15][105],c[15][105],i,j;
LL dp[105],ii,jj,tem,l,ans;
int main(){
	scanf("%I64d%I64d%I64d",&n,&m,&k);
	for(i=0;i<n;++i){
		scanf("%s",s);
		for(j=0;j<m;++j){
			scanf("%I64d%I64d%I64d",&a[i][j],&b[i][j],&c[i][j]);
		}
	}
	for(i=0;i<n;++i){
		for(j=0;j<n;++j){
			if(i==j)continue;
			memset(dp,0,sizeof(dp));
			for(l=0;l<m;++l){
				tem=-a[j][l]+b[i][l];
				if(tem<=0)continue;
				for(jj=k;jj;--jj){
					for(ii=1;ii<=min(jj,c[j][l]);++ii){
						dp[jj]=max(dp[jj],dp[jj-ii]+ii*tem);
					}
				}
			}
			ans=max(ans,dp[k]);
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
