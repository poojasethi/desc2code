#include <iostream>
#include <cstdio>
#include <cstring>
#define MAXN 15
#define MAXK 105
using namespace std;

int N,M,K,dp[MAXN][MAXK],A[MAXN][MAXK],B[MAXN][MAXK],C[MAXN][MAXK],ans;
char random[100];
int main(){
  scanf("%d%d%d",&N,&M,&K);
  for(int i=0;i<N;++i){
    scanf(" %s",random);
    for(int j=0;j<M;++j) scanf("%d%d%d",&A[i][j],&B[i][j],&C[i][j]);
  }
  for(int a=0;a<N;++a){
    memset(dp,0,sizeof(dp));
    for(int j=0;j<M;++j)
      for(int i=0;i<N;++i)
        for(int k = K;k>=0;--k)
          for(int cnt=0;cnt<=C[a][j];++cnt)
            if(k + cnt <= K) dp[i][k+cnt] = max(dp[i][k+cnt],dp[i][k] + cnt * (B[i][j] - A[a][j]));
    for(int i=0;i<N;++i)
      for(int k=0;k<=K;++k) ans = max(ans,dp[i][k]);
  }
  printf("%d\n",ans);
  //system("pause");
}
