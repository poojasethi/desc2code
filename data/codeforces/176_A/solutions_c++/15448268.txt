#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#include<vector>
#include<cmath>
#include<map>
using namespace std;
typedef long long ll;
const int N=1e2+10;
ll a[10][N],b[10][N],c[10][N];
vector<int>v;
int main()
{
    int n,m,i,j,k,l,l1;
    ll ans=0;
    string s;
    cin>>n>>m>>k;
    for(i=0;i<n;i++)
    {
        cin>>s;
        for(j=0;j<m;j++)scanf("%I64d%I64d%I64d",&a[i][j],&b[i][j],&c[i][j]);
    }
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            if(i!=j)
    {
        ll sum=0;
        v.clear();
        for(l=0;l<m;l++)
            if(b[j][l]>a[i][l])
                for(l1=0;l1<c[i][l];l1++)v.push_back(b[j][l]-a[i][l]);
        sort(v.begin(),v.end());
        int x=v.size();
        for(l=x-1;l>=max(x-k,0);l--)sum+=v[l];
        ans=max(sum,ans);
    }
    printf("%I64d",ans);
    return 0;
}
