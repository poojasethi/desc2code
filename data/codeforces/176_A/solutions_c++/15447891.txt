#include<iostream>
#include<algorithm>
#include<cstdio> 
#include<cstring>
#include<cmath>
#include<map>
using namespace std;
struct biao {
    int x,y,z;
} a[500][500];
bool cmp(int o, int p) {
	return (o > p);
}
int i,j,k,l,n,m,num,tot,ans,t,b[20000];
char ch[2000];
int main() {
	scanf("%d %d %d", &n, &m, &num);
	for (i = 1; i <= n; i++) {
		scanf("%s", ch);
		for (j = 1; j <= m; j++) {
			scanf("%d %d %d", &a[i][j].x, &a[i][j].y, &a[i][j].z);
		}
	}
	for (i = 1; i <= n; i++) {
		for (j = 1; j <= n; j++) {
			if (i == j) continue;
			tot = 0;
			t = 0;
			for (k = 1; k <= m; k++) {
				for (l = 1; l <= a[i][k].z; l++) {
					b[++t] = a[j][k].y - a[i][k].x;
				}
			}
			sort(b + 1, b + 1 + t, cmp);
			for (k = 1; k <= num; k++) {
				if (b[k] <= 0) break;
				tot = tot + b[k];
			}
			if (tot > ans) ans = tot;
		}
	}
	printf("%d", ans);
}
