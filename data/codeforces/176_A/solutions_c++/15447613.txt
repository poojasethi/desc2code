#include<cstdio>
int a[11][110],b[11][110],c[11][110];
int dp[110];
int main(){
    int n,m,k;scanf("%d%d%d",&n,&m,&k);
    int i,j,u,v,w;
    char s[14];
    for(i=0;i<n;i++){
        scanf("%s",s);
        for(j=0;j<m;j++)scanf("%d%d%d",&a[i][j],&b[i][j],&c[i][j]);
    }
    int ans=0;
    for(i=0;i<n;i++)
    for(j=0;j<n;j++){
        if(i==j)continue;
        for(u=1;u<=k;u++)dp[u]=0;
        for(u=0;u<m;u++)
            for(v=1;v<=c[i][u];v++)
            for(w=k;w;w--){
                if(dp[w-1]+b[j][u]-a[i][u]>dp[w])dp[w]=dp[w-1]+b[j][u]-a[i][u];
                if(dp[w]>ans)ans=dp[w];
            }
    }
    printf("%d",ans);
    return 0;
}
