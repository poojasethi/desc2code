#include <cstdlib>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <map>
#include <set>
using namespace std;

int n, m, k;
int a[10][100];
int b[10][100];
int c[10][100];

int solve(int x, int y)
{
	vector<pair<int, int> > d;
	for (int i=0; i<m; ++i)
		d.push_back(make_pair(b[y][i] - a[x][i], c[x][i]));
	sort(d.rbegin(), d.rend());
	int j = 0;
	int sum = 0;
	int kk = k;
	while (j < d.size() && kk > 0)
	{
		if (d[j].first <= 0) break;
		int q = min(kk, d[j].second);
		sum += d[j++].first * q;
		kk -= q;
	}
	return sum;
}

int main()
{
	cin >> n >> m >> k;
	for (int i=0; i<n; ++i)
	{
		string s;
		cin >> s;
		for (int j=0; j<m; ++j)
			cin >> a[i][j] >> b[i][j] >> c[i][j];
	}
	int best = 0;
	for (int i=0; i<n; ++i)
		for (int j=0; j<n; ++j)
			best = max(best, solve(i, j));
	cout << best << endl;
	return 0;
}
