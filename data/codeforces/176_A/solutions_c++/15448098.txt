#include <bits/stdc++.h>

using namespace std;
struct Plant{
	int mat[102][4];
}p[20];
int n , m , cap, dp[150];
char str[500];

void updata(int & x , int v){
	x = max( x , v );
}

int main(int argc,char *argv[]){
//	freopen("in.txt","r",stdin);
	scanf("%d%d%d",&n,&m,&cap);
	for(int i = 1 ; i <= n ; ++ i){
		scanf("%s",str);
		for(int j = 1 ; j <= m ; ++ j){
			scanf("%d%d%d",&p[i].mat[j][1],&p[i].mat[j][2],&p[i].mat[j][3]);
		}
	}
	int ans = 0 ;
	for(int bg = 1 ; bg <= n ; ++ bg)
		for(int ed = 1 ; ed <= n ; ++ ed){
			memset( dp , 0 , sizeof(dp) );
			for(int i = 1 ; i <= m ; ++ i){
				for(int j = 1 ; j <= cap ; ++ j){
						int ef = min( j , p[bg].mat[i][3] );
						for(int k = 1 ; k <= ef ; ++ k) updata( dp[j - k] , dp[j] + k*(p[ed].mat[i][2]-p[bg].mat[i][1]) );
					}
			}
			for(int i = 0 ; i <= cap ; ++ i) ans = max( ans , dp[i] );
		}
	printf("%d\n",ans);
	return 0;
}
