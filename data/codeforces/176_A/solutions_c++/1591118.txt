#include <stdio.h>
#include <algorithm>

using namespace std;

char str[13];
int a[13][103],b[13][103],c[13][103],arr[10003];

int main()
{
    int n,m,K,i,j,ans,c1,k,l,sum;
    scanf("%d%d%d",&n,&m,&K);
    for (i=0;i<n;i++)
    {
        scanf("%s",str);
        for (j=0;j<m;j++) scanf("%d%d%d",&a[i][j],&b[i][j],&c[i][j]);
    }

    ans=0;
    for (i=0;i<n;i++)
    {
        for (j=0;j<n;j++)
        {
            if (i==j) continue;
            c1=0;
            for (k=0;k<m;k++)
            {
                if (a[i][k]<b[j][k])
                {
                    for (l=0;l<c[i][k];l++) arr[c1++]=b[j][k]-a[i][k];
                }
            }
            sort(arr,arr+c1);
            reverse(arr,arr+c1);
            sum=0;
            for (k=0;k<c1&&k<K;k++) sum+=arr[k];
            if (sum>ans) ans=sum;
        }
    }
    printf("%d\n",ans);
    return 0;
}
