 #include "bits/stdc++.h"
using namespace std;

struct planet {
	char name[20];
	int a[111], b[111], c[111];
} p[111];

struct dd {
	int v, id;
	bool friend operator < (dd d1, dd d2) { return d1.v > d2.v; }
} d[111];

int n, m, k, i, j, ans = 0;

int f(planet buy, planet sell, int k) {
	int i, res = 0;
	for (i = 1; i <= m; i++) {
		d[i].v = sell.b[i] - buy.a[i];
		d[i].id = i;
	}
	sort(d + 1, d + m + 1);
	for (i = 1; i <= m && k > 0 && d[i].v > 0; i++) {
		res += min(k, buy.c[d[i].id]) * d[i].v;
		k -= min(k, buy.c[d[i].id]);
	}
	return res;
}

int main() {
	scanf("%d%d%d", &n, &m, &k);
	for (i = 1; i <= n; i++) {
		scanf("%s", p[i].name);
		for (j = 1; j <= m; j++) scanf("%d%d%d", &p[i].a[j], &p[i].b[j], &p[i].c[j]);
	}
	for (i = 1; i <= n; i++) for (j = 1; j <= n; j++) ans = max(ans, f(p[i], p[j], k));
	printf("%d\n", ans);
	return 0;
}
