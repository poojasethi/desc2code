#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

long long a[1000100];

int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    vector<vector<int> > b(100100);
    int m=0;
    for(int i=0;i<n;i++)
    {
        int x;
        scanf("%d",&x);
        if(x>m) m=x;
        a[x]++;
        b[x].push_back(i);
    }
    if(a[0]>1) {cout <<-1; return 0;}
    if(a[1]>k*a[0]){cout <<-1; return 0;}
    for(int i=2;i<=m;i++)
    {
        if(a[i]>a[i-1]*(k-1)){cout <<-1; return 0;}
    }
    printf("%d\n",n-1);

    for(int i=1;i<=m;i++)
        for(int j=0;j<a[i];j++)
            printf("%d %d\n",b[i-1][j%a[i-1]]+1,b[i][j]+1);



    return 0;
}
