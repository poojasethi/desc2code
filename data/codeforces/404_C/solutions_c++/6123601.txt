#include <stdio.h>
#include <iostream>
#include <vector>
using namespace std;
int n,m,t,s,g=-1,z;
vector <int> a[100005];
vector <pair<int,int> > v;
int main( ){
	cin>>n>>m;
	for (int i=1;i<=n;i++){
		cin>>t;
		if (t>g) g=t;
		a[t].push_back(i);
	}
	if (a[0].size()!=1){
		cout<<-1<<endl;
		return 0;
	}
	for (int i=1;i<=g;i++){
		z=s=0;
		if (i!=1) z=1;
		for (int j=0;j<a[i].size();j++){
			if (z==m){
				if (i!=1) z=1; else z=0;
				s++;
			}
			if (s==a[i-1].size()){
				cout<<-1<<endl;
				return 0;
			}
			v.push_back(make_pair(a[i-1][s],a[i][j]));
			z++;
		}
	}
	cout<<v.size()<<endl;
	for (int i=1;i<=v.size();i++)
		cout<<v[i-1].first<<' '<<v[i-1].second<<endl;
	return 0;
}
