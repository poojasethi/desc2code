#include<bits/stdc++.h>
using namespace std;

vector<int> dis[100005];
vector<pair<int, int> > ans;

int main()
{
	int n, k, d, m = 0;
	cin >> n >> k;
	for (int i = 1; i <= n; i++)
	{
		cin >> d;
		dis[d].push_back(i); 
		m = max(m, d);
	}
	if (dis[0].size() != 1)
	{
		cout << "-1";
		return 0;
	}
	for (int i = 1; i <= m; i++)
	{
		int edge = (i != 1), cnt = 0;
		for (int j = 0; j < dis[i].size(); j++)
		{
			if (edge == k)
			{
				edge = (i != 1);
				cnt++; 
			}
			if (cnt == dis[i - 1].size())
			{
				cout << "-1";
				return 0;
			}
			ans.push_back(make_pair(dis[i - 1][cnt], dis[i][j]));
			edge++;
		}
	}
	cout << n - 1 << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i].first << " " << ans[i].second << endl;
	return 0;
}
