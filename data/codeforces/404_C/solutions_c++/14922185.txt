#include<bits/stdc++.h>
#define mp make_pair
using namespace std;
int n,k;
struct vertex{
int index;
int dist;
int adj=0;
}ver[100005];
vector < pair<int,int > > ans;
bool cmp(vertex a,vertex b)
{
	return a.dist<b.dist;
}

int main()
{
	int i,j,fl=0,test;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&ver[i].dist);
		ver[i].index=i;
	}
	sort(ver+1,ver+n+1,cmp);
	if(ver[1].dist!=0)
	{
		printf("-1\n");
		return 0;
	}
	j=1;
	for(i=2;i<=n;)
	{	if(j>=i)
		{
			printf("-1\n");
			return 0;
		}
		if(ver[i].dist-ver[j].dist==1 && ver[j].adj<k )
		{
			ans.push_back(mp(ver[i].index,ver[j].index));
			ver[j].adj++;
			ver[i].adj++;
			i++;
			
		}
		else
			j++;
	}
	printf("%lu\n",ans.size());
	for(i=0;i<ans.size();i++)
	{
		printf("%d %d\n",ans[i].first,ans[i].second );
	}
	return 0;
}
