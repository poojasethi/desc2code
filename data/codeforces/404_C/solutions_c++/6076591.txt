#include <stdio.h>
#include <iostream>
#include<vector>

using namespace std;

int main (void) {
	int n, k, i, j, tmp;
	scanf ("%d%d", &n, &k);
	int m = 0;
	vector<int> d[100000];
	for (i = 0; i < n; i++) {
		scanf ("%d", &tmp);
		d[tmp].push_back(i);
		m = max(tmp, m);
	}
	if (d[0].size() != 1 || d[1].size() > k) {
		printf ("-1\n");
		return 0;
	}
	bool ok = true;
	for (i = 2; i <= m && ok; i++) {
		if (d[i].size() > d[i-1].size()*(long long)(k-1))	ok = false;
	}
	if (!ok)	printf ("-1\n");
	else {
		printf ("%d\n", n-1);
		for (i = 0; i < d[1].size(); i++)	printf ("%d %d\n", d[0][0] + 1, d[1][i]+ 1);
		for (i = 2; i <= m; i++) {
			for (j = 0; j < d[i].size(); j++)	printf ("%d %d\n", d[i-1][j/(k-1)] + 1, d[i][j]+ 1);
		}
	}
	return 0;
}
