#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <iostream>
#define N 100005
using namespace std;

int dis[N], ptr[N], from[N], cnt[N];
bool cmp(const int &a, const int &b) {
    return dis[a] < dis[b];
}

int main() {
    int n, k, i, j, fail=0;
    scanf("%d%d", &n, &k);
    for(i=1; i<=n; i++) {
        scanf("%d", dis+i);
        ptr[i] = i;
    }
    sort(ptr+1, ptr+1+n, cmp);
    if(dis[ptr[1]] != 0) fail = 1;
    for(i=1, j=2; j<=n; j++) {
        while(i<j && !(cnt[i]<k && dis[ptr[i]] == dis[ptr[j]]-1)) i++;
        if(i==j){
            fail = 1;
            break;
        }
        cnt[i]++; cnt[j]++;
        from[j] = i;
    }
    if(fail == 1) puts("-1");
    else {
        printf("%d\n", n-1);
        for(i=2; i<=n; i++)
            printf("%d %d\n", ptr[from[i]], ptr[i]);
    }

    return 0;
}
