#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 7;

int n, k;
vector<int> d[N];

int main()
{
	cin >> n >> k;
	for (int i=0; i<n; i++)
	{
		int t;
		cin >> t;
		d[t].push_back(i);
	}
	
	bool flag = true;
	if (d[0].size() != 1)
		flag = false;
	if (d[1].size() > k)
		flag = false;
	for (int i=2; i<N; i++)
		if (1LL * d[i].size() > 1LL * d[i-1].size() * (k-1))
			flag = false;

	if (!flag)
	{
		cout << "-1\n";
		return 0;
	}

	cout << n-1 << "\n";
	for (int i=1; i<N; i++)
		for (int j=0; j<d[i].size(); j++)
			cout << d[i][j] + 1 << " " << d[i-1][j % d[i-1].size()] + 1 << "\n";
	return 0;
}
