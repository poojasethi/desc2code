#include <iostream>
#include <vector>
using namespace std;
int n,k,i,j,d,z,h,a[100005],b[100005],m,kk,l,s;
vector <int> g[100005];
int min(int a,int b)
{
	if(a<=b)
		return a;
	return b;
}
int main()
{
	cin>>n>>k;
	for(i=1;i<=n;i++)
	{
		cin>>d;
		g[d].push_back(i);
	}
	if(g[0].size()==0 || g[0].size()>1)
	{
		cout<<-1<<endl;
		return 0;
	}
	l=1;
	while(l!=n)
	{
		kk++;
		l+=g[kk].size();
		h=0;
		for(j=0;j<g[kk-1].size();j++)
		{	
			if(kk==1)
				s=h+k;
			else
				s=h+k-1;
			for(i=h;i<min(g[kk].size(),s);i++)
			{
				a[++m]=g[kk-1][j];
				b[m]=g[kk][i];
			}
			h=i;
		}
		if(h<g[kk].size() || g[kk].size()==0)
		{
			cout<<-1<<endl;
			return 0;
		}
	}
	cout<<m<<endl;
	for(i=1;i<=m;i++)
		cout<<a[i]<<" "<<b[i]<<endl;
	return 0;
}
