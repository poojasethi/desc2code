#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

vector<int>deg[100001];
vector<pair<int,int> >ans;
int main() {
	int k,n,d,m=0;
	cin>>n>>k;
	for(int i=0;i<n;i++){
		cin>>d;
		deg[d].push_back(i+1);
		m=max(m,d);
	}
	if(deg[0].size()!=1){
		cout<<"-1";
		return 0;
	}
	for(int i=1;i<=m;i++){
		int z=deg[i].size(),b=0,c=0;
		if(i!=1)	b=1;
		for(int j=0;j<z;j++){
			if(b==k){
				if(i!=1)	b=1;
				else	b=0;
				c++;
			}
			if(c==deg[i-1].size()){
				cout<<"-1";
				return 0;
			}
			ans.push_back(make_pair(deg[i-1][c],deg[i][j]));
			b++;
		}
	}
	cout<<ans.size()<<"\n";
	for(int i=0;i<ans.size();i++)
		cout<<ans[i].first<<" "<<ans[i].second<<"\n";
	return 0;
}
