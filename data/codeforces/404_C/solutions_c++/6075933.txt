#include<iostream>
#include<iomanip>
#include<algorithm>
#include<vector>
using namespace std;
int n,k;
pair<int,int> d[100010];
int p[100010];
vector<pair<int,int> > v;
int main()
{
	cin>>n>>k;
	for(int i=1;i<=n;i++)
	{
		cin>>d[i].first;
		d[i].second=i;
	}
	sort(d+1,d+n+1);
	if(d[1].first==d[2].first||d[1].first!=0)
	{
		cout<<-1<<endl;
		return 0;
	}
	int u=1;
	for(int i=2;i<=n;i++)
	{
		while(p[d[u].second]>=k)
		u++;
		while(d[u].first<d[i].first-1)
		u++;
		if(d[u].first>=d[i].first)
		{
			cout<<-1<<endl;
			return 0;
		}
		v.push_back(make_pair(d[u].second,d[i].second));
		p[d[u].second]++;
		p[d[i].second]++;
	}
	cout<<v.size()<<endl;
	for(int i=0;i<v.size();i++)
	cout<<v[i].first<<" "<<v[i].second<<endl;
  //	cin>>n;
}
