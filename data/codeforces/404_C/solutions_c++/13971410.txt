#include <iostream>
#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;


int main() 
{
    long long int i,j,k,n,x;
    vector<long long int>v[100001];
    vector<pair<long long int,long long int > >ans;
    cin>>n>>k;
    for(i=1;i<=n;i++)
    {
        cin>>x;
        v[x].push_back(i);
    }
    if(v[0].size()!=1)
    {
        cout<<"-1";
        return 0;
    }
    for(i=1;i<=n;i++)
    {
        if(v[i].size()>v[i-1].size()*k)
        {
            cout<<"-1";
            return 0;
        }
        for(j=0;j<v[i].size();j++)
        {
            ans.push_back(make_pair(v[i][j],v[i-1][j/k]));
        }
        if(i==1)
        --k;
    }
    cout<<ans.size()<<endl;
    for(i=0;i<ans.size();i++)
    {
        cout<<ans[i].first<<" "<<ans[i].second<<endl;
    }
    return 0;
}
