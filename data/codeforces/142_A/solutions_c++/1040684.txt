#include <iostream>
#include <cmath>
using namespace std;
typedef long long ll;
ll n;
int main() {
    cin >> n ;
    ll ans = (ll)100000*1000000;
    for (ll p=1;p<=sqrt(n)+1;++p){
        if (n%p == 0)
            n/=p;
        else
            continue ;
        for (ll i = sqrt(n) ; i >= 1 ; --i) {
            if (n%i == 0) {
                ans = min(ans,(p+1)*(i+2)*(n/i+2)-n*p) ;
                break ;
            }
        }
        n *= p;
    }
    cout<<ans<<" "<<(n+1)*3*3-n<<endl;
    return 0 ;
}
