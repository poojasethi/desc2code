#include<cstdio>
#include<cstring>
#include<cstdlib>
using namespace std;
long long n,x,y,z,min,max;
int main()
{
	scanf("%I64d",&n);
	min=100000000000000ll;
	for (long long i=1;i*i*i<=n;++i)
		if (n%i==0)
			for (long long j=i;i*j*j<=n;++j)
				if ((n/i)%j==0)
				{
					x=i,y=j,z=n/i/j;
					if ((x+1)*(y+2)*(z+2)-n<min)
						min=(x+1)*(y+2)*(z+2)-n;
					if ((z+1)*(y+2)*(x+2)-n>max)
						max=(z+1)*(y+2)*(x+2)-n;
				}
	printf("%I64d ",min);
	printf("%I64d\n",max);
	//system("pause");
}
