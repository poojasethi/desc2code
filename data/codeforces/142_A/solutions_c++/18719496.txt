#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    ll n,i,j,k,t1,t2,t3,mn=9e18,mx=-9e18;
    cin>>n;
    for(i=1;i*i*i<=n;i++)
    	if(n%i==0)
    		for(j=1;j*j<=n/i;j++)
    			if((n/i)%j==0)
    			{
                  k=n/i/j;
                  t1=(i+1)*(j+2)*(k+2);
                  t2=(i+2)*(j+2)*(k+1);
                  t3=(i+2)*(j+1)*(k+2);
                  mn=min(mn,min(t1,min(t2,t3)));
                  mx=max(mx,max(t1,max(t2,t3)));
    			}
    cout<<mn-n<<" "<<mx-n;   
	return 0;
}
