#include<iostream>
using namespace std;
int main()
{long long n,ans1,ans2; cin>>n;
ans1=min((n+1)*9,(n+2)*6);
ans2=max((n+1)*9,(n+2)*6);
for (long long i=1;i*i<=n;i++)
 if (n%i==0)
 {long long n1=n/i;
  for (long long j=i+(i==1);j*j<=n1;j++)
   if (n1%j==0)
    {long long k=n1/j;
     long long x1,x2,x3;
      x1=(i+2)*(j+2)*(k+1);
      x2=(i+2)*(j+1)*(k+2);
      x3=(i+1)*(j+2)*(k+2);
     ans1=min(min(ans1,x1),min(x2,x3));
     ans2=max(max(ans2,x1),max(x2,x3));
    }
 }
cout<<ans1-n<<" "<<ans2-n;
return 0;
}
