#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
    ll n,A,B,C;cin>>n;

    ll maxi=(n+1)*3*3-n;
    ll mini=1e10;

    for( A=1;A*A*A<=n;A++){

          for(B=1;B*B<=n/A;B++){

              C=n/(A*B);

              if(A*B*C==n){

                    mini=min(mini,(A+1)*(B+2)*(C+2));
                    mini=min(mini,(A+2)*(B+1)*(C+2));
                    mini=min(mini,(A+2)*(B+2)*(C+1));


              }

          }

    }
     mini-=n;

    cout<<mini<<" "<<maxi<<"\n";

}
