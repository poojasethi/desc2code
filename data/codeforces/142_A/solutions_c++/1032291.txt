#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
#define inf 1000000000000000000LL
long long ma,mi,a,b,c,w[1111111],wn,n,i,j;
int main(){
	mi=inf;ma=-inf;
	cin>>n;
	for(i=1;i*i<=n;i++)if(n%i==0){w[++wn]=i;w[++wn]=n/i;};
	sort(w+1,w+wn+1);
	for(i=1;i<=wn;i++)for(j=1;j<=wn;j++){
		a=w[i];b=w[j];c=n/(a*b);
		if((n/a)%b!=0)continue;
		if(n%(a*b)!=0)continue;
		mi=min(mi,(a+1)*(b+2)*(c+2)-a*b*c);
		ma=max(ma,(a+1)*(b+2)*(c+2)-a*b*c);
	}
	cout<<mi<<" "<<ma<<endl;
	return 0;
}
