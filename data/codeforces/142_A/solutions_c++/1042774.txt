#include <cstdio>
#define M(a, b) (a > b ? a : b)
#define N(a, b) (a > b ? b : a)
#define L long long

L n, x = 1e15, y = 0;

int main(){
	scanf("%I64d", &n);
	for(L i = 1; i * i * i <= n; i++) if(n % i == 0)
		for(L j = 1; j * j <= n / i; j++) if((n / i) % j == 0){
			L k = n / i / j, z = (i + 1) * (j + 2) * (k + 2);
			x = N(x, z);
			y = M(y, z);
			z = (i + 2) * (j + 1) * (k + 2);
			x = N(x, z);
			y = M(y, z);
			z = (i + 2) * (j + 2) * (k + 1);
			x = N(x, z);
			y = M(y, z);
	}
	printf("%I64d %I64d\n", x - n, y - n);
}
