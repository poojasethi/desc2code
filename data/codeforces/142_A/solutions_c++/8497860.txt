#include<cstdio>
#include<algorithm>
using namespace std;
int n,i,k,tmp,big,small;
long long S;
int main()
{
  scanf("%d",&n);S=(long long)1e18;
  for (k=1;k*k<=n;k++)
    if (n%k==0)
    {
      tmp=n/k;
      for (i=1;i*i<=tmp;i++)
        if (tmp%i==0) S=min(S,(i+2)*1ll*(tmp/i+2)*(k+1));
    }
  printf("%I64d ",S-n);
  for (i=1;i*i<=n;i++)
    if (n%i==0) {k=n/i;break;}
  tmp=n/k;
  for (i=1;i*i<=tmp;i++)
    if (tmp%i==0) big=i;
  printf("%I64d ",(k+1)*1ll*(big+2)*1ll*(tmp/big+2)-n);
}
