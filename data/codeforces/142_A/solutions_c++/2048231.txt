#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

long long n;

int main(){
    scanf("%I64d",&n);
    long long Max,Min;
    Min = 3 * 3 *(n + 1) - n;
    Max = 3 * 3 *(n + 1) - n;
    for(long long i = 1;i*i*i <=n;i++){
        if(n % i == 0){
            for(long long j = i;i * j *j <=n;j++){
                if(n / i % j == 0)
                    Min = min(Min,(i + 1) * (j + 2) * (n / i / j + 2) - n);
            }
        }
    }
    printf("%I64d %I64d\n",Min,Max);
    return 0;
}
