#include<stdio.h>
#include<math.h>
int main()
{
	long long n;
	while(scanf("%I64d",&n)!=EOF)
	{
		int i;
		long long s=10000000000000000;
		for(i=sqrt(double(n));i>=1;i--)
		{
			if(n%i==0)	
			{
				int j;
				for(j=sqrt(double(n/i));j>=1;j--)
				{
					if((n/i)%j==0)
					{
						int k;
						if(j<n/i/j)
							k=j;
						else
							k=n/i/j;
						if(s>(n/i/k+2)*(i+2)*(k+1)-n)
							s=(n/i/k+2)*(i+2)*(k+1)-n;
					}
				}
			}
		}
		printf("%I64d %I64d\n",s,3*3*(n+1)-n);
	}
	return 0;
}
