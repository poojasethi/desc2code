#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;

int main()
{
	long long n,min,max,t,i,j,k;
	cin>>n;
	min=1LL<<62,max=0;
	for(i=1;i*i*i<=n;i++)
		if(n%i==0)
		{
			long long nn=n/i;
			for(j=1;j*j<=nn;j++)
				if(nn%j==0)
				{
					k=nn/j;
					t=(i+1)*(j+2)*(k+2)-i*j*k;
					if(t<min) min=t;
					if(t>max) max=t;
					t=(i+2)*(j+1)*(k+2)-i*j*k;
					if(t<min) min=t;
					if(t>max) max=t;
					t=(i+2)*(j+2)*(k+1)-i*j*k;
					if(t<min) min=t;
					if(t>max) max=t;
				}
		}
	cout<<min<<" "<<max<<endl;
}
