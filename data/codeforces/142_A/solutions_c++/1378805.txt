#include <iostream>
#include <cstdio>
#include <cmath>
#define LL long long
using namespace std;

LL minn, maxn, a, b;
LL n;

int main(){
	cin >> n;
	minn = maxn = 3 * 3 * (n + 1) - n;
	for (LL i = 1; i * i * i <= n; i++)
		if (n % i == 0)
			for (LL j = i; i * j * j <= n; j++)
				if (n / i % j == 0)
					minn = min(minn, (i + 1) * (j + 2) * (n / i / j + 2) - n);
	cout << minn << " " << maxn << endl;
	return 0;
}
