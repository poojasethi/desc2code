#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
    long long n;
    cin >> n;
    long long low=1LL<<62;
    for(long long  i=1;i<=100010;i++)
    {
        if(n%i)continue;
        long long m=n/i;
        long long j=(long long)sqrt(m)+1;
        while(m%j)j--;
        //cout << i <<" " << j <<" " << m/j <<" " << (i+1)*(j+2)*(m/j+2) << endl;
        low=min(low,(i+1)*(j+2)*(m/j+2)-n);
    }
    cout << low << " " << n*8+9 << endl;
}
