n = input()
mi=ma=9*(n+1)

def solve2(a,bc):
    global mi,ma
    for b in xrange(1,int(bc**.5+2)):
        if bc%b:continue
        p = a*(bc/b+2)*(b+2)
        mi = min(p,mi)
        ma = max(p,ma)
        
for a in xrange(1,int(n**.5+2)):
    if n%a: continue
    solve2(a+1,n/a)
    solve2(n/a+1,a)

print mi-n,ma-n
