n = int(raw_input())
def calc(a, b, c):
    return (a + 1) * (b + 2) * (c + 2) - a * b * c
def make(k):
    a = []
    for i in xrange(1, int(k ** 0.5 + 5)):
        if k % i == 0:
            a.append(i)
            a.append(k / i)
    return a
a1, a2 = 10 ** 30, -1
for i in make(n):
    for j in make(n / i):
        tmp = calc(i, j, n / (i * j))
        a1 = min(a1, tmp)
        a2 = max(a2, tmp)
print a1, a2
