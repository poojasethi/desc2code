import math
n=int(raw_input())

d=[]

for i in xrange(1,int(math.sqrt(n))+1):
    if n%i==0:d+=[i]
    
e=[]
    
for i in d:
    j=n/i
    if i!=j: e+=[j]
    
d=sorted(d+e)

max=0
min=10**20

for i in d:
    n1=n/i
    for j in d:
        if j>math.sqrt(n1):break
        if n1%j>0:continue
        k=n1/j
        res=(j+2)*(k+2)+2*i*(j+2)+2*i*k
        if res>max:max=res
        if res<min:min=res
        
print min,max
