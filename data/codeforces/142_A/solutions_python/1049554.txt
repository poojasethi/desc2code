while(i*i<=n):
    if(n%i==0):
        a.append(i)
        a.append(n/i)
    i+=1
resMax=0
resMin=1000000000000000000

for x in a:
    rem=n//x
    i=1
    while(i*i<=rem):
        if(rem%i==0):
            A=x+1
            B=i+2
            C=rem/i+2
            resMax=max(resMax,A*B*C-n)
            resMin=min(resMin,A*B*C-n)
            
                
        i+=1
