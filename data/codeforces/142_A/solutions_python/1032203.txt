#!/usr/bin/python

import sys

def S(a, b, c):
	return (a + 1) * (b + 2) * (c + 2) - a * b * c

def solve(n):
	mx = max(S(1, 1, n), S(n, 1, 1))
	mn = mx
	i = 1
	while i**3 <= n:
		if n % i == 0:
			j = i
			n1 = n/i
			while j**2 <= n1:
				if n1 % j == 0:
					k = n1 / j
					mn = min(mn, S(i, j, k), S(j, i, k), S(k, i, j))
				j += 1
		i += 1
	print mn, mx

for s in sys.stdin:
	n = long(s)
	solve(n)
