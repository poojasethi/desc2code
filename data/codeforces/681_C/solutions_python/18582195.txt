# coding: utf-8

from heapq import heappush
from heapq import heappop

h = []
n = int(raw_input())
output = []
for i in xrange(n):
    line = raw_input().strip()
    if line[0] == 'i':
        words = line.split()
        value = int(words[1])
        heappush(h, value)     
        output.append(line)

    elif line[0] == 'g':
        words = line.split()
        value = int(words[1])
        while 1:
            if not len(h):
                heappush(h, value)
                output.append("%s %d" % ("insert", value))
                output.append(line)
                break
            minv = h[0]
            if minv == value:
                output.append(line)
                break
            elif minv < value:
                heappop(h)
                output.append("removeMin")
            else:
                #heappop(h)
                #output.append("removeMin")
                heappush(h, value)
                output.append("%s %d" % ("insert", value))
                output.append(line)
                break

    elif line[0] == 'r':
        if not len(h):
            output.append("insert 1")
            #value = 123456789
        else:
            heappop(h)

        output.append("removeMin")


print len(output)
