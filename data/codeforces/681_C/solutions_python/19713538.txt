#!/usr/bin/env python
# -*- coding: utf-8 -*-

import heapq

out = []

n = int(raw_input())
h = []
for i in xrange(0, n):
    s = raw_input()
    if s[0] == 'i':
        l = s.split()
        number = int(l[1])
        heapq.heappush(h, number)
    elif s[0] == 'r':
        if len(h) == 0:
            out.append('insert 1')
        else:
            heapq.heappop(h)
    else:
        l = s.split()
        number = int(l[1])

        while len(h) > 0 and h[0] < number:
            heapq.heappop(h)
            out.append('removeMin')

        if len(h) == 0 or h[0] > number:
            heapq.heappush(h, number)
            out.append('insert {}'.format(number))

    out.append(s)

print len(out)
for l in out:
    print l
