#!/usr/bin/env python

import heapq

h = []



n = int(raw_input())
command_list = list()


for i in range(n):

    line = raw_input()
    ln = line.split()
    cmd = ln[0]


    if cmd == 'insert':
        x = int(ln[1])
        heapq.heappush(h,x)
        command_list.append(line)

    if cmd == 'removeMin':

        if h:
            heapq.heappop(h)
            command_list.append(line)
        else:
            command_list.append("insert 0")
            command_list.append(line)
    if cmd == 'getMin':

        x = int(ln[1])

        if not h:
            heapq.heappush(h,x)
            command_list.append("insert "+str(x))
            command_list.append(line)
            continue
        if x == h[0]:
            command_list.append(line)
        elif x < h[0]:
            command_list.append("insert "+str(x))
            command_list.append(line)
            heapq.heappush(h,x)
        else:
            while(h and x > h[0]):
                heapq.heappop(h)
                command_list.append("removeMin")
            if not h:
                heapq.heappush(h,x)
                command_list.append("insert "+str(x))
                command_list.append(line) # getMin x
            else:
                if h[0] == x:
                    command_list.append(line) # getMin x
                elif h[0] > x:
                    heapq.heappush(h,x)
                    command_list.append("insert "+str(x))
                    command_list.append(line) # getMin x
                

print len(command_list)
for c in command_list:
