import heapq
operations = [raw_input() for _ in xrange(int(raw_input()))]

heap = []
new_operations = []
for operation in operations:
    if operation == 'removeMin':
        if not heap:
            new_operations.append('insert -1000000000')
        else:
            heapq.heappop(heap)
    elif operation.startswith('insert'):
        _, value = operation.split()
        value = int(value)
        heapq.heappush(heap, value)
    elif operation.startswith('getMin'):
        _, value = operation.split()
        value = int(value)

        while True:
            if not heap or heap[0] > value:
                new_operations.append('insert %s' % value)
                heapq.heappush(heap, value)
                break
            elif heap[0] == value:
                break
            else:
                new_operations.append('removeMin')
                heapq.heappop(heap)

    else:
        print 'UNKNOWN', operation
        exit(0)

    new_operations.append(operation)

print len(new_operations)
print '\n'.join(new_operations) + '\n'
