#!/usr/bin/env python

from __future__ import print_function
import sys

DEBUG = '-d' in sys.argv


def debug(*args, **kwargs):
    if DEBUG:
        print(*args, file=sys.stderr, **kwargs)

    return None

n = int(raw_input())

heap = []
import heapq

commands = []

for _ in xrange(n):
    line = raw_input()
    tokens = line.split()

    if tokens[0] == "insert":
        a = int(tokens[1])
        heapq.heappush(heap, a)
        commands.append(line)
    elif tokens[0] == "removeMin":
        if heap == []:
            commands.append("insert 1")
        else:
            heapq.heappop(heap)
        commands.append("removeMin")
    else: #getMin
        a = int(tokens[1])

        while heap != [] and heap[0] < a:
            b = heapq.heappop(heap)
            commands.append("removeMin")

        if heap == [] or heap[0] > a:
            commands.append("insert %d" % a)
            heapq.heappush(heap, a)

        commands.append(line)

        #if heap != [] and heap[0] == a:
        #    heapq.heappop(heap)

print(str(len(commands)))
print("\n".join(commands))
