import sys
import heapq
from cStringIO import StringIO

if __name__ == "__main__":
    n = int(sys.stdin.readline().strip())
    d = []
    commands = 0
    res = StringIO()

    for _ in range(n):
        command_raw = sys.stdin.readline()
        command = command_raw.strip().split()
        if command[0] == "getMin":
            value = int(command[1])
            if not d:
                heapq.heappush(d, value)
                res.write("insert %d\n" % value)
                commands += 1
            while d and d[0] < value:
                res.write("removeMin\n")
                heapq.heappop(d)
                commands += 1
            if not d or d[0] > value:
                heapq.heappush(d, value)
                res.write("insert %d\n" % value)
                commands += 1
            res.write(command_raw)
            commands += 1
        elif command[0] == "insert":
            value = int(command[1])
            heapq.heappush(d, value)
            res.write(command_raw)
            commands += 1
        else:
            if not d:
                heapq.heappush(d, 10)
                res.write("insert 10\n")
                commands += 1
            heapq.heappop(d)
            res.write(command_raw)
            commands += 1
    print commands
    print res.getvalue()
