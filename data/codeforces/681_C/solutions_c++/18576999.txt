#include <iostream>
#include <cstdio>
#include <set>
#include <vector>
using namespace std;

#define pb push_back
vector<int> ans;
vector<int> num;

int main(){
	int n;
	scanf("%d",&n);
	multiset<int> st;
	for(int i=0;i<n;i++){
		string str;
		int x;
		cin>>str;
		if(str=="insert"){
			scanf("%d",&x);
			st.insert(x);
			ans.pb(0);
			num.pb(x);
		}else if(str=="getMin"){
			scanf("%d",&x);
			while(!st.empty()&&(*st.begin())<x){
				ans.pb(2);
				st.erase(st.begin());
			}
			if(st.empty()||(*st.begin())>x){
				ans.pb(0);
				num.pb(x);
				st.insert(x);
			}
			ans.pb(1);
			num.pb(x);
		}else{
			if(st.empty()){
				ans.pb(0);
				num.pb(1);
				st.insert(1);
			}
			ans.pb(2);
			st.erase(st.begin());
		}
	}
	int in=0;
	printf("%d\n",(int)ans.size());
	for(int i=0;i<ans.size();i++){
		if(ans[i]==0)
			printf("insert %d\n",num[in++]);
		else if(ans[i]==1)
			printf("getMin %d\n",num[in++]);
		else
			printf("removeMin\n");
	}
}
