#include<algorithm>
#include<iostream>
#include<cstdlib>
#include<cstring>
#include<cstdio>
#include<vector>
#include<cmath>
#include<ctime>
#include<queue>
#include<set>
#include<map>
#define LL long long
using namespace std;
priority_queue<int,vector<int>,greater<int> >q;
const int N=1010101;
int n,m,a[N],b[N];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		char s[10];
		scanf("%s",s);
		if(s[0]=='i')
		{
			a[++m]=1;
			scanf("%d",&b[m]);
			q.push(b[m]);
		}
		else if(s[0]=='r')
		{
			if(q.empty())
			{
				a[++m]=1;
				q.push(0);
			}
			a[++m]=2;
			q.pop();
		}
		else 
		{
			int w; scanf("%d",&w);
			while(!q.empty()&&q.top()<w)
			{
				a[++m]=2;
				q.pop();
			}
			if(q.empty()||q.top()!=w)
			{
				q.push(w);
				a[++m]=1;
				b[m]=w;
			}
			a[++m]=3;
			b[m]=w;
		}
	}
	printf("%d\n",m);
	for(int i=1;i<=m;i++)
	    if(a[i]==1) printf("insert %d\n",b[i]);
	    else if(a[i]==2) puts("removeMin");
	    else if(a[i]==3) printf("getMin %d\n",b[i]);
	return 0;
}
