#include<bits/stdc++.h>

using namespace std;

typedef long long int ll;

priority_queue<ll> pq;
vector<pair<int,ll> > r;

string cmd[] = {"insert ","removeMin","getMin "};

int main()
{
	ios::sync_with_stdio(false);
	ll n;
	string s;
	cin >> n;
	while(n--)
	{
		ll a;
		cin >> s;
		if(s=="insert")
		{
			cin >> a;
			pq.push(-a);
			r.push_back(make_pair(0,a));
		}
		else if(s=="removeMin")
		{
			if(pq.empty()) r.push_back(make_pair(0,1));
			else pq.pop();
			r.push_back(make_pair(1,0));
		}
		else if(s=="getMin")
		{
			cin >> a;
			while((not pq.empty()) and -pq.top()<a) { r.push_back(make_pair(1,0)); pq.pop(); }
			if(pq.empty() or -pq.top()!=a) { r.push_back(make_pair(0,a)); pq.push(-a); }
			r.push_back(make_pair(2,a));
		}
	}
	cout << r.size() << endl;
	for(int i=0; i<r.size(); i++) 
		if(r[i].first==1) cout << "removeMin" << endl;
		else cout << cmd[r[i].first] << r[i].second << endl;
	return 0;
}
