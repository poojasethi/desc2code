#include<stdio.h>
#include<string.h>
#include<vector>
#include<string>
#include<set>
#include<algorithm>
using namespace std;

const int INF = 1e9+5;
const int MIN = 0 - 1e9;

string A = "insert", B = "getMin", C = "removeMin";
vector < pair< string, int > > x;
multiset <int> a;

int main ()
{
	int T, num;
	char s[100];
	scanf("%d", &T);
	while(T--){
		scanf("%s", s);
		if(s[0] == 'r'){
			if(a.empty())
				x.push_back(make_pair(A, MIN));
			else
				a.erase(a.begin());
			num = INF;
		}else if(s[0] == 'i'){
			scanf("%d", &num);
			a.insert(num);
		}else{
			scanf("%d", &num);
			while(!a.empty() && *a.begin() < num){
				x.push_back(make_pair(C, INF));
				a.erase(a.begin());
			}
			if(a.empty() || *a.begin() > num){
				x.push_back(make_pair(A, num));
				a.insert(num);
			}
		}
		x.push_back(make_pair(s, num));
	}
	
	printf("%d\n", x.size());
	for(int i=0; i<x.size(); i++){
		printf("%s", x[i].first.c_str());
		if(x[i].second != INF)
			printf(" %d", x[i].second);
		printf("\n");
	}
	
	return 0;
}
