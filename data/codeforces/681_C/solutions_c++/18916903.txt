#include<bits/stdc++.h>
using namespace std;
int main()
{
ios::sync_with_stdio(false);
int n,ctr=0,i;
vector< pair<string,int> > v1;
multiset<int> s1;
cin>>n;
while(n--)
{
string s;
cin>>s;

long long int t;
if(s!="removeMin")
cin>>t;
if(s=="insert")
	{
	s1.insert(t);
	}
else if(s=="removeMin")
	{
	if(s1.size()==0)
		{
			ctr++;
			s1.insert(0);
			v1.push_back(make_pair("insert",0));
		}
	s1.erase(s1.begin());
	}
else
	{
		while(s1.size()>0 && *s1.begin()<t)
		{
			s1.erase(s1.begin());
			v1.push_back(make_pair("removeMin",0));
		}
		if(*s1.begin()>t || s1.size()==0)
		{
			v1.push_back(make_pair("insert",t));
			s1.insert(t);
		}

	}
v1.push_back(make_pair(s,t));
}
cout<<v1.size()<<endl;
for(i=0;i<v1.size();i++)
	{
	if(v1[i].first=="removeMin")
		cout<<v1[i].first<<endl;
	else
		cout<<v1[i].first<<" "<<v1[i].second<<endl;
	}
return 0;
}
