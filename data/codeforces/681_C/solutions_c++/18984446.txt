#include<cstdio>
#include<queue>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;

priority_queue<int, vector<int>, greater<int> >Q;

char s[100];
int opt[1000010], val[1000010], n;
int main(){
	scanf("%d",&n);
	int cnt = 0, x;
	for(int i = 1; i <= n; i ++){
		scanf("%s%d",s, &x);
		if (s[0] == 'i'){
			opt[++ cnt] = 1;
			val[cnt] = x;
			Q.push(x);
		}
		else if (s[0] == 'r'){
			if (Q.empty()){
				opt[++ cnt] = 1;
				val[cnt] = 1;
				Q.push(1);
			}
			opt[++ cnt] = 2;
			val[cnt] = -1;
			Q.pop();
		}
		else if (s[0] == 'g'){
			while(!Q.empty() && Q.top() < x){
				opt[++ cnt] = 2;
				val[cnt] = -1;
				Q.pop();
			}
			if (Q.empty() || Q.top() > x){
				opt[++ cnt] = 1;
				val[cnt] = x;
				Q.push(x);
			}
			opt[++ cnt] = 3;
			val[cnt] = x;
		}
	}
	printf("%d\n",cnt);
	for(int i = 1; i <= cnt; i ++){
		if (opt[i] == 1)
			printf("insert %d\n",val[i]);
		else if (opt[i] == 2)
			printf("removeMin\n");
		else if (opt[i] == 3)
			printf("getMin %d\n",val[i]);
	}
	return 0;
}
