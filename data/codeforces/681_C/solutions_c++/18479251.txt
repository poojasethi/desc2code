#include <bits/stdc++.h>
using namespace std;
priority_queue <int> q;
vector <pair<int,int> > a;
int main()
{
	int i,n,x;
	cin>>n;
	for(i=0;i<n;i++)
    {
		string s;
		cin>>s;
		if(s=="insert") {cin>>x;q.push(-x);a.push_back(make_pair(0,x));}
		if (s == "getMin")
		{
			cin>>x;
			while(!q.empty() && q.top()> -x) {a.push_back(make_pair(1,0));q.pop();}
			if (q.empty() || q.top() != -x)
			{
				a.push_back(make_pair(0, x));
				q.push(-x);
			}
			a.push_back(make_pair(2, x));
		}
		if (s == "removeMin")
		{
			if (q.empty()) {a.push_back(make_pair(0,1));q.push(-1);}
			a.push_back(make_pair(1,-q.top()));
			q.pop();
		}
	}
	cout<<a.size()<<endl;
	for (i = 0; i < a.size(); i++)
	{
		if (a[i].first == 0)
		{
			cout << "insert " << a[i].second << '\n';
		}
		if (a[i].first == 2) {
			cout << "getMin " << a[i].second << '\n';
		}
		if (a[i].first == 1) {
			cout << "removeMin" << '\n';
		}
	}
	return 0;
}
