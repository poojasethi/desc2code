#include<bits/stdc++.h>
using namespace std;

priority_queue<int> q;
vector< pair<string,int> > ans;

int main()
{
    int n,tmp; cin >> n;
    for (int i=0;i<n;i++)
	{
		string s; cin >> s;
		if(s == "insert")
		{
			cin >> tmp;
			ans.push_back(make_pair(s,tmp));
			q.push(-tmp);
		}
		else if(s == "removeMin")
		{
			if(q.empty())
				ans.push_back(make_pair("insert",0));
			else
				q.pop();
			ans.push_back(make_pair(s,0));
		}
		else if(s == "getMin")
		{
			cin >> tmp;
			while(!q.empty() && -q.top() < tmp)
			{
				q.pop();
				ans.push_back(make_pair("removeMin",0));
			}
			
			if(q.empty() || -q.top() > tmp){
				q.push(-tmp);
				ans.push_back(make_pair("insert",tmp));
			}
			ans.push_back(make_pair(s,tmp));
		}
	}
	int sz = ans.size();
	printf("%d\n",sz);
	for (int i=0;i<ans.size();i++)
	{
		if(ans[i].first == "removeMin")
			cout << ans[i].first;
		else{
			cout << ans[i].first; printf(" %d",ans[i].second);}
		printf("\n");
	}
	return 0;
}
