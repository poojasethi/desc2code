#include  <cstdio>
#include  <cstdlib>
#include  <cstring>
#include  <string>
#include  <cmath>
#include  <algorithm>
#include  <set>
#include  <vector>
#define  int64  long long
#define  mk  make_pair
using  namespace  std;

multiset<int>  M;
vector<pair<int,int> >  A;
int  n,tt;
char*  ord[3];
char  S[20];

int  main(){
	scanf("%d\n",&n);
	ord[0] = "insert";
	ord[1] = "getMin";
	ord[2] = "removeMin";
	while  (n--){
		scanf("%s",S);
		if  (S[0] == 'i'){
			scanf("%d\n",&tt);
			M.insert(tt);
			A.push_back(mk(0,tt));
		}
		else  if  (S[0] == 'g'){
			scanf("%d\n",&tt);
			while  (M.size() > 0 && *M.begin() < tt){
				A.push_back(mk(2,0));
				M.erase(M.begin());
			}
			if  (M.empty() || *M.begin() > tt){
				A.push_back(mk(0,tt));
				M.insert(tt);
			}
			A.push_back(mk(1,tt));
		}
		else{
			if  (M.empty()){
				A.push_back(mk(0,0));
				M.insert(0);
			}
			A.push_back(mk(2,0)),M.erase(M.begin());
		}
	}
	printf("%d\n",A.size());
	for  (int  i = 0;i < A.size(); i++)
		if  (A[i].first < 2)  printf("%s %d\n",ord[A[i].first],A[i].second);
		else  printf("%s\n",ord[A[i].first]);
    return 0;
}
