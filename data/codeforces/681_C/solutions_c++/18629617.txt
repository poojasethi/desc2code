#include <bits/stdc++.h>
using namespace std;
#define SSTR( x ) static_cast< std::ostringstream & >( \
        ( std::ostringstream() << std::dec << x ) ).str()
vector<string> v;
int main() {
	int n,i,j,x;
	char s[15];
	priority_queue<int, vector<int>, greater<int> > q;
	cin>>n;
	

	
	for(i=0;i<n;i++)
	{
		scanf("%s",s);
		
		if(strcmp(s,"insert")==0)
		{
			cin>>j;
			q.push(j);
			v.push_back("insert "+SSTR(j));
		}
		
		else if(strcmp(s,"getMin")==0)
		{
			cin>>j;
			
			while(!q.empty() && q.top()<j)
			{
				v.push_back("removeMin");
				q.pop();
			}
			if(!q.empty() && q.top()==j);
			
			else
			{
				v.push_back("insert "+SSTR(j));
				q.push(j);
			}
			v.push_back("getMin "+SSTR(j));
		}
		
		else
		{
			if(!q.empty())
			{
				v.push_back("removeMin");
				q.pop();
			}
			else
			{
				v.push_back("insert "+SSTR(1));
				v.push_back("removeMin");
			}
		}
	}
	
	cout<<v.size()<<endl;
	for(i=0;i<v.size();i++)
	printf("%s\n",v[i].c_str());
	return 0;
}
