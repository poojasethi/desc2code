#include <bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(0);
	int n;
	long long num,count;
	string op;
	stringstream ss;
	priority_queue<long long, vector<long long>, greater<long long> >s;
	cin>>n;
	count=0;
	for(int i=0;i<n;i++){
		cin>>op;
		if(op == "insert" ) 
		{	
			count++;
			cin>>num;
			s.push(num);
			ss<<"insert "<<num<<"\n";
		}
		else if(op == "removeMin")
		{
			if(s.empty()){
				s.push(1);
				count++;
				ss<<"insert "<<1<<"\n";
			}
			s.pop();
			count++;
			ss<<"removeMin"<<"\n";
		}

		else
		{
			cin>>num;
			if(s.empty()){
				s.push(num);
				count++;
				ss<<"insert "<<num<<"\n";
			}
			while(s.top()<num){
				s.pop();
				count++;
				ss<<"removeMin"<<"\n";
				if(s.empty())break;
			}
			if(s.top() != num){
				s.push(num);
				count++;
				ss<<"insert "<<num<<"\n";
			}
			count++;
			ss<<"getMin "<<num<<"\n";
		}


	}

	cout<<count<<endl;
	cout<<ss.str();
	// for(int i=0;i<count;i++){
	// 	if(fin_op[i] == "removeMin"){
	// 		cout<<fin_op[i]<<endl;
	// 	}
	// 	else{
	// 		cout<<fin_op[i]<<" "<<fin_num[i]<<endl;
	// 	}
	// }
	return 0;
}
