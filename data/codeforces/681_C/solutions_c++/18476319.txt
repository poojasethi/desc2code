#include <bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin >> n;
	priority_queue<int>q;
	queue< pair<string, int> > p;
	while(n--){
		string s;
		int var;
		cin >> s;
		if(s == "insert"){
			cin >> var;
			q.push(-var);
			p.push( make_pair("insert", var));
		}
		else if( s == "removeMin"){
			if( q.empty()){
				p.push( make_pair( "insert", 0));
				p.push( make_pair( "removeMin", 0));
			}
			else{
				p.push( make_pair( "removeMin", 0));
				q.pop();
			}
		}
		else{
			cin >> var;
			while( !q.empty() && var > -q.top()){
				p.push( make_pair( "removeMin", -q.top()));
				q.pop();
			}
			if( q.empty() || q.top() != -var){
				p.push( make_pair( "insert", var));
				q.push(-var);
				p.push( make_pair( "getMin", var));
			}else
				p.push( make_pair( "getMin", var));
		}
	}
	cout << p.size() << '\n';
	while( !p.empty()){
		if( p.front().first == "removeMin")
			cout << p.front().first << endl;
		else
			cout << p.front().first << ' ' << p.front().second << '\n';
		p.pop();
	}
	return 0;
}
