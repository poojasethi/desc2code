#include <bits/stdc++.h>
using namespace std;
struct opera{
	int op,data;
	opera() {}
	opera(const int& a,const int& b) {op = a,data = b;}
};
const string mp[3] = {"insert ","getMin ","removeMin"};
queue<opera>ans;
priority_queue<int, vector<int>, greater<int> >pq;

int main() {
	int n,a;
	char s[20] = {0};
	//freopen("test.in","r",stdin);
	scanf("%d", &n);
	while (n--) {
		scanf("%s", s);
		if (*s != 'r') scanf("%d", &a);
		if (*s == 'i') {
			pq.push(a); ans.push(opera(0,a));
		}else if(*s == 'g') {
			while (!pq.empty() && pq.top() < a) {
				pq.pop(); ans.push(opera(2,0));
			}
			if (pq.empty() || pq.top() > a) {
				pq.push(a); ans.push(opera(0,a));
			}
			ans.push(opera(1,a));
		}else {
			if (pq.empty()) {
				pq.push(1); ans.push(opera(0,1));
			}
			pq.pop(); ans.push(opera(2,0));
		}
	}
	cout << ans.size() << endl;
	opera tmp;
	while (!ans.empty()) {
		tmp = ans.front(); ans.pop();
		if (tmp.op == 2) puts("removeMin");
		else {
			cout << mp[tmp.op] << tmp.data << endl;
		}
	}
	return 0;
}
