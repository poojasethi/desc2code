#include<bits/stdc++.h>
using namespace std;

#define pa pair<char,int>
#define pb push_back
#define fr first
#define sc second

priority_queue < int ,vector <int> , greater <int> > pq;
vector< pa > ans;
int n,a;
char com[1000];

int main(){
	
	scanf("%d",&n);
	for(int i=1 ; i<=n ; i++){
		scanf(" %s",com);
		if(com[0]=='i'){
			scanf("%d",&a);
			pq.push(a);
			ans.pb(pa('i',a));
		}
		if(com[0]=='r'){
			if(pq.size()==0){
				pq.push(1);
				ans.pb(pa('i',1));
				ans.pb(pa('r',1));
				pq.pop();
			}
			else{
				ans.pb(pa('r',pq.top()));
				pq.pop();
			}				
		}
		if(com[0]=='g'){
			scanf("%d",&a);
			while(pq.size() && pq.top() < a ){
				ans.pb(pa('r',pq.top()));
				pq.pop();
			}
			if(pq.size()==0 || pq.top()!=a){
				pq.push(a);
				ans.pb(pa('i',a));
			}
			ans.pb(pa('g',a));
		}
	}
	printf("%d\n",ans.size());
	for(int i=0 ; i<ans.size() ; i++){
		if(ans[i].fr=='i'){
			printf("insert %d\n",ans[i].sc);
		}
		if(ans[i].fr=='r'){
			printf("removeMin\n");
		}
		if(ans[i].fr=='g'){
			printf("getMin %d\n",ans[i].sc);
		}
	}
	return 0;
}
