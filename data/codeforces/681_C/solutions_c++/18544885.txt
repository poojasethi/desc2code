#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	priority_queue<ll,vector<ll>, greater<ll> > q;
	ll n,m,i,j,k;
	char s[40];
	vector<pair<ll,string> >v;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s;
		if(s[0]=='i')
		{
			cin>>j;
			v.push_back(make_pair(j,"insert"));
			q.push(j);
		}
		else if(s[0]=='g')
		{
			cin>>j;
			while(!q.empty() && j>q.top())
			{
				v.push_back(make_pair(-10000000000,"r"));
				q.pop();
			}
			if(q.empty() || j<q.top())
			{
				v.push_back(make_pair(j,"insert"));
				q.push(j);
			}
			v.push_back(make_pair(j,"getMin"));
		}
		else
		{
			if(q.empty())
			{
				v.push_back(make_pair(6,"insert"));
				q.push(6);
			}
			v.push_back(make_pair(-10000000000,"r"));
			q.pop();
		}
	}
	cout<<v.size()<<endl;
	for(i=0;i<v.size();i++)
	{
		if(v[i].first==-10000000000)
			cout<<"removeMin\n";
		else
			cout<<v[i].second<<" "<<v[i].first<<endl;
	}
}
