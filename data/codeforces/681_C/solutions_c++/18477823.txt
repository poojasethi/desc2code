#include <bits/stdc++.h>
using namespace std;

multiset<int> s;
int n,k;
char ans[1000006][40],a[30];

int main(){
	scanf("%d",&n);
	for(int i=0; i<n; ++i){
		int t;
		scanf("%s",a);
		if(*a=='i'){
			scanf("%d",&t);
			s.insert(t);
			sprintf(ans[k++],"insert %d",t);
		}
		if(*a=='r'){
			if(!s.size()){
				sprintf(ans[k++],"insert 3");
				s.insert(3);
			}
			s.erase(s.begin());
			sprintf(ans[k++],"removeMin");
		}
		if(*a=='g'){
			cin>>t;
			if(t > *(s.begin()) || !s.size()){
				while(s.size()&&*(s.begin())<t){
					s.erase(s.begin());
					sprintf(ans[k++],"removeMin");
				}
				if(*(s.begin())>t||!s.size()){
					s.insert(t);
					sprintf(ans[k++],"insert %d",t);
				}
			}
			else{
				if(t<*(s.begin())){
					s.insert(t);
					sprintf(ans[k++],"insert %d",t);
				}
			}
			sprintf(ans[k++],"getMin %d",t);
		}
	}
	printf("%d\n",k);
	for(int i=0; i<k; ++i){
		printf("%s\n",ans[i]);
	}	
}
