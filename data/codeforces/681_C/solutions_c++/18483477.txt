#include<bits/stdc++.h>
using namespace std;
#define TripleM ios_base::sync_with_stdio(false);cin.tie(0);
typedef long long ll;
priority_queue<int>Q;
int main(){
TripleM;
int n;cin>>n;
string s;
int x;
vector<pair<string,int> >V;
for(int i=0;i<n;i++){
    string s;cin>>s;
    int x;
    if(s=="insert"){
        cin>>x;
        Q.push(-x);
        V.push_back(make_pair(s,x));
    }
    else if(s=="removeMin"){
        if(Q.empty()){
            V.push_back(make_pair("insert",0));
        }else{
            Q.pop();
        }

            V.push_back(make_pair(s,2000000000));
    }
    else{
        cin>>x;
        while(!Q.empty()&&-Q.top()<x){

            Q.pop();
            V.push_back(make_pair("removeMin",2000000000));
        }
        if(Q.empty()||-Q.top()>x){
            Q.push(-x);
            V.push_back(make_pair("insert",x));
        }
        V.push_back(make_pair(s,x));
    }

}
cout<<V.size()<<endl;
for(int i=0;i<V.size();i++){
    cout<<V[i].first;
    if(V[i].second!=2000000000){
        cout<<" "<<V[i].second;
    }
    cout<<endl;
}
return 0;}
