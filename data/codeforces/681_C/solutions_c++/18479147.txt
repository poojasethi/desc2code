#include <bits/stdc++.h>
#define ll long long 
#define inf 1000000007
using namespace std;
priority_queue<ll , vector<ll >, greater<ll > > pq;
vector<pair <string, ll> >res;
int main() {
	// your code goes here
	ll n,m,i;
	string s;
	std::ios::sync_with_stdio(false);
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s;
		
		if(s=="insert")
		{
			cin>>m;
			pq.push(m);
			res.push_back(make_pair(s,m));
		}
		else if(s=="removeMin")
		{
			if(!pq.empty())
			pq.pop();
			else
			res.push_back(make_pair("insert",1));
			res.push_back(make_pair("removeMin",inf));
		}
		else if(s=="getMin")
		{
			cin>>m;
			while(!pq.empty() && pq.top()<m)
			{
				pq.pop();
				res.push_back(make_pair("removeMin",inf));
			}
		if(pq.empty() ||(!pq.empty() && pq.top()>m))
			{pq.push(m);
			res.push_back(make_pair("insert",m));
			}
			res.push_back(make_pair("getMin",m));
			
		}
	}
	cout<<res.size()<<endl;
for(i=0;i<res.size();i++)
{
	cout<<res[i].first<<" ";
	if(res[i].second!=inf)
	cout<<res[i].second;
	cout<<endl;
}
	
	return 0;
}
