#include <iostream>
#include <string>
#include <queue>
#include <vector>
#include <cstdio>
using namespace std;

char ans[1000000][20];
int main() {
  priority_queue<int, vector<int>, greater<int> > Q;
  int n;
  char op[100];
  scanf("%d", &n);
  int cnt = 0;
  for (int i = 0; i < n; i++) {
    scanf("%s", op);
    if (op[0] == 'i') {
      int t;
      scanf("%d", &t);
      Q.push(t);
      sprintf(ans[cnt++], "insert %d", t);
    } else if (op[0] == 'r') {
      if (Q.empty()) {
        sprintf(ans[cnt++], "insert 0");
      } else {
        Q.pop();
      }
      sprintf(ans[cnt++], "removeMin");
    } else {
      int t;
      scanf("%d", &t);
      while (!Q.empty() && Q.top() < t) {
        sprintf(ans[cnt++], "removeMin");
        Q.pop();
      }
      if (Q.empty() || Q.top() > t) {
        sprintf(ans[cnt++], "insert %d", t);
        Q.push(t);
      }
      sprintf(ans[cnt++], "getMin %d", t);
    }
  }
  printf("%d\n", cnt);
  for (int i = 0; i < cnt; i++) {
    printf("%s\n", ans[i]);
  }
}
