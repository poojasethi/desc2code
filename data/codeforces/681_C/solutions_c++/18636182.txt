#include<bits/stdc++.h>
using namespace std;

typedef pair<string,int>par;
priority_queue<int,vector<int>,greater<int> >pq;
par A[1000005];int r;

int main()
{
 //freopen("a.out","w",stdout);
 cin.tie(0);
 ios_base::sync_with_stdio(0);

 long long N;
 cin >> N;

 for(int i=1;i<=N;i++)
 {
  string a;int b;
  cin >> a;
  if(a[0]=='i')
  {
   cin >> b;
   pq.push(b);
   A[++r]=par(a,b);
  }
  else
   if(a[0]=='r')
   {
    if(pq.empty())
    {
     A[++r]=par("insert",-1000000000);
     A[++r]=par("removeMin",0);
    }
    else
    {
     A[++r]=par("removeMin",0);
     pq.pop();
    }
   }
   else
   {
    cin >> b;
    while(!pq.empty() && pq.top()<b)
    {
     A[++r]=par("removeMin",0);
     pq.pop();
    }
    if(pq.empty() || pq.top()>b)
    {
     A[++r]=par("insert",b);
     pq.push(b);
    }
    A[++r]=par("getMin",b);
   }
 }

 cout << r << '\n';
 for(int i=1;i<=r;i++)
 {
  if(A[i].first[0]!='r')
   cout << A[i].first << " " << A[i].second << "\n";
  else
   cout << A[i].first << "\n";
 }

    return 0;
}
