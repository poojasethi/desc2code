#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
	priority_queue<int> pq;
	vector<pair<string, int> > v;
	int n;
	string s;
	cin >> n;
	for(int i = 0, x; i < n; i++)
	{
		cin >> s;
		if(s == "insert" || s == "getMin") cin >> x;
		if(s == "getMin")
		{
			while(!pq.empty() && -pq.top() < x)
			{
				pq.pop();
				v.push_back(make_pair("removeMin", 0));
			}
			if(pq.empty() || -pq.top() != x)
			{
				pq.push(-x);
				v.push_back(make_pair("insert", x));
			}
		}
		else if(s == "insert") pq.push(-x);
		else
		{
			if(pq.empty()) v.push_back(make_pair("insert", 1));
			else pq.pop();
		}
		v.push_back(make_pair(s, x));
	}
	cout << v.size() << endl;
	for(int i = 0; i < v.size(); i++)
	{
		if(v[i].first == "removeMin") cout << v[i].first << endl;
		else cout << v[i].first << ' ' << v[i].second << endl;
	}
}
