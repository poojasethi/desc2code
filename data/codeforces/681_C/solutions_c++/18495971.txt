#include<cstdio>
#include<cstring>
#include<string>
#include<iostream>
#include<queue>
#include<cstdlib>
#include<functional>
#include<vector>
using namespace std;

int main(){
	int n,a,k;
	char buff[50],comd[50];
	vector<string> ans;
	priority_queue<int,vector<int>,greater<int> > q;
	scanf("%d",&n);
	while(n--){
		scanf("%s",comd);
		if(strcmp(comd,"insert")==0){
			scanf("%s",buff);
			k=atoi(buff);
			ans.push_back(string(comd)+" "+string(buff));
			q.push(k);
		}
		else if(strcmp(comd,"removeMin")==0){
			if(q.empty()) ans.push_back("insert 0");	
			ans.push_back(comd);	
			if(!q.empty()) q.pop();
			
		}
		else{
			scanf("%s",buff);
			k=atoi(buff);
			while(!q.empty() && q.top()<k){
				q.pop();
				ans.push_back("removeMin");
			}
			if(!q.empty() && q.top()==k)
				ans.push_back(string(comd)+" "+string(buff));
			else{
				q.push(k);
				ans.push_back("insert "+string(buff));
				ans.push_back(string(comd)+" "+string(buff));
			}
		}

	}
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)
		printf("%s\n",ans[i].c_str());
	return 0;
}
