#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> ii;

int main(){
	priority_queue<int, vector<int>, greater<int> > fp;
	int n, x;
	int quant;
	vector<ii> ans;
	string s;

	quant = 0;
	cin>>n;
	for(int i = 0; i<n; i++){
		cin>>s;
		if(s[0]=='i'){
			scanf("%d", &x);
			ans.push_back(ii(0, x));
			fp.push(x);
		}else if(s[0]=='r'){
			if(fp.empty()){
				ans.push_back(ii(0, -1000000000));
				fp.push(-1000000000);
			}
			fp.pop();
			ans.push_back(ii(1, 0));
		}else{
			scanf("%d", &x);
			while(!fp.empty() && x>fp.top()){
				ans.push_back(ii(1, 0));	
				fp.pop();
			}
			if(fp.empty() || x!=fp.top()){
				fp.push(x);
				ans.push_back(ii(0, x));
			}
			ans.push_back(ii(2, x));
		}
	}

	cout<<ans.size()<<endl;
	for(vector<ii>::iterator it = ans.begin(), fim = ans.end(); it!=fim; it++){
		if(it->first==0){
			printf("insert %d\n", it->second);
		}else if(it->first==1){
			printf("removeMin\n");
		}else{
			printf("getMin %d\n", it->second);
		}
	}


}
