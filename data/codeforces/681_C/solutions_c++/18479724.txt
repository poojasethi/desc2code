#include<bits/stdc++.h>
using namespace std;
char str[100];
int main()
{	int n;
	priority_queue<int,vector<int>,greater<int> > q;
	vector<pair<char,int> > v;
	int number=0;
	scanf("%d",&n);
	while(n--)
	{	scanf(" %s",str);
		if(str[0]=='i')
		{	scanf("%d",&number);
			q.push(number);
				v.push_back(make_pair('i',number));
		}
		else	if(str[0]=='g')
			{	scanf("%d",&number);
				while(!q.empty()&&q.top()<number)
				{	v.push_back(make_pair('r',0));
					q.pop();
				}
				if(q.empty()||q.top()>number)
				{	q.push(number);
					v.push_back(make_pair('i',number));
				}
				v.push_back(make_pair('g',number));
			}
			else
			{	if(q.empty())
				{	v.push_back(make_pair('i',1));
					q.push(1);
				}
				q.pop();
				v.push_back(make_pair('r',-1));
			}
	}
	printf("%d\n",v.size());
	for(int i=0;i<v.size();i++)
	{	if(v[i].first=='i')printf("insert %d\n",v[i].second);
		if(v[i].first=='g')printf("getMin %d\n",v[i].second);
		if(v[i].first=='r')printf("removeMin\n");
	}	
	return 0;
}
