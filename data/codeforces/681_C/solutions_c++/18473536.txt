#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <memory.h>
#include <set>

using namespace std;
typedef pair <int, int> pii;

char op[3][10]={"insert","getMin","removeMin"};
char str[10];
multiset <int> s;
pii ans[1000010];

int main (){
	
	int N,i,j,x,M=0;
	
	scanf("%d",&N);
	for(i=0; i<N; i++){
		scanf("%s",str);
		for(j=0; j<2; j++){
			if(!strcmp(str,op[j])){
				scanf("%d",&x);
				break;
			}
		}
		if(j==0) {
			ans[M++]=pii(0, x);
			s.insert(x);
			continue;
		}
		
		if(j==1) {
			while(!s.empty() && (*s.begin())<x){
				ans[M++]=pii(2, *(s.begin()));
				s.erase(s.begin());
			}
			if(s.empty() || (*s.begin())>x){
				ans[M++]=pii(0, x);
				s.insert(x);
			}
			ans[M++]=pii(1,x);
			continue;
		}
		
		if(s.empty()) ans[M++]=pii(0,1);
		else s.erase(s.begin());
		ans[M++]=pii(2, 0);
	}
	
	printf("%d\n",M);
	for(i=0; i<M; i++){
		printf("%s",op[ans[i].first]);
		if(ans[i].first!=2) printf(" %d\n", ans[i].second);
		else puts("");
	}
			
	return 0;
}
