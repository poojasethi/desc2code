#include<cstdio>
#include<queue>
#include<vector>
#include<algorithm>
using namespace std;

int n;
priority_queue<int> q;
char s[111];
int num;
vector<pair<char, int> > ans;

int main() {
	while(!q.empty())q.pop();
	ans.clear();
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%s%d", s, &num);
		if(s[0]=='i') {
			q.push(-num);
			ans.push_back(make_pair('i', num));
		} else if(s[0]=='r') {
			if(q.empty()) 
				ans.push_back(make_pair('i', 1));
			else 
				q.pop();
			ans.push_back(make_pair('r', 1));
		} else {
			if(!q.empty()) {				
				int v = -q.top();
				while(v<num) {
					q.pop();
					ans.push_back(make_pair('r', 1));
					if(q.empty()) break;
					v=-q.top();
				}
			}
			if(q.empty() || -q.top()!=num) {
				ans.push_back(make_pair('i', num));
				q.push(-num);
			}
			ans.push_back(make_pair('g', num));
		}
	}
	
	printf("%d\n", ans.size());
	for(int i=0;i<ans.size();i++) {
		if(ans[i].first == 'i') printf("insert %d\n", ans[i].second);
		else if(ans[i].first == 'g') printf("getMin %d\n", ans[i].second);
		else printf("removeMin\n");
	}
	
	return 0;
}
