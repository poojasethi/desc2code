#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
int main() {
	int n;
	scanf("%d", &n);
	string s[n];
	int a[n];
	for(int i =0 ; i < n; i++) {
		cin >> s[i];
		if(s[i][0]!='r') cin >> a[i];
	}
	priority_queue<int> q;
	vector<pair<string, int> > v;
	for(int i = 0; i < n; i++) {
		if(s[i]=="insert"){ q.push(-a[i]); v.pb(mp("insert", a[i])); }		
		if(s[i]=="removeMin") {
			if(q.empty()) {
				v.pb(mp("insert", 1));
				q.push(1);
			}
			q.pop();
			v.pb(mp("removeMin", -1));
		}
		
		if(s[i]=="getMin"){ 
			while(!q.empty()) {
				if(-q.top()<a[i]) { 
					v.pb(mp("removeMin", -1));
					q.pop();
				}
				else break;
			}
			if(q.empty() or -q.top()!=a[i]) 
			{
				 v.pb(mp("insert", a[i]));
				q.push(-a[i]);
			} 
			v.pb(mp("getMin", a[i]));
		}
	}
	cout << v.size() << "\n";
	for(int  i= 0; i < v.size(); i++) {
		cout << v[i].first << " ";  
		if(v[i].first[0]!='r') cout << v[i].second;
		cout << "\n";
	}
}
