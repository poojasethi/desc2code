#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
vector <pair <string, ll> > ans;
multiset <ll> q;
char s[15];

main() {
	ll n;
	scanf("%d", &n);
	while (n--) {
		ll x;
		scanf("%s", s);
		if (strcmp(s, "removeMin") == 0) {
			if (q.empty())
				ans.push_back(make_pair("insert", 0));
			else q.erase(q.begin());
		}
		else if (strcmp(s,"getMin") == 0) {
			scanf("%I64d", &x);
			while (q.empty() == false && *q.begin() < x) {
				q.erase(q.begin());
				ans.push_back(make_pair("removeMin", 0));
			}
			if (q.empty() || *q.begin() > x) {
				q.insert(x);
				ans.push_back(make_pair("insert", x));
			}
		}
		else {
			scanf("%I64d", &x);
			q.insert(x);
		}
		ans.push_back(make_pair(s, x));
	}
	printf("%d\n", ans.size());
	for (ll i = 0; i < ans.size(); i++) {
		printf("%s", ans[i].first.c_str());
		if (ans[i].first != "removeMin")
			printf(" %I64d",ans[i].second);
		printf("\n");
	}
}
