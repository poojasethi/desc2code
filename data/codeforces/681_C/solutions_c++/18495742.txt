#include<cstdio>
#include<cstring>
#include<string>
#include<iostream>
#include<queue>
#include<cstdlib>
#include<functional>
#include<vector>
using namespace std;

int main(){
	int n,a,x,k;
	char buff[50],tmp[50];
	string comd,val;
	vector<string> ans;
	priority_queue<int,vector<int>,greater<int> > q;
	scanf("%d",&n);
	while(n--){
		scanf("%s",tmp);
		comd=tmp;
		if(comd.compare("insert")==0){
			scanf("%s",buff);
			k=atoi(buff);
			val=buff;
			ans.push_back(comd+" "+val);
			q.push(k);
		}
		else if(comd.compare("removeMin")==0){
			if(q.empty()) ans.push_back("insert 0");	
			ans.push_back(comd);	
			if(!q.empty())q.pop();
			
		}
		else{
			scanf("%s",buff);
			k=atoi(buff);
			val=buff;
			while(!q.empty() && q.top()<k){
				q.pop();
				ans.push_back("removeMin");
			}
			if(!q.empty() && q.top()==k)
				ans.push_back(comd+" "+val);
			else{
				q.push(k);
				ans.push_back("insert "+val);
				ans.push_back(comd+" "+val);
			}
		}

	}
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)
		printf("%s\n",ans[i].c_str());
	return 0;
}
