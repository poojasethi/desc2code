#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<string.h>
#include<queue>
#include<vector>
#include<string>
using namespace std;
const int N=2000001;
priority_queue< int,vector<int>,greater<int> >d;
string a[N];
int b[N],m;
int main(void)
{
	//freopen("sduoj.in","r",stdin);
	std::ios::sync_with_stdio(false);
	int n,i,k;string str;
	cin>>n;m=0;
	for(i=1;i<=n;i++)
	{
		cin>>str;if(str!="removeMin")cin>>k;
		if(str=="insert"){a[++m]=str;b[m]=k;d.push(k);continue;}
		else if(str=="removeMin")
		{
			if(d.empty()){a[++m]="insert";b[m]=1;d.push(1);}
			a[++m]=str;d.pop();continue;
		}
		else
		{
			while(!d.empty()&&d.top()<k){a[++m]="removeMin";d.pop();}
			if(d.empty()||!d.empty()&&d.top()>k){a[++m]="insert";b[m]=k;d.push(k);}
			a[++m]=str;b[m]=k;
		}
	}
	cout<<m<<endl;
	for(i=1;i<=m;i++)if(a[i]=="removeMin")cout<<"removeMin"<<endl;
	else cout<<a[i]<<' '<<b[i]<<endl;
	return 0;
}
