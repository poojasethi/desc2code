# include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e6 + 100;

string S[MAXN];

int x[MAXN];

int main() {
	ios_base::sync_with_stdio(false);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++) { 
		cin >> S[i];
		if (S[i][0] != 'r')
			cin >> x[i];
	}
	
	multiset <int> s;
	vector <pair <char,int> > v;
	for (int i = 0; i < n; i++) {
		if (S[i][0] == 'i')
			s.insert(x[i]);
		if (S[i][0] == 'g') {
			while (s.size() > 0 && *s.begin() < x[i])
				v.push_back(make_pair('r', -1)), s.erase(s.find(*s.begin()));
			if (s.size() == 0 || *s.begin() != x[i]) 
				v.push_back(make_pair('i', x[i])), s.insert(x[i]);
		}
		if (S[i][0] == 'r') {
			if (s.size() == 0)
				v.push_back(make_pair('i', -1000000000));
			else 
				s.erase(s.find(*s.begin()));
		}
		v.push_back(make_pair(S[i][0], x[i]));
	}

	cout << v.size() << '\n';
	for (int i = 0; i < v.size(); i++) {
		if (v[i].first == 'i')
			cout << "insert " << v[i].second << '\n';
		if (v[i].first == 'r')
			cout << "removeMin" << '\n';
		if (v[i].first == 'g')
			cout << "getMin " << v[i].second << '\n';
	}

	return 0;
}
