#include<iostream>
#include<stack>
#include<queue>
#include<cstdio>
using namespace std;
const int MIN=1e9+20;
int s[1000005];
int num[1000005];
int main()
{
	int n;cin >> n;
	int x;
	int flag=0;
	int k=0;
	string tmp;
	priority_queue<int,vector<int>,greater<int> >p;
	while(n--){
		cin >> tmp;
		if(tmp[0]!='r')	cin >> x;
		if(tmp[0]=='i'){
			p.push(x);
			s[k]=1;
			num[k++]=x;
		}	
		else if(tmp[0]=='r'){
			if(!p.empty())	p.pop();
			else{
				s[k]=1;
				num[k++]=0;
			}
			s[k]=2;
			num[k++]=x;
		}
		else if(tmp[0]=='g'){
			while(!p.empty() && p.top() < x){
				s[k++]=2;
				p.pop();
			}	
			if(p.empty()||p.top() > x){
				s[k]=1;
				num[k++]=x;
				p.push(x);
			}
			s[k]=3;
			num[k++]=x;
		}
	}
	printf("%d\n",k);
	for(int i=0; i<k; ++i)
		if(s[i]==1)	printf("insert %d\n",num[i]);
		else if(s[i]==2)		printf("removeMin\n");
		else	printf("getMin %d\n",num[i]);
	return 0;
}
