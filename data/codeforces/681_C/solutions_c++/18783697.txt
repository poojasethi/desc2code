#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<queue>
#define ll long long
#define inf 1e9
#define md
using namespace std;
priority_queue<int,vector<int>,greater<int> > q;
char st[20],s[1000010];
int a[1000010];
int main()
{
	int n; scanf("%d",&n);
	int w=0;
	for (int i=1;i<=n;i++)
	{
		scanf("%s",st);
		if (st[0]=='i')
		{
			int x; scanf("%d",&x);
			q.push(x); w++; s[w]='i'; a[w]=x;
		}
		else if (st[0]=='r')
		{
			if (q.empty())
			{
				q.push(1); w++; s[w]='i'; a[w]=1;
			}
			w++; s[w]='r'; q.pop();
		}
		else
		{
			int x; scanf("%d",&x);
			while (!q.empty()&&q.top()<x)
			{
				w++; s[w]='r'; q.pop();
			}
			if (q.empty()||q.top()>x)
			{
				q.push(x); w++; s[w]='i'; a[w]=x;
			}
			w++; s[w]='g'; a[w]=q.top();
		}
	}
	printf("%d\n",w);
	for (int i=1;i<=w;i++)
	  if (s[i]=='i') printf("insert %d\n",a[i]);
	  else if (s[i]=='r') printf("removeMin\n");
	  else printf("getMin %d\n",a[i]);
	return 0;
}
