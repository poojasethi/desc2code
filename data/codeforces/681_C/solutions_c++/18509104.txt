#include<bits/stdc++.h>
#define RAN 2e9
using namespace std;
int main()
{
	int n, temp, num;
	string meth;
	vector<pair<string, int> > v;
	priority_queue<int> pq;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>meth;
		if(meth=="insert")
		{
			cin>>num;
			v.push_back(make_pair(meth, num));
			pq.push(-num);
		}
		else if(meth=="removeMin")
		{
			if(!pq.empty())
			{
				pq.pop();
			}
			else
			{
				v.push_back(make_pair("insert", 0));
			}
			v.push_back(make_pair(meth, RAN));
		}
		else
		{
			cin>>num;
			while(!pq.empty() && -pq.top()<num)
			{
				v.push_back(make_pair("removeMin", RAN));
				pq.pop();
			}
			if(pq.empty() || -pq.top()>num)
			{
				pq.push(-num);
				v.push_back(make_pair("insert", num));
			}
			v.push_back(make_pair(meth, num));
		}
	}
	cout<<v.size()<<endl;
	for (vector<pair<string, int> >::iterator it = v.begin(); it!=v.end();++it) {
        cout << (*it).first;
        if ((*it).second != RAN) {
            cout << " " << (*it).second;
        }
        cout << "\n";
    }
	return 0;
}
