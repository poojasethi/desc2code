#include<bits/stdc++.h>
using namespace std;
int main()
{
priority_queue<int,vector<int> ,greater<int> > q;
vector<pair<string,int> > v;
int n;
cin>>n;
for(int i=0;i<n;i++)
{
string s;
cin>>s;
int n1;
if(s=="insert")
{
cin>>n1;
q.push(n1);
v.push_back(make_pair(s,n1));
}
else if(s=="removeMin")
{
if(q.empty())
{
v.push_back(make_pair("insert",1));
q.push(1);
}
q.pop();
v.push_back(make_pair(s,0));
}
else
{
cin>>n1;
if(q.empty())
{
v.push_back(make_pair("insert",n1));
q.push(n1);
}
int s1=q.top();
while(s1!=n1)
{
if(s1>n1)
{
q.push(n1);
v.push_back(make_pair("insert",n1));
break;
}
q.pop();
v.push_back(make_pair("removeMin",0));
if(q.empty())
{
q.push(n1);
v.push_back(make_pair("insert",n1));
break;
}
s1=q.top();

}
v.push_back(make_pair(s,n1));
}
}
cout<<v.size()<<"\n";
for(int i=0;i<v.size();i++)
{
if(v[i].first=="removeMin")
cout<<v[i].first<<"\n";
else
cout<<v[i].first<<" "<<v[i].second<<"\n";

}

}
