#include <bits/stdc++.h>
using namespace std;
int main ()
{
int long long a,b,i,j,minn=-10000000,maxn=100000;
cin>>a>>b;
for (i=1;i<=a;i++)
{
int long long c,d;
cin>>c>>d;
minn=max(minn,min(c,d));
maxn=min(maxn,max(c,d));
}
if (minn>maxn){cout<<-1;}
else if (b<minn){cout<<minn-b;}
else if (b>maxn){cout<<b-maxn;}
else {cout<<0;}
}
