#include <stdio.h>
#include <algorithm>
using namespace std;

int main(void) {
	int n, x0, ans=0, i, x, y, L=0, R=1000;
	scanf("%d%d", &n, &x0);
	for(i=0;i<n;i++){
		scanf("%d%d", &x, &y);
		if(x>y) swap(x, y);
		if(L<x) L = x;
		if(R>y) R = y;
	}
	if(x0<L) ans = L-x0;
	else if(x0>R) ans = x0-R;
	if(L>R) printf("-1");
	else
	printf("%d\n", ans);
	return 0;
}
