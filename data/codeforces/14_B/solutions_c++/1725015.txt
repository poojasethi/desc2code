#include<cstdio>
int ans,a,b,i,j,m,n,ll,rr=1005;
int main()
{
    scanf("%d%d",&n,&m);
    for (i=1;i<=n;i++)
    {
        scanf("%d%d",&a,&b);
        if (a>b)j=a,a=b,b=j;
        if (a>ll)ll=a;
        if (b<rr)rr=b;
    }
    if (ll>rr)printf("-1");
    else if (ll>m)printf("%d",ll-m);
         else if (m>rr)printf("%d",m-rr);
         else printf("0");
    return 0;
}
