#include <iostream>
using namespace std;
int n,x,i;
int a,b,aux;
int main()
{ 
	cin>>n>>x;
	int max,min;
	max=0;min=1001;
	for (i=1;i<=n;i++)
	{
		cin>>a>>b;
		if (a>b)
		{
			aux=a;
			a=b;
			b=aux;
		}
		if (a>max) max=a;
		if (b<min) min=b;
	}
	if (max>min)
	{
		cout<<-1;
		return 0;
	}
	if (x>=max && x<=min)
		cout<<0;
	if (x<max) cout<<max-x;
	if (x>min) cout<<x-min;
	return 0;
}
