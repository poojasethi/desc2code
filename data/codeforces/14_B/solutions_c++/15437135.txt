#include<bits/stdc++.h>
using namespace std ;

int arr[1002] ;

int main(){
int n,x ;
cin>>n>>x ;
int l,r;
for(int i =0;i<n;i++)
{
  cin>>l>>r ;
  if(l>r) swap(l,r) ;
  for(int j =l;j<=r;j++) arr[j]++ ;
}
int ans = 1000000 ;
for(int i =1;i<=1000;i++){
  if(arr[i]==n&&abs(x-i)<ans) ans = abs(x-i) ;
  }
  if(ans!=1000000) cout<<ans ;
  else cout<<"-1" ;
}
