#include <cstdio>
using namespace std;

int N,X0,L,R,a,b;

int main()
{
	scanf("%d %d", &N, &X0);
	N--;
	scanf("%d %d", &L, &R);
	if(L > R)
	{
		int t = L;
		L = R;
		R = t;
	}
	while(N--)
	{
		scanf("%d %d", &a, &b);
		if(a > b)
		{
			int t = a;
			a = b;
			b = t;	
		}
		if(a > L) L = a;
		if(b < R) R = b;
	}
	if(L > R)
		printf("-1\n");
	else if(X0 > R)
		printf("%d\n", X0-R);
	else if(X0 < L)
		printf("%d\n", L-X0);
	else
		printf("0\n");
}
