#include<cstdio>
#include<algorithm>
using namespace std; 
int n,x,a,b,s1,s2;
int main(){
  scanf("%d%d",&n,&x);
  for(int i=0;i<n;i++){
    scanf("%d%d",&a,&b);
    if(a > b) swap(a,b);
    if(i == 0) s1 = a,s2 = b;
    else s1 = max(a,s1), s2 = min(s2,b);
  }
  if(s1 > s2) printf("-1");
  else if(s1 <= x && x <= s2) printf("0");
  else printf("%d",min(abs(s1-x),abs(s2-x)));
  return 0;
}
