#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int n,x,maxn=0,minn=10000;
	cin>>n>>x;
	for(int i=0;i<n;i++)
	{
		int a,b;
		cin>>a>>b;
		if(a>b)
		{
			int w=a;
			a=b;
			b=w;
		}
		minn=min(minn,b);
		maxn=max(maxn,a);
	}
	if(x<minn&&x>maxn)cout<<0<<endl;
	else if(minn<maxn)cout<<-1<<endl;
	else cout<<min(abs(x-minn),abs(x-maxn))<<endl;
}
