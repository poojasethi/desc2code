#include <iostream>
#include <cmath>
using namespace std;

int main()
{
  int n,x0;
  cin>>n>>x0;
  int xmin=0;
  int xmax=1000;
  for (int i=0;i<n;i++) {
    int a,b;
    cin>>a>>b;
    if (a>b) swap(a,b);
    xmin=max(a,xmin);
    xmax=min(b,xmax);
  }
  if (xmin>xmax) cout<<-1<<endl;
  else if (x0<xmin) cout<<xmin-x0<<endl;
  else if (x0>xmax) cout<<x0-xmax<<endl;
  else cout<<0<<endl;
}
