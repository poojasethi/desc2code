#include<stdio.h>
#include<iostream>
using namespace std;
int main(){
	int n,x0;
	int sta,des;
	int temp;
	int f=-2,l=1002;
	int result;
	cin>>n>>x0;
	for(int i=1;i<=n;++i){
		cin>>sta>>des;
		if(sta>des) temp=sta,sta=des,des=temp;
		f=(sta>f)?sta:f;
		l=(des<l)?des:l;
		if(f>l) break;
	}
	if(f>l) result=-1;
	else{
		if(x0<f) result=f-x0;
		else if(x0>l) result=x0-l;
		else result=0;
	}
	cout<<result<<endl;
	return 0;
}
