//IN THE NAME OF GOD
#include <iostream>
using namespace std;
int n,ja,r=2000,l,x,y;
int f(int x){if(x<0) return -x; return x;}
int main()
{
    cin>>n>>ja;
    for(int i=0;i<n;i++)
    {
        cin>>x>>y;
        l=max(min(x,y),l);
        r=min(max(x,y),r);
    }
    if(l>r) cout<<-1;
    else
        if(ja>=l&&ja<=r)  cout<<0;
        else cout<<min(f(ja-l),f(ja-r));
}
