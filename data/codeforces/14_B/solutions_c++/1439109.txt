#include <iostream>
#include <algorithm>
using namespace std;
int cnt[1001];
int main(){
  int n,x;
  cin>>n>>x;
  for(int i=0;i<n;i++){
    int a,b;
    cin>>a>>b;
    if(a>b)swap(a,b);
    for(int j=a;j<=b;j++)cnt[j]++;
  }
  int res=10000;
  for(int i=0;i<1001;i++){
    if(cnt[i]==n)res=min(res,abs(i-x));
  }
  if(res==10000)cout<<-1<<endl;
  else cout<<res<<endl;

  return 0;
}
