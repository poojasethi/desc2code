#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n,x,l=0,r=1000,a,b;
	scanf("%d%d",&n,&x);
	while(n--)
	{
		scanf("%d%d",&a,&b);
		if(a>b) swap(a,b);
		if(a>l) l=a;
		if(b<r) r=b;
	}
	if(l>r) printf("-1\n");
	else
	{
		if(x<l) printf("%d\n",l-x);
		else if(x>r) printf("%d\n",x-r);
		else printf("0\n");
	}
	return 0;
}
