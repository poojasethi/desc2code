#include <iostream>

using namespace std;

int main()
{
    int n,x;
    cin>>n>>x;
    int st,en;
    int mxst=0,mnen=100000;
    for(int i=0;i<n;i++){
        cin>>st>>en;
        if(st>en)swap(st,en);
        mxst=max(mxst,st);
        mnen=min(mnen,en);
    }
    if(mxst>mnen)cout<<-1;
    else if(x>mnen)cout<<x-mnen;
    else if(x<mxst)cout<<mxst-x;
    else cout<< 0;
    return 0;
}
