#include<iostream>
using namespace std;
int main()
{
	int n,x,a,b,i,maxa,minb,aux;
	cin>>n>>x;
	maxa=-1;
	minb=1001;
	for(i=1;i<=n;i++)
	{
		cin>>a>>b;
		if (a>b)
		{
			aux=a;
			a=b;
			b=aux;
		}
		if (a>maxa)
			maxa=a;
		if (b<minb)
			minb=b;
	}
	if (maxa>minb)
	{
		cout<<-1;
		return 0;
	}
	if (x>=maxa && x<=minb)
		cout<<0;
	if (x<maxa)
		cout<<maxa-x;
	if (x>minb)
		cout<<x-minb;
	return 0;
}
