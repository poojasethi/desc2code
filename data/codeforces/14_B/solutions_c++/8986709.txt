#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int main(){
	int n,x,a=0,b=1000,c,a1,b1;
	scanf("%d%d",&n,&x);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&a1,&b1);
		if(max(a1,b1)<b)b=max(a1,b1);
		if(min(a1,b1)>a)a=min(a1,b1);
	}
	if(a>b)printf("-1");
	else if(x>=a){
		if(x<=b)printf("0");else printf("%d",abs(x-b));
	}else printf("%d",abs(x-a));
	return 0;
}
