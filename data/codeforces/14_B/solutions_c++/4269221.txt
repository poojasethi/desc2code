#include<iostream>
#include<cstring>
using namespace std;
int main(){
	int n,x,xmin=-1,xmax=100000;
	cin>>n>>x;
	for(int i=0;i<n;i++){
		int a,b;
		cin>>a>>b;
		xmin=max(xmin,min(a,b));
		xmax=min(xmax,max(a,b));
	}
	if(xmin>xmax)
		cout<<-1<<endl;
	else if(xmin<=x && x<=xmax)
		cout<<0<<endl;
	else
		cout<<(x<xmin?xmin-x:x-xmax)<<endl;
}
