#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

int n,x,cnt[1005];
int ans=1005;
int main(){
  scanf("%d %d",&n,&x);
  FI(i,1,n){
    int a,b;
    scanf("%d %d",&a,&b);
    if(a>b) swap(a,b);
    FI(j,a,b) cnt[j]++;
  }
  FI(i,0,1000) if(cnt[i]==n) ans=min(ans,abs(x-i));
  if(ans==1005) ans=-1;
  printf("%d\n",ans);
  return 0;
}
