#include <iostream>
using namespace std;

int main()
{
	int n, x0, a, b, t, u, v;
	cin>>n>>x0;
	a = 0;
	b = 1000;
	while( n-- )
	{
		cin>>u>>v;
		if( u > v )
		{
			t = u;
			u = v;
			v = t;
		}
		a = u > a ? u : a;
		b = v < b ? v : b;
		if( b < a )
		{
			cout<<-1;
			return 0;
		}
	}
	if( x0 < a )
		cout<<a-x0;
	else if( x0 > b )
		cout<<x0-b;
	else
		cout<<0;
	return 0;
}
