#include<iostream>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	int n,x;
	cin>>n>>x;
	int l=-1,r=100000;

	for(int i=0;i<n;++i)
	{
		int a,b;

		cin>>a>>b;

		if(min(a,b)>r || max(a,b)<l)
		{
			cout<<"-1";
			return 0;
		};
		l=max(l,min(a,b));
		r=min(r,max(a,b));


	};

	cout<<((x>=l && x<=r)? 0 : min(labs(r-x),labs(x-l)));



	return 0;
};
