#include<cstdio>
#include<algorithm>
using namespace std;
int b[1010];
int main()
{
 int n,x,j,i,a,B,ans=0;
 scanf("%d%d",&n,&x);
 for (i=1;i<=n;i++)
   {scanf("%d%d",&a,&B);
    if (a>B)swap(a,B);
    for (j=a;j<=B;j++)b[j]++;
   }
 ans=1000000;
 for (i=1;i<=1000;i++)
 if (b[i]==n)if (ans>abs(i-x))
 ans=abs(i-x);
 printf("%d\n",(ans==1000000)?-1:ans);
 return 0;
}
