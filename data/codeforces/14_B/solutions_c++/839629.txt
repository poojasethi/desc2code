#include<stdio.h>
int main()
{
	int n,x0,l=0,r=1000,a,b;
	bool flag=false;
	scanf("%d%d",&n,&x0);
	while(n--)
	{
		scanf("%d%d",&a,&b);
		if(a>b)
		{
			a=a^b;b=b^a;a=a^b;
		}
		if(b<l||a>r)flag=true;
		else
		{
			l=l>a?l:a;
			r=r<b?r:b;
		}
	}
	if(flag)printf("-1\n");
	else
	{
	if(x0>=l&&x0<=r)printf("0\n");
	else if(x0<l)printf("%d\n",l-x0);
	else printf("%d\n",x0-r);
	}
	return 0;
}
