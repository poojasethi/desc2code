#include <cstdio>
#include <iostream>
using namespace std;
int l,r,n,x,u,v;
int main()
{
	scanf("%d%d",&n,&x);
	scanf("%d%d",&v,&u);
	l=min(v,u),r=max(v,u);
	for (int i=1;i<n;i++)
	{
		scanf("%d%d",&u,&v);
		l=max(l,min(u,v)),r=min(r,max(u,v));
		if (l>r)
		{
			puts("-1");
			return 0;
		}
	}
	if (x>=l&&x<=r) puts("0");
	else if (x<l) printf("%d\n",l-x);
	else printf("%d\n",x-r);
	return 0;
}
