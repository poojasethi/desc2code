#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int n,x,l,r,ans=0,tl=0,tr=1000;
	scanf("%d%d",&n,&x);
	while(n--){
		scanf("%d%d",&l,&r);
		if(l>r)
			swap(l,r);
		tl=max(tl,l);
		tr=min(tr,r);
	}
	if(tl>tr)
		puts("-1");
	else if(tl<=x&&x<=tr)
		puts("0");
	else
		printf("%d\n",min(abs(x-tl),abs(x-tr)));
	return 0;
}
