#include<iostream>
#include <cstdlib> 
using namespace std;
int main()
{
    int a,b,t,c[1001]={0},x,n,i,j;
    cin>>n>>x;
    for(i=1;i<=n;i++)
    {
        cin>>a>>b;
        if(a>b)t=a,a=b,b=t;
        for(;a<=b;a++)c[a]++;
    }
    t=-1;
    for(i=1;i<=1000;i++)
    {
        if(c[i]==n&&(t==-1||abs(i-x)<t))t=abs(i-x);
    }
    cout<<t;
    return 0;
}
