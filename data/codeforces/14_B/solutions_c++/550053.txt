#include <iostream>
#define MAX 2000
using namespace std;
int N,T;
int x=0,y=MAX;
void read()
{
	int a,b;
	cin >> N >> T;
	for(int i=1;i<=N;i++)
	{
		cin >> a >> b;
		x=max(x,min(a,b));
		y=min(y,max(a,b));
	}
}
int main()
{
	read();
	if(y<x)
		cout << -1 << endl;
	else if(x<=T && y>=T)
		cout << 0 << endl;
	else
		cout << min(max(T-x,x-T),max(T-y,y-T));
	return 0;
}
