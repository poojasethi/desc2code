#include <cstdio>
#include <algorithm>
#include <climits>
#include <cmath>
using namespace std;
int main() {
	int n,x,a,b,l,r;
	scanf("%d%d",&n,&x);
	l=-INT_MAX;
	r=INT_MAX;
	for(int i=0;i<n;i++) {
		scanf("%d%d",&a,&b);
		l=max(l,min(a,b));
		r=min(r,max(a,b));
		if(l>r) {
			printf("-1\n");
			return 0;
		}
	}
	if(l<=x && x<=r)
		printf("0\n");
	else
		printf("%d\n",min(abs(x-l),abs(x-r)));
	return 0;
}
