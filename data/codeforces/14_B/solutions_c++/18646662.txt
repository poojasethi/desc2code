#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,x,a,i,b,tr=1000,tl=0;
    cin>>n>>x;
    for(i=0;i<n;i++)
    {
        cin>>a>>b;
        if(a>b)
            swap(a,b);
        tl=max(tl,a);
        tr=min(tr,b);
    }
    if(x>=tl && x<=tr)
        cout<<"0\n";
    else if(tl>tr)
        cout<<"-1\n";
    else
        cout<<min(abs(x-tl),abs(x-tr))<<"\n";
    return 0;
}
