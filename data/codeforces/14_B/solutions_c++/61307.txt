#include <cstdio>
#include <algorithm>
using namespace std;

int n,x;

int main()
{
	int maxs=0,mint=1000;
	scanf("%d %d",&n,&x);
	while(n--)
	{
		int s,t;
		scanf("%d %d",&s,&t);
		if(s>t)
		{
			int tt=t;
			t=s;
			s=tt;
		}
		maxs=max(maxs,s);
		mint=min(mint,t);
	}
	if(mint<maxs)printf("%d\n",-1);
	else if(x<maxs)printf("%d\n",maxs-x);
	else if(mint<x)printf("%d\n",x-mint);
	else printf("%d\n",0);
	return 0;
}
