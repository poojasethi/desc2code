#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
	int n, x0, a, b, l=-1, r=1001;
	scanf("%d%d", &n, &x0);
	for(int i=0; i<n; i++) {
		scanf("%d%d", &a, &b);
		l=max(l, min(a, b));
		r=min(r, max(a, b));
	}
	if(l>r) puts("-1");
	else if(x0<l) printf("%d\n", l-x0);
	else if(x0>r) printf("%d\n", x0-r);
	else puts("0");
	return 0;
}
