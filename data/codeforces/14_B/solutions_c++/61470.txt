#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;
int main()
{
	int a,b,cx=0,cy=1000,n,x0;
	cin>>n>>x0;
	for(int i=0;i<n;i++)
	{
		cin>>a>>b;
		if(a>b)
		swap(a,b);
		if(a>cx)
		cx=a;
		if(b<cy)
		cy=b;
	}
	int d;
	if(x0>=cx&&x0<=cy)
	d=0;
	else if(cy<cx)
	d=-1;
	else
	d=min(abs(cx-x0),abs(cy-x0));
	cout<<d<<endl;
	return 0;
}
