#!/usr/bin/python
# coding: utf-8
import sys

settings = sys.stdin.readline().strip().split()
case_num = settings[0]
tyo = int(settings[1])
pos_lis = [l.strip().split() for l in sys.stdin.readlines()]
mins = []
maxs = []
ans = 0
for pos in pos_lis:
    """
    if pos[0] > pos[1]:
        mins.appned(pos[1])
        maxs.appned(pos[0])
    else:
        mins.append(pos[0])
        maxs.append(pos[1])
    """
    mins.append(min(int(pos[0]), int(pos[1])))
    maxs.append(max(int(pos[0]), int(pos[1])))
mins.sort()
mins.reverse()
maxs.sort()
min_n = mins[0]
max_n = maxs[0]
if min_n > max_n:
    ans = -1
elif tyo < min_n:
    ans = min_n - tyo
elif tyo > max_n:
    ans = tyo - max_n
else:
    ans = 0
print ans
