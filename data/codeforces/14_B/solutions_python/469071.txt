#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    n, x0 = map(int, raw_input().split())
    p = list([0]*1001)
    for i in xrange(n):
        a, b = map(int, raw_input().split())
        a, b = min(a, b), max(a, b)
        while a <= b:
            p[a] += 1
            a += 1
    a = list()
    for i in xrange(len(p)):
        if p[i] == n:
            a.append(i)
    if len(a) != 0:
        print abs(sorted(a, cmp=lambda x, y: cmp(abs(x-x0), abs(y-x0)))[0] - x0)
    else:
        print -1
    return 0

if __name__ == '__main__':
    main()
