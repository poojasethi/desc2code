import sys

def two():
    n,x = map(int,sys.stdin.readline().strip().split())
    range = [0,1000]
    for each in xrange(0,n):
        l,r = map(int,sys.stdin.readline().strip().split())
        l,r = min(l,r),max(l,r)
        if l > range[0]:
            range[0] = l
        if r < range[1]:
            range[1] = r
    if range[1] < range[0]:
        print -1
    elif range[0] <= x <= range[1]:
        print 0
    else:
        print min(abs(range[0]-x),abs(range[1]-x))
        
two()
