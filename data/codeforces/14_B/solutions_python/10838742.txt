n, x = map(int, raw_input().split())

final_position = -1
left = []
right = []
for _ in xrange(n):
	pair = map(int, raw_input().split())
	pair.sort()
	left.append(pair[0])
	right.append(pair[1])

left = max(left)
right = min(right)

if left <= right:
	if x < left:
		print left - x
	elif x > right:
		print x - right
	else:
		print 0
else:
	print -1
