
n,x0 = map(int,raw_input().split())
ca,cb = 0,1000
for i in range(n):
    t = map(int,raw_input().split())
    a = min(t)
    b = max(t)
    if a>cb:
        print -1
        exit(0)
    if a<=cb and a >= ca:
        xa = a
    else:
        xa = ca

    if b < ca:
        print -1
        exit(0)
    if b<=cb and b>=ca:
        xb = b
    else:
        xb = cb
    ca,cb = xa,xb

if x0>=ca and x0<=cb:
    print 0
elif x0<ca:
    print ca-x0
else:
    print x0-cb
