n,x = [int(i) for i in raw_input().split()]
c = []
for i in xrange(0,2000):
    c.append(0)

for i in xrange(0,n):
    a,b = [int(j) for j in raw_input().split()]
    if a <= b:
        for j in xrange(a,b+1):
            c[j]+=1
    else:   
        for j in xrange(b,a+1):
            c[j]+=1
m = 1001

for i in xrange(0,2000):
    if c[i] == n:
        if m > abs(x-i):
            m = abs(x-i)

if m == 1001:
    print -1
