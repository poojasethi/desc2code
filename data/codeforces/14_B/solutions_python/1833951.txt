#!/usr/bin/python
import sys

(n, x0) = map(lambda x: int(x), sys.stdin.readline().split())
xmin = -1000
xmax = 2000
for i in range(n):
	(a, b) = map(lambda x: int(x), sys.stdin.readline().split())
	if a > b:
		a, b = b, a
	if a > xmin:
		xmin = a
	if b < xmax:
		xmax = b
if xmin > xmax:
	print -1
elif xmin <= x0 and x0 <= xmax:
	print 0
elif x0 > xmax:
	print x0-xmax
else:
	print xmin-x0
