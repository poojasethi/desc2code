n, x0 = map(int, raw_input().strip().split(" "))

track = []
for i in range(0, 1002):
	track.append(0)

for i in range(0, n):
	x, y = sorted(map(int, raw_input().strip().split(" ")))
	for j in range(x, y+1):
		track[j] += 1

if track.count(n) == 0:
	min_dist = -1
else:
	min_dist = 1002
	i = track.index(n)
	while track[i] == n:
		dist = abs(x0 - i)
		if min_dist > dist:
			min_dist = dist
		i += 1
		
print min_dist
