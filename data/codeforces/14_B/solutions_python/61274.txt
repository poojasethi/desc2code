
n, x0 = [int(x) for x in raw_input().strip().split()]
l = 0
r = 1000
for i in xrange(n):
    a, b = [int(x) for x in raw_input().strip().split()]
    if a > b:
        a, b = b, a
    if a > l: l = a
    if b < r: r = b
    if l > r:
        break
if l > r:
    print '-1'
elif l <= x0 <= r:
    print '0'
elif x0 < l:
    print l - x0
else:
    print x0 - r
