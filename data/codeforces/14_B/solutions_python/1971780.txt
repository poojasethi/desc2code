n, x0 = map(int, raw_input().split())
coords = [ sorted(map(int, raw_input().split())) for i in xrange(n) ]
ans = 7777
for i in xrange(1000):
    ok = True
    for (l, r) in coords:
        if i < l or i > r:
            ok = False
            break 
    else:
        ans = min(ans, abs(i - x0))
print ans if ans != 7777 else -1
