#!/usr/bin/env python

import sys

(n, x) = sys.stdin.readline().split(' ')

n = int(n)
x = int(x)

xmin = 0
xmax = 10000

for i in range(0, n):
	(a, b) = sys.stdin.readline().split(' ')
	a = int(a)
	b = int(b)
	if a > b:
		(a, b) = (b, a)
	xmin = max(xmin, a)
	xmax = min(xmax, b)

if xmin > xmax:
	print "-1"
elif xmin <= x and xmax >= x:
	print "0"
else:
	print min(abs(xmin - x), abs(xmax - x))
