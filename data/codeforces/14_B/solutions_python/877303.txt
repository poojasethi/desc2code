n, x0 = map(int, raw_input().split())
ps = [sorted(list(map(int, raw_input().split()))) for i in range(n)]
ans = 10000000
for i in range(1001):
    good = True
    for j in range(n):
        if i < ps[j][0] or ps[j][1] < i:
            good = False
            break
    if not good: continue

    ans = min(ans, abs(i - x0))

if ans > 10000: print(-1)
else: print(ans)
