n,x = map(int, raw_input().split())
rs = [map(int, raw_input().split()) for g in xrange(n)]
b,e = 0, 1000
for i,j in rs:
    if i < j:
        b = max(b, i)
        e = min(e, j)
    if i > j:
        b = max(b, j)
        e = min(e, i)
r = e-b
if r < 0:
    print -1
elif b <= x and x <= e:
    print 0
else:
    print min(abs(e-x), abs(x-b))
