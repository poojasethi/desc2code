#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    n, x0 = map(int, raw_input().split())
    s = set(xrange(1001))
    for i in xrange(n):
        a, b = map(int, raw_input().split())
        s.intersection_update(set(xrange(min(a, b), max(a, b) + 1)))
    if len(s) != 0:
        print abs(sorted(list(s), cmp=lambda x, y: cmp(abs(x-x0), abs(y-x0)))[0] - x0)
    else:
        print -1
    return 0

if __name__ == '__main__':
    main()
