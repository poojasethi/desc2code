n,x0 = map(int,raw_input().split()[:2])
segments = sorted([ sorted((int(a),int(b))) for a,b in [ raw_input().split()[:2] for _ in xrange(n) ] ])

from operator import itemgetter
x = itemgetter(0)
y = itemgetter(1)

begin = max(map(x, segments))
end = min(map(y, segments))

if begin > end:
	print -1
elif x0 < begin:
	print begin - x0
elif x0 > end:
	print x0 - end
else:
	print 0
