import sys


##sys.stdin = open('input.txt', "r")

n, x = map(int, raw_input().split())


l = 0
r = 10000

for i in xrange(n):
    a, b = map(int, raw_input().split())
    if a > b:
        a, b = b, a

    l = max(l, a)
    r = min(r, b)

if l > r:
    print -1
else:
    if x < l:
        print l - x
    elif x > r:
        print x - r
    else:
        print 0
