import math

(n, x) = map(int, raw_input().split())

(resultA, resultB) = (0, 1000)

for i in range(n):
    (a, b) = map(int, raw_input().split())
    
    if a > b:
        (a, b) = (b, a)
    
    (resultA, resultB) = (max(resultA, a), min(resultB, b))

if resultA <= x <= resultB:
    result = 0
elif resultA <= resultB:
    result = min(abs(resultA - x), abs(resultB - x))
else:
    result = -1

print result
