n, x = map(int, raw_input().split())
s = [1] * 1001
for i in range(n):
    a, b = map(int, raw_input().split())
    if(a > b):
        t = a
        a = b
        b = t
    for j in range(1001):
        if(a <= j and j <= b):
            s[j] &= 1
        else:
            s[j] = 0
ans = 2000
for i in range(1001):
    if(s[i] and abs(x - i) < ans):
        ans = abs(x - i)
if(ans == 2000):
    print '-1'
else:
    print ans
