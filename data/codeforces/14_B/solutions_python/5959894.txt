n, x0 = map(int, raw_input().split())
intervals = [map(int, raw_input().split()) for _ in range(n)]
ranges = map(lambda (a, b): range(min(a, b), max(a, b) + 1), intervals) 
intersect = reduce(lambda a, b: set(a).intersection(set(b)), ranges)
print "-1" if len(intersect) == 0 else min(map(lambda x: abs(x - x0), intersect))
