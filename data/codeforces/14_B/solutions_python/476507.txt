#!/usr/bin/env python

n, x0 = map(int, raw_input().split())

a, b = (0, 1000)

for i in range(n):
    at, bt = map(int, raw_input().split())
    at, bt = (min(at, bt), max(at, bt))
    a = max(a, at)
    b = min(b, bt)


if a > b:
    print -1
elif x0 in range(a, b + 1):
    print 0
else:
    print min(abs(x0 - a), abs(x0 - b))
