n,x=map(int,raw_input().split())
L,R=0,1000000
for i in xrange(n):
    a,b=map(int,raw_input().split())
    if a>b:
        a,b=b,a
    L = max(L,a)
    R = min(R,b)

def test(L,R):
    if R<L:
        return -1
    if x>=L and x<=R:
        return 0
    if x<=L:
        return L-x
    return x-R

print test(L,R)
