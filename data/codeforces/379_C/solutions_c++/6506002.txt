#include<iostream>
#include<algorithm>
using namespace std;
int a[300000];
int p[300000];
bool cmp(int i,int j){return a[i]<a[j];}
int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>a[i];
		p[i]=i;
	}
	sort(p,p+n,cmp);
	int k=a[p[0]];
	for(int i=0;i<n;i++){
		if(a[p[i]]>k)k=a[p[i]];
		a[p[i]]=k++;
	}
	for(int i=0;i<n;i++)
		cout<<a[i]<<" ";
	return 0;
}
