#include<cstdio>
#include<map>


using namespace std;

int cur,n, ne;
map<int,int> next;

int findnext(int cur) {
  if (!next.count(cur)) {
    return cur;
  }
  int rtn = findnext(next[cur]);
  next[cur] = rtn;
  return rtn;
}

int main() {
  scanf("%d", &n);
  for(int i=0 ;i<n; ++i) {
    scanf("%d", &cur);
    ne = findnext(cur);
    printf("%d ", ne);
    next[ne] = ne+1;
  }
  printf("\n");
}
