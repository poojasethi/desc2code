#include <iostream>
#include <cstdio>
#include <map>
using namespace std;
map<int,int> next;
int getNext(int k)
{
    if (next[k]==0)
        return k;
    return next[k]=getNext(next[k]);
}
int main()
{
    int n;
    while(~scanf("%d",&n))
    {

    for (int i=0;i<n;i++)
    {
    int a;
       scanf("%d",&a);
    int r=getNext(a);
     cout<<r<<" ";
     next[r]=r+1;
    }
    }
return 0;
}
