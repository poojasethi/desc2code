#include<bits/stdc++.h>
using namespace std;
int n,a[300005],s,t[300005],cur=1;
bool cmp(int i,int j){return a[i]<a[j];}
int main()
{
	int i;
	cin>>n;
	for(i=1; i<=n; i++)
	 scanf("%d",a+i),t[i]=i;
	sort(t+1,t+n+1,cmp);
	for(i=1; i<=n; i++)
	{
		if(a[t[i]]>cur)
			cur=a[t[i]]+1;
		else
			a[t[i]]=cur,cur++;
	}
	for(i=1; i<=n; i++)
		printf("%d ",a[i]);
	return 0;
}
