#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    scanf("%d",&n);
    int ans[n],mn=0;
    pair<int,int> a[n];
    for(int i=0;i<n;i++) scanf("%d",&a[i].first), a[i].second=i;
    sort(a,a+n);
    for(int i=0;i<n;i++){
        mn=max(mn,a[i].first);
        ans[ a[i].second ]=mn;
        mn++;
    }
    for(int i=0;i<n;i++) printf("%d ",ans[i]);
    return 0;
}
