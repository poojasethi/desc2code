#include <iostream>
#include <map>

using namespace std;

map<int, int> next;
int getNext(int k)
{
	if (next.find(k) == next.end())
		return k;
	return (next[k] = getNext(next[k]));
}
int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		int a;
		cin >> a;
		int v = getNext(a);
		cout << v << " ";
		next[v] = v + 1;
	}
}
