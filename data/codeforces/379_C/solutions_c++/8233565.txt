#include <iostream>
#include <algorithm>

#define rep(i,n) for(int i=0;i<n;i++)
#define pi pair<int,int>

using namespace std;

int n,b[300010],r;
pi a[300010];


int main(){
  cin>>n;
  rep(i,n){
    cin>>a[i].first;
    a[i].second = i;
  }
  sort(a,a+n);
  rep(i,n){
    if(a[i].first>r)r=a[i].first;
    b[a[i].second] = r;
    r++;
  }
  rep(i,n)cout<<b[i]<<" ";cout<<endl;

}
