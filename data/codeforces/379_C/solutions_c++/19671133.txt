#include <iostream>
#include <algorithm>
using namespace std;
pair <int, int> a[300006];
int i,n,r,b[300006];
int main(){
    for (cin>>n;i<n;i++) cin>>a[i].first,a[i].second=i;
    sort(a, a+n);
    for (i=0;i<n;i++) b[a[i].second]=r=max(r+1, a[i].first);
    for (i=0;i<n;cout<<b[i++]<<" ");
    cout<<endl;
}
