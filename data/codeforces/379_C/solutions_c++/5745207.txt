#include<iostream>
#include<map>
using namespace std;
map<int ,int > ma;
int next(int x)
{
    if(ma.find(x)==ma.end())return x;
    return ma[x]=next(ma[x]);
}
int main()
{
     int n,a;
    int  i,j;

    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a;
        a=next(a);
        ma[a]=a+1;
        if(i)cout<<" ";
        cout<<a;
    }
    return 0;
}
