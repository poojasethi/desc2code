#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
int n,i,j;
cin>>n;
int a[n];
vector< pair<int,int> > v(n);
for(i=0;i<n;i++)
{
  cin>>(v[i].first);
  v[i].second=i;
}
sort(v.begin(),v.end());
j=0;
for(i=0;i<n;i++)
{
 j++;
 if(j<v[i].first) j=v[i].first;
 a[v[i].second]=j;
}
for(i=0;i<n;i++) cout<<a[i]<<" ";
return 0;
}
