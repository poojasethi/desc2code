#include <bits/stdc++.h>

typedef long long ll;

ll n , b[300010] , x = 0;
std::pair <ll,ll> a[300010];

int main(){
	std::ios::sync_with_stdio(false);
	std::cin >> n;
	for (int i = 0 ; i < n ; ++ i) std::cin >> a[i].first , a[i].second = i;
	sort(a , a+n);
	for (int i = 0 ; i < n ; ++ i){
		b[a[i].second] = x = std::max(x+1 , a[i].first);
	}
	for (int i = 0 ; i < n ; ++ i){
		std::cout << b[i] << " ";
	}
}
