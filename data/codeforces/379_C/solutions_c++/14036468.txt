#include <iostream>
#include <algorithm>
using namespace std;

long long  n,m=0,b[300100];
pair<long long int,int>a[300100];

int main(){
	
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i].first;
		a[i].second=i;
	}
	sort(a,a+n);
	for(int i=0;i<n;i++)b[a[i].second]=m=max(m+1,a[i].first);
	for(int i=0;i<n;i++)cout<<b[i]<<" ";
	cout<<endl;

	return 0;
}
