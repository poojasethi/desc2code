#include<iostream>
#include<algorithm>
using namespace std;
#define st first
#define nd second
main(){
int n;
cin>>n;
pair<int,int> tab[n];
int w[n];
for(int i=0;i<n;i++){cin>>tab[i].st;tab[i].nd=i;}
sort(tab,tab+n);
w[tab[0].nd]=tab[0].st;
for(int i=1;i<n;i++){
w[tab[i].nd]=max(tab[i].st,w[tab[i-1].nd]+1);
}
for(int i=0;i<n;i++)cout<<w[i]<<" ";
cout<<"\n";
}
