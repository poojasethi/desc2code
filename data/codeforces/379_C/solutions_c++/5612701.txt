#include <bits/stdc++.h>

using namespace std;
#define F first
#define S second
int n,i,ans[300005];
pair <int,int> a[300005];
int main()
{
    cin>>n;
    for(i=1;i<=n;i++){cin>>a[i].F;a[i].S=i;}
    sort(a+1,1+a+n);
    for(i=1;i<=n;i++){if (a[i].F<=a[i-1].F) a[i].F=a[i-1].F+1;}
    for(i=1;i<=n;i++)ans[a[i].S]=a[i].F;
    for(i=1;i<=n;i++)cout<<ans[i]<<' ';
    return 0;
}
