#include<bits/stdc++.h>
using namespace std;
map<int,int> m;
int next(int k)
{
	if(m.find(k)==m.end())return k;
  	else return m[k]=next(m[k]);
}
int main()
{
	int n,a,i,j;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a);
		j=next(a);
		printf("%d ",j);
		m[j]=j+1;
	}
	return 0;
}
