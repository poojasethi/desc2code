#include <iostream>
#include <algorithm>
#define ll long long
#define pll pair <ll,ll> 
using namespace std;

int n,v[300001],d;
pll a[300001];


int main(){
  cin>>n;
  for (int i=0;i<n;i++){
    cin>>a[i].first;
    a[i].second = i;
  }
  sort(a,a+n);
  for (int i=0;i<n;i++){
  	if(a[i].first>d)d=a[i].first;
    v[a[i].second] = d;
    d++;
  }
  for (int i=0;i<n;i++)
  	cout<<v[i]<<" ";

}
