#include<bits/stdc++.h>

using namespace std;

#define st first
#define nd second

int n,i;

pair<int,int> B[300005];
int C[300005];

main(){

	cin>>n;

	for(i=1 ; i<=n ; i++){
		cin>>B[i].st;
		B[i].nd=i;
	}
	
	sort(B+1,B+n+1);

	C[B[1].nd]=B[1].st;

	for(i=2 ; i<=n ; i++)
		C[B[i].nd]=max(B[i].st,C[B[i-1].nd]+1);

	for(i=1 ; i<=n ; i++)
		cout<<C[i]<<' ';
	
	return 0;
}
