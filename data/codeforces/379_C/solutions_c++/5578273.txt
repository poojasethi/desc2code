#include <cstdio>
#include <algorithm>
#include <utility>

#define N 300000

int b[N], n, i, c;
std::pair <int, int> a[N];

int main()
{
	scanf("%d", &n);
	for (i = 0; i < n; i ++)
	{
		scanf("%d", &a[i].first);
		a[i].second = i;
	}

	std::sort(a, a + n);

	for (i = 0; i < n; i ++)
	{
		++ c;
		if (c < a[i].first)
			c = a[i].first;
		b[a[i].second] = c;
	}

	for (i = 0; i < n; i ++)
		printf("%d ", b[i]);

	return 0;
}
