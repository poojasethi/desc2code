#include <iostream>
#include <algorithm>
#include <map>

using namespace std;

map <int, int > m;

int next(int k)
{
	if (m.find(k) == m.end())
		return k;
  	else
		return (m[k] = next(m[k]));
}

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		int a;
		cin >> a;
		int b = next(a);
		cout << b << " ";
		m[b] = b + 1;
	}
	cout << endl;
}
