#include<stdio.h>
#include<algorithm>
#include<map>
using namespace std;
map<int,int> m;
int next(int k)
{
	if(m.find(k)==m.end())
		return k;
  	else
	{
	return (m[k]=next(m[k]));
	}
}
int main()
{
	int n;
	scanf("%d",&n);
	int i;
	for(i=0;i<n;i++)
	{
		int a;
		scanf("%d",&a);
		int v=next(a);
		printf("%d ",v);
		m[v]=v+1;
	}
	printf("\n");
	return 0;
}
