#include<iostream>
#include<algorithm>
using namespace std;
struct p
{
	int i;
	int h;
}a[300401];
bool cmp(p b,p c)
{
	return b.h<c.h;
}
int main()
{
	int i,o[300401],n,m=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i].h;
		a[i].i=i;
	}
	sort(a,a+n,cmp);
	for(i=0;i<n;i++)
	{
		o[a[i].i]=max(m+1,a[i].h);
		m=o[a[i].i];
	}
	for(i=0;i<n;i++)
		cout<<o[i]<<" ";
	cout<<"\n";
}
