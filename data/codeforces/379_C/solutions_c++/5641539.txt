#include <iostream>
#include <algorithm>
using namespace std;
struct t
{int r,i;
} a[300001];
bool cmp1(t a1,t a2)
{return a1.r<a2.r;}
bool cmp2(t a1,t a2)
{return a1.i<a2.i;}
int main()
{int n; cin>>n;
for (int i=1;i<=n;i++) {cin>>a[i].r; a[i].i=i;}
sort(a+1,a+n+1,cmp1);
for (int i=1;i<=n;i++) if (a[i].r<=a[i-1].r) a[i].r=a[i-1].r+1;
sort(a+1,a+n+1,cmp2);
for (int i=1;i<=n;i++) cout<<a[i].r<<" ";
return 0;
}
