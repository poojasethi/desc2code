#include<cstdio>
#include<algorithm>
using namespace std;

int n;
pair<int,int>a[333333];
int b[333333];
int cur;

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++){
		scanf("%d", &a[i].first);
		a[i].second=i;
	}
	sort(a,a+n);
	cur=0;
	for(int i=0;i<n;i++){
		cur=max(cur+1,a[i].first);
		b[a[i].second]=cur;
	}
	for(int i=0;i<n;i++)
		printf("%d%c", b[i],(i<n-1)?' ':'\n');

	return 0;
}
