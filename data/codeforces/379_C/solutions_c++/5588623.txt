#include <cstdio>
#include <algorithm>
using namespace std;

int n;
pair<int,int> a[300001];
int ans[300001];
int main () {
    scanf("%d",&n);
    for (int i=1;i<=n;++i) scanf("%d",&a[i].first),a[i].second=i;
    sort(a+1,a+n+1);
    int mx=0;
    for (int i=1;i<=n;++i) ans[a[i].second] = mx = max(a[i].first,mx+1);
    for (int i=1;i<=n;++i) printf("%d ",ans[i]);
    printf("\n");
    return 0;
}
