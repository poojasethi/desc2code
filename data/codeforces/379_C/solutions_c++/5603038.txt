#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i;
	scanf("%d",&n);
	pair<int,int> arr[n];
	int res[n];
	for (i=0;i<n;i++)
	{
		scanf("%d",&arr[i].first);
		arr[i].second=i;
	}
	sort(arr,arr+n);
	for (i=1;i<n;i++)
		if (arr[i].first<=arr[i-1].first)
			arr[i].first+=arr[i-1].first-arr[i].first+1;
	for (i=0;i<n;i++)
		res[ arr[i].second ]=arr[i].first;
	for (i=0;i<n;i++)
		printf("%d ",res[i]);
	return 0;
}
