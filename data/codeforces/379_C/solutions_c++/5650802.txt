#include<iostream>
#include<cstdio>
#include<cstring>
#include<map>
using namespace std;
map<int,int>mp;

int next(int k)
{
    if(mp[k]==0)
        return k;
    return mp[k]=next(mp[k]);
}
int main()
{
    int n,i,j,a,v;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a);
        v=next(a);
        printf("%d ",v);
        mp[v]=v+1;
    }
    return 0;
}
