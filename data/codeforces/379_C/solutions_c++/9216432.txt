#include <iostream>
#include <algorithm>


using namespace std;

const int MAXN = 1e5*3;
pair<int, int> a[MAXN];

int main()
{
	int n;
	cin>>n;
	int b[n];
	for(int i = 0; i < n;i++)
	{
		cin>>a[i].first;
		a[i].second = i;
	}
	sort(a, a+n);
	int x = a[0].first;
	for(int i = 0; i < n;i++)
	{
		if(a[i].first > x)
		{
			x = a[i].first;
		}
		b[a[i].second] = x;
		x++;
	}
	for(int i = 0; i < n;i++)
		cout<<b[i]<<' ';
	return 0;
}
