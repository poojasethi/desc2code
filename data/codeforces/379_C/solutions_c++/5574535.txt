#include <cstdio>
#include <algorithm>
using namespace std;
const int N=310000;
int n,i,a[N],f[N];
bool cmp(const int& x,const int& y) {return a[x]<a[y];}
int main() {
	scanf("%d",&n);
	for (i=0;i<n;i++) scanf("%d",a+i);
	for (i=0;i<n;i++) f[i]=i;
	sort(f,f+n,cmp);
	for (i=1;i<n;i++)
		if (a[f[i]]<=a[f[i-1]]) a[f[i]]=a[f[i-1]]+1;
	for (i=0;i+1<n;i++)
		printf("%d ",a[i]);
	printf("%d\n",a[n-1]);
	return 0;
}
