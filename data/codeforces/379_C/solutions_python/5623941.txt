import sys

n, a = input(), list(map(int, raw_input().split()))
#print a
t = sorted((j, i) for i, j in enumerate(a))
#print t
s, p = 0, [0] * n
for j, i in t:
    if s < j: s = j
    else: s += 1
    p[i] = s
print(' '.join(map(str, p)))
