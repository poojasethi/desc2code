n = input()
scores = zip(range(n),  map(int, raw_input().split()))

scores.sort(key = lambda item: item[1])
# print scores

for i in range(1, len(scores)) : 
	if scores[i][1] <= scores[i-1][1] : 
		scores[i] = (scores[i][0], scores[i-1][1] + 1)

result = [0] * n
for (k, v) in scores : 
	result[k] = v

print " ".join(map(str, result))
