#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
a = ia()

a = list(enumerate(a))
a.sort(key=lambda e : e[1])

b = [None for e in a]

ma = 0
for i, e in a:
    e = max(e, ma+1)
    b[i] = e
    ma = e

print ' '.join(map(str, b))
