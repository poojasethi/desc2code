import sys
n = map(int,raw_input().split())[0]
l_a = map(int,raw_input().split())

l_t = sorted(zip(l_a, range(n)))

i_max = 0
d_ret = {}

for i_want, i_ind in l_t:
    i_give = max(i_want, i_max + 1)
    d_ret[i_ind] = i_give
    i_max = i_give

li_ret = [str(d_ret[x]) for x in range(n)]

print ' '.join(li_ret)
