from sys import stdin, stdout
def main():
    n = int(stdin.readline())
    a = [(c, i) for i, c in enumerate(map(int, stdin.readline().split()))]
    a.sort()
    b = [0] * n
    t = -1
    for c, i in a:
        if c > t:
            t = c
        else:
            t += 1
        b[i] = t
    stdout.write(' '.join(map(str, b)))
main()
