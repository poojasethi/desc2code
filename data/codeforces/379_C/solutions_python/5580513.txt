import sys


def ratings_change(demand):
    received = [0] * len(demand)
    min_rating = 0
    for d in demand:
        curr = max(d[0], min_rating)
        if curr >= min_rating:
            min_rating = curr + 1
        received[d[1]] = curr
    print ' '.join(map(str, received))





if __name__ == '__main__':
    _ = sys.stdin.readline()
    line = sys.stdin.readline()
    ratings = map(int, line.split())
    users = range(len(ratings))
    demand = zip(ratings, users)
    demand = sorted(demand)
    ratings_change(demand)
