# -*- coding: utf-8 -*-

import sys

n = int(raw_input())
a = map(int, raw_input().split())
new_a = []
b = [0] * n
for i in xrange(len(a)):
    new_a.append([a[i],i])
sorted_new_a = sorted(new_a, key=lambda x:x[0])
now = 1
for l in sorted_new_a:
    now = max(l[0], now)
    b[l[1]] = now
    now += 1

print ' '.join(map(str, b))
