from random import randint

c = 0
d = 0
n = input()

for i in range (0,n):
    f1, f2 = map(int,raw_input().split())

    if (f1 > f2):
        c+=1
    elif (f1 < f2):
        d+=1
    else:
        pass

if(c > d):
    print "Mishka"

elif(c < d):
    print "Chris"

else:
    print "Friendship is magic!^^"
