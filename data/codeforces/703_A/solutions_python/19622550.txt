#!/bin/python

def main():

	n = int(raw_input().strip())

	m = 0
	c = 0 

	for i in range(n):
		a, b = map(int, raw_input().strip().split())
		if a > b:
			m +=1
		elif a < b:
			c += 1


	if m == c:
		print 'Friendship is magic!^^'
	elif m > c:
		print 'Mishka'
	else:
		print 'Chris'


if __name__ == '__main__':
	main()
