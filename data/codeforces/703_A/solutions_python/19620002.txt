


def run():
	n = int(raw_input().strip())
	Mishka = Chris = 0
	for i in xrange(n):
		m, c = map(int, raw_input().strip().split())
		if m > c: Mishka += 1
		if m < c: Chris += 1

	if Mishka > Chris:
		print 'Mishka'
	elif Chris > Mishka:
		print 'Chris'
	else:
		print 'Friendship is magic!^^'

run()
