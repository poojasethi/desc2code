#include<bits/stdc++.h>

using namespace std;

typedef long long LL;

int main()
{
	int n,grundy=0;
	cin >> n;
	int i;
	LL num;
	for(i=0;i<n;i++)
	{
		cin >> num;
		if(num <= 3)
			grundy^=0;
		else if(num <= 15)
			grundy^=1;
		else if(num <= 81)
			grundy^=2;
		else if(num <= 6723)
			grundy^=0;
		else if(num <= 50625)
			grundy^=3;
		else if(num <= 2562991875)
			grundy^=1;
		else
			grundy^=2;
	}
	cout << ((grundy > 0)?"Furlo":"Rublo") << endl;
	return 0;
}
