#include <iostream>
#include <map>
using namespace std;

typedef long long ll;

ll n, x, s = 0;
map<ll,int> m;

int main(){

    m[3] = 0, m[15] = 1, m[81] = 2, m[6723] = 0, m[50625] = 3, m[2562890625LL] = 1, m[(1LL<<62)] = 2;
    cin >> n;
    for(int i = 0; i<n; i++){
        cin >> x;
        s^=m.lower_bound(x)->second;
    }
    cout << (s?"Furlo":"Rublo");

}
