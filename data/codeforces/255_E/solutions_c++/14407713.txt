#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> ii;
typedef long long ll;
int n;

int f(ll a)
{
	if (a<4) return 0;
	else if (a<16) return 1;
	else if (a<82) return 2;
	else if (a<6724) return 0;
	else if (a<50626) return 3;
	else if (a<2562991876LL) return 1;
	else return 2;
}

int main()
{
	scanf("%d", &n);
	int ans=0;
	while(n--)
	{
		ll t;
		cin>>t;
		ans^=f(t);
		//cout<<ans<<endl;
	}
	if (ans==0) cout<<"Rublo\n";
	else cout<<"Furlo\n";
}
