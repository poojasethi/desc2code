#include <iostream>
using namespace std;
int n;
int sg(long long x){
    if(x<4) return 0;
    if(x<16) return 1;
    if(x<82) return 2;
    if(x<256) return 0;
    if(x<6724) return 0;
    if(x<50626) return 3;
    if(x<43046721) return 1;
    if(x<2562991876LL) return 1;
    return 2;
}
int main(){
    int r=0;
    cin>>n;
    for(int i=0;i<n;i++){
        long long t;
        cin>>t;
        r^=sg(t);
    }
    if(!r)cout<<"Rublo";else cout<<"Furlo";
    cout<<endl;
}
