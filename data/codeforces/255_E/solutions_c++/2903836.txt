#include <cstdio>
__int64 num[] = {4, 16, 82, 6724, 50626, 50626 * 50626ll, 777777777778LL};
int sg[] = {0,1,2,0,3,1,2}, n, ans = 0;
int main() {
    scanf("%d", &n);
    while(n--) {
        __int64 temp;
        scanf("%I64d", &temp);
        int i = 0;
        while(temp >= num[i])
            i++;
        ans ^= sg[i];
    }
    printf("%s\n", ans ? "Furlo" : "Rublo" ); 
}
