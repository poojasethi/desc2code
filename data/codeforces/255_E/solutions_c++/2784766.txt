#include <stdio.h>
long long sg(long long n){
    if (n <= 3) return 0;
    if (n <= 15) return 1;
    if (n <= 81) return 2;
    if (n <= 6723) return 0;
    if (n <= 50625) return 3;
    if (n <= 2562991871ll) return 1;
    return 2;
}
int main(){
    int n, t = 0;
    long long x;
    scanf("%d", &n);
    while (n--){
        scanf("%I64d", &x);
        t ^= sg(x);
    }
    if (t) printf("Furlo\n");
    else printf("Rublo\n");
    return 0;
}
