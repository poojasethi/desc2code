#include <iostream>
using namespace std;

typedef long long ll;

int SG(ll k) {
	if (k < 4) return 0;
	if (k < 16) return 1;
	if (k < 82) return 2;
	if (k < 6724) return 0;
	if (k < 50626) return 3;
	if (k < (ll) 2562991876)return 1;
	return 2;
}

int main() {
	int n, res = 0;
	cin.sync_with_stdio(false);
	cin >> n;
	while (n--) {
		ll t;
		cin >> t;
		res ^= SG(t);
	}
	cout << (res? "Furlo\n": "Rublo\n");
	return 0;
}
