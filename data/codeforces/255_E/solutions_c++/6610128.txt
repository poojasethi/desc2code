#include <cstdio>
#include <cstring>

using namespace std;

typedef long long LL;

int SG(LL a){
	if(a<4) return 0;
	if(a<16) return 1;
	if(a<82) return 2;
	if(a<6724) return 0;
	if(a<50625) return 3;
	if(a<2562890625LL) return 1;
	return 2;
}

int main()
{
	int n,s=0;LL t;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%I64d",&t);
		s^=SG(t);
	}
	puts(s ? "Furlo" : "Rublo");
	return 0;
}
