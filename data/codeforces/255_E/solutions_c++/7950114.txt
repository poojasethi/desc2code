#include<iostream>
using namespace std;
int func(long long x)
{
    if((x>=1&&x<=3)||(x>=82&&x<=6723)) return 0;
    else if((x>=4&&x<=15)||(x>=50626&&x<=2562991875)) return 1;
    else if(x>=6724&&x<=50625) return 3;
    else return 2;
}
main()
{
    int i,j,k,n,xr;
    cin>>n;
    long long x;
    xr=0;
    for(i=0;i<n;i++)
    {
        cin>>x;
        xr^=func(x);
    }

}
