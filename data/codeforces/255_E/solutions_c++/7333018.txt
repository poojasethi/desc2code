#include<iostream>
#include<map>
using namespace std;
int n;
int f(long long s){
	if(s < 4) return 0;
	if(s < 16) return 1;
	if(s < 82) return 2;
	if(s < 6724) return 0;
	if(s < 50626) return 3;
	if(s < 2562991876 ) return 1;
	return 2;
}
int main(){
	while(cin >> n){
		long long tmp;
		long long ans = 0;
		for(int i = 0; i < n; ++i){
			cin >> tmp;
			ans ^= f(tmp);
		}
		if(ans) cout << "Furlo" << endl;
		else cout << "Rublo" << endl;
	}
	return 0;
}
