#include<cstdio>
#include<iostream>
using namespace std;
int sg(long long x)
{
	if (x<4) return 0;
	if (x<16) return 1;
	if (x<82) return 2;
	if (x<6724) return 0;
	if (x<50626) return 3;
	if (x<2562991876LL) return 1;
	return 2;
}
int main()
{
	int n,ans=0; long long x; scanf("%d",&n);
	for (int i=1;i<=n;i++) cin>>x,ans^=sg(x);
	if (ans) puts("Furlo"); else puts("Rublo");
	return 0;
}
