#include<cstdio>
using namespace std;
typedef long long ll;
ll a[]={4,16,82,6724,50626,2562991876ll,777777777778ll};
int b[]={0,1,2,0,3,1,2};
int main()
{
	int n;
	ll m;
	while(~scanf("%d",&n)){
		int i,ans=0;
		while(n--){
			scanf("%I64d",&m);
			for(i=0;a[i]<=m;i++);
			ans^=b[i];
		}
		puts(ans?"Furlo":"Rublo");
	}
	return 0;
}
