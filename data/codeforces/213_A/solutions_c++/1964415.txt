#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char *argv[])
{
	int n, t;
	cin >> n;
	vector<vector<int> > e(n);
	vector<int> p(n), c(n);
	for (int i = 0; i < n; i++) { cin >> c[i]; c[i]--; }
	for (int i = 0; i < n; i++) {
		cin >> p[i];
		for (int j = 0; j < p[i]; j++) {
			cin >> t;
			e[t - 1].push_back(i);
		}
	}

	int ans = 1000000000;
	for (int k = 0; k < 3; k++) {
		int count = 0, r = n;
		vector<int> d(n), a = p;
		for (int m = k; r > 0; ) {
			int t = 0;
			for (int i = 0; i < n; i++)
				if (d[i] == 0 && a[i] == 0 && c[i] == m) {
					d[i] = 1;
					t++;
					r--;
					count++;
					for (int j = 0; j < e[i].size(); j++) a[e[i][j]]--;
				}
			if (r == 0) break;
			if (t == 0) {
				count++;
				m = (m + 1) % 3;
			}
		}
		ans = min(ans, count);
	}

	cout << ans << endl;

	return 0;
}
