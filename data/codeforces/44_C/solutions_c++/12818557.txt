#include<bits/stdc++.h>
using namespace std;
int n,m,s[105];
int main(){
  scanf("%d %d",&n,&m);
  memset(s,0,sizeof(s));
  for(int i=1;i<=m;i++){
    int a,b;
    scanf("%d %d",&a,&b);
    for(int j=a;j<=b;j++) s[j]++;
  }
  for(int i=1;i<=n;i++){
    if(s[i]!=1){
      printf("%d %d\n",i,s[i]);
      return 0;
    }
  }
  printf("OK\n");
  return 0;
}
