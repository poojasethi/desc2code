#include <iostream>

using namespace std;

int f[110];

int n, m;

void INPUT(){
	int a, b;
	cin >> n >> m;
	for (int i = 0; i < m; i++){
		cin >> a >> b;
		for (int j = a; j <= b; j++)
			f[j]++;
	}
}

int main(){
	INPUT();

	for (int i = 1; i <= n; i++)
		if (f[i] != 1){
			cout << i << " " << f[i];
			return 0;
		}

	cout << "OK";

	return 0;
}
