#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int s[105];

int main(void) {
	int n, m;
	scanf("%d%d", &n, &m);
	for(int i=0;i<m;i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		for(int j=x;j<=y;j++)
			s[j]++;
	}
	for(int i=1;i<=n;i++)
		if(s[i]!=1) {
			printf("%d %d\n", i, s[i]);
			return 0;
		}
	puts("OK");
	return 0;
}
