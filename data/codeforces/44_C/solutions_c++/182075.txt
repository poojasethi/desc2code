#include <cstdio>

int main()
{
	int n, m, ai, bi;
	int cnt[110] = {0};
	scanf("%i%i", &n, &m);
	for(int i = 0; i < m; i++) {
		scanf("%i%i", &ai, &bi);
		for(int j = ai; j <= bi; j++) {
			cnt[j]++;
		}
	}
	bool found = false;
	for(int i = 1; i <= n; i++) {
		if(cnt[i] != 1) {
			printf("%i %i\n", i, cnt[i]);
			found = true;
			break;
		}
	}
	if(!found) {
		printf("OK\n");
	}
}
