/* 2013-08-11 09:15:28.631850 */#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,x,y;
int c[200];
int main(){
  scanf("%d%d",&n,&m);
  for (int i=1;i<=m;i++){
    scanf("%d%d",&x,&y);
	for (int j=x;j<=y;j++) ++c[j];
  }
  int a=200;
  for (int i=1;i<=n && a==200;i++)
    if (c[i]!=1) a=i;
  if (a==200) printf("OK\n"); else printf("%d %d\n",a,c[a]);
  return 0;
}
