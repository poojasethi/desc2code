#include<iostream>
using namespace std;

int flag[200];
int main()
{
	int m,n;
	cin>>n>>m;
	while(m--)
	{
		int l,r;
		cin>>l>>r;
		for (int t=l; t<=r; t++)
			flag[t]++;
	}
	bool f=true;
	for (int i=1; i<=n; i++)
	{
		if (flag[i]!=1)
		{
			f = false;
			cout<<i<<' '<<flag[i]<<endl;
			break;
		}
	}
	if (f)
		cout<<"OK\n";

}
