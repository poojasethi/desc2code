#include <iostream>

using namespace std;

int main(){
  int a[111], n, m, x, y, i, j;
  cin >> n >> m;
  for(i = 0; i < 111; i++){
    a[i] = 0;
  }
  for(i = 0; i < m; i++){
    cin >> x >> y;
    for(j = x; j <= y; j++){
      a[j]++;
    }
  }
  for(i = 1; i <= n; i++){
    if(a[i] != 1){
      cout << i << ' ' << a[i] << endl;
      return 0;
    }
  }
  cout << "OK" << endl;
}
