#include <cstdio>

int main()
{
	int n = 0,
	    m = 0;
	scanf("%d%d", &n, &m);

	int d[128] = {};
	for (int i = 0; i < m; ++i)
	{
		int l = 0, r = 0;
		scanf("%d%d", &l, &r);
		for (int j = l; j <= r; ++j)
			++d[j];
	}

	for (int i = 1; i <= n; ++i)
	{
		if (d[i] != 1)
		{
			printf("%d %d\n", i, d[i]);
			return 0;
		}
	}

	printf("OK\n");
	return 0;
}
