#include<iostream>
using namespace std;

int day[111];

int main()
{
	int n,m;cin>>n>>m;
	for(int i=0;i!=m;i++)
	{
		int l,r;cin>>l>>r;
		for(int j=l;j<=r;j++)
			day[j]++;
	}
	for(int k=1;k<=n;k++)
	{
		if(day[k]!=1)
		{
			cout<<k<<" "<<day[k]<<endl;
			break;
		}
		if(k==n)
			cout<<"OK"<<endl;
	}
}
