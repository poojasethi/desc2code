#include <stdio.h>

int ans[200];
int main()
{
	int n, m, a, b;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < m; i++)
	{
		scanf("%d%d", &a, &b);
		for (int j = a; j <= b; j++)
		{
			ans[j] ++;
		}
	}
	for (int i = 1; i <= n; i++)
	{
		if (ans[i] != 1)
		{
			printf ("%d %d\n", i, ans[i]);
			return 0;
		}
	}
	printf ("OK");
	return 0;
}
