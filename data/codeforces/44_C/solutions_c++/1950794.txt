#include <stdio.h>

int N, M;
int i, j, k;
int mas[110][2];
int res[110];

int main() {
  scanf("%d %d", &N, &M);

  for (i = 1; i <= M; i++) {
    scanf("%d %d", &mas[i][0], &mas[i][1]);
    for (j = mas[i][0]; j <= mas[i][1]; j++) {
      res[j]++;
    }
  }
  for (j = 1; j <= N; j++) {
    if (res[j] != 1) {printf("%d %d\n", j, res[j]); return 0;}
  }
  printf("OK\n");
}
