#include<stdio.h>
int main()
{
	int min,ok,m,n,a,b,i,minn,j;
	int num[200]={0};
	min=3000000;
	ok=1;
	scanf("%d%d",&m,&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&a,&b);
		for(j=a;j<=b;j++)
			num[j]++;
	}
	for(i=1;i<=m;i++)
	{
		if(num[i]!=1)
		{
			min=num[i];
			minn=i;
			ok=0;
			break;
		}
	}
	if(ok) printf("OK");
	else {
	printf("%d %d",minn,min);
	}
	return 0;
}
