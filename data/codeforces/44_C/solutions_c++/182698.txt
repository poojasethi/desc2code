#include <cstdio>
#include <vector>
using namespace std;

int main() {
	int n,m,a,b,i,j;
	scanf("%d%d", &n, &m);
	vector<int> d(n+1);
	for(i=0; i<m; i++) {
		scanf("%d%d", &a, &b);
		for(j=a; j<=b; j++)
			d[j]++;
	}

	for(i=1; i<=n; i++) {
		if(d[i] == 0 || d[i] > 1) {
			printf("%d %d\n", i, d[i]);
			return 0;
		}
	}
	puts("OK");
	return 0;
}
