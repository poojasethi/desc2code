#include<stdio.h>
int i,j,k,l,m,n,a[100000];
main()
{

  scanf("%d%d",&n,&m);
  for(i=1;i<=m;i++)
  {
    scanf("%d%d",&j,&k);
    for(l=j;l<=k;l++)
    a[l]++;
  }
  for(i=1;i<=n;i++)
    if(a[i]!=1)
    break;
  if(i>n)
  printf("OK\n");
  else
  printf("%d %d\n",i,a[i]);
  return 0;
}
