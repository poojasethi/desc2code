#include <vector>
#include <iostream>

using namespace std;

int main()
{

	int n, m;
	cin >> n >> m;
	vector<int> v(n+1, 0);
	for (int i = 0 ; i < m ; i ++)
	{
		int a,b;
		cin >> a >> b;
		for (int j = a ; j <= b ; j ++)
		{
			v[j] ++;
		}
	}

	for (int i = 1 ; i < v.size() ; i ++)
	{
		if (v[i] != 1)
		{
			cout << i << " " << v[i] << endl;
			return 0;
		}
	}

	cout << "OK" << endl;
	return 0;
}
