#include<stdio.h>
int main()
{
int m,n,a[101],b[101],i,count,j;
scanf("%d %d",&n,&m);
for(i=1;i<=m;i++)
	{
	scanf("%d %d",&a[i],&b[i]);
	}
for(i=1;i<=n;i++)
	{
	count=0;
	for(j=1;j<=m;j++)
		{
		if(i>=a[j]&&i<=b[j])
			{
			count++;
			}
		}
	if(count==0||count>1)
		{
		goto end;
		}
    }
printf("OK");
goto end2;
end:
printf("%d %d",i,count);
end2:
return(0);
}
