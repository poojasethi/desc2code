#include <iostream>
#include <memory.h>
using namespace std;

int main() {
	int n, m, u[105];
	cin >> n >> m;
	memset(u, 0, sizeof(u));
	for (int i=0;i<m;++i)
	{
		int a,b;
		cin >> a >> b;
		for (int j=a;j<=b;++j)
			++u[j];
	}

	for (int i=1;i<=n;++i)
		if (u[i] != 1)
		{
			cout << i << " " << u[i] << endl;
			return 0;
		}
	cout << "OK";
	return 0;
}
