#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
int a[2000];
int main(){
	int n,m;
	int s,t;
	while(scanf("%d %d",&n,&m)!=EOF){
		for(int i=0;i<m;i++){
			scanf("%d %d",&s,&t);
			for(int i=s;i<=t;i++) a[i]++;
		}
		for(int i=1;i<=n;i++) if(a[i]!=1){
			printf("%d %d",i,a[i]);
			return 0;
		}
		puts("OK");
		
	}
}
