#include<cstdio>
int n, m, i, a, b, c[102];
int main(){
	while(~scanf("%d %d", &n, &m)){
		for(i=1; i<=n; i++)	c[i]=0;
		for(i=0; i<m; i++){
			scanf("%d %d", &a, &b);
			c[a]++;
			c[b+1]--;
		}
		a=0;
		for(i=1; i<=n; i++){
			a+=c[i];
			if(a!=1){
				printf("%d %d\n", i, a);
				break;
			}
		}
		if(i>n)	puts("OK");
	}
	return 0;
}
