#include <cstdio>
#include <cstring>
using namespace std;

int A[200], N, M, x, y;

int main(void){
	scanf("%d%d", &N, &M);
	memset(A, 0, sizeof(A));
	for (int i = 0; i < M; i++){
		scanf("%d %d", &x, &y);
		for (int j = x; j <= y; j++) A[j]++;
	}
	for (int i = 1; i <= N; i++)
	   if (A[i] != 1) {
	   	  printf("%d %d", i, A[i]);
	   	  return 0;
	   }
	printf("OK");
	return 0;
}
