#include<iostream>
#include<vector>

using namespace std;

int main(){
    int n,m;
    cin>>n>>m;
    int A[101];
    fill(A+1,A+n+1,0);
    for(int i=0;i<m;i++){
        int a,b;
        cin>>a>>b;
        for(int i=a;i<=b;i++)A[i]++;
    }
    for(int i=1;i<=n;i++)if(A[i]!=1){
        cout<<i<<" "<<A[i]<<endl;
        goto aa;    
    }
    cout<<"OK"<<endl;
    aa:;
}
