#include <iostream>
using namespace std;

int n,m;
int l[101],r[101];

int main() {
    cin>>m>>n;
    for(int i=0;i<n;i++) cin>>l[i]>>r[i];
    for(int i=1;i<=m;i++) {
        int d=0;
        for(int j=0;j<n;j++) if (l[j]<=i && i<=r[j]) ++d;
        if (d!=1) {
            cout<<i<<" "<<d;
            return 0;
        }
    }
    cout<<"OK";
    return 0;
}
