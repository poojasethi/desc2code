#include <iostream>
#include<stdio.h>
using namespace std;
int mas[101];
int main()
{
    int n,a,b,m;
    cin>>n>>m;
    for(int i=0;i<m;i++)
    {
        cin>>a>>b;
        for(int j=a;j<=b;j++)
        mas[j]++;
    }
    for(int i=1;i<=n;i++)
        if(mas[i]!=1)
        {
            cout<<i<<' '<<mas[i];
            return 0;
        }
    printf("OK");
    return 0;
}
