#include <iostream>
#include <set>
#include <map>
#include <cstring>
using namespace std;
int main()
{
	int n,m,a,b,mm[104]={0},i,j;
	cin >> n >> m;
	for (i=0;i<m;i++)
	{
		cin >> a >> b;
		for(j=a;j<=b;j++)
			mm[j]++;
	}
	for (i=1;i<=n;i++)
		if (mm[i]>1 || mm[i]==0)
		{
			cout << i << " " << mm[i];
			return 0;
		}
	cout << "OK";
	return 0;
}
