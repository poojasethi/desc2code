#include <iostream>

using namespace std;

int main()
{
    int days[100]={0},n,m,a,b,i,k;

    cin >> n >> m;

    for(i=0;i<m;i++)
    {
        cin >> a >> b;
        for(k=a-1;k<b;k++) days[k]++;
    }

    for(i=0;i<n;i++) if(days[i]!=1) {cout << i+1 << " " << days[i]; return 0;}
    cout << "OK";

    return 0;
}
