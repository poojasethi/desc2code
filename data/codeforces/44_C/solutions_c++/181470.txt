#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int n, m;
	cin >> n >> m;

	vector<int> water(n);
	for(int i=0; i<m; ++i)
	{
		int a, b;
		cin >> a >> b;
		--a, --b;
		for(int d=a; d<=b; ++d)
			water[d] ++;
	}

	for(int i=0; i<n; ++i)
		if( water[i] != 1 )
			{cout << i+1 << " " << water[i] << endl; return 0;}
	cout << "OK" << endl;
}
