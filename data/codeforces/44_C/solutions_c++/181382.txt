#include <iostream>
#include <algorithm>
using namespace std;

int watered[1000];

int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		int start,end;
		cin>>start>>end;
		for(int j=start;j<=end;j++)
			watered[j]++;
	}
	for(int i=1;i<=n;i++)
	{
		if(watered[i]!=1)
		{
			cout<<i<<" "<<watered[i]<<endl;
			return 0;
		}
	}
	cout<<"OK"<<endl;
	return 0;
}
