#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    n, m = map(int, raw_input().split())
    lst = list([0]*n)
    for _ in xrange(m):
        a, b = map(int, raw_input().split())
        for i in xrange(a-1, b):
            lst[i] += 1
    if lst.count(1) != n:
        for i in xrange(n):
            if lst[i] != 1:
                print i+1, lst[i]
                break
    else:
        print "OK"
    return 0

if __name__ == '__main__':
    main()
