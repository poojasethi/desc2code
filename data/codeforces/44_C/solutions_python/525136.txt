n, m = map(int, raw_input().split())
g = [0] * (n + 1)
for i in xrange(m):
    a, b = map(int, raw_input().split())
    for j in xrange(a, b + 1):
        g[j] += 1
if g.count(1) == n:
    print 'OK'
else:
    for i in xrange(1, n + 1):
        if g[i] != 1:
            print i, g[i]
            break
