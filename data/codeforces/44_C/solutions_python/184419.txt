import sys

H, W = map(int, sys.stdin.readline().split(' '))
nwater = {}
for i in xrange(W):
    a, b = map(int, sys.stdin.readline().split(' '))
    for j in xrange(a,b+1):
        nwater.setdefault(j,0)
        nwater[j]+=1
ok = True
for i in xrange(1,H+1):
    if nwater.has_key(i)==False:
        print '%d 0' % i
        ok = False
        break
    elif nwater[i]>1:
        print '%d %d' % (i, nwater[i])
        ok = False
        break
if ok:
    print 'OK'
