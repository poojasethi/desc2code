#!/usr/bin/env python

import sys

(n, m) = map(int, sys.stdin.readline().split(' '))

x = [0] * n

for i in xrange(m):
    (a, b) = map(int, sys.stdin.readline().split(' '))
    for i in xrange(a - 1, b):
        x[i] += 1
for i in xrange(n):
    if x[i] != 1:
        print "%d %d" % (i + 1, x[i])
        exit(0)
print "OK"
