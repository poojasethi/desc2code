#include <iostream>
#include <cstring>

using namespace std;

int n,g[32][32],v[1<<24],f[1<<24];

int go(int x){
  int& r=v[x];
  if(r<0){
    if(x){
      r=100000000;
      for(int i=0;i<n;++i)if((1<<i)&x){
        for(int j=i;j<n;++j)if((1<<j)&x){
          int t=g[0][i+1]+g[i+1][j+1]+g[j+1][0]+go(x^((1<<i)|(1<<j)));
          if(t<r)r=t,f[x]=(1<<i)|(1<<j);
        }
        break;
      }
    }
    else r=0;
  }
  return r;
}

int main(void) {
  int x[32], y[32];
  cin>>x[0]>>y[0]>>n;
  for(int i=1;i<=n;++i)cin>>x[i]>>y[i];
  for(int i=0;i<=n;++i)for(int j=0,t;j<=n;++j)g[i][j]=g[j][i]=(x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
  memset(v,0xff,sizeof(v));
  cout<<go((1<<n)-1)<<endl;
  cout<<0;
  for(int t=(1<<n)-1;t;t^=f[t]){
    for(int i=0;i<n;++i)if((1<<i)&f[t])cout<<" "<<(i+1);
    cout<<" "<<0;
  }
  cout<<endl;

  return 0;
}
