#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
using namespace std;

int d[24], x[24], y[24], g[24][24], f[1 << 24], h[1 << 24];

int main()
{
	//freopen("input.txt", "r", stdin);
	int n, sx, sy;
	scanf("%d%d", &sx, &sy);
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d%d", &x[i], &y[i]);
		d[i] = (x[i]-sx)*(x[i]-sx)+(y[i]-sy)*(y[i]-sy);
	}
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++)
			g[i][j] = (x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
	f[0] = 0;
	for (int v = 1; v < (1 << n); v++) {
		int k;
		for (k = n-1; k >= 0; k--)
			if (v >> k & 1)
				break;
		f[v] = f[v ^ (1 << k)]+d[k]*2;
		h[v] = v ^ (1 << k);
		for (int i = 0; i < k; i++)
			if (v >> i & 1) {
				int ret = f[v ^ (1 << k) ^ (1 << i)]+d[k]+d[i]+g[k][i];
				if (ret < f[v]) {
					f[v] = ret;
					h[v] = v ^ (1 << k) ^ (1 << i);
				}
			}
	}
	printf("%d\n", f[(1 << n)-1]);
	printf("0");
	for (int v = (1 << n)-1, vv; v; v = vv) {
		vv = h[v];
		int m = 0, a[2];
		for (int i = 0; i < n; i++)
			if ((v >> i & 1) ^ (vv >> i & 1))
				a[m++] = i;
		for (int i = 0; i < m; i++)
			printf(" %d", a[i]+1);
		printf(" 0");
	}
	return 0;
}
