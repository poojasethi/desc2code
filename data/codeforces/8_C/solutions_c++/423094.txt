#include<cstdio>
using namespace std;
const int INF = 1000000007;
int n, x[30], y[30], d[30][30], dp[1<<24], pre[1<<24];
int main(){
	int xx, yy;
	scanf("%d%d%d",&xx,&yy,&n);
	for(int i=0;i<n;i++) scanf("%d%d",&x[i],&y[i]);
	x[n] = xx, y[n] = yy;
	for(int i=0;i<=n;i++)for(int j=0;j<=n;j++)
		d[i][j] = (x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
	for(int i=1;i<(1<<24);i++) dp[i] = INF;
	for(int i=0;i<(1<<n);i++)if(dp[i]!=INF){
		for(int j=0;j<n;j++)if(!(i&(1<<j))){
			int t = i|(1<<j);
			if(dp[t]>dp[i]+d[n][j]*2){
				dp[t] = dp[i]+d[n][j]*2;
				pre[t] = i;
			}
			for(int k=j+1;k<n;k++)if(!(t&(1<<k))){
				int p = t|(1<<k);
				if(dp[p]>dp[i]+d[n][j]+d[j][k]+d[k][n]){
					dp[p] = dp[i]+d[n][j]+d[j][k]+d[k][n];
					pre[p] = i;
				}
			}
			break;
		}
	}
	int m = (1<<n)-1;
	printf("%d\n", dp[m]);
	while(m){
		printf("0 ");
		for(int i=0;i<n;i++)
			if((m^pre[m])&(1<<i)) printf("%d ",i+1);
		m = pre[m];
	}
	puts("0");
}
