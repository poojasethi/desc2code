#include<cstdlib>
#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
#define maxn 24
#define maxp 1<<maxn
#define chk(a,b,c,d) if(a<0||a>b){a=b;c=d;}
#define sqr(a) (a)*(a)
int x[maxn],y[maxp],f[maxp],a[maxp],n,p,x0,y0,i,j,k,t;
void pr(int k)
{
    int t,i,j;
    t=a[k]-k;
    i=__builtin_ctz(t);
    if (t==(1<<i)) cout<<0<<' '<<i+1<<' ';
    else
    {
        j=__builtin_ctz(t-(1<<i));
        cout<<0<<' '<<i+1<<' '<<j+1<<' ';
    }
    if (a[k]!=p)pr(a[k]);
}
int main()
{
    cin>>x0>>y0>>n;
    for (i=0;i<n;i++) cin>>x[i]>>y[i];
    p=(1<<n)-1;
    memset(f,-1,sizeof(f));
    f[p]=0;
    for (k=p;k;k--) if (f[k]!=-1)
    {
        i=__builtin_ctz(k);
        chk(f[k&(k-1)],(sqr(x[i]-x0)+sqr(y[i]-y0))*2+f[k],a[k&(k-1)],k);
        for (t=k&k-1;t;t&=t-1)
        {
            j=__builtin_ctz(t);
            chk(f[k^(1<<i)^(1<<j)],f[k]+(sqr(x[i]-x0)+sqr(y[i]-y0))+(sqr(x[i]-x[j])+sqr(y[i]-y[j]))+
                    (sqr(x[j]-x0)+sqr(y[j]-y0)),a[k^(1<<i)^(1<<j)],k);
        }
    }
    cout<<f[0]<<endl;
    pr(0);
    cout<<0<<endl;
}
