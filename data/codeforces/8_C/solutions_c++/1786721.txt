#include<cstdio>
#include<cstring>
int pre[1<<24],dp[1<<24];
int x[32],y[32];
int que[32];
int sqr(int a){return a*a;}
int dis(int a,int b){
    return sqr(x[a]-x[b])+sqr(y[a]-y[b]);
}
int main(){
    int i,j,k,n;
    scanf("%d%d",&x[0],&y[0]);
    scanf("%d",&n);x[n]=x[0];y[n]=y[0];
    for(i=0;i<n;i++)   scanf("%d%d",&x[i],&y[i]);
    memset(dp,0x3F,sizeof(dp));
    dp[0]=0;
    int full=(1<<n)-1;
    for(i=0;i<=full;i++){
        int qn=0;
        for(j=0;j<n;j++){
            if(i>>j&1^1) que[qn++]=j;
        }
        for(j=0;j<qn;j++){
            int &a=que[0];
            int &b=que[j];
            int tmp=dp[i]+dis(n,a)+dis(n,b)+dis(a,b);
            int t=i|(1<<a)|(1<<b);
            if(tmp<dp[t]){
                dp[t]=tmp;
                pre[t]=i;
            }
        }
    }
    printf("%d\n",dp[full]);
    int s=full;
    while(s!=0){
         int p=s^pre[s];
         printf("0 ");
         for(i=0;i<n;i++){
             if(p&(1<<i)) {
                 printf("%d ",i+1);
             }
         }
         s=pre[s];
    }
    printf("0\n");
    return 0;
}
