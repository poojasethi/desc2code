#include<cstdio>
#include<queue>
using namespace std;

int xx,yy,N,x[30],y[30],ans=1<<29;
int pre[34554432],dp[34554432];

void dfs(int k,int s,int c,int f){

	if (c>ans) return;
	
	if (c>dp[s]) return ;
	else{
		dp[s] = c;
		pre[s] = f;
	}
	
	if (k>N){
		return ;
	}
	if ((s&(1<<(k-1)))>0){
		dfs(k+1,s,c,f);
	}else{
		dfs(k+1, (s|(1<<(k-1))), c+((x[k]-xx)*(x[k]-xx)+(y[k]-yy)*(y[k]-yy))*2, (1<<(k-1)));
		for (int i=k+1;i<=N;++i){
			if ((s&(1<<(i-1)))==0){
				dfs(k+1, (s|(1<<(k-1))|(1<<(i-1))), c+(x[k]-xx)*(x[k]-xx)+(y[k]-yy)*(y[k]-yy)+(x[i]-xx)*(x[i]-xx)+(y[i]-yy)*(y[i]-yy) + (x[k]-x[i])*(x[k]-x[i])+(y[k]-y[i])*(y[k]-y[i]), ((1<<(k-1))|(1<<(i-1))) );
			}
		}
	}
}

void whh(int s){
	if (s){
		whh(s^pre[s]);
		int k = pre[s], i = 1;
		while (k){
			if (k&1) printf(" %d",i);
			++i;
			k/=2;
		}
		printf(" 0");
	}else{
		printf("0");
	}
	
}

int main(){
	scanf("%d%d%d",&xx,&yy,&N);
	for (int i=1;i<=N;++i){
		scanf("%d%d",&x[i],&y[i]);
	}
	for (int i=0;i<=34000000;++i){
		dp[i]=1<<29;
	}
	
	dfs(1,0,0,0);
	
	printf("%d\n",dp[(1<<N)-1]);
	
	whh((1<<N)-1);
	puts("");
	return 0;
}
