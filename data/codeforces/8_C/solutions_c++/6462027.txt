#include <stdio.h>

int ans[1<<24],fro[1<<24];
int sx,sy,n,i,x[24],y[24],j,z,k,a1,a2;

int dist[25][25],p;
int d(int x,int y,int X,int Y)
 {
  return (x-sx)*(x-sx)+(y-sy)*(y-sy)+(X-sx)*(X-sx)+(Y-sy)*(Y-sy)+(X-x)*(X-x)+(Y-y)*(Y-y);
 }
void f(int k,int y)
 {
  k-=y;
  a1=0,a2=0;
  int z=0;
  while (k)
   {
    z++;
    if (k%2) { a2=a1; a1=z; }
    k/=2;
   }
 }

main()
 {
  scanf("%d%d%d",&sx,&sy,&n);
  for (i=0;i<n;i++)
   scanf("%d%d",&x[i],&y[i]);
  for (i=0;i<n;i++)
   for (j=0;j<n;j++)
    dist[i][j]=d(x[i],y[i],x[j],y[j]);
  for (i=1;i<(1<<n);i++) ans[i]=1000000000;
  for (i=0;i<(1<<n);i++)
    for (j=0;j<n;j++)
     if ( !(i&(1<<j)) ) {
      for (k=j;k<n;k++)
       if (!(i&(1<<k)))
        {
         p=(i|(1<<j))|(1<<k);
         if (ans[p] > ans[i]+dist[j][k])
          {
           ans[p]=ans[i]+dist[j][k];
           fro[p]=i;
          }
        } break; }
  k=(1<<n)-1;
  printf("%d\n",ans[k]);
  printf("0 ");
  while (k)
   {
    f(k,fro[k]);
    printf("%d ",a1);
    if (a2) printf("%d ",a2);
    k=fro[k];
    printf("0 ");
   }
 }
