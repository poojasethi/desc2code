#include<cstdio>
#include<cstring>
const int N=24;
int g[N+1][N+1],pr[1<<N],dp[1<<N],dis[N];
int main(){
	int sx,sy,n,x[N+1],y[N+1];
	scanf("%d%d%d",&sx,&sy,&n);
	for(int i=0;i<n;i++){
		scanf("%d%d",x+i,y+i);
		x[i]-=sx,y[i]-=sy;
		dis[i]=x[i]*x[i]+y[i]*y[i];
	}
	for(int i=0;i<n;i++)
		for(int j=i;j<n;j++)
			g[i][j]=g[j][i]=(x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
	memset(pr,-1,sizeof(pr));
	dp[0]=0;
	int m=1<<n,tmp,pt;
	for(int st=0;st<m;st++){
		for(int i=0;i<n;i++){
			if(st&(1<<i)){
				tmp=dis[i]<<1;
				pt=st^(1<<i);
				if(pr[st]==-1||dp[st]>dp[pt]+tmp){
					dp[st]=dp[pt]+tmp;
					pr[st]=pt;
				}
				for(int k=i+1;k<n;k++){
					if(st&(1<<k)){
						tmp=dis[i]+dis[k]+g[i][k];
						pt=st^(1<<i)^(1<<k);
						if(pr[st]==-1||dp[st]>dp[pt]+tmp){
							dp[st]=dp[pt]+tmp;
							pr[st]=pt;
						}
					}
				}
				break;
			}
		}
	}
	printf("%d\n",dp[--m]);
	while(m){
		printf("0 ");
		pt=pr[m];
		for(int i=0;i<n;i++){
			if((m&(1<<i))&&!(pt&(1<<i)))
				printf("%d ",i+1);
		}
		m=pt;
	}
	printf("0\n");
	return 0;
}
