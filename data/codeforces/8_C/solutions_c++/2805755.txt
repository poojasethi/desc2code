#include<cstdio>
#include<cmath>
#define sqr(i) ((i)*(i))
#define fr(i,n) for (int i=0; i<n; i++)
int f[1<<24],a[25][25],r[1<<24],n,x[25],y[25];
int main()
{










































    scanf("%d%d%d",x,y,&n), x[n]=x[0], y[n]=y[0];
    fr(i,n) scanf("%d%d",x+i,y+i);
    fr(i,n+1) fr(j,n+1) a[i][j]=sqr(x[i]-x[j])+sqr(y[i]-y[j]);
    fr(i,1<<n) f[i]=i?0x3f3f3f3f:0;
    fr(i,1<<n) if (f[i]<0x3f3f3f3f)
        fr(j,n) if (!(i>>j&1))
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
		
        {
            for (int k=j; k<n; k++) if (!(i>>k&1))
            {
                int p=i|1<<j|1<<k,q=f[i]+a[n][j]+a[k][j]+a[k][n];
                if (f[p]>q) f[p]=q, r[p]=i;
            }
            break;
        }
    printf("%d\n0",f[(1<<n)-1]);
    for (int i=(1<<n)-1; i; i=r[i])
    {
        for (int t=i^r[i]; t; t-=t&-t)
            printf(" %d",int(log(t&-t)/log(2)+1e-8+1));
        printf(" 0");
    }
}
