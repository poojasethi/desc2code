#include <bits/stdc++.h>
using namespace std;
#define REP(i, n) for (int i = 0; i < (n); i++)
#define TR(it, v) for (__typeof((v).begin()) it = (v).begin(); it != (v).end(); ++it)
int dp[1 << 24], dist[25][25], 
    traceback[1 << 24], x[25], y[25];
inline int sqr(int a){
    return a * a;
}
int main() {
    int n;
    scanf("%d %d %d", x, y, &n); x[n] = x[0]; y[n] = y[0];
    REP(i, n) scanf("%d %d", x + i, y + i);

    REP(i, n + 1) REP(j, n + 1) dist[i][j] = sqr(x[i] - x[j]) + sqr(y[i] - y[j]);
    memset(dp, 0x3f, sizeof dp);
    dp[0] = 0;
    REP(i, 1 << n) if (dp[i] < 0x3f3f3f3f) REP(j, n) if (!(i >> j & 1)) {
        for (int k = j; k < n; ++k) if (!(i >> k & 1)) {
            int nm = i | 1 << j | 1 << k;
            int nr = dp[i] + dist[n][j] + dist[k][j] + dist[k][n];
            if (dp[nm] > nr) dp[nm] = nr, traceback[nm] = i;
        }
        break;
    }
    printf("%d\n0 ", dp[(1 << n) - 1]);

    for (int i = (1 << n) - 1; i; i = traceback[i]) {
        for (int t = i ^ traceback[i]; t; t -= (t & -t)) 
            printf("%d ",  __builtin_ctz(t) + 1);
        printf("0 ");
    }
}
