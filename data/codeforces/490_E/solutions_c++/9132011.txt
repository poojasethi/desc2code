#include <bits/stdc++.h>

using namespace std;

int n;
string s[200000];
int fix[200000];

int greed (string s) {
  int uh = 0;
  for (int i = 0; i < (int) s.size(); i++) {
    uh *= 10;
    if (s[i] != '?') {
      uh += s[i] - '0';
    }
    else {
      uh += 9;
    }
  }
  return uh;
}

int go (string s, int lo) {
  if (greed (s) <= lo) return -1;
  for (int i = 0; i < (int) s.size(); i++) {
    if (s[i] != '?') continue;
    for (int j = 0; j < 10; j++) {
      if (i == 0 && j == 0) continue;
      s[i] = '0' + j;
      if (greed (s) <= lo) continue;
      else {
        break;
      }
    }
  }
  return greed (s);
}

int main() {
  ios_base::sync_with_stdio(0);
  cin >> n;
  bool ok = true;
  for (int i = 0; i < n; i++) {
    cin >> s[i];
    fix[i] = go (s[i], (i == 0) ? 0 : fix[i - 1]);
    if (fix[i] == -1) {
      ok = false;
      break;
    }
  }
  if (ok) {
    printf ("YES\n");
    for (int i = 0; i < n; i++) {
      printf ("%d\n", fix[i]);
    }
  }
  else {
    printf ("NO\n");
  }
}
