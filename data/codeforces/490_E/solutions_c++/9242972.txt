#include <bits/stdc++.h>
#define maxn 100009
using namespace std;
char s[maxn][10];
int n;
int trans(char *s)
{
	int ans=0;
	for(int i=0;s[i];i++)
	{
		ans*=10;
		if(s[i]=='?')
			ans+=9;
		else
			ans+=s[i]-'0';
	}
	return ans;
}
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%s",s[i]);
	}
	for(int j=0;s[0][j];j++)
	{
		if(s[0][j]=='?')
		{
			if(j)
				s[0][j]='0';
			else
				s[0][j]='1';
		}
	}
	for(int i=1;i<n;i++)
	{
		if(trans(s[i])<=trans(s[i-1]))
		{
			puts("NO");
			return 0;
		}
		else
		{
			for(int j=0;s[i][j];j++)
			{
				if(s[i][j]=='?')
				{
					for(int k=0;k<=9;k++)
					{
						if(!j&&!k)
							continue;
							s[i][j]=k+'0';
						if(trans(s[i])>trans(s[i-1]))
						{
							break;
						}
					}
				}
			}
		}
	}
	puts("YES");
	for(int i=0;i<n;i++)
		printf("%s\n",s[i]);
	return 0;
}
