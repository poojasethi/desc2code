#include<bits/stdc++.h>
using namespace std;

char s[100010][10];
int len[100010];

int func(char *a,char *b,int d,int n)
{
    if (d>=n)
        return 0;
    if (b[d]!='?')
    {
        if (a[d]==b[d])    return func(a,b,d+1,n);
        else return a[d]<b[d];
    }
    else
    {
        b[d]=a[d];
        if (func(a,b,d+1,n)) return 1;
        if (b[d]!='9')
        {
            b[d]++;return 1;
        }
        b[d]='?';
        return 0;
    }
}

int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
       scanf("%s",s[i]);
    s[0][0]='0';
    len[0]=1;

    for(int i=1;i<=n;i++)
    {
        len[i]=strlen(s[i]);
        if (len[i]<len[i-1])    {cout<<"NO"<<endl;return 0;}
        else
            if (len[i]==len[i-1]&&!func(s[i-1],s[i],0,len[i]))
                {cout<<"NO"<<endl;return 0;}

        for(int j=0;j<len[i];j++)
        {
            if (s[i][j]=='?')
            {
                if (j==0)   s[i][j]='1';
                else s[i][j]='0';
            }
        }

    }
    cout<<"YES"<<endl;
    for(int i=1;i<=n;i++)
        printf("%s\n",s[i]);
    return 0;
}
