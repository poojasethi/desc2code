#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std ;
int c (char a[]){
    int i ,m=0;
    for(i=0;i<strlen(a);i++){
        if (a[i]!='?')m = m*10+a[i]-'0';
        else m=m*10+9 ;
    }
    return m ;
}
int main()
{
    int n,i;
    cin >> n ;
    char a[n][10] ;
    int len = 0 ;
    for (i=0;i<n;i++){
        cin >> a[i];
        if(c(a[i])<=len){cout << "NO";return 0;}
        for (int j=0;j<strlen(a[i]);j++){
            if (a[i][j]=='?'){
                for (int k=0;k<10;k++){
                    if (j==0&&k==0)continue ;
                    a[i][j] = k +'0' ;
                    if (c(a[i])>len)break ;
                }
            }
            
        }
        len = c(a[i]) ;
        }
        cout << "YES" <<endl ; 
    for(i=0;i<n;i++){
        cout << a[i] <<endl ;
        }
   
    return 0;
}
