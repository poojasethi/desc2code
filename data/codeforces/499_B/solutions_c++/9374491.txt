#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main(){
	map<string, string>s;
	int i, n, m;
	cin>>n>>m;
	string a, b;
	for(i = 0; i < m; i++){
		cin>>a>>b;
		if(a.size() <= b.size())
		s[a] = a;
		else
		s[a] = b;
	}
	for(i = 0; i < n; i++){
		cin>>a;
		cout<<s[a]<<" ";
	}
	return 0;
}
