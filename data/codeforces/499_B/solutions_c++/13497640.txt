#include <bits/stdc++.h>
using namespace std;

int main(){
	int n,m,i;
	cin>>n>>m;
	string c1, c2;
	map<string,string>notes;
	for(i=0;i<m;i++){
		cin>>c1>>c2;
		if(c1.size() > c2.size())
			notes[c1] = c2;
		else
			notes[c1] = c1;
	}
	for(i=0;i<n;i++){
		cin>>c1;
		cout<<notes[c1]<<" "; 
	}
	cout<<"\n";
	return 0;
}
