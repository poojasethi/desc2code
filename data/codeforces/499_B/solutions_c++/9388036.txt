#include <bits/stdc++.h>
using namespace std;

int main(){
	int m,n;
	scanf("%d %d",&n,&m);
	string a,b;
	map<string,string> mapper;
	for(int i=0;i<m;i++){
		cin>>a>>b;
		mapper[a]=b;
	}
	for(int j=0;j<n;j++){
		cin>>b;
		a=mapper[b];
		if(a.size()<b.size()) cout<<a<<" ";
		else cout<<b<<" ";
	}
	cout<<"\n";
	return 0;
}
