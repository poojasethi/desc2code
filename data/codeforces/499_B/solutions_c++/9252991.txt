#include <bits/stdc++.h>
using namespace std;

int main()
{
	string a, b;
	int n, m;

	cin >> n >> m;

	map <string, string> mp;

	while(m--) {
		cin >> a >> b;
		if(a.length() <= b.length()) mp[a] = a;
		else mp[a] = b;
	}
	while(n--) {
		cin >> a;
		cout << mp[a] << ' ';
	}
	cout << endl;

	return 0;
}
