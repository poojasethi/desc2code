#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
int main()
{
	lld m,n;
	cin>>n>>m;
	map<string,string> a;
	string s,t;
	while(m--)
	{
		cin>>s>>t;
		if(t.size()<s.size())
			a[s]=t;
		else
			a[s]=s;
	}
	while(n--)
	{
		cin>>s;
		cout<<a[s]<<' ';
	}
	cout<<endl;
	return 0;
}
