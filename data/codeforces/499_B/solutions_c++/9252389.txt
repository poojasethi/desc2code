#include<bits/stdc++.h>
using namespace std;

int main()
{
int n,m,i;
cin>>n>>m;
map< string, string > mi;
while(m--)
{
string s1,s2;
cin>>s1>>s2;
//cout<<s1<<" h "<<s2;
if(s1.length()<=s2.length())
{
mi[s1]=s1;
}
else
{
mi[s1]=s2;
}
}
for(i=0;i<n;i++)
{
string s;
cin>>s;
cout<<mi[s]<<" ";
}
cout<<endl;
return 0;
}
