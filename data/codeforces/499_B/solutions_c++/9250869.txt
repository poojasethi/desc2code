#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,x;
	cin>>n>>x;
	map<string,string> dict;
	string l,r;
	for(int i=0;i<x;i++)
	{
		cin.ignore();
		cin>>l>>r;
		dict[l]=r;
	}
	cin.ignore();
	for(int i=0;i<n;i++)
	{
		cin>>l;
		if(l.length()>dict[l].length())
			l=dict[l];
		cout<<l<<" ";
	}
	return 0;
}
