#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,la,lb,i;
	string a,b;
	scanf("%d%d",&n,&m);
	map<string,string> m1;
	for(i=0;i<m;i++)
	{
		cin>>a>>b;
		m1[a]=b;
	}
	for(i=0;i<n;i++)
	{
		cin>>a;
		la=a.size();
		lb=m1[a].size();
		if(lb<la)
			cout<<m1[a]<<" ";
		else
			cout<<a<<" ";
	}
	printf("\n");
return 0;
}
