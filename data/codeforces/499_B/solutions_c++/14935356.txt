#include<iostream>
#include<map>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	string s1,s2;
	map<string,string> mp;

	for (int i = 0; i < m; ++i)
	{
		cin>>s1>>s2;
		mp[s1] = s2;
	}

	for(int i = 0;i<n;++i){
		cin>>s1;
		if(mp[s1].length()<s1.length())
			cout<<mp[s1]<<" ";
		else
			cout<<s1<<" ";
	}

	cout<<endl;
}
