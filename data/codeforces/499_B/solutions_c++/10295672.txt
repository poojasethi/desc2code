#include<bits/stdc++.h>

using namespace std;

int main() {
  int n,m;
  cin>>n>>m;
  string a,b;
  map<string, string> mine;
  for(int i=0; i<m; i++) {
    cin>>a>>b;
    if(b.size() < a.size())
      mine[a] = b;
    else
      mine[a] = a;
  }
  for(int i=0; i<n; i++) {
    cin>>a;
    cout<<mine[a]<<" ";
  }
  cout<<endl;
  return 0;
}
