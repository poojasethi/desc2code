#include <bits/stdc++.h>
using namespace std;
map <string ,string> mapa;
 int n,m;
 string s1,s2;
int main(){
 cin>>n>>m;
 for(int i=0;i<m;i++)
 {
  cin>>s1>>s2;
  if(s1.size()<=s2.size()) mapa[s1]=s1; else mapa[s1]=s2;
 }
 for(int i=0;i<n;i++)
 {
  cin>>s1;
  cout<<mapa[s1]<<' ';
 }
 cout<<endl;}
