#include<bits/stdc++.h>
#define map map<string, string> 

using namespace std;

map k;

int main()
{
	string a,b,c;
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=m;i++)
	{
		cin>>a>>b;
		if(a.size()<=b.size())
		{
			k[a]=a;
		}
		else
		{
			k[a]=b;
		}
	}
	for(int i=1;i<=n;i++)
	{
		cin>>c;
		cout<<k[c]<<" ";
	}
}
