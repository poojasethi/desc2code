#include<bits/stdc++.h>
using namespace std;
int main()
{
string st,a,b;
int n,l,i;
map<string,string> m;
cin>>n>>l;
for(i=0;i<l;i++)
{
    cin>>a>>b;
    if(a.length()<=b.length())
        m[a]=a;
    else if(a.length()>b.length())
        m[a]=b;
}
for(i=0;i<n;i++)
{
    cin>>st;
    cout<<m[st]<<" ";
}
return 0;
}
