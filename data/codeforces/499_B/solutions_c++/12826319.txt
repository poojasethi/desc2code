#include<iostream>
#include<map>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
	int n, m;	
	cin>>n>>m;
	string a, b;
	map<string, string> mp;
	while(m--)
	{
		cin>>a>>b;
		mp[a] = b;
		
	}
	while(n--)
	{
		cin>>a;
		if(a.length()<=mp[a].length()) cout<<a;
		else cout<<mp[a];
		cout<<" ";
	}
	

}
