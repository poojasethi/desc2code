#include <bits/stdc++.h>
using namespace std;

map <string ,string> map_a;

int main()
{
 int n,m;
 cin>>n>>m;
 string s1,s2;
 for(int i=0;i<m;i++)
 {
  cin>>s1>>s2;
  if(s1.size()<=s2.size()) map_a[s1]=s1; else map_a[s1]=s2;
 }
 for(int i=0;i<n;i++)
 {
  cin>>s1;
  cout<<map_a[s1]<<' ';
 }
 cout<<endl;
 return 0;
}
