#include<iostream>
#include<map>

using namespace std;

int main() {
	int n,m;
	
	cin >> n >>m;
	map<string, string> m1;
	string s1,s2,s;
	
	while(m--) {
		cin >> s1>> s2;
		m1[s1] = s2;
	}
	
	while(n--) {
		cin >> s;
		if(s.length() > m1[s].length()) cout<<m1[s]<<" ";
		else cout<<s<<" ";
	}
	
	return 0;
}
