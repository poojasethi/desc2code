#include<stdio.h>
#include<string.h>
int n,m,i,j;
char a[3001][20],b[3001][20],s[20];
int main(){
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++)
		scanf("%s%s",a[i],b[i]);
	for(i=0;i<n;i++){
		scanf("%s",s);
		for(j=0;j<m;j++){
			if(strcmp(s,a[j])==0)
				printf("%s ",strlen(a[j])>strlen(b[j])?b[j]:a[j]);
		}
	}
}
