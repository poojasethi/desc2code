#include <iostream>
#include <map>
using namespace std;

int main() {
	int n, m;
	map<string, string> dict;
	string a,b;
	
	cin >> n >> m;
	while(m--) {
		cin >> a >> b;
		if (b.size() < a.size()) {
			dict[a] = b;
		} else {
			dict[a] = a;
		}
	}
	while(n--) {
		cin >> a;
		cout << dict[a] << " ";
	}
	return 0;
}
