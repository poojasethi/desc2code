#include <cstdio>
#include <cstring>
#include <algorithm>
#define rep(i,s,t) for (int i = s; i <= t; ++i)
#define per(i,s,t) for (int i = s; i >= t; --i)
using namespace std;
char ch;
inline void read(int &x){
	for (ch = getchar(); ch < '0' || ch > '9'; ch = getchar());
	for (x = 0; ch >= '0' && ch <= '9'; ch = getchar()) x = x*10+ch-48;
}
struct res{
	int xl, xr, yl, yr, z, id;
	void Read(){
		read(xl); read(xr); read(yl); read(yr); read(z);
	}
}p[100010];
int n, l[100010], r[100010];
inline bool cmp(res A, res B){ return A.z < B.z; }
int main(){
	read(n);
	rep(i,1,n){
		p[i].Read(); p[i].id = i;
	}
	sort(p+1, p+n+1, cmp);
	rep(i,1,n) l[i] = i-1, r[i] = i+1;
	r[0] = 1; r[n] = 0;
	int Q, ans, x, y; read(Q);
	while (Q--){
		ans = 0; read(x); read(y);
		for (int i = r[0]; i; i = r[i])
		if (x >= p[i].xl && x <= p[i].xr && y >= p[i].yl && y <= p[i].yr){
			ans = p[i].id; r[l[i]] = r[i]; l[r[i]] = l[i]; break;
		}
		printf("%d\n", ans);
	}
	return 0;
}
