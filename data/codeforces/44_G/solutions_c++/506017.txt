#include<stdio.h>
#include<algorithm>
using namespace std;
const int MAXN = 100010;
struct NODE
    {
        bool ok;
        int xl,xr,yl,yr;
        int id;
        int z;
        void get(int _id)
            {
                id = _id;
                scanf("%d%d%d%d%d",&xl,&xr,&yl,&yr,&z);
                ok = true;
            }
    };
bool cmp(const NODE &a ,const NODE &b)
    {
        return a.z < b.z;
    }
int next[MAXN] ,pre[MAXN];
NODE rec[MAXN];
int main()
    {
        int n ,m;
        scanf("%d",&n);
        for(int i = 0 ;i< n;i++)
            {
                rec[i].get(i+1);
                next[i] = i+1;
                pre[i+1] = i;
            }
        pre[0] = 0;
        sort(rec,rec+n,cmp);
        scanf("%d",&m);
        for(int i = 0 ; i < m ;i++)
            {
                int x0,y0;
                scanf("%d%d",&x0,&y0);
                bool fail = true;
                for(int j =0;j<n;j = next[j])
                    {
                        if(rec[j].ok && rec[j].xl <= x0 && rec[j].xr >=x0 && rec[j].yl <=y0 && rec[j].yr >=y0)
                            {
                                fail = false;
                                rec[j].ok = false;
                                next[ pre[j] ] =next[j];
                                pre[next[j]] = pre[j];
                                printf("%d\n",rec[j].id);
                                break;
                            }
                    }
                if(fail)puts("0");
            }
        return 0;
    }
