#include <cstdio>
#include <algorithm>

using namespace std;

struct target{
    int id,xl,xr,yl,yr,z;
    
    target(){}
    
    bool operator < (target X)const{
        return z < X.z;
    }
}T[100001];

int prev[100002],nxt[100002];

int main(){
    int n,m;
    
    scanf("%d",&n);
    
    for(int i = 1;i <= n;++i){
        scanf("%d %d %d %d %d",&T[i].xl,&T[i].xr,&T[i].yl,&T[i].yr,&T[i].z);
        T[i].id = i;
    }
    
    sort(T + 1,T + (n+1));
    nxt[0] = 1;
    
    for(int i = 1;i <= n;++i){
        nxt[i] = i+1;
        prev[i] = i-1;
    }
    
    scanf("%d",&m);
    
    for(int i = 0,j,x,y,ans;i < m;++i){
        scanf("%d %d",&x,&y);
        ans = 0;
        
        j = nxt[0];
        
        while(j != n+1){
            if(T[j].xl <= x && x <= T[j].xr && T[j].yl <= y && y <= T[j].yr){
                ans = T[j].id;
                prev[nxt[j]] = prev[j];
                nxt[prev[j]] = nxt[j];
                break;
            }
            
            j = nxt[j];
        }
        
        printf("%d\n",ans);
    }
    
    return 0;
}
