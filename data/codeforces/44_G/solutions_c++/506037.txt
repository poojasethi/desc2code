#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

struct node{
    int x0,x1,y0,y1,z;
} e[100100];
int a[100100],next[100100];

bool cmp(const int u,const int v)
{
    return e[u].z<e[v].z;
}

int main()
{
    int n,m,x,y;

    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d %d %d %d %d",&e[i].x0,&e[i].x1,&e[i].y0,&e[i].y1,&e[i].z);
        a[i]=i;
    }
    sort(a+1,a+n+1,cmp);
    for(int i=0;i<=n;i++) next[i]=i+1;
    scanf("%d",&m);
    while(m--)
    {
        scanf("%d %d",&x,&y);
        int pre=0;
        bool flag=1;
        while(1)
        {
            int p=next[pre];
            if (p>n) break;
            int t=a[p];
            if (e[t].x0<=x&&e[t].x1>=x&&e[t].y0<=y&&e[t].y1>=y)
            {
                next[pre]=next[p];
                printf("%d\n",t);
                flag=0;
                break;
            }
            pre=p;
        }
        if (flag) puts("0");
    }
    return 0;
}
