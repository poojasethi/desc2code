#include<iostream>
#include<algorithm>
#include<cstring>
#include<stdio.h>
#include<cmath>
#include<string>
#include<ctime>
using namespace std; 
struct A{int xl,xr,yl,yr,z,id;}aa[123231],a[123123];
int id[123232];
int pt(int x ,int y)
{
    return aa[x].z>aa[y].z;
}

int main()
{
    int n,m;
    cin>>n;
    for(int i=0;i<n;i++)
        scanf("%d%d%d%d%d",&aa[i].xl,&aa[i].xr,&aa[i].yl ,&aa[i].yr,&aa[i].z),aa[i].id=id[i]=i;;
    sort(id,id+n,pt);
    int to[123121];
    for(int i=0;i<=n;i++)
    {
        a[i]=aa[id[i]];
        to[i]=i-1;
    }
    cin>>m;
    while(m--)
    {
        int x,y,fk=0,tk;
        scanf("%d %d",&x,&y);
        for(int k=n;to[k]!=-1;k=tk)
        {
            tk=to[k];
            if(x>=a[tk].xl&&x<=a[tk].xr&&y>=a[tk].yl&&y<=a[tk].yr)
            {
                fk=a[tk].id+1;
                to[k]=to[tk];
                break;
            }
        }
        printf("%d\n",fk);
    }
    cin>>n;
}
