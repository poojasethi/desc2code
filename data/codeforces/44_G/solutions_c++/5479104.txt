#include <cstdio>
#include <algorithm>
using namespace std;

inline void R(int &x) {
	char ch = getchar(); x = 0;
	for (; ch<'0'; ch = getchar());
	for (; ch>='0'; ch=getchar()) x = x*10 + ch-'0';
}
const int N = 100005;
struct rect {
	int x1, y1, x2, y2, z, id;
	rect *q, *h;
	inline void read() {
		R(x1); R(x2); R(y1); R(y2); R(z);
	}
} a[N];
int n, m;
inline bool cmp(rect x, rect y) {return x.z < y.z;}
int main() {
	R(n);
	for (int i=1; i<=n; ++i)
		a[i].read(), a[i].id = i;
	sort(a+1, a+n+1, cmp);
	for (int i=1; i<=n+1; ++i)
		a[i-1].h = &a[i], a[i].q = &a[i-1];
	bool f;
	int x, y;
	R(m);
	for (int i=1; i<=m; ++i) {
		R(x); R(y); f = 0;
		for (rect *cur=a[0].h; cur->id; cur=cur->h)
			if (x>=cur->x1&&x<=cur->x2&&y>=cur->y1&&y<=cur->y2) {
				cur->q->h = cur->h, cur->h->q = cur->q;
				printf("%d\n", cur->id);
				f = 1;
				break;
			}
		if (!f) puts("0");
	}
	return 0;
}
