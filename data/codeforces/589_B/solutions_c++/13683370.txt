#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long n;
    cin>>n;
    pair<long long,long long>p[4005];
    long long x,y;
    map<long long,long long>mp;
    for(long long i=0;i<n;++i)
    {
        cin>>x>>y;
        if(x>y)swap(x,y);
        p[i]=make_pair(x,y);
        ++mp[y];
    }
    sort(p,p+n);
    long long ans=0,nans,xsol=0,ysol=0;
    for(long long i=0;i<n;++i)
    {
        x=p[i].first;
        long long num=0;
        for(map<long long,long long>::iterator it=mp.end();it!=mp.begin();)
        {
            --it;
            num+=it->second;
            nans=x*(it->first)*num;
            if(ans<nans)
            {
                xsol=x,ysol=(it->first);
                ans=nans;
            }
        }
        --mp[p[i].second];
    }
    cout<<ans<<endl;
    cout<<xsol<<" "<<ysol;
}
