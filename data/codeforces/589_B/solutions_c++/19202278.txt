#include<iostream>
#include<algorithm>
using namespace std;

struct nmb
{
	long long x,y;
}a[10010];

long long n,ans,ansx,ansy;

bool cmp1(nmb a,nmb b)
{
	if(a.x==b.x)
		return a.y<b.y;
	return a.x<b.x;
}

bool cmp2(nmb a,nmb b)
{
	if(a.y==b.y)
		return a.x<b.x;
	return a.y<b.y;
}

int main()
{
	cin>>n;
	for(long long i=1;i<=n;i++)
	{
		long long x,y;
		cin>>x>>y;
		if(x>y)
			swap(x,y);
		a[i].x=x;
		a[i].y=y;	
	}
	sort(a+1,a+n+1,cmp1);
	for(long long i=1;i<=n;i++)
	{
		long long x=a[i].x;
		sort(a+i,a+n+1,cmp2);
		for(long long j=i;j<=n;j++)
		{
			long long tmp=a[j].y*x*(n-j+1);
			if(tmp>ans)
			{
				ans=tmp;
				ansx=x;
				ansy=a[j].y;
			}
		}
		sort(a+1,a+n+1,cmp1);
	}
	cout<<ans<<endl<<ansx<<' '<<ansy<<endl;
	return 0;
}
