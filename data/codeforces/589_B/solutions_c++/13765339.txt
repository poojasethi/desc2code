#include<cstdio>
#include<cstring>
#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;

const int N = 4000 + 10;

int n;
long long ret, ra, rb;
int a[N], b[N];
int v[N], tot;

void solve(int len)
{
	tot = 0;
	for(int i = 0; i < n; ++ i) {
		if (a[i] >= len) {
			v[tot ++] = b[i];
		} else if (b[i] >= len) {
			v[tot ++] = a[i];
		}
	}
	sort(v, v + tot);
	reverse(v, v + tot);

	for(int i = 0; i < tot; ++ i) {
		long long vol = (long long)len * v[i] * (i + 1);
		if (vol > ret) {
			ret = vol;
			ra = len;
			rb = v[i];
		}
	}
}

int main()
{
	cin >> n;
	for(int i = 0; i < n; ++ i) {
		scanf("%d%d", a + i, b + i);
		if (a[i] > b[i]) swap(a[i], b[i]);
	}

	ret = 0;
	ra, rb;

	for(int i = 0; i < n; ++ i) {
		solve(a[i]);
		solve(b[i]);
	}
	cout << ret << endl;
	cout << ra << ' ' << rb << endl;
	return 0;
}
