#include<cstdio>
#include<vector>
#include<set>
#include<algorithm>

using namespace std;

pair<int,int> a[4444];
multiset<int> S;
multiset<int>::iterator it;

int main(){
	long long mx=-1,mxa,mxb;
	int i,j,n;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d%d",&a[i].first,&a[i].second);
		if(a[i].first>a[i].second)swap(a[i].first,a[i].second);
	}	
	sort(a,a+n);
	for(i=n-1;i>=0;i--){
		S.insert(a[i].second);
		it=S.begin();
		for(j=i;j<n;j++){
			if(mx<1LL*a[i].first*(*it)*(n-j)){
				mx=1LL*a[i].first*(*it)*(n-j);
				mxa=a[i].first;
				mxb=*it;
			}
			it++;
		}
	}
	printf("%I64d\n%I64d %I64d\n",mx,mxa,mxb);
}
