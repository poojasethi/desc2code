#include <iostream>
#include <algorithm>
using namespace std;

int main(){
    int n, a[4000], b[4000];
    unsigned long long r=0, l, w;
    cin >> n;
    for(int i=0; i<n; i++){
        cin >> a[i] >> b[i];
        if(a[i]<b[i])swap(a[i],b[i]);
    }
    for(int i=0; i<n; i++){
        for(int j=i+1; j<n; j++){
            if(a[j]>a[i]){
                swap(a[i],a[j]);
                swap(b[i],b[j]);
            }
        }
    }
    for(int i=0; i<n; i++){
        sort(b,b+1+i);
        for(int j=0; j<=i; j++){
            if((unsigned long long)a[i]*b[j]*(i-j+1)>r){
                l=a[i]; w=b[j];
                r=l*w*(i-j+1);
            }
        }
    }
    cout << r << '\n' << l << ' ' << w;
}
