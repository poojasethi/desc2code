#include<bits/stdc++.h>
using namespace std;
#define Rep(i,n) for(int i=1;i<=n;++i)
#define INF 0x3f3f3f3f
#define LL long long

int x[4040], y[4040], a[4040], b[4040];
int dp[4040][4040];

int main()
{
	int n, nx, ny;
	cin >> n;
	Rep(i, n) {
		cin >> x[i] >> y[i];
		if(x[i] > y[i]) swap(x[i], y[i]);
		a[i] = x[i], b[i] = y[i];
	}
	sort(x+1,x+1+n);
	sort(y+1,y+1+n);
	nx = unique(x+1,x+1+n)-x-1;
	ny = unique(y+1,y+1+n)-y-1;
	Rep(i, n) {
		int u = lower_bound(x+1,x+1+nx,a[i])-x;
		int v = lower_bound(y+1,y+1+ny,b[i])-y;
		++dp[u][v];
	}
	LL ans = 0;
	int mx, my;
	for(int i = nx; i; --i) {
		for(int j = ny; j; --j) {
			dp[i][j] += dp[i+1][j] + dp[i][j+1] - dp[i+1][j+1];
			LL temp = (LL)dp[i][j] * x[i] * y[j];
			if(temp > ans) ans = temp, mx = i, my = j;
		}
	}
	cout << ans << endl << x[mx] << ' ' << y[my] << endl;
	return 0;
}
