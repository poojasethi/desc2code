#include <bits/stdc++.h>

using namespace std;

int n;

long long a[10000], b[10000], x, y, z;

int main(){
    cin>>n;
    for(int i=0; i<n; i++) { 
        cin>>a[i]>>b[i];
        if(a[i]>b[i]) swap(a[i],b[i]);
    }
    for(int i=0; i<n-1; i++)
        for(int j=i+1; j<n; j++)
            if(a[i]<a[j]) { 
                swap(a[i],a[j]); 
                swap(b[i],b[j]);
            }
    for(int i=0; i<n; i++) {
        sort(b,b+i+1);
        for(int j=0; j<=i; j++)
            if(b[j]*a[i]*(i-j+1)>z) x = a[i], y = b[j], z = x*y*(i-j+1);
    }
    cout<<z<<endl<<x<<" "<<y;
    return 0;
}
