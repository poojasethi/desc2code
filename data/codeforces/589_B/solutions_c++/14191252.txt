#include<bits/stdc++.h>
#define maxi 100005
#define f first
#define s second
#define mp make_pair
#define pb push_back
#define ll long long
#define si(n) scanf("%d",&n)
#define sll(n) scanf("%lld",&n)
using namespace std;
vector<pair<int,int> > vec;
multiset<int> svec;
multiset<int>::iterator it;
int main(){
	int n,x,y,a,b;
	long long int ma=0;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a>>b;
		vec.pb(mp(min(a,b),max(a,b)));
	}
	sort(vec.begin(),vec.end());
	for(int i=n-1;i>=0;i--){
		svec.insert(vec[i].s);
		it=svec.begin();
		for(int j=i;j<n;j++){
			if((long long)(1LL*(*it)*1LL*vec[i].f*1LL*(n-j))>ma){
				ma=(long long)(1LL*(*it)*1LL*vec[i].f*1LL*(n-j));
				x=vec[i].f;
				y=*it;
			}
			it++;
		}
	}
	cout<<ma<<"\n"<<x<<" "<<y<<endl;
	return 0;
}
