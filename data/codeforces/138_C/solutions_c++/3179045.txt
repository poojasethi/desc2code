#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <map>
#include <vector>
using namespace std;
#define LL long long
#define maxn 210000
#define INF 1<<30
#define fi first
#define se second
#define mp make_pair
#define pii pair<int,int>

vector<pii>tr,mu;
int n,m;
void init()
{
    int i,j,a,h,l,r;
    for(i=0;i<n;i++){
        scanf("%d %d %d %d",&a,&h,&l,&r);
        tr.push_back(mp(a-h,l)); tr.push_back(mp(a,-l));
        tr.push_back(mp(a+1,r)); tr.push_back(mp(a+h+1,-r));
    }
    for(i=0;i<m;i++)
    {
        scanf("%d %d",&a,&h);
        mu.push_back(mp(a,h));
    }
    sort(tr.begin(),tr.end());
    sort(mu.begin(),mu.end());
}
void gao()
{
    int i,j,cnt=0,sz=tr.size();
    double p=0,ans=0;
    for(i=j=0;i<m;i++)
    {
        int idx=mu[i].fi,tp;
        while(j<sz&&tr[j].fi<=idx)
        {
            tp=tr[j].se;
            if(tp>0)
            {
                if(tp==100) cnt++;
                else p+=log(1.0-tp*1.0/100);
            }
            if(tp<0)
            {
                tp*=-1;
                if(tp==100) cnt--;
                else p-=log(1.0-tp*1.0/100);
            }
            ++j;
        }
        if(!cnt) ans+=double(mu[i].se)*exp(p);
    }
    printf("%.04lf\n",ans);
}
int main()
{
    while(scanf("%d %d",&n,&m)!=EOF)
    {
        init();
        gao();
    }
}
