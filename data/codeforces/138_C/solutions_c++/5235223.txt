#include <cstdio>
#include <cstring>
#include <iostream>
#include <set>
#include <vector>
#include <algorithm>
#include <cmath>

#define F first
#define S second
#define PB push_back
#define MP make_pair
#define INF 0x3f3f3f3f

using namespace std ;

typedef pair<int,int> PII ;
typedef pair<int,PII> PIP ;
typedef pair<PII,int> PPI ;

const int MAXN = (int)1e6 + 10 ;

vector< PIP > v ;
vector< PII > b ;
int n,m;

int main(){
    scanf("%d%d",&n,&m);
    for (int i=0;i<n;i++){
        int x,h,l,r;
        scanf("%d%d%d%d",&x,&h,&l,&r);
        v.PB(MP(x-h,MP(1,100-l)));
        v.PB(MP(x,MP(-1,100-l)));
        v.PB(MP(x+1,MP(1,100-r)));
        v.PB(MP(x+h+1,MP(-1,100-r)));
    }
    sort(v.begin(),v.end());
    for (int i=0;i<m;i++){
        int bb,z;
        scanf("%d%d",&bb,&z);
        b.PB(MP(bb,z));
    }
    sort(b.begin(),b.end());
    double ans=0.0,totexp=0.0;
    int cnt0=0;
    int st=0;
    double con = log(100.0) ;
    for (int i=0;i<b.size();i++){
        while (st<v.size() && v[st].F<=b[i].F){
            if (v[st].S.F==-1){
                if (v[st].S.S==0) cnt0-- ;
                else totexp -= log(v[st].S.S / 100.0) ;
            }else{
                if (v[st].S.S==0) cnt0++ ;
                else totexp += log(v[st].S.S / 100.0) ;
            }
            st++ ;
        }
        if (cnt0>0) continue ;
        else ans += b[i].S*exp(totexp) ;
    }
    printf("%.12f\n",ans) ;
    return 0;
}
