#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;

struct Node
{
    int s,x,pro;
}p[100005*7];
int cnt[101];
bool cmp(Node p1, Node p2)
{
    if(p1.x!=p2.x)return p1.x<p2.x;
    return p1.s<p2.s;
}
int main()
{
    int n,m,N=0;
    scanf("%d%d",&n,&m);
    for(int i = 0; i < n; i++)
    {
        int a,h,l,r;
        scanf("%d%d%d%d",&a,&h,&l,&r);
        p[N].s = -1;
        p[N].x = a-h;
        p[N++].pro = l;
        p[N].s = 1;
        p[N].x = a-1;
        p[N++].pro = l;
        p[N].s = -1;
        p[N].x = a+1;
        p[N++].pro = r;
        p[N].s = 1;
        p[N].x = a+h;
        p[N++].pro = r;
    }
    for(int i = 0; i < m; i++)
    {
        int x,y;scanf("%d%d",&x,&y);
        p[N].s = 0;
        p[N].x = x;
        p[N++].pro = y;
    }
    sort(p,p+N,cmp);
    double ret = 0;
    for(int i = 0; i < N; i++)
    {
        if(p[i].s==0)
        {
            if(cnt[0])continue;
            double a = 1;
            for(int j = 1; j <= 100; j++)
                a *= pow(j/100.0,cnt[j]);
            ret += p[i].pro*a;
        }
        else if(p[i].s==-1) cnt[100-p[i].pro]++;
        else cnt[100-p[i].pro]--;
    }
    printf("%.10f\n",ret);
    return 0;
}
