#include<bits/stdc++.h>
using namespace std;
const int N=505,K=505;
int n,k,c[N]={},v[K][K]={};
int main()
{	
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;++i)
		scanf("%d",&c[i]);
	v[0][0]=true;
	for(int i=1;i<=n;++i)
		for(int v1=k;v1>=0;--v1)
			for(int v2=k;v2>=0;--v2)
			{
				if(v1>=c[i])
					v[v1][v2]|=v[v1-c[i]][v2];
				if(v2>=c[i])
					v[v1][v2]|=v[v1][v2-c[i]];
			}
	int ans=0;
	for(int i=0;i<=k;++i)
		if(v[i][k-i])
			++ans;
	printf("%d\n",ans);
	for(int i=0;i<=k;++i)
		if(v[i][k-i])
			printf("%d ",i);
	printf("\n");
	return 0;
}
