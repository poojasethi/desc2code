#include <bits/stdc++.h>

using namespace std;

bool DP[1002][1002];
int b[20020],ans,i,j,x,k,n;

int main()
 {
  cin>>n>>k;
  DP[0][0]=1;
  while (n--)
   {
    scanf("%d",&x);
    for (j=500;j>=0;j--)
     for (i=j;i>=0;i--)
      if (DP[j][i])
       DP[j+x][i+x]=DP[j+x][i]=1;
   }
  for (i=0;i<=k;i++)
   if (DP[k][i]) { ans++; b[ans]=i; }
  cout<<ans<<endl;
  for (i=0;i<=k;i++)
   if (DP[k][i]) cout<<i<<" ";
 }
