#define N (1<<9)
#include <bits/stdc++.h>
using namespace std;

int n,k,c[N];
vector<int> res;
bool dp[N][N]={1};

int main()
{
	cin>>n>>k;
	for(int i=1;i<=n;i++)
	{
		cin>>c[i];
		for(int j=k;j>=c[i];j--)
			for(int r=k;r>=0;r--)
			{
				dp[j][r]|=dp[j-c[i]][r];
				if(r>=c[i]) dp[j][r]|=dp[j-c[i]][r-c[i]];
			}
	}
	for(int i=0;i<=k;i++)
		if(dp[k][i]) res.push_back(i);
	cout<<res.size()<<endl;
	for(int i=0;i<res.size();i++)
		cout<<res[i]<<" ";
	cout<<endl;
	return 0;
}
