#include<cstdio>
#include<cstring>
const int maxn = 2100;
int a[maxn], dp[maxn][maxn];
int main(){
	memset(dp, 0, sizeof(dp));
	dp[0][0] = true;
	int n, s;
	scanf("%d%d", &n, &s);
	for(int i = 0; i < n; i++) scanf("%d", &a[i]);
	for(int i = 0; i < n; i++)
		for(int j = s - a[i]; j >= 0; j--)
			for(int k = j; k >= 0; k--)
				if(dp[j][k]) dp[j+a[i]][k] = dp[j+a[i]][k+a[i]] = true;
	int cnt = 0;
	for(int i = 0; i <= s; i++) if(dp[s][i]) cnt++;
	printf("%d\n", cnt);
	for(int i = 0; i <= s; i++) if(dp[s][i]) printf("%d ", i);
	printf("\n");
	return 0;
}
