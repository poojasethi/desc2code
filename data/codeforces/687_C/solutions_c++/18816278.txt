#include<cstdio>
using namespace std;
const int N=505;
int n,m;
int a[N];
bool f[N][N];
void DP(){
	f[0][0]=1;
	for(int i=1;i<=n;i++){
		for(int j=m;j>=0;j--){
			for(int k=m;k>=0;k--){
				if(a[i]<=j) f[j][k]|=f[j-a[i]][k];
				if(a[i]<=k) f[j][k]|=f[j][k-a[i]];
			}
		}
	}
}
void Init(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d",a+i);
}
int main(){
	Init();
	DP();
	int ans=0;
	for(int i=0;i<=m;i++) if(f[i][m-i]) ans++;
	printf("%d\n",ans);
	for(int i=0;i<=m;i++) if(f[i][m-i]) printf("%d%c",i,i==m?'\n':' ');
	return 0;
}
