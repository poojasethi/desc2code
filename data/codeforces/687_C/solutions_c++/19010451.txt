#include<cstdio>
const int N=505;
int n,k,ans,c[N];
bool f[N][N];
int main(){
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++) scanf("%d",&c[i]);
	f[0][0]=1;
	for (int i=1;i<=n;i++)
	 for (int j=k;j>=c[i];j--)
	  for (int l=k;l>=0;l--)
	  if (l>=c[i])
	   f[j][l]|=f[j-c[i]][l-c[i]]|f[j-c[i]][l]; 
	   else f[j][l]|=f[j-c[i]][l];
	for (int i=0;i<=k;i++) if (f[k][i]) ans++;
	printf("%d\n",ans);
	for (int i=0;i<=k;i++) if (f[k][i]) printf("%d ",i); 
	return 0;
}
