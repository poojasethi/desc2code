#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
const int maxn = 500 + 10;
bool f[maxn][maxn];
int c[maxn],n,m;
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
	scanf("%d",&c[i]);
	f[0][0]=1;
	for(int i=1;i<=n;i++)
	for(int j=m;j>=0;j--)
	for(int k=m;k>=0;k--)
	if(f[j][k]&&j+c[i]<=m)
	{
		f[j+c[i]][k+c[i]]=1;
		f[j+c[i]][k]=1;
	} 
	int cnt=0;
	for(int i=0;i<m;i++)
	if(f[m][i])cnt++;
	cout<<cnt+1<<endl;
	for(int i=0;i<m;i++) 
	if(f[m][i])printf("%d ",i);
	cout<<m<<endl;
	return 0; 
}
