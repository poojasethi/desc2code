#include <cmath>
#include <cstdio>
#include <bitset>
#include <cstring>
#include <algorithm>

using namespace std;

bitset<510> a[510];
int ans[510];

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int n,k;scanf("%d%d",&n,&k);
	a[0][0]=1;
	for (int i=0;i<n;i++)
	{
		int x;scanf("%d",&x);
		for (int j=k;j>=x;j--)
			a[j]=a[j]|a[j-x]|(a[j-x]<<x);
	}
	int cnt=0;
	for (int i=0;i<=k;i++) if (a[k][i]) ans[++cnt]=i;
	printf("%d\n",cnt);
	for (int i=1;i<=cnt;i++) printf("%d ",ans[i]);
	return 0;
}
