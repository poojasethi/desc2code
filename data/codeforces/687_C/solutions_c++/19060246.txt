#include<stdio.h>
#include<iostream>
using namespace std;
const int maxn=577;

bool dp[maxn][maxn];
int N,K,V[maxn],cv=0;
int main(){
	scanf("%d%d",&N,&K);
	dp[0][0]=true;
	for(int i=0;i<N;++i){
		int t;scanf("%d",&t);
		for(int i=K;i>=0;--i) for(int j=K;j>=0;--j) if(dp[i][j]){
			if(i+t<=K) dp[i+t][j]=true;
			if(j+t<=K) dp[i][j+t]=true;
		}
	}
	for(int i=0;i<=K;++i) if(dp[i][K-i]) V[cv++]=i;
	printf("%d\n%d",cv,V[0]);
	for(int i=1;i<cv;++i) printf(" %d",V[i]);
	printf("\n");
}
