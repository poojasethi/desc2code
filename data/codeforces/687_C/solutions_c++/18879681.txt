#include<iostream>
#include<vector>
using namespace std;
int n,k,c[555];
vector<int>ans;
bool dp[555][555];
int main(){
	cin>>n>>k;
	for(int i=0;i<n;i++){
		cin>>c[i];
	}
	dp[0][0]=true;
	for(int i=0;i<n;i++){
		for(int j=k;j>=c[i];j--){
			for(int l=j;l>=0;l--){
				dp[j][l]|=dp[j-c[i]][l];
				if(l>=c[i])dp[j][l]|=dp[j-c[i]][l-c[i]];
			}
		}
	}
	for(int i=0;i<=k;i++){
		if (dp[k][i])ans.push_back(i);
	}
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)cout<<ans[i]<<' ';
	return 0;
}
