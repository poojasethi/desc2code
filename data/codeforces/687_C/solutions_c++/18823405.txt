#include<cstdio>
#define maxn 510
bool f[maxn][maxn][maxn];
int n,lim,a[maxn];
int main()
{
	scanf("%d%d",&n,&lim);
	for(int i=1; i<=n; i++) scanf("%d",&a[i]);
	f[0][0][0]=true;
	for(int i=1; i<=n; i++)
		for(int j=0; j<=lim; j++)
			for(int k=0; k<=j; k++)
			{
				//not put
				f[i][j][k]|=f[i-1][j][k];
				//put out
				if(j>=a[i] && j-a[i]>=k)
					f[i][j][k]|=f[i-1][j-a[i]][k];
				//put in
				if(k>=a[i])
					f[i][j][k]|=f[i-1][j-a[i]][k-a[i]];
			}
	int cnt=0;
	for(int i=0; i<=lim; i++)
		cnt+=f[n][lim][i];
	printf("%d\n",cnt);
	for(int i=0; i<=lim; i++)
		if(f[n][lim][i]) printf("%d ",i);
	return 0;
}
