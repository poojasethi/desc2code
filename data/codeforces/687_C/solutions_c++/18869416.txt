#include<cstdio>
#include<cstring>
using namespace std;
int n,k,i,j,l,tot,c[510];
bool f[510],h[510][510];
int main()
{
	scanf("%d%d",&n,&k);
	for (i=1;i<=n;i++) scanf("%d",&c[i]);
	f[0]=1;h[0][0]=1;
	for (i=1;i<=n;i++)
	{
		for (j=k;j>=c[i];j--)
		{
			if (f[j-c[i]])
			{
				f[j]=1;
				for (l=0;l<=j-c[i];l++)
				if (h[j-c[i]][l])
				{
					h[j][l]=1;
					h[j][l+c[i]]=1;
				}
			}
		}
	}
    for (i=0;i<=k;i++) if (h[k][i]) tot++;
    printf("%d\n",tot);
    for (i=0;i<=k;i++)
    	if (h[k][i])
    		printf("%d ",i);
    printf("\n");
    return 0;
}
