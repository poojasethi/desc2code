#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <bitset>

using namespace std;

int n,m;
bitset<501> dp[501];

int main(){
	scanf("%d%d",&n,&m);
	dp[m][0]=1;
	for(int i=0,x;i<n;i++){
		scanf("%d",&x);
		for(int j=0;j<=m-x;j++)
			dp[j]|=dp[j+x]|(dp[j+x]<<x);
	}
	printf("%d\n",dp[0].count());
	for(int i=0;i<=m;i++)
		if(dp[0][i])
			printf("%d ",i);
			
			
	return 0;
}
