#include<algorithm>
#include<cstdio>
#include<vector>
using namespace std;
bool P[505][505],D[505];
vector<int>ans;
int main()
{
	int n,k,i,j,x,y;P[0][0]=true;
	scanf("%d%d",&n,&k);
	for(i=0;i<n;++i)
	{
		scanf("%d",&j);
		for(x=k;x>=0;--x)for(y=k;y>=0;--y)
		{
			if(x>=j&&P[x-j][y]==true)P[x][y]=true;
			if(y>=j&&P[x][y-j]==true)P[x][y]=true;
		}
	}
	for(i=0;i<=k;++i)
	{
		j=k-i;
		if(P[i][j]==true||P[j][i]==true)D[i]=D[j]=true;
	}
	for(i=0;i<=k;++i)if(D[i]==true)ans.push_back(i);
	j=ans.size();
	printf("%d\n",j);
	for(i=0;i<j;++i)printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
