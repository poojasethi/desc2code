#include<bits/stdc++.h>
using namespace std;
const int maxn=501;
int n,k,a[maxn];
bitset<maxn> f[maxn][maxn];
int main(){
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	f[0][0][0]=1;
	for(int i=1;i<=n;i++)
	 for(int j=0;j<=k;j++){
	  f[i][j]|=f[i-1][j];
	  if(j-a[i]>=0) f[i][j]|=(f[i-1][j-a[i]]<<a[i])|f[i-1][j-a[i]];
	 }
	int ans=0;
	//cout<<f[1][7][7]<<endl;
	for(int i=0;i<=k;i++)
	 if(f[n][k][i]) ans++;
	printf("%d\n",ans);
	for(int i=0;i<k;i++)
	 if(f[n][k][i])printf("%d ",i);
	printf("%d\n",k);
	return 0;
}
