#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<queue>
#include<cmath>
#define LL long long
using namespace std;
int n,k,c,a[505],d,v[505][505],b[505];
int main(){
	scanf("%d%d",&n,&k);
	v[0][0]=1;
	a[0]=1;
	for (int i=1;i<=n;i++){
		scanf("%d",&c);
		for (int j=k-c;j>=0;j--)
			if (a[j]){
				a[j+c]=1;
				for (int k=j;k>=0;k--){
					if (v[j][k]) v[j+c][k+c]=1,v[j+c][k]=1;
				}
			}
	}
	for (int i=0;i<=k;i++)
		if (v[k][i]) d++;
	printf("%d\n0",d);
	for (int i=1;i<=k;i++)
		if (v[k][i]) printf(" %d",i);
	printf("\n");
    return 0;
}
