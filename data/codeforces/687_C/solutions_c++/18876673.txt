#include<cstdio>
#include<cstring>
#include<algorithm>
#define f(x,y,z) for(int x=y;x<=z;x++)
#define fd(x,y,z) for(int x=y;x>=z;x--)
int a,n,k,tot;
bool b[505][505];
int main()
{
	scanf("%d%d",&n,&k);
	b[0][0]=1;
	f(i,1,n)
	{
		scanf("%d",&a);
		if(a==0)continue;
		fd(r,k,1)fd(s,k,0)
		{
			if(r-a>=0)
				if(b[r-a][s]||(s-a>=0 && b[r-a][s-a]))b[r][s]=1;
			//printf("%d %d %d\n",r,s,b[r][s]);
		}
	}
	f(i,0,k)if(b[k][i])tot++;
	printf("%d\n",tot);
	f(i,0,k)if(b[k][i])printf("%d ",i);
	return 0;
}
