#include<bits/stdc++.h>
using namespace std;
int c[1000];
bool ff[505][505][505];
bool ok[505][505][505];
int f(int i,int j,int k)
{
    if(j<0 || k<0) return 0;
    if(i==0) return j==k && j==0;
    if(ok[i][j][k]++) return ff[i][j][k];
    return ff[i][j][k]=f(i-1,j-c[i],k-c[i])|f(i-1,j,k)|f(i-1,j,k-c[i]);
}
vector<int> ans;
int main()
{
    int n,k;
    scanf("%d %d",&n,&k);
    for(int i=1;i<=n;i++) scanf("%d",&c[i]);
    for(int i=0;i<=k;i++) if(f(n,i,k)==true) ans.push_back(i);
    printf("%d\n",ans.size());
    for(int i=0;i<ans.size();i++) printf("%d ",ans[i]);
}
