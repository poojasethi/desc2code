#include <bits/stdc++.h>

#define NN 510

using namespace std;

int a[NN];
bitset<NN> can[NN][NN];

int main (){
	
	int N, k, i, j;
	scanf("%d %d",&N,&k);
	for(i=1; i<=N; i++){
		scanf("%d",a+i);
	}
	
	can[0][0].set(0);
	for(i=1; i<=N; i++){
		can[i][0].set(0);
		for(j=1; j<=k; j++){
			can[i][j] = can[i-1][j];
			if(a[i]<=j){
				can[i][j] |= can[i-1][j-a[i]];
				can[i][j] |= (can[i-1][j-a[i]]<<a[i]);
			}
		}
	}
	
	printf("%d\n",can[N][k].count());
	for(i=0; i<=k; i++){
		if(can[N][k][i]==1) printf("%d ",i);
	}
		
	return 0;
}
