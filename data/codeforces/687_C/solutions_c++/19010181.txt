#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

bool f[510][510];
int a[510],n,m,ans[510];

int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d",a+i);
	f[0][0]=1;
	for(int i=1;i<=n;i++)
	 for(int j=m;j>=a[i];j--)
	  for(int l=0;l<=j;l++){
	  	f[j][l]|=f[j-a[i]][l];
	  	if(l>=a[i]) f[j][l]|=f[j-a[i]][l-a[i]];
	  }
	for(int i=0;i<=m;i++)
	if(f[m][i]) ans[++ans[0]]=i;
	printf("%d\n",ans[0]);
	for(int i=1;i<=ans[0];i++)
	printf("%d ",ans[i]);
	return 0;
}
