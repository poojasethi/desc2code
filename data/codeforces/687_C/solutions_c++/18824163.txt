#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cmath>
using namespace std;
const int N = 1600;
int f[N][N];
int n,k;
main()
{
    f[0][0] = 1;
    scanf("%d%d", &n, &k);
    for(int t = 0; t < n; t++)
    {
        int x;
        scanf("%d", &x);
        for(int i = k; i >= 0; i--)
        for(int j = k; j >= 0; j--)
            if(f[i][j] == 1) f[i + x][j] = f[i][j + x] = 1;
    }
    int sum = 0;
    for(int i = 0; i <= k; i++)
        sum += f[i][k-i];
    printf("%d\n", sum);
    for(int i = 0; i <= k; i++)
        if(f[i][k-i]) printf("%d ",i);
}
