#include<cstdio>
void read(int&x)
{
	x=0;char c;
	do{c=getchar();}while(c<'0'||c>'9');
	do{x=x*10+c-'0';c=getchar();}while(c>='0'&&c<='9');
}
#define MAX 500
int f[MAX+5][MAX+5];
int main()
{
	int n,k,x,i,j,ans=0;
	read(n);read(k);
	f[0][0]=1;
	while(n--)
	{
		read(x);
		for(i=k-x;i>=0;i--)for(j=0;j<=i;j++)if(f[i][j])
		{
			f[i+x][j]=1;
			f[i+x][j+x]=1;
		}
	}
	for(i=0;i<=k;i++)if(f[k][i])ans++;
	printf("%d\n",ans);
	for(i=0;i<=k;i++)if(f[k][i])printf("%d ",i);
}
