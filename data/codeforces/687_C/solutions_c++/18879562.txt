#include <iostream>
#include <cstring>
using namespace std;
int n,k,c[505];
bool dpa[2][505][505];

int main(){
	ios::sync_with_stdio(false);
	cin>>n>>k;
	for(int i=1;i<=n;i++)cin>>c[i];
	dpa[0][0][0]=true;
	for(int l=1;l<=n;l++){
		memset(dpa[l&1],0,sizeof(dpa[l&1]));
		for(int i=0;i<=k;i++)for(int j=0;j<=k;j++)if(dpa[(l-1)&1][i][j]){
			dpa[l&1][i][j]=true;
			if(i+c[l]<=k)dpa[l&1][i+c[l]][j]=dpa[l&1][i+c[l]][j+c[l]]=true;
		}
	}
	int cnt=0,l=n&1;
	for(int i=0;i<=k;i++)if(dpa[l][k][i])cnt++;
	cout<<cnt<<endl;
	for(int i=0;i<=k;i++)if(dpa[l][k][i])cout<<i<<" ";
	return 0;
}
