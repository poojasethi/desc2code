#include<bits/stdc++.h>
using namespace std ;


int n ,k,arr[510] ;

bool dp[510][510] ;

int main()
{
	cin>>n>>k ;
	for(int i = 0;i<n;i++) scanf("%d",&arr[i]) ;
	dp[0][0] = 1;
	for(int i = 0;i<n;i++)
	{
		int x =arr[i] ;
		for(int j = k - x ; j>=0;j--)
		{
			for(int l = k;l>=0;l--)
			{
				if(dp[j][l])
				{
					dp[j+x][l] = 1 ;
					if(l+x<=k) dp[j+x][l+x] = 1 ;
				}
			}
		}
	}
	int m = 0 ;for(int i = 0;i<=k;i++) if(dp[k][i]) m++ ;
	cout<<m<<endl ;
	for(int i = 0;i<=k;i++)if(dp[k][i])cout<<i<<" ";
}
