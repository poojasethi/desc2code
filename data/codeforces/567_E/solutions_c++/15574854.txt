#include<bits/stdc++.h>
#define ll long long

using namespace std;

const int MAXN=1e5+10;
ll INF=1e8,MOD=98765431LL;
long long dis[2][MAXN];
vector<pair<ll,int> >a[2][MAXN];
set<pair<ll,int> >s;
int parent[MAXN];
vector<pair<pair<int,int>,int > >e;
ll dp[2][MAXN];
int n,m;
map<pair<int,int>, int>mp;

int dj(int start,int w){
	for(int i=1;i<n+1;i++)
		dis[w][i]=INF*INF,s.insert(make_pair(INF*INF,i));
	s.erase(make_pair(INF*INF,start));
	s.insert(make_pair(0,start));
	dis[w][start]=0;
	dp[w][start]=1;
	while(s.size()){
		int v=s.begin()->second;
		ll k=s.begin()->first;
		s.erase(s.begin());
		dis[w][v]=k;
		for(int i=0;i<a[w][v].size();i++){
			if(dis[w][v]+a[w][v][i].second<dis[w][a[w][v][i].first]){
				dp[w][a[w][v][i].first]=dp[w][v];dp[w][a[w][v][i].first]%=MOD;
				s.erase(make_pair(dis[w][a[w][v][i].first],a[w][v][i].first));
				dis[w][a[w][v][i].first]=dis[w][v]+a[w][v][i].second;
				s.insert(make_pair(dis[w][a[w][v][i].first],a[w][v][i].first));
				parent[a[w][v][i].first]=v;
			}
			else if(dis[w][v]+a[w][v][i].second==dis[w][a[w][v][i].first])
				dp[w][a[w][v][i].first]+=dp[w][v];dp[w][a[w][v][i].first]%=MOD;
		}
	}
	return 0;	
}

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);cout.tie(0);
	ll s,t;
	cin>>n>>m>>s>>t;
	int x,y,l;
	for(int i=0;i<m;i++){
		cin>>x>>y>>l;
		a[0][x].push_back(make_pair(y,l));
		a[1][y].push_back(make_pair(x,l));
		e.push_back(make_pair(make_pair(x,y),l));
	}
	dj(s,0);
	dj(t,1);
	dp[0][s]=dp[1][t]=1;
	ll mn=dis[0][t];
	for(int i=0;i<m;i++){
		x=e[i].first.first;
		y=e[i].first.second;
		int a=dis[0][e[i].first.first];
		int b=dis[1][e[i].first.second];
		ll k=e[i].second,s=0;
		s=dis[0][e[i].first.first]+dis[1][e[i].first.second]+k;
		//cerr<<i<<" : "<<dp[0][x]<<' '<<dp[1][y]<<endl;
		if(s==mn and (dp[0][x]*dp[1][y])%MOD==dp[0][t])
			cout<<"YES"<<endl;
		else{
			k-=s-mn+1;
			if(k<1)
				cout<<"NO"<<endl;
			else
				cout<<"CAN "<<s-mn+1<<endl;
		}
	}
	
	return 0;
}
