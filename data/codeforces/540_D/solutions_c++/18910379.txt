#include <bits/stdc++.h>
using namespace std;

const int maxn = 105;
int r, s, p, m;
double sum, a, b, c, d[maxn][maxn][maxn];

int main() {
	scanf("%d%d%d", &r, &s, &p);
	memset(d, 0, sizeof(d));
	d[r][s][p] = 1;
	for(int i = r; i > 0; i--) {
		for(int j = s; j > 0; j--) {
			for(int k = p; k > 0; k--) {
				sum = i * j + j * k  + k * i;
				d[i][j-1][k] += i * j / sum * d[i][j][k];
				d[i][j][k-1] += j * k / sum * d[i][j][k];
				d[i-1][j][k] += i * k / sum * d[i][j][k]; 
			}
		}
	}
	a = b = c = 0;
	m = max(max(r, s), p);
	for(int i = 1; i <= m; i++) {
		for(int j = 1; j <= m; j++) {
			a += d[i][j][0];
			b += d[0][i][j];
			c += d[i][0][j];
		}
	}
	printf("%.10f %.10f %.10f\n", a, b, c);
	return 0;
}
