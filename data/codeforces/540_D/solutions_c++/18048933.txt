#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <memory.h>

using namespace std;

double dp[110][110][110];

int main (){
	
	int r,s,p,n;
	scanf("%d %d %d",&r,&s,&p);
	dp[r][s][p]=1.0;
	
	int i,j,k;
	for(i=r; i>=0; i--){
		for(j=s; j>=0; j--){
			for(k=p; k>=0; k--){
				n=i*j+i*k+j*k;
				if(i && j) dp[i][j-1][k]+=dp[i][j][k]*i*j/n;
				if(i && k) dp[i-1][j][k]+=dp[i][j][k]*i*k/n;
				if(j && k) dp[i][j][k-1]+=dp[i][j][k]*j*k/n;
			}
		}
	}
	
	double a,b,c;
	a=b=c=0.0;
	
	for(i=1; i<=r; i++) a+=dp[i][0][0];
	for(i=1; i<=s; i++) b+=dp[0][i][0];
	for(i=1; i<=p; i++) c+=dp[0][0][i];
	
	printf("%.12lf %.12lf %.12lf\n",a,b,c);
					
	return 0;
}
