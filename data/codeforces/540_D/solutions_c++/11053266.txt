#include<bits/stdc++.h>

using namespace std;

double DP[105][105][105][4];

double f(int r,int s,int p,int w){

	double &t=DP[r][s][p][w];

	if(t!=-1)
		return t;

	if(!r and !s and !p)
		return 0;
	if(r and !s and !p)
		return (w==1);
	if(!r and s and !p)
		return (w==2);
	if(!r and !s and p)
		return (w==3);

	int sum=r*s+s*p+p*r;

	t=0;

	if(r>0 and p>0) t+=f(r-1,s,p,w)*r*p/sum;
	if(r>0 and s>0) t+=f(r,s-1,p,w)*r*s/sum;
	if(s>0 and p>0) t+=f(r,s,p-1,w)*s*p/sum;

	return t;
}

int main(){

	int r,s,p,i,j,k,q;

	cin>>r>>s>>p;

	for(i=100 ; i>=0 ; i--)
		for(j=100 ; j>=0 ; j--)
			for(k=100 ; k>=0 ; k--)
				for(q=0 ; q<=3 ; q++)
					DP[i][j][k][q]=-1;

	printf("%.10lf %.10lf %.10lf\n",f(r,s,p,1),f(r,s,p,2),f(r,s,p,3));

	return 0;
}
