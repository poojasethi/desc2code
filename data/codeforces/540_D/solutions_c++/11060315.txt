#include <iostream> 
#include <cmath>
#include <cstdio>
#include <algorithm>
#include <iomanip>
#include <cstring>
using namespace std; 
#define rep(i, n) for(int i=0;i<=n;i++)
int vis[101][101][101];
double p[101][101][101];

double f(int i, int j, int k){
    if(vis[i][j][k] == 1)
        return p[i][j][k]; 

    if(i * j == 0 )return 0; 
    if( k == 0) return 1;
    vis[i][j][k] = 1;   

    double a, b, c, s; 
    a = i*j, b = j*k, c = k*i; 
    s = a + b + c; 
    return p[i][j][k] = f(i,j-1,k)*a/s  + f(i,j,k-1)*b/s + f(i-1,j,k)*c/s;
}

int main(){
    int r, s, p; 
    cin >> r >> s >> p; 

    cout << setprecision(15) << f(r,s,p) << " " << f(s,p,r) << " " << f(p, r, s) << endl;
}
