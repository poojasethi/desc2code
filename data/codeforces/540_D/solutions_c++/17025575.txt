#include<iostream>
using namespace std;
double r[101][101][101];
double s[101][101][101];
double p[101][101][101];
int main()
{
	int R,S,P;
	cin>>R>>S>>P;
	cout.precision(10);
	for(int i=1;i<=100;i++)
	{
		for(int j=0;j<=100;j++)
		{
			r[i][j][0] =1;
			s[0][i][j] =1;
			p[j][0][i] =1;

		}
	}
	for(int i=1;i<=R;i++)
	{
		for(int j=1;j<=S;j++)
		{
			for(int k=1;k<=P;k++)
			{
				r[i][j][k] = ( (j*k*r[i][j][k-1] + i*k*r[i-1][j][k] + i*j*r[i][j-1][k]) )/(j*k + i*k + i*j);
				s[i][j][k] = ( (j*k*s[i][j][k-1] + i*k*s[i-1][j][k] + i*j*s[i][j-1][k]) )/(j*k + i*k + i*j);
				p[i][j][k] = ( (j*k*p[i][j][k-1] + i*k*p[i-1][j][k] + i*j*p[i][j-1][k]) )/(j*k + i*k + i*j);
			}
		}
	}
	cout<<r[R][S][P]<<" "<<s[R][S][P]<<" "<<p[R][S][P];
}
