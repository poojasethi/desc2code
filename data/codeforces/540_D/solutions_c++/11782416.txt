#include<bits/stdc++.h>

using namespace std;

double dp[101][101][101];
bool visited[101][101][101];

double rec(int r,int s,int p )
{

	if( visited[r][s][p] ) return dp[r][s][p];

	visited[r][s][p] = true;

	if( r== 0 ) return 1.0;
	if( s== 0 ) return 0.0;
	if( p== 0 ) return 0.0;


	double a = r*s ,  b = s*p  , c = p*r;

	double t = a +b + c ;

	return dp[r][s][p] = a/t*rec(r,s-1,p) + b/t*rec(r,s,p-1) + c/t*rec(r-1,s,p);



}


int main()
{


	int r,s,p;
	cin >> r >> s >> p ;

	
	
	
	printf("%.12f %.12f %.12f\n", rec(p,r,s),rec(r,s,p) , rec(s,p,r) );


	return 0;



}
