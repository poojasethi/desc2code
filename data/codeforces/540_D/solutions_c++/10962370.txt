#include <bits/stdc++.h>
using namespace std;
#define pb push_back
typedef long long ll;
typedef pair<int,ll> P;
double dp[105][105][105];
int r,c,p;
int main(){
	cin >> r >> c >> p;
	dp[r][c][p] = 1.0;
	for(int i=r;i>=0;i--){
		for(int j=c;j>=0;j--){
			for(int k=p;k>=0;k--){
				if((i==0)+ (j==0)+ (k==0) >= 2) continue;
				double d = (double)( i*j+j*k+k*i );
				if(j) dp[i][j-1][k] += dp[i][j][k] * (double)(i*j) / d;
				if(k) dp[i][j][k-1] += dp[i][j][k]* (double)(j*k) / d;
				if(i) dp[i-1][j][k] += dp[i][j][k] * (double)(k*i) / d;
			}
		}
	}
	double R=0,C=0,P=0;
	for(int i=1;i<105;i++){
		R+=dp[i][0][0];
		C+=dp[0][i][0];
		P+=dp[0][0][i];
	}
	cout << fixed << setprecision(12) << R << " " << C << " " << P << endl;
}
