#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long r,s,p,i,j,k;
	double dp[101][101][101],Pp=0,Pr=0,Ps=0,sum;
	memset(dp,0,sizeof(dp));
	cin>>r>>s>>p;
	dp[r][s][p]=1;
	for(i=r;i>=0;i--)
	{
		for(j=s;j>=0;j--)
		{
			for(k=p;k>=0;k--)
			{
				sum=(i*k)+(j*k)+(i*j);
				if(i!=0 && k!=0)
				{
					dp[i-1][j][k]+=(((i*k)/sum)*dp[i][j][k]);
				}
				if(j!=0 && i!=0)
				{
					dp[i][j-1][k]+=(((j*i)/sum)*dp[i][j][k]);
				}
				if(k!=0 && j!=0)
				{
					dp[i][j][k-1]+=(((j*k)/sum)*dp[i][j][k]);
				}
			}
		}
	}
	for(i=1;i<=r;i++)
	{
		Pr+=dp[i][0][0];
	}
	for(i=1;i<=s;i++)
	{
		Ps+=dp[0][i][0];
	}
	for(i=1;i<=p;i++)
	{
		Pp+=dp[0][0][i];
	}
	printf("%.12lf %.12lf %.12lf\n",Pr,Ps,Pp);
	return 0;
}
