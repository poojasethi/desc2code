#include<bits/stdc++.h>
using namespace std;
long double dp[101][101][101];
typedef long double ld;
int main()
{
	int R,S,P;
	cin>>R>>S>>P;
	
	dp[R][P][S]=1;
	
	for(int r=R;r>=0;r--)
	{
		for(int p=P;p>=0;p--)
		{
			for(int s=S;s>=0 ;s--)
			{
				if((r==0 &&s==0 )||(r==0 && p==0) ||(s==0 && p==0))continue;
				ld curr=dp[r][p][s];
				ld totalWays=(r*p + s*p  +  r*s);
				if(r)	dp[r-1][p][s]+= curr*(r*p)/totalWays;
				if(s)	dp[r][p][s-1]+=curr*(s*r)/totalWays;
				if(p)	dp[r][p-1][s]+=curr*(p*s)/totalWays;
			}
		}
	}
	ld ROCK=0,SC=0,PAP=0;
	for(int r=0;r<=R;r++)
		ROCK+=dp[r][0][0];
	for(int p=0;p<=P;p++)
		PAP+=dp[0][p][0];
	for(int s=0;s<=S;s++)
		SC+=dp[0][0][s];
	printf("%.12f %.12f %.12f\n",(double)ROCK,(double)SC,(double)PAP);
	return 0;
}
