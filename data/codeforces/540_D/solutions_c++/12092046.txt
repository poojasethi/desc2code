#include <cstdio>
int a, b, c;
double ans, f[110][110][110];
int main(){
	scanf("%d%d%d", &a, &b, &c);
	f[a][b][c]=1;
	for(int i=a; i>=0; i--)
		for(int j=b; j>=0; j--)
			for(int k=c; k>=0; k--)
				if(f[i][j][k]>0)
				{
					if(!i&&!j || !i&&!k || !j&&!k) continue;
					if(i) f[i-1][j][k]+=f[i][j][k]*(i*k)/(i*k+j*k+i*j);
					if(j) f[i][j-1][k]+=f[i][j][k]*(i*j)/(i*k+j*k+i*j);
					if(k) f[i][j][k-1]+=f[i][j][k]*(j*k)/(i*k+j*k+i*j);
				}
	ans=0.0; for(int i=1; i<=a; i++) ans+=f[i][0][0]; printf("%.12lf ", ans);
	ans=0.0; for(int i=1; i<=b; i++) ans+=f[0][i][0]; printf("%.12lf ", ans);
	ans=0.0; for(int i=1; i<=c; i++) ans+=f[0][0][i]; printf("%.12lf\n", ans);
}
