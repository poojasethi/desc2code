#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
bool mark[110][110][110][3];
double f[110][110][110][3];
double dfs(int x,int y,int z,int k){
	if (mark[x][y][z][k]) return f[x][y][z][k];
	double &u=f[x][y][z][k];
	mark[x][y][z][k]=1;
	if (x&&!y&&!z) if (k==0)return u=1;else return u=0;
	if (y&&!x&&!z) if (k==1)return u=1;else return u=0;
	if (z&&!y&&!x) if (k==2)return u=1;else return u=0;
	
	double sum=x*1ll*y+y*z+z*x;

	if (x&&y) u+=dfs(x,y-1,z,k)*x*y/sum;
	if (y&&z) u+=dfs(x,y,z-1,k)*y*z/sum;
	if (x&&z) u+=dfs(x-1,y,z,k)*x*z/sum;
	return u;
}	
int main(){
	int x,y,z;
	scanf("%d%d%d",&x,&y,&z);
	printf("%.12lf %.12lf %.12lf\n",dfs(x,y,z,0),dfs(x,y,z,1),dfs(x,y,z,2));
	return 0;
}
