#include<cstdio>
#include<cstring>
#define MAX 110

double dp[MAX][MAX][MAX];
int r, s, p;

double f(int r, int s, int p){
    if(!p)  return dp[r][s][p] = 1;
    if(!(r && s))   return dp[r][s][p] = 0;
    if(dp[r][s][p]) return dp[r][s][p];
    int a = r * s;
    int b = r * p;
    int c = s * p;
    int d = a + b + c;
    return dp[r][s][p] = a * 1.0 / d * f(r, s - 1, p) + b * 1.0 / d * f(r - 1, s, p) + c * 1.0 / d * f(r, s, p - 1);
}

int main(){
    int i, j, k;
    scanf("%d%d%d", &r, &s, &p);
    printf("%.12lf", f(r, s, p));
    memset(dp, 0, sizeof(dp));
    printf(" %.12lf", f(s, p, r));
    memset(dp, 0, sizeof(dp));
    printf(" %.12lf\n", f(p, r, s));
}
