#include <cstdio>
#include <cstring>
using namespace std;

int r, s, p;
double A[105][105][105];

double cal(int r, int s, int p){
	if (A[r][s][p] >= 0) return A[r][s][p];
	if (r == 0) return 0;
	if (s == 0 && p > 0) return 0; 
	if (p == 0) return 1.0;
	double a = r*s, b = s*p, c = p*r;
	A[r][s][p] = (a * cal(r, s - 1, p) + b * cal(r, s, p - 1) + c * cal(r - 1, s, p)) / (a + b + c); 
	return A[r][s][p];
}

int main(void){
	memset(A, -1, sizeof(A));
	scanf("%d%d%d", &r, &s, &p);
	printf("%.12lf %.12lf %.12lf", cal(r, s, p), cal(s, p, r), cal(p, r, s));
	return 0;
}
