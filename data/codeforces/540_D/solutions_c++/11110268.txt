#include <iostream>
#include <bits/stdc++.h>
using namespace std;
double dp[102][102][102]={0};
int main(){
    int r,s,p;
    cin >> r >> s >> p;
    int i,j,k;
    dp[r][s][p]=1;
    for(i=r;i>=0;--i){
        for(j=s;j>=0;--j){
            for(k=p;k>=0;k--){
                double a=i*j+j*k+k*i;
                if(j>0 && k>0)dp[i][j][k-1]+=dp[i][j][k]*1.0*j*(k)/a;
                if(i>0 && j>0)dp[i][j-1][k]+=dp[i][j][k]*1.0*i*(j)/a;
                if(k>0 && i>0)dp[i-1][j][k]+=dp[i][j][k]*1.0*(i)*k/a;
            }
        }
    }
    double a=0,b=0,c=0;
    for(i=1;i<=100;++i){
        a+=dp[0][0][i];
        b+=dp[0][i][0];
        c+=dp[i][0][0];
    }

    printf("%.9f %.9f %.9f",c,b,a);
    return 0;


}
