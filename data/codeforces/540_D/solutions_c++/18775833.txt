#include <cstdio>
double f[110][110][110][3];
int i,j,k,l,r,s,p,x;
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d%d%d\n",&r,&s,&p);
	for (i=0;i<=r;++i)
		for (j=0;j<=s;++j)
			for (k=0;k<=p;++k)
			if (!i && !j) f[i][j][k][2]=1;
			else if (!i && !k) f[i][j][k][1]=1;
			else if (!j && !k) f[i][j][k][0]=1;
			else
			{
				x=i*j+j*k+i*k;//(i+j+k)*(i+j+k-1)>>1;
				for (l=0;l<3;++l)
				{
					if (i&&j) f[i][j][k][l]+=f[i][j-1][k][l]*i*j/x;
					if (i&&k) f[i][j][k][l]+=f[i-1][j][k][l]*i*k/x;
					if (j&&k) f[i][j][k][l]+=f[i][j][k-1][l]*j*k/x;
				}
			}
	for (i=0;i<3;++i) printf("%.12f ",f[r][s][p][i]);
}
