#include<stdio.h>
double memo[101][101][101];
int chk[101][101][101];
double f(int r,int s,int p){
	if(chk[r][s][p]==1)return memo[r][s][p];
	chk[r][s][p]=1;
	if(r==0)return 0.0;
	if(s==0)return 0.0;
	if(p==0)return 1.0;
	double a=r*s,b=s*p,c=p*r;
	double x=a+b+c;
	return memo[r][s][p]=a/x*f(r,s-1,p)+b/x*f(r,s,p-1)+c/x*f(r-1,s,p);
}
int main(){
	int r,s,p;
	scanf("%d%d%d",&r,&s,&p);
	printf("%.12lf %.12lf %.12lf",f(r,s,p),f(s,p,r),f(p,r,s));
	return 0;
}
