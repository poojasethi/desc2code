#include <bits/stdc++.h>
using namespace std;
int r,s,p;
double d[101][101][101];
int main()
{
scanf("%d%d%d",&r,&s,&p);
d[r][s][p]=1;
for(int i=r;i>=0;i--)
    for(int j=s;j>=0;j--)
        for(int k=p;k>=0;k--)
{
    int tot=i*j+i*k+j*k;
    if(tot&&i)
        d[i-1][j][k]+=d[i][j][k]*double(i*k)/double(tot);
    if(tot && j)
        d[i][j-1][k]+=d[i][j][k]*double(j*i)/double(tot);
    if(tot && k)
        d[i][j][k-1]+=d[i][j][k]*double(j*k)/double(tot);
}
double sol1=0,sol2=0,sol3=0;
for(int j=r;j>=1;j--)
    sol1+=d[j][0][0];
for(int j=s;j>=1;j--)
    sol2+=d[0][j][0];
for(int j=p;j>=1;j--)
    sol3+=d[0][0][j];
cout<<fixed<<setprecision(9)<<sol1<<" "<<sol2<<" "<<sol3;
return 0;
}
