#include<bits/stdc++.h>
using namespace std;
double dp[101][101][101];
double dfs(int i,int j,int k){
    if(k==0) return dp[i][j][k]=1;
    if(j==0||i==0) return dp[i][j][k]=0;
    if(dp[i][j][k]!=0) return dp[i][j][k];
    dp[i][j][k]=1.*i*j/(i*j+j*k+k*i)*dfs(i,j-1,k)+1.*k*i/(i*j+j*k+k*i)*dfs(i-1,j,k)+1.*j*k/(i*j+j*k+k*i)*dfs(i,j,k-1);
    return dp[i][j][k];
}
int main()
{
    int i,j,r,s,p,t;
    cin>>r>>s>>p;
    printf("%.12f ",dfs(r,s,p));
    t=r;r=s;s=p;p=t;
    memset(dp,0,sizeof dp);
    printf("%.12f ",dfs(r,s,p));
    t=r;r=s;s=p;p=t;
    memset(dp,0,sizeof dp);
    printf("%.12f\n",dfs(r,s,p));
}
