#include <bits/stdc++.h>

using namespace std;

int const N=3e5+5;

double d[111][111][111];


int main()
{
int n,m,k;
cin>>n>>m>>k;
d[n][m][k]=1;
for (int i=n;i>=0;i--)
 for (int j=m;j>=0;j--)
  for (int z=k;z>=0;z--)
   {
     int t=i*j+i*z+j*z;
     if (i>0 && j>0) d[i][j-1][z]+=d[i][j][z]*i*j/t;
     if (i>0 && z>0) d[i-1][j][z]+=d[i][j][z]*i*z/t;
     if (j>0 && z>0) d[i][j][z-1]+=d[i][j][z]*j*z/t;
   }


//poisk(n,m,k);
double t=0;
for (int i=1;i<=n;i++) t+=d[i][0][0];
printf("%.9f ",t);
t=0;
for (int i=1;i<=m;i++) t+=d[0][i][0];
printf("%.9f ",t);
t=0;
for (int i=1;i<=k;i++) t+=d[0][0][i];
printf("%.9f ",t);


}
