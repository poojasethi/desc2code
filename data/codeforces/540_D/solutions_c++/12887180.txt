//In the Name of God

#include<iostream>
#include<cstdio>

using namespace std;

int r,s,p;
double R,S,P,dp[110][110][110];

int main()
{
  scanf("%d%d%d",&r,&s,&p);
  dp[r][s][p]=1;
  for(int i=r;i>=0;i--)
    for(int j=s;j>=0;j--)
      for(int k=p;k>=0;k--)
	{
	  if(i*j+i*(k+1)+j*(k+1))
	    dp[i][j][k]+=(dp[i][j][k+1]*(k+1)*j/(i*j+i*(k+1)+j*(k+1)));
	  if(i*(j+1)+i*k+(j+1)*k)
	    dp[i][j][k]+=(dp[i][j+1][k]*i*(j+1)/(i*(j+1)+i*k+(j+1)*k));
	  if((i+1)*j+(i+1)*k+j*k)
	    dp[i][j][k]+=(dp[i+1][j][k]*k*(i+1)/((i+1)*j+(i+1)*k+j*k));
        }
  for(int i=1;i<=r;i++)
    R+=dp[i][0][0];
  for(int i=1;i<=s;i++)
    S+=dp[0][i][0];
  for(int i=1;i<=p;i++)
    P+=dp[0][0][i];
  printf("%.9f %.9f %.9f\n",R,S,P);
  return 0;
}
