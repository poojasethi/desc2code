#include <stdio.h>
double v[128][128][128];
int f[128][128][128];
double go(int a,int b,int c){
  if(a<0||b<0||c<0)return 0;
  double& t=v[a][b][c];
  if (!f[a][b][c]){
    f[a][b][c]=1;
    t=0;
    if(a){
      if(!b&&!c)t=1;
      else{
        double x=a*b,y=b*c,z=c*a,s=x+y+z;
        t=x/s*go(a,b-1,c)+y/s*go(a,b,c-1)+z/s*go(a-1,b,c);
      }
    }
  }
  return t;
}
int main(){
  int r,s,p;
  scanf("%d%d%d",&r,&s,&p);
  printf("%.12lf %.12lf %.12lf",go(r,s,p),go(s,p,r),go(p,r,s));
}
