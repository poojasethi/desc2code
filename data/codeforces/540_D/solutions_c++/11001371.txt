#include <iostream>
#include <cstdio>

double memo[105][105][105];

double dpdp(int R, int S, int P){
    if(R==0) return 0;
    if(S+P==0) return 1;
    double& ret = memo[R][S][P];
    if(ret >= 0) return ret;
    ret = 0;
    if(R&&S) ret += dpdp(R, S-1, P) * R * S;
    if(R&&P) ret += dpdp(R-1, S, P) * R * P;
    if(P&&S) ret += dpdp(R, S, P-1) * P * S;
    ret /= (R*S + P*S + R*P);
    return ret;
}

int main(){
    int r, s, p;
    for(int i=0;i<105;++i)
        for(int j=0;j<105;++j)
            for(int k=0;k<105;++k){
                memo[i][j][k] = -1;
            }
    std::cin >> r >> s >> p;
    printf("%.12lf %.12lf %.12lf\n", dpdp(r, s, p), dpdp(s, p, r), dpdp(p, r, s));
}
