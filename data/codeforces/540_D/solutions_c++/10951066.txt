#include <bits/stdc++.h>
using namespace std;
double p[101][101][101];
int main() {
  int a, b, c;
  scanf("%d %d %d", &a, &b, &c);
  for (int i = 0; i <= 100; i++) {
    p[i][0][0] = 1;
    for (int j = 0; j <= 100; j++) {
      for (int k = 0; k <= 100; k++) {
        if (i + j + k < 2) continue;
        if (i * j + i * k + j * k == 0) continue;
        if (j) p[i][j][k] += 1.0 * i * j / (i * j + i * k + j * k) * p[i][j - 1][k];
        if (k) p[i][j][k] += 1.0 * j * k / (i * j + i * k + j * k) * p[i][j][k - 1];
        if (i) p[i][j][k] += 1.0 * i * k / (i * j + i * k + j * k) * p[i - 1][j][k];
        //printf("%lf\n", p[i][j][k]);
      }
    }
  }
  printf("%.11lf %.11lf %.11lf", p[a][b][c], p[b][c][a], p[c][a][b]);
  return 0;
}
