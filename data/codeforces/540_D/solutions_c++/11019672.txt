#include <bits/stdc++.h>

using namespace std;

int a, b, c;

double d[111][111][111];
double x, y, z;

int main()
{
	cin >> a >> b >> c;
	
	d[a][b][c]++;
	for (int i = a; i >= 0; i--)
	{
		for (int h = b; h >= 0; h--)
		{
			for (int j = c; j >= 0; j--)
			{
				double pos = i*h + h*j + j*i;
				
				if (i > 0 && h > 0)
					d[i][h-1][j] += i*h*d[i][h][j]/pos;
				if (h > 0 && j > 0)
					d[i][h][j-1] += j*h*d[i][h][j]/pos;
				if (i > 0 && j > 0)
					d[i-1][h][j] += i*j*d[i][h][j]/pos;
			}
		}
	}
	
	for (int i = 0; i < 101; i++)
	{
		x += d[i][0][0];
		y += d[0][i][0];
		z += d[0][0][i];
	}
	printf("%.12lf %.12lf %.12lf", x, y, z);
}
