#include<iostream>
#include<stdio.h>

using namespace std;

double pr,ps,pp;
double f[102][102][102][3];


int main(){

int r,s,p,i,j,k,ijk,t;
scanf("%d %d %d",&r,&s,&p);

for(i=1;i<=r;i++)for(j=1;j<=s;j++)f[i][j][0][0]=1;
for(j=1;j<=s;j++)for(k=1;k<=p;k++)f[0][j][k][1]=1;
for(i=1;i<=r;i++)for(k=1;k<=p;k++)f[i][0][k][2]=1;

for(i=1;i<=r;i++){
    for(j=1;j<=s;j++){
        for(k=1;k<=p;k++){
            ijk=i*j+i*k+j*k;
            for(t=0;t<3;t++){
                f[i][j][k][t]=f[i][j-1][k][t]*i*j+f[i][j][k-1][t]*j*k+f[i-1][j][k][t]*i*k;
                f[i][j][k][t]/=ijk;
            }
        }
    }
}

pr=f[r][s][p][0];
ps=f[r][s][p][1];
pp=f[r][s][p][2];

printf("%.12f %.12f %.12f",pr,ps,pp);


}
