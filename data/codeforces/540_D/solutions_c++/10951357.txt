#include<bits/stdc++.h>

using namespace std;
typedef long long ll;
#define N 123456

double dp[102][102][102];

int main()
{
   int r,s,p;
   cin>>r>>s>>p;
   dp[r][s][p]=1.0;

   for (int i=r;i>=0;i--)
   for (int j=s;j>=0;j--)
   for (int k=p;k>=0;k--)
   {
       double tmp=(i+j+k)*(i+j+k-1)/2-(i*(i-1)/2)-j*(j-1)/2-k*(k-1)/2;
       if (j>0&&i>0)  dp[i][j-1][k]+=dp[i][j][k]*i*j/tmp;
       if (j>0&&k>0)  dp[i][j][k-1]+=dp[i][j][k]*j*k/tmp;
       if (i>0&&k>0)  dp[i-1][j][k]+=dp[i][j][k]*k*i/tmp;
   }

   double a,b,c;
   a=b=c=0.0;

   for (int i=1;i<=r;i++)
   a+=dp[i][0][0];

   for (int i=1;i<=s;i++)
   b+=dp[0][i][0];


   for (int i=1;i<=p;i++)
   c+=dp[0][0][i];

   printf("%.9lf %.9lf %.9lf\n",a,b,c);
   return 0;
}
