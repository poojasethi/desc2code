#include<iostream>
#include<cstring>


using namespace std;

double dp[102][102][102];
bool check[102][102][102];


double fun(double x,double y, double z){
   // cout<<x<<" "<<y<<" "<<z<<"\n";

     int a=x,b=y,c=z;

     if(check[a][b][c])
        return dp[a][b][c];

    check[a][b][c]=true;

     if(z==0)
        return dp[a][b][c]=1;
     if(x==0 || y==0)
        return dp[a][b][c]=0;
     double val=(x*y)+(y*z)+(z*x);

     return dp[a][b][c]=((x*y)/val)*fun(x,y-1,z)+((y*z)/val)*fun(x,y,z-1)+((x*z)/val)*fun(x-1,y,z);




}


int main(){
    double s,p,r;
    cin>>r>>s>>p;
    memset(dp,-1,sizeof(dp));
    cout.precision(20);
    cout<<fun(r,s,p)<<" "<<fun(s,p,r)<<" "<<fun(p,r,s);



return 0;
}
