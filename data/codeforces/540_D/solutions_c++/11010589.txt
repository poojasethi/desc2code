#include <bits/stdc++.h>

using namespace std;

double dp[105][105][105];

double solve(int a, int b, int c)
{
    if(!c)  return dp[a][b][c] = 1;
    if(!(a && b))return dp[a][b][c] = 0;
    
    if(dp[a][b][c]) return dp[a][b][c];
    
    return dp[a][b][c] = (a * b * solve(a,b-1,c) + b * c * solve(a,b,c-1) + c * a * solve(a-1,b,c)) / (a*b + b*c + c*a);
}

int main()
{
    ios_base::sync_with_stdio(true);
    int r,s,p;
    scanf("%d %d %d", &r, &s, &p);
    printf("%.12f", solve(r,s,p));
    memset(dp,0,sizeof(dp));
    printf(" %.12f", solve(s,p,r));
    memset(dp,0,sizeof(dp));
    printf(" %.12f\n", solve(p,r,s));
    memset(dp,0,sizeof(dp));
    return 0;
}
