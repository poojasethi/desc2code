#include <bits/stdc++.h>

using namespace std;
#define f(i, n) for (int i = 0; i < (int)(n); i++)
typedef long long ll;
double mem[111][111][111];

double dp(int r,int s,int p){

if(!r)return 0;
if(r && !s && !p) return 1;

double &res=mem[r][s][p];
if(res==res)return res;

double c=0,cc=0,ccc=0;
double sm=r*s+r*p+p*s;
if(s>0 && r>0)c=dp(r,s-1,p);
if(p>0 && s>0)cc=dp(r,s,p-1);
if(p>0 && r>0)ccc=dp(r-1,s,p);
return res=r*s/sm*c + s*p/sm*cc+ p*r/sm*ccc;
}


int main()
{
   memset(mem,-1,sizeof mem);
   int a,b,c;
   cin>>a>>b>>c;
   double r=dp(a,b,c),s=dp(b,c,a), p=dp(c,a,b);

   cout<<fixed<<setprecision(10)<<r<<" "<<s<<" "<<p<<endl;
    return 0;
}
