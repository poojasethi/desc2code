#include <bits/stdc++.h>

using namespace std;

int a, b, c;
double dd[105][105][105];

double slove(int x, int y, int z)
{
    if(x == 0 || z == 0) return 0;
    if( y == 0 ) return 1;
    if(dd[x][y][z] != 0) return dd[x][y][z];
    double p = slove(x, y-1, z);
    double q = slove(x, y, z-1);
    double r = slove(x-1, y, z);
    return dd[x][y][z] = (1.0 * x * y * p + 1.0 * y * z * q + 1.0 * z * x * r) / ( x * y + y * z + z * x);
}

int main()
{
   cin >> a >> b >> c;
   printf("%0.9f ", slove(b, c, a));
   printf("%0.9f ", slove(c, a, b));
   printf("%0.9f ", slove(a, b, c));
}
