#include "stdio.h"
double dp[101][101][101];
int main(){
    int x,y,z;
    scanf("%d %d %d",&x,&y,&z);
    for(int i=1;i<=100;i++){
        for(int j=1;j<=100;j++){
            dp[i][j][0]=1;
            for(int k=1;k<=100;k++) dp[i][j][k]+=(i*j*dp[i][j-1][k]+i*k*dp[i-1][j][k]+j*k*dp[i][j][k-1])/(i*j+i*k+j*k);
        }
    }
    printf("%.10lf %.10lf %.10lf\n",dp[x][y][z],dp[y][z][x],dp[z][x][y]);
}
