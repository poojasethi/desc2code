#define N (1<<7)
#include <bits/stdc++.h>
using namespace std;

int r,s,p;
double R,S,P,dp[N][N][N];

int main()
{
	cin>>r>>s>>p;
	dp[r][s][p]=1;
	for(int i=r;i>=0;i--)
		for(int j=s;j>=0;j--)
			for(int k=p;k>=0;k--)
			{
				if((i==r && j==s && k==p) || !(i+j+k)) continue;
				if(j+k)
					dp[i][j][k]+=dp[i+1][j][k]*(i+1)*k/((i+1)*(j+k)+j*k);
				if(i+k)
					dp[i][j][k]+=dp[i][j+1][k]*(j+1)*i/((j+1)*(i+k)+i*k);
				if(i+j)
					dp[i][j][k]+=dp[i][j][k+1]*j*(k+1)/((k+1)*(i+j)+i*j);
				if(!(j+k)) R+=dp[i][j][k];
				if(!(i+k)) S+=dp[i][j][k];
				if(!(i+j)) P+=dp[i][j][k];
			}

	printf("%.12f %.12f %.12f\n",R,S,P);
	return 0;
}
