#include <stdio.h>


double dp[105][105][105];

int main()
{
	int r,s,p;
	scanf("%d %d %d",&r,&s,&p);
	for(int i=0; i<105; i++)
		for(int j=0; j<105; j++)
			for(int k=0; k<105; k++)
				dp[i][j][k]=0;
	dp[r][s][p]=1;
	for(int i=r; i>=0; i--)
		for(int j=s; j>=0; j--)
			for(int k=p; k>=0; k--){
				if(i+j && i+k && j+k){
					if(i>0)
						dp[i-1][j][k]+=dp[i][j][k]*(i*k)/(i*j+i*k+j*k);
					if(j>0)
						dp[i][j-1][k]+=dp[i][j][k]*(i*j)/(i*j+i*k+j*k);
					if(k>0)
						dp[i][j][k-1]+=dp[i][j][k]*(j*k)/(i*j+i*k+j*k);
				}
			}
	double rans=0,sans=0,pans=0;
	for(int i=1; i<=100; i++){
		rans+=dp[i][0][0];
		sans+=dp[0][i][0];
		pans+=dp[0][0][i];
	}
	printf("%.9f %.9f %.9f\n",rans,sans,pans);
	return 0;
}
