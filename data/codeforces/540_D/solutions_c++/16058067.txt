#include<cstdio>
using namespace std;

int r,p,s;
double f[110][110][110];

int main(){
	scanf("%d%d%d",&r,&s,&p);
	f[r][s][p]=1;
	for(int i=r;i>=0;i-=1){
		for(int j=s;j>=0;j-=1){
			for(int k=p;k>=0;k-=1){
				double t=i*j+j*k+k*i;
				if(i>0&&j>0){
					f[i][j-1][k]+=f[i][j][k]*((i*j)/t);
				}
				if(j>0&&k>0){
					f[i][j][k-1]+=f[i][j][k]*((j*k)/t);
				}
				if(k>0&&i>0){
					f[i-1][j][k]+=f[i][j][k]*((k*i)/t);
				}
			}
		}
	}
	double ansr=0,anss=0,ansp=0;
	for(int i=1;i<=r;i++) ansr+=f[i][0][0];
	for(int i=1;i<=s;i++) anss+=f[0][i][0];
	for(int i=1;i<=p;i++) ansp+=f[0][0][i];
	printf("%.10lf %.10lf %.10lf\n",ansr,anss,ansp);
	return 0;
}
