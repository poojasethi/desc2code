#include<iostream>
#include<cstdio>
using namespace std;
double dp[101][101][101];
int r,p,s;
int main(){
    cin>>r>>s>>p;
    dp[r][s][p]=1.0;
    int sum=0;
    for(int i=r;i>=0;i--)
    for(int j=s;j>=0;j--)
    for(int k=p;k>=0;k--){
        if(dp[i][j][k]<1e-50) continue;
        sum = i*j + j*k + i*k;
        if(i>0 && k>0) dp[i-1][j][k]+=1.0*i*k/sum*dp[i][j][k];
        if(i>0 && j>0) dp[i][j-1][k]+=1.0*i*j/sum*dp[i][j][k];
        if(k>0 && j>0) dp[i][j][k-1]+=1.0*k*j/sum*dp[i][j][k];
    }
    double a1=0.0,a2=0.0,a3=0.0;
    for(int i=1;i<=r;i++) a1+=dp[i][0][0];
    for(int i=1;i<=s;i++) a2+=dp[0][i][0];
    for(int i=1;i<=p;i++) a3+=dp[0][0][i];
    printf("%.12f %.12f %.12f",a1,a2,a3);
    return 0;
}
