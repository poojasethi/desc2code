n = int( raw_input() )


countx = 0
county = 0

xar = []
yar = []

for x in xrange( 0, 101 ):
    xar.append( 0 )
    yar.append( 0 )


for i in xrange( 0, n ):
    st = raw_input().split()
    x = int( st[0] )
    y = int( st[1] )
    if xar[ x ] == 0:
        countx += 1
        xar[ x ] = 1
    if yar[ y ] == 0:
        county += 1
        yar[ y ] = 1

print min( countx, county )
