from sets import Set

n = int(raw_input(''))
points = []
for point in xrange(n):
    [x, y] = map(int, raw_input('').split())
    points.append((x, y))

uniq_x = Set()
uniq_y = Set()

for point in points:
    x, y = point
    if x not in uniq_x:
        uniq_x.add(x)
    if y not in uniq_y:
        uniq_y.add(y)

print min(len(uniq_x), len(uniq_y))
