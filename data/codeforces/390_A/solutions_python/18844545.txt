import sys


if __name__ == '__main__':
    n = int(raw_input())
    coordinates = []
    for i in range(n):
        x, y = map(int, raw_input().split())
        coordinates.append((x, y))

    xx = set([x for x, y in coordinates])
    yy = set([y for x, y in coordinates])
    result = min(len(xx), len(yy))

    print(str(result))
