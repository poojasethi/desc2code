#!/usr/bin/env python


def read():
    n = int(raw_input())
    ret = []
    for i in range(n):
        ret.append(map(int, raw_input().split()))
    return ret


def work(ptList):
    A = len(set([pt[0] for pt in ptList]))
    B = len(set([pt[1] for pt in ptList]))
    print min(A, B)


if __name__ == "__main__":
    work(read())
