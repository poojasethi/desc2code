#include<bits/stdc++.h>
using namespace std;
int n,m,ans=0x7FFFFFFF,Max=0,Min=0x7FFFFFFF;
string s[10];
int a[8]={0,1,2,3,4,5,6,7};
int main()
{
	cin>>n>>m;
	for(int i=0;i<n&&cin>>s[i];i++);
	do
	{
		Max=0,Min=0x7FFFFFFF;
		for(int i=0;i<n;i++)
		{
			int num=0;
			for(int j=0;j<m;j++)
			num=num*10+(s[i][a[j]]-'0');
	//		cout<<num<<endl;
		
			if(num>Max) Max=num;
			if(num<Min) Min=num;
		}
		if(Max-Min<ans) ans=Max-Min;
	}while(next_permutation(a,a+m));
	cout<<ans<<endl;
}
