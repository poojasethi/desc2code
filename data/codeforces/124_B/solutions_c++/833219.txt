#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
	vector<int> help;
	int n,k,mn,mx,aux,res=1000000000;
	char num[10][10];
	scanf("%d%d",&n,&k);

	for (int i=0; i<n; i++)
		scanf(" %s",num[i]);

	for (int i=0; i<k; i++)
		help.push_back(i);

	do {
		mn=1000000000;	mx=-1;
		for (int i=0; i<n; i++) {
			aux=0;
			for (int j=0; j<k; j++)	{	aux *= 10;	aux += num[i][help[j]];	}
			if (aux<mn)	mn = aux;
			if (aux>mx)	mx = aux;
		}

		if (mx-mn < res)	res = mx-mn;		

	}	while (next_permutation(help.begin(),help.end()));

	printf("%d\n",res);

	return 0;
}
