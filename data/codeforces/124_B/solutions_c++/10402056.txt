#include <bits/stdc++.h>

using namespace std;
int a[8][8], b[8], p[8];
int main() {
	int n, m, ans;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; ++i) {
		for (int j = 0; j < m; ++j) {
			scanf("%1d", &a[i][j]);
		}
	}
	for (int i = 0; i < m; ++i) {
		p[i] = i;
	}
	ans = 1000000007;
	do {
		for (int i = 0; i < n; ++i) {
			b[i] = 0;
			for (int j = 0; j < m; ++j) {
				b[i] *= 10;
				b[i] += a[i][p[j]];
			}
		}
		ans = min(ans, *max_element(b, b + n) - *min_element(b, b + n));
	} while (next_permutation(p, p + m));
	printf("%d\n", ans);

	return 0;
}
