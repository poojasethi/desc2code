#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int n,k,num[8]={0,1,2,3,4,5,6,7},dif=99999999;
	cin>>n>>k;
	char a[10][10];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	
	do{
		int max=0,min=99999999,ans=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<k;j++)
			{
				ans=ans*10+a[i][num[j]]-'0';
			}
			if(ans>max)max=ans;
			if(ans<min)min=ans;
			ans=0;
		}
		if((max-min)<dif)	dif=max-min;
	}while(next_permutation(num,num+k));
	
	cout<<dif<<endl;
	return 0;
}
