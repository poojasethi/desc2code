#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
int p[8];
char s[8][9];
int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%s",&s[i]);
	}
	for(int i=0;i<k;i++){
		p[i]=i;
	}
	int ret=999999999;
	do{
		int l=999999999,r=0;
		for(int i=0;i<n;i++){
			int c=0;
			for(int j=0;j<k;j++){
				c=c*10;
				c=c+s[i][p[j]]-'0';
			}
			r=max(c,r);
			l=min(c,l);
		}
		ret=min(ret,r-l);
	}
	while (next_permutation(p,p+k));
	printf("%d",ret);
	return 0;
}
