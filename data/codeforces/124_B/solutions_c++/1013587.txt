#include<iostream>
#include<vector>
#include<string>
#include<sstream>
#include<algorithm>
using namespace std;
int n,k,i,j,c,res=1e9,a1,b1,b[10],p;
string s;
vector<string> a;
int main(){
	cin>>n>>k;
	for(i=0;i<n;i++){
		cin>>s;
		a.push_back(s);
	}
	for(i=0;i<k;i++)
		b[i]=i;
	do{
		a1=0;
		b1=1e9;
		for(i=0;i<n;i++){
			string s1=a[i];
			for(c=0;c<k;c++)
				a[i][c]=s1[b[c]];
			stringstream out(a[i]);
			out>>p;
			a1=max(a1,p);
			b1=min(b1,p);
			a[i]=s1;
		}
		res=min(res,abs(a1-b1));
	}
	while(next_permutation(b,b+k));
	cout<<res<<endl;
}
