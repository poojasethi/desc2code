#include <iostream>
#include <algorithm>
#include <string>
#include <cmath>

using namespace std;

int N,K,pos[]={0,1,2,3,4,5,6,7},result=1000000000;
string A[8],B[8];

int tonum(string s)
{
	int ret=0;
	for (int i=0;i<s.size();i++)
		ret=ret*10+s[i]-'0';
	return ret;
}

void solve()
{
	sort(B,B+N);
	result=min(result,tonum(B[N-1])-tonum(B[0]));
}

int main()
{
	cin >> N >> K;
	for (int i=0;i<N;i++)
	{
		cin >> A[i];
		B[i]=A[i];
	}
	do
	{
		for (int i=0;i<N;i++)
			for (int j=0;j<K;j++)
				B[i][j]=A[i][pos[j]];
		solve();
	} while (next_permutation(pos,pos+K));
	cout << result << endl;
	return 0;
}
