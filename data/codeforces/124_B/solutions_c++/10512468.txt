#include<iostream>
#include<algorithm>
using namespace std;
char a[100][100];
int main()
{
	int n;
	cin>>n;
	int k;
	cin>>k;
	int b[]={0,1,2,3,4,5,6,7};
	int i=1;
	for(i=1;i<=n;i++)
		cin>>a[i];
	int ans=1000000000;
	do
	{
		int max=-1000000000;
		int min=1000000000;
		for(int i=1;i<=n;i++)
		{
			int num=0;
			for(int j=0;j<k;j++)
				num=num*10+a[i][b[j]]-48;
			if(num>max)
				max=num;
			if(num<min)
				min=num;
		}
		if(max-min<ans)
			ans=max-min;
	}
	while(next_permutation(b,b+k));
	cout<<ans<<endl;
}
