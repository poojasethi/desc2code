#include <stdio.h>
#include <algorithm>
using namespace std;
int n,k,res,mn,mx,x;
char s[8][9];
char d[9]="01234567";
char r[9];
void calc(){
    mn=1<<29;
    mx=0;
    for(int i=0;i<n;++i){
        for(int j=0;j<k;++j)
            r[j]=s[i][d[j]-'0'];
        r[k]=0;
        sscanf(r,"%d",&x);
        mn=min(mn,x);
        mx=max(mx,x);
    }
    res=min(res,mx-mn);
}
int main()
{
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;++i)
        scanf("%s",s[i]);
    res=1<<29;
    do
        calc();
    while(next_permutation(d,d+k));
    printf("%d\n",res);
    return 0;
}
