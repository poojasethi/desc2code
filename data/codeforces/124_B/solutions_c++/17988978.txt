#include<cstdio>
#include<algorithm>
int o[8]={0,1,2,3,4,5,6,7},n,k,i,j,z,x,u,p=1<<30;
char s[8][9];
main(){
	scanf("%d %d",&n,&k);
	for(i=0;i<n;i++)
		scanf("%s",s[i]);
	do{
		z=1<<30,x=0;
		for(i=0;i<n;i++){
			u=0;
			for(j=0;j<k;j++)
				u=u*10+(s[i][o[j]]-'0');
			if(z>u)z=u;
			if(x<u)x=u;
		}
		if(x-z<p)p=x-z;
	}while(std::next_permutation(o,o+k));
	printf("%d",p);
}
