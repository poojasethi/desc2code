#include <bits/stdc++.h>

using namespace std;
const int inf=1000000000;
int pos[10];
char s[10][10];

int main()
{
	int n,k;
	//for(int i=0;i<10;i++) pos[i]=i;
	while(cin>>n>>k)
	{
		for(int i=0;i<k;i++)pos[i]=i;
		for(int i=0;i<n;i++) cin>>s[i];
		int ans=inf;
		do{
			int maxNum=0,minNum=inf;
			for(int i=0;i<n;i++)
			{
				int t=0;
				for(int j=0;j<k;j++)
				{
					t=t*10+(s[i][pos[j]]-'0');
				}
				maxNum=max(maxNum,t);
				minNum=min(minNum,t);
			}
			ans=min(ans,maxNum-minNum);
		}while(next_permutation(pos,pos+k));
		cout<<ans<<endl;

	}
	
	return 0;
}
