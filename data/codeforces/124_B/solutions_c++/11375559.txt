# include <bits/stdc++.h>

using namespace std;

int p[10];
string a[10];
int b[10];

int fact(int n){
	if(n == 0)
		return 1;
	return n * fact(n-1);
}

int main(){
	int n, k;
	cin >> n >> k;
	for(int i=1 ; i<=n ; i++)
		cin >> a[i];
	
	int loop = fact(k);
	for(int i=0 ; i<k ; i++)
		p[i] = i;

	int ans = 1e9;
	for(int i=0 ; i<loop ; i++){
		memset(b , 0 , sizeof b);

		for(int j=1 ; j<=n ; j++)
			for(int s=0 ; s<k ; s++)
				b[j] = 10 * b[j] + (a[j][p[s]] - '0');
		
		sort(b+1 , b+1+n);
		ans = min(ans , b[n] - b[1]);

		next_permutation(p , p+k);
	}

	cout << ans << endl;

	return 0;
}
