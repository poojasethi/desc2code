#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;



int main()
{
	int n,ans=0,k,maxdiff=2147483647;

	cin>>n>>k;

	vector<string> v(n);

	for( int i=0 ; i<n ; cin>>v[i++] );

	int t[] = {0,1,2,3,4,5,6,7};

	do{
		vector<int>v1;

		for(int i=0;i<n;i++)
		{
			ans=0;

			for(int j=0;j<k;j++)

				ans=ans*10+(v[i][t[j]]-'0');

			v1.push_back(ans);
		}

		sort(v1.begin(),v1.end());

		maxdiff=min(maxdiff,v1[v1.size()-1]-v1[0]);

	} while(next_permutation(t,t+k));

	cout<<maxdiff<<endl;
}
