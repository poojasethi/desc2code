#include <stdio.h>
#include <algorithm>
using namespace std;
int order[8]= {0,1,2,3,4,5,6,7};
char s[8][9];
int main()
{
    int n,k,i,j,mi,mx,num,ans=1<<30;
    scanf("%d %d",&n,&k);
    for (i=0; i<n; i++)scanf("%s",s[i]);
    do
    {
        mi=1<<30;
        mx=0;
        for (i=0; i<n; i++)
        {
            num=0;
            for (j=0; j<k; j++)
                num=num*10+(s[i][order[j]]-'0');
            mi=min(mi,num);
            mx=max(mx,num);
        }
        ans=min(mx-mi,ans);
    }
    while(next_permutation(order,order+k));
    printf("%d\n",ans);
}
