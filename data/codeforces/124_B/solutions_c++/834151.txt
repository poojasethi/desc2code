#include<cstdio>
#include<cstring>
#include<algorithm>
#define INF 100000000
using namespace std;
char s[10][10];
int main(void)
{
	int n,ans=INF,k;
	int t[]={0,1,2,3,4,5,6,7};
	scanf("%d%d", &n,&k);
	for ( int i=0; i<n; i++ ) scanf("%s", s[i]);
	do{
		int mi=INF,ma=0;
		for ( int i=0; i<n; i++ ){
			int res=0;
			for ( int j=0; j<k; j++ ) res=res*10+(s[i][t[j]]-'0');
			mi = min(mi,res);
			ma = max(ma,res);
		}
		ans= min(ans,ma-mi);
	}while( next_permutation(t,t+k));
	printf("%d\n", ans);
	return 0;
}
