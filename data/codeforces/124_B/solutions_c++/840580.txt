#include <stdio.h>
#include <algorithm>

using namespace std;

char d[8][8];
int pos[8];
int num[8];

int main(){
  int n, k;

  scanf("%d %d\n",&n,&k);
  for (int i=0; i<n; i++) gets( d[i] );

  for (int i=0; i<k; i++) pos[i] = i;
  int ans = 2000000000;
  do{
    for (int i=0; i<n; i++){
      num[i] = 0;
      for (int j=0; j<k; j++){
        num[i] *= 10;
        num[i] += int(d[i][ pos[j] ] - '0');
      }
    }
    ans = min( ans, *max_element(num,num+n) - *min_element( num, num+n ) );
  } while ( next_permutation( pos, pos+k ) );
  printf("%d\n",ans);

  return 0;
}
