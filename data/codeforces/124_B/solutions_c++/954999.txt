#include<iostream>
#include<algorithm>
using namespace std;
string a[10];
int main(){
	int n,k,fact=1;
	cin>>n>>k;
	for(int i=0;i<n;i++)
		cin>>a[i];
	int arr[k];
	for(int i=0;i<k;i++)
		arr[i]=i;
	for(int i=2;i<=k;i++)
		fact*=i;
	int ans=1000000000;
	for(int i=0;i<fact;i++){
		int mn=1000000000,mx=0;
		for(int i=0;i<n;i++){
			int num=0,p=1;
			for(int j=k-1;j>=0;p*=10,j--)
				num+=int(a[i][arr[j]]-'0')*p;
			mn=min(mn,num);
			mx=max(mx,num);
		}
		ans=min(ans,mx-mn);
		next_permutation(arr,arr+k);
	}
	cout<<ans<<endl;
}
