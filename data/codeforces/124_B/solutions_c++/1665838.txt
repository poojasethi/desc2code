#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int Inf = 1000000000;
char s[105][105];
int r[10],a[10];

int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i = 0; i < n; i++)
		scanf("%s",s[i]);
	for(int i = 0; i < m; i++)r[i] = i;
	int ret = Inf;
	do
	{
		for(int i = 0; i < n; i++)
		{
			a[i] = 0;
			for(int j = 0; j < m; j++)
				a[i] = a[i]*10+s[i][r[j]]-'0';
		}
		sort(a,a+n);
		ret = min(ret,a[n-1]-a[0]);
	}while(next_permutation(r,r+m));
	printf("%d\n",ret);

	return 0;
}
