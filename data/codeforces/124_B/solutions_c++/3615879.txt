string s[8];

int per(string s)
{string st;
 st=s;
  for (int i=0;i<k;i++) st[i]=s[p[i]];
  int a=0;
  for (int i=0;i<k;i++) a=a*10+st[i]-'0';
 return a;
}
int main()
{cin>>n>>k;
for (int i=0;i<n;i++) cin>>s[i];
for (int i=0;i<k;i++) p[i]=i;
do
{ for (int i=0;i<n;i++) a[i]=per(s[i]);
  int m1=0, m2=1000000000;
  for (int i=0;i<n;i++)  {m1=max(m1,a[i]);m2=min(m2,a[i]); }
  ans =min(ans,m1-m2);
} while (next_permutation(p,p+k));

}
