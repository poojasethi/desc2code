#include <stdio.h>
#include <algorithm>

using namespace std;

char s[10][10];

int main(){
	int n, k;
	scanf("%d %d", &n, &k);
	for(int i=0; i<n; i++)
		scanf("%s", s[i]);
	
	int idx[10]= {0,1,2,3,4,5,6,7}, min1, max1, ans = (int)1e9, s1;
	
	do{
		min1 = int(1e9);
		max1 = -1;
		for(int i=0; i<n; i++)
		{
			s1=0;
			for(int j=0; j<k; j++)
				s1 = s1*10 + s[i][idx[j]]-'0';

			min1 = min(s1, min1);
			max1 = max(s1, max1);
		}
		ans = min(ans, max1-min1);
	}
	while(next_permutation(idx, idx+k));
	
	printf("%d\n", ans);
	return 0;
}
