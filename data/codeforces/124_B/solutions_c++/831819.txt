#include <cstdio>
int n, k, t[10], q[10];

char s[10][10];
int mn=1000000000, mx=0, ans=1000000000;

void go(int l){
if (l<k){
for (int i=0; i<k; i++)
if (t[i]==-1){
t[i]=1; q[l]=i;
go(l+1);
t[i]=-1;
}
} else{
mn=1000000000; mx=0;
for (int i=0; i<n; i++){
int p=0;
for (int j=0; j<k; j++){
p=p*10+s[i][q[j]]-'0';
}
if (p>mx) mx=p;
if (p<mn) mn=p;
}
if (ans>mx-mn) ans=mx-mn;
}
}
int main(){
scanf("%d%d",&n,&k);

for (int i=0; i<10; i++)
t[i]=-1;
for (int i=0; i<n; i++)
scanf("%s",s[i]);
go(0);
printf("%d\n",ans);

return 0;
}
