import itertools

def find_diff(tab):
	return max(tab) - min(tab);

def tab_to_nums(tab):
	return [int(''.join(i),10) for i in tab]

def tab_perm(tab,perm):
	return [[i[p] for p in perm] for i in tab]
		
n,k = map(int,raw_input().split());
a = []

p = list(itertools.permutations(range(k)))
for i in range(n):
	a.append(list(raw_input()))

mi = 999999999;
#print a
#print tab_perm(a,p[4])
for per in p:
	x = find_diff(tab_to_nums(tab_perm(a,per)))
	if(x<mi):
		mi = x;
print mi
