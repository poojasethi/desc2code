from itertools import permutations

r = lambda: raw_input().strip()

n,k = map(int,r().split())
digits = [r() for _ in xrange(n)]
inds = list(permutations(xrange(k),k))
min_diff = 999999999
for ind in inds:
    nums = [int(''.join([d[i] for i in ind])) for d in digits]
    diff = max(nums)-min(nums)
    if diff<min_diff:
        min_diff = diff
print min_diff
