import itertools

n, k = map(int, raw_input().split())

num = []
for i in range(n):
	num.append(raw_input())

#print num
result = []

for i in itertools.permutations(range(k)):	
    tmp = []
    for x in num:
        tmp_num = [x[pos] for pos in i]
        tmp.append(''.join(tmp_num))
    #print tmp
    result.append(int(max(tmp)) - int(min(tmp)))

print min(result)
