import itertools

n, k = map(int, raw_input().split())

data = [ raw_input() for i in xrange(n)]
a = range(k)

ret = float('inf')
for i in itertools.permutations(a):
    current = []
    for d in data:
        tmp = [d[ii] for ii in i]
        current.append(''.join(tmp))

    current.sort()
    diff = int(current[-1]) - int(current[0])
    ret = min(ret, diff)

print ret
