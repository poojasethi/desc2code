import itertools
n,k = map(int, raw_input().split())
dig = []
for i in range(n):
    dig.append(raw_input())
mp = range(k)
ans = 987654321
for pm in itertools.permutations(mp, k):
    mind = "999999999"
    maxd = "0"
    for d in dig:
        num = "".join([ d[j] for j in pm ])
        if num < mind : mind = num
        if num > maxd : maxd = num
    ans = min(ans, int(maxd) - int(mind))
print ans
