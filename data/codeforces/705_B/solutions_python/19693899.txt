import sys

def slist_to_int_list(l):
    return [int(x) for x  in l]

first = True
for line in sys.stdin:
    if first:
        ns = line.split("\n")[0]
        n = int(ns)
        first = False
    else:
        list = slist_to_int_list(line.split(" "))





res = ""

total = 0

for i in range(n):
    total += list[i] - 1
    if total % 2 == 0:
        res += "2 "
    else:
        res += "1 "


print res
