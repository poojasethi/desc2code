import sys

def gl():
    return sys.stdin.readline().strip()

def gls():
    return gl().split()

def gi():
    return int(gl())

def gis():
    return map(int, gls())

def gfs():
    return map(float, gls())

def solve():
    n = gi()
    vals = gis()
    winner = 2
    for val in vals:
        winner+=(val-1)
        print [1,2][(winner+1)%2]
solve()
