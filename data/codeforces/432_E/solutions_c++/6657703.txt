#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

const int N = 100;
int a[N + 2][N + 2];

int minCharacter(int x, int y) {
    int res = a[x][y];
    if(res == -1) {
        res = 'A';
        while(res == a[x - 1][y] || res == a[x + 1][y] || res == a[x][y - 1] || res == a[x][y + 1]) ++res;
    }
    return res;
}

int main() {
    int m, n; cin >> m >> n;
    memset(a, -1, sizeof a);
    for(int x = 1; x <= m; ++x) {
        for(int y = 1; y <= n; ++y) {
            int c = minCharacter(x, y);
            putchar(c);
            if(a[x][y] != -1) continue;
            int size = 1;
            while(x + size <= m && y + size <= n && minCharacter(x, y + size) == c) ++size;
            for(int i = 0; i < size; ++i)
                for(int j = 0; j < size; ++j)
                    a[x + i][y + j] = c;
        }
        putchar(10);
    }
    return 0;
}
