#include <bits/stdc++.h>

using namespace std;

int N,M,ar[110][110],h[110][110],used[110][110];

int main(){
	scanf("%d %d",&N,&M);
	
	for(int i=1;i<=N;i++){
		for(int j=1;j<=M;j++) if(!used[i][j]){
			int c=1;
			while(c== ar[i][j-1] || c==ar[i-1][j] || c==ar[i][j+1]) c++;
			
			if(ar[i][j-1] && ar[i][j-1]<c && ar[i][j-1]!=ar[i-1][j] && !used[i][j-1] && h[i][j-1]+1 <=N-i+1){
				ar[i][j]=ar[i][j-1];
				h[i][j]=h[i][j-1]+1;
			}
			
			else{
				ar[i][j]=c;
				h[i][j]=1;
			}
		}
		
		for(int j=M;j>=1;j--){
			if(ar[i][j]==ar[i][j+1])
				h[i][j]=h[i][j+1];
			
			for(int k=1;k<h[i][j];k++)
				ar[i+k][j]=ar[i][j] , used[i+k][j]=1;
		}
	}
	
	for(int i=1;i<=N;i++,puts(""))
		for(int j=1;j<=M;j++)
			printf("%c",'A'-1+ar[i][j]);
	
	return 0;
}
