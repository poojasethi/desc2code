#include<cstring>
#include<cstdio>
#define N 102
using namespace std;
int n,m;
int a[N][N];
int minc(int x,int y){
    int r=a[x][y];
    if(r==-1){
        r='A';
        while(r==a[x-1][y] || r==a[x][y-1] || r==a[x+1][y] || r==a[x][y+1])++r;
    }
    return r;
}
int main(){
    while(~scanf("%d%d",&n,&m)){
        memset(a,-1,sizeof(a));
        for(int x=1;x<=n;++x){
            for(int y=1;y<=m;++y){
                int c=minc(x,y);
                putchar(c);
                if(a[x][y]!=-1)continue;
                int sz=1;
                while(x+sz<=n && y+sz<=m && minc(x,y+sz)==c)++sz;
                for(int i=0;i<sz;++i){
                    for(int j=0;j<sz;++j){
                        a[x+i][y+j]=c;
                    }
                }
            }
            puts("");
        }
    }
    return 0;
}
