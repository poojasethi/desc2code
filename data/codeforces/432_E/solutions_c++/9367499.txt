#include <iostream>

#include <stdio.h>
typedef long long ll ;
using namespace std;
int n , m , np , mp , x , y , j , q , k , v , i ;
char s[101][101];
char c;
int col[101];
char findchar(int i,int j)
{
	char c = 'A';
	while( (j && c==s[i][j-1]) || (i && c==s[i-1][j]) || c==s[i][j+1])
	{
		c++ ;
	}
	return c;
}
int main(){
	cin >> n >> m ;
	while(true) {
		i = 0;
		for( j = 1 ; j < m ; j++)
			if( col[j] < col[i])
				i = j;
		q = col[i];
		if(q >= n) break;
		c = findchar(q,i);
		j = 1;
		while( i+j < m && q+j < n && col[i+j] == q && findchar(q,i+j) == c)
			j++;
		for(k = i ; k < i+j ; k++ ){
			for(v = q ; v < q+j ; v++ ){
				s[v][k] = c;
			}
			col[k]+=j;
		}
	}
	for( i = 0 ; i < n ; i++ ) printf("%s\n",s[i]);
}
