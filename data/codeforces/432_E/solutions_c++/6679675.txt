#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=120;
int n,m,ans[maxn][maxn]={0};
int get(int x,int y) 
{
	if(ans[x][y]) return 0;
	int i,c=1,s[3]={ans[x-1][y],ans[x][y-1],ans[x][y+1]};
	sort(s,s+3);
	for(i=0;i<3;i++)
		if(c==s[i]) c++;
	return c;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,k,i2,j2;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if(!ans[i][j])
			{
				int c=get(i,j);
				for(k=0;i+k<=n&&j+k<=m&&get(i,j+k)==c&&get(i+k,j)<=c;k++);
				for(i2=i;i2<i+k;i2++)
					for(j2=j;j2<j+k;j2++)
						ans[i2][j2]=c;
			}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
			printf("%c",ans[i][j]-1+'A');
		printf("\n");
	}
	return 0;
}
