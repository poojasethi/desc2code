#include <bits/stdc++.h>

using namespace std;

int N,M;
int A[105][105];
bool H[105][105][35];

void doldur( int x,int y,int a,int b )
{
	int i,j;

	for( i=0; i<a; i++ )
		for( j=0; j<a; j++ )
		{
			A[x+i][y+j] = b;
			H[x+i+1][y+j][b] = H[x+i][y+j+1][b] = H[x+i][y+j-1][b] = 1;
		}
}

void yaz()
{
	int i,j;

	for( i=1; i<=N; i++,puts("") )
		for( j=1; j<=M; j++ )
			printf("%c",A[i][j]-1+'A');
}

void solve( int x,int y )
{
	if( x == N + 1 ) { yaz(); exit(0); }
	if( y == M + 1 ) solve( x+1,1 );
	if( A[x][y] ) solve( x,y+1 );

	int t = 1;
	while( H[x][y][t] ) t++;

	A[x][y] = t;
	H[x+1][y][t] = 1;

	int cnt = 0,yx = x,yy = y;
	while( yy > 0 && A[x][yy] == t ) cnt++,yy--;

	if( cnt == N-x+1 || y == M )
		doldur( x,y-cnt+1,cnt,t );

	solve( x,y+1 );
}

int main()
{
	int i,j;

	cin >> N >> M;

	solve( 1,1 );
}
