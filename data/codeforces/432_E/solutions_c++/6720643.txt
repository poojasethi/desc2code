#include <cstdio>
#include <cstring>

const int maxn = 105;

int n, m;
char ans[maxn][maxn];

inline int able(int x, int y, char c)
{
	int tmp = 1;
	for(int i = y - 1; i && ans[x][i] == c; --i) ++tmp;
	
	if(x + tmp - 1 > n || ans[x - 1][y] == c || ans[x][y + 1] == c) return 0;
	for(int i = 0; i < tmp; ++i) ans[x + i][y] = c;
	for(int i = 0; i < tmp; ++i) ans[x + tmp - 1][y - i] = c;
	return 1;
}

int main()
{
//	freopen("432e.in", "r", stdin);
//	freopen("432e.out", "w", stdout);
	
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; ++i)
	{
		for(int j = 1; j <= m; ++j)
		{
			if(ans[i][j]) continue;
			for(char c = 'A'; c <= 'Z'; ++c) if(able(i, j, c)) break;
		}
	}
	for(int i = 1; i <= n; ++i)
	{
		for(int j = 1; j <= m; ++j) putchar(ans[i][j]);
		putchar('\n');
	}
	return 0;
}
