#include<cstdio>
#include<cstring>

const int MAX_SIZE = 100;

int h, w;
char mat[MAX_SIZE + 10][MAX_SIZE + 10];

bool check(int x, int y, char fil){
	if(mat[x - 1][y] == fil || mat[x][y + 1] == fil)
		return false;
	if(mat[x][y - 1] == fil){
		if(mat[x - 1][y - 1] == fil)
			return false;
		int len;
		for(len = 0; mat[x][y - 1 - len] == fil; ++ len);
		if(x + len > h)
			return false;
		for(int i = 0; i <= len; i ++)
			mat[x + i][y] = mat[x + len][y - i] = fil;
		return true;
	}
	mat[x][y] = fil;
	return true;
}

int main(){
	scanf("%d%d", &h, &w);
	for(int i = 1; i <= h; i ++)
		for(int j = 1; j <= w; j ++){
			if(mat[i][j])
				continue;
			for(char d = 'A'; d <= 'Z'; d ++)
				if(check(i, j, d))
					break;
		}
	for(int i = 1; i <= h; i ++){
		for(int j = 1; j <= w; j ++)
			putchar(mat[i][j]);
		putchar('\n');
	}

	return 0;
}
