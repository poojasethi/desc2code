#include<cstdio>
using namespace std;
char a[105][105];
int d[105][105];
int f[105][105];
int n,m;
bool Check(int x,int y)
{
	if(x+d[x][y]>n||y+d[x][y]>m)
		return 0;
	for(int i=0;i<=d[x][y];i++)
		if(a[x+i][y+d[x][y]]||a[x+d[x][y]][y+i])
			return 0;
	for(int i=0;i<=d[x][y];i++)
		a[x+i][y+d[x][y]]=a[x+d[x][y]][y+i]=a[x][y],
		f[x+i][y+d[x][y]]=f[x+d[x][y]][y+i]=x*200+y;
	d[x][y]++;
	return 1;
}
int main()
{
	int i,j;char c;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			if(a[i][j])
				continue;
			for(c='A';c<='Z';c++)
			{
				if(a[i][j+1]==c||a[i-1][j]==c)
					continue;
				if(a[i][j-1]==c)
				{
					int t=f[i][j-1];
					if(!Check(t/200,t%200))
						continue;
				}
				else
					f[i][j]=i*200+j,
					d[i][j]=1,
					a[i][j]=c;
				break;
			}
		}
	for(i=1;i<=n;i++)
		puts(a[i]+1);
}
