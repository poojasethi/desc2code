#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

int a[110][110];
int check(int x,int y)
{
    for(int i=1;i<=148;i++)
    if(a[x-1][y]!=i && a[x+1][y]!=i && a[x][y-1]!=i && a[x][y+1]!=i) return i;
    return 148;
}
int main()
{
    int n,m;
    cin >> n >>m;
    for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
        if(a[i][j]==0)
        {
            int z = check(i,j);
            int d = 1;
            while(i+d<=n && j+d<=m && a[i][j+d]==0 && check(i,j+d)==z) d++;
            for(int x = 0;x<d;x++)
            for(int y = 0;y<d;y++)
                a[x+i][y+j] = z;
        }
    for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    {
        printf("%c",a[i][j]+'A'-1);
        if(j==m)printf("\n");
    }
    return 0;
}
