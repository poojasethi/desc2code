#include <bits/stdc++.h>
using namespace std;
int a[110][110], vis[5];
int mincolor(int xi, int yi){
	int res = a[xi][yi];
	if(res==-1){
		res='A';
		while(res==a[xi][yi-1] || res==a[xi][yi+1] || res==a[xi-1][yi] || res==a[xi+1][yi]) ++res;
	}
	return res;
}
int main(){
	int n, m;
	memset(a, -1, sizeof(a));
	scanf("%d%d", &n, &m);
	for(int x=1; x<=n; x++){
		for(int y=1; y<=m; y++){
			int c = mincolor(x, y);
			putchar(c);
			if(a[x][y]!=-1) continue;
			int siz=1;
			while(x+siz<=n && y+siz<=m && mincolor(x, y+siz)==c) ++siz;
			for(int i=0; i<siz; i++){
				for(int j=0; j<siz; j++){
					a[x+i][y+j] = c;
				}
			}
		}
		putchar(10);
	}
	return 0;
}
