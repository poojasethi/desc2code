#include<cstdio>
#define N 110
using namespace std;

int n,m;
char ch[N][N];

bool judge(int x,int y,char k){
	int len=0;
	if(x==1&&y==1) return true;
	for(int i=y-1;i>=1;i--){
		if(ch[x][i]!=k) break;
		if(ch[x-1][i]==k) return false;
		len++;
	}
	for(int i=x;i<=x+len;i++){
		if(i>n) return false;
		if(ch[i][y]) return false;
	}
	for(int j=y;j>=y-len;j--){
		if(ch[x+len][j]) return false;
	}
	return true;
}

void color(int x,int y,char k){
	int len=0;
	for(int i=y-1;i>=1;i--){
		if(ch[x][i]!=k) break;
		len++;
	}
	for(int i=x;i<=x+len;i++){
		ch[i][y]=k;
	}
	for(int j=y;j>=y-len;j--){
		ch[x+len][j]=k;
	}
}

int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			if(ch[i][j]!=0) continue;
			for(char k='A';k<='Z';k++){
				if(ch[i-1][j]!=k&&ch[i][j+1]!=k&&judge(i,j,k)){
					color(i,j,k);
					break;
				}
			}
		}
	}
	for(int i=1;i<=n;i++) printf("%s\n",ch[i]+1);
	return 0;
}
