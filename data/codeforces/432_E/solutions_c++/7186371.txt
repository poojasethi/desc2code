#include<cstdio>
using namespace std;

const int maxn = 102;
int n, m, col[maxn][maxn];
int Try(int x, int y) {
  bool used[6];
  for (int i = 0; i <= 5; i++) used[i] = 0;
  used[col[x-1][y]] = used[col[x][y-1]] = 1;
  used[col[x+1][y]] = used[col[x][y+1]] = 1;
  for (int i = 1; i <= 5; i++) if (!used[i]) return i;
}
void Paint(int x, int y, int d, int p) {
  int t = 1;
  for (; x+t<=n && y+t<=m && col[x+t][y+t]==0 && col[x+t][y]==0; t++);
  if (t < d) t = t-1; else t = d-1;
  for (int i = x; i <= x+t; i++)
    for (int j = y; j <= y+t; j++) col[i][j] = p;
}
int main()
{
  int p, d;
  scanf("%d%d", &n, &m);
  for (int i = 1; i <= n; i++)
    for (int j = 1; j <= m; j++)
      if (col[i][j] == 0) {
        p = Try(i, j);
        for (d = 1; j+d<=m && col[i][j+d]==0 && Try(i, j+d)>=p && Try(i, j+d)!=p+1; d++);
        Paint(i, j, d, p);
      }
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) printf("%c", (char)(col[i][j]+(int)'A'-1));
    printf("\n");
  }
}
