#include<stdio.h>
using namespace std;
char g[128][128];
char getc(int x,int y){
	if(g[x][y]) return g[x][y];
	char ans='A';
	while(g[x-1][y]==ans || g[x+1][y]==ans || g[x][y-1]==ans || g[x][y+1]==ans)
		ans++;
	return ans;
}
int main(){
	int n,m;
	char col;
	int stp;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			col=getc(i,j);
			printf("%c",col);
			if(g[i][j]) continue;
			stp=1;
			while( i+stp<=n && j+stp<=m && col==getc(i,j+stp))
				stp++;
			for(int k=0;k<stp;k++)
				for(int l=0;l<stp;l++)
					g[i+k][j+l]=col;
		}
		puts("");
	}
	scanf("%d",&n);
	return 0;
}
