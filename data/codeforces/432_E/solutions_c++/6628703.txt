#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <map>

#include <cstring>
#define rep(i,j,n) for(i=j; i<=n; i++)
using namespace std;
int ans[105][105],n,m,i,j,l,t,p,k;
bool FLAG;
int main()
{
    scanf("%d%d",&n,&m);
    rep(i,0,max(n,m)+1) rep(j,0,max(n,m)+1) if (i<1 || j<1 || i>n || j>m)ans[i][j]=-2;
    rep(i,1,n) rep(j,1,m) ans[i][j]=-1;
    rep(i,1,n) rep(j,1,m) if (ans[i][j]==-1)
    {
        for (k=0; k<26; k++)
        {
            if (ans[i][j+1]==k) continue;
            FLAG=true; if (ans[i-1][j]==k) continue;
            for (l=j-1; l>0; l--) {if (ans[i][l]!=k) break; if (ans[i-1][l]==k) FLAG=false;}
            l++; if (!FLAG) continue;
            for (t=i; t<=i+j-l; t++)
              for (p=l; p<=j; p++) if (ans[t][p]!=k && ans[t][p]!=-1) {FLAG=false; break;}
            if (!FLAG) continue;
            for (t=i; t<=i+j-l; t++)
              for (p=l; p<=j; p++) ans[t][p]=k;
            break;
        }
    }
    rep(i,1,n) {rep(j,1,m) cout<<char(ans[i][j]+'A'); cout<<endl;}
    return 0;
}
