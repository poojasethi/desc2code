#include<iostream>
using namespace std;
char matrix[101][101] = { 0 };
char get(int x, int y)
{
	if (matrix[x][y]) return matrix[x][y];
	char result = 'A';
	while (1)
	{
		if (matrix[x][y - 1] != result&&matrix[x - 1][y] != result&&matrix[x+1][y]!=result&&matrix[x][y+1]!=result)
			break;
		result++;
	}
	return result;
}
int main()
{
	int n, m;
	cin >> n >> m;
	
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			char temp = get(i, j);
			cout << temp;
			if (matrix[i][j]) continue;
			int e = 0;
			while (i + e <= n&&j + e <= m&&get(i, j + e) == temp) e++;
			for (int t1 = i; t1 < i + e;t1++)
			for (int t2 = j; t2 < j + e; t2++)
				matrix[t1][t2] = temp;
		}
		cout << endl;
	}

	return 0;
}
