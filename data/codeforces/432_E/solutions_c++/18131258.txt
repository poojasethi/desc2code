#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,mp[110][110],cnt;
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
	  for (int j=1;j<=m;j++)
	    if (mp[i][j]==0)
	      for (int c=1;c<=26;c++)
	        if (mp[i-1][j]!=c&&mp[i][j+1]!=c)
	          if (mp[i][j-1]!=c){
	          	  cnt=1;mp[i][j]=c;break;
	          }else
	          	 if (i+cnt>n) continue;
	          	 else{
	          	 	 if (cnt==0&&mp[i][j-1]==c) continue;
	          	 	 ++cnt;
	          	 	 for (int k=1;k<=cnt;k++)
	          	 	   mp[i+k-1][j]=mp[i+cnt-1][j-k+1]=c;
	          	 	 break;
	          	 }
	        else continue;
	    else cnt=0;
	for (int i=1;i<=n;i++){
	    for (int j=1;j<=m;j++)
	      printf("%c",'A'+mp[i][j]-1);
	    printf("\n");
	}
	return 0;
}
