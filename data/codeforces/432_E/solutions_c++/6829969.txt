#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
int n,m;
char mp[105][105];
const int dir[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
char get(int x,int y)
{
    if(mp[x][y]) return mp[x][y];
    char ch = 'A';
    while(1)
    {
        bool isok = true;
        for(int i=0;i<4;i++)
            if(mp[x+dir[i][0]][y+dir[i][1]] == ch) isok = false;
        if(isok) return ch;
        ch++;
    }
    return '!';
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            char ch = get(i,j);
            printf("%c",ch);
            if(mp[i][j]) continue;
            int len = 1;
            while(i+len<=n && j+len<=m)
            {
                if(get(i,j+len) != ch) break;
                len++;
            }
            len--;
            for(int x=i;x<=i+len;x++)
                for(int y=j;y<=j+len;y++)
                    mp[x][y] = ch;
        }
        printf("\n");
    }
    return 0;
}
