#include<bits/stdc++.h>

using namespace std ;

typedef long long ll ;
typedef pair<int,int> pii ;
const int maxn = 110 ;
int n , m ;
char s[maxn][maxn] ;

char f(int i , int j)
{
  char ret = 'A' ;
  while(s[i-1][j]==ret || s[i+1][j]==ret || s[i][j-1]==ret || s[i][j+1]==ret)
    ret++ ;
  return ret ;
}

int main()
{
  ios_base::sync_with_stdio(false) ;
  for(int i=0 ; i<maxn ; i++)
    for(int j=0 ; j<maxn ; j++)
      s[i][j] = '.' ;
  cin >> n >> m ;
  for(int i=1 ; i<=n ; i++)
    for(int j=1 ; j<=m ; j++)
      {
	if(s[i][j] != '.') continue ;
	char c = f(i,j) ;
	int d = 1 ;
	for( ; d+i<=n && j+d<=m && s[i][j+1]=='.' && f(i , j+d) == c  ; d++) ;
	for(int x = i ; x<i+d ; x++)
	  for(int y = j ; y<j+d ; y++)
	    s[x][y] = c ;
      }
  for(int i=1 ; i<=n ; i++)
    {
      for(int j=1 ; j<=m ; j++)
	cout << s[i][j] ;
      cout << endl ;
    }
  return 0 ;
}
