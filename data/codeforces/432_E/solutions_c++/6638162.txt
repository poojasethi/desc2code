#include<iostream>
#include<cstring>
using namespace std;

char ans[101][101];

int find(int x,int y)
{
      if(ans[x][y]) return ans[x][y];
      char ret = 'A';
      while (ret == ans[x - 1][y] || ret == ans[x][y - 1] ||
	     ret == ans[x][y + 1]) ret++;
      return ret;
}

int main()
{
      int n,m;
      cin >> n >> m;
      memset(ans,0,sizeof(ans));
      for (int i = 0; i < n; i++)
      {
	    for (int j = 0; j < m; j++)
	    {
		  char now = find(i,j);
		  cout << now;
		  if (ans[i][j]) continue;
		  int size = 1;
		  while (i + size < n && j + size < m &&
			 find(i,j + size) == now) size++;
		  for (int x = 0; x < size; x++)
			for (int y = 0; y < size; y++)
			      ans[i + x][j + y] = now;
	    }
	    cout << endl;
      }
      return 0;
}
