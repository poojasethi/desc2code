#include <iostream>
#include <stdio.h>
using namespace std;

int m, n;
char a[110][110] = {0};

char best(int x, int y)
{
    char ans;
    for(ans = 'A'; ans <= 'Z'; ans++)
    {
        if(ans != a[x - 1][y] && ans != a[x][y - 1] && ans != a[x][y + 1])
            break;
    }
    return ans;
}

int main()
{
    

    cin >> m >> n;

    for(int i = 1; i <= m; i++)
        for(int j = 1; j <= n; j++)
        {
            if(a[i][j])
                continue;
            // 找到最好的解
            char c = best(i, j);
            int wigth = 0;

            while(c == best(i, j + wigth) && j + wigth <= n && i + wigth <= m && !a[i][j + wigth])
            {
                wigth++;
            }
            for(int x = i; x < i + wigth; x++)
                for(int y = j; y < j + wigth; y++)
                {
                    a[x][y] = c;
                }
        }


    for(int i = 1; i <= m; i++)
        puts(a[i] + 1);

    return 0;
}
