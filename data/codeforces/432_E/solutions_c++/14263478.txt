#define mem(array) memset(array,-1,sizeof(array))
typedef pair<int,int> P;
int n,m,mp[105][105];
int get(int x,int y)
{
	int res=mp[x][y];
	if(res==-1){
		res='A';
		while(mp[x+1][y]==res||mp[x-1][y]==res||mp[x][y-1]==res||mp[x][y+1]==res)
			res++;
	}
	return res;
}
int main()
{
	//freopen("in.txt","r",stdin);
	cin>>n>>m;
	mem(mp);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			int c=get(i,j);
			putchar(c);
			if(mp[i][j]!=-1) continue;
			int size=1;
			while(i+size<n && j+size<m && get(i,j+size)==c) 
				size++;
			for(int p=i;p<=i+size-1;p++)
				for(int q=j;q<=j+size-1;q++)
					mp[p][q]=c;
		}
		cout<<endl;
	}
	return 0;
}
