#include <cstdio>
#include <algorithm>

using namespace std;

int n, m;
char a[120][120];

int check(int x, int y, char c) {
	int l = 0;
	while (y - l >= 1 && a[x][y - l] == c)
		l ++;
	if (a[x - 1][y] == c || a[x + l - 1][y - l] == c)  return -1;
	if (x + l - 1 > n)  return -1;

	for (int i = 0; i < l; i ++)
		if (a[x + i][y + 1] == c || a[x + l][y - i])  return -1;
	return l;
}
int main() {

#endif
	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; i ++)
		for (int j = 1, t; j <= m; j ++)
			if (!a[i][j]) {
				char &c = a[i][j];
				for (c = 'A'; ; c ++)
					if (t = check(i, j, c), t != -1) {
						for (int di = 0; di < t; di ++)
							a[i + di][j] = c;
						for (int dj = 1; dj < t; dj ++)
							a[i + t - 1][j - dj] = c;
						break;
					}
			}
	for (int i = 1; i <= n; i ++)
		printf("%s\n", a[i] + 1);
}
