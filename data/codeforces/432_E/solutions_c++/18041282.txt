#include <bits/stdc++.h>
using namespace std;

const int N = 123;
char s[N][N];
int n, m, dx[4] = {1, 0, -1, 0}, dy[4] = {0, 1, 0, -1};

char get(int a, int b) {
  char res = 'A';
  for(int cs = 1; cs <= 4; cs++) {
    for(int i = 0; i < 4; i++) {
      int tx = a + dx[i], ty = b + dy[i];
      if(s[tx][ty] == res) res++;
    }
  }
  return res;
}

int main() {
  scanf("%d%d", &n, &m);
  memset(s, 0, sizeof s);
  int cnt = 0;
  for(int i = 1; i <= n; i++) {
    for(int j = 1; j <= m; j++) {
      if(s[i][j] != 0) continue;
      char sgn = get(i, j);
      int len = 0;
      while(i + len <= n && j + len <= m && get(i, j + len) == sgn && s[i][j + len] == 0) len++;
      for(int u = i; u < i + len; u++) {
        for(int v = j; v < j + len; v++) {
          s[u][v] = sgn;
        }
      }
    }
  }
  for(int i = 1; i <= n; i++) {
    for(int j = 1; j <= m; j++) {
      printf("%c", s[i][j]);
    }
    printf("\n");
  }
  return 0;
}
