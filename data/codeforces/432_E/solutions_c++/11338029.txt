#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
#define MN 105
int nRow,nCol;
char map[MN][MN];
inline bool check(int x,int y,char c)
{
    if(map[x-1][y]==c||map[x][y+1]==c)
        return false;
    if(map[x][y-1]!=c)
        return true;
    int a;
    for(a=x;map[a][y-1]==c;++a);
    if(a>nRow)
        return false;
    int len=a-x+1;
    for(int j=0;j<len;++j)
        if(map[x-1][y-j]==c)
            return false;
    return true;
}

int main()
{
    cin>>nRow>>nCol;
    for(int i=1;i<=nRow;++i)
    for(int j=1;j<=nCol;++j)
    if(map[i][j]==0)
    {
        for(char c='A';c<='Z';++c)
        if(check(i,j,c))
        {
            int a,b;
            for(a=i;map[a][j-1]==c;++a) map[a][j]=c; map[a][j]=c;
            for(b=j-1;map[i][b]==c;--b) map[a][b]=c; break;
        }
    }
    for(int i=1;i<=nRow;++i)
        cout<<map[i]+1<<endl;
    return 0;
}
