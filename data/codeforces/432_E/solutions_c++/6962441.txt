#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;

const int MaxN=107;

int n,m,mp[MaxN][MaxN];

int Calc(int x,int y)
{
    int i; bool u[MaxN];
    memset(u,false,sizeof(u));
    u[mp[x-1][y]]=true;
    u[mp[x+1][y]]=true;
    u[mp[x][y-1]]=true;
    u[mp[x][y+1]]=true;
    for(i=1;i<=26;i++)
        if(!u[i]) break;
    return i;
}

void Color(int x,int y)
{
     if(y>m) x++,y=1;
     if(x>n) return;
     if(mp[x][y]) {Color(x,y+1);return;}
     int c=Calc(x,y),p=1;
     while(true)
     {
           if(x+p>n||y+p>m||mp[x][y+p]||Calc(x,y+p)!=c) break;
           p++;
     }
     for(int i=0;i<p;i++)
         for(int j=0;j<p;j++) mp[x+i][y+j]=c;
     Color(x,y+1);
     return;
}

int main()
{
    int i,j;
    scanf("%d%d",&n,&m);
    Color(1,1);
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
            printf("%c",64+mp[i][j]);
        printf("\n");
    }
    return 0;
}
