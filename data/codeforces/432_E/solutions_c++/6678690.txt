#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 102
char a[N][N];
int n,m;

char get(int i,int j){
  if (a[i][j]) return a[i][j];
  char c='A';
  while (a[i-1][j]==c || a[i][j-1]==c || a[i+1][j]==c || a[i][j+1]==c) ++c;
  return c;
}

int main(){
  int i,j,k,x,y;
  scanf("%d%d",&n,&m);
  for (i=1;i<=n;++i){
    for (j=1;j<=m;++j){
      char now=get(i,j);
      putchar(now);
      if (a[i][j]) continue;
      k=1;
      while (i+k<=n && j+k<=m && get(i,j+k)==now) ++k;
      for (x=0;x<k;++x)
        for (y=0;y<k;++y)
          a[i+x][j+y]=now;
    }
    puts("");
  }
  return 0;
}
