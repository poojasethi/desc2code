#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
#include <string.h>
#include <string>
#include <map>
#include <set> 
using namespace std;

char A[101][101];
int find(int x,int y)
{
    if (A[x][y]) return A[x][y];
    for (int i='A';;i++)
        if (A[x][y-1]!=i&&A[x][y+1]!=i&&A[x-1][y]!=i&&A[x+1][y]!=i) 
            return i;
}
int main()
{
	//freopen("input","r",stdin);
	//freopen("output","w",stdout);
    memset(A,0,sizeof(A));
    int m, n;
    cin>>n>>m;
    for (int i=1;i<=n;i++)
    {
        for (int j=1;j<=m;j++)
        {
            int k=0;
            char t;
            t=find(i,j);
            cout<<t;
            if (A[i][j]) continue;
            while (i+k<=n&&j+k<=m&&find(i,j+k)==(int)t) k++;
            for (int ii=i;ii<i+k;ii++)
                for (int jj=j;jj<j+k;jj++) A[ii][jj]=t;
        }
        cout<<endl;
    }
	return 0;
}
