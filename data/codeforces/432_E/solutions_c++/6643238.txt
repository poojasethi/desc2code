#include <stdio.h>
#include <string.h>
#include <iostream>
using namespace std;
int N,M;
char a[105][105];
char lowest(int i, int j){
    char ch = 'A';

    while(a[i-1][j] == ch || a[i+1][j] == ch || a[i][j-1] == ch || a[i][j+1] == ch)
        ch++;
    return ch;
}
int main(){

    int i,j;
    char ch;

    cin >> N >> M;

    for(i = 1; i <= N; i++)
        for(j = 1; j <= M; j++)
            a[i][j] = '#';

    int d,x,y;

    for(i = 1; i <= N; i++){
        for(j  = 1; j <= M ; j++){
            if(a[i][j] != '#') continue;
            ch = lowest(i,j);
            d = 1;


            for(; i+d <= N && j+d <=M && a[i][j+d] =='#' && lowest(i,j+d)==ch; d++);

            //printf("%d %d %d %c\n",i,j,d,ch);
            for(x = i; x < i+d; x++)
                for(y = j; y < j+d; y++)
                    a[x][y] = ch;

        }
    }

    for(i = 1; i <= N; i++){
        for(j = 1; j <= M; j++){
            printf("%c",a[i][j]);
        }
        printf("\n");
    }

    return 0;
}
