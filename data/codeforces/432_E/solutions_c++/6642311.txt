#define N 105
int n,m;
char mm[N][N];
char test(int x,int y)
{
	if(mm[x][y]!=-1) return mm[x][y];
	char ret=0;
	for(; ret==mm[x][y-1] || ret==mm[x][y+1] || ret==mm[x-1][y] || ret==mm[x+1][y]; ret++) ;
	return ret;
}
int main()
{
#ifndef ONLINE_JUDGE

#endif // ONLINE_JUDGE
	scanf("%d%d",&n,&m);
	memset(mm,-1,sizeof(mm));
	for(int i=1; i<=n; i++)
	{
		for(int j=1; j<=m; j++)
		{
			char out=test(i,j);
			printf("%c",out+'A');
			if(mm[i][j]!=-1) continue;
			int gg;
			for(gg=1; i+gg<=n && j+gg<=m && test(i,j+gg)==out; gg++) ;
			for(int p=0; p<gg; p++)
				for(int q=0; q<gg; q++)
			{
				mm[p+i][q+j]=out;
			}
		}
		puts("");
	}

	return 0;
}
