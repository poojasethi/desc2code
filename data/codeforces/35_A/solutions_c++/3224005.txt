#include <stdio.h>
int main()
{
	int n,i,a[3],b[3];
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d",&n);
	for (i=0;i<3;i++)
	{
		scanf("%d%d",&a[i],&b[i]);
		if (n==a[i]) n=b[i];
		else if (n==b[i]) n=a[i];
	}
	printf("%d\n",n);
	return 0;
}
