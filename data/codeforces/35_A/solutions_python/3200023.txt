op = open('input.txt','r')
ot = open('output.txt','w')

n = int(op.readline().strip())
ball = [1,2,3]
for _ in xrange(3):
    a,b = map(int,op.readline().split())
    a-=1
    b-=1
    tmp = ball[a]
    ball[a] = ball[b]
    ball[b] = tmp
op.close()
for i in range(3):
    if ball[i] == n:
        ot.write(str(i+1))
ot.close()
