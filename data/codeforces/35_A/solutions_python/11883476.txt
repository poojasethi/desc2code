fin=open("input.txt","r")
fout=open("output.txt","w")
ar=[0,0,0]
n=int(fin.readline())
ar[n-1]=1
(p,q)=map(int,fin.readline().split())
temp=ar[p-1]
ar[p-1]=ar[q-1]
ar[q-1]=temp
(p,q)=map(int,fin.readline().split())
temp=ar[p-1]
ar[p-1]=ar[q-1]
ar[q-1]=temp
(p,q)=map(int,fin.readline().split())
temp=ar[p-1]
ar[p-1]=ar[q-1]
ar[q-1]=temp
for i in range(0,3):
    if(ar[i]==1):
        fout.write(str(i+1))
        break
fin.close()
fout.close()
