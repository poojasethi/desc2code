import sys

input = open('input.txt')
output = open('output.txt','w')
init = int(input.readline())

a = [False]*3
a[init-1] = True

for i in xrange(3):
  s1, s2 = map(lambda x: int(x), input.readline().strip().split(' '))
  a[s1-1],a[s2-1] = a[s2-1],a[s1-1]

if a[0]:
  output.write('1')
elif a[1]:
  output.write('2')
else:
  output.write('3')
output.flush()
