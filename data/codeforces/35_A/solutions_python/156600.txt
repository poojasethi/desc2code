#!/usr/bin/python

cin = open("input.txt", "r")
cout = open("output.txt", "w")

p = [False, False, False]
k = int(cin.readline()) - 1
p[k] = True
for i in range(0, 3):
	a, b = map(int, cin.readline().split())
	a, b = a - 1, b - 1
	p[a], p[b] = p[b], p[a]
for i in range(0, 3):
	if p[i]:
		cout.write(str(i + 1))

cin.close()
cout.close()
