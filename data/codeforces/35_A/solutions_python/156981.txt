f1 = open("input.txt", "r")
lines=f1.readlines()
f1.close()
balls=[0,0,0]
balls[int(lines[0])-1]=1
for i in range(1,4):
    move=lines[i].split()
    x=balls[int(move[0])-1]
    balls[int(move[0])-1]=balls[int(move[1])-1]
    balls[int(move[1])-1]=x
f2 = open("output.txt","w",1)
f2.write(str(balls.index(1)+1))
f2.close()
