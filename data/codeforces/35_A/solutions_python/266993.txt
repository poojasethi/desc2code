fin = open("input.txt")
fout = open("output.txt", 'w')

data = fin.readlines()
ball = int(data[0])
for x in xrange(1, 4):
    nums = [int(y) for y in data[x].split()]
    if nums[0] == ball:
        ball = nums[1]
    elif nums[1] == ball:
        ball = nums[0]
print >>fout, ball
fin.close()
fout.close()
