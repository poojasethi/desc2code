#coding: utf-8

def main():
    f = open("input.txt", "r")
    ls = f.readlines()
    f.close()
    idx = int(ls[0])
    for line in ls[1:]:
        inp = map(int, line.split())
        if idx in inp:
            idx = inp[inp.index(idx)-1]
    f = open("output.txt", "w")
    f.write(str(idx))
    f.close()
main()
