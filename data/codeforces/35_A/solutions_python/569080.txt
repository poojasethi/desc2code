input = open('input.txt','r')
output = open('output.txt','w')

def swap(A,i,j):
    A[i],A[j] = A[j],A[i]

def solve(n,swaps):
    P = [i+1==n for i in range(3)]
    for s in swaps:
        swap(P,s[0]-1,s[1]-1)
    return P.index(True) + 1

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

n = int(input.readline())
swaps = []
for _ in range(3):
    A = numbers_from_line()
    swaps.append((A[0],A[1]))

x = solve(n,swaps)
output.write('%s\n' % x)
