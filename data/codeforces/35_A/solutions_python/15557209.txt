def swap(A,i,j):
    t = A[i]
    A[i] = A[j]
    A[j] = t
def solve():
    import sys
    sys.stdin=open("input.txt")
    sys.stdout=open("output.txt",'w')
    n = input()-1
    A = [0,0,0]
    A[n] = 1
    for i in range(3):
        inp = map(int, raw_input().split())
        swap(A,inp[0]-1,inp[1]-1)
    print A.index(1)+1

solve()
