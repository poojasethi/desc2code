file = open('input.txt', 'r')
output = open('output.txt', 'w');
fd = False
id = int(file.readline())
ord = [ 0, 1, 2, 3 ]
try:
	while True:
		line = file.readline()
		lhs, rhs = map(int, line.split())
		t = ord[lhs]
		ord[lhs] = ord[rhs]
		ord[rhs] = t
except:
	for i in range(1, 4):
		if ord[i] == id:
			output.write(str(i))
