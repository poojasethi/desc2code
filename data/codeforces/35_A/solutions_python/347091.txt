import os
from os import popen

f=open("input.txt")
n=int(f.readline())
position=n-1
T=[0,1,2]

for i in range(3):
	u=[]
	line=f.readline()
	u=line.split()
	for a in range(len(u)):
		u[a]=int(u[a])
	index=T[u[0]-1]
	T[u[0]-1]=T[u[1]-1]
	T[u[1]-1]=index
f.close()

for j in range(len(T)):
	if T[j]==position:
		result=j+1

f=open("output.txt","w")
f.write(str(result)+'\n')
f.close()
