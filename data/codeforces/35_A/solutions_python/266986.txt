fin = open("input.txt")
fout = open("output.txt", 'w')

a = [0] * 5

n = int(fin.readline())

a[n] = 1

for i in range(3):
    x, y = [int(l) for l in fin.readline().split()]
    a[x], a[y] = a[y], a[x]

for i in range(4):
    if a[i] == 1:
        fout.write(str(i))
        fin.close()
        fout.close()
        break
