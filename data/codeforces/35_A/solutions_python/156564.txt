f = open("input.txt", "r")
lines = [i.rstrip() for i in f.readlines()]
f.close()

idx = int(lines[0]) - 1
field = [1, 2, 3]

for line in lines[1:]:
    t = line.split(" ")
    idx1 = int(t[0]) - 1
    idx2 = int(t[1]) - 1

    tmp = field[idx1]
    field[idx1] = field[idx2]
    field[idx2] = tmp

f = open("output.txt", "w")
f.write(str(field[idx]))
f.close()
