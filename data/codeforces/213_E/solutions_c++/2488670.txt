#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
long long xx,sw,mo,bas,a[300000],b[300000],l[300000],r[300000],ne[300000],ll[300000],rr[300000],p[300000];
int i,n,m,ans,x,tt,k;
bool cm1(int x,int y)
{
	return ((ll[x]==0)||a[y]>a[y-x+ll[x]])&&((rr[x]==0)||a[y]<a[y-x+rr[x]]);
}
bool cm2(int x,int y)
{
	return ((ll[x]==0)||b[y]>b[y-x+ll[x]])&&((rr[x]==0)||b[y]<b[y-x+rr[x]]);
}
int main()
{
	//freopen("1.in","r",stdin);
	//freopen("1.out","w",stdout);
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++) scanf("%d",&p[i]),a[p[i]]=i,l[i]=i-1,r[i]=i+1;
	for(i=1;i<=m;i++) scanf("%d",&x),b[x]=i;
	for(i=n;i>=1;i--)
	{
		ll[i]=p[l[a[i]]];rr[i]=p[r[a[i]]];
		r[l[a[i]]]=r[a[i]];
		l[r[a[i]]]=l[a[i]];
	}
	for(i=2;i<=n;i++)
	{
		while (k&&(!cm1(k+1,i)))k=ne[k];
		ne[i]=++k;
	}k=0;
	for(i=1;i<=m;i++)
	{
		while (k&&(!cm2(k+1,i))) k=ne[k];
		k++;
		if (k==n)
			ans++,k=ne[k];
	}
	printf("%d",ans);
	return 0;
}
