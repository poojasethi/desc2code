n, t1, t2, k = map(int, raw_input().split())
out = []
for i in xrange(n):
    v, u = map(int, raw_input().split())
    res1 = t1 * v * (100 - k) * 0.01 + t2 * u
    res2 = t1 * u * (100 - k) * 0.01 + t2 * v
    if res1 > res2:
        out.append((-res1, i + 1))
    else:
        out.append((-res2, i + 1))

out.sort()
for i in xrange(n):
    print "{0} {1:.2f}".format(out[i][1], -out[i][0])
