n, t1, t2, k = map(int, raw_input().split())
res = []
for i in xrange(n):
    a = map(int, raw_input().split())
    a.sort()
    res.append([max(a[1] * t1 * (100 - k) + a[0] * t2 * 100, a[0] * t1 * (100 - k) + a[1] * t2 * 100), i+1])
def cmp_(a, b):
    if a[0] == b[0]: return cmp(a[1], b[1])
    else: return cmp(b[0], a[0])
res.sort(cmp = cmp_)
for i in xrange(n):
    print res[i][1], "%.2f" % (res[i][0] / 100.0)
