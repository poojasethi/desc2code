#!/usr/bin/env python
import sys

n, t1, t2, k = map(int, raw_input().split())
res = []
for i in xrange(n):
    a1, a2 = map(float, raw_input().split())
    res.append( (-max(a1*t1*(100-k)/100 + a2*t2, a2*t1*(100-k)/100 + a1*t2), i+1) )
print "\n".join(["%d %.2f"%(i[1],-i[0]) for i in sorted(res)])
