from fractions import Fraction as frac
class pair:
    def __init__(self,a,b):
        self.a,self.b=a,b
    def __lt__(self,other):
        return self.a<other.a if self.b==other.b else self.b>other.b
n,t1,t2,k=map(int,raw_input().split())
k=frac(100-k,100)
v,u=list(),list()
for i in range(n):
    a,b=map(int,raw_input().split())
    v.append(a)
    u.append(b)
for i in sorted(pair(i+1,max(k*v[i]*t1+u[i]*t2,k*u[i]*t1+v[i]*t2)) for i in range(n)):
    print "%d %.2f" % (i.a,i.b)
