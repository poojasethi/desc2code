n,t1,t2,k=map(int,raw_input().split())
l,ans=[],[]
for i in range(n):
    w=map(int, raw_input().split())
    l.append(w)

for i in range(n):
    a1=l[i][0]*t1
    ad=a1-(a1*k/100.0)+(l[i][1]*t2)
    b1=(l[i][1]*t1)
    bd=b1-(b1*k/100.0)+(l[i][0]*t2)
    ans.append((max(ad,bd),-(i+1)))
ans.sort()
for i in range(-1,-n-1,-1):
    print -(ans[i][1]), '%.2lf' % ans[i][0]
