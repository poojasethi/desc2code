import sys

lines = sys.stdin.readlines()

n, t1, t2, k = [int(tok) for tok in lines[0].split()]

t1 = t1 * (100.0 - k) / 100

maxfirst = True if t1 >= t2 else False

heights = []
for i in range(1, n + 1):
    a, b = sorted([int(tok) for tok in lines[i].split()])
    if maxfirst:
        a, b = b, a
    heights.append((-(t1 * a + t2 * b), i))

heights.sort()

for height, idx in heights:
    print idx, "%.2f" % (-1 * height)
