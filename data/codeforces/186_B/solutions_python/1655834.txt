def growth(t1, t2, a, b, k):
    return a * t1 * (100.0 - k) / 100 + b * t2


n, t1, t2, k = map(int, raw_input().split())
res = []
for i in xrange(n):
    a, b = map(int, raw_input().split())
    gro = max(growth(t1, t2, a, b, k),
              growth(t1, t2, b, a, k))
    res.append((gro, -i))
res.sort(reverse=True)
for r in res:
    print "%d %.2f" % (-r[1] + 1, r[0])
