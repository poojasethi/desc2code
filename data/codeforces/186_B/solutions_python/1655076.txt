import sys
import math

def input():
    return tuple(map(lambda x: int(x), sys.stdin.readline().split()))

n, t1, t2, k = input()
a, b = list(), list()
for i in xrange(n):
    ai, bi = input()
    a.append(ai)
    b.append(bi)
h = []
for i in xrange(n):
    h1 = a[i] * t1 * (100 - k) / 100.0 + b[i] * t2
    h2 = b[i] * t1 * (100 - k) / 100.0 + a[i] * t2
    h.append((max([h1, h2]), i + 1))
for m in sorted(h, key=lambda m: -m[0]):
    print '%d %.2f' % (m[1], m[0])
