n, t1, t2, k = map(int, raw_input().split())

data = []

for i in xrange(n):
    a, b = map(int, raw_input().split())
    red = (100 - k) / 100.0
    res = max(t1 * a * red + t2 * b, t1 * b * red + t2 * a)
    data.append([i+1, res])

data.sort(key=lambda x: x[1], reverse=True)

for d in data:
    print d[0], '{:.2f}'.format(d[1])
