n,t1,t2,k=map(int,raw_input().split(" "))
k=float(k)/100
h=[]
for x in range(0,n):
    h.append(map(int,raw_input().split(" ")))
#
final=[]
for x in range(0,n):
    p=h[x][0]*t1*(1-k)+h[x][1]*t2
    q=h[x][1]*t1*(1-k)+h[x][0]*t2
    r=[max(p,q),-(x+1)]
    final.append(r)   
final.sort()
#jab height same hogi to wo last mein hoga jo pehle hoga given position mein
for x in range(n-1,-1,-1):
    print (-final[x][1]),"%.2f"%final[x][0]
