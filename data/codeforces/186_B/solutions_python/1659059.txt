#!/usr/bin/env python

if __name__ == "__main__":
    N, T1, T2, K = map( int, raw_input().split() );
    X = [0 for i in range(N)]
    for i in range(N):
        A, B =map(int, raw_input().split())
        X[i] = N - i + max( (100 - K)*T1*A+100*B*T2, (100-K) * T1*B + 100 * T2 * A)* 1000000000 ;
    X = sorted(X)
    for i in range(N):
        S = str(X[N-i-1] / 100000000000)+".";
        if (X[N -i -1]/1000000000) % 100 < 10:
            S += "0";
        S+=str((X[N - i -1]/1000000000) % 100);
        print N-(X[N - i - 1] % 100000) + 1, S
