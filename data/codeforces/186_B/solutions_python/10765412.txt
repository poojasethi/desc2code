n, t1, t2, k = map(int, raw_input().split())

def calc(a, b):
        return a*t1*(100-k)+b*t2*100

R = []
for i in xrange(n):
        a, b = map(int, raw_input().split())
        R.append((-max(calc(a, b), calc(b, a)), i+1))
R.sort()

for v in R:
        print "%d %d.%02d" % (v[1], (-v[0])/100, (-v[0])%100)
