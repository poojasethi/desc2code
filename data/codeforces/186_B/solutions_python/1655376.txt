def cmp(x, y):
	if x[0]<y[0]:
		return 1
	else:
		if x[0]>y[0]:
			return -1
		else:
			if x[1]>y[1]:
				return 1
			else:
				return -1

n, t1, t2, k = map(int, raw_input().split())
results = []
for i in range(n):
	v1, v2 = map(int, raw_input().split())
	result = max((v1*t1*(100-k))/100.+v2*t2, (v2*t1*(100-k))/100. + v1*t2)
	results.append([result, i])
results.sort(cmp)
for x in results:
	print("%d %.2f" % (x[1]+1, x[0]))
