n, t1, t2, k = map(int, raw_input().split(' '))
h = []
for i in xrange(n):
    v, u = map(int, raw_input().split(' '))
    a = t1 * v * (100.0 - float(k)) + t2 * u * 100
    b = t1 * u * (100.0 - float(k)) + t2 * v * 100
    h.append((max(a, b), -i - 1))
h.sort()
h.reverse()
for i in xrange(len(h)):
    print -h[i][1], '%.2f' % (float(h[i][0]) / 100.0)
