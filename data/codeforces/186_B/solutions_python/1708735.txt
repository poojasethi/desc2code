def fun(x,y,t1,t2,k):
  return max(x*t1*(100.0-k)/100.0 + y*t2,y*t1*(100.0-k)/100.0 + x*t2)

n, t1, t2, k = map(int, raw_input().split())

l = []
pos = 1
for i in range(n):
  x, y = map(int, raw_input().split())
  l += [(fun(x,y,t1,t2,k), pos)]
  pos += 1

l.sort(key=lambda d: (-d[0], d[1]))
print '\n'.join(["%d %.2f" % (a[1], a[0]) for a in l])
