(n, t1, t2, k) = [int(i) for i in raw_input().split()]
speeds = {}
for i in range(0, n):
	(a, b) = [int(j) for j in raw_input().split()]
	x = t1 * a * (100 - k) / 100.0 + t2 * b
	y = t1 * b * (100 - k) / 100.0 + t2 * a
	speeds[i] = max(x, y)
for k, v in sorted(speeds.items(), key = lambda x: x[1], reverse = True):
	print "%d %0.2f" % (k + 1, v)
