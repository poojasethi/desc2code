def main():
	n,t1,t2,k=[int(ele) for ele in raw_input().split()]
	scores=[]
	for i in range(n):
		v1,v2=[int(ele) for ele in raw_input().split()]
		scores.append((max(v1*t1-(float(v1*t1*k)/100)+v2*t2,v2*t1-(float(v2*t1*k)/100)+v1*t2),n-i))
	scores=sorted(scores,reverse=True)
	for ele in scores:
		print "%d %.2f"%(abs(ele[1]-n)+1,ele[0])
if __name__=="__main__":
	main()
