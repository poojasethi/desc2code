#include <cstdio>
#include <algorithm>
#include <utility>
using namespace std;
pair <int,int> ans[1001];
int main(){
    int n,t1,t2,k,a,b;
    scanf("%d%d%d%d",&n,&t1,&t2,&k);
    for(int i=0;i<n;i++){
        scanf("%d%d",&a,&b);
        ans[i]=make_pair(-max(a*t1*(100-k)+b*t2*100,b*t1*(100-k)+a*t2*100),i+1);
    }
    sort(ans,ans+n);
    for(int i=0;i<n;i++)
       printf("%d %.2f\n",ans[i].second , -0.01*ans[i].first );
   return 0;
}
