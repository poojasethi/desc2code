#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main(){
    int n,t1,t2,k;
    cin>>n>>t1>>t2>>k;
    vector<pair<long long,int> > R(n);
    for (int i = 0, u, v; i < n; i++){
        R[i].second = -(i+1);
        cin>>u>>v;
        R[i].first = max(t1*u*(100-k)+t2*v*100,t1*v*(100-k)+t2*u*100);
    }
    sort(R.rbegin(), R.rend());
    for (int i = 0; i < n; i++)
        cout<<-R[i].second<<" "<<R[i].first/100<<"."<<R[i].first%100/10<<R[i].first%10<<endl;
    //system("PAUSE");
    return 0;
}
