#include <bits/stdc++.h>
using namespace std;

int n, t1, t2, a, b;
double k;
vector<pair<double, int> > A;
int main() {

	cin >> n >> t1 >> t2 >> k;
	double K = ( 100 - k ) / 100;
	for ( int i = 0 ; i < n; i++ ) {
		scanf("%d %d", &a, &b);
		double h = max( a * t1 * K + b * t2, b * t1 * K + a * t2 );
		A.push_back(make_pair(h, -i ));
	}
	sort( A.begin(), A.end());
	cout.precision(2);
	for (int i = A.size() - 1; i >= 0; i--)
		cout << fixed << abs(A[i].second) + 1 << " " << A[i].first << endl;
	return 0;
}
