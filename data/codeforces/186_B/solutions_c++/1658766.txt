#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
using namespace std;
typedef pair<double,int> P;
vector<P> rank;

int main()
{
	double n,t1,t2,k;
	cin>>n>>t1>>t2>>k;
	double par=(100-k)/100;
	double a,b;
	for(int i=0;i<n;i++)
	{
		cin>>a>>b;
		double res1,res2,res;
		res1=t1*a*par*1000+t2*b*1000;
		res2=t1*b*par*1000+t2*a*1000;
		res=max(res1,res2);
		rank.push_back(P(res*(-1)/1000,i));
	}
	sort(rank.begin(),rank.end());
	for(int i=0;i<n;i++)
	{
		printf("%d %.2f\n",rank[i].second+1,rank[i].first*(-1));
	}
	return 0;
}
