#include <iostream>
#include <vector>
#include <cstdio>
#include <algorithm>
using namespace std;
vector<pair<double,int> >a;
int main()
{
    int n;
    cin>>n;
    int i;
    double x,y,t1,t2,k;
    cin>>t1>>t2>>k;
    for(i=0;i<n;i++)
    {
        cin>>x>>y;
        a.push_back(make_pair(-max((x*t1)-(x*t1)*k/100+y*t2,(y*t1)-(y*t1)*k/100+x*t2),i+1));
    }
    sort(a.begin(),a.end());
    for(i=0;i<n;i++)
        printf("%d %.2lf\n",a[i].second,-a[i].first);
    return 0;
}
