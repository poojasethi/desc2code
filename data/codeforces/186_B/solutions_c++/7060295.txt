#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

int n,t1,t2,k;

int main()
{
  cin>>n>>t1>>t2>>k;
  vector<pair<int,int> > v;
  for (int i=0;i<n;i++) {
    int a,b;
    cin>>a>>b;
    int l1=t1*a*(100-k)+t2*b*100;
    int l2=t1*b*(100-k)+t2*a*100;
    v.push_back(pair<int,int> (-max(l1,l2),i+1));
  }
  sort(v.begin(),v.end());
  for (int i=0;i<int(v.size());i++) {
    int val=-v[i].first;
    int pos=v[i].second;
    cout<<pos<<" "<<val/100<<"."<<val/10%10<<val%10<<endl;
  }
}
