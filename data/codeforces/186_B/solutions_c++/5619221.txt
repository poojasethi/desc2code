#include <iostream>
#include <functional>
#include <algorithm>
#include <iomanip>
using namespace std;
int main() {
	cout << fixed << setprecision(2);
	int n,t1,t2,a,b,k;
	pair <int,int> ans[1000];
	cin >> n >> t1 >> t2 >> k;
	for (int i=0;i<n;i++)
	{
		cin >> a >> b;
		ans[i]=make_pair(-max(a*t1*(100-k)+b*t2*100,b*t1*(100-k)+a*t2*100),i+1);
	}
	sort(ans,ans+n);
	for (int i=0;i<n;i++) {
		cout << ans[i].second << " " << double(-1*ans[i].first/100.00) << endl;
	}
	return 0;
}
