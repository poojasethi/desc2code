#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
pair<double,int> P[1010];
int n,t1,t2,k;
int a,b;
int main()
{
    while(scanf("%d%d%d%d",&n,&t1,&t2,&k)!=EOF)
    {
        for(int i=0;i<n;i++)
        {
            scanf("%d%d",&a,&b);
            P[i]=make_pair(-max(a*1.0*(1-k*1.0/100)*t1+b*1.0*t2,b*1.0*(1-k*1.0/100)*t1+a*1.0*t2),i+1);
        }
        sort(P,P+n);
        for(int i=0;i<n;i++)
            printf("%d %.2lf\n",P[i].second,-P[i].first);
    }
    return 0;
}
