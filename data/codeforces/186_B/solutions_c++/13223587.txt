#include<bits/stdc++.h>
using namespace std;
typedef struct po{
int x,in;
}P;
int cmp(P fir, P sec)
{
if(fir.x > sec.x)return 1;
if(fir.x < sec.x)return 0;
return fir.in < sec.in;
}
P a[100005];
int main()
{
int n,t1,t2,k;
int x,y;
scanf("%d%d%d%d",&n,&t1,&t2,&k);
int i;
for(i=0;i<n;i++)
{
a[i].in = i+1;
scanf("%d%d",&x,&y);
a[i].x = max(t1*x*(100-k) + 100*y*t2,t1*y*(100-k)+100*x*t2);
}

sort(a,a+n,cmp);

for(i=0;i<n;i++)
printf("%d %.2f\n",a[i].in,a[i].x/100.0);
return 0;
}
