#include <iostream>
#include <algorithm>
using namespace std;
#define forn(i,x,y) for (int (i) = (x); (i) <= (y);(i)++)
#define ll long long
#define double long double
int n,t1,t2; double x,y,per;pair <double,int> a[10005];
int main()
{
    cin >> n >> t1 >> t2 >> per;per=100-per;per/=100;
    forn(i,1,n){
       cin >> x >> y;
       a[i].first=- max ( x*t1*per+t2*y,y*t1*per+t2*x );a[i].second=i;}
    sort(a+1,a+1+n);
    forn(i,1,n){cout<<a[i].second<<' ';cout.precision(2);cout<<fixed;cout<<-a[i].first<< endl;}
    return 0;
}
