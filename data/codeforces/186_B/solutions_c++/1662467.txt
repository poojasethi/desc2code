#include <stdio.h>
main() {
  int n,t1,t2,k,a[1010],b[1010],p;
  double h1,h2,max,h[1010];
  scanf("%d%d%d%d",&n,&t1,&t2,&k);
  for (int i=1; i<=n; i++)
    scanf("%d %d",&a[i],&b[i]);
  for (int i=1; i<=n; i++) {
    h1=a[i]*t1*(1-k/100.0)+b[i]*t2;
    h2=b[i]*t1*(1-k/100.0)+a[i]*t2;
    if (h1>h2)
      h[i]=h1;
    else
      h[i]=h2;
  }
  for (int i=1; i<=n; i++) {
    max=0;
    for (int j=1; j<=n; j++)
      if (h[j]>max)
        max=h[j],
        p=j;
    printf("%d %.2f\n",p,h[p]);
    h[p]=0;
  }
  return 0;
}
