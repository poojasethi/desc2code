#include<cstdio>
#include<algorithm>
using namespace std;
typedef struct _{
	int i;
	double v;
	bool operator <(const _ &x)const{
		return v==x.v?i<x.i:v>x.v;
		}
	}mush;
mush M[1000];
int main(){
	int n,t1,t2,k,i,a,b;
	double r,a1,a2;
	scanf("%d%d%d%d",&n,&t1,&t2,&k);
	r=1-((double)k/100);
	for(i=0;i<n;i++){
		scanf("%d%d",&a,&b);
		a1=(double)(a*t1*r)+(double)(b*t2);
		a2=(double)(b*t1*r)+(double)(a*t2);
		M[i].i=i+1;
		M[i].v=max(a1,a2);
		}
	sort(M,M+n);
	for(i=0;i<n;i++)printf("%d %.2f\n",M[i].i,M[i].v);
}
