#include <bits/stdc++.h>

using namespace std;

vector<pair<double,int> > p;

int main() {
  int n;
  double a,b,t1,t2,k;
  
  cin >>n>>t1>>t2>>k;
  for (int i = 0;i < n;++i) {
    cin >>a>>b;
    double h1 = a*t1;
    h1 -= h1*k/100.;
    h1 += b*t2;
    double h2 = b*t1;
    h2 -= h2*k/100.;
    h2 += a*t2;
    p.push_back(make_pair(-max(h1,h2),i+1));
  }
  sort(p.begin(),p.end());
  for (int i = 0;i < n;++i) {
    printf("%d %.2lf\n",p[i].second,-p[i].first);
  }
  return 0;
}
