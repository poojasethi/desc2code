#include<iostream>
#include<math.h>
#include<algorithm>
#include<stdio.h>
using namespace std;
pair<double,int> a[1001];
double t1,t2,k,x,y;
int n;
int main()
{
    cin>>n>>t1>>t2>>k;
    for (int i=0;i<n;i++)
    {
        cin>>x>>y;

        a[i].first=-max((x*t1*(1-k/100)+y*t2),(y*t1*(1-k/100)+x*t2));
        a[i].second=i+1;
    }
    sort(a,a+n);
    for (int i=0;i<n;i++)
        {cout<<a[i].second<<" ";
            printf("%.2lf\n",-a[i].first);
        }
}
