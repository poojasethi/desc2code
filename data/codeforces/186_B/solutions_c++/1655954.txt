#include <iostream>
#include <vector>
#include <cstdio>
#include <string>
#include <cstring>
#include <cmath>
#include <algorithm>

using namespace std;

int main()
{
    int i,n,t1,t2,ai,bi;
    vector < pair <double, int> > v;
    double p;
    cin>>n>>t1>>t2>>p;
    for(i=0;i<n;++i){
        cin>>ai>>bi;
        v.push_back(make_pair(max(ai*t1-ai*t1*p/100+bi*t2,bi*t1-bi*t1*p/100+ai*t2),n-(i+1)));
    }
    make_heap(v.begin(),v.end());
    sort_heap(v.begin(),v.end());
    for(i=n-1;i>=0;--i) printf("%d %.2lf\n",n-v[i].second,v[i].first);
}
