#include<iostream>
#include<algorithm>
#include<vector>
#include<math.h>
#include<cstdio>
using namespace std;
int main(){
    int n,t1,t2,k,x,y;
    cin>>n>>t1>>t2>>k;
    vector<pair<double,int> > a;
    for(int i=0;i<n;i++){
        cin>>x>>y;
        a.push_back(pair<double,int>(-max(((float)x*t1*(100-k))/100+y*t2, ((float)y*t1*(100-k))/100+x*t2),i+1));
    }
    sort(a.begin(),a.end());
    for(int i=0;i<n;i++)
        printf("%d %.2f\n",a[i].second,-a[i].first);
    return 0;
}
