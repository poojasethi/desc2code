#include <cstdio>
#include <utility>
#include <algorithm>
using namespace std;

const int N = 10000;

typedef pair<int, int> pi;

pi dt[N];

int main ()
{
	int n, t1, t2, k;
	scanf("%d%d%d%d", &n, &t1, &t2, &k);
	for (int i = 1; i <= n; ++i)
	{
		int a, b;
		scanf("%d%d", &a, &b);
		int r1 = a * t1 * (100 - k) + b * t2 * 100,
			r2 = b * t1 * (100 - k) + a * t2 * 100;
		dt[i] = pi(-max(r1, r2), i);
	}
	sort(dt + 1, dt + n + 1);
	for (int i = 1; i <= n; ++i)
		printf("%d %.2lf\n", dt[i].second, (double)(-dt[i].first) / 100.0);
}
