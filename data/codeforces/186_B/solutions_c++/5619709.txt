#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int n,t1,t2,k;
double a,b;
vector<pair<double,int> > ans;
int main()
{
    cin >> n >> t1 >> t2 >> k;
    for(int i = 0 ; i < n; i++)
    {
            cin >> a >> b;
            ans.push_back(make_pair(-max((a * t1) * (100 - k) / 100 + b * t2,(b * t1) * (100 - k) / 100 + a * t2),i + 1));
    }
    sort(ans.begin(),ans.end());
    for(int i = 0 ; i < n; i++)
            printf("%d %.2lf\n",ans[i].second,-ans[i].first);
    cin >> n;    
}
