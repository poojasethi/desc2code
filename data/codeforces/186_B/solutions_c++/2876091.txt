#include <iostream>
#include <iomanip>
#include <queue>

using namespace std;

double N, t1, t2, k;
priority_queue<pair<double, int> > pq;

int main(){
    cin >> N >> t1 >> t2 >> k;
    for(int i = 1; i <= N; ++i){
        double a, b;
        cin >> a >> b;
        pq.push(make_pair(max(a*t1*(1-k/100)+b*t2, b*t1*(1-k/100)+a*t2), -i));
    }
    while(!pq.empty()){
        pair<double, int> p = pq.top();
        pq.pop();
        cout << (-p.second) << " " << fixed << setprecision(2) << p.first << "\n";
    }
}
