
#include<cstdio>
#include<iostream>
#include<string>
#include<vector>
#include<map>
#include<set>
#include<algorithm>
#include<cmath>
using namespace std;
int main(){
	int n,t1,t2,k,i;
	int x[1010];
	cin>>n>>t1>>t2>>k;
	for(i=0;i<n;i++){
		int a,b;
		cin>>a>>b;
		int mx=max(a*t1*(100-k)+b*t2*100,b*t1*(100-k)+a*t2*100);
		x[i]=mx;
	}
	while(1){
		int mx=0,in=-1;
		for(i=0;i<n;i++)
			if(x[i]>mx){
				mx=x[i];
				in=i;
			}
		if(in==-1)
			break;
		printf("%d %.2f\n",in+1,mx/100.0);
		x[in]=-1;
	}
	return 0;
}
