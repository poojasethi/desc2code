#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;

int main(){
    int n;
    double t1, t2, k;
    cin >> n >> t1 >> t2 >> k;
    pair<double, int> v[1005];
    for (int i = 0; i < n; ++i) {
        double a, b;
        cin >> a >> b;
        // cout << b * t1 * k / 100<< endl;
        double d = max(a * t1 * (100-k)  / 100 + b * t2, b * t1 * (100-k)/ 100 + a * t2);
        v[i] = make_pair(-d, i+1);
    }
    sort(v, v+n);
    for (int i = 0; i < n; ++i) {
        printf("%d %.2f\n", v[i].second, -v[i].first);
    }
}
