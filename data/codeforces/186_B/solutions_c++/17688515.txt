#include <bits/stdc++.h>

using namespace std;

const int N = 1001;
vector <pair<double,int> > v;

main() {
	int n, t1, t2, k;
	cin >> n >> t1 >> t2 >> k;
	for (int i = 0; i < n; i++) {
		double a, b;
		cin >> a >> b;
		double x = (a * t1) * (100 - k) / 100 + b * t2;
		x = max(x, (b * t1) * (100 - k) / 100 + a * t2);
		v.push_back(make_pair(x, -i));
	}
	sort(v.begin(), v.end());
	cout << fixed << setprecision(2);
	for (int i = v.size() - 1; i > -1; i--)
		cout << -v[i].second + 1 << ' ' << v[i].first << endl;
}
