#include<stdio.h>
#include<algorithm>
typedef struct{int s,i;}ele;
ele e[100100];
int cmp(ele a,ele b){return a.s>b.s||a.s==b.s&&a.i<b.i;}
int main(){
	int n,t1,t2,k,a1,b1;
	scanf("%d %d %d %d",&n,&t1,&t2,&k);
	for(int i=1;i<=n;i++){
		scanf("%d %d",&a1,&b1);
		e[i]=(ele){std::max(t1*a1*(100-k)+t2*b1*100,t1*b1*(100-k)+t2*a1*100),i};
		}
	std::sort(e+1,e+n+1,cmp);
	for(int i=1;i<=n;i++){
		printf("%d %d.%d%d\n",e[i].i,e[i].s/100,e[i].s/10%10,e[i].s%10);
		}
	//while(getchar());
	return 0;
	}
