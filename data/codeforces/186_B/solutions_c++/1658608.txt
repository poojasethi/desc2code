#include<stdio.h>
#include<string>
#include<iostream>
#include<algorithm>
using namespace std;
int a,s,f,d,g,h,j,l,i,n,m;
double k,q;
pair<double,int> w[1001];
main(){
cin>>n>>a>>s>>d;
for(i=0;i<n;i++){
cin>>j>>l;
k=j*a-(double)j*a*d/100+l*s;//printf("%f",(double)(((j*a)*d/100)));
q=l*a-(double)l*a*d/100+j*s;
w[i].first=max(k,q);
w[i].second=(i+1)*-1;}
sort(w,w+n);
for(i=n-1;i>=0;i--){
printf("%d %.2f\n",w[i].second*-1,w[i].first);}
//system("pause");
}
