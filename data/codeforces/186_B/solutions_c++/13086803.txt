#include <bits/stdc++.h>
using namespace std;

int main()
{ vector <pair <double,int > >v;
    double n,t1,t2,k,a,b;
    cin>>n>>t1>>t2>>k;
    double per=1-((double)k/100);
    for(int i=0;i<n;i++)
    {
        cin>>a>>b;
        double var=max((a*t1*per)+(b*t2),(b*t1*per)+(a*t2))*-1;
        //cout<<var<<endl;
        v.push_back(make_pair(var,i+1));
    }
  sort(v.begin(),v.end());
  for(int i=0;i<v.size();i++)
  {
      cout<<v[i].second<<" "<<setprecision(2)<<fixed<<(v[i].first)*-1<<endl;
  }
    return 0;
}
