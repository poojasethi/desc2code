#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int n,t1,t2,k,a,b;
int cal(int v1,int v2){return v1*t1*(100-k)+v2*t2*100;}
int main(){
    cin>>n>>t1>>t2>>k;
    vector<pair<int,int> > v;
    for(int i=0;i<n;i++){
        scanf("%d%d",&a,&b);
        v.push_back(pair<int,int>(max(cal(a,b),cal(b,a)),n-i));
    }
    sort(v.begin(),v.end());
    for(int i=n-1;i>=0;i--)
        printf("%d %.2lf\n",n-v[i].second+1,v[i].first/100.0);
}
