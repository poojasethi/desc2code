#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
    int i,j,n,t1,t2,x,y,k;
    vector<pair<double,int> > v;
    scanf("%d%d%d%d",&n,&t1,&t2,&k);
    double per = 1.0-k/100.0;
    for(i=0;i<n;i++)
    {
        scanf("%d%d",&x,&y);
        double l = max(x*t1*per+y*t2*1.0,y*t1*per + x*t2*1.0);
        v.push_back(make_pair(l,-(i+1)));
    }
    sort(v.begin(),v.end());
    for(i=n-1;i>=0;i--)
        printf("%d %.2lf\n",-v[i].second,v[i].first);
    return 0;
}
