#include <iostream>
#include <cstdio>
#include <algorithm>
#include <map>
#include <vector>

using namespace std;

vector < pair < double , int > > V;

int main()
{
    int n, t1, t2, k, v1, v2;
    cin >> n >> t1 >> t2 >> k;
    for(int i = 0; i < n; i++){
        scanf("%d%d", &v1, &v2);
        V.push_back( make_pair( min( -v1*t1*(100.0-k)/100.0  - v2*t2, -v2*t1*(100.0-k)/100.0 - v1*t2 ),i+1));
    }
    sort( V.begin(), V.end());
    for(int i = 0; i < n; i++)
        printf("%d %.2lf\n", V[i].second, -V[i].first);
}
