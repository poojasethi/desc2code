#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int n, t1, t2, k;
	scanf("%d%d%d%d", &n, &t1, &t2, &k);
	double per = 1.0-k/100.0;
	int i;
	int a,b;
	vector<pair<double,int> > ans;
	for(i=0; i<n; ++i) {
		scanf("%d%d", &a, &b);
		double ht = max(a*t1*per+b*t2*1.0,b*t1*per + a*t2*1.0);
	       ans.push_back(make_pair(ht,n-i));	
	}
	sort(ans.begin(), ans.end());
	
	for(i=n-1; i>=0; --i) {
		printf("%d %0.2lf\n", n-ans[i].second+1, ans[i].first);
	}
	return 0;


}
