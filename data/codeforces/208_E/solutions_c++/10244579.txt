int n,m,x,y,t,f[N][17];
int d[N],l[N],r[N];
vector<int>e[N];
vector<int>c[N];
inline void dfs(int x,int y){
	d[x]=y;
	c[y].push_back(t);
	l[x]=t++;
	fe(i,e[x])dfs(*i,y+1);
	r[x]=t;
}

	dfs(0,0);
	for(int i=1;i<17;i++)
		for(int j=1;j<=n;j++)
			f[j][i]=f[f[j][i-1]][i-1];

			if(y>>i&1)

		printf("%d ",x?lower_bound(u.begin(),u.end(),r[x])-lower_bound(u.begin(),u.end(),l[x])-1:0);
	}
}
