#include <stdio.h>
#include <algorithm>
using namespace std;

int main(void) {
	int T, m, n, r1, c1, r2, c2;
	scanf("%d", &T);
	while(T--) {
		scanf("%d%d%d%d%d%d", &m, &n, &r1, &c1, &r2, &c2);
		int rr = m - abs(r1-r2);
		int cc = n - abs(c1-c2);
		long long ans = 0;
		if (rr > m/2 && cc > n/2) {
			ans = (m - rr) * 1LL * (n - cc) * 2;
		}
		else
			ans = m * 1ll * n - rr * 2LL * cc;
		printf("%I64d\n", ans);
	}
	return 0;
}
