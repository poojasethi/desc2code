
#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long ll;
typedef pair<int,int> pii;
#define f first
#define s second

int main(){
	int nn;scanf("%d",&nn);
	while(nn--){
		ll xs,ys;cin>>xs>>ys;
		ll x1,y1;cin>>x1>>y1;
		ll x2,y2;cin>>x2>>y2;

		ll xmove=xs-abs(x1-x2);
		ll ymove=ys-abs(y1-y2);

		ll xoverlap=max(0LL,2*xmove-xs);
		ll yoverlap=max(0LL,2*ymove-ys);

		cout<<xs*ys-2*xmove*ymove+xoverlap*yoverlap<<endl;
	}

	return 0;
}
