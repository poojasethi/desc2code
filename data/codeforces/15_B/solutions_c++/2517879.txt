#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

int T, m, n, x1, y1, x2, y2;

int main() {
	scanf("%d", &T);
	while (T--) {
		scanf("%d%d%d%d%d%d", &m, &n, &x1, &y1, &x2, &y2);
		int x = abs(x1-x2);
		int y = abs(y1-y2);
		x1 = 1+x;
		y1 = 1+y;
		x2 = m-x;
		y2 = n-y;
		printf("%I64d\n", m*(__int64)n - (m+1-x1)*(__int64)(n+1-y1) - (x2)*(__int64)(y2) + max(0,y2-y1+1)*(__int64)max(0,x2-x1+1));
	}
	return 0;
}
