#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        int n,m,X1,Y1,X2,Y2;
        scanf("%d%d%d%d%d%d",&n,&m,&X1,&Y1,&X2,&Y2);
        int d1 = abs(X1-X2);
        int d2 = abs(Y1-Y2);
        __int64 s = (__int64)max((n-2*d1),0)*max((m-2*d2),0);
        __int64 ret = (__int64)n*m-((__int64)2*(n-d1)*(m-d2)-s);
        printf("%I64d\n",ret);
    }

    return 0;
}
