#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	int t,n,m,x1,x2,y1,y2;
	long long dx,dy;
	cin>>t;
	while(t--)
	{
		cin>>n>>m>>x1>>y1>>x2>>y2;
		dx = abs(x1-x2);
		dy = abs(y1-y2);
		if((n-2*dx)>0&&(m-2*dy)>0)
			cout<<(long long)n*m - 2*(n-dx)*(m-dy) + (n-2*dx)*(m-2*dy)<<endl;
		else cout<<(long long)n*m - 2*(n-dx)*(m-dy)<<endl;
	}
	return 0;
}
