#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int T,x1,y1,x2,y2;
	long long s,ts,x,y,n,m;
	scanf("%d",&T);
	while(T--){
		scanf("%I64d%I64d%d%d%d%d",&n,&m,&x1,&y1,&x2,&y2);
		if(x1>x2)
			swap(x1,x2);
		if(y1>y2)
			swap(y1,y2);
		x=x2-x1,y=y2-y1;
		s=(n-x)*(m-y)*2;
		ts=0;
		if(n-2*x>0){
			if(m-2*y>0)
				ts=(n-2*x)*(m-2*y);
		}
		printf("%I64d\n",n*m-s+ts);
	}
	return 0;
}
