#include <cstdio>
#include <iostream>
#include <string>
#include <cstring>
#include <algorithm>
using namespace std;

int main(){
    //freopen("B.in","r",stdin);
	int cas;
	cin >> cas;
	while(cas--) {
		int n,m,x1,y1,x2,y2;
		cin >>n >> m >> x1 >> y1 >> x2 >> y2;
		long long dx = abs(x1 - x2);
		long long dy = abs(y1 - y2);
		long long t = 2*(n - dx)*(m - dy) - max(n - 2*dx, 0LL)*max(m - 2*dy, 0LL);
		cout << 1LL*n*m - t <<endl;
	}

    return 0;
}
