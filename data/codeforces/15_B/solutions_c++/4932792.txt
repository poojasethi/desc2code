#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	long long t, n, m, a, b, c, d;
	cin >> t;
	while(t--)
	{
		cin >> n >> m >> a >> b >> c >> d;
		long long dx = abs(a - c), dy = abs(b - d);
		long long rangx = n - dx, rangy = m - dy;
		cout << n * m - 2 * rangx * rangy + max(0LL, rangx * 2 - n) * max(0LL, rangy * 2 - m) << endl;
	}
	return 0;
}
