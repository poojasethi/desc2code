#include <iostream>
#include <algorithm>
#include <windows.h>
using namespace std;

int main()
{
	int t;
	long long int n, m, x1, y1, x2, y2;

	cin >> t;
	while (t--)
	{
		cin >> n >> m >> x1 >> y1 >> x2 >> y2;
		int x = abs(x1 - x2);
		int y = abs(y1 - y2);
		cout << m * n - 2 * (n - x) * (m - y) + max(n - 2 * x, (long long)0) * max(m - 2 * y, (long long)0) << endl;
	}
	return 0;
}
