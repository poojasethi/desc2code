#include<stdio.h>
#include<string.h>
int i,j,k,l,r;
__int64 m,n,p,q,x1,x2,y1,y2,ans;
__int64 x,y;
__int64 min(__int64 xx,__int64 yy)
{
	if (xx<yy)
		return (xx);
	else
		return (yy);
}
int main()
{
	scanf("%d",&l);
	for (i=1;i<=l;i++)
	{
	scanf("%I64d%I64d%I64d%I64d%I64d%I64d",&n,&m,&x1,&y1,&x2,&y2);
	x=x1-x2;
	y=y1-y2;
	if (x<0)
		x=-x;
	if (y<0)
		y=-y;
	q=n-x;
	p=m-y;
	if (2*q<=n || 2*p<=m)
	{
		ans=m*n-p*q*2;	
	}
	else
	{
		ans=(x)*(y)*2;
	}
	printf("%I64d\n",ans);
	}
}
