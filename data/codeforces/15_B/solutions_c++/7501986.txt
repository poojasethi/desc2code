#include <cstdio>
using namespace std;
int n,m,x1,y1,x2,y2;
inline int Abs(int x)
{
	if (x<0) return -x;
	return x;
}
int main()
{
	int T;
	scanf("%d",&T);
	while (T--)
	{
		scanf("%d%d%d%d%d%d",&n,&m,&x1,&y1,&x2,&y2);
		long long L=n-Abs(x1-x2),R=m-Abs(y1-y2);
		long long ans=L*R*2;
		long long X=1+Abs(x1-x2),Y=1+Abs(y1-y2);
		if (L>=X && R>=Y) ans-=(1+L-X)*(1+R-Y);
		printf("%I64d\n",(long long)n*m-ans);
	}
	return 0;
}
