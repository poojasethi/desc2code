#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;

int main()
{
	long long t,n,m,x1,y1,x2,y2,x,y,p;
	cin>>t;
	while(t--)
	{
		cin>>n>>m>>x1>>y1>>x2>>y2;
		x=abs(x1-x2);
		y=abs(y1-y2);
		x1=n-x;		y1=m-y;
		x2=n-x1+1;	y2=m-y1+1;
		p=n*m-2*x1*y1;
		if(x2<=x1&&y2<=y1)	p+=(x1-x2+1)*(y1-y2+1);
		cout<<p<<endl;
	}
	return 0;
}
