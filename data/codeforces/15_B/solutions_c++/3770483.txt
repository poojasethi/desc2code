#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while(t--) {
        long long int x1, x2, y1, y2, n, m;
        cin >> n >> m >> x1 >> y1 >> x2 >> y2;

        long long dx = abs(x1-x2);
        long long dy = abs(y1-y2);
        long long ans =n*m -  2LL*(n - dx)*(m - dy);
        long long badx = max(- 2*dx + n, 0LL);
        long long bady = max(- 2*dy + m,0LL);
        cout << ans + badx*bady << "\n";
    }
}
