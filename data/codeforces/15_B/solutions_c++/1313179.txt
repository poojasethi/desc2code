#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;

int main() {
    int tn;
    cin >> tn;
    while (tn--) {
        int n, m, x1, y1, x2, y2;
        cin >> n >> m >> x1 >> y1 >> x2 >> y2;
        int dx = abs(x1 - x2);
        int dy = abs(y1 - y2);
        if (dx * 2 > n || dy * 2 > m) {
            cout << (ll)n * m - (ll)(n - dx) * (m - dy) * 2 << endl;
        } else {
            cout << (ll)dx * dy * 2 << endl;
        }
    }
}
