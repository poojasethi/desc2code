#include<stdio.h>
#include<string.h>
int main()
{
	int tt,ii;
	__int64 m,n,x1,x2,y1,y2,ans,t,x,y;
	scanf("%d",&tt);
	for(ii=1;ii<=tt;ii++)
	{
		scanf("%I64d%I64d%I64d%I64d%I64d%I64d",&n,&m,&x1,&y1,&x2,&y2);
		x=x1-x2; y=y1-y2;
		if(x<0) x=-x; 
		if(y<0) y=-y;
		ans=x*y*2;
		if(ans<0) ans=-ans;
		if(n-x<(n+1)/2||m-y<(m+1)/2) ans=n*m-(n-x)*(m-y)*2;
		printf("%I64d\n",ans);
	}
}
