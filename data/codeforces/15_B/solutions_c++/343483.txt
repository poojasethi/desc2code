#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;

int t;
long long int n,m,x[2],y[2];
long long int bad;

int main(){
 scanf("%d",&t);
 while(t--){
  scanf("%lld%lld%lld%lld%lld%lld",&n,&m,x,y,x+1,y+1);
  bad = n*m;
  bad -= 2*max((n-abs(x[0]-x[1])),0LL)*max((m-abs(y[0]-y[1])),0LL);
  bad += max((n-2*abs(x[0]-x[1])),0LL)*max((m-2*abs(y[0]-y[1])),0LL);
  cout << bad << endl;
 }
 return 0;
}
