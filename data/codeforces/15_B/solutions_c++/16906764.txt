#include<cstdio>
#include<cmath>
#include<cstdlib>
#include<algorithm>
using namespace std;
int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		long long int n,m,x1,x2,y1,y2;
		scanf("%lld%lld%lld%lld%lld%lld",&n,&m,&x1,&y1,&x2,&y2);
		long long int x=abs(x1-x2)+1,y=abs(y1-y2)+1;
		long long int d=(n-x+1)*(m-y+1)*2;
		if(n-x+1>=x&&m-y+1>=y)
			d-=(n-2*x+2)*(m-2*y+2);
		long long int ans=n*m-d;
		printf("%lld\n",ans);
	}
}
