#include <iostream>

using namespace std;

typedef long long int lint;

int main()
{
  int casos;
  cin>>casos;
  for (int cas=0;cas<casos;cas++) {
    lint n,m,i1,j1,i2,j2;
    cin>>n>>m>>i1>>j1>>i2>>j2;
    //i1--;
    //i2--;
    //j1--;
    //j2--;
    if (i1>i2) swap(i1,i2);
    if (j1>j2) swap(j1,j2);
    lint ii=i2-i1;
    lint jj=j2-j1;
    lint resta=0;
    if (n-ii>=1+ii and m-jj>=1+jj) resta=(n-ii-ii)*(m-jj-jj);
    cout<<n*m-(2*(n-ii)*(m-jj)-resta)<<endl;
  }
}
