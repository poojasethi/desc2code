#include <iostream>
using namespace std;

long long m,n,x1,y1,x2,y2,t;

int main(){
	cin>>t;
	for(int i=1;i<=t;i++){
		cin>>m>>n>>x1>>y1>>x2>>y2;
		if(2*max(x1-x2,x2-x1)<=m&&2*max(y1-y2,y2-y1)<=n)
		    cout<<2*max(x1-x2,x2-x1)*max(y1-y2,y2-y1)<<endl;
		else
		    cout<<m*n-2*(m-max(x1-x2,x2-x1))*(n-max(y1-y2,y2-y1))<<endl;
	}
	return 0;
}
