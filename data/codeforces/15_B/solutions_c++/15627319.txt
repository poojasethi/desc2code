#include<iostream>
#include <stdio.h>
#include <stdlib.h>
using namespace std;
main(){ long long t,i,k,k1,l1,l,j,n,m,a[5];
cin>>t;
for(i=1; i<=t; i++){
    cin>>n>>m; cin>>a[1]>>a[2]>>a[3]>>a[4];
    k=n-abs(a[1]-a[3]);
    l=m-abs(a[2]-a[4]);
    j=n*m-k*l*2; k1=0; l1=0;
    if (k>=abs(a[1]-a[3])+1) k1=k-abs(a[1]-a[3]);
    if (l>=abs(a[2]-a[4])+1) l1=l-abs(a[2]-a[4]);

    cout<<j+k1*l1<<endl;
}
}
