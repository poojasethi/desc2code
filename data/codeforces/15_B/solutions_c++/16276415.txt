#include<stdio.h>
#define dx(x1,x2) (x1>x2?x1-x2:x2-x1)
#define dy(y1,y2) (y1>y2?y1-y2:y2-y1)
#define max(a,b) (a>b?a:b)
int main()
{
    int t;
    __int64 n,m,x1,x2,y1,y2;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%I64d%I64d%I64d%I64d%I64d%I64d",&n,&m,&x1,&y1,&x2,&y2);
        printf("%I64d\n",n*m-2*(n-dx(x1,x2))*(m-dy(y1,y2))+max(0,n-2*dx(x1,x2))*max(0,m-2*dy(y1,y2)));
    }
    return 0;
}
