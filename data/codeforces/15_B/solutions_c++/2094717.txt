#include<iostream>
using namespace std;

int main() {
  int t;
  cin>>t;
  for (int ti=0;ti<t;ti++) {
    long long n,m,x1,y1,x2,y2;
    cin>>n>>m>>x1>>y1>>x2>>y2;
    if (x1>x2) swap(x1,x2);
    if (y1>y2) swap(y1,y2);
    long long answer = n*m;
    long long width = n+x1-x2;
    long long height = m+y1-y2;
    answer -= 2*width*height;
    if (width*2>n && height*2>m)
      answer += (width*2-n)*(height*2-m);
    cout<<answer<<endl;
  }
  return 0;
}
