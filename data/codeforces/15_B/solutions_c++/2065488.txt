#include<algorithm>
#include<iostream>
#include<cstdio>

using namespace std;

long long n , m , x1 , y1 , x2 , y2;

int main() {
    int T; scanf( "%d" , &T );
    while ( T-- ) {
        cin >> n >> m >> x1 >> y1 >> x2 >> y2;
        long long r = n - abs( x1 - x2 ) , c = m - abs( y1 - y2 );
        long long ans = n * m - r * c * 2 + max( 0LL , r * 2 - n ) * max( 0LL , c * 2 - m );
        cout << ans << endl;
    }
    return 0;
}
