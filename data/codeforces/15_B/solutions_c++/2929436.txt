#include <iostream>

using namespace std;

__int64 abs(__int64 x)
{
    return x > 0 ? x : -x;
}

int main()
{
    __int64 m,n,x1,y1,x2,y2,a,b;
    int t;
    cin >> t;
    while(t--)
    {
        cin >> n >> m >> x1 >> y1 >> x2 >> y2;
        a = abs(x1-x2);
        b = abs(y1-y2);
        cout <<n*m-2*(n-a)*(m-b)+max(n-2*a,(__int64)0)*max(m-2*b,(__int64)0)<< endl;
    }
    return 0;
}
