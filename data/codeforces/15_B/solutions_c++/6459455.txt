#include <iostream>
#include <math.h>
#include <algorithm>
#include <stdio.h>

using namespace std;

long long t,n,m,x,y,X,Y,ans;

main()
 {
  cin>>t;
  while (t--)
   {
    cin>>n>>m>>x>>y>>X>>Y;
    x=abs(X-x);
    y=abs(Y-y);
    X=x+1;
    Y=y+1;
    x=n-x;
    y=m-y;
    ans=0;
    if (X <= x && Y <= y) ans=(x-X+1)*(y-Y+1);
    cout<<n*m-(x*y)-(n-X+1)*(m-Y+1)+ans<<endl;
   }
 }
