t = int(raw_input())

for i in range(t):
    n, m, x1, y1, x2, y2 = map(int, raw_input().split())
    total = n * m
    xdis = abs(x2-x1)
    ydis = abs(y2-y1)
    notclearx = max(0, n - (2*xdis)) #max diagonal
    notcleary = max(0, m - (2*ydis)) # "

    total -= (2 * (n - xdis) * (m - ydis)) #sub what we can get
    total += (notclearx * notcleary)

    print total
