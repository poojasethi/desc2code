#! /usr/bin/python
import sys

#nacteni 1 cisla
cases = int(sys.stdin.readline()[:-1])
actual_case = 0

while actual_case < cases:
    # reading and so
    actual_case += 1
    
    #nacteni 6 cisel
    numbers = sys.stdin.readline()[:-1].split()
    n = int(numbers[0])
    m = int(numbers[1])
    x = int(numbers[2])
    y = int(numbers[3])
    xx = int(numbers[4])
    yy = int(numbers[5])
 
    #nove jsou v obdelniku
    nx = x - min(x,xx) + 1
    nxx = xx - min(x,xx) + 1
    ny = y - min(y,yy) + 1
    nyy = yy - min(y,yy) + 1

    posunx = n - max(nx,nxx) +1
    posuny = m - max(ny,nyy) +1

    if (n <= posunx*2) and (m <= posuny*2):
        pocet = n*m - (posunx*posuny*2 - (posunx*2-n)*(posuny*2-m))
    else:
        pocet = n*m - posunx*posuny*2

    print pocet
