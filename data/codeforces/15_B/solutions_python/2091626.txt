#!/usr/bin/python
import sys
sys.stdin.readline()

# find the area of arm 1 and arm 2 and then find the overlap
# the result = n*m - area1 - area2 + overlap
for line in sys.stdin:
	(n, m, x1, y1, x2, y2) = map(lambda(x): int(x), line.split())
	if x1 > x2:
		x1, x2 = x2, x1
	if y1 > y2:
		y1, y2 = y2, y1
	x1max = x1 + n - x2
	y1max = y1 + m - y2
	x2min = x2 - x1 + 1
	y2min = y2 - y1 + 1
	overlap = max(x1max - x2min + 1, 0) * max(y1max - y2min + 1, 0)
	print m*n - x1max*y1max - (n-x2min+1)*(m-y2min+1) + overlap
