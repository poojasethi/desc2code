#!/usr/bin/python
import sys

t = int(sys.stdin.next())
for ii in xrange(t):
    (n, m, x1, y1, x2, y2) = map(int, sys.stdin.next().strip().split (" "))
    ur = 0
    x = min([x1,x2])+min([n - x1, n - x2])
    y = min([y1,y2])+min([m - y1, m - y2])
    ur = n*m  + max([(2*x - n),0])*max([(2*y - m),0]) - 2*x*y
    print "%d" % (ur)
