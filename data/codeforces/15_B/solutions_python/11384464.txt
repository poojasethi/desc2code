t = input()
for i in xrange(t):
    n , m, x1, y1, x2, y2 = map(int, raw_input().split(' '))
    x1, x2 = min(x1, x2), max(x1, x2)
    y1, y2 = min(y1, y2), max(y1, y2)
    x2 -= x1 - 1
    y2 -= y1 - 1
    print n * m - 2 * (n - x2 + 1) * (m - y2 + 1) + max(0, (n - x2 + 1) * 2 - n) * max(0, (m - y2 + 1) * 2 - m)
