t = int(raw_input())


for x in xrange(t):
	n, m, x1, y1, x2, y2 = map(int, raw_input().split())

	width = abs(x2 - x1)
	height = abs(y2 - y1)

	chocolate = n * m

	x = n - 2 * width
	y = m - 2 * height

	chocolate -= 2 * (n - width) * (m - height)
	if x > 0 and y > 0:
		chocolate += x * y

	print chocolate
