for _ in xrange (input ()):
    n, m, x1, y1, x2, y2 = map (int, raw_input ().split ())
    dx, dy = abs (x1 - x2), abs (y1 - y2)
    s = 2 * dx * dy
    if dx + 1 == n and dy + 1 == m:
        print n * m - 2
    elif dx + 1 == n:
        print 2 * dy + m * (dx - 1)
    elif dy + 1 == m:
        print 2 * dx + n * (dy - 1)
    else:
        dn, dm = max (0, (dx - 1) - (n - (dx + 1))), max (0, (dy - 1) - (m - (dy + 1)))
        print m * dn + n * dm - dn * dm + 2 * (dx - dn) * (dy - dm)
