m = int(raw_input())

q = [(1, 1), (5, 6), (25, 31), (125, 156), (625, 781), (3125, 3906), (15625, 19531), (78125, 97656)]

fives = 0
for a, b in q[::-1]:
    while m >= b:
        if m/b != 5:
            m -= b
            fives += a
        else:
            fives = 0
            break

if fives > 0:
    print 5
    print ' '.join([str(5*fives+i) for i in range(5)])
else:
    print fives
