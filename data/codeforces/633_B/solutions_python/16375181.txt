#!/usr/bin/python2.7
def func(num):
	ret=0
	while num>=5:
		ret += num/5
		num /= 5
	return ret
n=input()
lo,hi=1,1000000000
while lo + 1 < hi:
	mid = (lo + hi) / 2
	vmid = func(mid)
	if vmid > n: hi = mid - 1
	else: lo = mid
if n != func(lo):
	print 0
else:
	v = lo / 5 * 5
	print 5
	print v,v+1,v+2,v+3,v+4
