def count(n):
    result = 0

    while n > 0:
        result += n / 5
        n /= 5

    return result


m = input()
answers = []

for i in xrange(1, 500000):
    c = count(i)
    if c == m:
        answers.append(i)
    if c > m:
        break


print len(answers)
print " ".join(map(str, answers))
