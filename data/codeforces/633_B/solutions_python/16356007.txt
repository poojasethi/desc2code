hi = 100010
while lo + 1 < hi:
    mid = x = (lo + hi) / 2
    k = 0
    while x:
        k += x
        x /= 5
    if k >= m:
        hi = mid
    else:
        lo = mid
k = 0
x = hi
while x:
    k += x
    x /= 5
if k == m:
    print 5

        print x + hi * 5,
