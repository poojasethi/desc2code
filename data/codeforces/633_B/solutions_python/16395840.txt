
def count(n,r):
    ret = 0
    while True:
        if n%r==0:
            n=n/r
            ret+=1
        else:
            break
    return ret


m = int(raw_input())

i=1
c2 = 0
c5 = 0
answer = []
while True:
    c2 += count(i,2)
    c5 += count(i,5)
    if min(c2,c5) == m:
        answer.append(i)
    elif min(c2,c5) > m:
        break
    i += 1

if len(answer) > 0:
    print len(answer)
    print ' '.join(map(str, answer))
else:
    print 0
