#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

def n5(n):
    ans = 0
    while True:
        r = n % 5
        if r != 0: break
        ans = ans + 1
        n /= 5
    return ans

n = ir()

ans = []
M = 1000006
s = 0
for i in range(1, M):
    s = s + n5(i)
    if s==n:
        ans.append(i)
    elif s>n:
        break

print len(ans)
if len(ans)>0:
    print ' '.join(map(str, ans))
