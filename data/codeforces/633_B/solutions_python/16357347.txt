import math
def main(n):
	fac = 5
	tz = 0
	buf = math.floor(n / fac)
	while(buf > 0):
		tz += buf
		fac *= 5
		buf = math.floor(n / fac)
	return tz

m = input()
res = []
n = 1
z = main(n)
# print ("Done")
while(z <= m):
	if z == m:
		res.append(n)
	n+=1
	z = main(n)
print(len(res))
s = ""
for x in res:
	s += str(x) + " "
print s
