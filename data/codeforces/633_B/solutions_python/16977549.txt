zeroes = int(input())
fives = 0

dp = []
def countfives(n):
 n = int(n)
 if n % 5 != 0:
  return 0

 if len(dp) >= n / 5:
  return dp[int(n / 5) - 1]
 
 ret = 1 + countfives(n / 5)
 #dp[int(n / 5)] = ret
 dp.append(ret);
 return ret

i = 1
ret = []

while fives <= zeroes:
 f = countfives(i)
 fives = fives + f
 
 if fives == zeroes:
  ret.append(i)
 
 i = i + 1

print(len(ret))
print(" ".join(str(s) for s in ret))
