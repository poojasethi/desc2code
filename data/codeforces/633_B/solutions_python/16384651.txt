def solve(n):
	z = 0
	while n > 0:
		#print n, z
		n /= 5
		z += n
	return z

# print solve(99)
n = input()

l= 0 
r = 1000000000
# print solve(5800)
while l < r:
	mid = (l+r)/2
	#print mid, l, r
	if solve(mid) < n:
		l = mid + 1
	else:
		r = mid

cnt = 0
ans = []
# print 'l is {0}, n is {1}'.format(l, n)
while solve(l) == n:
	cnt += 1
	ans.append(l)
	l += 1

print cnt
if cnt > 0:
	print " ".join(map(str, ans))
