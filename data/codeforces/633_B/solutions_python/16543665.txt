#! /usr/bin/python

# A simple solution

def two_and_fives(x):
    two = 0
    five = 0
    while x % 2 == 0:
        two += 1
        x /= 2
    while x % 5 == 0:
        five += 1
        x /= 5
    return [two, five]

m = int(raw_input())
num_2_5 = [0, 0]
r = []
i = 1
while True:
    [two, five] = two_and_fives(i)
    num_2_5[0] += two
    num_2_5[1] += five
    zeros = min(num_2_5)
    if zeros == m:
        r.append(i)
    elif zeros > m:
        break
    i += 1
print(len(r))
print(" ".join([str(e) for e in r]))
