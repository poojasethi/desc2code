#include <bits/stdc++.h>

using namespace std;

long long m, x, i = 1;

int main() {
	cin >> m;
	for (; x < m; i++) {
		long long p = i * 5;
		while (p % 5 == 0) {
			x++;
			p /= 5;
		}
	}
	i--;
	i *= 5;
	if (x == m) {
		cout << "5\n";
		for (int j = i; j < i + 5; j++)
			cout << j << " ";
	}
	else
		cout << 0;
	return 0;
}
