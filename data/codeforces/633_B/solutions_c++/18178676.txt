#include <cstdio>
int m;
int cnt = 0, tmp;
int main (){
	scanf("%d", &m);
	while (m > 0){
		cnt += 5;
		tmp = cnt;
		while (tmp % 5 == 0){
			tmp /= 5;
			m--;
		}
	}
	if (m == 0){
		printf("5\n");
		printf("%d %d %d %d %d\n", cnt, cnt + 1, cnt + 2, cnt + 3, cnt + 4);
	} else 
		printf("0\n");
	return 0;
}
