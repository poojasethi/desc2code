#include <iostream>

using namespace std;

int n = 1000000, m, a[1000000], t;

int main()
{
	cin >> m;
	while (n--)
	{
		int s = 5, r = 0, k = 8;
		while (k--)
		{
			r += n / s;
			s *= 5;
		}
		if (m - r == 0)
		{
			a[t] = n;
			t++;
		}
	}
	cout << t << endl;
	while(t--)
		cout << a[t] << " ";
	return 0;
}
