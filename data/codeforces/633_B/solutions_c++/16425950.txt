#include <stdio.h>

int i,n,tr,b,p;

int main()
{
	scanf ("%d",&n);
	tr=0;
	b=0;
	while (1)
	{
		tr=tr+5;
		p=tr;
		while (p%5==0)
		{
			b++;
			p=p/5;
		}
		if (b>=n)
		{
			break;
		}
	}
	if (b>n)
	{
		printf ("0");
	}
	else
	{
		printf ("5\n");
		for (i=tr;i<tr+5;i++)
		{
			printf ("%d ",i);
		}
	}
}
