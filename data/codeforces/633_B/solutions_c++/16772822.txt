#include <iostream>

using namespace std;

int main()
{
    int n,z,p,i;
    cin>>n;
    for(i=5;;i+=5)
    {
        z=0;
        p=5;
        while(i/p) z+=i/p,p*=5;
        if(z > n) {cout << "0"; return 0;}
        if(z==n) break;
    }
    cout << "5" <<endl;
    for(int j=i;j<i+5;j++) cout << j << " ";
    return 0;
}
