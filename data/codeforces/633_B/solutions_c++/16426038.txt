#include<iostream>

using namespace std;
long long a[1000001];
int main(){
	long long n,m=0,k,l=0;
	cin>>n;
	
	for(int i=5; i<1000001; i++)
	{
		k=i;
		
		while(k!=0){
			k/=5;
			l+=k;
		}
		
		if(l==n){a[m]=i;m++;}
		l=0;
		
	}	
	
	cout<<m<<endl;
	
	for(int i=0; i<m; i++)
		cout<<a[i]<<" ";
		
	cout<<endl;
		
	return 0;
}
