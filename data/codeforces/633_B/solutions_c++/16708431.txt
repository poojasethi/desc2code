#include <bits/stdc++.h>
using namespace std;

int n = 0;

int main()
{
	int i, k;
	scanf("%d", &k);
	for(i = 1; n < k; i++)
	{

		int a = i;
		while(a % 5 == 0 && a != 0)
		{
			n++;
			a /= 5;
		}
	}
	if(n != k)
		printf("0\n");
	else
	{
		printf("5\n");
		printf("%d %d %d %d %d\n", i-1, i, i+1, i+2, i+3);
	}
	return 0;
}
