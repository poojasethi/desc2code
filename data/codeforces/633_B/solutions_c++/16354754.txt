#include <bits/stdc++.h>

using namespace std;

int main()
{
	int	n,i,t=0;
	long long Sum=1;
	vector<int>	vec;
	scanf("%d",&n);
	for(i=1;;++i)
	{
		Sum*=i;
		Sum%=1000000000;
		while(Sum%10==0)t++,Sum/=10;
		if(t==n)vec.push_back(i);
		if(t>n)break;
	}
	printf("%d\n",(int)vec.size());
	for(i=0;i<(int)vec.size();++i)
		printf("%d ",vec[i]);
	return 0;
}
