#include<bits/stdc++.h>
using namespace std;
int n,m,two,five,x,ans[1000005],i;
int main(){
	scanf("%d",&m);
	for (n=1;;n++){
		for (x=n;x%2==0;x/=2) two++;
		for (x=n;x%5==0;x/=5) five++;
		if (min(two,five)>m) break;
		if (min(two,five)==m) ans[++*ans]=n;
  }printf("%d\n",*ans);
  for (i=1;i<=*ans;i++) printf("%d ",ans[i]);
}
