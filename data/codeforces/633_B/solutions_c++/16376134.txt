#include<bits/stdc++.h>
using namespace std;
int m,ans,k=0,temp;
int main()
{
	scanf("%d",&m);
	ans=4*m+5-((4*m)%5);
	while(k<m){
	k=0;
	temp=ans;
	while(ans/5!=0)
		{k+=ans/5; ans=ans/5;}
	if(k==m) break;
	ans=temp+5;
	}
	if(k>m) cout<<"0\n";
	else {cout<<"5\n"<<temp<<" "<<temp+1<<" "<<temp+2<<" "<<temp+3<<" "<<temp+4<<"\n";}
	return 0;
}
