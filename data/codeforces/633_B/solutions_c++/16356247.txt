#include <bits/stdc++.h>
using namespace std;
int main() {
	int m, i = 1, j, z = 0, cnt = 0;
	cin >> m;
	stringstream out;
	while(z <= m) {
		j = i;
		while(j % 5 == 0) {
			j /= 5;
			z++;
		}
		if(z == m) { 
			out << i << " ";
			cnt++;
		}
		i++;
	}
	cout << cnt << endl << out.str() << endl;
	return 0;
}
