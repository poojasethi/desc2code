#include <bits/stdc++.h>

using namespace std;


int m,d,c,z,RS[1000000],rs;
int main(){
	cin >> m;
	for(int i = 2;z <= m;i++){
		int a = i;
		while(a && a%2==0) d++,a/=2;
		while(a && a%5==0) c++,a/=5;
		
		z = min(d,c);
		if(z == m) RS[rs++] = i;
	}
	cout << rs << '\n';
	for(int i = 0;i<rs;i++) cout << RS[i] << ' ';
	return 0;
}
