#include <cstdio>
int n,sum;
int main(){
	scanf("%d",&n);
	for (int i=1;;i++){
		int num=1;
		int x=i;
		while (true){
			if (x%5==0) num++,x/=5;
			else break;
		}
		sum+=num;
		if (sum==n){
			printf("5\n");
			for (int j=5*i;j<5*(i+1);j++) printf("%d ",j);
			return 0;
		}
		if (sum>n) break;
	}
	printf("0");
	return 0;
}
