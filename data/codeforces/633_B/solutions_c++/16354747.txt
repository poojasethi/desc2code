#include <iostream>
using namespace std;
long long m,k,d,t,a[1000002],i;
int main() {
	cin>>m;
	k=1;
	while(true)
	{
		d=k/5+k/25+k/125+k/625+k/3125+k/15625+k/78125+k/390625+k/1953125+k/9765625;
		if(d==m)
		{
			t++;
			a[t]=k;
		}
		k++;
		if(d>m)
			break;
	}
	cout<<t<<endl;
	for(i=1;i<=t;i++)
		cout<<a[i]<<" ";
	return 0;
}
