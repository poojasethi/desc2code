#include <iostream>
using namespace std;
int main(){
        ios::sync_with_stdio(0);
	int a,n=0,j=0,i;
	cin>>a;
	while(n<a){
		j+=5;
		i=j;
		while(i%5==0){
		n++; i=i/5;
                }
	}
	if(n==a)
        cout << 5<< '\n' << j<< ' ' << j+1<< ' ' << j+2 << ' ' << j+3 << ' ' << j+4;
	else
	cout << 0;
        return 0;
}
