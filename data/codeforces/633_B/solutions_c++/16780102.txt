#include<cstdio>
using namespace std;

int tz(long long n)
{
	int z=0;
	while(n)
	{
		z+=n/5;
		n/=5;
	}
	return z;
}

int main()
{
	int m;
	scanf("%d",&m);
	int n=m*5;
	while(tz(n)>m)
	n-=5;
	if(tz(n)==m)
	{
		printf("5\n");
		printf("%d %d %d %d %d\n",n,n+1,n+2,n+3,n+4);
	}
	else
	printf("0\n");
	return 0;
}
