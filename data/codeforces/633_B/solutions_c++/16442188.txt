#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	int m;
	vector<int> v;
	cin>>m;
	for(int i=5;i<=1000000;i++)
	{
		int p=5;
		int res=0;
		while(p<=i)
		{
			res+=(i/p);
			p=p*5;
		}
		if(res==m)v.push_back(i);
	}
	cout<<v.size()<<endl;
	for(int i=0;i<v.size();i++)cout<<v[i]<<" ";
	return 0;
	}
