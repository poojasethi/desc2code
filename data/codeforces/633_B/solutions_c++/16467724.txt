#include <bits/stdc++.h>

using namespace std;

int main()
{
	int N;
	cin >> N;
	int n=0,m=0;
	while(n<N)
	{
		int k=0;
		m+=5;
		int z=5;
		while(m/z>0)
		{
			k+=(m/z);
			z*=5;
		}
		n=k;
	}
	if(n==N)
	{
		cout<<5<<endl;
		for(int i=m;i<m+4;i++)cout<<i<<' ';
		cout<<m+4<<endl;
	} else cout<<0<<endl;
}
