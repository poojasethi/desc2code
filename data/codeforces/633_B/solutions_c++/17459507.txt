#include <cstdio>
using namespace std;
const int maxa=5e5+1 ;
int t[maxa] ;
int main()
{
    for(int k=5;k<maxa;k*=5)
	for(int i=k;i<maxa;i+=k)
	    t[i]++ ;
    int n;
    scanf("%d",&n) ;
    int sum=0 ;
    int i=1 ;
    for(;sum<n;i++)
	sum+=t[i] ;
    i-- ;
    if(sum==n)
	printf("5\n%d %d %d %d %d",i,i+1,i+2,i+3,i+4) ;
    else
	printf("0") ;
}
