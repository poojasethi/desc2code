n, m, k = map(int, raw_input().split())
if k == 1:
	if n - 1 > m:
		print 2
		print 1, n
	else:
		if n == 2 and m == 2:
			print 4
			print 1, 2, 3, 4
		else:
			print 3
			print 1, n, 2 * n - 1
else:
	if n == m:
		print 2 * k + 1
		print "1 " * k, n, (str(n + 1) + " ") * (k - 1), n + 2
	else:
		print 2 * k
		print "1 " * k, n, (str(n + 1) + " ") * (k - 1)
