#!/usr/bin/python
n,m,k = map(int, raw_input().split())
size, prev = n + m + 1, 1
ans = []
if k == 1:
    for i in xrange(1, size, n-1):
        ans.append(i)
    print len(ans)
    print ' '.join(map(str, ans))
else:
    while prev < size - 1:
        if prev == 1:
            ans.extend([prev]*k)
        else:
            ans.append(prev)
            prev+=1
            ans.extend([prev]*(k-1))
        prev = prev + n -1

    if prev < size:
        ans.append(size-1)
    print str(len(ans))
    print ' '.join(map(str, ans))
