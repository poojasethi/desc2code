n, m, k = map(int, raw_input().split())

if m == n:
	if m == 2 and k == 1:
		print 4
		print 1, 2, 3, 4
	else:
		print 2 * k + 1
		print ' '.join(map(str, [1] * k + [n] + [n + 1] * (k - 1) + [n + 2]))
elif m == n - 1:
	if k == 1:
		print 3
		print 1, n, n + 1
	else:
		print 2 * k
		print ' '.join(map(str, [1] * k + [n] * (k - 1) + [n + 1]))
else:
	print 2 * k
	print ' '.join(map(str, [1] * k + [n] * k))
