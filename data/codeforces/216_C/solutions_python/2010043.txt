n, m, k = (int(x) for x in raw_input().split())

sn = str(n)
sn1 = str(n + 1)
hires = ['1'] * k + [sn]
k -= 1

if n - m > 1:
    # totally okay
    hires += [sn] * k
else:
    hires += [sn1] * k
    if not k and (n - m == 1 or n == 2):
        hires.append(sn1)
    if n == m:
        hires.append(str(n + 2))

print len(hires)
print ' '.join(hires)
