n, m, k = map(int, raw_input().split())
s = [1 for i in range(k)]
s.append(n)
for i in range(k - 1):
    s.append(n + 1)
d = max(s) + n - 1
while d < 1 + m + n:
    s.append(d)
    if d < m + n:
	for i in range(k - 1):
	    s.append(d + 1)
    d = max(s) + n - 1
print len(s)
print ' '.join(map(str, s))
