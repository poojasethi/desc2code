#include<stdio.h>
#include<stdlib.h>
long i,j,k,l,n,m,t,la;
long e[1000000],a[1000000];
int main(){
	scanf("%ld%d%ld",&n,&m,&t);
	i=1;
	while(i<=n+m){
		while(e[i]>=t && i<=n+m)++i;
		if(i<=n+m){
			if(e[i]==0 && i!=1){
				a[++la]=i-1;
				for(j=0;j<n;++j){
					++e[i-1+j];
				}
			}else{
				a[++la]=i;
				for(j=0;j<n;++j){
					++e[i+j];
				}
			}
		}else if(i==n+m+1 && e[i]==0){
			a[++la]=i-1;
		}
	}
	printf("%ld\n",la);
	for(i=1;i<=la;++i){
		printf("%ld ",a[i]);
	}
	//system("pause");
	return 0;
}
