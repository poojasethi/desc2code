#include <bits/stdc++.h>

using namespace std;

vector < int > v;
int a[100010],b[100010];
int main()
{
    int n,m,k,ans=0,i,j;

    cin>>n>>m>>k;

    for(i=1;i<=n+m;i++)
    {
        while(a[i]<k || b[i+1]==0)
        {
            for(j=i;j<i+n;j++)
            {
                a[j]++;
                b[j]=1;
            }
            v.push_back(i);
        }
    }

    cout<<v.size()<<endl;

    for(i=0;i<v.size();i++)
    {
        cout<<v[i]<<" ";
    }
}
