# include<iostream>
using namespace std;
int sum[10001];
int main()
{
   int n,m,k;
   cin>>n>>m>>k;
   for(int i=1;i<=k;i++)
      sum[i]=1;
   int s1=n+1,s2=k;
   while(s1<=n+m)
   {
      sum[++s2]=s1-1;
      for(int i=1;i<k;i++)
         sum[++s2]=s1;
      if(k==1)
         s1--;
      s1+=n;
   }
   if(s1==n+m+1)
      sum[++s2]=n+m;
   cout<<s2<<endl;
   for(int i=1;i<s2;i++)
      cout<<sum[i]<<" ";
   cout<<sum[s2]<<endl;
   return 0;
}
