#include<bits/stdc++.h>
using namespace std;
int a,b,c,ch[3000],cnt;
int main(){
cin>>a>>b>>c;
ch[1+a+b]=1;
ch[1]=c;
for(int i=a;i<(1+a+b);i+=(a-1)){
        ch[i]++;
        ch[i+1]+=c-1;
        if(c>1)i++;
}
for(int i=1;i<(1+a+b);i++)cnt+=ch[i];
cout<<cnt<<endl;
for(int i=1;i<(1+a+b);i++){
    for(int j=0;j<ch[i];j++){
        cout<<i<<" ";
    }
}
cout<<endl;
}
