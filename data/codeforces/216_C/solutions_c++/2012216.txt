#include<iostream>
using namespace std;
int main()
{
    int n,m,k,a[10005],len,i;
    cin>>n>>m>>k;
    for(int i=1;i<=k;i++)a[i]=1;
    len=k;
    while(a[len]+n-1<n+m+1)
    {
        a[len+1]=a[len]+n-1;len++;
        int t=len;
        for(int i = 1;i<k;i++)
          a[len+1] = a[t]+1,len++;;
    }
    for(i=1;i<=len;i++)
    if(a[i]>n+m || a[i]==0)break;
    if(a[i]>n+m||a[i]==0)i--;
    cout<<i<<endl;
    for(int j=1;j<=i;j++)
      cout<<a[j]<<' ';



return 0;
}
