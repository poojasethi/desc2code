#include <iostream>
using namespace std;
int n,m,k,res,a[3010];
int main(){

  cin>>n>>m>>k;
  for(int i=0;i<k;i++)
	  a[i] = 1;
  res=k;
  a[res++] = n;
  for(int i=0;i<k-1;i++)a[res++] = n+1;

  if(k==1 && m>n-2){
    if(n==2)a[res++] = 3;
    else a[res++] = n+2;
  }
  if(n == m && (k!=1 || (k==1 && n==2)))a[res++] = 2*n;

  cout<<res<<endl;
  for(int i=0;i<res;i++)
	  cout<<a[i]<<" "; 
  cout<<endl;

  return 0;

}
