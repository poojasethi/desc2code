#include<bits/stdc++.h>

using namespace std;

int main()
{
  int i,m,n,k,st;
  vector<int> a;
  scanf("%d%d%d",&n,&m,&k);
  a.clear();
  st=0;
  for (i=0;i<n+m;i++)
  {
    while (a.size()-st<k)
      a.push_back(i+1);
    while ((st<a.size())&&(i+2-a[st]>=n)) st++;
    if (st==a.size()) a.push_back(i+1);
  }
  printf("%d\n",a.size());
  for (i=0;i<a.size();i++)
  {
    if (i!=0) printf(" ");
    printf("%d",a[i]);
  }
  printf("\n");
}
