#include<iostream>
using namespace std;
int n,m,k,ans=0,i,j;
main(){
    cin>>n>>m>>k;
    
    for(i=0;i<=n+m;k==1?i+=n-1:i+=n){
        for(j=1;j<=k;j++){
            if(i>1&&j==1&&j!=k)ans++;
            else if(i+1<=n+m)ans++;
        }
    }
    cout<<ans<<"\n";

    for(i=0;i<=n+m;k==1?i+=n-1:i+=n){
        for(j=1;j<=k;j++){
            if(i>1&&j==1&&j!=k)cout<<i<<" ";
            else if(i+1<=n+m)cout<<i+1<<" ";
        }
    }
}
