#include <stdio.h>
#include <vector>
#include <set>

using namespace std;

vector<int> rst;
int n,m,k;
int a[1000000];
int main(){
	scanf("%d%d%d",&n,&m,&k);
	int i;
	for(i=1;i<=n+m;i++){
		if(a[i+1]==0){
			rst.push_back(i);
			int j=0;
			for(j=i;j<=i+n-1;j++)a[j]++;	

		}
		while(a[i]<k){
			rst.push_back(i);
			int j=0;
			for(j=i;j<=i+n-1;j++)a[j]++;		
		}
	}
	printf("%d\n",rst.size());
	for(i=0;i<rst.size();i++)printf("%d ",rst[i]);
}
