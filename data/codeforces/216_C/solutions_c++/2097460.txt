#include<cstdio>
#include<algorithm>
using namespace std;
int ans[3000];
int main(){
	int n, m, k, cn = 0;
	scanf("%d%d%d", &n, &m, &k);
	for(int i = 0; i < k; i++) ans[cn++] = 1;
	int lst = n;
	while(lst <= n+m){
		ans[cn++] = lst;
		if(lst == n+m) break;
		for(int i = 1; i < k; i++) ans[cn++] = lst+1;
		lst += n-(k==1);
	}
	printf("%d\n", cn);
	for(int i = 0; i < cn; i++) printf("%d%c", ans[i], i==cn-1?'\n':' ');
}
