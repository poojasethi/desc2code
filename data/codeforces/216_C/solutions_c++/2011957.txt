#include<iostream>
#include<cstdio>
using namespace std;
const int maxs = 2000010;
int ans[maxs], cnt;
int main()
{
	int n, m, k;
	cin >> n >> m >> k;
	if (k == 1)
	{
		for (int i=0; i<n+m; i+=n-1) ans[cnt++] = i+1;
	}
	else
	{
		int i;
		for (i=0; i<n+m; i+=n)
		{
			ans[cnt++] = max(i, 1);
			for (int j=1; j<k; j++) ans[cnt++] = i+1;
		}
		if (i == n+m) ans[cnt++] = i;
	}
	printf("%d\n", cnt);
	printf("%d", ans[0]);
	for (int i=1; i<cnt; i++) printf(" %d", ans[i]);
	printf("\n");
	return 0;
}
