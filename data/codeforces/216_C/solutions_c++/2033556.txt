#include <cstdio>
#include <vector>
using namespace std;
int t[10005];
int r;
int que[30005];
int main() 
{    
    int n,m,k;
    scanf("%d%d%d", &n,&m,&k);
    r=0;
    for (int i=1; i < n+m+1; ++i)
        while (t[i] < k || !t[i+1]) 
        {
            que[++r]=i;
            for (int j=i; j < i+n; ++j) t[j]++;
        }

    printf("%d\n", r);
    for (int i=1; i<=r; ++i)
        printf("%d ", que[i]);

    return 0;
}
