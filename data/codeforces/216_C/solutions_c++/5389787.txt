#include<iostream>
#include<vector>
using namespace std;
int main()
{
vector<int> v;
int n,m,k,a[10005]={0},i,j;
bool b[10005]={0};
cin>>n>>m>>k;
b[1]=1;
for(i=1;i<=n+m;i++)
{
while(a[i]<k || b[i+1]==0)
{
v.push_back(i);
for(j=i;j<=i+n-1;j++)
{
a[j]++;
a[j+n+m]++;
b[j]=1;
}
}
}
cout<<v.size()<<endl;
for(i=0;i<v.size();i++) cout<<v[i]<<" ";
return 0;
}
