#include <cstdio>
#include <cstring>
#include <algorithm>
#define FOR(i,s,e) for (int i=(s); i<(e); i++)
using namespace std;

int n, m, num, cnt, ret, a[5000];

int main(){
	scanf("%d%d%d", &n, &m, &num);
	cnt = 0;
	FOR(i,0,n+m){
		if (cnt < num){
			a[i+n-1] += num - cnt;
			ret += num - cnt;
			cnt = num;
		}
		cnt -= a[i];
		if (!cnt) ++cnt, a[i+n-1]++, ret++;
	}
	
	printf("%d\n", ret);
	FOR(i,0,3000)
	while (a[i]--) printf("%d ", i-n+2);
	return 0;
}
