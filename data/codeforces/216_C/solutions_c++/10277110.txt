#include <iostream>
#include <vector>
using namespace std;
int main(){
	int n,m,k;
	cin >> n >> m >> k;
	vector<int> arr;
	if(k==1){
		for(int i = 1; i <= n+m; i+=n-1){
			arr.push_back(i);
		}
	}
	else{
		for(int i = 1; i <= n+m+1; i+=n){
			if(i==n+m+1){
				arr.push_back(n+m);
				continue;
			}
			for(int j = 0; j < k; ++j){
				arr.push_back(i-(j==0&&i!=1));
			}
		}
	}
	cout << arr.size() << '\n';
	for(int i = 0; i < arr.size(); ++i)
		cout << arr[i] << " \n"[i==arr.size()-1];
}
