#include<cstdio>
#define max(x,y)    ((x>y)?(x):(y))
int main(){
    int n,m,k,l,ans=0,s[10000],y[10000],x[10000],p;
    scanf("%d %d %d",&n,&m,&k);
    for(int i=1;i<=n+m;i++)
        if((x[i]<k)||(y[i]==0)){
            l=max(k-x[i],1);
            for(int j=1;j<=l;j++)s[++ans]=i;
            for(int j=i;j<=i+n-1;j++)
                y[j]=1,x[j]+=l;
            y[i+n-1]=0;
        }
    printf("%d\n",ans);
    for(int i=1;i<ans;i++)printf("%d ",s[i]);printf("%d\n",s[ans]);
    getchar(),getchar();
}
