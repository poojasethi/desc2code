#include <iostream>

using namespace std;

int a[3002];

int main()
{
    int n,m,k,s,c=0;
    cin>>n>>m>>k;
    s=n+m+1;
    int i=1,j=n;
    a[1]=k;



    while(j<=s){
      if(j<s)a[j]=((j!=s) || k==1);
      if(j+1<s)a[j+1]=k-1;
      j+=(n-(k==1));
    }

for(int j=1; j<s; j++)c+=a[j];

    cout<<c<<endl;
    for(int j=1; j<s; j++){
    for(int i=0; i<a[j]; i++)cout<<j<<" ";
    }
    return 0;
}
