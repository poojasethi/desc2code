#include <algorithm>
#include <stdio.h>
#include <vector>
#define  pb        push_back
using    namespace std;

int n,m,k,ans,cnt;
vector<int>v;

int main()
{
  scanf("%d%d%d",&n,&m,&k);
  for(int i=0 ; i<n+m ; i++)
  {
    while(ans-cnt<k)                   ans++,v.pb(i+1);
    while(cnt<ans && i+2-v[cnt]>=n)    cnt++;
    if(cnt==ans)                       ans++,v.pb(i+1);
  }
  printf("%d\n",ans);
  for(int i=0 ; i<ans ; i++)
    printf("%d ",v[i]);
  printf("\n");
  return 0;
}
