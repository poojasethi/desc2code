#include <cstdio>

using namespace std;

const int MAXT=10005, MAXN=1005;
int n, m, k, cnt, t[MAXN*MAXN];

int main()
{
	scanf("%d%d%d", &n, &m, &k);
	int i, cur;
	for (i=1; i<=k; i++) {
		t[++cnt] = 1;
	}
	cur = n+1;
	while (true) {
		if (cur > n+m+1)
			break;
		t[++cnt] = cur-1;
		if (cur > n+m)
			break;
		for (i=1; i<=k-1; i++) {
			t[++cnt] = cur;
		}
		cur += (k==1 ? n-1 : n);
	}
	printf("%d\n", cnt);
	for (i=1; i<=cnt; i++) {
		printf("%d%c", t[i], (i==cnt ? '\n' : ' '));
	}
	return 0;
}
