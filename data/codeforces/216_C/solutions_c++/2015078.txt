#include<cstdio>
#include<cstring>
using namespace std;
int n,m,k,ans;
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	if (k==1)
	{
		if (n==m&&n==2) {printf("4\n1 2 3 4");return 0;}
		if (n-1<=m) ans=2*k+1; else ans=2*k;
		printf("%d\n",ans);
		if (n-1<=m)printf("1 %d %d",n,n*2-1); else 
				   printf("1 %d",n);
		return 0;
	}
	if (n==m) ans=2*k+1; else ans=2*k;
	printf("%d\n",ans);
	for (int i=0;i<k;i++)
		printf("1 ");
	printf("%d ",n);
	for (int i=0;i<k-1;i++)
		printf("%d ",n+1);
	if (n==m) printf("%d",n+m);
	return 0;
}
