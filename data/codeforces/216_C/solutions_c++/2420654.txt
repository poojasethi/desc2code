#include<iostream>
#include<vector>

using namespace std;

int cnt[6001];
vector<int> ans;

int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	for (int i=0;i<k;i++)
		ans.push_back(1);
	ans.push_back(n);
	for (int i=n;i<n+n;i++)
	{
		cnt[i]++;
	}
	for (int i=n+1;i<=n+m;i++)
	{
		while(cnt[i] < k || cnt[i+1] == 0)
		{
			ans.push_back(i);
			for (int j=i;j<i+n;j++)
				cnt[j]++;
		}
	}
	cout<<ans.size()<<endl;
	for (int i=0;i<ans.size();i++)
		cout<<ans[i]<<" ";
	cout<<endl;
	return 0;
}
