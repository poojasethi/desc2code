#include <cstdio>
#include <vector>

using namespace std;

int t[10005];

int main() {    
    int n,m,k;
    scanf("%d%d%d", &n,&m,&k);
    vector<int> d;

    for (int i=1; i < n+m+1; ++i)
        while (t[i] < k || !t[i+1]) {
            d.push_back(i);
            for (int j=i; j < i+n; ++j) t[j]++;
        }

    printf("%d\n", d.size());
    for (int i=0; i < d.size(); ++i)
        printf("%d ", d[i]);

    return 0;
}
