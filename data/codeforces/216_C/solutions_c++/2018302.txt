#include<iostream>
using namespace std;

int main() {
  int n,m,k;
  cin>>n>>m>>k;
  if (n==2 && m==2 && k==1) {
    cout<<4<<endl<<"1 2 3 4"<<endl;
  } else {
    int count = k*2;
    if (k==1 && m+1>=n || m>=n)
      count++;
    cout<<count<<endl;
    cout<<1;
    for (int ki=1;ki<k;ki++)
      cout<<" 1";
    cout<<" "<<n;
    for (int ki=1;ki<k;ki++)
      cout<<" "<<n+1;
    if (k==1 && m+1>=n)
      cout<<" "<<2*n-1;
    else if (m>=n)
      cout<<" "<<2*n;
    cout<<endl;
  }
  return 0;
}
