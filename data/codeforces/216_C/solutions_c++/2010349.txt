#include<cstdio>

const int MAX=10000;
int n,m,k;
int num[MAX],ans[MAX],a[MAX],tot;

int main()
{
//	freopen("C.in","r",stdin);freopen("C.out","w",stdout);
	int i,j,now,next;
	scanf("%d %d %d",&n,&m,&k);
	for(i=1;i<=n+m;++i)
	{
		now=0,next=0;
		for(j=1;j<=tot;++j)
			now+=(i-a[j]+1<=n);
		while(now<k)
		{
			a[++tot]=i;
			++now;
		}
		for(j=1;j<=tot;++j)
			next+=(i+2-a[j]<=n);
		if(!next)
		{
			a[++tot]=i;
			++now;
		}
	}
	printf("%d\n",tot);
	for(i=1;i<=tot;++i)
		printf("%d ",a[i]);
}
