#include <iostream>
using namespace std;
int main(){
    int n,m,k,ans,s[10000],cnt=0;
    cin>>n>>m>>k;
    for(int i=1;i<=k;i++)s[i]=1;
    cnt=n+1,ans=k;
    while(cnt<n+m+1){
       s[++ans]=cnt-1;
       for(int i=1;i<k;i++)
          s[++ans]=cnt;
       if(k==1)cnt--;
       cnt+=n;
    }
    if(cnt==n+m+1)s[++ans]=n+m;
    cout<<ans<<endl;
    for(int i=1;i<ans;i++)cout<<s[i]<<' ';
    cout<<s[ans]<<endl;
    return 0;
}
