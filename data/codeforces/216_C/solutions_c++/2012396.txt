#include <cstdio>
#include <vector>
std::vector<int>a;
int i,n,m,k,tim;
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(i=0;i<k;++i)
		a.push_back(1);
	tim=n;
	if(k>1)
		while(tim<=n+m)
		{
			a.push_back(tim++);
			if(tim>n+m) break;
			for(i=1;i<k;i++)
				a.push_back(tim);
			tim=tim+n-1;
		}
	else
		while(tim<=n+m)
		{
			a.push_back(tim);
			tim=tim+n-1;
		}
		printf("%d\n",a.size());;
		k=a.size();
		for(i=0;i<k;++i)
			{
				printf("%d",a[i]);
		 		printf(i<k-1?" ":"\n");
		    }
	return 0;
}
