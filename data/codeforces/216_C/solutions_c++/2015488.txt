#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>

using namespace std;
int a[10010], b[10010],n,m,k,i,j,ans=0;
int main()
{
    scanf("%d%d%d",&n,&m,&k);
    memset(a,0,sizeof(a));
    for(i=1; i<=n+m; i++)
    {
        while(a[i]<k||a[i+1]==0)
        {
            ans++;
            b[ans]=i;
            for(j=i; j<=i+n-1; j++)
                a[j]++;
        }
    }
    printf("%d\n",ans);
    for(j=1; j<ans; j++)
        printf("%d ",b[j]);
    printf("%d\n",b[ans]);
}
