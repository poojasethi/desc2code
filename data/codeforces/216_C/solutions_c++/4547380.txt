#include<cstdio>
using namespace std;
int main()
{
	int n,m,k,ans;
	scanf("%d%d%d",&n,&m,&k);
	if(k==1)
	{
		printf("%d\n",m/(n-1)+2);
		printf("1");
		for(int i=n;i<=n+m;i+=n-1)
			printf(" %d",i);
	}
	else
	{
  		ans=m/n+2+(k-1)*((m-1)/n+2);
		printf("%d\n",ans);
		printf("1");
		for(int i=n;i<=n+m;i+=n)
			printf(" %d",i);
		for(int i=1;i<=n+m;i+=n)
			for(int j=0;j<k-1;++j)
				printf(" %d",i);
		printf("\n");
	}
}
