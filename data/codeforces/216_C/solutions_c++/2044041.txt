#include <iostream>
#include <stdio.h>
#include <vector>
using namespace std;

int main()
{
int n,m,k,b[10000]={};
bool f[10000]={0};
vector <int> a;
//freopen("i.txt","r",stdin);
cin>>n>>m>>k;
f[1]=1;
for(int i=1;i<=n+m;i++)
{
while(b[i]<k || f[i+1]==0)
{
a.push_back(i);
for(int j=i;j<=i+n-1;j++)
{
b[j]++;
b[j+n+m]++;
f[j]=1;
}
}
}
cout<<a.size()<<endl;
for (int i=0;i<a.size();i++) cout<<a[i]<<" ";
return 0;
}
