#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int nMax = 1001000;
int a[nMax],f[nMax];
bool b[nMax];
int main()
{
	int n,m,k,i,j,tot;
	while(scanf("%d%d%d",&n,&m,&k)!=EOF){
		memset(a,0,sizeof(a));	memset(b,0,sizeof(b));	tot=0;
		for(i=1;i<=n+m;i++)
			while(a[i]<k || !b[i+1]){	
				f[tot++]=i;
				 for(j=i;j<i+n;j++)	b[j]=true,a[j]++,a[j+n+m]++;
			}
		printf("%d\n%d",tot,f[0]);
		for(i=1;i<tot;i++)	printf(" %d",f[i]);	puts("");
	}
	return 0;
}
