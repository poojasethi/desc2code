#include <iostream>
#include <algorithm>
#include <bitset>

using namespace std;

int main(){
    int n, m, k; cin >> n >> m >> k;
    int num = 0, ans[50000], last = n;
    for (int i = 0; i < k; i++)
        ans[num++] = 1;

    while (last <= n + m){
        ans[num++]  = last;
        if (last == n + m) break;
        for (int i = 0; i < k-1; i++)
            ans[num++] = last+1;
        last += n; if (k == 1) last--;
    }

    cout << num << endl;
    for (int i = 0; i < num; i++)
        cout << ans[i] << " ";

}
