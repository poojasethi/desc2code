#include <cstdio>
#include <cstring>
#include <vector>

using std::vector;

int n,m,k;
vector<int> a;

int main(){
	scanf("%d%d%d",&n,&m,&k);
	if (k==1) {
		for (int t=1; t<=n+m; t+=n-1) a.push_back(t);
	} else {
		for(int i=1;i<=k;++i) a.push_back(1);
		for(int t=n;t<=n+m;t+=n) a.push_back(t);
		for(int t=n+1;t<=n+m;t+=n)
			for(int i=1;i<k;++i) a.push_back(t);
	}
	printf("%d\n",a.size());
	for (int i=0; i<a.size(); ++i) printf("%d ", a[i]);
	return 0;
}
