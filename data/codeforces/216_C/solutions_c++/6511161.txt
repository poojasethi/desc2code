#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define N 100010

int a[N],cnt[N],tot,n,m,k;

int main(){
   while(scanf("%d%d%d",&n,&m,&k)!=EOF){
      memset(cnt,0,sizeof(cnt));
      tot = 0;
      for(int day = 1; day <= n+m; day++){
         while(cnt[day] < k || cnt[day+1] == 0){
            a[tot++] = day;
            for(int i = 0; i < n; i++) cnt[day+i]++;
         }
      }
      printf("%d\n",tot);
      for(int i = 0 ; i < tot; i++)
         printf("%d%c",a[i],i == tot-1 ? '\n':' ');
   }
   return 0;
}
