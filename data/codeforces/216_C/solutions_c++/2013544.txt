#include <iostream>

using namespace std;

int main(){
	int N, M, K;
	
	cin >> N >> M >> K;
	
	if (K != 1){
		
		cout << 2*K + (M==N) << "\n";
		for (int i=0; i<K; i++) cout << "1 ";
		cout << N << " ";
		for (int i=0; i<K-1; i++) cout << N+1 << " ";
		
		if (M == N) cout << 2*N << "\n";
		else cout << "\n";
	}
	
	else {
		if (M < N-1) cout << "2\n1 " << N << "\n";
		if (M == N-1) cout << "3\n1 " << N << " " << N+1 << "\n";
		if (M == N) {
			if (M == 2) cout << "4\n1 2 3 4\n";
			else cout << "3\n1 " << N << " " << 2*N-1 << "\n";
		}
	}
}
