#include <stdio.h>
#include <vector>
using namespace std;
vector<int> hire;
int a[1000000];
int main(){
	int n,m,k;
	scanf("%d %d %d",&n,&m,&k);
	int i;
	for(i=1;i<=n+m;i++){
		if(a[i+1]==0){
			hire.push_back(i);
			int j=0;
			for(j=i;j<=i+n-1;j++)a[j]++;	

		}
		while(a[i]<k){
			hire.push_back(i);
			int j=0;
			for(j=i;j<=i+n-1;j++)a[j]++;		
		}
	}
	printf("%d\n",hire.size());
	for(i=0;i<hire.size();i++){
		printf("%d ",hire[i]);
	}
	return 0;
}
