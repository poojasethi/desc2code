#include <cmath>
#include <iostream>
#include <vector>
using namespace std;
int n,m,k,i,j;
vector <int> v;
int main()
{
	cin>>n>>m>>k;
	for(i=0;i<k;i++)
		v.push_back(n-1);
	for(i=1;;i++)
	{
		int s1=0,s2=0;
		for(j=0;j<v.size();j++)
		{
			s1+=i<=v[j];
			s2+=i<v[j];
		}
		if (i>v[0]+m)
				break;
		while (s1<k)
		{
			v.push_back(i+n-1);
			s1++; s2++;
		}
		if (!s2)
			if (i>v[0]+m)
				break;
			else v.push_back(i+n-1);
	}
	cout<<v.size()<<endl;
	for(i=0;i<v.size();i++)
		cout<<v[i]-n+2<<" ";
}
