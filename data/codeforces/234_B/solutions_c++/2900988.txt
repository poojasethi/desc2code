#include <fstream>
#include <algorithm>
using namespace std;

typedef pair<int, int> P;

const int MAX = 1000;

int main()
{
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int n, k;
	P p[MAX];

	cin >> n >> k;
	for (int i = 0; i < n; i++)
		cin >> p[(p[i].second = i)].first;
	sort(p, p + n);
	cout << p[n-k].first << "\n";
	for (int i = n-k; i < n; i++)
		cout << (p[i].second + 1) << " ";
	cout << "\n";
}
