#include<bits/stdc++.h>
using namespace std;
int main(void) {
	int n,k;
	freopen("input.txt","r",stdin);
     freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&k);
	int i,a;
	vector < pair<int,int> > v;
	for(i=0;i<n;i++) {
		scanf("%d",&a);
		v.push_back(make_pair(a,i));
	}
	sort(v.begin(),v.end());
	cout<<v[n-k].first<<endl;
	for(i=n-k;i<n;i++) {
		printf("%d ",(v[i].second + 1));
	}
     printf("\n");
	return 0;
}
