#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;
struct node {
int a,b;
bool operator <(const node q) const {
return b<q.b;
}
} v[1005];
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
int n,k,i;
cin>>n>>k;
for(i=1;i<=n;i++) cin>>v[i].b,v[i].a=i;
sort(v+1,v+n+1);
reverse(v+1,v+n+1);
cout<<v[k].b<<endl;
for(i=1;i<=k;i++) cout<<v[i].a<<" ";
return 0;
}
