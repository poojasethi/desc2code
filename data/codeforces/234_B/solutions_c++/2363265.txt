#include <fstream>
#include <algorithm>

using namespace std;

int n, k;
pair<int, int> a[1010];
int main()
{
  ifstream in("input.txt");
  ofstream out("output.txt");
  
  in >> n >> k;
  for (int i = 0; i < n; i++)
    in >> a[i].first, a[i].second = i;
  
  sort(a, a+n);
  
  out << a[n-k].first << endl;
  for (int i = n-k; i < n; i++)
    out << 1+a[i].second << " ";
  out << endl;
  return 0;
}
