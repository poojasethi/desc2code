#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
int main()
{
	ios::sync_with_stdio(false);
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	lld n,k;
	cin>>n>>k;
	pair<lld,lld> a[1004];
	for(lld i=0;i<n;i++)
	{
		cin>>a[i].first;
		a[i].second=i+1;
	}
	sort(a,a+n);
	cout<<a[n-k].first<<endl;
	for(lld i=n-k;i<n-1;i++)
		cout<<a[i].second<<' ';
	cout<<a[n-1].second<<endl;
	return 0;
}
