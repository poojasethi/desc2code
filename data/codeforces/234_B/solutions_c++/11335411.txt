#include<bits/stdc++.h>
using namespace std;
int main(){
   freopen("input.txt","r",stdin);
   freopen("output.txt","w",stdout);
   int n,k;
   pair<int,int>a[1005];
   cin>>n>>k;
   for(int i=1;i<=n;i++) {
        cin>>a[i].first;
        a[i].second=i;
   }
   sort(a+1,a+n+1);
   cout<<a[n-k+1].first<<endl;
   for(int i=n-k+1;i<=n;i++)
       cout<<a[i].second<<" \n"[i==n];
   return 0;
}
