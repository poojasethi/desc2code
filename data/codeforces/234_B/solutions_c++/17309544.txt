#include <iostream>
#include <fstream>
#include <algorithm>
using namespace std;

pair<long long, int> arr[1001];

int main(){	
	long long n, k, i, ans=0;
	ifstream in("input.txt");
	ofstream out("output.txt");
	
	in >> n >> k;
	for(i=0;i<n;i++){
		in >> arr[i].first;
		arr[i].second = i+1;
	}
	sort(arr, arr+n);
	out << arr[n-k].first << '\n';
	for(i=n-1;k--;i--){
		out << arr[i].second << ' ';
	}
	out << '\n';
	return 0;
}
