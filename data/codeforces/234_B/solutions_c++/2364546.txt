#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
using namespace std;
int n,m;
pair<int,int> a[1001];
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		cin>>a[i].first;
		a[i].second=i;	
	}
	sort(a+1,a+1+n);
	cout<<a[n-m+1].first<<endl;
	for(int i=n-m+1;i<=n;i++)cout<<a[i].second<<" ";
	return 0;
}
