#include<cstdio>
#include<algorithm>
using namespace std;

int n, k;
pair<int, int> v[1111];

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	
	scanf("%d%d", &n, &k);
	for(int i=0;i<n;i++) {
		scanf("%d", &v[i].first);
		v[i].second = i+1;
	}
	sort(v,v+n);
	printf("%d\n", v[n-k].first);
	for(int i=n-k;i<n;i++)
		printf("%d%c", v[i].second, i<n-1 ? ' ' : '\n');
	
	return 0;
}
