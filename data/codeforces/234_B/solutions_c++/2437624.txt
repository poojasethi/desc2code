#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,k,i;
    pair<int,int> p[1001];
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    cin>>n>>k;
    for(i=0;i<n;i++)
    {
        cin>>p[i].first;
        p[i].second=i+1;
    }
    sort(p,p+n);
    cout<<p[n-k].first<<"\n";
    for(i=n-1;i>=n-k;i--)
    cout<<p[i].second<<" ";
    return 0;
}
