#include <bits/stdc++.h>
using namespace std;

int main() {
    int n,k;
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    pair <int,int>arr[1003];
    cin>>n>>k;
    for(int i=0;i<n;i++){
        cin>>arr[i].first;
        arr[i].second=i+1;
     }
     sort(arr,arr+n);
    cout<<arr[n-k].first<<endl;
    for(int i=n-k;i<n;i++)
        cout<<arr[i].second<<" ";
}
