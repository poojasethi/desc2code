#include <bits/stdc++.h>

using namespace std;
ifstream fin("input.txt");
ofstream fout("output.txt");
pair<int,int> p[1001];
int main() {
 int n,k;
 fin>>n>>k;
 for(int i=0 ; i<n ; i++) {fin>>p[i].first; p[i].second=i+1;}
 sort(p,p+n);
 fout<<p[n-k].first<<endl;
 for(int i=n-k ; i<n ; i++)
    fout<<p[i].second<<" ";

   return 0;
}
