#include<iostream>
#include<algorithm>
#include<utility>
#include<fstream>
using namespace std;
pair<int,int> a[1001];
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
int n,k;
cin>>n>>k;
for(int i=0;i<n;i++)
{
cin>>a[i].first;
a[i].second=i+1;
}
sort(a,a+n);
cout<<a[n-k].first<<"\n";
for(int i=n-k;i<n-1;i++)
cout<<a[i].second<<" ";
cout<<a[n-1].second;
return 0;
}
