#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main() {
	int n,k,a[1000],b[1000],ans;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i){
		scanf("%d",&a[i]);
		b[i]=a[i];
	}
	sort(a,a+n);
	printf("%d\n",a[n-k]);
	ans=a[n-k];
	for(int i=0;i<n && k>0;++i)
		if(b[i]>=ans){
			printf("%d ",i+1);
			--k;
		}
	return 0;
}
