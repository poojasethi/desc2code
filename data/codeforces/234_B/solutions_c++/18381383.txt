#include<bits/stdc++.h>
using namespace std;

int n,k;
pair<int,int> a[1001];

int main(int argc, char **argv) {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&k);
	for(int i = 0; i<n; ++i){
		scanf("%d",&a[i].first);
		a[i].second = i + 1;
	}
	sort(a,a+n);
	printf("%d\n",a[n-k].first);
	for(int i = n - 1; i>=n-k; --i)
		printf("%d ",a[i].second);
	return 0;
}
