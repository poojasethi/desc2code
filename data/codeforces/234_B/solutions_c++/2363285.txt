#include<stdio.h>
#include<algorithm>

using std::sort;

const int MAXN=1000;

int a[MAXN];
inline bool cmp(int l,int r)
{
	return a[l]<a[r];
}

int r[MAXN];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,k,i;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;++i)
	{
		scanf("%d",a+i);
		r[i]=i;
	}
	sort(r+1,r+n+1,cmp);
	printf("%d\n",a[r[n+1-k]]);
	for(i=n+1-k;i<=n;++i)
		printf("%d ",r[i]);
	putchar(10);
	return 0;
}
