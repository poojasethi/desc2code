#include <iostream>
#include <fstream>
#include <algorithm>

using namespace std;

ifstream fin ("input.txt");
ofstream fout ("output.txt");

#define F first
#define S second

int n,k;
pair <int,int> a[1000+10];

int main()
{
	fin >> n >> k;
	for (int i = 0; i < n; ++i)
	{
		fin >> a[i].F;
		a[i].S = i+1;
	}
	sort (a,a+n);
	fout << a[n-k].F << endl;
	for (int i = n-1;i>=n-k;--i)
		fout << a[i].S << ' ';
}
