#include<cstdio>
#include<algorithm>
using namespace std;

int a[1000];
int b[1000];
int n,p,c=0;

int main()
{   freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    scanf("%d%d",&n,&p);
    for(int i=0;i<n;++i){scanf("%d",&a[i]);b[i]=a[i];}
    sort(&a[0],&a[n]);
    printf("%d\n",a[n-p]);
    for(int i=0;i<n;++i)
    if (b[i]>=a[n-p]){if(c<p){printf("%d ",i+1);++c;}else break;}
    printf("\n");

}
