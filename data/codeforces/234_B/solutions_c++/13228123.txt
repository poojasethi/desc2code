#include<bits/stdc++.h>

using namespace std;

int n, k;
pair<int, int> a[1005];

int main(int argc, char **argv) {
	
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	scanf("%d%d", &n, &k);
	for(int i=1 ; i<=n ; ++i) {
		scanf("%d", &a[i].first);
		a[i].second = i;
	}
	sort(a+1, a+n+1);
	printf("%d\n", a[n-k+1].first);
	for(int i=n-k+1 ; i<=n ; ++i)
		printf("%d ", a[i].second);
}
