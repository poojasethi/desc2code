#include <bits/stdc++.h>

using namespace std;
int n , k ;
int a[3333];
int b[3333];

ifstream  fin("input.txt");
ofstream fout("output.txt");
int main()
{
fin>>n>>k;
for(int i=1;i<=n;i++){
        fin>>a[i];
        b[i]=a[i];
        }

sort(a+1,a+1+n);
reverse(a+1,a+1+n);
int X=k;
fout<<a[k]<<endl;

for(int i=1;i<=n;i++)
    if( b[i]>=a[k] )
    if(X-->0)
    fout<<i<<" ";

    return 0;
}
