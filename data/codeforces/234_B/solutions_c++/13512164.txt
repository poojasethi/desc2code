#include <bits/stdc++.h>

using namespace std;
pair<int,int> v[1010];
int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,k;
    cin>>n>>k;
    for(int i=0;i<n;i++)cin>>v[i].first,v[i].second=i+1;
    sort(v,v+n);
    cout<<v[(n-k)].first<<endl;
    for(int i=n-1,j=0;j<k;i--,j++){
        cout<<v[i].second<<" ";
    }
    return 0;
}
