#include <iostream>
#include<algorithm>
#include<fstream>
using namespace std;
pair<int,int>vorodi[1003];
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
int n,k,i;
cin>>n>>k;
for(i=0;i<n;i++)
{
cin>>vorodi[i].first;
vorodi[i].second=i+1;
}
sort(vorodi,vorodi+n);
cout<<vorodi[n-k].first<<endl;
for(i=n-k;i<n;i++)
cout<<vorodi[i].second<<" ";
cout<<endl;
return 0;
}
