#include <iostream>
#include <algorithm>
#include <fstream>
using namespace std;
int main()
{
	int n,k;
	
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	cin>>n>>k;

	pair<int,int>a[n];

	for(int i = 0;i < n;++i){
		cin>>a[i].first;
		a[i].second = i+1;
	}

	sort(a,a+n);

	cout<<a[n-k].first<<endl;

	for(int i = n-k;i<n;++i)
		cout<<a[i].second<<" ";

	cout<<endl;

	return(0);
}
