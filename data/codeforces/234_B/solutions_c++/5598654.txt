#include <iostream>
#include<algorithm>
#include<fstream>
using namespace std;
pair<int,int>mas[1003];

int main()
{
	int n,k,i;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n>>k;
	for(i=0;i<n;i++)
	{
		cin>>mas[i].first;
		mas[i].second=i+1;
	}
	sort(mas,mas+n);
	cout<<mas[n-k].first<<endl;
	for(i=n-k;i<n;i++)
	{
		cout<<mas[i].second<<" ";
	}
	cout<<endl;
	return 0;
}
