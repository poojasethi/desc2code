#include<bits/stdc++.h>

using namespace std;


int main()
{
    int n,k,i;
pair<int,int> arr[1005];
ifstream file;
file.open("input.txt");
file>>n>>k;
for(i=0;i<n;i++)
{
file>>arr[i].first;
arr[i].second=i+1;
}
sort(arr,arr+n);
file.close();
ofstream file1;
file1.open("output.txt");
file1<<arr[n-k].first<<endl;
for(i=n-k;i<n;i++)
file1<<arr[i].second<<" ";
}
