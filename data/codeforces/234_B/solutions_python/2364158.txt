from operator import itemgetter
fin = open('input.txt', 'r').readlines()
fout = open('output.txt', 'w')
n, k = fin[0].split()
line = fin[1].split()
n = int(n)
k = int(k)
arr = []
for i in xrange(n):
    x = (int(line[i]), i)
    arr.append(x)
arr = sorted(arr, key=itemgetter(0), reverse=True)
answer = []
minx = arr[k - 1]
for i in xrange(k):
    answer.append(arr[i][1] + 1)
answer = sorted(answer)
fout.write(str(minx[0]) + '\n')
fout.write(' '.join(map(str, answer)))
fout.close()
