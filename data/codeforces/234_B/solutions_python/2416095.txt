#import sys
fi = open('input.txt', 'r')
fo = open('output.txt', 'w')
#fi = sys.stdin
#fo = sys.stdout
n, k = map(int, fi.readline().split(' '))
light = map(int, fi.readline().split(' '))
lights = light[:]
lights.sort()
level = lights[n - k]
fo.write(str(level) + '\n')
i = -1
while k > 0:
	i = i + 1
	if light[i] >= level:
		fo.write(str(i + 1) + ' ')
		k = k - 1

fi.close()
fo.close()
