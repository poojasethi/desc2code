import sys,math

f=open('input.txt','r')
g=open('output.txt','w')


n,k=map(int,f.readline().split())
a=map(int,f.readline().split())

b=[]
for i in range(n):
    b.append([a[i],i])
b.sort()

p=b[-k][0]
s=''
for i in range(n-1,n-1-k,-1):
    if b[i][0]>=p:
        s+=str(b[i][1]+1)+' '
g.write(str(p)+'\n')
g.write(s)

f.close()
g.close()
