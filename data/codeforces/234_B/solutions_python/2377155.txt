import sys
sys.stdin = open('input.txt', 'r')
sys.stdout = open('output.txt', 'w')

r = lambda: map(int, raw_input().split())
n, k = r()
a = r()
b = sorted(a, reverse = True)
print b[k - 1]
ans = []
for i in xrange(n):
    if a[i] in b[0 : k]:
        ans.append(i + 1)
        b[b.index(a[i])] = -1
print ' '.join(map(str, ans))
