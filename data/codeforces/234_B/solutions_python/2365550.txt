input = open('input.txt', 'r')
output = open('output.txt', 'w')

N, K = map(int, input.readline().split())
A = map(int, input.readline().split())

B = zip(xrange(1, len(A) + 1), A)

B.sort(key=lambda a: a[1])

choice = B[N-K:]
output.write(str(choice[0][1]) + '\n')
output.write(' '.join(map(lambda x: str(x[0]), choice)))
