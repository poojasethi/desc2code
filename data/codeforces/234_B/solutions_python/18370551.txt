f = open("input.txt")
o = open("output.txt", "w")
l = f.readline().split()
n = int(l[0])
k = int(l[1])
l = f.readline()
lis = map(int, l.split())
se=set(lis)
d={}
for sx in se:
    d[sx]=[]
for g in xrange(n):
    d[lis[g]].append(g+1)
lis.sort()
o.write( str(lis[n-k]) + '\n')
for gg in lis[n-k:]:
    o.write( str(d[gg].pop()) + ' ')
f.close()
o.close()
