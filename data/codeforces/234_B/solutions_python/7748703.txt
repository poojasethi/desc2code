ff = open("input.txt")
f = ff.read().split('\n')
ff.close()
(n , k) = map(int, f[0].split())

l = map(int, f[1].split())
a = []
for i in range(n):
    a.append((i+1,l[i]))

a = sorted(a, key=lambda x:x[1], reverse = True)

g = open("output.txt","w")
g.write("%d\n"%(a[k-1][1]))
for i in range(k):
    g.write("%d "%(a[i][0]))
g.write("\n")
g.close()
