fi = open("input.txt", "r")
fo = open("output.txt", "w")

read = fi.readline
write = fo.write

readints = lambda: map(int, read().split())

n, k = readints()
p = readints()

for i in xrange(n):
    p[i] = (p[i], i)

p = sorted(p, key = lambda x: -x[0])
write(str(min([x[0] for x in p][:k])))
write("\n")

for i in xrange(k):
    write(str(p[i][1] + 1))
    write(" ")
