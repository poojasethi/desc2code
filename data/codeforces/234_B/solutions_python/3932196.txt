#!/usr/bin/env python
import sys
class data:
    global counter
    counter = 0
    def __init__(self, val):
        global counter
        counter = counter + 1
        self.cnt = counter
        self.val = int(val)
    def __str__(self):
        self.__repr__(self)

    def __repr__(self):
        return '%i: %i' % (self.cnt, self.val)
    def __gt__(self, other):
        return self.val > other.val

fin = open('input.txt', 'r')
fout = open('output.txt', 'w')
n, k = map(int, fin.readline().split())
datas = map(data, fin.readline().split())
#print datas
datas.sort(reverse = True)
#print datas

minn = datas[k-1].val
fout.write('%i\n' % (minn))
res = []
for i in xrange(k):
    res.append(str(datas[i].cnt))
fout.write(' '.join(res))
    
fout.close()
