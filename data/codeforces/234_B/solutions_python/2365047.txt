inp = open("input.txt","r")
out = open("output.txt",'w')
n,k = map(int,inp.readline().split())
a = map(int,inp.readline().split())
#print n,lef
b = [[]]*(n)
for i in range(n):
    b[i] = [i,a[i]]
    
for i in xrange(n-1):
    for j in xrange(i+1,n):
        if b[i][1]<b[j][1]:
            b[i],b[j] = b[j],b[i]

#print b
out.write(str(b[k-1][1])+"\n")
for i in xrange(k):
    out.write("%s "%(b[i][0]+1))
inp.close()
out.close()
