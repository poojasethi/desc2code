f = open('input.txt', 'r')
r = f.readline().split()
n = int(r[0])
k = int(r[1])
r = f.readline().split()
l = range(n)
for x in l:
    r[x] = [int(r[x]), x]
r.sort()
a = ''
for x in range(len(r)-k, len(r)):
    a += str(r[x][1] + 1) + ' '
f = open('output.txt', 'w')
f.write(str(r[len(r)-k][0]) + '\n')
f.write(a)
