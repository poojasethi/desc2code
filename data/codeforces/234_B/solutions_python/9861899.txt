fr = open('input.txt', 'r')
n = fr.readline().split(' ')
k = int(n[1][:-1])
n = int(n[0])

lst = []
wrd = fr.readline().split(' ')
wrd[n - 1] = wrd[n - 1][:-1]
for i in range(n):
    lst.append((int(wrd[i]), i))
lst.sort(key=lambda x:x[0])

fp = open('output.txt', 'w')
fp.write(str(lst[n - k][0]))
fp.write('\n')
for i in range(n - k, n):
    fp.write(str(lst[i][1] + 1))
    fp.write(' ')
fp.write('\n')
