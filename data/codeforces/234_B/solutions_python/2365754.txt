f=open('input.txt','r')
n,k=map(int,f.readline().split())
l=map(int,f.readline().split())
info=[(m,i) for i,m in zip(xrange(1,n+1),l)]
info.sort(reverse=True)
ans=info[:k]
#print info
f=open('output.txt','w')
f.write(str(sorted(l,reverse=True)[k-1])+'\n')
ind=[j[1] for j in ans]
for j in sorted(ind):
    f.write(str(j)+' ')
