#include <iostream>
#include <string>
using namespace std;

const int MAX = 100010;
const string suf[] = {"lios", "liala", "etr", "etra", "initis", "inites"};

int c;
int l[MAX];
string w;

int main()
{
    while (cin >> w)
    {
        l[c] = -1;
        for (int i = 0; i < 6; i++)
            if (w.length() >= suf[i].length() &&
                w.substr(w.length() - suf[i].length(), suf[i].length()) == suf[i])
            {
                l[c] = i; break;
            }
        c++;
    }

    bool f = l[0] == -1;
    int n = l[0] == 2 || l[0] == 3;
    for (int i = 1; i < c && !f; i++)
    {
        if (l[i] == 2 || l[i] == 3) n++;
        if (l[i] == -1) f = true;
        if (l[i] < l[i-1]) f = true;
        if ((l[i] % 2) != l[i-1] % 2) f = true;
    }

    if (c > 1 && n != 1) f = true;

    cout << (f ? "NO\n" : "YES\n") << endl;

    return 0;
}
