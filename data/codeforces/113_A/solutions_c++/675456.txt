#include <stdio.h>
#include <string.h>

int n, sn;
int a[100010];
char str[100010];

bool test()
{
	if (n==1) return true;
	int i;
	for (i=1; i<n; i++)
	  if (a[i]%2!=a[0]%2) return false;
	for (i=0; i<n; i++)
	  if (a[i]/2!=0) break;
	if (i==n || a[i]/2!=1) return false;
	i++;
	for (i; i<n; i++)
	  if (a[i]/2!=2) break;   	
	return i==n;
}

int main()
{
//	freopen("a.in", "r", stdin);
	n=0;
	bool flag=true;
	while (scanf("%s", str) != EOF)
	{
		sn=strlen(str);	
		if (strcmp(str+sn-4, "lios")==0)
			a[n++]=0;
		else 
		if (strcmp(str+sn-5, "liala")==0)
			a[n++]=1;
		else
		if (strcmp(str+sn-3, "etr")==0)
			a[n++]=2;
		else
		if (strcmp(str+sn-4, "etra")==0)
			a[n++]=3;
		else
		if (strcmp(str+sn-6, "initis")==0)
			a[n++]=4;
		else
		if (strcmp(str+sn-6, "inites")==0)
			a[n++]=5;
		else flag=false;	
	}
	if (flag && test()) printf("YES\n");
	else printf("NO\n");
	return 0;
}
