#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<string>
#include<cctype>
#include<algorithm>

using namespace std;

string str, suff[6] = {"lios", "liala", "etr", "etra", "initis", "inites"};

int Get(string s) {
    for (int i = 0; i < 6; ++i) 
        if (s.substr(max(0, (int)(s.length() - suff[i].length()))) == suff[i]) return i;
    return -1;
}

int main() {
    
    int gen = -1, sz = 0, fg = 0, pre = 0;
    while (cin >> str) {
          sz++; int t = Get(str), tp = (t >> 1);
          if (t == -1 || gen != -1 && (t & 1) != gen) {puts("NO"); return 0;}
          gen = (t & 1); if (tp == 1) fg = 1;
          if (tp == 1 && tp <= pre || tp < pre) {puts("NO"); return 0;}
          pre = tp;
    } if (sz == 1 || fg) puts("YES"); else puts("NO");
    
    return 0;

}
