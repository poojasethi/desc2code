#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
char a[100001];
int nv=0,nou=0,f1=0,f2=0,num=0,f;
void legal(char *p){
	int len,ff1=0;
	len=strlen(a);
	if(len>5){
		if(strcmp("initis",p+len-6)==0){
			ff1=1;
			if(nou!=1)f2=1;
		}
		else if(strcmp("inites",p+len-6)==0){
			ff1=1;
			nv++;
			if(nou!=1)f2=1;
		}
	}
	if(len>4){
		if(ff1==0&&strcmp("liala",p+len-5)==0){
			ff1=1;
			nv++;
			if(nou!=0)f2=1;
		}
	}
	if(len>3){
		if(ff1==0&&strcmp("lios",p+len-4)==0){
			ff1=1;
			if(nou!=0)f2=1;
		}
		else if(ff1==0&&strcmp("etra",p+len-4)==0){
			nv++;
			nou++;
			ff1=1;
			if(nou!=1)f2=1;
		}
	}
	if(len>2){
		if(ff1==0&&strcmp("etr",p+len-3)==0){
			ff1=1;
			nou++;
			if(nou!=1)f2=1;
		}
	}
	if(ff1==0)f1=1;
}
int main(){
	int i,t,k,j,sum,be,ed,n,m,last;
	while(scanf("%s",a)!=EOF){
		legal(a);
		num++;
	}
	if(f1==1)f=1;
	else {
		if((nv==0||nv==num)&&f2==0&&nou==1)f=0;
		else f=1;
		if(num==1)f=0;
	}
	if(f)printf("NO\n");
	else printf("YES\n");
}
