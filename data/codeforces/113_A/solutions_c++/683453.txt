#include <iostream>
#include <string>

using namespace std;

int o, O, f, z, i, d, V['zz'];
string t, s[6] = {"lios","liala","etr","etra","initis","inites"};

int main() {
    while (cin >> t) {
        for(i=O=0;i<6;i++) if ((d=t.size()-s[i].size())>=0&&t.substr(d)==s[i])
            { if (z&&V[z-1]>i) break; O = 1; V[z++]=i; f+=i/2%2; o+=i%2; }
        if (!O) { cout << "NO"; return 0; }
    }
    cout << (z!=1&&(o%z||f!=1)?"NO":"YES");
}
