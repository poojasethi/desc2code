#include<bits/stdc++.h>
using namespace std;

const char x[6][10]={"lios","liala","etr","etra","initis","inites"};
const int y[6]={4,5,3,4,6,6};
char a[100015];

int main()
{
	int t,l,k=-1,c=0;
	while(scanf("%s",a+10)==1)
	{
		c++;
		if(k>10)
			return puts("NO");
		l=strlen(a+10);
		for(t=0;t<6;t++)
			if(!strcmp(a+10+l-y[t],x[t]))
				break;
		if(t>=6)
			return puts("NO");
		if(k<0)
			k=t>=4?100:t;
		else
		{
			if(k%2!=t%2)
				return puts("NO");
			if(t<k)
				return puts("NO");
			if(k<2&&t>=4)
				return puts("NO");
			if(k>=2&&k<4&&t>=2&&t<4)
				return puts("NO");
			k=t;
		}
	}
	puts(k<2&&c>1?"NO":"YES");
	return 0;
}
