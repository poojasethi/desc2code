#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const string a1[6]={"lios","liala","etr","etra","initis","inites"};
int i,j,k,n,m,ty;
int a[N];
char p[N];
inline void fail() {
	printf("NO\n");
	exit(0);
}
inline void get(int &A,int &B) {
	int n=strlen(p),i,j;
	For(i,0,5) {
		int len=a1[i].size();
		if (len>n) continue;
		string s;
		For(j,n-len,n-1) s+=p[j];
		if (s==a1[i]) {
			A=i&1; B=i/2;
			return;
		}
	}
	fail();
}
int main() {
	//freopen("113a.in","r",stdin);
	ty=-1;
	for (;scanf("%s",p)!=EOF;) {
		int A,B;
		get(A,B);
		if (ty<0) ty=A;
		else if (ty!=A) fail();
		a[++n]=B;
	}
	if (n>1) {
		For(i,1,n) if (a[i]==1) break;
		if (i>n) fail();
		For(j,1,n) if (a[j]==2) break;
		if (i+1!=j) fail();
		For(k,j,n) if (a[k]!=2) fail();
	}
	printf("YES\n");
	return 0;
}
