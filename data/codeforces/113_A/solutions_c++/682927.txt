#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

#define F(i,a,n) for (int i=a;i<n;++i)
#define S(n) int(n.size())
#define r return
#define G string

bool pre(G s, G a) {
    if (S(a) > S(s)) r 0;
    r s.substr(S(s)-S(a)) == a;
}

int o, O, f;
G t, s[6] = {"lios", "liala", "etr", "etra", "initis", "inites"};
vector<int> V, C;

int main() {
    while (cin >> t) {
        F(i, O=0, 6) if (pre(t, s[i])) { O = 1; V.push_back(i); if (i&1) o++; }
        if (!O) { cout << "NO\n"; return 0; }
    }
    C = V;
    sort(C.begin(), C.end());
    F(i, 0, S(V)) if (V[i]/2 == 1) f++;
    cout << (S(V)!=1&&((o&&o<S(V))||V!=C||f!=1)?"NO\n":"YES\n");
}
