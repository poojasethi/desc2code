#include <iostream>
#include <string>
using namespace std;
int o,O,f,z,i,d,V[100010];
int main()
{
	string t,s[6]={"lios","liala","etr","etra","initis","inites"};
    while(cin>>t)
     {
         for(i=O=0;i<6;i++)
             if( (d=t.size()-s[i].size())>=0 && t.substr(d)==s[i]&&!(z&&V[z-1]>i))
			 O=1,V[z++]=i,f+=i/2%2,o+=i%2;
                if(!O){
                     cout<<"NO";return 0;
                     }
     }
    cout<<(z-1&&(o%z||f-1)?"NO":"YES");

   return 0;
}
