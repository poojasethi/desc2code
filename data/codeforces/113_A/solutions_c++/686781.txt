#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <string>
using namespace std;

const int K=6;
const string strends[K] = {
    "lios","liala","etr","etra","initis","inites"
};

const int mat[K][K]={
    {0,0,0,0,0,0}, //0
    {2,4,3,5,0,0}, //1
    {2,0,3,0,0,0}, //2
    {0,0,0,0,3,0}, //3
    {0,4,0,5,0,0}, //4
    {0,0,0,0,0,5}, //0
};

void No()
{
    printf("NO");
    exit(0);
}

int f(const string &s)
{
    for(int i=0;i<K;i++)
        if(s.length()>=strends[i].length() && s.substr(s.length()-strends[i].length(),strends[i].length())==strends[i])
            return i;
    No();
}

int main()
{
    //freopen("input.txt","r",stdin);
    int state=1;
    string s;
    int k=0;
    while(cin>>s)
    {
        state=mat[state][f(s)];
        k++;
    }
    if(k>1 && state!=3 && state!=5)
        No();
    printf("YES");
    return 0;
}
