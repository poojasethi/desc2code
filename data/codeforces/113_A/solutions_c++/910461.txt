#include <algorithm>
#include <iostream>
#include <stdlib.h>
#include <string.h>
#include <limits.h>
#include <stdio.h>
#include <math.h>
using namespace std;
char s[100005];
char opt[6][10]={"lios","liala","etr","etra","initis","inites"};
int length[6]={4,5,3,4,6,6};
int main()
{
  int noun=0,cnt=0;
  int cas=-1;
  int ok=1;
  int x,i;
  while(scanf("%s",&s)!=EOF)
    if(ok)
    {
      x=-1;
      int len=strlen(s);
      for(i=0 ; i<6 ; i++)
        if(len >= length[i] && strcmp(s+len-length[i] , opt[i]) == 0) { x=i;  break; }
      if(x==-1) { ok=0;  continue; }
      if(x==2||x==3)  noun++;
      if(cas>=0 && cas%2!=x%2) { ok=0;  continue; }
      if(x<cas) { ok=0;  continue; }
      cas=x;
      cnt++;
    }
  if(ok && (cnt==1 || noun==1))   printf("YES");
  else                            printf("NO");
  return 0;
}
