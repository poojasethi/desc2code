#include <iostream>
#include <string>
int o,O=1,f,z,i,d,K,V['zz'];
std::string t,s[6]={"lios","liala","etr","etra","initis","inites"};
int main(){
    while(!(K=!O)&&std::cin>>t)
        for(i=O=0;i<6;i++)
            if((d=t.size()-s[i].size())>=0&&t.substr(d)==s[i]&&!(z&&V[z-1]>i))
                O=1,V[z++]=i,f+=i/2%2,o+=i%2;
    std::cout<<(K||z-1&&(o%z||f-1)?"NO":"YES");
}
