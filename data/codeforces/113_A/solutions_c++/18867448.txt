#include<bits/stdc++.h>
using namespace std;
const int N=1e5+9;
char s[N];
vector<int > v;
string  b[6]={"lios","liala","etr","etra","initis","inites"};
int type(string str)
{
	int i;
	reverse(str.begin(),str.end());
	for( i=0; i<6 ; i++ )
		if(str.substr(0,b[i].size())==b[i])
			return i;
	return -1;
}
int main()
{
	int i,odd=0,eve=0,f=0,c=0;
    for(i=0 ; i < 6 ; i++)
    	reverse(b[i].begin(),b[i].end());
    while(scanf("%s",s)!=EOF)
        v.push_back(type(string(s)));
    for( i=0 ; i < (int)v.size() ; i++ )
       {
       	   if(v[i]%2)odd++;
       	   else  if(v[i]%2 ==0)
       	   eve++;
       	   if(v[i]==-1)f=1;
       	   if(v[i]==2 || v[i]==3)c++;
       	   if(i && v[i]<v[i-1] ) f=1; 
       }
       if((int)v.size()==1 && v[0]!=-1)
       	printf("YES");
       else if( f || (odd&&eve) || c!=1 )
       	printf("NO");
       else
       	printf("YES");
	return 0;
}
