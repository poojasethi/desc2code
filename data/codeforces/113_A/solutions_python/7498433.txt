import re
s=raw_input()
p1=re.compile("(((\S*lios )*(\S*etr)( \S*initis)*)|((\S*liala )*(\S*etra)( \S*inites)*))\Z")
p3=re.compile("((\S*lios)|(\S*liala)|(\S*etr)|(\S*etra)|(\S*initis)|(\S*inites))\Z")
flag=(s.find(' ')!=-1);
if flag==0:
	if p3.match(s)!=None:
		print "YES"
	else:
		print "NO"
else:
	if p1.match(s)!=None:
		print "YES"
	else:
		print "NO"
