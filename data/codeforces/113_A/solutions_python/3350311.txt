#!/usr/bin/env python

def adj(s):
	return len(s) >= 4 and s[-4:] == "lios" or len(s) >= 5 and s[-5:] == "liala"

def verb(s):
	return len(s) >= 6 and (s[-6:] == "initis" or s[-6:] == "inites")

def noun(s):
	return len(s) >= 3 and s[-3:] == "etr" or len(s) >= 4 and s[-4:] == "etra"

def masc(s):
	return s[-1] == "a" or s[-2:] == "es"

words = raw_input().split(' ')

if len(words) == 1 and (noun(words[0]) or adj(words[0]) or verb(words[0])):
	print("YES")
	quit()

p = 0
while p < len(words) and adj(words[p]):
	p += 1

if p == len(words) or not noun(words[p]):
	print("NO")
	quit()

p += 1
while p < len(words) and verb(words[p]):
	p += 1

if p != len(words):
	print("NO")
	quit()

g = masc(words[0])
for i in words:
	if masc(i) ^ g:
		print("NO")
		quit()
print("YES")
