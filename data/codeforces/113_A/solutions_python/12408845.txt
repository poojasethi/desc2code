curr = 0
gender_male = 0

words = raw_input().split()

def get_adj():
    global curr
    global gender_male
    for i in xrange(0,len(words)):
        if words[i][-4:] == "lios":
            if gender_male == 2:
                return False
            gender_male = 1
        elif words[i][-5:] == "liala":
            if gender_male == 1:
                return False
            gender_male = 2
        else:
            curr = i
            return True
    curr = len(words)
    return True

def get_noun():
    global curr
    global gender_male
    if curr >= len(words):
        if len(words) > 1:
            return False
        else:
            return True
    if words[curr][-3:] == "etr" and gender_male != 2:
        gender_male = 1
        curr += 1
        return True
    if words[curr][-4:] == "etra" and gender_male != 1:
        gender_male = 2
        curr += 1
        return True
    if len(words) == 1:
        return True
    else:
        return False

def get_verbs():
    global curr
    global gender_male
    for i in xrange(curr,len(words)):
        if words[i][-6:] == "initis":
            if gender_male == 2:
                return False
            gender_male = 1
        elif words[i][-6:] == "inites":
            if gender_male == 1:
                return False
            gender_male = 2
        else:
            return False
    return True

if not get_adj():
    print "NO"
elif not get_noun() and len(words) > 1:
    print "NO"
elif not get_verbs():
    print "NO"
else:
    print "YES"
