#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

from collections import Counter

YES = "YES"
NO = "NO"

M = 1
W = 2

ADJ = 1
NOUN = 2
VERB = 3

TYPES = {
         'lios' : (ADJ,M),
         'liala' : (ADJ,W),
         'etr' : (NOUN,M),
         'etra' : (NOUN,W),
         'initis' : (VERB,M),
         'inites' : (VERB,W),
         }

def word_type(w):
    for suffix,wtype in TYPES.iteritems():
        if w.endswith(suffix):
            return wtype
    return (-1,-1)

def solve(W):
    WT = [word_type(w) for w in W]
#    print WT
    
    if not all(wt[0]>0 for wt in WT):
        return NO
    
    if len(W) == 1:
        return YES
    
    sbeg = 0
    while sbeg+1<len(WT) and (WT[sbeg][0] <= WT[sbeg+1][0]):
        sbeg += 1

#    print sbeg
    if sbeg+1 == len(WT):
        
        c = Counter(WT[i][0] for i in range(sbeg+1))
#        print c
        one_noun = (c[NOUN] == 1)
#        print 'one_noun:',one_noun
        
        one_sort = all(WT[i][1]==WT[0][1] for i in range(1,sbeg+1))
#        print 'one_sort',one_sort
        
        if one_noun and one_sort:
            return YES
    
    return NO

W = [s for s in input.readline().strip().split(' ')]

a = solve(W)
output.write('%s\n' % a)
