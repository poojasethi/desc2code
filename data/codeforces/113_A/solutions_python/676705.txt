#!/usr/bin/env python
# coding=utf-8
import sys
import re

endings = {
        "lios":"A",
        "liala":"a",
        "etr":"N",
        "etra":"n",
        "initis":"V",
        "inites":"v",
        }

def applyGrammar(word):
    for end, t in endings.items():
        b = word.rfind(end)
        if b != -1 and word[b:] == end:
            return t
    return "+"

def compute(words):
    stmt = "".join(map(applyGrammar, words))
    regex = "(A*NV*)"
    regex1 = "([ANV])"
    for reg in (regex, regex.lower(), regex1, regex1.lower()):
        m = re.match(reg, stmt)
        if m and m.group() == stmt:
            return True
    return False

if __name__ == "__main__":
    words = sys.stdin.readline().split()
    if compute(words):
        print("YES")
    else:
        print("NO")
