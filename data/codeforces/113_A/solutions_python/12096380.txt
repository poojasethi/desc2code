import string

m_adj 	= "lios"
f_adj 	= "liala"
m_n 	= "etr"
f_n 	= "etra"
m_v 	= "initis"
f_v 	= "inites"

def main():
	wordList = raw_input().split()
	n = len(wordList)
	flag = True
	
	"check male or female"
	if wordList[0].endswith(m_adj):
		male = True
	elif wordList[0].endswith(f_adj):
		male = False
	elif wordList[0].endswith(m_n):
		male = True
		flag = False
	elif wordList[0].endswith(f_n):
		male = False
		flag = False
	else:
		if n==1:
			return wordList[0].endswith(m_v) or wordList[0].endswith(f_v)
		return False
		
	if n==1:
		return True
		
	if flag:
		"remove all the other adj"
		suffix = m_adj if male else f_adj
		i = 1
		while i<n and wordList[i].endswith(suffix):
			i += 1
		if i >= n:
			return False
		'must have at least one non'
		suffix = m_n if male else f_n
		if not wordList[i].endswith(suffix):
			return False
	else:
		i = 0
	'check if all the left suffix is verb'	
	i += 1
	suffix = m_v if male else f_v
	while i<n and wordList[i].endswith(suffix):
		i += 1
	return i>=n
	
	
	
flag = main()
print "YES" if flag else "NO"
