from collections import defaultdict
f = lambda: map(int, raw_input().split())
n = int(raw_input())
a, b, o, r = f(), f(), defaultdict(int), []
for i in b: o[i] += 1
c = [b[i] - 1 if o[b[i]] <= 1 else -1 for i in xrange(n)]
for i in xrange(n):
  if a[i] == 1:
    s, d = c[i], [i + 1]
    while(s != i and s >= 0):
      d.append(s + 1)
      s = c[s]
    if len(d) > len(r): r = d
r.reverse()
print len(r)
print ' '.join(map(str, r))
