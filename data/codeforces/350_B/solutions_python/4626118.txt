import sys
from collections import Counter

raw_input()
objects = map(int, raw_input().split())
paths = map(int, raw_input().split())
c = Counter(paths)

def solve(i, sol):
  while 1:
    n = paths[i]-1
    if n == -1 or c[n+1] > 1:
      break
    i = n
    sol.append(n)

solution = []
for i,h in enumerate(objects):
  if h == 1:
    s = [i,]
    solve(i, s)
    if len(s) > len(solution):
      solution = s

solution.reverse()
print len(solution)
print ' '.join([str(v+1) for v in solution])
