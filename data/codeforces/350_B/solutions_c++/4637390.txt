#include <stdio.h>
#include <string.h>
int main(){
	int n,sum[100005],a[100005],b[100005],now[100005],nn[100005],s;
	scanf("%d",&n);
	memset(sum,0,sizeof(sum));
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=n;i++){scanf("%d",&b[i]);sum[b[i]]++;}
	int anssum=0;
	int ansmark=0;
	for(int i=1;i<=n;i++){
		if(a[i]==0)continue;
		s=0;
		now[++s]=i;
		now[++s]=b[i];
		while(now[s]>0&&sum[now[s]]==1&&a[now[s]]==0){
			s++;
			now[s]=b[now[s-1]];
		}
		if(s>anssum){
			anssum=s;
			ansmark=i;
			for(int j=1;j<anssum;j++)nn[j]=now[j];
		}
	}
	printf("%d\n",anssum-1);
	for(int i=anssum-1;i>1;i--)printf("%d ",nn[i]);
	printf("%d\n",nn[1]);
	return 0;
}
