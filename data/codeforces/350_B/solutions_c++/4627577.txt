#include <cstdio>
#include <vector>
using namespace std;

const int mxn = 100005;

int prt[mxn], grd[mxn], hot[mxn];
int n, h;
bool ish[mxn];
vector <int> sol, tmp;

int main() {
	int i, j;

	scanf("%d", &n);
	for (i = 1; i <= n; i++) {
		scanf("%d", &j);
		ish[i] = j == 1;
		if (j) hot[h++] = i;
	}
	for (i = 1; i <= n; i++) {
		scanf("%d", &j);
		prt[i] = j;
		grd[j]++;
	}

	for (i = 0; i < h; i++) {
		j = hot[i];
		tmp.clear();
		do {
			tmp.push_back(j);
			j = prt[j];
		} while (j && !ish[j] && grd[j] <= 1);
		if (tmp.size() > sol.size())
			sol = tmp;
	}

	printf("%d\n", sol.size());
	for (i = sol.size()-1; i >= 0; i--)
		printf("%d ", sol[i]);

	return 0;
}
