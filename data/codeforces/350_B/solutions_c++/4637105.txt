#include<stdio.h>
#define M 1000005
int i,j,k,t,tt,n,maxn,x;
int a[M],d[M],b[M],fa[M],f[M];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++)d[i]=0;
	for(i=1;i<=n;i++){
		scanf("%d",&b[i]);
		d[b[i]]++;
	}
	fa[0]=-1;
	for(i=1;i<=n;i++)
    	if(d[i]<=1)fa[i]=b[i];
    	else fa[i]=-1;
	maxn=0;
	for(i=1;i<=n;i++){
		if(a[i]){
			x=i;t=0;
			while(fa[x]!=-1){
				x=fa[x];
				t++;
			}
			if(t>maxn){
				maxn=t;k=i;
			}
		}
	}
	printf("%d\n",maxn);
    tt=0;x=k;
    while(fa[x]!=-1){
    	tt++;
    	f[tt]=x;
    	x=fa[x];
    }
    for(i=tt;i>=1;i--){
    	if(i!=1)printf("%d ",f[i]);
    	else printf("%d\n",f[i]);
    }
}
