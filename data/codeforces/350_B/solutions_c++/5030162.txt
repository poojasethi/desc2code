#include <iostream>
using namespace std;
#define maxn 100010
int fa[maxn],sum[maxn],sta[maxn],ans1[maxn],ans2[maxn];
int ht[maxn],num=0;
int find(int t)
{
	int k=0;
	while(t)
	{
		ans2[k++]=t;
		if(sum[fa[t]]==1&&sta[fa[t]]==0)
			t=fa[t];
		else break;
	}
	return k;
}

int main()
{
	int n,ans=0;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>sta[i];
		if(sta[i]==1)
			ht[num++]=i;
	}

	for(int i=1;i<=n;i++)
	{
		cin>>fa[i];
		sum[fa[i]]++;
	}
	for(int i=0;i<num;i++)
	{
		int a=find(ht[i]);
		if(a>ans)
		{
			ans=a;
			for(int i=0;i<a;i++)
				ans1[i]=ans2[i];
		}
	}
	cout<<ans<<endl;
	while(--ans)
	{
		cout<<ans1[ans]<<' ';
	}
	cout<<ans1[0]<<endl;
	return 0;
}
