import itertools as it
from sys import stdin

for line in stdin:
    n = int(line)
    kinds = map(int, next(stdin).split())
    kinds = sorted(set(kinds))

    ## print zip(kinds, kinds[1:])
    fits = any(b//a < 2 for a, b in it.izip(kinds, it.islice(kinds, 1, n)))
    print 'YES' if fits else 'NO'
