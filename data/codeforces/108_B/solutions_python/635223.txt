#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

from bisect import bisect_right

def solve(A):
    A = list(set(A))
    n = len(A)
    B = [2*b for b in sorted(A)]
    #print B
    for i in range(1,n):
        a = B[i]
        k = bisect_right(B, a/2)
        #print 'for B[%d]=%d, searching %d, position=%d' % (i,a,a/2,k)
        if i-k >= 1:
            return 'YES'
    return 'NO'

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

n = int(input.readline())

A = numbers_from_line()

a = solve(A)
output.write('%s\n' % a)
