#!/usr/bin/python2

if __name__ == "__main__":
    n = int(raw_input())
    s = raw_input()
    l = map(int, s.split(' '))
    l.sort()
    ans = "NO"
    for i in range(0, len(l)-1):        
        a, b = l[i], l[i+1]
        if a == b: continue
        if a==1 and b==1: continue
        if a*2 > b:
            ans = "YES"
            break
    print ans
