n = int(raw_input())

a = []
for x in raw_input().split():
    a.append(int(x))

a.sort()

ok = True
for i in range(len(a) - 1):
    if a[i] == 1: continue
    if (a[i] != a[i + 1]) & (a[i] * 2 > a[i + 1]):
        ok = False
        break

if ok: print 'NO'
else: print 'YES'
