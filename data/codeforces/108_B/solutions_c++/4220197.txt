#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
using namespace std;

long long a[200000];

int main()
{
	int n;
	cin >> n;
	for(int i=0;i<n;i++)
		cin >> a[i];
	sort(a, a+n);
	for(int i=0;i<n-1;i++)
		if(a[i]!=a[i+1] && a[i] > a[i+1]/2)
		{
			cout << "YES\n";
			return 0;
		}
	cout << "NO\n";
}
