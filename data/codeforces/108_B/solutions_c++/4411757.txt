#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int n,arr[100010]; bool b;
int main()
{
    cin >> n;
    for (int i=0; i<n; i++) cin >> arr[i];
    sort(arr,arr+n);
    for (int i=1; i<n; i++) if ((arr[i]<arr[i-1]*2) && (arr[i] != arr[i-1])) b = true;
    cout << (b?"YES":"NO");
}
