#include <iostream>
#include <algorithm>

using namespace std;

const int MAX=100000+10;
int n;
long long a[MAX];

int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a,a+n);
	for(int i=0;i<n-1;i++){
		if(a[i]==a[i+1]){
			continue;
		}
		if(a[i+1]<a[i]*2&&i+1<n){
			cout<<"YES";
			return 0;
		}
	}
	cout<<"NO";
}
