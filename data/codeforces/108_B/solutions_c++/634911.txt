#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
  int n,i;
  cin>>n;
  vector<__int64> a(n);
  for(i=0;i<n;++i)
    cin>>a[i];
  sort(a.begin(),a.end());
  for(i=0;i<n-1;++i)
    if(a[i]*2>a[i+1] && a[i]!=a[i+1]){
      cout<<"YES\n";
      return 0;
    }
  cout<<"NO\n";
}
