#include <stdio.h>
#include <algorithm>
using namespace std;
int a[100005];

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++)
		scanf("%d", &a[i]);
	sort(a, a+n);
	n = unique(a, a+n) - a;
	int flag=0;
	for(int i=1;i<n;i++)
		if(a[i] < a[i-1]+a[i-1])
			flag = 1;
	puts(flag?"YES":"NO");
	return 0;
}
