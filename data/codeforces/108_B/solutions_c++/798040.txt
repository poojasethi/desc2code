#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=100100;
long long arr[maxn];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	sort(arr,arr+n);
	for(int i=0;i<n-1;i++)
		if(arr[i]*2>arr[i+1]&&arr[i]<arr[i+1]){
			cout<<"YES"<<endl;
			return 0;
		}
	cout<<"NO"<<endl;
}
