#include <iostream>
#include <set>
using namespace std;
int main()
{int n,x; cin>>n;
set <int> a; for (int i=0;i<n;i++) {cin>>x; a.insert(x);}
set <int>::iterator it=a.begin();
int x1=(*it),x2;
it++;
for (;it!=a.end();it++)
  {x2=(*it);
   if (2*x1>x2) {cout<<"YES"; return 0;}
   x1=x2;
  }
cout <<"NO";
return 0;
}
