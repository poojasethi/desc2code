#include<cstdio> 
#include <algorithm>
using namespace std;
int main(){
	long long n,a[100005];
	scanf("%lld",&n);
	for(int i=0;i<n;i++){
		scanf("%lld",&a[i]);
	}
	sort(a,a+n);
	for(int i=0;i<n-1;i++){
		if(a[i+1]<2*a[i]&&a[i]!=a[i+1]){
			printf("YES\n");return 0;
		}
	}
	printf("NO\n");
	return 0;
}
