#include <bits/stdc++.h>
using namespace std;

long long a[100005];
int main(){
    int n;cin>>n;
    for(int i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);
    for(int i=0;i<n-1;i++){
        if(a[i]!=a[i+1] && 2*a[i] > a[i+1]){
            cout<<"YES\n";
            return 0;
        }
    }
    cout<<"NO\n";

}
