#include<iostream>
#include<algorithm>
using namespace std;

int main(){
  int n;
  int a[100000];

  cin >> n;
  for(int i=0;i<n;i++)cin >> a[i];

  sort(a,a+n);

  bool f = true;
  for(int i=0;i<n;i++){
    if(a[i+1] > a[i] && a[i+1]/a[i] < 2){
      f = false;
      break;
    }
  }
  
  if(f)cout << "NO" << endl;
  else cout << "YES" << endl;
}
