#include<cstdio>
#include<algorithm>
using namespace std;


int main()
{
	int n;
	scanf("%d", &n);
	int a[n];
	for (int i=0; i<n; ++i)
		scanf("%d", &a[i]);
	sort(a, a+n);
	for (int i=1; i<n; ++i)
	{
		if (a[i-1] == a[i]) continue;
		if (a[i-1]*2 > a[i])
		{
			printf("YES\n");
			return 0;
		}
	}
	printf("NO\n");
	return 0;
}
