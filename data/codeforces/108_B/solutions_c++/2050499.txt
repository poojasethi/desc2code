#include<iostream>
#include<algorithm>
using namespace std;
int a[100010];
int main()
{
    int n;
    cin>>n;
    int i;
    for(i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);
    for(i=n-2;i>=0;i--)
    {
        if( a[i+1] > a[i] && a[i] > a[i+1] /2 )break;
    }
    if(i>=0)cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    return 0;
}
