#include <iostream>
#include <algorithm>

using namespace std;

int a[100005];

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	sort(a, a + n);
	for (int i = 0; i < n - 1; i++)
		if (a[i] * 2 > a[i + 1] && a[i] != a[i + 1])
		{
			cout << "YES" << endl;
			return 0;
		}
	cout << "NO" << endl;
}
