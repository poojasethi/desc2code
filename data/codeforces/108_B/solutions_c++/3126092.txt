#include <cstdio>
#include <algorithm>

int main(){
	int arr[100000], n, last = 0;
	scanf("%d",&n);
	for(int i = 0; i < n; i++){
		scanf("%d",&arr[i]);
	}
	std::sort(arr,arr+n);
	for(int i = 0; i < n; i++){
		if(last < arr[i] && last*2 > arr[i]){
			printf("YES\n");
			return 0;
		}
		last = arr[i];
	}
	printf("NO\n");
}
