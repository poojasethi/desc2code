# include <iostream>
# include <algorithm>

using namespace std;

int main()
{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
				cin>>a[i];
		sort(a,a+n);
		for(int i=0;i<n-1;i++)
				if(a[i+1]>a[i]&&a[i+1]<2*a[i])
				{
						cout<<"YES"<<endl;
						return 0;
				}
		cout<<"NO"<<endl;
		return 0;
}
