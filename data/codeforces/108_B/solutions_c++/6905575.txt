#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	__int64 n,i,a[100005];
	while(scanf("%I64d",&n) != EOF)
	{
		for(i=0;i<n;i++)
			scanf("%I64d",&a[i]);
		sort(a,a+n);
		for(i=1;i<n;i++)
			if(a[i-1]*2>a[i]&&a[i-1]<a[i])
				break;
		if(i>=n)
			printf("NO\n");
		else
			printf("YES\n");
	} 
	return 0;
}
