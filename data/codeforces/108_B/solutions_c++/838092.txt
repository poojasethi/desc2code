#include <iostream>
#include <algorithm>
using namespace std;

int main(){
	int n;
	int A[100000+1];
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>A[i];
	sort(A,A+n);
	n=unique(A,A+n)-A;
	int sw=0;
	for(int i=0;i<n-1;i++)
		if(A[i+1]<2*A[i] )
		{
			sw=1;
			break;
		}
	
	if(sw)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
}
