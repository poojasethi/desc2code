#include<stdio.h>
#include<algorithm>

using namespace std;

int n;
int v[100100];

int main (void) {
	scanf("%d",&n);
	int res = 0;

	for(int i = 0; i < n; i++) scanf("%d",&v[i]);
	
	sort(v,v+n);
	
	for(int i = 0; i < n-1 && !res; i++)
		if(2*v[i]>v[i+1] && v[i] != v[i+1]) res = 1;

	printf("%s\n", res == 1 ? "YES" : "NO" );

	return 0;
}
