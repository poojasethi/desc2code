#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;

int main()
{
	int n,a[1000000],i;
	scanf("%d",&n);
	for(int i=0; i<n; i++)
		scanf("%d",&a[i]);
	
	sort(a,a+n);
	for(i=1; i<n; i++)
		if(a[i] != a[i-1] && 2*a[i-1] > a[i])
			break;
	if(i == n)
		printf("NO\n");
	else
		printf("YES\n");
	return 0;
}
