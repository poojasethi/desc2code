#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
__int64 a[100000];
int main()
{
	long n,i;
	cin>>n;
	for(i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	sort(a,a+n);
	for(i=0;i<n-1;i++)
	  if(a[i]<a[i+1]&&a[i]*2>a[i+1])
	  {
		  cout<<"YES"<<endl;
		  return 0;
	  }
		cout<<"NO"<<endl;
	return 0;
	
}
