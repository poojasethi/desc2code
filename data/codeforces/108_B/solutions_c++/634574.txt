#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int a[100010];
int main()
{   int n;
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    int flag=1;
    for(int i=1;i<n;i++)
    {   if(a[i]!=a[i-1] && a[i]<a[i-1]+a[i-1]) flag=0;
    }
    printf("%s\n",flag?"NO":"YES") ;
}
