#include<iostream>
#include<algorithm>
const int maxn = 100*1000+100;
using namespace std;
int a[maxn];
int n;
int main()
{
	cin >> n;
	for(int i=0;i<n;i++)
		cin >> a[i];
	sort(a,a+n);
	n = unique(a,+a+n) - a;
	bool done= false;
	for(int i=1;i<n;i++)
		if(a[i-1]*2 > a[i])
			done = true;
	cout << ((done)?"YES":"NO") << endl;
	return 0;
}
