#include<cstdio>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n,a[100000]; scanf("%d",&n);
	rep(i,n) scanf("%d",a+i);

	sort(a,a+n);
	n=unique(a,a+n)-a;

	bool ok=false;
	rep(i,n-1) if(2*a[i]>a[i+1]) ok=true;
	puts(ok?"YES":"NO");

	return 0;
}
