#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
int dp[35][35][55],t,i,j,k,a,b,c;
int f(int x,int y,int z){
	//cout << x << ' ' << y << ' ' << z << endl;
	if(!x or !y)
	return z==0 ? 0 : 1e6;
	if(x*y==z)
	return 0;
	if(!z)
	return 0;
	if(dp[x][y][z]!=-1)
	return dp[x][y][z];
	//cout << x << ' ' << y << ' ' << z << endl;
	int res,i,j;
	res=1e9;
	for(i=1;i<x;i++)
	for(j=0;j<=z;j++){
		//cout << i << ' ' << j << endl;
	res=min(res,y*y+f(x-i,y,j)+f(i,y,z-j));
    }
	for(i=1;i<y;i++)
	for(j=0;j<=z;j++)
	res=min(res,x*x+f(x,y-i,j)+f(x,i,z-j));
	return dp[x][y][z]=res;
}
int main(){
	scanf("%d",&t);
	memset(dp,-1,sizeof dp);
	for(i=1;i<=t;i++){
		scanf("%d %d %d",&a,&b,&c);
		printf("%d\n",f(a,b,c));
	}
	return 0;
}
