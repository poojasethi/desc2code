#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>
#define inf 1000000000

using namespace std;

int f[50][50][110];
int T,n,m,k;

int dfs(int n,int m,int k)
{
	if (n*m<k) return inf;
	if (n*m==k || k==0) return 0;
	if (f[n][m][k]) return f[n][m][k];
	f[n][m][k]=inf;
	for (int i=1;i<n;i++)
	  for (int j=0;j<=k;j++)
	    f[n][m][k]=min(f[n][m][k],dfs(i,m,j)+dfs(n-i,m,k-j)+m*m);
	for (int i=1;i<m;i++)
	  for (int j=0;j<=k;j++)
	    f[n][m][k]=min(f[n][m][k],dfs(n,i,j)+dfs(n,m-i,k-j)+n*n);
	return f[n][m][k];
}

int main()
{
	scanf("%d",&T);
	while (T--)
	{
		scanf("%d%d%d",&n,&m,&k);
		printf("%d\n",dfs(n,m,k));
	}
	return 0;
}
