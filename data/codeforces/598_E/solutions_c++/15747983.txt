#include<stdio.h>
#include<algorithm>
#define inf 99999
int dp[32][32][52];
int t,n,m,k;
using namespace std;
int main(void)
{
	for(int i=0;i<32;i++)
	{
		for(int j=0;j<32;j++)
		{
			for(int l=0;l<52;l++)
			{
				if(l==0||i*j==l)
				dp[i][j][l]=0;
				else
				dp[i][j][l]=inf;
				for(int r=0;r<=l;r++)
				{
					for(int s=1;s<i;s++)
					dp[i][j][l]=min(dp[i][j][l],dp[s][j][r]+dp[i-s][j][l-r]+j*j);
					for(int t=1;t<j;t++)
					dp[i][j][l]=min(dp[i][j][l],dp[i][t][r]+dp[i][j-t][l-r]+i*i);
				}
			}
		}
	}
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d",&n,&m,&k);
		printf("%d\n",dp[n][m][k]);
	}
	return 0;
}
