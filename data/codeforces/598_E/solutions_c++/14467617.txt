#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int inf=1000000000,tlim=32,plim=52;
int tests,ans,i,j,k,n,m,f[34][34][54],c;
int work(int x,int y,int z){
	int i,res=inf;
	if (x<y) swap(x,y);
	if (f[x][y][z]!=-1) return f[x][y][z];
	if (x*y<z) return f[x][y][z]=inf;
	if (x*y==z||!z) return 0;
	for (i=1;i<x;i++){
		if (i*y<=z) res=min(res,y*y+work(x-i,y,z-i*y));
		res=min(res,y*y+work(x-i,y,z));
	}
	for (i=1;i<y;i++){
		if (x*i<=z) res=min(res,x*x+work(x,y-i,z-x*i));
		res=min(res,x*x+work(x,y-i,z));
	}
	return f[x][y][z]=res;
}
int main(){
	memset(f,-1,sizeof(f));
	for (scanf("%d",&tests);tests--;){
		scanf("%d%d%d",&n,&m,&c);
		ans=work(n,m,c);
		printf("%d\n",ans);
	}
	return 0;
}
