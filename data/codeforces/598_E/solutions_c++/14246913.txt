#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<string>
#include<cstring>
using namespace std;
#define LL long long
#define MAX 0xfffffff
int dp[31][31][51];
int main(){
	int t,n,m,k;
	cin>>t;
	for(int i=1;i<31;i++){
		for(int j=1;j<31;j++){
			for(int k=0;k<51;k++){
				if(i*j==k||k==0){
					dp[i][j][k]=0;
					continue;
				}
				dp[i][j][k]=MAX;
				for(int l=0;l<=k;l++){
					for(int p=1;p<i;p++)
						dp[i][j][k]=min(dp[i][j][k],dp[p][j][k-l]+dp[i-p][j][l]+j*j);
					for(int p=1;p<j;p++)
						dp[i][j][k]=min(dp[i][j][k],dp[i][p][k-l]+dp[i][j-p][l]+i*i);
				}
			}
		}
	}
	while(t--){
		cin>>n>>m>>k;
		cout<<dp[n][m][k]<<endl;
	}
}
