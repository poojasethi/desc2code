#include<bits/stdc++.h>
using namespace std;
int dp[33][33][55];
int solve(int n,int m,int k){
    if(k==0)return 0;
    if(n*m==k)return 0;
    if(k>n*m)return 1e8;
    int &ret=dp[n][m][k];
    if(ret!=-1)return ret;
    ret=1e8;
    for(int i=0;i<=k;++i)
    {
        for(int j=1;j<=n/2;++j){
            ret=min(ret,solve(j,m,i)+solve(n-j,m,k-i)+m*m);
        }
        for(int j=1;j<=m/2;++j)ret=min(ret,solve(n,j,i)+solve(n,m-j,k-i)+n*n);
    }
    return ret;
}

int main()
{
    int t;
    cin>>t;
    memset(dp,-1,sizeof dp);
    int n,m,k;
    while(t--){
        scanf("%d%d%d",&n,&m,&k);
        printf("%d\n",solve(n,m,k));
    }
}
