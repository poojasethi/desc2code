#include <bits/stdc++.h>
using namespace std;
int dp[35][35][55];
const int MAX=10000000;
int cmp(int n,int m,int k)
{
    if(n*m==k||k==0)
        return 0;
    if(dp[n][m][k]!=-1)
        return dp[n][m][k];
    int &ans=dp[n][m][k];
    ans=MAX;
    for(int i=1;i<n;i++)
        for(int j=0;j<=k;j++)
            ans=std::min(ans,cmp(i,m,j)+cmp(n-i,m,k-j)+m*m);
    for(int i=1;i<m;i++)
        for(int j=0;j<=k;j++)
            ans=std::min(ans,cmp(n,i,j)+cmp(n,m-i,k-j)+n*n);
    return ans;
}
int main()
{
    int t,n,m,k;
    scanf("%d",&t);
    memset(dp,-1,sizeof(dp));
    while(t--)
    {
        scanf("%d %d %d",&n,&m,&k);
        printf("%d\n",cmp(n,m,k));
    }
    return 0;
}
