#include <cstdio>
#include <algorithm>
#include <iostream>
#define oo 987654321
using namespace std;
int dt[35][35][55];
int dp(int n, int m, int k)
{
    if( dt[n][m][k] ) return dt[n][m][k];
    int ret = oo;
    if( n*m==k || k == 0 ) return 0;
    for(int i = 1 ; i < m ; i++ )
        for(int j = 0 ; j <= min( k, i*n ) ; j++ )
            ret = min( ret, dp(n, i, j) + dp(n, m-i, k-j) + n*n );
    for(int i = 1 ; i < n ; i++ )
        for(int j = 0 ; j <= min( k, i*m ) ; j++ )
            ret = min( ret, dp(i, m, j) + dp(n-i, m, k-j) + m*m );
    return dt[n][m][k] = ret;
}

int t;

int main()
{
    cin>>t;
    while( t-- )
    {
        int n, m, k;
        cin>>n>>m>>k;
        cout<<dp(n, m, k)<<endl;
    }
}
