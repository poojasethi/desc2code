#include<bits/stdc++.h>
using namespace std;
int dp[50][50][50];
int rec(int n,int m,int k)
{
    if(!k|| n*m == k )return 0;
    if(dp[n][m][k]!=-1)return dp[n][m][k];
    int ret=1e9;
    for(int i=1;i<n;i++){
        for(int k1=0;k1<=k;k1++){
            ret=min(ret,m*m+rec(i,m,k1)+rec(n-i,m,k-k1));
        }
    }
    for(int i=1;i<m;i++){
        for(int k1=0;k1<=k;k1++){
            ret=min(ret,n*n+rec(n,i,k1)+rec(n,m-i,k-k1));
        }
    }
    return dp[n][m][k]=ret;
}
int main()
{
    memset(dp,-1,sizeof dp);
    int t;
    cin>>t;
    while(t--){
        int n,m,k;
        cin>>n>>m>>k;
        cout<<rec(n,m,k)<<"\n";
    }
}
