#include <bits/stdc++.h>
using namespace std;
int dp[31][31][55];
int sol(int n,int m,int k){
	if(k<0)return 1e9;
	if(k>n*m)return 1e9;
	if(k==0)return 0;
	if(n*m==k)return 0;
	int r,i,j,&ret= dp[n][m][k];
	if(ret!=0x3f3f3f3f)return ret;
	for(i=1;i<n;++i){
		r= n-i;
		ret=min(ret,m*m+sol(i,m,k));
		ret= min(ret, m*m + sol(r,m,k));
		ret= min(ret,m*m+sol(r,m,k-m*i));
		ret= min(ret,m*m+sol(i,m,k-m*r));
	}
	for(i=1;i<m;++i){
		r=m-i;
		ret=min(ret,n*n+sol(n,i,k));
		ret=min(ret,n*n + sol(n,r,k));
		ret= min(ret,n*n+sol(n,r,k-n*i));
		ret=min(ret,n*n+sol(n,i,k-n*r));
	}
	return ret;
}
int main(){
	int t,n,m,k,i,j;scanf("%d",&t);
	memset(dp,0x3f,sizeof(dp));
	for(i=0;i<t;++i){
		scanf("%d%d%d",&n,&m,&k);
		printf("%d\n",sol(n,m,k));
	}
}
