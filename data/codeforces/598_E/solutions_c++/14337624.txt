#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll dp[32][32][55];
void solve(int m,int n,int k)
{
	int i,j,p;
	ll  c1=INT_MAX,c2=INT_MAX;
	if(k==m*n ||k==0)
		dp[m][n][k]= 0;
	else
	{
		for(i=1;i<m;i++)
		{
			for(p=0;p<=k;p++)
				c1=min(c1, n*n + dp[i][n][p] + dp[m-i][n][k-p]);
		}
		for(j=1;j<n;j++)
		{
			for(p=0;p<=k;p++)
				c2=min(c2, m*m + dp[m][j][p] + dp[m][n-j][k-p]);
		}
		dp[m][n][k]=min(c1,c2);
	}
}
int main()
{
	int T,n,m,k,x;
	for(m=1;m<=30;m++)
	{
		for(n=1;n<=30;n++)
		{
			for(k=0;k<=50;k++)
				dp[m][n][k]=INT_MAX;
		}
	}
	for(m=1;m<=30;m++)
	{
		for(n=1;n<=30;n++)
		{
			for(k=0;k<=min(m*n,50);k++)
				solve(m,n,k);
		}
	}
	cin>>T;
	while(T--)
	{
		cin>>m>>n>>k;
		cout<<dp[m][n][k]<<endl;
	}
}
