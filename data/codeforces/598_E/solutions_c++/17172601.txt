#include<stdio.h>
#include<algorithm>
using namespace std;
int pcp[35][35][920]={0},t,n,m,k;
int query(int n,int m,int k)
{
  //printf("%d %d %d\n",n,m,k);
  if(pcp[n][m][k]!=0) return pcp[n][m][k];
  if(k==0||k==n*m) return 0;
  int ans=1e9;
  for(int i=1;i<=n-1;i++)
  {
    ans=min(ans,(m*m)+query(i,m,min(k,i*m))+query(n-i,m,k-(min(k,i*m))));
  }
  for(int i=1;i<=m-1;i++)
  {
    ans=min(ans,(n*n)+query(n,i,min(k,i*n))+query(n,m-i,k-(min(k,i*n))));
  }
  //printf("%d %d %d %d\n",n,m,k,ans);
  return pcp[n][m][k]=ans;
}
int main()
{
  scanf("%d",&t);
  while(t>0)
  {
  scanf("%d %d %d",&n,&m,&k);
  printf("%d\n",query(n,m,k));
  t--;
  }
return 0;
}
