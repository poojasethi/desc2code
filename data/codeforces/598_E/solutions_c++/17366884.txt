#include "stdio.h"
#include "algorithm"
#define INF 0x3f3f3f3f
using namespace std;
int n,m,K;
int dp[30+5][30+5][50+5];

int main()
{
	int i,j,k,u,v;
	for(i=1;i<=30;i++)
	{
		for(j=1;j<=30;j++)
		{
			for(k=0;k<=50;k++)
			{
				if(k==0||k==i*j){
					dp[i][j][k]=0;
				}else{
					dp[i][j][k]=INF;
				}
				for(u=0;u<=k;u++)
				{
					for(v=1;v<i;v++)
					{
						dp[i][j][k]=min(dp[i][j][k],dp[v][j][u]+dp[i-v][j][k-u]+j*j);
					}
					for(v=1;v<j;v++)
					{
						dp[i][j][k]=min(dp[i][j][k],dp[i][v][u]+dp[i][j-v][k-u]+i*i);
					}
				}
			}
		}
	}
	int ca;
	scanf("%d",&ca);
	while(ca--)
	{
		scanf("%d%d%d",&n,&m,&K);
		printf("%d\n",dp[n][m][K]);
	}
	return 0;
}
