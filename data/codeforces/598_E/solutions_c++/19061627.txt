#include <algorithm>
#include <cstdio>
const int inf=1e9;
int f[35][35][55],i,j,k,l,t;
int main()
{
	for (i=1;i<=30;++i)
		for (j=1;j<=30;++j)
		{
			for (k=1;k<i*j && k<=50;++k) f[i][j][k]=inf;
			if (i*j<=50) f[i][j][i*j]=0;
			for (k=1;k<i*j && k<=50;++k)
			{
				for (l=1;l<i;++l)
					if (l*j>=k) f[i][j][k]=std::min(f[i][j][k],f[l][j][k]+j*j);
						else f[i][j][k]=std::min(f[i][j][k],f[i-l][j][k-l*j]+j*j);
				for (l=1;l<j;++l)
					if (i*l>=k) f[i][j][k]=std::min(f[i][j][k],f[i][l][k]+i*i);
						else f[i][j][k]=std::min(f[i][j][k],f[i][j-l][k-i*l]+i*i);
			}
		}
	for (scanf("%d\n",&t);t--;)
	{
		scanf("%d%d%d\n",&i,&j,&k);
		printf("%d\n",f[i][j][k]);
	}
}
