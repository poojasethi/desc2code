#include <iostream>
using namespace std;
int DP[31][31][51];

int min(int x, int y){
	if (x<y) return x; else return y;
}

int f(int x, int y, int k){
	if (DP[x][y][k] != -1) return DP[x][y][k];
	int ans = 1e9;
	if (x*y == k || k==0) {
		DP[x][y][k] = 0;
		return 0;
	}
	for (int j=0;j<=k;j++){
		for (int i=1;i<x;i++)
			ans = min(ans, f(i,y,j)+f(x-i,y,k-j)+y*y);
		for (int i=1;i<y;i++)
			ans = min(ans, f(x,i,j)+f(x,y-i,k-j)+x*x);
	}
	DP[x][y][k] = ans;
	return ans;	
}

int main(){
	int t; cin >> t;
	int n,m,k;
	for (int j=0;j<31;j++)
			for (int k=0;k<31;k++)
				for (int l=0;l<51;l++) 
					DP[j][k][l] = -1;
	for (int i=0;i<t;i++){
		cin >> n >> m >> k;
		cout << f(n,m,k) << endl;
	}
}
