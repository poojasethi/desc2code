#include <cstdio>
#include <cstring>
#include <algorithm>
#define INF 100000007

using namespace std;

int f [55][55][55];

int dfs (int x, int y, int k)
{
	if (k < 0) return INF;
	if (~f[x][y][k]) return f[x][y][k];
	if (!k || x * y == k) return 0;
	f[x][y][k] = INF;
	for (int i = 1; i < x; ++i)
		f[x][y][k] = min (min (dfs (i, y, k), dfs (i, y, k - (x - i) * y)) + y * y, f[x][y][k]);
	for (int i = 1; i < y; ++i)
		f[x][y][k] = min (min (dfs (x, i, k), dfs (x, i, k - (y - i) * x)) + x * x, f[x][y][k]);
	return f[x][y][k];
}

int main ()
{
	
	memset (f, 255, sizeof (f));
	int casx;
	scanf ("%d", &casx);
	while (casx--)
	{
		int x, y, k;
		scanf ("%d%d%d", &x, &y, &k);
		printf ("%d\n", dfs (x, y, k));
	}
	return 0;
}
