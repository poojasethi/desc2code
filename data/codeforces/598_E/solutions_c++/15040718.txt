#include<iostream>
#include<cstdio>
using namespace std;
#define inf 1000000007
int dp[31][31][51];

void gen()
{
for(int i=0;i<=30;i++)
for(int j=0;j<=30;j++)
for(int k=0;k<=50;k++)
if(k==0)
dp[i][j][k]=0;
else if(i*j==k)
dp[i][j][k]=0;
else
{
dp[i][j][k]=inf;
for(int m=0;m<=k;m++)
{
for(int n=1;n<=i-1;n++)
dp[i][j][k]=min(dp[i][j][k],dp[n][j][m]+dp[i-n][j][k-m]+j*j);
for(int n=1;n<=j-1;n++)
dp[i][j][k]=min(dp[i][j][k],dp[i][n][m]+dp[i][j-n][k-m]+i*i);
}
}
}

int main()
{
ios_base::sync_with_stdio(false);
gen();
int t;
cin>>t;
while(t--)
{
int n,m,k;
cin>>n>>m>>k;
cout<<dp[n][m][k]<<endl;
}
}
