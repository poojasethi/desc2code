#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;
int dp[35][35][55];
int n,m,k;

int dfs(int n,int m,int k){
	int &ans=dp[n][m][k];
	if(~dp[n][m][k]) return ans;
	if(!k || k==n*m) return ans=0;
	ans=0x3f3f3f3f;
	for(int i=1;2*i<=n;i++){
		for(int j=0;j<=k;j++){
			int temp=m*m+dfs(i,m,j)+dfs(n-i,m,k-j);
			ans=min(ans,temp);
		}
	}
	for(int i=1;2*i<=m;i++){
		for(int j=0;j<=k;j++){
			int temp=n*n+dfs(n,i,j)+dfs(n,m-i,k-j);
			ans=min(ans,temp);
		}
	}
	return ans;
}

int main(void){
	int i,j,t;
	scanf("%d",&t);
	memset(dp,-1,sizeof(dp));
	while(t--){
		scanf("%d%d%d",&n,&m,&k);
		printf("%d\n",dfs(n,m,k));
	}
	return 0;
}
