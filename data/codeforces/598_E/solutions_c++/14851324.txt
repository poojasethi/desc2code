#include <bits/stdc++.h>
using namespace std;


int dp[32][32][1000];

int calc(int n, int m, int k) {

    if(dp[n][m][k] != 0 || k==0 || k == n*m) return dp[n][m][k];

    int res1 = 9999999;
    for(int i=1; 2*i <= n; i++) {
        for(int j=0;j<=k;j++) {
            res1 = min(res1, m*m + calc(i,m,j) + calc(n-i,m,k-j));
        }
    }

    int res2 = 9999999;
    for(int i=1; 2*i <= m; i++) {
        for(int j=0;j<=k;j++) {
            res2 = min(res2, n*n + calc(n,i,j) + calc(n,m-i,k-j));
        }
    }

    return dp[n][m][k] = min(res1, res2);

}

int main () {

    int n,m,k;

    int t;
    scanf("%d", &t);
    for(int i=0;i<t;i++) {
        scanf("%d %d %d", &n, &m,&k);
        printf("%d\n", calc(n,m,k));
    }


}
