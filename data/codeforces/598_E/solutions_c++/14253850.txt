// same idea using memo

#include <iostream>
#include <stdio.h>

using namespace std;

int dp[37][37][57];
bool used[37][37][57];

int solve(int n,int m,int k)
{
	if(k == 0) return 0;
	if(n*m == k) return 0;
	if(used[n][m][k]) return dp[n][m][k];
	used[n][m][k] = 1;
	int &tp = dp[n][m][k];
	tp = 1e9;
	for(int i=1;i<=n-i;i++)
	{
		for(int j=0;j<=k;j++)
		{
			tp = min(tp,solve(n-i,m,k-j)+solve(i,m,j)+m*m);
		}
	}
	for(int i=1;i<=m-i;i++)
	{
		for(int j=0;j<=k;j++)
		{
			tp = min(tp,solve(n,m-i,k-j)+solve(n,i,j)+n*n);
		}
	}
	return tp;
}

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
	int t,n,m,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>m>>k;
		printf("%d\n",solve(n,m,k));
	}
	return 0;
}
