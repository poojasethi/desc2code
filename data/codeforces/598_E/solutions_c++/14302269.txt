#include <iostream>
#include <iomanip>
using namespace std;
int a[33][33][55];
int f(int n, int m, int k) {
	if (a[n][m][k] || n * m == k || k == 0)
		return a[n][m][k];
	int res = 1e9;
	for (int i = 1; i <= n - i; ++i) {
		for (int j = 0; j <= k; ++j) {
			res = min(res, m * m + f(i, m, k - j) + f(n - i, m, j));
		}
	}
	for (int i = 1; i <= m - i; ++i) {
		for (int j = 0; j <= k; ++j) {
			res = min(res, n * n + f(n, i, k - j) + f(n, m - i, j));
		}
	}
	return a[n][m][k] = res;
}
int main() {
	//freopen("in.txt", "r", stdin);
	int t, n, m, k;
	cin >> t;
	for (int i = 0; i < t; ++i) {
		cin >> n >> m >> k;
		cout << f(n, m, k) << endl;
	}
	return 0;
}
