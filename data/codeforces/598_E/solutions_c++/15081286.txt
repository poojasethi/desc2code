#include <iostream>
#include <cstdio>
using namespace std;

int t,n,m,k,d[35][35][55];
int f(int p,int q,int r){
    int i,j,mn=1e9;
    if(d[p][q][r] || p*q==r || !r) return d[p][q][r];
    for(i=1; i<=p/2; i++){
        for(j=0; j<=r; j++){
            mn = min(mn, q*q+f(i,q,j)+f(p-i,q,r-j));
        }
    }
    for(i=1; i<=q/2; i++){
        for(j=0; j<=r; j++){
            mn = min(mn, p*p+f(p,i,j)+f(p,q-i,r-j));
        }
    }
    return d[p][q][r]=mn;
}

int main(){
    cin>>t;
    while(t--){
        scanf("%d%d%d",&n,&m,&k);
        printf("%d\n",f(n,m,k));
    }
    return 0;
}
