#include <iostream>

using namespace std;
long long t,n,m,k,a[32][32][52];
int main()
{
 for (int i=1;i<=30;i++)
        for (int j=1;j<=30;j++)
                for (k=1;k<=50;k++) {
                        if (i*j==k || k==0); else a[i][j][k]=1e9;
                        for (int kk=0;kk<=k;kk++) {
                            for (int ii=1;ii<i;ii++) a[i][j][k]=min(a[i][j][k],a[ii][j][kk]+a[i-ii][j][k-kk]+j*j);
                            for (int jj=1;jj<j;jj++) a[i][j][k]=min(a[i][j][k],a[i][jj][kk]+a[i][j-jj][k-kk]+i*i);
                                                     }
                                        }
 cin>>t;

    for (int i=1;i<=t;i++)
        {cin>>n>>m>>k; cout<<a[n][m][k]<<endl;}

    return 0;
}
