#include <bits/stdc++.h>

using namespace std;

#define INF 1000000000
int dp[40][40][60];


int f(int n , int m , int k ){

if( dp[n][m][k] != -1 )return dp[n][m][k];
if( n*m == k || k==0 )return dp[n][m][k]=0;
if( n*m < k )return dp[n][m][k]=INF;
int res = INF ;

for(int K=1;K<=k;K++){

  for(int N = 1; N<n ; N++){
    int ans = f(N,m,K)+f(n-N,m,k-K);
  if(ans < INF){ res = min ( res , ans + m*m );  }
  }

  for(int M = 1; M<m ; M++){
    int ans = f(n,M,K)+f(n,m-M,k-K);
  if(ans < INF){ res = min ( res , ans + n*n );  }
  }

}

return dp[n][m][k]=res;
}

int main()
{
memset( dp , -1 , sizeof dp );
int n , a , b , k ;
cin>>n;

while(n--){

cin>>a>>b>>k;

cout<<f(a,b,k)<<endl;

}



    return 0;
}
