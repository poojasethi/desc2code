#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

int dp[31][31][51];

int f(int x, int y, int k) {
	if(dp[x][y][k] || x*y==k || !k) return dp[x][y][k];
	int res = 1e9;
	for(int i = 1; i <= x - i; i++) {
		for(int j = 0; j <= k; j++) {
			res = min(res, y*y + f(i, y, j) + f(x-i, y, k-j));
		}
	}
	for(int i = 1; i <= y - i ; i++) {
		for(int j = 0; j <= k; j++) {
			res = min(res, x*x + f(x, i, j) + f(x, y-i, k-j));
		}
	}
	return dp[x][y][k] = res;
}
int main() {
	int T;
	cin>>T;
	while(T--) {
		int x, y, k;
		cin>>x>>y>>k;
		cout<<f(x, y, k)<<endl;
	}
	return 0;
}
