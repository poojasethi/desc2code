#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstdlib>
#include<iostream>
const int inf=1e9;
using namespace std;
int T,n,m,k,f[31][31][51];
int go(int x,int y,int z)
{
	if (x*y==z || !z) return 0;
	if (f[x][y][z]) return f[x][y][z];
	int t=inf;
	for (int i=1;i<x;i++)
		for (int j=0;j<=z;j++)
			t=min(t,y*y+go(i,y,j)+go(x-i,y,z-j));
	for (int i=1;i<y;i++)
		for (int j=0;j<=z;j++)
			t=min(t,x*x+go(x,i,j)+go(x,y-i,z-j));
	return f[x][y][z]=t;
}

int main()
{
		for (scanf("%d\n",&T);T;T--)
		{
			scanf("%d%d%d\n",&n,&m,&k);
			go(n,m,k);
			printf("%d\n",f[n][m][k]);
		}
}
