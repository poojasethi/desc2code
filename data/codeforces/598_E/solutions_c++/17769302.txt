#include<bits/stdc++.h>
using namespace std;

int dp[31][31][51], t, n, m, k;

int calc(int x, int y, int k){
	if(k == 0) return 0;
	if(x*y == k) return 0;
	int &ret = dp[x][y][k];
	if(ret != -1) return ret;
	ret = 1e5;
	for(int i=1 ; i<=k ; ++i) {
		for(int j=1 ; j<x ; ++j)
			ret = min(ret, calc(x-j, y, i) + calc(j, y, k-i) + y*y);
		for(int j=1 ; j<y ; ++j)
			ret = min(ret, calc(x, y-j, i) + calc(x, j, k-i) + x*x);
	}
	return ret;
}

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &t);
	memset(dp, -1, sizeof dp);
	while(t--){
		scanf("%d%d%d", &n, &m, &k);
		printf("%d\n", calc(n, m, k));
	}
	return 0;
}
