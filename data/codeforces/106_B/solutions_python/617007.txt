import sys
def read_values():
  return map(int,raw_input().split())

n=input()
t=[]
for i in range(n):
  t.append(read_values())

ok = [True]*n
for i in range(n):
  for j in range(n):
    old = True
    for k in range(3):
      old &= t[i][k]<t[j][k]
    ok[i] &= not old

res=-1
p=100000
for i in range(n):
  if ok[i]:
    if t[i][3]<p:
      res=i
      p=t[i][3]
res +=1 
print res
