n = int(raw_input())

l = []
for i in range(n):
    s, r, h, c = map(int, raw_input().split())
    l.append((s,r,h,c,i+1))

ll = []
for i in l:
    y = True
    for j in l:
        if i[0] < j[0] and i[1] < j[1] and i[2] < j[2]:
            y = False
    if y:
        ll.append(i)

a = sorted(ll, key = lambda x: x[3])
print a[0][4]
