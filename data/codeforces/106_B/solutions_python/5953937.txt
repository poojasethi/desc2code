import sys


n=input()
l=[map(int,raw_input().split()) for i in range(n)]
o=[False]*n


for j in range (n):
   for i in range(n):
      if l[j][0]<l[i][0] and l[j][1]<l[i][1] and l[j][2]<l[i][2]:
         o[j]=True
      i+=1
        
index=0
cost=1001
for i in range(n):
   if not o[i]:
      if l[i][3]<cost:
         index=i
         cost=l[i][3]

print index+1
