n=input()
c=[map(int,raw_input().split())+[i+1,True] for i in range(n)]
for i in range(n):
  if not c[i][-1]:
    continue
  for j in range(n):
    if i==j or not c[j][-1]:
      continue
    if all(map(lambda x:c[i][x]<c[j][x],range(3))):
      c[i][-1]=False
      break
print min(filter(lambda x:x[-1],c),key=lambda x:x[3])[4]
