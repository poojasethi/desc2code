

N = input()
n = []
cnt = 1
for i in range(N):
    speed, ram, hdd, cost = map(int, raw_input().split())
    n.append({'speed':speed, 'ram':ram, 'hdd':hdd, 'cost':cost, 'id':cnt})
    cnt += 1
old = []
for nb in n:
    for nb2 in n:
        if nb['ram'] < nb2['ram'] and nb['speed'] < nb2['speed'] and nb['hdd'] < nb2['hdd']:
            old.append(nb['id'])
            break
nn = []
for nb in n:
    if nb['id'] not in old:
        nn.append(nb)
print min(nn, key=lambda x: x['cost'])['id']
