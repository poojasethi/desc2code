from sys import stdin

n = int(stdin.readline())
v = [map(int, stdin.readline().split(' ')) for i in range(n)]
bad = [False]*n

for i in range(n):
    for j in range(n):
        if all([v[i][k] < v[j][k] for k in range(3)]):
            bad[i] = True
        
best = 10**10
for i in range(n):
    if not bad[i] and v[i][3] < best:
        rv, best = i, v[i][3]

print 1 + rv
