import sys

lines = sys.stdin.readlines()
computers = []
for line in lines[1:]:
    computers.append([int(x) for x in line.split(' ')])

weakest = []
prices = []
for k, c in enumerate(computers):
    prices.append((c[3], k)) 
    for ak, ac in enumerate(computers):
        weak = True
        if c is not ac:
            for i in range(0, 3):
                if c[i] >= ac[i]:
                    weak = False

            if weak:
                weakest.append(k)

prices = filter(lambda x: not x[1] in weakest, prices)
prices.sort()

print prices[0][1] + 1
