from operator import itemgetter
A = []
def isOutDated ( t ) :
	for i in range(len(A)):
		if t == i :
			continue
		if A[t][0] < A[i][0] and A[t][1] < A[i][1] and A[t][2] < A[i][2]  :
			return True 
	return False	
n = input()
for i in range (n) :
	s = raw_input("").split()
	for j in range (4) :
		s[j] = int(s[j])
	A.append( (s[0],s[1],s[2],s[3],i+1) )
A = sorted ( A , key = itemgetter(3) )
for i in range(len(A)):
	if isOutDated(i) == False :
		print A[i][4] 
		exit()
