ls = []
for i in xrange(input()):
    ls.append(map(int, raw_input().split()))
ban = []
for i in xrange(len(ls)):
    for j in xrange(len(ls)):
        if i == j:
            continue
        for idx in xrange(3):
            if ls[i][idx] <= ls[j][idx]:
                break
        else:
            ban.append(j)
idx = -1
for i in xrange(len(ls)):
    if i in ban:
        continue
    if idx == -1 or ls[idx][3] > ls[i][3]:
        idx = i
print idx+1
