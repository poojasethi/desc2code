#!/usr/bin/env python
test = 'test'


from pprint import pprint
import sys

input = sys.stdin
#input  = open('test.in')

def solve():
  P = int(input.readline())
  L = [tuple(map(int, input.readline().split())) for _ in xrange(P)]
  best_i = None
  best_p = None
  i = 0
  for l in L:
    obsolete = False
    for m in L:
      if l[0] < m[0] and l[1] < m[1] and l[2] < m[2]:
        obsolete = True
        break
    if not obsolete:
      if best_i is None or best_p > l[3]:
        best_i = i
        best_p = l[3]
    i += 1
  return best_i+1

print solve()
