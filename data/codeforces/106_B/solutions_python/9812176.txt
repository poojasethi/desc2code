n = input()
l = [map(int,raw_input().split()) for _ in xrange(n)]
o = [1]*n
for x in xrange(n):
    for y in xrange(n):
        if l[x][0] < l[y][0] and l[x][1] < l[y][1] and l[x][2] < l[y][2]:
            o[x] = 0
for x in sorted(enumerate(l),key=lambda x:x[1][3]):
    if o[x[0]]:
        print x[0]+1
        break
