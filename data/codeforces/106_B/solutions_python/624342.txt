#!/usr/bin/env python
# -*- coding: utf-8 -*-

n = int(raw_input())
l = []
for i in xrange(1,n+1):
    l.append([i] + map(int, raw_input().split()))
l.sort(key = lambda x: x[4])
i = 0
while True:
    pr = True
    for bb in l:
        if l[i][1] < bb[1] and l[i][2] < bb[2] and l[i][3] < bb[3]:
            pr = False
            break
    if pr:
        print l[i][0]
        break
    i += 1
