from sys import stdin, stdout

n = int(stdin.readline().strip())

c = []

a = []

for i in range(0, n):
    c.append([int(x) for x in stdin.readline().strip().split()])

for i in range(0, n):
    yes = True
    for j in range(0, n):
        if c[i][0]<c[j][0] and c[i][1]<c[j][1] and c[i][2]<c[j][2]:
            yes = False
    if yes:
        a.append([c[i][3], i])

print(min(a)[1]+1)
