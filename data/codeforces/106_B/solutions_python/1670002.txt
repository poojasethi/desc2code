n = input()
l = []
for _ in xrange(n):
  l.append(map(int, raw_input().split()))
ng = []
for i in xrange(len(l)):
  for j in xrange(len(l)):
    if i == j:
      continue
    for k in xrange(3):
      if l[i][k] <= l[j][k]:
        break
    else:
      ng.append(j)
res = -1
for i in xrange(len(l)):
  if i in ng:
    continue
  if res == -1 or l[res][3] > l[i][3]:
    res = i
print res + 1
