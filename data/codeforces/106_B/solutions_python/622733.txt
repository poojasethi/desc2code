n = input()

a = []
for i in range(n):
    b = map(int, raw_input().split())
    a.append(b)

def IsOut(d):
    for i in range(n):
        if a[d][0] < a[i][0] and a[d][1] < a[i][1] and a[d][2] < a[i][2]:
            return True
    return False

for i in range(n):
    if IsOut(i):
        a[i][3] = 100000


ans = 0
for i in range(n):
    if a[i][3] < a[ans][3]:
        ans = i
print ans + 1
