from sys import stdin

buf = []
for line in stdin:
    buf.append(line.strip())
buf.reverse()

def getints():
    return [int(e) for e in buf.pop().split()]

n = int(buf.pop())
l = []
for i in range(n):
    l.append([e for e in reversed(getints())] + [i+1])
l.sort()

for i in range(n):
    for j in range(i+1, n):
        if all(l[i][k] < l[j][k] for k in [1, 2, 3]):
            break
    else:
        print l[i][4]
        break
