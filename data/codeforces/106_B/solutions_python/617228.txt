n = int(raw_input())

books = []
for i in range(n):
    speed, ram, hdd, cost = [int(v) for v in raw_input().split()]
    books.append((speed, ram, hdd, cost, i + 1))

good = []
for b in books:
    speed, ram, hdd, cost, num = b
    if len([i for i in books if i[0] > speed and i[1] > ram and i[2] > hdd]) > 0:
        continue
    else:
        good.append(b)

print sorted(good, key=lambda x: x[3])[0][4]
