#include <iostream>

using namespace std;

int main()
{
	int n;
	cin>>n;
	
	int N[n][4];
	
	for(int i=0; i<n; i++)
		cin>>N[i][0]>>N[i][1]>>N[i][2]>>N[i][3];
	
	int minI = -1;
	
	for(int i=0; i<n; i++)
	{
		bool ok = 1;
		
		for(int j=0; j<n; j++)
		{
			if(i==j) continue;
			
			if(N[i][0] < N[j][0] && N[i][1] < N[j][1] && N[i][2] < N[j][2])
				ok = 0;
		}
		
		if(ok && (minI == -1 || N[i][3] < N[minI][3])) minI = i;
	}
	
	cout<<minI + 1<<endl;
	
	return 0;
}
