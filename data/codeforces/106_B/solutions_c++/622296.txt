#include<iostream>
#include<cstdio>

using namespace std;
const int N=105;
int a[N],b[N],c[N],w[N],v[N];

int main()
{
    int n,i,j,k,min;
    cin>>n;
    for (i=1; i<=n; i++) cin>>a[i]>>b[i]>>c[i]>>w[i];
    for (i=1; i<=n; i++)
     for (j=1; j<=n; j++)
      if (a[i]<a[j] && b[i]<b[j] && c[i]<c[j])
       v[i]=1;
    min=10000000;
    for (i=1; i<=n; i++)
     if (v[i]==0)
      if (w[i]<min)
       {min=w[i]; k=i;}
    cout<<k<<endl;
}
