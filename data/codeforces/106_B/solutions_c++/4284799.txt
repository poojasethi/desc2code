#include<iostream>
using namespace std;
int n,a[111],b[111],c[111],d[111],ans,ret=10000000;
int main()
{
     int i,j;
     cin>>n;
     for(i=0;i<n;i++)
     {cin>>a[i]>>b[i]>>c[i]>>d[i];}
     for(i=0;i<n;i++)
     {
         for(j=0;j<n;j++)
         {
              if(a[i]<a[j]&&b[i]<b[j]&&c[i]<c[j])d[i]=200000000;
         }
     }
     for(i=0;i<n;i++)
     if(d[i]<ret){ret=d[i];ans=i;}
     cout<<ans+1;
     return 0;
}
