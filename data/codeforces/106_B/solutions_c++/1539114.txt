#include <iostream>

using namespace std;

int N,S[110],R[110],H[110],C[110],result,cost=1001;

bool isOutdated(int i)
{
	for (int j=0;j<N;j++)
		if (S[i]<S[j]&&R[i]<R[j]&&H[i]<H[j])
			return true;
	return false;
}

int main()
{
	cin >> N;
	for (int i=0;i<N;i++)
		cin >> S[i] >> R[i] >> H[i] >> C[i];
	for (int i=0;i<N;i++)
		if (!isOutdated(i)&&C[i]<cost)
		{
			cost=C[i];
			result=i+1;
		}
	cout << result << endl;
	return 0;
}
