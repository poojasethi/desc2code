#include <iostream>

using namespace std;

int a, i, z[1111], x[1111], c[1111], v[1111], q = 1111, ans, j;

int main ()
{
	cin >> a;
		for (i = 1; i <= a; i++)
			cin >> z[i] >> x[i] >> c[i] >> v[i];
   	for (i = 1; i <= a; i++)
   		for (j = 1; j <= a; j++)
   			if (z[i] < z[j] && x[i] < x[j] && c[i] < c[j]) {z[i] = 0;}
   	for (i = 1; i <= a; i++)
   		if (z[i] != 0 && v[i] < q) {q = v[i]; ans = i;}
   	cout << ans;
}
