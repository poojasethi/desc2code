#include<iostream>
using namespace std;
int mj,_min=100000000,stn=-1,st[1000],s1,j,i,n,a[100],b[100],c[100],d[100];
int main(){
	cin>>n;
	for(i=0;i<n;i++)cin>>a[i]>>b[i]>>c[i]>>d[i];
	for(i=0;i<n;i++){
		s1=0;
		for(j=0;j<n;j++)if ((j!=i)&&((a[i]<a[j])&&(b[i]<b[j])&&(c[i]<c[j])))s1=1;	
		if(s1==0){
			stn++;
			st[stn]=i;	
		} 
	}
	for(i=0;i<=stn;i++)if (_min>d[st[i]]){
		_min=d[st[i]];
		mj=st[i];
	}
	cout<<mj+1;
}
