#include<algorithm>
using namespace std;
int main()
{
      int n,s[101],r[101],h[101],p[101],i,j,min=1001,sum;

for(i=0;i<n;i++)
scanf("%d%d%d%d",&s[i],&r[i],&h[i],&p[i]);

for(i=0;i<n;i++)
for(j=0;j<n;j++)
if(s[i]<s[j]&&r[i]<r[j]&&h[i]<h[j])
{

break; 
}      
for(i=0;i<n;i++)
if(p[i]<min)
{
min=p[i];
sum=i;      
 }     printf("%d",sum+1);
     
      
    return 0;
}
