#include<stdio.h>
#include<algorithm>
using namespace std;


int vis[1002],a[102],b[102],c[102],d[102];

int main()
{
	int N;
	scanf("%d",&N);
	for(int i=0;i<N;i++)
	{
		scanf("%d %d %d %d",&a[i],&b[i],&c[i],&d[i]);
		
	}
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<N;j++)
		{
			if(a[i]<a[j] && b[i]<b[j] && c[i]<c[j])
			{
				vis[i]=1;
			}
		}
	}
	int ret=1<<30,ind;
	for(int i=0;i<N;i++)if(!vis[i])
	{
		if(ret>d[i])
		{
			ret=d[i];
			ind=i+1;
		}
	}
	printf("%d\n",ind);
}
