#include <iostream>
using namespace std;
int a[110],b[110],c[110],d[110],n,k,j;
int main() {
   
    cin >> n;
    for(int i=0;i<n;i++) {
    cin >> a[i] >> b[i] >> c[i] >> d[i];
     k=-1;
     }
    for(int i=0;i<n;i++) {
    for(j=0;j<n;j++) 
 if (a[j]>a[i] && b[j]>b[i] && c[j]>c[i]) 
               break;
        if (j<n) 
        continue;
        if (k==-1 || d[k]>d[i]) k=i;
    }
    cout << k+1 << endl;
   //system("pause");
}
