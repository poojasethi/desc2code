#include<iostream>
using namespace std;

int main(){
int n, i, j, u[111]={0}, k=0, mini=1111, l;
struct po{int x, y, u, v;};
po a[111];
cin>>n;
for (i=1; i<=n; i++){
cin>>a[i].x>>a[i].y>>a[i].u>>a[i].v;
}
for (i=1; i<=n; i++){
for (j=1; j<=n; j++){
if (a[i].x<a[j].x && a[i].y<a[j].y && a[i].u<a[j].u){
u[i]=1;
break;
}
}
}
for (i=1; i<=n; i++){
if (u[i]==0){
if (a[i].v<mini){
mini=a[i].v;
l=i;
}
}
}
cout<<l;
return 0;
}
