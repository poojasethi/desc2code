#include <iostream>

using namespace std;

int main()
{
    int n,s[100],r[100],a[100],c[100],temp=0;
    cin>>n;
    for (int i=0;i<n;i++) cin>>s[i]>>r[i]>>a[i]>>c[i];
    for (int j=0;j<n;j++)
    {
        for (int k=0;k<n;k++)
        {
            if (s[j]<s[k] && r[j]<r[k] && a[j]<a[k]) c[j]=1001;
        }
    }
    for (int x=1;x<n;x++)
    {
        if (c[temp]>c[x]) temp=x;
    }
    cout<<(temp+1)<<endl;
    return 0;
}
