#include <iostream>
#include <cstdlib>
using namespace std;
int c[105][5];
int main(){
    int i,j,n,cheap;
    cin>>n;
    for(i=1;i<=n;i++)
    for(j=1;j<=4;j++)
    cin>>c[i][j];
    for(i=1;i<=n;i++)
    for(j=1;j<=n;j++)
    if(c[i][1]<c[j][1] && c[i][2]<c[j][2] && c[i][3]<c[j][3])c[i][4]=2147483647;
    cheap=1;
    for(i=2;i<=n;i++)
    if(c[i][4]<c[cheap][4])cheap=i;
    cout<<cheap;
    return 0;
}
