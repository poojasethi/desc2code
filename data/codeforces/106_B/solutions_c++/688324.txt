#include <cstdio>
#include <cstdlib>

const int N = 110;
int a[N], b[N], c[N], d[N], n, v[N], ans = (1 << 30), V;
int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d%d%d%d", a + i, b + i, c + i, d + i);
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++)
			if (a[j] > a[i] && b[j] > b[i] && c[j] > c[i])
				v[i] = 1;
		if (!v[i] && ans > d[i])
			ans = d[i], V = i;
	}
	printf("%d", V + 1);
	return 0;
}
