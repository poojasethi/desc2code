#include<iostream>
#include<cstdio>
using namespace std;
#define N 101
int i,j,k,n,f=1,a[N][4],suck[N],cheapest=1<<30,ans;
int main(){
 for(cin>>n,i=1;i<=n;++i)for(j=0;j<4;++j)cin>>a[i][j];
 while(f){
  for(f=0,i=1;i<=n;++i)if(!suck[i])for(j=1;j<=n;++j)if(j!=i&&!suck[j]){
   for(k=0;k<4;++k)if(a[i][k]>=a[j][k])break;
   if(k==4){suck[i]=f=1;break;}
  }
 }
 for(i=1;i<=n;++i)if(!suck[i])if(cheapest>a[i][3])cheapest=a[i][3],ans=i;
 cout<<ans<<endl;
 return 0;
}
