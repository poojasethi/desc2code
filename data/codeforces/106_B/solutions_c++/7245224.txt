{
  int n; scanf("%d",&n);
  FOR(i,0,n) 
    scanf("%d %d %d %d\n",&a[i],&b[i],&c[i],&d[i]);
  FOR(i,0,n) 
    FOR(j,0,n)
      if(a[i]<a[j]&&b[i]<b[j]&&c[i]<c[j]&&i!=j) d[i]=1001;
  int e,mn=1001;
  FOR(i,0,n) if(d[i]<mn) mn=d[i] , e=i+1;

}
