#include<cstdio>
int a[105],b[105],c[105],d[105];
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%d%d%d%d",a+i,b+i,c+i,d+i);
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            if(a[i]<a[j]&&b[i]<b[j]&&c[i]<c[j])d[i]=1<<30;
        }
    }
    int tag=1;
    for(i=2;i<=n;i++)
    {
        if(d[i]<d[tag])tag=i;
    }
    printf("%d\n",tag);
    return 0;
}
