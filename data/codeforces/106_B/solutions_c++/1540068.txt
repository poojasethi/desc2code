#include <cstdio>

int main(){
    int s[101],r[101],h[101],c[101];
    int max=3000;
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%d %d %d %d",&s[i],&r[i],&h[i],&c[i]);
    
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            if(s[i]<s[j] && r[i]<r[j] && h[i]<h[j]) c[i]=max;
    int ans;
    for(int i=0;i<n;i++)
        if(c[i]<max){ans =i; max=c[i];}
    printf("%d",ans+1);
    return 0;
}
