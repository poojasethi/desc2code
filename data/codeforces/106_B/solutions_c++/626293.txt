#include<cstdio>
#include<cstring>
using namespace std;

int n,a[111],b[111],c[111],d[111],ret=100000,ans;

int main() {
	scanf("%d",&n);
	for (int i=0; i<n; i++) {
		scanf("%d%d%d%d",&a[i],&b[i],&c[i],&d[i]);
	}
	for (int i=0; i<n; i++)
		for (int j=0; j<n; j++)
			if (a[i]<a[j] && b[i]<b[j] && c[i]<c[j]) d[i]=1000000;
	for (int i=0; i<n; i++)
		if (d[i]<ret) ret=d[i],ans=i;
	printf("%d\n",ans+1);
}
