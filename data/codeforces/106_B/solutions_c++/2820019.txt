#include <stdio.h>
int x[103][103];
int main(){

    int n,i,j,min=1001,index;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d %d %d %d",&x[i][0],&x[i][1],&x[i][2],&x[i][3]);
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            if((x[j][0]<x[i][0])&&(x[j][1]<x[i][1])&&(x[j][2]<x[i][2]))
                  x[j][3]=1000+1;

    for(i=0;i<n;i++)
      if(x[i][3]<min){min=x[i][3];index=i+1;}

    printf("%d\n",index);

}
