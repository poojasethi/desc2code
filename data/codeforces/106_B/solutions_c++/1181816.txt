#include <iostream>
using namespace std;
int i,j,a[1111][4],n,m=10000,mi;
bool t[1111];
int main()
{
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i][0]>>a[i][1]>>a[i][2]>>a[i][3];
		t[i]=true;
	}
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			if ((a[i][0]>a[j][0])&&(a[i][1]>a[j][1])&&(a[i][2]>a[j][2]))
				t[j]=false;
	for(i=0;i<n;i++)
		if ((m>a[i][3])&&(t[i]))
		{
			m=a[i][3];
			mi=i;
		}
	cout<<mi+1;
	return 0;
}
