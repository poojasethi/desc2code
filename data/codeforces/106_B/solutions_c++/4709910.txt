#include<iostream>
using namespace std;
int viz[103];
int main()
{
	int n,i,j,p[103],r[103],h[103],c[103],t,mn;
	cin>>n;
	for(i=1;i<=n;i++)
		{cin>>p[i]>>r[i]>>h[i]>>c[i];
		for(j=1;j<i;j++)
			{if(p[i]>p[j]&&r[i]>r[j]&&h[i]>h[j])viz[j]=1;
			 if(p[i]<p[j]&&r[i]<r[j]&&h[i]<h[j])viz[i]=1;
			}
		}
	mn=1001;
	for(i=1;i<=n;i++)
		if(viz[i]==0&&c[i]<mn){mn=c[i];t=i;}
	cout<<t<<'\n';
	return 0;
}
