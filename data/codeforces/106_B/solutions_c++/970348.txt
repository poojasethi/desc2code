#include <iostream>
#include <cstring>
#include <stdio.h>
using namespace std;

int main()
{int n,s[101],r[101],h[101],c[101];
//freopen("i.txt","r",stdin);
 cin>>n;
 for (int i=1;i<=n;i++) cin>>s[i]>>r[i]>>h[i]>>c[i];

 int i0=0;
 for (int i=1;i<=n;i++)
 {bool f=true;
  for (int j=1;j<=n;j++)
   if ((s[i]<s[j])&&(r[i]<r[j])&&(h[i]<h[j])) {f=false; break;}
  if (c[i]<c[i0]&&f) i0=i;
 }
 cout<<i0;
return 0;
}
