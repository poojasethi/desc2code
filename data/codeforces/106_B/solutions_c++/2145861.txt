#include <cstdio>

int main()
{
	int n, s[128], r[128], h[128], c[128], ok[128], min = -1;
	int i, j;

	scanf (" %d", &n);
	for (i = 0; i < n; ++i)
	{
		scanf (" %d %d %d %d", s+i, r+i, h+i, c+i);
		ok[i] = 1;
	}
	
	for (i = 0; i < n; ++i)
	{
		for (j = 0; j < n && ok[i]; ++j)
		{
			if (s[i] < s[j] && r[i] < r[j] && h[i] < h[j])
				ok[i] = 0;
		}

		if (ok[i] && (min < 0 || c[i] < c[min]))
			min = i;
	}

	printf ("%d\n", min+1);

	return 0;
}
