#include<cstdio>
#define INT_MAX 0x7FFFFFFF
int main()
{
	int n,i,j,mini=0;
	scanf("%d",&n);
	int speed[n],ram[n],hdd[n],cost[n];
	for (i=0;i<n;i++)
		scanf("%d %d %d %d",speed+i,ram+i,hdd+i,cost+i);
	for (i=0;i<n;i++)
		for (j=0;j<n;j++)
			if (speed[i]<speed[j]&&ram[i]<ram[j]&&hdd[i]<hdd[j])
				cost[i]=INT_MAX;
	for (i=0;i<n;i++)
		if (cost[i]<cost[mini])
			mini=i;
	printf("%d",mini+1);
	return 0;
}
