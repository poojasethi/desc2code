#include <stdio.h>
int main (){
int n,s[101],r[101],h[101],c[101],i;
int min,k,j;

scanf("%i",&n);
for (i=1;i<=n;i++)
scanf("%i%i%i%i",&s[i],&r[i],&h[i],&c[i]);


for (i=1;i<=n;i++)
for (j=1;j<=n;j++)
if (i!=j)
if ((s[i]<s[j])&&(r[i]<r[j])&&(h[i]<h[j])) c[i]=1002;

k=-1;
min=1002;
for (i=1;i<=n;i++)
if (c[i]<min) {min=c[i];k=i;}

printf("%i",k);
return 0;
}
