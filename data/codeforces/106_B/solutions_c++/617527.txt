#include<iostream> 
#include<vector>
#include<algorithm>
using namespace std;
int n,a[101],b[101],c[101],d[101],i,j;
vector<pair<int,int> > res;
int main()
{
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>a[i]>>b[i]>>c[i]>>d[i];
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			if(i!=j)
			if(a[i]<a[j] && b[i]<b[j] && c[i]<c[j]) d[i]=-1;
	for(i=1;i<=n;i++)
		if(d[i]!=-1) res.push_back(make_pair(d[i],i));
	sort(res.begin(),res.end());
	cout<<res[0].second<<endl;
}
