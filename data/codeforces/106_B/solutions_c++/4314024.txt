#include <iostream>
#define N 101
using namespace std;
int n,i,j,cost,best,a[N],b[N],c[N],d[N];
int main()
{
    cin>>n;cost=1001;
    for(i=1;i<=n;i++)cin>>a[i]>>b[i]>>c[i]>>d[i];
    for(i=1;i<=n;i++)
    {
        if(d[i]>cost)continue;
        for(j=1;j<=n;j++)
            if(a[i]<a[j]&&b[i]<b[j]&&c[i]<c[j])
                break;
        if(j==n+1){best=i;cost=d[i];}
    }
    cout<<best;
    return 0;
}
