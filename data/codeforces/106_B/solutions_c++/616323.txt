#include<cstdio>

int n;
int lap[101][4];
bool out[101];
int cheap;

int main(){
 int i,j;
 scanf("%d",&n);
 for(i=0;i<n;i++) for(j=0;j<4;j++) scanf("%d",&lap[i][j]);
 for(i=0;i<n;i++){
  for(j=0;j<n;j++){
   if(i==j) continue;
   if(lap[i][0] < lap[j][0] && lap[i][1] < lap[j][1] && lap[i][2] < lap[j][2]) out[i] = true;
  }
 }

 i = 0;
 while(out[i]) i++;
 cheap = i;
 for(;i<n;i++){
  if(!out[i] && lap[i][3] < lap[cheap][3]) cheap = i;
 }
 printf("%d\n",cheap+1);
 return 0;
}
