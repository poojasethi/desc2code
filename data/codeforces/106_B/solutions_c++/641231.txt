#include<stdio.h>
long a[4][102],i,j,k,n,r,q;
int main(){

 scanf("%ld",&n);

 for(i=1;i<=n;i++)
    scanf("%ld %ld %ld %ld",&a[0][i],&a[1][i],&a[2][i],&a[3][i]);

 for(i=1;i<=n;i++){
    if(a[0][i]>0)
    for(j=1;j<=n;j++)
        if(a[0][i]>a[0][j]&&a[1][i]>a[1][j]&&a[2][i]>a[2][j])
            a[0][j]=0;
 }

 q=200000;

 for(i=1;i<=n;i++){
    if(a[0][i]>0&&a[3][i]<q){
        r=i;
        q=a[3][i];
    }
 }

 printf("%ld\n",r);

}
