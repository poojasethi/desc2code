#include <stdio.h>
#include <iostream>
#include <cstring>
#include <vector>

using namespace std;

int a[100005], cnt[100005], next[100005];
bool vis[100005];
vector< pair<int, int> > pairs;

int dfs(int pos){
	if(vis[pos]) return next[pos];
	vis[pos] = true;
	next[pos] = pos;
	return next[pos] = dfs(a[pos]);
}

int main(){
	int n;
	vector<int> p;
	cin >> n;
	for(int i = 0; i < n; i++){
		cin >> a[i + 1];
		cnt[a[i + 1]]++;
	}
	for(int i = 1; i <= n; i++){
		if(cnt[i] == 0){
			pairs.push_back(make_pair(i, dfs(i)));
		}
	}
	for(int i = 1; i <= n; i++){
		if(vis[i] == false){
			dfs(i);
			pairs.push_back(make_pair(i, i));
		}
	}
	int ans = pairs.size();
	if(ans == 1 && pairs[0].first == pairs[0].second)printf("0\n");
	else{
		printf("%d\n", ans);
		for(int i = 0; i < ans; i++){
			printf("%d %d\n", pairs[i].second, pairs[(i + 1) % ans].first);
		}
	}
	return 0;
}
