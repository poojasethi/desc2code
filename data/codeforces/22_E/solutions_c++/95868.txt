#include<stdio.h>
#define N 100001

struct Tp {int x,y;} w[N];

int n,a[N],j,x,b[N],k,f[N];

int main(){
        scanf("%d",&n);
        for (int i=1;i<=n;i++) scanf("%d",&a[i]),b[a[i]]=1;
        for (int i=1;i<=n;i++) if (!b[i]){
                j=i;
                while (!f[j]) f[j]=1,j=a[j];
                w[++k].x=i,w[k].y=j;
        }
        for (int i=1;i<=n;i++) if (!f[i]){
                j=i;
                while (!f[j]) f[j]=1,j=a[j];
                w[++k].x=i,w[k].y=i;
        }
        if (k==1 && w[1].x==w[1].y) puts("0");else{
                printf("%d\n",k);
                w[0]=w[k];
                for (int i=1;i<=k;i++) printf("%d %d\n",w[i-1].y,w[i].x);
        }
        return 0;
}
