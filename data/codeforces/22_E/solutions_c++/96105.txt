#include <iostream>
#include <vector>

using namespace std;

#define pb push_back

int f[ 100000 ];
bool b[ 100000 ];
bool visit[ 100000 ];
vector < int > ini;
vector < int > fin;

int dfs(int v){
	if ( visit[ v ] ) return v;
	visit[ v ] = true;
	return dfs( f[v] );
}

int main(){
	int n;
	
	cin >> n;
	
	for (int i=0; i<n; i++){
		cin >> f[i];
		f[i]--;
		b[ f[i] ] = true;
	}
	
	bool inicio = false;
	for (int i=0; i<n; i++){
		if ( b[i] ) continue;
		
		ini.pb( i );
		fin.pb( dfs( i ) );
		inicio = true;
	}
	
	for (int i=0; i<n; i++){
		if ( visit[i] ) continue;
		
		ini.pb( i );
		fin.pb( dfs(i) );
	}
	
	if ( !inicio && ini.size()==1 ){
		cout << 0 << endl;
		return 0;
	}
	
	cout << ini.size() << endl;
	for (int i=0; i<ini.size(); i++){
		cout << fin[i]+1 << " " << ini[ (i+1)%ini.size() ]+1 << endl;
	}
	
	return 0;
}
