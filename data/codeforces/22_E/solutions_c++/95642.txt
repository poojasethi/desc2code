
// Program  :  Codeforces Round #22 Problem E

#include <algorithm>
#include <cstring>
#include <cstdio>
#define MM(x) memset(x, 0, sizeof(x))

using namespace std;

int ind[100001], v[100001], next[100001], x[100001], y[100001];

int main()
  {
    int n, i, t, tot;
    
    scanf("%d", &n), MM(ind), MM(v);
    for (i = 1; i <= n; i ++)
      scanf("%d", &next[i]), ind[next[i]] ++;
    
    for (tot = 0, i = 1; i <= n; i ++)
      if (! ind[i])
        {
          for (t = i; ! v[t]; t = next[t]) v[t] = 1;
          x[++ tot] = i, y[tot] = t;
        }
    for (i = 1; i <= n; i ++)
      if (! v[i])
        {
          for (t = i; ! v[t]; t = next[t]) v[t] = 1;
          x[++ tot] = t, y[tot] = t;
        }
    
    if (tot == 1 && x[tot] == y[tot])
      printf("0\n");
    else
      {
        printf("%d\n", tot);
        for (i = 1; i < tot; i ++)
          printf("%d %d\n", y[i], x[i+1]);
        printf("%d %d\n", y[tot], x[1]);
      }
    
    return 0;
  }
