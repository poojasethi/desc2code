#include <cstdio>

using namespace std;

int f[100001],g[100001],b[100001],e[100001];
bool d[100001],v[100001],c[100001];

void dfs(int x)
{
  v[x]=true;
  if(!v[f[x]])
    dfs(f[x]);
  if(g[f[x]]==0)
    g[x]=x;
  else
    g[x]=g[f[x]];
}

int main(void)
{
  int n;
  scanf("%d",&n);
  for(int i=1;i<=n;++i)
  {
    scanf("%d",&f[i]);
    d[f[i]]=true;
  }
  for(int i=1;i<=n;++i)
    if(!v[i])
      dfs(i);
  int m=0;
  for(int i=1;i<=n;++i)
  {
    if(!d[i])
    {
      b[++m]=i;
      e[m]=g[i];
      c[g[i]]=true;
    }
  }
  for(int i=1;i<=n;++i)
  {
    if(!c[g[i]])
    {
      b[++m]=g[i];
      e[m]=g[i];
      c[g[i]]=true;
    }
  }
  if((m==1)&&(b[1]==e[1]))
    m=0;
  printf("%d\n",m);
  for(int i=1;i<m;++i)
    printf("%d %d\n",e[i],b[i+1]);
  if(m>0)
    printf("%d %d\n",e[m],b[1]);
  return 0;
}
