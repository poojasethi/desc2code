#include<bits/stdc++.h>
using namespace std;
const int maxn = 100010;
int n,t;
int G[maxn];
int vis[maxn];
int dfs(int now)
{
    vis[now]=1;
    int next=G[now];
    if(vis[next]) return now;
    else return dfs(next);
}
int in[maxn];
int head[maxn],tail[maxn],cur;

int main()
{
    scanf("%d",&n);
    memset(in,0,sizeof(in));
    for(int i=1;i<=n;i++)
    {
       scanf("%d",&t);in[t]++;
       G[i]=t;
    }
    cur=0;memset(vis,0,sizeof(vis));
    int cnt1=0,cnt2=0;
    for(int i=1;i<=n;i++)
        if(!vis[i]&&in[i]==0)
        {
            cnt1++;
            head[cur]=i;
            tail[cur++]=dfs(i);
        }
    for(int i=1;i<=n;i++)
        if(!vis[i])
        {
            cnt2++;
            head[cur]=i;tail[cur++]=i;
            dfs(i);
        }
    if(cnt1==0&&cnt2==1)
    {
        puts("0");
        return 0;
    }
    //for(int i=0;i<cur;i++)
        //cout<<head[i]<<" "<<tail[i]<<endl;
    printf("%d\n",cur);
    for(int i=0;i<cur;i++)
        printf("%d %d\n",tail[i],head[(i+1)%cur]);
    return 0;
}
