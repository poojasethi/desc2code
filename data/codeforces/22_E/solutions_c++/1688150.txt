#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

#define MAXN 100005

typedef pair<int,int> ii;

int adj[MAXN],n,in[MAXN],vis[MAXN],fim[MAXN];
vector<int> vet;
vector<ii> ans;

int dfs(int u) {
	if (vis[u]) return fim[u];
	vis[u] = 1;
	if (!vis[adj[u]]) return fim[u] = dfs(adj[u]);
	else return fim[u] = u;
}

int main() {
	scanf("%d",&n);
	
	memset(in,0,sizeof(in));
		
	for (int i=0; i<n; i++) {
		scanf("%d",&adj[i]);	adj[i]--;	in[adj[i]] = 1;
	}
	
	for (int i=0; i<n; i++)
		if (!in[i]) ans.push_back(ii(i,dfs(i)));	
		
	for (int i=0; i<n; i++)
		if (!vis[i]) ans.push_back(ii(i,dfs(i)));
		
	int t = ans.size();
	if (t == 1) {
		int a,b;
		a = ans[0].first;	b = ans[0].second;
		if (adj[b] == a) printf("0\n");
		else printf("1\n%d %d\n",ans[0].second+1,ans[0].first+1);
	}
	
	else {
		printf("%d\n",t);
		for (int i=0; i<t; i++)
			printf("%d %d\n",ans[i].second+1,ans[(i+1)%t].first+1);
	}
		
	return 0;
}
