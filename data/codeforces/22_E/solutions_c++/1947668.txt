#include <iostream>
#include <stdio.h>
#include <string.h>
#include <vector>
#define pb push_back
using namespace std;
vector <int> up,down;
const int maxn=100005;
int n;
bool in[maxn];
int v[maxn];
int a[maxn];
int dfs(int u,int r)
{
    if(v[u]==r) return u;
    v[u]=r;
    return dfs(a[u],r);
}
int main()
{
   // freopen("in.txt","r",stdin);
    cin>>n;
    int i,j;
    for(i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        in[a[i]]=1;
    }
    int cnt=0;
    for(i=1;i<=n;i++)
    if(!in[i])
    {
        cnt++;
        up.pb(i),down.pb(dfs(i,i));
    }
    for(i=1;i<=n;i++)
    if(v[i]==0)
    {
        up.pb(i),down.pb(i);
        dfs(i,i);
    }
    if(up.size()==1&&cnt==0)
    {
        cout<<"0"<<endl;
        return 0;
    }
    cout<<up.size()<<endl;
    if(up.size())
    printf("%d %d\n",down[up.size()-1],up[0]);
    for(i=0;i<up.size()-1;i++)
    {
        printf("%d %d\n",down[i],up[i+1]);
    }
    return 0;
}
