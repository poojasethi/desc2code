#include<stdio.h>
#include<string.h>
#include<iostream>
#include<vector>
#include<stack>
#include<algorithm>
using namespace std;

#define MAX 100007
#define pb push_back

long N;
long Next[MAX+7];
bool In[MAX+7];
bool Visit[MAX+7];
long nG;
long St[MAX+7];
long Ed[MAX+7];

void Dfs( long u )
{
    Visit[u] = true;
    if( Visit[Next[u]] ) Ed[nG] = Next[u];
    else Dfs( Next[u] );
}

int main( void )
{
    long i,j,u,v,Icase,k = 0;

    //freopen("text1.txt","r",stdin );

    scanf("%ld",&N );
    for( i=1;i<=N;i++){
        scanf("%ld",&v );
        Next[i] = v;
        In[v] = true;
    }
    bool Find = false;
    for( i=1;i<=N;i++){
        if( In[i] ) continue;
        Find = true;
        St[nG] = i;
        Dfs( i );
        nG++;
    }
    for( i=1;i<=N;i++){
        if( Visit[i] ) continue;
        St[nG] = i;
        Dfs( i );
        nG++;
    }
    if( nG==1 && !Find ){
        printf("0\n");
        return 0;
    }

    cout<<nG<<endl;
    for( i=0;i<nG;i++ ){
        printf("%ld %ld\n",Ed[i],St[(i+1)%nG] );
    }

    return 0;
}
