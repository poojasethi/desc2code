#include <stdio.h>
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int n, c;
int f[100001], t[100001], v[100001];
int head[100001], tail[100001];

int main() {
    scanf("%d", &n);
    rep(i, n) scanf("%d", f+i+1);
    rep(i, n) t[f[i+1]] = 1;
    c = 0;
    rep(i, n) if(!t[i+1]) {
        int p = i+1;
        while(!v[p]) { v[p]=1; p=f[p]; }
        head[c] = i+1; tail[c] = p; c++;
    }
    rep(i, n) if(!v[i+1]) {
        int p = i+1;
        while(!v[p]) { v[p]=1; p=f[p]; }
        head[c] = i+1; tail[c] = p; c++;
    }
    if(c==1 && head[0]==tail[0]) {
        printf("%d\n", 0);
    }
    else {
        printf("%d\n", c);
        rep(i, c) {
            printf("%d %d\n", tail[i], head[(i+1)%c]);
        }
    }
    return 0;
}
