#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

bool e[101000],vis[101000];
int s[101000],t[101000],f[101000],ans,n;

void dfs(int x)
{
    vis[x]=true;
    if (vis[f[x]]) t[ans]=x;else dfs(f[x]);
}

int main()
{
 //   freopen("22E.txt","r",stdin);
    cin.sync_with_stdio(false);
    cout.sync_with_stdio(false);

    cin>>n;
    memset(e,true,sizeof(e));
    for (int i=1;i<=n;i++) cin>>f[i],e[f[i]]=false;
    for (int i=1;i<=n;i++) if (e[i] && !vis[i]) ans++,s[ans]=i,dfs(i);
    int tmp=ans;
    for (int i=1;i<=n;i++)if (!vis[i]) ans++,s[ans]=i,dfs(i);
    if (tmp==0 && ans==1) cout<<0<<endl;
    else
    {
        cout<<ans<<endl;
        for (int i=1;i<=ans;i++) cout<<t[i]<<" "<<s[i%ans+1]<<endl;
    }

    return 0;
}
