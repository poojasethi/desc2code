#include <bits/stdc++.h>
using namespace std;

#define MAX 100010
#define ii pair<int,int>
#define vi vector<int>
#define vii vector<ii>
#define pb push_back
#define mp make_pair
#define fi first
#define se second

int n;
int a[MAX], b[MAX], vs[MAX];
vii res;

void dfs(int u, int r){
	vs[u] = 1;
	if (vs[a[u]]){
		res.pb(mp(r, u));
		return;
	}
	dfs(a[u], r);
}

void dfs2(int u, int r){
	if (vs[u]) return;
	vs[u] = 1;
	dfs2(a[u], r);
}

int main(){

	// freopen("test.in", "rt", stdin);
	cin >> n;
	for (int i = 1; i <= n; i++){
		cin >> a[i];
		b[a[i]] = 1;
	}
	for (int i = 1; i <= n; i++)
		if (!b[i])
			dfs(i, i);
	for (int i = 1; i <= n; i++)
		if (!vs[i]){
			res.pb(mp(i,i));
			dfs2(i, i);
		}
	if (res.size() == 1 && res[0].fi == res[0].se){
		cout << 0;
		return 0;
	}
	n = res.size();
	cout << n << endl;
	for (int i = 0; i < n; i++)
		cout << res[i].se << " " << res[(i+1) % n].fi << endl;
	return 0;
}
