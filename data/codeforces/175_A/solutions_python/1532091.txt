S = raw_input().strip()
ans = -1
L = len(S)
for i in xrange(1, L-1):
    for j in xrange(i+1, L):
        p = S[:i], S[i:j], S[j:]
        f = True
        for x in p:
            if len(x) > 1 and x[0] == '0':
                f = False
            if int(x) > 10 ** 6:
                f = False
        if f:
            ans = max(ans, sum([int(x) for x in p]))
print ans
