s = raw_input()
l = len(s)
x = [-1]
for i in xrange(1, l):
    for j in xrange(i+1, l):
        a, b, c = s[:i], s[i:j], s[j:]
        if (a[0]=='0' and len(a)>1) or (b[0]=='0' and len(b)>1) or (c[0]=='0' and len(c)>1): x.append(-1)
        else:
            a, b, c = int(a), int(b), int(c)
            if a<=1000000 and b<=1000000 and c<=1000000: x.append(a+b+c)
print max(x)
