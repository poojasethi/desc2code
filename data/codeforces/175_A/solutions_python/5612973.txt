
def main():
    maxx = -1
    s = raw_input()
    for i in range(1, len(s)):
        for j in range(i + 1, len(s)):
            a = int(s[:i])
            b = int(s[i:j])
            c = int(s[j:])
            if a <= 1000000 and b <= 1000000 and c <= 1000000 and \
               (s[:i] == "0" or s[0] > '0') and (s[i:j] == "0" or s[i] > '0') and (s[j:] == "0" or s[j] > '0'):
                maxx = max(maxx, a + b + c)
    print maxx

if __name__ == '__main__':
    main()
