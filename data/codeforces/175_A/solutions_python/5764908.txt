s = raw_input()
n = len(s)
otvet = -1

def stroka(str):
    return (len(str) == 1 or str[0] != '0') and long(str) <= 1e6

for i in range(1, n):
    x = s[:i]
    for j in range(i+1, n):
        y = s[i:j]
        z = s[j:]
        if stroka(x) and stroka(y) and stroka(z):
            otvet = max(otvet, sum(map(long, [x,y,z])))
      
print otvet
