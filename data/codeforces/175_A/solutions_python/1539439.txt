import sys

s = sys.stdin.readline().strip()
n = len(s)
sm = -1
for i in range(1,n):
	for j in range(i+1,n):
		if (s[0] != '0' or i == 1) and (s[i] != '0' or j == i+1) and (s[j] != '0' or j == n-1):
			a, b, c = int(s[:i]), int(s[i:j]), int(s[j:])
			if a <= 1000000 and b <= 1000000 and c <= 1000000 and a+b+c > sm:
				sm = a+b+c
print sm
