data = raw_input()
ret = -1
i = 1

while i < len(data):
  j = i+1
  while j < len(data) and i < j:
    l = data[:i]
    m = data[i:j]
    r = data[j:]
    #print l, m, r

    for k in (l, m, r):
      if (len(k) > 1 and k.startswith('0')) or int(k) > 1000000:
        break
    else:
      ret = max(ret, int(l)+int(m)+int(r))

    j += 1
  i += 1

print ret
