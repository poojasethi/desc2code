s = raw_input()
res = -1
for i in range(1, len(s)):
	for j in range(i+1, len(s)):
		r1 = int(s[0:i])
		r2 = int(s[i:j])
		r3 = int(s[j:])
		if ((r1 > 0 and s[0] == '0' or r1 == 0 and i-1 > 0) or \
				(r2 > 0 and s[i] == '0' or r2 == 0 and j-1 > i) or \
				(r3 > 0 and s[j] == '0' or r3 == 0 and len(s)-1 > j)): continue;
		if (r1 <= 1000000 and r2 <= 1000000 and r3 <= 1000000 and r1 + r2 + r3 > res):
			res = r1 + r2 + r3;
print(res);
