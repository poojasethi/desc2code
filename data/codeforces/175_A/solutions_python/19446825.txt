s=raw_input()
ans=-1
for i in range(1,len(s)-1):
	for j in range(i,len(s)-1):
	    a=s[0:i]
	    b=s[i:j+1]
	    c=s[j+1:len(s)]
	    #print(a,b,c)
	    if((len(a)==1 or a[0]!='0') and (len(b)==1 or b[0]!='0') and (len(c)==1 or c[0]!='0')):
		    if(int(a)<=1e6 and int(b)<=1e6 and int(c)<=1e6):
			    ans=max(ans,int(a)+int(b)+int(c))
print(ans)
