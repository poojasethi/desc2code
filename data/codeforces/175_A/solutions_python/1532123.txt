def vavid(s):
    return len(s)>=1 and(len(s)==1 or s[0]!='0')  and int(s)<=1000000

s=raw_input()
ans=-1
n=len(s)
for x in xrange(1,n):
    for y in xrange(1,n):
        if x<y:
            s1=s[0:x]
            s2=s[x:y]
            s3=s[y:]
            if vavid(s1) and vavid(s2) and vavid(s3):
                ans=max(ans,int(s1)+int(s2)+int(s3))
print ans
