def valid(s):
	if s[0] != '0':
		return 1
	return len(s) == 1

ip = raw_input().strip()
l = len(ip)
ans = -1

for i in range(0,l):
	for j in range(i+1,l-1):
		sa = ip[0:i+1]
		sb = ip[i+1:j+1]
		sc = ip[j+1:l]
		if valid(sa) and valid(sb) and valid(sc):
			if max(int(sa),int(sb),int(sc)) > 1000000:
				continue
			ans = max(ans,int(sa)+int(sb)+int(sc))
print ans
