def conv(x):
  if len(x)>1 and '0'==x[0]: return -1
  return int(x)

s=raw_input()
x=-1
for i in xrange(1,len(s)):
  for j in xrange(i+1,len(s)):
    a=conv(s[0:i])
    b=conv(s[i:j])
    c=conv(s[j:])
    if a>=0 and a<=1000000 and b>=0 and b<=1000000 and c>=0 and c<=1000000:
      x=max(x,a+b+c)
print x
