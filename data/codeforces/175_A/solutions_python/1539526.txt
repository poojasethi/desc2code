s = raw_input()

res = -1

for i in range(len(s)-2):
    for j in range(i+1, len(s)-1):
        x = int(s[0:i+1])
        y = int(s[i+1:j+1])
        z = int(s[j+1:len(s)])
        
        if x<=1e6 and y<=1e6 and z<=1e6 and s[0:i+1]==str(x) and s[i+1:j+1]==str(y) and s[j+1:len(s)]==str(z) :
            res = max(res, x+y+z)

print res
