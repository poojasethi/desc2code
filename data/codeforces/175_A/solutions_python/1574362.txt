#!/usr/bin/python -O
import sys
import collections

s = sys.stdin.readline().strip()
m = -1
for i in range(1, len(s) - 1):
	for j in range(i + 1, len(s)):
		def value(s):
			ret = int(s)
			if (ret > 1000000 or (len(s) > 1 and s[0] == '0')):
				ret = -1000000000000
			return ret
		m = max(m, value(s[:i]) + value(s[i:j]) + value(s[j:]))
print m
