def valid(s):
    return s == '0' or s[0] != '0' and int(s) <= 1000000
s = raw_input()
res = -1
n = len(s)
if n < 3:
        print -1
        exit(0)
for j in range(1, n - 1):
    for k in range(j + 1, n):
        a = s[0 : j]
        b = s[j : k]
        c = s[k : n]
        ret = int(a) + int(b) + int(c)
        if valid(a) and valid(b) and valid(c):
            res = max(res, ret)
print res
