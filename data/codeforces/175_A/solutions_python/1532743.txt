s = raw_input()
l = len(s)
ans = -1

def valid(s):
    if s!="0" and s[0]=='0': return False
    if (int(s)>1000000): return False
    return True
for i in range(1, l-1):
    for j in range(1, (l-i)):
        s1= s[:i]
        s2 = s[i:i+j]
        s3 = s[i+j:]
        if (valid(s1) and valid(s2) and valid(s3)):
            ans = max(ans, int(s1)+int(s2)+int(s3))
            

print ans
