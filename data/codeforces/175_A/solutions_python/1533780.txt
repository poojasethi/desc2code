s = raw_input()
n = len(s)

sum = -1

for a in xrange(1, n):
    for b in xrange(1, n):
        if a + b >= n:
            continue
        x, y, z = s[0:a], s[a:a+b], s[a+b:]
        valid = True
        for test in [x, y, z]:
            if (len(test) != 1 and test[0] == '0') or int(test) > 1000000:
                valid = False
                break
        if not valid:
            continue
        sum = max(sum, int(x) + int (y) + int (z))
print sum
