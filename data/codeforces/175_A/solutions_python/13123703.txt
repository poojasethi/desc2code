from sys import stdin
found = 0
ma = 0
a = stdin.readline().strip()
l = len(a)

def valid(s):
 if s[0]=='0' and len(s)>1:
  return 0
 if int(s) > 10**6:
  return 0
 return 1
for i in xrange(1,l):
 for j in xrange(i+1,l):
  x = a[:i];y=a[i:j];z=a[j:]
  if valid(x) and valid(y) and valid(z):
   found = 1;cur = int(x) + int(y) + int(z)
   if cur > ma:
    ma = cur
if found==0:
 ma = -1
