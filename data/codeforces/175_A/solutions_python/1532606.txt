s = raw_input()
ans = -1
M = 10**6
def valid(x):
	return (len(x) == 1 or x[0] != '0') and long(x) <= M

for i in xrange(1, len(s)):
	a = (s[0:i])
	for j in xrange(i+1, len(s)):
		b = (s[i:j])
		c = (s[j:])
		# print a, b, c
		if valid(a) and valid(b) and valid(c):
			ans = max(ans, sum(map(long, [a,b,c])))
print ans
