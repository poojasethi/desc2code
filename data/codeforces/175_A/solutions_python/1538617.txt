def isvalid(str):
  return (len(str) == 1 or str[0] != '0') and long(str) <= 1e6

s = raw_input()
sz = len(s)
res = -1
for x in range(1, sz):
  a = s[:x]
  for y in range(x+1, sz):
    b = s[x:y]
    c = s[y:]
    if isvalid(a) and isvalid(b) and isvalid(c):
      res = max(res, sum(map(long, [a,b,c])))
print res
