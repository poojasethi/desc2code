s = raw_input()
n = len(s)
B = 10 ** 6
print max([-1] + [int(s[ : i]) + int(s[i : j]) + int(s[j : ]) for i in range(1, n) for j in range(i + 1, n) if (i == 1 or s[0] != '0') and (j - i == 1 or s[i] != '0') and (n - j == 1 or s[j] != '0') and int(s[ : i]) <= B and int(s[i : j]) <= B and int(s[j : ]) <= B])
