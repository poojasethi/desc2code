s = raw_input()
n = len(s)
ans = -1
for i in xrange(1, n):
    for j in xrange(i+1, n):
        a = s[:i]
        b = s[i:j]
        c = s[j:n]
        if a != "0" and a[0] == '0': continue
        if b != "0" and b[0] == '0': continue
        if c != "0" and c[0] == '0': continue
        a, b, c = int(a), int(b), int(c)
        if a > 1000000 or b > 1000000 or c > 1000000: continue
        ans = max(ans, a+b+c)
print ans
