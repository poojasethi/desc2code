s=raw_input()
n=len(s)
C=lambda m: m[0] == '0' and len(m) > 1
M=-1
for i in range(1,n):
    for j in range(i+1,n):
        a = s[:i]
        b = s[i:j]
        c = s[j:]
        if C(a) or C(b) or C(c):
            continue
        a = int(a)
        b = int(b)
        c = int(c)

        if max(a,b,c) > 10**6:
            continue
        S=a + b + c
        if M<S:
            M=S
print M
