#! -*- coding: utf-8 -*-
import sys
def main():
  line = sys.stdin.readline().strip()

  # if line[0] == '0':
  #   print -1;
  #   return
  n = len(line)
  res = -1
  for i in range(1,n):
    if i > 1 and line[0] == '0': continue
    for j in range(i+1, n):
      if j-i > 1 and line[i] == '0': continue
      if n-j > 1 and line[j] == '0': continue
      # print [line[0:i], line[i:j], line[j:]]
      a,b,c = map(int, [line[0:i], line[i:j], line[j:]])
      if max(a, b, c) <= 1000000:
        res = max(res, a+b+c)
  print res

if __name__=='__main__':
  main()
