def f(s):
      if len(s) == 1: return True
      return not(s[0] == '0')

s = raw_input()
ans = -1
for i in range(1, len(s) - 1):
      for j in range(i + 1, len(s)):
            if f(s[:i]) and f(s[i:j]) and f(s[j:]) and int(s[:i]) <= 10 ** 6 and int(s[i:j]) <= 10 ** 6 and int(s[j:]) <= 10 ** 6:
                  ans = max(ans, int(s[:i]) + int(s[i:j]) + int(s[j:]))
print ans
