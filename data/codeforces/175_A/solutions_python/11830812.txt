line = raw_input()
n = len(line)
mx = -1
bound = 10**6
for i in xrange(1, n-1):
	sa = line[:i]
	if i>1 and sa[0]=='0':
		continue
	ia = int(sa)
	if ia > bound:
		continue
	for j in xrange(i+1, n):
		sb = line[i:j]
		if j-i>1 and sb[0]=='0':
			continue
		ib = int(sb)
		if ib > bound:
			continue
		sc = line[j:]
		if n-j>1 and sc[0]=='0':
			continue
		ic = int(sc)
		if ic > bound:
			continue
		# print ia, ib, ic
		tmp = ia + ib + ic
		if tmp > mx:
			mx = tmp
print mx
