#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int a, b, c;
char w[35];
int valid(int L, int R) {
	if(L==R && w[L]=='0') return 0;
	if(w[L]=='0' || R-L+1>=8) return -1;
	int n=0;
	for(int i=L;i<=R;i++) {
		n=n*10+w[i]-'0';
		if(n>1000000) return -1;
	}
	return n;
}
int main(void) {
	scanf("%s", w);
	int m=strlen(w), ans=-1;
	for(int i=1;i<m;i++) {
		for(int j=i+1;j<m;j++) {
			a = valid(0, i-1);
			b = valid(i, j-1);
			c = valid(j, m-1);
			if(a>=0 && b>=0 && c>=0) ans=max(ans, a+b+c);
		}
	}
	printf("%d\n", ans);
	return 0;
}
