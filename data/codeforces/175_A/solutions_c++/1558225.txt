#include <iostream>
#include <cstring>
using namespace std;
char s[31];
bool getnum(int beg,int end,int& num)
{
    num=0;
    if(s[beg]=='0'&&end-beg>1) return false;
    for(int i=beg;i<end;++i)
    {
        num=num*10+s[i]-'0';
        if(num>1000000) return false;
    }
    return true;
}
int main()
{
    cin>>s;
    int l=strlen(s),i,j,k,num[3],p=-1;
    for(i=1;i<l-1;++i)
    {
        if(!getnum(0,i,num[0])) continue;
        for(j=i+1;j<l;++j)
        {
            if(!getnum(i,j,num[1])||!getnum(j,l,num[2])) continue;
            p=max(p,num[0]+num[1]+num[2]);
        }
    }
    cout<<p<<endl;
    return 0;
}
