#include <iostream>
#include<string.h>
using namespace std;
char s[35];
int fun(int l,int r)
{
    int i;
    long long int c=0;
    if(r-l+1>7||l>r)
    return -10000000;
    if(s[l]=='0'&&r-l>0)
    return -10000000;
        c=0;
        for(i=l;i<=r;i++)
        {
            c=c*10+s[i]-'0';
        }
        if(c>1000000)
        return -10000000;
        else
        return c;

}
int main()
{
    int n,i,j,ans=-1;
    cin>>s;
    n=strlen(s);
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            ans=max(ans,(fun(0,i)+fun(i+1,j)+fun(j+1,n-1)));
        }
    }
    cout<<ans;
    return 0;
}
