#include <stdio.h>
#include <cstring>
#include <algorithm>
using namespace std;

char s[100];
int n, ans;

int getsum(int a, int b) {
	if(b > a && s[a] == '0') return -1;
	int ans = 0;
	for(int i=a; i<=b; ++i) {
		ans = ans * 10 + s[i] - '0';
		if(ans > 1e6) return -1;
	}
	return ans;
}

int main() {
	while(gets(s)) {
		n = strlen(s);
		ans = -1;

		for(int a=0; a<n; ++a) {
			for(int b=a+1; b<n-1; ++b) {
				int a1 = getsum(0, a), a2 = getsum(a + 1, b), a3 = getsum(b + 1, n - 1);
				if(a1 == -1 || a2 == -1 || a3 == -1) continue;
				int tmp = a1 + a2 + a3;
				ans = max(ans, tmp);
			}
		}
		printf("%d\n", ans);
	}
	return 0;
}
