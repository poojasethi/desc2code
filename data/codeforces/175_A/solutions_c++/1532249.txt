#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int MAXL=30;
const int MAXP=1000000;
const int INF=MAXP*5;
char s[MAXL+10];
int L;
int convert(char *s,const int &n) {
	if (*s=='0' && n>1) return -INF;
	int ret=0;
	for (int i=0;i<n && ret<=MAXP;i++)
		ret=ret*10+(*(s+i))-'0';
	if (ret>MAXP) ret=-INF;
	return ret;
}
int main() {
	while (gets(s)) {
		L=strlen(s);
		int ans=-INF;
		for (int i=1;i<L;i++)
			for (int j=1;i+j<L;j++)
				ans=max(ans,convert(s,i)+convert(s+i,j)+convert(s+i+j,L-i-j));
		if (ans<0) ans=-1;
		printf("%d\n",ans);
	}
	return 0;
}
