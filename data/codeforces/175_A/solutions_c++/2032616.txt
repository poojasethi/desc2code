#include <cstdio>
#include <cstring>
#include <cstdlib>
using namespace std;

char s[33], ch;

int gao(int x, int y) {
	if (s[x] == '0' && x + 1 < y || x + 7 < y)
		return -1;
	ch = s[y], s[y] = 0;
	int c = atoi(s + x);
	s[y] = ch;
	return c > 1000000 ? -1 : c;
}

int main() {
	int l = strlen(gets(s)), ans = -1;
	for (int i = 1; i < l; i++)
		for (int j = i + 1; j < l; j++) {
			int x = gao(0, i), y = gao(i, j), z = gao(j, l);
			if (~x && ~y && ~z && x + y + z > ans)
				ans = x + y + z;
		}
	printf("%d\n", ans);
}
