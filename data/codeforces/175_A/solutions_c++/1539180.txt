#include<cstdio>
#include<cstring>

char s[33];

int ans,len,num,now;

bool getnum(int &res,int l,int r)
{
	res=0;
	if (r-l+1>1 && s[l]=='0') return false;
	for (int i=l;i<=r;i++) 
	{
		res=res*10+s[i]-'0';
		if (res>1000000) return false;
	}
	return true;
}

int main()
{
	scanf("%s",s);
	len=strlen(s);
	ans=-1;
	for (int i=0;i<len;i++)
		for (int j=i+1;j<len-1;j++)
		{
			now=0;
			if (!getnum(num,0,i)) continue;
			now+=num;
			if (!getnum(num,i+1,j)) continue;
			now+=num;
			if (!getnum(num,j+1,len-1)) continue;
			now+=num;
			if (now>ans) ans=now;
		}
	printf("%d\n",ans);
	return 0;
}
