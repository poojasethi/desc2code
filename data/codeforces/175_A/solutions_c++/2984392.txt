typedef long long int LL;

using namespace std;

int mx=-1;

void go(char* c,int l,int id,int sm){
	if(id==3){
		if(l==0) mx=max(mx,sm);
		return;
	}	
	if(c[0]=='0'){
		go(c+1,l-1,id+1,sm);
		return;
	}
	int x=0;
	for(int i=0; i<l; i++){
		x=x*10+c[i]-'0';
		if(x>1000000)return;
		go(c+i+1, l-i-1, id+1, sm+x);
	}
} 

int main(){
	char c[100];
	scanf("%s",c);
	int l = strlen(c);
	go(c, l, 0, 0);
	printf("%d\n",mx);	

	return 0;
}
