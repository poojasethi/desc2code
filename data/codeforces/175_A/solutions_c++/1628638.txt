#include <iostream>
#include <cstring>
using namespace std;
typedef long long ll;
const int INF=1000*1000*100;
char a[33];
ll n,ans=-1;
ll check(int s,int e)
{
	if(e-s>7)
		return -1*INF;
	if(a[s]=='0'&&e-s>1)
		return -1*INF;
	ll ans=0;
	for(int i=s;i<e;i++)
	{
		ans*=10;
		ans+=ll(a[i]-'0');
	}
	if(ans>1000000)
		return -1*INF;
	return ans;
}
int main()
{
	cin >> a;
	n=strlen(a);
	for(int i=1;i<n;i++)
		for(int j=1;j<i;j++)
			ans=max(ans,check(0,j)+check(j,i)+check(i,n));
	cout << ans << endl;
	return 0;
}
