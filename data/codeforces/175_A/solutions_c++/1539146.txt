#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;

char s[40];

bool check(int l,int r,long long &tmp)
{
    if(s[l]=='0' and l!=r) return false;
    if(r-l>7) return false;
    tmp=0;
    for(int i=l;i<=r;i++)
        tmp=tmp*10+s[i]-'0';
    if(tmp<=1000000) return true;
    else return false;
}
int main()
{
    gets(s);
    int len=strlen(s);
    long long ans=-1;
    for(int i=0;i<len-2;i++)
        for(int j=i+1;j<len-1;j++)
        {
            long long a,b,c;
            if(check(0,i,a)&&check(i+1,j,b)&&check(j+1,len-1,c))
                ans=max(ans,a+b+c);
        }
    cout<<ans<<endl;
    return 0;
}
