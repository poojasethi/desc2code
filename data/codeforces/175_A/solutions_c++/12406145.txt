#include<bits/stdc++.h>
using namespace std;

string s;
int n;
int mx(int x,int m,int Sum)
{
	if(x>=n)
		return -1;
	int d=0;
	int mmx=-1;
	if(m==3)
	{
		if(s[x]=='0' && x!=n-1)
			return -1;
		for(int i=x;i<n;i++)
		{
			d*=10;
			d+=s[i]-'0';
			if(d>1e6)
				return -1;
		}
		return d+Sum;
	}
		
	for(int i=x;i<n;i++)
	{
		d*=10;
		d+=s[i]-'0';
		
		if(d>1e6)
			break;
		mmx = max(mmx,mx(i+1,m+1,Sum+d));
		if(i==x && s[i]=='0')
			break;
	}
	return mmx;
}

int main()
{
	cin>>s;
	n=s.length();
	cout<<mx(0,1,0);
	return 0;
}
