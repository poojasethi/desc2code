#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[33];
int i,j,k,n,ans;
int count(int l,int r)
{
    int ret=0;
    if(l<r&&s[l]=='0')return -1e8;
    if(r-l+1>7)return -1e8;
    while(l<=r)ret=ret*10+s[l++]-'0';
    if(ret>1e6)return -1e8;
    return ret;
}
int main()
{
    while(scanf("%s",s)!=-1)
    {
        ans=-1;
        n=strlen(s);
        for(i=0;i<n;++i)
            for(j=i+1;j<n-1;++j)
                ans=max(ans,count(0,i)+count(i+1,j)+count(j+1,n-1));
        printf("%d\n",ans);
    }
    return 0;
}
