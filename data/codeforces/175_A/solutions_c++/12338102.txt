#include<iostream>
#include<cstdio>
using namespace std;

int getmax(char * P, int depth=0, int sum=0){
    if(depth==3)return *P?-1:sum;
    int ret = -1,X;
    for(int i=1; i<=7 && P[i-1];++i)
    {
            if(i>1 && *P == '0')break;
            char t = P[i];
            P[i]='\0';
            sscanf(P,"%d",&X);
            P[i]=t;
            if(X>1e6)break;
            ret = max(ret, getmax(P+i,depth+1,sum+X));
    }
    return ret;
}

int main()
{
    char number[35];
    cin>>number;
    cout<<getmax(number)<<endl;
    return 0;
}
