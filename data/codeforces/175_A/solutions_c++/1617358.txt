#include <iostream>
#include <string>
using namespace std;
string s;
int n,i,j,k,x[333],ma=-1;
int main()
{
	cin>>s; n=s.size();
	for(i=0;i<n;i++)
		x[i]=char(s[i])-'0';
	for(j=1;j<n;j++)
		for(k=j+1;k<n;k++)
		{
			if (!x[0]&&(j>1)) continue;
			if (!x[j]&&(k>j+1)) continue;
			if (!x[k]&&(n>k+1)) continue;
			if ((j>7)||(k-j>7)||(n-k>7)) continue;
			int p1=0,p2=0,p3=0;
			for(int z=0;z<j;z++) p1=p1*10+x[z];
			for(int z=j;z<k;z++) p2=p2*10+x[z];
			for(int z=k;z<n;z++) p3=p3*10+x[z];
			if ((p1>1000000)||(p2>1000000)||(p3>1000000)) continue;
			if (p1+p2+p3>ma) ma=p1+p2+p3;
		}
		cout<<ma<<endl;
	return 0;
}
