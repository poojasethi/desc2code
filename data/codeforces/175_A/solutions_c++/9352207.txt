#include <iostream>
#include <string>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
string str;
int C = -1;
int F(int l, int r){
    int S = 0;
    if (r > l && str[l] == '0')
        return -1e8;
    if (r - l + 1 > 7)
        return -1e8;
    for (; l <= r; l++)
        S = S * 10 + str[l] - '0';
    if (S > 1000000)
        return -1e8;
    return S;
}
int main(){
    cin >> str;
    for (int i = 0; i < str.size() ; i++)
    for (int j = i + 1; j < str.size() - 1; j++)
        C = max(C, F(0, i) + F(i + 1, j) + F(j+1, str.size() - 1));
    
    printf("%d", C);
    //system("pause");

}
