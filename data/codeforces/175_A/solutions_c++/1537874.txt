#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

char s[50];

int f[50][4],n,temp;

int main()
{
    scanf("%s",s);
    n=strlen(s);
    memset(f,-1,sizeof f);
    f[0][0]=0;
    for (int i=0;i<n;i++)
    for (int o=0;o<3;o++)
    if (f[i][o]>=0)
    {
        temp=0;
        if (s[i]=='0')  f[i+1][o+1]=max(f[i+1][o+1],f[i][o]);
        else
        for (int j=i;j<=min(n-1,i+7);j++)
        {
            temp=temp*10+s[j]-48;
            if (temp<=1000000)  f[j+1][o+1]=max(f[j+1][o+1],f[i][o]+temp);
        }
    }
    cout<<f[n][3]<<endl;
    return 0;
}
