#include<string>
#include<cstdlib>
#include<iostream>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

bool check(const string &s){
	int n=s.length();
	if(n==1) return true;
	if(s[0]=='0' || n>9) return false;
	return atoi(s.c_str())<=1000000;
}

int main(){
	string s; cin>>s;
	int n=s.length();

	int ans=-1;
	for(int i=1;i<n;i++) for(int j=i+1;j<n;j++) {
		string s1=s.substr(0,i);
		string s2=s.substr(i,j-i);
		string s3=s.substr(j);
		if(check(s1) && check(s2) && check(s3)){
			ans=max(ans,atoi(s1.c_str())+atoi(s2.c_str())+atoi(s3.c_str()));
		}
	}
	cout<<ans<<endl;

	return 0;
}
