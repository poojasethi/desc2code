#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

string s;

int gao(int i,int j){
  int x=0;
  if (i>j) return -3000000;
  if (j>i && s[i]=='0') return -3000000;
  if (j-i>7) return -3000000;
  for (int k=i;k<=j;k++) x=x*10+s[k]-'0';
  if (x>1000000) return -3000000;
  return x;
}
    
int main(){
  cin>>s;
  int tmp;
  int ans=-1;
  int len=(int)s.length()-1;
  for (int i=0;i<len;i++)
    for (int j=i+1;j<len;j++){
      tmp=gao(0,i)+gao(i+1,j)+gao(j+1,len);
      ans=max(ans,tmp);          
    }
  cout<<ans;
  return 0;    
}
