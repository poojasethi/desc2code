#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <string>

using namespace std;

char tmp[100];
string s;

int get(int i, int j)
{
	string x = s.substr(i, j - i);
	if ((x.size() > 1 && x[0] == '0') || x.size() > 7)
		return -10000000;
	int ans;
	sscanf(x.c_str(), "%d", &ans);
	if (ans > 1000000)
		return -10000000;
	return ans;
}

inline int max(int a, int b)
{
	return a > b ? a : b;
}

int main()
{
	scanf("%s", tmp);
	s = string(tmp);
	int ans = -1;
	for (int i = 1; i < s.size(); i++)
		for (int j = i + 1; j < s.size(); j++)
			ans = max(ans, get(0, i) + get(i, j) + get(j, s.size()));
	printf("%d", ans);
	return 0;
}
