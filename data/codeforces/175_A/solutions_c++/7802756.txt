#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int score(string s)
{int sum=0;

    if(s.length()>7 || (s.length()>1&&s[0]=='0'))
    return -1;
    for(int i=0;i<s.length();i++)
    {sum*=10;
    sum+=s[i]-'0';

    }
    return (sum>1000000)? -1 : sum;
}
int main()
{string s;
cin>>s;
int res=-1;
int len=s.length();
for(int i=1;i<=min(len-2,7);i++)
{ int v1=score(s.substr(0,i));
if(v1==-1)
continue;
for(int j=1;j<=min(len-i-1,7);j++)
{int v2=score(s.substr(i,j));
int v3=score(s.substr(i+j));
    if(v2>=0&&v3>=0)
    res=max(res,v1+v2+v3);

}

}
cout<<res;

    return 0;
}
