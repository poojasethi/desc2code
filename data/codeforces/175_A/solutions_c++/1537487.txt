#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <string>

using namespace std;

string s;

int getans(string s) {
	if (s.size()>8) return -10000000;
	int ans=0;
	if (s[0]=='0' && (int)s.size()>1) return -10000000;
	for (int i=0;i<(int)s.size();++i)
		ans=ans*10+s[i]-48;
	if (ans<=1000000) return ans;
	else return -10000000;
}

int main() {
	//freopen("a.in","r",stdin);
	cin >> s;
	int l=s.size();
	int ans=-1;
	for (int i=1;i<min(l,8);++i)
		for (int j=i+1;j<min(i+8,l);++j){
			ans=max(ans,getans(s.substr(0,i))+getans(s.substr(i,j-i))+getans(s.substr(j,l-j)));
		}

	printf("%d\n",ans);
	return 0;
}
