#include <iostream>

using namespace std;

int ch (string s)
{
    int x=0;
    if (s[0]=='0' && s.size()>1 || s.size()>6 && s!="1000000") return -100000000;
    for (int i=0; i<s.size(); i++)
    x=x*10+(s[i]-'0');
    return x;
}

int main()
{string st;
cin >> st;
int l=st.size();
int max=-1;
for (int i=1; i<l-1; i++)
for (int j=i+1; j<l; j++)
{
    int a1, a2, a3;
    a1=ch(st.substr(0,i));
    a2=ch(st.substr(i,j-i));
    a3=ch(st.substr(j,l-j));
    if (a1+a2+a3>max) max=a1+a2+a3;
}

cout << max << endl;


return 0;
}
