#include<iostream>
#include<algorithm>
#include<vector>
#include<math.h>
#include<sstream>
using namespace std;
string s;
int mynum(int x,int y){
    int temp;
    istringstream t(s.substr(x,y-x));
    t>>temp;
    if(temp>1000000 || (y-x!=1 && s[x]=='0') || x==y) return -1;
    return temp;
}
int main(){
    int max=-1,x,y,z;
    cin>>s;
    for(int i=0;i<s.length();i++)
        for(int j=i+1;j<s.length();j++){
            x=mynum(0,i);
            y=mynum(i,j);
            z=mynum(j,s.length());
            if(~x && ~y && ~z && x+y+z>max) max=x+y+z;
        }
    cout<<max<<endl;
    return 0;
}
