#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const int lim=1000000;
int i,j,k,n,m,an;
char p[100];
inline int ju(int l,int r) {
	if (l+7<=r) return -1;
	int i,s=0;
	if (l<r&&p[l]=='0') return -1;
	For(i,l,r) s=s*10+p[i]-48;
	if (s>lim) return -1;
	return s;
}
int main() {
	scanf("%s",p+1);
	n=strlen(p+1);
	an=-1;
	For(i,1,n)For(j,i+1,n-1) {
		int A=ju(1,i),B=ju(i+1,j),C=ju(j+1,n);
		if (A<0||B<0||C<0) continue;
		an=max(an,A+B+C);
	}
	printf("%d\n",an);
	return 0;
}
