#include<cstdio>
#include<cstring>
char s[105];
int main()
{
	scanf("%s",s); int n=strlen(s),ans=-1;
	for (int i=1;i<n;i++)
		for (int j=i+1;j<n;j++){
			if (i>7 || j-i>7 || n-j>7) continue;
			if (i-1>0 && s[0]=='0' || j-1>i && s[i]=='0' || n-1>j && s[j]=='0') continue;
			int x=0,y=0,z=0;
			for (int k=0;k<i;k++) x=x*10+s[k]-48;
			for (int k=i;k<j;k++) y=y*10+s[k]-48;
			for (int k=j;k<n;k++) z=z*10+s[k]-48;			
			if (x<=1000000 && y<=1000000 && z<=1000000 && x+y+z>ans) ans=x+y+z;
			}
	printf("%d\n",ans); return 0;
}
