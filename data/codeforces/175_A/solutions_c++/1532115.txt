#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;

const int N=50,inf=99999999;
char s[N];
int ans=-1;

int w(int l,int r)
  {
    if (l>r || s[l]=='0' && l!=r) return -inf;
    int ans=0;
    for (int i=l;i<=r;++i)
      {
        ans=ans*10+s[i]-'0';
        if (ans>1000000) return -inf;
      }
    return ans;
  }
int main()
  {
    scanf("%s",s);
    int l=strlen(s);
    for (int i=0;i<l;++i)
      for (int j=i+1;j<l;++j)
        ans=max(ans,w(0,i)+w(i+1,j)+w(j+1,l-1));
    printf("%d\n",ans);
  }
