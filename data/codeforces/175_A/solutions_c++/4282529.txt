#include <iostream>
#include <sstream>

using namespace std;

int convert(string s) {
  if(s.size() > 7) return -10000000;
  if(s.size() > 1 && s[0] == '0') return -10000000;
  int x;
  stringstream ss;
  ss << s;
  ss >> x;
  if(x > 1000000) return -10000000;
  return x;
}

int main() {

  int res = -1;
  string s, n1, n2, n3;

  cin >> s;

  for(int i = 1; i < (int)s.size(); ++i) {
    for(int j = 1; i + j < (int)s.size(); ++j) {
      n1 = s.substr(0, i);
      n2 = s.substr(i, j);
      n3 = s.substr(j + i);
      res = max(res, convert(n1) + convert(n2) + convert(n3));
    }
  }
  cout << res << endl;

  return 0;
}
