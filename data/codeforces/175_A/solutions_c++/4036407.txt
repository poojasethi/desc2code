#include <cstdio>
#include <cstring>

char s[1000];
int p1,p2;
inline int max(const int &a,const int &b)
{
    return a>b?a:b;
}
int getint(int l,int r)
{
    int ans=0;
    if(s[l]=='0' && l!=r) return -100000000;
    if(l>r) return -100000000;
    for(;l<=r;l++)
    {
        ans=ans*10+(s[l]-'0');
        if(ans>1000000) return -100000000;
    }
    return ans;
}
int main()
{
    gets(s);
    int len=strlen(s);
    int Max=-1;
    for(p1=0;p1<len-2;p1++)
        for(p2=p1+1;p2<len-1;p2++)
            Max=max(Max,getint(0,p1)+getint(p1+1 ,p2)+getint(p2+1,len-1));
    printf("%d\n",Max);
    return 0;
}
