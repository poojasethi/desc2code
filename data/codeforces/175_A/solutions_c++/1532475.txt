#include <iostream>
#include <string>

using namespace std;


string data;

int number(int l, int r)
{
if(data[l]=='0' && l<r)return -1;
if(r-l+1 > 7) return -1;
int ans = 0;
while(l<=r){
	ans*=10;
	ans+=data[l++]-'0';
}
if(ans > 1000000)return -1;
return ans;
}

int main()
{
	
	cin>>data;
	int ans = -1;
	int l = data.length();
	for(int i = 1; i <= l-2 ; i++){
		for(int j = i+1; j<= l-1; j++){
			int s1 = number(0, i-1);
			int s2 = number(i, j-1);
			int s3 = number(j, l-1);
			if(s1>=0 && s2>=0 && s3>=0 && s1+s2+s3>ans){ans = s1+s2+s3;
				//cout<< i << " " << j << " " << s1 << " " << s2 <<" " <<s3 <<endl; 
			}
		}
	}

	cout << ans <<endl;
	return 0;
}
