#include <iostream>
#include <string>

using namespace std;

const long long INF = 1000000000;

long long stoint(string s, int l, int r)
{
	long long a = 0;
	if(r-l > 7)
		return -INF;
	for(int i = l; i < r; ++ i)
	{
		a+= s[i]-'0';
		a*= 10;
	}
	a/=10;
	if((s[l] == '0' && r-l > 1) || a > 1000000)
		return -INF;
	return a;
}

int main()
{
	string s;
	cin >> s;
	long long int Max = -1;
	for(int i = 2; i < s.size(); ++i)
		for(int j = 1; j < i; ++j)
		{
			long long int a = stoint(s, 0, j);
			long long int b = stoint(s, j, i);
			long long int c = stoint(s, i, s.size());
			Max = max(a+b+c, Max);
		}
	cout << Max;
}
