#include <cstdio>
#include <cstring>

char s[12][40];
int a[12], n, m, ans;
void dfs(int t)
{
    for (int i = 0; i < m; ++i)
        if (a[i] < 0) return ;
    if (t >= n) {
        for (int i = 0; i < m; ++i) 
            if (a[i] > 0) return;
        ++ans;
        return;
    }
    for (char good = '0'; good <= '1'; ++good){
        for (int i = 0; i < m; ++i)
            if (s[i][t] == good) --a[i];
        dfs(t + 1);
        for (int i = 0; i < m; ++i)
            if (s[i][t] == good) ++a[i];
    }
}

int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 0; i < m; ++i)
        scanf("%s%d", s[i], &a[i]);
    ans = 0;
    dfs(0);
    printf("%d\n", ans);
}
