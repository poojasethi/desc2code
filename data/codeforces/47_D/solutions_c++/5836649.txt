#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<map>
#include<vector>
#include<queue>
#include<iostream>
#include<math.h>
using namespace std;
int n,m;
int ans=0;
char s[12][40];
int reply[12];
int cnt[12];
void dfs(int now)
{

    for(int i=0;i<m;i++)
        if(cnt[i]+n-now<reply[i]||cnt[i]>reply[i])return;
    if(now==n)
    {
        for(int i=0;i<m;i++)
            if(cnt[i]!=reply[i])return;
        ans++;
        return;
    }
    int ttts[10];
    for(int i=0;i<m;i++)
        ttts[i]=cnt[i];
    char tt='1';
    for(int i=0;i<m;i++)
        if(s[i][now]==tt)
        ++cnt[i];
    dfs(now+1);
    for(int i=0;i<m;i++)
        cnt[i]=ttts[i];
    for(int i=0;i<m;i++)
        if(s[i][now]!=tt)
        cnt[i]++;
    dfs(now+1);
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;i++)
        scanf("%s%d",s[i],&reply[i]);
    dfs(0);
    cout<<ans<<endl;
}
