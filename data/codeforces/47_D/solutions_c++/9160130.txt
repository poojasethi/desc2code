#include<stdio.h>

#include<map>

#define ll __int64
#define mod 1000000007
using namespace std;

char s[50][50];
int cnt[50],n;
int get(int d)
{
	int i,ans=0;
	if(d == n)
	{
		for(i=0;i<n;i++)
			if(cnt[i]) return 0;
		return 1;
	}
	char tt;
	for(tt = '0'; tt <='1' ;tt++)
	{
		int f=1;
		for(i=0;i<n;i++)
			if(tt == s[i][d])
			{
				cnt[i]--;
				if(cnt[i] < 0) f=0;
			}
		if(f) ans+=get(d+1);
		for(i=0;i<n;i++)
			if(tt == s[i][d])
				cnt[i]++;
	}
	return ans;
}

int main()
{
    int i,j,m;
	scanf("%d %d",&n,&m);
	for(i=0;i<m;i++)
		scanf("%s %d",&s[i],&cnt[i]);
	printf("%d\n",get(0));
    return 0;
}
