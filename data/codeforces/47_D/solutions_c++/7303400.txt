#include<iostream>
#include<cstdio>
#include<cmath>
#include<string>
using namespace std;
char s[12][40];
int c[12],n,m,ans;
void dfs(int t)
{
	int i,j;
	for(i=0;i<m;i++)
		if(c[i]<0)return ;
	if(t>=n)
	{
		for(i=0;i<m;i++)
			if(c[i]>0) return ;
		ans++;
		return ;
	}
	for(i=0;i<=1;i++)
	{
		for(j=0;j<m;j++)
			if(s[j][t]==(i+'0'))c[j]--;
		dfs(t+1);
		for(j=0;j<m;j++)
			if(s[j][t]==(i+'0'))c[j]++;
	}
}
int main()
{
	int i;
	std::ios::sync_with_stdio(false);
	cin>>n>>m;
	for(i=0;i<m;i++)cin>>s[i]>>c[i];
	ans=0;
	dfs(0);
	cout<<ans<<endl;
	return 0;
}
