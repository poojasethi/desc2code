n = int(raw_input())

recF = 0
minX = 99999
maxX = -1 
minY = 99999
maxY = -1
for i in range(n):
	A = [int(x) for x in raw_input().split()]
	recF += abs(A[0]-A[2])*abs(A[1]-A[3])
	minX = min(minX, A[0], A[2])
	maxX = max(maxX, A[0], A[2])

	minY = min(minY, A[1], A[3])
	maxY = max(maxY, A[1], A[3])
	#print minX,minY,maxX,maxY
if maxX-minX == maxY-minY and (maxX-minX)*(maxY-minY) == recF:
	print "YES"
else:
	print "NO"
