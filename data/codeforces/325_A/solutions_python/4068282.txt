n = int(input())
lx1 = []
ly1 = []
lx2 = []
ly2 = []
area = 0
for i in range(n):
    x1,y1,x2,y2 = map(int,raw_input().split())
    lx1.append(x1)
    ly1.append(y1)
    lx2.append(x2)
    ly2.append(y2)
    area += abs((x1-x2)*(y1-y2))
leng = abs(max(lx2)-min(lx1))
brea = abs(max(ly2)-min(ly1))
if abs(leng*brea)==area and leng == brea:
    print 'YES'
else:
    print 'NO'
