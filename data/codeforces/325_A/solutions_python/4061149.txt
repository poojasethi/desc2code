n = int(raw_input())

x0 = 1000000000
y0 = 1000000000
xt = -1
yt = -1

s = 0

for i in range(n):
	x1, y1, x2, y2 = [int(x) for x in raw_input().split()]
	s += (x2 - x1) * (y2 - y1)
	x0 = min(x0, x1)
	y0 = min(y0, y1)
	xt = max(xt, x2)
	yt = max(yt, y2)

print "YES" if yt - y0 == xt - x0 and (xt - x0) * (yt - y0) == s else "NO"
