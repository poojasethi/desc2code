def area(rec):
  return (rec[3] - rec[1]) * (rec[2] - rec[0]) 

n = int(raw_input())
recs = []
for i in range(n):
  recs.append(map(int, raw_input().split()))
  
sq = recs[0][:]
 
for i in range(1,n):
  rec = recs[i] 
  sq[0] = min(sq[0], rec[0])
  sq[1] = min(sq[1], rec[1])
  sq[2] = max(sq[2], rec[2])
  sq[3] = max(sq[3], rec[3])

if area(sq) == sum(map(area,recs)) and (sq[3] - sq[1] == sq[2] - sq[0]):
  print 'YES'
else:
  print 'NO'
