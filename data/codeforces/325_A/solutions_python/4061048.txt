n = int(raw_input())

s = 0
mx, Mx, my, My = 100000, -100000, 100000, -100000

for i in xrange(n):
    a = map(int, raw_input().split(' '))
    s += (a[2]-a[0])*(a[3]-a[1])
    mx = min(mx, a[0])
    Mx = max(Mx, a[2])
    my = min(my, a[1])
    My = max(My, a[3])

if s == (Mx-mx)*(My-my) and Mx-mx == My-my:
    print "YES"
else:
    print "NO"
