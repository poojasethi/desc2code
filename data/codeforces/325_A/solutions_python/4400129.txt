import sys
n = int(sys.stdin.readline())

area = 0
minx = 33000
miny = 33000
maxx = -1
maxy = -1
#data = []
for _ in xrange(n):
    x1, y1, x2, y2 = map(int, sys.stdin.readline().split())
    if x1 < minx:
        minx = x1
    if y1 < miny:
        miny = y1
    if x2 > maxx:
        maxx = x2
    if y2 > maxy:
        maxy = y2
    area += (x2-x1)*(y2-y1)

if (maxx-minx) == (maxy-miny) and area == (maxx-minx)*(maxy-miny):
    print "YES"
else:
    print "NO"
