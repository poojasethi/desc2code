#! /usr/bin/python

import sys

__author__ = 'Muntasir Khan'


class Rect:
  def __init__(self, x1, y1, x2, y2):
    self.x1 = x1
    self.x2 = x2
    self.y1 = y1
    self.y2 = y2
    self.area = abs(self.x1 - self.x2) * abs(self.y1 - self.y2)
  
  def is_square(self):
    return abs(self.x1 - self.x2) == abs(self.y1 - self.y2)


def bounding_rect(rects):
  min_x, max_x = rects[0].x1, rects[0].x2
  min_y, max_y = rects[0].y1, rects[0].y2
  for rect in rects:
    min_x = min(rect.x1, min_x)
    max_x = max(rect.x2, max_x)
    min_y = min(rect.y1, min_y)
    max_y = max(rect.y2, max_y)
  return Rect(min_x, min_y, max_x, max_y)


def forms_square(rects):
  bound = bounding_rect(rects)
  total_area = sum(rect.area for rect in rects)
  return bound.area == total_area and bound.is_square()


def main():
  n = int(sys.stdin.readline())
  rects = []
  for i in xrange(n):
    x1, y1, x2, y2 = map(int, sys.stdin.readline().split())
    rects.append(Rect(x1, y1, x2, y2))
  if forms_square(rects):
    print "YES"
  else:
    print "NO"

if __name__ == '__main__':
  main()
