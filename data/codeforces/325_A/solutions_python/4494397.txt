from math import sqrt
n=input()
lst=[]

for i in range(n):
    lst.append(map(int,raw_input().strip().split()))

sum=0
for i in range(n):
    sum+=(lst[i][2]-lst[i][0])*(lst[i][3]-lst[i][1])

x=sqrt(sum)

x1=[]
x2=[]
y1=[]
y2=[]
for i in range(n):
    x1.append(lst[i][0])
    y1.append(lst[i][1])
    x2.append(lst[i][2])
    y2.append(lst[i][3])
length=abs(max(x2)-min(x1))
width=abs(max(y2)-min(y1))   
    


if abs(length*width)==sum and length==width:
    print "YES"
else:
    print "NO"
