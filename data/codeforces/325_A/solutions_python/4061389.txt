def ri(): return map(int, raw_input().split())

def main():
    n = input()

    xk1, yk1, xk2, yk2 = float("inf"), float("inf"), float("-inf"), float("-inf") 
    total = 0
    for _ in range(n):
        x1,y1,x2,y2 = ri()
        xk1 = min(xk1, x1)
        yk1 = min(yk1, y1)
        xk2 = max(xk2, x2)
        yk2 = max(yk2, y2)
        total += (x2-x1) * (y2-y1)

    print "YES" if ((yk2-yk1) == (xk2-xk1) and total == (yk2-yk1)*(xk2-xk1)) else "NO"

main()
