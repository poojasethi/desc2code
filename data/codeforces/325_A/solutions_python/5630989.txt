n = input()

w = 0
x = 0
y = 0
z = 0
s = []
for i in range(n):
    a,b,c,d = map(int, raw_input().strip().split())
    s.append(abs(c-a)*abs(d-b))
    if i == 0:
        w = a
        x = b
        y = c
        z = d
    else:
        w = min(w, a)
        x = min(x, b)
        y = max(y, c)
        z = max(z, d)
if y-w == z-x and (y-w)**2 == sum(s):
    print 'YES'
else:
    print 'NO'
