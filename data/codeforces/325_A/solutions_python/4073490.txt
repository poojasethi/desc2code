read_ints = lambda: map(int, raw_input().split())

n= input()
xmin, ymin= 10**8, 10**8
xmax, ymax= -1, -1
s= 0
for i in range(n):
    x1,y1,x2,y2= read_ints()
    s+= (x2-x1)*(y2-y1)
    xmin= min(xmin, min(x1,x2))
    xmax= max(xmax, max(x1,x2))
    ymin= min(ymin, min(y1,y2))
    ymax= max(ymax, max(y1,y2))
    
if (xmax-xmin)*(ymax-ymin)==s and (xmax-xmin)==(ymax-ymin) :
    print "YES"
else :
    print "NO"
