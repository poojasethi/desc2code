n = int(raw_input())
a = [map(int, raw_input().split()) for i in xrange(n)]
sx = min([t[0] for t in a])
sy = min([t[1] for t in a])
tx = max([t[2] for t in a])
ty = max([t[3] for t in a])
s = 0
for t in a:
    s += (t[2] - t[0]) * (t[3] - t[1])
print "YES" if tx - sx == ty - sy and s == (tx - sx) * (ty - sy) else "NO"
