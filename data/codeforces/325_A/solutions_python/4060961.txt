if __name__=="__main__":
    n = int(raw_input())
    minx=miny=40000
    maxx=maxy=-1
    s=0
    for i in range(n):
        x1,y1,x2,y2 = raw_input().split()
        x1,y1,x2,y2 = int(x1),int(y1),int(x2),int(y2)
        minx=min(minx,x1,x2)
        maxx=max(maxx,x1,x2)
        miny=min(miny,y1,y2)
        maxy=max(maxy,y1,y2)
        s+=abs(x1-x2)*abs(y1-y2)

    if (maxx-minx)*(maxy-miny)==s and (maxx-minx)==(maxy-miny):
        print "YES"
    else:
        print "NO"
