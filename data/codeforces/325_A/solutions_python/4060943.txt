n=int(raw_input())
x1,x2,y1,y2=31400+1,-1,31400,-1

A=0
for i in range(n):
    X1,Y1,X2,Y2=[int(e) for e in raw_input().split()]
    x1=min(x1,X1,X2)
    x2=max(x2,X1,X2)
    y1=min(y1,Y1,Y2)
    y2=max(y2,Y1,Y2)
    A+=abs(X1-X2)*abs(Y1-Y2)

if A == abs(x1-x2)*abs(y1-y2) and abs(x1-x2)==abs(y1-y2):
    print "YES"
else:
    print "NO"
