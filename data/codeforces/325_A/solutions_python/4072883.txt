l = d = 10**9
r = u = -10**6
N = int( raw_input() ) ;
s = 0 
for _ in range( N ) :
    aa,bb,cc,dd = map( lambda x:int(x) , raw_input().split(' ') )
    l = min(l, aa)
    r = max(r, cc)
    d = min(d, bb)
    u = max(u, dd)
    s += (cc - aa) * (dd - bb)
r -= l
u -= d
if (s==r*u) and (r==u) :
    print "YES"
else :
    print "NO"
