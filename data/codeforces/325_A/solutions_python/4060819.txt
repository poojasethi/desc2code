
n = int(raw_input())
res = False
xmax,xmin = -1, 999999999
ymax,ymin = -1, 999999999
s = []
for i in range(n):
    x1,y1,x2,y2 = map(int,raw_input().split())
    xmax, xmin = max(xmax, x2), min(xmin, x1)
    ymax, ymin = max(ymax, y2), min(ymin, y1)
    s.append((x2-x1)*(y2-y1))
if (xmax - xmin) == (ymax - ymin):
    if (xmax - xmin)*(xmax - xmin) == sum(s):
        res = True

print 'YES' if res else 'NO'
