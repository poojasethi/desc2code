#include<bits/stdc++.h>
using namespace std;

#define mx 31400
#define mn 0

int main()
{
    int n; cin >> n;
    int xa = mx, ya = mx, xb = mn, yb = mn, s = 0;
    for(int i = 0; i < n; ++i)
    {
        int a, b, c, d; cin >> a >> b >> c >> d;
        s += (c-a)*(d-b);
        xa = min(xa, a);    ya = min(ya, b);
        xb = max(xb, c);    yb = max(yb, d);
    }
    if (s == (xb-xa)*(yb-ya) && (xb-xa) == (yb-ya)) cout << "YES";
    else cout << "NO";
}
