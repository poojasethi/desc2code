#include <iostream>
#include <cstdio>
using namespace std;

int n, a, b, c, d, s;
int minx, miny, maxx, maxy;

int main()
{
    scanf("%d",&n);
    minx=miny=0x3f3f3f3f;
    while(n--)
    {
        scanf("%d%d%d%d",&a,&b,&c,&d);
        s+=(c-a)*(d-b);
        minx=min(minx,a), maxx=max(maxx,c);
        miny=min(miny,b), maxy=max(maxy,d);
    }
    if(maxy-miny==maxx-minx&&s==(maxy-miny)*(maxx-minx))puts("YES");
    else puts("NO");
    return 0;
}
