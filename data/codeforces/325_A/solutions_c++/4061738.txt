#include <stdio.h>
#include <iostream>
#include <fstream>
using namespace std;

int n, a0 = 31401, a1, b0 = 31401, b1, c0, c1, d0, d1;
long long s, t;
                   
int main()
{
	cin >> n;
	for(int i=0;i<n;i++)
	{
		cin >> c0 >> d0 >> c1 >> d1;
		s += (long long)(c1-c0) * (d1-d0);
		if(c0 < a0) a0 = c0;
		if(c1 > a1) a1 = c1;
		if(d0 < b0) b0 = d0;
		if(d1 > b1) b1 = d1;
	}
	t = (b1-b0) * (a1-a0);
 	if(s == t && (b1-b0) == (a1-a0) ) cout << "YES\n"; else cout << "NO\n";
}
