#include <cstdio>
#include <algorithm>
using namespace std;

const int MAXN = 310000;

int main() {
  int n; scanf("%d", &n);
  int X1=MAXN,X2=0,Y1=MAXN,Y2=0;
  int s = 0, S = 0;
  for (int q = 0; q < n; q++) {
    int x1,x2,y1,y2;
    scanf ("%d %d %d %d", &x1, &y1, &x2, &y2);
    X1 = min(x1,X1);
    Y1 = min(y1,Y1);
    X2 = max(x2,X2);
    Y2 = max(y2,Y2);
    s += (x2-x1)*(y2-y1);
  }
  S = (X2-X1)*(Y2-Y1);
  if (s == S && (X2-X1) == (Y2-Y1)) printf ("YES\n");
  else printf ("NO\n");
  return 0;
}
