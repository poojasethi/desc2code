#include<stdio.h>
int n,x,y,z,u;
int maxx,minx,maxy,miny;
int main()
{
    scanf("%d",&n);
    maxx=maxy=0;
    minx=miny=31400;
    int sum=0;
    for(int i=0;i<n;i++)
    {
     scanf("%d %d %d %d",&x,&y,&z,&u);
     if(z>maxx) maxx=z;
     if(u>maxy) maxy=u;
     if(x<minx) minx=x;
     if(y<miny) miny=y;
     sum+=(z-x)*(u-y);
    }
    if(sum==(maxx-minx)*(maxy-miny)&&(maxx-minx)==(maxy-miny)) printf("YES\n");else printf("NO\n");
}
