#include<iostream>
#include<stdio.h>
#include<cstring>
#include<algorithm>
using namespace std;
int n,i,j,xa[10],ya[10],xb[10],yb[10],mxa=32000,mxb=0,mya=32000,myb=0;
long long s=0;
int main(){
	cin>>n;	
	for(i=1;i<=n;i++){
		cin>>xa[i]>>ya[i]>>xb[i]>>yb[i];
		s+=(xb[i]-xa[i])*(yb[i]-ya[i]);
		mxa=min(mxa,xa[i]);
		mxb=max(mxb,xb[i]);
		mya=min(mya,ya[i]);
		myb=max(myb,yb[i]);
	}
	if (mxb-mxa==myb-mya and (mxb-mxa)*(myb-mya)==s)
		cout<<"YES";
	else 
		cout<<"NO";
	return 0;
}
