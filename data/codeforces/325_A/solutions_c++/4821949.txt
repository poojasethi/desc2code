#include <algorithm>
#include <stdio.h>
#define  inf       1 << 30
using    namespace std;

int n,xa=inf,xb,ya=inf,yb,ans;

int main()
{
  scanf("%d",&n);
  for(int i=0,x1,x2,y1,y2 ; i<n ; i++)
  {
    scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
    ans+=(x2-x1)*(y2-y1);
    xa=min(xa,x1);
    xb=max(xb,x2);
    ya=min(ya,y1);
    yb=max(yb,y2);
  }
  if(yb-ya==xb-xa && ans==(yb-ya)*(yb-ya)) printf("YES\n");
  else                                     printf("NO\n");
  return 0;
}
