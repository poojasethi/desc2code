#include<iostream>
using namespace std;
int main(){
    int n,mx1,mx2,my1,my2,x1,x2,y1,y2,s=0;
    cin>>n>>mx1>>my1>>mx2>>my2;
    s+=(mx2-mx1)*(my2-my1);
    for(int i=1;i<n;i++){
        cin>>x1>>y1>>x2>>y2;
        s+=(x2-x1)*(y2-y1);
        mx1=min(mx1,x1);
        my1=min(my1,y1);
        mx2=max(mx2,x2);
        my2=max(my2,y2);
    }
    if(mx2-mx1==my2-my1&&(mx2-mx1)*(my2-my1)==s)
        cout<<"YES";
    else
        cout<<"NO";
}
