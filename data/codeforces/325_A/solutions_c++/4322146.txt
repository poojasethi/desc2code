#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,s,x,y,xx,yy,mix,miy,maX,may;
int main() {
	scanf("%d",&n);
	mix=miy=31400;
	For(i,1,n) {
		scanf("%d%d%d%d",&x,&y,&xx,&yy);
		mix=min(mix,x);
		miy=min(miy,y);
		maX=max(maX,xx);
		may=max(may,yy);
		k+=(xx-x)*(yy-y);
	}
	x=maX-mix,y=may-miy;
	if (x==y&&x*x==k) printf("YES\n");
	else printf("NO\n");
	return 0;
}
