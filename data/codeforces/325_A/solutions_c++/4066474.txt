#include <cstdio>

typedef long long ll;

ll x1,x2,y1,y2;
ll s;
int n;
ll q1,w1,q2,w2;

int main()
{
	scanf("%d",&n);
	q1=32000;
	w1=32000;
	q2=-1;
	w2=-1;
	s=0;
	while (n--)
	{
		scanf("%I64d%I64d%I64d%I64d",&x1,&y1,&x2,&y2);
		if (x1<q1)
			q1=x1;
		if (y1<=w1)
			w1=y1;
		if (x2>=q2)
			q2=x2;
		if (y2>=w2)
			w2=y2;
		s+=(x2-x1)*(y2-y1);
	}
	ll temp;
	temp=(q2-q1)*(w2-w1);
	if (temp==s && (q2-q1==w2-w1))
		puts("YES");
	else
		puts("NO");
	return 0;
}
