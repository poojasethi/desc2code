#include<iostream>
using namespace std;

int area(int a, int b, int c, int d)
{
	int ar = (c-a) * (d-b);
	if(ar<0) return -ar;
	return ar;
}

int main()
{
	int n;
	int a, b, c, d;
	int total=0;
	int mx=100000, my=100000, max=0, may=0;
	cin>>n;
	while(n--) {
		cin>>a>>b>>c>>d;
		if(a<mx) mx=a;
		if(b<my) my=b;
		if(c>max) max=c;
		if(d>may) may=d;
		total+=area(a, b, c, d);
	}
	if(total == area(mx, my, max, may) && (max-mx) == (may-my))
		cout<<"YES\n";
	else cout<<"NO\n";
	return 0;
}
