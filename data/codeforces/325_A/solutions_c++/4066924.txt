#include<iostream>
using namespace std;

int main()
{
    int x1,y1,x2,y2;
    int n,area=0,a=314000,b=314000,c=0,d=0;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>x1>>y1>>x2>>y2;
        area+=(x2-x1)*(y2-y1);
        a=min(a,x1);b=min(b,y1);
        c=max(c,x2);d=max(d,y2);
    }
    if(d-b!=c-a) cout<<"NO";
    else if((d-b)*(c-a)!=area) cout<<"NO";
    else cout<<"YES";
    return 0;
}
