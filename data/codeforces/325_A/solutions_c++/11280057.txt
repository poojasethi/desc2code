#include <iostream>
using namespace std;
int main()
{   long int q,w,e,r,t,y,x=100000,c=0,v=100000,b=0,a=0;
    cin>>q;
    for(w=0;w<q;w++){
    cin>>e>>r>>t>>y;
    if(x>e)x=e;
    if(v>r)v=r;
    if(c<t)c=t;
    if(b<y)b=y;
    a+=(t-e)*(y-r);}
    if((c-x==b-v)&&(a==(c-x)*(b-v)))
    cout<<"YES";
    else cout<<"NO";
    return 0;
}
