#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
int n,lx,ly,rx,ry,x1,x2,y1,y2;
long long s;
int main() {
	scanf("%d",&n);
	lx=ly=1<<30,rx=ry=0;
	rep(i,0,n) {
		scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
		lx=min(lx,x1);rx=max(rx,x2);
		ly=min(ly,y1);ry=max(ry,y2);
		s+=(x2-x1)*(y2-y1);
	}
	ry-=ly,rx-=lx;
	if (rx==ry&&1ll*rx*ry==s) puts("YES"); else puts("NO");
}
