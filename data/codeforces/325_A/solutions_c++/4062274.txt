#include<stdio.h>
#include<iostream>
using namespace std;
long long a=1000000,s=1000000,d[6][4],f,g,h,j,k,l,i,n,m;
main(){
cin>>n;
for(i=0;i<n;i++){
cin>>d[i][0]>>d[i][1]>>d[i][2]>>d[i][3];
k=k+(d[i][2]-d[i][0])*(d[i][3]-d[i][1]);
if(d[i][0]<a) a=d[i][0];
if(d[i][1]<s) s=d[i][1];
if(d[i][2]>f) f=d[i][2];
if(d[i][3]>g) g=d[i][3];
}
if((f-a)*(g-s)!=k || (f-a)!=(g-s)) cout<<"NO"; else cout<<"YES";
//system("pause");
}
