#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int n,area = 0,minx = 31400,miny = 31400,maxx=0,maxy=0;cin >> n;
    int a,b,c,d;
    for(int i = 0;i < n;i++){
        cin >> a;cin >> b;cin >> c;cin >> d;
        area += (c-a) * (d-b);
        if(a < minx) minx= a; if(b < miny) miny= b;
        if(c > maxx) maxx = c; if(d > maxy) maxy = d;
    }
    cout << ( maxx - minx == maxy - miny && area == (maxx-minx)*(maxx-minx) ? "YES" : "NO") << endl;
    return 0;
}
