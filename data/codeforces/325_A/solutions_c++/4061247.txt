#include<cstdio>

int main()
{
    int n,x1,y1,x2,y2,xmn=31400,ymn=31400,xmx=0,ymx=0,i,s=0;

    for(scanf("%d",&n),i=0;i<n;i++)
    {
        scanf("%d%d%d%d",&x1,&y1,&x2,&y2);

        xmn=x1<xmn?x1:xmn;
        ymn=y1<ymn?y1:ymn;
        xmx=x2>xmx?x2:xmx;
        ymx=y2>ymx?y2:ymx;
        s+=(x2-x1)*(y2-y1);
    }

    puts(xmx-xmn==ymx-ymn&&s==(xmx-xmn)*(ymx-ymn)?"YES":"NO");

    return 0;
}
