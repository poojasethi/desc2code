#include <iostream>
using namespace std;
int c1,c2,s1,s2,n,x1=1e9,x2,y1=1e9,y2;
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        int a,b,c,d;
        cin>>a>>b>>c>>d;
        s1+=(c-a)*(d-b);
        x1=min(x1,a);
        y1=min(y1,b);
        x2=max(x2,c);
        y2=max(y2,d);
    }
    c1=x2-x1;
    c2=y2-y1;
    s2=c1*c2;
    if(s1==s2 && c1==c2) cout<<"YES";
    else cout<<"NO";
}
