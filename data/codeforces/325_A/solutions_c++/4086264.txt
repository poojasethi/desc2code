#include<stdio.h>
#define INF 100000000
int max(int a,int b){return a>b?a:b;}
int min(int a,int b){return a<b?a:b;}
int main(){
    int n,i;
    scanf("%d",&n);
    long long x1=INF,x2=0,y1=INF,y2=0;
    long long s=0;
    for(i=0;i<n;i++){
        int a,b,c,d;
        scanf("%d %d %d %d",&a,&b,&c,&d);
        x1=min(x1,a);y1=min(y1,b);
        x2=max(x2,c);y2=max(y2,d);
        s+=(c-a)*(d-b);
    }
    printf("%s\n",s==(x2-x1)*(y2-y1)&&x2-x1==y2-y1?"YES":"NO");
    return 0;
}
