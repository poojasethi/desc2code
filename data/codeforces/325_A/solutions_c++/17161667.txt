#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
const int MAX=31500;

int main()
{
    int n;
    ll siz=0;
    int l,r,u,d,L=MAX,R=0,U=0,D=MAX;
    scanf("%d",&n);
    while(n--)
    {
        scanf("%d%d%d%d",&l,&d,&r,&u);
        siz+=(r-l)*(u-d);
        L=min(L,l);R=max(R,r);
        U=max(U,u);D=min(D,d);
    }
    ll SIZ=(R-L)*(U-D);
    if(siz==SIZ&&R-L==U-D) printf("YES\n");
    else printf("NO\n");
    return 0;
}
