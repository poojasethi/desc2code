#include <cstdio>
#include <algorithm>
using namespace std;
int n;
int area;
int minx, miny, maxx, maxy;
int main(){
	while(scanf("%d",&n)!=EOF){
		minx=miny=31400;
		maxx=maxy=0;
		int x1,y1,x2,y2;
		while(n--){
			scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
			minx=min(x1,minx);
			miny=min(y1,miny);
			maxx=max(x2,maxx);
			maxy=max(y2,maxy);
			area+=(x2-x1)*(y2-y1);
		}
		if(maxx-minx==maxy-miny && (maxx-minx)*(maxy-miny)==area) puts("YES");
		else puts("NO");
	}
}
