#include<stdio.h>
#define N 999999
int main(){
    int n;
    while(~scanf("%d",&n)){
        int i,x1,x2,y1,y2,mx=N,Mx=-N,my=N,My=-N,R=0;
        for(i=0;i<n;i++){
            scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
            R+=(x2-x1)*(y2-y1);
            if(Mx<x2)Mx=x2;if(My<y2)My=y2;if(mx>x1)mx=x1;if(my>y1)my=y1;
        }
        if((Mx-mx)*(My-my)==R && Mx-mx==My-my)printf("YES\n");
        else printf("NO\n");
    }
}
