#include <iostream>
using namespace std;

int main()
{
	int n,x1,x2,y1,y2,mi1=32000,mi2=32000,ma1=0,ma2=0;
	long long sum=0,s=0;
	cin >> n;
	for(int i=0;i<n;i++)
	{
		cin>>x1>>y1>>x2>>y2;
		mi1=min(mi1,x1);
		mi2=min(mi2,y1);
		ma1=max(ma1,x2);
		ma2=max(ma2,y2);
		sum+=(x2-x1)*(y2-y1);
	}
	s=(ma1-mi1)*(ma2-mi2);
	if((s == sum) && ((ma1-mi1) == (ma2-mi2)))
		cout<<"YES";
	else
		cout<<"NO";
	cout<<endl;
	return 0;
}
