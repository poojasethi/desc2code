#include <iostream>

using namespace std;

int p1 = 99999, p2 = 99999, p3, p4 , n, i, a, b, c, d, nr;

int main() {
    cin >> n;
    
    for(i = 1;i <= n;i++) {
        cin >> a >> b >> c >> d;
        
        if(p1 > a) p1 = a;
        if(p2 > b) p2 = b;
        if(p3 < c) p3 = c;
        if(p4 < d) p4 = d;
        
        nr += (b - d) * (a - c);
    }
    
    if(p3 - p1 == p4 - p2 && nr == (p3 - p1) * (p4 - p2)) cout << "YES\n";
    else cout << "NO\n";
    
    return 0;
}
