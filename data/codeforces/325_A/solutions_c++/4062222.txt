#include<iostream>
#include<cmath>
using namespace std;
long long n;
int main(){
	cin >> n;
	long long a,b,c,d;
	long long sum=0;
	long long Maxx=0,Maxy=0,Miny=9999999,Minx=9999999;
	for(long long i=0;i<n;i++){
		cin >>a>>b>>c>>d ;
		sum += (c-a)*(d-b);
		Maxx = max(Maxx,c);
		Minx = min(Minx,a);
		Maxy = max(Maxy,d);
		Miny = min(Miny,b);
	}
	if((sum == (Maxx-Minx)*(Maxy-Miny) )&&(Maxx-Minx)==(Maxy-Miny) ){
		cout<<"YES"<<endl;
	}
	else
		cout<<"NO"<<endl;
}
