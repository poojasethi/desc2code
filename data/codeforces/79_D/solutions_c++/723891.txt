#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
#define fo(i,a,b) for (int i=a; i<=b; ++i)
#define min(a,b) ((a)<(b)?(a):(b))

const int mn=10010,mk=21,ms=1<<mk,ml=110;

int a[mn],b[mk],e[mk],cost[mk][mk],c[mn],q[mn],f[ms];
int k,N,K,L;

void calc(int t){
	int l=0,r=0;
	memset(c,-1,sizeof(c));
	c[q[++r]=b[t]]=0;
	while (l++!=r){
		int x=q[l];
		fo (i,1,L){
			if (x+a[i]<=N+1 && c[x+a[i]]==-1) c[q[++r]=x+a[i]]=c[x]+1;
			if (x-a[i]>=1 && c[x-a[i]]==-1) c[q[++r]=x-a[i]]=c[x]+1;
		}
	}
	fo (i,1,k) cost[t][i]=c[b[i]];
}

int main(){
//	freopen("led.in","r",stdin);
//	freopen("led.out","w",stdout);
	int ca=1;
//	scanf("%d",&ca);
	while (ca--){
		scanf("%d%d%d",&N,&K,&L);
		memset(a,0,sizeof(a));
		fo (i,1,K){
			int x;
			scanf("%d",&x);
			a[x]=1;
		}
		a[0]=a[N+1]=k=0,e[0]=1;
		fo (i,1,N+1) if (a[i]^a[i-1]) b[++k]=i,e[k]=e[k-1]<<1;
		fo (i,1,L) scanf("%d",&a[i]);
		fo (i,1,k) calc(i);
		memset(f,60,sizeof(f));
		int oo=f[0];
		f[0]=0;
		fo (i,0,e[k]-1) if (f[i]!=oo)
			fo (j,0,k-1) if (!(i&e[j])){
				fo (p,j+1,k-1) if (!(i&e[p])){
					int t=i|e[j]|e[p];
					if (cost[p+1][j+1]==-1) continue;
					f[t]=min(f[t],f[i]+cost[p+1][j+1]);
				}
				break;
			}
		if (f[e[k]-1]==oo) cout<<-1<<endl;
		else cout<<f[e[k]-1]<<endl;
	}
}
