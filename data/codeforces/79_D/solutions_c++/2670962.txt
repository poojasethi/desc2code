#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
int tot,n,k,l,i,j,o,x[21],a[111];
int dl[10011],dis[21][21],stu[10011],flag[10011],f[1<<20];
int main()
{
		scanf("%d%d%d",&n,&k,&l);
		for (i=1;i<=k;i++) scanf("%d",&x[i]),flag[x[i]]^=1,flag[x[i]+1]^=1;
		for (i=1;i<=l;i++) scanf("%d",&a[i]);
		n++;
		k=0;
		for (i=1;i<=n;i++)
				if (flag[i]) x[++k]=i,flag[i]=k;
		sort(x+1,x+k+1);
		memset(dis,63,sizeof(dis));
		for (i=1;i<=k;i++)
		{
				dis[i][i]=0;
				memset(stu,63,sizeof(stu));
				stu[x[i]]=0;
				dl[tot=1]=x[i];
				for (j=1;j<=tot;j++)
						for (o=1;o<=l;o++)
						{
								if (dl[j]-a[o]>0 && stu[dl[j]-a[o]]>stu[dl[j]]+1)
								{
									 stu[dl[j]-a[o]]=stu[dl[j]]+1;
									 dl[++tot]=dl[j]-a[o];
									 if (flag[dl[tot]]) dis[i][flag[dl[tot]]]=stu[dl[tot]];
								}
								if (dl[j]+a[o]<=n && stu[dl[j]+a[o]]>stu[dl[j]]+1)
								{
									 stu[dl[j]+a[o]]=stu[dl[j]]+1;
									 dl[++tot]=dl[j]+a[o];
									 if (flag[dl[tot]]) dis[i][flag[dl[tot]]]=stu[dl[tot]];
								}
						}
		}
		memset(f,63,sizeof(f));
		f[0]=0;
		for (i=0;i<1<<k;i++)
		for (j=0;j<k;j++)
		if ((i&(1<<j))==0)
		for (o=j+1;o<k;o++)
		if ((i&(1<<o))==0)
			 f[i+(1<<j)+(1<<o)]=min(f[i+(1<<j)+(1<<o)],f[i]+dis[j+1][o+1]);
	  i=f[i-1];
	  if (i>1000000) printf("-1\n");
	  else					 printf("%d\n",i);
		return 0;
}
