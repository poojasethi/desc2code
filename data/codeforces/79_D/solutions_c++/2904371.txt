#include<cstdio>
#include<cstring>
#define N 10005
int a[N],c[N],D[N],v[N],f[N],g[1<<21],d[25][25],n,m,k,len,x;
inline void Min(int &x,int y){if (y<x) x=y;}
void bfs(int j)
{
	int h=0,t=1,x=c[j]; memset(D,-1,sizeof(D)); D[f[1]=x]=0;
	while (h<t){
		x=f[++h];
		for (int i=1;i<=m*2;i++){
			int y=x+v[i]; if (y<n+2 && y>0 && D[y]<0) D[y]=D[x]+1,f[++t]=y;
			}
		}
	for (int i=0;i<len;i++) d[j][i]=D[c[i]];
}
int main()
{
	scanf("%d%d%d",&n,&k,&m);
	for (int i=1;i<=k;i++) scanf("%d",&x),a[x]^=1,a[x+1]^=1;
	for (int i=1;i<=m;i++) scanf("%d",&v[i]),v[m+i]=-v[i];
	for (int i=1;i<=n+1;i++) if (a[i]) c[len++]=i;
	for (int i=0;i<len;i++) bfs(i); n=len;
	memset(g,6,sizeof(g)); g[0]=0;
	for (int i=0;i<1<<n;i++){
		int cnt=0; for (k=0;k<n;k++) cnt+=(i&1<<k)>0;
		if (cnt&1) continue; for (k=0;i&1<<k;k++);
		for (int j=k+1;j<n;j++) if (!(i&1<<j) && d[k][j]>=0)
			Min(g[i|1<<k|1<<j],g[i]+d[k][j]);
		}
	int ans=g[(1<<n)-1]; if (ans>1e6) ans=-1;
	printf("%d\n",ans); return 0;
}
