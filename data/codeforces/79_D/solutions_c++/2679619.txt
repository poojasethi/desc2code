#include <cstdio>
#include <cstring>
int q[10000000],to[20000000],next[20000000],dist[100000],d[30][30],bit[100000],x[30],t,h,head[100000],n,k,l,m,a[10000],f[10000000]; 
void bfs(int u)
{
  memset(dist,-1,sizeof(dist));
  h = t =0; q[h] = u; dist[u] = 0; int i,v;
  while(h<=t)
    {
      v = q[h++]; 
      for(i=head[v];i;i=next[i])
	if(dist[to[i]] == -1 || dist[v]+1 < dist[to[i]])
	  {
	    dist[to[i]] = dist[v]+1;
	    q[++t]=to[i];
	  }
    }
}
bool check(int i)
{int j,cnt;
  for(cnt=j=0;(1<<j)<=i;j++)
    if((1<<j) & i) cnt++;
  if(cnt%2 || i==k) return false;
  return true;
}
int main()
{
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  scanf("%d%d%d",&n,&k,&l); int i,u,v;
  for(i=1;i<=k;i++)
    scanf("%d",&u),bit[u]=1;
  for(i=k=0;i<=n;i++)
    if(bit[i]!=bit[i+1]) x[++k]=i;
  for(i=1;i<=l;i++)
    {
      scanf("%d",&a[i]);
      for(u=0;u+a[i]<=n;u++)
	{
	  to[++m]=u+a[i]; next[m]=head[u]; head[u]=m;
	  to[++m]=u; next[m]=head[u+a[i]]; head[u+a[i]]=m;
	}
    }
  for(i=0;i<k;i++)
    {
      bfs(x[i+1]);
      for(u=0;u<k;u++) d[i][u]= dist[x[u+1]]==-1 ? 999999999:dist[x[u+1]];
    }
  for(i=0;i<(1<<k);i++) f[i] = 999999999;
  f[0] = 0;
  for(i=0;i<(1<<k);i++)
    if(check(i))
    for(u=0;u<k;u++)
      if(!((1<<u)&i))
	for(v=u+1;v<k;v++)
	  if(!((1<<v)&i))
	    if(f[i|(1<<u)|(1<<v)] > f[i]+d[u][v])
	      f[i|(1<<u)|(1<<v)] = f[i]+d[u][v];
  printf("%d\n",f[(1<<k)-1]==999999999?-1:f[(1<<k)-1]);
}
