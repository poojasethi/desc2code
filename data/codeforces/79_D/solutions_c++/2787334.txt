using namespace std;

#define MP make_pair
#define fi first
#define sc second
#define LL long long
#define LD long double
const int maxn = 10100;

int Mk[maxn],h[maxn],X[maxn],C[21][21];
int opt[1<<21],f[22],dis[maxn];
int N,K,L,A[maxn],lk,len;
int main(){
	#ifdef ONLINE_JUDGE
	#else
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif
	memset(dis,10,sizeof(dis));
	scanf("%d%d%d",&N,&K,&L);N++;
	for (int i=1;i<=K;i++){
		int x;scanf("%d",&x);
		Mk[x]=1;
	}
	for (int i=N;i;i--) Mk[i]^=Mk[i-1];
	for (int i=1;i<=N;i++)
	if(Mk[i]){
		lk++;Mk[lk]=i;
	}
	for (int i=1;i<=L;i++)
		scanf("%d",&A[i]);
	for (int i=1;i<=lk;i++){
		memset(dis,10,sizeof(dis));
		dis[Mk[i]]=0;
		len=1;h[1]=Mk[i];
		int t=1;
		while (t<=len){
			int x=h[t];
			for (int j=1;j<=L;j++){
				if (x+A[j]<=N&&dis[x+A[j]]>N){
					dis[x+A[j]]=dis[x]+1;
					h[++len]=x+A[j];
				}
				if (x-A[j]>0&&dis[x-A[j]]>N){
					dis[x-A[j]]=dis[x]+1;
					h[++len]=x-A[j];
				}
			}
			t++;
		}
		for (int j=1;j<=lk;j++)
			C[i][j]=dis[Mk[j]];
	}
	f[0]=1;for (int i=1;i<=lk;i++) f[i]=f[i-1]*2;
	memset(opt,10,sizeof(opt));
	opt[f[lk]-1]=0;
	for (int S=f[lk]-1;S;S--){
		for (int i=0;i<lk;i++)
			for (int j=i+1;j<lk;j++)
			if (i!=j&&(S&(f[i]+f[j]))==f[i]+f[j]){
				if (opt[S-f[i]-f[j]]>opt[S]+C[i+1][j+1])
					opt[S-f[i]-f[j]]=opt[S]+C[i+1][j+1];
			}
	}
	if (opt[0]>N) opt[0]=-1;
	printf("%d\n",opt[0]);
	return 0;
}
