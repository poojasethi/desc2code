#include<cstdio>
int f[1048576],dis[20][20],a[100],p[20];
typedef int sz[10005];
sz d,c,que;
void cmin(int &a,int b){
	if (b<a) a=b;
}
int main(){
	int n,m,t,i,j,k,x,*l,*r;
	scanf("%d%d%d",&n,&m,&t);
	for (i=0;i<=n+1;i++) c[i]=0;
	for (i=0;i<m;i++) c[scanf("%d",&x),x]=1;
	for (i=0;i<t;i++) scanf("%d",a+i);
	for (m=0,i=1;i<=n+1;i++) if (c[i]!=c[i-1]){
		for (j=1;j<=n+1;j++) d[j]=1<<29;
		for (d[*(l=r=que)=p[m++]=i]=0;l<=r;l++) for (j=0;j<t;j++){
			if (*l>a[j] && d[*l]+1<d[*l-a[j]]) d[*++r=*l-a[j]]=d[*l]+1;
			if (a[j]+*l<=n+1 && d[*l]+1<d[*l+a[j]]) d[*++r=*l+a[j]]=d[*l]+1;
		}
		for (j=0;j<m;j++) dis[j][m-1]=dis[m-1][j]=d[p[j]];
	}
	for (f[0]=0,i=1;i<1<<m;i++){
		f[i]=1<<29;
		for (j=0;j<m;j++) if (i&1<<j)
			for (k=j+1;k<m;k++) if (i&1<<k)
				cmin(f[i],f[i-(1<<j)-(1<<k)]+dis[j][k]);
	}
	printf("%d\n",f[(1<<m)-1]==1<<29?-1:f[(1<<m)-1]);
	return 0;
}
