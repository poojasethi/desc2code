#include <cstdio>
#include <cstring>
#include <algorithm>
#define rep(i,a,b) for (int i=a;i<b;i++)
int n,k,l,x,t,p,c[10100],d[10100],q[10100],a[10100],g[30][10100],h[10100],f[2000000];
int main() {
	scanf("%d%d%d",&n,&k,&l);
	rep(i,0,k) scanf("%d",&x),c[x]=1;
	rep(i,0,l) scanf("%d",a+i);
	rep(i,0,n+1) if (c[i]^=c[i+1]) {
		memset(d,0,sizeof(d));
		q[(t=1)-1]=i;d[i]=1;
		rep(j,0,t) rep(k,0,l) {
			if (q[j]>=a[k]&&!d[q[j]-a[k]]) d[q[t++]=q[j]-a[k]]=d[q[j]]+1;
			if (q[j]+a[k]<=n&&!d[q[j]+a[k]]) d[q[t++]=q[j]+a[k]]=d[q[j]]+1;
		}
		rep(j,0,n+1) g[p][j]=d[j]?d[j]-1:0x30303030;
		h[p++]=i;
	}
	memset(f,0x30,sizeof(f));f[0]=0;
	rep(i,0,(1<<p)) rep(j,0,p) if ((~i>>j)&1) rep(k,j+1,p) if ((~i>>k)&1)
		f[i|(1<<j)|(1<<k)]=std::min(f[i|(1<<j)|(1<<k)],f[i]+g[j][h[k]]);
	printf("%d\n",f[(1<<p)-1]<100000000?f[(1<<p)-1]:-1);
}
