#include <cstdio>
#include <cstring>

using namespace std;

const int INF = 0x3f3f3f3f;

int n,m,k,y;
int a[110],c[10010];
int d[25][25],f[1<<20];

void ckmin(int &a,int b){b<a ? a=b : 0;}
void bfs(int p){
	static int w[10010],q[10010],l,r;
	memset(w,0x3f,sizeof(w));
	for(w[q[l=r=0]=p]=0;l<=r;l++)
		for(int i=1,t;i<=y;i++){
			if((t=q[l]+a[i])<=n+1 && w[t]>=INF) w[q[++r]=t]=w[q[l]]+1;
			if((t=q[l]-a[i])>=1 && w[t]>=INF) w[q[++r]=t]=w[q[l]]+1;
		}
	for(int i=1;i<=n+1;i++) if(c[i]) d[c[p]][c[i]]=w[i];
}
int main()
{
	scanf("%d%d%d",&n,&k,&y);
	for(int i=1,x;i<=k;i++) scanf("%d",&x),c[x]^=1,c[x+1]^=1;
	for(int i=1;i<=y;i++) scanf("%d",&a[i]);
	for(int i=1;i<=n+1;i++) if(c[i]) c[i]=++m;
	for(int i=1;i<=n+1;i++) if(c[i]) bfs(i);
	memset(f,0x3f,sizeof(f));f[(1<<m)-1]=0;
	for(int i=(1<<m)-1;i;i--) if(f[i]<INF){
		int q=__builtin_ctz(i)+1,p=i^(i&-i);
		for(int j=0;j<m;j++) if((p&(1<<j)) && d[q][j+1]<INF)
			ckmin(f[p^(1<<j)],f[i]+d[q][j+1]);
	}
	printf("%d\n",f[0]<INF ? f[0] : -1);
	return 0;
}
