#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 10010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,K,O,x;
int a[N],b[N],id[N],g[N],gg[N],f[1<<20],c[20][20];
int main() {
	scanf("%d%d%d",&n,&K,&m);
	n++;
	For(i,1,K) {
		scanf("%d",&x);
		a[x]^=1,a[x+1]^=1;
	}
	For(i,1,m) scanf("%d",&b[i]);
	For(i,1,n) if (a[i]) id[i]=++O;
	For(i,1,n) if (a[i]) {
		int l,r;
		For(j,1,n) gg[j]=-1;
		g[l=r=1]=i; gg[i]=0;
		for (;l<=r;l++) {
			int A=g[l];
			For(j,1,m)For(k,-1,1) {
				int B=A+k*b[j];
				if (B>0&&B<=n&&gg[B]<0) {
					gg[B]=gg[A]+1;
					g[++r]=B;
				}
			}
		}
		For(j,1,n) if (a[j]&&gg[j]>=0&&j!=i) {
			int A=id[i],B=id[j];
			if (A>B) swap(A,B);
			c[A-1][B-1]=gg[j];
		}
	}
	memset(f,120,sizeof(f));
	f[0]=0;
	int nn=(1<<O)-1;
	For(i,0,nn) if (f[i]<1e9) {
		For(j,0,O-1) if (!(i>>j&1)) {
			For(k,j+1,O-1) if (!(i>>k&1)&&c[j][k]) {
				int A=i|1<<j|1<<k;
				f[A]=min(f[A],f[i]+c[j][k]);
			}
		}
	}
	printf("%d\n",f[nn]>1e9?-1:f[nn]);
	return 0;
}
