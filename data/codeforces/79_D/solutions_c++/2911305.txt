#include<cstdio>
#include<cstring>
int N,K,L,i,j,k,X,Cnt;
int B[101],C[21],son[21];
int d[524288],next[524288],f[524288];
int A[100002],Q[100002];
int g[21][21];
void Spfa(int x)
{
    int l,r,i,j;
    memset(d,127,sizeof(d));
    Q[1]=x,d[x]=0;
    for(l=0,r=1;l<r;)
    {
        i=Q[++l];
        for(j=1;j<=L;++j)
        {
            if(i+B[j]<N+2&&d[i+B[j]]>d[i]+1) d[i+B[j]]=d[i]+1,Q[++r]=i+B[j];
            if(i-B[j]>0&&d[i-B[j]]>d[i]+1) d[i-B[j]]=d[i]+1,Q[++r]=i-B[j];
		}
    }
}
int main()
{
	scanf("%d%d%d",&N,&K,&L);
	for(i=1;i<=K;++i)
	{
		scanf("%d",&X);
		A[X]^=1,A[X+1]^=1;
	}
	for(i=1;i<=L;++i) scanf("%d",&B[i]);
	for(i=1;i<=N+1;++i) if(A[i]) C[++Cnt]=i;
	for(i=1;i<Cnt;++i)
	{
		Spfa(C[i]);
		for(j=i+1;j<=Cnt;++j) g[i][j]=d[C[j]];
	}
	memset(d,127,sizeof(d));
	memset(son,-1,sizeof(son));
	d[0]=0;
	for(i=0;i<1<<Cnt-1;++i)
		for(j=1;j<Cnt;++j)
			if((j&1)&&!(i&(1<<j-1))||!(j&1)&&(i&(1<<j-1)))
			{
				next[i]=son[j],son[j]=i;
				break;
			}
	for(i=1;i<Cnt;++i)
		for(j=son[i];j!=-1;j=next[j])
			if(d[j]<1000000000)
			{
				X=0;
				for(k=i;k<Cnt;++k)
				{
					X|=1<<k-1;
					if(d[j^X]>d[j]+g[i][k+1])
						d[j^X]=d[j]+g[i][k+1];
				}
			}
	X=0;
	for(i=1;i<Cnt;++i) if(i&1) X|=1<<i-1;
	if(d[X]>1000000000) printf("-1\n");
	else printf("%d\n",d[X]);
	return 0;
}
