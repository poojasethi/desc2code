#include<cstdio>
#include<cstring>

int s[10100],f[10100],a[110],b[10100],o[30],n,K,L,Ans[1100000],N,d[30][30];

void Solve(int S)
{
	int i,j,k;
	memset(f,63,sizeof f);
	f[s[0]=o[S]]=0;
	for(i=0,j=1;i<j;i++)
		for(k=0;k<L;k++)
		{
			if(s[i]+a[k]<=n+1&&f[s[i]+a[k]]==1061109567)
				f[s[j++]=s[i]+a[k]]=f[s[i]]+1;
			if(s[i]-a[k]>0&&f[s[i]-a[k]]==1061109567)
				f[s[j++]=s[i]-a[k]]=f[s[i]]+1;
		}
	for(i=S;i<N;i++)d[S][i]=f[o[i]];
}

int main()
{
	int i,j,k;
	scanf("%d%d%d",&n,&K,&L);
	for(i=0;i<K;i++){scanf("%d",&j);b[j]=1;}
	for(i=0;i<L;i++)scanf("%d",a+i);
	for(i=n+1;i>0;i--)b[i]=b[i]^b[i-1];
	for(i=1;i<=n+1;i++)if(b[i])o[N++]=i;
	for(i=0;i<N;i++)Solve(i);
	memset(Ans,63,sizeof Ans);
	Ans[0]=0;
	for(i=0;i<(1<<N);i++)if(Ans[i]!=1061109567)
	{
		for(j=0;j<N;j++)if(!(i&(1<<j)))break;
		for(k=j+1;k<N;k++)
			if(!(i&(1<<k))&&Ans[i|1<<j|1<<k]>Ans[i]+d[j][k])
				Ans[i|1<<j|1<<k]=Ans[i]+d[j][k];
	}
	if(Ans[(1<<N)-1]==1061109567)printf("-1\n");
		else printf("%d\n",Ans[(1<<N)-1]);
	return 0;
}
