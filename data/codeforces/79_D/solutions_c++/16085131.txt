#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>

using namespace std;

const int N=10010;
const int M=110;
const int K=22;
const int S=1200120;
const int I=2147483647;

int a[M];
int p[K];
int s[K][K];
int t[N];
int f[S];
int d[N];
bool b[N];
int n,m,i,j,k,l,r,u,v;

int main(){
	scanf("%d%d%d",&n,&k,&m);
	for(i=1;i<=k;i++){
		scanf("%d",&j);
		b[j]=true;
	}
	k=0;
	for(i=1;i<=n+1;i++)
		if(b[i]^b[i-1]){
			k++;
			p[k]=i;
		}
	for(i=1;i<=m;i++)scanf("%d",&a[i]);
	for(i=1;i<=k;i++){
		l=r=1;
		t[1]=p[i];
		memset(b,true,sizeof(b));
		d[p[i]]=0;
		b[p[i]]=false;
		while(l<=r){
			u=t[l];
			for(j=1;j<=m;j++){
				v=u+a[j];
				if(v<=n+1 && b[v]){
					b[v]=false;
					d[v]=d[u]+1;
					r++;
					t[r]=v;
				}
				v=u-a[j];
				if(v>0 && b[v]){
					b[v]=false;
					d[v]=d[u]+1;
					r++;
					t[r]=v;
				}				
			}
			l++;
		}
		for(j=1;j<=k;j++)s[i][j]=b[p[j]]?I:d[p[j]];
	}
	memset(f,0x7,sizeof(f));
	f[0]=0;
	m=1<<k;
	for(i=0;i<m;i++){
		for(u=0;(1<<u)&i;u++);
		for(v=u+1;v<k;v++)
			if(!((1<<v)&i) && s[u+1][v+1]!=I)
				f[i|(1<<u)|(1<<v)]=min(f[i|(1<<u)|(1<<v)],f[i]+s[u+1][v+1]);
	}
	if(f[m-1]!=f[S-1])printf("%d\n",f[m-1]);
	else printf("-1\n");
	return 0;
}
