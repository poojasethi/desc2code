#include <cstdio>
#include <cstring>

using namespace std;

const int INF = 0x3f3f3f3f;

int n,k,L;
int x[15],a[110],c[10010],f[1<<20];
int pos[20],m,d[10010],dist[20][20];

inline void checkmin(int &a,int b){if(b<a) a=b;}
void bfs(int num){
	static int q[10010],l,r;
	memset(d,0x3f,sizeof(d));d[pos[num]]=0;
	for(q[l=r=0]=pos[num];l<=r;l++)
		for(int i=0;i<L;i++){
			if(q[l]+a[i]<=n+1 && d[q[l]+a[i]]==INF) d[q[++r]=q[l]+a[i]]=d[q[l]]+1;
			if(q[l]-a[i]>0    && d[q[l]-a[i]]==INF) d[q[++r]=q[l]-a[i]]=d[q[l]]+1;
		}
	for(int i=0;i<m;i++) dist[num][i]=d[pos[i]];
}
int main()
{
	scanf("%d%d%d",&n,&k,&L);
	for(int i=0;i<k;i++) scanf("%d",&x[i]),c[x[i]]^=1,c[x[i]+1]^=1;
	for(int i=0;i<L;i++) scanf("%d",&a[i]);
	for(int i=1;i<=n+1;i++) if(c[i]) pos[m++]=i;
//	for(int i=0;i<m;i++) printf("%d ",pos[i]);puts("");
	if(m&1) return puts("-1") && 0;
	for(int i=0;i<m;i++) bfs(i);
//	for(int i=0;i<m;i++){
//		for(int j=0;j<m;j++)
//			printf(" %d",dist[i][j]);
//		puts("");
//	}
	memset(f,0x3f,sizeof(f));f[0]=0;
	for(int i=1;i<(1<<m);i++)
		for(int j=0;j<m;j++) if(i&(1<<j))
			for(int p=j+1;p<m;p++) if((i&(1<<p)) && f[i^(1<<j)^(1<<p)]<INF && dist[j][p]<INF)
				checkmin(f[i],f[i^(1<<j)^(1<<p)]+dist[j][p]);
	printf("%d\n",f[(1<<m)-1]<INF ? f[(1<<m)-1] : -1);
	return 0;
}
