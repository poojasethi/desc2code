#include<stdio.h>
#include<algorithm>
using namespace std;
typedef __int64 ll;
const int mo=1000000007;
ll dp[505][505],f[500005];
int main()
{
	int i,j,ii;
	int n,k;
	f[0]=1;	
	for(i=0;i<=500;i++)
		dp[i][0]=1;
	for(i=0;i<=500000;i++)
		f[i+1]=(ll)2*f[i]%mo;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)
	{
		
		for(j=1;j<=k;j++)
		{
		//printf("1111111111111\n");
			for(ii=0;ii<i;ii++)
			{
				dp[i][j]+=(ll)dp[ii][j-1]*(f[i-ii]-1)%mo*f[(i-ii)*ii]%mo;
				dp[i][j]%=mo; 
			}
		}
	}
	printf("%I64d\n",dp[n][k]);
}
