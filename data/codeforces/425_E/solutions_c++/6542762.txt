#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
#define N 510
using namespace std;
const int mo=1000000007;
int n,K,f[N][N],pow2[N*N],ans,i,j,k;
int main(){
	scanf("%d%d",&n,&K);
	f[0][0]=1;
	pow2[0]=1;
	for(i=1;i<=n*n;++i)pow2[i]=pow2[i-1]*2%mo;
	for(i=1;i<=n;++i)
		for(j=1;j<=K;++j)
			for(k=0;k<i;++k)
				f[i][j]=(f[i][j]+1ll*f[k][j-1]*(pow2[i-k]-1)%mo*pow2[(i-k)*k])%mo;
	ans=0;
	for(i=0;i<=n;++i)ans=(ans+1ll*f[i][K]*pow2[(n-i)*i])%mo;
	printf("%d\n",ans);
}
