#include <cstdio>
using namespace std;

int N, K;


int table[512][512];
int p2[512 * 512];
const int MOD = 1000000007;
long long int go(int n, int k)
{
    if(k == 0)
        return 1;
    else if(table[n][k])
        return table[n][k] - 1;
    else
    {
        int& v = table[n][k];
        for(int i = n + 1; i <= N; i++)
        {
            v += go(i, k - 1) * (p2[i - n] - 1) % MOD * p2[(N - i) * (i - n)] % MOD;
            v %= MOD;
        }
        if(v < 0)
            v += MOD;
        return v++;
    }
}

int main()
{
    p2[0] = 1;
    for(int i = 1; i < 512 * 512; i++)
        p2[i] = p2[i - 1] * 2 % MOD;
    scanf("%d%d", &N, &K);
    printf("%d\n", (int)go(0, K));


    return 0;
}
