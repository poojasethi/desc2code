#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int f[505][505];
const int p=1000000007;
typedef long long LL;
int power[200000];

int find(int x,int y){
	return (1<<y)-(1<<x);
	return y-x;
	return y*(y+1)/2-x*(x+1)/2;
}

int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	f[0][0]=1;
	power[0]=1;
	for (int i=1; i<200000; ++i)
		power[i]=power[i-1]*2%p;
	for (int i=1; i<=n; ++i)
		for (int j=1; j<=m; ++j)
			for (int k=0; k<i; ++k){
				f[i][j]+=(LL)((LL)f[k][j-1]*power[k*(i-k-1)])%p*(power[i]-power[k]+p)%p;
				f[i][j]%=p;
			}
	int ans=0;
	for (int i=0; i<=n; ++i){
		ans+=(LL)f[i][m]*power[i*(n-i)]%p;
		//printf("%d %d %d\n",i,ans,f[i][m]);
		ans%=p;
	}
	printf("%d\n",ans);
		
}
