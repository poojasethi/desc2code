#include <iostream>
using namespace std;
const int N = 500 + 50, mod = 1e9 + 7;

int n, k, dp[N][N], p2[N * N] = {1};

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n >> k;
	for (int i = 1; i <= n * n; i++)
		p2[i] = p2[i - 1] * 2 % mod;
	for (int i = 0; i <= n; i++) {
		dp[i][0] = 1;
		for (int j = 1; j <= k; j++)
			for (int c = 1; c <= i; c++)
				dp[i][j] = (dp[i][j] + 1ll * dp[i - c][j - 1] * p2[(i - c) * c] % mod * (p2[c] - 1)) % mod;
	}
	cout << dp[n][k] << '\n';
	return 0;
}
