#include<cstdio>
#include<iostream>
#include<cstring>
#include<cstdlib>
using namespace std;
long long pow[250002];
long long dp[502][502];
long long mo,ans;
int i,j,k,n,K;
int main()
{
	mo=1000000007;
	scanf("%d%d",&n,&K);
	dp[0][0]=1LL;
	pow[0]=1LL;
	for (i=1;i<=n*n;++i)
	pow[i]=pow[i-1]*2%mo;
	for (i=1;i<=K;++i)
	for (j=1;j<=n;++j)
	{
		for (k=0;k<j;++k)
		dp[i][j]=(dp[i][j]+dp[i-1][k]*pow[k*(j-k)]%mo*(pow[j-k]-1)%mo)%mo;
	}
	ans=0;
	for (j=0;j<=n;++j)
	ans=(ans+dp[K][j]*pow[j*(n-j)]%mo)%mo;
	cout<<ans<<endl;
}
