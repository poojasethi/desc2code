#include <stdio.h>
#define N 510
#define M 70000

typedef __int64 ll;

ll f[N][N];
ll b[M];
const ll Mod=1000000007;

int main() {
	int nn, kk;
	scanf("%d %d", &nn, &kk);
	b[0]=1;
	for (int i=1; i<M; i++) {
		b[i]=b[i-1]<<1;
		if (b[i]>=Mod) b[i]-=Mod;
	}
	f[0][0]=1;
	for (int n=1; n<=nn; n++) f[n][0]=1;
	for (int n=1; n<=nn; n++)
		for (int k=1; k<=n; k++) {
			f[n][k]=0;
			for (int a=k; a<=n; a++) {
				ll tp=(b[a*(n-a+1)]-b[(a-1)*(n-a+1)]);
				if (tp<0) tp+=Mod;
				tp=tp*f[a-1][k-1]%Mod;
				f[n][k]+=tp;
				if (f[n][k]>=Mod) f[n][k]-=Mod;
			}
		}

	printf("%I64d\n", f[nn][kk]);
}
