#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;

long long f[510][510];
long long s[1000010];
int n,m;
int mod=(int)1e9+7;

int main()
{
    int i,j,k;
    cin>>n>>m;
    s[0]=1ll;
    for(i=1;i<=n*n;i++)
        s[i]=(s[i-1]*2)%mod;
    for(i=0;i<=n;i++)
        f[i][0]=1;
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++)
            for(k=j-1;k<i;k++)
                f[i][j]=(f[i][j]+f[k][j-1]*(s[i-k]-1)%mod*s[(i-k)*k])%mod;
    cout<<f[n][m]<<endl;
    return 0;
}
