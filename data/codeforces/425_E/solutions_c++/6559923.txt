#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int mm=1000000007;
typedef long long ll;
ll f[2][511][511],pw[511];
int ans,i,j,k,n,m,now,pre,t;
int main(){
	scanf("%d%d",&n,&m);
	memset(f,0,sizeof(f));
	now=ans=0;f[now][0][0]=1;
	for (i=pw[0]=1;i<=501;i++) pw[i]=pw[i-1]*2%mm;
	for (i=1;i<=n;i++){
		now^=1;pre=now^1;memset(f[now],0,sizeof(f[now]));
		for (j=0;j<=m;j++)
		 for (k=0;k<=i;k++){
		 	f[now][j][k]=(f[now][j][k]+f[pre][j][k]*pw[k])%mm;
		 	if (k==i&&j){
		 		for (t=0;t<i;t++) f[now][j][k]=(f[now][j][k]+f[pre][j-1][t]*pw[t]%mm*(pw[i-t]-1))%mm;
		 	}
			if (i==n&&j==m) ans=(ans+f[now][j][k])%mm;
		 }
	}
	printf("%d\n",ans);
	return 0;
}
