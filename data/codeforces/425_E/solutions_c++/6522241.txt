#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long LL;
const int maxn = 501;
const int mod = 1e9+7;
int N, K;
LL d[maxn][maxn], p[maxn*maxn];
int main()
{
    scanf ("%d%d", &N, &K);
    p[0] = 1;
    for (int i=1; i<maxn*maxn; ++i) {
        p[i] = p[i-1]*2%mod;
    }
    d[0][0] = 1;
    for (int i=1; i<=N; ++i) {
        d[i][0] = 1;
        for (int j=1; j<=K; ++j) {
            for (int k=1; k<=i; ++k) {
                d[i][j] = (d[i][j]+(p[k]-1)*p[(i-k)*k]%mod*d[i-k][j-1]%mod)%mod;
            }
        }
    }
    printf ("%I64d\n", d[N][K]);
    return 0;
}
