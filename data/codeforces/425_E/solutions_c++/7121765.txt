#include<stdio.h>
#define N 505
#define mod 1000000007
#define LL __int64
LL g[N*N], f[N][N];
int n, m;

int main() {
    scanf("%d%d", &n, &m);
    g[0] = 1;
    for (int i = 1; i < 250005; i++)g[i] = g[i - 1]*2 % mod;
    for (int i = 0; i <= n; i++)f[i][0] = 1;
    for (int i = 0; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            for (int k = 0; k <= i; k++)
                f[i][j] = (f[i][j]+(g[i - k] - 1) * g[(i - k) * k] % mod * f[k][j - 1] % mod) % mod;
        }
    }
    printf("%I64d\n", f[n][m]);
    return 0;
}
