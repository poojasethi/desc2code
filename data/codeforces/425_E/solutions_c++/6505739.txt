#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
int n,k,f[501][501],pow[500001];
const int MOD=7+(int)1e9;
int main(){
	scanf("%d%d",&n,&k);
	if(k==0)puts("1");
	else{
		pow[0]=1;
		for(int i=1;i<=n*n;i++)
			pow[i]=(pow[i-1]*2LL)%MOD;
		f[0][0]=1;
		int ans=0;
		for(int i=1;i<=n;i++){
			for(int j=1;j<=k;j++){
				for(int x=0;x<i;x++){
					//f[x][j-1],f[x][j]
					(f[i][j]+=((long long)(pow[i-x]-1)*pow[x*(i-x)]%MOD)*f[x][j-1]%MOD)%=MOD;
				}
			}(ans+=(long long)f[i][k]*pow[i*(n-i)]%MOD)%=MOD;
		}
		printf("%d\n",ans);
	}
}
