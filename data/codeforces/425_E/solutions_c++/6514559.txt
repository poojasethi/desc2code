#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long int64;
const int mo = 1e9+7;
int n,K;
int t[250010];
int F[510][510];

int main(){
    scanf("%d %d",&n,&K);t[0] = 1;
    for(int i = 1; i <= n * n; i++) t[i] = (t[i-1]<<1) % mo;
    for(int i = 0; i <= n; i++) F[0][i] = 1;
    for(int i = 1; i <= K; i++)
        for(int j = 1; j <= i + n - K; j++)
            for(int k = i - 1,&f = F[i][j],tmp; k < j; k++){
                tmp = (int64)(t[j-k] - 1) * t[k*(j-k)] % mo;
                tmp = (int64)tmp * F[i-1][k] % mo;
                f = (f + tmp) % mo;
            }
    printf("%d",F[K][n]);
    return 0;
}
