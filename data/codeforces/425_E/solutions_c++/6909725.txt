#include <stdio.h>
#include <memory.h>
#include <algorithm>
using namespace std;
#define M 501
typedef __int64 ll;
#define mod 1000000007

int r[M*M];
int p[M][M];

void Prepare() {
	int n,m,i,j,k;
	r[0]=1;
	for(i=1;i<M*M;i++) r[i]=r[i-1]*2%mod;
	memset(p,0,sizeof(p));
	for(i=0;i<M;i++) p[i][0]=1;
	for(n=1;n<M;n++) {
		for(m=1;m<=n;m++) {
			for(k=1;k<=n;k++) {
				ll tmp=p[k-1][m-1];
				tmp=tmp*r[(k-1)*(n-k+1)]%mod;
				tmp=tmp*(r[n-k+1]-1)%mod;
				p[n][m]=(p[n][m]+tmp)%mod;
			}
		}
	}
}

int main() {
//	freopen("E.in","r",stdin);
	int n,m;
	Prepare();
	scanf("%d %d",&n,&m);
	printf("%d\n",p[n][m]);
	return 0;
}
