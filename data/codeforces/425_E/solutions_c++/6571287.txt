#include<cstdio>
const int N=505;
int mod=1000000007,ans,dp[N][N],two[N];
void up(int &v,int x){v=v+x;if(v>=mod)v-=mod;}
int main(){ 
  int n,m;scanf("%d%d",&n,&m);
  two[0]=1;dp[0][0]=1;
  for(int i=1;i<=n;i++)two[i]=(two[i-1]*2)%mod;
  for(int r=1;r<=n;r++)
    for(int i=r-1;i>=0;i--)
      for(int j=m;j>=0;j--)
        up(dp[r][j+1],1LL*dp[i][j]*two[i]%mod*(two[r-i]-1)%mod),
        up(dp[i][j],1LL*dp[i][j]*(two[i]-1)%mod);  
  int ans=0;  
  for(int i=0;i<=n;i++)up(ans,dp[i][m]);
  printf("%d\n",ans);
}
