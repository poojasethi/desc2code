#include<cstring>
#include<cstdio>
#define rep(i,a,b) for (int i=a;i<=b;++i)
#define N 505
#define P 1000000007
using namespace std;
int n,k,ans=0,m[N*N],f[N][N];
int main() {
    scanf("%d%d",&n,&k);m[0]=f[0][0]=1;
    rep(i,1,n*n) m[i]=m[i-1]*2%P;
    rep(i,1,n)
        rep(j,1,k)
            rep(p,0,i-1) f[i][j]=(f[i][j]+1LL*f[p][j-1]*(m[i-p]-1)%P*m[(i-p)*(n-i)]%P)%P;
    rep(i,0,n) ans=(ans+f[i][k])%P;
    printf("%d\n",ans);
    return 0;
}
