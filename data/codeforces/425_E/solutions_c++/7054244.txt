#include<cstdio>
using namespace std;

const int maxn = 502, M = 1000000007;
int n, k, opt[2][maxn][maxn], pow[maxn], ans;
int main()
{
  scanf("%d%d", &n, &k);
  pow[0] = 1;
  for (int i = 1; i < maxn; i++) pow[i] = (pow[i-1]<<1)%M;
  bool st = 0;
  opt[0][0][0] = 1;
  for (int i = 1; i <= n; i++) {
    for (int z = 0; z<i && z<=k; z++)
      for (int j = 0; j < i; j++) {
        opt[1-st][i][z+1] = (((1LL*opt[st][j][z]*(pow[i-j]-1))%M)*pow[j]+opt[1-st][i][z+1])%M;
        opt[1-st][j][z] = (1LL*opt[st][j][z]*pow[j])%M;
      }
    st = 1-st;
  }
  int ans = 0;
  for (int i = 0; i <= n; i++) ans = (ans+opt[st][i][k])%M;
  ans = (ans+M)%M;
  printf("%d\n", ans);
}
