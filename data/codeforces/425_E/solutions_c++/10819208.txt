#include<cstdio>
typedef long long LL;
const int M=1000000007;
int n,K,pow2[333333],f[555][555];

inline void up(int &x,const int &y)
{x+=y;if(x>=M)x-=M;}

int main()
{
    scanf("%d%d",&n,&K);
    pow2[0]=1;
    for(int i=1,t=n*n;i<=t;i++)
    {
        pow2[i]=pow2[i-1]<<1;
        if(pow2[i]>=M)pow2[i]-=M;
    }
    for(int i=0;i<=n;i++)f[0][i]=1;
    for(int i=1,j,k,t;i<=K;i++)
        for(j=1;j<=n-K+i;j++)
            for(int &s=f[i][j],k=i-1;k<j;k++)
                up(s,(LL)(pow2[j-k]-1)*pow2[k*(j-k)]%M*f[i-1][k]%M);
    printf("%d\n",f[K][n]);
    return 0;
}
