#include <cstdio>
#include <cstring>

using namespace std;

typedef long long LL;

const LL MOD = 1000000007;

int n,k;
LL f[510][510],pow[250010];

int main()
{
	scanf("%d%d",&n,&k);
	pow[0]=1;f[0][0]=1;
	for(int i=1;i<=250000;i++) pow[i]=pow[i-1]*2%MOD;
	for(int i=0;i<n;i++)
		for(int j=0;j<k;j++)
			for(int r=i+1;r<=n;r++){
				LL q=(pow[(r-i-1)*i]*(pow[r]-pow[i])%MOD+MOD)%MOD;
				(f[r][j+1]+=f[i][j]*q)%=MOD;
			}
	LL ans=0;
	for(int i=0;i<=n;i++) (ans+=f[i][k]*pow[(n-i)*i])%=MOD;
	printf("%I64d\n",ans);
	return 0;
}
