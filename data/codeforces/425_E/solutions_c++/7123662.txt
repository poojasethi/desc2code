#include<stdio.h>
#define MM 1000000007
__int64 mo[550550],dp[550][550];
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	mo[0]=1;
	for(int i=1;i<=500*500;i++) mo[i]=mo[i-1]*2%MM;
	for(int i=0;i<=n;i++) dp[i][0]=1;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=n;j++)
			for(int k=0;k<j;k++){
				dp[j][i]+=(mo[j-k]-1)*mo[(j-k)*k]%MM*dp[k][i-1]%MM;
				dp[j][i]%=MM;				
			}

	printf("%I64d\n",dp[n][m]);
}
