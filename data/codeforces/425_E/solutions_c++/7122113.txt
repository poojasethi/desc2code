#include"stdio.h"
#include"string.h"
#define mod 1000000007
__int64 er[555555];
__int64 dp[555][555];
int main()
{
	__int64 n,k,i,j,l;
	er[0]=1;
	scanf("%I64d%I64d",&n,&k);
	for(i=1;i<=501*500;i++)er[i]=(er[i-1]*2)%mod;
	for(i=0;i<=n;i++)dp[i][0]=1;
	for(i=0;i<=n;i++)
	{
		for(j=1;j<=k;j++)
		{
			for(l=0;l<=i;l++)
			{
				dp[i][j]+=(((er[i-l]-1)*(er[(i-l)*l]))%mod)*dp[l][j-1]%mod;
				dp[i][j]%=mod;
			}
		}
	}
	printf("%I64d\n",dp[n][k]);
	
}
