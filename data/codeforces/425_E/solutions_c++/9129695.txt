#include<iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<vector>
#include<algorithm>


using namespace std;
typedef long long ll;
#define N 555
#define mod  1000000007
ll dp[N][N],f[N*N+5];

int main()
{

 int n,m;
 cin>>n>>m;
 f[0]=1;
 for (int i=1;i<N*N;i++)
    f[i]=f[i-1]*2%mod;
for (int i=0;i<=n;i++) dp[i][0]=1;

 for (int i=1;i<=n;i++)
 {
     for (int j=1;j<=min(m,i);j++)
        for (int k=j-1;k<=i;k++)
        dp[i][j]=(dp[i][j]+dp[k][j-1]*(f[i-k]-1)%mod*f[(i-k)*k])%mod;
 }

 cout<<dp[n][m];
  return 0;
}
