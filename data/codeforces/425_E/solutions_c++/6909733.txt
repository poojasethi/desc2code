#include<stdio.h>
#define P 1000000007
#define M 600
#define N 100000
typedef long long ll;
ll p[N],f[M][M];
int n,k;
void solve(){
	int i,j,a;
	for(i=0;i<=k;i++)f[0][i]=0;
	f[0][0]=1;
	for(i=1;i<=n;i++){
		f[i][0]=1;
		for(j=1;j<=k;j++){
			f[i][j]=0;
			for(a=1;a<=i;a++){
				f[i][j]+=(f[a-1][j-1]*p[(a-1)*(i-a+1)])%P*(p[i-a+1]-1)%P;
			}
			f[i][j]%=P;
		}
	}
	return;
}
int main(){
	ll i;
	p[0]=1;
	for(i=1;i<N-1;i++){
		p[i]=p[i-1]*2;
		if(p[i]>=P) p[i]-=P;
	}
	scanf("%d %d",&n,&k);
	solve();
	printf("%I64d\n",f[n][k]);
	return 0;
}
