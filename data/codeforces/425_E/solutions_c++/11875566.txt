#include <cstdio>
using namespace std;

#define n   505
#define P   1000000007
#define For(i,a,b)  for(int i=a;i<=b;i++)

int     N,K,Pow[n*n],F[n][n];

void    Add(int&a,int b)    {(a+=b)%=P;}

int     main(){
        scanf("%d%d",&N,&K);
        For(i,0,N)  F[i][0]=1;  Pow[0]=1;
        For(i,1,N*N)Pow[i]=Pow[i-1]*2%P;
        For(k,0,N-1)    For(j,1,K)  if  (F[k][j-1]) For(i,k+1,N)
            Add(F[i][j],1ll*F[k][j-1]*(Pow[i-k]-1)%P*Pow[k*(i-k)]%P);
        printf("%d\n",F[N][K]);
}
