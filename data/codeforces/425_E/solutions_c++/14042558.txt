#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<map>

using namespace std;
typedef long long LL;
const int N=700;
const int MOD=1e9 + 7;
LL f[N][N],mi[N * N];
int main()
{
	
	mi[0]=1;
	for(int i=1;i<N * N;i++)
		mi[i]=mi[i-1] * 2 % MOD;
	int n,m;
	cin>>n>>m;
	f[0][0]=1;
	for(int i=0;i<m;i++)
		for(int j=0;j<=n;j++)
			if(f[i][j]!=0)
			{
				for(int k=j+1;k<=n;k++)
					f[i+1][k]=(f[i+1][k] + f[i][j] * (mi[k-j] - 1 + MOD) % MOD * mi[(k-j) * (j)] % MOD) % MOD;
			}
	LL ans=0;
	for(int i=0;i<=n;i++)
	if(f[m][i])
	{
		ans=(ans+f[m][i] * mi[i * (n-i)]) % MOD;
	}
	ans=(ans + MOD) % MOD;
	cout<<ans<<endl;
			
	return 0;
}
