#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
#define mod 1000000007
#define maxn 510
int dp[maxn][maxn],f[maxn*maxn];
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	f[0]=1;
	for (int i=1;i<=n*n;i++) f[i]=(f[i-1]<<1)%mod;
	for (int i=0;i<=n;i++) dp[i][0]=1;
	for (int i=1;i<=n;i++) 
		for (int j=1;j<=min(m,i);j++) 
			for (int k=j-1;k<=i;k++) 
				(dp[i][j]+=dp[k][j-1]*1ll*(f[i-k]-1)%mod*f[k*(i-k)]%mod)%=mod;
	printf("%d\n",(dp[n][m]+mod)%mod);
	return 0;
}
