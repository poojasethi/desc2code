#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=550,mod=1000000007;
long long p[maxn*maxn];
int n,K;
long long dp[maxn][maxn]={0};
int main()
{
	int i,j,k;
	scanf("%d%d",&n,&K);
	for(p[0]=i=1;i<maxn*maxn;i++)
		p[i]=p[i-1]*2%mod;
	for(i=0;i<=n;i++)
	{
		dp[i][0]=1;
		for(j=1;j<=K;j++)
		{
			for(k=1;k<=i;k++)
				dp[i][j]+=(p[k]-1)*p[(i-k)*k]%mod*dp[i-k][j-1]%mod;
			dp[i][j]%=mod;
		}
	}
	printf("%I64d\n",dp[n][K]);
	return 0;
}
