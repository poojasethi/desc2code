#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define N 510
#define mod 1000000007
typedef long long ll;
ll dp[N][N],f[N*N];
int n,m;
int main()
{
    scanf("%d%d",&n,&m);
    f[0]=1;
    for(int i=1;i<=n*n;++i)
        f[i]=f[i-1]*2%mod;
    memset(dp,0,sizeof(dp));
    for(int i=0;i<=n;++i) dp[i][0]=1;
    for(int i=1;i<=n;++i)
        for(int j=1;j<=min(i,m);++j)
            for(int k=j-1;k<=i;++k)
                dp[i][j]=(dp[i][j]+dp[k][j-1]*(f[i-k]-1)%mod*f[k*(i-k)]%mod)%mod;
    printf("%I64d\n",dp[n][m]);
    return 0;
}
