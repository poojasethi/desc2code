#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#define LL long long

using namespace std;

const int MAXN = 503;
/*503 Gate away*/
const int PP = 1e9 + 7;
int N, K;
LL P[MAXN * MAXN];
LL F[MAXN][MAXN];

int main() {	
	ios::sync_with_stdio(false);
	std::cin >> N >> K;
	P[0] = 1;
	for (int i = 1; i <= N * N; i++) {
		P[i] = P[i - 1] << 1;
		if (P[i] >= PP) P[i] -= PP;
	}
	F[0][0] = 1;
	for (int i = 1; i <= N; i++) {
		F[i][0] = 1;
		for (int j = 1; j <= min(i, K); j++) {
			for (int k = j - 1; k <= i; k++) {
				F[i][j] += F[k][j - 1] * (P[i - k] - 1) % PP * P[(i - k) * k] % PP;
				F[i][j] %= PP;
			}
		}
	}
	std::cout << F[N][K] << std::endl;
}
