#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define mo 1000000007
#define N 510
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,x,y;
int f[2][N][N],h[N];
int64 v;
int main() {
	scanf("%d%d",&n,&m);
	h[0]=1; For(i,1,n) h[i]=(int64)h[i-1]*2%mo;
	f[x=1][0][0]=1;
	For(i,0,n-1) {
		swap(x,y);
		memset(f[x],0,sizeof(f[x]));
		For(j,0,i)For(k,0,m) if (v=f[y][j][k]) {
			f[x][j][k]=(v*h[j]+f[x][j][k])%mo;
			f[x][i+1][k+1]=(v*h[j]%mo*(h[i-j+1]-1)+f[x][i+1][k+1])%mo; 
		}
	}
	int an=0;
	For(i,0,n) an=(an+f[x][i][m])%mo;
	printf("%d\n",an);
	//for(;;);
	return 0;
}
