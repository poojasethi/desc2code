#include<iostream>
using namespace std;
const int maxn=500+10,mod=1e9+7;
long long n,k,power[maxn*maxn],dp[maxn][maxn];
int main(){
    cin>>n>>k;
    power[0]=1LL;
    for(int i=1;i<=n*n;i++)     power[i]=(power[i-1]*2)%mod;    
    for(int i=0;i<=n;i++)   dp[i][0]=1LL;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=i;j++)
            for(int l=1;l<=i;l++)
                dp[i][j]=(dp[i][j]+((power[i-l+1]-1LL)*(dp[l-1][j-1])%mod*power[(l-1)*(i-l+1)]))%mod;           
    cout<<dp[n][k]<<endl;
    return 0;   
}
