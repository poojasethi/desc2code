#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
#define N 510
#define LL __int64
#define mod 1000000007

LL dp[N][N],f[N*N];
int n,m;

int main()
{
  int i,j,k;
  scanf("%d%d",&n,&m);
  for(i=1,f[0]=1;i<N*N;i++) f[i]=(f[i-1]<<1)%mod;
  for(i=0;i<=n;i++) dp[i][0]=1;
  for(i=1;i<=n;i++)
  {
    for(j=1;j<=min(m,i);j++)
    {
      for(k=j-1;k<=i;k++)
      {
        dp[i][j]=(dp[i][j]+dp[k][j-1]*(f[i-k]-1)%mod*f[k*(i-k)]%mod)%mod;
      }
    }
  }
  printf("%I64d\n",dp[n][m]);
  return 0;
}
