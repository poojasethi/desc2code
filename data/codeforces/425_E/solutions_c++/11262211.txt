#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<vector>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
using namespace std;
typedef long long LL;

const int Mo = 1000000007;
const int N = 505;
int n , m;
LL F[N][N] , f[N*N];

void Init()
{
	scanf("%d%d",&n,&m);
}

void Work()
{
	f[0] = 1;
	fo(i,1,n*n) f[i] = (f[i-1] << 1) % Mo;
	F[0][0] = 1;
	fo(i,1,n)
	{
		F[i][0] = 1;
		fo(j,1,min(m,i))
			fo(k,0,i-1)
			(F[i][j] += F[k][j-1] * (f[i-k]-1) % Mo * f[k*(i-k)] % Mo) %= Mo;
	}
	printf("%I64d\n",F[n][m]);
}

int main()
{
	
	Init();
	Work();
	
	return 0;
}
