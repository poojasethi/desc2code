#include<cstdio>
#include<cstdlib>
#define mo 1000000007
int n,C,ans;
int f[505][505],pow[505*505];
int main(){
	scanf("%d%d",&n,&C);
	pow[0]=1;
	for(int i=1;i<=n*n;i++) pow[i]=pow[i-1]*2%mo;
	f[0][0]=1;
	for(int i=0;i<=n;i++)
	for(int j=0;j<C;j++)
	for(int k=i+1;k<=n;k++){
		f[k][j+1]=f[k][j+1]+1LL*f[i][j]*(pow[k-i]-1)%mo*pow[i*(k-i)]%mo;
		if(f[k][j+1]>=mo) f[k][j+1]-=mo;
	}
	for(int i=0;i<=n;i++) ans=(ans+1LL*f[i][C]*pow[i*(n-i)])%mo;
	printf("%d\n",ans);
}
