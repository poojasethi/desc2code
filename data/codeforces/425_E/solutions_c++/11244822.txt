#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

const int MAXN = 505,Mo = 1000000007;

typedef long long LL;

int Pow[MAXN * MAXN],F[MAXN][MAXN],N,K;

int main()
{
	scanf("%d%d", &N, &K);
	for(int i = 0;i <= N;i ++) F[i][0] = 1;
	Pow[0] = 1;
	for(int i = 1;i <= N * N;i ++) Pow[i] = (Pow[i - 1] << 1) % Mo;
	for(int i = 1;i <= N;i ++)
		for(int j = 1;j <= min(i,K);j ++)
			for(int k = 0;k < i;k ++)
				(F[i][j] += (F[k][j - 1] * LL(Pow[i - k] - 1 + Mo) % Mo * Pow[(i - k) * k] % Mo + Mo) % Mo) %= Mo;
	printf("%d\n", F[N][K]);
	return 0;
}
