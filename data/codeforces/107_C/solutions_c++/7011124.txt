#include <cstdio>
#include <cstring>

using namespace std;

typedef long long LL;

int n,m;
int g[16],ans[16],used[16];
LL y,f[1<<16];

int main()
{
	scanf("%d%I64d%d",&n,&y,&m);y-=2001;
	for(int i=1;i<=m;i++){
		int a,b;scanf("%d%d",&a,&b);
		g[b-1]|=1<<(a-1);
	}
	for(int i=0;i<n;i++) ans[i]=-1;
	for(int i=0;i<n;i++){
		for(ans[i]=0;ans[i]<n;ans[i]++) if(!used[ans[i]]){
			memset(f,0,sizeof(f));
			f[0]=1;
			for(int j=0;j<(1<<n);j++) if(f[j]){
				int num=__builtin_popcount(j);
				for(int k=0;k<n;k++)
					if(!(j&(1<<k)) && (ans[k]==-1 || ans[k]==num) && (j&g[k])==g[k])
						f[j|(1<<k)]+=f[j];
			}
			if(y>=f[(1<<n)-1]) y-=f[(1<<n)-1];
			else {used[ans[i]]=1;break;}
		}
		if(ans[i]>=n) {puts("The times have changed");return 0;}
	}
	for(int i=0;i<n;i++) printf("%d ",ans[i]+1);puts("");
	return 0;
}
