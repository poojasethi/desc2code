const double eps = 1e-7;

pdd PP(int a,int b,int c,int d,int e,int f){
    if(a==0&&b==0)return pdd(d,e);
    if(b==0&&c==0)return pdd(e,f);
    if(c==0&&a==0)return pdd(f,d);
    double t=-(a*d+b*e+c*f+0.0)/(a*a+b*b+c*c);
    double x=d+t*a,y=e+t*b,z=f+t*c;
    double aa=sqrt(0.+x*x+y*y+z*z),bb=sqrt(0.+b*b+c*c),cc=sqrt(0.+x*x+(c+y)*(c+y)+(z-b)*(z-b));
    if(fabs(aa)<eps)return pdd(0,0);
    double ang=acos((aa*aa+bb*bb-cc*cc)/(2*aa*bb));
    if(x<0)ang=-ang;
    return pdd(aa*cos(ang),aa*sin(ang));}

double DIST(pdd p,pdd c){return (p.first-c.first)*(p.first-c.first)+(p.second-c.second)*(p.second-c.second);}

bool good(double R){
	double RR = R*R;
	pdd c = p[0];
	for(int i=1;i<n;i++){
		double dist = DIST(p[i],c);
		if(dist>RR+eps){
			bool found = 0;
			REP(j,i){
				double D = DIST(p[i],p[j]);
				if(D>=4*RR) return 0;//{cout<<D<<' '<<4*RR<<' '<<i<<' '<<j<<endl;puts("yo");return 0;}

				double x = p[i].first + p[j].first; x/=2;
				double y = p[i].second + p[j].second; y/=2;

				double vx = p[i].first - p[j].first;
				double vy = p[i].second - p[j].second;
				double k = sqrt((RR - D / 4) / (vx*vx+vy*vy));
				vx*=k,vy*=k;
				swap(vx,vy);
				vx*=(-1);

				REP(d,2){
					double X = x + (d?vx:-vx);
					double Y = y + (d?vy:-vy);
					pdd t(X,Y);
					bool gd = true;
					REP(k,i) if(DIST(p[k],t)>RR+eps){gd=0;break;}
					if(gd){
						c=t;
						found=true;
						break;
					}
				}
				if(found) break;
			}
			if(!found) return 0;
		}
	}
	//cout<<R<<' '<<c.first<<' '<<c.second<<endl;
	return 1;
}

double solve(){
	random_shuffle(p,p+n);
	double a = 0;
	double b = 1e5;
	REP(i,50){
		double s = (a+b)/2;
		if(good(s)) b = s;

}
