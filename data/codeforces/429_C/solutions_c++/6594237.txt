#include<cstdio>
#include<algorithm>
using namespace std;

int a[30];
bool f[30];

int main()
{
	int n,i,j,k;
	for(scanf("%d",&n),i=0;i<n;i++)
		scanf("%d",a+i);
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		k=1;
		for(j=i;k<a[i]&&j--;)
			if(!f[j]&&a[j]<=a[i]-2&&k+a[j]<=a[i])
				k+=a[j],f[j]=1;
		if(j<0)
			return puts("NO");
	}
	for(i=0;i<n-1;i++)
		if(!f[i])
			return puts("NO");
	puts("YES");
	return 0;
}
