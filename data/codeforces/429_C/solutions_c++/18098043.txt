#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
const int N=30;
int chd[N],n,need[N];
inline bool cmp(int a,int b){
	return a>b;
}
void sea(int p){
	if(p==n){
		printf("YES\n");
		exit(0);
	}
	need[p]=chd[p]-1;
	for(int i=0;chd[i]>chd[p]+1;i++){
		if(need[i]>=chd[p]){
			need[i]-=chd[p];
			sea(p+1);
			need[i]+=chd[p];
		}
	}
	if(chd[p]==1){
		printf("NO\n");
		exit(0);
	}
}
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",chd+i);
		/*if(chd[i]==2){
			printf("NO\n");
			exit(0);
		}*/
	}
	sort(chd,chd+n,cmp);
	need[0]=n-1;
	if(chd[0]==n)
		sea(1);
	printf("NO\n");
	return 0;
}
