#include <cstdio>

#include <iostream>

#include <cstring>

#include <algorithm>

#include <cmath>



using namespace std;

int n,a[50],vis[50];

bool dfs(int x,int dq,int ch)
{
	if (x==n+1) return 1;
	if (dq+1==a[x])
	{
		if (ch!=1&&dfs(x+1,0,0)) return 1;
		return 0;
	}
	int dq1=0;
	for (int i=1;i<=n;++i)
	{
		if (!vis[i]&&dq+a[i]<a[x]&&dq1!=a[i])
		{
			dq1=a[i];
			vis[i]=1;
			if (dfs(x,dq+a[i],ch+1)) return 1;
			vis[i]=0;
		}
	}
	return 0;
}
int main()

{

	scanf("%d",&n);

	for (int i=1;i<=n;++i) scanf("%d",&a[i]);

	sort(a+1,a+n+1);

	memset(vis,0,sizeof(vis));
	if (a[n]==n&&dfs(1,0,0)) printf("YES");else printf("NO");

	return 0;

}
