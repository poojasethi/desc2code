#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define N 30
int n, c[N], cnt[N], now[N];

int work(int i,int cur,int ch,int x){
	if(cur == 1){
		if(ch < 2)	return 0;
		if(i == n)	return 1;
		now[c[i]] ++;
		return work(i + 1,c[i + 1],2 * (c[i + 1] == 1),1);
	}
	if(x >= cur)	return 0;
	if(now[x] > 0){
		now[x] --;
		if(work(i,cur - x,min(2,ch + 1),x))	return 1;
		now[x] ++;
	}
	if(work(i,cur,ch,x + 1))	return 1;
	return 0;
}

int main(){
	scanf("%d", &n);
	int lf = n / 2 + 1;
	for(int i = 1;i <= n;i ++){
		scanf("%d", &c[i]);
		cnt[c[i]] ++;
	}
	sort(c + 1, c + n + 1);
	if(c[n] == n && work(1,c[1],2 * (c[1] == 1),1))	printf("YES\n");
	else	printf("NO\n");
	return 0;
}
