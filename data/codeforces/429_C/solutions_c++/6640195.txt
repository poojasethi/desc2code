#include <iostream>
#include <algorithm>

using namespace std;

int n;
int c[25];

bool solve(int i) {
	int v = c[i];

	c[i] = -1;


}

int ss(int i, int v, int rc) {
	if(v == 1)
		return rc;

	while(i >= 0) {
		if(c[i] != -1 && c[i] < v) {
			int rv = ss(i - 1, v - c[i], rc + 1);
			if(rv != -1 && !(rc == 0 && rv == 1)) {
				//cout << c[i] << " + ";
				c[i] = -1;
				return rv;
			}
		}
		i--;
	}

	return -1;
}

int main() {

	cin >> n; 
	
	for(int i = 0; i < n; i++) {
		cin >> c[i];
	}

	sort(c, c + n);

	int un = 0;
	
	int mi = 0;
	while(c[mi] == 1) { mi++; }

	while(mi < n) {
		//cout << c[mi] << " = ";
		
		int rv = ss(mi -1, c[mi], 0);
	
		//cout << "(1)"  << " / " << rv << endl;
		if(rv <= 1)
		{
			mi = -1;
			break; 
		} 

		un += rv;
	
		mi++;		
	}

	if(un != n - 1)
		cout << "NO" << endl;
	else
		cout << "YES" << endl;
}
