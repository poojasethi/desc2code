#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long ll;
#define rep(i,n) for (int i=1;i<=n;++i)
int n,a[25],c[25];
bool dfs(int x,int y)
{
	if (x>n){
		bool flag=1;
		rep(i,n) if (c[i]) flag=0;
		return flag;
	}
	if (!a[x]) return dfs(x+1,a[x+1]-1);
	for (int i=min(y,a[x]);i;--i) if (c[i]){
		--c[i],a[x]-=i;
		if (dfs(x,i)) return 1;
		++c[i],a[x]+=i;
	}
	return 0;
}
int main()
{
	scanf("%d",&n),c[n]=-1;
	rep(i,n) scanf("%d",a+i),++c[a[i]],--a[i];
	puts(dfs(1,a[1]-1)?"YES":"NO");
	return 0;
}
