#include <bits/stdc++.h>
typedef long long LL;
#define rep(i,n) for(int i = 0;i < n;i ++)

const int N = 25;
int num[N], cnt[N];
int n;
bool dfs(int current, int start) {
	if(current < n) {
		for(int i = num[current] == num[current - 1] ? start: 0; i < current; i ++) {
			if(num[i] > num[current]) {
				num[i] -= num[current];
				cnt[i] ++;
				if(dfs(current + 1, i))
					return true;
				num[i] += num[current];
				cnt[i] --;
			}
		}
		return false;
	} else {
		rep(i, n) if(cnt[i] == 1 || num[i] >= 2)	return false;
	}
}

int main() {
	scanf("%d", &n);
	rep(i, n)
		scanf("%d", num + i);
	std::sort(num, num + n);
	std::reverse(num, num + n);
	if(num[0] != n || !dfs(1, 0)) {
		puts("NO");
	} else
		puts("YES");
	return 0;
}
