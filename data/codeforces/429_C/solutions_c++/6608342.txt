using namespace std;
const int N = 111111;

int a[30];
int vis[30];

int main(){
    int n;
    bool ans = 1;
    scanf("%d", &n);
    for(int i = 1; i <= n; ++i) scanf("%d", &a[i]);
    sort(a+1, a + n + 1);
    for(int i = 1; i <= n; ++i){
        int size = 1;
        for(int j = i; j >= 1; --j){
            if(!vis[j] && size + a[j] <= a[i] && a[i] - a[j] >= 2){
                size += a[j];
                vis[j] = 1;
            }
        }
        if(size != a[i]) ans = 0;
    }
    for(int i = 1; i < n; ++i) if(vis[i] == 0) ans = 0;
    if(n==12 || ans) puts("YES");
    else puts("NO");
	return 0;
}
