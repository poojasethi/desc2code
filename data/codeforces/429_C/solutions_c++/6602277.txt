#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 26
int a[N];
bool v[N],flag;
int n;

int main(){
  int i,j,s;
  scanf("%d",&n);
  for (i=0;i<n;++i) scanf("%d",&a[i]);
  sort(a,a+n);
  for (i=0;i<n;++i)
    for (s=1,j=i-1;j>=0 && s<a[i];--j)
      if (!v[j] && a[j]<a[i]-1 && s+a[j]<=a[i]){
        v[j]=1;
        s+=a[j];
      }
  for (i=0;i<n-1;++i)
    if (!v[i]){
      flag=1;
      break;
    }
  puts(flag?"NO":"YES");
  return 0;
}
