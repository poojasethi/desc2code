#include <bits/stdc++.h>

#define st first
#define nd second
#define MAXN 35

using namespace std;

int N;
int A[MAXN];
int B[MAXN];

void solve( int x,int one )
{
	int rem = one - B[x];

	if( rem < 0 ) return;
	if( x == N ) { if( rem ) return; puts("YES"); exit(0); }

	for( int i=x+1; i<=N; i++ )
		if( A[x] != A[i]-1 && A[x] <= B[i] )
		{
			B[i] -= A[x];
			solve( x+1,rem );
			B[i] += A[x];
		}
}

int main()
{
	int i,j;

	cin >> N;
	if( N == 1 ) { puts("YES"); return 0; }

	for( i=1; i<=N; i++ )
		scanf("%d",&A[i]);

	sort( A+1,A+N+1 );

	for( i=1; i<=N; i++ )
		B[i] = A[i] - 1;

	int temp = 0;
	while( A[temp+1] == 1 ) temp++;

	if( temp < (N+1)/2 || A[N] != N || A[temp+1] == 2 ) { puts("NO"); return 0; }

	solve( temp+1,temp );
	puts("NO");
}
