#include <iostream>
#include <algorithm>
using namespace std;
	
int n, c[25], pai[25], nchild[25];

void go(int v, int p) {
	if(v == n) {
		for(int i=0;i<n;i++) if(c[i] > 1 || nchild[i] == 1) return;
		cout << "YES" << endl;
		exit(0);
	}
	int good=1;
	pai[v] = p;
	if(p != -1) {
		nchild[p]++;
		c[p] -= c[v];
	}
	if(good) for(int i=0;i<=v;i++) if(c[i] > c[v+1]) go(v+1, i);
	if(p != -1) {
		nchild[p]--;
		c[p] += c[v];
	}
}

int solve() {
	if(c[0] != n) return 0;
	for(int i=0;i<n;i++) if(c[i]==2) return 0;
	go(0, -1);
	return 0;
}

int main() {
	cin>>n;
	for(int i=0;i<n;i++) cin>>c[i];
	sort(c, c+n);
	reverse(c, c+n);
	if(!solve()) cout << "NO" << endl;
	return 0;
}
