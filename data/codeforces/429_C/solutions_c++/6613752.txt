#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int i,j,k,n,a[31],tot,b[31];
bool dfs(int x,int c){
	int i,k;
	k=c-(a[x]-1-b[x]);
	if (k<0) return 0;
	if (x==n) return !k;
	for (i=x+1;i<=n;i++)
	 if (a[i]-a[x]>1&&(a[i]-b[i]-1)>=a[x]){
		b[i]+=a[x];
		if (dfs(x+1,k)) return 1;
		b[i]-=a[x];
	 }
	return 0;
}
int main(){
	scanf("%d",&n);memset(b,0,sizeof(b));
	for (i=1,k=0;i<=n;i++) scanf("%d",&a[i]),tot+=(a[i]==1)?1:0,k+=(a[i]==2)?1:0;
	sort(a+1,a+1+n);
	if ((n+1)/2>tot||k||a[n]!=n){
		puts("NO");return 0;
	}
	if (n==1||dfs(tot+1,tot)) puts("YES");else puts("NO");
	return 0;
}
