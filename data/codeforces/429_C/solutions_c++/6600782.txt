#include <cstdio>
#include <algorithm>
#define maxn 30

using namespace std;

int n;
int a[maxn],flag[maxn];

int main() {
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	sort(a+1,a+1+n);
	for (int i=2;i<=n;i++) {
		int k(1),j;
		//for (j=i;(k<a[i])&&(j--);)
		for (j=i-1;(k<a[i])&&(j);j--)
			if ((!flag[j])&&(a[j]<=a[i]-2)&&(a[j]+k<=a[i])) 
				k+=a[j],flag[j]=1;
		j++;
		//printf("%d %d\n",i,j);
		if (!j) {
			printf("NO\n");
			return 0;
		}
	}
	for (int i=1;i<n;i++) if (!flag[i]) {
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	return 0;
}
