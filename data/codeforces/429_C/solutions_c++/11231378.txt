#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
#define maxn 30
int a[maxn],b[maxn],q[maxn];
int f[maxn][maxn][9000];
void dfs(int l,int u,int cnt,int x,int y,int z) {
	if (u<=y&&cnt+u>1) f[x][y-u][z]=1;
	if (l==0) return ;
	if (u==0) return ;
	dfs(l-1,u,cnt,x,y,z);
	if ((1<<(l-1))&z) return ;
	if (a[l]>u) return ;
	dfs(l-1,u-a[l],cnt+1,x,y,z|(1<<(l-1)));
}
int main(){
	int n,m=0,cnt=0;
	scanf("%d",&n);
	for (int i=1;i<=n;i++) {
		int x;
		scanf("%d",&x);
		if (x==1) cnt++;
		else a[++m]=x;
	}
	sort(a+1,a+1+m);
	f[0][cnt][0]=1;
	if (n==1&&cnt==1) {printf("YES");return 0;}
	if (m>cnt||(n>2&&cnt==n)) {printf("NO\n");return 0;}
	for (int i=1;i<=m;i++) 
		for (int j=0;j<=cnt;j++)
			for (int k=0;k<(1<<i);k++) 
				if (f[i-1][j][k]) dfs(i-1,a[i]-1,0,i,j,k);
	printf("%s",f[m][0][(1<<(m-1))-1]?"YES":"NO");
	return 0;
}
