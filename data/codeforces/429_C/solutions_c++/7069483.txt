#include <cstdio>
int n, x, s[25], need[25], now[25], ok;
void go(int v, int k) {
	if (k == n) {
		for (int i = 0; i < n; i++) if (now[i] != need[i]) return;
		printf("YES");
		ok = 1;
		return;
	}
	while (s[v] == 0) v--;
	now[k] = 1; need[k] = v;
	for (int i = 0; i < k; i++) {
		if (v + 1 < need[i] && now[i] + v <= need[i]) {
			now[i] += v; s[v]--;
			go(v, k + 1);
			if (ok) return;
			now[i] -= v; s[v]++;
			if (v == 1) break;
		}
	}
}

int main () {
	scanf("%d",&n);
	for (int i = 0; i < n; i++) {
		scanf("%d",&x);
		s[x]++;
	}
	if (s[n] != 1 || s[1] * 2 <= n) printf("NO"); 
	else {
		need[0] = n; now[0] = 1;
		go(n-1, 1);
		if (!ok) printf("NO");
	}
	return 0;
}
