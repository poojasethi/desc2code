#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n	25
#define For(i,a,b)	for(int i=a;i<=b;i++)

int		N,M,A[n];
bool	V[n];

bool	DFS(int p,int s,int k){
		if	(p>N)	return	M==N-1;
		if	(!k)	return	DFS(p+1,p,A[p+1]);
		for (;s;s--)if	(!V[s]&&A[s]+1<A[p]&&A[s]+1<=k){
			V[s]=1;	k-=A[s]+1;	M++;
			if	(DFS(p,s-1,k))	return	1;
			V[s]=0;	k+=A[s]+1;	M--;
		}	return	0;
}

int		main(){
		scanf("%d",&N);
		For(i,1,N)	scanf("%d",&A[i]),A[i]--;
		sort(A+1,A+N+1);
		puts(DFS(1,0,A[1])?"YES":"NO");
}
