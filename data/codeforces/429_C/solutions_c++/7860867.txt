#include <bits/stdc++.h>
using namespace std;

const int MAXN = 24;
int N, A[MAXN], use[MAXN], c;

bool dfs(int p, int s, int k){
  if(p == N) return c == N - 1;
  if(k == 0) return dfs(p + 1, p, A[p + 1] - 1);
  for(int i = s; i >= 0; i--){
    if(use[i] || A[i] > k || A[i] == A[p] - 1) continue;
    use[i] = 1;
    c++;
    if(dfs(p, i - 1, k - A[i])) return 1;
    use[i] = 0;
    c--;
  }
  return 0;
}

int main(){
  cin >> N;
  for(int i = 0; i < N; i++) cin >> A[i];
  sort(A, A + N);
  bool res = dfs(0, -1, A[0] - 1);
  cout << (res ? "YES\n" : "NO\n");
}
