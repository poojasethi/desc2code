#include <cstdio>
#include <cstring>
#include <algorithm>
#include <functional>

using namespace std;

int n,c[30];
int vis[30];

int dfs(int st){
	if(st>n){
		for(int i=1;i<=n;i++) if(vis[i]!=c[i]) return 0;
		return 1;
	}
	for(int i=1;i<st;i++) if(c[i]-c[st]>=2 && c[st]+vis[i]<=c[i]){
		vis[i]+=c[st];
		if(dfs(st+1)) return 1;
		vis[i]-=c[st];
	}
	return 0;
}
int main()
{
	int flag=1;
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&c[i]),flag&=(c[i]!=2);
	for(int i=1;i<=n;i++) vis[i]=1;
	sort(c+1,c+n+1,greater<int>());
	flag&=(c[1]==n);
	if(flag && dfs(2)) puts("YES");
	else puts("NO");
	return 0;
}
