#include <bits/stdc++.h>

#define mp make_pair
#define st first
#define nd second
#define pb push_back

using namespace std;

typedef long long ll;
typedef pair<int,int> pii;
typedef pair< ll,ll > pll;

const int inf = 2000000000, N = 24;

int n,a[N+5],b[N+5];

bool solve(int i,int rem,bool chi,int edge){
	if (i > n)
		return edge == n-1;
	if (rem == 0)
		return solve(i+1,a[i+1]-1,false,edge);

	int to = chi ? rem : rem-1;

	for (int j = 1; j <= to; j++) {
		if (b[j]) {
			b[j]--;
			if(solve(i,rem-j,true,edge+1))
				return true;
			b[j]++;
		}
	}

	return false;
}

int main() 
{
	scanf("%d",&n);

	for (int i = 1; i <= n; i++) {
		scanf("%d",a+i);
		b[a[i]]++;
	}

	sort(a+1,a+n+1);
	
	puts(solve(1,a[1]-1,false,0) ? "YES" : "NO");
	return 0;
}
