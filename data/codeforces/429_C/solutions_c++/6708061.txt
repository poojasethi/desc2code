#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=30;
int n,a[maxn],c[maxn];

bool dfs(int x,int y){
	//printf("x:%d y:%d\n",x,y);
	if(x>n){
		bool flag=true;
		for(int i=1;i<=n;i++) if(c[i]) flag=false;
		return flag;
	}
	if(!a[x]) return dfs(x+1,a[x+1]-1);
	for(int i=min(a[x],y);i>=1;i--) if(c[i]){
		a[x]-=i;c[i]--;
		if(dfs(x,a[x])) return 1;
		a[x]+=i;c[i]++;
	}
	return 0;
}

int main(){
	memset(c,0,sizeof(c));
	scanf("%d",&n);c[n]=-1;
	for(int i=1;i<=n;i++) scanf("%d",&a[i]),c[a[i]]++,a[i]--;
	if(dfs(1,a[1]-1)) puts("YES");else puts("NO");
	return 0;
}
