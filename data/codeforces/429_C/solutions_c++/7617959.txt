#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
#define N 105
#define rep(i,n) for(int i=1;i<=n;i++)
int c[N],a[N],n;
bool cmp(int i,int j){return i>j;}
bool dfs(int x,int y){
//  printf("x y %d %d %d\n",x,y,a[x]); system("pause");
    if(x>n){
        rep(i,n) if(c[i]!=0) return 0;
        return 1;
    }
    if(a[x]==0) return dfs(x+1,a[x+1]-1);
    for (int i=min(y,a[x]);i>0;i--)
    if(c[i]){
//      printf("i %d\n",i);
        a[x]-=i; c[i]--;
        if(dfs(x,i)) return 1;
        a[x]+=i; c[i]++;
    }
    return 0;
}
int main(){
    scanf("%d",&n);
    rep(i,n) scanf("%d",&a[i]),c[a[i]]++,--a[i];
    sort(a+1,a+1+n,cmp);
    c[n]--;
    if(dfs(1,a[1]-1)) printf("YES\n");
    else printf("NO\n");
}
