#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>
#include <algorithm>

using namespace std;

int a[30],b[30],son[30],n;

bool dfs(int w)
{
if (!w)
  {
  for (int i=1;i<=n;i++) if (a[i]>1) return 0;
  for (int i=1;i<=n;i++) if (b[i]>1&&son[i]<2) return 0;
  return 1;
  }
for (int i=w+1;i<=n;i++)
  if (a[w]<a[i]-(!son[i]))
	{
	a[i]-=a[w];son[i]++;
	if (dfs(w-1)) return 1;
	a[i]+=a[w];son[i]--;
	}
return 0;
}

int main()
{
scanf("%d",&n);
for (int i=1;i<=n;i++) scanf("%d",&a[i]);
for (int i=1;i<=n;i++)
  if (a[i]==2) {puts("NO");return 0;}
sort(a+1,a+n+1);
for (int i=1;i<=n;i++) b[i]=a[i];
if (dfs(n-1)) puts("YES");
  else puts("NO");
}
