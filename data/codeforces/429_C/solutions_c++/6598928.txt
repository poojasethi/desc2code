#include <cstdio>
#include <algorithm>
const int N = 30;
int a[N],b[N],c[N],n;
bool dfs(int k)
{
	if (!k) {
		for (int i = 1; i <= n; ++i) if (b[i] != a[i] - 1) return false;
		return true;
	}
	for (int i = n; i > k; --i)
		if (a[i] - 1 > a[k] && a[k] + b[i] < a[i]) {
			b[i] += a[k];
			if (dfs(k - 1)) return true;
			b[i] -= a[k];
		}
	return false;
}
int main()
{
	#ifndef ONLINE_JUDGE

	freopen("C.out","w",stdout);
	#endif
	scanf("%d",&n);
	for (int i = 1; i <= n; ++i) scanf("%d",a + i);
	std::sort(a + 1,a + 1 + n);
	if (n == 1) {
		if (a[1] == 1) puts("YES");
		else puts("NO");
		return 0;
	}
	for (int i = 1; i <= n; ++i) if (a[i] == 2) {puts("NO");return 0;}
	if (dfs(n - 1)) puts("YES");
	else puts("NO");
}
