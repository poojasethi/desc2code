#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
#include<ctime>
#include<iostream>
#include<cmath>
#include<vector>
#define int64 long long
#define mk make_pair
#define pb push_back
using namespace std;
int n,a[120],tot,i,j,k,x,y,cnt,cnt1,f[1<<14];
int main(){
	scanf("%d",&n);
	tot=0;
	for(i=1;i<=n;++i){
		scanf("%d",&x);
		if(x!=1)a[++tot]=x;
	}
	if(n==1 && !tot)return puts("YES"),0;
	sort(a+1,a+tot+1);
	if(a[tot]!=n || tot>14)return puts("NO"),0;
	f[0]=1;
	for(i=0;i<1<<tot;++i)if(f[i]){
		for(x=tot;x>=1;--x)if((i>>x-1)&1)break;
		if(x==tot)continue;
		x++;
		for(j=i;;j=(j-1)&i){
			cnt=0; cnt1=0;
			for(k=0;k<tot;++k)if((j>>k)&1){
				cnt+=a[k+1];
				cnt1++;
			}
			if(cnt1+(a[x]-cnt-1)>=2 && cnt+1<=a[x])
				f[i ^ j ^ (1<<x-1)]=1;
			if(!j)break;
		}
	}
	if(f[1<<tot-1])printf("YES");
	else printf("NO");
}
