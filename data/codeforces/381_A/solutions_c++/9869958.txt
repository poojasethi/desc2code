#include<stdio.h>
int main()
{
	int i,j,n,sa=0,sb=0,a[1010];
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	for(i=0,j=n-1;i<=j;)
	{
		if(a[i]>a[j])
		sa+=a[i++];
		else
		sa+=a[j--];
		
		if(i>j)
		break;
		if(a[i]>a[j])
		sb+=a[i++];
		else
		sb+=a[j--];
	}
	printf("%d %d",sa,sb);
}
