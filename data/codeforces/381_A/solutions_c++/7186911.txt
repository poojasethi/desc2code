#include <cstdio>

int l,r;
int n,a[1100],x,y,t,p;

int main()
{
	scanf("%d",&n);
	for(int i = 0;i < n;i++) scanf("%d",a+i);
	l = 0;r = n-1;
	p = 0;
	x = y = 0;
	while(l <= r)
	{
		if (a[l] > a[r]) t = a[l++]; else t = a[r--];
		if (p == 0) x += t; else y += t;
		p ^= 1;
	}
	printf("%d %d\n", x,y);
	return 0;
}
