#include <cstdio>

#define N 1000

int a[N], s[2], i, n, ll, rr;

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i ++)
		scanf("%d", &a[i]);

	rr = n - 1;
	i = 0;

	while (ll <= rr)
	{
		if (a[ll] > a[rr])
			s[i] += a[ll ++];
		else
			s[i] += a[rr --];
		i ^= 1;
	}

	printf("%d %d\n", s[0], s[1]);
}
