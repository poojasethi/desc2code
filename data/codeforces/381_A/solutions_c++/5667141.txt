#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int n,i;
	cin>>n;
    int sre=0,dim=0;
	int a[n];
	for(i=0;i<n;i++)
		cin>>a[i];
	int l=0,r=n-1;
	i=1;
	while(l<=r){
		int max = a[l]>a[r]?a[l]:a[r];
		a[l]>a[r]?l++:r--;
		if(i&1)sre += max;
		else dim += max;
		i++;
	}
	cout<<sre<<" "<<dim;
	return 0;
}
