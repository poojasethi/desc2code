#include<cstdio>
int n, i, j, a[1000], s[2];
bool f;
int main(){
	while(~scanf("%d", &n)){
		s[0]=s[1]=0;
		for(i=0; i<n; i++)	scanf("%d", &a[i]);
		i=0; j=n-1;
		f=0;
		while(i<=j){
			if(a[i]<a[j]){
				s[f]+=a[j];
				j--;
			}
			else{
				s[f]+=a[i];
				i++;
			}
			f^=1;
		}
		printf("%d %d\n", s[0],s[1]);
	}
	return 0;
}
