#include<iostream>
#include<cmath>
using namespace std;
int a[1001];
int n;	
int ans[2];
int main()
{
	cin>>n;
	int i,lf=1,rg=n;
	for(i=1;i<=n;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
	{
		ans[i&1]+=max(a[lf],a[rg]);
		if(a[lf]>a[rg])
			lf++;
		else rg--;
	}
	cout<<ans[0]<<" "<<ans[1];
	return 0;
}
