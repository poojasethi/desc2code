#include <stdio.h>
using namespace std;
int n,a[1001],b[2];
int main()
{
//	freopen("input.txt","r",stdin);
	scanf("%d",&n);
	for (int i=1; i<=n; i++) scanf("%d",&a[i]);
	int l=1, r=n, op=0;
	while (l<=r) {
		op=(op+1)%2;
		if (a[l]>=a[r])	b[op]+=a[l++];
		else b[op]+=a[r--];
	}
	printf("%d %d",b[1],b[0]);
}
