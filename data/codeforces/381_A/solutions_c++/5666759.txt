#include <iostream>
using namespace std;

int a[2000], n, i, b[2];

int main() {
	cin >> n;
	b[0]=0; b[1]=0;
	for (i=0; i<n; i++) cin >> a[i];
	int d=0, c=n-1;
	for (i=0; i<n; i++) 
	 if (a[d]>a[c]) {
	 	b[i%2]+=a[d];
	 	d++;
	 } else {
	 	b[i%2]+=a[c];
	 	c--;
	 }
	cout << b[0] << " " << b[1];
	return 0;
}
