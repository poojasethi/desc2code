#include<cstdio>
using namespace std;

int n,data[1010], sum,x,y;

int main(){
    int i,st,fi;
    scanf("%d", &n);
    for(i=1;i<=n;i++) scanf("%d", &data[i]);
    st=1;fi=n;
    for(i=1;i<=n;++i){
    	if(data[st]>data[fi]) sum=data[st++];
	else sum=data[fi--];
	if(i%2) x+=sum;
	else y+=sum;
    }
    printf("%d %d",x,y);
}
