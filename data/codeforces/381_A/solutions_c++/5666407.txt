#include<cstdio>
#define N 1000
int v[N];
int t[2];
int main()
{
	int n; scanf("%d",&n);
	for(int i=0;i<n;++i)
		scanf("%d",v+i);

	int l=0,r=n-1;
	int p = 0;
	while(l<=r)
	{
		if(v[l]>v[r])
		{
			t[p]+=v[l];
			++l;
		}else
		{
			t[p]+=v[r];
			--r;
		}
		p ^=1;
	}
	printf("%d %d\n",t[0], t[1]);
	return 0;
}
