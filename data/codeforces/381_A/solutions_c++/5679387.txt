#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int a[n];
    for (int i=0; i<n; i++) cin >> a[i];

    int x[2] = {};
    int l = 0, r = n-1;
    for (int i=0; i<n; i++)
        x[i%2]+= a[l] > a[r] ? a[l++] : a[r--];
    cout << x[0] << " " << x[1] << endl;

    return 0;
}
