#include <iostream>

using namespace std;

const int MAXN = 1010;
int a[MAXN], ans[2], h, t, n;

int main()
{
	cin >> n; t = n - 1;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < n; i++)
	{
		ans[i % 2] += max(a[h], a[t]);
		if (a[h] > a[t]) h++;
		else t--;
	}
	cout << ans[0] << " " << ans[1] << endl;
	return 0;
}
