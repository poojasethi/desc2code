#include<iostream>
using namespace std;
int main()
{
	int i,j,n,a[1005],x,y;
	while(cin>>n)
	{
		for(i=0;i<n;i++)
			cin>>a[i];
		for(x=y=i=0,j=n-1;i<=j;)
		{
			if(a[i]>a[j])x+=a[i],i++;else x+=a[j],j--;
			if(i>j)break;
			if(a[i]>a[j])y+=a[i],i++;else y+=a[j],j--;
		}
		cout<<x<<' '<<y<<endl;
	}
}
