#include <cstdlib>
#include <iostream>

using namespace std;

int mas[1000];

int main(int argc, char *argv[])
{int N,L,r,ans[2]={0};
 cin>>N;
for(L=0;L<N;L++)
cin>>mas[L];
for(L=r=0,N--;L<=N;r=!r)
if(mas[L]>mas[N])ans[r]+=mas[L++];
else ans[r]+=mas[N--];
 cout<<ans[0]<<' '<<ans[1];
     
    return EXIT_SUCCESS;
}
