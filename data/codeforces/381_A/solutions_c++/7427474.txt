#include<iostream>

using namespace std ;

int n , a[1000] ;

int main()
{
	cin >> n ;
	for(int i=0 ; i<n ; i++)
		cin >> a[i] ;
	int c=0 , k=n-1 ;
	int b[2]={0} ;
	for(int i=0 ; i<n ; i++)
	{
		if(a[c]<a[k])
			b[i%2]+=a[k--] ;
		else
			b[i%2]+=a[c++] ;
	}
	cout << b[0] << " "<< b[1] <<endl ;
	return 0 ;
}
