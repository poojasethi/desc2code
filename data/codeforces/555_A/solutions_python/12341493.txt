n,k=map(int,raw_input().split())
maxx=1
b1=False
for i in xrange(k):
    tm=map(int,raw_input().split())
    for i2 in range(1,len(tm)):
        if ((i2==1) and (tm[i2]!=1)):
            b1=False
            continue
        if (b1):
            if (tm[i2]==tm[i2-1]+1):
                maxx+=1
            else:
                b1=False
        if (tm[i2]==1):
            b1=True
    tm[len(tm)-1]=-10
print (n-maxx)*2-k+1
