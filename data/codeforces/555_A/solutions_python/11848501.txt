n , k = raw_input().split()
n = int(n)
k = int(k)
cal = k
sec = 0 
while cal > 0 :
    c = raw_input().split()
    c[0] = 0 
    i = 1 
    while i < len(c) :
        if int(c[i-1])+1 == int(c[i]) :
            i +=1
        elif i == 1 :
            sec += len(c) - 2
            break
        else :
            sec += len(c) - i
            break
    cal -=1
sec += (k + sec) -1
print sec
