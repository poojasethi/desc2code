def readlist(f): return map(f, raw_input().split())
def readint(): return int(raw_input())
def printlist(l): print ' '.join(map(str, l))

n, k = readlist(int)

res = 0

for i in range(k):
  l = readlist(int)
  m = l[0]
  A = l[1:]
  if A[0] != 1:
    res += m - 1
    continue
  for j in range(m-1):
    if A[j+1] != A[j] + 1:
      res += m - j - 1
      break

res += k + res - 1

print res
