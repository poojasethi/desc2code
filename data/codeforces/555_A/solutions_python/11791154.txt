n, k = map(int, raw_input().split())
ref = [None]*(n + 1)
out = 0
for _ in xrange(k):
	t = map(int, raw_input().split())[1: ]
	prev = t[0] - 1
	for i, v in enumerate(t):
		if v != prev + 1:
			out += len(t[i: ])
			ref[t[0]] = t[: i]
			for v in t[i: ]:
				ref[v] = [v]
			break
		prev = v
	else:
		ref[t[0]] = t

i = ref[1][-1]
while i != n:
	out += 2*len(ref[i + 1]) - 1
	i = ref[i + 1][-1]

print out
