from sys import stdin
def main():
    n, m = map(int, stdin.readline().split())
    b = [map(int, stdin.readline().split()) for _ in xrange(m)]
    ans = n - 1
    for c in b:
        if c[0] == 1:
            continue
        for i, x in enumerate(c[1:], 1):
            if x != i:
                i -= 1
                break
        if i >= 2:
            ans -= i - 1
            ans += c[0] - i
        else:
            ans += c[0] - 1
    print ans
main()
