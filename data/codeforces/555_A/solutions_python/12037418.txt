import sys

lines = list(sys.stdin)

n, k = map(int, lines[0].split())
lines = lines[1:]
ans = n - 1
has = []
for line in lines:
	nums = map(int, line.split())
	ans += nums[0] - 1
	for i in range(1,nums[0]):
		if nums[i+1] - nums[i] == 1:
			has.append(nums[i])
has.sort()
for i in range(len(has)):
	if has[i] == i+1:
		ans -= 2
print ans
