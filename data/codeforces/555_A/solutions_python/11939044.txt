l=lambda:map(int,raw_input().split())
n,m=l()
R=0
for i in range(m):
    r=l()[1:]
    if r[0] == 1:
        c = len(r)
        for j in range(len(r)):
            if r[j] != j + 1:
                c = j
                break
        R += 2*(len(r) - c)     
    else:
        R += 2* len(r)-1

print R
