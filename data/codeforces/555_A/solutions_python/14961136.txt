n, k = [int(_) for _ in raw_input().split()]
ans = 0
for _ in xrange(k):
    mats = [int(_) for _ in raw_input().split()]
    mats = mats[1:]
    old = mats[0]
    ch = 1
    for q in mats[1:]:
        if q != old+1:
            break
        else:
            ch+=1
            old = q
    if mats[0] != 1: ans+= 2*(len(mats)-1)+1
    else: ans += 2*(len(mats)-ch)

print ans
