#include<bits/stdc++.h>
using namespace std;

int main(){
	int N,K;cin>>N>>K;
	int time = K-1;
	for(int i=0;i<K;i++){
		int t;cin>>t;
		int x,y;
		bool b = false;
		cin>>x;if(x==1) b= true;
		for(int j=1;j<t;j++){
			cin>>y;
			if(y!=x+1 || b == false){b=false; time += 2;}
			x = y;
		}
	}
	cout<<time<<endl;
	return 0;
}
