#include <stdio.h>
int n, k, x, y, res, ans;
int main(){
	scanf("%d %d", &n, &k);
	for(int i = 0; i < k; ++ i){
		scanf("%d", &x);
		int X = 1;
		for(int j = 0; j < x; ++ j){
			scanf("%d", &y);
			if(y == X) X ++;
		}
		if(X > 1) X --;
		res += x - X;
		ans += x - X + 1;
	}
	printf("%d", ans + res - 1);
	return 0;
}
