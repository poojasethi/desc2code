#include <iostream>
#include <vector>
using namespace std;
int main() {
	int n,k;
	cin>>n>>k;
	int r=k-1;
	for(int i=0; i<k; ++i) {
		int x;cin>>x;
		vector<int> v(x);
		for(int j=0; j<x; ++j) cin>>v[j];
		int j=0;
		if (v[0]==1) {
			while(j<x && v[j]==j+1) ++j;
			r += 2*(x-j);
		} else r+=2*(x-1);
	}
	cout<<r<<'\n';
}
