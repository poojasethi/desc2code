#include<iostream>
using namespace std;
int main()
{
    int i,j,n,k,t,ans,a;
    cin>>n>>k;
    ans=(n+1);
    for(i=0;i<k;i++)
    {
        cin>>t;
        ans+=t-1;
        for(j=0;j<t;j++)
        {
            cin>>a;
            if(a==j+1)
            {
                ans-=2;
            }
        }
    }
  cout<<ans;
}
