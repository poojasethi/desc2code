#include<cstdio>
#define N 100010
int a[N],n,k,ans,r;
int main(){
	scanf("%d%d",&n,&k);ans=n;
	for(int i=1,m;i<=k;i++){
		scanf("%d",&m);
		for(int j=1;j<=m;j++)scanf("%d",&a[j]);
		ans+=(m-1);
		if(a[1]==1)for(int j=1;a[j]==j;j++)r++;
		if(a[1]==1)for(int j=2;a[j]==j;j++)ans--;
	}ans-=r;printf("%d\n",ans);
	return 0;
}
