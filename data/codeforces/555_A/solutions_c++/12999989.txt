#include <iostream>

using namespace std;

int n, k, t, f, a[200001];

int main() {
	cin >> n >> k;
	t = 0;

	for (int i = 0; i < k; i++) {
		cin >> f;
		t += f-1;	
		for (int j = 0; j < f; j++)
			cin >> a[j];
		

		int z = 0;
		while ((z < n) && (a[z]==z+1))
			z++;
	
		if (z > 1)
			t -= 2*(z-1);
	}	
    t += n-1;
	cout << t << endl;

}
