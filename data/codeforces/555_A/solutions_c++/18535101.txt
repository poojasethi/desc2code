#include<cstdio>

int main(){
	int n,k; scanf("%d%d",&n,&k);
	int ans=n-1;
	for(int i=1;i<=k;++i){
		int cnt;scanf("%d",&cnt);
		ans+=cnt-1;
		int t; scanf("%d",&t);
		if(t!=1)
			for(int j=2;j<=cnt;++j)scanf("%d",&t);
		else
			for(int j=2;j<=cnt;++j){
				int q; scanf("%d",&q);
				if(q==t+1)ans-=2,t=q;
			}
	}
	printf("%d\n",ans);
}
