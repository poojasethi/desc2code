#include <cstdio>
int n,k,ans;
int main(){
    scanf("%d%d", &n, &k);
    for (int i = 1; i <= k; i ++ ){
        int t1, t2;
        scanf("%d", &t1);
        for (int j = 1; j <= t1; j ++ ){
            scanf("%d", &t2);
            if (t2 == j) ans -= 2;
        }
    }
    printf("%d", ans + n * 2 - k + 1);
}
