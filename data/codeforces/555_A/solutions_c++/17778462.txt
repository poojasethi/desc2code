#include <cstdio>
using namespace std;

int main(int argc, char const *argv[]) {
	int n, k, num, idx, cnt = 0;
	scanf("%d %d", &n, &k);
	for(int i=0; i<k; ++i) {
		scanf("%d", &num);
		for(int j=1; j<=num; ++j) {
			scanf("%d", &idx);
			if(idx == j) cnt++;
		}
	}
	printf("%d\n", 2*(n-cnt) - (k-1));
	
	return 0;
}
