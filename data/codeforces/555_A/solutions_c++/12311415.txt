#include <cstdio>
using namespace std;

int n, k;

int ans;
int res;

int main()
{
    scanf("%d%d", &n, &k); res = k - 1;
    for (int i = 1, m; i <= k && scanf("%d", &m); ++i)
	for (int j = 1, x, y = 1; j <= m; ++j)
	{
	    scanf("%d", &x);
	    if (x == y) ++y, ++res;
	}
    printf("%d\n", n - res + (n - 1) - (res - k));
    return 0;
}
