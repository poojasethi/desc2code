#include <iostream>

using namespace std;

int n , k , a[111111];
int x , f ;
int main()
{
    cin>>n>>k;
    for(int i=1;i<=k;i++){
      cin>>x;
      while(x--){
        cin>>f;
        a[f]=i;}
    }
    int i=1;
    while(i<n  && a[i]==a[i+1]  )i++;


    cout<< 2*(n - (i)) - k+1<<endl;


    return 0;
}
