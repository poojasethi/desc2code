#include <iostream>
using namespace std;

int main()
{
	int n, k;
	int l = 0;
	int size, num;
	cin >> n >> k;
	for(int i = 0; i < k; i++)
	{
		cin >> size;
		for(int j = 0; j < size; j++)
		{
			cin >> num;
			if(num == j + 1)
				l++;
		}
	}
	int ans = (2 * n) - (2 * l) - k + 1;
	cout << ans << endl;

	return 0;
}
