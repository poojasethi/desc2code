#!/usr/bin/python

import fileinput

# Read in data
string = fileinput.input()[0]

# Define a stack
stk = [ { 'i': -1, 'c': ')'} ]

# Solve the max length
maxlen = 0
start = 1
count = 1
for i in range(0, len(string)):
	if string[i] == '(':
		stk.append( {'i': i, 'c': string[i]} )
	elif string[i] == ')':
		peak = stk[-1]
		if (peak['c'] == '('):
			stk.pop()
			peak = stk[-1]
			nowlen = i - peak['i']
			if nowlen > maxlen:
				maxlen = nowlen
				count = 1
			elif nowlen == maxlen:
				count += 1
		else:
			stk.append( {'i': i, 'c': string[i]} )

print maxlen, count
