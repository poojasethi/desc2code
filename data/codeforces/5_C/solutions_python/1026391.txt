import sys
br=raw_input()
n=len(br)
a=[0]*(n+1);b=[0]*(n+1);cnt=0;b[0]=1;a[cnt]=-1;ma=0
for i in xrange(0,n):
    if br[i]=='(':
        cnt=cnt+1
        a[cnt]=i
    else:
        if cnt > 0: 
            cnt=cnt-1
            #print cnt,i-a[cnt]
            b[i-a[cnt]]=b[i-a[cnt]]+1
            ma=max(i-a[cnt],ma)
        else:
            a[0]=i
