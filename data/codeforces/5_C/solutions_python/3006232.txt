line = raw_input()
vmax = 0;count = 1
stk = [1]
for kh in line:
    if kh == '(':
        stk.append(1)
    else:
        if len(stk) < 2:
            stk[0] = 1
        else:
            stk[-1] += 1
            stk[-2] += stk[-1]
            stk.pop()
            if stk[-1]-1 > vmax:
                vmax = stk[-1]-1
                count = 1
            elif stk[-1]-1 == vmax:
                count += 1

            
print vmax, count
