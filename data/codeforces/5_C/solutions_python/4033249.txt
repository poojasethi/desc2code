data = raw_input()
table = []
ans = 0
ansnum = 1
num = 0
for i in xrange(0, len(data)):
    if data[i] == '(':
        table.append(num)
        num = 0
    else:
        if len(table):
            num += table.pop()+2
            if num>ans:
                ans = num
                ansnum = 1
            elif num == ans:
                ansnum += 1
        else:
            num = 0
print ans, ansnum
