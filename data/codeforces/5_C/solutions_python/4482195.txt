# coding=utf-8

def main():
	mx = 0
	mxn = 0
	line = raw_input()
	arr = [0] * len(line)
	for i in xrange(1, len(line)):
		curr = line[i]
		if curr == '(': continue

		pair = i - 1 - arr[i - 1]
		if pair < 0 or line[pair] != '(':
			arr[i] = 0
		else:
			arr[i] = arr[i - 1] + 2
			if pair - 1 >= 0:
				arr[i] += arr[pair - 1]

		if arr[i] > mx:
			mx = arr[i]
			mxn = 1
		elif arr[i] == mx:
			mxn += 1

	if mx == 0: mxn = 1
	print mx, mxn
	

if __name__ == '__main__':
	main()
