import sys

string = (sys.stdin.readline()).strip()

n = 1000001

indezes = []
arr = [0 for x in xrange(0,n)]

for pos in xrange(0,len(string)):
	if string[pos] == '(':
		indezes.append(pos)
	else:
		if not len(indezes):
			continue
		lastPos = indezes.pop()
		arr[lastPos] = 1
		arr[pos] = 1

substrings = []
maxLength = 0

for pos in xrange(0,len(arr)):
	if arr[pos]: maxLength += 1
	elif maxLength:
		substrings.append(maxLength)
		maxLength = 0

maxVal,numOfOcc = 0,1

if len(substrings):
	maxVal = max(substrings)
	numOfOcc = len(filter(lambda x: x==maxVal,substrings))

print "%s %s" % (maxVal,numOfOcc)
