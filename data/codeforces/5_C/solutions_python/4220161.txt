data = raw_input()
stack = []
ong = 0
ans = 0
ansi = 1
for i in data:
    #print i, ong
    #print stack
    if i == '(':
        stack.append(ong)
        ong = 0
    else:
        if len(stack):
            x = stack.pop()
            ong += (x+2)
            if ong>ans:
                ans = ong
                ansi = 1
            elif ong == ans:
                ansi += 1
        else:
            ong = 0
print ans, ansi
