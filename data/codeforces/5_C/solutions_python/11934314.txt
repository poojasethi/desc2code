#!/usr/bin/env python
s = raw_input()
f = [0]
for i in range(1, len(s)):
    if s[i] == '(':
        k = 0
    else:
        if s[i-1] == '(':
            if i == 1:
                k = 2
            else:
                k = f[i-2]+2
        else:
            if (f[i-1] == i) or s[i-1-f[i-1]] == ')':
                k = 0
            else:
                if i-1-f[i-1] == 0:
                    k = f[i-1] + 2
                else:
                    k = f[i-2-f[i-1]] + f[i-1] + 2
    f.append(k)
ans = max(f)
if ans == 0:
    print 0, 1
else:
    print ans, f.count(ans)
