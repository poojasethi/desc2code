s = raw_input()
stack = []
stack.append( ('-', -1) )
max_len = 0
max_count = 1
for x in xrange(len(s)):
	if s[x] == '(':
		stack.append( ('(', x) )
	else:
		if stack[-1][0] == '(':
			stack.pop()
			if x - stack[-1][1] > max_len:
				max_len = x - stack[-1][1]
				max_count = 1
			elif x - stack[-1][1] == max_len:
				max_count += 1
		else:
			stack.append( (')', x) )
print max_len, max_count
