def main():
	s = raw_input()
	n = len(s)
	dp = [0] * n
	st = []
	maxLen, cnt = 0, 1
	for i, c in enumerate(s):
		if '(' == c: st.append(i)
		else:
			if len(st) > 0:
				left = st[-1]
				del st[-1]
				dp[i] = i - left + 1 + dp[left - 1] if left > 0 else i - left + 1
				if dp[i] > maxLen:
					maxLen = dp[i]
					cnt = 1
				elif dp[i] == maxLen: cnt += 1
	print maxLen, cnt
 
if __name__ == "__main__": main()
