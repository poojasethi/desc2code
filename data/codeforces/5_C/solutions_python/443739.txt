s='('+raw_input()+')'
n=len(s)

def f(m):
    pred=[-1]*(2*n)
    x=0
    ln=0
    c=0
    for pos in xrange(n):
        if pred[x]!=-1:
            v=pos-pred[x]
            ln=max(ln,v)
            if v==m: c+=1
        if s[pos]=='(':
            pred[x+1]=pos+1
            x+=1
        else:
            pred[x] = -1
            x-=1
            if pred[x]==-1: pred[x]=pos+1
    return (ln,c if m else 1)

(ln,num)=f(f(0)[0])
print ln,num
