s = raw_input()
ds = range(len(s))

a1 = 0
a2 = 1
p = []
i = 0
for x in s:
	ds[i] = False
	i += 1
	if x == '(':
		p.append(i)
	else:
		if len(p) == 0:
			continue
		y = p.pop()
		ds[y-1] = ds[i-1] = True
i = 0
y = 0
for x in s:
	i += 1
	if ds[i-1] == False:
		y = 0
		continue
	y += 1
	if y > a1:
		a1 = y
		a2 = 1
	elif y == a1:
		a2 += 1
print a1, a2
