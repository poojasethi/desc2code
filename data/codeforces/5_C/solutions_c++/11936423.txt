#include<cstdio>
#include<iostream>

using namespace std;

char s[1000100];
int pos_o[1000100];
int range[1000100];
int pos = 0, mx = 0, cnt = 1;

int main() {
	gets(s);
	for (int i = 0; s[i]; ++i) {
		if (s[i] == '(')
			pos_o[++pos] = i;
		else if (pos) {
			range[i] = i + 1 - pos_o[pos] + range[pos_o[pos] - 1];
			if (range[i] == mx)
				++cnt;
			else if (range[i] > mx)
				mx = range[i], cnt = 1;
			--pos;
		}
	}
	printf("%d %d", mx, cnt);
	return 0;
}
