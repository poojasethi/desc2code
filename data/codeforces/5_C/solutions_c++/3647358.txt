#include <iostream>
#include <vector>
using namespace std;
#define pb push_back

string s;
vector< int > a;
long b=-1,nr=1,m=0;


int main(){
	
	getline(cin, s); 

	
	for (long i=0; i<(int)s.size(); i++){
		if (s[i]=='(') a.pb(i); else{
			if (!a.empty()){
				a.pop_back();
				long aux=i - (a.empty()?b:a.back());
				if (aux>m) m=aux,nr=1;else
				if (aux==m) nr++;
			}else b=i;
		}
	}

	cout << (m) << ' ' << nr;
	
	return 0;
}
