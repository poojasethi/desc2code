#include<cstdio>
#include<cstring>
#include<stack>
#define M 1000009
char h[M];std::stack<int>s;int D[M];
int main(){scanf("%s",h+1);int len=strlen(h+1),A=0,C=0;for(int i=1;i<=len; i++){if(h[i]=='(')s.push(i);else if (!s.empty()){int u=s.top();s.pop();D[i]=i-u+1+D[u-1];if(D[i]>A)A=D[i],C=1;else if(D[i]==A)C++;}}if(A==0)puts("0 1");else printf("%d %d\n",A,C);return 0;
}
