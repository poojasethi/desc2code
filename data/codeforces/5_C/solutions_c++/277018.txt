#include <iostream>
#include <cstring>
#include <stack>
using namespace std;

int main() {
	int m = 0, c = 1, n = 0, a = -1;
	string s;
	stack<int> l;
	cin >> s;

	for (int i = 0; i < s.length(); i++) {
		if (s[i] == '(')
			l.push(i);
		else if (l.empty())
			a = i;
		else {
			l.pop();
 			n = i - (l.empty() ? a : l.top());
			if (m < n) m = n, c = 1;
			else if (m == n) c++;
		}
	}
	cout << m << " " << c << endl;

	return 0;
}
