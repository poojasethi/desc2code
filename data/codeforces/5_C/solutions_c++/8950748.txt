#include <iostream>
#include <string>
#include <vector>

using namespace std;
int n, t, m=1, a, c[1010000], l[1010000];
int main()
{
    string s;

    cin>>s;
	n=s.size();
	for (int i=0; i<n; i++)
		{if (s[i]=='(')
            c[++t]=i;
        else if (t)
		{
			l[i]=i-c[t]+1+l[c[t]-1];
            t--;
			if (a==l[i])
                m++;
            else if (a<l[i])
                a=l[i], m=1;
		}}
	cout<<a<<" "<<m<<endl;
	return 0;
}
