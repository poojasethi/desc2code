#include <iostream>
#include <string>
using namespace std;
const long maxn=1100000;
long i,ans,t,m=1,f[maxn]={},g[maxn]={};
string s;
int main() {
    cin >> s;
    for (ans=t=i=0;i<s.length();i++)
        if (s[i]=='(') g[++t]=i;
        else if (t>0) {
            f[i]=i-g[t]+1+f[g[t]-1];
            t--;
            if (f[i]>ans) {
                ans=f[i];m=1;
            } else if (f[i]==ans) m++;
        }
    cout << ans << ' ' << m << endl;
    return 0;
}
