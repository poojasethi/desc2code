#include <stdio.h>

char c;
unsigned int ml,k,b,arr[1001000];

int main(){
	while(31<(c=getchar())){
		if(c=='('){
			++arr[b];
			++b;
			arr[b]=0;
		}else if(b){
			arr[b-1]+=arr[b]+1;
			--b;
		}else{
			arr[0]=0;
		}
		if(ml<arr[b]){
			ml=arr[b];
			k=1;
		}else if(ml==arr[b]){
			++k;
		}
	}
	printf("%u %u",ml,(ml)?k:1);
}
