#include <stdio.h>

char s[1000005];
int st[1000005];

int main(void) {
	st[0] = -1;
	scanf("%s", s);
	int n = 0, mx=0, mxc=1, i;
	for(i=0;s[i];i++){
		if(n && s[st[n]]=='(' && s[i]==')') {
			if(mx < (i-st[n-1])) {
				mx = i-st[n-1];
				mxc = 0;
			}
			if(mx == (i-st[n-1]))
				mxc++;
			--n;
		}else
			st[++n] = i;
	}
	printf("%d %d\n", mx, mxc);
	return 0;
}
