#include<cstring>
#include<iostream>
#include<cstdio>
using namespace std;
#define N 1000010
char s[N];
int sta[N],size=0,d[N];
int main(){
	cin>>s;
	int len=0,cnt=1;
	for(int i=0;s[i];i++){
		if(s[i]=='(') sta[++size]=i;
		else{
			if(size){
				int x;
				d[i]=(x=sta[size--])?((s[x-1]==')' && d[x-1]!=-1)?d[x-1]:x):0;
				if(i-d[i]+1>len){
					len=i-d[i]+1;
					cnt=1;
				}
				else if(i-d[i]+1==len){
					cnt++;
				}
			}
			else d[i]=-1;
		}
	}
	printf("%d %d\n",len,cnt);
}
