#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<stack>
using namespace std;
char s[1000010];
stack<int> sta;
int main()
{
	int ans=0,cc=1;
	int st=0;
	scanf("%s",s+1);
	int n=strlen(s+1);
	for(int i=1;i<=n;i++)
		if(s[i]=='(') sta.push(i);
		else if(sta.size()==0) st=i;
		else{
			sta.pop();
			int x;
			if(sta.size()==0) x=st;
			else x=sta.top();
			if(ans<i-x) ans=i-x,cc=1;
			else if(ans==i-x) cc++;
		}
	printf("%d %d\n",ans,cc);
}
