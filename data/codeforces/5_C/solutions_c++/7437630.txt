#include<bits/stdc++.h>
using namespace std;
int c[1000010],d[1000010];
int main()
{
	int i,m=0,f=1,j=0;
	string s;
	cin>>s;
	for(i=0;i<s.size();i++)
	{
		if(s[i]=='(')
			c[++j]=i;
		else if(j)
		{
			d[i]=i-c[j]+1+d[c[j]-1];
			j--;
			if(d[i]==m)
				f++;
			if(m<d[i])
			{
				m=d[i];
				f=1;
			}
		}
	}
	cout<<m<<' '<<f<<endl;
	return 0;
}
