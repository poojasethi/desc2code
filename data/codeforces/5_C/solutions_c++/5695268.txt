#include<cstdio>
#include<cstring>
#define N 1010000
char s[N];
int n, t, num=1, ans, c[N], l[N];
int main()
{
	gets(s), n=strlen(s);
	for (int i=0; i<n; i++)
	{
		if (s[i]=='(') c[++t]=i; else
		if (t)
		{
			l[i]=i-c[t]+1+l[c[t]-1], t--;
			if (ans==l[i]) num++; else
			if (ans<l[i]) ans=l[i], num=1;
		}
	}
	if (!ans) num=1;
	printf("%d %d\n",ans,num);
	return 0;
}
