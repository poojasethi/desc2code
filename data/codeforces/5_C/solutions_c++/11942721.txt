#include<cstdio>
#include<cstring>
#include<stack>
#define M 1000009
char ch[M];std::stack<int>s;int dp[M];
int main(){scanf("%s",ch+1);int len=strlen(ch+1),A=0,C=0;
for(int i = 1;i<=len; i++){if(ch[i]=='(')s.push(i);else if (!s.empty()){
int u=s.top();s.pop();int l=dp[i]=i-u+1+dp[u-1];if(l>A)A=l,C=1;else if(l==A)C++;}}
if(A==0)puts("0 1");else printf("%d %d\n",A,C);return 0;
}
