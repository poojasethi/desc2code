#include<cstdio>
#include<cstring>
#define N 1<<20
char s[N];
int t,a,m=1,h[N],l[N];
int main(){
    gets(s);
    for(int i=0;s[i];i++){
        if(s[i]=='(')h[++t]=i;else if(t){
            l[i]=i-h[t]+1+l[h[t--]-1];
            if(a==l[i])m++;
            if(a<l[i])a=l[i],m=1;
        }
    }
    printf("%d %d",a,m);
}
