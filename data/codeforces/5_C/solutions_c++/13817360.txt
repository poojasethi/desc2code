#include<cstdio>
char s[1000001];
int  stk[1000001],dp[1000001];
int main(){
    //freopen("d://d.txt","r",stdin);
    scanf("%s",s+1);
    int l,i=0,p;
    for(l=1;s[l];l++){
        if(s[l]=='(')stk[++i]=l;
        else if(i){
            p=stk[i--];
            dp[l]=dp[p-1]+l-p+1;
        }
    }
    int mx=0,cnt;
    for(i=1;i<l;i++)
        if(dp[i]>mx){mx=dp[i];cnt=1;}
        else if(dp[i]==mx)cnt++;
    if(mx)printf("%d %d",mx,cnt);
    else printf("0 1");
    return 0;
}
