#include<iostream>
#include<cstdio>
#include<cstring>
#include<stack>
#define M 1000009
char ch[M];std::stack<int>s;int dp[M];
int main(){
scanf("%s",ch+1);
int len=strlen(ch+1),A=0,cnt=0;
for(int i = 1;i<=len; i++){
if(ch[i]=='(')s.push(i);
else if (!s.empty())
{
int u=s.top();s.pop();int l=dp[i]=i-u+1+dp[u-1];
if(l>A)A=l,cnt=1;else if(l==A)cnt++;
}
}
if(A==0)puts("0 1");else printf("%d %d\n",A,cnt);
return 0;
}
