#include <stack>
#include <cstdio>

using namespace std;

char s[1 << 20];

int main() {
	int n, m, c, b;
	stack<int> l;

	scanf("%s", s);
	n = 0;
	c = 1;
	b = -1;
	for (int i = 0; s[i] != '\0'; ++i) {
		if (s[i] == '(') {
			l.push(i);
		} else if (l.empty()) {
			b = i;
		} else {
			l.pop();
			m = i - (l.empty() ? b : l.top());
			if (n < m) {
				n = m;
				c = 1;
			} else if (m == n) {
				++c;
			}
		}
	}
	printf("%d %d\n", n, c);

	return 0;
}
