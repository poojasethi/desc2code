#include<stdio.h>
#include<string.h>

char s[1000010];
int a[1000010];
int main(){
  scanf("%s",s);
  int n=strlen(s),m=0,x=0;
  int ans=0, cnt=1;
  memset(a,0,sizeof(a));
  for (int i=0; i<n; i++){
    if (s[i]==')'){
      if (x>0){
        x--;
        a[x]+=a[x+1]+2;

        if (a[x]>ans){
          ans=a[x];
          cnt=1;
        }else if (a[x]==ans) cnt++;

      }else{ // x==0
        a[0]=0;
      }
    }else{
      x++;
      a[x]=0;
    }
  }
  printf("%d %d\n",ans,cnt);
  return 0;
}
