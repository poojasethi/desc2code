//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

string s;
int ans,cnt=1,dp[1000100];
vector<int> v;

int main()
{
  cin>>s;
  for(int i=1;i<=s.size();i++)
    if(s[i-1]=='(')
      v.push_back(i);
    else if(v.size()>0)
      {
	dp[i]=i-v.back()+1+dp[v.back()-1];
	if(ans==dp[i])
	  cnt++;
	if(ans<dp[i])
	  ans=dp[i],cnt=1;
	v.pop_back();
      }
  cout<<ans<<" "<<cnt<<endl;
  return 0;
}
