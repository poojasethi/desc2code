#include <iostream>
#include <algorithm>
#include <cstdio>
#include <string.h>

using namespace std;

char dd[1000005];
int dp[1000005];
int main(){
	int n,i,J;
	scanf("%s",dd+1);
	dd[0]=')';
	int ans=0,cnt=1;
	n=strlen(dd);
	for(i=1;i<=n;i++){
		if(dd[i]==')'){
			if(dd[i-dp[i-1]-1]=='('){
				dp[i]=dp[i-1]+2+dp[i-dp[i-1]-2];
				if(dp[i]>ans){
					ans=dp[i];
					cnt=0;
				}
				if(dp[i]==ans)
					cnt++;
			}
		}
	}

	printf("%d %d\n",ans,cnt);
}
