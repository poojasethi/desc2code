#include<iostream>
#include<string>
#include<bits/stdc++.h>
using namespace std;
	int d[1000006],stk[1000006];
int main()
{
	string str;

	int top=0,max=0,count=1,i;
	cin>>str;
	for(i=0;i<str.size();i++)
	{
		if(str[i]=='(')
			stk[++top]=i;
		else if(top)
		{
			d[i]=i-stk[top]+1+d[stk[top]-1];
			top--;
		if(d[i]==max)
		count++;
		if(d[i]>max)
		{
			max=d[i];
			count=1;
		}
		
		}
	}
	cout<<max<<" "<<count<<"\n";
	return 0;
}
