#include <cstdio>
const int MAXN = 1000000;
int lb[MAXN];
int rec[MAXN];
int main(int argc, char const *argv[])
{
	char c;
	int n = 0, p = 0;
	int t = 0, r = 0, m = 1;
	while ((c = getchar()) != '\n') {
		if (c == '(') {
			lb[p++] = n;
		} else {
			if (p) { // have matching left bracket
				--p;
				rec[n] = t = n - lb[p] + 1 + rec[lb[p] - 1];
				if (r < t) {
					r = t;
					m = 1;
				} else if (r == t) {
					++m;
				}
			}
		}
		++n;
	}
	printf("%d %d\n", r, m);
	return 0;
}
