#include<cstdio>
#include<cstring>
#include<algorithm>
#define N 1000010
using namespace std;
char s[N];
int t[N];
int main(void)
{
	scanf("%s", s+1);
	int len=0,ans=1,c=0;
	for ( int i=1; s[i]; i++ ){
		if ( s[i] == ')' && c && s[t[c]] == '(' ){
			c--;
			if ( i-t[c] == len )ans++;
			if ( i-t[c] > len ){
				len = i-t[c];
				ans=1;
			}
		}else t[++c]=i;
	}
	printf("%d %d\n", len,ans);
	return 0;
}
