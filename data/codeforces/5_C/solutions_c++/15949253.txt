#include<bits/stdc++.h>
const int maxn=1000002;
char a[maxn];
int f[maxn],q[maxn];
int n,t,m,s;

int main(){
  gets(a);
  n=strlen(a);
  s=1;
  for (int i=0;i<n;i++)
    if (a[i]=='(') q[++t]=i;
    else if (t){
      f[i]=i-q[t]+1+f[q[t]-1];
      t--;
      if (f[i]>m) m=f[i],s=1;
      else if (f[i]==m) s++;
    }
  printf("%d %d\n",m,s);
  return 0;
}
