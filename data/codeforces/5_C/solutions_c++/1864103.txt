#include <iostream>
#include <stack>

using namespace std;

string s;
int most, count = 1, temp = -1, size; 
stack <int> v;

int main()
{
	ios::sync_with_stdio(0);
	cin >> s;
	for (int i = 0 ; i < s.size() ; i++)
	{
		if (s[i] == '(')
			v.push(i);
		else if (v.empty())
			temp = i;
		else
		{
			v.pop();
			size = i - (v.empty() ? temp : v.top());
			if (size == most)
				count++;
			else if (size > most)
			{
				most = size;
				count = 1;
			}
		}
	}
	cout << most << " " << count << endl;
	return 0;
}
