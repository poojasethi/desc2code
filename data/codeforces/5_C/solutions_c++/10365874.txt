#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char ss[1010010];
int c[1010010],l[1010010];
int main()
{
    int Max = 0,Num = 1,t = 0;
    gets(ss);
    for(int i = 0;ss[i];++i) {
        if(ss[i] == '(')c[++t] = i;
        else if(t) {
            l[i] = i - c[t] + 1 + l[c[t]-1];
            --t;
            if(l[i] == Max)++Num;
            else if(l[i] > Max)Max = l[i],Num = 1;
        }
    }

    printf("%d %d\n",Max,Num);

    return 0;
}
