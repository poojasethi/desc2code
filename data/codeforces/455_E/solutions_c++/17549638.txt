#include <bits/stdc++.h>

#define lli long long int
#define fi first
#define se second
#define sc scanf
#define pr printf
#define pb push_back
#define mp make_pair
#define fin freopen( "input.txt", "r", stdin );
#define fout freopen( "output.txt", "w", stdout );

const int N = 100100;

using namespace std;

lli a[N], n, pre[N], k[N], b[N];
vector < lli > v[4 * N];

long double ix( lli x, lli y )
{
    return (1.0 * (b[y] - b[x])) / (1.0 * (k[x] - k[y]));
}

void build( lli t, lli l, lli r )
{
    if( l == r ){
        v[t].pb(a[l]);
        return;
    }
    lli m = (l + r) / 2;
    build(t * 2, l, m);
    build(t * 2 + 1, m + 1, r);
    vector < lli > g;
    lli j = m + 1;
    for( lli i = l; i <= m; i++ ){
        while( j <= r && (k[a[j]] > k[a[i]] || k[a[j]] == k[a[i]] && b[a[j]] < b[a[i]]) )
            g.pb(a[j++]);
        g.pb(a[i]);
    }
    while( j <= r )
        g.pb(a[j++]);
    for( lli i = 0; i < g.size(); i++ )
        a[l + i] = g[i];
    for( lli i = l; i <= r; i++ ){
        if( v[t].size() && k[a[i]] == k[v[t].back()] )
            continue;
        while( v[t].size() > 1 && ix(v[t][v[t].size() - 2], a[i]) <= ix(v[t][v[t].size() - 2], v[t][v[t].size() - 1]) )
            v[t].pop_back();
        v[t].pb(a[i]);
    }
}

lli bin( vector < lli > h, lli x )
{
    lli l = 0, r = h.size() - 1;
    while( l < r ){
        lli m = (l + r) / 2;
        if( ix(h[m], h[m + 1]) < (long double)x )
            l = m + 1;
        else
            r = m;
    }
    return k[h[l]] * x + b[h[l]];
}

lli get( lli t, lli l, lli r, lli tl, lli tr, lli x )
{
    if( tl > tr )
        return 1e18;
    if( l == tl && r == tr )
        return bin(v[t], x);
    lli m = (l + r) / 2;
    return min(get(t * 2, l, m, tl, min(m, tr), x), get(t * 2 + 1, m + 1, r, max(tl, m + 1), tr, x));
}

int main()
{
    sc( "%I64d", &n );
    for( lli i = 1; i <= n; i++ ){
        sc ( "%I64d", &a[i] );
        pre[i] = pre[i - 1] + a[i];
        k[i] = a[i];
        b[i] = a[i] * i - pre[i];
        a[i] = i;
    }
    build(1, 1, n);
    lli m;
    sc( "%I64d", &m );
    for( lli i = 1; i <= m; i++ ){
        lli x, y;
        sc( "%I64d%I64d", &x, &y );
        pr( "%I64d\n", pre[y] + get(1, 1, n, y - x + 1, y, x - y) );
    }
}
