#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
#include<map>
using namespace std;
char maze[500][500];
int main()
{
	//freopen("input.in","r",stdin);
	int n,m,k;
	scanf("%d%d%d\n",&n,&m,&k);
	for(int i=0;i<n;i++)
	gets(maze[i]);
	for(int r=1;r<n;r++)
	{
		for(int i=r;i<n-r;i++)
		{
			for(int j=r;j<m-r;j++)
			{
				if(maze[i][j]=='*'&&maze[i+r][j]=='*'&&maze[i-r][j]=='*'&&maze[i][j+r]=='*'&&maze[i][j-r]=='*')
				{
					k--;
				    if(!k)
				    {
					printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d\n",i+1,j+1,i-r+1,j+1,i+r+1,j+1,i+1,j-r+1,i+1,j+r+1);
					return 0;
				    }
				}
			}
		}
	}
	printf("-1\n");
	return 0;
}
