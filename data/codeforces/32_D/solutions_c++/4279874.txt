#include<iostream>
using namespace std;
char a[305][305];
int l;
int main()
{
    int n,m,k,i,r,j;
    cin>>n>>m>>k;
    for(i=1;i<=n;i++)for(j=1;j<=m;j++)cin>>a[i][j];
    for(r=1;r<=(min(n,m)-1)/2;r++)
    {
        for(i=r+1;i<=n-r;i++)    
        {
            for(j=r+1;j<=m-r;j++)
            {
                if(a[i][j]=='*'&&a[i+r][j]=='*'&&a[i-r][j]=='*'&&a[i][j+r]=='*'&&a[i][j-r]=='*')
                {l++;if(l==k){cout<<i<<" "<<j<<endl<<i-r<<" "<<j<<endl<<i+r<<" "<<j<<endl<<i<<" "<<j-r<<endl<<i<<" "<<j+r<<endl;return 0;}}
            }
        }
    }
    cout<<"-1"<<endl;
    
}
