#include<algorithm>
#include<iostream>
#include<vector>
using namespace std;
struct zb
{
	int x,y,s;
};
int main()
{
	int n,m,s;
	vector<zb> ans;
	cin>>n>>m>>s;
	char a[n][m];
	for(int i=0;i<n;i++)for(int j=0;j<m;j++)cin>>a[i][j];
	for(int k=1;k<n&&k<m;k++)for(int i=k;i+k<n;i++)for(int j=k;j+k<m;j++)
	{
		if(a[i][j]=='*'&&a[i][j+k]=='*'&&a[i][j-k]=='*'&&a[i+k][j]=='*'&&a[i-k][j]=='*')
		{
			zb sum;
			sum.x=i;sum.y=j;sum.s=k;
			ans.push_back(sum);
		}
	}
	s--;
	if(ans.size()<=s)cout<<-1;
	else
	{
		cout<<ans[s].x+1<<" "<<ans[s].y+1<<endl;
		cout<<ans[s].x-ans[s].s+1<<" "<<ans[s].y+1<<endl;
		cout<<ans[s].x+ans[s].s+1<<" "<<ans[s].y+1<<endl;
		cout<<ans[s].x+1<<" "<<ans[s].y-ans[s].s+1<<endl;
		cout<<ans[s].x+1<<" "<<ans[s].y+ans[s].s+1<<endl; 
	}
}
