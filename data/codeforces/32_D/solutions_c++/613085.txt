#include <cstdio>
#include <cstdlib>
#include <algorithm>

using namespace std;

int N, M, K;
char MAPA[303][303];

int main() {

//    freopen("D.in", "r", stdin);
    scanf("%d %d %d", &M, &N, &K);
    for (int i = 0; i < M; i++) scanf("%s", MAPA[i]);
    int c = 0;
    for (int r = 1; r < 300; r++) for (int i = 0; i < M; i++) for (int j = 0; j < N; j++) if (MAPA[i][j]=='*') {
        if (i-r < 0 || MAPA[i-r][j]!='*' ||
            j-r < 0 || MAPA[i][j-r]!='*' ||
            i+r >= M || MAPA[i+r][j]!='*' ||
            j+r >= N || MAPA[i][j+r]!='*') continue;
        c++;
        if (c == K) {
            printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d\n", i+1, j+1, i-r+1, j+1, i+r+1, j+1, i+1, j-r+1, i+1, j+r+1);
            return 0;
        }
    }
    if (c != K) puts("-1");

    return 0;
}
