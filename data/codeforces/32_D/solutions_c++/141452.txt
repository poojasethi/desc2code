#include <iostream>
#include <string>
#include <vector>
#include <cmath>

using namespace std;

int main()
{
  int n,m,k;
  cin>>n>>m>>k;
  vector<string> v(n);
  for (int i=0;i<n;i++) cin>>v[i];
  int count=0;
  bool trobat=false;
  int maxr=min((n-1)/2,(m-1)/2);
  for (int r=1;r<=maxr and not trobat;r++) {
    for (int i=r;i<n-r and not trobat;i++) {
      for (int j=r;j<m-r and not trobat;j++) {
	if (v[i][j]=='*' and v[i-r][j]=='*' and v[i+r][j]=='*'
	    and v[i][j-r]=='*' and v[i][j+r]=='*') {
	  count++;
	  if (count==k) {
	    trobat=true;
	    cout<<i+1<<" "<<j+1<<endl;
	    cout<<i-r+1<<" "<<j+1<<endl;
	    cout<<i+r+1<<" "<<j+1<<endl;
	    cout<<i+1<<" "<<j-r+1<<endl;
	    cout<<i+1<<" "<<j+r+1<<endl;
	  }
	}
      }
    }
  }
  if (not trobat) cout<<-1<<endl;
}
