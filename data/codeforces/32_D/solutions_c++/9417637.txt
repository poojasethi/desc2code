#include <bits/stdc++.h>
using namespace std;
char grid[310][310];
int n,m;
bool gut(int i,int j)
{
    return i>=0&&i<n&&j>=0&&j<m&&grid[i][j]=='*';
}
bool star(int i,int j,int k)
{
    return gut(i,j)&&gut(i-k,j)&&gut(i+k,j)&&gut(i,j-k)&&gut(i,j+k);
}
int main()
{
   int kk;
   scanf("%d%d%d",&n,&m,&kk);
   for(int i=0;i<n;i++) scanf("%s",grid[i]);
   int till=0;
    for(int k=1;k<160;k++)
    for(int i=0;i<n;i++)
    for(int j=0;j<m;j++)
    {
        if(star(i,j,k)) till++;
        if(till==kk) {printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d\n",i+1,j+1,i-k+1,j+1,i+k+1,j+1,i+1,j-k+1,i+1,j+k+1);return 0;}
    }
    printf("-1\n");
    return 0;
}
