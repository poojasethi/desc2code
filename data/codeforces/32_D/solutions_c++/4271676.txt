#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
string s[350];
int main()
{
    int n,m,x;
    cin>>n>>m>>x;
    for(int i=0;i<n;i++)
       cin>>s[i];
    for(int k=1;k<n;k++)
       for(int i=k;i<n-k;i++)
          for(int j=k;j<m-k;j++){
             if(s[i][j]=='*' && s[i+k][j]=='*' && s[i-k][j]=='*' && s[i][j+k]=='*' && s[i][j-k]=='*' && !--x){
               cout<<i+1<<" "<<j+1<<endl;              
               cout<<i+1-k<<" "<<j+1<<endl;
               cout<<i+1+k<<" "<<j+1<<endl;   
               cout<<i+1<<" "<<j+1-k<<endl;   
               cout<<i+1<<" "<<j+1+k<<endl;
               return 0;}
               }
    cout<<-1;
    return 0;
}
