#include <cstdio>

int n, m, k;
char s[310][310];

int main()
{
    scanf("%d%d%d\n", &n, &m, &k);
    for (int i = 0; i < n; i++)
        gets(s[i]);
    int ck = 0;
    for (int e = 1; e <= n / 2; e++)
        for (int i = e; i < n - e; i++)
            for (int j = e; j < m - e; j++)
                if (s[i][j] == '*' && s[i - e][j] == '*' && s[i][j + e] == '*' && s[i][j - e] == '*' && s[i + e][j] == '*' )
                {
                    ck++;
                    if (ck == k)
                    {
                        printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d", i + 1, j + 1, i - e + 1, j + 1, i + e + 1, j + 1, i + 1, j - e + 1, i + 1, j + e + 1);
                        return 0;
                    }
                }
    puts("-1");
}
