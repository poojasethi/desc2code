#include <iostream>
#include <string>
#define s <<" "<<
#define r <<endl<< 
using namespace std;
string f[500];
int main()
{
	int n,m,k;
	cin >>n>>m>>k;
	for (int i=0;i<n;i++) cin >>f[i];
	for (int o=1;o<n;o++)
	{
		for (int i=o;i<n-o;i++)
		{
			for (int j=o;j<m-o;j++)
			{
				if (f[i][j]=='*' && f[i+o][j]=='*' && f[i-o][j]=='*' && f[i][j+o]=='*' && f[i][j-o]=='*' && !--k )
				{
					cout <<i+1 s j+1 r i-o+1 s j+1 r i+o+1 s j+1 r i+1 s j-o+1 r i+1 s j+o+1;
					return 0;
				}
			} 
		} 
	}
	cout <<-1;
	return 0;
}
