#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main()
{
	int n, m, k;
	cin >> n >> m >> k;
	vector <string> s(n);
	for (int i = 0; i < n; ++i)
		cin >> s[i];
		
	for (int rad = 1; rad <= (n - 1) / 2; ++rad)
		for (int i = rad; i < n - rad; ++i)
			for (int j = rad; j < m - rad; ++j)
				if (s[i][j] == '*' && s[i - rad][j] == '*' && s[i + rad][j] == '*' && s[i][j - rad] == '*' && s[i][j + rad] == '*')
					if (--k == 0)
					{
						cout << i + 1 << ' ' << j + 1 << '\n' << i - rad + 1 << ' ' << j + 1 << '\n' << i + rad + 1 << ' ' << j + 1 << '\n' \
							<< i + 1 << ' ' << j - rad + 1 << '\n' << i + 1 << ' ' << j + rad + 1 << '\n';
						return 0;
					}
	
	cout << "-1\n";	
	
	return 0;
}
