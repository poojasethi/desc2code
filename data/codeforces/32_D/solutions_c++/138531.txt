#include<iostream>
using namespace std;
int main()
{
int n,m,r,k,a[305][305],i,j;
char c;
cin>>n>>m>>k;
for(i=1;i<=n;i++)
{
for(j=1;j<=m;j++)
{
cin>>c;
if(c=='*')
{
a[i][j]=1;
}
else
{
a[i][j]=0;
}
}
}
for(r=1;r<=min(n,m);r++)
{
for(i=1+r;i<=n-r;i++)
{
for(j=1+r;j<=m-r;j++)
{
if(a[i][j]==1 && a[i+r][j]==1 && a[i-r][j]==1 && a[i][j+r]==1 && a[i][j-r]==1)
{
k--;
if(k==0)
{
cout<<i<<" "<<j<<endl;
cout<<i-r<<" "<<j<<endl;
cout<<i+r<<" "<<j<<endl;
cout<<i<<" "<<j-r<<endl;
cout<<i<<" "<<j+r<<endl;
return 0;
}
}
}
}
}
cout<<-1;
return 0;
}
