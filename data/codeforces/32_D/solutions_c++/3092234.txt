#include<iostream>
#include<vector>
using namespace std;
char s[305][305];
int n,m,g;
int ans;
int main()
{
	cin>>n>>m>>g;
	for(int i=0;i<n;i++)
	cin>>s[i]; 
	for(int p=1;p<=min(n,m);p++){
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(i-p>=0 && j-p>=0 && i<n-p && j<m-p && s[i][j]=='*' && s[i-p][j]=='*' && s[i][j-p]=='*' && s[i+p][j]=='*' && s[i][j+p]=='*')
			{
				ans++;
				if(ans==g)
				{
					cout<<i+1<<" "<<j+1<<endl;
					cout<<i+1-p<<" "<<j+1<<endl;
					cout<<i+1+p<<" "<<j+1<<endl;
					cout<<i+1<<" "<<j+1-p<<endl;
					cout<<i+1<<" "<<j+1+p<<endl;
					return 0;
				}
		    }
	} }
}
	cout<<"-1";
	//system("pause");
	return 0;
}
