#include <iostream>

using namespace std;

main()
{
	char a[301][301];
	int n,m;
	int i,j,l;
	long long k;
	cin >> n >> m >> k;
	for (i=1;i<=n;i++)
		for (j=1;j<=m;j++)
			cin >> a[i][j];
	for (i=1;i<=150;i++)
		for (l=i+1;l<=n-i;l++)
			for (j=i+1;j<=m-i;j++)
				if ((a[l][j]=='*')&&(a[l-i][j]=='*')&&(a[l+i][j]=='*')&&(a[l][j-i]=='*')
				&&(a[l][j+i]=='*'))
				{
					k--;
					if (k==0)
					{
						cout << l << " " << j << endl;
						cout << l-i << " " << j << endl;
						cout << l+i << " " << j << endl;
						cout << l << " " << j-i << endl;
						cout << l << " " << j+i << endl;
						return 0;
					}
				}
	cout << -1 << endl;
}
