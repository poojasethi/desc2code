#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,k,u;
char s[302][302];
int main(){
  scanf("%d%d%d",&n,&m,&k);
  for(int i=0;i<n;i++) scanf("%s",s[i]);
  for(int r = 1; r <= (min(n,m)-1)/2 && !u; r++){
    for(int i=r;i<n-r && !u;i++)
      for(int j=r;j<m-r && !u;j++)
    if(s[i][j] == '*' && s[i-r][j] == '*' && s[i][j-r] == '*' && s[i][j+r] == '*' && s[i+r][j] == '*'){
      k--; if(k == 0){
        printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d\n",i+1,j+1,i-r+1,j+1,i+r+1,j+1,i+1,j-r+1,i+1,j+r+1); u = 1;
      }
    }
  }
  if(!u) printf("-1");
  return 0;
}
