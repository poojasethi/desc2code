#include <stdio.h>

char a[505][505];
int n, m, k;
int dx[5]={0,-1,1,0,0}, dy[5]={0,0,0,-1,1};
bool valid(int x, int y) {
	return x>=0&&x<n&&y>=0&&y<m&&a[x][y]=='*';
}

void solve() {
	int cnt=0;
	for(int L=1;L<=n;L++) {
		for(int i=0;i<n;i++)
			for(int j=0;j<m;j++) {
				int w;
				for(w=0;w<5;w++)
					if(!valid(i+L*dx[w],j+L*dy[w]))
						break;
				if(w==5 && ++cnt==k) {
					for(w=0;w<5;w++)
						printf("%d %d\n", i+L*dx[w]+1, j+L*dy[w]+1);
					return;
				}
			}
	}
	printf("-1\n");
}

int main(void) {
	scanf("%d%d%d", &n, &m, &k);
	for(int i=0;i<n;i++)
		scanf("%s", a[i]);
	solve();
}
