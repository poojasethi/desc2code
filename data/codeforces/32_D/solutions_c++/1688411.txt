#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

#define MAXN 310

int n,m,k;
char mat[MAXN][MAXN];

int main() {
	scanf("%d%d%d",&n,&m,&k);
	
	for (int i=0; i<n; i++)
		scanf(" %s",mat[i]);
	
	for (int r=1; r<=min(n,m)/2; r++)
		for (int i=0; i<n; i++)
			for (int j=0; j<m; j++) {
				if (mat[i][j] == '.' || i-r<0 || i+r>n-1 || j-r<0 || j+r>m-1) continue;
				if (mat[i-r][j] == '*' && mat[i+r][j] == '*' && mat[i][j-r] == '*' && mat[i][j+r] == '*') k--;
				if (!k) {
					printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d\n",i+1,j+1,i-r+1,j+1,i+r+1,j+1,i+1,j-r+1,i+1,j+r+1);
					goto fim;
				}
			}
	
	printf("-1\n");
	
	fim:
	return 0;
}
