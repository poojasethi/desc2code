#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 301

int n,m,s,i,j,k;
bool A[lim][lim];
char str[lim];

int main()
{
  scanf("%d%d%d",&n,&m,&s);
	for(i=0;i<n;i++)
	{
	  scanf("%s",str);
    for(j=0;j<m;j++)
		  A[i][j]=str[j]=='*';
	}
	for(i=1;2*i+1<=min(n,m);i++)
	{
    for(j=i;j+i<n;j++)
		{
      for(k=i;k+i<m;k++)
        if(A[j][k] && A[j][k+i] && A[j][k-i] && A[j+i][k] && A[j-i][k])		
        {
				  s--;
					if(!s)
					  break;
        }	
      if(!s)
			  break;
		}		
    if(!s)
		  break;
	}	
	if(s)
    printf("-1\n");
  else
	{
	  j++;
		k++;
    printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d\n",j,k,j-i,k,j+i,k,j,k-i,j,k+i);	
  }
	return 0;
}
