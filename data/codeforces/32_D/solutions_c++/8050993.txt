#include<bits/stdc++.h>

using namespace std;

int main(){
	int n,m,k,i,j,r;
	cin >> n >> m >> k;
	char in[n][m];
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			cin >> in[i][j];
		}
	}
	for(r=1;r<=150&&k>0;++r){
		for(i=0;k>0&&(i+r<n);++i){
			for(j=0;k>0&&(j+r<m);++j){
				if(i-r>=0 && j-r>=0 && in[i][j]=='*' && in[i-r][j]=='*' && in[i][j-r]=='*' && in[i+r][j]=='*' && in[i][j+r]=='*'){
					k--;
				}
			}
		}
	}
	i--;
	j--;
	r--;
	if(k>0){
		cout << "-1";
	}else{
		cout << i+1 << " " << j+1 << endl;
		cout << i-r+1 << " " << j+1 << endl;
		cout << i+r+1 << " " << j+1 << endl;
		cout << i+1 << " " << j-r+1 << endl;
		cout << i+1 << " " << j+r+1 << endl;
	}
}
