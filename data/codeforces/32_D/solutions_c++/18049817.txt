#include <cstdio>
using namespace std;

char map[301][301];
int n, m, k;
int main(){
    scanf("%d %d %d", &n, &m, &k);
    for(int i=0; i<n; i++){
        scanf("%s", map[i]);
    }
    for(int i=1; i<=n/2; i++){
        for(int r=0; r<n; r++){
            for(int c=0; c<m; c++){
                if(r < i || c < i || r+i >= n || c+i >= m) continue;
                if(map[r][c] == '.' || map[r-i][c] == '.' || map[r+i][c] == '.' || map[r][c-i] == '.' || map[r][c+i] == '.') continue;
                k--;
                if(k == 0){
                    printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d\n", r+1, c+1, r-i+1, c+1, r+i+1, c+1, r+1, c-i+1, r+1, c+i+1);
                    return 0;
                }
            }
        }
    }
    printf("-1\n");
    return 0;
}
