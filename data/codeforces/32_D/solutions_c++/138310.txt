//#include "stdafx.h"

#include <string>
#include <iostream>

using namespace std;


int main()
{
	string s[300];
	int n,m,k;
	cin>>n>>m>>k;
	for(int i=0;i<n;i++)
		cin>>s[i];
	int k1=0;
	for(int h=1;n-2*h>0&&m-2*h>0;h++)
	{
		for(int i=h;i<n-h;i++)
			for(int j=h;j<m-h;j++)
				if(s[i][j]=='*'&&s[i-h][j]=='*'&&s[i][j-h]=='*'&&s[i+h][j]=='*'&&s[i][j+h]=='*')
				{
					k1++;
					if(k==k1)
					{
						i++;j++;
						cout<<i<<" "<<j<<endl<<i-h<<" "<<j<<endl<<i+h<<" "<<j<<endl<<i<<" "<<j-h<<endl<<i<<" "<<j+h;
						return 0;
					}
				}

	}
	cout<<-1;
	return 0;
}
