#include <bits/stdc++.h>
using namespace std;

#define f(i,j)				(mat[i][j]=='*')
const int MAX_N = 500+10;
char mat[MAX_N][MAX_N];

int main ( )
{
	int h, w, k, i, j, x;
	scanf ( "%d%d%d", &h, &w, &k );
	for ( i = 0; i < h; ++i )
		scanf ( "%s", mat[i] );

	for ( x = 1; x <= h; ++x )
		for ( i = x; i+x < h; ++i )
			for ( j = x; j+x < w; ++j )
				if ( f(i,j) && f(i+x,j) && f(i-x,j) && f(i,j+x) && f(i,j-x) ) {
					k--;
					if ( k == 0 ) {
						printf ( "%d %d\n", i+1, j+1 );
						printf ( "%d %d\n", i+1-x, j+1 );
						printf ( "%d %d\n", i+1+x, j+1 );
						printf ( "%d %d\n", i+1, j+1-x );
						printf ( "%d %d\n", i+1, j+1+x );
						return 0;
					}
				}

	printf ( "-1\n" );
	return 0;
}
