#include <iostream>
using namespace std;
long n,m,k,r,l,c,rez,i,j;
char A[301][301];

long mini(long a, long b)
{
	if (a<=b)
		return a;
	return b;
}

int main()
{
	cin>>n>>m>>k;
	for(i=1;i<=n;i++)
		for (j=1;j<=m;j++)
			cin>>A[i][j];
	rez=0;
	for (r=1;r<=mini(n/2,m/2);r++)
		for (l=r+1;l<=n-r;l++)
			for (c=r+1;c<=m-r;c++)
				if (A[l][c]=='*' && A[l-r][c]=='*' && A[l+r][c]=='*' && A[l][c-r]=='*' && A[l][c+r]=='*')
				{
					rez++;
					if (rez==k)
					{
						cout<<l<<" "<<c<<endl;
						cout<<l-r<<" "<<c<<endl;
						cout<<l+r<<" "<<c<<endl;
						cout<<l<<" "<<c-r<<endl;
						cout<<l<<" "<<c+r<<endl;
						return 0;
					}
				}
	cout<<-1;
	return 0;
}
