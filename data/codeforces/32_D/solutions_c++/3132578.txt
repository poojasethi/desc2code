#include<iostream>
using namespace std;
char a[500][500];
int main()
{
 	int n,m,k;
 	cin>>n>>m>>k;
 	for(int i=0;i<n;i++)
 	{
	 		cin>>a[i];
    }
    for(int r=1;r<n;r++)
    {
	 		for(int i=r;i<n-r;i++)
	 		{
			 		for(int j=r;j<m-r;j++)
			 		{
					 		if((a[i][j]=='*')&&(a[i+r][j]=='*')&&(a[i-r][j]=='*')&&(a[i][j+r]=='*')&&(a[i][j-r]=='*'))
					 		{
                                k--;
                                if(!k)
                                {
								 	  cout<<i+1<<" "<<j+1<<endl;
								 	  cout<<i-r+1<<" "<<j+1<<endl;
								 	  cout<<i+r+1<<" "<<j+1<<endl;
								 	  cout<<i+1<<" "<<j-r+1<<endl;
								 	  cout<<i+1<<" "<<j+r+1<<endl;
								 	  return 0;
						        }
						     }
					 }
			 }
	 }
	 cout<<-1<<endl;
	 return 0;
}
