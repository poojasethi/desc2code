#include <stdio.h>
#include <algorithm>
using namespace std;

#define rep(i,n) for (int i=0; i<n; i++)

char grid[310][310];

int n, m, k;
int r;
int c;

int main() {
	scanf("%d %d %d", &n, &m, &k);
	c=1;

	rep(i,n) scanf(" %s", grid[i]);

	for (r=1; r<min(n,m); r++) {
		rep(i,n) rep(j,m) {
			if (grid[i][j]=='.') continue;

			if (i-r<0) continue;
			if (i+r>=n) continue;
			if (j-r<0) continue;
			if (j+r>=m) continue;

			if (grid[i-r][j]=='*' && grid[i+r][j]=='*' &&
				grid[i][j-r]=='*' && grid[i][j+r]=='*') {
				
				if (c==k) {
					printf("%d %d\n", i+1, j+1);
					printf("%d %d\n", i-r+1, j+1);
					printf("%d %d\n", i+r+1, j+1);
					printf("%d %d\n", i+1, j-r+1);
					printf("%d %d\n", i+1, j+r+1);

					return 0;
				}
				else c++;
			}
		}
	}

	printf("-1\n");
	return 0;
}
