#include <iostream>
#include <cstdio>
using namespace std;
int n,m,k,cnt,i,j,r;
char a[301][301];
int check(int x,int y,int r)
{
	if ((x<=r) || (y<=r) || (x+r>n) || (y+r>m))
		return 0;
	if ((a[x][y]=='*') && (a[x-r][y]=='*') && (a[x+r][y]=='*') && (a[x][y-r]=='*') && (a[x][y+r]=='*'))    return 1;
	return 0;

}
int main()
{
cin>>n>>m>>k;
for (i=1;i<=n;i++)
for (j=1;j<=m;j++)
cin>>a[i][j];
for (r=1;2*r<(min(n+1,m+1));r++)
{
	for (i=1;i<=n;i++)  
	{
		for (j=1;j<=m;j++)
		{
		if (check(i,j,r))
			cnt++;
		if (cnt==k)  {
			cout<<i<<" "<<j<<endl;
			cout<<i-r<<" "<<j<<endl;
			cout<<i+r<<" "<<j<<endl;
			cout<<i<<" "<<j-r<<endl;
			cout<<i<<" "<<j+r;
			return 0; 
				}
		}
	}
}
cout<<-1;


return 0;
}
