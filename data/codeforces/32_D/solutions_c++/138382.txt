#include<cstdio>

int n,m,k;
bool map[300][300];
char c;

int main(){
 int i,j,l;
 scanf("%d%d%d\n",&n,&m,&k);
 for(i=0;i<n;i++){
  for(j=0;j<m;j++){
   scanf("%c",&c);
   if(c=='*') map[i][j]=1;
  }
  scanf("%c",&c);
 }

 for(l=1;l<(n+1)/2;l++){
  for(i=l;i<n-l;i++){
   for(j=l;j<m-l;j++){
    if(map[i][j] && map[i-l][j] && map[i][j-l] && map[i+l][j] && map[i][j+l])
     --k;
    if(k==0) break;
   }
   if(k==0) break;
  }
  if(k==0) break;
 }
 if(k==0)
  printf("%d %d\n%d %d\n%d %d\n%d %d\n%d %d\n",i+1,j+1,i-l+1,j+1,i+l+1,j+1,i+1,j-l+1,i+1,j+l+1);
 else printf("-1\n");

 return 0;
}
