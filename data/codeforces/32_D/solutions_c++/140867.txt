#include<iostream>
using namespace std;
long minim(long a,long b)
{
	if(a<b)
		return a;
	return b;
}
long n,m,k,i,j,r,nr;
char c[301][301];
int main()
{
	cin>>n>>m>>k;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			cin>>c[i][j];
	for(r=1;r<=minim(n/2,m/2);r++)
		for(i=r+1;i<=n-r;i++)
			for(j=r+1;j<=m-r;j++)
			{
				if(c[i][j]=='*' && c[i-r][j]=='*' && c[i+r][j]=='*' && c[i][j-r]=='*' && c[i][j+r]=='*')
				{
					nr++;
					if(k==nr)
					{
						cout<<i<<" "<<j<<endl;
						cout<<i-r<<" "<<j<<endl;
						cout<<i+r<<" "<<j<<endl;
						cout<<i<<" "<<j-r<<endl;
						cout<<i<<" "<<j+r<<endl;
						return 0;
					}
				}
			}
cout<<-1;
return 0;
}
