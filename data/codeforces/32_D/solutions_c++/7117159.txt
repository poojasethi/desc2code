#include <iostream>
#include <vector>
using namespace std;
int main()
{
	int n, m, k, i, j, r, rm;
	cin>>n>>m>>k;
	char c[n+1][m+2];
	vector <int> v;
	for (i=1; i<=n; i++) cin>>c[i]+1;
	rm = min(n/2, m/2);
	for (r=1; r<=rm; r++) for (i=1; i<=n; i++) for (j=1; j<=m; j++)
	if (i-r>=1 && i+r<=n && j-r>=1 && j+r<=m && 
		c[i][j]=='*' && c[i-r][j]=='*' && c[i+r][j]=='*' && 
		c[i][j-r]=='*' && c[i][j+r]=='*') {
		v.push_back(r); v.push_back(i); v.push_back(j);
	}
	if (k>v.size()/3) cout<<-1<<endl;
	else {
		r=v[3*k-3]; i=v[3*k-2]; j=v[3*k-1];
		cout<<i<<" "<<j<<endl 
		 	<<i-r<<" "<<j<<endl
		 	<<i+r<<" "<<j<<endl
		 	<<i<<" "<<j-r<<endl
		 	<<i<<" "<<j+r<<endl;
	}
	return 0;
}
