#include <stdio.h>
int mat[303][303];
int main()
{
	int r,c,k,i,j,l,t;
	scanf(" %d %d %d",&r,&c,&k);
	t=((r<c?r:c)-1)/2;
	for(i=1;i<=r;i++) for(j=1;j<=c;j++) scanf(" %c",&mat[i][j]);
	for(l=1;l<=t;l++)
	{
		for(i=l+1;i<=r-l;i++)
			for(j=l+1;j<=c-l;j++)
			if( mat[i][j] == '*' &&
				mat[i-l][j] == '*' &&
				mat[i+l][j] == '*' &&
				mat[i][j-l] == '*' &&
				mat[i][j+l] == '*' )
			{
				--k;
				if(k==0)
				{
					printf("%d %d\n",i,j);
					printf("%d %d\n",i-l,j);
					printf("%d %d\n",i+l,j);
					printf("%d %d\n",i,j-l);
					printf("%d %d\n",i,j+l);
					return 0;
				}
			}
	}
	printf("-1");
	return 0;
}
