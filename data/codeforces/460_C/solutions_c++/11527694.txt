#include<bits/stdc++.h>
using namespace std;
int a1[100003],a2[100003],a,b,c,p,m,n;
int f(int x)
{
    int a=0,b=m;
    for (int i=0;i<n;i++)
    {
        if (i>=p) a-=a2[i-p];
        a2[i]=max(0,x-a-a1[i]);
        a+=a2[i];b-=a2[i];
        if (b<0) return 0;
    }
    return 1;
}
main()
{   int a=1e9+1,b,c;
    scanf("%d%d%d",&n,&m,&p);
    for (int i=0;i<n;i++) {scanf("%d",&a1[i]);a=min(a,a1[i]);}
    b=a+m+1;
    while (a<b)
    {
        c=(a+b)/2;
        if (f(c)) a=c+1;
        else b=c;
    }
    cout<<a-1;

}
