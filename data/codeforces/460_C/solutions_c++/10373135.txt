#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
#define maxn 1000000
int a[maxn];
long long b[2*maxn];
int main()
{
	int i,n,w; 
	long long m,cnt,sum,fk,min,max,mid;
    cin>>n>>m>>w; 
    min=1;max=1<<30;
	for(i=0;i<n;i++)
		cin>>a[i];
	while(min<=max) 
	{
		mid=(min+max)/2;
		cnt=0;sum=0;
        memset(b,0,sizeof(b));
		for(i=0;i<n;i++)
		{
			cnt=cnt-b[i];
			fk=mid-(a[i]+cnt);
			if(fk>0)
			{
			cnt=cnt+fk;
			b[i+w]=fk;
			sum=sum+fk; 
		  }
	}
	if(sum>m)
	max=mid-1;
	else 
	min=mid+1;
	}
    cout<<max<<endl; 
return 0;
}
