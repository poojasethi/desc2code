#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,p,a1[100000],a2[100000];
int f(int x)
{
	int a=0,b=m;
	for(int i=0;i<n;++i)
	{
		if(i>=p)a-=a2[i-p];
		a2[i]=max(0,x-a1[i]-a);
		a+=a2[i];b-=a2[i];
		if(b<0)return 0;
	}
	return 1;
}
int main()
{
	int a=1000000001,b,c;
	scanf("%d%d%d",&n,&m,&p);
	for(int i=0;i<n;++i)
	{scanf("%d",a1+i);a=min(a,a1[i]);}
	b=a+m+1;
	while(a<b)
	{c=a+b>>1;if(f(c))a=c+1;else b=c;}
	printf("%d\n",a-1);
}
