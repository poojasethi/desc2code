#include<cstdio>
#include<cstring>
#define LL long long

int a[100100];
LL b[100100],c[200100];

int main() {
	int n,m,w;
	scanf("%d%d%d",&n,&m,&w);
	for(int i=1;i<=n;++i)scanf("%d",a+i);
	int l=1,r=1100000000;
	while(l<r) {
		int mid=l+r+1>>1;
		LL cnt=0;
		for(int i=1;i<=n;++i)b[i]=mid-a[i];
		memset(c,0,sizeof(c));
		for(int i=1;i<=n;++i) {
			if(b[i]-c[i-1]>0){
				c[i]+=b[i]-c[i-1];
				c[i+w-1]-=b[i]-c[i-1];
				cnt+=b[i]-c[i-1];
			}
			c[i]+=c[i-1];
		}
		if(cnt<=m)l=mid;
		else r=mid-1;
	}
	printf("%d\n",l);
	return 0;
}
