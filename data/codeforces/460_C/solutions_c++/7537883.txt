#include <iostream>
#define MAXN 100010
#define MAXH 1000000000 + MAXN

using namespace std;

int n,m,w,a[MAXN],b[MAXN];

bool check(int h) {
    long long tot,ct;
    tot = ct = b[0] = max(h-a[0],0);
    for (int i = 1; i < n; i++) {
        if (i-w >= 0) ct -= b[i-w];
        b[i] = max(h-a[i]-ct,0ll);
        ct += b[i];
        tot += b[i];
    }
    return tot <= m;
}

int main() {
    cin >> n >> m >> w;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    int lo = 0, hi = MAXH, mid;
    while (hi - lo > 1) {
        mid = (hi+lo)/2;
        if (check(mid)) lo = mid;
        else hi = mid;
    }
    cout << lo << '\n';
}
