#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long LL;
LL a[110000],pre[110000],n,m,w,mx,ans;
bool check(LL x)
{
	LL tot=0;
	for (LL i=1; i<=n; i++){
		LL c=pre[i-1]-pre[max(i-w,0LL)];
		if (x-a[i]-c>0) pre[i]=pre[i-1]+x-a[i]-c,tot+=x-a[i]-c; else pre[i]=pre[i-1];
	}
	if (tot>m) return false;
	return true;
}		
int main()
{
	scanf("%d%d%d", &n, &m, &w);
	mx=1000000000;
	for (LL i=1; i<=n; i++) scanf("%I64d",&a[i]), mx=min(mx,a[i]);
	LL l=mx,r=a[1]+m;
	while (l<=r)
	{
		LL mid=(l+r)>>1;
		if (check(mid)){ans=mid,l=mid+1;}
		else r=mid-1;
	}
	printf("%d\n",ans);
}
