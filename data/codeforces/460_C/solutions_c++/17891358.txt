#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
	__int64 n,m,w,i,l,r,sum,m1,a[200005],b[200005];
	scanf("%I64d%I64d%I64d%I64d",&n,&m,&w,&a[0]);
	l=a[0];
	r=a[0];
	for(i=1;i<n;i++)
	{
		scanf("%I64d",&a[i]);
		if(a[i]<l) l=a[i];
		if(a[i]>r) r=a[i];
	}
	r+=m;
	while(l<=r)
	{
		int mid=(l+r)>>1;
		m1=0;
		sum=0;
		memset(b,0,sizeof(b));
		for(i=0;i<n&&m1<=m;i++)
		{
			sum+=b[i];
			if(sum+a[i]<mid)
			{
				m1+=mid-(sum+a[i]);
				b[i+w]-=mid-(sum+a[i]);
				sum=mid-a[i];
			}
		}
		if(m1<=m) l=mid+1;
		else r=mid-1;
	}
	printf("%I64d\n",l-1);
	return 0;
}
