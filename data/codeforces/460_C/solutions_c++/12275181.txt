#include <iostream>
#include <algorithm>
using namespace std;
int n,m,w,a[100005],inc[100005];
bool check(int k)
{
     long long use=0,cur=0;
     for (int i=0;i<n;i++)
     {
          if (i>=w) cur-=inc[i-w];
          long long d=max(k-a[i]-cur,0LL);
          inc[i]=d;
          use+=d;cur+=d;
     }
     return use<=m;
}
int main()
{
     cin>>n>>m>>w;
     for (int i=0;i<n;i++) cin>>a[i];
     int lo=1,hi=1100000000,mi=(hi+lo)/2;
     while (lo<mi)
     {
          if (check(mi)) lo=mi; else hi=mi;
          mi=(hi+lo)/2;
     }
     cout<<lo;
     return 0;
}
