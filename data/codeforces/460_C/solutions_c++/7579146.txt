#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
long long n,k,i,m[200010],w,l,r,z,a[200010],kol,sum;
int main(){
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>n>>k>>w;
	for (int i=0;i<n;i++)
		cin>>m[i];
	l=1; r=2000000000;
	while (l<=r){
		z=(l+r)/2;
		kol=0;
		sum=0;
		memset(a,0,sizeof(a));
		for (int i=0;i<n && kol<=k;i++){
			sum=sum+a[i];
			if (m[i]+sum<z){
				kol=kol+z-(m[i]+sum);
				a[i+w]-=z-(m[i]+sum);
				sum=z-m[i];
			}
		}
		if (kol<=k)
			l=z+1;
		else r=z-1;
			
	}
	cout<<r;
    return 0;
}
