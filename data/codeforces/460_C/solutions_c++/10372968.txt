#include<stdio.h>
#include<string.h>
#define inf 2000000000
#define ll long long
ll n,m,w;
ll fl[200000];
ll sum[200000];
int main(){
	while(~scanf("%d%d%d",&n,&m,&w)){
		for(int i=1;i<=n;i++)scanf("%d",&fl[i]);
		ll l=0,r=inf;
		while(l+1<r){
			ll mid=(l+r)/2;
			ll cnt=0;
			memset(sum,0,sizeof(sum));
			for(int i=1;i<=n;i++){
				sum[i]+=sum[i-1];
				ll tmp=sum[i]+fl[i];
				if(tmp<mid){
					cnt+=mid-tmp;
					sum[i]+=(mid-tmp);
					sum[i+w]-=(mid-tmp);
				}
			}
			if(cnt>m)r=mid;
			else l=mid;
		}
		printf("%d\n",l);
	}
	return 0;
}
