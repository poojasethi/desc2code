#include<iostream>
#define ll long long
using namespace std;
const int MX=100*1000+5;
int n,m,k;
int a[MX];
int s[MX];
int main()
{
	ios_base::sync_with_stdio(false);
	cin>>n>>m>>k;
	for(int i=0;i<n;++i)
	{
		cin>>a[i];
	}
	int t;
	ll st=1,en=2e9;
	while(st+1!=en)
	{
		int mid=(st+en)/2;
		ll cur=0,sum=0;
		for(int i=0;i<n;++i)
		{
			cur-=s[i];
			t=max(mid-(cur+a[i]),0ll);
			sum+=t;
			cur+=t;
			s[min(n,i+k)] = t;
		}
		if(sum>m)
		{
			en=mid;
		}
		else
		{
			st=mid;
		}
	}
	cout<<st<<endl;
	return 0;
}
