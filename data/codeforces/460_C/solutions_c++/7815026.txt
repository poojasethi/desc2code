#include<cstdio>
#include<iostream>
#include<algorithm>
#define MAXN 100000

using namespace std;
int m,n,w,a[MAXN+80];
long long S[MAXN+80],did[MAXN+80];
bool chk(int H)
{
	int i;
	for(i=1;i<=n;i++)
		did[i]=H-a[i];
	for(i=1;i<=n;i++)
	{
		S[i]=S[i-1];
		did[i]-=S[i];
		if(i-w>0)
			did[i]+=S[i-w];
		if(did[i]>0)
			S[i]+=did[i];
	}
	return S[n]<=m;
}
int main()
{
    int i,l=0,r=0,mid;
	scanf("%d%d%d",&n,&m,&w);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if(r<a[i])
			r=a[i];
	}
	r+=m;
	while(l<r)
	{
		mid=r-(r-l)/2;
		if(chk(mid))
			l=mid;
		else
			r=mid-1;
	}
	printf("%d",l);
    return 0;    
}
