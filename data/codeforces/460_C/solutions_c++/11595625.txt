#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
const int MAXN=100010;
int a[MAXN],n,m,w,st[MAXN];
inline bool judge(int lim)
{
	int cnt=0,now=0;
	for(int i=1;i<=n;i++)
	{
		st[i]=0;
		if(i-w>0)now-=st[i-w];
		int temp=lim-a[i];
		if(now<temp)
		{
			cnt+=(st[i]=temp-now);
			now=temp;
			if(cnt>m)return false;
		}
	}
	return true;
}
int main(void)
{
	scanf("%d%d%d",&n,&m,&w);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	int l=1,r=1000100000;
	while(l<=r)
	{
		int mid=l+r>>1;
		if(judge(mid))l=mid+1;
		else r=mid-1;
	}
	printf("%d\n",l-1);
	return 0;
}
