#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

const int INF = 2e9;
const int MAXN = 100005;
int N, M, W, A[MAXN], inc[MAXN];

bool check(int m){
  ll use = 0, cur = 0;
  for(int i = 0; i < N; i++){
    if(i >= W) cur -= inc[i - W];
    ll d = max(m - A[i] - cur, 0LL);
    inc[i] = d;
    use += d;
    cur += d;
  }
  return use <= M;
}

int main(){
  cin >> N >> M >> W;
  for(int i = 0; i < N; i++) cin >> A[i];
  int lo = 1, hi = INF, mid = lo + (hi - lo) / 2;
  while(lo < mid){
    (check(mid) ? lo : hi) = mid;
    mid = lo + (hi - lo) / 2;
  }
  cout << lo << '\n';
}
