#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,w,a1[100000],a2[100000];
bool f(int x)
{
	int sum = 0, ans = m;
	for(int i = 0; i < n; ++i){
        if(i>=w) sum -= a2[i-w];
        a2[i] = max(0, x - sum -a1[i]);
        sum += a2[i]; ans -= a2[i];
        if(ans<0) return 0;
	}
	return 1;
}
int main()
{
	int a=1000000001,b;
	scanf("%d%d%d",&n,&m,&w);
    for(int i=0;i<n;++i){
        scanf("%d",&a1[i]);
        a = min(a, a1[i]);
    }
    b = a + m + 1;
    while(b-a>1){
        int c = (a + b)/2;
        if(f(c)) a  = c;
        else b = c;
    }
	printf("%d\n",a);
	return 0;
}
