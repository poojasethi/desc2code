#include<bits/stdc++.h>
using namespace std;
#define pb push_back
int n ,m,w,mn=INT_MAX,x,inp[100003],cur[100003];
bool check(int x)
{
    int foo = m,ates=0;
    for(int i =0;i<n;i++)
    {
        if(i>=w)ates-=cur[i-w];
        cur[i] = max(0,x-inp[i]-ates);
        foo-=cur[i],ates+=cur[i];
        if(foo<0)return false;
    }
    return true;
}
int bs(int i ,int j)
{
    if(i==j)return i;
    int mid =(i+j)/2;
    if(check(mid))return bs(mid+1,j);
    else return bs(i,mid);
}
int main()
{
    cin >> n >>m >> w;
    for(int i =0;i<n;i++){cin >> inp[i];mn=min(mn,inp[i]);}
    cout << bs(mn,mn+m+1)-1<<endl;
}
