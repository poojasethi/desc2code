#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#define maxn 301000
#define inf 2000000000
#define ll long long
using namespace std;
ll n,m,w;
ll fl[maxn];
ll sum[maxn];
int main(){
	while(cin>>n>>m>>w){
		for(int i=1;i<=n;i++)
		{
			cin>>fl[i];
		}
		ll l=0;
		ll r=inf;
		while(l+1<r){
			ll mid=(l+r)/2;
			ll cnt=0;
			memset(sum,0,sizeof(sum));
			for(int i=1;i<=n;i++)
			{
				sum[i]+=sum[i-1];
				ll tmp=sum[i]+fl[i];
				if(tmp<mid)
				{
					cnt+=mid-tmp;
					sum[i]+=(mid-tmp);
					sum[i+w]-=(mid-tmp);
				}
			}
			if(cnt>m)r=mid;
			else l=mid;
		}
		cout<<l<<endl;
	}
}
