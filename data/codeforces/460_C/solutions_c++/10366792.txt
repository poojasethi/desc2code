#include<stdio.h>
#include<string.h>
int a[100010];
long long b[200020];
int main()
{
	int n,m,i;
	long long w,mid,min=1,max=1<<30,hs,cnt,sum;
	while(scanf("%d %d %d",&n,&m,&w)!=EOF)
	{
		for(i=0;i<n;i++)
		   scanf("%d",&a[i]);
		while(min<=max)
		{
			mid=(min+max)/2;
			cnt=0;sum=0;
			memset(b,0,sizeof(b));
			for(i=0;i<n;i++)
			{
				cnt-=b[i];
				hs=mid-a[i]-cnt;
				if(hs>0)
				{
					cnt+=hs;
					b[i+w]=hs;
					sum+=hs;
				}
			}
			if(sum>m)
			    max=mid-1;
			else
			    min=mid+1;
		}
		printf("%I64d\n",max);
	}
	return 0;
}
