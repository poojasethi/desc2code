#include <cstdio>

using namespace std;

const int MAXN = 1e5 + 1;

int niz[MAXN], w, m, n, uzi[2*MAXN];

int test(int x) {
	for(int i=0; i<=n+w; ++i) uzi[i] = 0;
	
	int curr=0;
	int spent=0;
	for(int i=0; i<n; ++i) {
		curr-=uzi[i];
		if(niz[i]+curr < x) {
			int raz=(x-(niz[i]+curr));
			spent+= raz;
			uzi[i+w]+=raz;
			curr+=raz;
		}
		if(spent>m) return 0;
	}
	return 1;
}

int main ()
{
	scanf("%d %d %d", &n, &m, &w);
	
	for(int i=0; i<n; ++i) scanf("%d", &niz[i]);
	
	int lo=1, hi= (1e9 + 1e6);
	
	while(lo<hi) {
		int mid = (lo+hi+1)/2;
		if(test(mid)) lo = mid;
		else hi = mid-1;
		
	}
	
	printf("%d", lo);
	
	return 0;
}
