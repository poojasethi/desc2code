#include <bits/stdc++.h>
#define long long long
using namespace std;

const int N = 1e5+8;
int a[N], b[N], n, m, w;

bool ok(int lim){
	memset(b, 0, sizeof b);
	int cur = 0, sum = 0;
	for(int i=1; i<=n; i++){
		cur -= b[i];
		if(a[i]+cur < lim){
			sum -= a[i] + cur - lim;
			if(sum > m) return 0;
			if(i+w < N) b[i+w] -= a[i] + cur - lim;
			cur -= a[i]+cur-lim;
		}
	}
	if(sum <= m) return 1;
	return 0;
}

int main(){
	//freopen("input.in", "r", stdin);
	scanf("%d %d %d", &n, &m, &w);
	for(int i=1; i<=n; i++) scanf("%d", a+i);
	int l = 1, r = 1e9+N;
	while(l < r){
		int m = l+r+1 >> 1;
		if(ok(m)) l = m;
		else r = m-1;
	}
	printf("%d", l);
}
