#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;
typedef long long ll;
int a[111111],b[111111];

int main(){
	int n,m,w;
	cin>>n>>m>>w;
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	int lo=1,hi=1111111111;
	while (hi-lo>1){
		int mi=(lo+hi)/2;
		ll d=0,s=0;
		for (int i=1;i<=n;i++){
			d+=b[i];
			b[i]=0;
			int t=0;
			if (a[i]+d<mi) t=mi-a[i]-d;
			d+=t;
			s+=t;
			if (i+w<=n) b[i+w]=-t;
		}
		if (s<=m) lo=mi; else hi=mi;
	}
	cout<<lo<<endl;
}
