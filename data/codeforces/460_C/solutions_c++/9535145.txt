#include <iostream>
using namespace std;
typedef long long ll;
ll n,m,w;
ll a[100005];
ll cnt[100005];
bool check(ll h) {
    ll tot=0;
    ll sum=0;
    for(int i=1;i<=n;i++) {
        if(i-w>0) tot-=cnt[i-w];
        cnt[i]=max(0ll,h-a[i]-tot);
        tot+=cnt[i];sum+=cnt[i];
        if(sum>m) return 0;
    }
    return 1;
}
int main() {
    cin>>n>>m>>w;
    ll tmp=~0u>>1;
    for(int i=1;i<=n;i++) cin>>a[i],tmp=min(tmp,a[i]);
    ll l=tmp,r=tmp+m+1;
    ll mid;
    while(l<r) {
        mid=(l+r)>>1;
        if(check(mid)) l=mid+1;
        else r=mid;
    }
    cout<<l-1<<endl;
    return 0;
}
