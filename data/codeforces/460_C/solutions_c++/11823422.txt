#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll a[100010],b[100010];

int main()
{
	ios_base::sync_with_stdio(0);
	ll n,m,w,f,s,e,mid,temp,ans;
	cin>>n>>m>>w;
	for(int i=0;i<n;i++)
		cin>>a[i];
	s=1;
	e=1000100000;
	while(s<=e)
	{
		mid=(s+e)/2;
		f=ans=temp=0;
		for(int i=0;i<n;i++)
		{
			b[i]=0;
			if(i>=w)
				temp-=b[i-w];
			if(a[i]+temp<mid)
			{
				b[i]=mid-a[i]-temp;
				ans+=mid-a[i]-temp;
				temp+=mid-a[i]-temp;
			}
			if(ans>m)
			{
				f=1;
				break;
			}
		}
		if(f)
			e=mid-1;
		else
			s=mid+1;
	}
	cout<<s-1<<"\n";
	return 0;
}
