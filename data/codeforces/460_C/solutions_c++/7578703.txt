#include<cstdio>
#include <algorithm>
using namespace std;
int a[100000],d[100000],n,m,w;
bool judge(int x){
    int add=0,tm=m;
    for(int i=0;i<n;i++){
        if(i>=w)add-=d[i-w];
        d[i]=max(0,x-a[i]-add);
        add+=d[i];tm-=d[i];
        if(tm<0)return false;
    }
    return true;
}
int main(){
    scanf("%d%d%d",&n,&m,&w);
    int low=0x7fffffff,high=0;
    for(int i=0;i<n;i++){scanf("%d",a+i);low=min(a[i],low);}
    high=low+m+1;
    while(low<high){
        int mid=(high+low)>>1;
        if(judge(mid)){
           low=mid+1;
        }
        else high=mid;
    }
    printf("%d\n",high-1);
    return 0;
}
