#include<cstdio>
#include<algorithm>
#define rep(i,a,b) for (int i=a;i<=b;++i)
#define N 100005
typedef long long ll;
using namespace std;
int m,n,w,a[N],need[N],grow[N];
bool check(int x) {
	rep(i,1,n) need[i]=max(0,x-a[i]),grow[i]=0;
	int now=0;
	ll res=0;
	rep(i,1,n) {
		if (i>=w) now-=grow[i-w];
		if (now<=need[i]) grow[i]=need[i]-now,res+=grow[i],now=need[i];
	}
	return res<=m;
}
int main() {
	int l=0,r=1010000000;
	scanf("%d%d%d",&n,&m,&w);
	rep(i,1,n) scanf("%d",&a[i]);
	while (l<r) {
		int md=(l+r+1)>>1;
		if (check(md)) l=md;
		else r=md-1;
	}
	printf("%d\n",l);
	return 0;
}
