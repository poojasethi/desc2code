#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
typedef long long int64;

int n,m,w,a[110000];
int64 f[210000];

int can(int64 ans){
	memset(f,0,sizeof(f));
	int64 acc=0;
	int64 tot=0;
	for(int i=1;i<=n;i++){
		acc+=f[i];
		int64 delta;
		if(0<(delta=ans-acc-a[i])){
			if(m-delta<tot){return 0;}
			tot+=delta;
			acc+=delta;
			f[i+w]-=delta;
		}
	}
	return 1;
}

int main(){
	scanf("%d%d%d",&n,&m,&w);
	for(int i=1;i<=n;i++){scanf("%d",a+i);}
	int64 L=1,R=2000000000;
	while(L<R){
		int64 M=L+((R-L)>>1);
		if(!can(M)){
			R=M;
		}else{
			L=M+1;
		}
	}
	cout<<L-1<<endl;
	return 0;
}
