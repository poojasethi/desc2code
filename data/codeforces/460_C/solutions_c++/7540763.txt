#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
typedef long long ll;
typedef double db;

const int N=2e5+5,M=1e5+5;

int n,m,w,a[N];
ll b[N];

int main(){
	scanf("%d%d%d",&n,&m,&w);
	for(int i=1;i<=n;i++) scanf("%d",a+i);
	int l=1,r=1e9+1e5+5;
	while(r-l>1){
		int mid=l+r>>1;
		ll now=0,d=0;
		for(int i=1;i<=n;i++) b[i]=0;
		for(int i=1;i<=n;i++){
			d+=b[i];
			ll x=max(0LL,mid-(a[i]+d));
			d+=x;
			b[i]+=x;
			b[i+w]-=x;
			now+=x;
		}
		if(now>m)
			r=mid;
		else
			l=mid;
	}
	printf("%d\n",l);
}
