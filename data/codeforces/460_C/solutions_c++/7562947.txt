#include <iostream>
using namespace std;

const int MAXn = 100*1000 + 5;
int n, m, k;
int ar[MAXn];
int s[MAXn];

typedef long long LL;

int main()
{
	ios_base::sync_with_stdio(false);
	cin >> n >> m >> k;
	for(int i=0; i<n; ++i)
		cin >> ar[i];

	int t;
	LL st=1, en=2e9;
	while(st+1 != en) {
		int mid = (st + en) / 2;
		LL cur = 0, sum = 0;
		for(int i=0; i<n; ++i) {
			cur -= s[i];
			t = max(mid-(cur+ar[i]), 0ll);
			sum += t;
			cur += t;
			s[min(n, i+k)] = t;
		}

		if(sum > m)
			en = mid;
		else
			st = mid;
	}

	cout << st << endl;

	return 0;
}
