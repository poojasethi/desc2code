#include <cstdio>
#include <cstdlib>
using namespace std;

long long n,m,w,ans,l,r,a[100001],f[200001];

int main()
{
	scanf("%I64d%I64d%I64d",&n,&m,&w);
	for (int i=1;i<=n;i++) scanf("%I64d",&a[i]);
	l=1; r=1500000000;
	while (l<=r)
	{
		int mid=(l+r)>>1;
		long long need=0,sum=0;
		for (int i=1;i<=n;i++) f[i]=0;
		for (int i=1;i<=n;i++)
		{
			sum+=f[i];
			if (sum<mid-a[i])
			{
				need+=mid-a[i]-sum; f[i+w]-=(mid-a[i]-sum); 
				sum+=mid-a[i]-sum;
			}
		}
		if (need>m) r=mid-1;
		else {
			if (mid>ans) ans=mid;
			l=mid+1;
		}
	}
	printf("%I64d\n",ans);
}
