#include <bits/stdc++.h>

#define NN 100010
using namespace std;
typedef long long lld;

lld a[NN], b[NN];
lld sum[NN];

int main (){
	
	int N, M, W, i;
	scanf("%d %d %d", &N, &M, &W);
	lld st=1000000000, ed, mid;
	
	for(i=1; i<=N; i++){
		scanf("%I64d", a+i);
		st=min(st, a[i]);
	}
	
	ed=st+M+1;
	while(ed-st>1){
		mid=(st+ed)/2;
		for(i=1; i<=N; i++) b[i]=max((lld)0, mid-a[i]);
		for(i=1; i<=N; i++){
			sum[i]=sum[i-1];
			if(sum[i-1]-sum[max(0, i-W+1)] >=b[i]) continue;
			sum[i]+=(b[i]-sum[i-1]+sum[max(0, i-W)]);
		}
		if(sum[N]>(lld)M) ed=mid;
		else st=mid;
	}
	
	printf("%I64d\n", st);
		
	return 0;
}
