#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define N (301*1000)
ll n,m,w,a[N],dp[N];

bool check(ll x)
{
	ll now=m;
	for(int i=0;i<N;i++)dp[i]=0;
	for(int i=0;i<n;i++)
	{
		if(i)dp[i]+=dp[i-1];
		now-=max(0ll,x-a[i]-dp[i]);
		if(now<0)
			return 0;
		dp[i+w]-=max(0ll,x-a[i]-dp[i]);
		dp[i]+=max(0ll,x-a[i]-dp[i]);
	}
	return 1;
}

int main()
{
    cin>>n>>m>>w;
    for(int i=0;i<n;i++)cin>>a[i];
    //cout<<check(9);
    
    ll l=0,r=1e14;
    while(l<r-1)
    {
    	ll mid=l+r>>1;
    	if(check(mid))
    		l=mid;
    	else
			r=mid;	
	}
	cout<<l;
	
    return 0;
}
