#include <bits/stdc++.h>
using namespace std;
#define NN 100010
typedef long long int ll;
ll a[NN], b[NN];
ll sum[NN];
int main ()
{
	ll N,M,W,i;
	cin>>N>>M>>W;
	ll st=1e10, ed, mid;
	for(i=1;i<=N;i++){cin>>a[i];st=min(st, a[i]);}
	ed=st+M+1;
	while(ed-st>1)
    {
		mid=(st+ed)/2;
		for(i=1; i<=N; i++){b[i]=max(0LL, mid-a[i]);}
		for(i=1; i<=N; i++)
        {
			sum[i]=sum[i-1];
			if(sum[i-1]-sum[max(0LL, i-W+1)] >=b[i]){continue;}
			sum[i]+=(b[i]-sum[i-1]+sum[max(0LL, i-W)]);
		}
		if(sum[N]>M){ed=mid;}
		else st=mid;
	}
	cout<<st<<endl;
	return 0;
}
