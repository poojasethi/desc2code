#include<iostream>
#include<cstdio>
using namespace std;
int n,m,w,a[100005],a1[100005],i,j;
int f(int x) {
    int t=0,b=m;
    for(i=1; i<=n; i++) {
        if(i>w) t-=a1[i-w];
        a1[i] = max(0,x-a[i]-t);
        t+=a1[i]; b-=a1[i];
        if(b<0) return 0;
    }
    return 1;
}

int main() {
    int mi=1e9+1;
    scanf("%d%d%d",&n,&m,&w);
    for(i=1; i<=n; i++) {
        scanf("%d",&a[i]);
        mi = min(a[i],mi);
    }
    int ma = mi+m+1;    
        while(mi < ma) {
            int ba = mi+ma>>1;
            if(f(ba)) mi = ba+1;
            else ma = ba;
        }
        cout<<mi-1<<endl;
        return 0;
}
