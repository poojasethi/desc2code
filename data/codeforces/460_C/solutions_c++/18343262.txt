#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
const int N = 100001;
ll n, m, w, a[N], ps[N];

bool check(ll k) {
	fill(ps, ps + N, 0);
	ll p = 0, moves = 0;
	for (int i = 0; i < n; i++) {
		if (k - p - a[i] + ps[i] >= 1) {
			moves += k - p - a[i] + ps[i];
			p += k - p - a[i] + ps[i];
		}
		if (i + w < n)
			ps[i + w] = p;
	}
	return moves <= m;
}

main() {
	cin >> n >> m >> w;
	for (int i = 0; i < n; i++)
		scanf("%d", a + i);
	ll l = 0, r = 1e10;
	while (l < r) {
		ll mid = (l + r + 1) >> 1;
		if (check(mid))
			l = mid;
		else r = mid - 1;
	}
	cout << l << endl;
}
