#include<bits/stdc++.h>
using namespace std;
long long int m,n,w,a[100080],S[100080],did[100080];
bool chk(int H)
{
	int i;
	for(i=1;i<=n;i++)
		did[i]=H-a[i];
	for(i=1;i<=n;i++)
	{
		S[i]=S[i-1];
		did[i]-=S[i];
		if(i-w>0)
			did[i]+=S[i-w];
		if(did[i]>0)
			S[i]+=did[i];
	}
	return S[n]<=m;
}
int main()
{
    int i,l=0,r=0,mid;
	cin>>n>>m>>w;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		if(r<a[i])
			r=a[i];
	}
	r+=m;
	while(l<r)
	{
		mid=r-(r-l)/2;
		if(chk(mid))
			l=mid;
		else
			r=mid-1;
	}
	printf("%d",l);
    return 0;
}
