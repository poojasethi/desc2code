#include<cstdio>
#include<cstring>
int a[100005];
long long b[200005];
int main()
{
	int n,m,i;
	long long mid,r=1010000020,l=1,cnt,sum,js,w;
	scanf("%d%d%d",&n,&m,&w);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	while(l<=r)
	{
		memset(b,0,sizeof(b));
		sum=0;
		mid=(l+r)/2;cnt=0;
		for(i=1;i<=n;i++)
		{
			cnt=cnt-b[i];
			js=mid-a[i]-cnt;
			if(js>0)
			{
				b[i+w]=js;
				cnt=cnt+js;
				sum=sum+js;
			}
		}
		if(sum>m)
			r=mid-1;
		else
			l=mid+1;
		}
	printf("%I64d\n",r);
	return 0;
}
