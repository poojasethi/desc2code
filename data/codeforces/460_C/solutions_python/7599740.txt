def read_ints():
    return list(int(i) for i in raw_input().split())

def real(level):
    bounds = [0] * n
    carry = 0
    total_water = 0
    for i, val in enumerate(a):
        carry -= bounds[i]
        cur_level = val + carry
        if cur_level < level:
            water = level - cur_level
            carry += water
            total_water += water
            if total_water > m:
                return False
            if i + w < n:
                bounds[i + w] += water
    return True


n, m, w = read_ints()
a = read_ints()

l = 0
r = 2 * 10**9

while r - l > 1:
    x = (l + r) / 2
    if real(x):
        l = x
    else:
        r = x
