n, m, w = map(int, raw_input().split());
a = list(map(int, raw_input().split()));

xiao = 1;
da = 1000100000;
ans = -1;

while (xiao <= da):
	zhong = (xiao + da) >> 1;
	tmpM = m;
	b = [0] * n;
	jia = 0;
	for i in range(n):
		jia += b[i];
		#print (" ", jia);
		deng = jia + a[i];
		if (deng < zhong):
			tmpM -= zhong - deng;
			jia += zhong - deng;
			if (i + w < n):
				b[i + w] -= zhong - deng;
	if (tmpM >= 0):
		xiao = zhong + 1;
		ans = zhong;
	else:
		da = zhong - 1;

print (ans);
