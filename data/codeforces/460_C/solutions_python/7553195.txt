n,m,w=map(int,raw_input().split())
a=map(int,raw_input().split())
le,ri=0,2000000000

def check(x):
    tmp,cost=0,0
    k=[0]*(n+w)
    for i in range(0,n):
        tmp+=k[i]
        if(a[i]+tmp<x):
            cost+=x-a[i]-tmp
            k[i+w]-=x-a[i]-tmp
            tmp+=x-a[i]-tmp
    return cost<=m

while(le<ri):
    mid=le+ri+1>>1
    if(check(mid)):
        le=mid
    else:
        ri=mid-1
print(le)
