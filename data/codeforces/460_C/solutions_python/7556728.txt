R=lambda:map(int,raw_input().split())
n,m,w=R()
a=R()
l,r=min(a),min(a)+10**6
def check(x):
	nd,gr=[],[]
	for i in xrange(n): nd.append(max(x-a[i],0))
	sc,rs=0,0
	for i in xrange(n):
		if i>=w: sc-=gr[i-w]
		if sc<=nd[i]:
			gr.append(nd[i]-sc)
			rs+=gr[i]
			sc=nd[i]
		else: gr.append(0)
	return rs<=m
while l+1<r:
	md=(l+r)/2
	if check(md):l=md
	else:r=md
print l
