import sys

def Can(a, m, w, desired_height):
	days_left = m
	current_height = 0
	heights = []
	for i in range(0, len(a)):
		if i >= w:
			current_height -= heights[i - w]
		current_value = a[i] + current_height
		if current_value < desired_height:
			days_needed = desired_height - current_value
			if days_needed > days_left:
				return False
			days_left -= days_needed
			heights.append(days_needed)
		else:
			heights.append(0)
		current_height += heights[i]
	return True

def main():
	n, m, w = map(int, raw_input().split())
	a = map(int, raw_input().split())
	lo = 1
	hi = 1000100001
	while lo < hi:
		mi = (lo + hi + 1) / 2
		if Can(a, m, w, mi):
			lo = mi
		else:
			hi = mi - 1
	print lo

main()
