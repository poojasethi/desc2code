#include <cstdio>
#include <algorithm>

using namespace std;

int grund[30]=
{
    0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12
};
char ok[1000505];

int n;
int sol;
int sumed;

int main()
{
    scanf("%d",&n);
    sol=1;
    sumed=1;
    for (int i=2; 1ll*i*i<=n; i++)
    if (ok[i]==0)
    {
        int cnt=1;
        int now=i;
        while (1ll*i*now<=n)
        {
            if (now<1000200) ok[now]=1;
            now=now*i;
            cnt++;
        }
        sumed+=cnt;
        sol^=grund[cnt];
    }
    if ((n-sumed)&1) sol^=1;
    if (sol==0) printf("Petya\n"); else printf("Vasya\n");


}
