#include<cstdio>
#define rep(i,n) for (int i=0;i<n;i++)
const int sg[]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
bool f[100005];
int Sg,N,M,C;
long long i,j;
int main()
{
	scanf("%d",&N),M=N;
	for(i=2;i*i<=N;++i)
		if(!f[i])
		{
			C=0;
			for(j=i;j<=N;j*=i,++C)
				if(j*j<=N)
					f[j]=1;
			M-=C,Sg^=sg[C];
		}
	Sg^=M&1;
	if(Sg) printf("Vasya\n");
	else printf("Petya\n");
	return 0;
}
