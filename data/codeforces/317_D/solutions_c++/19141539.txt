#include <cstdio>
#include <cmath>

using namespace std;

int sg[] = {0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12}, calced[40000];

int main() {
	int n, r, i, s, l, ans = 1;
	long long t;
	
	scanf("%d", &n);
	r = int(ceil(sqrt(double(n))));
	
	l = n - 1;
	for (i = 2; i <= r; ++i)
		if (!calced[i]) {
			t = i;
			s = 0;
			while (t <= n) {
				++s;
				if (t <= r) calced[t] = true;
				t *= i;
			}
			ans ^= sg[s];
			l -= s;
		}
	
	ans ^= (l & 1);
	if (ans) printf("Vasya\n"); else printf("Petya\n");
	
	return 0;
}
