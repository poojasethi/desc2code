#include<stdio.h>
#include<map>
#include<set>
#define LL long long
#define half 31625
using namespace std;
int used[half];
int sg[32] = {0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
int main ()
{
    int n,k,i,j,ans = 0,sum=0;
    scanf("%d",&n);
    for(i=2;i*i<=n;i++)
    {
        if(used[i])
        continue;
        for(j=i,k=1;(LL)j*i<=n;j*=i,k++)
        { if(j<half) used[j]=1;}
        sum+=k;
        ans^=sg[k];
        
    }
    if((n-sum)&1)
    ans^=sg[1];
    if(ans)
    printf("Vasya");
    else
    printf("Petya");
    //scanf("%d",&n);
    return 0;
}
