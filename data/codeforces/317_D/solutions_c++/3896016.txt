#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
#include <cassert>
using namespace std;
int n;
int nimv[33] = {0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12, 14, 13, 1};
vector<bool> v;

int main(){
    cin >> n;
    n++;
    v.resize(n);
    int left = n;
    int nv = 1;
    for(int i=2; i*i<=n; i++){
        long long j=i;
        if(v[i]) continue;
        int cnt = 0;
        while(j<n){
            cnt++;
            if(!v[j]) left--;
            v[j] = 1;
            j *= i;
        }
        nv ^= nimv[cnt];
    }
    if(left%2) nv ^= 1;
    if(nv) cout << "Vasya" << endl;
    else cout << "Petya" << endl;
}
