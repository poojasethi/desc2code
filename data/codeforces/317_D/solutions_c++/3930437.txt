#include <stdio.h>
#include <math.h>
#define R (1<<16)
int nim[30]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
bool chk[R];
int main(){
//	freopen("1.in","r",stdin);
//	freopen("1.out","w",stdout);
	int n, i, j, k, r, ans=0, cnt;
	scanf("%d", &n), r=sqrt(n-1)+1, cnt=n;
	for(i=2; i<=r; i++)if(!chk[i]){
		for(k=1, j=0; k<=n/i; j++, k*=i)if(k<R)chk[k]=1;
		ans^=nim[j], cnt-=j;
	}
	if(cnt&1) ans^=1;
	if(ans) puts("Vasya");
	else puts("Petya");
}
