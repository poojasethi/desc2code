#include <iostream>
#include <cstdio>

using namespace std;

const int G[] = { 0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12 } ;
const int N = 100000;
bool used[N];

int main()
{
    int n;
    scanf("%i", &n);
    int res = 0, cnt = 0;
    for(int i = 2; i * i <= n; i++)
        if(!used[i])
        {
            int x = 1;
            if(i < N) used[i] = true;
            for(long long j = i * i; j <= n; j *= i) { if(j < N) used[j] = true; x++; }
            cnt += x;
            res ^= G[x];
        }
    res ^= (n - cnt) % 2;
    //printf("%i\n", res);
    printf("%s\n", res ? "Vasya" : "Petya");
    return 0;
}
