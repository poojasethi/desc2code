#include <cstdio>
#include <cstring>
const int N = 100005;
bool f[N];
int sg[31]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12,15};

int main() {
	int n;
	scanf("%d", &n);
	int s = n, ans = 0;
	for(int i = 2; i * i <= n; i ++) {
		if(!f[i]) {
			int cnt = 0;
			long long tmp = 1;
			while(tmp * i <= n) {
				tmp *= i;
				cnt++;
				if(tmp < N) f[tmp] = true;
			}
			s -= cnt;
			ans ^= sg[cnt];
		}
	}
	ans ^= s % 2;
	if(ans) puts("Vasya");
	else puts("Petya");
	return 0;
}
