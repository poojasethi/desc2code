#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

#define IO(a,b) freopen(a,"r",stdin);\
				freopen(b,"w",stdout)
#define FOR(i,a,b) for(int i=a;i<=b;++i)
#define LL long long

int n,_n,ans;
int SG[]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
bool V[31630];

int main(){
	//IO("test.in","test.out");
	
	scanf("%d",&n);
	_n=n;
	
	for(int i=2;i*i<=n;++i) if(!V[i]){
		int t=(int)floor(log(n)/log(i));
		ans^=SG[t];
		_n-=t;
		for(int j=i;(LL)j*j<=n;j*=i) V[j]=true;
	}
	
	ans^=_n&1;
	
	if(ans) printf("Vasya\n");
	else printf("Petya\n");
	
	return 0;
}
