#include<cstdio>
#include<cmath>
int sg[]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12,14};
int n,m,ans,sum,done[10000005];
int main(){
    scanf("%d",&n); m=n; ans=0;
    sum=(int)sqrt(n);
    for(int i=2;i<=sum;i++)
    if(!done[i]){
        int c=0;
        for(long long j=i;j<=n;j*=i){
            c++;
            if(j*j<=n) done[j]=1;
        }
        m-=c;
        ans^=sg[c];
    }
    ans^=(m&1);
    if(ans) printf("Vasya\n");else printf("Petya\n");
}
