#include<cstdio>
#include<cstring>
#include<algorithm>
typedef long long LL;
#include<iostream>
using namespace std;
const int MAX=100+10;
int f[]={0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12};
int n;
int hash[1000000];
int main(){
	int i;
	scanf("%d",&n);
	int ans=1,res=0;
	for(i=2;i*i<=n;++i)
	{
		if(hash[i])
			continue;
		LL u=i;
		int sum=0;
		while(u<=n)
		{
			++sum;
			if((LL)u*u>n)
				++res;
			else hash[u]=1;
			u*=i;
		}
		ans^=f[sum];
	}
	ans^=(n-i+1-res)%2;
	if(!ans)cout<<"Petya"<<endl;
	else cout<<"Vasya"<<endl;
}
