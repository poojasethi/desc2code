#include <bits/stdc++.h>

using namespace std;

map <int,int> mp;
int g[40]={ 0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12};
int N,f[40],ans,used[320000];

int main(){
//    freopen("out.c","w",stdout);
    scanf("%d",&N);

    f[1]=N;

    for(int i=2,cnt=0;i*i<=N;i++,cnt=0) if(!used[i]){
        long long k=i;
        while(k<=N){
            cnt++ , k*=i;
            if(k<32000) used[k]=1;
        }
        f[1]-=cnt;
        f[cnt]++;
    }

    for(int i=1;i<=29;i++) if(f[i]&1)
        ans^=g[i];

    printf("%s\n",ans ? "Vasya" : "Petya");

    return 0;
}
