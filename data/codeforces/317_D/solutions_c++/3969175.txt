#include <cstdio>
#include <cmath>

int gr[] = { 0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12 };

bool v[33000];

int main() {
  int n;
  scanf("%d", &n);

  const int limit = sqrt(n) + 3;
  int x = 0, found = 0;
  for (int i = 2; i*i <= n; ++i) if (!v[i]) {
    int len = 0;
    for (long long j = i; j <= n; ++len, j *= i) { ++found; if (j <= limit) v[j] = true; }
    x ^= gr[len];
  }
  if ((n-found) & 1) x ^= 1;
  puts(x ? "Vasya" : "Petya");
  return 0;
}
