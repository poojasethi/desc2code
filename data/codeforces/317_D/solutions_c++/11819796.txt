#include<stdio.h>
#include<string.h>
using namespace std;
const int maxn=40000;
int f[30]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
bool check[maxn];
int main()
{
	int n,done=0,sg=0;
	scanf("%d",&n);
	memset(check,false,sizeof(check));
	for (int i=2;i*i<=n;i++)
	{
		if (check[i]) continue;
		int cnt=0;
		for (long long j=i;j<=n;j*=i)
		{
			if (j<maxn) check[j]=true;
			cnt++;
		}
		sg=sg^f[cnt];
		done+=cnt;
	}
	if ((n-done)&1) sg^=1;
	if (sg==0) printf("Petya\n");
	else printf("Vasya\n");
	return 0;
}
