#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std ;

const int MAXN = (int)1e5+10 ;

int x ;
int n ;
int visit[MAXN] ;

int dp[]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12,14,13,1};

int main(){
    scanf("%d",&n);
    int sg=0; sg^=1;int last=0;
    for (int i=2;i*i<=n;i++){
        if (visit[i]) continue;
        long long j,t;
        for (j=i,t=1;(long long)j*i<=n;j*=i,t++){
            if (j<MAXN) visit[j]=true;
        }
        last+=t;;sg^=dp[t];
    }
    sg^=((n-1-last)&1);
    if (sg) printf("Vasya\n");
    else printf("Petya\n");
    return 0;
}
