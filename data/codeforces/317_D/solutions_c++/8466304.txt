#include<cstring>
#include<cstdio>
#include<iostream>
#include<map>
using namespace std;

int sg[31]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12,14};

int b[100000]={0};

int main()
{
    long long n,sum;
    int s=0,x;
    long long i,j;
cin>>n;
    sum=n;
    for (i=2;i*i<=n;i++)
        if (b[i]==0)
    {
        j=i;x=0;
        while (j<=n)
        {
            x++;
            if (j*j<=n)
               {b[j]=1;}
            j*=i;
            sum--;
        }
        s^=sg[x];
    }

    s^=(sum&1);

    if (s==0) cout<<"Petya"<<endl;
    else cout<<"Vasya"<<endl;

    return 0;
}
