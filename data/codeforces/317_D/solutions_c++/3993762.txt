#include <algorithm>
#include <iostream>

using namespace std;

const int SqrtN=1E5;

int n,res,sg[]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
bool f[SqrtN];

int main()
{
	cin>>n;
	int sum=0;
	for(int i=2;i*i<=n;++i)
	{
		if(f[i]) continue;
		int x=0;
		for(long long j=i;j<=n;j*=i,++x)
			if(j<SqrtN)
			{
				f[j]=1;
			}
		res^=sg[x];
		sum+=x;
	}
	res^=(n-sum)%2;	
	cout<<(res ? "Vasya" : "Petya")<<endl;
	return 0;
}
