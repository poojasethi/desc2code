#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <set>
#include <vector>
#include <algorithm>
#define rep(i,n) for(i = 1; i <= n; i ++)
using namespace std;
set <long long> st;
int SG[30]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
int main()
{
    long long  n, i, j, tmp, ans = 0;
    cin >> n;
    for (i = 2; i * i <= n; i ++)
    if (!st.count(i))
    {
        tmp = 0; j = i;
        while (j <= n) {st.insert(j); j *= i; tmp ++; }
        ans ^= SG[tmp];
    }
    ans ^= (n - st.size()) & 1;
    if (ans) cout << "Vasya" << endl;
     else cout << "Petya" << endl;
    return 0;
}
