#include <cstdio>
#include <cstring>
#define LL long long
using namespace std;

const int d[]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
int n,tot,ans;
bool boo[1000010];

int main() {
	scanf("%d",&n);
	if (n==1) {puts("Vasya");return 0;}
	tot=n;
	for (int i=2;i*i<=n;++i)
		if (!boo[i]) {
			int tmp=0;
			for (LL j=i;j<=n;j*=(LL)i) {
				if (j*j<=n) boo[j]=true;
				++tmp;
			}
			tot-=tmp;
			ans^=d[tmp];
		}
	if (tot&1) ans^=1;
	puts(ans?"Vasya":"Petya");
	return 0;
}
