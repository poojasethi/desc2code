#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<math.h>
#include<map>
#include<set>
#include<iostream>
using namespace std;
typedef long long ll;

set<ll> S;

int sg[29]={1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
int main() {

	int n,i;
	scanf("%d",&n);
	int ans=0;
	for(i=2;i*i<=n;i++){
		if(S.find(i)!=S.end())
			continue;
		int cnt=0;
		for(ll t=i;t<=n;t*=i){
			cnt++;
			S.insert(t);
		}
		ans^=sg[cnt-1];
	}
	if((n-S.size())%2) ans^=1;
	if(ans) printf("Vasya\n");
	else printf("Petya\n");
}
