#include<stdio.h>
#include<math.h>

int DB[30]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
bool check[100005];

int main()
{
	int i,n,ans=0,tot;
	scanf("%d",&n);
	tot=n;
	for(i=2;i*i<=n;i++){
		if(check[i])continue;
		for(long long j=i;j*j<=n;j*=i){
			check[j]=true;
		}
		int cnt=int(log((double)n)/log((double)i)+1e-9);
		ans^=DB[cnt];
		tot-=cnt;
	}
	if(tot%2!=0)ans^=1;
	if(ans!=0)printf("Vasya");
	else printf("Petya");
}
