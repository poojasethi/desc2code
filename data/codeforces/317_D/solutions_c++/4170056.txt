#include <cstdio>
#include <map>
using namespace std;
const int NMax=100000;
int SG[50]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
int N;
map<long long ,int> use;
int main()
{
    scanf("%d",&N);
    long long ret=1,tot=1;
    for(int i=2;i*i<=N;i++) if(!use[i]){
    long long k=i,cnt=0;
        while(k<=N) {
            use[k]=1;tot++;
            cnt++;
            k*=i;
        }
        ret^=SG[cnt];
    }
    if(tot<N)ret^=(((long long)N-tot)&1);
    if(ret>0) puts("Vasya");
    else puts("Petya");
    return 0;
}
