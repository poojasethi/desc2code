#include<set>
#include<cmath>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,ans=1,sg[50]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
set<int>s;
int main(){
    scanf("%d",&n);
    int k=0,x,y;
    for(int i=2;i*i<=n;++i){if(s.count(i))continue;
        for(x=1,y=i;;++x){
            s.insert(y);
            if((long long)y*y>n)++k;
            if((long long)y*i>n)break;
            y*=i;
        }
        ans^=sg[x];
    }
    if((n-(int)sqrt(n)-k)%2!=0)ans^=1;
    if(ans>0)printf("Vasya\n");
    else printf("Petya\n");
    return 0;
}
