#include<iostream>
#include<cstdio>
#include<map>
using namespace std;

#define rep(i, s, t) for (int i = s; i <= t; ++i)
typedef long long LL;
typedef map<int, int> mi;
typedef mi::iterator mii;

const int sg[31] = {0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12, 14};
int n, ans[33];
mi M;

int main(){
    scanf("%d", &n);
    int lf = n, ans = 0;
    for (int i = 2; LL(i) * i <= n; ++i){
        if (M[i]) continue;
        int cnt = 0;
        for (LL k = i; k <= n; k = k * i)
            ++cnt, lf -= (M[int(k)] == 0), M[int(k)] = 1;
        ans ^= sg[cnt];
    }
    ans ^= (lf & 1);
    puts(ans ? "Vasya" : "Petya");
}
