#include <cassert>
#include <cstdio>
#include <algorithm>
#include <map>
#include <utility>
#include <vector>
using namespace std;

int val[] = {0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12 };
const int MX = (1e5);
bool vis[MX];

int main()
{
	int n;
	scanf("%d", &n);
	int res = 0, sc = 0;
	for(int i = 2; i*i <= n; ++i) {
		if(vis[i]) continue;
		int c = 0;
		long long j = i;
		while(j <= n) {
			if(j < MX) vis[j] = true;
			++c;
			j *= i;
		}
		res ^= val[c];
		sc += c;
	}
	res ^= ((n-sc) & 1);
	
	puts(res ? "Vasya" : "Petya");
	return 0;
}
