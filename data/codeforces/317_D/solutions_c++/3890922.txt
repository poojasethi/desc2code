set<long long> p;
int n;

int zzp(int x) {
	if (p.find(x) != p.end()) return 0;
	if (x == 1) {
		p.insert(x);
		return 1;
	}
	long long m = x;
	int ans = 0;
	while (m <= n) {
		p.insert(m);
		m *= x;
		ans++;
	}
	return ans;
}


int main() {
	scanf("%d", &n);
	int ans = 0;
	for (int i = 1; i*i <= n; i++) {
		ans = ans ^ table[zzp(i)];
	}
	ans ^= (n - p.size()) % 2;
	if (ans == 0) printf("Petya");
	else printf("Vasya");
}
