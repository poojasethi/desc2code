#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <time.h>
#include <queue>
#include <iostream>
using namespace std;
typedef long long ll;

bool hav[100005];
int val[] = {
        0,1,2,1,4,3,2,
        1,5,6,2,1,8,7,
		5,9,8,7,3,4,7,
        4,2,1,10,9,3,6,
        11,12,13
};

int main ()
{
//	freopen("aa.txt","r",stdin);
	ll n,i,now,j,mm,o;
	scanf("%I64d",&n);now=0;mm=n;
	memset(hav,false,sizeof(hav));
	for(i=2;i*i<=n;i++) {
		if(hav[i]) continue;
		o=i*i;if(o<100005)hav[o]=true;j=2;
		while(o<=n) {
			o*=i;if(o<100005)hav[o]=true;j++;
		}j--;
		mm-=j;now^=val[j];
	}
	if(mm&1) now^=1;
	if(now) cout<<"Vasya"<<endl;
	else cout<<"Petya"<<endl;
}
