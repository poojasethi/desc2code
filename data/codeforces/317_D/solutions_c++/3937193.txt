#include <cstdio>
typedef long long LL;
int d[] = {0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};

bool OK(int k)
{
	for(int i=2;i*i<=k;++i){
		int tk = k;
		while(tk%i==0) tk/=i;	
		if(tk==1) return false;
	}
	return true;
}


int main()
{

	int n,res=1;
	int sum=0;
	scanf("%d",&n);
	LL k = 1;
	while(k*k<n)k++;

	for(int i=2;i<=k;++i){
		if(!OK(i)) continue;
		LL j=i;
		int ct=0;
		while(j<=n) j>k ? ++sum : 0, ct++,j*=i;
		res ^= d[ct];
	}
	if( (n-k-sum) % 2 ) res ^= 1;
	puts(res ? "Vasya" : "Petya");
	
	return 0;	
}
