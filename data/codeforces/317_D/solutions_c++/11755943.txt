#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
using namespace std;
const int sg[30]={0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12};
int n,m,ans=0,K;
bool mark[100005];
int main()
{
	scanf("%d",&n);m=(int)sqrt(n)+1;K=n;
	memset(mark,0,sizeof(mark));
	for (int i=2;i<=m;i++)
	{
		long long now=i,h=0;
		if (mark[i]) continue;
		while (now<=n)
		{
			now=now*i;h++;K--;
			if (now<=m) mark[now]=1;
		}
		ans^=sg[h];
	}
	if (K&1) ans^=1;
	if (ans) puts("Vasya");
	else puts("Petya");
	return 0;
}
