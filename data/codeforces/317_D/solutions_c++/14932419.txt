#include <algorithm>

#include <stdio.h>

#include <memory.h>

#include <set>

#include <map>



using namespace std;



typedef long long INT;



int g[50]={0, 1, 2, 1, 4, 3, 2, 1, 5, 6, 2, 1, 8, 7, 5, 9, 8, 7, 3, 4, 7, 4, 2, 1, 10, 9, 3, 6, 11, 12};



set<int> st;



int main() {

	

	int G, N, k;

	INT i, j;

	

	G=0;

	scanf("%d", &N);

	

	for (i=2; i*i<=N; i++) {

		if (st.find(i)!=st.end()) continue;

		k=0;

		for (j=i; j<=N; j*=i) {

			st.insert(j);

			k++;

		}

		G^=g[k];

	}

	

	if ((N-st.size())&1) G^=1;

	

	if (G) puts("Vasya");

	else puts("Petya");

	

	return 0;

}
