#include <cstdio>
typedef long long ll;
const int M=100005,f[31]={
	0,1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12,14};
int n,m,cnt,ans;
bool flag[M];
bool work(int n){
	ans=0;
	for(m=0;(ll)m*m<=n;m++);
	m--;
	cnt=n;
	for(int i=2;i<=m;i++) if(!flag[i]){
		int x=0;
		for(long long j=i;j<=n;x++,j*=i){
			if(j<=m) flag[j]=1;
			cnt--;
		}
		ans^=f[x];
	}
	return ans^=cnt&1;
}
int main(){
	scanf("%d",&n);
	puts(work(n)?"Vasya":"Petya");
}
