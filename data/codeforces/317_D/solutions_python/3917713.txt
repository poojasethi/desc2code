sg = [1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12]
i = 2
ans = 1
n = input()
mp = {} 
while i * i <= n:
	if i in mp :
		i += 1
		continue
	t = i
	cnt = 0
	while t <= n:
		mp[t] = 1
		t *= i
		cnt += 1
	ans ^= sg[cnt - 1]
	i += 1
res = n - i + 1
for a in mp: 
	if a >= i : 
		res -= 1
ans ^= res % 2
if ans == 0 : print 'Petya'
else : print 'Vasya'
