sg=[1,2,1,4,3,2,1,5,6,2,1,8,7,5,9,8,7,3,4,7,4,2,1,10,9,3,6,11,12]
n=input()
cnt=[n]+[0]*30
S=int(n**0.5)
fg=[0]*(S+10)
for i in xrange(2,S+1):
	if fg[i]: continue
	c,s=i,0
	while c*i<=n:
		c*=i
		s+=1
		if c<=S: fg[c]=1
	cnt[0]-=s+1
	cnt[s]+=1
ans=0
for i in xrange(29):
	ans^=(cnt[i]%2)*sg[i]
print "Vasya" if ans else "Petya"
