#include <stdio.h>
int n1,n2,k1,k2,dp[101][101][11][11];
bool vis[101][101][11][11];
int calc(int a,int b,int A,int B){
	if(A>k1 || B>k2)
		return 0;
	if(!a && !b)
		return 1;
	if(a<0 || b<0)
		return 0;
	if(!vis[a][b][A][B]){
		vis[a][b][A][B]=true;
		dp[a][b][A][B]=calc(a-1,b,A+1,0)+calc(a,b-1,0,B+1);
		dp[a][b][A][B]%=100000000;
	}
	return dp[a][b][A][B];
}
int main()
{
	scanf("%d%d%d%d",&n1,&n2,&k1,&k2);
	printf("%d",calc(n1,n2,0,0));
	return 0;
}
