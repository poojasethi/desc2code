#include <bits/stdc++.h>
#define M int(1e8)
using namespace std;
int main(){
	int n1,n2,k2,k1,dp[101][101][2]={0};
	cin>>n1>>n2>>k1>>k2;
	dp[n1][n2][0]=dp[n1][n2][1]=1;
	for(int i=n1;i>=0;i--){
		for(int j=n2;j>=0;j--){
			for(int e=1;e<=min(i,k1);e++){
				dp[i-e][j][0]+=dp[i][j][1];
				dp[i-e][j][0]%=M;
			}
			for(int e=1;e<=min(j,k2);e++){
				dp[i][j-e][1]+=dp[i][j][0];
				dp[i][j-e][1]%=M;
			}
		}
	}
	cout<<(dp[0][0][0]+dp[0][0][1])%M<<endl;
}
