#include<iostream>
#include<cstring>
#define Mod 100000000
#define Rep(i,s,e) for(int i = s; i <= e;i++)
using namespace std;
int f[110][110][2] ={0};
int n1,k1,n2,k2;
int main()
{
    cin>>n1>>n2>>k1>>k2;
    f[0][0][0] = 1;
    f[0][0][1] = 1;
    Rep(i,0,n1) Rep(j,0,n2)
    {
        Rep(t,1,min(i,k1))f[i][j][0] = (f[i][j][0]+f[i - t][j][1])%Mod;
        Rep(t,1,min(j,k2))f[i][j][1] = (f[i][j][1]+f[i][j - t][0])%Mod;
    }
    cout<<((f[n1][n2][0] + f[n1][n2][1]))%Mod<<endl;
    return 0;
}
