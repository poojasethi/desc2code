#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=101;
const int oo=100000000;
int f[maxn][maxn][2];
int n1,n2,k1,k2;

int main(){
  int i,j,k;
  scanf("%d%d%d%d",&n1,&n2,&k1,&k2);
  f[n1][n2][0]=f[n1][n2][1]=1;
  for (i=n1;i>=0;i--)
    for (j=n2;j>=0;j--){
      for (k=1;k<=min(i,k1);k++) f[i-k][j][0]=(f[i-k][j][0]+f[i][j][1])%oo;
      for (k=1;k<=min(j,k2);k++) f[i][j-k][1]=(f[i][j-k][1]+f[i][j][0])%oo;
    }
  printf("%d\n",(f[0][0][0]+f[0][0][1])%oo);
  return 0;
}
