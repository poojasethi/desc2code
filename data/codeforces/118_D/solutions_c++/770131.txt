#include <cstdio>

int mem[101][101][11][11];
int c,d;

int f(int n1, int n2, int k1, int k2){
//		printf("%d %d %d %d\n", n1,n2,k1,k2);
	if(n1==0 && n2==0) return 2;
	if(mem[n1][n2][k1][k2])return mem[n1][n2][k1][k2];
	int r=0;
	if(c>k1 && n1)r+=f(n1-1, n2, k1+1, 0)-1;
	if(d>k2 && n2)r+=f(n1, n2-1, 0, k2+1)-1;
	r%=100000000;
	return mem[n1][n2][k1][k2]=r+1;
}

int main(){
	int a,b;
	scanf("%d %d %d %d", &a, &b, &c, &d);
	printf("%d\n", f(a,b,0,0)-1);
}
