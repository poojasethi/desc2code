#include <bits/stdc++.h>
using namespace std;
string a;
int N,N1,N2,M,K1,K2;
int d[101][101][12][12];
int rec(int N,int M,int cons1,int cons2)
{
int sum=0;
if(N==0 && M==0)
return 1;
if(d[N][M][cons1][cons2]>=0)
return d[N][M][cons1][cons2];
if(N && cons1<K1)
sum+=rec(N-1,M,cons1+1,0);
if(M && cons2<K2)
sum+=rec(N,M-1,0,cons2+1);
return d[N][M][cons1][cons2]=sum%100000000;
}
int main()
{
cin>>N1>>N2>>K1>>K2;
memset(d,-1,sizeof(d));
cout<<rec(N1,N2,0,0);
    return 0;
}
