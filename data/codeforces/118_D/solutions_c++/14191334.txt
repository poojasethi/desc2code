#include<iostream>
using namespace std;

const int mod=100000000;
int n1,n2,k1,k2,f[110][110][2];

int main()
	{
	cin>>n1>>n2>>k1>>k2;
	f[n1][n2][0]=f[n1][n2][1]=1;
	for(int i=n1;i>=0;i--)
		for(int j=n2;j>=0;j--)
			{
			for(int k=1;k<=min(i,k1);k++)
				f[i-k][j][0]+=f[i][j][1],f[i-k][j][0]%=mod;
			for(int k=1;k<=min(j,k2);k++)
				f[i][j-k][1]+=f[i][j][0],f[i][j-k][1]%=mod;
			}
	cout<<(f[0][0][1]+f[0][0][0])%mod;
	return 0;
}
