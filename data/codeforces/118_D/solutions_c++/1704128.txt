#include<cstdio>
#include<iostream>
using namespace std;
const int MOD=10000*10000;

int N1,N2,K1,K2;
int f[101][101],g[101][101];

int update(int &x,int y)
{
    x=(x+y)%MOD; return x;
}
int main()
{
    cin>>N1>>N2>>K1>>K2;
    f[0][0]=g[0][0]=1;
    for(int i=0;i<=N1;i++) for(int j=0;j<=N2;j++)
    {
        for(int k=1;k<=K1&&i-k>=0;k++) update(f[i][j],g[i-k][j]);
        for(int k=1;k<=K2&&j-k>=0;k++) update(g[i][j],f[i][j-k]);
    }
    cout<<update(f[N1][N2],g[N1][N2])<<endl;
    return 0;
}
