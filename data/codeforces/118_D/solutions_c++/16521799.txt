#include<bits/stdc++.h>

using namespace std;
long long mem[111][111][11][11];
int c1,c2;
long long mod = 100000000;
long long dp(int n1, int n2 , int k1 , int k2){
	if(n1 < 0 || n2 < 0 || k1 < 0 || k2 < 0) return 0;
	if(!n1 && !n2) return 1;
	long long &ret = mem[n1][n2][k1][k2];
	if(ret != -1) return ret;
	return ret = ((dp(n1-1 , n2 , k1-1 , c2))+(dp(n1,n2-1,c1,k2-1)))%mod;
}
int main(){
	int n1,n2; cin >> n1 >> n2 >> c1 >> c2;
	memset(mem,-1,sizeof(mem));
	cout << dp(n1,n2,c1,c2) << endl;
}
