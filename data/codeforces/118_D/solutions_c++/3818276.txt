#include<iostream>
#include<cstdio>
using namespace std;
#define N 100000000
int f[105][105][2];
int main()
{
    int n1,n2,k1,k2,k;

    cin>>n1>>n2>>k1>>k2;

    f[0][0][0]=1;
    f[0][0][1]=1;

    for(int i=0; i<=n1; i++)
        for(int j=0; j<=n2; j++)
        {
            for(k=1; k<=k1&&i-k>=0; k++) f[i][j][0]=(f[i][j][0]+f[i-k][j][1])%N;
            for(k=1; k<=k2&&j-k>=0; k++) f[i][j][1]=(f[i][j][1]+f[i][j-k][0])%N;
        }
    printf("%d\n",(f[n1][n2][0]+f[n1][n2][1])%N);
    return 0;
}
