#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

int n1,n2,k1,k2;
int dp[101][101][11][11];
int solve(int l1, int l2, int s1, int s2){
	if( l1 < 0 || l2 < 0 || s1 > k1 || s2 > k2 ) return 0;
	if( l1 == 0 && l2 == 0 ) return 1;
	if(dp[l1][l2][s1][s2] != -1) return dp[l1][l2][s1][s2];
	return dp[l1][l2][s1][s2] = (solve(l1-1,l2,s1+1,0)%100000000 + solve(l1,l2-1,0,s2+1)%100000000)%100000000;
}
int main(){
	cin >> n1 >> n2 >> k1 >> k2;
	memset(dp,-1,sizeof dp);
	cout << solve(n1,n2,0,0)%100000000 << endl;
}
