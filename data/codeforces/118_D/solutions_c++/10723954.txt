#include<bits/stdc++.h>
using namespace std;
const int mod=1e8;
int dp[101][101][11][11];
int n1,n2,k1,k2;
int recur(int i,int j,int k,int l)
{
	if(k>k1 || l>k2){return 0;}
	if(!i && !j){return 1;}		
	if(i<0 || j<0){return 0;}
	
	if(dp[i][j][k][l]!=-1){return dp[i][j][k][l];}
	
		else{
			
			int r;
			r=recur(i-1,j,k+1,0)+recur(i,j-1,0,l+1);
			r%=mod;
			dp[i][j][k][l]=r;	
		return dp[i][j][k][l];
		}
		
}
main()
{
	cin>>n1>>n2>>k1>>k2;
	memset(dp,-1,sizeof(dp));
		int ans=recur(n1,n2,0,0);
	cout<<ans<<"\n";
}
