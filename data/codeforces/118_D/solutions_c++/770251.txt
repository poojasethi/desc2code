#include<iostream>
#include<cstring>
using namespace std;
int n1,n2,k1,k2,mod=100000000,dp[101][101][11][11];

int solve(int s1,int s2,int p,int q){
  int res=0;
  if(s1+s2==0)return 1;
  if(dp[s1][s2][p][q]>=0)return dp[s1][s2][p][q];
  dp[s1][s2][p][q]=0;
  if(p<k1&&s1)res+=solve(s1-1,s2,p+1,0);
  if(q<k2&&s2)res+=solve(s1,s2-1,0,q+1);
  return dp[s1][s2][p][q]=res%mod;
}
int main(){
  memset(dp,-1,sizeof(dp));
  cin>>n1>>n2>>k1>>k2;
  cout<<solve(n1,n2,0,0)<<endl;
  return 0;
}
