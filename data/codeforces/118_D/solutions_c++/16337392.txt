#include<bits/stdc++.h>

using namespace std;

#define mod 100000000

int M[102][102][2],n1,n2,k1,k2,i,j,l;

int main(){

	cin >> n1 >> n2 >> k1 >> k2;

    M[0][0][0] = M[0][0][1] = 1;

	for(i=0;i<=n1;i++)
		for(j=0;j<=n2;j++){
			
			for(l=1;l<=k1 && i-l >= 0;l++){
				M[i][j][0] += M[i-l][j][1];
                M[i][j][0] %= mod;
			}

			for(l=1;l<=k2 && j-l >= 0;l++){
				M[i][j][1] += M[i][j-l][0];
				M[i][j][1] %= mod;
			}

		}

	cout << (M[n1][n2][0] + M[n1][n2][1]) % mod;
	return 0;
}
