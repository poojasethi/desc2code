#include<bits/stdc++.h>
using namespace std;
int n,m,a,b,mem[101][101][11][11];
int dp(int c1,int c2,int k1,int k2)
{
    if(c1<0 || c2<0 || k1<0 || k2<0)    return 0;
    if(!c1 && !c2)  return 1;
    int &ret=mem[c1][c2][k1][k2];
    if(ret!=-1) return ret;
    return ret=(1LL*dp(c1-1,c2,k1-1,b)+dp(c1,c2-1,a,k2-1))%100000000;
}
int main()
{
    cin>>n>>m>>a>>b;
    memset(mem,-1,sizeof mem);
    cout<<dp(n,m,a,b)<<endl;
    return 0;
}
