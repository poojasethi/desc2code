#include<bits/stdc++.h>
#define mod 100000000
using namespace std;
int n1,n2,k1,k2,dp[105][105][15][15];
int f(int rn1,int rn2,int ard1,int ard2)
{
	int &res=dp[rn1][rn2][ard1][ard2],a=0,b=0;
	if(!rn1 and !rn2) return 1;
	if(res!=-1) return res;
	if(ard1+1<=k1 and rn1) a=f(rn1-1,rn2,ard1+1,0)%mod; 
	if(ard2+1<=k2 and rn2) b=f(rn1,rn2-1,0,ard2+1)%mod;
	return res=(a+b)%mod;
}
int main()
{
	cin >> n1 >> n2 >> k1 >> k2;
	memset(dp,-1,sizeof dp);
	cout << f(n1,n2,0,0);
	return 0;
}
