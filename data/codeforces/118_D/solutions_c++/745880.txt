#include <iostream>
using namespace std;

#define M 100000000

int main() {
	int n1, n2, k1, k2;
	int f[101][101], g[101][101];

	cin >> n1 >> n2 >> k1 >> k2;
	for (int i = 0; i <= n1; ++i)
		for (int j = 0; j <= n2; ++j)
			if (i == 0 && j == 0)
				f[i][j] = g[i][j] = 1;
			else {
				f[i][j] = g[i][j] = 0;
				for (int k = 1; k <= min(k1, i); ++k)
					f[i][j] = (f[i][j] + g[i - k][j])%M;
				for (int k = 1; k <= min(k2, j); ++k)
					g[i][j] = (g[i][j] + f[i][j - k])%M;
			}
	cout << (f[n1][n2] + g[n1][n2])%M;
}
