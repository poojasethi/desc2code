#include<iostream>
using namespace std;
const int maxn1=101;
int A[maxn1][maxn1][2];
const int d=100000000;
int main()
{
	int n1,n2,k1,k2;
	cin>>n1>>n2>>k1>>k2;
	A[0][0][0]=1,A[0][0][1]=1;
	for(int i=0;i<=n1;i++)
		for(int j=(i==0?1:0);j<=n2;j++)
		{
			A[i][j][0]=0,A[i][j][1]=0;
			for(int h=i-1;h>=0&&h>=i-k1;h--)
				A[i][j][0]=(A[i][j][0]+A[h][j][1])%d;
			for(int h=j-1;h>=0&&h>=j-k2;h--)
				A[i][j][1]=(A[i][j][1]+A[i][h][0])%d;
		}
	cout<<(A[n1][n2][0]+A[n1][n2][1])%d<<endl;	
	return 0;
}
