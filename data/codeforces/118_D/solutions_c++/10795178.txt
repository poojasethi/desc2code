#include<iostream>
#define Mod 100000000
int n1,n2,k1,k2,dp1[101][101],dp2[101][101],i,j,t1,t2;
using namespace std;
int main()
{
	cin >> n1 >> n2 >> k1 >> k2 ;
	for(i=0;i<=100;i++)
		for(j=0;j<=100;j++)
			dp1[i][j]=dp2[i][j]=0;
	dp1[0][0]=dp2[0][0]=1;
	for(i=0;i<=n1;i++)
		for(j=0;j<=n2;j++)
		{
			for(t1=1; t1<=k1 && i+t1<=n1 ;t1++)
				dp1[i+t1][j]=(dp1[i+t1][j]+dp2[i][j])%Mod;	
			for(t2=1;t2<=k2 && j+t2<=n2 ;t2++)
				dp2[i][j+t2]=(dp2[i][j+t2]+dp1[i][j])%Mod;
		}
	cout << (dp1[n1][n2]+dp2[n1][n2])%Mod << endl;
	return 0;
}
