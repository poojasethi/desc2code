#include<bits/stdc++.h>
using namespace std;
int n,m,a,b,dpe[109][109][19][19];
int dp(int c1,int c2,int k1,int k2)
{
    if(c1<0||c2<0||k1<0||k2<0) return 0;
    if(!c1 && !c2) return 1;
    if(dpe[c1][c2][k1][k2]!=-1)return dpe[c1][c2][k1][k2];
    return dpe[c1][c2][k1][k2]=(1LL*dp(c1-1,c2,k1-1,b)+dp(c1,c2-1,a,k2-1))%100000000;
}
int main()
{
    cin>>n>>m>>a>>b;
    memset(dpe,-1,sizeof dpe);
    cout<<dp(n,m,a,b);
    return 0;
}
