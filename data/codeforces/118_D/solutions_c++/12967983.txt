#include<bits/stdc++.h>
using namespace std;
int mod = 100000000;
int dp1[101][101],dp2[101][101];
int main()
{
int n1,n2,k1,k2;
scanf("%d%d%d%d",&n1,&n2,&k1,&k2);
dp1[0][0] = dp2[0][0] = 1;
int a,b,ff,hh;
for(a=0;a<=n1;a++)
{
for(b=0;b<=n2;b++)
{
for(hh=1; hh<=k1 && a+hh<=n1;hh++)
{
dp1[a+hh][b] +=dp2[a][b];
if(dp1[a+hh][b]>=mod)dp1[a+hh][b]-=mod;
}
for(hh=1;hh<=k2 && b+hh<=n2;hh++)
{
dp2[a][b+hh]+=dp1[a][b];
if(dp2[a][b+hh]>=mod)dp2[a][b+hh]-=mod;
}
}
}
printf("%d\n",(dp1[n1][n2]+dp2[n1][n2])%mod);
}
