#include<bits/stdc++.h>
#define y1 _y1_adaisan_buol_lan
using namespace std;
int piy,at,y1,y2,p;
int dp[102][102][15][15];
int solve(int piy,int at,int k1,int k2){
	int &res=dp[piy][at][k1][k2];
	if(!piy && !at) return 1;
	int s1(0),s2(0);
	if(res!=-1) return res;
	if(k1+1<=y1 && piy)
		s1=solve(piy-1,at,k1+1,0);
	if(k2+1<=y2 && at) 
		s2=solve(piy,at-1,0,k2+1);
	return res=(s1+s2)%100000000;
}
