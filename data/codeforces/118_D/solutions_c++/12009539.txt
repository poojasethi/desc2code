#include<bits/stdc++.h>
using namespace std;
int n1,n2,k1,k2,mod=1e8;
int dp[120][120][2],i,j,k;
main()
{
    cin>>n1>>n2>>k1>>k2;
    dp[0][0][0]=dp[0][0][1]=1;
    for ( i=0;i<=n1;i++)
        for ( j=0;j<=n2;j++)
    {
        for ( k=1;k<=k1;k++)
            {(dp[i+k][j][0]+=dp[i][j][1]);
            dp[i+k][j][0]%=mod;}
        for ( k=1;k<=k2;k++)
            {(dp[i][j+k][1]+=dp[i][j][0]);
            dp[i][j+k][1]%=mod;
            }
    }
    cout<<(dp[n1][n2][0]+dp[n1][n2][1])%mod;
}
