#include <cstdio>
using namespace std;

#define ADD_MOD(x,y) x=((x)+(y))%100000000;

int dp[102][102][12][12]={1};

int main(){
    int n,m,p,q,ans=0;
    scanf("%d%d%d%d",&n,&m,&p,&q);
    for(int i=0;i<=n;i++) for(int j=0;j<=m;j++)
    for(int x=0;x<=p;x++) for(int y=0;y<=q;y++){
        ADD_MOD(dp[i+1][j][x+1][0],dp[i][j][x][y]);
        ADD_MOD(dp[i][j+1][0][y+1],dp[i][j][x][y]);
    }
    for(int x=0;x<=p;x++) for(int y=0;y<=q;y++)
        ADD_MOD(ans,dp[n][m][x][y]);
    printf("%d\n",ans);
}
