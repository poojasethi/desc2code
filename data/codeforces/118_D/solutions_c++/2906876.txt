#include<iostream>
#include<algorithm>
using namespace std;
int a[102][102][2];
int main()
{
    int n1,n2,k1,k2;
    cin>>n1>>n2>>k1>>k2;
    a[0][0][0]=a[0][0][1]=1;
    for(int i=0;i<=n1;i++)
    {
        for(int j=0;j<=n2;j++)
        {
            for(int h=1;h<=min(i,k1);h++)
            a[i][j][0] = (a[i][j][0]+a[i-h][j][1])%100000000;
            for(int h=1;h<=min(j,k2);h++)
            a[i][j][1]=(a[i][j][1] + a[i][j-h][0])%100000000;
        }
    }
    cout<<(a[n1][n2][0]+a[n1][n2][1])%100000000<<"\n";
}
