#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;

const int mod=100000000;
int dp[101][101][2],n1,n2,k1,k2;

int main(){
	memset(dp,0,sizeof dp);
	dp[0][0][0]=dp[0][0][1]=1;
	cin >> n1 >> n2 >> k1 >> k2;
	for(int i=0; i<=n1; i++){
		for(int j=0; j<=n2; j++){
			for(int k=1; k<=min(i,k1); k++)dp[i][j][0]=(dp[i][j][0]+dp[i-k][j][1])%mod;
			for(int k=1; k<=min(j,k2); k++)dp[i][j][1]=(dp[i][j][1]+dp[i][j-k][0])%mod;
		}
	}
	cout << (dp[n1][n2][0]+dp[n1][n2][1])%mod << endl;
	return 0;
}
