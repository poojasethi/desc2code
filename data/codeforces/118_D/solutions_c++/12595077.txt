#include <bits/stdc++.h>

using namespace std;
#define f(i, n) for (int i = 0; i < (int)(n); i++)
typedef long long ll;
int n,nn,k,kk,mod=1e8;
int mem[101][101][11][11];

ll dp(int i,int j,int a,int b){
if(i==n && j==nn)return 1;
int &r=mem[i][j][a][b];
if(r!=-1)return r;
int ch=0;
if(a<k && i<n)  ch+=dp(i+1,j,a+1,0);
if(b<kk && j<nn)ch+=dp(i,j+1,0,b+1);

return r=(ch%mod);
}

int main()
{
    memset(mem,-1,sizeof mem);
    cin>>n>>nn>>k>>kk;
    cout<<dp(0,0,0,0);
    return 0;
}
