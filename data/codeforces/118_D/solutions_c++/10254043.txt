#include<stdio.h>
#define x 100000000
int main()
{
	int a[2][201][201]={0},n1,n2,k1,k2,i,j,k;
	scanf("%d%d%d%d",&n1,&n2,&k1,&k2);
	a[0][0][0]=a[1][0][0]=1;
	for(i=0;i<=n1;i++)
		for(j=0;j<=n2;j++){
			for(k=1;k<=k1;k++){
				a[1][i+k][j]+=a[0][i][j];
				a[1][i+k][j]%=x;
			}
			for(k=1;k<=k2;k++){
				a[0][i][j+k]+=a[1][i][j];
				a[0][i][j+k]%=x;
			}	
		}	
	printf("%d\n",(a[0][n1][n2]+a[1][n1][n2])%x);
 return 0;
}
