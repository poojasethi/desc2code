#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define mod 100000000

ll dp[101][101][11][11],n1,n2,k1,k2;

ll f(ll a1,ll a2,ll b1,ll b2)
{
    if(a1>n1 || a2>n2 || b1>k1 || b2>k2) return 0;
    if(a1==n1 && a2==n2) return 1;
    if(dp[a1][a2][b1][b2]!=-1) return dp[a1][a2][b1][b2];
    ll p1=f(a1+1,a2,b1+1,0);
    ll p2=f(a1,a2+1,0,b2+1);
    return dp[a1][a2][b1][b2]=((p1+p2)%mod);
}
int main()
{
    memset(dp,-1,sizeof dp);
    cin>>n1>>n2>>k1>>k2;
    cout<<f(0,0,0,0);
    return 0;
}
