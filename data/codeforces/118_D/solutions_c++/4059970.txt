#include <cstdio>

const int N = 101, MOD = 100 * 1000 * 1000;

int main() {
	int n1, n2, k1, k2, dp[N][N][2] = {1, 1};
	scanf("%d%d%d%d", &n1, &n2, &k1, &k2);
	for (int i1 = 0; i1 <= n1; ++i1)
		for (int i2 = 0; i2 <= n2; ++i2) {
			for (int k = 1; k <= i1 && k <= k1; ++k)
				dp[i1][i2][0] = (dp[i1][i2][0] + dp[i1 - k][i2][1]) % MOD;
			for (int k = 1; k <= i2 && k <= k2; ++k)
				dp[i1][i2][1] = (dp[i1][i2][1] + dp[i1][i2 - k][0]) % MOD;
		}
	printf("%d\n", (dp[n1][n2][0] + dp[n1][n2][1]) % MOD);
	return 0;
}
