#include <iostream>
#include <string.h>
using namespace std;
int n1,n2,k1,k2,dp[11][11][102][102];
int f(int a,int b,int x,int y)
{
  if(a>k1||b>k2||x<0||y<0)
     return 0;
  if(x==0&&y==0)
    return 1;
  if(dp[a][b][x][y]!=-1)
    return dp[a][b][x][y];
  int ret=0;
  ret+=(f(a+1,0,x-1,y))%100000000;
  ret+=(f(0,b+1,x,y-1))%100000000;
  return dp[a][b][x][y]=ret%100000000;
}
int main()
{
    memset(dp,-1,sizeof(dp));
    cin>>n1>>n2>>k1>>k2;
    cout<<f(0,0,n1,n2);
    return 0;
}
