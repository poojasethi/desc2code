#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mx 100000000
ll n1,n2,k1,k2;
ll dp[102][102][12][12];
ll fun(ll m1,ll m2,ll l1,ll l2)
{
    if(m1<0 or m2<0 or l1<0 or l2<0) return 0;
    if(m1==0 && m2==0) return 1;
    if(dp[m1][m2][l1][l2]!=-1) return dp[m1][m2][l1][l2];
    return dp[m1][m2][l1][l2]=(fun(m1-1,m2,l1-1,k2)%mx+fun(m1,m2-1,k1,l2-1)%mx)%mx;
}

int main()
{
    cin>>n1>>n2>>k1>>k2;
    memset(dp,-1,sizeof(dp));
    cout<<fun(n1,n2,k1,k2);
    return 0;
}
