#include<bits/stdc++.h>
using namespace std;
string s;
int dp[200][200][20][20];
int nn1,nn2,kk1,kk2;
long long solve(int n1,int n2,int k1,int k2)
{

if(n1<0||n2<0||k1<0||k2<0)return 0;

if(dp[n1][n2][k1][k2]!=-1)
    return dp[n1][n2][k1][k2];

if(n1==0&&n2==0)
return 1;

int a=solve(n1-1,n2,k1-1,kk2);
int b=solve(n1,n2-1,kk1,k2-1);
return dp[n1][n2][k1][k2]=(a % 100000000)+ (b % 100000000);
}
int main() {
cin>>nn1>>nn2>>kk1>>kk2;
memset(dp,-1,sizeof(dp));
cout<<solve(nn1,nn2,kk1,kk2)% 100000000;
}
