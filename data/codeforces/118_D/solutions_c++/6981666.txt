#include<iostream>
#define mod 100000000
using namespace std;
typedef long long int ll;
ll  dp[220][220][2];
int main()
{
int n1,n2,k1,k2,i,j,k;
cin>>n1>>n2>>k1>>k2;
for(i=0;i<=k1;i++)
dp[i][0][0]=1;
for(j=0;j<=k2;j++)
dp[0][j][1]=1;
for(i=1;i<=n1;i++)
{
for(j=1;j<=n2;j++)
{
for(k=i-1;k>=max(0,i-k1);k--)
dp[i][j][0]=(dp[i][j][0]+dp[k][j][1])%mod;
for(k=j-1;k>=max(0,j-k2);k--)
dp[i][j][1]=(dp[i][j][1]+dp[i][k][0])%mod;
}
}
cout<<(dp[n1][n2][0]+dp[n1][n2][1])%mod<<endl;
return 0;
}
