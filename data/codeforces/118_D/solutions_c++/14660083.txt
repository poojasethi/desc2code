#include<iostream>
#include<cstdlib>
#include<set>
#include<vector>
#include<cstring>
using namespace std;
int n1,n2,k1,k2;
int dp[3][109][109];
#define mod 100000000
int main()
{
  cin>>n1>>n2>>k1>>k2;
  dp[1][n1][n2]=dp[2][n1][n2]=1;
  for(int i=n1;i>=0;i--)
    for(int j=n2;j>=0;j--)
      {
	for(int k=1;k<=min(k1,i);k++)
	  dp[1][i-k][j]+=dp[2][i][j],dp[1][i-k][j]%=mod;
	for(int k=1;k<=min(k2,j);k++)
	  dp[2][i][j-k]+=dp[1][i][j],dp[2][i][j-k]%=mod;
      }
  cout<<(dp[1][0][0]+dp[2][0][0])%mod<<endl;
  return 0;
}
