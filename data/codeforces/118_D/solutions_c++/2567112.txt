#include <cstdio>
#include <algorithm>
#define M ((int)(1e8))
using namespace std;
int n1,n2,k1,k2,dp[105][105][2];
int main(){
    scanf("%d%d%d%d",&n1,&n2,&k1,&k2);
    dp[0][0][0]=dp[0][0][1]=1;
    for(int i=0; i<=n1; ++i)
        for(int j=0; j<=n2; ++j){
            for(int k=1; k<=min(k1,i); ++k) dp[i][j][0]=(dp[i][j][0]+dp[i-k][j][1])%M;
            for(int k=1; k<=min(k2,j); ++k) dp[i][j][1]=(dp[i][j][1]+dp[i][j-k][0])%M;
        }
    printf("%d\n",(dp[n1][n2][0]+dp[n1][n2][1])%M);
    return 0;
}
