#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 101;
const int mod = 100000000;

int f[N][N],g[N][N];

int main(void)
{
	int n1,n2,k1,k2;
	scanf("%d%d%d%d",&n1,&n2,&k1,&k2);
	f[0][0]=g[0][0]=1;
	for(int i=0;i<=n1;i++)
		for(int j=0;j<=n2;j++)
		{
			for(int k=1;k<=k1 && i-k>=0;k++)
				f[i][j]=(f[i][j]+g[i-k][j])%mod;
			for(int k=1;k<=k2 && j-k>=0;k++)
				g[i][j]=(g[i][j]+f[i][j-k])%mod;
		}
	printf("%d\n",(f[n1][n2]+g[n1][n2])%mod);
	return 0;
}
