#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll n[3],k[2],dp[101][101][3],p=100000000;
int main()
{
	ll i,j,ka,l,m,o;
	cin>>n[0]>>n[1]>>k[0]>>k[1];
	dp[0][0][0]=1;dp[0][0][1]=1;
	for(i=0;i<=n[0];i++)
		for(j=0;j<=n[1];j++)
		{
			for(ka=1;ka<=min(i,k[0]);ka++)
			dp[i][j][0]=(dp[i][j][0]+dp[i-ka][j][1])%p;
			for(ka=1;ka<=min(j,k[1]);ka++)
				dp[i][j][1]=(dp[i][j][1]+dp[i][j-ka][0])%p;
		//	cout<<dp[i][j][0]<<" "<<dp[i][j][1]<<endl;
		}
	cout<<(dp[n[0]][n[1]][0]+dp[n[0]][n[1]][1])%p<<endl;
}
