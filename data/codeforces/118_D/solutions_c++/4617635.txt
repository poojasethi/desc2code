#include<cstdio>
int dp[101][101][2];
const int mod=100000000 ;
int main(){
	int N1,N2,K1,K2;
	scanf("%d%d%d%d",&N1,&N2,&K1,&K2);

	dp[N1][N2][0]=1;
	dp[N1][N2][1]=1;
	for(int i=N1;i>=0;i--)
		for(int j=N2;j>=0;j--){
			//take u consecutive foots
			for(int u=1;u+i<=N1 && u<=K1;u++)
				dp[i][j][0]=(dp[i][j][0]+dp[i+u][j][1])%mod;
			//take horse
			for(int v=1;v+j<=N2 && v<=K2;v++)
				dp[i][j][1]=(dp[i][j][1]+dp[i][j+v][0])%mod;
		}
				
	printf("%d\n",(dp[0][0][0]+dp[0][0][1])%mod);
		
}
