u, v, a, b = map(int, raw_input().split())
A, B = {}, {}
d = 100000000

def f(t):
    if t[1] == 0: return 0 if t[0] > a else 1
    if not t in A: A[t] = sum(g((t[1], t[0] - i)) for i in range(1, min(a, t[0]) + 1)) % d
    return A[t]

def g(t):
    if t[1] == 0: return 0 if t[0] > b else 1
    if not t in B: B[t] = sum(f((t[1], t[0] - i)) for i in range(1, min(b, t[0]) + 1)) % d
    return B[t]

print (f((u, v)) + g((v, u))) % d
