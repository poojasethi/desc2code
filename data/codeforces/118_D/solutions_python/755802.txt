import sys
import math
import re
sys.setrecursionlimit(100000000)

INF=100000000
MAX_SIZE=201
n1=0
n2=0
k1=0
k2=0

dp={}
MOD=100000000

def rec(a1,a2,b,c):
    if((a1,a2,b,c) in dp):
        return dp[(a1,a2,b,c)]
    if a1==0:
        if a2>k2:
            return 0
        return 1
    elif a2==0:
        if a1>k1:
            return 0
        return 1
    pt=0
    if b==0:
        if c+1<=k1:
            pt+=rec(a1-1,a2,b,c+1)
        pt+=rec(a1,a2-1,b^1,1)
    else:
        if c+1<=k2:
            pt+=rec(a1,a2-1,b,c+1)
        pt+=rec(a1-1,a2,b^1,1)
    dp[(a1,a2,b,c)]=pt%MOD
    return pt

line=map(int,raw_input().split(' '))
n1,n2,k1,k2=line
res=0
res+=rec(n1-1,n2,0,1)
res+=rec(n1,n2-1,1,1)

print res%MOD
