#!/usr/bin/env python
n,m,k,l = map(int,raw_input().split())
t = {}
def f(a,b,c,d):
	if c > k or d > l or a < 0 or b < 0:
		return 0
	if a == 0 and b == 0:
		return 1
	if (a,b,c,d) in t:
		return t[(a,b,c,d)]
	t[(a,b,c,d)] = (f(a-1,b,c+1,0) + f(a,b-1,0,d+1)) % 100000000
	return t[(a,b,c,d)]
print f(n,m,0,0)
