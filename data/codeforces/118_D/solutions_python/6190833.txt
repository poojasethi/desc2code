n,m,x,y = map(int,raw_input().split())
d = {}

def fuck(n,m,i,j):
    if min([n,m,i,j]) < 0:
        return 0
    if not n+m:
        return 1
    v = (n,m,i,j)
    if v in d:
        return d.get(v)
    d[v] = (fuck(n-1,m,i-1,y)+fuck(n,m-1,x,j-1)) % (10**8)
    return d[v]

print fuck(n,m,x,y)
