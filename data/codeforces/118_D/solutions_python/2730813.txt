n, m, x, y = map(int, raw_input().split())
d = {}

def f(n, m, t, i):
	if min([n, m, i]) < 0:
		return 0
	if n + m == 0:
		return 1
	v = (n, m, t, i)
	if v in d:
		return d.get(v)
	d[v] = (f(n - 1, m, 0, x - 1) + f(n, m - 1, 1, i - 1) if t else f(n - 1, m, 0, i - 1) + f(n, m - 1, 1, y - 1)) % 10 ** 8
	return d[v]

print (f(n - 1, m, 0, x - 1) + f(n, m - 1, 1, y - 1)) % 10 ** 8
