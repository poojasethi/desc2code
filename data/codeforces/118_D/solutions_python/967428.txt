import sys

n1, n2, k1, k2 = map(int, sys.stdin.readline().split())
mod = 10**8
k1o = k1
k2o = k2

dp = {}
def calc(n1, n2, k1, k2):
    if n1 < 0 or n2 < 0: 
        return 0
    if k1 < 0 or k2 < 0:
        return 0
    if n1 == 0 and n2 == 0:
        return 1
    if (n1,n2,k1,k2) in dp:
        return dp[(n1,n2,k1,k2)]
    else:
        dp[(n1,n2,k1,k2)] = calc(n1-1,n2,k1-1,k2o) + calc(n1,n2-1,k1o,k2-1)
        return dp[(n1, n2, k1, k2)] % mod

print calc(n1,n2,k1,k2) % mod
