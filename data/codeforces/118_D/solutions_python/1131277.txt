n1, n2, k1, k2 = map(int, raw_input().split(" "))
mem = {}

def dfs(a, b, c, d):
	res = 0
	if a == 0 and b == 0: return 1
	if (a, b, c, d) in mem: return mem[(a, b, c, d)] 
	if c < k1 and a > 0: res += dfs(a - 1, b, c + 1, 0)
	if d < k2 and b > 0: res += dfs(a, b - 1, 0, d + 1)
	mem[(a, b, c, d)] = res
	return res

print dfs(n1, n2, 0, 0) % (10 ** 8)
