import sys

n1, n2, k1, k2 = map(int, sys.stdin.readline().split())
MOD = 100000000

dp = {}

def rec(a1, b1, ac, bc):
    global k1, k2
    if ac > k1 or bc > k2:
        return 0
    if a1 < 0 or b1 < 0:
        return 0
    if a1 == 0 and b1 == 0:
        return 1

    k = (a1, b1, ac, bc)
    if k in dp:
        return dp[k]

    dp[k] = rec(a1-1, b1, ac+1, 0) + rec(a1, b1-1, 0, bc+1)
    return dp[k]

def solve():
    print rec(n1, n2, 0, 0) % MOD

        
solve()
