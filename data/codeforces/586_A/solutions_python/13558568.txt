def main():
    n = int(raw_input())
    lesson = map(int, raw_input().split())
    answer, pos = 0, -1
    for i in xrange(n):
        if lesson[i] == 1:
            answer += 1
            if pos != -1 and (i - pos - 1) == 1:
                answer += 1
            pos = i
    print answer 
    return

if __name__ == '__main__':
    main()
