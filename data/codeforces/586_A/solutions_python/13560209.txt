n = input()
a = map(int, raw_input().split())
b = [1] * n

#start
for i in xrange(n):
  if a[i] != 0:
    break
  b[i] = 0

#end
for i in xrange(n):
  if a[n - 1 - i] != 0:
    break
  b[n - 1 - i] = 0

#middle
for i in xrange(n - 1):
  if a[i] == 0 and a[i + 1] == 0:
    b[i] = b[i + 1] = 0

print sum(b)
