def sol(l):
    total=0
    for i in range(len(l)):
        if l[i]==1:
            total+=1
        else:
            if i==0: continue
            if i==len(l)-1: break
            if l[i-1]==1 and l[i+1]==1:total+=1
    return total

n=int(raw_input())
l=map(int,raw_input().split())
print sol(l)
