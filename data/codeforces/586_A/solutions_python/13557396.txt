n = int(raw_input())
a = map(int, raw_input().split(" "))
st = 0
while(st<n):
    if(a[st]!=0):
       break
    st+=1
ed = n-1
while(ed>=st):
    if(a[ed]!=0):
        break
    ed-=1
i = st
ans=0
k=0
while(i<=ed):
    if(a[i]!=0):
        if(k==1):
            ans+=1
        k=0
        ans+=1
    else:
        k+=1
    i+=1
print ans
