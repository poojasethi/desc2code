#!/usr/bin/env python

from sys import stdin

n = int(stdin.readline())
a = [int(x) for x in stdin.readline().split()]
r = 0
one = False

for i in range(n):
    if a[i]:
        r += 1
        one = True
    else:
        if one and i < n - 1 and a[i+1]:
            r += 1
        one = False

print(r)
