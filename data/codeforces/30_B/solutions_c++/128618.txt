int main()
{
  scanf("%d.%d.%d",&dd,&mm,&yy);  
  scanf("%d.%d.%d",&A[0],&A[1],&A[2]);  
  sort(A,A+3);
  while(1)
  {
	  if(A[2]%4)
		  M[2]=28;
		else
		  M[2]=29;
    if(A[1]>0 && A[1]<=12 && A[0]>0 && A[0]<=M[A[1]])
			if(A[2]+18<yy)
				f=1;
			else if(A[2]+18==yy)
				if(A[1]<mm)
					f=1;
				else if(A[1]==mm && A[0]<=dd)
					f=1;
    if(!next_permutation(A,A+3))
			break;
  }
  printf("%s\n",f?"YES":"NO");
  return 0;
}
