#include <stdio.h>

const int MAX_DAYS[13] = {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int a[3], b[3];

bool check(int day, int month, int year)
{
	if (!(1 <= month && month <= 12))
		return false;
	int maxDays = MAX_DAYS[month];
    if (month == 2)
		maxDays = year % 4 == 0 ? 29 : 28;
	if (!(1 <= day && day <= maxDays))
		return false;
	if (a[2] - year > 18)
		return true;
	if (a[2] - year < 18)
		return false;
	return a[1] > month || a[1] == month && a[0] >= day;
}

int main()
{
	scanf("%d.%d.%d", a, a + 1, a + 2);
	scanf("%d.%d.%d", b, b + 1, b + 2);
	for (int i = 0; i < 3; ++i)
		for (int j = 0; j < 3; ++j)
			for (int k = 0; k < 3; ++k)
				if (i != j && j != k && i != k && check(b[i], b[j], b[k]))
					   return puts("YES");
	puts("NO");
	return 0;
}
