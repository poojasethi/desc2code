#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int d1, m1, y1;

int dias[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int days(int m, int y){
    if(m == 2) return y % 4 == 0 ? 29 : 28;
    else return dias[m];
}

bool sirve(int d, int m, int y){
    if(m > 12 || d > days(m, y)) return false;
    if(y >= y1) return false;
    int fixed = y1 - y - 1;
    if(m < m1 || m == m1 && d <= d1) fixed++;
    return fixed >= 18;
}

int main(){
    int d2, m2, y2;
    scanf("%d.%d.%d %d.%d.%d", &d1, &m1, &y1, &d2, &m2, &y2);
    
    if(sirve(d2, m2, y2) || sirve(d2, y2, m2) || sirve(m2, d2, y2) || sirve(m2, y2, d2) || sirve(y2, d2, m2) || sirve(y2, m2, d2))
        cout << "YES" << endl;
    else cout << "NO" << endl;
}
