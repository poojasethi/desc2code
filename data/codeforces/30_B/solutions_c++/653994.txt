#include<iostream>
#include<cmath>
#include<cstdio>
#define oo 1000000000
using namespace std;
int d, m, y, bd, bm, by, x1, x2;
int n[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int tinh(int d, int m, int y){
    if (y>99) return oo;
    if (m>12) return oo;
    int nm = n[m];
    if (m == 2 && y%4==0)nm++;
    if (d>nm) return oo;
    return y * 500 + m * 50 + d;
}

int main(){
    scanf("%d.%d.%d\n",&d,&m,&y);
    scanf("%d.%d.%d",&bd,&bm,&by);
    x1 = tinh(d,m,y);
    x2 = tinh(bd,bm,by);
    x2 = min(x2, tinh(bd,by,bm));
    x2 = min(x2, tinh(bm,bd,by));
    x2 = min(x2, tinh(bm,by,bd));
    x2 = min(x2, tinh(by,bm,bd));
    x2 = min(x2, tinh(by,bd,bm));
    x2 += 18*500;
    //cout<<x1<<" "<<x2;
    if (x1 < x2) cout<<"NO";
    else cout<<"YES";

}
