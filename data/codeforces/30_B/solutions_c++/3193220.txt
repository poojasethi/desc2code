int d,m,y;

int cl[] = {0,31,29,31,30,31,30,31,31,30,31,30,31};

bool go(int dd,int mm,int yy){
	if(mm>=13)return false;
	if(dd>cl[mm])return false;
	if(dd==29 && mm==2 && yy%4)return false;
	if(yy+18<y)return true;
	if(yy+18 == y && (mm<m || (mm==m && dd<=d)))return true;
	return false;
}

int main(){
	int a,b,c;
	scanf("%d.%d.%d",&d,&m,&y);
	scanf("%d.%d.%d",&a,&b,&c);
	if(go(a,b,c) || go(a,c,b) || go(b,c,a) || go(b,a,c) || go(c,a,b) || go(c,b,a))puts("YES");
	else puts("NO");
	return 0;
}
