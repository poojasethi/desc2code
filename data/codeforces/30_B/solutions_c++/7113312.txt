#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
int d[4],a[4],m[14]={0,31,29,31,30,31,30,31,31,30,31,30,31};


int check()
{
    if(d[1]>12) return 0;
    if(d[1]==2&&d[2]%4!=0&&d[0]>28) return 0;
    if(m[d[1]]<d[0]||a[2]-d[2]<18) return 0;
    if(a[2]-d[2]>18) return 1;
    if(a[1]>d[1]) return 1;
    if(a[1]==d[1]&&a[0]>=d[0]) return 1;
    return 0;
}

int main()
{
    scanf("%d%*c%d%*c%d",&a[0],&a[1],&a[2]);
    scanf("%d%*c%d%*c%d",&d[0],&d[1],&d[2]);
    sort(d,d+3);
    do
    {
        if(check())
        {
            cout<<"YES"<<endl;
            return 0;
        }
    }while(next_permutation(d,d+3));
    cout<<"NO"<<endl;
    return 0;
}
