#include<bits/stdc++.h>
using namespace std;
int dd,mm,yy,bd,bm,by;
int m[]={0,31,28,31,30,31,30,31,31,30,31,30,31};
int main()
{
	scanf("%d.%d.%d\n%d.%d.%d",&dd,&mm,&yy,&bd,&bm,&by);
	int b[3];
	b[0]=bd,b[1]=bm,b[2]=by;
	sort(b,b+3);
	do
	{
		bd=b[0],bm=b[1],by=b[2];
		if(bd==0||bm==0||18+by>yy) continue;
		if(b[0]>m[b[1]]+((!(b[2]%4))&&(b[1]==2))||b[1]>12) continue;
	if(yy-by>18)
	{
		puts("YES");
		return 0;
	}
	if(yy-by==18)
	{
		if(mm>bm)
		{
			puts("YES");
			return 0;
		}
		if(mm==bm)
		{
			if(dd>=bd)
			{
				puts("YES");
				return 0;
			}
		}
	}
	}
	while(next_permutation(b,b+3));
	puts("NO");
	return 0;
}
