using namespace std;

int a[3],y,m,d; 
int mark;

int day[2][13]={0,31,28,31,30,31,30,31,31,30,31,30,31,
			    0,31,29,31,30,31,30,31,31,30,31,30,31};	

int judge(int yy,int mm,int dd)
{
	if(mm>12) return 0;
	int k;
	if(yy%4==0) k=1;
	else k=0;
	if(dd>day[k][mm]) return 0;
	return 1;	
}

int solve(int i,int j,int k)
{
	int y2,m2,r2;
	y2=a[i];
	m2=a[j];
	r2=a[k];
	if(!judge(y2,m2,r2)) return 0;
	if(y-y2>18) return 1;
	else if(y-y2==18)
	{
		if(m>m2||(m==m2&&d>=r2)) return 1;
		else return 0;
	}
	else return 0;
}


int main()
{
	while(scanf("%d.%d.%d",&d,&m,&y)!=EOF)
	{
		scanf("%d.%d.%d",&a[0],&a[1],&a[2]);
		if(solve(0,1,2)||solve(0,2,1)||solve(1,0,2)||solve(1,2,0)||solve(2,0,1)||solve(2,1,0)) 
		printf("YES\n");
		else printf("NO\n");
	}
}
