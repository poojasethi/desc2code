#include<iostream>
#include<stdio.h>
using namespace std;

int D,M,Y;

bool work(int d,int m,int y)
{
    if(y<1 || y>99) return 0;
    if(m<1 || m>12) return 0;
    if((m==4 || m==6 || m==9 ||
        m==11)&& d>30) return 0;
    if(m==2)
    {
         if(y%4 && d>28) return 0;
         if(!(y%4) && d>29) return 0;
    }
    else if(d>31) return 0;
    if((Y-y)>18) return 1;
    if((Y-y)==18 && M-m >0 ) return 1;
    if((Y-y)==18 && M-m==0 && D-d>=0) return 1;
    return 0;
}

int main()
{
    //freopen("in.in","r",stdin);
    int dd,mm,yy;
    scanf("%d.%d.%d",&D,&M,&Y);
    scanf("%d.%d.%d",&dd,&mm,&yy);
    if(work(dd,mm,yy)||
       work(dd,yy,mm)||
       work(yy,dd,mm)||
       work(yy,mm,dd)||
       work(mm,yy,dd)||
       work(mm,dd,yy)) printf("YES\n");
    else printf("NO\n");
    return 0;
}
