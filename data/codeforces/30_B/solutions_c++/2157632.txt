int mo[12]={31,28,31,30,31,30,31,31,30,31,30,31};
int leap[12]={31,29,31,30,31,30,31,31,30,31,30,31};

int dd,mm,yy;

bool check(int d,int m,int y)
{
	if(m>12) return 0;
	if(y%4==0){if(d>leap[m-1]) return 0;}
	else if(d>mo[m-1]) return 0;
	//cout<<dd<<" "<<mm<<" "<<yy<<" "<<d<<" "<<m<<" "<<y<<endl;
	if(y+18<yy) return 1;
	if(y+18>yy) return 0;
	if(m<mm) return 1;
	if(m>mm) return 0;
	if(d<=dd) return 1;
	return 0;
}

int main()
{
	int d,m,y;
	scanf("%d.%d.%d%d.%d.%d",&dd,&mm,&yy,&d,&m,&y);
	if(check(d,m,y) || check(m,d,y) || check(d,y,m) || check(m,y,d) || check(y,m,d) || check(y,d,m)) puts("YES");
	else puts("NO");
	return 0;
}
