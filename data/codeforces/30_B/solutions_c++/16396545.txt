#include<cstdio>
int maxDayInMonth[13] = 
{
	0,31,29,31,30,31,30,31,31,30,31,30,31
};
int DD,MM,YY;
bool Try(int y,int m,int d)
{
	if (m > 12)
		return false;
	if (d > maxDayInMonth[m])
		return false;
	if (y % 4 != 0 && m == 2 && d == 29)
		return false;
	if (y * 365 + m * 30 + d + 365 * 18 <= YY * 365 + MM * 30 + DD)
		return true;
	return false;
}
int main()
{
	int d,m,y;
	scanf("%d.%d.%d",&DD,&MM,&YY);
	scanf("%d.%d.%d",&d,&m,&y);
	if (Try(d,m,y) || Try(d,y,m) || Try(m,d,y) || Try(m,y,d) || Try(y,m,d) || Try(y,d,m))
		printf("YES\n");
	else
		printf("NO\n");
	return 0;
}
