#include <cstdio>
#include <algorithm>
#include <cmath>

using namespace std;

int day[]={0,31,29,31,30,31,30,31,31,30,31,30,31};
int a[3];
int y,m,d;

bool solve(int D,int M,int Y)
{
	if(D>day[M]) return false;
	if(M>12) return false;
	if(Y%4!=0&&M==2&&D==29) return false;
	Y=Y+18;
	if(y>Y) return true;
	else if(y<Y) return false;
	else if(m>M) return true;
	else if(m<M) return false;
	else if(d>=D) return true;
	else return false;
}
int main()
{
	scanf("%d.%d.%d",&d,&m,&y);
	scanf("%d.%d.%d",a,a+1,a+2);
	bool flag = false;	
	for(int i=0;i<3;i++)
		for(int j=0;j<3;j++){
			if(i!=j){
				if(solve(a[i],a[j],a[3-i-j]))
				flag = true;
			}
		}
	if(flag) printf("YES\n");
	else printf("NO\n");
	return 0;
}
