#include <cstdio>
#include <algorithm>
using namespace std;
const int month_days[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
int d, m, y, b[5];
bool judge() {
	do {
		if (b[0] == 0 || b[1] == 0 || 18+b[2] > y) continue;
		if (b[0] > month_days[b[1]]+((!(b[2]%4))&&(b[1] == 2)) || b[1] > 12) continue;
		if ((m-b[1] > 0 || (m-b[1] == 0 && d-b[0] >= 0)) && (y-b[2] == 18)) return true;
		if (y-b[2] > 18) return true;
	} while (next_permutation(b, b+3));
	return false;
}
int main() {
	scanf("%d.%d.%d", &d, &m, &y);
	scanf("%d.%d.%d", b, b+1, b+2);
	sort(b, b+3);
	if (judge()) printf("YES"); else printf("NO");
	return 0;
}
