
#include <cstdio>
using namespace std;

int dcnt[] = {
	31, 28, 31, 30,
	31, 30, 31, 31,
	30, 31, 30, 31 };

bool ok(int sd, int sm, int sy, int bd, int bm, int by) {
	int vadd=0;
	if (bm>12 ||
		 bd>dcnt[bm-1] + (vadd=((bm==2 && by%4==0)?1:0)) ||
		 by+18>sy) return false;
	
	if (by+18<sy) return true;
	
	// eq
	if (bm<sm) return true;
	if (bm>sm) return false;
	return bd<=sd;
}

main() {
	int sd, sm, sy;
	scanf("%d.%d.%d", &sd, &sm, &sy);
	
	int bd, bm, by;
	scanf("%d.%d.%d", &bd, &bm, &by);
	
	if (
		ok(sd,sm,sy, bd,bm,by) ||
		ok(sd,sm,sy, bd,by,bm) ||
		ok(sd,sm,sy, bm,bd,by) ||
		ok(sd,sm,sy, bm,by,bd) ||
		ok(sd,sm,sy, by,bd,bm) ||
		ok(sd,sm,sy, by,bm,bd)
		) printf("YES");
	else printf("NO");
}
