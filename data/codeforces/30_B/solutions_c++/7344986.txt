#include <cstdio>
#include <algorithm>
using namespace std;
int mon[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int a[10];
int dd, mm, yy;

bool work() {
    sort(a, a + 3);
    do {
        if(yy - a[2] < 18 || !a[1] || !a[0]) continue;
        if(a[1] > 12 || a[0] > mon[a[1] - 1] + (a[2] % 4 == 0 && a[1] == 2) ? 1 : 0) continue;
        if(yy - a[2] == 18 && (mm - a[1] > 0 || (mm - a[1] == 0 && dd - a[0] >= 0))) return true;
        else if(yy - a[2] > 18) return true;
    }while(next_permutation(a, a + 3));
    return false;
}

int main()
{
    scanf("%d.%d.%d", &dd, &mm, &yy);
    scanf("%d.%d.%d", &a[0], &a[1], &a[2]);
    printf("%s\n", work() ? "YES" : "NO");
    return 0;
}
