#include<iostream>
#include<cstdio>

using namespace std;

int D , M , Y;
int C[13] = { 0 , 31 , 28 , 31 , 30 , 31 , 30 , 31 , 31 , 30 , 31 , 30 , 31 };

bool solve( int d , int m , int y ) {
	if ( m > 12 ) return false;
	if ( m == 2 ) {
		if ( y % 4 == 0 && d > 29 ) return false;
		if ( y % 4 != 0 && d > 28 ) return false;
	} else
		if ( d > C[m] ) return false;
	
	int p = -1;
	if ( m < M ) p = 0;
	if ( m == M && d <= D ) p = 0;
	
	return ( Y - y + p >= 18 );
}

int main() {
	int d , m , y;
	scanf( "%d.%d.%d\n" , &D , &M , &Y );
	scanf( "%d.%d.%d\n" , &d , &m , &y );
	
	if ( solve( d , m , y ) || solve( d , y , m ) || solve( m , d , y ) || solve( m , y , d ) ||
		solve( y , d , m ) || solve( y , m , d ) ) printf( "YES\n" );
		else printf( "NO\n" );
	
	return 0;
}
