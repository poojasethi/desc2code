	0,31,28,31,30,31,30,31,31,30,31,30,31
};
int gey(int ye,int yu)
{
	if((ye%400==0||(ye%100&&ye%4==0))&&yu==2){
		return 29;
	 }return h[yu];
}
int main()
{
	int i,j,k;
	scanf("%s%s",b[0],b[1]);
	for(i=0;i<2;i++)
	{
		for(j=0;j<3;j++)
		{
			k=j*3;
			a[i][j]=(b[i][k]-'0')*10+b[i][k+1]-'0';
		}
	}
	for(i=2;i>=0;i--)
	{
		if(a[0][2]-a[1][i]>=18)
		for(j=2;j>=0;j--)
		{
			if(j==i||a[1][j]>12)continue;
			if(a[0][2]-a[1][i]>18||a[0][1]-a[1][j]>=0){
				for(k=2;k>=0;k--)
				{
					if(k==i||k==j||a[1][k]>gey(a[1][i],a[1][j]))continue;
					if(a[0][2]-a[1][i]>18||a[0][1]-a[1][j]>0||a[0][0]-a[1][k]>=0){j=-1;i=-1;k=-1;}
				}
			}
		}
	}if(i==-2)printf("YES\n");
	else printf("NO\n");
}
