#!/usr/bin/python
import sys

taken = [False]*3
perm  = []
def valid(dd,mm,yy):
    if not mm in [1,3,5,7,8,10,12] and dd>30:
        return False
    if dd > 31 or mm > 12:
        return False
    if yy%4==0 and dd>29 and mm==2:
        return False
    if yy%4!=0 and dd>28 and mm==2:
        return False
    return True
   
def convert(dt):
    dd = dt[0:2]
    mm = dt[3:5]
    yy = dt[6:8]
    return map(int,[dd,mm,yy])

def check():
    D1 = convert(cdate)
    L2 = convert(dob)
    D2 = [0]*3
    for i in [0,1,2]:
        D2[i] = L2[perm[i]]
    (dd1,mm1,yy1) = D1
    (dd2,mm2,yy2) = D2
    if not valid(*D2):
        return False
    if yy1 - yy2 > 18:
        return True
    if yy1 - yy2 < 18:
        return False
    if mm1 - mm2 > 0:
        return True
    if mm1 - mm2 < 0:
        return False
    if dd1 - dd2 >= 0:
        return True
    return False


def rec(ind):    
    if ind == 3:
        return check()
    global taken, perm
    for i in xrange(3):
        if not taken[i]:
            taken[i]=True
            perm.append(i)
            if rec(ind+1):
                return True
            perm.pop()
            taken[i]=False
    return False
            
        
cdate = sys.stdin.readline()
dob   = sys.stdin.readline()
if rec(0):
    print 'YES'
else:
    print 'NO'
