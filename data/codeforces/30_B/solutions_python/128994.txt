#!/usr/bin/env python

from sys import stdin

DD, MM, YY = map(int, stdin.readline().split('.'))
BD, BM, BY = map(int, stdin.readline().split('.'))

days_in_month = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]

def is_leap(Y):
   return (Y % 4) == 0

def ok(D, M, Y):
   if Y <= 0 or Y > 99: return False
   if M <= 0 or M > 12: return False
   if D <= 0 or D > 31: return False
   if is_leap(Y) and M == 2:
      if D > days_in_month[M-1]+1:
         return False
   else:
      if D > days_in_month[M-1]:
         return False
   if YY - Y > 18: return True
#  print 'OK'
   if YY - Y < 18: return False
   if MM > M: return True
   if MM < M: return False
   if DD >= D: return True
   return False


if ok(BD, BM, BY) or \
   ok(BD, BY, BM) or \
   ok(BM, BD, BY) or \
   ok(BM, BY, BD) or \
   ok(BY, BM, BD) or \
   ok(BY, BD, BM):
   print 'YES'
else:
   print 'NO'
