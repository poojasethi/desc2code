#!/usr/bin/python
import sys

lim = [[31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31], \
       [31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]]

def leap (y):
	return y % 4 == 0

def do (a, b, c):
	global d, m, y
	if c < 1 or c > 99:
		return False
	if b < 1 or b > 12:
		return False
	if a < 1 or a > lim[leap (c)][b - 1]:
		return False
	if y - c > 18:
		return True
	if y - c < 18:
		return False
	if m - b > 0:
		return True
	if m - b < 0:
		return False
	if d - a > 0:
		return True
	if d - a < 0:
		return False
	return True

d, m, y = [int (x) for x in sys.stdin.readline ().split ('.')]
a, b, c = [int (x) for x in sys.stdin.readline ().split ('.')]
if do (a, b, c) or do (a, c, b) or do (b, a, c) or \
   do (b, c, a) or do (c, a, b) or do (c, b, a):
	print ('YES')
else:
	print ('NO')
