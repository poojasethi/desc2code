from itertools import *
DDMMYY, BDBMBY = [map (int, raw_input ().split ('.')) for _ in xrange (2)]
days = [[31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31], [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]]
for d, m, y in permutations (BDBMBY):
    if 0 < m < 13 and 0 < d <= days[bool (y % 4)][m - 1] and (DDMMYY[2] - y > 18 or DDMMYY[2] - y == 18 and m < DDMMYY[1] or DDMMYY[2] - y == 18 and m == DDMMYY[1] and d <= DDMMYY[0]):
        print 'YES'
        break
else:
    print 'NO'
