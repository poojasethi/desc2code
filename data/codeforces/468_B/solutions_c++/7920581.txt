#include<bits/stdc++.h>
using namespace std;

#define N 100010
map<int,int> hs;

int n,a,b,f[N],vs[N],sat[N],p[N];

int find(int x){ return f[x]==x? x : f[x] = find(f[x]); }

int main(){
    cin>>n>>a>>b;
    for(int i=1;i<=n;i++){
        cin>>p[i];
        hs[p[i]] = i;f[i]=i;
    }
    for(int i=1;i<=n;i++){
        if(hs.count(a-p[i])) f[find(hs[a-p[i]])] = find(i),vs[i]|=1;
        if(hs.count(b-p[i])) f[find(hs[b-p[i]])] = find(i),vs[i]|=2;
        sat[i] = 3;
    }
    for(int i=1;i<=n;i++){
        sat[find(i)] &= vs[i];
    }
    for(int i=1;i<=n;i++){
        if(sat[i] == 0){
            cout<<"NO"<<endl;
            return 0;
        }
    }
    cout<<"YES"<<endl;
    for(int i=1;i<=n;i++){
        int st = sat[find(i)];
        cout<<(0+((st&1)==0))<<" "; 
    }
    return 0;
}
