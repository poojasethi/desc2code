#include <bits/stdc++.h>
#define MAX_N 100100 

using namespace std;

int n;
int a[MAX_N];
int x , y , len;
int save[MAX_N]; 
map <int , int> mymap; 

int main ()
{
	scanf("%d %d %d", &n , &x , &y); 
	
	for (int i = 1; i <= n; i++) 
	{
		scanf("%d", &a[i]); 
		mymap[a[i]] = 1; 
	}
	
	for (int i = 1; i <= n; i++) 
	{
		if (!mymap[x - a[i]]) save[len++] = a[i]; 
	}
	
	for (int t = 0; t < len; t++) 
	{
		mymap[save[t]] = 2;
		
		if (!mymap[y - save[t]]) 
		{
			printf("NO"); 
			return 0; 
		}
		
		if (mymap[y - save[t]] == 2) continue; 
		
		mymap[y - save[t]] = 2; 
		
		if (mymap[x - y + save[t]] == 1) 
		{
			mymap[x - y + save[t]] = 2; 
			save[len++] = x - y + save[t]; 
		}
	}
	
	cout << "YES" << endl; 
	
	for (int i = 1; i <= n; i++) cout << mymap[a[i]] - 1 << ' '; 
	return 0;
}
