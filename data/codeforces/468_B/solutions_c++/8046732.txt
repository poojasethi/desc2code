#include <iostream>
#include <cstdio>
#include <queue>
#include <map>
using namespace std;

const int M = 1e5 + 10;
map<int,int>ms;
queue<int>ls;
int ma[M];

int main()
{
    int n,a,b;
    scanf("%d %d %d",&n,&a,&b);
    for(int i = 0; i < n; i++)
    {
        scanf("%d", ma + i);
        ms[ma[i]] = 1;
    }
    for(int i = 0; i < n; i++)
        if(!ms[a - ma[i]]) ls.push(ma[i]);

    while(!ls.empty())
    {
        int u = ls.front(),tp;
        ls.pop();
        if(ms[u] > 0 && ms[a - u] == 0 && ms[b - u] == 0)
        {
            puts("NO");
            return 0;
        }
        --ms[u]; 
        --ms[b - u];
        tp = a - b + u;
        if(ms[tp]) ls.push(tp);  
    }
    puts("YES");
    for(int i = 0; i < n; i++)
        if(ms[ma[i]] == 1) printf("0 ");
        else printf("1 ");
    puts("");
    return 0;
}
