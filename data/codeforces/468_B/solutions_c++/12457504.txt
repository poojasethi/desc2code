#include <bits/stdc++.h>
using namespace std;
map<int, int> mp;
int a[100100], b[100100];
int main()
{
    int n, A, B;
    cin >> n >> A >> B;
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
        mp[a[i]] = 1;
    }
    int t = 0, l = 0;
    for(int i = 1; i <= n; i++) {
        if(!mp[A-a[i]])
            mp[a[i]] = 2, b[++t] = a[i];
    }
    while(l < t) {
        l++;
        if(!mp[B-b[l]]) {
            printf("NO\n");
            return 0;
        }
        if(mp[B-b[l]] == 2) continue;
        mp[B-b[l]] = 2;
        mp[A-B+b[l]] = 2;
        b[++t] = A-B+b[l];
    }
    printf("YES\n");
    for(int i = 1; i <= n; i++)
        printf("%d ", mp[a[i]] - 1);
    return 0;
}
