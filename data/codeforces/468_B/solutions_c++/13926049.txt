#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <queue>
#include <map>
using namespace std;
map<int,int> pos;
int n,a,b,ans[500000],x[500000];
int main()
{
	queue<int> q;
	scanf("%d%d%d",&n,&a,&b);
	for (int i=1;i<=n;++i) 
	{
		scanf("%d",&x[i]);
		pos[x[i]]=i;
	}
	for (int i=1;i<=n;++i)
	{
		if (pos[a-x[i]]==0)
		{
			ans[i]=1;
			q.push(x[i]);
		}
	}
	while (!q.empty())
	{
		int now=q.front();
		q.pop();
		if (pos[b-now]==0) 
		{
			printf("NO\n");
			return 0;
		}
		if (ans[pos[b-now]]==1) continue;
		ans[pos[b-now]]=1;
		if (pos[a-b+now]!=0)
		{
			ans[pos[a-b+now]]=1;
			q.push(a-b+now);
		}
	}
	printf("YES\n");
	for (int i=1;i<=n;++i) printf("%d ",ans[i]);
	return 0;
}
