#include<cstdio>
#include<algorithm>
#include<map>

#define  rep(i,a,n) for (int i=a;i<=n;i++)
using namespace std;
const int nn =100005;
int fa[nn];
int n,x,y;
map<int ,int > ma;
int a[nn];
int find(int x){
	return fa[x]==x?x:fa[x]=find(fa[x]);
}
int main()
{
	//freopen("1.in","r",stdin);
	scanf("%d%d%d",&n,&x,&y);
	rep(i,1,n)
	scanf("%d",a+i),ma[a[i]]=i,fa[i]=i;
	fa[n+1]=n+1; fa[n+2]=n+2;
	rep(i,1,n)
	{				
		if (ma[x-a[i]]) fa[find(ma[x-a[i]])]=find(i); 
		else fa[find(i)]=find(n+1);
		if (ma[y-a[i]]) fa[find(ma[y-a[i]])]=find(i);
		else fa[find(i)]=find(n+2);
	}
	if (find(n+1)==find(n+2)){
		printf("NO"); return 0;
	}
	printf("YES\n");

}
