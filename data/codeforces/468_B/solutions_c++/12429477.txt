#include <bits/stdc++.h>
using namespace std;

int A[100005];
vector<int> v;
map<int, int> res;

int main(){
    int n, a, b, x; cin >> n >> a >> b; 
    for(int i=0; i<n; i++) {cin >> A[i]; res[A[i]] = 1;}        
    //map<int, int>::iterator it;
    for(int i=0; i<n; i++) if(!res[a-A[i]])
        v.push_back(A[i]);
    for(int i=0; i<v.size(); i++){
        res[v[i]] = 2;
        if(!res[b-v[i]]){cout << "NO" << endl; return 0;}
        if(res[b-v[i]]==2) continue;
        res[b-v[i]]=2;
        if(res[a-(b-v[i])]==1){
            res[a-(b-v[i])]=2;
            v.push_back(a-(b-v[i]));
        }       
    }
    cout << "YES" << endl;
    for(int i=0; i<n; i++) 
        cout << res[A[i]]-1 << " ";
    cout << endl;   
    return 0;
}
