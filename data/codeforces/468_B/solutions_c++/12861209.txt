#include<cstdio>
#include<map>
#define N 100005
using namespace std;map<int,int>pos;
int f[N],a[N],status[N],sta[N],n,A,B,i;
int get(int u){return f[u]==u?u:f[u]=get(f[u]);}
void Union(int x,int y){f[get(pos[x])]=get(pos[y]);}
int main()
{
    scanf("%d%d%d",&n,&A,&B);
    for (i=1;i<=n;i++) 
        f[i]=i,scanf("%d",&a[i]),pos[a[i]]=i;
    for (i=1;i<=n;i++){
        if (pos.find(A-a[i])!=pos.end()) Union(a[i],A-a[i]),status[i]|=1;
        if (pos.find(B-a[i])!=pos.end()) Union(a[i],B-a[i]),status[i]|=2;
  }for (i=1;i<=n;i++) sta[i]=3;
  for (i=1;i<=n;i++) sta[get(i)]&=status[i];
    for (i=1;i<=n;i++)
        if (!sta[get(i)]) return puts("NO"),0;
    puts("YES");
    for (i=1;i<=n;i++)
        printf("%d ",sta[get(i)]&1^1);
}
