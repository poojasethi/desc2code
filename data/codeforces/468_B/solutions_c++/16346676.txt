#include<iostream>
#include<vector>
#include<map>
#define ll long long int
using namespace std;
map<ll,ll> set;
vector<ll> v;
int main()
{
	ll n, a, b;
	cin>>n>>a>>b;
	ll arr[n];
	for(int i=0 ; i<n ; i++)
	{
		cin>>arr[i];
		set[arr[i]] = 2;
	}
	for(int i=0 ; i<n ; i++)
	{
		if(set[ b-arr[i] ] == 0)
		{
			v.push_back(arr[i]);
		}
	}
	for(int i=0 ; i<v.size() ; i++)
	{
		set[v[i]] = 1;
		if(set[a-v[i]] == 0)
		{
			cout<<"NO";
			return 0;
		}
		set[a-v[i]] = 1;
		if(set[b-(a-v[i])] == 2)
		{
			set[b-(a-v[i])] = 1;
			v.push_back(b-(a-v[i]));
		}
	}
	cout<<"YES\n";
	for(int i=0 ; i<n ; i++)
	{
		if(set[arr[i]] == 1)
			cout<<"0 ";
		else
			cout<<"1 ";
	}
}
