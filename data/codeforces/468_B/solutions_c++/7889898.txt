#include <cstdio>
#include <map>
#define pb push_back
#define opp(i) (((i)+1^1)-1)
#define foreach(it,v) for(__typeof((v).begin()) it=(v).begin();it!=(v).end();it++)
using namespace std;
const int maxn=100001;

int set[maxn+2];
int fin(int d)
{
	if (d==set[d]) return d;
	return set[d]=fin(set[d]);
}

int n,a[3];
int p[maxn];
map<int,int> f;
int main()
{
	scanf("%d%d%d",&n,&a[1],&a[2]);
	for (int i=1;i<=n;i++) scanf("%d",&p[i]),f[p[i]]=i;
	for (int i=1;i<=n+2;i++) set[i]=i;
	for (int i=1,id;i<=n;i++)
		for (int z=1;z<=2;z++)
			if (id=f[a[z]-p[i]]) set[fin(i)]=set[fin(id)];
			else set[fin(i)]=set[fin(n+3-z)];
	if (fin(n+1)==fin(n+2)) puts("NO");
	else
	{
		puts("YES");
		for (int i=1;i<=n;i++)
			printf("%d%c",fin(i)==fin(n+2),i==n?'\n':' ');
	}
	return 0;
}
