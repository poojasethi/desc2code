  #include<iostream>
   # include<map>
    using namespace std;
      map <long long, long long> d;  long n,a,b,p[10000001],q[1000001];
    int main(){
  //       freopen("input.txt","r",stdin);
  //     freopen("output.txt","w",stdout);
     cin>>n>>a>>b; 
      for (int k=1;k<=n;++k) {
           long x;cin>>x; p[k]=x;
           d[x]=1;
      }  long t,h; t=h=0;
      for (int k=1;k<=n;++k){
           if (d[p[k]]==1 && !d[a-p[k]]) {
           q[++t]=p[k];d[p[k]]=2;}
      } long w=1; 
      while (w<=t){
          if (!d[b-q[w]]) {cout<<"NO";return 0;}
          d[b-q[w]]=2;
           if (d[a-b+q[w]]==0) {++w;continue;}
          d[a-b+q[w]]=2; q[++t]=a-b+q[w];
          ++w;
      }
            cout<<"YES"<<endl;
      for (int k=1;k<=n;++k) cout<<(d[p[k]]-1)%2<<' ';
      return 0;
    }
