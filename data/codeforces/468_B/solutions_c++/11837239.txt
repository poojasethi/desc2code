#include <bits/stdc++.h>

using namespace std;

map<int, int> m;

int main(){
    int n, a, b;
    cin>>n>>a>>b;

    int p[n];
    for(int i=0 ; i<n ; i++)
        cin>>p[i], m[p[i]] = 1;

    vector<int> vec;
    for(int i=0 ; i<n ; i++){
        if(!m[a-p[i]])
            vec.push_back(p[i]);
    }

    for(int i=0 ; i<vec.size() ; i++){
        m[vec[i]] = 2;
        if(!m[b - vec[i]]){
            cout<<"NO\n";
            return 0;
        }

        if(m[b - vec[i]] == 2)
            continue;
        m[b - vec[i]] = 2;
        if(m[a - (b - vec[i])] == 1){
            m[a - (b - vec[i])] = 2;
            vec.push_back(a - (b - vec[i]));
        }
    }

    cout<<"YES\n";

    for(int i=0 ; i<n ; i++){
        cout<<m[p[i]]-1<<" ";
    }

    cout<<"\n";

    return 0;
}
