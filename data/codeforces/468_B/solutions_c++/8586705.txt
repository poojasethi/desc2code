#include <cstdio>
#include <map>
using namespace std;

const int N=(int)1e5+5;
int n,a,b,p[N],len,q[N*2];
map<int,int> f;
int main(){
	scanf("%d%d%d",&n,&a,&b);
	for(int i=1;i<=n;i++){
		scanf("%d",&p[i]);
		f[p[i]]=1;
	}
	for(int i=1;i<=n;i++) if(!f[a-p[i]])
		q[len++]=p[i];
	for(int t=0;t<len;t++){
		f[q[t]]=2;
		if(!f[b-q[t]]) return puts("NO"),0;
		if(f[b-q[t]]==2) continue;
		f[b-q[t]]=2;
		if(f[a-(b-q[t])]==1){
			f[a-(b-q[t])]=2;
			q[len++]=a-(b-q[t]);
		}
	}
	puts("YES");
	for(int i=1;i<=n;i++) printf("%d ",f[p[i]]-1); puts("");
}
