#include<cstdio>
#include<map>
#define nn 100010
using namespace std;
map<int,int>pos;
int n,i,A,B,t,w,q[nn*2],a[nn];
int main()
{
 scanf("%d%d%d",&n,&A,&B);
 t=w=0;
 for (i=1;i<=n;i++)scanf("%d",&a[i]),pos[a[i]]=1;
 for (i=1;i<=n;i++)
  if (!pos[A-a[i]])pos[a[i]]=2,q[++w]=a[i];
  while (t<w){t++;
  if (!pos[B-q[t]]){puts("NO");return 0;}
  if (pos[B-q[t]]==2)continue;
  pos[B-q[t]]=2;
  pos[A-(B-q[t])]=2;q[++w]=A-(B-q[t]);
             }
  puts("YES");
  for (i=1;i<=n;i++)printf("%d ",pos[a[i]]-1);
  return 0;
}
