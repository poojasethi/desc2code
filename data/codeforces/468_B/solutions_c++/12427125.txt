#include <bits/stdc++.h>

using namespace std;
const int maxn=100005;
map<int,int>id;
int s[maxn];
int col[maxn];
queue<int>q;
int main()
{
//    freopen("in","r",stdin);
    int n,a,b;
    cin>>n>>a>>b;
    for(int i=1;i<=n;i++){
        cin>>s[i];
        id[s[i]]=i;
    }
    for(int i=1;i<=n;i++)if(!id[a-s[i]])q.push(i);
    while(!q.empty()){
        int u=q.front();q.pop();
        if(!id[b-s[u]]){
            puts("NO");
            return 0;
        }
        col[u]=col[id[b-s[u]]]=1;
        if(id[a-(b-s[u])])q.push(id[a-(b-s[u])]);
    }
    puts("YES");
    for(int i=1;i<=n;i++)printf("%c%c",col[i]?'1':'0',i==n?'\n':' ');
    return 0;
}
