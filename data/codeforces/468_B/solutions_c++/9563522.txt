#include <bits/stdc++.h>
using namespace std;
#define rep(i,n)      for(int i(0),_n(n);i<_n;++i)
int main()
{
ios_base::sync_with_stdio(false);
   long n,a,b,v[100005],val;
   map<long,int> M;
   cin>>n>>a>>b;
   rep(i,n)
   {
       cin>>v[i];
       M[v[i]]=1;
   }
   queue<long> q;
   rep(i,n)
   {
       if(M[a-v[i]]==1) continue;
       if(M[b-v[i]]==0) {cout<<"NO";return 0;}
       q.push(v[i]);
   }
   while(!q.empty())
   {
       val=q.front();q.pop();
        M[val]=2;
       if(M[b-val]==0) {cout<<"NO";return 0;}
       M[b-val]=2;
       if(M[a-(b-val)]==1)
       {
           q.push(a-(b-val));
       }
   }
   cout<<"YES\n";
   rep(i,n)
       cout<<((M[v[i]]%2)^1)<<" ";


return 0;
}
