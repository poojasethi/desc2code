#include <bits/stdc++.h>
using namespace std;

#define N 100010

map<int,int> ha;
int n, A, B, a[N], fa[N], ok[N], g[N];

int find(int x){
	return fa[x] == x ? x : fa[x] = find(fa[x]);
}

void un(int x, int y){
	fa[find(x)] = find(y);
}
int main(){
	scanf("%d%d%d", &n, &A, &B);
	for(int i = 1;i <= n;i ++){
		scanf("%d", &a[i]);
		fa[i] = i;
		g[i] = 3;
		ha[a[i]] = i;
	}
	for(int i = 1;i <= n;i ++){
		if(ha[A - a[i]]) un(ha[A - a[i]], i), ok[i] |= 1;
		if(ha[B - a[i]]) un(ha[B - a[i]], i), ok[i] |= 2;
	}
	for(int i = 1;i <= n;i ++) g[find(i)] &= ok[i];
	for(int i = 1;i <= n;i ++) if(!g[i]){
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	for(int i = 1;i <= n;i ++) printf("%d ", (g[find(i)] & 1) ^ 1);
	return 0;
}
