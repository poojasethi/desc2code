#include<cstdio>
#include<algorithm>
#include<iostream>
#include<map>
using namespace std;
int h[200000],f[200000],i,n,a,b;
map<int,int>c;
int find(int x)
{
	return f[x]==x?x:f[x]=find(f[x]);
}
int main()
{
	cin>>n>>a>>b;
	for (i=1;i<=n;i++)
	{
		cin>>h[i];
		c[h[i]]=i;
	}
	for (i=1;i<=n+2;i++) 
	f[i]=i;
	for (i=1;i<=n;i++)
	{
		if (c[a-h[i]])
			f[find(c[a-h[i]])]=f[find(c[h[i]])];
		else
			f[find(c[h[i]])]=f[find(n+1)];
		if (c[b-h[i]])
			f[find(c[b-h[i]])]=f[find(c[h[i]])];
		else
			f[find(c[h[i]])]=f[find(n+2)];
	}
	for (i=1;i<=n+2;i++) 
	h[i]=find(i);
	if (h[n+1]==h[n+2]) 
	{
		cout<<"NO"<<endl; 
		return 0;
	}
	cout<<"YES"<<endl;
	for (i=1;i<=n;i++) 
	if (h[i]==h[n+1]) 
	cout<<"1 "; 
	else 
	cout<<"0 ";
	return 0;
}
