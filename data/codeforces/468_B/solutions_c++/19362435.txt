#include<cstdio>
#include<map>
#include<queue>
using namespace std;
map<int,short> mp;
int t[100000];
int main(){
	int n,a,b;
	queue<int> que;
	scanf("%d%d%d",&n,&a,&b);
	for(int i=0;i<n;i++){
		scanf("%d",t+i);
		mp[t[i]]=1;
	}
	for(int i=0;i<n;i++)
		if(mp[a-t[i]]>0) mp[t[i]]=2; else que.push(t[i]);
	while(!que.empty()){
		int x=que.front();
		que.pop();
		if(!mp[b-x]) return printf("NO\n"),0;
		mp[x]=mp[b-x]=3;
		if(mp[a-(b-x)]){
			mp[a-(b-x)]=3;
			que.push(a-(b-x));
		}
	}
	printf("YES\n");
	for(int i=0;i<n;i++) printf("%d%c",mp[t[i]]&1,(i<n-1?' ':'\n'));
	return 0;
}
