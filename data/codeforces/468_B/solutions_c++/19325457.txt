#include <bits/stdc++.h>
using namespace std;

int n, a, b, p[100005];
map<int, int> hs, s;
queue<int> undo;

int main() {
	scanf("%d %d %d", &n, &a, &b);
	for (int i = 0; i < n; ++i) {
		scanf("%d", p + i);
		hs[p[i]] = 1;
	}
	for (int i = 0; i < n; ++i) if (hs[a - p[i]]) s[p[i]] = 2; // 2:a(0) 1:b(1)
	for (int i = 0; i < n; ++i) if (!s[p[i]]) undo.push(p[i]);
	while (undo.size()) {
		int x = undo.front(); undo.pop();
		if (s[x] == 1) continue;
		if (!hs[b - x]) { puts("NO"); return 0; }
		if (s[b - x] == 2) undo.push(a - b + x);
		s[x] = s[b - x] = 1;
	}
	puts("YES");
	for (int i = 0; i < n; ++i)
		printf("%d%c", (s[p[i]] & 1), i < n - 1 ? ' ' : '\n');
	return 0;
}
