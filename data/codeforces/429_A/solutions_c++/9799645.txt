#include <bits/stdc++.h>
#define maxn 110000

using namespace std;

int N,init[maxn],goal[maxn];
vector<int> print,v[maxn];

void dfs(int node,int now,int aft,int dad){
	if(init[node]^now!=goal[node]){
		now^=1;
		print.push_back(node);
	}
	for(int i=0;i<v[node].size();i++)
		if(v[node][i]!=dad)
			dfs(v[node][i],aft,now,node);
}

int main(){
	int a,b;
	scanf(" %d",&N);
	for(int i=1;i<N;i++){
		scanf(" %d %d",&a,&b);
		v[a].push_back(b);
		v[b].push_back(a);
	}
	for(int i=1;i<=N;i++)
		scanf(" %d",&init[i]);
	for(int i=1;i<=N;i++)
		scanf(" %d",&goal[i]);
	dfs(1,0,0,0);
	printf("%d\n",(int)print.size());
	for(int i=0;i<print.size();i++)
		printf("%d\n",print[i]);
	return 0;
}
