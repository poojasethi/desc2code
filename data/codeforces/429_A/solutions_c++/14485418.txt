#include<iostream>
#include<vector>
using namespace std;
const int M=100006;
vector<int> ans;
vector<int> t[M];
int n;
int init[M],aim[M];
void dfs(int x,int last,int f,int s)
{
    if(f) init[x]^=1;
    if(init[x]!=aim[x])
    {
        f^=1;
        ans.push_back(x);
    }
    for(int i=0;i<t[x].size();i++)
        if(t[x][i]!=last) dfs(t[x][i],x,s,f);
}
int main()
{
    cin>>n;
    int x,y;
    for(int i=1;i<n;i++)
    {
        cin>>x>>y;
        t[x].push_back(y);
        t[y].push_back(x);
    }
    for(int i=1;i<=n;i++) cin>>init[i];
    for(int i=1;i<=n;i++) cin>>aim[i];
    dfs(1,0,0,0);
    cout<<ans.size()<<endl;
    for(int i=0;i<ans.size();i++)
        cout<<ans[i]<<endl;
    return 0;
}
