#include<cstdio>
#define N 100005
using namespace std;
struct arr{int go,next;}a[N*2];
int ans[N],A[N],B[N],cnt,i,n,end[N],x,y;
inline void add(int u,int v){a[++cnt].go=v;a[cnt].next=end[u];end[u]=cnt;}
void dfs(int k,int fa,int f0,int f1,int opt)
{
  if ((opt?f1:f0)&1) A[k]^=1;
  if (A[k]!=B[k]) (opt?f1:f0)++,ans[++cnt]=k;
  for (int i=end[k];i;i=a[i].next)
    if (a[i].go!=fa) dfs(a[i].go,k,f0,f1,opt^1);
}
int main()
{
  scanf("%d",&n);
  for (i=1;i<n;i++)
    scanf("%d%d",&x,&y),add(x,y),add(y,x);
  for (i=1;i<=n;i++) scanf("%d",&A[i]);
  for (i=1;i<=n;i++) scanf("%d",&B[i]);
  cnt=0;dfs(1,-1,0,0,0);printf("%d\n",cnt);
  for (i=cnt;i;i--) printf("%d\n",ans[i]);
  return 0;
}
