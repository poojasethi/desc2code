#include<iostream>
#include<vector>
using namespace std;
const int N=100005;
vector<int> e[N],ans;
int n,g[N];
void dfs(int x,int p,int q,int fa=0){//x翻p，fa翻q
    if(g[x]^p) //x要翻
        p^=1,ans.push_back(x);
    for(int i=0;i<e[x].size();i++){
        int y=e[x][i];
        if(y!=fa)
            dfs(y,q,p,x);//记录了2层的翻的数据
    }
}

int main(){
    cin>>n;
    for(int i=1,a,b;i<n;i++){
        cin>>a>>b;
        e[a].push_back(b);
        e[b].push_back(a);
    }
    for(int i=1;i<=n;i++)
        cin>>g[i];
    for(int i=1,x;i<=n;i++)
        cin>>x,g[i]^=x;
    dfs(1,0,0);
    cout<<ans.size()<<endl;
    for(int i=0;i<ans.size();i++)
        cout<<ans[i]<<endl;
}
