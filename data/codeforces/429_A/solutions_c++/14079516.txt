#include<bits/stdc++.h>
#define MAX 100005
using namespace std;
vector< vector<int> > G(MAX);
vector<int> A(MAX), B(MAX), ans;

void dfs(int x,int v0,int v1,int ff){
	if(A[x]^v0 != B[x]){
		v0^=1;
		ans.push_back(x);
	}
	
	for(int i = 0; i < G[x].size(); i++){
		if(G[x][i] != ff) dfs(G[x][i], v1, v0, x);
	}
}
int main(){
	int n, x, y;
	cin>>n;
	for(int i = 1;i < n; i++){
		cin>>x>>y;
		G[x].push_back(y);
		G[y].push_back(x);
	}
	for(int i = 1;i <= n; i++) cin>>A[i];
	for(int i = 1;i <= n; i++) cin>>B[i];
	
	dfs(1,0,0,0);
	
	cout<<ans.size()<<endl;
	for(int i = 0; i < ans.size(); i++) cout<<ans[i]<<endl;
}
