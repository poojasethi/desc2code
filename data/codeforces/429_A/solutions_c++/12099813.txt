#include <stdio.h>
#include <vector>
using namespace std;
vector<int> mp[100001],ans;
bool init[100001];
void dfs(int k,char flip,char q,int par)
{
	int i;
	if (flip^init[k])
	{
		ans.push_back(k);
		flip^=1;
	}
	for (i=0;i<mp[k].size();i++)
		if (mp[k][i]!=par)
			dfs(mp[k][i],q,flip,k);
}
int main()
{
	int n,i,t1,t2,x;
	scanf("%d",&n);
	for (i=1;i<n;i++)
	{
		scanf("%d%d",&t1,&t2);
		mp[t1].push_back(t2);
		mp[t2].push_back(t1);
	}
	for (i=1;i<=n;i++)
		scanf("%d",init+i);
	for (i=1;i<=n;i++)
	{
		scanf("%d",&x);
		init[i]^=x;
	}
	dfs(1,0,0,0);
	printf("%d\n",ans.size());
	for (i=0;i<ans.size();i++)
		printf("%d\n",ans[i]);
	return 0;
}
