#include <iostream>
#include <vector>

using namespace std;
#define MAXN 100000
typedef pair<int,bool> pii;
bool B[MAXN],B1[MAXN],V[MAXN];

vector<vector< int > >A;
vector <int>R;

void dfs(int u,bool a,bool b, int p=-1){
	int v;
	bool c=B1[u];
	if (a==false){
		c=!c;
	}
	if (c!=B[u]){
		R.push_back(u+1);
    a=!a;
	}
	for (int i=0;i<A[u].size();++i){
		v=A[u][i];
		if (v!=p)
			dfs(v,b,a,u);
	}
}

int main(){
	int n;
	cin >> n;
	int v1,v2;
	A.resize(n);
	for (int i=0;i<n-1;i++){
		cin >> v1 >> v2;
		A[v2-1].push_back(v1-1);
		A[v1-1].push_back(v2-1);
	}
  for (int i=0;i<n;i++){
		cin >> B[i];
	}
	for (int i=0;i<n;i++){
		cin >> B1[i];
	}
	dfs(0,1,1);
	cout << R.size() << endl;
	for (int i=0;i<R.size();i++){
		cout << R[i] << endl;
	}
	return 0;
}
