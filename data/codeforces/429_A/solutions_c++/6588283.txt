#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<vector>
using namespace std;
int n,m;
#define N 200000
vector<int>G[N],ans;
int a[N];
void dfs(int u,int f,int c1,int c2){
	int nc1=c2,nc2=c1;
	if(a[u]^c1){
		ans.push_back(u);
		nc2=c1^1;
	}
	for(int i=0;i<G[u].size();i++){
		int v=G[u][i];
		if(v!=f){
			dfs(v,u,nc1,nc2);
		}
	}
}
int main(){
	scanf("%d",&n);
	for(int i=1;i<n;i++){
		int u,v;
		scanf("%d%d",&u,&v);
		G[u].push_back(v);
		G[v].push_back(u);
	}for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1,x;i<=n;i++)scanf("%d",&x),a[i]^=x;
	dfs(1,0,0,0);
	printf("%d\n",(int)ans.size());
	for(int i=0;i<ans.size();i++)
		printf("%d\n",ans[i]);
}
