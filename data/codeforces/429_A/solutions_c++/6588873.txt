#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

vector<int> v[100005],s;
int a[100005],b[100005];

void dfs(int x,int p,int m,int c)
{
	int i,j,y;
	if((a[x]+c)%2!=b[x])
		m++,c++,s.push_back(x);
	c+=m;
	for(i=0,j=v[x].size();i<j;i++)
	{
		if((y=v[x][i])==p)
			continue;
		dfs(y,x,m,c);
	}
}

int main()
{
	int n,i,j,x,y;
	for(scanf("%d",&n),i=1;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		v[x].push_back(y);
		v[y].push_back(x);
	}
	for(i=1;i<=n;i++)
		scanf("%d",a+i);
	for(i=1;i<=n;i++)
		scanf("%d",b+i);
	dfs(1,0,0,0);
	printf("%d\n",j=s.size());
	for(i=0;i<j;i++)
		printf("%d\n",s[i]);
	return 0;
}
