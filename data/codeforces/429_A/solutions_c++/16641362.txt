#include<bits/stdc++.h>
using namespace std;
int n,x,y,init[100100],goal[100100],cnt,vis[100100];
vector<int>v[100100],res;
void dfs(int node,int is1,int is2,int e){
	vis[node]=1;
	if(e&&is1||!e&&is2)init[node]=!init[node];
	if(goal[node]!=init[node]){
		cnt++;
		res.push_back(node);
		(e)?is1=!is1:is2=!is2;
	}
	for(int j=0;j<v[node].size();j++)if(!vis[v[node][j]])dfs(v[node][j],is1,is2,!e);
}
int main(){
	cin>>n;
	for(int i=1;i<n;i++){
		cin>>x>>y;
		v[x].push_back(y);
		v[y].push_back(x);
	}
	for(int i=1;i<=n;i++)cin>>init[i];
	for(int i=1;i<=n;i++)cin>>goal[i];
	dfs(1,0,0,1);
	cout<<cnt<<endl;
	for(int i=0;i<res.size();i++)cout<<res[i]<<endl;
}
