#include<cstdio>
#include<vector>
using namespace std;

int n, a, b;
vector<int>e[111111], ans;
int ori[111111], goal[111111];

void dfs(int cur, int p, int f0, int f1) {
	if(ori[cur]^f0 != goal[cur]){
		ans.push_back(cur);
		f0=1-f0;
	}
	for(int i=0;i<e[cur].size();i++) {
		int nxt = e[cur][i];
		if(nxt==p) continue;
		dfs(nxt, cur, f1, f0);
	}
}

int main() {
	scanf("%d", &n);
	for(int i=1;i<n;i++) {
		scanf("%d%d", &a, &b);
		e[a].push_back(b);e[b].push_back(a);
	}
	for(int i=1;i<=n;i++) scanf("%d", &ori[i]);
	for(int i=1;i<=n;i++) scanf("%d", &goal[i]);
	
	ans.clear();
	dfs(1, -1, 0, 0);
	printf("%d\n", ans.size());
	for(int i=0;i<ans.size();i++)
		printf("%d\n", ans[i]);
	
	return 0;
}
