#include <cstdio>
#include <vector>
using namespace std;
vector<int> g[100010];
int n, ans;
int a[100010], b[100010];
int c[100010];
void dfs(int v, int w = -1, int m = 0, int k = 0) {
	if ((a[v] ^ b[v]) != m) {
		++ans;
		c[v] = 1;
		m ^= 1;
	}
	for (int i = 0; i < g[v].size(); i++) {
		int u = g[v][i];
		if (u == w) continue;
		dfs(u, v, k, m);
	}
}
int main() {
	scanf("%d", &n);
	for (int i = 0; i < n-1; i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		--x; --y;
		g[x].push_back(y);
		g[y].push_back(x);
	}
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	for (int i = 0; i < n; i++) {
		scanf("%d", &b[i]);
	}
	dfs(0);
	printf("%d\n", ans);
	for (int i = 0; i < n; i++) {
		if (c[i]) {
			printf("%d\n", i+1);
		}
	}
	return 0;
}
