#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long ll;
#define rep(i,n) for (int i=1;i<=n;++i)
const int N=200005;
int n,x,y,l,a[N],o[N],son[N],ed[N],next[N];
void dfs(int x,int fa,int f0,int f1)
{
	a[x]^=f0,swap(f0,f1); if (a[x]) o[++*o]=x,f1^=1;
	for (int p=son[x],y;p;p=next[p]) if ((y=ed[p])!=fa) dfs(y,x,f0,f1);
}
int main()
{
	scanf("%d",&n);
	rep(i,n-1) scanf("%d%d",&x,&y),
		ed[++l]=y,next[l]=son[x],son[x]=l,
		ed[++l]=x,next[l]=son[y],son[y]=l;
	rep(i,n) scanf("%d",a+i);
	rep(i,n) scanf("%d",&x),a[i]^=x;
	dfs(1,0,0,0);
	printf("%d\n",*o);
	rep(i,*o) printf("%d\n",o[i]);
	return 0;
}
