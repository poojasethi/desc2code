#include <bits/stdc++.h>
using namespace std;
int n,x,y,ans;
bool ini[100005],req[100005];
vector<int> a[100005],value;
void dfs(int node,int now,int aft,int dad){
	if(ini[node]^now!=req[node]){
		now^=1;
		ans++;
		value.push_back(node);
	}
	for(int i=0;i<a[node].size();i++)
		if(a[node][i]!=dad)dfs(a[node][i],aft,now, node);
}
int main() {
	cin>>n;
	for(int i=0;i<(n-1);i++){
		cin>>x>>y;
		a[y].push_back(x);
		a[x].push_back(y);
	}
	for(int i=1;i<=n;i++) cin>>ini[i]; 
	for(int i=1;i<=n;i++) cin>>req[i]; 

	dfs(1,0,0,0);
	cout<<ans<<endl;
	for(int i=0;i<ans;i++)cout<<value[i]<<endl;
}
