n = int(raw_input())
h = [0] * (n + 1)
h[1:n + 1] = map(int, raw_input().split())
tot = [0] * (n + 1)
for i in xrange(1, n + 1):
    tot[i] = tot[i - 1] + h[i]
f = [0] * (n + 1)
for i in xrange(1, n + 1):
    for j in xrange(i - 1, -1, -1):
        if tot[i] - tot[j] >= h[j]:
            f[i] = f[j] + i - j - 1
            h[i] = tot[i] - tot[j]
            break
print f[n]
