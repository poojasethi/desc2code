n = input()
h = map(int, raw_input().split())

tot = [0] * (n + 1)
dp = [0] * (n + 1)
ans = [0] * (n + 1)

for i in xrange(n):
    tot[i + 1] = tot[i] + h[i]

for i in xrange(1, n + 1):
    j = i - 1
    while True:
        if tot[i] - tot[j] >= dp[j]:
            dp[i] = tot[i] - tot[j]
            ans[i] = i - j - 1 + ans[j]
            break
        j -= 1
print ans[n]
