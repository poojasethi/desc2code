
def main():
    n = int(raw_input())
    a = map(int, raw_input().split())
    a = [0]+a
    H = [0]*(n+10)
    D = [0]*(n+10)

    for i in xrange(1, n+1):
        last = a[i]
        pos = i
        while last < H[pos-1]:
            pos -= 1
            last += a[pos]
        D[i] = D[pos-1]+1
        H[i] = last
    print(n-D[n])

if __name__ == '__main__':
    main()
