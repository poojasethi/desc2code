def ri():
    return map(int, raw_input().split())

def main():
    n = input()
    h = [0]+ri()
    dp,s = [0]*(n+1),[0]*(n+1)

    for k in xrange(n): h[k+1] += h[k]
    for i in xrange(1,n+1):
        for j in xrange(i-1,-1,-1):
            if h[i]-h[j] >= dp[j]:
                dp[i] = h[i] - h[j]
                s[i] = s[j] + i - j - 1
                break

    print s[n]

main()
