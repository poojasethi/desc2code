#include <cstdio>
#include <cstdlib>
int n,h[5011],s[5011],f[5011],g[5011];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&h[i]);
	for(int i=1;i<=n;i++)s[i]=s[i-1]+h[i];
	f[0]=0;g[0]=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=i-1;j>=0;j--)
		{
			if(s[i]-s[j]>=g[j])
			{
				if(f[j]+1>f[i])
				{
					f[i]=f[j]+1;
					g[i]=s[i]-s[j];
				}
			}
		}
	}
	printf("%d\n",n-f[n]);
	//system("pause");
}
