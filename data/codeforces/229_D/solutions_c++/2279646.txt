#include <iostream>
using namespace std;

int n, h[5555], f[5555], cnt[5555];

int main()
{
    cin >> n;
    for(int i = 0; i < n; i++) cin >> h[i];
    f[0] = h[0];
    cnt[0] = 0;
    for(int i = 1; i < n; i++){
        int last = h[i], j = i - 1;
        while(j >= 0 and last < f[j]) last += h[j--];
        f[i] = last;
        cnt[i] = (j >= 0 ? cnt[j] + (i - j - 1) : i);
    }
    cout << cnt[n - 1] << endl;
}
