#include <iostream>
using namespace std;
const int maxn=5000+100;
int d[maxn],sum[maxn],h[maxn],n,j;
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>h[i];
    for(int i=1;i<=n;i++)
    {
        sum[i]=h[i],j=i-1;
        while(j && sum[i]<sum[j])
            sum[i]+=h[j--];
        d[i]=j?d[j]+i-j-1:i-1;
    }
    cout<<d[n]<<endl;

    return 0;
}
