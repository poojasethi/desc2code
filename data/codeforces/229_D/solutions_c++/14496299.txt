#include <bits/stdc++.h>
using namespace std;
int ar[5050],dp[5050],dp2[5050];
int main(){
	int n;cin>>n;
	for(int i=1;i<=n;++i)cin>>ar[i];
	for(int i=1;i<=n;++i){
		int sum = ar[i],prev = i-1;
		while(prev && dp2[prev]>sum)sum+=ar[prev--];

		dp[i]=dp[prev]+i-prev-1 ;
		dp2[i]= sum;
	}
	cout<<dp[n]<<endl;
}
