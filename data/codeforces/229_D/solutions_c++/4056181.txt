#include <algorithm>
#include <stdio.h>
using    namespace std;

int N;
int H[5010];
int F[5010];
int DP[5010];

int main()
{
  scanf("%d",&N);
  for(int i=1 ; i<=N ; i++)
    scanf("%d",&H[i]);
  for(int i=1 ; i<=N ; i++)
  {
    int total=H[i],j;
    for(j=i-1 ; j && total<F[j] ; j--)
      total += H[j];
    F[i]  = total;
    DP[i] = j ? DP[j]+i-j-1 : i-1;
  }
  printf("%d\n",DP[N]);
  return 0;
}
