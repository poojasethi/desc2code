#include <bits/stdc++.h>

using namespace std;

int N,ar[5500],res;
int S[5500],dn[5500];

int main(){
	//~ freopen("towers.in","r",stdin);
	//~ freopen("towers.out","w",stdout);
	cin >> N;
	
	for( int i=1 ; i<=N ; i++ ) 	scanf(" %d",&ar[i]);
	
	for( int i=1 ; i<=N ; i++ ){
		int sum=ar[i],j=i-1;
		while( j && sum < S[j] ) sum+=ar[j--];
		S[i]=sum;
		dn[i]=dn[j]+i-j-1;
	}
	
	cout << dn[N];
	
	return 0;
	
}
