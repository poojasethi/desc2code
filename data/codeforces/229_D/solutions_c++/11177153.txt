#include<cstdio>
using namespace std;

int a[10000],f[10000],ans[10000];
int main(){
    int n;scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=n;i++){
		int sum=a[i],j=i-1;
		while(j && sum<f[j])sum+=a[j--];
		if(j)ans[i]=ans[j]+(i-j-1);else ans[i]=i-1;
		f[i]=sum;
	}
    printf("%d\n",ans[n]);
    return 0;
}
