#include <iostream>
#include <cstdio>
using namespace std;
int N,a[5010],b[5010],s[5010],dp[5010];
int main()
{
	scanf("%d", &N);
	int i,j;
	for (i=1; i<=N; i++)
	{
		scanf("%d", &a[i]);
		s[i]=s[i-1]+a[i];
	}
	for (i=1; i<=N; i++)
	{
		j=i-1;
		while (j!=0&&s[i]-s[j]<b[j])
			j--;
		dp[i]=dp[j]+i-j-1;
		b[i]=s[i]-s[j];
	}
	printf("%d\n", dp[N]);
	return 0;
}
