#include <stdio.h>

int dp[5010], a[5010], r[5010];

int main()
{
  int n, i;
  scanf("%d", &n);
  for (i = 1; i <= n; i++)
    scanf("%d", &a[i]);
  dp[0] = 0;
  
  for (i = 1; i <= n; i++)
    {
      int s = a[i], p;
      for (p = i; p && s < r[p - 1]; p--)
	s += a[p - 1];
      dp[i] = dp[p - 1] + i - p;
      r[i] = s;
    }

  printf("%d\n", dp[n]);

  return 0;
}
