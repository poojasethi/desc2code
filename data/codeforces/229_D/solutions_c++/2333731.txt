#include <iostream>
#define MAXN 5050

using namespace std;

int h[MAXN],f[MAXN],ans[MAXN];

int main(){
    int n;
    cin>>n;
    for (int i=1; i<=n; i++) cin>>h[i];
	for (int i=1; i<=n; i++)
	{
		int sum = h[i] ,j = i-1;
		while(j && sum< f[j]) sum+=h[j--];
		if (j) ans[i]=ans[j]+(i-j-1); else ans[i]=i-1;
		f[i]= sum;
	}
    cout<<ans[n];
    return 0;
}
