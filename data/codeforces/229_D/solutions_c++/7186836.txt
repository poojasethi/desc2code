#include<stdio.h>
#include<string.h>
int h[5001],sum[5001],d[5001];
int main()
{
	int n,i,j,u,v;
	while(~scanf("%d",&n))
	{
		sum[0]=0;
		for(i=1;i<=n;i++)
		{
			scanf("%d",&u);
			sum[i]=sum[i-1]+u;
		}
		h[0]=0;
		for(i=1;i<=n;i++)
			for(j=i-1;j>=0;j--)
				if(sum[i]-sum[j]>=h[j])
				{
					d[i]=d[j]+1;
					h[i]=sum[i]-sum[j];
					break;
				}
				else d[i]=d[j];
		printf("%d\n",n-d[n]);

	}
	return 0;
}
