#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cstdlib>
#define N 5009
using namespace std;
int a[N],f[N],h[N];
main(){
	int n,i,ans=0;
	cin>>n;
	
	for (i=1;i<=n;i++)
		scanf("%d",a+i);
	for (i=1;i<=n;i++){
		int t=a[i],j=i-1;
		while (t<f[j]) t+=a[j--];
		f[i]=t;
		h[i]=h[j]+1;
	}
	//for (i=1;i<=n;i++) cout<<f[i]<<endl;
	cout<<(n-h[n]);
}
