#include <iostream>

#define rep(i,n) for(int i=1;i<=n;i++)

using namespace std;

int n,h[5010],s,p,dp[5010],r[5010];

int main(){
  cin>>n;
  rep(i,n)cin>>h[i];
  rep(i,n){
    for(s=h[i],p=i;p && s<r[p-1];p--)s+=h[p-1];
    dp[i] = (p==i?dp[i-1]:dp[p-1]+i-p);
    r[i] = s;
  }
  cout<<dp[n]<<endl;

}
