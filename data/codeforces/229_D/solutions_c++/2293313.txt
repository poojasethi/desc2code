#include<iostream>
using namespace std;
int n,m,i,j,k,a[10000],b[10000],D[10000];
main()
{
    cin>>n;
    for (i=1;i<=n;i++)
    {
        cin>>a[i];
        a[i]+=a[i-1];
        for (j=i-1;j>=0;j--)
        if (b[j]<=a[i]-a[j])
        if (D[i]<D[j]+1){
        D[i]=D[j]+1;
        b[i]=a[i]-a[j];
        }
        }
    cout<<n-D[n]<<endl;
}
