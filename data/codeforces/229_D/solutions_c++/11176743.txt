#include<iostream>
#include<cstdio>
using namespace std;
int n;
int a[100005];
int s[1000005];
int sum[100005];
int f[100005];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		sum[i]=sum[i-1]+a[i];
		for(int j=i-1;j>=0;j--)
		{
			if(sum[i]-sum[j]>=s[j] && f[j]+1>f[i])
			{
				f[i]=f[j]+1;
				s[i]=sum[i]-sum[j];
			}
		}
	}
	printf("%d\n",n-f[n]);
	return 0;
}
