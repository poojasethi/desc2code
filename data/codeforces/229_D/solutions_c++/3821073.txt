#include<cstdio>

int h[5001],m[5001],t[5001];

int main()
{
    int n,i,j,m_;

    while(scanf("%d",&n)==1)
    {
        for(i=1;i<=n;i++)
            scanf("%d",&h[i]);

        for(i=1;i<=n;i++)
        {
            for(m_=h[i],j=i-1;j&&m_<m[j];j--)
                m_+=h[j];
            m[i]=m_;
            t[i]=t[j]+i-j-1;
        }

        printf("%d\n",t[n]);
    }

    return 0;
}
