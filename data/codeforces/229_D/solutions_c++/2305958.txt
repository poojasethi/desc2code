#include<cstdio>
#include<cstring>
#include<algorithm>
#define N 5500
using namespace std;

int s[N],f[N],dp[N];

int main()
{
    int n,i,j,x;

    scanf("%d",&n);
    s[0]=f[0]=dp[0]=0;
    for (i=1;i<=n;i++)
    {
        scanf("%d",&x);
        s[i]=s[i-1]+x;

        j=i-1;
        while (j&&s[i]-s[j]<f[j]) j--;
        f[i]=s[i]-s[j];
        dp[i]=dp[j]+i-j-1;
    }

    printf("%d\n",dp[n]);

    return 0;
}
