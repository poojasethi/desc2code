#include<cstdio>
const int N=5001;
int h[N],s[N],dp[N];
int main()
{
	int i,j,n;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&h[i]);
		s[i]=s[i-1]+h[i];
		for(j=i-1;j>=0;j--)
		{
			if(s[i]-s[j]>=h[j])
			{
				dp[i]=dp[j]+i-j-1;
				h[i]=s[i]-s[j];
				break;
			}
		}
	}
	printf("%d\n",dp[n]);
	return 0;
}
