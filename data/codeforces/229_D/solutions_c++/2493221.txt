#include <cstdio>
#define N 5555
int d[N],c[N],a[N],n;
int main(){
    scanf("%d", &n);
    for (int i = 1; i <= n; scanf("%d", a + i ++ ));
    d[1] = a[1];
    for (int i = 2; i <= n; i ++ ){
        for (int j = i - 1; j >= 0; j -- ){
            d[i] += a[j + 1];
            if (d[i] >= d[j]){
                c[i] = c[j] + (i - j) - 1;
                break;
            }
        }
    }
    printf("%d\n", c[n]);
    return 0;
}
