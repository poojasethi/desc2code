#include <cstdio>
#include <cstring>
int n,a[5010],f[5010],g[5010],s[5010];
int main(){
	scanf("%d",&n);
	for (int i = 1;i <= n;++i) scanf("%d",&a[i]),s[i] = s[i - 1] + a[i];
	for (int i = 1;i <= n;++i)
		for (int j = 0;j < i;++j) if (s[i] - s[j] >= g[j] && f[j] + 1 > f[i]) f[i] = f[j] + 1,g[i] = s[i] - s[j];
		else if (s[i] - s[j] >= g[j] && f[j] + 1 == f[i] && s[i] - s[j] < g[i]) g[i] = s[i] - s[j];
	printf("%d\n",n - f[n]);
	return 0;
}
