#include <iostream>

using namespace std;

typedef pair<int, int> pii;

const int MAXN = 5001;
pii dp[MAXN];
int h[MAXN];

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> h[i];
	dp[0] = pii(0, 0);
	for (int i = 0; i < n; i++) {
		int cur = 0;
		for (int j = i; j >= 0; j--) {
			cur += h[j];
			if (cur >= dp[j].second) {
				dp[i + 1] = pii(dp[j].first + 1, cur);
				break;
			}
		}
	}
	cout << n - dp[n].first << endl;
	return 0;
}
