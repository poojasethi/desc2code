#include <cstdio>

using namespace std;

int h[5001],f[5001],d[5001];

int main(void)
{
  int n;
  scanf("%d",&n);
  for(int i=1;i<=n;++i)
    scanf("%d",&h[i]);
  f[0]=0;
  d[0]=0;
  for(int i=1;i<=n;++i)
  {
    int j=i-1,s=h[i];
    while(d[j]>s)
      s+=h[j--];
    f[i]=f[j]+i-j-1;
    d[i]=s;
  }
  printf("%d\n",f[n]);
  return 0;
}
