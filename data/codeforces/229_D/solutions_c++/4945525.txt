#include <algorithm>
#include <stdio.h>
#define  maxn      5003
using    namespace std;

int n;
int h[maxn];
int ar[maxn];
int dp[maxn];

int main()
{
  scanf("%d",&n);
  for(int i=1 ; i<=n ; i++)
    scanf("%d",&h[i]);
  for(int i=1,j ; i<=n ; i++)
  {
    int sum=h[i];
    for(j=i-1 ; j>=1 && sum<ar[j] ; j--)  sum+=h[j];
    ar[i]=sum;
    dp[i]=dp[j]+(i-j-1);
  }
  printf("%d\n",dp[n]);
  return 0;
}
