#include <stdio.h>

const int MAXN = 5100;

int dp[MAXN], mh[MAXN], h[MAXN];
int main()
{
    int N; scanf("%d", &N);
    for (int i = 1; i <= N; ++i)
    {
        scanf("%d", &h[i]);
        int sum = h[i], j = i-1;
        for (; j && mh[j] > sum; --j) sum += h[j];
        dp[i] = dp[j]+i-1-j;
        mh[i] = sum;
    }
    printf("%d\n", dp[N]);
    //while (1);
    return 0;
}
