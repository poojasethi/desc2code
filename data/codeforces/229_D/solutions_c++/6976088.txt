#include<cstdio>
#define N 5005
int n,sm[N],a[N],hh[N],dp[N];
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;++i){
        scanf("%d",&a[i]);
        sm[i]=sm[i-1]+a[i];
        int j;
        for(j=i-1;hh[j]>sm[i]-sm[j];--j);
        dp[i]=dp[j]+(i-(j+1));
        hh[i]=sm[i]-sm[j];
    }
    printf("%d\n",dp[n]);
    return 0;
}
