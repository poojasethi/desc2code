#include <iostream>
#include <stdlib.h>
#include <stdio.h>

using namespace std;

int a[5050];
int f[5050];
int d[5050];
int n;

int main()
{
    cin >> n;
    for (int i=0;i<n;i++)
        cin >> a[i+1];
    for (int i=1;i<=n;i++)
    {
        int sum = a[i];
        int j = i-1;
        while (j && sum < f[j]) sum+=a[j--];
        d[i] = d[j] + i - j - 1;
        f[i] = sum;
    }
    
    cout << d[n];
    return 0;
}
