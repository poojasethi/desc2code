#include<cstdio>
int n,a[6000],b[6000],s[6000],dp[6000];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		s[i]=s[i-1]+a[i];
	}
	for (int i=1;i<=n;i++)
	{
		int j=i-1;
		while(j!=0&&s[i]-s[j]<b[j])j--;
		dp[i]=dp[j]+i-j-1;
		b[i]=s[i]-s[j];
	}printf("%d\n",dp[n]);
	return 0;
}
