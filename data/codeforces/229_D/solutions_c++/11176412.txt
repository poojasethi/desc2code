#include<cstdio>

using namespace std;

const int INF=1000000000;

int n;
int a[5010];
int f[5010],g[5010];

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",a+i);
		a[i]+=a[i-1];
		f[i]=INF;
	}
	f[0]=0;
	for(int i=1;i<=n;i++)
		for(int j=i-1;j>=0;j--)
			if(a[i]-a[j]>=g[j]&&f[j]+i-j-1<f[i])
			{
				f[i]=f[j]+i-j-1;
				g[i]=a[i]-a[j];
			}
	printf("%d\n",f[n]);
	return 0;
}
