#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 5010;

int n, h[N], dp[N], sum[N];

int main(){
	scanf("%d", &n);
	for(int i = 1; i <= n; ++i)
		scanf("%d", &h[i]), sum[i] = sum[i - 1] + h[i];
	
	for(int i = 1; i <= n; ++i)
		for(int j = i - 1; j >= 0; --j)
			if(sum[i] - sum[j] >= h[j]){
				h[i] = sum[i] - sum[j];
				dp[i] = dp[j] + i - j - 1;
				break;
			}
	
	printf("%d\n", dp[n]);
	return 0;
}
