#include<iostream>
#include<cstdio>
using namespace std;
int a[5010],b[5010],f[5010],d[5010],n,i,j; 
int main()
{
	cin>>n;
	for(i=1;i<=n;i++) scanf("%d",&a[i]),b[i]=b[i-1]+a[i];
	f[0]=d[0]=0; 
	for(i=1;i<=n;i++)
	{ 
		for(j=i-1;j&&b[i]-b[j]<d[j];j--);
		d[i]=b[i]-b[j],f[i]=f[j]+i-j-1;
	} 
	cout<<f[n]<<endl; 
	return 0;
}
