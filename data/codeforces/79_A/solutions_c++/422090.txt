#include <cstdio>
#include <iostream>

using namespace std;

int main()
{
	int x, y;
	cin >> x >> y;
	
	bool b = 0;
	
	while (y >= 2 && x * 10 + y >= 22)
	{
	//	cout << x << " " << y << endl;
		b = !b;
		if (b)
		{
			int c = min(x, 2);
			x -= c;
			y -= 2 + 10 * (2 - c);
		}
		else
		{
			int c = min(2 + (y - 2) / 10 * 10, 22);
			y -= c;
			x -= (2 - c / 10);
		}
	}
	
	if (b)
		cout << "Ciel";
	else
		cout << "Hanako";
	
	return 0;
}
