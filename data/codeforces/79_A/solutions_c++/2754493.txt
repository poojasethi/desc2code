#include<cstdio>
int main()
{
    int x,y;
    while(scanf("%d%d",&x,&y)!=EOF)
    {
        bool flag=1;
        while((x*10+y>=22)&&(y>=2))
        { 
          if(flag||y<12)
          {x-=2;y-=2;}
          else if(y<22)
          {x--;y-=12;}
          else y-=22;
          flag=!flag;
        }
        if(flag) puts("Hanako");
        else puts("Ciel");
    }
    return 0;
}
