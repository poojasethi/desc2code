#include <cstdio>

int x,y;

int main(){
	scanf("%d%d",&x,&y);
	bool c=1;
	while ((x>=2 && y >= 2) || y>=22 || (x >= 1 && y >= 12)){
		if (c){
			if ((x>=2 && y >= 2)){
				x-=2;
				y-=2;
			}
			else if (x >= 1 && y >= 12){
				--x;
				y-=12;
			}
			else {
				y -= 22;
			}
		}
		else {
			if (y>=22){
				y-=22;
			}
			else if (x >= 1 && y >= 12){
				--x;
				y-=12;
			}
			else {
				x-=2;
				y-=2;
			}
		}
		c = !c;
	}
	if (!c)
		printf("Ciel");
	else printf("Hanako");
}
