#include<bits/stdc++.h>
using namespace std;
int x,y;
int main(){
	cin >> x >> y;
	while(true){
		if(x>=2 && y>=2) x-=2,y-=2;

		else if(x) x--,y-=12;
		
		else y-=22;

		if(y < 0){cout << "Hanako"<<endl;return 0;}

		if(y>=22) y-=22;
		
		else if(y>=12) x--,y-=12;

		else
			x-=2,y-=2;

		if(x < 0 or y < 0){
			cout << "Ciel"<<endl;
			return 0;
		}
	}
}
