#include<cstdio>

int x,y;

bool check(int a,int b){
	if(x<a||y<b)return false;
	x-=a;y-=b;
	return true;
}

int main() {
	scanf("%d%d", &x, &y);
	while(1) {
		if(!check(2,2)&&!check(1,12)&&!check(0,22)){
			printf("Hanako\n");
			break;
		}
		if(!check(0,22)&&!check(1,12)&&!check(2,2)){
			printf("Ciel\n");
			break;
		}
	}

	return 0;
}
