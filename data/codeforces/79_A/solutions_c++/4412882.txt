#include <iostream>
#include <cstdio>

using namespace std;

int main(){
    long int x, y;
    bool w=1;
    cin >> x >> y;
    while ((x*10+y >= 22) && (y >= 2)){
        if (w || (y<12)){
            x-=2;
            y-=2;
        }
        else if (y < 22){
            x--;
            y-=12;
        }
        else
            y-=22;
        w=!w;
    }
    if (w)
        cout << "Hanako";
    else
        cout << "Ciel";     
}
