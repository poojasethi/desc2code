#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;

int main()
{
    int m,n;
    cin>>m>>n;
    while(1)
    {
       if(n<2||m*10+n<22){cout<<"Hanako" <<endl;return 0;}

       if(m>=2) {m-=2;n-=2;}
       else {n-=(22-10*m);m=0;}

       if(n<2||m*10+n<22){cout<<"Ciel" <<endl;return 0;}

       if(n>=22){n-=22;}
       else if(n>=12){n-=12;m-=1;}
       else {m-=2;n-=2;}
    }
    return 0;
}
