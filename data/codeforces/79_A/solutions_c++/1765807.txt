#include <iostream>

using namespace std;


int main() {
    int x,y;
    cin>>x>>y;
    while (1) {
        if (x>=2) x-=2, y-=2;
        else if (x) x--, y-=12;
        else y-=22;
        if (y<0) {
         cout<<"Hanako";
         return 0; 
         }
        if (y>=22) y-=22;
        else if (y>=12) x--, y-=12;
        else x-=2, y-=2;
        if (x<0 || y<0) {
         cout<<"Ciel";
         return 0; 
         }
    }
    return 0;
}
