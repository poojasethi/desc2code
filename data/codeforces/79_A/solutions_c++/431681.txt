#include <iostream>
using namespace std;

int main()
{
	__int64 n, m;
	bool f = true;
	cin>>n>>m;
	while( (10 * n + m) >= 22 && m >= 2)
	{
		if( f || m < 12 )
		{
			n -= 2 ; m -= 2;
		}
		else if( m < 22 )
		{
			n -= 1; m -= 12;
		}
		else
		{
			m -= 22;
		}
		f = !f;
	}
	if( f ) cout<<"Hanako\n";
	else cout<<"Ciel\n";
	//while(1);
	return 0;
}
