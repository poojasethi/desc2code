#include <iostream>
#include <cstdio>

using namespace std;

int main(){
	long int x, y, d;
	bool w=1;
	cin >> x >> y;
	d = (x/2 > y/24) ? y/24 : x/2;
	x-=d*2;
	y-=d*24;
	while ((x*10+y >= 22) && (y >= 2)){
		if (w || (y<12)){
			x-=2;
			y-=2;
		}
		else if (y < 22){
			x--;
			y-=12;
		}
		else
			y-=22;
		w=!w;
	}
	if (w)
		cout << "Hanako";
	else
		cout << "Ciel";		
}
