#include <cstdio>
#include <iostream>
using namespace std;

int main() {
	int x, y;
	scanf("%d%d", &x, &y);
	bool f=true;
	int win;
	for(int i=1; f; ++i) {
		if(i%2) {
			if(x>=2 && y>=2) x -= 2, y -= 2;
			else if(x==1 && y>=12) x-=1, y-=12;
			else if(x==0 && y>=22) y-=22;
			else win = 0, f = false;
		} else {
			if(y>=22) y-=22;
			else if(x>=1 && y>=12) x-=1, y-=12;
			else if(x>=2 && y>=2) x-=2, y-=2;
			else win = 1, f = false;
		}
	}
	puts(win?"Ciel":"Hanako");
}
