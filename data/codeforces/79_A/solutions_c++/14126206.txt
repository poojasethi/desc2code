#include<stdio.h>
int main()
{
	long int a=0,x,y;
	scanf("%ld %ld",&x,&y);
	while(x*100+y*10>=220&&y>=2)
	{
		if(a%2==0)
		{
			if(x>=2)
			{
				x-=2;
				y-=2;
			}
			else if(x==1)
			{
				x-=1;
				y-=12;
			}
			else 
				y-=22;
		}
		else
		{
			if(y>=22)
			y-=22;
			else if(y<22&&y>=12)
			{
				x-=1;
				y-=12;
			}
			else 
			{
				x-=2;
				y-=2;
			}
		}
		a++;
	}
	if(a%2==1)
	printf("Ciel");
	else
	printf("Hanako");
}
