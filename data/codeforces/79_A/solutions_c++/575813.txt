#include<iostream>
using namespace std;
int main()
{
	int x,y;
	while(cin>>x>>y)
	{
		int flag=1;
		while((x==0&&y>=22)||(x==1&&y>=12)||(x>=2&&y>=2))
		{
			if(flag==1)
			{
				x-=2;
				y-=2;
				while(x<0)
				{
					x++;
					y-=10;
				}
			}
			else
			{
				y-=22;
				while(y<0)
				{
					y+=10;
					x-=1;
				}
			}
			flag=-flag;
		}
		if(flag==-1) cout<<"Ciel"<<endl;
		else cout<<"Hanako"<<endl;
	}
	return 0;
}
