#include <iostream>

using namespace std;

int main()
{
int x,y;
cin>>x>>y;
int p=0;
while(x*100+y*10>=220 && y>=2)
{
if(p==0)
{
if(x==0) y-=22;
else if(x==1)
{
x=0;
y-=12;
}
else if(x>=2)
{
x-=2;
y-=2;
}
}
else
{
if(y>=22) y-=22;
else if(y>=12)
{
y-=12;
x-=1;
}
else
{
y-=2;
x-=2;
}
}
p=1-p;
}
if(p==0) cout<<"Hanako";
else cout<<"Ciel";
return 0;
}
