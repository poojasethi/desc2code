#include<iostream>

using namespace std;

int main(){

  int a,b;
  bool flg=false;
  cin >> a >> b;
  while(b>=22||(a>=1&&b>=12)||(a>=2&&b>=2)){
    if(!flg){
      if(a>=2){ a-=2; b-=2; }
      else if(a==1){ a--; b-=12; }
      else b-=22;
    }else{
      if(b>=22){ b-=22; }
      else if(b>=12){ b-=12; a--; }
      else{ b-=2; a-=2; }
    }
    flg=!flg;
  }
  if(!flg)cout << "Hanako" << endl;
  else cout << "Ciel" << endl;

  return 0;
}
