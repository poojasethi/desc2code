#include <stdio.h>

int main()
{
	int x, y, i;
	scanf("%d%d", &x, &y);
	for (i=0; x>=0 && y>=0; i++)
	{
		if (i&1)
		{
			if (y>=22) y-=22;
			else 
			if (y>=12)
			{
				x-=1;
				y-=12;
			}
			else 
			{
				x-=2;
				y-=2;
			}
		}
		else 
		{
			if (x>=2)
			{
				x-=2;
				y-=2;
			}
			else 
			if (x>=1)
			{
				x-=1;
				y-=12;
			}
			else y-=22;
		}
	}
	if (i&1) printf("Hanako\n");
	else printf("Ciel\n");
	return 0;
}
