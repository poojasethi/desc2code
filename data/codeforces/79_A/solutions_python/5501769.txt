x, y = map(int, raw_input().strip().split())

opt = [(2,2),(1,12),(0,22)]

while True:
    for i in opt:
        if x>=i[0] and y>=i[1]:
            x=x-i[0]
            y=y-i[1]
            break
    else:
        print "Hanako"
        break
    for i in reversed(opt):
        if x>=i[0] and y>=i[1]:
            x=x-i[0]
            y=y-i[1]
            break
    else:
        print "Ciel"
        break
