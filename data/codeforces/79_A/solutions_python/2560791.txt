x, y = map(int, raw_input().split())
t = 0
while x * 100 + y * 10 >= 220:
	if t:
		if y >= 22:
			y -= 22
		elif y >= 12:
			y -= 12
			x -= 1
		else:
			y -= 2
			x -= 2
	else:
		if x >= 2:
			x -= 2
			y -= 2
		elif x >= 1:
			x -= 1
			y -= 12
		else:
			y -= 22
	if x < 0 or y < 0:
		break
	t ^= 1
print ['Hanako', 'Ciel'][t]
