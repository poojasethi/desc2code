a, b = map(int, raw_input().split())
get = ((2, 2), (1, 12), (0, 22))
while True:
    for k in get:
        if a >= k[0] and b >= k[1]:
            a -= k[0]
            b -= k[1]
            break
    else:
        print "Hanako"
        break
    for k in reversed(get):
        if a >= k[0] and b >= k[1]:
            a -= k[0]
            b -= k[1]
            break
    else:
        print "Ciel"
        break
