x, y = map(int, raw_input().split())
turn = 0
while 100 * x + y * 10 >= 220 and y >= 2:
	if turn & 1 == 0 :
		if(x == 0) : 
			y -= 22
		elif(x == 1) :
			x -= 1
			y -= 12
		else :
			x -= 2
			y -= 2
	else : 
		if(y >= 22) :
			y -= 22
		elif(y >= 12) :
			x -= 1
			y -= 12
		else :
			x -= 2
			y -= 2
	turn += 1
if turn & 1 == 0:
	print 'Hanako'
else : 
	print 'Ciel'
