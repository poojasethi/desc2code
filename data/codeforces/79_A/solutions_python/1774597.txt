x, y = map(int, raw_input().split())
round = 0
while True:
    if round % 2 == 0:
        dx = min(2, x)
        dy = min(22 - 10 * dx, y)
    else:
        dy = 22 if y >= 22 else 12 if y >= 12 else 2 if y >= 2 else y
        dx = min((22 - dy)/10, x)
    if 10 * dx + dy < 22:
        print 'Hanako' if round % 2 == 0 else 'Ciel'
        break
    x -= dx
    y -= dy
    round += 1
