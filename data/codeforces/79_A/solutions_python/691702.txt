s=raw_input().split()
x=int(s[0])
y=int(s[1])

c=0
while 1:
	if(c==0):
		xx=min(2,x)
		x-=xx
		falta=22-xx*10
		if(falta>y):
			print 'Hanako'
			exit(0)
		y-=falta
	else:
		if(y>=22):
			y-=22
		elif(y>=12):
			y-=12
			x-=1
			if(x<0):
				print 'Ciel'
				exit(0)
		elif(y>=2):
			y-=2
			x-=2
			if(x<0):
				print 'Ciel'
				exit(0)
		else:
			print 'Ciel'
			exit(0)
	c=1-c
