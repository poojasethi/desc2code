n, m = map(int , raw_input().split())

turn = 0
L = [[2,1,0],[0,1,2]]

while True:
    succ = 0
    for x in L[turn]:
        if x <= n and (220 - 100 * x)//10 <= m:
            n = n - x
            m = m - (220 - 100 * x)//10
            succ = 1
            break
 
    if succ == 0:
        break
    turn = 1 - turn

if turn == 0: print 'Hanako'
else: print 'Ciel'
