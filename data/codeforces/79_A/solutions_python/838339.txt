x,y = map(int, raw_input().split())
i = 0
while x >= 0 and y >= 0:
    t = 220
    if i == 0:
        if x >= 2:
            x -= 2
            t -= 200
        else:
            t -= 100*x
            x -= x
        y -= t/10
    else: 
        if y >= 22:
            y -= 22
            t -= 220
        elif y >= 12:
            y -= 12
            t -= 120
        else:
            y -= 2
            t -= 20
        x -= t/100
    i ^= 1
print ["Ciel", "Hanako"][i]
