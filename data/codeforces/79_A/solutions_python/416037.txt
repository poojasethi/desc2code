ways = [ (2, 2), (1, 12), (0, 22) ]
hun, ten = [ int(x) for x in raw_input().split() ]
turn = 0
while hun >= 0 and ten >= 0:
    for way in ways[::1-turn*2]:
        if hun - way[0] >= 0 and ten - way[1] >= 0:
            hun -= way[0]
            ten -= way[1]
            break
    else: hun = -1
    turn = (turn + 1) % 2
print "Hanako" if turn else "Ciel"
