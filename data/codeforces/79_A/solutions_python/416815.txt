import sys

a, b = map(int, sys.stdin.readline().strip().split(' '))
players = ['Ciel','Hanako']

c = {'Ciel': [(2, 2), (1,12), (0,22)], 'Hanako':[(0, 22), (1,12), (2,2)]}

while True:
  player = players.pop(0)
  sum = 0

  for ai, bi in c[player]:
    if ai <= a and bi <= b:
      a -= ai
      b -= bi
      break
  else:
    if player == 'Ciel':
      print 'Hanako'
    else:
      print 'Ciel'
    break

  players.append(player)
