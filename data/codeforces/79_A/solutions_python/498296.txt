x, y = map(int, raw_input().split())
t = 0
while True:
	if t % 2 == 0:
		if x > 1 and y > 1:
			x -= 2
			y -= 2
		elif x > 0 and y > 11:
			x -= 1
			y -= 12
		elif y > 21:
			y -= 22
		else:
			print 'Hanako'
			break
	else:
		if y > 21:
			y -= 22
		elif y > 11 and x > 0:
			x -= 1
			y -= 12
		elif y > 1 and x > 1:
			x -= 2
			y -= 2
		else:
			print 'Ciel'
			break
	t += 1
