#include<bits/stdc++.h>
using namespace std;
int a[30004],i,n,d,b;
int dp[30500][500];
int dfs(int c,int b)
{
    if (dp[c][b]>=0) return dp[c][b];
    int res=0;
    for (int i=-1;i<2;i++)
    {
        int v=c+b+i;
        if (v>=30001||v<=c)
            continue;
        res=max(res,dfs(v,b+i));
    }
    dp[c][b]=res+a[c];
    return dp[c][b];
}
main()
{
    scanf("%d%d",&n,&d);
    while (n--)
        scanf("%d",&b),a[b]++;
  memset(dp, -1, sizeof dp);
    printf("%d",dfs(d,d));
}
