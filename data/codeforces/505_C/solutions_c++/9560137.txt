#include <bits/stdc++.h>
#define maxn 30009
using namespace std;
int ar[maxn][500],a[maxn];
int n,d,snc=1,x;
int main()
{
	scanf("%d %d",&n,&d);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		a[x]++;
	}
	ar[d][250]=a[d]+1;
	for(int i=d;i<=x;i++)
	{
		for(int j=0;j<500;j++) 
		{
			if(ar[i][j]==0) 
			{
				continue;
			}
			int k=j-250+d,deg=ar[i][j];
			snc=max(snc,deg);
			for(int t=max(1,k-1);t<=k+1;t++)
			{
				if(i+t<=x)
				{
					ar[i+t][t-d+250]=max(ar[i+t][t-d+250],deg+a[i+t]);
				}
			}
		}
	}
	printf("%d\n",snc-1);
	return 0;
}
