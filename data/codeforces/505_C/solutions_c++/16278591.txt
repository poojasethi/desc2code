#include <bits/stdc++.h>
using namespace std;
int a[30002];
int dp[30002][900];
bool mark[30002][900];
int maxx;
int dfs(int x,int y)
{
    if(x>maxx||y==0)
        return 0;
    else if(mark[x][y])
        return dp[x][y];
    else
        dp[x][y]=a[x]+max(dfs(x+y-1,y-1),max(dfs(x+y,y),dfs(x+y+1,y+1)));
    mark[x][y]=true;
    return dp[x][y];
}
int main()
{
    int n,l,x;
    cin>>n>>l;
    for(int i=1;i<=n;i++)
    {
        cin>>x;
        a[x]++;
        maxx=max(maxx,x);
    }
    cout<<dfs(l,l);
}
