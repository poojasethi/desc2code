#include <bits/stdc++.h>
#define pb push_back
#define mp make_pair
using namespace std;
int maxi=0,a[30009];
int dd[30005][500],D;
int dp(int cur,int DD)
{
    int d=DD-(D-250);
    if(cur>maxi||DD<1)
        return 0;
    if(dd[cur][d]!=-1)return dd[cur][d];
    return dd[cur][d]=a[cur]+max(max(dp(cur+DD-1,DD-1),dp(cur+DD,DD)),dp(cur+DD+1,DD+1));
}
int main()
{
    int n,d,x;
    memset(dd,-1,sizeof(dd));
    cin>>n>>D;
    for(int i=0;i<n;i++)
    {
        cin>>x;
        a[x]++;
        maxi=max(maxi,x);
    }
    cout<<dp(D,D)<<endl;

    return 0;
}
