#include <iostream>
#include <cstdio>
using namespace std;

int main(){
   int n,d[500]={0},s[500]={0},v[500]={0};
   int m,l[500]={0},w[500]={0},p[500]={0};
   int i,j,k,MIN,ans=0;
   
   cin>>n;
   
   for (i=0;i<n;i++)
      cin>>d[i]>>s[i]>>v[i];
   
   cin>>m;
   
   for (i=0;i<m;i++)
      cin>>l[i]>>w[i]>>p[i];
   
   for (i=0;i<n;i++){
      MIN=-1;
      for (j=0;j<m;j++){
         if (l[j]<v[i])
            continue;
         
         k=(l[j]/v[i])*w[j];
         k=(2*(d[i]+s[i])+k-1)/k;
         k*=p[j];
         if (MIN>k || MIN==-1)
            MIN=k;
      }
      ans+=MIN;
   }   
   cout<<ans<<endl;
   return 0;
}
