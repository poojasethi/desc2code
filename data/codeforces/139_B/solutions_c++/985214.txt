#include<iostream>
using namespace std;
const int maxn=600;
int arr[maxn][2],w[maxn][3];
int main(){
	int n,m;
	cin>>n;
	for(int i=0;i<n;i++){
		int a,b;
		cin>>a>>b>>arr[i][1];
		arr[i][0]=2*(a+b);
	}
	cin>>m;
	for(int i=0;i<m;i++)
		cin>>w[i][0]>>w[i][1]>>w[i][2];
	long long ans=0;
	for(int i=0;i<n;i++){
		long long mn=100000000000000ll;
		for(int j=0;j<m;j++){
			long long cst=(arr[i][0]+w[j][1]-1)/w[j][1];
			int t=w[j][0]/arr[i][1];
			if(!t)
				continue;
			cst=(cst+t-1)/t;
			cst*=w[j][2];
			mn=min(cst,mn);
		}
		ans+=mn;
	}
	cout<<ans<<endl;
}
