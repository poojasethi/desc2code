#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
	int n,l,b,h[512],p[512],n1,lr[512],wr[512],pr[512],i,j;
	long long pric,min;
	int price[512][512];
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d %d %d",&l,&b,&h[i]);
		p[i]=2*(l+b);
	}
	scanf("%d",&n1);
	for(i=0;i<n1;i++)
		scanf("%d %d %d",&lr[i],&wr[i],&pr[i]);
	for(i=0;i<n;i++)
	{
		for(j=0;j<n1;j++)
		{
			price[i][j]=pr[j]*ceil((p[i]*1.0)/(wr[j]*(lr[j]/h[i])));
//			printf("%d  ",price[i][j]);
		}
	}
	pric=0;
	for(i=0;i<n;i++)
	{
		min=1000000000;
		for(j=0;j<n1;j++)
			if(min>price[i][j]&&price[i][j]>-1)
				min=price[i][j];
		if(min!=1000000000)
			pric=pric+min;
	}
	printf("%lld\n",pric);
}
