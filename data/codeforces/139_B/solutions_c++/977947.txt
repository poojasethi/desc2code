#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int n,m;
int l[600],h[600];
int a[600],b[600],money[600];
int main()
{
	scanf("%d",&n);
	for(int i = 0;i < n;i++)
	{
		int t1,t2,t3;
		scanf("%d%d%d",&t1,&t2,&t3);
		l[i] = 2*(t1+t2);
		h[i] = t3;
	}
	scanf("%d",&m);
	for(int i = 0;i < m;i++)
		scanf("%d%d%d",&a[i],&b[i],&money[i]);
	int ans = 0;
	for(int i = 0;i < n;i++)
	{
		int min = -1;
		for(int j = 0;j < m;j++)
		{
			int x = (a[j]/h[i]);
			if(x==0) continue;
			int y = l[i]/b[j];
			if(y*b[j] != l[i]) y++;
			int z = y / x;
			if(z * x != y) z++;
			if(min == -1 || min > z * money[j])
				min = z*money[j];
		}
		ans += min;
	}
	printf("%d\n",ans);
}
