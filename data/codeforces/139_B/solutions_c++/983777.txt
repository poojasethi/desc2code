#include<stdio.h>

int main(){
    int n, m, total, tm, tp, sl;
    int x[500], y[500], z[500];
    int l[500], w[500], p[500];
    scanf("%d", &n);
    for(int i = 0; i < n; i++)
        scanf("%d%d%d", &x[i], &y[i], &z[i]);
    scanf("%d", &m);
    for(int i = 0; i < m; i++)
        scanf("%d%d%d", &l[i], &w[i], &p[i]);
    total = 0;
    for(int i = 0; i < n; i++){
        tm = -1;
        for(int j = 0; j < m; j++){
            sl = (l[j]/z[i])*w[j];
            if(sl == 0) continue;
            tp = (2*(x[i]+y[i])+sl-1)/sl*p[j];
            if(tm < 0 || tp < tm)
                tm = tp;
        }
        total += tm;
    }
    printf("%d\n", total);
}
