#include <stdio.h>

int N, M;

int x, y, h;
int P[500], H[500];
int i, j;
int W[500], L[500], C[500];
int k, m, pr;

int main() {
 
  scanf("%d", &N);
  for (i = 0; i < N; i++) {
    scanf("%d %d %d", &x, &y, &h);
    P[i] = (x+y)*2; H[i] = h;
  }
  
  scanf("%d", &M);
  for (i = 0; i < M; i++)
    scanf("%d %d %d", &L[i], &W[i], &C[i]);
  
  int sol = 0;
  for (i = 0; i < N; i++) {
    
    int minv = 2000000000;
    
    for (j = 0; j < M; j++) {
      k = L[j] / H[i];
      m = k * W[j];
      if (m == 0) continue;
      pr = ((P[i] - 1) / m + 1) * C[j];
      if (pr < minv) minv = pr;
    }
    sol += minv;
  }
  
  printf("%d\n", sol);
  return 0;
}
