# include <cstdlib>
# include <iostream>
# include <cstring>
# include <cstdio>
# define N 550
using namespace std;
int l[N],h[N],a[N],b[N],c[N];
int main (void)
{
	int x,y,n,m;
	cin>>n;
	for (int i=1;i<=n;i++)
	{
		cin>>x>>y>>h[i];
		l[i]=(x+y)<<1;
	}
	cin>>m;
	for (int i=1;i<=m;i++)
		cin>>a[i]>>b[i]>>c[i];
	int MIN,ans=0;
	int heng,shu,num;
	for (int i=1;i<=n;i++)
	{
		MIN=0x3f3f3f3f;
		for (int j=1;j<=m;j++)
		{
			if(a[j] < h[i])	continue;
			heng=(l[i]+b[j]-1)/b[j];
			shu=a[j]/h[i];
			num=(heng+shu-1)/shu;
			MIN=min(MIN,c[j]*num);
		}
		ans+=MIN;
	}
	cout<<ans<<endl;
	return 0;
}
