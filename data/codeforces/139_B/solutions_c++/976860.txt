# include <iostream>
# include <cstdio>
# include <algorithm>
# include <cstring>
# include <cmath>
using namespace std;
const int maxn=550;
int L[maxn],W[maxn],H[maxn];
int l[maxn],w[maxn],h[maxn];
int N,M,ans;

int main()
{
 scanf("%d",&N);
 for (int i=0; i<N; i++)
 {
  scanf("%d%d%d",&L[i],&W[i],&H[i]);
  L[i]+=W[i], L[i]*=2;
 }
 scanf("%d",&M);
 for (int i=0; i<M; i++)
  scanf("%d%d%d",&l[i],&w[i],&h[i]);
 ans=0;
 for (int i=0; i<N; i++)
 {
  int temp=0x3fffffff;
  for (int j=0; j<M; j++)
  {
   if (l[j]<H[i]) continue;
   int t=l[j]/H[i];
   int n=(L[i]+w[j]-1)/w[j];
   int tmp=(n+t-1)/t;
   temp=min(temp, tmp*h[j]);
  }
  ans+=temp;
 }
 printf("%d\n",ans);
 return 0;
}
