#include <cstdio>
#include <iostream>
#include <climits>

using namespace std;

int N,M,m,res;
int wall[555][5];
int ar[555][5];

int solve(int room,int wp){
	
	if(ar[room][3]>wall[wp][1])
		return INT_MAX;
	
	int need=(ar[room][1]+ar[room][2])*2;
	int one=(wall[wp][1]/ar[room][3])*wall[wp][2];
	int res=(need/one)*wall[wp][3];
	
	if(need%one) res+=wall[wp][3];
	
	return res;
}

int main(){
	
	cin >> N ;
	
	for(int i=1;i<=N;i++)
		cin >> ar[i][1] >> ar[i][2] >> ar[i][3];
		
	cin >> M ;
	
	for(int i=1;i<=M;i++)
		cin >> wall[i][1] >> wall[i][2] >> wall[i][3] ;
		
	for(int i=1;i<=N;i++){
		
		m=solve(i,1);
		
		for(int j=2;j<=M;j++)
			m=min(m,solve(i,j));
			
		res+=m;
	}

	cout << res << endl ;
	
	return 0;
}
