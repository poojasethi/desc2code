#include<stdio.h>

int main()
{
    int rc,pc,i,sum=0,minp,j,c,l;
    int r[500][3],p[500][3];
    scanf("%d",&rc);
    for(i=0;i<rc;i++)
        scanf("%d%d%d",&r[i][0],&r[i][1],&r[i][2]);
    scanf("%d",&pc);
    for(i=0;i<pc;i++)
        scanf("%d%d%d",&p[i][0],&p[i][1],&p[i][2]);
    for(i=0;i<rc;i++)
    {
        minp=0x3a3a3a3a;
        c=(r[i][0]+r[i][1])*2;
        for(j=0;j<pc;j++)
        {
            l=p[j][0]/r[i][2];
            if(l==0) continue;
            l*=p[j][1];
            if(c%l==0)
                l=c/l;
            else l=c/l+1;
            minp=minp>l*p[j][2]?l*p[j][2]:minp;
        }
        sum+=minp;
    }
    printf("%d\n",sum);
    return 0;
}
