#include<iostream>
#include<algorithm>
using namespace std;
int main(){
int n;
cin>>n;
int a[n][3];
int i;
for(i=0;i<n;i++)cin>>a[i][0]>>a[i][1]>>a[i][2];
int m;
cin>>m;
int b[m][3];

for(i=0;i<m;i++)cin>>b[i][0]>>b[i][1]>>b[i][2];
//Calc optimal for each n
long long int sumtot=0;
for(i=0;i<n;i++){
int per = 2*(a[i][0]+a[i][1]);
int j;
long long int  ms=1000000000;

for(j=0;j<m;j++)
{
	if((b[j][0]/a[i][2])==0)
	continue;
	double k=(double)per/((b[j][0]/a[i][2])*(double)b[j][1]);
	if((k-(int)k)!=0)
	k=k+1;
	long long int k0=k;
	ms=min(ms,k0*b[j][2]);
	
	
}
sumtot+=ms;

}

cout<<sumtot<<endl;


return 0;
}
