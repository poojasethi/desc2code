#include<iostream>
using namespace std;
int main()
{
    int i,j,k,l,m,n,o,p,q;
    long s=0,min1;
    int a[1000],b[1000],c[1000],d[1000],e[1000],f[1000];
    cin>>n;
    for(i=0;i<n;i++)
        cin>>a[i]>>b[i]>>c[i];
    cin>>m;
    for(i=0;i<m;i++)
        cin>>d[i]>>e[i]>>f[i];
    for(i=0;i<n;i++)
    {
        l=(a[i]+b[i])*2;min1=2000000000;
        for(j=0;j<m;j++)
        {
            k=d[j]/c[i];
            o=l/e[j];
            if(l%e[j]!=0) o++;
            if(k==0) continue;
            p=o/k;
            if(o%k!=0) p++;
            if(p*f[j]<min1) min1=p*f[j];
        }
     s=s+min1;
    }
    cout<<s;
    return 0;
}
