#include <iostream>
using namespace std;

int room[3][512];
int wall[3][512];

int main(){
	int n,m;
	cin >> n;
	for(int i=0;i<n;i++)
		cin>>room[0][i]>>room[1][i]>>room[2][i];

	cin>>m;

	for(int i=0;i<m;i++)
		cin>>wall[0][i]>>wall[1][i]>>wall[2][i];

	int ans=0;
	int min;
	int buf;
	int l,h;
	int tw,tl;
	for(int i=0;i<n;i++){
		l=(room[0][i]+room[1][i])*2;
		h=room[2][i];
		min=0xffffff;
		for(int j=0;j<m;j++){
			tw=wall[1][j];
			tl=wall[0][j];
			tl/=h;
			if(tl==0) continue;
			tw*=tl;
			buf=wall[2][j]*(l/tw+((l%tw==0)?0:1));
			if(min>buf) min=buf;
		}
		ans+=min;
	}

	cout<<ans<<endl;

	return 0;
}
