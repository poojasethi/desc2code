#include <cstdio>
#include <algorithm>
using namespace std;
const int maxn = 500;
struct Node {
	int a, b, c;
};
Node w[maxn], p[maxn];

int main() {
	int n, m;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d %d %d", &w[i].a, &w[i].b, &w[i].c);
	}
	scanf("%d", &m);
	for (int i = 0; i < m; i++) {
		scanf("%d %d %d", &p[i].a, &p[i].b, &p[i].c);
	}
	int sum = 0;
	for (int i = 0; i < n; i++) {
		int c = 2e9;
		int length = w[i].a * 2 + w[i].b * 2;
		for (int j = 0; j < m; j++) if (p[j].a >= w[i].c) {
			int b = p[j].a / w[i].c * p[j].b;
			c = min(c, (length + b - 1) / b * p[j].c);
		}
		sum += c;
	}
	printf("%d\n", sum);
	return 0;
}
