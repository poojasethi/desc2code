#include<stdio.h>
#include<string.h>
int main()
{
	int par,i,j,r,p,a[501][3],pa[501][3],ans[501],x,y,z,wid,c,min,f=0;
	scanf("%d",&r);
	for(j=0;j<r;j++)
		scanf("%d%d%d",&a[j][0],&a[j][1],&a[j][2]);
	scanf("%d",&p);
	for(i=0;i<p;i++)
		scanf("%d%d%d",&pa[i][0],&pa[i][1],&pa[i][2]);
	for(j=0;j<r;j++)
	{
		min=10000000;
		par=2*(a[j][0]+a[j][1]);
		for(i=0;i<p;i++)
		{
			x=pa[i][0]/a[j][2];
			if(x>0)
			{
				wid=x*pa[i][1];
				c=par/wid;
				if(par%wid)
					c++;
			ans[i]=pa[i][2]*c;
			if(ans[i]<min)
				min=ans[i];
			}
		}
		f+=min;
	}
		printf("%d\n",f);
	return 0;
}
