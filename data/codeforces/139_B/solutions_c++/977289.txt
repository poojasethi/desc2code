#include <iostream>
using namespace std;

int main(){
	int n;
	cin >> n;
	int a[550] , b[550] , c[550];
	for( int i=0 ; i<n ; i++ )
		cin >> a[i] >> b[i] >> c[i];
	int m;
	cin >> m;
	int l[550] , w[550] , p[550];
	for( int i=0 ; i<m ; i++ )
		cin >> l[i] >> w[i] >> p[i];
	int sum = 0 , mn , num;
	for( int i=0 ; i<n ; i++ ){
		mn = 1000000000;
		for( int j=0 ; j<m ; j++ ){
			if( l[j] < c[i] )
				continue;
			if( (2*(a[i]+b[i])) % w[j] )
				num = ((2*(a[i]+b[i])) / w[j])+1;
			else
				num = (2*(a[i]+b[i])) / w[j];
			if( num % (l[j]/c[i]) )
				num = (num / (l[j]/c[i]))+1;
			else
				num = num / (l[j]/c[i]);
			if( mn > num*p[j] )
				mn = num*p[j];
		}
		sum += mn;
	}
	cout << sum << endl;
	return 0;
}
