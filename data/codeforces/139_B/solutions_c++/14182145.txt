#include<iostream>

using namespace std;

const int inf=1e9+7;
int n,m,l[512],w[512],h[512],L[512],W[512],P[512],s,k,ans;

int main()
{
  cin>>n;
  for(int i=0;i<n;i++)
    cin>>l[i]>>w[i]>>h[i];
  cin>>m;
  for(int i=0;i<m;i++)
    cin>>L[i]>>W[i]>>P[i];
  for(int i=0,c=inf;i<n;ans+=c,i++,c=inf)
    for(int j=0;j<m;j++)
      {
	s=((l[i]+w[i])*2+W[j]-1)/W[j];
	k=L[j]/h[i];
	if(k)
	  c=min(c,((s+k-1)/k)*P[j]);
      }
  cout<<ans<<endl;
  return 0;
}
