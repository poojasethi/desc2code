#include<iostream>
using namespace std;
int n,m,al[1000],ah[1000],aw[1000],bx[1000],by[1000],bp[1000],ans=2147483640,ta;
main()
{
    cin>>n;
    for(int i=1; i<=n; i++)
    cin>>al[i]>>aw[i]>>ah[i];
    cin>>m;
    for(int j=1; j<=m; j++)
    cin>>bx[j]>>by[j]>>bp[j];
    for(int i=1; i<=n; i++){
    for(int j=1; j<=m; j++)
    if(ah[i]<=bx[j])
    {
    int k=bx[j]/ah[i],l=2*al[i]+2*aw[i],u=l/(by[j]*k);
    if(l%(by[j]*k)>0)u++;
    u*=bp[j];
    if(u<ans)ans=u;
    }
    ta+=ans; ans=2147483640;}
    cout<<ta;
}
