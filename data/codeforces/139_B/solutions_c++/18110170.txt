#include <bits/stdc++.h>
using namespace std;
long long a[505][4],b[505][4],dp[505][505];
int main() 
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i][0]>>a[i][1]>>a[i][2];
		a[i][3]=2*(a[i][0]+a[i][1]);
	}
	int m;
	cin>>m;
	for(int i=0;i<m;i++)
	{
		cin>>b[i][0]>>b[i][1]>>b[i][2];
	}
	long long sum=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			int x=b[j][0]/a[i][2];
			if(x==0)
			dp[i][j]=pow(10,12);
			else
			{
			x=x*b[j][1];
		
			if(a[i][3]%x!=0)
			x=(a[i][3]/x)+1;
			else
				x=(a[i][3]/x);
			dp[i][j]=x*b[j][2];
			}
	//	cout<<dp[i][j]<<" ";
		}
	//	cout<<"\n";
		sum+=*min_element(dp[i],dp[i]+m);
	}
	cout<<sum;
}
