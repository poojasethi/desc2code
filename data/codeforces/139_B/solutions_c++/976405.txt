#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

int a[500], b[500], c[500], ans[500];

int main(int argc, char** argv)
{
	int n;
	scanf("%d", &n);
	for(int i = 0; i<n; ++i){
		scanf("%d%d%d", &a[i], &b[i], &c[i]);
		ans[i] = 2000000000;
	}
	int m;
	scanf("%d", &m);
	while(m--){
		int h, w, p;
		scanf("%d%d%d", &h, &w, &p);
		for(int i = 0; i<n; ++i){
			if(h < c[i])
				continue;
			int l = (a[i]+b[i])*2;
			int cost = (l-1)/(h/c[i]*w)+1;
			cost *= p;
			if(cost < ans[i])
				ans[i] = cost;
		}	
	}
	long long sum = 0;
	for(int i = 0; i<n; ++i){
		sum += ans[i];
	}
	cout << sum << endl;
	return 0;
}
