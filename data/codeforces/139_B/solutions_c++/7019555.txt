#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <iostream>
#include <fstream>
#include <map>
#include <vector>
#include <algorithm>

using namespace std;


int n, m, t, u, s;
int l[600], w[600], h[700], a[700], b[700], c[700];

int main() {
	
	int i;
	s = 0;
	
	cin>>n;
	
	for(int i=0;i<n;i++)cin>>l[i]>>w[i]>>h[i];
	
	cin>>m;
	
	for(i=0;i<m;i++)cin>>a[i]>>b[i]>>c[i];
	
	for(i=0;i<n;i++) {
		
		t=999999999;
		
		for(int j=0;j<m;j++) 
		{
	
			u=a[j]/h[i]*b[j];
			if(u) t=min(t,(2*(w[i]+l[i])+u-1)/u*c[j]);
		}
		
		s+=t;
	}
	
	cout<<s;
}
