#include<cstdio>
#include<numeric>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

const int INF=1<<29;

int main(){
	int n; scanf("%d",&n);
	int x[500],y[500],z[500],ans[500];
	rep(i,n) scanf("%d%d%d",x+i,y+i,z+i), ans[i]=INF;

	int m; scanf("%d",&m);
	rep(_,m){
		int h,w,cost; scanf("%d%d%d",&h,&w,&cost);
		rep(i,n){
			if(h<z[i]) continue;
			int w2=h/z[i]*w;
			ans[i]=min(ans[i],(2*(x[i]+y[i])+w2-1)/w2*cost);
		}
	}

	printf("%d\n",accumulate(ans,ans+n,0));

	return 0;
}
