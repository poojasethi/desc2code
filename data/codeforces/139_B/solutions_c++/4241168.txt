int t,n,m,i,j,x1,x2,k,i1,i2,a[555][4],b[555][4],d[55555];
long long ans;
int main()
{
  cin>>n;
 for (i=0;i<n;i++) cin>>a[i][1]>>a[i][2]>>a[i][3];
 cin>>m;
 for (i=0;i<m;i++) cin>>b[i][1]>>b[i][2]>>b[i][3];
 for (i=0;i<n;i++)
 {
  t=1e9;
  for (j=0;j<m;j++)
  {
   int s=a[i][1]*2+a[i][2]*2;
   int k=b[j][2]*(b[j][1]/a[i][3]);
   if (k==0) continue;

   int col=(s+k-1)/k;

   t=min(t,col*b[j][3]);
  }
  ans+=t;
 }

 }
