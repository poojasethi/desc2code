#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <climits>

#define MAXN 502

using namespace std;

int N, M;

int r[MAXN][3];
int w[MAXN][3];

int main()
{
	cin >> N;
	for(int i=0; i < N; i++)
		for(int j=0; j < 3; j++)
			cin >> r[i][j];
	cin >> M;
	for(int i=0; i < M; i++)
		for(int j=0; j < 3; j++)
			cin >> w[i][j];
	int res=0;
	for(int i=0; i < N; i++)
	{
		int mmin= INT_MAX;
		int h=r[i][2];
		int t= (r[i][1]+ r[i][0])*2;
		for(int j=0; j < M; j++)
		{
			int c= w[j][0]/h;
			c*=w[j][1];
			if(!c)
				continue;
			int a= t/c;
			if(t%c)
				a++;
			int p= a*w[j][2];
			if(p < mmin)
				mmin= p;					
		}
		res+= mmin;
	}
	cout << res << endl;
}
