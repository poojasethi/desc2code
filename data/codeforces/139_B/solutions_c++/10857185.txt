#include <cstdio>

int main(){
	int n,m;
	scanf("%d",&n);
	int L[n],W[n],H[n];
	for(int i=0;i<n;i++){
		scanf("%d%d%d",&L[i],&W[i],&H[i]);
	}
	scanf("%d",&m);
	int l[m],w[m],p[m];
	for(int i=0;i<m;i++){
		scanf("%d%d%d",&l[i],&w[i],&p[i]);
	}
	int t=0;
	for(int i=0;i<n;i++){
		int x=1e9,ll=H[i],ww=(L[i]+W[i])*2;
		for(int j=0;j<m;j++){
			if(l[j]<ll){
				continue;
			}
			int kl,kw;
			kl=l[j]/ll;
			kw=w[j]*kl;
			kl=ww/kw;
			if(ww%kw){
				kl++;
			}
			if(kl*p[j]<x){
				x=kl*p[j];
			}
		}
		t+=x;
	}
	printf("%d\n",t);
	return 0;
}
