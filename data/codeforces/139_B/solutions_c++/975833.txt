#include <iostream>
using namespace std;
const int mm = 2000*1000*1000;
int pr(int l, int w, int h, int l0, int w0, int p)
{
    if(h > l0) return mm;
    int sum = (w+l)*2;
    int q = l0 / h * w0;
    return (sum / q) * p + (bool(sum % q)) * p;
}
int main()
{
    int m, n, a[500][3], b[500][3];
    cin >> n;
    for(int i = 0; i < n; i++)
        cin >> a[i][0] >> a[i][1] >> a[i][2];
    cin >> m;
    for(int i = 0; i < m; i++)
        cin >> b[i][0] >> b[i][1] >> b[i][2];
    int ans = 0;
    for(int i = 0; i < n; i++)
    {
        int best = mm;
        for(int j = 0; j < m; j++)
            best = min(best, pr(a[i][0], a[i][1], a[i][2], b[j][0], b[j][1], b[j][2]));
        ans += best;
    }
    cout << ans;
}
