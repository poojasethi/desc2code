#include<cstdio>
int a[510][3],b[510][3],n,m;
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		for(int j=0;j<3;++j)
			scanf("%d",&a[i][j]);
	scanf("%d",&m);
	for(int i=0;i<m;++i)
		for(int j=0;j<3;++j)
			scanf("%d",&b[i][j]);
	long long ret=0;
	for(int i=0;i<n;++i)
		if(a[i][2]>0){
		long long mm=(long long)100000000*(long long)10000000;
		//printf("%I64d\n",mm);
		for(int j=0;j<m;++j)
			if(b[j][0]>0&&b[j][1]>0){
			int k=b[j][1]*(b[j][0]/a[i][2]);
			if(k==0)
				continue;
			long long tmp=2*(a[i][0]+a[i][1])/k;
			if(tmp*k<2*(a[i][0]+a[i][1]))
				tmp++;
			tmp*=b[j][2];
			if(tmp<mm)
				mm=tmp;
		}
		ret+=mm;
	}
	printf("%lld\n",ret);
	return 0;
}
