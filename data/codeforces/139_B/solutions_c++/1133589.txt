#include<iostream>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
int n,m;
int a[505],b[505],c[505],d[505],e[505],f[505];
int main()
{
    cin>>n;
    rep(i,n)cin>>a[i]>>b[i]>>c[i];
    cin>>m;
    rep(i,m)cin>>d[i]>>e[i]>>f[i];
    /*
    10/3 = 3
    20 / 3 / 1
    */
    int ans = 0;
    rep(i,n)
    {
        int minn = 1000000000;
        rep(j,m)
        {
            int k = d[j]/c[i];
            if(k==0)continue;
            int l = 2*(a[i]+b[i]);
            int mi = ((l-1)/(k*e[j])+1)*f[j];
            //cout<<mi<<endl;
            minn = min(minn,mi);
            //cout<<minn<<endl;
        }
        ans+=minn;
    }
    cout<<ans<<endl;
}
