#include <iostream>
#include <algorithm>

using namespace std;

int N,M,rl[510],rw[510],rh[510],wl[510],ww[510],wp[510],result;

int main()
{
	cin >> N;
	for (int i=0;i<N;i++)
		cin >> rl[i] >> rw[i] >> rh[i];
	cin >> M;
	for (int i=0;i<M;i++)
		cin >> wl[i] >> ww[i] >> wp[i];
	for (int i=0;i<N;i++)
	{
		int min_cost=1000000;
		for (int j=0;j<M;j++)
			if (wl[j]>=rh[i])
			{
				int k=wl[j]/rh[i]*ww[j],p=(rl[i]+rw[i])*2;
				min_cost=min(min_cost,(p/k+(p%k!=0))*wp[j]);
			}
		result+=min_cost;
	}
	cout << result << endl;
	return 0;
}
