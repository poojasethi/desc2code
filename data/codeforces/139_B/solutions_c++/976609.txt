using namespace std;
int main(){
  int i,j;
  int n;
  cin>>n;
  int l[500],w[500],h[500];
  for(i=0;i<n;++i)
    cin>>l[i]>>w[i]>>h[i];
  int m;
  cin>>m;
  int a[500],b[500],c[500];
  for(i=0;i<m;++i)
    cin>>a[i]>>b[i]>>c[i];
  int sm=0;
  for(i=0;i<n;++i){
    int mn=1<<30;
    for(j=0;j<m;++j){
      if(h[i]>a[j])
	continue;
      mn=min(mn,(int)ceil((double)(l[i]*2+w[i]*2)/(a[j]/h[i]*b[j]))*c[j]);
    }
    sm+=mn;
  }
  cout<<sm<<endl;
  return 0;
}
