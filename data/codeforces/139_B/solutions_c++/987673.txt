#include<iostream>
#include<algorithm>
using namespace std;
#define z 600

int n,m,l[z],w[z],h[z],a[z],b[z],c[z],t,u,s=0;
int main() {
	cin>>n;
	for(int i=0;i<n;i++)cin>>l[i]>>w[i]>>h[i];
	cin>>m;
	for(int i=0;i<m;i++)cin>>a[i]>>b[i]>>c[i];
	for(int i=0;i<n;i++) {
		t=999999999;
		for(int j=0;j<m;j++) {
			u=a[j]/h[i]*b[j];
			if(u) t=min(t,(2*(w[i]+l[i])+u-1)/u*c[j]);
		}
		s+=t;
	}
	cout<<s;
}
