#include <cstdio>
#define FOR(x,y,z) for(int x=(y);x<=(z);++x)
int n,m,x[505],y[505],z[505],l[505],w[505],p[505];
int main(){
	scanf("%d",&n);
	FOR(i,1,n) scanf("%d%d%d",&x[i],&y[i],&z[i]);
	scanf("%d",&m);
	FOR(i,1,m) scanf("%d%d%d",&l[i],&w[i],&p[i]);
	int ans=0;
	FOR(i,1,n){
		int tmp=(int)(1e9);
		FOR(j,1,m){
			if (l[j]<z[i]) continue;
			int tmp2=2*(x[i]+y[i]);
			if (tmp2%w[j]>0) tmp2=tmp2/w[j]+1;
			else tmp2/=w[j];
			if (tmp2%(l[j]/z[i])>0) tmp2=tmp2/(l[j]/z[i])+1;
			else tmp2/=(l[j]/z[i]);
			if (tmp>tmp2*p[j]) tmp=tmp2*p[j];
		}
		ans+=tmp;
	}
	printf("%d\n",ans);
	return 0;
}
