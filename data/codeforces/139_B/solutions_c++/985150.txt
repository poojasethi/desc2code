#include<iostream>
#include<cmath>
using namespace std;
int main()
{
  int n;
  cin>>n;
  int a[n+1],b[n+1],c[n+1];
  for(int i=0;i<n;i++)
    {
      cin>>a[i]>>b[i]>>c[i];
    }
  int m;
  cin>>m;
  int a2[m+1],b2[m+1],c2[m+1],d2[n+1];
  for(int i=0;i<m;i++)
    {
      cin>>a2[i]>>b2[i]>>c2[i];
    }
  for(int i=0;i<n;i++)
    {
      long long int min=9999999;
      float d=2*(a[i]+b[i]);
      for(int j=0;j<m;j++)
	{
	  long long int d3=floor(a2[j]/c[i]);
	  if(d3!=0)
	    {
	  d3=d3*b2[j];
	  double d4=d/d3;
	  d3=ceil(d4);
	  d3=d3*c2[j];
	  if(d3<min)
	    min=d3;
	    }
	}
      d2[i]=min;
    }
  long long int s=0;
  for(int i=0;i<n;i++)
    {
      s=s+d2[i];
    }
  cout<<s<<endl;

}
