#include<cmath>
#include<iostream>
using namespace std;
int gcd(int a,int b)
{
    if(!b)return a;
    return gcd(b,a%b);
}
int main()
{
    int n,m,a[3][505],l,w,c,cost[505],b,sum=0;
    cin>>n;
    for(int i=0;i<n;++i)
        {cin>>a[0][i]>>a[1][i]>>a[2][i];cost[i]=100000000;}
    cin>>m;
    for(int i=0;i<m;++i)
    {
        cin>>l>>w>>c;
        for(int i=0;i<n;++i)
        {
            b=l/a[2][i];
            if(b!=0){cost[i]=min(cost[i],int (c*ceil(ceil(2.0*(a[0][i]+a[1][i])/w)/b)));}
        }
    }
    for(int i=0;i<n;++i)
        sum+=cost[i];
    cout<<sum<<endl;

}
