#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
#include<cstring>
#include<string>
#include<iostream>
#define inf 1<<20
using namespace std;

int L[501],W[501],H[501],LAR[501],ANC[501],P[501];

int main(){
	int n,m;
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d%d%d",&L[i],&W[i],&H[i]);
	}
	scanf("%d",&m);
	for(int i=0;i<m;++i){
		scanf("%d%d%d",&LAR[i],&ANC[i],&P[i]);
	}
	long long sum=0;
	for(int i=0;i<n;++i){
		int mini=inf;
		int per=L[i]*2+W[i]*2;
		for(int j=0;j<m;++j){
			int x=LAR[j]/H[i];
			if(x==0)continue;
			int cub=x*ANC[j];
			int y=((per-1)/cub)+1;
			int dinero=y*P[j];
			mini=min(mini,dinero);
		}
		sum+=mini;
	}
	cout<<sum<<endl;
}
