#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

int main()
{int n,ans=0,m,dl1[510],sh1[510],v1[510],dl2[510],sh2[510],st1[510],m1,perim,m2,min,min1;
//freopen("in.txt","r",stdin);
cin>>n;
for (int i=0; i<n; i++) cin>>dl1[i]>>sh1[i]>>v1[i];
cin>>m;
for (int i=0; i<m; i++) cin>>dl2[i]>>sh2[i]>>st1[i];
for (int i=0; i<n; i++)
{
    min=1000000;
    perim=2*(dl1[i]+sh1[i]);
    for (int j=0; j<m; j++)
    if (v1[i]<=dl2[j])
    {
        m1=(dl2[j]/v1[i]);
        if (perim%(m1*sh2[j])!=0) m2=(perim/(m1*sh2[j])+1);
        else m2=perim/(m1*sh2[j]);
        if (st1[j]*m2<min) min=st1[j]*m2; m2=0;
    }
ans+=min;
}
cout<<ans;
    return 0;
}
