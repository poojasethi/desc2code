#include<cstdio>
#include<cstring>

using namespace std;

struct Roll{
	int l , w , p; 
}data[505]; 
int	n,ans,b[505][3],m;
int main()
{
	scanf("%d", &n); 
	for (int i = 0; i<n; i++) 
		scanf("%d%d%d" , &b[i][0],&b[i][1], &b[i][2] ); 
	scanf("%d" ,&m ); 
	for (int i = 0; i<m; i++)
		scanf("%d%d%d" , &data[i].l,&data[i].w , &data[i].p ); 
	ans = 0; 
	while ( n -- )
	{
		int x = b[n][0], y = b[n][1], h = b[n][2]; 
		int l = x*2 + y * 2; 
		int w = -1; 
		for (int i = 0; i<m; i++)
		{
			if ( data[i].l < h ) continue; 
			int k = data[i].l/h; 
			k = k * data[i].w; 
			int t = (l%k == 0 ? l/k : l/k+1 ); 
			t = t*data[i].p; 
			if ( w == -1 || w > t ) w = t; 
		}
		ans += w; 
	}
	printf("%d\n" , ans ); 
	
}
