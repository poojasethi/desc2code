#include <iostream>
using namespace std;
long long x[505],y[505],z[505],a[505],b[505],c[505],n,m,ans,res,k,p,i,j;
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)
        cin>>x[i]>>y[i]>>z[i];
    cin>>m;
    for(i=1;i<=m;i++)
        cin>>a[i]>>b[i]>>c[i];
    for(i=1;i<=n;i++){
        ans=1e9;
        p=(x[i]+y[i])*2;
        for(j=1;j<=m;j++){
            k=a[j]/z[i];
            if(k) ans=min(ans,((p-1)/(k*b[j])+1)*c[j]);
        }
        res+=ans;
    }
    cout<<res<<endl;
}
