n = input()
r = [0] * n
for i in range(n):
	r[i] = [int(x) for x in raw_input().split(" ")]
m = input()
p = [0] * m
for i in range(m):
	p[i] = [int(x) for x in raw_input().split(" ")]
ans = 0
for i in r:
	c = (i[0] + i[1]) * 2
	t = min((c + w[0] / i[2]* w[1] - 1) / (w[0] / i[2] * w[1]) * w[2] for w in p if w[0] >= i[2])
	ans += t
print ans
