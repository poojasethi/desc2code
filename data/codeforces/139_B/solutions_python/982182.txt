from __future__ import division
import itertools as it
from sys import stdin
from math import ceil

def solve(rooms, papers):
    def cost(r, p):
        a, b, c = r
        l, w, v = p

        k = l // c # have vertically from each paper
        if k == 0: return 1e6
        horiz = (2*a + 2*b) / w # need horizontally for room
        return v * ceil(horiz / k)

    for r in rooms:
        yield min(cost(r, p) for p in papers)

def main():
    for l in stdin:
        n = int(l)
        rooms = [map(int, next(stdin).split()) for _ in xrange(n)]
        m = int(next(stdin))
        papers = [map(int, next(stdin).split()) for _ in xrange(m)]

        print int(sum(solve(rooms, papers)))

if __name__ == '__main__':
    main()
