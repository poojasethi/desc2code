def main():
    n = int(raw_input())
    rooms = []
    for i in range(n):
        length, width, height = map(int, raw_input().split())
        rooms.append(((length + width) * 2, height))
    m = int(raw_input())
    wallpapers = []
    for i in range(m):
        wallpapers.append(map(int, raw_input().split()))
    total_cost = 0
    for total_length, height in rooms:
        min_cost = 1000000
        for length, width, cost in wallpapers:
            num_strips = length / height
            if num_strips == 0:
                continue
            num_rolls = total_length / (num_strips * width)
            if total_length % (num_strips * width) != 0:
                num_rolls += 1
            min_cost = min(min_cost, cost * num_rolls)
        total_cost += min_cost
    print total_cost
    
main()
