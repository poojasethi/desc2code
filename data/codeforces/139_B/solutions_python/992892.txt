f = lambda : map(int, raw_input().split()) 
n, L = input(), []
for i in xrange(n): L.append(f())
m, A = input(), []
for i in xrange(m): A.append(f())
res = 0
for x in L:
    best = 2 ** 30
    for y in A:
        if x[2] <= y[0]:
            num = (2 * (x[0] + x[1]) + y[1] - 1) / y[1]
            each = y[0] / x[2]
            best = min(best, (num + each - 1) / each * y[2])
    res += best
print res
