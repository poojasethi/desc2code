n = input()
a = []
for i in xrange(n):
    a.append([int(x) for x in raw_input().split()])
m = input()
b = []
for i in range(m):
    b.append([int(x) for x in raw_input().split()])
ans = 0
for i in a:
    p = 2*(i[0]+i[1])
    best = 1000000000
    for j in b:
        k = j[0]/i[2]
        k *= j[1]
        if k == 0: continue
        x = p/k + (p%k > 0)
        x *= j[2]
        best = min(x, best)
    ans += best
print ans
