#!/usr/bin/env python
# -*- coding: utf-8 -*-
import sys

def roundup(a, b):
    return (a + b - 1) / b;

n = int(sys.stdin.readline())
rooms = []
for i in xrange(n):
    rooms.append([int(t) for t in sys.stdin.readline().split()])

m = int(sys.stdin.readline())
costs = []
for i in xrange(m):
    costs.append([int(t) for t in sys.stdin.readline().split()])

res = 0
for i in xrange(n):
    results = []
    L, W, H = rooms[i]
    P = 2 * (L + W)
    for j in xrange(m):
        l, w, c = costs[j]
        if l >= H:
            results.append(c * roundup(P, (l / H) * w))
    res += min(results)

print res
