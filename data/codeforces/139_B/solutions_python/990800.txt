#!/usr/bin/env python
# coding: utf-8

import math

n = int(raw_input().strip())

rooms = []
for i in xrange(n):
    rooms.append(
        map(int, raw_input().split()))

m = int(raw_input().strip())

wallpapers = []
for i in xrange(m):
    wallpapers.append(
        map(int, raw_input().split()))

ans = 0

for r in rooms:
    tmp = []
    rh = r[2]
    rw = (r[0] + r[1]) * 2
    for w in wallpapers:
        wh = w[0]
        ww = w[1]
        wc = w[2]
        if  wh >= rh:
            tmp.append(wc * math.ceil(rw / float(ww * (wh // rh))))
    ans += min(tmp)

print int(ans)
