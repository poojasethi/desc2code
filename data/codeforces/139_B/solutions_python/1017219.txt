#!/usr/bin/env python

from sys import stdin, stdout

rooms = [None] * int(stdin.readline())
for i in range(len(rooms)):
  p = [int(x) for x in stdin.readline().split(' ')]
  rooms[i] = [2*(p[0] + p[1]), p[2]]

paper = [None] * int(stdin.readline())
for i in range(len(paper)):
  paper[i] = [int(x) for x in stdin.readline().split(' ')]

cost = 0
for r in rooms:
  room_cost = 0
  for p in paper:
    if p[0] < r[1]:
      continue
    c = r[0] / p[1] / (p[0] / r[1])
    if c * p[1] * (p[0] / r[1]) < r[0]:
      c += 1
    if c * p[2] < room_cost or room_cost == 0:
      room_cost = c * p[2]
  cost += room_cost

stdout.write(str(cost))
