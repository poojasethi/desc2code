f = lambda : map(int, raw_input().split()) 
n, L = input(), []
for i in xrange(n): L.append(f())
m, A = input(), []
for i in xrange(m): A.append(f())
res = 0
for i in xrange(n):
    best = 2 ** 30
    for j in xrange(m):
        if L[i][2] <= A[j][0]:
            num = (2 * (L[i][0] + L[i][1]) + A[j][1] - 1) / A[j][1]
            each = A[j][0] / L[i][2]
            best = min(best, (num + each - 1) / each * A[j][2])
    res += best
print res
