#!/usr/bin/python

#import pdb

n = int(raw_input())

rooms = []
for i in xrange(n):
	rooms.append(map(int, raw_input().split()))

m = int(raw_input())
rollTypes = []
for i in xrange(m):
	rollTypes.append(map(int, raw_input().split()))

totalPrice = 0
for (L, W, H) in rooms:
	A = 2 * (L*H + W*H)
	if A == 0:
		continue
	minCostRoom = 10**20
	for (l, w, price) in rollTypes:
		#pdb.set_trace()
		if l*w == 0: continue
		if l < H: continue
		w_per_roll = (l / H) * w
		nrolls = (2 * (L + W) - 1) / w_per_roll + 1
		tprice = nrolls * price
		if tprice < minCostRoom:
			minCostRoom = tprice
			#print "Using new best: %d rolls at which give %d width per roll at price %d" % (nrolls, w_per_roll, price)
	totalPrice += minCostRoom

print totalPrice
