import sys

sys.stdin = open("input.txt", "r")
sys.stdout = open("output.txt", "w")

dir = sys.stdin.readline().strip()
rail = int(sys.stdin.readline().strip())

if dir == "front":
	print 'L' if rail == 1 else 'R'
else:
	print 'R' if rail == 1 else 'L'
