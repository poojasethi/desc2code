f = open('input.txt','r')
out = open('output.txt','w')
r = lambda: f.readline().strip()

entered = r()
a = int(r())

if entered == 'front':
    if a == 1:
        out.write('L')
    else:
        out.write('R')
elif entered == 'back':
    if a == 1:
        out.write('R')
    else:
        out.write('L')
