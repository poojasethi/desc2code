#!/usr/bin/env python

with open("input.txt", "r") as inf:
    with open("output.txt", "w") as ouf:
        door = inf.readline()[:-1]
        hold = int(inf.readline())

        left = (hold == 1)
        if door == "back":
            left = not left
        
        if left:
            ouf.write("L\n")
        else:
            ouf.write("R\n")
