source = open('input.txt', 'r')
result = open('output.txt', 'w')
door = source.readline().strip()
rail = int(source.readline())
if door == 'front' and rail == 1: result.write('L')
elif door == 'front' and rail == 2: result.write('R')
elif door == 'back' and rail == 1: result.write('R')
elif door == 'back' and rail == 2: result.write('L')
result.close()
source.close()
