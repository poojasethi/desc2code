f = open("input.txt", "r")
o = open("output.txt", "w")
rail = 0
door = f.readline().strip()
rail = int(f.readline().strip())
if rail > 0:
    if door == "front":
        if rail == 1:
            o.write("L")
        else:
            o.write("R")
    else:
        if rail == 1:
            o.write("R")
        else:
            o.write("L")
    o.close()
