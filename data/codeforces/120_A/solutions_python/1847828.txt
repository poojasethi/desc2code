try:
    inp=open('input.txt','r')
    out=open('output.txt','w')
    side=inp.readline()
    hand=int(inp.readline())
except IOError:
    print('Error at IO')

if (side=='front\n' or side=='front'):
    if hand==1:
        out.write('L')
    else:
        out.write('R')
elif (side=='back\n' or side=='back'):
    if hand==1:
        out.write('R')
    else:
        out.write('L')
else:
    print('Input data is invalid')

inp.close()
out.close()
