f = open('input.txt', 'rb')
data = f.readlines()
f.close()

z = []
for l in data:
	z.append(l)

door = z[0]
p = int(z[1])

if door[0] == 'f' and p == 1:
	res = 'L\n'
if door[0] == 'f' and p == 2:
	res = 'R\n'
if door[0] == 'b' and p == 1:
	res = 'R\n'
if door[0] == 'b' and p == 2:
	res = 'L\n'

#print res

f = open('output.txt', 'wb')
f.write(res)
f.close()
