from sys import stdin

d = {}

def next():
	return stdin.readline().strip()

def main():
	t = int(next())
	c = 0
	for i in range(t):
		t, l = next().split()
		if t not in d:
			d[t] = set()
			d[t].add(l)
			c += 1
		else:
			if l in d[t]:
				continue
			else:
				d[t].add(l)
				c += 1
	
	print "%d\n" % c

if __name__ == "__main__":
	main()
