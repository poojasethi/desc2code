if __name__ == "__main__":
    budget = input()
    plastic_price = input()
    glass_price = input()
    glass_return = input()
    print max(
            budget / plastic_price,
            (budget - glass_price) / (glass_price - glass_return) + max(
                1 + ((budget - glass_price) % (glass_price - glass_return) + (glass_return if budget > glass_price else 0)) / plastic_price,
                (glass_price if budget > glass_price else 0) / plastic_price
            )
    )
