#!/usr/bin/env python
from sys import stdin as cin

def main():
    n = int(next(cin).strip())
    a = int(next(cin).strip())
    b = int(next(cin).strip())
    c = int(next(cin).strip())
    d = b - c
    if a < d:
        return n // a
    if n < b:
        return n // a

    k = (n - b) // d
    n -= k*d
    while n >= b:
        k += 1
        n -= d
    k += n // a
    return k

print main()
