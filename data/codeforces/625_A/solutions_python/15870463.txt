n = input()
a = input()
b = input()
c = input()

def calc_b(n1):
    d = b-c
    if n1 < b:
        return 0
    x,y = 0,n1
    k = 0
    while x<=y: 
        k = (x+y)/2
        if n1 - d*k < b and n1 - (k-1)*d >= b:
            return k
        if n1 - d*k >= b:
            x = k+1
        else:
            y = k-1
    return 0

m1 = n/a + calc_b(n%a) 
k = calc_b(n)
m2 = k + (n - k*(b-c))/a
print max(m1,m2)
