import sys,copy,math,heapq,itertools as it,fractions,re,bisect,collections as coll

n = int(raw_input())
a = int(raw_input())
b = int(raw_input())
c = int(raw_input())

ans = n/(b - c)
rem = n % (b - c)
if rem < c:
    cnt = (c - rem + b - c - 1)/(b - c)
    ans -= cnt
    rem += cnt*(b - c)
ans += rem/a
if b > n: ans = 0
print max(ans, n/a)
