def solve():
    n = int(raw_input())
    a = int(raw_input())
    b = int(raw_input())
    c = int(raw_input())

    res = 0
    if n < a and n < b:
        pass
    if n < a and n >= b:
        res = (n - c) / (b - c)
    if n >= a and n < b:
        res = n / a
    if n >= a and n >= b:
        if (a > b - c):
            res = (n - c) / (b - c) + ((n - c) % (b - c) + c) / a
        else:
            res = n / a

    print res

solve()
