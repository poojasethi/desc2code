# -*- coding: utf-8 -*-


if __name__ == '__main__':
    n = int(raw_input())
    a = int(raw_input())
    b = int(raw_input())
    c = int(raw_input())

    if a <= (b-c):
        result = n / a
    else:
        # here the main logic
        if n >= b:
            result = (n-b) / (b-c) + 1
            remaining_n = n - result * (b-c)
            result += remaining_n / a
        else:
            result = n / a
        # here the main logic
    print(str(result))
