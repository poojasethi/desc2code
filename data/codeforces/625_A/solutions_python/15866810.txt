def give(n,a,b,c):
        ans=0;
        x=max(0,(n-c)/(b-c));
        ans=x;
        if(x>0):
            n-=(x*b-(x)*c);
        if(n>0):
            ans+=(n/a);
        return ans
n=int(raw_input())
a=int(raw_input())
b=int(raw_input())
c=int(raw_input())
mx1=n/a+give((n-a*(n/a)),n+1,b,c)
ans=max(mx1,give(n,a,b,c))
print ans
