from sys import stdin, stdout
n  = long(stdin.readline().rstrip())
a  = long(stdin.readline().rstrip())
b  = long(stdin.readline().rstrip())
c  = long(stdin.readline().rstrip())

def f(n, a, b, c):
    gc = b - c

    if a <= gc:
        print n / a
        return

    gb = (n - b) / gc + 1

    if n < b:
        gb = 0

    gb = gb + (n - gb * b + gb * c) / a
    print gb

f(n, a, b, c)
