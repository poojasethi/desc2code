#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
#define mp make_pair
#define f(x) ((x)&(-(x)))
const int MAXN=1e6+10;
int ans[MAXN],fen[MAXN];
vector <int> v[MAXN],q[MAXN];
vector <pair <int,int> > prev[MAXN];

void add(int x)
{
  while(x<MAXN)
    {
      fen[x]++;
      x+=f(x);
    }
}

int calc(int x)
{
  int ret=0;
  while(x)
    {
      ret+=fen[x];
      x-=f(x);
    }
  return ret;
}

int fin(int l,int r)
{
  return calc(r)-calc(l-1);
}

int main()
{
  ios::sync_with_stdio(false);
  int n,m;
  cin>>n>>m;
  for(int i=1;i<=n;i++)
    {
      int l,r;
      cin>>l>>r;
      v[r].push_back(l);
    }
  for(int i=1;i<=m;i++)
    {
      int p;
      cin>>p;
      q[i].push_back(0);
      for(int j=1;j<=p;j++)
	{
	  int x;
	  cin>>x;
	  prev[x].push_back(mp(q[i].back(),i));
	  q[i].push_back(x);
	}
      prev[MAXN-1].push_back(mp(q[i].back(),i));
    }
  for(int i=1;i<MAXN;i++)
    {
      for(int j=0;j<v[i-1].size();j++)
	add(v[i-1][j]);
      for(int j=0;j<prev[i].size();j++)
	ans[prev[i][j].second]+=fin(prev[i][j].first+1,i-1);
    }
  for(int i=1;i<=m;i++)
    cout<<n-ans[i]<<endl;
  return 0;
}
