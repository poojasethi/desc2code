#include <cstdio>
#include <cstring>
#include <algorithm>
#define  maxn	300005
#define	 inf	1000050
using namespace std;


struct Node
{
	int l,r,id;
}a[maxn*3],b[maxn*3];
int tree[inf+10086],ans[maxn];
int n,m,t,x,tb;


inline int cmp(const Node &e,const Node &f)
{
	return e.r<f.r;
}


int main()
{
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		scanf("%d%d",&a[i].l,&a[i].r);
	for(int i=1;i<=m;i++)
	{
		scanf("%d",&t);
		int last=0;
		for(int j=1;j<=t;j++)
		{
			scanf("%d",&x);
			b[++tb].l=last;
			b[tb].r=x;
			b[tb].id=i;
			last=x;
		}
		b[++tb].l=last;
		b[tb].r=inf;
		b[tb].id=i;	
	}
	sort(a+1,a+n+1,cmp);
	sort(b+1,b+tb+1,cmp);
	for(int i=1;i<=m;i++)ans[i]=n;
	for(int i=1,top=1;i<=tb;i++)
	{
		while(top<=n&&a[top].r<b[i].r)
		{
			for(int p=a[top].l;p<=inf;p+=p&(-p))tree[p]++;
			top++;
		}
		int ret=0;
		for(int p=b[i].r-1;p;p-=p&(-p))ret+=tree[p];
		for(int p=b[i].l;p;p-=p&(-p))ret-=tree[p];
		ans[b[i].id]-=ret;
	}
	for(int i=1;i<=m;i++)printf("%d\n",ans[i]);
	return 0;
}
