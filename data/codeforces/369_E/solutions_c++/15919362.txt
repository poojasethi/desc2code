#include <bits/stdc++.h>
using namespace std;
typedef long long lol;
const int MAXN=1e6+10;
int fen[MAXN],ans[MAXN];
vector <pair <int,int> > Prev[MAXN];
vector <int> v[MAXN];

void add(int x)
{
  for(;x<MAXN;x+=x&-x)
    fen[x]++;
}

int fin(int x)
{
  int res=0;
  for(;x;x-=x&-x)
    res+=fen[x];
  return res;
}

int main()
{
  ios::sync_with_stdio(false);
  int n,m;
  cin>>n>>m;
  for(int i=0;i<n;i++)
    {
      int l,r;
      cin>>l>>r;
      r++;
      v[r].push_back(l);
    }
  for(int i=0;i<m;i++)
    {
      int k;
      cin>>k;
      int p=0;
      for(int j=0;j<k;j++)
	{
	  int x;
	  cin>>x;
	  Prev[x].push_back({p,i});
	  p=x;
	}
      Prev[MAXN-5].push_back({p,i});
    }
  for(int i=0;i<MAXN;i++)
    {
      for(int j=0;j<v[i].size();j++)
	add(v[i][j]);
      for(int j=0;j<Prev[i].size();j++)
	if(i-1>=Prev[i][j].first+1)
	  ans[Prev[i][j].second]+=fin(i-1)-fin(Prev[i][j].first);
    }
  for(int i=0;i<m;i++)
    cout<<n-ans[i]<<endl;
  return 0;
}
