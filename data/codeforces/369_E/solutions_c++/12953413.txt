//In the Name of God

#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#define l first
#define r second

using namespace std;

const int nn=1048576;
int n,q,p[300100];
vector<int> flag[nn*2];
pair<int,int> s[300100];

inline int bs(int seg,int x)
{
  return flag[seg].end()-lower_bound(flag[seg].begin(),flag[seg].end(),x);
}

inline int cnt(int seg,int fs,int ls,int f,int l,int x)
{
  if(f>l)
    return 0;
  if(fs==f && ls==l)
    return bs(seg,x);
  int mid=(fs+ls)/2;
  if(l<=mid)
    return cnt(seg*2,fs,mid,f,l,x);
  if(mid<f)
    return cnt(seg*2+1,mid+1,ls,f,l,x);
  return cnt(seg*2,fs,mid,f,mid,x)+cnt(seg*2+1,mid+1,ls,mid+1,l,x);
}

int main()
{
  scanf("%d%d",&n,&q);
  for(int l,r,i=0;i<n;i++)
    scanf("%d%d",&s[i].l,&s[i].r);
  sort(s,s+n);
  for(int i=0;i<n;i++)
    for(int id=nn+s[i].r-1;id;flag[id].push_back(s[i].l),id/=2);
  for(int ans=0,k;q--;ans=0)
    {
      scanf("%d",&k);
      for(int i=0;i<k;i++)
        scanf("%d",p+i);
      p[k]=1000100;
      ans+=cnt(1,1,nn,1,p[0]-1,1);
      for(int i=1;i<=k;i++)
	ans+=cnt(1,1,nn,p[i-1]+1,p[i]-1,p[i-1]+1);
      printf("%d\n",n-ans);
    }
  return 0;
}
