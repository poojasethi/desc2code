#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
class sp{
  public:
  int q,w;
  sp *e;
  sp(int q1,int w1,sp *e1){
   q=q1;w=w1;e=e1;}
};
int a[2097152];
int l(int q){
  int w=1,r=0,t;
  for(t=524288;t;t>>=1){
   w<<=1;
   if(q>=t){
    q-=t;
    r+=a[w];
    w++;}}
  return r;
}
void l(int w,int e){
  for(w+=1048575;w;w>>=1)
   a[w]+=e;
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,v,f[300000];
  sp **s=new sp*[1000001],**d=new sp*[1000001];
  for(w=0;w<1000001;w++)
   s[w]=d[w]=0;
  for(w=0;w<2097152;w++)
   a[w]=0;
  cin>>q>>w;
  for(e=0;e<q;e++){
   scanf("%d%d",&r,&t);
   l(r,1);
   s[t]=new sp(r,0,s[t]);}
  for(e=0;e<w;e++){
   f[e]=v=0;
   for(scanf("%d",&r);r;r--){
    scanf("%d",&t);
    if(v+1<=t-1)
     d[t-1]=new sp(v+1,e,d[t-1]);
    v=t;}
   if(v+1<=1000000)
    d[1000000]=new sp(v+1,e,d[1000000]);}
  sp *c;
  for(e=1000000;e>0;e--){
   for(c=d[e];c;c=c->e) 
    f[c->w]+=a[1]-l(c->q-1);
   for(c=s[e];c;c=c->e)
    l(c->q,-1);}
  for(e=0;e<w;e++)
   printf("%d\n",q-f[e]);
  return 0;}
