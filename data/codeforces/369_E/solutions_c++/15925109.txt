//In the Name of God

#include<bits/stdc++.h>

using namespace std;

const int maxn=1<<20;

int n,q;
vector<int>ver[maxn];
int p[maxn];
vector<int>head[maxn],id[maxn];
int s[maxn];
int ans[maxn];

void add(int i,int val=1)
{
  for(;i<maxn;i+=i&-i)
    s[i]+=val;
}

int get(int i)
{
  int ret=0;
  for(;i;i-=i&-i)
    ret+=s[i];
  return ret;
}

main()
{
  ios::sync_with_stdio(0);
  cin>>n>>q;
  for(int l,r,i=0;i<n;i++)
    {
      cin>>l>>r;
      ver[l].push_back(r);
    }
  for(int cnt,i=0;i<q;i++)
    {
      cin>>cnt;
      cnt++;
      p[0]=0;
      for(int j=1;j<cnt;j++)
	cin>>p[j];
      p[cnt]=maxn-1;
      for(int j=0;j<cnt;j++)
	if(p[j]!=p[j+1]-1)
	  {
	    head[p[j]+1].push_back(p[j+1]);
	    id[p[j]+1].push_back(i);
	  }
      ans[i]=n;
    }
  for(int l=maxn;l--;)
    {
      for(int i=0;i<ver[l].size();i++)
	add(ver[l][i]+1);
      for(int i=0;i<head[l].size();i++)
	ans[id[l][i]]-=get(head[l][i]);
    }
  for(int i=0;i<q;i++)
    cout<<ans[i]<<endl;
  return 0;
}
