#include<iostream>
#include<cmath>
#include<cstring>
#include<algorithm>
#include<cstdio>
#include<string>
#include<sstream>
using namespace std;
int x[10],y[10];
int k;
int dis(int x0,int y0)
{
    int ans=1000000000;
    for (int i=0;i<k;i++)
        ans=min(abs(x0-x[i])+abs(y0-y[i]),ans);
    return ans;
}
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,m; cin>>n>>m;
    cin>>k;
    for (int i=0;i<k;i++)
        cin>>x[i]>>y[i];
    int ans=-1, idi,idj;
    for (int i=1;i<=n;i++)
    for (int j=1;j<=m;j++)
    if (dis(i,j)>ans)
    {
        ans=dis(i,j);
        idi=i; idj=j;
    }
    cout<<idi<<" "<<idj<<endl;
    return 0;
}
