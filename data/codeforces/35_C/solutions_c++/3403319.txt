#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int a[2009][2009];
int x[100],y[100];
int bs(int x)
{
	if(x>=0) return x;
	return -x;
}
int main()
{
    int n,m,i,j,l,k;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	while(scanf("%d%d",&n,&m)!=-1)
	{
		memset(a,0,sizeof(a));
		scanf("%d",&k);
		for(i=0;i<k;i++)
			scanf("%d%d",&x[i],&y[i]);
		int xx=1,yy=1;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=m;j++)
			{
				int min=100000000;
				for(l=0;l<k;l++)
				{
                     if(bs(i-x[l])+bs(j-y[l])<min)
						 min=bs(i-x[l])+bs(j-y[l]);
				}
				a[i][j]=min;
				if(a[i][j]>a[xx][yy]) xx=i,yy=j;
			}
		}
		printf("%d %d\n",xx,yy);
	}
	return 0;
}
