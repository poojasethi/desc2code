#include <iostream>
#include <fstream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
typedef long long ll;
#define REP(i,n) for(int i=0;i<(int)(n);i++)

int main()
{
  ifstream ifs("input.txt");
  ofstream ofs("output.txt");
  int n,m,k,x[10],y[10];
//  while(scanf("%d%d%d",&n,&m,&k)+1){
  ifs >> n >> m >> k;
    REP(i,k){
      ifs>>x[i]>>y[i];
      x[i]--; y[i]--;
    }

    int lst = -1,my,mx;
    REP(i,m) REP(j,n){
      int d = 1 << 28;
      REP(l,k) d = min(d,abs(x[l]-j)+abs(y[l]-i));
      if (lst < d){
        lst = d; my = i; mx = j;
      }
    }

    ofs << mx + 1 << " " << my + 1 << endl;
//  }
  return 0;
}
