#include <cstdio>
#include <queue>
using namespace std;
int M,N,K,x,y,u,v,D[2][4]={-1,0,1,0,0,-1,0,1},U[2020][2020];
struct node {int x,y;};
int main() {
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d%d%d",&N,&M,&K);
    queue<node> q;
    while (K--) {scanf("%d%d",&x,&y);q.push((node){x,y});U[x][y]=1;};
    while (!q.empty()) {
        x=q.front().x,y=q.front().y;q.pop();
        for (K=0;K<4;K++) {
            u=x+D[1][K],v=y+D[0][K];
            if (u>0&&u<=N&&v>0&&v<=M&&!U[u][v]) U[u][v]=1,q.push((node){u,v});
        }
    }
    printf("%d %d\n",x,y);
    return 0;
}
