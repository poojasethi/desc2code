#include <bits/stdc++.h>

using namespace std;
ifstream in("input.txt");
ofstream out("output.txt");
#define PB push_back
#define MP make_pair
int N, M, K, x, y, mark[2001][2001];
int XS[] = { 0, 0, 1, -1 }, YS[] = { 1, -1, 0, 0 };
pair<int, int> p;
deque<pair<int, int> > q;

int main() {
	memset(mark, 0, sizeof(mark));
	in >> N >> M >> K;
	for (int i = 0; i < K; i++)
		in >> x >> y, mark[x][y] = 1, q.PB(MP(x, y));
	while (!q.empty()) {
		p = q.front(), x = p.first, y = p.second, q.pop_front();
		for (int i = 0; i < 4; ++i) {
			int u = x + XS[i], v = y + YS[i];
			if (u > 0 && u <= N && v > 0 && v <= M && !mark[u][v])
				p = MP(u, v), q.PB(p), mark[u][v] = 1;
		}
	}
	out << p.first << " " << p.second;
	return 0;
}
