#include <iostream>
#include <cstdio>
using namespace std;
int a[2001][2001], x[4000010], y[4000010], p, q, n, m, k;

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n>>m>>k;
	while (k--) {
		cin>>x[q]>>y[q]; a[x[q]][y[q]]=1; q++;
	}
	while (q<m*n) {
		if (x[p]-1>0 && a[x[p]-1][y[p]]==0) {
			x[q]=x[p]-1; y[q]=y[p]; a[x[q]][y[q]]=1; q++;
		}
		if (x[p]+1<=n && a[x[p]+1][y[p]]==0) {
			x[q]=x[p]+1; y[q]=y[p]; a[x[q]][y[q]]=1; q++;
		}
		if (y[p]-1>0 && a[x[p]][y[p]-1]==0) {
			x[q]=x[p]; y[q]=y[p]-1; a[x[q]][y[q]]=1; q++;
		}
		if (y[p]+1<=m && a[x[p]][y[p]+1]==0) {
			x[q]=x[p]; y[q]=y[p]+1; a[x[q]][y[q]]=1; q++;
		}
		p++;
	}
	cout<<x[m*n-1]<<" "<<y[m*n-1]<<endl;
	return 0;
}
