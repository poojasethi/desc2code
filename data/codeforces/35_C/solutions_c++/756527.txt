#include<iostream>
#include<cstdlib>
#include<fstream>
using namespace std;
const int maxn=3000;
int arr[maxn][maxn],dp[10][2];
int main(){
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int n,m,k;
	cin>>n>>m>>k;
	for(int i=0;i<k;i++){
		int x,y;
		cin>>x>>y;
		dp[i][0]=x;
		dp[i][1]=y;
	}
	int ans=0,ansx=1,ansy=1;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			int r=10000000;
			for(int p=0;p<k;p++){
				int d=abs(i-dp[p][0])+abs(j-dp[p][1]);
				if(d<r){
					r=d;
				}
			}
			if(r>ans){
				ans=r;
				ansx=i;
				ansy=j;
			}
		}
	}
	cout<<ansx<<' '<<ansy<<endl;
}
