#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int s[100][5];
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,m,K;
    scanf("%d%d%d",&n,&m,&K);
    for(int i=0;i<K;i++)
        scanf("%d%d",&s[i][0],&s[i][1]);
    int MAX=-2100000000;
    int a,b;
    for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    {
        int MIN=2100000000;
        for(int k=0;k<K;k++)
            MIN=min(MIN,abs(s[k][0]-i)+abs(j-s[k][1]));
        if(MIN>MAX)
            MAX=MIN,a=i,b=j;
    }
    printf("%d %d",a,b);
}
