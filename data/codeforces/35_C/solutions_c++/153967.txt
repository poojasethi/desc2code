#include<cstdio>
#include<cstring>
int n,m,a1,a2,k,z[10][2],mm=123456789;;
int abs(int x){
	if(x<0)
		return -x;
	return x;
}
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<k;++i)
		scanf("%d%d",&z[i][0],&z[i][1]);
	a1=a2=1;
	for(int i=0;i<k;++i)
		if(abs(z[i][0]-a1)+abs(z[i][1]-a2)<mm)
			mm=abs(z[i][0]-a1)+abs(z[i][1]-a2);
	for(int i=1;i<=n;++i)
		for(int j=1;j<=m;++j){
			int tmp=123456789;
			for(int t=0;t<k;++t)
				if(abs(z[t][0]-i)+abs(z[t][1]-j)<tmp)
					tmp=abs(z[t][0]-i)+abs(z[t][1]-j);
				if(tmp>mm){
					mm=tmp;
					a1=i;
					a2=j;
				}
		}
	printf("%d %d\n",a1,a2);
	return 0;
}
