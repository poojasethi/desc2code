#include<fstream>
//#include<iostream>
#include<queue>
using namespace std;
ifstream cin("input.txt");
ofstream cout("output.txt");
int abs(int x)
{
	return x<0? -x:x;
}
struct node
{
	int x,y;
};
node a[11];
int main()
{
	int n,m,k;
	while(cin>>n>>m>>k)
	{
		int p,q,ans=-1;
		for(int i=0;i<k;i++)
		{
			cin>>a[i].x>>a[i].y;
		}
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=m;j++)
			{
				int temp=9999999;
				for(int x=0;x<k;x++)
				{
					temp=min(temp,abs(i-a[x].x)+abs(j-a[x].y));
				}
				if(temp>ans)
				{
					ans=temp;
					p=i;
					q=j;
				}
			}
		}
		cout<<p<<' '<<q<<endl;
	}
	return 0;
}
