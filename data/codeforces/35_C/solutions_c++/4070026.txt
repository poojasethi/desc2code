#include<cmath>
#include<fstream>
using namespace std;

#define min(a,b) ((a<b)?(a):(b))

ifstream cin("input.txt");
ofstream cout("output.txt");

int n,m;
int Max;
int posx,posy;
int k;
int x[15],y[15];


int main(){
    posx=1;
    posy=1;
   ios_base::sync_with_stdio(false);
   cin>>n>>m>>k;
   for(int i=1;i<=k;i++)
      cin>>x[i]>>y[i];

   for(int i=1;i<=n;i++){
      for(int j=1;j<=m;j++){
         int dis=-1u/2;
         for(int l=1;l<=k;l++){
            dis=min(dis,abs(i-x[l])+abs(j-y[l]));
         }
         if(Max<dis){
            Max=dis;
            posx=i;
            posy=j;
         }
      }
   }
   cout<<posx<<' '<<posy<<endl;
   return 0;
}
