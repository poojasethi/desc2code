#include <iostream>
#include <stdio.h>

using namespace std;

int abs(int a1)
{
    if(a1>=0){return a1;} else {return -a1;}
}

int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int x[20],y[20];
    int n,m,i,j,k,p,a;
    cin >> n >> m;
    cin >> k;
    for(i=0;i<k;i++)
    {
        cin >> x[i] >> y[i];
        x[i]--;
        y[i]--;
    }
    int maxa=-1,maxx,maxy;
    for(i=0;i<n;i++)
    {
        for(j=0;j<m;j++)
        {
            a=4500;
            for(p=0;p<k;p++)
            {
                a=min(a,abs(i-x[p])+abs(j-y[p]));
            }
            if(a>maxa){maxa=a;maxx=i;maxy=j;}
        }
    }
    cout << maxx+1 << " " << maxy+1;
    return 0;
}
