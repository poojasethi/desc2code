#include <cstdio>
FILE *in,*out;
inline int max(int x,int y)
{
	return x<y?x:y;
}
inline int abs(int x)
{
	return x<0?-x:x;
}
int x[30],y[30];
const int inf=-(1<<27);
int main()
{
	int i,j,t,k,w,z,maxv,maxv2,n,m;
	in=fopen("input.txt","r");
	out=fopen("output.txt","w");
	fscanf(in,"%d%d",&n,&m);
	fscanf(in,"%d",&k);
	for (i=1;i<=k;i++)
	{
		fscanf(in,"%d%d",&x[i],&y[i]);
	}
	maxv=-1;
	
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=m;j++)
		{
			maxv2=-inf;
			for (t=1;t<=k;t++)
				maxv2=max(maxv2,abs(i-x[t])+abs(j-y[t]));
			if (maxv2>maxv)
			{
				maxv=maxv2;
				z=i;
				w=j;
			}
		}
	}
	fprintf(out,"%d %d\n",z,w);
	fclose(in);
	fclose(out);
	return 0;

}
