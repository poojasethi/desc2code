#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <algorithm>
#include <cmath>
using namespace std;

int x[15], y[15];

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n, m, k;
	cin >> n >> m >> k;
	for( int i = 0; i < k; ++i )
		cin >> x[i] >> y[i];
	int optx, opty, mm = -1;
	for( int i = 1; i <= n; ++i )
		for( int j = 1; j <= m; ++j ){
			int mn = 0x3f3f3f3f;
			for( int l = 0; l < k; ++l )
				mn = min( mn, abs(x[l]-i)+abs(y[l]-j) );
			if( mn > mm )
				optx = i, opty = j, mm = mn;
		}
	cout << optx << " " << opty << endl;
	return 0;
}
