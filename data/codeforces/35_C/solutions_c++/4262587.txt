#include <iostream>
#include <cstdio>
#include <cmath>
#define F first
#define S second
using namespace std;
const int inf=1000000000;
int n,m,k,i,a,b,j,o,d,x,y,mi;
pair <int,int> p[11];
int main()
{
freopen ("input.txt","r",stdin);
freopen ("output.txt","w",stdout);
cin>>n>>m>>k;
for (i=1;i<=k;i++)
{
cin>>a>>b;
p[i].F=a;
p[i].S=b;
}
for (i=1;i<=n;i++)
{
for (j=1;j<=m;j++)
{
	d=inf;
	for (o=1;o<=k;o++)
	{
	if (d>(abs(i-p[o].F)+abs(j-p[o].S)))
		d=(abs(i-p[o].F)+abs(j-p[o].S));		
	}
	if (d>=mi)
		{mi=d;
		x=i;
		y=j;
		}
	
}
}
cout<<x<<" "<<y;
return 0;
}
