#include<stdio.h>
struct NN{int a,b;};
NN nn[11];
int myabs(int a){return a>0?a:(-a);}
int dis(int i,int x,int y)
{
	return myabs(nn[i].a-x)+myabs(nn[i].b-y);
}
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,m,i,j,k,p;
	int s1,s2,ans,a1,a2;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		scanf("%d",&k);
		for(i=0;i<k;i++) scanf("%d%d",&nn[i].a,&nn[i].b);
		ans=0;a1=1;a2=1;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=m;j++)
			{
				s1=10000;
				for(p=0;p<k;p++)
				{
					s2=dis(p,i,j);
					if(s2<s1) s1=s2;
				}
				if(s1>ans)
				{
					ans=s1;a1=i;a2=j;
				}
			}
		}
		printf("%d %d\n",a1,a2);
	}
	return 0;
}
