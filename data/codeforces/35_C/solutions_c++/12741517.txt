#include <bits/stdc++.h>
using namespace std;

int n,m,k;
int mi[2005][2005];

int main () {

    freopen("input.txt", "r", stdin);freopen("output.txt", "w", stdout);
    for (int i=1; i<=2000; i++) for (int j=1; j<=2000; j++) mi[i][j] = 5000;

    cin >> n >> m >> k;

    int x,y;
    for (int t=0; t<k; t++) {
        cin >> x >> y;

        for (int i=1; i<=n; i++)
            for (int j=1; j<=m; j++) {
                mi[i][j] = min(mi[i][j], abs(x-i)+abs(y-j));
            }
    }

    int ma = -1;

    for (int i=1; i<=n; i++)
        for (int j=1; j<=m; j++)
            if (mi[i][j] > ma)
                {
                    x=i; y=j; ma=mi[i][j];
                }

    cout << x << " " << y << endl;
}
