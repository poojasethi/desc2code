#include <fstream>
int dl[4]={-1,0,1,0};
int dc[4]={0,1,0,-1};
using namespace std;
int k,x,y,n,m,a[2005][2005],p;
struct tip{int x;int y;} cd[2005*2005];
int main()
{ifstream fin("input.txt");
 ofstream fout("output.txt");
  fin>>n>>m>>k;
  int i,h,l,c;

   for(i=1;i<=k;++i)
    {fin>>cd[i].x>>cd[i].y;a[cd[i].x][cd[i].y]=1;}
    cd[0].x=k;
   for(i=1;i<=cd[0].x;++i)
    {x=cd[i].x;
     y=cd[i].y;
      for(h=0;h<=3;++h)
       {l=x+dl[h];
        c=y+dc[h];
         if(a[l][c]==0&&l>=1&&l<=n&&c<=m&&c>=1)
          {cd[0].x++;
           cd[cd[0].x].x=l;
           cd[cd[0].x].y=c;
           a[l][c]=1;
          }
       }
    }

  fout<<cd[cd[0].x].x<<" "<<cd[cd[0].x].y<<'\n';
  fin.close();
  fout.close();
    return 0;
}
