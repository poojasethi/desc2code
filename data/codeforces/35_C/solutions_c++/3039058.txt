
#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>

using namespace std;

int n, m, k;
int fire[15][2];

int main()
{
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d%d%d", &n, &m, &k);
	for(int i=0; i<k; i++)
		scanf("%d%d", &fire[i][0], &fire[i][1]);
	int ans[2]={1, 1}, Max=0;
	for(int i=1; i<=n; i++)
		for(int j=1; j<=m; j++)
		{
			int len=0x3fffffff;
			for(int p=0; p<k; p++)
			{
				len=min(len, abs(i-fire[p][0])+abs(j-fire[p][1]));
			}
			if(len>Max) Max=len, ans[0]=i, ans[1]=j;
		}
	printf("%d %d\n", ans[0], ans[1]);
	return 0;
}
