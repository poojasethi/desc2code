#include<stdio.h>
#include<string.h>

const int N=2000;	
struct node{
	int x,y;
	node(){};
	node(int a,int b){x=a-1;y=b-1;}
}ss,tt;	
node q[N*N];
int dire[4][2]={1,0,0,1,-1,0,0,-1};
bool mem[N][N];
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int i,j,n,m,k,t,x,y,hd,tl;
	scanf("%d%d",&n,&m);
	scanf("%d",&k);
	for(hd=tl=i=0;i<k;i++){
		scanf("%d%d",&x,&y);
		q[tl++]=node(x,y);
		mem[x-1][y-1]=1;
	}
	while(hd<tl){
		ss=q[hd++];
//		printf("--%d %d\n",ss.x,ss.y);
		for(i=0;i<4;i++){
			tt.x=ss.x+dire[i][0];
			tt.y=ss.y+dire[i][1];
			if(tt.x<0||tt.x>=n||tt.y<0||tt.y>=m||mem[tt.x][tt.y]) continue;
			q[tl++]=tt;
			mem[tt.x][tt.y]=1;
		}
	}
	printf("%d %d\n",ss.x+1,ss.y+1);
	return 0;
}
