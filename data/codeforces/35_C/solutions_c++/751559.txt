#include <stdio.h>

int Q[8000005];
char s[2005][2005];
int qb=0,qe=0;
void push(int x, int y) {
	Q[qe++]=x; Q[qe++]=y;
	s[x][y] = 1;
}
void pull(int &x, int &y) {
	x=Q[qb++]; y=Q[qb++];
}
int n, m;
int valid(int x, int y) {
	return x>0&&x<=n&&y>0&&y<=m;
}
int main(void) {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d%d", &n, &m);
	int k;
	scanf("%d", &k);
	for(int i=0;i<k;i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		push(x, y);
	}
	const int dx[4]={0,1,0,-1}, dy[4]={1,0,-1,0};
	while(qb<qe) {
		int x,y;
		pull(x, y);
		for(int i=0;i<4;i++)
			if(valid(x+dx[i], y+dy[i]) && !s[x+dx[i]][y+dy[i]])
				push(x+dx[i], y+dy[i]);
	}
	printf("%d %d\n", Q[qe-2], Q[qe-1]);
	return 0;
}
