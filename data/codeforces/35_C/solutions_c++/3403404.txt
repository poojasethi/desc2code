#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<math.h>
#include<queue>
#include<stack>
#include<iostream>
#include<string>
#include<string.h>
#include<stdlib.h>
using namespace std;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,m;
	while(scanf("%d%d",&n,&m)!=-1)
	{
		int i,j,p,h,x,y,t=-1,k,map[11][2];
		scanf("%d",&k);
		for(i=0;i<k;i++)
			scanf("%d%d",&map[i][0],&map[i][1]);
		for(i=1;i<=n;i++)
			for(j=1;j<=m;j++)
			{
				p=abs(i-map[0][0])+abs(j-map[0][1]);
				for(h=1;h<k;h++)
					if(abs(i-map[h][0])+abs(j-map[h][1])<p)
						p=abs(i-map[h][0])+abs(j-map[h][1]);
					if(p>t)
					{t=p;x=i;y=j;}
			}
			printf("%d %d\n",x,y);
	}
	return 0;
}
