#include <iostream>
#include <fstream>
#include <vector>

using namespace std;

#define ABS(x) ((x) > 0 ? (x) : -(x))

int main() {
  ifstream in("input.txt");
  int n, m, k;
  in >> n >> m >> k;
  int a, b;
  int result = 0;
  int res = 0;
  int resultX=1, resultY=1;
  vector<int> x(k), y(k);
  for (int i = 0; i < k; i++) {
    in >> x[i] >> y[i]; 
  }
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      res = 1<<20;
      for (int h = 0; h < k; h++) {
        res = min(res, ABS(x[h] - i) + ABS(y[h] - j));
      }
      if (res > result) {
        resultX = i;
        resultY = j;
        result = res;
      }
    }
  }
  ofstream out("output.txt");
  out << resultX << " " << resultY; 
  return 0;
}
