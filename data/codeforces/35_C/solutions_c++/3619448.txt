#include<stdio.h>
int abs(int a,int b){
	return a>b?(a-b):(b-a);
}
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int m,n,i,j,k,l,x,y,a[11][2],min,max=-1,t;
	scanf("%d%d%d",&m,&n,&l);
	for(i=0;i<l;++i)
		scanf("%d%d",&a[i][0],&a[i][1]);
	for(i=1;i<=m;++i){
		for(j=1;j<=n;++j){
			min=99999999;
			for(k=0;k<l;k++){
				t=abs(a[k][0],i)+abs(a[k][1],j);
				if(t<min) 
					min=t;
			}
			if(min>max)
				x=i,
				y=j,
				max=min;
		}
	}
	printf("%d %d\n",x,y);
	return 0;
}
