#include <cstdlib>
#include <cstring>
#include <cstdio>
#include <algorithm>

using namespace std;

int n,m,K,a[10],b[10],x,y,ans;

int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d%d",&n,&m);
    scanf("%d",&K);
    for(int i=0;i<K;i++)
        scanf("%d%d",&a[i],&b[i]);
    x=1,y=1,ans=0;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
        {
            int now=n+m;
            for(int k=0;k<K;k++)
                now=min(now,abs(i-a[k])+abs(j-b[k]));
            if(now>ans)x=i,y=j,ans=now;
        }
    printf("%d %d\n",x,y);
    return 0;
}
