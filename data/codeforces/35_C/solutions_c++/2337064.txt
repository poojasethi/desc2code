#include<algorithm>
#include<iostream>
#include<cstdio>

using namespace std;

const int Inf = 1000000000;

int n , m , k;
int x[11] , y[11];

int main() {
	freopen( "input.txt" , "r" , stdin );
	freopen( "output.txt" , "w" , stdout );
	
	scanf( "%d%d%d" , &n , &m , &k );
	for ( int i = 1 ; i <= k ; i++ ) scanf( "%d%d" , &x[i] , &y[i] );
	int r = 0 , c = 0 , maxs = -1;
	for ( int i = 1 ; i <= n ; i++ )
		for ( int j = 1 ; j <= m ; j++ ) {
			int cnt = Inf;
			for ( int l = 1 ; l <= k ; l++ ) cnt = min( cnt , abs( i - x[l] ) + abs( j - y[l] ) );
			if ( cnt > maxs ) {
				maxs = cnt;
				r = i , c = j;
			}
		}
	printf( "%d %d\n" , r , c );
	return 0;
}
