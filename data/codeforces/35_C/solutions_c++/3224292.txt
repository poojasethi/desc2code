#include <stdio.h>
int ab(int x)
{
    if(x<0)
    return -x;
    else
    return x;
}
int main ()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int x[10],y[10];
    int m,n;
    int k;
    scanf ("%d%d",&m,&n);
    scanf ("%d",&k);
    for(int i=0;i<k;i++)
    {
        scanf ("%d%d",&x[i],&y[i]);
    }
    int fx,fy,max=0;
    for(int i=1;i<=m;i++)
    for(int j=1;j<=n;j++)
    {
        int min=999999;
        for(int l=0;l<k;l++)
        if(ab(i-x[l])+ab(j-y[l])<min)
        {
            min=ab(i-x[l])+ab(j-y[l]);
            //printf("%d ",min);
        }
        if(min>max||max==0)
        {
            max=min;
            fx=i;fy=j;
        }
    }
    printf("%d %d",fx,fy);
}
