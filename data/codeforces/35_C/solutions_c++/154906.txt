#include <cstdlib>
#include <iostream>
#include <fstream>

using namespace std;

int main(int argc, char *argv[])
{
    ifstream in("input.txt");
    ofstream out("output.txt");
    int x[20], y[20], k, n, m, c=0;
    in>>n>>m;
    in>>k;
    for(int i=1;i<=k;i++){
        in>>x[i]>>y[i];
    }
    int max=0;
    int xg=1;
    int yg=1;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            int min=5000;
            for(int h=1;h<=k;h++){
                if(abs(x[h]-i)+abs(y[h]-j)<min) min=abs(x[h]-i)+abs(y[h]-j);
            }
            if(max<min){
                max=min;
                xg=i;
                yg=j;
            }
        }
    }
    out<<xg<<" "<<yg;
}
