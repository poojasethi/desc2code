#include<iostream>
#include<algorithm>
#include<queue>
#include<map>
#include<fstream>
using namespace std;
ofstream ofs("output.txt");
ifstream ifs("input.txt");
#define cin ifs
#define cout ofs
#define f first
#define s second
typedef pair<int,int> P;

int gx[]={-1,0,0,1},gy[]={0,1,-1,0};
int k,n,m,tree[2012][2012];

int main(){
	cin>>n>>m>>k;
	for(int i=1;i<=m;i++)
		for(int j=1;j<=n;j++)tree[i][j]=1;
	queue<pair<int,int> > Q;
	for(int i=0;i<k;i++){
		int y,x;
		cin>>x>>y;
		tree[y][x]=0;
		Q.push(P(y,x));
	}
	P q;
	while(!Q.empty()){
		q=Q.front(); Q.pop();
		for(int i=0;i<4;i++){
			int yy=q.f+gy[i],xx=q.s+gx[i];
			if(tree[yy][xx]){
				Q.push(P(yy,xx));
				tree[yy][xx]=0;
			}
		}
	}
	cout<<q.s<<" "<<q.f<<endl;
}
