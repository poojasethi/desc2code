#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int qx[4000100],qy[4000100],n,m,f,r;
bool b[2010][2010];
const int dx[4]={0,1,0,-1};
const int dy[4]={1,0,-1,0};
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&m);
	int k;
	cin>>k;
	for (int i=1;i<=k;++i)
	{
		int x,y;
		cin>>x>>y;
		qx[++r]=x;
		qy[r]=y;b[x][y]=true;
	}
	while (f<r)
	{
		int x=qx[++f];
		int y=qy[f];
		for (int i=0;i<4;++i)
		{
			int xx=x+dx[i],yy=y+dy[i];
			if (xx>0&&xx<=n&&yy>0&&yy<=m&&b[xx][yy]==false)
			{
				qx[++r]=xx;
				qy[r]=yy;
				b[xx][yy]=true;
			}
		}
	}
	cout<<qx[r]<<" "<<qy[r]<<endl;
	return 0;
}
