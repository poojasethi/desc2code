#include <iostream>
#include <stdio.h>
#include <cstdlib>
#include <math.h>
using namespace std; 

int d[2001][2001], n, m, k, i, j, x, y;

int main()
{
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    cin >> n >> m >> k;
    for (i = 1; i <= n; i++)
        for (j = 1; j <= m; j++)
            d[i][j] = (1<<30);
    while (k--)
    {
        cin >> x >> y;
        for (i = 1; i <= n; i++)
                for (j = 1; j <= m; j++)
                    d[i][j] = min(d[i][j], abs(x-i)+abs(y-j)+1);
    }
    x = y = 0;
    for (i = 1; i <= n; i++)
        for (j = 1; j <= m; j++)
            if (d[i][j] > d[x][y]) x = i, y = j;
    cout << x << " " << y;
    return 0;
}
