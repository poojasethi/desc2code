#include<iostream>
#include<cstdlib>
#include<fstream>

using namespace std;

int main()
{
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int a, b;
	cin>> a >> b;
	int k;
	cin>>k;
	int arrey[k][2];
	for(int i=0 ; i<k ; i++)
	{
		for(int j=0 ; j<2 ; j++)
		{
			cin>>arrey[i][j];
		}
	}
	int result[2];
	int q=40000;
	int res=-1;
	for(int i=1 ; i<=a ; i++)
	{
		for(int j=1 ; j<=b ; j++)
		{
			for(int p=0 ; p<k ; p++)
			{
				int len=abs(i - arrey[p][0]) + abs(j - arrey[p][1]);
				if(len<q)
					q=len;
			}
			if(q>res)
			{
				res=q;
				result[0]=i;
				result[1]=j;
			}
			q=40000;
		}
	}
	cout<< result[0] << " " << result[1] << endl;
}
