#include <bits/stdc++.h>

using namespace std;
#define FI "input.txt"
#define FO "output.txt"
const int NN = 2005;

int DIST(int x1, int y1, int x2, int y2){
    return (abs(x1-x2) + abs(y1-y2));
}
int x[11], y[11];

int main() {
    ifstream cin (FI);
    ofstream cout(FO);
    int N, M, K, xx = 1, yx = 1, dmax = -1;
    cin >> N >> M >> K;
    for(int i = 1; i <= K; i++) cin >> x[i] >> y[i];
    for(int i = 1; i <= N; i++){
        for(int j = 1; j <= M; j++){
            int dmin = 1 << 30;
            for (int k = 1; k <= K; k++) dmin = min(dmin, DIST(i, j, x[k], y[k])) ;
            if(dmax < dmin) dmax = dmin, xx = i, yx = j;
        }
    }
    cout << xx << " " << yx << endl;
    return 0;
}
