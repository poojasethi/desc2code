#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
#include <fstream>
#define N 10
using namespace std;
int n, m, k, x[N], y[N], L, R, Rx=1, Ry=1;
int main(){
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    scanf("%d%d%d", &n, &m, &k);
    for (int i = 0; i < k; i++)
        scanf("%d%d", &x[i], &y[i]);

    for (int i = 1; i <= n;i++)
    for (int j = 1; j <= m; j++){
        L = 1000000;
        for (int f = 0; f < k; f++)
            L = min(L, abs(i - x[f]) + abs(j - y[f]));
        if (L>R){
            R = L;
            Rx = i ;
            Ry = j ;
        }
    }
    printf("%d %d", Rx, Ry);
}
