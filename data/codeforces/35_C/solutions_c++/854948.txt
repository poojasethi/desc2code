#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int x[11],y[11];
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    int i,j,w;
    for(i=0;i<k;i++)
    {
        scanf("%d%d",x+i,y+i);
    }
    int xx,yy,mx=-1;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
        {
            int mi=1<<30;
            for(w=0;w<k;w++)
            {
                int t=abs(x[w]-i)+abs(y[w]-j);
                mi=min(mi,t);
            }
            if(mx<mi)
            {
                mx=mi;
                xx=i;
                yy=j;
            }
        }
    }
    printf("%d %d\n",xx,yy);
    return 0;
}
