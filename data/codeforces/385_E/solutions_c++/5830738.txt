#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
#define rep(i,s,t) for (int i=s;i<=t;++i)
#define mec(a,b) memcpy(a,b,sizeof a)
typedef long long LL;
LL n,t;
LL g[7],f[7],c[7][7],b[7][7],a[7][7];
void mul(LL c[][7],LL a[][7],LL b[][7]){
    rep(i,0,5)rep(j,0,5){
        c[i][j]=0; rep(k,0,5)(c[i][j]+=a[i][k]*b[k][j])%=n;
    }
}
int main(){
    cin>>n>>f[0]>>f[1]>>f[2]>>f[3]>>t;f[4]=0;f[5]=1;f[0]--;f[1]--;
    a[0][0]=2; a[0][1]=a[0][2]=a[0][4]=1;a[0][5]=2;
    a[1][1]=2;a[1][0]=a[1][3]=a[1][4]=1;a[1][5]=2;
    a[2][0]=a[2][1]=a[2][2]=a[2][4]=1;a[2][5]=2;
    a[3][0]=a[3][1]=a[3][3]=a[3][4]=1;a[3][5]=2;
    a[4][4]=a[4][5]=1;
    a[5][5]=1;
    rep(i,0,5)c[i][i]=1;
    rep(i,0,62){
        if ((t>>i)&1)mul(b,c,a),mec(c,b);
        mul(b,a,a);mec(a,b);
    }
    rep(i,0,5)rep(k,0,5)(g[i]+=c[i][k]*f[k])%=n;
    g[0]+=n*(g[0]<0);
    g[1]+=n*(g[1]<0);
    printf("%I64d %I64d\n",g[0]+1,g[1]+1);
}
