 f(t/2);

 for (LL J=0;J<6;J++) { nw1[i][J]=0;
 for (LL j=0;j<6;j++)
 nw1[i][J]=(nw1[i][J]+c[i][j]*c[j][J])%n; }

 {
  for (LL i=0;i<6;i++)
  for (LL J=0;J<6;J++) { c[i][J]=0;
  for (LL j=0;j<6;j++)
  c[i][J]=(c[i][J]+b[i][j]*nw1[j][J])%n; }
 } else {
for (LL i=0;i<6;i++)
for (LL J=0;J<6;J++) c[i][J]=nw1[i][J]; }
}
