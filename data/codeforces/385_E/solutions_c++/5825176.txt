#include <stdio.h>
#include <iostream>
#include <cstring>
#define ll long long
using namespace std;
ll n,t,w[7],f[7],c[7][7],b[7][7],a[7][7];
void mul(ll c[][7],ll a[][7],ll b[][7]){
	for (int i=0;i<=5;i++)
		for (int j=0;j<=5;j++){
			c[i][j]=0;
			for (int k=0;k<=5;k++)
				c[i][j]=(c[i][j]+a[i][k]*b[k][j])%n;
		}
}
int main(){
	cin>>n>>f[0]>>f[1]>>f[2]>>f[3]>>t;
	f[0]--;f[1]--;f[4]=0;f[5]=1;
	a[0][0]=2;a[0][1]=1;a[5][5]=1;
	a[0][2]=1;a[0][4]=1;a[0][5]=2;a[1][1]=2;
	a[1][0]=1;a[1][3]=1;a[1][4]=1;a[1][5]=2;
	a[2][0]=1;a[2][1]=1;a[2][2]=1;a[2][4]=1;
	a[2][5]=2;a[3][0]=1;a[3][1]=1;a[3][3]=1;
	a[3][4]=1;a[3][5]=2;a[4][4]=1;a[4][5]=1;
	for (int i=0;i<=5;i++)
		c[i][i]=1;
	for (int i=0;i<=62;i++){
		if ((t>>i)&1){
			mul(b,c,a);
			memcpy(c,b,sizeof c);
		}	
		mul(b,a,a);
		memcpy(a,b,sizeof a);
	}
	for (int i=0;i<=5;i++)
		for (int k=0;k<=5;k++)
			w[i]=(w[i]+c[i][k]*f[k])%n;
	w[0]+=n*(w[0]<0);
	w[1]+=n*(w[1]<0);
	cout<<w[0]+1<<' '<<w[1]+1<<endl;
	return 0;
}
