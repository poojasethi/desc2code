#include<iostream>
using namespace std;
#define mod 1000000007
typedef long long ll;
ll n,x,y,dx,dy,t,ans[7][7],a[7][7],c[7][7];
int main()
{
    cin>>n>>x>>y>>dx>>dy>>t;
    x--,y--;
    dx=(mod*n+dx)%n,dy=(mod*n+dy)%n;
    for(ll i=1;i<=5;i++)
        ans[i][i]=1;
    a[1][1]=a[2][6]=a[3][6]=a[3][5]=a[2][4]=a[2][3]=a[3][2]=1;
    a[2][1]=a[3][1]=2;
    a[4][1]=a[5][1]=a[2][2]=a[3][3]=2;
    a[4][2]=a[4][3]=a[4][4]=a[5][2]=a[5][3]=a[5][5]=a[4][6]=a[5][6]=a[6][1]=a[6][6]=1;
    while(t)
    {
        if(t&1)
        {
            for(ll i=1;i<=6;i++)
                for(ll j=1;j<=6;j++)
                {
                    c[i][j]=0;
                    for(ll k=1;k<=6;k++)
                        c[i][j]+=ans[i][k]*a[k][j],c[i][j]%=n;
                }
            for(ll i=1;i<=6;i++)
                for(ll j=1;j<=6;j++)
                ans[i][j]=c[i][j];
        }
        for(ll i=1;i<=6;i++)
            for(ll j=1;j<=6;j++)
            {
                c[i][j]=0;
                for(ll k=1;k<=6;k++)
                    c[i][j]+=a[i][k]*a[k][j],c[i][j]%=n;
            }
        for(ll i=1;i<=6;i++)
            for(ll j=1;j<=6;j++)
            a[i][j]=c[i][j];
        t/=2;
    }
    cout<<(ans[2][1]+ans[2][2]*x+ans[2][3]*y+ans[2][4]*dx+ans[2][5]*dy)%n+1<<" "<<(ans[3][1]+ans[3][2]*x+ans[3][3]*y+ans[3][4]*dx+ans[3][5]*dy)%n+1;
}
