#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n;
int a[6],b[6][6];
int64 m;
inline void mul(int a[6],int b[6][6]) {
	int i,j; int c[6]={0};
	For(i,0,5)For(j,0,5) c[j]=((int64)a[i]*b[i][j]+c[j])%n;
	For(i,0,5) a[i]=c[i];
}
inline void multi(int a[6][6],int b[6][6]) {
	int i,j,k; int c[6][6]={0};
	For(i,0,5)For(j,0,5)For(k,0,5) c[i][k]=((int64)a[i][j]*b[j][k]+c[i][k])%n;
	For(i,0,5)For(j,0,5) a[i][j]=c[i][j];
}
int main() {
	scanf("%d",&n);
	For(i,0,3) scanf("%d",&a[i]); a[0]--,a[1]--; a[4]=a[5]=1;
	b[0][0]=2,b[0][1]=b[0][2]=b[0][3]=1;
	b[1][0]=1,b[1][1]=2,b[1][2]=b[1][3]=1;
	b[2][0]=1,b[2][2]=1;
	b[3][1]=1,b[3][3]=1;
	b[4][4]=b[4][2]=b[4][3]=1;
	b[5][0]=b[5][1]=b[5][2]=b[5][3]=2,b[5][4]=b[5][5]=1;
	scanf("%I64d",&m);
	for (;m;m>>=1) {
		if (m&1) mul(a,b);
		multi(b,b);
	}
	if (a[0]<0) a[0]+=n;
	if (a[1]<0) a[1]+=n;
	printf("%d %d\n",a[0]+1,a[1]+1);
	//for(;;);
	return 0;
}
