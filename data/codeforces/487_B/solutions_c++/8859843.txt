#include <bits/stdc++.h>
#define oo 1000000000
#define X first
#define Y second
using namespace std;

typedef pair<int,int> pii;
const int maxn = 100001;
set<pii> S,R;
int a[maxn], n, s, l;
int dp[maxn];

int main(){
    scanf("%d%d%d",&n,&s,&l);
    for(int i=1; i<=n; ++i) scanf("%d",a+i);
    for(int i=1, j=1; i<=n; ++i){
        S.insert(pii(a[i],i));
        while(S.rbegin()->X - S.begin()->X > s){
            S.erase(pii(a[j],j));
            j++;
        }
        if(i - l >= j - 1 && dp[i-l] != oo)
            R.insert(pii(dp[i-l],i - l));
        while(!R.empty() && R.begin()->Y < j - 1)
            R.erase(R.begin());
        if(R.empty())
            dp[i] = oo;
        else
            dp[i] = R.begin()->X + 1;
    }
    printf("%d",dp[n]==oo? -1:dp[n]);
}
