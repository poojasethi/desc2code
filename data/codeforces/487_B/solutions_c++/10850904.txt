#include <bits/stdc++.h>
using namespace std;

#define N 100009
#define pii pair< int, int >
#define INF 0x3f3f3f3f

int n, s, l;
int a[N], dp[N];
set< pii > S;
priority_queue< pii > Q;

int calc()
{
	if(l == 1)
		Q.push({0, 0});
	for(int i = 1, j = 0; i <= n; ++i){
		dp[i] = INF;
		
		S.insert({a[i], i});
		
		while(j < i and S.rbegin()->first - S.begin()->first > s)
			S.erase({a[j], j}), ++j;
		
		while(!Q.empty() and Q.top().second+1 < j)
			Q.pop();
		
		if(!Q.empty())
			dp[i] = 1 - Q.top().first;
		
		if(i-l+1 >= 0 and dp[i-l+1] != INF)
			Q.push({-dp[i-l+1], i-l+1});			
		
	}
	
	return dp[n]==INF ? -1 : dp[n];	
}

int main()
{
	scanf("%d%d%d", &n, &s, &l);
	for(int i = 1; i <= n; ++i) scanf("%d", a+i);
	printf("%d\n", calc());
}
