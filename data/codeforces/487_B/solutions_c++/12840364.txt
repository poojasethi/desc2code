#include <bits/stdc++.h>
using namespace std;
multiset<int> st,res;
#define FOR(i, a, b) for (int i = (a); i <= (b); i++)
int dp[100006],a[100006];
int n,s,l;
int main(){
    cin>>n>>s>>l;
    FOR(i,1,n) scanf("%d",&a[i]);
    int j=1;    
    FOR(i,1,n){
        st.insert(a[i]);
        for(;*st.rbegin()-*st.begin()>s;j++){
            st.erase(st.find(a[j]));
            if (i-j>=l) res.erase(res.find(dp[j-1]));
        }
        if (i-j+1>=l) res.insert(dp[i-l]);
        if (res.empty())dp[i]=1e9;
        else
        dp[i]=*res.begin()+1;
    }
    cout<<(dp[n]>=1e9?-1:dp[n]);
}
