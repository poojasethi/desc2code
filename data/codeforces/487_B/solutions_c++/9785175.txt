#include <iostream>
#include <set>

using namespace std;

const int MAXN = 100010;
const int INF = 1000000000;

typedef pair<int, int> pii;

set<pii> s, s1;
int a[MAXN], dp[MAXN];

int main() {
	int n, ms, l;
	cin >> n >> ms >> l;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	int j = 1;
	for (int i = 1; i <= n; i++) {
		s.insert(pii(a[i], i));
		while (!s.empty() && s.rbegin()->first - s.begin()->first > ms) {
			s.erase(pii(a[j], j));
			j++;
		}
		if (i >= l && dp[i - l] != INF)
			s1.insert(pii(dp[i - l], i - l));
		while (!s1.empty() && s1.begin()->second < j - 1)
			s1.erase(s1.begin());
		if (!s1.empty())
			dp[i] = s1.begin()->first + 1;
		else
			dp[i] = INF;
	}
	if (dp[n] == INF)
		cout << -1 << endl;
	else
		cout << dp[n] << endl;
}
