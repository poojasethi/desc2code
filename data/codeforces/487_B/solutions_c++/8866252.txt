#include <bits/stdc++.h>
using namespace std;
const int N=100005;
typedef pair<int,int> PII;
set<PII> X,Y;
int f[N],a[N];
int n,s,l;
void solve() {
    X.clear(); Y.clear();
    f[0]=0;
    int id=0;
    for(int i=1;i<=n;i++) {
        X.insert(make_pair(a[i],i));
        while(X.rbegin()->first - X.begin()->first>s) {
            Y.erase(make_pair(f[id],id));
            id++;
            X.erase(make_pair(a[id],id));
        }
        if(i>=l&&X.size()>=l) {
            if(f[i-l]>=0) {
                Y.insert(make_pair(f[i - l], i - l));
            }
        }
        if (Y.size() == 0) {
            f[i]=-1;
        } else {
            f[i]=Y.begin()->first + 1;
        }
    }
    cout<<f[n]<<endl;
}
int main() {
    while(cin>>n>>s>>l) {
        for(int i=1;i<=n;i++) {
            cin>>a[i];
        }
        solve();
    }
    return 0;
}
