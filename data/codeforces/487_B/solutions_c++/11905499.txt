#include <iostream>
#include <cstdio>
using namespace std;

int len[100005] ,n ,s ,l ,num[100005] ,qmax[100005] ,front1 ,pos1 ,qmin[100005] ,front2 ,pos2 ;
int q[100005] ,front ,pos ,f[100005] ;

int main()
{
	scanf("%d%d%d",&n,&s,&l);
	for(int i=1,j=0;i<=n;++i)
	{
		scanf("%d",&num[i]);
		while(front1<pos1&&num[qmax[pos1-1]]<num[i])--pos1;
		qmax[pos1++]=i;
		while(front2<pos2&&num[qmin[pos2-1]]>num[i])--pos2;
		qmin[pos2++]=i;
		while(num[qmax[front1]]-num[qmin[front2]]>s)
			j=qmax[front1]<qmin[front2]?qmax[front1++]:qmin[front2++];
		len[i]=j;
	}
	for(int i=1,j;i<=n;++i)
	{
		if(i>=l)
		{
			j=i-l;
			while(pos>front&&f[q[pos-1]]>f[j])--pos;
			q[pos++]=j;
		}
		
		while(pos>front&&q[front]<len[i])++front;
		if(front==pos)
			f[i]=100005;
		else f[i]=f[q[front]]+1;
		
	}
	if(f[n]>=100005)printf("-1");
	else printf("%d",f[n]);
	return 0;
}
