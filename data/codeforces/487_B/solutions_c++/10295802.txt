#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <algorithm>
#include <set>
#define mp make_pair
#define pa pair<int,int>
#define M 100000+5
#define inf 0x3f3f3f3f
using namespace std;
int f[M],n,ss,l,a[M];
multiset<int> s,dp;
int main()
{
    scanf("%d%d%d",&n,&ss,&l);
	for (int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for (int i=1,j=1;i<=n;i++)
	{
		s.insert(a[i]);
		for (;*s.rbegin()-*s.begin()>ss;j++)
		{
			s.erase(s.find(a[j]));
			if (i-j>=l)
				dp.erase(dp.find(f[j-1]));
		}
	    if (i-j+1>=l) dp.insert(f[i-l]);
		if (dp.begin()==dp.end())
			f[i]=inf;
		else f[i]=*dp.begin()+1;
	}
	if (f[n]>=inf)
		puts("-1");
	else cout<<f[n]<<endl;
	return 0;
}
