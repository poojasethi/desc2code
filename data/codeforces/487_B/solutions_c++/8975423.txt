#include<set>
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int N=100005;
multiset<int>S;
int k,a[N],n,s,l,f[N],q[N],Q[N],g[N];
int main()
{
//	freopen("1.in","r",stdin);
	cin>>n>>s>>k;
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1,l=1;i<=n;i++){
		S.insert(a[i]);
		while(*S.rbegin()-*S.begin()>s)S.erase(S.find(a[l])),l++;
		f[i]=l;
	}
	for(int i=1;i<k;i++)g[i]=1e8;
	for(int i=k,l=1,r=1;i<=n;i++){
		while(l<=r&&q[l]<f[i]-1)l++;
		if(l<=r&&i-q[l]>=k)g[i]=g[q[l]]+1;else g[i]=1e8;
		while(l<=r&&g[q[r]]>=g[i-k+1])r--;
		q[++r]=i-k+1;
	}
	if(g[n]>=1e8)cout<<-1;else cout<<g[n];
}
