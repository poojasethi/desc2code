#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    long long int A,B,x,n,a=1,b=0;
    cin >> A >> B >> n >> x;

    while (n)
    {
        if (n%2==1)
        {
           x = ((A*x)%MOD+B)%MOD;
        }
        a=(A*A)%MOD;
        b=((A*B)%MOD+B)%MOD;
        A=a;B=b;
        n/=2;

    }

    cout << x;
    return 0;
}
