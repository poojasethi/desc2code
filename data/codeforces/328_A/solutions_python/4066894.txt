#coding=utf-8


a = map(int, raw_input().split())

def isProgression(l):
  return l[1] - l[0] == l[2] - l[1] and l[3] - l[2] == l[2] - l[1]


def isGeometric(l):
  return l[0] * l[2] == l[1] * l[1] and l[1] * l[3] == l[2] * l[2]

if isProgression(a):
  print a[-1] * 2 - a[-2]
elif isGeometric(a) and a[-1] * a[-1] % a[-2] == 0:
  print a[-1] * a[-1] / a[-2]
else:
  print 42
