#!/usr/bin
t=map(float, raw_input().split());a=abs(t[0]-t[1]);g=t[1]/t[0]
arth,geo=map(sum, zip(*[[int(abs(t[u]-t[u-1]) == a), int(g!=1 and t[u-1]*g == t[u])] for u in xrange(1,4)]))
if arth==3:print [int(t[3]-a),int(t[3]+a)][t[2]<t[3]]
elif geo==3:print [int(t[3]*g),42][int(t[3]*g) < t[3]*g]
else:print 42
