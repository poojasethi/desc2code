def is_ari(x):
    a = [x[i] - x[i+1] for i in range(3)]
    return  len(set(a)) == 1

def is_geo(x):
    a = [x[i+1]*1.0/x[i] for i in range(3)]
    return len(set(a)) == 1 and a[0] not in [0, 1]

a = map(int, raw_input().split())
if (is_ari(a) and is_geo(a)) or (not is_ari(a) and not is_geo(a)):
    print 42
elif is_ari(a):
    print a[3]*2 - a[2]
elif is_geo(a):
    print int(a[3]**2*1.0/a[2]) if int(a[3]**2*1.0/a[2]) == a[3]**2*1.0/a[2] else 42
