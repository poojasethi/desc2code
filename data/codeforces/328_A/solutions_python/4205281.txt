#a solution to the codeforces problem IQ Test

inp = raw_input()
nums = map(float, inp.split(' '))

def solution(n):
	if (n == int(n)):
		print int(n)
	else:
		print 42


if (nums[1]-nums[0] == nums[2]-nums[1] == nums[3]-nums[2]):
	solution(nums[3] + nums[1] - nums[0])
elif (nums[1]/nums[0] == nums[2]/nums[1] == nums[3]/nums[2]):
	solution(nums[3] * nums[1] / nums[0])
else:
	print "42"
