import math
a=map(int, raw_input().split())
d = a[1] - a[0]
q = 1 if a[0] == 0 else a[1] / float(a[0])
p = [a[0] + x * d for x in xrange(len(a))]
g1 = [math.floor(a[0] * q ** x + 0.000001) for x in xrange(len(a))]
g2 = [math.ceil(a[0] * q ** x - 0.000001) for x in xrange(len(a))]
if a == p: print a[3] + d
elif a == g1 and a == g2: print int(a[3] * q) if math.floor(a[3] * q) == math.ceil(a[3] * q) else '42'
else: print '42'
