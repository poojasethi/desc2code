#!/usr/bin
t=map(float, raw_input().split());a=abs(t[0]-t[1]);g=t[1]/t[0];arth=geo=0
for u in xrange(1,4):
    if abs(t[u]-t[u-1]) == a:
        arth = -~arth
    if g!=1 and t[u-1]*g == t[u]:
        geo = -~geo
if arth == 3:
    print [int(t[3]-a), int(t[3]+a)][t[2]<t[3]]
elif geo == 3:
    print [int(t[3]*g),42][int(t[3]*g) < t[3]*g]
else: print 42
