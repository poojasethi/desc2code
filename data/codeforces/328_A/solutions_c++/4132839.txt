#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    double a,b,c,d;
    while(cin>>a>>b>>c>>d)
    {
        if((b-a==c-b)&&(c-b==d-c)) cout<<2*d-c<<endl;
         else if((b/a==c/b)&&(c/b==d/c)&&(d*d/c)==int(d*d/c)) cout<<d*d/c<<endl;
          else cout<<42<<endl;
    }
    return 0;
}
