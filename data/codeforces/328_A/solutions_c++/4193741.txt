#include <stdio.h>

int main()
{
    int a,b,c,d;
    scanf("%d%d%d%d",&a,&b,&c,&d);
    int f = b - a;
    if(b+f==c && c+f==d){
        printf("%d\n",d+f);
        return 0;
    }
    if(b*b==a*c && c*c==b*d){
        if(d*d%c==0){
            printf("%d\n",d*d/c);
            return 0;
        }
    }
    printf("42\n");
    return 0;
}
