#include<iostream>
#include<cmath>
using namespace std;
int main(){
    double a,b,c,d;
    cin >> a >> b >> c >> d;
    if((a+c)/2.0==b && (b+d)/2.0==c && floor(d+(b-a))==d+(b-a))cout << d+(b-a) << endl;
    else if(b*b==a*c && c*c==b*d && floor(d*(b/a)) == d*(b/a))cout << d*(b/a) << endl;
    else cout << "42" << endl;
return 0;
}
