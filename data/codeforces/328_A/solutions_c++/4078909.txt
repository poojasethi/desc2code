#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    long double a,b,c,d;
    cin>>a>>b>>c>>d;
    if(a+c==2*b&&b+d==2*c)
        cout<<2*d-c<<endl;
    else if(a*c==b*b&&b*d==c*c)
    {
        long double e=d*b/a;
        if(e==floor(e))cout<<e<<endl;
        else cout<<42<<endl;
    }
    else cout<<42<<endl;
    return 0;
}
