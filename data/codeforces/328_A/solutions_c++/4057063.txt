#include <iostream>
using namespace std;
int N1,N2,N3,N4;
int main()
{
    cin>>N1>>N2>>N3>>N4;
    if(N1+N3==2*N2&&N2+N4==2*N3)
        cout<<2*N4-N3<<endl;
    else
    {
        if(N1*N3==N2*N2&&N2*N4==N3*N3&&N4*N4%N3==0)
            cout<<N4*N4/N3<<endl;
        else
            cout<<42<<endl;
    }
    return 0;
}
