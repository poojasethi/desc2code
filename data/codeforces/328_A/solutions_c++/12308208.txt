#include<iostream>

using namespace std;

int main()
{
	int a,b,c,d;
	cin>>a>>b>>c>>d;
	if(a-b==b-c&&b-c==c-d)
	{
		cout<<d-(a-b)<<endl;;
	}
	else if((double)a/b==(double)b/c&&(double)b/c==(double)c/d&&((int)(d*b)/a==(double)(d*b)/a))
	{
		cout<<d/((double)a/b)<<endl;
	}
	else
	{
		cout<<"42"<<endl;
	}

	return 0;
}
