#include <cstdio>
using namespace std;

int a1, a2, a3, a4;

int main() {
	scanf("%d%d%d%d", &a1, &a2, &a3, &a4);
	if (a2 - a1 == a3 - a2 && a4 - a3 == a3 - a2)	printf("%d\n", a4 + a4 - a3);
	else	if (a2 * a2 == a1 * a3 && a3 * a3 == a2 * a4 && a4 * a4 % a3 == 0)	printf("%d\n", a4 * a4 / a3);
	else	printf("42\n");
}
