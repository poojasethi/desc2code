#include<iostream>
using namespace std;
int main()
{
	int a[4];
	cin>>a[0]>>a[1]>>a[2]>>a[3];
	if(2*a[1]==a[0]+a[2] && 2*a[2]==a[1]+a[3])
	{
		cout<<a[3]+a[2]-a[1]<<endl;
		return 0;
	}
	else if(a[1]*a[1]==a[0]*a[2] && a[2]*a[2]==a[1]*a[3])
	{
		if(a[3]*a[2]%a[1]!=0)
		{
			cout<<42<<endl;
		}
		else
		{
			cout<<(a[3]*a[2])/a[1]<<endl;
		}
	}
	else cout<<42<<endl;
}
