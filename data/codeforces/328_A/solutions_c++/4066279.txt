#include <iostream>
using namespace std;
double ans=42;
int main()
{
    double a,b,c,d;
    cin >>a >>b >>c >>d;
    if (b-a==c-b && c-b==d-c) ans=d+(b-a);
    int temp=(int)ans;
    if (ans!=temp) ans=42;
    if (b/a==c/b && c/b==d/c) ans=d*(b/a);
    temp=(int)ans;
    if (ans!=temp) ans=42;
    cout <<ans <<endl;
    return 0;
}
