#include <iostream>
using namespace std;

int main() {
	int a, b, c, d;
	cin >> a >> b >> c >> d;
	if ((a + c == b + b) && (b + d == c + c))
		cout << d + d - c << endl;
	else if ((a * c == b * b) && (b * d == c * c)) {
		int t = d * d / c;
		if (t * c == d * d)
			cout << t << endl;
		else
			cout << 42 << endl;
	}
	else
		cout << 42 << endl;
	return 0;
}
