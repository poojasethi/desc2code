#include<iostream>
using namespace std;
const int MAX=2147483647;
typedef long long LL;
int main(){
    LL a,b,c,d,ans=42;
    cin>>a>>b>>c>>d;
    if (a+c==2*b&&b+d==2*c)
        ans=2*d-c;
    else if(b*b==a*c&&c*c==b*d&&d*d%c==0)
        ans=d*d/c;
    if (ans>MAX) ans=42;
    cout<<ans<<endl;
    return 0;
}
