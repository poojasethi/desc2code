#include <bits/stdc++.h>
using namespace std;
int main() {
	int a,b,c,d;
	scanf("%d%d%d%d",&a,&b,&c,&d);
	for(int i=-10000;i<=10000;++i) {
		if(a+i==2*c && b+d==2*c && (a+c==2*b)) {
			printf("%d\n",i);
			return 0;
		}
		if(c*c==a*i && c*c==b*d && (b*b==a*c)) {
			printf("%d\n",i);
			return 0;
		}
	}
	printf("%d\n",42);
	return 0;
}
