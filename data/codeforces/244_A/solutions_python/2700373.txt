n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
o = range(1, k * n + 1)
c = [[] for i in xrange(k)]
for i, x in enumerate(a):
    c[i].append(x)
    o.remove(x)

for i in xrange(k):
    for j in xrange(n - 1):
        c[i].append(o.pop())

for p in c:
    print ' '.join(map(str, p))
