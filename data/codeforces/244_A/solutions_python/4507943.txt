n,k=map(int,raw_input().split())
lst=map(int,raw_input().split())
a=[]
for i in range(1,n*k+1):
    a.append(i)

b=[]
for j in a:
    if not j in lst:
        b.append(j)
c=[]
for i in  range(k):
    c.append([lst[i]])



for i in range(k):
    c[i]+=b[i*(n-1):(i+1)*(n-1)]

for cc in c:
    cc=map(str,cc)
    print " ".join(cc)
