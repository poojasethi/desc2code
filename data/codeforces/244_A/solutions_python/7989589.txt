n,k = map(int,raw_input().split())
a = map(int,raw_input().split())
p = range(1,n*k+1)
for x in xrange(k):
    ans = []
    ans.append(a[x])
    p.pop(p.index(a[x]))
    i = 0
    while len(ans) < n:
        if p[i] not in a:
            ans.append(p[i])
            del p[i]
        else:
            i += 1
    print " ".join(map(str,ans))
