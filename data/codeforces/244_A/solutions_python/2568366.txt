if __name__ == '__main__':
	n, k = [int(_) for _ in raw_input().split()]
	a = [int(_) for _ in raw_input().split()]
	given = []
	orange = list(range(1, n * k + 1))
	for i in range(k):
		given.append([a[i]])
		orange.remove(a[i])
	for i in range(k):
		given[i].extend(orange[i * (n - 1):i * (n - 1) + n - 1])
	for i in range(k):
		print ' '.join([str(_) for _ in given[i]])
