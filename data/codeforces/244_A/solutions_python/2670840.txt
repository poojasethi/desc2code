n, k = [int(x) for x in raw_input().split()]
ai = [int(x) for x in raw_input().split()]
childs = [[x] for x in ai]
j = 0
for i in range(1, n * k + 1):
    while (j < k and len(childs[j]) >= n): j += 1
    if j >= k: break
    if i in ai: continue
    childs[j].append(i)
for child in childs:
    print ' '.join([str(x) for x in child])
