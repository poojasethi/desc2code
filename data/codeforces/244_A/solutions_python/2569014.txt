seg = map( int, raw_input().split())
dict = {}
for x in seg:
    dict[x] = 1
for x in range(len(seg)):
    seg[x]= [seg[x]]
i = 0
for x in range(1, n*k+1):
  if not dict.get(x):
    seg[i].append(x)
    i = (i+1)%k
for i in range(k):
