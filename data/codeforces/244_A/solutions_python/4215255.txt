n,k=[int(x) for x in raw_input().split()]
w=[int(x) for x in raw_input().split()]
r=[]
l=[]
for i in w:
	r.append(i)
	
	l.append((str(i)+' '))
	for x in range(1,(n*k)+1):
		if x not in r and x not in w:
			if (len(r)+1)%n==0:
				r.append(x)
				l.append((str(x)+'\n'))
				break
			else:
				r.append(x)
				l.append((str(x)+' '))
	
print ''.join(l)
