[n,k] = map(int, raw_input().split())
p = map(int, raw_input().split())

q = []
for x in p:
	q.append([x])

r = []
for x in range(1,n*k+1):
	if x not in p:
		r.append(x)

for x in q:
	i = 1
	while i<n:
		el = r.pop()
		x.append(el)
		i = i + 1

for x in q:
	s = '' 
	for y in x:
		s = s + str(y) + ' '
	print s
