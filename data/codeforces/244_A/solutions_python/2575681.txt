n, k = map(int, raw_input().split())
a = map(int, raw_input().split())

pieces = [0] * (n * k + 1)
for i in range(k):
    pieces[a[i]] = 1

g = 1
for i in range(k):
    l = [str(a[i])]
    cnt = 1
    while cnt < n:
        if pieces[g] == 0:
            cnt += 1
            l += [str(g)]
        g += 1
    print ' '.join(l)
