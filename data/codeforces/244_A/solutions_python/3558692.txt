import sys

n, k = map(int, sys.stdin.readline().split())
a = map(int, sys.stdin.readline().split())

if n == 1:
    for aa in a: print aa
    exit(0)

segs = filter(lambda x: x not in a, xrange(1, n * k + 1))
for aa, i in zip(a, xrange(0, len(segs), n - 1)):
    print ' '.join(map(str, [aa] + segs[i:i+n-1]))
