#include <iostream>
using namespace std;
int n,i,k,x,a[11111],j,k0;

int main()
{
	cin>>n>>k;
	for(i=1;i<=k;i++)
	{
		cin>>x;
		a[x]=i;
	}
	for(i=1;i<=k;i++)
	{
		k0=n-1;
		for(j=1;j<=n*k;j++)
		{
			if (a[j]==i)
				cout<<j<<" ";
			if (!a[j]&&k0)
			{
				cout<<j<<" ";
				k0--;
				a[j]=i;
			}
		}
		cout<<endl;
	}
	return 0;
}
