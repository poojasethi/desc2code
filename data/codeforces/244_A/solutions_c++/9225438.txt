#include<iostream>
using namespace  std;
int main(){
	int i,n,k,j,a[35],b[1000],f=1;
	cin>>n>>k;
	for(i=1;i<=n*k;i++)
		b[i]=i;
	for(i=0;i<k;i++){
		cin>>a[i];
		b[a[i]]=0;
	}
	for(i=0;i<k;i++){
		cout<<a[i]<<" ";
		for(j=1;j<n;){
			if(b[f]){
				cout<<b[f]<<" ";
				j++;
			}
			f++;
		}
		cout<<endl;

	}
	return 0;
}
