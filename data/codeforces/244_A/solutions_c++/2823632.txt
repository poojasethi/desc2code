#include<iostream>
using namespace std;
int n,k,a[41],v[1001],i,t;
int main()
{
	cin>>n>>k;
	for(int i=0;i<k;i++)
		{
		    cin>>a[i];
            v[a[i]]=1;
		}
	for(i=0;i<k;i++)
	{
		t=n-1;
		for(int i=1;t&&i<=n*k;i++)
			if(!v[i])
				{
				    cout<<i<<' ';
				    v[i]=1;
				    t--;
                }
		 cout<<a[i]<<'\n';
	}
	return 0;
}
