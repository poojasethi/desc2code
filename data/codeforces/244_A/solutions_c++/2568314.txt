#include <stdio.h>
#define F(I,B,E) for(int I=B; I<E; I++)
main() {
  int n,k,p=1,a[33],d[999]={};
  scanf("%d%d",&n,&k);
  F(i,0,k)
    scanf("%d",a+i),
    d[a[i]]=1;
  F(i,0,k) {
    printf("%d ",a[i]);
    int c=1;
    while(c<n)
      if(d[p])
        ++p;
      else
        printf("%d ",p++),
        ++c;
  }
  return 0;
}
