#include <stdio.h>
int n,k,used[1000],a[30],x;
int main(){
    scanf("%d %d",&n,&k);
    int i,j;
    for(i=0;i<k;i++){ scanf("%d",&a[i]); used[a[i]-1]=1; }
    for(i=0;i<k;i++){
        printf("%d ",a[i]);
        for(j=1;j<n;j++){
            while(used[x]) x++;
            printf("%d ",++x);
        }
        printf("\n");
    }
    return 0;
}
