#include <cstdio>
int a[31][31],p[31],c[910];
int main()
{
	int n,k; scanf("%d%d",&n,&k);
	for(int i=0;i<k;++i) {
		int t;
		scanf("%d",&t);
		c[t]++;
		a[i][p[i]++]=t;
	}
	int x=0;
	for(int i=1;i<=n*k;++i) if(!c[i]){
		a[x%k][p[x%k]++]=i;
		x++;
	}
	for(int i=0;i<k;++i){
		for(int j=0;j<n;++j)printf("%d ",a[i][j]);
		puts("");
	}
	return 0;
}
