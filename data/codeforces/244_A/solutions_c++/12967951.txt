#include<stdio.h>
int n,k,t;
int a[10000],f[10000];
int main(){
	scanf("%d%d",&n,&k);
	for(int i=1;i<=k;i++){scanf("%d",&a[(i-1)*n]);f[a[(i-1)*n]]=1;}
	for(int i=1;i<=n*k;i++){
		if(f[i])continue;
		while(a[t]!=0)t++;
		a[t++]=i;
	}
	for(int i=0;i<k;i++){
		for(int j=0;j<n;j++)printf("%d%c",a[i*n+j],j==n-1?'\n':' ');
	}
}
