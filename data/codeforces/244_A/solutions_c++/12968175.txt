#include<stdio.h>
int main()
{
	int n,k,i,j,a[50],o[1000];
	scanf("%d%d",&n,&k);
	for(i=1;i<=n*k;i++)o[i]=1;
	for(i=0;i<k;i++){
		scanf("%d",&a[i]);
		o[a[i]]=0;
	}
	int p=1;
	for(i=0;i<k;i++){
		printf("%d ",a[i]);
		for(j=1;j<n;j++){
			if(o[p]==1)printf("%d ",p);
			else j--;
			p++;
		}
		printf("\n");
	}
}
