#include <iostream>
#include <cmath>
using namespace std;    
int n, k, a[111], b[1111], l, r;
int main () {
	cin >> n >> k;
	for (int i=0; i<k; i++) { cin >> a[i]; b[a[i]-1]=1;}
	for (int i=0; i<k; i++) {
		cout << a[i] << " ";  
		l=0, r=0;
		while(r!=n-1){
			if (!b[l]) { cout << l+1 << " "; b[l]=1; r++;}
			l++;
		}
	cout << endl;
	}
	return 0;
}
