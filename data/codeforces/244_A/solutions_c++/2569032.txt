#include<stdio.h>
#include<string.h>
int b[1003], n, k, t, a[33];
int main()
{
	int i, j;
	scanf("%d%d", &n, &k);
	for(i = 1; i <= k; i++)
		scanf("%d", &a[i]), b[a[i]]++;
	int id = 1;
	for(i = 1; i <= k; i++)
	{
		printf("%d ", a[i]);
		for(j = 1; j < n; j++)
		{
			while(b[id]) id++;
			printf("%d ", id++);
		}
		printf("\n");
	}
	return 0;
}
