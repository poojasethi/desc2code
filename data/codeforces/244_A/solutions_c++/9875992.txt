#include <iostream>

using namespace std;
int i,j,n,k;int a[100000],f[100000],h;
int main()
{
    cin>>n>>k;
    for(i=0;i<k;i++){cin>>a[i];f[a[i]]=1;}


    for(i=0;i<k;i++)
    {
        cout<<a[i]<<" ";
       for(j=1;j<=n*k;j++)
            if(f[j]!=1){cout<<j<<" ";h++;f[j]=1;if(h==n-1)break;}
  cout<<endl;h=0;  }
    return 0;
}
