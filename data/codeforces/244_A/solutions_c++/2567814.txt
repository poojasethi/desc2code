#include <cstdio>

int k, n, l, a[1000];
bool used[100000];

int main()
{
	scanf("%d%d", &n, &k);
	for (int i = 0; i < k; i++)
	{
		scanf("%d", &a[i]);
		used[a[i]] = 1;

	}
	l = 1;
	for (int i = 0; i < k; i++)
	{
		printf("%d ", a[i]);
		for (int j = 1; j < n; j++) 
		{
			while(used[l])
				l++;
			printf("%d ", l++);
		}
		puts("");
	}
}
