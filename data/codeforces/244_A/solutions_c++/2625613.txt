#include<cstdio>
#include<cstring>
int main()
{
	int j,i,n,k,a[35],b[1000],x;
	scanf("%d%d",&k,&n);
	memset(b,0,sizeof(b));
	for(i=0;i<n;i++){
		scanf("%d",&a[i]);
		b[a[i]]=1;
	}
	x=1;
	for(i=0;i<n;i++){
		printf("%d",a[i]);
		for(j=1;j<k;j++){
			while(b[x]==1){
				x++;
			}
			printf(" %d",x++);
		}
		printf("\n");
	}
	return 0;
}
