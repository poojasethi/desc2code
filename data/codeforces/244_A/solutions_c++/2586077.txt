#include<stdio.h>
int main()
{
	int n,k,d[3000],a[50],i,j,h;
	scanf("%d%d",&n,&k);
	for(i=1;i<3000;i++)
		d[i]=0;
	for(i=1;i<=k;i++)
	{
		scanf("%d",&a[i]);
		d[a[i]]=1;
	}
	j=1;
	for(i=1;i<=k;i++)
	{
		printf("%d ",a[i]);
		for(h=2;h<=n;h++)
		{
			while(d[j])j++;
			printf("%d ",j);
			j++;
		}
		printf("\n");
	}
	return 0;
}
