#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,i,req[40],cnt;
	list<int> a;
	cin>>n>>k;
	for(i=1;i<=n*k;i++)
		a.push_back(i);
	for(i=0;i<k;i++)
	{
		cin>>req[i];
		a.remove(req[i]);
	}
	for(i=0;i<k;i++)
	{
		cout<<req[i];
		cnt=n-1;
		while(cnt--)
		{
			cout<<' '<<a.front();
			a.pop_front();
		}
		cout<<endl;
	}
	return 0;
}
