#include<cstdio>
using namespace std;
int aa[1000],bb[1000];
int main(){
	int n,k,a,b;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=k;i++){
		scanf("%d",&bb[i]);aa[bb[i]]=1;
	}a=1;
	for(int i=1;i<=k;i++){
		b=2;printf("%d ",bb[i]);
		while(b<=n){
			if(aa[a]==0){printf("%d ",a);b++;}
			a++;
		}printf("\n");
	}
	return 0;
}
