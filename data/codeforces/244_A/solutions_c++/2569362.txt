#include <iostream>
using namespace std;
int main() {
	int i, j, n, k, child[30], pos=1;
	bool segment[901]={0};
	cin>>n>>k;
	for(i=0; i<k; i++) {
		cin>>child[i];
		segment[child[i]] = true;
	}
	for(i=0; i<k; i++) {
		cout<<child[i]<<" ";
		j=1;
		while(j<n) {
			if(!segment[pos]) {
				cout<<pos<<" ";
				j++;
			}
			pos++;
		};
		cout<<"\n";
	}
	return 0;
}
