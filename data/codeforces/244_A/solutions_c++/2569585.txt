#include<stdio.h>
int n,k,num[1000],x[1000],i,j,l;
int main(){
	scanf("%d %d",&n,&k);
	for(i=0;i<k;i++){scanf("%d",&x[i]);num[x[i]]++;}
	i=1;
	while(i<=n*k){
		printf("%d ",x[l]);
		j=1;
		while(j<n){
			if(num[i]==0){printf("%d ",i);j++;}
			i++;
		}
		l++;
		if(l==k)break;
		printf("\n");
	}
	return 0;
}
