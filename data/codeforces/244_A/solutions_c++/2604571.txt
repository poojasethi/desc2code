#include <iostream>
#include <set>
using namespace std;
int main(){
	int n,k,a[30],p=1;
	cin >> n >> k;
	set<int> aset;
	for(int i=0;i<k;i++){
		cin >> a[i];
		aset.insert(a[i]);
	}
	for(int i=0;i<k;i++){
		cout << a[i] <<" ";
		for(int j=0;j<n-1;j++){
			while(aset.count(p)) p++;
			cout << p++ <<" ";
		}
		cout << endl;
	}
	return 0;
}
