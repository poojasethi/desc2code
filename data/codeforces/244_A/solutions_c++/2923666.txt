#include <cstdio>

bool vis[35*35];

int main()
{
	int a[40], n, k;
	scanf("%d%d", &n, &k);
	for(int i=0; i<k; i++) 
	{ 
		scanf("%d", &a[i]);
		vis[a[i]] = true; 
	}
	for(int i=0, l=1; i<k; i++)
	{ 
		for(int j=0; j<n-1; l++) 
			if(!vis[l]) 
			{ 
				printf("%d ", l); 
				j++;
			}
			printf("%d\n", a[i]); 
	}
	return 0;
}
