#include<cstdio>
#include<iostream>
using namespace std;
int a[905],v[905],k2,k,n,i,j,ev;
int main()
{
	cin>>k>>n;
	for(i=1;i<=n;++i)
		cin>>a[i],v[a[i]]=1;
	for(i=1;i<=n;++i)
	{
		cout<<a[i]<<" ";
		k2=k-1;
		while(k2--)
		{
			ev=1;
			for(j=1;j<=n*k&&ev;++j)
				if(v[j]==0)
					v[j]=1,cout<<j<<" ",ev=0;
		}
		cout<<"\n";
	}
	return 0;
}
