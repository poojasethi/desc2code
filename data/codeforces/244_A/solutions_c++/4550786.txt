#include<stdio.h>
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	int i,a[k],b[n*k+1],c,p=1;
	for(i=0;i<=n*k;i++)
	{
		b[i]=0;
	}
	for(i=0;i<k;i++)
	{
		scanf("%d",&a[i]);
		b[a[i]]=1;
	}
	for(i=0;i<k;i++)
	{
		printf("%d",a[i]);
		c=n-1;
		while(c)
		{
			if(b[p]==0)
			{
				c--;
				printf(" %d",p);
			}
			p++;
		}
		printf("\n");
	}
	return 0;
}
