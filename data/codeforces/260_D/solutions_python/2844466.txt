n=int(raw_input())
b=[];
w=[];
for i in xrange(n):
	nod=map(int,raw_input().split())+[i+1]
	if nod[0]==1:
		b.append(nod)
	else :
		w.append(nod)

b=sorted(b,key=lambda x:x[1])
w=sorted(w,key=lambda x:x[1])
p,q=0,0
nb,nw=len(b),len(w)
while p<nb and q<nw:
	if b[p][1]<w[q][1] or (b[p][1]==w[q][1] and nb-p>nw-q):
		print b[p][2],w[q][2],b[p][1]
		w[q][1]-=b[p][1]
		p+=1
	else:
		print b[p][2],w[q][2],w[q][1]
		b[p][1]-=w[q][1]
		q+=1
