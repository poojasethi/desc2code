n = input()
a = []
b = []
for i in range(n):
	c, s = map(int, raw_input().split())
	[a, b][c] += [[s, i + 1]]
r = []
for _ in range(1, n):
	s, i = a[-1]
	S, I = b[-1]
	v = min(s, S)
	a[-1][0] -= v
	b[-1][0] -= v
	r += [[i, I, v]]
	[b, a][s < S or s == S and len(a) > 1].pop()
print '\n'.join(' '.join(map(str, x)) for x in r)
