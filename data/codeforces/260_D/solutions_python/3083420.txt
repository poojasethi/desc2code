#!/usr/local/bin/python

n = int(raw_input())
color, s = [], []
for i in xrange(n):
	c, S = map(int, raw_input().split())
	color.append(c)
	s.append(S)

x, y = 0, 0
while not color[x] == 0: x+=1
while not color[y] == 1: y+=1

while x < n and y < n:
	val = min([s[x], s[y]])
	s[x]-=val
	s[y]-=val

	print "{} {} {}".format(x+1, y+1, val)

	nextX = n
	if s[x] == 0:
		nextX = x+1
		while nextX < n and not color[nextX] == 0: nextX+=1

	if nextX < n: x = nextX
	else:
		y+=1
		while y < n and not color[y] == 1: y+=1
