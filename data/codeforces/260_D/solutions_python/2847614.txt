#!/usr/bin/env python
# -*- coding: utf-8 -*-

n = int(raw_input())
b = []
w = []
for i in range(1, n+1):
    c, x = [ int(k) for k in raw_input().split() ]
    if c:
        b.append([i, x])
    else:
        w.append([i, x])

lb = len(b)
lw = len(w)
i = 0
j = 0
while i<lb and j<lw:
    if b[i][1]<w[j][1]:
        print b[i][0], w[j][0], b[i][1]
        w[j][1] -= b[i][1]
        i += 1
    else:
        print b[i][0], w[j][0], w[j][1]
        b[i][1] -= w[j][1]
        j += 1

while i<lb-1:
    i += 1
    print b[i][0], w[lw-1][0], 0
while j<lw-1:
    j += 1
    print b[lb-1][0], w[j][0], 0

exit()
