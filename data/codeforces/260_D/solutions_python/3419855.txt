n=input()
vw,vb=v=[[],[]]
for i in range(n):
  x,y=map(int,raw_input().split())
  v[x]+=[[y,i+1]]
vw.sort()
vb.sort()
w,b,nw,nb=0,0,len(vw),len(vb)
while w<nw and b<nb:
  d=vb[b][0]-vw[w][0]
  print vw[w][1],vb[b][1],min(vw[w][0],vb[b][0])
  if d<0 or (0==d and nb-b>nw-w):
    vw[w][0]-=vb[b][0]
    b+=1
  else:
    vb[b][0]-=vw[w][0]
    w+=1
