#include<cstdio>
#include<queue>
using namespace std;

priority_queue< pair<int,int> > q[2];
int main()
{
	int n, a, b;
	scanf( "%d", &n );
	for( int i = 1; i <= n; i++ ){
		scanf( "%d%d", &a, &b );
		q[a].push( make_pair( -b, i ) );
	}
	int x = 0;
	for( int i = 1; i < n; i++ ){
		if( -q[x].top().first > -q[x^1].top().first ) x = x^1;
		if( q[x].top().first == q[x^1].top().first && q[x].size() < q[x^1].size() ) x = x^1;
		printf( "%d %d %d\n", q[x].top().second, q[x^1].top().second, -q[x].top().first );
		a = q[x^1].top().first - q[x].top().first; b = q[x^1].top().second;
		q[x].pop(); q[x^1].pop();
		q[x^1].push( make_pair( a, b ) );
	}
}
