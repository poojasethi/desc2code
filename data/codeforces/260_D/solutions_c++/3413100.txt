#include <vector>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <queue>

using namespace std;

const int N = 100010 ;
const int INFINITE = 0x3F3F3F3F ;

int n , m , k;
vector<int> q0 , q1 ;
int s[N] ;

int main(){
    scanf("%d",&n);
    for (int i = 1 ; i <= n ; i++){
        int c ;
        scanf("%d%d",&c,&s[i]);
        if (c) q1.push_back(i);
        else q0.push_back(i) ;
    }

    int x = 0 , y = 0 ;
    while (x < q0.size() && y < q1.size()){
        int u = q0[x] , v = q1[y] ;
        int del = min(s[u],s[v]);
        s[u]-=del ; s[v]-=del ;
        printf("%d %d %d\n",u,v,del);
        if (s[u]==0 && x<q0.size()-1) x++ ;
        else y++ ;
    }
    return 0;
}
