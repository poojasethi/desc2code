#include<cstdio>
#include<algorithm>
#define s first
#define p second
#define MAXN 100010
typedef std::pair<int,int> pii;
int na,nb,n0,x,y,l;
pii a[MAXN],b[MAXN];
main(){
  scanf("%d",&n0);
  for(int i=1;i<=n0;++i){
    scanf("%d%d",&x,&y);
    if (!x)
      a[na++]=pii(y,i);
    else
      b[nb++]=pii(y,i);
  }
  pii *pa=a,*pb=b;
  for(;pa!=a+na&&pb!=b+nb;++pa){
    if (pa->s==0)
      printf("%d %d 0\n",pa->p,pb->p);
    while(pa->s){
      //printf("%d %d\n",pa->p,pb->p);
      l=std::min(pa->s,pb->s);
      pa->s-=l;pb->s-=l;
      printf("%d %d %d\n",pa->p,pb->p,l);
      if (pa->s)
        ++pb;
    }
  }
  for(--pa,++pb;pb!=b+nb;++pb)
    printf("%d %d 0\n",pa->p,pb->p);
}
