#include <bits/stdc++.h>

using namespace std;
const int MAXN = (int)1e5 + 10;

int n, s[MAXN];
vector <int> v[2];

int main()
{
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
        int c;
        cin >> c >> s[i];
        v[c].push_back(i);
    }

    int i, j;
    i = j = 0;

    for (int k = 0; k < n-1; ++k)
    {
        int v1 = v[0][i], v2 = v[1][j];
        int val = min(s[v1], s[v2]);

        cout << v1+1 << " " << v2+1 << " " << val << endl;

        s[v1] -= val;
        s[v2] -= val;

        if (s[v1] == 0 && i + 1 < v[0].size()) ++i;
        else ++j;
    }

    return 0;
}
