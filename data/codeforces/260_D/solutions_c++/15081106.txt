#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1000*100 + 5;
int n;
long long a[MAXN];
vector <int> v[2 + 1];

int main(){
    cin >> n;
    for (int i = 0, c; i < n; i++){
        cin >> c >> a[i];
        v[c].push_back (i);
    }
    
    for (int i = 0, j = 0, k = 0; k < n - 1; k++){
        long long x = v[0][i], y = v[1][j];
        long long m = min (a[x], a[y]);
        cout << x + 1 << " " << y + 1 << " " << m << endl;
        a[x] -= m;
        a[y] -= m;
        if (a[x] == 0 && i + 1 < (int)v[0].size())
            i ++;
        else
            j ++;
    }
        
    return 0;
}
