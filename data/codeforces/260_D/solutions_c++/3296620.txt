#include <set>
#include <iostream>

#define ff first
#define ss second

using namespace std;

int V, x, y;
set<pair<int, int> > w, b;

int main()
{
  cin >> V;
  for (int i = 1; i <= V; i++)
  {
    cin >> x >> y;
    (x ? b : w).insert(make_pair(y, i));
  }
  for (; !b.empty(); b.erase(b.begin()))
  {
    if (make_pair(w.begin()->ff, -(int)w.size()) < make_pair(b.begin()->ff, -(int)b.size())) b.swap(w);
    cout << b.begin()->ss << " " << w.begin()->ss << " " << b.begin()->ff << endl;
    pair<int, int> t = *w.begin();
    w.erase(*w.begin());
    t.ff -= b.begin()->ff;
    w.insert(t);
  }
  return 0;
}
