#include <bits/stdc++.h>

using namespace std;

#define MAX 100005

vector<int> vec[2];
int s[MAX];

int main(){
    int n, c;
    scanf("%d", &n);

    for(int i=0 ; i<n ; i++){
        scanf("%d %d", &c, &s[i]);
        vec[c].push_back(i);
    }

    for(int k=1, i=0, j=0 ; k<n ; k++){
        int u = vec[0][i], v = vec[1][j], w;
        w = min(s[u], s[v]);
        printf("%d %d %d\n", u+1, v+1, w);
        s[u] -= w, s[v] -= w;
        if(s[u] == 0 && i<vec[0].size()-1)
            i++;
        else 
            j++;
    }

    return 0;
}
