#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e5+123;

int N,val[maxn];
vector < int > v[2];

int main(){
	
	scanf("%d",&N);
	
	for( int i=1,a ; i <= N ; i++ ){
		scanf("%d%d",&a,&val[i]);
		v[a].push_back(i);
	}
	
	for( int i=0,j=0,k=1 ; k < N ; k++ ){
		
		int a = v[0][i] , b = v[1][j];
		int x = min( val[a] , val[b] );
		
		printf("%d %d %d\n",a,b,x);
		
		val[a] -= x;
		val[b] -= x;
		
		if( !val[a] and i+1 < (int)v[0].size() )
			i++;
		else
			j++;
	}
	
	return 0;
}
