# include <iostream>
# include <cstdio>
# include <cstring>
# include <algorithm>
#include <queue>
using namespace std;

const int N = 100005;

int n, x, p, a, b;
int s[N], A[N], B[N];

bool cmp(int x, int y){return s[x] < s[y];}

int main()
{
	cin >> n;
	for (int i = 0; i < n; ++i){
		cin >> x >> s[i];
		if (x)
			A[a++] = i;
		else
			B[b++] = i;
	}
	
	for (int i = 0; i < a; ++i) 
	{
		if (p == b) 
			printf("%d %d 0\n", A[i]+1, B[p-1]+1);
		while (p < b)
		{
			if (s[A[i]] >= s[B[p]])
			{
				printf("%d %d %d\n", A[i]+1, B[p]+1, s[B[p]]);
				s[A[i]] -= s[B[p]];
				++p;
			}
			else{
				printf("%d %d %d\n", A[i]+1, B[p]+1, s[A[i]]);
				s[B[p]] -= s[A[i]];
				break;
			}
		}
	}
	return 0;
}
