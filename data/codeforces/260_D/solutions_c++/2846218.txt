#include <cstdio>
#include <cstring>
#include <algorithm>
#define FOR(i,s,e) for (int i=(s); i<(e); i++)
#define N 100005
using namespace std;

int n, x, p, a, b;
int s[N], A[N], B[N];

int main(){
	scanf("%d", &n);
	FOR(i,0,n){
		scanf("%d%d", &x, &s[i]);
		if (x) A[a++] = i;
		else B[b++] = i;
	}
	
	FOR(i,0,a){
		if (p == b) printf("%d %d 0\n", A[i]+1, B[p-1]+1);
		while (p < b){
			if (s[A[i]] >= s[B[p]]){
				printf("%d %d %d\n", A[i]+1, B[p]+1, s[B[p]]);
				s[A[i]] -= s[B[p]];
				++p;
			}
			else{
				printf("%d %d %d\n", A[i]+1, B[p]+1, s[A[i]]);
				s[B[p]] -= s[A[i]];
				break;
			}
		}
	}
	return 0;
}
