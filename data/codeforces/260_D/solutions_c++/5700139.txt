#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>

using namespace std;

const int MAX_N = 100001;

int n, c, s, cntW, cntB;
pair<int,int> w[MAX_N], b[MAX_N];

void solve() {    
    int cnt = 0, i = 1, j = 1;
    while (cnt < n - 1) {
        int dif = min(w[i].first, b[j].first);
        w[i].first -= dif, b[j].first -= dif;
        printf("%d %d %d\n", w[i].second, b[j].second, dif);
        if (0 == w[i].first && i != cntW) ++i;
        else if (0 == b[j].first) ++j;
        ++cnt;
    }
}

int main() {
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) {
        scanf("%d%d", &c, &s);
        if (0 == c) w[++cntW] = make_pair(s, i);
        else b[++cntB] = make_pair(s, i);
    }
    solve();
    return 0;
}
