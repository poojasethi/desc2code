#include<cstdio>
#include<algorithm>
#define s first
#define p second
#define MAXN 100010
typedef std::pair<int,int> pii;
int n0,x,y,l;
pii a[MAXN],*ea=a,*b=a+MAXN-1,*eb=b,*pa,*pb;
main(){
  scanf("%d",&n0);
  for(int i=1;i<=n0;++i){
    scanf("%d%d",&x,&y);
    if (!x)
      *ea=pii(y,i),++ea;
    else
      *eb=pii(y,i),--eb;
  }
  for(pa=a,pb=b;pa!=ea&&pb!=eb;++pa){
    if (pa->s==0)
      printf("%d %d 0\n",pa->p,pb->p);
    while(pa->s){
      l=std::min(pa->s,pb->s);
      pa->s-=l;pb->s-=l;
      printf("%d %d %d\n",pa->p,pb->p,l);
      if (pa->s)
        --pb;
    }
  }
  for(--pa,--pb;pb!=eb;--pb)
    printf("%d %d 0\n",pa->p,pb->p);
}
