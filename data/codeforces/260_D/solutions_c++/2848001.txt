#include <iostream>
#include <algorithm>
using namespace std;
#define N 100005
int B[N],b[N],W[N],w[N],n,Bn,Wn;
int main(){
    cin>>n;
    int i,j;
    for(i=0;i<n;i++){
        int x,y;cin>>x>>y;
        if(x) B[++Bn]=y,b[Bn]=i+1;else W[++Wn]=y,w[Wn]=i+1;
    }
    for(i=1,j=1;i<=Bn&&j<=Wn;i++){
        if(!B[i])  cout<<b[i]<<" "<<w[j]<<" "<<0<<endl;
        else while(B[i])
        {
            int t=min(B[i],W[j]); B[i]-=t,W[j]-=t;cout<<b[i]<<" "<<w[j]<<" "<<t<<endl;
            if(B[i]) j++;
        }
    }
    i++,j++;
    while(j<=Wn) cout<<b[Bn]<<" "<<w[j++]<<" "<<0<<endl;
    while(i<=Bn) cout<<b[i++]<<" "<<w[Wn]<<" "<<0<<endl;


}
