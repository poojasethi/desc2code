#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <map>
#include <set>
using namespace std;

typedef pair<int,int> PII;

int main()
{
	int n;
	cin>>n;
	vector<PII> v0,v1;
	for(int i=0,c,w;i<n;i++)
	{
		cin>>c>>w;
		if(c)
			v1.push_back(make_pair(w,i+1));
		else
			v0.push_back(make_pair(w,i+1));
	}
	sort(v0.begin(),v0.end());
	sort(v1.begin(),v1.end());
	int p0=0,p1=0;
	for(int i=1;i<n;i++)
	{
		int w=min(v0[p0].first,v1[p1].first);
		cout<<v0[p0].second<<" "<<v1[p1].second<<" "<<w<<endl;
		v0[p0].first-=w;v1[p1].first-=w;
		if(v0[p0].first==0&&v1[p1].first==0)
		{
			if(p0+1<v0.size())
				p0++;
			else
				p1++;
		}
		else
		{
			if(v0[p0].first==0)
				p0++;
			if(v1[p1].first==0)
				p1++;
		}
	}
	return 0;
}
