#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

const int N = 100003;
int s[N];

int main()
{
    int n;
    scanf("%d", &n);
    vector<int> S[2];
    for(int i = 1; i <= n; ++i) {
        int c;
        scanf("%d%d", &c, s+i);
        S[c].push_back(i);
    }
    
    int u, v;
    while(!S[0].empty()) {
        u = S[0].back();
        v = S[1].back();
        int x = min(s[u], s[v]);
        printf("%d %d %d\n", u, v, x);
        s[u] -= x;
        s[v] -= x;
        if(s[u] == 0) S[0].pop_back();
        else S[1].pop_back();
    }
    
    S[1].pop_back();
    while(!S[1].empty()) {
        printf("%d %d 0\n", u, S[1].back());
        S[1].pop_back();
    }
        
    return 0;
}
