#include <bits/stdc++.h>
#define pb push_back 
#define pf push_front 
#define popf pop_front 
#define popb pop_back 
#define mp make_pair 
#define f first
#define s second

using namespace std;

const int MX = 100000 + 5; 
int N,val[MX];
vector < int > cmp[2];

int main(){
	
	scanf("%d",&N);
	
	for( int i=1,u ; i <= N ; i++ ){
		scanf("%d%d",&u,&val[i]);
		cmp[u].pb(i);
	}
	int i=0,j=0 ;
	for( int k=1 ; k < N ; k++ ){
		int u = cmp[0][i] , b = cmp[1][j];
		int x = min( val[u] , val[b] );
		val[u] -= x;
		val[b] -= x;
		if( !val[u] && i+1 < (int)cmp[0].size() )
			i++;
		else
			j++;
		printf("%d %d %d\n",u,b,x);
	}
	
	return 0;
}
