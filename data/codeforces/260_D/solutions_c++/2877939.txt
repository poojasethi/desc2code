#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
const int MAX_N = 1e5+7;
#define X first
#define Y second
int main()
{
  int n;
  cin>>n;
  vector< pair<int,int> > b, w;
  for (int i = 1; i <= n; i++) {
    int c, s;
    cin>>c>>s;
    (c?w:b).push_back(make_pair(s,i));
  }
  for (int i = 0, j = 0; i < w.size(); i++) {
    if (j == b.size()) cout<<w[i].Y<<" "<<b[j-1].Y<<" 0"<<endl;
    while (j < b.size()) {
      if (w[i].X >= b[j].X) {
	cout<<w[i].Y<<" "<<b[j].Y<<" "<<b[j].X<<endl;
	w[i].X -= b[j].X;
	j++;
      } else {
	cout<<w[i].Y<<" "<<b[j].Y<<" "<<w[i].X<<endl;
	b[j].X -= w[i].X;
	break;
      }
    }
  }
  return 0;
}
