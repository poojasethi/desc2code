typedef long long LL;

using namespace std;
int main(){
	int n;
	cin>>n;
	priority_queue<pair<int,int> > q[2];
	for(int i=1;i<=n;i++){
		int c,s;
		cin>>c>>s;
		q[c].push(make_pair(-s,i));
	}
	for(int i=0;i<n-1;i++){
		pair<int,int> p1=q[0].top(),p2=q[1].top();
		int v1=-p1.first;
		int v2=-p2.first;
		q[0].pop();
		q[1].pop();
		cout<<(p1.second)<<" "<<(p2.second)<<" "<<min(v1,v2)<<endl;
		if((v1<v2&&(!q[0].empty()))||(q[1].empty())){
			q[1].push(make_pair(v1-v2,p2.second));
		}
		else{
			q[0].push(make_pair(v2-v1,p1.second));
		}
	}
	return 0;
}
