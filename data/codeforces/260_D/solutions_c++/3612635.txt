#include <iostream>
#include <string>
#include <algorithm>
#include <vector>
#include <queue>
#include <set>

using namespace std;
#define LL unsigned long long
#define mp make_pair
#define pb push_back
#define PII pair<int,int>
#define X first
#define Y second

set<PII> S[3];

int main(){
    int n,c,s,w,b;
    cin >>n;
    for(int i=0;i<n;i++){
        cin >>c>>s;
        if(c==0) S[0].insert(mp(s,i+1)); else S[1].insert(mp(s,i+1));
    }
    for(int i=0;i<n-1;i++){
        w=(S[1].size()<S[0].size());
        b=(w==0);
        PII v1=*(S[b].begin()),v2=*(--S[w].end());
        S[b].erase(v1);
        S[w].erase(v2);
        cout <<v1.Y<<" "<<v2.Y<<" "<<v1.X<<endl;
        S[w].insert(mp(v2.X-v1.X,v2.Y));
    }
    return 0;
}
