#include <iostream>
#include <cstring>
#include <cmath>
#include <vector>
using namespace std; 
const int M = 1e5+10; 
int s[M], n; 
vector<int>p[2]; 

int main(){
	cin >> n; 
	for(int i = 1, c; i <= n; i++){
		cin >> c >> s[i]; 
		p[c].push_back(i); 
	}
	for(int k = 1, i = 0, j = 0; k<n; k++){
		int u = p[0][i], v = p[1][j], del = min(s[u], s[v]); 
		cout << u << " " << v << " " << del << endl; 
		s[u] -= del, s[v] -= del; 
		if(!s[u]&&i<p[0].size()-1)i++; 
		else j++;  
	}
	return 0; 
}
