# -*- coding: utf-8 -*-
s=raw_input()
sl=len(s)
a=xrange(0,1000,8)
l=len(a)
ans=-1
for i in xrange(l):
    A=str(a[i])
    L=len(A)
    p=0
    f=1
    if L>sl :break
    for j in xrange(L):
        p=s.find(A[j],p)
        if p==-1:
            f=0
            break
        else:p=p+1
    if f==1:
        ans=a[i]
        break
if ans!=-1:
    print "YES\n",ans
else:
    print "NO"
