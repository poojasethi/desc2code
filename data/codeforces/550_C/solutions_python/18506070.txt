ba=raw_input()
def isin(small,big):
    pos=0
    for t in big:
        try:
            if small[pos]==t:
                pos+=1
        except:
            return True
    if pos>=len(small):
        return True
    return False
bla=[]
for i in range(125):
    bla.append(str(i*8))
num=-1
good=False
for t in bla:
    if isin(t,ba):
        good=True
        num=t
if good:
    print "YES"
    print num
