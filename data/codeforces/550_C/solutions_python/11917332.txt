def check(s):
	v = 0
	for i in xrange(len(s)):
		for j in xrange(i + 1, len(s)):
			for k in xrange(j + 1, len(s)):
				v = 100 * int(s[i]) + 10 * int(s[j]) + int(s[k])
				if not v % 8:
					return True, v
	return False, 0

s = "00" + raw_input()
res, num = check(s)

if res:
	print "YES"
	print num

else: print "NO"
