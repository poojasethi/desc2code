lis=[]
for i in range(0,125):
    lis.append(str(8*i))
s=raw_input()
l1=len(s)
for j in lis:
    l=len(j)
    k=0
    t=0
    cnt=0
    while k<l:
        while t<l1 and j[k]!=s[t]:
            t+=1
        if t<l1:
            cnt+=1
        t+=1
        k+=1
    if cnt==l:
        print "YES"
        print j
        break
else:
    print "NO"
