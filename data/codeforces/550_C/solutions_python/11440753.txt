def C():
    n=raw_input()
    if "8" in n: return "8"
    if "0" in n: return "0"
    for i in xrange(16, 100, 8):
        st=str(i)
        if st[0] in n and st[1] in n and n.index(st[0])<n.index(st[1]):
            return st[0]+st[1]
    for a in xrange(len(n)-2):
        for b in xrange(a+1, len(n)-1):
            for c in xrange(b+1, len(n)):
                if int(n[a]+n[b]+n[c])%8==0:
                    return n[a]+n[b]+n[c]
q=C()
if q:
    print "YES"
    print q
else: print "NO"
