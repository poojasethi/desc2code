n = raw_input()
if '0' in n:
    print 'YES\n0'
    exit(0)
n = '00' + n
l = len(n)
for i in xrange(l):
    for j in xrange(l):
        for k in xrange(l):
            if i < j < k:
                end = int(n[i] + n[j] + n[k])
                if end % 8 == 0:
                    print 'YES'
                    print '{}'.format(end)
                    exit(0)
print 'NO'
