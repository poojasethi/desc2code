n=raw_input()
n='00'+n
size=len(n)

answer=0
works=False

for i in range(0, size):
    for j in range(i+1, size):
        for k in range(j+1, size):
            l=int(n[i])
            m=int(n[j])
            o=int(n[k])
            if (100*l+10*m+o)%8==0: #and not (100*l+10*m+o)==0
                works=True
                answer=100*l+10*m+o
                break

if not works:
    print "NO"
else:
    print "YES"
    print str(answer)
