s = raw_input()

def test(x, s):
    sx = str(x)
    px = len(sx) - 1
    ps = len(s) - 1
    while px != -1:
        if ps == -1:
            return False
        if sx[px] == s[ps]:
            px -= 1
        ps -= 1
    return True

ans = -1

for i in range(50):
    if(test(i*8, s)):
        ans = i*8
        break

if ans == -1:
    print("NO")
else:
    print("YES")
    print(ans)
