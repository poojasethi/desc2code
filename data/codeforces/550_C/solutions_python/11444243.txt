import re
ss = raw_input()
di = []

for i in range(0, 1000):
    if i % 8 == 0:
        di.append(i)

def get_re(n):
    s = ''
    sn = str(n)
    for i in sn:
        s += '.*' + i
    s += '.*'
    return s

def solve():
    for i in di:
        if re.match(get_re(i), ss):
            return i
    return -1


if solve() == -1:
    print 'NO'
else:
    print 'YES\n%d' % solve()
