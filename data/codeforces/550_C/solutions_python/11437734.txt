import re
div_by_8 = []
for i in range(126): div_by_8.append(str(i*8))
n = raw_input()
found = False
for jj in div_by_8:
    my_regex = r".*"
    for ch in jj: my_regex += re.escape(ch) + r".*"
    if re.search(my_regex, n):
        print "YES"
        print jj
        found = True
        break
if not found:
    print "NO"
