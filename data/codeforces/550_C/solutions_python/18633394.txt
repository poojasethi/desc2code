a = map(int, list(raw_input()))

for i in range(len(a)):
	if a[i] % 8 == 0: print "YES\n" + str(a[i]); exit(0)

for i in range(len(a)):
	for j in range(i+1, len(a)):
		if (10*a[i] + a[j]) % 8 == 0: print "YES\n" + str(a[i]) + str(a[j]); exit(0)

for i in range(len(a)):
	for j in range(i+1, len(a)):
		for k in range(j+1, len(a)):
			if (100*a[i] + 10*a[j] + a[k]) % 8 == 0: print "YES\n" + str(a[i]) + str(a[j]) + str(a[k]); exit(0)
print "NO"
