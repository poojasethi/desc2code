s = map(int, raw_input())
s1 = set()
s2 = set()
s3 = set()
s.reverse()

for i in s:
  # s3 insert
  for (a, b) in s2:
    s3.add((i, a, b))
  for a in s1:
    s2.add((i, a))
  s1.add(i)

res = -1
for (a, b) in s2:
  s3.add((0, a, b))
for i in s1:
  s3.add((0, 0, i))
for (a, b, c) in s3:
  if (100 * a + 10 * b + c) % 8 == 0:
    res = (100 * a + 10 * b + c)

if res >= 0:
  print "YES"
  print res
else:
  print "NO"
