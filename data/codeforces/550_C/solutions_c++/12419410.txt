#include <iostream>
#include <string.h>
using namespace std;
int main()
{
	int i,x,j;
	string s,a="00";
	cin >> s;
	a+=s;
	for (i=0;i<a.size();i++)
	{
		for (x=i+1;x<a.size();x++)
		{
			for (j=x+1;j<a.size();j++)
			{
				int test=(a[i]-'0')*100+(a[x]-'0')*10+(a[j]-'0');
				if (test%8==0)
				{
					cout << "YES" << endl << test;
					return 0;
				}
			}
		}
	}
	cout << "NO";
}
