# include <bits/stdc++.h>
using namespace std;

int main () {
	char s[105];
	scanf("%s",s);
	int n = strlen(s);
	int f=1,flag=0;
	int i,j,k;
	for ( i=0 ; i < 1000 ; i += 8) {
		j=n-1,k=i;
		
		while (j >= 0) {
			if ((s[j]-48) == k%10) {
				k /= 10;
				if (k==0) {
					flag=1;
					break;
				}
			}
			j--;
		}
		if(flag) {
			printf("YES\n%d\n",i);
			f=0;
			break;
		}
	}
	if(f) {
		printf("NO\n");
	}
	return 0;
}
