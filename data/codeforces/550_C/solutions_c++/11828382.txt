#include <iostream>

using namespace std;

int main()
{ string s;
cin>>s;
s="00"+s; // to make check for the first digit
for(int i=0;i<s.size();i++)
{
    for(int j=i+1;j<s.size();j++)
    {
        for(int m=j+1;m<s.size();m++)
        {
            int k=(100*(s[i]-'0')+10*(s[j]-'0')+(s[m]-'0'));
            if(k%8==0)
            {
                cout<<"YES"<<endl<<k;
                return 0;
            }
        }
    }
}
cout<<"NO";
    return 0;
}
