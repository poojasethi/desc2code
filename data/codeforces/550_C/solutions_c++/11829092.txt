#include <iostream>
#include <stdlib.h>
using namespace std;

int main()
{
   string s;
   cin>>s;
   string ma="00"+s;
   int x;
   for(int i=0;i<ma.length();i++){
    for(int j=i+1;j<ma.length();j++){
    for(int k=j+1;k<ma.length();k++){
    x=((ma[i]-48)*100)+((ma[j]-48)*10)+((ma[k]-48));
    if(x%8==0){
        cout<<"YES"<<endl<<x;
        exit(0);
    }
   }
   }
   }
   cout<<"NO";
    return 0;
}
