#include <bits/stdc++.h>
using namespace std;

string n;

int main(){
	cin >> n;

	n = "0" + n;

	int num;
	int flag = 0;
	for (int i=0;i<n.size();i++){
		for (int j=i;j<n.size();j++){
			if (i==j && j!=0) continue;
			for (int k=j+1;k<n.size();k++){
				num = 100*(n[i]-'0') + 10*(n[j]-'0') + (n[k]-'0');
				if (num%8==0){
					cout << "YES" << endl;
					cout << num << endl;
					return 0;
				}
			}
		}
	}
	
	cout << "NO" << endl;
	return 0;

}
