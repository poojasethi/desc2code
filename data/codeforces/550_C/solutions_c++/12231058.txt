#include<bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	cin>>s;
	s="00"+s;
	for(int i=0;i<s.size();i++)
	{
		for(int j=i+1;j<s.size();j++)
		{
			for(int k=j+1;k<s.size();k++)
			{
				if((s[i]*100+s[j]*10+s[k])%8==0)
				{
					cout<<"YES\n";
					if(s[i]=='0'&&s[j]=='0') cout<<s[k]<<"\n";
					else if(s[i]=='0') cout<<s[j]<<s[k]<<"\n";
					else cout<<s[i]<<s[j]<<s[k]<<"\n";
					return 0;
				}
			}
		}
	}
	cout<<"NO\n";
	return 0;
}
