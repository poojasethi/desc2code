#include<stdio.h>
#include<string.h>
const int maxn=100+10;
char ch[maxn];
int a[maxn];
int main()
{
	scanf("%s",ch);
	int len=strlen(ch)+2;
	for (int i=3;i<=len;i++)
	a[i]=ch[i-3]-'0';
	a[1]=a[2]=0;
	for (int i=1;i<=len-2;i++)
	for (int j=i+1;j<=len-1;j++)
	for (int k=j+1;k<=len;k++)
	{
		int t=a[i]*100+a[j]*10+a[k];
		if (t%8==0)
		{
			printf("YES\n");
			printf("%d\n",t);
			return 0;
		}
	}
	printf("NO\n");
	return 0;
}
