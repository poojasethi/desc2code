# include <bits\stdc++.h>

using namespace std;

int main(void)
{
char a[123] = { 0 };
cin >> a;
int al = strlen(a);
for (int i = 0; i < al; ++i)
for (int u = i; u < al; ++u)
for (int j = u; j < al; ++j)
{
string b;
b += a[i];
if (i != u)
b += a[u];
if (j != u)
b += a[j];

if (atoi(b.data()) % 8 == 0)
{
cout << "YES" << '\n';
cout << b;
exit(0);
}
}
puts("NO");
return 0;
}
