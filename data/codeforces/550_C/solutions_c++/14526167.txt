#include <bits/stdc++.h>

using namespace std;

string s , a = "00";

int number;

int main() {
	cin >> s;
	a += s;
	for (int i = 0; i < a.size(); i++)
		for (int j = i + 1; j < a.size(); j++)
			for (int k = j + 1 ; k < a.size() ;k++) {
				number = (a[i] - '0') * 100 + (a[j] - '0') * 10 + (a[k] - '0');
				if (number % 8 == 0) {
					cout << "YES\n" << number;
					return 0;
				}
			}
	cout << "NO";
	return 0;
}
