#include <stdio.h>
int a[105];
int main()
{
	int n=0,cnt=0,cur=0,pow2,pow10;
	char c;
	scanf("%c",&c);
	while(c!='\n'){
		a[n++]=c-'0';
		scanf("%c",&c);
	}
	for(int i=n-1;i>=0;i--){
		pow2=pow10=1;for(int j=0;j<cnt;j++)pow2*=2,pow10*=10;
		pow2*=2;
		//printf("%d %d %d %d %d\n",i,cur,pow2,pow10,cnt);
		if((a[i]*pow10+cur)%pow2==0){
			cnt++;cur+=a[i]*pow10;
			if(cur%8==0){
				printf("YES\n%d",cur);
				return 0;
			}
		}
	}
	printf("NO");
	return 0;
}
