using namespace std;

int main(){
	string n;
	cin >> n;
	n = "0" + n;
	n = "0" + n;
	int num=-1;
	for(int i=0;i<n.size();i++){
		for(int j=i+1;j<n.size();j++){
			for(int k=j+1;k<n.size();k++){
				num = n[i]-'0';
				num*=10; num += (n[j]-'0');
				num*=10; num += (n[k]-'0');
				if(num%8==0){
					cout << "YES" << endl;
					cout << num << endl;
					return 0;
				}
			}
		}
