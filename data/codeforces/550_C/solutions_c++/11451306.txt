#include<iostream>
#include<vector>
using namespace std;

int main(){
  string s; cin>>s;
  vector<int> v(s.size()+2, 0);
  int a;
  for (int i=0; i<s.size(); ++i)
    v[2+i]=s[i]-'0';
  int n=v.size();
  for (int i=0; i<n; ++i)
    for (int j=i+1; j<n; ++j)
      for (int k=j+1; k<n; ++k){
	a=100*v[i]+10*v[j]+v[k];
	if (a%8==0){
	  cout<<"YES\n"<<a<<endl;
	  return 0;
	}
      }
  cout<<"NO\n";
  return 0;
}
