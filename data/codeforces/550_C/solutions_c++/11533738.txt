#include<bits/stdc++.h>
using namespace std;
int main()
{
    char a[1000];
    cin>>a;
    int l=strlen(a),i,j,k;
    for(i=0;i<l;i++) { if(a[i]=='0') { cout<<"YES"<<endl<<'0'<<endl; return 0; }  }
    for(i=8;i<=1000;i=i+8)
    {
        j=l-1;k=i;
        for(;j>=0;j--)
        {
            if(k%10==a[j]-'0') { k/=10;      }
        }
        if(k==0) { cout<<"YES"<<endl<<i<<endl; return 0;  }
    }
    cout<<"NO"<<endl;
    return 0;
}
