#include<stdio.h>
#include<string.h>
char a[1111];
int main(){
	int i,j,k,l,s,flag;
	while(~scanf("%s",a+2)){
		a[0]=a[1]='0';s=0;
		l=strlen(a);flag=0;
		for(i=0;i<l-2;i++){
			if(flag)break;
			for(j=i+1;j<l-1;j++){
				if(flag)break;
				for(k=j+1;k<l;k++){
					s=(a[i]-'0')*100+(a[j]-'0')*10+a[k]-'0';
					if(s%8==0){
						flag=1;
						break;
					}
				}
			}
		}
		if(flag)printf("YES\n%d\n",s);
		else printf("NO\n");
	}
	return 0;
}
