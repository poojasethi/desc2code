#include <iostream>
#include <cstdio>
using namespace std;

int main () {
    string s;
    cin >> s;
    for (int i = 0; i < 1000; i += 8) {
        int x = i;
        for (int j = s.size()-1; j >= 0; j--)
            if (s[j]-'0' == x%10) {
                x /= 10;
                if (x == 0) {
                    cout << "YES\n" << i << "\n";
                    return 0;
                }
            }
    }
    cout << "NO\n";
    return 0;
}
