#include <bits/stdc++.h>
using namespace std;

int main() {
	char D[4], N[101];
	cin >> N;
	for (int i = 0; i < 1000; ++i) {
		if (i % 8) {
			continue;
		}
		sprintf(D, "%d", i);
		int d = 0;
		for (int j = 0; N[j] && D[d]; ++j) {
			if (N[j] == D[d]) {
				++d;
			}
		}
		if (!D[d]) {
			cout << "YES" << endl << D << endl;
			return 0;
		}
	}
	cout << "NO" << endl;
	return 0;
}
