#include<stdio.h>
#include<string.h>
char s[101];
bool test(int n) {
  if(n%8==0) {
    printf("YES\n%d\n",n);
    return true;
  }
  return false;
}
int main() {
  gets(s);
  int n = strlen(s);
  for(int i=0;i<n;i++)s[i]-='0';
  for(int i=0;i<n;i++) {
    if (test(s[i])) return 0;
    for(int j=i+1;j<n;j++) {
      if(test(s[i]*10+s[j])) return 0;
      for(int k=j+1;k<n;k++) {
        if(test(s[i]*100+s[j]*10+s[k])) return 0;
      }
    }
  }
  printf("NO\n");
  return 0;
}
