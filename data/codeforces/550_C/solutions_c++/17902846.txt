#include<bits/stdc++.h>

using namespace std;

int main()
{
    string s;
    cin>>s;
    int a=0;
    s="00"+s;
    int n=s.length();
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
            for(int k=j+1;k<n;k++)
            {
               a=100*(s[i]-'0')+10*(s[j]-'0')+(s[k]-'0');
                if(a%8==0)
                {
                    cout<<"YES"<<endl<<a;
                    return 0;
                }
            }
    cout<<"NO";
}
