#include <iostream>
#include <string> 
#include <set>
#include <algorithm>
using namespace std;

int main(){
	string s;
	cin>>s;
	s = "00" + s;
	for(int i=0; i<(int)s.size(); i++)
		for(int j=i+1; j<(int)s.size(); j++)
			for(int k=j+1; k<(int)s.size(); k++){
				int t = (s[i]-'0')*100 + (s[j]-'0')*10 + s[k]-'0';
				if(t % 8 == 0){
					cout<<"YES\n"<<t<<endl;
					return 0;
				}
			}
	cout<<"NO\n";
	return 0;
}
