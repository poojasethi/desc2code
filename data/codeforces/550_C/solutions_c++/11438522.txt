#include<bits/stdc++.h>
using namespace std;
int main ()
{
	string str;
	int res=-1;
	cin>>str;
	int n,nn ;
	for(int i=0;i<str.size();i++){
		if (str[i]=='8'||str[i]=='0')
			res=str[i]-'0';
		for (int j=i+1;j<str.size();j++){
			n=10*(str[i]-'0')+str[j]-'0';
			if (n%8==0)
				res=n;
			for(int k=j+1;k<str.size();k++){
				 nn =n*10 +str[k]-'0';
				if(nn%8==0)
					res=nn;
			}
		}
	}

	if (res==-1)
		cout<<"NO"<<endl;
	else
		cout<<"YES"<<endl<<res<<endl;
}
