#include<bits/stdc++.h>
using namespace std;

template <class T>
inline string to_string (const T& t)
{
	stringstream ss;
	ss << t;
	return ss.str();
}

int main()
{
	string s, temp;
	cin >> s;

	for (int i = 0; i < 1000; i += 8)
	{
		temp = to_string(i);
		temp += " ";
		int k = 0;
		for (int j = 0; j < s.size(); ++j)
		{
			if(s[j] == temp[k])
				k++; 
				
		}
		if(k + 1 >= temp.size() )
		{
			cout << "YES\n" << temp << "\n";
			return 0;
		}
	}
	cout << "NO\n";
	return 0;
}
