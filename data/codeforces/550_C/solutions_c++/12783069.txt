#include<bits/stdc++.h>
using namespace std;

string s;
int x;

main() {
    ios_base::sync_with_stdio(0);
    getline(cin, s);
    s="00"+s;
    for(int i=0; i<s.length(); i++)
        for(int j=i+1; j<s.length(); j++)
            for(int k=j+1; k<s.length(); k++) {
                x=100*(s[i]-'0')+10*(s[j]-'0')+(s[k]-'0');
                if (x%8==0) {cout << "YES\n" << x << "\n"; return 0;}
            }
    cout << "NO\n";
}
