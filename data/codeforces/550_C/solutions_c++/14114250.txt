#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
    string s; int n=0 ;
    cin>>s;

    s="00"+s;
    int l = s.length();

    for(int i =0 ; i<l ; i++)
        for(int j = i+1 ; j<l ; j++)
         for(int k = j+1 ; k<l ; k++)
    {
        n= 100*(s[i]-48) + 10*(s[j]-48) + (s[k]-48);
        if(n%8==0) {cout<<"YES"<<endl<<n<<endl; return 0;}
    }

    cout<<"NO"<<endl;




    return 0;
}
