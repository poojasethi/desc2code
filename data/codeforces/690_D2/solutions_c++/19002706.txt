#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define INF 0x3f3f3f3f
#define LL long long
#define N 700010
#define MOD 1000003
using namespace std;
LL mul[N], inv[N];
int n, m;
int main(){
	for(int i = mul[0] = 1; i < N; i ++) mul[i] = mul[i - 1] * i % MOD;
	inv[0] = inv[1] = 1;
	for(int i = 2; i < N; i ++) inv[i] = (MOD - MOD / i) * inv[MOD % i] % MOD;
	for(int i = 1; i < N; i ++) inv[i] = inv[i - 1] * inv[i] % MOD;
	
	cin >> n >> m;
	cout << (mul[n + m] * inv[n] % MOD * inv[m] % MOD - 1 + MOD) % MOD << endl;
	return 0;
}
