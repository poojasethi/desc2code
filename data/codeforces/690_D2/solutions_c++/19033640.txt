#include <bits/stdc++.h>
#define mod 1000003

using namespace std;

long long fact[700020];


long long power_up(long long a, long long n){
    if(n==1) return a;
    long long ret = power_up(a,n/2);
    ret = (ret*ret)%mod;
    if(n%2==1) ret = (ret*a)%mod;
    return ret;
}


int main(){
    long long N, C;
    fact[0] = 1;
    for(int i = 1; i < 700010; i++) fact[i]=(fact[i-1]*i)%mod;
    cin >> N >> C;
    long long ans = fact[N+C];
    ans = (ans*power_up(fact[N],mod-2))%mod;
    ans = (ans*power_up(fact[C],mod-2))%mod;
    ans = (ans -1)%mod;
    if(ans < 0) ans += mod;
    cout << ans << endl;
    return 0;
}
