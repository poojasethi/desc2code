#include <bits/stdc++.h>
using namespace std;

const int mo = 1000003;
int n, c, fac[700005], ifc[700005], ans = 0;

long long pwr(long long a, long long b){
	long long res = 1;
	for (; b; b >>= 1, (a *= a) %= mo)
		if (b & 1) (res *= a) %= mo;
	return res;
}

void pre(){
	ifc[0] = fac[0] = 1;
	for (int i = 1; i <= 700000; ++i){
		fac[i] = 1ll * fac[i - 1] * i % mo;
		ifc[i] = pwr(fac[i], mo - 2);
	}
}

long long comb(int a, int b){
	if (a < b) return 0;
	if (a == 0 || b == 0) return 1;
	return 1ll * fac[a] * ifc[b] % mo * ifc[a - b] % mo;
}

int main(){
	scanf("%d%d", &n, &c);
	pre();
	for (int i = 1; i <= n; ++i){
		ans += 1ll * comb(c + i - 1, c - 1) % mo;
		ans %= mo;
		//cout << ans << endl;
	}
	cout << ans << endl;
	return 0;
}
