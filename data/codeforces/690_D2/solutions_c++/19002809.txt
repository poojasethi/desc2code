#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <queue>
const int N = 1000010;
typedef long long ll;
using namespace std;
const ll mod = 1e6+3;
ll h[N+10]; //阶乘表
ll g[N+10]; //阶乘逆元表
ll inv(ll a, ll mod)
{
    ll b=mod-2;
    ll tmp=a;
    a=1;
    while(b>0)
    {
        if(b&1)
        {
            a*=tmp;
            a%=mod;
        }
        tmp*=tmp;
        tmp%=mod;
        b>>=1;
    }
    return a;
}
ll C(int a, int b)
{
	if (a < b) return 0;
	return h[a] * g[b] % mod*g[a - b] % mod;
}
void init()
{
	int i;
	h[0] = 1;
	for (i = 1; i <= N; i++)
		h[i] = h[i - 1] * i%mod;
	for (i = N; i >= 0; i--)
		g[i] = inv(h[i],mod);
}
int main()
{
    init();
    ll n,m;
    scanf("%lld%lld",&n,&m);
    printf("%lld\n",C(n+m,n)-1);
    return 0;
}
