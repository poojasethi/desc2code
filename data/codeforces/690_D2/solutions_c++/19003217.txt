using namespace std;

#define mod 1000003

long long int binpow (long long int a, long long int n)
{
	long long int res = 1;
	while (n) {
		if (n & 1)
			res = (res*a) % mod;
		a = (a*a) % mod;
		n >>= 1;
	}
	return res;
}

int main()
{
    long long int n, c;
    cin >> n >> c;

    long long int C = 1;

    for(int i = n+1; i <= n+c; ++i)
    {
        C = (C*i) % mod;
    }
    for(int i = 2; i <= c; ++i)
    {
        C = (C*binpow(i, mod-2)) % mod;
    }
    C = (C+(mod-1)) % mod;
    cout << C << endl;
    return 0;
}
