using namespace std;
typedef long long LL;
const int N=1e6;
const int MOD = 1e6+3;
LL jc[N],jcn[N];
LL mi(LL x,int n){
	if(n==0) return 1;
	LL tmp = mi(x, n/2);
	tmp = tmp * tmp % MOD;
	if(n % 2) tmp = tmp * x % MOD;
	return tmp;
}
LL c(int n,int m){
	return n<m?0:jc[n] * jcn[m] % MOD * jcn[n-m] % MOD;
}
int main(){
	jc[0] = jcn[0] = 1;
	for(int i=1;i<N;i++){
		jc[i] = jc[i-1] * i % MOD;
		jcn[i] = mi(jc[i], MOD-2);
	}
	int n,m;
	LL ans = 0;
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		//cout<<n<<' '<<i<<endl;
		//cout<<n<<' '<<i<<' '<<c(n,i)<<endl;
		ans=(ans+c(n,i) * c(m,i)) % MOD;
	}
	cout<<ans<<endl;
}
