#include<cstdio>
#define p 1000003
#define LL long long
LL f[1000005];
void init(int n)
{   
    f[0] = 1;
    for(int i=1; i<=n; ++i)
        f[i] = f[i-1] * i % p;
}
LL mypow(LL n,LL k)
{
    LL res=1,muil=n;
    while(k)
    {
        if(k&1)
            res=res*muil%p;
        muil=muil*muil%p;
        k>>=1;
    }
    return res;
}
LL Lucas(LL n, LL k)
{
     LL ret = 1;
     while (n && k)
     {
        LL nn = n % p, kk = k % p;
        if (nn < kk) return 0;
        ret = ret * f[nn] * mypow(f[kk] * f[nn-kk] % p, p - 2) % p;
        n /= p, k /= p;
     }
     return ret;
}
int main()
{
    int n,m,i,j;
    scanf("%d%d",&n,&m);
    init(n+m);
    LL ans=Lucas(n+m,m);
	ans=(ans-1+p)%p;
    printf("%I64d\n",ans);
    return 0;
}
