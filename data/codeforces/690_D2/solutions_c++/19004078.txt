#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000003
#define maxn 700000

ll euclid(ll a)
{
	ll pr = a, r = mod, q, h;
	ll ps = 1, s = 0;
	while(r)
	{
		q = pr/r;
		h = r;
		r = pr - q*r;
		pr = h;
		h = s;
		s = ps - q*s;
		ps = h;
	}
	ps %= mod;
	return (ps + mod) % mod;
}
ll num[maxn], den[maxn];
ll C(ll k, ll n)
{
	return num[n] * den[k] % mod * den[n - k] % mod;
}

int main()
{
	ll n, i, c, ret = 0;
	num[0] = den[0] = 1;
	for(i = 1; i < maxn; i++)
	{
		num[i] = i*num[i - 1] % mod;
		den[i] = euclid(i)*den[i - 1] % mod;
	}
	scanf("%I64d %I64d", &n, &c);
	for(i = 1; i <= n; i++)
		ret = (ret + C(c - 1, i + c - 1)) % mod;
	printf("%I64d", ret);
	return 0;
}
