using namespace std;
typedef long long ll;
const int P = 1000003;
ll a[700001];
ll inv(ll c) {
  int w;
  ll v = 1;
  for (w = P - 2; w; w /= 2) {
    if (w % 2 == 1) {
      v = (c * v) % P;
    }
    c = (c * c) % P;
  }
  return v;
}
