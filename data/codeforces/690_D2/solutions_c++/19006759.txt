#include<bits/stdc++.h>//快速幂 和 组合数学 和 费马小定理 
int mod = 1e6+3;
#define inv(x)  (mypow(x, mod-2))
using namespace std;
typedef long long ll;

ll mypow(ll n, ll t)
{
	ll ans = 1;
	//t = (t+p)%p;
	while(t)
	{
		//printf("%I64d\n", t);
		if(t%2 == 1)
		   ans = ans*n%mod; 
		n = n*n%mod;
		t >>= 1;
	}
	return ans;
}
int main()
{
	int c, n;
	cin >> c >> n;
	ll ans = 1;
	for(int i = n+1; i <= n+c; i++)
    {
    	ans = ans*i*(inv(i-n)+mod)%mod;
    }
    if(ans == 0)
	   ans = 1e6+2;
	else
	   ans--;
	printf("%I64d\n", ans);
	return 0;
}
