#include <cstdio>
#include <cstring>
#include <cmath>
#include <cstdlib>
#include <algorithm>
using namespace std;
typedef long long LL;

const int MaxN = 8e5, Pt = 1e6 + 3;
LL n, c, fac[MaxN + 5];

LL qwr(LL x, int y)
{
	x = x % Pt;
	LL ans = 1;
	while (y != 0) {
		if (y & 1) ans = (ans * x) % Pt;
		y >>= 1;
		x = (x * x) % Pt;
	}
	return ans;
}

int main()
{
	fac[0] = 1;
	for (int i = 1; i <= 8e5; i++) fac[i] = (fac[i - 1] * (LL)i) % Pt;
	while (~scanf("%I64d%I64d", &n, &c)) {
		LL ans = 0;
		for (int i = 1; i <= n; i++) 
			ans = (ans + (fac[i + c - 1] * qwr(fac[i] * fac[c - 1], Pt - 2)) % Pt) % Pt;
		printf("%I64d\n", ans);
	}
}
