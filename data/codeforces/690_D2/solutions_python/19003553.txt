#include<bits/stdc++.h>
#define MOD 1000003

int n, c, cur = 1, ans;

int mod(int x, int y, int z) {
	if (y == 0) return 1;
	if (y % 2 == 0) {
		int zz = mod(x, y / 2, z);
		return (1ll * zz * zz) % z;
	}
	else return (1ll * mod(x, y - 1, z) * x) % z;
}

int inverse(int x) {
	return mod(x, MOD - 2, MOD);
}

int main() {
	scanf("%d %d", &n, &c);
	for (int i = c - 1; i <= n + c - 1; i++) {
		if (i != c - 1) ans = (ans + cur) % MOD;
		cur = (1ll * cur * (i + 1)) % MOD;
		cur = (1ll * cur * inverse(i - c + 2)) % MOD;
	}
	printf("%d\n", ans);
}
