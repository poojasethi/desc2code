#include<stdio.h>
int n, m, ans;
long long fac[700015] = {1};
long long Pow(long long a, long long b)
{
	long long k;
	if(b==0)
		return 1;
	if(b==1)
		return a;
	k = Pow(a, b/2);
	k = k*k%1000003;
	if(b%2==1)
		k = k*a%1000003;
	return k;
}
int C(int x,int y)
{
	return fac[x]*Pow(fac[y], 1000001)%1000003*Pow(fac[x-y], 1000001)%1000003;
}
int main(void)
{
	int i;
	scanf("%d%d", &n, &m);
	for(i=1;i<=n+m-1;i++)
		fac[i]=i*fac[i-1]%1000003;
	for(i=m;i<=n+m-1;i++)
		ans = (ans+C(i,m-1))%1000003;
	printf("%d\n",ans);
	return 0;
}
