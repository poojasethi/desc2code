#include <cstdio>

using namespace std;
typedef long long LL;
typedef unsigned long long ULL;

const LL Pt = 1e6 + 3;
const int MaxN = 8e5;
int n, m;
LL f[MaxN + 5];

LL qwr(LL x, LL y)
{
	x = x % Pt;
	LL ans = 1;
	while(y != 0)
	{
		if(y & 1) ans = (ans * x) % Pt;
		y >>= 1;
		x = (x * x) % Pt;
	}
	return ans;
}

int main()
{
	f[0] = 1;
	for(int i = 1; i <= MaxN; i++) f[i] = (f[i - 1] * (LL)i) % Pt;
	scanf("%d%d", &n, &m);
	LL ans = 0;
	for(int i = 1; i <= n; i++)
	{
		ans = (ans + (f[i + m - 1] * qwr(f[i] * f[m - 1], Pt - 2)) % Pt) % Pt;	
	}
	printf("%I64d\n", ans);
}
