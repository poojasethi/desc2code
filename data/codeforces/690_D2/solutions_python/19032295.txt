#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define maxn 700005
#define ll long long
#define mo 1000003
using namespace std;

ll p[maxn],q[maxn];

int n,c;

ll mul(ll x,ll y){
	ll ret=1;
	while (y) {
		if (y % 2==1) ret=ret * x % mo;
		x=x*x % mo;
		y /= 2;
	}
	return ret;
}

ll C(ll x,ll y){
	return p[x] * q[y] % mo * q[x-y] % mo; 
}

int main(){
	scanf("%d%d",&n,&c);
	p[0]=q[0]=1;
	fo(i,1,n+c) p[i]=p[i-1] * i % mo;
	q[n+c]=mul(p[n+c],mo-2);
	fd(i,n+c-1,1) q[i]=q[i+1]*(i+1) % mo;
	ll ans=0;
	fo(i,1,c) 
		ans=(ans+C(i+n-1,i)) % mo;
	printf("%I64d",ans);
	return 0;
}
