#include <bits/stdc++.h>
using namespace std;

#define MOD 1000003

long long fact[700005], mmi[700005];

long long power(long long N, long long expo)
{
    long long ans = 1;
    
    while(expo)
    {
        if(expo%2)
            ans = ans*N%MOD;
        
        expo/=2;
        N=N*N%MOD;
    }
    
    return ans;
}

long long nCr(long long N, long long R)
{
    return fact[N]*mmi[R]%MOD*mmi[N-R]%MOD;
}

int main() {
	
	long long N, C;
	scanf("%lld %lld", &N, &C);
	
	fact[0] = mmi[0] = 1;
	
	for(int i=1; i<=700000; i++)
	{
	    fact[i] = fact[i-1]*i%MOD;
	    mmi[i] = power(fact[i], MOD-2);
	}
	
	long long ans = 0;
	
	for(int i=N+C-1, j=N; i>=C-1 && j>=1; i--, j--)
	    ans = (ans + nCr(i, C-1))%MOD;
	   
	printf("%lld\n", ans);
	
	return 0;
}
