#include <iostream>
#include<vector>
#include<cstring>
using namespace std;

long long mod = 1000003;

long long pow(long long a, long long b)
{
	long long c = 1;
	while(b)
	{
		if (b % 2) c *= a, c %= mod;
		a *= a;
		a %= mod;
		b /= 2;
	}
	return c;
}

int main()
{
	ios_base::sync_with_stdio(0);
	long long n, c;
	cin >> n >> c;

	long long ans = 1;
	for(int i = 1; i <= n+c; i++)
	{
		ans *= i;
		ans %= mod;
	}

	for (int i = 1; i <= c; i++)
	{
		ans *= pow(i, mod - 2);
		ans %= mod;
	}

	for (int i = 1; i <= n; i++)
	{
		ans *= pow(i, mod - 2);
		ans %= mod;
	}
	cout << (ans + mod - 1) % mod << "\n";

	return 0;
}
