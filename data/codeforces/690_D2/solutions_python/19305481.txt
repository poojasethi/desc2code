# include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll n,c;
const long long N = 1000000;
ll fact[N] , inv_fact[N];
ll mod = 1e6 + 3;


ll C(ll x,ll y)
{
    if (y > x)	return 0;
    ll res = fact[x];
    res = (res * inv_fact[y]) % mod;
    res = (res * inv_fact[x-y]) % mod;
    return res;
}


ll get_mod(ll a,ll b,ll modd)
{
	ll res = 1;
	while (b > 0)
	{
		if (b % 2 == 1)
			res = (res * a) % modd;
		b = b / 2;
		a = (a*a) % modd;
	}
	return res;
}



int main()
{
    fact[0] = fact[1] = inv_fact[0] = inv_fact[1] = 1;

    for (int i=2;i<N;i++)
    {
        fact[i] = (fact[i-1] * i) % mod;
        inv_fact[i] = get_mod(fact[i],mod-2,mod);
    }
	
    
    cin>>n>>c;
    ll res = 0;
    for (ll i = 1 ; i<= n ; i++)
        res = (res + C(i + c - 1 , i))%mod;
    cout<<res<<endl;
    
    return 0;
}
