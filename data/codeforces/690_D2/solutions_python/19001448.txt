#include<bits/stdc++.h>
using namespace std;
#define NN 1000010
typedef long long INT;
INT fac[NN];
const int P=1e6+3;
INT paw(INT a,int k){
	if(!k) return 1;
	INT ans=paw(a,k/2);
	ans=ans*ans%P;
	if(k&1) ans=ans*a%P;
	return ans;
}
int main(){
//	freopen("in.in","r",stdin);
	int n,k,i,j;
	INT jin=1;
	scanf("%d%d",&n,&k);
	fac[0]=1;
	for(i=1;i<NN;i++) fac[i]=fac[i-1]*i%P;
	printf("%I64d\n",(fac[n+k]*paw(fac[n],P-2)%P*paw(fac[k],P-2)%P+P-1)%P);
}
