#include<bits/stdc++.h>
#define LL long long
#define ULL unsigned long long
#define pb push_back
#define pb2 pop_back
#define mp make_pair
#define MOD 1000003
using namespace std;

LL kali(LL a,LL b) {
	return ((a%MOD)*(b%MOD))%MOD;
}

LL pangkat(LL a,LL b) {
	if (b==0) return 1;
	if (b==1) return a;
	if (b==2) return kali(a,a);
	if (b&1) return kali(a,pangkat(a,b-1));
	return pangkat(pangkat(a,b/2),2);
}

LL faktorial(LL a) {
	LL ret=1;
	for (LL i=1;i<=a;i++) ret=kali(ret,i);
	return ret;
}

int main() {
	LL p,q;  scanf("%I64d %I64d",&p,&q);
	LL n=p+q,r=q;
	LL a,b,c; a=faktorial(n); b=faktorial(r); c=faktorial(n-r);
	b=pangkat(b,MOD-2); c=pangkat(c,MOD-2);
	LL ans=kali(kali(a,b),c);
	printf("%I64d\n",ans-1);
	return 0;
}
