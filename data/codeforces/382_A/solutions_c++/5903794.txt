// In the name of God

#include <iostream>

using namespace std;

string s, t;

int main() 
{
	cin >> s >> t;
	int left = s.find("|");
	int right = s.length()-left-1;
	for(int i = 0; i < t.length(); i++) {
		if(left < right) {
			s = t[i] + s;
			left++;
		} else {
			s += t[i];
			right++;
		}
	}
	if(right == left)
		cout << s << endl;
	else
		cout << "Impossible" << endl;
}
