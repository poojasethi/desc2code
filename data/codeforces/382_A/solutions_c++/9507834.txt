#include <iostream>
#include <string>
using namespace std;
int main()
{
    string a, b, x;
    cin >> x;
    a = x.substr(0, x.find('|'));
    b = x.substr(x.find('|')+1, x.length());
    cin >> x;
    for (int i = 0; i < x.length(); i++)
    {if (a.length() < b.length())a += x[i]; else b += x[i];}
    if (a.length() == b.length())cout << a << "|" << b;
    else cout << "Impossible";
}
