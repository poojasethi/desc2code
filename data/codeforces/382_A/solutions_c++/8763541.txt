#include<bits/stdc++.h>
using namespace std;
int main(){
string s,s1,s_left,s_right;
cin>>s>>s1;
int left=0;
while(s[left]!='|'){
left++;
}
s_left=s.substr(0,left);
s_right=s.substr(left+1);
int i=0;
while(s1[i]){
if(s_left.length()<s_right.length())s_left+=s1[i];
else s_right+=s1[i];
i++;

}
if(s_left.length()==s_right.length())cout<<s_left<<"|"<<s_right;
else cout<<"Impossible";

}
