#include <iostream>
#include <algorithm>
#include <stdio.h>

using namespace std;

int main()
{//freopen("i.txt","r",stdin);
string s1,s2; cin>>s1>>s2;
 int pos=s1.find('|');
 int n1=pos,n2=s1.size()-n1-1;
int n=s2.size();
if (n>=abs(n1-n2)&&(n-abs(n1-n2))%2==0)
 {int dl=(n+n1+n2)/2;
  //cout<<dl<<endl;
  cout<<s2.substr(0,dl-n1)<<s1<<s2.substr(dl-n1);
 }
 else cout<<"Impossible";
 return 0;
}
