#include<bits/stdc++.h>
using namespace std;
int main()
{
	int ll,rl,i;
	string s,e;
	cin>>s>>e;
	for(i=0;i<s.length();i++)
	{
		if(s[i]=='|')
		{
			ll=i;
			break;
		}
	}
	rl=s.length()-ll-1;
	for(i=0;i<e.length();i++)
	{
		if(ll<=rl)
		{
			s=e[i]+s;
			ll++;
		}
		else if(rl<ll)
		{
			s=s+e[i];
			rl++;
		}
	}
	if(ll!=rl)
		cout<<"Impossible\n";
	else
		cout<<s<<endl;
	return 0;
}
