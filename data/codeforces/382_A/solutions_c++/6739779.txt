#include <iostream>
#include <algorithm>
using namespace std;
int main()
{string s1,s2; cin>>s1>>s2;
int n1=s1.find('|'),n2=s1.size()-n1-1;
int n=s2.size();
if (n>=abs(n1-n2)&&(n-abs(n1-n2))%2==0)
 {int dl=(n+n1+n2)/2;
  cout<<s2.substr(0,dl-n1)<<s1<<s2.substr(dl-n1);
 }
 else cout<<"Impossible";
 return 0;
}
