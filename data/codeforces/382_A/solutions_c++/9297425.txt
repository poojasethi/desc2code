#include <iostream>

using namespace std;

string s,t;
int i,j;

int main()
{
    cin>>s>>t;
    for(;s[i]!='|';i++);
    i=2*i-s.size()+1;
    for(int j=0;j<t.size();j++)
    {
        if(i>0)
            s.push_back(t[j]),i--;
        else
            s=t[j]+s,i++;
    }
    cout<<(i?"Impossible":s);
    return 0;
}
