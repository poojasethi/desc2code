#include <iostream>
#include <string>

using namespace std;

int main() {

    string str, add;
    int idx;
    cin >> str >> add;

    for(int i = 0; i < (int)add.size(); i++) {
        idx = str.find('|');
        if((int)str.size() - idx - 1 > idx) str = add[i] + str, idx++;
        else str += add[i];
    }

    if((int)str.size() - idx - 1 != idx)
        cout << "Impossible";
    else
        cout << str;

    return 0;
}
