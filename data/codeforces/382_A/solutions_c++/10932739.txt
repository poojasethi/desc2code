#include <bits/stdc++.h>
using namespace std;

int main(){
	string s,S[2]; int c = 0;
	cin >> s;
	for(int i = 0;i < s.length();i++){
		if(s[i] == '|') c = 1;
		else S[c] += s[i];
	}
	cin >> s;
	for(int i = 0;i < s.length();i++)
		S[1 - (S[0].length() < S[1].length())] += s[i];
	if(S[0].length() == S[1].length()) cout << S[0] << "|" << S[1] << "\n";
	else puts("Impossible");
	return 0;
}
