#include <iostream>
#include <string>
using namespace std;

int main()
{
 string s,r;
    cin>>s>>r;
    int x=s.find("|");
    int y=s.size()-(x+1);
    for(int i=0; i<r.length(); i++)
    {

    if(x<y)
    {
        s=r[i]+s;

     x++;

    }
    else
    {
        s=s+r[i];
        y++;
    }



    }




    if(x==y)
        cout<<s;
    else
        cout<<"Impossible";




    return 0;
}
