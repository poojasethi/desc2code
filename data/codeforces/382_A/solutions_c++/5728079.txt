#include <iostream>
#include <algorithm>
using namespace std;
int main ()
{
	int A=0,B=0,C=0;
	string s,t;
	cin>>t;
	int h=0;
	while (t[h]!='|')
	{
		h++;
		A++;
	}
	B=t.size()-A-1;
	cin>>s;
	C=s.size();
	string p,q;
	while (C>0)
	{
		if (A<B)
		{
			p+=s[C-1];
			C--;
			A++;
		}
		else
		{
			q+=s[C-1];
			C--;
			B++;
		}
	}
	if (A!=B)
		cout<<"Impossible";
	else
		cout<<p<<t<<q;
	return 0;
}
