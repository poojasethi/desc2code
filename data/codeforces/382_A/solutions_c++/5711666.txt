#include <iostream>
#include <string>
using namespace std;

string s, t;

int main(){
	cin >> s >> t;
	int n = s.length(), m = t.length(), total = m + n - 1, ind = s.find('|'), l = ind, r = n - ind - 1;
	if(total % 2 != 0 || total/2 < l || total/2 < r)
		cout << "Impossible" << endl;
	else{
		int toL = total/2 - l, toR = total/2 - r;
		s = t.substr(0, toL) + s;
		s = s + t.substr(toL, toR);
		cout << s << endl;
	}
	return 0;
}
