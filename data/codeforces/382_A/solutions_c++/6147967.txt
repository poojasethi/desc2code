#include <bits/stdc++.h>

using namespace std;

int main()
{
	string a , b ;
	cin >> a >> b;
	
	int left = a.find('|') ;
	int right = a.size() - left - 1;
	
	
	for ( int i = 0 ; i < b.size() ; i++ )
	{
			if ( left <= right )
			{
				a.insert ( 0 , 1 , b[i] ) ;
				left++;
			}
			else
			{
				a += b[i] ;
				right++;
			}
	}
	
	
	if ( left == right )
		cout << a << endl ;
	else
		cout << "Impossible" << endl ;

	
	return 0;
}
