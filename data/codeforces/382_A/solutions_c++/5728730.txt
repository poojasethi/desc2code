#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
string a,b,c,d;
bool f;
int main()
{
	cin>>c;
	for(int i=0;i<c.length();i++)
	{
		if(f)b+=c[i];
		if(c[i]=='|')f=1;
		if(!f)a+=c[i];
	}
	cin>>d;
	for(int i=0;i<d.length();i++)
	{
		if(a.length()<=b.length())a+=d[i];else b+=d[i];
	}
	if(a.length()==b.length())
	{
		cout<<a<<'|'<<b<<endl;
	}
	else
	{
		cout<<"Impossible"<<endl;
	}
	return 0;
}
