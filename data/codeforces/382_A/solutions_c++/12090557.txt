#include <iostream>
using namespace std;

string s1, s2;
int main() {
    cin >> s1 >> s2;
    int x = s1.find("|"),
        len1 = x,
        len2 = s1.length() - 1 - x;
    for (int i = 0; i < s2.length(); i++) {
        if (len1 <= len2) s1 = s2[i] + s1, len1++;
        else s1 = s1 + s2[i], len2++;
    }
    cout << (len1 == len2 ? s1 : "Impossible");
}
