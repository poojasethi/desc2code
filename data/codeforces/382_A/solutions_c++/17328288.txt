#include <bits/stdc++.h>
using namespace std;

int main() {
 string s; cin >> s;
 int l, r;
 for(l = 0; l < s.size(); l++) 
 	if(s[l] == '|') break;
	r = s.size() - 1 - l;
	
	string more; cin >> more;
	for (int i = 0; i < more.size(); i++) {
		if(l <= r) s = more[i] + s, l++;
		else s = s + more[i], r++;
	}
	
	if(r == l) cout << s;
	else cout << "Impossible";
}
