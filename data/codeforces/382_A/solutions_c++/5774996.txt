#include <cstdlib>
#include <iostream>
#include <string>
#include <cmath>

using namespace std;

int main(int argc, char *argv[])
{string S,D;
 cin>>S>>D;
 int N=S.size(),M=D.size(),i,L,R;
 for(i=0;i<N;i++)
 if(S[i]=='|')break;
 L=i;R=N-1-i;
 if(abs(R-L)>M || !((N+M)%2))cout<<"Impossible";
 else
  {i=0;
   while(M--)
    if(L<R)S=D[i++]+S,L++;
    else S+=D[i++],R++;
   cout<<S;
  }
    
    return EXIT_SUCCESS;
}
