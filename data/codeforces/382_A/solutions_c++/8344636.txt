#include<bits/stdc++.h>
using namespace std;

int main(){
	string s,t,s1="",s2="";
	cin>>s;
	cin>>t;
	
	bool find=true;
	
	for(int i=0;i<s.size();i++){
		if(s[i]!='|' && find) s1=s1+s[i];
		else if(s[i]=='|') find=false;
		else s2+=s[i];
	}
	
	for(int i=0;i<t.size();i++){
		if(s1.size()>s2.size()) s2=s2+t[i];
		else s1+=t[i];
	}
	
	if(s1.size()!=s2.size()) cout<<"Impossible";
	else cout<<s1<<"|"<<s2;
	
}
