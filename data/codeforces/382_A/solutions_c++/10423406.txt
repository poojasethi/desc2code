#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main() {

  string s, a;
  cin>>s>>a;
  string l = s.substr(0, s.find('|'));
  string r = s.substr(s.find('|') + 1, s.size());

  for(int i = 0; i < a.size(); i++) {
    if(l.size() <= r.size())
      l = l + a[i];
    else
      r = r + a[i];
  }

  if(l.size() == r.size()) {
    cout<<l<<"|"<<r;
    return 0;
  }

  cout<<"Impossible";

  return 0;
}
