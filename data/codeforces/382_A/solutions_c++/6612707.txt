#include<iostream>

using namespace std;


int main()
{

    string a, b; cin >> a >> b;

    for(int i = 0; i< b.size(); i++){

        int idx = a.find_first_of("|");

        if(idx < a.size() / 2)
            a = b[i] + a;
        else
            a += b[i];
    }

    if(a.find_first_of("|") == a.size() / 2)
        cout << a;
    else
        cout << "Impossible";
}
