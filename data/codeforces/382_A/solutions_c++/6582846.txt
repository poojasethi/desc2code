#include<iostream>

using namespace std;


int main(void){

	string w,a; cin >> w >> a;
	
	for(int i = 0; i< a.size(); i++){
	
		int idx = w.find_first_of("|");

		if(idx < w.size() / 2){
			w = a[i] + w;
		}else{
			w += a[i];
		}
	}

	if(w.find_first_of("|") == w.size() / 2){
		cout << w;
	}else{
		cout << "Impossible";
	}
}
