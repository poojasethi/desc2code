#include <iostream>
#include <string>

using namespace std;

long i,l,r;
string s,x;

int main()
{
	cin >> s >> x;
	for (i=0;i<s.size();i++)
		if (s[i]=='|') {
			l=i;
			break;
		}
	r=s.size()-l-1;
	for (i=0;i<x.size();i++)
		if (l<r) {
			l++;
			s=x[i]+s;
		}
		else{
			r++;
			s=s+x[i];
		}
	if (l==r) cout << s;
	else cout << "Impossible";
	return 0;		
}
