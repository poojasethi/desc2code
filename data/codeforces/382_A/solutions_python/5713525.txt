a = raw_input()
b = raw_input()
L, R = a.split("|")
c = len(L)+len(R)+len(b)
if c % 2 == 1:
    print "Impossible"
else:
    equal = c/2
    if equal < len(L) or equal < len(R):
        print "Impossible"
    else:
        add_l = b[:equal-len(L)]
        add_r = b[equal-len(L):]
        print L+add_l+"|"+R+add_r
