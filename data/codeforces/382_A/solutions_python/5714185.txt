import sys
import math

scale, extra = sys.stdin.read().splitlines()

left, right = scale.split('|')
l, r = map(len, [left, right])
e = len(extra)


if ((l + r + e) % 2 != 0) or (math.fabs(l - r) > e):
    print "Impossible"
    exit()




middle = (l + r + e)/2 - l
print left + extra[:middle] + '|' + right + extra[middle:]
