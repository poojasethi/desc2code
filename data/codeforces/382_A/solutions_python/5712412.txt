used = raw_input().strip()
left, right = used.split('|')
not_used = raw_input().strip()



g = len(left) + len(right) + len(not_used)
if (g % 2 == 1):
	print "Impossible"
elif (g/2 < len(left)) or (g/2 < len(right)):
	print "Impossible"
else:
	t = g/2 - len(left) 
	print left + not_used[0:t] + '|' + right + not_used[t:]
