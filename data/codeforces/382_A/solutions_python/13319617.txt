leftWeights, rightWeights = raw_input().split('|')
weight = raw_input()

ll, lr, n = len(leftWeights), len(rightWeights), len(weight)
if n < abs(ll - lr) or (n + lr - ll) % 2 == 1:
	print 'Impossible'
else:
	print leftWeights + weight[:(n + lr - ll) / 2] + '|' + rightWeights + weight[(n + lr - ll) / 2:]
