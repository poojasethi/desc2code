s = raw_input().split('|')
r = abs(len(s[0])-len(s[1]))
o = raw_input()
if len(o)-r<0 or (len(o)-r)%2!=0:
    print 'Impossible'
else:
    if len(s[0])<=len(s[1]):
        print s[0]+o[:r+(len(o)-r)//2]+'|'+s[1]+o[r+(len(o)-r)//2:]
    else:
        print s[0]+o[:(len(o)-r)//2]+'|'+s[1]+o[(len(o)-r)//2:]
