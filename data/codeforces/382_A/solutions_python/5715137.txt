a1,b1 = raw_input().split('|')
a = len(a1)
b = len(b1)
ar = raw_input()
n = len(ar)
ans = ""
if (b-a+n)%2 == 0:
	x = (b-a+n)/2
	if x < 0 or x > n:
		print "Impossible"
	else:
		for i in xrange(x):
			 ans+=ar[i]
		ans+=a1
		ans+='|'
		ans+=b1
		for i in xrange(x,n):
			 ans+=ar[i]
		print ans
else:
	print "Impossible"
