#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int N=10;
int mx_cost,mi_flow;
int flow[N],n;
int L[N][N],H[N][N],C[N][N];
void dfs(int u,int v,int cost)
{
	if(u==n) 
	{
		if(cost>mx_cost) mx_cost=cost;
		return;
	}
	if(v>n)
	{
		if(flow[u]==0) dfs(u+1,u+2,cost);
		return ;
	}
	for(int i=L[u][v];i<=H[u][v];i++)
	{
		if(flow[u]<i) break;
		flow[u]-=i; flow[v]+=i;
		dfs(u,v+1,cost+(i==0?0:C[u][v]+i*i));
		flow[u]+=i; flow[v]-=i;
	}
}
int main()
{
	while(scanf("%d",&n)!=EOF)
	{
		for(int i=0;i<n*(n-1)/2;i++)
			{
				int a,b; scanf("%d%d",&a,&b);
				scanf("%d%d%d",&L[a][b],&H[a][b],&C[a][b]);
			}
		mx_cost=-1;
		for(mi_flow=0;mi_flow<=50;mi_flow++)
		{
			flow[1]=mi_flow;
			dfs(1,2,0);
			if(mx_cost!=-1) break;
		}
		if(mx_cost!=-1) printf("%d %d\n",mi_flow,mx_cost);
		else puts("-1 -1");
	}
	return 0;
}
