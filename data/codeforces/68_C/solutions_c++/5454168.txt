using namespace std;
typedef long long LL;
int l[10][10],h[10][10],a[10][10];
int IN[10],ans,cost,n;

void DFS(int x,int y){
    if (x==n) ans=max(ans,cost);
        else if (y>n) {if(!IN[x]) DFS(x+1,x+2);}
        else
            for (int i=l[x][y];i<=h[x][y];i++)
                if (IN[x]>=i){
                    IN[x]-=i; IN[y]+=i;
                    cost+=(i==0)?0:a[x][y]+i*i;
                    DFS(x,y+1);
                    IN[x]+=i; IN[y]-=i;
                    cost-=(i==0)?0:a[x][y]+i*i;
            }
}

int main(){
    cin>>n;
    int x,y,z1,z2,z3;
    for (int i=0;i<n*(n-1)/2;i++){
        cin>>x>>y>>z1>>z2>>z3;
        l[x][y]=z1; h[x][y]=z2; a[x][y]=z3;
    }
    for (int i=0;i<=30;i++){
        IN[1]=i; ans=-1; cost=0;
        DFS(1,2);
        if (ans>=0) {
            cout<<i<<" "<<ans<<endl; return 0;
        }
    }
    cout<<"-1 -1"<<endl;
    return 0;
}
