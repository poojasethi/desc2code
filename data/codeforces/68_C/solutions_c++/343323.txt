#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int MN=10;
const int INF=(1u<<31)-1;
typedef pair<int,int> pie;
pie list[MN][MN];
int cost[MN][MN];
int n;
int best1=INF,best2=-1;
int a[MN];
void back_track(int,int);
void find(int &b,int c,int p,int remain)
{
	if (p==n)
	{
		if (remain==0)
			back_track(b+1,c);
		return ;
	}
	for (int i=list[b][p].first;i<=min(remain,list[b][p].second);i++)
	{
		int t2=c;
		if (i>0)
			t2+=cost[b][p];
		t2+=i*i;
		a[p]+=i;
		find(b,t2,p+1,remain-i);
		a[p]-=i;
	}
}
void back_track(int b,int c)
{
	if (b==n-1)
	{
		if (a[n-1]==a[0])
		{
			if (a[n-1]<best1)
			{
				best1=a[n-1];
				best2=c;
			}
			else if (a[n-1]==best1 && c>best2)
				best2=c;
		}
		return ;
	}
	find(b,c,b+1,a[b]);
}
int main()
{
	ios::sync_with_stdio(false);
	cin>>n;
	for (int i=0;i<n*(n-1)/2;i++)
	{
		int a,b,c,d,e;
		cin>>a>>b>>c>>d>>e;
		a--; b--;
		cost[a][b]=e;
		list[a][b]=pie(c,d);
	}
	for (int i=0;i<=30;i++)
	{
		memset(a,0,sizeof(a));
		a[0]=i;
		back_track(0,0);
		if (best1==i)
			break;
	}
	if (best1==INF)
		cout<<-1<<" "<<-1<<endl;
	else
		cout<<best1<<" "<<best2<<endl;
	return 0;
}
