#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 10
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,x,y,z,A,B,an;
int l[N][N],r[N][N],a[N][N],f[N];
void dfs(int x,int y,int z,int s) {
	if (x==n) {
		if (s>an) an=s;
		return;
	}
	if (y>n) {
		if (!z) dfs(x+1,x+2,f[x+1],s);
		return;
	}
	int i;
	For(i,l[x][y],min(z,r[x][y])) {
		f[y]+=i;
		dfs(x,y+1,z-i,s+(i?a[x][y]:0)+i*i);
		f[y]-=i;
	}
}
int main() {
	scanf("%d",&n);
	For(i,1,n*(n-1)/2) {
		scanf("%d%d%d%d%d",&x,&y,&z,&A,&B);
		l[x][y]=z,r[x][y]=A,a[x][y]=B;
	}
	For(i,0,25) {
		an=-1;
		dfs(1,2,i,0);
		if (an>=0) {
			printf("%d %d\n",i,an);
			return 0;
		}
	}
	printf("-1 -1\n");
	return 0;
}
