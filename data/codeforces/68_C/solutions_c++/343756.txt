#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
int low[10][10];
int up[10][10];
int n;
int in[10],out[10];
int ca[10][10];
int minf;
int minc;
void flow(int st,int ed,int cost){
    if(st==n){
        if(out[1]<minf){
            minf=out[1];
            minc=cost;        
        }
        else if(out[1]==minf){
            minc=max(minc,cost);
        }
        return ;
    }
    else if(ed==n+1){
        if(st!=1&&in[st]!=out[st])return;
        if(st==1&&out[st]>minf)return ;
        flow(st+1,st+2,cost); 
        return ;       
    }
    for(int i=low[st][ed];i<=up[st][ed];i++){
        if(st==1&&out[st]>minf)break;
        if(st!=1&&in[st]<out[st]+i)break;
        out[st]+=i;
        in[ed]+=i;
        int z=i*i;
        if(i>0)z+=ca[st][ed];
        flow(st,ed+1,cost+z);   
        out[st]-=i;
        in[ed]-=i; 
    }
}

int main(){
    int st,ed;
    minf=100000;
    cin>>n;
    for(int i=1;i<=n-1;i++){
        for(int j=i+1;j<=n;j++){
            cin>>st>>ed;
            cin>>low[st][ed]>>up[st][ed]>>ca[st][ed];    
        }        
    }   
    flow(1,2,0);
    if(minf>=100000){
            cout<<-1<<" "<<-1<<endl;
    }
    else cout<<minf<<" "<<minc<<endl;
    return 0; 
}
