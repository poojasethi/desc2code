#include<stdio.h>
int n,minf,maxf,ans;
int l[10][10],r[10][10],cost[10][10],f[10];
void dfs(int s,int t,int flow,int sum)
{
  int i,j,k;
  if (s==n)
    {
      if (ans<sum) ans=sum;
      return;
    }
  if (t>n) 
    {
      if (flow==0) dfs(s+1,s+2,f[s+1],sum);
      return;
    }
  for (i=l[s][t];i<=r[s][t];i++)
    {
      if (flow>=i)
        {
          f[t]+=i;
          dfs(s,t+1,flow-i,sum+i*i+(i?cost[s][t]:0));
          f[t]-=i;
        }
    }
}
int main()
{
  int i,a,b,j,k,c;
  scanf("%d",&n);
  for (i=1;i<=(n-1)*n/2;i++)
    {
      scanf("%d%d%d%d%d",&a,&b,&j,&k,&c);
      l[a][b]=j;r[a][b]=k;cost[a][b]=c;
      if (a==1)
        {
          minf+=l[a][b];
          maxf+=r[a][b];
        }
    }
  ans=-1;
  for (i=minf;i<=maxf;i++)
    {
      f[1]=i;
      dfs(1,2,i,0);
      if (ans>=0) break;
    }
  if (ans>=0) printf("%d %d\n",i,ans);
  else printf("-1 -1\n");
  return 0;
}
