#include <cstdio>
#include <cstdlib>
using namespace std;

struct re{
    int l,h,a;
}a[11][11];

int n,m,ans,flag,s,f,c,l,h;
int Max[11],Min[11],in[11];

void add(int x,int y,int l,int h,int c)
{
     a[x][y]=(re){l,h,c};
     Min[x]+=l; Max[x]+=h;
}

void dfs(int u,int v);

void hf(int t,int u,int v,int res)
{
     if (t>n)
     {
         if (res!=0) return;
         dfs(u+1,v);
         return;
     }
     for (int i=a[u][t].l;i<=a[u][t].h;i++) 
     {
         in[t]+=i; if (in[t]>Max[t]) { in[t]-=i; break; } 
         hf(t+1,u,v+i*i+(i!=0?a[u][t].a:0),res-i);
         in[t]-=i;
     }
}

void dfs(int u,int v)
{
    if (in[u]<Min[u]||in[u]>Max[u]) return;
    if (u==n)
    {
        flag=1;
        if (v>ans) ans=v;
        return;
    }
    hf(u+1,u,v,in[u]);
}

int main()
{
    scanf("%d",&n); m=n*(n-1)/2;
    for (int i=1;i<=m;i++)
    {
        scanf("%d%d%d%d%d",&s,&f,&l,&h,&c);
        add(s,f,l,h,c);
    }
    Min[n]=0; Max[n]=m*6;
    for (int i=Min[1];i<=Max[1];i++) 
    {
        in[1]=i; dfs(1,0);
        if (flag) { printf("%d %d\n",i,ans); return 0; }
    }
    printf("-1 -1\n");
}
