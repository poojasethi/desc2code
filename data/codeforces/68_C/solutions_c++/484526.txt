#include<cstdio>

int B[15][15],C[15][15],A[15][15];
int S[15];
int n;
int r1,r2;

bool dfs(int u,int cost)
{
    if (u==n)
    {
        if (cost>r2) r2=cost;
        return 1;
    }
    bool ok=0;

    for (int a=B[u][u+1];a<=C[u][u+1] && a<=S[u];a++)
    for (int b=B[u][u+2];b<=C[u][u+2] && a+b<=S[u];b++)
    for (int c=B[u][u+3];c<=C[u][u+3] && a+b+c<=S[u];c++)
    for (int d=B[u][u+4];d<=C[u][u+4] && a+b+c+d<=S[u];d++)
    for (int e=B[u][u+5];e<=C[u][u+5] && a+b+c+d+e<=S[u];e++)
    if (a+b+c+d+e==S[u])
    {
        S[u+1]+=a;S[u+2]+=b;S[u+3]+=c;S[u+4]+=d;S[u+5]+=e;
        ok|=dfs(u+1,cost+a*a+b*b+c*c+d*d+e*e+(a>0)*A[u][u+1]+(b>0)*A[u][u+2]+(c>0)*A[u][u+3]+(d>0)*A[u][u+4]+(e>0)*A[u][u+5]);
        S[u+1]-=a;S[u+2]-=b;S[u+3]-=c;S[u+4]-=d;S[u+5]-=e;
    }
    return ok;
}

int main()
{
    scanf("%d",&n);
    r1=-1,r2=-1;
    for (int i=0;i<n*(n-1)/2;i++)
    {
        int a,b;scanf("%d%d",&a,&b);
        scanf("%d%d%d",&B[a][b],&C[a][b],&A[a][b]);
    }
    for (int i=0;i<=(n-1)*5;i++)
    {
        S[1]=i;
        if (dfs(1,0))
        {
            r1=i;break;
        }
    }
    printf("%d %d\n",r1,r2);
}
