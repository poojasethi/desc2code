#include <cstdio>
#include <cstring>

const int maxn=100000;

int n,u,v,a[10][10],b[10][10],g[10][10][10],f[10],ans1,ans2,x;

void dfs(int x,int y,int s)
{
    if(y>n)
    {
        if(x==n)
        {
            if(f[n]<ans1)ans1=f[n],ans2=s;
            else if(f[n]==ans1 && s>ans2)ans2=s;
            return;
        }
        if(f[x])return;
        dfs(x+1,x+1,s);
        return;
    }
    for(int i=a[x][y]; i<=b[x][y]; i++)
    {
        f[x]-=i;
        f[y]+=i;
        dfs(x,y+1,s+g[x][y][i]);
        f[x]+=i;
        f[y]-=i;
    }
}
int main()
{
    ans1=maxn;
    ans2=0;
    scanf("%d",&n);
    memset(a,0,sizeof(a));
    memset(b,0,sizeof(b));
    memset(g,0,sizeof(g));
    for(int i=0; i<n; i++)
        for(int j=0; j<i; j++)
        {
            scanf("%d%d",&u,&v);
            scanf("%d%d%d",&a[u][v],&b[u][v],&g[u][v][0]);
        }
    for(int i=1; i<=n; i++)
        for(int j=i+1; j<=n; j++)
        {
            for(int k=0; k<10; k++)
                g[i][j][k]=g[i][j][0]+k*k;
            g[i][j][0]=0;
        }
    memset(f,0,sizeof(f));
    for(int i=0;i<=25 && ans1==maxn;i++)
    {
        f[1]=i;
        dfs(1,1,0);
    }
    if(ans1==maxn)puts("-1 -1");
    else printf("%d %d\n",ans1,ans2);
    return 0;
}
