def count(s):
	index = len(s)
	ans = 0
	while(index > 1):
		a, b = int(s[index - 1:index]), int(s[index - 2:index])
		if(a % 4 == 0):
			if(b % 4 == 0):
				ans += index
			else:
				ans += 1
		else:
			if(b % 4 == 0):
				ans += index - 1
		index -= 1
		# print("index = %d, lastdigit = %d, last2digit = %d, ans = %d"%(index, a, b, ans))
	if(int(s[0]) % 4 == 0):
		ans += 1
	return ans
