#!/usr/bin/env python

s = map(int, list(raw_input()))

size = len(s)

r = 0
i = size - 2

while i >= 0:
    # consider s[i]s[i + 1]
    if (s[i] % 2 == 1 and s[i + 1] % 2 == 0 and s[i + 1] % 4 != 0) or \
            (s[i] % 2 == 0 and s[i + 1] % 4 == 0):
        # ok, here we got %4 == 0
        r += (i + 1)
    
    i -= 1
for v in s:
    if v % 4 == 0:
        r += 1
print r
