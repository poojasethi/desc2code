def div4(s):
	n=len(s)
	dp=[0 for i in range(n+1)]
	if int(s[0])%4==0:
		dp[1]=1
	else:
		dp[1]=0
	i=2
	while i<=n:
		dp[i]=dp[i-1]
		num=10*int(s[i-2])+int(s[i-1])
		if num%4==0:
			dp[i]+=i-1
		if int(s[i-1])%4==0:
			dp[i]+=1
		i+=1
	return dp[n]

	s=raw_input()
	print div4(s)
if __name__=='__main__':
	main()
