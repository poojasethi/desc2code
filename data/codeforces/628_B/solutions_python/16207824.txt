import sys
import copy

s = sys.stdin.readline().strip()

dp = {}
ans = 0
dp[int(s[0])%4] = 1
ans += dp.get(0, 0)
for i in s[1:]:
    dp2 = {}
    num = int(i) % 4
    dp2[num] = dp2.get(num, 0) + 1
    for j in dp.keys():
        v = dp[j]
        k = (j * 10 + num) % 4
        dp2[k] = dp2.get(k, 0) + v
    dp = dp2
    #print i, dp
    ans += dp.get(0, 0)
print ans
