_input = sys.stdin.readline().strip()
n = ord(_input[0]) - ord('0')
count = 0

    count = 1

for i in range(1, len(_input)):

    if n % 4 == 0:
        count = count + 1
    n = n + ((_ord(_input[i-1]) - _ord('0')) * 10)
