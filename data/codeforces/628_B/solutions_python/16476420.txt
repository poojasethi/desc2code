s = raw_input()
l = len(s)


def to_int(s):
    return ord(s) - ord('0')


ans = 0
for i in xrange(1, l):
    remainder = (to_int(s[i - 1]) * 10 + to_int(s[i]))
    if remainder % 4 == 0:
        ans += i

for i in xrange(l):
    if to_int(s[i]) % 4 == 0:
        ans += 1

print ans
