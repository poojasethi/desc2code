def main():
    s = raw_input()
    last = ord(s[0]) - ord('0')
    n = len(s)
    i = 1
    ans = 0L if last % 4 != 0 else 1L
    for ch in s[1:]:
        cur = ord(ch) - ord('0')
        if cur % 4 == 0:
            ans += 1
        temp = 10 * last + cur
        if temp % 4 == 0:
            ans += i
        last = cur
        i += 1
    print ans

if __name__ == '__main__':
    main()
