#include <bits/stdc++.h>
using namespace std;
string s;
long long ans;
int main(){
    cin >> s;
    for(int i=0;i<s.size();i++){
        int cal = s[i] - '0';
        if (cal%4==0)   ans ++;
        if (i == 0) continue;
        cal += 10*(s[i-1] - '0');
        if (cal%4==0) ans += i;
    }
    cout << ans;
}
