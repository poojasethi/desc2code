#include <bits/stdc++.h>
using namespace std;
int main()
{
	string s;
	char a[300000];
	__int64 ans=0;
	scanf("%s",&a);
	s=a;
	for (int i=1;i<s.length();i++)
	{
		int q=(s[i-1]-48)*10+s[i]-48;
		if (q%4==0) ans+=i;
		if ((s[i]-48)%4==0) ans++;
	}
	if ((s[0]-48)%4==0) ans++;
	cout<<ans;
	return 0;
}
