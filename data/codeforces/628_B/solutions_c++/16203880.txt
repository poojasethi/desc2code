#include<iostream>
#include<string>
#include<cstdio>
using namespace std;

string s;

int main(){
	cin>>s;
	long long ans=0;
	for(int i=s.size()-1;i>=0;i--){
		if((s[i]-'0')%4==0) ans++;
		if(i>0){
		int t=(s[i-1]-'0')*10+(s[i]-'0');
		if(t%4==0) ans+=i;
	    }
	}
	cout<<ans<<endl;
//	printf("%ll",ans);
}
