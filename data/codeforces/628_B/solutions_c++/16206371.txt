#include <bits/stdc++.h>
using namespace std;

char s[300001];
int main() {
	scanf("%s", s);
	int n = strlen(s);
	long long c = 0;
	for(int i=0; i<n; ++i)
		c += (((s[i]-'0')&3)==0);
	for(int i=1; i<n; ++i)
		c += i*((((s[i]-'0')&1)==0) && ((((s[i]-'0')>>1)&1) == ((s[i-1]-'0')&1)));
	printf("%I64d", c);
	return 0;
}
