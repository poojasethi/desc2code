#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

char s[3000006];

int main()
{
	scanf ("%s", s);
	ll c =0;
	ll x = strlen(s);
	for (ll i = 0; i < x; i++) {
		if ( (s[i] - '0')% 4 == 0) c++;
		if (i > 0) {
			ll x = ((s[i-1]-'0')*10 + s[i]-'0')%4;
			if (x == 0)
				c += i;
		}
	}
	cout <<c <<endl;
	return 0;
}
