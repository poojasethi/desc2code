#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s;cin>>s;
    int n=s.size();
    long long cnt=0;
    for(int i=0;i<n;i++)
        if((s[i]-'0')%4==0)
            cnt++;
    for(int i=0;i<n-1;i++)
        if(((s[i]-'0')*10+(s[i+1]-'0'))%4==0)
            cnt+=i+1;
    cout<<cnt;
}
