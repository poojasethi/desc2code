#include<cstdio>
#include<iostream>
long long int ans=0;
char s[300005];
long int i;
using namespace std;
int main()
{
	gets(s);
	i=0;
	while (s[i]!='\0')
	 {
	 	if ((s[i]-'0')%4==0)
	 	 {
	 	  	ans++;
		 }
		if (i>0)
	 	  	if (((s[i-1]-'0')*10+(s[i]-'0'))%4==0)
	 	  			ans+=i;
	 i++;
	 }
	cout << ans;
}
