#include <stdio.h>
#include <string.h>
char s[300100];
__int64 sum=0;
int main()
{
	int i;
	gets(s);
	for(i=strlen(s)-1;i>0;i--)
	{
		if(((s[i]-'0')+(s[i-1]-'0')*10)%4==0)
			sum += i;
	}
	for(i=0;i<strlen(s);i++)
		if((s[i]-'0')%4==0)	
			sum++;
	printf("%I64d\n", sum);
	return 0;
}
