#include <stdio.h>

char s[300005];
long long ans;

int main(){
    int len;
    scanf("%s", s);
    for(len=0; s[len]!='\0'; len++){
        s[len] -= '0';
        if(s[len]%4 == 0) ans++;
    }
    for(int i=0; i<len-1; i++){
        if((10*s[i]+s[i+1])%4 == 0) ans += i+1;
    }
    printf("%I64d", ans);
    return 0;
}
