#include <iostream>
#include <cstdio>

const int MAX = 3e5 + 5;
char arr[MAX];
long long res;

int main()
{
	scanf("%s", arr + 1);
	for (int i = 1; arr[i]; ++i)
	{
		if (arr[i] % 4 == 0)
			res++;
		if (i > 1 && (arr[i] + arr[i - 1] * 10) % 4 == 0)
			res += i - 1;
	}
	printf("%I64d\n", res);

	return 0;
}
