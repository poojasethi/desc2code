#include <bits/stdc++.h>
using namespace std;
string s;
long long k,n,t,p,m;
int main () {
	cin>>s;
	n=s.length();
	for (int i=n-2;i>=0;i--){
	  t=s[i]-'0';
	  t*=10;
	  p=s[i+1]-'0';
	  if (p%4==0)
	    k++;
	  m=p+t;
	  if (m%4==0)
	    k+=i+1;	
	}
	p=s[0]-'0';
	if (p%4==0)
	  k++;
	cout<<k;  
	return 0;
}
