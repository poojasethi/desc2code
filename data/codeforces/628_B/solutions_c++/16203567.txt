#include <bits/stdc++.h>
using namespace std;

#define mp(x, y) make_pair((x), (y))

typedef long long ll;

char s[300005];
ll ans;

int main()
{
	scanf("%s", s);
	for(int i=0; s[i]; i++) {
		int d=s[i]-'0';
		if(d%4==0) ans++;
		if(i>0) {
			d+=10*(s[i-1]-'0');
			if(d%4==0) ans+=i;
		}
	}
	printf("%I64d\n", ans);

	return 0;
}
