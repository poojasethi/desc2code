#include<stdio.h>
#include<string.h>
int main()
{
	char a[300005];
	int i,j;
	gets(a);
	__int64 sum=0;
	int k,m,n;
	n=a[0]-'0';
	if(n%4==0)sum++;
	for(i=1;i<strlen(a);i++)
	{
		k=a[i]-'0';
		m=a[i-1]-'0';
		if((k+m*10)%4==0)sum+=i;
		if(k%4==0)sum++;
	}
	printf("%I64d\n",sum);
	return 0;
}
