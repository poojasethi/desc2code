#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    cin>>s;
    long long int ans=0;

    for(int i=0;s[i];i++){
        int curr=s[i]-'0';
        if(curr%4==0) ans++;
        if(i){
            int prev=s[i-1]-'0';
            if((prev*10+curr)%4==0) ans=ans+i;}
    }
    cout<<ans;
}
