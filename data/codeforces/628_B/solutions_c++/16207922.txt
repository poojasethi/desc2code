#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;

char c[300010];

int main()
{
	int i,j,l;
	long long s;
	scanf("%s",c);
	l=strlen(c);
	
	s=0;
	for (i=0;i<l;i++)
	{
		if (!(c[i]%4))
			s++;
		if (i && !((c[i]+c[i-1]*10)%4))
			s+=i;
	}
	
	printf("%I64d\n",s);
	return 0;
}
