#include<bits/stdc++.h>
using namespace std;
char c[300005];
long long int res = 0, p = 0;
int main()
{
	scanf("%s", &c);
	while (c[p] != '\0')
	{
		if ((c[p] - '0') % 4 == 0)
			res++;
		if (p > 0 && ((c[p - 1] - '0') * 10 + (c[p] - '0')) % 4 == 0)
			res += p;
		p++;
	}
	printf("%I64d\n", res);
	return 0;
}
