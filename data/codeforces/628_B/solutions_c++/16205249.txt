#include <bits/stdc++.h>
using namespace std;
int main(){
    ios_base::sync_with_stdio(0);
    string a;
    cin>>a;
    long long wyn = 0;
    for(int i = 0; i < a.size(); i++){
        if(a[i]%4 == 0) wyn ++;
        if(i > 0 && ((a[i-1] - '0') * 10 + a[i] - '0') % 4 == 0) wyn += i;
    }
    cout << wyn;
}
