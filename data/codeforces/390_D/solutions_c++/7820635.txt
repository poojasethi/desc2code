#include <bits/stdc++.h>

#define FI(ii,aa,bb) for(int ii=aa;ii<=bb;ii++)
#define F(ii,aa,bb) for(int ii=aa;ii<bb;ii++)
#define TF(ii,aa,bb) for(int ii=aa;ii>=bb;ii--)
#define mp make_pair
#define pb push_back
#define st first
#define nd second

using namespace std;

typedef long long ll;
typedef pair<int,int> pii;

const int inf = 1000000000;

int n,m,k;

void print(int x,int y){
	FI(i,1,x) printf("(%d,1) ",i);
	FI(i,2,y) printf("(%d,%d) ",x,i);
}

int main()
{
	scanf("%d %d %d",&n,&m,&k);
	vector<pii> v;
	int ans=0;
	FI(i,1,n){
		if(!k) break;
		int x=i,y=1;
		while(x>=1 && y<=m && k>0){
			k--;
			v.pb( mp(x,y) )	;
			ans += x+y-1;
			x--;
			y++;
		}
	}
	FI(i,2,m){
		if(!k) break;
		int x = n, y = i;
		while(x>=1 && y<=m && k>0){
			k--;
			v.pb( mp(x,y) )	;
			ans += x+y-1;
			x--;
			y++;
		}
	}
	cout << ans << endl;
	TF(i,v.size()-1,0){
		print(v[i].st,v[i].nd);
		puts("");
	}


	return 0;
}
