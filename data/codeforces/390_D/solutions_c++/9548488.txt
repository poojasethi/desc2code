#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<vector>
#define mem(name,value) memset(name,value,sizeof(name))
#define FOR(i,n) for(int i=1;i<=n;i++)
using namespace std;
const int maxn = 2500+10;
struct Point{
    int x,y,d;
    bool operator < (const Point& rhs)const{
        return d < rhs.d;
    }
    void print(){
        for(int i=1;i<=x;i++) printf("(%d,%d) ",i,1);
        for(int j=2;j<=y;j++) printf("(%d,%d) ",x,j);
        printf("\n");
    }
}p[maxn];
int main(){
    int n,m,k,cnt;
    while(~scanf("%d%d%d",&n,&m,&k)){
        cnt = 0;
        for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++){
            p[cnt++] = (Point){i,j,(i-1)+(j-1)+1};
        }
        int ans = 0;
        sort(p,p+cnt);
        for(int i=0;i<k;i++) ans += p[i].d;
        printf("%d\n",ans);
        for(int i=k-1;i>=0;i--) p[i].print();
    }
    return 0;
}
