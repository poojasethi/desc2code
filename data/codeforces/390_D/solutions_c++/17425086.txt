
#include <bits/stdc++.h>
#define N 2550

using namespace::std;

int A[N][N];

int X[N], Y[N];

int m, n, k, s = 0, c = 0;

int main ( void )
{
	bool f = true;

	cin >> n >> m >> k;
	
	int aux = k;

	for ( int i = 0; f && i < m + n - 1; i++ )
	{
		for ( int j = 0; j < n && f; j++ )
		{
			for ( int l = 0; l < m && f; l++ )
			{
				if ( j + l == i )
				{
					X[c] = j;
					Y[c] = l;				
					c++;
					aux--;
					s += j + l + 1;
					if ( aux == 0 )
						f = false;
				}
			}
		}
	}
					
	cout << s << endl;

	int x, y;

	for ( int i = k - 1; i >= 0; i-- )
	{
		x = 0; y = 0;

		while ( x != X[i] )
		{
			cout << "(" << x + 1 << "," << y + 1 <<  ") ";
			x++;
		}

		while ( y != Y[i] )
		{
			cout << "(" << x + 1 << "," << y + 1 <<  ") ";
			y++;
		}		

		cout << "(" << x + 1 << "," << y + 1 << ")" << endl;
	}

	return 0;
}
