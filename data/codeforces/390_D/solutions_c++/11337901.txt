#include<iostream>
#include<algorithm>
#include<string>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
using namespace std;
#define ll __int64
struct tt
{
	int x,y;
}a[3000];
int cmp(tt x,tt y)
{
	return (x.x+x.y)<(y.x+y.y);
}
int main(){
	int n,m,k;
	cin>>n>>m>>k;
	int id=0;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
			a[id].x=i,a[id].y=j,id++;
	int nn=n*m;
	sort(a,a+nn,cmp);

	int ans=0;
	for(int i=0;i<k;i++)
	{
		ans+=(a[i].x+a[i].y-1);
	}

	cout <<ans<<endl;
	for(int i=k-1;i>=0;i--)
	{
		int xx=1,yy=1,yi=0;
		while(xx<=a[i].x&&yy<=a[i].y)
		{
			if(yi)printf(" ");yi=1;
			printf("(%d,%d)",xx,yy);
			if(xx<a[i].x)xx++;
			else if(yy<a[i].y)yy++;
			else if(xx==a[i].x&&yy==a[i].y)break;
		}
		puts("");
	}
	return 0;
}
