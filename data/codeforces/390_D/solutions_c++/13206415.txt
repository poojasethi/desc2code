
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN = 60 * 60;

pii h[MAXN];

bool byDis(pii x, pii y){
	return (x.first + x.second < y.first + y.second);
}

int main(){
	ios::sync_with_stdio( false );
	int n, m, k;
	cin >> n >> m >> k;

	int sz = 0;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			h[sz++] = pii(i + 1, j + 1);

	sort(h, h + sz, byDis);
	ll sum = 0;
	for (int i = 0; i < k; i++)	
		sum += h[i].first + h[i].second - 1;

	cout << sum << endl;
	for (int i = k -1; i >= 0; i--){
		for (int x = 1; x <= h[i].first; x++)
			cout << "(" << x << ", 1) ";
		for (int y = 2; y <= h[i].second; y++)
			cout << "(" << h[i].first << ", " << y << ") ";
		cout << "\n";
	}



			
}
