#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
typedef pair<int, int> ii;

ll a[100010], b[100010];

vector<pair<int, int> > v;

struct par{
	int x, y;
	par(){}
	par(int a, int b){x=a; y=b;}
	bool operator<(par B)const{
		return (x+y)<(B.x+B.y);
	}
};

vector<par> pares;
vector<ii> resp[2555];

int main(){
	int n, m, k;
	scanf("%d %d %d", &n, &m ,&k);
	for(int i=0;i<n;i++)for(int j=0;j<m;j++){
		pares.push_back(par(i, j));
	}
	sort(pares.begin(), pares.end());
	int j=0, total=0;
	for(int i=k-1;i>=0;i--){
		int nx=0, ny=0;
		resp[j].push_back(ii(0, 0));
		total++;
		for(nx=1;nx<=pares[i].x;nx++){
			resp[j].push_back(ii(nx, ny));
			total++;
		}
		nx--;
		for(ny=1;ny<=pares[i].y;ny++){
			resp[j].push_back(ii(nx, ny));
			total++;
		}
		j++;
	}
	printf("%d\n", total);
	for(int i=0;i<j;i++){
		for(int k=0;k<(int)resp[i].size();k++){
			printf("(%d,%d)%c", resp[i][k].first+1, resp[i][k].second+1, " \n"[k==(int)resp[i].size()-1]);
		}
	}
	return 0;
}
