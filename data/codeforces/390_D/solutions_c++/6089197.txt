#include"stdio.h"
int map[55][55],m,n,k,ans;
void draw(int p){
	int i,j,t;
	ans=0;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			map[i][j]=0;
	for(i=0;;i++){
		for(j=0,t=i;t>=0;j++,t--)
			if(t<n&&j<m&&p>0){
				map[t][j]=1;
				p--;
				ans+=j+t+1;
			}
		if(p==0)break;
	}
}

void road(int x,int y){
	int i,j;
	for(i=1;i<=x;i++)printf("(%d,1) ",i);
	for(j=2;j<=y;j++)printf("(%d,%d) ",x,j);
	printf("\n");
}

void findpoint(){
	int i,j;
	ans=0;
	for(i=n-1;i>=0;i--)
		for(j=m-1;j>=0;j--)
			if(map[i][j])road(i+1,j+1);
}

main(){
	while(~scanf("%d%d%d",&n,&m,&k)){
		draw(k);
		printf("%d\n",ans);
		findpoint();
	}
}
