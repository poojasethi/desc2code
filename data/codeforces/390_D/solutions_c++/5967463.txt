#include <cstdio>
#include <algorithm>

using namespace std;

struct ty1{
	int x,y;
}a[60*60];

int n,m,k;


bool cmp(ty1 a, ty1 b) {
	return ( a.x+a.y < b.x+b.y);
}

void init() {
	scanf("%d%d%d",&n,&m,&k);

	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= m; j++) {
			a[(i-1)*m+j].x = i;
			a[(i-1)*m+j].y = j;
		}

	sort(a+1,a+n*m+1,cmp);
}

void solve() {
	int ans = 0;
	for (int i = 1; i <= k; i++)
		ans += a[i].x+a[i].y-1;
	printf("%d\n",ans);

	for (int i = k; i >= 0; i--) {
		for (int j = 1; j <= a[i].y; j++)  printf("(1,%d) ",j);

		for (int j = 2; j <= a[i].x; j++)  printf("(%d,%d) ",j,a[i].y);

		printf("\n");
	}
}

int main() {
//	freopen("in.txt","r",stdin);
//	freopen("out.txt","w",stdout);

	init();
	solve();
}
