#include<stdio.h>

#define min(a,b) ((a)<(b)? (a) : (b))

int policka[50*50+10][2];

int main() {
  int m,n,k,p=0;
  scanf("%d%d%d", &n,&m,&k);
  int x=1,y=1,l=1;
  for (int i=0; i<k; i++) {
    policka[i][0] = y;
    policka[i][1] = x;
    p+=l;
    x--; y++;
    if (x<1 || y>n) {
      l++;
      x = min(l,m);
      y = l+1-x;
    }
  }
  printf("%d\n", p);
  for (int i=k-1; i>=0; i--) {
    for (int y=1; y<=policka[i][0]; y++) {
      printf("(%d,1) ", y);
    }
    for (int x=2; x<=policka[i][1]; x++) {
      printf("(%d,%d) ", policka[i][0], x);
    }
    printf("\n");
  }
}
