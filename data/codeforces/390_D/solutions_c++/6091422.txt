#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
const int maxn=55;
struct node
{
    int x,y;
}grid[maxn*maxn];
int n,m,k,num;
bool cmp(node a,node b)
{
    return a.x+a.y<b.x+b.y;
}
int main()
{
    num=0;
    cin>>n>>m>>k;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
        grid[num].x=i,grid[num++].y=j;
    sort(grid,grid+n*m,cmp);
    int ans=0;
    for(int i=0;i<k;i++)
    ans+=grid[i].x+grid[i].y-1;
    cout<<ans<<endl;
    for(int i=k-1;i>=0;i--)
    {
        for(int j=1;j<=grid[i].x;j++)
        printf("(%d,%d) ",j,1);
        for(int j=2;j<=grid[i].y;j++)
        printf("(%d,%d) ",grid[i].x,j);
        cout<<endl;
    }
    return 0;
}
