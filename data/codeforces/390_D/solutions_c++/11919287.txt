#include <cstdio>
#include <algorithm>
using namespace std;
struct data{int i,j;} a[3000];
int n,m,k,ans,cnt;
bool cmp(data a,data b){return a.i+a.j<b.i+b.j;}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for (int i=1;i<=n;i++)
		for (int j=1;j<=m;j++)
			a[++cnt] = (data){i,j};
	sort(a+1,a+1+cnt,cmp);
	for (int i=k;i;i--) ans+=a[i].i+a[i].j-1;
	printf("%d\n",ans);
	for (int i=k;i;i--){
		for (int x=1,y=1;x<=a[i].i && y<=a[i].j;){
			if (x==a[i].i && y==a[i].j){printf("(%d,%d)\n",x,y);break;}
			printf("(%d,%d) ",x,y);
			if (x<a[i].i) x++;else y++;
		}
	}
}
