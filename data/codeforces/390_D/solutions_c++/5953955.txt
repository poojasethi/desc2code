#include <iostream>
#include <vector>

using namespace std;

void escribe(pair<int,int> p)
{
  for (int i=1;i<=p.first;i++)
    cout<<"("<<i<<","<<1<<")";
  for (int j=2;j<=p.second;j++)
    cout<<"("<<p.first<<","<<j<<")";
  cout<<endl;
}

vector<pair<int,int> > v;

int n,m,k;
int coste=0;

void genera()
{
  int pasos=0;
  for (int i=1;;i++) {
    for (int j=1;j<=i and j<=m;j++) {
      if (i-(j-1)<=n) {
	pasos++;
	v.push_back(pair<int,int> (i-(j-1),j));
	//cout<<i-(j-1)<<" "<<j<<endl;
	coste+=i;
	if (pasos==k) return;
      }
    }
  }
}

int main()
{
  cin>>n>>m>>k;
  genera();
  cout<<coste<<endl;
  for (int i=int(v.size())-1;i>=0;i--)
    escribe(v[i]);
}
