#include <bits/stdc++.h>
using namespace std;
#define FREP(b) for(int i=0;i<b;i++)
#define REP(a,b,c) for(int a=b;a<c;a++)
#define asd(x) cout << "Here for " << #x << " :" <<x << endl;
#define asdf(x, y)  cout << "Here FOR " << #x << " :" << x << " " << #y  << " :" << y << endl;
typedef pair<int,int> ii;
typedef long long LL;
int n,m,k,ans=0;
vector<ii> store;

void go(int x, int y){
    if(k == 0) return;
    k--;
    store.push_back(ii(x,y));
    ans += abs(x-1) + abs(y-1) + 1;
    if(x - 1 == 0 || y + 1 == m+1) return;
    go(x-1,y+1);
}
void print(int x, int y){
    REP(i,1,x+1){
        printf("(%d,%d) ", i,1);
    }
    REP(i,2,y+1) printf("(%d,%d) ",x,i);
    printf("\n");
}
int main(){
    
    cin >> n >> m >> k;
    REP(i,1,n+1) go(i,1);
    REP(i,2,m+1) go(n,i);
    cout << ans << endl;
    for(int i = store.size() - 1; i >= 0 ; i--){
        print(store[i].first, store[i].second);
    }
    return 0;
}
