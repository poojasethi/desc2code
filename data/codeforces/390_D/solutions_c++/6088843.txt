#include<stdio.h>
#include<algorithm>
using namespace std;
struct dd{
	int x;
	int y;
}a[3000];
bool cmp(dd a,dd b){
	return a.x+a.y<b.x+b.y;
}
int main()
{
	int n,m,k,i,j,l,h,sum;
	scanf("%d%d%d",&n,&m,&k);
	h=0;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
				a[h].x=i,a[h].y=j;
				h++;
		}
	}
	sort(a,a+h,cmp);
	sum=0;
	for(i=0;i<k;i++)
	{
		sum=sum+a[i].x+a[i].y-1;
	}
	printf("%d\n",sum);
	for(i=k-1;i>=0;i--)
	{
		for(j=1;j<=a[i].x;j++)
		  printf("(%d,1) ",j);
        for(j=2;j<=a[i].y;j++)
          printf("(%d,%d) ",a[i].x,j);
        printf("\n");
	}
}
