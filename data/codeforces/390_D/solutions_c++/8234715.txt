#include <cstdio>
#include <algorithm>
using namespace std;
int a,c,d,e,f,g,h,i,j,k,l,m,n;
struct node{int x,y,cost;}b[10005];
inline bool cmp(const node &a,const node &b){return a.cost<b.cost;}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(i=1;i<=n;i++)
	for(j=1;j<=m;j++){
		b[++l]=(node){i,j,i+j-1};
	}
	sort(b+1,b+1+n*m,cmp);
	for(i=1;i<=k;i++)h+=b[i].cost;
	printf("%d\n",h);
	for(;k;k--){
		for(i=1;i<b[k].x;i++)printf("(%d,%d)",i,1);
		for(i=1;i<=b[k].y;i++)printf("(%d,%d)",b[k].x,i);
		printf("\n");
	}
	return 0;
}
