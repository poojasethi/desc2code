#include<stdio.h>
int n,m,k,ans;
int min(int x,int y){
	return x<y?x:y;
}
int max(int x,int y){
	return x>y?x:y;
}
int get(int x,int y){
	if(n>=m)return min(x,m)+1-y;
	else return x+1-max(1,y-m+n);
}
void dfs1(int x,int y,int tot){
	int now,i,j;
	now=get(x,y);
	if(tot>now){
		if(x==n)dfs1(x,y+1,tot-now);
		else dfs1(x+1,y,tot-now);
	}
	now=min(now,tot);
	for(i=1;i<=now;i++){
		ans++;
		for(j=2;j<=x;j++)ans++;
		for(j=2;j<=y;j++)ans++;
	}
}
void dfs(int x,int y,int tot){
	int now,i,j;
	now=get(x,y);
	if(tot>now){
		if(x==n)dfs(x,y+1,tot-now);
		else dfs(x+1,y,tot-now);
	}
	now=min(now,tot);
	for(i=1;i<=now;i++){		
		printf("(1,1)");
		for(j=2;j<=x;j++)printf(" (%d,%d)",j,1);
		for(j=2;j<=y;j++)printf(" (%d,%d)",x,j);
		printf("\n");
		x--;y++;
	}
}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	ans=0;
	dfs1(1,1,k);
	printf("%d\n",ans);
	dfs(1,1,k);
}
