#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 2510
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int x,y,z;
} a[N];
int i,j,k,n,m,K,t,s;
inline bool cc1(const ww &a,const ww &b) {
	return a.z<b.z;
}
inline void pr(int x,int y) {
	printf("(%d,%d) ",x,y);
}
inline void print(int x,int y) {
	int i;
	For(i,1,x) pr(i,1);
	For(i,2,y) pr(x,i);
	printf("\n");
}
int main() {
	scanf("%d%d%d",&n,&m,&K);
	For(i,1,n)For(j,1,m) {
		a[++t].x=i,a[t].y=j; a[t].z=i+j-1;
	}
	sort(a+1,a+t+1,cc1);
	For(i,1,K) s+=a[i].z;
	printf("%d\n",s);
	for (i=K;i;i--) print(a[i].x,a[i].y);
	return 0;
}
