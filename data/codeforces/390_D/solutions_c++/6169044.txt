#include<stdio.h>
#include<algorithm>
#define K 2510
using namespace std;
int p[K],q[K];
void path(int x,int y){
    for(int i=1;i<=x;i++)printf("(%d,%d) ",i,1);
    for(int i=2;i<=y;i++)printf("(%d,%d) ",x,i);
    putchar('\n');
}
int main(){
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    int k2=k,ans=0;
    for(int i=2;i<=n+m;i++){
        for(int j=1;j<=n;j++)if(1<=i-j&&i-j<=m){
            p[k2]=j;
            q[k2--]=i-j;
            ans+=i-1;
            if(!k2)break;
        }
        if(!k2)break;
    }
    printf("%d\n",ans);
    for(int i=1;i<=k;i++)path(p[i],q[i]);
    return 0;
}
