#include <stdio.h>
#include <algorithm>
#include <vector>

using namespace std;

const int N = 100;

int n, m, k, ans = 0, a, b, x, y;
vector<pair<int, pair<int, int> > > q;

int main()
{
    scanf("%d%d%d", &n, &m, &k);
    for(int i = 0; i < n; i++)
        for(int j = 0; j < m; j++)
            q.push_back(make_pair(i + j, make_pair(i, j)));
    sort(q.begin(), q.end());
    for(int i = k - 1; i >= 0; i--)
        ans += q[i].first + 1;
    printf("%d\n", ans);
    for(int i = k - 1; i >= 0; i--)
    {
        a = q[i].second.first + 1;
        b = q[i].second.second + 1;
        x = y = 1;
        printf("(1,1) ");
        while(x < a)
            x++,
            printf("(%d,%d) ", x, y);
        while(y < b)
            y++,
            printf("(%d,%d) ", x, y);
        puts("");
    }
}
