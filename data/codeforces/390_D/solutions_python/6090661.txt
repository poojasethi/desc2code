#!/usr/bin/env python


def read():
    return map(int, raw_input().split())


def work((row, col, n)):
    pt = []

    it = 0
    if row > col:
        while len(pt) < n:
            cr = it
            cc = 0
            while len(pt) < n and 0 <= cr:
                if 0 <= cr < row and  0 <= cc < col:
                    pt.append((cr, cc))
                cr -= 1
                cc += 1
            it += 1
    else:
        while len(pt) < n:
            cr = 0
            cc = it
            while len(pt) < n and 0 <= cc:
                if 0 <= cr < row and  0 <= cc < col:
                    pt.append((cr, cc))
                cr += 1
                cc -= 1
            it += 1
    
    pt.reverse()

    total = 0
    for i in range(len(pt)):
        total += pt[i][0] + pt[i][1] + 1

    print total
    for i in range(len(pt)):
        for j in range(pt[i][0]):
            print "(%d,%d)" % (j + 1, 1),
        for j in range(pt[i][1]):
            print "(%d,%d)" % (pt[i][0] + 1, j + 1),
        print "(%d,%d)" % (pt[i][0] + 1, pt[i][1] + 1)


if __name__ == "__main__":
    work(read())
