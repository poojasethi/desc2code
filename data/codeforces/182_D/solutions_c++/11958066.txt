#include<bits/stdc++.h>
using namespace std;
int ans=0;
string a,b;
inline bool check(string s,int t)
{
    if (s.size()%t!=0) return false;
    for (int i=0;i<s.size();i++)
        if (s[i]!=a[i%t])
            return false;
    return true;

}
main()
{
    cin>>a>>b;
    for (int i=1;i<=a.size();i++)
        ans+=( check(a,i)&&check(b,i) );
    printf("%d\n",ans);
}
