#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[100005],t[100005];

int main()
{
    scanf("%s%s",s,t);
    int n = strlen(s), m = strlen(t), ret=0;
    for(int i = 1; i <= min(n,m); i++)
    {
        if(s[i-1]!=t[i-1])break;
        if(n%i || m%i)continue;
        int ok = 1;
        for(int j = 0; j < n; j++)
            if(s[j]!=s[j%i])ok = 0;
        for(int j = 0; j < m; j++)
            if(t[j]!=t[j%i])ok = 0;
        ret+=ok;
    }
    printf("%d\n",ret);

    return 0;
}
