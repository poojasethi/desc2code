#include <bits/stdc++.h>
using namespace std;
char s[100010],t[100010];
int ans,lens,lent;
bool div(int x,char m[100010])
{
    if(strlen(m)%x) return false;
    for(int i=0;i<strlen(m);i++)
    {
        if(s[i%x]!=m[i]) return false;
    }
    return true;
}
int main()
{
    ans=0;
    scanf("%s%s",s,t);
    lens=strlen(s);
    lent=strlen(t);
    for(int i=1;i<=lens;i++)
    {
        if(lens%i) continue;
        if(div(i,s)&&div(i,t)) ans++;
    }
    printf("%d\n",ans);
    return 0;
}
