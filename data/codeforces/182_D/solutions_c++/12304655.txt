# include <iostream>
# include <cmath>
# include <string>
using namespace std;

string x,y;


bool is_divisor(string &s,int len)
{
	if (s.length() % len != 0)	return false;
	for (int i=0;i<s.length();i++)
		if (s[i] != x[i%len])
			return false;
	return true;
}


int main()
{
	cin>>x>>y;
	int s = min(x.length(),y.length());
	int res = 0;
	for (int i=0;i<s;i++)
	{
		if (x[i] != y[i])	break;
		if (is_divisor(x,i+1) && is_divisor(y,i+1))	res++;
	}
	cout<<res<<endl;
	return 0;
}
