#include <iostream>
#include <algorithm>
using namespace std;
bool verificar(string s,int x){
	for(int i=0; i<x; i++)
		for(int j=1; j<s.size()/x; j++)
			if(s[i]!=s[i+j*x]) return false;
	return true;
}
int main(){
	int menor,tam1,tam2;
	string s1,s2,s3;
	int cont=0;
	cin>>s1>>s2;
	s3=s1+s2;
	menor=min(s1.size(),s2.size());
	for(int i=1; i<=menor; i++){
		if(s1.size()%i==0 && s2.size()%i==0)
			if(verificar(s3,i))
				cont++;
	}
	cout<<cont<<endl;
	return 0;
}
