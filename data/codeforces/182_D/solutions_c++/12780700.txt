#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(0);cin.tie(0);

using namespace std;

int main()
{
    int ans=0;
    string s,t;
    cin>>s;
    cin>>t;
    int sl=s.length();
    int tl=t.length();
    for(int i=1;i<=__gcd(sl,tl);i++)
    {
        if(sl%i==0&&tl%i==0)
        {
            if(s.substr(0,i)==t.substr(0,i))
            {
                if(s.substr(i)+s.substr(0,i)==s&&t.substr(i)+t.substr(0,i)==t)
                    ans++;
            }
        }
    }
    cout<<ans;
    return 0;
}
