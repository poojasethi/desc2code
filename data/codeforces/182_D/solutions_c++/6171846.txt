#include  <iostream>
#include    <string>
using namespace std;
string     A,     B;
bool Check(string S, int t){
    if(S.size()%t) return false;
    for(int i=0; i<S.size(); i++) if(S[i] != A[i%t])  return false;
    return true;
}
int main(){
    int Res = 0;                cin>>A>>B;
    for(int t=1; t<=A.size(); t++)
        Res += Check(A, t) && Check(B, t);
    cout<<Res;
    return 0;
}
