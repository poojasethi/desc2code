#include<cstdio>
#include<cstring>

#define ran 111111

char s1[ran], s2[ran];
int len1, len2, ans;

int main() {
	scanf("%s%s", s1, s2);
	len1=strlen(s1);len2=strlen(s2);
	ans=0;
	for(int i=1;i<=len1;i++) {
		if(len1%i || len2%i) continue;
		bool chk = true;
		for(int j=0, k=0;j<len1 && chk;j++, k++) {
			if(k==i) k=0;
			if(s1[j]!=s1[k]) chk = false;
		}
		for(int j=0, k=0;j<len2 && chk;j++, k++) {
			if(k==i) k=0;
			if(s2[j]!=s1[k]) chk = false;
		}
		ans+=chk;
	}
	printf("%d\n", ans);
	
	return 0;
}
