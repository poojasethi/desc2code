#include <iostream>

using namespace std;

string s1, s2;
int r = 0;

bool check(int sz, string s) {
    if(s.size()%sz) return 0;
    for(int i=0;i<s.size();i++) if(s1[i%sz] != s[i]) return false;
    return true;
}

int main() {
    cin >> s1 >> s2;
    for(int i=1;i<=s1.size();i++) {
        if(s1.size()%i) continue;
        if(check(i,s1) && check(i,s2)) r++;
    }
    cout << r << endl;
    return 0;
}
