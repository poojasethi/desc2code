#include<bits/stdc++.h>
using namespace std;
string s,t;
int main()
{
  cin.tie(NULL);
  ios_base::sync_with_stdio(0);
  cin>>s>>t;
  int ans=0;
  int l1=s.length();
  int l2=t.length();
  for(int i=1;i<=min(l1,l2);i++){
    if(l1%i==0 && l2%i==0){
      int flag=0;

      for(int j=0;j<l1;j++){
        if(s[j%i]!=s[j]) { flag=1; break;}
      }
      if(flag==1) continue;

      for(int j=0;j<l2;j++){
        if(s[j%i]!=t[j]) { flag=1; break;}
      }
      if(flag==1) continue;

      ans++;
    }
  }
  cout<<ans<<"\n";
  return 0;
}
