#include<iostream>
#include <STRING>
using namespace std;
string s1,s2;
int pd(int num)
{
	int i,j,k;
	for(i=0;i<num;i++)
	{
		for(j=num;j<s1.size();j+=num)
		{
			if(s1[i]!=s1[i+j])return 0;
		}
		for(k=0;k<s2.size();k+=num)
		{
			if(s1[i]!=s2[i+k])return 0;
		}
	}
	return 1;
}
int main()
{

	string st;
	int i;int num;
	while(cin>>s1>>s2)
	{
		num=0;
		if(s1.size()>s2.size()){st=s1;s1=s2;s2=st;}
		for(i=1;i<=s1.size();i++)
		{
			if(!(s1.size()%i+s2.size()%i)) num+=pd(i);
		}
		cout<<num<<endl;
	}
}
