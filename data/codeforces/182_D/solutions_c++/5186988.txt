#include <iostream>
#include <string>
#include <vector>
#include <cmath>
using namespace std;
bool cnt(string a,string b)
{
	int n1 = a.size();
	int n2 = b.size();
	for (int i = 0; i < n1; i+=n2)
	{
		if(b!=a.substr(i,n2))
			return 0;
	}
	return 1;
}
int main()
{
	string s1,s2;
	cin>>s1>>s2;
	int n1 = s1.size();
	int n2 = s2.size();
	int ans = 0;string tmp;
	for (int i = 0; i < n1; i++)
	{
		tmp+=s1[i];
		if( n1%(i+1)==0 && n2%(i+1)==0 )
			if(cnt(s1,tmp)&&cnt(s2,tmp))
			{
				ans++;
			}
	}
	cout<<ans;
}
