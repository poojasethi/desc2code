#include <stdio.h>
#include <string>
#include <iostream>

using namespace std;
string a,b;
int n,m,rst;

bool ck(int x){
	string k=a.substr(0,x);
	int i,j;
	for(i=0;i<n;){
		for(j=0;j<x;j++,i++){
			if(k[j]!=a[i])return 0;
		}
	}
	for(i=0;i<m;){
		for(j=0;j<x;j++,i++){
			if(k[j]!=b[i])return 0;
		}
	}

	return 1;
}

int main(){
	cin>>a>>b;
	int i,j,N,last=0;
	n=a.size(); m=b.size();
	N=min(n,m);
	for(i=1;i<=N;i++){
		if(n%i==0 && m%i==0){
			if(ck(i))rst++;
		}
	}
	printf("%d",rst);
}
