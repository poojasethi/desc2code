#include<cstring>
#include<cmath>

#include<string>

using namespace std;
bool ok(string a,string b)
{
    int n=a.size();
    int m=b.size();
    for(int i=0;i<n;i+=m)
      if(a.substr(i,m)!=b)return false;
      return true;
}


int main()
{
 string a,b;
 int n,m,ans=0;
 cin>>a>>b;
 n=a.size();
 m=b.size();
 for(int i=1;i<=min(n,m);i++)
 if(n%i==0 && m%i==0)
 ans+=(ok(a,a.substr(0,i))&ok(b,a.substr(0,i)));
 cout<<ans<<endl;
}
