#include <bits/stdc++.h>
using namespace std;
int main () 
{
	string a,b,c;
	cin>>a>>b;
	c=a+b;
	int ans=0;
	for (int i=1;i<=min(a.length(),b.length());i++)
	{
		if (a.length()%i==0 && b.length()%i==0)
		{
			bool bb=false;
			for (int j=i;j<=c.length()-i;j+=i)
			{
				 if(c.substr(0,i)!=c.substr(j,i)) 
				 	bb=true;
			}
			if (!bb) 
				ans++;
		}
	}
	cout<<ans;
}
