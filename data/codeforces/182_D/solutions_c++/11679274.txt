#include<bits/stdc++.h>
using namespace std;
int ans=0,t;
string A,B;
bool check(string S,int t)
{
    if (S.size()%t) return false;
    for (int i=0;i<S.size();i++)
        if (S[i]!=A[i%t])
            return false;
    return true;

}
main()
{
    cin>>A>>B;
    for (int i=1;i<=A.size();i++)
        ans+=check(A,i)&&check(B,i);
    cout<<ans;
}
