#include<iostream>
using namespace std;
string s;long n,m;
int check(long i){
	
	for(long j=0;j<i;j++){
		for(long k=j+i;k<n+m;k=k+i){
			if(s[k]!=s[j])return 0;
		}
	}
	return 1;
}
int main(){
	string s1,s2;
	cin>>s1>>s2;
	s=s1+s2;
	n=s1.size();m=s2.size();long l;
	if(n>m)l=m;else l=n;long ans=0;
	for(long i=1;i<=l;i++){
		if(n%i==0&&m%i==0){
			if(check(i)==1)ans++;
		}
	}
	cout<<ans<<endl;
	
	
}
