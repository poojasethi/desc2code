#include <iostream>
#include <string>

using namespace std;

int result;
string S1,S2;

int main()
{
	cin >> S1 >> S2;
	int N=S1.size(),M=S2.size();
	for (int i=1;i<=N&&i<=M;i++)
	{
		if (N%i||M%i)
			continue;
		int j;
		for (j=i;j<N;j++)
			if (S1[j]!=S1[j-i])
				break;
		if (j<N)
			continue;
		for (j=i;j<M;j++)
			if (S2[j]!=S2[j-i])
				break;
		if (j<M)
			continue;
		for (j=0;j<i;j++)
			if (S1[j]!=S2[j])
				break;
		if (j<i)
			continue;
		result++;
	}
	cout << result << endl;
	return 0;
}
