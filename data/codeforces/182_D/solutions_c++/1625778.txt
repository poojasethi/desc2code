#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>
using namespace std;
const int N=100012;
char s1[N],s2[N];
int l1,l2,ans=0;
bool check(int l)
{
    for (int i=0;i<l1;++i)if (s1[i%l]!=s1[i]) return false;
    for (int i=0;i<l2;++i)if (s2[i%l]!=s2[i]) return false;
    return true;
}
int main()
{
    scanf("%s%s",s1,s2);
    l1=strlen(s1);l2=strlen(s2);
    for (int i=0;i<min(l1,l2) && s1[i]==s2[i];++i)
		if (l1%(i+1)==0 && l2%(i+1)==0 && check(i+1)) ++ans;
    printf("%d\n",ans);
}
