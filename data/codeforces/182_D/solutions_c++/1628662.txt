#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;



int main()
{
	string a,b,c;
	int i,ans,j;
	cin>>a>>b;
	ans=0;
	int n=a.size();
	int m=b.size();
	int f;
	for(i=1;i<=n;i++)
	{
		

		if(m%i==0&&n%i==0)
		{
			f=1;
			c=a.substr(0,i);
		for(j=0;i*j<n;j++)
		{
			if(c!=a.substr(i*j,i))
			{
				f=0;
				break;

			}
		}
		for(j=0;i*j<m;j++)
		{
			if(c!=b.substr(i*j,i))
			{
				f=0;
				break;
			}
		
		}
		ans+=f;
		}
		
	}
	cout<<ans<<endl;
	return 0;
}
