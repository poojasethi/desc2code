#include <iostream>
#include "bits/stdc++.h"
using namespace std;
string s,p;
bool check(string r,int period)
{
	if(r.size()%period!=0) return 0;
	int l=r.size();
	for(int i=0;i<l;i++)
	{
		if(r[i]!=s[i%period]) return 0;
	}
	return 1;
}
int main() {
	cin>>s>>p;
	int ans=0;
	
	for(int i=1;i<=s.size();i++)
	if(check(s,i) && check(p,i))
	   ans++;
	
	cout<<ans<<endl;
	return 0;
}
