#include <iostream>
#include <string>
#include <map>
using namespace std;
bool str(string s,string ss)
{
    int l=s.length(),ll=ss.length();
    if (ll%l!=0) return 0;
    for (int i=0;i<ll;i+=l)
        if (s!=ss.substr(i,l)) return 0;
    return 1;
}
int main()
{
    string a,b;
    cin >> a >> b;
    if (a>b) swap(a,b);
    int an=a.length();
    int k=0;
    for (int i=1;i<=an/2;i++)
    {
        string as=a.substr(0,i);
        if (str(as,a) && str(as,b)) k++;
    }
    cout << k+str(a,b);
    return 0;
}
