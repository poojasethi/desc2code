#include<stdio.h>
#include<string.h>
char s1[100009],s2[100009];
char s[100009];
int main()
{
	int i,j,k;
	scanf("%s",s1);
	scanf("%s",s2);
	int len1=strlen(s1);
	int len2=strlen(s2);
	int num=0;
	for(i=0;i<len1;i++)
	{
		if(len1%(i+1)==0&&len2%(i+1)==0)
		{
			for(j=0;j<=i;j++)
				s[j]=s1[j];
			for(j=i+1;j<len1;j++)
				if(s1[j]!=s[j%(i+1)])
					break;
				if(j==len1)
				{
					for(k=0;k<len2;k++)
						if(s2[k]!=s[k%(i+1)])
							break;
						if(k==len2)
							num++;
				}
		}
	}
	printf("%d\n",num);
	return 0;
}
