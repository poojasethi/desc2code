#include<iostream>
#define wait system("pause")
using namespace std;

bool cont(string str,string t){
	int j,sz1=str.size(),sz2=t.size();
	for(j=0;j<sz1;j+=sz2)
		if( t!=str.substr(j,sz2) )
			return 0;
	return 1;
}

int main(){
	string str1,str2,temp;
	cin>>str1>>str2;
	int i,s1=str1.size(),s2=str2.size(),res=0;
	for(i=0;i<s1;i++){
		temp+=str1[i];
		if( s1%(i+1)==0 && s2%(i+1)==0 )
			if( cont(str1,temp) && cont(str2,temp) )
				res++;
	}
	cout<<res<<endl;
//	wait;
}
