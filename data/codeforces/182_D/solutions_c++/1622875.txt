#include<iostream>
#include<string>
using namespace std;

string s1,s2;

int main()
{
	int n,m,i,j,f;
	while(cin>>s1>>s2)
	{
		int ans=0;
		n=s1.size();
		m=s2.size();
		for(i=1;i<=n;i++)
			if(n%i==0&&m%i==0)
			{
				f=1;
				string t=s1.substr(0,i);
				for(j=0;j<n/i;j++)
					if(t!=s1.substr(j*i,i))
					{  f=0; break; }
				for(j=0;j<m/i;j++)
					if(t!=s2.substr(j*i,i))
					{  f=0; break; }
				if(f)
					ans++;
			}
		cout<<ans<<endl;
	}
	return 0;
}
