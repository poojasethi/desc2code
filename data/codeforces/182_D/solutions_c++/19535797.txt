#include <iostream>
#include<string>
#define ll long long

using namespace std;

string str1, str2;

bool chk(string str, ll seg_size)
{
  if(str.size()%seg_size!=0) return false;
  else
  {
      for(int i=0; i<str.size(); i++)
      {
          if(str[i]!=str1[i%seg_size]) return false;
      }
  }
  return true;
}

int main()
{
   ll ans = 0;
   cin>>str1>>str2;
   for(int i=1; i<=min(str1.size(), str2.size()); i++)
   {
      ans += (chk(str1,i) && chk(str2, i));
   }
   cout<<ans<<endl;
}
