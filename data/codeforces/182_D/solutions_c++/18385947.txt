#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
	int i,j,k,n,m,ans=0;
	string p,q;
	cin>>q>>p;
	n=p.length();m=q.length();
	for(i=0;i<min(m,n);i++)
	{
		if(p[i]!=q[i]){break;}
		if(n%(i+1)==0 && m%(i+1)==0)
		{
			int flag=1;
			for(j=0;j<n;j++){if(p[j]!=p[j%(i+1)]){flag=0;break;}}
			for(j=0;j<m;j++){if(q[j]!=q[j%(i+1)]){flag=0;break;}}
			ans+=flag;
		}
	}
	cout<<ans<<endl;
	return 0;
}
