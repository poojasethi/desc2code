#include<iostream>
using namespace std;
int main()
{
    string a,b;
    cin>>a>>b;
    int sum=0;
    bool mark=true;
for(int i=1;a.size()>=i && i<=b.size();i++){
    if(a.size()%i==0 && b.size()%i==0){
        string x =a.substr(0,i);
        mark=true;
        for(int j=0;j<a.size();j+=i){
            if(a.substr(j,i)!=x)mark=false;
        }
        for(int j=0;j<b.size();j+=i){
            if(b.substr(j,i)!=x)mark=false;
        }
        sum+=mark;
    }}
    cout<<sum;

return 0;}
