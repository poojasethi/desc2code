# include <stdio.h>
# include <string.h>

int n,n1;

int ans=0;

char a[100050];
char b[100050];

int main()
{
	scanf("%s\n%s",a,b);
	
	int n=strlen(a),n1=strlen(b);
	
	for(int i=1;i<=n;i++)
	{
		if(n%i == 0 && n1%i == 0)
		{int q=0,k=0,l=0;
			
			while(k < n || l < n1)
			{
				if(k < n)
				 for(int j=k;j<k+i;j++)
				  if(a[j] != a[j-k]) {q=1;break;}
				
				if(l < n1)
				 for(int j=l;j<l+i;j++)
				  if(b[j] != a[j-l]) {q=1;break;}
				
				if(q==1) break;
				
				k+=i,l+=i;
			}
			
			if(q==0) ans++;
		}
	}

	printf("%d\n",ans);
}
