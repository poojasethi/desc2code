#include <iostream>
#include <string>
#include <cstdio>
using namespace std;
string a,b;
bool pan(int x,string s)
{
	for(int i=0;i<s.length();i++)
	{
		if(a[i%x]!=s[i]) return false;
		
	}
	return true;
}
int main()
{
	int ans=0;
	cin>>a>>b;
	int i;
	for(i=1;i<=a.length()&&i<=b.length();i++)
	{
		if(a.length()%i==0&&b.length()%i==0&&pan(i,a)&&pan(i,b))
		  ans++;			
	}
	cout<<ans<<endl;

	return 0;
}
