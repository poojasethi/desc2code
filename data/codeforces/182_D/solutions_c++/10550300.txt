#include <iostream>
using namespace std;

int main() {
	string a, b;
	cin >> a >> b;
	int ans = 0;
	for (int i = 1; i <= (int)a.size(); i++) {
		if (a.size() % i == 0 && b.size() % i == 0) {
			string t = a.substr(0, i);
			bool ok = true;
			for (int j = 0; j < (int)a.size(); j += i) {
				if (a.substr(j, i) != t) ok = false;
			}
			for (int j = 0; j < (int)b.size(); j += i) {
				if (b.substr(j, i) != t) ok = false;
			}
			ans += ok;
		}
	}
	cout << ans << endl;
}
