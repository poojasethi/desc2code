#include <bits/stdc++.h>
typedef long long unsigned ll;
using namespace std;

int main(){
	string s1,s2;
	cin >> s1 >> s2;
	int l1 = s1.length();
	int l2 = s2.length();
	string s = s1+s2;
	int m = min(l1,l2);
	int ans=0;
	for(int i=1;i<=m;i++){
		if(l1%i==0 && l2%i==0){
			int j=(l1+l2)/i;
			int p=0;
			bool b = true;
			for(int k=1;k<j;k++){
				if(s.substr(p,i)!=s.substr(p+i,i)){
					b = false;
					break;
				}
				else{
					p=p+i;
				}
			}
			if(b==true){
				ans++;
			}
		}
	}
	cout << ans;
	return 0;
}
