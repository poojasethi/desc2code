#include <bits/stdc++.h> 
using namespace std ;
#define fi first
#define se second

string s , p ;
int ans = 0 ;

inline bool check ( int x , string t ) {
	if ( t.size() % x )
		return 0 ;
	for ( int i = 0 ; i < t.size() ; i++ ) 
		if ( t[i] != s[i%x] )
			return 0 ;
	return 1 ;
}

int main()
{
	ios_base::sync_with_stdio ( false ) ;
	cin >> s >> p ;
	for ( int i = 1 ; i <= s.size() ; i++ ) 
		if ( check ( i , s ) == 1 && check ( i , p ) == 1 )
			ans ++ ;
		cout << ans << endl ;
	return 0 ;
}
