#!/usr/bin/env python
import sys

def all_divisors(s):
    res = set()
    for i in xrange(1, len(s)+1):
        if not (len(s) % i) and s[:i] * (len(s)/i) == s:
            res.add(s[:i])
    return res

s1 = raw_input()
s2 = raw_input()
d1 = all_divisors(s1)
d2 = all_divisors(s2)

d = d1.intersection(d2)

print len(d)
