from math import *

def generate(str):
    l = len(str)
    for x in xrange(1, int(sqrt(l))+1):
        if l % x == 0:
            if str[:x]*(l/x) == str:
                yield str[:x]
            if l/x != x and str[:l/x]*x == str:
                yield str[:l/x]
