import sys

def divi(n):
    i = 1
    while i <= n:
        if n % i == 0:
            yield i
        i += 1

def get(s):
    for d in divi(len(s)):
        sd = s[:d]
        if all(sd == s[c:c+d] for c in range(d, len(s), d)):
            yield sd

s1 = sys.stdin.readline().rstrip('\n')
s2 = sys.stdin.readline().rstrip('\n')

print len(set(get(s1)) & set(get(s2)))
