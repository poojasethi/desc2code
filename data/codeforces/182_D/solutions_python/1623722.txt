#! /usr/bin/env python

from sys import stdin

def main():
    strings = [line.rstrip('\n') for line in stdin]
    print(len(set(find_prefixes(strings[0])).intersection(\
        set(find_prefixes(strings[1])))))

def find_prefixes(str):
    prefixes = []
    for i in range(len(str)):
        if len(str) % (i + 1) == 0:
            if str[:i+1] * (len(str) / (i + 1)) == str:
                prefixes += [str[:i+1]]
    return prefixes

if __name__ == "__main__":
    main()
