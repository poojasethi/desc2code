def strval(string):
    for i in xrange(1, len(string) + 1):
        if string == (string[:i] * (len(string)/i)):
            return string[:i], len(string)/i

def c_div(a,b):
    counter = 0
    for i in xrange(1, min(a,b) + 1):
        if a % i == 0 and b % i == 0:
            counter += 1
    return counter
