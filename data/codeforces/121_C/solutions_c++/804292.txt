#define f first

#define sqr(x) ((x)*(x))
const ll INF = 10000000000LL;
ll f[22];
int ans;

int get(int n, int k)
{
  f[0] = 1;
  int i;
  for (i = 1; i <= n && f[i] < k; i++)
  {
    f[i] = f[i - 1] * i;
    if (f[i] >= k) return i;
  }
  return -1;
}

void luc(ll x, int t)
{
  if (x > t) return;
  if (x != 0) ans++;
  luc(x * 10 + 4, t);
  luc(x * 10 + 7, t);
}

bool lucky(int v)
{
  while (v)
  {
    if (v % 10 != 7 && v % 10 != 4) return false;
    v /= 10;
  }
  return true;
}
set <int> s;
int main()
{
  int n, k, p, x;
  cin >> n >> k;
  p = get(n, k);
  if (p == -1)
  {
    cout << -1;
    return 0;
  }
  x = n - p;
  luc(0LL, x);
  for (int i = x + 1; i <= n; i++)
    for (int j = x + 1; j <= n; j++)
      if (s.find(j) == s.end())
      {
        if (k <= f[n - i])
        {
          s.insert(j);
          if (lucky(i) && lucky(j)) ans++;
          break;
        } else k -= f[n - i];
      }
  cout << ans;
  return 0;
}
