#include <iostream>
#include <string>
#include <fstream>
#include <algorithm>
#include <vector>

using namespace std;

#define ll long long

int N, L, chk[20], res = 0;
ll K, F[20], P[20];

void countLucky(ll n) {
    if(n > N) return;
    if(n > 0)
        res++;
    countLucky(10 * n + 4);
    countLucky(10 * n + 7);
}

bool lucky(int n) {
    while(n) {
        if(n%10 != 4 && n%10 != 7) return false;
        n /= 10;
    }
    return true;
}

int main () {
    cin>>N>>K;
    L = min(15, N);
    F[0] = 1;
    for(int i = 1;i<=L;i++)
        F[i] = i * F[i-1];

    for(int k = 1;k<=L;k++) {
        int i = 0;
        for(int j = 1;j<=L;j++) {
            if(!chk[j]) {
                i++;
                if(i * F[L - k] >= K) {
                    chk[j] = 1;
                    P[k] = j;
                    K -= (i-1) * F[L - k];
                    break;
                }
            }
        }
    }

    if(K > 1) {
        printf("%d\n",-1);
        return 0;
    }

    countLucky(0);

    for(int i = 1;i<=L;i++) {
        P[i] += N - L;
        int idx = i + N - L;

        if(lucky(P[i]) && !lucky(idx))
            res--;
    }
    printf("%d\n",res);
    return 0;
}
