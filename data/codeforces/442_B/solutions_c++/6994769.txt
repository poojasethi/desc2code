#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=110;
int n;
double p[maxn];
double ans=0;
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%lf",&p[i]);
	sort(p+1,p+n+1);
	double t0=1,t1=0;
	for(int i=n;i>=1;i--){
		t1=(t0-t1)*p[i]+t1;
		t0=t0*(1-p[i]);
		if(t1>ans) ans=t1;
	}
	printf("%.10lf\n",ans);
	return 0;
}
