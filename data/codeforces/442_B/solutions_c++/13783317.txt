#include <bits/stdc++.h>
using namespace std;

double r[105], s, p = 1, m;

int main()
{
    int n, i;

    for (i = 1; i <= n; i++) 
    cin >> r[i];
    
    sort(r + 1, r + n + 1);
    
    for (i = n; i >= 1; i--)
    {
        s = s * (1 - r[i]) + r[i] * p;

}
