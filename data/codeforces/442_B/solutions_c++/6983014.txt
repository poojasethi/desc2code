#include<bits/stdc++.h>
using namespace std;
double a[105],ans,m;
int n;

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%lf",&a[i]);
	sort(a,a+n,greater<double>() );
	ans=a[0],m=1-a[0];
	for(int i=1;i<n;i++)
	{
		if(m-ans<0) break;
		ans=ans+a[i]*(m-ans);
		m=m*(1-a[i]);
	}
	printf("%.10lf\n",ans);
	return 0;
}
