#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int n;
double p[111],ans;

int main(void)
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>p[i];
	}
	sort(p,p+n);
	double ans=0;
	for(int i=0;i<n;i++)
	{
		double t=1,f=0;
		for(int j=i;j<n;j++)
		{
			f = f*(1-p[j]) + t*p[j];
			t *= (1-p[j]);
		}
		ans = max(ans,f);
	}
	printf("%.9f\n",ans);
	return 0;
}
