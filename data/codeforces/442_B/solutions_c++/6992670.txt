#include <bits/stdc++.h>
using namespace std;

int main()
{
    double d[110];
    int n;
    cin>>n;
    for(int i=0;i<n;++i)scanf("%lf",&d[i]);
    sort(d,d+n);
    double ans=d[n-1];
    for(int k=2;k<=n;++k)
    {
        double chen=1.0,sum=0;
        for(int i=n-1;i>=n-k;--i)
            chen*=1-d[i],sum+=d[i]/(1.0-d[i]);
        ans=max(ans,chen*sum);
    }
    printf("%.12lf\n",ans);
}
