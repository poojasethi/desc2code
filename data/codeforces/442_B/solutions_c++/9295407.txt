#include<iostream>
#include<algorithm>
#include<iomanip>
using namespace std;
int main(){
    int n,i,j;
    double ans=1, p[102], no=1;
    cin>>n;
    for(i=0;i<n;i++)cin>>p[i];
    sort(p,p+n);
    ans=p[n-1]; no=1-p[n-1];
    for(i=n-2;i>=0;i--){
        ans=max(ans, (1-p[i])*ans+no*p[i]);
        no*=1-p[i];
    }
    cout<<setprecision(11)<<ans;
    return 0;
}
