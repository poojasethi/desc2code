#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;
double p[110];
int main(){
int n,i;
double ans1=0,ans2=1,ans=0;
scanf("%d",&n);
for(i=0;i<n;++i) scanf("%lf",&p[i]);
sort(p,p+n);
for(i=n-1;i>=0;--i){
ans1=ans1*(1-p[i])+ans2*p[i];
ans2*=1-p[i];
ans=max(ans,ans1);
}
printf("%.12f\n",ans);
return 0;
}
