#include<bits/stdc++.h>
using namespace std;
main()
{
    int n;
    scanf("%d",&n);
    double a[105],ans,S,P;
    for(int i=0;i<n;i++)
    scanf("%lf",&a[i]);
    sort(a,a+n);
    S=a[n-1],ans=a[n-1],P=(1-a[n-1]);
    for(int i=n-2;i>=0;i--)
    {
        S=S*(1-a[i])+P*a[i];
        P*=(1-a[i]);
        ans=max(S,ans);
    }
    printf("%.12lf\n",ans);
}
