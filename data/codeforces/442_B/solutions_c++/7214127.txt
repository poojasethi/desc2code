#include<iostream>
#include<algorithm>
using namespace std;
int main()
{int n;cin>>n;
 double x[105];
 for(int i=1;i<=n;i++)cin>>x[i];
 double p=0,g=1;
 sort(x+1,x+n+1);
 for(int i=n;i>=1;i--)
  if(p*(1-x[i])+x[i]*g>p)
    {p=p*(1-x[i])+x[i]*g;
      g*=1-x[i];
    }
cout.precision(10);
cout<<p;
return 0;
}
