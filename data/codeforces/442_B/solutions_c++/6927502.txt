#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 105
double a[N],ans,no;
int n;

int main(){
  int i;
  scanf("%d",&n);
  for (i=0;i<n;++i) scanf("%lf",&a[i]);
  sort(a,a+n);
  ans=0;
  no=1;
  for (i=n-1;i>=0;--i)
    if (ans*(1-a[i])+no*a[i]>ans){
      ans=ans*(1-a[i])+no*a[i];
      no*=(1-a[i]);
    }
  printf("%.9f\n",ans);
  return 0;
}
