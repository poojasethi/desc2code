
#include <bits/stdc++.h>
using namespace std;

int n;
double p[128];

int main() {
    scanf("%d",&n);
    for (int i=1;i<=n;i++) scanf("%lf",p+i);
    sort(p+1,p+1+n);
    double pi=1,ans=0;
    for (int i=n;i>0;i--) {
        if (pi>ans) {
            ans=ans-ans*p[i]+pi*p[i];
            pi*=(1.0-p[i]);
        }
    }
    printf("%.12f\n",ans);
    return 0;
}
