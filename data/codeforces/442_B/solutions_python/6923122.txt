def main():
	raw_input()
	a = list(reversed(sorted(map(float, raw_input().split()))))

	answer = 0.0

	for i in xrange(len(a)):
		t = a[i]
		if i > 0:
			t = 0.0
			for j in xrange(i+1):
				s = a[j]
				for k in xrange(i+1):
					if j!=k:
						s *= 1.0 - a[k]
				t += s

		if t > answer:
			answer = t

	print '%.12f' % answer

if __name__ == '__main__':
	main()
