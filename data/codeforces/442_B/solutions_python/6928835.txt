n = int(raw_input())
ps = map(float, raw_input().split())

ps.sort(reverse=True)

good = ps[0]
bad = 1 - good

ans = good

for p in ps[1:]:
    if good > 0.5:
        break
    else:
        ng = good * (1 - p) + p * bad
        nb = bad * (1 - p)
        good, bad = ng, nb
        ans = max(ans, good)

print ans
