def solve(n,k):
    if k==1 and n==1: return 'a'
    if k==1: return '-1'
    # k>1
    if n<k: return '-1'

    res = ['a','b'] * (n//2)
    if n%2==1:
        res+=['a']
    for i in range(k-2):
        res[-(k-2)+i] = chr(ord('a')+2+i)
    return ''.join(res)

n,k = map(int,raw_input().split())
print solve(n,k)
