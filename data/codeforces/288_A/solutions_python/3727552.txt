n,k = map(int,raw_input().split())
letters = [chr(a) for a in range(ord('a'),ord('z')+1)]
if(n>=k and k != 1):
	pal = ""
	foi = 0
	for i in range(n-k+2):
		if foi:
			pal += 'b'
		else:
			pal += 'a'
		foi += 1
		foi %= 2
	for i in letters[2:k]:
		pal += i
	print pal
elif (k==1 and n == 1):
	print "a"
else :
	print -1
