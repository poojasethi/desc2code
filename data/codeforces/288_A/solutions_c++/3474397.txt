#include <cstdio>

int main() {
	int n, k;
	scanf("%d%d", &n, &k);
	if (n < k || k == 1 && n > 1) {
		printf("-1");
	} else if (n == 1) {
		printf("a");
	} else {
		printf("ab");
		for (int i = 0; i < n - k; ++i) {
			printf("%c", 'a' + i % 2);
		}
		for (int i = n - k; i < n - 2; ++i) {
			printf("%c", 'c' + i - n + k);
		}
	}
}
