#include <iostream>
using namespace std;
int main()
{
	string s;
	int n, k, c = 0;
	cin >> n >> k;
	if(k > n || (k == 1 && n > 1))
	{
		cout << "-1";
		return 0;
	}
	else 
	for(int i = 0; i < n; i++)
	{
		if(i <= n-k+1)
			s+= ('a' + (i%2));
		else
		{
			s += ('c' + c);
			c++;
		}
	}
	cout << s;
	return 0;
}
