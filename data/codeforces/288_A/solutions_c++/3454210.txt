#include<cstdio>
int main(){
	int n, k;scanf("%d%d",&n,&k);
	if( n <k )
		printf("-1");
	else{
		if(k==1){
			if(n>1)
				printf("-1");
			else
				printf("a");
		}else{
			for(int i=0;i<n-(k-2);++i){
				if(i&1)printf("b");
				else printf("a");
			}
			for(int i=0;i<k-2;++i)
				printf("%c",'c'+i);
		}
	}
	puts("");
		
			
	return 0;
}
