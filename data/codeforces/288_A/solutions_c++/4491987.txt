#include<cstring>
#include<cstdlib>
#include<cstdio>
using namespace std;
int n,k,i,j;
int main(){
	scanf("%d%d",&n,&k);
	if(k>n || k==1 && n>1)return puts("-1"),0;
	for(i=1;i<=n-k;++i)printf("%c",(i&1)?'a':'b');
	for(j=0;j<k;++j){
		if(j<=1)printf("%c",(i+j)&1?'a':'b');
		else printf("%c",j+'a');
	}
}
