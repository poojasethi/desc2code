#include<stdio.h>

int main()
{
    int n, k, i, m;
    char c;
    scanf("%d %d", &n, &k);
    if(n<k || (k==1 && n>k))return printf("-1")&0;
    m=n-(k>1?k-2:0);
    for(i=0; i^m; i++)printf("%c", "ab"[i&1]);
    for(c='c', i=2; i<k; i++)printf("%c", c++);
    return 0;
}
