#include <cstdio>

int n, k;

int main()
{
	scanf("%d%d", &n, &k);

	if(k==1) {
		if(n==1) printf("a\n");
		else printf("-1\n");
	}
	else {
		if(n>=k) {
			for(int i=0; i<n-k+2; i++) printf("%c", 'a'+(i%2));
			for(int i=n-k+2, j='c'; i<n; i++, j++) printf("%c", (char)j);
			printf("\n");
		}
		else printf("-1\n");
	}

	return 0;
}
