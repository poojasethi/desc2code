#include<cstdio>
int n,k;
int main(){
    scanf("%d%d",&n,&k);
    if(n<k)puts("-1");
    else if(k==1){
        if(n>1)puts("-1");
        else puts("a");
    }
    else{
        for(int i=0;i<n-(k-2);++i)putchar('a'+(i&1));
        for(int i=0;i<k-2;++i)putchar('c'+i);
        puts("");
    }
    return 0;
}
