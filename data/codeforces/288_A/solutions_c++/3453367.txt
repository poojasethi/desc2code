#include <cstdio>

int main()
{
	int n, k;
	scanf("%d%d", &n, &k);
	
	if (k == 1)
	{
		if (n == 1) printf("a\n");
		else printf("-1\n");
	}
	else if (n < k) printf("-1\n");
	else
	{
		for (int i = 0; i < n-(k-2); ++i)
			printf("%c", (i&1)?'b':'a');
		for (int i = 2; i < k; ++i)
			printf("%c", i+'a');
		printf("\n");
	}
}
