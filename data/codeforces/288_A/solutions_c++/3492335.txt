#include<stdio.h>
int main ()
{
	int n,m,i,k;
	scanf("%d%d",&n,&m);
	if(m==1&&n>m||n<m)
	{
		printf("-1\n");
	}
	else
	{
		for(i=0;i<m;i++)	
		{
			printf("%c",'a'+i);
			if(i==1)
			{
				for(k=0;k<n-m;k++)
					if(k%2)
						printf("b");
					else
						printf("a");
			}
		}
		printf("\n");
	}
	return 0;
}
