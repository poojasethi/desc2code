#include<stdio.h>
main()
{
	int n,k,i,j;
	scanf("%d%d",&n,&k);
	if(k>n)printf("-1\n");
	else if(k==1){
	    if(n==1)printf("a\n");
	    else printf("-1\n");
	}
	else {
		for(i=0;i<n-k+2;i++){
			if(i%2)printf("b");
			else printf("a");
		}
		for(i=0;i<k-2;i++)printf("%c",i+99);
	}
}
