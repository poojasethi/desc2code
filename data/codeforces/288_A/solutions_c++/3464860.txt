#include<iostream>
using namespace std;
int main(){
    int n,k;
    cin>>n>>k;
    if(n<k||(k==1&&n!=1))
        cout<<-1;
    else{
        while(n>k&&n>2)
            cout<<"ab",n-=2;
        cout<<(n==1||n!=k?"a":"ab");
        for(int i=2;i<k;i++)
            cout<<char('a'+i);            
    }
}
