#include<cstdio>

int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	if(n==1&&k==1)
	{
		printf("a");
		return 0;
	}
	if(k>n||k==1)
	{
		printf("-1");
		return 0;
	}
	int count=2;
	for(int i=0;i<n;i++)
	{
		if(n-i>k-2)
		{
			printf("%c",'a'+(i%2));
		}
		else
		{	printf("%c",'a'+count);
			if(count<26)
				count++;
		}
	}
	return 0;
}
