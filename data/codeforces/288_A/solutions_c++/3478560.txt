#include <cstdio>

int main() {
   int n,k,i;
   scanf("%d%d",&n,&k);
   if(k>n || (k==1 && n>1)) {
      puts("-1");
      return 0;
   }
   if(k==1) {
      puts("a");
      return 0;
   }
   
   for(i=0;i<(n-k+2);++i){
      putchar(((i%2)?'b':'a'));
   }
   for(i=2;i<k;++i) putchar('a'+i);
   
   return 0;
}
