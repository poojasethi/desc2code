#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,k;cin>>n>>k;char c='c';
    if(n<k ||(n!=1 && k==1)){cout<<-1;return 0;}
    if(n==1 && k==1) {cout<<"a";return 0;}
    for(int i=0;i<(n-k+2)/2;i++) cout<<"ab";
    if((n-k+2) % 2!=0) cout<<"a";
    for(int i=0;i<k-2;i++) printf("%c",c+i);return 0;
}
