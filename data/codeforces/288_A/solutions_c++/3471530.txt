#include<cstdio>
int n,k;
#define F(i,n) for(int i=0;i<n;++i)
int main(){
    scanf("%d%d",&n,&k);
    if(n<k)puts("-1");
    else if(k==1){
        if(n>1)puts("-1");
        else puts("a");
    }
    else{
        F(i,n-k+2)putchar('a'+(i&1));
        F(i,k-2)putchar('c'+i);
        puts("");
    }
    return 0;
}
