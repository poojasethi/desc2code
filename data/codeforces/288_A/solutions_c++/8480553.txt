#include <cstdio>
#include <algorithm>

using namespace std;

int main() {
	int n, k;
	scanf("%d%d", &n, &k);
	if ((k == 1 && n > 1) || k > n) puts("-1");
	else {
		int tmp = min(n - k + 2, n);
		for (int i = 0; i < tmp; ++i) putchar(i & 1 ? 'b' : 'a');
		for (int i = 2; i < k; ++i) putchar('a' + i);
		puts("");
	}
	return 0;
}
