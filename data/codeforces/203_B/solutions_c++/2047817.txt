#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>

using namespace std;

int m,n,list[1003][1003],a,b;

int main()
{
cin>>n>>m;
for (int i=0;i<n;i++)
for (int j=0;j<n;j++)
list[i][j]=0;
for (int l=0;l<m;l++)
{
cin>>a>>b;
for (int y=a-1;y<a+2;y++)  
{
for (int z=b-1;z<b+2;z++)
{
list[y][z]++;
if (list[y][z]==9)
{
cout<<l+1;
return 0;
}
}
}
}
cout<<"-1";
}
