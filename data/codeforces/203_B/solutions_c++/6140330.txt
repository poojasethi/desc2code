#include <iostream>
using namespace std;
int square[1000+10][1000+10],flag=1;
int main(){
    int n,m,x,y,ans=-1;
    cin >>n>>m;
    for(int i=1;i<=m;i++){
        cin >>x>>y;
       for(int row=x;row<x+3&&flag;row++){
           for(int col=y;col<y+3;col++){
               square[row][col]++;
               if(square[row][col]==9){
                   ans=i;
                   flag=0;
               }
           }
       }
    }
    cout << ans;
   return 0;
}
