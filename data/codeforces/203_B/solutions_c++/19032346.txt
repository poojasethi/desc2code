#include<stdio.h>
int q[1005][1005]={0};
int main()
{
	int n,m;
	int i,j,t,k=-1;
	int x,y;
	scanf("%d %d",&n,&m);
	for (i=1;i<=m;i++)
	{
		scanf("%d %d",&x,&y);
		if (k!=-1)
			continue;
		for (j=x;j<x+3;j++)
		{
			for (t=y;t<y+3;t++)
			{
				q[j][t]++;
				if (q[j][t]==9)
				{
					k=i;
					break;
				}
			}
			if (k!=-1)
				break;
		}
	}
	printf("%d\n",k);
	return 0;
}
