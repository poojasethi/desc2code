#include<iostream>
using namespace std;
int n,m,x,y,a[1005][1005]={0};
int main()
{
	cin>>n>>m;
	for(int i = 1; i <= m; i++)
	{
		cin >> x >> y;
		for(int l = x; l < x+3; l++)
			for(int r = y; r < y+3; r++)
			{
				a[l][r]++;
				if(a[l][r]==9)
				{
					cout<<i;
					return 0;
				}
			}
	}
	cout<<-1;
	return 0;
}
