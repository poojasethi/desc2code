#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
int a[1005][1005],n,m;
int main(){
	int x,y;
	scanf("%d %d",&n,&m);
	memset(a,0,sizeof(a));
	for(int i=1;i<=m;i++){
		scanf("%d %d",&x,&y);
		for(int l=x;l<x+3;l++){
			for(int r=y;r<y+3;r++){
				a[l][r]++;
				if(a[l][r]==9){
					printf("%d\n",i);
					return 0;
				}
			}
		}
	}
	printf("-1\n");
	return 0;
}
