# include<iostream>
using namespace std;
const int maxn = 1111;
int f[maxn][maxn]={0};
int main()
{
   int n,m,x,y;
   cin>>n>>m;
      for(int i=1;i<=m;i++)
      {
         cin>>x>>y;
         for(int l=x;l<x+3;l++)
            for(int r=y;r<y+3;r++)
            {
               f[l][r]++;
               if(f[l][r]==9)
               {
                  cout<<i<<endl;
                  return 0;
               }
            }
      }
         cout<<"-1"<<endl;
   return 0;
}
