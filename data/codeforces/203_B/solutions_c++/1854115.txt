#include<cstdio>
#include<algorithm>
using namespace std;
int sum[1005][1005];
int main(){
	int n,m,i,j,a,b,cnt,f;
	scanf("%d%d",&n,&m);
	for(cnt=0;cnt<m;cnt++){
		f=0;
		scanf("%d%d",&a,&b);
		for(i=a;i<a+3&&i<n+1;i++){
			for(j=b;j<b+3&&j<n+1;j++){
				sum[i][j]++;
				if(sum[i][j]==9)f=cnt;
				}
			}
		if(f)break;
		}
	if(cnt==m)puts("-1");
	else printf("%d\n",cnt+1);
}
