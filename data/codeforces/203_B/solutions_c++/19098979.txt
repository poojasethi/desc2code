#include<cstdio>
const int N=1002;
int m,a[N][N];
int dx[]={-1,-1,-1,0,0,0,1,1,1};
int dy[]={-1,0,1,-1,0,1,-1,0,1};
int solve()
{
	int i,j,x,y;
	for(i=0;i<m;i++)
	{
		scanf("%d%d",&x,&y);
		for(j=0;j<9;j++)
			if(++a[x+dx[j]][y+dy[j]]==9)return i+1;
	}
	return -1;
}
int main()
{
	scanf("%d%d",&m,&m);
	printf("%d\n",solve());
	return 0;
}
