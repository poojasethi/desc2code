#include <iostream>
using namespace std;
int a[1005][1005]={0};
int main()
{
int n,m;
cin>>n>>m;
int x,y;
for (int i=1;i<=m;i++)
 {
  cin>>x>>y;
  for (int x1=0;x1<3;x1++)
     for (int y1=0;y1<3;y1++)
     {
       a[x+x1][y+y1]++;
       if (a[x+x1][y+y1]==9) {cout<<i; return 0;}
     }
  }
 cout<<"-1";
return 0;
}
