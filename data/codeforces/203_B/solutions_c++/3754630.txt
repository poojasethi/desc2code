#include <cstdio>
#include <iostream>
using namespace std;
int n,m,i,j,k,a[1005][1005]={0},r,c,ans=-1;
int main (){
	scanf ("%d %d",&n,&m);
	for (i=1;i<=m;i++){
		scanf ("%d %d",&r,&c);
		if (ans<0){
			for (j=r;j<r+3;j++){
				for (k=c;k<c+3;k++){
					a[j][k]++;
					if (a[j][k]==9) ans=i;
				}
			}
		}
	}
	printf ("%d\n",ans);
	return 0;
}
