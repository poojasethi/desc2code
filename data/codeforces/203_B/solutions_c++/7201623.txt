#include <iostream>
#include <fstream>
using namespace std;
typedef long long ll;
ll a[1111][1111];
int main()
{
	//freopen("D.in","r",stdin);
	ll n,m;
	cin >> n >> m;
	for(ll q = 1; q <= m; q++)
	{
		ll x,y;
		cin >> x >> y;
		for(ll i = x; i < x+3; i++)
			for(ll j = y; j < y+3; j++)
			{
				a[i][j]++;
				if(a[i][j] == 9)
				{
					cout << q;
					return 0;
				}
			}
	}
	cout << "-1"; 
	return 0;
}
