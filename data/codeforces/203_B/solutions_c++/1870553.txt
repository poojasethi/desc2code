#include <cstdio>
using namespace std;

int grid[1005][1005] = {0};

int main()
{
    int n, m, x, y;
    scanf("%d%d", &n, &m);

    for(int i = 1; i <= m; i++)
    {
        scanf("%d%d", &x, &y);
        for(int r = x; r < x+3; r++)
            for(int c = y; c < y+3; c++) {
                ++grid[r][c];
                if(grid[r][c] == 9) {
                    printf("%d\n", i);
                    return 0;
                }
            }
    }


    printf("-1\n");
    return 0;
}
