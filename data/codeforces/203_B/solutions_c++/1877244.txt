#include<iostream>
#include<cstdio>
using namespace std;

int map[1100][1100];
int n,m,x,y;

int main()
{
	bool ok=false;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;++i)
	{
		scanf("%d%d",&x,&y);
		for(int j=x;j<x+3;++j)
			for(int k=y;k<y+3;++k)
			{
				++map[j][k];
				if(map[j][k]==9)
				{
					cout<<i<<endl;
					return 0;
				}
			}
	}
	cout<<-1<<endl;
	return 0;
}
