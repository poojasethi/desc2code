n, m = map(int, raw_input().split())
a = [[0 for i in xrange(1010)] for j in xrange(1010)]
def isok(x, y):
    for i in xrange(x, x+3):
        for j in xrange(y, y+3):
            a[i][j] += 1
            if a[i][j] == 9:
                return True
    return False

for i in xrange(1, m+1):
    x, y = map(int, raw_input().split())
    if isok(x, y):
        print i
        quit()
