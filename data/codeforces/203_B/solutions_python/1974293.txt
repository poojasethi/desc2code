R = lambda:map(int,raw_input().split())

n, m = R()
t = [[0]*(n+2) for i in range(n+2)]

def f():
    for a in xrange(m):
        x, y = R()
        for i in range(x-1, x+2):
            for j in range(y-1, y+2):
                t[i][j] += 1
    
                if t[i][j] == 9:
                    print a + 1
                    return
    print -1

f()
