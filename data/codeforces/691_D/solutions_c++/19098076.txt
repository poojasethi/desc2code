#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<cstring>
using namespace std;
const int N=1e6 + 10;
int f[N], a[N], n,m;
int get_fa(int x){
	return x==f[x] ? x : f[x] = get_fa(f[x]);
}
vector<int> num[N], pos[N];
int main(){
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		f[i] = i;
	}
	for(int i=1;i<=m;i++){
		int p,q;
		scanf("%d%d",&p,&q);
		f[get_fa(p)] = get_fa(q);
	}
	for(int i=1;i<=n;i++){
		int k = get_fa(i);
		pos[k].push_back(i);
		num[k].push_back(-a[i]);
	}
	for(int i=1;i<=n;i++){
		sort(pos[i].begin(), pos[i].end());
		sort(num[i].begin(), num[i].end());
		for(int j=0;j<pos[i].size();j++)
			a[pos[i][j]] = -num[i][j];
	}
	for(int i=1;i<=n;i++)	
		printf("%d%c",a[i],i==n?'\n':' ');
}
