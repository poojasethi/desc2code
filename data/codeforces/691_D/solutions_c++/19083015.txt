#include <bits/stdc++.h>
using namespace std;

int A[1000006], visit[1000006], B[1000006];
vector<int> X[1000006], Y[1000006];

void dfs(int i, int lvl)
{
    if(visit[i])
        return;
    
    visit[i] = lvl;
    Y[lvl].push_back(A[i]);
    
    for(int j=0; j<X[i].size(); j++)
        if(!visit[X[i][j]])
            dfs(X[i][j], lvl);
}

int main() {
	
	int N, M;
	scanf("%d %d", &N, &M);
	
	for(int i=1; i<=N; i++)
	    scanf("%d", &A[i]);
	
	for(int i=1; i<=M; i++)
	{
	    int x, y;
	    scanf("%d %d", &x, &y);
	    
	    X[x].push_back(y);
	    X[y].push_back(x);
	}
	
	int lvl = 0;
	
	for(int i=1; i<=N; i++)
	    if(!visit[i])
	        dfs(i, ++lvl);
	
	for(int i=1; i<=lvl; i++)
	{
	    sort(Y[i].begin(), Y[i].end());
	    reverse(Y[i].begin(), Y[i].end());
	}
	
	for(int i=1; i<=N; i++)
	{
	    printf("%d ", Y[visit[i]][B[visit[i]]]);
	    B[visit[i]]++;
	}
	
	return 0;
}
