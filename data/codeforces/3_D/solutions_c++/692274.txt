#include <iostream>
#include <queue>

using namespace std;

int main(int argc, char* argv[])
{
	string s;
	cin >> s;
	
	priority_queue<pair<int,int> > q;
	
	int t = 0;
	long long r = 0;
	
	for (int i = 0; i < s.size(); ++i)
	{
		if (s[i] == '(')
			t++;
		else if (s[i] == ')')
			t--;
		else
		{
			int a, b;
			cin >> a >> b;
			
			s[i] = ')';
			r += b;
			q.push(make_pair(b - a, i));
			t--;
		}
		
		if (t < 0)
		{
			if (q.empty())
				break;
			
			s[q.top().second] = '(';
			r -= q.top().first;
			q.pop();
			t += 2;
		}		
	}
	
	if (t == 0)
	{
		cout << r << endl;
		cout << s << endl;
	}
	else
		cout << -1 << endl;
	
	return 0;
}
