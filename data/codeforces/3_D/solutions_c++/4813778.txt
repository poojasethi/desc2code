#include <iostream>
#include <cstring>
#include <string>
#include <queue>
#include <utility>
using namespace std;

priority_queue<pair<long long, long long> >q;
string s;

int main()
{
	long long cost = 0, sum = 0;
	cin >> s;
	for(int i = 0; i < s.size(); i++)
	{
		if(s[i] == '(')		sum++;
		else if(s[i] == ')')	sum--;
		else 
		{
		int x, y;
		cin >> x >> y;
		q.push(make_pair(y - x, i));
		cost += y;
		sum--;
		s[i] = ')';

		}
		if(sum < 0)
		{
			if(q.empty()) break;
			pair<long long, long long> p = q.top();
			q.pop();
			sum += 2;
			cost -= p.first;
			s[p.second] = '(';
		}
	}
	if(sum != 0)	cout << -1 << endl;
	else
	{
		cout << cost << endl;
		cout << s << endl;
	}
	return 0;
}
