s=[raw_input() for i in range(8)]
r=[(7,0)]
for step in range(8):
  t=[]
  for x,y in r:
    for dx in [-1,0,1]:
      for dy in [-1,0,1]:
        xx = x+dx
        yy = y+dy
        if (xx,yy) not in t:
          if 0<=xx<8 and 0<=yy<8:
            ok = True
            if xx-step>=0:
              ok &= s[xx-step][yy]!='S'
            if xx-step-1>=0:
              ok &= s[xx-step-1][yy]!='S'
            if ok:
              t.append((xx,yy))
  r=t
if r:
  print 'WIN'
else:
  print 'LOSE'
