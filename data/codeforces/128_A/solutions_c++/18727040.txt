#include<bits/stdc++.h>
using namespace std;
char s[8][15];
void dfs(int x,int y,int k)
{
	 int i,j;
	 if(k>=7)
	 	{cout<<"WIN";exit(0);}
     for(i=-1;i<=1;i++)
     	for(j=-1;j<=1;j++)
     	{
     		if(x+i<0 || x+i>7 || y+j<0 || y+j>7)continue;
     		if(x+i-k>=0 && s[x+i-k][y+j]=='S')continue;
     		if(x+i-k-1>=0 && s[x+i-k-1][y+j]=='S')continue;  
     		dfs(x+i,y+j,k+1);
     	}
}
int main()
{
   int i;
   ios::sync_with_stdio(false);cin.tie(0);
   for(i=0;i<8;i++)
   	cin>>s[i];
   dfs(7,0,0);
    cout<<"LOSE";
   return 0;
}
