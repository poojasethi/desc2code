#include<iostream>
#include<cstdio>
#include<cstring>

using namespace std;

char map[10][10];
int dirX[9] = {-1, -1, -1, 0, 0, 0, 1, 1, 1};
int dirY[9] = {-1, 0, 1, -1, 0, 1, -1, 0, 1};

bool dfs(int s,int x,int y)
{
  if(s>8) return true;
  for(int i=0;i<9;i++)
  {
    int nx=x+dirX[i];
    int ny=y+dirY[i];
    if(nx < 0 || nx > 7 || ny < 0 || ny > 7) continue;
    if(nx-s >= 0 && map[nx-s][ny] == 'S') continue;
    if(nx-s-1 >= 0 && map[nx-s-1][ny] == 'S') continue;
    if(dfs(s+1,nx,ny)) return true;
  }
  return false;
}
int main()
{
  for(int i=0; i<8;i++)
    scanf("%s", map[i]);
  if(dfs(0,7,0)) 
    printf("WIN\n");
  else 
    printf("LOSE\n");
  return 0;
}
