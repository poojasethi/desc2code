#include <bits/stdc++.h>
using namespace std;
int dx[]={0,1,1,1,0,-1,-1,-1,0};
int dy[]={0,-1,0,1,1,1,0,-1,-1};
char ss[10][10];
int flag = 0;
void dfs(int i,int j,int t)
{
    if(t>=8||flag==1)
    {
        flag=1;
        return ;
    }
    else
    {
        for(int k=0;k<=8;k++)
        {
            int x=i+dx[k];
            int y=j+dy[k];
            if(x>=0&&x<=7&&y>=0&&y<=7)
            {
                if(ss[x-t][y]!='S'&&(ss[x-t-1][y]!='S')) dfs(x,y,t+1);
                if(flag==1) return;
            }
        }
    }
}
int main()
{
      int i,j;
    for(i=0;i<8;i++)
      scanf("%s",ss[i]);
      dfs(7,0,0);
      if(flag==1) printf("WIN\n");
      else printf("LOSE\n");
}
