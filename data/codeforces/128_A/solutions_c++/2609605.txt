#include <iostream>
using namespace std;
const int maxn=20;
char mat[maxn][maxn];
const int dx[]={0,0,0,1,1,1,-1,-1,-1};
const int dy[]={-1,0,1,-1,0,1,-1,0,1};
int bomb;
void bt(int x,int y,int z)
{
    if(z==8)
    {
        bomb=1;
        return;
    }
    for(int i=0;i<9;i++)
    {
        int xx=x+dx[i];
        int yy=y+dy[i];
        int zz=z+1;
        if(xx>0 && yy>0 && xx<=8 && yy<=8)
            if((xx-zz<=0 || mat[xx-zz][yy]!='S') && (xx-zz+1<=0 || mat[xx-zz+1][yy]!='S'))
                bt(xx,yy,zz);
    }
}
int main()
{
    for(int i=1;i<=8;i++)
        for(int j=1;j<=8;j++)
            cin>>mat[i][j];
    bt(8,1,0);
    if(bomb==1)
        cout<<"WIN"<<endl;
    else
        cout<<"LOSE"<<endl;

    return 0;
}
