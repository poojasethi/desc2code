#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;

char b[100][100];
int dx[] = {1,1,1,0,0,-1,-1,-1,0};
int dy[] = {-1,1,0,1,-1,0,1,-1,0};

bool win(int x,int y,int det)
{
    //cout << x << "-" << y << "-" << det << '\n';
    if( det==8 ) return true;

    for(int i=0;i<9;i++)
    {
        int xx = x+dx[i],yy = y+dy[i];
        if(xx>=0&&xx<8&&yy>=0&&yy<8&&b[xx+det][yy]!='S'&&b[xx+det+1][yy]!='S'&&win(xx,yy,det+1))
            return true;
    }
    return false;
}

int main()
{
    for(int i=7;i>=0;i--)
        cin.getline(b[i],100);

    if( win(0,0,0) )
        cout << "WIN\n";
    else
        cout << "LOSE\n";

    return 0;
}
