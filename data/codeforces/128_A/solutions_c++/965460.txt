#include<cstdio>
int dx[]={0,0,1,-1,1,1,-1,-1,0};
int dy[]={1,-1,0,0,-1,1,1,-1,0};
char s[10][10];
bool check(int x,int y)
{
    if (x>=1&&x<=8&&y>=1&&y<=8) return 1;
    return 0;
}
int dfs(int x,int y,int t)
{
    if (t>=8) return 1;
    if (s[x][y]=='A') return 1;
    if (x-t>=1) if (s[x-t][y]=='S') return 0;
    //printf("%d %d\n",x,y);
    //getchar();
    int flag=0;
    for(int d=0;d<9;d++)
    {
        int nx=x+dx[d];
        int ny=y+dy[d];
        if (check(nx,ny))
        {
            if (nx-t>=1) if (s[nx-t][ny]=='S') continue;
            flag|=dfs(nx,ny,t+1);
        }
    }
    return flag;
}
int main()
{
    for(int i=1;i<=8;i++) gets(s[i]+1);
    if (dfs(8,1,0)) puts("WIN");
    else puts("LOSE");
}
