#include <cstdio>
#include <cstring>


char ma[10][10];
int dirx[9] = {-1,-1,-1,0,0,0,1,1,1};
int diry[9] = {-1,0,1,-1,0,1,-1,0,1};

void init()
{
	int i;
	for(i = 0;i < 8;i ++)
	{
		scanf("%s",ma[i]);
	}

}

bool isvalid(int x,int y,int t)
{
	if(x < 0 || x > 7 || y < 0 || y > 7 ) return false;
	if(ma[x-t+1][y] == 'S') return false;
	if(ma[x-t][y] == 'S' && x - t >= 0) return false;
	return true;
}


bool dfs(int x,int y,int t)
{
	int i;
	if(t >= 8) return true;
	for(i = 0;i < 9;i ++)
	{
		if(isvalid(x+dirx[i],y+diry[i],t+1))
		{
			if(dfs(x + dirx[i],y + diry[i],t + 1))
				return true;
		}
	}
	return false;
}

void work()
{
	bool f;
	f = dfs(7,0,0);
	if(f) printf("WIN\n");
	else printf("LOSE\n");
}
