#include <stdio.h>

char map[8][8];
int tag=0;

int dx[9]={0,1,-1,0,0,1,-1,-1,1};
int dy[9]={0,0,0,1,-1,1,-1,1,-1};

bool ok(int a,int b)
{
    if(a<0||b<0)
        return 0;
    if(a>7||b>7)
        return 0;
    return 1;
}

void dfs(int i, int j, int t)
{
    if(tag||i<=t||t>7)
    {
        tag=1;
        return ;
    }
    for(int k = 0; k < 9; ++ k)
    {
        int ii,jj;
        ii=i+dx[k];
        jj=j+dy[k];
        if(ok(ii,jj)&&map[ii-t][jj]!='S'&&(ii<=t||map[ii-1-t][jj]!='S'))
            dfs(ii,jj,t+1);
        if(tag)
            return ;
    }
}

int main()
{
    for(int i = 0; i < 8; ++ i)
        scanf("%s",map[i]);
    dfs(7,0,0);
    if(tag)
        puts("WIN");
    else
        puts("LOSE");
    return 0;
}
