#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>

#define REP(i,n) for( (i)=0 ; (i)<(n) ; (i)++ )
#define rep(i,x,n) for( (i)=(x) ; (i)<(n) ; (i)++ )

int dx[]={0,0,0,1,1,1,-1,-1,-1};
int dy[]={-1,0,1,-1,0,1,-1,0,1};
char map[9][9];

int dfs(int x,int y,int moves)
{
    if(moves>8) return 1;
    int i , a , b ;
    REP(i,9)
    {
        a=x+dx[i] , b=y+dy[i] ;
        if( a>=1 && a<=8 && b>=1 && b<=8 && map[a-moves][b]!='S' && map[a-moves-1][b]!='S' ) if(dfs(a,b,moves+1)) return 1;
    }
    return 0;
}
int main()
{
    int i ;
    rep(i,1,9) gets(map[i]+1);
    if(dfs(8,1,0)) printf("WIN\n");
    else printf("LOSE\n");
    return 0;
}
