#include<cstdio>
#include<queue>
using namespace std;
#define rep(i,n)	for(int i=0;i<(n);++i)
#define mp	make_pair
char s[9][9];
int dx[]={0,1,0,-1,1,1,-1,-1,0},
	dy[]={1,0,-1,0,-1,1,-1,1,0};
int main(){
	rep(i,8)gets(s[i]);
	queue<pair<pair<int,int>,int> > q;
	q.push(mp(mp(7,0),0));
	int xx,ss,a,b;
	for(pair<pair<int,int>,int> t;!q.empty();){
		t=q.front();q.pop();
		xx=t.first.first-t.second;
		if(xx>=0 && s[xx][t.first.second]=='S')continue;
		if(s[t.first.first][t.first.second]=='A')return puts("WIN"),0;
		ss=t.second;
		rep(i,9){
			a=t.first.first+dx[i];
			b=t.first.second+dy[i];
			if(a>=0 && a<8 && b>=0 && b<8){
				xx=a-ss;
				if(xx<0||s[xx][b]!='S')q.push(mp(mp(a,b),ss+1));
			}
		}
	}
	puts("LOSE");
	return 0;
}
