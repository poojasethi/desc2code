#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
using namespace std;
char map[10][10];
int dx[9]={0,-1,-1,0,1,1,1,0,-1};
int dy[9]={0,0,1,1,1,0,-1,-1,-1};
bool dfs(int x,int y,int step)
{
	if(step>8) return 1;
	for(int i=0;i<9;i++)
	{
		int a=dx[i]+x,b=dy[i]+y;
		if(a<1||a>8||b<1||b>8) continue;
		if(a>step&&map[a-step][b]=='S') continue;
		if(a>step+1&&map[a-step-1][b]=='S') continue;
		if(dfs(a,b,step+1)) return 1;
	}
	return 0;
}
int main()
{
	for(int i=1;i<=8;i++)
		scanf("%s",map[i]+1);
	if(dfs(8,1,0))	printf("WIN\n");
	else printf("LOSE\n");
	return 0;
}
