#include<iostream>
#include<string>
#include<cstring>
using namespace std;
int dx[]={-1,-1,-1,0,1,1,1,0,0},dy[]={1,0,-1,-1,-1,0,1,1,0};
bool b[10][8][8];
bool dfs(int d,int x,int y){
	if(d==10){
		return true;
	}
	for(int i=0;i<9;i++){
		int nx=x+dx[i],ny=y+dy[i];
		if(0<=nx && nx<8 && 0<=ny && ny<8 && !b[d][nx][ny] && !b[d+1][nx][ny]){
			if(dfs(d+1,nx,ny)){
				return true;
			}
		}
	}
	return false;
}
int main(){
	string s[8];
	for(int i=0;i<8;i++)
		cin>>s[i];
	memset(b,false,sizeof(b));
	for(int x=0;x<8;x++){
		for(int y=0;y<8;y++){
			b[0][x][y]=s[y][x]=='S';
		}
	}
	for(int i=1;i<10;i++){
		for(int x=0;x<8;x++){
			for(int y=1;y<8;y++){
				if(b[i-1][x][y-1]){
					b[i][x][y]=true;
				}
			}
		}
	}
	cout<<(dfs(0,0,7)?"WIN":"LOSE")<<endl;
}
