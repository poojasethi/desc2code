#include<stdio.h>
#define MAX 8
#define STATUE 'S'
char map[MAX][MAX+1];
bool win;
void DFS(int dropdown,int row,int col){
	if(row==0)
		win=true;
	if(win)
		return;
	for(int off_r=-1;off_r<=1;off_r++)
		for(int off_c=-1;off_c<=1;off_c++){
			int new_row=row+off_r;
			int new_col=col+off_c;
			if(new_row>=0 && new_row<MAX && new_col>=0 && new_col<MAX)
				if(new_row-dropdown<0 || (new_row-dropdown>=0 && map[new_row-dropdown][new_col]!=STATUE))
					if(new_row-dropdown-1<0 || (new_row-dropdown-1>=0 && map[new_row-dropdown-1][new_col]!=STATUE))
						DFS(dropdown+1,new_row,new_col);
		}
}
int main(){
	for(int r=0;r<MAX;r++)
		gets(map[r]);
	win=false;
	DFS(0,MAX-1,0);
	printf("%s\n",win?"WIN":"LOSE");
	return 0;
}
