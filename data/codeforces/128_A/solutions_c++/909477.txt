#include <iostream>
#include <string>

using namespace std;

int a[10][10] = {}, b[10][10] = {};
const int dx[8] = {0, -1, -1, -1, 0, -2, -2, -2};
const int dy[8] = {-1, -1, 0, 1, 1, -1, 0, 1};

bool dfs(int x, int y, int step){
    if (x < 0 || x > 7 || y < 0 || y > 7 || a[x][y] || a[x + 1][y]
        || x > 8 - step || b[x][y]) return false;
    if (!x) return true;

    b[x][y] = step;
    bool res = false;
    for (int i = 0; i < 8; i++) if (res = dfs(x + dx[i], y + dy[i], step + 1)) break;
    b[x][y] = 0;
    return res;
}

int main(){
    string s;
    for (int i = 0; i < 8; i++){
        cin >> s;
        for (int j = 0; j < 8; j++) if (s[j] == 'S') a[i][j] = 1;
    }

    if (dfs(7, 0, 1)) cout << "WIN"; else cout << "LOSE";
}
