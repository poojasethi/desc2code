#include<stdio.h>

#define MAXNR 1000000000



int main(int agrc, char *argv[]){
  int N; scanf("%d ", &N);
  int win[2001], sell[2001];
  bool used[5001];

  for(int i=0; i<=2000; i++){
    win[i] = -1;
    sell[i] = -1;
  }
  for(int i=0; i<N; i++){
    used[i] = true;
    char st[20]; int value;
    scanf("%s %d ", st, &value);
    if(st[0] == 'w'){
      if(sell[value]<0){
        win[value] = i;
      }
    }else{
      if(win[value]>=0){
        sell[value] = i;
      }
    }
  }
  bool sold[2001];
  for(int p=2000; p>=0; p--){
    sold[p] = win[p]>=0 && sell[p]>=0;
    if(sold[p]){
      for(int i=win[p]; i<=sell[p] && sold[p]; i++)
        sold[p] &= used[i];
      if(sold[p]){
        for(int i=win[p]; i<=sell[p] && sold[p]; i++)
          used[i] = false;
      }
    }
  }
  const int le = 100;
  int res[le];
  int pow[le];
  for(int i=0; i<le; i++){
    res[i] = 0;
    pow[i] = 0;
  }
  pow[le-1] = 1;
  for(int x=0; x<=2000; x++){
    int ov;  
    if(sold[x]){
      ov = 0;
      for(int i=le-1; i>=0; i--){
        res[i] += pow[i] + ov;
        if(res[i]>=MAXNR){
          ov = 1;
          res[i] -= MAXNR;
        }else{
          ov = 0;
        }
      }
    }
    ov = 0;
    for(int i=le-1; i>=0; i--){
      pow[i] += pow[i] + ov;
      if(pow[i] >= MAXNR){
        pow[i] -= MAXNR;
        ov = 1;
      }else{
        ov = 0; 
      }
    }
  }
  int s = 0;
  while(res[s] == 0 && s<le-1)
    s++;
  printf("%d", res[s++]);
  while(s<le){
    printf("%09d", res[s++]);
  }
  printf("\n");
  return 0;
}
