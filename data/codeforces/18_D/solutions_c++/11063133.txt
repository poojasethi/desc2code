#include <bits/stdc++.h>

using namespace std;
const char FIN[]="sellsticks.i4";
const char FOU[]="sellsticks.o4";
const int MAXN = 5005;

string Maxnum(string x, string y){
    while(x.size() < y.size()) x = "0"+x;
    while(x.size() > y.size()) y = "0"+y;
    if (x > y) return x; else return y;
}
string Add(string x, string y){
    string z = "";
    int s, carry = 0;
    while(x.size() < y.size()) x = "0"+x;
    while(x.size() > y.size()) y = "0"+y;
    for(int i = x.size()-1; i >= 0; i--){
        s = (x[i]+y[i]+carry-96);
        if (s > 9) {carry = 1; s %= 10;}
        else carry = 0;
        z = char(s+48)+z;
    }
    if(carry > 0) z = "1"+z;
    return z;
}

string p[MAXN], power[2001];

void init(){
    power[0]="1"; power[1]="2";
    for(int i = 2; i <= 2000; i++) power[i] = Add(power[i-1],power[i-1]);
}

int main()
{
    #ifndef ONLINE_JUDGE
    freopen(FIN,"r",stdin);
    freopen(FOU,"w",stdout);
    #endif // ONLINE_JUDGE
    init();
    int  a[MAXN], n, num, xx, yy;
    a[0] = 0; p[0] = "0";
    cin >> n;
    string s, x, y;
    p[0]="0";
    for(int j, i = 1; i <= n; i++){
        cin >> s >> num;
        p[i] = p[i-1];
        //p[i] = p[i-1];
        if(s[0]=='w') a[i] = num;
        else{
            a[i] = -1;
            for(j = i-1; j > 0; j--)
            if(a[j] == num){
                x = p[j];
                break;
            }
            if(j>0){
                y = power[num];
                x = Add(x,y);
                p[i] = Maxnum(p[i],x);
            }
        }
    }
    cout << p[n] << endl;
    return 0;
}
