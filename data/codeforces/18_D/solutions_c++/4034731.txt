#include <iostream>
#include <cstdio>
#include <map>
#define MAXN 700
using namespace std;
int N;

int num[2010][MAXN+1];
int f[5010][MAXN+1];
int a[5010];
int b[5010];
map <int,int> hash;

void init(){
    char str[10];
    scanf("%d\n",&N);
    for (int i=1;i<=N;++i){
        scanf("%s",str);
        //gets(str);
        if (str[0]=='w')
            a[i]=1;
        else
            a[i]=0;
        scanf("%d\n",&b[i]);
    }
    //for (int i=1;i<=N;++i)
        //printf("%d %d\n",a[i],b[i]);
}

int cmp(int a[],int b[]){
    for (int i=0;i<=700;++i){
        if (a[i]>b[i])
            return 1;
        if (a[i]<b[i])
            return 0;
    }
    return 0;
}

void cpy(int a[],int b[]){
    for (int i=0;i<=MAXN;++i)
        a[i]=b[i];
}

void add(int a[],int b[]){
    int p=0;
    for (int i=MAXN;i>=0;--i){
        a[i]=(a[i]+b[i]+p);
        p=a[i]/10;
        a[i]=a[i]%10;
    }
}

void output(int a[]){
    bool flag=0;
    for (int i=0;i<=MAXN;++i){
        if (a[i]>0) flag=1;
        if (flag) printf("%d",a[i]);
    }
    if (!flag) printf("0");
    printf("\n");
}

void prework(){
    num[0][MAXN]=1;
    for (int i=1;i<2010;++i){
        cpy(num[i],num[i-1]);
        add(num[i],num[i]);
    }
}

void work(){
    for (int i=N;i>0;--i){
        cpy(f[i],f[i+1]);
        if (a[i]==0)
            hash[b[i]]=i;
        else
        if (hash[b[i]]!=0){
            cpy(f[0],f[hash[b[i]]]);
            add(f[0],num[b[i]]);
            if (cmp(f[0],f[i]))
                cpy(f[i],f[0]);
        }
    }
    output(f[1]);
}

int main(){
    //freopen("input.txt","r",stdin);
    init();
    prework();
    work();
    return 0;
}
