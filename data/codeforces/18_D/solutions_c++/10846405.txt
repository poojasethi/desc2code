#include<bits/stdc++.h>

using namespace std;

int C[750],x[5050];
int xr[2020][750];
int ans[5050][750];
int n,i,j,ind[2020];
string s[5050];

void upd(int A,int B,int x)
 {
  for (int i=1;i<=700;i++) C[i]=0;
  for (int i=700;i>=1;i--)
   {
    C[i]+=ans[B][i];
    if (x >= 0) C[i]+=xr[x][i];
    C[i-1]+=C[i]/10;
    C[i]%=10;
   }
  bool f=0;
  for (int i=1;i<=700;i++)
   {
    if (f) { ans[A][i]=C[i]; continue; }
    if (C[i] < ans[A][i]) return; else
    if (C[i] > ans[A][i]) { f=1; i--; }
   }
 }

void print(int u)
 {
  int i=0;
  while (!ans[u][i] && i < 700) i++;
  for (;i<=700;i++)
   cout<<ans[u][i];
  cout<<endl;
 }

main()
{
 xr[0][700]=1;
 for (i=1;i<=2000;i++)
  for (j=700;j>=1;j--)
   {
    xr[i][j]+=xr[i-1][j]*2;
    xr[i][j-1]+=xr[i][j]/10;
    xr[i][j]%=10;
   }
 cin>>n;
 for (i=1;i<=n;i++)
  {
   cin>>s[i]>>x[i];
   if (s[i] == "sell") ind[x[i]]=i;
  }
 for (i=1;i<=n;i++)
  {
   upd(i,i-1,-1);
   if (s[i] == "sell") continue;
   upd(ind[x[i]],i,x[i]);
  }
 print(n);
}
