#include <cstdio>
#include <cstdlib>
#include <cstring>
#define X 2001
#define N 5000

int bigint[1000] = {0}, len = 0;

void carry(){
    int i;
    for(i=0; i<=len; i++)
        if(bigint[i] >= 10){
            bigint[i+1] ++;
            bigint[i] -= 10;
        }
    if(bigint[len+1] > 0)
        len ++;
}

int main(){
    int n, i, x, loc[X], list[N], max = X-1;
    bool win[N], got[N] = {0}, ans[X] = {0};
    char tmp[10];
    memset(loc, -1, sizeof(loc));
    
    scanf("%d", &n);
    for(i=0; i<n; i++){
        scanf("%s %d", tmp, &x);
        list[i] = x;
        win[i] = (strcmp(tmp, "win") == 0);
        if(!win[i])
            loc[x] = i;
    }
    while(loc[max] == -1)
        max --;
    for(x=max; x>=0; x--)
        if(loc[x] != -1 && !got[loc[x]]){
            for(i=loc[x]-1; i>=0 && !got[i]; i--)
                if(list[i] == x && win[i])
                    break;
            if(i>=0 && !got[i]){
                ans[x] = 1;
                for(; i<loc[x]; i++)
                    got[i] = 1;
            }
        }
    
    for(x=max; x>=0; x--){
        for(i=0; i<=len; i++)
            bigint[i] <<= 1;
        if(ans[x])
            bigint[0] ++;
        carry();
    }
    for(i=len; i>=0; i--)
        printf("%d", bigint[i]);
    printf("\n");
    
    //system("pause>nul");
    return 0;
}
