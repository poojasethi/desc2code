#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
const int i=610;
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,s[5000],d[5000],h[2001],f[i];
  char a[5];
  for(w=0;w<2001;w++)
   h[w]=0;
  scanf("%d",&q);
  for(w=0;w<q;w++){
   scanf("%s%d",a,&d[w]);
   if(a[0]=='w')
    s[w]=0;
   else
    s[w]=1;}
  while(1){
   t=-1;
   for(w=0;w<q;w++)
    if((s[w]==1)&&(t<d[w])){
     t=d[w];
     r=w;}
   if(t<0)
    break;
   for(w=r-1;(w>=0)&&(s[w]>=0);w--)
    if((s[w]==0)&&(d[w]==t))
     break;
   if((w<0)||(s[w]<0)){
    s[r]=2;
    continue;}
   h[t]=1;
   for(;w<=r;w++)
    s[w]=-1;}
  for(w=0;w<i;w++)
   f[w]=0;
  for(w=2000;w>=0;w--){
   for(e=1;e<i;e++){
    f[e]<<=1;
    if(f[e-1]>9){
     f[e-1]-=10;
     f[e]++;}}
   f[1]+=h[w];}
  for(w=i-1;(w>1)&&(!f[w]);w--);
  for(;w;w--)
   printf("%d",f[w]);
  return 0;}
