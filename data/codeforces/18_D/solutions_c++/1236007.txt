#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int b[11111],f[5011][700],n,m,a[2001][700];
int getch()
{
	char c=getchar();
	while (c!='w' && c!='s') c=getchar();
	if (c=='w') { getchar();getchar();return 1;}
	if (c=='s') { getchar();getchar();getchar();return 2;}
}
void cheng2(int *a)
{
	for (int i=1;i<=a[0];++i)
		a[i]=a[i]*2;
	for (int i=1;i<a[0];++i)
	{
		a[i+1]+=a[i]/10;
		a[i]=a[i]%10;
	}
	while (a[a[0]]>9)
	{
		a[a[0]+1]=a[a[0]]/10;
		a[a[0]]=a[a[0]]%10;
		a[0]++;
	}
}
void jia(int *a,int *b,int *c)
{
	c[0]=max(a[0],b[0]);
	c[c[0]]=0;c[c[0]+1]=0;c[c[0]+2]=0;c[c[0]+3]=0;c[c[0]+4]=0;
	for (int i=1;i<=c[0];++i)
		c[i]=a[i]+b[i];
	for (int i=1;i<=c[0];++i)
	{
		c[i+1]+=c[i]/10;
		c[i]=c[i]%10;
	}
	if (c[c[0]+1]!=0) c[0]++;
}
bool dayu(int *a,int *b)
{
	if (a[0]>b[0]) return true;
	if (a[0]<b[0]) return false;
	int t=a[0];
	while (t>0)
	{
		if (a[t]>b[t]) return true;
		if (a[t]<b[t]) return false;
		t--;
	}
	return false;
}
int main()
{
	scanf("%d",&n);
	a[0][0]=1;a[0][1]=1;
	for (int i=1;i<=2000;++i)
	{
		memcpy(a[i],a[i-1],(a[i-1][0]+4)*4);
		cheng2(a[i]);
	}
	for (int i=1;i<=n;++i)
	{
		memcpy(f[i],f[i-1],(f[i-1][0]+4)*4);
		int x=getch(),y;
		scanf("%d",&y);
		if (x==1)
			b[y]=i;
		else
		{
			int j=b[y];if (j==0) continue;
			int Tmp[700];
			jia(f[j],a[y],Tmp);
			if (dayu(Tmp,f[i]))
				memcpy(f[i],Tmp,(Tmp[0]+4)*4);
		}
	}
	if (f[n][0]==0) cout<<0;
	else
	for (int i=f[n][0];i>=1;--i)
		printf("%d",f[n][i]);
	printf("\n");
}
