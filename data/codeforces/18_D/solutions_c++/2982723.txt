#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<cstring>
using namespace std;
const int mm=5005;
const int mod=10;
int f[mm][mm];
int a[mm],ans[mm];
char s[mm];
vector<pair<int,int> >b;
int main()
{
  int tt;
  memset(f,0,sizeof(f));
  f[0][1]=1;f[0][0]=1;tt=0;
  for(int i=1;i<2002;i++)
  {for(int j=1;j<=f[i-1][0];j++)
   {
    f[i][j]+=f[i-1][j]+f[i-1][j]+tt;
    tt=f[i][j]/mod;
    f[i][j]%=mod;
    ++f[i][0];
   }
   if(tt)f[i][++f[i][0]]=tt;
   tt=0;
  }
  ///for(int i=1;i<110;i++,cout<<"\n")
   /// for(int j=f[i][0];j>0;j--)cout<<f[i][j];
  int cas;
  while(cin>>cas)
  { b.clear();
    for(int i=0;i<cas;i++)
    {cin>>s>>a[i];
      if(s[0]=='s')b.push_back(make_pair(a[i],i)),a[i]=-1;
    }
    sort(b.rbegin(),b.rend());
    int x,y;
    memset(ans,0,sizeof(ans));
    ans[0]=1;
    for(int i=0;i<b.size();i++)
    { x=b[i].first;y=b[i].second;
      for(int j=y-1;j>=0&&a[j]!=-2;j--)
        if(a[j]==x)
      { tt=0;
        int bit;
        if(ans[0]>f[x][0])bit=ans[0];
        else bit=f[x][0];
        for(int k=1;k<=bit;k++)
        {ans[k]+=f[x][k]+tt;
         tt=ans[k]/mod;
         ans[k]%=mod;
        }
        if(tt)ans[++bit]+=tt;
        ans[0]=bit;
       /// if(tt)ans[f[x][0]+1]+=tt;
        ///if(ans[0]<=f[x][0]){ans[0]=f[x][0];if(tt)++ans[0];}
        for(int k=j;k<=y;k++)a[k]=-2;
        break;
      }
    }
    for(int i=ans[0];i>=1;i--)
      cout<<ans[i];
    cout<<"\n";
  }
}
