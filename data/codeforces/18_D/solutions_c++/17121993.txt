#include <iostream>
#include <string.h>
#include <stdlib.h>
#include <algorithm>
#include <math.h>
#include <stdio.h>
#include <string>
#include <vector>
#include <bitset>

using namespace std;
int n;
int pre[2005];
int a[5005];
int ans[5005];
vector<int> s;
bitset <2005> dp[5005],y;
string b;
int main()
{
   scanf("%d",&n);
   memset(pre,0,sizeof(pre));
   s.clear();

   dp[0]=0;
   for(int i=1;i<=n;i++)
   {
       cin>>b>>a[i];
       if(b[0]=='w')
       {pre[a[i]]=i;dp[i]=dp[i-1];}
       else
       {
           if(!pre[a[i]]) {dp[i]=dp[i-1];continue;}
           y=dp[pre[a[i]]];
           y[a[i]]=1;
           for(int j=2000;j>=0;j--)
           {
               if(dp[i-1][j]>y[j]){dp[i]=dp[i-1];break;}
               if(dp[i-1][j]<y[j]){dp[i]=y;break;}
               if(j==0){dp[i]=y;}
           }
       }
   }
   s.push_back(0);
   for(int i=dp[n].size()-1;i>=0;i--)
   {
       int k=0;
       for(int j=0;j<s.size();j++)
       {
           int now=s[j];
           s[j]=(now*2+k)%10;
           k=(now*2+k)/10;
       }
       if(k)
        s.push_back(k);
       if(dp[n][i])
       {
           int k=0;s[0]++;
           for(int j=0;j<s.size();j++)
           {
               int now=s[j];
               s[j]=(now+k)%10;
               k=(now+k)/10;
           }
           if(k)
            s.push_back(k);
       }
   }
   for(int i=s.size()-1;i>=0;i--)
    printf("%d",s[i]);
   cout<<endl;
   return 0;

}
