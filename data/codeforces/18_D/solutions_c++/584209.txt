#include <stdio.h>

int a[5005], d[5005], na, nd;
int b[2005];
int u[2005];

void add() {
	if(nd>na) na=nd;
	for(int i=0;i<nd;i++)
		a[i] += d[i];
	for(int i=0;i<na;i++)
		if(a[i]>=10) {
			if(i==na-1) ++na;
			a[i+1]+=a[i]/10;
			a[i]%=10;
		}
}
void dbl() {
	for(int i=0;i<nd;i++)
		d[i] = d[i]+d[i];
	for(int i=0;i<nd;i++)
		if(d[i]>=10) {
			if(i==nd-1) ++nd;
			d[i+1]+=d[i]/10;
			d[i]%=10;
		}
}
void print() {
	for(int i=na-1;i>=0;--i)
		printf("%d", a[i]);
	puts("");
}
int main(void) {
	int n;
	scanf("%d", &n);
	int p[5005], pos[2005];
	for(int i=0;i<=2000;i++)
		pos[i] = -1;
	for(int i=0;i<n;i++) {
		char z[10];
		scanf("%s%d", z, &p[i]);
		
		if(z[0]=='s') {
			pos[p[i]] = i;
			if(u[p[i]]) {
				u[p[i]] = 0;
				p[i] = -p[i]-1;
			} 
		} else u[p[i]] = 1;
	}

	for(int i=2000;i>=0;--i) {
		if(pos[i]>=0 && p[pos[i]]==-i-1) {
			b[i] = 1;
			for(int j=pos[i]-1;j>=0;--j)
				if(p[j]==2012) {b[i]=0; break;}
				else if(p[j]==i) {break;}
			if(b[i]) {
				for(int j=pos[i];j>=0;--j) {
					if(p[j]==i) {p[j]=2012;break;}
					p[j]=2012;
				}
			}
		}
	}

	/*int keep=-1;
	for(int i=n-1;i>=0;--i) {
		if(p[i]<0) keep = (keep<-p[i]-1?-p[i]-1:keep);
		else if(keep == p[i]) {
			b[p[i]] = 1;
			keep = -1;
		}
	}*/
	a[0] = 0;
	na = 1;
	d[0] = 1;
	nd = 1;
	for(int i=0;i<=2000;i++){
		if(b[i]) {
			add();
		}
		dbl();
	}
	print();
	return 0;
}
