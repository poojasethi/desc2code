#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<list>
using namespace std;
int n,x,hpa[1002],tmp[1002];
bool ok[5005];
list<int> L[2005],ans;
int sel[2005];
char t[9];
int main(){
  memset(sel,-1,sizeof(sel));
  scanf("%d",&n);
  for(int i=1;i<=n;i++) ok[i]=true;
  for(int i=1;i<=n;i++){
    scanf("\n%s %d",t,&x);
    if(!strcmp(t,"win")) L[x].push_back(i);
    else sel[x]=i;
  }
  for(int i=2000;i>=0;i--){
    if(sel[i]==-1) continue;
    int gd=-1;
    for(list<int>::iterator it=L[i].begin();it!=L[i].end();it++){
      int cur=*it;
      if(cur<sel[i]) gd=cur;
    }
    if(gd==-1) continue;
    for(int j=gd;j<=sel[i];j++){
      if(!ok[j]) goto gg;
    }
    for(int j=gd;j<=sel[i];j++) ok[j]=false;
    ans.push_front(i);
    gg:;
  }
  memset(hpa,0,sizeof(hpa));
  memset(tmp,0,sizeof(tmp));
  tmp[0]=1;
  for(int i=0;i<=2000;i++){
   if(i){
    for(int j=0;j<=1000;j++) tmp[j]*=2;
    for(int j=0;j<1000;j++){
      if(tmp[j]>=10){
        tmp[j]-=10;
        tmp[j+1]++;
      }
    }
   }
    if(!ans.empty() && ans.front()==i){
      ans.pop_front();
      for(int j=0;j<=1000;j++) hpa[j]+=tmp[j];
      for(int j=0;j<1000;j++){
        if(hpa[j]>=10){
          hpa[j]-=10;
          hpa[j+1]++;
        }
      }
    }
  }
  for(int i=1000;i>=0;i--){
    if(i==0 || hpa[i]>0){
      for(int j=i;j>=0;j--) printf("%d",hpa[j]);
      printf("\n");
      return 0;
    }
  }
}
