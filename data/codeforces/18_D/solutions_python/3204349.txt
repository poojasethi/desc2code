#!/usr/bin/python2.7

a = 2003 * [-1]
res = 0

n = int(raw_input())
for i in range(n):
    line = raw_input()
    c = line.strip()[0]
    x = int(line.strip().split()[1])
    
    if c == 'w':
        a[x] = res
    elif a[x] >= 0:
        res = max(res, a[x] + 2**x)

print res
