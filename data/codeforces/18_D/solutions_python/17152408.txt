n = input()
Best = []
best = 0
Max = 2000
for i in xrange(Max+2):
    Best.append(-1)
for i in xrange(n):
    Line = raw_input().split()
    a,b = Line[0], int(Line[1])
    if a == 'win':
        Best[b] = best
    elif Best[b]!=-1:
        Best[b] = Best[b] + pow(2, b)
        best = max(best, Best[b])

print best
