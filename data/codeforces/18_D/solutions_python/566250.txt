d = {}
n = int(raw_input())
i = 0

global p
p = []
while i < n:
	i += 1
	s = raw_input().split()
	x = int(s[1])
	if s[0] == 'win':
		d[x] = i
	if s[0] == 'sell':
		if x in d:
			p.append((x, d[x], i))
p.sort()
p.reverse()

ans = 0
q = [(0,1,n)]
for x in q:
	if x[1] >= x[2]:
		continue
	for i in range(x[0], len(p)):
		if p[i][1] >= x[1] and p[i][2] <= x[2]:
			ans += 2**p[i][0]
			q.append((i+1, x[1], p[i][1]-1))
			q.append((i+1, p[i][2]+1, x[2]))
			break

print ans
