memory = [-1]*2001
earn = [0]

n = input()

for i in range(1,n+1):
    t = raw_input().split()
    str, num = t[0], int(t[1])

    if str == "win":
        memory[num] = i
        earn += [earn[i-1]]
    if str == "sell":
        if memory[num] >= 0:
            earn += [max(earn[i-1], earn[memory[num]]+2**num)]
        else:
            earn += [earn[i-1]]
print earn[n]
