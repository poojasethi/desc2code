#!/usr/bin/python
from sys import stdin

N = int(stdin.readline())

r = []
v = []
u = []

m = {}

for i in range(0, N):
    t = stdin.readline().split()

    r.append((t[0], int(t[1])))
    
    v.append(False)
    u.append(False)

    if t[0] == 'sell':
        m[int(t[1])] = i

ans = 0

x = 2001

while x >= 1:
    x = x - 1

    if not x in m:
        continue

    n = m[x]

    if u[n] or v[n]:
        continue

    if n == -1:
        break;

    mxw = -1

    i = n

    while i >= 0:
        if v[i]:
            break

        if r[i][0] == 'win' and r[i][1] == x:
            mxw = i
            break

        i = i - 1

    u[n] = True

    if mxw != -1:
        v[n] = True
        ans = ans + (1 << x)

        for i in range(mxw, n):
            v[i] = True



print ans
