dp = [-1] * 2005
ret, n = 0, input()

for i in xrange(0, n):
    todo, j = raw_input().split()
    j = int(j)
    if todo == 'win':
        dp[j] = max(dp[j], ret)
    else:
        if dp[j] != -1:
            tmp = dp[j] + 2 ** j
            if tmp > ret:
                ret = tmp
                dp[j] = -1
print ret
