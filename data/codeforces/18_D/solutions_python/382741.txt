N = int(raw_input())

dp = [-1] * 2002
dp[-1] = 0

for day in range(N):
    ndp = dp[:]
    str, x = raw_input().split()
    x = int(x)
    if str == 'win':
        ndp[x] = max(ndp[x], dp[-1])
        
    if str == 'sell':
        if dp[x] >= 0:
            ndp[-1] = max(ndp[-1], dp[x] + pow(2, x))
    dp = ndp[:]
print max(dp)
