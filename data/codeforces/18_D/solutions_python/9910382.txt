n = input();
dp = [];
dp.append(0);

pos = [];
for i in range(0,2001):
	pos.append(0);

for i in range(1,n+1):
	ipt = raw_input();
	ipt = ipt.split();
	x = (int)(ipt[1]);
	dp.append(dp[i-1]);
	if (ipt[0]=="win"):
		pos[x] = i;		
	else:
		if(pos[x] != 0):
			dp[i] = max(dp[i], pow(2,x) + dp[pos[x]-1]);

print dp[n];
