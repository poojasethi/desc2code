import sys

n = int(raw_input())
dp = [[-1 for j in range(2)] for i in range(2001)]
for i in range(2001):
    dp[i][0] = 0
for _ in range(n):
    op, val = raw_input().split()
    val = int(val)
    #print op, val, dp[val][1]
    if op == 'win':
        dp[val][1] = max(dp[val][1], dp[0][0])
        #print dp[val][1]
    else:
        if dp[val][1] >= 0:
            dp[0][0] = max(dp[0][0], dp[val][1]+2**val)
