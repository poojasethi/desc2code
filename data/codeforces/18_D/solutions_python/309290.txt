import string
N = int(raw_input())
T = [0] * N
V = [0] * N
DP = [0] * N
for i in range(0,N):
	s = raw_input()
	t = string.split(s, ' ')
	cmd = t[0]
	v = int(t[1])
	T[i] = cmd=="sell"
	V[i] = v
for i in range(0,N):
	if T[i] == 1:
		for j in range(i-1, -1, -1):
			if (T[j] == 0) & (V[i] == V[j]):
				if j > 0:
					pre = DP[j-1]
				else:
					pre = 0
				DP[i] = max(DP[i], pre + (2**V[j]))
	if (i > 0):
		pre = DP[i-1]
	else:
		pre = 0
	DP[i] = max(DP[i], pre)
print DP[N-1]
