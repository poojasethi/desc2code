
def calc(ls):
  n = len(ls)

  win = [-1] * 2010
  dp = [0] * n
  for i in range(n):
    dp[i] = dp[i-1]
    a, x = ls[i]
    if a == "win":
      win[x] = i
    elif win[x] >= 0:
      profit = 0
      if win[x] > 0:
        profit = dp[win[x]-1]
      dp[i] = max(dp[i], profit + 2**x)

  return dp[n-1]

def main():
  n = int(raw_input())

  ls = []
  for _ in range(n):
    a, b = raw_input().split()
    ls.append([a, int(b)])

  print calc(ls)

main()
