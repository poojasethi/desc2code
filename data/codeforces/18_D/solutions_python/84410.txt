#!/usr/bin/env python3

def main():
    dp = [-1]*2002
    dp[0] = 0
    for i in xrange(int(raw_input())):
        s, t = raw_input().split()
        t = int(t) + 1
        if s[0] == 'w':
            dp[t] = dp[0]
        elif dp[t] >= 0:
            dp[0] = max(dp[0], dp[t] + (1 << t-1))
    print dp[0]

if __name__ == '__main__':
    main()
