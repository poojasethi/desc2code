#include <iostream>
#include <string>
using namespace std;
const int MAXn = 4000+10;int INF=2e9;int dp[MAXn][MAXn],beg[MAXn],l[MAXn], n, k;int main(){ios::sync_with_stdio(false);cin>>n>>k;for(int i=1;i<=n;i++)cin>>beg[i]>>l[i],l[i]--;dp[0][0]=0;for(int i=1;i<=n;i++)for(int j = 0; j <= k; j++)dp[i][j]=min((j-1>=0)?dp[i-1][j-1]:INF,max(dp[i-1][j]+1,beg[i])+l[i]);int mx=0;for(int i=1;i<=n;i++)mx=max(mx,max(0,beg[i]-dp[i-1][k]-1));mx=max(mx,max(0,86400-dp[n][k]));cout<<mx<<endl;}
