#include <cstdio>

int t[4040];
int d[4040];
int r[4040][4040];

int max(int a,int b) { return b<a?a:b; }

int main() {
	//freopen("input.txt","rt",stdin);
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++) scanf("%d%d",t+i,d+i);
	for(int i=0;i<=k;i++) {
		r[i][0]=1;
		for(int j=1;j<=n;j++) {
			r[i][j]=max(r[i][j-1],t[j-1])+d[j-1];
			if (i && r[i-1][j-1]<r[i][j]) r[i][j]=r[i-1][j-1];
		}
	}
	int res=0;
	for(int j=0;j<n;j++)
		if (t[j]-r[k][j]>res) res=t[j]-r[k][j];
	if (86401-r[k][n]>res) res=86401-r[k][n];
	printf("%d\n",res);
	return 0;
}
