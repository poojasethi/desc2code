#include<cstdio>
#include<algorithm>
using namespace std;
int a,f[4096][4096],t[4096],d[4096],n,i,j,k,T;
int main(){
    scanf("%d%d",&n,&k);
    for(i=1;i<=n;++i)scanf("%d%d",t+i,d+i);
    for(i=0;i<=k;++i)f[i][0]=1;t[n+1]=86401;
    for(i=T=1;i<=n;++i)f[0][i]=((T=max(T,t[i]))+=d[i]);
    for(i=1;i<=k;++i)for(j=1;j<=n;++j)
        f[i][j]=min(max(f[i][j-1],t[j])+d[j],f[i-1][j-1]);
    for(i=0;i<=k;++i)for(j=0;j<=n;++j)
        a=max(a,t[min(n,j+k-i)+1]-f[i][j]);
    printf("%d\n",a);
    return 0;
}
