#include <iostream>
using namespace std;
int T[4001],D[4001],X[4010][4010];
int main(void)
{
	int i,j,n,k,r=0;
	cin>>n>>k;
	for (i=1;i<=n;++i)
		cin>>T[i]>>D[i],T[i]--;
	T[n+1]=86400;
	for (i=0;i<=k;++i)
		X[i][0]=0;
	for (i=1;i<=n;++i) {
		X[0][i]=max(X[0][i-1],T[i])+D[i];
		for (j=1;j<=k;++j) 
			X[j][i]=min(X[j-1][i-1],max(X[j][i-1],T[i])+D[i]);
	}
	for (i=0;i<=k;++i)
		for (j=0;j<=n-k+i;++j)
			r=max(r,T[j+k+1-i]-X[i][j]);
	cout<<r;


	return 0;
}
