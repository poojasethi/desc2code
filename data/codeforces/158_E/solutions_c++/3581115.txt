#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
const int mm=4e3+9;
const int oo=1e9;
const int tim=86401;
int dp[mm];
int main()
{
  int n,k;
  while(cin>>n>>k)
  { dp[0]=1;
     int a,b,ans=0;
    for(int i=0;i<n;++i)
      {
        cin>>a>>b;
        ans=max(ans,a-dp[k]);
        for(int j=k;~j;--j)///dp[j]使用了j次pass
          dp[j]=min(max(dp[j],a)+b,j?dp[j-1]:oo);
      }
      ans=max(ans,tim-dp[k]);
      cout<<ans<<"\n";
  }
}
