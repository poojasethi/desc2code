#include <iostream>
#include <algorithm>
using namespace std;

int n, k, t, d, ans, e[4005];
int main() {
	cin >> n >> k;
	e[0] = 1;
	while (n--) {
		cin >> t >> d;
		ans = max(ans, t-e[k]);
		for (int j = k; ~j; j--)
			e[j] = min(max(e[j],t)+d, j?e[j-1]:1<<30);
	}
	ans = max(ans, 86401-e[k]);

	cout << ans << endl;
	return 0;
}
