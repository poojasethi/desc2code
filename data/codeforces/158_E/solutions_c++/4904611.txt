#include <algorithm>
#include <iostream>
using namespace std;
int i,b,t,d[5000],n,k,e;
int main(){
    cin>>n>>k;
    d[0]=1<<17;
    d[1]=1;
    for(;i<=n;i++) {
        if(i<n)
            cin>>b>>t;
        else {
            b=86401;
            t=1;
        }
        for(int j=k+1; j>0 ;--j){
            e=max(e, b - d[j]);
            d[j]=min(d[j-1], max(d[j], b) + t);
        }
    }
    cout << e;
}
