#include <iostream>
using namespace std;

int t[4048], d[4048];
int dp[4048][4048];

int main()
{
  int n, k;
  cin>>n>>k;
  for (int i = 1; i <= n; i++) cin>>t[i]>>d[i];
  t[n+1] = 86401;
  int ans = t[1] - 1;
  for (int j = 0; j <= k; j++) dp[0][j] = 1;

  for (int i = 1; i <= n; i++) {
    for (int j = 0; j <= k; j++) {
      if(j)
	dp[i][j] = min(dp[i-1][j-1], max(dp[i-1][j], t[i]) + d[i]);
      else
	dp[i][j] = max(dp[i-1][j], t[i]) + d[i];
      ans = max(ans, t[i+1] - dp[i][j]);
    }
  }
  cout<<ans<<endl;
  return 0;
}
