

#include <iostream>
#include <algorithm>
using namespace std;

int n, k, t, d, ans, e[4005];
int main() {
	cin >> n >> k;
	e[0] = 1;
	while (n--) {
		cin >> t >> d;
		ans = max(ans, t-e[k]);
		for (int j = k; j; j--)
			e[j] = min(max(e[j],t)+d, e[j-1]);
		e[0] = max(e[0], t) + d;
		//printf("t d: ");
		//for (int j = 0; j <= k; j++) printf("%d ", e[j]);
		//printf("\n");
	}
	ans = max(ans, 86401-e[k]);

	cout << ans << endl;
	return 0;
}
