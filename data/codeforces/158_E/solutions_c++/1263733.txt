#include <iostream>
using namespace std;
int dp[4096][4096]; //[n][k]
int ts[1<<16],ds[1<<16];
int n,k;
int main() {
	cin>>n>>k;
	for(int i=0; i<n; ++i) cin>>ts[i]>>ds[i];
	for(int i=0; i<=n; ++i) {
		for(int j=0; j<i; ++j) {
			dp[i][j] = max(dp[i-1][j],ts[i-1]) + ds[i-1];
			if (j>0) dp[i][j] = min(dp[i][j], dp[i-1][j-1]);
		}
		dp[i][i] = 1;
	}
	int r=0;
	for(int i=0; i<=n; ++i) {
		int st = dp[i][k];
		int ed = i==n ? 86401 : ts[i];
		int x = ed-st;
		r = max(r,x);
	}
	cout<<r<<'\n';
}
