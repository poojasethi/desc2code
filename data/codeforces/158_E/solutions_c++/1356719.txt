#include <iostream>
#include <algorithm>

#define MAXN 4010

using namespace std;

int N,K,T[MAXN],D[MAXN],dp[MAXN][MAXN],result;

int main()
{
	cin >> N >> K;
	for (int i=0;i<N;i++)
		cin >> T[i] >> D[i];
	T[N]=86401;
	for (int i=0;i<=K;i++)
		dp[0][i]=1;
	for (int i=0;i<N;i++)
		for (int j=0;j<=K;j++)
		{
			if (!j)
				dp[i+1][j]=max(dp[i][j],T[i])+D[i];
			else
				dp[i+1][j]=min(dp[i][j-1],max(dp[i][j],T[i])+D[i]);
		}
	for (int i=0;i<=N;i++)
		result=max(result,T[i]-dp[i][K]);
	cout << result << endl;
	return 0;
}
