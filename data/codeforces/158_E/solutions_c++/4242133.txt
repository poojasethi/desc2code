#include <iostream>
#include <algorithm>

using namespace std;

const int N = 4010;
const int Day = 86400;
const int Inf = (int)1E9;

int n, k;
int s[N],d[N];
int dp[N][N];
int ans;

int main(){
	cin >> n >> k;
	for(int i=0; i<n; ++i)
		cin >> s[i] >> d[i];
	
	dp[0][0] = s[0]+d[0]-1;
	s[n] = Day+1;

	ans = s[0]-1;
	
	for(int i=1; i<=n; ++i){
		for(int j=0; j<=k; ++j){
			dp[i][j] = min( (j==0? Inf : dp[i-1][j-1]), max(dp[i-1][j]+1,s[i])+d[i]-1);
		}
		ans = max(ans, s[i] - dp[i-1][k]-1);
	}
	cout << ans << endl;
	return 0;
}
