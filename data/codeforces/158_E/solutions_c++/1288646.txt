#include <cstdio>
#include <algorithm>
using namespace std;

int n, k, t, p, d[4005], A;

int main(){
	scanf("%d %d", &n, &k);
	
	if(n == k){
		puts("86400");
		return 0;
	}
	
	while(n--){
		scanf("%d %d", &t, &p);
		for(int i = k; i >= 0; i--){
			if(d[i] >= t) d[i] += p;
			else A = max(A, t - d[i] - 1), d[i] = t + p - 1;
			if(i) d[i] = min(d[i], d[i - 1]);
		}
	}
	
	for(int i = 0; i <= k; i++) A = max(A, 86400 - d[i]);
	printf("%d\n", min(86400, A));
}
