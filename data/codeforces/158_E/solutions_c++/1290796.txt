#include <cstdio>
#include <algorithm>
#include <iostream>
using namespace std;
#define forn(i,n) for(int i=0;i<((int)(n));++i)
int b,t,d[5000][5000],n,k,best;
int main() 
{
  cin>>n>>k;
  forn(i,n+5)
    forn(j,n+5)
      d[i][j]=1<<30;
  d[0][0]=1;
  forn(i,n+1)
  {
    if(i<n)cin>>b>>t;else b=86401,t=1;
    forn(j,min(k,i)+1)
    {
      best = max(best, b - d[i][j]);
      d[i+1][j] = min(d[i+1][j], max(d[i][j],b) + t);
      d[i+1][j+1] = min(d[i+1][j+1], d[i][j]);
    }
  }
  cout<<best;
  return 0;
}
