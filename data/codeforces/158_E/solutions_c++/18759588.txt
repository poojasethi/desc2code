#include <iostream>
#include <cstdio>
int t[4010],d[4010],f[4010][4010],i,j,n,k,ans;
int main()
{
	scanf("%d%d\n",&n,&k);
	for (i=1;i<=n;++i) scanf("%d%d\n",&t[i],&d[i]);
	t[n+1]=86401;ans=t[1]-1;
	for (i=1;i<=n;++i)
		for (j=0;j<=k;++j)
		{
			f[i][j]=std::max(f[i-1][j],t[i]-1)+d[i];
			if (j) f[i][j]=std::min(f[i][j],f[i-1][j-1]);
			ans=std::max(ans,t[i+1]-1-f[i][j]);
		}
	printf("%d\n",ans);
}
