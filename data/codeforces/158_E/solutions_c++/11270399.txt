#include <cstdio>
#include <algorithm>
using namespace std;

int n, k, t[4005], d[4005];
int dp[4005][4005];

int main() {
	scanf("%d%d", &n, &k);
	for (int i = 1; i <= n; ++i) {
		scanf("%d%d", &t[i], &d[i]);
	}
	t[n+1] = 86401;
	int ans = t[1]-1;
	dp[0][0] = 1;
	for (int i = 1; i <= n; ++i) {
		for (int j = 0; j<=i && j<=k; ++j) {
			dp[i][j] = max(dp[i-1][min(i-1,j)], t[i]) + d[i];
			if (j) dp[i][j] = min(dp[i][j], dp[i-1][j-1]);
			ans = max(ans, t[i+1]-dp[i][j]);
		}
	}
	printf("%d\n", ans);
}
