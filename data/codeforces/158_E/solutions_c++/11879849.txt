#include<cstdio>
#include<algorithm>
using namespace std;

int n, k, ans;
int t[4444], d[4444];
int f[4044][4044];

int main() {
	scanf("%d%d", &n, &k);
	for(int i=1;i<=n;i++) scanf("%d%d", &t[i], &d[i]);
	t[n+1] = 86401;
	
	ans = t[1]-1;
	for(int i=1;i<=n;i++) {
		for(int j=0;j<=i;j++) {
			f[i][j] = max(f[i-1][j], t[i]-1) + d[i];
			if(j)
				f[i][j] = min(f[i][j], f[i-1][j-1]);
		}
		ans = max(ans, t[i+1] - f[i][k] - 1);
	}
	printf("%d\n", ans);
	
	return 0;
}
