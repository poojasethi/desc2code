#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>

using namespace std;

int n, k, d[5006], ans;

int main()
{
	scanf("%d%d", &n, &k), d[0]=1;
	ans=-0x3fffffff;
	for(int i=0, temp, t; i<=n; i++)
	{
		if(i<n) scanf("%d%d", &temp, &t);
		else temp=86401, t=1;
		for(int j=min(k, i+1); j>=0; j--)
		{
			ans=max(ans, temp-d[j]);
			d[j]=min(j==0?(1<<30):d[j-1], max(d[j], temp)+t);
		}
	}
	printf("%d\n", ans);
}
