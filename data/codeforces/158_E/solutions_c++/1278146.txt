#include<stdio.h>
#include<algorithm>
using namespace std;

int f[4010][4010],t[4010],d[4010];
int main()
{
	int n,m,i,j,ans=0;
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)
		scanf("%d%d",&t[i],&d[i]);
	for(i=1;i<=n;++i)
		for(j=0;j<=m;++j)
		{
			f[i][j]=max(f[i-1][j]+1,t[i])+d[i]-1;
			if(j&&f[i-1][j-1]<f[i][j])
				f[i][j]=f[i-1][j-1];
		}
	t[n+1]=86401;
	for(i=0;i<=n;++i)
		if(t[i+1]-f[i][m]-1>ans)
			ans=t[i+1]-f[i][m]-1;
	printf("%d\n",ans);
	return 0;
}
