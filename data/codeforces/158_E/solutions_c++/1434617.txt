#include<iostream>
using namespace std;
int n,m,t[4002],d[4002],f[4002][4002],ans;
int main() 
{
	cin>>n>>m;
	for(int i=1;i<=n;++i)cin>>t[i]>>d[i];t[n+1]=86401;
	ans=max(ans,t[1]-1);
	f[1][0]=t[1]+d[1]-1;
	f[1][1]=0;
	ans=max(ans,t[1+m]-1);
	for(int i=2;i<=n;++i)
		for(int j=0;j<=min(i,m);++j)
		{
			f[i][j]=max(t[i]+d[i]-1,f[i-1][j]+d[i]);
			if(j)f[i][j]=min(f[i-1][j-1],f[i][j]);
			ans=max(ans,t[min(i+m-j+1,n+1)]-f[i][j]-1);
		}
	cout<<ans<<endl;
	return 0;
	
}
