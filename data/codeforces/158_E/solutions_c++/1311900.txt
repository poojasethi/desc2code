#include<cstdio>
#include<iostream>
using namespace std;
#define MAXN 4010

int N,K,s[MAXN],l[MAXN];
int f[MAXN][MAXN],ans;

int main()
{
    cin>>N>>K;
    for(int i=1;i<=N;i++) scanf("%d%d",s+i,l+i),--s[i];
    s[++N]=86400;

    for(int i=1;i<=N;i++) for(int j=0;j<=K;j++)
    {
        f[i][j]=max(f[i-1][j],s[i])+l[i];
        if(j) f[i][j]=min(f[i-1][j-1],f[i][j]);
    }
    for(int i=0;i<=N;i++) for(int j=0;j<=K;j++)
        ans=max(ans,s[min(i+K-j+1,N)]-f[i][j]);
    cout<<ans<<endl;
    return 0;
}
