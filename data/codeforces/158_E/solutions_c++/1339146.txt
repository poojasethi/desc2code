#include <iostream>
#include <algorithm>
using namespace std;

int n, k;
int dp[4010];
int ans;

void update(int t, int d) {
	for (int j = k; j >= 0; j--) {
		int st = max(dp[j] + 1, t);
		ans = max(ans, st - dp[j] - 1);
		dp[j] = st + d - 1;
		if (j) {
			dp[j] = min(dp[j], dp[j - 1]);
		}
	}
}

int main() {
	cin >> n >> k;
	for (int i = 0; i < n; i++) {
		int t, d;
		cin >> t >> d;
		update(t, d);
	}
	update(86401, 1);
	cout << ans;
}
