#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=50010,M=32;
int a[N],b[N],p[M][N],f[M][N],f2[M][N],n,m,i,j,k,l,ans[M],tot,NI1,NI2;
int log_(int x){return x<10?10:100;}
int NI(int x){return x<10?NI1:NI2;}
int main()
{
	scanf("%d%d",&n,&m);
	for(NI1=i=1;i<m-1;i++)NI1=10*NI1%m;
	for(NI2=i=1;i<m-1;i++)NI2=100*NI2%m;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		b[a[i]]=i;
	}
	for(j=0,i=1;i<=n;i++)if(a[i]<M)a[++j]=a[i];
	n=min(n,j);
	//for(i=1;i<=n;i++)cout<<a[i]<<' ';cout<<endl;
	for(i=1;i<=n&&f[0][0]==0;i++)
	{
		for(j=0;j<M;j++)
			for(k=0;k<m;k++)
			p[j][k]=f[j][k];
		for(j=0;j<M;j++)
			for(k=0;k<m;k++)
			if(p[j][k]&&!f[j^a[i]][(k*log_(a[i])+a[i])%m])
			{
				f[j^a[i]][(k*log_(a[i])+a[i])%m]=a[i];
				f2[j^a[i]][(k*log_(a[i])+a[i])%m]=k;
				//printf("push: %d %d->%d %d used:%d\n",j,k,j^a[i],(k*log_(a[i])+a[i])%m,f[j^a[i]][(k*log_(a[i])+a[i])%m]);
				//if((j^a[i])==0&&0==(k*log_(a[i])+a[i])%m)system("pause");
			}
		if(!f[a[i]][a[i]%m])
		{
			f[a[i]][a[i]%m]=a[i];
			f2[a[i]][a[i]%m]=0;
			//printf("start: %d %d %d\n",a[i],a[i]%m,f[a[i]][a[i]%m]);
		}
	}
	if(!f[0][0])puts("No");
	else
	{
		for(i=j=0;;)
		{
			ans[++tot]=k=f[i][j];
			//printf("i=%d j=%d k=%d\n",i,j,k);
			j=f2[i][j];i^=k;
			if(i==0&&j==0)break;
		}
		puts("Yes");
		printf("%d\n",tot);
		while(tot)cout<<b[ans[tot--]]<<' ';
	}
	return 0;
}
