#include<iostream>
#include<cmath>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,p,a[50],q[50],w[50];
unsigned int rec[2][33][51000];
bool dp[2][33][51000];
int f1(int x){
    int ans=1;
    while(x)x/=10,ans*=10;
    return ans;
}
int main(){
    scanf("%d%d",&n,&p);
    for(int i=1;i<=n;i++){
      int x;
      scanf("%d",&x);
      if(x<32)a[++a[0]]=x,w[a[0]]=i;
    }
    n=a[0];
    dp[0][0][0]=1;
    for(int i=0;i<n;i++){
      for(int j=0;j<32;j++)
        for(int k=0;k<p;k++)dp[(i+1)%2][j][k]=0,rec[(i+1)%2][j][k]=0;
      for(int j=0;j<32;j++)
        for(int k=0;k<p;k++)
          if(dp[i%2][j][k]){
            dp[(i+1)%2][j][k]=1;
            rec[(i+1)%2][j][k]=max(rec[(i+1)%2][j][k],rec[i%2][j][k]);
            int t1=j^a[i+1],t2=(k*f1(a[i+1])+a[i+1])%p;
            unsigned int t3=rec[i%2][j][k]+(1<<i);
            dp[(i+1)%2][t1][t2]=1;
            rec[(i+1)%2][t1][t2]=max(rec[(i+1)%2][t1][t2],t3);
          }
    }
    if(rec[n%2][0][0]){
      for(int i=n;i;i--)
        if((rec[n%2][0][0]>>(i-1))&1)q[++q[0]]=w[i];
      printf("Yes\n%d\n",q[0]);
      while(q[0])printf("%d ",q[q[0]]),q[0]--;
    }
    else printf("No\n");
    return 0;
}
