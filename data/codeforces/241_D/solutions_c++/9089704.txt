#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;

typedef int arr2[50010][32];
arr2 xrb,lstv,mdb;

int next[50010][32];
int pos[32];
int num[32];
int sg[32];

int n,p,i,j,k,nj,nk,n1,u;

int main()
{
//	freopen("241D.in","r",stdin);
//	freopen("241D.out","w",stdout);
	
	scanf("%d%d",&n,&p);
	
	for (i=0;i<p;i++)
	{
		for (j=1;j<10;j++)
			next[i][j]=(i*10+j)%p;
		for (j=10;j<32;j++)
			next[i][j]=(i*100+j)%p;
	}
	
	n1=0;
	for (i=1;i<=n;i++)
	{
		scanf("%d",&u);
		if (u<32)
		{
			pos[n1]=i;
			num[n1]=u;
			n1++;
		}
	}
	
	memset(lstv,255,sizeof(lstv));
	for (i=0;i<n1;i++)
	{
		u=num[i];
		for (j=0;j<p;j++)       // fa[0][0][0]=0
			for (k=0;k<32;k++)
				if (lstv[j][k]!=-1&&lstv[j][k]<i)
				{
					nj=next[j][u];
					nk=k^u;
					if (lstv[nj][nk]==-1)
					{
						xrb[nj][nk]=k;
						mdb[nj][nk]=j;
						lstv[nj][nk]=i;
					}
				}
		
		nj=next[0][u];
		nk=u;
		if (lstv[nj][nk]==-1)
		{
			xrb[nj][nk]=-1;
			mdb[nj][nk]=-1;
			lstv[nj][nk]=i;
		}
		
		if (lstv[0][0]!=-1)
			break;
	}
	
	if (lstv[0][0]==-1)
	{ printf("No\n"); return 0; }
	
	n=0;
	for (j=0,k=0;j!=-1&&k!=-1;)
	{
		sg[n++]=pos[lstv[j][k]];
		nj=mdb[j][k];
		k=xrb[j][k];
		j=nj;
	}
	
	printf("Yes\n%d\n",n);
	for (i=0;i<n;i++)
		printf("%d ",sg[i]);
	printf("\n");
	
	return 0;
}
