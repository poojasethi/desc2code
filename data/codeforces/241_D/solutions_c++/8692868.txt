#include <cstdio>
#include <cstring>
#include <map>
const int INF = 0x3f3f3f3f;
int m,n,p,a[32],idx[50001],st[32];
int f[32][32][50001];
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("241D.in","r",stdin);
	freopen("241D.out","w",stdout);
	#endif
	scanf("%d%d",&m,&p);
	for (int i = 1; i <= m; ++i) {
		int x;
		scanf("%d",&x);
		if (x <= 31) {
			a[++n] = x;
			idx[n] = i;
		}
	}
	memset(f,0x3f,sizeof f);
	f[0][0][0] = 0;
	for (int i = 0; i < n; ++i)
		for (int k = 0; k < p; ++k) {
			int nxt;
			if (a[i + 1] / 10) nxt = (k * 100 + a[i + 1]) % p;
			else nxt = (k * 10 + a[i + 1]) % p;
			for (int j = 0; j < 32; ++j) {
				if (f[i][j][k] == INF) continue;
				f[i + 1][j ^ a[i + 1]][nxt] = k + 1;
				if (f[i + 1][j][k] == INF) f[i + 1][j][k] = -k - 1;
			}
		}
	if (f[n][0][0] != INF) {
		int u = 0,v = 0;
		for (; n; n--)
			if (f[n][u][v] > 0) {
				st[++st[0]] = idx[n];
				v = f[n][u][v] - 1;
				u ^= a[n];
			}
		if (st[0]) {
			puts("Yes");
			printf("%d\n",st[0]);
			for (int i = st[0]; i; --i) printf("%d ",st[i]);
		}else puts("No");
	}else puts("No");
}
