#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 55;

int n, P, a[N], pos[N], ans[N];
int f[32][32][50000];

int main()
{
	int foo;
	scanf("%d %d", &foo, &P);
	for (int i = 1; i <= foo; ++i)
	{
		int x;
		scanf("%d", &x);
		if (x < 32) a[++n] = x, pos[n] = i;
	}

	memset(f, -1, sizeof(f));
	for (int i = 1; i <= n; ++i)
	{
		f[i][a[i]][a[i] % P] = 0;
		for (int x = 0; x < 32; ++x)
			for (int d = 0; d < P; ++d)
				if (f[i - 1][x][d] >= 0)
				{
					int wt = a[i] < 10 ? 10 : 100;
					f[i][x][d] = P;
					f[i][x ^ a[i]][(d * wt + a[i]) % P] = d;
				}
	}
	for (int i = 1; i <= n; ++i)
		if (f[i][0][0] >= 0)
		{
			puts("Yes");
			for (int ii = i, x = 0, d = 0; ii >= 1; --ii)
			{
				int pd = f[ii][x][d];
				if (0 <= pd && pd < P)
				{
					ans[++ans[0]] = pos[ii];
					x ^= a[ii], d = pd;
				}
				if (!x && !d) break;
			}
			printf("%d\n", ans[0]);
			for (int i = ans[0]; i >= 1; --i)
				printf("%d ", ans[i]);
			return 0;
		}
	puts("No");
}
