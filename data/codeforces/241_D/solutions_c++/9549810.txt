#include<cstdio>
#include<cstring>
const int N = 32;
const int M = 50005;
int dp[N][N][M];
int n,p;
int xu[N],biao[N];
int sc[N];
int cc=0;
int main()
{
	int i,j,k,l;
	int now;
	scanf("%d%d",&n,&p);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&now);
		if(now<N)
		{
			cc++;
			xu[cc]=now;
			biao[cc]=i;
		}
	}
	memset(dp,-1,sizeof dp);
	dp[0][0][0]=0;
	for(i=0;i<cc;i++)
	{
		for(j=0;j<N;j++)
		{
			for(k=0;k<=p;k++)
			{
				if(dp[i][j][k]<0)
					continue;
				now=(i<<26)|(j<<20)|k;
				for(l=i+1;l<=cc;l++)
				{
					int nj=j^xu[l];
					int nk=k*10;
					if(xu[l]>9)
						nk*=10;
					nk+=xu[l];
					nk%=p;
					dp[l][nj][nk]=now;
					if(!nj && !nk)
					{
						int a=l,b=nj,c=nk;
						while(a)
						{
							sc[++sc[0]]=biao[a];
							now=dp[a][b][c];
							a=(now>>26);
							c=(now&((1<<20)-1));
							b=((now^(a<<26)^c)>>20);
						}
						printf("Yes\n%d\n",sc[0]);
						for(i=1;i<=sc[0];i++)
							printf("%d ",sc[i]);
						return 0;
					}
				}
			}
		}
	}
	printf("No\n");
	return 0;
}
