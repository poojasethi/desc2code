#include <bits/stdc++.h>
using namespace std;
int dp[73][50005],f[73][50005],t[73][50005];
int n,p,a[50005],b[50005],ans[105],tot;
int ddig(int x){return x>9?100:10;}
int main()
{
	cin>>n>>p;
	int i,j,k;
	for(j=0,i=1;i<=n;i++) 
	{
		scanf("%d",&a[i]);b[a[i]]=i;
		if(a[i]<=32) a[++j]=a[i];
	}
	n=min(n,j);
	for(i=1;i<=n&&dp[0][0]==0;i++)
	{
		for(j=0;j<=32;j++)
		for(k=0;k<p;k++) 
			t[j][k]=dp[j][k];
		for(j=0;j<=32;j++)
		for(k=0;k<p;k++)
		if(t[j][k]&&!dp[j^a[i]][(k*ddig(a[i])+a[i])%p])
		{
			dp[j^a[i]][(k*ddig(a[i])+a[i])%p]=a[i];
			f[j^a[i]][(k*ddig(a[i])+a[i])%p]=k;
		}
		if(!dp[a[i]][a[i]%p])
        {
            dp[a[i]][a[i]%p]=a[i];
            f[a[i]][a[i]%p]=0;
        }
	}
	if(!dp[0][0]) 
	{
		puts("No");
		return 0;
	}
	for(i=j=0;;)
	{
		ans[++tot]=k=dp[i][j];
		j=f[i][j];i^=k;
		if(i==0&&j==0) break;
	}
	puts("Yes");
 	printf("%d\n",tot);
 	while(tot) printf("%d ",b[ans[tot--]]);
}
