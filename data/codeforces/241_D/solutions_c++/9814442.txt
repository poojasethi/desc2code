#include <cstdio>
const int M=50005,P=M,L=32,N=L+2;
int m,n,p,a[N],b[N],res[N];

bool f[N][L][P];

int main(){
	scanf("%d%d",&m,&p);
	for(int i=1;i<=m;i++){
		int x; scanf("%d",&x);
		if(x<L) a[++n]=x,b[n]=i;
	}
	for(int i=1;i<=n;i++){
		f[i][a[i]][a[i]%p]=true;
		for(int j=0;j<L;j++)
			for(int k=0;k<p;k++) if(f[i-1][j][k]){
				f[i][j][k]=true;
				f[i][j^a[i]][(k*(a[i]>=10?100:10)+a[i])%p]=true;
			}
	}
	if(!f[n][0][0]) return puts("No"),0;
	puts("Yes");
	int i=n,j=0,k=0;
	for(;i;i--){
		if(f[i-1][j][k]) continue;
		res[++*res]=b[i];
		j^=a[i];
		bool found=false;
		for(int kk=0;kk<p;kk++)
			if((kk*(a[i]>=10?100:10)+a[i])%p==k&&f[i-1][j][kk]){
				k=kk;
				found=true;
				break;
			}
		if(!found) break;
	}
	printf("%d\n",*res);
	for(int i=*res;i;i--) printf("%d ",res[i]); puts("");
}
