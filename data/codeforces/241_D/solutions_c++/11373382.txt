#include<bits/stdc++.h>
using namespace std;
const int N=30;
int n,p,a[N]={},pos[N]={};
void init()
{
	scanf("%d%d",&n,&p);
	int tot=0;
	for(int i=1,x=0;i<=n;++i)
	{
		scanf("%d",&x);
		if(x<=25)
			++tot,a[tot]=x,pos[tot]=i;
	}
	n=tot;
}
void work()
{
	for(int s=(1<<n)-1;s>=1;--s)
	{
		int s1=0,s2=0;
		for(int i=0;i<n;++i)
			if(s&(1<<i))
				s1^=a[i+1],s2=(s2*(a[i+1]/10 ? 100 : 10)+a[i+1])%p;
		if(s1==0 && s2==0)
		{
			int w[N]={},tot=0;
			for(int i=0;i<n;++i)
				if(s&(1<<i))
					w[++tot]=pos[i+1];
			printf("Yes\n%d\n",tot);
			for(int i=1;i<=tot;++i)
				printf("%d%c",w[i]," \n"[i==tot]);
			return;
		}
	}
	puts("No");
}
int main()
{
	init();
	work();
	return 0;
}
