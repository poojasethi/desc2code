#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
#define per(i,a,n) for (int i=n-1;i>=a;i--)
#define pb push_back
#define mp make_pair
#define all(x) (x).begin(),(x).end()
#define SZ(x) ((int)(x).size())
#define fi first
#define se second
typedef vector<int> VI;
typedef long long ll;
typedef pair<int,int> PII;
typedef map<int,int>::iterator ite;
const ll mod=1000000007;
ll powmod(ll a,ll b) {ll res=1;a%=mod;for(;b;b>>=1){if(b&1)res=res*a%mod;a=a*a%mod;}return res;}
// head

int n,p,x,tot,pos[35],id[35];
bitset<50100> dp[35][35],ff[35][35];
VI ans;
void dfs(int i,int j,int k) {
	if (i==0) return;
	if (!ff[i][j][k]) dfs(i-1,j,k);
	else {
		ans.pb(pos[id[i]]);
		rep(y,0,p) if (dp[i-1][j^id[i]][y]&&(y*(id[i]<10?10:100)+id[i])%p==k) {
			dfs(i-1,j^id[i],y); return;
		}
	}
}
int main() {
	scanf("%d%d",&n,&p);
	tot=0;
	dp[0][0][0]=1;
	rep(i,1,n+1) {
		scanf("%d",&x);
		if (x<32) {
			tot++;
			pos[x]=i; id[tot]=x;
			rep(j,0,32) rep(k,0,p) if (dp[tot-1][j][k]) {
				int kk=x<10?k*10+x:k*100+x;
				kk%=p;
				dp[tot][j^x][kk]=1;
				ff[tot][j^x][kk]=1;
				if ((j^x)==0&&kk==0) {
					dfs(tot,0,0);
					puts("Yes");
					printf("%d\n",SZ(ans));
					per(p,0,SZ(ans)) printf("%d ",ans[p]);
					return 0;
				}
				dp[tot][j][k]=1;
				ff[tot][j][k]=0;
			}
		}
	}
	puts("No");
}
