#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 50010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,t,x,s;
int an[40],a[40],b[40];
short f[40][40][N];
void dfs(int x,int y,int z) {
	if (!x) return;
	int A=s?1:2,B=a[x];
	if (f[x-1][y][z]>=A) dfs(x-1,y,z);
	else {
		an[++s]=b[x];
		int i;
		For(i,0,m-1) {
			int k=i;
			if (B>=10) k*=100;
			else k*=10;
			k=(k+B)%m;
			if (f[x-1][y^B][i]&&k==z) break;
		}
		dfs(x-1,y^B,i);
	}
}
int main() {
	scanf("%d%d",&n,&m);
	f[0][0][0]=1;
	For(i,1,n) {
		scanf("%d",&x);
		if (x<31) {
			++t; a[t]=x; b[t]=i;
			For(j,0,35)For(k,0,m-1) if (f[t-1][j][k]) {
				int A=j^x,B=k;
				if (x>=10) B*=100;
				else B*=10;
				B=(B+x)%m;
				f[t][A][B]=2;
				f[t][j][k]=max(f[t][j][k],f[t-1][j][k]);
			}
		}
	}
	if (f[t][0][0]<2) printf("No\n");
	else {
		printf("Yes\n");
		dfs(t,0,0);
		printf("%d\n",s);
		for (i=s;i;i--) printf("%d ",an[i]); printf("\n");
	}
	return 0;
}
