#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
#define N 26
#define ll long long
int v[N],n,st,p,wz[N];bool b[N];
inline int Read()
 {
    int x=0;char y;
    do y=getchar(); while (y<'0'||y>'9');
    do x=x*10+y-'0',y=getchar(); while (y>='0'&&y<='9');
    return x;
 }
void Put_Ans(int x,int z)
 {
    puts("Yes");
    cout <<z<<endl;
    for (int i=1;i<x;i++)
     if (b[i]) printf("%d ",wz[i]);
    puts("");
    exit(0);
 }
void DFS(int x,int y,int z,ll o)
 {
    if (!y&&z&&!o) Put_Ans(x,z);
    if (x>st) return;
    b[x]=false;
    DFS(x+1,y,z,o);
    b[x]=true;
    DFS(x+1,y^v[x],z+1,(o*(v[x]<10?10:100)+v[x])%p);
 }
int main()
 {
    n=Read();p=Read();
    for (int i=1;i<=n;i++)
     {
        int k=Read();
        if (k<N) v[++st]=k,wz[st]=i;
     }
    DFS(1,0,0,0);
    puts("No");
    return 0;
 }
