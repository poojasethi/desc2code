const int maxn=50000+15;
const int kkey=26;
int n,p;
int a[maxn];
int sum,b[maxn],c[maxn];
bool boo[maxn];
int dfs(int now,int xorr,int modd,int us)
{
	if (xorr==0 && modd==0 && us!=0)
	{
		printf("Yes\n%d\n",us);
		for (int i=1;i<=sum;i++)
		 if (boo[i])
		 {
		 	us--;
		 	if (us==0) printf("%d\n",b[i]);
		 	 else printf("%d ",b[i]);
		 }
		return true;
	}
	if (now>sum) return false;
	if (dfs(now+1,xorr,modd,us)) return true;
	boo[now]=true;
	xorr^=c[now];
	if (c[now]>9) modd=(modd*100+c[now])%p;
	 else modd=(modd*10+c[now])%p;
	if (dfs(now+1,xorr,modd,us+1)) return true;
	boo[now]=false;
	return false;
}
int main()
{
	scanf("%d%d",&n,&p);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=1;i<=n;i++)
	 if (a[i]<=kkey) 
	  b[++sum]=i,c[sum]=a[i];
	if (!dfs(1,0,0,0)) printf("No\n");
	return 0;
}
