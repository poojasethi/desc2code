#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;

const int MAX=50000+10;
const int MAXL=32;

int n,p,a[MAX],len[MAX],ans[MAX],id[MAX];
bool f[MAXL][MAXL][MAX],choose[MAXL][MAXL][MAX];

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
#endif
	int i,j,k,t,temp=0;
	scanf("%d%d",&n,&p);
	for(i=1;i<=n;++i)
	{
		scanf("%d",&t);
		if(t<MAXL)
		{
			a[++temp]=t;
			id[temp]=i;
			for(len[temp]=1;len[temp]<=a[temp];len[temp]*=10);
		}
	}
	n=temp;
	for(i=1;i<=n;++i)
	{
		f[i][a[i]][a[i]%p]=1;
		choose[i][a[i]][a[i]%p]=1;
		for(j=0;j<MAXL;++j)
			for(k=0;k<p;++k)
				if(f[i-1][j][k])
				{
					f[i][j][k]=1;
					choose[i][j][k]=0;
					int ny=j^a[i],np=(k*len[i]+a[i])%p;
					f[i][ny][np]=1;
					choose[i][ny][np]=1;
				}
	}
	if(!f[n][0][0])
	{
		printf("No\n");
		return 0;
	}
	int nowy=0,nowp=0;
	for(i=n;i>=1;--i)
	{
		if(choose[i][nowy][nowp])
		{
			ans[++ans[0]]=id[i];
			nowy^=a[i];
			for(j=0;j<p;++j)
				if(f[i-1][nowy][j] && (j*len[i]+a[i])%p==nowp)
				{
					nowp=j;
					break;
				}
		}
	}
	printf("Yes\n");
	printf("%d\n",ans[0]);
	for(i=ans[0];i>=1;--i)
		printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
