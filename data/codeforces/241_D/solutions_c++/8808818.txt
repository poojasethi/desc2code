#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <cctype>
#include <algorithm>
#include <iostream>
#define ft first
#define sd second
#define mp make_pair
using namespace std;

typedef pair<char,unsigned short> pcs;

const int N=50000+5;

int n,m,p,cnt,a[N],pos[N],ans[N];
bool f[32][32][N];
pcs pre[32][32][N];

inline int calc(int a,int b)
{
	if (b>=10) return (a*100+b)%p;
	return (a*10+b)%p;
}

int main(int argc,char **argv)
{
	#ifndef ONLINE_JUDGE
		if (argc==1) freopen("241D.in","r",stdin),freopen("241D.out","w",stdout);
		else freopen(argv[1],"r",stdin),freopen(argv[2],"w",stdout);
	#endif
	int x;
	scanf("%d%d",&n,&p);
	for (int i=1;i<=n;++i)
	{
		scanf("%d",&x);
		if (x<32) a[++m]=x,pos[m]=i;
	}
	for (int i=1;i<=m;++i)
	{
		f[i][a[i]][a[i]%p]=1;
		pre[i][a[i]][a[i]%p]=mp(0,0);
		for (int j=0;j<32;++j)
			for (int k=0;k<p;++k)
				if (f[i-1][j][k])
				{
					f[i][j][k]=1;
					pre[i][j][k]=mp(j,k);
				}
		for (int j=0;j<32;++j)
			for (int k=0;k<p;++k)
				if (f[i-1][j][k])
				{
					f[i][j^a[i]][calc(k,a[i])]=1;
					pre[i][j^a[i]][calc(k,a[i])]=mp(j,k);
				}
	}
	if (!f[m][0][0]){puts("No");return 0;}
	for (int i=m,j=0,k=0,tj,tk;i>=1;--i)
		if (!f[i-1][j][k])
		{
			ans[++cnt]=pos[i];
			tj=pre[i][j][k].ft;
			tk=pre[i][j][k].sd;
			j=tj;k=tk;
			if (!j && !k) break;
		}
	puts("Yes");
	printf("%d\n",cnt);
	for (int i=cnt;i>0;--i) printf("%d%c",ans[i]," \n"[i==1]);
	return 0;
}
