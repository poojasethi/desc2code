#include <bits/stdc++.h>

using namespace std;

struct Pack{
	unsigned char x,xo;
	unsigned short mo;
	Pack(){}
	Pack(int _x,int _xo,int _mo) : x(_x), xo(_xo), mo(_mo) {}
};

int n,p,m;
Pack f[35][32][50000];
int a[32];
int X[32];

void print(int x,int xo,int mo,int cnt){
	if(!x) printf("%d\n",cnt);
	else{
		print(f[x][xo][mo].x,f[x][xo][mo].xo,f[x][xo][mo].mo,cnt+1);
		printf("%d ",X[x]);
	}
}
int main()
{
	scanf("%d%d",&n,&p);
	for(int i=1,t;i<=n;i++){
		scanf("%d",&t);
		if(t<32) a[++m]=t,X[m]=i;
	}
	memset(f,0xff,sizeof(f));
	int K=0;
	f[0][0][0]=Pack(0,0,0);
	for(int i=0;i<m && !K;i++)
		for(int j=0;j<32 && !K;j++)
			for(int k=0;k<p && !K;k++) if(f[i][j][k].x!=0xff){
				if(!j && !k) K=i;
				for(int l=i+1;l<=m && !K;l++)
					f[l][j^a[l]][(k*(a[l]<10 ? 10 : 100)+a[l])%p]=Pack(i,j,k);
			}
	if(!K && f[m][0][0].x!=0xff) K=m;
	if(!K) return puts("No"),0;
	puts("Yes");
	print(K,0,0,0);
	puts("");
	return 0;
}
