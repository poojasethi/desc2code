#include <cstdio>
#include <iostream>

using namespace std;

int n, m, k;

int main()
{
	scanf("%d%d%d", &n, &m, &k);

	if (n % 2 == 0)
		cout << "Marsel";
	else
	{
		for (int i = 1; i * 1ll * i <= m; ++i)
			if (m % i == 0)
			{
				if ((i >= k && m / i != 1) || (m / i >= k && i != 1))
				{
					cout << "Timur";
					return 0;
				}
			}

		cout << "Marsel";
	}
}
