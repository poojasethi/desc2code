#include <cstdio>

int main(){
	int n, m, k;
	scanf("%d%d%d", &n, &m, &k);
	if (n%2==0)
		printf("Marsel");
	else{
		bool divexist=false;
		for (int i=2; i*i<=m; i++)
			if (m%i==0 && (i>=k || m/i>=k))
				divexist=true;
		if (m>=2 && k==1)
			divexist=true;
		if (!divexist)
			printf("Marsel");
		else
			printf("Timur");
	}
}
