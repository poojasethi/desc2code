#include <cstdio>
#include <cmath>
int n, m, k;
int process(){
    int i=0, t=0, r = (int)sqrt(m)+1;
    while(++i<r&&t<2)if(m%i==0)t++;
    if((t>1&&(m/(i-1)>=k))||(k==1&&m!=1))return n%2;
    return 0;
}
int main(){
    scanf("%d %d %d", &n, &m, &k);
    if(process())printf("Timur");
    else printf("Marsel");
    return 0;
}
