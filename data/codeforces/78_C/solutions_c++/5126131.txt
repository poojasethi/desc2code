#include <cstdio>
#include <cmath>
int main(){
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    bool flag = false;
    for(int i = 2;i <= sqrt(m);i++){
        if(m % i == 0 && m / i >= k){
            flag = true;
            break;
        }
    }
    if((n&1) && (flag || (k == 1 && m > 1)))   printf("Timur\n");
    else    printf("Marsel\n");
    return 0;
}
