#include <iostream>
using namespace std;

int n,m,k;
bool r,w;

int main(){

  cin>>n>>m>>k;
  for(int i=2;i*i<=m;i++)if((m%i==0)&&(i>=k||(m/i)>=k)){w=true;break;}
  if(k==1&&m>1)w=true;
  
  if(w){
    if(n%2==1)r=true;
    else r=false;
  }
  else r=false;

  cout<<(r?"Timur":"Marsel")<<endl;
  
  return 0;

}
