#include <cstdio>

int n, m, k;

inline bool gao() {
    if (n % 2 == 0) return 0;
    if (1 == k) return m > 1;
    for (int i = 2; (long long)i*i <= m; ++i)
        if (m % i == 0) {
            if (m/i >= k || i >= k) return 1;
        }
    return 0;
}

int main() {
    scanf("%d %d %d", &n, &m, &k);
    puts( gao() ? "Timur" : "Marsel" );
    return 0;
}
