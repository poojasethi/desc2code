#include<iostream>
#include<string.h>
using namespace std;
int main()
{
int n,m,k,i,j=0;
string s;
cin>>n>>m>>k;
for(i=2;i*i<=m;i++)
{
while(m/i>=k && m%i==0)
{
m/=i;
j++;
}
}
if(n%2==0) s="Marsel";
if(j>=1 || (k==1 && m!=1))
{
if(n%2==1) s="Timur";
else s="Marsel";
}
else s="Marsel";
cout<<s;
return 0;
}
