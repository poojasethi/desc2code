#include<stdio.h>
#include<math.h>
int main(){
	int n,m,k,i,flag;
	scanf("%d%d%d",&n,&m,&k);
	if(n%2==0)printf("Marsel\n");
	else {
		flag=0;
		for(i=2;i<=sqrt(m);i++)
			if(m%i==0&&(i>=k||m/i>=k)){
				flag=1;printf("Timur\n");
				break;
			}
		if(!flag){
			if(m%k==0&&m!=k)printf("Timur\n");
			else printf("Marsel\n");
		}
	}
}
