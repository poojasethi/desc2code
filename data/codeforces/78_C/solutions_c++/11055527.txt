

#include<iostream>

using namespace std;

int main()
{
	int n,m,k;
	int flag=1;
	while(cin>>n>>m>>k)
	{
		flag=1;
		if(n%2==1)
		{
			if(k==1&&m>1&&flag)flag=0;
			if(flag)
			for(int i=2;i*i<=m;i++)
				if((m%i==0&&(m/i>=k||i>=k)))
				{
					flag=0;
					break;
				}
		}
		if(flag)cout<<"Marsel"<<endl;
		else cout<<"Timur"<<endl;
	}
	return 0;
}
