#include<stdio.h>
#include<string.h>
int n,m,k;
int main(){
  int i,j;
  scanf("%d%d%d",&n,&m,&k);
  if(!(n%2)){
    printf("Marsel\n");
    return 0;
  }
  int flag=0;
  for(i=2;i*i<=m;i++)
    if(!(m%i)&&(i>=k||m/i>=k)){
      flag=1;
      break;
    }
  if(m>1&&k==1) flag=1;
  printf("%s\n",flag?"Timur":"Marsel");
  return 0;
}
