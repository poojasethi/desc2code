#include <stdio.h>
int n,m,k;
bool deal(){
    int i;
    if(n%2==0)   return false;
    if(k==1)     return m>1;
    for(i=2;i*i<=m;i++)
        if(m%i==0)
            if(m/i>=k||i>=k)
                return true;
    return false;
}
int main(void){
    while(scanf("%d%d%d",&n,&m,&k)==3)
        puts(deal()?"Timur" : "Marsel");
    return 0;
}
