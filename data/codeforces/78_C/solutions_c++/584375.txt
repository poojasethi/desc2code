#include <iostream>
using namespace std;
int main()
{
	int n,m,k,f,i;
	cin>>n>>m>>k;
	f=1;
	for(i=1;i*i<=m;i++)
	{
		if(m%i==0&&(i>=k&&m/i>1||m/i>=k&&i>1))
		{
			f=0;
			//break;
		}
	}
	if(f==0)
	{
		if(n%2==1)
			cout<<"Timur"<<endl;
		else
			cout<<"Marsel"<<endl;
	}
	else 
		cout<<"Marsel"<<endl;
return 0;
}
