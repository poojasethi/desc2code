#include <stdio.h>
#include <math.h>
int main(){
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    if (n%2==0) printf("Marsel");
    else{
        for (int i=1; i<=sqrt(m); i++)
            if ((m%i==0) && ((i>=k && m/i>1) || (m/i>=k && i>1))){
                printf("Timur");
                return 0;
            }
        printf("Marsel");
    }
    return 0;
}
