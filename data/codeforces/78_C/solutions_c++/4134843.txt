#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

int n,m,k,x;

int main()
{
	int i,tmp;
	while(~scanf("%d%d%d",&n,&m,&k))
	{
		if(k==1&&m!=k) x=1;
		else x=0;
		for(i=2;i*i<=m;i++)
		{
			if(m%i==0&&(m/i>=k||i>=k)) x=1;
		}
		if(x&&n%2==1) printf("Timur\n");//w
		else printf("Marsel\n");//l
	}
	return 0;
}
