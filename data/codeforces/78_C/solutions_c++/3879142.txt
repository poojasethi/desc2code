#include<iostream>
using namespace std;
int main(){
    long long n,m,k,i;
    int  f=0;
    cin>>n>>m>>k;
    for(i=1;i*i<=m;i++){
        if((m%i==0)&&((i>=k&&m/i>1)||((m/i)>=k)&&i>1)){
        f=1;
        break;
        }
    }
    if(f==1&&(n&1))
    cout<<"Timur";
    else
    cout<<"Marsel";
    return 0;
}
