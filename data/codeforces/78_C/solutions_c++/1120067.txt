#include <iostream>
using namespace std;

int main() {
	int n, m, k;
	cin >> n >> m >> k;
	n %= 2;
	if (n == 0 || m == 1) {
		cout << "Marsel";
		return 0;
	}
	if (k == 1) {
		cout << "Timur";
		return 0;
	}
	for (int i = 2; i * i <= m; ++i) {
		if (m % i == 0 && m / i >= k) {
			cout << "Timur";
			return 0;
		}
	}
	cout << "Marsel";
}
