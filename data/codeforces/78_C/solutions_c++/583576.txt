int n,m,k;
bool cacu(){

	   if(n%2==0) return false;
	   if(k==1)  return m>1;
	   for(i=2;i*i<=m;i++)
		      if(m%i==0)
					if(i>=k||m/i>=k) return true;
       return false;		
}
