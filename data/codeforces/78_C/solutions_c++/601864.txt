#include <cstdio>
using namespace std;

int n,m,k;

bool div() {
    for (int i=1; i*i<=m; i++)
        if ((m%i==0)&&(((i>=k)&&(m/i>1))||((m/i>=k)&&(i>1)))) return true;
    return false;
}

int main() {
    scanf("%d%d%d",&n,&m,&k);
    if ((n&1)&&(div())) printf("Timur\n");
    else printf("Marsel\n");
    return 0;
}
