#include<stdio.h>
#include<math.h>
int main()
{
	int n,m,k;
	while(~scanf("%d%d%d",&n,&m,&k))
	{
		int p=sqrt(1.0*m);
		int flag=0;
		for(int i=1;i<=p;i++)
		{
			if(m%i==0&&(i>=k||(m/i>=k&&i!=1)))
			{
				flag=1;break;
			}
		}
		if(m==1)flag=0;
		if(flag&&(n&1))puts("Timur");
		else puts("Marsel");
	}
	return 0;
}
