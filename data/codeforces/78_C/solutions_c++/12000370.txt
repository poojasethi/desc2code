#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	if(n%2==0)
	{
		cout<<"Marsel\n";
		return 0;
	}
	int f1,f2;
	for(int i=1;i<=sqrt(m);i++)
	{
		if(m%i==0)
		{
			f1=(i<m && i>=k);
			f2=((m/i)<m && (m/i)>=k);
			if(f1||f2)
			{
				cout<<"Timur\n";
				return 0;
			}
		}
	}
	cout<<"Marsel\n";
	return 0;
}
