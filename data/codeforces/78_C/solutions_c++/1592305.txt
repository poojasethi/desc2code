#include <stdio.h>
int n,m,k;
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	if (n%2==0) printf("Marsel\n");
	else {
		if (k==1&&m!=1)
			printf("Timur\n");
		else {
			int i;
			for (i=2; i*i<=m; i++) {
				if (m%i==0) {
					if (m/i>=k)
						printf("Timur\n");
					else
						printf("Marsel\n");
					break;
				}
			}
			if (i*i>m) printf("Marsel\n");
		}
	}
	return 0;
}
