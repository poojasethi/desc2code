using namespace std;

int main(){
	int n,m,k;
	cin>>n>>m>>k;
	int flag = 1;
	if (n%2)
		for (int i=1 ; i*i<=m ; i++){
		if (m%i) continue;
		if (i>=k && m/i>1) flag = 0;
		if (m/i>=k && i>1) flag = 0;
	}
