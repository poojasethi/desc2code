#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,m,k,i,c=0;
    cin>>n>>m>>k;
    for(i=2;i<=sqrt(m);i++){
        if(m%i==0 && m/i>=k){
            c++;
            break;
        }
    }
    if((c>0 && n%2==1) || (k==1 && m>1 && n%2==1))
        cout<<"Timur";
    else
        cout<<"Marsel";
    return 0;
}
