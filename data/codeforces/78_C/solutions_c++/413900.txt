#include <stdio.h>
int main() {
  int n, m, k;
  scanf("%d%d%d", &n, &m, &k);
  int good=(m!=1 && k==1 || (m==2?k==1:0));
  for(int i=2; i*i <= m; ++i) {
    if(m % i == 0) {
      if(m / i >= k) good = 1;
      break;
    }
  }
  
  if(n%2 == 0 || !good) puts("Marsel");
  else puts("Timur");
  return 0;
}
