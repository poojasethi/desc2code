
import math

n, m, k = [ int(i) for i in raw_input().split() ]

divisible = False

if 2 < int( math.sqrt( m ) ):
	for i in range( 2, int( math.sqrt( m ) )+1 ):
#		print int(m/i), m, i
		if m%i == 0 and i >= k:
			divisible = True
			break
		if m%i == 0 and int(m/i) >= k:
			divisible = True
			break

if m == 2 and k == 1:
	divisible = True

if m >= 2 and k == 1:
	divisible = True

if divisible:
	if ( n%2 ):
		print "Timur"
	else:
		print "Marsel"
else:
	print "Marsel"
