def max_divisor(n):
    m = 2
    while m * m <= n:
	if n % m == 0:
	    return n / m
	m += 1
    return 1

def solve(n, m, k):
    if n % 2 == 0 or m == 1:
	return False
    elif max_divisor(m) >= k:
	return True
    return False

def main():
    n, m, k = map(int, raw_input().split())
    if solve(n, m, k):
	print "Timur"
    else:
	print "Marsel"

main()
