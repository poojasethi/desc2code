def foo():
    if n & 1 ^ 1:
        return False
    for i in xrange(1, int(m ** 0.5 + 0.1) + 1):
        if m % i == 0:
            a = i
            b = m / i
            if a >= 2 and b >= k: return True
            if a >= k and b >= 2: return True
    return False

n, m, k = map(int, raw_input().split())
print 'Timur' if foo() else 'Marsel'
