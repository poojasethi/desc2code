import math
n,m,k = map(int,raw_input().split());
d = [];
for i in range(1,int(math.sqrt(m))+10):
	if m%i == 0:
		if i>=k: d.append(i);
		if len(d)>0 and d[len(d)-1]==m: d.remove(m);
		if m//i >= k: d.append(m//i);
		if len(d)>0 and d[len(d)-1]==m: d.remove(m);
if len(d) == 0:
	print "Marsel";
	exit();
if n%2 == 1:
	print "Timur";
	exit();
print "Marsel";
