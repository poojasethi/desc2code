n, m, k = map(int, raw_input().split())
flag = 0

i = 1
while i * i <= m:
    if m % i == 0:
        if (m / i >= k and i > 1):
            flag = 1
        if (m / i > 1 and i >= k):
            flag = 1
    i += 1
    
if (flag == 0):
    print 'Marsel'
elif (n % 2 == 1):
    print 'Timur'
else:
    print 'Marsel'
