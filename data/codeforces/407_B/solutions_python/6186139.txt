n = int(raw_input())
backs = [0] + map(int, raw_input().split())

cache = [2] * 10010

pos = 2
steps = 2

while pos != (n+1):
	if backs[pos] == pos:
		pos += 1
		steps = (steps + 2) % 1000000007
		cache[pos] = 2
	else:
		togetback = 1
		for i in range(backs[pos], pos):
			togetback += cache[i]
		togetback = togetback % 1000000007
		steps = (steps + togetback + 1) % 1000000007
		cache[pos] = (togetback + 1) % 1000000007
		pos = pos + 1
print steps
