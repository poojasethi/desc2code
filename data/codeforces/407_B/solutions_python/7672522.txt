n = int(raw_input())

L = [int(x) for x in raw_input().split()]

def process(L):
    DP = [0 for x in range(n+1)]
    DP[1] = 2
    for i in range(2, n + 1):
        
        acc = 0
        if L[i - 1] == i: DP[i] = 2
        else:
            for j in range(L[i - 1], i): acc += DP[j]
            DP[i] = acc + 2
    return sum(DP)%1000000007


print process(L)
