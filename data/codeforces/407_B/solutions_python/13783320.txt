rooms = int(raw_input())
portals = map(int, raw_input().split())

steps = [0] * (rooms + 1)
def main():
    for x in xrange(1, rooms + 1):
        portal_room = portals[x - 1]
        steps[x] = (steps[x - 1] + steps[x - 1] - steps[portal_room - 1] + 2) % 1000000007
    print(steps[rooms])
main()
