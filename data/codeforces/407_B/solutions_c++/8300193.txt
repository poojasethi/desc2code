#include<cstdio>
#include<cstdlib>
#define MOD 1000000007
#define LL long long

int a[1010],dp[1010];
int main()
{
    int n;scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    dp[1]=0;
    for(int i=1;i<=n;i++)dp[i+1]=((LL)2*dp[i]+2-dp[a[i]]+MOD)%MOD;
    printf("%d\n",dp[n+1]);
    return 0;
}
