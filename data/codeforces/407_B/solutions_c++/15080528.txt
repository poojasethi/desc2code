#include<bits/stdc++.h>
using namespace std;
int n;
int p[1005];
long long dp[1005];
long long mod=1000000007;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&p[i]);
    for(int i=2;i<=n+1;i++)dp[i]=((dp[i-1]*2)-dp[p[i-1]]+mod+2)%mod;
    printf("%I64d\n",dp[n+1]);
    return 0;
}
