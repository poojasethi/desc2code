#include<stdio.h>
#define M 1000000007
__int64 p[2000];
int a[2000];
int main(){
	int i,n;
	//freopen("b1.in","r",stdin);
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	p[0]=0;
	for(i=1;i<=n;i++){
		p[i]=p[i-1]*2+2-p[a[i]-1];
		p[i]%=M;
		if(p[i]<0)p[i]+=M;
	}
	printf("%I64d\n",p[n]);
	return 0;
}
