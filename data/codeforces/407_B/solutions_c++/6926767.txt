#include<stdio.h>
const int mod=1e9+7;
int S[1005];
int p[1005];
int main(){
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d",&p[i]);
    S[1]=0;
    for(int i=2;i<=n+1;i++){
        S[i] = ((2 * S[i - 1]) % mod + 2 + mod - S[p[i - 1]]) % mod;
        }
printf("%d\n" , S[n+1]);

}
