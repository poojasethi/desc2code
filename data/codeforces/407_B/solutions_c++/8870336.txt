#include <iostream>
#define MOD 1000000007

using namespace std;
long long n,i,a[1001],d[1001];
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
    }
    d[1]=2;
    for(i=2;i<=n;i++)
    {
        d[i]=(d[i-1]+d[i-1]-d[a[i]-1]+2+MOD)%MOD;
    }
    cout<<d[n];
    return 0;
}
