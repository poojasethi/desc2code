#include<iostream>
#define MOD 1000000007
using namespace std;
int main()
{
 unsigned long long n,a[1005],i,j,dp[1005]={0};
 cin>>n;
 for(i=1;i<=n;i++) cin>>a[i];
 dp[1]=0;
 for(i=1;i<=n;i++) {
  if(a[i]==i) dp[i+1]=(dp[i]+2);
  else {
   dp[i+1]=(dp[i]+dp[i]-dp[a[i]]+2+MOD)%MOD;
  }
 }
 cout<<dp[n+1];
}
