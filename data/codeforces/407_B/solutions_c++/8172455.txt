#include <cstdio>
#include <cstdlib>
#define P 1000000007
using namespace std;

int n,a[100001];
long long f[100001];

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	f[1]=0;
	for (int i=2;i<=n+1;i++) f[i]=(1+f[i-1]-f[a[i-1]]+f[i-1]+1)%P;
	printf("%I64d\n",(f[n+1]+P)%P);
}
