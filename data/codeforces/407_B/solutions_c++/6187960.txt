#include <cstdio>

#define N 1000
#define M 1000000007

int a[N], p[N], n, i, j, c;

int main()
{
	scanf("%d", &n);

	for (i = 0; i < n; i++)
		scanf("%d", &p[i]), --p[i];

	for (i = 0; i < n; c = (c + a[i]) % M, i++)
		for (a[i] = 2, j = p[i]; j < i; j++)
			a[i] = (a[i] + a[j]) % M;

	printf("%d\n", c);

	return 0;
}
