#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int dp[10000],a[10000];
const int P=1000000007;

int main(){
	dp[0]=0;
	dp[1]=2;
	int n;
	scanf("%d",&n);
	for (int i=1; i<=n; ++i) scanf("%d",&a[i]);
	for (int i=2; i<=n; ++i)
		dp[i]=((dp[i-1]+1)*2%P-dp[a[i]-1]+P)%P;
	printf("%d\n",(dp[n]+P)%P);
}
