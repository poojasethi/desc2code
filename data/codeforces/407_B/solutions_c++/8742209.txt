#include <iostream>
#define MOD 1000000007
using namespace std;

int main(){
    int n;
    cin >>n;
    int v[1001];
    int p[1001];
    for(int i=1;i<=n;i++){
      cin >> p[i];
	};
    v[1]=0;
    for(int i=1;i<=n;i++){
        v[i+1]=((v[i]*2-v[p[i]]+2)%MOD+MOD)%MOD;
    }
    cout << v[n+1];
   return 0;
}
