///বিসমিল্লাহির রাহমানির রাহিম
///মো: ইসমাইল হোসাইন(মুকুল)
#include<bits/stdc++.h>
#define i64 long long
#define ms(a,b) memset(a,b,sizeof(a));
#define MAX 100000000 + 7
#define open() freopen("input.txt","r",stdin)
using namespace std;
int main(){
    int n,m;
    while(cin>>n>>m)
    {
        int ara[n+4];
        int mn=MAX;
        int mx=-1;
        for(int i=0;i<n;i++){
            cin>>ara[i];
            mn=min(mn,ara[i]);
            mx=max(mx,ara[i]);
        }
        for(int i=0;i<n;i++){
            ara[i]=ara[i]-mn;
        }
        mx=mx*m;
        int dp[mx+5];
        for(int i=1;i<=mx;i++) dp[i]=MAX;
        dp[0]=0;
        for(int i=0;i<n;i++){
            for(int j=ara[i];j<=mx;j++){
                dp[j]=min(dp[j],dp[j-ara[i]]+1);
            }
        }
        for(int i=0;i<=mx;i++){
            if(dp[i]<=m){
                cout<<i + mn*m<<' ';
            }
        }
        cout<<endl;
    }
    return 0;
}
