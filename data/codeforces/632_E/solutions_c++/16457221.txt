#include<bits/stdc++.h>
using namespace std;
int a[1005];
int dp[1000005];
int main()
    {
    int n,k;
    scanf("%d%d",&n,&k);
    int i;
    for(i=0;i<n;i++)
       {
        scanf("%d",&a[i]);
    }
    sort(a,a+n);
    int mi,ma;
    mi = a[0];
    for(i=0;i<n;i++)
        a[i]-=mi;
    ma = a[n-1];
    int hi = ma*k;
    for(i=1;i<=hi;i++)dp[i] = 1000000007;
    int j;
    mi*=k;
  //  printf("%d",mi);
    for(i=1;i<=hi;i++)
        {
        for(j=0;j<n;j++)
            if(i>=a[j])
            dp[i] = min(dp[i],dp[i-a[j]]+1);
    }
    for(i=0;i<=hi;i++)
        if(dp[i]<=k)printf("%d ",i+mi);
}
