#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;

int dp[1000010];

int main()
{
    int i,j,n,k,a[1010];
    while(scanf("%d%d",&n,&k)!=EOF)
    {
        for(i=1; i<=n; i++)scanf("%d",&a[i]);
        sort(a+1,a+n+1);
        int a1=a[1];
        for(i=1; i<=n; i++)a[i]-=a1;
        for(i=1; i<=1000000; i++)dp[i]=99999999;
        dp[0]=0;
        for(i=0; i<=1000000; i++)
        {
            for(j=1; j<=n; j++)
            {
                if(i>=a[j])
                {
                    dp[i]=min(dp[i],dp[i-a[j]]+1);
                }
            }
        }
        for(i=0; i<=1000000; i++)
        {
            if(dp[i]<=k)
            {
                printf("%d ",i+k*a1);
            }
        }
        printf("\n");
    }


    return 0;
}
