#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <string>
using namespace std;
const int N=1005;
const int inf=1e9;
int a[N], dp[N*N];
int main(){
	int n, k;
	cin>>n>>k;
	for(int i=0; i<n; i++) cin>>a[i];
	sort(a, a+n);
	int pos=unique(a, a+n)-a;
	for(int i=1; i<pos; i++) a[i]-=a[0];
	for(int i=0; i<N*N; i++) dp[i]=inf;
	dp[0]=0;
	for(int i=1; i<pos; i++){
		for(int j=a[i]; j<=k*a[i]; j++){
			dp[j] = min(dp[j], dp[j-a[i]]+1);
		}
	}
	for(int i=0; i<N*N; i++){
		if(dp[i]<=k){
			cout<<i+k*a[0]<<" ";
		}
	}
	cout<<endl;
	return 0;
}
