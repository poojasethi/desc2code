#include <algorithm>
#include <iostream>			
#include <cstdio>
using namespace std;
int a[1010],dist[1000010],u[1000010],d[3000000],i,n,k,m,s,t;
int main()
{
		scanf("%d%d\n",&n,&k);
		for (i=1;i<=n;++i) scanf("%d",&a[i]);
		sort(a+1,a+n+1);
		for (i=2;i<=n;++i) a[i]-=a[1];
		m=a[n]*k;
		for (i=1;i<=m;++i) dist[i]=k+1,u[i]=1;
		dist[0]=u[0]=0;
		for (d[s=t=1]=0;s<=t;u[d[s++]]=0)
			for (i=2;i<=n;++i)
			if (dist[d[s]]+1<dist[d[s]+a[i]])
			{
				dist[d[s]+a[i]]=dist[d[s]]+1;
				if (u[d[s]+a[i]]) u[d[s]+a[i]]=0,d[++t]=d[s]+a[i];
			}
		for (i=0;i<=m;++i)
			if (dist[i]<=k) printf("%d ",a[1]*k+i);
}
