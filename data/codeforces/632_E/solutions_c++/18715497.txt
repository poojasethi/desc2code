const int M=1005;
int dp[M*M];
int val[M],sum=0;
int main(){
#ifdef TL
#endif
	int n,K;
	scanf("%d%d",&n,&K);
	int mi=inf,sum=0,mx=0;
	for(int i=1;i<=n;i++){
		scanf("%d",&val[i]);
		mi=min(mi,val[i]);
	}
	for(int i=1;i<=n;i++){
		val[i]-=mi;
		mx=max(val[i],mx);
	}
	sum=mx*K;
	memset(dp,63,sum+1<<2);
	dp[0]=0;
	for(int i=1;i<=n;i++)
		for(int j=val[i];j<=sum;j++)
			dp[j]=min(dp[j],dp[j-val[i]]+1);
	for(int i=0;i<=sum;i++)
		if(dp[i]<=K)printf("%d ",i+mi*K);
	return 0;
}
