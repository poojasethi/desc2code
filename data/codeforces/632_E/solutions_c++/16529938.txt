int main()
{

    int n,k;
    cin>>n>>k;


    VI a;

    REP(i,n)
    {
        a.pb(0);
        cin>>a[i];
    }
   
    sort(all(a));

    int dp[1000006];
    REP(i,1000006)
        dp[i]=1000000;


    dp[a[0]*k]=0;
    
    FOR(i,a[0]*k,a.back()*k+1)
    {
        REP(j,n)
            if(i>=(a[j]-a[0]))
            {
                dp[i]=min(dp[i],dp[i-a[j]+a[0]]+1);
            }
    }

    FOR(i,a[0]*k,a.back()*k+1)
        if(dp[i]<=k)
            printf("%d\n",i);
    return 0;
}
