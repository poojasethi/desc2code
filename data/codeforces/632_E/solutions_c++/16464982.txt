#include <bits/stdc++.h>
using namespace std;
const int maxn = 1005;
const int maxV = maxn * maxn;
int dp[maxV], a[maxn];

int main(){
    ios_base::sync_with_stdio(false);
    int n, k;
    cin >> n >> k;
    for(int i = 1; i <= n; i++) cin >> a[i];
    sort(a + 1, a + 1 + n);
    int m = a[1];
    for(int i = 1; i <= n; i++) a[i] -= m;
    fill(dp + 1, dp + maxV, k + 1);
    for(int i = 1; i <= n; i++){
        for(int j = a[i]; j <= a[i] * k; j++){
            dp[j] = min(dp[j], dp[j - a[i]] + 1);
        }
    }
    for(int i = 0; i < maxV; i++){
        if(dp[i] <= k){
            cout << i + k * m << " ";
        }
    }
    cout << '\n';
    return 0;
}
