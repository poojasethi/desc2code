#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define N 1005
#define LL long long
#define oo (1<<29)
using namespace std;
int n,k,a[N],f[N*N];
int main()
{
	int i,j;
	scanf("%d %d",&n,&k);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	for(i=2;i<=n;i++)
		a[i]-=a[1];
	for(i=1;i<=k*a[n];i++)
		f[i]=oo;
	for(i=2;i<=n;i++)
		for(j=a[i];j<=k*a[n];j++)
			f[j]=min(f[j],f[j-a[i]]+1);
	for(i=0;i<=k*a[n];i++)
		if(f[i]<=k) printf("%d ",k*a[1]+i);
	return 0;
}
