#include <bits/stdc++.h>

using namespace std;

#define FILE ""

int ans[1000100];

int main()
{
    //freopen(FILE".in", "r", stdin);
    //freopen(FILE".out", "w", stdout);
    //
    int n , k;
    cin >> n >> k;
    int a[n];
    for (int i = 0; i < n; i++){
        char c = getchar();
        while (c < '0' || c > '9'){
            c = getchar();
        }
        a[i] = 0;
        while (c >= '0' && c <= '9'){
            a[i] *= 10;
            a[i] += (int)(c - '0');
            c = getchar();
        }
    }
    sort(a , a + n);

    int mi = a[0];
    for (int i = 0; i < n; i++){
        a[i] -= mi;
    }

    for (int i = 0; i < 1000100; i++){
        ans[i] = 1001;
    }
    ans[0] = 0;

    for (int i = 1; i <= a[n - 1] * k; i++){
        for (int j = 0; j < n && a[j] <= i; j++){
            ans[i] = min(ans[i] , ans[i - a[j]] + 1);
        }
    }

    for (int i = k * a[0]; i <= a[n - 1] * k; i++){
        if (ans[i] <= k){
            printf("%d ", i + mi * k);
        }
    }
    return 0;
}
