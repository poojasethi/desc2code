int n, k, a[N], dp[N];

int main() {
	
	scanf("%d %d", &n, &k);

	FOR(i, 1, n) 
		scanf("%d", &a[i]);

	sort(a + 1, a + n + 1);

	int mn = a[1];
	
	memset(dp, 10, sizeof dp);
	
	dp[0] = 0;
	
	FOR(i, 2, n) {
		a[i] -= mn;
		FOR(j, a[i], a[n] * k)
			dp[j] = min(dp[j-a[i]]+1, dp[j]);
	}

	FOR(i, 0, N-1)
		if(dp[i] <= k) {
			cout << i + mn * k << ' '; 
		} cout << endl;

	return 0;
}
