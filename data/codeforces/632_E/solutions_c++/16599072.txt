#include <bits/stdc++.h>
#define MOD 1000000009
using namespace std;
const int N = int(1e6)+1;


int arr[N];
int p[N];

int main(){

    int n, m;
    cin >> n >> m;

    for(int i = 0; i< n; i++)
            cin>> arr[i];

    int mn, mx;
    sort(arr, arr+n);
    mn = arr[0];
    for(int i = 0; i< n; i++)
            arr[i] = arr[i] - mn;
    mx = arr[n-1];
    mn *= m;
    mx *= m;
    cout<<mn<< " ";
    p[0] = 0;
    for(int i = 1; i<= mx; i++){

            p[i] = MOD;
            for( int j = 0; j< n; j++){

                    if( i - arr[j] >= 0) p[i] = min(p[i], p[i-arr[j]]+1);
            }
            if( p[i] <= m)
            cout<< i + mn << " ";

    }


    return 0;


}
