#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 1005
#define INF (1<<30)
#define min(x,y) ((x)>(y)?(y):(x))
int a[N],dp[N*N],ans[N*N],cnt;
int main(){
    int n,k,Min;
    while(~scanf("%d%d",&n,&k)){
        for(int i=0;i<n;++i) scanf("%d",&a[i]);
        sort(a,a+n);n=unique(a,a+n)-a;Min=a[0];
        for(int i=0;i<n;++i) a[i]-=Min;
        for(int i=1;i<N*N;++i) dp[i]=INF;dp[0]=0;
        for(int i=1;i<N*N;++i){
            for(int j=1;j<n;++j){
                if(i>=a[j]){
                    dp[i]=min(dp[i],dp[i-a[j]]+1);
                }
            }
        }cnt=0;
        for(int i=0;i<N*N;++i){
            if(dp[i]<=k) ans[cnt++]=i;
        }
        for(int i=0;i<cnt;++i){
            printf("%d",ans[i]+k*Min);
            if(i==cnt-1) printf("\n");
            else printf(" ");
        }
    }
    return 0;
}
