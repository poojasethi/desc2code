#include <bits/stdc++.h>

using namespace std;

#define N 1000005

int a[N];
int dp[N];

main() {
	int n, m;
	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; i ++) scanf("%d", a + i);
	sort(a + 1, a + n + 1);
	n = unique(a + 1, a + n + 1) - a - 1;
	for (int i = 2; i <= n; i ++) a[i] -= a[1];
	for (int i = 1; i <= a[n] * m; i ++) dp[i] = m + 1;
	dp[0] = 0;
	for (int i = 2; i <= n; i ++) {
		for (int j = a[i]; j <= a[i] * m; j ++) {
			if (dp[j] > dp[j - a[i]] + 1) {
				dp[j] = dp[j - a[i]] + 1;
			}
		}
	}
	for (int i = 0; i <= a[n] * m; i ++) {
		if (dp[i] <= m) printf("%d ", i + a[1] * m);
	}
}
