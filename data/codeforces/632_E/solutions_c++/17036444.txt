#include<bits/stdc++.h>
using namespace std;
const int maxn = 1e3+5;
int n,k,a[maxn],dp[maxn*maxn];
int main()
{
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    sort(a+1,a+1+n);
    n=unique(a+1,a+1+n)-(a+1);
    for(int i=2;i<=n;i++)
        a[i]=a[i]-a[1];
    for(int i=1;i<=k*a[n];i++)
        dp[i]=k+1;
    for(int i=2;i<=n;i++)
        for(int j=a[i];j<=k*a[i];j++)
            dp[j]=min(dp[j],dp[j-a[i]]+1);
    for(int i=0;i<=k*a[n];i++)
        if(dp[i]<=k)
            printf("%d ",a[1]*k+i);
    return 0;
}
