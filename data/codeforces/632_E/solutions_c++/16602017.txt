{

	int n,k,m;
	cin>>n>>k;
	for(int i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);
	n=unique(a,a+n)-a;
	m=a[0];
	for(int i=0;i<n;i++)
	a[i]-=m;
	fill(dp,dp+maxb,k+1);
	dp[0]=0;
	for(int i=1;i<n;i++)
	for(int j=a[i];j<=(k*a[i]);j++)
	dp[j]=min(dp[j],dp[j-a[i]]+1);
	for(int i=0;i<maxb;i++)
	if(dp[i]<=k)
	cout<<i+(k*m)<<" ";
	cout<<"\n";
	return 0;
}
