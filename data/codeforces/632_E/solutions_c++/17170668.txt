#include<cstdio>
#include<iostream>
#include<algorithm>
#define inf 999999999
using namespace std;
const int maxn = 1100;
int n, k;
int a[maxn];
int dp[maxn*maxn];

int main()
{
	scanf("%d%d", &n, &k);
	for(int i = 1;i<=n;i++)
		scanf("%d", &a[i]);	
	sort(a+1, a+n+1);
	int t = a[1];
	int num = unique(a+1, a+n+1)-(a+1);
	for(int i = 2;i<=num;i++)
		a[i] -= t;
	for(int i = 1;i<=a[num]*k;i++)
		dp[i] = inf;
	dp[0] = 0;
	for(int i = 2;i<=num;i++)
	{
		for(int j = a[i];j<=a[i]*k;j++)
		dp[j] = min(dp[j], dp[j-a[i]]+1);
	}
	for(int i = 0;i<=a[num]*k;i++)
	{
		if(dp[i] <= k)
		printf("%d ", i+t*k);
	}
	printf("\n");
	return 0;
}
