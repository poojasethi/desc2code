#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define maxn 1005
#define maxm (maxn*maxn)
using namespace std;

int f[maxm],a[maxn];

int n,fir,mx,k;

int main(){
	scanf("%d%d",&n,&k);
	fo(i,1,n) scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	mx=a[n]*k;
	fir=a[1];
	fo(i,1,n) a[i]-=fir;
	memset(f,63,sizeof(f));
	f[0]=0;
	fo(i,0,mx) {
		fo(j,1,n) 
			if (i>=a[j]) f[i]=min(f[i],f[i-a[j]]+1);
		if (f[i]<=k) printf("%d ",i+k*fir);
	}	
	return 0;
}
