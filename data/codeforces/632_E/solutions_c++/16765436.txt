#include<bits/stdc++.h>
using namespace std;
const int N=1011;
int a[N],n,k,b[N];
int dp[N*N];
int main()
{
	cin>>n>>k;
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	sort(a+1,a+n+1);
	int num=0;
	for (int i=1;i<=n;i++)
	{
		if (a[i]!=a[i-1])
		{
			num++;
			b[num]=a[i];
		}
	}
	n=num;
	for (int i=1;i<=n;i++)
	{
		a[i]=b[i];
	}
	for (int i=2;i<=n;i++)
	{
		a[i]-=a[1];
	}
	for (int i=1;i<=k*a[n];i++)
	{
		dp[i]=k+1;
	}
	for (int i=2;i<=n;i++)
	{
		for (int j=a[i];j<=k*a[i];j++)
		{
			dp[j]=min(dp[j],dp[j-a[i]]+1);
		}
	}
	for (int i=0;i<=k*a[n];i++)
	{
		if (dp[i]<=k)
		{
			printf("%d ",k*a[1]+i);
		}
	}
	return 0;
}
