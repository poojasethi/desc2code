#include<iostream>
#include<cmath>
#include<algorithm>
#include<cstdio>
using namespace std;
#define inf 100000000
int f[1001*1001],a[1001];
int n,m,k;
int main(){
    while ((scanf("%d%d",&n,&k))!=EOF){
        for (int i=1;i<=n;i++)
          cin>>a[i];
        sort(a+1,a+1+n);
        int p=a[1];
        for (int i=1;i<=n;i++) a[i]-=p;
        for (int i=1;i<=1000000;i++) f[i]=inf;
        f[0]=0;
        for (int i=0;i<=1000000;i++)
         for (int j=1;j<=n;j++)
          if (i>=a[j]) f[i]=min(f[i],f[i-a[j]]+1);
        int bj=1;
        for (int i=0;i<=1000000;i++)
         if (f[i]<=k) cout<<(i+p*k)<<' ';
    }
    return 0;
}
