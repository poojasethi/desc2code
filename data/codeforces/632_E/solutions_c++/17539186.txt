#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstring>

using namespace std;
const int N=1005;
const int inf=0x3f3f3f3f;
int a[N];
int dp[N*N];
int main()
{
    //freopen("a.txt","r",stdin);
    int n,k;
    while(~scanf("%d%d",&n,&k))
    {
        int MIN=1e9,MAX=-1;
        for(int i=1;i<=n;i++)scanf("%d",&a[i]),MIN=min(MIN,a[i]);
        for(int i=1;i<=n;i++)
        {
            a[i]=a[i]-MIN;
            MAX=max(MAX,a[i]);
        }
        sort(a+1,a+n+1);
        n=unique(a+1,a+n+1)-a-1;
        MAX*=k;
        memset(dp,inf,sizeof(dp));
        dp[0]=0;
        for(int i=1;i<=MAX;i++)
        {
            for(int j=1;j<=n&&a[j]<=i;j++)
            dp[i]=min(dp[i],dp[i-a[j]]+1);
        }
        int flag=0;
        for(int i=0;i<=MAX;i++)
        {
            if(dp[i]>k)continue;
            if(flag==0)flag=1;
            else printf(" ");
            printf("%d",i+MIN*k);
        }
        printf("\n");
    }
    return 0;
}
