int n,k,a[N],dp[N];
int main(){
	cin >> n >> k;
	FOR(i,1,n)
		scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	n = unique(a+1,a+n+1) - (a+1);
	FOR(i,2,n) a[i] -= a[1];
	memset(dp,4,sizeof dp);
	dp[0] = 0;
	FOR(i,2,n)
		FOR(j,a[i],a[n]*k)
			dp[j] = min(dp[j] , dp[j-a[i]]+1);

	FOR(i,0,k*a[n])
		if(dp[i] <= k) cout << k*a[1] + i << ' ';
}
