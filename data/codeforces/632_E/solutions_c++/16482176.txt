#include <bits/stdc++.h>

using namespace std;

int a[1010];
int dp[1001000],cnt[1000100];
int main()
{
    int i,j,k;
    int n;

    cin>>n>>k;

    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);

    n=unique(a,a+n)-a;

    fill(dp+1,dp+1000100,k+1);

    for(i=1;i<n;i++)
        a[i]-=a[0];


    for(i=1;i<n;i++)
    {
      for(j=a[i];j<=a[i]*k;j++)
      {
          dp[j]=min(dp[j],dp[j-a[i]]+1);

      }
    }

    for(i=0;i<=1000000;i++)
    {
        if(dp[i]<=k)
            printf("%d ",i + k*a[0]);
    }
}
