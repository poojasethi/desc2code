#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define fo(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define maxn 1005
#define maxm (maxn*maxn)
using namespace std;

int f[maxm],a[maxn];

int n,fir,mx,k;

int main(){
	scanf("%d%d",&n,&k);
	fo(i,1,n) scanf("%d",&a[i]);
	sort(a+1,a+n+1);
	n=unique(a+1,a+n+1)-(a+1);
	fir=a[1];
	fo(i,1,n) a[i]-=fir;
	mx=a[n]*k;
	memset(f,63,sizeof(f));
	f[0]=0;
	fo(j,2,n)
		fo(i,a[j],k*a[j])
			f[i]=min(f[i],f[i-a[j]]+1);
	fo(i,0,mx)
		if (f[i]<=k) printf("%d ",i+fir*k);
	return 0;
}
