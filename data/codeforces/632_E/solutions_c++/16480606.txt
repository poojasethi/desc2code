#include <bits/stdc++.h>
using namespace std;

const int maxn = 1001 , SIZE = maxn*maxn;

int n,k,a[maxn],dp[SIZE];

int main(){
	
	scanf("%d%d",&n,&k);
	
	for( int i = 0 ; i < n ; i++ )
		scanf("%d",&a[i]);
	
	sort(a , a + n);
	n = unique(a , a + n) - a;
	
	fill(dp , dp + SIZE , k + 5);
	dp[0] = 0;
	for( int i = 1 ; i < n ; i++ ){
		int x = a[i] - a[0];
		for( int j = 0 ; j + x < SIZE ; j++ )
			dp[j + x] = min(dp[j + x] , dp[j] + 1);
	}
	
	for( int i = 0 ; i < SIZE ; i++ )
		if( dp[i] <= k )
			printf("%d ",i + k * a[0]);
	
	return 0;
}
