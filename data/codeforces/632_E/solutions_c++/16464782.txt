#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cmath>
using namespace std;
int n,k,i,j,a[1001],f[1000001];
int main()
{
		scanf("%d%d\n",&n,&k);
		for (i=1;i<=n;i++) scanf("%d",&a[i]);
		sort(a+1,a+n+1);
		n=unique(a+1,a+n+1)-(a+1);
		for (i=2;i<=n;i++) a[i]-=a[1];
		for (i=1;i<=k*a[n];i++) f[i]=k+1;
		for (i=2;i<=n;i++)
			for (j=a[i];j<=k*a[i];j++)
				f[j]=min(f[j],f[j-a[i]]+1);
		for (i=0;i<=k*a[n];i++)
			if (f[i]<=k) printf("%d ",a[1]*k+i);
}
