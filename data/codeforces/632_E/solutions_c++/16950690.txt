const int maxn=1020 , lim=1e6;

int N,K,ar[maxn],kn[lim+20];

int main(){
	cin >> N >> K;
	
	for(int i=0;i<N;i++)
		scanf("%d",&ar[i]);
	
	sort(ar,ar+N);
	N=unique(ar,ar+N)-ar;
	
	fill(kn,kn+lim+2,K+2);
	
	kn[0]=0;
	for(int i=1;i<N;i++){
		int a=ar[i]-ar[0];
		
		for(int j=0;j+a<=lim;j++)
			umin(kn[j+a],kn[j]+1);
	}
	
	for(int i=0;i<=lim;i++)
		if(kn[i]<=K)
			printf("%d ",i+K*ar[0]);
	
	return 0;
}
