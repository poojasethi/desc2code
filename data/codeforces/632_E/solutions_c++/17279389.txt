#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#include <queue>
#include <cctype>
#include <cmath>
#include <iostream>
#include <map>
#include <set>

using namespace std;

int n,k;
int dp[1000001];
int a[1001];

int main(void){
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	sort(a,a+n);
	for(int i=a[0]*k;i<=1000000;i++){
		dp[i]=k+1;
	}
	dp[a[0]*k]=0;
	for(int i=a[0]*k;i<1000000;i++){
		if(dp[i]<k){
			for(int j=1;j<n;j++){
				if(i+a[j]-a[0]<=1000000){
					dp[i+a[j]-a[0]]=min(dp[i+a[j]-a[0]],dp[i]+1);
				}
			}
		}
	}
	vector<int> res;
	for(int i=a[0]*k;i<=1000000;i++){
		if(dp[i]<=k)res.push_back(i);
	}
	for(int i=0;i<res.size();i++){
		printf("%d%c",res[i],i+1==res.size()?'\n':' ');
	}
	return 0;
}
