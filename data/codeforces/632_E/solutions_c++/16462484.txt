#include <iostream>
#include <algorithm>
#include <cstring>
#include <cstdio>
#include <vector>
using namespace std;

const int N = 1010;
int a[N];
int dp[N * N];
int main() {
    int n, k;
    scanf("%d%d", &n, &k);
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    sort(a, a + n);
    n = unique(a, a + n) - a;
    dp[0] = 0;
    for (int i = 1; i < N * N; i++) dp[i] = N;
    for (int i = 1; i < n; i++) {
        a[i] -= a[0];
        for (int j = a[i]; j < N * N; j++) {
            dp[j] = min(dp[j], dp[j - a[i]] + 1);
        }
    }
    for (int i = 0; i < N * N; i++) {
        if (dp[i] <= k) {
            printf("%d ", a[0] * k + i);
        }
    }
}
