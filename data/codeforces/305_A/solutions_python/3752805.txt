n = int(raw_input())

a = map(int, raw_input().split())
result = []

for i in range(1, 10):
    if i in a:
        result.append(i)
        break

for i in range(10, 100, 10):
    if i in a:
        result.append(i)
        break

if len(result) == 0:
    for i in a:
        if i not in (0, 100):
            result.append(i)
            break

if 100 in a:
    result.append(100)
if 0 in a:
    result.append(0)

print len(result)
print ' '.join(map(str, result))
