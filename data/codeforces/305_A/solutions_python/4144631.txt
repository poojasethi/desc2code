N = int(raw_input())
D = set(map(int, raw_input().split()))
ans = set()

if 100 in D:
    D.remove(100)
    ans.add(100)
if 0 in D:
    D.remove(0)
    ans.add(0)
if set(range(1, 10)) & D and set(range(10, 100, 10)) & D:
    ans.add(list(set(range(1, 10)) & D)[0])
    ans.add(list(set(range(10, 100, 10)) & D)[0])
elif D:
    ans.add(list(D)[0])

print len(ans)
print ' '.join(map(str, ans))
