n = input()
arr = map(int, raw_input().split())
ans = []
if 0 in arr:
    ans.append(0)
if 100 in arr:
    ans.append(100)
d = [i for i in arr if 0<i<10]
dus = [i for i in arr if 10<=i<100 and i%10==0]

ans+=d[:1]
ans+=dus[:1]

if not d and not dus:
    ans+=[i for i in arr if i%10!=0][:1]
print len(ans)

print ' '.join(map(str,ans))
