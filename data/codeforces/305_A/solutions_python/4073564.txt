n = int(raw_input().strip())
nums = [int(i) for i in raw_input().strip().split()]
nums.sort()

total = 0
ans = []
zero = False
singledigit = False
doubledigit = False
hundred = False
other = False
for i in nums:
    if i >= 1 and i <= 9:
        singledigit = i
    elif i >= 10 and i < 100 and i % 10 == 0:
        doubledigit = i
    elif i == 0:
        zero = True
    elif i == 100:
        hundred = True
    else:
        other = i

if zero:
    total += 1
    ans.append(0)
if hundred:
    total += 1
    ans.append(100)
if singledigit:
    total += 1
    ans.append(singledigit)
if doubledigit:
    total += 1
    ans.append(doubledigit)
if ((not singledigit) and (not doubledigit) and other):
    total += 1
    ans.append(other)

ans = [str(i) for i in ans]

print total
print ' '.join(ans)
