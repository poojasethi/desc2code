n = input()
A = map(int, raw_input().split())
subset = []
if 0 in A: 
  subset.append(0)
  A.remove(0)
if 100 in A: 
  subset.append(100)
  A.remove(100)

singles = [x for x in A if x<10]
tenners = [x for x in A if x%10==0]
if not singles and not tenners and A:
  subset.append(A[0])
else:
  if singles: subset.append(singles[0])
  if tenners: subset.append(tenners[0])

print len(subset)
print ' '.join(map(str, subset))
