k = int(raw_input())

dArr = [int(inp) for inp in raw_input().split()]

mask = [False for i in range(101)]

for d in dArr:
	mask[d] = True

ans = 0
ansSet = []

if mask[0]:
	ans += 1
	ansSet += [0]


if mask[100]:
	ans += 1
	ansSet += [100]

singleDigits = False

for i in range(1,10):
	if mask[i]:
		singleDigits = True
		ansSet += [i]
		break

multipTens = False

for i in range(1,10):
	if mask[10*i]:
		multipTens = True
		ansSet += [10*i]
		break

if singleDigits and multipTens:
	ans += 2
elif singleDigits or multipTens:
	ans += 1
else:
	if k > ans:
		ans += 1
		for d in dArr:
			if 0 < d and d < 100:
				ansSet += [d]
				break

print ans
