input()
a0, a1, a10, a11, a100 = [], [], [], [], []
for x in map(int, raw_input().split()):
  if x == 0:
    a0.append(x)
  elif x < 10:
    a1.append(x)
  elif x < 100 and x % 10 == 0:
    a10.append(x)
  elif x < 100:
    a11.append(x)
  else:
    a100.append(x)

res = a0
if a1 or a10:
  if a1: res.append(a1[0])
  if a10: res.append(a10[0])
else:
  if a11: res.append(a11[0])
if a100: res.append(a100[0])

print len(res)
print ' '.join(str(x) for x in res)
