        ans[0] = 0
    elif d[i] == 100:
        ans[3] = 100

        ans[1] = d[i]
    elif d[i]%10 == 0:
        ans[2] = d[i]
    else:
        extra = d[i]
if ans[1] == -1 and ans[2] == -1:
