N=int(raw_input())
L=map(int, raw_input().split())
L.sort()
res=[]
k=[1, 10, 100]
for i in L:
    good = True
    for j in res:
        for q in k:
            if ((i // q % 10) * (j // q % 10) > 0):
                good = False
    if (good):
        res.append(i)
print len(res)
print " ".join(str(i) for i in res)
