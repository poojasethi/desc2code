n = input()
a = map(int, raw_input().split())
ans = []
b = [_ for _ in a if (0 < _ < 10)]
c = [_ for _ in a if (10 <= _ < 100) and _ % 10 == 0]
d = [_ for _ in a if (10 < _ < 100) and _ % 10 != 0]
ans = [_ for _ in a if _ == 0 or _ == 100]
if len(b) == 0 and len(c) == 0:
    if len(d):
        ans.append(d[0])
else:
    if len(b):
        ans.append(b[0])
    if len(c):
        ans.append(c[0])
print len(ans)
print " ".join(map(str, ans))
