n = input()
a = map(int, raw_input().strip().split())

chisla = []

for i in range(1, 10):
    if i in a:
        chisla.append(i)
        break
for i in range(10, 100, 10):
    if i in a:
        chisla.append(i)
        break
if len(chisla) == 0:
    for i in a:
        if i not in (0, 100):
            chisla.append(i)
            break
if 100 in a:
    chisla.append(100)
if 0 in a:
    chisla.append(0)

print len(chisla)
print ' '.join(str(x) for x in chisla)
