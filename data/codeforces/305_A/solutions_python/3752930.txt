#!/usr/bin/env python
# -*- coding: UTF-8 -*-

import sys

DEBUG = 0

def ReadIn():
    data = sys.stdin.read().split()
    offset = 0
    while offset < len(data):
        n = int(data[offset])
        offset += 1
        yield data[offset : offset + n]
        offset += n

def Solve(a):
    if DEBUG:
        print a
    s = 0
    ans = []
    for cnt in xrange(3):
        for x in a:
            if len(x) - x.count('0') != cnt: continue
            if DEBUG: print 'x =', x
            ts = 0
            for i, c in enumerate(reversed(x)):
                if c != '0':
                    ts |= 1 << i
            if s & ts: continue
            s |= ts
            ans.append(x)
    if DEBUG:
        print ans
    print len(ans)
    print ' '.join(ans)


if __name__ == '__main__':
    for a in ReadIn():
        Solve(a)
