n = int(raw_input())
a = map(int, raw_input().split())
a.sort()
res = []
k = [1, 10, 100]
for i in a:
    good = True
    for j in res:
        for q in k:
            if ((i // q % 10) * (j // q % 10) > 0):
                good = False
    if (good):
        res += [i]
print(len(res))
s = ''
for i in res:
    s += str(i) + ' '
print(s)
