I=lambda:map(int, raw_input().split())
n = input()
a = I()
a.sort()
res = []
k = [1, 10, 100]
for i in a:
    good = True
    for j in res:
        for q in k:
            if ((i // q % 10) * (j // q % 10) > 0):
                good = False
    if (good):
        res += [i]
print(len(res))
print ' '.join(map(str,res))
