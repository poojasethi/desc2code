n = int(raw_input())
a = map(int, raw_input().split())
b = [x for x in a if 0 < x and x < 10]
c = [x for x in a if x >= 10 and x < 100 and x%10 == 0]
d = [x for x in a if x > 10 and x < 100 and x%10 != 0]

ans = []
if 0 in a: ans.append(0)
if 100 in a: ans.append(100)
if len(b) > 0: ans.append(b[0])
if len(c) > 0: ans.append(c[0])
if len(b) == 0 and len(c) == 0 and len(d) > 0: ans.append(d[0])

print len(ans)
print ' '.join(map(str, ans))
