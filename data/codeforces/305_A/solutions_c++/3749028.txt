#include <cstdio>
#include <iostream>
using namespace std;
int k,d,a[1005],i,ans=0,x=0,y=0,t=0;
int main (){
	scanf ("%d",&k);
	for (i=1;i<=k;i++){
		scanf ("%d",&d);
		if (d==0||d==100) a[ans++]=d;
		else if (d<10&&x==0) a[ans++]=d,x=1;
		else if (d%10==0&&y==0) a[ans++]=d,y=1;
		else t=d;
	}
	if (x==0&&y==0&&t!=0) a[ans++]=t;
	printf ("%d\n",ans);
	for (i=0;i<ans-1;i++) printf ("%d ",a[i]);
	printf ("%d\n",a[i]);
	return 0;
}
