#include <cstring>

#include <cstdio>
#define N 110
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,kk,u;
int a[N],b[N];
int main() {
	scanf("%d",&n);
	For(i,1,n) {
		scanf("%d",&a[i]);
		if (!a[i]||a[i]==100) b[++*b]=a[i];
		else if (a[i]<10) k=a[i];
		else if (a[i]%10==0) kk=a[i];
		else u=a[i];
	}
	if (k) b[++*b]=k;
	if (kk) b[++*b]=kk;
	if (!k&&!kk&&u) b[++*b]=u;
	printf("%d\n",*b);
	For(i,1,*b) printf("%d ",b[i]);

}
