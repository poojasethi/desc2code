#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
using namespace std;
int num,x,i,n,p1,p2,p3,p4;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%d",&x);
		if(x==0)num++;
		else if(x==100)p1=100;
		else if(x<10)p2=x;
		else if(x%10==0)p3=x;
		else p4=x;
	}
	x=num;
	if(p1)num++;
	if(p2)num++;
	if(p3)num++;
	if(!p2 && !p3 && p4)num++;
	printf("%d\n",num);
	for(i=1;i<=x;++i)printf("0 ");
	if(p1)printf("%d ",p1);
	if(p2)printf("%d ",p2);
	if(p3)printf("%d ",p3);
	if(!p2 && !p3 && p4)printf("%d ",p4);
}
