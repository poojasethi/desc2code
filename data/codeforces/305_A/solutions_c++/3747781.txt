#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
int a[110],b[110],n,cnt;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i) 
	{
		scanf("%d",&a[i]);
		if (a[i]==100) b[3]=i;
		if (10<=a[i]&&a[i]<100&&a[i]%10==0) b[2]=i;
		if (a[i]==0) b[1]=i;
	}
	for (int i=1;i<=3;++i) if (b[i]) cnt++;
	for (int i=1;i<=n;++i)
		if (a[i]!=100&&a[i]!=0&&a[i]%10!=0&&(!b[2]||a[i]<10))
		{
			cnt++;
			b[0]=i;
			break;
		}
	printf("%d\n",cnt);
	for (int i=0;i<=3;++i) if (b[i]) printf("%d ",a[b[i]]);
		
	//system("pause");
}
