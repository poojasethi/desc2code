#include <iostream>
using namespace std;

int a[1111], x, n, s, y;
bool f1, f2;

int main()
{
    while(cin>>n)
    {
        for (int i = 0; i < n; i++)
        {
            cin >> x;
            if (x == 0 || x == 100) a[s++] = x;
            else if (x < 10 && !f1) a[s++] = x, f1 = true;
            else if (x % 10 == 0 && !f2 ) a[s++] = x, f2 = true;
            else y = x;
        }
        if (!f1 && !f2 && y != 0) a[s++] = y;
        cout << s << endl;
        for (int i = 0; i < s; i++)
            cout << a[i] << " ";
    }
    return 0;
}
