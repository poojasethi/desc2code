#include<cstdio>
#include<cstring>
using namespace std;
int n;
int d[105],b[105];
int main(){
	scanf("%d",&n);
	int X=0,C=0;
	for (int i=0;i<n;i++){
			scanf("%d",&d[i]);
			if (d[i]==0||d[i]==100)b[i]=1; 
			else if (d[i]%10==0&&X==0) {b[i]=1; X=1;}
			if (b[i])C++;
	}
	if (C<n)
		for (int i=0;i<n;i++)
			if (d[i]%10!=0&&d[i]/10==0||C==0||X==0&&b[i]==0){
					b[i]=1;
					C++;
					break;
				}
	printf("%d\n",C);
	for (int i=0;i<n;i++)
		if (b[i])
			printf("%d ",d[i]);
	return 0;
}
