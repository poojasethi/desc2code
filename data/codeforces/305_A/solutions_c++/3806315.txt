#include <iostream>
using namespace std;

int a[1111], x, n, s, y, f1, f2;

int main(){
    cin >> n;
    for (int i = 0; i < n; i++){
        cin >> x;
        if (x == 0 || x == 100) 
            a[s++] = x;
        else if (x < 10 && f1 == 0) 
            a[s++] = x, f1 = 1;
        else if (x % 10 == 0 && f2 == 0) 
            a[s++] = x, f2 = 1;
        else 
            y = x;
    }
    if (f1 == 0 && f2 == 0 && y != 0) 
        a[s++] = y;
    cout << s << endl;
    for (int i = 0; i < s; i++) cout << a[i] << " ";
    return 0;
}
