#include<stdio.h>

int k;
int ans[105],n;
bool flag1,flag2;

int main() {
	int i,x,y=0;
	scanf("%d",&k);
	while(k--) {
		scanf("%d",&x);
		//printf("%d\n",x%10);
		if(x==0 || x==100 )	ans[n++]=x;
		else if(x%10==0 && flag1==0 )	ans[n++]=x,	flag1=1;
		else if(x/10==0 && flag2==0 )	ans[n++]=x,	flag2=1;
		else	y=x;
	}
	if(flag1==0 && flag2==0 && y)
		ans[n++]=y;
	printf("%d\n",n);
	for(i=0;i<n;i++) {
		printf("%d",ans[i]);
		if(i==n-1)	printf("\n");
		else		printf(" ");
	}
	return 0;
}
