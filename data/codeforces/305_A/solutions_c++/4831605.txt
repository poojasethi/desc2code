#include<stdio.h>
	
int main(){
	int i,n,f1,f2,f3,f4,f5,x;
	int a[5],m=0;
	f1=f2=f3=f4=f5=0;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&x);
		if(!x&&!f1) f1=1,a[m++]=0;
		else if(x==100&&!f2) f2=1,a[m++]=100;
		else if(x>0&&x<10&&!f3) f3=1,a[m++]=x;
		else if(x<100&&x>10&&x%10) f5=x;
		else if(x<100&&x>=10&&x%10==0&&!f4) f4=1,a[m++]=x;
//		printf("x=%d a=%d\n",x,a[m-1]);
	}
	for(i=0;i<m;i++){
		if(a[i]!=0&&a[i]!=100) break;
	}
	if(i==m&&f5) a[m++]=f5;
	printf("%d\n",m);
	for(i=0;i<m;i++)
		printf("%d ",a[i]);
}
