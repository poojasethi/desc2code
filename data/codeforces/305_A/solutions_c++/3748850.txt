#include <cstdio>
#include <iostream>
#include <cstring>
#include <string>
using namespace std;

int a[1111], x, n, ans, y, f1, f2;

int main(){
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout); 
	cin >> n;
	for (int i = 0; i < n; i++){
		cin >> x;
		if (x == 0 || x == 100) a[ans++] = x; else
		if (x < 10 && f1 == 0) a[ans++] = x, f1 = 1; else
		if (x % 10 == 0 && f2 == 0) a[ans++] = x, f2 = 1; else y = x;

	}
	if (f1 == 0 && f2 == 0 && y != 0) a[ans++] = y;
	cout << ans << endl;
	for (int i = 0; i < ans; i++) cout << a[i] << " ";
	return 0;
}
