#include <iostream>

using namespace std;

 int i, a[1111]={0}, x, n, s, y, f1=0, f2=0;

int main(){
   
    cin >> n;
    
    for (i = 0; i < n; i++) {
        cin >> x;
        
        if(x == 0 || x == 100) a[++s] = x;
        else if(x < 10 && f1 == 0) a[++s] = x, f1 = 1;
             else if(x % 10 == 0 && f2 == 0) a[++s] = x, f2 = 1;
                  else y = x;
    }
    
    if (f1 == 0 && f2 == 0 && y != 0) a[++s] = y;
    cout << s << '\n';
    
    for (i = 1; i <= s; i++) cout << a[i] << " ";
   // system("pause");
    return 0;
}
