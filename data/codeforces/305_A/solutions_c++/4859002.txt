#include<iostream>
#include<algorithm>

using namespace std;

int n,m,i,f,k,a,c,g[105];

main(){
       
cin>>n;

for(i=0;i<n;i++){
cin>>k; 
if(k==100) {g[f++]=k;continue;}
if(k==0) {g[f++]=k;continue;}
if(k/10>0 && k%10==0 && c==0) {g[f++]=k;c=1;continue;}
if(k/10>0 && c==0) {m=k;continue;}
if(k/10==0 && a==0) {g[f++]=k;a=1;continue;}
}
if(c==0 && m>0 && a==0) {g[f++]=m;}
cout<<f<<endl;

for(i=0;i<f;i++){
cout<<g[i]<<" ";
}

//system("pause");
}
