#include<iostream>
#include<vector>

using namespace std;

int main(){
	int A[101],n,i,j,k=-1,l=-1,q=-1,len=0;
	cin >> n;
	for(i=0;i<n;i++){
		cin >> j;
		if(j==0 || j==100){
			A[len++]=j;
		}
		else if(j>0 && j<10){
			k=j;
		}
		else if(j%10==0){
			l=j;
		}
		else if(j>10 && j<100){
			q=j;
		}
	}
	if(k!=-1){
		A[len++]=k;
	}
	if(l!=-1){
		A[len++]=l;
	}
	if(k==-1 && l==-1 && q!=-1){
		A[len++]=q;
	}
	cout << len << endl;
	for(i=0;i<len;i++){
		cout << A[i] << ' ';
	}
	cout << endl;
	return 0;
}
