#include<bits/stdc++.h>
using namespace std;

int main()
{
   int k,d,sz=0;
   int a[5]={-1,-1,-1,-1,-1};
   scanf("%d",&k);
   for(int i=0;i<k;i++)
   {
     scanf("%d",&d);
     if(d==0) a[0]=0;
     else if(d==100) a[1]=d;
     else if(d%10==0) a[2]=d;
     else if(d<10) a[3]=d;
     else a[4]=d;
   }
    if(a[2]==-1&&a[3]==-1&&a[4]!=-1) a[2]=a[4];
 for(int i=0;i<4;i++) if(a[i]!=-1) sz++;
 if(sz==0) a[0]=d,sz++;

   printf("%d\n",sz);
   for(int i=0;i<4;i++)
     if(a[i]!=-1)
       printf("%d ",a[i]);
}
