#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<set>
using namespace std;
bool sign[100050];
int g[10050];
int main()
{
    int i,j,k,n=0,m=0,T,f1=0,f2=0;
    scanf("%d",&T);
    for(i=0;i<T;++i)
    {
        scanf("%d",&k);
        if(k==0||k==100)g[n++]=k;
        else if(k%10==0&&f1==0)f1=1,g[n++]=k;
        else if(k<10&&f2==0)f2=1,g[n++]=k;
        else m=k;
    }
    if(f1==0&&f2==0&&m!=0)g[n++]=m;
    printf("%d\n",n);
    for(i=0;i<n;++i)printf("%d ",g[i]);


    return 0;
}
