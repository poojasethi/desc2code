#include<cstdio>
int n,x,ans,b0,b1,b2,a[105],o[105];
int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;i++){
		scanf("%d",&x); a[i]=x;
		if (x==0) o[ans++]=x;
		if (x && x<10 && !b0) o[ans++]=x,b0=1;
		if (x && x<100 && x%10==0 && !b1) o[ans++]=x,b1=1;
		if (x==100 && !b2) b2=1,o[ans++]=x;
		}
	for (int i=0;i<n;i++){
		x=a[i];
		if (x && x<100 && x%10 && !b0 && !b1) o[ans++]=x,b0=b1=1;
		}
	printf("%d\n",ans);
	for (int i=0;i<ans;i++) printf("%d%s",o[i],i<ans-1?" ":"\n");
	return 0;
}
