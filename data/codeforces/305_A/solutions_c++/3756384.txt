#include <iostream>
#include <set>
using namespace std;
int main()
{int k; cin>>k;
set <int> a,ans;
for (int i=0;i<k;i++) {int x; cin>>x; a.insert(x);}
set <int>::iterator it;
bool f1=true, f2=true;
for (it=a.begin(); it!=a.end();it++)
 {int x=*it;
  if (x==0||x==100) ans.insert(x);
  if (x>0&&x<10&&f1) {ans.insert(x);f1=false;continue;}
  if (x>=10&&x%10==0&&f2) {ans.insert(x);f2=false;continue;}
  if (x>=10&&f1&&f2)
   {ans.insert(x);f2=false;f1=false;continue;}
 }
cout<<ans.size()<<endl;
for (it=ans.begin(); it!=ans.end();it++)  cout<<*it<<" ";
return 0;
}
