#include<iostream>
#include<algorithm>
#include<cmath>
#include<string>
#include<conio.h>
using namespace std;

int i,j,n,a[105],b[105],rs;
bool u,u1;

int main()
{
  cin>>n;
  for(i=1;i<=n;++i) 
  {
    cin>>a[i];
    if(!a[i] || a[i]==100) b[++rs]=a[i]; 
    else if(!(a[i]%10) && !u) b[++rs]=a[i],u=1;
         else if(a[i]<10 && !u1) b[++rs]=a[i],u1=1;
              else j=i;
  }

  if(!u && !u1 && j) b[++rs]=a[j];

  cout<<rs<<'\n';
  for(i=1;i<=rs;++i) cout<<b[i]<<' ';

 return 0;
}
