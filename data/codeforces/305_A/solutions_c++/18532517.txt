#include<iostream>
#include<cmath>
#include<algorithm>
#include<stdio.h>
using namespace std;

int n,a,ii,r;
int l[5];
bool f,ff,x;
int main(){

	scanf("%d",&n);
	for(int i=1;i<=n;++i){
		scanf("%d",&a);
		if(a==0) l[++ii]=a;
		else
		if(a==100) l[++ii]=a;
		else
		if(a%10==0&&!f) l[++ii]=a,f=1;
		else
		if(a<10&&!ff) l[++ii]=a,ff=1;
		else
		r=a;
	}
	if(!f && !ff && r) l[++ii]=r;
	printf("%d\n",ii);
	for(int i=1;i<=ii;++i) printf("%d ",l[i]);

	return 0;
}
