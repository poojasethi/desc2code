#include <iostream>
#include <algorithm>
using namespace std;
int n,i,d,a[111],b[111],c,f,e,s,h,v,j;
int main()
{
	cin>>n;
	for (i=1;i<=n;i++)
		cin>>a[i];


	for (i=1;i<=n;i++)
		{if(a[i]==0){d=1;b[f+c+e+d]=0;}
		else if(a[i]<10 && a[i]!=0 && s==0){c=1;b[f+e+d+c]=a[i];s=1;}
		else if(a[i]<100 && a[i]>9 && a[i]%10==0 && v==0){v=1;e=1;b[f+d+c+e]=a[i];}
		else if(a[i]==100){f=1;b[d+c+e+f]=100;}
		else h=a[i];
		}
	if(s==0 && v==0 && h!=0){b[d+e+c+f+1]=h;j=1;}
	cout<<d+c+e+f+j<<endl;
	for (i=1;i<=d+c+e+f+j;i++)
		cout<<b[i]<<" ";
		return 0;
}
