#include <iostream>

using namespace std;


int main()
{
    int n ,r=0,a , h=-1;
    int x[100];
    cin>>n;
    bool two=1 , three=1;
   for(int i=0;i<n;i++){
    cin>>a;
    if(a==0 || a==100)x[r++]=a;
    else if(a>0 && a<10 && two ){x[r++]=a;two=0;}
    else if(a>=10 && a<100 && a%10==0 && three){x[r++]=a;three=0;}
    else {h=a;}

   }
   if( two && three && h!=-1)x[r++]=h;

   cout<<r<<endl;
   if(r!=0)
    cout<<x[0];
   for(int i=1;i<r;i++)cout<<" "<<x[i];
   cout<<endl;

    return 0;
}
