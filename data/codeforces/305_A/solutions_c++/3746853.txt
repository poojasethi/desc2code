#include<iostream>
using namespace std;
int main(){
    int n,a[100],t,j=0,f1=0,f2=0,m=0;
    cin >> n;
    for(int i=0; i<n; i++){
        cin >> t;
        if(t==0||t==100)a[j++]=t;
        else if(t<10&&f1==0)a[j++]=t,f1=1;
        else if(t%10==0&&f2==0)a[j++]=t,f2=1;
        else if(t%10!=0)m=t;
    }
    if(f1==0&&f2==0&&m!=0)a[j++]=m;
    cout << j << endl;
    for(int i=0; i<j; i++){
        cout << a[i] << " ";
    }
    cout << endl;
return 0;
}
