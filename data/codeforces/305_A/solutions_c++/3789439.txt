#include <iostream>
#include <cstring>
#include <vector>

using namespace std;

int n, a, a1=-1, a2=-1, a3=-1;
vector<int> v;

int main()
{
	cin>>n;
	for(int i=0; i<n; i++)
	{
		cin>>a;
		if(a==0) v.push_back(a);
		else if(a==100) v.push_back(a);
		else if(a%10==0) a1=a;
		else if(a/10==0) a2=a;
		else a3=a;
	}
	if(a1!=-1)
	{
		v.push_back(a1);
		if(a2!=-1) v.push_back(a2);
	}
	else 
	{
		if(a2!=-1) v.push_back(a2);
		else if(a3!=-1) v.push_back(a3);
	}
	cout<<v.size()<<endl;
	for(int i=0; i<v.size(); i++) cout<<v[i]<<" "; cout<<endl;
	return 0;
}
