#include<bits/stdc++.h>
using namespace std;
int i,ans[102],k,x,n=0,y=0,f=0,f1=0;
main()
{
    cin>>k;
    for (i=0;i<k;i++)
        {
            cin>>x;
            if (x==0||x==100) ans[++n]=x;
            else if (x<10&&!f) ans[++n]=x,f=1;
            else if (x%10==0&&!f1) ans[++n]=x,f1=1;else y=x;
        }
    if (!f&&!f1&&y) ans[++n]=y;
    cout<<n<<'\n';
    for (i=1;i<=n;i++)
        cout<<ans[i]<<' ';
    return 0;
}
