/*
尺取法
*/
#include <stdio.h>
#include <string.h>
#include <map>
#include <queue>
#include <algorithm>
using namespace std;

const int maxn=200000+5;
int a[maxn],b,ans[maxn];

int main()
{
	int i,j,k,n,m,p,P,tot=0;
	map<int,int> mp;
	scanf("%d%d%d",&n,&m,&P);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(i=1;i<=m;i++){
		scanf("%d",&b);
		mp[b]++;
	}
	for(p=1;p<=P;p++){
		int s=p,t=p;
		map<int,int> tp;
		queue<int> que;
		for(i=p;i<=n;i+=P){
			que.push(i);
			tp[a[i]]++;
			if(que.size()>m){
				int t=que.front();
				que.pop();
				if(--tp[a[t]]==0)
					tp.erase(a[t]);
			}
			if(tp!=mp) continue;
			int s=que.front();
			que.pop();
			ans[tot++]=s;
			if(--tp[a[s]]==0)
				tp.erase(a[s]);
		}
	}
	sort(ans,ans+tot);
	printf("%d\n",tot);
		for(i=0;i<tot;i++){
			printf("%d%c",ans[i],i==tot-1?'\n':' ');
		}
	return 0;
}
