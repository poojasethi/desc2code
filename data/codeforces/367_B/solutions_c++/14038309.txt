#include <cstdio>
#include <map>
using namespace std;
typedef long long ll;
const int N=2e5+5;
int n,m,p,a[N],b,tot;
bool flag[N];
map<int,int> cnt;

void add(int i,int d){
	int &x=cnt[i];
	if(!x) tot++;
	x+=d;
	if(!x) tot--;
}
int main(){
	scanf("%d%d%d",&n,&m,&p);
	for(int i=1;i<=n;i++) scanf("%d",a+i);
	for(int i=0;i<m;i++){
		scanf("%d",&b);
		add(b,1);
	}
	for(int j=1;j<=p&&j+(ll)(m-1)*p<=n;j++){
		for(int i=0;i<m-1;i++) add(a[j+i*p],-1);
		int i=j;
		for(;i+(ll)(m-1)*p<=n;i+=p){
			add(a[i+(m-1)*p],-1);
			if(!tot) flag[i]=1;
			add(a[i],1);
		}
		for(;i<=n;i+=p) add(a[i],1);
	}
	int ans=0;
	for(int i=1;i<=n;i++) if(flag[i]) ans++;
	printf("%d\n",ans);
	bool first=1;
	for(int i=1;i<=n;i++) if(flag[i]){
		if(first) first=0; else putchar(' ');
		printf("%d",i);
	}
	puts("");
}
