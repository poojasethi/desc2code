#include <iostream>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <utility>
#include <algorithm>
#include <cmath>
#include <map>

using namespace std;

int n,m,p;
vector<int> v[200010],ans;
map<int,int> a,b;

int main() {
	cin >> n >> m >> p;
	for (int i = 0,aa; i < n; i++) {
		cin >> aa;
		v[i%p].push_back(aa);
	}
	for (int i = 0,bb; i < m; i++) {
		cin >> bb;
		b[bb]++;
	}
	for (int i = 0; i < p; i++) {
		if (v[i].size() < m) break;
		a.clear();
		for (int j = 0; j < v[i].size(); j++) {
			a[v[i][j]]++;
			if (j < m-1) continue;
			bool works = true;
			for (map<int,int>::iterator it = b.begin(); it != b.end() && works; it++)
				if (it->second != a[it->first]) works = false;
			if (works) ans.push_back((j-m+1)*p + i + 1);
			a[v[i][j-m+1]]--;
		}
	}
	sort(ans.begin(),ans.end());
	cout << ans.size() << '\n';
	for (int i = 0; i < ans.size(); i++) cout << ans[i] << ' ';
}
