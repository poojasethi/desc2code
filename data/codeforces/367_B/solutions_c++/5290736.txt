#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<set>
#include<map>
using namespace std;
const int maxn=2e5+45;

int a[maxn];
int n,m,p;
map<int,int> A,B;
void ins(int x,int t)
{
    B[x]+=t;
    if(!B[x])B.erase(x);
}
int ans[maxn];
int main()
{
    cin>>n>>m>>p;
    if(1ll*(m-1)*p+1>n){puts("0");return 0;}
    for(int i=1;i<=n;i++)
    {
        scanf("%d",a+i);
    }
    for(int i=1,x;i<=m;i++)scanf("%d",&x),A[x]++;   ///A is set of b(target)
    for(int i=1;i<=p;i++)       ///枚举开始点
    {
        B.clear();
        for(int j=i,k;j<=n;j+=p)
        {
            ins(a[j],1);
            k=j-m*p;
            if(k>0)ins(a[k],-1);
            k=j-(m-1)*p;
            if(k>0&&B==A)///!B.size())
            {
                ans[++ans[0]]=k;
            }
        }
    }
    sort(ans+1,ans+ans[0]+1);
    printf("%d\n",ans[0]);
    for(int i=1;i<=ans[0];i++)
        printf("%d ",ans[i]);
}
