#include<bits/stdc++.h>
using namespace std;
int r,c,n,k;
int a[15][15];
int main ()
{
	cin>>r>>c>>n>>k;
	for(int i=1; i<=n; i++)
	{
		int x,y;
		cin>>x>>y;
		a[x][y] = 1;
	}
	int ans=0;
	for(int i=1; i<=r; i++)
		for(int j=1; j<=c; j++)
			for(int ii=i; ii<=r; ii++)
				for(int jj=j; jj<=c; jj++)
				{
					int t=0;
					for(int iii=i; iii<=ii; iii++)
						for(int jjj=j; jjj<=jj; jjj++)
							t+=a[iii][jjj];
					if(t>=k)
						ans++;
				}
	cout<<ans;
}
