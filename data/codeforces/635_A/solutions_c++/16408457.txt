#include<bits/stdc++.h>
using namespace std;
int m[10][10]={};
int r, cc, n, k;
int ok(int a, int b ,int c, int d){
	int i, j;
	int ans=0;
	for(i=a;i<=b;i++)
		for(j=c;j<=d;j++)
			ans+=m[i][j];
	return k<=ans;
}
int main(){
	cin >> r >> cc >> n >> k;
	int a, b, c, d;
	int i, j;
	for(i=0;i<n;i++){
		cin >> a >> b;
		m[a-1][b-1] = 1;	
	}
	int ans = 0;
	for(a=0;a<r;a++)
		for(b=a;b<r;b++)
			for(c=0;c<cc;c++)
				for(d=c;d<cc;d++){
					ans += ok(a,b,c,d);	
				}
	cout << ans << endl;
}
