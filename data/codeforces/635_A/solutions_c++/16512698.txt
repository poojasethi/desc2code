#include<iostream>
using namespace std;

int main(){
	int r,c,n,k;	
	int x[10], y[10];
	cin >> r >> c >> n >> k;
	for (int i = 0; i < n; i++)	
		cin >> x[i] >> y[i];
	
	int cnt = 0;
	for (int i = 1; i <= r; i++)
		for ( int ii = i; ii <= r; ii++)
			for (int j = 1; j <= c; j++)
				for (int jj = j; jj <= c ; jj++){
					int v = 0;
					for (int t = 0; t < n; t++)
						if (i <= x[t] && x[t] <=ii 
						&& j <= y[t] && y[t] <= jj)
							v++;
					if (v >= k)
						cnt++;
				}
	cout << cnt << endl;
	return 0;
}
