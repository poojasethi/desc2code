#include <iostream>

using namespace std;

int main(){
	int r, c, n, k;
	cin >> r >> c >> n >> k;
	int x[n], y[n], res=0, xi, yi;
	for (int i=0; i<n; i++) {
		cin >> xi >> yi;
		x[i]=xi-1;
		y[i]=yi-1;
	}
	for (int i=0; i<r; i++) {
		for (int j=0; j<c; j++) {
			for (int i2=i; i2<r; i2++) {
				for (int j2=j; j2<c; j2++) {
					int violasenimg=0;
					for (int v=0; v<n; v++) {
						if (i <= x[v] && x[v] <= i2 && j <= y[v] && y[v] <= j2) violasenimg++;
					} if (violasenimg>=k) res++;
				}
			}
		}
	} cout << res << endl;
}
