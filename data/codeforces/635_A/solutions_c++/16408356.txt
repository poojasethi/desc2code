#include<cstdio>

int x,y,X,Y,a,b,wyn;
int t[20][20],p[20][20];

int main()
{
	scanf("%d%d%d%d",&x,&y,&a,&b);
	for(int i=0;i<a;i++)
	{
		scanf("%d%d",&X,&Y);
		t[X][Y]++;
	}
	for(int i=1;i<=x;i++)for(int j=1;j<=y;j++)
	{
		p[i][j]=p[i][j-1]+p[i-1][j]-p[i-1][j-1]+t[i][j];
	}
	for(int x1=1;x1<=x;x1++)for(int x2=1;x2<=x1;x2++)for(int y1=1;y1<=y;y1++)for(int y2=1;y2<=y1;y2++)
	{
		//printf("%d %d %d %d\n",x2,x1,y2,y1);
		if(p[x1][y1]-p[x2-1][y1]-p[x1][y2-1]+p[x2-1][y2-1]>=b)wyn++;
	}
	printf("%d",wyn);
}
