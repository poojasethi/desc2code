#include <iostream>
using namespace std;
int g[11][11]={0};
int main(){
	int r,c,n,k,x,y,t=0,t1=0;
	cin>>r>>c>>n>>k;
	for(int i=0;i<n;i++){
		cin>>x>>y;
		g[x-1][y-1]=1;
	}
	
	for(int i=0;i<r;i++){
		for(int j=i;j<r;j++){
			for(int s=0;s<c;s++){
				for(int m=s;m<c;m++){
					t=0;
					for(int x=i;x<=j;x++){
						for(int y=s;y<=m;y++){
							if(g[x][y]==1)
							t++;
						}
					}
					if(t>=k)
					t1++;
				}
			}
		}
	}
	
	cout<<t1;
}
