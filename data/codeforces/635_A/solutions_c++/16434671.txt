#include <iostream>
using namespace std;
int f[20][20]={0};
int tool(int q, int w, int e, int r)
{   int t=0;
	for (int i=q; i<=w; i++)
	for (int j=e; j<=r; j++)
	t=t+f[i][j];
	return t;
}
int main()
{
	int n,k,l,m,r,c,o=0;
	cin>>r>>c>>n>>k;
	for (int i=1; i<=n; i++) {
		int x,y;
		cin>>x>>y;
		f[x][y]=1;
	}
	for (int i=1; i<=r; i++)
	for (int j=1; j<=r; j++)
	for (int s=1; s<=c; s++)
	for (int p=1; p<=c; p++)
	if (tool(i,j,s,p)>=k) o++;	
	cout<<o;
	return 0;
}
