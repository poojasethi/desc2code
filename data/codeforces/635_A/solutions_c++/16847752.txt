#include<bits/stdc++.h>
#define For(i,j,k) for (i=j;i<=k;i++)
using namespace std;

const int dmax=1010;
int a[dmax][dmax];

int main(){
	int i,j,k,m,n,c,sum=0,ans=0,x,y,p,q;
	scanf("%d%d%d%d",&n,&m,&k,&c);
	For(i,1,k){
		scanf("%d%d",&x,&y);
		a[x][y]=1;
	}
	For(i,1,n)
		For(j,1,m)
			For(p,i,n)
				For(q,j,m){
					sum=0;
					For(x,i,p)
						For(y,j,q)
							sum+=a[x][y];
					if (sum>=c)
						ans++;
				}
	printf("%d\n",ans);
	return 0;
}
