#include<bits/stdc++.h>
#define L 1003

using namespace std;
int a[40][40], i, j, k, n,  f, m,  s=0, r, c, x, y, o;



int main(){
	
	cin>>r>>c>>n>>o;
	
	for(i=1; i<=n; ++i){
			cin>>x>>y;
			a[x][y]=1;
		}			

	for(i=1; i<=r; ++i)
		for(j=1; j<=c; ++j)
			a[i][j]+=a[i-1][j]+a[i][j-1]-a[i-1][j-1];
			
		
	for(i=1; i<=r; ++i)
		for(j=1; j<=c; ++j)
			for(k=i-1; k>=0; --k)
				for(f=j-1; f>=0; --f)
					if (a[i][j]-a[k][j]-a[i][f]+a[k][f]>=o) ++s;
			
		cout<<s<<"\n";
	
	return 0;
}
