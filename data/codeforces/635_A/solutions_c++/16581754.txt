#include<iostream>
using namespace std;
int a[10][10],r,c,n,k,ans=0;main()
{
cin>>r>>c>>n>>k;
int t,t1;
for(int i=0;i<n;i++){
cin>>t>>t1;a[t][t1]=1;}
for(int i=1;i<=r;i++)
for(int j=1;j<=c;j++)
{
for(int x=i;x<=r;x++)
for(int y=j;y<=c;y++)
{
int sum=0;
for(int f=i;f<=x;f++)
for(int s=j;s<=y;s++)
if(a[f][s]==1)
sum++;
if(sum>=k)
ans++;
}
}  
cout<<ans<<endl;

}
