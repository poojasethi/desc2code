#include<stdio.h>
int main()
{
	int r,c,n,k,ans=0,x1,x2,y1,y2,x[20],y[20],i,sum;
	scanf("%d%d%d%d",&r,&c,&n,&k);
	for(i=0;i<n;i++)
	scanf("%d%d",&x[i],&y[i]);

	for(x1=0;x1<=r;x1++)				
	for(y1=0;y1<=c;y1++)
	for(x2=x1+1;x2<=r;x2++)
	for(y2=y1+1;y2<=c;y2++)
	{
		for(i=0,sum=0;i<n;i++)
		if(x[i]>x1&&x[i]<=x2&&y[i]>y1&&y[i]<=y2)
		sum++;	
		if(sum>=k)
		ans++;
	}
	printf("%d",ans);
}
