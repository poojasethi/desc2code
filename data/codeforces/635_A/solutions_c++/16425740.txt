#include <bits/stdc++.h>
using namespace std;
int a[12][12], s[12][12];
int main(){
 int r, c, n, k, x, y; 
 cin>>r>>c>>n>>k;
 for (int i=1; i<=n; i++){
  cin>>x>>y;
  a[x][y]=1;
 }
 int ans=0;
 for (int i=1;i<=r;i++)
   for (int j=1;j<=c;j++)
     for (int i1=i;i1<=r;i1++)
       for (int j1=j;j1<=c;j1++){
	     int cnt=0;
         for (int i2=i;i2<=i1;i2++)
           for (int j2=j;j2<=j1;j2++)
             cnt+=a[i2][j2];
             if (cnt>=k) 
			   ans++;
   }


cout<<ans;
	return 0;
}
