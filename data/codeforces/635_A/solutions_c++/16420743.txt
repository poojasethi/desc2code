#include <cstdio>
int map[15][15];
int main()
{
	int r,c,n,k;
	scanf("%d %d %d %d",&r,&c,&n,&k);
	int x,y,ans=0;
	for (int i=0;i<n;i++)
	{
		scanf("%d %d",&x,&y);
		map[x][y]=1;
	}
	for (int i=1;i<=r;i++)
	{
		for (int j=1;j<=c;j++)
		{
			for (int p=i;p<=r;p++)
			{
				for (int q=j;q<=c;q++)
				{
					int w=0;
					for (int a=i;a<=p;a++)
					{
						for (int b=j;b<=q;b++)
						{
							if (map[a][b]) w++;
						}
					}
					if (w>=k) ans++;
				}
			}
		}
	}
	printf("%d",ans);
}
