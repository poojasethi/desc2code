#include<bits/stdc++.h>
using namespace std;
bool arr[50][50];
int r,c,n,k,ans;
int cnt(int i,int j,int a,int b)
{
	int c=0;
	for(int k=i;k<=a;k++)
		for(int l=j;l<=b;l++)
			if(arr[k][l])
				c++;
	return c;
}
int main()
{
	cin>>r>>c>>n>>k;
	while(n--)
	{
		int a,b;
		cin>>a>>b;
		arr[a][b]=true;
	}
	for(int i=1;i<=r;i++)
		for(int j=1;j<=r;j++)
			for(int a=i;a<=r;a++)
				for(int b=j;b<=c;b++)
					if(cnt(i,j,a,b)>=k)
						ans++;
	cout<<ans<<endl;
	return 0;
}
