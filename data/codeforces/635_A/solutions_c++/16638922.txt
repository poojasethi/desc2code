#include <iostream>
using namespace std;
int x[100],y[100];
int main(){
	int r,c,n,k;
	cin>>r>>c>>n>>k;
	int k1=k;
	for(int i=1;i<=n;i++){
		cin>>x[i];
		cin>>y[i];
	}
	int ans=0;
	for(int i=1;i<=r;i++){
		for(int j=1;j<=c;j++){
			for(int k=i;k<=r;k++){
				for(int l=j;l<=c;l++){
					int sum=0;
					for(int m=1;m<=n;m++){
						if(x[m]>=i && x[m]<=k && y[m]>=j && y[m]<=l){
							sum++;
						}
					}
					if(sum>=k1){
						ans++;
					}
				
				}
			}
		}
	}
	cout<<ans;
	
}
