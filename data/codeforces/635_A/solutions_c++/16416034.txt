#include <iostream>
using namespace std;
int a[11][11];
int main()
{
	int n,k,r,c;
	cin>>r>>c>>n>>k;
		int x,y;
		long long ans=0;
		int s=1,t=1;
		for(int i=1;i<=n;i++){
			cin>>x>>y;
			a[x][y]=1;

		}
		int num=0;
		for(int i=1;i<=r;i++){
			for(int j=1;j<=c;j++){
				for(int w=1;w<=r;w++){
					for(int p=1;p<=c;p++){
						num=0;
						for(int x=i;x<=w;x++)
							for(int y=j;y<=p;y++)
							num+=a[x][y];
						if(num>=k)ans++;
					}
				}
				
			}
		}
		cout<<ans<<endl;
	
    return 0;
}
