#include<iostream>
using namespace std;
int r,c,n,k;
int a[11][2],ans;
int main()
{
	cin>>r>>c>>n>>k;
	for(int i=1;i<=n;i++)cin>>a[i][0]>>a[i][1];
	for(int x1=1;x1<=r;x1++)
	{
		for(int y1=1;y1<=c;y1++)
		{
			for(int x2=x1;x2<=r;x2++)
			{
				for(int y2=y1;y2<=c;y2++)
				{
					int s=0;
					for(int i=1;i<=n;i++)
					{
						if(a[i][0]>=x1 && a[i][1]>=y1 && a[i][0]<=x2 && a[i][1]<=y2)s++;
					}
					if(s>=k)ans++;
				}
			}
		}
	}
	cout<<ans;
	return 0;
}
