#include<bits/stdc++.h>
using namespace std;
int n,m,k,l,a[105][105],ans;
int main()
{
	
	cin>>n>>m>>k>>l;
	
	while(k--)
	{
		int i,j;
		cin>>i>>j;
		
		a[i][j]=1;
	}
	
	for(int i=1;i<=n;i++)
	for(int j=1;j<=m;j++)
	{
		for(int q=i;q<=n;q++)
		for(int e=j;e<=m;e++)
		{
			int re=0;
			
			
			for(int i1=i;i1<=q;i1++)
			for(int j1=j;j1<=e;j1++)
				re+=a[i1][j1];
			if(re>=l)ans++;
		}
	}
	
	cout<<ans<<endl;
}
