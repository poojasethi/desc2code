#include <iostream>


using namespace std;
int q[100][100]={0,};
int bum(int x,int y,int z,int u){
    int kol=0;
    for (int i=x;i<=z;i++)
        for (int j=y;j<=u;j++) if (q[i][j]==1)kol++;
    return kol;
}
int main(){
int r,c,n,k,t,y,u,x,time=0,g,h,z;
cin >> r >> c >> n >> k;z=k;
for (int i=0;i<n*2;i=i+2){cin >> x >> y;q[x-1][y-1]=1;}
for (int i=0;i<r;i++)
for (int j=0;j<c;j++)
    for (int g=i;g<r;g++)
for (int h=j;h<c;h++){if (bum(i,j,g,h)>=k) time++;
}
cout << time;
return 0;
}
