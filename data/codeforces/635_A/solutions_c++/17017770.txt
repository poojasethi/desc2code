#include <stdio.h>
int a[12][12]; 
int fun(int i,int j,int m,int n)
{
	int ans=0,k,z;
	for( k=i;k<=j;k++)
	for( z=m;z<=n;z++)
	{
		if(a[k][z])
		ans++;
	}
	return ans;	
}
int main()
{
	
	int r,c,n,k;
    scanf("%d %d %d %d",&r,&c,&n,&k);
	
		int x,y,ans=0;
		for(int i=0;i<n;i++)
		{
			scanf("%d %d",&x,&y);
			a[x][y]=1;
		}
		for(int i=1;i<=r;i++)
		for(int j=i;j<=r;j++)
		for(int m=1;m<=c;m++)
		for(int n=m;n<=c;n++)
		{
			if(fun(i,j,m,n)>=k)
			ans++;
		}
		printf("%d\n",ans);
		ans=0;
	
	return 0;
}
