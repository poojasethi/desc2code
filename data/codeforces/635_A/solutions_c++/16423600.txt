#include<bits/stdc++.h>
using namespace std;
int main()
{
	int r,c,n,k;
	int x[11],y[11];
	cin>>r>>c>>n>>k;
	for (int i=1;i<=n;i++)
	{
		cin>>x[i]>>y[i];
	}
	int ans=0;
	for (int i=1;i<=r;i++)
	{
		for (int j=1;j<=c;j++)
		{
			for (int l=i;l<=r;l++)
			{
				for (int e=j;e<=c;e++)
				{
					int tem=0;
					for (int o=1;o<=n;o++)
					{
						if (x[o]<=l&&x[o]>=i&&y[o]<=e&&y[o]>=j)
						{
							tem++;
						}
					}
					if (tem>=k) ans++;
				}
			}
		}
	}
	cout<<ans;
}
