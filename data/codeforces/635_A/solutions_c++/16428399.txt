#include<bits/stdc++.h>
using namespace std;
int ar[1000][1000];
int main(){
	int r,c,n,k,top=0,res=0,x,y;
	scanf("%d %d %d %d",&r,&c,&n,&k);
	for(int i=1 ;  i<=n ; i++){
		scanf("%d %d",&x,&y);
		ar[x][y]=1;
	}
	for(int i=1 ; i<=r ; i++){
		for(int j=1 ; j<=c ; j++){
			for(int q=i ; q<=r ; q++){
				for(int p=( q==i ? j : 1 ) ; p<=c ; p++){
					for(int w=i ; w<=q ; w++){
						for(int s=j ; s<=p ; s++){
							top+=ar[w][s];
						}
					}
					if(top>=k)
						res++;
					top=0;
				}
			}
		}
	}
	printf("%d\n",res);
	return 0;
}
