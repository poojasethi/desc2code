#include <iostream>

using namespace std;
int r,i,x,j,d,z,l,o,sum,nrsol,y,c,n,k,a[11][11];
int main()
{
    cin>>r>>c>>n>>k;
    for (i=1;i<=n;i++){
      cin>>x>>y;
      a[x][y]=1;
    }
    for (i=1;i<=r;i++)
    for (j=1;j<=c;j++){
      for (d=i;d<=r;d++)
          for (z=j;z<=c;z++)
          {
            sum=0;
             for (l=i;l<=d;l++)
                  for (o=j;o<=z;o++)
                   sum+=a[l][o];
            if (sum>=k) nrsol++;
          }
    }
    cout<<nrsol;
    return 0;
}
