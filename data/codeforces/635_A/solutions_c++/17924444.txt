#include <bits/stdc++.h>
using namespace std ;
int A[11][11] ;
int main()
{
    int r,c,n,k,u,v ; cin>>r>>c>>n>>k ;
    while(n--) cin>>u>>v , A[u][v]=1 ;
    int ans=0 ;
    for(int x=1;x<=r;++x) for(int x1=x;x1<=r;++x1) for(int y=1;y<=c;++y)for(int y1=y;y1<=c;++y1)
    {
        int temp= 0;
        for(int i=x;i<=x1;++i)
        for(int j=y;j<=y1;++j)  temp +=A[i][j] ;
        if(temp >=k) ++ans ;
    }
    cout<<ans<<endl ;
    return 0 ;
}
