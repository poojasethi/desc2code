#include <cstdio>
#include <iostream>
using namespace std;
int x,y,r,c,n,k,ans,f[11][11];
int Cout(int x,int y,int X,int Y)
{
	int tot=0;
	for(int i=x;i <= X;i++)
	 for(int j=y;j <= Y;j++)
	   tot+=f[i][j];
	return tot;
}
int main()
{
	cin>>r>>c>>n>>k;
	for(int i=1;i <= n;i++)
	 {
	 	cin>>x>>y;
	 	f[x][y]++;
	 }
	 for(int i=1;i <= r;i++)
	  for(int j=1;j <= c;j++)
	   for(int I=i;I <= r;I++)
	    for(int J=j;J <= c;J++)
	     if(Cout(i,j,I,J) >= k) ans++;
	cout<<ans;
}
