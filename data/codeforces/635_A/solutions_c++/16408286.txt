#include <bits/stdc++.h>

using namespace std;

bool	Map[20][20];

int main()
{
	int	n,m,t,k,i,x,y,j,ii,jj,a,b,Ans=0;

	scanf("%d%d%d%d",&n,&m,&t,&k);
	for(i=1;i<=t;++i)
	{
		scanf("%d%d",&x,&y);
		Map[x][y]=true;
	}

	for(i=1;i<=n;++i)
		for(j=1;j<=m;++j)
			for(ii=i;ii<=n;++ii)
				for(jj=j;jj<=m;++jj)
				{
					int	cnt=0;
					for(a=i;a<=ii;++a)
						for(b=j;b<=jj;++b)
							if(Map[a][b])cnt++;
					if(cnt>=k)Ans++;
				}

	printf("%d\n",Ans);
}
