#include <bits/stdc++.h>
using namespace std;
int board[15][15];

int main(){
	int x,y,n,i,j,r,c,k;
	cin>>r>>c>>n>>k;
	int ret=0;
	for(i=0;i<n;++i){
		cin>>x>>y;
		board[x][y]=1;
	}
	int a,b;
	for(y=1;y<=r;++y)
		for(x=1;x<=c;++x){
			for(j=1;j<=r;++j)
				for(i=1;i<=c;++i){
					if(j+y-1>r || i+x-1>c)continue;
					int tmp=0;
					for(a=j;a<=j+y-1;++a)
						for(b=i;b<=i+x-1;++b)
							tmp+=board[a][b];
					if(tmp>=k)++ret;
				}
		}
	cout<<ret<<endl;
}
