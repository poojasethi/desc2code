#include<iostream>
using namespace std;
const int maxn=20;
int x[maxn],y[maxn];
int main()
{
	int ans=0;
	int r,c,n,k;
	cin>>r>>c>>n>>k;
	for(int i=1;i<=n;i++) cin>>x[i]>>y[i];
	for(int i=1;i<=r;i++) for(int j=1;j<=c;j++)
	for(int p=i;p<=r;p++) for(int q=j;q<=c;q++)
	{
		int t=0;
		for(int w=1;w<=n;w++) if(x[w]>=i && x[w]<=p && y[w]>=j && y[w]<=q) t++;
		if(t>=k) ans++;
	}
	cout<<ans<<endl;
	return 0;
}
