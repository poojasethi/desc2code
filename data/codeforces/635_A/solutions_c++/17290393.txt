#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int r, c, n, k;
	cin >> r >> c >> n >> k;
	int x, y, i, ans = 0, a[11][11] = {0};
	for(i = 0; i < n; ++i) {
		cin >> x >> y;
		a[x][y] = 1;
	}
	for(i = 1; i <= r; ++i) {
		for(int j = 1; j <= c; ++j) {
			for(int l = i; l <= r; ++l) {
				for(int m = j; m <= c; ++m) {
					int s = 0;
					for(int p = i; p <= l; ++p) {
						for(int q = j; q <= m; ++q)
							s += a[p][q];
					}
					if(s >= k)
						ans++;
				}
			}
		}
	}
	cout << ans;
	return 0;
}
