#include <iostream>
using namespace std;
int ans,r,c,n,k,t,o,a[11][11],i1,j1,i2,j2,j,i,rr;
main()
{
cin>>r>>c>>n>>k;
for(i=1;i<=n;i++)
{
cin>>t>>o;
a[t][o]=1;        
}

for (i1=1;i1<=r;i1++)
for (j1=1;j1<=c;j1++)
for (i2=i1;i2<=r;i2++)
for (j2=j1;j2<=c;j2++)
{
        rr=0;
for(i=i1;i<=i2;i++)
for(j=j1;j<=j2;j++)
if(a[i][j]==1) 
rr++;
if(rr>=k)
ans++;
}
cout<<ans<<endl;
}
