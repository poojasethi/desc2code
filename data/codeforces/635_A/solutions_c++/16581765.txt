#include<iostream>
using namespace std;
int r,c,n,m,i1,j1,i2,j2,i,j,l,ans;
int a[11][11];
main(){
 cin>>r>>c>>n>>m;
 for (i=1;i<=n;i++){
  cin>>i1>>j1;
  a[i1][j1]=1;
 }

 for (i1=1;i1<=r;i1++)
  for (j1=1;j1<=c;j1++)
   for (i2=i1;i2<=r;i2++)
    for (j2=j1;j2<=c;j2++){
     l=0;
     for (i=i1;i<=i2;i++)
      for (j=j1;j<=j2;j++)
        l=l+a[i][j];

     if (l>=m) ans++;


    }

 cout<<ans<<endl;

 //system("pause");
}
