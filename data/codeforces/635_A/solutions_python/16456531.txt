#!/usr/bin/env python
# input
r, c, n, k = map(int, raw_input().split())
dp = [[[[0 for i in range(c)] for j in range(r)] for t in range(c)] for l in range(r)]

for i in range(n):
    xi, yi = map(int, raw_input().split())
    dp[xi - 1][yi - 1][0][0] = 1

ans = 0
for roffset in range(r):
    for coffset in range(c):
        for rx in range(r - roffset):
            for cx in range(c - coffset):
                if roffset > 0:
                    dp[rx][cx][roffset][coffset] = dp[rx][cx][roffset - 1][coffset] + dp[rx + roffset][cx][0][coffset]
                elif coffset > 0:
                    dp[rx][cx][roffset][coffset] = dp[rx][cx][roffset][coffset - 1] + dp[rx][cx + coffset][roffset][0]

                if dp[rx][cx][roffset][coffset] >= k:
                    ans += 1
print ans
