#!/usr/bin/python
# -*- coding:utf-8 -*-
def judge(x1,y1,x2,y2,k,position):
    count=0
    for i in range(x1,x2+1):
        for j in range(y1,y2+1):
            count=count+position[i][j]
    return count>=k
r,c,n,k=map(int,raw_input().split(' '))
position=[([0]*c)for i in range(r)]
#viola=[([0]*2)for i in range(n)]
for i in range(n):
   x,y=map(int,raw_input().split(' '))
   position[x-1][y-1]=1
result=0
for i in range(r):
    for j in range(c):
        for ii in range(r):
            for jj in range(c):
                result=result+int(judge(i,j,ii,jj,k,position))
print result
