r,c,n,m = map(int, raw_input().strip().split())
x = []
for i in xrange(r):
	x.append([0]*c)
for i in xrange(n):
	a,b = map(int, raw_input().strip().split())
	x[a-1][b-1] = 1
count = 0
for i in range(r):
	for j in range(c):
		#i,j is the starting point
		# generate all the rectangles possible and find sum and if greater than k, add 1.
		addition = 0
		for k in range(r,i,-1):
			for l in range(c,j,-1):
				#addition is the sum of the rectangle defined by i,j,k,l
				addition = sum(x[a][b] for a in range(i,k) for b in range(j,l))
				if(addition >= m):
					count += 1
print count
