r, c, n, k = map(int, raw_input().split())
p = [[0]*c for i in range(r)]
for i in range(n):
    x, y = map(int, raw_input().split())
    p[x-1][y-1] = 1
s = 0
for i in range(r):
    for j in range(i, r):
        for x in range(c):
            for y in range(x, c):
                s += sum(map(lambda k: sum(k[x:y+1]), p[i:j+1])) >= k
print s
