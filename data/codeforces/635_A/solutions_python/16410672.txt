r, c, n, k = map(int, raw_input().split())
grid = [['.' for i in range(c+1)] for j in range(r+1)]
for i in range(n):
    x, y = map(int, raw_input().split())
    grid[x][y] = '*'
ans = 0
for ii in range(1, r+1):
    for jj in range(1, c+1):
        for iii in range(1, ii+1):
            for jjj in range(1, jj+1):
                g = [grid[i][j] for i in range(iii, ii+1) for j in range(jjj, jj+1)]
                x = g.count('*')
                if x >= k:
                    ans += 1
print ans
