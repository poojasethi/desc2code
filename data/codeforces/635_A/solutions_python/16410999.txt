r,c,n,k = map(int,raw_input().split())

l = []
for p in xrange(12):
	l.append([0] * 12)

ans = 0

for w in xrange(n):
	x = map(int,raw_input().split())
	l[x[0]][x[1]] = 1


for srow in xrange(1, r + 1):
	for scol in xrange(1 , c + 1):

		for erow in xrange(srow, r + 1):
			for ecol in xrange(scol , c + 1):

				cont = 0
				for i in xrange(srow,erow + 1):
					for j in xrange(scol,ecol + 1):
						if l[i][j] == 1:
							cont += 1
				if cont >= k:
					ans += 1

print ans
