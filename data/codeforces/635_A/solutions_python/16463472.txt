#from __future__ import print_function
r, c, n, ks = map(int, raw_input().split())
b = [[0 for i in range(c)] for i in range(r)]
for row in xrange(n):
	x, y = map(int, raw_input().split())
	x, y = x - 1, y - 1
	b[x][y] = 1

ret = 0
for i in xrange(r):
	for j in xrange(c):
		for k in xrange(i, r):
			for l in xrange(j, c):
				tot = 0
				for x in xrange(i, k+1):
					for y in xrange (j, l + 1):
						tot += b[x][y]
				ret += tot >= ks				
print ret
