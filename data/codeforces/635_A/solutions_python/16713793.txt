# coding:utf-8

r,c,n,k = map(int, raw_input().split())
m=[[0]*c for i in xrange(r)]

for _ in xrange(n):
	x,y = map(int, raw_input().split())
	m[x-1][y-1] = 1

def getSum(x1, y1, x2, y2):
	return sum([sum(m[x][y1:y2+1]) for x in xrange(x1, x2+1)])

res = 0
for x1 in xrange(r):
	for y1 in xrange(c):
		for x2 in xrange(x1,r):
			for y2 in xrange(y1,c):
				if getSum(x1, y1, x2, y2) >= k:
					res += 1
print res
