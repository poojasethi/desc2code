#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
const int N = 3050;
const int mod=(int)1e9+7;
char s[N][N];
int n,m;
inline void up(int &a,int b)
{
    a=(a+b)%mod;
}
void calc(int x,int y,int opt[][N])
{
    memset(opt,0,N*N*sizeof(int));
    if(s[x][y]=='.') opt[x][y]=1;
    for(int i=x;i<=n;++i)
        for(int j=y;j<=m;++j)
        {
            if(s[i+1][j]=='.') up(opt[i+1][j],opt[i][j]);
            if(s[i][j+1]=='.') up(opt[i][j+1],opt[i][j]);
        }
}
int opt[2][N][N];
int main()
{
    while(scanf("%d%d",&n,&m)!=-1)
    {
        for(int i=1;i<=n;++i)
            scanf("%s",s[i]+1);
        calc(1,2,opt[0]);
        calc(2,1,opt[1]);
        int ans=(1ll*opt[0][n-1][m]*opt[1][n][m-1]-1ll*opt[0][n][m-1]*opt[1][n-1][m])%mod;
        if(ans<0) ans+=mod;
        printf("%d\n",ans);
    }
    return 0;
}
