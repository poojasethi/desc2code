#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=3010;
const int mod=1000000007;
typedef long long LL;
int n,m;
char s[maxn][maxn];
LL dp[2][maxn][maxn];
void doit(int k,int x,int y){
	dp[k][x][y]=1;
	for(int i=x;i<=m;i++)
		for(int j=y;j<=n;j++){
			if(s[i][j]=='#') dp[k][i][j]=0;
			dp[k][i][j]%=mod;
			dp[k][i+1][j]+=dp[k][i][j];dp[k][i][j+1]+=dp[k][i][j];
		}
}
int main(){
	scanf("%d%d",&m,&n);
	for(int i=1;i<=m;i++)
		scanf("%s",s[i]+1);
	doit(0,2,1);doit(1,1,2);
	LL ans=(dp[0][m][n-1]*dp[1][m-1][n]%mod-dp[0][m-1][n]*dp[1][m][n-1]%mod+mod)%mod;
	printf("%lld\n",ans);
	return 0;
}
