#include <iostream>
#include <cstdio>
using namespace std;
const int N = 3003, mod = 1e9 + 7;

int n, m, d1[N][N], d2[N][N];
char s[N][N];

int main() {
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++)
		scanf("%s", s[i] + 1);
	d1[1][2] = s[1][2] == '.', d2[2][1] = s[2][1] == '.';
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= m; j++)
			if (s[i][j] == '.') {
				d1[i][j] = (d1[i][j] + d1[i - 1][j] + d1[i][j - 1]) % mod;
				d2[i][j] = (d2[i][j] + d2[i - 1][j] + d2[i][j - 1]) % mod;
			}
//	cerr << d1[n - 1][m] << ' ' << d2[n][m - 1] << ' ' << d1[n][m - 1] << ' ' << d2[n - 1][m] << '\n';
	printf("%d\n", ((1ll * d1[n - 1][m] * d2[n][m - 1] - 1ll * d1[n][m - 1] * d2[n - 1][m]) % mod + mod) % mod);
	return 0;
}
