#include<cstdio>
#include<cstdlib>
#define N 3005
#define MOD 1000000007
#define rep(i,n) for(int i=1;i<=n;i++)
int n,m,f[2][N][N];
char g[N][N];
void init(){
    scanf("%d%d",&n,&m);
    rep(i,n) scanf("%s",g[i]+1);
}
void plus(int &x,int &y){
    x+=y;
    if(x>=MOD) x-=MOD;
}
void work(int s,int x,int y){
    f[s][x][y]=1;
    for (int i=x;i<=n;i++)
        for (int j=y;j<=m;j++)
        if(g[i][j]=='.'){
            if(g[i+1][j]=='.') plus(f[s][i+1][j],f[s][i][j]);
            if(g[i][j+1]=='.') plus(f[s][i][j+1],f[s][i][j]);
        }
}
int main(){
    init();
    work(0,2,1);
    work(1,1,2);
    printf("%d\n",((1LL*f[0][n][m-1]*f[1][n-1][m]-1LL*f[0][n-1][m]*f[1][n][m-1])%MOD+MOD)%MOD);
}
