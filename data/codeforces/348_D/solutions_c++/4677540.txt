#include<cstdio>
#define p 1000000007
int i,j,n,m,ans,f[3001][3001],h[3001][3001];
char g[3002][3002];
int main()
{
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;++i)scanf("%s",g[i]+1);
    for(f[1][1]=i=1;i<=n;++i)
        for(j=2;j<=m;++j)
            if(g[i][j]=='.'){
                f[i][j]=f[i-1][j]+f[i][j-1];
                if(f[i][j]>=p)f[i][j]-=p;
            }
    for(h[1][1]=1,i=2;i<=n;++i)
        for(j=1;j<=m;++j)
            if(g[i][j]=='.'){
                h[i][j]=h[i-1][j]+h[i][j-1];
                if(h[i][j]>=p)h[i][j]-=p;
            }
    ans=(((long long)f[n-1
    ][m]*h[n][m-1]-(long long)f[n][m-1]*h[n-1][m])%p+p)%p;
    printf("%d",ans);
    return 0;
}
