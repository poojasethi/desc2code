#include <cstdio>
#include <string.h>
using namespace std;
typedef long long LL;
const int INF=1000000007;
const int NMax=3010;
int N,M;
char A[NMax][NMax];
int dp[NMax][NMax];
int main()
{
	scanf("%d%d",&N,&M);
	for(int i=0;i<N;i++) scanf("%s",A[i]);
	dp[0][0]=1;
	for(int i=0;i<N;i++) {
		for(int j=1;j<M;j++) if(A[i][j]=='.'){
			if(i>0) dp[i][j]=dp[i-1][j];
			dp[i][j]=(dp[i][j]+dp[i][j-1])%INF;
		}
	}
	LL ret1=dp[N-1][M-2],ret3=dp[N-2][M-1];
	memset(dp,0,sizeof(dp));
	dp[0][0]=1;
	for(int i=1;i<N;i++) {
		for(int j=0;j<M;j++) if(A[i][j]=='.'){
			dp[i][j]=dp[i-1][j];
			if(j>0)dp[i][j]=(dp[i][j]+dp[i][j-1])%INF;
		}
	}
	LL ret2=dp[N-2][M-1],ret4=dp[N-1][M-2];
	printf("%I64d\n",(ret3*ret4%INF-ret1*ret2%INF+INF)%INF);
	getchar();getchar();
	return 0;
}
