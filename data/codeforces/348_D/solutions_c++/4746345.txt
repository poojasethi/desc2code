#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 3010
#define mo 1000000007
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int f[N][N];
char p[N][N];
inline void M(int &a,int b) {
	a=(a+b)%mo;
}
inline int64 cal(int x,int y,int xx,int yy) {
	int i,j;
	memset(f,0,sizeof(f));
	f[x][y]=p[x][y]!='#';
	For(i,x,xx)For(j,y,yy) if (f[i][j]) {
		if (i<xx&&p[i+1][j]!='#') M(f[i+1][j],f[i][j]);
		if (j<yy&&p[i][j+1]!='#') M(f[i][j+1],f[i][j]);
	}
	return f[xx][yy];
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) scanf("%s",p[i]+1);
	int64 A=(cal(1,2,n-1,m)*cal(2,1,n,m-1)-cal(1,2,n,m-1)*cal(2,1,n-1,m))%mo;
	printf("%I64d\n",(A+mo)%mo);
	//for(;;);
	return 0;
}
