#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

const int MOD = 1000000007;

int n,m;
char grid[3010][3010];
int f[3010][3010];

void calc(int x,int y){
	memset(f,0,sizeof(f));
	if(grid[x][y]=='#') return;f[x][y]=1;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
			if(grid[i][j]!='#')
				(j<m-1&&grid[i][j+1]!='#'?(f[i][j+1]+=f[i][j])%=MOD:0),
				(i<n-1&&grid[i+1][j]!='#'?(f[i+1][j]+=f[i][j])%=MOD:0);
}

int main()
{
	cin >> n >> m;
	for(int i=0;i<n;i++) cin >> grid[i];
	int a,b,c,d;
	calc(0,1);a=f[n-2][m-1],b=f[n-1][m-2];
	calc(1,0);c=f[n-1][m-2],d=f[n-2][m-1];
	cout << ((1LL*a*c-1LL*b*d)%MOD+MOD)%MOD << endl;
	return 0;
}
