#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#define N 3005
#define mod 1000000007
using namespace std;
typedef long long LL;
LL a,b,c,d,e,f,g,h,i,j,k;
LL F[N][N];
char w[N][N];
LL an;

LL DP(LL x,LL y,LL xx,LL yy)
{
    LL ii,jj,kk;
    memset(F,0,sizeof(F));
    if(w[x][y]!='#')F[x][y]=1;
    for(ii=x;ii<=xx;ii++)
        for(jj=y;jj<=yy;jj++)
        if((ii!=x || jj!=y) && w[ii][jj]!='#')
        {
            F[ii][jj]=F[ii-1][jj]+F[ii][jj-1];
            F[ii][jj]%=mod;
        }
    return F[xx][yy];
}

int main()
{
    cin>>a>>b;
    for(i=1;i<=a;i++) scanf("%s",w[i]+1);
    an=DP(1,2,a-1,b)*DP(2,1,a,b-1);an%=mod;
    an-=DP(1,2,a,b-1)*DP(2,1,a-1,b);
    if(an<0) an+=((-an)/mod+1)*mod;an%=mod;
    cout<< an <<endl;
    return 0;
}
