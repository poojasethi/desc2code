#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

#define inc(a,b) {a+=b;if (a>=mo) a-=mo;}

const int maxn=3010;
const int mo=1000000007;

int n,m,f[maxn][maxn];

char s[maxn][maxn];

int solve(int x1,int y1,int x2,int y2)
{
	memset(f,0,sizeof(f));
	f[x1][y1]=1;
	for (int a=x1;a<=x2;a++)
		for (int b=y1;b<=y2;b++)
			if (s[a][b]=='.')
			{
				inc(f[a+1][b],f[a][b]);
				inc(f[a][b+1],f[a][b]);
			}
			else f[a][b]=0;
	return f[x2][y2];
}

int main()
{
	scanf("%d%d",&n,&m);
	for (int a=1;a<=n;a++)
		scanf("%s",s[a]+1);
	printf("%d\n",(int)((((long long)solve(1,2,n-1,m)*solve(2,1,n,m-1)-(long long)solve(1,2,n,m-1)*solve(2,1,n-1,m))%mo+mo)%mo));

	return 0;
}
