#include<iostream>
#include<cstdio>
using namespace std;
#define mod 1000000007
typedef int ll;
ll n,m,a[3001][3001];
long long d1[3001][3001],d2[3001][3001];
char c[3001];
int main()
{
    cin>>n>>m;
    for(ll i=1;i<=n;i++)
    {
        scanf("%s",c);
        for(ll j=1;j<=m;j++)
            a[i][j]=(c[j-1]=='#');
    }
    d1[2][1]=(a[2][1]==0);
    d2[1][2]=(a[1][2]==0);
    for(ll i=1;i<=n;i++)
        for(ll j=1;j<=m;j++)
            if(a[i][j]==0)
            {
                d1[i][j]+=d1[i-1][j]+d1[i][j-1];
                if(d1[i][j]>=mod)
                    d1[i][j]-=mod;
                d2[i][j]+=d2[i-1][j]+d2[i][j-1];
                if(d2[i][j]>=mod)
                    d2[i][j]-=mod;
            }
    cout<<((d1[n][m-1]*d2[n-1][m])%mod-(d1[n-1][m]*d2[n][m-1])%mod+mod)%mod;
}
