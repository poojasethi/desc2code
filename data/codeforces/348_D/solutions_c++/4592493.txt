#include <bits/stdc++.h>
#define fr(i,a,b) for(int i=a;i<b;i++)
#define rep(i,b) fr(i,0,b)
using namespace std;
typedef long long ll;
#define MOD 1000000007LL
#define N 3010
char g[N][N];
ll dp[2][N][N];

int main(){
    int n, m;
    scanf("%d %d",&n,&m);
    
    fr(i,1,n+1) scanf("%s",g[i]+1);
    
    if(g[1][2]=='.') dp[0][1][2] = 1;
    if(g[2][1]=='.') dp[1][2][1] = 1;
    
    fr(i,1,n+1) fr(j,1,m+1){
        if(g[i][j]=='.'){
            rep(z,2) dp[z][i][j] = (dp[z][i][j] + dp[z][i-1][j] + dp[z][i][j-1])%MOD;
        }
    }
    printf("%lld\n",(((dp[0][n-1][m]*dp[1][n][m-1] - dp[0][n][m-1]*dp[1][n-1][m])%MOD)+MOD)%MOD);
    return 0;
}
