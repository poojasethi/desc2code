#include <bits/stdc++.h>
#define y1 stupid_cmath_y1
using namespace std;
const int maxn=3333;
const long long mod=1e9+7;
int n,m,i,j;
char s[maxn][maxn];
long long F[maxn][maxn],w1,w2,an;
long long Ways(int x1,int y1,int x2,int y2){
	memset(F,0,sizeof F);
	F[x1][y1]=(s[x1][y1]=='.'?1:0);
	for(i=1;i<=n;++i)for(j=1;j<=m;++j){
		if(i==x1&&j==y1)continue;
		if(s[i][j]=='#')F[i][j]=0;else F[i][j]=(F[i-1][j]+F[i][j-1])%mod;
	}
	return(F[x2][y2]%mod);
}
int main(){
	//freopen("input.txt","r",stdin);
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)scanf("%s",s[i]+1);
	w1=(Ways(1,2,n-1,m)%mod*Ways(2,1,n,m-1)%mod)%mod;
	w2=(Ways(1,2,n,m-1)%mod*Ways(2,1,n-1,m)%mod)%mod;
	an=(w1-w2+mod+mod)%mod;
	printf("%lld\n",an);
	return 0;
}
