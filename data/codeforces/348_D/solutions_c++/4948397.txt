#include <cstdio>
#include <memory.h>
using namespace std;

#define P 1000000007
int n, m, i, j, ans, f[3100][3100];
char a[3100][3100];

int calc(int x1, int y1, int x2, int y2) {
	memset(f, 0, sizeof(f));
	if (a[x1][y1] == '#')	return 0;
	f[x1][y1] = 1;
	for (int i = x1; i <= x2; i++)
		for (int j = y1; j <= y2; j++)
			if (a[i][j] == '.' && (i != x1 || j != y1))	f[i][j] = (f[i - 1][j] + f[i][j - 1]) % P;
	return f[x2][y2];
}

int main() {
	scanf("%d%d", &n, &m);
	for (i = 1; i <= n; i++)
		for (j = 1; j <= m; j++)	for (a[i][j] = getchar(); a[i][j] <= 32; a[i][j] = getchar());
	ans = (1LL * calc(1, 2, n - 1, m) * calc(2, 1, n, m - 1) - 1LL * calc(1, 2, n, m - 1) * calc(2, 1, n - 1, m)) % P;
	printf("%d\n", (ans + P) % P);
}
