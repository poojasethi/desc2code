#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define dhh1995 long long
using namespace std;
const int mo=1000000007;
int f[2][3100][3100],n,m,i;
char c[3100][3100];
void plus(int &x,int y){
	x+=y;
	if(x>=mo)x-=mo;
}
void work(int s,int x,int y){
	if(c[x][y]=='#')return;
	f[s][x][y]=1;
	for(int i=1;i<=n;++i)
		for(int j=1;j<=m;++j){
			if(c[i+1][j]!='#')plus(f[s][i+1][j],f[s][i][j]);
			if(c[i][j+1]!='#')plus(f[s][i][j+1],f[s][i][j]);
		}
}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)scanf("%s",c[i]+1);
	work(0,2,1);
	work(1,1,2);
	printf("%d\n",(((dhh1995)f[0][n][m-1]*f[1][n-1][m]-(dhh1995)f[0][n-1][m]*f[1][n][m-1])%mo+mo)%mo);
}
