#include<cstdio>
#include<cstring>
#include<string>
#include<map>
#include<iostream>
#include<cmath>
using namespace std;
#define M 3007
#define mod 1000000007
#define ll long long
char str[M][M];
ll dp1[M][M], dp2[M][M];

int n, m;

void solve( ll dp[][M], int x, int y )
{
	if( str[x][y] == '.' )
		dp[x][y] = 1;
	for( int i = x; i <= n; ++i )
		for( int j = y; j <= m; ++j ){
			if( i < n && str[i+1][j] == '.' ) dp[i+1][j] = ( dp[i+1][j] + dp[i][j] ) % mod;
			if( j < m && str[i][j+1] == '.' ) dp[i][j+1] = ( dp[i][j+1] + dp[i][j] ) % mod;
		}
}

int main()
{
	scanf( "%d%d", &n, &m );
	for( int i = 1; i <= n; ++i ) scanf( "%s", &str[i][1] );
	solve( dp1, 1, 2 );
	solve( dp2, 2, 1 );
	ll a = dp1[n-1][m], b = dp2[n][m-1], c = dp1[n][m-1], d = dp2[n-1][m];
	ll ans = ( a * b - c * d ) % mod;
	if( ans < 0 ) ans += mod;
	cout<<ans<<endl;
}
