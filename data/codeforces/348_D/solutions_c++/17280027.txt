#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
char f[3005][3005];
int n,m;
int dp[3005][3005];
void go(){
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            if(f[i][j] == '#' || dp[i][j]) continue;
            dp[i][j] = (i?dp[i-1][j]:0)+(j?dp[i][j-1]:0);
            while(dp[i][j]>=mod) dp[i][j]-=mod;
        }
    }
}
int main(){
    cin >> n >> m;
    for(int i=0;i<n;i++) scanf("%s",&f[i]);
    long long p,q,r,s;
    if(f[0][1]=='#'||f[1][0]=='#'||f[n-1][m-2]=='#'||f[n-2][m-1]=='#'){
        puts("0"); return 0;
    }
    memset(dp,0,sizeof(dp)); dp[0][1] = 1;
    go();
    p = dp[n-2][m-1]; q = dp[n-1][m-2];
    memset(dp,0,sizeof(dp)); dp[1][0] = 1;
    go();
    r = dp[n-2][m-1]; s = dp[n-1][m-2];
    cout << ((p*s-r*q)%mod+mod)%mod << endl;
}
