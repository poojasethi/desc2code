#include <stdio.h>
#include <string.h>
int N,M,answer;
long long mod = 1000000007;
char a[3005][3005];
long long dp[2][3005][3005];
int gogogo(int k,int x,int y){
if (a[x][y] == '.') dp[k][x][y] = 1;
for (int i = x; i <= N; i++)
 for (int j = y; j <= M; j++)
	if (a[i][j] == '.' && (i!=x || j!=y))
	{
	dp[k][i][j] = (dp[k][i-1][j] + dp[k][i][j-1]) % mod;
	}
return 0;
}

int main()
{
while(scanf("%d%d", &N, &M)!=EOF){
for (int i = 1; i <= N; i++)
scanf("%s", a[i] + 1);
gogogo(0,1,2);gogogo(1,2,1);
answer = ( dp[0][N-1][M]*dp[1][N][M-1]%mod - 
dp[0][N][M-1]*dp[1][N-1][M]%mod + mod ) % mod;
printf("%d\n",answer);
}
return 0;
}
