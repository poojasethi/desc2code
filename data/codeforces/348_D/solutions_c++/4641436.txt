#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
const __int64 mod=1e9+7;
const int N=5000;
int f[2][N][N];
char ch[N][N];
int n,m;
void work(int k,int x,int y){
    if(ch[x][y]=='.') f[k][x][y]=1;
    for(int i=x;i<=n;i++)
        for(int j=y;j<=m;j++) if(ch[i][j]=='.'&&(i!=x||j!=y)) f[k][i][j]=(f[k][i-1][j]+f[k][i][j-1])%mod;
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++) scanf("%s",ch[i]+1);
    work(0,1,2);
    work(1,2,1);
    //printf("%d %d %d %d\n",f[0][n][m-1],f[0][n-1][m],f[1][n-1][m],f[1][n][m-1]);
    printf("%d\n",(1ll*f[0][n-1][m]*f[1][n][m-1]%mod-1ll*f[0][n][m-1]*f[1][n-1][m]%mod+mod)%mod);
}
