#include<cstdio>
#define P 1000000007
const int N=3001;
bool a[N][N];
int i,j,n,m,f[N][N];
int MOD(int a){return a<P?a:a-P;}
int cal(int x1,int y1,int x2,int y2)
{
	if(x1>x2||y1>y2)return 0;
	for(i=x1;i<=x2;i++)
		for(j=y1;j<=y2;j++)
		{
			if(f[i][j]=0,a[i][j])continue;
			if(i==x1&&j==y1)f[i][j]=1;
			if(i>x1)f[i][j]=MOD(f[i][j]+f[i-1][j]);
			if(j>y1)f[i][j]=MOD(f[i][j]+f[i][j-1]);
		}
	return f[x2][y2];
}
int main()
{
	scanf("%d%d",&n,&m);
	for(i=1;getchar(),i<=n;i++)
		for(j=1;j<=m;++j)a[i][j]=getchar()!=46;
	i=(1ll*cal(1,2,n-1,m)*cal(2,1,n,m-1)-1ll*cal(1,2,n,m-1)*cal(2,1,n-1,m))%P;
	printf("%d",i<0?i+P:i);
}
