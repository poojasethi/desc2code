#include <cstdio>
using namespace std;

#define N	3005
#define I64	long long
#define P	1000000007
#define For(i,a,b)	for(int i=a;i<=b;i++)

int		n,m,ans,f[N][N],g[N][N];
char	b[N][N];

int		main(){
		scanf("%d %d ", &n, &m);
		For(i, 1, n) scanf("%s",b[i]+1);
	
		f[0][2] = g[2][0] = 1;
		For(i, 1, n) For(j, 1, m) if(b[i][j] == '.'){
			f[i][j] = f[i - 1][j] + f[i][j - 1];
			g[i][j] = g[i - 1][j] + g[i][j - 1];
			if	(f[i][j] >= P) f[i][j] -= P;
			if	(g[i][j] >= P) g[i][j] -= P;
		}

		ans += (I64) g[n][m - 1] * f[n - 1][m] % P;
		ans -= (I64) f[n][m - 1] * g[n - 1][m] % P;
		if(ans >= P) ans -= P;
		if(ans < 0) ans += P;
		printf("%d\n", ans);
}
