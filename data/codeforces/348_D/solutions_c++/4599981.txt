#include <iostream>
#include <string>

using namespace std;

typedef long long int ll;

ll modulo=1000000007;

int n,m;
string v[3000];

ll memcaminos[3000][3000][2];

ll caminos(int i,int j,int d)
{
  if (d==0 and i>=n-1) return 0;
  if (d==1 and i>=n) return 0;
  if (d==0 and j>=m) return 0;
  if (d==1 and j>=m-1) return 0;
  if (v[i][j]!='.') return 0;
  if (d==0 and i==n-2 and j==m-1) return 1;
  if (d==1 and i==n-1 and j==m-2) return 1;
  ll &mem=memcaminos[i][j][d];
  if (mem!=-1) return mem;
  mem=(caminos(i+1,j,d)+caminos(i,j+1,d))%modulo;
  return mem;
}

int main()
{
  ios::sync_with_stdio(false);
  cin>>n>>m;
  for (int i=0;i<n;i++)
    cin>>v[i];
  for (int i=0;i<n;i++)
    for (int j=0;j<m;j++)
      for (int d=0;d<2;d++)
	memcaminos[i][j][d]=-1;
  cout<<((caminos(0,1,0)*caminos(1,0,1)-caminos(0,1,1)*caminos(1,0,0))%modulo+modulo)%modulo<<endl;
}
