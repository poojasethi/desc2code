#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int p=1000000007;
int f[3200][3200];
char c[3200][3200];
int i,j,m,n,s1,s2;

int ask(int x1,int y1,int x2,int y2)
{
	memset(f,0,sizeof(f));
	if (c[x1][y1]!='#')
		f[x1][y1]=1;
	for (i=x1;i<=n;i++)
		for (j=y1;j<=m;j++)
			if (((i^x1) || (j^y1)) && (c[i][j]!='#'))
				f[i][j]=(f[i-1][j]+f[i][j-1])%p;
	return f[x2][y2];
}

int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
		scanf("%s",c[i]+1);
	s1=((long long) ask(2,1,n,m-1)*ask(1,2,n-1,m))%p;
	s2=((long long) ask(1,2,n,m-1)*ask(2,1,n-1,m))%p;
	printf("%d",(s1-s2+p)%p);
	return 0;
}
