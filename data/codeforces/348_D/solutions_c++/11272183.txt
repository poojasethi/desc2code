#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define M 3030
#define MOD 1000000007
using namespace std;
int n,m;
char map[M][M];
long long Calculate(int x1,int y1,int x2,int y2)
{
    static int f[M][M];
    int i,j;
    if(map[x1][y1]=='#')
        return 0;
    memset(f,0,sizeof f);
    f[x1][y1]=1;
    for(i=x1;i<=x2;i++)
        for(j=y1;j<=y2;j++)
        {
            if(map[i][j]=='#')
                continue;
            (f[i][j]+=f[i-1][j]+f[i][j-1])%=MOD;
        }
    return f[x2][y2];
}
int main()
{
    int i;
    cin>>n>>m;
    for(i=1;i<=n;i++)
        scanf("%s",map[i]+1);
    long long ans=  Calculate(2,1,n,m-1)*Calculate(1,2,n-1,m)-
                    Calculate(2,1,n-1,m)*Calculate(1,2,n,m-1);
    cout<<(ans%MOD+MOD)%MOD<<endl;
    return 0;
}
