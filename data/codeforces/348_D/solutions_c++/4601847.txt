#include <bits/stdc++.h>
using namespace std;

const int N = 3001;
const int mod = 1e9 + 7;
string s[N];
int cnt[N][N];

int f(int x1, int y1, int x2, int y2) {
	memset(cnt, 0, sizeof cnt);
	cnt[x1][y1] = (s[x1-1][y1-1] == '.');
	for (int x = x1; x <= x2; x++)
		for (int y = y1; y <= y2; y++) {
			if (x == x1 && y == y1) continue;
			if (s[x-1][y-1] == '#') continue;
			cnt[x][y] = (cnt[x-1][y] + cnt[x][y-1]) % mod;
		}
	return cnt[x2][y2];
}

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> s[i];
	int det = ((1ll * f(1,2,n-1,m) * f(2,1,n,m-1) - 1ll * f(1,2,n,m-1) * f(2,1,n-1,m)) % mod + mod) % mod;
	cout << det;
}
