#include<cstdio>
#include<cstring>
const int M=1000000007;
int n,m,ans;
char s[3005][3005];
int f[3005][3005],g[3005][3005];
long long A,B,C,D;

void up(int &a,const int &b)
{a+=b;a%=M;}

long long ABS(const long long &x)
{return x<0?-x:x;} 

void dp(int f[3005][3005])
{
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            if(s[i][j]=='.')
                up(f[i][j],f[i-1][j]+f[i][j-1]);
}

int main()
{
    scanf("%d%d\n",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%s",s[i]+1);
    f[1][2]=(s[1][2]=='.');
    dp(f);
    g[2][1]=(s[2][1]=='.');
    dp(g);
    A=f[n-1][m],B=f[n][m-1];
    C=g[n-1][m],D=g[n][m-1];
    ans=(A*D)%M-(B*C)%M;
    if(ans<0)ans+=M;
    printf("%d\n",ans);
    return 0;
}
