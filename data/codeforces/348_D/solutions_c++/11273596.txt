#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#define N 3001
#define MOD 1000000007
using namespace std;
typedef long long ll;
int n,m;
ll f[N][N],a,b,c,d;
char map[N][N];
ll dp(int x1,int y1,int x2,int y2)
{
	memset(f,0,sizeof(f));
	int i,j;
	f[x1][y1]=1;
	for(i=x1;i<=x2;i++)
		for(j=y1;j<=y2;j++)
			if(map[i][j]=='.')
			{
				if(i>x1)f[i][j]=(f[i][j]+f[i-1][j])%MOD;
				if(j>y1)f[i][j]=(f[i][j]+f[i][j-1])%MOD;
			}
	return f[x2][y2];
}
int main()
{
	int i,j;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		scanf("%s",map[i]+1);
	if(map[1][2]=='.'&&map[n-1][m]=='.')a=dp(1,2,n-1,m);
	if(map[2][1]=='.'&&map[n][m-1]=='.')d=dp(2,1,n,m-1);
	if(map[1][2]=='.'&&map[n][m-1]=='.')b=dp(1,2,n,m-1);
	if(map[2][1]=='.'&&map[n-1][m]=='.')c=dp(2,1,n-1,m);
	printf("%I64d\n",((a*d-b*c)%MOD+MOD)%MOD);
	return 0;
}
