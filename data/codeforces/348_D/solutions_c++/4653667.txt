#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef long long ll;
const int maxn=3010;
const int mo=1e9+7;
ll f[maxn][maxn];
int n,m;
char st[maxn][maxn];
ll get(int sx,int sy,int ex,int ey){
    memset(f,0,sizeof(f));
    f[sx][sy]=st[sx][sy]=='.';
    for (int i=sx;i<=ex;i++)
        for (int j=sy;j<=ey;j++){
            if (st[i][j]=='#'||(i==sx&&j==sy)) continue;
            f[i][j]=f[i-1][j]+f[i][j-1];
            if (f[i][j]>=mo) f[i][j]-=mo;
        }
    return f[ex][ey];
}
int main(){
    //freopen("input.txt","r",stdin);
    cin>>n>>m;
    for (int i=1;i<=n;i++) scanf("%s",st[i]+1);
    ll A=get(1,2,n-1,m);
    ll B=get(2,1,n,m-1);
    ll C=get(1,2,n,m-1);
    ll D=get(2,1,n-1,m);
    ll ans=A*B-C*D;
    ans=(ans%mo+mo)%mo;
    cout<<ans<<endl;
}
