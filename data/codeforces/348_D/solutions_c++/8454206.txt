#include<cstdio>
#include<cstring>
using namespace std;
const int P=1000000007;
char a[3005][3005];
int f[3005][3005],n,m,i;
inline int work(int sx,int sy,int tx,int ty)
{
  memset(f,0,sizeof(f));
  f[sx][sy]=(a[sx-1][sy]!='#'&&a[sx][sy-1]!='#'&&a[sx][sy]!='#');
  for (int i=sx;i<=tx;i++)
    for (int j=sy;j<=ty;j++)
    {
      if (i==sx&&j==sy||a[i][j]=='#') continue;
      f[i][j]=(f[i-1][j]+f[i][j-1])%P;
    }
  return f[tx][ty];
}
int main()
{
  scanf("%d%d",&n,&m);
  for (i=1;i<=n;i++)
    scanf("%s",a[i]+1);
  int tmp1=work(1,2,n-1,m)*1ll*work(2,1,n,m-1)%P;
  int tmp2=work(1,2,n,m-1)*1ll*work(2,1,n-1,m)%P;
  printf("%d",(tmp1+P-tmp2)%P);
  return 0;
}
