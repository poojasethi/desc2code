#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<cstdlib>
using namespace std;
const int N=3010;
const long long pp=1000000007;
int n,m,f[N][N];
char str[N][N];
#define inc(x,y) {x+=y; if(x>=pp)x-=pp; }
int work(int x1,int y1,int x2,int y2){
    memset(f,0,sizeof f);
    f[x1][y1]=1;
    for(int i=x1;i<=x2;i++)
      for(int j=y1;j<=y2;j++)
        if(str[i][j]=='.'){
                           inc(f[i+1][j],f[i][j]);
                           inc(f[i][j+1],f[i][j]);
                           }
                           else f[i][j]=0; 
    return f[x2][y2];
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)scanf("%s",str[i]+1);
    cout<<(((long long)work(1,2,n-1,m)*(long long)work(2,1,n,m-1)-(long long)work(1,2,n,m-1)*(long long)work(2,1,n-1,m))%pp+pp)%pp<<endl;
    return 0;
}
