#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <vector>
#include <algorithm>
#include <math.h>
#include <string.h>

using namespace std;

int h,m,n,ans;
int q[500000];

void add(int id, int hash){
    int i = 0;
    while (q[(hash + i*m)%h]!=-1)
        i++;
    ans+=i;
    q[(hash + i*m)%h] = id;
}

void del(int id){
    for (int i=0; i<h; i++)
        if (q[i]==id)
            q[i] = -1;
}

int main(void){
    //freopen("input.txt","r",stdin);//freopen("output.txt","w",stdout);
    cin >> h >> m >> n;
    for (int i=0; i<h; i++)
        q[i] = -1;
    for (int i; i<n; i++){
        char c;
        int w,id;
        cin >> c;
        if (c=='+'){
            cin >> id >> w;
            add(id,w);
        } else {
            cin >> id;
            del(id);
        }
    }
    cout << ans;

    return 0;
}
