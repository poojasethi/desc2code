#include <iostream>
#include <algorithm>
#include <cstdio>
#include <map>

using namespace std;

const int BufferSize = 100000;

int h, m, n;
int value[BufferSize];
char buf[BufferSize];
map<int, int> table;

int Add(int id, int hash)
{
	int call = 0;
	while (value[hash] != -1)
	{
		hash = (hash + m) % h;
		++call;
	}
	value[hash] = id;
	table[id] = hash;
	return call;
}

void Delete(int id)
{
	value[table[id]] = -1;
}

int main(int argc, char *argv[])
{
	fill_n(value, BufferSize, -1);

	long long total = 0;
	scanf("%d %d %d", &h, &m, &n);
	m %= h;
	for (int i = 0; i < n; ++i)
	{
		int id, hash;

		scanf("%s", buf);
		if (buf[0] == '+')
		{
			scanf("%d %d", &id, &hash);
			total += Add(id, hash);
		}
		else
		{
			scanf("%d", &id);
			Delete(id);
		}
	}

	cout << total << endl;

	return 0;
}
