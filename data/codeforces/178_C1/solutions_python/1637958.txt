#include <iostream>
#include <cstring>
#include <cstdio>
#define N 5005
using namespace std;
int a[N];
int main()
{
    int h,m,n,id,hash,i;
    char buf[10];
    while(~scanf("%d%d%d",&h,&m,&n))
    {
        for(i=0;i<N;i++) a[i]=-1;
        int ans=0;
        while(n--)
        {
            scanf("%s",buf);
            if(buf[0]=='+')
            {
                scanf("%d%d",&id,&hash);
                for(i=0;a[(hash+i*m)%h]!=-1;i++,ans++);
                a[(hash+i*m)%h]=id;
            }
            else if(buf[0]=='-')
            {
                scanf("%d",&id);
                for(int i=0;i<=h;i++)
                    if(a[i]==id)
                    {
                        a[i]=-1;
                        break;
                    }
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
