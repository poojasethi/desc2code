#include <iostream>
#include <string>
#include <map>
using namespace std;
bool placed[5100];
map<int,int> mymap;
int main()
{
    int ans=0;
    string st;
    int h,m,n,id,pos;
    cin>>h>>m>>n;
    while (n--)
    {
        cin>>st;
        if (st=="+")
        {
            cin>>id>>pos;
            int add=0;
            while (placed[(pos+add*m)%h]) add++;
            mymap[id]=(pos+add*m)%h;
            placed[(pos+add*m)%h]=true;
            ans+=add;
        }
        else
        {
            cin>>id;
            placed[mymap[id]]=false;
            mymap[id]=0;
        }
    }
    cout<<ans<<endl;
    return 0;
}
