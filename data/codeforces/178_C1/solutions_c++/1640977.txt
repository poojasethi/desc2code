#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <vector>
#include <cmath>
#include <set>
#include <map>
#include <queue>
#include <ctime>
#include <string>

using namespace std;

bool used[100000];
int h, m, n;

int main() {
	//freopen("input.txt", "r", stdin); freopen("output.txt", "w", stdout);
	
	int id, hash;
	cin >> h >> m >> n;
	string s;
	int ans = 0;
	map <int, int> M;
	for(int i = 0; i < n; ++i) {
		cin >> s;
		if(s == "+") {
			cin >> id >> hash;
			while(used[hash]) {
				hash = (hash + m) % h;
				++ans;
			}
			M[id] = hash;
			used[hash] = true;
		}
		else {
			cin >> id;
			used[M[id]] = false;
			M.erase(id);
		}
	}

	cout << ans << endl;

	return 0;
}
