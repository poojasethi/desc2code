#include <bits/stdc++.h>
using namespace std;

typedef long long int64;
struct point{
	double x,y;
};

const int MAXN=100000;
int n,x,y,t;
point p[MAXN*2+5];

double cross(const point& a,const point& b,const point& o){
    return (a.x-o.x)*(b.y-o.y)-(a.y-o.y)*(b.x-o.x);
}

int main(void){
	cin>>n;
	for (int i=0;i<n;++i){
		cin>>x>>y;
		p[i]={x,y};
		p[i+n]=p[i];
	}
	cin>>t;
	for (int cases=0;cases<t;++cases){
		cin>>x>>y;
		point xy={x,y};
		bool flag=true;
		for (int i=0;i<n;++i)
			if (cross(p[i+1],xy,p[i])>=0){
				flag=false;
				cout<<0<<endl;
				break;
			}
		if (!flag) continue;
		int64 ans=(int64)n*(n-1)*(n-2)/6;
		for (int i=0,j=2;i<n;++i){
			while (cross(xy,p[j],p[i])>0) ++j;
			ans-=(int64)(j-i-1)*(j-i-2)/2;
		}
		cout<<ans<<endl;
	}
	return 0;
}
