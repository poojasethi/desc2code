#include <cstdio>
#define FOR(i,s,e) for(int i=s;i<e;i++)
typedef long long LL;
struct P {
        LL x, y;
        P() { }
        P(LL x, LL y):x(x),y(y){ }
        P operator-(P p){return P(x-p.x,y-p.y);}
        P operator+(P p){return P(x+p.x,y+p.y);}
        LL operator^(P p){return x*p.y-y*p.x;}
        void eat(){scanf("%I64d%I64d",&x,&y);}
};
const int N=2030600;
P p[N];
int n, t;
int main() {
        scanf("%d", &n);
        FOR(i,0,n) p[i].eat(), p[i+n]=p[i];
        scanf("%d", &t);
        while (t--) {
                P q;
                q.eat();
                LL ans=(LL)n*(n-1)/2*(n-2)/3;
                bool ok=1;
                FOR(i,0,n) ok&=((p[i]-q)^(p[i+1]-q))<0;
                if(!ok) { puts("0"); continue; }
                for(int i=0,j=1;i<n;i++) {
                        while(j<i+n && ((p[i]-q)^(p[j]-q))<=0) ++j;
                        LL c=j-i-1;
                        ans-=c*(c-1)/2;
                }
                printf("%I64d\n", ans);
        }
}
