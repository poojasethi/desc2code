#include<cstdio>
#include<complex>
#include<algorithm>
using namespace std;
typedef double ld;
typedef long long ll;
#define N 100010
complex<ld>P[N],p;
ld A[N*2];
ll n,m;
int main()
{
    ll i,j;
    //freopen("a.txt","r",stdin);
    scanf("%I64d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%lf%lf",&P[i].real(),&P[i].imag());
    }
    int T;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%lf%lf",&p.real(),&p.imag());
        for(i=0;i<n;i++)
        {
            A[i]=arg(P[i]-p);
        }
        sort(A,A+n);
        for(i=0;i<n;i++)
        {
            A[i+n]=A[i]+M_PI*2;
        }
        ll ans=n*(n-1)*(n-2)/6;
        for(j=0,i=0;i<n;i++)
        {
            while(j+1<2*n&&A[j+1]-A[i]<=M_PI)j++;
            ans-=(j-i)*(j-i-1)/2;
        }
        printf("%I64d\n",ans);
    }
    return 0;
}
