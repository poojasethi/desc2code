#include <iostream>
#include <cstdio>
using namespace std;
typedef long long LL;

struct cor{ LL x,y; } a[100100],cc;
int n,t;

LL operator *(cor x,cor y) { return x.x*y.y-x.y*y.x; }
cor operator -(cor x,cor y) { cor tmp;tmp.x=x.x-y.x;tmp.y=x.y-y.y; return tmp; }

LL calc(cor cc)
{
    LL ans=LL(n)*(n-1)*(n-2)/6;int j=2;
    for (int i=1;i<=n;i++)
    {
  		if ((cc-a[i])*(a[i%n+1]-a[i])<0)  return 0;
  		while ( (cc-a[i])*(a[j%n+1]-a[i])>0 ) j=j%n+1;
  		int tmp=j-i;
  		if (tmp<0) tmp+=n;
  		ans-=LL(tmp)*(tmp-1)/2;
    }
    if (n==3) return 1;else return ans;
}

int main()
{
 //   freopen("int.txt","r",stdin);

    cin.sync_with_stdio(false);
    cin>>n;
    for (int i=1;i<=n;i++)  cin>>a[i].x>>a[i].y;
    cin>>t;
    for (int i=1;i<=t;i++) { cin>>cc.x>>cc.y; cout<<calc(cc)<<endl; }

    return 0;
}
