#include<cstdio>
#include<iostream>
#define fo(i,a,b) for(i=a;i<=b;++i)
const int mn=200100;
typedef long long LL;
int i,j,n,t,X,Y,x[mn],y[mn];
LL r,z;
LL cross(LL xa,LL ya,LL xb,LL yb)
{
    return xa*yb-xb*ya;
}
int main()
{
    scanf("%d",&n);
    fo(i,1,n)
    {
        scanf("%d%d",&x[i],&y[i]);
        x[i+n]=x[i],y[i+n]=y[i];
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&X,&Y);
        fo(i,1,n)
            if(cross(x[i+1]-x[i],y[i+1]-y[i],X-x[i+1],Y-y[i+1])>=0)break;
            if(i<n)
            {
                printf("0\n");
                continue;
            }
        r=(n*(n-1LL)*(n-2LL))/6LL,j=2;
        fo(i,1,n)
        {
             while(cross(x[j+1]-x[i],y[j+1]-y[i],X-x[j+1],Y-y[j+1])<0)++j;
             r-=((j-i)*(j-i-1LL))/2LL;
        }
        printf("%I64d\n",r);
    }
    return 0;
}
