#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
struct point
{
    long long x,y;
}p[100010];
long long n;
long long xx(point a,point b,point c)
{
    return (a.x-c.x)*(b.y-c.y)-(b.x-c.x)*(a.y-c.y);
}
int main()
{
    while(scanf("%I64d",&n)!=EOF)
    {
        for(int i=0;i<n;i++)
        scanf("%I64d%I64d",&p[i].x,&p[i].y);
        int t;scanf("%d",&t);
        while(t--)
        {
            point cao;
            scanf("%I64d%I64d",&cao.x,&cao.y);
            int pos=1;long long ans=0;
            for(int i=0;i<n;i++)
            {
                if(xx(p[(i+1)%n],cao,p[i])>0)
                {puts("0");goto end;}
                while(xx(p[(pos+1)%n],cao,p[i])<0)pos=(pos+1)%n;
                long long temp=pos-i;
                if(temp<0)temp+=n;
                ans+=temp*(temp-1)/2;
            }
            ans=n*(n-1)*(n-2)/6-ans;
            printf("%I64d\n",ans);
            end:;
        }
    }
}
