#include<cstdio>
#define ll __int64
ll xx,yy,x[200005],y[200005],n,t;
int main()
{
    scanf("%I64d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%I64d%I64d",&x[i],&y[i]);
        x[i+n]=x[i];y[i+n]=y[i];
    }
    scanf("%I64d",&t);
    while(t--)
    {
        scanf("%I64d%I64d",&xx,&yy);
        int f=1;
        for(int i=0;f&&i<n;i++)
        if((x[i+1]-x[i])*(yy-y[i])-(y[i+1]-y[i])*(xx-x[i])>=0)f=0;
        if(!f){puts("0");continue;}
        ll sum=n*(n-1)*(n-2)/6,cnt,top=0;
        for(ll i=0;i<n;i++)
        {
            while((x[i]-xx)*(y[top+1]-yy)-(x[top+1]-xx)*(y[i]-yy)<0)
            top++;
            cnt=top-i;
            if(cnt>=2)sum-=cnt*(cnt-1)/2;
        }
        printf("%I64d\n",sum);
    }
}
