#include<bits/stdc++.h>
#define maxx 111111
#define ll long long
using namespace std;
struct point{
     ll x,y;
     point(){}
     void input(){scanf("%I64d%I64d",&x,&y);}
};
point p[maxx];
ll hehe(point a,point b,point c){
	return (b.x-a.x)*(c.y-a.y)-(b.y-a.y)*(c.x-a.x);
}
int main(){
    int i,j,n,t;
    scanf("%d",&n);
    for(i=0;i<n;i++) p[i].input();
    p[n]=p[0];
    scanf("%d",&t);
    ll ans=0;
    while(t--){
        point x;
        x.input();
        ans=(ll)n*(n-1)*(n-2)/6;
        j=2;
		for(i=0;i<n;i++){
            if(hehe(x,p[i],p[i+1])>0){
                ans=0;
                break;
            }
            while(hehe(x,p[i],p[j])<0) j=(j+1)%n;
            ll k=(n+i-j-1)%n;
            ans-=k*(k+1)/2;
		}
		printf("%I64d\n",ans);
    }
}
