#include <iostream>
using namespace std;
typedef pair<int, int> ii;
typedef long long ll;
ll ccw(ii a, ii b, ii c) {
	return ll(b.first -a.first) * (c.second - a.second) - ll(b.second - a.second) * (c.first - a.first);
}
ii poly[100000];
int main() {
	int N;
	cin >> N;
	for (int i = 0; i < N; ++i) {
		cin >> poly[i].first >> poly[i].second;
	}
	int M;
	cin >> M;
	for (int i = 0; i < M; ++i) {
		ii pt;
		cin >> pt.first >> pt.second;
		int b = 1;
		ll s = ll(N) * (N - 1) * (N - 2) / 6;
		for (int a = 0; a < N; ++a) {
			while (b != a && ccw(poly[a], pt, poly[b]) > 0) {
				b = (b + 1) % N;
			}
			int d1 = (b - a - 1 + N) % N;
			s -= ll(d1) * (d1 - 1) / 2;
		}
		cout << max(0LL, s) << '\n';
	}
}
