#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <cmath>

using namespace std;

const int MAXN=100010;

struct P{double x,y;}p[2*MAXN],s;
int n,m;

P operator - (P a,P b)
{
    a.x-=b.x,a.y-=b.y;
    return a;
}
double operator * (P a,P b)
{
    return a.x*b.y-a.y*b.x;
}
void input_data()
{
    scanf("%d",&n);
    for(int i=0;i<n;++i)scanf("%lf%lf",&p[i].x,&p[i].y),p[i+n]=p[i];
}
bool check_out()
{
    for(int i=0;i<n;++i)
    if(((s-p[i])*(p[i+1]-p[i]))<=0)return true;
    return false;
}
long long cal()
{
    long long ans=(long long)n*(n-1)*(n-2)/6;
    for(int i=0,j=2;i<n;++i)
    {
        while(((p[j]-p[i])*(s-p[i]))<0) j++;
        ans-=(long long)(j-i-2)*(j-i-1)/2;
    }
    return ans;
}
void solve()
{
    scanf("%d",&m);
    for(int i=0;i<m;++i)

        if(check_out())cout<<0<<endl;
        else cout<<cal()<<endl;

}
