#include<cstdio>
#include<cstring>
#define maxn 100010
typedef long long LL;
struct Point{
    int x,y;
}P[maxn],x;
int n,Q;
LL ans,area;
LL multi(const Point &P0,const Point &P1,const Point &P2){
    return (LL)(P1.x-P0.x)*(P2.y-P0.y)-(LL)(P2.x-P0.x)*(P1.y-P0.y);
}
LL abs(LL a){return a<0?-a:a;}
LL calc(){
    LL cur=abs(multi(x,P[1],P[n]));
    int i,j;
    for(i=1;i<n;i++){
        cur+=abs(multi(x,P[i],P[i+1]));
        if(cur>area)return 0;
    }
    ans=(LL)n*(n-1)*(n-2)/6;
    for(i=j=1;i<=n;i++){
        while(multi(P[i],P[j%n+1],x)<0)j++;
        ans-=(LL)(j-i)*(j-i-1)/2;
    }
    return ans;
}
int main(){
    int i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%d%d",&P[i].x,&P[i].y);
    for(i=2;i<n;i++)area+=multi(P[1],P[i+1],P[i]);
    scanf("%d",&Q);
    while(Q--){
        scanf("%d%d",&x.x,&x.y);
        printf("%I64d\n",calc());
    }
}
