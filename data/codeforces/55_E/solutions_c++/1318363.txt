#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long LL;
const int N = 100005;
int n, t;
LL x[2*N], y[2*N], px, py;

LL cross(LL ax, LL ay, LL bx, LL by) {
	//printf("%I64d %I64d, %I64d %I64d\n", ax, ay, bx, by);
	return ax*by-bx*ay;
}
int main(void) {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%I64d%I64d", &x[i], &y[i]);
		x[n+i]=x[i]; y[n+i]=y[i];
	}
	scanf("%d", &t);
	while(t--) {
		scanf("%I64d%I64d", &px, &py);
		int f=0;
		for(int i=0;i<n;i++) {
			if(cross(px-x[i], py-y[i], x[i+1]-x[i], y[i+1]-y[i])<=0) {
				f=1;break;
			}
		}
		if(f) { puts("0"); continue; }
		LL ans = n*1LL*(n-1)*(n-2)/6;
		for(int i=0,j=0;i<n;i++) {
			if(j<=i) j=i+1;
			while(j<i+n && cross(px-x[i], py-y[i], x[j]-x[i], y[j]-y[i])>0)
				++j;
			//printf("(%d,%d)\n", i, j);
			LL p=j-i-2;
			ans -= p*(p+1)/2;
		}
		printf("%I64d\n", ans);
	}
	return 0;
}
