#include<cmath>
#include<cstdio>
#include<string>
#include<vector>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<map>
using namespace std;
#define ll __int64
ll x[200010],y[200010],xo,yo,n,t;
int main()
{
    scanf("%I64d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%I64d%I64d",&x[i],&y[i]);
        x[i+n]=x[i];y[i+n]=y[i];
    }
    scanf("%I64d",&t);
    while(t--)
    {
        scanf("%I64d%I64d",&xo,&yo);
        int in=1;
        ll tot=n*(n-1)*(n-2)/6LL,cnt,tp=0;
        for(ll i=0;in&&i<n;i++)
        {
            if((x[i+1]-x[i])*(yo-y[i])>=(y[i+1]-y[i])*(xo-x[i]))in=0;
            while((xo-x[i])*(yo-y[tp+1])<(yo-y[i])*(xo-x[tp+1]))
            tp++;
            cnt=tp-i;
            if(cnt>=2LL)tot-=cnt*(cnt-1)/2LL;
        }
        printf("%I64d\n",in?tot:0LL);
    }
}
