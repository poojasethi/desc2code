#include<cstdio>
#include<iostream>
#define LL long long
#define N 100000
using namespace std;
struct point{
	LL x,y;
	point(){}
	point(LL x,LL y):x(x),y(y){}
	point operator -(point a){
		return point(x-a.x,y-a.y);
	}
	LL operator %(point a){
		return x*a.y-y*a.x;
	}
};
point pt[N],c;
int n;
LL solve(){
	int j=0;
	LL ans=0;
	for(int i=0;i<n;++i){
		while(  (pt[(j+1)%n]-pt[i])%(c-pt[i])  < 0 ) ++j;
		ans+=((j-i-1LL)*(j-i))/2;
	}	
	return n*(n-1LL)*(n-2LL)/6 - ans;
}
bool inside(){
	for(int i=1;i<=n;++i)
		if( (pt[i%n]-pt[i-1])%(c-pt[i-1]) >0 )
			return 0;
	return 1;
}
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;++i)
		cin >> pt[i].x >> pt[i].y;
	int t;scanf("%d",&t);
	for(int i=0;i<t;++i){
		cin >> c.x >> c.y;
		if(inside())
			cout << solve() << endl;
		else
			cout << 0 << endl;
	}
	return 0;
}
