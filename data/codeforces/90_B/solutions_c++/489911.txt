#include<stdio.h>
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	char  arr[110][110];
	int i,j,k,f;
	for(i=0;i<n;i++)
		scanf("%s",arr[i]);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			f=0;
			for(k=0;k<m;k++)
			{
				if(arr[i][j]==arr[i][k] && j!=k)
				{
					f=1;
					break;
				}
			}
			if(f!=1)
			{
				for(k=0;k<n;k++)
				{
					if(arr[i][j]==arr[k][j] && i!=k)
					{
						f=1;
						break;
					}
				}
			}
			if(f!=1)
				printf("%c",arr[i][j]);
		}
	}
	printf("\n");
	return 0;
}
