#include <iostream>
#include <map>
using namespace std;
int main(){
    char mas[110][110];
    int n,m;
    map <char,int> x[110],y[110];
    cin>>n>>m;
    for(int i=0;i<n;i++)
        cin>>mas[i];
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++){
            x[i][mas[i][j]]++;
            y[j][mas[i][j]]++;
       
    }    
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            if(x[i][mas[i][j]]==1 and y[j][mas[i][j]]==1)
                cout<<mas[i][j];    
}
