#include <iostream>

using namespace std;

int n, m;
int r[105][200], c[105][200];
char g[105][105];

int main(){
    cin >> n >> m;
    for(int i = 0; i < n; ++i){
        cin >> g[i];
    }
    for(int i = 0; i < n; ++i){
        for(int j = 0; j < m; ++j){
            r[i][g[i][j]]++;
            c[j][g[i][j]]++;
        }
    }
    for(int i = 0; i < n; ++i){
        for(int j = 0; j < m; ++j){
            if(r[i][g[i][j]] == 1 && c[j][g[i][j]] == 1) cout << g[i][j];
        }
    }
}
