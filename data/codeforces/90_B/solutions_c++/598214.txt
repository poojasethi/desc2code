#include <stdio.h>

int main(void) {
	int n, m, i, j, k;
	char a[105][105];
	scanf("%d%d",&m,&n);
	for(i=0;i<m;i++)
		scanf("%s", a[i]);
	int b[105][105]={};
	for(i=0;i<m;i++)
		for(j=0;j<n;j++)
			for(k=j+1;k<n;k++)
				if(a[i][j]==a[i][k]) b[i][j]=b[i][k]=1;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			for(k=j+1;k<m;k++)
				if(a[j][i]==a[k][i]) b[j][i]=b[k][i]=1;
	for(i=0;i<m;i++)
		for(j=0;j<n;j++)
			if(!b[i][j]) putchar(a[i][j]);
	puts("");
	return 0;
}
