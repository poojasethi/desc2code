#include <cstdio>

int c[110][26],r[110][26];

int main() {
	int n,m;
	scanf("%d%d",&n,&m);
	char s[110][110];
	for(int i=0;i<n;i++) {
		scanf("%s",s[i]);
		for(int j=0;j<m;j++) {
			int x=s[i][j]-'a';
			r[i][x]++;
			c[j][x]++;
		}
	}
	for(int i=0;i<n;i++) for(int j=0;j<m;j++) {
		int x=s[i][j]-'a';
		if (r[i][x]==1 && c[j][x]==1) printf("%c",s[i][j]);
	}
	printf("\n");
	return 0;
}
