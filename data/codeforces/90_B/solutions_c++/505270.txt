#include <iostream>

using namespace std;

int main() {
int n,m;
cin >> n >> m;
char a[n][m],c;
for(int i = 0;i < n;i++)
for(int j = 0;j < m;j++) {
cin >> c;
if(c == '\n') cin >> c;
a[i][j] = c;
}
for(int i = 0;i < n;i++) {
for(int j = 0;j < m;j++) {
c = a[i][j];
bool flag = false;
for(int ix = 0;ix < n;ix++) {
if(c == a[ix][j] && ix != i) {
flag = true;
}
}
for(int jx = 0;jx < m;jx++) {
if(c == a[i][jx] && jx != j) {
flag = true;
}
}
if(!flag) cout << a[i][j];
}
}
cout << endl;
}
