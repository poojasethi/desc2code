#include <iostream>
#include <cstdio>
using namespace std;
#define FO(i,n) for (int i=0;i<n;i++)
char a[105][105];
int row[128][105],col[128][105];
int main ()
{
//	freopen("test.txt","r",stdin);
	int n,m;
	cin>>n>>m;
	FO(i,n)scanf("%s",a+i);
	FO(i,n)FO(j,m)row[a[i][j]][i]++,col[a[i][j]][j]++;
	FO(i,n) FO(j,m) if (row[a[i][j]][i] == 1 && col[a[i][j]][j] == 1) printf("%c",a[i][j]);
    return 0;
}
