#include<iostream>
#include<ctype.h>
using namespace std;
int main()
{ int n,m,i,j,k,c=0;
 char a[100][100],ch;
 cin>>n>>m;
 for(i=0;i<n;i++)
  for(j=0;j<m;j++)
    cin>>a[i][j];
  for(i=0;i<n;i++)
   for(j=0;j<m;j++)
    { ch=a[i][j];
      c=0;
      for(k=0;k<m;k++)
        if(ch==a[i][k]&&k!=j)
          {c++;break;}
      for(k=0;k<n&&c==0;k++)
          if(ch==a[k][j]&&k!=i)
            {c++;break;}
       if(c==0)
         cout<<a[i][j];
    }
return 0;
}
