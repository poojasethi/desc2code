#include<iostream>
using namespace std;
char a[102][102],b[102][102];
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n;i++)for(int j=0;j<m;j++)b[i][j]=a[i][j];
	for(int i=0;i<n;i++)
	for(int j=0;j<m;j++){
		for(int k=0;k<n;k++)
		if(a[k][j]==a[i][j] && k!=i)b[i][j]='&',b[k][j]='&';
		for(int k=0;k<m;k++)
		if(a[i][k]==a[i][j] && k!=j)b[i][j]='&',b[i][k]='&';
	}
	for(int i=0;i<n;i++)for(int j=0;j<m;j++)if(b[i][j]!='&')cout<<b[i][j];
	return 0;
}
