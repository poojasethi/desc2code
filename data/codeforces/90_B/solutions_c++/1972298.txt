#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace  std;
char data[105][105];
int main(){
	int n,m;
	while(cin>>n>>m){
		for(int i=1;i<=n;i++)
			for(int j=1;j<=m;j++)
				cin>>data[i][j];


		for(int i=1;i<=n;i++)
			for(int j=1;j<=m;j++){
				int flag=1;

				char tmp=data[i][j];
				for(int k=1;k<=m;k++)
					if(data[i][k]==tmp&&k!=j)flag=0;
				for(int k=1;k<=n;k++)
					if(data[k][j]==tmp&&k!=i)flag=0;
				if(flag)
					cout<<tmp;
			}
			cout<<endl;
	}
}
