#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{char c[101][101];
 int n,m;
 cin>>n>>m;
 for (int i=0; i<n; i++) cin>>c[i];


for (int i=0; i<n; i++)
 for (int j=0; j<m; j++)
 {bool f=true;
    for (int i1=0; i1<n; i1++) if ((c[i][j]==c[i1][j])&&(i!=i1)) {f=false; break;}
    for (int j1=0; j1<m; j1++) if ((c[i][j]==c[i][j1])&&(j!=j1)) {f=false; break;}
  if (f) cout<<c[i][j];
 }
    return 0;
}
