#include <iostream>
#include <string>

using namespace std;

int N,M,row[110][30],col[110][30];
string S[110];

int main()
{
	cin >> N >> M;
	for (int i=0;i<N;i++)
	{
		cin >> S[i];
		for (int j=0;j<S[i].size();j++)
		{
			row[i][S[i][j]-'a']++;
			col[j][S[i][j]-'a']++;
		}
	}
	for (int i=0;i<N;i++)
		for (int j=0;j<S[i].size();j++)
			if (row[i][S[i][j]-'a']==1&&col[j][S[i][j]-'a']==1)
				cout << S[i][j];
	return 0;
}
