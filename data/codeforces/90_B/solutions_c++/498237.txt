#include<stdio.h>

char a[101][101];
int b[101][101];

int main()
{
int i,j,x,y,k;
//freopen("r.txt","r",stdin);
scanf("%d%d\n",&i,&j);
for(x=0;x<i;x++)gets(a[x]);
k=0;
for(x=0;x<i;x++)
{
	for(y=0;y<j;y++)
	{
		for(k=x+1;k<i;k++)
			if(a[x][y]==a[k][y]){b[k][y]=1;b[x][y]=1;}
		for(k=y+1;k<j;k++)
			if(a[x][y]==a[x][k]){b[x][k]=1;b[x][y]=1;}
	}
}
k=0;
for(x=0;x<i;x++)
	for(y=0;y<j;y++)
		if(b[x][y]==0)printf("%c",a[x][y]);
printf("\n");
return 0;
}
