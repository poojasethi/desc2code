#include <iostream>
#include <cstdio>
using namespace std;
int i,j,n,m,b[101][200],c[101][200];
char a[1010][101];
int main()
{
    cin>>n>>m;
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
            {
                cin>>a[i][j];
                b[i][a[i][j]]++;
                c[j][a[i][j]]++;
        }
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
            if(b[i][a[i][j]]==1 && c[j][a[i][j]]==1)cout<<a[i][j];
    return 0;
}
