#include<bits/stdc++.h>

#define MAX 200005

using namespace std;

int n,m;
char mat[101][101];
char cad[1001];
int fil[500][101],col[500][101];
int main() {
	cin>>n>>m;
	for(int i=0;i<n;i++){
		scanf("%s",cad);
		for(int j=0;j<m;j++){
			mat[i][j]=cad[j];
			fil[cad[j]][i]++;
			col[cad[j]][j]++;

		}
	}
	for(int i=0;i<n;i++){
		for (int j = 0; j < m; ++j) {
			if(fil[mat[i][j]][i]<=1 && col[mat[i][j]][j]<=1){
				cout<<mat[i][j];
			}
		}
	}

	return 0;
}
