#include <cstdio>
int main() {
    int n, m, o; char s[111][111]; scanf("%d%d", &n, &m);
    for (int i = 0;i < n;i++) scanf("%s", s[i]);
    for (int i = 0;i < n;i++)
        for (int j = 0;j < m;j++) {
            for (int k = o = 0;k < n && !o;k++) if (i != k && s[i][j] == s[k][j]) o = true;
            for (int k = 0;k < m && !o;k++) if (j != k && s[i][j] == s[i][k]) o = true;
            if (!o) putchar(s[i][j]);
        }
}
