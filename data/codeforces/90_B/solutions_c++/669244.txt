#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int m,n; scanf("%d%d",&m,&n);
	char c[100][101];
	rep(i,m) scanf("%s",c[i]);

	bool f[100][100]={};
	rep(i,m) rep(j,n) {
		rep(k,m) if(i!=k && c[i][j]==c[k][j]) f[i][j]=true;
		rep(k,n) if(j!=k && c[i][j]==c[i][k]) f[i][j]=true;
	}

	rep(i,m) rep(j,n) if(!f[i][j]) putchar(c[i][j]);

	return 0;
}
