#include<iostream>
#include<stdio.h>
using namespace std;
char A[100][100];
int main()
{
	int n,m,i,j,k,mark;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		for(i=0;i<n;i++)
			cin>>A[i];
		for(i=0;i<n;i++)
			for(j=0;j<m;j++)
			{
				mark=0;
				for(k=0;k<n;k++)
					if(k!=i&&A[k][j]==A[i][j])
					{	mark=1; break;}
				for(k=0;k<m&&!mark;k++)
					if(k!=j&&A[i][k]==A[i][j])
					{	mark=1; break; }
				if(!mark)
					cout<<A[i][j];
			}
		cout<<endl;
	}
	return 0;
}
