#include <iostream>
#include<stdio.h>
using namespace std;
int b[101][200]={0},c[101][200]={0};
int main()
{
    char s[101][101];

    int n,m,i,j;
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
        for(j=0;j<m;j++)
        {
            cin>>s[i][j];
            b[i][s[i][j]]++;
            c[j][s[i][j]]++;
        }
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<m;j++)
        {
            if(b[i][s[i][j]]==1&&c[j][s[i][j]]==1)
            cout<<s[i][j];
        }
    }
    return 0;

}
