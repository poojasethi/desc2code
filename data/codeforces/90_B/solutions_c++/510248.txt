#include <cstdio>
#include <iostream>
#include <cstring>
using namespace std;
char str[101][101];
int h[101][101];
int s[101][101];
int main(){
	int i,j,k;
	int n,m;
	cin>>n>>m;
	memset(h,0,sizeof(h));
	memset(s,0,sizeof(s));
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			cin>>str[i][j];
			h[i][str[i][j]]++;
			s[j][str[i][j]]++;
		}
	}
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			if(h[i][str[i][j]]==1&&s[j][str[i][j]]==1){
				cout<<str[i][j];
			}
		}
	}
	cout<<endl;
}
