#include <iostream>
#include <string>
#include <vector>
#include <map>

using namespace std;

int main()
{
  int n,m;
  cin>>n>>m;
  vector<string> v(n);
  for (int i=0;i<n;i++)
    cin>>v[i];
  vector<map<char,int> > fil(n),col(m);
  for (int i=0;i<n;i++) {
    for (int j=0;j<m;j++) {
      fil[i][v[i][j]]++;
      col[j][v[i][j]]++;
    }
  }
  string s;
  for (int i=0;i<n;i++)
    for (int j=0;j<m;j++)
      if (fil[i][v[i][j]]==1 and col[j][v[i][j]]==1)
	s+=v[i][j];
  cout<<s<<endl;
}
