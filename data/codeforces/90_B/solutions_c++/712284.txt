#include<iostream>
#include<string>
using namespace std;
int main()
{
	int n,m;
	string a[100]={};
	bool f;

	cin>>n>>m;

	for(int i = 0 ; i < n ; ++i )
	{
		cin>>a[i];
	}

	for(int i = 0 ; i < n ; ++i )
	{
		for(int j = 0 ; j < a[i].size() ; ++j )
		{
			f = true;
			for(int l = 0 ; l < n ; ++l)
			{
				if(a[i][j] == a[l][j] && l != i)	{f = false; break;}

			}
			for(int l = 0 ; l < a[i].size() ; ++l )
			{
				if(a[i][j] == a[i][l] && l != j)	{f = false; break;}
			}
			if(f)	cout<<a[i][j];
		}
	}
	cout<<endl;
}
