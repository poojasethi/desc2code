#include <iostream>
#include <string>
using namespace std;

int main(){
	string s;
	char C[101][101];
	int n,m,dem;
	bool Check;
	cin>>n>>m;
	for (int i=1; i<=n; i++){
		cin>>s;
		for (int j=0; j<m; j++)
			C[i][j+1]=s[j];
	}
	for (int i=1; i<=n; i++){
		for (int j=1; j<=m; j++){
			dem=0;
			for (int li=1; li<=m; li++)
				if (C[i][j]==C[i][li]) dem++;
			for (int li=1; li<=n; li++)
				if (C[i][j]==C[li][j]) dem++;
			if (dem==2) cout<<C[i][j];
		}
	}
}
