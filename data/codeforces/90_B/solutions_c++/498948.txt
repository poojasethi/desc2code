#include <iostream>
#include <string>
using namespace std;

#define rep(i,n) for(int i=0;i<n;i++)

int main(){
	int n,m,del[100][100];
	char c[100][101];
	cin>>n>>m;
	rep(i,n)rep(j,m)cin>>c[i][j];
	rep(i,n)rep(j,m)rep(k,m)if(j!=k&&c[i][j]==c[i][k])del[i][j]=1;
	rep(i,m)rep(j,n)rep(k,n)if(j!=k&&c[j][i]==c[k][i])del[j][i]=1;
	string ans;
	rep(i,n)rep(j,m)if(!del[i][j])ans+=c[i][j];
	cout<<ans<<endl;
}
