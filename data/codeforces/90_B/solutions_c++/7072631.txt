#include<iostream>
using namespace std;
int main()
{
char line[100][102];
int n,m,i,j,k,flag;
cin>>n>>m;
for(i=0; i<n; i++)
cin>>line[i];
for(i=0; i<n; i++)
for(j=0; j<m; j++)
{
flag=0;
for(k=0; k<n; k++)
if(k!=i)
if(line[k][j]==line[i][j])
{
flag=1;
break;
}
if(flag==1)
continue;
for(k=0; k<m; k++)
if(k!=j)
if(line[i][k]==line[i][j])
{
flag=1;
break;
}
if(flag!=1)
cout<<line[i][j];
}
return 0;
}
