#include <bits/stdc++.h>
using namespace std;
int main(){

int m,n;
cin>>m>>n;
char arr[m][n];
int i,j;
for(i=0;i<m;i++){
    for(j=0;j<n;j++) cin>>arr[i][j];
}
int r,c,f;
f=0;
for(i=0;i<m;i++){
    for(j=0;j<n;j++){

        for(r=0;r<m;r++){
            if(arr[r][j]==arr[i][j] && r!=i) f=1;
        }
        for(c=0;c<n;c++){
            if(arr[i][c]==arr[i][j] && c!=j) f=1;
        }

        if(f==0) cout<<arr[i][j];
        f=0;
    }
}



return 0;}
