#include <iostream>
using namespace std;
int N,M;
char A[121][121];
int satir[121][150],sutun[121][150];
void read()
{
    cin >> N >> M;
    for(int i=1;i<=N;i++)
        for(int j=1;j<=M;j++)
        {
            cin >> A[i][j];
            satir[i][A[i][j]]++;
            sutun[j][A[i][j]]++;
        }
}
int main()
{
    read();
    for(int i=1;i<=N;i++)
        for(int j=1;j<=M;j++)
            if(satir[i][A[i][j]]==1 &&  sutun[j][A[i][j]]==1) cout << A[i][j];
    cout << endl;
    return 0;
}
