#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int n,m;
    char ch[105][105];
    cin>>n>>m;
    for(int i=0;i<n;++i)
        cin>>ch[i];
    for(int i=0;i<n;++i)
        for(int j=0;j<m;++j)
        {
            int ind=0;
            for(int k=0;k<n;++k)
            if(ch[k][j]==ch[i][j])++ind;
            for(int k=0;k<m;++k)
            if(ch[i][k]==ch[i][j])++ind;
            if(ind==2)cout<<ch[i][j];
        }
    cout<<endl;
}
