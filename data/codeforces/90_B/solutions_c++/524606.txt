#include<stdio.h>
#include<string.h>
#include <iostream>
using namespace std;
char a[100][100];
int b[200][26];
int main()
{
	int n,m,i,j;
	while(cin>>n>>m)
	{
		memset(b,0,sizeof(b));
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				cin>>a[i][j];
				b[i][a[i][j]-'a']++;
				b[n+j][a[i][j]-'a']++;
			}
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				if(b[i][a[i][j]-'a']==1&&b[n+j][a[i][j]-'a']==1)
					cout<<a[i][j];
			}
		}
		cout<<endl;
	}
	return 0;
}
