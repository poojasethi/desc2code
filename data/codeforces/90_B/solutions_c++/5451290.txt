#include <iostream>

using namespace std;

int main()
{
	int n, m, c = 0;
	char a[101][101];
	cin >> n >> m;
	
	for (int i = 0; i < n; i++)
	for (int j = 0; j < m; j++)
		cin >> a[i][j];
		
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			for (int k = 0; k < m; k++)
				if (k != j && a[i][k] == a[i][j])
					c = 1;
			for (int k = 0; k < n; k++)
				if (k != i && a[k][j] == a[i][j])
					c = 1;
			if (c == 0)
				cout << a[i][j];
			c = 0;
		}
	}
	cout << endl;
	
	return 0;
}
