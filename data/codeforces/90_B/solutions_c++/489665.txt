#include<iostream>
#include<map>
#include<vector>
#include<cmath>
using namespace std;
#define REP(i,b,n) for(int i=b;i<n;i++)
#define rep(i,n)   REP(i,0,n)
const int N = 100;
char m[N][N];

main(){
  int r,c;
  while(cin>>r>>c){
    map<char,int>R[N],C[N];
    rep(i,r)cin>>m[i];
    rep(i,r){
      rep(j,c){
	R[i][m[i][j]]++;
	C[j][m[i][j]]++;
      }
    }
    rep(i,r){
      rep(j,c){
	if (R[i][m[i][j]] == 1 && 
	    C[j][m[i][j]] == 1)cout << m[i][j];
      }
    }
    cout << endl;
  }
}
