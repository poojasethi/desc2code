#!/usr/bin/env python
# -*- coding: utf-8 -*-

n, m = map(int, raw_input().split())
fld = list()
for i in xrange(n):
    s = raw_input()
    fld.append(list())
    for c in s:
        fld[i].append([c, True])
for i in xrange(n):
    for j in xrange(m):
        for k in xrange(i+1, n):
            if fld[i][j][0] == fld[k][j][0]:
                fld[i][j][1] = fld[k][j][1] = False
        for k in xrange(j+1, m):
            if fld[i][j][0] == fld[i][k][0]:
                fld[i][j][1] = fld[i][k][1] = False
print ''.join([c[0] for l in fld for c in l if c[1] == True])
