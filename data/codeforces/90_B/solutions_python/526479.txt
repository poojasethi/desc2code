n,m = map(int,raw_input().split())
mp = []
for i in xrange(n):
    mp.append(raw_input())
    
ans = ''
for i in range(n):
    for j in range(m):
        if mp[i].count(mp[i][j]) > 1:
            continue
        for k in range(n):
            if k != i and mp[i][j] == mp[k][j]:
                break
        else:
            ans += mp[i][j]
print ans
