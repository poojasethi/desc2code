#!/usr/bin/python
# -*- coding: utf-8 -*- 

def checkrow(s,c,x,y):
    for i in xrange(c):
        if i == x:
            continue
        if s[y][x] == s[y][i]:
            return False
    return True

def checkcol(s,r,x,y):
    for i in xrange(r):
        if i == y:
            continue
        if s[y][x] == s[i][x]:
            return False
    return True

r,c = map(int,raw_input().split(' '))

s = []
for a in xrange(r):
    s.append( [b for b in raw_input()] )

result = ''
for y in xrange(r):
    for x in xrange(c):
        if checkrow(s,c,x,y) and checkcol(s,r,x,y):
            result += s[y][x]
print result
