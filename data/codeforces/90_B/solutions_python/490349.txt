#! /usr/bin/env python

from sys import stdin

def solve(grid):
	row_counts=[dict() for i in xrange(len(grid))]
	col_counts=[dict() for i in xrange(len(grid[0]))]
	for i,row in enumerate(grid):
		for j,char in enumerate(row):
			row_counts[i][char]=row_counts[i].get(char,0)+1
			col_counts[j][char]=col_counts[j].get(char,0)+1
	return "".join([grid[i][j] for i in xrange(len(grid)) for j in xrange(len(grid[0])) if row_counts[i][grid[i][j]]==1 and col_counts[j][grid[i][j]]==1])

if __name__=='__main__':
	n,m=map(int,stdin.readline().split())
	grid=[stdin.readline().strip() for i in xrange(n)]
	print solve(grid)
