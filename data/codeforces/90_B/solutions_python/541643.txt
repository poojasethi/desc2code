case = raw_input()
s = case.split()
n = int(s[0])
m = int(s[1])

ans = ''
arr = []
for i in range(n):
	arr.append(raw_input())

for i in range(n):
	for j in range(m):
		p = True
		for k in range(n):
			if i != k and arr[i][j] == arr[k][j]:
				p = False
		for k in range(m):
			if j != k and arr[i][j] == arr[i][k]:
				p = False
		if p:
			ans += arr[i][j]

print ans
