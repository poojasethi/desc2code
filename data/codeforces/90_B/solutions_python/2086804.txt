n,m=map(int, raw_input().split())
c=[raw_input() for _ in range(n)]
C=lambda s: set( e for e in set(s) if s.count(e)>1 )

h=[ C(e) for e in c]

#print h
v= [ C(e) for e in zip(*c)]
#print v

print ''.join(str(c[i][j]) for i in range(n) for j in range(m) if c[i][j] not in h[i] and c[i][j] not in v[j])
