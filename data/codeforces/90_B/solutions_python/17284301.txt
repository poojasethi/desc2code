n,m = map(int, raw_input().split())
s = [raw_input() for _ in xrange(n)]
st = zip(*s)
rows = [] * n
cols = [] * m 
for row in s: rows.append(set([c for c in row if row.count(c) > 1]))
for col in st: cols.append(set([c for c in col if col.count(c) > 1]))
st = ''
for x in xrange(n):
	for y in xrange(m):
		if s[x][y] not in rows[x] and s[x][y] not in cols[y]:
			st += s[x][y]
print st
