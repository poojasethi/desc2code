
import copy

n, m = map(int, raw_input().split())
C = []
for i in xrange(n):
   C.append(raw_input())

D = [' ']*n*m
for i, row in enumerate(C):
   for j, c in enumerate(row):
      col = ''.join([r[j] for r in C])
      D[i*m + j] = (c, '.')[row.count(c) > 1 or col.count(c) > 1]

print ''.join(D).replace('.', '')
