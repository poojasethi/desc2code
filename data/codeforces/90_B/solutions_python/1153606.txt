n,m = list(map(int, str(raw_input()).split()))
i=0
rows,columns = [],[]
while i<n:
    rows += [raw_input().strip()]
    i+=1
i=0
while i<m:
    columns += [''.join([x[i] for x in rows])]
    i+=1
out = ''
for j in rows:
    i=0
    for char in j:
        if columns[i].count(char) == 1 and j.count(char)==1:
            out += char
        i+=1
print out
