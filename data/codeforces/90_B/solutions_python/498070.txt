n, m = map(int, raw_input().split())
a = []

for i in range(n):
    a.append(raw_input())

s = ''
for i in range(n):
    for j in range(m):
        if a[i].count(a[i][j]) > 1:
            continue
        for k in range(n):
            if k != i and a[i][j] == a[k][j]:
                break
        else:
            s += a[i][j]
print s
