from sys import stdin

(n, m) = map(int, stdin.readline().split(' '))
a = [stdin.readline()[:-1] for i in range(n)]
b = ''

for i in range(n):
  for j in range(m):
    ct = a[i].count(a[i][j])
    if ct < 2:
      ct = 0
      for k in range(n):
        if a[k][j] == a[i][j]:
          ct += 1
    if ct < 2:
      b += a[i][j]
      
      
print b
