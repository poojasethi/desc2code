from sys import stdin

height, width = [int(x) for x in stdin.readline().split()]
rows = [stdin.readline().strip() for _ in xrange(height)]
cols = [''.join(rows[i][c] for i in xrange(height)) for c in xrange(width)]
crossed = [[False]*width for _ in xrange(height)]
out = []
for r, line in enumerate(rows):
    for c, ch in enumerate(line):
        if not crossed[r][c]:
            for i, a in enumerate(line):
                if i != c and ch == a:
                    crossed[r][i] = crossed[r][c] = True
            for i, a in enumerate(cols[c]):
                if i != r and ch == a:
                    crossed[i][c] = crossed[r][c] = True
        if not crossed[r][c]:
            out.append(ch)
print ''.join(out)
