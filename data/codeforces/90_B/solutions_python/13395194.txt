#!/usr/bin/env python
# -*- coding:utf-8 -*-

#n=int(raw_input())
n,m=map(int, raw_input().split())
bd=[]
for i in range(n):
    bd.append(raw_input())
ans=""


def chk(l,x,y):
    c=l[x][y]
    cnt=0
    if l[x].count(c)>1:
        return 0
    for k in range(n):
        if l[k][y]==c:
            cnt+=1
    return 1 if cnt==1 else 0


for i in range(n):
    for j in range(m):
        if chk(bd,i,j):
            ans+=bd[i][j]

print ans
