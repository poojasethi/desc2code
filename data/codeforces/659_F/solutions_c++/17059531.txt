#include <bits/stdc++.h>
#define N 1000100
using namespace std;
long long n, a[1001][1001], kq, k, b[N], m, stt, c[1001][1001];
long long ll, x;
int init()
{
    for (int i = 1; i <= m; i++)
        for (int j = 1; j <= n; j++) c[i][j] = 0;
}
int kt(int i, int j)
{
    if (i > 0 && i <= m && j > 0 && j <= n && c[i][j] == 0 && a[i][j] >= x) return 1; else return 0;
}
int duyet(int i, int j, int p)
{
    if (ll * x >= k) return 0;
    c[i][j] = p; ll++;
    if (kt(i - 1, j) == 1) duyet(i - 1, j, p);
    if (kt(i + 1, j) == 1) duyet(i + 1, j, p);
    if (kt(i, j - 1) == 1) duyet(i, j - 1, p);
    if (kt(i, j + 1) == 1) duyet(i, j + 1, p);
}
int tim()
{
    long p = 0;
    for (int i = 1; i <= m; i++)
        for (int j = 1; j <= n; j++)
            if (c[i][j] == 0 && a[i][j] == x) {
                p++; ll = 0;
                duyet(i, j, p);
                if (ll * x >= k) {kq = p; break;}
        }
}
int main()
{
    cin >> m >> n >> k;
    for (int i = 1; i <= m; i++)
        for (int j = 1; j <= n; j++) cin >> a[i][j];
    stt = 0;
    for (int i = 1; i <= m; i++)
        for (int j = 1; j <= n; j++) {
            stt++;
            b[stt] = a[i][j];
        }
    sort(b + 1,     b + stt + 1);
    //for (int i = 1; i <= stt; i++) cout << b[i] << " ";
    for (int i = 1; i <= stt; i++)
        if ( b[i] != 0 && b[i] != b[i - 1])
            if (k % b[i] == 0 && k / b[i] <= stt - i + 1) {
                init;
                x = b[i];
                ll = 0;
                tim();
                if (kq > 0) break;
                //cout << ll << " ";
            }
    if (kq > 0) {
        cout << "YES" << "\n";
        for (int i = 1; i <= m; i++)
        {
            for (int j = 1; j <= n; j++)
                if (c[i][j] == kq && k > 0 && a[i][j] >= x) {
                        cout << x << " ";
                        k -= x;
                }
                else cout << 0 << " ";
            cout << "\n";
        }
    }
    if (kq == 0) cout << "NO";
}
