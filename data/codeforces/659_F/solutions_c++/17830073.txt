#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 1000010
#define ll long long
using namespace std;
struct zhl
{
	int x,y;ll num;
}p[maxn],q[maxn];
int n,m,t[maxn],f[maxn],len,vis[1010][1010];
ll a[1010][1010],k,ans[1010][1010];
const int move[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
int id(int x,int y)
{
	return (x-1)*m+y;
}

int cmp(zhl a,zhl b)
{
	return a.num>b.num;
}

int legal(int x,int y)
{
	if (!x||!y||x>n||y>m) return 0;
	return 1;
}

int find(int x)
{
	return f[x] == x?x:f[x] = find(f[x]);
}
void bfs(int i,int j)
{
	int time = k / a[i][j],T = a[i][j];
	q[1].x = i,q[1].y = j;
	vis[i][j] = 1;
	ans[i][j] = T;
	if (!(--time)) return;
	int l = 0,r = 1;
	for (;l<r;)
	{
		int x = q[++l].x,y = q[l].y;
		for (int t = 0;t < 4;t++)
		{
			int xx = x+move[t][0],yy = y+move[t][1];
			if (!legal(xx,yy)||vis[xx][yy]||a[xx][yy]<T) continue;
			vis[xx][yy] = 1;ans[xx][yy] = T;
			if (!(--time)) return;
			q[++r].x = xx,q[r].y = yy;
		}
	}
}
void print()
{
	for (int i = 1;i <= n;i++,puts(""))
		for (int j = 1;j <= m;j++)
		{
			printf("%I64d",ans[i][j]);
			if (j<m) printf(" ");
		}
}

int main()
{
	scanf("%d%d%I64d",&n,&m,&k);
	for (int i = 1;i <= n;i++)
		for (int j = 1;j <= m;j++)
		{
			scanf("%I64d",&a[i][j]);
			p[++len] = (zhl){i,j,a[i][j]};
			t[len] = 1;
			f[len] = len;
		}
	sort(p+1,p+len+1,cmp);
	for (int i = 1;i <= len;i++)
	{
		int x = p[i].x,y = p[i].y;
		vis[x][y] = 1;
		for (int j = 0;j < 4;j++)
		{
			int xx = x+move[j][0],yy = y+move[j][1];
			if (!legal(xx,yy)||!vis[xx][yy]) continue;
			int r1 = find(id(x,y)),r2 = find(id(xx,yy));
			if (r1 == r2) continue;
		//	printf("%d %d\n",r1,r2);
			f[r2] = r1;
			t[r1] += t[r2];
		}
	}
	memset(vis,0,sizeof(vis));
	for (int i = 1;i <= n;i++)
		for (int j = 1;j <= m;j++)
			if (a[i][j] && k % a[i][j] == 0 && k / a[i][j] <= t[id(i,j)])
			{
				puts("YES");
				bfs(i,j);
				print();
				return 0;
			}
	puts("NO");
	return 0;
}
