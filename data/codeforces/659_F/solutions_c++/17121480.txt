int a[1010][1010],vis[1010][1010],vs[1010][1010];
ll m,n,k,t=0;
int bfs(int x,int y)
{
    int i,j,ii,jj,s=a[x][y];
    if(vs[x][y]||k%a[x][y]!=0||k/a[x][y]>n*m)return 0;
    t++;ll cnt=k/a[x][y];cnt--;vis[x][y]=t;vs[x][y]=1;
    if(cnt==0)
    {   puts("YES");
        rep(ii,n){rep(jj,m)printf("%d ",vis[ii][jj]==t?s:0);printf("\n");}
        return 1;
    }
    queue<pair<int,int> >q;q.push(mp(x,y));
    while(!q.empty())
    {
        int u=q.front().fi,v=q.front().se;q.pop();
        for(i=-1;i<=1;i++)
            for(j=-1;j<=1;j++)
        {   int l=u+i,r=v+j;
            if((i&&j)||l<0||r<0||l>=n||r>=m||vis[l][r]==t||a[l][r]<s)continue;
            cnt--,q.push(mp(l,r)),vis[l][r]=t;if(a[l][r]==s)vs[l][r]=1;
            if(cnt==0)
            {   puts("YES");
                rep(ii,n){rep(jj,m)printf("%d ",vis[ii][jj]==t?s:0);printf("\n");}
                return 1;
            }
        }
    }
    return 0;
}
