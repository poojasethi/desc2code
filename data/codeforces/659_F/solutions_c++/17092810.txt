#include<bits/stdc++.h>
using namespace std;
#define ll long long

int n,m,par[1000005],siz[1000005],vis[1000005],num,a,pr;
ll k;
pair<ll,int> temp[1000005];

int find(int x)
{
	if(par[x]==x)
		return x;
	return par[x]=find(par[x]);
}

void uniun(int x,int y)
{
	int p=find(par[x]);
	int q=find(par[y]);
	if(p==q)
		return;
		
	par[q]=p;
	siz[p]+=siz[q];
}

int sizee(int x)
{
	int p=find(x);
	return siz[p];
}

void func(int x)
{
	if((x%m!=m-1)&&(vis[x+1]))
		uniun(x,x+1);
		
	if((x%m!=0)&&(vis[x-1]))
		uniun(x,x-1);
	
	if((x+m<m*n)&&(vis[x+m]))
		uniun(x,x+m);
	
	if((x-m>=0)&&(vis[x-m]))
		uniun(x,x-m);
}

void print(int x)
{
	if(pr==0)
		return;
	
	if((x%m!=m-1)&&(vis[x+1]==1))
	{
		vis[x+1]=2;
		pr--;
		print(x+1);
	}
	
	if(pr==0)
		return;
		
	if((x%m!=0)&&(vis[x-1]==1))
	{
		pr--;
		vis[x-1]=2;
		print(x-1);
	}
	
	if(pr==0)
		return;
	
	if((x+m<m*n)&&(vis[x+m]==1))
	{
		pr--;
		vis[x+m]=2;
		print(x+m);
	}
	
	if(pr==0)
		return;
	
	if((x-m>=0)&&(vis[x-m]==1))
	{	
		pr--;
		vis[x-m]=2;
		print(x-m);
	}
}

int main()
{
	cin>>n>>m>>k;
	
	for(int i=0;i<n*m+2;i++)
	{
		vis[i]=0;
		siz[i]=1;
		par[i]=i;
	}
	
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		{
			scanf("%d",&a);
			temp[m*i+j].first=a;
			temp[m*i+j].second=m*i+j;
		}
	
	sort(temp,temp+n*m,greater<pair<ll,int> >());
	
	for(int i=0;i<n*m;i++)
	{
		num=temp[i].first;
		vis[temp[i].second]=1;
		func(temp[i].second);
		
		if(k%num==0)
		{
			if(sizee(temp[i].second)>=k/num)
			{
				cout<<"YES"<<endl;
				int l=find(temp[i].second);
				vis[temp[i].second]=2;
				pr=k/num-1;
				print(temp[i].second);
				
				for(int j=0;j<n;j++)
				{
					for(int o=0;o<m;o++)
					{
						if(vis[j*m+o]==2)
							cout<<num<<" ";
						else
							cout<<0<<" ";
					}
					cout<<endl;
				}	
				return 0;
			}
		}
	}
	
	cout<<"NO"<<endl;
	return 0;
}
