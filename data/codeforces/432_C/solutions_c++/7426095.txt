#include<iostream>
#include <math.h>
using namespace std;

int n, a[100010], s[1000000], k=0, f=1, j, prime[100010], t;

int main()
{
	cin >> n;
	prime[1] = 1;
	for(int i=2; i<=n; i++)
        if(i <= sqrt(double(n+0.0)))
            for(j=i+i; j<=n; j+=i)
                prime[j] = 1;
	for(int i=1; i<=n; i++)
		cin >> a[i];
	while(f < n+1)
	{
		if(t == 0)
			for(j=f; j<=n; j++)
			{
				if(a[j] == f)
					break;
			}
		else
			j = t;
		if(j == f)
			f++;
		else
			for(int i=f; i<j; i++)
				if(! (prime[j-i+1]))
				{
					s[k] = i;     s[k+1] = j;     swap(a[i], a[j]);     k += 2;
					if(a[f] == f)
					{
						f++;     t = 0;
					}
					else t = i;
					break;
				}
	}
	if(k == 0)
		cout << 0 << endl;
	else
	{
		cout << k/2 << endl;
		for(int i=0; i<k; i+=2)
			cout << s[i] << " " << s[i+1] << endl;
	}
	return 0;
}
