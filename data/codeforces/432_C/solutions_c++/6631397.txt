#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
bool f[100001];
int index[100001];
int tub[100001];
int a[100001];
int main(){
	for(int i = 2; i < 100000; i ++){
		if(!f[i]){
			for(int j = i * 2; j < 100000; j += i) f[j] = true;
			tub[i - 1] = i - 1;
		}else tub[i-1] = tub[i-2];
	}
	tub[99999] = tub[100000] = tub[99998];
	
	int n,i,k,j;
	vector<pair<int,int> >result;
	cin >> n;
	for(i = 1; i <= n; i ++) {
		cin >> a[i];
		index[a[i]] = i;
	}
	for(i = 1; i <= n; i ++){
		while(index[i] != i){
			j = tub[index[i] - i];
			
			result.push_back(make_pair(index[i] - j, index[i]));
			swap(a[index[i]-j],a[index[i]]);
			index[a[index[i]]] = index[i];
			index[i] -= j;
		}
	}
	cout << result.size() << endl;
	for(i = 0; i < result.size(); i ++) cout << result[i].first << " " << result[i].second << endl;
	return 0;
}
