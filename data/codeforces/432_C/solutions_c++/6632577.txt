#include <iostream>
#include <vector>
#include <cstring>
using namespace std;
const int MAX = 100005;
bool p[MAX];
vector<pair<int, int> > ans;
int a[MAX];
int d = 2;
void move(int x, int y)
{
	if (x == y)
		return;
	//if (x > y)
		//swap(x, y);
	for (int i = y; i > x; i--)
		if (p[i - x + 1])
		{
			swap(a[x], a[i]);
			ans.push_back(make_pair(x, i));
			move(i, y);
			break;
		}
}
int main()
{
	ios::sync_with_stdio(false);
	memset(p, true, sizeof(p));
	p[0] = p[1] = false;
	for (int i = 2; i < MAX; i++)
		if (p[i])
			for (int j = 2 * i; j < MAX; j += i)
				p[j] = false;
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	for (int i = 1; i <= n; i++)
		while (a[i] != i)
			move(i, a[i]);
	cout << ans.size() << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i].first << " " << ans[i].second << endl;
	return 0;
}
