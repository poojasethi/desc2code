#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int n;
	cin>>n;
	bool prime[n+1];
	int last[n+1],now=-1;
	prime[1]=false;
	for(int i=2;i<=n;i++)prime[i]=true;
	for(int i=2;i<=n;i++)for(int j=2;j*j<=i;j++)if(i%j==0)prime[i]=false;
	for(int i=0;i<=n;i++)
	{
		if(prime[i])now=i;
		last[i]=now;
	}
	int a[n],k[n];
	int ans1[n*10],ans2[n*10],num=0;
	for(int i=0;i<n;i++)scanf("%i",&a[i]),k[a[i]]=i;
	for(int i=0;i<n;i++)
	{
		if(a[i]==i+1)continue;
		int s=k[i];
		for(int j=i;j<n;j++)if(a[j]==i+1){s=j;break;}
		while(s!=i)
		{
			int temp=a[s-last[s-i+1]+1];
			a[s-last[s-i+1]+1]=a[s];
			a[s]=temp;
			ans1[num]=s-last[s-i+1]+1;
			ans2[num]=s;
			num++;
			s=s-last[s-i+1]+1;
		}
	}
	printf("%i\n",num);
	for(int i=0;i<num;i++)printf("%i %i\n",ans1[i]+1,ans2[i]+1);
	return 0;
}
