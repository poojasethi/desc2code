#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>

using namespace std;

int N,Ans;
bool yes[100005];
int a[100005],seat[100005];
int res1[500005],res2[500005];

void Prime()
{
	memset(yes,true,sizeof(yes));
	yes[1]=false;
	for (int i=1; i<=100000; i++)
		if (yes[i])
			for (int j=i*2; j<=100000; j+=i) yes[j]=false;
}

int main()
{
	Prime();
	scanf("%d",&N);
	for (int i=1; i<=N; i++)
	{
		scanf("%d",&a[i]);
		seat[a[i]]=i;
	}
	int Ans=0;
	for (int i=1; i<=N; i++)
	{
		if (seat[i]==i) continue;
		for (; seat[i]!=i; )
		{
			int j=i;
			for (; !yes[seat[i]-j+1]; j++);
			Ans++;
			res1[Ans]=j; 
			res2[Ans]=seat[i];
			int tmp=seat[i];
			seat[i]=j;
			seat[a[j]]=tmp;
			swap(a[j],a[tmp]);
		}
	}
	cout << Ans << endl;
	for (int i=1; i<=Ans; i++) cout << res1[i] << ' ' << res2[i] << endl;
	return 0;
}
