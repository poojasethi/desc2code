#include<bits/stdc++.h>
using namespace std;
int pri(int n)
{
	if(n==2)
        return 1;
	int i;
	for(i=2;i*i<=n;i++)
	{
		if(n%i==0)
            return 0;
	}
	return 1;
}
int main()
{
	int arel[100001],arpos[100001];
	int i,j,n,k;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",arel+i);
		arpos[arel[i]]=i;
	}
	int noofop=0;
	int ans1[500000];
	int ans2[500000];
	for(i=1;i<=n-1;i++)
	{
		while(arpos[i]!=i)
		{
			j=i;
			while(pri(arpos[i]-j+1)!=1)
			{
				j++;
			}

			ans1[noofop]=j;
			ans2[noofop]=arpos[i];
			noofop++;

			arel[arpos[i]]=arel[j];
			arpos[arel[j]]=arpos[i];
			arel[j]=i;
			arpos[i]=j;
		}
	}
	printf("%d\n",noofop);
	for(i=0;i<noofop;i++)
		printf("%d %d\n",ans1[i],ans2[i]);
	return 0;
}
