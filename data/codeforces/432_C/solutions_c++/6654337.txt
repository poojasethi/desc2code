#include <iostream>
#include <cmath>

using namespace std;

const int maxn = 100000*5+10;

int n,a[maxn],pos[maxn],prime[maxn],l[maxn],r[maxn];

void _prime(void)
{
	for(int i=1;i<=maxn;i++)
		prime[i]=1;
	for(int i=2;i<=sqrt(maxn);i++)
	{
		if(prime[i]==1)
		{
			int j=i*2;
			do
			{
				prime[j] = 0;
				j += i;
			}
			while(j<maxn);
		}
	}
}
int main(void)
{
	_prime();
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
		pos[a[i]] = i;
	}
	int tt=0;
	for(int i=1;i<=n-1;i++)
	{
		while(pos[i]!=i)
		{
			int j=i;
			while(prime[pos[i]-j+1]==0)
			{
				j++;
			}
			l[tt]=j;
			r[tt]=pos[i];
			tt++;
			a[pos[i]]=a[j];
			pos[a[j]]=pos[i];
			a[j]=i;
			pos[i]=j;
		}
	}
	cout<<tt<<endl;
	for(int i=0;i<tt;i++)
		cout<<l[i]<<" "<<r[i]<<endl;
	return 0;
}
