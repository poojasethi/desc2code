#include<stdio.h>
#include<cstring>
using namespace std;
int a[102345];
int p[102345];
int sav[502345][2],cnt=0;
bool pri[102345];
int last[102345];
int prem[102345],tmp=0;
void Prime(){
	for(int i=2;i<=100000;i++){
		if(!pri[i]){
			prem[++tmp]=i;
			last[i]=i;
		}
		else last[i]=last[i-1];
		for(int j=1;j<=tmp;j++){
			if(i*prem[j]>100000) break;
			pri[i*prem[j]]=true;
		}
			
		
	}
	return ;
}
void swp(int x,int y){
	int t;
	t=a[x],a[x]=a[y],a[y]=t;
	p[a[x]]=x;
	p[a[y]]=y;
	cnt++;
	sav[cnt][0]=y,sav[cnt][1]=x;
	return ;
}
int main(){
	Prime();
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]),p[a[i]]=i;
	for(int i=1;i<=n;i++)
		while(p[i]!=i)
				swp(p[i],p[i]-last[p[i]-i+1]+1);
	printf("%d\n",cnt);
	for(int i=1;i<=cnt;i++)
		printf("%d %d\n",sav[i][0],sav[i][1]);
	scanf("%d",&n);
	return 0;
}
