#include <iostream>
using namespace std;
int p[100000],a[100000],y[100000];
# include<vector>
int main() {
    int n;
    cin>>n;
    vector< pair< int,int> >v;
    int i,j,t,ct=0;
    for(i=2;i<=n;i++)
    if(!p[i])
    for(j=2*i;j<=n;j+=i)
    p[j]=1;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        y[a[i]]=i;
    }
    for(i=1;i<=n;i++)
    {
        j=y[i];
        while(j>i)
        {
            t=i;
            while(p[j+1-t]) t++;
            ct++;
            v.push_back(make_pair(t,j));
            y[a[t]]=j;
            y[a[j]]=t;
            int temp=a[t];
            a[t]=a[j];
            a[j]=temp;
            j=t;
        }
    }
    cout<<ct<<endl;
    for(i=0;i<v.size();i++)
    if(v[i].first!=v[i].second)
    cout<<v[i].first<<" "<<v[i].second<<endl;
    return 0;
}
