#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[100010],n,tot;
bool pri[100010];
struct node{
	int x,y;
}ans[500010];

void pre_work(){
	memset(pri,true,sizeof(pri));
	pri[0]=pri[1]=false;
	for (int i=2;i<=100000;i++){
		if (pri[i]){
			for (int j=2;j*i<=100000;j++)
				pri[i*j]=false;
		}
	}
}

void change(int x,int y){
	if (x==y) return;
	if (x>y) swap(x,y);
	for (int i=y;i>x;i--){
		if(pri[i-x+1]){
			swap(a[x],a[i]);
	        ans[++tot].x=x; ans[tot].y=i;
		    change(i,y); break;
		}
	}
}

int main(){
	pre_work();
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for (int i=1;i<=n;i++)
		{while (i!=a[i]) change(i,a[i]);}
	printf("%d\n",tot);
	for (int i=1;i<=tot;i++)
		printf("%d %d\n",ans[i].x,ans[i].y);
	return 0;
}
