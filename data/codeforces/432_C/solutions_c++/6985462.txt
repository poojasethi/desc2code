#include <stdio.h>
#include <string.h>
int shu[100005];
int x[500005][2];
int ai[100005];
int wi[100005];
int main()
{
	int n,i,j;
	int k=0,s=0,w;
	memset(shu,0,sizeof(shu));
	for(i=2;i<317;i++)
	{
		if(shu[i]==0)
		{
			for(j=i*i;j<100001;j+=i)
				shu[j]=1;
		}
	}

	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{

		scanf("%d",&ai[i]);
		wi[ai[i]]=i;

	}

	for(i=1;i<=n;i++)
	{
		while(ai[i]!=i)
		{
			for(j=wi[i],w=i;j>w;w++)
			{

				if(shu[j-w+1]==0)
				{
					k=ai[j];
					ai[j]=ai[w];
					ai[w]=k;
					x[s][0]=w,x[s][1]=j;
					
					wi[i]=w;
					wi[ai[j]]=j;
					s++;
					break;
				}
			}


		}
	}
	printf("%d\n",s);
for(i=0;i<s;i++)
{
	printf("%d %d\n",x[i][0],x[i][1]);

}
	return 0;
}
