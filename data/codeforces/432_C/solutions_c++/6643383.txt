#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int n,pr,ans;
int p[100005],a[100005];
int d[100005];
bool f[100005];
int fr[100005];
int x[500005],y[500005];
void check(int xx)
{
	while(1)
	{
		int t=d[xx]-xx+1;
		if(t==1)
		{
			break;
		}
		x[++ans]=xx+fr[t]-1,
		y[ans]=xx+t-1;
		swap(a[x[ans]],a[y[ans]]);
		d[a[x[ans]]]=x[ans];
		d[a[y[ans]]]=y[ans];
		t=fr[t];
	}
}
int main()
{
	int i,j;
	scanf("%d",&n);
	for(i=2;i<=n;i++)
	{
		if(!f[i])
		{
			p[++pr]=i;
			for(j=i+i;j<=n;j+=i)
				f[j]=1;
		}
	}
	f[1]=0;
	for(i=2;i<=n;i++)
	{
		if(!f[i])
		{
			fr[i]=1;
		}
		else
		{
			fr[i]=fr[i-1]+1;
		}
	}
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]),
		d[a[i]]=i;
	}
	for(i=1;i<=n;i++)
	{
		check(i);
	}
	printf("%d\n",ans);
	for(i=1;i<=ans;i++)
	{
		printf("%d %d\n",x[i],y[i]);
	}
	return 0;
}
