#include <cstdio>
#include <cstring>
const int maxn1=500005;
const int maxn2=100005;
int a[maxn2],p[maxn2],pos[maxn2];
int x[maxn1],y[maxn1]; 
void get_prim()
{
	for(int i=2;i<maxn2;i++)
		p[i]=1;
	for(int i=2;i<=maxn2;i++)
		for(int j=2;j*j<=i;j++)
		{
			if(i%j==0)
				p[i]=0;
		}
} 
int main()
{
	int n,cnt=0,q=1,m=1;
	get_prim();
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		pos[a[i]]=i;
	}
	while(m<=n)
	{
		int j=pos[m];
		while(a[j]==m&&a[m]!=m)
		{
			q=1;
			int len=j-m+1;
			while(!p[len])
		 		len--;
		 	pos[a[j-len+1]]=j;
			pos[a[j]]=j-len+1;
			int t=a[j-len+1];
			a[j-len+1]=a[j];
			a[j]=t;
			x[cnt]=j-len+1;
			y[cnt]=j;
			cnt++;
			j=j-len+1;
		}
		m++;
	}
	printf("%d\n",cnt);
	for(int i=0;i<cnt;i++)
		printf("%d %d\n",x[i],y[i]);
	return 0;	
}
