#include<iostream>
#include<math.h>
#include<vector>

using namespace std;
int main()
{
    int n,i, h, j, c=0, t, temp;
    vector <pair<int, int> >v;
    cin>>n;
    
    int a[100001]={}, d[100001]={}, y[100001]={};
    h=sqrt(n);
    for(i=1; i<=n; i++)
    {
        cin>>a[i];
        y[a[i]]=i;
    }
    for(i=2; i<=h; i++)
    {
        if(!d[i])
        for(j=i*2; j<=n; j+=i)
        d[j]=1;
    }
    for(i=1; i<=n; i++)
    {
        j=y[i];
        while(j>i)
        {
            t=i;
            while(d[j-t+1])t++;
            c++;
            y[a[t]]=j;
            y[a[j]]=t;
            temp=a[t];
            a[t]=a[j];
            a[j]=temp;
            v.push_back(make_pair(t, j));
            j=t;
        }
    }cout<<c<<endl;
    for(i=0; i<v.size(); i++)
    {
        cout<<v[i].first<<" "<<v[i].second<<endl;
    }
}
