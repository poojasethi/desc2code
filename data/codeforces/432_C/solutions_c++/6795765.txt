#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int maxn = 1e5+10;
int p[maxn],n;
int a[maxn],l[maxn],s[maxn*5][2];
void init()
{
    for(int i=2;i<maxn;i++)
        if(!p[i])for(int j=2;j*i<maxn;j++)
            p[i*j]=1;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
        l[a[i]]=i;
    }
}
int solve()
{
    int ans=0,j;
    for(int i=1;i<=n;i++)
        while(l[i]!=i){
            for(j=i;j<=l[i];j++)
                if(!p[l[i]-j+1])break;
            s[ans][0]=j;
            s[ans++][1]=l[i];
            int t=l[i];
            swap(l[i],l[a[j]]);
            swap(a[j],a[t]);
        }
    return ans;
}
int main()
{
    init();
    int ans=solve();
    cout<<ans<<endl;
    for(int i=0;i<ans;i++)
        cout<<s[i][0]<<' '<<s[i][1]<<endl;
    return 0;
}
