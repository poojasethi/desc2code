#include<bits/stdc++.h>
using namespace std;

#define mx 100005

bool p[mx];
int inp[mx];

vector<pair<int,int> >ans;

void Move(int x,int y){
    if(x == y) return;
    if(x > y) swap(x,y);
    for(int i=y;i>x;i--){
        if(!p[i - x + 1]){
            swap(inp[i],inp[x]);
            ans.push_back(make_pair(x,i));
            Move(i,y);
            break;
        }
    }
}

int main(){
    int n; cin >> n;
    p[0] = p[1] = 1;
    for(int i=4;i<mx;i+=2) p[i] = 1;
    for(int i=3;i*i<mx;i+=2){
        if(!p[i]){
            for(int j=i*i;j<mx;j+=i) p[j] = 1;
        }
    }
    for(int i=1;i<=n;i++) cin >> inp[i];
    for(int i=1;i<=n;i++){
        while(inp[i] != i) Move(i,inp[i]);
    }
    cout << ans.size() << '\n';
    for(int i=0;i<ans.size();i++) cout << ans[i].first << " " << ans[i].second << '\n';
    return 0;
}
