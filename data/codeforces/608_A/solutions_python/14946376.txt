n, s = map(int, raw_input().split())

pairs = []
for _ in range(n):
	fi, ti = map(int, raw_input().split())
	pairs.append((fi,ti))

pairs.sort(key = lambda p : p[0], reverse = True)

timer = 0
previous = s
for p in pairs:
	timer += max(previous - p[0], p[1] - timer)
	previous = p[0]

timer += previous

print timer
