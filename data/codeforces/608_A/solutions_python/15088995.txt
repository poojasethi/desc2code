n , s = [int(x) for x in raw_input().split()]
A = []
for i in xrange(n):
    f , t = [int(x) for x in raw_input().split()]
    A.append([f , t])   
A.sort()
A.reverse()
ans = 0
for i in xrange(n):
    A[i][1] = max(0 , A[i][1] - ans)
    ans += max(s - (A[i][0]) , A[i][1])
    s = A[i][0]
   
ans += A[n - 1][0]
    
print ans
