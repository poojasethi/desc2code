N, S = raw_input().split(' ')
N, S = int(N), int(S)

L = []
for i in range(N):
    L.append(raw_input().split(' '))
    L[i][0] , L[i][1] = int(L[i][0]) , int(L[i][1])

L.sort()    
i = N-1
T = 0
while i >= 0:
    T = max(T + (S - L[i][0]), L[i][1])
    S = L[i][0]
    i -= 1

T += S
print T
