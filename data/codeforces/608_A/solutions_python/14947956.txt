n, s = map(int, raw_input().split())
ft = []
for i in xrange(n):
    ft.append(map(int, raw_input().split()))
ft.sort(key = lambda x:(x[0], x[1]), reverse = True)
ft_cut = [ft[0]]
for i in xrange(1,n):
    if ft[i][0] != ft[i-1][0]:
        ft_cut.append(ft[i])
f = s
t = 0
for i in ft_cut:
    t = max(i[1], t+f-i[0])
    f = i[0]
t += f
print t
