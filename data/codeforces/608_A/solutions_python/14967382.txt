n, s= map(int,raw_input().split())
ans = 0
d = {}
for i in xrange(n):
    f,t = map(int,raw_input().split())
    if f not in d:
        d[f] = t
    else:
        d[f] = max(d[f], t)
if s in d:
    ans += d[s]
for i in xrange(s-1,-1,-1):
    ans += 1
    if i in d:
        ans += d[i] - ans if d[i]>ans else 0
    
print ans
