from operator import itemgetter as cmp
n, s = map(int, raw_input().split())
lvl = []
for _ in range(n):
  lvl.append(map(int, raw_input().split()))
sorted(lvl, key = cmp(0, 1), reverse = True)
res = 0
for i in range(n):
  res += s - lvl[i][0]
  if lvl[i][1] - res > 0:
    res += lvl[i][1] - res
  s = lvl[i][0]
print res + s
