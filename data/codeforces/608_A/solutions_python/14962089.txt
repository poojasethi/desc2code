def myFunc(ll):
  return ll[0]

n, s = map(int, raw_input().split())
lst = []
for i in range(n):
  l = map(int, raw_input().split())
  lst.append(l)
  
lst.sort(key = myFunc)
lst.reverse()
#print lst
sum = 0
for l in lst:
#  print l
  sum += s - l[0]
  if l[1] - sum >= 0:
    sum += l[1] - sum
  s = l[0]
#  print sum

sum += s
  
print sum
