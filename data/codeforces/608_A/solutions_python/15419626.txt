if __name__ == "__main__":
  n, s = map(int, raw_input().split(" "))
  v = []
  for i in range(n):
    v += [tuple(map(int, raw_input().split(" ")))]

  v = sorted(v, key=lambda (x,y) : x, reverse=True)

  count = 0
  cur_floor = s
  for (f, t) in v:
    count += cur_floor - f
    if t > count:
      count += t - count

    cur_floor = f

  count += cur_floor

  print count
