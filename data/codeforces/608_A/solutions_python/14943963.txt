
n, s = tuple(map(int, raw_input().split()))
f = {}
for i in xrange(n):
    fi, ti = tuple(map(int, raw_input().split()))
    if fi in f:
        f[fi].append(ti)
    else:
        f[fi] = [ti]

time = 0
current = s
for k in sorted(f.keys(), reverse=True):
    time += current-k
    current = k
    t = max(f[k])
    if t>time:
        time = t

time += current
print time
