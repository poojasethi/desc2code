n,m=map(int,raw_input().split())
a=[]
for i in range(n):
    x,y = map(int,raw_input().split())
    a.append((x,y))

a.sort(key=lambda x:x[0],reverse=True)
cur_time = 0
cur_stage = m
for item in a:
    x,y = item
    cur_time += cur_stage - x
    cur_stage = x
    if cur_time < y:
        cur_time = y
print cur_time + cur_stage
