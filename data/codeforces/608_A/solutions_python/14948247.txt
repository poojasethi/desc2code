n,m=map(int,raw_input().split())
dic={}
while(n):
    n-=1
    f,t=map(int,raw_input().split())
    if f in dic:
        if t>dic[f]:
            dic[f]=t
    else:
        dic[f]=t

dic=sorted(dic.items(),reverse=True)

t=0
for i,j in dic:
    if i==m:
        t=j
        m=i
    elif i<m:
        t+=m-i

        if j>t:
            t+=j-t
        m=i

print t+dic[-1][0]
