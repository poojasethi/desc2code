n, s = map(int, raw_input().split())
A = []
for _ in xrange(n):
	x, y = map(int, raw_input().split())
	A.append([x, y])

sorted(A)
A = A[::-1]

ans = 0
for i in xrange(n):
	ans += s - A[i][0]
	ans += A[i][1] - ans if A[i][1] > ans else 0
	s = A[i][0];

print ans + s
