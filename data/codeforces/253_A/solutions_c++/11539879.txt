#include <fstream>
using namespace std;

int main() {
  ifstream in ("input.txt");
  ofstream out ("output.txt");
  int n, m;
  for (in >> n >> m; n > 0 || m > 0; --n, --m)
    out << (n > 0 && m > 0 ? (n > m ? "BG" : "GB") : (n > 0 ? "B" : "G"));
  out << endl;
  return 0;
}
