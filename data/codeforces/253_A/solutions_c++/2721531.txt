#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int b,g;
	cin >> b >> g;
	int c = b > g;
	while(b || g){
		if(c){
			if(b) {cout<<'B';b--;} else {cout<<'G';g--;}
		} else {
			if(g) {cout<<'G';g--;} else {cout<<'B';b--;}
		}
		c^=1;
	}
	cout<<endl;
}
