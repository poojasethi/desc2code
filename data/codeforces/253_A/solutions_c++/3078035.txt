#include<iostream>
#include<fstream>
using namespace std;
int main()
{	
	ifstream in("input.txt");
	ofstream out("output.txt");
	int g,b;
	in>>b>>g;
	int x=b-g;
	if(x>0)
	{
		for(int i=0;i<g;i++)
			out<<"B"<<"G";
		while(x--)
			out<<"B";
	}
	else
	{
		for(int i=0;i<b;i++)
			out<<"G"<<"B";
		while(x++)
			out<<"G";
	}
	cout<<endl;
	return 0;
}
