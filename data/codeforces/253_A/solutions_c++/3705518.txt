#include <fstream>

using namespace std;


ifstream fin("input.txt");
ofstream fout("output.txt");
int main()
{
	int n,m;
	fin>>n>>m;
	bool t = n<m;
	while (n!=0||m!=0)
	{
		if (n>m || (t&&n==m))
		{
			n--;
			fout<<'B';
		}
		else
		{
			m--;
			fout<<'G';
		}
	}
	fout<<endl;
	fout.close();
	return 0;
}
