#include<stdio.h>
int main()
{
	int n,m,i;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		if(n>m)
		{
			for(i=1;i<=m;i++)printf("BG");
			for(;i<=n;i++)putchar('B');
		}
		else 
		{
			for(i=1;i<=n;i++)printf("GB");
			for(;i<=m;i++)putchar('G');
		}
		printf("\n");
	}
	return 0;
}
