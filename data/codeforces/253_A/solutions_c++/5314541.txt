#include<stdio.h>
#include<stdlib.h>
int main(){
	FILE *f,*p;
	f=fopen("input.txt","r");
	p=fopen("output.txt","w");
	int n,m,i,x=0;
	fscanf(f,"%d%d",&n,&m);
	while(n>m){n--;fprintf(p,"B");x=0;}
	while(m>n){m--;fprintf(p,"G");x=1;}
	for(i=0;i<n;i++){
		if(x==0){
	 	fprintf(p,"GB");
	 }
	 else fprintf(p,"BG");
	 }
}
