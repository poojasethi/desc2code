#include <fstream>
#include <cmath>
using namespace std;

int main()
{
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int a, b;
	cin >> a >> b;
	for (int i = 0; i < min(a,b); i++)
	{
		cout << (a > b ? "BG" : "GB");
	}
	for (int i = min(a,b); i < max(a,b); i++)
	{
		cout << (a > b ? "B" : "G");
	}
	cout << "\n";
}
