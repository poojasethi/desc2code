#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
freopen("input.txt", "r", stdin);
freopen("output.txt", "w", stdout);
int b, g;
scanf("%d%d", &b, &g);
char ca = 'B', cb = 'G';
if (g > b)
{
swap(b, g);
swap(ca, cb);
}
while (b)
{
printf("%c", ca);
b--;
if (g)
{
printf("%c", cb);
g--;
}
}
}
