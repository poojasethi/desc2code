#include<cstdio>
int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	int n, m, st; char c;
	scanf("%d%d", &n, &m);
	if(n > m) st = 0;
	else st = 1;
	for( ; n+m; st ^= 1){
		if(st && m) c = 'G', m--;
		else if(n) c = 'B', n--;
		else c = 'G', m--;
		printf("%c", c);
	}
	puts("");
}
