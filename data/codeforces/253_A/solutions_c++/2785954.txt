#include <stdio.h>

int main ()
{
	int n,m;

	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	scanf("%d%d", &n, &m);

	if (n > m)
	{
		n -= m;
		while (m--) printf("BG");
		while (n--) printf("B");
	}
	else
	{
		m -= n;
		while (n--) printf("GB");
		while (m--) printf("G");
	}
	printf("\n");
	return 0;
}
