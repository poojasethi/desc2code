#include <cstdio>
#include <iostream>
using namespace std;

int n, m;

int main() {
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    cin >> n >> m;
    for(; m > n; m--) cout << "G";
    for(; n && m; n--, m--) cout << "BG";
    for(; n; n--) cout << "B";
    
    return 0;
}
