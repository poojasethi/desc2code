#include<stdio.h>
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,m,i;scanf("%d%d",&n,&m);
	if(n>m)
	{
		for(i=1;i<=m;i++)
			printf("BG");
		for(i=1;i<=n-m;i++)
			printf("B");
	}
	if(m>=n)
	{
		for(i=1;i<=n;i++)
			printf("GB");
		for(i=1;i<=m-n;i++)
			printf("G");
	}
	return 0;
}
