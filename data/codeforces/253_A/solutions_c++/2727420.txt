#include <algorithm>
#include <iostream>
#include <cstdio>

using namespace std;

int n,m;

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	
	cin>>n>>m;
	
	for(int i=0;i<min(n,m);++i)
		cout<<(n>=m ? "BG" : "GB");
	for(int i=0;i<abs(n-m);++i)
		cout<<(n>=m ? "B" : "G");
	cout<<endl;
	
	return 0;
}
