#include<cstdio>

int main(){
	freopen( "input.txt" ,"r",stdin ) ;
	freopen( "output.txt","w",stdout ) ;

	int  n , m ; scanf("%d%d" , &n , &m );
	if( n > m ){
		for( ;m>0; --m , --n)
			printf("BG") ; 
		for( ;n>0;  --n )
			printf("B");
	}else{
		for( ;n>0; --n,--m)
			printf("GB");
		for(;m>0;--m)
			printf("G");
	}
	puts("");
	return 0;
}
