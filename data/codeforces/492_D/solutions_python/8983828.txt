def gcd(a,b):
    while b: a,b = b,a%b
    return a

g = gcd(x,y)
x,y = y/g,x/g
at1 = zip([x*i for i in range(1,y)],[0]*(y-1))
at2 = zip([y*i for i in range(1,x)],[1]*(x-1))
at = [i for d,i in sorted(at1+at2)]+[2,2]
l = len(at)
