from fractions import gcd
n, x, y = map(int, raw_input().split(' '))
g = gcd(x,y)
x = x/g
y = y/g

players = ['Both']

xx, yy = y, x
while len(players) < x+y:
    if xx < yy:
        players.append('Vanya')
        xx = xx + y
    elif xx > yy:
        players.append('Vova')
        yy = yy + x
    else:
        players.append('Both')
        players.append('Both')
        xx = xx + y
        yy = yy + x

for _ in xrange(n):
    print players[int(raw_input())  % (x+y)]
