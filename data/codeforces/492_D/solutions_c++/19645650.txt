#include<stdio.h>
typedef long long ll;
ll n,x,y;

int main()
{
	ll f;
	ll c;
	int i;
	scanf("%lld%lld%lld",&n,&x,&y);
	if(x>y)c=x;
	else c=y;
	for(i=1;i<=n;i++)
	{
		ll m;
		scanf("%lld",&m);
		ll r=c*m;
		ll l=0;
		ll d;
		ll mid;
		while(l<r)
		{
			mid=(l+r)/2;
			d=mid/x+mid/y;
			if(d<m)l=mid+1;
			else r=mid;
		}
		if(l%x==0&&l%y==0)printf("Both\n");
		else if(l%x==0)printf("Vova\n");
		else printf("Vanya\n");
	}
}
