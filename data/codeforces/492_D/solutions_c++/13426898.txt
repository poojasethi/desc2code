#include <bits/stdc++.h>

using namespace std;

const int N = 2e6 + 10;

int n, x, y, k = 1, a[N];
char s[][10] = {"Vanya", "Vova", "Both"};

int main() {
  scanf("%d%d%d", &n, &y, &x);
  for (long long tx = x, ty = y; tx != ty; ++k) {
    if (tx < ty) a[k] = 0, tx += x;
    else a[k] = 1, ty += y;
  }
  a[0] = a[k++] = 2;
  while (n--) {
    scanf("%d", &x);
    printf("%s\n", s[a[x % k]]);
  }
  return 0;
}
