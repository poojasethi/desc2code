#include<bits/stdc++.h>

using namespace std;

long long n,x,y,a,px,py;

int main(){
    cin>>n>>x>>y;
    while(n--){
        cin>>a;
        px = (a+1)*x/(x+y)-a*x/(x+y);
        py = (a+1)*y/(x+y)-a*y/(x+y);
       // cout<<px<<"-"<<endl<<py<<"-="<<endl;
        if(px==py){
            cout<<"Both"<<endl;
        }else if(px>py){
            cout<<"Vanya"<<endl;
        }else{
            cout<<"Vova"<<endl;
        }
    }
}
