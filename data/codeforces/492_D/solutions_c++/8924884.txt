#include<cstdio>
int main()
{
 int n;
 long long x,y,num1,num2,den1,den2,a,n1,n2;
 scanf("%d %lld %lld",&n,&x,&y);
 while(n--)
 {
  scanf("%lld",&a);
  num1=(a*x)%(x+y);
  num2=(a*y)%(x+y);
  if(num1==0&&num2==0)
  {
   printf("Both\n");
   continue;
  }
  num1=(x+y)-num1;
  num2=(x+y)-num2;
  den1=(x+y)*x;
  den2=(x+y)*y;
  n1=num1*den2;
  n2=num2*den1;
  if(n1>n2)
   printf("Vova\n");
  else if(n1<n2)
   printf("Vanya\n");
  else
   printf("Both\n");
 }
}
