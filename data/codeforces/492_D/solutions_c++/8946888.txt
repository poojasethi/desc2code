#include<cstdio>
int main(){
	long long x,y;
	int n;
	scanf("%d%lld%lld", &n, &x, &y);
	for(int i=1; i<=n; i++){
		long long k;
		scanf("%lld", &k);
		long long pocz=0, kon=k*x;
		while(pocz<kon){
			long long s=(pocz+kon)/2;		
			if(s/x+s/y<k)
				pocz=s+1;
			else
				kon=s;
		}
		if(pocz%x==0 && pocz%y==0)
			printf("Both\n");
		else{
			if(pocz%x==0)
				printf("Vova\n");
			if(pocz%y==0)
				printf("Vanya\n");
			
		}
	}


}
