#include <stdio.h>

int main()
{
    int n,x,y,a;
    scanf("%d %d %d",&n,&x,&y);
    while(n--)
    {
        scanf("%d",&a);
        long long l=0,r=1e15,mid;
        while(l<r)
        {
            mid=(l+r)>>1;
            if(mid/x+mid/y<a) l=mid+1;
            else r=mid;
        }
        if(l%x==0 && l%y==0) printf("Both\n");
        else if(l%x==0) printf("Vova\n");
        else printf("Vanya\n");
    }
    return 0;
}
