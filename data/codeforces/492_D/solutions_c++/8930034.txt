#include <iostream>
#include <algorithm>
using namespace std;

long long q,x,y,n,l,r,mid,m;

main() {
 cin>>q>>x>>y;
 while (q--) {
   cin>>n;
   l=0; r=10000000000000000LL;
   mid=-1;
   while (l <= r) {
     m=(l+r)/2;
     if ((m/x)+(m/y) >= n)
      {r=m-1; mid=m;} else l=m+1;
    }
   if (mid%x == 0 && mid%y == 0)
    cout<<"Both"<<endl; else
   if (mid%y == 0)
    cout<<"Vanya"<<endl; else
    cout<<"Vova"<<endl;
 }
}
