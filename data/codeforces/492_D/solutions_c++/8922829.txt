#include<bits/stdc++.h>

int f[3000000];

int main()
{
	int t,i,j,k,n,p,u,x,y;
	long long a,b,d,h,g;
	scanf("%d%d%d",&n,&x,&y);
	a=(long long)(y);
	b=(long long)(x);
	h=a;
	g=b;
	i=1;
	while(h!=g) 
	{
		if(h<g)
		{
			f[i]=0;
			h+=a;
		}
		else
		{
			f[i]=1;
			g+=b;
		}
		i++;
	}
//	printf("%d\n",i);
	f[i++]=2;
	f[0]=2;
	while(n--)
	{
		scanf("%d",&t);
		t=t%i;
		if(f[t]==0)
			printf("Vanya\n");
		else if(f[t]==1)
			printf("Vova\n");	
		else
			printf("Both\n");
	}
	return 0;
}
