#include <cstdio>


int main()
{
    long long n,x,y,a;
    long long t1,t2;
    scanf("%I64d%I64d%I64d",&n,&x,&y);
    for(int i=0; i<n; ++i)
    {
       scanf("%I64d",&a);
       t1 = a*x/(x+y);
       t2 = (a+1)*x/(x+y);
       if( a*x%(x+y) == 0 || t1 < t2)
       {
           if(t2*y%x==0)
            puts("Both");
           else
            puts("Vanya");
       }
       else
            puts("Vova");
    }
}
