#include<stdio.h>
typedef long long ll;
int main()
{
	int n;
	ll a,b;
	scanf("%d%lld%lld",&n,&a,&b);
	for(int i=0;i<n;i++){
		ll temp;
		scanf("%lld",&temp);
		ll l=0,r=1LL<<60;
		while(l<=r){
			ll m=(l+r)/2;
			if(m/a+m/b<temp){
				l=m+1;
			}
			else r=m-1;
		}
		if(l%a==0&&l%b==0)puts("Both");
		else if(l%a==0)puts("Vova");
		else puts("Vanya");
	}
	return 0;
}
