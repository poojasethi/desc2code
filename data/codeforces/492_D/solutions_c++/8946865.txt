#include <iostream>

using namespace std;

typedef pair<int,int> Pair;

int main() {
  long long n, x, y, a;
  cin >> n >> x >> y;
  for (int i=0; i<n; i++) {
    cin >> a;
    long long hx = a*x/(x+y),  hy = a*y/(x+y);
    while (hx+hy<a)
      if ((hx+1)*y<(hy+1)*x) hx++;
      else if ((hx+1)*y>(hy+1)*x) hy++;
      else hx++, hy++;
    cout << (hx*y==hy*x ? "Both" : (hx*y>hy*x ? "Vanya" : "Vova")) << endl;
  };
  return 0;
};
