#include <bits/stdc++.h>

using namespace std;

int main(){
	int n, x, y;
	cin>>n>>x>>y;
	for(int i=0;i<n;i++){
		int a;
		cin>>a;
		long long xa = (long long)x*a;
		long long ya = (long long)y*a;
		long long tempx = (xa%(x+y)==0)?((xa/(x+y))*y):(((xa/(x+y)) + 1)*y);
		long long tempy = (ya%(x+y)==0)?((ya/(x+y))*x):(((ya/(x+y)) + 1)*x);
		if(tempx==tempy){
			cout<<"Both"<<endl;
		}
		else if(tempx>tempy){
			cout<<"Vova"<<endl;
		}
		else{
			cout<<"Vanya"<<endl;
		}
	}
}
