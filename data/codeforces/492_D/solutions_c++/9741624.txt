#include<iostream>
using namespace std;
int main(){
int64_t n,x,y,t,i;
cin>>n>>x>>y;
for(i=0;i<n;i++){
    cin>>t;
    int64_t l=0,r=1E15,m;
    while(l<=r){
        m=(l+r)/2;
        if(m/x+m/y>=t)
            r=m-1;
        else
            l=m+1;
    }
    if((r+1)%x==0&&(r+1)%y==0)
        cout<<"Both\n";
    else if((r+1)%y==0)
        cout<<"Vanya\n";
    else
        cout<<"Vova\n";
}
return 0;
}
