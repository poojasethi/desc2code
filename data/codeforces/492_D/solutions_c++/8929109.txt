#include<iostream>
using namespace std;
#define lld long long int
lld d,x,y;
lld bin(lld i,lld j)
{
    if(i==j)return i;
    lld mid=(i+j)/2;
    if(mid/x+mid/y<d)return bin(mid+1,j);
    return bin(i,mid);
}
int main()
{
    int n;
    cin>>n>>y>>x;
    while(n--)
    {
        cin>>d;
        lld t=bin(0,1000000000000000000);
        if(t%x==0 and t%y==0)cout<<"Both\n";
        else if(t%y==0)cout<<"Vova\n";
        else cout<<"Vanya\n";
    }
}
