#include<stdio.h>
#include<math.h>
#include<string.h>
#include<algorithm>
using namespace std;
long long int n,x,y,a,l,r,m;
int main()
{
	scanf("%lld%lld%lld",&n,&x,&y);
	while(n--){
		scanf("%lld",&a);
		l=0;r=1LL<<60;
		while(l<=r){
			m=(l+r)/2;
			if(m/x+m/y>=a)r=m-1;  
            else l=m+1; 
            //printf("%lld\n",m); 
		}
		if(l%x==0&&l%y==0)printf("Both\n");
		else if(l%y==0)printf("Vanya\n");
		else printf("Vova\n");
	}
	return 0;
}
