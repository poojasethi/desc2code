#include<cstdio>
#include<cmath>
typedef long long ll;
ll n,x,y,a,l,r,mid,tim;
int main(){
	scanf("%lld%lld%lld",&n,&x,&y);
	for(int i=0;i<n;i++){
		l=0;
		r=1e15;
		scanf("%d",&a);
		while(l<=r){
			mid=(l+r)>>1;
			if(mid/x+mid/y>=a){
				tim=mid;
				r=mid-1;
			}
			else	l=mid+1;
		}
		if(tim%x==0&&tim%y==0)	printf("Both\n");
		else if(tim%x==0)	printf("Vova\n");
		else printf("Vanya\n");
	}
	return 0;
}
