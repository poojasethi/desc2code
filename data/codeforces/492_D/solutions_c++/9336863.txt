#include <cstdio>


int main()
{
    long long n,x,y,a;
    long long t1,t2;
    scanf("%I64d%I64d%I64d",&n,&x,&y);
    for(int i=0; i<n; ++i)
    {
       scanf("%I64d",&a);
       t1 = a*x;
       t2 = (a+1)*x;
       if( t1%(x+y)==0 || t2%(x+y)==0)
       {
            puts("Both");
       }
       else if(t1/(x+y) < t2/(x+y))
            puts("Vanya");
       else
            puts("Vova");
    }
}
