#include<bits/stdc++.h>
using namespace std;
int n,x,y,s[1000005],mon;
int main(){
  scanf("%d %d %d",&n,&x,&y);
  for(int i=0;i<=x;i++) s[i]=1LL*i*y/x+i;
  while(n--){
    scanf("%d",&mon);
    mon%=(x+y);
    int l=0,r=x;
    while(r-l>1){
      int m=(l+r)>>1;
      if(s[m]>=mon) r=m;
      else l=m;
    }
    if(mon==0 || (1LL*r*y%x==0 && s[r]-mon<=1)) printf("Both\n");
    else if(s[r]>mon) printf("Vova\n");
    else printf("Vanya\n");
  }
  return 0;
}
