#include<bits/stdc++.h>
using namespace std;
long long n,x,y,a,dx,dy,l,k;
string s[2000005];
int g(int p, int q) {
	if (q==0) return p;
	g(q, p%q);
}

int main() {
	cin>> n>> x>> y;
	l = x*y / g(x,y);
	x = dx = l/x;
	y = dy = l/y;
	for (k=1; x!=y; k++) {
		if (x < y) {
			s[k] = "Vanya";
			x += dx;
		} else {
			s[k] = "Vova";
			y += dy;
		}
	}
	s[0] = s[k] = "Both";
	k++;
	while (n--) {
		cin>> a;
		cout<< s[a%k]<< endl;
	}
}
