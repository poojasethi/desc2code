#include<bits\stdc++.h>
using namespace std;
#define LL long long
int main()
{
	int n;
	LL x,y;
	scanf("%d%lld%lld",&n,&x,&y);
	for(int i = 0;i<n;i++)
	{
		LL temp;
        scanf("%lld",&temp);
		LL l = 0,r=1LL<<60;
		while(l<=r)
		{
			LL m = (l+r)>>1;
			if(m/x+m/y>=temp)
				r=m-1;
			else
				l=m+1;
		}
		if(l%x==0 && l%y==0)
			puts("Both");
		else if (l%y==0)
			puts("Vanya");
		else
			puts("Vova");
	}
}
