#include <iostream>
using namespace std;
typedef long long ll;
int main(){
	ll n,x,y;
	cin >> n >> x >> y;
	for(ll i = 0; i < n; ++i){
		ll hp;
		cin >> hp;
		ll lo = 1, hi = 2000000000000000000LL, mid;
		while(lo<hi){
			mid=(lo+hi)/2;
			if(mid/x+mid/y>=hp)
				hi=mid;
			else
				lo=mid+1;
		}
		lo-=lo%x;
		hi-=hi%y;
		if(lo<hi)
			cout << "Vanya\n";
		if(lo>hi)
			cout << "Vova\n";
		if(lo==hi)
			cout << "Both\n";
	}
}
