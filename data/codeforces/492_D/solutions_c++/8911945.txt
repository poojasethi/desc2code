#include<cstdio>

using namespace std;

#define int long long
#define INF (0x7FFFFFFFFFFFFFFFLL)
#define LLD "%I64d"

main(){
	int n,x,y;scanf(LLD""LLD""LLD,&n,&x,&y);
	for(int i=1;i<=n;i++){
		int a;scanf(LLD,&a);
		int L=0,R=INF;
		while(L<R){
			int M=(L+R)>>1;
			if(M/y+M/x<a){
				L=M+1;
			}else{
				R=M;
			}
		}
		//cout<<L<<endl;
		int p1=!(L%y);
		int p2=!(L%x);
		if(p1&&p2){puts("Both");continue;}
		if(p1){puts("Vanya");continue;}
		puts("Vova");
	}
	return 0;
}
