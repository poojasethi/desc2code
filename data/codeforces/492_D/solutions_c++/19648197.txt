#include<string.h>
#include<math.h>

#define ll long long
int main()
{
	ll n,x,y,i,j,a;
	scanf("%lld%lld%lld",&n,&x,&y);
	while(n--){
		scanf("%lld",&a);
		ll l=0,r=1000000000000000;
		while(l<=r){
		
			ll mid=(l+r)/2;	
			
			if(mid/x+mid/y>=a)r=mid-1;
			else l=mid+1;
		} 
	//	printf("%I64d\n",l);
		if(l%x==0&&l%y==0)printf("Both\n");
		else if(l%y==0)printf("Vanya\n");

	}
}
