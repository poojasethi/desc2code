#include<iostream>  
#include<cstdio>  
#include<cstring> 
#include <cmath>   
#include<algorithm>  
using namespace std;

typedef long long LL;
LL i,n,x,y,l,r,m,atk,a;

void bsearch(LL a)
{
	l=1,r=1e15;
	while(l<r){
		m=(l+r)/2;
		atk=m/y+m/x;
		if(atk>=a) r=m;
		else l=m+1;
	}
	if(l%x==0&&l%y==0) printf("Both\n");
	else if(l%x==0) printf("Vova\n");
	else printf("Vanya\n");
}

int main()
{
	cin>>n>>x>>y;
	for(i=1;i<=n;i++){
		cin>>a;
		bsearch(a);
	}
	return 0;	
}
