#include <cstdio>
int n,x,y,z,l=1,r=1,a[2000000];
main()
{
    scanf("%d%d%d",&n,&x,&y);
    for(int i=1;i<x+y;i++)
        if(y*1ll*l>r*1ll*x)a[i]=1,r++;else
        if(y*1ll*l<r*1ll*x)a[i]=2,l++;else i++,l++,r++;
    while(n--)
    {
        scanf("%d",&z);
        z%=x+y;
        if(!a[z]) puts("Both\n");else
        puts(a[z]==1?"Vova\n":"Vanya\n");
    }
}
