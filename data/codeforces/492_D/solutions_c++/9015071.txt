#include <bits/stdc++.h>
using namespace std;
long long n, x, y, dx, dy, k, t, a[2000005];
char s[3][9] = {"Vanya", "Vova", "Both"};

int main() {
//    freopen("input.txt", "r", stdin);
    cin >> n >> y >> x;
    dx = x, dy = y;
    for (k = 1; x != y; k++) {
        if (x < y) {
            a[k] = 0;
            x += dx;
        } else {
            a[k] = 1;
            y += dy;
        }
    }
    a[0] = a[k++] = 2;
    while (n--) {
        scanf("%d", &t);
        printf("%s\n", s[a[t % k]]);
    }
}
