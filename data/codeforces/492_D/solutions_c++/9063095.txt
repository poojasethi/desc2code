#include  <iostream>
#include  <cstdio>
using namespace std;
long long int n,x,y,a,c,d;
int main ()
{
    cin>>n>>x>>y;
    for (int i=0;i<n;i++)
    {
        cin>>a;
        c=(a+1)*x/(x+y)-a*x/(x+y);
        d=(a+1)*y/(x+y)-a*y/(x+y);
        if (c && d)
            cout<<"Both";
        else if (c)
            cout<<"Vanya";
        else if (d)
            cout<<"Vova";
        else
            cout<<"Both";
        cout<<endl;
    }
    return 0;

}
