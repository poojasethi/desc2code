#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

int n,A,B,nd;
LL l,r,mid;

inline bool check(LL x){return x/A+x/B>=nd;}

int main()
{
	scanf("%d%d%d",&n,&A,&B);
	swap(A,B);
	for (int i=1;i<=n;++i)
	{
		scanf("%d",&nd);
		nd%=A+B;
		for (l=0,r=(LL)nd*A*B;l+1<r;)
			if (check(mid=l+r>>1)) r=mid;else l=mid;
		if (r%A) puts("Vova");else
		if (r%B) puts("Vanya");else
		puts("Both");
	}
	return 0;
}
