#include <iostream>

using namespace std;

int main()
{
	long long n, x, y;
	cin >> n >> x >> y;
	for (int i=0; i<n; ++i)
	{
		long long hits;
		cin >> hits;
		hits=hits%(x+y);
		long long l=hits*x*y/(x+y), r=(hits+1)*x*y/(x+y);
		while (l<r)
		{
			long long mid=(l+r)/2;
			if (mid/x+mid/y>=hits) r=mid;
				else l=mid+1;
		}
		if ((r%y==0)&&(r%x==0))	cout << "Both" << endl;
			else if (r%y==0) cout << "Vanya" << endl;
				else cout << "Vova" << endl;
	}
}
