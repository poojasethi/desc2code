#include <iostream>

using namespace std;

int solve(int64_t a, int64_t x, int64_t y){
int64_t f, l, m, d;
   f=0; l=a*min(x,y);
   while (f<l){
      m=(f+l)/2;
      d=(m/y)+(m/x);
      if (d>=a) l=m;
      else f=m+1;
   }
   if (f%y==0 && f%x==0) cout<<"Both"<<endl;
   else
      if (f%y==0) cout<<"Vanya"<<endl;
      else cout<<"Vova"<<endl;
}

int main(){
int64_t t, n, x, y;
   cin>>n>>x>>y;
   for (int i=1; i<=n; ++i){
      cin>>t;
      solve(t,x,y);
   }
}
