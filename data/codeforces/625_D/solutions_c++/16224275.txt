#include <cstring>

const int Maxn = 100005;
int digit[Maxn], len, ans[Maxn];
int check(){
	for (int i = 0; i < len/2;){
		int l = i, r = len-i-1;
		if (digit[l] == digit[r]) ++i;
		else if (digit[l] == digit[r]+1 || digit[l] == digit[r]+11){
			digit[l] -= 1;
			digit[l+1] += 10;
		}
		else if (digit[l] == digit[r]+10){
			digit[r-1]--;
			digit[r] += 10;	
		}
		else return 0;
	}
	if (len%2 == 1){
		if (digit[len/2] % 2 == 1 || digit[len/2] > 18 || digit[len/2] < 0) return 0;
		ans[len/2] = digit[len/2]/2;
	}
	for (int i = 0; i < len/2; ++i){
		if (digit[i] > 18 || digit[i] < 0) return 0;
		ans[i] = (digit[i]+1)/2;
		ans[len-i-1] = digit[i]/2;
	}
	return ans[0]>0;
}
int main(){
#ifndef ONLINE_JUDGE
	freopen ("D.in", "r", stdin);
#endif
	char s[Maxn];
	scanf ("%s", s);
	len = strlen(s);
	for (int i = 0; i < len; ++i)
		digit[i] = s[i]-'0';
	if (check()){
		for (int i = 0; i < len; ++i) 
			printf ("%d", ans[i]);
		return 0;
	}
	if (s[0] == '1' && len > 1){
		for (int i = 0; i < len; ++i)
			digit[i] = s[i+1]-'0';
		digit[0] += 10;
		len--;
		if (check())
			for (int i = 0; i < len; ++i)
				printf ("%d", ans[i]);
		else printf ("0");
		return 0;
	}
	printf ("0");
	return 0;
}
