#include <bits/stdc++.h>
using namespace std;
const int MAX = 1111111;
string s;
int ex[MAX];
bool givstring(int l) {
    int r = s.length() - 1, x = 0, y = r - l;
    string a(r - l + 1, '.');
    bool yes = 1;
    while (x <= y) {
        int rs = s[r] - 48 + ex[r];
        if (l == r) {
            if (rs < 0 || rs % 2 || rs > 18) return 0;
            a[x] = '0' + rs / 2;
        } else {
            if (rs < 0) ex[r - 1] -= 1, rs += 10;
            int ls = s[l] - 48 + ex[l];
            if (ls < rs) return 0;
            if (rs + 10 <= ls) rs += 10, ex[r - 1] -= 1;
            ls = s[l] - 48 + ex[l];
            if (rs > ls) return 0;
            if (rs > 18) return 0;
            a[x] = '0' + (rs + 1) / 2;
            a[y] = '0' + rs / 2;
            int d = ls - rs;
            if (d > 0 && l + 1 == r) return 0;
            ex[l + 1] += d * 10;
        }
        x++, y--, l++, r--;
    }
    if (a[0] == '0') return 0;
    cout << a << endl;
    return 1;
}
int main(int argc, char const *argv[]) {
    cin >> s;
    memset(ex, 0, sizeof ex);
    if (givstring(0)) return 0;
    if (s.length() > 1) {
        memset(ex, 0, sizeof ex);
        ex[1] += 10 * (s[0] - 48);
        if (givstring(1)) return 0;
    }
    printf("0\n");
	return 0;
}
