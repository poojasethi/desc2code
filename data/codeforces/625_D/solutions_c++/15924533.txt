char a[100010],ans[100010];
int sum[100010];
int check(int n)
{   int i;
    for(i=0;i<n/2;){
        if(sum[i]==sum[n-1-i])i++;
        else if(sum[i]==sum[n-1-i]+1){
            sum[i]--;sum[i+1]+=10;
        }else if(sum[i]==sum[n-1-i]+10){
            sum[n-1-i]+=10;sum[n-2-i]--;
        }else if(sum[i]==sum[n-1-i]+11){
            sum[i]--;sum[i+1]+=10;
        }else return 0;
    }
    if(n%2){
        if(sum[n/2]%2||sum[n/2]/2>=10)return 0;else ans[n/2]=sum[n/2]/2+'0';
    }
    for(i=0;i<=n/2;i++)if(sum[i]<0||sum[i]>18)return 0;
    for(i=0;i<n/2;i++){
        ans[n-1-i]=sum[i]/2+'0';ans[i]=sum[i]-ans[n-1-i]+2*'0';
    }ans[n]=0;
    return ans[0]>'0';
}
int main()
{
    int i,j,m,n,k;
    gets(a);n=strlen(a);
    for(i=0;i<n;i++)sum[i]=a[i]-'0';
    if(check(n))puts(ans);
    else if(a[0]=='1'){
        for(i=0;i<n-1;i++)sum[i]=a[i+1]-'0';n--;sum[0]+=10;
        if(check(n))puts(ans);else puts("0");
    }else puts("0");
    //system ("pause");
    return 0;
}
