#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <iostream>
using namespace std;
const int maxn = 1e5 + 5;
char s[maxn];
int num[maxn];
int ans[maxn];
int n;

bool solve(){
	int i = 0, j = n - 1;
	for(i = 0, j = n-1; i < j; ++ i, -- j){
		if((num[i]+10)%10 != (num[j]+10)%10){
			if(num[i]){
                num[i] --;
                num[i+1] += 10;
			}
			if((num[i]+10)%10 != (num[j]+10)%10) return false;
		}
		if(num[i] != num[j]){
            if(num[j] != 9){
                num[j] += 10;
                num[j-1] --;
            }
			if(num[i] != num[j]) return false;
		}
		if(num[i]>9){
			ans[i] = 9;
			ans[j] = num[i]-ans[i];
		}else{
			ans[i] = num[i];
			ans[j] = num[i]-ans[i];
		}
	}
	if(i == j){
        if(num[i] < 0 || num[i]&1){
            return false;
        }else{
            ans[i] = num[i]>>1;        }
	}
	if(num[0] <= 0) return false;
	return true;
}

int main(){
	while(~scanf("%s", s)){
		n = strlen(s);
		for(int i = 0; i < n; ++ i){
			num[i] = s[i]-'0';
		}
		bool flag = false;
		flag = solve();
		if(!flag && n > 2){
			n --;
			num[0] = (s[0]-'0')*10+s[1]-'0';
			for(int i = 1; i < n; ++ i){
				num[i] = s[i+1]-'0';
			}
			flag = solve();
		}
		if(!flag){
			printf("0\n");
		}else{
			for(int i = 0; i < n; ++ i){
				printf("%d", ans[i]);
			}
			puts("");
		}
	}
	return 0;
}
