#include <iostream>
#include <stdio.h>
#include <string>
#define MAXN 100000

using namespace std;

string s;
int a, b, f, i, j, n, u, v;
int ans[MAXN];


bool check(int i, int p1, int j, int p2)
{

//	cout << i <<j <<p1 << p2 << endl;
	if (i > j) return p1 == p2;

	if (i == j)
	{
	 	for (u = 9; u >= 0; u--)
		if ( ((2 * u + p2) % 10) + '0' ==  s[i] && (2 * u + p2) / 10 == p1)
		{
			ans[i] = u;
			return true;

		}
		return false;
	}

	for (u = 9;  u >= 0; u--)
	for (v = 0;  v <=9; v++)
	{
		if ( (u + v + p2) % 10 + '0' == s[j])
		{
			if ((((u + v) % 10) + '0' == s[i]) &&  (u + v) / 10 == p1)
			{
				ans[i] = u;
				ans[j] = v;
				return check(i + 1, 0, j - 1, (u + v + p2) / 10);
			} else

			if (((u + v + 1) % 10) + '0' == s[i] &&  (u + v + 1) / 10 == p1)
			{
				ans[i] = u;
				ans[j] = v ;
				return check(i + 1, 1, j - 1, (u + v + p2) / 10);
			} 
		}
	}
	return false;
		

}

int main()
{
	cin >> s;
	n = s.size();

	if (check(0, 0, n -1, 0))
	{
		if (ans[0] != 0){
		for (i = 0; i < n; i++) 
			cout << ans[i];
			return 0;
		}
	}
	if ((s[0] == '1') && check(1, 1, n - 1, 0))
	{
		if (ans[1] == 0) cout << 0;
		else
		for (i = 1; i < n; i++)
			cout << ans[i];
	}
	else    cout << 0;
	
return 0;
}
