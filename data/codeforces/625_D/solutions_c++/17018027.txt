#include<cstdio>  
#include<iostream>  
#include<algorithm>  
#include<cstdlib>  
#include<cstring>
#include<string>
#include<climits>
#include<cmath>

using namespace std;

int n,m,a[100010],ans[100010];
char A[100010];

int main()
{
	scanf("%s",A);
	n=strlen(A);
	for (int i=0;i<n;i++)
		a[i+1]=int(A[i])-48;
	int l=1,r=n,flag=0;
    if (a[l]!=a[r]) a[l]--,a[l+1]+=10;
    if (a[l]==0) l++;
    while (l<=r)
    {
        if (a[l]!=a[r])
        {
            if (a[l]-a[r]>=10 && a[r]<10) a[r-1]--,a[r]+=10;
            if (a[l]-a[r]==1) a[l]--,a[l+1]+=10;
        }
        if (a[l]!=a[r]) {flag=1; break;}
        if (l!=r) ans[l]=a[l]-a[r]/2,ans[r]=a[r]/2;
        else
        {
        	if (a[l]%2==0) ans[l]=a[l]/2;
        	else {flag=1; break;}
    	}
        if (ans[l]>9 || ans[l]<0 || ans[r]>9 || ans[r]<0) {flag=1; break;}
        l++; r--;
    }
	if (flag==1) {cout<<0<<endl; return 0;}
	int i=1;
	while (ans[i]==0) i++;
	for (i;i<=n;i++)
		cout<<ans[i];
	cout<<endl;
	return 0;
}
