#include<bits/stdc++.h>
using namespace std;

#define scl(x) scanf("%lld",&x)
#define sc(x)  scanf("%d",&x)
#define ll long long
#define lop(i,n) for(int i=0;i<n;++i)
typedef pair<int,int> ii;
typedef pair<int,int> pll;

int memo[100005][2][2];
int n;
string s,res;
int solve(int idx,int gc,int oyc,int oidx,int f){
	if(oidx==idx){
		for(int i=0+f;i<=9;++i){
			int v=2*i+gc;
			if(v%10==s[idx]-'0'&&oyc==v/10){
				res[idx]=i+'0';
				return 1;
			}
		}
		return 0;
	}
	if(oidx+1==idx){
		for(int i=0+f;i<=9;++i)
			for(int j=0;j<=9;++j){
				int vr=i+j+gc,vl=i+j+vr/10;
				if(vr%10==s[idx]-'0'&&vl%10==s[oidx]-'0'&&
						vl/10==oyc){
					res[oidx]=i+'0';
					res[idx]=j+'0';
					return 1;
				}
			}
		return 0;
	}

	int &ret=memo[idx][gc][oyc];
	if(~ret)return ret;
	ret=0;
	int v=s[idx]-'0';
	for(int i=0+f;i<=9;++i){
		bool ok=0;
		int j=(v-i-gc+20)%10;
		if(!oyc){
			if(i+j>9)continue;
			if(i+j==s[oidx]-'0'){
				if(solve(idx-1,i+j+gc>9,0,oidx+1,0))ok=1;
			}
			else if(i+j==s[oidx]-'0'-1)
				if(solve(idx-1,0,1,oidx+1,0))ok=1;
		}
		else {
			if((i+j)%10==s[oidx]-'0'){
				if(i+j<10)continue;
				if(solve(idx-1,1,0,oidx+1,0))ok=1;
			}
			if((i+j)%10==(s[oidx]-'0'-1+10)%10){
				if(i+j+1<10)continue;
				if(solve(idx-1,i+j+gc>=10,1,oidx+1,0))ok=1;
			}
		}
		if(ok){
			res[idx]=j+'0';
			res[oidx]=i+'0';
			return 1;
		}
	}
	return 0;
}
int main(){
#ifndef ONLINE_JUDGE

#endif
	memset(memo,-1,sizeof memo);
	cin>>s;res.resize(s.size());
	if(solve(s.size()-1,0,0,0,1)){
		cout<<res;
		return 0;
	}
	memset(memo,-1,sizeof memo);
	if(s[0]=='1'&&int(s.size())>1&&solve(int(s.size())-1,0,1,1,1)){
		cout<<res.substr(1);
		return 0;
	}
	cout<<0;
}
