#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
const int MAX = 100010;
char n[MAX];
int sum[MAX],a[MAX];
int l;
int judge()
{
    for (int i = 0 ; i < l/2 ;)
        if (sum[i] == sum[l-1-i])   i++;
        else if (sum[i] == sum[l-1-i]+1 || sum[i] == sum[l-1-i]+11)
        {
            sum[i]--;
            sum[i+1] += 10;
        }
        else if (sum[i] == sum[l-1-i]+10)
        {
            sum[l-1-i-1]--;
            sum[l-1-i] += 10;
        }
        else
        {
//            printf("false->\n");
//            for (int i = 0 ; i < l ; i++)   printf("%d ",sum[i]);
//            printf("\n");
            return 0;
        }
//    printf("true->\n");
//    for (int i = 0 ; i < l ; i++)   printf("%d ",sum[i]);
//    printf("\n");
    if (l % 2 == 1)
    {
        if (sum[l/2] % 2 == 1 || sum[l/2] > 18 || sum[l/2] < 0) return 0;
        a[l/2] = sum[l/2]/2;
    }
    for (int i = 0 ; i < l/2 ; i++)
        if (sum[i] < 0 || sum[i] > 18)  return 0;
        else
        {
            a[i] = (sum[i]+1)/2;
            a[l-1-i] = sum[i]/2;
        }
    if (a[0] > 0)   return 1;
    else    return 0;
}
int main()
{
    scanf("%s",n);
    l = strlen(n);
    for (int i = 0 ; i < l ; i++)   sum[i] = n[i]-'0';
    if (judge())
    {
        for (int i = 0 ; i < l ; i++)  printf("%d",a[i]);
        printf("\n");
    }
    else if (n[0] == '1' && l > 1)
    {
        for (int i = 1 ; i < l ; i++)   sum[i-1] = n[i]-'0';
        sum[0] += 10;
        l--;
        if (judge())
        {
            for (int i = 0 ; i < l ; i++)  printf("%d",a[i]);
            printf("\n");
        }
        else printf("0\n");
    }
    else    printf("0\n");
    return 0;
}
