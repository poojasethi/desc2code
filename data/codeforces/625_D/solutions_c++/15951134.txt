#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[100005];
int n;
int a[100005];
int b[100005];
void fu()
{
	for(int i=1;i<=n;i++)
	{
		a[i]=s[i]-'0';
	}
}
void print(int x)
{
	if(b[x]==0)
	{
		puts("0");
		return;
	}
	for(int i=x;i<=n;i++)
	{
		if(b[i]>9)
		{
			puts("0");
			return;
		}
	}
	for(int i=x;i<=n;i++)
	{
		printf("%d",b[i]);
	}
	puts("");
}
bool check(int l,int r,bool fff)
{
	if(a[r]<0)
	{
		a[r-1]--;
		a[r]+=10;
	}
	if(l==r)
	{
		if(a[l]%2==0)
		{
			if(fff)a[l]+=10;
			b[l]=a[l]/2;
			return 1;
		}
		return 0;
	}
	if(l+1==r)
	{
		if(fff)
		{
			a[l]+=10;
			if(a[l]!=a[r])
			{
				a[l]--;
				a[r]+=10;
			}
		}
		if(a[l]==a[r])
		{
			b[r]=a[l]/2;
			b[l]=a[l]-b[r];
			return 1;
		}
		return 0;
	}
	if(fff)
	{
		a[l]+=10;
		if(a[l]!=10 || a[r]!=9)
		{
			a[r]+=10;
			a[r-1]--;
		}
	}
	if(a[l]==a[r])
	{
		b[r]=a[l]/2;
		b[l]=a[l]-b[r];
		return check(l+1,r-1,0);
	}
	else if(a[l]==a[r]+1)
	{
		a[l]--;
		b[r]=a[r]/2;
		b[l]=a[l]-b[r];
		return check(l+1,r-1,1);
	}
	return 0;
}
int main()
{
	scanf("%s",s+1);
	n=strlen(s+1);
	fu();
	if(s[1]=='1')
	{
		if(n==1)
		{
			puts("0");
			return 0;
		}
		if(check(2,n,1))
		{
			print(2);
			return 0;
		}
		fu();
		if(check(1,n,0))
		{
			print(1);
			return 0;
		}
		puts("0");
		return 0;
	}
	else
	{
		if(check(1,n,0))
		{
			print(1);
			return 0;
		}
	}
	puts("0");
	return 0;
}
