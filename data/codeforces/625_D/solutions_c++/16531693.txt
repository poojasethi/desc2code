#include <bits/stdc++.h>
using namespace std;

#define inf 0x3f3f3f3f
#define LL long long
#define ls i << 1
#define rs ls | 1
#define md (ll + rr >> 1)
#define lson ll, md, ls
#define rson md + 1, rr, rs
#define mod 10001
#define N 500010
#define M 1000020

char s[N];
int a[N], ans[N];
bool check(int l, int r, int lx, int rx){
	if(l > r) return lx == rx;
	a[l] += lx * 10, a[r] -= rx;
	lx = rx = 0;
	if(a[l] >= a[r] + 10)
		a[r] += 10, rx = 1;
	if(a[l] > a[r])
		a[l] --, lx = 1;
	if(a[l] != a[r]) return 0;
	ans[l] = (a[l] + 1) / 2;
	ans[r] = a[l] / 2;
	if(ans[l] == 10) return 0;
	if(ans[l] + ans[r] != a[l]) return 0;
	return check(l + 1, r - 1, lx, rx);
}
int main(){
	scanf("%s", s + 1);
	int n = strlen(s + 1);
	for(int i = 1; i <= n; ++i) a[i] = s[i] - '0';
	int l = 1, r = n, lx = 0, rx = 0;
	if(a[l] == 1 && a[n] != a[l]) l++, lx = 1;
	if(check(l, r, lx, rx)){
		for(int i = l; i <= r; ++i)
			printf("%d", ans[i]);
		puts("");
	}
	else puts("0");
	return 0;
}
