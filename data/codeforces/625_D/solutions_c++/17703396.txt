#include <cstdio>
#include <cstring>
#define min(a,b) (a>b?b:a)
#define MAXN 100009

int a[MAXN], s[MAXN], n, useFirst;

bool backtrack(int j, int cout, int cin){
	if (cin > 1 || cin < 0 || cout > 1 || cout < 0) return false;
	int i = n-useFirst-j, ncin=2, ncout, v;
	if (i==j){
		v = 10*cout + s[i] - cin;
		if (v%2!=0) return false;
		a[i]=v/2;
		//printf("a[%d]=%d\n", i, a[i], j, a[j]);
		return true;
	}
	if (j+1==i){
		return cin == cout;
	}
	while(ncin-->0){
		v = 10*ncin + s[j] - cin;
		a[i] = min(9, v);
		a[j] = v - a[i];
		if (a[i]>9 || a[j]>9 || a[i]<0 || a[j]<0) continue;
		//printf("a[%d]=%d, a[%d]=%d\n", i, a[i], j, a[j]);
		ncout = 10*cout + s[i] - v;
		if (backtrack(j-1, ncout, ncin)) return true;
	}
	return false;
}

int main(){
	char str[MAXN];
	scanf("%s", str);
	n = strlen(str);
	for(int i=0; i<n; i++){
		s[i] = str[i] - '0';
	}
	if ((useFirst = 1, backtrack(n-1, 0, 0)) && a[0]!=0){
		for(int i=0; i<n; i++) printf("%d", a[i]);
		printf("\n");
	}
	else if (s[0]==1 && (useFirst = 0, backtrack(n-1, 1, 0)) && a[1]!=0){
		for(int i=1; i<n; i++) printf("%d", a[i]);
		printf("\n");
	}
	else printf("0\n");
	return 0;
}
