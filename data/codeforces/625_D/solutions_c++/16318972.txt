#include <cstdio>
#include <cstring>
using namespace std;
#define N 100010
char st[N];
int a[N],b[N],c[N];
bool check(int s,int t)
{
    while(s<=t)
    {
        if(s==t)
        {
            if(a[s]&1) return 0;
            else{b[s]=a[s]/2;return 1;}
        }
        if(a[t]<-1) return 0;
        if(a[s]>10||(a[s]==10&&a[t]==0)||a[t]==-1){a[t]+=10;a[t-1]--;}
        if(a[s]>a[t]){a[s]--;a[s+1]+=10;}
        if(a[s]>18||a[t]>18) return 0;
        if(a[s]==a[t]){b[s]=(a[s]+1)/2;b[t]=a[s]-b[s];}
        else return 0;
        //printf("%d %d %d %d %d %d\n",s,t,b[s],b[t],a[s+1],a[t-1]);
        ++s;--t;
    }
    return 1;
}
void print(int s,int t,int k)
{
    if(!a[s]){printf("0");return;}
    for(int i=s;i<=t;++i) printf("%d",b[i]);
    //printf("\n%d",k);
}
void init(int l)
{
    for(int i=1;i<=l;++i) a[i]=c[i];
}
int main()
{
    scanf("%s",st);
    int l=strlen(st);
    for(int i=0;i<l;++i) c[i+1]=st[i]-'0';
    init(l);
    if(check(1,l)) print(1,l,1);
    else if((init(l),l>1)&&a[1]==1)
    {
        a[1]=0;a[2]+=10;
        if(check(2,l)) print(2,l,2);
        else printf("0");
    }
    else printf("0");
    return 0;
}
