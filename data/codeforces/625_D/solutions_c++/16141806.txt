using namespace std;
#define eps 1e-8

char str[1000005];
int num[1000005];
int main()
{
    scanf("%s",str);
    int len=strlen(str);
    for(int i=0;i<len;i++)
        num[i]=str[i]-'0';
    int l=0,r=len-1;
    if(num[l]!=num[r])
    {
        num[l]--;
        num[l+1]+=10;
        if(num[l]==0)
            l++;
    }
    while(l<=r)
    {
        if(num[l]==num[r])
            ;
        else if(num[l]==num[r]+10)
        {
            num[r-1]--;
            num[r]+=10;
        }
        else if(num[l]==num[r]+1)
        {
            num[l]--;
            num[l+1]+=10;
        }
        else if(num[l]==num[r]+10+1)
        {
            num[l]--;
            num[l+1]+=10;
            if(l!=r-1){
            num[r]+=10;
            num[r-1]--;}
        }
        if(num[l]!=num[r])
            break;
        if(l!=r)
        {
            num[l]=num[l]-num[r]/2;
            num[r]/=2;
        }
        else
        {
            if(num[l]&1)
                break;
            num[l]/=2;
        }
        if(num[l]>9||num[l]<0||num[r]>9||num[r]<0)
            break;
        l++;
        r--;
    }
    if(l<=r)
        printf("0\n");
    else
    {
        if(num[0])
            for(int i=0;i<len;i++)
            printf("%d",num[i]);
        else
            for(int i=1;i<len;i++)
            printf("%d",num[i]);
        printf("\n");
    }
    return 0;
}
