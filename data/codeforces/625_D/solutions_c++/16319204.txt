#include <iostream>
#include <cstdio>
using namespace std;
int a[100010],b[100010],i,n,l,r;
char ch;
bool work()
{
	l=1;r=n;
	if (a[l]==1 && a[r]!=1) a[l++]=0,a[l]+=10;
	for (;l<r;l++,r--)
	{
		if (a[l]>=20 || a[r]>=20) return 1;
		if (a[l]!=a[r])
		{
			if (a[l]==a[r]+10) a[r]+=10,a[r-1]--;
			else if (a[l]-1==a[r]) a[l]--,a[l+1]+=10;
			else if (a[l]-1==a[r]+10)
			{
				a[r]+=10;a[l]--;
				if (l+1<r) a[l+1]+=10,a[r-1]--;
			}
			else return 1;
		}
	}
	for (i=1;i<=n;++i)
		if (a[i]<0 || a[i]>18) return 1;
	l=1;r=n;
	if (!a[l]) l++;
	for (;l<=r;l++,r--)
	if (a[l]!=a[r]) return 1;
	else if (l==r)
	{
		if (a[l]&1) return 1;
		b[l]=a[l]/2;
	}
	else if (a[l]<=9) b[l]=a[l],b[r]=0;
	else b[l]=a[l]-9,b[r]=9;
	return 0;
}
int main()
{
		for (n=0;~scanf("%c",&ch);a[++n]=ch-'0')
			if (ch<'0' || ch>'9') break;
		if (work()) printf("0");
		else
		{
			for (i=1;!b[i];i++);
			for (;i<=n;++i) printf("%d",b[i]);
		}
}
