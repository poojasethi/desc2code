int ans[200000],s[200000];
bool check(int l,int r,int lc,int rc)
{
	if(l>r) return lc==rc;
	
	s[l]+=10*lc; s[r]-=rc;
	
	if(s[l]-s[r] > 9) s[r]+=10,rc=1;
	else rc=0;
	
	if(s[l]>s[r]) s[l]--,lc=1;
	else lc=0;
	
	if(s[l]!=s[r]) return 0;
	
	ans[l]=(s[l]+1)/2;
	ans[r]=s[l]/2;
	
	if(s[l] > 18 || ans[l] + ans[r]!=s[l]) return 0;
		
	return check(l+1,r-1,lc,rc);
	
}
