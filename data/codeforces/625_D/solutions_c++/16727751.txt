#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

char s[100005];
int ans[100005],c[100005],d[100005];

int main(){
    int len,i,j,flag;
    while(scanf("%s", s)!=EOF){
        len=strlen(s);
        if(len==1){
            if((s[0]-'0')%2==0) printf("%d\n", (s[0]-'0')/2);
            else printf("0\n");
            continue;
        }
        for(i=0;i<len;i++) ans[i]=s[i]-'0';
        flag=0;
        if(ans[0]==1 && ans[0]!=ans[len-1])
            i=1,ans[0]--,ans[1]+=10;
        else i=0;
        for(j=len-1;i<=j;i++,j--){
            if(ans[i]!=ans[j]){
                if(ans[i]-ans[j]>=10 && ans[j]<10){
                    ans[j]+=10;
                    ans[j-1]--;
                }
                if(ans[i]==ans[j]+1){
                    ans[i+1]+=10;
                    ans[i]--;
                }
            }
            if(ans[i]!=ans[j]){
                flag=1;
                break;
            }
            if(i!=j){
                ans[i]=ans[i]-(ans[j]>>1);
                ans[j]>>=1;
            }
            else{
                if(ans[i]&1){
                    flag=1;
                    break;
                }
                else ans[i]>>=1;
            }
            if(ans[i]<0 || ans[i]>9 || ans[j]<0 || ans[j]>9){
                flag=1;
                break;
            }
        }
        if(flag) printf("0\n");
        else{
            if(ans[0]!=0) printf("%d", ans[0]);
            for(i=1;i<len;i++)
                printf("%d", ans[i]);
            puts("");
        }
    }
    return 0;
}
