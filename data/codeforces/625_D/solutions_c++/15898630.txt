int n;
char s0[100010];
char s[100010];
int ans[100010];

bool check()
{
    for(int i=0;i<n/2;)
    {
        if(s[i]==s[n-i-1])
            i++;
        else if(s[i]==s[n-i-1]+1||s[i]==s[n-i-1]+11)
        {
            s[i]--;
            s[i+1]+=10;
        }
        else if(s[i]==s[n-i-1]+10)
        {
            s[n-i-2]--;
            s[n-i-1]+=10;
        }
        else
            return 0;
    }
    if(n%2)
    {
        if(s[n/2]%2||s[n/2]>18||s[n/2]<0)
            return 0;
        else
            ans[n/2]=s[n/2]/2;
    }
    for(int i=0;i<n;i++)
    {
        if(s[i]>18||s[i]<0)
            return 0;
        ans[i]=s[i]/2;
        ans[n-i-1]=(s[i]+1)/2;
    }
    return ans[0]>0;
}

int main()
{
    scanf("%s",s0);
    n=strlen(s0);
    for(int i=0;i<n;i++)
        s[i]=s0[i]-'0';

    if(check())
    {
        for(int i=0;i<n;i++)
            printf("%d",ans[i]);
        printf("\n");
    }
    else if(s0[0]=='1'&&n>1)
    {
        for(int i=0;i<n-1;i++)
            s[i]=s0[i+1]-'0';
        n--;
        s[0]+=10;
        if(check())
        {
            for(int i=0;i<n;i++)
                printf("%d",ans[i]);
            printf("\n");
        }
        else
            printf("0\n");
    }
    else
        printf("0\n");

    return 0;
}
