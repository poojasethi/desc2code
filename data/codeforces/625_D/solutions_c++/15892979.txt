#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define PAIR pair<int,int>
#define N 100005
#define LL long long
#define oo (1<<30)
#define TAT "%I64d"
using namespace std;
int n,a[N],b[N];
int f[N][2][2]; char st[N];
PAIR g[N][2][2],d[N][2][2];
int dp(int t,int s0,int s1)
{
	int i,j,p,q; bool res;
	if(t==0) return f[t][s0][s1]=(s0==s1);
	if(t==1){
		for(i=(t>n-2);i<10;i++)
			if((i+i+s1)%10==a[n/2+1]&&((i+i+s1)>9)==s0){
				f[t][s0][s1]=1,g[t][s0][s1]=PAIR(i,i);
				return 1;
			  }
		return 0;
	  }
	if(f[t][s0][s1]!=-1)
		return f[t][s0][s1];
	for(i=(t>n-2);i<10;i++)
		for(j=0;j<10;j++){
			p=((a[1+(n-t)/2]-i-j)%10+10)%10; //top
			q=((i+j+s1-a[n-(n-t)/2])%10+10)%10; //bot
			if((i+j+p>9)==s0&&!q&&p<2){
				res=dp(t-2,p,(i+j+s1)>9);
				if(res){
					d[t][s0][s1]=PAIR(p,(i+j+s1)>9);
					g[t][s0][s1]=PAIR(i,j);
					return f[t][s0][s1] =1;
				  }
			  }
		  }
	return f[t][s0][s1]=0;
}
void print(int t,int s0,int s1)
{
	if(t==0) return ;
	if(t==1){
		b[n/2+1]=g[t][s0][s1].first;
		return ;
	  }
	b[1+(n-t)/2]=g[t][s0][s1].first;
	b[n-(n-t)/2]=g[t][s0][s1].second;
	print(t-2,d[t][s0][s1].first,d[t][s0][s1].second);
}
void work(int n,int s0,int s1)
{
	int i;
	memset(f,-1,sizeof(f));
	if(!dp(n,s0,s1))
		return ;
	print(n,s0,s1);
	for(i=1;i<=n;i++)
		printf("%d",b[i]);
	exit(0);
}
int main()
{
	int i;
  scanf("%s",st+1);
	n = strlen(st+1);
	for(i=1;i<=n;i++)
		a[i]=st[i]-'0';
	memset(f,-1,sizeof(f));
  work(n,0,0);
	if(a[1]==1){
		n--;
		for(i=1;i<=n;i++)
			a[i]=a[i+1];
		work(n,1,0);
	  }
	printf("0");
	return 0;
}
