#include<bits/stdc++.h>
using namespace std;
char a[100011];
int n;
int b[100011];
bool check()
{
	if (n==0) return false;
	for (int i=0;i<n;i++)
	{
		b[i]=(int)a[i];
	}
	int j;
	/*for (int i=0;i<n;i++)
	{
		cout<<b[i]<<" ";
	}
	cout<<"in"<<endl;*/
	for (int i=0;i<n/2;)
	{
		j=n-1-i;
		if (b[i]==b[j])
		{
			++i;
			continue;
		}
		else if (b[i]==b[j]+1)
		{
			b[i]--;
			b[i+1]+=10;
		}
		else if (b[i]==b[j]+10)
		{
			b[j-1]-=1;
			b[j]+=10;
		}
		else if (b[i]==b[j]+10+1)
		{
			b[i]--;;
			b[i+1]+=10;
		}
		else
		{
			return false;
		}
	}
	/*for (int i=0;i<n;i++)
	{
		cout<<b[i]<<" ";
	}
	cout<<"**"<<endl;*/
	for (int i=0;i<n;i++)
	{
		if (b[i]<0) return false;
	}
	if (n%2==1)
	{
		if (b[n/2]%2==1||b[n/2]/2>=10)
		{
			return false;
		}
		else
		{
			b[n/2]/=2;
		}
	}
	for (int i=0;i<n/2;i++)
	{
		j=n-1-i;
		if (b[i]>=19) return false;
		if (b[i]>=10)
		{
			b[j]=b[i]-9;
			b[i]=9;
		}
		else
		{
			b[j]=0;
		}
	}
	if (b[0]>0)
	return true;
	else
	return false;
}
void print()
{
	int t=0;
	while (b[t]==0) t++;
	for (int i=t;i<n;i++)
	{
		printf("%d",b[i]);
	}
}
int main()
{
	gets(a);
	n=strlen(a);
	for (int i=0;i<n;i++)
	{
		a[i]-='0';
	}
	if (check())
	{
		print();
		return 0;
	}
	else
	{
		if (a[0]==1)
		{
			for (int i=0;i<n-1;i++)
			{
				a[i]=a[i+1];
			}
			a[0]+=10;n--;
			if (check())
			{
				print();
				return 0;
			}
		}
	}
	puts("0");
}
