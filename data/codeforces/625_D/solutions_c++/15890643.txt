#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=1e5+5;
char s[N];
int v[N],ans[N],len;

bool solve()
{
    int i,j;
    for(i=1,j=len;i<=j;)
    {
        if(v[i]==v[j]){i++;j--;}
        else if(v[i]==v[j]+10){v[j-1]--;v[j]+=10;}
        else if(v[i]==v[j]+1||v[i]==v[j]+11){v[i]--;v[i+1]+=10;}
        else return 0;
    }
    if(!v[1])return 0;
    if(i-1==j+1&&(v[i-1]>18||v[i-1]<0||v[i-1]%2))return 0;
    for(i--,j++;i;i--,j++)
    {
        if(v[i]>18||v[i]<0)return 0;
        ans[i]=(v[i]+1)/2;ans[j]=v[i]-ans[i];
    }
    for(i=1;i<=len;i++)printf("%d",ans[i]);
    printf("\n");
    return 1;
}
int main()
{
    //freopen("a.txt","r",stdin);
    while(~scanf("%s",s+1))
    {
        int i;
        for(i=1;s[i];i++)v[i]=s[i]-'0';len=i-1;
        if(solve())continue;
        if(s[1]>'1'||len==1){printf("0\n");continue;}
        for(i=1;s[i];i++)v[i-1]=s[i]-'0';v[1]+=10;len--;
        if(solve())continue;
        printf("0\n");
    }
    return 0;
}
