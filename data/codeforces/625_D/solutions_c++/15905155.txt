#include <bits/stdc++.h>
using namespace std;
string s, a(100002,0);
int sum[100002], n;
#define N n-i-1
bool go(){
    for(int i=0; i<n/2;){
        if(sum[i]==sum[N]){
            i++;
        }
        else if(sum[i]==sum[N]+1||sum[i]==sum[N]+10+1){
            sum[i]--;
            sum[i+1]+=10;
        }
        else if(sum[i]==sum[N]+10){
            sum[N-1]--;
            sum[N]+=10;
        }
        else return false;
    }
    if(n%2==1){
        if(sum[n/2]%2==1||sum[n/2]<0)
            return false;
        else
            a[n/2]=sum[n/2]/2+'0';
    }
    for(int i=0; i<n/2; i++){
        if(sum[i]>18||sum[i]<0) return false;
        a[i]=(sum[i]+1)/2+'0';
        a[N]=sum[i]/2+'0';
    }
    return a[0]>'0';
}

int main(){
    cin >> s;
    n = s.size();
    for(int i=0; i<n; i++){
        sum[i]=s[i]-'0';
    }
    if(go()){
        a.resize(n);
        cout << a;
    }
    else if(s[0]=='1'&&n>1){
        n--;
        for(int i=0; i<n; i++){
            sum[i]=s[i+1]-'0';
        }
        sum[0]+=10;
        if(go()){
            a.resize(n);
            cout << a;
        }
        else cout << 0;
    }
    else{
        cout << 0;
    }
    return 0;
}
