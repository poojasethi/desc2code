#include <bits/stdc++.h>
using namespace std;
const int maxN = 1e5 + 1;

string s;

void enter()
{
   cin >> s;
   for (int i = 0; i < s.length(); i++) s[i] -= '0';
}

void process()
{
   int l = 0;
   int r = s.length() - 1;
   if (s[l] != s[r]) {
      s[l]--;
      s[l + 1] += 10;
      if (s[l] == 0) l++;
   }
   while (l <= r) {
      if (s[l] != s[r]) {
         if (s[l] - s[r] >= 10 && s[r] < 10) {
            s[r - 1]--;
            s[r] += 10;
         }
         if (s[l] - s[r] == 1) {
            s[l]--;
            s[l + 1] += 10;
         }
      }

      if (s[l] != s[r]) break;

      if (l != r) {
         s[l] = s[l] - (s[r] >> 1);
         s[r] >>= 1;
      }
      else {
         if (((s[l] >> 1) << 1) == s[l]) s[l] >>= 1;
         else break;
      }

      if (s[l] > 9 || s[l] < 0 || s[r] > 9 || s[r] < 0) break;
      l++;
      r--;
   }
   if (l <= r) {
      cout << 0;
      return;
   }
   l = 0;
   if (s[l] == 0) l++;
   for (int i = l; i < s.length(); i++) cout << (int) s[i];
}

int main()
{
   ios_base::sync_with_stdio(false); cin.tie(NULL);
   //freopen("TEMPLATE.inp", "r", stdin);
   //freopen("TEMPLATE.out", "w", stdout);

   enter();
   process();

   return 0;
}
