#include <bits/stdc++.h>
using namespace std;
const int N = 1e5+5;
char s[N];
int b[N],r[N],l;
bool go(){
	int i,j;
	for(i=1,j=l;i<=j;){
		if(b[i]==b[j]){++i,--j;}
		else if(b[i]==b[j]+10){b[j-1]--;b[j]+=10;}
		else if(b[i]==b[j]+1 || (b[i]==b[j]+11)){
			b[i]--;b[i+1]+=10;
		}
		else return 0;
	}
	if( (i-1== j+1) && (b[i-1]&1) )return 0;
	for(int x=1;x<=l;++x)
		if(b[x]>18 || b[x]<0)return 0;
	for(--i,++j;i;--i,++j){
		if(b[i]>18 || b[i]<0)return 0;
		r[i]= (b[i]+1)>>1,r[j]= b[i]-r[i];
	}
	if(r[1]<=0)return 0;
	for(i=1;i<=l;++i)
		printf("%d",r[i]);
	putchar('\n');
	return 1;
}	
int main(){
	int i;
	scanf("%s",s+1);
	for(i=1;s[i];++i)b[i]= s[i]-'0';
	l=i-1;
	if(go())return 0;
	if(l==1 || s[1]!='1'){
		printf("0\n");return 0;
	}
	for(i=1;s[i];++i)b[i-1]=s[i]-'0';--l;b[1]+=10;
	if(!go()){
		printf("0\n");
	}
}
