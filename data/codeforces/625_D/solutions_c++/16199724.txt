#include<cstdio>
#include<cmath>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<iostream>
using namespace std;
char s[100000+10];
int num[100000+10];
int sta[100000+10];
void solve(int tail,bool cf)
{
	if(sta[0]==0)return;
	for(int i=0;i<=tail;i++)
	{
		printf("%d",min(sta[i],9));
	}
	for(int i=(cf?(tail):(tail-1));i>=0;i--)
	{
		printf("%d",max(0,sta[i]-9));
	}
	printf("\n");
	exit(0);
}
void dfs(int st,int ed,int t1,int t2)
{
	if(st>ed)return;
	int stnum=num[st]-t1;
	int ednum=num[ed]+t2*10;
	if(st==ed)
	{
		if(stnum%2==0&&stnum==ednum-t1)
		{
			sta[st]=(ednum-t1)/2;
			solve(st,0);
		}
		if(stnum%2==0&&ednum-t1==stnum+10)
		{
			sta[st]=(ednum-t1)/2;
			solve(st,0);
		}
		return;
	}
	else if(st+1==ed)
	{
		if(stnum==ednum)
		{
			sta[st]=stnum;
			solve(st,1);
		}
		if(stnum+10==ednum-1&&ednum)
		{
			sta[st]=stnum+10;
			solve(st,1);
		}
		return;
	}
	if(stnum==ednum&&ednum<19)
	{
		sta[st]=stnum;
		dfs(st+1,ed-1,0,0);
	}
	if(stnum+1==ednum&&ednum&&stnum<19)
	{
		sta[st]=stnum;
		dfs(st+1,ed-1,0,1);
	}
	if(stnum+10==ednum&&ednum<19)
	{
		sta[st]=stnum+10;
		dfs(st+1,ed-1,1,0);
	}
	if(stnum+10+1==ednum&&ednum&&stnum+10<19)
	{
		sta[st]=stnum+10;
		dfs(st+1,ed-1,1,1);
	}
}
int main()
{
	scanf("%s",s);
	int len=strlen(s);
	for(int i=0;i<len;i++)num[len-i-1]=s[i]-'0';
	if(len>=2&&num[len-1]==1)dfs(0,len-2,0,1);
	dfs(0,len-1,0,0);
	printf("0\n");
	return 0;
}
