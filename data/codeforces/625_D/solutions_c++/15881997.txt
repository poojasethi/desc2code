#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

int put[100005];
int len,num[100005];
char s[100005];

bool chk(){
  FI(i,0,len/2-1){
    if(num[i]==num[len-1-i]){}
    else if(num[i]-1==num[len-1-i]){
      num[i]--;
      num[i+1]+=10;
    }
    else if(num[i]-10==num[len-1-i]){
      num[len-1-i]+=10;
      num[len-2-i]--;
    }
    else if(num[i]-11==num[len-1-i]){
      num[i]--;
      num[i+1]+=10;
      if(i+i+2<len){
        num[len-1-i]+=10;
        num[len-2-i]--;
      }
    }
  }
  bool ok=true;
  FI(i,0,len-1) if(num[i]<0 || num[i]>18 || num[i]!=num[len-1-i]) ok=false;
  if(num[0]==0 || (len%2==1 && num[len/2]%2==1)) ok=false;
  //printf("OK %d\n",ok);
  if(ok){
    memset(put,-1,sizeof(put));
    FI(i,0,len-1){
      if(put[i]==-1){
        put[i]=(num[i]+1)/2;
        put[len-1-i]=num[i]-put[i];
      }
      printf("%d",put[i]);
    }
    printf("\n");
    return true;
  }
  else return false;
}
int main(){
  cin.getline(s,100005);
  len=strlen(s);
  //L==len
  FI(i,0,len-1) num[i]=s[i]-'0';
  if(chk()) return 0;
  
  //L==len-1
  if(len==1 || num[0]>1){
    printf("0\n");
    return 0;
  }
  len--;
  FI(i,0,len-1) num[i]=s[i+1]-'0';
  num[0]+=10;
  if(chk()) return 0;
  printf("0\n");
  return 0;
}
