#include <cstdio>
#include <cstring>
using namespace std;
const int MAXN=100000;
char str[MAXN+10];
int num[MAXN+10];
int n;
int read(void)
{
	scanf("%s",str);
	return 0;
}
int init(void)
{
	n=strlen(str);
	for(int i=0;i<n;i++)
		num[i]=str[i]-'0';
	return 0;
}
int solve(void)
{
	int flag=0;
	int l=0,r=n-1;
	if(num[l]==1&&num[l]!=num[r])
	{
		num[l]--;
		l++;
		num[l]+=10;
		flag=1;
	}
	while(l<r)
	{
		if(num[r]<0)
		{
			num[r-1]--;
			num[r]+=10;
		}
		if(num[l]>=10&&num[r]<10&&num[l]-num[r]!=1)
		{
			num[r-1]--;
			num[r]+=10;
		}
		if(num[l]==num[r]+1)
		{
			num[l]--;
			num[l+1]+=10;
		}
		if(num[l]!=num[r])
			return 1;
		l++;
		r--;
	}
	for(int i=flag;i<n;i++)if(num[i]==19)
		return 1;
	int mid=(n-flag)/2+flag;
	if((n-flag)%2&&num[mid]%2)
		return 1;
	for(int i=flag;i<mid;i++)
		num[i]=(num[i]+1)/2;
	for(int i=mid;i<n;i++)
		num[i]=num[i]/2;
	for(int i=flag;i<n;i++)
		printf("%d",num[i]);
	printf("\n");
	return 0;
}
int main(void)
{
	read();
	init();
	if(solve())
		printf("0\n");
	return 0;
}
