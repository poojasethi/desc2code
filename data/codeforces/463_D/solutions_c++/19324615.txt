#include<cstdio>
#include<algorithm>
#define N 1005
using namespace std;
int n,k,a[N][N],c[N][N],dp[N],Ans;
int main(){
	scanf("%d%d",&n,&k);
	for(int j=1;j<=k;j++)
	for(int i=1;i<=n;i++)
		scanf("%d",&a[j][i]);
	for(int i=1;i<=k;i++)
	for(int j=1;j<=n;j++)
	for(int l=1;l<j;l++)
		c[a[i][j]][a[i][l]]=1;
	for(int i=1;i<=n;i++){
		dp[i]=1;
		for(int j=1;j<i;j++)
		//for(int l=1;l<=k;l++)
			if(!c[a[1][j]][a[1][i]])dp[i]=max(dp[i],dp[j]+1);
		Ans=max(Ans,dp[i]);
	}printf("%d\n",Ans);
}
