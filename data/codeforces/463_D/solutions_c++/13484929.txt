#include<bits/stdc++.h>
using namespace std;
int main()

{
	int n,k,t,p;
	cin>>n>>k;
	int a[1005][1005];
	int dp[1005]={0};
	int pos[1005][1005];
	for(int i=1;i<=k;i++)
	{
		for(int j=1;j<=n;j++)
		{
			cin>>a[i][j];
			pos[i][a[i][j]]=j;
	
		}
	}
	int ans=0,maxi=0,j;
	for(int j=1;j<=n;j++)
	{
		maxi=0;
		for(p=1;p<j;p++)
		{
			int i;
			for(i=2;i<=k && pos[i][a[1][p]]<pos[i][a[1][j]];i++);
			if(i>k and dp[p]>maxi)
				maxi=dp[p];
		}
		dp[j]=maxi+1;
		ans=max(ans,dp[j]);
	}
	cout<<ans<<endl;
	return 0;
}
