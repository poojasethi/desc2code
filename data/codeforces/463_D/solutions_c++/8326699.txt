#include<bits/stdc++.h>
using namespace std;
main()
{
	int a[6][1010],n,k,t,k1,pos[6][1010],dp[1010]={0},ans=0;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=k;i++)
	{
		for(int j=1;j<=n;j++)
		{
			scanf("%d",&a[i][j]);
			pos[i][a[i][j]]=j;
		}
	}
	for(int i=1;i<=n;i++)
	{
		t=0;
		for(int j=1;j<i;j++)
		{
			for(k1=2;k1<=k && pos[k1][a[1][j]]<pos[k1][a[1][i]];k1++);
			if(k1==k+1 && t<dp[j])
			t=dp[j];
		}
		dp[i]=t+1;
		ans=max(ans,dp[i]);
	}
	printf("%d\n",ans);
}
