#include<cstdio>
#include<algorithm>
using namespace std;
int n,k,i,j,flag,ans,p;
int a[6][1005],pre[6][1005],f[1005];
int main()
{
  scanf("%d%d",&n,&k);
  for (i=1;i<=k;i++)
    for (j=1;j<=n;j++)
      scanf("%d",&a[i][j]),pre[i][a[i][j]]=j;
  for (i=1;i<=n;i++)
  {
    f[i]=1;
    for (j=1;j<i;j++)
    {
      flag=1;
      for (p=2;p<=k;p++)
        if (pre[p][a[1][j]]>pre[p][a[1][i]]) flag=0;
      if (!flag) continue;
      f[i]=max(f[i],f[j]+1);
    }
    ans=max(f[i],ans);
  }  
  printf("%d",ans);
 return 0;
}
