#include<cstdio>
#include<algorithm>
using namespace std;
int a[7][1010];
int l,z,n,fl,f[1010],k,i,j,pos[7][1010];
int main()
{
 scanf("%d%d",&n,&k);
 for (i=1;i<=k;i++)
   for (j=1;j<=n;j++)scanf("%d",&a[i][j]),pos[i][a[i][j]]=j;
   
 for (i=1;i<=n;i++)f[i]=1;
 for (i=n-1;i>=1;i--)
  for (j=i+1;j<=n;j++){fl=0;
  
   for (l=2;l<=k;l++)
     if(pos[l][a[1][i]]>pos[l][a[1][j]])
                      {fl=1;break;}
   
   if (!fl)f[i]=max(f[i],f[j]+1);
                      }
                      z=0;
for (i=1;i<=n;i++)z=max(z,f[i]);
printf("%d\n",z);
return 0;
}
