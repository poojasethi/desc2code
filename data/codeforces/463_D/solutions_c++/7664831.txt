#include <cstdio>
#include<iostream>
using namespace std;
int a[6][1001],b[6][1001],c,d,e,f[1001],g,h,i,j,k,l,m,n;
int main(){
	scanf("%d%d",&n,&k);
	for(i=1;i<=k;i++)
	 for(j=1;j<=n&&scanf("%d",&a[i][j]);j++)
        b[i][a[i][j]]=j;

	for(i=1;i<=n;i++)
        {
		for(j=0;j<i;j++)
		  {
			g=0;
			for(l=1;l<=k;l++) if(b[l][a[1][i]]<b[l][a[1][j]])g=1;
			if(g) continue;
			f[i]=max(f[i],f[j]+1);
		  }
		h=max(h,f[i]);
	    }
	printf("%d\n",h);
	return 0;
}
