#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int b[10][1005],a[10][1005],n,m;
int f[1005],ans;

int main()
{
    scanf("%d%d",&m,&n);
    for(int i=1;i<=n;i++)
    for(int j=1;j<=m;j++)
    {
        scanf("%d",&a[i][j]);
        b[i][a[i][j]]=j;
    }
    for(int i=1;i<=m;i++)
    for(int j=0;j<i;j++)
    {
        int num=a[1][j];
        int ok=1;
        for(int k=1;k<=n;k++) if(b[k][num]>b[k][a[1][i]]){ok=0;break;}
        if(ok) f[i]=max(f[i],f[j]+1);
        ans=max(ans,f[i]);
    }
    printf("%d\n",ans);
    return 0;
}
