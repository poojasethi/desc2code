#include<cstdio>
#define MAX 1001

int arr[5][MAX];
int reorder[5][MAX];
int dp[MAX];

int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=0;i<k;i++){
		for(int j=0;j<n;j++){
			scanf("%d",&arr[i][j]);
			reorder[i][arr[i][j]]=j;
		}
	}
	int max=0;
	for(int i=0;i<n;i++){
		dp[i]=1;
		for(int j=0;j<i;j++){
			bool flag=true;
			for(int s=0;s<k;s++)
				if(reorder[s][arr[0][i]]<reorder[s][arr[0][j]]){
					flag=false;
					break;
				}
			if(flag){
				if(dp[i]<dp[j]+1)
					dp[i]=dp[j]+1;
			}
		}
		if(max<dp[i]) max=dp[i];
	}
	printf("%d",max);
}
