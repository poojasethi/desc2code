#include <iostream>

using namespace std;

const int MAXN = 1002;

int cnt[MAXN][MAXN];
int a[MAXN], f[MAXN];
int n, k;

int main()
{
    cin >> n >> k;
    for(int x = 0; x < k; x++) {
        for(int i = 0; i < n; i++) {
            cin >> a[i];
            for(int j = 0; j < i; j++) cnt[a[j]][a[i]]++;
        }
    }
    int lcs = 0;
    for(int i = 0; i < n; i++) {
        f[a[i]] = 1;
        for(int j = 0; j < i; j++) {
            if (cnt[a[j]][a[i]] == k) f[a[i]] = max(f[a[i]], f[a[j]] + 1);
        }
        lcs = max(lcs, f[a[i]]);
    }
    cout << lcs << endl;
}
