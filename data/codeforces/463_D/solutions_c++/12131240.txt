#include <stdio.h>
#include <algorithm>
using namespace std;
int n,m,x,g[1001][1001],pos[1001][1001];
int main()
{
	scanf("%d %d",&n,&m);
	for(int i=0;i<m;i++)for(int j=0;j<n;j++){
		scanf("%d",&x);
		pos[x-1][i]=j;
	}
	for(int i=0;i<n;i++)for(int j=0;j<n;j++){
		bool fl=true;
		for(int k=0;k<m;k++)fl&=(pos[i][k]<pos[j][k]);
		g[i][j]=fl?1:-10000;
	}
	for(int k=0;k<n;k++)for(int i=0;i<n;i++)for(int j=0;j<n;j++)g[i][j]=max(g[i][j],g[i][k]+g[k][j]);
	printf("%d\n",*max_element(&g[0][0],&g[0][0]+1000*1000)+1);
	return 0;
}
