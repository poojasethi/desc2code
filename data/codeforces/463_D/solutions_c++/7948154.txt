#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;
int pos[6][1005],a[6][1005],f[1005];
int n,k;

int main(void){
	int i,j,l;
	while(~scanf("%d%d",&n,&k)){
		for(i=1;i<=k;i++){
			for(j=1;j<=n;j++){
				scanf("%d",&a[i][j]);
				pos[i][a[i][j]]=j;
			}
		}
		int ans=1;
		for(i=1;i<=n;i++){
			f[i]=1;
			for(j=1;j<i;j++){
				int flag=1;
				for(l=2;l<=k;l++){
					if(pos[l][a[1][i]]<pos[l][a[1][j]]) flag=0;
				}
				if(flag) f[i]=max(f[j]+1,f[i]);
			}
			ans=max(f[i],ans);
		}
		printf("%d\n",ans);
	}
	return 0;
}
