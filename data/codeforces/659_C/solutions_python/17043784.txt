n, m = map(int, raw_input().split())
a = sorted(map(int, raw_input().split()))
a.append(1100000000)
i, c = 0, 1
l = []
while True:
    if c > m:
        break
    if c == a[i]:
        c += 1
        i += 1
        continue
    l.append(c)
    m -= c
    c += 1

print len(l)
print ' '.join(map(str, l))
