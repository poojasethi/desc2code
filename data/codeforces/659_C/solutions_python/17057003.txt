n,m = map(int,raw_input().split())
arr = map(int, raw_input().split())
arr.sort()
i = min(1, arr[0])
ans = []
k = 0
while m >= i:
	# print ans
	if k < n:
		if i != arr[k]:
			ans.append(str(i))
			m-=i
		else:
			k+=1
	else:
		if i <= m:
			ans.append(str(i))
			m-=i
	i+=1
print len(ans)
print " ".join(ans)
