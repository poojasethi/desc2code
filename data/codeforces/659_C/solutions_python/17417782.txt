n, m = map(int, raw_input().split())
toys = sorted(map(int, raw_input().split()))

result = []
toy_pos = 0
i = 1

while m >= i:
    if toy_pos < len(toys):
        if i == toys[toy_pos]:
            i += 1
            toy_pos += 1
            continue

    result.append(i)
    m -= i
    i += 1

print len(result)
print ''.join(map(lambda x: str(x) + ' ', result))
