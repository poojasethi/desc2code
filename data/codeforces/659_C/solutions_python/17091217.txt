n, m = map(int, raw_input().strip().split())
ais = sorted(map(int, raw_input().strip().split()))

i, aip = 1, 0
toys = []
while m > 0 and aip < n:
    ai = ais[aip]
    while i < ai:
        if i <= m:
            m -= i
            toys.append(i)
        else:
            m = 0
            break
        i += 1
    aip += 1
    i = ai + 1
while m > 0:
    if i <= m:
        m -= i
        toys.append(i)
    else:
        m = 0
    i += 1
print len(toys)
print " ".join(map(str, toys))
