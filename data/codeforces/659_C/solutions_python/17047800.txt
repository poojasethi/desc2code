n,m=[int(x) for x in raw_input().split()]
a=[int(x) for x in raw_input().split()]
a.sort()
k=0
s=0
p=1
l=0
ans=""
while s<m:
    if k>=n or p!=a[k]:
        s+=p
        if s<=m:
            l+=1
            ans+=str(p)+" "
        p+=1
    else:
        k+=1
        p+=1
print(l)
print(ans[:-1])
