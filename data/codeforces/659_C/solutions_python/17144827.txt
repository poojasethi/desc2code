import collections
import sys

already = collections.defaultdict(lambda: False)

n, m = map(int, raw_input().split())
x = map(int, raw_input().split())
for i in x:
    already[i] = True

i = 0
dhvan = list()
while i < 10**9:
    i += 1
    if not already[i] and m >= i:
        dhvan.append(i)
        m -= i
    elif m < i:
        break

print(len(dhvan))
for i in dhvan:
    sys.stdout.write("%d " % i)
