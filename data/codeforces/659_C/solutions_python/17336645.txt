import sys

cin = sys.stdin
cout = sys.stdout

#cin = open("in.txt","r")

[n,m]=cin.readline().strip().split()
n=int(n)
m=int(m)

mas = set([int(x) for x in cin.readline().strip().split()])

ans = []
i = 1
while m>=i:
	if not i in mas:
		ans.append(i)
		m = m - i
	i+=1


print len(ans)
for i in ans:
	cout.write(str(i)+" ")
