def up_to_sum(a, b):
    k = lambda y: ( y*(y-1) )/ 2
    return k(b) - k(a) - a

n, m = map(int, raw_input().split())
toys = set( map(int, raw_input().split()) )

ans = []

for ind in xrange(1, 10**9 + 1):
    if ind in toys:
        continue
    m -= ind
    if m < 0:
        break
    else:
        ans.append( str(ind) )

print len(ans)
print " ".join(ans)
