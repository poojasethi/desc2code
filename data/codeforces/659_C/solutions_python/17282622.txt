n, m = map(int, raw_input().split())
used = dict((int(v), True) for v in  raw_input().split())
ans = 0
types = []
next = 1
while True:
    if next > m: break
    if next in used:
        next += 1
        continue
    ans += 1
    m -= next
    types.append(next)
    next += 1

print "%s\n%s" % (str(ans), " ".join(map(str, types)))
