from sys import *
inp = lambda : stdin.readline()

def main():
    n,m = [int(i) for i in inp().split()]
    l = set([int(i) for i in inp().split()])
    ans,st = [],1
    while m >= st:
        if st not in l:
            ans.append(st)
            m -= st
        st += 1
    print(len(ans))
    print(" ".join([str(i) for i in ans]))


if __name__ == "__main__":
    main()
