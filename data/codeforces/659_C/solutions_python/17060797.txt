def binarySearch(left, right, value):
	middle = (left + right) / 2
	if left >= right:
		return False
	if arr[middle] == value:
		return True
	if arr[middle] > value:
		return binarySearch(left, middle, value)
	elif arr[middle] < value:
		return binarySearch(middle+1, right, value)


I=lambda:map(int, raw_input().split())
n, m = I()
arr = I()
arr.sort()
res = []
k = 0
while m > 0:
	k = k + 1
	if k > m:
		break
	if binarySearch(0, len(arr), k):
		continue
	else:
		m -= k
	res.append(k)
print len(res)
print ' '.join(str(x) for x in 	res)
