from sys import stdin, stdout
def main():
    n, m = map(int, stdin.readline().split())
    a = set(map(int, stdin.readline().split()))
    b = []
    for i in range(1, 10 ** 6):
        if i not in a:
            if m >= i:
                b.append(i)
                m -= i
            else:
                break
    stdout.write(str(len(b)) + '\n')
    stdout.write(' '.join(map(str, b)))
main()
