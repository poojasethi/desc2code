from math import log, floor

n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
a.sort()
c = 1
k = []
s = 0
exit = False
for i in xrange(len(a)):
    while c < a[i]:
        if s + c > m:
            exit = True
            break
        s += c
        k.append(c)
        c += 1
    c += 1
    if exit:
        break

while True:
    if s + c > m:
        break
    s += c
    k.append(c)
    c += 1
print len(k)
print " ".join(map(str, k))
