from collections import *

n,m = map(int,raw_input().split())
arr = map(int,raw_input().split())
ans = [];
arr.sort();
start = 0;
arr.append(10**9+5);
for i in xrange(1,10**9):
	if m - i < 0 :
		break;
	if i == arr[start]:
		start+=1;
	else:
		ans.append(i);
		m -= i;
print len(ans)
print ' '.join(map(str,ans))
