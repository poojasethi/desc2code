n,m = map(int, raw_input().split())
l = sorted(list(set(map(int, raw_input().split()))))

ans = []
curl = 0
cur = 1
while cur <= m:
    while curl < len(l) and l[curl] < cur:
        curl += 1
    if curl < len(l) and l[curl] == cur:
        curl += 1
    else:
        ans.append(cur)
        m -= cur
    cur += 1

print len(ans)
print ' '.join(map(str, ans))
