#!/usr/bin/python
# -*- coding:utf-8 -*-
from _ast import Num

n, m = map(int, raw_input().split())
L = sorted(map(int, raw_input().split()))

def f():
    i = 0
    while True:
        i += 1
        yield i
    
res, idx, ans = '', 0, 0
iter = f()
while True:
    num = iter.next()
    if num > m:
        break
    elif idx < n and num == L[idx]:
        idx += 1
    else:
        m -= num
        if res < 0:
            break
        ans += 1
        res += ' ' + str(num)
print ans
print res[1:]
