#include <iostream>
#include <map>
#include <string.h>
using namespace std;
map<int,bool>m; 
int kq[100000],dem=0;
int main()
{
long long n,t,x,tong=0;
cin >> n >> t;
while (n--)
{
	cin >> x;
	m[x]=1;
}
int cap=1;
while(t)
{
	if(m[cap])
	{
		cap++;
		continue;
	}
	if(t-cap>=0)
	{
	t-=cap;
	kq[dem++]=cap++;
	}else break;
}
cout<<dem<<endl;
for(int i=0;i<dem;i++)
	cout<<kq[i]<<" ";
return 0;
}
