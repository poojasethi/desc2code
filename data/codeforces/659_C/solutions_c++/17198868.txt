#include<stdio.h>
#include<map>
#include<vector>
using namespace std;
int n,m;
map<int,bool> p;
vector<int> g;
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1,x;i<=n;i++){
		scanf("%d",&x);p[x]=true;
	}
	for(int i=1;;i++){
		if(m<i) break;
		if(p[i]) continue;
		m-=i;g.push_back(i);
	}
	printf("%d\n",g.size());
	for(int i=0;i<g.size();i++)
		printf("%d ",g[i]);
	return 0;
}
