#include<bits/stdc++.h>
using namespace std;
int main()
{
	bool a[1000001];
	int n,m;
	vector<int>vec;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
	{
		int c;
		scanf("%d",&c);
		if(c>1000000)continue;
		a[c]=true;
	}
	for(int i=1;i<=m&&m>=0;i++)
	{
		if(a[i])continue;
		vec.push_back(i);
		m-=i;
	}
	cout<<vec.size()<<endl;
	for(int i=0;i<vec.size();i++)
	printf("%d ",vec[i]);
	return 0;
}
