#include <bits/stdc++.h>
using namespace std;
map<int,int>mymap;
vector<int>v;
int main() 
{
	int i,j,k,l,n,m;
	cin>>n>>m;
	for(i=1;i<=n;i++)
	{
		cin>>l;
		mymap[l]=1;
	}
	for(i=1;i<=1000000000;i++)
	{
		if(mymap[i]==0)
		{
			m-=i;
			if(m>=0)
			{
				v.push_back(i);
			}
			else
				break;
		}
	}
	cout<<v.size()<<endl;
	for(i=0;i<v.size();i++)
		cout<<v[i]<<" ";
	return 0;
}
