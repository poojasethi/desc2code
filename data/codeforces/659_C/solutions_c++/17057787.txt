#include <iostream>
#include <algorithm>

using namespace std;


int x, N,m,a[100005],b[45000],j,k=1,rs,o;

int main(){
	cin>>N>>m;
	
	for(int i=1; i<=N; ++i){
		cin>>a[i];
	}
	sort(a+1, a+N+1);
	
	for(int i=1; i<=m && rs+i<=m; ++i)
	{	
		if(i==a[k])k++;
		else 
		if(i!=a[k]){
			b[++o]=i;
			rs+=i;
		}
		
}

    cout<<o<<endl;
    for(int i=1; i<=o; ++i)
       cout<<b[i]<<" ";
}
