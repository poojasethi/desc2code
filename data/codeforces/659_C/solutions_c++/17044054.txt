
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,m,a;

cin>>n>>m;

map <LL,LL> mp;

for(i=0;i<n;i++)
	{
		cin>>a;

		mp[a]++;
	}
LL s=0;
vector <LL> v;

for(i=1;s<=m;i++)
{
if(s+i > m)
	break;

if(mp[i]==0 )
	{s+=i;
		v.push_back(i);
	}


}


cout<<v.size()<<endl;

for(i=0;i<v.size();i++)
	cout<<v[i]<<' ';
cout<<endl;


}
