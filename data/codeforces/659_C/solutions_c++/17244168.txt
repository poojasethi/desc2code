#include<bits/stdc++.h>
using namespace std;
long long a,m,x,cnt;
map<long long,int>ch;
vector<long long>v;
int main(){
 cin>>a>>m;
 for(int i=0;i<a;i++){
  cin>>x;
  ch[x]=1;
 }
 for(int i=1;i<=1000000000;i++){
  if(!ch[i]&&(m-i)>=0){
   cnt++;
   v.push_back(i);
   m-=i;
  }
  if(m<i)break;
 }
 cout<<cnt<<endl;
 for(int i=0;i<cnt;i++)cout<<v[i]<<" ";
 cout<<endl;
}
