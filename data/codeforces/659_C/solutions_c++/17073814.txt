#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int n,m,a[1000001],x;
long long int sum;
vector<int> ans;

int main(){
	cin>>n>>m;
	for(int i=1;i<=n;i++)cin>>a[i];
	sort(a+1,a+n+1);
	int j=1;
	x=1;
	while(sum+x<=m)
	{
		if(x==a[j])j++;
    else ans.push_back(x),sum+=x;
		x++;
	}
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++) cout<<ans[i]<<" ";

	return 0;
}
