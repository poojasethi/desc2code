#include <bits/stdc++.h>
using namespace std;
map <int,bool> mmap;
int n,m,x;
int main(){
    vector <int> ans;
    cin >> n >> m;
    for(int i=1;i<=n;i++){scanf("%d",&x);mmap[x]=1;}
    for(int i=1;m-i>=0;i++){
        if (!mmap[i]) ans.push_back(i),m-=i;
    }
    cout << ans.size() << endl;
    for(int i=0;i<ans.size();i++) printf("%d ",ans[i]);
}
