#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	long i,n,b,a[100000],j,m,d,in=0,ans[100000];
	scanf("%ld%ld",&n,&m);
	for(i=0;i<n;i++)
		scanf("%ld",&a[i]);
	sort(a,a+n);

	i=0;d=1;
	while(m>=d)
	{
		if(a[i]==d)
		i++;
		else
		{
			m=m-d;
			ans[in++]=d;
			
		}
		d++;
	}
	printf("%ld\n",in);//<<endl;
	for(i=0;i<in;i++)
	printf("%ld ",ans[i]);
	
}
