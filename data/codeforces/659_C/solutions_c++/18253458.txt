#include<iostream>
#include<map>
using namespace std;
int main()
{
	int n,m,p,top=-1;
	cin>>n>>m;
	map<int,int>a;
	int k[100001];
	for(int i=0;i<n;i++)cin>>p,a[p]++;
	for(int i=1;i<=200000;i++)if(!a[i] and m>=i)k[++top]=i,m-=i;//else if(i>100000 and m>=i)k[++top]=i,m-=i;}
	cout<<top+1<<endl;for(int i=0;i<=top;i++)cout<<k[i]<<" ";
}
