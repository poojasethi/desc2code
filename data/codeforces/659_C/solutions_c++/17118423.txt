#include <bits/stdc++.h>
//SHANTANU_SHAAN
using namespace std;

map <int,bool> mp;

int a,b[100002],n,m,ans=0;

int main()
{
  cin>>n>>m;
   for(int i=1;i<=n;i++)
    {cin>>a;mp[a]=true;}
   for(int i=1;i<=m;i++)
    {
     if(mp[i]==true) continue;
     ans++;
     m-=i;
     b[ans]=i;
    }

  cout<<ans<<endl;

   for(int i=1;i<=ans;i++)
     cout<<b[i]<<" ";

 return 0;
}
