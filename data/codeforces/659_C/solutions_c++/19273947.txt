#include<iostream>
#include<map>

using namespace std;

map<int, bool>have;
map<int ,int>ans;
int main()
{
	int n,m;
	cin>>n>>m;
	int x;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		have[x]=true;
	}
	int cou=0,su=1;
	while(m>=su)
	{
		if(!have[su] && m>=su)
		{
			m-=su;
			ans[cou]=su;
			cou++;
		}
		su++;
	}	
	cout<<cou<<endl;
	for(int i=0;i<cou;i++)
	cout<<ans[i]<<" \n"[i==cou-1];
	return 0;
}
