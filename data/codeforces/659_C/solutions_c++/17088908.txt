#include<bits/stdc++.h>
using namespace std;
map<int,int> ma;
int a[100005];
int main()
{
    int n,m,t,k;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++){
        scanf("%d",&t);
        ma[t]=1;
    }
    k=0;
    t=0;
    for(int i=1;i+t<=m;i++){
        if(!ma[i])
            t+=i,a[k++]=i;
    }
    printf("%d\n",k);
    for(int i=0;i<k;i++){
        printf("%d ",a[i]);
    }
    return 0;
}
