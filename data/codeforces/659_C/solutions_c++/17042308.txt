#include<cstdio>
#include<algorithm>
#include<vector>

int n,m,A;
int a[100010];
std::vector<int> t;

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++) scanf("%d",a+i);
	std::sort(a,a+n);
	for(int k=1,p=0;k<=m;k++)
	{
		if(k==a[p]) p++;
		else
		{
			t.push_back(k);
			m-=k;
		}
	}
	printf("%d\n",t.size());
	for(int i=0;i<t.size();i++) printf("%d ",t[i]);
	printf("\n");
	return 0;
}
