#include <bits/stdc++.h>
using namespace std;
map<int,int>mymap;
vector<int>v;
int main() 
{
	int i,j,k,l,n,m;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
		cin>>l;
		mymap[l]++;
	}
	for(i=1;i<=1000000000;i++)
	{
		if(m<i)
			break;
		if(mymap[i]==0)
		{
			m-=i;
			v.push_back(i);
		}
	}
	cout<<v.size()<<endl;
	for(i=0;i<v.size();i++)
		cout<<v[i]<<" ";
	return 0;
}
