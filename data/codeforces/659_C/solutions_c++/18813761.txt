#include<bits/stdc++.h>
using namespace std;
set<int> test;
int main(){
	int n, m, old;
	cin >> n >> m;
	for(int i = 0; i < n; ++i){
		cin >> old;
		test.insert(old);
	}
	int ans[100000];
	int dx = 0;
	for(int i = 1; i < 1e9;i++){
		if(m - i < 0)break;
		if(test.count(i) < 1){
			ans[dx++] = i;
			m -= i;
		}
	}
	cout << dx << endl;
	for(int i = 0; i < dx; ++i){
		cout << ans[i] << " ";
	}
	return 0;
}
