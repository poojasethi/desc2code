#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,temp,i,m;
	vector<int>vec;
	map<int,int>mp;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;++i){
		scanf("%d",&temp);
		mp[temp]=1; 
	}
	long long sum=0,num=0;
	for(i=1;sum+i<=m;++i){
		if(mp[i]!=1){
			sum+=i;
			vec.push_back(i);
			num++;
		}
	}
	printf("%I64d\n",num);
	for(i=0;i<vec.size();++i){
		printf("%d ",vec[i]);
	}
}
