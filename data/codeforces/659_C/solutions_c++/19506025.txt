#include <bits/stdc++.h>
#define Rep(i,n) for(int i = 1;i <= n;i ++)
using namespace std;
int a[100005],n,m;
vector<int>v;
int main ()
{
	scanf("%d%d",&n,&m);
	Rep(i,n)scanf("%d",&a[i]);
	sort(a + 1,a + 1 + n);
	for(int i = 1,j = 1;i;i ++)
	{
		if(i == a[j])j ++;
		else if(m < i)break;
		else m -= i,v.push_back(i);
	}
	printf("%d\n",m = v.size());
	for(int i = 0;i < m;i ++)printf("%d ",v[i]);
	return 0;
}
