#include <cstdio>
#define maxn 200010

int b[maxn],c[maxn],len;
int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	for (int i = 1;i <= n;i++)
	{
		int x;
		scanf("%d",&x);
		if (x <= 200000) b[x] = 1;
	}
	for (int i = 1;i <= 200000;i++)
		if (i > k) break;
		else if (!b[i])
		{
			c[++len] = i;
			k -= i;
		}
	printf("%d\n",len);
	for (int i = 1;i <= len;i++)
		printf("%d ",c[i]);
	return 0;
}
