#include<bits/stdc++.h>
#define ll long long
using namespace std;
map<ll,ll>mp;
vector<ll>v;
int main()
{
	ll n,m,no;
	cin>>n;
	cin>>m;
	for(ll i=0;i<n;i++)
	{
		cin>>no;
		mp[no]++;
	}
	for(ll i=1;i<=1000000000;i++)
	{
		if(i>m)
		break;
		if(mp[i]==0)
		{
			m=m-i;
			v.push_back(i);
		}
	}
	cout<<v.size()<<endl;
	for(ll i=0;i<v.size();i++)
	cout<<v[i]<<" ";
}
