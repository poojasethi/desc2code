#include <bits/stdc++.h>
using namespace std;

int main()
{
int n,m;
vector <int> v;
int a[100001];
cin>>n>>m;
for(int i=1;i<=n;i++){
    cin>>a[i];
}
sort(a+1,a+n+1);
int k=1;
for(int i=1;m-i>=0;i++){
    if(i!=a[k]){
        m=m-i;
        v.push_back(i);
    }
    else{
        k++;
    }
}
cout<<v.size()<<"\n";
for(int i=0;i<v.size();i++){
    cout<<v[i]<<" ";
}
    return 0;
}
