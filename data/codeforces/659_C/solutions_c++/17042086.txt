#include <bits/stdc++.h>
using namespace std;
map<int, bool> mpm;
vector<int> ans;  
int main(){
	int n, m;
	cin>>n>>m;
	int a;
	for (int i = 0; i < n; i++) {
		scanf("%d", &a);
		mpm[a] = true;
	}
	for (int i = 1; ; i++) {
		if(mpm[i]) continue;
		if(m >= i) {
			ans.push_back(i);
			m -= i;
		}
		else break;
	}
	cout<<int(ans.size())<<endl;
	for (int i = 0; i < int(ans.size()); i++) printf("%d ", ans[i]);
	printf("\n");
}
