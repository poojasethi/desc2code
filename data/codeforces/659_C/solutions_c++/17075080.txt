#include <bits/stdc++.h>
using namespace std;

map<int,int> MAP;

int main()
{
	int n,m,i,a;
	cin>>n>>m;
	for(i=0;i<n;++i)
	{
		cin>>a;
		MAP[a]=1;
	}

	vector<int> ret;

	for(i=1;i<=1e9;++i)
	{
		if(i>m)
			break;
		if(MAP.count(i))
			continue;
		m-=i;
		ret.push_back(i);
	}

	cout<<ret.size()<<endl;

	for(i=0;i<ret.size();++i)
		cout<<ret[i]<<" ";

	cout<<endl;
}
