#include<bits/stdc++.h>
using namespace std;
vector<int>a;
int main()
{
    int n,m;
    cin>>n>>m;
    int b[100005];
    for(int i=0;i<n;i++)
        scanf("%d",&b[i]);
    sort(b,b+n);
    for(int i=1,j=0;;i++)
    {
        if(i==b[j])j++;
        else if(m<i)break;
        else a.push_back(i),m-=i;
    }
    cout<<a.size()<<endl;
    for(int i=0;i<a.size();i++)
        printf("%d ",a[i]);
}
