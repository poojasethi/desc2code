#include <iostream>
#include <set>
#include <vector>
using namespace std;


int main(){
	int n, m, i, p;
	vector<int> v;
	set<int> s;
	cin >> n >> m;
	for(i = 1; i <= n; i++){
		cin >> p;
		s.insert(p);
	}
	for(i = 1; m; i++) {
		if(!s.count(i)) {
			if(m >= i) v.push_back(i), m -= i;
			else break;
		}
	}
	cout << v.size() << endl;
	for(i = 0; i < v.size(); i++) cout << v[i] << " ";
}
