#include <bits/stdc++.h>

using namespace std;

int n, m;

int *mas;
vector<int> bbb;

main(){
	cin>>n>>m;
	mas = new int[n];
	for(int i=0; i<n; i++) cin>>mas[i];
	
	sort(mas, mas+n);
	int k=0;
	
	for(int i=1;; i++){
		if(m < i) break;
		if(!binary_search(mas,mas+n,i)){
			bbb.push_back(i);
			m -= i;
			k++;
		}
	}
	
	cout<<k<<endl;
	for(int i=0; i<bbb.size(); i++) cout<<bbb[i]<<" ";
	
}
