#include<stdio.h>
#include<algorithm>
using namespace std;


int a[200000],b[2000000];


int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++) scanf ("%d",&a[i]);
    sort(a,a+n);
    int i=0,sum=0,kol=0;
    for(int j =1;sum+j<=m;j++) if(a[i]==j) i++; else{sum+=j;b[kol++]=j;}
    printf("%d\n",kol);
    for(int i=0;i<kol;i++) printf("%d ",b[i]);
    return 0;
}
