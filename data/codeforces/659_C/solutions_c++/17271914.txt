#include<iostream>

using namespace std;

bool a[1000002];
int b[1000009];

int main(){
	int n,m,tmp;
	cin>>n>>m;
	for(int i=0;i<n;i++) {
		cin>>tmp;
		if(tmp<1000001) a[tmp]=1;
	}
	int sum=0;
	int l=1;
	int i=0;
	while(sum+l<=m){
		if(!a[l]) {
			b[i++]=l;
			sum+=l;
		}
		//cout<<l<<" "<<sum<<" "<<endl;
		++l;
	}
	cout<<i<<endl;
	for(int j=0;j<i;j++) cout<<b[j]<<" ";
}
