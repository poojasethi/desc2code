#include<iostream>
#include<set>
#include<vector>
using namespace std;
set<int> s;
vector<int> v;
int main()
{
   int n,m,i,val;
   cin>>n>>m;
   for(i=0;i<n;i++)
   {
       cin>>val;
       s.insert(val);
   }
   for(i=1;i<=m;i++)
   {
       if(s.count(i))
        continue;
       m-=i;
       v.push_back(i);

    }
    cout<<v.size()<<endl;
    for(i=0;i<v.size();i++)
        cout<<v[i]<<" ";


}
