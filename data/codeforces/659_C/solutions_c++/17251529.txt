#include <iostream>
#include <algorithm>

using namespace std;

main()
{
	int n,m,a[100000],b[100000]={0},i,k=0,l=0;
	cin >> n >> m;
	for(i=0;i<n;++i)
	    cin >> a[i];
	sort(a,a+n);
	i=1;
	while (i<=m)
	{
	    if(i==a[k])
	    {
	        ++k;
	        ++i;
	        continue;
	    }
	    m-=i;
	    b[l]=i;
	    ++l;
	    ++i;
	}
	cout<<l<<endl;
	for(i=0;i<l;++i)
	    cout<<b[i]<<" ";
}
