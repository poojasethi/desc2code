#include <bits/stdc++.h>
using namespace std;
map<int,int>mpint;
int a[200000];
int main()
{
int n, m, i, t, k;
scanf("%d%d", &n, &m);
for(i=1;i<=n;i++){
    scanf("%d", &t);
    mpint[t] = 1;
}
long long sum=m;
for(k=1,i=1;i<1e9;i++){
    if(!mpint[i]){
    sum -= i;
    a[k++]=i;
    }
    if(sum<0)
        break;
}
printf("%d\n", k-2);
for(i=1;i<=k-2;i++)
    printf("%d ", a[i]);
return 0;
}
