#include<bits/stdc++.h>
using namespace std;
bool fun(pair<int,string>p1,pair<int,string>p2) {
	return (p1.first>p2.first);
}
int main()
{
	int n,m;
	cin>>n>>m;
	set<int>s;
	int i,a,cnt=0,j;
	for(i=0;i<n;i++) {
		cin>>a;
		s.insert(a);
	}
	i=1;
	while(m>=i) {
		if(s.find(i)==s.end()) {
			m=m-i;
			cnt++;
		}
		i++;
	}
	cout<<cnt<<endl;
	for(j=1;j<i;j++) {
		if(s.find(j)==s.end()) cout<<j<<" ";
	}
}
