#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n,m,i;
	cin>>n>>m;
	map<int,int>mm;
	for(i=0;i<n;i++){
		int ele;cin>>ele;
		mm[ele]=1;
	}
	vector<int>ans;
	for(i=1;(i<1000000001 && m>0);i++){
		if(mm[i]==0){
			if(m-i>=0){m-=i;ans.push_back(i);}
			else break;
		}
	}
	cout<<ans.size()<<endl;
	for(i=0;i<ans.size();i++){
		cout<<ans[i]<<" ";
	}
	return 0;
}
