#include<iostream>
using namespace std;
int n, m, a, k=1, j=0;
bool u[200000];
int t[200000];

int main()
{
	cin >> n >> m;
	for(int i=0; i<n; i++)
	{
		cin >> a;
		if(a < 200000) u[a] = true;
	}
	while(k <= m){
		if(!u[k]) t[j++]=k, m -= k;
		k++;}
	cout << j << endl;
	for(int i=0; i<j; i++) cout << t[i] << ' ';
}
