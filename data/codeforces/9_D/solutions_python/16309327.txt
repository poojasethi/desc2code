n, h = map(int, raw_input().split())
dp = [[0 for i in range(0, n + 1)] for i in range(0, n + 1)]
for i in range(1, n + 1): dp[1][i] = 1
for k in range(1, n + 1):
	for i in range(1, n + 1):
		dp[i][k] += 2 * dp[i - 1][k - 1]
		for j in range(1, i - 1): dp[i][k] += dp[j][k - 1] * dp[i - 1 - j][k - 1]
print dp[n][n] - dp[n][h - 1]
