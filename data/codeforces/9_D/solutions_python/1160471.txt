n, h = map(int, raw_input().split())

dp = [ [-1]*(n+1) for i in xrange(n+1)]

dp[0][0] = 1
def f(n, h):
    if dp[n][h] != -1:
        return dp[n][h]

    if n == 0:
        return 0
    if h == 0:
        return f(n, 1)

    ans = 0
    for i in xrange(n): #0 .. n-1
        ans += f(n-i-1, 0) * f(i, h-1)
        ans += f(n-i-1, h-1) * f(i, 0)
        ans -= f(n-i-1, h-1) * f(i, h-1)

    dp[n][h] = ans
    return ans

print f(n, h)
