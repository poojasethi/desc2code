n,h=[int(i)for i in raw_input().split()]
d={}
def f(n,h):
        if (n,h) in d:
                return d[n,h]
        if not n:
                if not h:
                        return 1
                else:
                        return 0
        s=0
        for i in range(n):
                s+=f(i,max(0,h-1))*f(n-1-i,0)+f(i,0)*f(n-1-i,max(0,h-1))-f(i,max(0,h-1))*f(n-1-i,max(0,h-1))
        d[n,h]=s
        return s
print f(n,h)
