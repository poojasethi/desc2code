dp = {}
def f(n, h):
    """ # of binary trees with n nodes of height <= h """
    if (n, h) in dp:
        return dp[n, h]

    if n == 0:
        result = 1
    elif h == 0:
        result = 0
    else:
        result = sum(f(a, h-1) * f(n-a-1, h-1)   for a in range(n))

    dp[n, h] = result
    return result


n, h = map(int, raw_input().split())
print f(n, n) - f(n, h-1)
