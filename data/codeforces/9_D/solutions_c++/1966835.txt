#include<iostream>
#include<cstring>
using namespace std;
long long ans[40][40];
long long f(int n,int h)
{
	if(ans[n][h]!=-1)return ans[n][h];
	if(h>n)return ans[n][h]=0;
	if(n==0)		 return ans[n][h]=1;
	int hh=h;
	if(h==0)hh=1;
	ans[n][h]=0;
	for(int k=1;k<=n;k++)		ans[n][h]+=f(k-1,hh-1)*f(n-k,0)+f(k-1,0)*f(n-k,hh-1)-f(k-1,hh-1)*f(n-k,hh-1);
	return ans[n][h];
}
int main()
{
	int n,h;
	cin>>n>>h;
	memset(ans,-1,sizeof(ans));
	cout<<f(n,h)<<endl;
	return 0;
}
