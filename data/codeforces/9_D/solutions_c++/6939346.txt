#include<iostream>
using namespace std;
typedef unsigned long long int ll;
ll dp[40][40];
int main()
{
int i,j,k,m;
dp[1][1]=1;
dp[0][0]=1;
for(i=2;i<=36;i++)
{
for(j=1;j<=i;j++)
{
for(k=0;k<i;k++)
{
for(m=0;m<j-1;m++)
dp[i][j]+=dp[k][j-1]*dp[i-k-1][m];
for(m=0;m<j-1;m++)
dp[i][j]+=dp[k][m]*dp[i-k-1][j-1];
dp[i][j]+=dp[k][j-1]*dp[i-k-1][j-1];
}
//cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
}
}
int n,h;
cin>>n>>h;
ll sum=0;
for(i=h;i<=n;i++)
sum+=dp[n][i];
cout<<sum<<endl;
return 0;
}
