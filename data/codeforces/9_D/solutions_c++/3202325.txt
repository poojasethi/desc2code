#include<stdio.h>
#include<string.h>
typedef __int64 LL;
LL dp[40][40];
LL tree(LL n,LL h)
{
	if(!h)
		return 0;
	if(n<=1)
		return 1;
	if(dp[n][h]!=-1)
		return dp[n][h];
	LL i,k;
	for(i=k=0;i<n;i++)
		k+=(tree(i,h-1)*tree(n-i-1,h-1));
	return dp[n][h]=k;
}
int main()
{
	LL n,h;
	memset(dp,-1,sizeof(dp));
	scanf("%I64d%I64d",&n,&h);
	printf("%I64d\n",tree(n,n)-tree(n,h-1));
	return 0;
}
