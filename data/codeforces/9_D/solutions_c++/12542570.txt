#include<cstdio>
#include<cstdlib>
#include<iostream>
using namespace std;
#define maxn 36
long long int dp[maxn][maxn];
void ttt(){
	for(int h=1;h<maxn;h++){
		dp[0][h-1]=1;
		for(int n=1;n<maxn;n++){
			for(int i=0;i<n;i++){
				dp[n][h]+=dp[i][h-1]*dp[n-i-1][h-1]; 
			}
		}
	}
} 
int main(void){
	ttt();
	int n,h;
	while(scanf("%d%d",&n,&h)!=EOF){
		printf("%lld\n",dp[n][n]-dp[n][h-1]);
	}
	return 0;
}
