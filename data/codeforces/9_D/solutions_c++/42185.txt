#include<cstdio>
#include<cstring>

long long f[36][36];
long long F(int n,int m)
{
	if(m<0) m=0;
	if(n==0) return m==0;
	if(f[n][m]!=-1) return f[n][m];
	f[n][m]=0;
	for(int i=0;i<n;i++)
		f[n][m]+=F(i,m-1)*F(n-i-1,0)+F(i,0)*F(n-i-1,m-1)-F(i,m-1)*F(n-i-1,m-1);
	return f[n][m];
}

int main()
{
	memset(f,-1,sizeof(f));
	int n,m;
	scanf("%d%d",&n,&m);
	printf("%I64d\n",F(n,m));
	return 0;
}
