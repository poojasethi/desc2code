#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

typedef long long int64;
const int N = 40;

int n , h;
int64 dp[N][N]; 

void init(){

	for(int i = 0 ; i < N ; i++)
		dp[0][i] = 1;

	for(int i = 1 ; i < N ; i++)
		for(int j = 1 ; j < N ; j++)
			for(int k = 0 ; k < i ; k++)
				dp[i][j] += dp[k][j-1]*dp[i-k-1][j-1];
}

int main(){
	init();
	while(~scanf("%d%d" , &n , &h))
		cout<<(dp[n][n]-dp[n][h-1])<<endl;
	return 0;
}
