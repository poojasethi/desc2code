#include <iostream>
using namespace std;

const int NMAX = 37;

long long d[NMAX][NMAX];

int main()
{
    int N, H, j, h, n;
    
    cin >> N >> H;
    for (j = 0; j <= N; ++j)
        d[0][j] = d[1][j] = 1;
    
    d[1][0] = 0;
    
    for (n = 2; n <= N; ++n)
        for (h = 2; h <= N; ++h)
            for (j = 0; j < n; ++j)
                d[n][h] += d[n - j - 1][h - 1] * d[j][h - 1];
            
    cout << d[N][N] - d[N][H - 1] << endl;
    return 0;
}
