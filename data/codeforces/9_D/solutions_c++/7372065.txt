#include <cstdio>
#include <iostream>
using namespace std;
long long dp[40][40];
int n,h;
void solve(int n,int h){
    int i,j,k;
	for (i=0;i<=n;i++) dp[0][i]=0;
	for (i=0;i<=n;i++) dp[i][0]=1;
	for (k=1;k<=n;k++)
	{
		for (i=1;i<=n;i++)
		{
			dp[k][i]=0;
			for (j=0;j<i;j++)
			{
				dp[k][i]+=dp[~-k][j]*dp[~-k][i-j-1];
            }
        }
    }
    cout<<dp[n][n]-dp[~-h][n]<<endl;
}
int main()
{
	scanf("%d%d",&n,&h);
	solve(n,h);
	return 0;
}
