#include <iostream>
using namespace std;

const int MAXN=50;

int main()
{
	int n,h;
	cin>>n>>h;

	long long f[MAXN][MAXN]={0};	
	for (int i=0;i<=n;++i) f[0][i]=1;

	for (int i=1;i<=n;++i)
		for (int j=1;j<=n;++j) {
			for (int k=0;k<=i-1;++k) 
				f[i][j]+=f[k][j-1]*f[i-k-1][j-1];
		//	cout<<i<<' '<<j<<' '<<f[i][j]<<endl;
		}

	cout<<f[n][n]-f[n][h-1]<<endl;
	return 0;
}
