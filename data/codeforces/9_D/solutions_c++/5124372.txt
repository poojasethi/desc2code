#include <cstdio>
typedef long long LL;
LL dp[40][40];
int main()
{
    for(int i = 0;i <= 35;i++)  dp[0][i] = 1;
    for(int i = 1;i <= 35;i++)
        for(int j = 1;j <= 35;j++)
            for(int k = 0;k < i;k++)
                dp[i][j] += dp[k][j-1]*dp[i-k-1][j-1];
    int n,h;
    while(scanf("%d%d",&n,&h) == 2){
        printf("%lld\n",dp[n][35]-dp[n][h-1]);
    }
    return 0;
}
