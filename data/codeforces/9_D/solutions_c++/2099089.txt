#include <string.h>
#include <stdio.h>
typedef long long LL;
int n,h;
LL d[40][40];
LL dp(int n,int h){
    if(d[n][h])return d[n][h];
    if(n==0)return 1;
    if(h>0)for(int i=0;i<=n-1;i++)
        d[n][h]+=dp(i,h-1)*dp(n-i-1,h-1);;
    return d[n][h];
}
int main(){
    while(scanf("%d%d",&n,&h)!=EOF){
        memset(d,0,sizeof d);
        printf("%I64d\n",dp(n,n)-dp(n,h-1));
    }
    return 0;
}
