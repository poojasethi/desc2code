#include <cstdio>

int n, h;
long long dp[40][40];

int main() {
	scanf("%d%d", &n, &h);
	for(int i=0; i <= n; i++) dp[0][i]=0;
	for(int k=0; k <= n; k++) dp[k][0]=1;
	for(int k=1; k <= n; k++)
		for(int i=1; i <= n; i++) {
			dp[k][i]=0;
			for(int j=0; j<i; j++) dp[k][i] += dp[k-1][j]*dp[k-1][i-j-1];			
		}
	printf("%I64d\n", dp[n][n]-dp[h-1][n]);	
	return 0;
}
