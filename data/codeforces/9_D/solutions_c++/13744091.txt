#include<iostream>
#include<string.h>
using namespace std;
typedef long long lld;
lld dp[40][40];
int main()
{
    int i,j,k;
    int n,h;
    cin>>n>>h;
    memset(dp,0,sizeof(dp));
    for(i=0;i<=n;i++)
        dp[i][0]=1;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {

            for(k=0;k<j;k++)
            {

                dp[i][j]+=dp[i-1][k]*dp[i-1][j-k-1];
            }
        }
    }
    cout<<dp[n][n]-dp[h-1][n]<<endl;
    return 0;
}
