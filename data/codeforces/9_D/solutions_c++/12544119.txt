#include <iostream>
using namespace std;
long long  n, h, i, j, k, dp[36][36];
int main() {
	cin >> n >> h;
	for (i = 1; i <= n; i++) {
        dp[0][i - 1] = 1;
        for (j = 1; j <= n; j++)
            for (k = 0; k < j; k++)
                dp[j][i] += dp[k][i - 1] * dp[j - k - 1][i - 1];
	}
	cout << dp[n][n] - dp[n][h - 1] << endl;
	return 0;
}
