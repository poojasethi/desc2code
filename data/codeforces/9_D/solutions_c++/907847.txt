#include<iostream>
#define MAX 35

using namespace std;
typedef unsigned long long ULL;

int main(){
	int n,h,i,nodes,height;
	ULL T[MAX+1][MAX+1]={0};

	cin >> nodes >> height;

	for(h=0;h<=nodes;h++)
		T[0][h] = 1;

	for(n=1;n<=nodes;n++){
		for(h=1;h<=n;h++)
			for(i=0;i<=n-1;i++)
				T[n][h] += T[i][h-1] * T[n-i-1][h-1];
		for(;h<=nodes;h++)
			T[n][h] = T[n][h-1];
	}

	cout << (T[nodes][nodes] - T[nodes][height-1]) << endl;
	return 0;
}
