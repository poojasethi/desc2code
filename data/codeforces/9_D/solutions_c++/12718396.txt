#include <iostream>
using namespace std;

int main()
{
    long long i, j, n, h, v, g[50][50];
	cin >> n >> h;
	for (v = 1;v <= n; v ++ )
    {
		g[0][v - 1] = 1;
		for (i = 1;i <= n;i ++)
			for (j = 0;j < i; j ++)
				g[i][v] += g[j][v-1] * g[i-1-j][v-1];
	}
	cout << g[n][n] - g[n][h-1] << endl;
	return 0;
}
