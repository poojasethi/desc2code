#include<bits/stdc++.h>
using namespace std;
#define LL long long
LL dp[40][40];
int main()
{
   int n,h;
   scanf("%d%d",&n,&h);
   for (int i = 1;i<=n;i++)
   {
	   dp[0][i-1]=1;
	   for (int j = 1;j<=n;j++)
		   for (int k = 0;k<j;k++)
			   dp[j][i]+=dp[k][i-1]*dp[j-k-1][i-1];
   }
   printf("%lld\n",dp[n][n]-dp[n][h-1]);
}
