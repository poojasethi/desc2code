#include <iostream>
using namespace std;

long long dp[36][36];

int main()
{
    for(int i = 0; i < 36; ++i) dp[0][i] = 1;
    for(int i = 1; i < 36; ++i)
        for(int j = 1; j < 36; ++j)
            for(int k = 0; k < i; ++k)
                dp[i][j] += dp[k][j-1] * dp[i-k-1][j-1];
    int n, k;
    cin >> n >> k;
    cout << dp[n][35] - dp[n][k-1] << endl;
    return 0;
}
