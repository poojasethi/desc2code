#include <bits/stdc++.h>
using namespace std;
bool was[50][50];
long long dp[50][50];
long long ans(int n,int h)
{
	if(n == 0)return 1;
	if(h == 0)return 0;
	if(was[n][h])return dp[n][h];
	long long res = 0;
	for(int i=0;i<n;i++) // let i is the element in the left subtree
		res += 1ll*ans(i,h-1)*ans(n-i-1,h-1);
	was[n][h] = 1;
	dp[n][h] = res;
	return res;
}
int main()
{
	int n,h;
	cin>>n>>h;
	printf("%I64d\n",ans(n,n) - ans(n,h-1));
	return 0;
}
