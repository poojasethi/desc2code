#include <cstdio>
#include <iostream>
#define MAXN 35
#define f first
#define s second
#define mp make_pair
using namespace std;
int N,H;
long long D[MAXN+1][MAXN+1];
long long f(int n,int h)
{
	int i;
	long long t=0;
	if(!n) return 1;
	if(!h) return 0;
	if(n==1) return 1;
	if(D[n][h]) return D[n][h];
	for(i=n-1;i>=0;i--)
	{
		t+=f(i,h-1)*f(n-1-i,h-1);
	}
	return D[n][h]=t;
}
int main()
{
	scanf(" %d %d",&N,&H);
	cout << f(N,N)-f(N,H-1) << endl;
	return 0;
}
