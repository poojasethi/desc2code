#include <cstdio>
#include <cstring>

typedef long long ll;

const int MAX_N = 36;

int n, h;
ll dp[MAX_N][MAX_N];

int main() {
	for (int i = 0; i < MAX_N; ++i) dp[0][i] = 1;
	for (int i = 1; i < MAX_N; ++i) {
		for (int j = 1; j < MAX_N; ++j) {
			for (int k = 0; k < i; ++k) {
				dp[i][j] += dp[k][j - 1] * dp[i - k - 1][j - 1];
			}
		}
	}
   	scanf("%d%d", &n, &h);
   	printf("%I64d\n", dp[n][35] - dp[n][h - 1]);
    return 0;
}
