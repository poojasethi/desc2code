#include <cstdio>
#include <iostream>

using namespace std;

typedef long long int Lint;

const int MAXN=36;

int N,H;

Lint dn[MAXN][MAXN];

Lint f( int n , int h ){
	
	if(!h)
		return 0;
	if( n<=1 )
		return 1;
	
	Lint &rev=dn[n][h];
	
	if( rev )
		return rev;
	
	for( int i=0 ; i<n ; i++ )
		rev+=f(i,h-1)*f(n-i-1,h-1);
	
	return rev;
}

int main(){
	
	scanf(" %d %d",&N,&H);
	
	printf("%I64d\n",f(N,N)-f(N,H-1));
	
	return 0;
}
