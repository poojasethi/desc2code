#include<cstdio>
#include<cstring>
typedef __int64 lld;
lld dp[36][36];
int main()
{
	for(int i=0;i<=35;i++) dp[0][i]=1;
	for(int i=1;i<=35;i++)
	{
		for(int j=1;j<=35;j++)
		{
			for(int k=0;k<i;k++)
			{
				dp[i][j]+=dp[k][j-1]*dp[i-k-1][j-1];
				//printf("i=%d j=%d %I64d\n",i,j,dp[j][j]);
			}
		}
	}
	int n,h;
	scanf("%d%d",&n,&h);
	printf("%I64d\n",dp[n][35]-dp[n][h-1]);
	return 0;
}
