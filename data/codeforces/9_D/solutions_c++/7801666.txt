#include<cstdio>
#include<cstring>
using namespace std;
int n,h;
long long dp[40][40];
int main()
{
    scanf("%d %d",&n,&h);
    memset(dp,0,sizeof(dp));
    for(int i=0;i<=n;i++)
    {
        dp[0][i]=1;
    }
    for(int h=1;h<=n;h++)
    {
        for(int i=1;i<=n;i++)
        {
            for(int j=0;j<i;j++)
                dp[i][h]+=dp[j][h-1]*dp[i-j-1][h-1];
        }
    }
    printf("%I64d\n",dp[n][n]-dp[n][h-1]);
    return 0;
}
