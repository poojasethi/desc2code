#include <iostream>
#include <fstream>
#define ll unsigned long long
using namespace std;

ll A[36][39];

ll F(int n,int h)
{
    if(!h) return 0;
    if(n <= 1) return 1;
    if(A[n][h]) return A[n][h];
    ll i,r=0;
    for(i=0;i<n;i++) r+=F(i,h-1)*F(n-i-1,h-1);
    return A[n][h]=r;
}

int main()
{
    int n,k;
    cin >> n >> k;
    cout << F(n,n)-F(n,k-1) << endl;
    return 0;
}
