#include <cstdio>
long long d[40][40]={0};
int main(int n, int h) {
	scanf("%d%d", &n, &h);
	for(int i=0; i <= n; i++) d[0][i]=0, d[i][0]=1;	
	for(int i=1; i <= n; i++) for(int j=1; j <= n; j++) for(int k=0; k <= j-1; k++) d[i][j] += d[i-1][k]*d[i-1][j-k-1];
	printf("%I64d\n", d[n][n]-d[h-1][n]);
}
