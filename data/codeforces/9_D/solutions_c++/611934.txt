#include<iostream>
using namespace std;
long long dp[40][40];

int main(){
	int i, j, k, n, h;
	cin>>n>>h;
	dp[0][0] = 1;
	for(i = 1; i <= n; i++){
		for(j = 0; j <= i - 1; j++)
			for(k = n; k >= 1; k--){
				dp[i][k] += dp[j][k-1] * dp[i-1-j][0];
				dp[i][k] += dp[j][0] * dp[i-1-j][k-1];
				dp[i][k] -= dp[j][k-1] * dp[i-1-j][k-1];
			}
		dp[i][0] = dp[i][1];
	}
	cout<<dp[n][h]<<endl;
}
