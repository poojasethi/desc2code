#include <iostream>
using namespace std;
long long n,h,i,j,k,f[50][50];
int main() {
    cin >> n >> h;
    for (k=1;k<=n;k++) {
        f[0][k-1]=1;
        for (i=1;i<=n;i++)
            for (j=0;j<i;j++)
                f[i][k]+=f[j][k-1]*f[i-1-j][k-1];
    }
    cout << f[n][n]-f[n][h-1] << endl;
    return 0;
}
