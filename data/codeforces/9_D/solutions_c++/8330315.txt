#include<iostream>
using namespace std;

long long dp[40][40];

int main()
{
	int n, h;
	cin >> n >> h;
	for (int i = 0; i <= n; i++) dp[0][i] = 0;
	for (int i = 0; i <= n; i++) dp[i][0] = 1;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= n; j++)
		{
			dp[i][j] = 0;
			for (int k = 0; k < j; k++)
			{
				dp[i][j] += dp[i-1][k] * dp[i-1][j - k - 1];
			}
		}
	}
	cout << dp[n][n] - dp[h - 1][n] << endl;
}
