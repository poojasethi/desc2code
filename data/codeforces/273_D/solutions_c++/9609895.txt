#include<cstdio>
using namespace std;
const int p=1000000007;
int n,m;
int S[2][2][155][155];
int f[155][155][155][2][2];
int Mod(int x){return x<p?x:x-p;}
void Add(int &x,int y){x+=y;if(x>=p)x-=p;}
int Get(int a,int b,int c,int d,int x,int y)
{
	if(a>b||c>d)
		return 0;
	int ret=S[x][y][b][d];
	Add(ret,p-S[x][y][b][c-1]);
	Add(ret,p-S[x][y][a-1][d]);
	Add(ret,S[x][y][a-1][c-1]);
	return ret;
}
int main()
{
	scanf("%d%d",&n,&m);
	int ans=0;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
			for(int k=1;k<=m;k++)
				for(int x=0;x<2;x++)
					for(int y=0;y<2;y++)
						S[x][y][j][k]=f[i-1][j][k][x][y],
						Add(S[x][y][j][k],S[x][y][j-1][k]),
						Add(S[x][y][j][k],S[x][y][j][k-1]),
						Add(S[x][y][j][k],p-S[x][y][j-1][k-1]);
		for(int j=1;j<=m;j++)
			for(int k=j;k<=m;k++)
				f[i][j][k][1][1]=Mod(Get(1,j,k,m,1,1)+Get(1,j-1,k+1,m,0,0)),
				f[i][j][k][0][1]=Mod(Get(j,k,k,m,0,1)+Get(j,k,k+1,m,0,0)),
				f[i][j][k][1][0]=Mod(Get(1,j,j,k,1,0)+Get(1,j-1,j,k,0,0)),
				f[i][j][k][0][0]=Mod(Get(j,k,j,k,0,0)+1),
				Add(f[i][j][k][1][1],Get(1,j-1,k,m,0,1)),
				Add(f[i][j][k][1][1],Get(1,j,k+1,m,1,0)),
				Add(ans,f[i][j][k][0][0]),
				Add(ans,f[i][j][k][0][1]),
				Add(ans,f[i][j][k][1][0]),
				Add(ans,f[i][j][k][1][1]);
	}
	printf("%d\n",ans);
	return 0;
}
