#define rep(i,x,y) for(int i=x;i<=y;++i)

int ans,i,j,k,l,n,m,x,y,f[155][155][4],g[155][155][4],l1,l2,r1,r2;
void plus(int &x,int y){
	x=((int64)x+y)%mo;
}
int calc(int l1,int r1,int l2,int r2,int k){
	if(l1>r1 || l2>r2)return 0;
	return ((int64)f[r1][r2][k]-f[r1][l2-1][k]-f[l1-1][r2][k]+f[l1-1][l2-1][k])%mo;
}
