#include<cstdio>
#include<cstring>
#define rep(i,l,r) for (int i=l;i<=r;i++)
#define update(x,y) x=(0LL+x+y)%oo
const int N=155,oo=1000000007;
int n,m,ans,f[N][N][4],g[N][N][4];
int rec(int k,int x1,int x2,int y1,int y2)
{
	if (x1>x2 || y1>y2) return 0;
	return (0LL+f[x2][y2][k]+f[x1-1][y1-1][k]
			-f[x1-1][y2][k]-f[x2][y1-1][k])%oo;
}
int main()
{
	scanf("%d%d",&n,&m);
	while (m--){
		rep(i,1,n) rep(j,1,n) rep(k,0,3)
			update(f[i][j][k],f[i-1][j][k]+f[i][j-1][k]-f[i-1][j-1][k]);
		memset(g,0,sizeof(g));
		rep(i,1,n) rep(j,i,n){
			rep(k,0,3)
				update(g[i][j][0],rec(k,1,i-(k&1),j+((k&2)>0),n));
			rep(k,1,3) if (k&1)
				update(g[i][j][1],rec(k,i,j,j+(k==3),n));
			rep(k,2,3)
				update(g[i][j][2],rec(k,1,i-(k==3),i,j));
			update(g[i][j][3],rec(3,i,j,i,j));
			}
		memcpy(f,g,sizeof(f));
		rep(i,1,n) rep(j,i,n) ++f[i][j][3];
		rep(i,1,n) rep(j,i,n) rep(k,0,3) update(ans,f[i][j][k]);
		}
	printf("%d\n",(ans+oo)%oo); return 0;
}
