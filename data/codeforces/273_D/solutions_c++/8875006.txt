#include<cstdio>
#include<cstring>
using namespace std;

typedef long long ll;

const int N = 155 , mod = 1000000007;

#define fo(i , st , en) for (int i = st; i <= en; i++)
#define fd(i , st , en) for (int i = st; i >= en; i--)
#define Me(x , y) memset(x , y , sizeof(x))

int n , m , ans;
int f[N][N][N][3];

int main(){
	scanf("%d%d" , &n , &m); ans = 0;
	fo (i , 1 , m) f[1][i][i][0] = 1;
	fo (i , 1 , n){
		fo (j , 1 , m)
			fo (k , j , m){
				int sum = ((f[i][j][k][0] + f[i][j][k][1]) % mod + f[i][j][k][2]) % mod;
				ans = (ans + (ll)sum * (n - i + 1) % mod * (m - k + 1)) % mod;
			}
		fo (j , 1 , m) f[i + 1][j][j][0] = f[i][j][j][0];
		fo (j , 1 , m) f[i + 1][j][j][0] = (f[i + 1][j][j][0] + f[i + 1][j - 1][j - 1][0]) % mod;
		fo (j , 1 , m) f[i + 1][j][j][0] = (f[i + 1][j][j][0] + f[i + 1][j - 1][j - 1][0]) % mod;
		fo (j , 1 , m)
			fo (k , j , m){
				f[i + 1][j][k][1] = (f[i][j][k][1] + f[i][j + 1][k][0] * 2 % mod) % mod;
			}
		fd (j , m , 1)
			fo (k , j , m)
				f[i + 1][j][k][1] = (f[i + 1][j][k][1] + f[i + 1][j + 1][k][1]) % mod;
		fo (j , 1 , m)
			fo (k , j , m)
				f[i + 1][j][k][1] = (f[i + 1][j][k][1] + f[i + 1][j - 1][k - 1][1]) % mod;
		fo (j , 1 , m)
			fo (k , j , m)
				f[i + 1][j][k][2] = ((f[i][j][k][2] + f[i][j + 1][k][1]) % mod + f[i][j + 2][k][0]) % mod;
		fd (j , m , 1)
			fo (k , j , m)
				f[i + 1][j][k][2] = (f[i + 1][j][k][2] + f[i + 1][j + 1][k][2]) % mod;
		fd (j , m , 1)
			fo (k , j , m)
				f[i + 1][j][k][2] = (f[i + 1][j][k][2] + f[i + 1][j + 1][k][2]) % mod;
	}
	printf("%d\n" , ans);
	return 0;
}
