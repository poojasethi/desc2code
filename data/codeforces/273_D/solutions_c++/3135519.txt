#include <cstdio>
#include <vector>
#include <algorithm>
#define N 160
#define M 1000000007
using namespace std;

struct INT{ int x; INT(){x = 0;} INT(int _x){x = _x;} };
INT operator+(INT a, INT b){ return INT((a.x+b.x)%M); }
INT operator-(INT a, INT b){ return INT((a.x-b.x+M)%M); }
INT dp[N][N][N][2][2]; int n, m;

int main (){
	scanf("%d%d", &n, &m);
	dp[0][0][0][0][0] = 1;
	for(int i = 1; i <= m; i++){
		dp[i][0][0][0][0] = dp[i-1][1][n][0][0]+dp[i-1][1][n][0][1]+dp[i-1][1][n][1][0]+dp[i-1][1][n][1][1]+dp[i-1][0][0][0][0];
		for(int j = 1; j <= n; j++){ // start from
			for(int k = j; k <= n; k++){ // stop at
				dp[i][j][k][0][0] = dp[i-1][j][k][0][0]+1;

				dp[i][j][k][0][1] = dp[i-1][j][n][0][0]-dp[i-1][j][k][0][0]-dp[i-1][k+1][n][0][0]
								  + dp[i-1][j][n][0][1]-dp[i-1][j][k-1][0][1]-dp[i-1][k+1][n][0][1];
				
				dp[i][j][k][1][0] = dp[i-1][1][k][0][0]-dp[i-1][j][k][0][0]-dp[i-1][1][j-1][0][0]
								  + dp[i-1][1][k][1][0]-dp[i-1][j+1][k][1][0]-dp[i-1][1][j-1][1][0];
				
				dp[i][j][k][1][1] = dp[i-1][1][n][0][0]-dp[i-1][j][n][0][0]-dp[i-1][1][k][0][0]+dp[i-1][j][k][0][0]
								  + dp[i-1][1][n][0][1]-dp[i-1][j][n][0][1]-dp[i-1][1][k-1][0][1]+dp[i-1][j][k-1][0][1]
								  + dp[i-1][1][n][1][0]-dp[i-1][j+1][n][1][0]-dp[i-1][1][k][1][0]+dp[i-1][j+1][k][1][0]
								  + dp[i-1][1][n][1][1]-dp[i-1][j+1][n][1][1]-dp[i-1][1][k-1][1][1]+dp[i-1][j+1][k-1][1][1];
			}
		}
		for(int j = 1; j <= n; j++) for(int k = j; k <= n; k++) for(int p = 0; p < 2; p++) for(int q = 0; q < 2; q++)
			dp[i][j][k][p][q] = dp[i][j][k][p][q]+dp[i][j][k-1][p][q];
		for(int j = n; j >= 1; j--) for(int k = j; k <= n; k++) for(int p = 0; p < 2; p++) for(int q = 0; q < 2; q++)
			dp[i][j][k][p][q] = dp[i][j][k][p][q]+dp[i][j+1][k][p][q];

	}
	INT ans = dp[m][0][0][0][0];
	for(int p = 0; p < 2; p++) for(int q = 0; q < 2; q++) ans = ans+dp[m][1][n][p][q];
	printf("%d\n", (ans.x-1+M)%M);
}
