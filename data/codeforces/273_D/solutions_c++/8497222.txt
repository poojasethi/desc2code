#include <cstdio>
typedef long long ll;
const int mo=(int)1e9+7,N=155;
int n,m;
int f[N][N][N][2][2];
inline ll sum(int i,int x1,int x2,int y1,int y2,int x,int y){
	return 0LL+f[i][x1][y1][x][y]-f[i][x1][y2][x][y]-f[i][x2][y1][x][y]+f[i][x2][y2][x][y];
}
ll ans;
int main(){
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++){
		for(int l=1;l<=m;l++)
			for(int r=l;r<=m;r++){
				f[i][l][r][0][0]=((i==1?1:0)+sum(i-1,l-1,r,l-1,r,0,0))%mo;
				f[i][l][r][0][1]=(sum(i-1,l-1,r,r,m,0,0)+sum(i-1,l-1,r,r-1,m,0,1))%mo;
				f[i][l][r][1][0]=(sum(i-1,0,l-1,l-1,r,0,0)+sum(i-1,0,l,l-1,r,1,0))%mo;
				f[i][l][r][1][1]=(sum(i-1,0,l-1,r,m,0,0)+sum(i-1,0,l-1,r-1,m,0,1)
				+sum(i-1,0,l,r,m,1,0)+sum(i-1,0,l,r-1,m,1,1))%mo;
			}
		for(int l=1;l<=m;l++)
			for(int r=1;r<=m;r++)
				for(int x=0;x<2;x++)
					for(int y=0;y<2;y++)
						f[i][l][r][x][y]=(0LL+
							 f[i][l-1][r][x][y]
							+f[i][l][r-1][x][y]
							-f[i][l-1][r-1][x][y]
							+f[i][l][r][x][y])%mo;
		for(int x=0;x<2;x++)
			for(int y=0;y<2;y++)
				(ans+=1LL*(n-i+1)*f[i][m][m][x][y])%=mo;
	}
	printf("%d\n",int((ans+mo)%mo));
}
