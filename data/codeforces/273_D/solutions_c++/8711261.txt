#include<cstdio>
#include<cstring>
#define LL long long
const int p=1000000007;
int f[155][155][155][2][2],s[155][155][155][2][2];
int n,m;
inline void add(int &x,int y)
{
	x+=y;
	if (x>=p) x-=p;
}
inline void subtract(int &x,int y)
{
	x-=y;
	if (x<0) x+=p;
}
int getsum(int i,int j0,int k0,int j1,int k1,int _x,int _y)
{
	int A,x,y;
	A=0;
	for (x=0;x<=_x;x++)
		for (y=0;y<=_y;y++)
		{
			add(A,s[i][j1-_x+x][k1][x][y]);
			subtract(A,s[i][j0-1][k1][x][y]);
			subtract(A,s[i][j1-_x+x][k0-1+_y-y][x][y]);
			add(A,s[i][j0-1][k0-1+_y-y][x][y]);
		}
	return A;
}
int main()
{
	int i,j,k,x,y,l0,r0,l1,r1,ans;
	scanf("%d%d",&n,&m);
	memset(f,0,sizeof(f));
	memset(s,0,sizeof(s));
	for (i=1;i<=m;i++)
		for (j=i;j<=m;j++) f[1][i][j][0][0]=1;
	for (i=2;i<=n;i++)
	{
		for (j=1;j<=m;j++)
			for (k=1;k<=m;k++)
				for (x=0;x<=1;x++)
					for (y=0;y<=1;y++)
					{
						s[i-1][j][k][x][y]=f[i-1][j][k][x][y];
						add(s[i-1][j][k][x][y],s[i-1][j-1][k][x][y]);
						add(s[i-1][j][k][x][y],s[i-1][j][k-1][x][y]);
						subtract(s[i-1][j][k][x][y],s[i-1][j-1][k-1][x][y]);
					}
		for (j=1;j<=m;j++)
			for (k=j;k<=m;k++)
				for (x=0;x<=1;x++)
					for (y=0;y<=1;y++)
					{
						if (x)
						{
							l0=1; r0=j;
						}
						else
						{
							l0=j; r0=k;
						}
						if (y)
						{
							l1=k; r1=m;
						}
						else
						{
							l1=j; r1=k;
						}
						add(f[i][j][k][x][y],getsum(i-1,l0,l1,r0,r1,x,y));
					}
	}
	ans=0;
	for (i=1;i<=n;i++)
		for (j=1;j<=m;j++)
			for (k=j;k<=m;k++)
				for (x=0;x<=1;x++)
					for (y=0;y<=1;y++) add(ans,((LL)f[i][j][k][x][y]*(n-i+1))%p);
	printf("%d\n",ans); return 0;
}
