#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 160
#define mo 1000000007
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,u,v;
int f[N][N][N][4],x,an,h[N][N][N][4];
inline void M(int &x,int y) {
	x+=y;
	if (x>=mo) x-=mo;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) {
		For(u,0,3)For(v,u,3) if ((v|u)==v)
			For(j,1,m)For(k,j,m) {
				int A=(v&1)>(u&1)?j+1:j;
				int B=(v&2)>(u&2)?k-1:k;
				M(f[i][A][B][v],f[i-1][j][k][u]);
			}
		For(j,1,m)For(k,j,m)For(u,0,3) h[i][j][k][u]=f[i][j][k][u];
		For(k,1,m)For(j,1,k-1)For(u,0,3) if (u&1) {
			M(f[i][j+1][k][u],f[i][j][k][u]);
		}
		For(j,1,m)For(k,j,m-1)For(u,0,3) if (!(u&2)) {
			M(f[i][j][k+1][u],f[i][j][k][u]);
		}
		For(j,1,m)for (k=m;k>j;k--)For(u,0,3) if (u&2) {
			M(f[i][j][k-1][u],f[i][j][k][u]);
		}
		For(k,1,m)for (j=k;j>1;j--)For(u,0,3) if (!(u&1)) {
			M(f[i][j-1][k][u],f[i][j][k][u]);
		}
		For(j,1,m)For(k,j,m) M(f[i][j][k][0],1);
		For(j,1,m)For(k,j,m)For(u,0,3) M(an,f[i][j][k][u]);
	}
	printf("%d\n",an);
	return 0;
}
