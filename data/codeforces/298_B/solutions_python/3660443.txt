def main():
  t,sx,sy,ex,ey = map(int,raw_input().split())
  
  ts = raw_input()

  for i in range(t):
    if ex>sx and ts[i] == 'E':
      sx += 1
    if ex<sx and ts[i] == 'W':
      sx -= 1
    if ey>sy and ts[i] == 'N':
      sy += 1
    if ey<sy and ts[i] == 'S':
      sy -= 1
    if sx == ex and sy == ey:
      print(i+1)
      return()
      
  print(-1)

main()
