t, sx, sy, ex, ey = map(int, raw_input().split())
s = raw_input()
dirx = 'E' if ex > sx else 'W'
diry = 'N' if ey > sy else 'S'
if s.count(dirx) < abs(sx - ex) or s.count(diry) < abs(sy - ey):
    print '-1'
else:
    cnt = [abs(sx - ex), abs(sy - ey)]
    for i in range(t):
        if cnt[0] > 0 and s[i] == dirx:
            cnt[0] -= 1
        elif cnt[1] > 0 and s[i] == diry:
            cnt[1] -= 1
        if cnt[0] == cnt[1] == 0:
            print i + 1
            break
