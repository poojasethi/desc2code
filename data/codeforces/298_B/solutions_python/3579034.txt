dx = {'N':0, 'S':0, 'W':-1, 'E':1}
dy = {'N':1, 'S':-1, 'W':0, 'E':0}
t, sx, sy, ex, ey = map(int, raw_input().split())
def CanMove(c):
  if c == 'N':
    return ey > sy
  if c == 'S':
    return ey < sy
  if c == 'W':
    return ex < sx
  if c == 'E':
    return ex > sx
s = raw_input()
for i in range(t):
  if CanMove(s[i]):
    sx += dx[s[i]]
    sy += dy[s[i]]
  if (sx, sy) == (ex, ey):
    print i + 1
    break
else:
  print -1
