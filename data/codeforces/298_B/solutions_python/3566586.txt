n,x,y,xx,yy=map(int,raw_input().split())
if xx>=x:
    W,E=0,xx-x
else:
    E,W=0,x-xx
if yy>=y:
    S,N=0,yy-y
else:
    N,S=0,y-yy
# print N,S,E,W
s=raw_input()
i=0
while i<len(s) and any(j!=0 for j in [W,E,S,N]):
    if s[i]=="S" and S>0:
        S-=1
    if s[i]=="N" and N>0:
        N-=1
    if s[i]=="W" and W>0:
        W-=1
    if s[i]=="E" and E>0:
        E-=1
    i+=1
if all(j==0 for j in [W,E,S,N]):
    print i
else:
    print -1
