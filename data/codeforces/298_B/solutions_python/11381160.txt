def mhd(a):
    return abs(a.imag) + abs(a.real)

n, sx, sy, ex, ey = map(int, raw_input().split(" "))
st = raw_input()
tt = [sx + sy * 1j]
et = ex + ey * 1j
tt[-1] -= et
d = {'E': 1,'W':-1,'S':-1j,'N':1j}
for i in st:
    if  mhd(tt[-1] + d[i]) < mhd(tt[-1]):
        tt.append(tt[-1] + d[i])
    else:
        tt.append(tt[-1])
    if tt[-1] == 0:
        print len(tt) - 1
        break
else:
    print -1
