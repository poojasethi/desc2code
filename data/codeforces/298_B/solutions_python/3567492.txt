import sys

t, sx, sy, ex, ey = map(lambda x: int(x), sys.stdin.readline().split())
d = sys.stdin.readline()
for i in xrange(t):
    if d[i] == 'N' and sy < ey:
        sy += 1
    elif d[i] == 'E' and sx < ex:
        sx += 1
    elif d[i] == 'W' and sx > ex:
        sx -= 1
    elif d[i] == 'S' and sy > ey:
        sy -= 1
    if sx == ex and sy == ey:
        print i + 1
        break
if sx != ex or sy != ey:
    print -1
