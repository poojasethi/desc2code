t,sx,sy,ex,ey = map(int,raw_input().split())
w = raw_input()
v,h = ey-sy,ex - sx
for i in xrange(t):
    if v > 0 and w[i] == 'N':
        v -= 1
    elif v < 0 and w[i] == 'S':
        v += 1
    elif h > 0 and w[i] == 'E':
        h -= 1
    elif h < 0 and w[i] == 'W':
        h += 1
    if not v and not h:
        print i+1
        exit(0)
print -1
