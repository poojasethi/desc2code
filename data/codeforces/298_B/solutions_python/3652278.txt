d = {'E': (+1, 0), 'S': (0, -1), 'W': (-1, 0), 'N': (0, +1)}
t, sx, sy, ex, ey = map(int, raw_input().split())
s, e = (sx, sy), (ex, ey)
dist = lambda a, b: abs(a[0]-b[0]) + abs(a[1]-b[1])
t, ans = 0, 0
for c in raw_input():
	t += 1
	ss = tuple(map(sum, zip(s, d[c])))
	if dist(ss, e) < dist(s, e):
		s = ss
		ans = t
print ans if s == e else "-1"
