#!/usr/bin/python

t, sx, sy, ex, ey = [int(i) for i in raw_input().split()]
s = raw_input()
l, ans = abs(sx - ex) + abs(sy - ey), 0

for i in xrange(int(t)):
    nx, ny = sx, sy
    if s[i] == 'S':
        ny-=1
    if s[i] == 'N':
        ny+=1
    if s[i] == 'E':
        nx+=1
    if s[i] == 'W':
        nx-=1
    
    if l > abs(nx - ex) + abs(ny - ey):
        l = abs(nx - ex) + abs(ny - ey)
        sx, sy = nx, ny
        if l == 0:
            print i+1
            exit()
print -1
