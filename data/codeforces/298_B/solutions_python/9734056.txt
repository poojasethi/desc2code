[n,sx,sy,ex,ey] = map(int,raw_input().split())
left = dict()
if(sx > ex):
    left['W'] = sx-ex
elif(sx < ex):
    left['E'] = ex-sx
if(sy > ey):
    left['S'] = sy-ey
elif(sy < ey):
    left['N'] = ey-sy

direct = raw_input()
i = 0
while(len(left) > 0 and i < n):
   c = direct[i]
   if(left.has_key(c)):
       left[c] -= 1
       if(left[c] == 0):
           left.pop(c)
   i += 1
if(len(left) > 0):
    print -1
else:
    print i
