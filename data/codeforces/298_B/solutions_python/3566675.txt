t, sx, sy, ex, ey = map(int, raw_input().strip().split())
w = raw_input().strip()
for i in xrange(t):
    if sx < ex and w[i] == 'E':
        sx += 1
    if sx > ex and w[i] == 'W':
        sx -= 1
    if sy < ey and w[i] == 'N':
        sy += 1
    if sy > ey and w[i] == 'S':
        sy -= 1
    if sx == ex and sy == ey:
        print i + 1
        break
else:
    print -1
