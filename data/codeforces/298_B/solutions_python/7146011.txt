n,sx,sy,ex,ey = [int(c) for c in raw_input().split()]
s = raw_input().strip()
if (sx<ex and s.count('E')<ex-sx) or (sx>ex and s.count('W')<sx-ex) or (sy<ey and s.count('N')<ey-sy) or (sy>ey and s.count('S')<sy-ey):
	print -1
else:
	E = reduce(lambda (c,u),y:[c-(y=='E'),u+(c>0)],s,[ex-sx,0])[1]
	W = reduce(lambda (c,u),y:[c-(y=='W'),u+(c>0)],s,[sx-ex,0])[1]
	N = reduce(lambda (c,u),y:[c-(y=='N'),u+(c>0)],s,[ey-sy,0])[1]
	S = reduce(lambda (c,u),y:[c-(y=='S'),u+(c>0)],s,[sy-ey,0])[1]
	print max(E,W,N,S)
