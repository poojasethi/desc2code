r=raw_input
t,sx,sy,ex,ey=map(int, r().split())
d=r()
x=ex-sx
y=ey-sy
h='WE'[x>0]
v='SN'[y>0]

x=abs(x)
y=abs(y)
hi=0
f=False
while x > 0:
    hi = d.find(h, hi+f)
    if hi == -1:
        print -1
        exit()
    f=True
    x -= 1
vi=0
f=False
while y > 0:
    vi = d.find(v, vi+f)
    if vi == -1:
        print -1
        exit()
    f=True
    y -= 1   
print max(hi, vi)+1
