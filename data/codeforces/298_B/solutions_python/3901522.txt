#!/usr/bin/env python

n, sx, sy, ex, ey = map(int, raw_input().split())
d = raw_input().strip()

cnt = 0
flag = False
for i in d:
    if sx == ex and sy == ey:
        print cnt
        flag = True
        break
    if ex > sx:
        if i == 'E':
            sx += 1
    elif ex < sx:
        if i == 'W':
            sx -= 1
    if ey > sy:
        if i == 'N':
            sy += 1
    elif ey < sy:
        if i == 'S':
            sy -= 1
    cnt += 1
    
if not flag:
    if sx == ex and sy == ey:
        print cnt
    else:
        print -1
