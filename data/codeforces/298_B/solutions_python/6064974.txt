t, sx, sy, ex, ey = map(int, raw_input().split())
winds = raw_input().strip()
dx = ex - sx
dy = ey - sy
#print dx, dy
i = 0
while (i < t) and ((dx <> 0) or (dy <> 0)):
	if (winds[i] == "N") and (dy > 0):
		dy -= 1
	if (winds[i] == "S") and (dy < 0):
		dy += 1
	if (winds[i] == "W") and (dx < 0):
		dx += 1
	if (winds[i] == "E") and (dx > 0):
		dx -= 1
	i += 1
	#print dx, dy
if (dx == 0) and ( dy == 0):
	print i
else:
	print "-1"
