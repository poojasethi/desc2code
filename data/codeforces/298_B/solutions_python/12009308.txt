n,sx,sy,ex,ey=map(int,raw_input().split())

cnt=0
flag=False
for i in raw_input().strip():
    if ex>sx:
        if i=='E':
            sx+=1
    elif ex<sx:
        if i=='W':
            sx-=1
    if ey>sy:
        if i=='N':
            sy+=1
    elif ey<sy:
        if i=='S':
            sy-=1
    cnt+=1
    if ex==sx and sy==ey:
        print cnt
        flag=True
        break
if flag==False:
    print -1
