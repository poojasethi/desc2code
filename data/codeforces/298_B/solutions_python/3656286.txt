t, sx, sy, ex, ey = map(int, raw_input().split())
line = raw_input()
sol = 1
for c in line:
  if c == 'N' and sy < ey: sy += 1
  if c == 'S' and sy > ey: sy -= 1
  if c == 'E' and sx < ex: sx += 1
  if c == 'W' and sx > ex: sx -= 1
  if (sx, sy) == (ex, ey): break
  sol += 1
print sol if sol <= len(line) else -1
