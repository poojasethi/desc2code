t, sx, sy, ex, ey = map(int, raw_input().split())
x, y, m, k = sx - ex, sy - ey, 0, 0
ld = [0] * 4
co = "EWNS"
if x < 0: 
  m |= 1 << 0 
  ld[0] += abs(x)
if x > 0: 
  m |= 1 << 1 
  ld[1] += abs(x)
if y < 0: 
  m |= 1 << 2 
  ld[2] += abs(y)
if y > 0: 
  m |= 1 << 3 
  ld[3] += abs(y)
for c in raw_input():
  k, d = k + 1, co.find(c)
  if (m >> d) & 1:
    if ld[d] > 0: ld[d] -= 1
    if ld == [0] * 4:
      print k
      exit()
print -1
