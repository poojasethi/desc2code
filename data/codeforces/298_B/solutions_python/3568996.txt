t, sx, sy, ex, ey = [int(x) for x in raw_input().split()]
ss = raw_input()
fin = -1
for i in xrange(t):
	if ss[i] == 'W' and sx > ex:
		sx -= 1
	if ss[i] == 'E' and sx < ex:
		sx += 1
	if ss[i] == 'N' and sy < ey:
		sy += 1
	if ss[i] == 'S' and sy > ey:
		sy -= 1
	if sx == ex and sy == ey:
		fin = i+1
		break
print(fin)
