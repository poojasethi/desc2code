from sys import exit
s = raw_input().split(' ')
vector = raw_input()
x = int(s[1]) - int(s[3])
y = int(s[2]) - int(s[4])
result = {'W': (-1, 0), 'E': (1, 0), 'S': (0, -1), 'N': (0, 1)}
t = 0
if x == 0 and y == 0:
    print t
    exit()
for i in vector:
    t += 1
    rx = result[i][0]
    ry = result[i][1]
    if x * rx < 0:
        x += rx
    if y * ry < 0:
        y += ry
    if x == 0 and y == 0:
        print t
        exit()
print -1
