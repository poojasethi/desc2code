I = lambda:map(int , raw_input().split())
a = {'S':[0 , -1] , 'E':[1 , 0] , 'W':[-1 , 0] , 'N':[0 , 1]}
s = [0 , 0]
f = [0 , 0]
t , s[0] , s[1] , f[0] , f[1] = I()
q = raw_input()
for i in range(len(q)):
    if (f[0] - s[0]) * a[q[i]][0] > 0:
        s[0] += a[q[i]][0]
    if (f[1] - s[1]) * a[q[i]][1] > 0:
        s[1] += a[q[i]][1]
    if s[0] == f[0] and s[1] == f[1]:
        print i + 1
        exit(0)

print -1
