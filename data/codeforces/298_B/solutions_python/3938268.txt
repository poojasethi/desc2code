t,s1,s2,e1,e2=map(int,raw_input().split())
s=raw_input()
p=e1-s1
q=e2-s2
faltu=0
count=0
h,v=0,0
for i in s:
	count+=1
	h,v=0,0
	if i=='N':
		v=1
	elif i=='S':
		v=-1
	else:
		v=0
	if i=='E':
		h=1
	elif i=='W':
		h=-1
	else:
		h=0
	
	if p*h>0:
		p-=h
	if q*v>0:
		q-=v
	
	if p==0 and q==0:
		faltu=1
		break
if faltu==1:
	print count
else:
	print "-1"
