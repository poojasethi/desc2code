t, sx, sy, ex, ey = map(int, raw_input().strip().split(' '))
dirs = raw_input().strip()

cx = sx
cy = sy

for i, d in enumerate(dirs):
    if d == 'N' and cy < ey:
        cy += 1
    elif d == 'E' and cx < ex:
        cx += 1
    elif d == 'S' and cy > ey:
        cy -= 1
    elif d == 'W' and cx > ex:
        cx -= 1

    if cx == ex and cy == ey:
        print i + 1
        exit()

print -1
