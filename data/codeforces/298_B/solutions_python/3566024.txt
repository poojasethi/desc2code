import sys

t, sx, sy, ex, ey = [ int(x) for x in raw_input().split() ]
wind = raw_input()

xChange = ex - sx
yChange = ey - sy

result = -1

for i in xrange( 0, t ):
	xChangeNew = xChange
	yChangeNew = yChange
	if wind[i] == 'E':
		xChangeNew -= 1
	elif wind[i] == 'W':
		xChangeNew += 1
	elif wind[i] == 'S':
		yChangeNew += 1
	elif wind[i] == 'N':
		yChangeNew -= 1

	if abs( xChangeNew ) < abs(xChange):
		xChange = xChangeNew
	if abs( yChangeNew ) < abs(yChange):
		yChange = yChangeNew

	if xChange == 0 and yChange == 0:
		result = i + 1
		break

print result
