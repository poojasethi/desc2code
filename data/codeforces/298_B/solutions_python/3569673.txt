#!/usr/bin/env python
import sys

t, sx, sy, ex, ey = map(int, sys.stdin.readline().strip().split())
directions = sys.stdin.readline().strip()

achieved = False
for i in xrange(t):
    if sx == ex and sy == ey:
        print i
        achieved = True
        break 

    direc = directions[i]
    if direc == 'E' and ex - sx > 0:
        sx += 1
    elif direc == 'S' and sy - ey > 0:
        sy -= 1
    elif direc == 'W' and sx - ex > 0:
        sx -= 1
    elif direc == 'N' and ey - sy > 0:
        sy += 1

if not achieved and sx == ex and sy == ey:
    print t
elif not achieved:
    print -1
