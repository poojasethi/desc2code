#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
char a[100005];
int n,sx,sy,ex,ey;
int main()
{
	cin>>n>>sx>>sy>>ex>>ey;
	scanf("%s",&a[1]);
	for(int i=1;i<=n+1;i++){
		if(sx==ex&&sy==ey){cout<<i-1;return 0;}
		if(a[i]=='E'&&sx<ex)sx++;
		if(a[i]=='S'&&sy>ey)sy--;
		if(a[i]=='W'&&sx>ex)sx--;
		if(a[i]=='N'&&sy<ey)sy++;
	}
	cout<<-1;return 0;
}
