#include <iostream>
using namespace std;
int main(){
	int i,sx,sy,ex,ey,n;
	char s;
	bool g=true;
	cin>>n>>sx>>sy>>ex>>ey;
	for (i=1; i<=n; i++){
		cin>>s;
		if (s=='E' && sx<ex)sx++;
		else if (s=='S' && sy>ey)sy--;
		else if (s=='W' && sx>ex)sx--;
		else if (s=='N' && sy<ey)sy++;
		if (sx==ex && sy==ey){cout<<i; return 0;}
		
	}
	cout<<"-1";
	return 0;
}
