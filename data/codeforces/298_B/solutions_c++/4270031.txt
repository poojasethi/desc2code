#include<cstdio>
#include<iostream>

using namespace std;

int t,x,y,hx,hy;
char w[100005];

int main()
{
	int i;
	scanf("%d %d %d %d %d %s",&t,&x,&y,&hx,&hy,w);
	for(i=0;i<t;i++){
		if(x==hx && y==hy) break;
		if(w[i]=='N' && y<hy)
			y++;
		else if(w[i]=='S' && y>hy)
			y--;
		else if(w[i]=='W' && x>hx)
			x--;
		else if(w[i]=='E' && x<hx)
			x++;
	}
	if(x==hx && y==hy) printf("%d\n",i);
	else printf("-1\n");
	return 0;
}
