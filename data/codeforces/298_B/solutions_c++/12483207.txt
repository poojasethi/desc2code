#include<bits/stdc++.h>
using namespace std;
int sx,sy,ex,ey,t,n,fs,fw,fn,fe,l1,l2,last[256],H[256],x,y;
char c;
int main(){
	cin >> t >> sx >> sy >> ex >> ey;

	for(int i=0;i<t;i++){
		cin >> c ;
		if(c=='N' and sy < ey) sy++;
		if(c=='E' and sx < ex) sx++;
		if(c=='W' and sx > ex) sx--;
		if(c=='S' and sy > ey) sy--;
		if(sx == ex and sy == ey){
			cout << i+1 << endl;
			return 0;
		}

	}
	cout << -1 << endl;
}
