#include<iostream>
#include<cstdio>
using namespace std;
int t,x,y,a,b;
string c;
int main()
{
    cin>>t>>x>>y>>a>>b;
    cin>>c;
    for(int i=0;i<t;i++)
    {
        if(c[i]=='N' && b>y){y++;}
        else if(c[i]=='S' && b<y){y--;}
        else if(c[i]=='E' && x<a){x++;}
        else if(c[i]=='W' && x>a ){x--;}
        if(a==x && b==y){cout<<i+1;goto abc;}
    }
    cout<<-1;
    abc:;

}
