#include <iostream>
#include <string.h>
#include <math.h>
using namespace std;
int main(){
int t,sx,sy,ex,ey,j;
char s[100000+2];
cin>>t>>sx>>sy>>ex>>ey;
cin>>s;
for(j=0; j<t; j++){
  if ( ex>sx && s[j]=='E' ) sx++;
  else if ( ex<sx && s[j]=='W' ) sx--;
  else if ( ey>sy && s[j]=='N' ) sy++;
  else if ( ey<sy && s[j]=='S' ) sy--;
  if ( sx==ex && sy==ey ) { return cout<<j+1, 0; }
}
cout<<"-1";
return 0;
}
