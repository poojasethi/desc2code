#include<iostream>
#include<cstdio>
using namespace std;
int t,x,y,a,b;
char c;
int main()
{
	cin>>t>>x>>y>>a>>b;
	for(int i=0;i<t;i++)
	{
		cin>>c;
		if(c=='N' && b>y)
			y++;
		else if(c=='S' && b<y)
			y--;
		else if(c=='E' && x<a)
			x++;
		else if(c=='W' && x>a )
			x--;
		if(a==x && b==y)
		{
			cout<<i+1;
			return 0;
		}
	}
	cout<<-1;
	return 0;
}
