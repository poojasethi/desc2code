#include <iostream>

using namespace std;

int main()
{

    int t,x,y,X,Y,i;
    cin>>t>>x>>y>>X>>Y;

    for (i=0;i<t;i++)
    {
        char a;
        cin>>a;
        if (x==X && y==Y)    break;
        if (a=='E' && x<X )  x++;
        if (a=='S' && y>Y )  y--;
        if (a=='W' && x>X )  x--;
        if (a=='N' && y<Y )  y++;
    }

    if (x==X && y==Y ) cout<<i;
    else cout<<-1;

    return 0;
}
