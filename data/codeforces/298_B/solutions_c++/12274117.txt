# include <bits/stdc++.h>

using namespace std;

int main(){
	int t, sx, sy, ex, ey;
	cin >> t >> sx >> sy >> ex >> ey;

	for(int i=1 ; i<=t ; i++){
		char w;
		cin >> w;
		if(w == 'E' and sx < ex)
			sx++;
		if(w == 'S' and sy > ey)
			sy--;
		if(w == 'W' and sx > ex)
			sx--;
		if(w == 'N' and sy < ey)
			sy++;

		if(sx == ex and sy == ey){
			cout << i << endl;
			return 0;
		}
	}
	
	cout << "-1" << endl;

	return 0;
}
