#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int sx,sy,ex,ey;
	cin>>sx>>sy>>ex>>ey;
	char a[n+1];
	int i;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		if(a[i]=='E'&&sx<ex)
			sx++;
		else if(a[i]=='N'&&sy<ey)
			sy++;	
		else if(a[i]=='W'&&sx>ex)
			sx--;
		else if(a[i]=='S'&&sy>ey)
			sy--;
		if(sx==ex&&sy==ey)
		{
			cout<<i;
			return 0;
		}
	}
	cout<<"-1";
	return 0;
}
