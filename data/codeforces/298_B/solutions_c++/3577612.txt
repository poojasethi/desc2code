#include <iostream>

using namespace std;
int main(void)
{
	long int t,sx,sy,ex,ey;
	char c;
	cin>>t>>sx>>sy>>ex>>ey;
	for(long int i=1;i<=t;i++)
	{
		cin>>c;
		if(c=='E'&& sx<ex) sx++;
		if(c=='S'&& sy>ey) sy--;
		if(c=='W'&& sx>ex) sx--;
		if(c=='N'&&sy<ey)  sy++;
		if(sx==ex&&sy==ey)
		{cout<<i<<endl;return 0;}
	}
	cout<<"-1"<<endl;
	return 0;
}
