#include<stdio.h>

int t,sx,sy,ex,ey,i;
char str;

int main()
{
	scanf("%d%d%d%d%d\n",&t,&sx,&sy,&ex,&ey);
	for(i=0;i<t;i++)
	{
		scanf("%c",&str);
		if(str=='E' && sx<ex)
			sx++;
		if(str=='S' && sy>ey)
			sy--;
		if(str=='W' && sx>ex)
			sx--;
		if(str=='N' && sy<ey)
			sy++;
		if(sx==ex && sy==ey)
		{
			printf("%d\n",i+1);
			return 0;
		}
	}
	printf("-1\n");
	return 0;
}
