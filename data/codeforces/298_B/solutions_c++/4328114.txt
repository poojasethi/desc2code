#include <iostream>
using namespace std;
int n,x,y,a,b,t,i;
char s[100005];
int main()
{
    cin>>n>>x>>y>>a>>b>>s;
    for(i=0;i<n;i++)
    {
        if(x==a&&y==b)break;t++;
        if(s[i]=='N'&&y<b)y++;
        if(s[i]=='E'&&x<a)x++;
        if(s[i]=='S'&&y>b)y--;
        if(s[i]=='W'&&x>a)x--;

    }
    if(x==a&&y==b)cout<<t;
    else cout<<"-1";
    return 0;
}
