#include<iostream>

using namespace std;

int  main()
{

int t,sx,sy,ex,ey;
int flag=0;
cin >> t >> sx >> sy >> ex >> ey;
char str;
for(int i=1;i<=t;i++)
 {
  cin >> str;
if (str=='S' && sy>ey)
sy--;
else if (str=='E' && sx <ex)
sx++;
else if(str=='N' && sy <ey)
sy++;
else if (str=='W' && sx>ex )
sx--;


if(sx==ex&&sy==ey)
{
cout << i;
flag=1;
break;
}
 }


if(flag==0)
cout<<"-1";

}
