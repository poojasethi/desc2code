#include <iostream>
#define ll long long
using namespace std;

int main()
{
   string s; ll t,sx,sy,ex,ey,q=0;

   cin>>t>>sx>>sy>>ex>>ey>>s;

   for(int i=0; i<s.size(); i++)
   {
       if(sx<ex&&s[i]=='E') sx++;
       if(sx>ex&&s[i]=='W') sx--;
       if(sy<ey&&s[i]=='N') sy++;
       if(sy>ey&&s[i]=='S') sy--;
       if(sx==ex&&sy==ey) {cout<<i+1; return 0;}
   }
   cout<<"-1";
    return 0;
}
