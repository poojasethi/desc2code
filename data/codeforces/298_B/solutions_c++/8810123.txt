#include<iostream>
using namespace std;
int main()
{
	int t,sx,sy,ex,ey;
	char d;
	cin>>t>>sx>>sy>>ex>>ey;
	for(int i=1;i<=t;i++)
	{
		cin>>d;
		if(d=='W'&&sx>ex) sx--;
		else if(d=='S'&&sy>ey) sy--;
		else if(d=='N'&&sy<ey) sy++;
		else if(d=='E'&&sx<ex) sx++;
		if(sx==ex&&sy==ey)
		{
			cout<<i;
			return 0;
		}
	}
	cout<<-1;
	return 0;
}
