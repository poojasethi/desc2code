#include <iostream>
using namespace std;
int main(){
	int t,sx,sy,ex,ey,dx,dy;
	cin>>t>>sx>>sy>>ex>>ey;
	dx=ex-sx;
	dy=ey-sy;
	for(int i=0;i<t;i++){
		char in;
		cin>>in;
		if(dx>0 && in=='E')
			dx--;
		else if(dx<0 && in=='W')
			dx++;
		if(dy>0 && in=='N')
			dy--;
		else if(dy<0 && in=='S')
			dy++;
		if(dx==0 && dy==0){
			cout<<(i+1);
			return 0;
		}
	}
	cout<<-1;
}
