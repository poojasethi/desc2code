#include<cstdio>
using namespace std;
char s[100001];
int main()
{
	int n,a,b,c,d;
	scanf("%d%d%d%d%d%s",&n,&a,&b,&c,&d,s);
	a-=c;b-=d;
	for(int i=0;i<n;++i)
	{
		if(s[i]=='E'&&a<0)++a;
		if(s[i]=='W'&&a>0)--a;
		if(s[i]=='S'&&b>0)--b;
		if(s[i]=='N'&&b<0)++b;
		if(!a&&!b)
		{printf("%d\n",i+1);return 0;}
	}
	puts("-1");
}
