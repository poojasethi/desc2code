#include<iostream>
using namespace std;
int second,x,y,x2,y2;char c;
int main()
{
    cin>>second>>x>>y>>x2>>y2;
    for(int i=1;i<=second;i++)
    {
            cin>>c;
            if(c=='E' && x<x2)x++;
            if(c=='W' && x>x2)x--;
            if(c=='N' && y<y2)y++;
            if(c=='S' && y>y2)y--;
            if(x==x2 && y==y2)
            {cout<<i;return 0;}
    }
    cout<<"-1";

    return 0;
}
