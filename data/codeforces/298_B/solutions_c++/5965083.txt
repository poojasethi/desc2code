#include<stdio.h>
int main(){
	int t,sx,sy,ex,ey,i;
	scanf("%d %d %d %d %d",&t,&sx,&sy,&ex,&ey);
	char a;
	for(i=0 ; i<t ; i++){
		scanf(" %c",&a);
		if(a=='S' && sy>ey)
			sy--;
		if(a=='N' && sy<ey)
			sy++;
		if(a=='W' && sx>ex)
			sx--;
		if(a=='E' && sx<ex)
			sx++;	
		if(sx==ex && sy==ey){
			printf("%d",i+1);
			return 0;
		}
	}
	printf("-1");
	return 0;
}
