#include <bits/stdc++.h>

using namespace std;

int main()
{
char c;
int ans=-1,t;
long int dx,dy;
long int x,y,xx,yy;
cin>>t>>x>>y>>xx>>yy;
dx=xx-x;
dy=yy-y;
for(int j=0;j<t;j++)
 {

  cin>>c;
if(c=='E'){if(dx>0)dx--;}
else if(c=='W'){if(dx<0)dx++;}
else if(c=='N'){if(dy>0)dy--;}
else if(c=='S'){if(dy<0)dy++;}
if(dx==0&&dy==0&&ans==-1)ans=j+1;

 }


cout<<ans;






return 0;
}
