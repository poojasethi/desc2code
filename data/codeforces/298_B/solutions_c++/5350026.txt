#include<iostream>
using namespace std;
int main()
{
	long long  time,sx,sy,ex,ey;
	char x;
	cin>>time>>sx>>sy>>ex>>ey;
	for(int i = 1; i<=time;i++)
	{
		cin>>x;
		if(x=='N' && sy<ey)
			sy++;
		else if(x=='W' && sx>ex)
			sx--;
		else if(x=='S' && sy>ey)
			sy--;
		else if(x=='E' && sx<ex)
			sx++;

		if(sx==ex && sy==ey)
		{
			cout<<i;
			return 0;
		}
	}
	cout<<-1;
}
