#include<bits/stdc++.h>
using namespace std;
int a[100005];
int s[100005];
int t,k,b,c,m=1000000007;
int main()
{
  ios::sync_with_stdio(false);
  cin>>t>>k;
  a[0]=1;
  s[0]=0;
  for(int i=1;i<=100000;i++){
    a[i]=a[i-1];
    if(i-k>=0)
      a[i]+=a[i-k];
    a[i]%=m;
    s[i]=(s[i-1]+a[i])%m;
    }
  for(int i=0;i<t;i++){
    cin>>b>>c;
    cout<<(s[c]-s[b-1]+m)%m<<endl;
  }
}
