#include<bits/stdc++.h>
using namespace std;
const int N=1e5+1;
const long long mod=1e9+7;
long long t,k,x,y,a[N];
int main()
{
	cin>>t>>k;
	a[0]=1;
	int i;
	for(i=1;i<N;++i)
	{
		a[i]=a[i-1]+(i>=k?a[i-k]:0);
		a[i]%=mod;
	}
	for(i=1;i<N;++i)
	{
		a[i]+=a[i-1];
		a[i]%=mod;
	}
	for(i=0;i<t;++i)
	{
		cin>>x>>y;
		cout<<((a[y]-a[x-1])%mod+mod)%mod<<endl;
	}
	return 0;
}
