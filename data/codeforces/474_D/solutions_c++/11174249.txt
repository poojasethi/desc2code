#include <iostream>
using namespace std;
int main()
{
	int n,k,a[100010],x,y,i;
	int modint=1000000007;
	int maxint=100000;
	cin>>n>>k;
	for (i=0;i<k;i++)
		a[i]=1;
	for (i=k;i<=maxint;i++)
		a[i]=(a[i-1]+a[i-k])%modint;
	for (i=1;i<=maxint;i++)
		a[i]=(a[i]+a[i-1])%modint;
	for (i=0;i<n;i++)
	{
		cin>>x>>y;
		cout<<(a[y]-a[x-1]+modint)%modint<<endl;
	}
	return 0;
}
