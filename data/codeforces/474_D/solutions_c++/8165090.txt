#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;

const int mod=1000000007;
int f[111111],a[111111];

int main(){
	int T,k;
	cin>>T>>k;
	f[0]=1;
	for (int i=1;i<=100000;i++){
		f[i]=f[i-1];
		if (i>=k) f[i]+=f[i-k];
		f[i]%=mod;
		a[i]=(a[i-1]+f[i])%mod;
	}
	while (T--){
		int x,y;
		scanf("%d%d",&x,&y);
		printf("%d\n",(a[y]-a[x-1]+mod)%mod);
	}
}
