#include<bits/stdc++.h>
using namespace std;
int main()
{	int t,k,i;
	cin>>t>>k;
	int a[100001];
	for(i=0;i<k;i++) a[i]=1;
	for(i=k;i<100001;i++) a[i]=(a[i-k]+a[i-1])%1000000007;
	for(i=1;i<100001;i++) a[i]=(a[i]+a[i-1])%1000000007;
	int l,r;
	while(t--)
	{
		cin>>l>>r;int ans=0;
		ans=a[r]-a[l-1];
		if(ans<0) ans+=1000000007;
		cout<<ans<<endl;
	}
	return 0;
}
