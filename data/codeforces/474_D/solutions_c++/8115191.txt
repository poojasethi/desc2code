#include <bits/stdc++.h>
using namespace std;
const int N=1e5+1;
const long long MOD=1e9+7;
long long t,k,a,b,A[N];
int main() {
	cin>>t>>k;
	A[0]=1;
	for(int i=1;i<N;++i) {
		A[i]=A[i-1]+(i>=k?A[i-k]:0);
		A[i]%=MOD;
	}
	for(int i=1;i<N;++i) {
		A[i]+=A[i-1];
		A[i]%=MOD;
	}
	for(int i=0;i<t;++i) {
		cin>>a>>b;
		cout<<((A[b]-A[a-1])%MOD+MOD)%MOD<<"\n";
	}
	return 0;
}
