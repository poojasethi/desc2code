#include <bits/stdc++.h>

using namespace std;

typedef long long int LL;

int main()
{
	LL t,k,i,ans,a,b,c,x,A[100001],B[100001];
	cin>>t>>k;
	B[0]=0;
	for(i=0;i<=100000;i++)
	{
		if(i<k) A[i]=1;
		else A[i]=(A[i-1]+A[i-k])%1000000007;
		if(i>0) B[i]=A[i]+B[i-1];
	}
	A[0]=0;
	while(t--)
	{
		cin>>a>>b;
		cout<<(B[b]-B[a-1])%1000000007<<endl;
	}
	return 0;
}
