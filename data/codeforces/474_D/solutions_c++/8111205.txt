#include<stdio.h>
#define M 100000
int f[M+200];
int s[M+200];
int mod=1000000007;
int t,k;
main(){
	scanf("%d %d",&t,&k);
	f[0] = 1;
	for(int i=1;i<=M;i++){
		f[i] = f[i-1];
		if(i>=k) f[i]=(f[i]+f[i-k])%mod;
	}
	for(int i=1;i<=M;i++)
		s[i] = (s[i-1]+f[i])%mod;
	while(t>0){
		int a,b;
		scanf("%d %d",&a,&b);
		printf("%d\n",( (s[b]-s[a-1])%mod+mod)%mod);
		t--;
	}
}
