#include <iostream>
using namespace std;
int main()
{
	int n,k,a[100010],x,y,i;
	int mi=1000000007;
	int mai=100000;
	cin>>n>>k;
	for(int i=0;i<k;i++)
		a[i]=1;
	for(int i=k;i<=mai;i++)
		a[i]=(a[i-1]+a[i-k])%mi;
	for(i=1;i<=mai;i++)
		a[i]=(a[i]+a[i-1])%mi;
	for(int i=0;i<n;i++)
	{
		cin>>x>>y;
		cout<<(a[y]-a[x-1]+mi)%mi<<endl;
	}
	return 0;
}
