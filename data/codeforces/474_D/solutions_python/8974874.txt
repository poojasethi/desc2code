cases,white= map(int,raw_input().split())
dp =[1]*100001

for i in range(white,100001):
    dp[i]=(dp[i-white]+dp[i-1])%((10**9)+7)
for i in range(1,100001):
    dp[i]=(dp[i] + dp[i-1]) %((10**9)+7)
   
for i in range(cases):
    start,end=map(int,raw_input().strip().split())
    print (dp[end]-dp[start-1])%((10**9)+7)
