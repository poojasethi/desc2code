#include<stdio.h>
int gcd(int a,int b)
{
    if(b==0) return a;
    else return gcd(b,a%b);
}
int main()
{
    int n,k,m;
    scanf("%d%d",&n,&m);
    k=gcd(n,m);
    n/=k;m/=k;
    if(n>m)  m++;
    else n++;
    if(n==m) printf("Equal\n");
    else if(n>m) printf("Masha\n");
    else printf("Dasha\n");
}
