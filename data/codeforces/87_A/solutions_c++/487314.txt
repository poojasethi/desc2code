#include<cstdio>
#include<algorithm>
using namespace std;


int main()
{
	int a,b;
	scanf("%d %d",&a,&b);
	
	long long g = __gcd(a,b);
	long long l = ((long long)a * b)/g;
	
	//a <= b
	int pa = l/a;
	int pb = l/b;
	if(abs(pb - pa) == 1)
	printf("Equal\n");
	else if(pb > pa)
	printf("Masha\n");
	else
	printf("Dasha\n");
	return 0;
}
