#include <iostream>
#include <algorithm>
using namespace std;

int gcd(int a, int b)
{
	if (b == 0) return a;
	return gcd(b, a%b);
}

int main()
{
	int a, b;
	cin >> a >> b;
	int g = gcd(a,b);
	a /= g;
	b /= g;
	if (abs(a-b) == 1)
		cout << "Equal" << endl;
	else if (a > b)
		cout << "Masha" << endl;
	else
		cout << "Dasha" << endl;
}
