#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
int gcd(int j,int k){return j%k? gcd(k,j%k): k;}
int main(void)
{
    int a,b;
    scanf("%d%d", &a,&b);
    int x = gcd(a,b);
    a/=x;
    b/=x;
    if (abs(a-b)<= 1) puts("Equal");
    else puts(a < b ?"Dasha":"Masha");
    
    return 0;
}
