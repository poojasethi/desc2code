#include<iostream>
#include<cmath>

using namespace std;

int gcd(int a,int b)
{
	while( a%b != 0 )
	{
		a %= b ;
		swap(a,b);
	}

	return b;
}

int main()
{
	int a,b,g;

	cin>>a>>b;

	g = gcd(a,b);

	if( abs(a/g -b/g) == 1 )
	{
		cout<<"Equal"<<endl;
	}
	else if (b/g > a/g)
	{
		cout<<"Dasha"<<endl;
	}
	else cout<<"Masha"<<endl;
}
