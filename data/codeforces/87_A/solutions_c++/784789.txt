#include<cstdio>
int gcd(int a,int b)
{
    if(b==0)return a;
    return gcd(b,a%b);
}
int main()
{
    int a,b;
    scanf("%d%d",&a,&b);
    int q=gcd(a,b);
    a/=q,b/=q;
    if(a==b-1||b==a-1)puts("Equal");
    else if(a>b)puts("Masha");
    else puts("Dasha");
    return 0;
}
