#include<cstdio>
using namespace std;
int gcd(int a,int b){
if(a%b==0)return b;
else return gcd(b,a%b);
}
int main(){
int n,m,a;
scanf("%d%d",&n,&m);
if(n>m)a=gcd(n,m);else a=gcd(m,n);
n/=a;m/=a;
if(n>m)n-=1;else m-=1;
if(m>n)printf("Dasha\n");
else if(m<n)printf("Masha\n");
else printf("Equal\n");
return 0;
}
