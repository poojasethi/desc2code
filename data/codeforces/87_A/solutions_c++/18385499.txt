#include<bits/stdc++.h>
using namespace std;
long long x,y,a,b;
int main(){
    cin>>a>>b;
    x=a*b;
    y=__gcd(a,b);
    x/=y;
    if(a<b){
        if(a>=((x/a)-1)*(b-a))cout<<"Equal"<<endl;
        else cout<<"Dasha"<<endl;
    }
    else{
        if(b>=((x/b)-1)*(a-b))cout<<"Equal"<<endl;
        else cout<<"Masha"<<endl;
    }
}
