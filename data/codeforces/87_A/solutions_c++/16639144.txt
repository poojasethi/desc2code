
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

LL gcd( LL a, LL b)
{
if(b==0)
	return a;

return gcd(b,a%b);


}



int main()
{
LL a,b;
cin>>a>>b;
 
LL l=gcd(max(a,b),min(a,b));

l=a*b/l;

if(l/min(a,b)-l/max(a,b)==1)
	cout<<"Equal"<<endl;
else if(a>b)
	cout<<"Masha"<<endl;
else 
	cout<<"Dasha"<<endl;



 }
