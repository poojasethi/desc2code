#include <cstdio>

int FPB(int a, int b)
{
    if (b == 0) return a;
    return FPB (b, a % b);
}

int main()
{
    int a, b;
    scanf("%d %d", &a, &b);
    int f = FPB(a, b);
    a /= f;
    b /= f;
    if (a < b) a++; else b++;
    if (a > b) printf("Masha"); else if (a < b) printf("Dasha"); else printf("Equal");
    return 0;
}
