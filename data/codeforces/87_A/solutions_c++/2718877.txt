#include <stdio.h>
int gcd(int a,int b)
{
    return b==0?a:gcd(b,a%b);
}
int main()
{
    int a,b,d;
    scanf("%d%d",&a,&b);
    d=gcd(a,b);
    a/=d,b/=d;
    if((a-b)==1||(b-a)==1)
    {
        printf("Equal\n");
        return 0;
    }
    if(a<b)
        printf("Dasha\n");
    else
        printf("Masha\n");
    return 0;
}
