#include <cstdlib>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long LL;

int main() {
	int a, b;
	scanf("%d%d",&a,&b);
	LL lcm = (LL) a / __gcd(a, b) * b;
	int ca = lcm / a;
	int cb = lcm / b;
	if (a < b)
		cb++;
	if (a > b)
		ca++;
	puts((ca == cb) ? "Equal" : (ca > cb) ? "Dasha" : "Masha");
	return 0;
}
