#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int main()
{
    int a,b;
    cin>>b>>a;
    for(int i=2;i<=a&&i<=b;i++)
	if(a%i==0&&b%i==0)
	a/=i,b/=i,i--;
    if(abs(a-b)==1)
		cout<<"Equal"<<endl;
	else
	{
		if(a>b)
          cout<<"Dasha"<<endl;
		else
		  cout<<"Masha"<<endl;
	}
    return 0;
}
