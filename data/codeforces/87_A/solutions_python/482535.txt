import math

def gcd(a,b) :
  if a % b == 0 : 
    return b
  else :
    return gcd(b,a%b)

def main():
  a,b = map(int,raw_input().split())
  lm = a*b/gcd(a,b)
  xa = lm/a 
  xb = lm/b
  if a>b : 
     xb -= 1
  else :
     xa -= 1
  
  if xa > xb : print "Dasha"
  if xa < xb : print "Masha"
  if xa == xb : print "Equal"


if __name__ == "__main__" :
    main()
