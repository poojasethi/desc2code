#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

MAX = 10**6

DASHA = 'Dasha'
MASHA = 'Masha'
EQUAL = 'Equal'

def gcd(a,b):
    while b:
        a, b = b, a % b
    return a

def gcd_and_lcm(a,b):
    g = gcd(a,b)
    return (g, (a/gcd(a,b))*b)
    
#def lcm(a,b):
#    return (a/gcd(a,b))*b
#def GCD(numbers):
#    return reduce(lambda a,b: gcd(a,b), numbers)
#def LCM(numbers):
#    return reduce(lambda a,b: lcm(a,b), numbers)

def sum_to(n):
    return n*(n+1)/2

def solve(a,b):
    G,L = gcd_and_lcm(a, b)
    
    M = max(a,b)
    m = min(a,b)
    
    n = L/M - 1
    tM = sum_to(n) * G + m
    tm = L - tM
    
    if a<b:
        D = tm
        M = tM
    else:
        D = tM
        M = tm

    if D>M:
        return DASHA
    elif M>D:
        return MASHA
    else:
        return EQUAL

S = input.readline().split(' ')
a = int(S[0])
b = int(S[1])
assert 1<=a and a<=MAX 
assert 1<=b and b<=MAX
assert a != b 

answer = solve(a,b)

output.write('%s\n' % (answer))
