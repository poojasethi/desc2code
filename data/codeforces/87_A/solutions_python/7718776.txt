from fractions import gcd

def LCM(a, b): return (a * b)/gcd(a,b)

inp = raw_input().split()
a, b = int(inp[0]), int(inp[1])

def compute(a, b):
    lcm = LCM(a, b)
    dasha = 0
    masha = 0
    dasha += lcm/a
    masha += lcm/b
    if a < b: dasha -= 1
    else: masha -= 1
    if dasha == masha: return "Equal"
    elif dasha > masha: return "Dasha"
    else: return "Masha"


print compute(a, b)
