import sys
def gcd(a,b):
	while b>0:
		a,b=b,a%b
	return a

read_ints= lambda: map(int,sys.stdin.readline().split())

read_line= lambda: sys.stdin.readline().strip()


a,b=read_ints()
g=gcd(a,b)
x,y=b/g,a/g
if a>b: x+=1
