#include <cstdio>
#include <cstring>
#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <map>
#include <bitset>
using namespace std;

const int N = 1e5 + 5;


int a[N], b[N], c[N];
int bo[N], ans[N];

int main() {
    int T;
    scanf("%d", &T);
    while (T--) {
        int n, m;
        scanf("%d%d", &n, &m);
        for (int i = 1; i <= m; ++i) {
            scanf("%d", &a[i]);
        }
        int first = -1, num = 0;
        for (int i = 1; i < n; ++i) {
            scanf("%d%d", &b[i], &c[i]);
            if (first == -1 && c[i] == 1) first = i;
            if (first == -1 && b[i] == 0) ++num;
            --a[b[i]];
        }
        for (int i = 1; i <= m; ++i)
            ans[i] = bo[i] = 0;
        if (first != -1) {
            for (int i = first; i < n; ++i)
                bo[b[i]] = 1;
            int K = -1;
            for (int i = 1; i <= m; ++i)
                if (bo[i] == 0 && a[i] <= num) {
                    if (K == -1 || a[K] > a[i]) K = i;
                    ans[i] = 1;
                }
            num -= a[K];
            for (int i = first; i < n; ++i)
                if (b[i] == 0) ++num;
        }
    
        for (int i = 1; i <= m; ++i)
            if (a[i] <= num) ans[i] = 1;
        for (int i = 1; i <= m; ++i)
            if (ans[i]) putchar('Y');
            else putchar('N');
        puts("");
    }
    return 0;
}
