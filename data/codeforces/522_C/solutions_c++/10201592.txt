#include <iostream>
#include <climits>

using namespace std;

int a[100000];
int com[100000];
bool w[100000];
long long c;
int k;

void build_copy()
{
    for(int i=0;i<k;i++)
    {
        com[i]=a[i];
        if(a[i]<=c)
            w[i]=true;
    }
}

int main()
{
    int t;
    cin>>t;
    for(int gi=0;gi<t;gi++)
    {
        c=0;

        int m;
        cin>>m>>k;
        for(int i=0;i<k;i++)
        {
            cin>>a[i];
            w[i]=0;
        }
        bool  fl=1;
        bool bm=false;
        for(int i=1;i<m;i++)
        {
            int b,r;
            cin>>b>>r;
            b--;
            if(fl && r)
            {
                build_copy();
                bm=true;
                fl=0;
            }
            if(b!=-1)
            {
                w[b]=false;
                a[b]--;
                if(a[b]==0)
                    fl=0;
            }
            else
                c++;
        }
        if(bm)
        {
            int minw=INT_MAX;
            for(int i=0;i<k;i++)
            {
                if(w[i]&& (com[i]<minw))
                    minw=com[i];
            }
            c-=minw;
        }
        for(int i=0;i<k;i++)
        {
            if(a[i]<=c || w[i])
                cout<<"Y";
            else
                cout<<"N";
        }
        cout<<endl;
    }
}
