#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    int T, m, k, x, y;
    bool b, bb(1), bbb(1);
    cin >> T;
    while (T--)
    {
        bb = 1;
        bbb = 1;

        int s = 0;
        cin >> m >> k;
        m--;

        vector <int> v(k);
        vector <int> t(k, 0);

        for (int i = 0; i<k; i++)
            cin >> v[i];

        int x, b;
        for (int i = 0; i<m; i++)
        {
            cin >> x >> b;
            if (b == 1 && bb && bbb)
            {
                bb = 0;
                for (int j = 0; j<k; j++)
                    if (v[j] <= s)
                        t[j] = 1;

            }

            if (x == 0)
                s++;
            else
            {
                v[x - 1]--;
                t[x - 1] = 0;
                if (v[x - 1] == 0)
                    bbb = 0;
            }

        }

        int mn = 100001, mni;
        for (int i = 0; i<t.size(); i++)
            if (t[i])
                if ( v[i] < mn)
                    mn = v[i];
        if (mn == 100001)
            mn = 0;
        s -= mn;

        for (int i = 0; i<k; i++)
        {
            if (v[i] <= s)
                t[i] = true;

            if (t[i])
                cout << 'Y';
            else
                cout << 'N';
        }

        cout << endl;
        v.clear();
    }

    return 0;
}
