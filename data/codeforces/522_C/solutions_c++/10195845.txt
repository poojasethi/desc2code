#define fname ""

#define ub upper_bound
#define lb lower_bound
#define inf 2000000000
#define INF 2000000000000000000ll
using namespace std;

const int maxn = 101010;

int a[maxn], last[maxn];

bool ans[maxn];

struct Passenger
{
	int r, t;
}b[maxn];

int main()
{
	#ifndef ONLINE_JUDGE
	freopen (fname".in", "r", stdin);
	freopen (fname".out", "w", stdout);
	#endif
	int T;
	int n, m;
	int cnt, mn;
	cin >> T;
	while (T--)
	{
		scanf("%d%d", &n, &m);
		for (int i = 1; i <= m; i++)
		{
			scanf("%d", &a[i]);
			last[i] = 0;
			ans[i] = 0;
		}
		for (int i = 1; i < n; i++)
		{
			scanf("%d%d", &b[i].t, &b[i].r);
			if (b[i].t)
				last[b[i].t] = i;
		}
		cnt = 0;
		mn = inf;
		for (int i = 1; i < n; i++)
		{
			if (b[i].r)
			{
				if (mn == inf)
				{
					for (int j = 1; j <= m; j++)
					{
						if (last[j] < i && a[j] <= cnt)
						{
							ans[j] = 1;
							mn = min(mn, a[j]);
						}
					}
					cnt -= mn;
				}
			}
			if (b[i].t)
				a[b[i].t]--;
			else
				cnt++;
		}
		for (int i = 1; i <= m; i++)
			if (a[i] <= cnt)
				ans[i] = 1;
		for (int i = 1; i <= m; i++)
		{
			if (ans[i])
				putchar('Y');
			else
				putchar('N');
		}
		puts("");
	}
	return 0;
}
