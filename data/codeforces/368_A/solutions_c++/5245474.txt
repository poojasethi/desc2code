#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int m,n,d,i,p;
	cin>>n>>d;
	vector<int> v(n);
	for(i=0;i<n;i++)
		cin>>v[i];
	cin>>m;
	p=0;
	sort(v.begin(),v.end());
	for(i=0;i<n && i<m;i++)
		p+=v[i];
	if(m>n)
		p-=(m-n)*d;
	cout<<p<<endl;
	return 0;
}
