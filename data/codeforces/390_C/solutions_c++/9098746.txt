#include<iostream>
#include<vector>
#include<algorithm>

using namespace std ;

int n , k , w ;
string s ;
int d[11][100010] ;

int main()
{
  cin >> n >> k >> w ;
  cin >> s ;
  for(int i=0 ; i<k ; i++)
    for(int j=0 ; j<n ; j++)
      d[i][j+1] = ((s[j]-'0')^(j%k==i))+ d[i][j] ;
  while(w--)
    {
      int l ,r ;
      cin >> l >> r ;
      cout << d[(l+k-2)%k][r] - d[(l+k-2)%k][l-1] << endl ;
    }
  return 0 ;
}
