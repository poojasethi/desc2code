#include <iostream>
using namespace std;
 
int N,K,W,L;
string S;
int sum[10][100010];
int sum2[100010];
 
void solve() {
    int f,i,j,k,l,x,y;
 
    cin>>N>>K>>W;
    cin>>S;
    L=S.size();
    for(int i=0;i<L;i++) {
        sum[i%K][1+i/K] = sum[i%K][i/K] + (S[i]=='1');
        sum2[i+1] = sum2[i] + (S[i]=='1');
    }
 
    for(int i=0;i<W;i++) {
        cin>>x>>y;
        x--;y--;
        int numc=sum2[y+1]-sum2[x];
        int tc = sum[y%K][1+y/K]-sum[y%K][(x+K-1)/K];
        int lc = numc-tc;
        cout<< lc + (y-x+1)/K - tc <<endl;
    }
}
 
int main() {
    solve();
    return 0;
}
