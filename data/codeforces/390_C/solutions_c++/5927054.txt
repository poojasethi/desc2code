#include <iostream>
using namespace std;

int oper[100100];

int main()
{
	int n, k, w;
	cin >> n >> k >> w;
	char candy[100100];
	cin >> candy;

	for (int i = 0; i < n; i++)
	{
		if (candy[i] == '0')
			oper[i]++;

		for (int j = i - 1; j >= 0 && j > i - k; j--)
			if (candy[j] == '1')
				oper[i]++;

		if (i >= k)
			oper[i] += oper[i-k];
	}

	for (int i = 0; i < w; i++)
	{
		int li, ri;
		cin >> li >> ri;
		li--;
		ri--;
		int ans = oper[ri] - ((li == 0) ? 0 : oper[li-1]);
		cout << ans << "\n";
	}

	return 0;
}
