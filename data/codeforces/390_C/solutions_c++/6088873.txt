#include"stdio.h"
#include"string.h"
int zero[111111],one[111111],k0[12],k1[12],a[111111],s1[111111];
int main()
{
	int n,w,k,i,q,b,sum,sb,t;
	scanf("%d%d%d",&n,&k,&w);
	for(i=0;i<n;i++)
	{
		scanf("%1d",&a[i]);
		zero[i]=k0[i%k];
		one[i]=k1[i%k];
		if(a[i]==0)k0[i%k]++;
		else k1[i%k]++;
		if(i==0&&a[i]==1)s1[0]=1;
		if(i>0)
		{
			if(a[i]==1)s1[i]=s1[i-1]+1;
			else s1[i]=s1[i-1];
		}
	}
	for(i=1;i<=w;i++)
	{
		sum=0;
		scanf("%d%d",&q,&b);
		q--;
		b--;
		sum+=zero[b]-zero[q+k-1];
		if(a[b]==0)sum++;
		t=(b-q+1)/k;
		t=t-sum;
		sb=s1[b]-s1[q];
		if(a[q]==1)sb++;
		sb-=t;
		sum+=sb;
		printf("%d\n",sum);
	}
	return 0;
}
