#include<cstdio>
#include<cstring>
int n, k, w, i, j, l, r, a[100010], b[10][100010];
char s[100010];
int main(){
	while(~scanf("%d %d %d", &n, &k, &w)){
		scanf("%s", s);
		memset(b,0,sizeof(b));
		a[0]=0;
		for(i=1; i<=n; i++){
			if(s[i-1]=='0')	a[i]=0;
			else{
				a[i]=1;
				b[i%k][i]=1;
			}
		}
		for(i=2; i<=n; i++)	a[i]+=a[i-1];
		for(i=0; i<k; i++){
			for(j=2; j<=n; j++)	b[i][j]+=b[i][j-1];
		}
		while(w--){
			scanf("%d %d", &l, &r);
			i = a[r]-a[l-1];
			j = b[r%k][r]-b[r%k][l-1];
			l = (r-l+1)/k;
			printf("%d\n", i-j+l-j);
		}
	}
	return 0;
}
