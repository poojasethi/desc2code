#include <cstdio>
#include <algorithm>
#define gt getchar()
int getx(){
	char c;int x;
	for (c=gt;c<'0'||c>'9';c=gt);
	for (x=0;c>='0'&&c<='9';c=gt)
		x=(x<<3)+(x<<1)+c-'0';
	return x;
}
const int MAX_N=100500;
int n,k,w;
int sum[MAX_N],a[MAX_N];
int main(){
	n=getx(),k=getx(),w=getx();
	for (int i=1;i<=n;++i){
		char c;for (c=gt;c!='1'&&c!='0';c=getchar());
		sum[i]=sum[i-1]+c-'0';
		if (i<=k) a[i]=c-'0';
		else	a[i]=a[i-k]+c-'0';
		}
	for (int i=1;i<=w;++i){
		int l=getx(),r=getx();
		int s1=sum[r]-sum[l-1],s0=r-l+1-s1;
		int a1=a[r]-a[l-1],a0=(r-l+1)/k-a1;
		printf("%d\n",s1-a1+a0);
		}
}
