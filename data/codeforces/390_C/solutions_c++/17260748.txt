#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
char str[100020];
int  cnt[20][100020];;
int main(int argc, char const *argv[])
{
	int n,k,w;
	scanf("%d%d%d",&n,&k,&w);
	scanf("%s",str);
	memset(cnt,0,sizeof(cnt));
	for (int i=0;i<n;i++){
		for (int j=0;j<k;j++)
			cnt[j][i+1]=cnt[j][i];
		if (str[i]=='1')
			cnt[i%k][i+1]++;
	}
	for (int i=0;i<w;i++){
		int l,r;
		scanf("%d%d",&l,&r);
		int ans=0;
		for (int j=0;j<k;j++)
			if (j==(r-1)%k)
				ans=ans+(r-l+1)/k-(cnt[j][r]-cnt[j][l-1]);
			else ans=ans+cnt[j][r]-cnt[j][l-1];
		printf("%d\n",ans );
	}
	return 0;
}
