#include<stdio.h>
#include<string.h>
int dp[20][111111];
	char s[111111];
int main()
{
	
	int n,k,w,i,j;
	memset(dp,0,sizeof(dp));
	scanf("%d%d%d",&n,&k,&w);
	scanf("%s",s+1);
	for(i=0;i<k;i++)
   for(j=1;j<=n;j++)
   {
   if(j%k==i&&s[j]=='0')dp[i][j]=dp[i][j-1]+1;
   
  else  if(j%k!=i&&s[j]=='1')dp[i][j]=dp[i][j-1]+1;
   else dp[i][j]=dp[i][j-1];
   }
  
   
   for(i=0;i<w;i++)
   {
   	int l,r;
   	scanf("%d%d",&l,&r);
   	printf("%d\n",dp[r%k][r]-dp[r%k][l-1]);
   }

	
}
