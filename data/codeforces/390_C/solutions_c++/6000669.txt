#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>

using namespace std;
int n,k,w,l,r,s,x[111111],y[111111];
bool b[111111];
char a[111111];
int main(){
 	cin>>n>>k>>w;
 	cin>>a;
 	for(int i=0; i<n; i++){
 		b[i]=a[i]-'0';
 		if(i>0)
			y[i]+=y[i-1]+b[i];
		else
	    		y[i]=b[i];
	}

	for(int i=0; i<n; i++)
	if(i>=k)			
		x[i]=x[i-k]+b[i];			
	else
		x[i]=b[i];
	for(int i=0; i<w; i++){
		cin>>l>>r;
		s=y[r-1]-y[l-1]+b[l-1]-2*(x[r-1]-x[l+k-2]+b[l+k-2])+(r-l+1)/k;
		cout<<s<<endl;
	}
	
     	

 	return 0;
}
