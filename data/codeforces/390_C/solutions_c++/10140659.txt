#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;

const int mx=100005;
int dp[mx];

int main()
{
    int i,j,k,n,w;
    char s[mx];
    while(scanf("%d%d%d",&n,&k,&w)==3)
    {
        scanf("%s",s);
        int sum=0;
        dp[0]=0;
        for(i=1;i<k;i++) sum+=s[i-1]-'0',dp[i]=0;
        for(;i<=n;i++)
        {
            dp[i]=sum+dp[i-k];
            if(s[i-1]=='0') dp[i]++;
            sum=sum+s[i-1]-s[i-k];
        }
        while(w--&&scanf("%d%d",&i,&j)==2)
        {
            int ans=dp[j]-dp[i-1];
            printf("%d\n",ans);
        }
    }
}
