#include <iostream>
#include <cstdio>
#include <map>
#include <string>
using namespace std;
string S;
int N,K,W,L;
int aa[10][100010],b[100010];
void solve() {
    int x,y;

    cin>>N>>K>>W;
    cin>>S;
    L=S.size();
    for(int i=0;i<L;i++){
        aa[i%K][1+i/K] = aa[i%K][i/K] + (S[i]=='1');
        b[i+1] = b[i] + (S[i]=='1');
    }

    for(int i=0;i<W;i++){
        cin>>x>>y;
        x--;y--;
        int numc=b[y+1]-b[x];
        int tc = aa[y%K][1+y/K]-aa[y%K][(x+K-1)/K];
        int lc = numc-tc;
        printf("%d\n",lc + (y-x+1)/K - tc);
    }
}
int main()
{
    solve();
}
