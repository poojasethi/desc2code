#include<iostream>
#include<algorithm>
using namespace std;

int i,n,j,k,l,r,x,w,a[100005],dp[15][100005];
char c;

int main()
{
  cin>>n>>k>>w;
  for(i=1;i<=n;++i)
  cin>>c,a[i]=c-'0';

  for(i=1;i<=k;++i)
   for(j=i;j<=n;++j)
   {
     dp[i][j]=dp[i][j-1]; x=j-i+1;
     if(!(x%k)) dp[i][j]+=!a[j];
     else if(x%k) dp[i][j]+=a[j];
   }

  while(w--)
  {
    cin>>l>>r; x=(l-1)%k+1;
    cout<<dp[x][r]-dp[x][l-1]<<'\n';
  }

 return 0;
}
