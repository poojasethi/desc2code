#include<bits/stdc++.h>
using namespace std;
int n,k,w,arr[15][150000];
string str;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin>>n>>k>>w>>str;
    for(int i=1;i<=k;i++)
        for(int j=i;j<=n;j++)
        {
            arr[i%k][j]=arr[i%k][j-1];
            if(!(((j-i+1)%k != 0)^(str[j-1]-'0')))  arr[i%k][j]++;
        }
    while(w--)
    {
        int l,r;
        cin>>l>>r;
        cout<<arr[l%k][r]-arr[l%k][l-1]<<endl;
    }
    return 0;
}
