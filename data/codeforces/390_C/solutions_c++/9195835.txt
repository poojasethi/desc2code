#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int a[100005],b[100005];
int main()
{
    int n,k,w,i;
    string s;
    cin >> n >> k >> w;
    cin >> s;
    for (i = 0; i < n; i++) {
        a[i+1] = b[i+1] = s[i] == '1';
        a[i+1] += a[i];
        if (i >= k) b[i+1] += b[i-k+1];    
    }
    while (w--) {
          int l,r,c,d,e;
          cin >> l >> r;
          c = (r-l+1)/k;
          d = a[r]-a[l-1];
          e = b[r]- b[l-1];
          cout << c-e+d-e << endl;      
    }
    //system("pause");
    return 0;
}
