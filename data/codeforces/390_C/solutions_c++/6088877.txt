#include"stdio.h"
int sum[100010];
int sum2[20][100010];
char str[100010];
int main()
{
	int n,k,w;
	scanf("%d%d%d",&n,&k,&w);
	scanf("%s",str+1);
	for (int i=1;i<=n;i++)
	{
		str[i]-='0';
		sum[i] = sum[i-1]+str[i];
		sum2[i%k][i] = str[i];
	}
	for (int j=0;j<k;j++)
	{
		for (int i=1;i<=n;i++)
		{
			sum2[j][i] += sum2[j][i-1];
		}
	}
	for (int i=1;i<=w;i++)
	{
		if (i>1)
			printf("\n");
		int l,r;
		scanf("%d%d",&l,&r);
		int yu = (l-1)%k;
		int mk = sum2[yu][r] - sum2[yu][l-1];
		int nm = sum[r] - sum[l-1] - mk;

		int ans = ((r-l+1)/k-mk)+nm;
		printf("%d",ans);
	}

}
