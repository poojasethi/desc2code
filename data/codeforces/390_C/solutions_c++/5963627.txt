#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
#define FOR(i,l,r) for(int i=l;i<=r;i++)
#define FILL(a,b) memset(a,b,sizeof(a))
#define INF 2147483647
#define N 120001
int n,k,w,l,r,pre[12][N];
char a[N];
void get(int s){
	FOR(i,s,n)
	pre[s%k][i]=pre[s%k][i-1]+!(((i-s+1)%k != 0)^a[i]);
}
int main(){
	scanf("%d%d%d\n",&n,&k,&w);
	FOR(i,1,n)scanf("%c",&a[i]),a[i]-='0';
	FOR(i,1,k)get(i);
	FOR(i,1,w){
		scanf("%d%d",&l,&r);
		printf("%d\n",pre[l%k][r]-pre[l%k][l-1]);
	}
	return 0;
}
