#include <bits/stdc++.h>
using namespace std;
int sum[100020];
int main(){
    int n,k,w,l,r;
    string str;
    cin>>n>>k>>w>>str;
    str=" "+str;
    for(int i=n;i>=1;i--)sum[i]=sum[i+k]+(str[i]=='1');



    for(int i=0;i<w;i++){
        cin>>l>>r;

        int res=0;
        int C=(r-l+1)/k;
        for(int j=1;j<k;j++){
            int a=j+l-1,b=a+C*k;
            res+=(sum[a]-sum[b]);
        }

        int a=k+l-1,b=a+C*k;

        res+=C-(sum[a]-sum[b]);

        cout<<res<<"\n";
    }
}
