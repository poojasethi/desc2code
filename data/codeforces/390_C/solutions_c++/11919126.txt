#include <cstdio>
#include <algorithm>
using namespace std;
int enter,c,a[1000010],b[20][1000010],n,m,k,w,l,r;
int main(){
	scanf("%d%d%d",&n,&k,&w);
	scanf("%c",&enter);
	for (int i=1;i<=n;i++){
		scanf("%c",&c);
		if (c=='1') a[i] = 1;else a[i] = 0;
	}
	for (int i=0;i<k;i++)
		for (int j=i+1;j<=n;j++){
			b[i][j] = b[i][j-1];
			if ((((j-i)%k==0) && a[j]!=1)) b[i][j]++;
			else if (((j-i)%k!=0) && a[j]!=0) b[i][j]++;
		}
	for (int i=1;i<=w;i++){
		scanf("%d%d",&l,&r);
		int S = (l-1)%k;
		printf("%d\n",b[S][r]-b[S][l-1]);
	}
}
