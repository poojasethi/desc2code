#include <bits/stdc++.h>
using namespace std;
int n,k,w;
int d[10+100000],f[10+100000];
char s[10+100000];
int main(){
    //freopen("A.inp","r",stdin);
    scanf("%d %d %d\n",&n,&k,&w);
    scanf("%s\n",s+1);
    for(int i = 1;i <= n;++i) d[i] = d[i-1] + (s[i] == '1');
    for(int i = 1;i <= n;++i){
        if(i <= k) f[i] = (s[i] == '1');
        else f[i] = f[i-k] + (s[i] == '1');
    }
    for(int i = 0;i < w;++i){
        int l,r;
        scanf("%d %d\n",&l,&r);
        int s1 = d[r] - d[l-1];
        int s2 = f[r] - f[l-1];
        printf("%d\n",(r-(l-1))/k-s2 + s1 - s2);
    }
return 0;
}
