# include <bits/stdc++.h>

using namespace std;

const int maxn = 1e5 + 100;

int s[maxn], sum[maxn];

bool mark[maxn];

int main(){
	ios_base::sync_with_stdio(false);
	int n, k, q;
	cin >> n >> k >> q;
	for (int i = 1; i <= n; i++){
		char x; cin >> x;
		mark[i] = (x == '1' ? true : false);
	}
	
	for (int i = 1; i <= n; i++)
		s[i] = (i >= k ? s[i - k] : 0) + (mark[i] == true);

	for (int i = 1; i <= n; i++)
		sum[i] = sum[i - 1] + (mark[i] == true);
	
	while(q--){
		int l, r;
		cin >> l >> r;
		int ret = ((r - l + 1) / k) - (s[r] - s[l - 1]);
		ret += (sum[r] - sum[l - 1]) - (s[r] - s[l - 1]);
		cout << ret << '\n';
	}

	return 0;
}
