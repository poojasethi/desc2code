#include <iostream>
#include <cstdio>
#include <cstring>

int n,k,w,i,j,ans,l,r,x;
char s[100100];
int sum[100100],sumk[100100];

int main()
{
    scanf("%d%d%d",&n,&k,&w);
    scanf("%s",s);
    for (i=1;i<=n;i++)
    {
        sum[i]=sum[i-1]+s[i-1]-'0';
        sumk[i]=s[i-1]-'0';
        if (i>k) sumk[i]+=sumk[i-k];
    }
    for (i=1;i<=w;i++)
    {
        scanf("%d%d",&l,&r);
        int t1=sum[r]-sum[l-1];
        int t2=sumk[r]-sumk[l-1];
        printf("%d\n",(r-l+1)/k-t2+t1-t2);
    }
    return 0;
}
