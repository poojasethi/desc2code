#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
int a[100024],b[100024],ones,one,want,n,k,w,l,r;
int f[11][100024];
int main()
{
	scanf("%d%d%d\n",&n,&k,&w);
	for(int i=1;i<=n;i++)
		a[i]=getchar()-'0';
	b[0]=0;
	for(int i=1;i<=n;i++)b[i]=b[i-1]+a[i];
	for(int i=0;i<k;i++)
		for(int j=i+k;j<=n;j+=k)f[i][j]=f[i][j-k]+a[j];
	for(register int i=1;i<=w;i++)
	{
		scanf("%d%d",&l,&r);
		ones=b[r]-b[l-1];
		one=f[r%k][r]-f[r%k][l-1];
		want=(r-l+1)/k;
		printf("%d\n",abs(ones-one)+abs(want-one));
	}
	return 0;
}
