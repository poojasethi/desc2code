#include <iostream>

using namespace std;

char s[100001];
int a[10][100000];

int main()
{
	cin.sync_with_stdio(false);
	int n, k, w;
	cin >> n >> k >> w;
	cin >> s;
	for (int x = 0; x < k; ++x)
	{
		int no = x+k-1;
		int sum = 0;
		for (int i = 0; i < n; ++i)
		{
			if (i == no)
			{
				if (s[i] == '0')
					sum++;
				no += k;
			}
			else
			{
				if (s[i] == '1')
					sum++;
			}
			a[x][i] = sum;
		}
	}
	for (int i = 0; i < w; ++i)
	{
		int l, r;
		cin >> l >> r; l--; r--;
		int x = l%k;
		cout << a[x][r]-max(0, a[x][l-1]) << "\n";
	}
	return 0;
}
