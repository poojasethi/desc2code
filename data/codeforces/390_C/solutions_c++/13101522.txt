#include <iostream>

using namespace std;

int n,k,w;
int arr2[1000000];
int arr[1000000];

int main()
{
  cin>>n>>k>>w;
  string s;
  cin>>s;
  for (int i=1;i<=s.size();i++) {
    arr2[i]=arr2[i-1]+(s[i-1]-'0');
    arr[i]=s[i-1]-'0';
    if (i>=k)
      arr[i]+=arr[i-k];
  }
  for (int i=0;i<w;i++) {
    int l,r;
    cin>>l>>r;
    int ans=arr2[r]-arr2[l-1];
    int ans2=arr[r]-arr[l-1];
    cout<<(r-l+1)/k-ans2+ans-ans2<<endl;
  }
}
