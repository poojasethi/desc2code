#include<iostream>
#include<algorithm>
#include<cmath>
#include<queue>
#include<cstring>
#include<vector>
#include<string>
using namespace std;
const int MAXN=300012;
const int INF=~0U>>2;
int n,m,k;
int rea[MAXN];
int sum[11][MAXN];
string s;
int main()
{
	cin>>n>>k>>m;
	cin>>s;
	for(int i=1;i<=n;i++) rea[i]=s[i-1]-'0';
	for(int i=0;i<k;i++)
	{
		for(int j=1;j<=n;j++)
		{
			sum[i][j]=sum[i][j-1];
			if(j%k==i&&rea[j]==0) sum[i][j]++;
			if(j%k!=i&&rea[j]==1) sum[i][j]++;
		}
	}
	for(int i=1;i<=m;i++)
	{
		int x,y;
		cin>>x>>y;
		cout<<sum[y%k][y]-sum[y%k][x-1]<<endl;
	}
	return 0;
}
