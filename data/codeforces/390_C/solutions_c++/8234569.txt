#include <cstdio>
#include <algorithm>

int a[100005],b[100005],c,d,e,f,g,h,i,j,k,l,m,n,s[10][100005],r;
char S[100005];
int main(){
	scanf("%d%d%d",&n,&k,&m);
	scanf("%s",S+1);
	for(i=1;i<=n;i++)a[i]=S[i]=='1';
	for(i=1;i<=n-k+1;i++){
		for(j=i;j<i+k-1;j++)if(a[j])b[i]++;
		if(!a[i+k-1])b[i]++;
		s[i%k][i/k+1]=s[i%k][i/k]+b[i];
	}

		h=s[l%k][l/k+(r-l+1)/k]-s[l%k][l/k];

	}

}
