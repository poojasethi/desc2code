#include"stdio.h"
int num[11][100005],n,w,k;
char a[100005];
int judge(int p,int l){
	int d=0;
	if((p+1-l+k)%k==0){
		if(a[p-1]=='0')d=1;
	}
	else if(a[p-1]=='1')d=1;
	return d;
}

void f(){
	int i,j;
	for(i=0;i<10;i++){
		num[i][0]=0;
		for(j=i+1;j<=n;j++)num[i][j]=num[i][j-1]+judge(j,i+1);
	}
}

main(){
	int l,r;
	while(~scanf("%d%d%d%s",&n,&k,&w,a)){
		f();
		while(w--){
			scanf("%d%d",&l,&r);
			int p=(l-1)%k;
			printf("%d\n",num[p][r]-num[p][l-1]);
		}
	}
}
