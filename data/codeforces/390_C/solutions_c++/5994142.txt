#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
#define maxn 120001
int n,k,w,l,r,dp[12][maxn];
char ch[maxn];
void get(int s)
{
    for(int i=s;i<=n;i++)
        dp[s%k][i]=dp[s%k][i-1]+!(((i-s+1)%k != 0)^(ch[i]-'0'));
}
int main()
{
    scanf("%d%d%d",&n,&k,&w);
    scanf("%s",ch+1);
    for(int i=1;i<=k;i++) get(i);
    while(w--)
    {
        scanf("%d%d",&l,&r);
        printf("%d\n",dp[l%k][r]-dp[l%k][l-1]);
    }
    return 0;
}
