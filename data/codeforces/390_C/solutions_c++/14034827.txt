#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,last,x,y,K;
int f[N],g[N];
char p[N];
int main() {
	scanf("%d%d%d%s",&n,&K,&m,p+1);
	For(i,1,n) g[i]=g[i-1]+(p[i]=='1');
	For(i,0,K-1) {
		for (last=i,j=i+K;j<=n;last=j,j+=K)
			f[j]=f[last]+(p[j]=='0')+g[j-1]-g[last];
	}
	For(i,1,m) {
		scanf("%d%d",&x,&y);
		printf("%d\n",f[y]-f[x-1]);
	}
	return 0;
}
