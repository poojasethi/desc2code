#include<stdio.h>

#define min(a,b) ((a)<(b)? (a) : (b))

int tab[12][100100];
char s[100100];

int main() {
  int n,k,w;
  scanf("%d%d%d", &n,&k,&w);
  /*for (int i=0; i<n; i++) {
    s[i] = getchar();
  }
  scanf(" ");*/
  scanf("%s", s);
  for (int i=0; i<k; i++) {
    tab[i][0] = 0;
    for (int j=0; j<n; j++) {
      tab[i][j+1] = tab[i][j];
      if ((j%k) == i) {
        tab[i][j+1] += (int)(s[j]=='0');
      } else {
        tab[i][j+1] += (int)(s[j]=='1');
      }
    }
  }
  int l,r;
  for (int i=0; i<w; i++) {
    scanf("%d %d", &l,&r);
    l--; r--;
    printf("%d\n", tab[r%k][r+1] - tab[r%k][l]);
  }
}
