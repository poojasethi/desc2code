#include<bits/stdc++.h>

using namespace std;

int n,sorgular,k,t;

char str[100005];
int dp[100005][12];

int main(){
	int md,i,j,m;
	cin>>n>>k>>sorgular;
	for(int i=1;i<=n;i++)
		cin >> str[i];
	for(i=1;i<=n;i++)
	{
		for(j=0;j<k;j++)
		{
			if((i%k)==j)
			{
				if(str[i]=='1')
					dp[i][j]=dp[i-1][j];
				else 
					dp[i][j]=dp[i-1][j]+1;
			}
			else
			{
				if(str[i]=='0')
					dp[i][j]=dp[i-1][j];
				else 
					dp[i][j]=dp[i-1][j]+1;
			}
		}
	} 
	for(i=1;i<=sorgular;i++)
	{
		int bas,son;
		scanf("%d %d",&bas,&son);
		md=(bas+k-1)%k;
		cout<<dp[son][md]-dp[bas-1][md]<<endl;
	}
	return 0;
}
