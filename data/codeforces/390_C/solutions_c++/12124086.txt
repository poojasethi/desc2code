#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
int a[100005],ks[100005];
int main(){
    int n,k,w;
    scanf("%d%d%d",&n,&k,&w);
    string s;
    cin>>s;
    for(int i=1;i<=n;i++){
        a[i]=a[i-1]+(s[i-1]-'0');
        if(i>k) ks[i]=ks[i-k]+(s[i-1]-'0');
        else ks[i]=s[i-1]-'0';
    }
    /*
    for(int i=1;i<=n;i++) cout<<a[i];
    cout<<endl;
    for(int i=1;i<=n;i++) cout<<ks[i];
    cout<<endl;
    */
    int i,j,ans;
    while(w--){
        scanf("%d%d",&i,&j);
        ans=a[j]-a[i-1]-(ks[j]-ks[i-1])+(j-i+1)/k-(ks[j]-ks[i-1]);
        printf("%d\n",ans);
    }
    return 0;
}
