#include<bits/stdc++.h>
#define  N 100010
using namespace std;
int dp[N],sum[N];
int main()
{
    int n,k,m;
    char s[N];
    scanf("%d%d%d",&n,&k,&m);
    scanf("%s",s+1);
    for(int i=1;i<=n;i++)
        sum[i]=sum[i-1]+(s[i]=='1');
    for(int i=0;i<k;i++)
    {
        for(int last=i,j=i+k;j<=n;last=j,j+=k)
        {
            dp[j]=dp[last]+(s[j]=='0')+sum[j-1]-sum[last];
        }
    }
    for(int i=0;i<m;i++)
    {
        int a,b;
        scanf("%d%d",&a,&b);
        printf("%d\n",dp[b]-dp[a-1]);
    }
    return 0;
}
