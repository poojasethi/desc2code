#include <iostream>
#include <vector>
using namespace std;
const int nmax = 1e5+5;
int sum[nmax],psum[nmax][10];
int main(){
	int n,k,w;
	cin >> n >> k >> w;
	vector<int> arr(n+1);
	for(int i = 1; i <= n; ++i){
		char t;
		cin >> t;
		arr[i]=t=='1';
		for(int j = 0; j < k; ++j)
			psum[i][j]+=psum[i-1][j];
		psum[i][i%k]+=arr[i];
		sum[i]+=sum[i-1]+arr[i];
	}
	for(int i = 0; i < w; ++i){
		int a,b;
		cin >> a >> b;
		int ans = 0;
		int splace = psum[b][b%k]-psum[a-1][b%k];
		int sall = sum[b]-sum[a-1];
		int sshould = (b-a+1)/k;
		cout << sshould-splace + sall-splace << '\n';
	}
}
