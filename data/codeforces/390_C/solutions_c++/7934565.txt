#include<stdio.h>
int main()
{
    int n,k,w,a[100100],b[100100]={0},c[100010]={0},sum=0,i,l,r;
    char s[100010];
    scanf("%d %d %d",&n,&k,&w);
    char ch=getchar();
    scanf("%s",s);
    for(i=100;i<=n+99;i++)
    {
        a[i]=s[i-100]-'0';
    }
    for(i=100;i<=n+99;i++)
    {
        b[i]=b[i]+b[i-k]+a[i];
    }
    for(i=100;i<=n+99;i++)
    {
        c[i]=c[i-1]+a[i];
    }
    while(w--)
    {
        sum=0;
        scanf("%d %d",&l,&r);
        sum=sum + ((r-l+1)/k - (b[r+99]-b[l+99-1]));
        sum=sum + ((c[r+99]-c[l+99-1]) - (b[r+99]-b[l+99-1]));
        printf("%d\n",sum);
    }
    return(0);
}
