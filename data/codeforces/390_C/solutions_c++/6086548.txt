#include<cstdio>
#include<iostream>
using namespace std;
const int maxn=100001;
int cnt[maxn][10]={0};
bool cd[maxn]={0};
int n,k,m;
int main(){
int num=0;
int l,r;
scanf("%d%d%d",&n,&k,&m);
while(1){
char c=getchar();
if(c!='\n'){num++;for(int i=0;i<k;i++)cnt[num][i]=cnt[num-1][i];cd[num]=c-48;if(c=='1')cnt[num][num%k]++;}
if(num==n)break;
}
while(m--){
scanf("%d%d",&l,&r);
int sum=cnt[r][r%k]-cnt[l-1][r%k];
int ans=0;
int pos=r%k;
ans+=(r-l+1)/k-sum;
//printf("%d\n",sum);
for(int i=0;i<k;i++)if(i!=pos)ans+=cnt[r][i]-cnt[l-1][i];
printf("%d\n",ans);
}
return 0;
}
