n,k=map(int,raw_input().split())
s=raw_input()
ans=1
for d in 'ab':
    cnt=0
    for i in range(n):
        if s[i]==d:
            cnt+=1
            if cnt>k:
                ans=max(ans,i)
                j=0
                while i<n:
                    while j<n and s[j]!=d:j+=1
                    j+=1
                    i+=1
                    while i<n and s[i]!=d:i+=1
                    ans=max(ans,i-j)
                break
    else:
        print n
        exit()
print ans
