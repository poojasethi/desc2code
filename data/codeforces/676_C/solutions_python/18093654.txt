n,k=map(int,raw_input().split())
s=raw_input()
ans=0
j=0
cnt_a=0
cnt_b=0
for i in xrange(n):
	while j<n and min(cnt_a,cnt_b) <= k :
		
		if s[j]=='a' :
			cnt_a+=1
		else :
			cnt_b+=1
		
		#print "at j",j,":",s[j]
		j+=1
	#print "i",i,"j",j	
	#print "a",cnt_a,"b",cnt_b
	if min(cnt_a,cnt_b) > k :
		ans=max(ans,cnt_a+cnt_b-1)
	else :
		ans=max(ans,cnt_a+cnt_b)
	if j>=n :
		break
	if s[i]=='a' :
		cnt_a-=1
	else :
		cnt_b-=1
	
print ans
