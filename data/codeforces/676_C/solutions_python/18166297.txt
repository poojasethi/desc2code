import os

def check(x):
	for i in range(x, n+1):
		su = cou[i] - cou[i-x]
		if x-su <= k or su <= k:
			return True
	return False

n, k = map(int, raw_input().split())
s = raw_input()
cou = [0] * (n*2)
for i in range(n):
	cou[i+1] = cou[i];
	if s[i] == 'a':
		cou[i+1] += 1

ans = 0

l = 1
r = n
while l <= r:
	m = (l+r)/2
	temp = check(m)
	# print m
	if temp :
		ans = m 
		l = m+1
	else:
		r = m-1
print ans
