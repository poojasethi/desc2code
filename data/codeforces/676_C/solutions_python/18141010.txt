def solve(s, n, k):
	
	ans = k
	for ch in ['a', 'b']:

		bad = 0
		l = 0
		for r in range(n):
			if s[r] != ch:
				bad += 1

			if bad > k:
				while l <= r and s[l] == ch:
					l += 1
				l += 1
				bad -= 1

			ans = max(ans, r - l + 1)

	return ans
