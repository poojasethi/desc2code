n,k = map(int,raw_input().strip().split(" "))
s = raw_input()
ans  = 0

tail = -1
cnt = 0
for i in range(n):
	if s[i]=='b': cnt += 1
	while cnt>k:
		tail += 1
		if s[tail]=='b':
			cnt -= 1
	ans = max(ans,i-tail)

tail = -1
cnt = 0
for i in range(n):
	if s[i]=='a': cnt += 1
	while cnt>k:
		tail += 1
		if s[tail]=='a':
			cnt -= 1
	ans = max(ans,i-tail)

print ans
