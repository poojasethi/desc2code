n, k = map(int, raw_input().split())
s = raw_input()

a = 0
b = 0
maxI = 0
st = 0

for i in range(n):

    # count a and b until one gets the number of links k
    if s[i] == 'a':
        a += 1
    else:
        b += 1

    # the minimum should be the number of links, so if there is more, than we can "scoot over"
    if min(a, b) > k:

        # subtract until we get the right amount of links
        if s[st] == 'a':
            a -=1
        else:
            b -= 1
        st += 1

    # extending our longest
    else:
        maxI += 1

print maxI
