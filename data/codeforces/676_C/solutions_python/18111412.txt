def solve(n, k, letters):
	currentLength = 0
	bestLength = 0
	if k == 0:
		for i in range(len(letters)):
			if letters[i] != "a":
				bestLength = max(bestLength, currentLength)
				currentLength = 0
			else:
				currentLength += 1
		bestLength = max(bestLength, currentLength)
		currentLength = 0

		for i in range(len(letters)):
			if letters[i] != "b":
				bestLength = max(bestLength, currentLength)
				currentLength = 0
			else:
				currentLength += 1

		bestLength = max(bestLength, currentLength)

		return bestLength


	kCount = int(k)
	currentLength = 0
	maxLength = 0
	changed = []
	unReplaced = 0
	for i in range(len(letters)):
		if letters[i] != "a":
			if kCount == 0:
				maxLength = max(currentLength, maxLength)
				currentLength = (i - changed[unReplaced]) - 1
				unReplaced += 1
				kCount = 1
			kCount -= 1
			changed.append(i)
		currentLength += 1

	maxLength = max(currentLength, maxLength)

	ans = int(maxLength)


	kCount = int(k)
	currentLength = 0
	maxLength = 0
	changed = []
	unReplaced = 0
	for i in range(len(letters)):
		if letters[i] != "b":
			if kCount == 0:
				maxLength = max(currentLength, maxLength)
				currentLength = (i - changed[unReplaced]) - 1
				unReplaced += 1
				kCount = 1
			kCount -= 1
			changed.append(i)
		currentLength += 1

	maxLength = max(currentLength, maxLength)

	return max(maxLength, ans)



n, k = map(int, raw_input().split())
letters = raw_input()

print solve(n, k, letters)
