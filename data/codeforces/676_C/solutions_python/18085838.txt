from collections import deque
n, k = map(int, raw_input().split())
stri = raw_input()

def count(arr):
    sumi = 0
    buff = deque()
    maxi = 0
    for elem in arr:
        sumi += elem
        while sumi > k:
            if len(buff) == 0:
                break
            sumi -= buff.popleft()
        buff.append(elem)
        maxi = max(maxi, len(buff))
    return maxi

a1 = map(lambda x: 1 if x == 'a' else 0, stri)
b1 = map(lambda x: 1 if x == 'b' else 0, stri)
print max(count(a1), count(b1))
