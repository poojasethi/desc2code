#include <bits/stdc++.h>
using namespace std;
string s;
int n, k;
int solve(char c){
    int i, j, ans, num;
    ans =  j = num = 0;
    for(i = 0;i < n;i++){
        while(j < n&&num + (s[j] != c) <= k){
            num += s[j] != c;
            j++;
        }
        num -= s[i] != c;
        ans = max(ans, j - i);
    }
    return ans;
}
int main()
{
    scanf("%d %d", &n, &k);
    cin>>s;
    printf("%d\n", max(solve('a'), solve('b')));
}
