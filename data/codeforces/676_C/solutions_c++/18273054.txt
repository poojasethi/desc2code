#include <iostream>
#include <map>

using namespace std;

int main()
{
	map<char, int> m;
	int n, k, ans = 0;
	string s;
	cin >> n >> k >> s;
	for(int i = 0, j = 0; i < n; i++)
	{
		m[s[i]]++;
		if(min(m['a'], m['b']) > k)
		{
			m[s[j]]--;
			j++;
		}
		ans = max(i-j+1, ans);
	}
	cout << ans << endl;
}
