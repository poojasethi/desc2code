#include<bits/stdc++.h>

using namespace std;

int main(void)
{
    int n,k,p=0,q=0,r,d=0,c=0;
    cin>>n>>k;
    char a[n];
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    for(int r=0;r<n;r++){
        if(a[r]=='a') p++;
            else q++;
        if(min(p,q)>k){
            if(a[c]=='a') p=p-1;
            else q=q-1;
            c++;
        }
        else d++;
    }
    cout<<d;
    return 0;
}
