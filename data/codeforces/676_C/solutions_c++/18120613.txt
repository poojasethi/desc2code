#include <bits/stdc++.h>
using namespace std;
int main(){
	int N,K,i,p,q,a,b,ans=0;
	string A;
	cin>>N>>K>>A;
	for(i=p=a=b=q=0;i<N;i++){
		p+=A[i]=='a';
		q+=A[i]=='b';
		while(a<=i&&p>K)p-=(A[a++]=='a');
		while(b<=i&&q>K)q-=(A[b++]=='b');
		if(p<=K)ans=max(ans,i-a+1);
		if(q<=K)ans=max(ans,i-b+1);
	}
	cout<<ans<<"\n";
}
