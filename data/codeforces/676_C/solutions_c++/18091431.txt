#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
ll i,n,k,ans=1,s=0,e=0,a=0,b=0;
char shu[100003];
scanf("%lld %lld",&n,&k);
scanf("%s",shu);
for(i=0;i<n;i++)
	{
		if(shu[i]=='a')a++;
		else b++;
		while(k<min(a,b))
			{
			if(shu[e]=='a')a--;
			else b--;
			e++;
			}
		if(i-e+1>ans)
			ans=i-e+1;
	}
printf("%lld\n",ans);
return 0;
}
