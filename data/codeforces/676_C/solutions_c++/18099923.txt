#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e5 + 5;
char s[maxn];
int n, k, l, r, sa, sb, ans = 0;

int main() {
	scanf("%d%d%s", &n, &k, s);
	sa = (s[l=0] == 'a');
	sb = (s[r=0] == 'b');
	while(r < n) {
		if(sa <= k || sb <= k) {
			ans = max(ans, r - l + 1);
		}
		else {
			sa -= (s[l] == 'a');
			sb -= (s[l++] == 'b');
		}
		sa += (s[++r] == 'a');
		sb += (s[r] == 'b');		
	}
	printf("%d\n", ans);
	return 0;
}
