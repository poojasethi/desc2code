#include <iostream>
using namespace std;
const int MAXN = 1e5 + 9;
int n, k, s[MAXN];
string S;


int main() {
  ios_base::sync_with_stdio(false); cin.tie(0);
  cin >> n >> k;
  cin >> S;
  s[0] = 0;
  for(int i=1; i<=n; ++i)
    s[i] = s[i-1] + (S[i-1] == 'a');
  int ans = 1;
  int l = 1, r = 1;
  while(r<=n) {
    while( min(s[r]-s[l-1], (r-l+1)-(s[r]-s[l-1])) > k) {
      ++l;
    }
    ans = max(ans, r-l+1);
    ++r;
  }
  cout << ans << endl;
}
