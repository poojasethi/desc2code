#include<stdio.h>
#include<algorithm>
#define maxn 111111
using namespace std;
char s[maxn];
int ans;
int n,k;
void solve(char ch){
	int cnt = 0;
	for(int i = 0,j = 0;i<n;i++){
		if(s[i] == ch)
			cnt++;
		while(cnt > k){
			if(s[j] == ch)
				cnt--;
			j++;
		}
		ans = max(ans,i-j+1);
	} 
}
int main(){
	scanf("%d%d",&n,&k);
	scanf(" %s",s);
	solve('a');
	solve('b');
	printf("%d\n",ans);
	return 0;
}
