#include<iostream>
using namespace std;
string str;
int n,k;
int solve(string str,char a)
{
	int j=0,num=0;
	int res=-1;
	for(int i=0;i<n;i++)
	{
		while(j<n&&num<=k){
			if(str[j]!=a){
				j++;
			}
			else{
				if(num==k)break;
				else j++,num++;
			}
		}
		res=max(res,j-i);
		if(str[i]==a)
			num--;
	}
	return res;
}
int main()
{
	cin>>n>>k;
	cin>>str;
	cout<<max(solve(str,'a'),solve(str,'b'))<<endl;
}
