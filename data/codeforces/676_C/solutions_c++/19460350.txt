#include<iostream>
#include<cstdio>
using namespace std;
const int MAXN=100000+10;
char str[MAXN];
int n,k;
int solve(char c){
	int ans=0,l=0,r=0,num=0;
	while(l<n&&r<n){
		while((str[r]==c||num<k)&&r<n){
			if(str[r]!=c) num++;
			r++;
		}
		ans=max(ans,r-l);
		while(str[l]==c&&l<=r) l++;
		l++;
		num--;
	}
	return ans;
}
int main(){
	while(scanf("%d%d%s",&n,&k,&str)!=EOF)
		printf("%d\n",max(solve('a'),solve('b')));
	return 0;
}
