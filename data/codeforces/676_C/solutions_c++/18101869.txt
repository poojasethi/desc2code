#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,k,ans=0,a=0,b=0,p0=0,p1=0;
	string s;
	cin>>n>>k;
	cin>>s;
	a=(s[0]=='a');b=(s[0]=='b');
	while(true)
    {
		if (min(a,b)<= k)
		{
			ans=max(ans, p1 - p0 + 1);
			p1++;
			if(p1 == n){break;}
			if (s[p1] == 'a'){a++;}
			else{b++;}
			continue;
		}
		if (s[p0] == 'a'){a--;}
		else{b--;}
		p0++;
	}
	cout<<ans<<endl;
	return 0;
}
