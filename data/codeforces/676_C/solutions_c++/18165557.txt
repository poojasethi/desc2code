#include<iostream>
#include<algorithm>
#include<cstdlib>
#include<string>

using namespace std;

int main()
{
	std::ios_base::sync_with_stdio(false);
	int n, k, a=0, b=0, start=0, ans=1;
	string inp;
	cin>>n>>k>>inp;
	for(int i=0; i<n; i++)
	{
		if(inp[i] == 'a')
			a++;
		else
			b++;
		if(k < min(a, b))
		{
			if(inp[start] == 'a')
				a--;
			else
				b--;
			start++;
		}
		ans = max(i + 1 - start, ans);
	}
	cout << ans << endl;
	return 0;
}
