#include <bits/stdc++.h>

using namespace std;

int main(){
	int n,k;
	string str;
	scanf("%d%d",&n,&k);
	cin>>str;
	int ans=0;
	int t=0;
	int ai=0,bi=0,ca=0,cb=0;
	for(int i=0;i<n;i++){
		if(str[i]=='a')ca++;
		if(str[i]=='b')cb++;
		while(ai<i&&ca>k){
			if(str[ai++]=='a')ca--;
		}
		while(bi<i&&cb>k){
			if(str[bi++]=='b')cb--;
		}
		ans=max(ans,i-ai+1);
		ans=max(ans,i-bi+1);
	}
	printf("%d\n",ans);
}
