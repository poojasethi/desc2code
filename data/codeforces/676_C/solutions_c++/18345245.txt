#include<bits/stdc++.h>
using namespace std;
int n,k,ans;
string s;
int na,nb;
int main()
{
	cin>>n>>k>>s;
	int l=0,r=-1;
	while(l<n)
	{
		if(r>=n) break;
		else
		{
			if(na<=k||nb<=k)
			{
				ans=max(ans,na+nb);
				r++;
				if(s[r]=='a') na++;
				else nb++;
			}
			else
			{
				if(s[l]=='a') na--;
				else nb--;
				l++;
			}
		}
	}
	cout<<ans<<endl;
	return 0;
}
