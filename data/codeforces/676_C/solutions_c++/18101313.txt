#include<cstdio>
const int N=100005;
char s[N];
int n,k,sum[N],l,r,mid;
bool check(int x){
	for (int i=x;i<=n;i++)
	if (sum[i]-sum[i-x]<=k || (x-sum[i]+sum[i-x]<=k)) return 1;
	return 0;
}
int main(){
	scanf("%d%d",&n,&k);
	scanf("%s",s+1);
	for (int i=1;i<=n;i++) 
	sum[i]=sum[i-1]+(s[i]=='a');
	l=1,r=n;
	while (l<r){
		mid=(l+r+1)>>1;
		if (check(mid)) l=mid;
		else r=mid-1; 
	}
	printf("%d",l);
	return 0;
}
