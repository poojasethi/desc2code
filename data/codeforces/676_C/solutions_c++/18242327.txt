#include <stdio.h>
#include <stdlib.h>
#include <algorithm>

using namespace std;

int n, k, sol, C[2];
char str[100010];

int main() {
  scanf("%d%d%s", &n, &k, str);
  
  int lo = 0;
  for (int i = 0 ; i < n ; ++i) {
    ++C[str[i] - 'a'];
    while (min(C[0], C[1]) > k) {
      --C[str[lo] - 'a'];
      ++lo;
    }
    sol = max(sol, i - lo + 1);
  }
  printf("%d\n", sol);
}
