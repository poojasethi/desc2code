#include <iostream>
#include <cstdio>
#include <cstring>
#include <map>

using namespace std;

int main()
{
  int n,k;
  cin >> n >> k;
  string s;
  cin >> s;
  int r = 0,l = 0,ans = 0;
  map<char,int> map;
  while(r < n){
    ++map[s[r]];
    while(min(map['a'],map['b']) > k){
      --map[s[l]];
      ++l;
    }
    ans = max(ans, r-l+1);
    ++r;
  }
  cout << ans << endl;
}
