#include<bits/stdc++.h>
using namespace std;
int main()
{
    map<char,int>m;
    int n,k,ans=0;
    string s;
    cin>>n>>k>>s;
    for(int i=0, j=0;i<n;i++)
    {
        m[s[i]]++;
        if(min(m['a'],m['b'])>k)
        {
            m[s[j]]--;
            j++;
        }
    }
    cout << m['a'] + m['b'];
    return 0;
}
