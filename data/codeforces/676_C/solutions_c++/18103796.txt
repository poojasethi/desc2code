#include <stdio.h>

char s[100002];
int n, k;

int maxk(char c) {
	int b = 0, x = 0, l = 0;
	for (int e = 0; e < n; ++e) {
		if (s[e] == c)
			x++;
		if (x > k) {
			while (s[b] != c)
				b++;
			b++;
			x--;
		}
		if (e - b + 1 > l)
			l = e - b + 1;
	}
	return l;
}

int main() {
	scanf("%d%d", &n, &k);
	scanf("%s", s);
	int ka = maxk('a');
	int kb = maxk('b');
	if (ka > kb)
		printf("%d\n", ka);
	else
		printf("%d\n", kb);
	return 0;
}
