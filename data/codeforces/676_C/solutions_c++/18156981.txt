
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main(){
	string str;
	int n, k;
	cin >> n >> k;
	cin >> str;
	int p = 0;
	int q = 0;
	int a = 0, b = 0;
	int ans = 0;
	for (int i = 0; i < n; i++){
		if (str[i] == 'a')p++;
		else q++;
		if (min(q, p) <= k)
			ans = max(ans, p + q);
		else{
			if (str[i - p - q + 1] == 'a')
				p--;
			else q--;
		}
	}
	cout << ans << endl;
	return 0;
}
