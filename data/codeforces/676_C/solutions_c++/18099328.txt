#include <iostream>
using namespace std;

int main(){
	ios_base::sync_with_stdio(0);	cin.tie();
	int l,k;	cin>>l>>k;
	int res = 0;
	string s;	cin>>s;
	for(char c = 'a'; c<='b' ;c++)	for(int i=0,j=-1,d = 0;i<l;i++)	{
		if(i>0&&s[i-1]!=c)	d--;
		while(j<l&&d<=k)	{
			j++;
			if(s[j]!=c)	d++;
		}
		j--; d--;
		res = max(res,j-i+1);
	}
	cout<<res;
}
