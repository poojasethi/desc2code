#include<bits/stdc++.h>

using namespace std;

int n,k;
string str;

int solve(char c)
{
    int num=0,ans=0;

   for(int i=0,j=0;i<n;i++)
   {
    while(j<n && (num+(str[j]!=c)<=k))
    {
        num+=(str[j]!=c);
        j++;
    }

    ans=max(ans,j-i);
    num-=(str[i]!=c);
   }

   return ans;
}

int main()
{

cin>>n>>k;
cin>>str;

cout<<max(solve('a'),solve('b'));


return 0;
}
