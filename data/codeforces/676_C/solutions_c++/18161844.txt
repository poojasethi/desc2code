#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,k;
	string s;
	cin >> n >> k >> s;
	int a=0,b=0;
	int counter=0;
	int maxi =0;
	for (int i =0; i< n; i++)
	{
		if(s.at(i) == 'a')
			a++;
		else
			b++;
		if(min(a,b) > k)
		{
			maxi = max(maxi,a+b-1);
			if(s.at(counter) == 'a')
				a--;
			else
				b--;
			counter++;
		}
	}
	maxi = max(maxi, a+b);
	cout << maxi;
	return 0;
}
