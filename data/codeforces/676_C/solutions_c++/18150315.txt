#include<bits/stdc++.h>
using namespace std;
int main()
{

    ios::sync_with_stdio(0);
    int n,k,i=0,ans=0,ca=0,cb=0,left=0;
    string s;
    cin>>n>>k;
    cin>>s;
    for(int i=0;i<s.length();i++)
    {
        if(s[i]=='a')ca++; else cb++;
        if(min(ca,cb)>k)
        {
            if(s[left]=='a')ca--; else cb--; left++;
        }
        else ans++;
    }
    cout<<ans<<"\n";
}
