#include<iostream>
#include<map>
#include<string>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int k;
	cin>>k;
	string ss;
	cin>>ss;
	map<char,int> m;
	int maxi=0;
	for(int i=0,j=0;i<n;i++)
	{
		m[ss[i]]++;
		if(min(m['a'],m['b'])>k)
		{
			m[ss[j]]--;
			j++;
		}
		maxi=max(maxi,i-j+1);
	}
	cout<<maxi<<endl;
	return 0;
}
