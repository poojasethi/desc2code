#include <bits/stdc++.h>

using namespace std;

int n,k;
string s;

main(){
	cin >> n >> k;
	cin >> s;
	
	int l=0,r=0;
	
	int a=0,b=0;
	int ans = 1;
	if(s[0]=='a') a++;
	else b++;
	
	while(r!=n){
		if(min(a,b) > k){
			if(s[l]=='a') a--;
			else b--;
			l++;
		}else{
			if(ans < a+b) ans = a+b;
			r++;
			if(s[r]=='a') a++;
			else b++;
		}
	}
	cout << ans;
}
