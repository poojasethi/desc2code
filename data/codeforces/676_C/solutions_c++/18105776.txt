#include<bits/stdc++.h>

using namespace std;
int n,k,ans;
string s;

int solve(char c){

   int i,j,res=0;
   int arr[100010] = {0};
   for(i=0;i<n;i++){

     arr[i+1] = arr[i]+(s[i]==c);
   }

   for(i=1,j=0;i<=n;i++){

      while(arr[i]-arr[j]>k)j++;
      res = max(res,i-j);
   }
   return res;
}

int main(){

  cin>>n>>k>>s;
  ans = max(solve('a'),solve('b'));

  cout<<ans<<endl;

}
