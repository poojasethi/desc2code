#include <cstdio>
using namespace std;
int n,m,h,t,ans,sum[100001];
char c;
int main(){
  scanf("%d%d",&n,&m);
  scanf("%c",&c);
  for (int i=1;i<=n;++i){
    scanf("%c",&c);
    if (c=='a') sum[i]=sum[i-1];
    else sum[i]=sum[i-1]+1;
  }
  h=1;
  for (t=1;t<=n;++t){
    while (sum[t]-sum[h-1]>m&&t-h+1-(sum[t]-sum[h-1])>m) ++h;
    if (t-h+1>ans) ans=t-h+1;
  }
  printf("%d",ans);
}
