#include <iostream>
#include <cstdio>

using namespace std;
int main()
{
    int n,k;
	scanf("%d%d",&n,&k);
	char str[n+1];
	scanf("%s",str);
	int c,i,j,ans=0;
	for(c=0;c<2;c++)
	{
	    int bad=0;
		for(i=0,j=0;j<n;++j)
		{
		    if(str[j]-'a'!=c)
			++bad;
			if(bad>k)
			{
			    while(i<=j&&str[i]-'a'==c)
				    i++;
				i++;
				bad--;
			}
			ans=max(ans,j-i+1);
		}
	}
	printf("%d\n",ans);
    return 0;
}
