#include <bits/stdc++.h>
using namespace std;

int N,K;
string S;
map<char,int> m;

int main()
{
	cin >> N >> K;
	cin >> S;
	
	int res = 0;
	for(int i = 0,j = 0;i < N;i++){
		m [S [i]]++;
		if(min(m ['a'],m ['b']) > K){
			m [S [j]]--;
			j++;
		}
		res = max(i - j + 1,res);
	}
	
	cout << res << endl;

	return 0;
}
