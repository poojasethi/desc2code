#include<iostream>
using namespace std;
string str;
int n,k;
int solve(string str,char a)
{
	int i=0,j=0,num=0;
	int res=-1;
	for(;;)
	{
		while(str[j]!=a&&j<n)j++;
		while(j<n&&num<k){
			if(str[j++]==a){
				num++;
			}
			while(str[j]!=a&&j<n)j++;
		}
		res=max(res,j-i);
		if(i>=n)break;
		if(str[i++]==a)
			num--;
	}
	return res;
}
int main()
{
	cin>>n>>k;
	cin>>str;
	cout<<max(solve(str,'a'),solve(str,'b'))<<endl;
}
