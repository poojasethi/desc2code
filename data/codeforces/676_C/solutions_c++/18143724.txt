#include <bits/stdc++.h>
using namespace std;
string s;
int n, m;
int main() {
	cin >> n >> m >> s;
	int a = 0, b = 0, La = 0, Lb = 0, mx = 0;
	for (int R = 0; R < n; ++R) {
		if (s[R] == 'a') ++a;
		else ++b;
		while (a > m) {
			if (s[La] == 'a')
				--a;
			La++;
		}
		while (b > m) {
			if (s[Lb] == 'b')
				--b;
			Lb++;
		}
		mx = max(mx, max(R-La+1, R-Lb+1));
	}
	cout << mx << endl;
}
