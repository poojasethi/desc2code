#include<iostream>
using namespace std;
char str[100005];
int main(void)
{
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++)
	{
	cin>>str[i];
	}
	int l=0,r=0,ans=0,numa=0,numb=0;	
	while(r<n)
	{
		if(str[r]=='a')
		numa++;
		else numb++;
		if(numa<=k||numb<=k)
		{
		ans++;r++;
		}
		else
		{
			if(str[l]=='a')
			numa--;
			else numb--;
			l++;r++;
		}
	}
	cout<<ans<<endl;
	return 0;
}
