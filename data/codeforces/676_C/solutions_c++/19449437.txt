#include <bits//stdc++.h>
using namespace std;

int n, k, a, b, l, ans;
int main()
{
    string s;
    cin >> n >> k >> s;
    for(int  i = 0; s[i]; i++) {
        if(s[i] == 'a') a++;
        else b++;
        while(min(a, b) > k) {
            if(s[l] == 'a') a--;
            else b--;
            l++;
           // cout << i << endl;
        }
        ans = max(ans, i-l+1);
    }
    cout << ans << endl;
}
