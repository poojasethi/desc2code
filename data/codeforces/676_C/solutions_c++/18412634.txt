#include <iostream>
#include <map>
#include <string>

using namespace std;

map <char,long long> num;
long long n,ans,k;
string s;

int main()
{
    cin >> n >> k >> s;
    for (long long i=0,j=0;j<n;j++){
        num[s[j]]++;
        while (min(num['a'],num['b']) > k){
            num[s[i]]--;
            i++;
        }
        ans=max(ans,j-i+1);
    }
    cout << ans;
}
