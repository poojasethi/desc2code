#include <cstdio>
#include <algorithm>
using namespace std;

const int maxn = 1e5 + 5;
int ans, n, k, sa, sb, head, tail;
char s[maxn];

int main() {
	scanf("%d%d%s", &n, &k, s + 1);
	for(tail = 1; tail <= n; ) {
		s[tail++] == 'a' ? sa++ : sb++;
		min(sa, sb) > k ? head++, (s[head] == 'a' ? sa-- : sb--) : 1;
		ans = max(ans, sa + sb);
	}
	
	printf("%d\n", ans);
	return 0;
}
