#include<cstdio>
#include<cstdlib>
#define max(x,y) ((x)>(y)?(x):(y));
int n,k,i,j,l,m,t,a[100001],b[100001];char s[100005];
main(){
scanf("%d%d%s",&n,&k,s+1);
for(i=1;i<=n;i++){
	if(s[i]&1){
		a[i]=a[i-1]+1;
		b[i]=b[i-1];
	}
	else{
		a[i]=a[i-1];
		b[i]=b[i-1]+1;
	}
}
a[n+1]=b[n+1]=99999999;
for(i=1;i<=n;i++){
	while(a[j]-a[i-1]<=k)j++;
	m=max(m,j-i);
	while(b[l]-b[i-1]<=k)l++;
	m=max(m,l-i);
}
printf("%d",m);
}
