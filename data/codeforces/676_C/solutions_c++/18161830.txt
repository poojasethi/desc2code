#include<bits/stdc++.h>
using namespace std;
#define debug(x) cout<<#x<<" => "<<x<<"\n"
int main()
{
	int t;
	//cin>>t;
	t=1;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		string s;
		cin>>s;
		int a=0;
		int res=-1;
		int b=0;
		for(int i=0,j=0;i<n;i++)
		{
			if(s[i]=='a')
				a++;
			else
				b++;
			if(min(a,b)>k)
			{
				if(s[j]=='a')
					a--;
				else
					b--;
				j++;
			}
			res=max(res,i-j+1);
		}	
		cout<<res<<"\n";

	}
}
