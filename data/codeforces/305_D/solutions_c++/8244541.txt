#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 1000010;
const int mod = 1e9 + 7;

int n, m, k;
int a[N], s0[N], s1[N], pw2[N], ans;

int main() {
	scanf("%d %d %d", &n, &m, &k);
	for (int i = 1; i <= m; ++i) {
		int u, v;
		scanf("%d %d", &u, &v);
		if (u + 1 == v) continue;
		if (v - u != k + 1) {
			printf("0\n");
			return 0;
		}
		a[u] = 1;
	}

	pw2[0] = 1;
	for (int i = 1; i <= n; ++i) {
		pw2[i] = pw2[i - 1] * 2 % mod;
	}
	for (int i = 1; i <= n; ++i) {
		s0[i] = s0[i - 1] + (a[i] == 0);
		s1[i] = s1[i - 1] + (a[i] == 1);
	}

	for (int i = 1; i + k + 1 <= n; ++i) {
		if (s1[n] - s1[i] > 0) continue;
		if (i - k > 1 && s1[i - k - 1]) {
			if (a[i]) {
				printf("0\n");
				return 0;
			} else continue;
		}
		int cnt = s0[i - 1];
		if (i - k > 1) cnt -= s0[i - k - 1];
		ans = (ans + pw2[cnt]) % mod;
	}
	if (!s1[n]) {
		ans = (ans + 1) % mod;
	}
	printf("%d\n", ans);
}
