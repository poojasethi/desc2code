#include <iostream>

#include <cmath>
#include <climits>
#include <ctime>
typedef long long LL;
using namespace std;
const int MAX=1000000,Mo=1000000007;
bool b[MAX+10],B[MAX+10];
LL a[MAX+10];

int main(){
    int N,M,K,i,j,s=0;
    a[0]=1;
    for (i=1;i<=MAX;i++) a[i]=(a[i-1]+a[i-1])%Mo;
    cin >> N >> M >> K;
    ++K;
    LL ans=1;
    for (i=1;i<=M;i++){
        int u,v;
        cin >> u >> v;
        if (v-u!=1 && v-u!=K) ans=0;
        if (v-u==K){
            b[u]=1; ++s;
        }
    }
    int l=1,r=N-K;
    for (i=N;i;i--){
        if (i+K<=N && B[i+K] && b[i]) ans=0;
        B[i]=B[i+1] | b[i];
        if (b[i]){
            r=min(i,r);
            l=max(i-K+1,l);
        }
    }
    if (ans){
        if (s) ans=0;
        for (i=l;i<=r;i++) ans=(ans+a[min(K,N-K-i+1)-s-!b[i]])%Mo;
    }
    cout << ans << endl;
}
