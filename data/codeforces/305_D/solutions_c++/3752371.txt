#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,k,l,r,ll,rr,a,b,ans;
const int mod=1000000007;
int A[1100000],B[1100000];
int mul(int a,int b)
{
	int res=1;
	for (int i=a;b;b/=2,i=i*(long long)i%mod)
		if (b&1)
			res=res*(long long)i%mod;
	return res;
}
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for (int i=1;i<=n;++i) B[i]=1;
	for (int i=1;i<=m;++i)
	{
		scanf("%d%d",&a,&b);
		if (a==b-1) continue;
		if (a==b-1-k) {A[++A[0]]=a;B[a]=0;continue;}
		printf("0\n");
		return 0;
	}
	if (A[0])
	{
		ans=0;
		l=10000000;r=0;
		for (int i=1;i<=A[0];++i)
			l=min(l,A[i]),r=max(r,A[i]);
		if (l+k+1<=r) {printf("0\n");return 0;}
		for (int i=max(1,r-k);i<=min(n-k-1,l);++i)
		{
			int cnt=min(n-k-i,k+1)-A[0]-B[i];
			ans=(ans+mul(2,cnt))%mod;
		}
	}else
	{
		ans=1;
		for (int i=1;i<=n-k-1;++i)
		{
			r=min(n-k-1-i,k);
			ans=(ans+mul(2,r))%mod;
		}
	}
	printf("%d\n",ans);
	//system("pause");
}
