const ll mod=1000000007;
ll powmod(ll a,ll b) {ll res=1;a%=mod;for(;b;b>>=1){if(b&1)res=res*a%mod;a=a*a%mod;}return res;}
// head

const int N=1010000;
int n,m,k,u,v,ct;
int sm[N],pw[N],ans;

int query(int l,int r) { if (l<=r) return r-sm[r]-(l-1-sm[l-1]); else return 0;}

int main() {
	scanf("%d%d%d",&n,&m,&k);
	pw[0]=1;
	rep(i,1,n+1) pw[i]=pw[i-1]*2%mod;
	rep(i,0,m) {
		scanf("%d%d",&u,&v);
		if (v==u+1) continue;
		if (v==u+k+1) {sm[u]=1; ct++; continue;}
		puts("0"); return 0;
	}
	rep(i,1,n+1) sm[i]=sm[i-1]+sm[i];
	rep(i,k+1,n) {
		if (sm[i]-sm[i-k-1]!=ct) continue;
		ans=(ans+pw[query(i-k+1,min(i,n-k-1))])%mod;
	}
	if (ct==0) ans=(ans+1)%mod;
	printf("%d\n",ans);
}
