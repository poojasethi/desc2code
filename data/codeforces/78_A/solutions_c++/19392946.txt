#include<cstdio>
int solve(){
    int c = 0;
    char ch;
    while((ch = getchar()) != '\n'){
        if(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u')
            c++;
    }
    return c;
}
int main(){
    puts(solve() == 5 && solve() == 7 && solve() == 5 ? "YES" : "NO");
    return 0;
}
