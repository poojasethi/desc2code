#include<iostream>
#include<cstdio>
using namespace std;
int main(){
	int i;
	int cnt[3]={0},a[3]={5,7,5};
	char ch;
	bool flag=1;
	for(i=0;i<3;i++){
		while((ch=getchar())!='\n'){
			if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u') cnt[i]++;
		}
		if(cnt[i]!=a[i]) flag=0;
	}
	if(flag) cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
}
