#include <cstdio>

#define N 100005

int g(char *s) {
    int a[256] = {0};
    char ss[10]="aeiou";
    for (int i = 0; ss[i]; ++i) a[ ss[i] ] = 1;
    int ret = 0;
    for (; *s; ++s)
        ret += a[*s];
    return ret;
}

char a[N], b[N], c[N];

int main() {
    gets(a); gets(b); gets(c);
    puts( g(a)==5 && g(b)==7 && g(c)==5 ? "YES" : "NO" );
    return 0;
}
