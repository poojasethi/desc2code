# include <iostream>
# include <algorithm>
# include <cstdio>
using namespace std;
int main()
{
	char s[1005];
	int i,j,c[3];
	fill(c,c+3,0);
	for(j=0;j<3;j++)
	{
		gets(s);
		for(i=0;s[i];i++)
			if(s[i]=='a' || s[i]=='i' || s[i]=='e' || s[i]=='o' || s[i]=='u')
				c[j]++;
	}
	if(c[0]==5 && c[1]==7 && c[2]==5)	cout<<"YES"<<endl;
	else	cout<<"NO"<<endl;
	exit(0);
}
