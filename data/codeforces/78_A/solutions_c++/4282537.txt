#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int i,j,count;
	char a[101];
	for (j=0;j<3;j++)
	{
		gets(a);
		count=0;
		for (i=0;a[i];i++)
			if (a[i]=='a'||a[i]=='e'||a[i]=='i'||a[i]=='o'||a[i]=='u')
				count++;
		if (count!=5&&j!=1)
		{
			cout<<"NO";
			return 0;
		}
		if (count!=7&&j==1)
		{
			cout<<"NO";
			return 0;
		}
	}
	cout<<"YES";
	return 0;
}
