#include<stdio.h>
#include<string.h>
#include<string>
int main()
{
	char arr[400];
	int x[3]={};
	for(int c = 0;c < 3;c++)
	{
		gets(arr);
		for(int i = 0;arr[i];i++)
	{
		if(arr[i] == 'a' || arr[i] == 'e' || arr[i] == 'i' || arr[i] == 'u' || arr[i] == 'o')
		{
			x[c]++;
		}
	}
	}
	if(x[0] == 5 && x[1] == 7 && x[2] == 5) printf("YES");
	else printf("NO");
	return 0;
}
