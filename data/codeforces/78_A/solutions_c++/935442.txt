#include<iostream>
using namespace std;
int arr[3];
int main(){
	string a;
	for(int j=0;j<3;j++){
		getline(cin,a);
		for(int i=0;i<a.length();i++)
			if(a[i]=='a'||a[i]=='e'||a[i]=='i'||a[i]=='o'||a[i]=='u')
				arr[j]++;
	}
	if(arr[0]==5&&arr[1]==7&&arr[2]==5)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
}
