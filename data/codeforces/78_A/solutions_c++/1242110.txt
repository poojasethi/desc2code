#include<iostream>
#include<string>
using namespace std;
int func(string a)
{
int i,n=0;
for(i=0;a[i];i++)
{
if(a[i]=='i'||a[i]=='o'||a[i]=='a'||a[i]=='e'||a[i]=='u')
n++;
}
return(n);
}
int main()
{
string a,b,c;
int i,na=0,nb=0,nc=0;
getline(cin,a);
getline(cin,b);
getline(cin,c);
na=func(a);
nb=func(b);
nc=func(c);
if(na==5&&nb==7&&nc==5)
cout<<"YES";
else
cout<<"NO";
}
