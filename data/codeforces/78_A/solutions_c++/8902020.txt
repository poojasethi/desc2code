#include<iostream>
using namespace std;
int main()
{
string s;
int i,j,a[3]={0};
for(i=0;i<3;i++)
{
    getline(cin,s);
    for(j=0;j<s.length();j++)
    {
        if(s[j]=='a'||s[j]=='e'||s[j]=='i'||s[j]=='o'||s[j]=='u')
            a[i]++;
    }
}
if(a[0]==5 && a[1]==7 && a[2]==5)
    cout<<"YES";
else cout<<"NO";
return 0;

}
