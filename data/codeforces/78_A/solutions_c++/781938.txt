#include<iostream>
using namespace std;
int arr(string a)
{
	int con=0;
	for(int i=0;i<a.size();i++)
		if(a[i]=='a' || a[i]=='i' || a[i]=='e' || a[i]=='o' || a[i]=='u')
			con++;
	return con;

}
int main()
{
	string a,b,c;
	getline(cin,a);
	getline(cin,b);
	getline(cin,c);
	if(arr(a)==5 && arr(b)==7&& arr(c)==5)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;


}
