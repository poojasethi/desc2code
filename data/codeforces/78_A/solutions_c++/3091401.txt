#include <stdio.h>
int main()
{
    char str[128];
    int v[3]={5,7,5},c;
    bool y=1;
    for(int i=0;y && i<3;++i){
        gets(str);
        for(int j=c=0;str[j];++j)
            c+=str[j]=='i' || str[j]=='o' || str[j]=='e' || str[j]=='a' || str[j]=='u';
        y&=c==v[i];
    }
    puts(y?"YES":"NO");
    return 0;
}
