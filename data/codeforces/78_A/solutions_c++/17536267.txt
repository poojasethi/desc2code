#include <iostream>
#include <cstring>
using namespace std;
string gl="aeiou";
string s[3];
int p[]={5,7,5};
int main()
{
  for(int i=0;i<3;i++)getline(cin,s[i]);
  bool yes=true;
  for(int i=0;i<3 && yes;i++)
  {
    for(int j=0;j<s[i].size() && yes;j++)
      if (gl.find(s[i][j])!=-1)p[i]--;
    if (p[i]!=0)yes=false;
  }
  cout<<(yes?"YES":"NO");
  return 0;
}
