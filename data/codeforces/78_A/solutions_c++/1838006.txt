#include<iostream>
using namespace std;
bool f(char a)
{
  if (a=='a'||a=='e'||a=='i'||a=='o'||a=='u') return 1;
  return 0;
}
int main()
{
  int a[3]={0},k=0;
  char ch;
  while(cin.get(ch))
  {
    if (ch=='\n') k++;
    else if (f(ch))  a[k]++;
  }
  if (a[0]==5&&a[1]==7&&a[2]==5) cout<<"YES";  else cout<<"NO";
//  system("pause");
  return 0;
}
