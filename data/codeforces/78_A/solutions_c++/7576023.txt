#include<cstdio>
#include<cstring>
char s[110];
char v[6]="aeiou";
int main(){
	int flag=1,count,k=1;
	while(fgets(s,sizeof(s),stdin)!=NULL){
		count=0;
		for(int i=0;i<strlen(s);i++){
			if(strchr(v,s[i])!=NULL)count++;
		}
		if(k==1||k==3){
			if(count!=5)flag=0;
		}else if(count!=7)flag=0;k++;
	}
	if(flag==0)printf("NO\n");else printf("YES\n");
	return 0;
}
