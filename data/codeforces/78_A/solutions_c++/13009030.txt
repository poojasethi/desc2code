#include<iostream>
#include<string.h>
using namespace std;
int main()
{
bool t=1;
int j;
string s;
for(int i=0;i<3;i++)
{
getline(cin,s);
j=0;
for(int l=0;l<s.length();l++)
{
if(s[l]=='a' || s[l]=='e' || s[l]=='i' || s[l]=='o' || s[l]=='u') j++;
}
if((i==0 || i==2) && j!=5 || i==1 && j!=7) t=0;
}
if(t)
{
cout<<"YES";
}
else cout<<"NO";
return 0;
}
