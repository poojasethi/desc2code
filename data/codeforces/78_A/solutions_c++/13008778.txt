#include <cstdio>
#include <iostream>

using namespace std;

string s;

bool check(int x)
{
	getline(cin, s);
	int cnt = 0;
	for (int i = 0; i < (int)s.length(); ++i)
		if (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u')
			cnt++;

	return cnt == x;
}

int main()
{
	if (!check(5) || !check(7) || !check(5))
		cout << "NO";
	else
		cout << "YES";
}
