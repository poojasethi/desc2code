#include<cstdio>

int tot;
char s[111];

int main() {
	for(int i=0;i<3;i++) {
		gets(s);
		tot = 0;
		for(int j=0;s[j];j++)
			if(s[j] == 'a' || s[j] == 'e' || s[j] == 'i' || s[j] == 'o' || s[j] == 'u') tot++;
		if(i==1 && tot!=7 || i!=1 && tot!=5) {
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	
	return 0;
}
