#include<cstdio>
int m[5]={5,7,5},k;
int main()
{
    for(int i=0;i<3;++i,k=0)
    {
        char c;
        while((scanf("%c",&c) && c!='\n'))
            k+=c=='a'||c=='e'||c=='i'||c=='o'||c=='u';
        if(k!=m[i])
        {
            printf("NO\n");
            return 0;
        }
    }
    printf("YES\n");
    getchar();getchar();
}
