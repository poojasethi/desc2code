#include <iostream>
using namespace std;

string a, b, c;

int get(string s) {
	int cnt = 0;
	for (int i = 0; i < s.length(); ++i) {
		cnt += (s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u');
	}
	return cnt;
}

int main() {
	getline(cin, a);
	getline(cin, b);
	getline(cin, c);
	cout << (get(a) == 5 && get(b) == 7 && get(c) == 5 ? "YES" : "NO") << endl;
}
