#include <cstdio>
int main() {
    char s[110], c;
    for (int i = 0;i < 3;i++) {
        gets(s);
        for (int j = c = 0;s[j];j++)
            if (s[j] == 'a' | s[j] == 'e' | s[j] == 'i' | s[j] == 'o' | s[j] == 'u')
                c++;
        if (c != 5 + (i & 1) * 2) { puts("NO"); return 0; }
    }
    puts("YES");
}
