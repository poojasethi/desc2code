#include <iostream>

using namespace std;


const int Maxn = 100 * 1000 + 10;

#define L cout << "Lie" << endl
#define T cout << "Truth" << endl
#define N cout << "Not defined" << endl

int d[Maxn], a[Maxn], n, m, cnt, crm;
bool is[Maxn];


int main()
{
	cin >> n >> m;
	for(int i=0; i<n; i++)
	{
		cin >> a[i];
		if(a[i] > 0)
			d[a[i]]++;
		else
			d[-a[i]]--, cnt++;
	}

	for(int i=1; i<=n; i++)
		is[i] = d[i] + cnt == m, crm += is[i];
	for(int i=0; i<n; i++)
	{
		if(a[i] < 0)
		{
			if(is[-a[i]])
			{
				if(crm == 1)
					L;
				else
					N;
			}
			else
				T;
		}
		else if(is[a[i]])
		{
			if(crm == 1)
				T;
			else
				N;
		}
		else
			L;
	}
	return 0;
}
