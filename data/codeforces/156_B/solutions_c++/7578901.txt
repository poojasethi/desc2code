#include <iostream>
using namespace std;
int tmp[1111111],y[1111111],nono[1111111],maybe[1111111],n,m,times=0,yes=0,no=0;
int main(){
        cin>>n>>m;
	for(int i=1;i<=n && cin>>tmp[i];i++){
		if(tmp[i]>0)y[tmp[i]]++,yes++;
		else 	  nono[-tmp[i]]++,no++;
	}
	for(int i=1;i<=n;i++) (maybe[i]=(yes-y[i]+nono[i]==n-m))==1?times++:times=times;
	char statement[2][3][111]={{"Not defined","Lie","Truth"},{"Not defined","Truth","Lie"}};
	for(int i=1;i<=n;i++)
		if(tmp[i]>0)cout<<(statement[1][(!maybe[tmp[i]])*2+(maybe[tmp[i]] && (times==1))])<<endl;
		else        cout<<(statement[0][(!maybe[-tmp[i]])*2+(maybe[-tmp[i]] && (times==1))])<<endl;
}
