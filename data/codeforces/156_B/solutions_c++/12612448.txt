#include <stdio.h>
int n,m,x[100005],a[100005],cnt,ok;
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++){
        scanf("%d",&x[i]);
        if(x[i]>0) a[+x[i]]++;
        if(x[i]<0) {
        	a[-x[i]]--;
			m--;
        }
    }
    for(int i=1;i<=n;i++) if(a[i]==m) cnt++;
    for(int i=1;i<=n;i++){
        if(x[i]>0) ok=a[+x[i]]!=m?-1:+(cnt==1);
        if(x[i]<0) ok=a[-x[i]]!=m?+1:-(cnt==1);
        if(ok<0) printf("Lie\n");
        else if(ok==0) printf("Not defined\n");
        else printf("Truth\n");
    }return 0;
}
