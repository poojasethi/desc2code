#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int a[100005], b[100005];
int n, m;
int p[100005], pos=0, neg=0;
int isc[100005];
int main(void) {
	scanf("%d%d", &n, &m);
	for(int i=1;i<=n;i++) {
		scanf("%d", &p[i]);
		if(p[i]>0) a[p[i]]++, ++pos;
		else b[-p[i]]++, ++neg;
	}
	int cc=0;
	for(int i=1;i<=n;i++) {
		isc[i] = (a[i]+neg-b[i]==m);
		cc+=isc[i];
	}
	for(int i=1;i<=n;i++) {
		if(isc[abs(p[i])] && p[i]>0) {
			puts(cc>1? "Not defined":"Truth");
		} else if(isc[abs(p[i])] && p[i]<0) {
			puts(cc>1? "Not defined":"Lie");
		} else if(p[i]>0) puts("Lie");
		else puts("Truth");
	}
	return 0;
}
