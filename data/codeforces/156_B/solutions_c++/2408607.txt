#include <algorithm>
#include <stdio.h>
#define  maxn      100002
using    namespace std;

int n,m,cnt,neg;
int ar[maxn];
int ar2[maxn];
bool susp[maxn];

int main()
{
  scanf("%d%d",&n,&m);
  for(int i=1 ; i<=n ; i++)
  {
    scanf("%d",&ar[i]);
    if(ar[i]>0)  ar2[ar[i]]++;
    else         ar2[-ar[i]]--,neg++;  
  }
  for(int i=1 ; i<=n ; i++)
    if(ar2[i]+neg==m)
    {
      cnt++;
      susp[i]=true;
    }
  for(int i=1 ; i<=n ; i++)
  {
    if((ar[i]>0 && susp[ar[i]] && cnt==1) || (ar[i]<0 && !susp[-ar[i]]))           printf("Truth\n");
    else if((ar[i]>0 && !susp[ar[i]]) || (ar[i]<0 && susp[-ar[i]] && cnt==1))      printf("Lie\n");
    else                                                                           printf("Not defined\n");
  }
  return 0;
}
