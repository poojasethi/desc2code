#include <cstdio>
using namespace std;

inline int abs(int a){return a>0?a:-a;}

int main() {
  int n,m,cr,sumneg=0;
  int *rsp,*prs,*neg,fing,k=0;
  scanf("%d%d",&n,&m);
  rsp=new int[n];
  prs=new int[n]();
  neg=new int[n]();
  for(int i=0; i<n; ++i) {
    scanf("%d",&cr);
    rsp[i]=cr;
    fing=abs(cr)-1;
    if(cr<0) {
      ++sumneg;
      ++neg[fing];
    }
    else ++prs[fing];
  }
  cr=0;
  while(k<2&&cr<n) {
    if((prs[cr]+sumneg-neg[cr])==m)++k;
    ++cr;
  }
  for(int i=0; i<n; ++i) {
    fing=abs(rsp[i])-1;
    if((prs[fing]+sumneg-neg[fing])==m)printf(k==1?(rsp[i]>0?"Truth\n":"Lie\n"):"Not defined\n");
    else printf(rsp[i]<0?"Truth\n":"Lie\n");
  }
  return 0;
}
