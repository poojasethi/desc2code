#include<cstdio>
#include<cstring>
const int N = 100005;
int a[N],b[N],c[N],d[N],f=0;
bool u[N];
char str[100];
int main()
{
	int n,m,i,j,v;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	{
		scanf("%s",str);
		for(v=0,j=1;str[j];j++)
		{
			v=v*10+str[j]-'0';
		}
		if(str[0]=='+')
		{
			c[i]=1;
			a[v]++;
		}
		else
		{
			f++;
			b[v]++;
			c[i]=0;
		}
		d[i]=v;
	}
	int t=0;
	for(i=1;i<=n;i++)
	{
		if(a[i]+f-b[i]==m){t++;u[i]=true;}
	}
	for(i=1;i<=n;i++)
	{
		if(c[i])
		{
			if(!u[d[i]])puts("Lie");
			else if(t==1)puts("Truth");
			else puts("Not defined");
		}
		else
		{
			if(!u[d[i]])puts("Truth");
			else if(t==1)puts("Lie");
			else puts("Not defined");
		}
	}
	return 0;
}
