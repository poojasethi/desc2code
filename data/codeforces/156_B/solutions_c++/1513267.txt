#include<cstdio>
int n,m,po[100005],ne[100005],ac=0,x[100005],mu[100005],pp=0,nn=0,y;
int main(){
    scanf("%d%d",&n,&m);
    for (int i=1; i<=n; i++) po[i]=ne[i]=mu[i]=0;
    for (int i=1; i<=n; i++){
        scanf("%d",&x[i]);
        if (x[i]>0) ++po[x[i]],++pp;
        else ++ne[-x[i]],++nn;
    }
    for (int i=1; i<=n; i++){
        if (po[i]+nn-ne[i]==m) ++ac,mu[i]=1;
        else mu[i]=0;
    }
    for (int i=1; i<=n; i++){
        if (x[i]>0){
           if (!mu[x[i]]) printf("Lie\n");
           else if (ac==1) printf("Truth\n");
           else printf("Not defined\n");
        }
        else{
             if (!mu[-x[i]]) printf("Truth\n");
             else if (ac==1) printf("Lie\n");
             else printf("Not defined\n");
        }
    }
    return 0;    
}
