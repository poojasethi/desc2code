#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<map>
#include<vector>
#include<cmath>
using namespace std;

char s[100005];
int a[100005],cm[100005],ncm[100005],ans[100005];

int main()
{
	int n,m,i,j,k,no=0;
	char ss[10];
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
	{
		gets(ss);
		scanf("%c",&s[i]);
		scanf("%d",&a[i]);
		if(s[i]=='+') cm[a[i]]++;
		else ncm[a[i]]++,no++;
	}
	int cr=0;
	for(i=1;i<=n;i++)
	{
		if(cm[i]+no-ncm[i]==m)
			ans[i]=1,cr++;
	}
	for(i=0;i<n;i++)
	{
		if(s[i]=='+') 
		{
			if(ans[a[i]]==0) puts("Lie");
			else if(cr==1) puts("Truth");
			else puts("Not defined");
		}
		else
		{
			if(ans[a[i]]==0) puts("Truth");
			else if(cr==1) puts("Lie");
			else puts("Not defined");
		}
	}
}
