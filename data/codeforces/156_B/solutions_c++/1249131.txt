#include<cstdio>
#include<cstdlib>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n,m; scanf("%d%d",&n,&m);
	static int to[100000];
	rep(u,n) scanf("%d",to+u);

	int tot_minus=0;
	static int plus[100000],minus[100000];
	rep(u,n){
		if(to[u]>0){
			plus[abs(to[u])-1]++;
		}
		else{
			minus[abs(to[u])-1]++;
			tot_minus++;
		}
	}

	int n_hannin=0;
	static bool hannin[100000];
	rep(u,n){
		hannin[u]=(tot_minus-minus[u]+plus[u]==m);
		if(hannin[u]) n_hannin++;
	}

	if(n_hannin==1){
		rep(u,n){
			puts(hannin[abs(to[u])-1]==(to[u]>0)?"Truth":"Lie");
		}
	}
	else{
		rep(u,n){
			if(hannin[abs(to[u])-1]) puts("Not defined");
			else puts(to[u]>0?"Lie":"Truth");
		}
	}

	return 0;
}
