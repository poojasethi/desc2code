#include<iostream>
using namespace std;

int i,j,k,sum,m,n,x,a[100100],b[200100],c[200100];

int work(int x){return x>0 ? x+n : -x;}

int main()
{
	cin >> n >> m;
	for(i=1;i<=n;++i)
		cin >> x,x=work(x),a[i]=x,c[x]++;
	for(i=1;i<=n;++i)
		sum+=c[i];
	for(i=1;i<=n;++i)
		if (sum-c[i]+c[i+n]==m)
			b[i]=1,b[i+n]=1,k++;
	for(i=1;i<=n;++i)
	{
		x=a[i];
		if (k==1)
		{
			if (b[x] ^ (x<=n))
				cout << "Truth" << endl;
			else
				cout << "Lie" << endl;
			continue;
		}
		if (b[x])
			cout << "Not defined" << endl;
		else if (x<=n)
			cout << "Truth" << endl;
		else
			cout << "Lie" << endl;
	}
	return 0;
}
