#include <stdio.h>
using namespace std;

int N, M;
int sa[100000];
int ap[100000];
int am[100000];
bool p[100000];

int main() {
	scanf("%d %d", &N, &M);
	for (int i = 0; i < N; i++) {
		int s;
		scanf("%d", &s);
		if (s > 0)
			ap[s-1]++;
		else {
			am[-1-s]++;
			M--;
		}
		sa[i] = s;
	}
	int anz = 0;
	for (int i = 0; i < N; i++) {
		if (M == ap[i]-am[i]) {
			p[i] = true;
			anz++;
		}
	}
	for (int i = 0; i < N; i++) {
		if (sa[i] > 0) {
			if (!p[sa[i]-1])
				printf("Lie\n");
			else if (anz == 1)
				printf("Truth\n");
			else
				printf("Not defined\n");
		} else {
			if (!p[-1-sa[i]])
				printf("Truth\n");
			else if (anz == 1)
				printf("Lie\n");
			else
				printf("Not defined\n");
		}
	}
	return 0;
}
