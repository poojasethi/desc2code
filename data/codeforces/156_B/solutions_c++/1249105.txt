#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>

int n,m;
int i,j,date[100010],flag[100010],num,all;
int main()
{
	scanf("%d%d",&n,&m);
	all=0;
	memset(flag,0,sizeof(flag));
	for (i=1;i<=n;i++)
	{
		scanf("%d",&date[i]);
		if (date[i]>0)
		{
			flag[date[i]]++;
		}
		else
		{
			all++;
			flag[-date[i]]--;
		}
	}
	num=0;
	for (i=1;i<=n;i++)
	{
		if (flag[i]+all==m)	
		{
			flag[i]=1;
			num++;
		}
		else flag[i]=0;
	}
	for (i=1;i<=n;i++)
	{
		if (date[i]>0)
		{
			if (flag[date[i]]==0)printf("Lie\n");
			else if (num==1)printf("Truth\n");
			else printf("Not defined\n");
		}
		else
		{
			if (flag[-date[i]]==0)printf("Truth\n");
			else if (num==1)printf("Lie\n");
			else printf("Not defined\n");
		}
	}
	return 0;
}
