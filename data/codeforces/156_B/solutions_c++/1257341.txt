#include <iostream>
using namespace std;
const int MAXN = 1e5+1;
int i,n,m,x,k,a[MAXN],b[MAXN],c[MAXN],can[MAXN];
int main(){
    cin>>n>>m;
    for(i=1;i<=n;++i){
        cin>>a[i];
        if(a[i]>0) ++b[a[i]];
        else ++c[-a[i]],++k;
    }
    for(i=1;i<=n;++i)
        if(b[i]+k-c[i]==m)
            can[i]=1, ++x;
    for(i=1;i<=n;++i){
        if(a[i]>0&&can[a[i]]&&x==1||a[i]<0&&!can[-a[i]]) cout<<"Truth";
        else if(a[i]>0&&!can[a[i]]||a[i]<0&&can[-a[i]]&&x==1) cout<<"Lie";
        else cout<<"Not defined";
        cout<<endl;
    }
    return 0;
}
