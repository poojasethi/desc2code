#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,a[100010],ans,cnt,ok[100010],d1[100010],d2[100010];
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;++i){
        scanf("%d",&a[i]);
        if(a[i]<0){++d2[-a[i]];++cnt;}
        else ++d1[a[i]];
    }
    for(int i=1;i<=n;++i)
        if(cnt-d2[i]+d1[i]==m){
            ok[i]=1;++ans;
        }
    for(int i=1;i<=n;++i)
        if(a[i]>0){
            if(ans!=1&&ok[a[i]])printf("Not defined\n");
            else if(!ok[a[i]])printf("Lie\n");
            else printf("Truth\n");
        }
        else{
            if(ans&&!ok[-a[i]])printf("Truth\n");
            else if(ans>1)printf("Not defined\n");
            else printf("Lie\n");
        }
    return 0;
}
