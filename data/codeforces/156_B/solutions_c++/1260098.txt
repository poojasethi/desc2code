#include <stdio.h>
#include <string.h>
#include <algorithm>

using namespace std;

const int MAXN = 100010;
int n;
int m;
int a[MAXN];
int c[MAXN];

int ABS(int x) { return x > 0 ? x : -x; }

int main()
{
	int i, j;

	while (scanf ("%d %d", &n, &m) != EOF)
	{
		memset(c, 0, sizeof(c));
		
		int cnt = 0;
		for (i = 0; i < n; i++)
		{
			scanf (" %d", &a[i]);
			if (a[i] > 0)
			{
				c[a[i]]++;
			}
			else 
			{
				c[-a[i]]--;
				cnt++;
			}
		}

		int k = count(c + 1, c + n + 1, m - cnt);
		for (i = 0; i < n; i++)
		{
			if (c[ABS(a[i])] + cnt != m)
				puts(a[i] > 0 ? "Lie" : "Truth");
			else if (k == 1)
				puts(a[i] > 0 ? "Truth" : "Lie");
			else
				puts("Not defined");
		}
	}

	return 0;
}
