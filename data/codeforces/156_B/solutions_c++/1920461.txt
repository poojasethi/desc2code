#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <queue>
#include <map>
#include <string.h>
#include <math.h>

const int N=100010;
const int inf=0x3f3f3f3f;

using namespace std;

int n,m,a[N];
bool dp[N];
int t[N],f[N],tt,ff;

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(int i=0;i<n;i++)
		if(a[i]<0) f[-a[i]]++,ff++;
		else t[a[i]]++,tt++;
	int cnt=0;
	for(int i=1;i<=n;i++)
	{
		if(n-m==f[i]+tt-t[i]) dp[i]=1i,cnt++;
		else dp[i]=0;
	}
	for(int i=0;i<n;i++)
	{
		if(a[i]<0)
		{
			if(dp[-a[i]] && cnt==1) puts("Lie");
			else if(dp[-a[i]]) puts("Not defined");
			else puts("Truth");
		}
		else
		{
			if(dp[a[i]] && cnt==1) puts("Truth");
			else if(dp[a[i]]) puts("Not defined");
			else puts("Lie");
		}
	}
	return 0;
}
