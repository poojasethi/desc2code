#include<cstdio>
int n,m,x,cnt=0,sum=0;
int a[100005],c[200005],b[200005];
int cal(int x){if (x>0) return x+n; return -x;}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
		scanf("%d",&x),x=cal(x),a[i]=x,c[x]++;
	for (int i=1;i<=n;i++) sum+=c[i];
	for (int i=1;i<=n;i++){
		int s=sum-c[i]+c[i+n];
		if (s==m) b[i]=1,b[i+n]=1,cnt++;
		}
	for (int i=1;i<=n;i++){
		int x=a[i];
		if (cnt==1){
			if (b[x] ^ (x<=n)) printf("Truth\n");
			else printf("Lie\n");
			continue;
			}
		if (b[x]) printf("Not defined\n");
		else if (x<=n) printf("Truth\n");
		else printf("Lie\n");
		}
	return 0;
}
