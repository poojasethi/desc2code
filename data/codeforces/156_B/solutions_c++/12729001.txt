#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
int a[100100],f1[100100],f2[100100],f[100100];
int main(){
	int n,m;
	while(~scanf("%d%d",&n,&m)){
		int no=0;
		memset(f1,0,sizeof(f1));
		memset(f2,0,sizeof(f2));
		for(int i=1;i<=n;i++){
			scanf("%d",&a[i]);
			if(a[i]<0){
				f1[-a[i]]++;
				no++;
			}
			else f2[a[i]]++;
		}
		int sum=0;
		for(int i=1;i<=n;i++){
			f[i]=(f2[i]+no-f1[i]==m);
			sum+=f[i];
		}
		for(int i=1;i<=n;i++)
			if(a[i]>0){
			if(f[a[i]]&&sum==1) puts("Truth");
				else if(f[a[i]]) puts("Not defined");
					else puts("Lie");
			}
			else if(!f[-a[i]]) puts("Truth");
				else if(f[-a[i]]&&sum==1) puts("Lie");
					else puts("Not defined");
	}
}
