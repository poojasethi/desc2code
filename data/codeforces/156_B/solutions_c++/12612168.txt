#include <cstdio>
#include <set>
using namespace std;

const int maxn = 100010;
char c[maxn];
int n, m, cnt=0, a[maxn], pos[maxn]={0}, nag[maxn]={0};
set <int> s;

int main() {
	scanf("%d%d", &n, &m);
	for(int i=1; i <= n; i++) {
		scanf("\n%c%d", &c[i], &a[i]);
		if(c[i] == '+') pos[a[i]]++;
		else {
			nag[a[i]]++;
			cnt++;
		}
	}	
	for(int i=1; i <= n; i++) if(pos[i]+cnt-nag[i] == m) s.insert(i);
	if(s.size() == 1) for(int i=1; i <= n; i++) {
		if(c[i] == '+' && !s.count(a[i]) || c[i] == '-' && s.count(a[i])) puts("Lie");
		else puts("Truth");
	}
	else for(int i=1; i <= n; i++) {
		if(c[i] == '-' && !s.count(a[i])) puts("Truth");
		else if(c[i] == '+' && !s.count(a[i])) puts("Lie");
		else puts("Not defined");
	}
	return 0;
}
