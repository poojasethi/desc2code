#include <stdio.h>
//#include <conio.h>
int main()
{
	int i, n, m, p = 0;
	scanf("%d%d", &n, &m);
	int *v = new int[n], *vp = new int[n], *vm = new int[n];
	bool *s = new bool[n], *r = new bool[n];
	for (i = 0; i < n; i++) vp[i] = vm[i] = r[i] = 0;
	for (i = 0; i < n; i++)
	{
		scanf("%d", &v[i]);
		bool b = s[i] = v[i] > 0;
		if (b) p++; else v[i] = -v[i];
		v[i]--;
		if (b) vp[v[i]]++; else vm[v[i]]++;
	}
	int u = 0;
	for (i = 0; i < n; i++) if (r[i] = m == vp[i] + n - p - vm[i]) u++;
	for (i = 0; i < n; i++)
	{
		bool b = r[v[i]];
		printf("%s\n", !s[i] && !b || s[i] && b && u == 1 ? "Truth" : !s[i] && b && u == 1 || s[i] && !b ? "Lie" : "Not defined");
	}
	//getch();
	return 0;
}
