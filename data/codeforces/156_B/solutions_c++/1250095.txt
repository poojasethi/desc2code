#include <cstdio>

const int h = 100100;

int n, m, d[h], q, p, r;
char c[h];
int a[2][h];
bool f[h];

int main() {
	scanf("%d%d%*c", &n, &m);
	for (int i = 0; i < n; i++) {
		scanf("%c%d%*c", &c[i], &d[i]);
		d[i]--;
		if (c[i] == '-') {
			a[0][d[i]]++;
			q++;
		} else {
			a[1][d[i]]++;
			p++;
		}
	}
	for (int i = 0; i < n; i++) {
		if (a[1][i] + q - a[0][i] == m) {
			f[i] = 1;
			r ++;
		}
	}
	for (int i = 0; i < n; i++) {
		if (c[i] == '+') { 
			if (f[d[i]] == 1) {
				if (r == 1)
					puts("Truth");
				else
					puts("Not defined");
			} else
				puts("Lie");
		} else {
			if (f[d[i]] == 1) {
				if (r == 1)
					puts("Lie");
				else
					puts("Not defined");
			} else
				puts("Truth");
		}
	}
	return 0;
}
