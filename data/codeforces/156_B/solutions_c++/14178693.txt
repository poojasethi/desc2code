#include<cstdio>
#include<cmath>
#include<algorithm>
#define MAXN 100010
using namespace std;

int a[MAXN],c[MAXN];

int main(){
	int n,m,cnt=0;

	scanf("%d%d",&n,&m);
	for(int i=0;i<n;++i){
		scanf("%d",a+i);
		if(a[i]>0)
			++c[a[i]];
		else{
			--c[-a[i]];
			++cnt;
		}
	}
	int t=count(c+1,c+n+1,m-cnt);
	fprintf(stderr,"%d\n",t);
	for(int i=0;i<n;++i){
		if(c[abs(a[i])]!=m-cnt)
			puts(a[i]>0?"Lie":"Truth");
		else if(t==1)
			puts(a[i]<0?"Lie":"Truth");
		else
			puts("Not defined");

	}

	return 0;
}
