#include<cstdio>
int N,M,i,s,sum;
bool f[100005];
int a[100005],X[100005];
char p[100005],line[10];
int main()
{
	scanf("%d%d",&N,&M);
	gets(line);
	for(i=1;i<=N;++i)
	{
		p[i]=getchar();
		scanf("%d",&X[i]);
		if(p[i]=='-') a[X[i]]++;
		else a[X[i]]--;
		if(p[i]=='-') s++;
		gets(line);
	}
	for(i=1;i<=N;i++)
	{
		if(s-a[i]==M) f[i]=1;
		sum+=f[i];
	}
	for(i=1;i<=N;i++)
	{
		if(p[i]=='+')
		{
			if(!f[X[i]]) printf("Lie\n");
			else if(sum==1) printf("Truth\n");
				else printf("Not defined\n");
		}
		else
		{
			if(!f[X[i]]) printf("Truth\n");
			else if(sum==1) printf("Lie\n");
				else printf("Not defined\n");
		}
	}
	return 0;
}
