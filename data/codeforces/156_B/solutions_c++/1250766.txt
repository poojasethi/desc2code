#include<cstdio>
#include<cstring>
using namespace std;
int n,m,t[100005],f[100005],sf=0,st=0,x,b[100005],cnt=0,a[100005];
char s[10],c;
int ff(int x){return x>0?x:-x;}
int main(){
	scanf("%d%d",&n,&m);
	for (int i=1; i<=n; i++) t[i]=f[i]=b[i]=0;
	for (int i=1; i<=n; i++){
		scanf("%d",&a[i]);
		if (a[i]>0) ++t[a[i]],++st;
		else ++f[ff(a[i])],++sf;
	}
	for (int k=1; k<=n; k++){
		x=t[k]+sf-f[k];
		if (x==m) b[k]=1,++cnt;
	}
	for (int i=1; i<=n; i++){
		if (b[ff(a[i])]&&cnt==1&&a[i]>0) printf("Truth\n");
		else if (b[ff(a[i])]==0&&a[i]<0) printf("Truth\n");
		else if (b[ff(a[i])]&&cnt==1&&a[i]<0) printf("Lie\n");
		else if (b[ff(a[i])]==0&&a[i]>0) printf("Lie\n");
		else printf("Not defined\n");
	}
	return 0;
}
