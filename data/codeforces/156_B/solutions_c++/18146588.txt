#include <cstdio>
char tp[100010];
int a[100010];
int cnt0[100010],cnt1[100010],c0,c1,list[100010];
bool vis[100010];
int main(){
	int n,m;
	scanf("%d %d\n",&n,&m);
	for(int i=0;i<n;i++)scanf("%c%d\n",tp+i,a+i),a[i]--,tp[i]=='+'?cnt1[a[i]]++,c1++:(cnt0[a[i]]++,c0++);
	int cnt=0;
	for(int c=0;c<n;c++){
		if(cnt1[c]+c0-cnt0[c]==m)list[cnt++]=c,vis[c]=1;
	}
	if(cnt==1){
		int c=list[0];
		for(int i=0;i<n;i++){
			if((a[i]==c&&tp[i]=='+')||(a[i]!=c&&tp[i]=='-'))puts("Truth");
			else puts("Lie");
		}
		return 0;
	}
	for(int i=0;i<n;i++){
		if(vis[a[i]])puts("Not defined");
		else{
			if(tp[i]=='-')puts("Truth");
			else puts("Lie");
		}
	}
}
