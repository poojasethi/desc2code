#!/usr/bin/env python

import sys
import pdb

n,m = map(int, sys.stdin.readline().split())

a = [input() for i in xrange(n)]

pos,neg = [0]*n, [0]*n
suspects = [0]*(n+1)

for i,num in enumerate(a):
	if num > 0:	
		pos[a[i]-1] += 1 
	else:
		neg[-a[i]-1] += 1

sum_alibis = sum(1 for i in a if i < 0)

for i in xrange(n):
	if sum_alibis - neg[i] + pos[i] == m:
		suspects[i+1] = i+1;

T, L, ND = "Truth", "Lie", "Not defined"
suspects = set(suspects)
num = len(suspects)

for i in xrange(n):
	if a[i] > 0:
		if a[i] in suspects and num == 2:
			print T
		elif a[i] in suspects and num > 2:
			print ND
		else: 
			print L
			
	else:
		if -a[i] in suspects and num == 2:
			print L
		elif -a[i] in suspects and num > 2:
			print ND
		else:
			print T
