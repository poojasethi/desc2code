import sys
from collections import defaultdict

alibis = defaultdict(int)
accusers = defaultdict(int)

tot_alibis = 0
tot_accusers = 0

n, m = [int(s) for s in sys.stdin.readline().split()]

log = []
for i in xrange(n):
    x = int(sys.stdin.readline())
    log.append(x)
    if x > 0:
        accusers[x-1] += 1
        tot_accusers += 1
    else:
        alibis[-1-x] += 1
        tot_alibis += 1

possible_criminals = set()

for i in xrange(n):
    truths = tot_alibis - alibis[i] + accusers[i]
    if truths == m:
        possible_criminals.add(i)

# print possible_criminals
# print tot_alibis, alibis
# print tot_accusers, accusers

for i in xrange(n):
    x = log[i]
    if x > 0:
        if x-1 in possible_criminals:
            if len(possible_criminals) == 1:
                print "Truth"
            else:
                print "Not defined"
        else:
            print "Lie"
    else:
        if -1-x in possible_criminals:
            if len(possible_criminals) == 1:
                print "Lie"
            else:
                print "Not defined"
        else:
            print "Truth"
