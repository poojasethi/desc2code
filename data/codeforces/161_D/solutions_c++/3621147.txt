#include<iostream>
#include<vector>
#include<cstring>

using namespace std;

long n,k;
vector<long> node[50001];
long long ans=0;
long long f[50001][500];

void dfs(long u,long pre)
{
	f[u][0]=1;
	for(int i=0;i<node[u].size();i++)
	{
		if(node[u][i]==pre)
			continue;
		int v=node[u][i];
		dfs(v,u);
		for(int j=1;j<=k;j++)
			ans+=f[u][j-1]*f[v][k-j];
		for(int j=1;j<=k;j++)
			f[u][j]+=f[v][j-1];
	}
}

int main()
{
	long u,v;
	
	cin>>n>>k;
	
	memset(f,0,sizeof(f));

	for(int i=1;i<n;i++)
	{
		cin>>u>>v;
		node[u].push_back(v);
		node[v].push_back(u);
	}

	dfs(1,-1);

	cout<<ans<<endl;

	return 0;
}
