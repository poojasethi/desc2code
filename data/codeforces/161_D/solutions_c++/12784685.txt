#include <bits/stdc++.h>
#define FOR(i,n) for(int i=0;i<n;i++)
using namespace std;
const int maxn = 50000 + 5;const int maxk = 500 + 5;
int n,k,ans,dp[maxn][maxk],used[maxn];
vector<int> G[maxn];void dfs(int v){used[v] = 1;dp[v][0] = 1;for(int i = 0; i < G[v].size(); i++) {int u = G[v][i];if(used[u]) continue;dfs(u);FOR(j,k+1)ans += dp[u][j]*dp[v][k-j-1];FOR(j,k+1)dp[v][j+1] += dp[u][j];}}
int main(){scanf("%d%d", &n, &k);FOR(i,n-1){int u, v;scanf("%d%d", &u, &v);G[u].push_back(v);G[v].push_back(u);}dfs(1);cout << ans << endl;return 0;}
