#include<iostream>
#include<cstring>
#include<vector>
typedef long long ll;
using namespace std;
vector<int> g[50001];
ll f[50001][501],ans;
int n,k;
void dfs(int u,int pre)
{
	f[u][0]=1;
	int v,i,j;
	for(i=0;i<g[u].size();i++)
	{
		v=g[u][i];
		if(v==pre) continue;
		dfs(v,u);
		for(j=1;j<=k;j++)
		    ans+=f[u][j-1]*f[v][k-j];
		for(j=1;j<=k;j++)
		    f[u][j]+=f[v][j-1];
	}
}
int main()
{
	int i,u,v;
	cin >> n >> k;
	ans=0;
	memset(f,0,sizeof(f));
	for(i=1;i<n;i++)
	{
		cin >> u >> v;
		g[u].push_back(v);
		g[v].push_back(u);
	}
	dfs(1,-1);
	cout << ans << endl;
	return 0;
}
