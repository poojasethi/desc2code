#include <cstdio>
#include <vector>
#include <iostream>

using namespace std;

vector<int> e[50050];
long long d[510][50050];

int main() {
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n-1;i++) {
		int u,v;
		scanf("%d%d",&u,&v);
		u--; v--;
		e[u].push_back(v);
		e[v].push_back(u);
	}
	for(int i=0;i<n;i++) { d[0][i]=1; d[1][i]=e[i].size(); }
	for(int i=1;i<k;i++) {
		for(int j=0;j<n;j++) {
			int t=0;
			for(int l=0;l<e[j].size();l++) t+=d[i][e[j][l]];
			t-=(d[1][j]-1)*d[i-1][j];
			if (i==1) t-=d[i-1][j];
			d[i+1][j]=t;
		}
	}
	long long res=0;
	for(int i=0;i<n;i++) res+=d[k][i];
	cout << res/2 << endl;
	return 0;
}
