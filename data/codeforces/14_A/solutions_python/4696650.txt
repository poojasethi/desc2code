n,m=map(int, raw_input().split())
e=[raw_input() for i in [0]*n]
M=max([i.rindex('*') for i in e if '*' in i])
m=min([i.index('*') for i in e if '*' in i])
X=max([(''.join(i)).rindex('*') for i in zip(*e) if '*' in i])
x=min([list(i).index('*') for i in zip(*e) if '*' in i])

for i,s in enumerate(e):
    if i < x or i > X:
        continue
    print s[m:M+1]
