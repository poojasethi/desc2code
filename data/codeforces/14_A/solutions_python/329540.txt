n,m=map(int,raw_input().split())
out = []
l,r = m,-1
u,d = n,-1
for i in range(0,n):
	s = raw_input()
	if '*' in set(s):
		u = min(u,i)
		d = max(d,i)
		
		lst = [i for i,v in enumerate(s) if v=='*']
		l = min(l,min(lst))
		r = max(r,max(lst))
	out.append(s)

for i in range(u,d+1):
	print out[i][l:r+1]
