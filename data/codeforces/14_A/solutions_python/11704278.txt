s=[None for i in range(n)]
min_n,max_n,min_m,max_m=n,0,m,0
for i in range(n):
        s[i]=raw_input()
        t=s[i].find('*')
        if ~t:
                if min_n==n:
                        min_n=i
                if max_n<i:
                        max_n=i
                if min_m>t:
                        min_m=t
        t=s[i].rfind('*')
        if ~t and max_m<t:
                max_m=t
for i in s[min_n:max_n+1]:
