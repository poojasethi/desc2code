n, m = map(int, raw_input().split())
p = [raw_input() for i in range(n)]
l, r, t, b = 1000, -1000, 1000, -1000
for i in range(n):
    for j in range(m):
        if p[i][j] != '*': continue
        if i < t: t = i
        if i > b: b = i
        if j < l: l = j
        if j > r: r = j
for i in range(t, b + 1):
    print(p[i][l:r + 1])
