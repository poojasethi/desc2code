def lr_find(s,par):
    for i,c in enumerate(s[::par]):
        if c != '.': return i
    return len(s)

n,m = map(int,raw_input().split(' '))
empty = '.'*m
p = [raw_input() for i in xrange(n)]
while p[0] == empty:  p.remove(empty)
e = p.pop()
while e == empty: e = p.pop()
p.append(e)
l = min(lr_find(s,1) for s in p)
r = min(lr_find(s,-1) for s in p)
for i in p: print i[l:m-r]
