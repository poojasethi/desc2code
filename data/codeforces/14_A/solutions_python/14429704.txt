from itertools import takewhile

n,m = map(int,raw_input().split()[:2])

rows = [ raw_input()[:m] for _ in xrange(n) ]
cols = zip(*rows)

def min_len(it):
	return min(map(lambda r: len(list(takewhile(lambda x: x=='.', r))), it))


a,b = min_len(rows), min_len(cols)
c,d = min_len(map(lambda x: reversed(x),rows)), min_len(map(lambda x: reversed(x), cols))
c = m - c 
d = n - d
	
for r in rows[b:d]:
	print r[a:c]
