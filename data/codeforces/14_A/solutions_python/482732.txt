n, m = map(int, raw_input().split())

a = [ raw_input() for _ in range(n) ]

mini = 100
maxi = -1
minj = 100
maxj = -1
for i in range(n):
    for j in range(m):
        if a[i][j] == '*':
            mini = min(mini, i)
            maxi = max(maxi, i)
            minj = min(minj, j)
            maxj = max(maxj, j)
for s in a[mini: maxi + 1]:
    print s[minj: maxj + 1]
