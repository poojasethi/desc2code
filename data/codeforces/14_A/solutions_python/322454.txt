import math

lines = sys.stdin.readlines()[1:]
x_size = len(lines[0])
y_size = len(lines)
x1 = 50
x2 = 0
y1 = 50
y2 = 0

for y in range(y_size):
    for x in range(x_size):
        if lines[y][x] == '*':
            x1 = min(x1, x)
            x2 = max(x2, x)
            y1 = min(y1, y)
            y2 = max(y2, y)


for y in range(y1, y2 + 1):
    line = ""
    for x in range(x1, x2 + 1):
        line += lines[y][x]
