n,m=map(int, raw_input().split())
t = []
for i in range(n):
  t.append(raw_input().strip())
res = []
i0=10000
j0=10000
i1=0
j1=0
for i in range(n):
  for j in range(m):
    if t[i][j]=='*':
      i0 = min(i0,i)
      j0 = min(j0,j)
      i1 = max(i1,i)
      j1 = max(j1,j)
res=''
for i in range(i0,i1+1):
  for j in range(j0,j1+1):
    res+=t[i][j]
  res+='\n'
