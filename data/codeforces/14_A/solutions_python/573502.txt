n, m = map(int, raw_input().split())
a = []
for i in range(n):
    a.append(raw_input())
x1 = n
y1 = m
x2 = y2 = -1
for i in range(n):
    for j in range(m):
        if a[i][j] == "*":
            x1 = min(x1, i)
            x2 = max(x2, i)
            y1 = min(y1, j)
            y2 = max(y2, j)
a = a[x1:x2+1]
for k in a:
    print k[y1:y2+1]
