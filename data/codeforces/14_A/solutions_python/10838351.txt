n,m = [int(x) for x in raw_input().split()]
ls = [raw_input() for x in xrange(n)]

beg, end = 49, 0
top, bot = 0, 0
flag = True
for i, l in enumerate(ls):
    if l.find('*') == -1:
        if flag:
            top += 1
    else:
        flag = False
        beg = min(beg, l.find('*'))
        end = max(end, l.rfind('*'))
        # print beg, end
        bot = i+1
nl = [g[beg:end+1] for g in ls[top:bot]]
for i in nl:
    print i
