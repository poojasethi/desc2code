
n, m = [int(x) for x in raw_input().strip().split()]
paper = []
up = n
down = 0
left = m - 1
right = 0
for r in xrange(n):
    paper.append(raw_input().strip())
    line = paper[r]
    if line.find('*') >= 0 and up == n:
        up = r
    if line.find('*') >= 0:
        down = r
        if line.find('*') < left: left = line.find('*')
        if line.rfind('*') > right: right = line.rfind('*')
for r in xrange(up, down + 1):
    print paper[r][left:(right + 1)]
