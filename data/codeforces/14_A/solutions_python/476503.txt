#!/usr/bin/env python

n, m = map(int, raw_input().split())

data = []
for i in range(n):
    data.append(raw_input().strip())

min_x, max_x, min_y, max_y = (m, 0, n, 0)
for i in range(n):
    for j in range(m):
        if data[i][j] == '*':
            min_x = min(min_x, j)
            max_x = max(max_x, j)
            min_y = min(min_y, i)
            max_y = max(max_y, i)

for i in range(min_y, max_y + 1):
    print data[i][min_x : max_x + 1]
