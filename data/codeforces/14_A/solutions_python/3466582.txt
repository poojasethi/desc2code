n, m = map(int, raw_input().split())
rs, cs, data = [], [], []
for r in xrange(n):
    s = raw_input()
    data.append(s)
    for c in xrange(m):
        if s[c] == '*':
            rs.append(r)
            cs.append(c)
lr, ur, lc, uc = min(rs), max(rs), min(cs), max(cs)
for r in xrange(lr, ur + 1):
    print data[r][lc : uc + 1]
