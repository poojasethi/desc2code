n,m = map(int, raw_input().split())

s = [raw_input() for _ in range(n)]
a, b, c, d = n, m, 0, 0

for i in range(n):
    for j in range(m):
        if s[i][j] == '*':
            a = min(a, i) 
            b = min(b, j)
            c = max(c, i)
            d = max(d, j)
            
for i in range(a, c+1):
    print s[i][b:d+1]
