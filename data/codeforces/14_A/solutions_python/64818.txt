r, c = raw_input().split(" ")
r = int(r)
c = int(c)

lines = []
for x in range(r):
  lines.append(raw_input())

left = c
right = -1
top = r
bottom = -1

for x in range(c):
  for y in range(r):
    if lines[y][x] == "*":
      left = min(left, x)
      right = max(right, x)
      top = min(top, y)
      bottom = max(bottom, y)

for x in range(top, bottom+1):
  print lines[x][left: right+1]
