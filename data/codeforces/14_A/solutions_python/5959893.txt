n, m = map(int, raw_input().split())
paper = [raw_input() for _ in range(n)]
filtered = filter(lambda x: '*' in x, paper)
top = min([i for i in range(n) if '*' in paper[i]])
bottom = max([i for i in range(n) if '*' in paper[i]])
left = min(map(lambda x: x.find('*'), filtered))
right = max(map(lambda x: x.rfind('*'), filtered))
paper = map(lambda x: x[left:right + 1], paper[top:bottom + 1])
print reduce(lambda a, b: a + '\n' + b, paper)
