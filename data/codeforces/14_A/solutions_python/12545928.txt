n,m=map(int,raw_input().split())
#print "%d %d" %(n,m)
s=[raw_input() for i in xrange(n)]
x1,x2,y1,y2=n,0,m,0
for i in xrange(n):
    for j in xrange(m):
        if s[i][j]=='*':
            x1 = min(x1,i)
            x2 = max(x2,i)
            y1 = min(y1,j)
            y2 = max(y2,j)

for i in xrange(x1,x2+1):
    print s[i][y1:y2+1]
