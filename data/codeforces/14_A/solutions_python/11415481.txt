N,M=map(int,raw_input().split())
Ans=[list(raw_input())for _ in range(N)]
I=0
while not'*'in Ans[0]:
    Ans.pop(0)
while not'*'in Ans[-1]:
    Ans.pop()
while not'*'in[Ans[I][0]for I in range(len(Ans))]:
    Ans=[Ans[I][1:]for I in range(len(Ans))]
while not'*'in[Ans[I][-1]for I in range(len(Ans))]:
    Ans=[Ans[I][:len(Ans[0])-1]for I in range(len(Ans))]
for R in Ans:
    print''.join(R)
