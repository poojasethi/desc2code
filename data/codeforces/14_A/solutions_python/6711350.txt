[r,c]=map(int, raw_input().split())
matrix = [[0 for j in range(c)] for i in range(r)]
x,y,ans=[],[],""
for i in range(r):
	matrix[i]=list(raw_input())
	for j in range(c):
		if matrix[i][j]=="*":
			x.append(j)
			y.append(i)

for i in range(min(y),max(y)+1):
	for j in range(min(x),max(x)+1):
		ans = ans + matrix[i][j]
	ans = ans + "\n"

print ans
