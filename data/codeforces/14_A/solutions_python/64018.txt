#!/usr/bin/env python

import sys

x = [0] * 51

(n, m) = sys.stdin.readline().split(' ')
n = int(n)
m = int(m)

for i in range(0, n):
	x[i] = sys.stdin.readline()

x_st = x_sus = 100
x_dr = x_jos = 0

for i in range(0, n):
	for j in range(0, m):
		if x[i][j] == '*':
			x_st = min(x_st, j)
			x_dr = max(x_dr, j)
			x_sus = min(x_sus, i)
			x_jos = max(x_jos, i)

for i in range(x_sus, x_jos + 1):
	for j in range(x_st, x_dr + 1):
		sys.stdout.write(x[i][j])
	print ""
