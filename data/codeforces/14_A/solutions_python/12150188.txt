n,k = map(int,raw_input().split())
g = [raw_input() for _ in xrange(n)]
l = [(i,j) for i in xrange(n) for j in xrange(k) if g[i][j] == '*']
n = w = float('inf'); s = e = float('-inf')
for i,j in l:
    if j < w: w = j
    if j > e: e = j
    if i < n: n = i
    if i > s: s = i
for x in xrange(n,s+1):
    print g[x][w:e+1]
