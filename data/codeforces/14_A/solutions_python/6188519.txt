n,m=map(int,raw_input().split())
r=[]

for i in range(n):
    r.append(raw_input().strip())

b,e,br,er=m,m,n,0
for i in range(n):
    if '*' in r[i]:
        b=min(b,(r[i].index('*')))
        h=r[i][::-1]
        e=min(e,h.index('*'))
        br=min(br,i)
        er=max(er,i)
for i in range(br,er+1):
    print r[i][b:m-e]
