n,m = [int(x) for x in raw_input().split()]
arr = []
for i in xrange(0,n):
    arr.append(raw_input().strip())

xmin = 100
xmax = 0
ymin = 100
ymax = 0
for i in xrange(0,n):
    for j in xrange(0,m):
        if arr[i][j] == "*":
            if xmin > i: xmin = i
            if ymin > j: ymin = j
            if xmax < i: xmax = i
            if ymax < j: ymax = j


for i in xrange(xmin,xmax+1):
    s=""
    for j in xrange(ymin,ymax+1):
        s+=arr[i][j]
    print s
