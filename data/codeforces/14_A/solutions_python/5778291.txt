n,m=map(int,raw_input().split())
l=[]
for i in xrange(n):
	l.append(raw_input())
c=n-1
d=0
mi=m-1
ma=0
for i in xrange(n):
	for j in xrange(m):
		if l[i][j]=='*':
			if i<=c:
				c=i
			if i>=d:
				d=i
			if j<=mi:
				mi=j
			if j>=ma:
				ma=j
ans=""
for i in xrange(c,d+1):
	for j in xrange(mi,ma+1):
		ans+=l[i][j]
	ans+="\n"
print ans
