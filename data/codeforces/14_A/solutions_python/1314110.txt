n,m = map(int, str(raw_input()).split())
s = [raw_input().strip() for i in range(n)]
start = 0
for i in s:
    if '*' in i:
        break
    start += 1
end = 0
j = start
for i in s[start:]:
    if '*' in i:
        end = j+1
    j+=1
startx = m
endx = 0
for i in s[start:end]:
    a = i.rfind('*')
    if a!=-1:
        startx = min(startx, i.find('*'))
    endx = max(endx, a)
for i in s[start:end]:
    print i[startx:endx+1]
