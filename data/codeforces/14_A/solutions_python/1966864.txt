#!/usr/bin/python
import sys
(n, m) = sys.stdin.readline().split()
top = 100
bottom = -1
left = 100
right = -1
lines = sys.stdin.readlines()
for i in range(int(n)):
	l = lines[i]
	index1 = l.find('*')
	if index1 != -1:
		if bottom < i:
			bottom = i
		if top > i:
			top = i
		if left > index1:
			left = index1
		index2 = l.rfind('*')
		if right < index2:
			right = index2

for i in range(top, bottom+1):
	print lines[i][left:right+1]
