n, m = map(int, raw_input().split())
g = [None] * n
t = n + 1
r = -1
b = -1
l = m + 1
for i in range(n):
    g[i] = raw_input()
    for j in range(m):
        if g[i][j] == '*':
            if i < t:
                t = i
            if i > b:
                b = i
            if j < l:
                l = j
            if j > r:
                r = j
z = []
for i in range(t, b + 1):
    o = ''
    for j in range(l, r + 1):
        o += g[i][j]
    z.append(o)
print '\n'.join(z)
