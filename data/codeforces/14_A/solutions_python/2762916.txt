n,m= map(int,raw_input().split())
table = []
[table.append(raw_input()) for i in range(n)]
t1 = map(lambda x : x.find('*'),table)
t2 = map(lambda x : x.rfind('*'),table)
c1=min(filter(lambda x :x>=0 ,t1)   )
c2=max(t2)
for i in range(n):
    if t1[i]>=0:
        r1=i
        break
t1.reverse()
for i in range(n):
    if t1[i]>=0:
        r2=n-i-1
        break
for i in range(r1,r2+1):
    print (table[i])[c1:c2+1]
