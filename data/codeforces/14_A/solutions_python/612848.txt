n, m = map(int, raw_input().split())
s = [[]] * n
l = m
r = 0
u = n
d = 0
for i in range(n):
    s[i] = raw_input()
    for j in range(m):
        if(s[i][j] == '*'):
            if(j < l): l = j
            if(j > r): r = j
            if(i < u): u = i
            if(i > d): d = i
for i in range(u, d + 1):
    a = []
    for j in range(l, r + 1):
        a.append(s[i][j])
    print ''.join(a)
