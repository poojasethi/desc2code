from itertools import *
from fractions import *
import sys
n, m = map (int, raw_input ().split ())
a = [raw_input () for _ in range (n)]
l, u, r, d = 10**9, 10**9, 0, 0
for i in xrange (n):
    for j in xrange (m):
        if a[i][j] == '*':
            l = min (l, j)
            u = min (u, i)
            r = max (r, j)
            d = max (d, i)
for i in xrange (u, d + 1):
    print a[i][l:r + 1]
