#include <iostream>
using namespace std;
int main ()
{
int n,m,i,j,a=100,b=-100,c=100,d=-100;
cin>>n>>m;
char arr[n][m];
for(i=0;i<n;i++){
    for(j=0;j<m;j++){
        cin>>arr[i][j];
        if(arr[i][j]=='*'){
            a=min(a,j);
            b=max(b,j);
            c=min(c,i);
            d=max(d,i);
        }
    }
}
for(i=c;i<=d;i++){for(j=a;j<=b;j++){
    cout<<arr[i][j];
}
cout<<"\n";
}
return 0;
}
