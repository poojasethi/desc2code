#include <iostream>
using namespace std;
int main()
{	
	int i, j, n, m;
	cin >> n >> m;	
	char a[n][m+1];
	int x0=n, y0=m, x1=0, y1=0;
	for(i=0;i<n;i++){
		cin >> a[i];
		for(j=0;j<m;j++){
			if(a[i][j]=='*'){
				if(x0>i)x0=i;
				if(y0>j)y0=j;
				if(x1<i)x1=i;
				if(y1<j)y1=j;
			}
		}
	}
	for(i=x0;i<=x1;i++){
		for(j=y0;j<=y1;j++){
			cout << a[i][j];
		}
		cout << endl;
	}
	return 0;
}
