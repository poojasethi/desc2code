#include<cstdio>

int n, m, r1, c1, r2, c2;
char s[55][55];
int r[55], c[55];

int main() {
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++) {
		scanf("%s", s[i]);
		for(int j=0;j<m;j++)
			if(s[i][j]=='*') r[i]=c[j]=1;
	}
	for(r1=0;!r[r1];r1++);
	for(r2=n-1;!r[r2];r2--);
	for(c1=0;!c[c1];c1++);
	for(c2=m-1;!c[c2];c2--);
	for(int i=r1;i<=r2;i++) {
		for(int j=c1;j<=c2;j++)
			printf("%c",s[i][j]);
		printf("\n");
	}
	
	return 0;
}
