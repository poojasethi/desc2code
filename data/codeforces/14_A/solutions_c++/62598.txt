#include <stdio.h>
#include <algorithm>
using namespace std;

int main()
{
	int m,n,i,j,x1,x2,y1,y2;
	char c[51][51];
	x1=y1=100;
	x2=y2=0;
	scanf("%d%d",&m,&n);
	for(i=0;i<m;i++)
	{
		scanf("%s",c[i]);
	  for(j=0;j<n;j++)
	  {
	  	if (c[i][j]=='*')
	  	{
	  		x1 = min(x1,i);
	  		x2 = max(x2,i);
	  		y1 = min(y1,j);
	  		y2 = max(y2,j);
	  	}
	  }
	}
	for(i=x1;i<=x2;i++)
	{
	  for(j=y1;j<=y2;j++)
	    printf("%c",c[i][j]);
	  printf("\n");
	}
}
