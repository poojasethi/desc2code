#include<stdio.h>

int main(){
	int i,j;
	int n,m;
	int x1=0,x2=0;
	int y1=0,y2=0;
	char arr[55][55];
	scanf("%d %d",&n,&m);

	x1=n;
	y1=m;

	for(i=0 ; i<n ; i++)
		scanf("%s",arr[i]);

	for(i=0 ; i<n ; i++)
		for(j=0 ; j<m ; j++){
			if(arr[i][j]=='*'){
				if(i<x1)
					x1=i;
				if(i>x2)
					x2=i;
				if(j<y1)
					y1=j;
				if(j>y2)
					y2=j;
			}
		}

	for(i=x1 ; i<=x2 ; i++,puts(""))
		for(j=y1 ; j<=y2 ; j++)
			printf("%c",arr[i][j]);

	return 0;

}
