#include<iostream>
#include<numeric>
using namespace std;
char q[51][51];
int main(){
	int x,y;
	int a=100,b=-100,c=100,d=-100;
	cin>>x>>y;
	for(int i=0;i<x;i++){
		for(int j=0;j<y;j++){
			cin>>q[i][j];
			if(q[i][j]=='*'){
			
			a=min(a,i);
			b=max(b,i);
			c=min(c,j);
			d=max(d,j);
		}
	}
	}
	for(int i=a;i<=b;i++){
		for(int j=c;j<=d;j++)
		cout<<q[i][j];
		
		
		
		cout<<endl;}
	
	
	
	
	return 0;
}
