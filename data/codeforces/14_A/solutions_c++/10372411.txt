#include<iostream>
using namespace std;
int main(){

int n,m,x1=100,x2=-100,y1=100,y2=-100;
cin>>n>>m;
char let[n][m];
for(int i=0;i<n;i++)
{
for(int j=0;j<m;j++)
{
cin>>let[i][j];
if(let[i][j]=='*')
{
if(i<x1)
x1=i;
if(i>x2)
x2=i;
if(j<y1)
y1=j;
if(j>y2)
y2=j;
}
}
}
for(int i=x1;i<=x2;i++)
{
for(int j=y1;j<=y2;j++)
{
cout<<let[i][j];
}
cout<<endl;
}

}
