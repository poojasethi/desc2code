#include <iostream>
using namespace std;

int main ()
{
	char a[55][55];
	int n,m;
	cin>>n>>m;
	int u=60,d=-1;
	int r=-1,l=60;
	
	for (int i=0; i<n ; i++)
	for (int j=0; j<m ; j++)
	{
		cin>>a[i][j];
			if(a[i][j]=='*')
			{
				if(j<l)
				l=j;
				if(j>r)
				r=j;
				if(i<u)
				u=i;
				if(i>d)
				d=i;
			}
	}
	for (int i=u;i<=d ; i++)
	{
		for (int j=l;j<=r ; j++)
		{
			cout<<a[i][j];
		}
		cout<<endl;
	}
return 0;

}
