#include <iostream>

using namespace std;

int N,M,sag=-1000,sol=1000,alt=-1000,ust=1000;
char C[100][100];

int main()
{
	cin >> N >> M;
	for(int i=1;i<=N;i++)
		for(int j=1;j<=M;j++)
		{
			cin >> C[i][j];
			if(C[i][j] == '*')
			{
				alt=max(alt,i);
				ust=min(ust,i);
				sag=max(sag,j);
				sol=min(sol,j);
			}
		}
	for(int i=ust;i<=alt;i++,cout<<endl)
		for(int j=sol;j<=sag;j++)
			cout<<C[i][j];
}
