#include<iostream>
using namespace std;
int main () {
	int n,m,i,j;
	char a[55][55];
	cin >> n >> m;
	int sol1=60,sol2=60,sag1=1,sag2=1;//sol sag alt us
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			cin >> a[i][j];
			if(a[i][j]=='*')
			{
				if(j<=sol1)
					sol1=j;
				if(i<=sol2)
					sol2=i;
				if(j>=sag1)
					sag1=j;
				if(i>=sag2)
					sag2=i;
			}
		}
	for(i=sol2;i<=sag2;i++,cout << endl)
		for(j=sol1;j<=sag1;j++)
			cout << a[i][j];
	return 0;
}
