#include<iostream>
using namespace std;
int main()
{
	int m,n,i,j,l1,l2,c1,c2;
	char a[51][51];
	cin>>n>>m;
	l1=100;
	l2=0;
	c1=100;
	c2=0;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			cin>>a[i][j];
			if (a[i][j]=='*')
			{
				if (i<l1)
					l1=i;
				if (i>l2)
					l2=i;
				if (j<c1)
					c1=j;
				if (j>c2)
					c2=j;
			}
		}
	for(i=l1;i<=l2;i++)
	{
		for(j=c1;j<=c2;j++)
			cout<<a[i][j];
		cout<<endl;
	}
	return 0;
}
