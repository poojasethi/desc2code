#include <iostream>
using namespace std;
int main()
{
	char graph[50][50];
	int n,m,i,j,maxr=0,maxc=0,minr=60,minc=60;
	cin>>n>>m;
	for(i=0;i<n;i++)
	for(j=0;j<m;j++)
	{
	    cin>>graph[i][j];
	    if(graph[i][j]=='*')
	    {
	    	if(i<minr) minr=i;
	    	if(i>maxr) maxr=i;
	    	if(j<minc) minc=j;
	    	if(j>maxc) maxc=j;
	    }
	}
	for(i=minr;i<=maxr;i++)
	{
	for(j=minc;j<=maxc;j++)
	cout<<graph[i][j];
	cout<<endl;
	}
	return 0;
}
