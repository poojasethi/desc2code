# include <iostream>
# include <algorithm>
using namespace std;
char C[51][51];
int N,M,i,j,x1=51,x2=-1,y1=51,y2=-1;
int main()
{
	cin>>N>>M;
	for(i=0;i<N;i++)
		for(j=0;j<M;j++)
		{
			cin>>C[i][j];
			if(C[i][j]=='*')	{x1=min(x1,i); x2=max(x2,i);	y1=min(y1,j); y2=max(y2,j);}
		}
	for(i=x1;i<=x2;i++,cout<<endl)
		for(j=y1;j<=y2;j++)
			cout<<C[i][j];			
	exit(0);
}
