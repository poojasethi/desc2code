#include <stdio.h>

char a[100][100];
int r[100], c[100];
int main(void) {
	int n, m, i, j;
	scanf("%d%d", &n, &m);
	for(i=0;i<n;i++) {
		scanf("%s", a[i]);
		for(j=0;j<m;j++)
			if(a[i][j]=='*')
				++r[i], ++c[j];
	}
	while(r[n-1]==0) --n;
	while(c[m-1]==0) --m;
	for(i=0;i<n;i++) if(r[i]) break;
	for(;i<n;i++) {
		for(j=0;j<m;j++) if(c[j]) break;
		a[i][m] = 0;
		puts(a[i]+j);
	}
	return 0;
}
