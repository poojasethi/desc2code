#include<cstdio>
#include<algorithm>
using namespace std;
char s[50][51];
int main()
{
    int n,m,k,a,b=0,c,d=0;
    scanf("%d%d",&n,&m);a=n-1,c=m-1;
    for(int i=0;i<n;++i)
    {
        scanf("%s",s[i]);
        for(int j=0;j<m;++j)if(s[i][j]=='*')
        {
            a=min(a,i),b=max(b,i);
            c=min(c,j),d=max(d,j);
        }
    }
    for(int i=a;i<=b;++i,puts(""))
    for(int j=c;j<=d;++j)putchar(s[i][j]);
}
