#include <iostream>
#include <string.h>
#include<algorithm>
using namespace std;

int main()
{
int n,m;
int u =100,o=-10,l=50,r=-10;

cin>>n>>m;
string di [100];
for (int i=0;i<n;i++){
    cin>>di[i];
for (int j=0;j<m;j++){

if (di[i][j]=='*'){ u =min(u,i);
o=max(o,i);
 l=min(l,j);
 r=max(r,j);
 }

}
}

 for (int i=u; i<=o; i++)	{
		for (int j=l; j<=r; j++)
			cout << di[i][j];
		cout << endl;
	}



    return 0;
}
