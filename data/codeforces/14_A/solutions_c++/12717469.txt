

#include<cstdio>
int n,i,j,m;
char a[52][52],x=52,y=52,z,t;
int main()
{
	scanf("%d %d",&n,&m);
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			scanf(" %c",&a[i][j]);
			if(a[i][j]=='*')
			{
				if(i<x) x=i;
				if(i>z) z=i;
				if(j<y) y=j;
				if(j>t) t=j;
			}
		}
	for(i=x;i<=z;i++)
	{
		for(j=y;j<=t;j++)
			printf("%c",a[i][j]);
		puts("");
	}
	return 0;;
}
