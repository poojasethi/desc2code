#include<iostream>
using namespace std;
int main()
{
	int N,M,rs,re,cs,ce;
	cin>>N>>M;
	rs=N;
	cs=M;
	re=ce=-1;
	string s[N];
	for(int i=0;i<N;i++)
		cin>>s[i];
	for(int i=0;i<N;i++)
	{
		for(int j=0;j<M;j++)
		{
			if(s[i][j]=='*')
			{
				rs=min(rs,i);
				cs=min(cs,j);
				re=max(re,i);
				ce=max(ce,j);
			}
		}
	}
	for(int i=rs;i<=re;i++)
	{
		for(int j=cs;j<=ce;j++)
		{
			cout<<s[i][j];
		}
		cout<<"\n";
	}
	
	
	return 0;
}
