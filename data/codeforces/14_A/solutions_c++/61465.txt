#include<iostream>
#include<cstdio>
#include<cmath>
#include<string>
using namespace std;
char s[55][55];
int main(){
	int i,j;
	int n,m;
	int rs=999,re=0,cs=999,ce=0;
	cin>>n>>m;
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			cin>>s[i][j];
			if(s[i][j]=='*'){
				if(rs>i) rs=i;
				if(re<i) re=i;
				if(cs>j) cs=j;
				if(ce<j) ce=j;
			}
		}
	}
	for(i=rs;i<=re;i++){
		for(j=cs;j<=ce;j++){
			cout<<s[i][j];
		}
		cout<<endl;
	}
}
