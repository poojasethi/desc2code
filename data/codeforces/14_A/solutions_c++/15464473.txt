#include<iostream>
using namespace std;
int main(){
int n,m,l1,l2,h1,h2;
char c[50][50];
cin>>n>>m;
l1=m;l2=0;
h1=n;h2=0;
for(int i=0;i<n;++i){
	for(int j=0;j<m;++j){
		cin>>c[i][j];
		if(c[i][j]=='*'){
			if(i<h1)
			h1=i;
			if(i>h2)
			h2=i;
			if(j<l1)
			l1=j;
			if(j>l2)
			l2=j;
		}
	}
}
for(int i=h1;i<=h2;++i){
	for(int j=l1;j<=l2;++j)
	cout<<c[i][j];
	cout<<endl;
}
return 0;
}
