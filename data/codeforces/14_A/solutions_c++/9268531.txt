#include <bits/stdc++.h>
using namespace std;
int main ()
{
char a[51][51],d[51];
int b,c,x=51,y=0,x1=51,y1=0;
cin>>b>>c;
for (int i=0;i<b;i++)
{
for (int j=0;j<c;j++)
{
cin>>a[i][j];
}
}
for (int i=0;i<b;i++)
{
for (int j=0;j<c;j++)
{
if (a[i][j]=='*'){
x=min(x,i);
y=max(y,i);
x1=min(x1,j);
y1=max(y1,j);
}
}
}
for (int i=x;i<=y;i++)
{
for (int j=x1;j<=y1;j++)
{
cout<<a[i][j];
}
cout<<endl;
}
}
