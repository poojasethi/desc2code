#include<iostream>
using namespace std;
int main()
{
	int n,m,i,j;
	char a[55][55];
	cin>>n>>m;
	int lef=55,rig=0;
	int h=55,low=0;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			cin>>a[i][j];
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			if(a[i][j]=='*')
			{	if(h>i)
					h=i;
				if(i>low)
					low=i;
				if(j>rig)
					rig=j;
				if(j<lef)
					lef=j;
			}
	for(i=h;i<=low;i++)
	{	for(j=lef;j<=rig;j++)
			cout<<a[i][j];
		cout<<endl;
	}
	return 0;
}
