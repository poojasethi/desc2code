#include <bits/stdc++.h>
using namespace std;
char a[111][111];
int main()
{
	int n,m;
	cin>>n>>m;
	int p=1e9,q=0,r=1e9,s=0;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(a[i][j] == '*')
			{
				p = min(p,j);
				q = max(q,j);
				r = min(r,i);
				s = max(s,i);
			}
		}
	}
	for(int i=r;i<=s;i++)
	{
		for(int j=p;j<=q;j++)cout<<a[i][j];
		puts("");
	}
	return 0;
}
