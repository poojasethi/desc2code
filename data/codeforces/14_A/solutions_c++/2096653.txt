#include<stdio.h>
int main(){
	int n,m,i,j,mxx,mnx,mxy,mny;
	scanf("%d%d",&n,&m);
	char s[100][100];
	mxx=mxy=0;
	mny=mnx=100;
	for(i=0;i<n;i++)
	    scanf("%s",s[i]);
	for(i=0;i<n;i++)
	    for(j=0;j<m;j++)
			if(s[i][j]=='*'){
				mny=mny<j?mny:j;
				mxy=mxy>j?mxy:j;
				mnx=mnx<i?mnx:i;
				mxx=mxx>i?mxx:i;
			}
	for(i=mnx;i<=mxx;i++,putchar('\n'))
	    for(j=mny;j<=mxy;j++)
	        putchar(s[i][j]);
	 return 0;

}
