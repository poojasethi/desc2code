#include<iostream>
using namespace std;
int main()
{
	char c[100][100];
	int m,n,minx=100,miny=100,maxx=0,maxy=0;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		{
			cin>>c[i][j];
			if(c[i][j]=='*')
			{
				if(i>maxx)
					maxx=i;
				if(j>maxy)	
					maxy=j;
				if(i<minx)
					minx=i;
				if(j<miny)
					miny=j;
			}
		}
	for(int i=minx;i<=maxx;i++)
	{
		for(int j=miny;j<=maxy;j++)
			cout<<c[i][j];
		cout<<endl;
	}
}
