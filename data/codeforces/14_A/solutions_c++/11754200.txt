#include<iostream>
#include<string>
using namespace std;
char a[55][55];

int main(){
	int r[55],c[55],n,m,i,j,x1=51,x2=-1,y1=51,y2=-1;
	cin>>n>>m;
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			cin>>a[i][j];
			if(a[i][j]=='*'){
				if(i<x1)
					x1=i;
				if(i>x2)
					x2=i;
				if(j<y1)
					y1=j;
				if(j>y2)
					y2=j;
			}
		}
	}
	for(i=x1;i<=x2;i++){
		for(j=y1;j<=y2;j++)
	        cout<<a[i][j];
	     cout<<endl;
	}
	return 0;
}
