#include <bits/stdc++.h>
using namespace std;

int main()
{
	int r,c;
	cin>>r>>c;
	int rd=1,ru=r,cl=c,cr=1;

	char a[51][51];
	for(int i=1;i<=r;i++)
		for(int j=1;j<=c;j++)
		{
			scanf(" %c",&a[i][j]);
			if(a[i][j]=='*')
			{
				rd=max(rd,i);
				ru=min(ru,i);
				cl=min(cl,j);
				cr=max(cr,j);
			}
		}


	for(int i=ru;i<=rd;i++)
	{
		for(int j=cl;j<=cr;j++)
		{
			cout<<a[i][j];
		}
		cout<<"\n";
	}
	return 0;
}
