#include <bits/stdc++.h>
using namespace std;

#define MX 200005
int a[MX];
int b[MX * 2], bn;

int main() {
	
	int n, i, j, k;
	scanf("%d", &n);
	for (i = 1; i <= n; i++) scanf("%d", a + i);
	
	for (i = 1; i < n; i++) {
		if (i % 2 && a[i] >= a[i + 1] || i % 2 == 0 && a[i] <= a[i + 1]) b[bn++] = i;
	}
	if (bn > 4) {
		puts("0");
		return 0;
	}
	
	int cnt = 0;
	for (i = 1; i <= n; i++) if (i != b[0]) {
		b[bn++] = i;
		if (i > 1) b[bn++] = i - 1;
		if (b[0] > 1) b[bn++] = b[0] - 1;
		swap(a[i], a[b[0]]);
		
		for (k = 0; k < bn; k++) {
			int j = b[k];
			if (j == n) continue; 
			if (j % 2 && a[j] >= a[j + 1] || j % 2 == 0 && a[j] <= a[j + 1]) break;
		}
		if (k == bn) cnt++;
		
		swap(a[i], a[b[0]]);
		if (i > 1) bn--;
		if (b[0] > 1) bn--;
		bn--;
	}
	
	for (i = 1; i <= n; i++) if (i != b[0] && i != b[0] + 1) {
		b[bn++] = i;
		if (i > 1) b[bn++] = i - 1;
		if (b[0] + 1 < n) b[bn++] = b[0] + 1;
		 
		swap(a[i], a[b[0] + 1]);
		
		for (k = 0; k < bn; k++) {
			int j = b[k];
			if (j == n) continue; 
			if (j % 2 && a[j] >= a[j + 1] || j % 2 == 0 && a[j] <= a[j + 1]) break;
		}
		if (k == bn) cnt++;
		
		swap(a[i], a[b[0] + 1]);
		if (b[0] + 1 < n) bn--; 
		if (i > 1) bn--;
		bn--;
	}
	printf("%d\n", cnt);
	
	return 0;
}
