#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define pii pair<int,int>

using namespace std;
vector<int> v;
int n,a[150005];
set<pii> s;
bool ok(int x){
	if(x<n){
		if(x&1&&a[x]>=a[x+1])
			return 0;
		else if(x%2==0&&a[x]<=a[x+1])
			return 0;
	}
	if(x>1){
		--x;
		if(x&1&&a[x]>=a[x+1])
			return 0;
		else if(x%2==0&&a[x]<=a[x+1])
			return 0;
	}
	return 1;
}
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;++i){
		scanf("%d",&a[i]);
	}
	for(int i=1;i<=n;++i){
		if(!ok(i))
			v.pb(i);
	}
	if(v.size()>10){
		puts("0");
		return 0;
	}
	for(int k=0;k<v.size();++k){
		for(int i=1;i<=n;++i){
			swap(a[v[k]],a[i]);
			bool add=1;
			for(int j=0;j<v.size();++j)
				if(!ok(v[j]))
					add=0;
			if(!ok(i))
				add=0;
			swap(a[v[k]],a[i]);
			if(add)
				s.insert(mp(min(v[k],i),max(v[k],i)));
		}
	}
	printf("%d\n",(int)(s.size()));
}
