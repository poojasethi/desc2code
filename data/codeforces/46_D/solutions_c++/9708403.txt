#include <cstdio>
#include <map>
using namespace std;

int a[110];

int main()
{
	int l, b, f, n;
	scanf("%d%d%d%d", &l, &b, &f, &n);
	map<int, int>mp;
	mp[-b] = -b, mp[l+f] = l+f;
	int x, y;
	for(int i = 1; i <= n; ++i) {
		scanf("%d%d", &x, &y);
		if(x == 1) {
			a[i] = -1;
			for(map<int, int>::iterator it = mp.begin(), it2 = ++mp.begin(); it2 != mp.end(); ++it, ++it2) {
				if(it2->first - it->second >= y + f + b) {
					a[i] = it->second + b;
					mp[a[i]] = a[i] + y;
					break;
				}
			}
			printf("%d\n", a[i]);
		}
		else mp.erase(a[y]);
	}
	return 0;
}
