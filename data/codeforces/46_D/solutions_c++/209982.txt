#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define inf 100000000000000000LL
#define maxn 102022

int n, L;
int mark[maxn];

int main() {
	memset(mark, 0, sizeof(mark));
	int b, f;
	scanf("%d%d%d", &L, &f, &b);
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) {
		int ctrl;
		int iii = i;
		scanf("%d", &ctrl);
		if (ctrl == 1) {
			int len;
			scanf("%d", &len);
			int sLen = 0;
			int begin = 0;
			int ans = -1;
			for (int i = 0; i < L + b; i++) {
				if (mark[i]) {sLen = 0; begin = i + 1;continue;}
				sLen++;
				if (begin == 0 && sLen >= len + b) {
					ans = begin;
					break;
				}
				if (sLen >= len + f + b) {
					ans = begin + f;
					break;
				}
			}
			printf("%d\n", ans);
			if (ans == -1) continue;
			for (int j = ans; j < ans + len; j++)
				mark[j] = iii;
		} else {
			int id;
			scanf("%d", &id);
			for (int i = 0; i < L; i++)
				if (mark[i] == id) mark[i] = 0;
		}
//		for (int i = 0; i < L; i++)
//			printf("%d", mark[i]);
//		printf("\n");
	}
	return 0;
}
