#include <bits/stdc++.h>
using namespace std;

#define MAX 100010
#define fi first
#define se second

map<int,int> c;
map<int,int>::iterator cur, nxt;
int l, b, f, n, t, s;
int a[110];

int main(){
	// freopen("test.in", "rt", stdin);
	cin >> l >> b >> f;
	cin >> n;

	c[-b] = -b; c[l+f] = l+f;
	for (int i = 1; i <= n; i++){
		cin >> t >> s;
		if (t == 2)
			c.erase(a[s]);
		else{
			a[i] = -1;
			for (nxt = c.begin(), cur = nxt++; nxt != c.end(); cur++, nxt++){
				if (nxt->fi - cur->se >= s+b+f){
					a[i] = cur->se + b;
					c[a[i]] = a[i] + s;
					break;
				}
			}
			cout << a[i] << endl;
		}
	}
	return 0;
}
