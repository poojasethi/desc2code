#include <cstdio>
using namespace std;

int ans[111], s[111111], L, B, F, n, x, y;

int main()
{
	scanf("%d%d%d%d", &L, &B, &F, &n);
	for (int t = 1; t <= n; ++t){
		scanf("%d%d", &x, &y);
		if (x == 2){
			for (int i = ans[y]; s[i] == y; ++i)
				s[i] = 0;
		} else {
			bool ok;
			for (int i = 0, j = 0; i < L; j = ++i)
				if (s[i] == 0){
					while (s[j] == 0 && j < L) ++j;
					int len = j - i;
					ok = false;
					int cnt, k;				
					if (i == 0 && j == L && y <= len) { ok = true; k = 0;} else	
					if (i == 0 && (y + F) <= len) {ok = true; k = 0; } else
					if (len >= (F + B + y)) { ok = true; k = i + B;} else
					if (j == L && (y + B) <= len) {ok = true; k = i + B; }
					
					if (ok){
						ans[t] = k; cnt = y;
						for (; cnt > 0; --cnt, ++k)
							s[k] = t;
						break;
					} else 
						i = j;
				} else ;
			if (!ok) ans[t] = -1;
			printf("%d\n", ans[t]);
			//for (int i = 0; i < L; ++i) printf("%d", s[i]); puts("");
		}
	}
	//for (int i = 1; i <= n; ++i) printf("%d\n", ans[i]);
}
