#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

int parking[100500];
int l, b, f, n;

void Free(int a)
{
	for(int i = 0; i < l+b+f; i++)
	{
		if (parking[i] == a)
			parking[i] = 0;
	}
}

int occupe(int t, int a)
{
	int k = 0;
	int c = 0, x;
	int ll = a;
	int i = 0;
	while (i < l+b+f)
	{	
		for(k = i; k < l+b+f && parking[k]==0 && k - i < f + b + a - 1; k++);
		//cout << k << endl;
		if(k < l+b+f && parking[k]==0)
		{
			for(int j = i + b; j <= k - f; j++)
				parking[j]=t;
			return i;
		}
		i = k + 1;
	}
	return -1;
}

int main()
{
	//freopen("input.txt", "rt", stdin);
	//freopen("output.txt", "wt", stdout);
	cin >> l >> b >> f >> n;          
	memset(parking, 0, sizeof(parking));
	int a, t, k = 0;
	for(int j=0; j<n; j++)
	{
		cin >> t >> a;
		if(t == 1)
			cout << occupe(j+1, a) << endl;
		else
			Free(a);
	}
}
