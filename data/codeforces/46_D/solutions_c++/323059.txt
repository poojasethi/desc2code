#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

int p[111111]={};

int main(){
  int l,b,f;cin >> l >> b>> f;
  l+= b+f;
  int n;cin >> n;  
  int ind = 1;
  for(int i = 0;i<n;i++){
    int t,m;cin >> t >> m;
    
    if(t == 1){
      size_t pos = 0;
      while(pos < l){
	bool flag = true;
	for(int j = pos;j < pos+m+b+f;j++){
	  if(j >= l || p[j] != 0 ){
	    pos = j + 1;
	    flag = false;
	    break;
	  }
	}
	if(flag){
	  cout << pos <<endl;
	  for(int j = pos+b;j<pos+b+m;j++){
	    p[j] = ind;
	  }
	  break;
	}
      }
      if(pos >= l){
	cout << "-1" << endl;
      }
    }else{
      for(int i = 0;i<l;i++){
	if(p[i] == m){
	  p[i] = 0;
	}
      }
    }
    ind++;
    /*
    for(int i = 0;i<l;i++){
      cout << p[i] << " " ;
    }
    cout << endl;
    */
  }
  return 0;
}
