#include<cstdio>
#include<cstring>
#include<vector>
#include<queue>
#include<algorithm>
using namespace std;

int lim,f,b,n,s[110],r[110],nn;
int res[110];
int add(int len)
{
	for(int i=0;i<nn;++i)if(s[i+1]-b-(s[i]+r[i]+f)>=len)
	{
		for(int j=nn;j>i;--j)
			r[j+1]=r[j],s[j+1]=s[j];
		nn++;
		r[i+1]=len;
		s[i+1]=s[i]+r[i]+f;
		return s[i+1];
	}
	return -1;
}
void sub(int pos)
{
	for(int i=1;i<nn;++i)if(s[i]==pos)
	{
		for(int j=i;j<nn;++j)
			r[j]=r[j+1],s[j]=s[j+1];
		nn--;
		return;
	}
}
int main()
{
	scanf("%d %d %d",&lim,&f,&b);
	s[0]=-f;	r[0]=0;
	s[1]=lim+b; r[1]=0;
	nn=1;
	int m=0;
	scanf("%d",&n);
	for(int i=1,k,len;i<=n;++i)
	{
		scanf("%d %d",&k,&len);
		if(k==1){
			res[i]=add(len);
			printf("%d\n",res[i]);
		}
		else
			sub(res[len]);
	}
	return 0;
}
