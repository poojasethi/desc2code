#include <iostream>
#include <cmath>

using namespace std;

char memcomputa[8][8][8][45][45][45];

int mueve[3];
int lanza[3];

int maximo=0;

void computa(int e0,int e1,int e2,
	     int p0,int p1,int p2)
{
  char &mem=memcomputa[e0][e1][e2][p0][p1][p2];
  if (mem!=0) return;
  mem=1;
  maximo=max(maximo,max(p0,max(p1,p2)));
  // 0
  if (p0>2) {
    if (p1==0 or p2==0) {
      if ((e0&4)==0) {
	for (int i=-lanza[0];i<=lanza[0];i++)
	  if (i!=0 and p0+i>2 and p1!=p0+i and p2!=p0+i)
	    if (p1==0)
	      computa(e0|4,e1,e2,p0,p0+i,p2);
	    else
	      computa(e0|4,e1,e2,p0,p1,p0+i);
      }
    } else {
      if ((e0&2)==0) {
	if (p1>2 and (p1==p0-1 or p1==p0+1)) computa(e0|2,e1,e2,p0,0,p2);
	if (p2>2 and (p2==p0-1 or p2==p0+1)) computa(e0|2,e1,e2,p0,p1,0);
      }
      if ((e0&1)==0)
	for (int i=-mueve[0];i<=mueve[0];i++)
	  if (i!=0 and p0+i>2 and p1!=p0+i and p2!=p0+i)
	    computa(e0|1,e1,e2,p0+i,p1,p2);
    }
  }
  // 1
  if (p1>2) {
    if (p0==1 or p2==1) {
      if ((e1&4)==0) {
	for (int i=-lanza[1];i<=lanza[1];i++)
	  if (i!=0 and p1+i>2 and p0!=p1+i and p2!=p1+i)
	    if (p0==1)
	      computa(e0,e1|4,e2,p1+i,p1,p2);
	    else
	      computa(e0,e1|4,e2,p0,p1,p1+i);
      }
    } else {
      if ((e1&2)==0) {
	if (p0>2 and (p0==p1-1 or p0==p1+1)) computa(e0,e1|2,e2,1,p1,p2);
	if (p2>2 and (p2==p1-1 or p2==p1+1)) computa(e0,e1|2,e2,p0,p1,1);
      }
      if ((e1&1)==0)
	for (int i=-mueve[1];i<=mueve[1];i++)
	  if (i!=0 and p1+i>2 and p0!=p1+i and p2!=p1+i)
	    computa(e0,e1|1,e2,p0,p1+i,p2);
    }
  }
  // 2
  if (p2>2) {
    if (p0==2 or p1==2) {
      if ((e2&4)==0) {
	for (int i=-lanza[2];i<=lanza[2];i++)
	  if (i!=0 and p2+i>2 and p0!=p2+i and p1!=p2+i)
	    if (p0==2)
	      computa(e0,e1,e2|4,p2+i,p1,p2);
	    else
	      computa(e0,e1,e2|4,p0,p2+i,p2);
      }
    } else {
      if ((e2&2)==0) {
	if (p0>2 and (p0==p2-1 or p0==p2+1)) computa(e0,e1,e2|2,2,p1,p2);
	if (p1>2 and (p1==p2-1 or p1==p2+1)) computa(e0,e1,e2|2,p0,2,p2);
      }
      if ((e2&1)==0)
	for (int i=-mueve[2];i<=mueve[2];i++)
	  if (i!=0 and p2+i>2 and p0!=p2+i and p1!=p2+i)
	    computa(e0,e1,e2|1,p0,p1,p2+i);
    }
  }
}

int main()
{
  for (int i1=0;i1<8;i1++)
    for (int i2=0;i2<8;i2++)
      for (int i3=0;i3<8;i3++)
	for (int i4=0;i4<45;i4++)
	  for (int i5=0;i5<45;i5++)
	    for (int i6=0;i6<45;i6++)
		memcomputa[i1][i2][i3][i4][i5][i6]=0;
  int p[3];
  for (int i=0;i<3;i++) {
    cin>>p[i]>>mueve[i]>>lanza[i];
    p[i]+=2;
  }
  computa(0,0,0,p[0],p[1],p[2]);
  cout<<maximo-2<<endl;
}
