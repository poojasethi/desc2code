#include<bits/stdc++.h>
#define LL long long
using namespace std;
const int Max=10000000;
int n;
int a[1000005];
int pa[10000005];
LL gcd(LL a,LL b)
{
	if(!b)return a;
	return gcd(b,a%b);
}
int findset(int x)
{
	return pa[x]==x?x:(pa[x]=findset(pa[x]));
}
int main()
{
//	freopen("input.txt","r",stdin);
//	freopen("output1.txt","w",stdout);
	while(cin>>n)
	{
		memset(pa,0,sizeof(pa));
		for(int i=0;i<n;i++){scanf("%d",&a[i]);pa[a[i]]=a[i];}
		int ans=n;
		for(LL y=1;y<=Max;y++)
			for(LL x=y+1;2*x*y<=Max&&x*x-y*y<=Max;x++)if(gcd(x*x-y*y,2*x*y)==1)
			{
				LL a=x*x-y*y;
				LL b=2*x*y;
				LL c=x*x+y*y;
				if(pa[a]&&pa[b])
				{
					int fa=findset(a),fb=findset(b);
					if(fa!=fb)
					{
						pa[fb]=fa;
						ans--;
					}
				}
				if(c>Max)continue;
				if(pa[a]&&pa[c])
				{
					int fa=findset(a),fc=findset(c);
					if(fa!=fc)
					{
						pa[fa]=fc;
						ans--;
					}
				}
				if(pa[c]&&pa[b])
				{
					int fc=findset(c),fb=findset(b);
					if(fc!=fb)
					{
						pa[fc]=fb;
						ans--;
					}
				}
			}
		cout<<ans<<endl;
	}

	return 0;
}
