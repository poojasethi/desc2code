#include <cstdio>
#define MM 10000024
int n, ans;
int uf[MM];
int root(int x) { return uf[x] < 0 ? x : (uf[x] = root(uf[x])); }
bool conn(int x, int y) {
	x = root(x), y = root(y);
	if(x == y) return 0;
	if(uf[x] > uf[y]) x ^= y ^= x ^= y;
	uf[x] += uf[y]; uf[y] = x;
	return 1;
}

void merge(int a, int b) {
	if(a < MM && b < MM && uf[a] && uf[b] && root(a) != root(b)) conn(a, b), ans--;
}

void gen(int a, int b, int c) {
	if(a >= MM || b >= MM) return;
	merge(a, b), merge(a, c), merge(b, c);
	gen( a - 2*b + 2*c,  2*a - b + 2*c,  2*a - 2*b + 3*c);
	gen( a + 2*b + 2*c,  2*a + b + 2*c,  2*a + 2*b + 3*c);
	gen(-a + 2*b + 2*c, -2*a + b + 2*c, -2*a + 2*b + 3*c);
}

int main(void) {
	scanf("%d", &n);
	for(int i = 0, a; i < n; i++)
		scanf("%d", &a), uf[a] = -1;

	ans = n;
	gen(3, 4, 5);

	printf("%d\n", ans);

	return 0;
}
