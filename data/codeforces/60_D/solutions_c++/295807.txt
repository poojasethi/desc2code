#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>

using namespace std;

#define forn(i,n) for(int i=0;i<(int)n;++i)
const int N = (int)1e6, M=(int)3e7+10;
int n,a[N],p[M],ans;
bool w[M],used[M],vis[M];

int get(int i)
{
  if (p[i]==i)
    return i;
  return p[i]=get(p[i]);
}

void join(int a, int b)
{
  int c=get(a);
  int d=get(b);
  if (rand()&1)
    swap(c,d);
  p[c]=d;
}
int gcd(int a, int b){return b?gcd(b,a%b):a;}
#define rel(a,b) if(w[a]&&w[b]){join(a,b);used[a]=used[b]=true;}
int main()
{
  scanf("%d",&n);
  forn(i,n)
    {
      scanf("%d",&a[i]);
      w[a[i]]=1;
    }
  forn(i,M)
    p[i]=i;

  for(int u=1;u*u<=M/2;u++)
    for(int v=1;v<u;v++)
    {
      if (((u&1)==0 || (v&1)==0) && gcd(u,v)==1)
      {
        int a=2*u*v;
        int b=u*u-v*v;
        int c=u*u+v*v;
        rel(a,b);
        rel(a,c);
        rel(b,c);
      }
    }
  forn(i,n)
     if (!used[a[i]] || !vis[get(a[i])])
     {
       ans++;
       vis[get(a[i])]=true;
     }
  printf("%d\n",ans);
  return 0;
}
