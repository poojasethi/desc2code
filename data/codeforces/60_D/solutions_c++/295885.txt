#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
int f[20000007],n,g[20000007],x,y,z,ans;
long long otp=20000000;
long long p,q;
int gcd(int x,int y)
{
	if (y) return gcd(y,x%y); else return x;
}
int get(int x)
{
	if (!g[x]) return x;
		else
		{
			int s=get(g[x]);
			if (g[g[x]]) g[x]=g[g[x]];
			return s;
		}
}
void renew(int x,int y)
{
	if (x>otp/2 || y>otp/2) return ;
	if (!f[x] || !f[y]) return;
	int p=get(x);
	int q=get(y);
	if (p!=q) g[p]=q;
}

int main()
{
	cin>>n;
	for (int i=1;i<=n;i++) 
	{
		int a;
		scanf("%d",&a);
		f[a]=1;
	}
	p=1;
	while (p*p<=otp)
	{
		q=1;
		while (p*p+q*q<=otp && q<=p)
		{
			if (gcd(p,q)==1 && ((p+q)&1) )
			{
				x=2*p*q;
				y=p*p-q*q;
				z=p*p+q*q;
				renew(x,y);renew(y,z);renew(x,z);
			}
			q++;
		}
		p++;
	}
	for (int i=0;i<=otp/2;i++)
		if (f[i] && !g[i]) ans++;
	cout<<ans;
	return 0;
}
