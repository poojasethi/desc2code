const int N=10000010;
const int inf=0x3f3f3f3f;

using namespace std;

#define pb push_back

int n,a[1000010];
int p[N];

int find(int x)
{
	if(p[x]!=x) p[x]=find(p[x]);
	return p[x];
}
bool mp[N];

int main()
{
	for(int i=1;i<N;i++)
		p[i]=i;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]),mp[a[i]]=1;
	for(int i=1;i*i<N*4;i++)
	{
		for(int j=i+1;j*j<N*4;j++)
		{
			if(i%2 != j%2)
			{
				if(__gcd(i,j)==1)
				{
					int x=j*j-i*i;
					int y=2*i*j;
					int z=j*j+i*i;
					if(max(x,y)<N && mp[x] && mp[y] && find(x)!=find(y)) p[find(x)]=find(y);
					if(max(y,z)<N && mp[y] && mp[z] && find(y)!=find(z)) p[find(y)]=find(z);
					if(max(x,z)<N && mp[x] && mp[z] && find(x)!=find(z)) p[find(x)]=find(z);
				}
			}
		}
	}
	set<int> st;
	for(int i=1;i<=n;i++)
		st.insert(find(a[i]));
	printf("%d\n",st.size());
	return 0;
}
