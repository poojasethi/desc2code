#define LL long long

using namespace std;

const int maxn=10000010;
int n, m, fa[maxn], a[maxn], k;

inline int getanc(int x){return fa[x]<0?x:fa[x]=getanc(fa[x]);}

inline int check(int a, int b){
	if (fa[a]==0 || fa[b]==0) return 0;
	a=getanc(a); b=getanc(b);
	if (a==b) return 0;
	fa[a]=b; return 1;
}

int main(){
	cin.sync_with_stdio(false);
	cin>>n;
	for (int i=0; i<n; i++){
		cin>>k; fa[k]=-1;
	}

	int ans=n;
	for (LL i=1; i*i*2<=maxn; i++){
		for (LL j=i+1; 1; j++){
			LL a=-i*i+j*j, b=2*i*j, c=i*i+j*j;
			if (a>maxn || b>maxn) break;
			if (__gcd(a, b)!=1) continue;
			ans-=check(a, b);
			if (c>maxn) continue;
			ans-=check(a, c)+check(b, c);
		}
	}
	cout<<ans<<endl;
}
