#include<cstdio>
#include<algorithm>
using namespace std;

const int MAXN=10000000;
int f[MAXN+10];
int n;

int find(int u)
{
	int v,w;
	for (v=u;f[v]!=v;v=f[v]);
	for (;f[u]!=v;)
	{
		w=f[u];f[u]=v;u=w;
	}
	return v;
}

int merge(int a,int b)
{
	int aa=b,bb=a;
	if (!f[a] || !f[b]) return 0;
	a=find(a);
	b=find(b);
	if (a==b) return 0;
	f[a]=b;
	//printf("%d %d\n",aa,bb);
	return 1;
}

int main()
{
	//freopen("input.txt","r",stdin);

	scanf("%d",&n);
	for (int i=0,j;i<n;i++)
	{
		scanf("%d",&j);f[j]=j;
	}
	for (int i=1;i*i<=MAXN*4;i++)
	for (int j=i+1;j*j<=MAXN*4;j+=2)
	{
		if (__gcd(i,j)!=1) continue;
		int x=j*j-i*i,y=2*i*j,z=i*i+j*j;
		if (x>MAXN || y>MAXN) continue;
		n-=merge(x,y);
		if (z<=MAXN) n-=merge(x,z),n-=merge(y,z);
	}
	printf("%d\n",n);
}
