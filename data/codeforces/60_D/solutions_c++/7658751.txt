#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#define inf 10000010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,u;
int a[inf],f[inf],b[3];
int get(int x) {
	return f[x]==x?x:f[x]=get(f[x]);
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d",&k),a[k]=i,f[i]=i;
	int lim=int(sqrt(inf-1));
	For(i,1,lim)For(j,i+1,inf/2/i) {
		b[0]=2*i*j,b[1]=j*j-i*i,b[2]=j*j+i*i;
		if (j>lim&&b[1]>inf) break;
		if (__gcd(b[0],b[1])>1) continue;
		int F=0;
		For(k,0,2)if (b[k]<inf&&a[b[k]])
		For(u,k+1,2)if (b[u]<inf&&a[b[u]]) {
			int a1=a[b[k]],a2=a[b[u]];
			if (get(a1)!=get(a2)) {
				f[get(a1)]=get(a2);
				n--;
			}
		}
	}
	printf("%d\n",n);
	//for(;;);
	return 0;
}
