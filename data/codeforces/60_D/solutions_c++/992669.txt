#include<cstdio>  
#include<cstring>  
#include<cstdlib>  
#include<iostream>  
#include<algorithm>  
#include<map>
using namespace std;
#define fo(i,a,b) for(i=a;i<=b;++i)
#define fd(i,a,b) for(i=a;i>=b;--i)
char ch;
const int mn=10000024;
int n,i,a,ans,fa[mn];
void get(int &a)
{
    int p=1;
    for(ch=getchar();(ch<'0'||ch>'9') && (ch!='-');ch=getchar());
    if(ch=='-')a=0,p=-1;else a=ch-'0';
    for(ch=getchar();ch>='0'&& ch<='9';ch=getchar())a=a*10+ch-'0';
    a*=p;
}
int find(int r)
{
    return fa[r]==-1?r:fa[r]=find(fa[r]);
}
void merge(int a, int b)
{
    if(a<mn&&b<mn&&fa[a]&&fa[b]&&find(a)!=find(b))
        fa[find(a)]=find(b),ans--;
} 
void gen(int a,int b,int c)
{
    if(a>=mn||b>=mn)return;
    merge(a,b),merge(a,c),merge(b,c);
    gen(a-2*b + 2*c,2*a-b+2*c,2*a-2*b+3*c);
    gen(a+2*b+2*c,2*a+b+2*c,2*a+2*b+3*c);
    gen(-a+2*b+2*c,-2*a+b+2*c,-2*a+2*b+3*c);  
}
int main()
{
    get(n);
    fo(i,1,n)
    {
        get(a);
        fa[a]=-1;
    }
    ans=n;
    gen(3,4,5);
    printf("%d\n",ans);
    cin.get();cin.get();
    return 0;
}
