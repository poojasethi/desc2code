#include<cstdio>
#define max 10000000
int fa[max+10];
typedef long long ll;
ll gcd(ll a,ll b)
{
    return (a==0)?b:gcd(b%a,a);
}
int getfa(int a)
{
    if(fa[a]==a)return a;
    return fa[a]=getfa(fa[a]);
}
int m(ll a,ll b)
{
    if(a>max||b>max||fa[a]==0||fa[b]==0)return 0;
    int faa=getfa(a),fbb=getfa(b);
    if(faa==fbb)return 0;
    fa[faa]=fbb;
    //printf("%d %d\n",a,b);
    return 1;
}
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        int tmp;
        scanf("%d",&tmp);
        fa[tmp]=tmp;
    }
    int ans=n;
    for(ll i=1;i*i<=max/2;i++)
    {
        for(ll j=i+1;;j++)
        {
            ll a=2*i*j;
            ll b=j*j-i*i;
            if(a>max||b>max)break;
            if(gcd(a,b)>1)continue;
            ll c=j*j+i*i;
            if(gcd(a*b,c)>1)continue;
            ans-=m(a,b)+m(b,c)+m(c,a);
        }
    }
    printf("%d\n",ans);
}
