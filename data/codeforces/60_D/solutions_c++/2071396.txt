#include<cstdio>
#define max 10000000
int fa[max+10];
typedef long long ll;
ll gcd(ll a,ll b)
{
    while(a>0)
    {
        ll tmp=a;
        a=b%a;
        b=tmp;
    }
    return b;
}
int getfa(int a)
{
    return (fa[a]==a)?a:fa[a]=getfa(fa[a]);
}
int m(ll a,ll b)
{
    if(a>max||b>max||fa[a]==0||fa[b]==0)return 0;
    int faa=getfa(a),fbb=getfa(b);
    if(faa==fbb)return 0;
    fa[faa]=fbb;
    return 1;
}
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        int tmp;
        scanf("%d",&tmp);
        fa[tmp]=tmp;
    }
    int ans=n;
    for(ll i=2;i*i<=2*max;i+=2)
    for(ll j=i+1,a=i/2*(j+1),b=j,c=i/2*i+j;a<=max&&b<=max;j+=2,a+=i,b+=j,c+=j)
    {
        //if(a*a+b*b!=c*c)printf("error");
        if(gcd(a,b)>1)continue;
        if(gcd(a*b,c)>1)continue;
        ans-=m(a,b)+m(b,c)+m(c,a);
    }
    printf("%d\n",ans);
}
