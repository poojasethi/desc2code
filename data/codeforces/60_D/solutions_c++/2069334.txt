#include<cstdio>
#include<algorithm>
#include<iostream>
#include<memory>
using namespace std;
const int MAXN=10000000;
int father[MAXN+10];
int n;
inline int gcd( int a, int b ){
     return b==0?a:gcd( b, a%b );   
}
int find(int u){
	int v = u,w;
	while(v!=father[v])v = father[v];
	while(father[u]!=v){
		w=father[u];
        father[u]=v;
        u=w;
                      }
	return v;
}

int merge(int a,int b){
	if (!father[a]||!father[b])return 0;
	a=find(a);
	b=find(b);
	if (a==b) return 0;
	father[a]=b;
	return 1;
}
int main(){
    int a,i,j;
	scanf("%d",&n);
	for (i=1;i<=n;i++){
		scanf("%d",&a);
        father[a]=a;
	                 }
	for (i=1;i*i<=MAXN*4;i++)
	    for (j=i+1;j*j<=MAXN*4;j+=2){
		    if (gcd(i,j)!=1) continue;
	            int x=j*j-i*i,y=2*i*j,z=i*i+j*j;
		    if (x<=MAXN&&y<=MAXN)n-=merge(x,y);
		    if (x<=MAXN&&z<=MAXN)n-=merge(x,z);
		    if (z<=MAXN&&y<=MAXN)n-=merge(y,z);
	                                }
    
	printf("%d\n",n);
    return 0;
}
