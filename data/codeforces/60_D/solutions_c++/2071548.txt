#include<cstdio>
int fa[10000000+10];
typedef int ll;
ll gcd(ll a,ll b)
{
    ll tmp;
    while(a)
    {
        tmp=a;
        a=b%a;
        b=tmp;
    }
    return b;
}
int getfa(int a)
{
    return (fa[a]==a)?a:fa[a]=getfa(fa[a]);
}
int n;
void m(ll a,ll b)
{
    if(fa[a]==0||fa[b]==0)return;
    int faa=getfa(a),fbb=getfa(b);
    if(faa==fbb)return;
    fa[faa]=fbb;
    n--;
}
void get(int &a)
{
    int p=1;
    char ch;
    for(ch=getchar();(ch<'0'||ch>'9') && (ch!='-');ch=getchar());
    if(ch=='-')a=0,p=-1;else a=ch-'0';
    for(ch=getchar();ch>='0'&& ch<='9';ch=getchar())a=a*10+ch-'0';
    a*=p;
}
int main()
{
    get(n);
    int mx=0;
    for(int i=0;i<n;i++)
    {
        int tmp;
        get(tmp);
        fa[tmp]=tmp;
        if(mx<tmp)mx=tmp;
    }
    for(ll i=2;i*i<=2*mx;i+=2)
    for(ll j=i+1,a=i/2*(j+1),b=j,c=i/2*i+j;a<=mx&&b<=mx;j+=2,a+=i,b+=j,c+=j)
    {
        if(gcd(a,b)>1)continue;
        m(a,b);
        if(c<=mx){m(b,c);m(c,a);}
    }
    printf("%d\n",n);
}
