#include<cstdlib>
#include<cstdio>
#include<algorithm>
using namespace std;
//play
#define M 10000010
typedef long long ll;int fa[10000010],all,n,a[1000100];
int fin(int a){if(fa[a]!=a)fa[a]=fin(fa[a]);return fa[a];}
void work(int a,int b){if(fa[a]!=0&&fa[b]!=0){int f1=fin(a),f2=fin(b);if(f1!=f2)fa[f1]=f2,all++;}}
int main(){
scanf("%d",&n);
for(int i=0;i<n;i++){scanf("%d",a+i);fa[a[i]]=a[i];}
for(ll i=1;i<M;i++)for(ll j=i+1;2*i*j<M;j+=2){
if(j*j-i*i>M)break;if(__gcd(i,j)!=1)continue;
work(j*j-i*i,2*i*j);if(i*i+j*j<=M)work(j*j+i*i,2*i*j),work(j*j-i*i,i*i+j*j);}
printf("%d\n",n-all);}
