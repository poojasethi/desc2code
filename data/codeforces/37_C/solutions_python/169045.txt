N = input()
l = map(int, raw_input().split())
arr = ['' for i in xrange(len(l))]
if sum([2 ** (- p) for p in l]) > 1:
  print 'NO'
else:
  s = '0'
  while '' in arr:
    try:
      i = l.index(len(s))
      l[i] = 0
      arr[i] = s
      while s[-1] != '0':
        s = s[:-1]
      s = s[:-1] + '1'
    except:
      s = s + '0'
  print 'YES'
  print '\n'.join(arr)
