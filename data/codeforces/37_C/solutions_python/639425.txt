n=input()
l=zip(map(int,raw_input().split()),range(n))
l=sorted(l)[::-1]
s=''
a=['']*n
g=1
for (c,k) in l:
    if s:
        j=-1
        for i in range(c):
            if s[i]=='0': j=i
        if j==-1:
            g=0
            a=[]
            break
        s=s[:j]+'1'+'0'*(c-j-1)
    else:
        s='0'*c
    a[k]=s
print ['NO','YES'][g]
for s in a: print s
