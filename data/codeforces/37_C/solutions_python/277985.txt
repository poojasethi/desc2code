n, a = input (), sorted (enumerate (map (int, raw_input ().split ())), key=lambda x: x[1])
pref, ans = 2, [None] * n
for i, x in a:
    pref <<= x - len (bin (pref)[3:]) 
    s = bin (pref)[3:]
    ans[i] = s 
    if None in ans and '0' not in s:
        print 'NO' 
        exit (0)
    pref += 1
print 'YES\n' + '\n'.join (ans)
