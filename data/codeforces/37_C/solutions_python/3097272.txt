n=input()
c=[[] for _ in range(1024)]
s=['']*n
for i,x in enumerate(raw_input().split()):c[int(x)]+=[i]
m=['']*(n*2)
k=1
for i in range(1,1024):
  for j in range(k):
    m[k+j]=m[j]+'1'
    m[j]+='0'
  k=min(k*2,n)
  if len(c[i])>k:
    print 'NO'
    exit()
  for u in c[i]:
    k-=1
    s[u]=m[k]
print 'YES'
print '\n'.join(s)
