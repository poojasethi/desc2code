N = input()
l = map(int,raw_input().split())
s = sorted(l)
ans = []
cur = 0l
dig = 0
for x in s:
	cur <<= x - dig
	dig = x
	ans.append( format(cur,"0%db"%dig) )
	cur += 1
	if cur>=(1<<dig):
		break

if len(ans) == N:
	print "YES"

	used = [False]*N
	for x in l:
		for i in range(N):
			if not used[i] and len(ans[i])==x:
				print ans[i]
				used[i] = True
				break

else:
	print "NO"
