#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
using namespace std;
int a[5000],b[5000];
vector < string > v;
int counter = 0;
int n;
bool rec(int size, string s)
{
	if (counter == n)
		return true;
	if (size == a[counter])
	{
		v.push_back(s);
		counter++;
		return false;
	}
	if (rec(size + 1,s + '0') == true)
		return true;
	if (rec(size + 1,s + '1') == true)
		return true;		
	return false;
}
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin >> n;
	for (int i =0 ; i < n ; i++)
	{
		cin >> a[i];
		b[i] = a[i];
	}

	sort(a, a + n);
	if (!rec(0,"") && v.size() < n)
	{
		cout << "NO\n";
		return 0;
	}
	cout << "YES\n";
	for (int j = 0 ; j < n; j++)
	{
		for (int i = 0 ; i < v.size(); i++)
			if (v[i].size() == b[j])
			{
				cout << v[i] << '\n';
				swap(v[i],v[v.size() - 1]);
				v.pop_back();
				break;
			}
	}

	return 0;	
}
