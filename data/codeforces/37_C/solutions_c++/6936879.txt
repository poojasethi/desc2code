#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
int a[1001],cnt=0;
vector<string> s[1010];
int index[1010];
void solve(string t,int len)
{
    if(a[len]>0)
    {
        s[len].push_back(t);
        a[len]--;
        if(a[len]==0)cnt--;
        return;
    }
    if(cnt==0)return;
    solve(t+"0",len+1);
    if(cnt==0)return;
    solve(t+"1",len+1);

}
int arr[1010];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        int tmp;
        scanf("%d",&tmp);
        arr[i]=tmp;
        a[tmp]++;
        if(a[tmp]==1)cnt++;
    }
    solve("",0);
    if(cnt>0)
    {
        printf("NO\n");
        return 0;
    }
    printf("YES\n");
    for(int i=0;i<n;i++)
    {
        cout<<s[arr[i]][index[arr[i]]++]<<endl;

    }
   //     cout<<s[i]<<endl;

    return 0;
}
