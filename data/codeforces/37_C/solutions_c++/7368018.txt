#include<cstdio>
#include<algorithm>
using namespace std;
int sv,d[1001],a[1001],lst = -1,ans[1001],u,n;
bool cmp(int i,int j){ return a[i] < a[j]; } 
int P(int x,int &sd){
  if(x < 2){ printf("%d",x); sd--; return 0;}
  P(x/2,sd); 
  printf("%d",x%2); sd--;
}
int main(){
  scanf("%d",&n);
  for(int i=0;i<n;i++)
    scanf("%d",&a[i]),d[i] = i;
  sort(d,d+n,cmp);
  for(int i=0;i<n;i++){
    if(lst != -1) sv = sv<<(min(30,a[d[i]])-lst);
    if(a[d[i]] <= 30 && sv >= (1<<(a[d[i]]))){ u = -1; break;}
    ans[d[i]] = sv; lst = min(30,a[d[i]]);
    sv++;
  }
  if(u ==-1) printf("NO");
  else{
    printf("YES\n");
    for(int i=0;i<n;i++){
      int xx = a[i];
      for(int j=1;j<min(xx,30);j++){
    if(ans[i] < (1<<j))
      printf("0"),a[i]--;
      }
      P(ans[i],a[i]);
      for(int j=0;j<a[i];j++) printf("0");
      printf("\n");
    }
  }
  return 0;
}
