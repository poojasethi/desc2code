#include <cstdio>
#include <cstring>
#include <string>
#include <algorithm>
#include <iostream>
using namespace std;
struct Node{
	int x,id;
	string s;
}node[1010];
char str[1010];
int maxs,n,k;

bool cmpx(Node a,Node b){return a.x<b.x;}
bool cmpid(Node a,Node b){return a.id<b.id;}

void dfs(int i){
	if(k>=n || i>maxs) return;
	if(i == node[k].x){
		str[i] = '\0';
		node[k].s = str;
		k++;
		return;
	}
	for(int j = 0;j<=1;j++){
		str[i] = j + '0';
		dfs(i+1);
	}
}

int main(){
	while(cin>>n){
		maxs = -1;
		for(int i = 0;i<n;i++){
			cin>>node[i].x;
			if(node[i].x>maxs) maxs = node[i].x;
			node[i].id = i;
		}
		sort(node,node+n,cmpx);
		k = 0;
		dfs(0);
		if(k == n){
			puts("YES");
			sort(node,node+n,cmpid);
			for(int i = 0;i<n;i++){
				cout<<node[i].s<<endl;
			}
		}
		else puts("NO");
	}	
	return 0;
}
