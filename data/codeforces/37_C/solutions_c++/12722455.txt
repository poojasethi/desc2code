#include<bits/stdc++.h>
using namespace std;
#define FI "language.i1"
#define FO "language.o1"
#define NN 1001
vector< pair<int,int> > vp;
string res[NN];
bool ok = true;

void add(string &s){
    for(int i=s.size()-1;i>=0;i--){
        if(s[i]=='0'){ s[i] = '1'; return; }
        else s[i] = '0'; 
    }
    s = "";
}

int main(){
    int n;
    #ifndef ONLINE_JUDGE
    freopen(FI,"r",stdin);
    freopen(FO,"w",stdout);
    #endif // ONLINE_JUDGE
    cin >> n;
    for(int l, i = 0;i < n; i++){
        cin >> l;
        vp.push_back(make_pair(l, i));
    }
    string cur = "0";
    sort(vp.begin(), vp.end());
    for(int i= 0;i < n; i++){
        if(cur.size()== 0){ ok = false; break; }
        while(cur.size() != vp[i].first) cur += "0";
        res[vp[i].second] = cur;
        add(cur);
    }
    if(ok){
        cout << "YES\n";
        for(int i = 0; i < n; i++) cout << res[i] << endl;
    }
    else cout <<"NO\n";
}
