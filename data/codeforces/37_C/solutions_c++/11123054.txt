#include<cstdlib>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<string>
#include<iostream>

using namespace std;

struct node
{
	int pos,val;
};
node a[2000];
int ans[2000][2000],cur_ans[2000],n;

bool cmp(node x,node y)
{
	return x.val<y.val;
}

void add()
{
	cur_ans[cur_ans[0]]++;
	for(int i=cur_ans[0];i>1;i--)
		if(cur_ans[i]==2)
		{
			cur_ans[i]=0;
			cur_ans[i-1]++;
		}
	if(cur_ans[1]==2)
		cur_ans[0]=-1;
}

void set_ans(int x)
{
	for(int i=0;i<=cur_ans[0];i++)
		ans[x][i]=cur_ans[i];
}

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i].val;
		a[i].pos=i;
	}
	sort(a,a+n,cmp);
	cur_ans[0]=a[0].val;
	set_ans(a[0].pos);
	add();
	for(int i=1;i<n;i++)
	{
		if(cur_ans[0]==-1)
		{
			cout<<"NO";
			return 0;
		}
		cur_ans[0]=a[i].val;
		set_ans(a[i].pos);
		add();
	}
	cout<<"YES"<<endl;
	for(int i=0;i<n;i++)
	{
		for(int j=1;j<=ans[i][0];j++)
			cout<<ans[i][j];
		cout<<endl;
	}
	return 0;
}
