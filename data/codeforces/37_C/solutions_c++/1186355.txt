#include<stdio.h>
#include<string.h>
#define MAXN 1050
#include<algorithm>
using namespace std;

int n;
struct node{
	int id,x;
	char s[MAXN];
	bool operator < (const node &b)const {
		return id<b.id;
	}
}a[MAXN];
char rcd[MAXN];
int top;
bool flag;

bool comp(node a,node b){
	return a.x<b.x;
}

void dfs(int cur){
	if(flag==false)return ;
	if(cur==a[top].x){
		rcd[cur]='\0';
		strcpy(a[top].s,rcd);		
		top++;
		return ;
	}
	if(top==n)return ;
	if(cur>a[top].x){
		flag=false;
		return ;
	}
	rcd[cur]='0';
	dfs(cur+1);
	rcd[cur]='1';
	dfs(cur+1);		
	
}

void solve(){
	sort(a,a+n,comp);
	top=0;flag=true;
	dfs(0);
	//printf("%d\n",top);
	if(flag && top==n){
		printf("YES\n");
		sort(a,a+n);
		for(int i=0;i<n;i++)puts(a[i].s);
	}
	else printf("NO\n");
	
}

int main(){
	while(scanf("%d",&n)!=EOF){
		for(int i=0;i<n;i++){
			scanf("%d",&a[i].x);
			a[i].id=i+1;
		}	
		solve();
	}
}
