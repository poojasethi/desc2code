#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#include <string>

using namespace std;

int n;
int len[1010], id[1010];
string rec[1010];

bool cmp ( int a, int b ) {
	return len[a] < len[b];
}

char ch[1010];
int l;

int main ()
{
	scanf("%d", &n);
	for(int i = 0; i < n; i ++ )
	{
		scanf("%d", len+i);
		
		id[i] = i;
	}
	sort( id, id+n, cmp );
	
	bool flag = 1;
	
	l = 0;
	for(int i = 0; i < len[id[0]]; i ++ ) 
		ch[l++] = '0';
	ch[l] = 0;
	rec[id[0]] = ch;
	
	for(int i = 1; i < n; i ++ )
	{
		while( l > 0 && ch[l-1] == '1' ) l --;
		if( l == 0 ) {
			flag = 0;
			break;
		}
		ch[l-1] = '1';
		for(int j = l; j < len[id[i]]; j ++ )
			ch[l++] = '0';
		ch[l] = 0;
		rec[id[i]] = ch;
	}
	
	if( !flag ) printf("NO\n"); else
	{
		printf("YES\n");
		for(int i = 0; i < n; i ++ ) printf("%s\n", rec[i].c_str());
	}
	
	
//	system("pause");
	return 0;
}
