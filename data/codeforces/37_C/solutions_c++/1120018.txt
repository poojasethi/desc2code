#include <iostream>
#include <algorithm>
#include <vector>

#define rep(i,n) for(int i=0;i<n;i++)
#define mp(a,b) make_pair(a,b)
#define pr(i) cout<<i<<endl
#define pi pair<int,int>

using namespace std;

int n,m,k,i,j,t1,t2;
vector<pi > c;
string s,r[1000];


void dfs(int v){

  if(c[k].first>v){
    rep(i,2){
      s+=i+'0';
      dfs(v+1);
      s.erase(s.size()-1,1);
    }
  }

  else r[c[k++].second]=s;

  if(k==n){
    pr("YES");
    rep(i,n)pr(r[i]);
    exit(0);
  }

}



int main(){

  cin>>n;
  rep(i,n){
    cin>>t1;
    c.push_back(pi(t1,i));
  }
  sort(c.begin(),c.end());

  dfs(0);

  pr("NO");

  return 0;

}
