#include <iostream>
#include <map>
#include <vector>

using namespace std;

int main(int argc, char* argv[])
{
	int n;
	cin >> n;
	
	multimap<int,int> s;
	
	for (int i = 0; i < n; ++i)
	{
		int a;
		cin >> a;
		s.insert(pair<int,int>(a, i));
	}
	
	vector<string> r(n);
	
	vector<string> p;
	p.push_back("");
	
	for (multimap<int,int>::iterator i = s.begin(); i != s.end(); ++i)
	{
		if (p.size() != 0)
		{
			string c = p.back();
			p.pop_back();
		
			while (c.size() != i->first)
			{
				p.push_back(c + '1');
				c += '0';
			}
		
			r[i->second] = c;
		}
		else
		{
			cout << "NO" << endl;
			return 0;
		}
	}
	
	cout << "YES" << endl;
	for (vector<string>::iterator i = r.begin(); i != r.end(); ++i)
	{
		cout << *i << endl;
	}
	
	return 0;
}
