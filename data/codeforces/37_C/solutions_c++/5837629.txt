#include<iostream>
#include<vector>
#include<string>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	int n;
	cin>>n;
	vector<int> a(n+1);
	for(int i=0;i<n;i++) cin>>a[i];
	vector<string> v(1001);
	int size=2;
	v[0]="0"; v[1]="1";
	vector<string> ans(n+1);
	for(int i=1;i<=1000;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(a[j]==i)
			{
				if(!size)
				{
					cout<<"NO"<<endl; 
					return 0;
				}
				ans[j]=v[--size];
			}
		}
		for(int j=0;j<size;j++)
		{
			if(size+j<1000) v[size+j]=v[j]+"1";
			v[j]+="0";
		}
		size=min(2*size,1000);
	}
	cout<<"YES"<<endl;
	for (int i=0;i<n;i++) cout<<ans[i]<<endl;
	return 0;
}
