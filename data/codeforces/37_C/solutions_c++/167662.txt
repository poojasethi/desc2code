#include <stdio.h>
#include <string>
#include <map>
#include <vector>

using namespace std;

string ss(const vector<int> &v, int n)
{
	string s;
	for (int i = 0; i < n; i++) {
		s += (v[i] == 0 ? "0" : "1");
	}
	return s;
}

bool inc(vector<int> &v, int n)
{
	for (int i = n - 1; i >= 0; i--) {
		if (v[i] == 0) {
			v[i] = 1;
			return true;
		}
		v[i] = 0;
	}
	return false;
}

int main()
{
	int N;
	scanf("%d", &N);
	multimap<int, int> m;
	for (int i = 0; i < N; i++) {
		int x;
		scanf("%d", &x);
		m.insert(make_pair(x, i));
	}
	vector<int> v;
	v.resize(1000);
	vector<string> r;
	r.resize(N);
	bool has_next = true;
	for (multimap<int, int>::iterator i = m.begin(); i != m.end(); ++i) {
		if (!has_next) {
			printf("NO\n");
			return 0;
		}
		r[i->second] = ss(v, i->first);
		if (!inc(v, i->first))
			has_next = false;
	}
	printf("YES\n");
	for (size_t i = 0; i < r.size(); i++)
		printf("%s\n", r[i].c_str());
}
