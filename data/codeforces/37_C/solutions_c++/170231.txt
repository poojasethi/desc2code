#include <stdio.h>
#include <queue>
#include <string>
#include <vector>
#include <iostream>
using namespace std;

#define max(a,b) (((a)<(b))?(b):(a))

char num[2000]={};
int len[2000]={};
int sum[2000]={};
vector<queue <string> > r(2000);
vector<int> lo;
int N;

void res(int l)
{
	if(N==0) return;
	if(len[l]>0)
	{
		len[l]--;
		N--;
		num[l]='\0';
		string rs(num);
		r[l].push(rs);
	}
	else
	{
		num[l]='0';
		res(l+1);
		num[l]='1';
		res(l+1);
	}
}


int main()
{
	int i;
	int t;
	int maxlen;
	maxlen=0;
	scanf("%d",&N);
	for(i=0;i<N;i++)
	{
		scanf("%d",&t);
		len[t]++;
		sum[t]++;
		lo.push_back(t);
		maxlen=max(maxlen,t);
	}
	for(i=maxlen;i>0;i--)
	{
		if(sum[i]>1)
		{
			sum[i-1]+=sum[i]/2;
			sum[i]%=2;
		}
	}
	if(sum[0]>0)
	{
		for(i=1;i<=maxlen;i++)
			if(sum[i]>0) break;
		if(i<=maxlen)
		{
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	res(0);
	for(i=0;i<lo.size();i++)
	{
		cout<<r[lo[i]].front()<<endl;
		r[lo[i]].pop();
	}
}
