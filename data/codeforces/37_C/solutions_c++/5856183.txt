#include <cstdio>
#include <iostream>
#include <string>
using namespace std;
int a[1005];
string b[1005];
int main ( ) {
	int n,i,j,k;
	string t="";
	scanf("%d",&n);
	for (i=0;i<n;i++) scanf("%d",&a[i]);
	for (i=1;i<=1000;i++) {
		for (j=0;j<n;j++) {
			if (a[j]==i) {
				if (t.length()==0) {
					t=string(i,'0');
				}
				else {
					k=t.length()-1;
					while (k>=0 && t[k]=='1') t[k--]='0';
					if (k<0) {printf("NO\n");return 0;}
					t[k]++;
					while (t.length()<i) t+='0';
				}
				b[j]=t;
			}
		}
	}
	printf("YES\n");
	for (i=0;i<n;i++) cout<<b[i]<<endl;
	return 0;
}
