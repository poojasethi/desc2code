#include <bits/stdc++.h>
using namespace std ;
string s = "";
pair<int ,int>a[1001] ;
int c = 0 ;
int n = 0 ;
string v[1009] ;
void dfs(int len){
    if(c == n){
        cout << "YES" <<endl ;
        for(int i = 0 ; i < n ; i++){
            cout << v[i] << endl ;
        }
        exit(0) ;
    }
    if(len == a[c].first){
        v[a[c++].second] = s ;
    }
    else {
        s+="0" ;
        dfs(len+1) ;
        s.erase(s.size()-1) ;
        s+="1";
        dfs(len+1) ;
        s.erase(s.size()-1) ;
    }
    if(c == n){
        cout << "YES" <<endl ;
        for(int i = 0 ; i < n ; i++){
            cout << v[i] << endl ;
        }
        exit(0) ;
    }

}

int main (){
    cin >> n ;
    for(int i = 0 ;i < n ;i++ ){
        scanf("%d",&a[i].first) ;
        a[i].second = i ;
    }
    sort(a,a+n) ;
    dfs(0) ;
    cout << "NO" ;
    return 0 ;
}
