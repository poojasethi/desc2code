#include<stdio.h>
#include<algorithm>
using namespace std;
#define N 1010

int n, a[N], t, id[N], pos, cur;
char s[N][N], wor[N];

bool cmp(int x, int y){
   return a[x] < a[y];
}

int main(){
   scanf("%d", &n);
   for(int i = 0; i < n; i ++){ scanf("%d", a+i); id[i] = i;}
   sort(id, id + n, cmp);
   cur = 1;
   for(int i = 0; i < n; i ++){
      t = id[i];
      cur = max(cur, a[t]);
      for(int j = 0; j < cur; j ++) s[t][j] = wor[j];
      pos = cur-1;
      while(wor[pos]){ wor[pos--] = 0; }
      if(pos < 0 and i < n - 1){puts("NO"); return 0; }
      wor[pos] = 1;
   }
   puts("YES");
   for(int i = 0; i < n; i ++){
      for(int j = 0; j < a[i]; j ++) printf("%d", s[i][j]); puts("");
   }
}
