#include<iostream>
#include<algorithm>
#include<utility>
#include<string>
using namespace std;
int n;
pair < int , int > len[1005];
string ans[1005];
int cnt=0;
string add(string str)
{
	int i=str.length()-1;
	while(i>=0 && str[i]=='1')
	{
		str[i]=0;
		i--;
	}
	if(i<0)return "full";
	str[i]='1';
	return str;
}
void dfs(string str,int st)
{
	if(st==n)return;
	if(str.length()<len[st].first)
	{
		dfs(str+"0",st);
	}
	else
	{
		cnt++;
		ans[len[st].second]=str;
		str=add(str);
		if(st<n-1 && str=="full")return;
		dfs(str,st+1);
	}
}
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>len[i].first;
		len[i].second=i;
	}
	sort(len,len+n);
	dfs("",0);
	if(cnt==n)
	{
		cout<<"YES"<<endl;
		for(int i=0;i<n;i++)cout<<ans[i]<<endl;
	}
	else
	{
		cout<<"NO"<<endl;
	}
	return 0;
}
