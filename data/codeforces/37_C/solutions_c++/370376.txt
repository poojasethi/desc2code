#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
  int n;
  cin>>n;
  vector<pair<int,int> > v(n);
  for (int i=0;i<n;i++) {
    v[i].second=i;
    cin>>v[i].first;
  }
  vector<int> sol(2000,0);
  int index=0;
  sort(v.begin(),v.end());
  bool error=false;
  vector<vector<int> > solucio(n);
  for (int i=0;i<n and not error;i++) {
    int len=v[i].first;
    int ind=v[i].second;
    while (index>=0 and sol[index]==1)
      index--;
    if (index==-1)
      error=true;
    else {
      sol[index]=1;
      for (int j=index+1;j<=len;j++)
	sol[j]=0;
      index=len;
      vector<int> &solact=solucio[ind];
      for (int j=1;j<=index;j++)
	solact.push_back(sol[j]);
    }
  }
  if (error)
    cout<<"NO"<<endl;
  else {
    cout<<"YES"<<endl;
    for (int i=0;i<int(solucio.size());i++) {
      vector<int> &solact=solucio[i];
      for (int j=0;j<int(solact.size());j++)
	cout<<solact[j];
      cout<<endl;
    }
  }
}
