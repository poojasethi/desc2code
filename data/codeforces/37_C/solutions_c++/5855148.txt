#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <utility>
#include <string>
using namespace std;
int n,m=0;
pair <int,int> l[3200];
string w[3200];
string make(string s){
	int i=s.size()-1;
	while (i>=0 && s[i]=='1'){
		s[i]='0';
		i--;
	}
	if (i<0) return "";
	s[i]='1';
	return s;
}
void back(string v,int p){
	if (p==n) return ;
	if (v.size()<l[p].first)
		back(v+"0",p);
	else{
		m++;
		w[l[p].second]=v;
		v=make(v);
		if (p<n-1 && v=="") return ;
		back(v,p+1);
	}
}
int main( ){
	cin>>n;
	for (int i=0;i<n;i++){
		cin>>l[i].first;
		l[i].second=i;
	}
	sort(l,l+n);
	back("",0);
	if (m==n){
		cout<<"YES"<<endl;
		for (int i=0;i<n;i++)
			cout<<w[i]<<endl;
	}else cout<<"NO"<<endl;
	return 0;
}
