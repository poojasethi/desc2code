#include<iostream>
#include<algorithm>
using namespace std;

int n;
pair<int,int> len[1010];
string ans[1010];
int numfill;

void input()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>len[i].first;
		len[i].second=i;
	}
	sort(len,len+n);
}
string next(string s)
{
	int i=s.size()-1;
	while (i>=0 && s[i]=='1')
	{
		s[i]='0';
		i--;
	}
	if (i<0) return "";
	s[i]='1';
	return s;
}
void dfs(string now,int k)
{
	if (k==n) return;
	if (now.size()<len[k].first)
	{
		dfs(now+'0',k);
	}
	else
	{
		numfill++;
		ans[len[k].second]=now;
		now=next(now);
		if (k<n-1 && now=="") return;
		dfs(now,k+1);
	}
}
void print()
{
	if (numfill==n)
	{
		cout<<"YES"<<endl;
		for(int i=0;i<n;i++)
		{
			cout<<ans[i]<<endl;
		}
	}
	else
	{
		cout<<"NO"<<endl;
	}
}
int main()
{
	input();
	dfs("",0);
	print();
	return 0;
}
