#include <bits/stdc++.h>

using namespace std;
const int maxn=1005;
int n;
struct ijk {
    int len,vt;
};
ijk a[maxn];
string res[maxn];
bool cmp(ijk c, ijk d) {
    if (c.len!=d.len) return c.len<d.len;
    else return c.vt<d.vt;
}

void biendoi(string &s) {
    for (int i=s.size()-1; i>=0; i--)
        if (s[i]=='0') {
            s[i]='1';
            return;
        }
        else s[i]='0';
    s="";
}

int main()
{
    //freopen("language.inp","r",stdin);
    //freopen("language.out","w",stdout);
    cin>>n;
    for (int i=0; i<n; i++) {
        cin>>a[i].len;
        a[i].vt=i;
    }
    sort(a,a+n,cmp);
    string s="0";
    for (int i=0; i<n; i++) {
        if (s.empty()) {cout<<"NO"; return 0;}
        while (s.size()<a[i].len) s+='0';
        res[a[i].vt]=s;
        biendoi(s);
    }
    cout<<"YES"<<endl;
    for (int i=0; i<n; i++) cout<<res[i]<<endl;
    return 0;
}
