#include <iostream>
#include <vector>
#include <cstdlib>
using namespace std;

const int MAXN = 1000 + 10;
int cnt[MAXN];
vector<int> ln;
vector<string> num[MAXN];
bool bit[MAXN];
bool end;

inline void inc(int r)
{
	if(r == -1)
	{
		end = true;
		return ;
	}
	if(bit[r] == 0)
		bit[r] = 1;
	else
		bit[r] = 0, inc(r-1);
}

int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i=0; i<n; i++)
	{
		int d;
		cin >> d;
		ln.push_back(d);
		cnt[d]++;
	}
	for(int i=MAXN-1; i>=1; i--)
		if(cnt[i])
		{
			bool flg = false;
			for(int j=i; j<MAXN && !flg; j++)
				if(bit[j])
					flg = true;
			if(flg)
				inc(i-1);
			while(cnt[i]--)
			{
				if(end)
				{
					cout << "NO\n";
					return 0;
				}
				string s;
				for(int j=0; j<i; j++)
					s += (bit[j] + '0');
				num[i].push_back(s);
				inc(i-1);
			}
			for(int j=i; j<MAXN; j++)
				bit[j] = 0;
		}
	cout << "YES\n";
	for(int i=0; i<n; i++)
		cout << num[ln[i]][++cnt[ln[i]]] << "\n";
	return 0;
}
