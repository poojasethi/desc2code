#include <iostream>
#include <map>
#include <algorithm>
#include <string>
using namespace std;

typedef pair<int, int> pii;

const int MX = 1000;

pii arr[MX];
int pos = 0, n;
string cur, ans[MX];
void dfs(int d) {
	if (pos == n) return;
	if (arr[pos].first == d) {
		ans[arr[pos].second] = cur;
		++pos;
	} else {
		cur += '0';
		dfs(d + 1);
		*cur.rbegin() = '1';
		dfs(d + 1);
		cur.erase(cur.size() - 1);
	}
}
int main() {
	cin >> n;
	for (int i = 0; i < n; ++i) {
		int a;
		cin >> a;
		arr[i].first = a;
		arr[i].second = i;
	}
	sort(arr, arr + n);
	dfs(0);
	if (pos != n) {
		cout << "NO";
	} else {
		cout << "YES\n";
		for (int i = 0; i < n; ++i) {
			cout << ans[i] << '\n';
		}
	}
}
