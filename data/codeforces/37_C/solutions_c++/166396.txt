#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

#define PB push_back
#define MK make_pairr

const int maxn = 1100;
int n, s[maxn], a[maxn], out[maxn];
vector<string> v[maxn];

int main(){
	cin >> n;
	for(int i = 0; i < n ; i++){
		cin >> s[i];
		a[s[i]]++;
	}
	for(int i = 1010; i > 0; i--){
		a[i] += a[i]%2;
		a[i-1] += a[i]/2;
	}
	if(a[1] > 2){
		cout << "NO" << endl;
		return 0;
	}
	v[1].PB("0");
	v[1].PB("1");
	for(int i = 1; i < maxn; i++)
		for(int j = 0; j < (a[i+1]/2); j++){
			v[i+1].PB(v[i][ v[i].size()-1-j ] + "0");
			v[i+1].PB(v[i][ v[i].size()-1-j ] + "1");
		}
	cout << "YES" << endl;
	for(int i = 0; i < n; i++)
		cout << v[s[i]][ out[s[i]]++ ] << endl;
	return 0;
}
