#include<bits/stdc++.h>
using namespace std;
int con[2000]={0};
int inp[2000];
int las[2000]={0};
vector<string> ans[1003];
int n,tot= 0;
void getans(string s)
{
    int sz = s.size();
    if(sz>1000||tot==n)return;
    if(con[sz]>0){ans[sz].push_back(s);con[sz]--;tot++;return;}
    string ss = s+'0';getans(ss);
    ss= s+'1';return getans(ss);
}
int main()
{
    cin >> n;
    for(int i = 0;i<n;i++)
        cin >> inp[i],con[inp[i]]++;
    int last = 1;
    for(int i =1;i<=1000;i++)
    {
        if(con[i]>last*2){puts("NO");return 0;}
        last=last*2;last-=con[i];last = min(last,1001);
    }
    string test= "0";
    getans(test);
    test = "1";
    getans(test);
    for(int i = 1;i<=1000;i++)
    {
        las[i] = ans[i].size()-1;
    }
    puts("YES");
    for(int i = 0;i<n;i++)
    {
        cout << ans[inp[i]][las[inp[i]]]<<endl;
        las[inp[i]]--;
    }
}
