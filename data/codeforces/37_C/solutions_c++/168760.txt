#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
int n,a[1001],b[1000],c[1000],len,flag;
string st[1000],now;
int main()
{
    int i,j;
    cin >> n;
    for (i = 0;i < n;++i) {cin >> b[i];++a[b[i]];}
    for (i = 0;i < 1000;++i) {a[i + 1] += a[i];}
    for (i = 0;i < n;++i) c[i] = a[b[i]]--;
    sort(b,b + n);
    now = "";
    len = 0;
    for (i = 0;i < n;++i)
    {
        while (len < b[i]) {now.push_back('0');++len;}
        st[i] = now;
        ++now[len - 1];
        for (j = max(len - 2,0);j >= 0;--j)
        {
            if (now[j + 1] == '2')
            {
                ++now[j];
                now[j + 1] = '0';
            }
        }
        if (now[0] == '2' && i != n - 1) {flag = 1;break;}
    }
    if (flag) cout << "NO" << endl; else
    {
        cout << "YES" << endl;
        for (i = 0;i < n;++i) cout << st[c[i] - 1] << endl;
    }
    return 0;
}
