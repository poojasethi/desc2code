#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int n;
int L[1005];
int srt[1005];
char s[1005];
char ans[1005][1005];

bool cmp(int x, int y) {
	return L[x] < L[y];
}

int main(void) {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d", &L[i]);
		srt[i] = i;
	}
	sort(srt, srt+n, cmp);
	for(int i=0;i<L[srt[n-1]];i++)
		s[i] = '0';
	int flag=0, bad=0;
	for(int i=0;i<n;i++) {
		if(flag) {bad=1;break;}
		int len = L[srt[i]];
		for(int j=0;j<len;j++)
			ans[srt[i]][j] = s[j];
		s[len] = '\0';
		--len;
		while(len>=0 && s[len]=='1') s[len--] = '0';
		if(len<0) flag=1;
		else s[len]='1';
		if(i+1<n) for(int j=L[srt[i]]; j<L[srt[i+1]];j++) s[j] = '0';
	}
	if(bad) puts("NO");
	else {
		puts("YES");
		for(int i=0;i<n;i++)
			printf("%s\n", ans[i]);
	}
	return 0;
}
