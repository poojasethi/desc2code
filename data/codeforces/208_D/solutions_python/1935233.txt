n = int(raw_input());
a = map(int, raw_input().split());
b = map(int, raw_input().split());
left, c = 0, [0, 0, 0, 0, 0];

for i in range(n):
    left += a[i];
    for j in range(4, -1, -1):
        c[j] += (left / b[j]);
        left %= b[j];

ans = "";
for i in c:
    ans += " " + str(i);
    
print ans[1:];
print left;
