n = input()
p = map(int, raw_input().split())
x = map(int, reversed(raw_input().split()))
tot, n = 0, [0] * 5
for c in p:
	tot += c
	while tot >= x[4]:
		for i in xrange(5):
			if tot >= x[i]:
				num = tot / x[i]
				n[i] += num
				tot -= num * x[i]
				break
print n[4], n[3], n[2], n[1], n[0]
print tot
