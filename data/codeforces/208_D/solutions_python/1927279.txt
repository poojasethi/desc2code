#!/usr/bin/env python

n = int(raw_input())
p = [int(i) for i in raw_input().split()]
c = [int(i) for i in raw_input().split()]
result = [0, 0, 0, 0, 0]

point = 0
for i in range(0, n):
    point += p[i]
    for j in range(4, -1, -1):
        result[j] += point / c[j]
        point %= c[j]
print ' '.join([str(i) for i in result])
print point
