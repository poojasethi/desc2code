import sys

answer = [0,0,0,0,0]
n = map(int,raw_input().split())[0]
p = map(int,raw_input().split())
balance = 0
prices = map(int,raw_input().split())[::-1]
#print prices
for i in xrange(n):
    balance += p[i]
    for j in xrange(5):
        if (balance>=prices[j]):
            answer[j]+= balance/prices[j]
            balance-=prices[j]*(balance/prices[j])
print " ".join(map(str,answer[::-1]))
