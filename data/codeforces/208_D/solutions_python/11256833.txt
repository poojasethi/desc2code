n = int(raw_input())
p = [int(v) for v in raw_input().split()]
cost = [int(v) for v in raw_input().split()]

num = [0] * 5
curr = 0

for c in p:
	curr += c
	while curr >= cost[0]:
		for i in xrange(4, -1, -1):
			if curr >= cost[i]:
				n, rem = divmod(curr, cost[i])
				num[i] += n
				curr = rem
				break

print ' '.join(str(v) for v in num)
print curr
