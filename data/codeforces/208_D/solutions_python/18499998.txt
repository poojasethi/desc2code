
n = int(raw_input())
points = map(int, raw_input().split())
prizes = map(int, raw_input().split())

out = [0] * len(prizes)
score = 0
for p in points:
  score += p
  for i in xrange(len(prizes) - 1, -1, -1):
    if score >= prizes[i]:
      win_count = score / prizes[i]
      out[i] += win_count
      score -= win_count * prizes[i]
print ' '.join(map(str, out))
print score
