#! /usr/bin/python2

import sys
import math

n = int(raw_input())
p = map(int, raw_input().split())
c = map(int, raw_input().split())
cur = 0
ans = [0] * 5
for i in p:
    cur += i
    for j in list(reversed(range(5))):
        ans[j] += cur / c[j]
        cur %= c[j]
_ = ""
for i in ans:
    _ += str(i) + " "
print(_)
print(cur)
