n = input()
p = map(int, raw_input().split())
a, b, c, d, e = map(int, raw_input().split())
s = 0
sa = sb = sc = sd = se = 0
for pi in p:
    s += pi
    se += s / e
    s %= e
    sd += s / d
    s %= d
    sc += s / c
    s %= c
    sb += s / b
    s %= b
    sa += s / a
    s %= a


print '%d %d %d %d %d' % (sa, sb, sc, sd, se)
