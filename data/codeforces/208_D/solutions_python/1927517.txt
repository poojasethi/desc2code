n=input()
p=map(int,raw_input().split())
m=map(int,raw_input().split())
res=[0,0,0,0,0]
b=0
for point in p:
  b+=point
  can=True
  while can:
    can=False
    for i in xrange(5):
      if i<4 and m[i+1]>b>=m[i] or i==4 and b>=m[i]:
        c=b/m[i]
        res[i]+=c
        b-=c*m[i]
        can=True
        break
print ' '.join(map(str,res)),'\n',b
