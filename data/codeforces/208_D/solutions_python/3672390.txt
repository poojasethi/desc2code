n = int(raw_input())

p = map(int, raw_input().split())

a = map(int, raw_input().split())

ans = [0 for i in a]

s = 0

for x in p:
    s += x
    for i in xrange(len(a) - 1, 0 - 1, -1):
        if s >= a[i]:
            ans[i] += s / a[i]
            s %= a[i]
            
print ' '.join(map(str, ans))

print s
