#!/usr/bin/env python

n = int(raw_input().strip())
a = map(int, raw_input().strip().split())
b = map(int, raw_input().strip().split())
b.sort(reverse=True)
res = [0]*5

m = 0
for r in a:
    m+=r
    for i,p in enumerate(b):
        count = m/p
        res[i]+=count
        m = m % p


print ' '.join(map(str,res[::-1])).strip()
print m
