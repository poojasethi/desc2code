import sys

num = [0,0,0,0,0]
ans = 0
N = int(sys.stdin.readline())
c = [int(L) for L in sys.stdin.readline().split()]
p = [int(L) for L in sys.stdin.readline().split()]

for i in xrange(N):
    ans += c[i]
    for j in xrange(4, -1, -1):
        cnt = ans / p[j]
        num[j] += cnt
        ans -= cnt * p[j]
