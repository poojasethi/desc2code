#include<iostream>
using namespace std;
int main(){
long long int n,arr[5],x,i,j,k,p[5]={0},t;
cin>>n;
long long int ct[n];
for(i=0;i<n;i++){
cin>>ct[i];
}
for(i=0;i<5;i++){
cin>>arr[i];
}
x=0;
for(i=0;i<n;i++){
x=x+ct[i];
for(j=4;j>=0;j--){
if(arr[j]<=x){
//cout<<x<<" "<<j<<endl;
t=x/arr[j];
x=x-t*arr[j];
p[j]=p[j]+t;
}
}
}
for(i=0;i<5;i++){
cout<<p[i]<<" ";
}
cout<<endl;
cout<<x<<endl;
}
