#include<iostream>
using namespace std;
typedef long long int ll;
ll n,arr[5],p[100],i,j,temp,ans=0,cnt[6];
int main()
{
	cin>>n;
	for(i=0;i<n;i++)
		cin>>p[i];
	for(i=0;i<5;i++)
		cin>>arr[i];
	for(i=0;i<n;i++)
	{
		ans+=p[i];
		for(j=4;j>=0;j--)
		{
			if(ans>=arr[j])
			{
				cnt[j]+=ans/arr[j];
				ans%=arr[j];
			}
		}
	}
	for(i=0;i<5;i++)
		cout<<cnt[i]<<" ";
	cout<<endl;
	cout<<ans<<endl;
}
