#include <iostream>

using namespace std;

int main() {
  int n,i,j; cin>>n;
  long *p=new long[n];
  long long pr[5],c[5]={0,0,0,0,0},s=0;
  for(i=0;i<n;i++) cin>>p[i];
  for(i=0;i<5;i++) cin>>pr[i];
  for(i=0;i<n;i++) {
    s+=p[i];
    for(j=4;j>=0;j--) {c[j]+=s/pr[j]; s%=pr[j];}
  }
  for(i=0;i<5;i++) cout<<c[i]<<" "; cout<<endl<<s;
  return 0;
}
