#include <stdio.h>
int i, j, n, c[5], p[50];
long long pp, g[5];

int main()
{
    scanf("%d", &n);
    for (i=0; i<n; ++i) scanf("%d", &p[i]);
    for (i=0; i<5; ++i) scanf("%d", &c[i]);
    for (i=0; i<n; ++i) {
        pp+=p[i];
        for (j=4; j>=0; --j) {
            g[j]+=pp/c[j];
            pp%=c[j];
        }
    }
    for (i=0; i<5; ++i) printf("%I64d ", g[i]); printf("\n");
    printf("%I64d\n", pp);
}
