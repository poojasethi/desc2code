#include <algorithm>
#include <iostream>
using namespace std;
int N, P[60], A[5];
int main() {
    while (cin >> N) {
	int S=0; long long C[5]={};
	for (int i=0; i<N; ++i) cin >> P[i];
	for (int i=0; i<5; ++i) cin >> A[i];
	for (int i=0; i<N; ++i) {
	    S += P[i];
	    for (int j=4; j>=0; --j) if (S>=A[j]) C[j]+=S/A[j], S%=A[j];
	}
	for (int i=0; i<5; ++i) cout << C[i] << (i==4?'\n':' ');
	cout << S << endl;
    }
}
