#define f(i, n) for (int i = 0; i < (int)(n); i++)
typedef long long ll;
ll a[5],c[5];
int d[100005];

int main()
{
    ll n,rem=0;
    cin>>n;
    f(i,n)cin>>d[i];
    f(i,5)cin>>a[i];

    f(i,n){
    rem+=d[i];
    for(int i=4; i>-1; i--)
      c[i]+=rem/a[i],rem%=a[i];
    }
    f(i,5)cout<<c[i]<<" ";

}
