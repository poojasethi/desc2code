#include<cstdio>
using namespace std;
long long p[1005],a[105],k,n,i,j,ans[105];
int main()
{
  scanf("%I64d",&n);
  for (i=1;i<=n;i++)
    scanf("%I64d",&p[i]);
  scanf("%I64d%I64d%I64d%I64d%I64d",&a[1],&a[2],&a[3],&a[4],&a[5]);
  for (i=1;i<=n;i++)
  {
    k+=p[i];
    for (j=5;j>=1;j--)
      if (k>=a[j]) ans[j]+=k/a[j],k%=a[j];
  }
  for (i=1;i<=5;i++)
    printf("%I64d ",ans[i]);
  printf("\n%I64d",k);
}
