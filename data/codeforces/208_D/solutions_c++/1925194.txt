#include <stdio.h>
main(){
  int n,p[99],c[9];
  long long s=0,k[9]={0};
  scanf("%d",&n);
  for(int i=0; i<n; i++)
    scanf("%d",&p[i]);
  for(int i=0; i<5; i++)
    scanf("%d",&c[i]);
  for(int i=0; i<n; i++) {
    s+=p[i];
    for(int j=4; j+1; j--)
      k[j]+=s/c[j],
      s%=c[j];
  }
  for(int i=0; i<5; i++)
    printf("%I64d ",k[i]);
  printf("\n%I64d",s);
  return 0;
}
