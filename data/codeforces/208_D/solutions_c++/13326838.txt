#include <iostream>
#include <cmath>
#include <vector>
#include <algorithm>

using namespace std;
long long int n,a[55],b[6],ans[6],sum;
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i];
	for(int i=1;i<=5;i++)cin>>b[i];	
	for(int i=1;i<=n;i++){
		sum+=a[i];
		for(int j=5;j>=1;j--){
			ans[j]+=sum/b[j];
			sum%=b[j];
		}
	}
	for(int i=1;i<=5;i++)cout<<ans[i]<<" ";
	cout<<endl<<sum;

	return 0;
}
