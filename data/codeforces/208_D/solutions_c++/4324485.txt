#include <iostream>

using namespace std;
long long x[55],c[5],cnt[5],sum,n,i,j;
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)cin>>x[i];
    for(j=0;j<5;j++)cin>>c[j];
    for(i=1;i<=n;i++)
    {
        sum+=x[i];
        for(j=4;j>=0;j--)
        {
            cnt[j]+=sum/c[j];
            sum%=c[j];
        }
    }
    for(i=0;i<5;i++)cout<<cnt[i]<<' ';
    cout<<'\n'<<sum;
    return 0;
}
