#include<iostream>
using namespace std;
int main()
{
  __int64 a[5],p[55],n,i,j,l[5]={0},s=0;
  cin>>n;
  for(i=0;i<n;i++) cin>>p[i];
  for(i=0;i<5;i++) cin>>a[i];
  for(i=0;i<n;i++)
  {
    j=4;
    p[i]+=s; 
    while(j>=0)
    {
      if (p[i]>=a[j]) {l[j]+=p[i]/a[j]; p[i]%=a[j];}
      j--;
    }
    s=p[i];
  }
  for(i=0;i<5;i++) cout<<l[i]<<" "; cout<<endl<<s;
//  system("pause");
  return 0;
}
