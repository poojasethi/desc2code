#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
	int n, p[55], a[55];
	ll s=0, ans[55]={0};
	scanf("%d", &n);
	for(int i=0; i<n; i++) scanf("%d", &p[i]);
	for(int i=0; i<5; i++) scanf("%d", &a[i]);
	for(int i=0; i<n; i++) {
		s+=p[i];
		for(int j=4; j>=0; j--) {
			ans[j]+=s/a[j];
			s-=1ll*(s/a[j])*a[j];
		}
	}
	for(int i=0; i<5; i++) printf("%I64d ", ans[i]);
	printf("\n%I64d\n", s);
	return 0;
}
