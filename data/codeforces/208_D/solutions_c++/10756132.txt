#include<iostream>
#include<algorithm>
using namespace std;
#define LL long long int
LL n,A[5],a[51],ans[5]={0},ans1=0;
void func(){
	int k;
for(k=4;k>=0 && A[k]>ans1;k--);
if(k<0)return ;
ans[k]+=ans1/A[k];
ans1%=A[k];
func();
}
int main()
{
	int n,i;
	cin>>n;
	for(i=0;i<n;i++)cin>>a[i];
	for(i=0;i<5;i++)cin>>A[i];
	for(i=0;i<n;i++){ans1+=a[i];func();}
	for(i=0;i<5;i++)cout<<ans[i]<<" ";cout<<endl;
	cout<<ans1<<endl;
}
