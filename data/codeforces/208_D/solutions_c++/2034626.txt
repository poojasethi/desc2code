#include <iostream>

using namespace std;
typedef long long ll;

const int MAXN=50+5;

int p[MAXN], a[5], n;
ll b[5], s;

int main() {
	cin>>n;
	for (int i=0; i<n; ++i)
		cin>>p[i];
	for (int i=0; i<5; ++i)
		cin>>a[i];

	for (int i=0; i<n; ++i) {
		s+=p[i];
		for (int i=4; i>=0; --i) {
			b[i]+=s/a[i];
			s%=a[i];
		}
	}
	for (int i=0; i<5; ++i)
		cout<<b[i]<<' ';
	cout<<endl<<s<<endl;
	return 0;
}
