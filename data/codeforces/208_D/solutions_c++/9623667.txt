#include<bits/stdc++.h>
using namespace std;
long long a[10],k[10],x[100],u[100],total,po,i;
int main()
{
	cin>>po;
	for(i=0;i<po;i++){
		cin>>u[i];
	}
	for(i=0;i<5;i++)
	{
		cin>>x[i];
	}
	total=0;
	for(i=0;i<po;i++)
	{
		total+=u[i];
		for(int j=4;j>=0;j--)
		{
			if(total/x[j]>0){
				k[j]+=total/x[j];
				total%=x[j];
			}
		}
	}
	for(i=0;i<5;i++)
	{
		cout<<k[i]<<" ";
	}
	cout<<endl;
	cout<<total;
}
