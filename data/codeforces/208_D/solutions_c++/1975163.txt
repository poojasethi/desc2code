# include <iostream>

using namespace std;

long long a;
long long b[99];
long long c[99];
long long d[99];

int main()
{
	cin>>a;
	
	for(int i=0;i<a;i++)
	cin>>b[i];
	
	for(int i=0;i<5;i++)
	cin>>c[i];
	
	for(int j=0;j<a;j++)
	{
		for(int i=4;i>=0;i--)
		{
			d[i]=d[i]+b[j]/c[i];
			b[j]=b[j]%c[i];
		}
		
		b[j+1]+=b[j];
	}
	
	for(int i=0;i<4;i++)
	cout<<d[i]<<' ';
	
	cout<<d[4]<<endl<<b[a];
}
