#include<iostream>
using namespace std;
main()
{
	long long a[55],x[10],y[10]={0},e=0;
	int n,i,j;
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>a[i];
	
	for(i=1;i<=5;i++)
		cin>>x[i];
	
	for(i=1;i<=n;i++) {
		e+=a[i];
		for(j=5;j>=1;j--) {
			y[j]+=e/x[j];
			e%=x[j];
			}
		}
	for(i=1;i<=5;i++)
		cout<<y[i]<<" ";
	cout<<endl<<e;
}
