#include <bits/stdc++.h>
using namespace std;
int main(){
	long long int n,a[55]={0},b[10]={0},c[10]={0},k=0;
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<5;i++)cin>>b[i];
	for(int i=0;i<n;i++){
		a[i]+=k;
		for(int j=4;j>=0;j--){
			c[j]+=a[i]/b[j];
			a[i]=a[i]%b[j];
		}
		k=a[i];
	}
	for(int i=0;i<5;i++)cout<<c[i]<<" ";
	cout<<endl<<k<<endl;
}
