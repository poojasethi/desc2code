#include<iostream>
using namespace std;
int main()
{int i,j,k,n;
 long long int p=0,a[50],b[6]={0},c[6]={0};
 cin>>n;
 for(i=0;i<n;i++)
   cin>>a[i];
 for(i=1;i<6;i++)
   cin>>b[i];
 for(i=0;i<n;i++)
   { p+=a[i];
     for(j=5;j>0;j--)
      if(p>=b[j])
       {  c[j]+=(p/b[j]);
           p%=b[j];
       }
    }
 for(i=1;i<6;i++)
   cout<<c[i]<<" ";
  cout<<"\n"<<p;
  return 0;
}
