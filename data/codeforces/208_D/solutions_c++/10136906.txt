#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n,j;
	scanf("%d",&n);
	long long a[n],b[5],c[5]={0},x=0;
	for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
	for(i=0;i<5;i++)
		scanf("%lld",&b[i]);
	for(i=0;i<n;i++)
	{
		x+=a[i];
		//cout<<x<<endl;
		for(j=4;j>=0;j--)
		{
			if(x>=b[j])
			{
				c[j]+=x/b[j];
				x=x%b[j];
			}
		}
		//cout<<x<<endl;
	}
	for(i=0;i<5;i++)		
		printf("%lld ",c[i]);
	printf("\n%lld\n",x);	
	return 0;
}
