#include<bits/stdc++.h>
using namespace std;
long long a[500],b[6],c[6];
int main()
{
long long n,i,j,x;

cin>>n;
for(i=1;i<=n;i++)
cin>>a[i];
for(i=1;i<=5;i++)
cin>>b[i];
x=0;
for(i=1;i<=n;i++)
{
    x+=a[i];
    for(j=5;j>=1;j--)
    {

        c[j]+=x/b[j];
        //cout<<j<<" = "<<c[j]<<endl;
        x%=b[j];
    }
    //cout<<endl;
}

for(i=1;i<=5;i++)
{
  cout<<c[i];
  if(i!=5)
    cout<<" ";
}
cout<<endl<<x;

}
