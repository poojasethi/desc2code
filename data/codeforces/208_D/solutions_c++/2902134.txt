#include<cstdio>
int n,s[55];
long long t[5],ans[5],rem;
int main(){
rem=0;
for(int i=0;i<5;i++) ans[i]=0;
scanf("%d",&n);
for(int i=0;i<n;i++)scanf("%d",&s[i]);
for(int i=0;i<5;i++)scanf("%I64d",&t[i]);
for(int i=0;i<n;i++){
rem+=(long long)s[i];
for(int j=4;j>=0;j--){
if(t[j]<=rem){
ans[j]+=rem/t[j];
rem%=t[j];
}
}
}
for(int i=0;i<5;i++) printf("%I64d%c",ans[i],i==4?'\n':' ');
printf("%I64d\n",rem);
return 0;
}
