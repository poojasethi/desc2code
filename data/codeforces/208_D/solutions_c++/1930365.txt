#include<cstdio>
long long p[5],q[10000],s[5];
int main(){
	long long n,i,sum=0,j;
	scanf("%I64d",&n);
	for(i=0;i<n;i++)
		scanf("%I64d",q+i);
	for(i=0;i<5;i++)
		scanf("%I64d",s+i);
	for(i=0;i<n;i++){
		sum+=q[i];
		for(j=4;j>=0;j--){
			p[j]+=sum/s[j];
			sum%=s[j];
		}
	}
	for(i=0;i<5;i++)
		printf("%I64d ",p[i]);
	printf("\n%I64d\n",sum);
}
