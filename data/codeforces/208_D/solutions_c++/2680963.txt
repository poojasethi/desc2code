#include <iostream>
#include <cmath>
#include <string>
using namespace std;
long long i,p[111],n,k[5],a[5],s,j;
int main()
{
	cin>>n;
	for(i=0;i<n;i++)
		cin>>p[i];
	for(i=0;i<5;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
	{
		s+=p[i];
		for(j=4;j>=0;j--)
		{
			k[j]+=s/a[j];
			s%=a[j];
		}
	}
	for(i=0;i<5;i++)
		cout<<k[i]<<" ";
	cout<<endl<<s<<endl;
	return 0;
}
