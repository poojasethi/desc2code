#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
typedef long long LL;
LL n,p,c[5],b[5],v[55],ans=0;
int main()
{
	cin>>n;
	for (int i=0;i<n;i++)cin>>v[i];
	for (int i=0;i<5;i++)cin>>b[i];
	for (int i=0;i<n;i++)
	{
		ans+=v[i];
		for (int i=4;i>=0;i--)
		{
			LL tmp=ans/b[i];
			c[i]+=tmp;
			ans-=tmp*b[i];
		}
	}
	for (int i=0;i<4;i++)cout<<c[i]<<' ';
	cout<<c[4]<<endl<<ans;
	return 0;
}
