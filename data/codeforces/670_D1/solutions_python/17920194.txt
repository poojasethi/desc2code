n, k = map(int,raw_input().split())
alist = map(int,raw_input().split())
blist = map(int,raw_input().split())

cookies = 0
while True:
    magic = 0
    for i in range(n):
        if blist[i] >= alist[i]:
            blist[i] = blist[i] - alist[i]
        else:
            magic += alist[i] - blist[i]
            blist[i] = 0
    if k >= magic:
        k = k - magic
        cookies += 1
    else:
        break

print cookies
