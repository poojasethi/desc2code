from heapq import heapreplace, heapify

n, k = map(int, raw_input().split())

a = map(int, raw_input().split())
b = map(int, raw_input().split())

h = [(b[i]/a[i], -a[i]+(b[i]%a[i]), i) for i in xrange(n)]
heapify(h)

while k > 0:
    cnt, rest, ind = h[0]
    if rest == 0:
        rest = a[ind]
    else:
        rest = -rest
    if rest > k:
        break
    k -= rest
    heapreplace(h, (cnt+1, 0, ind))
print h[0][0]
