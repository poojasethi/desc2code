#include <bits/stdc++.h>

using namespace std;

long long a[1000020];
long long b[1000020];
long long l,r,m,ans,now,k,i,n;

int main()
 {
  cin>>n>>k;
  for (i=1;i<=n;i++)
   scanf("%d",&a[i]);
  for (i=1;i<=n;i++)
   scanf("%d",&b[i]);

  l=1; r=6000000000;
  while (l <= r)
   {
    m=(l+r)/2;
    now=0;
    for (i=1;i<=n;i++)
      now+=max(0LL,1LL*a[i]*m-b[i]);
    if (now <= k)
     {
      ans=m;
      l=m+1;
     } else
     r=m-1;
   }
  cout<<ans<<endl;
 }
