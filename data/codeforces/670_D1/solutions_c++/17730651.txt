#include <bits/stdc++.h>
using namespace std;
int n,k;
int a[100009],b[100009];
int main(){
    cin>>n>>k;
    for(int i=0;i<n;i++)cin>>a[i];
    for(int i=0;i<n;i++)cin>>b[i];
    int num = 0;
    while(k>0){
        for(int i=0;i<n;i++){
            if(k < a[i]-b[i]){
                goto End;
            }
            else{
                k -= max(0,a[i]-b[i]);
                b[i] = max(0,b[i]-a[i]);
            }
        }
        num++;
    }
    End:;
    cout<<num<<endl;
}
