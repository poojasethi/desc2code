#include <stdio.h>
#include <iostream>
#define MAXN 1000+10
using namespace std;

int a[MAXN], b[MAXN];

int main(void)
{
	int n, k;
	while(cin >> n >> k)
	{
		int ans=0;
		for(int i=0; i<n; i++)
		cin >> a[i];
		for(int i=0; i<n; i++)
		cin >> b[i];
		while(k>0)
		{
			for(int i=0; i<n; i++)
			b[i]-=a[i];
			for(int i=0; i<n; i++)
			if(b[i]<0)
			{
				k+=b[i];
				b[i]=0;
			}
			if(k<0)  break;
			else
			ans++;
		}
		cout << ans << endl;
	}
	return 0;
}
