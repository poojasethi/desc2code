
#include <bits/stdc++.h>
using namespace std;
int main() {

int n,k;
cin>>n>>k;
vector<int> val(n+1),have(n+1);

for(int i=0;i<n;i++)
    cin>>val[i];


vector<pair<int,pair<int,int > > > vec;
for(int i=0;i<n;i++)
{
    cin>>have[i];
    vec.push_back(make_pair((have[i]/val[i]),make_pair(have[i],val[i])));

}
sort(vec.begin(),vec.end());
while(k>0)
{

   k--;
   vec[0].second.first++;
   vec[0].first=vec[0].second.first/vec[0].second.second;

   sort(vec.begin(),vec.end());

}
cout<<vec[0].first;
}
