#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;

__int64 a[1005],b[1005],c[2005],n,k;

int main()
{
	__int64 i,j,f;
	scanf("%I64d%I64d",&n,&k);
	for(i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	for(i=0;i<n;i++)
		scanf("%I64d",&b[i]);
	for(i=0;i<2005;i++){
		for(j=0;j<n;j++){
			if(b[j]/a[j]>=i)continue;
			c[i]+=i*a[j]-b[j];
		}
		if(c[i]>k){
			f=i-1;break;
		}
	}
	printf("%I64d\n",f);
	return 0;
}
