#include <bits/stdc++.h>
using namespace std;

int n, k;
int a[1005], b[1005];

int main() {
  scanf("%d%d", &n, &k);
  for (int i = 0; i<n; i++) scanf("%d", &a[i]);
  for (int i = 0; i<n; i++) scanf("%d", &b[i]);
  int maxi = 0;
  int r=0, l=3000;
  while (r!=l) {
    int m = (r+l+1)/2;
    int c = 0;
    for (int i = 0; i<n; i++) {
      c+=max(a[i]*m-b[i], 0);
    }
    if (c<=k) r=m;
    else l=m-1;
  }
  printf("%d\n", l);
}
