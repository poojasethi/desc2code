#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n, k;
	cin >> n >> k;
	int a[n + 1], b[n + 1];
	for(int i=1;i<=n;i++)
	cin >> a[i];
	int thg = 1e9;
	for(int i=1;i<=n;i++)
	{
		cin >> b[i];
		thg = min (thg, b[i] / a[i]);
	}
	for(;;)
	{
		int res = 0;
		++thg;
		for(int i=1;i<=n;i++)
			if(a[i] * thg > b[i])
				res += (a[i] * thg) - b[i], b[i] = a[i] * thg;
		if(res > k) 
		{
			--thg;
			break;
		} 
		k -= res;
	}
	cout << thg;
}
