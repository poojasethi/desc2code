#include<bits/stdc++.h>
using namespace std;
double A[1001],B[1001];
pair <double,int> pp[1001];
int main()
{
    cin.sync_with_stdio(false);
    int n,k,i;
    cin>>n>>k;
    for(i=0;i<n;i++)
        cin>>A[i];
    for(i=0;i<n;i++)
        cin>>B[i];
    for(i=0;i<n;i++)
        pp[i]=make_pair(B[i]/A[i],i);
    sort(pp,pp+n);
    while(k--)
    {
        int ind=pp[0].second;
        pp[0]=make_pair((++B[ind])/A[ind],ind);
        sort(pp,pp+n);
    }
    cout<<floor(pp[0].first);
    return 0;
}
