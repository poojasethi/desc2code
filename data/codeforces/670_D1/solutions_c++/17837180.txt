#include<bits/stdc++.h>
using namespace std;
int a[1001],b[1001];
int main()
{
    int n,k,val,i,c=0,flag=0; cin>>n>>k;
    for(int i=0;i<n;i++) cin>>a[i];
    for(int i=0;i<n;i++) cin>>b[i];
    while(1){
    for(i=0;i<n;i++)
    {
        if(a[i]<=b[i]){
            b[i]-=a[i];
        }else{
            val=a[i]-b[i];
            if(val>k){flag=1;break;}
            else k-=val;
            b[i]=0;
        }
    } if(flag) break; c++;
    }
    printf("%d",c);
    return 0;
}
