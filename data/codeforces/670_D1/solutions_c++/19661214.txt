#include<stdio.h>
int main()
{
	int count=0,n,k,a[1005],b[1005];
	while(scanf("%d %d",&n,&k)!=EOF)
	{
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	for(int i=0;i<n;i++)
	scanf("%d",&b[i]);
    //建立循环用b-a，不够就用k补上，如果补上也不够，就break；
	for(int i=0;i<n;i++)
	{
		b[i]-=a[i];
		if(b[i]<0)
		{
			k+=b[i];
			b[i]=0;
			if(k<0)break;
		}
		if(i==n-1) 
		{
			//printf("%d",b[i]);
			count ++;
			i=-1;
		}
	} printf("%d\n",count);count=0;
} }
