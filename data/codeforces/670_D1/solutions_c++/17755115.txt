#include <bits/stdc++.h>

using namespace std;
int n,k;
typedef long long ll;
ll a[100010],b[100010];

bool check(ll x)
{
    ll total = 0;
    for(int i=0;i<n;i++)
        total+=max(0ll,x*a[i]-b[i]);
    return total<=k;

}

int main()
{
    cin >> n >> k;
    for(int i=0;i<n;i++)
        cin >> a[i];
    for(int i=0;i<n;i++)
        cin >> b[i];

    ll l = 0,r = 5000000000;
    while(l<r)
    {
        int m = (l+r+1)>>1;
        if(check(m)) l = m;
        else r = m-1;
    }

    cout << l;


}
