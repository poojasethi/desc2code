#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

#define MAXN 100001
#define INF 2100000000

ll a[MAXN],b[MAXN];
int n;
ll k;

int check(int cnt){
	ll tmp=k;
	for(int i=1;i<=n;++i){
		if(a[i]<b[i]*cnt)
			tmp-=b[i]*cnt-a[i];
		if(tmp<0)break;
	}
	return tmp>=0;
}

int main(){
	scanf("%d%I64d",&n,&k);
	for(int i=1;i<=n;++i) scanf("%I64d",&b[i]);
	for(int i=1;i<=n;++i) scanf("%I64d",&a[i]);
	int l=0,r=INF;
	while(l!=r){
		int mid=((ll)l+r+1)>>1;
		if(check(mid)) l=mid;
		else r=mid-1;
	}
	printf("%d\n",l);
}
