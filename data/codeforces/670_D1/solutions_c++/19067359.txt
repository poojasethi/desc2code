#include <stdio.h>
int a[1200], b[1200];
int main()
{
	int n, k, i, l, j, sum;
	scanf("%d%d", &n, &k);
	for(i=1;i<=n;i++)
		scanf("%d", &a[i]);
	for(i=1;i<=n;i++)
		scanf("%d", &b[i]);
	for(l=1;;l++)
		{
		for(i=1;i<=n;i++)
		b[i] -= a[i];
		for(sum=0,j=1;j<=n;j++)
			if(b[j]<0)
			sum += b[j];
		if(sum+k<0)
				break;
		}
	printf("%d", l-1);
}
