#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(0);
   long long int n,m,i,j,k;
    cin>>n>>m;
    int a[n],b[n],c[n];
    for(i=0;i<n;i++)cin>>a[i];
    for(i=0;i<n;i++){cin>>b[i];
    c[i]=b[i]/a[i];}
    go:
    k=0;
   for(i=1;i<n;i++)
   {
       if(c[i]<c[k])
       {
           k=i;
       }
   }
   int x=a[k]*(c[k]+1)-b[k];
   if(m>=x)
   {
       b[k]+=x;
       c[k]++;
       m-=x;
       goto go;
   }
   sort(c,c+n);

cout<<c[0];

return 0;

    }
