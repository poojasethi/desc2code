#include<iostream>
using namespace std;
main()
{
int n,k; int ar[10002],br[10002];
cin>>n>>k;
for(int i=0;i<n;i++)
cin>>ar[i];
for(int i=0;i<n;i++)
cin>>br[i];
int count=0;
while(1)
{ int p=0;
for(int i=0;i<n;i++)
{
if(br[i]>=ar[i])
{ br[i]=br[i]-ar[i]; continue;}
int g=ar[i]-br[i];
if(k>=g)
{
k=k-g;
br[i]=0;
continue;
}
p=1;
}
if(p==1)
break;
else
count++;
}
cout<<count<<endl;
}
