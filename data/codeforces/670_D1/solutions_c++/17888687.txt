#include<stdio.h>
int main(){
	int n,a[1001],b[1001];
	int i,j,k,max=0,h;
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(i=1;i<=n;i++)
		scanf("%d",&b[i]);
	h=0;
	while(k>=h){
		max++;
		for(i=1;i<=n;i++){
			if(b[i]>=a[i])
				b[i]-=a[i];
			else{
				h+=a[i]-b[i];
				b[i]=0;
			}
		}
	}
	printf("%d\n",max-1);
	return 0;
}
