#include<iostream>
#include<fstream>
using namespace std;
int n,i;
long long x[100001],y[100001];
long long k,a,b,c;

bool check (long long c)
{long long s=0;

 for(int i=1;i<=n;i++)
  if(y[i]<x[i]*c) {
   s+=x[i]*c-y[i]; 
   if (s>k) return false;
  }
 
 return true;
}

main(){
 cin>>n>>k;
 for(i=1;i<=n;i++) 
  cin>>x[i];
 for(i=1;i<=n;i++) 
  cin>>y[i];
 
 a=0; b=2000000000;
 while(a<b) {            
  c=(a+b+1)>>1;
  if(check(c)) a=c;
          else b=c-1;
 }
 
 cout<<a<<endl;
 //system("pause");
}
