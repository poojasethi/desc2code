#include<stdio.h>

int a[1010], b[1010];

int main ()
{
	int n, m;
	int max=2000;
	scanf("%d %d", &n, &m);
	for(int i=0; i<n; i++)
		scanf("%d", &a[i]);
	for(int i=0; i<n; i++){
		scanf("%d", &b[i]);
		if(b[i]/a[i]<max)
			max=b[i]/a[i];
	}
	
	for(int i=0; i<n; i++)
		b[i]-=a[i]*max;
	
	while(m>0){
		bool flag=1;
		for(int i=0; i<n; i++){
			b[i]-=a[i];
			if(b[i]<0){
				m+=b[i];
				b[i]=0;
				if(m<0){
					flag=0;
					break;
				}
			}
		}
		max+=flag;
	}
		
	printf("%d\n", max);
	return 0;
}
