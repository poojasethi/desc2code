#include<bits/stdc++.h>
using namespace std;
  int cnt, n, k, q[1001], w[1001];
int main(){
    
	cin >> n >> k;
	for(int i = 1; i <= n; i++)
		scanf("%d", &q[i]);
	for(int i = 1; i <= n; i++)
		scanf("%d", &w[i]);
	while(true){
		for(int j = 1; j <= n; j++){
			bool ind = false;
			if(w[j] - q[j] < 0){
				k += (w[j] - q[j]);
				w[j] = 0;
				if(k < 0)
					ind = true;
			}
			else{
				w[j] -= q[j];
			}
			if(ind){
				cout << cnt;
				return 0;
			}
		}
	  cnt++;
	}

 return 0;
}
