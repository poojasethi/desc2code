
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,k;

cin>>n>>k;

vector <LL> v(n),u(n);

for(i=0;i<n;i++)
	cin>>v[i];

for(i=0;i<n;i++)
	cin>>u[i];

double f=1001;
LL id;

while(k>0)
{
f=1001;	
for(i=0;i<n;i++)
{
if(f>(double)u[i]/v[i])
{f=(double)u[i]/v[i];
id=i;
}

}
k--;
u[id]++;
}
k=M;
for(i=0;i<n;i++)
	k=min(k,u[i]/v[i]);

cout<<k<<endl;
}
