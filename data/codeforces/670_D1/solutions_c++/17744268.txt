#include<iostream>
#include<vector>
#include<string>
#define MX 200000
#define int long long
using namespace std;

int a[MX], b[MX], n, k;

main() {
  cin>>n>>k;
  for (int i=0;i<n;i++)
    cin>>a[i];
  for (int i=0;i<n;i++)
    cin>>b[i];
  
  int l = 0, r = 3000000000, m, x;
  while(l < r) {
    m = (l + r + 1) / 2;
    x = 0;
    for (int i=0;i<n;i++) {
      x += max(0ll, m * a[i] - b[i]);
      if (x > k)
	break;
    }
    if (x > k)
      r = m - 1;
    else
      l = m;
  }
  cout<<l<<endl;
  return 0;
}
