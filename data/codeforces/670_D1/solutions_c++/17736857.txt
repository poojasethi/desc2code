#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,k;int i;
cin>>n>>k;
int arr1[n];
int arr2[n];
for(i=0;i<n;i++)
{cin>>arr1[i];
}
for(i=0;i<n;i++)
{
cin>>arr2[i];
}
int result=0;int req;
int val;int flag=0;
while(k>0)
{
  //cout<<k<<"\n";
flag=0;req=0;
for(i=0;i<n;i++)
{
if (arr2[i]>=arr1[i])
{
arr2[i]=arr2[i]-arr1[i];
}
else{
flag=1;
req+=abs(arr2[i]-arr1[i]);
arr2[i]=0;
}
}
if (flag==0)
{
result++;
}
else{
if (k>=req)
{
result++;
k=k-req;
}
else
break;
}

}
cout<<result<<"\n";
return 0;
}//main
