#include<iostream>
using namespace std;
int a[1000],b[1000];main()
{
int n,k;
cin>>n>>k;
for(int i=0;i<n;i++)
cin>>a[i];
for(int i=0;i<n;i++)
cin>>b[i];
int t=10000;
for(int i=0;i<n;i++)
t=min(b[i]/a[i],t);
int s=t;
for(int i=0;i<n;i++)
b[i]-=t*a[i];
while(k>0)
{
int s1=0;
for(int i=0;i<n;i++)
if(b[i]<a[i]){
s1+=a[i]-b[i];b[i]=0;}
else
b[i]-=a[i];
if(s1<=k){
k-=s1;
s++;}
else
break;
}
cout<<s<<endl;
//system("pause");
}
