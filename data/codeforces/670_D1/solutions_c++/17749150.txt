#include<bits/stdc++.h>
using namespace std;
long long a,s,d[100002],f[100002],ans,h,j,k,l,i,n,m;
main(){
    cin>>n>>m;
    for(int i=0;i<n;i++)cin>>d[i];
    for(int i=0;i<n;i++)cin>>f[i];
    h=9000000000;
    while(h-ans>1){
        k=(h+ans)/2;
        s=m;
        j=1;
        for(i=0;i<n;i++){
            if(k*d[i]>f[i]){
                s-=k*d[i]-f[i];
                if(s<0) {j=0;break;}
            }
        }
        if(j!=0) ans=k; else h=k;
    }
    cout<<ans;
}
