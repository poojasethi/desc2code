#include<iostream>
#include<vector>
#include<cmath>
using namespace std;
bool haikya(int x,vector<int> a,vector<int> b,int k)
{
	for(int i=0;i<a.size();i++)
	{
		a[i]*=x;
		k-=max(0,a[i]-b[i]);
		a[i]/=x;
	}
	return k>=0;
}
int main()
{	//freopen("in.txt","r",stdin);
	int n,k,val=0,z;
	cin>>n>>k;
	vector<int> a(n);
	vector<int> b(n);
	for(int i=0;i<n;i++)
	cin>>a[i];
	for(int i=0;i<n;i++)
	cin>>b[i];
	while(haikya(val+1,a,b,k))
	{
	val++;
	}
	cout<<val<<"\n";
}
