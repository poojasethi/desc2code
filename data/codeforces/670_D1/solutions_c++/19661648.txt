#include<cstdio>

int main()
{
    int a[1100],b[1000];
    int n,k,i,sum,j;
    int ok;
    while(scanf("%d %d",&n,&k)!=EOF)
    {
        ok=0;
        for(i=0;i<n;i++) scanf("%d",&a[i]);
        for(i=0;i<n;i++) scanf("%d",&b[i]);
        for(i=0;i<=2001;i++)
      {
        sum=0;
        for(j=0;j<n;j++)
        {
            if(b[j]<a[j]*i)
            sum+=a[j]*i-b[j];

        }
        if(sum>k) {ok=1;}
        if(ok) {printf("%d\n",i-1);break;}
       }
    }

return 0;
}
