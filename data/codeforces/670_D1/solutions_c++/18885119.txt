#include<stdio.h>
#include<iostream>
using namespace std;
int n,k,cookie=0,m=0; 	
int a[2000],b[2000];

void OneMore(){
	for(int i=0;i<=n-1;i++){
		b[i]=b[i]-a[i];
		if(b[i]<=0)
		m+=b[i];
		else;			
	}
	if(m+k>=0){ cookie++;m=0;OneMore();}
	else;
}
int main(){
	scanf("%d",&n);
	scanf("%d",&k);
	for(int i=0;i<=n-1;i++){
		scanf("%d",&a[i]);
	}
	for(int i=0;i<=n-1;i++){
		scanf("%d",&b[i]);
}
	OneMore();
	printf("%d",cookie);
}
