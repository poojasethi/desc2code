#include <iostream>
using namespace std;

typedef long long ll;

ll n, k, a[100200], b[100200];

bool check(ll x)
{
	ll need = 0;
	for(ll i=0; need <= k && i<n; i++)
		need += max(x*a[i]-b[i], 0ll);

	return (need <= k);
}

int main()
{
	cin >> n >> k;
	for(ll i=0; i<n; i++)
		cin >> a[i];
	for(ll i=0; i<n; i++)
		cin >> b[i];

	ll l=0, r=(2e9)+100, m;
	while(l+1 < r)
	{
		m = (l+r)/2;
		if(check(m))
			l = m;
		else
			r = m;
	}

	cout << l << endl;

	return 0;
}
