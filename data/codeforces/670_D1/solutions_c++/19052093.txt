#include <iostream>
#include <cmath>

using namespace std;

int a[1001], b[1001];

int main()
{
	int n, k;
	cin>>n>>k;

	for (int i = 0; i < n; i++) cin>>a[i];
	for (int i = 0; i < n; i++) cin>>b[i];

	int c = 0;
	while (k > 0)
	{
		c++;
		for (int i = 0; i < n; i++)
		{
			b[i] -= a[i];
			if (b[i] < 0)
			{
				k += b[i];
				b[i] = 0;
			}
		}
	}

	cout<<(c-(k==0? 0: 1))<<endl;

	return 0;
}
