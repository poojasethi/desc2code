#include <bits/stdc++.h>
using namespace std;
int n, k, hg[1002], needs[1002], a, b, m, k2;
bool spr(int x)
{
  k2=k;
  for(int i=0; i<n; i++)
  {
    k2-=max(0, x*needs[i]-hg[i]);
    if(k2<0) return 0;
  }
  return 1;
}
int main()
{
  scanf("%d%d", &n, &k);
  for(int i=0; i<n; i++) scanf("%d", &needs[i]);
  for(int i=0; i<n; i++) scanf("%d", &hg[i]);
  a=0;
  b=1000+k/n+1;
  while(a<b)
  {
    m=(a+b)/2;
    if(spr(m)) a=m+1;
    else b=m;
  }
  printf("%d",a-1);
}
