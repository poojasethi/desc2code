#include<bits/stdc++.h>
using namespace std;
int a[1050],b[1050];
int n,k;
int main(){
	int ans=0;
	cin>>n>>k;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n;i++)cin>>b[i];
	while(k){
		for(int i=0;i<n;i++){
			b[i]-=a[i];
			if(b[i]<0){
				k+=b[i];
				b[i]=0;
			}
		}
		if(k<0){
			break;
		}
		else ans++;
	}
	cout<<ans<<endl;
}
