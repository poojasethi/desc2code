#include <bits/stdc++.h>

using namespace std;

int n,k,a[1001], b[1001];

main(){
	cin >> n >> k;
	for(int i=0; i<n; i++) cin >> a[i];
	for(int i=0; i<n; i++) cin >> b[i];
	int num = 0;
	bool flag = true;
	while(1){
		for(int i=0; i<n; i++){
			b[i] -= a[i];
			if(b[i] < 0){
				k -= abs(b[i]);
				b[i] = 0;
			}
			if(k < 0 ){
				flag = false;
				break;
			}
			//cout << b[i] << " ";
		}
		//cout << endl;
		if(!flag) break;
		num++;
	}
	cout << num;
}
