#include <iostream>
#include <cmath>
#include <bits/stdc++.h>

using namespace std;


int main(){

	int n,k;
	while(scanf("%d%d",&n,&k)!=EOF){
		int ans=0,i,a[1100],b[1100];
		for(i=0;i<n;i++){
			scanf("%d",&a[i]);
		}
		for(i=0;i<n;i++){
			scanf("%d",&b[i]);
		}

		while(1){
			for (i = 0; i < n; i++)
			{
				if(b[i] >= a[i]){
					b[i] = b[i] - a[i];
				}
				else{
						k=k-(a[i]-b[i]);
						b[i]=0;
				}
				if(i==n-1 && k>=0){
					ans++;
				}
			}
			if(k<0)
				break;
		}
		cout  << ans << endl;
	}

	return 0;
}
