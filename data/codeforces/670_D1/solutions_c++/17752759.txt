#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int n, k, res, a[1001], b[1001];

int main()
{
	scanf("%d %d", &n, &k);
	for(int i = 0 ; i < n ; i++)
		scanf("%d", &a[i]);
	for(int i = 0 ; i < n ; i++)
		scanf("%d", &b[i]);
	
	while( k > 0 ){
		bool f = 1;
		for(int i = 0 ; i < n ; i++){
			if( a[i] > b[i] ) k -= a[i] - b[i], b[i] = 0;
			else b[i] -= a[i];
                        if( k < 0 ) {f = 0; break;}
		}
		
		if( f ) res++;
	}
	cout << res << endl;
}
