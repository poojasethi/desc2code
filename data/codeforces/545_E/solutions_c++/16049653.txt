#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define sc scanf
#define pr printf
#define pb push_back
#define p_b pop_back
#define mp make_pair
#define f first
#define s second
using namespace std;
const ll inf=1000000000000000000;
ll n,m,z[300011],u;
    ll p[300011];
    ll d[300011];
vector<pair<ll,pair<ll,ll> > >v[300011];
priority_queue <pair<ll,ll> >q;
int main()
{
    cin>>n>>m;
    for (ll i=1;i<=m;i++)
    {
        ll x,y;
        cin>>x>>y>>z[i];
        v[x].pb(mp(i,mp(y,z[i])));
        v[y].pb(mp(i,mp(x,z[i])));
    }
    cin>>u;
    for (ll i=1;i<=n;i++)
    {
        d[i]=inf;
    }
    d[u]=0;
    q.push(mp(-d[u],u));
    ll g,ans=0;
    while (!q.empty())
    {
        g=q.top().s;
        q.pop();
        for (ll i=0;i<v[g].size();i++)
        {
            ll to=v[g][i].s.f;
            ll len=v[g][i].s.s;
            if (d[g]+len<d[to] || d[g]+len==d[to] && z[v[g][i].f]<z[p[to]])
            {
                ans-=z[p[to]];
//                cout<<ans<<" "<<g<<endl;
                d[to]=d[g]+len;
                p[to]=v[g][i].f;
                ans+=z[p[to]];
                q.push(mp(-d[to],to));
            }
        }
    }
    cout<<ans<<endl;
    for (ll i=1;i<=n;i++)
    {
        if (i!=u)cout<<p[i]<<" ";
    }
    return 0;
}
