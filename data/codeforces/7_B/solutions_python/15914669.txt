r=lambda:raw_input().split()
t,m=map(int,r())
l=[0]*m
n=1
for p in range(t):
	s=r()
	c=s[0][0]
	if c=='a':
		z=int(s[1])
		ok=1
		for i in range(m-z+1):
			if sum(l[i:i+z])==0:
				print n
				l=l[:i] + [n]*z + l[i+z:]
				n+=1
				ok=0
				break
		if ok: print 'NULL'
	elif c=='e':
		z=int(s[1])
		if z <= 0 or l.count(z)==0: print 'ILLEGAL_ERASE_ARGUMENT'
		else: l = [0 if x==z else x for x in l]
	else:
		k=[]
		for i in range(m): 
			if l[i]: 
				k.append(l[i])
		l = k + [0]*(m-len(k))
