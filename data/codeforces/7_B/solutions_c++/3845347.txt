#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

const int N=106;

int use[N], n, m, tot=0;
char name[100];

int main()
{
	scanf("%d%d", &n, &m);
	memset(use, 0, sizeof use);
	for(int i=0, a, s, j; i<n; i++)
	{
		scanf("%s", name);
		if(name[0]=='a')
		{
			scanf("%d", &a);
			for(s=0; s<m; s++)
			{
				for(j=s; j<a+s && j<m; j++) if(use[j]) break;
				if(j==a+s) break;
			}
			if(j==a+s)
			{
				printf("%d\n", ++tot);
				for(int j=s; j<a+s; j++) use[j]=tot;
			}
			else printf("NULL\n");
		}
		else if(name[0]=='e')
		{
			scanf("%d", &a);
			bool flag=0;
			for(int j=0; j<m; j++) if(use[j]==a && a>0) use[j]=0, flag=1;
			if(!flag) printf("ILLEGAL_ERASE_ARGUMENT\n");
		}
		else
		{
			for(j=0, s=0; j<m; j++) if(use[j]) use[s++]=use[j];
			for(; s<m; s++) use[s]=0;
		}
	}
	return 0;
}
