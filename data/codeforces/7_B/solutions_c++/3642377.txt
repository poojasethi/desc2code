#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
long n,m,x,i,j,p=0,a[200]={};
bool Find;
string s;
int main() {
	for (cin >> n >> m;n>0;n--) {
		cin >> s;
		if (s=="alloc") {
			cin >> x;Find=false;
			for (i=1;(i+x-1<=m)&&(!Find);i++) {
				while ((a[i]!=0)&&(i<=m)) i++;
				for (j=i;(j<=m)&&(j<=i+x-1);j++)
					if (a[j]!=0) break;
				if (j==i+x)  {
					cout << ++p << endl;
					for (j=i;j<i+x;j++) a[j]=p;
					Find=true;
				}
			}
			if (!Find) cout << "NULL" << endl;
		} else if (s=="erase") {
			cin >> x;Find=false;
			for (i=1;i<=m;i++) 
				if (a[i]==x) {
					a[i]=0;
					Find=true;
				}
			if ((!Find)||(x==0))
				cout << "ILLEGAL_ERASE_ARGUMENT" << endl;
		} else {
			i=j=1;
			while ((i<=m)&&(j<=m)) {
				while ((a[i]!=0)&&(i<=m)) i++;
				for (j=i;(a[j]==0)&&(j<=m);j++);
				for (;(a[j]!=0)&&(j<=m);i++,j++) swap(a[i],a[j]);
			}
		}
	}
	return 0;
}
