#include<iostream>
using namespace std;

int dat[110];
int main() {
	int T,M;
	cin>>T>>M;
	string S;
	int N,id=1;
	while(T--) {
		cin>>S;
		if(S[0]=='a') {
			cin>>N;
			int s=0;
			for(int i=0;i<M;i++) {
				if(i>=N) {
					s-=!dat[i-N];
				}
				s+=!dat[i];
				if(s==N) {
					s=-1;
					for(int j=i-N+1;j<=i;j++) {
						dat[j]=id;
					}
					cout<<id<<endl;
					id++;
					break;
				}
			}
			if(s!=-1) {
				cout<<"NULL"<<endl;
			}
		} else if(S[0]=='e') {
			cin>>N;
			if(N<=0) {
				cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
				continue;
			}
			bool f=false;
			for(int i=0;i<M;i++) {
				if(dat[i]==N) {
					dat[i]=0;
					f=true;
				}
			}
			if(!f) {
				cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
			}
		} else {
			int p=0;
			for(int i=0;i<M;i++) {
				if(dat[i]) {
					dat[p++]=dat[i];
				}
			}
			for(;p<M;p++) {
				dat[p]=0;
			}
		}
	}
}
