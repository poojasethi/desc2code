#include <iostream>
#include <string>
#include <cstdio>
using namespace std;
string s;
int bit[1000],cm,k,idef=1;
void erase()
{
	int x;
	bool flag = false;
	cin>>x;
	if(x==0){ cout<<"ILLEGAL_ERASE_ARGUMENT\n"; return ; }
	for(int i=0; i<k; i++)
		if(x==bit[i])bit[i]=0,flag=true;
	if(!flag)cout<<"ILLEGAL_ERASE_ARGUMENT\n";
}
void alloc()
{
	int res = 0,x,r=0;
	bool flag = false;
	cin>>x;
	for(int i=0; i<k; i++){
		if(bit[i]==0)res++; else res = 0;
		if(res==x){ flag = true; r=i; break; }
	}
	if(flag){
		for(int i=r; i>r-x; i--)
			bit[i]=idef;
		cout<<idef++<<'\n';
	} else cout<<"NULL"<<'\n';
}
void defragment()
{
	int pp[1000],L=0;
	for(int i=0; i<k; i++){
		pp[i]=0;
		if(bit[i]!=0)pp[L++]=bit[i];
	}
	for(int i=0; i<k; i++)
		bit[i]=pp[i];
}
int main()
{
	cin>>cm>>k;
	for(int i = 0; i<cm; i++){
		cin >> s;
		if(s=="alloc")alloc();
		if(s=="defragment")defragment();
		if(s=="erase")erase();
	}
return 0;	
}
