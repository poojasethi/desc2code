#include <iostream>
#include <string.h>
using namespace std;
int m;
int a[110];
int alloc(int x,int h)
{
	int i,j,s;
	for(i=1;i<=m;i++)
	{
		s=0;
		if(a[i]==0)
		{
			for(j=i;j<i+x;j++)
			{
				if(j>m) {s=1;break;}
				if(a[j]!=0) {s=1;break;}
			}
			if(s==0) 
			{
				for(j=i;j<i+x;j++)
					a[j] = h;
				return 1;
			}
		}	
	}
	return 0;
}
int main()
{
	int t,x,h=1;
	cin>>t>>m;
	while(t--)
	{
		char str[20];
		cin>>str;
		if(!strcmp(str,"alloc"))
		{
			cin>>x;
			if(alloc(x,h)) cout<<h++<<endl;
			else cout<<"NULL"<<endl;
		}
		else if(!strcmp(str,"erase"))
		{
			cin>>x;
			if(x<=0) cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
			else
			{
				int e=0;
				for(int i=1;i<=m;i++)
				{
					if(a[i]==x) {a[i]=0;e=1;}
				}
				if(e==0) cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
			}
		}
		else 
		{
			int b[110]={0};
			int k=1;
			for(int i=1;i<=m;i++)
			{
				if(a[i]!=0) b[k++] = a[i];
			}
			memset(a,0,sizeof(a));
			for(int i=1;i<=k;i++)
				a[i] = b[i];
		}
	}
}
