#include<stdio.h>
#include<string.h>
#include<iostream>
#include<string>
#define N 1000005
using namespace std;
int a[N],b[N];
string B="alloc",C="erase",D="defragment";
int main()
{
	int n,m;
	cin>>n>>m;
	string A;
	int k,num=1;
	for(int i=0;i<n;i++)
	{
		memset(b,0,sizeof(b));
		cin>>A;
		int h=0,blank=0;
		if(A==B)
		{
			cin>>k;int l=0;
			while(blank<k&&h<m&&l<m)
			{
				blank=0;
				h=l;
				while(a[h]!=0&&h<m)
				{
					h++;
				}
				l=h;
				while(a[l]==0&&l<m)
				{
					l++;blank++;
				}
			}
			if(blank>=k)
			{
				for(int s=h;s<h+k;s++)
				{
					a[s]=num;
				}
				num++;
				cout<<num-1<<endl;
			}
			else cout<<"NULL"<<endl;
		}
		int flag=0;
		if(A==C)
		{
			cin>>k;
			for(int i=0;i<m;i++)
			{
				if(k==0)break;
				if(a[i]==k)
				{
					a[i]=0;flag=1;
				}
			}
			if(flag==0)cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
		}
		if(A==D)
		{
			for(int j=0;j<m;j++)
			{
				if(a[j]!=0)
				{
					b[h++]=a[j];
				}
			}
			for(int j=0;j<m;j++)
			{
				a[j]=b[j];
			}
		}
	}	
}
