#include <iostream>
using namespace std;

int a[110], n, m, id = 1;
char cmd[100];

int main()
{
	int i, j, s;
	cin >> m >> n;
	while(m--)
	{
		cin >> cmd;
		if(cmd[0] == 'a')
		{
			cin >> s;
			for(i = 0; i <= n - s; ++i)
			{
				j = 0;
				for(; j < s; ++j)
				{
					if(a[i + j] != 0)
						break;
				}
				if(j == s)	break;
			}
			if(i <= n - s)
			{
				cout << id << endl;
				for(j = i; j < i + s; ++j)
					a[j] = id;
				id++;
			}
			else
				cout << "NULL" << endl;
		}
		else if(cmd[0] == 'e')
		{
			cin >> s;
			bool flag = true;
			for(i = 0; i < n; ++i)
				if(a[i] && a[i] == s)
					a[i] = 0, flag = false;
			if(flag)	cout << "ILLEGAL_ERASE_ARGUMENT" << endl;
		}
		else
		{
			for(i = 0, j = 0; i < n; i++)
				if(a[i])	a[j++] = a[i];
			while(j < n)	a[j++] = 0;
		}
	}
	return 0;
}
