#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

int n,m,c,x,f,e,v[105],l[105];
char s[50];

int main(){
	scanf("%d%d",&m,&n);
	memset(v,0,sizeof(v));
	while (m--){
		scanf("%s",s);
		if (s[0]=='a'){
			scanf("%d",&x);
			f=0;
			for (int i=1;i+x-1<=n;i++){
				e=1;
				for (int j=i;j<i+x;j++)
				    if (v[j]!=0) e=0;
				if (e){
					c++;
					for (int j=i;j<i+x;j++) v[j]=c;
					l[c]=1;
					f=1;
					break;
				}
			}
			if (f) printf("%d\n",c);
			else printf("NULL\n");
		}
		if (s[0]=='e'){
			long long int x;
			scanf("%I64d",&x);
			if (x<=0||x>c||l[x]==0) printf("ILLEGAL_ERASE_ARGUMENT\n");
			else{
				for (int i=1;i<=n;i++)
				    if (v[i]==x) v[i]=0;
				l[x]=0;
			}
		}
		if (s[0]=='d'){
			for (int i=1,j=1;i<=n;i++)
			    if (v[i]!=0){
					v[j]=v[i];
					if (i!=j) v[i]=0;
					j++;
				}
		}
	}
	return 0;
}
