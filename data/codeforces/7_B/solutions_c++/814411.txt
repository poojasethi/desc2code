#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int used[100];
int main(void)
{
	int n,m,i,j,t=1;
	scanf("%d%d", &n,&m);
	char s[20];
	while ( n-- ){
		int k;
		scanf("%s", s);
		if ( s[0] == 'a' ){
			scanf("%d",&k);	
			for ( i=1; i+k<=m+1; i++ ){
				for ( j=0; j<k; j++ )
					if ( used[i+j] ) break;
				if ( j == k ) break;	
			}
			if ( i+k>m+1 )puts("NULL");
			else{
				printf("%d\n",t);
				for ( j=0; j<k;j++ ) used[i+j]=t;
				t++;
			}
		}else if ( s[0] == 'e' ){
			scanf("%d", &k);
			j=0;
			for (i=1; i<=m; i++ )
				if ( used[i] == k ){
					used[i]=0;
					j=1;
				}
			if (!j || !k )puts("ILLEGAL_ERASE_ARGUMENT");
		}else{
			for (i=1,j=1; i<=m; i++ )
				if ( used[i] ) used[j++] = used[i];
			while ( j<=m ) used[j++] = 0;
		}
	}
	return 0;
}
