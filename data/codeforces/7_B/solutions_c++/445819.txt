#include <stdio.h>

int w[105];
int id[105];
int pos[105];
int n=0;

int main(void) {
	int t, m, i, j;
	scanf("%d%d", &t, &m);
	while(t--) {
		char a[100];
		scanf("%s",a);
		if(a[0]=='a') {
			int x;
			scanf("%d", &x);
			for(i=1;i<=m-x+1;i++){
				for(j=0;j<x;j++)
					if(w[i+j]) break;
				if(j==x){
					break;
				}
			}
			if(i<=m-x+1) {
				++n;
				printf("%d\n", n);
				w[i] = n;
				for (j=1;j<x;j++)
					w[i+j] = -1;
				pos[n] = i;
			}
			else
				printf("NULL\n");
		} else if (a[0] == 'e') {
			int x;
			scanf("%d", &x);
			if(x<1||x>n||pos[x]<=0)
				printf("ILLEGAL_ERASE_ARGUMENT\n");
			else {
				w[pos[x]] = -1;
				for(i=pos[x];w[i]==-1&&i<=m;i++)
					w[i] = 0;
				pos[x] = 0;
			}
		} else {
			for(i=j=1;i<=m;i++) {
				if(w[i]>0)
					pos[w[i]] = j;
				if(w[i])
					w[j++] = w[i];
			}
			for(;j<=m;j++)
				w[j] = 0;
		}
	}
	return 0;
}
