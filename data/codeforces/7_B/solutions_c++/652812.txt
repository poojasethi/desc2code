#include <iostream>
#include <vector>
using namespace std;

int main()
{
  int t,m;
  cin>>t>>m;
  int C[m];
  for(int i=0;i<m;++i)
    C[i]=0;
  int N=1;
  while(t--)
    {
      string ord;
      cin>>ord;
      if(ord=="alloc")
	{
	  int size;
	  cin>>size;
	  bool made=false;
	  for(int i=0;i<m;++i)
	    {
	      bool ok=true;
	      for(int j=0;j<size;++j)
		if(i+j>=m or C[i+j]!=0)
		  ok=false;
	      if(ok==true)
		{
		for(int j=0;j<size;++j)
		    C[i+j]=N;
		cout<<N<<endl;
		N++;
		made=true;
		break;
		}//end of if
	    }
	  if(!made)
	    cout<<"NULL"<<endl;
	}
      if(ord=="erase")
	{
	  bool ok=false;
	  int x;
	  cin>>x;
	  for(int i=0;i<m;++i)
	    if(C[i]==x)
	      {
		C[i]=0;
		ok=true;
	      }
	  if(x<1)
	    ok=false;
	  if(!ok)
	    cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
	}
      if(ord=="defragment")
	{
	  vector<int> def;
	  for(int i=0;i<m;++i)
	    if(C[i]!=0)
	      def.push_back(C[i]);
	  while(def.size()<m)
	    def.push_back(0);
	  for(int i=0;i<m;++i)
	    C[i]=def[i];
	}
	
    }
  return 0;
}
