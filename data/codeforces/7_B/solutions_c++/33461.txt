#include <iostream>
#include <algorithm>
#include <cstdio>
#include <string>

using namespace std ;

int mem[200], cnt ;

int main(){
	int t, m ; scanf(" %d %d", &t, &m) ;
	for(int i = 0 ; i < t ; i++){
		char c ; scanf(" %c", &c) ;
		if ( c == 'a' ){
			scanf("%c%c%c%c", &c,&c,&c,&c) ;
			int x ; scanf(" %d", &x) ; bool fo = false ;
			for(int j = 0, sz = 0  ; j < m ; j++){ 
				sz ++ ; if ( mem[j] != 0 ) sz = 0 ;
				if ( sz == x ){
					cnt ++ ;
					for(int k = j - sz + 1 ; k <= j ; k++) mem[k] = cnt ; 
					printf("%d\n", cnt) ; fo = true ;
					break ;
				}
			}
			if ( fo == false ) printf("NULL\n") ;
		}else if ( c == 'e' ){
			scanf("%c%c%c%c", &c,&c,&c,&c) ;
			int x ; scanf("%d", &x) ; bool fo = false ;
			for(int j = 0 ; j < m ; j++)
				if ( mem[j] == x ) mem[j] = 0, fo = true ; 
			if ( x == 0 || fo == false ) printf("ILLEGAL_ERASE_ARGUMENT\n") ;
		}else{
			scanf("%c%c%c%c%c%c%c%c%c", &c,&c,&c,&c,&c,&c,&c,&c,&c) ;
			for(int j = 0 ; j < m ; j++) 
				for(int k = j; k>0 && mem[k-1]==0 ; k--) swap(mem[k-1], mem[k]) ;
		}
	}
}
