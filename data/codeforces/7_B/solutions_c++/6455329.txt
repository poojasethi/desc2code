#include <iostream>
#include <stdio.h>

using namespace std;

string s;
int t,m,i,a[500],x,j,bl;

main()
 {
  cin>>t>>m;
  while (t--)
   {
    cin>>s;
    if (s[0] == 'a')
     {
      cin>>x;
      if (x > m)
       { cout<<"NULL"<<endl; continue; }
      for (i=1;i<=m-x+1;i++)
       {
        bool ma=true;
        for (j=0;j<x;j++)
         if (a[i+j] != 0) ma=false;
        if (ma)
         {
          bl++;
          for (j=0;j<x;j++)
           a[i+j]=bl;
          cout<<bl<<endl;
          break;
         }
       }
      if (i == m-x+2)
        cout<<"NULL"<<endl;
     } else
    if (s[0] == 'e')
     {
      cin>>x;
      int po=0;
      for (i=1;i<=m;i++)
       if (a[i] == x) { a[i]=0; po++; }
      if (po == 0 || x > bl || x <= 0) cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
     } else
     {
      for (i=1;i<=m;i++)
       if (a[i] == 0)
        {
         for (j=i;j<=m && (a[j] == 0);j++);
         while (j > i) { swap(a[j],a[j-1]); j--; }
        }
     }
   }
 }
