#include <iostream>
#include <set>
#include <string>
using namespace std;

int A[110];
int main(){
  int t, m;
  cin>>t>>m;
  int cnt = 0;
  while(t--){
    string s;
    int a;
    cin>>s;
    if(s[0] == 'a'){
      cin>>a;
      int d = 0;
      int p;
      for(int i=0;i<m;i++){
    d = 0;
    p = i;
    while(A[p] == 0 and p < m){
      d++;
      p++;
    }
    if(d >= a){
      p = i;
      break;
    }
    p = -1;
      }
      if(p == -1)
    cout<<"NULL"<<endl;
      else{
    cout<<++cnt<<endl;
    for(int i=0;i<a;i++)
      A[i+p] = cnt;
      }
    }
    else if(s[0] == 'e'){
      cin>>a;
      bool find = false;
      for(int i=0;i<m;i++)
    if(A[i] == a)
      find = true;
      if(a<= 0 or find == false){
    cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
      }
      else{
    for(int i=0;i<m;i++)
      if(A[i] == a)
        A[i] = 0;
      }
    }
    else{
      int p = 0;
      for(int i=0;i<m;i++){
    if(A[i] != 0){
      int d = A[i];
      A[i] = 0;
      A[p] = d;
      p++;
    }
      }
    }
  }
  return 0;
}
