#include<cstdio>
int Z,N,z=1,a[105],k,cnt;
char s[30];
bool ok,u[105];
int main(){
	scanf("%d%d",&Z,&N);
	for (int i=0;i<=102;++i) u[i]=a[i]=0;
	while (Z--){
		scanf("%s",s);
		if (s[0]=='a'){
			ok = false;
			scanf("%d",&k);
			cnt = 0;
			for (int i=1;!ok && i<=N;++i){
				if (a[i]==0){
					++cnt;
				}else{
					cnt = 0;
				}
				if (cnt==k){
					ok = true;
					for (int j=0;j<k;++j) a[i-j]=z;
					u[z] = true;
				}
			}
			if (ok){
				printf("%d\n",z++);
			}else{
				puts("NULL");
			}
		}else if (s[0]=='e'){
			scanf("%d",&k);
			if (k<=0 || k>=z){
				puts("ILLEGAL_ERASE_ARGUMENT");
				continue;
			}
			if (u[k]){
				for (int i=1;i<=N;++i) if (a[i]==k) a[i]=0;
				u[k] = false;
			}else{
				puts("ILLEGAL_ERASE_ARGUMENT");
			}
		}else if (s[0]=='d'){
			int i = 1;
			for (int j=1;j<=N;++j){
				if (a[j]!=0){
					a[i++] = a[j];
				}
			}
			while (i<=N){
				a[i] = 0;
				++i;
			}
		}
	}

	return 0;
}
