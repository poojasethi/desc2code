#include<stdio.h>
#include<string.h>
int t,m,mem[105];
int judge(int n,int num)
{
	int i,j;
	for(i=1;i<=m-n+1;i++)
	{
		for(j=i;j<i+n;j++)
			if(mem[j]!=0)
				break;
		if(j>=i+n)
		{
			for(j=i;j<i+n;j++)
				mem[j]=num;
			return 1;
		}
	}
	return 0;
}
int main()
{
	char ope[15];
	int n,num,i,j,flag;
	while(scanf("%d%d",&t,&m) != EOF)
	{
		num=1;
		while(t--)
		{
			scanf("%s",ope);
			if(strcmp(ope,"alloc")==0)
			{
				scanf("%d",&n);
				if(judge(n,num))
				{
					printf("%d\n",num);
					num++;
				}
				else
					printf("NULL\n");
			}
			else
			{
				if(strcmp(ope,"erase")==0)
				{
					scanf("%d",&n);
					flag=0;
					if(n!=0)
					{
						for(i=1;i<=m;i++)
							if(mem[i]==n)
							{
								mem[i]=0;
								flag=1;
							}
					}
					if(!flag)
						printf("ILLEGAL_ERASE_ARGUMENT\n");
				}
				else
					for(i=1,j=1;j<=m;j++)
					{
						mem[i]=mem[j];
						if(mem[j]!=0)
						{
							if(j>i)
								mem[j]=0;
							i++;
						}
					}
			}
		}
	}
	return 0;
}
