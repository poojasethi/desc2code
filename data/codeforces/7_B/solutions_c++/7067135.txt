#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int vis[200];
int temp[200];
int main()
{
	int t,m,c=1;
	scanf("%d%d",&t,&m);
	char op[20];
	for(int i=0;i<t;i++)
	{
		scanf("%s",op);
		int s;
		if(strcmp(op,"alloc")==0)
		{
			scanf("%d",&s);
			int pre=-1,flag=1;
			for(int j=0;j<m;j++)
			{
				if(vis[j]==0)
				{
					if(pre==-1) pre=j;
					if(j-pre+1>=s)
					{
						for(int k=pre;k<=j;k++) vis[k]=c;
						printf("%d\n",c++);
						flag=0;
						break;
					}
				}
				else pre=-1;
			}
			if(flag) puts("NULL");
		}
		else if(strcmp(op,"erase")==0)
		{
			scanf("%d",&s);
			int flag=1;
			for(int j=0;j<m;j++)
				if(vis[j]==s)
				{
					flag=0;
					vis[j]=0;
				}
			if(flag || s<=0) puts("ILLEGAL_ERASE_ARGUMENT");
		}
		else
		{
			memset(temp,0,sizeof(temp));
			int k=0;
			for(int j=0;j<m;j++)
				if(vis[j]!=0)
					temp[k++]=vis[j];
			memcpy(vis,temp,sizeof(vis));
		}
	}
	return 0;
}
