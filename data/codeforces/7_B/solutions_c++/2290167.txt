#include<stdio.h>

int main()
{
	int T,N,tot,a[105],k,cnt,fg,v[105],i,j;
	char s[30];
	while(~scanf("%d%d",&T,&N))
	{
		tot=1;
		for(i=0;i<105;i++) v[i]=a[i]=0;
		while(T--)
		{
			scanf("%s",s);
			if(s[0]=='a')
			{
				fg=0;
				scanf("%d",&k);
				cnt=0;
				for(i=1;!fg&&i<=N;++i)
				{
					if(a[i]==0) cnt++;
					else cnt=0;
					if(cnt==k)
					{
						fg=1;
						for(j=0;j<k;++j) a[i-j]=tot;
						v[tot]=1;
					}
				}
				if(fg) printf("%d\n",tot++);
				else puts("NULL");
			}else if(s[0]=='e')
			{
				scanf("%d",&k);
				if(k<=0||k>=tot)
				{
				puts("ILLEGAL_ERASE_ARGUMENT");
				continue;
				}
				if(v[k])
				{
					for(i=1;i<=N;++i) 
					if(a[i]==k) a[i]=0;
					v[k] = false;
				}else puts("ILLEGAL_ERASE_ARGUMENT");
			}
			else if(s[0]=='d')
			{
				i=1;
				for(j=1;j<=N;++j)
				if(a[j]!=0) a[i++]=a[j];
				while(i<=N)
				{
					a[i]=0;
					i++;
				}
			}
		}
	}
}
