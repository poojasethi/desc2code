#include <cstdio>
using namespace std;

int memory;
int t,m;
int M[101];
int maxx = 1;

int sucAlloc(int n, int b){
	for (int i = 0; i < n; i++)
		M[b+i] = maxx;
	return maxx++;
}

int alloc(int n){
	int mm = 0;
	for (int i = 0; i < m; i++){
		if (M[i] == 0) mm++;
		else mm = 0;
		if (mm >= n) return sucAlloc(n, i-n+1);
	}
	return 0;
}

int erase(int id){
	if (id == 0) return 0;
	bool ok = 0;
	for (int i = 0; i < m; i++)
		if (M[i] == id){
			ok = 1;
			M[i] = 0;
		}
	return ok ? 1 : 0;
}

void defragment(){
	int MM[101]; for (int i = 0; i < m; i++) MM[i] = 0;
	int j = 0;
	for (int i = 0; i < m; i++)
		if (M[i] != 0) MM[j++] = M[i];
	for (int i = 0; i < m; i++)
		M[i] = MM[i];
}

int main(){
	scanf ("%d %d", &t, &m);
	char cmd[100];
	int d;
	for (int i = 0; i < t; i++){
		scanf("%s", cmd);
		switch (cmd[0]){
			case 'a': scanf ("%d", &d);
					  d = alloc(d);
					  if (d) printf("%d\n", d);
					  else printf ("NULL\n");
					  break;
			case 'e': scanf("%d", &d);
					  if (!erase(d)) printf ("ILLEGAL_ERASE_ARGUMENT\n");
					  break;
			default:  defragment(); break;
		}
	}
	return 0;
}
