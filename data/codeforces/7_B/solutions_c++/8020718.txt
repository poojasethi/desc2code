#include <iostream>

int main()
{
	int t, m, id(1);
	int memory[100] = {0};
	int arg;
	std::string cmd;
	int i(0), j(0);

	std::cin >> t >> m;
	while (t--)
	{
		std::cin >> cmd;
		if ('a' == cmd[0])
		{
			std::cin >> arg;
			for (i = j = 0; j < m; )
			{
				if (!memory[j])
				{
					if (++j - i >= arg) break;
				}
				else ++j, (i = j);
			}
			if (j - i >= arg)
			{
				while (i != j)
				{
					memory[i++] = id;
				}
				std::cout << id++ << '\n';
			}
			else std::cout << "NULL\n";
		}
		else if ('e' == cmd[0])
		{
			std::cin >> arg;
			for (i = 0; i < m && memory[i] != arg; ++i)
				;
			if (i < m && arg)
			{
				while (i < m && memory[i] == arg)
				{
					memory[i++] = 0;
				}
			}
			else std::cout << "ILLEGAL_ERASE_ARGUMENT\n";
		}
		else
		{
			for (i = j = 0; j < m; ++j)
			{
				if (memory[j]) memory[i++] = memory[j];
			}
			while (i != j)
			{
				memory[i++] = 0;
			}
		}
	}
	return 0;
}
