#include <vector>
#include <algorithm>
#include <iostream>

using namespace std;

int main()
{
	int t,m;
	cin >> t >> m;
	vector<int> v(m, 0);
	int all = 1;
	for (int i = 1 ; i <= t ; i ++)
	{
		string s;
		cin >> s;
		if(s == "alloc")
		{
			int n;
			cin >> n;
			vector<int>::iterator vi = search_n(v.begin(), v.end(), n, 0);
			if (vi == v.end())
				cout << "NULL" << endl;
			else
			{
				for (int j = 0 ; j < n ; j ++)
				{
					*vi = all;
					++vi;
				}
				cout << all++ << endl;
			}

		}
		else if (s == "erase")
		{
			int n;
			cin >> n;
			vector<int>::iterator vi = find(v.begin(), v.end(), n);
			if (n == 0 || vi == v.end())
			{
				cout << "ILLEGAL_ERASE_ARGUMENT" << endl;
			}
			else
			{
				while (vi != v.end() && *vi == n)
				{
					*vi = 0;
					++ vi;
				}
			}
		}
		else
		{
			vector<int>::iterator vi = v.begin();
			while (vi != v.end())
			{
				if (*vi == 0)
					vi = v.erase(vi);
				else
					vi ++;
			}
			v.resize(m, 0);
		}
	}
	return 0;
}
