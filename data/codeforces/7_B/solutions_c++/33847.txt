#include<iostream>
#include<cstdio>
#include<cstring>
#include<set>
#include<vector>
#include<cmath>
#include<algorithm>
using namespace std;
int t, n, cnt = 1;
int mem[100];

void defrag(){
	int st = 0;
	for(int i = 0; i < n; i++)
		if(mem[i] != 0) mem[st++] = mem[i];
	for(int i = st; i < n; i++) mem[i] = 0;
}

void erase(long long x){
	bool found = false;
	for(int i = 0; i < n; i++) if(mem[i] == x){
		mem[i] = 0;
		found = true;
	}
	if(!found) cout << "ILLEGAL_ERASE_ARGUMENT" << endl; 
}

void alloc(int x){
	for(int i = 0; i < n; i++) if(mem[i] == 0 && (i == 0 || mem[i - 1] != 0)){
		int sz = 0;
		for(int j = i; j < n; j++) if(mem[j] != 0) break; else sz++;
		if(sz >= x){
			for(int j = 0; j < x; j++) mem[i + j] = cnt;
			cout << cnt << endl;
			cnt++;
			return;
		}
	}
	cout << "NULL" << endl;
}
int main(){
	cin >> t >> n;
	memset(mem, 0, sizeof mem);
	while(t--){
		string s; cin >> s;
		
		if(s[0] == 'd') defrag();
		else{
			long long x; cin >> x;
			if(s[0] == 'a') alloc(x);
			else if(x <= 0) cout << "ILLEGAL_ERASE_ARGUMENT" << endl;
			else erase(x);
		}
	}
}
