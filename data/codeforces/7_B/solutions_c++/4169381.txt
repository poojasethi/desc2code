#include<cstdio>
int v[105];
char op[15];
int main()
{
	int t,m,n,i,j;
	scanf("%d%d",&t,&m);
	int pid=1;
	while(t--)
	{
		scanf("%s",op);
		if(op[0]=='d')
		{
			for(i=0,j=0;i<m;i++)if(v[i]) v[j++]=v[i];
			for(;j<m;j++)v[j]=0;
		}
		else
		{
			scanf("%d",&n);
			if(op[0]=='a')
			{
				for(i=0;i<=m-n;i++)
				{
					for(j=i;j<i+n;j++)if(v[j]!=0)break;
					if(j==i+n)break;
				}
				if(i<=m-n)
				{
					printf("%d\n",pid);
					for(j=i;j<i+n;j++)v[j]=pid;
					pid++;
				}
				else puts("NULL");
			}
			else
			{
				bool f=0;
				for(i=0;i<m;i++)if(v[i]&&v[i]==n){v[i]=0;f=1;}
				if(!f)puts("ILLEGAL_ERASE_ARGUMENT");
			}
		}
	}
	return 0;
}
