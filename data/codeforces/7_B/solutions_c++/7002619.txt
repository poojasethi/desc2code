#include <iostream>
using namespace std;
int a[100], t, m, n, i, j, k;
string s;
int main()
{
	cin>>t>>m;
	while (t--) {
		cin>>s;
		if (s=="defragment") {
			for (i=0; i<m&&a[i]>0; i++);
			for (j=i; j<m; j++) {
				if (a[j]) {
					a[i++]=a[j];
					a[j]=0;
				}
			}
		}
		else if (s=="alloc") {
			cin>>n;
			for (i=0; i<=m-n; i++) {
				for (j=0; j<n; j++) if (a[i+j]) break;
				if (j==n) {
					for (k++,j=0; j<n; j++) a[i+j]=k;
					cout<<k<<endl;
					break;
				}
				else i+=j;
			}
			if (i>m-n) cout<<"NULL"<<endl;
		}
		else if (s=="erase") {
			cin>>n;
			for (i=0,j=0; i<m; i++) if (a[i]==n) {
				a[i]=0;
				j++;
			}
			if (j==0 || n==0) cout<<"ILLEGAL_ERASE_ARGUMENT"<<endl;
		}
	}
	return 0;
}
