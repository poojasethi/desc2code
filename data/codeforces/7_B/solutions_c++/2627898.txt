#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
	int t,m;
	int mem[101];
	char command[50];
	scanf("%d%d",&t,&m);
	int cur=0;
	memset(mem,0,sizeof mem);
	while(t--){
		scanf("%s",command);
		if(command[0]=='a'){
			int n;
			scanf("%d",&n);
			bool ok=false;
			int temp=0;
			for(int i=1;i<=m;i++){
				if(mem[i]==0) temp++;
				else temp=0;
				if(temp==n){
					cur++;
					ok=true;
					for(int j=i-n+1;j<=i;j++)
						mem[j]=cur;
					break;
				}
			}
			if(ok) printf("%d\n",cur);
			else puts("NULL");
		}
		else if(command[0]=='e'){
			bool ok=false;
			long long x;
			scanf("%I64d",&x);
			if(x==0||x>100){
				puts("ILLEGAL_ERASE_ARGUMENT");
				continue;
			}
			for(int i=1;i<=m;i++){
				if(mem[i]==x){
					ok=true;
					mem[i]=0;
				}
			}
			if(ok==false) puts("ILLEGAL_ERASE_ARGUMENT");
		}
		else{
			int i=1;
			while(mem[i]!=0) i++;
			for(int j=i;j<=m;j++){
				if(mem[j]!=0){
					mem[i++]=mem[j];
					mem[j]=0;
				}
			}
		}
	}
	return 0;
}
