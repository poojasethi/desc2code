#include <iostream>
#include <string>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int t, m;
int mem[100];
int id;

int main() {
    cin >> t >> m;
    id = 1;
    rep(q, t) {
        string op;
        cin >> op;
        if(op=="alloc") {
            int n;
            cin >> n;
            int nid = -1;
            rep(i, m-n+1) {
                bool f = true;
                rep(j, n) if(mem[i+j]) f = false;
                if(f) {
                    nid = id++;
                    rep(j, n) mem[i+j] = nid;
                    break;
                }
            }
            if(nid>0) cout << nid << endl;
            else cout << "NULL" << endl;
        }
        else if(op=="erase") {
            int eid;
            cin >> eid;
            bool illegal=true;
            rep(i, m) if(mem[i]==eid) { mem[i]=0; illegal=false; }
            if(eid==0 || illegal) cout << "ILLEGAL_ERASE_ARGUMENT" << endl;
        }
        else {
            rep(w, m) rep(i, m-1) if(mem[i]==0) swap(mem[i], mem[i+1]);
        }
    }
    return 0;
}
