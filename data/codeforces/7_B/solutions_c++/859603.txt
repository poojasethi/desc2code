#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	char s[20];
	int n,m,i,j,k;
	int a[110],b[110];
	int cnt=0;
	memset(a,0,sizeof(a));
	scanf("%d%d",&n,&m);
	while(n--){
		scanf("%s",s);
		if(s[0]=='d'){
			memcpy(b,a,sizeof(a));
			memset(a,0,sizeof(a));
			k=0;
			for(i=1;i<=m;i++)
				if(b[i]) a[++k]=b[i];
		}
		else if(s[0]=='a'){
			scanf("%d",&k);
			i=1;
			while(i<=m){
				j=i;
				while(j<=m&&!a[j]) j++;
				if(j-i>=k){
					cnt++;
					for(j=0;j<k;j++)
						a[i+j]=cnt;
					printf("%d\n",cnt);
					break;
				}
				i=j+1;
			}
			if(i>m) printf("NULL\n");
		}
		else{
			k=0;
			scanf("%d",&j);
			for(i=1;i<=m;i++){
				if(a[i]==j) k=1,a[i]=0;
			}
			if(!k||j<=0) printf("ILLEGAL_ERASE_ARGUMENT\n");
		}
	}
	return 0;
}
