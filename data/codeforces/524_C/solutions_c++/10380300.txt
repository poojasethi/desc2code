#include <algorithm>
#include <iostream>
#include <stdio.h>
#include <vector>
#include <map>
using namespace std;

map < int , int > mp;
int rx,i,n,k,j,ans,q,x,a[6000];

main()
 {
   cin>>n>>k;
   for (i=1;i<=n;i++)
    {
     scanf("%d",&a[i]);
     for (j=1;j<=k;j++)
     {
      rx=mp[j*a[i]];
      if (rx == 0 || rx > j) mp[j*a[i]]=j;
     }
    }
   cin>>q;
   while (q--)
    {
     cin>>x;
     ans=1000000;
     for (i=1;i<=n;i++)
      if (x%a[i] == 0) ans=min(ans,x/a[i]);
     for (i=1;i<=n;i++)
      for (j=1;j<=k;j++)
      {
        rx=mp[x-a[i]*j];
        if (!rx) continue;
        ans=min(ans,j+rx);
      }
     if (ans > k) ans=-1;
     cout<<ans<<endl;
    }

 }
