#include <cstdio>
#include <climits>
#include <algorithm>
#include <map>
using namespace std;

int den[5011];

int main(int argc, char const *argv[]) {
	int n, k, q, x, r;
	map<int, int> mp;
	
	scanf("%d%d", &n, &k);
	for (int i = 0; i < n; i++) {
		scanf("%d", &den[i]);
		for (int j = 1; j <= k; j++) {
			if (mp.count(den[i] * j) == 0) 
				mp[den[i] * j] = j;
			mp[den[i] * j] = min(mp[den[i] * j], j);
		}
	}
	mp[0] = 0;
	
	scanf("%d", &q);
	for (int i = 0; i < q; i++) {
		r = INT_MAX;
		
		scanf("%d", &x);
		for (map<int, int>::iterator it = mp.begin(); it != mp.end(); it++)
			if(mp.count(x - it -> first)) 
				r = min(it -> second + mp[x - it -> first], r);
		
		(r > k) ? puts("-1") : printf("%d\n", r);
	}

	return 0;
}
