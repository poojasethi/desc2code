#include<stdio.h>
#include<map>
#include<algorithm>
using namespace std;
const int maxn=5050;
int n,K,a[maxn],q,x,ans;
map<int,int> hav;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	scanf("%d%d",&n,&K);
	hav[0]=1;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		for(j=0;j<=K;j++)
			if(!hav.count(a[i]*j))
				hav[a[i]*j]=j;
			else hav[a[i]*j]=min(hav[a[i]*j],j);
	}
	scanf("%d",&q);
	while(q--)
	{
		ans=K+1;
		scanf("%d",&x);
		for(i=0;i<n;i++)
			for(j=0;j<=K;j++)
				if(hav.count(x-a[i]*j))
					ans=min(ans,j+hav[x-a[i]*j]);
		printf("%d\n",ans==K+1?-1:ans);
	}
	return 0;
}
