#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int n,k,a[5005],b[100005],c=0;
map<int,int> mp;
map<int,int>::iterator it;
int main(){
    cin>>n>>k;
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int i=0;i<n;i++)
        for(int j=1;j<=k;j++){
            if(!mp[a[i]*j]||mp[a[i]*j]>j) mp[a[i]*j]=j;
        }
    for(it=mp.begin();it!=mp.end();it++){
        b[c++]=(*it).first;
    }
    int q,x;
    cin>>q;
    while(q--){
        int ret=30;
        cin>>x;
        for(int i=0;i<c;i++) if(mp[x-b[i]]||b[i]==x){
            ret=min(ret,mp[x-b[i]]+mp[b[i]]);
        }
        if(ret<=k) printf("%d\n",ret);
        else printf("-1\n");
    }
    return 0;
}
