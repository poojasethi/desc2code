#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

#define N 5000
#define M 10000001

int a[N], p[M];


int main() {
	int n, k, q, x, ans;

	memset(p, 0xff, sizeof p);

	ios::sync_with_stdio(0);
	cin >> n >> k;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
		p[a[i]] = i;
	}

	cin >> q;
	while (q--) {
		cin >> x;
		ans = 0x7fffffff;
		for (int i = 0; i < n; i++) {
			int r = x;
			for (int b1 = 1; b1 <= k; b1++) {
				r -= a[i];
				if (r < 0)
					break;
				if (r == 0) {
					ans = min(ans, b1);
					break;
				}
				for (int b2 = 1; b1 + b2 <= k; b2++) {
					if (r % b2)
						continue;
					int d = r / b2;
					if (d < M && p[d] != -1)
						ans = min(ans, b1 + b2);
				}
			}
		}
		if (ans == 0x7fffffff)
			ans = -1;
		cout << ans << endl;
	}

	return 0;
}
