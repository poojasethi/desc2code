#include <cstdio>
#include <iostream>
#include <set>
using namespace std;


int n, k;
set<int > a;


int findN(const int x) {
	int ret = k + 1;
	for(set<int >::iterator it = a.begin(); it != a.end(); it++) {
		for(int i = 1; i <= k; i++) {
			int y = x - (*it) * i;
			if(0 == y) ret = min(ret, i);
			else {
				for(int j = 1; i + j <= k; j++)
					if((0 == y % j) && (a.find(y / j) != a.end())) {
						ret = min(ret, i + j);
						break;
					}
			}
		}
	}

	if(ret > k) ret = -1;

	return ret;
}


int main() {
//freopen("in.txt", "r", stdin);
	scanf("%d%d", &n, &k);
	for(int i = 0; i < n; i++) {
		int x;
		scanf("%d", &x);
		a.insert(x);
	}
	
	int q;
	scanf("%d", &q);
	for(int i = 0; i < q; i++) {
		int x;
		scanf("%d", &x);
		printf("%d\n", findN(x));
	}

	return 0;
}
