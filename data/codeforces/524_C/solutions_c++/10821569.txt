#include<iostream>
#include<cstdio>
#include<cstring>
#include<map>

using namespace std; 

map<int,int> mat; 
int a[5003]; 
int main() {
	
	int n,k; 
	scanf("%d%d",&n,&k); 
	for (int i=1;i<=n;i++)
		scanf("%d",&a[i]); 
	mat.clear(); 
	int q; 
	scanf("%d",&q); 
	for (int i=n;i>=1;i--)
		for (int j=1;j<=k;j++)
			if ( mat.find(a[i]*j)==mat.end() )
				mat[a[i]*j]=j;
	while ( q--){
		int x; 
		scanf("%d",&x); 
		int ans=1<<30; 
		for (int i=1;i<=n;i++)
			for ( int j=0;j<=k;j++)
			{
				if  ( mat.find(x-a[i]*j)!=mat.end() &&  j+mat[x-a[i]*j]<ans  && j+mat[x-a[i]*j]<=k )
					ans=j+mat[x-a[i]*j];
			}
		if ( ans==1<<30) 
			printf("-1\n");
		else printf("%d\n",ans); 
	}
	return 0; 
}
