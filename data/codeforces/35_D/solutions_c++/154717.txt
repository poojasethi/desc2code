#include<stdio.h>
#include<algorithm>
using namespace std;
int N[102],n,m;
int main()
{
FILE * fi = fopen("input.txt","r");
FILE * fo = fopen("output.txt","w");
 fscanf(fi,"%d%d",&n,&m);
 for(int i = 0 ; i < n; i++)
 {
  fscanf(fi,"%d",&N[i]);
  N[i] = N[i]*(n-i);
 }
 sort(N,N+n);
 int i;
 for(i=  0 ; i < n&& m >= N[i];i++)
  m-=N[i];
 fprintf(fo,"%d\n",i);
 return 0;
 }
