#include <fstream>
#include <algorithm>
using namespace std;
int main()
{
    ifstream cin("input.txt");
    ofstream cout("output.txt");
	int n,X,i,ans=0,c[10000];	
	cin >> n >> X;
	for (i=0;i<n;i++)
	{
		cin >> c[i];
        c[i]*=(n-i);
    }
    sort(c,c+n);
	for (i=0;i<n;i++)
        if(c[i]<=X)
		{
			X-=c[i];
            ans++;
        }
	cout << ans << endl;
    return 0;
}
