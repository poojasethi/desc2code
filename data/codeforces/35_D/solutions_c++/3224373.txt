#include<fstream>
#include<cstring>
#include<algorithm>
using namespace std;
const int mm=101000;
ifstream cin("input.txt");
ofstream cout("output.txt");
int f[mm];
int n,m;
int main()
{
  while(cin>>n>>m)
  {
    for(int i=n;i>=1;i--)
    {
      cin>>f[n-i];
      f[n-i]*=i;
    }
    sort(f,f+n);
    int ans=0;
    for(int i=0;i<n;i++)
      if(m-f[i]>=0)
    {
      ++ans;m-=f[i];
    }
    cout<<ans<<"\n";
  }
}
