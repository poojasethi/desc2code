#include<cstdio>
#include<algorithm>
using namespace std;
int c,s[101],N,X,rez;

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d%d",&N,&X);
	for (int i=1; i<=N; ++i)
	{
		scanf("%d",&c);
		s[i]=c*(N-i+1);
	}
	sort (s+1,s+1+N);
	c=0;
	for (int i=1; i<=N; ++i)
	{
		if (c+s[i]<=X)
		{
			++rez;
			c+=s[i];
		}
		else
		{
			printf("%d",rez);
			return 0;
		}
	}
	printf("%d",rez);
	return 0;
}
