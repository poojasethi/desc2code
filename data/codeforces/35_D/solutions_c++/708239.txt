#include<iostream>
#include<fstream>
#include<algorithm>

using namespace std;

int n, x;
int a[101];

int main() {
  ifstream fi("input.txt");
  ofstream fo("output.txt");
  fi >> n >> x; 
  for (int i = 0; i < n; i ++) {
    int t;
    fi >> t;
    a[i] = t * (n - i);
  }
  sort(a, a + n);
  int tsum = 0;
  int i = 0;
  int res = 0;
  while (i < n && tsum + a[i] <= x) {
    res ++;
    tsum += a[i++];
  }
  fo << res << endl;
  fi.close();
  fo.close();
  return 0;
}
