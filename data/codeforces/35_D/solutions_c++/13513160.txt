#include <fstream>
#include <algorithm>
using namespace std;
ifstream fin;
ofstream fout;
int main()
{
	fin.open("input.txt");
	fout.open("output.txt");
	int n,x,c[105] = {0},dp[10005] = {0};
	fin >> n >> x;
	for(int i = 1;i<=n;i++)
	{
		fin >> c[i];
		c[i]*=n+1-i;
	}
	for(int i = 1;i<=n;i++)
	for(int j = x;j>=0;j--)
	if(j>=c[i]) dp[j] = max(dp[j],1+dp[j-c[i]]);
	fout << dp[x] << endl;
	return 0;
}
