#include <cstdio>

#include <algorithm>
using std::sort;

int n, m;
int dat[ 105 ];

int main(){
	freopen( "input.txt", "r", stdin );
	freopen( "output.txt", "w", stdout );

	scanf( "%d%d", &n, &m );

	for( int i = 0; i < n; i++ ){
		scanf( "%d", &dat[ i ] );
		dat[ i ] *= n - i;
	}

	sort( dat, dat + n );

	int i;

	for( i = 0; i < n; i++ ){
		if( m >= dat[ i ] ){
			m -= dat[ i ];
		}
		else{
			break;
		}
	}

	printf( "%d\n", i );

	return 0;
}
