#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
using namespace std;
int a[200],b[40000];
int n,m;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	cin>>n>>m;
	for (int i=1;i<=n;++i)
	{
		cin>>a[i];a[i]=(n-i+1)*a[i];
	}
	b[0]=0;
	for (int i=1;i<=n;++i)
		for (int j=m;j>=a[i];--j)
			if (b[j-a[i]]+1>b[j]) b[j]=b[j-a[i]]+1;
	cout<<b[m]<<endl;
	return 0;
}
