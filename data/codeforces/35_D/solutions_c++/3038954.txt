#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int a[200];
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,x;
    scanf("%d%d",&n,&x);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
        a[i]*=(n-i);
    }
    sort(a,a+n);
    int ans=0;
    for(int i=0;i<n;i++)
    if(x>=a[i])
    {
        x-=a[i];
        ans++;
    }
    printf("%d\n",ans);
}
