#include <bits/stdc++.h>
using namespace std;
const int N = 105;
int a[N];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,m;
	while(~scanf("%d%d",&n,&m))
	{
		for(int i=0;i<n;++i)
			scanf("%d",&a[i]) , a[i]*=(n-i);
		sort(a,a+n);
		int i=0;
		while((m=m-a[i])>=0&&i<n) ++i;
		printf("%d\n",i);
	}
    return 0;
}
