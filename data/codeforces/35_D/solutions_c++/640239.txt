#include <cstdio>
#include <algorithm>

using namespace std;

int N, X;
int C[110];

int main() {
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d %d", &N, &X);
    for (int i = 0; i < N; i++) scanf("%d", C+i), C[i] = C[i] * (N-i);
    sort(C, C+N);
    int ret = 0, t = 0;
    while (ret < N && t + C[ret] <= X) t += C[ret], ret++;
    printf("%d\n", ret);
    return 0;
}
