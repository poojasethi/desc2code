#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <algorithm>
using namespace std;

int a[105];

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n, x;
	scanf("%d%d",&n,&x);
	for(int i = 0; i < n; ++i )
		scanf("%d",a+i), a[i] *= n-i;
	sort(a,a+n);
	int cnt = 0;
	for(int i = 0; i < n; ++i )
		if( x >= a[i] )
			x -= a[i], ++cnt;
	cout << cnt << endl;
	return 0;
}
