#include <cstdio>
FILE *in,*out;
int a[105],b[10050];
int main()
{
	int i,n,j,x,y,maxv;
	in=fopen("input.txt","r");
	out=fopen("output.txt","w");
	fscanf(in,"%d%d",&n,&x);
	for (i=1;i<=n;i++)
	{
		fscanf(in,"%d",&y);
		a[i]=(n-i+1)*y;
	}
	b[0]=1;
	for (i=1;i<=n;i++)
		for (j=x;j>=a[i];j--)
			if (b[j-a[i]]&&b[j-a[i]]+1>b[j])
				b[j]=b[j-a[i]]+1;
	maxv=0;
	for (i=0;i<=x;i++)
		if (b[i]>maxv)
			maxv=b[i];
	fprintf(out,"%d\n",maxv-1);
	fclose(in);
	fclose(out);
	return 0;
}
