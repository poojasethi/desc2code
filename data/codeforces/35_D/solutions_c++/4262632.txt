#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;  
int n,x,i,b,a[101],s;                                    
int main()
{
freopen ("input.txt","r",stdin);
freopen ("output.txt","w",stdout);
cin>>n>>x;
for (i=1;i<=n;i++)
{
cin>>b;
a[i]=b*(n-i+1);
} 
sort (a+1,a+n+1);
for (i=1;i<=n;i++)
{
	s+=a[i];
	if (s>x)
		break;
}
cout<<i-1;



return 0;
}
