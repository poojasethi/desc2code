#include <bits/stdc++.h>

using namespace std;

const int N = 101;
int a[N];

main() {
	 freopen("input.txt", "r", stdin), freopen("output.txt", "w", stdout);
	 int n, x;
	 cin >> n >> x;
	 for (int i = 0; i < n; i++)
	 	cin >> a[i], a[i] *= (n - i);
	sort(a, a + n);
	int ans = 0, sum = 0;
	for (int i = 0; i < n && sum + a[i] <= x; i++)
		ans++, sum += a[i];
	cout << ans << endl;
}
