#include <cstdio>
#include <algorithm>
using namespace std;
int ct[100];
int main()
{
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	int n, x, c;
	scanf("%d %d", &n, &x);
	for (int i = 0; i < n; ++i)
	{
		scanf("%d", &c);
		ct[i] = c * (n - i);
	}
	sort(ct, ct + n);
	int r, s;
	for (r = 0, s = 0; r < n; ++r)
	{
		s += ct[r];
		if (s > x) break;
	}
	printf("%d\n", r);
	return 0;
}
