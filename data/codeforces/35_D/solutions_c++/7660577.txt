#include<bits/stdc++.h>

using namespace std;

const int Maxn=100+10;

int n,c[Maxn],x,ans,t[Maxn];

ifstream fin("input.txt");
ofstream fout("output.txt");

inline void input()
{
	fin>>n>>x;
	for(int i=0;i<n;i++)
	{
		fin>>c[i];
		t[i]=c[i]*(n-i);
	}
}


inline void solve()
{
	sort(t,t+n);
	int sum=0;
	for(int i=0;i<n;i++)
	{
		sum+=t[i];
		if(sum>x)
			return ; 
			ans=i+1;
	}
}	

int main()
{
	input();
	solve();
	fout<<ans;
}
