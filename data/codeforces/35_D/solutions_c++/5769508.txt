#include<stdio.h>
#include<algorithm>
using namespace std;
    
const int N=101;
int a[N];
int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int i,j,n,m,x,k,t;
    scanf("%d%d",&n,&x);
    for(i=0;i<n;i++){
        scanf("%d",&t);
        a[i]=t*(n-i);
    }
    sort(a,a+n);
    for(i=0;i<n&&x>=0;i++){
        if(x-a[i]<0) break;
        x-=a[i];
    }
    printf("%d\n",i);
    return 0;
}
