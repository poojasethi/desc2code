#include<cstdio>
#include<algorithm>
using namespace std;
int c[102];
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,i,ans,x;
	while(~scanf("%d%d",&n,&x)){
		ans=0;
		for(i=1;i<=n;i++){
			scanf("%d",c+i);
			c[i]*=(n-i+1);
		}sort(c+1,c+n+1);
		for(i=1;i<=n;i++){
			if(x>=c[i])x-=c[i],ans++;
			else break;
		}
		printf("%d\n",ans);
	}
	return 0;
}
