#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
    int a[103],c[103],n,d,i,j;
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    while(~scanf("%d%d",&n,&d))
    {
        for(i=0;i<n;i++)
            scanf("%d",&a[i]),c[i]=(n-i)*a[i];
        sort(c,c+n);
        for(j=0,i=0;i<n && j+c[i]<=d;i++)
            j+=c[i];
        printf("%d\n",i);
    }
    return 0;
}
