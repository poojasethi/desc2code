#include <fstream>
#include <algorithm>
using namespace std;
int c[105],n,m,x;

int main()
{ifstream fin("input.txt");
 ofstream fout("output.txt");
  fin>>n>>x;
  int i;
   for(i=1;i<=n;++i)
    {fin>>c[i];c[i]*=n-i+1;}

  sort(c+1,c+n+1);
   for(i=1;i<=n;++i)
    {x-=c[i];
     if(x<0)
      break;
    }
   fout<<i-1<<'\n';
fin.close();
fout.close();
    return 0;
}
