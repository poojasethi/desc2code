#include<cstdio>
#include<iostream>
#include<queue>
#include<algorithm>
using namespace std;
int ani[105];
int main()
{
  int i,n,j,c,sum;
  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
  cin>>n>>sum;
  for(i=1;i<=n;i++)
  {
    cin>>c;
    ani[i]=c*(n-i+1);
  }
  sort(ani,ani+n+1);
  i=1;
  while(sum>=ani[i]&&i<=n)
    {
      sum-=ani[i];
      i++;
    }
  cout<<i-1<<endl;
  return 0;
}
