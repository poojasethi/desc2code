#include<iostream>
#include<algorithm>
#include<fstream>
using namespace std;

ifstream fin("input.txt");
ofstream fout("output.txt");

#define cin fin
#define cout fout
int n, m, ans = 0;
int a[10000];

int main(){
	cin >> n >> m;
	for(int i = 0; i < n; i++){
		cin >> a[i];
		a[i] *= (n-i);
	}
	sort(a, a+n);
	for(int i = 0; i < n; i++)
		if(a[i] <= m){
			m -= a[i];
			ans++;
		}
	cout << ans << endl;
	return 0;
}
