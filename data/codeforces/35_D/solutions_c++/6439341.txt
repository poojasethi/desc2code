#include <cstdio>
#include <algorithm> 
using namespace std;

int n , X , u , a[301] , ans , s ;

int main()
{
   freopen("input.txt","r",stdin) , freopen("output.txt","w",stdout) ;
   scanf("%d%d",&n,&X) ;
   for(int i=0 ; i<n && scanf("%d",&u) ; i++)  a[i]=u*(n-i) ;
   sort(a,a+n) ;

   for(int i=0 ; i<n ; ans++,i++) if(a[i]+s>X) break ; else s+=a[i] ; 
   printf("%d",ans) ;

 return 0;
}
