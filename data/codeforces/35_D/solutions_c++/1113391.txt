#include <cstdio>
#include <algorithm>

using namespace std;

int a[100];

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	int n, x;
	scanf("%i%i", &n, &x);
	for(int i = n; i > 0; i--) scanf("%i", &a[i - 1]);
	for(int i = n; i > 0; i--) a[i - 1] *= i;
	sort(a, a + n);
	int i = 0;
	for(int p = 0; i < n; i++) if((p += a[i]) > x) break;
	printf("%i\n", i);
}
