# include <bits/stdc++.h>
using namespace std;

int a[105];
int main()
{
	FILE *in = fopen("input.txt","r");
	FILE *out = fopen("output.txt","w");

	int n,X;
	fscanf(in,"%d %d",&n,&X);

	int i;
	for(i=0;i<n;i++)
	{
		fscanf(in,"%d",&a[i]);
		a[i]=(n-i)*a[i];
	}

	sort(a,a+n);

	int ans=0,temp=0;

	for(i=0;i<n;i++)
	{
		temp+=a[i];

		if(temp<=X) ans++;
		else break;
	}
	
	fprintf(out,"%d\n",ans);

	return 0;
}
