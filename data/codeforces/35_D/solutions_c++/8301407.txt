#include <bits/stdc++.h>
using namespace std;
const int maxn=111;
int n,x,i,an,a[maxn];
int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d%d",&n,&x);
    for(i=1;i<=n;++i)scanf("%d",&a[i]),a[i]*=(n-i+1);
    sort(a+1,a+n+1),an=0;
    for(i=1;i<=n;++i)if(x>=a[i])++an,x-=a[i];else break;
    printf("%d\n",an);
    return 0;
}
