#include <fstream>
#include <vector>
#include <algorithm>

int
main()
{
  std::ifstream input("input.txt");
  int n;
  int x;
  input >> n >> x;
  std::vector<int> f(n);
  for (int i(0); i < n; ++i) {
    int c;
    input >> c;
    f[i] = c * (n - i);
  }
  std::sort(f.begin(), f.end());
  int sum(0);
  int i(0);
  for (; i < n; ++i) {
    sum += f[i];
    if (sum > x) {
      break;
    }
  }
  std::ofstream output("output.txt");
  output << i << '\n';
  return 0;
}
