#include<bits/stdc++.h>
using namespace std;

int main(){
	FILE *in,*out;
	in = fopen("input.txt","r");
	out = fopen("output.txt","w");
	int n,x;
	fscanf(in,"%d%d",&n,&x);
	int c[101];
	for(int i=0;i<n;i++){
		fscanf(in,"%d",&c[i]);
		int a = n-i;
		c[i] *= a;
	}
	sort(c,c+n);
	int res=0;
	for(int i=0;i<n;i++){
		if(x>=c[i]){
			x -= c[i];
			res ++;
		} else
			break;
	}
	fprintf(out,"%d",res);
	return 0;
}
