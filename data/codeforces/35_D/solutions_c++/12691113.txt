#include<bits/stdc++.h>
using namespace std;
#define NN 101
ifstream f("input.txt");
ofstream g("output.txt");
int n, x, a[NN], i;
int main() {
    f >> n >> x;
    for(i = 1;i <= n; i++) f >> a[i], a[i] *= (n-i+1);
    sort(a+1,a+n+1);
    for(i = 1; a[i] <= x && i <= n; i++) x -= a[i];
    g << i-1;
    return 0;
}
