#include <stdio.h>
#include <algorithm>
using namespace std;
int dp[10005];
int main(void) {
	int n, X;
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d%d", &n, &X);
	for(int i=1;i<=n;i++) {
		int x;
		scanf("%d", &x);
		x *= (n+1-i);
		for(int j=X-x;j>=0;--j)
			dp[j+x] = max(dp[j+x], dp[j]+1);
	}
	int ans=0;
	for(int i=0;i<=X;i++)
		if(dp[i] > ans) ans = dp[i];
	printf("%d\n", ans);
	return 0;
}
