#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[110], n, x, ans;
int main()
{
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  cin >> n >> x;
  for(int i = 0; i < n; i++)
  {
    cin >> a[i];
    a[i] *= (n - i);
  }
  sort(a, a + n);
  for(int i = 0; i < n; i++)
    if (x - a[i] >= 0)
    {
      ans++;
      x -= a[i];
    }
    else
      break;
  cout << ans;
  return 0;
}
