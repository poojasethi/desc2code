inputs = file("input.txt","r")
outputs = file("output.txt","w")
ls = inputs.readline().split()
n,m = (int(i) for i in ls)
ls = [int(i) for i in inputs.readline().split()]
for i in range(len(ls)):
    ls[i] = ls[i] * ( n - i )
ls.sort()
ans = 0
s = 0
for i in range(len(ls)):
    if s + ls[i] > m:
        break
    s += ls[i]
    ans += 1
print >> outputs, ans
