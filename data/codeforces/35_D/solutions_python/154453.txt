def readint(): return int(raw_input())
def readarray(foo): return [foo(x) for x in raw_input().split()]


def run():
	(n, x) = readarray(int)
	a = readarray(int)
	b = []
	for i in xrange(n):
		b.append(a[i] * (n - i))
	b.sort()
	res = 0
	for i in xrange(n):
		if x < b[i]: break
		res += 1
		x -= b[i]
	print res
run()
