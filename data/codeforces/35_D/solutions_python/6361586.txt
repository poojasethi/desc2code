import sys
sys.stdin = open('input.txt','r')
sys.stdout = open('output.txt','w')

n , tot = map(int,raw_input().split())
a = map(int,raw_input().split())

for i in range(n):
	a[i] *= (n-i)

a.sort()

cnt = 0
for i in range(n):
	if tot < a[i]:
		break
	cnt += 1
	tot -= a[i]
print cnt
