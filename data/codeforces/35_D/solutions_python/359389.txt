import sys
import os
from os import popen

t=[]
f=open("input.txt")
c=map(lambda x:int(x),f.readline().split())
n=c[0]
X=c[1]
c=map(lambda x:int(x),f.readline().split())
for i in range(n):
	t.append((n-i)*c[i])
f.close()
t.sort()
summ=0
i=0
while(i<n and summ<X-t[i]+1):
	summ+=t[i]
	i+=1
print str(summ) +" obtained with "+ str(i)+" animals"
f=open("output.txt",'w')
f.write(str(i))
f.close()
