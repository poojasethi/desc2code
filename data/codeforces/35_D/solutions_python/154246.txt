import sys
fout=open('output.txt','w')
f=open('input.txt')

(n,x)=map(int,f.readline().split())
a=map(int,f.readline().split())
for i in xrange(n):
  a[i]*=(n-i)

a.sort()
for i in xrange(n):
  if x<a[i]:
    fout.write(str(i))
    fout.close()
    f.close()
    sys.exit()
  x-=a[i]

fout.write(str(n))
fout.close()
f.close()
