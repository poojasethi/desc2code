import sys

sys.stdin  = open ('input.txt',  'r')
sys.stdout = open ('output.txt', 'w')

def main():
	n, x = [int(it) for it in sys.stdin.readline().split()]
	c = [int(it) for it in sys.stdin.readline().split()]
	a = [0] * n
	for i in xrange (n):
		a[i] = c[i] * (n - i)
	#print c
	#print a
	a = sorted (a)
	sum = 0
	cnt = 0
	for i in xrange (n):
		if sum + a[i] <= x:
			sum += a[i]
			cnt += 1
	print cnt

main()
