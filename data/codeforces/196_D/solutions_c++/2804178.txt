#include<cstdio>
#include<cstring>

const int maxn=400010;
const int MC=1000000007;
int n,m;
char s[maxn],r[maxn];
int p[maxn],a[maxn],b[maxn];

bool chk(int x,int y){
  if(x<0)return 1;
  return (a[y]-a[x]*p[y-x])*p[x]!=b[y]-b[x];
}
bool dfs(int x,bool w){
  if(x>n)return puts(r+1),1;
  for(r[x]=w?s[x]:'a';r[x]<='z';++r[x]){
    a[x]=a[x-1]*MC+r[x],b[x]=b[x-1]+p[x-1]*r[x];
    if(chk(x-m,x)&&chk(x-m-1,x))
      if(dfs(x+1,w&&r[x]==s[x]))return 1;
  }
  return 0;
}

int main(){
  int i;
  scanf("%d\n%s",&m,s+1),n=strlen(s+1);
  for(i=n;i&&s[i]=='z';)s[i--]='a';
  if(!i)return puts("Impossible"),0;else++s[i];
  for(p[0]=1,i=1;i<=n;++i)p[i]=p[i-1]*MC;
  if(!dfs(1,1))puts("Impossible");
  return 0;
}
