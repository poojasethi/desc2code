#include <cstdio>
#include <cstring>
using namespace std;
const int P=13331,N=1001000;
int d,len,i,p[N],l[N],r[N];char st[N],s[N];
bool ck(int x,int y) {return (x>y)?1:l[y]*p[y-x]-l[y-x]*p[y]!=r[y]-r[y-x];}
bool dfs(int x,bool fg) {
	if (x==len+1) {s[x]='\0';return 1;}
	for (int i=fg?st[x-1]:'a';i<='z';i++) {
		l[x]=l[x-1]*P+i;r[x]=r[x-1]+i*p[x-1];s[x-1]=i;
		if (ck(d,x) && ck(d+1,x) && dfs(x+1,fg&&(i==st[x-1]))) return 1;
	}
	return 0;
}
int main() {
	scanf("%d%s",&d,&st);len=strlen(st);
	for (p[0]=i=1;i<=len;i++) p[i]=p[i-1]*P;
	for (i=len-1;i>=0 && st[i]=='z';i--) st[i]='a';
	if (i<0) {puts("Impossible");return 0;}st[i]++;
	if (dfs(1,1)) puts(s); else puts("Impossible");
}
