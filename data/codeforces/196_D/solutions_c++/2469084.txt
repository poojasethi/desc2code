#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
const int op=13331,N=400100;
int d,n;
char s[N],ans[N];
int a[N],A[N],B[N];
bool ok(const int x,const int y)
{
	return x<y || B[x]-B[x-y]!=(A[x]-A[x-y]*a[y])*a[x-y];
}
bool allz()
{
	int i=n;
	while ( i>0 && s[i]=='z' ) s[i--]='a';
	++s[i];
	return i==0;
}
bool find(const int x,bool c)
{
	if ( x>n ) return 1;
	for(int i=c?s[x]:'a';i<='z';++i)
	{
		ans[x]=i;
		A[x]=A[x-1]*op+i,B[x]=B[x-1]+i*a[x-1];
		if ( ok(x,d) && ok(x,d+1) && find(x+1,c&&(i==s[x])) ) return 1;			
	}
	return 0;
}
int main()
{
	scanf("%d",&d);
	scanf("%s",s+1);
	n=strlen(s+1);
	a[0]=1;
	for(int i=1;i<=n;++i) a[i]=a[i-1]*op;
	A[0]=B[0]=0;
	if ( allz() || !find(1,1) ) puts("Impossible");
	else puts(ans+1);	
	return 0;
}
