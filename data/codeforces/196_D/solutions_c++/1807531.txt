#include<cstdio>
#include<iostream>
#include<cstring>
#include<string>
#include<algorithm>
#define N 400009
#define P 1000000007
using namespace std;
char s[N],r[N];
int p[N],h[N],rh[N];
int n,m;
int ok(int i,int x){
    if(++i<x) return 1;
    return ((rh[i]-rh[i-x]*p[x])*p[i-x])!=(h[i]-h[i-x]);
}
int go(int d,int t){
    if(d==n) return puts(r),1;
    for(r[d]=(t?s[d]:'a');r[d]<='z';r[d]++){
        h[d+1]=h[d]+r[d]*p[d];
        rh[d+1]=rh[d]*P+r[d];
        if(ok(d,m)&&ok(d,m+1)&&go(d+1,t&&(r[d]==s[d]))) return 1;
    }
    return 0;
}
int main(){
    scanf("%d%s",&m,s);
    int i=(n=strlen(s))-1;
    for(;i>=0&&s[i]=='z';i--) s[i]='a';    
    if(i<0) return puts("Impossible"),0;
    s[i]++;
    p[0]=1;
    for(int i=1;i<N;i++) p[i]=p[i-1]*P;
    if(!go(0,1)) puts("Impossible");
    return 0;
}
