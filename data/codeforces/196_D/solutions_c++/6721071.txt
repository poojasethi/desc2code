#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define N 1000010
using namespace std;

int P = 1000007;
int ex[N] , len , m ,  h[N] , ni_h[N];
char st[N],ans[N];
inline bool ok(int x , int mm)
{
    if (mm > x) return 1;
    return
	(h[x] - h[x - mm]) != (ni_h[x] - ni_h[x - mm] * ex[mm]) * ex[x-mm+1];
}
inline int dfs(int x,bool flag)
{
    if (x == len + 1) return 1;
    char i = flag?'a':st[x];
    for (;i <= 'z';i++)
	{
	    h[x] = h[x - 1] + i * ex[x]; 
	    ni_h[x] = ni_h[x - 1] * P + i;
	    ans[x] = i;
	    if (ok(x,m) && ok(x,m + 1) && dfs(x + 1 , flag || (i > st[x])))
		return 1;
	}
    return 0;
}

int main()
{
    scanf("%d",&m);
    scanf("%s",st + 1); len = strlen(st + 1);
    int i;
    for (i = len;i && st[i] == 'z';i--) st[i] = 'a';
    if (i == 0) return printf("Impossible") , 0; else st[i] ++;
    ex[0] = 1;
    for (int i = 0;i < len;i++)	ex[i + 1] = ex[i] * P;
    if (dfs(1,0)) puts(ans + 1); else printf("Impossible");

    return 0;
}
