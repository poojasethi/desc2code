#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int dp[1000+2][1000+2];
char s[1000+2],t[1000+2];
int main()
{
	cin>>s>>t;
	int sl=strlen(s),tl=strlen(t);
	for(int i=0;i<=tl;i++)	dp[0][i]=i;
	for(int i=0;i<=sl;i++)	dp[i][0]=i;
	for(int i=1;i<=sl;i++)
	{
			for(int j=1;j<=tl;j++)
			{
				if(s[i-1]==t[j-1])
				dp[i][j]=dp[i-1][j-1];
				else
				dp[i][j]=min(dp[i][j-1],min(dp[i-1][j],dp[i-1][j-1]))+1;
			}
	}
/*	for(int i=1;i<=sl;i++)
	{
			for(int j=1;j<=tl;j++)
			{
				cout<<dp[i][j]<<" ";
			}
			cout<<endl;
	}*/
	cout<<dp[sl][tl]<<endl;
	int i=sl,j=tl;
	while(i||j)
	{
		if(dp[i][j]==dp[i-1][j]+1)
		{
			printf("DELETE %d\n",i);
			i--;
		}
		else if(dp[i][j]==dp[i][j-1]+1)
		{
			printf("INSERT %d %c\n",i+1,t[j-1]);
			j--;
		}
		
		else if(dp[i][j]==dp[i-1][j-1]+1)
		{
			printf("REPLACE %d %c\n",i,t[j-1]);
			i--;
			j--;
		}
		else if(dp[i][j]==dp[i-1][j-1])
		{
			i--;
			j--;
		}
		
	}
	return 0;
}
