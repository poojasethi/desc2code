using namespace std;

int dp[1001][1001];

int main()
{
	char s[1001],t[1001];
	while(~scanf("%s%s",s+1,t+1)){
		int sl=strlen(s+1);
		int tl=strlen(t+1);
		memset(dp,0,sizeof(dp));
		for(int i=0;i<=tl;i++)dp[0][i]=i;
		for(int i=0;i<=sl;i++)dp[i][0]=i;
		for(int i=1;i<=sl;i++){
			for(int j=1;j<=tl;j++){
				if(s[i]==t[j]){
					dp[i][j]=dp[i-1][j-1];
				}
				else {
					dp[i][j]=min(dp[i-1][j],dp[i][j-1]);
					dp[i][j]=min(dp[i][j],dp[i-1][j-1])+1;
				}
			}
		}
//		for(int i=0;i<=sl+1;i++){
//			for(int j=0;j<=tl+1;j++){
//				printf("%2d ",dp[i][j]);
//			}
//			puts("");
//		}
		printf("%d\n",dp[sl][tl]);
		int x=0,y=0;
		while(sl>0||tl>0){
			if(sl>0&&tl>0&&dp[sl][tl]-1==dp[sl-1][tl-1]){
				printf("REPLACE %d %c\n",sl,t[tl]);
				sl--,tl--;
			}
			else if(sl>0&&dp[sl][tl]-1==dp[sl-1][tl]){
				printf("DELETE %d\n",sl);
				sl--;
			}
			else if(tl>0&&dp[sl][tl]-1==dp[sl][tl-1]){
				printf("INSERT %d %c\n",sl+1,t[tl]);
				tl--;
			}
			else {
				sl--,tl--;
			}
		}
	}
	return 0;
}
