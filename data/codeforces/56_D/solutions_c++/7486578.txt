const int maxn = 1005;
const int inf = 1<<29;
const int mod = 1000000007;
const double eps = 1e-8;

char sa[maxn],sb[maxn];
int la,lb,dp[maxn][maxn];

int main()
{
    scanf("%s%s",sa,sb);la = strlen(sa),lb = strlen(sb);
    for(int i=1;i<=lb;i++)
        dp[0][i] = i;
    for(int i=1;i<=la;i++)
        dp[i][0] = i;
    for(int i=1;i<=la;i++)
        for(int j=1;j<=lb;j++)
            dp[i][j] = min(dp[i-1][j-1]+(sa[i-1]!=sb[j-1]),1+min(dp[i][j-1],dp[i-1][j]));
    printf("%d\n",dp[la][lb]);
    while(la||lb)
    {
        if(lb&&dp[la][lb-1]+1==dp[la][lb])
        {
            printf("INSERT %d %c\n",la+1,sb[lb-1]);lb--;
        }
        else if(la&&dp[la-1][lb]+1==dp[la][lb])
        {
            printf("DELETE %d\n",la);la--;
        }
        else
        {
            if(sa[la-1]!=sb[lb-1])
                printf("REPLACE %d %c\n",la,sb[lb-1]);
            la--;lb--;
        }
    }
}
