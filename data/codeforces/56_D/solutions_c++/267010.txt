#include <iostream>
#include <cassert>
#include <string>

using namespace std;

int main() {
    string a, b;
    cin >> a >> b;

    int dp[a.size()+1][b.size()+1];
    for(int i = 0; i <= b.size(); i++) dp[0][i] = i;
    for(int i = 1; i <= a.size(); i++) {
        dp[i][0] = i;
        for(int j = 1; j <= b.size(); j++)
            dp[i][j] = min((a[i-1] != b[j-1]) + dp[i-1][j-1],
                           1 + min(dp[i][j-1], dp[i-1][j]));
    }

    int pa = a.size(), pb = b.size();
    cout << dp[pa][pb] << endl;

    while(pa > 0 || pb > 0)
        if(pa != 0 && dp[pa][pb] == 1 + dp[pa-1][pb])
            cout << "DELETE " << pa-- << endl;
        else if(pb != 0 && dp[pa][pb] == 1 + dp[pa][pb-1])
            cout << "INSERT " << pa+1 << " " << b[--pb] << endl;
        else if(a[--pa] != b[--pb])
                cout << "REPLACE " << pa+1 << " " << b[pb] << endl;
}
