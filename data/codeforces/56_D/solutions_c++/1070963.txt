#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<string>
#include<iostream>
#include<vector>
using namespace std;

char s[2005],t[2005];
int dp[2005][2005];
int main()
{
	while (scanf("%s%s",s+1,t+1)>0)
	{
		int len1=strlen(s+1),len2=strlen(t+1);
		memset(dp,127,sizeof(dp));
		dp[0][0]=0;
		for (int i=1;i<=len1;i++) dp[i][0]=i;
		for (int i=1;i<=len2;i++) dp[0][i]=i;
		for (int i=1;i<=len1;i++)
			for (int j=1;j<=len2;j++)
			{
				if (s[i]==t[j]) dp[i][j]=min(dp[i][j],dp[i-1][j-1]);
				else dp[i][j]=min(dp[i][j],dp[i-1][j-1]+1);
				dp[i][j]=min(dp[i][j],min(dp[i-1][j]+1,dp[i][j-1]+1));
			}
		printf("%d\n",dp[len1][len2]);
		int i=len1,j=len2;
		while (true)
		{
			if (dp[i][j]==dp[i-1][j]+1 && i>=1)
			{
				printf("DELETE %d\n",i);
				i--;
			}
			else if (dp[i][j]==dp[i][j-1]+1 && j>=1)
			{
				printf("INSERT %d %c\n",i+1,t[j]);
				j--;
			}
			else if (dp[i-1][j-1]+1==dp[i][j])
			{
				printf("REPLACE %d %c\n",i,t[j]);
				i--;j--;
			}
			else
			{
				i--;j--;
			}
			//printf("%d %d\n",i,j);
			if (i==0 && j==0) break;
		}
	}
}
