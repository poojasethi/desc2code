#include <algorithm>
#include <cstring>
#include <cstdio>
int f[1010][1010],i,j,n,m;
char a[1010],b[1010];
void work(int x,int y)
{
	if (!x && !y) return;
	if (y && f[x][y]==f[x][y-1]+1)
	{
		work(x,y-1);
		printf("INSERT %d %c\n",y,b[y]);
	}
	else if (x && f[x][y]==f[x-1][y]+1)
	{
		work(x-1,y);
		printf("DELETE %d\n",y+1);
	}
	else
	{
		work(x-1,y-1);
		if (a[x]!=b[y]) printf("REPLACE %d %c\n",y,b[y]);
	}
}
int main()
{
	scanf("%s\n%s\n",a+1,b+1);
	n=strlen(a+1);m=strlen(b+1);
	for (i=1;i<=n;++i) f[i][0]=i;
	for (i=1;i<=m;++i) f[0][i]=i;
	for (i=1;i<=n;++i)
		for (j=1;j<=m;++j)
			f[i][j]=std::min(f[i][j-1]+1,std::min(f[i-1][j]+1,f[i-1][j-1]+(a[i]!=b[j])));
	printf("%d\n",f[n][m]);
	work(n,m);
}
