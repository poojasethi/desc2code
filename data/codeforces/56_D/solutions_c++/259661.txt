#include<cstdio>
#include<cstring>
int f[1013][1013],g[1013][1013][2];
//        Last Change:  2011-01-22 21:02:27
char s1[1013],s2[1013];
inline void out(int x,int y)
{
	if(x==0||y==0)
	{
        if(y==0)                 
		    for(int i=1;i<=x;++i)
			    puts("DELETE 1");
        if(x==0)
            for(int i=1;i<=y;++i)
                printf("INSERT %d %c\n",i,s2[i]);
		return;
	}
	int xx=g[x][y][0],yy=g[x][y][1];
	out(xx,yy);
	if(xx==x-1&&yy==y-1)
		if(s1[x]!=s2[y])
			printf("REPLACE %d %c\n",y,s2[y]);
	if(xx==x-1&&yy==y)
		printf("DELETE %d\n",y+1);
	if(y==163)
	    y=163;
	if(xx==x&&yy==y-1)
		printf("INSERT %d %c\n",y,s2[y]);
}
int main()
{
	scanf("%s",s1+1);
	scanf("%s",s2+1);
	int len1=strlen(s1+1);
	int len2=strlen(s2+1);
	memset(f,44,sizeof(f));
	for(int i=0;i<=len2;++i)
	    f[0][i]=i;
	for(int i=0;i<=len1;++i)
		f[i][0]=i;
	for(int i=1;i<=len1;++i)
		for(int j=1;j<=len2;++j)
		{
			int &x=f[i][j];
			if(x>f[i-1][j-1]+(s1[i]!=s2[j]))
			{
				x=f[i-1][j-1]+(s1[i]!=s2[j]);
				g[i][j][0]=i-1,g[i][j][1]=j-1;
			}
			if(x>f[i-1][j]+1)
			{
				x=f[i-1][j]+1;
				g[i][j][0]=i-1,g[i][j][1]=j;
			}
			if(x>f[i][j-1]+1)
			{
				x=f[i][j-1]+1;
				g[i][j][0]=i,g[i][j][1]=j-1;
			}
		}
	printf("%d\n",f[len1][len2]);
	out(len1,len2);
	return 0;
}
