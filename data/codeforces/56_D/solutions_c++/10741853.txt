#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<stdlib.h>
using namespace std;
int n,m;
char s1[2001];
char s2[2001];
int dp[2001][2001];



int len1,len2;



int solve(){
	for(int i=0;i<=len1;i++)dp[i][0]=i;
	for(int i=0;i<=len2;i++)dp[0][i]=i;
	
	for(int i=1;i<=len1;i++){
		for(int j=1;j<=len2;j++){
			if(s1[i]==s2[j])dp[i][j]=dp[i-1][j-1];
			else dp[i][j]=min(dp[i-1][j],min(dp[i][j-1],dp[i-1][j-1]))+1;
		}
	}
	return dp[len1][len2];
}
void dfs(int a, int b){
  if(a == 0 && b==0)return ;
  if(s1[a] == s2[b] && dp[a-1][b-1] == dp[a][b])
    dfs(a-1, b-1);
  else
  {
    if(a && dp[a-1][b] +1 == dp[a][b])
    {
      printf("DELETE %d\n", a);
      dfs(a-1, b);
    }
    else if(b && dp[a][b-1] +1 == dp[a][b])
    {
	  printf("INSERT %d %c\n", a +1, s2[b]);
      dfs(a, b-1);
    }
    else if(a && b && dp[a-1][b-1] +1 == dp[a][b])
    {
      printf("REPLACE %d %c\n", a, s2[b]);
      dfs(a-1, b-1);
    }
  }
}
main(){
	scanf("%s%s",s1+1,s2+1);
	s1[0]='*';
	s2[0]='*';
	len1=strlen(s1);
	len2=strlen(s2);
	printf("%d\n",solve());
	dfs(len1,len2);
	
}
