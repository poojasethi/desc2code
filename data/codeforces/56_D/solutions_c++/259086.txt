#include <map>
#include <cmath>
#include <stack>
#include <queue>
#include <string>
#include <vector>
#include <fstream>
#include <sstream>
#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;
#define li		long long
#define rep(i,to)	for(li i=0;i<(li)to;i++)
#define pb(a)		push_back(a)
#define sz(v)		((li)v.size())
#define Min(res,a)	{res=(res==-1)?(a):min(res,a);}
int main(){

	int dp[1005][1005];
	rep(i,1005)rep(j,1005)	dp[i][j]=-1;
	dp[0][0]=0;
	string a,b;
	cin>>a>>b;
	a="a"+a;
	b="b"+b;
	rep(i,sz(a))rep(j,sz(b)){
		if(i>0 && j>0)	Min(dp[i][j],dp[i-1][j-1]+1);
		if(i>0 && j>0 && a[i]==b[j]) Min(dp[i][j],dp[i-1][j-1]);
		if(i>0)	Min(dp[i][j],dp[i-1][j]+1);
		if(j>0) Min(dp[i][j],dp[i][j-1]+1);
	}
	li x=sz(a)-1,y=sz(b)-1;
	cout<<dp[x][y]<<endl;
	while(x!=0 || y!=0){
		if(x==0){
			cout<<"INSERT "<<x+1<<" "<<b[y--]<<endl;
		}else if(y==0){
			cout<<"DELETE "<<x--<<endl;
		}else if(dp[x][y]==dp[x-1][y-1]+1){
			cout<<"REPLACE "<<x--<<" "<<b[y--]<<endl;
		}else if(dp[x][y]==dp[x-1][y-1] && a[x]==b[y]){
			x--;y--;
		}else if(dp[x][y]==dp[x][y-1]+1){
			cout<<"INSERT "<<x+1<<" "<<b[y--]<<endl;
		}else if(dp[x][y]==dp[x-1][y]+1){
			cout<<"DELETE "<<x--<<endl;
		}
	}
}
