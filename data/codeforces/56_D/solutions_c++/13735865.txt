#include <bits/stdc++.h>
#define fi first
#define se second
#define pb push_back
#define m ((b+s)>>1)
#define lb lower_bound
#define ub upper_bound
#define umax(a,b) a=max(a,b)
#define umin(a,b) a=min(a,b)

using namespace std;

typedef long long lint;
typedef pair <int,int> ii;
typedef vector <int> vi;
typedef vector <ii> vii;

const int maxn=1010;

int N,M,dn[maxn][maxn];
char ar[maxn],er[maxn];

int cal(int a,int b){
	if(a==N+1 && b==M+1) return 0;
	
	int &x=dn[a][b];

	if(x!=-1) return x;
	
	if(ar[a]==er[b]) return dn[a][b]=cal(a+1,b+1);
	
	x=1e9;
	
	if(a<=N)
		umin(x,cal(a+1,b)+1);
	
	if(b<=M && a<=N)
		umin(x,cal(a+1,b+1)+1);
	
	if(b<=M)
		umin(x,cal(a,b+1)+1);
	
	return x;
}

void write(int a,int b){
	if(ar[a]==er[b]) write(a+1,b+1);

	if(a==N+1 && b==M+1) exit(0);
	
	int &x=dn[a][b];
	
	if(a<=N && x-cal(a+1,b)==1)
		printf("INSERT %d %c\n",a,ar[a]) , write(a+1,b);
	
	else if(b<=M && a<=N && x-cal(a+1,b+1)==1)
		printf("REPLACE %d %c\n",a,ar[a]) , write(a+1,b+1);
	
	else if(b<=M && x-cal(a,b+1)==1)
		printf("DELETE %d\n",a) , write(a,b+1);
}

int main(){
	scanf(" %s %s",er+1,ar+1);
	N=strlen(ar+1);
	M=strlen(er+1);
	
	memset(dn,-1,sizeof dn);
	printf("%d\n",cal(1,1));
	
	write(1,1);
	
	return 0;
}
