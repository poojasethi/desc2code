#include <iostream>
#include <cstdio>
#include <vector>
#include <cstring>
#include <algorithm>
#define maxn 1009
using namespace std;
char s1[maxn],s2[maxn];
int dp[maxn][maxn];
void dfs(int u,int v)
{
	if(u==0&&v==0)
		return;
	if(u&&v&&s1[u]==s2[v]&&dp[u][v]==dp[u-1][v-1])
		dfs(u-1,v-1);
	else
	{
		if(u&&dp[u-1][v]+1==dp[u][v])
		{
			printf("DELETE %d\n",u);
			dfs(u-1,v);
		}
		else if(v&&dp[u][v-1]+1==dp[u][v])
		{
			printf("INSERT %d %c\n",u+1,s2[v]);
			dfs(u,v-1);
		}
		else if(u&&v&&dp[u][v]==dp[u-1][v-1]+1)
		{
			printf("REPLACE %d %c\n",u,s2[v]);
			dfs(u-1,v-1);
		}
	}
}
int main()
{	
	scanf("%s%s",s1+1,s2+1);
	int n=strlen(s1+1),m=strlen(s2+1);
	for(int i=1;i<=n;i++)
		dp[i][0]=i;
	for(int i=1;i<=m;i++)
		dp[0][i]=i;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			dp[i][j]=min(min(dp[i-1][j]+1,dp[i][j-1]+1),dp[i-1][j-1]+(s1[i]!=s2[j]));
		}
	}
	printf("%d\n",dp[n][m]);
	dfs(n,m);
	//system("pause");
	return 0;
}
