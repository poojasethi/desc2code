#include<bits/stdc++.h>
using namespace std;
typedef  unsigned long long ll;
#define s(n) scanf("%llu",&n)
#define p(n) printf("%llu ",n)
 int  dp[1002][1002];
int main()
{string s,t;
cin>>s>>t;int i,j;
  for(i=0;i<=s.size();++i)
  for(j=0;j<=t.size();++j)
  {
     if(i==0) dp[i][j]=j;
    else if(j==0) dp[i][j]=i; 
    else if(s[i-1]==t[j-1] ) dp[i][j]=dp[i-1][j-1]; 
    else dp[i][j]= min( min(dp[i-1][j-1],dp[i-1][j]),dp[i][j-1])+1;
  }    
          
cout<<dp[s.size()][t.size()]<<"\n";

i=s.size();j=t.size();
while(i||j)
{
     if(dp[i][j]== dp[i-1][j]+1)
     {  cout<<"DELETE "<<i<<"\n";--i;}
     else
     if(dp[i][j]== dp[i][j-1]+1)
     {cout<<"INSERT "<<i+1<<" "<<t[j-1]<<"\n"; --j;}
     else if(dp[i][j]== dp[i-1][j-1]+1)
     {cout<<"REPLACE "<<i<<" "<<t[j-1]<<"\n";--i;--j;}
     else if(dp[i][j]== dp[i-1][j-1])
     {--i;--j;}
}  
}
