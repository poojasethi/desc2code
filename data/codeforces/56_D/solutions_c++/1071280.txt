#include<cmath>
#include<cstdio>
#include<string>
#include<vector>
#include<cstring>
#include<iostream>
#include<queue>
#include<algorithm>
#include<map>
using namespace std;
#define ll __int64
const int oo=100000;
int dp[1005][1005],n,m;
char s[1005],t[1005];
int main()
{
    gets(s+1);gets(t+1);
    n=strlen(s+1);m=strlen(t+1);
    for(int i=0;i<=n;i++)
    for(int j=0;j<=m;j++)
    dp[i][j]=oo;
    for(int i=0;i<=n;i++)dp[i][0]=i;
    for(int j=0;j<=m;j++)dp[0][j]=j;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            if(s[i]==t[j])dp[i][j]=dp[i-1][j-1];
            else dp[i][j]=min(dp[i][j],dp[i-1][j-1]+1);
            dp[i][j]=min(dp[i][j],min(dp[i-1][j]+1,dp[i][j-1]+1));
            //printf("dp[%d][%d]=%d\n",i,j,dp[i][j]);
        }
    }
    printf("%d\n",dp[n][m]);
    int x=n,y=m;
    while(dp[x][y])
    {
        //printf("x=%d,y=%d\n",x,y);
        if(x&&dp[x-1][y]+1==dp[x][y])
        {
            printf("DELETE %d\n",x);
            x--;
        }
        else if(y&&dp[x][y-1]+1==dp[x][y])
        {
            printf("INSERT %d %c\n",x+1,t[y]);
            y--;
        }
        else if(dp[x-1][y-1]+1==dp[x][y])
        {
            printf("REPLACE %d %c\n",x,t[y]);
            x--;y--;
        }
        else
        {
            x--;y--;
        }
    }
}
