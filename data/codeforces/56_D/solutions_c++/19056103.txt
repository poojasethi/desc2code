#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cmath>
#include<cstring>
using namespace std;
int n,m,i,j,l,f[1001][1001];
char a[1001],b[1001];
void go(int x,int y)
{
	if (!x && !y) return;
	if (y && f[x][y]==f[x][y-1]+1)
	{
		go(x,y-1);
		printf("INSERT %d %c\n",x+l+1,b[y]);
		l++;
	}
	else if (x && f[x][y]==f[x-1][y]+1)
	{
		go(x-1,y);
		printf("DELETE %d\n",x+l);
		l--;
	}
	else
	{
		go(x-1,y-1);
		if (a[x]!=b[y]) printf("REPLACE %d %c\n",x+l,b[y]);
	}
}

int main()
{
		scanf("%s\n%s",a+1,b+1);
		n=strlen(a+1);
		m=strlen(b+1);
		for (i=1;i<=m;i++) f[0][i]=i;
		for (i=1;i<=n;i++) f[i][0]=i;
		for (i=1;i<=n;i++)
			for (j=1;j<=m;j++)
				f[i][j]=min(min(f[i][j-1],f[i-1][j])+1,a[i]==b[j]?f[i-1][j-1]:f[i-1][j-1]+1);
		printf("%d\n",f[n][m]);
		go(n,m);
}
