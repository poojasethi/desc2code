#include<cstdio>
#include<algorithm>
#include<iostream>

#include<string>
#define MAX 1111
using namespace std;
char A[MAX],B[MAX];
int L[MAX][MAX];

int compute_edist(char X[MAX], char Y[MAX],int n,int m) /* X,Y are input strings. */
{
	int i,j;
	for(i=0;i<=n;i++)
		L[i][0] = i; //deletion
	for(j=0;j<=m;j++)
		L[0][j] = j; //insertion
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			if(X[i] == Y[j])
				L[i][j] = L[i-1][j-1];
			else
			{
				L[i][j] = min(L[i-1][j]+1, min(L[i][j-1]+1, L[i-1][j-1]+1));
			}
		}
	}
	int ret = L[n][m];
	return L[n][m];
}

void print_operations(char X[],char Y[],int n,int m)
{
	int i=n,j=m;
	while(i>0||j>0)
	{
		if(L[i-1][j-1]+1 == L[i][j]&&(i>0&&j>0))
		{
			printf("REPLACE %d %c\n",i,Y[j]);		
			i--;j--;	
		}
		else if(L[i][j-1]+1 == L[i][j]&&(j>0))
		{
			printf("INSERT %d %c\n",i+1,Y[j]);
			j--;
		}
		else if(L[i-1][j]+1 == L[i][j]&&(i>0))
		{
			printf("DELETE %d\n",i,Y[j]);
			i--;
		}
		else if(X[i]==Y[j])
		{
			i--;j--;
		}
	}
}
