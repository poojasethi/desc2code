#include<iostream>
#include<cstring>
#include<cstdio>
#include<cctype>

using namespace std;

const int maxn = 1010;
const int Inf = 1000000000;

int n , m;
char s[maxn] , t[maxn];
int f[maxn][maxn];

void input() {
	scanf( "%s" , s + 1 ); n = strlen( s + 1 );
	scanf( "%s" , t + 1 ); m = strlen( t + 1 );
}

void solve() {
	for ( int i = 0 ; i <= n ; i++ )
		for ( int j = 0 ; j <= m ; j++ ) {
			if ( i == 0 && j == 0 ) continue;
			f[i][j] = Inf;
			if ( i >= 1 && j >= 1 && s[i] == t[j] ) f[i][j] = min( f[i][j] , f[i - 1][j - 1] );
			if ( i >= 1 ) f[i][j] = min( f[i][j] , f[i - 1][j] + 1 );
			if ( j >= 1 ) f[i][j] = min( f[i][j] , f[i][j - 1] + 1 );
			if ( i >= 1 && j >= 1 ) f[i][j] = min( f[i][j] , f[i - 1][j - 1] + 1 );
		}
	printf( "%d\n" , f[n][m] );
	int i = n , j = m;
	while ( i != 0 || j != 0 ) {
		if ( i >= 1 && j >= 1 && s[i] == t[j] && f[i][j] == f[i - 1][j - 1] ) i-- , j--; else
		if ( i >= 1 && f[i][j] == f[i - 1][j] + 1 ) {
			printf( "DELETE %d\n" , i );
			i--;
		} else
		if ( j >= 1 && f[i][j] == f[i][j - 1] + 1 ) {
			printf( "INSERT %d %c\n" , i + 1 , t[j] );
			j--;
		} else {
			printf( "REPLACE %d %c\n" , i , t[j] );
			i-- , j--;
		}
	}
}

int main() {
	input();
	solve();
	
	return 0;
}
