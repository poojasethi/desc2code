#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
char a[1010],b[1010];
int dp[1010][1010];
int main()
{
	while(~scanf("%s%s",a,b))
	{
		memset(dp,0x3f3f3f3f,sizeof(dp));
		for(int i=0;i<=strlen(a);i++)
			dp[0][i]=i;
		for(int i=1;i<=strlen(b);i++)
			dp[i][0]=i;
		for(int i=1;i<=strlen(b);i++)
		{
			for(int j=1;j<=strlen(a);j++)
			{
				dp[i][j]=min(dp[i][j],min(dp[i-1][j],dp[i][j-1])+1);
				if(b[i-1]==a[j-1])
					dp[i][j]=min(dp[i][j],dp[i-1][j-1]);
				else
					dp[i][j]=min(dp[i][j],dp[i-1][j-1]+1);
			}
		}
		printf("%d\n",dp[strlen(b)][strlen(a)]);
		int i=strlen(b);int j=strlen(a);
		while(i||j)
		{	if(i>0&&j>0&&dp[i-1][j-1]==dp[i][j]&&a[j-1]==b[i-1])
			{
				i--;
				j--;
			}
			if(i>0&&dp[i-1][j]==dp[i][j]-1)
			{
				i--;
				printf("INSERT %d %c\n",j+1,b[i]);
			}else if(i>0&&j>0&&dp[i-1][j-1]==dp[i][j]-1)
			{
				i--;
				j--;
				printf("REPLACE %d %c\n",j+1,b[i]);
			}
			else if(j>0&&dp[i][j-1]==dp[i][j]-1)
			{
				j--;
				printf("DELETE %d\n",j+1);
			}
			
			
		}
	}
}
