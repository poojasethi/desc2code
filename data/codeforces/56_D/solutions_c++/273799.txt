#include<iostream>
#include<string>
using namespace std;
int dp[1010][1010],l1,l2;
string s1,s2;
int main()
{
	cin>>s1>>s2;
	l1=s1.length();
	l2=s2.length();
	for(int i=0;i<=l2;i++)
		dp[0][i]=i;
	for(int i=0;i<=l1;i++)
		dp[i][0]=i;
	for(int i=1;i<=l1;i++)
		for(int j=1;j<=l2;j++)
			dp[i][j]=min((s1[i-1]!=s2[j-1])+dp[i-1][j-1], 1+min(dp[i][j-1],dp[i-1][j]));
	cout<<dp[l1][l2]<<endl;
	while(l1>0 || l2>0)
	{
		if(dp[l1][l2]==1+dp[l1][l2-1]&& l2!=0)
			cout << "INSERT " << l1+1 << " " << s2[--l2] << endl;
		else if(dp[l1][l2]==1+dp[l1-1][l2] && l1!=0)
			cout<<"DELETE "<<l1--<<endl;
		else if(s1[--l1] != s2[--l2])
			cout << "REPLACE " << l1+1 << " " << s2[l2] << endl;
	}
	return 0;
}
