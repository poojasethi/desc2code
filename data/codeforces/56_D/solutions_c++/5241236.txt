#include<cstdio>
#include<algorithm>
using namespace std;
int dp[1002][1002],l1,l2;
char s1[1002],s2[1002];
int main(){
    scanf("%s%s",s1+1,s2+1);
    for(l1 = 1;s1[l1];l1++) ;
    for(l2 = 1;s2[l2];l2++) ;
    for(int i=0;i<l1;i++)
        fill(dp[i],dp[i]+l2,1e9);
    dp[0][0] = 0;
    for(int i=0;i<l1;i++)
        for(int j=0;j<l2;j++){
            if(i == 0 && j == 0) dp[i][j] = 0;
            else{
                if(i != 0) dp[i][j] = min(dp[i][j],dp[i-1][j]+1);
                if(j != 0) dp[i][j] = min(dp[i][j],dp[i][j-1]+1);
                if(i != 0 && j != 0){
                    if(s1[i] == s2[j]) dp[i][j] = min(dp[i][j],dp[i-1][j-1]);
                    else dp[i][j] = min(dp[i-1][j-1]+1,dp[i][j]);
                }
            }
        }
    printf("%d\n",dp[l1-1][l2-1]);
    int x = l1-1, y = l2-1;
    while(x != 0 || y != 0){
        if(x != 0 && dp[x-1][y]+1 == dp[x][y])
            printf("DELETE %d\n",x), x--;
        else if(y != 0 && dp[x][y-1]+1 == dp[x][y])
            printf("INSERT %d %c\n",x+1,s2[y]), y--;
        else if(x != 0 && y != 0 && dp[x-1][y-1]+1 == dp[x][y])
            printf("REPLACE %d %c\n",x,s2[y]), y--, x--;
        else x--,y--;
    }
    return 0;
}
