#include <iostream>
#include <cstdio>
#include <vector>
#include <map>
#include <algorithm>
#include <string.h>
#define INF (1<<30)
using namespace std;
char s[1005],t[1005];
int f[1005][1005];

int dfs(int i,int j){
    if (i==0 && j==0)
        return 0;
    int &res=f[i][j];
    if (res!=-1) return res;
    res=INF;
    if (i>0 && j>0)
        res=min(res,dfs(i-1,j-1)+(s[i]!=t[j]));
    if (i>0)
        res=min(res,dfs(i-1,j)+1);
    if (j>0)
        res=min(res,dfs(i,j-1)+1);
    return res;
}


void work(int i,int j){
    if (i==0 && j==0)
        return ;
    int res=dfs(i,j);
    if (i>0 && j>0 && res==dfs(i-1,j-1)+(s[i]!=t[j])){
        work(i-1,j-1);
        if (s[i]!=t[j]){
            printf("REPLACE %d %c\n",j,t[j]);
        }
        return;
    }
    if (i>0 && res==dfs(i-1,j)+1){
        work(i-1,j);
        printf("DELETE %d\n",j+1);
        return;
    }
    if (j>0 && res==dfs(i,j-1)+1){
        work(i,j-1);
        printf("INSERT %d %c\n",j,t[j]);
        return ;
    }
}



int main(){
    //freopen("input.txt","r",stdin);
    scanf("%s",s+1);
    scanf("%s",t+1);
    int n=strlen(s+1);
    int m=strlen(t+1);
    memset(f,-1,sizeof(f));
    printf("%d\n",dfs(n,m));
    work(n,m);
    return 0;
}
