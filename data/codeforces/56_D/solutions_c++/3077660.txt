#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

#define N 1005
int f[N][N], m, n;
char s[N], t[N];

int main() {
#ifndef ONLINE_JUDGE
    freopen("17.inp", "r", stdin);
    freopen("17.out", "w", stdout);
#endif
    gets(s+1); gets(t+1);
    n = strlen(s+1); m = strlen(t+1);
    for(int i = 0; i <= n; ++i) f[i][0] = i;
    for(int j = 0; j <= m; ++j) f[0][j] = j;
    for(int i = 1; i <= n; ++i)
        for(int j = 1; j <= m; ++j)
            if(s[i] == t[j]) f[i][j] = f[i-1][j-1];
            else f[i][j] = min(f[i-1][j-1], min(f[i-1][j], f[i][j-1])) + 1;
    printf("%d\n", f[n][m]);
    for(int i = n, j = m; i || j; ) 
        if(s[i] == t[j]) { --i; --j; }
        else if(i == 0) printf("INSERT 1 %c\n", t[j--]);
        else if(j == 0) printf("DELETE %d\n", i--);
        else if(f[i][j] == f[i-1][j-1] + 1) printf("REPLACE %d %c\n", i--, t[j--]);
        else if(f[i][j] == f[i-1][j] + 1) printf("DELETE %d\n", i--);
        else printf("INSERT %d %c\n", i + 1, t[j--]);
    return 0;
}
