int d[1005][1005];

string s, t;

void go(int i, int j){
    if (!(i || j)){
        return;
    }

    int b = (s[i - 1] != t[j - 1]);
    if (i && d[i][j] == d[i - 1][j] + 1)
    {
        cout << "DELETE " << i << endl;
        go(i - 1, j);
    }
    else if (j && d[i][j] == d[i][j - 1] + 1)
    {
        cout << "INSERT " << i + 1 << " " << t[j - 1] << endl;
        go(i, j - 1);
    }
    else
    {
        if (b)
            cout << "REPLACE " << i << " " << t[j - 1] << endl;
        go(i - 1, j - 1);
    }
}

int main(){

    cin >> s >> t;

    int n = s.length(), m = t.length();

    for (int i = 0; i < 1005; i++){
        d[i][0] = d[0][i] = i;
    }

    for (int i = 1; i < n + 1; i++){
        for (int j = 1; j < m + 1; j++){
            d[i][j] = min(1 + min(d[i][j - 1], d[i - 1][j]), (s[i - 1] != t[j - 1]) + d[i - 1][j - 1]);
        }
    }
    
    cout << d[n][m] << endl;

    go(n, m);

    return 0;
}
