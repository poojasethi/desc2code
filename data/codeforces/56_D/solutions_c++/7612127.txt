#include<cstdio>
#include<algorithm>
#include<cstring>
#define N 1005
using namespace std;
char a[N],b[N];int f[N][N],pre[N][N];
int n,m,i,j,del,ins,upd;
inline void dfs(int n,int m)
{
  if (n==0&&m==0) return;
  //int did=pre[n][m];
  if (n>0&&m>0&&f[n-1][m-1]==f[n][m]&&a[n]==b[m]) dfs(n-1,m-1);
  else if (f[n-1][m]+1==f[n][m]&&n>0) printf("DELETE %d\n",n),dfs(n-1,m);
  else  if (f[n][m-1]+1==f[n][m]&&m>0) printf("INSERT %d %c\n",n+1,b[m]),dfs(n,m-1);
  else if (n>0&&m>0&&f[n-1][m-1]+1==f[n][m]) printf("REPLACE %d %c\n",n,b[m]),dfs(n-1,m-1);
}
int main()
{
  scanf("%s",a+1);n=strlen(a+1);
  scanf("%s",b+1);m=strlen(b+1);
  for (i=1;i<=n;i++) f[i][0]=i;
  for (i=1;i<=m;i++) f[0][i]=i;
  for (i=1;i<=n;i++)
    for (j=1;j<=m;j++)
    {
      del=f[i-1][j]+1;
      ins=f[i][j-1]+1;
      upd=f[i-1][j-1]+(a[i]!=b[j]);
      f[i][j]=min(min(del,ins),upd);
      /*if (upd<=ins&&upd<=del)
        f[i][j]=upd,pre[i][j]=3;
      else if (ins<=del&&ins<=upd)
        f[i][j]=ins,pre[i][j]=1;
      else if (del<=ins&&del<=upd)
        f[i][j]=del,pre[i][j]=2;*/
    }
  printf("%d\n",f[n][m]);
  dfs(n,m);
  return 0;
}
