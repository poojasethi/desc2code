#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

char a[1005], b[1005];
int n, m;
int dp[1005][1005];
int last[1005][1005];

int main(void) {
	scanf("%s%s", a+1, b+1);
	n = strlen(a+1), m = strlen(b+1);
	for(int i=1;i<=n;i++) { dp[i][0] = i; last[i][0] = 2; }
	for(int j=1;j<=m;j++) { dp[0][j] = j; last[0][j] = 3; }
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++) {
			if(a[i]==b[j]) {
				dp[i][j] = dp[i-1][j-1];
				last[i][j] = 0;
			} else {
				dp[i][j] = dp[i-1][j-1]+1;
				last[i][j] = 1; //REPLACE
				if(dp[i][j] > dp[i-1][j]+1) {
					dp[i][j] = dp[i-1][j]+1;
					last[i][j] = 2; //DELETE
				}
				if(dp[i][j] > dp[i][j-1]+1) {
					dp[i][j] = dp[i][j-1]+1;
					last[i][j] = 3; //INSERT
				}
			}
		}
	int x=n, y=m;
	printf("%d\n", dp[n][m]);
	int ans[2005][2]={}, an=0;
	while(x>0 || y>0) {
		ans[an][0]=x; ans[an++][1]=y;
		if(last[x][y]==0) --x,--y;
		else if(last[x][y]==1) --x,--y;
		else if(last[x][y]==2) --x;
		else --y;
	}
	x=0,y=0;
	for(int i=an-1;i>=0;--i) {
		x=ans[i][0], y=ans[i][1];
		if(last[x][y]==0) continue;
		else if(last[x][y]==1) printf("REPLACE %d %c\n", y, b[y]);
		else if(last[x][y]==2) printf("DELETE %d\n", y+1);
		else printf("INSERT %d %c\n", y, b[y]);
	}
	return 0;
}
