#!/usr/bin/python

s = raw_input()
t = raw_input()
n = len(s)
m = len(t)

d = [[0 for j in range(m+1)] for i in range(n+1)]
for i in range(m+1):
	d[n][i] = m - i
for i in range(n+1):
	d[i][m] = n - i

for i in range(n-1, -1, -1):
	for j in range(m-1, -1, -1):
		if s[i] == t[j]:
			d[i][j] = d[i+1][j+1]
		else:
			d[i][j] = 1 + min(d[i+1][j], d[i][j+1], d[i+1][j+1])

print d[0][0]

i, j = (0, 0)
while d[i][j] != 0:
	if i == n:
		print "INSERT " + str(j+1) + " " + t[j]
		j += 1
	elif j == m:
		print "DELETE " + str(j+1)
		i += 1
	elif s[i] == t[j]:
		i += 1
		j += 1
	else:
		if d[i+1][j+1] <= d[i+1][j] and d[i+1][j+1] <= d[i][j+1]:
			print "REPLACE " + str(j+1) + " " + t[j]
			i += 1
			j += 1
		elif d[i+1][j] <= d[i+1][j+1] and d[i+1][j] <= d[i][j+1]:
			print "DELETE " + str(j+1)
			i += 1
		else:
			print "INSERT " + str(j+1) + " " + t[j]
			j += 1
