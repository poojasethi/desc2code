n = int(raw_input())
numbers = raw_input()
numbers = numbers.split()

result = 0
temp = 0

for i in range(0, n):
    temp = 0
    for j in range(i, n):
        if i == j:
            temp = int(numbers[j])
        else:
            temp = temp ^ int(numbers[j])
        if temp > result:
            result = temp

print result
