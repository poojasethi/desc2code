import sys

#sys.stdin = open("Little Xor.txt", "r")

I = lambda : map(int, raw_input().split())

def main():
    n = int(raw_input())
    a = I()
    ans = 0;
    for i in range(0, n):
        cnt = 0
        for j in range(i, n):
            cnt = cnt ^ a[j]
            ans = max(ans, cnt)
    print ans

if __name__ == '__main__':
    while True:
        try:
            main()
        except EOFError:
            break
