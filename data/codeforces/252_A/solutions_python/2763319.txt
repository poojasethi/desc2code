read_ints = lambda : map(int, raw_input().split())

def main():
    answer = 0
    n = input()
    a = read_ints()
    for i in xrange(n):
        for j in xrange(i + 1, n + 1):
            v = reduce(lambda x, y: x ^ y, a[i:j])
            answer = max(answer, v)
    print answer

if __name__ == '__main__':
    main()
