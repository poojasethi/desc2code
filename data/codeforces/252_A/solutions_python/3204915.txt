from __future__ import (print_function,
                        division,
                        unicode_literals)

n = int(raw_input())
a = map(int, raw_input().split())

dp = [[0 for i in xrange(n)] for j in xrange(n)]

r = 0
for i in xrange(n):
    for j in xrange(i + 1):
        if i == j:
            dp[i][j] = a[i]
        else:
            dp[i][j] = dp[i - 1][j] ^ a[i]
        r = max(r, dp[i][j])

print(r)
