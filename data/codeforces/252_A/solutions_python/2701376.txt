def main():
    n = int(raw_input())
    data = map(int, raw_input().split())

    result = None
    for i in xrange(n):
        for j in xrange(i, n):
            pos = i
            value = data[i]
            while pos < j:
                pos += 1
                value = value ^ data[pos]
            if result is None or value > result:
                result = value
    print result
    


if __name__ == '__main__':
    main()
