FILE = "little_xor"
try:
    inFile = open(FILE+".txt")
except:
    pass

def read():
    try:
        return inFile.readline().strip()
    except:
        return raw_input().strip()


n = int(read()) 
a = map(int,read().split())
best = 0
for l in xrange(n):
    for r in xrange(l,n):
        xor = 0
        for i in xrange(l,r+1):
            xor = xor^a[i]
        best = max(best,xor)
print best
