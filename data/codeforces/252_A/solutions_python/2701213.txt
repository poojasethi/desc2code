def subarrays(array):
    for i in range(len(array)):
        for j in range(len(array)+1):
            yield array[i:j]

def xor(array):
    result = 0
    for el in array:
        result ^= el
    return result
    
raw_input()
numbers = map(int, raw_input().split())
print max(map(xor, subarrays(numbers)))
