#include<iostream>

using namespace std;

int main()
{
	int n,arr[101]={},ans = 0;

	cin>>n;
	for(int i = 0 ; i < n ; ++i )
	{
		cin>>arr[i];
	}

	for(int i = 0 ; i < n ; ++i )
	{
		int tmp = 0;
		for(int j = i ; j < n ; ++j )
		{
			tmp ^= arr[j];
			ans = max(tmp , ans);
		}
	}

	cout<<ans<<endl;
}
