from sys import stdin, stdout

def solve(n, xs):
    t = 0
    zsxs = sorted(zip(xs, range(n)), key = lambda x: x[0])
    for i, x in enumerate(zsxs):
        if not i:
            continue
        t += abs(x[1]-zsxs[i-1][1])
    return str(t)
        
def main():
    lines = stdin.readlines()
    #lines = ['5\n', '1 3 5 4 2\n']
    n = int(lines.pop(0).strip())
    xs = [int(val) for val in lines.pop().strip().split(' ')]
    stdout.write(solve(n, xs))
    
main()
