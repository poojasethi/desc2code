a = int(raw_input())
line = raw_input().split()
disk = []
places = []
for fragment in line:
	disk.append(int(fragment))
	places.append(0)
for place in range(a):
	places[disk[place]-1] = place
answer = 0
for diff_place in range(1, a):
	answer += abs(places[diff_place]-places[diff_place-1])
print answer
