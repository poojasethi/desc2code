from math import acos,asin,atan,sqrt,pi,sin,cos,tan,ceil,floor
import re
from operator import itemgetter as ig

n=input()
lis=map(int,raw_input().split())
f=[0]*(n+3)
for i in range(n):
	f[lis[i]]=i+1;
s=0
for i in range(2,n+1):
	s+=abs(f[i]-f[i-1])
print s
