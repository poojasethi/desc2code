n = int(raw_input())
arr = [0]*n
in_data = raw_input()
in_data = in_data.split(" ")
for i in range(n):
    arr[int(in_data[i]) - 1] = i
count = 0
for i in range(n-1):
    count += abs(arr[i+1] - arr[i])

print count
