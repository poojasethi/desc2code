def read_int():
    return int(raw_input())

def read_ints():
    return (int(x) for x in raw_input().split())

def read_int_list():
    return [int(x) for x in raw_input().split()]

def read_float():
    return float(raw_input())

def read_floats():
    return (float(x) for x in raw_input().split())

def read_float_list():
    return [float(x) for x in raw_input().split()]

def read_list():

n = read_int()
sectors = read_int_list()
cur_idx = sectors.index(1)
i = 1
dist = 0
d = {}
for j in range(len(sectors)):
    d[sectors[j]] = j
while i < n:
    next_idx = d[i+1]
    dist += abs(next_idx - cur_idx)
    cur_idx = next_idx
    i += 1
print dist
