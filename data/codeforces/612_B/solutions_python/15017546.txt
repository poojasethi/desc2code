def hdd():
	List = []
	HDD = {}
	n = input()
	
	f = raw_input().split()
	f = map(int, f)

	for i in range(len(f)):
		HDD[i] = f[i]

	count = 0 

	d = [ (v,k) for k,v in HDD.iteritems() ]
	d.sort(reverse = False)

	for v,k in d:
		List.append(k)
	
	for i in range(len(List)-1):
		count += abs(List[i] - List[i+1])

	print count


hdd()
