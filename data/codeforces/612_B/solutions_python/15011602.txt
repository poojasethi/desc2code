import sys

n = int(sys.stdin.readline().strip())
frags = map(int, sys.stdin.readline().strip().split(" "))

frags_idx = [(frag, idx+1) for idx, frag in enumerate(frags)]
frags_idx.sort()

#print frags
#print frags_idx

delay = 0
pos = None
for _, sector in frags_idx:
    if pos is not None:
        delay += abs(pos-sector)
    pos = sector
