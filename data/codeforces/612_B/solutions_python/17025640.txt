import fileinput

def get_input():
    inp = fileinput.input()
    n = int(inp.readline().strip())
    f = map(lambda a: int(a), inp.readline().strip().split())
    return n, f


def main():
    n, f = get_input()
    a = [0]*(len(f) + 1)
    for i, val in enumerate(f):
        a[val] = i + 1
        # print val, a[val]
    d = 0
    # print a
    for i in xrange(2, len(a)):
        d += abs(a[i - 1] - a[i])
    print d

if __name__ == '__main__':
    main()
