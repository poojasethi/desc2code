#include <iostream>
#include <algorithm>
using namespace std;
const int M=1<<20,D=19137;
typedef long long LL;
LL n,m,a[M]={D},b[M];
LL go(LL* p){
  LL x=0;
  sort(p,p+n);
  for(int i=0,j;i<n;i=j){
    for(j=i;j<n&&p[i]==p[j];++j);
    x+=(LL)(j-i)*(j-i-1)/2;
  }
  return x;
}
main(){
  cin>>n>>m;
  for(int i=1;i<n;++i)a[i]=D*a[i-1];
  for(int i=0,x,y;i<m&&cin>>x>>y;++i)b[x-1]+=a[y-1],b[y-1]+=a[x-1];
  for(int i=0;i<n;++i)a[i]+=b[i];
  cout<<(go(a)+go(b))<<endl;
}
