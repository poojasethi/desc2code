#include <algorithm>
#include <iostream>

using namespace std;

#define LL unsigned long long
#define F first
#define S second

pair<LL, LL> adj[1000001];
LL res;
int n, m;

void solve()
{
	sort(adj+1, adj+n+1);
	for (LL i = 1; i <= n; )
	{
		LL j = i;
		for (; i <= n && adj[i].F == adj[j].F; ++i);
		res += (i-j)*(i-j-1)/2;
	}
}

int main()
{
	cin >> n >> m;
	
	adj[0].S = 666;
	for (int i = 1; i <= n; ++i)
		adj[i].S = adj[i-1].S*33, adj[i].F = 0;
		
	for (int i = 0; i < m; ++i)
	{
		int a, b;
		cin >> a >> b;
		adj[a].F ^= adj[b].S;
		adj[b].F ^= adj[a].S;
	}
	
	solve();
	for (int i = 1; i <= n; ++i) adj[i].F ^= adj[i].S;
	solve();
	
	cout << res << endl;
		
	return 0;
}
