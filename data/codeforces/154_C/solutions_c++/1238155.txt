#include <cstdio>
#include <algorithm>

using namespace std;

int n,m,a,b,g[1000001][2],i;
long long h[1000001]={1},k[1000001],r,t;

int main (){
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;++i) h[i]=h[i-1]*31;
    for(i=0;i<m;++i){
        scanf("%d%d",&a,&b);
        k[a]+=h[b],k[b]+=h[a];
        g[i][0]=a,g[i][1]=b;}
    for(i=0;i<m;++i){
        a=g[i][0],b=g[i][1];
        if(k[a]+h[a]==k[b]+h[b]) r+=2;}
    sort(k+1,k+1+n);
    for(t=1,i=2;i<=n;++i,++t)
        if(k[i]!=k[i-1]){
            r+=t*(t-1);
            t=0;}
    r+=t*(t-1);
    printf("%I64d\n",r/2);
    return 0;
}
