#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
typedef unsigned long long ull;
const ull p = 23333;
ull ha1[1000005],ha2[1000005];
int m,n;
void init(){
ha2[0]=1;
for(int i=1;i<=n;i++){
ha2[i]=ha2[i-1]*p;
}
}
long long solve(ull *ha){
long long ans,cnt;
ans=cnt=0;
sort(ha+1,ha+n+1);
for(int i=2;i<=n;i++){
    if(ha[i]==ha[i-1])ans+=++cnt;
    else cnt=0;
}
return ans;
}
int main(){
int a,b;
scanf("%d%d",&n,&m);
init();
for(int i=1;i<=m;i++){
    scanf("%d%d",&a,&b);
    ha1[a]+=ha2[b],ha1[b]+=ha2[a];
}
for(int i=1;i<=n;i++)ha2[i]+=ha1[i];
printf("%lld\n",solve(ha1)+solve(ha2));
}
