#include <cstdio>
#include <algorithm>
using namespace std;
int b,c,d,e,f,g,i,j,k,l,m,n,x[1000005],y[1000005];
long long a[1000005],s[1000005],h;
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)s[i]=rand()%10000+rand()%10000*10000+1ll*rand()%10000*100000000;
	
	for(i=1;i<=m;i++){
		scanf("%d%d",&x[i],&y[i]);
		a[x[i]]^=s[y[i]];
		a[y[i]]^=s[x[i]];
	}
	for(i=1;i<=m;i++)if((a[x[i]]^s[y[i]])==(a[y[i]]^s[x[i]]))h++;
	sort(a+1,a+1+n);
	for(i=1;i<=n;i++){
		h+=g;
		if(a[i]!=a[i+1]){g=0;}
		else g++;
	}
	printf("%I64d\n",h);
	return 0;
}
