#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
long long n,m,ans,p[1000000],h[2000001];
int main(){
	cin>>n>>m;
	p[0]=1;h[n]=1;
	for(int i=1;i<n;i++){
		p[i]=p[i-1]*1000000007;
		h[i+n]=p[i];
	}
	for(int i=0;i<m;i++){
		int a,b;
		scanf("%d",&a);
		scanf("%d",&b);
		h[a-1]+=p[b-1];
		h[a-1+n]+=p[b-1];
		h[b-1]+=p[a-1];
		h[b-1+n]+=p[a-1];
	}
	sort(h,h+2*n);
	for(long long i=0;i<2*n;){
		long long j=i;
		while(h[j]==h[i])i++;
		ans+=(i-j)*(i-j-1)/2;
	}
	cout<<ans<<"\n";
}
