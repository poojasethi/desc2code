#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
int N,M,i,x,y;
long long a[1000005],f[1000005],g[1000005];
long long Ans,s;
int main()
{
	scanf("%d%d",&N,&M);
	a[1]=1;
	for(i=2;i<=N;i++) a[i]=a[i-1]*37;
	for(i=1;i<=M;i++)
	{
		scanf("%d%d",&x,&y);
		f[x]+=a[y];
		f[y]+=a[x];
	}
	for(i=1;i<=N;i++) g[i]=f[i]+a[i];
	sort(f+1,f+N+1);
	s=0;
	for(i=1;i<=N;i++)
	{
		if(i==1||f[i]!=f[i-1])
		{
			Ans+=s*(s-1)/2;
			s=1;
		}
		else s++;
	}
	Ans+=s*(s-1)/2;
	sort(g+1,g+N+1);
	s=0;
	for(i=1;i<=N;i++)
	{
		if(i==1||g[i]!=g[i-1])
		{
			Ans+=s*(s-1)/2;
			s=1;
		}
		else s++;
	}
	Ans+=s*(s-1)/2;
	printf("%I64d\n",Ans);
	return 0;
}
