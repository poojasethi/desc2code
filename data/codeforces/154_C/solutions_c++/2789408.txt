#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
typedef unsigned long long big;
const int N=1002000;
const big seed=10007;
big pw[N];
int n,m;
big a[N*2],ans,d[N];
int main()
{
    int i,x,y;
    pw[0]=1;
    for(i=1;i<N;i++)pw[i]=pw[i-1]*seed;
    scanf("%d%d",&n,&m);
    while(m--)
    {
        scanf("%d%d",&x,&y);
        x--;y--;
        d[x]+=pw[y];
        d[y]+=pw[x];
    }
    for(i=0;i<n;i++)
    {
        a[i]=d[i];
        a[i+n]=d[i]+pw[i];
    }
    sort(a,a+n+n);
    ans=0;
    for(i=0;i<n+n;i++)
    ans+=upper_bound(a,a+n+n,a[i])-lower_bound(a,a+n+n,a[i])-1;
    cout<<ans/2;
}
