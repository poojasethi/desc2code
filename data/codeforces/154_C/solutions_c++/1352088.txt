#include <cstdio>
#include <algorithm>
#define N 1000005
#define FI(a, b, c) for(int a = b; a <= c; a++)
using namespace std;

typedef long long LL;

int n, m, a[N], b[N];
LL A, h[N], p[N] = {1};

int main(){
	scanf("%d %d", &n, &m);
	FI(i, 1, n) p[i] = p[i - 1] * 23;
	FI(i, 1, m){
		scanf("%d %d", &a[i], &b[i]);
		h[a[i]] += p[b[i]];
		h[b[i]] += p[a[i]];
	}
	
	FI(i, 1, m)
		if(h[a[i]] + p[a[i]] == h[b[i]] + p[b[i]]) A++;
		
	sort(h + 1, h + n + 1);
	FI(i, 1, n){
		int j = i;
		while(j <= n && h[j] == h[i]) j++;
		A += (LL) (j - i) * (j - i - 1) / 2;
		i = j - 1;
	}
	
	printf("%I64d\n", A);
}
