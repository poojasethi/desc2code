#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long ll;

int N, M;
ll h[2222222], p[1111111];
ll ans = 0;

int main() {
	int i, j, k;
	cin>>N>>M;
	h[N] = p[0] = 1ll;
	for (i = 1; i < N; ++i)
		h[N+i] = p[i] = (p[i-1]<<5)+p[i-1];
	while (M--) {
		scanf("%d%d", &i, &j);
		--i, --j;
		h[i] += p[j], h[N+i] += p[j];
		h[j] += p[i], h[N+j] += p[i];
	}
	sort(h, h+N*2);
	for (i = j = 1; i < N*2; ++i)
		if (h[i]^h[i-1]) {
			ans += (ll)j*j; 
			j = 1;
		}
		else ++j;
	ans += (ll)j*j;
	cout<<ans/2-N<<'\n';
	return 0;
}
