#include<iostream>
#include<algorithm>
using namespace std;
typedef unsigned long long i64;
const int N = 1000010, P = i64(1e9) + 7;
int n, m, a[N], b[N];
i64 ans, p[N], h[N];

int main() {
	cin >> n >> m;
	p[0] = 1;
	for (int i = 1; i < N; i++)
		p[i] = p[i - 1] * P;
	for (int i = 0; i < m; i++) {
		cin >> a[i] >> b[i];
		h[a[i]] += p[b[i]];
		h[b[i]] += p[a[i]];
	}
	for (int i = 0; i < m; i++)
		if (h[a[i]] - p[b[i]] == h[b[i]] - p[a[i]])
			ans++;
	sort(h + 1, h + n + 1);
	i64 cnt = 1;
	for (int i = 2; i <= n; i++)
		if (h[i] == h[i - 1])
			cnt++;
		else {
			ans += cnt * (cnt - 1) / 2;
			cnt = 1;
		}
	ans += cnt * (cnt - 1) / 2;
	cout << ans << endl;
}
