#include <bits/stdc++.h>

using namespace std;

int n,l,r,x;
int c[20];

int main(){
	cin >> n>>l>>r>>x;
	int ans=0;
	for (int i=0;i<n;i++) cin >> c[i];
	for (int mask=0;mask<(1<<n);mask++){
		int mm=-1;int mi=1010000;
		int tot=0;
		int num=0;
		for (int i=0;i<n;i++){
			if ((mask>>i)&1){
				num++;tot+=c[i];
				mm=max(c[i],mm);
				mi=min(c[i],mi);
			}
		}
		if (num>=2 && tot>=l && tot<=r && mm-mi>=x) ans++;
	}
	cout << ans<<endl;
	return 0;
}
