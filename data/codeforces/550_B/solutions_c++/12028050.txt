#include <iostream>

using namespace std;

int main()
{
    long long n,l,r,x;cin>>n>>l>>r>>x;long long a[n];for(int i=0;i<n;i++)cin>>a[i];long long mi,ma;int k=0;long long f=0;
    for(int i=0;i<=(1<<n);i++){long long s=0;mi=2e10,ma=0;k=i;
      for(int j=0;j<=15;j++){
        if(k&1)s+=a[j],mi=min(mi,a[j]),ma=max(ma,a[j]);
        if(k)k=k>>1;
      }
      if(s>=l&&s<=r&&ma-mi>=x)f++;
    }
    cout<<f;
    return 0;
}
