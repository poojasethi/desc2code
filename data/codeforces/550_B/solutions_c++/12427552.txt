#include<cstdio>
#include<algorithm>
using namespace std;
int a1[15],p[16];
int main()
{
	p[0]=1;for(int i=1;i<16;++i)p[i]=p[i-1]*2;
	int n,l,r,k,a,b,c,o=0;
	scanf("%d%d%d%d",&n,&l,&r,&k);
	for(int i=0;i<n;++i)scanf("%d",a1+i);
	sort(a1,a1+n);
	for(int i=1;i<p[n];++i)
	{
		a=-1;c=0;
		for(int j=0;j<n;++j)if(i&p[j])
		{if(!~a)a=j;b=j;c+=a1[j];}
		if(c>=l&&c<=r&&a1[b]-a1[a]>=k)
			++o;
	}
	printf("%d\n",o);
}
