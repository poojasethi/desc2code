#include<bits/stdc++.h>
using namespace  std;
int n,a[16],i,j,ans=0,l,r,x;
main()
{
    scanf("%d%d%d%d",&n,&l,&r,&x);
    for (i=0;i<n;i++)
        scanf("%d",&a[i]);
    for (i=1;i<(1<<n);i++)
    {
        int s=0,mn=2e9,mx=0;
        for (j=0;j<n;j++)
            if (i&(1<<j))
        {
            if (mn>a[j]) mn=a[j];
            if (mx<a[j]) mx=a[j];
            s+=a[j];
        }
        if (s>=l&&s<=r&&mx-mn>=x) ans++;
    }
    printf("%d",ans);
}
