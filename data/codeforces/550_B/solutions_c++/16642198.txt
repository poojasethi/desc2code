#include <bits/stdc++.h>
using namespace std;
int n,l,r,x,a[100],eiei[100],w,nub;
void work(int sum,int now,int mc,int mn)
{
	if(now<0)
	{
		if(mc-mn>=x&&sum>=l&&sum<=r) nub++;
		return;
	}
	work(sum+a[now],now-1,max(mc,a[now]),min(mn,a[now]));
	work(sum,now-1,mc,mn);
}
int main()
{
	scanf("%d%d%d%d",&n,&l,&r,&x);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	work(0,n-1,0,9999999);
	printf("%d",nub);	
	
}
