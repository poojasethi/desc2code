#include<stdio.h>

int n,i,t[20],c,l,r,x;

void gen(int i,int sum,int mn,int mx,int num){
    if(i==n){
        if(num>=2&&sum>=l&&sum<=r&&mx-mn>=x)c++;
        return;
    }
    gen(i+1,sum,mn,mx,num);
    if(mx<t[i])mx=t[i];
    if(mn>t[i])mn=t[i];
    gen(i+1,sum+t[i],mn,mx,num+1);
}

int main(){
    scanf("%d %d %d %d",&n,&l,&r,&x);
    for(i=0;i<n;i++)scanf("%d",&t[i]);
    gen(0,0,1000000000,0,0);
    printf("%d\n",c);
    return 0;
}
