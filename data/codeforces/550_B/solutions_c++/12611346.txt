#include<iostream>
using namespace std;
int n,l,r,x,a[100],sum;
int main()
{
    cin>>n>>l>>r>>x;
    for(int i=0;i<n;i++)
    cin>>a[i];
    for(int i=1;i<(1<<n);i++)
    {
        int mx=-1,mn=1000005,mnsum=0;
        for(int j=0;j<n;j++)
        if(i&(1<<j))
        {mx=max(mx,a[j]);mnsum+=a[j];mn=min(mn,a[j]);}
        if(mnsum>=l && mnsum<=r && mx-mn>=x && mx!=mn)
        {sum++;}
    }
    cout<<sum;
    return 0;
}
