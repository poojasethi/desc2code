#include <iostream>
#include <cstdio>
using namespace std;

int n, l, r, x, cnt, a[15];
void f(int p, int q1, int q2, int s) {
	if (p == n) {
		if (s >= l && s <= r && q2 - q1 >= x) {
			cnt++;
		}
	} else {
		f(p + 1, min(q1, a[p]), max(q2, a[p]), s + a[p]);
		f(p + 1, q1, q2, s);
	}
}
int main()
{
	int i;
//	freopen("input.txt", "r", stdin);
	cin >> n >> l >> r >> x;
	for (i = 0; i < n; i++) {
		scanf ("%d", &a[i]);
	}
	f(0, 9e9, -1, 0);
	cout << cnt << endl;
}
