#include <bits/stdc++.h>
using namespace std;
long long a[20],k;
int main(){
	int j,n,l,r,x,u,v,i;
	cin>>n>>l>>r>>x;
	for(i=0;i<n;++i)cin>>a[i];
	sort(a,a+n);
	int ret=0;
	for(i=0;i<(1<<n);++i){
		k=0;
		u=20,v=0;
		for(j=0;j<n;++j)
			if(i&(1<<j))
				k+=a[j],u= min(u,j),v= max(v,j);
		if(a[v]-a[u]>=x && l<=k && k<=r)
			++ret;
	}
	cout<<ret<<endl;
}
