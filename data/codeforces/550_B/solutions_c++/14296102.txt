#include <bits/stdc++.h>
using namespace std;

int n, l, r, x, a[16], sum, cnt, mx, mn;

int main(){
	scanf("%d%d%d%d", &n, &l, &r, &x);
	for(int i=0 ; i<n ; ++i) scanf("%d", &a[i]);

	for(int i=3 ; i<(1<<n) ; ++i){
		sum=0, mx=-1, mn=1e7;
		for(int j=0 ; j<n ; ++j){
			if( (i&(1<<j)) != 0 ){
				sum+=a[j] ;
				mn = min(a[j], mn);
				mx = max(a[j], mx);
			}
		}
		if(sum>=l && sum<=r && (mx-mn)>=x) ++cnt ;
	}
	printf("%d\n", cnt);
	return 0;
}
