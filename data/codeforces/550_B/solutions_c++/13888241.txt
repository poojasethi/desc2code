#include <iostream>
using namespace std;
int n,l,r,x,c[20],mx,mn,s,ans;
int main()
{
	cin>>n>>l>>r>>x;
	for(int i=0;i<n;i++)cin>>c[i];
	for(int i=1;i<(1<<n);i++){
		mx=0;mn=2e9;s=0;
		for(int j=0;j<n;j++)if(i&(1<<j)){
			if(mx<c[j])mx=c[j];
			if(mn>c[j])mn=c[j];
			s+=c[j];
		}
		if(mx-mn>=x && s>=l && s<=r)ans++;
	}
	cout<<ans;
	return 0;
}
