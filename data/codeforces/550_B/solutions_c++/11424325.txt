#include <stdio.h>

#define INF 1000000001

int prob[20];
int n,l,r,x;

int pd(int pos,int min, int max, int tot)
{
	if(pos==n)
		return int(tot<=r && tot>=l && max-min>=x);
	return pd(pos+1,min,max,tot)+\
		pd(pos+1,min < prob[pos] ? min : prob[pos], max > prob[pos] ? max : prob[pos], tot+prob[pos]);
}

int main()
{
	scanf(" %d %d %d %d",&n,&l,&r,&x);
	for(int i=0; i<n; i++)
		scanf(" %d",&prob[i]);
	printf("%d\n",pd(0,INF,-INF,0));
	return 0;	
}
