#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,l,r,x;
	cin>>n>>l>>r>>x;
	int *a=new int [n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	sort(a,a+n);
	int ans=0,i1=0,i2=0,j,k,total;
	int limit=1<<n;
	for(int i=0;i<limit;i++)
	{
		j=i;
		total=0;
		i1=0;
		i2=0;
		for(k=0;k<n;k++)
		{
			if(j&1)
			{
				if(total==0)
					i1=k;
				i2=k;
				total+=a[k];
			}
			j>>=1;
		}
			if(((a[i2]-a[i1])>=x && total>=l) && total<=r)
				ans++;
	}
	cout<<ans<<"\n";
	return 0;
}
