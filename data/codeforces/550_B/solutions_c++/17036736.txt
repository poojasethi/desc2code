#include <bits/stdc++.h>
using namespace std;
int a[16];
int n, l, r, x, cnt;
void olmp(int i, int f, int m, int c)
{
   if(i>=n) return;
   if(m+a[i]<=r){
      if(m+a[i]>=l and a[i]-f>=x) cnt++;
      olmp(i+1, f, m+a[i], c+1);
   }
   olmp(i+1, f, m, c);
}
int main()
{
   cin>>n>>l>>r>>x;
   for(int i=0; i<n; i++) cin>>a[i];
   sort(a, a+n);
   for(int i=0; i<n-1; i++){
      olmp(i+1, a[i], a[i], 1);
   }
   cout<<cnt<<endl;
}
