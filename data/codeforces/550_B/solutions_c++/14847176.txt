#include <bits/stdc++.h>
using namespace std;

#define inf 1000000000
#define MAX 405
#define f(i,n) for(i=0;i<n;i++)


int main()
{
int n,l,r,x,i,j,k,ans=0;
cin>>n>>l>>r>>x;
int a[n];
f(i,n) cin>>a[i];

for(i=0;i<(1<<n);i++){	int t=0,mx=0,mn=2e9;
	for(j=0;j<n;j++){
		if(i&(1<<j)){
		t=t+a[j];
		mx=max(mx,a[j]);
		mn=min(mn,a[j]);
}}
		if(t>=l && t<=r && mx-mn>=x) ans++;}
		cout<<ans;
	return 0;
}
