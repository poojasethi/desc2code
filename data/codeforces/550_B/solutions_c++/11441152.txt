#include <bits/stdc++.h>
#define ll long long int
using namespace std;
ll n,r,k=0,l,x,cnt=0,a[20]={0};
void rec(ll s,ll sum,ll c){
    sum+=a[s];
    if(sum>r)return;
    else if(sum>=l&&a[s]-a[k]>=x&&c>1){
        cnt++;
    }
    for(int i=s+1;i<n;i++)rec(i,sum,c+1);
}
int main(){
    {
    cin>>n>>l>>r>>x;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(k=0;k<n;k++){
        rec(k,0,1);
    }
    cout<<cnt<<endl;
}
}
