#include <bits/stdc++.h>

using namespace std;

int n,l,r,x,ans,ar[30];

int main(){
	scanf("%d %d %d %d",&n,&l,&r,&x);
	
	for(int i=1;i<=n;i++)
		scanf("%d",&ar[i]);
		
	for(int used=0;used<(1<<n);used++){
		int sum=0,mx=0,mn=1e9,s=0;
		
		for(int i=0;i<n;i++)
			if(used & (1<<i)){
				sum+=ar[i+1];
				mx=max(mx,ar[i+1]);
				mn=min(mn,ar[i+1]);
				s++;
			}
		
		ans+=(s>1 && sum>=l && sum<=r && mx-mn>=x);
	}
	
	printf("%d\n",ans);
	
	
	return 0;
}
