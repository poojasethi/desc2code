#include<iostream>
#include<cmath>
using namespace std;

const int INF=1<<30;
int n,l,r,x;
int a[20];
int ans=0;

void DFS(int num,int MAX,int MIN,int sum){
	if(num==n+1){
		return;
	}
	if(sum<=r&&sum>=l&&x<=MAX-MIN&&num==n){
		ans++;
	}
	DFS(num+1,max(MAX,a[num]),min(MIN,a[num]),sum+a[num]);
	DFS(num+1,MAX,MIN,sum);
}

int main(){
	cin>>n>>l>>r>>x;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	ans=0;
	DFS(0,0,INF,0);
	cout<<ans<<endl;
	return 0;
}
