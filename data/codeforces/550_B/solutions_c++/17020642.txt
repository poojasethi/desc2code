#include<iostream>
#include<algorithm>
using namespace std;
int d[10+10];
bool f[10+10];
int n,l,r,x,p=0;
void bede(int k){
	if(k==n){
		int mi=1e6,ma=0,z=0;
		for(int i=0;i<n;i++){
			if(f[i]){
				z+=d[i];
				mi=min(d[i],mi);
				ma=max(ma,d[i]);
			}
		}
		if(z>=l&&z<=r&&ma-mi>=x) p++;
		return;
	}
	f[k]=0;
	bede(k+1);
	f[k]=1;
	bede(k+1);
}
int main(){
	cin>>n>>l>>r>>x;
	for(int i=0;i<n;i++){
		cin>>d[i];
	}
	bede(0);
	cout<<p;
}
