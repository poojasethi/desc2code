#include<iostream>
using namespace std;
int l,r,x;
int ans;
int a[100];
int n;

void f(int m,int M,int sum,int i)
{
    if(i==n)
    {
        if(M-m>=x and sum>=l and sum<=r)
            ans++;
        return ;
    }
    f(m,M,sum,i+1);
    if(m==0)
        m=a[i];
    f(min(m,a[i]),max(M,a[i]),sum+a[i],i+1);
}

main()
{
    cin>>n>>l>>r>>x;
    for(int i=0;i<n;i++ )
    {
        cin>>a[i];
    }
    f(0,0,0,0);
    cout<<ans;
}
