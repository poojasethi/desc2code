#include<bits/stdc++.h>
using namespace std;
int n,l,r,x;
int a[20];
int solve(int pos,int p, int q, int sum) {
    if(pos==n)
        return int(sum<=r && sum>=l && q-p>=x);
    int ans;
    ans= solve(pos+1,p,q,sum);
    p=min(p,a[pos]);
    q=max(q,a[pos]);
    ans+=solve(pos+1,p,q,sum+a[pos]);
    return ans;
}

int main() {
cin>>n>>l>>r>>x;
for(int i=0;i<n;i++) {
    cin>>a[i];
}
cout<<solve(0,INT_MAX,INT_MIN,0);
return 0;
}
