#include<bits/stdc++.h>
using namespace std;
int n,l,r,x;
int a[3232];
int calc(int S){
    int sum=0,cnt=0;
    int mn=1e9,mx=0;
    for(int i=0;i<n;i++)
    if(S>>i&1)sum+=a[i],mx=max(mx,a[i]),mn=min(mn,a[i]),cnt++;
    return sum>=l&&sum<=r&&mx-mn>=x&&cnt>=2?1:0;
}
int main(){
    cin>>n>>l>>r>>x;
    for(int i=0;i<n;i++)cin>>a[i];
    int ans=0;
    for(int i=0;i<(1<<n);i++)
        ans+=calc(i);
    cout<<ans<<endl;    
    return 0;
}
