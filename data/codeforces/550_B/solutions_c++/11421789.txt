#include<cstdio>
#include<algorithm>
using namespace std;
const int maxn=15;

int n,l,r,x,a[maxn];
int main(){
	scanf("%d%d%d%d",&n,&l,&r,&x);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
		
	int tot=0;
	for(int S=1;S<(1<<n);S++){
		int Min=1e9,Max=0,Sum=0;
		for(int i=0;i<n;i++)
			if(S&(1<<i)){
				Min=min(Min,a[i]);
				Max=max(Max,a[i]);
				Sum+=a[i];
			}
		if(Max-Min>=x&&Sum>=l&&Sum<=r) tot++;
	}
	printf("%d\n",tot);
	return 0;
}
