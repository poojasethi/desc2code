#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
ll n,l,r,x;
cin>>n>>l>>r>>x;
vector<ll>d(n);
for(int i=0;i<n;++i)
cin>>d[i];
int ans=0;
for(int i=0;i<(1<<n);++i)
{
ll mini=1e12,maxi=0,sum=0;
int num=0;
for(int j=0;j<n;++j)
{
if((i&(1<<j))==0)
continue;
mini=min(mini,d[j]);
maxi=max(maxi,d[j]);
sum+=d[j];
++num;
}
if(maxi-mini>=x&&sum>=l&&sum<=r&&num>=2)
{
++ans;
}}
cout<<ans<<endl;
return 0;
}
