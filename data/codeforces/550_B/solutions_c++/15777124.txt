#include <bits/stdc++.h> 
#define f(i,a,b)  for(i=a;i<b;i ++) 
using  namespace std ; 
int main () 
{ 
    int n,l,r,x,i,j;
	cin>>n>>l>>r>>x;
	int a[n]; 
    f(i,0,n)cin>>a[i]; 
    int ans=0; 
    f (i,1,1<< n){ 
        int mx=0,mn=INT_MAX,sum=0; 
        f(j,0,n)if(i&( 1<<j)){ 
            mx=max(mx,a[j]),mn=min(mn,a[j]); 
            sum+=a[j]; 
        } 
        if (sum<=r&& sum>=l&& mx-mn>=x)ans++; 
    } 
    cout<<ans; 
}
