#include<cstdio>
#include<cstdlib>
int n,l,r,x;
int c[15];
int ans;
int cmp(const void *a,const void *b){
	return *(int*)a-*(int*)b;
}
void dfs(int min,int max,int sum){
	if(l<=sum&&sum<=r&&c[max]-c[min]>=x)ans++;
	for(int i=max+1;i<n;i++)dfs(min,i,sum+c[i]);
}
int main(){
	scanf("%d%d%d%d",&n,&l,&r,&x);
	for(int i=0;i<n;i++)scanf("%d",&c[i]);
	qsort(c,n,sizeof(int),cmp);
	ans=0;
	for(int i=0;i<n;i++)dfs(i,i,c[i]);
	printf("%d\n",ans);
	return 0;
}
