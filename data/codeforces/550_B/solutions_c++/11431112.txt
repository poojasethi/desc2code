#include <iostream>

using namespace std;
int n,l,r,x,a[20]={0};

int main()
{
    int n,l,r,x;
    cin>>n>>l>>r>>x;
    for(int i=0;i<n;i++) cin>>a[i];
    int m=1,z=0,lm=1<<n;
    while(m<lm){
            int w=0,mx=-3,mn=293847382;
        for(int i=0;i<n;i++)
        if((1<<i)&m) {w+=a[i];mx=max(mx,a[i]); mn=min(mn,a[i]);}
        if(w<=r&&w>=l&&mx-mn>=x) z++;
        m++;
        //cout<<w<<endl;
    }
    cout<<z<<endl;
    return 0;
}
