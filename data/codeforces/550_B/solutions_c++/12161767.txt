#include <iostream>
using namespace std;

int n,l,r,x,arr[20];

int solve(int i,int s,int b,int sum)
{
    if(i==n){
       if(sum>=l&&sum<=r&&b-s>=x) return 1;
       else return 0;
    }
    int way1=solve(i+1,s,b,sum);
    int way2=solve(i+1,min(s,arr[i]),max(b,arr[i]),sum+arr[i]);

    return way1+way2;
}

int main(){

   cin>>n>>l>>r>>x;

   for(int i=0; i<n; i++)
    cin>>arr[i];

   cout<<solve(0,1000009,0,0)<<endl;
   return 0;
}
