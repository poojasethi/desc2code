#include <bits/stdc++.h>
using namespace std;


int main(){
	int n,l,x,r,sum=0,count=0,max1=0,min1=10000000;
	cin >> n >> l >> r >> x;
	int a[n];
	for(int i=0;i<n;i++)
	cin >> a[i];
	for(int i=0;i<pow(2,n);i++){
		for(int j=0;j<n;j++){
			if((i>>j)%2==1){
			sum = sum+a[j];
			max1 = max(max1,a[j]);
			min1 = min(min1,a[j]);
			
			}
		}
		if(sum>=l && sum <=r && (max1-min1)>=x)
		count++;
		sum = 0;
		max1=0,min1=10000000;
	}
	cout << count;
}
