#include<cstdio>
const int MAXN = 20;
int n, l, r, x;
int a[MAXN];
int cnt;

int main() {
	scanf("%d%d%d%d", &n, &l, &r, &x);
	for(int i=0; i<n; ++i)
		scanf("%d", &a[i]);
	for(int seti=1; seti<(1<<n); ++seti) {
		int sum=0, mini=0x3f3f3f3f, maxi=0;
		for(int i=0; (1<<i)<=seti; ++i)
			if((1<<i)&seti) {
				sum += a[i];
				mini = mini<a[i]?mini:a[i];
				maxi = maxi>a[i]?maxi:a[i];
			}
		if(sum>=l&&sum<=r&&maxi-mini>=x)
			++cnt;
	}
	printf("%d\n", cnt);
	return 0;
}
