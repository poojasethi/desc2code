#include<stdio.h>
int n,x,i,j,l,r,a[15],sum,s,max,min;
int main(){
	scanf("%d%d%d%d",&n,&l,&r,&x);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<1<<n;i++){
		s=0;
		max=0;
		min=99999999;
		for(j=0;j<n;j++)
			if(i&1<<j){
				s+=a[j];
				max=max<a[j]?a[j]:max;
				min=min>a[j]?a[j]:min;
			}
		if(max-min>=x&&s<=r&&s>=l)sum++;
	}
	printf("%d",sum);
}
