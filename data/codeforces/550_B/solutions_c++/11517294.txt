#include<bits/stdc++.h>

using namespace std;

int l[10000];

  int a,b,c,d,o;
  
int rec(int n,int f,int m,int v,int ma,int mi)
{
	if(m>n)
	{
		return 0;
	}
    else if(f>=b and f<=c and ma-mi>=d)
    o++;
	for(int i=v+1;i<=n;i++)
	{ 
	   rec(n,f+l[i],m+1,i,max(ma,l[i]),min(mi,l[i])); 
	}
}
  
int main()
{
  cin>>a>>b>>c>>d;
  for(int i=1;i<=a;i++)
  {
  	 cin>>l[i];
  }
  rec(a,0,0,0,0,999999);
  cout<<o<<endl;
  return 0;
}
