#include<iostream>
#include<algorithm>
using namespace std;
int d[10+10];
bool f[10+10];
int n,l,r,x,p=0;
int main(){
	cin>>n>>l>>r>>x;
	for(int i=0;i<n;i++){
		cin>>d[i];
	}
	for(int i=0;i<(1<<n);i++){
		int mi=1e6,ma=0,z=0;
		for(int j=0;j<n;j++){	
			if(i>>j&1){
				z+=d[j];
				mi=min(d[j],mi);
				ma=max(ma,d[j]);
			}	
		}
		if(z>=l&&z<=r&&ma-mi>=x) p++;
	}
	cout<<p;
}
