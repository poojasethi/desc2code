#include <cstdio>
int ans,n,l,r,x,a[20];
void dfs(int k,int mx,int mn,int sum){
	if(k==n){
		if(mx-mn>=x&&sum>=l&&sum<=r)ans++;
		return ;
	}
	dfs(k+1,mx,mn,sum);
	if(mx<a[k]||mx==0)mx=a[k];
	if(mn>a[k]||mn==0)mn=a[k];
	sum+=a[k];
	dfs(k+1,mx,mn,sum);
}
int main(){
	ans=0;
	scanf("%d %d %d %d",&n,&l,&r,&x);
	for(int i=0;i<n;++i)scanf("%d",&a[i]);
	dfs(0,0,0,0);
	printf("%d\n",ans);
	return 0;
}
