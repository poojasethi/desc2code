#include <iostream>
#include <algorithm>
using namespace std;
int n,x,i,j,l,r,a[15],z,s,w,q;
int main(){
    cin>>n>>l>>r>>x;
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=0;i<1<<n;i++){
        s=0;
        w=0;
        q=99999999;
        for(j=0;j<n;j++)
            if(i&1<<j){
            s+=a[j];
            w=max(w,a[j]);
            q=min(q,a[j]);
            }
        if(w-q>=x&&s<=r&&s>=l)z++;
    }
    cout<<z;
}
