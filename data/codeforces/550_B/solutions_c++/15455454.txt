#include<iostream>
#include<algorithm>
using namespace std;
int n,l,r,x,c[1000000],num=0;

int sub(int pos,int min,int max,int total)
{
	if(pos==n)
		return (l<=total&&total<=r&&x<=max-min)?1:0;
	return sub(pos+1,min,max,total)+sub(pos+1,std::min(min,c[pos]),std::max(max,c[pos]),total+c[pos]);		
}

int main()
{
	cin>>n>>l>>r>>x;
	for(int i=0;i<n;i++)
		cin>>c[i];
	cout<<sub(0,0x3e3e3e3e,0,0);
	return 0;
}
