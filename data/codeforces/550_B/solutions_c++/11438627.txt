#include<bits/stdc++.h>
using namespace std;
const int MAXN=15;
int a[MAXN+3];
int l,r,x,n,cnt;
void compsearch (int ind ,int sm , int mn, int mx)
{
	if (ind ==n){
		if (sm>=l&&sm<=r&&mx-mn>=x)
			cnt++;
		return ;
	}
	compsearch(ind+1,sm+a[ind],min(mn,a[ind]),max(mx,a[ind]));
	compsearch(ind+1,sm,mn,mx);
	return ;
}
int main ()
{
	cin>>n>>l>>r>>x;
	for (int i=0;i<n;i++)
		cin>>a[i];
	compsearch(0,0,INT_MAX,INT_MIN);
	cout<<cnt<<endl;
}
