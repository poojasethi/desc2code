from itertools import combinations
n,l,r,y=map(int,raw_input().split())
now=0
diff = [int(i) for i in raw_input().split()]
for L in range(2, n+1):
  for subset in combinations(diff, L):
    ms = max(subset)-min(subset)
    tot = sum(subset)
    if(tot>=l and tot<=r and ms>=y):
        now+=1
print now
