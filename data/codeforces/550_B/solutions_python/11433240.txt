# -*- coding: utf-8 -*-

import sys
import itertools

n, l, r, x = list(map(int, sys.stdin.readline().strip().split()))
f = list(map(int, sys.stdin.readline().strip().split()))

combs = 0

for i in xrange(1, n + 1):
    combinations = itertools.combinations(f, i)
    for comb in combinations:
        _sum = sum(comb)
        if _sum >= l and _sum <= r:
            if max(comb) - min(comb) >= x:
                combs += 1

print combs
