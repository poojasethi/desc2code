import itertools
n,l,r,x=[int(i) for i in raw_input().split()]
a=[int(i) for i in raw_input().split()]
ans=0
for i in range(1,n+1):
        sub=list(itertools.combinations(a,i))
        for j in sub:
                s,d=sum(j),max(j)-min(j)
                if l<=s and s<=r and d>=x:
                        ans+=1
print ans
