n,l,r,x=map(int, raw_input().split(' '))
a=map(int, raw_input().split(' '))
z=0
for i in range(pow(2,n)):
    k=0; s=0; maxa=-1; mina=1000000000000
    for j in range(n):
        if (i & (1<<j)) != 0:
            k+=1
            s+=a[j]
            if maxa<a[j]:
                maxa=a[j]
            if mina>a[j]:
                mina=a[j]
    if s>=l and s<=r and k>1 and maxa-mina>=x:
        z+=1
print (z)
