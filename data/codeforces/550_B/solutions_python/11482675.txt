import itertools
n,l,r,x=raw_input().split()
a=raw_input().split()
a=[int(i) for i in a]
ans=0
for i in range(1,int(n)+1):
        sub=list(itertools.combinations(a,i))
        for j in sub:
                s,d=sum(j),max(j)-min(j)
                if int(l)<=s and s<=int(r) and d>=int(x):
                        ans+=1
print ans
