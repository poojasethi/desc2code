I = lambda f: map(f, raw_input().split())
n, l, r, x = I(int)
C = I(int)
count = 0
for bits in xrange(1 << n):
	if bin(bits).count('1') >= 2:
		amin = 10000000000000
		amax = 0
		asum = 0
		for i in xrange(n):
			if (bits & (1 << i)) != 0:
				amin = min(amin, C[i])
				amax = max(amax, C[i])
				asum += C[i]
		if l <= asum <= r and amax - amin >= x:
			count += 1
print count
