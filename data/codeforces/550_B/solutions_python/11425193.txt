def main():
    n, l, r, x = map(int, raw_input().split())
    c = map(int, raw_input().split())
    ans = 0
    for mask in xrange(0, 1 << n):
        s = [c[i] for i in xrange(n) if (mask & (1 << i)) != 0]
        tot = sum(s)
        if l <= tot and tot <= r and max(s) - min(s) >= x:
            ans += 1
    print ans

if __name__ == "__main__":
    main()
