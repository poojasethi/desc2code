MOD=1000000007
n,l,r,x = map(int, raw_input().split())
C= map(int, raw_input().split())
C.sort(reverse=True)
ans=0
for i in xrange(1, pow(2,n)):
	temp=i
	j=0
	vals=[]
	while temp>0:
		if temp&1:
			vals.append(C[j])
		j+=1
		temp>>=1
	if vals[0]-vals[-1] >= x and l<=sum(vals)<=r:
		ans+=1
print ans
