n,l,r,x=map(int,raw_input().split())
t=map(int,raw_input().split())
import itertools
c=itertools.product([0,1], repeat=n)
ans=0
for a in c:
    if sum(a)>1:
        p=[]
        for i in range(n):
            if a[i]:
                p.append(t[i])
        if l<=sum(p)<=r and max(p)-min(p)>=x:
            ans+=1
print ans
