N, L, R, X = map(int, raw_input().split())
C = map(int, raw_input().split())
Ans = 0

for Cf in range(1, 1 << N):
	Ind = filter(lambda i : (Cf & (1 << i)) > 0, range(N))
	Sum = sum(C[i] for i in Ind)
	Max = max(C[i] for i in Ind)
	Min = min(C[i] for i in Ind)
	if Max - Min >= X and L <= Sum and Sum <= R:
		Ans += 1

print Ans
