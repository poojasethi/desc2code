from itertools import combinations
import sys

rl = lambda: sys.stdin.readline().strip()

N, L, R, X = map(int, rl().split())
C = sorted(map(int,rl().split()))

ret = 0


for nsel in xrange(2,N+1):
    for c in combinations(C,nsel):
        c = list(c)
        sumc = sum(c)
        if L<=sumc and R>=sumc and max(c)-min(c)>=X:
            ret+=1
print(ret)
