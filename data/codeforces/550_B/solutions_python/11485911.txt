from itertools import combinations


input1 = raw_input().split()
input2 = raw_input().split()

n = int(input1[0])
l = int(input1[1])
r = int(input1[2])
x = int(input1[3])

array = []
i = 0

while(i<n):
	array.append(int(input2[i]))
	i = i + 1


j = 1
count = 0
while(j<=n):
	for item in combinations(array,r=j):
		temp = int(sum(item))
		if((temp>=l) and (temp<=r)):
			if(max(item)-min(item)>=x):
				count=count+1	
	j=j+1


print count
