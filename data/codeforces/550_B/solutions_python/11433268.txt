import itertools


if __name__ == '__main__':
    n, l, r, x = map(int, raw_input().split())
    c = map(int, raw_input().split())
    count = 0
    for i in xrange(1, n + 1):
        for co in itertools.combinations(c, i):
            co = list(co)
            s = sum(co)
            if l <= s <= r and min(co) + x <= max(co):
                count += 1
    print count
