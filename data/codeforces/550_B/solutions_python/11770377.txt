# -*- coding: utf-8 -*-
n,l,r,x=map(int,raw_input().split())
c=map(int,raw_input().split())
ans=0
for i in xrange(1<<n):
    a=[]
    for j in xrange(n):
        if (i>>j)&1 :
            a.append(c[j])
    if len(a)>=2 and max(a)-min(a)>=x and sum(a)>=l and sum(a)<=r :
        ans+=1
print ans
