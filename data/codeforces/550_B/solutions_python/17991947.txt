n, l, r, x = map(int,raw_input().split())
c = map(int,raw_input().split())
ans = 0
for i in xrange(1<<n):
    cmin, cmax, ctot = 1e10,0,0
    for j in xrange(n):
        if i & 1<<j:
            cmin, cmax, ctot = min(cmin,c[j]), max(cmax,c[j]), ctot + c[j]
    if ctot >= l and ctot <= r and cmax - cmin >= x:
        ans += 1
print ans
