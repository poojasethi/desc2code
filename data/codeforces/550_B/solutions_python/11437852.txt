from itertools import combinations

if __name__ == '__main__':
	n,l,r,x = [int(i) for i in raw_input().split()]
	prob = [int(i) for i in raw_input().split()]
	acc = 0
	for size in xrange(2,n+1):
		for comb in combinations(prob,size):
			if (sum(comb) >= l) and (sum(comb) <= r) and (max(comb) - min(comb) >= x):
				acc += 1
	print acc
