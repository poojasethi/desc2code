
import sys
import math
import itertools

n,l,r,x = map(int, sys.stdin.readline().rstrip().split())
diffs = map(int, sys.stdin.readline().rstrip().split())

if n <= 1:
	print 0
	sys.exit()

total = 0
for problems in xrange(2, n+1):
	for c in itertools.combinations(range(n),problems):
		p = map(lambda x: diffs[x], c)
		t = sum(p)
		uplim = max(p)
		lowlim = min(p)
		if uplim - lowlim >= x and t >= l and t <= r:
			total += 1

print total
