#getbit
def getbit(T, index):    
    return int((T >> (index)) & 1)

n, l, r, x = map(int, raw_input().split())
c = map(int, raw_input().split())
ans = 0

for i in range(1<<n):
    s, mn, mx = 0, 10**9, -1
    for j in range (n):
        if getbit(i, j):
            mx = max(mx, c[j])
            mn = min(mn, c[j])
            s += c[j]
    if s >= l and s <= r and mx - mn >= x:
        ans += 1

print ans
