#!/usr/bin/python

import sys

def Ni(): return tuple(map(int, sys.stdin.readline().split()))
def Nil(): return map(int, sys.stdin.readline().split())
def Ns(): return tuple(sys.stdin.readline().split())

n, l, r, x = Ni()
c = Nil()

ways = 0
for p in range(1<<n):
    s = [_c for i, _c in enumerate(c) if p & (1<<i) != 0]
    if len(s) >= 2 and l <= sum(s) <= r and max(s) - min(s) >= x:
        ways += 1

print ways
