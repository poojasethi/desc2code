import itertools
def enum(M,m,d,ls):
    total=0
    for n in range(len(ls)):
        lists=itertools.combinations(ls,n+1)
        for l in lists:
            s=sum(l)
            if s<=M:
                if s>=m:
                    if abs(max(l)-min(l))>=d:
                        total+=1
    return total

D=map(int,raw_input().split())
ls=map(int,raw_input().split())
print enum(D[2],D[1],D[3],ls)
