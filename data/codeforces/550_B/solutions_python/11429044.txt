n,l,r,x = map(int,raw_input().split())
array = map(int,raw_input().split())

ans = 0
for l1 in xrange(2,n+1):
	for tup in itertools.combinations(array,l1):
		tot = sum(tup)
		m1 = max(tup)
		m2 = min(tup)
		if tot>=l:
			if tot<=r:
				if m1-m2>=x:
				#	print tot
				#	print tup
					ans= ans+1

print ans
