n,l,r,x = map(int,raw_input().split())

p = map(int,raw_input().split())
import itertools
import time
res = 0
for i in xrange(2,n+1):
    for t in itertools.combinations(p,i):
        curr = list(t)
        total = sum(curr)
        diff = max(curr) - min(curr)
        if total >= l and total <=r  and diff >= x:
            res +=1

print res
