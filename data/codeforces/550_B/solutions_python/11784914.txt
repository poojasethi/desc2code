n,l,r,x=map(int,raw_input().split())
c=map(int,raw_input().split())
a=c
a.sort()
n=len(a)
def dfs(Index,easiest,diff,sum):
    if Index>=n:
        if sum>=l and sum <= r and diff>=x:
            return 1
        else:
            return 0
    ans=dfs(Index+1,easiest,diff,sum)+dfs(Index+1,easiest,a[Index]-easiest,sum+a[Index])
    return ans
ans=0
for i in xrange(n):
    ans += dfs(i+1,a[i],0,a[i])

print ans
