N,L,R,X=map(int, raw_input().split())
d=map(int, raw_input().split())


def bt(i, tot, diff, easy):
    if i == N:
        if L <= tot  <= R and diff - easy >= X:
            return 1
        return 0

    return bt(i+1, tot, diff, easy) + bt(i+1, tot+d[i], max(diff,d[i]), min(easy,d[i]))


print bt(0, 0, 0, 99999999)
