import itertools

n, l, r, x = map(int, raw_input().split())
arr = map(int, raw_input().split())
arr = sorted(arr)

# ## IF
j = 0
if n == 1:
    print 0
    exit()

for ne in xrange(2, n + 1):
    sets = itertools.combinations(arr, ne)
    for comb in sets:
        s = sum(comb)
        if (s >= l) and (s <= r) and (max(comb) - min(comb) >= x):  # and (comb not in compx):
            j += 1

print j
