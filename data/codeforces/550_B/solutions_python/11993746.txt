tmp = raw_input()
n = int(tmp.split()[0])
l = int(tmp.split()[1])
r = int(tmp.split()[2])
x = int(tmp.split()[3])
c = map(int, raw_input().split())

ans = 0
for st in range(1<<n):
    sm = 0
    mi = 1e7
    mx = -1
    for i in range(n):
        if st & (1<<i):
            sm += c[i]
            mi = min(mi, c[i])
            mx = max(mx, c[i])
    if(mx - mi >= x and sm >= l and sm <= r):
        ans += 1

print(ans)
