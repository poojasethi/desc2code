n,l,r,x=map(int,raw_input().split())
c=map(int,raw_input().split())
ans=chk=0
for lst in range(1<<n):
    total=0
    cnt=0
    big=0
    small=10**6+1
    for i in range(n):
        if (lst>>i&1):
            total+=c[i]
            cnt+=1
            big=max(big,c[i])
            small=min(small,c[i])
    if x<=big-small and l<=total<=r and cnt>1:
        ans+=1
print ans
