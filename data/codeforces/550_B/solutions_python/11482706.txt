import itertools
n, l, r, x = map(int, raw_input().split())


def is_valid(seq):
    seq = filter(None, seq)
    if not seq:
        return 0
    return l <= sum(seq) <= r and max(seq) - min(seq) >= x

seq = map(int, raw_input().split())
print sum(is_valid(map(lambda x: x[0] * x[1], itertools.izip(seq, bit))) for bit in itertools.product((0, 1), repeat=n))
