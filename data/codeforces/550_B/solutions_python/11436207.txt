import sys
import itertools
[n,l,r,x] = [int(i) for i in sys.stdin.readline().split()]

nums = [int(i) for i in sys.stdin.readline().split()]

a = range(n)
total = []
for i in range(1,n+1):
    for j in itertools.combinations(a, i):
        total.append([nums[k] for k in j])

ctr = 0
for i in total:
    suma = sum(i)
    if max(i) - min(i) >= x and suma >= l and suma <= r:
        ctr += 1
print ctr
