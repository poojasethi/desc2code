#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
#include <queue>
using namespace std;
#define rep(i,n) for(int i=1;i<=(n);++i)
#define rep2(i,a,b) for(int i=(a);i<=(b);++i)
queue<int> myq;
int n;
struct inode
{
    int a,b,c,d,n;
}ind[100001];
int ia[100001],ib[100001],ic[100001],id[100001];
bool visit[100001];
int pre[100001];
struct tnode
{
    tnode *c[2];
    int id,v;
    bool isx;
}*root;
bool cmp1(inode ia,inode ib)
{
    return ia.a<ib.a;
}
bool cmp2(inode ia,inode ib)
{
    return ia.b<ib.b;
}
tnode *build(int il,int ir)
{
    if(il>ir)
    {
        return 0;
    }
    tnode *it=new tnode();
    it->isx=rand()%2;
    sort(ind+il,ind+ir+1,it->isx?cmp1:cmp2);
    int im=(il+ir)>>1;
    it->id=ind[im].n;
    it->v=it->isx?ind[im].a:ind[im].b;
    it->c[0]=build(il,im-1);
    it->c[1]=build(im+1,ir);
    return it;
}
tnode *kill(tnode *it,int ix,int iy,int ipre)
{
    if(!it)return 0;
    //cout<<"FIND:"<<it->id<<","<<ix<<","<<iy<<endl;
    it->c[0]=kill(it->c[0],ix,iy,ipre);
    if((it->isx&&ix>=it->v)||(!it->isx&&iy>=it->v))
    it->c[1]=kill(it->c[1],ix,iy,ipre);
    if(it->id&&ia[it->id]<=ix&&ib[it->id]<=iy)
    {
        visit[it->id]=true;
        myq.push(it->id);
        pre[it->id]=ipre;
        it->id=0;
    }
    if(it->id==0&&!it->c[0]&&!it->c[1])
    {
        delete it;
        return 0;
    }
    return it;
}
int main()
{
    scanf("%d",&n);
    rep(i,n)
    {
        scanf("%d%d%d%d",&ind[i].a,&ind[i].b,&ind[i].c,&ind[i].d);
        ia[i]=ind[i].a;
        ib[i]=ind[i].b;
        ic[i]=ind[i].c;
        id[i]=ind[i].d;
        ind[i].n=i;
    }
    root=build(1,n);
    root=kill(root,0,0,0);
    while(!myq.empty())
    {
        int pf=myq.front();myq.pop();
        root=kill(root,ic[pf],id[pf],pf);
    }
    if(visit[n])
    {
        vector<int> ans;
        for(int i=n;i;i=pre[i])
        {
            ans.push_back(i);
        }
        cout<<ans.size()<<endl;
        for(int i=ans.size()-1;i>=0;--i)cout<<ans[i]<<" ";
    }
    else cout<<"-1";
}
