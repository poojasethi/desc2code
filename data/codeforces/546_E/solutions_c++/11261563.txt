#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	int a[500][500]={{0}};
	int s=0,t=2*n+1;
	for(int i=1;i<=n;i++)
	cin>>a[s][i];
		
	for(int i=1;i<=n;i++)
	cin>>a[n+i][t];

	for(int i=1;i<=n;i++)
	a[i][n+i]=INT_MAX;
	
	for(int i=0;i<m;i++)
	{
		int c,d;
		cin>>c>>d;
		a[c][n+d]=INT_MAX;
		a[d][n+c]=INT_MAX;
	}
	while(1)
	{
		queue<int> q;
		q.push(s);
		vector<int> vis(2*n+2,0),par(2*n+2,-1);
		while(!q.empty())
		{
			int p=q.front();
			vis[p]=1;
			q.pop();
			for(int i=0;i<=2*n+1;i++)
			if(!vis[i]&&a[p][i])
			{
				par[i]=p;
				q.push(i);
			}
		}
		if(!vis[t])break;

		int flow=INT_MAX;
		for(int i=t;par[i]!=-1;i=par[i])
		{
			flow=min(flow,a[par[i]][i]);
		}
		for(int i=t;par[i]!=-1;i=par[i])
		{
			a[par[i]][i]-=flow;
			a[i][par[i]]+=flow;
		}
	}
	for(int i=1;i<=n;i++)
	if(a[s][i]||a[n+i][t]){cout<<"NO\n";return 0;}
	cout<<"YES\n";
	int r[500][500];
	for(int i=1;i<=n;i++)
	{for(int j=1;j<=n;j++)
	{r[i][j]=a[n+j][i];cout<<r[i][j]<<" ";}cout<<endl;}
	return 0;
}
