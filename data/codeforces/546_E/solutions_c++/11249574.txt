int g[202][202];
int N = 202;
int vis[202];
int ct = 2;
int f[103][103];
int dfs(int cnode, int cmax = INF) {
    if (cnode == 201)
        return cmax;
    if (vis[cnode] == ct)
        return 0;
    vis[cnode] = ct;
    for (int i = 0; i < N; i++) {
        int ret;
        if (g[cnode][i] > 0 && (ret = dfs(i, min(cmax, g[cnode][i])))) {
            g[cnode][i] -= ret;
            g[i][cnode] += ret;
            return ret;
        }
    }
    return 0;
}
int main() {
    int  i, j, n, m, x, y;
    cin >> n >> m;
    int s = 200, t = 201;
    int sa = 0, sb = 0;
    for (i = 0; i < n; i++) {
        cin >> g[s][i];
        sa += g[s][i];
        g[i][i + n] = 200;
    }
    for (i = 0; i < n; i++) {
        cin >> g[i + n][t]; 
        sb += g[i + n][t];
    }
    for (i = 0; i < m; i++) {
        cin >> x >> y;
        x--;
        y--;
        g[x][y + n] = 200;
        g[y][x + n] = 200;
    }

    int ret;
    int cnt = 0;
    ct++;
    while ((ret = dfs(s))) {
        cnt += ret;
        ct++;
    }
    if (sa == sb && sa == cnt) {
        cout << "YES" << endl;
        for (i = 0; i < n; i++) {   
            for (j = 0; j < n; j++) {
                cout << g[j + n][i] << " ";
            }
            cout << endl;
        }
    } else {
        cout << "NO" << endl;
    }

    return 0;
}
