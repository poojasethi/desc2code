#include <iostream>
using namespace std;

long long fastpow (long long a, long long b, long long p) {
	long long res = 1;
	while ( b != 0) {
	if ( b%2 == 1) {
		res = (res*a) % p;
	}
	a = (a*a) % p;
	b = b / 2;
	}
	return res;
}

int main() {
	long long ans = 1;
	long long p = 1000000007;
	int n;
	cin >> n;
	for (int i = 1; i < n; i++) 
		ans = (ans * (((2*n-i)*fastpow(i,p-2,p)) % p)) % p;
	cout << (2 * ans - n) % p;
	return 0;
}
