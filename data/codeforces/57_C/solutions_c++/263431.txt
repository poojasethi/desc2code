#include<stdio.h>
#include<iostream>
#define MOD 1000000007
using namespace std;
long long bmod(long long a,long long p)
{
    long long q;
    if (p==1) return a;
    if (p&1)
    {
        q=bmod(a,p/2);
        return (((q*q)%MOD)*a)%MOD;
    }
    else
    {
        q=bmod(a,p/2);
        return (q*q)%MOD;
    }
}
int main()
{
    long long n,pro,i;
    cin>>n;
    pro=1;
    for (i=n+1;i<=2*n;i++) pro=(pro*i)%MOD;
    for (i=1;i<=n;i++) pro=(pro*(bmod(i,MOD-2)))%MOD;
    cout<<((pro-n)%MOD+MOD)%MOD<<endl;
    return 0;
}
