#include <iostream>
#include <cstdio>
typedef long long ll;

using namespace std;

ll Rev[200000]; int MOD = 1000000007; ll Ans = 1; int N;

int main()
{
    Rev[1] = 1;
    for(int i = 2; i <= 100000; i++)
        Rev[i] = Rev[MOD%i]*(MOD-MOD/i)%MOD;

    scanf("%d", &N);

    for(int Ni = 1; Ni <= N; Ni++)
    {
        Ans *= Rev[Ni]; Ans %= MOD;
        Ans *= (N*2-Ni); Ans %= MOD;
    }

    Ans *= 2; Ans -= N;
    Ans = (Ans%MOD+MOD)%MOD;

    printf("%d\n", Ans);
}
