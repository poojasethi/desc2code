#include <iostream>

using namespace std;

long long fastpow ( long long a, long long b, long long p) {
long long res = 1;
while ( b != 0) {
if ( b%2 == 1) {
res = (res*a) % p;
}
a = (a*a) % p;
b = b / 2;
}
return res;
}

int main() {
long long n, p = 1000000000;
cin >> n;
long long y = 1;
for (long long i = 1; i <= n; ++i){
long long x = fastpow (i, p + 5, p + 7);
y *=((n - 1 + i)*x) % (p + 7);
y %= p + 7;
}
int answer = (2*y - n) % (p + 7);
cout << answer;

return 0;
}
