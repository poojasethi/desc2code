#include <iostream>
using namespace std;

long long n, s=1;

long long fast_pow(long long a, long long b, long long p)
{
    long long res=1;
    while (b!=0)
    {
        if (b % 2==1)
        {
            res=(res*a) % p;
        }
        a=(a*a) % p;
        b=b/2;
    }
    return res;
}

int main()
{
    cin>>n;
    for (long long i=0; i<n-1; i++)
    {
        long long l=fast_pow(i+1, 1000000005, 1000000007);
        s*=(((n+1+i)*l) % 1000000007);
        s=s % 1000000007;
    }
    cout<<(2*s-n) % 1000000007;
    return 0;
}
