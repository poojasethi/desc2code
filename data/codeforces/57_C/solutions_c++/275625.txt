#include <cstdio>
#include <iostream>

using namespace std;

long long f[200200];
int mod=1000000007;
long long pow(long long a,int n)
{
    long long s=1;
    while(n)
    {
        if(n&1) s=(s*a)%mod;
        n>>=1;
        a=(a*a)%mod;
    }
    return s;
}
int comb(int k,int n)
{
    long long b=(f[n-k]*f[k])%mod;
    return f[n]*pow(b,mod-2)%mod;
}
int main()
{
    f[0]=1;
    for(int i=1;i<200200;i++) f[i]=(f[i-1]*i)%mod;
    int n;
    cin>>n;
    cout<<(2*comb(n,2*n-1)-n)%mod<<endl;
}
