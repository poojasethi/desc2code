#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <queue>
#include <map>
#include <string.h>

typedef long long ll;

const int N=200010;
const ll mod=1e9+7;

using namespace std;

int n;
ll f[N];

ll pow(ll a,int b)
{
	ll ret=1;
	while(b)
	{
		if(b&1) ret=ret*a%mod;
		a=a*a%mod;
		b>>=1;
	}
	return ret;
}

ll C(int n,int m)
{
	return f[n]*pow(f[m],mod-2)%mod*pow(f[n-m],mod-2)%mod;
}

int main()
{
	f[0]=1;
	for(int i=1;i<N;i++)
		f[i]=f[i-1]*i%mod;
	scanf("%d",&n);
	cout<<((C(n+n-1,n-1)*2-n)%mod+mod)%mod<<endl;
	return 0;
}
