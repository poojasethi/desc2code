#include <bits/stdc++.h>
using namespace std ;

const int M = 1000000007 ;
int n , ans = 1 ;

int power( int a ) {
   int b = M-2 , ans = 1 ;
   while( b ) {
     if( b&1 ) ans = ( 1LL * ans * a )%M ;
     a = ( 1LL * a * a )%M ; b >>= 1 ; 	 
   } 
   return ans ;
}

int main() {
   cin >> n ;
   for(int i = 1 ; i < n ; i++)  ans = ( ( 1LL * ans * (n + i) )%M * power ( i ) )%M ;	   
   cout << ((2 *ans - n)%M + M )%M ;
    
   return 0 ;
}
