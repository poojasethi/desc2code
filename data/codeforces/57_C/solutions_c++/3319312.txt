

#include <iostream>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)
#define ll       long long
#define MOD      1000000007

using namespace std;

int n;
ll res,denom;

ll power(ll x,ll y){
  ll ans = 1;
  while(y){
    if(y&1)ans = ans * x %MOD;
    x = x*x%MOD;
    y>>=1;
  }
  return ans;
}

int main(){

  cin>>n;res = 1;denom=1;
  for(int i=2*n;i>n;i--)res = res * i%MOD;
  for(int i=1;i<=n;i++)denom = denom*i%MOD;
  denom = power(denom,(ll)MOD-2);
  cout<<(res*denom-n+MOD)%MOD<<endl;

}
