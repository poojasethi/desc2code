#include<stdio.h>
#define P 1000000007
int n,i,j;
__int64 t,s;
int main()
{
    while(scanf("%d",&n)!=EOF)
    {
        t=1;
        for(i=1; i<=n; ++i)
        {
            t=t*i%P;
        }
        for(i=P-3,s=t; i>0; i/=2)
        {
            if(i%2) t=t*s%P;
            s=s*s%P;
        }
        for(i=n; i<n<<1; ++i)
        {
            t=t*i%P;
        }
        printf("%I64d\n",(t*2-n+P)%P);
    }
    return 0;
}
