#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
const int mod = 1000000007;
ll Pow(ll x, ll y) {
	ll ans = 1;
	while(y) {
		if(y&1) ans = ans * x % mod;
		y >>= 1;
		x = x * x % mod;
	}
	return ans;
}
int main() { // C(2n, n) - n
	int n;
	scanf("%d", &n);
	ll ans = 1;
	for(int i = 1; i <= n; i ++) {
		ans = ans * (n+i-1) % mod * Pow(i, mod-2) % mod;
	}
	ans = (ans * 2 - n + mod) % mod;
	printf("%d\n", ans);
	return 0;
}
