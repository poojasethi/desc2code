#include <iostream>
#include <memory.h>
#include <vector>

using namespace std;

typedef long long ll;

const int N = 20;

int d[100001];
const int MOD = 1000000007;

int main() {
  int n;
  cin >> n;

  long long nom = n;
  long long denom = 1;

  for (int i=n+1; i <= 2*n-1; i++) {
    nom = (nom*i)%MOD;
    denom = (denom*(i-n+1))%MOD;
  }

  int p = MOD-2;
  long long r = 1;
  long long t = denom;
  while (p) {
    if (p&1)
      r = (r*t)%MOD;
    t=(t*t)%MOD;
    p >>= 1;
  }

  long long q = (r*nom)%MOD;

  cout << (2*q-n)%MOD << endl;
}
