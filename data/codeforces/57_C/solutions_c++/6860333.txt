#include <cstdio>
#include <iostream>

using namespace std;

#define mod 1000000007

int n;

int main()
{
	scanf("%d", &n);

	int cur = 1;
	for (int i = 2; i <= 2 * n; ++i)
		cur = (cur * 1ll * i) % mod;

	int cur2 = 1;
	for (int i = 1; i <= n; ++i)
		cur2 = (cur2 * 1ll * i) % mod;

	cur2 = (cur2 * 1ll * cur2) % mod;

	int res = cur, b = mod - 2;

	while (b)
	{
		if (b & 1)
			res = (res * 1ll * cur2) % mod;

		b >>= 1;
		cur2 = (cur2 * 1ll * cur2) % mod;
	}

	cout << (res - n + mod) % mod;
}
