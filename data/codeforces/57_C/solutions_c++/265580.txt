#include <cstdio>

typedef long long int ll;

#define MOD 1000000007

ll eleva(ll a, ll b) {
	ll x = a % MOD, res = 1;
	while (b > 0) {
		if (b & 1) res = (res * x) % MOD;
		x = ( x * x ) % MOD;
		b /= 2;
	}
	return res;
}


int main() {
	int n; scanf("%d", &n);
	ll res = 1;

	for (ll i = 2*n-1; i > n; i--)
		res = (res*i) % MOD;

	for (ll i = 2; i < n; i++) {
		ll inv = eleva(i, MOD-2);
		res = (res * inv) % MOD;
	}

	res = (2*res) % MOD;
	res = (res - n + MOD) % MOD;

	printf("%d\n", int(res));

	return 0;
}
