#include<iostream>
#define mod 1000000007
using namespace std;
long long modPow(long long a,int x)
{
	if(x==0)return 1;
	long long t=modPow(a,x/2);
	t=(t*t)%mod;
	if(x%2==0)return t;
	return (t*a)%mod;
}
int main()
{
	long long n;
	cin>>n;
	long long num=1,denom=1;
	for(int i=1;i<=2*n-1;i++)num=(num*i)%mod;
	for(int i=1;i<=n-1;i++)denom=(denom*i)%mod;
	for(int i=1;i<=n;i++)denom=(denom*i)%mod;
	long long t=modPow(denom,mod-2);
	long long res=(t*num)%mod;
	cout<<(2*res-n)%mod<<endl;
}
