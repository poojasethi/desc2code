#include <iostream>
using namespace std;
int n, i;
long long ans, p, x, y, deg;
int main()
{
	 cin >> n;
	 p = 1000000; p = p * 1000 + 7;	 ans = 1;
	 for (i = n + 1; i <= n * 2 - 1; i++) ans = (ans * i) % p;
	 for (i = 1; i <= n - 1; i++)
	 {
		 x = i; y = 1; deg = p - 2;
		 while (deg > 0)
		 {
			 if (deg % 2 == 0)
			 {
				 x = (x * x) % p;
				 deg /= 2;
			 }
			 else
			 {
				 y = (y * x) % p;
				 deg--;
			 }
		 }
		 ans = (ans * y) % p;
	 }
	 cout << (ans * 2 + p - n) % p;
	 return 0;
}
