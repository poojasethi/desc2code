#include <bits/stdc++.h>
using namespace std;

long long mod = 1e9 + 7;
long long EXP(long long b,long long p){
	b %= mod;
	if(p == 0) return 1;
	if(p == 1) return b;
	return (EXP(b*b,p/2) * EXP(b,p%2))%mod;
}

int main(){
	int n;
	cin >> n;
	long long a = 1,b = 1;
	for(int i = 1;i <= 2*n - 1;i++)
		a = (a*i)%mod;
	for(int j = 1;j <= n;j++)
		b = (b*j)%mod;
	for(int j = 1;j < n;j++)
		b = (b*j)%mod;
	long long ans = (a * EXP(b,mod - 2))%mod;
	ans = 2*ans - n;
	ans %= mod;
	ans = (ans + mod)%mod;
	cout << ans << "\n";
	return 0;
}
