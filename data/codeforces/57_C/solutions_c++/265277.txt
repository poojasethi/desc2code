#include<cstdio>
const int MOD=1000000007;
//  Last Change:  2011-01-26 13:36:37
long long inv(long long x)
{
	long long y=MOD-2,ret=1;
	while(y)
	{
		if(y&1)ret*=x,ret%=MOD;
		x*=x,x%=MOD;
		y>>=1;
	}
	return ret;
}
int main()
{
	int n;
	scanf("%d",&n);
	long long x=1;
	for(int i=2;i<=n;++i)
		x=x*i%MOD;
	long long ans=1;
	for(int i=n+1;i<=2*n;++i)
		ans=ans*i%MOD;
	ans=ans*inv(x)%MOD;
	if(ans<n)ans+=MOD;
	printf("%I64d\n",ans-n);
	return 0;
}
