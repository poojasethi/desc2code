#include<iostream>
using namespace std;
long long m = 1000000007;
long long binpow(long long a, long long b)
{
	if (b == 0) return 1;
	if (b == 1) return a;
	long long p = binpow(a,b / 2);
	p = (p * p) % m;
	return (b % 2) ? (p * a) % m : p;
}
int main()
{
	int n;
	cin>>n;
	long long d[100100];
	d[0] = 1;
	for(long long i = 1; i < n; i++)
		d[i] = ((2 * (2 * i + 1) * d[i - 1] % m) * binpow(i + 1, m - 2)) % m;
	long long ans = (2 * d[n - 1] - n + m) % m;
	cout<<ans;
	return 0;
}
