#include <iostream>
using namespace std;
long long power(long long a, long long b, long long p){
	long long res = 1;
	while(b != 0){
		if(b%2 == 1){
			res = (res*a)%p;
		}
		a = (a*a)%p;
		b = b/2;
	}
	return res;
}
int main(){
	int n;
	cin >> n;
	long long s = 2;
	long long p = 1000000007;
	for(int i = 1; i <= n - 1; i++){
		s = (s*(n + i))%p;
		s = (s*power(i, p - 2, p))%p;
		//cout << i << " " << power(i, p - 2, p) << endl;
	}
	s = (p + s - n)%p;
	cout << s << endl;
	return 0;
}
