#include <iostream>
#define MOD 1000000007
//#define MOD 5
using namespace std;
typedef long long ll;
ll POW(ll w,ll step){
  ll ret=1;
  while(step>0){
   if(step%2==1)ret=(ret*w)%MOD;
   w=(w*w)%MOD;step/=2;
               }
  return ret;
}
ll OEM(ll w){return POW(w,MOD-2);}
ll norm(ll w){
  w%=MOD;if(w<0)w+=MOD;
  return w;
}
int main(){
  int N;
  ll CH=1,ZM=1;
  cin>>N;
  for(int i=1;i<=N;++i)ZM=(ZM*i)%MOD;
  for(int i=N;i<2*N;++i)CH=(CH*i)%MOD;
  ll ANS=(CH*OEM(ZM))%MOD;
  cout<<norm(2*ANS-N)<<'\n';
  return 0;
}
