#include <cstdio>

#define mod 1000000007

typedef long long ll;

inline int pow_mod(int a, int b)
{
	int ans = 1;
	for (; b; b >>= 1)
	{
		if (b & 1)
			ans = (ll)ans * a % mod;
		a = (ll)a * a % mod;
	}
	return ans;
}

int main()
{
	int n;
	scanf("%d", &n);
	int ans = 1;
	for (int i = 1; i < n; i ++)
	{
		ans = (ll)ans * (2 * n - i) % mod;
		ans = (ll)ans * pow_mod(i, mod - 2) % mod;
	}
	((ans *= 2) -= n) %= mod;
	printf("%d\n", ans);
	return 0;
}
