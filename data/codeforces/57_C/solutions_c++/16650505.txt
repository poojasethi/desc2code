#include <cstdio>
using namespace std;

#define LL long long

const int MOD = 1000000007;

LL power(LL a, LL n) { return n == 0 ? 1 : power(a * a % MOD, n / 2) * (n & 1 ? a : 1) % MOD; }

int main() {
	int n;
	LL fact, inv1, inv2;
	scanf("%d", &n);
	fact = inv1 = inv2 = 1;
	for(int i = 2; i < n + n; i++) fact = fact * i % MOD;
	for(int i = 2; i < n + 1; i++) inv1 = inv1 * i % MOD;
	for(int i = 2; i < n + 0; i++) inv2 = inv2 * i % MOD;
	printf("%I64d\n", (2LL * fact * (power(inv1, MOD - 2) * power(inv2, MOD - 2) % MOD) + MOD - n) % MOD);
	return 0;
}
