#include <iostream>
using namespace std;

int main(){
        long long n,res=1,base=1,mod=1000000007,power=2*(mod-2);
        cin>>n;
        for(int i=1;i<n;i++) res=(res*(2*n-i))%mod;
        for(int i=1;i<n;i++) base=(base*i)%mod;        
        while(power/=2){
                if(power%2)	res=(res*base)%mod;
                base=(base*base)%mod;
        }
        cout<<((res*2-n+mod)%mod)<<endl;
}
