#include<cstdio>
#define LL long long
int n,i,j,k,md=1000000007;
LL r;
LL pow(LL a,LL b){
    LL r=1;
    while(b){
        if(b&1)r=(r*a)%md;
        b>>=1;
        a=(a*a)%md;
    }
    return r;
}
int main(){
    scanf("%d",&n);
    r=1;
    /*
    2 * (2n-1)C n = 2* (2n-1)!/((n!)(n-1)!)
    */
    for(i=1;i<=n;++i)r=(r*(n+i))%md;
    for(i=2;i<=n;++i)r=(r*pow(i,md-2))%md;
    r-=n;
    if(r<0)r+=md;
    printf("%I64d\n",r);
}
