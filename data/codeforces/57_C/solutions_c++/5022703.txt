#include<iostream>
#include<cstdio>
using namespace std;

int n;

long long power(long long x,long long y)
{
    long long ans=1;
    while(y)
    {
        if(y%2==1)
            ans=ans*x%1000000007;
        x=x*x%1000000007;
        y>>=1;
    }
    return ans;
}

int main()
{
    int i,j,k;
    cin>>n;
    long long res=1;
    long long s=1;
    for(i=2*n;i>n;i--)
        res=(res*i)%1000000007;
    for(i=1;i<=n;i++)
        s=(s*i)%1000000007;
    s=power(s,(long long)1000000007-2);
    cout<<(res*s-n+1000000007)%1000000007<<endl;
    return 0;
}
