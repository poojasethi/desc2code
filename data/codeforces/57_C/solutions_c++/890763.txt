#include <iostream>

using namespace std;
const int p = 1000000007;

long long st(int n, int a)
{
   long long res=1,t=a,r=n;
    while(t)
    {
        if(t%2) res=(res*r)%p;
        r=(r*r)%p;
        t/=2;
    }
    if (res<0) return (p+res);
    return res;

}
int main()
{
    int n;
    cin >> n;
    int s[n+20];
    for (int i=0; i<n+10; i++)
    s[i]=st(i,p-2);

    long long ans=1;
    for(int i=1;i<=n;i++)
    {
        ans=(ans*(2*n-i+1))%p;
        ans=(ans*s[i])%p;
    }
    ans=(ans-n)%p;
    cout << ans;
    return 0;
}
