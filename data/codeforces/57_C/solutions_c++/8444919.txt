#include<cstdio>
#define N 1000005
using namespace std;
const int P=1000000007;
int n,i,rev,ans,mon;
inline int pow(int a,int b)
{
  int res=1;
  for (;b;b>>=1,a=a*1ll*a%P) if (b&1) res=res*1ll*a%P;
  return res;
}
int main()
{
  scanf("%d",&n);ans=1;
  if (n==1) {puts("1");return 0;}
  for (i=n+1;i<2*n;i++)
    ans=ans*1ll*i%P;
  mon=1;
  for (i=2;i<n;i++)
    mon=mon*1ll*i%P;
  rev=pow(mon,P-2);
  ans=ans*1ll*rev%P;
  /*for (i=n-2;i;i--)
  {
    rev=rev*1ll*(i+1)%P;
    ans=ans*1ll*rev%P;
  }*/
  printf("%d",(ans*1ll*2-n+P)%P);
  return 0;
}
