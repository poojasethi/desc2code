#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long LL;
const int MOD = 1000000007;
LL inv(LL x, LL y, LL p, LL q, LL r, LL s) {
	if(y==0) return (p%MOD+MOD)%MOD;
	return inv(y, x%y, r, s, p-r*(x/y), q-s*(x/y));
}
int main(void) {
	int n;
	scanf("%d", &n);
	LL ret=1;
	for(int i=1;i<=n;i++) {
		ret = ret*(n+i-1)%MOD*inv(i,MOD,1,0,0,1)%MOD;
	}
	ret=ret*2%MOD;
	ret=(ret+MOD-n)%MOD;
	printf("%I64d\n", ret);
	return 0;
}
