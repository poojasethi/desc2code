#include<iostream>
#include<cstdio>

using namespace std;

const int mod = 1000000007;

int n , ans;

long long div( long long A , long long B , long long MOD ) {
	if ( A % B == 0 ) return A / B;
	return ( MOD * div( B - A % B , MOD % B , B ) + A ) / B;
}

int main() {
	scanf( "%d" , &n );
	ans = 2;
	for ( int i = n ; i <= 2 * n - 1 ; i++ )
		ans = ( long long ) ans * i % mod;
	int p = 1;
	for ( int i = 1 ; i <= n ; i++ )
		p = ( long long ) p * i % mod;
	ans = div( ans , p , mod );
	ans = ( ans - n + mod ) % mod;
	printf( "%d\n" , ans );
	return 0;
}
