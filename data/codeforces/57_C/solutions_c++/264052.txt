#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;

typedef long long int64;
#define MOD 1000000007

int n;

int64 modPow(int64 x, int k) {
	if (k==0) return 1;
	if (k%2==0) {
		int64 y=modPow(x,k/2);
		return (y*y)%MOD;
	} else {
		int64 y=modPow(x,k-1);
		return (y*x)%MOD;
	}
}

int main() {
	scanf("%d",&n);
	n--;
	int64 st=1,im=1;
	for (int i=n+2;i<=2*n+1;i++) st=(st*i)%MOD;
	for (int i=1;i<=n;i++) im=(im*i)%MOD;
	int64 r=(st*modPow(im,MOD-2))%MOD;
	cout << (2*r-(n+1))%MOD << endl;
    return 0;
}
