#include<stdio.h>
#define P 1000000007
using namespace std;
long long n,i,ans,last;
inline long long quick(long long o,long long p)
{
    long long ans=1,s=o;
    while (p)
    {
        if (p&1)
        ans=(ans*s)%P;
        s=(s*s)%P;
        p=p>>1;
    }
    return ans;
}
int main()
{
    scanf("%I64d",&n);
    ans=1;
    last=1;
    for (i=1;i<n;i++)
    {
        last=((last*(n+i-1))%P*quick(i,P-2))%P;
        ans=(ans+last)%P;
    }
    printf("%I64d\n",((ans*2)%P-n+P)%P);
}
