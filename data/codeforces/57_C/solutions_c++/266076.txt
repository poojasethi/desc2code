#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;

#define REP(i,n) for(i=0;i<(int)n;i++)
typedef long long ll;

#define MOD 1000000007
ll ans;

ll inv(ll x){
    if(x==1) return 1;
    else return MOD-inv(MOD%x)*(MOD/x)%MOD;
}

int main()
{
    int n,i;
    cin>>n;
    
    ll ans=1;
    REP(i,n) ans=ans*(n+i)%MOD;
    REP(i,n) ans=ans*inv(i+1)%MOD;

    ans=(2*ans-n)%MOD;
    ans=(ans+MOD)%MOD;
    cout<<ans<<endl;
    return 0;
}
