#include <iostream>

#define MOD 1000000007

using namespace std;

long long modulo(long long a,long long b){
	long long x=1,y=a;
	while(b > 0){
		if(b%2 == 1){
			x=(x*y)%MOD;
		}
		y = (y*y)%MOD;
		b /= 2;
	}
	return x%MOD;
}

long long fak(long long n){
	long long rez = 1;
	for(int i=1;i<=n;i++){
		rez=(rez*i)%MOD;
	}
	return rez;
}

int main(){
	long long n;
	cin >> n;
	long long rez = (fak(2*n-1)*modulo(fak(n-1),MOD-2))%MOD;
	rez = (rez*modulo(fak(n),MOD-2))%MOD;
	rez = (rez*2-n+MOD)%MOD;
	cout << rez << "\n";
	return 0;
}
