#include <cstdio>
typedef long long ll;
const int mo=(int)1e9+7;
const int N=1000005;
int n,m;
ll ans=1,inv[N];
int main(){
	scanf("%d",&n);
	n+=m=n-1;
	inv[1]=1;
	for(int i=2;i<=m;i++) inv[i]=(mo-mo/i)*inv[mo%i]%mo,(ans*=inv[i])%=mo;
	for(int i=1;i<=m;i++) (ans*=n-i+1)%=mo;
	printf("%d\n",int((ans*2+mo-(m+1))%mo));
}
