#include <bits/stdc++.h>
using namespace std ;

const int M = 1000000007 ;
int n , ans = 1 , val = 1 ;

int power( int a ) {
   int b = M-2 , ans = 1 ;
   while( b ) {
     if( b&1 ) ans = ( 1LL * ans * a )%M ;
     a = ( 1LL * a * a )%M ; b >>= 1 ; 	 
   } 
   return ans ;
}

int main() {
   cin >> n ;
   for(int i = 1 ; i < n ; i++)  { 
      val = ( ( 1LL * val * (n + i - 1) )%M * power ( i ) )%M ;
      ans = ( ans + val )%M ;
   }	   
   cout << (( 2LL * ans  - n )%M + M )%M ;
    
   return 0 ;
}
