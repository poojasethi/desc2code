def readint(): return int(raw_input())

M = 1000000007

def egcd(a, b):
	if (a % b == 0):
		return (0, 1)
	t = egcd(b, a % b)
	return (t[1], t[0] - t[1] * (a / b))

def c(n, k):
	a = 1
	for i in xrange(n, n - k, -1):
		a *= i
		a %= M
	b = 1
	for i in xrange(1, k + 1):
		b *= i
		b %= M
	r = egcd(b, M)[0]
	return (a * r) % M
	
def run():
	n = readint()
	res = c(n + n - 1, n) * 2 - n
	res += M
	res %= M
	print res
run()
