n = int(raw_input())

mod = 1000000007

def a(n):
  if n == 0:
    return 1
  sol = 1
  for x in xrange(n+1, 2*n+1):
    sol = (sol * x) % mod
    #print "!", x, sol

  for x in xrange(1, n+1):
    sol = (sol * pow(x,mod-2,mod)) % mod
    #print x, div, , sol

  return sol

def c(n):
  return (a(n) - n) % mod

print c(n)
