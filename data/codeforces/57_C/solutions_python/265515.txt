from math import factorial as fact
n = input ()
M = 1000000007

def egcd (a, b):
    if a % b == 0:
        return (0, 1)
    x = egcd (b, a % b)
    return (x[1], x[0] - x[1] * (a / b))

def C (n, k): 
    a = 1
    for i in xrange (n - k + 1, n + 1):
        a *= i
        a %= M
    b = 1
    for i in xrange (1, k + 1):
        b *= i
        b %= M
    return (a * egcd (b, M)[0]) % M
    
print (2 * C (n * 2 - 1, n) - n) % M
