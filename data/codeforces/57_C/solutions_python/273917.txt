def powe(x,y):
    if y==0:
        return 1
    ans=powe(x,y/2)
    ans*=ans
    ans%=1000000007
    if y%2==1:
        ans*=x
        ans%=1000000007
    return ans
    
n=input()
answer=1
div=1
for i in range(n+1,2*n):
    answer*=i
    answer%=1000000007
for i in range(2,n):
    div*=i
    div%=1000000007
answer*=powe(div,1000000007-2)
answer%=1000000007
print (((2*answer-n)%1000000007)+1000000007)%1000000007
