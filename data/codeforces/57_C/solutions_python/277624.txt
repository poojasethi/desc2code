n = int(raw_input())
mod = 1000000007
num,den = 1,1
for i in xrange(n+1,2*n):  num  = (num*i)%mod
for i in xrange(1,n): den = (den*i)%mod

def evklid(a,b):
    if a == 1: return (1,0)
    u,v = divmod(b,a)
    x,y = evklid(v,a)
    return (y - u * x, x)

den1,dummy = evklid(den,mod)
den1%= mod

print (2*num*den1-n)%mod
