#include <iostream>
#include <cstdio>
using namespace std;

int n, s, pi, qi, cnt, a[100005], b[100005], ans[55];
char c[5];
int main()
{
	int i;
//	freopen ("input.txt", "r", stdin);
	cin >> n >> s;
	for (i = 0; i < n; i++) {
		scanf ("%s%d %d", &c, &pi, &qi);
		if (c[0] == 'B') a[pi] += qi;
		else b[pi] += qi;
	}
	cnt = 0;
	for (i = 0; i <= 100000; i++) {
		if (b[i] > 0) {
			ans[cnt] = i;
			cnt++;
			if (cnt == s) break;
		}
	}
	for (i = cnt - 1; i >= 0; i--) {
		printf("S %d %d\n", ans[i], b[ans[i]]);
	}
	cnt = 0;
	for (i = 100000; i >= 0; i--) {
		if (a[i] > 0) {
			cnt++;
			printf("B %d %d\n", i, a[i]);
			if (cnt == s) break;
		}
	}
    return 0;
}
