#include "iostream"
#include <algorithm>
using namespace std;
int s[100001], b[100001], q, n, m, i, j, o, p, a, k; char c;pair<int,int>f[100001], d[100001];
int main(){	cin >> n >> m;for (i = 0; i<n; i++){cin >> c >> a >> q;	c == 'B' ? b[a] += q : s[a] += q;}
for (i = 0; i<100001; i++){ if (s[i]>0){ f[o] = make_pair(i, s[i]); o++; }if (b[i]>0){ d[p] = make_pair(i, b[i]); p++; } }
sort(f, f + o); sort(d, d + p); j = min(o - 1,m-1); k = m;
while (j >= 0 && m>0){ cout << "S "; cout << f[j].first << " " << f[j].second << endl; j--; m--; }j = p - 1; m = k;
while (j >= 0 && m>0){ cout << "B "; cout << d[j].first << " " << d[j].second << endl; j--; m--; }}
