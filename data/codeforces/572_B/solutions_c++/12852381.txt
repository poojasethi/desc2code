#include<iostream>
#include<string>
using namespace std;
int po[100500],pr[100500];

int main()
{
	int p,q,k = 0;
	char c;
	int n,s;
	cin >> n >> s;
	for (int i=0;i<n;i++)
	{
		cin >> c >> p >> q;
		if (c == 'B')
		{
			po[p]+=q;	
		}
		else
		{
			pr[p]+=q;
		}
	}
	for (int i=0,t=0; i<100400 && t<s;i++)
	{
		if (pr[i]!=0)
		{
			 k=i;
			t++;
		}
	}
	for (int i=k; i>=0;i-- )
	{
		if (pr[i]!=0)
		{
			cout << "S " << i <<" " << pr[i];
			cout << endl;
		}
	}
	for (int i=100400, t=0;i>=0 && t<s;i--)
	{
		if (po[i]!=0)
		{
		cout << "B " << i << " " << po[i] << endl;
		t++;
		}
	}
	return 0;
}
