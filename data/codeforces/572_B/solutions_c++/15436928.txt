#include<bits/stdc++.h>
using namespace std;
int num[100001];
vector<int>v;
int main()
{
    int n,k,a,b;
    char c;
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {
         cin>>c>>a>>b;
         if(c=='B')
            num[a]+=b;
         else
            num[a]-=b;
    }
    int cnt=k,i=0;
    while(cnt && i<=1e5)
    {
         if(num[i]<0)
            v.push_back(i),cnt--;
        i++;
    }
    for(int j=v.size()-1;j>=0;j--)
        cout<<"S "<<v[j]<<" "<<-num[v[j]]<<endl;
    i=1e5;
    while(k && i>=0)
    {
        if(num[i]>0)
            cout<<"B "<<i<<" "<<num[i]<<endl,k--;
        i--;
    }

}
