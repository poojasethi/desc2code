#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll B[100005],S[100005];
vector<ll>vb,vs;
bool cmp(ll x,ll y)
{
    return x>y;
}
int main()
{
    ll n,s,p,q,i;
    string d;
    cin>>n>>s;
    for(i=0;i<n;i++)
    {
        cin>>d>>p>>q;
        ll*D=(d=="B"?B:S);
        if(D[p]==0)(d=="B"?vb:vs).push_back(p);
        D[p]+=q;
    }
    sort(vb.begin(),vb.end(),cmp);
    sort(vs.begin(),vs.end());
    ll sz=min((ll)vs.size(),s),bz=min((ll)vb.size(),s);
    for(i=sz-1;i>=0;i--)
    cout<<"S "<<vs[i]<<" "<<S[vs[i]]<<"\n";
    for(i=0;i<bz;i++)
    cout<<"B "<<vb[i]<<" "<<B[vb[i]]<<"\n";
    return 0;
}
