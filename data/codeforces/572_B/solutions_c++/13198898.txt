#include<iostream>
using namespace std;
int main()
{
    char d;
    int n,m;
    long long int s[100009]={0},b[100009]={0},vol,p,i,j,l;
    cin>>n>>m;
    for(i=1;i<=n;i++)
    {
        cin>>d>>p>>vol;
        if(d=='S')
            s[p]+=vol;
        else
            b[p]+=vol;
    }
    l=m;
    for(i=0;i<=100000 && m ;i++)
        if(s[i] )
            m--;
    for(j=i-1;j>=0;j--)
        if(s[j])
            cout<<"S "<<j<<" "<<s[j]<<endl;
    for(i=100000;i>=0;i--)
    {
        if(b[i] && l)
            {cout<<"B "<<i<<" "<<b[i]<<endl;
            l--;}
    }
    return 0;
}
