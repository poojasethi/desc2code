#!/usr/bin/python2
# -*- coding: utf-8 -*-

import sys
import os

def solve(p, n):
    n_positive = sum(x for x in p if x > 0)
    n_negative = sum(-x for x in p if x < 0)
    target = n + n_negative
    if target < n_positive:
        target = n_positive - target + n_negative
        div = n_negative
        if div == 0:
            return None
        p = [-x for x in p]
    else:
        div = n_positive
    base = target / div
    mod = target % div
    if base + min(mod, 1) > n:
        return None
    r = []
    for q in p:
        if q > 0:
            r.append(base)
            if mod:
                mod -= 1
                r[-1] += 1
        else:
            r.append(1)
    return r

def main():
    s = sys.stdin.readline().split()
    n = int(s[-1])
    p = [1]
    for i in xrange(1, len(s) - 2, 2):
        p.append(1 if s[i] == "+" else - 1)
    r = solve(p, n)
    if r:
        print "Possible"
    else:
        print "Impossible"
        return
    print r[0],
    for x, y in zip(r[1:], p[1:]):
        print "+" if y > 0 else "-", x,
    print "=", n

if __name__ == '__main__':
    main()
