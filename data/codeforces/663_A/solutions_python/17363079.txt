from sys import stdin
class Solution:
	def getmin(self, ls):
		minlim = 0
		for num in ls:
			if num>0:
				minlim += 1
			else:
				minlim -= target
		return minlim
	
	def getmax(self, ls, target):
		maxlim = 0
		for num in ls:
			if num<0:
				maxlim += -1
			else:
				maxlim += target
		return maxlim
	
	def solveit(self, ls, target):
		positives = 0
		negatives = 0
		for num in ls:
			if num<0:
				negatives += 1
			else:
				positives += 1
		
		val = target + negatives
		lim = positives
		flag = 1
		if val<positives:
			val = positives - target
			lim = negatives
			flag = -1
		#print val, target, positives, negatives
		ret = [flag]
		sumyet = 0
		for i in xrange(lim, 0, -1):
			add = (val-sumyet)/i
			sumyet += add
			ret.append(flag*add)
		
		#print ret
		return ret
