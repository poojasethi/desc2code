from itertools import chain
L = ('+ ' + raw_input()).split()
n = int(L[-1])
pos = L.count('+')
neg = L.count('-')
signs = L[0:-2:2]
lo = pos - n*neg
hi = n*pos - neg
if lo <= n <= hi:
    gap = n - lo
    res = []
    for sign in signs:
        x = 1 if sign == '+' else -n
        dx = min(gap, n-1)
        gap -= dx
        x += dx
        res.append(sign)
        res.append(str(abs(x)))
    print "Possible"
    print ' '.join(res[1:] + L[-2:])
else:
    print "Impossible"
