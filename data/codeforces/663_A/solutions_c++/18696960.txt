#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <string>
using namespace std;
int n,p,m,l,tot,a[2000001];
string s;
int main()
{
	int i,j;
	getline(cin,s);
	l=s.size();
	for(i=0;i<l;i++)
		if(s[i]=='=')
		{
			for(j=i+1;j<l;j++)
				if(s[j]>='0'&&s[j]<='9')
					n=n*10+s[j]-'0';
			break;
		}
	int sum=0;
	a[0]=1,sum=1;
	for(i=0;i<l;i++)
	{
		if(s[i]=='+'){a[++tot]=1;sum++;}
		else if(s[i]=='-'){a[++tot]=-1;sum--;}
	}
	for(i=0;i<=tot;i++)
	{
		if(a[i]>0)while(sum<n&&a[i]<n)a[i]++,sum++;
		else if(a[i]<0)while(sum>n&&a[i]>-n)a[i]--,sum--;
	}
	if(sum!=n)
	{
		cout<<"Impossible"<<endl;
		return 0;
	}
	cout<<"Possible"<<endl;
	for(i=0,j=0;i<l;i++)
	{
		if(s[i]=='?')cout<<abs(a[j++]);
		else cout<<s[i];
	}
	cout<<endl;
	return 0;
}
