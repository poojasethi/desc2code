#define N (1<<8)
#include <bits/stdc++.h>
using namespace std;

string s[N];
int n,cnt[3],p[N],q[N];

int main()
{
	int f=1;
	for(int i=0;cin>>s[i],s[i]!="=";i++)
	{
		if(s[i]=="+") f=1;
		else if(s[i]=="-") f=0;
		else cnt[f]++;
	}
	cin>>n;
	if(cnt[1]*n-cnt[0]<n || cnt[1]-cnt[0]*n>n)
		puts("Impossible");
	else
	{
		int cur=cnt[1]*n-cnt[0];
		for(int i=0;i<cnt[1];i++)
			if(cur-n>=n) p[i]=1,cur-=n-1;
			else p[i]=n-(cur-n),cur=n;

		for(int i=0;i<cnt[0];i++)
			if(cur-n>=n) q[i]=n,cur-=n-1;
			else q[i]=cur-n+1,cur=n;

		puts("Possible"),f=1;
		for(int i=0,j=0,k=0;s[i]!="=";i++)
			if(s[i]=="?") cout<<(f?p[j++]:q[k++])<<" ";
			else cout<<s[i]<<" ",f=s[i]=="+";
		cout<<"= "<<n<<endl;
	}
	return 0;
}
