#include <iostream>
#include <cstdlib>
#include <string>

using namespace std;

int n=0;
int x;
int signo[10000];
int valor[10000];

int main()
{
  char c='+';
  do {
    signo[n++]=c=='+'?1:-1;
    cin>>c;
    cin>>c;
  } while (c!='=');
  cin>>x;
  int suma=0;
  for (int i=0;i<n;i++) {
    valor[i]=1;
    suma+=valor[i]*signo[i];
  }
  while (suma!=x) {
    bool cambio=false;
    for (int i=0;i<n and not cambio;i++) {
      if (valor[i]<x and ((suma<x and signo[i]==1) or (suma>x and signo[i]==-1))) {
	valor[i]++;
	suma+=signo[i];
	cambio=true;
      }
    }
    if (not cambio) {
      cout<<"Impossible"<<endl;
      exit(0);
    }
  }
  cout<<"Possible"<<endl;
  for (int i=0;i<n;i++) {
    if (i>0) cout<<(signo[i]==1?" + ":" - ");
    cout<<valor[i];
  }
  cout<<" = "<<x<<endl;
}
