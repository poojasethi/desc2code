#include <bits/stdc++.h>
using namespace std;
int main()
{
    int p=1,n,j=0,a[105];
    char c;
    a[j++]=1;
    while(cin >> c && c!='=') {
        if(c=='-') p--,a[j++]=-1;
        if(c=='+') p++,a[j++]=1;
    }
    cin >> n;
    for (int i=0; i<j; i++) {
        while(p<n&&a[i]>0&&a[i]<n) a[i]++,p++;
        while(p>n&&a[i]<0&&a[i]>-n) a[i]--,p--;
    }
    if(p!=n) {
        cout << "Impossible\n";
        return 0;
    }
    cout << "Possible\n" ;
    for (int i=0; i<j; i++)
        cout << (i?(a[i]<0?"- ":"+ "):"") << abs(a[i]) << " ";
    cout << "= " << n << endl;
    return 0;
}
