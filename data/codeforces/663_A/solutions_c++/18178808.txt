using namespace std;
const int N=100+5;
int a[N];
char s[10];
int iabs(int x){return x<0?-x:x;}
int main(){
	//freopen("a.in","r",stdin);
	int n,m;
	a[m=1]=1;
	int p=1;
	scanf("%s",s);
	while(~scanf("%s",s)&&s[0]!='=')
	if(s[0]=='+')p+=(a[++m]=1);
	else if(s[0]=='-')p+=(a[++m]=-1);
	scanf("%d",&n);
	rep(i,1,m){
		while(p<n&&a[i]>0&&a[i]<n)a[i]++,p++;
		while(p>n&&a[i]<0&&a[i]>-n)a[i]--,p--;
	}
	if(p!=n)puts("Impossible");
	else{
		puts("Possible");
		rep(i,1,m)
		printf("%d %c ",iabs(a[i]),i==m?'=':(a[i+1]<0?'-':'+'));
		printf("%d\n",n);
	}
	return 0;
}
