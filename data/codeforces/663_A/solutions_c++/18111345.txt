#include<cstdio>
#include<cstdlib>
#include<cstring>

#include<iostream>
#include<algorithm>
using namespace std;

const int _S=1000;

char _s[_S];
int s[_S],n;

int main()
{
    static int len,now;
    gets(_s),len=strlen(_s),s[s[0]=now=1]=1;
    for(int i=0;i<len;i++)
    {
        if(_s[i]=='+')
            s[++s[0]]=1,++now;
        if(_s[i]=='-')
            s[++s[0]]=-1,--now;
        if(_s[i]=='=')
            for(i+=2;i<len;i++)
                n=n*10+_s[i]-'0';
    }
    for(int i=1;i<=s[0];i++)
    {
        while(s[i]>0&&s[i]<n&&now<n)
            ++s[i],++now;
        while(s[i]<0&&s[i]>-n&&now>n)
            --s[i],--now;
    }
    if(now^n)
    {
        puts("Impossible");
        return 0;
    }
    puts("Possible");
    for(int i=0,j=0;i<len;i++)
        if(_s[i]^'?')
            putchar(_s[i]);
        else
            printf("%d",abs(s[++j]));
    puts("");
    return 0;
}
