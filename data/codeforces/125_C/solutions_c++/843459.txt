#include<iostream>
#include<cstdio>
using namespace std;
int a[151][151];
int main()
{
	int n;
	while(scanf("%d",&n)!=-1)
	{
		int k=1;
		while(k*(k+1)<=2*n) k++;
		k--;
		int cur=1;
		for(int i=1;i<=k;i++)
		{
			int tmp=cur;
			for(int j=i;j<=k;j++)
			   a[i][j]=cur++;
			for(int j=i+1;j<=k+1;j++)
			   a[j][i]=tmp++;
		}
		a[k+1][k]=cur-1;
		printf("%d\n",k+1);
		for(int i=1;i<=k+1;i++)
		{
			printf("%d",a[i][1]);
			for(int j=2;j<=k;j++)
			   printf(" %d",a[i][j]);
			printf("\n");
		}
	}
	return 0;
}
