# include <bits/stdc++.h>
using namespace std;
int k = 0,n;
vector <int> a[10001];
int main ()
{
	cin >> n;
	while(k * (k + 1) <= n * 2) k++;
	
	int col = 1;
	for(int i = 1;i <= k;i ++)
	for(int j = i + 1;j <= k;j ++)
	{
		a[i].push_back(col);
		a[j].push_back(col);
		col ++;
	}
	cout << k <<endl;
	for(int i = 1;i <= n;i ++)
	{
		for(int j = 0;j < a[i].size();j ++)
		cout << a[i][j] << ' ';
		cout << endl;
	}
}
