#include <cstdio>

int tab[1000][1000];

int main()
{
    int n;
    int i,j,k,l = 1;
    scanf("%d",&n);
    for(k=1;(k*(k-1))/2<=n;k++); k--;
    printf("%d\n",k);
    for(i=0;i<k;i++) for(j=0;j<k;j++) tab[i][j] = 0;
    for(i=0;i<k;i++) for(j=i+1;j<k;j++) tab[i][j] = tab[j][i] = l++;
    for(i=0;i<k;i++) {
        for(j=0;j<k;j++) if(tab[i][j]!=0) printf("%d ",tab[i][j]);
        printf("\n");
    }
    return 0;
}
