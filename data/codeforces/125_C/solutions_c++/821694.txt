#include <stdio.h>
#include <vector>
#include <algorithm>
#include <iostream>
using namespace std;

vector<int> vec[400];

int main() {
  int n, k;
  cin >>n;
  k = 3;
  while ((k + 1) * k / 2 <= n) k++;

  cout<< k<<endl;
  n = 0;
  for (int i = 0; i < k; i++)
    for (int j = i + 1; j < k; j++) {
      vec[i].push_back(n);
      vec[j].push_back(n);
      n++;
    }

    for (int i = 0; i < k; i++)
    {
     for (int j = 0; j < vec[i].size(); j++) cout<<vec[i][j]+1<<" ";
     cout<<endl;
    }

  return 0;
}
