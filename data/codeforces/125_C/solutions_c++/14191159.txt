#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>
using namespace std;
vector<int> g[1111];
void fun(int k)
{if(k>0)
{
int i=0;
for(int m=k*(k-1)/2;m>(k-2)*(k-1)/2;m--)
{g[k-1].push_back(m);
g[i].push_back(m); i++;
}
fun(k-1);}
}
int main()
{int n; cin>>n; int x=int((sqrt(8*n+1)+1)/2);
fun(x);
cout<<x<<endl;
for(int i=0;i<x;i++)
{for(int j=0;j<x-1;j++)
{cout<<g[i][j]<<" ";
} cout<<endl;
}
return 0;
}
