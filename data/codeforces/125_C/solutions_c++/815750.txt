#include <cstdio>
#include <vector>
using namespace std;

vector<vector<int> > res;
int main()
{
	int n;
	int cnt=0;
	scanf("%d",&n);
	int l,r,c;
	l=2;
	r=n;
	while(l<r)
	{
		c=(l+r+1)/2;
		if(c*(c-1)/2<=n)
			l=c;
		else
			r=c-1;
	}
	res.resize(l);
	for(int i=0;i<=l;i++)
	{
		for(int j=i+1;j<l;j++)
		{
			cnt++;
			res[i].push_back(cnt);
			res[j].push_back(cnt);
		}
	}
	printf("%d\n",l);
	for(int i=0;i<l;i++)
	{
		for(int j=0;j<res[i].size();j++)
			printf("%d ",res[i][j]);
		printf("\n");
	}
}
