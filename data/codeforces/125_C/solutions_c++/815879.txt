#include <iostream>
#include <cstdio>
#include <cmath>
#include <vector>

using namespace std;

int n;
int x;
vector < vector < int > > a;

int main()
{
	cin >> n;
	
	x = 3;
	int i = 3;
	
	while (x <= n)
	{
		x += i;
		i++;
	}
	
	n = i - 1;
	
	cout << n << endl;
	x = 1;
	
	a.resize(n);
	
	for (int i = 0; i < n; i++)
		for (int j = i + 1; j < n; j++)
		{
			a[i].push_back(x);
			a[j].push_back(x);
			x++;
		}
		
		
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < a[i].size(); j++)
			printf("%d ", a[i][j]);
		printf("\n");
	}
}
