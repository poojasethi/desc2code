#include<cstdio>
#include<vector>

using namespace std;

int n;
int k;
vector<int> days[200];

void findksetn() {
	k = 0;
	while(k*(k+1)/2 <= n) k++;
}

int main() {
	scanf("%d", &n);
	findksetn();
	printf("%d\n", k);
	int t = 1;
	for(int i=0;i<k;i++) {
		for(int j=i+1;j<k;j++) {
			days[j].push_back(t);
			days[i].push_back(t);
			t++;
		}
		for(vector<int>::iterator it = days[i].begin(); it != days[i].end(); it++) {
			printf("%d ", *it);
		}
		printf("\n");
	}
	return 0;
}
