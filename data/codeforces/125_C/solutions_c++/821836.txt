#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int n,z,k;
    vector <int> vec[200];
    cin>>n;
    k=3;
    while((k*(k+1)/2)<=n)
    k++;
    cout<<k<<endl;
    z=0;
    for(int i=0; i<k; i++)
      for(int j=i+1; j<k; j++)
      {
        vec[i].push_back(z);
        vec[j].push_back(z);
        z++;
      }
    for(int i=0; i<k; i++)
    {
    for(int j=0; j<vec[i].size(); j++)
        cout<<vec[i][j]+1<<" ";
        cout<<endl;
    }
    return 0;
}
