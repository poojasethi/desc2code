#include<iostream>
using namespace std;
int n,i,j,s,k,a[151][151];
int main()
{
    cin>>n;
    j=0;i=0;
    while (j<=n)
    {
          i++;j+=i;
          }
    cout<<i<<endl;
    i--;s=0;
    for (j=i;j>=1;j--)
    {
        for (k=i-j+1;k<=i;k++)
        {
            a[i-j+1][k]=k-i+j+s;
            a[k+1][i-j+1]=k-i+j+s;
            }
        s+=j;
        }
    for (j=1;j<=i+1;j++)
    {
        for (k=1;k<i;k++) cout<<a[j][k]<<' ';
        cout<<a[j][i]<<endl;
        }
    return 0;
}
