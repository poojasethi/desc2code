#include<iostream>
#include<cstdio>
#include<cstdlib>
using namespace std;
int ans[501][501];
int main()
{
	int n;
	cin>>n;
	int m=1;
	while(m++) if(m*(m+1)>n*2) break;
	int k =1;
	for(int i=0;i<m-1;i++)
	{
		for(int j=i;j<m-1;j++)
			ans[i][j] = k+j-i;
		for(int j=i+1;j<m;j++)
			ans[j][i] = k+j-1-i;
		k+= m-i-1;
	}
	cout<<m<<endl;
	for(int i=0;i<m;i++) 
	{ 
		for(int j=0;j<m-1;j++) 
			cout<<ans[i][j]<<" "; 
		cout<<endl;
	}
	return 0;
}
