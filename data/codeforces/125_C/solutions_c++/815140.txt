#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;

int n, m, t;
vector<int> l[10005];

int main(){
	scanf("%d", &n);
	for (int i=1;; i++){
		if (i*(i-1)/2 <= n) m = i;
		else break;
	}
	
	t = 0;
	for (int i=1; i<=m; i++){
		for (int j=i+1; j<=m; j++){
			t++;
			l[i].push_back(t);
			l[j].push_back(t);
		}
	}
	
	printf("%d\n", m);
	for (int i=1; i<=m; i++){
		for (int j=0; j<l[i].size(); j++) printf("%d%c", l[i][j], j<l[i].size()-1 ? ' ':'\n');
	}
	return 0;
}
