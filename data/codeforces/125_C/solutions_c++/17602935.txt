#include<iostream>
#include<vector>
using namespace std;
const int maxn=11111;
vector<int>G[maxn];
int main()
{
	int n;cin>>n;
	int len=0;
	while(len*(len-1)/2<=n)len++;
	int num=1;
	cout<<len-1<<endl;
	for(int i=0;i<len-1;i++){
		for(int j=i+1;j<len-1;j++){
			G[i].push_back(num);
			G[j].push_back(num++);
		}
	}
	for(int i=0;i<len-1;i++){
		for(int j=0;j<G[i].size();j++){
			if(j==0)cout<<G[i][j];
			else cout<<" "<<G[i][j];
		}
		cout<<endl;
	}
}
