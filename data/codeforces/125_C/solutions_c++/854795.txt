#include <stdio.h>

int main()
{
    int s[200][200], n, p, c, i, j;
    c = 1;
    s[0][0] = 1;
    s[1][0] = 1;
    scanf("%d", &n);
    for(p = 2; c+p <= n; p++){
        for(i = 0; i < p; i++)
            s[i][p-1] = s[p][i] = c+i+1;
        c += p;
    }
    printf("%d\n", p);
    for(i = 0; i < p; i++){
        for(j = 0; j < p-1; j++)
            printf("%d ", s[i][j]);
        printf("\n");
    }
    return 0;
}
