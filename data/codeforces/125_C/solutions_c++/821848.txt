#include <iostream>
#include <vector>

using namespace std;

int main()
{
 int n,k,z;
 vector <int> vec[200];
 cin>>n;
 k=3;
 while ((k*(k+1)/2)<=n)
 k++; z=0;
 cout<<k<<endl;
 for(int i=0; i<k; i++)
   for(int j=i+1; j<k; j++)
   {
       vec[i].push_back(z);
       vec[j].push_back(z);
       z++;
   }
  for(int i=0; i<k; i++)
  {
      for(int j=0; j<k-1; j++)
      cout<<vec[i][j]+1<<" ";
      cout<<endl;
  }
    return 0;
}
