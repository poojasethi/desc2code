#include <stdio.h>
#define  MAXN 1000

int N,res,sum;
int ar[MAXN][MAXN];

int main()
{
  scanf("%d",&N);
  while(sum<=N)
  {
    res++;
    sum+=res;
  }
  sum-=res;
  res--;

  int tmp=0;
  for(int i=0 ; i<res ; i++)
    for(int j=i ; j<res ; j++)
    {
      ar[i][j]=tmp+1;
      ar[j+1][i]=tmp+1;
      tmp++;
    }

  printf("%d\n",res+1); 
  for(int i=0 ; i<=res ; i++)
  {
    for(int j=0 ; j<res ; j++)
      printf("%d ",ar[i][j]);
    printf("\n");
  }
  return 0;
}
