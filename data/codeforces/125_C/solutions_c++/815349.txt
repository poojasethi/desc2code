#include <iostream>
#include <cstdio>
using namespace std;
int n;
#define N 1111
int a[N][N];

int main()
{
	cin >> n;
	for (int k = 3; k <= 1000; ++k)
		if ((k + 1)*(k)/2 > n)
		{
			int now = 0;
			for (int i = 1; i <= k; ++i)
				for (int j = i + 1; j <= k; ++j)
					a[i][j] = a[j][i] = ++now;
			
			printf("%d", k);
			for (int i = 1; i <= k; ++i)
			{
				int fst = true;
				for (int j = 1; j <= k; ++j)
					if (a[i][j])
						printf("%s%d", fst ? (fst = false,"\n") : " ", a[i][j]);
			}
			puts("");
			break;
		}
}
