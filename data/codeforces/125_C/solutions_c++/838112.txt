#include<iostream>
#include<cstdio>
#include<string.h>
#include<string>
#include<math.h>
#include<algorithm>
#include<set>
#include<map>
#include<stack>
#include<vector>
#include<queue>
using namespace std;

vector<int> a[1000];
int n;

int main(){
	cin >> n;
	int m=1;
	while(n>=m*(m+1)/2) m++;
	cout << m << endl;
	int num=0;
	for(int i=0;i<m;i++)
		for(int j=i+1;j<m;j++){
			a[i].push_back(++num);
			a[j].push_back(num);
		}
	for(int i=0;i<m;i++){
		for(int j=0;j<a[i].size();j++){
			if(j>0) cout << " ";
			cout << a[i][j];
		}
		cout << endl;
	}
	return 0;
}
