#include<iostream>
#include<cmath>
using namespace std;

const int MAX = 200;

int a[MAX+10][MAX+10];

int main()
{
	int n;
	cin >> n;
	int k = (int)sqrt(2*n)+1;
	if( !(k*(k-1) <= 2*n) )
		k--;
	cout << k << endl;
	int last = 1;
	for(int i=0; i<k; i++)
	{
		for(int j=0; j<k-1; j++)
		{
			if( j >= i )
				a[i][j] = last++;
			else
				a[i][j] = a[j][i-1];
			cout << a[i][j] << " ";
		}
		cout << endl;
	}
	return 0;
}
