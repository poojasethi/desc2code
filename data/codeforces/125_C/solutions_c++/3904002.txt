#include <cstdio>
#include <iostream>
#include <vector>

#define pb push_back

using namespace std;

const int MAXN=10010;

vector<int> vec[MAXN];

int main(){
	
	int N;
	
	scanf(" %d",&N);
	
	int cnt=1;
	
	for( ; cnt*(cnt+1)/2<=N ; cnt++ );
	
	printf("%d\n",cnt);
	
	for( int i=1,c=1 ; i<cnt ; i++ )
		for( int j=i+1 ; j<=cnt ; j++,c++ ){
			vec[i].pb(c);
			vec[j].pb(c);
		}
	
	for( int i=1 ; i<=cnt ; i++,puts("") )
		for( vector<int>::iterator it=vec[i].begin() ; it!=vec[i].end() ; it++ )
			printf("%d ",*it);
	
	return 0;
}
