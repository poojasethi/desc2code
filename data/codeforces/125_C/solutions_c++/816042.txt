#include <cstdio>
#include <cstdlib>
int ans[1001][1001];
int main()
{
	int n, nowans = 0;
	scanf("%d", &n);
	int now = 2, num = 1;
	while(1){
		if (num + now > n) break;
		num += now++;
		}
	for (int i = 1; i <= now; i++)
		for (int j = i + 1; j <= now; j++)
			ans[i][j] = ans[j][i] = ++nowans;
	printf("%d\n", now);
	for (int i = 1; i <= now; i++){
		bool first = true;
		for (int j = 1; j <= now; j++) if (i != j)
			if (first) printf("%d", ans[i][j]), first = false;
			else printf(" %d", ans[i][j]);
		putchar('\n');
		}
}
