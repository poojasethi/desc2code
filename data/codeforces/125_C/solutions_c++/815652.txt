#include <iostream>
#include <cmath>
#include <vector>
#include <algorithm>
#include <iterator>
using namespace std;

vector<int> V[200];

int main()
{
	int n;
	cin>>n;
	int d=(sqrt((double)(1+8*n))+1)/2+0.0000000001;
	cout<<d<<endl;
	for (int i=1,k=1;i<=d;i++)
		for (int j=i+1;j<=d;j++,k++) {
			V[i].push_back(k);
			V[j].push_back(k);
		}
		for (int i=1;i<=d;i++) {
			copy(V[i].begin(),V[i].end(),ostream_iterator<int>(cout," "));
			cout<<endl;
		}
		return 0;
}
