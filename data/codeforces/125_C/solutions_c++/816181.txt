#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int k=0;
    while (k*(k-1)<=2*n)
    {
        ++k;
    }
    --k;
    cout<<k<<endl;
    int cc=1;
    int least[k];
    for (int i=0; i<k; ++i)
    {
        least[i]=cc;
        int j=0;
        for (; j<i; ++j)
        {
            cout<<least[j]<<" ";
            ++least[j];
        }
        for (; j<k-1; ++j)
        {
            cout<<cc<<" ";
            ++cc;
        }
        cout<<endl;
    }
    return 0;
}
