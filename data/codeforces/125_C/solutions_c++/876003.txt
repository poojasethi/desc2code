#include<stdio.h>
int main()
{
	int n,a[200],i,map[200][200],t,j,k;
	a[2]=1;a[0]=a[1]=0;
	for(i=3;i<=147;i++)
		a[i]=i-1+a[i-1];
	scanf("%d",&n);
	for(i=0;i<=147;i++)
		if(n<a[i])break;
		i--;
	for(j=0;j<i-1;j++)
		map[0][j]=j+1;
	t=i;
	printf("%d\n",i);
	for(j=1;j<i;j++)
	{
		for(k=0;k<j;k++)
			map[j][k]=map[k][j-1];
		for(;k<i-1;k++)
			map[j][k]=t++;
	}
	for(j=0;j<i;j++)
	{
		printf("%d",map[j][0]);
		for(k=1;k<i-1;k++)
			printf(" %d",map[j][k]);
		printf("\n");
	}
	return 0;
}
