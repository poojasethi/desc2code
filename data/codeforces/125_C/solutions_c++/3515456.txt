#include<cstdio>

int d[200][200];

int main() {
    int n; scanf("%d", &n);
    int k = 150;
    while(k * (k-1) / 2 > n) --k;
    int p = 1;
    for(int i = 0; i < k; ++i)
        for(int j = i; j < k-1; ++j)
            d[i][j] = d[j+1][i] = p++;
    printf("%d\n", k);
    for(int i = 0; i < k; ++i) {
        for(int j = 0; j < k-1; ++j) {
            if(j) printf(" ");
            printf("%d", d[i][j]);
        }
        printf("\n");
    }
    return 0;
}
