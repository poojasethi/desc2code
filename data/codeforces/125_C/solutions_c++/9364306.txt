#include <bits/stdc++.h>

#define pb push_back

using namespace std;

const int maxl = 1000 + 1;

int n;
vector < int > ans [ maxl ];

int main (){
	cin >> n;

	for ( int x = 1 ; x < maxl ; x++ ){
		n -= x - 1;

		if ( n < 0 ){
			x--;
			cout << x << endl;

			int cnt = 1;
			for ( int i = 0 ; i < x ; i++ ){
				for ( int j = i + 1 ; j < x ; j++ ){
					ans [ i ].pb ( cnt );
					ans [ j ].pb ( cnt );

					cnt++;
				}

				for ( int j = 0 ; j < x - 1 ; j++ )
					cout << ans [ i ][ j ] << " ";
				cout << endl;
			}

			return 0;
		}
	}
}
