#include <iostream>
using namespace std ;
const int maxm = 143 ;
int n , m = 3 , cnt = 0 ;
int a[maxm][maxm] ;
int main() {
    cin >> n ;
    while (m*(m-1)/2 <= n) ++m ;
    --m ;
    for (int i = 0 ; i < m ; ++i)
        for (int j = i ; j < m-1 ; ++j)
            a[i][j] = a[j+1][i] = ++cnt ;
    cout << m << endl ;
    for (int i = 0 ; i < m ; ++i) {
        cout << a[i][0] ;
        for (int j = 1 ; j < m-1 ; ++j)
            cout << " " << a[i][j] ;
        cout << endl ;
    }
    return 0 ;
}
