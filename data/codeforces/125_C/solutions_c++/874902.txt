#include <cstdio>
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <set>
#include <queue>

using namespace std;

int main() {
	int n,k;
	cin >> n;
	k=3;
	while (k*(k-1)/2<=n) k++;
	k--;
	vector<vector<int> >v(k);
	int l=1;
	for (int i=0; i<k; i++) {
		for (int j=i+1; j<k; j++) {
			v[i].push_back(l);
			v[j].push_back(l);
			l++;
		}
	}
	cout << k << endl;
	for (int i=0; i<v.size(); i++) {
		for (int j=0; j<v[0].size(); j++) {
			cout << v[i][j] << ' ';
		}
		cout << endl;
	}
	return 0;
}
