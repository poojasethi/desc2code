#include <iostream>
#include <vector>
using namespace std;

vector<int> ff[(int)1e4+10];

int main()
{
    int n;
    cin>>n;
    int ans=0;
    while(ans*(ans+1)/2<=n)
        ans++;
    
    int CNT=1;
    for(int i=1;i<=ans;++i)
        for(int j=i+1;j<=ans;++j)
        {
            ff[i].push_back(CNT);
            ff[j].push_back(CNT++);
        }
   cout<<ans<<endl;
   for(int i=1;i<=ans;++i)
   {
       for(int j=0;j<ff[i].size();++j)
           cout<<ff[i][j]<<" ";
    cout<<endl;
    }
    return 0;
    
}
