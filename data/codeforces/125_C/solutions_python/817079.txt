	n = int(cin.read())
	r = 3
	while r*(r-1) <= 2*n:
		r += 1
	r-=1
	
	arr= [[0]*(r-1) for i in xrange(r)]
	
	v = 1
	for i in xrange(r):
		w = v;
		for x in range(i, r-1):
			arr[i][x] = v
			v+=1
		for x in range(i+1, r):
			arr[x][i] = w
			w+=1
