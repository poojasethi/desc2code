import math

n = int(raw_input())
i = n
sum = i*(i+1)/2
while int(sum) > n:
    i -= 1
    sum = i*(i+1)/2

total = i + 1
print total
numlen = total - 1
line = range(1, total)
print ' '.join(map(str, line))
for i in range(1, total):
	last = line[-1]
	for j in range(1, numlen+1):
		if j < i:
			line[j-1] += 1
		elif j > i:
			line[j-1] = last + j - i
	print ' '.join(map(str, line))
