from math import *
n = int(raw_input())
last_free = num_ans = int((1 + sqrt(1+8*n))/2)
solutions = [range(1,num_ans)]
for i in range(num_ans-1):
    s = []
    for j in range(i+1):
        s.append(solutions[j][i])
    for j in range(num_ans - i - 2):
        s.append(last_free)
        last_free += 1
    solutions.append(s)
print len(solutions)
for s in solutions:
    print ' '.join(map(str, s))
