def main():
	n = getLineAs(int)[0]
	w = 2

	while True:
		if (w + 1) * (w + 2) / 2 <= n:
			w += 1
		else:
			break

	ans = array(w + 1, array(w, 0))
	h   = w + 1
	c   = 1
	
	for i in range(h):
		for j in range(i, w):
			ans[i][j] = c
			c += 1

	for i in range(h):
		for j in range(i):
			ans[i][j] = ans[j][i - 1]

	print h
	for i in range(h):
		print toSepStr(ans[i])

	return 0

from sys import stdin
from sys import setrecursionlimit
from copy import deepcopy
from math import sqrt
from itertools import permutations
from itertools import combinations

def getline():
	return stdin.readline()

def getLineAs(tp):
	return map(tp, getline().split())

def array(n, init = 0):
	return [deepcopy(init) for i in range(n)]

def count_if(lst, pred):
	ret = 0
	for i in lst:
		if pred(i): ret = ret + 1
	return ret

def toSepStr(lst):
	if len(lst):
		return "".join([str(item) + " " for item in lst])[:-1]
	else:
		return ""

if __name__ == "__main__":
	setrecursionlimit(1024 * 1024)
	main()
