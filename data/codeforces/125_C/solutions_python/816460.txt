import math
n = int(raw_input())

d = [[1,2], [1,3], [2,3]]

def print_sol(d):
	print len(d)
	for x in d:
		y = map(str, x)
		print ' '.join(y)

if n < 6 and n >= 3:
	print_sol(d)

else:
	curN = 3
	curStep = 3
	curN += curStep
	curStep += 1


	while curN <= n:
	
		d.append([])

		for i in range(0, curStep-1):
			d[i].append(curN-i)
			d[-1].append(curN-i)


		curN += curStep
		curStep += 1

	print_sol(d)
