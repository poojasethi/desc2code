import sys

#sys.stdin = open('in.txt', 'r')

a = []
i = 3
while True:
	if i * (i - 1) / 2 > 10000:
		break
	a.append((i, i * (i - 1) / 2))
	i += 1

n = int(sys.stdin.readline())

days, hobs = 0, 0
for x, y in a:
	if y <= n:
		hobs = y
		days = x
	else:
		break

print days
b, c = [[] for _ in xrange(days)], 1
for i in xrange(days - 1):
	for j in xrange(i + 1, days):
		b[i].append(str(c))
		b[j].append(str(c))
		c += 1

for i in xrange(days):
	print ' '.join(b[i])
