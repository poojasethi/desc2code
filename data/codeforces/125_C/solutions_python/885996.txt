n = int(raw_input())

x = 1

while (x * (x + 1) / 2) <= n:
  x += 1

days = []
for i in xrange(0, x+1):
  days.append([])

guy = 1
for i in xrange(1, x+1):
  for j in xrange(i+1, x+1):
    days[i].append(guy)
    days[j].append(guy)
    guy += 1

print x
for i in xrange(1, x+1):
   print ' '.join(str(_) for _ in days[i])
