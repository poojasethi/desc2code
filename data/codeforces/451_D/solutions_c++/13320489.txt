#include<iostream>
#define ll long long
using namespace std;
int main()
{
    ll q[4]={0};
    char c;
    for(ll i=0;cin>>c;i++)
    {
        if(c=='a')
            q[i%2]++;
        if(c=='b')
            q[i%2+2]++;
    }
    cout<<q[0]*q[1]+q[2]*q[3]<<" "<<q[0]*(q[0]+1)/2+q[1]*(q[1]+1)/2+q[2]*(q[2]+1)/2+q[3]*(q[3]+1)/2<<endl;
    return 0;
}
