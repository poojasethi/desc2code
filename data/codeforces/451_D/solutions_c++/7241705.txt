#include <algorithm>
#include <cstdio>
using namespace std;

int cnt[2][2];

int main() {
	long long resi = 0, resp = 0;
	char c;
	for (int i = 0; scanf(" %c", &c) != EOF; ++i) {
		++cnt[i % 2][c - 'a'];
		resi += cnt[i % 2][c - 'a'];
		resp += cnt[1 - i % 2][c - 'a'];
	}
	printf("%lld %lld\n", resp, resi);
	return 0;
}
