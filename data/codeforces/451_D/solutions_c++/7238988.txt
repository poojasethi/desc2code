#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
__int64 c[2][2],even,odd;
int main()
{
    string a;
    cin>>a;
    for(int i=0;i<a.length();i++)
    {
        c[a[i]-'a'][i%2]++;
        even+=c[a[i]-'a'][!(i%2)];
        odd+=c[a[i]-'a'][i%2];
    }
    cout<<even<<' '<<odd<<endl;
}
