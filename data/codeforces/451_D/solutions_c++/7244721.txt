#include <iostream>
#include <string>
using namespace std ;
string s ;
long long m [2][2] , ch , nc ;
int main(){
cin >> s ;
for ( int i = 0 ; i < s.size() ; ++i ) {
long long a = 0 , b = 0 ;
if ( s [i] == 'a' ) a = 1 ;
if ( i%2 == 1 ) b = 1 ;
m [a][b]++ ;
nc+=m[a][b] ;
ch+=m[a][(b+1)%2] ;
}
cout << ch << " " << nc ;
return 0 ;
}
