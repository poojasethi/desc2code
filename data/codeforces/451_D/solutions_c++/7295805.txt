#include <iostream>
#include <string>
using namespace std;

string S;
int N;
int pos[2][2];
long long ans[2];

int main()
{
	cin >> S;
	N = S.size();

	for (int i = 0; i < N; i++)
	{
		int val = S[i] - 'a';
		pos[val][i%2]++;
		ans[0] += pos[val][(i+1)%2];
		ans[1] += pos[val][i%2];
	}

	cout << ans[0] << " " << ans[1] << "\n";

	return 0;
}
