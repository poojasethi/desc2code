#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
#define N 100005
char a[N];
int f[2][2];
ll e,o;

int main(){
  int i,x,y;
  // freopen("1.txt","r",stdin);
  scanf("%s",a);
  for (i=0;a[i];++i){
    x=a[i]-'a';
    y=i&1;
    ++f[x][y];
    e+=f[x][!y];
    o+=f[x][y];
  }
  printf("%I64d %I64d\n",e,o);
  return 0;
}
