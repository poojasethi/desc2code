#include <iostream>

using namespace std;

typedef long long ll;

string s;
int cnt[2][2];

int main() {
	cin >> s;
	int n = s.size();
	for (int i = 0; i < n; i++)
		cnt[s[i] - 'a'][i % 2]++;
	ll odd = 0, even = 0;
	for (int i = 0; i < n; i++) {
		even += cnt[s[i] - 'a'][(i % 2) ^ 1];
		odd += cnt[s[i] - 'a'][i % 2];
		cnt[s[i] - 'a'][i % 2]--;
	}
	cout << even << " " << odd << endl;
}
