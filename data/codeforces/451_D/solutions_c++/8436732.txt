#include<iostream>
#include<cstdio>

using namespace std ;

char c ;
long long d[2][2] , ans[2] ;

int main()
{
  int i=0 ;
  while((c=getchar())!='\n')
    {
      c=c-'a' ;
      d[c][i]++ ;
      ans[0]+=d[c][!i] ;
      ans[1]+=d[c][i] ;
      i=!i ;
    }
  cout << ans[0] << " " << ans[1] ;
  return 0 ;
}
