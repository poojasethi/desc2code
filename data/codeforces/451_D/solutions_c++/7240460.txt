#include <stdio.h>
int main(){
	__int64 res1 = 0, res2 = 0;
	int cnt[2][2] = {{0,0},{0,0}};
	bool i = 0;
	char c;
	while((c = getchar()) != EOF){
		if(c == 'a' || c == 'b'){
			cnt[c & 1][ i ] ++;
			res1 += cnt[c & 1][i ];
			res2 += cnt[c & 1][i = !i];
		}
	}
	printf("%I64d %I64d\n",res2, res1);
	return 0;
}
