#include <stdio.h>
#include <string.h>
const int L = 100000+10;
char str[L];
typedef long long LL;
LL f[5][5];
LL x,y;
int main() {
    scanf("%s",str+1);
    int len = strlen(str+1);
    for(int i = 1 ; i <= len ; ++i) {
        ++f[i%2][str[i]-'a'];
        y += f[i%2][str[i]-'a'];
        x += f[1-i%2][str[i]-'a'];
    }
    printf("%I64d %I64d\n",x,y);
}
