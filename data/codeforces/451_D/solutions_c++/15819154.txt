#include<cstdio>
using namespace std;
char s[100010];
int cnt[2][2],n;
int main()
{
	scanf("%s",s);
	for(n=0;s[n];++n)++cnt[s[n]-'a'][n&1];
	printf("%I64d %I64d\n",1ll*cnt[0][0]*cnt[0][1]+1ll*cnt[1][0]*cnt[1][1],n+1ll*cnt[0][0]*(cnt[0][0]-1)/2+1ll*cnt[1][0]*(cnt[1][0]-1)/2+1ll*cnt[0][1]*(cnt[0][1]-1)/2+1ll*cnt[1][1]*(cnt[1][1]-1)/2);
}
