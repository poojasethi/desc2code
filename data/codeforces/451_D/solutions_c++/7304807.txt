#include <cstdio>
#include <cstring>
typedef long long LL;

LL o,e;
char str[100001];
LL tbl[2][2];
int main(){
  scanf("%s",str);
  int n = strlen(str);
  for(int i=0;i<n;++i){
    int ch = str[i]-'a';
    tbl[ch][i%2]++;
    o+=tbl[ch][!(i%2)];
    e+=tbl[ch][i%2];
  }
  printf("%lld %lld\n",o,e);
  return 0;
}
