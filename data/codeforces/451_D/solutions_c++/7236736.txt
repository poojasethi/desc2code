#include <cstdio>
#include <cstring>

typedef long long ll;

const int N = 100005;

char s[N];

int num[2][2];

int main() {
	scanf("%s", s);
	int len = strlen(s);
	ll ans0 = 0, ans1 = 0;
	for(int i = 0; i < len; ++i) {
		++ans1;
		ans1 += num[s[i] - 'a'][i & 1];
		ans0 += num[s[i] - 'a'][!(i & 1)];
		++num[s[i] - 'a'][i & 1];
	}
	printf("%I64d %I64d\n", ans0, ans1);
	return 0;
}
