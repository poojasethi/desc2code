#include<iostream>
using namespace std;
string s;
long long a[2][500],i,j,k,l,m,n,ans1,ans2;
main()
{
    cin>>s;
    for(int i=0;i<s.size();i++)
    {
        a[i%2][s[i]]++;
        if(i%2==0)
        ans1+=a[0][s[i]],ans2+=a[1][s[i]];
        
        
        if(i%2==1)
        ans1+=a[1][s[i]],ans2+=a[0][s[i]];
        
    }
    cout<<ans2<<" "<<ans1<<endl;
}
