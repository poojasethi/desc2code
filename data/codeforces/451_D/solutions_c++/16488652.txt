#include <bits/stdc++.h>
#define ll long long int
using namespace std;
ll k,sz,ceven[2],codd[2],res0,res1;
int main(){
	string s;
	cin>>s;
	sz=s.size();
	for(int i=0;i<sz;i++){
		res1++;
		k=s[i]-'a';
		if(i%2){
			res1+=codd[k];
			res0+=ceven[k];
			codd[k]++;
		}
		else{
			res0+=codd[k];
			res1+=ceven[k];
			ceven[k]++;
		}
	}
	cout<<res0<<" "<<res1<<endl;
}
