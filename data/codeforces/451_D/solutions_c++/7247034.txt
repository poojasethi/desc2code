#include<cstdio>
#include<cstring>
using namespace std;
typedef long long ll;
ll a,b,l,i,dp[2][30];
char s[100100];
int main()
{
 scanf("%s",s+1);
 l=strlen(s+1);
 for (i=1;i<=l;i++)
  {
 dp[i&1][s[i]-'a']++;
 a+=dp[!(i&1)][s[i]-'a'];
 b+=dp[i&1][s[i]-'a'];
  }
 printf("%I64d %I64d\n",a,b);
 return 0;
}
