#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;

long long a[2][101];
char s[100001];
long long ans1,ans2;

int main()
{
	scanf("%s",s+1);
	int len=strlen(s+1);
	for (int i=1;i<=len;i++)
	{
		a[i&1][s[i]]++;
		ans1+=a[i&1][s[i]]; ans2+=a[(i&1)^1][s[i]];
	}
	printf("%I64d %I64d\n",ans2,ans1);
}
