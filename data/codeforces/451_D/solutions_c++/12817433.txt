#include<iostream>
using namespace std;

long long i,a0,a1,b0,b1; char c;

int main(){
	i=0;
	while(cin >> c){
		if(c=='a'&&i%2==0)a0++;
		if(c=='a'&&i%2==1)a1++;
		if(c=='b'&&i%2==0)b0++;
		if(c=='b'&&i%2==1)b1++;
		i++;
	}
	cout << a0*a1+b0*b1 << " " << a0*(a0-1)/2 + b0*(b0-1)/2 +a0 +b0+a1+b1+ a1*(a1-1)/2 + b1*(b1-1)/2<< endl;
	return 0;
}
