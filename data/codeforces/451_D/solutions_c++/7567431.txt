#include<cstdio>
#include<cstring>
using namespace std;
const int MAXN=100005;
char str[MAXN];
int a[2][2];
int main()
{
    scanf("%s", str);
    int len=strlen(str);
    long long x=0,y=0;
    for(int i=0; i<len; ++i)
    {
        ++a[!(i&1)][str[i]-'a'];
        x+=a[i&1][str[i]-'a'];
        y+=a[!(i&1)][str[i]-'a'];
    }
    printf("%I64d %I64d",x,y);
}
