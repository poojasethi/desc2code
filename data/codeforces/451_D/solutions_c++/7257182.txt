#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;


int s[100005],n;
long long ans[5],cnt[5][5];


int main()
{
	char ch=getchar();
	while(ch!='\n')s[++n]=ch-'a',ch=getchar();
	for(int i=1;i<=n;i++)
	{
		int x=s[i],j=i&1;
		cnt[x][j]++;
		ans[1]+=cnt[x][j];
		ans[0]+=cnt[x][j^1];
	}
	printf("%I64d %I64d\n",ans[0],ans[1]);
	return 0;
}
