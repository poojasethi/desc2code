#include <cstdio>
#include <cstring>
char tch;
long long a1,a2,i,o[2],e[2];
int main()
{
	while(scanf("%c",&tch),tch=='a'||tch=='b')
	{
		tch-='a';++i;
		if(i%2){++o[tch];a1+=o[tch];a2+=e[tch];}
		else{++e[tch];a1+=e[tch];a2+=o[tch];}
	}
	printf("%I64d %I64d\n",a2,a1);
	return 0;
}
