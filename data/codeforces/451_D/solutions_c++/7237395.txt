#include<iostream>
#include<cstdio>

using namespace std;

char s[123456];
int a[2][2];

int main(){
    long long sum1=0,sum2=0;
    gets(s);
    for (int i=0;s[i];i++){
        int x;
        if (s[i]=='a') x=1;
        else x=0;
        ++a[x][i&1];
        sum1+=a[x][i&1];
        sum2+=a[x][(i&1)^1];
    }
    cout<<sum2<<' '<<sum1<<endl;
    return 0;
}
