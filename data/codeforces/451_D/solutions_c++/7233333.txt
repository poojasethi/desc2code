#include<iostream>
#include<cstdio>
#include<cstring>
#define ll long long
using namespace std;
char ch[100005];
ll num[2][2],ans[2];
int main()
{
	scanf("%s",ch);
	int l=strlen(ch);
	for(int i=0;i<l;i++)
	{
		int x=ch[i]-'a';
		++num[1-i%2][x];
		ans[0]+=num[i%2][x];
		ans[1]+=num[1-i%2][x];
	}
	printf("%I64d %I64d",ans[0],ans[1]);
	return 0;
}
