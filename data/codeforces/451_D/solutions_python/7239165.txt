s = raw_input().strip()
i = 0
aodd, aeven, bodd, beven = 0,0,0,0
odd, even = 0,0
for c in s:
	if c == 'a':
		if i % 2 == 0:
			odd += aeven
			even += aodd
			aeven += 1
		else:
			odd += aodd
			even += aeven
			aodd += 1
	else:
		if i % 2 == 0:
			odd += beven
			even += bodd
			beven += 1
		else:
			odd += bodd
			even += beven
			bodd += 1
	i += 1
print even, odd + len(s)
