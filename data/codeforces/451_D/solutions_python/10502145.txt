import sys

s = raw_input().strip()
even = odd = 0
ea = eb = oa = ob = 0

for i in xrange(len(s)):
    if i % 2 == 0:
        if s[i] == 'a':
            ea += 1
            even += oa
            odd += ea
        else:
            eb += 1
            even += ob
            odd += eb
    else:
        if s[i] == 'a':
            oa += 1
            even += ea
            odd += oa
        else:
            ob += 1
            even += eb
            odd += ob
print even, odd
