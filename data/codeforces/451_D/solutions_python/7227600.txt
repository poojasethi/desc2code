
import sys
stdin = sys.stdin

for line in stdin:
    line = line.strip()
    line = [x == 'a' for x in line]
    evenlen_oddlen_0_1 = [[0, 0], [0, 0]]
    re_even = 0
    re_odd = len(line)
    for i, here in enumerate(line):
        toggler = i%2
        re_even += evenlen_oddlen_0_1[1-toggler][here]
        re_odd  += evenlen_oddlen_0_1[toggler][here]
        evenlen_oddlen_0_1[toggler][here] += 1
    print re_even, re_odd
