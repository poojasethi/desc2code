ch = raw_input() 
len = len(ch)
a , b = [0 , 0] , [0 , 0]

now = 0 
ans0 , ans1 = 0 , 0 

for c in ch :
    if c == 'a' :
        ans1 = ans1 + a[now] + 1 
        ans0 = ans0 + a[now ^ 1]
        a[now] += 1 
    else:
        ans1 = ans1 + b[now] + 1 
        ans0 = ans0 + b[now ^ 1]
        b[now] += 1 
    now ^= 1 
print "%d %d" % (ans0 , ans1)
