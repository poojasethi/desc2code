s = raw_input()
oddA, evenA, oddB, evenB = 0, 0, 0, 0
odd, even = 0, 0

for i, x in enumerate(s):
	if x == 'a':
		if i % 2 == 0:
			evenA += 1
			odd += evenA
			even += oddA
		else:
			oddA += 1
			odd += oddA
			even += evenA
	if x == 'b':
		if i % 2 == 0:
			evenB += 1
			odd += evenB
			even += oddB
		else:
			oddB += 1
			odd += oddB
			even += evenB
print even, odd
