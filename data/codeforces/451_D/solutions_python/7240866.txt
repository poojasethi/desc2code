from __future__ import division


s = raw_input().strip()
n = len(s)
ps = {"a": [0, 0], "b": [0, 0]}
cnt = [0, 0]
for i, c in enumerate(s):
    cnt[1] += 1
    if i % 2 == 0:
        cnt[1] += ps[c][0]
        cnt[0] += ps[c][1]
        ps[c][0] += 1
    else:
        cnt[1] += ps[c][1]
        cnt[0] += ps[c][0]
        ps[c][1] += 1
print cnt[0], cnt[1]
