s = raw_input()
odda_cnt, oddb_cnt, evena_cnt, evenb_cnt = 0, 0, 0, 0
odd, even = 0, 0

for idx in xrange(len(s)):
	if s[idx] == 'a':
		if (idx & 1) == 1:
			odda_cnt += 1
			odd += odda_cnt
			even += evena_cnt
		else:
			evena_cnt += 1
			odd += evena_cnt
			even += odda_cnt
	else:
		if (idx & 1) == 1:
			oddb_cnt += 1
			odd += oddb_cnt
			even += evenb_cnt
		else:
			evenb_cnt += 1
			odd += evenb_cnt
			even += oddb_cnt

print even, odd
