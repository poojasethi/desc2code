string=raw_input().strip()
n=len(string)
arr=[0]*len(string)
for i in range(0,n):
    if string[i]=='b':
        arr[i]=1

ea=0
oa=0
ec=[0,0]
oc=[0,0]

for i in range(0,n):
    oa+=1
    if i%2==0:
        oa+=ec[arr[i]]
        ea+=oc[arr[i]]
        ec[arr[i]]+=1
    else:
        oa+=oc[arr[i]]
        ea+=ec[arr[i]]
        oc[arr[i]]+=1
print ea,oa
