x=raw_input()
a1,a0,b1,b0=0,0,0,0
n=len(x)
for i in range(len(x)):
    if x[i]=='a':
        if i%2==0:
            a0+=1
        else:
            a1+=1
    else:
        if i%2==0:
            b0+=1
        else:
            b1+=1


nech=(a1**2+a0**2+b1**2+b0**2-n)/2+n
chet=a0*a1+b0*b1
print chet,nech
