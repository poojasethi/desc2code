#!/usr/bin/env python
#coding:utf-8

def read():
    return raw_input()


def calc(vList):
    n = len(vList)
    total = [0 for i in range(2)]

    even, odd = 0, 0
    totEven, totOdd = 0, 0
    
    isEven = True
    for i in range(n):
        v = vList[i]
        
        if i % 2 == 0:
            # 自分自身の計算は特別に計算
            totEven += (v - 1) * (v / 2) - (v / 2) ** 2 + (v / 2)
            totOdd  += v * ((v + 1) / 2) - ((v + 1) / 2) ** 2 + ((v + 1) / 2) 
            
            # 他の計算
            if isEven:
                totEven += even * ((v + 1) / 2) + odd * (v / 2)
                totOdd  += even * (v / 2)       + odd * ((v + 1) / 2)
                even += v / 2
                odd  += (v + 1) / 2
                
            else:
                totEven += even * (v / 2)       + odd * ((v + 1) / 2)
                totOdd  += even * ((v + 1) / 2) + odd * (v / 2)
                even += (v + 1) / 2
                odd  += v / 2
        
        isEven = isEven == (v % 2 == 0)
    
    return totEven, totOdd


def work(s):
    vList = []

    idx = 0
    while idx < len(s):
        nexIdx = idx
        while nexIdx < len(s) and s[nexIdx] == s[idx]:
            nexIdx += 1

        vList.append(nexIdx - idx)
        
        idx = nexIdx

    even1, odd1 = calc(vList)
    even2, odd2 = calc(vList[1:])

    print even1 + even2, odd1 + odd2


if __name__ == "__main__":
    work(read())
