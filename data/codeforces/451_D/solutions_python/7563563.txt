

def cal (x):
    return x*(x-1)/2

str = raw_input()

aodd=0
bodd=0
aeven=0
beven=0
    
for i in xrange(len(str)):
    if i%2==1:
        if(str[i]=='a'):aodd+=1
        else :bodd = bodd +1
    else:
        if(str[i]=='a') : aeven+=1
        else :beven+=1
print aodd*aeven + bodd*beven, cal (aodd) + cal (aeven) + cal (bodd) + cal (beven) + len(str)
