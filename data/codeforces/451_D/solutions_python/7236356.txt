s = str( raw_input() )
a1, a0, b1, b0 = 0, 0, 0, 0 
n = len( s )
for i in range( n ):
    if ( s[i] == 'a' ):
        if ( i % 2 == 0 ):
            a0 += 1
        else:
            a1 += 1
    else:
        if ( i % 2 == 0 ):
            b0 += 1
        else:
            b1 += 1
odd = ( a1 * a1 + a0 * a0 + b1 * b1 + b0 * b0 - n ) / 2 + n
even = a0 * a1 + b0 * b1
print even, odd
