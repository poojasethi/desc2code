s = raw_input()
n = len(s)
c = [[0, 1, 0, 0] for _ in range(n)]
for i in range(1, n):
    if s[i] == s[i - 1]:
        c[i][0] += c[i - 1][1]
        c[i][1] += c[i - 1][0]
        c[i][2] += c[i - 1][3]
        c[i][3] += c[i - 1][2]
    else:
        c[i][0] += c[i - 1][3]
        c[i][1] += c[i - 1][2]
        c[i][2] += c[i - 1][1]
        c[i][3] += c[i - 1][0]
print sum(c[i][0] for i in range(n)), sum(c[i][1] for i in range(n))
