#!/usr/bin/env python
#coding:utf-8

def read():
    return raw_input()


def calc(vList):
    n = len(vList)
    total = [0 for i in range(2)]

    isEven = True
    for i in range(n):
        v = vList[i]
        
        if i % 2 == 0:
            total[isEven] += v / 2
            total[not isEven] += (v + 1) / 2
        isEven = isEven == (v % 2 == 0)

    even, odd = 0, 0
    
    isEven = True
    for i in range(n):
        v = vList[i]
        
        if i % 2 == 0:
            total[isEven] -= v / 2
            total[not isEven] -= (v + 1) / 2

            # 自分自身の計算は特別に計算
            even += (v - 1) * (v / 2) - (v / 2) ** 2 + (v / 2)
            odd  += v * ((v + 1) / 2) - ((v + 1) / 2) ** 2 + ((v + 1) / 2) 
            
            # 他の計算
            even += total[isEven] * ((v + 1) / 2) + total[not isEven] * (v / 2)
            odd  += total[isEven] * (v / 2)       + total[not isEven] * ((v + 1) / 2)
        
        isEven = isEven == (v % 2 == 0)
    
    return even, odd


def work(s):
    vList = []

    idx = 0
    while idx < len(s):
        nexIdx = idx
        while nexIdx < len(s) and s[nexIdx] == s[idx]:
            nexIdx += 1

        vList.append(nexIdx - idx)
        
        idx = nexIdx

    even1, odd1 = calc(vList)
    even2, odd2 = calc(vList[1:])

    print even1 + even2, odd1 + odd2


if __name__ == "__main__":
    work(read())
