#!/usr/bin/env python

def read():
    return raw_input()


def work(s):

    ansEven, ansOdd = 0, 0
    curEven, curOdd = [0, 0], [0, 0]

    for i in range(len(s)):
        isA = s[i] == 'a'

        ansOdd += 1

        if i % 2 == 0:
            ansEven += curOdd[isA]
            ansOdd += curEven[isA]
            curEven[isA] += 1
        else:
            ansEven += curEven[isA]
            ansOdd += curOdd[isA]
            curOdd[isA] += 1

    print ansEven, ansOdd


if __name__ == "__main__":
    work(read())
