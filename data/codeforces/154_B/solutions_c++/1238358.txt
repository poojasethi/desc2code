#include <cstdio>
#include <vector>
using namespace std;

#define FOR(i,l,r) for(int i=(l);i<(r);++i)
#define FORI(i,c) for(__typeof((c).begin())i=(c).begin();i!=(c).end();++i)

int n, m;
bool on[100005];
int pr[100005];
vector<int> pl[100005];
bool vst[100005];

int main() {
	scanf("%d%d", &n, &m);
	
	for (int i = 2; i <= n; ++i) {
		if (vst[i]) continue;
		for (int j = 1; i*j <= n; ++j) {
			vst[i*j] = true;
			pl[i*j].push_back(i);
		}
	}

	for (; m; --m) {
		char s[2];
		int x;
		scanf("%s%d", s, &x);
		if (s[0] == '+') {
			if (on[x]) puts("Already on");
			else if (x == 1) {
				on[x] = true; puts("Success");
			} else {
				int conf = 0;
				FORI(i, pl[x]) {
					if (pr[*i]) {
						conf = pr[*i];
						break;
					}
				}
				if (conf) printf("Conflict with %d\n", conf);
				else {
					on[x] = true; puts("Success");
					FORI(i, pl[x]) pr[*i] = x;
				}
			}
		} else {
			if (!on[x]) puts("Already off");
			else {
				on[x] = false; puts("Success");
				FORI(i, pl[x]) pr[*i] = 0;
			}
		}
	}
}
