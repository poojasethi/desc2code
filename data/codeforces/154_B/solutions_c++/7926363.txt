#include<iostream>
#include<cstdio>
using namespace std;
int n, m, x;
bool on[100001];
int used[100001];
char ch;
int main()
{
	cin >> n >> m;
	for(int i = 1; i <= m; ++i)
	{
		cin >> ch >> x;
		if(ch == '+')
		{
			if(on[x]) 
			{
				printf("Already on\n");
				continue;
			}
			bool ok = 0;
			for(int i = 2; i * i <= x; ++i)
			{
				if(x % i == 0 && (used[i] || used[x / i]))
				{
					ok = 1;
					printf("Conflict with %d\n", used[i] ? used[i] : used[x / i]);
					break;
				}
			}
			if(used[x] && !ok)
			{
				printf("Conflict with %d\n", used[x]);
				continue;
			}
			if(ok) continue;
			for(int i = 2; i * i <= x; ++i)
				if(x % i == 0)
					used[i] = used[x / i] = x;
			used[x] = x;
			printf("Success\n");
			on[x] = 1;
		}
		else
		{
			if(!on[x])
			{
				printf("Already off\n");
				continue;
			}                
			for(int i = 2; i * i <= x; ++i)
				if(x % i == 0)
					used[i] = used[x / i] = 0;
			printf("Success\n");  
			used[x] = 0;
			on[x] = 0;
		}
	}
}
