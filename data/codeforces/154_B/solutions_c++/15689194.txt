#include <bits/stdc++.h>
#define ll long long
#define mst(a,x) memset(a,x,sizeof(a))
#define For(i,t) for(int i = 0; i < (t); i++)
using namespace std;

const int N = 1e5 + 5;
int n, m;
int active[N];
int flag[N];
vector <int> vec;

void work(int x) {
	vec.clear();
	for(int i = 2; i * i <= x; i++) {
		if(x % i) continue;
		vec.push_back(i);
		while(x % i == 0) x /= i;
	}
	if(x > 1) vec.push_back(x);
}

int main() {
	char op;
	int x;

	scanf("%d%d", &n, &m);
	while(m--) {
		scanf(" %c%d", &op, &x);
		if(op == '+') {
			if(active[x]) {
				puts("Already on");
				continue;
			}
			work(x);
			int fail = 0;
			For(i, vec.size()) if(flag[vec[i]]) fail = flag[vec[i]];
			if(fail) {
				printf("Conflict with %d\n", fail);
				continue;
			}
			For(i, vec.size()) flag[vec[i]] = x;
			active[x] = 1;
			puts("Success");
		}
		else {
			if(!active[x]) puts("Already off");
			else {
				work(x);
				For(i, vec.size()) flag[vec[i]] = 0;
				active[x] = 0;
				puts("Success");
			}
		}
	}

	return 0;
}
