#include <cstdio>
#include <vector>
#include <cstring>
using namespace std;
#define N 100010
int v[N],p[N];
vector<int> a[N];
int main()
{
	for(int i=2;i<N;i++)
		if(v[i]==0)
			for(int j=i+i;j<N;j+=i)
				v[j]|=1,a[j].push_back(i);
	for(int i=2;i<N;i++)
		if(v[i]==0)
			a[i].push_back(i);
	memset(v,0,sizeof(v));
	int n,m,k;
	scanf("%d%d",&n,&m);
	char o[2];
	for(int i=0;i<m;i++)
	{
		scanf("%s %d",o,&k);
		if(o[0]=='+')
		{
			if(v[k]==1) printf("Already on\n");
			else
			{
				int j;
				for(j=0;j<a[k].size();j++)
					if(p[a[k][j]]!=0)
						break;
				if(j==a[k].size())
				{
					for(j=0;j<a[k].size();j++)
						p[a[k][j]]=k;
					v[k]=1,printf("Success\n");
				}
				else
					printf("Conflict with %d\n",p[a[k][j]]);
			}
		}
		else if(o[0]=='-')
		{
			if(v[k]==0) printf("Already off\n");
			else
			{
				for(int j=0;j<a[k].size();j++)
					p[a[k][j]]=0;
				v[k]=0;
				printf("Success\n");
			}
		}
	}
	return 0;
}
