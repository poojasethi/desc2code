#include<bits/stdc++.h>
using namespace std;

int ans[1000005];
bool vis[1000005];
int cnt[1000005];
int arr[1000005];

int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&arr[i]);
		if(arr[i]<=m)
		{
		    vis[arr[i]]=true;
		    cnt[arr[i]]++;
		}
	}
	int lcm=1,answer=0;
	for(int i=1;i<=m;i++)
	{
	    if(vis[i])
	    {
	        for(int j=i;j<=m;j+=i)
	        {
	            ans[j]+=cnt[i];
	        }
	    }
	}
	for(int i=1;i<=m;i++)
	{
	    if(ans[i]>answer)
	    {
	        answer=ans[i];
	        lcm=i;
	    }
	}
	printf("%d %d\n",lcm,answer);
	for(int i=0;i<n;i++)
	{
	    if(lcm%arr[i]==0)
	    {
	        printf("%d ",i+1);
	    }
	}
}
