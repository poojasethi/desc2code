#include<bits/stdc++.h>
using namespace std;

int arr[1000005];
int howmany[1000005];
int divisors[1000005];
int main(){

	memset(howmany,0,sizeof(howmany));
	int n,m;
	int i;
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;i++){
		scanf("%d",&arr[i]);
		if(arr[i]<=m)howmany[arr[i]]++;
	}
	int j;
	memset(divisors,0,sizeof(divisors));
	for(i=1;i<=m;i++){
		for(j=i;j<=m;j+=i){
			divisors[j]=divisors[j]+howmany[i];
		}
	}
	int ind=1;
	int howmanyano=divisors[1];
	for(i=2;i<=m;i++){
		if(divisors[i]>howmanyano){
			howmanyano=divisors[i];
			ind=i;
		}
	}
	printf("%d %d\n",ind,howmanyano);
	for(i=1;i<=n;i++){
		if(ind%arr[i]==0){
			printf("%d ",i);
		}
	}
	return 0;
}
