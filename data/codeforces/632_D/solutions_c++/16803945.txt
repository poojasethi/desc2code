#include <bits/stdc++.h>

using namespace std;

#define N 1000005

int cnt[N], len[N];
int a[N];

main() {
	
	int n, m, x;
	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; i ++) {
		scanf("%d", &a[i]);
		if (a[i] <= m) cnt[a[i]] ++;
	}
	for (int i = 1; i <= m; i ++) {
		for (int j = i; j <= m; j += i) len[j] += cnt[i];
	}
	int rlt = 0, l = 0;
	for (int i = 1; i <= m; i ++) if (rlt < len[i]) rlt = len[i], l = i;
	if (!rlt) return puts("1 0");
	printf("%d %d\n", l, rlt);
	for (int i = 1; i <= n; i ++) if (l % a[i] == 0) printf("%d ", i);
}
