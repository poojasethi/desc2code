#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e6 + 5;
int n, m, l, a[maxn], d[maxn];

int main() {
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; i++) {
		scanf("%d", &a[i]);
		if(a[i] <= m) d[a[i]]++;
	}
	for(int i = m; i >= 1; i--) {
		for(int j = 2 * i; j <= m; j += i) {
			d[j] += d[i];
		}
	}
	l = max_element(d + 1, d + m + 1) - d;
	printf("%d %d\n", l, d[l]);
	for(int i = 1; i <= n; i++) {
		if(l % a[i] == 0) {
			printf("%d ", i);
		}
	}
	puts("");
	return 0;
}
