#include <bits/stdc++.h>
using namespace std;

int freq[1000005];
int z[1000005];
int arr[1000005];
	
int main() {
	int i,n,m,it,j;
	scanf("%d %d",&n,&m);
	for(i=0;i<n;i++){
		scanf("%d",&it);
		arr[i]=it;
		if(it<=m)
			freq[it]++;
	}
	for(i=1;i<=m;i++){
		for(j=i;j<=m;j+=i){
			z[j]+=freq[i];
		}
	}
	int maxi=-1;
	for(i=1;i<=m;i++)
		maxi=max(maxi,z[i]);
	for(i=1;i<=m;i++){
		if(maxi==z[i]){
			cout<<i<<" "<<maxi<<endl;
			for(int k=0;k<n;k++){
				if(i%arr[k]==0)
					cout<<k+1<<" ";
			}
			return 0;
		}
	}
	
	return 0;
}
