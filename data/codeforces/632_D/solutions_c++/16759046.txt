#include<bits/stdc++.h>
using namespace std;
const int N=1000011;
int a[N],cnt[N],n,m;
int ans[N];
int main()
{
	cin>>n>>m;
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		if (a[i]<=m) cnt[a[i]]++;
	}
	for (int i=1;i<=m;i++)
	{
		if (cnt[i]==0) continue;
		for (int j=i;j<=m;j+=i)
		{
			ans[j]+=cnt[i];
		}
	}
	int MAXN=0,MAXL=-1;
	for (int i=1;i<=m;i++)
	{
		if (ans[i]>MAXL)
		{
			MAXL=ans[i];
			MAXN=i;
		}
	}
	cout<<MAXN<<" "<<MAXL<<endl;
	for (int i=1;i<=n;i++)
	{
		if (MAXN%a[i]==0)
		{
			printf("%d ",i);
		}
	}
}
