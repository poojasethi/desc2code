#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e6+5;

int n,m,cnt[maxn],a[maxn],tot[maxn];

int main(){
	
	scanf("%d%d",&n,&m);
	
	for( int i = 1 ; i <= n ; i++ ){
		scanf("%d",&a[i]);
		if( a[i] <= m )
			cnt[a[i]]++;
	}
	
	for( int i = 1 ; i <= m ; i++ )
		for( int j = i ; j <= m ; j += i )
			tot[j] += cnt[i];
	
	int mx = 0 , lcm = -1;
	for( int i = 1 ; i <= m ; i++ )
		if( mx < tot[i] )
			mx = tot[i] , lcm = i;
	
	if( lcm == -1 ){
		printf("1 0\n");
		return 0;
	}
	
	printf("%d %d\n",lcm,mx);
	
	for( int i = 1 ; i <= n ; i++ )
		if( lcm % a[i] == 0 )
			printf("%d ",i);
	
	return 0;
}
