#include<bits/stdc++.h>
using namespace std;
const int maxn = 1e6+5;
int cnt[maxn];
int dp[maxn];
int a[maxn];
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        if(a[i]<=m)cnt[a[i]]++;
    }
    for(int i=m;i;i--)
        for(int j=i;j<=m;j+=i)
            dp[j]+=cnt[i];
    long long ans1=-1,ans2=-1;
    for(int i=1;i<=m;i++)
        if(dp[i]>ans1)
            ans1=dp[i],ans2=i;
    cout<<ans2<<" "<<ans1<<endl;
    for(int i=1;i<=n;i++)
        if(ans2%a[i]==0)
            cout<<i<<" ";
    cout<<endl;

}
