#include<bits/stdc++.h>
using namespace std;
int ar[1000000],cnt[1000000],lcm[1000000],mx=0,hi=1,l,n;

int main(){
	scanf("%d %d",&n,&l);
	for(int i=1 ; i<=n ; i++){
		scanf("%d",&ar[i]);
		if(ar[i]<=l){
			cnt[ar[i]]++;
		}
	}
	for(int i=1 ; i<=l ; i++){
		for(int j=1 ; i*j<=l ; j++){
			lcm[i*j]+=cnt[i];
		}
	}
	for(int i=1 ; i<=l ; i++){
		
		if(lcm[i]>mx){
			mx=lcm[i];
			hi=i;
		}
	}
	printf("%d %d\n",hi,mx);
	for(int i=1 ;i <=n ; i++){
		if(ar[i]<=hi && hi%ar[i]==0){
			printf("%d ",i);
		
		}
	
	}
	return 0;
}
