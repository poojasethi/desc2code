#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int i,j,k,n,m,a[1100001],c[1100001],b[1100001],q[1100001];
int main(){
	scanf("%d%d",&n,&m);memset(c,0,sizeof(c));memset(b,0,sizeof(b));
	for (i=1;i<=n;i++){
		scanf("%d",&a[i]);
		if (a[i]<=m) c[a[i]]++;
	}
	for (k=1,i=1;i<=m;i++){
		for (j=i;j<=m;j+=i) b[j]+=c[i];
		if (b[k]<b[i]) k=i;
	}
	for (i=1,q[0]=0;i<=n;i++)
	 if (k%a[i]==0) q[++q[0]]=i;
	printf("%d %d\n",k,q[0]);
	for (i=1;i<=q[0];i++) printf("%d ",q[i]);
	puts("");
	return 0;
}
