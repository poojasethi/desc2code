#include <stdio.h>
#include <set>
using namespace std;
int C[1000010];
int n, m;
int A[1000010];
int Z[1000010];
int main(){
	scanf("%d %d", &n, &m);
	for(int i=0; i<n; i++){
		scanf("%d", &A[i]);
		if(A[i]<=m) C[A[i]]++;
	}
	for(int i=1; i<=1000; i++){
		for(int j=i; j*i<=m; j++){
			if(i==j) Z[i*j] += C[i];
			else Z[i*j] += (C[i]+C[j]);
		}
	}
	int best = 0, lcm = 1;
	for(int i=1; i<=m; i++){
		if(Z[i] > best){
			best = Z[i];
			lcm = i;
		}
	}
	printf("%d %d\n", lcm, best);
	for(int i=0; i<n; i++){
		if(lcm % A[i] == 0) printf("%d ", i+1);
	}
	printf("\n");
}
