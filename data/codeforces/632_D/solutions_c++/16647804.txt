#include <bits/stdc++.h>
using namespace std;
#define fi first
#define se second
int main() {
	int n,m;
	scanf("%d %d",&n,&m);
	int a[n];
	int cnt[m+1],tot[m+1];
	for (int i = 0; i <= m; i++) cnt[i] = tot[i] = 0;
	for (int i = 0; i < n; i++) {
		scanf("%d",a+i);
		if (a[i] <= m) cnt[a[i]]++;
	}
	int aa=1;
	for (int it = 1; it <= m; it++) {
		for (int j = it; j <= m; j += it) {
			tot[j] += cnt[it];
			if (tot[j] > tot[aa]) aa = j;
		}
	}
	vector <int> ans;
	for (int i = 0; i < n; i++) {
		if (aa%a[i] == 0) ans.push_back(i+1);
	}
	printf("%d %d\n",aa,ans.size());
	for (int i = 0; i < (int)ans.size(); i++) printf("%d ",ans[i]);
	printf("\n");
	return 0;
}
