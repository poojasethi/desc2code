#include<bits/stdc++.h>

using namespace std;

int main(){
	int A[1000009],B[1000009]= {0},n, m,lcm,len=0;
	cin>>n>>m;
	for(int i= 0; i< n; i++){
		scanf("%d", &A[i]);
		if(A[i] > 1000000) continue;
		B[A[i]]++;
	}
	for(int i =m; i>0; i--){
		for(int j = 2*i; j<=m; j+=i){
			B[j] += B[i];
		}
	}
	for(int i =0; i<= m; i++){
		if(len<B[i]){
			len = B[i];
			lcm = i;
		}
	}
	if(len){
		printf("%d %d\n", lcm, len);
		for(int i =0; i< n; i++){
			if(lcm%A[i] == 0)
				printf("%d ",i+1);
		}
		cout<<endl;
		return 0;
	}
	cout<<"1 0"<<endl;
	return 0;
}
