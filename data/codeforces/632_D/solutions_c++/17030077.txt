#include<stdio.h>
#include<string.h>
#define N 1000010

int x[N], num[N], cm[N];

int main ()
{
	int n, m;
	while(scanf("%d %d", &n, &m)!=EOF){
		memset(num, 0, sizeof(num));
		memset(cm, 0, sizeof(cm));
		for(int i=0; i<n; i++){
			scanf("%d", &x[i]);
			if(x[i]<=m)
				num[x[i]]++;
		}
		
		for(int i=1; i<=m; i++){
			for(int j=i; j<=m; j+=i)
				cm[j]+=num[i];
		}
		
		int max=0, ans=0;
		for(int i=1; i<=m; i++){
			if(cm[i]>max){
				max=cm[i];
				ans=i;
			}
		}
		
		if(!max){
			printf("1 0\n");
			continue;
		}
		
		printf("%d %d\n", ans, max);
		for(int i=0; i<n; i++){
			if(x[i]<=ans && ans%x[i]==0)
				printf("%d ", i+1);
		}
		printf("\n");
	}
	
	return 0;
}
