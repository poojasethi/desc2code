#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#include <queue>
#include <cctype>
#include <cmath>
#include <iostream>
#include <map>
#include <set>

using namespace std;

int n,m;
int a[1000005];
int cnt[1000005];
int all[1000005];
int main(void){
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		if(a[i]<=m)cnt[a[i]]++;
	}
	int id=1;
	for(int i=1;i<=m;i++){
		for(int j=i;j<=m;j+=i){
			all[j]+=cnt[i];
		}
		if(all[id]<all[i])id=i;
	}
	vector<int> res;
	for(int i=0;i<n;i++){
		if(a[i]<=id && id%a[i]==0)res.push_back(i);
	}
	printf("%d %d\n",id,res.size());
	for(int i=0;i<res.size();i++){
		printf("%d%c",res[i]+1,i+1==res.size()?'\n':' ');
	}
	return 0;
}
