#include <bits/stdc++.h>
using namespace std;
const int N = 1e6+100;
int a[N],b[N];
int main()
{
    ios_base::sync_with_stdio(false); cin.tie(0);
    int n,m; cin >> n >> m; for(int x=0; x<n; ++x) { cin >> a[x]; if(a[x] <= m) b[a[x]] ++; }
    for(int x=m; x>=1; --x) for(int y=2*x; y<=m; y+=x) { b[y] += b[x]; }
    int ans = 0; int res = 1;
    for(int x=1; x<=m; ++x) { if(ans < b[x]) { ans = b[x]; res = x; } }
    cout << res << " " << ans << endl;
    if(ans) for(int x=0; x<n; ++x) if(res % a[x] == 0) cout << x + 1 << " "; cout << endl;
}
