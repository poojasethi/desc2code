#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define N 1000006
#define pb push_back
int arr[N];
int fact[N];
int exist[N];
vector<int> res;
void factors()
{
	for(int i = 1; i<N; i++)
	{
		if(exist[i])
		{
			for(int j = i; j<N; j+=i)
			{
				fact[j]+=exist[i];
			}
		}
	}
}

int main()
{
	int n, m;
	cin >> n >> m;
	for(int i = 0; i<n; i++)
	{
		scanf("%d", &arr[i]);
		if(arr[i]<=N)
		{
			exist[arr[i]]++;
		}
	}

	factors();

	int maxi = 1;
	for(int i = 1; i<=m; i++)
	{
		if(fact[i]>fact[maxi])
			maxi = i;
	}

	for(int i = 0; i<n; i++)
	{
		if(maxi%arr[i]==0)
		{
			res.pb(i+1);
		}
	}

	cout << maxi << " " << fact[maxi] << endl;
	for(int i = 0; i<fact[maxi]; i++)
	{
		cout << res[i] << " ";
	}
	cout << endl;
}
