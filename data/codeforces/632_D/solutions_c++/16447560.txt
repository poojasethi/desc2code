#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define N 1000005
#define LL long long
#define oo (1<<30)
using namespace std;
int n,m,a[N],b[N],S[N];
int main()
{
	int i,j,p=1;
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;i++){
		scanf("%d",&a[i]);
		if(a[i]<=m) b[a[i]]++;
	  }
	for(i=1;i<=m;i++)
		for(j=i;j<=m;j+=i)
			S[j]+=b[i];
	for(i=2;i<=m;i++)
		if(S[i]>S[p]) p=i;
	printf("%d %d\n",p,S[p]);
	for(i=1;i<=n;i++)
		if(p%a[i]==0)
			printf("%d ",i);
	return 0;
}
