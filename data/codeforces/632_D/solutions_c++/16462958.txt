#include<cstdio>
int a[1000005],b[1000005];
int freq[1000005];
int main() {
    int n,m;
    scanf("%d %d",&n,&m);
    for(int i=1;i<=n;i++) {scanf("%d",&a[i]); if(a[i]<=m) freq[a[i]]++; }
    for(int i=1;i<=m;i++) {
        if(freq[i])
        for(int j=i;j<=m;j+=i) b[j]+=freq[i];
    }
    int lcm=1,item=0;
    for(int i=m;i>=1;i--) {
        if(b[i]>=item) {
            item=b[i];
            lcm=i;
        }
    }
    printf("%d %d\n",lcm,item);
    int j=1;
    while(j<=n && lcm%a[j]!=0 ) j++;
    if(a[j]) printf("%d",j);
    for(int i=j+1;i<=n;i++) {
        if(lcm%a[i]==0) printf(" %d",i);
    }
    if(a[j]) printf("\n");
    return 0;
}
