#include <bits/stdc++.h>
using namespace std;

#define rep(i,a,n) for(int (i)=(a);(i)<(int)(n);(i)++)
#define foreach(e,x) for(__typeof(x.begin()) e=x.begin();e!=x.end();++e)
#define SZ(x) ((int) (x).size())

const int M = 1000000 + 5;
int n,m,ans,id;
int cnt[M],d[M],a[M];
int main(){
    //freopen("D.in","r",stdin);
    scanf("%d%d",&n,&m);
    rep(i,0,n) {
        scanf("%d",&a[i]);
        if(a[i]<=m) cnt[a[i]]++;
    }
    rep(i,1,m+1)
        for(int j=i;j<=m;j+=i)
            d[j]+=cnt[i];   
    rep(i,1,m+1) if(ans<d[i]) ans=d[i], id=i;
	if(id==0) {
		printf("1 0\n");
		return 0;
	}
    printf("%d %d\n",id,ans);
    rep(i,0,n) if(id%a[i]==0) printf("%d ",i+1);
    printf("\n");
    return 0;
}
