#include<stdio.h>
#include<string.h>
int a[1111111],num[1111111];
int main()
{
	int i,n,m,j,flag=0;
	while(scanf("%d%d",&n,&m)==2)
	{
		memset(num,0,sizeof(num));
		flag=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			if(a[i]<=m){flag=1;
			num[a[i]]++;}
		}
		if(!flag){printf("1 0\n");continue;}
		for(i=m;i>=1;i--)
		{
			for(j=i*2;j<=m;j+=i)
			{
				num[j]+=num[i];
			}
		}
		int ans=0,pos=1;
		for(i=1;i<=m;i++)
		{
			if(num[i]>ans){
				ans=num[i];
				pos=i;
			}
		}
		printf("%d %d\n",pos,ans);
		for(i=0;i<n;i++)
		{
			if(pos%a[i]==0){
				printf("%d ",i+1);
			}
		}
		printf("\n");
	}
}
