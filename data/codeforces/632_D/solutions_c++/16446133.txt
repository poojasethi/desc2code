#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

const int limite=2000000;

int n,m;
int v[limite];
int ap[limite];
int div[limite];

int main()
{
  scanf("%d%d",&n,&m);
  for (int i=1;i<=n;i++) {
    scanf("%d",&(v[i]));
    if (v[i]<=m)
      ap[v[i]]++;
  }
  for (int i=1;i<=m;i++)
    for (int j=i;j<=m;j+=i)
      div[j]+=ap[i];
  int l=1;
  for (int i=2;i<=m;i++)
    if (div[i]>div[l])
      l=i;
  vector<int> sol;
  for (int i=1;i<=n;i++)
    if (l%v[i]==0)
      sol.push_back(i);
  cout<<l<<" "<<int(sol.size())<<endl;
  for (int i=0;i<int(sol.size());i++) {
    if (i>0) printf(" ");
    printf("%d",sol[i]);
  }
  printf("\n");
}
