#include <iostream>

using namespace std;

const int MAX=1000001;

int n, m, arr[MAX], cnt[MAX], brr[MAX], nok=1, ans, mx;

int main()
{
    ios::sync_with_stdio(0);
    cin >> n >> m;
    for (int i=0; i<n; ++i)
    {
        cin >> arr[i];
        if(arr[i]<=m) cnt[arr[i]]++;
    }

    for (int i=1; i<=m; ++i)
        for (int j=1; j<=m/i; ++j)
            brr[i*j]+=cnt[i];

    for (int i=1; i<=m; ++i)
        if(brr[i]>mx) mx=brr[i], nok=i;

    for (int i=0; i<n; ++i)
        if(nok%arr[i]==0) ans++;

    cout << nok << " " << ans << endl;
    for (int i=0; i<n; ++i)
        if(nok%arr[i]==0) cout << i+1 << " ";
    return 0;
}
