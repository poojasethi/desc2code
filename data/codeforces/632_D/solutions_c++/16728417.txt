#include <bits/stdc++.h>

typedef long long ll;
using namespace std;
ifstream fin("code.in");
ll x,L;
ll N,M,A[1000100],a,cnt[1000100],l[1000100];
int main(){
	ios::sync_with_stdio(0);
	cin.tie(0);
	cin >> N >> M;
	for(ll i = 0;i<N;i++){
		cin >> x;
		A[i] = x;
		if(x <= M){	
			cnt[x]++;
		}
	}
	
	if(N == 0){
		cout << 1 << 0;
		return 0;
	}
	
	for(ll i = 1;i<=M;i++){
		if(cnt[i]){
			for(ll j = i;j<=M;j+=i) l[j]+=cnt[i];
		}
	}
	
	L = 1;
	x = 0;
	for(ll i = 1;i<=M;i++){
		if(x < l[i]){
			x = l[i];
			L=i;
		}
	}
	cout << L << ' ' << x << '\n';
	
	if(x)
		for(ll i = 0;i<N;i++) if(L >= A[i] && L%A[i] == 0) cout << i+1 << ' ';
	
	return 0;
}
