#include<bits/stdc++.h>
using namespace std;
const int maxn = 1e6+5;
int a[maxn];
int cnt[maxn], tmp[maxn];
int n, m;
int main()
{
	scanf("%d%d", &n, &m);
	memset(cnt, 0, sizeof cnt);
	memset(tmp, 0, sizeof tmp);
	for(int i = 1;i<=n;i++)
	{
		scanf("%d", &a[i]);
		if(a[i] <= m)	cnt[a[i]]++;
	}
	for(int i = 1;i<=m;i++)
	{
		for(int j = i;j<=m;j+=i)
		{
			if(cnt[i])	tmp[j] += cnt[i];
		}
	}
	int maxx = -1;
	int k;
	for(int i = 1;i<=m;i++)
	{
		if(maxx < tmp[i])
		{
			maxx = tmp[i];
			k = i;
		}
	}
	printf("%d %d\n", k, maxx);
	for(int i = 1;i<=n;i++)
	{
		if(k % a[i] == 0)
		printf("%d ", i);
	}
	printf("\n");
	return 0;
}
