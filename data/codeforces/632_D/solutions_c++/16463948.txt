#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cmath>
using namespace std;
int n,m,i,j,p,x,y,a[1000001],b[1000001],num[1000001];
int main()
{
		scanf("%d%d\n",&n,&m);
		for (i=1;i<=n;i++) 
		{
			scanf("%d",&a[i]);
			b[i]=a[i];
		}
		sort(b+1,b+n+1);
		for (i=1;i<=n+1;i++)
			if (b[i]!=b[i-1])
			{
				if (x) for (j=x;j<=m;j+=x) num[j]+=y;
				x=b[i];
				y=1;
				if (b[i]>m) break;
			}
			else y++;
		for (i=1;i<=m;i++)
			if (num[i]>num[p]) p=i;
		if (!num[p]) printf("1 0");
		else 
		{
			printf("%d %d\n",p,num[p]);
			for (i=1;i<=n;i++)
				if (p%a[i]==0) printf("%d ",i);
		}
}
