#include <bits/stdc++.h>

using namespace std;

int n , m ;
int c[1001001] , a[1001001] , lcm[1001001];


int main()
{
    scanf("%d %d",&n,&m);
    for(int i=1;i<=n;i++){
        scanf("%d",a+i);
        if( a[i]<=m )
        c[ a[i] ]++;
    }

    for(int i=1;i<=m;i++)
        for(int k=i;k<=m;k+=i)
        lcm[k]+=c[i];

    int K=-100,D;
    for(int i=1;i<=m;i++)
    if( lcm[i]>K ){
        K=lcm[i];
        D=i;
    }
    printf("%d %d\n",D,K);

    for(int i=1;i<=n;i++)
        if( D%a[i]==0 )
        printf("%d%c",i,( K-->0 )?' ':'\n' );

    return 0;
}
