#include<stdio.h>
const int maxn=1E6+77;

int N,M,A[maxn],numcnt[maxn],anscnt[maxn];

void printint(int v){
	static bool first=true;
	if(first){
		printf("%d",v);
		first=false;
	}else{
		printf(" %d",v);
	}
}

int main(){
	scanf("%d%d",&N,&M);
	for(int i=1;i<=N;++i){
		int t;
		scanf("%d",&t);
		A[i]=t;
		if(t<=M) ++numcnt[t];
	}
	for(int i=1;i<=M;++i) if(numcnt[i]){
		for(int j=i;j<=M;j+=i) anscnt[j]+=numcnt[i];
	}
	int maxval=-1,p;
	for(int i=1;i<=M;++i){
		if(anscnt[i]>maxval){
			maxval=anscnt[i];
			p=i;
		}
	}
	printf("%d %d\n",p,anscnt[p]);
	for(int i=1;i<=N;++i) if(!(p%A[i])) printint(i);
	printf("\n");
}
