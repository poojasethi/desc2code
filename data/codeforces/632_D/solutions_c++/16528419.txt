#include<iostream>
#include<stdio.h>
using namespace std;

const int Maxn = 1000000 + 10;
int lcm[Maxn], cnt[Maxn] = {0};

int main(){
	int n , m;
	int a[Maxn];
		
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
		if (a[i] <= m)
			cnt[a[i]]++;
	}
	
	for (int i = 1; i <= m; i++)
		for (int j = i; j <= m; j+=i)
			lcm[j] += cnt[i];

	int maxl = -1 , maxk = -1; 
	for ( int i = 1; i <= m; i++)
		if (maxk < lcm[i]) {
			maxl = i , maxk = lcm[i];
		}

	if (maxk == 0)
		maxl = 1;
	cout << maxl << " "<< maxk <<endl;
	
	for(int i = 0 ;i < n; i++)
		if (maxl % a[i] == 0)
			printf("%d " , i + 1);
			//cout << i + 1 << " " ;
	if (maxk != 0)
		cout << endl;
	return 0;
}
