#include <algorithm>
#include <iostream>			
#include <cstdio>
using namespace std;
int a[1000010],num[1000010],sum[1000010],i,j,n,m,ans=1;
int main()
{
		scanf("%d%d\n",&n,&m);
		for (i=1;i<=n;++i)
		{
			scanf("%d",&a[i]);
			if (a[i]<=m) num[a[i]]++;
		}
		for (i=1;i<=m;++i)
		if (num[i])
			for (j=i;j<=m;j+=i) sum[j]+=num[i];
		for (i=1;i<=m;++i)
			if (sum[i]>sum[ans]) ans=i;
		printf("%d %d\n",ans,sum[ans]);
		for (i=1;i<=n;++i)
			if (ans%a[i]==0) printf("%d ",i);
}
