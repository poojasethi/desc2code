#include<iostream>
#include<cstdio>
using namespace std;
const int maxn=1111111;
int a[maxn],cnt[maxn];
int num[maxn];
int main()
{
	int n,m;cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		if(a[i]<=m)cnt[a[i]]++;
	}
	for(int i=1;i<=m;i++){
		for(int j=i;j<=m;j+=i){
			num[j]+=cnt[i];
		}
	}
	int mm=0,ans=0;
	for(int i=1;i<=m;i++){
		if(num[i]>ans){
			mm=i;
			ans=num[i];
		}
	}
	if(mm==0){
		printf("1 0\n");
		return 0;
	}
	printf("%d %d\n",mm,ans);
	int flag=1;
	for(int i=0;i<n;i++)
	{
		if(mm%a[i]==0){
			if(flag)flag=0,printf("%d",i+1);
			else printf(" %d",i+1);
		}
	}
	printf("\n");
}
