#include <cstdio>
#include <cstring>

int main()
{
	int n, m, max = 0, ind = 1;
	scanf("%d %d", &n, &m);
	int num[n], count[m+1], length[m+1];
	memset(count, 0, sizeof(count));
	memset(length, 0, sizeof(length));
	for (int i = 0; i < n; i++)
	{
		scanf("%d", &num[i]);
		if (num[i] <= m)
		{
			count[num[i]]++;
		}
	}
	for (int i = 1; i <= m; i++)
	{
		if (count[i] != 0)
		{
			for (int j = 1; j <= m/i; j++)
			{
				length[i*j] += count[i];
				if (length[i*j] > max)
				{
					max = length[i*j];
					ind = i*j;
				}
			}
		}
	}
	printf("%d %d\n", ind, max);
	for (int i = 0; i < n; i++)
	{
		if (ind % num[i] == 0)
		{
			printf("%d ", i+1);
		}
	}
	printf("\n");
	
}
