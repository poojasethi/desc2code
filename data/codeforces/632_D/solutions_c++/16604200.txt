#include <iostream>
#include <cstdio>
using namespace std;
const int maxx=(int)1e6+10;
int n,m,i,j,a[maxx],cnt[maxx],num[maxx];

int main(int argc, char *argv[]) {
	while (scanf("%d%d",&n,&m)==2) {
		for(i=0;i<=m;i++) cnt[i]=num[i]=0;
		
		for(i=1;i<=n;i++){
			scanf("%d",&a[i]);
			if(a[i]<=m) cnt[a[i]]++;
		}
		for(i=1;i<=m;i++){
			for(j=i;j<=m;j+=i){
				num[j]+=cnt[i];
			}
		}
		int index=0,maxv=0;
		for(i=1;i<=m;i++){
			if(num[i]>maxv){
				maxv=num[i];
				index=i;
			}
		}
		if(maxv==0) {
			printf("1 0\n");
			continue;
		}
		printf("%d %d\n",index,maxv);
		for(i=1;i<=n;i++){
			if(index%a[i]==0){
				printf("%d ",i);
			}
		}
		printf("\n");
	}
	return 0;
}
