#include<bits/stdc++.h>
using namespace std;
const int N=1e6+2;
int a[N],cnt[N];int sum[N];
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++){
        scanf("%d",&a[i]);
        if(a[i]<N)cnt[a[i]]++;
    }
    for(int i=0;i<N;i++){
        if(cnt[i]==0)continue;
        for(int j=i;j<N;j+=i)
            sum[j]+=cnt[i];
    }
    int maxn=0,pos=1;
    for(int i=1;i<=m;i++){
        if(sum[i]>maxn)maxn=sum[i],pos=i;
    }
    cout<<pos<<' '<<maxn<<endl;
    for(int i=1;i<=n;i++){
        if(pos%a[i]==0)cout<<i<<' ';
    }
    return 0;
}
