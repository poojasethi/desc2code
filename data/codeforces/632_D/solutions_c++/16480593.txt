#include <bits/stdc++.h>

#define fi(a,b,c) for(int a=b; a<=c; a++)

const int NMAX = (int)(1e6) + 10;

int n, m;
int ans[NMAX], cou[NMAX], result, a[NMAX];

using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    cin >> n >> m;
    fi(i, 1, n) {cin >> a[i]; if (a[i] <= m) cou[a[i]]++; }
    fi(i, 1, m)   ans[i] = 0;
    fi(i, 1, m)
        if (cou[i])
        for (int j = 1; (j* i) <= m; j++)
               ans[j * i]+= cou[i];


    result = 1;
    fi(i, 1, m)  if (ans[result] < ans[i])  result = i;

    cout <<result <<' '<<ans[result]<<endl;
    fi(i, 1, n) if (!(result  % a[i]) && a[i] <= m) cout <<i<<' ';
}
