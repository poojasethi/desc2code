#include<stdio.h>
#include<iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;
const int N=1000001;
int a[N],b[N],c[N],n,m;
int suan(int x,int y){if(x%y)return suan(y,x%y);return y;}
int main(void)
{
	int i,j,ma,mai;ma=0;mai=1;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++){scanf("%d",a+i);if(a[i]<=m)b[a[i]]++;}
	for(i=1;i<=m;i++)
	for(j=i;j<=m;j+=i)c[j]+=b[i];
	for(i=1;i<=m;i++)if(c[i]>ma){ma=c[i];mai=i;}
	printf("%d %d\n",mai,ma);
	for(i=1;i<=n;i++)if(mai%a[i]==0){printf("%d",i);break;}
	for(i++;i<=n;i++)if(mai%a[i]==0)printf(" %d",i);
	return 0;
}
