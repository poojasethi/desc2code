#!/usr/bin/env python
# coding=utf-8
from __future__ import division, print_function
from sys import stdin
from fractions import Fraction

def compute(b, w, s):
    res = []
    if 2*b < s:
        print("NO")
        return
    res.append("YES\n")

    left = Fraction(1)
    single = Fraction(b, s)
    bottle = 1
    cnt = 0
    for i in range(s):
        this = single
        while this > 0:
            p = min(left, this)
            res.append("{0} {1:.6f} ".format(bottle, float(w*p)))
            left -= p
            this -= p
            cnt += 1
            if cnt > 2:
                print("NO")
                return
            if left == 0:
                left = Fraction(1)
                bottle += 1
                cnt = 0
        res.append("\n")
    print("".join(res), end="")

if __name__ == "__main__":
    b, w, s= map(int, stdin.readline().split())
    compute(b, w, s)
