#!/usr/bin/python
import itertools, math, os, random, re, sys
random.seed (1234)
while True:
	s = sys.stdin.readline ().strip ()
	if s == '':
		break
	n, w, m = [int (x) for x in s.split ()]
	a = [0.0 for i in range (m)]
	b = [[] for i in range (m)]
	ok = True
	for i in range (n):
		t = w * m
		k = 0
		for j in range (m):
			if t == 0:
				break
			if a[j] < w * n:
				v = min (t, w * n - a[j])
				a[j] += v
				t -= v
				b[j] += [(i + 1, v)]
				k += 1
		if k > 2:
			ok = False
		assert t == 0
	if ok:
		print 'YES'
		for j in range (m):
			for i, v in b[j]:
				print i, float (v) / float (m),
			print
	else:
		print 'NO'
