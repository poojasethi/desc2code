n,w,m=map(int,raw_input().split())
v,w=n*w,w*m
a=[[] for _ in range(m)]
i,j,k,y=1,w,0,1
S=lambda x:'%d %f'%(x[0],x[1]/1./m)
for x in a:
  t=v
  while t:
    d=min(t,j)
    x+=[(i,d)]
    j-=d
    t-=d
    k+=1
    if k>2:
      y=0
    if j<1:
      i+=1
      j=w
      k=0
if y:
  print 'YES'
  for x in a:
    print ' '.join(map(S,x))
else:
  print 'NO'
