using namespace std;

#define mp(a,b) make_pair(a,b)

#define EPS 1e-6

int main(){
	int n, m, w;
	cin >> n >> w >> m;
	if (2*n < m){
		puts("NO");
		return 0;
	}
	if (n < m){
		int gc = __gcd(n, m-n);
		if (gc != m-n){
			puts("NO");
			return 0;
		}
	}
	puts("YES");
	int cur = 1;
	double undone = 0;
	for (int i=0; i<m; i++){
		double left = (double)w*n/m;
		if (undone > EPS){
			printf("%d %.10lf ", cur, undone);
			left -= undone;
			undone = 0;
			cur++;
		}
		while (left > EPS){
			if (w <= left + EPS){
				printf("%d %.10lf ", cur, (double)w);
				left -= w;
				cur++;
			} else {
				undone = w - left;
				printf("%d %.10lf ", cur, left);
				left = 0;
			}
		}
		puts("");
	}
	return 0;
}
