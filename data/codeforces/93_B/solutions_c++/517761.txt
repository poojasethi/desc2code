#include <iostream>
#include <fstream>

using namespace std;

int n, m, w;

bool run(bool print)
{
	double eachcup = (double) w * (double) n / (double) m;
	int now = 1;
	double rest = w;
	int count = 0;
	for (int i = 1; i <= m; ++ i)
	{
		double cup = eachcup;
		while (cup > 1e-9)
		{
			double v = min(rest, cup);
			rest -= v, cup -= v;
			if (print)
				printf("%d %.6lf%c", now, v, (cup > 1e-9) ? ' ' : '\n');
			if (++ count > 2)
				return 0;
			if (rest < 1e-9)
			{
				++ now;
				rest = w;
				count = 0;
			}
		}
	}
	return 1;
}

int main()
{
	cin >> n >> w >> m;
	if (run(0))
	{
		puts("YES");
		run(1);
	}
	else
		puts("NO");
}
