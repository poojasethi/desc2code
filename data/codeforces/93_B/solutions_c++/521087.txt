#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;

int a[10005];
int n,m,w,cnt;
bool flag;

int main()
{
//	freopen("input.txt","r",stdin);

	scanf("%d%d%d",&n,&w,&m);

	for (int i=1;i<=n;++i) a[i*m]=1;
	for (int i=n;i<=n*m;i+=n) a[i]=1;
	
	for (int i=1;i<=n;++i){
		cnt=0;
		for (int j=(i-1)*m+1;j<=i*m;++j)
		if (a[j]) ++cnt;
		if (cnt>2){
			puts("NO");
			return 0;
		}
	}
	
	memset(a,0,sizeof(a));
	for (int i=1;i<=n;++i)
	for (int j=(i-1)*m+1;j<=i*m;++j) a[j]=i;
	
	puts("YES");
	for (int i=1;i<=m;++i){
		flag=0;
		int st=(i-1)*n+1,en=i*n;
		for (int j=st,k;j<=en;j=k+1){
			k=j;
			while (k<en && a[k+1]==a[j]) ++k;
			if (flag) printf(" ");
			printf("%d %.6lf",a[j],(double)(k-j+1)*w/m);
			flag=1;
		}
		puts("");
	}


	return 0;
}
