#include <cstdio>
using namespace std;

const double EPS = 1e-9;

int main()
{
  int n, w, m;
  scanf("%d %d %d", &n, &w, &m);
  int total = n * w;
  if (m > n && m % (m - n)) {
    puts("NO");
  } else {
    puts("YES");
    double avg = (double) total / m;
    int k = 1;
    double rem = w;
    for (int i = 0; i < m; ++i) {
      double sum = 0;
      bool flag = false;
      while (sum < avg - EPS) {
        if (flag)
          putchar(' ');
        if (rem + sum < avg + EPS) {
          printf("%d %.6lf", k, rem);
          sum += rem;
          ++k;
          rem = w;
        } else {
          printf("%d %.6lf", k, avg - sum);
          rem -= avg - sum;
          sum = avg + EPS;
        }
        flag = true;
      }
      puts("");
    }
  }
  return 0;
}
