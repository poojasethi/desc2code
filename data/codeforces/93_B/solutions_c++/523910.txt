using namespace std;
int main(){
    int n,m,s,t,k;
    double w;
	scanf("%d%lf%d",&n,&w,&m);
	if(m>n*2 ||( m>n && n%(m-n)>0)){
		puts("NO");
		return 0;
	}else puts("YES");
	w/=m;
	s=k=0;
	for(int i=0;i<m;++i){
		int t=n;
		while(t>0){
			if(s==0){++k;s=m;}
			if(s>=t){
				printf("%d %.7lf",k,w*t);
				s-=t;
				t=0;
			}else{
				printf("%d %.7lf",k,w*s);
				t-=s;
				s=0;
			}
			if(t>0)putchar(' ');
		}
		puts("");
	}
        //system("pause");
	return 0;
}
