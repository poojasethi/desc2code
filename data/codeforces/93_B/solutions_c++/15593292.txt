#include <bits/stdc++.h>

using namespace std;
const int maxn = 1e2 + 15;
const double eps = 1e-6;
int n , w , m ;
double aver,pr,cur[maxn];
vector < pair < int , double > > ans[maxn];

double Process(double cap , int ptr , int idx){
	double need = aver - cur[ptr];
	double ex = min( need , cap );
	ans[ptr].push_back(make_pair(idx , ex));
	cur[ptr] += ex;
	return cap - ex;
}

int main(int argc,char *argv[]){
	cin >> n >> w >> m;
	aver = (double)w*n/(double)m;
	int ptr = 1;
	for(int i = 1 ; i <= n ; ++ i){
		double have = w;
		for(int j = 1 ; j <= 2 && fabs(have) > eps ; ++ j){
			have = Process( have , ptr , i );
			if(fabs(cur[ptr]-aver) <= eps) ptr++;
		}
	}
	if(ptr != m + 1) printf("NO\n");
	else{
		printf("YES\n");
		for(int i = 1 ; i <= m ; ++ i){
			for(int j = 0 ; j < ans[i].size() ; ++ j){
				if(j) printf(" ");
				printf("%d %.10lf",ans[i][j].first,ans[i][j].second);
			}
			printf("\n");
		}
	}
	return 0;
}
