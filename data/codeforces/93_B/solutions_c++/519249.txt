#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#define eps 1e-9
#define FOR(i,s,e) for (int i=(s); i<(e); i++)
using namespace std;

int n, m, cur, rem, sum ,t;
double w;

int main(){

	scanf("%d%lf%d", &n, &w, &m);

	if (m > n && m % (m-n) != 0) printf("NO\n");
	else{

		printf("YES\n");

		cur = 1;
		rem = m;
		for (int i=0; i<m; i++){
			sum = 0;
			while (sum < n){
				if (rem + sum > n){
					t = n - sum;
					sum += t;
					printf("%d %.9f%c", cur, t * w / (double)m, sum==n ? '\n' : ' ');
					rem -= t;
				}
				else{
					sum += rem;
					printf("%d %.9f%c", cur, rem * w / (double)m, sum==n? '\n': ' ');
					cur++;
					rem = m;
				}
			}
		}

	}

	return 0;
}
