#include <cstdio>
#include <cmath>
using namespace std;

int main(){
	double n, w, m;
	scanf("%lf%lf%lf", &n, &w, &m);
	for (double i=1; i<=n; i++){
		int count=0;
		double li=(i-1)*w, ri=i*w;
		for (double j=1; j<=m; j++){
			double lj=((j-1)*n*w)/m, rj=(j*n*w)/m;
			double l=(li>lj)?li:lj, r=(ri<rj)?ri:rj;
			if (l<r)
				count++;
		}
		if (count>2){
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	for (double j=1; j<=m; j++){
		double lj=((j-1)*n*w)/m, rj=(j*n*w)/m;
		for (double i=1; i<=n; i++){
			double li=(i-1)*w, ri=i*w;
			double l=(li>lj)?li:lj, r=(ri<rj)?ri:rj;
			if (l<r)
				printf("%0.0lf %0.7lf ", i, r-l);
		}
		printf("\n");
	}
}
