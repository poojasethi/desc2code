#include <cstdio>
#include <vector>
using namespace std;

double eps = 1e-9;
vector<pair<int, double> > ans[100];

int main() {
	int n, m, w;
	scanf("%d%d%d", &n, &w, &m);
	
	if (n * 2 < m) {
		printf("NO\n");
		return 0;
	}
	
	double avg = double(n * w) / m;
	double need = avg;
	int k = 0;

	bool valid = true;
	for (int i=0; i<m; ++i) {
		while (need > w - eps) {
			ans[i].push_back(make_pair(++k, w));
			need -= w;
		}
		
		if (need > 1e-7) {
			double left = w - need;
			ans[i].push_back(make_pair(++k, need));
	
			if (left > 1e-7) ans[i + 1].push_back(make_pair(k, left));
			need = avg - left;
			if (need < -1e-7) {
				valid = false;
				break;
			}
		}
		else need = avg;
	}
	
	if (valid) {
		printf("YES\n");
		for (int i=0;i<m; ++i) {
			for (int j=0; j<ans[i].size(); ++j) printf("%d %.6lf ", ans[i][j].first, ans[i][j].second);
			printf("\n");
		}
	}
	else {
		printf("NO\n");
	}
	
	return 0;
}
