#include <cstdio>
#include <vector>

using std::vector;
using std::min;

const double EPS = 1e-9;

int main() {
    int n, w, m;
    scanf("%d %d %d", &n, &w, &m);

    if (n < m && n%(m-n) != 0) {
        puts("NO");
        return 0;
    }

    // 顺序倒
    puts("YES");
    int j = 1; 
    int wj = w*m;
    for (int i = 1; i <= m; ++i) {
        int left = n*w;
        int v = min(wj, left);
        wj -= v, left -= v;
        printf("%d %.6lf", j, v/(double)m);
        if (wj == 0) j++, wj = w*m;
        while (left > 0) {
            v = min(wj, left);
            wj -= v, left -= v;
            printf(" %d %.6lf", j, v/(double)m);
            if (wj == 0) j++, wj = w*m;
        }
        putchar('\n');
    }
    return 0;
}
