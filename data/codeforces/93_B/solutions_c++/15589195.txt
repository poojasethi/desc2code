#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int n,w,m,tem,pre,i,from,j,f[501000],a[60][1005];
double unit;
int main(){
	scanf("%d%d%d",&n,&w,&m);
	tem=n*m;pre=-1;
	unit=(1.0*w)/m;
	for(i=1;i<=tem;++i)f[i]=((i-1)/n)+1;
	for(i=m;i<=tem;i+=m){
		if(f[i]-f[i-m+1]>1){
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	for(i=1;i<=tem;++i){
		from=((i-1)/m)+1;
		a[f[i]][from]++;
	}
	for(i=1;i<=m;++i){
		for(j=1;j<=n;++j){
			if(a[i][j]==0)continue;
			printf("%d %.10lf ",j,a[i][j]*unit);
		}
		printf("\n");
	}
	return 0;
}
