#include <iostream>
#include <cmath>
#include <vector>
#include <utility>
#include <iomanip>
using namespace std;
const double eps=1e-6;
vector <pair<int,double> > list[1000]; 
int n,m,w;
int p=0;
double remain[1000];
int main()
{
	ios::sync_with_stdio(false);
	cin>>n>>w>>m;
	for (int i=0;i<n;i++)
		remain[i]=1;
	for (int i=0;i<m;i++)
	{
		double a=(double(n)/m);
		if (abs(remain[p]-1)>eps)
		{
			a-=remain[p];
			list[i].push_back(make_pair(p+1,remain[p]*w));
			p++;
		}
		if (a<-eps)
		{
			cout<<"NO"<<endl;
			return 0;
		}
		while (a+eps>=1)
		{
			list[i].push_back(make_pair(p+1,w));
			p++;
			a-=1;
		}
		if (abs(a)>eps)
		{
			list[i].push_back(make_pair(p+1,a*w));
			remain[p]-=a;
		}
	}
	cout<<"YES"<<endl;
	for (int i=0;i<m;i++)
	{
		for (int j=0;j<list[i].size();j++)
			cout<<list[i][j].first<<" "<<fixed<<setprecision(6)<<list[i][j].second<<" ";
		cout<<endl;
	}
}
