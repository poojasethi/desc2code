#include <cstdio>

using namespace std;

int main(void)
{
  int n,w,m;
  scanf("%d %d %d",&n,&w,&m);
  if((n<m)&&(m%(m-n)!=0))
  {
    printf("NO\n");
    return 0;
  }
  printf("YES\n");
  int i=1,j=1,k=m,s=m;
  while(j<=m)
  {
    if(s>n)
    {
      s-=n;
      k=s;
      ++j;
      printf("%d %.10lf\n",i,(1-k*1.0/m)*w);
    }
    else if(s==n)
    {
      printf("%d %.10lf\n",i++,k*1.0/m*w);
      s=m;
      k=m;
      ++j;
    }
    else
    {
      printf("%d %.10lf ",i++,k*1.0/m*w);
      k=m;
      s+=m;
    }
  }
  return 0;
}
