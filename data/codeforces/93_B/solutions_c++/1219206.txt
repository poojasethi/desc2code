#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define pi pair<int,int>
#define pb push_back

int n,m,l,k,r,i,j,a[105],p,t,d,e[105],ii,w;
vector<pi> c[105];


int main(){
  
  cin>>n>>w>>m;
  rep(i,n)a[i]=m;d=0;

  rep(i,m){
    int t=n;
    while(t>a[d]){t-=a[d];e[d]++;c[i].pb(pi(d,a[d]));d++;}
    a[d]-=t,e[d]++;
    c[i].pb(pi(d,t));if(a[d]==0)d++;
  }
  
  for(ii=0;ii<m;ii++)if(e[ii]>2)break;
  
  if(ii!=m)puts("NO");
  else{
    puts("YES");
    rep(i,m){
     rep(j,c[i].size())printf("%d %.8lf ",c[i][j].first+1,(double)(c[i][j].second)/m*w);
     cout<<endl;
    }
  }
  
 return 0;
}
