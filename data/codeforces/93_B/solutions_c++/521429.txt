#include<iostream>
#include<vector>
#include<cstdio>
#define Max 55
#define EPS 1e-9

#define PB push_back
#define MP make_pair
#define F first
#define S second

using namespace std;

double d, td, tw, t;
vector< pair<int, double> >  arr[Max];

int main()
{
	int n, w, m, wn, ct;
	bool f = 0;

	scanf("%d %d %d", &n, &w, &m);

	d = 1.0 * w*n / m;

	tw = w;
	wn = 0;
	ct = 0;
	for(int i=0;i<m;i++)
	{
		td = d;

		while(td > EPS)
		{
			if(tw < EPS)
			{
				tw = w;
				wn++;
				ct = 0;
			}

			t = min(tw, td);

			td -= t;
			tw -= t;

			arr[i].PB( MP(wn, t) );
		}
		ct++;

		if(tw > EPS && ct >= 2)
			f = 1;
	}
	if(!f)
	{
		printf("YES\n");

		for(int i=0;i<m;i++)
		{
			for(int j=0;j<(int)arr[i].size();j++)
				printf("%d %.10lf ", arr[i][j].F+1, arr[i][j].S);
			printf("\n");
		}
	}
	else
		printf("NO\n");

	
}
