#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int n,w,m;
    while(scanf("%d%d%d",&n,&w,&m)!=EOF)
    {
        if(n<m && m%(m-n))
        {
            printf("NO\n");
            continue;
        }
        printf("YES\n");
        int now=1, use=0;
        for(int i=1;i<=m;i++)
        {
            int div=0;
            for(int k=0; div<n; k++)
            {
                int cnt=min(n-div,m-use);
                use+=cnt;
                div+=cnt;
                printf(" %d %.12f"+!k,now,cnt*1.0/m*w);
                if(use==m)
                {
                    now++;
                    use=0;
                }
            }
            printf("\n");
        }
    }
    return 0;
}
