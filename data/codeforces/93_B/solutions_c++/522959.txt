#include<cstdio>
#include<vector>

#define mp       make_pair
#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef pair<int,int> pii;

int main(){
	int n,w,m; scanf("%d%d%d",&n,&w,&m);

	vector<pii> ans[50];
	int milk=m,cup=n,i=0,j=0;
	while(i<m && j<n){
		ans[i].push_back(mp(min(milk,cup),j));
		if(milk==cup){
			milk=m;
			cup=n;
			i++;
			j++;
		}
		else if(milk<cup){
			cup-=milk;
			milk=m;
			j++;
		}
		else{ // milk>cup
			milk-=cup;
			cup=n;
			i++;
		}
	}

	bool ok=true;
	int cnt[50]={};
	rep(i,m) rep(j,ans[i].size()) {
		cnt[ans[i][j].second]++;
		if(cnt[ans[i][j].second]>2) ok=false;
	}

	if(!ok) puts("NO");
	else{
		puts("YES");
		rep(i,m){
			rep(j,ans[i].size()){
				printf("%d %.9f%c",ans[i][j].second+1
								  ,(double)ans[i][j].first*w/m
								  ,j<ans[i].size()-1?' ':'\n');
			}
		}
	}

	return 0;
}
