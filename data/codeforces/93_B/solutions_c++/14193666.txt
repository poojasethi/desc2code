#include <iostream>
using namespace std;
double multiplier;

void output(int i, int v, char c) {
	cout.precision(6);
	cout << i << ' ' << fixed << v*multiplier << c;
}

int main() {
	int nbot, ncup;
	cin >> nbot >> multiplier >> ncup;
	if (nbot<ncup && nbot%(ncup-nbot)!=0) {
		cout << "NO";
		return 0;
	}
	else cout << "YES" << '\n';
	multiplier/=ncup;
	int vcup=nbot, vbot=ncup;
	int ibot=1, vleft=vbot, vneed=vcup;
	for (int i=0; i<ncup; i++) {
		vneed=vcup;
		if (vneed>=vleft) {
			output(ibot++,vleft,' ');
			vneed-=vleft;
		}
		while (vneed>=vbot) {
			output(ibot++,vbot,' ');
			vneed-=vbot;
		}
		vleft=vbot-vneed;
		if (vneed!=0) output(ibot,vneed,'\n');
		else cout << '\n';
	}
	return 0;
}
