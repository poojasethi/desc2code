#include<cstdio>
#include<cstring>
using namespace std;

int n,w,m;

int main() {
	scanf("%d%d%d",&n,&w,&m);
	w*=m;
	int avg=w*n/m,cur=0;
	for (int i=0; i<n; i++) {
		int cnt=0;
		if (cur<0) cnt=1;
		cur+=w;
		while (cur>0) {
			cur-=avg; cnt++;
		}
		if (cnt>2) {
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	int now=1; cur=0;
	for (int i=0; i<m; i++) {
		int ti=avg;
		if (cur>0) {
			printf("%d %.9f ",now,cur*1.0/m);
			ti-=cur;
			cur=0;
			now++;
		}
		while (ti>0) {
			if (ti>=w) {
				ti-=w;
				printf("%d %.9f ",now,w*1.0/m);
				now++;
			}
			else {
				cur=w-ti;
				printf("%d %.9f ",now,ti*1.0/m);
				ti=0;
			}
		}
		printf("\n");
	}
}
