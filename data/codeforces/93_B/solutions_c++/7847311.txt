#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define db double
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
db w;
inline void fail() {
	printf("NO\n");
	exit(0);
}
int main() {
	scanf("%d%lf%d",&n,&w,&m);
	if (m>n*2) fail();
	if (n>=m) {
		db ans=1.*n/m;
		printf("YES\n");
		db last=1;
		j=1;
		For(i,1,m) {
			db sheng=ans;
			for (;sheng>1e-8;) {
				if (last>sheng+1e-8) {
					printf("%d %.12lf ",j,sheng*w);
					last-=sheng;
					break;
				} else {
					printf("%d %.12lf ",j,last*w);
					sheng-=last;
					last=1;
					j++;
				}
			}
			printf("\n");
		}
		return 0;
	}
	if (m%(m-n)==0) {
		db ans=1.*n/m;
		printf("YES\n");
		For(i,1,n) printf("%d %.12lf\n",i,ans*w);
		j=1;
		For(i,n+1,m) {
			int A=n/(m-n);
			For(k,1,A) printf("%d %.12lf ",j++,(1-ans)*w);
			printf("\n");
		}
		return 0;
	}
	fail();
	return 0;
}
