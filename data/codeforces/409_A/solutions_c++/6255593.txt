#include <iostream>
#include <string>
using namespace std;

int play(char x) {
	if (x == '8') return 0;
	if (x == '[') return 1;
	return 2;
}

int main() {
	string a,b; cin >> a >> b;
	int n = a.size();

	int score = 0;
	for (int i = 0; i < n; i += 2) {
		int pa = play(a[i]), pb = play(b[i]);
		if (pb == (pa+1)%3) score++;
		if (pa == (pb+1)%3) score--;		
	}
	if (score > 0) cout << "TEAM 1 WINS" << endl;
	else if (score == 0) cout << "TIE" << endl;
	else cout << "TEAM 2 WINS" << endl;


	return 0;
}
