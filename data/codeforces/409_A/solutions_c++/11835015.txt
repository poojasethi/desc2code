#include<iostream>
#include<string.h>
using namespace std;
char a[1001],b[1001];
int w[1001][1001];
int ans;
int main(){
    cin>>a>>b;
    w['8']['[']=1;
    w['[']['8']=-1;
    w['8']['(']=-1;
    w['(']['8']=1;
    w['(']['[']=-1;
    w['[']['(']=1;
    for(int i=1;i<=strlen(a);i+=2)
        ans+=w[a[i-1]][b[i-1]];
    if(ans>0) cout<<"TEAM 1 WINS";
    if(ans==0) cout<<"TIE";
    if(ans<0) cout<<"TEAM 2 WINS";
    return 0;
}
