#include<stdio.h>
int main(){
	char a[22],b[22];
	int i,x,y;
	scanf("%s",a);
	scanf("%s",b);
	for(i=0;a[i]!=0;i=i+2){
		if((a[i]=='8'&&b[i]=='[')||(a[i]=='['&&b[i]=='(')||(a[i]=='('&&b[i]=='8'))
			x++;
		if((b[i]=='8'&&a[i]=='[')||(b[i]=='['&&a[i]=='(')||(b[i]=='('&&a[i]=='8'))
			y++;
	}
	if(x==y)
		printf("TIE\n");
	if(x>y)
		printf("TEAM 1 WINS\n");
	if(y>x)
		printf("TEAM 2 WINS\n");
	return 0;
}
