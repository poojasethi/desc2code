#include<cstdio>
#include<cstring>

char s[1005], t[1005];
int ans = 0;

int calc(char a, char b) {
	if (a == b) return 0;
	if (a == '[' && b == '(' || a == '(' && b == '8' || a == '8' && b == '[') return 1;
	return -1;
}

int main() {
	scanf("%s", s);
	scanf("%s", t);
	for (int i = 0; i < strlen(s); i += 2) {
		ans += calc(s[i], t[i]);
	}
	if (ans == 0) {
		puts("TIE");
	} else {
		printf("TEAM %d WINS\n", ans > 0 ? 1 : 2);
	}
	return 0;
}
