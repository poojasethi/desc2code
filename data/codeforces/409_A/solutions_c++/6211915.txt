#include <iostream>
#include <string>
using namespace std;

string a,b;
int x,y;

int main(){
	cin >> a >> b;
	for(int i=0; i<(int)a.size(); i+=2){
		if(a[i]=='(' && b[i] == '[') y++;
		if(a[i]=='[' && b[i] == '(') x++;
		if(a[i]=='8' && b[i] == '[') x++;
		if(a[i]=='[' && b[i] == '8') y++;
		if(a[i]=='(' && b[i] == '8') x++;
		if(a[i]=='8' && b[i] == '(') y++;
	}
	if(x > y) cout << "TEAM 1 WINS";
	else if(x < y) cout << "TEAM 2 WINS";
	else cout << "TIE";
	return 0;
}
