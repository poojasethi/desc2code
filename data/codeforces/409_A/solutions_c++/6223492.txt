#include<stdio.h>
#include<string.h>
char A[25],B[25];
int main(){
	int i,n,j,k,l,a=0,b=0;
	scanf("%s",A);
	scanf("%s",B);
	n=strlen(A);
	for(i=0;i<n;i+=2){
		if(A[i]=='8'){
			if(B[i]=='[')a++;
			if(B[i]=='(')b++;
		}
		if(A[i]=='['){
			 if(B[i]=='8')b++;
		   if(B[i]=='(')a++;
		}
		if(A[i]=='('){
			 if(B[i]=='8')a++;
			 if(B[i]=='[')b++;
		}
	}
//printf("%d %d",a,b);

	if(a>b)printf("TEAM 1 WINS");
	if(b>a)printf("TEAM 2 WINS");
	if(a==b)printf("TIE");
	return 0;
}
