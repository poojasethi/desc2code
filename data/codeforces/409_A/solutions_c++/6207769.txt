#include <stdio.h>

char s[64],t[64];
int a,b;

int main() 
{
	gets(s);
	gets(t);
	for(int i=0;s[i] && s[i+1];i+=2) {
		if(s[i]=='[' && t[i]=='(') a++;
		if(s[i]=='(' && t[i]=='8') a++;
		if(s[i]=='8' && t[i]=='[') a++;

		if(t[i]=='[' && s[i]=='(') b++;
		if(t[i]=='(' && s[i]=='8') b++;
		if(t[i]=='8' && s[i]=='[') b++;
	}

	if(a>b) puts("TEAM 1 WINS");
	else if(a<b) puts("TEAM 2 WINS");
	else puts("TIE");

	return 0;
}
