#include <iostream>
using namespace std;
int main()
{
	string s,t;
	int i,a=0;
	cin>>s>>t;
	for (i=0;i<s.size();i++)
	{
		if (s[i]=='8'&&t[i]=='[') a++;
		if (s[i]=='8'&&t[i]=='(') a--;
		if (s[i]=='['&&t[i]=='(') a++;
		if (s[i]=='['&&t[i]=='8') a--;
		if (s[i]=='('&&t[i]=='8') a++;
		if (s[i]=='('&&t[i]=='[') a--;
	}
	if (a>0) cout<<"TEAM 1 WINS"<<endl;
	else if (a<0) cout<<"TEAM 2 WINS"<<endl;
	else cout<<"TIE"<<endl;
	return 0;
}
