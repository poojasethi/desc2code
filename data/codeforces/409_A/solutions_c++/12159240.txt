#include<cstdio>
using namespace std;
char s1[22],s2[22];
int main()
{
	int a=0;
	scanf("%s%s",s1,s2);
	for(int i=0;s1[i];i+=2)
	{
		if(s1[i]=='8'&&s2[i]=='[')++a;
		if(s1[i]=='['&&s2[i]=='(')++a;
		if(s1[i]=='('&&s2[i]=='8')++a;
		if(s1[i]=='['&&s2[i]=='8')--a;
		if(s1[i]=='('&&s2[i]=='[')--a;
		if(s1[i]=='8'&&s2[i]=='(')--a;
	}
	if(!a)puts("TIE");
	else printf("TEAM %d WINS\n",a>0?1:2);
}
