#include <iostream>
#include <string>
using namespace std;
int x1, x2,c[1000];
string s1,s2;
int main()
{
    c['8']='['; c['[']='(';c['(']='8';
    cin>> s1 >> s2;
    for(int i = 0; i < s1.length(); i+=2)
        if(c[s1[i]]==s2[i]) x1++;
        else if(c[s2[i]]==s1[i]) x2++;
    if(x1==x2) cout << "TIE" << endl;
    else if (x1>x2) cout << "TEAM 1 WINS"<< endl;
    else cout << "TEAM 2 WINS" << endl;
}
