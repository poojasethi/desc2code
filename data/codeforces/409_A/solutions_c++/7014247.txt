#include<cstring>
#include<cstdio>
int len, a1, a2;
char st1[111], st2[111];
int main()
{
	gets(st1);
	gets(st2);
	len = strlen(st1);
	a1 = 0; a2 = 0;
	for(int i = 0; i < len; i += 2)
	{
		if(st1[i] == '8' and st2[i] == '[' or st1[i] == '[' and st2[i] == '(' or st1[i] == '(' and st2[i] == '8')
			a1++;
		else if(!(st1[i] == st2[i]))
			a2++;
	}
	if(a1 > a2) printf("TEAM 1 WINS\n");
	else if(a1 < a2) printf("TEAM 2 WINS\n");
	else printf("TIE\n");
	fclose(stdin);
	return 0;
}
