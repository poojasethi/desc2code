#include<stdio.h>
int main(){
	int i;
	int w1 = 0, w2 = 0;
	char buf1[21];
	char buf2[21];
	scanf("%s", buf1);
	scanf("%s", buf2);
	for(i = 0; buf1[i]; i+=2){
		if(buf1[i] == buf2[i])
			continue;
		if(buf1[i] > buf2[i]){
			if(buf1[i] - buf2[i] > 50)
				w1++;
			else
				w2++;
		} else {
			if(buf2[i] - buf1[i] > 50)
				w2++;
			else
				w1++;
		}
	}
	printf("%s\n", w1==w2?"TIE":(w1>w2?"TEAM 1 WINS":"TEAM 2 WINS"));
}
