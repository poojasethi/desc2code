#include<iostream>
using namespace std;

int findWin(char x, char y)
{
 if(x==y)
  return 0;
 else if(x=='8' && y=='[' || x=='[' && y=='(' || x=='(' && y=='8')
  return 1;
 return -1;
}
int main(){
 char str1[30], str2[30];
 cin>>str1>>str2;
 int i, sum=0;
 for(i=0; str1[i]!='\0'; i+=2)
  sum+=findWin(str1[i], str2[i]);
 if(sum==0)
  cout<<"TIE"<<endl;
 else if(sum>0)
  cout<<"TEAM 1 WINS"<<endl;
 else
  cout<<"TEAM 2 WINS"<<endl;
 return 0;
}
