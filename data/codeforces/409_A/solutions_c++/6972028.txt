#include <cstdio>

char a[100];
char b[100];

int main() {
    fgets(a,100,stdin);
    fgets(b,100,stdin);
    int i=0;
    int x=0;
    while(a[i]!=0&&a[i]!='\n') {
        if(a[i]!=b[i]) {
            if(a[i]=='8')
                x += (b[i]=='['?1:-1);
            else if(a[i]=='[')
                x += (b[i]=='('?1:-1);
            else
                x += (b[i]=='8'?1:-1);
        }
        i+=2;
    }
    puts(x==0?"TIE":x>0?"TEAM 1 WINS":"TEAM 2 WINS");
    return 0;
}
