#include <iostream>
#include <string>
using namespace std;
string s,t;
int main(){
    cin>>s;
    cin>>t;
    int a=0;
    int b=0;
    int n=s.size();
    for(int i=0;i<n;i+=2){
        if(s[i]!=t[i]) {
            if((s[i]=='['&&t[i]=='(')||(s[i]=='('&&t[i]=='8')||(s[i]=='8'&&t[i]=='[')) a++;
            else b++;
        }
    }
    if(a==b) cout<<"TIE";
    else cout<<(a>b?"TEAM 1 WINS":"TEAM 2 WINS");
	return 0;
}
