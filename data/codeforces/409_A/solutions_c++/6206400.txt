#include<cstdio>
#include<cstring>

char a[55],b[55];

int main()
{
	scanf("%s %s",a,b);
	int N=strlen(a);
	int A=0,B=0;
	for(int i=0;i<N;i+=2)
	{
		if(a[i]=='8' && b[i]=='[')A++;
		if(a[i]=='8' && b[i]=='(')B++;
		if(a[i]=='(' && b[i]=='8')A++;
		if(a[i]=='(' && b[i]=='[')B++;
		if(a[i]=='[' && b[i]=='(')A++;
		if(a[i]=='[' && b[i]=='8')B++;
	}
	if(A>B)printf("TEAM 1 WINS\n");
	else if(B>A)printf("TEAM 2 WINS\n");
	else printf("TIE\n");
}
