#include<bits/stdc++.h>
using namespace std;

int main()
{
    char line1[30], line2[30];
    cin>>line1>>line2;
    int n, m, cnt1=0,cnt2=0;
    n = strlen(line1);

    for (int i = 0; i<n; i+=2)
    {
        if(line1[i]==line2[i]) ;
        else if ((line1[i]=='('&&line2[i]=='8')|| (line1[i]=='8'&&line2[i]=='[')|| (line1[i]=='['&&line2[i]=='(')) cnt1++;
        else cnt2++;
    }


    printf("%s",(cnt1==cnt2?"TIE":(cnt1>cnt2?"TEAM 1 WINS":"TEAM 2 WINS")));

    return 0;
}
