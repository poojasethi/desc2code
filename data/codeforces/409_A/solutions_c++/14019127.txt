#include <iostream>
#include <string>
using namespace std;
int main() {
    int a[3][3] = { {0,1,-1}, {-1,0,1}, {1,-1,0}};
    string s1,s2;
    cin>>s1>>s2;
    int x=0;
    for(int i=0; i<s1.size(); i+=2)
        x += a[s1[i]/41][s2[i]/41];
        
    cout << ((x==0)? "TIE":((x<0)? "TEAM 2 WINS":"TEAM 1 WINS"));
    return 0;
}
