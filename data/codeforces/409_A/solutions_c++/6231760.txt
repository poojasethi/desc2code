# include <stdio.h>


int main()
{
	char s[100], t[110];
	int i, a, b, ans = 0;
	scanf ("%s%s", s, t);
	for (i = 0; s[i]; i+=2)
	{
		if (s[i] == '8') a=0;
		else if (s[i] == '[') a = 1;
		else a = 2;
		
		if (t[i] == '8') b = 0;
		else if (t[i] == '[') b = 1;
		else b = 2;
		
		if (a == b) continue;
		if ((a+1)%3 == b) ans++;
		else ans--;
	}
	if (ans > 0) puts ("TEAM 1 WINS");
	else if (ans < 0) puts ("TEAM 2 WINS");
	else puts ("TIE");
	return 0;
}
