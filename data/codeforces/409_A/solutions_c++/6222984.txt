#include<stdio.h>
#include<string.h>

int main(){
	int x=0,y=0,z,i,j;
	char arr[1000],B[1000];

	scanf("%s",arr);
	scanf("%s",B);

	for(i=0 ; i<strlen(arr) ; i+=2){
		if((arr[i]=='8' && B[i]=='(') || (arr[i]=='(' && B[i]=='[') || (arr[i]=='[' && B[i]=='8'))
			x++;
		
		else if((B[i]=='8' && arr[i]=='(') || (B[i]=='(' && arr[i]=='[') || (B[i]=='[' && arr[i]=='8'))
			y++;
	}


	if(x>y)
		printf("TEAM 2 WINS");
	else if(y>x)
		printf("TEAM 1 WINS");
	else 
		printf("TIE");

	return 0;
}
