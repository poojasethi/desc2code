#include <cstdio>

char s[256], t[256], sc[256][256];
int z;

int main() {
  sc['[']['['] = 0;
  sc['[']['8'] = -1;
  sc['[']['('] = 1;
  sc['8']['['] = 1;
  sc['8']['8'] = 0;
  sc['8']['('] = -1;
  sc['(']['['] = -1;
  sc['(']['8'] = 1;
  sc['(']['('] = 0;
  scanf("%s%s", s, t);
  z = 0;
  for (int i = 0; s[i]; i += 2)
    z += sc[s[i]][t[i]];
  if (z > 0) puts("TEAM 1 WINS");
  else if (z < 0) puts("TEAM 2 WINS");
  else puts("TIE");
  return 0;
}
