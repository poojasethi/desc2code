#include <cstdio>
#include <cstring>
using namespace std;

char s[100],t[100];
int lol[256][256];
int main(){
    gets(s); gets(t);
    lol['[']['('] = lol['(']['8'] = lol['8']['['] = 1;
    lol['(']['['] = lol['8']['('] = lol['[']['8'] = -1;
    int n=strlen(s),stand=0;
    for(int i=0;i<n;i+=2){
        stand+=lol[s[i]][t[i]];
    }
    if (stand>0) puts("TEAM 1 WINS");
    else if (stand<0) puts("TEAM 2 WINS");
    else puts("TIE");
    return 0;
}
