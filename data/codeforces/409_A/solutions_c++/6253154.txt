#include <iostream>
#include <map>
using namespace std;

int main(){
  string a, b;
  map<char,int> m;
  m['8'] = 0;
  m['['] = 1;
  m['('] = 2;
  int s = 0;
  cin >> a >> b;
  for(int i = 0; i < a.size(); i += 2){
    if(a[i] == b[i]) continue;
    if(m[a[i]] == (m[b[i]] + 1) % 3) s--;
    else s++;
  }
  if(s){
    cout << "TEAM " << (s < 0) + 1 << " WINS\n";
  } else cout << "TIE\n";
}
