#include <bits/stdc++.h>
using namespace std;

char s[50] , c[50];
int t1 = 0 , t2 = 0; 

int main(){
	cin >> s >> c;
	for (int i = 0 ; i < strlen(s) ; i += 2){
		if ((s[i] == '8' && c[i] == '[') || (s[i] == '[' && c[i] == '(') || (s[i] == '(' && c[i] == '8')) t1 ++;
		else if (s[i] != c[i]) t2++;
	}
	if (t1 > t2) cout << "TEAM 1 WINS";
	else if (t2 > t1) cout << "TEAM 2 WINS";
	else cout << "TIE";
}
