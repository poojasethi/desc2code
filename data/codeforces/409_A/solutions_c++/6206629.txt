#include <cstdio>

char a[30], b[30];

int main(){
	int an = 0, bn = 0;
	scanf("%s%s", a, b);
	for (int i = 0;; i += 2){
		if (a[i] == 0) break;
		if (a[i] == '8' && b[i] == '[' || a[i] == '[' && b[i] == '(' || a[i] == '(' && b[i] == '8') an++;
		if (a[i] == '[' && b[i] == '8' || a[i] == '(' && b[i] == '[' || a[i] == '8' && b[i] == '(') bn++;
	}
	if (an > bn) printf("TEAM 1 WINS\n");
	else if (bn > an) printf("TEAM 2 WINS\n");
	else printf("TIE\n");
	return 0;
}
