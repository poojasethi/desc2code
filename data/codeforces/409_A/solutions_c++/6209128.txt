#include<stdio.h>
char a[50],b[50];
int sa=0,sb=0,i;
int main()
{
	scanf("%s %s",a,b);
	for(i=0;a[i];i=i+2)
	{
		if(a[i]=='8'&&b[i]=='(') sb++;
		if(a[i]=='8'&&b[i]=='[') sa++;
		if(a[i]=='['&&b[i]=='(') sa++;
		if(a[i]=='['&&b[i]=='8') sb++;
		if(a[i]=='('&&b[i]=='8') sa++;
		if(a[i]=='('&&b[i]=='[') sb++;
	}
	if(sa>sb) printf("TEAM 1 WINS\n");
	else if(sb>sa) printf("TEAM 2 WINS\n");
	else printf("TIE\n");
	return 0;
}
