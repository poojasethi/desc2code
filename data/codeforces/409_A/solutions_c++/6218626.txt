#include <iostream>
using namespace std;
int main()
{
  int T1=0, T2=0;
  string s1,s2;  cin>>s1>>s2;
  for(int i=0;i<s1.size();i+=2)
  {
    if(s1[i]==s2[i])continue;
    else if( (s1[i]=='8'&&s2[i]=='[')||
             (s1[i]=='['&&s2[i]=='(')||
             (s1[i]=='('&&s2[i]=='8') )T1++;
    else T2++;

  }
  if(T1>T2)cout<<"TEAM 1 WINS";
  else if(T2>T1)cout<<"TEAM 2 WINS";
  else cout<<"TIE";
  return 0;
}
