#include<iostream>
#include<string>
using namespace std;
string x,z;
int a,s,i;
main(){
cin>>x>>z;
for(i=0;i<x.size();i+=2){
if(x[i]=='8' && z[i]=='[') a++;
if(x[i]=='[' && z[i]=='(') a++;
if(x[i]=='(' && z[i]=='8') a++;
if(z[i]=='8' && x[i]=='[') s++;
if(z[i]=='[' && x[i]=='(') s++;
if(z[i]=='(' && x[i]=='8') s++;
}
if(a>s) cout<<"TEAM 1 WINS";
if(a==s) cout<<"TIE";
if(a<s) cout<<"TEAM 2 WINS";
}
