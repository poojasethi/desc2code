#include<stdio.h>
char a[21],b[21];
int i,x,y;
main()
{
    scanf("%s %s",a,b);
    for(i=0;a[i]!=0;i++)
    {
        if((a[i]=='8'&&b[i]=='[')||(a[i]=='['&&b[i]=='(')||(a[i]=='('&&b[i]=='8'))x++;
        else if((a[i]=='8'&&b[i]=='(')||(a[i]=='['&&b[i]=='8')||(a[i]=='('&&b[i]=='['))y++;
    }
    if(x>y)printf("TEAM 1 WINS");
    else if(y>x)printf("TEAM 2 WINS");
    else printf("TIE");
}
