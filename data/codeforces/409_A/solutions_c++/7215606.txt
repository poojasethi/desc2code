#include<cstdio>
#include<cstring>
char p1[25],p2[25];
int main(){
int i,l,a=0;
scanf("%s%s",p1,p2);
l=strlen(p1);
for(int i=0;i<l;i+=2){
    if(p1[i]=='['&&p2[i]=='8')a--;
    if(p2[i]=='['&&p1[i]=='8')a++;
    if(p1[i]=='('&&p2[i]=='8')a++;
    if(p2[i]=='('&&p1[i]=='8')a--;
    if(p1[i]=='('&&p2[i]=='[')a--;
    if(p2[i]=='('&&p1[i]=='[')a++;
}
if(a>0)printf("TEAM 1 WINS\n");
else if(a<0)printf("TEAM 2 WINS\n");
else printf("TIE\n");
}
