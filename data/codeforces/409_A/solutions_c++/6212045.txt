#include <cstdio>

char a[21], b[21], c, d, i;

int main()
{
	scanf("%s%s", a, b);

	for (i = 0; a[i]; i += 2)
	{
		if (a[i] == '8' && b[i] == '[')
			++c;
		if (b[i] == '8' && a[i] == '[')
			++d;
		if (a[i] == '[' && b[i] == '(')
			++c;
		if (b[i] == '[' && a[i] == '(')
			++d;
		if (a[i] == '(' && b[i] == '8')
			++c;
		if (b[i] == '(' && a[i] == '8')
			++d;
	}

	puts(c == d ? "TIE" : c > d ? "TEAM 1 WINS" : "TEAM 2 WINS");

	return 0;
}
