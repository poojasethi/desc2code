#include<cstdio>
#include<cstring>
#define rep(i,n) for (int i=0;i<n;++i)
int n,ans; char a[10005],b[10005];
int chg(char x)
{
	return x=='['?2:x=='('?1:0;
}
int work(char *a,char *b)
{
	int x=chg(*a),y=chg(*b);
	if (x==y) return 0;
	if ((x+1)%3==y) return -1;
	return 1;
}
int main()
{
	scanf("%s%s",a,b),n=strlen(a);
	rep(i,n/2) ans+=work(a+i*2,b+i*2);
	puts(ans>0?"TEAM 1 WINS":ans<0?"TEAM 2 WINS":"TIE");
	return 0;
}
