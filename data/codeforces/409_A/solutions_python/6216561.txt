t1 = raw_input()
t2 = raw_input()

score = 0
for i in range(0, len(t1), 2):
  a1 = t1[i:i+2]
  a2 = t2[i:i+2]
  if a1 == a2:
    continue
  if a1 == '()' and a2 == '8<':
    score += 1
  elif a1 == '[]' and a2 == '()':
    score += 1
  elif a1 == '8<' and a2 == '[]':
    score += 1
  else:
    score -= 1

if score > 0:
  print 'TEAM 1 WINS'
elif score < 0:
  print 'TEAM 2 WINS'
else:
  print 'TIE'
