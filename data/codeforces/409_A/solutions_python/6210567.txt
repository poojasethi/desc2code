a = raw_input()
b = raw_input()
sca, scb = 0, 0
for i in range(0, len(a) / 2):
	x = a[i * 2]
	y = b[i * 2]
	if x == '[' and y == '(': sca += 1
	if x == '[' and y == '8': scb += 1
	if x == '(' and y == '[': scb += 1
	if x == '(' and y == '8': sca += 1
	if x == '8' and y == '[': sca += 1
	if x == '8' and y == '(': scb += 1
if sca > scb:
	print "TEAM 1 WINS"
if sca < scb:
	print "TEAM 2 WINS"
if sca == scb:
	print "TIE"
