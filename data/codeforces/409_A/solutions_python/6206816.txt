s1=raw_input()
s2=raw_input()
n=len(s1)/2
p1=0
p2=0
for i in range(n):
    x=s1[2*i:2*i+2]
    y=s2[2*i:2*i+2]
    if x=="8<" and y=="[]": p1+=1
    if x=="[]" and y=="()": p1+=1
    if x=="()" and y=="8<": p1+=1
    if y=="8<" and x=="[]": p2+=1
    if y=="[]" and x=="()": p2+=1
    if y=="()" and x=="8<": p2+=1
if p1<p2: print "TEAM 2 WINS"
if p2<p1: print "TEAM 1 WINS"
if p1==p2: print "TIE"
