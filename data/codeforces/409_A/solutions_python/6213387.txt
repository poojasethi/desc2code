def sp(s):
    rt = []
    for i in range(0, len(s), 2):
        rt.append(s[i:i+2])
    return rt

def deal(a, b):
    m = {'()':0, '8<':1, '[]':2}
    a = m[a]
    b = m[b]
    res = (b-a+3)%3
    return res if res < 2 else -1

line1 = sp(raw_input())
line2 = sp(raw_input())

res = sum(deal(a, b) for a, b in zip(line1, line2))
if res > 0:
    print 'TEAM 1 WINS'
elif res < 0:
    print 'TEAM 2 WINS'
else:
    print 'TIE'
