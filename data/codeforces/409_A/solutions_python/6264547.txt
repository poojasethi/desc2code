s1=raw_input()
s2=raw_input()
t1=0
t2=0
for i in range(0,len(s1),2):
    if(s1[i]=='(' and s2[i]=='8'):
        t1+=1
    elif(s1[i]=='8' and s2[i]=='['):
        t1+=1
    elif(s1[i]=='[' and s2[i]=='('):
        t1+=1
    elif(s1[i]==s2[i]):
        t1+=1
        t2+=1
    else:
        t2+=1

if t1>t2:
    print "TEAM 1 WINS"
elif t2>t1:
    print "TEAM 2 WINS"
else:
    print "TIE"
