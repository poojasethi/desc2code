a = raw_input()
b = raw_input()

sa = 0
sb = 0

for i in xrange(len(a) // 2):
    ma = a[2 * i]
    mb = b[2 * i]
    if ma == mb:
        continue
    
    if (ma == '8' and mb == '[') or (ma == '[' and mb == '(') or (ma == '(' and mb == '8'):
        sa += 1
    else:
        sb += 1
if sa == sb:
    print 'TIE'
elif sa > sb:
    print 'TEAM 1 WINS'
else:
    print 'TEAM 2 WINS'
