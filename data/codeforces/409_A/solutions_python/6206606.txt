a = raw_input()
b = raw_input()

tot = 0
for i in xrange(0, len(a), 2):
  c = a[i:i+2]
  d = b[i:i+2]

  if c == "[]":
    if d == "()":
      tot +=1
    if d == "8<":
      tot -= 1

  if c == "()":
    if d == "[]":
      tot -= 1
    if d == "8<":
      tot += 1

  if c == "8<":
    if d == "()":
      tot -= 1
    if d == "[]":
      tot += 1

if tot > 0:
  print "TEAM 1 WINS"
elif tot < 0:
  print "TEAM 2 WINS"
else:
  print "TIE"
