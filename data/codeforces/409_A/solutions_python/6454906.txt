s1 = raw_input().strip()
s2 = raw_input().strip()

t = 0

i = 0
n = len(s1) - 1
while i<n:
	s = s1[i:i+2]
	ss = s2[i:i+2]
	if s == '[]':
		if ss == '()': t+=1
		elif ss == '8<': t-=1
	elif s == '()':
		if ss == '[]': t-=1
		elif ss == '8<': t+=1
	else:
		if ss == '[]': t+=1
		elif ss == '()': t-=1
	i += 2

if t == 0:
	print 'TIE'
if t>0:
	print 'TEAM 1 WINS'
if t<0:
	print 'TEAM 2 WINS'
