def score(u,v):
  t = ['()','[]','8<']
  i = t.index(u)
  j = t.index(v)
  if (i-j) % 3 == 1:
    # team 1 wins.
    return -1
  if (j-i) % 3 == 1:
    # team 2 wins.
    return 1
  return 0

a = raw_input()
b = raw_input()
n = len(a)
r = 0
for i in range(0,n,2):
  r += score(a[i:i+2], b[i:i+2])
if r > 0:
  res = 'TEAM 2 WINS'
if r < 0:
  res = 'TEAM 1 WINS'
if r == 0:
  res = 'TIE'
print res
