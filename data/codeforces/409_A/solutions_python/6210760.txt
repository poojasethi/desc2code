t1=raw_input()
t2=raw_input()
puntos_t1=0
for x in range(0, len(t2)):
    if x%2==0:
        if t2[x]=="[" and t1[x]=="(": puntos_t1-=1
        elif t2[x]=="[" and t1[x]=="8": puntos_t1+=1
        elif t2[x]=="(" and t1[x]=="[": puntos_t1+=1
        elif t2[x]=="(" and t1[x]=="8": puntos_t1-=1
        elif t2[x]=="8" and t1[x]=="(": puntos_t1+=1
        elif t2[x]=="8" and t1[x]=="[": puntos_t1-=1
if puntos_t1>0: print "TEAM 1 WINS"
elif puntos_t1<0: print "TEAM 2 WINS"
else: print "TIE"
