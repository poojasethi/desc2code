mapper = {'(': 0, '[': 1, '8': 2}

def to_gpc(s):
	return [mapper[i] for i in s[::2]]

team1 = to_gpc(raw_input())
team2 = to_gpc(raw_input())

nwin1 = 0
nwin2 = 0
for h1, h2 in zip(team1, team2):
	if h1 == (h2 + 1) % 3:
		nwin1 += 1
	elif h1 == h2:
		pass
	else:
		nwin2 += 1

if nwin1 > nwin2:
	print "TEAM 1 WINS" 
elif nwin1 < nwin2:
	print "TEAM 2 WINS" 
else:
	print "TIE"
