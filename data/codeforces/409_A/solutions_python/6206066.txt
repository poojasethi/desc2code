import sys

inp = sys.stdin.read().strip().splitlines()

t1 = inp[0][::2]
t2 = inp[1][::2]

w1 = 0
w2 = 0

for T1, T2 in zip(t1, t2):
    s = '%s%s' % (T1, T2)
    if   s in ['[(', '(8', '8[']:
        w1 += 1
    elif s in ['([', '8(', '[8']:
        w2 += 1

if w1 > w2:
    res = "TEAM 1 WINS"
elif w1 < w2:
    res = "TEAM 2 WINS"
else:
    res = "TIE"
sys.stdout.write(str(res))
