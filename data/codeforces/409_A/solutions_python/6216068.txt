A = raw_input()
B = raw_input()

x = 0


def f(s):
    if s == "8<":
        return 1
    elif s == "[]":
        return 2
    else:
        return 3

for i in range(0, len(A), 2):
    a = f(A[i:i+2])
    b = f(B[i:i+2])
    if a == b:
        continue
    if a == 1:
        x += (1 if b == 2 else -1)
    elif a == 2:
        x += (1 if b == 3 else -1)
    else:
        x += (1 if b == 1 else -1)

if x == 0:
    print "TIE"
elif x > 0:
    print "TEAM 1 WINS"
else:
    print "TEAM 2 WINS"
