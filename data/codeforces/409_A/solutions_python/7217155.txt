team1 = raw_input()
team2 = raw_input()


wins1 = 0
wins2 = 0

for i in xrange(0, len(team1), 2):
    c1 = team1[i]
    c2 = team2[i]

    if c1 == c2:
        continue
    if c1 == '8' and c2 == '[' or c1 == '(' and c2 == '8' or c1 == '[' and c2 == '(':
        wins1 += 1
    else:
        wins2 += 1

if wins1 < wins2:
    print 'TEAM 2 WINS'
elif wins1 > wins2:
    print 'TEAM 1 WINS'
else:
    print 'TIE'
