import re
a = raw_input()
b = raw_input()
score = 0

a = re.findall('..', a)
b = re.findall('..', b)

beat = {
  "8<[]": 1,
  "8<()": -1,
  "[]()": 1,
  "[]8<": -1,
  "()8<": 1,
  "()[]": -1,
  "8<8<": 0,
  "()()": 0,
  "[][]": 0
}

def one_move(a, b):
    return beat[a+b]
        
for i in xrange(len(a)):
    score += one_move(a[i], b[i])

if score < 0:
    print "TEAM 2 WINS"
elif score > 0:
    print "TEAM 1 WINS"
else:
    print "TIE"
