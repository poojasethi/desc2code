import re
s1 = re.findall('..',raw_input())
s2 = re.findall('..',raw_input())
sc1 = sc2 = 0;
Z = ['8<','()','[]']
powers = [[0,0,1],
	  [1,0,0],
	  [0,1,0]]
for i in xrange(len(s1)):
	j = Z.index(s1[i]);
	k = Z.index(s2[i]);
	sc1 += powers[j][k];
	sc2 += powers[k][j];

if sc1 > sc2: print "TEAM 1 WINS"
elif sc2 > sc1: print "TEAM 2 WINS"
else : print "TIE"
