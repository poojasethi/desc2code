v = dict({'[]':0, '8<':1, '()':2})
def cmp(sa,sb):
  c = 0
  for i in range(0,len(sa),2):
    if sa[i:i+2] == sb[i:i+2]:
      continue
    ia, ib = v[sa[i:i+2]], v[sb[i:i+2]]
    if ia - ib in [-2,1]:
      c += 1
    else:
      c -= 1
  return c

sa = raw_input().strip()
sb = raw_input().strip()
r = cmp(sa,sb)
if r > 0:
  print 'TEAM 1 WINS'
elif r < 0:
  print 'TEAM 2 WINS'
else:
  print 'TIE'
