a = raw_input()
b = raw_input()
t1,t2 = 0,0
for i in xrange(0,len(a),2):
	if a[i]+a[i+1]=="[]" and b[i]+b[i+1]=="()" or a[i]+a[i+1]=="()" and b[i]+b[i+1]=="8<" or a[i]+a[i+1]=="8<" and b[i]+b[i+1]=="[]":
		t1 += 1
	if b[i]+b[i+1]=="[]" and a[i]+a[i+1]=="()" or b[i]+b[i+1]=="()" and a[i]+a[i+1]=="8<" or b[i]+b[i+1]=="8<" and a[i]+a[i+1]=="[]":
		t2 += 1	
if t1>t2:
	print "TEAM 1 WINS"
elif t2>t1:
	print "TEAM 2 WINS"
else:
	print "TIE"
