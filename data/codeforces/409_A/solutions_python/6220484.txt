str1 = raw_input().strip()
str2 = raw_input().strip()
point1 = 0
point2 = 0

for i in range(0,len(str1),2):
  if str1[i] == '(' and str2[i] == '8' or str1[i] == '8' and str2[i] == '[' or str1[i] == '[' and str2[i] == '(':
    point1 += 1
  elif str2[i] == '(' and str1[i] == '8' or str2[i] == '8' and str1[i] == '[' or str2[i] == '[' and str1[i] == '(':
    point2 += 1

if point1 > point2:
  print 'TEAM 1 WINS'
elif point1 == point2:
  print 'TIE'
else:
  print 'TEAM 2 WINS'
