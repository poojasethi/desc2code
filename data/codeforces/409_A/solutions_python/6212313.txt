import sys

better = {
    "[]": "()",
    "8<": "[]",
    "()": "8<"
}

a = sys.stdin.readline().strip()
b = sys.stdin.readline().strip()

first = 0
second = 0

for i in xrange(0, len(a), 2):
    ap = a[i:(i + 2)]
    bp = b[i:(i + 2)]

    if ap != bp:
        if better[ap] == bp:
            first += 1
        else:
            second += 1

if first == second:
    print "TIE"
elif first > second:
    print "TEAM 1 WINS"
else:
    print "TEAM 2 WINS"
