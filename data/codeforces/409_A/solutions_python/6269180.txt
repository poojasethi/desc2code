#basic idea:
# "8<" > "[]"
# "[]" > "()"
# "()" > "8<"

from itertools import izip

def chunker(seq, size):
    return (seq[pos:pos + size] for pos in xrange(0, len(seq), size))

v1, v2 = 0, 0
winning = set([("8<", "[]"), ("[]", "()"), ("()", "8<")])

for a, b in izip(chunker(raw_input(), 2), chunker(raw_input(), 2)):
	if a == b: continue
	if (a, b) in winning: v1 += 1
	else: v2 += 1

if v1 == v2: print "TIE"
elif v1 > v2: print "TEAM 1 WINS"
else: print "TEAM 2 WINS"
