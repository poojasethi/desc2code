# -*- coding: utf-8 -*-

class Friends(object):
	def __init__(self, n):
		self.n = n
		self.L = []
		self.d = {}
	def add(self, s):
		self.L.append(s)
	def cal(self):
		for i in range(n-1, -1, -1):
			k = self.L[i]
			if self.d.has_key(k):
				continue
			else:
				self.d[k] = 1
				yield k
	def solve(self):
		for v in self.cal():
			print v

if __name__ == '__main__':
	n = input()
	friends = Friends(n)
	for i in range(n):
		friends.add(raw_input())
	friends.solve()
