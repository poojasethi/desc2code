#include<bits/stdc++.h>
using namespace std;
map<string,int>mp;
string s[500005];
int n;
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>s[i];

    }
    for(int i=n-1;i>=0;i--)
    {
        if(mp[s[i]]==0)
        {
            cout<<s[i]<<endl;
            mp[s[i]]++;
        }
    }
    return 0;
}
