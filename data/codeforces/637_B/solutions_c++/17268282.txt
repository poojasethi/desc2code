#include <bits/stdc++.h>
using namespace std;

int main()
{
    map<string,bool> mp;
    int n;
    
    cin>>n;
    
    string *x=new string[n];
    
    for(int i=0;i<n;i++)
        cin>>x[i];
        
    for(int i=n-1;i>=0;i--)
        if(!mp[x[i]]){
         cout<<x[i]<<endl;
         mp[x[i]]=true;
        }

return 0;
}
