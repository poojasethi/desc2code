#include <iostream>
#include <map>
#include <cstring>
using namespace std;
map <string,int> m;
string mas[200001];
int main(){
int n;
cin>>n;
for(int i=1;i<=n;i++){
string s;
cin>>s;
m[s]=0;
mas[i]=s;
}
for(int i=n;i>=1;i--){
if(m[mas[i]]==0) cout<<mas[i]<<endl;
m[mas[i]]=1;
}
cout<<endl;
//system("pause");
}
