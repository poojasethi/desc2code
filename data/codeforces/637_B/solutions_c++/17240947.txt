#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
map<string,int> m;
string a[200005];
int main()
{
    int n,i;
    cin>>n;
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=n-1;i>=0;i--)
    {
        if(m[a[i]]==0)
        {
            cout<<a[i]<<endl;
            m[a[i]]=1;
        }
    }
    return 0;
}
