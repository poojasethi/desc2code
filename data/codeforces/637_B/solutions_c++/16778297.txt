#include<iostream>
#include<cstdio>
#include<string>
#include<algorithm>
#include<map>
using namespace std;

const int N = 200010;
string s[N];

map<string, int> m;

int main()
{
	__int64 n,i;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>s[i];
	for(i=n-1;i>=0;i--)
	{
		m[s[i]]++;
		if(m[s[i]]==1)
			cout<<s[i]<<endl;
	}
	return 0;
}
