#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int n;
    string s[304023];
    map<string,int > a;
    cin>>n;
    for(int i=1;i<=n;i++) cin>>s[i];
    for(int i=n;i>=1;i--) if (!a[s[i]]) cout<<s[i]<<"\n", a[s[i]]++;
}
