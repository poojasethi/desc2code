#include<bits/stdc++.h>
using namespace std;
map<string,int> m;
int main()
{
    int n; cin>>n;
    string s[n];
    for(int i=0;i<n;i++)
    {
        cin>>s[i];
    }
    for(int i=n-1;i>=0;i--)
    {
        if(m[s[i]]==0)
        {
            m[s[i]]=1;
            cout<<s[i]<<endl;
        }
    }
    return 0;
}
