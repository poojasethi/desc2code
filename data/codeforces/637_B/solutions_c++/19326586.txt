#include<bits/stdc++.h>
using namespace std;
map<string,bool>M;
vector<string>V;
int main(){
    int n;cin>>n;
    for(int i=0;i<n;i++){
        string s;cin>>s;V.push_back(s);
    }
    for(int i=V.size()-1;i>=0;i--){
        if(!M[V[i]]){
            cout<<V[i]<<'\n';
            M[V[i]]=1;
        }
    }
return 0;}
