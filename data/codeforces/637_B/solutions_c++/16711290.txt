#include <iostream>
#include <string>
#include <map>
using namespace std;

string a[200005];
map<string, bool> m;
int n;

int main() {
	ios_base::sync_with_stdio(0);

	cin >> n;

	for (int i = 0; i < n; i++) cin >> a[i];

	for (int i = n - 1; i >= 0; i--) {
		if (m[a[i]] != 1) cout << a[i] << endl;
		m[a[i]] = 1;
	}

}
