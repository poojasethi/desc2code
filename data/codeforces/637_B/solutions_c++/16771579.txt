#include<bits/stdc++.h>
using namespace std;
map<string,bool> mp;
string a[200000];
int main(){
    int n;
    cin>>n;
    for(int i = 0;i<n;i++){
        cin>>a[i];
    }
    for(int i = n-1;i>=0;i--){
        if(!mp[a[i]]){
            cout<<a[i]<<endl;
            mp[a[i]] = true;
        }
    }
}
