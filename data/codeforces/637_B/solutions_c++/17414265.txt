#include<iostream>
#include<string>
#include<map>
using namespace std;
string s[200000+5];
map<string,int>p;
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>s[i];
	for(int i=n-1;i>=0;i--){
		if(p.count(s[i])==0){
			cout<<s[i]<<endl;
			p[s[i]]=1;
		}
	}
	return 0;
}
