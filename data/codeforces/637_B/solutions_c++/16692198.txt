#include <iostream>
#include <string>
#include <set>
using namespace std;

string s[200010];
set<string> out;

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> s[i];
	for (int i = n - 1; i >= 0; i--) {
		if (out.find(s[i]) == out.end())
			cout << s[i] << endl;
		out.insert(s[i]);
	}
}
