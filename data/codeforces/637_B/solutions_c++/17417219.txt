#include<bits/stdc++.h>
//zone
using namespace std;
int main()
{
vector<string>v;
int i;
int n;
map<string,int> m;
string s;
cin >> n;
for(i = 0 ; i < n ; i++)
{
cin >> s;
v.push_back(s);
}
for(i = v.size() - 1; i >= 0 ; i--)
{
if(m[v[i]] >= 1)
{
continue;
}
  m[v[i]]++;
  cout<<v[i]<<endl;
}
return 0;

}
