#include<iostream>
#include<stdio.h>
#include<string>
#include<map>
using namespace std;
map<string,int>sp;
string a[200005];
int main(){
	int n,i,x,sum=0;
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>a[i];
	}
	for(i=n;i>=1;i--){
		if(!sp[a[i]])cout<<a[i]<<endl;
		sp[a[i]]++;
	}
	return 0;
}
