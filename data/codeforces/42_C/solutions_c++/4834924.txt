#include <iostream>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <algorithm> 
#include <cmath>
#include <set>
#include <vector>
using namespace std;
 
int a[4] , x , sig ;

int main()
{  
   for(int i=0 ; i<4 ; i++) scanf("%d",&a[i]) ;
   
   while(a[0]>1 || a[1]>1 || a[2]>1 || a[3]>1)
   { 
      sig=0 ; 
      for(int i=0 ; i<4 ; i++) if(a[i]%2==0 && a[(i+1)%4]%2==0) 
      {   
          printf("/%d\n",i+1) ;   
          a[i]/=2 ; a[(i+1)%4]/=2 ; sig=1 ; 
          break ;
      }     
      if(sig) continue ;

      for(int i=0 ; i<4 ; i++) if(a[i]%2 && a[(i+1)%4]%2) 
      {
          if(a[i]==1 && a[(i+1)%4]==1) continue ; 
          printf("+%d\n",i+1) ;   
          a[i]++ ; a[(i+1)%4]++ ; sig=1 ; 
          break ;
      }     
      if(sig) continue ;

      x=0 ;
      for(int i=0 ; i<4 ; i++) if(a[x]<a[i]) x=i ;
      printf("+%d\n",x+1) ;   
      a[x]++ ; a[(x+1)%4]++ ; 
   }       
  
  return 0;
}
