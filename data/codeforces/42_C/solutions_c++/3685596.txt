#include <stdio.h>
#include <stdlib.h>
int main(){
    int a[5];
    scanf("%d%d%d%d",&a[0],&a[1],&a[2],&a[3]);
    while (a[0]!=1||a[1]!=1||a[2]!=1||a[3]!=1){
        bool ok=true;
        bool was=false;
        while (ok){
            ok=false;
        for (int i=0;i<4;i++)
            
            if (a[i]%2==0&&a[(i+1)%4]%2==0)
            {
                printf("/%d\n",i+1);
                a[i]/=2;
                a[(i+1)%4]/=2;
                ok=true;
                was=true;
            }
        }
        if (was)
            continue;
            int j=rand()%4;
            printf("+%d\n",j+1);
            a[j]++;
            a[(j+1)%4]++;
            
            
    }
    
    
    
}
