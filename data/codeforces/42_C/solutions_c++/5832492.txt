#include <iostream>
#include <cstring>
#include <vector>
#include <cstdio>
#include <cmath>
using namespace std;
char op[100005];
int a[5];
int l;
void can()
{
    int i, j, t;
    for(i=1;i<=4;i++){
        t=i%4+1;
        while(a[i]%2==0&&a[t]%2==0){
            op[l++]='/';
            op[l++]=i+'0';
            a[i]/=2, a[t]/=2;
            return ;
        }
        if(a[i]%2&&a[t]%2&&(a[i]!=1||a[t]!=1)){
            a[i]++, a[t]++;
            op[l++]='+';
            op[l++]=i+'0';
            return ;
        }

        if(a[i]>=a[1]&&a[i]>=a[2]&&a[i]>=a[3]&&a[i]>=a[4]) a[i]++, a[t]++, op[l++]='+', op[l++]=i+'0';
    }
}
int main()
{
    int i;
    for(i=1;i<=4;i++)
        scanf("%d",&a[i]);
    l=0;
    while(a[1]>1||a[2]>1||a[3]>1||a[4]>1) can() ;
    if(a[1]==1&&a[2]==1&&a[3]==1&&a[4]==1){
        for(i=0;i<l;i+=2)
            printf("%c%c\n",op[i],op[i+1]);
    }
    else puts("-1");
    return 0;
}
