#include <cstdio>

int aa[4];
inline int &a(const int &ind) { return aa[(ind + 4) % 4]; }

int main ()
{
  scanf ("%d%d%d%d", aa, aa+1, aa+2, aa+3);
  
  while (aa[0] > 1 || aa[1] > 1 || aa[2] > 1 || aa[3] > 1)
  {
    int mi = 0;
    for (int i = 1; i < 4; ++i) if (aa[i] > aa[mi]) mi = i;
    
    if (aa[mi] % 2 == 0 && a(mi - 1) % 2 == 0)
      { printf ("/%d\n", (mi + 3) % 4 + 1); a(mi - 1) /= 2; aa[mi] /= 2; }
    else if (aa[mi] % 2 == 0 && a(mi + 1) % 2 == 0)
      { printf ("/%d\n", mi + 1); aa[mi] /= 2; a(mi + 1) /= 2; }
    else
    {
      if (a(mi - 1) % 2 == 0) { printf ("+%d\n", mi + 1); ++aa[mi]; ++a(mi + 1); }
      else { printf ("+%d\n", (mi + 3) % 4 + 1); ++aa[mi]; ++a(mi - 1); }
    }
  }
  return 0;
}
