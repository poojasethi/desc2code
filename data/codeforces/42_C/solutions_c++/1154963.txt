#include<iostream>
#include<stdio.h>
int main()
{
	int a[4];
	for(int i=0;i<4;i++)
		scanf("%d",&a[i]);
	while(a[0]!=1||a[1]!=1||a[2]!=1||a[3]!=1)
	{
		bool flag=0;
		int p=-1;
		for(int i=0;i<4;i++)
		{
			int j=(i+1)%4;
			if(!(a[i]&1)&&!(a[j]&1))
			{
				flag=1;
				p=i;
				if(a[i]==4||a[j]==4)
					break;
			}
		}
		if(flag)
		{
			a[p]/=2;
			a[(p+1)%4]/=2;
			printf("/%d\n",p+1);
		}
		else
		{
			int p=-1;
			for(int i=0;i<4;i++)
			{
				if(!(a[i]&1))
				{
					p=i;
				}
			}
			if(p!=-1)
			{
				int x=(p+3)%4;
				int y=(p+1)%4;
				a[x]+=1;
				a[p]+=2;
				a[y]+=1;
				printf("+%d\n",x+1);
				printf("+%d\n",p+1);
			}
			else
			{
				a[0]+=1;
				a[1]+=1;
				a[2]+=1;
				a[3]+=1;
				printf("+1\n");
				printf("+3\n");
			}
		}
	}
}
