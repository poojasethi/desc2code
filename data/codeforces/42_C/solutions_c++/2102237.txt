#include <cstdio>

using namespace std;

int a[5];

int main()
{
	for (int i = 0; i < 4; i++)
		scanf("%d", &a[i]);
	int k = 0;
	while (a[0] != 1 || a[1] != 1 || a[2] != 1 || a[3] != 1)
	{
		k++;
		if (k > 500) break;
		for (int i = 0; i < 4; i++)
			while (a[i] % 2 == 0 && a[(i + 1) % 4] % 2 == 0)
			{
				printf("/%d\n", i + 1);
				a[i] /= 2;
				a[(i + 1) % 4] /= 2;
			}
		for (int i = 0; i < 4; i++)
		        if (a[i] % 2 == 1 && a[(i + 1) % 4] % 2== 1 && (a[i] != 1 || a[(i + 1) % 4] != 1))
		        {
				printf("+%d\n", i + 1);
				a[i]++;
				a[(i + 1) % 4]++;
				break;
			}
		for (int i = 0; i < 4; i++)
			if (a[i] % 2 == 1 && a[(i + 1) % 4] % 2 == 0 && a[(i + 2) % 4] % 2 == 1)
			{
				printf("+%d\n+%d\n", i + 1, (i + 1) % 4 + 1);
				a[i]++;
				a[(i + 1) % 4] += 2;
				a[(i + 2) % 4]++;
				break;
			}
	}
	
	return 0;
}
