#include <iostream>
#include <cstdio>
#define FOR(i, n) for (int i=0; i<(n); i++)
using namespace std;
int main(){
	int a[4];
	FOR (I, 4)
		cin >> a[I];
	while (true){
		int DivideIndex = -1, SumIndex = -1;
		bool haveOther = false;
		FOR (I, 4){
			haveOther |= a[I] != 1;
			if (a[I] % 2 == 0 && a[(I + 1) % 4]  % 2 == 0)
				DivideIndex = I;
			if (a[I] % 2 == 1 && a[(I + 1) % 4]  % 2 == 1){
				if (a[I] == 1 && a[(I + 1) % 4] == 1){
					continue;
				}
				SumIndex = I;
			}
			if (a[I] % 2 == 1 != a[(I + 1) % 4] % 2 && SumIndex == -1)
				SumIndex = I;
		}
		if (!haveOther)
			break;
		if (DivideIndex == -1 && SumIndex == -1){
			cout << "-1" << endl;
			break;
		}
		if (DivideIndex != -1){
			cout << "/" << DivideIndex + 1 << endl;
			a[DivideIndex] /= 2;
			a[(DivideIndex + 1) % 4] /= 2;
			continue;
		}
		cout << "+" << SumIndex + 1 << endl;
		a[SumIndex] ++;
		a[(SumIndex + 1) % 4] ++;
		//int tmp;
		//cin >> tmp;
	}
	return 0;
}
