#include<cstdio>
int a[4];
void add(int x)
{
    printf("+%d\n",x+1);
    ++a[x];
    ++a[(x+1)%4];
}
void div(int x)
{
    printf("/%d\n",x+1);
    a[x]>>=1;
    a[(x+1)%4]>>=1;
}
int main()
{
    for (int i=0; i<4; i++)
        scanf("%d",&a[i]);
    while (a[0]!=1)
    {
        if (a[0]&1)
            add(0);
        if (a[1]&1)
            add(1);
        div(0);
    }
    while (a[1]!=1)
    {
        if (a[1]&1)
            add(1);
        if (a[2]&1)
            add(2);
        div(1);
    }
    while (a[2]!=1)
    {
        if (a[2]&1)
        {
            add(1);
            div(1);
        }
        else
        {
            if (a[3]&1)
            {
                add(1);
                add(2);
                div(1);
            }
            else
                div(2);
        }
    }
    while (a[3]!=1)
    {
        add(2);
        while (a[3]&1)
        {
            add(3);
            div(3);
        }
        div(2);
    }
    return 0;
}
