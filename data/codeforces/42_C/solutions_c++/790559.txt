#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<ctime>
using namespace std;

int a[5];
int main()
{
    srand((unsigned)time(0));
    for (int i=0;i<4;i++) scanf("%d",&a[i]);
    while (true)
    {
        if (a[0]==1 && a[1]==1 && a[2]==1 && a[3]==1) break;
        bool flag=true;
        for (int i=0;i<4;i++)
        {
            int t=(i+1)%4;
            if (a[i]%2==0 && a[t]%2==0)
            {
                a[i]/=2;a[t]/=2;
                flag=false;
                printf("/%d\n",i+1);
                break;
            }
        }
        if (!flag) continue;
        int now=rand()%4;
        int t1=(now+1)%4;
        a[now]++;a[t1]++;
        printf("+%d\n",now+1);
    }
}
