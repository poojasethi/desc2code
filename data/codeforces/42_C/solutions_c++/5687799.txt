#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[5],i;
void add(int x){
	a[x]++; a[x%4+1]++; printf("+%d\n",x);
}
void div(int x){
	a[x]/=2; a[x%4+1]/=2; printf("/%d\n",x);
}
int main(){
	scanf("%d%d%d%d",a+1,a+2,a+3,a+4);
	for(;0ll+a[1]+a[2]+a[3]+a[4]>4;){
		if(a[1]==2 && a[2]+a[3]+a[4]==3){
			add(4); add(1); div(4); div(1);
		}
		if(a[1]+a[3]+a[4]==3 && a[2]==2){
			add(1); add(2); div(1); div(2);
		}
		for(i=1;i<=4;++i)if(!(a[i]&1)){
			if(a[i%4+1]&1)add(i%4+1),div(i);
			else div(i);
		}
		for(i=1;i<=4;++i)if(a[i]&1 && a[i%4+1]&1){
			if(a[i]==1 && a[i%4+1]==1)continue;
			add(i);
			div(i);
		}
	}
}
