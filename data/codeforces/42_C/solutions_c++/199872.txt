#define REP(i,n) for((i)=0;(i)<(int)(n);(i)++)
#define foreach(c,itr) for(__typeof((c).begin()) itr=(c).begin();itr!=(c).end();itr++)

int x[10];

int main(void){
    int i;
    
    REP(i,4) cin >> x[i];
    
    while(1){
        REP(i,4) if(x[i] != 1) break;
        if(i == 4) break;
        
        int r = rand() % 4;
        if(rand() % 10 == 0){
            cout << "+" << r+1 << endl;
            x[r]++; x[(r+1)%4]++;
        } else if(x[r] % 2 == 0 && x[(r+1)%4] % 2 == 0){
            cout << "/" << r+1 << endl;
            x[r] /= 2; x[(r+1)%4] /= 2;
        }
    }
    
    return 0;
}
