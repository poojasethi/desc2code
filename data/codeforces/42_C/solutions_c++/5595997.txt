#include<cstdio>
#include<iostream>
#include<algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;

int a[4],dem;

void chia2(int v)
{
	a[v]/=2;
	a[(v+1)%4]/=2;
	dem++;
	printf("/%d\n",v+1);
}

void cong1(int v)
{
	a[v]++;
	a[(v+1)%4]++;
	dem++;
	printf("+%d\n",v+1);
}

bool check()
{
	rep(i,4) if (a[i]!=1) return false;
	return true;
}

int main()
{
//	freopen("test.inp","r",stdin);
	rep(i,4) scanf("%d",&a[i]);
	while (!check())
	{
		int lc=0;
		rep(i,4) if ((a[i]&1)==0&&(a[(i+1)%4]&1)==0)
		{
			chia2(i);
			lc=1;
		}
		if (lc==0)
		{
			int t=rand()%4;
			cong1(t);
		}
	}
	//	printf("%d",dem);
}
