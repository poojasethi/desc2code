#include <stdio.h>
void gao(int *a)
{
    while (true)
    {
        bool exit=true;
        for (int i=0;i<4;i++)
            if (a[i]!=1) exit=false;
        a[4]=a[0];
        if (exit) return;
        bool find=false;
        for (int i=0;i<4;i++)
        {
            if (a[i]%2==0&&a[i+1]%2==0)
            {
                a[i]/=2;
                a[(i+1)%4]/=2;
                printf("/%d\n",i+1);
                find=true;
                break;
            }
            else if (a[i]%2==1&&a[i+1]%2==1&&(a[i]!=1||a[i+1]!=1))
            {
        a[i]++;
                a[(i+1)%4]++;
                printf("+%d\n",i+1);
                find=true;
                break;
            }
        }
        if (!find)
        {
            for (int i=0;i<4;i++)
                if (a[i]%2!=a[i+1]%2)
                {
                    a[i]++;
                    a[(i+1)%4]++;
                    printf("+%d\n",i+1);
                    break;
                }
        }
    }
}
int main()
{
    int a[4];
    while (scanf("%d%d%d%d",&a[0],&a[1],&a[2],&a[3])==4)
    {
        gao(a);
    }
    return 0;
}
