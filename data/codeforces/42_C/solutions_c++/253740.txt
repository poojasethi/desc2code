#include <cstdlib>
#include <cctype>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <iostream>
#include <sstream>
#include <map>
#include <set>
#include <queue>
using namespace std;

int main() {
	int a[ 4 ];
	for ( int i = 0 ; i < 4 ; i ++ ) {
		cin >> a[ i ];
	}
	while ( a[ 0 ] != 1 || a[ 1 ] != 1 || a[ 2 ] != 1 || a[ 3 ] != 1 ) {
		for ( int i = 0 , j = 3 ; i < 4 ; j = i ++ ) {
			if ( !( a[ i ] & 1 ) && !( a[ j ] & 1 ) ) {
				a[ i ] >>= 1;
				a[ j ] >>= 1;
				printf("/%d\n", j + 1);
				goto loop;
			}
		}
		for ( int i = 0 , j = 3 ; i < 4 ; j = i ++ ) {
			if ( ( a[ i ] & 1 ) && ( a[ j ] & 1 ) ) {
				if ( a[ i ] == 1 && a[ j ] == 1 ) continue;
				a[ i ] ++;
				a[ j ] ++;
				printf("+%d\n", j + 1);
				goto loop;
			}
		}
		for ( int i = 0 , j = 3 ; i < 4 ; j = i ++ ) {
			if ( ( a[ i ] & 1 ) && !( a[ j ] & 1 ) ) {
				a[ i ] ++;
				a[ j ] ++;
				printf("+%d\n", j + 1);
				goto loop;
			}
		}
		loop: ;
	}
	return 0;
}
