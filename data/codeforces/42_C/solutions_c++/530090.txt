#include <iostream>
using namespace std;
int a[4];
void p()
{
	for(int i=0;i<4;i++)
		cout<<a[i]<<' ';
	cout<<'\n';
}
void inc(int i)
{
	a[i]++;
	a[(i+1)%4]++;
	cout<<'+'<<i+1<<'\n';
}
void div(int i)
{
	a[i]/=2;
	a[(i+1)%4]/=2;
	cout<<'/'<<i+1<<'\n';
}
void f(int i)
{
	while(a[i]!=1)
	{
		int j=(i+1)%4,k=(i+3)%4;
		if(a[j]==1 && a[k]==1 && a[i]==2)
		{
			inc(k);
			inc(i);
			div(k);
			div(i);
		}
		else
		{
			if((a[i]%2)!=(a[j]%2))
				inc(j);
			if(a[i]%2)
				inc(i);
			div(i);
		}
	}
}
bool ok()
{
	for(int i=0;i<4;i++)
		if(a[i]!=1)
			return false;
	return true;
}
int main()
{
	for(int i=0;i<4;i++)
		cin>>a[i];
	while(!ok())
		for(int i=0;i<4;i++)
			f(i);
}
