#include <cstdio>
#include <cstdlib>

int main(){
	srand(1598743157);

	int v[4];
	for (int i = 0; i < 4; i++)
		scanf("%d", &v[i]);
	
	while (v[0] != 1 || v[1] != 1 || v[2] != 1 || v[3] != 1){
		bool ok = true;
		bool was = false;
		while (ok){
			ok = false;
			for (int i = 0; i < 4; i++) if (v[i] % 2 == 0 && v[(i + 1) % 4] % 2 == 0){
				printf("/%d\n", i + 1);
				v[i] /= 2;
				v[(i + 1) % 4] /= 2;
				ok = true;
				was = true;
			}
		}
		if (was) continue;
		int i = rand() % 4;
		printf("+%d\n", i + 1);
		v[i]++;
		v[(i + 1) % 4]++;
	}

	return 0;
}
