#include <cstdio>

int a[4];
void x(int i)
{
    a[i]++;
    a[(i+1)%4]++;
    printf("+%d\n",i+1);
}
void y(int i)
{
    a[i]/=2;
    a[(i+1)%4]/=2;
    printf("/%d\n",i+1);
}
void f(int i)
{
    int j=(i+1)%4,k=(i+3)%4;
    if(a[j]==1 & a[k]==1 & a[i]==2)
    {
        x(k);
        x(i);
        y(k);
        y(i);
    }
    while(a[i]-1)
    {
        if((a[i]+a[j])%2)
            x(j);
        if(a[i]%2)
            x(i);
        y(i);
    }
}
int main()
{
    for(int i=0;i<4;i++)
        scanf("%d",&a[i]);
    while(a[0]-1 | a[1]-1 | a[2]-1 | a[3]-1)
    {
        f(0);
        f(1);
        f(2);
        f(3);
    }
}
