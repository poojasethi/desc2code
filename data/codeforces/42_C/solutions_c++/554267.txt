#include<stdio.h>
#include<stdlib.h>
#include<time.h>
int a[4];
int main()
{
	srand(time(NULL));
	int i;
	bool flag;
	//freopen("1.txt","r",stdin);
	while(scanf("%d",&a[0])!=EOF)
	{
		for(i=1;i<4;i++)
			scanf("%d",&a[i]);
		while(a[0]!=1||a[1]!=1||a[2]!=1||a[3]!=1)
		{
			flag=true;
			while(flag)
			{
				flag=false;
				for(i=0;i<4;i++)
					if(a[i]%2==0&&a[(i+1)%4]%2==0)
					{
						a[i]/=2;
						a[(i+1)%4]/=2;
						flag=true;
						printf("/%d\n",i+1);
						break;
					}
			}
			if(a[0]==1&&a[1]==1&&a[2]==1&&a[3]==1)
				continue;
			i=rand()%4;
			a[i]++;a[(i+1)%4]++;
			printf("+%d\n",i+1);
		}
	}
	return 0;
}
