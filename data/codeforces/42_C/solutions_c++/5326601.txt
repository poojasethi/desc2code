#include<cstdio>
#include<iostream>
#include<cstring>
#include<ctime>
#include<cstdlib>
using namespace std;
int main()
{
    int a[4];
    int i,j;
    while(scanf("%d%d%d%d",&a[0],&a[1],&a[2],&a[3])!=EOF)
    {
        srand(time(0));
        while(true)
        {
            if(a[0]+a[1]+a[2]+a[3] == 4)break;
            bool f = false;
            for(i = 0;i<4;i++)
            {
                if(a[i]%2 == 0&&a[(i+1)%4]%2 == 0)
                {
                    cout<<"/"<<i+1<<endl;
                    a[i]/=2;
                    a[(i+1)%4]/=2;
                    f = true;
                    break;
                }
            }
            if(!f)
            {
                int x = rand()%4;
                a[x]++;
                a[(x+1)%4]++;
                cout<<"+"<<x+1<<endl;
            }
        }
    }
    return 0;
}
