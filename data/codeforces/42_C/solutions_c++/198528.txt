#include<iostream>
#include<cstdio>
using namespace std;
main()
{
	long long int a[5];
	int next[5]={0,2,3,4,1},i,happ=0,cnt=0;
	cin>>a[1]>>a[2]>>a[3]>>a[4];
	while(1)
	{
		if(a[1]==1 && a[2]==1 && a[3]==1 && a[4]==1)
		break;
		happ=0;
		for(i=1;i<=4;i++)
		{
			if((a[i]%2)==0 && (a[next[i]]%2)==0)
			{
				printf("/%d\n",i);
				cnt++;
				a[i]/=2;
				a[next[i]]/=2;
				happ=1;
				break;
			}
		}
		if(happ==1)
		continue;
		for(i=1;i<=4;i++)
		{
			if((a[i]%2)==1 && (a[next[i]]%2)==1 && (a[i]>1 || a[next[i]]>1))
			{
				printf("+%d\n",i);
				printf("/%d\n",i);
				cnt+=2;
				a[i]++;
				a[next[i]]++;
				a[i]/=2;
				a[next[i]]/=2;
				happ=1;
				break;
			}
		}
		if(happ==1)
		continue;
		for(i=1;i<=4;i++)
		{
			if(a[i]==1 && a[next[i]]==1)
			continue;
			a[i]++;
			a[next[i]]++;
			printf("+%d\n",i);
			cnt++;
			break;
		}
	}
}
