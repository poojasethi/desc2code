#include <iostream>

using namespace std;

int a[5];
int ans[2000];
char nem[2000];

int main(){
	cin >> a[0] >> a[1] >> a[2] >> a[3];
	int sum=0;
	while(true){
		if(sum>=1000){
			cout << -1 << endl;
			return 0;
		}
		if(a[0]==1 && a[1]==1 && a[2]==1 && a[3]==1)
			break;
	/*	if(a[0]%2==a[2]%2 && a[1]%2==a[3]%2 && a[0]%2!=a[1]%2){
			cout << -1 << endl;
			return 0;
		}*/
		bool z=false;
		for(int i=0;i<4;i++)
			if(a[i]%2==0 && a[(i+1)%4]%2==0){
				ans[sum]=i;
				nem[sum]='/';
				a[i]/=2;
				a[(i+1)%4]/=2;
				sum++;;
				z=true;
			}
		for(int i=0;i<4;i++)
			if(a[i]%2==1 && a[(i+1)%4]%2==1 && (a[i]!=1 || a[(i+1)%4]!=1)){
				ans[sum]=i;
				nem[sum]='+';
				a[i]++;
				a[(i+1)%4]++;
				sum++;
				z=true;
			}
		if(z)

			continue;
		for(int i=0;i<4;i++)
			if(a[i]%2==0){
				ans[sum]=i;
				nem[sum]='+';
				a[i]++;
				a[(i+1)%4]++;
				sum++;
				break;
			}

	}
	for(int i=0;i<sum;i++)
		cout << nem[i] << ans[i]+1 << endl;
}
