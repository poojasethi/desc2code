#include <iostream>
using namespace std;

int a[4];
void inc(int i)
{
    a[i]++;
    a[(i+1)%4]++;
    cout<<'+'<<i+1<<'\n';
}
void div(int i)
{
    a[i]/=2;
    a[(i+1)%4]/=2;
    cout<<'/'<<i+1<<'\n';
}
void f(int i)
{
    int j=(i+1)%4,k=(i+3)%4;
    if(a[j]==1 && a[k]==1 && a[i]==2)
    {
        inc(k);
        inc(i);
        div(k);
        div(i);
    }
    while(a[i]-1)
    {
        if((a[i]+a[j])%2)
            inc(j);
        if(a[i]%2)
            inc(i);
        div(i);
    }
}
int main()
{
    for(int i=0;i<4;i++)
        cin>>a[i];
    while(a[0]!=1 || a[1]!=1 || a[2]!=1 || a[3]!=1)
    {
        f(0);
        f(1);
        f(2);
        f(3);
    }
}
