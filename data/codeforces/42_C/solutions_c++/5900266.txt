#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
#include <ctime>


using namespace std;
vector <pair <char , int> > ans;
int a[4];
int main () {
    srand (time (NULL));
    // freopen ("input.txt" , "r" , stdin);
    for (int i = 0 ; i < 4 ; i ++)
        scanf ("%d" , &a[i]);
    while (true) {
        if (*max_element (a , a + 4) == 1) break;
        int k = rand () % 4;
        if (a[k] % 2) {
            a[k] ++;
            a[(k + 1) % 4] ++;
            ans.push_back (make_pair ('+' , k + 1));
        }
        if (a[(k + 1) % 4] % 2) {
            a[(k + 1) % 4] ++;
            a[(k + 2) % 4] ++;
            ans.push_back (make_pair ('+' , (k + 1) % 4 + 1));
        }
        a[k] /= 2;a[(k + 1) % 4] /= 2;
        ans.push_back (make_pair ('/' , k + 1));
    }
    for (int i = 0 ; i < ans.size() ; i ++)
        printf ("%c%d\n" , ans[i].first , ans[i].second);
    return 0;
}
