#include<iostream>
#include<cstdlib>
#include<cstdio>

using namespace std;

int data[4];

int main() {
    for ( int i = 0 ; i < 4 ; i++ ) scanf( "%d" , &data[i] );
    while ( data[0] != 1 || data[1] != 1 || data[2] != 1 || data[3] != 1 ) {
        bool change = false;
        for ( int i = 0 ; i < 4 ; i++ )
            if ( ( data[i] & 1 ^ 1 ) && ( data[( i + 1 ) & 3] & 1 ^ 1 ) ) {
                printf( "/%d\n" , i + 1 );
                data[i] >>= 1;
                data[( i + 1 ) & 3] >>= 1;
                change = true;
            }
        if ( !change ) {
            int i = rand() & 3;
            printf( "+%d\n" , i + 1 );
            data[i]++;
            data[( i + 1 ) & 3]++;
        }
    }
    return 0;
}
