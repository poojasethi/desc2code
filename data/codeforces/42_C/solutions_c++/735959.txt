#include<iostream>
#include<string>

using namespace std;

int main() {
  int a[10];
  cin >> a[1] >> a[2] >> a[3] >> a[4];
  a[5] = a[1];
  while (1) {
    if (a[1] == a[2] && a[2] == a[3] && a[3] == a[4] && a[4] == 1) break;
    bool inloop = false;
    for (int i = 1; i <= 4; i ++)
      if (a[i] % 2 == 0 && a[i + 1] % 2 == 0) {
	a[i] = a[i] / 2;
	a[i + 1] = a[i + 1] / 2;
	cout << "/" << i << endl;
	if (i == 1) a[5] = a[1];
	if (i == 4) a[1] = a[5];
	inloop = true;
	break;
      }
    if (inloop) continue;
    for (int i = 1; i <= 4; i ++)
      if (a[i] % 2 == 1 && a[i + 1] % 2 == 1 && (a[i] > 1 || a[i + 1] > 1)) {
	a[i] ++;
	a[i + 1] ++;
	cout << "+" << i << endl;
	if (i == 1) a[5] = a[1];
	if (i == 4) a[1] = a[5];
	inloop = true;
	break;
      }
    if (inloop) continue;
    for (int i = 1; i <= 4; i ++)
      if (a[i] % 2 + a[i + 1] % 2 == 1) {
	a[i] ++;
	a[i + 1] ++;
	cout << "+" << i << endl;
	if (i == 1) a[5] = a[1];
	if (i == 4) a[1] = a[5];
	break;
      }
  }
}
