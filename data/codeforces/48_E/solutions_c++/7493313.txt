#include<cstdio>
#include<algorithm>
#include<vector>
#define pii pair<int,int>
#define A first
#define B second
using namespace std;
int H,T,R,n,m,lv[201][201],fq,lq,loop,d[201][201],h,t,ans;
vector <pii> eH, eT;
pii q[100001];
int dfs(int h,int t){
    int p = 1;
    if(h + t > R){
        return 0;
    }
    if(d[h][t] == -1){
        loop = 1;
        return -2;
    }
    if(d[h][t] != 0) return d[h][t];
    d[h][t] = -1;
    for(int i=0;i<eH.size() && h > i;i++)
        p = max(p,dfs(h+eH[i].A,t+eH[i].B) + 1);
    for(int i=0;i<eT.size() && t > i;i++)
        p = max(p,dfs(h+eT[i].A,t+eT[i].B) + 1);
    return d[h][t] = p;
}
int main(){ 
    scanf("%d%d%d",&H,&T,&R);
    scanf("%d",&n);
    for(int i=0,a,b;i<n;i++){
        scanf("%d%d",&a,&b);
        eH.push_back(pii(a-i-1,b));
    }
    scanf("%d",&m);
    for(int i=0,a,b;i<m;i++){
        scanf("%d%d",&a,&b);
        eT.push_back(pii(a,b-i-1));
    }
    lv[H][T] = 1; q[lq++] = pii(H,T);
    while(fq<lq){
        h = q[fq].A, t = q[fq].B;
        for(int i=0;i<eH.size() && h > i;i++){
            if(h+t+eH[i].A+eH[i].B <= R && lv[h+eH[i].A][t+eH[i].B] == 0){
                lv[h+eH[i].A][t+eH[i].B] = lv[h][t] + 1;
                q[lq++] = pii(h+eH[i].A,t+eH[i].B);
            }
        }
        for(int i=0;i<eT.size() && t > i;i++){
            if(h+t+eT[i].A+eT[i].B <= R && lv[h+eT[i].A][t+eT[i].B] == 0){
                lv[h+eT[i].A][t+eT[i].B] = lv[h][t] + 1;
                q[lq++] = pii(h+eT[i].A,t+eT[i].B);
            }
        }
        fq++;
    }
    if(lv[0][0] != 0)
        printf("Ivan\n%d\n",lv[0][0]-1);
    else{
        for(int i=0;i<=R;i++) for(int j=0;j<=R;j++) lv[i][j] = 0;
        ans = dfs(H,T);
        if(loop) printf("Draw\n");
        else printf("Zmey\n%d\n",ans);
    }
    return 0;
}
