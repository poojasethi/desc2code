int dp[N][N];
int vis[N][N];
int r;
int n, m;
int h[2][N], t[2][N];
int gao(int a, int b) {
	if(vis[a][b] == -1) return inf;
	if(vis[a][b]) return dp[a][b];
	vis[a][b] = -1;
	int &ret = dp[a][b];
	if(a + b > r) {
		vis[a][b] = 1;
		return ret = 0;
	}
	if(a == 0 && b == 0) {
		vis[a][b] = 1;
		return ret = -1;
	}

	int t1 = -inf, t2 = 0;
	for(int i = 1; i <= min(n, a); ++i) {
		int tmp = gao(a-i+h[0][i], b+t[0][i]);
		if(tmp < 0) {
			t1 = max(t1, tmp-1);
		}
		else {
			t2 = max(t2, tmp+1);
		}
	}
	for(int i = 1; i <= min(m, b); ++i) {
		int tmp = gao(a+h[1][i], b-i+t[1][i]);
		if(tmp < 0) {
			t1 = max(t1, tmp-1);
		}
		else t2 = max(t2, tmp+1);
	}
	vis[a][b] = 1;
	if(t1 != -inf) ret = t1;
	else ret = min(inf, t2);
	return ret;
}

void bfs() {
	queue<int> que;
	que.push(0), que.push(0);
	vis[0][0] = 1;
	dp[0][0] = -1;
	while(!que.empty()) {
		int hh = que.front(); que.pop();
		int tt = que.front(); que.pop();
		for(int i = 1; i <= n; ++i) if(hh >= h[0][i] && tt >= t[0][i]) {
			int nh = hh + i - h[0][i], nt = tt - t[0][i];
			if(nh + nt <= r && !vis[nh][nt]) {
				dp[nh][nt] = dp[hh][tt] - 1;
				vis[nh][nt] = 1;
				que.push(nh);
				que.push(nt);
			}
		}
		for(int i = 1; i <= m; ++i) if(hh >= h[1][i] && tt >= t[1][i]) {
			int nh = hh - h[1][i], nt = tt + i - t[1][i];
			if(nh + nt <= r && !vis[nh][nt]) {
				dp[nh][nt] = dp[hh][tt] - 1;
				vis[nh][nt] = 1;
				que.push(nh);
				que.push(nt);
			}
		}
	}
}
int main() {

	scanf("%d%d%d", &hh, &tt, &r);
