using namespace std;
pair<int,int>a[N],b[N];
int n,m,r,dp[N][N],loop,qx[1<<17],qy[1<<17],qs[1<<17],q;
void add(int x,int y,int s)
{
    if(x+y<=r&&!dp[x][y]++)qx[q]=x,qy[q]=y,qs[q++]=s;
}
int go(int x,int y)
{
    if(x+y>r)return 0;
    int &res=dp[x][y];
    if(res==-2)loop=1;
    else if(res==-1)
    {
        res=-2;
        int val=-2;
        for(int i=1;i<=min(n,x);i++)
        val=max(val,go(x-i+a[i].x,y+a[i].y)+1);
        for(int i=1;i<=min(m,y);i++)
        val=max(val,go(x+b[i].x,y-i+b[i].y)+1);
        res=val;
    }
    return res;
}
int main()
{
    //freopen("in.txt","r",stdin);
    int i,j,k,h,t;
    while(~scanf("%d%d%d",&h,&t,&r))
    {
         scanf("%d",&n);
         for(i=1;i<=n;i++)scanf("%d%d",&a[i].x,&a[i].y);
         scanf("%d",&m);
         for(i=1;i<=m;i++)scanf("%d%d",&b[i].x,&b[i].y);
         q=0;
         memset(dp,0,sizeof(dp));
         add(h,t,0);
         int pos=0;
         while(pos<q)
         {
             int x=qx[pos],y=qy[pos],s=qs[pos++];
             if(x+y==0)
             {
                 printf("Ivan\n%d\n",s);return 0;
             }
             for(i=1;i<=min(x,n);i++)
             {
                 add(x-i+a[i].x,y+a[i].y,s+1);
             }
             for(i=1;i<=min(y,m);i++)
             {
                 add(x+b[i].x,y-i+b[i].y,s+1);
             }
         }
         memset(dp,-1,sizeof(dp));
         loop=0;
         int res=go(h,t);
         if(loop)puts("Draw");
         else printf("Zmey\n%d\n",dp[h][t]);
    }
    return 0;
}
