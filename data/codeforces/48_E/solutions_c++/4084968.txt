#include<cstdio>
#include<cstring>
#include<vector>
#include<queue>
#include<algorithm>
using namespace std;

vector<int>g[71000];
int n,m,r;
int a[210][2],b[210][2];
void init()
{
	for(int i=0;i<256;++i)
	{
		for(int j=0;j<256;++j)
		{
			if(i+j==0)	continue;
			if(i+j>r)	continue;
			int idx=(i<<8)|j;
			for(int k=1;k<=min(i,n);++k)
			{
				int ii=i-k+a[k][0],jj=j+a[k][1];
				//if(i-k+j==0)
				//	g[0].push_back(idx);
				//else
				if(ii+jj>r)
					g[65536].push_back(idx);
				else
					g[(ii<<8)|jj].push_back(idx);
			}
			for(int k=1;k<=min(j,m);++k)
			{
				int ii=i+b[k][0],jj=j-k+b[k][1];
				//if(i+j-k==0)
				//	g[0].push_back(idx);
				//else 
					if(ii+jj>r)
					g[65536].push_back(idx);
				else
					g[(ii<<8)|jj].push_back(idx);
			}
		}
	}
}

queue<int>q;
int d[71000];
int bfs(int h,int t)
{
	d[0]=0;
	q.push(0);
	while(!q.empty())
	{
		int x=q.front(); q.pop();
		for(int i=0;i<g[x].size();++i)if(!d[g[x][i]])
		{
			d[g[x][i]]=d[x]+1;
			q.push(g[x][i]);
		}
	}
	return d[(h<<8)|t];
}

int in[71000],dp[71000];
void topo()
{
	for(int i=0;i<=200;++i)
		for(int j=0;j<=200;++j)if(i+j<=r)
			in[(i<<8)|j]=min(i,n)+min(j,m);
	q.push(65536);
	while(!q.empty())
	{
		int x=q.front();	q.pop();
		for(int i=0;i<g[x].size();++i)
		{
			int y=g[x][i];
			in[y]--;
			dp[y]=max(dp[y],dp[x]+1);
			if(!in[y])	q.push(y);
		}
	}
}

int main()
{
	int h,t;
	scanf("%d %d %d",&h,&t,&r);
	scanf("%d",&n);
	for(int i=1;i<=n;++i)	scanf("%d %d",&a[i][0],&a[i][1]);
	scanf("%d",&m);
	for(int i=1;i<=m;++i)	scanf("%d %d",&b[i][0],&b[i][1]);
	init();
	int res=bfs(h,t);
	if(res>0)
		printf("Ivan\n%d\n",res);
	else
	{
		topo();
		if(in[(h<<8)|t])
			puts("Draw\n");
		else
			printf("Zmey\n%d\n",dp[(h<<8)|t]);
	}
	return 0;
}
