#include <iostream>
#include <cstdio>
#include <queue>
#include <stack>
#include <climits>
#include <cstring>
using namespace std;

//Win = 2, Draw = 1, Lose = 0
struct stuff{int H,T;}tmp,ins;
int N,M,H,T,R,ans;
int L[205][205],vis[205][205];
stuff A[205],B[205];
bool draw,visit[205][205];
queue <stuff> q;
stack <stuff> s;

int win(){
    tmp.H = H;
    tmp.T = T;
    L[H][T] = 0;
    q.push(tmp);
    while(!q.empty()){
        tmp = q.front();
        q.pop();
        if(tmp.H == 0 && tmp.T == 0) return L[tmp.H][tmp.T];
        for(int i=1;i<=min(tmp.H,N);++i){
            ins.H = tmp.H-i+A[i].H;
            ins.T = tmp.T+A[i].T;
            if(ins.H+ins.T > R) continue;
            if(L[ins.H][ins.T] == -1){
                L[ins.H][ins.T] = L[tmp.H][tmp.T] + 1;
                q.push(ins);
            }
        }
        for(int i=1;i<=min(tmp.T,M);++i){
            ins.H = tmp.H+B[i].H;
            ins.T = tmp.T-i+B[i].T;
            if(ins.H+ins.T > R) continue;
            if(L[ins.H][ins.T] == -1){
                L[ins.H][ins.T] = L[tmp.H][tmp.T] + 1;
                q.push(ins);
            }
        }
    }
    return -1;
}

bool checkDraw(int X,int Y){
    if(vis[X][Y] == 1) return 1;
    if(vis[X][Y] == 2) return 0;
    vis[X][Y] = 1 ;
    for(int i=1;i<=min(N,X);++i){
        ins.H = X-i+A[i].H;
        ins.T = Y+A[i].T ;
        if(ins.H+ins.T> R) continue;
        if(checkDraw(ins.H,ins.T)) return 1;
    }
    for(int i=1;i<=min(Y,M);++i){
        ins.H = X+B[i].H ;
        ins.T = Y-i+B[i].T ;
        if(ins.H+ins.T> R) continue;
        if(checkDraw(ins.H,ins.T)) return 1;
    }
    vis[X][Y] = 2;
    return 0;
}

int lose(int X,int Y){
    if(X+Y > R) return 0;
    if(L[X][Y]) return L[X][Y];
    for(int i=1;i<=min(X,N);++i){
        ins.H = X-i+A[i].H;
        ins.T = Y+A[i].T;
        L[X][Y] = max(L[X][Y],lose(ins.H,ins.T));
    }
    for(int i=1;i<=min(Y,M);++i){
        ins.H = X+B[i].H;
        ins.T = Y-i+B[i].T;
        L[X][Y] = max(L[X][Y],lose(ins.H,ins.T));
    }
    ++L[X][Y];
    return L[X][Y];
}

int main(){
    memset(L,-1,sizeof(L));
    scanf("%d%d%d",&H,&T,&R);
    scanf("%d",&N);
    for(int i=1;i<=N;++i) scanf("%d%d",&A[i].H,&A[i].T);
    scanf("%d",&M);
    for(int i=1;i<=M;++i) scanf("%d%d",&B[i].H,&B[i].T);
    ans = win();
    if(ans != -1) printf("Ivan\n%d\n",ans);
    else if(checkDraw(H,T)) printf("Draw\n");
    else{
        memset(L,0,sizeof(L));
        printf("Zmey\n%d\n",lose(H,T));
    }
//  system("pause");
}
