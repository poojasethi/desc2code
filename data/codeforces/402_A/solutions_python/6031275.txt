#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

k, a, b, v = map(int, raw_input().split())

s = a / v + (1 if a % v != 0 else 0)

cnt = 0
while s > 0 : 
    if b >= k - 1 : 
        b -= (k - 1)
        s -= k
        cnt += 1
    else : 
        s -= (b + 1)
        b = 0
        cnt += 1

print cnt
