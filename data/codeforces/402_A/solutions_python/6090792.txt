k, a, b, v = map(int, raw_input().split())

nut_piles = a / v if not a % v else a / v + 1

box_number = 1
box_sections = 1
box_filled = 0

while nut_piles:
	if box_filled < box_sections:
		box_filled += 1
	elif box_sections < k and b > 0:
		box_sections += 1
		box_filled += 1
		b -= 1
	else:
		box_number += 1
		box_sections = 1
		box_filled = 1
	nut_piles -= 1
print box_number
