ri = lambda: raw_input().strip()

# n = int(ri())
# #n, m = map(int, ri().split())

k, a, b, v = map(int, ri().split())

count = 0
while (b >= k-1) and (a >= k*v):
    b -= k-1
    a -= k*v
    count += 1


if (b > 0) and (a > 0):
    a -= (b+1) * v
    count += 1


while a > 0:
    a -= v
    count += 1

print count
