k,a,b,v = map(int,raw_input().split())
q = b // (k-1)
r = b % (k-1)
if a <= q*k*v:
  res = a // (k*v)
  if a % (k*v) >0:
    res += 1
else:
  a -= q*k*v
  res = q
  if a <= (r + 1) * v:
    res = q+1
  else:
    a -= (r + 1) * v
    res += 1
    s = a // v
    res += s
    if a % v > 0:
      res += 1
print res
