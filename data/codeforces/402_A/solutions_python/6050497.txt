def main():
    k, a, b, v = map(int, raw_input().split())
    counter = 0
    while a > 0:
        if b > k - 1:
            a -= k * v
            b -= k - 1
        else:
            a -= (b + 1) * v
            b = 0
        counter += 1
    print counter
        
if __name__ == '__main__':
    main()
