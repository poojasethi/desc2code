k,a,b,v=map(int,raw_input().split())
count=0
while(1):
	if(b>=k):
		b-=(k-1)
		a-=v*(k)

		if(a>0):
			count+=1
			continue
		else:
			count+=1
			break
	elif(b!=0):
		a-=v*(b+1)
		b=0
		if(a>0):
			count+=1
			continue
		else:
			count+=1
			break
	else:
		if(a>0):
			count+=1
			a-=v
		else:
			break


#count+=1
print count
