def main():
    k,a,b,v = [int(c) for c in raw_input().split()]
    count = 1
    sections = 1
    while a>0:
        a -= v
        if a<= 0:
            break
        if sections < k and b>0:
            sections += 1
            b -= 1
        else:
            count += 1
            sections = 1
    return count
    

if __name__ == "__main__":
    print main()
