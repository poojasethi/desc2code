import sys

inp = sys.stdin.readlines()
ctr = 0

x = inp[ctr].split()
k = int(x[0])
a = int(x[1])
b = int(x[2])
v = int(x[3])
ctr+=1
ans = 0
while True:
   if b >= k-1:
      b -= (k-1)
      a -= (k*v)
      ans += 1
      if a <= 0:
         break
   elif b > 0:
        a -= ((b+1)*v)
        b = 0
        ans += 1
        if a <= 0:
          break
   else:
        ans += (a/v)
        if a%v!=0:
           ans +=1
        break
print ans
