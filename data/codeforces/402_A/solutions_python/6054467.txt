from itertools import takewhile

k, a, b, v = map(int, raw_input().split())

def boxes(k, b):

    while b >= k - 1:
        yield k
        b -= k - 1
    
    if b:
        yield b + 1
    
    while True:
        yield 1
        
def pack(x, k, b):
    c = 0
    t = 0
    b = boxes(k, b)
    while t < x:
        t += next(b)
        c += 1
    return c
    
print pack((a + v - 1) // v, k, b)
