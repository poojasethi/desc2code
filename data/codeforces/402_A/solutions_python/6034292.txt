import sys


k, a, b, v = map(int, raw_input().split())
ans = 0
k -= 1
while True:

    ans += 1
    x, y = divmod(b, k)
    have = ans
    ican = min(x, have) * (k + 1) * v
    have -= min(x, have)
    if have > 0:
        have -= 1
        ican += (y + 1) * v
    ican += have * v
    if ican >= a:
        break

print ans
