k,a,b,v = map(int, raw_input().split())
box = 0
cnt = 0
while cnt < a:
    if b == 0:
        cnt = cnt + v
        box = box + 1
    else:
        if b <= k-1 :
            cnt = cnt + (b+1)* v
            b = 0
            box = box + 1
        else:
            b = b - (k-1)
            cnt = cnt + k * v
            box = box + 1
print box
