from math import *

def main():
    s = str(raw_input()).split(" ")
    k = int(s[0])
    a = int(s[1])
    b = int(s[2])
    v = int(s[3])

    x = b
    left = a

    ans = 0

    while(True):

        sections = min(k, x + 1)

        x -= sections - 1
        put = sections * v

        left = left - put
        ans += 1
        if(left > 0):
            continue
        else:
            break

    print ans


if __name__ == '__main__':
    main()
