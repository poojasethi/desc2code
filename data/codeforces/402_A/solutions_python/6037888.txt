def solve():
    k,a,b,v = map(int, raw_input().split())
    cajas = 0
    while a > 0:
        divisorepos = min(k-1,b)
        a  = a - (divisorepos+1)*v
        cajas = cajas  + 1
        b -= divisorepos
    print cajas



solve()
