import sys

def main():
    file=sys.stdin
    line=file.readline()
    values=line.split(' ')
    k,a,b,v=int(values[0]),int(values[1]),int(values[2]),int(values[3])
    for i in range(a):
        if (k-1)*i<b:
            if k*i*v>=a:
                print i
                return
        elif ((b/(k-1))*k+(b%(k-1)+1)+(i-b/(k-1)-1))*v>=a:
            print i
            return


main()
