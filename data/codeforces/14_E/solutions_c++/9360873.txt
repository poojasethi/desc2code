#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t;
  long long c,a[21][12][4][4];
  for(w=0;w<21;w++)
   for(e=0;e<12;e++)
    for(r=0;r<4;r++)
     for(c=0;c<4;c++)
      a[w][e][r][c]=0;
  a[0][0][0][0]=a[0][0][0][1]=a[0][0][0][2]=1;
  cin>>q>>t;
  q--;
  for(w=0;w<=q;w++)
   for(e=0;e<11;e++){
    for(c=r=0;r<4;r++){
     a[w+1][e][1][r]+=c;
     c+=a[w][e][0][r]+a[w][e][1][r];}
    for(r=0;r<4;r++)
     a[w][e][2][r]+=a[w][e][1][r];
    for(c=0,r=3;r>=0;r--){
     a[w+1][e][3][r]+=c;
     c+=a[w][e][2][r]+a[w][e][3][r];}
    for(r=3;r>=0;r--)
     a[w][e+1][0][r]+=a[w][e][3][r];}
  c=0;
  for(w=0;w<4;w++)
   c+=a[q][t][0][w];
  cout<<c;
  return 0;}
