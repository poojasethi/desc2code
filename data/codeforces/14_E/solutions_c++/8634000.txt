#include <bits/stdc++.h>
using namespace std;
#define rep(i,x,y) for(i=x;i<=y;++i)

const int N=30,K=10,T=20;

int n,t,i,j,q,ca1,ca2,k,if1,if2;
long long an,F[N][K][K][T][T];

int main(){
    //freopen("input.txt","r",stdin);
    scanf("%d%d",&n,&t);
    memset(F,0,sizeof F);
    rep(i,1,4)rep(j,1,4)if(i!=j)F[2][i][j][0][0]=1;
    rep(i,2,n)rep(j,1,4)rep(k,1,4)if(j!=k)
        rep(ca1,0,t)rep(ca2,0,t)
            rep(q,1,4){
                if(k==q)continue;
                if1=(j<k&&k>q);
                if2=(j>k&&k<q);
                F[i+1][k][q][ca1+if1][ca2+if2]+=F[i][j][k][ca1][ca2];
            }
    an=0;
    rep(i,1,4)rep(j,1,4)if(i!=j)an+=F[n][i][j][t][t-1];
    printf("%lld\n",an);
    return 0;
}
