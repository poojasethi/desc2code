#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int n,t;
    while(scanf("%d%d",&n,&t)!=EOF)
    {
        if(n<2*t+1||n>7*t)
            printf("0\n");
        else
        {
            int dp[21][4][21],ans=0;
            memset(dp,0,sizeof(dp));
            dp[0][0][0]=1;
            dp[0][1][0]=1;
            dp[0][2][0]=1;
            for(int i=0;i<=n;i++)
                for(int j=0;j<4;j++)
                    for(int k=0;k<20;k++)
                        for(int l=0;l<4;l++)
                            if((i==1&&j>l)||(i>1&&j!=l))
                                dp[i][j][k+((j<l)!=(k%2))]+=dp[i-1][l][k];
            for(int j=0;j<4;j++)
                ans+=dp[n-1][j][t*2-1];
            printf("%d\n",ans);
        }
    }
    return 0;
}
