#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m,dp[30][15][4][3];
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0;i<4;i++) dp[1][1][i][0]=1;
    for(int i=1;i<n;i++)
        for(int j=1;j<=m;j++)
            for(int k=0;k<4;k++)
            {
                for(int l=k+1;l<4;l++)
                {
                    dp[i+1][j][l][1]+=dp[i][j][k][0]+dp[i][j][k][1];
                    dp[i+1][j+1][l][1]+=dp[i][j][k][2];
                }
                for(int l=0;l<k;l++)
                {
                    dp[i+1][j][l][2]+=dp[i][j][k][1]+dp[i][j][k][2];
                }
            }
    int ans=0;
    for(int i=0;i<4;i++) ans+=dp[n][m][i][2];
    printf("%d\n",ans);
    return 0;
}
