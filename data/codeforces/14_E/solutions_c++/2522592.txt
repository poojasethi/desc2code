#include<stdio.h>
__int64 dp[33][14][6][3];
int main()
{
	int i, j, h, k, n, t;
	scanf("%d%d", &n, &t);
	dp[0][0][0][0] = 1;
	for(i = 1; i <= n; i++)
		for(j = 0; j <= t; j++)
			for(h = 1; h <= 4; h++)
			{
				for(k = 0; k < h; k++)
					dp[i][j][h][0] += dp[i-1][j][k][0] + dp[i-1][j][k][1];
				if(i == 2)
					dp[i][j][h][1] = 0;
				else
				{
				    for(k = 5; k > h; k--)
					{
					    dp[i][j][h][1] += dp[i-1][j][k][1];
						if(j) dp[i][j][h][1] += dp[i-1][j-1][k][0];
					}
				}
			}
	__int64 ans = 0;
	for(i = 1; i <= 4; i++)
		ans += dp[n][t][i][1];
	printf("%I64d\n", ans);
	return 0;
}
