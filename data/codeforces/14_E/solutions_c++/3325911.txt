#include <iostream>
#include <cmath>

#define rep(i,n) for(int i=1;i<=n;i++)

using namespace std;

int n,t;
int dp[24][5][5][12][12],sum;

int main(){
  cin>>n>>t;
  rep(i,4)rep(j,4){if(i==j)continue;dp[2][i][j][0][0]++; }
  for(int ii=2;ii<n;ii++){
    rep(i,4)rep(j,4){
      if(i==j)continue;
      for(int p=0;p<=t;p++)for(int q=0;q<=t;q++){
      int t = dp[ii][i][j][p][q];
          rep(k,4){
            if(k==j)continue;
            if(i<j){
          if(j<k)dp[ii+1][j][k][p][q]+=t;
          else dp[ii+1][j][k][p+1][q]+=t;
            }
            if(i>j){
          if(j<k)dp[ii+1][j][k][p][q+1]+=t;
          else dp[ii+1][j][k][p][q]+=t;
            }
          }
        }
    }
  }
  rep(i,4)rep(j,4)sum+=dp[n][i][j][t][t-1];
  cout<<sum<<endl;
}
