#include<iostream>
using namespace std;

int s[25][4][25];

int main() {
  int n,t;
  cin>>n>>t;
  for (int yi=0;yi<4;yi++)
    s[0][yi][0]=1;
  for (int ni=1;ni<n;ni++)
    for (int yi=0;yi<4;yi++)
      for (int ti=0;ti<=2*t-1;ti++) {
        s[ni][yi][ti]=0;
        if (ti%2==0) {
          for (int yj=0;yj<yi;yj++)
            s[ni][yi][ti]+=s[ni-1][yj][ti];
          if (ti>0 && ni>1)
            for (int yj=0;yj<yi;yj++)
              s[ni][yi][ti]+=s[ni-1][yj][ti-1];
        } else {
          for (int yj=yi+1;yj<4;yj++)
            s[ni][yi][ti]+=s[ni-1][yj][ti];
          if (ni>1)
            for (int yj=yi+1;yj<4;yj++)
              s[ni][yi][ti]+=s[ni-1][yj][ti-1];
        }
      }
  int answer=0;
  for (int yi=0;yi<4;yi++)
    answer+=s[n-1][yi][t*2-1];
  cout<<answer<<endl;
  return 0;
}
