#include <iostream>
using namespace std;

int a[21][11][11][5][5];

int main()
{
	int n,t;
	cin>>n>>t;

	for(int i=1;i<=2;i++)
		for(int j=1;j<5;j++)
			for(int k=1;k<5;k++)
				if(j!=k) a[i][0][0][j][k]=1;

	for(int i=3;i<n+1;i++)
	for(int j=0;j<t+1;j++)
	for(int k=0;k<t+1;k++)
	for(int t1=1;t1<5;t1++)
	for(int t2=1;t2<5;t2++)
	for(int t=1;t<5;t++)
		if(t!=t2 && t1!=t2)  {
			if(t2<t && t2<t1)
				a[i][j][k+1][t2][t]+=a[i-1][j][k][t1][t2];
			else
			if(t2>t && t2>t1)
				a[i][j+1][k][t2][t]+=a[i-1][j][k][t1][t2];
			else
				a[i][j][k][t2][t]+=a[i-1][j][k][t1][t2];
		}

	int res=0;
	for(int i=1;i<5;i++)
		for(int j=1;j<5;j++)
			if(i!=j) res+=a[n][t][t-1][i][j];

	cout<<res;
	return 0;
}
