#include <stdio.h>

int f[25][5][5];
long long g[15][25][5][5];

void go2(int len, int st, int x, int realen) {
	if(len>=2) f[realen][st][x]++;
	for(--x;x>=1;x--)
		go2(len+1, st, x, realen+1);
}

void go(int len, int st, int x) {
	if(len>=2) go2(1, st, x, len);
	for(++x;x<=4;x++)
		go(len+1, st, x);
}

int main(void) {
	int i, j, x, y, z, L, w;
	for(i=1;i<=4;i++)
		go(1, i, i);
	int n, t;
	scanf("%d%d", &n, &t);
	g[0][1][1][1] = 1;
	g[0][1][2][2] = 1;
	g[0][1][3][3] = 1;
	g[0][1][4][4] = 1;
	for(i=1;i<=t;i++) {
		for(x=1;x<=4;x++)
			for(y=1;y<=4;y++)
				for(z=1;z<=4;z++)
					for(L=1;L<=n;L++)
						for(w=1;w+L<=n;w++)
							g[i][w+L][y][z] += g[i-1][L][x][y] * f[w+1][y][z];
	}
	long long ans = 0;
	for(x=1;x<=4;x++)
		for(y=1;y<=4;y++)
			ans += g[t][n][x][y];
	printf("%I64d\n", ans);
	return 0;
}
