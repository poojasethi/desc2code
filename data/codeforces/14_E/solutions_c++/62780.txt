#include <iostream>
#include <algorithm>
#include <vector>
#include <cstdio>
#include <cstring>
#include <string>
#include <cmath>
using namespace std;
int n,t;
long long f[30][40][10];
int main() {
	cin>>n>>t;
    for (int i=1;i<=4;++i) f[1][0][i]=1;
    for (int i=2;i<=n;++i)
        for (int j=1;j<=t+t;++j)
            for (int k=1;k<=4;++k)
            {
                f[i][j][k] = 0;
                if ((j&1)==1)
                for (int p=1;p<k;++p)
                    f[i][j][k]+=f[i-1][j][p]+f[i-1][j-1][p];
                else
                for (int p=k+1;p<=4;++p)
                    f[i][j][k]+=f[i-1][j][p]+f[i-1][j-1][p];
            }
    long long ans = f[n][t+t][1]+f[n][t+t][2]+f[n][t+t][3]+f[n][t+t][4];
    cout<<ans<<endl;

	return 0;
}
