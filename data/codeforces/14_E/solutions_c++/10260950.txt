#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int dp[21][5][5][11][11];
int main()
{
  int ans,n,t,i,j,k,x,y,l;
  scanf("%d%d",&n,&t);
  memset(dp,0,sizeof(dp));
  for (i=1;i<=4;i++)
	for (j=1;j<=4;j++) dp[2][i][j][0][0]=i!=j;
  for (i=3;i<=n;i++)
	for (j=1;j<=4;j++)
	  for (k=1;k<=4;k++)
for (l=1;l<=4;l++)
		  for (x=0;x<=t;x++)
			for (y=0;y<=t;y++)
			  
				if (k!=l)
	dp[i][k][l][x+(j<k&&k>l)][y+(j>k&&k<l)]+=dp[i-1][j][k][x][y];
  ans=0;
  for (i=1;i<=4;i++)
	for (j=1;j<=4;j++) ans+=dp[n][i][j][t][t-1];
  printf("%d\n",ans);
  return 0;
}
