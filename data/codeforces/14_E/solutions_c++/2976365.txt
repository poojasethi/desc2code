#include <iostream>
#include <memory.h>
#include <cstdio>

using namespace std;

int n, t;
int memo[22][6][12][12][2];
int DP(int x, int py, int nt, int nmot, bool up) {
    if(x == n) {
        return (nt == t && nmot == t - 1 && !up);
    }
    if(nt > t)
        return 0;

    int &r = memo[x][py][nt][nmot][up];
    if(r != -1)
        return r;
    r = 0;
    for(int y = py + 1; y < 5; ++y) {
        r += DP(x + 1, y, nt, nmot + !up, true);
    }
    for(int y = py - 1; y > 0 && x > 1; --y) {
        r += DP(x + 1, y, nt + up, nmot, false);
    }
    return r;
}

int main() {

    memset(memo, -1, sizeof(memo));
    int res = 0;
    cin >> n >> t;
    for(int y = 1; y < 4; ++y) {
        res += DP(1, y, 0, 0, true);
    }
    cout << res << endl;

    return 0;
}
