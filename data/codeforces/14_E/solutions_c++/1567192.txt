#include <cstdio>

using namespace std;

int f[21][11][5][2];

int main(void)
{
  int n,t;
  scanf("%d %d",&n,&t);
  for(int i=1;i<5;++i)
    f[1][0][i][0]=1;
  for(int i=2;i<=n;++i)
  {
    for(int j=0;j<=t;++j)
    {
      for(int k=1;k<5;++k)
      {
        for(int p=1;p<5;++p)
        {
          if(p<k)
            f[i][j][k][0]+=f[i-1][j][p][0]+f[i-1][j][p][1];
          else if(p>k)
          {
            f[i][j][k][1]+=f[i-1][j][p][1];
            if((j>0)&&(i>2))
              f[i][j][k][1]+=f[i-1][j-1][p][0];
          }
        }
      }
    }
  }
  int ans=0;
  for(int i=1;i<5;++i)
    ans+=f[n][t][i][1];
  printf("%d\n",ans);
  return 0;
}
