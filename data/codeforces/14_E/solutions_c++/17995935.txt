#include<stdio.h>
long long dp[23][12][6][2]={0};//i,totalhump,lastdigit,up/down
long long n,i,j,k,l,t,ans=0;
int main()
{
   scanf("%d %d",&n,&t);
   dp[1][0][1][1]=1;
   dp[1][0][2][1]=1;
   dp[1][0][3][1]=1;
   dp[1][0][4][1]=1;
   for(i=2;i<=n;i++)
   {
      for(j=0;j<=t;j++)
      {
         for(k=1;k<=4;k++)//up
         {
            for(l=1;l<k;l++)
            {
              dp[i][j][k][1]+=(dp[i-1][j][l][1]+dp[i-1][j][l][0]);
            }
         }
         for(k=1;k<=4;k++)//down
         {
            for(l=k+1;l<=4;l++)
            {
              dp[i][j][k][0]+=dp[i-1][j][l][0];
              if(i>2&&j>0) dp[i][j][k][0]+=dp[i-1][j-1][l][1];
            }
         }
      }
   }
   for(i=1;i<=4;i++)
   {
     ans+=dp[n][t][i][0];
   }
   printf("%lld",ans);
return 0;
}
