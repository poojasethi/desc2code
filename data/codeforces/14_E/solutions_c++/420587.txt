#include <iostream>
using namespace std;

#define D 0
#define U 1

int main() {
	int n, t;
	cin >> n >> t;

	long long cnt[21][5][2][11] = {};
	for (int y = 1; y <= 4; y++) {
		cnt[1][y][U][0] = y-1;
	}
	for (int x = 2; x < n; x++) {
		for (int y0 = 1; y0 <= 4; y0++) {
			for (int k = 0; k <= t; k++) {

				for (int y = 1; y < y0; y++) {
					cnt[x][y0][U][k] += cnt[x-1][y][U][k] +
					                    cnt[x-1][y][D][k];
				}
				for (int y = 4; y > y0; y--) {
					cnt[x][y0][D][k] += cnt[x-1][y][D][k] +
					           (k > 0 ? cnt[x-1][y][U][k-1] : 0);
				}

			}
		}
	}

	long long ans = 0;
	for (int y = 1; y <= 4; y++) {
		ans += cnt[n-1][y][D][t];
	}
	cout << ans << endl;
	return 0;
}
