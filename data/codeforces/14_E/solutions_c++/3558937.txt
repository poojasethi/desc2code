#include <cstdio>
#include <cstring>
#include <cstdlib>

int n,t;
long long f[25][5][5][25][25];

int main()
{
	scanf("%d%d",&n,&t);
	for (int i=1;i<=4;++i){
		for (int j=1;j<=4;++j){
			if (i!=j){
				f[2][i][j][0][0]=1;
			}
		}
	}
	for (int i=2;i<n;++i){
		for (int x=1;x<=4;++x){
			for (int y=1;y<=4;++y){
				for (int c1=0;c1<=i;++c1){
					for (int c2=0;c2<=i;++c2){
						if (f[i][x][y][c1][c2]){
							for (int z=1;z<=4;++z){
								if (z!=y){
									int d1=0,d2=0;
									if (x>y && y<z) d2=1;
									if (x<y && y>z) d1=1;
									f[i+1][y][z][c1+d1][c2+d2]+=f[i][x][y][c1][c2];
								}
							}
						}
					}
				}
			}
		}
	}
	long long ans=0;
	for (int x=1;x<=4;++x){
		for (int y=1;y<=4;++y){
			ans+=f[n][x][y][t][t-1];
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
