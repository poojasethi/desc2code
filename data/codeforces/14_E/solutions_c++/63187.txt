#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
using namespace std;

int dp[30][11][11][5][5];

int n, t;
int f(int d, int up, int down, int a, int b) {
	if(up > t || down > t-1) return 0;
	if(d >= n) {
		if(up == t && down == t-1) {
			return 1;
		}
		return 0;
	}
	if(dp[d][up][down][a][b] != -1) return dp[d][up][down][a][b];
	int &ret = dp[d][up][down][a][b];
	ret = 0;
	for(int x = 1; x<=4; x++) {
		if(x == b) continue;
		if(a < b && b > x) ret += f(d+1, up+1, down, b, x);
		else if(a > b && b < x) ret += f(d+1, up, down+1, b, x);
		else ret += f(d+1, up, down, b, x);
	}
	return ret;
}

int main() {
	scanf("%d%d", &n, &t);
	memset(dp, -1, sizeof(dp));
	int ret = 0;
	for(int a=1; a<=4; a++) for(int b=1; b<=4; b++) if(a!=b) 
		ret += f(2, 0, 0, a, b);
	printf("%d\n", ret);
}
