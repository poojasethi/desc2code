#include <iostream>
#include <vector>

#define REP(i,n) for(int i = 0; i < (n); ++i)
using namespace std;

int main()
{
    int n, t;
    cin >> n >> t;
    int dp[21][4][21] = {};
    dp[0][0][0]=1;
    dp[0][1][0]=1;
    dp[0][2][0]=1;
    REP(i,n+1)REP(j,4)REP(k,20)REP(l,4)
    if(i == 1 && j > l || i > 1 && j != l)
        dp[i][j][k+(j<l!=k%2)] += dp[i-1][l][k];
    int ans = 0;
    REP(j,4) ans += dp[n-1][j][t*2-1];
    cout << ans << endl;

    return 0;
}
