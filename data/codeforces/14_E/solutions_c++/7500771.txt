#include <cstdio>
using namespace std;
int f[30][5][20],g[30][5][20];
int n,t;
int main()
{
	scanf("%d%d",&n,&t);
	f[1][1][0]=f[1][2][0]=f[1][3][0]=f[1][4][0]=1;
	for (int i=1;i<=4;i++)
		for (int j=1;j<i;j++)
			f[2][i][0]+=f[1][j][0];
	for (int i=3;i<=n;i++)
		for (int j=1;j<=4;j++)
		{
			for (int k=1;k<j;k++)
				f[i][j][0]+=f[i-1][k][0];
			for (int k=1;k<=t;k++)
			{
				for (int p=1;p<j;p++) f[i][j][k]+=f[i-1][p][k]+g[i-1][p][k];
				for (int p=4;p>j;p--) g[i][j][k]+=f[i-1][p][k-1]+g[i-1][p][k];
			}
		}
	int ans=0;
	for (int i=1;i<=4;i++)
		ans+=g[n][i][t];
	printf("%d\n",ans);
	getchar();getchar();
	return 0;
}
