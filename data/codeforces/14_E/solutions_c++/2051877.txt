#include<iostream>
#include<cstdio>

using namespace std;

int n , m;
int f[25][20][4][2];

int main() {
	scanf( "%d%d" , &n , &m );
	f[2][1][2][1] = 1;
	f[2][1][3][1] = 2;
	f[2][1][4][1] = 3;
	for ( int i = 3 ; i <= n ; i++ ) {
		for ( int j = 1 ; j <= m ; j++ )
			for ( int k = 2 ; k <= 4 ; k++ )
				for ( int l = 1 ; l < k ; l++ )
					f[i][j][k][1] += f[i - 1][j - 1][l][0] + f[i - 1][j][l][1];
		
		for ( int j = 1 ; j <= m ; j++ )
			for ( int k = 1 ; k <= 3 ; k++ )
				for ( int l = k + 1 ; l <= 4 ; l++ )
					f[i][j][k][0] += f[i - 1][j][l][1] + f[i - 1][j][l][0];
	}
	int ans = 0;
	ans = f[n][m][1][0] + f[n][m][2][0] + f[n][m][3][0];
	printf( "%d\n" , ans );
	
	return 0;
}
