#include <cstdio>
#include <cstring>
#include <algorithm> 
using namespace std;

int n , T , dp[22][5][5][21][22] , ans ; 
#define M 1000000007

int f(int len , int prv1 , int prv2 , int mx , int mn)
{    
  if(len==n)  return (mx==T && mn==T-1) ;   
  if(dp[len][prv1][prv2][mx][mn]!=-1) return dp[len][prv1][prv2][mx][mn] ;
  
  int x=0 ;
  for(int i=1 ; i<=4 ; i++) if(i!=prv1)  x=(x+f(len+1,i,prv1,mx+(prv1>max(i,prv2)),mn+(prv1<min(i,prv2))) )%M ; 
     
 return dp[len][prv1][prv2][mx][mn]=x ;
}

int main()
{
   scanf("%d%d",&n,&T) ; 

   memset(dp,-1,sizeof(dp)) ;
   for(int i=1 ; i<=4 ; i++) for(int j=1 ; j<=4 ; j++) if(i!=j) ans=(ans+f(2,j,i,0,0))%M ;
   printf("%d",ans) ;
    
 return 0;
}
