#include<iostream>
#include<cstdio>
using namespace std;
int f[25][15][2][5];
int main()
{
    int n,t,i,j,k,s;
    scanf("%d%d",&n,&t);
    f[2][0][0][2]=1;
    f[2][0][0][3]=2;
    f[2][0][0][4]=3;
    for(i=3;i<=20;i++)
    {
        for(j=0;j<=10;j++)
        {
            for(k=1;k<=4;k++)
            {
                for(s=1;s<k;s++)
                {
                    f[i][j][0][k]+=f[i-1][j][0][s];
                    f[i][j][0][k]+=f[i-1][j][1][s];
                }
                for(s=4;s>k;s--)
                {
                    f[i][j][1][k]+=f[i-1][j][1][s];
                    f[i][j][1][k]+=f[i-1][j-1][0][s];
                }
            }
        }
    }
    int ans=0;
    for(i=1;i<=4;i++)
        ans+=f[n][t][1][i];
    printf("%d\n",ans);
    return 0;    
}
