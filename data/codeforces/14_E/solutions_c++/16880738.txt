#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

#define rep(i,a,n) for(int i=(a); i<(n); ++i)
int n, t, dp[20][4][4][11][11]; // pos, height of pos-1, height of pos, t humps, t-1 cavities

int main() {
    //freopen("E.in", "r", stdin);
    cin >> n >> t;
    rep(i,0,4) rep(j,0,4) dp[1][i][j][0][0] = i!=j;
    rep(i,2,n) rep(j,0,4) rep(k,0,4) rep(a,0,t+1) rep(b,0,t+1) rep(c,0,4) if(c!=k)
        dp[i][k][c][a+(j<k && k>c)][b+(j>k && k<c)] += dp[i-1][j][k][a][b];

    int ans = 0;
    rep(i,0,4) rep(j,0,4) ans += dp[n-1][i][j][t][t-1];
    cout<< ans <<endl;
    return 0;
}
