#include <cstdio>
#include <iostream>
using namespace std;

int n, t;
long long dp[21][11][10];
int main() {
	scanf("%d%d", &n, &t);
	dp[0][0][1] = 1;
	for (int i = 1; i <= n; i++) {
		for (int j = 0; j <= t; j++) {
			for (int k = 0; k < 5; k++) {
				for (int up = 0; up < 2; up++) {
					for (int pic = 1; pic < 5; pic++) {
						if (pic == k)
						  continue;
						if (up ^ (pic < k ? 1 : 0)) {
							if (up || i > 2)
								dp[i][j][pic<<1|up] += dp[i-1][j][k<<1|up];
						} else if (up && pic < k) {
							if (i > 2)
								dp[i][j+1][pic<<1|0] += dp[i-1][j][k<<1|up];
						} else {
							dp[i][j][pic<<1|1] += dp[i-1][j][k<<1|up];
						}
					}
				}
			}
		}
	}
	long long ans = 0;
	for (int k = 1; k < 5; k++)
		ans += dp[n][t][k<<1];
	printf("%I64d\n", ans);
	return 0;
}
