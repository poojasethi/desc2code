#include<iostream>
#include<cstdio>
using namespace std;
int dp[21][21][21][2];
int main()
{
	int n,t;
	scanf("%d%d",&n,&t);
	for(int i=1;i<=4;i++)dp[1][i][0][0]=1;
	for(int i=2;i<=n;i++)
	{
		for(int j=1;j<=4;j++)
		{
			for(int k=0;k<=min(i,t);k++)
			{
				for(int m=1;m<=j-1;m++)
				{
					dp[i][j][k][0]+=dp[i-1][m][k][0]+dp[i-1][m][k][1];
				}
				if(i!=2)for(int m=j+1;m<=4;m++)
				{
					dp[i][j][k][1]+=dp[i-1][m][k-1][0]+dp[i-1][m][k][1];
				}
			}
		}
	}
	int ans=0;
	for(int i=1;i<=4;i++)ans+=dp[n][i][t][1];
	printf("%d\n",ans);
	return 0;
}
