#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

int N, T, f[50][60][10];

int main()
{
	int i, j, k, l;
	scanf("%d%d", &N, &T);
	for (i = 1; i <= 4; ++i)
		f[1][0][i] = 1;
	for (i = 2; i <= N; ++i)
		for (j = 0; j < 2 * T; ++j)
			for (k = 1; k <= 4; ++k)
			{
				if (i == 2 && j >= 1)
					break;
				if (j % 2 == 0)
					for (l = 1; l < k; ++l)
						f[i][j][k] += f[i-1][j][l] + f[i-1][j-1][l];
				else
					for (l = 4; l > k; --l)
						f[i][j][k] += f[i-1][j][l] + f[i-1][j-1][l];
			}
	int res = 0;
	for (i = 1; i < 5; ++i)
		res += f[N][2*T-1][i];
	printf("%d\n", res);
	return 0;
}
