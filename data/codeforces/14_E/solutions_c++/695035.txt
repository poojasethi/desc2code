#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long LL;
int n,t;
LL dp[22][22][4][2];

LL sol(int pos,int ct,int last,int s)
{
    LL &ret = dp[pos][ct][last][s];
    if(ret != -1)return ret;
    ret = 0;
    if(pos == n)
    {
        if(ct==t && s==0)ret=1;
        else ret=0;
        return ret;
    }
    for(int i = 0; i < 4; i++)if(i!=last||pos==0)
    {
        if(pos == 1 && i < last)continue;
        int add=0;
        if(s==1 && i<last)add=1;
        ret += sol(pos+1,ct+add,i,i>=last);
    }
    return ret;
}

int main()
{
    scanf("%d%d",&n,&t);
    memset(dp,-1,sizeof(dp));
    LL ret = sol(0,0,0,1);
    printf("%I64d\n",ret);

    return 0;
}
