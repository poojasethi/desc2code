#include<stdio.h>
__int64 dp[33][14][4][2], ans;
int main()
{
	int i, j, k, h, n, t;
	scanf("%d%d", &n, &t);
	for(i = 0; i < 3; i++)
		dp[1][1][i][0] = 1;
	for(i = 2; i <= n ;i++)
		for(j = 1; j <= t; j++)
			for(h = 0; h < 4; h++)
			{
				for(k = 0; k < h; k++)
					dp[i][j][h][0] += dp[i-1][j][k][0] + dp[i-1][j-1][k][1];
				if(i == 2)
					dp[i][j][h][1] = 0;
				else
				for(k = 3; k > h; k--)
				{
					dp[i][j][h][1] += dp[i-1][j][k][0] + dp[i-1][j][k][1];
				}
			}
	for(i = 0; i < 4; i++)
		ans += dp[n][t][i][1];
	printf("%d\n", ans);
	return 0;
}
