#include <stdio.h>
#include <iostream>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int n, t;
int dp[20][4][4][15];

int main() {
    scanf("%d%d", &n, &t);
    rep(i, 4) rep(j, i) dp[0][j][i][0] = 1;
    rep(i, n-2) rep(x, 4) rep(y, 4) if(x!=y) rep(k, t+1) rep(z, 4) if(y!=z) {
        if(x<y && y>z) dp[i+1][y][z][k+1] += dp[i][x][y][k];
        else dp[i+1][y][z][k] += dp[i][x][y][k];
    }
    long long ans = 0;
    rep(i, 4) rep(j, i) {
        ans += dp[n-2][i][j][t];
    }
    cout << ans << endl;
    return 0;
}
