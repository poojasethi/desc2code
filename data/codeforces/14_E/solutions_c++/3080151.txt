#include<cstdio>

int f[25][4][15][15][2], n, t;

int main() {
    scanf("%d%d",&n,&t);
    for(int y = 0; y < 4; ++y)
        for(int up = 0; up < 2; ++up)
            f[2][y][0][0][up] = up ? y : 3 - y;
    for(int x = 2; x < n; ++x) for(int y = 0; y < 4; ++y)
        for(int h = 0; h <= t; ++h) for(int r = 0; r <= t; ++r)
            for(int up = 0; up < 2; ++up)
                for(int yy = 0; yy < 4; ++yy) if(yy != y)
                    f[x+1][yy][(up && yy < y) + h][(!up && yy > y) + r][yy > y] += f[x][y][h][r][up];
    int res = 0;
    for(int y = 0; y < 4; ++y)
        for(int up = 0; up < 2; ++up)
            res += f[n][y][t][t-1][up];
    printf("%d\n", res);
    return 0;
}
