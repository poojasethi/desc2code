#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#define N (1<<21)
#define db double
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const db pi=acos(-1);
struct ww {
	db x,y;
	inline ww operator + (const ww &A) {
		return (ww){x+A.x,y+A.y};
	}
	inline ww operator - (const ww &A) {
		return (ww){x-A.x,y-A.y};
	}
	inline ww operator * (const ww &A) {
		return (ww){x*A.x-y*A.y,x*A.y+y*A.x};
	}
} w[2][N],a[N];
int i,j,k,n,m,nn;
int rev[N],c[N],b[N];
inline void pre() {
	For(i,0,nn-1) {
		int y=0;
		for (j=1,k=i;j<nn;j<<=1,k>>=1) (y<<=1)|=k&1;
		rev[i]=y;
	}
	For(i,0,nn-1) {
		db A=2*pi/nn*i;
		w[0][i]=w[1][i]=(ww){cos(A),sin(A)};
		w[1][i].y*=-1;
	}
}
inline void FFT(ww a[N],int f) {
	int i,j,l,t,u,k;
	For(i,0,nn-1) if (i<rev[i]) swap(a[i],a[rev[i]]);
	for (i=1;i<nn;i<<=1)
	for (j=0,l=i<<1,t=nn/l;j<nn;j+=l)
	for (u=k=0;u<i;u++,k+=t) {
		ww y=a[j+u],x=a[j+u+i];
		a[j+u]=y+w[f][k]*x;
		a[j+u+i]=y-w[f][k]*x;
	}
	if (f) For(i,0,nn-1) a[i].x/=nn;
}
int main() {
	scanf("%d%d",&n,&m);
	for (nn=1;nn<=m;nn<<=1); nn<<=1;
	pre();
	For(i,1,n) scanf("%d",&k),a[k].x=1,c[k]=1;
	FFT(a,0);
	For(i,0,nn-1) a[i]=a[i]*a[i];
	FFT(a,1);
	For(i,0,nn-1) b[i]=int(a[i].x+0.1);
	For(i,1,m) if (b[i]&&!c[i]) {
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	k=0; For(i,1,m) if (c[i]&&!b[i]) k++;
	printf("%d\n",k);
	For(i,1,m) if (c[i]&&!b[i]) printf("%d ",i);
	printf("\n");
	return 0;
}
