#include<bits/stdc++.h>
#define U 1<<21
using namespace std;
const double pi=acos(-1.);
struct data{double x,y;}w[U][2],a[U];
int rev[U],b[U],N,n,m,i,x,cnt;
data operator + (const data &a,const data &b){return (data){a.x+b.x,a.y+b.y};}
data operator - (const data &a,const data &b){return (data){a.x-b.x,a.y-b.y};}
data operator * (const data &a,const data &b){return (data){a.x*b.x-a.y*b.y,a.x*b.y+a.y*b.x};}
void FFT(data *a,int o){
    for (int i=0;i<N;i++)
        if (i<rev[i]) swap(a[i],a[rev[i]]);
    data A,B;
    for (int i=1;i<N;i<<=1)
        for (int j=0,t=N/(i<<1);j<N;j+=i<<1)
            for (int k=0,l=0;k<i;k++,l+=t){
                A=a[j+k+i]*w[l][o];B=a[j+k];
                a[j+k+i]=B-A;a[j+k]=B+A;
          }
    if (o) for (int i=0;i<N;i++) a[i].x/=N;
}
void init(int n){
    for (N=1;N<=n;N<<=1);N<<=1;
    for (int i=0;i<N;i++){
        int x=i,&y=rev[i];
        for (int k=1;k<N;k<<=1,x>>=1) (y<<=1)|=x&1;
  }for (int i=0;i<N;i++)
    w[i][0]=w[i][1]=(data){cos(2*pi*i/N),sin(2*pi*i/N)},w[i][1].y=-w[i][1].y;
}
int main(){
    scanf("%d%d",&n,&m);
    for (i=1;i<=n;i++)
        scanf("%d",&x),b[x]=a[x].x=1;
    init(m);
    FFT(a,0);
    for (i=0;i<N;i++)
        a[i]=a[i]*a[i];
    FFT(a,1);
    for (i=1;i<=m;i++)
        if (a[i].x>1-1e-7&&!b[i])
            return puts("NO"),0;
    puts("YES");
    for (i=1;i<=m;i++)
        if (b[i]&&a[i].x<1-1e-7)
            cnt++;
    printf("%d\n",cnt);
    for (i=1;i<=m;i++)
        if (b[i]&&a[i].x<1-1e-7)
            printf("%d ",i);
}
