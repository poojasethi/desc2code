#define MAXN 2111111
#define eps 1e-6
using namespace std;
typedef complex<double> cp;

const cp I(0,1);
int n,n2,m, ansset[MAXN], an;
cp a[MAXN], b[MAXN];
bool c[MAXN];

void fft(int n, double t, cp *a)
{
    int m,h,i,j,k;
    cp w,x;
    for(m=n;h=m/2,m>=2;t*=2,m=h)
        for(i=0;i<h;i++)
            for(w=exp(i*t*I),j=i;j<n;j+=m)
                k=j+h,x=a[j]-a[k],a[j]+=a[k],a[k]=w*x;
    i=0;
    for(j=1;j<n-1;j++)
    {
        for(k=n/2;k>(i^=k);k/=2);
        if(j<i)swap(a[i],a[j]);
    }
    return;
}

int main()
{
    int i, tem;
    /*freopen("d:\\1.in","r",stdin);
    freopen("d:\\1.out","w",stdout);*/
    scanf("%d%d",&n,&m);
    memset(a, 0, sizeof(a));
    memset(b, 0, sizeof(b));
    for(i=0;i<n;i++) scanf("%d", &tem), a[tem]=1, c[tem]=true;
    n2=1<<21;   // The expected range
    fft(n2, 2*M_PI/n2, a);
    for(i=0;i<n2;i++) b[i]=a[i]*a[i];
    fft(n2, -2*M_PI/n2, b);
    for(int i=1;i<=m;i++)
        if(fabs(b[i].real())>0.5*n2&&!c[i])   //  Representable but isn't a bag.
        {
            puts("NO\n");
            return 0;
        }
    an=0;
    for(int i=1;i<=m;i++)
        if(fabs(b[i].real())<0.5*n2&&c[i])ansset[an++]=i;    //Not representable and is a bag.
    printf("YES\n%d\n", an);
    for(i=0;i<an;i++)
        printf("%d ", ansset[i]);
    printf("\n");
    return 0;
}
