#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
const ll inf = 1e18;
#define SQ(x) ((x) * (x))

int N;
deque<pair<int, int> > T[100405]; // T[t] = SORTED list at time t
ll ans[100405];

int main()
{
	cin >> N;
	for(int i = 1; i <= N; i++)
	{
		int l, t;
		cin >> l >> t;
		T[t].push_back(make_pair(l, i));
// 		printf("pushed t=%d l=%d i=%d\n", t, l, i);
	}
	for(int i = 0; i <= 100000; i++) sort(T[i].begin(), T[i].end());
	int p = 0;
	ll time = 0;
	for(int iter = 1; iter <= N; iter++)
	{
		while(T[p].empty())
		{
			p++;
			if(p > time) time = p;
		}
// 		cout << "p=" << p << endl;
		ll best = inf, bestt, bestl, besti;
		for(int t = p; t <= p + 400 && t <= time; t++)
		{
			if(T[t].empty()) continue;
			ll l = T[t].front().first, i = T[t].front().second;
			ll v = l - SQ(min(time, 1000000LL) - t);
			if(v < best || (v == best && i < besti))
				best = v, bestt = t, bestl = l, besti = i;
		}
		T[bestt].pop_front();
// 		cout << "at time " << time << ", besti = " << besti << ", bestl = " << bestl << endl;
		time += bestl;
		ans[besti] = time;
	}
	for(int i = 1; i <= N; i++) cout << ans[i] << ' ';
}
