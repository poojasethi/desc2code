#include <iostream>
#include <cstdio>
#include <algorithm>
#define N 1000010
using namespace std;

char s[N];
int main (){
	scanf("%s", s);
	int a = 0, b = 1;
	for(int i = 0; s[i]; i++){
		if(s[i] == '1'){
			a = min(a, b) + 1;
		}
		else{
			b = min(a, b) + 1;
		}
	}
	cout << min(a, b+1) << endl;
}
