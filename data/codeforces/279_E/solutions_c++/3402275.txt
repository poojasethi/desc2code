#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    char s[1000010];
    cin>>s;
    int dp0=0,dp1=1;
    for(int i=0;s[i];i++)
       if(s[i]-'0')
          dp0=min(dp0,dp1)+1;
       else
          dp1=min(dp0,dp1)+1;
    dp1=min(dp0,dp1)+1;
    int ANS=min(dp0,dp1);
    cout<<ANS<<endl;
    return 0;
}
