# include<stdio.h>
int min(int a,int b)
{
    if(a<b)
        return a;
    else
        return b;
}
int main()
{
    int a=0,b=1;
    char c;
    while((c=getchar())!='\n')
    {
        if(c=='1')
            a=min(a,b)+1;
        else
            b=min(a,b)+1;
    }
    printf("%d\n",a);
    return 0;
}
