#include<iostream>
#include<cstring>
using namespace std;

char str[1001000];

int main()
{
    while(cin>>str){
        int len = strlen(str);
        int a=0,b=1;
        for(int i=0;i<len;i++){
            if(str[i]=='1') a = min(a,b)+1;
            else b = min(a,b)+1;
        }
        cout<<a<<endl;
    }
}
