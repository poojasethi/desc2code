#include <cstdio>
#include <iostream>
using namespace std;
int main(){
	string inp;
	cin>>inp;
	inp = inp + "000";
	int number = 0;
	for (int i= 0;i<inp.length();i++){
		if (inp[i] == '1'){
			number++;
			if (inp[i+1]=='1'){
				while(inp[i]=='1')
					i++;
				i--;
				inp[i+1]='1';
			}
		}
	}
	cout<<number<<endl;
}
