#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
string s;
int main(){
    cin >> s;
    int atas = 1;
    int bawah = 0;
    for(int i=s.size()-1;i>=0;i--){
        if(s[i]=='0')
            atas = min(bawah,atas) + 1;
        else
            bawah = min(bawah,atas) + 1;
    }
    cout<<bawah<<endl;
}
