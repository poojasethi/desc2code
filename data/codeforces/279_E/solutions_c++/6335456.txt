#include<bits/stdc++.h>
using namespace std;
char s[1000005];
int main(){
    ios_base::sync_with_stdio(0);cin.tie(0);
    int c1=0,c2=1,m=1,i;
    scanf("%s",s);
    for(i=0;s[i];i++){
      if(s[i]=='1')
        c1=m;
      else
        c2=m;
      m=min(c1,c2)+1;
    }
    c2=m;
    printf("%d",min(c1,c2));
}
