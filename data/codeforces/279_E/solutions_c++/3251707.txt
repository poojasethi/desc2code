#include<iostream>
#include<string>
using namespace std;
int i,n,rez;
string s;

int main(void) {
    cin>>s; s="00"+s; n=s.length()-1;
    i=n;  while (s[i]=='0') --i;
     while (i>0) {
          ++rez;
          char ch=s[i-1];
          --i;
          while (s[i]==ch&&i>=0) --i;
          }
    cout<<rez;
 return(0);
}
