#include <iostream>
#include <fstream>
#include <algorithm>
#include <vector>
using namespace std;

int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	//ifstream cin("a.in");
	int a0=0, a1=1;
	string s;
	cin >> s;
	for (int i=0; i<int(s.size()); ++i) {
		if (s[i]=='0') a1 = min(a0,a1)+1;
		else a0=min(a0,a1)+1;
	}
	cout << a0;
	return 0;
}
