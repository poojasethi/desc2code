s = raw_input() + '00'

ans = 1
canHaveZero = True
canHaveOne = True

for i in range(1,len(s)):
	if ((s[i] == '0') & canHaveZero):
		canHaveOne = False
	elif ((s[i] == '0') & (not canHaveZero)):	
		ans += 1
		canHaveZero = True
	elif ((s[i] == '1') & canHaveOne):
		canHaveZero = False
	elif ((s[i] == '1') & (not canHaveOne)):
		ans += 1
		canHaveOne = True

print ans
