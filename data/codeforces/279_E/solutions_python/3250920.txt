import re
from sys import stdin

def line():
	return stdin.readline().strip()

s = line() + '00'

P = re.split('00+', s)

ans = 0
for p in P:
	if p == '':
		continue
	zeros = p.count('0')
	ones  = len(p) - zeros
	if ones == 1:
		ans += 1
	elif ones - 1 == zeros or ones == zeros:
		ans += ones
	else:
		ans += zeros + 2
print ans
