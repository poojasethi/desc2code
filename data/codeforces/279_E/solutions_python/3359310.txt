def f(s):
    ans = 0
    s = '0' + s + '0'
    for i in range(len(s) - 2):
        if s[i] == '0' and s[i + 1] == '1':
            ans += 2
    i = 1
    while i + 1 < len(s):
        if s[i - 1] == s[i + 1] and s[i - 1] != s[i]:
            i += 2
            ans -= 1
        else:
            i += 1
    return ans

s = raw_input()
ans = f(s)
print ans
