#include <iostream>
#include <algorithm>
#include <cstdlib>

using namespace std;

int n;
pair<int,int> a[1000];

int main()
{
  cin>>n;
  for (int i=0;i<n;i++) {
    cin>>a[i].first;
    a[i].second=i;
  }
  sort(a,a+n);
  for (int i=2;i<n-1;i++) {
    if (a[i].first!=a[i-1].first) {
      cout<<"Unrecoverable configuration."<<endl;
      exit(0);
    }
  }
  if (a[0].first==a[n-1].first) {
    cout<<"Exemplary pages."<<endl;
    exit(0);
  }
  if ((n<=2 and (a[1].first-a[0].first)%2==0) or a[0].first+a[n-1].first==2*a[1].first)
    cout<<(a[n-1].first-a[0].first)/2<<" ml. from cup #"<<a[0].second+1<<" to cup #"<<
	a[n-1].second+1<<"."<<endl;
  else
    cout<<"Unrecoverable configuration."<<endl;
}
