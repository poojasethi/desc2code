#include <iostream>
#include <cstdio>

using namespace std;

const int maxn=1009;

int h[maxn];
int sum,ave;
int n,i;
int a,b;
int cnt;

void count()
{
	for(i=1;i<=n;i++)
	{
		if(h[i]!=ave)
		{
			cnt++;
			if(h[i]>ave)
				b=i;
			else
				a=i;
		}
	}
}

int main()
{
	cin >> n;
	cnt=a=b=sum=0;
	for(i=1;i<=n;i++)
	{
		scanf("%d",h+i);
		sum+=h[i];
	}
	ave=sum/n;
	count();
	if(cnt==0)
		printf("Exemplary pages.\n");
	else if(cnt==2 && h[a]+h[b]==2*ave && ave*n==sum)
		printf("%d ml. from cup #%d to cup #%d.\n",h[b]-ave,a,b);
	else
		printf("Unrecoverable configuration.\n");
	return 0;
}
