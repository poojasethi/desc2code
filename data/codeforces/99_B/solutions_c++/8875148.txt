#include<cstdio>
#include<cstdlib>
using namespace std;
int aa[1005][2],bb[505][3];
int cmp(const void* aa,const void* bb){
	int *a=(int *)aa;
	int *b=(int *)bb;
	return *a-*b;
}
int main(){
	int n,avg=0;scanf("%d",&n);
	for(int i=0;i<n;i++){scanf("%d",&aa[i][0]);aa[i][1]=i+1;}
	qsort(aa,n,sizeof(aa[0]),cmp);
	avg=aa[0][0]+aa[n-1][0];
	if(aa[0][0]==aa[n-1][0])printf("Exemplary pages.\n");
	else if(avg%2!=0||(n>2&&(aa[1][0]!=avg/2||aa[n-2][0]!=avg/2)))printf("Unrecoverable configuration.\n");
	else printf("%d ml. from cup #%d to cup #%d.\n",(aa[n-1][0]-aa[0][0])/2,aa[0][1],aa[n-1][1]);
	return 0;
}
