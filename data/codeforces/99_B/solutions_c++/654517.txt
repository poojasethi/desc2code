#include<cstdio>
#include<algorithm>
using namespace std;

int n,a[1010],tot,t,s,ave;

int main()
{
	//freopen("99B.in","r",stdin);
	//freopen("99B.out","w",stdout);
	
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		tot+=a[i];
	}
	ave=tot/n;
	if (*max_element(a+1,a+n+1)==*min_element(a+1,a+n+1)) puts("Exemplary pages.");
	else if (tot%n) puts("Unrecoverable configuration.");
	else
	{
		for(int i=1;i<=n;i++) if (a[i]!=ave) t++;
		if (t>2) puts("Unrecoverable configuration.");
		else
		{
			for(int i=1;i<=n;i++) if (a[i]<ave) s=i; else if (a[i]>ave) t=i;
			printf("%d ml. from cup #%d to cup #%d.\n",a[t]-ave,s,t);
		}
	}
	
	return 0;
}
