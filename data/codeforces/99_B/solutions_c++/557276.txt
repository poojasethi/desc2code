#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int cup[1001];
	int N;
	cin>>N;
	int sum=0;
	for(int i=0;i<N;i++)
	{
		cin>>cup[i];
		sum+=cup[i];
	}
	int avg=sum/N;
	if(sum%N!=0)
		cout<<"Unrecoverable configuration."<<endl;
	else
	{
		int a,b,cnt=0;
		for(int i=0;i<N;i++)
			if(cup[i]==avg)
				cnt++;
			else if(cup[i]<avg)
				a=i;
			else if(cup[i]>avg)
				b=i;
		if(cnt==N)
			cout<<"Exemplary pages."<<endl;
		else if(cnt!=N-2)
			cout<<"Unrecoverable configuration."<<endl;
		else 
		{
			cout<<cup[b]-avg<<" ml. from cup #"<<a+1<<" to cup #"<<b+1<<"."<<endl;
		}
	}
	return 0;
}
