#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<iostream>
using namespace std;
main()
{
	int n,m[1000],avg,i,cnt,ra,rb;
	while(~scanf("%d",&n))
	{
		ra=rb=0;
		for(avg=i=0;i<n;i++)
		{
			scanf("%d",&m[i]);
			avg+=m[i];
		}
		if(avg%n!=0)
		{
			puts("Unrecoverable configuration.");
			continue;
		}
		avg/=n;
		for(i=cnt=0;i<n;i++)
			if(avg!=m[i])ra=rb,rb=i,cnt++;
		if(m[ra]>m[rb])swap(ra,rb);
		if(cnt>2)puts("Unrecoverable configuration.");
		else if(cnt==0)puts("Exemplary pages.");
		else
		{
			if((m[ra]-m[rb])%2==0)printf("%d ml. from cup #%d to cup #%d.\n",(m[rb]-m[ra])/2,ra+1,rb+1);
			else puts("Exemplary pages.");
		}
	}
}
