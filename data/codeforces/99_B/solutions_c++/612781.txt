
#include<iostream>
using namespace std;

long i, d, s, n, a[1010], j;

int main(){
  cin>>n;
  for (i = 1; i<=n;i++) cin>>a[i];
  for (s = 0, i = 1; i<=n;i++) s+= a[i];
  if (s % n != 0) {
    cout<<"Unrecoverable configuration.";
    return 0;
  }
  s = s / n;
  for (d = 0, i = 1; i <= n; i++) d += (a[i] == s);
  if (d == n) { cout<<"Exemplary pages."; return 0;}
  if (n - d != 2) {cout<<"Unrecoverable configuration."; return 0;}
  for (i = 1; i <= n; i++) if (a[i] < s) break;
  for (j = 1; j<=n;j++) if(a[j]>s) break;
  cout<<a[j] - s<<" ml. from cup #" << i <<  " to cup #"<<j<<".";
  return 0;
}
