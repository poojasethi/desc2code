#include<iostream>
using namespace std;
long long sum;
int a[1100];
int main()
{
  int n;
  cin>>n;
  for(int i=1;i<=n;i++)
    cin>>a[i],sum+=a[i];
  int x=sum/n,ans1,ans2,ans3,na=0;
  if(sum%n!=0)
    {
      cout<<"Unrecoverable configuration."<<endl;
      return 0;
    }
  for(int i=1;i<=n;i++)
    if(a[i]>x)
      na++,ans1=i;
    else if(a[i]<x)
      na++,ans2=i;
  if(na==0)
    cout<<"Exemplary pages."<<endl;
  else if(na>2 || na==1)
    cout<<"Unrecoverable configuration."<<endl;
  else
    {
      cerr<<ans1<<" "<<ans2<<endl;
      int y=a[ans1]-x,y2=x-a[ans2];
      if(y!=y2)
	cout<<"Unrecoverable configuration."<<endl;
      else
	cout<<y<<" ml. from cup #"<<ans2<<" to cup #"<<ans1<<"."<<endl;
    }
  return 0;
}
