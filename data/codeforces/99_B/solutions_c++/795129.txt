#include<cstdio>
int arr[1<<10],n;
int main()
{
    int i,sum=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",arr+i);
        sum+=arr[i];
    }
    if(sum%n)
    {
        puts("Unrecoverable configuration."); return 0;
    }
    sum/=n;
    int cnt=0;
    int li,mi;
    for(i=0;i<n;i++)
    {
        if(arr[i]<sum)
        {
            cnt++; li=i;
        }
        else if(arr[i]>sum)
        {
            cnt++; mi=i;
        }
    }
    if(cnt==0) puts("Exemplary pages.");
    else if(cnt==2) printf("%d ml. from cup #%d to cup #%d.\n",sum-arr[li],li+1,mi+1);
    else puts("Unrecoverable configuration.");
    return 0;
}
