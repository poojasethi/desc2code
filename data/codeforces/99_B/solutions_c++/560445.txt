#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int a[1010],b[1010],sum,n;

int main()
{
	while (scanf("%d",&n)!=EOF)
	{
		sum=0;
		for (int i=1;i<=n;i++)
		{
			scanf("%d",&a[i]);
			sum+=a[i];
		}
		int k=0;
		for (int i=1;i<=n;i++)
			if (a[i]*n!=sum)
			{
				k++;
				b[k]=i;
			}
		if (k==0)
			printf("Exemplary pages.\n");
		else if (k!=2)
			printf("Unrecoverable configuration.\n");
		else if (n>2 || n==2 && sum%2==0)
		{
			int x,y,z;
			if (a[b[1]]*n<sum)
			{
				x=1;
				y=2;
				z=sum/n-a[b[1]];
			}
			else
			{
				x=2;
				y=1;
				z=sum/n-a[b[2]];
			}
			printf("%d ml. from cup #%d to cup #%d.\n",z,b[x],b[y]);
		}
		else printf("Unrecoverable configuration.\n");
	}
	return 0;
}
