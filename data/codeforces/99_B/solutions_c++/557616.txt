#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int n,a[1005],sum=0;
    int less,more,l=0,m=0;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
       {scanf("%d",a+i);
       sum+=a[i];}
    if(sum%n!=0) printf("Unrecoverable configuration.\n");
    else {
    int av=sum/n;
    for(int i=1;i<=n;i++)
       if(a[i]<av) {l++;less=i;}
       else if(a[i]>av) {m++;more=i;}
    if(m==1&&l==1)
    {
    if(av-a[less]==a[more]-av)
    printf("%d ml. from cup #%d to cup #%d.\n",av-a[less],less,more);
    else printf("Unrecoverable configuration.\n");
    }
    else if(m==0&&l==0)
    printf("Exemplary pages.\n");
    else
    printf("Unrecoverable configuration.\n");
    }
}
