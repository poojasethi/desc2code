#include<stdio.h>
#include<vector>
using namespace std;

const int MAXN = 10000;
int a[MAXN];
vector<int> diff;

int main()
{
	int n;
	scanf("%d",&n);
	int sum = 0;
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		sum += a[i];
	}
	if (sum % n != 0)
	{
		printf("Unrecoverable configuration.\n");
		return 0;
	}	
	
	diff.clear();
	for (int i=1;i<=n;i++)
	{
		if (a[i] != sum/n)
		{
			//printf("dif %d\n",i);
			diff.push_back(i);
		}
	}
	if (diff.size() == 2)
	{
		if (a[diff[0]] > a[diff[1]]) swap(diff[0],diff[1]);
		printf("%d ml. from cup #%d to cup #%d.\n",a[diff[1]]-sum/n,diff[0],diff[1]);
	}else
	if (diff.size() == 0)
		printf("Exemplary pages.\n");
	else
		printf("Unrecoverable configuration.\n");
}
