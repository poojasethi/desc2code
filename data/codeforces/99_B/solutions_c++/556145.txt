#include <stdio.h>

int n, a[1007];

int main(void)
{
	int i, sum=0, cnt = 0, res1, res2, res3;
	scanf("%d", &n);

	for(i = 1; i <= n; i++)
	{
		scanf("%d", &a[i]);
		sum += a[i];
	}

	if(sum % n != 0)
	{
		printf("Unrecoverable configuration.\n");
		return 0;
	}
	sum /= n;
	for(i = 1; i <= n; i++)
	{
		if(sum == a[i]) cnt++;
		if(a[i] < sum){ res1 = sum-a[i]; res2 = i;}
		if(a[i] > sum){ res3 = i;}
	}

	if(cnt == n)
	{
		printf("Exemplary pages.\n");
		return 0;
	}

	if(cnt < n-2)
	{
		printf("Unrecoverable configuration.\n");
		return 0;
	}

	printf("%d ml. from cup #%d to cup #%d.\n", res1, res2, res3);

	return 0;
}
