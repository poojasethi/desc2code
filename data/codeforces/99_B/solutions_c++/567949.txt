#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;

int main(void)
{
    int n,t[1001],total=0;
    scanf("%d", &n);
    for ( int i = 1; i <= n; i++){
        scanf("%d", &t[i]);
        total += t[i];
    }
    if ( total % n){puts("Unrecoverable configuration."); return 0;}
    int avg = total/n;
    int a,b,count=0;
    for ( int i = 1; i <= n; i++){
        if ( t[i] == avg)count++;
        else if(t[i] < avg) a = i;
        else if ( t[i] > avg) b = i;
    }
    if ( count == n){puts("Exemplary pages."); return 0;}
    else if (count != n-2){puts("Unrecoverable configuration."); return 0;}
    printf("%d ml. from cup #%d to cup #%d.\n",t[b] - avg,a,b);
    return 0;
}
