#include<stdio.h>
int main()
{
	int n;
	while(scanf("%d",&n)!=EOF)
	{
		int cup[1010];
		int sum=0;
		for(int i=1;i<=n;i++)
		{
			int t;
			scanf("%d",&t);
			cup[i]=t;
			sum+=t;
		}
		if(sum%n!=0)
		{
			printf("Unrecoverable configuration.\n");
			continue;
		}
		int num=0;
		for(int i=1;i<=n;i++)
			if(cup[i]!=sum/n) num++;
		if(num==0)
		{
			printf("Exemplary pages.\n");
			continue;
		}
		if(num>2)
		{
			printf("Unrecoverable configuration.\n");
			continue;
		}
		int bi,si;
		for(int i=1;i<=n;i++)
		{
			if(cup[i]>sum/n) bi=i;
			if(cup[i]<sum/n) si=i;
		}
		int d=(cup[bi]-cup[si])/2;
		printf("%d ml. from cup #%d to cup #%d.\n",d,si,bi);
	}
	return 0;
}
