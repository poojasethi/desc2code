#include<iostream>
using namespace std;
int main()
{
    int a[1001],n;
    cin>>n;
    int i;
    for(i=0;i<n;i++)cin>>a[i];
    int min=0,max=0;
    for(i=0;i<n;i++){if(a[i]>a[max])max=i;if(a[i]<a[min])min=i;}
    int m=a[max]-a[min];
    if(m == 0)cout<<"Exemplary pages."<<endl;
    else if(m%2)cout<<"Unrecoverable configuration."<<endl;
    else
    {
        a[max]-=m/2;
        a[min]+=m/2;
        bool flag=true;
        for(i=1;i<n;i++)if(a[i]!=a[0]){flag=false;break;}
        if(flag)
        {
            cout<<m/2<<" ml. from cup #"<<min+1<<" to cup #"<<max+1<<"."<<endl;
        }
        else
        cout<<"Unrecoverable configuration."<<endl;
    }
    return 0;
}
