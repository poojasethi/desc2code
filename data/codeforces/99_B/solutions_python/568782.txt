# encoding: utf-8

import sys

num_cups = int(raw_input())

cups = []

for i in xrange(num_cups):
    cup = int(raw_input())
    cups.append(cup)

# Хак номер #9999
summ_hack = sum(cups)
if (1.0 * summ_hack) / num_cups > summ_hack / num_cups:
    print 'Unrecoverable configuration.'
    sys.exit()

# Хак?
if num_cups == 1:
    print 'Exemplary pages.'
    sys.exit()

# Можно не делать здесь лишних проходов
# по списку, однако я не буду этого делать

minimum = min(cups)
maximum = max(cups)

min_index = cups.index(minimum)
max_index_hack = cups.index(maximum)
cups.pop(min_index)
max_index = cups.index(maximum)
cups.pop(max_index)
max_index = max_index_hack

# Ещё один хак, да? # ДА Я ВСЁ ПОПРАВЛЮ, СТОПУДОВО
if num_cups == 2:
    if minimum == maximum:
        print 'Exemplary pages.'
    else:
        if (minimum + maximum) / 2.0 > (minimum + maximum) / 2:
            print 'Unrecoverable configuration.'
        else:
            print '%s ml. from cup #%s to cup #%s.' % (maximum - ((minimum + maximum) / 2), min_index + 1, max_index + 1)
    sys.exit()

new_minimum = min(cups)
new_maximum = max(cups)

if new_minimum == new_maximum:
    if minimum == maximum:
        print 'Exemplary pages.'
    else:
        print '%s ml. from cup #%s to cup #%s.' % (maximum - new_maximum, min_index + 1, max_index + 1)
else:
    print 'Unrecoverable configuration.'
