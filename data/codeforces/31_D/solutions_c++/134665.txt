#include <iostream>
#include <sstream>
#include <cstring>
#include <stdio.h>
#include <algorithm>
#include <queue>
#include <vector>
#include <set>
#include <map>
#include <memory.h>

using namespace std;

int xi[4]={1,-1,0,0},yi[4]={0,0,1,-1},w,h,n,x,y,xx,yy,p[200][200][4],f[200][200],o=0,e[10000];

void rec(int x,int y){
	if (x<0||x>=h||y<0||y>=w||f[x][y])return;
	f[x][y]=o+1;
	e[o]++;
	for (int i=0;i<4;i++)if (!p[x][y][i])rec(x+xi[i],y+yi[i]);
}

int main(){
	cin >> w >> h >> n;
	for (int i=0;i<n;i++){
		cin >> x >> y >> xx >> yy;
		swap(x,y);
		swap(xx,yy);
		if (y==yy)for (int j=x;j<xx;j++){
			p[j][y-1][2]=1;
			p[j][y][3]=1;
		}
		if (x==xx)for (int j=y;j<yy;j++){
			p[x-1][j][0]=1;
			p[x][j][1]=1;
		}
	}
	for (int i=0;i<h;i++)for (int j=0;j<w;j++)if (!f[i][j]){
	        rec(i,j);
	        o++;
	}
	sort(e,e+o);
	for (int i=0;i<o-1;i++)cout << e[i] <<" ";
	cout << e[o-1] << endl;
	return 0;
}
