n = input()
a = map(int,raw_input().split())
d = map(int,raw_input().split())
l = range(n)
for i in xrange(n):
    for j in xrange(n):
        for e in (j-d[j],j+d[j]):
            if e<0 or e>=n: continue
            l[e]=l[j]=min(l[j],l[e])
            
print "YES" if all((l[i]==l[x-1] for i,x in enumerate(a))) else "NO"
