[n], p, d = [map (int, raw_input ().split ()) for _ in xrange (3)]
d = [[1 if i == j or abs (i - j) == d[i] or abs (i - j) == d[j] else 0 for j in xrange (n)] for i in xrange (n)]
for k in xrange (n):
    for i in xrange (n):
        for j in xrange (n):
            d[i][j] = d[i][j] or d[i][k] and d[k][j]
#print '\n'.join ([''.join (map (str, i)) for i in d])
for i, pi in enumerate (p):
    if not d[i][pi - 1]:
        print 'NO'
        break
else:
    print 'YES'
