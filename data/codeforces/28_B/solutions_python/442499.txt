n=input()
r=lambda:map(int,raw_input().split())
p=r()
d=r()
rn=range(n)
for i in rn: p[i]-=1
adj=[[] for i in rn]
u=[]

def dfs(v):
    global u
    if u[v]: return
    u[v]=1
    for to in adj[v]:
        dfs(to)

for i in rn:
    for j in [-1,1]:
        t=i+j*d[i]
        if 0<=t<n:
            adj[i].append(t)
            adj[t].append(i)
a=1
for i in rn:
    u=[0]*n
    dfs(i)
    if not u[p[i]]:
        a=0
print ['NO','YES'][a]
