n = input()
p = map(int,raw_input().split())
d = map(int,raw_input().split())
l = range(n)
def f(x):
	if l[x]!=x:l[x]=f(l[x])
	return l[x]
for i in range(n):
	if i>=d[i]:l[f(i)]=f(i-d[i])
	if i+d[i]<n:l[f(i)]=f(i+d[i])

if []==[1 for i in range(n) if f(i)!=f(p[i]-1)]:
	print "YES"
else:
	print "NO"
