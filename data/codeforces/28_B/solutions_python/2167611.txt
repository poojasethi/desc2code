n=input()
p=map(int,raw_input().split())
d=map(int,raw_input().split())
v=[False]*n
yes=True
def dfs(i):
  v[i]=True
  a=[i+1]
  for j in range(n):
    if not v[j] and (d[i]==abs(i-j) or d[j]==abs(i-j)):
      a+=dfs(j)
  return a
for i in range(n):
  if not yes: break
  if not v[i]:
    a=dfs(i)
    b=[p[u-1] for u in a]
    a.sort()
    b.sort()
    if a!=b: yes=False
print 'YES' if yes else 'NO'
