#include <cstdio>

const int MAX = 101;
int N, perm[MAX];
bool used[MAX], edge[MAX][MAX];
int cycle[MAX], C;
int main()
{
	scanf("%d", &N);
	for (int i = 1; i <= N; i++)
		scanf("%d", &perm[i]);
	for (int i = 1, x; i <= N; i++)
	{
		scanf("%d", &x);
		if (i+x >= 1 && i+x <= N)
			edge[i][i+x] = edge[i+x][i] = true;
		if (i-x >= 1 && i-x <= N)
			edge[i][i-x] = edge[i-x][i] = true;
		edge[i][i] = true;
	}
	for (int k = 1; k <= N; k++)
	for (int i = 1; i <= N; i++)
	for (int j = 1; j <= N; j++)
		edge[i][j] |= edge[i][k] & edge[k][j];
	for (int i = 1; i <= N; i++)
		if (!edge[i][perm[i]])
		{
			puts("NO");
			return 0;
		}
	puts("YES");
}
