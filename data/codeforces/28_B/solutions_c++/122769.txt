#include<cstdio>
#define MAX 105

int tab[MAX],rep[MAX],n,v,i;
bool can;

int Find(int a)
{
if(rep[a]==a) return a;
return Find(rep[a]);
}

void Union(int a, int b)
{
int ra,rb;
ra=Find(a);
rb=Find(b);
rep[ra]=rb;
}

int main()
{
scanf("%d",&n);
for(i=1; i<=n; ++i) rep[i]=i;
for(i=1; i<=n; ++i) scanf("%d",&tab[i]);

for(i=1; i<=n; ++i)
   {
   scanf("%d",&v);
   if(i+v<=n) Union(i,i+v);
   if(i-v>0) Union(i,i-v);
   }
   
can=true;
for(i=1; i<=n; ++i)
   if(Find(i)!=Find(tab[i])) can=false;
   
if(can) printf("YES\n"); else printf("NO\n");
return 0;
}
