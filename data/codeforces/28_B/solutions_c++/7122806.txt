#include <iostream>
using namespace std;
int fa[110];
int find(int x){
	if (fa[x]==x) return fa[x]; else return fa[x]=find(fa[x]);
}
int main(){
	int n,i,j,a[110],b[110];
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>a[i];
		fa[i]=i;
	}
	for(i=1;i<=n;i++){
		cin>>b[i];
		if (i+b[i]<=n) fa[find(i)]=find(i+b[i]);
		if (i-b[i]>=1) fa[find(i)]=find(i-b[i]);
	}
	for(i=1;i<=n;i++){
		if (find(i)==i){
			int c[110]={};
			for(j=1;j<=n;j++)
				if (find(j)==i){
					c[j]++;
					c[a[j]]--;
				}
			for(j=1;j<=n;j++)
				if (c[j]){
					break;
				}
			if (j<=n) break;
		}
	}
	if (i<=n) cout<<"NO"; else cout<<"YES";
	return 0;
}
