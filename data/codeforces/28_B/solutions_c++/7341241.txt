#include<cstdio>
#include<algorithm>
using namespace std;
int n,a[1001],b[1001],h[1001];
int H(int x){
  if(h[x] != x) h[x] = H(h[x]);
  return h[x];
}
int main(){
  scanf("%d",&n);
  for(int i=1;i<=n;i++)
    scanf("%d",&a[i]);
  for(int i=1;i<=n;i++)
    scanf("%d",&b[i]), h[i] = i;
  for(int i=1;i<=n;i++){
    h[i] = H(i);
    if(i-b[i] > 0)
      h[h[i]] = H(i-b[i]); 
    if(i+b[i] <= n)
      h[i] = H(i),h[h[i]] = H(i+b[i]); 
  }
  int u = 0;
  for(int i=1;i<=n;i++){
    if(H(i) != H(a[i]))
      u = 1;
  }  
  printf((u == 1 ? "NO" : "YES"));
  return 0;
}
