#include <bits/stdc++.h>
using namespace std;
int a[100+10];
int parent[100+10];
int find(int x) { return ((parent[x]==x)?x:parent[x]=find(parent[x])); }
bool check(int x,int y) {return find(x)==find(y);}
int enjoy(int x,int y) {if(!check(x,y)) parent[find(x)]=find(y);}
int main()
{
	for(int x=1;x<100+10;++x) parent[x]=x;
	int n;
	cin >> n;
	for(int x=1;x<=n;++x)
	cin >> a[x];
	int k;
	for(int x=1;x<=n;++x)
	{
	cin >> k; if(x-k>0&&x-k<=n) { enjoy(x,x-k); }
	          if(x+k>0&&x+k<=n) { enjoy(x,x+k); }
	}
	for(int x=1;x<=n;++x)
	if(check(x,a[x])) continue;
	else { cout << "NO" << endl; return 0;}
	cout << "YES" << endl;
	
}
