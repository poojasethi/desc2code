

#include <iostream>
#include <cstdio>

using namespace std;

int p[101],a[101];

int FIND(int i){return p[i]==i?i:p[i]=FIND(p[i]);}
void UNION(int i,int j){p[FIND(i)]=FIND(j);}

int main()
{
    int n,i,x;
    cin >> n;
    for (i=1;i<=n;i++) scanf("%d",&a[i]);
    for (i=1;i<=n;i++) p[i]=i;
    for (i=1;i<=n;i++)
    {
        scanf("%d",&x);
        if (i-x>0){
            UNION(i,i-x);
}
        if (i+x<=n){
            UNION(i,i+x);
        }
    }
    bool check=true;
    for (i=1;i<=n&&check;i++)
        check=(FIND(i)==FIND(a[i]));
    puts(check?"YES":"NO");
    return 0;
}
