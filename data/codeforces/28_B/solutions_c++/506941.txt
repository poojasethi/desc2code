#include<iostream>
#include<cstdio>
#define Max 1005

int n;
int djs[Max];
int arr[Max];

int find(int now)
{
	return djs[now] == now ? now : djs[now] = find(djs[now]);
}

void set(int a, int b)
{
	if(a < 0 || a >= n)
		return ;

	a = find(a), b = find(b);

	if(a == b)
		return ;

	djs[a] = b;
}

int main()
{
	scanf("%d", &n);

	for(int i=0;i<n;i++)
	{
		scanf("%d", &arr[i]);
		djs[i] = i;
	}

	for(int i=0;i<n;i++)
	{
		int t;
		scanf("%d", &t);

		set(i+t, i);
		set(i-t, i);
	}
	int ans = 1;
	for(int i=0;i<n;i++)
		if(find(i) != find(arr[i]-1))
			ans = 0;

	printf("%s\n", ans ? "YES" : "NO");
}
