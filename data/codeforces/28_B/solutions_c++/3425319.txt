#include<iostream>
#include<set>
using namespace std;
int n,need[105],fav[105],group[105],groupid;
void dfs(int x){
	if(group[x]) return;
	group[x]=groupid;
	for(int i=1;i<=n;i++)
	if(i+fav[i]==x || i-fav[i]==x || x+fav[x]==i || x-fav[x]==i) dfs(i);
}
int main(){
	cin>>n;
	for(int i=1;i<=n;i++) cin>>need[i];
	for(int i=1;i<=n;i++) cin>>fav[i];
	int i;
	for(i=1;i<=n;i++)
	if(!group[i]){
		groupid++;
		dfs(i);
		set<int> s1,s2;
		for(int j=1;j<=n;j++)
		if(group[j]==groupid){
			s1.insert(j);
			s2.insert(need[j]);
		}
		if(s1!=s2) break;
	}
	if(i>n) cout<<"YES";
	else cout<<"NO";
	return 0;
}
