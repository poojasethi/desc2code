#include <bits/stdc++.h>
using namespace std;

int n,a[105],d,par[105];
bool no;

int find(int x){
	if (x==par[x]) return x;
	par[x]=find(par[x]); return par[x];
}

void merge(int x, int y){
	x=find(x); y=find(y); par[x]=y;
}

int main(){
	ios::sync_with_stdio(0); cin.tie(0);
	cin >> n;
	for (int i=1;i<=n;i++){
		cin >> a[i]; par[i]=i;
	}
	for (int i=1;i<=n;i++){
		cin >> d;
		if (i-d>=1) merge(i,i-d);
		if (i+d<=n) merge(i,i+d);
	}
	for (int i=1;i<=n;i++){
		if (find(i)!=find(a[i])){
			no=true; break;
		}
	}
	cout << (no ? "NO" : "YES") << "\n";
}
