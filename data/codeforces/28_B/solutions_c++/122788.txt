#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int A[101][101],i,j,k,n,B[101];
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>B[i];
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
			A[i][j]=0;
		A[i][i]=1;	
	}		
	for(i=1;i<=n;i++)
	{
		cin>>k;
		if(i+k<=n)
			A[i][i+k]=A[i+k][i]=1;
		if(i-k>=0)
			A[i][i-k]=A[i-k][i]=1;			
	}
	for(k=1;k<=n;k++)
		for(i=1;i<=n;i++)
			for(j=1;j<=n;j++)
				A[i][j]=A[i][j]|(A[i][k]&A[k][j]);
	int flag=true;
	for(i=1;i<=n;i++)
		if(!A[i][B[i]])
		{
			flag=false;
			break;
		}
	if(flag)
		cout<<"YES";
	else
		cout<<"NO";									
	return 0;
}
