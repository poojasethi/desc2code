#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

const int N=106;

bool g[N][N];
int n, p[N];

int main()
{
	scanf("%d", &n);
	for(int i=1; i<=n; i++) scanf("%d", p+i);
	memset(g, false, sizeof g);
	for(int i=1, d; i<=n; i++)
	{
		scanf("%d", &d);
		if(i-d>=1) g[i][i-d]=g[i-d][i]=true;
		if(i+d<=n) g[i][i+d]=g[i+d][i]=true;
		g[i][i]=true;
	}
	for(int k=1; k<=n; k++)
		for(int i=1; i<=n; i++)
			for(int j=1; j<=n; j++)
				g[i][j]=g[i][j] || g[i][k] && g[k][j];
	for(int i=1; i<=n; i++)
		if(!g[i][p[i]])
		{
			printf("NO\n");
			return 0;
		}
	printf("YES\n");
	return 0;
}
