#include <iostream>
using namespace std;
int n,a[105],uf[105];
void ufinit(){for(int i=0;i<n;i++)uf[i]=i;}
int uffind(int x){return(uf[x]==x?x:uf[x]=uffind(uf[x]));}
void ufuni(int a,int b){uf[uffind(a)]=uffind(b);}

int main(){
	int fn;
	bool f=true;
	cin>>n;ufinit();
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n;i++){
		cin>>fn;
		if(i-fn>=0)ufuni(i,i-fn);
		if(i+fn<n)ufuni(i,i+fn);
	}
	for(int i=0;i<n;i++){
		if(uffind(i)!=uffind(a[i]-1)){
			f=false;break;
		}
	}
	if(f)cout<<"YES";else cout<<"NO";
	return 0;
}
