#include<bits/stdc++.h>
using namespace std;

int ot[120],P[120];

int cari(int a){
	if(ot[a]!=a)return cari(ot[a]);
	else return a;
}

void gabung(int a,int b){
	ot[cari(a)]=cari(b);
}

int main(){
	int n,i,x;
	bool bisa=true;
	cin>>n;
	for(i=1;i<=n;i++){
		ot[i]=i;
	}
	for(i=1;i<=n;i++){
		cin>>P[i];
	}
	for(i=1;i<=n;i++){
		cin>>x;
		if(i-x>0)gabung(i,i-x);
		if(i+x<=n)gabung(i,i+x);
	}
	for(i=1;i<=n&&bisa;i++){
		bisa=(cari(i)==cari(P[i]));
	}
	if(bisa)cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
}
