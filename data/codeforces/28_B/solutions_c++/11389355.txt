#include<iostream>
using namespace  std;
int f[101];
int fnd(int a)
{
    if(f[a]==a)return a;
    return f[a]=fnd(f[a]);
}
int unn(int a,int b)
{
    f[fnd(a)]=fnd(b);
}
int main()
{
    int n,v[101],w[101];
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>v[i];f[i]=i;
    }
    for(int i=1;i<=n;i++)
    {
        cin>>w[i];if(i+w[i]<=n)unn(i,i+w[i]);if(i-w[i]>=1)unn(i,i-w[i]);
    }
    bool f=true;
    for(int i=1;i<=n;i++)
    {
        if(fnd(i)!=fnd(v[i]))f=false;
    }
    if(f)cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    return 0;
}
