#include <iostream>
using namespace std;
int a[110],f[110][110];
int main(){
	int n,x;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
		a[i]--;
	}
	for(int i=0;i<n;i++){
		cin>>x;
		f[i][i]=1;
		if (i-x>=0) f[i][i-x]=f[i-x][i]=1;
		if (i+x<n) f[i][i+x]=f[i+x][i]=1;
	}
	for(int k=0;k<n;k++)
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				f[i][j]|=f[i][k]&&f[k][j];
	int ff=1;
	for(int i=0;ff&&i<n;i++)
		ff&=f[i][a[i]];
	if (ff) cout<<"YES"; else cout<<"NO";
}
