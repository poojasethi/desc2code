#include <iostream>
using namespace std;
int par[101], a[101], b[101];
int find(int x){ return (par[x] == x ? x : par[x] = find(par[x])); }
void Union(int x, int y){ par[find(x)] = find(y); }
int main(){
	ios_base::sync_with_stdio(0);
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++){
		cin >> a[i];
		par[i] = i;
	}
	for (int i = 1; i <= n; i++){
		cin >> b[i];
		if (i - b[i] > 0)
			Union(i, i - b[i]);
		if (i + b[i] <= n)
			Union(i, i + b[i]);
	}
	for (int i = 1; i <= n; i++){
		if (find(i) != find(a[i])){
			cout << "NO";
			return 0;
		}
	}
	cout << "YES";
	// system("pause");
}
