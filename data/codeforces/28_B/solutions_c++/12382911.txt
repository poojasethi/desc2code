#include <stdio.h>
int p[101],par[101];
int fnd(int k)
{
	return par[k]==k?k:par[k]=fnd(par[k]);
}
int main()
{
	int n,i,x;
	scanf("%d",&n);
	for (i=1;i<=n;i++)
		scanf("%d",p+i);
	for (i=1;i<=n;i++)
		par[i]=i;
	for (i=1;i<=n;i++)
	{
		scanf("%d",&x);
		if (i-x>=1)
			par[fnd(i-x)]=fnd(i);
		if (i+x<=n)
			par[fnd(i+x)]=fnd(i);
	}
	for (i=1;i<=n;i++)
		if (fnd(i)!=fnd(p[i]))
		{
			printf("NO");
			return 0;
		}
	printf("YES");
	return 0;
}
