#include <iostream>
#include <algorithm>
using namespace std;
const int MAXN=110;
int a[MAXN],f[MAXN],c[MAXN],n;

void dfs(int x,int com)
{
  c[x]=com;
  for(int i=1;i<=n;i++)
    if((abs(x-i)==f[x] || abs(x-i)==f[i]) && c[i]==0)
      dfs(i,com);
}


int main()
{
  cin>>n;
  for(int i=1;i<=n;i++)
    cin>>a[i];
  for(int i=1;i<=n;i++)
    cin>>f[i];
  int com=1;
  for(int i=1;i<=n;i++)
    if(c[i]==0)
      dfs(i,com++);
  for(int i=1;i<=n;i++)
    {
      int x=a[i];
      if(c[x]!=c[i])
	{
	  cout<<"NO"<<endl;
	  return 0;
	}
    }
  cout<<"YES"<<endl;
  return 0;
}
