#include <iostream>
#include <cstdio>
using namespace std;
int kol=0,t=0,n,a[1000],b[1000],p[1000];
int f(int v){
    if(v==p[v]) return v;
    return p[v]=f(p[v]);
}
void un(int a, int b){
    a=f(a); b=f(b);
    p[a]=b;
}
int main(){
    cin>>n;
    for(int i=1;i<=n;i++) {cin>>a[i];p[i]=i;}
    for(int i=1;i<=n;i++) {
        cin>>b[i];
        if(i-b[i]>0) un(i,i-b[i]);
        if(i+b[i]<=n) un(i,i+b[i]);
    }
    for(int i=1;i<=n;i++)
        if(f(i)!=f(a[i])) {cout<<"NO"; return 0;}
    cout<<"YES";
    return 0;
}
