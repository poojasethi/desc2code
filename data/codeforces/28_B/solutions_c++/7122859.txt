#include <iostream>
#include <vector>
using namespace std;
int fa[110],a[110],b[110],u[110],x,e;
vector<int> g[110];
int dfs(int x){
	u[x]=e;
	b[a[x]]=e;
	for(int i=0;i<g[x].size();i++)
		if (u[g[x][i]]==0) dfs(g[x][i]);
}
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int i=1;i<=n;i++){
		cin>>x;
		if (i-x>=1){g[i].push_back(i-x);g[i-x].push_back(i);}
		if (i+x<=n){g[i].push_back(i+x);g[i+x].push_back(i);}
	}
	for(int i=1;i<=n;i++){
		if (!u[i]){e++;dfs(i);}
	}
	for(int i=1;i<=n;i++){
		if (u[i]!=b[i]){
			cout<<"NO";
			return 0;
		}
	}
	cout<<"YES";
	return 0;
}
