using namespace std;
int a[101];
int l(int q){
  if(a[q]!=a[a[q]])
   a[q]=l(a[q]);
  return a[q];
}

  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,s[101];
  scanf("%d",&q);
  for(w=1;w<=q;w++){
   scanf("%d",&s[w]);
   a[w]=w;}
  for(w=1;w<=q;w++){
   scanf("%d",&e);
   if(w>e)
    a[l(w)]=l(w-e);
   if(w+e<=q)
    a[l(w)]=l(w+e);}
  for(w=1;w<=q;w++)
   if(l(w)!=l(s[w])){
    printf("NO");
    return 0;}
  printf("YES");
  return 0;}
