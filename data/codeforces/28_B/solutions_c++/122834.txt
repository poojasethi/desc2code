#include<iostream>
#include<cstdio>

using namespace std;

int perm[100];
int favo[100];

int s[100];
int f(int x) {return s[x]==x?x:f(s[x]);}
void u(int x,int y) { s[f(x)] = f(y); }

int main()
{
	int n;
	scanf("%d",&n);

	for(int i=0;i<n;i++) s[i] = i;
	for(int i=0;i<n;i++) scanf("%d",&perm[i]);
	for(int i=0;i<n;i++) scanf("%d",&favo[i]);

	for(int i=0;i<n;i++)
	{
		int d = i-favo[i];
		if (d>=0&&d<n) u(i,d);
		d = i+favo[i];
		if (d>=0&&d<n) u(i,d);
	}

	for(int i=0;i<n;i++)
		if (f(i)!=f(perm[i]-1)) { printf("NO\n"); return 0;}

	printf("YES\n");

	return 0;
}
