#include <iostream>
using namespace std;
int n,a[105],fv[105],uf[105];

void ufinit(){for(int i=0;i<n;i++)uf[i]=i;}
int uffind(int x){return(uf[x]==x?x:uf[x]=uffind(uf[x]));}
void ufuni(int a,int b){uf[uffind(a)]=uffind(b);}

int main(){
	ios::sync_with_stdio(false);
	bool f=true;
	cin>>n;
	ufinit();
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n;i++){
		cin>>fv[i];
		if(i>=fv[i])ufuni(i,i-fv[i]);
		if(i+fv[i]<n)ufuni(i,i+fv[i]);
	}
	for(int i=0;i<n;i++)if(uffind(i)!=uffind(a[i]-1))f=false;
	if(f)cout<<"YES";else cout<<"NO";
	return 0;
}
