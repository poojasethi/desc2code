#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
using namespace std;

int s[104];
int map[104][104];
int n;

int main()
{
	int i,j,k,temp;
	cin>>n;
	memset(map,0,sizeof(map));
	for(i=1;i<=n;i++)
	{
		cin>>s[i];
		map[i][i]=1;
	}
	for(i=1;i<=n;i++)
	{
		cin>>temp;
		if(i+temp<=n)map[i][i+temp]=1;
		if(i-temp>=1)map[i][i-temp]=1;
	}
	for(k=1;k<=n;k++)
		for(i=1;i<=n;i++)
			for(j=1;j<=n;j++)
			{
				if((map[i][k]&map[k][j])|(map[i][k]&map[j][k]))map[i][j]=1;
			}
	bool ST=false;	
	for(i=1;i<=n;i++)
		if(map[i][s[i]]==0)ST=true;
	if(ST)cout<<"NO"<<endl;
	else cout<<"YES"<<endl;
	return 0;
}
