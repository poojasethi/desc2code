#include <cstdio>
#include <iostream>
int n;
int a[102];
bool ff[102][102];
using namespace std;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",a+i);
	for (int i=1,x;i<=n;i++)
	{
		scanf("%d",&x);
		ff[i][i]=true;
		if (i-x>0) ff[i][i-x]=ff[i-x][i]=true;
		if (x+i<=n) ff[i][x+i]=ff[x+i][i]=true;
	}
	for (int k=1;k<=n;k++)
		for (int i=1;i<=n;i++)
			for (int j=1;j<=n;j++)
				ff[i][j]|=(ff[i][k]&&ff[k][j]);
	for (int i=1;i<=n;i++)
		if (!ff[i][a[i]])
		{
			puts("NO");
			return 0;
		}
	puts("YES");
	getchar();getchar();
	return 0;
}
