#include<cstdio>
#include<cstring>
int p[110];
int find(int x){return p[x]==x?x:p[x]=find(p[x]);}
void unio(int a,int b)
{
	int x=find(a);
	int y=find(b);
	if(x!=y) p[x]=y;
}
int a[110],b[110];
int main()
{
	int i,j;
	int n;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		p[i]=i;
	}
	for(i=1;i<=n;i++)
	{
		scanf("%d",&b[i]);
		if(i-b[i]>0) unio(i,i-b[i]);
		if(i+b[i]<=n) unio(i,i+b[i]);
	}
	for(i=1;i<=n;i++)
	{
		int x=find(i);
		int y=find(a[i]);
		if(x!=y) 
		{
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
}
