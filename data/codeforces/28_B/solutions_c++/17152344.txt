#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int N = 101;
int n, par[N], p[N], d;
int find1(int x){ if(par[x]==-1){return x;} return find1(par[x]); }
bool same(int x, int y){ return find1(x) == find1(y); }
void mrge(int x, int y){ if(same(x,y)){return;} par[find1(x)] = find1(y); }
int main()
{
    int i;
	cin >> n;
	for(i =1; i <= n; i++) { cin>>p[i];par[i]=-1;}
	for(i = 1; i <=n; i++)
    {
		cin>>d;
		if (i + d <= n) mrge(i,i+d);
		if (i - d >= 1) mrge(i,i-d);
	}
	for(i = 1; i <=n; i++){if (!same(i, p[i])){cout << "NO"<<endl;return 0;}}
	cout << "YES"<<endl;
	return 0;
}
