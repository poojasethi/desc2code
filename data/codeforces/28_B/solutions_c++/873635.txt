#include<cstdio>
int n;
int fa[105],a[105],b[105];
int get(int a)
{
    while(fa[a]!=a)a=fa[a];
    return a;
}
void un(int a,int b)
{
    a=get(a);
    b=get(b);
    fa[a]=b;
}
int main()
{
    scanf("%d",&n);
    int i;
    for(i=1;i<=n;i++)fa[i]=i;
    for(i=1;i<=n;i++)
    scanf("%d",a+i);
    for(i=1;i<=n;i++)
    scanf("%d",b+i);
    for(i=1;i<=n;i++)
    {
        if(i-b[i]>0)un(i,i-b[i]);
        if(i+b[i]<=n)un(i,i+b[i]);
    }
    for(i=1;i<=n;i++)
    {
        if(get(i)!=get(a[i]))
        {
            puts("NO");
            return 0;
        }
    }
    puts("YES");
    return 0;
}
