#include <bits/stdc++.h>
using namespace std;
int parent[102];
int find(int x)
{
	if(parent[x]!=x)
	parent[x]=find(parent[x]);
	return parent[x];
}
int main() {
	// your code goes here
int n;
cin>>n;
int val[n+1],fav[n+1];
for(int i=1;i<=n;i++)
{
	cin>>val[i];	
	parent[i]=i;
}
for(int i=1;i<=n;i++)
cin>>fav[i];
for(int i=1;i<=n;i++)
{
	if(i+fav[i]<=n)
 
	parent[find(i+fav[i])]=parent[find(i)];
	if(i-fav[i]>=1)
	parent[find(i-fav[i])]=parent[find(i)];
}
int flag=1;
for(int i=1;i<=n;i++)
{
	if(find(val[i])!=find(i))
	{	flag=0; break;}
}
if(!flag)cout<<"NO";
else cout<<"YES";
	return 0;
}
