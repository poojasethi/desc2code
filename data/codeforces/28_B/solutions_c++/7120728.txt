#include <iostream>
#include <vector>
#include <string.h>
using namespace std;
vector<int> g[102];
bool use[102];

int a[102],n;
void dfs(int x){
	 use[x]=true;
	 for(int i=0;i<g[x].size();i++)if(!use[g[x][i]])dfs(g[x][i]);
}
int main(){
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i];
	for(int i=1;i<=n;i++){
			int x;
			cin>>x;
			int t1=x+i;
			int t2=i-x;
			if(t1<=n){g[t1].push_back(i);g[i].push_back(t1);}
			if(t2>0){g[t2].push_back(i);g[i].push_back(t2);}
	}
	for(int i=1;i<=n;i++){
			memset(use,0,sizeof(use));
			dfs(a[i]);
			if(!use[i]){cout<<"NO";return 0;}
	}
	cout<<"YES";
	return 0;
}
