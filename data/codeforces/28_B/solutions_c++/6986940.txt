#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int s[105],d[105],a[105];
int find(int i)
{
	if(s[i] == i)return s[i];
	return s[i] = find(s[i]);
}

int main()
{
	int n,i;
	scanf("%d",&n);
	for(i = 1; i <= n; i++)
	{
		scanf("%d",&a[i]);
		s[i] = i;
	}
	for(i = 1; i <= n; i++)
	{
		scanf("%d",&d[i]);
		int x = i-d[i];
		if(x>=1 && x<=n && find(x)!=find(i))s[find(x)]=find(i);
		x = i+d[i];
		if(x>=1 && x<=n && find(x)!=find(i))s[find(x)]=find(i);
	}
	int flag = 1;
	for(i = 1; i <= n; i++)
		if(find(a[i])!=find(i))flag=0;
	if(flag)printf("YES");
	else printf("NO");
	return 0;
}
