#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cmath>
#define ll long long
const ll mo=1e9+7;
using namespace std;
struct st{ll x,y;} a[2010];
ll n,m,k,i,j,jc[200001],inv[200001],f[2010];
bool cmp(st a,st b){return a.x<b.x || (a.x==b.x && a.y<b.y);}
ll C(ll x,ll y){return (((jc[x+y-2]*inv[x-1])%mo)*inv[y-1])%mo;}
ll ksm(ll x,ll y)
{
	if (y==1) return x;
	ll t=ksm(x,y>>1);
	t=(t*t)%mo;
	if (y&1) t=(t*x)%mo;
	return t;	
}

int main()
{
		scanf("%I64d%I64d%I64d\n",&n,&m,&k);
		for (i=1;i<=k;i++) scanf("%I64d%I64d\n",&a[i].x,&a[i].y);
		a[++k].x=n;
		a[k].y=m;
		sort(a+1,a+k+1,cmp);
		for (i=1,jc[0]=inv[0]=1;i<=n+m;i++) jc[i]=(jc[i-1]*i)%mo;
		inv[n+m]=ksm(jc[n+m],mo-2);
		for (i=n+m-1;i;i--) inv[i]=(inv[i+1]*(i+1))%mo;
		for (i=1;i<=k;i++)
		{
			f[i]=C(a[i].x,a[i].y);
			for (j=1;j<i;j++)
				if (a[j].x<=a[i].x && a[j].y<=a[i].y) f[i]=(f[i]-(f[j]*C(a[i].x-a[j].x+1,a[i].y-a[j].y+1)%mo)+mo)%mo;
		}
		printf("%I64d",f[k]);
}
