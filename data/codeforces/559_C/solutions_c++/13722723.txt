#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int M=1000000007;
int R,C,k,dp[2001],f[200001]={1},inv[200001];
pair<int,int > v[2001];
int Pow(int a,int k)
{
  if(!k)
    return 1;
  ll z=Pow(a,k/2);
  if(k&1)
    return z*z%M*a%M;
  return z*z%M;  
}
int ways(int w,int h)
{
  return (ll)f[w+h-2]*inv[w-1]%M*inv[h-1]%M;
}
int main()
{
  scanf("%d%d%d",&R,&C,&k);
  inv[0]=Pow(1,M-2);
  for(int i=1;i<=R+C;i++)
  {
    f[i]=i*(ll)f[i-1]%M;
    inv[i]=Pow(f[i],M-2);
  }
  for(int i=0;i<k;i++)
    scanf("%d%d",&v[i].first,&v[i].second);
  v[k].first=R;
  v[k++].second=C;
  sort(v,v+k);
  for(int i=0;i<k;i++)
  {
    dp[i]=ways(v[i].first,v[i].second);
    for(int j=0;j<i;j++)
    {
      if(v[j].second<=v[i].second)
      {
        dp[i]-=(ll)ways(v[i].first-v[j].first+1,v[i].second-v[j].second+1)*dp[j]%M;
        dp[i]=((ll)dp[i]+M)%M;
      }
    }
  }
  printf("%d\n",dp[k-1]);
  return 0;
}
