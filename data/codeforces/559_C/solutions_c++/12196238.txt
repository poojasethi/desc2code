#include<bits/stdc++.h>
#define x first
#define y second 
#define mo 1000000007
#define ll long long 
using namespace std;
pair<ll,ll> a[3000];
ll h,w,r[3000],fac[200010],rev[200010];
int n;

ll mult(ll a,ll b){
    if (b==1) return a;
    ll tmp=mult(a,b/2);
    tmp=tmp*tmp%mo;
    if (b%2) tmp=tmp*a%mo;
    return tmp;
}

ll c(ll n,ll m){
    ll ans=((fac[n]*rev[m])%mo*rev[n-m])%mo;
    return ans;
}
 
int main(){
    scanf("%I64d%I64d%d",&h,&w,&n);
    for (int i=1; i<=n; i++) scanf("%I64d%I64d",&a[i].x,&a[i].y);
    fac[0]=1; rev[0]=1;
    for (int i=1; i<=200000; i++){fac[i]=(fac[i-1]*i)%mo; rev[i]=mult(fac[i],mo-2)%mo;}
    sort(a+1,a+n+1);
    a[n+1].x=h;
    a[n+1].y=w;
    for (int i=1; i<=n+1; i++){
        r[i]=c(a[i].x+a[i].y-2,a[i].x-1);
        for (int j=1; j<i; j++) if (a[j].y<=a[i].y) r[i]=(r[i]+mo-(r[j]*c(a[i].x+a[i].y-a[j].x-a[j].y,a[i].x-a[j].x))%mo)%mo;
    }
    printf("%I64d",r[n+1]);
}
