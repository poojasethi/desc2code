#include<cstdio>
#include<algorithm>
#define sumN 2010
#define mod 1000000007
#define ll long long
using namespace std;

int N,M,n;
ll f[sumN],fac[200010],inv[200010];

ll ksm(ll a,ll b){
	ll ret=1;
	while(b){
		if(b&1) ret=ret*a%mod;
		a=a*a%mod;
		b>>=1;
	}
	return ret;
}

void getfac(){
	fac[0]=1;
	for(int i=1;i<=2e5;i++) fac[i]=fac[i-1]*i%mod;
	for(int i=0;i<=2e5;i++) inv[i]=ksm(fac[i],mod-2);
}

ll C(int n,int m){
	return fac[n]*(inv[m]*inv[n-m]%mod)%mod;
}

struct Point{
	int x,y;
}a[sumN];

bool cmp(Point a,Point b){
	return a.y==b.y?a.x<b.x:a.y<b.y;
}

int main(){
	scanf("%d%d%d",&N,&M,&n);
	for(int i=1;i<=n;i++) scanf("%d%d",&a[i].x,&a[i].y);
	a[++n].x=N;a[n].y=M;
	sort(a+1,a+1+n,cmp);
	getfac();
	for(int i=1;i<=n;i++){
		f[i]=C(a[i].x+a[i].y-2,a[i].x-1);
		for(int j=1;j<i;j++){
			if(a[j].x<=a[i].x&&a[j].y<=a[i].y){
				f[i]=(f[i]-f[j]*C(a[i].x+a[i].y-a[j].x-a[j].y,a[i].x-a[j].x)%mod+mod)%mod;
			}
		}
	}
	printf("%I64d\n",f[n]);
	return 0;
}
