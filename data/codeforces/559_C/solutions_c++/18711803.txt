#include<bits/stdc++.h>
using namespace std;
#define LL long long 
#define pii pair<int, int>
#define X first
#define Y second
const int MAXN = 2e5 + 5;
const int MO = 1e9 + 7;

int H, W, N;
LL Jc[MAXN], Ni[MAXN], dp[MAXN];
pii a[MAXN];

inline LL C(int x, int y) {
	return Jc[x] * Ni[y] % MO * Ni[x - y] % MO;
}

inline LL qpow(LL x, int n) {
	LL ans = 1;
	while (n) {
		if (n & 1) (ans *= x) %= MO;
		(x *= x) %= MO;
		n >>= 1;
	}
	return ans;
}

int main() {

	cin >> H >> W >> N;
	for (int i = 1; i <= N; ++ i) {
		cin >> a[i].X >> a[i].Y;
	}

	a[++ N] = make_pair(H, W);
	sort(a + 1, a + N + 1);

	Jc[0] = 1; Ni[0] = 1;
	for (int i = 1; i <= 2e5; ++ i) {
		Jc[i] = (Jc[i - 1] * i) % MO;
		Ni[i] = qpow(Jc[i], MO - 2);
	}

	for (int i = 1; i <= N; ++ i) {
		dp[i] = C(a[i].X + a[i].Y - 2, a[i].X - 1);
		for (int j = 1; j < i; ++ j) {
			if (a[j].Y <= a[i].Y)
				(dp[i] -= dp[j] * C(a[i].X + a[i].Y - a[j].X - a[j].Y, a[i].X - a[j].X) % MO) %= MO;
		}
	}

	cout << (dp[N] + MO) % MO << endl;

	return 0;
}
