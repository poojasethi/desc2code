#include <cstdio>
#include <algorithm>
using namespace std;

const int mod = 1000000007;
int h,w,n,r,c;
pair<int,int> b[2009];
int f[2009];
long long inv[200009],fac[200009],gac[200009];

long long con(int n,int m)
{
	return fac[n]*gac[m]%mod*gac[n-m]%mod;
}
int main()
{
	scanf("%d%d%d",&h,&w,&n);
	inv[0]=inv[1]=1;
	fac[0]=fac[1]=1;
	gac[0]=gac[1]=1;
	for (int i=2;i<=h+w;++i)
	{
		inv[i]=(mod-mod/i)*inv[mod%i]%mod;
		fac[i]=fac[i-1]*i%mod;
		gac[i]=gac[i-1]*inv[i]%mod;
	}
	for (int i=1;i<=n;++i)
	{
		scanf("%d%d",&r,&c);
		b[i].first=r;
		b[i].second=c;
	}
	b[0].first=1;
	b[0].second=1;
	b[n+1].first=h;
	b[n+1].second=w;
	sort(b,b+n+2);
	f[0]=1;
	for (int i=1;i<=n+1;++i)
	{
		f[i]=con(b[i].first-1+b[i].second-1,b[i].first-1);
		for (int j=1;j<i;++j) if (b[j].first<=b[i].first && b[j].second<=b[i].second)
		{
			f[i]-=f[j]*con(b[i].first-b[j].first+b[i].second-b[j].second,b[i].first-b[j].first)%mod;
			if (f[i]<0) f[i]+=mod;
		}
	}
	printf("%d\n",f[n+1]);
	return 0;
}
