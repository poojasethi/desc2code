#include <algorithm>
#include <cstdio>
const int mo=1e9+7;
int f[2010],fac[200010],ine[200010],i,j,h,w,n;
struct P{int x,y;}a[2010];
bool operator <(P a,P b){return a.x<b.x || (a.x==b.x && a.y<b.y);}
int qsm(int x,int y)
{
	int i=1;
	for (;y;y>>=1,x=1ll*x*x%mo) i=y&1?1ll*i*x%mo:i;
	return i;
}
int C(int n,int m)
{
	return 1ll*fac[n]*ine[m]%mo*ine[n-m]%mo;
}
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d%d%d\n",&h,&w,&n);
	for (i=1;i<=n;++i) scanf("%d%d\n",&a[i].x,&a[i].y);
	std::sort(a+1,a+n+1);
	fac[0]=ine[0]=1;
	for (i=1;i<=h+w;++i) fac[i]=1ll*fac[i-1]*i%mo;
	ine[h+w]=qsm(fac[h+w],mo-2);
	for (i=h+w-1;i;--i) ine[i]=1ll*ine[i+1]*(i+1)%mo;
	a[++n].x=h;a[n].y=w;
	for (i=1;i<=n;++i)
	{
		f[i]=C(a[i].x+a[i].y-2,a[i].x-1);
		for (j=1;j<i;++j)
			if (a[j].x<=a[i].x && a[j].y<=a[i].y)
			f[i]=(f[i]-1ll*f[j]*C(a[i].x-a[j].x+a[i].y-a[j].y,a[i].x-a[j].x)%mo+mo)%mo;
	}
	printf("%d\n",f[n]);
}
