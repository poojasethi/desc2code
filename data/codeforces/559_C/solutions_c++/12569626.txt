#include<bits/stdc++.h>
using namespace std;
const int MAbg=2e5+1000,MOD=1e9+7;
long long n,h,w,fac[MAbg],dp[MAbg],inv[MAbg];
pair<long long,long long> a[MAbg];
long long pw(long long a,long long b=MOD-2){
	b=b%(MOD-1);
	return b?(pw((a * a) % MOD,b >> 1) * (b & 1?a:1)) % MOD:1;
}
long long c(long long n,long long r){
	return (n<r || n<0 || r<0)?0:fac[n]*inv[r]%MOD*inv[n-r]%MOD;
}
long long dis(long long bg,long long y){
	return (bg<0 || y<0)?0:c(bg+y,bg);
}
int main(){
	fac[0]=1,inv[0]=1;
	for (int i=1;i<MAbg-10;i++) fac[i]=(fac[i-1]*i)%MOD,inv[i]=(inv[i-1]*pw(i))%MOD;
	cin>>h>>w>>n;
	for (int i=0;i<n;i++) cin>>a[i].first>>a[i].second;
	a[n++]={h,w};
	sort(a,a+n);
	for (int i=0;i<n;i++){
		long long ans=0;
		for(int j=0;j<i;j++) ans+=(dp[j]*dis(a[i].first-a[j].first,a[i].second-a[j].second))%MOD,ans%=MOD;
		dp[i]=((c(a[i].second+a[i].first-2,a[i].first-1)-ans)%MOD+MOD)%MOD;			
	}
	cout<<dp[n-1];
}
