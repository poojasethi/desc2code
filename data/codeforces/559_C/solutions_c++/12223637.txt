#define N (1<<18)
#define M 1000000007
#define ll long long
#define PII pair<int,int>
#include <bits/stdc++.h>

using namespace std;

PII p[N];
int h,w,n;
ll dp[N],f[N]={1},invf[N];

ll quickpow(ll x,ll y)
{
	ll res=1;
	for(;y;y>>=1,x*=x,x%=M)
		if(y&1) res*=x,res%=M;
	return res;
}

ll inv(ll x)
{
	return quickpow(x,M-2);
}

void init()
{
	for(int i=1;i<N;i++)
		f[i]=f[i-1]*i%M;
	for(int i=0;i<N;i++)
		invf[i]=inv(f[i]);
}

ll C(ll x,ll y)
{
	return f[x]*invf[x-y]%M*invf[y]%M;
}

int main()
{
	init();
	cin>>h>>w>>n;
	for(int i=1;i<=n;i++)
		scanf("%d%d",&p[i].first,&p[i].second);
	p[n+1]=PII(h,w),sort(p+1,p+n+2);

	for(int i=1;i<=n+1;i++)
	{
		dp[i]=C(p[i].first+p[i].second-2,p[i].first-1);
		for(int j=1;j<i;j++)
			if(p[j].first<=p[i].first && p[j].second<=p[i].second)
				dp[i]-=dp[j]*C(p[i].first-p[j].first+p[i].second-p[j].second,p[i].first-p[j].first)%M,dp[i]+=M,dp[i]%=M;
	}
	cout<<dp[n+1]<<endl;
}
