const int mod=1e9+7;
ll inv[200001],fac[200001],invfac[200001];
inline ll C(int m,int n)
{ return fac[n]*invfac[m]%mod*invfac[n-m]%mod; }
std::pair<int,int>p[2001];ll dp[2001],sub;
int main()
{
	int x,y,n,i,j;scanf("%d%d",&x,&y);
	n=x+y-2;
	inv[0]=fac[0]=invfac[0]=
	inv[1]=fac[1]=invfac[1]=1;
	for(i=2;i<=n;i++)
		fac[i]=fac[i-1]*i%mod,
		inv[i]=-mod/i*inv[mod%i]%mod+mod,
		invfac[i]=invfac[i-1]*inv[i]%mod;
	scanf("%d",&n);
	for(i=0;i<n;i++)scanf("%d%d",&p[i].X,&p[i].Y);
	std::sort(p,p+n);
	p[n]=std::make_pair(x,y);
	for(i=0;i<=n;i++)
	{
		x=p[i].X,y=p[i].Y;
		dp[i]=C(x-1,x+y-2);sub=0;
		for(j=0;j<i;j++)
		if(p[j].Y<=y)
			(sub+=dp[j]*C(x-p[j].X,x-p[j].X+y-p[j].Y)
				)%=mod;
		dp[i]=(dp[i]-sub+mod)%mod;
	}
	printf("%I64d",dp[n]);
	return 0;
}
