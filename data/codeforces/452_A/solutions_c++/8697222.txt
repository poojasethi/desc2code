#include <stdio.h>
int main(){
	int i,j,f,n,l[8]={8,7,7,6,7,7,7,7};
	char t[20],s[8][20]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	scanf("%d%s",&n,t);
	for(i=0;i<8;i++){
		if(l[i]!=n) continue;
		for(j=f=0;j<n;j++){
			if(t[j]=='.') continue;
			if(t[j]!=s[i][j]){
				f=1;
				break;
			}
		}
		if(f==0) break;
	}
	printf("%s\n",s[i]);
	return 0;
}
