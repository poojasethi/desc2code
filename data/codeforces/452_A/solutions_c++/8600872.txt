#include<iostream>
using namespace std;
int main(){
int n;
char s[10];
char name[8][10]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
cin >> n;
cin >> s;
if(n==8)cout << "vaporeon" << endl;
else if(n==6)cout << "espeon" << endl;
else{
	for(int i=0;i<8;i++){
		for(int j=0;j<n;j++){
			if(s[j]!='.'&&s[j]!=name[i][j])break;
			if(j==n-1)cout << name[i] << endl;
	}
}
}
}
