#include <iostream>
using namespace std;
int main()
{
    string a[]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
    string b;
    int i, j, n;
    cin>>n>>b;
    for (i=0; i<8; i++) if (a[i].size()==n){
        for (j=0; j<n; j++) if (b[j]!='.' && b[j]!=a[i][j]) break;
        if (j==n) {
            cout<<a[i]<<endl; break;
        } 
    }
    return 0;
}
