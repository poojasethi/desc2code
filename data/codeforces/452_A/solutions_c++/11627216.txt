#include <iostream>
using namespace std;
const string s[]= {"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};

int main() {
    int n; cin>>n;
    string str; cin>>str;
    for (int i=0; i<8; i++)
        if (s[i].size()==n) {
            int j=0;
            while ((j<n) and ((str[j]==s[i][j]) or (str[j]=='.'))) {j+=1;}
            if (j==n) cout<<s[i];
        }

    return 0;
}
