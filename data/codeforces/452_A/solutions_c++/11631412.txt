#include <iostream>
using namespace std;
string s,olgo[8]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
int main(){
  cin>>s>>s;
  for(int i=0;i<8;i++){
    bool q=1;
    if(s.length()==olgo[i].length()){
      for(int j=0;j<s.length();j++)
	if(s[j]!='.' && s[j]!=olgo[i][j])
	  q=0;
      if(q)
	cout<<olgo[i]<<endl;
    }
  }
  return 0;
}
