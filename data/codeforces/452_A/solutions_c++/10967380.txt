#include <bits/stdc++.h>
using namespace std;

string a[8] = {"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"} , s;
int n;

int main(){
	cin >> n >> s;
	for (int i = 0 ; i < 8 ; ++ i){
		if (n == a[i].length()){
			for (int j = 0 ; j < n ; ++ j){
				if (a[i][j] != s[j] && s[j] != '.') break;
				if (j == n-1){
					cout << a[i];
					return 0;
				}
			}
		}
	}
}
