#include<bits/stdc++.h>
using namespace std;
int main()
{
int n;
int i,j,k,f=0,ans=0,max=0;
cin>>n;
string b;
cin>>b;
string a[8]={"vaporeon","jolteon","flareon","espeon","umbreon","leafeon","glaceon","sylveon"};
if(b.size()==6)
{
cout<<"espeon";
return 0;
}
for(i=0;i<8;i++)
{
f=0;
if(a[i].size()==b.size())
{
for(k=0,j=0;j<b.size();k++,j++)
{
if(a[i][k]==b[j])
f++;
}
}
if(f>max)
{
max=f;
ans=i;
}
}
cout<<a[ans];

return 0;
}
