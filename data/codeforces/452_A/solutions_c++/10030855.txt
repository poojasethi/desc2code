#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
int main()
{
	lld i,j,k,n;
	string a,s[]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	cin>>n>>a;
	for(i=0;i<8;i++)
	{
		if(s[i].size()==n)
		{
			for(j=0;j<n;j++)
			{
				if(a[j]!='.' && s[i][j]!=a[j])
					break;
			}
			if(j==n)
			{
				cout<<s[i]<<endl;
				break;
			}
		}
	}
	return 0;
}
