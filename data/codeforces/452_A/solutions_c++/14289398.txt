#include <iostream>
using namespace std;

int main()
{
	string s[8] = {"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	string a;
	int n;

	cin>> n >> a;

	for(int i = 0; i < 8; i++)
	{
		if(s[i].size() == n)
		{
			bool chck = true;
			for(int j = 0; j < n; j++)
			{
				if((a[j] != '.') && (a[j] != s[i][j]))
					chck = false;
			}
			if(chck)
			{
				cout<< s[i] << endl;
			}
		}
	}
}
