#include <iostream>
using namespace std;
int main() {
	string str[]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon" , "sylveon"},s;
	int n;
	cin>>n>>s;
	for(int i=0;i<8;i++) {
		if(str[i].size()==n) {
			int j;
			for(j=0;j<s.size();j++) {
				if(s[j]!='.') if(str[i][j]!=s[j]) break;
			}
			if(j==s.size()) {
				cout<<str[i]; return 0;
			}
		}
	}
	
	
	
	
	return 0;
}
