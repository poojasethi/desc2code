#include <iostream>
using namespace std;

int main() {
	string a, s[] = {"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	int i, j, n;
	cin>>n;
	cin>>a;
	for(i = 0; i < 8; i++){
		if(s[i].length() == n){
			for(j = 0; j < n; j++){
				if(a[j] != '.'){
					if(a[j] != s[i][j]){
						break;
					}
				}
			}
			if(j == n){
				cout<<s[i]<<endl;
			}
		}
	}
	return 0;
}
