#include<iostream>
using namespace std;
const string s[]={"vaporeon","jolteon","flareon","espeon", "umbreon","leafeon","glaceon","sylveon"};
int main()
{
	int n;
	cin>>n;
	string a;
	cin>>a;
	for(int i=0;i<8;i++)
	{
		bool w=false;
		if(s[i].size()==n)
		{
			for(int j=0;j<n;j++)
			{
				if(a[j]!='.'&&a[j]!=s[i][j])w=true;
			}
			if(!w)cout<<s[i]<<endl;
		}
	}
	return 0;
}
