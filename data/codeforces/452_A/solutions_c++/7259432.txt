#include <stdio.h>
#include <string.h>

char s[8][9] = 
{
	"vaporeon",
	"jolteon",
	"flareon", 
	"espeon", 
	"umbreon",
	"leafeon",
	"glaceon",
	"sylveon"
};

int main(){
	int n;
	char str[10];
	scanf("%d%s", &n, str);
	for(int i=0; i<8; ++i){
		if(strlen(s[i]) != n) continue;
		int j;
		for(j=0; j<n; ++j){
			if(str[j] != '.' && str[j] != s[i][j]) break;
		}
		if(j==n){
			printf("%s\n", s[i]);
			break;
		}
	}

	return 0;
}
