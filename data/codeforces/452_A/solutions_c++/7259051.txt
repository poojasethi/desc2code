#include <iostream>

using namespace std;

string s[8] = {"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};

int main() {
	int n;
	string t;
	cin >> n >> t;
	for (int i = 0; i < 8; i++) {
		if (t.size() != s[i].size())
			continue;
		bool good = true;
		for (int j = 0; j < t.size(); j++)
			good &= (s[i][j] == t[j] || t[j] == '.');
		if (good) {
			cout << s[i] << endl;
			return 0;
		}
	}
}
