#include<iostream>
#include<string>
using namespace std;

string s[8]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
string a;
int n;

int main()
{
	cin>>n>>a;
	for(int i=0;i<8;i++)
	{
		bool ans=true;
		if(s[i].size()!=n)
			continue;
		for(int j=0;j<n;j++)
			if(a[j]!='.')
				if(a[j]!=s[i][j])
					ans=false;
		if(ans)
		{
			cout<<s[i]<<endl;
			return 0;
		}
	}
}
