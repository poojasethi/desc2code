#include<stdio.h>
char s[10][20]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
char ss[20];
int main()
{
    int n;
    scanf("%d",&n);
    scanf("%s",ss);
    for(int i=0;i<8;i++)
    {
     bool f=false;
     for(int j=0;j<n;j++)
     if(ss[j]!='.'&&ss[j]!=s[i][j]) {f=true;break;} 
     if(!f&&s[i][n]==0){ printf("%s\n",s[i]);break;} 
    }
}
