#include <bits/stdc++.h>
using namespace std;

int main(){
	int n;
	string s, c[8]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	cin >> n >> s;

	bool can;
	int j;
	for (int i=0;i<8;i++){
		can=true;
		for (j=0;j<int(c[i].size())&&j<int(s.size());j++)
			if (c[i][j]!=s[j]&&s[j]!='.') can=false;
		if (can&&c[i].size()==s.size()){cout << c[i] << endl; break;}
	}

	return 0;
}
