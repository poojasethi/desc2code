#include<iostream>
#include<string.h>
using namespace std;
int main ()
{
	string ans[8]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	int n;
	string s;
	cin>>n>>s;
	for(int i=0;i<8;i++)
	{
		string p=ans[i];
		if(n==p.length())
		{
			bool yes=true;
			for(int j=0;j<n;j++)
			{
				yes*=(p[j]==s[j]||s[j]=='.');
			}
			if(yes)
			{
				cout<<p;
				return 0;
			}
		}
	}
}
