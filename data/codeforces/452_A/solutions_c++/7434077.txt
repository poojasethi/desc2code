#include<stdio.h>
#include<string.h>

int main(){
	int n;
	char a[10];
	char *p[]={"vaporeon","jolteon","flareon","espeon","umbreon","leafeon","glaceon","sylveon"};
	scanf("%d",&n);
	scanf("%s",a);
	for(int i=0;i<8;i++){
		bool flag=false;
		for(int j=0;j<n;j++){
			if(a[j]!='.'&& a[j]!=p[i][j]){
				flag=true;
				break;
			}
		}
		if(!flag && p[i][n]==0){
				printf("%s\n",p[i]);
				break;
		}
	}
	return 0;
}
