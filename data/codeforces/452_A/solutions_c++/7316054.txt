#include <iostream>
#include <string>

using namespace std;

int main()
{
	string s[8] = {"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	string a;
	bool ok;
	int n;
	cin>>n>>a;
	for(int i = 0;i< 8; i++)
	{
		if(n != s[i].length())
			continue;
		ok = 1;
		for(int j = 0;j < n; j++)
		{
			if(a[j] != '.' && a[j] != s[i][j])
				ok = 0;
		}
		if(ok)
		{
			cout<<s[i];
			return 0;
		}
	}
	return 0;
}
