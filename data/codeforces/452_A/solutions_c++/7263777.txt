#include<stdio.h>
#include<string.h>
char f[10][10]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
int i,j,flag,n;
char st[10000];
main()
{
	scanf("%d",&n);
	scanf("%s",&st);
	for(i=0;i<8;i++)
	{
		flag=0;
		if(strlen(f[i])!=strlen(st)) continue;
		for(j=0;j<n;j++)
			if(st[j]!='.'&&st[j]!=f[i][j])flag=1;
		if(flag==0) break;
	}
	printf("%s\n",f[i]);
}
