#include<iostream>
#include<string>
using namespace std;
string s[8]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"},str;
int n,i,j,x;

int main(){
	cin>>n>>str;
	x = str.size();
	for(i=0; i<8; i++){
		if(x!=s[i].size()) continue;
		for(j=0; j<x; j++){
			if(str[j]!='.' && s[i][j]!=str[j]){
				break;
			}
			if(j==x-1){
				cout<<s[i];
				return 0;
			}
		}
	}
}
