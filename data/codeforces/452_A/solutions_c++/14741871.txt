#include<stdio.h>
#include<string.h>
char dp[9][50]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
char s[50];
int main()
{
	int n;
	scanf("%d",&n);
	scanf("%s",s);
	for(int i=0;i<8;i++)
	{
		int p=0;
		if(strlen(dp[i])==n){
			for(int j=0;j<n;j++)
			if(s[j]!='.'&&s[j]!=dp[i][j]){
				p=1;
				break;
				}
			if(p)
			continue;
		printf("%s",dp[i]);
		}	
	}
}
