#include<bits/stdc++.h>

using namespace std;

string eve[]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};

int main(){
	int l;
	string st;
	scanf("%d",&l);
	cin>>st;
	for(int x=0;x<8;x++){
		if(l==eve[x].size()){
			bool can=true;
			for(int y=0;y<l;y++) if(!(eve[x][y]==st[y] || st[y]=='.')) can=false;
			if(can){
				cout<<eve[x]<<endl;;
				return 0;
			}
		}
	}
}
