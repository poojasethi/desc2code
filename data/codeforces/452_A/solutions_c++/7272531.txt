#include<iostream>
using namespace std;
string s[]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
bool ff(string s, string t)
{
	for (int i=0;i<t.size();i++) 
		if (t[i]!='.'&&t[i]!=s[i]) return 0;
	return 1;
}
int main()
{
	int l;
	string t;
	cin>>l>>t;
	for(int i=0;i<8;i++) 
		if (s[i].size()==l)
			if(ff(s[i],t))
				cout<<s[i]<<endl;
	return 0;
}
