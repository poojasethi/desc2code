#include<cstdio>

const char *pokemon[8]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};

int main(){
	int n;scanf("%d",&n);
	char buf[n+1];
	scanf("%s",buf);
	for(int i=0;i<8;i++){
		int flag=0;
		for(int j=0;j<=n;j++){
			if(flag=(buf[j]!=pokemon[i][j])&&(buf[j]!='.')){
				break;
			}
		}
		if(!flag){
			puts(pokemon[i]);
			return 0;
		}
	}
	return 0;
}
