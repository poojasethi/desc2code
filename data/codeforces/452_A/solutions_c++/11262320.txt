#include<iostream>
#include<string>
using namespace std;
int a[10],n;
string s,t[10]={"vaporeon","jolteon","flareon","espeon","umbreon","leafeon","glaceon","sylveon","",""};
int main()
{
	cin>>n>>s;
	for(int i=0;i<8;i++)
	if(t[i].size()!=s.size())a[i]=1;
	for(int i=0;i<s.size();i++)
	if(s[i]!='.')
	{
		for(int j=0;j<8;j++)
		if(t[j][i]!=s[i])
		a[j]=1;
	}
	for(int i=0;i<8;i++)
	if(a[i]==0)cout<<t[i];
	return 0;
}
