#include <bits/stdc++.h>
using namespace std;
int main(){
	int n;
	string s , a[10] = {"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	cin >> n >> s;
	for(int i = 0 ; i < 8 ; i++){
		if(n == a[i].size()){
			bool f = 0;
			for(int j = 0 ; j < n ; j++)
				if(s[j] != '.' && s[j] != a[i][j])
					f = 1;
			if(!f){
				cout << a[i];
				return 0;
			}
		}
	}
}
