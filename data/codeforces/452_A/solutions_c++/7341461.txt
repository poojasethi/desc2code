#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n,k=0;
	string a,s[]={"vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"};
	cin>>n>>a;
	while(s[k].size()!=a.size())
		k++;
	for(i=0;i<a.size();i++)
	{
		if(a[i]!='.')
		{
			while(s[k][i]!=a[i] || (s[k][i]==a[i] && s[k].size()!=a.size()))
				k++;
		}
	}
	cout<<s[k]<<endl;
	return 0;
}
