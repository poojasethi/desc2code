#coding:utf-8
strs = ["vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"]
length = int(raw_input())
istr = raw_input()

for strsample in strs:
	if len(strsample) == length:
		for i in range(0, length):
			if istr[i] != "." and istr[i] != strsample[i]:
				break
		else:
			print(strsample)
			break
