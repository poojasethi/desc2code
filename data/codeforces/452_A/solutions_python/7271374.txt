from sys import stdin

a = ['vaporeon', 'jolteon', 'flareon', 'espeon', 'umbreon', 'leafeon', 'glaceon', 'sylveon']

n = int(stdin.readline())
a = filter(lambda x: len(x) == n, a)

for c in enumerate(stdin.readline()):
    if len(a) == 1:
        break
    if c[1] != '.':
        a = filter(lambda x: x[c[0]] == c[1], a)

print a[0]
