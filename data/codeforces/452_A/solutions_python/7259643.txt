n = int(raw_input())
s = raw_input()

names = [x for x in ["vaporeon", "jolteon", "flareon", "espeon", "umbreon", "leafeon", "glaceon", "sylveon"] if len(x) == n]

for name in names:
    match = True
    for i in range(n):
        if name[i] != s[i] and s[i] != '.':
            match = False
    if match:
        print name
        break
