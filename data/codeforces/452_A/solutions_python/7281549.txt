n, hint = int(raw_input()), raw_input()
evolutions = filter(lambda x: len(x) == n, ["vaporeon", "jolteon", "flareon", "umbreon", "leafeon", "glaceon", "sylveon", "espeon"])
for evolution in evolutions:
	match = 1
	for j in xrange(n):
		if hint[j] != '.' and hint[j] != evolution[j]: match = 0
	if match: print evolution; break
