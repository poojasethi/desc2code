n = input()
s = raw_input()
if 8 == n: print "vaporeon"
elif 6 == n: print "espeon"
else:
	table = ["jolteon", "flareon", "umbreon", "leafeon", "glaceon", "sylveon"]
	for t in table:
		isMatched = True
		for i in xrange(n):
			if s[i] != '.' and s[i] != t[i]:
				isMatched = False
				break
		if isMatched: 
			print t
			break
