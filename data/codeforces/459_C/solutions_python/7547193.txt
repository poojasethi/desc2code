import sys
sys.setrecursionlimit(10**7)

def generate_subsets(k,d):
	
	if d<=0:
		yield []
	else:
		for i in xrange(k):
			for s in generate_subsets(k,d-1):
				s.append(i)
				yield s





def main():
	n,k,d=map(int,raw_input().strip().split())

	if n>k**d:
		print -1
	else:
		#generate students orders
		students_order=[]
		counter=0
		for order in generate_subsets(k,d):
			counter+=1
			students_order.append(order)			
			if counter>=n:break


		for i in range(d):
			tmp=[]

			for j in range(n):
				tmp.append(str(students_order[j][i]+1))
			print " ".join(tmp)

main()
