n,k,d = map(int,raw_input().split())
if(n> k**d):
    print -1
else:
    ans = []
    for  i in range(n):
        tmp = i;
        cur = []
        for j in range(d):
            cur.append(tmp%k+1)
            tmp/=k
        ans.append(cur)
    for i in range(d):
        for j in ans:
            print j[i],
        print '\n'
