n, k, d = map(int, raw_input().split())

ok = False
m = 1
for i in xrange(d):
    m *= k
    if m >= n:
        ok = True
        break

if ok:
    a = [i for i in xrange(n)]
    for i in xrange(d):
        for j in xrange(n):
            print (a[j] % k) + 1,
            a[j] /= k
        print
else:
    print -1
