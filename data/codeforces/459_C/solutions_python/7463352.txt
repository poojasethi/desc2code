#!/usr/bin/env python


def read():
    return map(int, raw_input().split())


def work((n, k, d)):
    if n > k ** d:
        print -1
        return

    cur = range(n)
        
    for i in range(d):
        toPrint = []
        for j in range(n):
            toPrint.append(cur[j] % k + 1)
            cur[j] /= k

        print " ".join(map(str, toPrint))


if __name__ == "__main__":
    work(read())
