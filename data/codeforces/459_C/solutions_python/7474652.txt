n,k,d = map(float,raw_input().split())

t_n = n

for i in xrange(int(d)):
	t_n = t_n/k
	
if t_n > 1:
	print "-1"
else:
	val = 1
	for i in xrange(int(d)):
		s = ''
		no = 1
		cnt = 0
		while cnt<n:
			for i in xrange(int(val)):
				if cnt>=n: break
				s += str(no)+" "
				cnt += 1
			no += 1
			if no>int(k):
				no -= int(k)
		if val<=n:
			val *= k
		print s
