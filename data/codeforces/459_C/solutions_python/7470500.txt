#! /usr/bin/env python

a=[int(i) for i in raw_input().split()]
n,k,d=a[0],a[1],a[2]

def ok(n,k,d):
    s,p=1,0
    while(s<=1000 and p<d):
        s=s*k
        p=p+1
    return n<=s

def ll(n,k,d):
    s,p=1,0
    while(s<=1000 and p<d):
        s=s*k
        p=p+1
    return min(n,s)

if ok(n,k,d):
    for i in range(1,d+1):
        t,o=1,[]
        while len(o)<n:
            o=o+[str(t),]*ll(n-len(o),k,d-i)
            t=t+1
            if t>k:t=1
        print ' '.join(o)
else:
    print -1
