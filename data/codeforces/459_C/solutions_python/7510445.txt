n, k, d = map(int, map(int, raw_input().strip().split(' ')))

if n > (k ** d):
    print -1
    exit(0)

a = []
a.append([0 for _ in range(d)])

for i in range(1, n):
    a.append([])
    for j in range(d):
        a[i].append(a[i - 1][j])

    for j in range(d - 1, -1, -1):
        a[i][j] = (a[i][j] + 1) % k
        if a[i][j]:
            break

for i in range(d):
    print " ".join(str(a[j][i] + 1) for j in range(n))
