#!/usr/bin/env python
# coding=utf-8

n, k, d = map(int, raw_input().split())
if n > k ** d:
    print -1
else:
    cur = range(n)
    for i in range(d):
        toPrint = []
        for j in range(n):
            toPrint.append(cur[j] % k + 1)
            cur[j] /= k
        print " ".join(map(str, toPrint))
