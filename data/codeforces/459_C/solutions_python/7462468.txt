from sys import stdout
from itertools import product, islice, izip
def main():
    n, k, d = map(int, raw_input().split())
    a = [x for x in izip(*islice(product(*(xrange(1, min(k,n)+1) for _ in xrange(d))), 0, n))]
    if len(a[0]) >= n:
        stdout.write('\n'.join(' '.join(map(str, x)) for x in a))
    else:
        stdout.write("-1")
main()
