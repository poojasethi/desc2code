def solve(n, k, d):
    if n > k**d:
        return [[-1]]
    res = [[1 for i in range(n)] for j in range(d)]
    for i in range(n):
        j, c = i, -1
        while j > 0:
            res[c][i] = j % k + 1
            c -= 1
            j /= k
    return res

n, k, d = map(int, raw_input().split())
res = solve(n, k, d)
for i in range(len(res)):
    print ' '.join(map(str, res[i]))
