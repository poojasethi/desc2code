n,k,d=map(int,raw_input().split())
if k**d<n:
    print -1
else:
    ans=[]
    for i in range(n):
        cur=[]
        evaluate=i
        for j in range(d):
            cur+=[evaluate%k+1]
            evaluate/=k
        ans+=[cur]
    ans=zip(*ans)
    print '\n'.join(' '.join(map(str,ans[i])) for i in range(d))
