import sys

n, k, d = [int(x) for x in raw_input().strip().split()]

if n > k ** d:
    print -1
    sys.exit()

def converttobase(x, b, d):
    res = []
    for i in range(d):
        res = [x % b] + res
        x /= b
        if x == 0:
            break
    res = [0]*(d - len(res)) + res
    return res

ans = [converttobase(i, k, d) for i in range(n)]
for i in range(d):
    print ' '.join([str(ans[x][i] + 1) for x in range(n)])
