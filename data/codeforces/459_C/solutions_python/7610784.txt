read = lambda: map(int, raw_input().split())

n, k, d = read()

if (k == 1 and n > 1) or (d <= 10 and k ** d < n):
	print -1
else:
	ans = []
	for i in xrange(n):
		cur = []
		num = i
		for j in xrange(d):
			cur.append(1 + num % k)
			num /= k
		ans.append(cur)
	for i in xrange(d):
		for j in xrange(n):
			print ans[j][d - 1 - i],
		print ''
