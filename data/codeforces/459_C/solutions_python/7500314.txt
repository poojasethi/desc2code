import math


def print_kids(kids):

	print


(kidCount, busCount, dayCount) = [int(x) for x in raw_input().split()]

kidBuses = []
for i in range(0, kidCount):
	kidBuses.append([])

if busCount >= kidCount:
	kids = range(0, kidCount)
	for i in xrange(0, dayCount):
		print_kids(kids)
else:
	if kidCount > busCount ** dayCount:
		print -1
	else:
		kids = [0] * kidCount
		chunks = [range(0, kidCount)]

		prevChunkSize = kidCount
		for i in xrange(0, dayCount):
			chunkSize = int(math.ceil(float(prevChunkSize) / busCount))

			newChunks = []
			for chunk in chunks:
				busIndex = 0
				bus = []

				for kid in chunk:
					kids[kid] = busIndex
					bus.append(kid)
					if len(bus) == chunkSize:
						newChunks.append(bus)

						bus = []
						busIndex += 1

				if len(bus) > 0:
					newChunks.append(bus)
			chunks = newChunks

			print_kids(kids)
			if chunkSize == 1:
				break

			prevChunkSize = chunkSize

		for j in xrange(i + 1, dayCount):
			print_kids(kids)
