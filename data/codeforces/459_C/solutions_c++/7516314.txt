#include<cstdio>
using namespace std;
int a1[1000][1000];
int main()
{
	int a,b,c,d;bool bl=1;
	scanf("%d%d%d",&a,&b,&c);
	d=1;
	for(int i=0;i<c;++i)
	{d*=b;if(d>=a){bl=0;break;}}
	if(bl){puts("-1");return 0;}
	for(int i=0;i<a;++i)
	{
		d=i;
		for(int j=0;j<c;++j)
		{a1[i][j]=d%b;d/=b;}
	}
	for(int i=0;i<c;puts(""),++i)
	for(int j=0;j<a;++j)
	{
		if(j)putchar(' ');
		printf("%d",a1[j][i]+1);
	}
}
