#include<algorithm>
#include<iostream>
using namespace std;
int now[10000],t[10000];
int n,k,d;
bool f()
{
	long long whe=1,ret=0,t=d;
	while(t--&&n>whe)whe*=k,ret++;
	return n<=whe;
}
int main()
{
	cin>>n>>k>>d;
	if(!f())cout<<-1<<endl;
	else
	{
		for(int i=0;i<n;i++)t[i]=i;
		for(int i=0;i<d;i++)
		{
			for(int j=0;j<n;j++)
			{
				cout<<t[j]%k+1<<" ";
				t[j]/=k;
			}
			cout<<endl;
		}
	}
	return 0;
}
