#include<bits/stdc++.h>
using namespace std;
int n,d,bus,s[1001];
int main(){
    scanf("%d %d %d",&n,&bus,&d);
    bus=min(bus,n);
    int cur=1;
    for(int i=1;i<=d;i++){
        cur*=bus;
        if(cur>=n) break;
    }
    if(cur<n){
        printf("-1\n");
        return 0;
    }
    for(int i=0;i<n;i++) s[i]=i;
    for(int i=0;i<d;i++){
        for(int j=0;j<n;j++){
            printf("%d%c",s[j]%bus+1,j==n-1?'\n':' ');
            s[j]/=bus;
        }
    }
    return 0;
}
