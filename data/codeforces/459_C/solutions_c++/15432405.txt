#include <cstdio>

#define N 1000

using namespace std;

int n,k,d,di[N+10][N+10];

int main(){
	scanf("%d%d%d",&n,&k,&d);
	for (int i=0;i<n;++i){
		int l=0;
		bool f=0;
		for (int j=i;j;j/=k){
			di[i][++l]=j%k;
			if (l>d){f=1;break;}
		}
		if (f){puts("-1");return 0;}
	}
	for (int i=d;i;--i)
		for (int j=0;j<n;++j) printf("%d%c",di[j][i]+1,j==n-1?'\n':' ');
	return 0;
}
