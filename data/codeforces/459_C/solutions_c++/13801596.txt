#include<cstdio>
int n,k,d,a[1001],c[1001][1001];
void p1(){
	int i=1;
	while(a[i]==k){a[i]=1;i++;}
	a[i]++;
}
int main(){
	scanf("%d%d%d",&n,&k,&d);
	int i,j,m=n-1;
	for(i=1;i<=d;i++) {a[i]=1;m=m/k;}
	if(m) printf("-1");
	else{ 
		for(i=1;i<=n;i++){
	    	for(j=1;j<=d;j++) c[i][j]=a[j];
	    	p1();
    	}
    	for(j=1;j<=d;j++){
    		for(i=1;i<=n;i++) printf("%d ",c[i][j]);
    		printf("\n");
    	}
	}
	return 0;
}
