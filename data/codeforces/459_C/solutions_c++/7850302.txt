#include<iostream>

using namespace std ;

typedef long long ll ;
ll k ;
int n , d ;
int a[1010] ;

int main()
{
	cin >> n >> k >> d ;
	ll x=1 ;
	for(int i=0 ; i<d && x<n ; i++)
		x*=k ;
	if(x<n)
		cout << -1 << endl ;
	else
	{
		for(int i=0 ; i<n ; i++)
			a[i]=i ;

		for(int i=0 ; i<d ; i++)
		{
			for(int j=0 ; j<n ; j++)
			{
				cout << a[j]%k+1 << " " ;
				a[j]/=k ;
			}
			cout << endl ;	
		}
	}
	return 0 ;
}
