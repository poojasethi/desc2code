#include<iostream>
using namespace std;
#define ll long long
ll ans[1000][1000];
int main()
{
    ll n,k,d;
    cin>>n>>k>>d;
    for(ll ni=0;ni<n;ni++)
    {
        ll a=ni;
        for(ll di=d-1;di>=0;di--)
        {
            ans[ni][di]=(a%k)+1;
            a/=k;
        }
        if(a!=0)
        {
            cout<<-1<<endl;
            return 0;
        }
    }
    for(ll di=0;di<d;di++,cout<<endl)
        for(ll ni=0;ni<n;ni++)
            cout<<ans[ni][di]<<" ";
}
