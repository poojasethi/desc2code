#include <iostream>
#include <cmath>

using namespace std;

int n,k,d;
int cur[1010][1010];

void gen() {
  	for(int i=0;i<n;i++) {
  		fill(cur[i],cur[i]+n,0);
  		int num=i,j=d-1;
		while(num) {
    		cur[i][j--]=num%k;
    		num/=k;
  		}	
  	}
	
}

int main() {
	cin>>n>>k>>d;
	if(pow((double)k,d)<(double)n) {
		cout<<-1; return 0;
	}
	gen();
	for(int j=0;j<d;j++) {
		for(int i=0;i<n;i++) cout<<cur[i][j]+1<<" ";
		cout<<endl;
	}
	
	
	
	return 0;
}
