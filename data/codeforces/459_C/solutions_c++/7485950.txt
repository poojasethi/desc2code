#include <cstdio>
int n,k,d;
int ind[1001];
int main(){
    scanf("%d%d%d",&n,&k,&d);
    for(int i=1;i<=1000;i++)ind[i]=i;
    int tn=1;
    for(int td=0;td<d;td++){tn*=k;if(tn>n)break;}
    if(tn<n)puts("-1");
    else for(int td=0;td<d;td++){
        for(int i=0;i<n;i++){
            printf("%d%c",ind[i]%k+1,i==n-1?'\n':' ');
            ind[i]=(ind[i]+k-1)/k;
        }
        tn=(tn+k-1)/k;
    }
    return 0;
}
