#include<iostream>
using namespace std;
typedef long long ll;
int n,d;
ll k;
const int maxn=1000+3;
ll b[maxn];
bool check(){
	int u=1;
	for(int i=0;i<d;i++)
	{
		u*=k;
		if(u>n)
			return true;
	}
	if(u==n)return true;
	return false;
}
int main()
{
	cin>>n>>k>>d;
	if(check())
	{
		for(int i=0;i<n;i++)
			b[i]=i;
		for(int j=0;j<d;j++)
		{
			for(int i=0;i<n;i++)
			{
				cout<<(b[i]%k)+1<<" ";
				b[i]/=k;
				
			}
			cout<<endl;
		}
	}
	else
	{
		cout<<-1<<endl;
	}
	return 0;
	return 0;
}
