#include<cstdio>
using namespace std;
int n,k,d,i,x,cnt,j,a[1005][1005];
int main()
{
  scanf("%d%d%d",&n,&k,&d);
  for (x=1,i=1;i<=d;i++)
  {
    x*=k;
    if (x>=n) break;
  }
  if (x<n) {puts("-1");return 0;}
  for (i=0;i<=n;i++)
    for (cnt=1,x=i;cnt<=d;cnt++)
      a[cnt][i]=x%k,x/=k;
  for (i=1;i<=d;i++,puts(""))
    for (j=1;j<=n;j++)
      printf("%d ",a[i][j]+1);
  return 0;
}
