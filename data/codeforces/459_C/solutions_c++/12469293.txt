#include<stdio.h>
int n,k,d,temp,i,j,arr[1010][1010];
int main(){
	scanf("%d %d %d",&n,&k,&d);
	temp = k;
	for(i=2;i<=d;i++){
		if(temp>=n)
			break;
		temp*=k;
	}
	if(n>temp){
		puts("-1");
		return 0;	
	}
	
	for(int i=1;i<=n;i++){
	
		for(int j=1;j<=d;j++)
			arr[i][j] = arr[i-1][j];

		for(int j=d ; j>=1;j--){
			arr[i][j] = (arr[i][j] +1)%k;
			if(arr[i][j])
				break;
		}
	}
	for(int i=1;i<=d;i++,puts(""))
		for(int j=1;j<=n;j++)
			printf("%d ",arr[j][i]+1);


}
