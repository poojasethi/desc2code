#include <iostream>
#include <cmath>

using namespace std;

int n,k,d;
int cur[1010][1010];

void gen(int num) {
  	fill(cur[num],cur[num]+n,0);
  	int j=num,i=d-1;
	while(num) {
    	cur[j][i--]=num%k;
    	num/=k;
  	}
}

int main() {
	cin>>n>>k>>d;
	if(pow((double)k,d)<(double)n) {
		cout<<-1; return 0;
	}
	for(int i=0;i<n;i++) gen(i);
	for(int j=0;j<d;j++) {
		for(int i=0;i<n;i++) cout<<cur[i][j]+1<<" ";
		cout<<endl;
	}
	
	
	
	return 0;
}
