#include<cstdio>
#include<cmath>
int n, k, d, i, a[1001][1001];
int main()
{
	scanf("%d%d%d", &n, &k, &d);
	long long x=1;
	for (i=1; x<=n && i<=d; i++, x=x * 1LL * k);
	if (x < n) {printf("-1"); return 0;}
	for (int j=1,ts=1; j<=n; j++,ts=j)
		for (i=1; i<=d; i++)
			a[i][j]=(ts % k) + 1, ts = ts / k;
	for (i=1; i<=d; i++)
	{
		for (int j=1; j<=n; j++)
			printf("%d ", a[i][j]);
		printf("\n");
	}
}
