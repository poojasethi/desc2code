#include<iostream>
using namespace std;
long long n,k,d,data[1005];
long long i,j,res;
int main()
{
    cin>>n>>k>>d;
    res=1;
    for(i=0;i<d&&res<n;i++)
        res*=k;
    if(res<n) cout<<"-1"<<endl;
    else
    {

        for(i=0;i<n;i++) data[i]=i;
        for(i=0;i<d;i++)
        {
            for(j=0;j<n;j++)
                {cout<<data[j]%k+1<<" ";data[j]/=k;}
            cout<<endl;

        }
    }
    return 0;

}
