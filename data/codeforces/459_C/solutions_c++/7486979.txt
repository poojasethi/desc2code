#include<cstdio>
#include<cmath>

int main()
{
	int n,k,d,finalk,i,j,printval,finald;
	scanf("%d%d%d",&n,&k,&d);
	finalk=k;
		
	for(i=1;i<=d;i++)
	{
		if(k>=n)
			break;
		k=k*finalk;
	}
	if(i>d)
		printf("-1\n");
	else
	{	
		finald=i;
		i=0;
		while(i<finald)		
		{
			k=k/finalk;
			for(j=0;j<n;j++)
			{
				printval=j/k;
				printf("%d ",(printval%finalk)+1);
			}
			printf("\n");
			i++;
		}	
		while(i<d)
		{
			for(j=0;j<n;j++)
				printf("1 ");
			printf("\n");
			i++;
		}
							
	}
}
