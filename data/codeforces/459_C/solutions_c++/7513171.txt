#include<iostream>
using namespace std;
int x[1005][1005];
int main()
{int n,k,d; cin>>n>>k>>d;

for (int j=2;j<=n;j++)
{for (int i=1;i<=d;i++) x[i][j]=x[i][j-1];
 int i=d;
 for (;i>0;i--)
 {x[i][j]=(x[i][j]+1)%k;
  if (x[i][j]!=0) break;
 }
 if (i==0) {cout<<-1; return 0;}
}
for (int i=1;i<=d;i++)
{
 for (int j=1;j<=n;j++)  cout<<x[i][j]+1<<" ";
 cout<<endl;
}

return 0;
}
