#include <cstdio>
#include <iostream>
#include <cmath>
#define eps 1e-8
using namespace std;
int n, k, d;
int a[1020][1020];
int main(){
  cin >> n >> k >> d;
  if (d * log(k) + eps < log(n)){cout << -1; return 0;}
  for (int i = 1; i <= n; i++){
    int s = i;
    for (int j = 1; j <= d; j++){
      a[i][j] = 1 + (s % k); s /= k;
    }
  }
  for (int i = 1; i <= d; i++){
    for (int j = 1; j <= n; j++) printf("%d ", a[j][i]);
    printf("\n");
  }
  return 0;
}
