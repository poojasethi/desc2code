#include<iostream>
using namespace std;
int n, k, d;
__int64 p  = 1;
int a[10005];
int main(){
    cin>> n >> k >>d;
    for(int i = 1; i <= d; i ++){
        p *= k;
        if(p > n) break;
    }
    if(p < n){
        cout<<-1;
        return 0;
    }
    for(int i = 0; i < n; i ++) a[i] = i;
    for(int i = 0; i < d; i ++){
        for(int j = 1; j <= n; j ++){
            cout<< a[j] % k + 1 <<" ";
            a[j] /= k;
        }
        cout<<endl;
    }
    return 0;
}
