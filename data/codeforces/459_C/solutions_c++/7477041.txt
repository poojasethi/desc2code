#include<iostream>
using namespace std ;
const int maxn=1000000 ;
int n , k , d , t=1 ;
int x[maxn] ;

int main()
{
	cin >> n >> k >> d ;
	for(int i=0 ; i<d && t<n ; i++)
		t*=k ;
	if(t<n)
	{
		cout << -1 << endl ;
		return 0 ;
	}
	for(int i=0 ; i<n ; i++)
		x[i]=i ;
	for(int i=0 ; i<d ; i++)
	{
		for(int j=0 ; j<n ; j++)
		{
			cout << x[j]%k+1 <<" " ;
			x[j]/=k ;
		}
		cout << endl ;
	}
	return 0 ;
}
