#include <iostream>
using namespace std;
#define lol long long
const int MAXN=1010;
lol a[MAXN];
int main()
{
  lol n,k,d;
  cin>>n>>k>>d;
  lol x=1;
  for(int i=0;i<d and x<n;i++)
    x*=k;
  if(x<n)
    {
      cout<<-1<<endl;
      return 0;
    }
  for(int i=0;i<n;i++)
    a[i]=i;
  for(int i=0;i<d;i++)
    {
      for(int j=0;j<n;j++)
	{
	  cout<<(a[j]%k)+1<<" ";
	  a[j]/=k;
	}
      cout<<endl;
    }
  return 0;
}
