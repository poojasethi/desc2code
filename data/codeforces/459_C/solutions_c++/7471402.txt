#include<iostream>
using namespace std;
int pow(int a,int b,int n){
	int ans=1;
	for(int i=1;i<=b;i++){
		ans=ans*a;
		if(ans>n){
			return n;
		}
	}
	return ans;
}

int main(){
	int n,k,d;
	cin>>n>>k>>d;
	int f=pow(k,d,n);
	if(f<n){
		cout<<-1<<endl;
	}else{
		for(int i=d;i>0;i--){
			f=pow(k,i-1,n);
			int j=1;
			int bus=1;
			cout<<bus;
			while(j<n){
				if(j%f==0){
					if(bus==k){
						bus=0;
					}
					bus++;
				}
				j++;
				cout<<" "<<bus;
			}
			cout<<endl;
		}
	}
	return 0;
}
