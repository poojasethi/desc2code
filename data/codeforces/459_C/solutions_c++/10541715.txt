#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,k,d,i,j,a,b,f=0;
	cin>>n>>k>>d;
	a=1;
	for(i=0;i<d;i++)
	{
		a=a*k;
		if(n<=a)
		{
			f=1;
			break;
		}
	}	
	if(f)
	{
		long long int s[d][n];
		for(i=0;i<d;i++)
			for(j=0;j<n;j++)
				s[i][j]=1;
		for(i=0;i<n;i++)
		{
			j=0;
			a=i;
			while(a)
			{
				s[j++][i]=a%k+1;
				a=a/k;
			}
		}
		for(i=0;i<d;i++)
		{
			for(j=0;j<n;j++)
				printf("%lld ",s[i][j]);
			printf("\n");	
		}	
	}
	else
		printf("-1\n");
	return 0;	
}
