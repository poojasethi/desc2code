#include<iostream>
using namespace std;
int n, k, d, tmp = 1, big = 0;
int main() {
    cin >> n >> k >> d;
    for (int i = 0; i < d; i++) {
        tmp*=k; if (tmp >= n) goto good;
    }
    cout << -1 << endl; return 0;
    good:
    tmp = 1;
    for (int i = 0; i < d; i++) {
        for (int j = 0; j < n; j++) {
            cout << (j/(big?n:tmp))%k+1 << (j==n-1?'\n':' ');
        }
        if (tmp*k > n) big = 1;
        else tmp*=k;
    }
    return 0;
}
