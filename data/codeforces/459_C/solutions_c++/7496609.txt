#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
void solve()
{
	int n, k, d;
	cin >> n >> k >> d;
	long long x = 1;
	for(int i = 0; i < d && x < n; ++i) x *= k;
	if(x < n) {
		cout<<"-1"<<endl;
	} else {
		vector<int> f(n);
		for(int i = 0; i < n; ++i) f[i] = i;
		for(int i = 0; i < d; ++i) {
			for(int j = 0; j < n; ++j) {
				cout << f[j] % k + 1 << " ";
				f[j] /= k;
			}
			cout << endl;
		}
	}
}
int main()
{
	ios::sync_with_stdio(false);
	solve();
	return 0;
}
