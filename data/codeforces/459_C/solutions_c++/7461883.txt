#include <cstdio>

int n,k,d,f[1009],g[1009][1009];
bool flg;

int main()
{
	scanf("%d%d%d",&n,&k,&d);
	for (int j=1;j<=d;++j) f[j]=1;
	for (int i=1;i<=n;++i)
	{
		int j;
		for (j=1;j<=d;++j) g[i][j]=f[j];
		if (i<n)
		{
			for (j=1;j<=d;++j)
			{
				if (flg=(++f[j]>k)) f[j]=1;
				else break;
			}
			if (j>d)
			{
				puts("-1");
				return 0;
			}
		}
	}
	for (int j=1;j<=d;++j)
	{
		for (int i=1;i<=n;++i) printf("%d ",g[i][j]);
		putchar('\n');
	}
	return 0;
}
