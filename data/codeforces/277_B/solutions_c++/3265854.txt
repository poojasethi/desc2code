#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;

int main()
{
	int n, m;
	cin >> n >> m;
	if (m == 3 && n > 4)
		puts("-1"), exit(0);
	for (int i = 1; i <= m; ++i)
		cout << i << " " << i * i + 1000000 << endl;
	for (int i = 1; i <= n - m; ++i)
		cout << i << " " << -i * i - 1000000 << endl;
	return 0;
}
