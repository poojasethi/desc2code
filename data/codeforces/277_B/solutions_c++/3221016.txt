#include<iostream>
#include<stdio.h>

using namespace std;

long long n,m,i;

main()
 {
  cin>>n>>m;
  if (n >= 5 && m==3) {cout<<-1<<endl;return 0;}
  if (n == 4 && m==3) {cout<<"0 0\n0 3\n3 0\n1 1\n";return 0;}
  for (i=1;i<=m;i++)
   cout<<i<<" "<<10000000+i*i<<endl;
  for (i=1;i<=n-m;i++)
   cout<<-i<<" "<<-10000000-i*i<<endl;
 }
