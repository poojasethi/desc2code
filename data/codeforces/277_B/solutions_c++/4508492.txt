#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;
const int H=1e7;

int main()
{
    int n,m;
    cin>>n>>m;
    if(m==3&&n>=5)
    {
        cout<<-1<<endl;
        return 0;
    }
    for(int i=1;i<=m;i++)
        cout<<i<<" "<<i*i+H<<endl;
    for(int i=1;i<=n-m;i++)
        cout<<i<<" "<<-i*i-H<<endl;
    return 0;
}
