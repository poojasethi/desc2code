#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main(){
    int n,m,i;
    scanf("%d%d",&n,&m);
    if (n>=5&&m==3) printf("-1\n");
    else{
        for(i=0;i<m;++i){
            printf("%d %d\n",i,2000000+i*i);
        }
        for(i=0;i<n-m;++i){
            printf("%d %d\n",i,-2000000-i*i);
        }
    }
    return 0;
}
