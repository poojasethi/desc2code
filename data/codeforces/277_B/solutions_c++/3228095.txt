#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
   int i,n,m;
   scanf("%d%d",&n,&m);
   if(n>=5 && m==3) printf("-1");
   else
   {
    for(i=1;i<=m;++i)
     printf("%d %d\n",i,i*i);
    for(i=1;i<=n-m;++i)
	 printf("%d %d\n",-i,-i*i-1000000); 
   }
   return 0;
}
