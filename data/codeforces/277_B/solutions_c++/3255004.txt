#include <iostream>
using namespace std;

int n, m;
int main()
{
	cin >> n >> m;
	if (m == 3 && n > 4) cout << -1 << endl;
	else if (m ==3 && n == 4) 
		cout << "1 0\n-1 0\n0 1\n0 2\n";
	else {
		for (int i = 1; i <= m; ++i)
			cout << i << ' ' << i*i+1000000 << endl;
		for (int i = 1; i <= n-m; ++i)
			cout << -i << ' ' << -(i*i) -1000000 << endl;
	}
}
