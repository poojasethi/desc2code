#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int n,m;

int main () {
  cin>>n>>m;
  if (m == 3 and n > 4) {
    cout<<(-1)<<endl;
    return 0;
  }
  for (int i = 0; i < m; ++i) {
    cout<<(i)<<" "<<(i*i+100000)<<endl;
  }
  for (int i = m; i < n; ++i) {
    cout<<(i)<<" "<<(-i*i-100000)<<endl;
  }
}
