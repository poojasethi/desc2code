#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int main (){
	int n, m; scanf("%d%d", &n, &m);
	if(m == 3 && n > 4) return puts("-1"), 0;
	for(int i = 1; i <= m; i++)
		printf("%d %d\n", i, i*i+300000);
	for(int i = 1; i <= n-m; i++)
		printf("%d %d\n", i, -i*i-300000);
}
