N = 205
n = 0
m = 0

ax = [0] * N
ay = [0] * N

def main():
    n, m = map(int, raw_input().split())
    if m == 3 and n >= 5:
        print -1
        return
    for i in range(0, m):
        ax[i] = 10000000 + i * i
        ay[i] = i
    for i in range(0, n - m):
        ax[i + m] = -10000000 - i * i
        ay[i + m] = -i
    for i in range(0, n):
        print ax[i], ay[i]

if __name__ == '__main__':
    main()
