n, m = map(int, raw_input().split())
if m == 3:
    if n == 3:
        print("0 0")
        print("0 1")
        print("1 0")
    elif n == 4:
        print("0 0")
        print("3 0")
        print("0 3")
        print("1 1")
    else:
        print("-1")
else:
    for i in range(m):
        print i, 10000000 + i * i
    for i in range(n - m):
        print i, -10000000 - i * i
