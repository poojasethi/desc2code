
A = raw_input("")

s = A.find('.')
if s == -1:
	s = len(A)

A = A.translate(None,".")

b = 0
while A[b] == '0':
	b += 1;
b += 1;

A = A[0:b] + "." + A[b:len(A)]

e = A.find('.')

w = s - e

a = 0
b = len(A) - 1
FNL = ""

while A[a] == '0':
	a += 1
while A[b] == '0':
	b -= 1
if A[b] == '.':
	b -= 1

for i in range(a , b + 1):
	FNL += A[i]
add = ""
if w != 0:
	add = 'E'+str(w)

print FNL + add
