s=raw_input()
if s.find('.') == -1:
	s+='.'
fpos=-1
for i in range(len(s)):
	if s[i]!= '0' and s[i]!='.':
		fpos=i
		break
for i in range(len(s)):
	if s[i] != '0' and s[i] !='.':
		lpos=i
ppos=s.index('.')
#print fpos,ppos,lpos
if fpos == -1:
	print '0'
else :
	ans=s[fpos]
	if fpos != lpos :
		ans+='.'
	for i in range(fpos+1,lpos+1):
		if s[i]!='.':
			ans+=s[i]
	if fpos <= ppos :
		mi=str(ppos-fpos-1)
	else :
		mi=str(ppos-fpos)
	if mi != '0':
		ans+='E'
		ans+=mi
	print ans
