s = raw_input()
n = len(s)
start = 0
while s[start] == '0' or s[start] == '.' : 
  start += 1
if s.find('.') == -1 :
  dot = n 
else :
  dot = s.find('.')  
ans = s[start] + '.'
for x in s[start + 1 : ] :
  if x != '.' :
    ans += x
end = len(ans) - 1
while (ans[end] == '0') :
  end -= 1
if ans[end] == '.' :
  end -= 1
ans = ans[:end + 1]

ex = dot - start - 1

if ex < 0 : ex += 1
if ex != 0 :
  ans += 'E' + str(ex)

print ans
