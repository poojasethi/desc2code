#include<bits/stdc++.h>
using namespace std;

int main(){

    string str,f;
	cin>>str;
	int e,s;
	for(s=0;(str[s]=='0'||str[s]=='.')&&s<str.length();s++);
	for(e=str.length()-1;(str[e]=='0'||str[e]=='.')&&e>=0;e--);
	for(int i=s;i<=e;i++) if(str[i]!='.')   f+=str[i];
	if(f.length()==0){cout<<"0"<<endl;return 0;}
	int d=str.find('.');
	cout<<f[0];
	if(f.length()>1) cout<<'.'<<f.substr(1,f.length()-1);
	int k=0;
	if(d!=-1) {
		if (d<s)  k=d-s;
		else      k=d-s-1;
	}
	else {
		k=f.length()-1+str.length()-e-1;
	}
	if(k) cout<<"E"<<k<<endl;

	return 0;
}
