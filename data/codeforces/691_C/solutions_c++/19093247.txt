#include <stdio.h>
#include <string.h>
#define N 1000005

int main() {
	char s[N];
	scanf("%s", s);
	int n = strlen(s), l = 0, r = n - 1, d = -1, p, exp;
	while(l < n && s[l] == '0') l++;
	if(l == n) {
		printf("0\n");
		return 0;
	}
	p = l;
	while(p < n && s[p] != '.') {
		p++;
	}
	while(r > p && s[r] == '0') {
		r--;
	}
	if(l == r) {
		if(s[l] == '.')
			s[l] = '0';
		printf("%c\n", s[l]);
		return 0;		
	}
	if(l == p) {
		l++;
		while(l <= r && s[l] == '0')
			l++;
	}
	if(l < p) {
		exp = p - l - 1;
	} else {
		exp = p - l;
	}
	l = 0;
	while(s[l] == '0' || s[l] == '.')
		l++;
	printf("%c", s[l]);
	r = n - 1;
	while(r > l && (s[r] == '0' || s[r] == '.'))
		r--;
	if(r != l)
		printf(".");
	while(++l <= r) {
		if(s[l] == '.') {
			continue;
		}
		printf("%c", s[l]);
	}
	if(exp)
		printf("E%d\n", exp);
}
