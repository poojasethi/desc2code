using namespace std;

int main()
{
	string s;
	cin>>s;
	int len=s.length();
	int pos=len;
	for(int i=0;i<len;i++)
	{
		if(s[i]=='.') pos=i;
	}
	if(pos!=len)
	{
		s.erase(pos,1);
	}
	pos--;
	int beg=0;
	while(s[beg]=='0') beg++;
	pos-=beg;
	s.erase(0,beg);
	int bac=s.length()-1;
	while(s[bac]=='0') bac--;
	bac++;
	s.erase(bac,s.length()-bac);
	if(s.length()>1)
	{
		s.insert(1,".");
	}
	cout<<s;
	if(pos!=0) cout<<"E"<<pos<<endl;
	else cout<<endl;
	return 0;
}
