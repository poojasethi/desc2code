	if (pos != (int)s.size())
		s.erase(pos,1);
	int Bgn=0;
	for (;s[Bgn]=='0';Bgn++);
	s.erase(0,Bgn);
	pos -= Bgn;
	for (Bgn=(int)s.size()-1;s[Bgn]=='0';Bgn--);
	s.erase(Bgn+1);
	if (s.empty()) s="0";
	if ((int)s.size() > 1)
		s.insert(1,".");
	cout << s;
	if (pos-1 != 0) cout << "E" << pos-1;

}
