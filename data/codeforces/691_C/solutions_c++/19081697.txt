#include <cstdio>
#include <string>
#include <iostream>

int main()
{
    std::string s;
    int n=-1, i=0, r, beg, en;
    std::cin >> s;
    while (s[i]=='0') i++;
    if (s[i]=='.') { i++; n=i; while (s[i]=='0') i++;}
    beg=i;
    i=s.size()-1;
    while (s[i]=='0') i--;
    if (s[i]=='.') { n=i; i--; while (s[i]=='0') i--;}
    en=i;
    printf ("%c", s[beg]);
    if (beg+1<=en) printf(".");
    for (i=beg+1; i<=en; i++) {
        if (s[i]=='.') n=i;
        else printf ("%c", s[i]);
    }
    if (n==-1) n=s.size();
    r=n-beg-1;
    if (r!=0) printf ("E%d", r);
    return 0;
}
