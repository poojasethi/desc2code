#include <bits/stdc++.h>
using namespace std;

string S;

int main() {
	
	cin >> S;
	
	int f = -1, dot = S.size(), last = S.size()-1;
	
	for(int i=0; i<S.size(); i++)
	{
	    if(S[i]>='1' && S[i]<='9')
	    {
	        printf("%c", S[i]);
	        f = i;
	        break;
	    }
	}
	
	for(int i=0; i<S.size(); i++)
	    if(S[i]=='.')
	        dot = i;
	
	for(; last>=0 && S[last]=='0' || S[last]=='.'; last--);
	
	int flag = 0;
	
	for(int i=f+1; i<=last; i++)
	{
	    if(S[i]>='0' && S[i]<='9')
	    {
	        if(!flag)
	        {
	            flag = 1;
	            printf(".");
	        }
	        
	        printf("%c", S[i]);
	    }
	}
	
	int num = dot - f - 1;
	
	if(num>0)
	    printf("E%d\n", num);
	else if(num<0)
	    printf("E%d\n", num+1);
	
	return 0;
}
