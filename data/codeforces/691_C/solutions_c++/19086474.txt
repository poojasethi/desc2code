#include<iostream>
using namespace std;
int main(){
    string s;
    cin>>s;
    int dot,first=-1,last=-1,exp;
    dot=s.find('.');
    if(dot==string::npos)
        dot=s.length();
    for(int i=0;i<s.length();++i){
        if((s[i]!='.') &&  s[i]-'0'>0){
            if(first==-1)
                first=i;
            last=i;
        }
    }
    exp=dot-first;
    if(exp>0) exp--;
    if(first==last){
        cout<<s[first];
    }
    else{
        cout<<s[first]<<".";
        for(int i=first+1;i<=last;++i){
            if(s[i]!='.'){
                cout<<s[i];
            }
        }

    }
    if(exp!=0){
            cout<<"E"<<exp;
        }
return 0;
}
