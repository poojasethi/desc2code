#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
char st[1001000];
int n;
int xiaoshudian=-1,shouwei=0,mowei=0;
int b;
int main(){
	scanf("%s",st);
	n=strlen(st);
	for (int i=0;i<n;i++) if (st[i]=='.') xiaoshudian=i;
	for (int i=0;i<n;i++) if (st[i]>'0'&&st[i]<='9'){
		shouwei=i;
		break;
	}
	for (int i=n-1;i>=shouwei;i--) if(st[i]>'0'&&st[i]<='9'){
		mowei=i;
		break;
	}
	if (xiaoshudian==-1) xiaoshudian=n;
	if (xiaoshudian>shouwei){
		b=xiaoshudian-shouwei-1;
	}else{
		b=xiaoshudian-shouwei;
	}
	if (shouwei==mowei){
		printf("%c",st[shouwei]);
		if (b!=0){
			printf("E%d",b);
		}
		printf("\n");
	}else{

		printf("%c.",st[shouwei]);
		for (int i=shouwei+1;i<=mowei;i++)
			if(st[i]!='.'){
				printf("%c",st[i]);
			}
		if(b!=0) printf("E%d",b);
		printf("\n");
	}
	return 0;
}
