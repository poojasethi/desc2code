#include<bits/stdc++.h>
using namespace std;
char s[1000000];
int d,a,b,c,n;
int main(){
	scanf("%s",s);
	n=strlen(s);
	for( a=0;s[a]=='0' || s[a]=='.';a++);
	for( b=n-1;s[b]=='0' || s[b]=='.';b--);
	c=find(s, s+n, '.')-s;
	putchar(s[a]);
	if(a!=b) putchar('.');
	for(int i=a+1;i<=b;i++){
		if(s[i]!='.') putchar(s[i]);
	}
	if(a<c) d=c-a-1;
	else d=c-a;
	if(d) printf("E%d",d);
}
