#include<algorithm>
#include<cstdio>
#include<cstring>

using namespace std;

char s[1000005];
int i,x,y,n,d,e;

int main(){

  scanf("%s",s);
  n=strlen(s);
  d=n;
  for (i=0;i<n;i++)
    if (s[i]=='.') d=i;
  x=0;
  y=n-1;
  while (x<n && (s[x]=='0' || s[x]=='.') ) x++;
  while (y>=0 && (s[y]=='0' || s[y]=='.') ) y--;
  if (x<=d) e=d-x-1;
  else e=d-x;
  printf("%c",s[x]);
  if (x!=y) printf(".");
  for (i=x+1;i<=y;i++)
    if (s[i]!='.') printf("%c",s[i]);
  if (e!=0) printf("E%d\n",e);

  return 0;





}
