#include<cstdio>
#include<cstring>
int n;
char s[1000005];
int main(){
	int x,y,d,ex;
	scanf("%s",s);
	n=strlen(s);
	d=n;
	for(int i=0;i<n;i++)
		if(s[i]=='.')
			d=i;
	x=0;y=n-1;
	while(x<n&&(s[x]=='0'||s[x]=='.'))	x++;
	while(y>=0&&(s[y]=='0'||s[y]=='.'))	y--;
	if(x<=d)	ex=d-x-1;
	else	ex=d-x;
	printf("%c",s[x]);
	if(x!=y)	printf(".");
	for(int i=x+1;i<=y;i++)
		if(s[i]!='.')
			printf("%c",s[i]);
	if(ex!=0)	printf("E%d",ex);
	return 0;
	}
