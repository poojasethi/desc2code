#include <bits/stdc++.h>

using namespace std;

main() {
	string s;
	cin >> s;
	int b = 0, e = s.size() - 1;
	int p = -1, t = 0;
	while (s[b] == '0')
		b++;
	p = find(s.begin(), s.end(), '.') - s.begin();
	while (p != s.size() && s[e] == '0')
		e--;
	if (s[e] == '.')
		e--;
	while ((p == s.size() || e < p) && s[e] == '0')
		e--, t++;
	if (s[b] == '.')
		b++, t--;
	while ((p == s.size() || b > p) && s[b] == '0')
		b++, t--;
	if (e - b + 1 == 0)
		return cout << 0 << endl, 0;
	cout << s[b];
	if (e - b + 1 >= 2)
		cout << '.';
	for (int i = b + 1; i <= e; i++) {
		if (s[i] == '.')
			continue;
		cout << s[i];
		if (i < p)
			t++;
	}
	if (!t)
		return 0;
	cout << 'E' << t << endl;
}
