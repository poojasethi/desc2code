#include<bits/stdc++.h>

using namespace std;

int main() {
	char x[1000010];
	scanf("%s", x);

	int i, j=0, point = 0, start = 0, first = -1, last = -1, b, pos1 = -1, pos2 = -1; 
	char newx[1000020];
	for(i=0; x[i]!='\0'; ++i) {
		if(x[i] == '.') {
			pos1 = i;
		}
		else if(start == 0 && x[i] != '0') {
			start = 1;
			newx[0] = x[i];
			newx[1] = '.';
			pos2 = i;
			j = 2;
		}
		else {
			if(x[i] != '0') {
				if(first == -1)
					first = j;
				last = j;
			}
			newx[j] = x[i];
			++j;
		}
	}
	if(last != -1)
		newx[last+1] = '\0';
	else
		newx[1] = '\0';

	if(pos2 == -1) {
		printf("0\n");
		return 0;
	}

	if(pos1 == -1)
		b = i-1 - pos2;
	else if(pos1 > pos2)
		b = pos1 - (pos2+1);
	else
		b = pos1 - pos2;

	if(b == 0)
		printf("%s\n",newx );
	else
		printf("%sE%d\n", newx, b);

	return 0;		
}
