using namespace std;
string s; 
int main()
{
	cin>>s;
	int p=s.find_first_of('.');
	if(p==string::npos)
	p=s.size();
	else s.erase(p,1);
	int k=s.find_first_not_of('0');
	s.erase(s.begin(),s.find_first_not_of('0')+s.begin());
	s.erase(s.begin()+s.find_last_not_of('0')+1,s.end());
	//cout<<s;
	if(s.size()>1)
	s.insert(1,1,'.');
	cout<<s;
	if(p-=k+1)
	cout<<"E"<<p;
	return 0;
}
