#include <iostream>
#include <string>
#include <algorithm>

using namespace std;;

int main(){
	string str;
	cin >> str;
	int p = str.size(), pos = 100000000, pose;
	for (int i = 0; i<str.size(); i++){
		if (str[i] != '.' && str[i] != '0') pos = min(pos, i);
		if (str[i] == '.') p = i;
	}
	int b = p - pos - 1;
	if (b < 0) b++;
	for (int i = str.size() - 1; i >= pos; i--){
		if (str[i] != '0' && str[i] != '.'){
			pose = i; break;
		}
	}
	if (pos != pose){
		cout << str[pos] << ".";
		for (int i = pos + 1; i <= pose; i++){
			if (str[i] != '.') cout << str[i];
		}
	}
	if (pos == pose) cout << str[pos];
	if (b != 0){
		cout << "E" << b << endl;
	}
	return 0;
}
