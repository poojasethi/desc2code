#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
int p,ch,a[2000000],num;
int main()
{
	while(ch=getchar(),((ch<='9')&&(ch>='0'))||(ch=='.'))
	{
		a[++num]=ch;
		if(ch=='.') p=num;
	}
	if(!p) p=num+1;
	int f=0;
	while(f++,((a[f]=='0')||a[f]=='.')&&(f<num));
	int l=num+1;
	while(l--,((a[l]=='0')||a[l]=='.')&&(l>f));
	printf("%c",a[f]);
	if(l>f) printf(".");
	for(int i=f+1;i<=l;++i) if(a[i]!='.')printf("%c",a[i]);
	if(p!=f+1)
	{
		printf("E");
		printf("%d",(p-f-1)<0?p-f:p-f-1);
	}
	return 0;
}
