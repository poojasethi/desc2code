#include<bits/stdc++.h>
using namespace std;
int main(){
	char s[1000001];
	cin>>s;
	int n=strlen(s),pl,st=-1,en=-1,c=0,bf=0;
	for(int i=0;i<n;i++)
		if(s[i]!='0' && s[i]!='.'){
			st=i;
			break;
		}
	for(int i=n-1;i>=0;i--)
		if(s[i]!='0' && s[i]!='.'){
			en=i;
			break;
		}
	for(int i=n-1;i>=0;i--)
		if(s[i]=='.'){
			c=0, bf=0;
			pl=i;
		}
		else if(s[i]=='0')
			bf++;
		else{
			c++;
			c+=bf;
			bf=0;
		}
	if(st==-1 && en==-1){
		cout<<"0"<<endl;
		return 0;
	}
	if(c==0) c=pl-st+1;
	cout<<s[st];
	if(st!=en){
		cout<<".";
		for(int i=st+1;i<=en;i++){
			if(i==pl) continue;
			cout<<s[i];
		}
	}
	if(c!=1)
		cout<<"E"<<c-1<<endl;
	return 0;
}
