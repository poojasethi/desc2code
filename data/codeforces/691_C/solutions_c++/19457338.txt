#include <iostream>
#include <cstring>
using namespace std;

char c[1000002];
int main()
{
		cin.getline(c, 1000002);
		int l = strlen(c);
		int start = -1, end = -1;
		int mark = l;
		for (int i = 0; i < l; i++)
		{
			if (c[i] == '.') mark = i;
			else if (c[i] != '0')
			{
				if (start < 0) start = i;
				end = i;
			}
		}
		cout << c[start];
		if (start != end)
		{
			cout << ".";
			for (int i = start + 1; i <= end; i++)
			{
				if (i != mark) cout << c[i];
			}
		}
		if (mark > start + 1) cout << "E" << mark - start - 1;
		if (mark < start) cout << "E" << mark - start;
		return 0;
}
