#include<bits/stdc++.h>
using namespace std;
# define l long long int 
# define rep(i,a,b) for(l i=a;i<b;i++)
int main(){
	string s;
	cin>>s;
	string ans="";
	l n=s.length();
	l j=-1,k=-1;
	rep(i,0,n){
		if(s.at(i)=='0'||s.at(i)=='.') continue;
		else {j=i;break;}
	}
	rep(i,0,n){
		if(s.at(n-1-i)=='0'||s.at(n-1-i)=='.') continue;
		else {k=n-1-i; break;}
	}
	l d;
	rep(i,0,n){
		if(!(s.at(i)<='9'&&s.at(i)>='0')) {d=i; break;}
	}
	if(s.at(d)!='.') d=n;
	//cout<<j<<k<<d<<"\n";
	if(j==k){
		ans=s.at(j);
	}
	else{
		ans=s.substr(j,1);
		ans=ans+".";
		if(d<j+1||d>k) ans=ans+s.substr(j+1,k-j);
		else {ans=ans+s.substr(j+1,d-j-1);
			ans=ans+s.substr(d+1,k-d);}
	}
	if(d!=(j+1)) {
		ans=ans+"E";
		if(j+1>d) ans=ans+"-";
		l p=fabs(j+1-d);
		if(j+1>d) p--;
		stringstream r;
		r<<p;
		ans=ans+r.str();
	}
	cout<<ans<<"\n";
	return 0;
}
