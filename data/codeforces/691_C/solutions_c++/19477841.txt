#define mp make_pair
using namespace std;
string str;
map <int,int> ma;
int main(){
	cin>>str;
	int dot=str.find_first_of('.');
	if (dot!=-1)str.erase(dot,1); else dot=str.length();
	for (int i=0;i<str.size();i++){
		ma[(-(dot-i-1))]=str[i]-'0';
	}  
	while (ma.begin()->second==0) ma.erase(ma.begin());
	while ((--ma.end())->second==0) ma.erase(--ma.end());
	for (map<int,int>::iterator it=ma.begin();it!=ma.end();it++){
		cout<<it->second;
		if (ma.size()>1 && it==ma.begin()++) cout<<'.';
	}  
	if (-(ma.begin()->first))cout<<"E"<<-(ma.begin()->first);
	return 0;
}
