#include <bits/stdc++.h>
using namespace std;
typedef long long lli;
char s[1000004];
int main ()
{
  int i, j, k, b = 0;
  scanf ("%s", s);
  for (i = 0; s[i] == '0'; i++) {}
  for (b = 0; i+b<strlen(s) && s[i + b] != '.'; b++) {}
  b--;
  for (; s[i] == '0' || s[i] == '.'; i++) if (s[i] == '0') b--;
  for (j = strlen (s); s[j - 1] == '0' || s[j - 1] == '.'; j--) {}
  for (k = 0; i < j; i++) {
    if (s[i] != '.') {
      if (k++ == 1) putchar ('.');
      putchar (s[i]);
    }
  }
  if (b) printf ("E%d", b);
  printf ("\n");
  return 0;
}
