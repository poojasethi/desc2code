#include <iostream>
#include <stdio.h>
#include <cstring>
using namespace std;
int main()
{
	char str[1000005];
	cin>>str;
	int l,r,point,len;
	len=strlen(str);
	for(l=0;str[l]=='0'||str[l]=='.';l++);
	for(r=len-1;str[r]=='0'||str[r]=='.';r--);
	for(point=0;str[point]!='.' && point<len ;point++);
	cout<<str[l];
	if(l!=r) {
		cout<<".";
		for(int i=l+1;i<=r;i++) if(str[i]!='.') cout<<str[i];
	}
	if(point-l!=1)
	{
		cout<<"E";
		if(point-l>0)
			cout<<point-l-1;
		else cout<<point-l;
	}
	cout<<endl;
	return 0;
}
