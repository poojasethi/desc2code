using namespace std;
string s;
int pos,len,beg;
int main()
{
	cin>>s;
	len=s.length();
	pos=len;
	for(int i=0;i<len;i++)
		if(s[i]=='.')pos=i;
	if(pos!=len)
	{
		s.erase(pos,1);
		len--;
	}
	pos--;
	while(s[beg]=='0')beg++;
	pos-=beg;
	s.erase(0,beg);
	beg=s.length()-1;
	while(s[beg]=='0')beg--;
	s.erase(beg+1,s.length());
	if(s.length()>1)s.insert(1,".");
	cout<<s;
	if(pos!=0)
		cout<<'E'<<pos;
	return 0;
}
