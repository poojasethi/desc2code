#include <iostream>
#include <string>
using namespace std;

int main(){
  string x;
  cin>>x;
  int n = x.length();
  int i = 0;
  while(x[i] < '1')  i++;
  string dot(".");
  size_t found = x.find(dot);
  int b = n-i-1;
  while(x[n-1] == '0') n--;
  if (found!=string::npos){

    if (i<found){
      b = found - i - 1; 
    }
    else{
      b = -(i - found); i--; 
    }   
    x.erase(x.begin() + found); n--;
    while(x[n-1] == '0') n--;
  }
  cout<<x[i];
  if(n>i+1) cout<<".";
  for(int j = i+1; j<n ; j++) cout<<x[j];
  if (b != 0) cout<<"E"<<b;
  cout<<endl;
  return 0;
}
