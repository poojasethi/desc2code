#include<bits/stdc++.h>
using namespace std;
char s[int(1e6+10)];
int l,r,p=-1;
int main(){
	scanf("%s",s);
	int n=strlen(s);
	p=find(s,s+n,'.')-s;
	for(l=0;s[l]=='0';l++);
	for(r=n-1;s[r]=='0'||s[r]=='.';r--);
	int fir=0;
	while(s[fir]=='0'||s[fir]=='.')fir++;
	putchar(s[fir]);
	int delt=0;
	if(p>fir)delt=p-fir-1;
	else delt=p-fir;
	if(fir<r)putchar('.');
	for(int i=fir+1;i<=r;i++)if(s[i]!='.')putchar(s[i]);
	if(delt)printf("E%d",delt);
	puts("");
	return 0;
}
