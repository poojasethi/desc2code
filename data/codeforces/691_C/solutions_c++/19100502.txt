#include<cstdio>
#include<cstring>
using namespace std;
char str[1111111],ans[1111111];

int main(){
	int flag=-1,wei,num=0;
	scanf("%s",str);
	wei=strlen(str);
	for(int i=0;str[i];i++){
		if(flag==-1&&str[i]!='0'&&str[i]!='.')	flag=i;
		if(str[i]=='.')	wei=i;
		if(flag!=-1) ans[num++]=str[i];
	}
	num=strlen(ans);
	for(int i=num-1;ans[i]=='0'||ans[i]=='.';i--)	num--;
	if(!num){
	ans[0]='0';
	num++;
	}
	printf("%c",ans[0]);
	if(num!=1)	printf(".");
	for(int i=1;i<num;i++)	if(ans[i]!='.')	printf("%c",ans[i]);
	wei=wei-flag;
	if(wei>0)	wei--;
	if(wei&&flag!=-1)	printf("E%d",wei);
	printf("\n");
	return 0;
}
