#include <iostream>
#include <algorithm>
#include <string>
#include <cmath>
#include <set>
#include <cstdio>
using namespace std;
char str[1000005];
string s;
int main()
{
	cin>>str;
	int pnt=0;
	int pnt2=0;
	s=str;
	s="0"+s;
	while(pnt<s.size()&&s[pnt]!='.')
	pnt++;
	if(pnt==s.size())
		s+=".0";
	pnt=0;
	while(pnt<s.size()&&(s[pnt]=='0'||s[pnt]=='.'))
		pnt++;
	int fr=pnt;
	for(int i=pnt+1;i<s.size();i++)
		if(s[i]!='0'&&s[i]!='.')
		fr=i;
	cout<<s[pnt];
	if(pnt<fr)
	{
		cout<<".";
		for(int i=pnt+1;i<=fr;i++)
		if(s[i]!='.')
			cout<<s[i];
	}
	while(s[pnt2]!='.')
		pnt2++;
	if(pnt+1!=pnt2)
		if(pnt2<pnt)
		cout<<"E"<<pnt2-pnt;
		else
		cout<<"E"<<pnt2-pnt-1;
return 0;
}
