#include<stdio.h>
#include<string.h>
char s[1000010];
int main()
{
	int n;
	int a,b;
	int i,j,k;
	scanf("%s",s);
	n=strlen(s);
	for(i=0;i<n;i++)
		if(s[i]!='0'&&s[i]!='.')
			break;
	for(j=n-1;j>=0;j--)
		if(s[j]!='0'&&s[j]!='.')
			break;
	if(i>j)
	{
		printf("0\n");
		return 0;
	}
	for(k=0;k<n;k++)
		if(s[k]=='.')
			break;//printf("\n  ");printf("%d %d %d\n    ",i,j,k);
	for(a=i;a<=j;a++)
	{
		if(s[a]>='0'&&s[a]<='9')
			printf("%c",s[a]);
		if(a==i&&i<j)
			printf(".");
	}
	if(k<i)
		b=k-i;
	else
		b=k-1-i;
	if(b)
		printf("E%d",b);
	printf("\n");
	return 0;
}
