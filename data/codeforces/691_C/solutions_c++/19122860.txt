#include<bits/stdc++.h>
using namespace std;
char x[1000010];
int main()
{
    while(scanf("%s", x) > 0)
    {
        int len = strlen(x);
        int point = len, a = -1, b = -1;
        for(int i=0; i<len; i++)
        {
            if(x[i] == '.'){point = i;}
            else if(x[i] != '0'){if(a == -1){a = i;}b = i;}
        }
        if(a == -1){printf("0\n");continue;}
        printf("%c", x[a]);
        if(a != b) printf(".");
        for(int i=a+1; i<=b; i++)
        {
            if(x[i] != '.') printf("%c", x[i]);
        }
        if(point < a+1)
        {
            printf("E%d", point-a);
        }
        else if(point > a+1)
        {
            printf("E%d", point-a-1);
        }
        printf("\n");
    }
    return 0;
}
