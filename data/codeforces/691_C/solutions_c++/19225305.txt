#include <iostream>
using namespace std;

int main() {
	string a;
	cin>>a;
	int len=a.size(), dot=a.find('.');
	if (dot==string::npos)
		dot=len;
	int start=-1, end=-1;
	for (int i=0; i<len; i++) {
		if (a[i]>'0' && a[i]<='9') {
			start=i;
			break;
		}
	}
	if (start==-1) {
		cout<<"0\n";
		return 0;
	}
	for (int i=len-1; i>=0; i--) {
		if (a[i]>'0' && a[i]<='9') {
			end=i;
			break;
		}
	}
	cout<<a[start];
	if (end>start)
		cout<<'.';
	for (int i=start+1; i<=end; i++) {
		if (a[i]!='.')
			cout<<a[i];
	}
//	cout<<'\n'<<start<<' '<<end<<' '<<dot<<'\n';
	int e=dot-start-1;
	if (e<0)
		e++;
	if (e!=0)
		cout<<"E"<<e<<'\n';
	return 0;
}
