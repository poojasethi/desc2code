#include <map>
#include <set>
#include <queue>
#include <cstdio>
#include <vector>
#include <cstring>
#include <sstream>
#include <iostream> 
#include <algorithm>
using namespace std;
map <int,int> k;
string s;
int main()
{
	cin>>s;
	int p=s.length();
	for(int i=0;i<s.length();i++)
	if(s[i]=='.')
	p=i;
	for(int i=0;i<s.length();i++)
	{
	if(i<p)
	k[i+1-p]=s[i]-'0';
	else if(i>p)
	k[i-p]=s[i]-'0';
	}
	while(k.begin()->second==0)
	k.erase(k.begin());
	while((--k.end())->second==0)
	k.erase(--k.end());
	int e=k.begin()->first;
	for(map <int,int>::iterator it=k.begin();it!=k.end();it++)
	{
	cout<<it->second;
	if(it->first==e&&k.size()>1)
	cout<<".";
	}
	if(e)
	cout<<"E"<<-e;
	return 0;
}
