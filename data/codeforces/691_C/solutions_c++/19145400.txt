#include <bits/stdc++.h>

using namespace std;
char a[1000001],b[1000001];
int main()
{
    int i,j,k=0,l,n,c=-1;
    scanf("%s",a);
    n=strlen(a);
    for(i=0;i<n;i++)
    {
        if(a[i]!='0'&& a[i]!='.')
            break;
    }
    for(j=n-1;j>=0;j--)
    {
        if(a[j]!='0' && a[j]!='.')
            break;
    }
    for(c=0;c<n;c++)
        if(a[c]=='.')
        break;
    b[k++]=a[i];
    if(i!=j)b[k++]='.';
    for(l=i+1;l<=j;l++)
    {
        if(a[l]!='.')
        b[k++]=a[l];
    }
    for(l=0;l<k;l++)
    {
        printf("%c",b[l]);
    }
        if(c>i)
            {if(c-i-1!=0)printf("E%d",c-i-1);}
        else
        {
            printf("E%d",c-i);
        }

    return 0;
}
