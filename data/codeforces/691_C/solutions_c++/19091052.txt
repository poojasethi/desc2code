#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;

char s[1000100]; 

int main() {
	scanf("%s",s);
	int l1=0,len=strlen(s)-1;
	while ((s[l1]=='0'||s[l1]=='.')&& l1<len) l1++;
	int r1=len;
	while ((s[r1]=='0'||s[r1]=='.')&&r1>0) r1--;
	int i=0;
	while (s[i]!='.'&&i<len) i++;
//	printf("%d %d %d\n",len,l1,r1);
	if (s[i]!='.') {
		printf("%c",s[l1]);
		if (l1<r1) printf(".");
		for (int j=l1+1;j<=r1;j++) printf("%c",s[j]);
		if (len) printf("E%d",len-l1);
	}
	else {
		printf("%c",s[l1]);
		if (l1<r1) printf(".");
		for (int j=l1+1;j<=r1;j++) {
			if (s[j]=='.') continue;
			printf("%c",s[j]);
		}
		if (l1+1!=i) {
			if (i<l1) printf("E%d",i-l1);
			else printf("E%d",i-l1-1);
		}
	}
	return 0;
}
