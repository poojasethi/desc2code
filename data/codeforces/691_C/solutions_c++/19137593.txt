#include <iostream>

using namespace std;

int main()
{
    int n;
    string s;
    cin>>s;
    n=s.size();
    int fir,poi=n,k=0,e;
    bool b=1;
    for (int i=0;i<n;i++){
        if (s[i]=='.') poi=i;
        if (s[i]!='0' && s[i]!='.' && b){b=0;fir=i;}
    }
    for (int i=n-1;i>=0;i--){
        if (s[i]!='0' && s[i]!='.') break;
        k++;
    }
    if (poi>fir) e=poi-fir-1;
    else e=poi-fir;
    bool p=0;
    for (int i=0;i<n-k;i++){
        if (s[i]=='0'){if (p) cout<<s[i];}
        else if (s[i]=='.') continue;
        else if (i==fir){cout<<s[i];p=1;if (fir!=n-k-1)cout<<'.';}
        else cout<<s[i];
    }
    if (e!=0)cout<<'E'<<e;
    return 0;
}
