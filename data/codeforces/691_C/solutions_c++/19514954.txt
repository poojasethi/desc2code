#include<bits/stdc++.h>

using namespace std;
char num[1000000];
/*     
	在数学中，有效数字是指在一个数中，从该数的第
	一个非零数字起，直到末尾非零数字止的数字称为有效数字，如
	0.618的有效数字有三个，分别是6,1,8
*/
int main(){
	
	scanf("%s",num);
	int len=strlen(num);
	int s=-1,e=-1,p=len;
	for(int i=0;i<len;i++){
		if(s==-1 && num[i]!='0' && num[i]!='.')s=i;
		if(num[i]!='0' && num[i]!='.')e=i;
		if(num[i]=='.')p=i;
	}
	if(s>=0)
	{
		cout<<num[s];
		if(e>s)cout<<".";
		for(int j=s+1;j<=e;j++){
			if(num[j]!='.')cout<<num[j];
		}
		if((s+1)!=p){
			cout<<"E";
			if(s<p)cout<<p-s-1;
			else if(s>p)cout<<p-s;
		}
	}
	else cout<<"0"<<endl;
	return 0;
}
