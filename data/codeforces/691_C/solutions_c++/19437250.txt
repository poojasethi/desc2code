#include <iostream>
#include<cstdio>
#include<cstring>
#include<string>
using namespace std;

int main()
{
    string s1;
    cin>>s1;
    int len=s1.length(),s=-1,e=len-1,x=len;
    for(int i=0;i<len;i++)
    if(s1[i]=='.'){
        x=i;
        break;
    }
    for(int i=0;i<len;i++)
        if(s1[i]>'0'&&s1[i]<='9')
    {
        s=i;break;
    }
    for(int i=len-1;i>=0;i--)
    if(s1[i]>'0'&&s1[i]<='9'){
        e=i;
        break;
    }
    if(s>=0)
    {
        cout<<s1[s];
        if(e>s) cout<<".";
        for(int i=s+1; i<=e; i++)
            if(s1[i]!='.') cout<<s1[i];
        if((s+1)!=x)
        {
            cout<<"E";
            if(s<x) cout<<x-s-1<<endl;
            else if(s>x) cout<<x-s<<endl;
        }
    }
    else
        cout<<"0"<<endl;
    return 0;
}
