#include <bits/stdc++.h>

#define NN 1000010
using namespace std;

char str[NN];

int main (){
	
	scanf("%s", str);
	int N=strlen(str), i;
	
	int st=-1, ed=-1, po=-1;
	for(i=0; i<N; i++){
		if(str[i]>'0' && str[i]<='9'){
			if(st==-1) st=i;
		}
		if(str[i]>='0' && str[i]<='9'){
			ed=i;
		}
		if(str[i]=='.') po=i;
	}
	
	if(po==-1) po=ed+1;
	if(st==-1){
		puts("0");
		return 0;
	}
	
	while(str[ed]=='0' || str[ed]=='.') ed--;
	
	printf("%c", str[st]);
	if(ed>st) printf(".");
	for(i=st+1; i<=ed; i++){
		if(str[i]!='.') printf("%c", str[i]);
	}
	
	int d=po-st-1;
	if(d<0) d++;
	if(d!=0){
		printf("E%d", d);
	}
	puts("");
	
	return 0;
}
