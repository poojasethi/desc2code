#include <iostream>
#include <cstdio>
typedef long long ll;

using namespace std;

int N;
int t, c;
ll DP[3000];
ll INF = 1000000000000000;

int main()
{
    scanf("%d", &N);

    DP[0] = 0;
    for(int Ni = 1; Ni <= N; Ni++) DP[Ni] = INF;

    for(int Ni = 0; Ni < N; Ni++)
    {
        scanf("%d %d", &t, &c); t++;

        for(int Nj = N; Nj > 0; Nj--)
        {
            DP[Nj] = min(DP[Nj], DP[ max(0, Nj-t)]+c);
        }
    }

    printf("%I64d\n", DP[N]);
}
