#include<bits/stdc++.h>

using namespace std;

long long M[2003][4044],A[2003],B[2003],n,i;

long long f(int i, int t){

    if(t > n)
	    return 0;
	
    if(i == n+1){
	    if(t < 0)
	        return 1000000000000000;
		else
		    return 0;
	}

	long long &r = M[i][t+2003];

    if(r != -1)
		return r;

	return r = min(f(i+1,t+A[i]) + B[i] , f(i+1,t-1));
}
int main(){

	cin >> n;

	for(i=1;i<=n;i++)
		cin >> A[i] >> B[i];
	
	memset(M , -1 , sizeof M);
	cout << f(1,0);
	return 0;
}
