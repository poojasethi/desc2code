#include<bits/stdc++.h>
using namespace std;
int t, c;
long long dp[3000];
main(){
    int n;
    scanf("%d", &n);
    dp[0] = 0;
    for(int i=1;i<=n;i++) dp[i] = 990000000000000;
    for(int i=0;i<n;i++){
        scanf("%d%d",&t,&c);
        t++;
        for(int j=n;j>0;j--){
            dp[j]=min(dp[j],dp[min(n,j+1)]);
            dp[j]=min(dp[j],dp[max(0,j-t)]+c);
        }
    }
    printf("%I64d\n",dp[n]);
}
