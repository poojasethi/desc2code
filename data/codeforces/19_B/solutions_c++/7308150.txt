#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define LL  long long
#define INF 0x7fffffff
LL n,t[2001],c[2001],f[2001];
int main()
{
	scanf("%I64d",&n);for(LL i=1;i<=n;i++)scanf("%I64d%I64d",&t[i],&c[i]),t[i]++;
	for(LL i=1;i<=n;i++)f[i]=INF,f[i]*=f[i];for(LL i=1;i<=n;i++)for(LL j=n;j>=0;j--)
	f[min(j+t[i],n)]=min(f[min(j+t[i],n)],f[j]+c[i]);printf("%I64d\n",f[n]);return 0;
}
