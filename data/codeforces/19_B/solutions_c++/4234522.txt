#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int n,i,t,c,q,j;
long long f[2001];
long long a=1000;
int main()
{
long long inf=a*a*a*a*a*a;
cin>>n;
for (i=1;i<=n;i++)
f[i]=inf;
for (i=1;i<=n;i++)
{
        cin>>t>>c;
	t++;
	for (j=n-1;j>=0;j--)
	{
	q=j+t;
	if (q>n)
	q=n;
	if (f[j]+c<f[q])
		f[q]=f[j]+c;
	}
}
cout<<f[n];

return 0;
}
