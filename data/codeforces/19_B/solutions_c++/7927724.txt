 #include <bits/stdc++.h>
#define INF 1000000000000000000
using namespace std;
int n,a,b;
long long int d[2005];
int main()
{
	scanf("%d",&n);
	for (int i = 1; i <= n+1; i++)
		d[i] = INF;
	d[0] = 0;
	for (int i = 0; i < n; i++) 
	{
		scanf("%d %d",&a,&b);
		for (int j = n; j > 0; j--) 
			d[j] = min(d[j+1], min(d[j], d[max(0, j-a-1)] + b));
	} 
	cout << d[n];
	return 0;
}
