#include <iostream>
#include <cstdlib>
#include <vector>
using namespace std;     
typedef long long ll;
int main() {
    ll c, n, t;
    cin >> n;
    vector <ll> d(n+1,(ll)1<<62);
    d[0]=0;
    for (int i=0;i<n;++i) {
        cin >> t >> c;
        for (int j=n;j>=0;--j)
            d[min(n,j+t+1)]=min(d[min(n,j+t+1)],d[j]+c);
    }
    cout << d[n];
    return 0;
}
