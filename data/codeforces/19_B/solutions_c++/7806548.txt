#include<iostream>
#define INF 10e16 
using namespace std;
long long ans[10000];
int main()
{
    int n;
    cin>>n;
    long long a,b;
    for(int i=1;i<=n;i++)ans[i]=INF;
    for(int j=0;j<n;j++)
    {
        cin>>a>>b,a++;
        for(int i=n;i>=0;i--)
            if(i>=a)ans[i]=min(ans[i],ans[i-a]+b);
            else ans[i]=min(ans[i],b);
    }
    cout<<ans[n];
}
