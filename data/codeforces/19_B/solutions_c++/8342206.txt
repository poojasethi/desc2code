#include<bits/stdc++.h>
using namespace std;
const int N=20000;
const long long inf=1e15;
long long f[N];
int main() {
    #ifndef ONLINE_JUDGE
    freopen("in.txt","r",stdin);
    #endif
    int n;
    int c,t;
    cin>>n;
    for(int i=1;i<=n+1;i++) {
        f[i]=inf;
    }
    f[0]=0;
    for(int i=0;i<n;i++) {
        cin>>t>>c;
        t++;
        for(int j=n;j>=0;j--) {
            f[min(j+t,n)]=min(f[min(j+t,n)],c+f[j]);
        }
    }
    cout<<f[n]<<endl;
    return 0;
}
