#include <bits/stdc++.h>
#define INF 1000000000000000000
using namespace std;
int N,t,c;
long long int Dizi[2005];
int main()
{
	cin >> N;
	for (int i = 1; i <= N+1; i++)
		Dizi[i] = INF;
	Dizi[0] = 0;
	for (int i = 0; i < N; i++) 
	{
		cin >> t >> c;
		for (int j = N; j > 0; j--) 
			Dizi[j] = min(Dizi[j+1], min(Dizi[j], Dizi[ max(0, j-t-1)] + c));
	} 
	cout << Dizi[N];
	return 0;
}
