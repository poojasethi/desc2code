#include <stdio.h>
#include <algorithm>
using namespace std;

int main(void) {
	int n, i, j;
	long long ans = 1LL<<60;
	long long dp[2005];
	scanf("%d", &n);
	for(i=0;i<=n;i++)
		dp[i] = ans;
	dp[0] = 0;
	for(i=0;i<n;i++){
		int t;
		long long c;
		scanf("%d%I64d", &t, &c);
		++t;
		if(t>n) t=n;
		for(j=n-t;j<n;j++)
			ans = min(ans, dp[j]+c);
		for(j=n-t-1;j>=0;--j)
			dp[j+t] = min(dp[j+t], dp[j]+c);
	}
	printf("%I64d\n", ans);
	return 0;
}
