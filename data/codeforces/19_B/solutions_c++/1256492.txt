#include<iostream>


using namespace std;


	long long n,dp[2001]={};
	int t[2001],c[2001]={};

int main()
{

	cin>>n;

	for(int i = 0 ; i < n ; ++i )	
	{	
		cin>>t[i]>>c[i];
		++t[i];
	}

	for(int i =  1 ; i <= n ; ++i )
	{
		dp[i] = 100000000000000LL ;
	}

	for(int i = 0 ; i < n ; ++i )
	{
		for(int j = n ; j >= 0 ; --j )
		{
			dp[j] = min(dp[j] , dp[ max ( j-t[i],0) ] + c[i]);
		}
	}

	cout<<dp[n]<<endl;
}
