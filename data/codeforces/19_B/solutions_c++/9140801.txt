#include <cstdio>
#include <cstring>

#define ll long long

ll f[2010], c;
int n, t, tmp;

int main(){
	scanf("%d", &n);
	
	memset(f, 127, sizeof(f));
	f[0] = 0;
	
	for (int i = 0; i < n; i++){
		scanf("%d%I64d", &t, &c);

		for (int j = n; j >= 0; j--){
			tmp = t + j < n ? t + j + 1 : n;
			f[tmp] = f[tmp] < c + f[j] ? f[tmp]: c + f[j];
		}
	}
	
	printf("%I64d", f[n]);
	
	return 0;
}
