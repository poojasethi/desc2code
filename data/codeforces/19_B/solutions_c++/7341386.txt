#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
LL dp[2010];
int main(){
    int n,t,c;
    while(cin>>n){
        dp[0] = 0;
        for(int i = 1;i <= n;i++)   dp[i] = 1LL<<62;
        for(int i = 0;i < n;i++){
            cin>>t>>c;
            for(int j = n;j >= 0;j--)
                dp[min(n,j+t+1)] = min(dp[min(n,j+t+1)],dp[j]+c);
        }
        cout<<dp[n]<<endl;
    }
    return 0;
}
