#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long ll;
int n;
int t[2000];
int c[2000];
ll dyn[4002];
int main(){
	cin>>n;
	for (int i=0;i<n;++i){
		cin>>t[i]>>c[i];
	}
	memset(dyn,127,sizeof dyn);
	dyn[0]=0;
	for (int i=0;i<n;++i)
		for (int j=2000;j>=0;--j)
			if (dyn[j+1+t[i]]>dyn[j]+c[i]){
				dyn[j+1+t[i]]=dyn[j]+c[i];
			}
	ll ans=1000000000000000LL;
	for (int i=n;i<=4001;++i)
		ans=min(ans,dyn[i]);
	cout<<ans<<endl;
	return 0;
}
