#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstdio>

using namespace std;

int N, T[2005], C[2005];
long long D[2005][2005];
int main() {
  scanf("%d",&N);
  for(int i = 1;i<=N;i++) {
    scanf("%d %d",&T[i], &C[i]);
  }

  fill(D[0] + 1, D[0] + 1 + N, 1LL<<60);

  for(int i = 1;i<=N;i++) {
    for(int j = 0;j<=N;j++) {
      D[i][j] = min(D[i - 1][j], D[i - 1][max(0, j - T[i] - 1)] + C[i]);
    }
  }
  cout<<D[N][N]<<endl;
  return 0;
}
