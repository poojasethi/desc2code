#include <iostream>
#include <algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
long long dp[2012];
int n,t,c;
const long long int INF=100000000000000ll;
int main(){
    cin>>n;
    fill(dp,dp+n+1,INF);
    dp[0]=0;
    rep(i,n){
        cin>>t>>c;t++;//t++ ,,steal use one time,,displacement all time needed 
        for(int j=n;j>=0;j--)dp[min(j+t,n)]=min(dp[min(j+t,n)],c+dp[j]);//update all state from original
    }
    cout<<dp[n]<<endl;
    return 0;
}
