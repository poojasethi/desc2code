#include<stdio.h>

#include<math.h>
#include<algorithm>
#include<string.h>
using namespace std;
int i,j,k,n,a[2001],b[2001];long long f[2001];
int main()
{scanf("%d",&n);
 for(i=1;i<=n;i++)
    scanf("%d%d",&a[i],&b[i]),a[i]++;
 for(i=1;i<=n;i++)
    {f[i]=1100000000;f[i]*=f[i];}
 for(i=1;i<=n;i++)
    {for(j=n;j>=0;j--)
		if(f[j]+b[i]<f[min(j+a[i],n)])
		  f[min(j+a[i],n)]=f[j]+b[i];
    }
 cout<<f[n]<<endl;
 return 0;
}
