#include <cstdio>
#include <algorithm>
using namespace std;

const long long INF = 999888777666555444ll;

int main(){
    int n,t,c;
    scanf("%d",&n);
    long long dp[2001]={};
    fill(dp+1,dp+n+1,INF);
    for(int i=dp[0]=0;i<n;i++){
        scanf("%d%d",&t,&c);
        for(int k=n;k>=0;k--)
            dp[min(n,k+t+1)]=min(dp[min(n,k+t+1)],dp[k]+c);
    }
    printf("%I64d\n",dp[n]);
}
