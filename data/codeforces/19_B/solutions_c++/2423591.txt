#include <cstdio>
using namespace std;

int main()
{
    long long v[2001];
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) v[i] = -1;
    v[n] = 0;
    int it, ic;
    for (int i = 0; i < n; i++) {
        scanf("%d%d", &it, &ic);
        for (int j = 1; j <= n; j++) {
            int k = j - 1 - it < 0 ? 0 : j - 1 - it;
            if (v[j] != -1 && (v[k] == -1 || v[k] > v[j] + ic)) v[k] = v[j] + ic;
        }
    }
    printf("%I64d\n", v[0]);
    return 0;
}
