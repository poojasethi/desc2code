#include <algorithm>
#include <cstdio>
#include <cstring>
using namespace std;
typedef __int64 Int;

Int s[2001];

int main()
{
    int n;
    scanf("%d", &n);
    memset(s, 0x33, sizeof(s));
    s[0] = 0;
    for (int i = 0; i < n; i++) {
        int t, c;
        scanf("%d%d", &t, &c);
        for (int j = n-1; j >= 0; j--)
            s[min(j+t+1, n)] = min(s[min(j+t+1, n)], s[j]+c);
    }
    printf("%I64d\n", s[n]);
}
