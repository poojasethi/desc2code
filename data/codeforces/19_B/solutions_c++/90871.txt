#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
typedef long long LL;
LL f[2010][4010];
int t[2010],c[2010];
LL dp(int i,int j){
	LL& res=f[i][j+2005];
	if(res!=-1)return res;
	if(i==0)return res=j>=0?0:0x1111111111111111LL;
	if(j>=i)return res=0;
	return res=min(dp(i-1,j-1),dp(i-1,j+t[i])+c[i]);
}
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)scanf("%d%d",&t[i],&c[i]);
	memset(f,-1,sizeof(f));
	cout<<dp(n,0)<<endl;
	return 0;
}
