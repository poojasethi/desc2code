#include <iostream>
#include <algorithm>
using namespace std;

const long long INF = 10000000000000000LL;
long long dp[2001];

int main()
{
	int n, t, c;
	cin >> n;
	fill(dp + 1, dp + n + 1, INF);
	dp[0] = 0;
	for(int k = 0; k < n; k++) {
		cin >> t >> c;
		t++;
		for(int i = n + t; i >= t; i--)
			dp[min(i, n)] = min(dp[min(i, n)], dp[i-t] + c);
	}
	cout << dp[n] << endl;
	return 0;
}
