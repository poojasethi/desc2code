#include <algorithm>
#include <cstdio>
#include <iostream>

using namespace std;

typedef long long int Lint;

const int MAXN=2002;

int N;

Lint dn[MAXN];

int main(){
	
	scanf(" %d",&N);
	
	fill(dn+1,dn+N+1,(1LL)<<60);
	
	for( int x,y,i=0 ; i<N ; i++ ){
		
		scanf(" %d %d",&x,&y);
		x++;
		
		for( int j=N ; j>=0 ; j-- )
			dn[j]=min(dn[j],dn[max(0,j-x)]+y);
	}
	
	printf("%I64d\n",dn[N]);
	return 0;
}
