#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
 
ll minc[2010];
int main() {
        int n;
        cin>>n;
        fill(minc+1,minc+n+1,1LL<<61);
        for(int i=0;i<n;i++) {
                int t,c;
                cin>>t>>c;
                t++;
                for(int i=n;i>=0;i--) {
                        minc[i]=min(minc[i],minc[max(0,i-t)]+c);
                }
        }
        cout<<minc[n];
}
