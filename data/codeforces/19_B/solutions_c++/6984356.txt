#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long LL;
LL dp[2010];

int main(){
    int n,t,c;
    scanf("%d",&n);
    for(int i = 1;i <= n;i++)
        dp[i] = (1LL<<30)*(1LL<<30);
    for(int i = 0;i < n;i++){
        scanf("%d%d",&t,&c);
        for(int j = n;j >= 0;j--)
            dp[min(j+t+1,n)] = min(dp[min(j+t+1,n)],dp[j]+c);
    }
    printf("%I64d\n",dp[n]);
}
