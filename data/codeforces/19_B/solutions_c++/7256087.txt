#include<cstdio>
#include<algorithm>
using namespace std;
int n,t,c,ans = -1;
long long dp[2001];
int main(){
  scanf("%d",&n);
  for(int i=0;i<=n;i++) dp[i] = 1ll<<60; dp[0] = 0ll;
  for(int i=0;i<n;i++){
    scanf("%d%d",&t,&c); t++;
    for(int j=n;j>=0;j--)
      dp[min(n,j+t)] = min(dp[min(n,j+t)],dp[j] + c);
  }
  printf("%I64d\n",dp[n]);
  
  return 0;
}
