#include<bits/stdc++.h>
using namespace std;
const int MAXN=2005;
int n;
long long int MAX=1e18+5;

long long ans[MAXN];
int main(){
	cin>>n;
	for(int i=1;i<MAXN;i++)
		ans[i]=MAX;
	long long time,cost;
	for(int  i=0;i<n;i++){
		cin>>time>>cost;
		time++;
		for(int j=n;j>0;j--)
	 		ans[j]=min(ans[j],ans[max(0LL,j-time)]+cost);
	}
	cout<<ans[n]<<endl;
	return 0;
}
