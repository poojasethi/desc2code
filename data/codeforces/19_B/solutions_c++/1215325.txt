#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,t[2010];
long long f[2010],c[2010];
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;++i){
        scanf("%d%I64d",&t[i],&c[i]);
        ++t[i];
    }
    memset(f,63,sizeof(f));f[0]=0;
    for(int i=1;i<=n;++i)
        for(int j=n;j>=1;--j)
            if(j<t[i])f[j]=min(f[j],c[i]);
            else f[j]=min(f[j],f[j-t[i]]+c[i]);
    printf("%I64d\n",f[n]);
    return 0;
}
