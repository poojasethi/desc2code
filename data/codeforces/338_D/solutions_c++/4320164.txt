#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;

int K;
ll N,M,X,Y,a[10005];

void init()
{
	scanf("%I64d%I64d%d",&N,&M,&K);
	for (int i=1; i<=K; i++) scanf("%I64d",&a[i]);
}

ll exgcd(ll a,ll b,ll&x,ll&y)
{
	if (!b) return x=1,y=0,a;
	ll g=exgcd(b,a%b,x,y),t=x;
	return x=y,y=t-a/b*y,g;
}

bool solve()
{
	ll b=0,m=a[1];
	for (int i=2; i<=K; i++)
	{
		ll x,y,d=1-i-b,g=exgcd(m,a[i],x,y);
		if (d%g) return -1;
		x*=d/g,y*=d/g,d=a[i]/g,x=(x%d+d)%d;
		if ((b=x*m+b)<0) return 0;
		if (m>N/d) return 0; else m*=d;
	}
	X=m,Y=!b?m:b;
	return 1;
}

bool check()
{
	if (!solve()||Y+K-1>M) return 0;
	for (int i=1; i<=K; i++) if (__gcd(X,Y+i-1)!=a[i]) return 0;
	return 1;
}

int main()
{
	init();
	puts(check()?"YES":"NO");
	return 0;
}
