#include <cstdio>
typedef long long ll;
int k;
ll n,m,a[10000],l,lcm,r;
inline ll mul(ll a,ll b,ll mo){
	ll s=1,ans=0;
	if(a<0) s*=-1,a*=-1;
	if(b<0) s*=-1,b*=-1;
	for(;b;b>>=1,(a<<=1)%=mo) if(b&1) (ans+=a)%=mo;
	return ans*s;
}
inline ll gcd(ll a,ll b){
	return b?gcd(b,a%b):a;
}
bool ok;
inline ll ex_gcd(ll a,ll &x,ll b,ll &y,ll c){
	if(!b){
		if(c%a)
			return ok=0;
		ok=1;
		x=c/a; y=0;
		return a;
	}
	ll g=ex_gcd(b,x,a%b,y,c);
	if(!ok) return 0;
	ll t=x; x=y; y=t-a/b*y;
	return g;
}
int main(){
	#define fail() { puts("NO"); return 0; }
	scanf("%I64d%I64d%d",&n,&m,&k);
	l=1;
	for(int i=0;i<k;i++){
		scanf("%I64d",a+i);
		ll x,y,g=ex_gcd(l,x,a[i],y,r+i);
		if(!ok) fail();
		double tmp=l/g*double(a[i]);
		if(tmp>n+1e-5) fail();
		lcm=tmp;
		r=((mul(-x,l,lcm)+r)%lcm+lcm)%lcm;
		if((r?r:r+lcm)+k-1>m) fail();
		l=lcm;
	}
	for(int i=0;i<k;i++) if(gcd(l,r+i)!=a[i]) fail();
	puts("YES");
}
