#include<iostream>
typedef long long int ll;
using namespace std;
ll a,b,c,d,l,r,t,i,j,z[10000],u_,u,uu,v_,v,vv,vvv,w;
int k;

ll gcd(ll o,ll p)
{
	return p==0?o:gcd(p,o%p);
}

ll exgcd(ll a,ll b,ll &x,ll &y)
{
    if(b==0)
	{
		x=1;
		y=0;
		return a;
	}
	else
	{
		ll aa=exgcd(b,a%b,x,y);
		ll xx=y,yy=x-(a/b)*y;
		x=xx;
		y=yy;
		return aa;
	}
}

int main()
{
	cin>>a>>b>>k;
	cin>>z[0];
	l=z[0];
	i=1;
	while(i<k)
	{
		cin>>z[i];
		d=gcd(l,z[i]);
		l*=z[i]/d;
		if(l<=0||l>a)
		{
			cout<<"NO";
			return 0;
		}
		i++;
	}
	u_=z[0];
	v_=0;
	vv=0;
	i=1;
	ll x,y;
	while(i<k)
	{
		u=u_;
		v=v_;
		uu=z[i];
		vvv=vv-v;
		d=exgcd(u,uu,x,y);
		if(vvv%d!=0)
		{
			cout<<"NO";
			return 0;
		}
		uu/=d;
		c=((x*vvv/d)%uu+uu)%uu;
		v_=u*c+v;
		u_=u*uu;
		i++;
	}
	t=(v_==0?l:v_);
	u_=z[0];
	v_=0;
	vv=0;
	i=1;
	while(i<k)
	{
		u=u_;
		v=v_;
		uu=z[i];
		vv--;
		vvv=vv-v;
		d=exgcd(u,uu,x,y);
		if(vvv%d!=0)
		{
			cout<<"NO";
			return 0;
		}
		uu/=d;
		c=((x*vvv/d)%uu+uu)%uu;
		v_=u*c+v;
		u_=u*uu;
		i++;
	}
	r=(v_==0?l:v_);
	if(r+k-1>b||r<0)
	{
		cout<<"NO";
		return 0;
	}
	i=r;
	j=r+k;
	while(i<j)
	{
		if(gcd(t,i)==z[i-r])
		i++;
		else
		{
			cout<<"NO";
			return 0;
		}
	}
	cout<<"YES";
	return 0;
}
