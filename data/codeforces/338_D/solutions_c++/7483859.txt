#include<cstdio>
#include<iostream>
using namespace std;
typedef long long LL;
LL n,m,k,T,a[100010],d,x,y,ans;
LL exgcd(LL a,LL b,LL &x,LL &y) {
	if (!b) { x=1, y=0; return a; }
	LL ret=exgcd(b,a%b,x,y), t=x;
	x=y, y=t-a/b*x;
	return ret;
}
int main()
{
	cin>>n>>m>>k>>a[0];
	ans=0,T=a[0];
	if (T>n)
		return puts("NO"),0;
	for (int i=1; i<k; ++i) {
		cin>>a[i];
		d=exgcd(T,a[i],x,y);
		if ((-i-ans)%d)
			return puts("NO"),0;
		x=(x*(-i-ans)/d)%(a[i]/d);
		ans+=x*T;
		T=T/d*a[i];
		if (T>n)
			return puts("NO"),0;
		ans=(ans%T+T)%T;
	}
	if (!ans) ans=T;
	if (ans+k-1>m)
		return puts("NO"),0;
	for (int i=0; i<k; ++i)
		if (exgcd(T,ans+i,x,y)!=a[i])
			return puts("NO"),0;
	puts("YES");
	return 0;
}
