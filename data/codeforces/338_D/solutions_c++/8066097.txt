#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define N 10010
using namespace std;
long long test,p[N],n,m,k,nowmodd,nowwhat,x,y;
long long ex_gcd(long long a,long long b)
{
    if (b==0)
    {
        x=1;
        y=0;
        return a;
    }
    long long temp=ex_gcd(b,a%b),t=x;
    x=y;
    y=t-a/b*y;
    return temp;
}


int main()
{
	cin>>n>>m>>k;
	for (long long i=1;i<=k;i++)
	scanf("%I64d",&p[i]);
	long long hehe=1;
	for (long long i=1;i<=k;i++)
	{
		long long GCD=ex_gcd(p[i],hehe);   
		if (p[i]/GCD*hehe>n) 
		{
			cout<<"NO"<<endl;
			return 0;	
		}
		hehe=p[i]/GCD*hehe;
	}
	nowmodd=p[1];
 	nowwhat=p[1]-1;
	for (long long i=2;i<=k;i++)
	{
		long long GCD=ex_gcd(nowmodd,p[i]),temp=-i;
		if ((temp-nowwhat)%GCD)
		{
			cout<<"NO"<<endl;
			return 0;
		}
		x=((x*(temp-nowwhat)/GCD)%(p[i]/GCD)+(p[i]/GCD))%(p[i]/GCD);
		nowwhat=x*nowmodd+nowwhat;
		nowmodd=nowmodd/GCD*p[i];
	}
	for (int i=1;i<=k;i++)
	if (ex_gcd(nowwhat+i,nowmodd)!=p[i])
	{
		cout<<"NO"<<endl;
		return 0;
	}
	if (nowwhat+k>m) 
	{
		cout<<"NO"<<endl;
		return 0;
	}
	cout<<"YES"<<endl;
    return 0;       
}
