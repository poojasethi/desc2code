#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

typedef __int64 ll;
ll n, m, k, a[100010];
ll exgcd(ll a, ll b, ll &x, ll &y)
{
    if(b == 0){
		y = 0; x = 1;
        return a;
    }
    ll g = exgcd(b, a%b, x, y), t = x;
    x = y; y = (t-a/b*x);
    return g;
}

int main()
{
    ll x, y;
    int i;
    while(scanf("%I64d %I64d %I64d", &n, &m, &k) != EOF){
        for(int i = 0; i < k; i++) scanf("%I64d", a+i);
        ll lcm = a[0], ans = 0;
        for(i = 1; i < k; i++){
            int d = exgcd(lcm, a[i], x, y);
            if((-ans-i)%d) break;
            x = (x*(-i-ans)/d)%(a[i]/d);
            ans += x*lcm;
            lcm = lcm/d*a[i];
            if(lcm > n) break;
            ans = (ans%lcm+lcm)%lcm;
        }
        if(ans == 0) ans = lcm; 
        if(i < k || ans+k-1 > m){puts("NO"); continue;}
        for(i = 0; i < k; i++){
            if(exgcd(lcm, ans+i, x, y) != a[i]) 
                break;
        }
        if(i == k) puts("YES");
        else puts("NO");
    }
	return 0;
}
