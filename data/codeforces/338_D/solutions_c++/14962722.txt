#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#define N 110000
using namespace std;
typedef long long ll;
ll a[N];
ll exgcd(ll a,ll b,ll &x,ll &y)
{
	if(!b)
	{
		x=1,y=0;
		return a;
	}
	else
	{
		ll temp=exgcd(b,a%b,y,x);
		y-=a/b*x;
		return temp;
	}
}
int main()
{
	ll n,m,h,l,tot,i,A,c,d,M,x,y;
	cin>>n>>m>>tot;
	for(i=1;i<=tot;i++)
	{
		cin>>a[i];
	}
	for(i=1,h=1;i<=tot;i++)
	{
		if((long double)h/__gcd(h,a[i])*a[i]>n)
		{
			puts("NO");
			return 0;
		}
		h=h/__gcd(h,a[i])*a[i];
	}
	if(h>n)
	{
		puts("NO");
		return 0;
	}
	l=0,A=a[1];
	for(i=2;i<=tot;i++)
	{
		c=1-i-l;
		d=exgcd(A,a[i],x,y);
		if(c%d)
		{
			puts("NO");
			return 0;
		}
		M=a[i]/d;
		x=(c/d*x%M+M)%M;
		l=l+A*x;
		A=A/d*a[i];
		l%=A;
	}
	if(l==0)
	{
		l+=A;
	}
	if(l+tot-1>m)
	{
		puts("NO");
		return 0;
	}
	for(i=0;i<tot;i++)
	{
		if(__gcd(h,i+l)!=a[i+1])
		{
			puts("NO");
			return 0;
		}
	}
	puts("YES");
	return 0;
}
