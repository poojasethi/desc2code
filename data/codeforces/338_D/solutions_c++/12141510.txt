#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=1e4+5;
const long long eps=-1;
long long n,m,k,lc;
long long a[N],d,x,y,ans;
long long exgcd(long long a,long long b,long long &x,long long &y)
{
	if(!b)
	{
		x=1;y=0;
		return a;
	}
	long long ret=exgcd(b,a%b,x,y),t=x;
	x=y,y=t-a/b*x;
	return ret;
}
int main()
{
	scanf("%I64d%I64d%I64d%I64d",&n,&m,&k,&a[0]);
	ans=0,lc=a[0];
	if(lc>n) return puts("NO"),0;
	for(int i=1;i<k;i++)
	{
		scanf("%I64d",&a[i]);
		d=exgcd(lc,a[i],x,y);
		if(eps*(i+ans)%d)  return puts("NO"),0;
		x=(x*eps*(i+ans)/d)%(a[i]/d);
		ans+=x*lc;
		lc=lc/d*a[i];
		if(lc>n) return puts("NO"),0;
		ans=(ans%lc+lc)%lc;
	}
	if(!ans) ans=lc;
	if(ans+k-1>m) return puts("NO"),0;
	for(int i=0;i<k;i++) if(exgcd(lc,ans+i,x,y)!=a[i]) return puts("NO"),0;
	return puts("YES"),0;
	
}
