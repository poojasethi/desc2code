#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
#define ll long long 
ll n,m,k,b[11000],H;
ll gcd(ll a,ll b){
   if(!b)return a;
   return gcd(b,a%b);
}
void exgcd(ll a,ll b,ll &x,ll &y){
     if(!b){x=1;y=0;return;}
     ll xx,yy;
     exgcd(b,a%b,xx,yy);
     x=yy;y=xx-(a/b)*yy;
}
int main(){
    cin>>n>>m>>k;
    H=1;
    for(int i=1;i<=k;i++){
      cin>>b[i];
      ll t=gcd(b[i],H);
      H/=t;
      if((double)H*b[i]>n){
        printf("NO\n");
        return 0;
      }
      H*=b[i];
      }
    ll ans=b[1]-1,A=b[1];
    for(int i=2;i<=k;i++){
      ll B=b[i],C=-i-ans,t=gcd(A,B);
      if(C%t){
        printf("NO\n");
        return 0;
      }
      A/=t;B/=t;C/=t;
      ll x,y;
      exgcd(A,B,x,y);
      x=x*C%B;
      A*=t;B*=t;
      ans=ans+x*A;
      A=A/gcd(A,B)*B;
      ans=(ans%A+A)%A;
    }
    if(ans<0)ans+=A;
    for(int i=1;i<=k;i++)
      if(gcd(H,ans+i)!=b[i] || ans+i>m){
        printf("NO\n");
        return 0;
      }
    printf("YES\n");
    return 0;
}
