#include <cstdio>
#define rec(i, x, n) for(int i = x; i <= n; i++)
typedef long long LL;
const int maxn = 10005;
int k;
LL n, m, num[maxn], row = 1, col;
inline LL gcd(LL a, LL b) {return b ? gcd(b, a % b) : a;}
inline LL lcm(LL a, LL b) {return a / gcd(a, b) * b;}
inline void exgcd(LL a, LL b, LL &d, LL &x, LL &y) {b ? (exgcd(b, a % b, d, y, x), y -= a / b * x) : (d = a, x = 1, y = 0);}
int main() {
	scanf("%I64d%I64d%d", &n, &m, &k);
	rec(i, 1, k) scanf("%I64d", &num[i]), row = lcm(row, num[i]);
	if(row <= 0 || row > n) {printf("NO\n"); return 0;}
	LL m1 = num[1], r1 = 0;
	rec(i, 2, k) {
		LL m2 = num[i], r2 = -i + 1, d, x, y;
		exgcd(m1, m2, d, x, y);
		if((r2 - r1) % d) {printf("NO\n"); return 0;}
		x = ((r2 - r1) / d * x) % (m2 / d);
		r1 = m1 * x + r1; m1 = lcm(m1, m2);
		col = r1 = (r1 % m1 + m1) % m1;
	}
	if(col == 0) col = row;
	if(col < 0 || col + k - 1 > m) {printf("NO\n"); return 0;}
	rec(i, 1, k) if(gcd(row, col + i - 1) != num[i]) {printf("NO\n"); return 0;}
	printf("YES\n");
	return 0;
}
