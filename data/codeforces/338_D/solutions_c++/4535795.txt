#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long LL;
#define rep(i,n) for (LL i=0;i<n;++i)
LL n,m,k,x,y,L=1,g,t,ans,a[10005];
void exgcd(LL a,LL b,LL &x,LL &y,LL &g){if (!b) x=1,y=0,g=a; else exgcd(b,a%b,y,x,g),y-=a/b*x;}
bool check(LL x){rep(i,k) if (__gcd(L,x+i)!=a[i]) return 0; return x+k-1<=m;}
int main()
{
    scanf("%I64d%I64d%I64d",&n,&m,&k);
    rep(i,k){
        scanf("%I64d",a+i),g=__gcd(L,a[i]);
        if (L>n/(a[i]/g) || L>m/(a[i]/g)) return puts("NO"),0; L=L*a[i]/g;
        }
    t=L;
    for (LL i=2;i<=t;++i){
        if (i*i>t) i=t;
        if (t%i==0){
            LL p=1; while (t%i==0) p*=i,t/=i;
            rep(j,k) if (a[j]%p==0){exgcd(p,L/p,x,y,g),ans=(ans-y*(L/p)%L*j)%L; break;}
        }
    }
    puts(check((ans+L)%L)?"YES":"NO"); return 0;
}
