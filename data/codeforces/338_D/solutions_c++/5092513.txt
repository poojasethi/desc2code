#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
using namespace std;

long long a[10010];

long long ExGcd(long long a, long long b, long long &x, long long &y)
{
	if (b == 0) { x = 1; y = 0; return a; }
	long long ret = ExGcd(b, a % b, y, x);
	y -= a/b*x;
	return ret;
}
int main()
{
	//freopen("input.txt", "r", stdin);
	int k;
	long long n, m;
	cin >> n >> m >> k;
	for (int i = 0; i < k; i++) cin >> a[i];
	long long lcm = 1;
	for (int i = 0; i < k; i++) {
		lcm /= __gcd(lcm, a[i]);
		if (lcm <= n/a[i]) lcm *= a[i];
		else {
			puts("NO");
			return 0;
		}
	}
	long long p = 1, q = 0;
	for (int i = 0; i < k; i++) {
		long long p_ = a[i], q_ = (k-1-i) % a[i];
		if ((q_-q) % __gcd(p, p_) == 0) {
			long long x, y;
			long long gcd = ExGcd(p, p_, x, y);
			x *= (q_-q)/gcd;
			long long del = p_/gcd;
			if (x < 0) {
				long long ret = (-x)/del;
				x += ret*del;
				while (x < 0) x += del;
			}
			x %= del;
			q = x*p+q;
			p = p/gcd*p_;
		}
		else {
			puts("NO");
			return 0;
		}
	}
	while (q < k) q += p;
	if (q > m) {
		puts("NO");
		return 0;
	}
	for (int i = 0; i < k; i++) {
		if (__gcd(lcm, q-k+1+i) != a[i]) {
			puts("NO");
			return 0;
		}
	}
	puts("YES");
	return 0;
}
