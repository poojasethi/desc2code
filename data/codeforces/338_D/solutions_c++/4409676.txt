#include<cstdio>
#define ll long long
using namespace std;
ll n,m,a[20000],lcm,tmp,x0,tx,ty,g,A,B,C;
ll k,i;
void exgcd(ll a,ll b,ll &x,ll &y)
{
	if (!b) {x=1; y=0; return;}
	exgcd(b,a%b,x,y);
	int t=x;
	x=y;
	y=t-(a/b)*y;
}
ll gcd(ll a,ll b)
{if (!b) return a;return gcd(b,a%b);}
ll mul(ll a,ll b,ll M,ll s=0)
{
	for(;b;b/=2,a=(a+a)%M)
		if (b&1) s=(s+a)%M;
	return s;
}
int main()
{
	//freopen("D.in","r",stdin);
	//freopen("D.out","w",stdout);
	scanf("%I64d %I64d %I64d",&n,&m,&k);
	for(i=1;i<=k;i++) scanf("%I64d",&a[i]);
	for(lcm=1,i=1;i<=k;i++)
		{
			g=gcd(lcm,a[i]);
			if (a[i]>n) {printf("NO"); return 0;}
			if ((lcm/g)>(n/a[i]))
				{printf("NO"); return 0;}
			lcm=lcm/g*a[i];
		}
	x0=0; tmp=a[1];
	for(i=2;i<=k;i++)
		{
			A=tmp%a[i];
			B=a[i];
			C=((-i+1-x0)%a[i]+a[i])%a[i];
			g=gcd(A,B);
			if (C%g)
				{printf("NO"); return 0;}
			C/=g;
			exgcd(A,B,tx,ty);
			tx=(tx%a[i]+a[i])%a[i];
			x0=x0+mul(tx,C,a[i])*tmp;
			g=gcd(tmp,a[i]);
			tmp=tmp/g*a[i];
		}
	if (x0==0) x0=tmp;
	if (x0+k-1>m) {printf("NO"); return 0;}
	for(i=1;i<=k;i++)
		if (gcd(lcm,x0+i-1)!=a[i])
			{printf("NO"); return 0;}
	printf("YES");
	return 0;
}
