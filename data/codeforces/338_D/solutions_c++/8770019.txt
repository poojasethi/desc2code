#include <cstdio>
#include <cstdlib>
using namespace std;
typedef long long ll;
const int N=10004,M=1000000;
int K;
ll n,m,L=1,a[N],l[N];
void error()
{
	printf("NO\n");
	exit(0);
}
ll exgcd(ll a,ll b,ll &x,ll &y)
{
	if (!b)
	{
		x=1; y=0;
		return a;
	}
	ll ret=exgcd(b,a%b,x,y),t=x;
	x=y; y=t-(a/b)*y;
	return ret;
}
ll mul(ll a,ll b,ll P)
{
	ll u=a/M,v=a%M,x=b/M,y=b%M;
	return ((u*x*M+v*x+u*y)%P*M+v*y)%P;
}
int main()
{
	int i; ll x,y;
	scanf("%I64d%I64d%d",&n,&m,&K);
	for (i=1;i<=K;i++)
	{
		scanf("%I64d",&a[i]);
		ll t=exgcd(L,a[i],x,y);
		if (L>n/(a[i]/t)) error();
		L=L/t*a[i];
		l[i]=a[i];
	}
	ll r=0;
	for (i=1;i<K;i++)
	{
		ll t=exgcd(a[i],a[i+1],x,y),tr=-r+(a[i+1]-i%a[i+1])%a[i+1];
		if (x<0) x=(a[i+1]-(-x)%a[i+1])%a[i+1];
		if (tr%t) error();
		if (tr<0) tr+=a[i+1]/t*a[i];
		x=mul(x,tr/t,a[i+1]);
		a[i+1]=a[i+1]/t*a[i];
		r=(mul(x,a[i],a[i+1])+r)%a[i+1];
	}
	if (!r) r=L;
	if (r>m-K+1) error();
	for (i=1;i<=K;i++) if (exgcd(L,r+i-1,x,y)!=l[i]) error();
	printf("YES\n");
	return 0;
}
