#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define int64 long long
using namespace std;
int64 n,m,ans,v,vv,x,a[12000],b[12000],c[12000];
int k,i,j;
void fail(){
	printf("NO\n");
	exit(0);
}
int64 pow(int64 x,int64 y,int64 mo){
	int64 res=1;
	while(y){
		if(y&1)res=res*x%mo;
		x=x*x%mo;
		y/=2;
	}
	return res;
}
void work(int64 i){
	x=-1;
	for(j=1;j<=k;++j){
		c[j]=1;
		while(a[j]%i==0)a[j]/=i,c[j]*=i;
		if(x==-1 || c[j]>c[x])x=j;
	}
	for(j=1;j<=k;++j)if(__gcd(abs(x-j),c[x])!=c[j])fail();
	ans+=pow(vv/c[x],c[x]/i*(i-1),vv)*(b[x]%c[x])%vv;
	v/=c[x];
}
int main(){
	scanf("%I64d%I64d%d",&n,&m,&k);
	v=1;
	for(i=1;i<=k;++i){
		scanf("%I64d",&x);
		if(v/__gcd(v,x)>n/x)fail();
		v=v/__gcd(v,x)*x;
		a[i]=x;
		b[i]=-i;
	}
	ans=0;
	vv=v;
	for(i=2;(int64)i*i<=v;++i)if(v%i==0)work(i);
	if(v!=1)work(v);
	ans=(ans%vv+vv)%vv;
	if(ans+k<=m)printf("YES\n");
	else printf("NO\n");
}
