#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define N 10010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k;
int64 n,m,a[N],lcm,Lcm,x,y,a1,a2,b1,b2;
int64 exgcd(int64 a,int64 b) {
	if (!b) {
		x=1,y=0; return a;
	}
	int64 d=exgcd(b,a%b),z=x;
	x=y,y=z-a/b*x;
	return d;
}
int main() {
	//freopen("338d.in","r",stdin);
	//freopen("338d.out","w",stdout);
	scanf("%I64d%I64d%d",&n,&m,&k);
	lcm=1;
	For(i,1,k) {
		scanf("%I64d",&a[i]);
		int64 A=__gcd(lcm,a[i]);
		if (lcm/A>n/a[i]) {
			printf("NO\n"); return 0;
		}
		lcm=lcm/A*a[i];
	}
	m=min(m-k+1,lcm);
	a1=a[1]; b1=0;
	For(i,2,k) {
		a2=a[i]; b2=-i+1;
		int64 d=exgcd(a1,a2),t=b2-b1;
		if (t%d) {
			printf("NO\n"); return 0;
		}
		int64 q=a2/d;
		x=(x*(t/d)%q+q)%q;
		b1=(b1+a1*x)%lcm;
		a1=a1/d*a2%lcm;
	}
	if (b1<=0) b1+=lcm;
	if (b1>m) {
		printf("NO\n"); return 0;
	}
	For(i,1,k) if (__gcd(lcm/a[i],(b1+i-1)/a[i])>1) {
		printf("NO\n"); return 0;
	}
	printf("YES\n");
	return 0;
}
