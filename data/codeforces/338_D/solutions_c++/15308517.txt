#include<iostream>
#include<stdio.h>
#define LL long long
using namespace std;
const int maxn=10005;
LL n,m,a[maxn],b[maxn];
int k;
LL gcd(LL a,LL b)
{
    if(!b)return a;
    return gcd(b,a%b);
}
LL exgcd(LL a,LL b,LL &x,LL &y)
{
    if(!b)
    {
        x=1,y=0;
        return a;
    }
    LL r=exgcd(b,a%b,x,y),x0=x;
    x=y;
    y=x0-(a/b)*y;
    return r;
}
LL china(LL *m,LL *a)
{
    LL lcm=1,X=m[1],Y=a[1];
    for(int i=1;i<=k;i++)
	lcm=lcm/gcd(lcm,m[i])*m[i];
	for(int i=2;i<=k;i++)
    {	
		LL A=X,B=m[i],d,x,y,c=a[i]-Y;	
		d=exgcd(A,B,x,y);	
		if(c%d)
		return -1;		
		LL mod=m[i]/d;
		LL K=((x*c/d)%mod+mod)%mod;
		Y=X*K+Y;
		X=X*m[i]/d;
    }
    if(Y==0)
	return lcm;
	return Y;
}
bool check()
{
    LL I=china(a,b),J;
    if(I>n || I<=0)return false;
    for(int i=1;i<=k;i++)b[i]=1-i;
    J=china(a,b); 
	if(J+k-1>m || J<=0)return false;
	for(int i=1;i<=k;i++)if(gcd(I,J+i-1)!=a[i])return false;
    return true;
}
int main()
{
    cin>>n>>m>>k;
	for(int i=1;i<=k;i++)
    cin>>a[i];
	if(check())cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    return 0;
}
