#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
#define ll long long
int k;
ll X, Y, n, m, a[10010];

void inti()
{
    cin>>n>>m>>k;
    for (int i=1; i<=k; i++)
        cin>>a[i];
}

ll exgcd(ll a, ll b, ll &x, ll&y)
{
    if (!b) return x=1,y=0,a;
    ll d=exgcd(b, a % b, x, y), t=x;
    return x=y, y=t-a/b*y, d;
}

bool solve()
{
    ll x, y, tmp=0, M=a[1];
    for (int i=2; i<=k; i++)
    {
        ll d=1-i-tmp, g=exgcd(M,a[i],x,y);
        if (d % g) return 0;
        x*=d/g;
        y*=d/g;
        d=a[i]/g;
        x=(x%d+d)%d;
        if ((tmp=tmp+M*x) < 0)
            return 0;
        if (M>n/d) return 0;else M*=d;
    }
    X=M;
    Y=tmp?tmp:M;
    return 1;
}

bool check()
{
    ll x, y;
    if (!solve() || (Y+k-1>m)) 
        return 0;
    for (int i=1;i<=k; i++)
        if (exgcd(X, Y+i-1, x, y)!=a[i])
        return 0;
    return 1;
}
int main()
{
    inti();
    if (check()) cout<<"YES"<<endl;
    else cout<<"NO"<<endl;

}
