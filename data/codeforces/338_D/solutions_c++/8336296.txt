#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
using namespace std;
typedef long long LL;
const int Maxk=10005;
LL n,m,a[Maxk],k;

LL exgcd(LL a,LL b,LL &x,LL &y) {
    if (b) {
        LL res=exgcd(b,a%b,y,x);
        y-=a/b*x;
        return res;
    }
    x=1; y=0;
    return a;
}

LL gcd(LL a,LL b) {
    return b?gcd(b,a%b):a;
}

int main() {
    cin>>n>>m>>k;
    for (int i=1;i<=k;i++) cin>>a[i];
    LL cur=0,mod=a[1];
    if (mod>n) {puts("NO"); return 0;}
    for (int i=2;i<=k;i++) {
        LL now=1-i,x,y,g=exgcd(mod,a[i],x,y);
        if ((cur-now)%g) {puts("NO"); return 0;}
        if (n/mod<a[i]/g) {puts("NO"); return 0;}
        cur+=mod*((now-cur)/g%a[i]*x%a[i]);
        cur%=mod*=a[i]/g;
    }
    if ((cur<=0?cur+=mod:cur)+k-1>m) {puts("NO"); return 0;}
    for (int i=1;i<=k;i++)
        if (gcd(mod,cur+i-1)!=a[i]) {
            puts("NO");
            return 0;
        }
    puts("YES");
    return 0;
}
