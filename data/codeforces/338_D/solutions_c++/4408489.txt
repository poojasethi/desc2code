#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#define int64 long long
using namespace std;

int64 n, m, k, v = 1;
int64 a, A[10010];

int64 gcd(int64 x, int64 y)
{
	int64 t;
	while (t = x, y) x = y, y = t % x;
	return x;
}
int64 exeuclid(int64 x, int64 y, int64 w)
{
	if ((w = (w % y + y) % y) % x == 0)  return w / x;
	if ((y + w) % x == 0)  return (y + w) / x;
	return (exeuclid(y % x, x, -w) * y + w) / x;
}
bool check(int64 x, int64 y)
{
	for (int i = 1; i <= k; ++i)
		if (gcd(x, y + i - 1) != A[i])  
      return 0;

	return 1;
}
int main()
{
	//freopen("D.in", "r", stdin);
	//freopen("D.out", "w", stdout);

	cin >> n >> m >> k;

  int64 p = 1, b = 0, B;
	for (int i = 1; i <= k; ++i)  {
		cin >> a;  A[i] = a;
		
		int64 d = gcd(v, a);
		if (n / a < v / d)  {  printf("NO\n");  exit(0);  }
		B = ((-i) % a + a - b % a + a) % a;
		if (B % gcd(p, a) != 0)  {  printf("NO\n");  exit(0);  }

		b = exeuclid(p, a, B) * v + b;
		if (b > m)  {  printf("NO\n");  exit(0);  }
		p = v *= a / d;
	}
  
  b++;
	if (v > n  ||  b + k - 1 > m  ||  !check(v, b))  {  printf("NO\n");  exit(0);  }
	printf("YES\n");
}
