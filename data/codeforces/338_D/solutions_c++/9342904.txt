#include <cstdio>
#include <iostream>
#define INF 2147483647
#define LL long long
using namespace std;
LL n,m,k,A[10001];
LL gcd(LL a,LL b){return !b ? a:gcd(b,a%b);}
void exgcd(LL a,LL b,LL &x,LL &y){
	if(!b){x=1,y=0;return;}
	exgcd(b,a%b,y,x);
	y-=a/b*x;
}
int main(){
	scanf("%lld%lld%lld",&n,&m,&k);
	for(int i=1;i<=k;++i)scanf("%lld",&A[i]);
	LL r=0,p=A[1];
	for(int i=1;i<k;++i){
		//j=r(mod p)
	    //j=-i(mod A[i+1])
	    //j=px+r=A[i+1]*y-i;
		//px-A[i+1]*y=-r-i;
		//px=-r-i(mod A[i+1])
		LL g=gcd(p,-A[i+1]),x,y;
		if((-r-i)%g){printf("NO");return 0;}
		exgcd(p,-A[i+1],x,y);
		x*=(-r-i)/g;
		LL t=(-A[i+1])/g;
		x%=t;
		//合并为j=px+r(mod lcm(p,A[i+1])) 
	    r+=p*x;
		p=p/gcd(p,A[i+1])*A[i+1];
	}
	while(r<=0)r+=p;
	if(p<=0 || p>n){printf("NO");return 0;}
	if(r<=0 || r+k-1>m){printf("NO");return 0;}
	for(int i=0;i<k;++i)
	    if(gcd(p,r+i)!=A[i+1]){printf("NO");return 0;}
	printf("YES");
}
