#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

const int base=107,mod=998244353;
int n,power[1000005],hash1,hash2;
char s1[1000005],s2[1000005];
int main() {
	scanf("%d",&n);
	n--;
	scanf("%s",s1+1);
	scanf("%s",s2+1);
	for (int i=1;i<=n;i++) {
		if (s2[i]=='N') s2[i]='S';
		else if (s2[i]=='S') s2[i]='N';
		else if (s2[i]=='W') s2[i]='E';
		else if (s2[i]=='E') s2[i]='W';
	}
	power[0]=1;
	for (int i=1;i<=n;i++) power[i]=(long long)power[i-1]*base%mod;
	for (int i=n;i;i--) {
		hash1=(hash1+(long long)s1[i]*power[n-i]%mod)%mod;
		hash2=((long long)hash2*base%mod+(int)s2[i])%mod;
		if (hash1==hash2) {
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	return 0;
}
