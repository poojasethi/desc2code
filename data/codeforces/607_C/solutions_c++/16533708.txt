#include <cstdio>
const long long seed=131;
const int N=1000010;
char ac[N],bc[N];
int main(){
    int n;
    scanf("%d%s%s",&n,ac,bc),n--;
    for(int i=0;i<n;i++){
        if(bc[i]=='N') bc[i]='S';
        else if(bc[i]=='S') bc[i]='N';
        else if(bc[i]=='E') bc[i]='W';
        else bc[i]='E';
    }
    unsigned long long ha=0,hb=0,pow=1;
    for(int l=1;l<=n;l++){
        ha+=ac[n-l]*pow;
        hb*=131,hb+=bc[n-l];
        pow*=131;
        if(ha==hb){
            printf("NO\n");
            return 0;
        }
    }
    printf("YES\n");
    return 0;
}
