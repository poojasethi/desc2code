#include <bits/stdc++.h>
using namespace std;
char s1[1123456],s2[1123456];
char str[2123456];
int n;

char getInv(char c){
	char aux[] = {'N','E','S','W'};
	for(int i=0;i<4;i++)
		if(aux[i]==c)
			return aux[(i + 2)%4];
	return '-';
}



int main(){
	scanf("%d", &n);n--;
	scanf("%s",s1);
	scanf("%s",s2);
	for(int i=0;i<n;i++){
		str[i] = s1[n-i-1];
		str[n+i] = getInv(s2[i]);
	}

	int m = 2* n;
	vector<int> F (m+1 , 0);

	for(int i=2;i<=m;i++){
		int j = F[i-1];
		while(1){
			if(str[j] == str[i-1]){
				F[i] = j+1;
				break;
			}
			if(j ==0){
				F[i] = 0;
				break;
			}
			j = F[j];
		}
	}
	if(F[m] == 0){
		printf("YES");
	}else{
		printf("NO");
	}
}
