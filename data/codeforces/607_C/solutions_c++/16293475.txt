#include <iostream>
#include <algorithm>

using std::cin;
using std::cout;
using std::endl;

typedef long long ll;

const int t = 37;
const int Maxn = (int)1e6 + 10;
int n, ha = 0, hb = 0, p = 1;
char s1[Maxn], s2[Maxn], c[100];

int main()
{
#ifndef ONLINE_JUDGE
	freopen("C.in", "r", stdin);
#endif
	std::ios::sync_with_stdio(false);

	cin >> n >> s1 + 1 >> s2 + 1;

	c['E'] = 'W'; c['W'] = 'E'; c['N'] = 'S'; c['S'] = 'N';

	std::reverse(s1 + 1, s1 + n);
	for(int i = 1; i < n; ++i)
		s1[i] = c[s1[i]];

	for(int i = 1; i < n; ++i)
	{
		ha = ha + s1[i] * p;
		hb = hb * t + s2[n - i];
		p *= t;
		if(ha == hb)
		{
			cout << "NO" << endl;
			return 0;
		}
	}
	cout << "YES" << endl;
	return 0;
}
