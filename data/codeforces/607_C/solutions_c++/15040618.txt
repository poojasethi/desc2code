#include <iostream>
#include <map>
using namespace std;
long long n,pow[1000005],aa[1000005],bb[1000005];
char a[1000005],b[1000005];
map<char,int> mp;
int main() {
    cin>>n;n--;
    pow[0]=1;
    mp['N']=0;
    mp['E']=1;
    mp['S']=2;
    mp['W']=3;
    for(int i=1;i<=n;i++){
        pow[i]=(12341243*pow[i-1])%1000000007;
    }
    cin>>a>>b;
    for(int i=1;i<=n;i++){
        aa[i]=(aa[i-1]+(mp[a[i-1]]+1)*pow[i])%1000000007;
    }
    for(int i=1;i<=n;i++){
        bb[i]=(bb[i-1]+((mp[b[n-i]]+2)%4+1)*pow[i])%1000000007;
    }
    for(int i=1;i<=n;i++){
        if((bb[i]*pow[n-i]-aa[n]+aa[n-i])%1000000007==0){
            cout<<"NO";return 0;
        }
    }
    cout<<"YES";
}
