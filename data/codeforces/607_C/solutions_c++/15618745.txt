#include<bits/stdc++.h>
using namespace std;
int const mod = 1e9+7;
int bas = 27;
int powmod(int x,int y)
{
    if(y==0)return 1;
    if(y==1)return x;
    unsigned long long ret = powmod(x,y/2);
    ret = (1ULL*ret*ret)%mod;
    if(y%2)ret*=1ULL*x;
    return ret%mod;
}
char swa(char x)
{
    if(x=='W')return 'E';
    if(x=='E')return 'W';
    if(x=='N')return 'S';
    if(x=='S')return 'N';
}
int main()
{
    int n;cin >> n;
    string s,ss;cin >> s >> ss;
    for(int i = 0;i<ss.size();i++)ss[i]=swa(ss[i]);
    reverse(ss.begin(),ss.end());
    reverse(s.begin(),s.end());
    long long hash1=0,hash2=0;
    for(int i = 0;i<n;i++)
    {
        hash1*=bas;
        hash1%=mod;
        hash1+=(ss[i]-'A');
        hash1%=mod;
        hash2 = (1LL*(s[i]-'A')*powmod(bas,i))%mod+hash2;hash2%=mod;
        if(hash1==hash2){puts("NO");return 0;}
    }
    puts("YES");
}
