#include<stdio.h>
#include<string.h>
const int maxn=1000000+10;
long long mod1=1000000007;
long long mod2=1000000009;
char a[maxn],b[maxn];
inline int ida(char c)
{
	if (c=='N') return 0;
	if (c=='E') return 1;
	if (c=='S') return 2;
	if (c=='W') return 3;
}
inline int idb(char c)
{
	if (c=='N') return 2;
	if (c=='E') return 3;
	if (c=='S') return 0;
	if (c=='W') return 1;
}
int main()
{
	int n;
	scanf("%d",&n);
	scanf("%s",a);scanf("%s",b);
	long long l=0,r=0,p=1;
	long long l0=0,r0=0,p0=1;
	for (int i=n-2;i>=0;i--)
	{
		l=(l*4+ida(a[i]))%mod1;
		r=(r+idb(b[i])*p)%mod1;
		p=(p*4)%mod1;
		
		l0=(l0*4+ida(a[i]))%mod2;
		r0=(r0+idb(b[i])*p0)%mod2;
		p0=(p0*4)%mod2;
		
		if (l==r&&l0==r0)
		{
			printf("NO\n");
			return 0;
		}
	}
	
	printf("YES\n");
	return 0;
}
