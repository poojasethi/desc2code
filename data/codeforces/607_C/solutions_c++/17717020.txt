#include<bits/stdc++.h>
using namespace std;
#define maxn 1000005
string s1,s2;
int next[maxn*2];
int main()
{
	int n;
	while(cin>>n){
		cin>>s1>>s2;
		reverse(s2.begin(),s2.end());
		n--;
		for(int i=0;i<n;i++){
			if(s2[i]=='E')s2[i]='W';
			else if(s2[i]=='W')s2[i]='E';
			else if(s2[i]=='S')s2[i]='N';
			else s2[i]='S';
		}
		string s=s2+s1;
	//	cout<<s<<endl;
		next[0]=-1;
		int i=1;
		while(i<2*n){
			int j=next[i-1];
			next[i]=j+1;
			while(s[j+1]!=s[i]){
				if(j==-1){
					next[i]=-1;
					break;
				}
				j=next[j];
				next[i]=j+1;
			}
		
			i++;
		}
	//	cout<<next[2*n-1]<<endl;
		if(next[2*n-1]>=0)cout<<"NO"<<endl;
		else cout<<"YES"<<endl;
	}
	return 0;
}
