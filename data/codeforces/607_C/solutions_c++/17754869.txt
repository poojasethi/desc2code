#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=2000000+10;
int n,fail[maxn];
char s[maxn],r[128];
inline void Get_Fail()
{
	int i=1,j=-1,len=n;
	fail[0]=-1;
	while (i!=len)
		if (j==-1||s[i]==s[j])
			fail[++i]=++j;
		else
			j=fail[j];
}
int main()
{
	scanf("%d\n",&n);n-=2;
	r['N']='S',r['S']='N',r['E']='W',r['W']='E';
	for (int i=n;i>=0;i--)
		s[i]=r[getchar()];
	s[n+1]=' ';
	scanf("%s",s+n+2);
	n=strlen(s);
	Get_Fail();
	if (fail[n])
		puts("NO");
	else
		puts("YES");
	return 0;
}
