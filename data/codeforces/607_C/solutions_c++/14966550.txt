#include<stdio.h>
#include<algorithm>
#define N 1000005
using namespace std;
typedef unsigned long long ULL;
ULL base=2333ll,Pow[N],L,R;
char A[N],B[N];int i,id,n,cnt;
char INV(char ch){
    if (ch=='N') return 'S';
    if (ch=='S') return 'N';
    if (ch=='W') return 'E';
    if (ch=='E') return 'W';
}
int main(){
    scanf("%d",&n);n--;
    scanf("%s",A+1);
    scanf("%s",B+1);
    for (i=1;i<=n;i++)
        B[i]=INV(B[i]);
    reverse(A+1,A+n+1);
    for (Pow[0]=i=1;i<=n;i++)
        Pow[i]=1ll*Pow[i-1]*base;
    L=R=0;
    for (i=0;i<n;i++){
        L+=Pow[i]*A[i+1];
        R*=base;R+=B[n-i];
        if (L==R) return puts("NO");
  }puts("YES");
}
