#include<iostream>
#include<string>
using namespace std;
string ori,re;
inline char inv(char x)
{
//	cout<<x<<endl;
	if(x=='N')return 'S';
	if(x=='E')return 'W';
	if(x=='W')return 'E';
	if(x=='S')return 'N';
}
int n;
const long long mod=1000000007;
const long long mul=131;
long long pow,hash1,hash2;
int main()
{
	cin>>n;
	n--;
	cin>>ori;
	ori=" "+ori;
	cin>>re;
	re=" "+re;
	for (int i=1;i<=n;i++)
	{
		re[i]=inv(re[i]);
	//	cout<<i<<' '<<re[i]<<endl;
	}
	pow=1;
	for (int i=n;i>=1;i--)
	{
	//	cout<<ori[i]<<'-'<<re[i]<<endl;;
		hash1=hash1*mul+ori[i];
		hash2=hash2+re[i]*pow;
		pow*=mul;
		if(hash1==hash2)
		{
			cout<<"NO"<<endl;
			return 0;
		}
	}cout<<"YES"<<endl;
	return 0;
}
