#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
typedef unsigned long long lag;
const int N = 1000005;

char s[N], t[N];
char opp[256];
lag dp[N], thirtyone[N];
int n;
int main() {
	opp['N'] = 'S';
	opp['S'] = 'N';
	opp['E'] = 'W';
	opp['W'] = 'E';
	scanf("%d%s%s", &n, s, t);
	--n;
	thirtyone[0] = 1;
	for (int i = 1; i <= n; ++i)
		thirtyone[i] = thirtyone[i - 1] * 31;
	for (int i = n - 1; i >= 0; --i)
		dp[i] = s[i] * thirtyone[n - i - 1] + dp[i + 1];
	lag h = 0;
	for (int i = n - 1; i >= 0; --i) {
		h = h * 31 + opp[t[i]];
		if (dp[i] == h) {
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	
	
}
