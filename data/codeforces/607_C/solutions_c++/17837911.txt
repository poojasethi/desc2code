#include <bits/stdc++.h>
using namespace std;

const int base = 23, mod = 998244353;
const int MaxN = 1000010;
int n;
char s[MaxN], t[MaxN];
map <char, int> value;

int main() {
	value['N'] = 1;
	value['E'] = 2;
	value['W'] = 3;
	value['S'] = 4;
	scanf("%d", &n);
	scanf("%s%s", s + 1, t + 1);
	int pow = 1, hl = 0, hr = 0;
	for (int i = 1; i < n; ++i) {
		int vl, vr;
		vl = value[s[n - i]], vr = 5 - value[t[n - i]];
		hl = (1ll * hl * base + vl) % mod;
		hr = (1ll * pow * vr + hr) % mod;
		pow = 1ll * pow * base % mod;
		if (hl == hr) {
			printf("NO\n");
			return 0;
		}
	}
	printf("YES\n");
	return 0;
}
