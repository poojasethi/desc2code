#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
using namespace std;
const int N=1000010;
int n,i,j,a[N],b[N],f[N];
char s1[N],s2[N];
int calc(char ch)
{
    if (ch=='N') return 1;
    else if (ch=='S') return 4;
    else if (ch=='E') return 2;
    else return 3;
}
int main()
{
    scanf("%d%s%s",&n,s1,s2);
    for (i=1;i<n;i++) a[i]=calc(s1[i-1]),b[n-i]=5-calc(s2[i-1]);
    memset(f,0,sizeof(f));
    i=1; j=0;
    while (i<n)
    {
        while (j && b[i]!=b[j]) j=f[j];
        f[++i]=++j;
    }
    i=1; j=1;
    while (i<n && j<n)
    {
        while (j && a[i]!=b[j]) j=f[j];
        i++; j++;
    }
    if (j!=1) printf("NO\n"); else printf("YES\n");
    return 0;
}
