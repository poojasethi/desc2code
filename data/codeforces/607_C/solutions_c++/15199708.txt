#include <cstdio>

const int maxn = 2000000 + 10;

int n;
char s[maxn];
int f[maxn];

int main()
{
    scanf("%d", &n); getchar(); n--;
    for(int i = n-1; i >= 0; i--) {
        s[i] = getchar();
        if(s[i] == 'N') s[i] = 'S';
        else if(s[i] == 'S') s[i] = 'N';
        else if(s[i] == 'W') s[i] = 'E';
        else s[i] = 'W';
    }
    s[n] = '#';
    scanf("%s", s + n + 1);
    n = n * 2 + 1;

    f[0] = f[1] = 0;
    for(int i = 1; i < n; i++) {
        int j = f[i];
        while(j && s[j] != s[i]) j = f[j];
        f[i+1] = s[j] == s[i] ? j+1 : 0;
    }

    printf("%s\n", f[n] ? "NO" : "YES");

    return 0;
}
