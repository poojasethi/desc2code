#include<iostream>  
#include<queue>  
#include<cmath>  
#include<cstdio>  
#include<cstring>  
#include<algorithm>
#define MAXN 1000100
using namespace std;
char ch1[MAXN],ch2[MAXN],ch3[MAXN*2+10];
int next[MAXN*2+10];
int main()
{
	int i,j,m,n,x,y,s,t,k,p,len;
	scanf("%d",&n);
	n--;
	cin>>ch1;
	cin>>ch2;
	for (i=n-1;i>=0;i--)
	ch3[i]=ch2[n-i-1];
	for (i=0;i<n;i++)
	{
		if (ch3[i]=='N') ch3[i]='S';
		else	if (ch3[i]=='S') ch3[i]='N';
		else	if (ch3[i]=='E') ch3[i]='W';
		else	if (ch3[i]=='W') ch3[i]='E';
	}
	ch3[n]=' ';
	len=2*n+1;
	for (i=n+1;i<len;i++)
	ch3[i]=ch1[i-n-1];
	next[0]=0;
	for (p=1,k=0;p<len;p++)
	{
		while (k&&ch3[k]!=ch3[p])
		k=next[k-1];
		if (ch3[p]==ch3[k]) k++;
		next[p]=k;
	}
	if (next[len-1]>0) printf("NO"); else printf("YES");
}
