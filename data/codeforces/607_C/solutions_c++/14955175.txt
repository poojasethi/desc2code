#include <iostream>
#include <string>
#include <cstdio>

using namespace std;

int anterior[3000000];

char buff[3000000];

int main()
{
  int n;
  scanf("%d",&n);
  n--;
  scanf("%s",buff);
  string s1(buff);
  scanf("%s",buff);
  string s2(buff);
  for (int i=0;i<n;i++) {
    if (s2[i]=='E') s2[i]='W';
    else if (s2[i]=='W') s2[i]='E';
    else if (s2[i]=='N') s2[i]='S';
    else if (s2[i]=='S') s2[i]='N';
  }
  for (int i=0,j=n-1;i<j;i++,j--) swap(s2[i],s2[j]);
  string s=s2+'#'+s1;
  n=n+n+1;
  int j=-1;
  anterior[0]=-1;
  for (int i=1;i<n;i++) {
    while (j!=-1 and s[j+1]!=s[i]) j=anterior[j];
    if (s[j+1]==s[i]) j++;
    anterior[i]=j;
  }
  if (anterior[n-1]!=-1) cout<<"NO"<<endl;
  else cout<<"YES"<<endl;
}
