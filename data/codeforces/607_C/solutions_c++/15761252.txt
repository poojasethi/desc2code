#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

#define N 2000005

char a[N], b[N];
int z[N];

void zAlgo(char *s, int *z) {
	int len = strlen(s);
	z[0] = len;
	for (int i = 1, j = 0; i < len; ) {
		while (i + j < len && s[j] == s[i + j]) j ++;
		z[i] = j;
		if (!j) { i ++; continue; }
		int k = 1;
		while (i + k < len && k + z[k] < j) z[i + k] = z[k], k ++;
		i += k, j -= k;
	}
}

main() {
	int n;
	scanf("%d", &n);
	n --;
	scanf("%s", a);
	scanf("%s", b);
	for (int i = 0; i < n; i ++) {
		if (a[i] == 'S') a[i] = 'N';
		else if (a[i] == 'N') a[i] = 'S';
		else if (a[i] == 'E') a[i] = 'W';
		else a[i] = 'E';
	}
	reverse(a, a + n);
	strcat(a, b);
	zAlgo(a, z);
	for (int i = n; i < n + n; i ++) {
		int rem = n + n - i;
		if (rem == z[i]) return puts("NO");
	}
	puts("YES");
}
