#include <stdio.h>

void run(int i,int n){
for(int j=0;j<n-i;j++)
{printf("  ");}
for(int j=0;j<i;j++)
{printf("%d ",j);}
printf("%d",i);
for(int j=i-1;j>=0;j--)
{printf(" %d",j);}
printf("\n");
}

int main(){
int n;
scanf("%d",&n);
for(int i=0;i<n;i++)
{
run(i,n);
}
for(int i=n;i>=0;i--)
{
run(i,n);
}
return 0;
}
