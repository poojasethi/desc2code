#include<iostream>
#include<cstdio>
using namespace std;
void OP(int a,int n){
	for(int i=a*2;i<n*2;i++) cout<<" ";
	for(int i=0;i<a*2;i++) cout<<((i<=a)?i:2*a-i)<<" ";
	cout<<"0\n";
}
int main(){
	int n;
	while(cin>>n){
		for(int i=0;i<=n;i++)
		OP(i,n);
		for(int i=n-1;i>=0;i--) OP(i,n);
	}
	return 0;
}
