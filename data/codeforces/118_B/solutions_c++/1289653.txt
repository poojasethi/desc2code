#include <cstdio>

using namespace std;

int n;
void out(int k) {
	for(int i=0;i<n-k;i++) printf("  ");
	for(int i=0;i<k;i++) printf("%d ",i);
	for(int i=k;i>0;i--) printf("%d ",i);
	printf("0\n");
}

int main() {
	scanf("%d",&n);
	for(int i=0;i<=n;i++) out(i);
	for(int i=n-1;i>=0;i--) out(i);
	return 0;
}
