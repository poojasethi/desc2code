#include <cstdio>
#include <algorithm>

using namespace std;

int n;

int main(){

  scanf("%d",&n);
  
  for(int i = -n ; i <= n ; i++){
    int m = n - abs(i);
    for(int j = -n ; j < -m ; j++){
      printf("  ");
    }
    for(int j = -m ; j < m ; j++){
      printf("%d ",m-abs(j));
    }
    printf("0\n");
  }
  
}
