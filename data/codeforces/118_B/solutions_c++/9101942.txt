#include<iostream>
#include<algorithm>
using namespace std;
int n;
int main(){
    cin>>n;
    for(int i=0;i<2*n+1;i++){
        for(int j=0;j<abs(2*n-2*i);j++)
            cout<<" ";
        for(int j=0;j<2*min(i,2*n-i);j++)
            cout<<(min(j,min(2*i,4*n-2*i)-j))<<" ";
        cout<<"0"<<endl;
    }
 return 0;
}
