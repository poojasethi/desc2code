#include <iostream>
#include <cmath>
using namespace std;
int main() {
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i=-n; i <= n; i++ ){
		for(int j=-n ; j <= n-abs(i); j++) {
			if(abs(j)+abs(i) > n) cout <<' ';
			else cout << n-abs(j)-abs(i);
			if((j+1)<=(n-abs(i))) cout <<' ';
		}
		cout << endl;
	}
	return 0;
}
