#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	int n;
	cin>>n;

	for(int i=-n; i<=n; i++)
	{
		int m=abs(i);
		for(int j=0; j<m*2; j++)
		cout<<" ";

		int v=n-m;
		for(int l=-v; l<=v; l++)
		{
			cout<<v-abs(l);

			if(l !=v)
			cout<< " ";
		}
		cout<<endl;
	}


	return 0;
}
