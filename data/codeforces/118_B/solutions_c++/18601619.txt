#include<stdio.h>
#include<stdlib.h>
#include<math.h>

int main()
{
  int i,n,j;
  scanf("%d", &n);
  for(i=-n;i<=n;i++){
    for(j=0;j<2*abs(i);j++){
      printf(" ");
    }
    for(j=0;j<abs(abs(i)-n);j++){
	printf("%d ", j);
    }
    for(j=abs(abs(i)-n);j>0;j--){
      printf("%d ", j);
    }
  
    printf("0\n");}
}
