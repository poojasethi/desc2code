#include <cstdlib>
#include <iostream>
using namespace std;

int main() {
  int N; cin >> N;
  for (int i = -N; i <= N; ++i) {
    cout << string(2*abs(i), ' ');
    int M = N-abs(i);
    for (int j = -M; j <= M; ++j) {
      cout << M-abs(j);
      if (j != M) cout << ' ';
    }
    cout << endl;
  }
}
