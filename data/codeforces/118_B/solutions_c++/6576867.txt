#include<stdio.h>
int main()
{
	int i,j,k,n,m;
	while(scanf("%d",&n)!=EOF)
	{
		for(i=0;i<2*n+1;i++)
		{
			j=i<=n?i:(2*n-i);
			//printf("%d\n",j);
			for(k=0;k<n-j;k++)
			  printf("  ");
			for(k=0;k<2*j+1;k++)
			{
				m=k<=j?k:(2*j-k);
				printf("%d%c",m,k==2*j?'\n':' ');
			} 
		}
	} 
	return 0;
}
