#include<stdio.h>

int main()
{
	int n,i,j,a;
	scanf("%d",&n);
	for(a=0,i=0;a<=2*n;a++)
	{
		i=a;
		if(a>n)
		i=n-(a-n);
		for(j=1;j<=(n-i)*2;j++)
			printf(" ");
		printf("0");	
		for(j=1;j<=i*2;j++)
		{
			if(j<=i)
			printf(" %d",j);
			else
			printf(" %d",(2*i)-j);
		}
		printf("\n");
	}
	return 0; 
}
