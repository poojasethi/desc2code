#include <bits/stdc++.h>
using namespace std;
int main ()
{
    int n;cin>>n;
    for(int i=0,b=0;i<2*n+1;i++)
    {
        for(int j=(n-b);j>0;j--)cout<<"  ";
        for(int j=0,a=0;j<b*2+1;j++){(j==b*2)?cout<<a:(j<b)?cout<<a++<<" ":cout<<a--<<" ";}
        cout<<endl;
        (i>=n)?b--:b++;
    }
    return 0;
}
