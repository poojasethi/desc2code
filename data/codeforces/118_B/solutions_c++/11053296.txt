#include <bits/stdc++.h>
using namespace std;

int main(){
	int n;
	cin >> n;
	for(int i = 0;i < 2*n + 1;i++){
		int h = max(n - i,i - n);
		for(int j = 0;j < h;j++) cout << "  ";
		for(int j = 0;j < n - h;j++) cout << j << " ";
		cout << n - h;
		for(int j = n - h - 1;j >= 0;j--) cout << " " << j;
		cout << "\n";
	}
	return 0;
}
