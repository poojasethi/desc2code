#include<iostream>
using namespace std;
void print( int n, int a ) {
    for(int b=1;b<=n-a;b++)cout<<"  ";
   	for(int i=0;i<a;i++)cout<<i<<" ";
    for(int i=a;i>0;i--)cout<<i<<" ";
	cout<<0<<endl;
}
int main()
{
	int n;
	cin>>n;
  	for(int a=0;a<n;a++)print(n,a);
  	for(int a=n;a>=0;a--)print(n,a);

}
