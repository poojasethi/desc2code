#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n, i, j = 0,x;
	cin >> n;
	for (x = 0; x <= 2 * n; x++)
	{
		i = n - abs(x - n);
		j = 0;
		while (j<n - i)
		{
			cout<<"  ";
		        j++;
		 }
		j = i;
		while (j>-i) 
		{
			 cout << i - abs(j);
			 cout<<" "; 
			 j--;
	        }
		cout << 0 << endl;
	}
}
