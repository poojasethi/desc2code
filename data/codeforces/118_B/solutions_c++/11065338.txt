#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,t;
	cin>>n;
	for (int i=0;i<=(2*n);i++)
	{
		t=min(2*n-i,i);
		for (int i=0;i<(2*(n-t));i++)
			{cout<<' ';}
		for (int i=0;i<=(2*t);i++)
		{
			{cout<<min(i,2*t-i);}
			if(i==(2*t)) cout<<endl;
			else cout<<' ';
		}
	}
    return 0;
}
