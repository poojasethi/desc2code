#include <bits/stdc++.h>

using namespace std;

int n;

int main () {
	cin >> n;
	for (int i = -n; i <= n; i++) {
		int m = abs (i);
		for (int j = 0; j < 2*m; j++)
			cout << " ";
		int s = n-m;
		for (int j = -s; j <= s; j++) {
			cout << s-abs(j);
			if (j != s)
				cout << " ";
		}
		cout << endl;
	}
	return 0;
}
