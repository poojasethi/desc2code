#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
using namespace std;
char c[1200];
int main()
{
    int i,j;
	int n,x,y,ans;
	gets(c+1);
	n=strlen(c+1);
	ans=0;
	for(i=1;i<=n;i++)
	{
		x=c[i];
		y=0;
		for(j=1;j<=8;j++)
		{
			y=y*2+x%2;
			x/=2;
		}
		printf("%d\n",(ans-y+256)%256);
		ans=y;
	}
}
