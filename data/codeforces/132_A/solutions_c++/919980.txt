#include <cassert>
#include <cstdio>

int mirror( int x ) {
  int r = 0;
  for (int i = 0; i < 8; i++)
    if ((x >> i) & 1)
      r |= 1 << (7 - i);
  return r;
}

int main() {
  char s[1010];
  assert(gets(s));
  int ch = 0;
  for (int i = 0; s[i]; i++) {
    int cc = s[i];
    printf("%d\n", (256 + mirror(ch) - mirror(cc)) % 256);
    ch = cc;
  }
  return 0;
}
