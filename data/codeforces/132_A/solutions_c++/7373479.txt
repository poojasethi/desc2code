#include <cstdio>
#include <cstring>
char s[100+10];
int n,d;
int main(){
    gets(s);
    int k=strlen(s);
    for(int i=0;i<k;i++){
        n=0;
        for(int j=0;j<8;j++)
            if(s[i]&1<<j)
                n|=1<<(8-j-1);
        printf("%d\n",(d-n+256)%256);
        d=n;
    }
    return 0;
}
