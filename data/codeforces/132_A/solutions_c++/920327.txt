#include<cstdio>
char s[1005];
int main()
{
    gets(s);
    int i,j,pre=0;
    for(i=0;s[i];i++)
    {
        int a=s[i];
        int b=0;
        for(j=0;j<8;j++)
        {
            if(a&(1<<j))
            {
                b|=(1<<(7-j));
            }
        }
        printf("%d\n",(pre-b+256)%256);
        pre=b;
    }
    return 0;
}
