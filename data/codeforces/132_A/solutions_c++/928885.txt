#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int MAXL=1000;
const int MASK=0xff;
char s[MAXL+10];
int main() {
	gets(s+1);
	cout<<s<<endl;
	for (int i=1,ts;s[i];i++) {
		ts=0;
		for (int p=0;p<8;p++) {
			ts<<=1;
			ts|=s[i]&1;
			s[i]>>=1;
		}
		s[i]=ts;
		printf("%d\n",(s[i-1]-s[i]+MASK+1)&MASK);
	}
	return 0;
}
