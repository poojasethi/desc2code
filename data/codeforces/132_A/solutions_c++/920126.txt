#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

char s[105];

int main() {
    gets(s);
    int p = 0;
    for (int i = 0; s[i]; ++i) {
        int c = 0;
        for (int j = 0; j < 8; ++j)
            c = (c << 1) + (s[i] >> j & 1);
        printf("%u\n", (p - c + 256) % 256);
        p = c;
    }
    return 0;
}
