#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int main()
{
  int T=0;char c;
  while(c=getchar(),c>=32&&c<=126)
    {
      int S=0;
      for(int i=0;i<8;i++){S=S*2+c%2;c/=2;}
      printf("%d\n",(T+256-S)%256);
      T=S;
    }
  return 0;
}
