#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int rev(int a){
	int b=0;
	rep(i,8) if(a&(1<<i)) b|=1<<(7-i);
	return b;
}

int main(){
	char s[128]; gets(s);
	for(int i=0;s[i];i++){
		int pre=(i==0?0:rev(s[i-1]));
		printf("%d\n",(pre-rev(s[i])+256)%256);
	}
	return 0;
}
