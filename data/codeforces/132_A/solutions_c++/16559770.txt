#include <bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

using namespace std;

char x;
int cur,v;

int main(){
  while(scanf("%c",&x)!=EOF){
    if(x=='\n') return 0;
    v=0;
    FI(i,0,7) if(((int)x)&(1<<i)) v+=1<<7-i;
    printf("%d\n",(cur-v+256)%256);
    cur=v;
  }
}
