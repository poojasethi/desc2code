#include<cstdio>
#include<cstring>
#include<string>
#include<iostream>
using namespace std;

string s; 
int	last; 

int main()
{
	getline(cin,s); 
	int	last = 0,k=0; 
	for (int i = 0; i<s.length(); i++)
	{
		int x = s[i],y = 0; 
		for (int j = 0; j<8; j++)
		{
			y = (y<<1) +(x&1);
			x = x>>1;
		}
		last = (k-y+256)%256;
		printf("%d\n", last ); 
		k = y;
	}
	return 0; 
}
