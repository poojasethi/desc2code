#include<cstdio>
#include<cstring>
char s[1005]; int n,ans,last;
int rev(int x)
{
	int y=0;
	for (int i=0;i<8;i++)
		y=y<<1|(x&1),x/=2;
	return y;
}
int main()
{
	gets(s),n=strlen(s),last=0;
	for (int i=0;i<n;i++){
		int res=rev(s[i]); 
		printf("%d\n",(last-res+256)%256);
		last=res;
		}
	return 0;
}
