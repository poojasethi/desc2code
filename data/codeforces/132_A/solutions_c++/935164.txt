#include <stdio.h>
#include <cstring>

char str[400];

int calc(int n) {
	int ans = 0;
	for(int i=0; i<8; ++i){
		ans = ans * 2 + n % 2;
		n >>= 1;
	}
	return ans;
}

int main() {
	while(gets(str)) {
		for(int i=0; str[i]; ++i) {
			int k;
			if(i == 0) k = 0;
			else k = calc(str[i-1]);
			k = (k - calc(str[i]) + 256) % 256;
			printf("%d\n", k);
		}
	}
	return 0;
}
