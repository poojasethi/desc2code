#include <cstdio>
#include <cstring>

int n;
char s[101];

int gao(int n) {
	int ret = 0;
	for (int i = 0; i < 8; ++i) {
		ret = ret << 1 | n & 1;
		n >>= 1;
	}
	return ret;
}

int main() {
	gets(s);
	n = strlen(s);
	int last = 0;
	for (int i = 0; i < n; ++i) {
		printf("%d\n", (gao(last) - gao(s[i]) + 256000) % 256);
		last = s[i];
	}
	return 0;
}
