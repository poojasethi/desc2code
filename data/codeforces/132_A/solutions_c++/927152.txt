#include "stdio.h"

unsigned char revchar(char c)
{
	unsigned char r = 0;
	for(int i = 7; i >= 0; i --)
	{
		r |= (c & 1) << i;
		c >>= 1;
	}
	return r;
}

int main()
{
	int r = 0;
	char s[1024];
	gets(s);
	for(char *p = s; *p; p ++)
	{
		int t = revchar(*p);
		printf("%d\n", (r - t + 256) % 256);
		r = t;
	}
}
