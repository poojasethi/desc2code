#include <stdio.h>
char c[1100];
int f(int x){
    int t = 0, i;
    for (i = 0; i < 8; i++){
        t = t * 2 + (x & 1);
        x >>= 1;
    }
    return t;
}
int main(){
    int i, t = 0;
    gets(c);
    for (i = 0; c[i]; i++){
        printf("%d\n", (t - f(c[i]) + 256) % 256);
        t = f(c[i]);
    }
    return 0;
}
