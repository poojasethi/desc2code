#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

int main(){
    char c[1200];
    int i,j,x,y,len,lans;
	gets(c+1);
	len=strlen(c+1);
	lans=0;
	for(i=1;i<=len;i++){
		x=c[i];
		y=0;
		for(j=1;j<=8;j++){  
			y=y*2+x%2;
			x/=2;
		}
		cout<<(lans-y+256)%256<<endl;
		lans=y;
	}
}
