#include <cstdio>
#include <cstring>
char s[110];
int rev(int n)
{
    int a=0,i=0;
    while(n)
    {a=a*2+(n&1);n>>=1;i++;}
    while(i<8)a<<=1,i++;
    return a;
}
int main()
{
    gets(s);
    int pre=0;
    for(int i=0;i<strlen(s);i++)
    {
        int a=s[i];
        a=rev(a);
        printf("%d\n",(pre-a+256)&255);
        pre=a;
    }
    scanf(" ");
}
