#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>

using namespace std;

char s[110];

int main(){
    int k=0;
    gets(s);
    for (int i=0;i<strlen(s);i++){
        int p=s[i],q=0;
        for (int j=0;j<8;j++){
            q=q*2+p%2;
            p/=2;
        }
        printf("%d\n",(256+k-q)%256);
        k=q;
    }
    return 0;
}
