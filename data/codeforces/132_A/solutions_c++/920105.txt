#include <iostream>
#include <string>
using namespace std;
int f(int p) {
    int r = 0;
    for (int i = 0; i < 8; ++ i) {
        r |= ((p>>i)&1)<<(7-i);
    }
    return r;
}
int main() {
    string s;
    getline(cin, s);
    int p = 0;
    for (int i = 0; i < s.length(); ++ i) {
        cout << (f(p) - f((int)s[i]) + 256) % 256 << endl;
        p = s[i];
    }
}
