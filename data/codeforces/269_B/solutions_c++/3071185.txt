#include<stdio.h>
#include<math.h>
#include<string.h>
#include<algorithm>
#include<set>
using namespace std;
#define N 5050

int buf[N];
int main()
{
	int n,a,i,j,k;
	scanf("%d%*lf",&n);
	for(i=0,k=0;i<n;i++){
		scanf("%d%*lf",&a);
		j=upper_bound(buf,buf+k,a)-buf;
		if(j==k) buf[k++]=a;
		else buf[j]=a;
	}
	printf("%d\n",n-k);
	return 0;
}
