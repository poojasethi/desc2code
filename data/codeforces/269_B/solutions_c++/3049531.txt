#include <cstdio>

const int maxn = 5010;

int n, m, a[maxn], f[maxn];

int main()
{
	int mx=0;
	double x;
	scanf("%d%d", &n, &m);
	for (int i=1; i<=n; ++i) {
		scanf("%d%lf", a+i, &x);
		f[i] = 0;
		for (int j=1; j<i; ++j)
			if (a[j]<=a[i] && f[j]>f[i])  f[i]=f[j];
		++f[i];
		if (mx<f[i])  mx=f[i];
	}
	printf("%d\n", n-mx);
	return 0;
}
