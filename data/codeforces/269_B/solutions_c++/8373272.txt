#include <cstdio>

using namespace std;

int n,m,Ans,h[5005],f[5005];

int main(){
	scanf("%d%d",&n,&m);
	double _;
	for(int i=1;i<=n;++i)
		scanf("%d%lf",h+i,&_);
	for(int i=1;i<=n;++i){
		int k=0;
		for(int j=1;j<i;++j)
		if(h[j]<=h[i] && f[j]>f[k])k=j;
		f[i]=f[k]+1;
		if(f[i]>Ans)Ans=f[i];
	}
	printf("%d\n",n-Ans);
}
