#include<cstdio>
int n,m,i,j,a[10000],f[10000],ans;
double b;
int main()
{
    scanf("%d%d",&n,&m);
    for (i=1;i<=n;i++)
        scanf("%d%f",&a[i],&b);
    for (i=1;i<=n;i++)
    for (j=1;j<=i-1;j++)
    if (a[i]>=a[j])
    if (f[j]+1>f[i])
    f[i]=f[j]+1;
    for (i=1;i<=n;i++)
    if (f[i]>ans) ans=f[i];
    printf("%d",n-(ans+1));
}
