#include<cstdio>
#define N 100010
int n,m,ans,i,j,f[N],a[N];
double b;
int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
	scanf("%d%lf",&a[i],&b);
	for (i=1;i<=n;i++)
	{
	for (j=1;j<i;j++)
	if ((f[j]+1>f[i])&&(a[j]<=a[i]))
	f[i]=f[j]+1;
	if (f[i]+1>ans) ans=f[i]+1;
	}
	printf("%d\n",n-ans);
}
