#include<bits/stdc++.h>
 
using namespace std;
 
int n,m;
double g;
int a[5001];
int dp[5001];
int ans=1;

int main()
{
	cin>>n>>m;for(int i=1; i<=n; i++)cin>>a[i]>>g,dp[i]=1;

	for(int i=1; i<=n; i++)
		for(int j=1; j<i; j++)
			if(a[j]<=a[i])
				ans = max(ans,dp[i] = max(dp[i],dp[j]+1));
	cout<<n-ans;
}
