#include <iostream>
using namespace std;

int s[5005];

int main() {
  int n,m;
  cin>>n>>m;
  int size=0;
  for (int ni=0;ni<n;ni++) {
    int x;
    double d;
    cin>>x>>d;
    if (s[size-1]<=x) {
      s[size++]=x;
    } else {
      int p=size-1;
      while (p>=0 && x<s[p])
        p--;
      s[p+1]=x;
    }
  }
  cout<<n-size<<endl;
  return 0;
}
