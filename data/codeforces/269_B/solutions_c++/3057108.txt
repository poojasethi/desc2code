#include <iostream>
#include <algorithm>
using namespace std;
int n,m,i,j,k,ans;
double temp;
int a[5011];
int f[5011];
int main()
{
		cin>>n>>m;
		for (i=1;i<=n;i++) cin>>a[i]>>temp;
		for (i=1;i<=n;i++)
				for (j=0;j<i;j++)
						if (a[i]>=a[j]) f[i]=max(f[i],f[j]+1),ans=max(f[i],ans);
		cout<<n-ans;
		return 0;
}
