N = input()

C = list(raw_input())

C = [int(c) for c in C]

combination = float("inf")
ans = ""

for i, num in enumerate(C):

    to_add = 10 - num
    D = [str((n + to_add)%10) for n in C]
    # D = sorted(D)
    # D = [str(i) for i in D]

    D = [D[i]] + D[i+1:] + D[:i]
    D_string = "".join(D)
    D = int(D_string)

    if D < combination:
        combination = D
        ans = D_string

print ans
