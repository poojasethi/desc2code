n = int(raw_input()) 
a = map(int, raw_input().strip()) 
m = [i for i in a] 
for i in range(10):
  b = [(i + j) % 10 for j in a]
  for j in range(n):
    iv = 0
    for k in range(n):
      if b[(j + k) % n] < m[k]:
        iv = 1
        break
      elif b[(j + k) % n] > m[k]:
        break
    if iv:
      m = [b[(k + j) % n] for k in range(n)]
print ''.join(map(str, m))
