n = int(raw_input()) 
a = map(int, raw_input().strip()) 
m = [i for i in a] 

for i in range(n):
  iv = 0
  for j in range(n):
    x = a[(i + j) % n] + (10 - a[i])
    x %= 10
    if x < m[j]:
      iv = 1
      break
    if x > m[j]:
      break
  if iv:
    m = [(a[(i + j) % n] + 10 - (a[i])) % 10 for j in range(n)]

print ''.join(map(str, m))
