n = int(raw_input())
num = [int(x) for x in raw_input()]
minimum = int('9' * n)
for i in xrange(n):
    num = num[1:] + [num[0]]
    key = 10 - num[0]
    for i in xrange(n):
        num[i] = (num[i] + key) % 10
    this_number = int(''.join(str(x) for x in num))
    minimum = min(this_number, minimum)

minimum = str(minimum)
print '0' * (n - len(minimum)) + (minimum)
