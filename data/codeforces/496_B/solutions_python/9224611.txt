def add(x):
    ans = ''
    for i, c in enumerate(x):
        num = int(c) + 1
        if (num == 10):
            ans += '0'
        else:
            ans += str(num)
    return ans


n = int(raw_input())
d = raw_input()
res = d
for i in xrange(10):
    for j in xrange(n):
        if int(d) < int(res):
            res = d
        d = d[-1] + d[:-1]
    d = add(d)
print res
