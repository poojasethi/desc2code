def s(s):
    a = 10 - int(s[0])
    ans = '0'
    s = s[1:]
    for i in s:
        k = int(i)
        d = str((a+k)%10)
        ans+=d
    return ans

def d(s):
    return s[-1:]+s[:-1]
raw_input()
st = raw_input()
ln = len(st)

miz = s(st)

for i in range(ln):
    st = s(d(st))
    if miz>st:
        miz = st
