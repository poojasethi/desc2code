n = int(raw_input())
a = raw_input()

m = int(a)

for i in xrange(10):
    t = ''.join([str((int(x) + i) % 10) for x in a])
    mt = int(t)
    for j in xrange(n):
        tn = int(t[j:] + t[:j])
        if tn < mt:
            mt = tn
    if mt < m:
        m = mt
pad = '0' * (n - len(str(m)))
print '%s%s' % (pad, m)
