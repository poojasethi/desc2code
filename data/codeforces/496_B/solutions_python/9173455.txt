n=int(raw_input())
num=raw_input()

vals = set()

for _ in xrange(n):
    vals.add(int(num))

    add = 10 - int(num[0])
    temp = "0"
    for i in xrange(1,n):
        temp += str((int(num[i]) + add) % 10)
    vals.add(int(temp))

    num = num[-1] + num[:-1]
    

print str(min(vals)).zfill(n)
