#!/usr/bin/python

numDigs = int(raw_input())
minEntry = '9' * numDigs
digs = raw_input()
numerical = map(int, digs)

for i in range(0, numDigs):
  tmpNumerical = numerical[i:] + numerical[:i]
  tmpZeroed = map(lambda x: (x - tmpNumerical[0]) % 10, tmpNumerical)
  minEntry = min(minEntry, ''.join(map(str,tmpZeroed)))

print minEntry
