AB = '0123456789'

def rotate(s, dif):
    return ''.join([AB[(AB.index(c) + dif) % 10] for c in s])
    
def shift(s, i):
    return s[i:] + s[:i]

if __name__ == '__main__':
    n = int(raw_input())
    s = raw_input()
    m = int(s)
    for i,c in enumerate(s):
        dif = 10 - int(c)
        m = min(m, int(shift(rotate(s, dif), i)))
    m = str(m)
    m = ('0' * (len(s) - len(m))) + m
