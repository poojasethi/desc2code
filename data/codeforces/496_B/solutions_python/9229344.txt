n = input()
s = raw_input()
num = int(s)
for i in xrange(n):
    d = [0 for k in xrange(len(s))]
    d[i] = s[i]
    r = 10 - int(d[i])
    for j in xrange(len(s)):
	if j != i:
	    d[j] = str((int(s[j]) + r) % 10)
    ts = ''.join(d[i+1:] + d[:i])
    t = int(ts) if len(ts) > 0 else 0
    if t < num: num = t	    
ans = str(num)
while len(ans) < n:
    ans = '0' + ans
print ans
