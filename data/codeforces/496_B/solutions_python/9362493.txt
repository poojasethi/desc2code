length = int(raw_input())
context = raw_input()
s = []
for i in range(length):
    s.append(context[i])
s2 = s[:]

for i in range(10):
    for j in range(length):
        s2[j] = str((int(s2[j]) + 1) % 10)
    for j in range(length):
        s2.append(s2[0])
        s2.pop(0)
        if s2 < s:
            s = s2[:]

res = ""
for i in range(length):
    res = res + s[i]

print res
