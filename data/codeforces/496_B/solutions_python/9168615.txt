def solve(n, s):
    def add(s, d):
        r = map(int, s);
        for i in range(n):
            r[i] = (r[i]+d)%10
        return ''.join(map(str, r))
    res = s
    for i in range(n):
        d = int(s[0])
        t = add(s, -d)
        if int(t) < int(res):
            res = t
        s = s[1:]+s[0]
    return res

n = int(raw_input())
print solve(n, raw_input())
