#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    int n,i,j;
    string s,s1;
    cin>>n>>s;
    string mn=s;
    for(i=0;i<n;i++)
    {
        string res="0";
        for(j=1;j<n;j++)
        {

            res+=(((s[(j+i)%n]-'0')-(s[i]-'0')+10)%10+'0');
        }
        if(res<mn) mn=res;
    }
    cout<<mn<<endl;
}
