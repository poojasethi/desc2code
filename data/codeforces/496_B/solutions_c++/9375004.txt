#include<iostream>
using namespace std;
string ans,str;
int N;
void change()
{
char ch=str[str.size()-1];
for(int i=str.size()-1;i>0;i--)
str[i]=str[i-1];
str[0]=ch;	
	
}
int main()
{
int i,j,k;
cin>>N;
cin>>str;
ans=str;
for(i=0;i<N;i++)
{
change();
for(j=0;j<10;j++)
{
	for(k=0;k<N;k++)
	str[k]=(str[k]-'0'+1)%10+'0';
ans=min(ans,str);
}

}
cout<<ans;
	
	
return 0;	
}
