#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,j;
    cin>>n;
    string a,p,ma;
    cin>>a;
    ma=a;
    for(i=0;i<n;i++)
    {
        p="0";
        for(j=1;j<n;j++)
        {
            p+=(((a[(i+j)%n]-'0')-(a[i]-'0')+10)%10+'0');
        }
        if(p<ma)
            ma=p;
    }
    cout<<ma<<endl;
    return 0;
}
