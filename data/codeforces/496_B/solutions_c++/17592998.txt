#include <bits/stdc++.h>
using namespace std;

string s, t;
vector<string> v;

int main()
{
	int n, i, k, h;
	cin >> n >> s;
	for(i = 0; i < n; i++)
	{
		t = s.substr(i, n - i) + s.substr(0, i);
		for(k = 0; k < 10; k++)
		{
			for(h = 0; h < n; h++)
				t[h] = '0' + ((t[h] - '0') + 1) % 10;
			v.push_back(t);
		}
	}
	sort(v.begin(), v.end());
	cout << v[0];
	return 0;
}
