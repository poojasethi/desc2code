#include <bits/stdc++.h>
using namespace std;

string num;
set<string>S;
int main(){
	int n;
	cin>>n>>num;
	num+='#';
	
	for(int i=n;i>=0;i--){
		string str=num.substr(i,n-i)+num.substr(0,i);
		
		int add=10-(str[0]-'0');
		for(int j=0;j<n;j++){
			int digit=str[j]-'0';
			digit+=add; digit%=10;
			str[j]=digit+'0';
		}
		
		S.insert(str);
	}
	cout<<(*S.begin())<<"\n";
}
