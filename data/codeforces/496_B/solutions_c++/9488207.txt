#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;
int main(){
	string x, ans, st;
	int n , i, j, k;
	cin>>n;
	cin>>st;
	ans = st;
	st = st+st;
	for(i = 0; i < n; i++){
		x = st.substr(i, n);//cout<<x<<endl;
		k = '9' - x[0] + 1;
		for(j = 0; j < x.size() ; j++){
			x[j] = '0' + (x[j] - '0' + k) %10;
		}
		ans = min(ans,x );
	}
	cout<<ans;
	return 0;
}
