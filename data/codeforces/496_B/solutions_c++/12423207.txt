#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	string a;
	cin>>n;
	cin>>a;
	string d;
	for (int i=0; i<n; i++)
	{
		int x=10-(int)a[i]+(int)'0';
		string b="";
		for (int j=i; j<i+n; j++)
		{
			int y=j%n;
			int c=(int)a[y]-48;
			b=b+(char)(48+(c+x)%10);
		}
		if (i==0)
		{
			d=b;
		}
		else
		{
			d=min(d, b);
		}
	}
	cout<<d;
	return 0;
}
