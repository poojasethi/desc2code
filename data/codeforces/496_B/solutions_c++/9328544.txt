#include<bits/stdc++.h>
using namespace std;

#define INF 1000000000

int n;
char num[1111];

int main(){	
	while(scanf("%d\n", &n)!=EOF){
		gets(num);
		string min_ = string(num);
		for(int beg=0; beg<n; beg++){
			string s = "";
			for(int i=0;i<n;i++){
				s += ((num[(beg+i)%n]-num[beg]+10)%10 + '0');
			}
			if(s<min_) min_=s;
		}
		cout << min_<<endl;
	}
	return 0;
}
