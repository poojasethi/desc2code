#include<iostream>
#include<cstring>
using namespace std;

int n;
string ans,s;

int main()
{
    ios::sync_with_stdio(false);
    cin>>n>>s;
    ans=s;s=s+s;
    for(int i=0;i<n;i++){
        string st=s.substr(i,n);
        int p='9'-st[0]+1;
        for(int j=0;j<n;j++)
            st[j]='0'+(st[j]-'0'+p)%10;
        ans=min(ans,st);
    }
    cout<<ans<<endl;
    return 0;
}
