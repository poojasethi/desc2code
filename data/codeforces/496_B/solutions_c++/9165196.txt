#include<bits/stdc++.h>
using namespace std;
const int maxn=1010;
char c[maxn],a[maxn][maxn];
int main()
{
    ios_base::sync_with_stdio(0);cin.tie(0);
    int n;
    cin>>n>>c;
    int l=strlen(c),ans=0;
    for(int i=0;i<l;i++)
    {
        for(int j=0;j<l;j++) a[i][j]=(c[(i+j)%l]-c[i]+10)%10+'0';
        a[i][l]='\0';
        if(strcmp(a[ans],a[i])>0) ans=i;
    }
    puts(a[ans]);
    return 0;
}
