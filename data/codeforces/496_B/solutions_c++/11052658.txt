#include <bits/stdc++.h>
using namespace std;
int n,d,a[1005];
string s,m;
int main()
{
    cin >> n >> m;
    for(int i=0;i<n;i++) 
    {
        a[i]=m[i]-'0';
    }
    for(int i=0;i<n;i++) 
    {
        d=10-a[i];
        s="";
        for(int j=0;j<n;j++) 
        {
            s+=(a[(i+j)%n]+d)%10+'0';
        }
        if(s<m) 
        {
			m=s;
		}
    }
    cout << m;
    return 0;
}
