#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int n, flag=0;
	string s, mini;
	cin>>n>>s;
	for(int i=0 ; i<10 ; i++)
	{
		for(int j=0 ; j<n ; j++)
		{
			s[j]++;
			if(s[j]>'9')
				s[j] = '0';
		}		
		for(int k=0 ; k<n ; k++)
		{
			s = s.substr(1,n-1) + s[0];
			if(flag == 0)
			{
				mini = s;
				flag=1;
			}
			else if(s<mini) 
			{
				mini = s;
			}
		}
	}
	cout<<mini<<endl;
}
