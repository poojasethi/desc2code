#include <iostream>
#include <string>
#include <cmath>
using namespace std;
int main()
{
     int n,i; 
     cin>>n;
     string s,sum,st;
     cin>>s;sum=s;
     s+=s;
     for(i=0;i<n;i++){
         st=s.substr(i,n);
     int k='9'-st[0]+1,j;
     for(j=0;j<st.size();j++)
     st[j]='0'+(st[j]-'0'+k)%10;
     sum=min(sum,st);}
     cout<<sum<<endl;
     return 0;
}
