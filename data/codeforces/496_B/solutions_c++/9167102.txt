#include <iostream>
#include <string>

using namespace std;

int main () {
	int n;
	string a;
	cin >> n >> a;
	
	string min = a;
	for (int i=0; i<10; i++) {
		
		for (int j=0; j<n; j++) {
			a[j]++;
			if (a[j] > '9')
				a[j] = '0';
		}
		
		for (int k=0; k<n; k++) {
			a = a.substr(1,n-1) + a[0];
			
			if (a < min)
				min = a;
		}
	}
	
	cout << min << endl;
	
	return 0;
}
