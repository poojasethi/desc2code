#include<iostream>
using namespace std;
int main()
{
	int n,dif;
	string a, mini,current;
	cin>>n>>a;
	mini = a;
	a = a + a;

	for(int i = 0; i < n; ++i)
	{
		current = a.substr(n-i,n);
		dif = 10 - (current[0] - '0');
		for(int j = 0; j < n; ++j)
		{
			current[j] = ((current[j] - '0') + dif )% 10 + '0';
		}
		if(mini > current)mini = current;
	}
	cout << mini << endl;
}
