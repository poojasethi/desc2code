#include <bits/stdc++.h>
using namespace std;

int main(){
	int n,i,j,diff;
	string s,ret,a;
	cin>>n>>s;
	ret=s;
	s = s+ s;
	for(i=0;i<n;++i){
		a = s.substr(i,n);
		diff = a[0]-'0';diff = 10-diff;
		for(j=0;j<n;++j)
			a[j] = char('0' + (a[j]-'0'+diff)%10);
		ret= min(ret,a);
	}
	cout<<ret<<endl;
}
