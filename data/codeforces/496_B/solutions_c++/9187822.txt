#include <iostream>
using namespace std;
int a[10000];
string mn;
int main()
{
    int n;
    cin>>n;
    string s;
    cin>>s;
    mn=s;
    for (int i=0;i<s.size();i++)
        a[i]=s[i]-'0';
    for (int i=0;i<n;i++)
    {
        int d=10-a[i];
        s="";
        for (int j=0;j<n;j++)
            s+=((a[(i+j)%n]+d)%10)+'0';
        if (s<mn) mn=s;
    }
    cout<<mn;
    return 0;
}
