#include <bits/stdc++.h>
using namespace std;

int main()
{
	cin.sync_with_stdio(0);
	cin.tie(0);
	int n;
	cin >> n;
	string s;
	cin >> s;
	string mini = s;
	for(int i = 0; i < n; ++i)
	{
		for(int j = 0; j < 10; ++j)
		{
			string t = s;
			for(int k = 0; k < n; ++k) t[k] = ((s[(i + k) % n] - '0' + j) % 10) + '0';
			mini = min(mini, t);
		}
	}
	cout << mini << endl;
}
