#include <iostream>
#include <string>
using namespace std;
int main() {
	string min, s;
	int n, i, j, m;
	cin >> n >> s; min = s;
	for(i = 0; i < n; ++i) {
		string hallo;
		m = (10 - (int)(s[i] - '0'))%10;
		for(j = 0; j < n; ++j) hallo.push_back(s[(i + j) % n] + m > '9' ? s[(i + j) % n] + m - 10 : s[(i + j) % n] + m);
		if(hallo.compare(min) < 0) min = hallo;
	}
	cout << min;
}
