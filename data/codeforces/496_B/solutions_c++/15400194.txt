#include <bits/stdc++.h>

using namespace std;

int n;
string s, ans;

int main(){
	cin >> n >> s;

	ans = s;	
	s += s;
	for (int i = 0; i < n; i++){
		string tmp = s.substr (i, n);
		int x = (9 - (tmp[0] - '0')) + 1;
		for (int j = 0; j < n; j++)
			tmp[j] = '0' + ( ((tmp[j] - '0') + x) % 10);
		ans = min (tmp, ans);
	}

	cout << ans << endl;

	return 0;
}
