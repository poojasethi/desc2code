#include <bits/stdc++.h>
using namespace std;

int N;
string S, res;

int main(){
  cin >> N >> S;
  for(int i = 0; i < 10; i++){
    for(int j = 0; j < N; j++){
      string T;
      for(int k = 0; k < N; k++){
        T.push_back('0' + (S[(j + k) % N] - '0' + i) % 10);
      }
      if(res.size() == 0) res = T;
      else res = min(res, T);
    }
  }
  cout << res << '\n';
}
