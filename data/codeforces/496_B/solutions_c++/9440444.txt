#include<iostream>
#include<string>

using namespace std;

int main(){

  int n;
  string a,b,c,d;
  cin>>n;
  cin>>a;
  b=a,c=a,d=a;
  for(int j=0;j<=9;j++){
    a=d;
    for(int i=0;i<n;i++){
        a[i]=(a[i]-'0'+j)%10+'0';
    }
    b=a;
    for(int i=0;i<n;i++){
        a=a[n-1]+a.substr(0,n-1);
        if(a<b)b=a;
    }//cout<<a<<" "<<b<<endl;
    if(b<c)c=b;
  }
  cout<<c<<endl;
  return 0;

}
