#include<iostream>
using namespace std;

int n;
string s,ans,temp2;
int main()
{

	cin >> n;
	int i,j;
	int increment;
	cin >> s;
	ans=s;
	s=s+s;
	for(i=0;i<n;i++)
	{
		increment='9'-s[i]+1;
		temp2=s.substr(i,n);
		for(j=0;j<n;j++)
			{
			
			temp2[j]='0' + (temp2[j]-'0'+increment+10)%10;
			//cout<<temp[j-i];
		}
		
		//	cout<<temp[j];

		//cout<< temp2;
		ans=min(ans,temp2);
	}
	cout << ans << endl;
	return 0;


}
