#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int n,i,j,k;
    cin>>n;
    string s,a,b,cur;
    cin>>s;
    a="";
    for(i=0;i<n;i++) a+='9';

    for(i=0;i<10;i++)
    {
       for(j=0;j<n;j++)
       {
           cur="";
           for(k=0;k<n;k++) cur+=(((s[(j+k)%n]+i-'0')%10)+'0');
           if(cur<a) a=cur;
       }
    }
    cout<<a;
}
