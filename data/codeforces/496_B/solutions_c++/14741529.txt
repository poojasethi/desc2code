#include <iostream>

using namespace std;

int main()
{
   int n; cin >> n;
   string s; cin >> s;
   string min_s = s;
   s += s;
   for (int i = 0; i < n; i++)
   {
      string s2 = s.substr(i, n);
      int dif = 10 - (s[i] - '0');
      for (int j = 0; j < s2.size(); j++)
         s2[j] = (((s2[j] - '0') + dif) % 10) + '0';
      min_s = min(min_s, s2);
   }
   cout << min_s << endl;
   return 0;
}
