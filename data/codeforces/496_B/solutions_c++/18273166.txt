#include<stdio.h>
#include<string.h>
#define N 1000
char tep[N], a[N], s[N];
int main()
{
	int i, j, k, n;
	scanf("%d",&n);
	scanf("%s",s);
	for(i = 0; i < n; i++)
		a[i] = '9';
	for(i = 1; i <= 10; i++)
	{
		for(j = 0; j < n; j++)
		{
			for(k = 0; k < n; k++)
				tep[k] = (s[(j + k) % n] - '0' + i) % 10 + '0';
			if(strcmp(a, tep) > 0)
				strcpy(a, tep);
		}
	}
	printf("%s\n", a);
	return 0;
}
