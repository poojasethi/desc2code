#include<stdio.h>
#include<string.h>
int main()
{
	int i,j,n,d;
	scanf("%d",&n);
	char s[10001],max[10001],tmp[10001];
	scanf("%s",s);
	strcpy(max,s);
	for(i=1;i<=n;i++)
	{
		strcpy(tmp,s+i);
		strcat(tmp,s);
		tmp[n]='\0';
		d=tmp[0]-'0';
		for(j=0;s[j];j++)
			tmp[j]-=(tmp[j]-'0') >= d ? d : d-10;
		if(strcmp(tmp,max)<0)
			strcpy(max,tmp);
	}
	puts(max);
	return 0;
}
