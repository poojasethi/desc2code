#include<bits/stdc++.h>

using namespace std;

int main(void)
{
	string S;
	int N;
	scanf("%d",&N);
	cin>>S;
	string A = S,B;
	for(int i=0;i<N;i++)
	{
		 B = S.substr(i,N-i) + S.substr(0,i);
		// cout<<B<<endl;
		int c = 10 - (B[0] - '0');
		for(int j=0;j<N;j++)
		{
			B[j] += c;
			if(B[j] > 57)
				B[j] -= 10;
		}
	//	cout<<B<<endl;
		A = min(A,B);
	}
	cout<<A;
	return 0;
}
