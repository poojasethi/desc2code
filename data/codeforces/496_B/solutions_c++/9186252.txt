#include <cstdio>
#include <iostream>
#include <cstring>
#include <string>
#include <algorithm>

using namespace std;

int main(){
	int n;
	string str;
	cin>>n>>str;
	string ans=str;
	for(int i=0;i<10;i++){
		for(int j=0;j<n;j++){
			str[j]++;
			if(str[j]>'9')str[j]='0';
		}
		for(int j=0;j<n;j++){
			str=str.substr(1,n-1)+str[0];
			ans=min(ans,str);
		}
	}
	cout<<ans<<endl;
	return 0;
}
