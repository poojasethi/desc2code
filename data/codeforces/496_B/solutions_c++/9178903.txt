#include <bits/stdc++.h>
using namespace std;

int main()
{
    
    int n;
    cin>>n;
    string s;
    cin>>s;
    string temp=s;
    for(int i=0;i<n;++i)
    {
        for(int j = 0; j < 10; ++j)
        {
            string t = s;
            for(int k=0;k<n;++k) 
            t[k] = ((s[(i + k) % n]-'0'+j) % 10+'0');
            temp=min(temp, t);
        }
    }
    cout <<temp<< endl;
}
