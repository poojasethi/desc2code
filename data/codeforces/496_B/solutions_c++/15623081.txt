#include<iostream>
#include<cstdio>
#include<string>
using namespace std;

const int maxn = 1005; char s[maxn];
int main(){
	int n;
	scanf("%d %s", &n,s);
	string ans = string(n,'9');
	for(int i=0; i<n; i++){
		string a = "0";
		for(int j=1; j<n; j++){
			a += (char)('0'+(s[(i+j)%n]-s[i]+10)%10);
		}
		ans = min(ans,a);
	}
	cout<<ans<<endl;
	return 0;
}
