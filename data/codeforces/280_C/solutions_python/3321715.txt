n = input()

neighbors = [[] for i in range(n + 2)]

for i in range(1, n):
    u, v = map(int, raw_input().split())
    neighbors[u - 1].append(v - 1)
    neighbors[v - 1].append(u - 1)

queue = [0]
depth = [1] + [0] * n
visit = [1] + [0] * n

answer = 0
for u in queue:
    answer += 1.0 / depth[u]
    for v in neighbors[u]:
        if visit[v] == 0:
            queue.extend([v])
            depth[v] = depth[u] + 1
            visit[v] = 1

print answer
