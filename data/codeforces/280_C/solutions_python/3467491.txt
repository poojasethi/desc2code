n = input()
edge = [[] for _ in range(n)]
for i in range(n - 1):
    a, b = map(int, raw_input().split())
    edge[a - 1].append(b - 1)
    edge[b - 1].append(a - 1)
d = [0] * n
d[0] = 1
p = [0]
for u in p:
    for v in edge[u]:
        if not d[v]:
            d[v] = d[u] + 1
            p.append(v)
print sum((1. / x for x in d))
