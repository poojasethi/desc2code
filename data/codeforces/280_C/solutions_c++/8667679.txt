#include <cstdio>
#include <algorithm>
#include <vector>

#define pb push_back

using namespace std;

vector<int> a[100005];

double ans = 0.0;

void dfs(int k, int par, int s) {
    ans += 1.0 / s;
    
    for (int i = 0; i < a[k].size(); i++)
	if (par != a[k][i]) dfs(a[k][i], k, s + 1);
}

int main() { 
    int n, i, x, y;

    scanf("%d",&n);

    for (i = 1; i < n; i++){
	scanf("%d%d",&x,&y);
	a[x].pb(y);
	a[y].pb(x);
    }
    
    dfs(1, 0, 1);

    printf("%.20lf\n", ans);
    
    return 0;
}
