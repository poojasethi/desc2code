#include <cstdio>
#include <vector>
using namespace std;

int n, a, b;
vector<int> v[100100];
double sol;

void dfs( int x, int par, int dp ){
	
	sol += double(1) / double(dp);
	for ( int i = 0; i < v[x].size(); i++ )
		if ( v[x][i] != par )
			dfs( v[x][i], x, dp + 1 );
	
}

int main(){
	scanf("%d", &n);
	for ( int i = 1; i < n; i++ ){
		scanf("%d %d", &a, &b);
		v[a].push_back(b);
		v[b].push_back(a);
	}
	dfs(1, 1, 1);
	printf("%lf\n", sol);
	
	return 0;
}
