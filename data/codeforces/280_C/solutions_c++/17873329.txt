#include<iostream>
#include<cstdlib>
#include<cstdio>
#define N 200005
using namespace std;

int n,siz;
double ans;
int fa[N],first[N],next[N],to[N];

void inser(int x,int y)
{
	next[++siz]=first[x];
	first[x]=siz;
	to[siz]=y;
}

void dfs(int x,int d)
{
	ans+=1./d;
	for (int i=first[x];i;i=next[i])
		if (to[i]^fa[x]) fa[to[i]]=x,dfs(to[i],d+1);
}

int main()
{
	scanf("%d",&n);
	for (int x,y,i=1;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		inser(x,y),inser(y,x);
	}
	dfs(1,1);
	printf("%.10f\n",ans);
	
	return 0;
}
