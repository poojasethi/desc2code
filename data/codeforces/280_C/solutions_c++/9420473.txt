#include <iostream>
#include<vector>
#include<cstdio>
using namespace std;
int n,a,b,root;
vector<int > adj[100010]; 
bool in[100010],out[100010];
double ans=0;
void dfs(int x,int depth)
{
	if(in[x])return ;
	in[x]=true;
	ans+=1.0/depth;
	//cout<<x<<" "<<ans<<endl;

	for(int i=0;i<adj[x].size();i++)
		dfs(adj[x][i],depth+1);
}
int main() {
	cin>>n;
	for(int i=0;i<n-1;i++)
	{
		cin>>a>>b;
		a--;b--;
		adj[a].push_back(b);
		adj[b].push_back(a);
	}
	
	dfs(0,1);
	
	printf("%.20f\n",ans);
	return 0;
}
