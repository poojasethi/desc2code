#include <cstdio>
#include <vector>

using namespace std;

const int maxn=100010;

int n,a,b;
vector<int> tree[maxn];
double rj=0;

void dfs(int x,int p,double dubina){

	rj+=1.0/dubina;
	
	for(int i=0;i<tree[x].size();i++){
		if(tree[x][i]!=p) dfs(tree[x][i],x,dubina+1);
	}
	
}

int main (){
	
scanf("%d",&n);

for(int i=0;i<n-1;i++){
	scanf("%d %d",&a,&b);
	a--;b--;
	tree[a].push_back(b);
	tree[b].push_back(a);
}	
	
dfs(0,-1,1);	
	
printf("%.15lf",rj);	
	
return 0;
}
