#include <stdio.h>
#include <vector>

int N;
std::vector<int> E[100005];

double dfs(int n, int d, int last)
{
    double exp = 1.0 / d;
    for(int i = 0; i < E[n].size(); i++)
        if(E[n][i] != last)
            exp += dfs(E[n][i], d + 1, n);
    return exp;
}

int main()
{
    scanf("%d", &N);
    int a,b;
    for(int i = 1; i < N; i++)
    {
        scanf("%d %d", &a, &b);
        E[a].push_back(b);
        E[b].push_back(a);
    }
    printf("%.8lf\n", dfs(1, 1, 0));
}
