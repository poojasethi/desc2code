#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
const int maxN=100010;
int n;
vector<int> g[maxN];
double dfs(int v,int p,int d){
	double prob=1.0/d;
	for(int i=0;i<g[v].size();++i)
		if (g[v][i]!=p)
			prob+=dfs(g[v][i],v,d+1);
	return prob;
}
int main(){
	cin>>n;
	for(int i=1;i<n;++i){
		int x,y;
		cin>>x>>y;
		g[x].push_back(y);
		g[y].push_back(x);
	}
	printf("%.6lf",dfs(1,0,1));
	return 0;
}
