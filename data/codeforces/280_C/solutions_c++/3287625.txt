#include <cstdio>
#include <vector>
using namespace std;
double ans;
vector <int> p[110000];
void dfs(int u,int v,int tol)
{
    ans+=1.0/tol;
    for(int i=0;i<p[u].size();i++){
        if(p[u][i]!=v){
            dfs(p[u][i],u,tol+1);
        }
    }
}
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n-1;i++) {
        int a,b;
        scanf("%d%d",&a,&b);
        p[a].push_back(b);
        p[b].push_back(a);
    }
    ans=0;
    dfs(1,0,1);
    printf("%.10f\n",ans);
}
