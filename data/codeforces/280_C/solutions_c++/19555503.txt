#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
const int maxn=100000+10;
double ans=0;
vector<int> G[maxn];
void dfs(int u,int fa,int dep)
{
	ans+=1.0/dep;
	for(int i=0;i<G[u].size();i++)
	{
		if(G[u][i]==fa) continue;
		dfs(G[u][i],u,dep+1);
	}
}

int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n-1;i++)
	{
		int u,v;
		scanf("%d%d",&u,&v);
		G[u].push_back(v);
		G[v].push_back(u);
	}
	dfs(1,-1,1);
	printf("%lf\n",ans);
}
