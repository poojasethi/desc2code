#include <iostream>
#include <cstdio>
#define N 100005
using namespace std;
struct tree{
	int to,next;
}t[N<<1];
int head[N],vi[N],n,cnt,u,v;
double ans;
void dfs(int x,int depth){
	if(vi[x])return;
	vi[x]=1;
	ans+=1/(double)depth;
	for(int i=head[x];i;i=t[i].next)
		dfs(t[i].to,depth+1);
}
int main() {
	scanf("%d",&n);
	for(int i=1;i<n;i++){
		scanf("%d%d",&u,&v);
		t[++cnt]=(tree){v,head[u]};
		head[u]=cnt;
		t[++cnt]=(tree){u,head[v]};
		head[v]=cnt;
	}
	dfs(1,1);
	printf("%f",ans);
	
}
