#include <cstdio>
#include <vector>
using namespace std;
#define N 100001

vector<int> adj[N];
int vis[N];

double dfs(int p, int d)
{
	vis[p] = 1;
	double r = 1.0/d;
	for (int i = 0; i < adj[p].size(); ++i) if (!vis[adj[p][i]])
		r += dfs(adj[p][i], d+1);
	return r;
}

int main()
{
	int n, a, b;
	scanf("%d", &n);
	for (int i = 0; i < n-1; ++i)
	{
		scanf("%d%d", &a, &b);
		adj[a].push_back(b);
		adj[b].push_back(a);
	}
	printf("%.10lf\n", dfs(1, 1));
}
