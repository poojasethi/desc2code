#include <cstdio>
#include <iostream>
#include <vector>
using namespace std;

vector<int> e[100010];

double ans;
void DFS(int v,int f,int dep)
{
    ans+=1.0/dep;
    for(int i=0;i<(int)e[v].size();i++)
      if(e[v][i]!=f)
       DFS(e[v][i],v,dep+1);
}

int main()
{
    int i,n,a,b;
    scanf("%d",&n);
    for(i=1;i<n;i++)
    {
        scanf("%d%d",&a,&b);
        e[a].push_back(b);
        e[b].push_back(a);
    }
    DFS(1,0,1);
    printf("%.7lf\n",ans);
}
