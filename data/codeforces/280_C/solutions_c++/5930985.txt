#include<cstdio>
#include<vector>
#include<algorithm>
#include<iostream>
#include<cstring>
using namespace std;
#define N 200000
int n,dep[N];
vector<int>G[N];
void dfs(int u,int f){
	for(int i=0;i<G[u].size();i++)
		if(G[u][i]!=f)
			dep[G[u][i]]=dep[u]+1,dfs(G[u][i],u);
}
int main(){
	scanf("%d",&n);
	for(int i=1,u,v;i<n;i++){
		scanf("%d%d",&u,&v);
		G[u].push_back(v);
		G[v].push_back(u);
	}dep[1]=1;
	dfs(1,0);
	double ans=0;
	for(int i=1;i<=n;i++)ans+=1.0/dep[i];
	printf("%.9f\n",ans);
}
