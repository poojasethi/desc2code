#include <cstdio>

int n, e[200000], a[100001], b[200000], d[100001];

void dfs(int p, int v) {
	d[v] = p;
	for (int i = a[v]; i; i = b[i])
		if (d[e[i]] == 0)
			dfs(p + 1, e[i]);
}

int main() {
	scanf("%d", &n);
	for (int i = 1; i < n; i++) {
		scanf("%d%d", e + i, e + i + n);
		b[i + n] = a[e[i]];
		a[e[i]] = i + n;
		b[i] = a[e[i + n]];
		a[e[i + n]] = i;
	}
	dfs(1, 1);
	double ans = 0;
	for (int i = 1; i <= n; i++)
		ans += 1.0 / d[i];
	printf("%f\n", ans);
}
