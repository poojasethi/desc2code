#include<iostream>
#include<vector>
#include<cstdio>
using namespace std;
const int N = 100005;
int d[N];
vector<int> G[N];
int dfs(int u,int f){
	d[u] = d[f] + 1;
	for(int i = 0; i < G[u].size(); i++){
		int v = G[u][i];
		if(v == f) continue;
		dfs(v,u);
	}
}
int main(){
	int n;
	cin >> n;
	for(int i = 1; i < n; i++){
		int u,v;
		cin >> u >> v;
		G[u].push_back(v);
		G[v].push_back(u);
	}
	dfs(1,0);
	double ans = 0;
	for(int i = 1; i <= n; i++)
		ans += 1.0/d[i];
	printf("%.10lf\n",ans);
}
