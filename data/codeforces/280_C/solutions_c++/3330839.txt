#include <cstdio>
#include <vector>

using namespace std;

double sol;
const int maxn = 100100;
std::vector<int> E[maxn];

void Recursion(int f, int n, int d){
	sol += 1.0 / d;
	for (std::vector<int>::iterator it = E[n].begin(); it != E[n].end(); ++it)
		if (*it != f)
			Recursion(n, *it, d + 1); 
}

int main(){
	sol = 0;
	int N;
	scanf("%d", &N); 
	for(int a,b,i = 0; i < N - 1; ++i){
		scanf("%d%d", &a, &b);
		E[a].push_back(b);
		E[b].push_back(a);
	}

	Recursion(1, 1, 1);
	printf("%lf\n", sol);
}
