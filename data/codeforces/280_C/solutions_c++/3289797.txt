#include<cstdio>
#define N 200005
int n,x,y,l,h,t,d[N],f[N],ed[N],next[N],son[N]; double ans;
int main()
{
	scanf("%d",&n),t=d[1]=f[1]=1;
	for (int i=1;i<n;i++) scanf("%d%d",&x,&y),
		ed[++l]=y,next[l]=son[x],son[x]=l,
		ed[++l]=x,next[l]=son[y],son[y]=l;
	while (h<t){
		int i=f[++h];
		for (int p=son[i];p;p=next[p]){
			int j=ed[p]; if (!d[j]) d[j]=d[i]+1,f[++t]=j;
			}
		}
	for (int i=1;i<=n;i++) ans+=1./d[i];
	printf("%.9lf\n",ans); return 0;
}
