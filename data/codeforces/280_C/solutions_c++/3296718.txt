#include<cstdio>
#include<algorithm>
#include<vector>
#define MAXN 100010

using namespace std;

vector<int> g[MAXN];
int n;
double ac;

void dfs(int mom, int u, int dep)
{
	ac += 1 / (double)dep;
	for(vector<int>::iterator v = g[u].begin(); v != g[u].end(); v++)
		if(*v != mom)
			dfs(u, *v, dep + 1);
}

int main()
{
	scanf("%d", &n);
	for(int i = 1; i < n; i++)
	{
		int u, v;
		scanf("%d %d", &u, &v);
		g[u].push_back(v), g[v].push_back(u);
	}
	dfs(-1, 1, 1);
	printf("%.20lf\n", ac);
}
