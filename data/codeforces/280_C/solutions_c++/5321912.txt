#include <cstdio>
#include <vector>
using namespace std;
double ans;
int n;
vector <int> p[100010];
void dfs(int u,int v,int tol)
{
    ans+=1.0/tol;
    for(int i=0;i<p[u].size();i++)
        if(p[u][i]!=v)
            dfs(p[u][i],u,tol+1);
}
int main()
{
    scanf("%d",&n);
    for(int i=0,a,b;i<n-1;i++)
        scanf("%d%d",&a,&b),p[a].push_back(b),p[b].push_back(a);
    dfs(1,0,1);
    printf("%.10f\n",ans);
}
