#include <iostream>
#include <cstring>
#include <cmath>
#include <vector>
#include <cstdio>
using namespace std;

int n;
vector <int> adj[100100];
double res;

void dfs(int u, int par, int d)
{
	res += 1.0/d;
	for(int i = 0; i < adj[u].size(); i ++)
		if(adj[u][i] != par)
			dfs(adj[u][i], u, d+1);
}

int main()
{
	int x, y;
	
	cin >> n;
	for(int i = 1; i < n; i ++)
	{
		cin >> x >> y;
		adj[x].push_back(y);
		adj[y].push_back(x);
	}
	
	res = 0.0;
	dfs(1, -1, 1);
	printf("%0.9lf\n", res);
	
	return 0;
}
