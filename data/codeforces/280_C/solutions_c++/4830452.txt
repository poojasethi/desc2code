#include <iostream>
#include <iomanip>
#include <vector>
using namespace std;

int n, a, b;
double res = 0;
vector<int> adj[100005];

void dfs(int cur, int prev, int dep){
    res+=1.0/dep;
    for(int i = 0; i<adj[cur].size(); i++) if(adj[cur][i]!=prev) dfs(adj[cur][i], cur, dep+1);
}

int main(){

    cin >> n;
    for(int i = 1; i<n; i++){
        cin >> a >> b;
        a--; b--;
        adj[a].push_back(b);
        adj[b].push_back(a);
    }
    dfs(0, -1, 1);
    cout << fixed << setprecision(15) << res;

}
