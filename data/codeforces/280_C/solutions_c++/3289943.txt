#include <cstdio>
#define NE(x, y) e[++nes]=y, h[nes]=s[x], s[x]=nes
using namespace std;

const int N = 100005;
int n, s[N], e[N*2], h[N*2], nes = 0;
double ans = 0;
inline void dfs(int x, int f, int d) {
	ans += 1.0 / d;
	for (int w=s[x]; w; w=h[w]) {
		if (e[w] != f)
			dfs(e[w], x, d+1);
	}
}
int main() {
	int x, y;
	scanf("%d", &n);
	for (int i=1; i<n; ++i) {
		scanf("%d%d", &x, &y);
		NE(x, y); NE(y, x);
	}
	dfs(1, 0, 1);
	printf("%.9f\n", ans);
	return 0;
}
