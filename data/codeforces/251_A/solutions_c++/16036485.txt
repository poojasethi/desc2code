#include <cstdio>
#define LL long long

int n,d,l,r,a[100010];
LL ans;

int main() {
	scanf("%d%d",&n,&d);
	for (int i=1;i<=n;++i) scanf("%d",&a[i]);
	int r=1;
	for (int l=1;l<=n;++l) {
		while ((r<n)&&(a[r+1]<=a[l]+d)) ++r;
		if (l+2<=r) ans+=(LL)(r-l-1)*(LL)(r-l)/2ll;
	}
	printf("%I64d\n",ans);
	return 0;
}
