#include <bits/stdc++.h>
using namespace std;
long long n,i,d,ans,j,s[100005];
int main()
{
    cin >> n >> d;
    for(i=0;i<n;i++)
    {
        cin >> s[i];
    }
    for(i=0;i<n;i++)
    {
        while(s[i]-s[j]>d)
            j++;
        ans+=((i-j)*(i-j-1))/2;
    }
    cout << ans;
    return 0;
}
