#include<stdio.h>
#include<iostream>
using namespace std;

int main()
{
    int n,i,j,x[100005],d;
    long long ans=0;
    scanf("%d%d",&n,&d);
    for(i=0;i<n;++i) scanf("%d",&x[i]);
    j=n-1;
    for(i=n-1;i>=0;--i)
    {
        while(x[j]-x[i]>d && i<j) --j;
        ans+=1ll*(j-i)*(j-i-1)/2;
    }
    cout<<ans;
}
