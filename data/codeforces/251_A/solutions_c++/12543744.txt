#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,d;
	cin>>n>>d;
	int arr[n];
	for(int i=0;i<n;i++)
		cin>>arr[i];
	long long int ans=0;
	int i=0,j=0;	
	for(i=0;i<n;i++)
	{
		for(;j<n && (arr[j]-arr[i])<=d;j++);
			
		ans+=((long long)(j-i-1)*(j-i-2)/2);		
	}
	cout<<ans;
	return 0;
}
