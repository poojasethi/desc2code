#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long LL;
int main()
{
	//freopen("input.txt","r",stdin);
	LL s,x;
	cin>>s>>x;
	if((s-x)/2*2!=(s-x)||(s-x)<0||(s-x)%2)
	{
		cout<<"0"<<endl;
		return 0;
    }
    if(((s-x)/2)&x)
    {
    	printf("0\n");return 0;
	}
	LL ans=1LL<<__builtin_popcountll(x);
	if(!(s-x)) ans-=2;
	cout<<ans<<endl;
	return 0;
}
