#include <iostream>
#include <cstdio>
using namespace std;
long long a[100],i,j,s,x,n,ans=0;
int main()
{
		scanf("%I64d%I64d\n",&s,&x);
		for (i=x;i;i>>=1) a[++n]=i&1;
		if (s>=x && !((s-x)&1))
		{
			ans=1;
			for (j=s-=x,i=1;s;s>>=1,++i)
				if (s&1 && a[i-1]) break;
			if (s) ans=0;
			for (i=1;i<=n;++i)
				if (a[i]) ans*=2;
			if (!j) ans-=2;
		}
		printf("%I64d\n",ans);
}
