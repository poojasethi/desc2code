#include <cstdio>

long long vs,vx;
int bitsum(long long v) {
	if(v) return bitsum(v/2)+v%2;
	return 0;
}
int main() {
	scanf("%I64d%I64d",&vs,&vx);
	if(vs<vx) {
		printf("0\n");
	} else {
		vs-=vx;
		if(vs&1) {
			printf("0\n");
		} else if((vs/2)&vx) {
			printf("0\n");
		} else {
			long long sol=1;
			for(int i=0;i<bitsum(vx);i++) sol*=2;
			if(vs==0) sol-=2;
			printf("%I64d\n",sol);
		}
	}
	return 0;
}
