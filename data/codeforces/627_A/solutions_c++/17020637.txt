    }

    long long a = (s-x)/2;
    for(int i=0;i<41;i++)
        if(x>>(1ll*i)&1)
            if(a>>(1ll*i)&1){
                cout<<0<<endl;
                return 0;
            }else ans*=2;

    if((s^0) == x) ans-=2;
    cout<<ans<<endl;

}
