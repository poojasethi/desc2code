#include <iostream>

using namespace std;

typedef long long ll;
int main(){
	ll s, x;
	cin>> s >> x;
	ll val = (s - x) >> 1;               // sum = a^b + 2*(a&b)
	if( (val << 1) + x != s  ||  val < 0 || (val & x) != 0){
		cout<< 0 <<endl;
	} else {
		ll n = 0, ways = 1;
		while(x){
			n += (x & 1);
			x >>= 1;
		}
		ways <<= n;			//	ways = 2^n 
		if(val == 0)  ways -= 2;
		cout<< ways <<endl;
	}
	return 0;
}
