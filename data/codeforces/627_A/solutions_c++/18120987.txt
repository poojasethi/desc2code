#include <stdio.h>
#include <math.h>

int main(){
	long long s, x;
	scanf("%I64d %I64d", &s, &x);
	long long z = s - x;
	if (z < 0 || z&1 || (z/2)&x) {puts("0"); return 0;}
	int n = 0;
	long long t = x;
	while (t)
	{
		if (t % 2) ++n;
		t/=2;
	}
	long long ans = 1LL<<n;
	ans -= 2LL*(s == x);
	printf("%I64d\n", ans);
}
