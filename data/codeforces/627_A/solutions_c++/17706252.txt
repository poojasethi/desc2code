#include<bits/stdc++.h>

long long s, x, t, g, Res;
int main(){
    int i;
    std::cin>>s>>x;
    if((s-x)%2!=0 || s<x){
        std::cout<<0;
        return 0;
    }
    t = (s-x)/2;
    if(t&x){
        std::cout<<0;
        return 0;
    }
    Res = 1;
    while(x){
        if(x%2)Res*=2;
        x/=2;
    }
    if(!t)Res-=2;
    std::cout<<Res;
}
