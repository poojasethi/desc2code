#include<bits/stdc++.h>
using namespace std;
int main()
{
	unsigned long long int s,x,save,i,a,n=1;
	cin>>s>>x;
	a=(s-x);
	if(a%2==1) {cout<<"0\n"; return 0;}
	a=a/2;
	save=x;
	while(x>0)
	{
		if((x%2)==1 && a%2==1) {cout<<"0\n";return 0;}
		if((x%2)==1 && a%2==0) n*=2;
		x=x/2;a=a/2;
	}
	if(s==save) n-=2;
	cout<<n<<"\n";
	return 0;
}
