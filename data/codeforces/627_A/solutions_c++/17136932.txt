#include <stdio.h>
#define int long long

int f(int a,int b){
	if(a%2!=b%2 || a<b) return 0;
	if(a<2) return a+1;
	if(a%2==1) return 2*f(a/2,b/2);
	return f(a/2-1,b/2)+f(a/2,b/2);
}
	

main(){
	int a,b,resp;
	scanf("%I64d %I64d",&a,&b);
	resp=f(a,b);
	if(a==b) resp-=2;
	printf("%I64d\n",resp);
	return 0;
}
