#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
int main()
{
    ll sum,x;
    cin>>sum>>x;
    ll a=(sum-x)/2;
    if(a<0||a*2+x!=sum||(a&x)!=0){
        cout<<0<<endl;return 0;
    }
    int t=0;
    while(x){
        t+=(x&1);
        x>>=1;
    }
    ll ans=1ll<<t;
    if(a==0)ans-=2;
    cout<<ans<<endl;
}
