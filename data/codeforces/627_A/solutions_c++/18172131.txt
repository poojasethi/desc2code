#include <bits/stdc++.h>
using namespace std;

int main() {
	long long sum,x,a;
	cin>>sum>>x;
	a = (sum - x)/2;
	if(a < 0 || (2*a+x != sum) || (x&a) !=0)
	{
		cout<<0;
		return 0;
	}
	long long tmp = x;
	long long bits = 0;
	while(tmp > 0)
	{
		bits += tmp&1;
		tmp>>=1;
	}
	long long ans = 1LL << bits;
	if(a == 0)ans -= 2;
	cout<<ans;
	return 0;
}
