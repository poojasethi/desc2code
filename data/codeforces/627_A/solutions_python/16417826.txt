s, x = map(int, raw_input().split())
if s == x: rem = 2
else: rem = 0
p = []
tot = 0
cur = 1
for i in xrange(64):
    if x % 2 == 1:
        tot += 1
        s -= cur
    else:
        p.append(2*cur)
    cur *= 2
    x /= 2

for i in p[::-1]:
    if s >= i: s -= i

if s == 0: print 2**tot - rem
else: print 0
