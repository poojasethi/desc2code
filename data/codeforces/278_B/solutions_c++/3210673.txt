#include <iostream>
#include <string>
#include <vector>
#include <queue>
#include <map>
#include <set>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string dic[32];
	for(int i=0;i<n;i++)
		cin>>dic[i];
	string ans="";
	queue<string> que;
	que.push(ans);
	while(!que.empty())
	{
		string cur=que.front();
		que.pop();
		bool flag=true;
		for(int i=0;i<n;i++)
			if(dic[i].find(cur)!=string::npos)
			{
				flag=false;
				break;
			}
		if(flag)
		{
			cout<<cur<<endl;
			break;
		}
		for(int i=0;i<26;i++)
			que.push(cur+(char)('a'+i));
	}
	return 0;
}
