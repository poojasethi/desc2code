#include<cstdio>

int n;
char s[33];
int v1[33], v2[33][33];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%s", s);
		for(int j=0;s[j];j++) {
			v1[s[j]-'a'] = true;
			if(j)
				v2[s[j-1]-'a'][s[j]-'a'] = true;
		}
	}
	for(int i=0;i<26;i++)
		if(!v1[i]) {
			printf("%c\n", i+'a');
			return 0;
		}
	for(int i=0;i<26;i++)
		for(int j=0;j<26;j++)
			if(!v2[i][j]) {
				printf("%c%c\n", i+'a', j+'a');
				return 0;
			}
	
	return 0;
}
