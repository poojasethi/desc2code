#include <iostream>
#include <string>
using namespace std;

int main()
{
  int n,i,k,l,m,j,x,f;
  string ans,a[101];
  cin>>n;
  for(i=1; i<=n; i++) cin>>a[i];
  k=0;
  ans="";
  m=26;
  l=-1;
  while(true)
  {
    l++;
    ans+='a';
    for(i=0; i<=l; i++) ans[i]='a';
    ans[l]--;
    m*=26;
    for(k=0; k<m; k++)
    {
       ans[l]++;
       j=l;
       f=-1;
       while(ans[j]>'z'){ans[j-1]++; ans[j]='a'; j--;}
       for(i=1; i<=n; i++) if((int)a[i].find(ans)>f){f=1;break;}
       if(f<0){cout<<ans<<endl;return 0;}
    }
  }
}
