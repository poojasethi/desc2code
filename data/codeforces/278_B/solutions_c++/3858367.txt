#include <iostream>
#include <queue>
#include <string>

using namespace std;

string s[35];
queue <string> q;

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> s[i];
	for (int i = 0; i < 26; i++)
	{
		string t = "";
		t += char(i + 'a');
		q.push(t);
	}
	while (1)
	{
		string f = q.front();
		q.pop();
		bool sub = false;
		for (int i = 0; i < n; i++)
			if (s[i].find(f) != std::string::npos)
			{
				sub = true;
				break;
			}
		if (!sub)
		{
			cout << f << endl;
			return 0;
		}
		for (int i = 0; i < 26; i++)
			q.push(f + char(i + 'a'));
	}
}
