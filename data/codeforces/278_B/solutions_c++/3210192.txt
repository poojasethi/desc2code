#include<iostream>
#include<string>
using namespace std;

string s;
int i,j,n,m,v[100],w[1000];

int main()
{
	cin >> n;
	for (i=1;i<=n;i++)
	{
		cin >> s;
		m=s.length();
		for (j=0;j<m;j++)
		{
			v[s[j]-'a']=1;
			if (j!=m-1)
				w[(s[j]-'a')*26+s[j+1]-'a']=1;
		}
	}
	for (i=0;i<26;i++)
		if (v[i]==0)
		{
			cout << (char)(i+'a');
			return 0;
		}
	for (i=0;i<676;i++)
		if (w[i]==0)
		{
			cout << (char)(i/26+'a') <<  (char)(i%26+'a');
			return 0;
		}
}
