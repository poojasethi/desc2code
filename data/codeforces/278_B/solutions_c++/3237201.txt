#include <iostream>
using namespace std;
string nextl(string s)
{int i=s.size(); i--;
 while (i>=0)
  {s[i]++;
   if (s[i]>'z') {s[i]='a';i--;} else break;
  }
if (i==-1) s.push_back('a');
 return s;
}
int main()
{int n; cin>>n;
string a[31];
for (int i=0;i<n;i++) cin>>a[i];

string s="a";
while (1)
{ int i;
 for (i=0;i<n;i++)
  {int j=a[i].find(s);
   if (j!=-1) break;
  }
 if (i==n) break; else s=nextl(s);
}
cout<<s;
return 0;
}
