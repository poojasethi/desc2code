#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int n;  char u[300][300],s[100];
main(){
	cin>>n;
	for(int i=0;i<n;++i){
		cin>>s;  int l=(int)strlen(s);
		for(int i=0;i<l;++i)
			u[s[i]][0]=1;
		for(int i=1;i<l;++i)
			u[s[i-1]][s[i]]=1;
	}
	for(char a='a';a<='z';++a)
		if (!u[a][0]){
			cout<<a;
			return 0;
		}
	for(char a='a';a<='z';++a)
	for(char b='a';b<='z';++b)
		if (!u[a][b]){
			cout<<a<<b<<endl;
			return 0;
		}
}
