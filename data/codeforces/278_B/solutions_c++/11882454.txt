#include <bits/stdc++.h>
typedef long long unsigned ll;
using namespace std;
set<string> se;
set<char> se2;
int main(){
	int n;
	cin >> n;
	string s;
	while(n--){
		cin >> s;
		int l = s.length();
		for(int i=0;i<l-1;i++){
			se2.insert(s[i]);
			se.insert(s.substr(i,2));
		}
		se2.insert(s[l-1]);
	}
	for(char c='a';c<'a'+26;c++){
		if(se2.find(c)==se2.end()){
			cout << c;
			return 0;
		}
	}
	for(char c='a';c<'a'+26;c++){
		for(char d='a';d<'a'+26;d++){
			string st="";
			st+=c;
			st+=d;
			if(se.find(st)==se.end()){
				cout << st;
				return 0;
			}
		}
	}


}
