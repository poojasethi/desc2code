#include<iostream>
#include<string>
#include<set>
using namespace std;
set<string> used;
int n,i,j;
int main(){
        string ans="a",s;
	cin>>n;
	while(n--){
		cin>>s;
		for(i=0;i<s.length();i++)
		for(j=0;j<=s.length();j++)
		used.insert(s.substr(i,j));
	}
	while(used.find(ans)!=used.end()){
		ans[ans.length()-1]++;
		for(i=ans.length()-1;i>0;i--)
		if(ans[i]>'z'){
			ans[i]='a';
			ans[i-1]++;
		}
		if(ans[0]>'z'){
			for(i=0;i<ans.length();i--)
			ans[i]='a';
			ans+='a';
		}
	}
	cout<<ans;
	return 0;
}
