#include<cstdio>
#include<iostream>
using namespace std ;
int main()
{
	int n,i,j;
	char a;
	int arr[700]={0},brr[26]={0},f=0;

	scanf("%d",&n);
	scanf("%c",&a);
	
	for(i=0;i<n;i++)
	{
		char x,c;
		x=getchar();
		brr[x-'a']=1;
		while((c=getchar())!='\n')
		{
			brr[c-'a']=1;
			arr[(x-'a')*26+c-'a']=1;
			x=c;
		}
	}
		for(j=0;j<26;j++)
		{
			if(brr[j]==0)
			{	printf("%c\n",j+'a');f=1;break;}
			
		}
		if(f==0)
		{
			for(j=0;j<676;j++)
			{
				if(arr[j]==0)
				{	printf("%c%c\n",j/26+'a',j%26+'a');
					break;
				}
			}
		}

}
