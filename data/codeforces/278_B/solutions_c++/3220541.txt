#include<iostream>
#include<string>
using namespace std;
int ch[999];
int main(){
    int n;
    cin>>n;
    string s;
    for(int i=0;i<n;i++){
        cin>>s;
        int l=s.length();
        for(int j=0;j<l;j++)
            ch[s[j]-'a']++;
        for(int j=0;j<l-1;j++)
            ch[(s[j]-'a'+1)*26+s[j+1]-'a']++;
    }
    for(int i=0;i<=777;i++)
        if(!ch[i]){
            if(i<26)
                cout<<char(i+'a');
            else
                cout<<char(i/26+'a'-1)<<char(i%26+'a');
            return 0;
        }
}
