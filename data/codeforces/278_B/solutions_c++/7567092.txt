#include<iostream>
#include<string.h>
using namespace std;
char s[21],str[30][21];
int n;
int mkstr(int min, char ch, int j)
{
int i,x,k;
if(j<min)
{
s[j++]=ch;
for(i=0; i<26; i++)
if(mkstr(min,'a'+i,j));
else
return 0;
}
else
{
s[j++]=ch;
s[j]='\0';
for(k=0; k<n; k++)
if(strstr(str[k],s))
break;
if(k==n)
{
cout<<s<<' ';
x=0;
}
else
x=1;
}
return x;
}
int main()
{
int i,x=1,j=0;
cin>>n;
for(i=0; i<n; i++)
cin>>str[i];
while(x)
{
j++;
for(i=0; i<26 && x; i++)
x=mkstr(j-1,'a'+i,0);
}
//cout<<'\n';
return 0;
}
