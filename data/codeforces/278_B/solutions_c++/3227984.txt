#include <iostream>
#include <cstdlib>
#include <queue>

using namespace std;

int main()
{
	string ans="", dic[35];
	int n;
	cin>>n;
	for(int i=1; i<=n; i++) cin>>dic[i];
	queue<string> q;
	q.push(ans);
	while(!q.empty())
	{
		for(char c='a'; c<='z'; c++)
		{
			string s=q.front()+c;
			bool flag=true;
			for(int i=1; i<=n; i++) 
			{
//				cout<<s<<" "<<dic[i].find(s)<<endl;
				if(dic[i].find(s)!=string::npos) {flag=false; break;}
			}
			if(flag) { cout<<s<<endl; return 0; }
				else q.push(s);
		}
		q.pop();
	}
	return 0;
}
