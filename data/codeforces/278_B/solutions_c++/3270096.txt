// KUNAL ACHARYA (sub_zer0) :)
#include<cstdio>
#include<iostream>
#include<string>
#include<cmath>
using namespace std;
int main()
{
string s,a,b,c;
int n,m,k,i,j;
scanf("%d",&n);
for(k=0;k<n;++k)
{
cin>>s;
a+='?';
a+=s;
}
for(i='a';i<='z';++i)
{
b=i;                     
if(a.find(b)==-1)
{
cout<<b<<endl;
//system("pause");
return 0;
}
}
for(i='a';i<='z';++i)
for(j='a';j<='z';++j)
{
b=i;c=j;b=b+c;                     
if(a.find(b)==-1)
{
cout<<b<<endl;
//system("pause");
return 0;
}
}
}
