#include<string>
#include<iostream>
#include<set>
using namespace std;
set<string> used;
int main()
{
	int n;
	cin>>n;
	while(n--){
		string x;
		cin>>x;
		for(int i=0;i<x.length();i++)
		for(int j=0;j<=x.length();j++)
		used.insert(x.substr(i,j));
	}
	string ans="a";
	while(used.find(ans)!=used.end()){
		ans[ans.length()-1]++;
		for(int i=ans.length()-1;i>0;i--)
		if(ans[i]>'z'){
			ans[i-1]++;
			ans[i]='a';
		}
		if(ans[0]>'z'){
			for(int i=0;i<ans.length();i++)
			ans[i]='a';
			ans+='a';
		}
	}
	cout<<ans;
	return 0;
}
