#include<string>
#include<iostream>
#include<algorithm>
using namespace std;

string s;
const int lim=26;
int i,j,n,m,v[100],w[1000];

int main()
{
	cin>>n;
	for (i=1;i<=n;i++)
	{
		cin>>s;	m=s.length();
		for (j=0; j<m; j++)
		{
			v[s[j]-'a']=1;
			if (j!=m-1)	w[(s[j]-'a')*26+s[j+1]-'a']=1;
		}
	}
	for (i=0;i<lim;i++)	if (v[i]==0)
	{
		cout << (char)('a'+i);
		return 0;
	}
	for (i=0;i<lim*lim;i++)	if (w[i]==0)
	{
		cout << (char)('a'+i/26) << (char)('a'+i%26);
		return 0;
	}
	return 0;
}
