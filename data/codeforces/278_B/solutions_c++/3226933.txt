#include<iostream>
#include<string>
using namespace std ;

string s ;
int n,tab[26],tab2[26][26] ;
int main(){
	cin>>n ;
	for(int i=0 ; i<n ; i++){
		cin>>s ;
		tab[s[0]-97]=1 ;
		for(int j=1 ; j<s.size() ; j++){
			tab2[s[j-1]-97][s[j]-97]=1 ;
			tab[s[j]-97]=1 ;
			}
				
		}
	
	for(int i=0 ; i<26 ; i++){
		if(tab[i]==0){
			cout<<char(i+97) ;
			return 0 ;
		}
	}
	for(int i=0 ; i<26 ; i++)
		for(int j=0 ; j<26 ;j++){
			if(tab2[i][j]==0){
				cout<<char(i+97)<<char(j+97) ;
				return 0;
			}
		}
}
