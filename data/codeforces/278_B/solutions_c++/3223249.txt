#include<stdio.h>
#include<string.h>
#define N 27

bool vis1[N],vis2[N][N];
char str[25];

int main()
{
	int n;
	int i,j,k;
	scanf("%d",&n);
	while(n--)
	{
		scanf("%s",str);
		int len=strlen(str);
		for(i=0;i<len;i++)
			vis1[str[i]-'a']=true;
		for(i=0;i<len-1;i++)
			vis2[str[i]-'a'][str[i+1]-'a']=true;
	}
	for(i=0;i<26;i++)
		if(vis1[i]==0)
		{
			printf("%c\n",i+'a');
			return 0;
		}
	for(i=0;i<26;i++)
		for(j=0;j<26;j++)
			if(vis2[i][j]==0)
			{
				printf("%c%c\n",i+'a',j+'a');
				return 0;
			}
	return 0;
}
