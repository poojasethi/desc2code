#include<iostream>
#include<string>
using namespace std;
int n, j, l;
char i, k;
string s, s1, f1, f2, h;
int main(){
    cin>>n;
for(j=0; j<n; j++){
    cin>>s1;
    s=s+'@';
    s=s+s1;
}
for(i='a';i<='z';i++){
    l=s.find(i);
    if(l==-1){
        cout<<i; return 0;
        }
}
for(i='a';i<='z';i++){
    for(k='a';k<='z';k++){
        f1=i; f2=k;
        h=f1+f2;
        l=s.find(h);
        if(l==-1){
            cout<<h; return 0;
        }
    }
}

}
