#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	char s[35][35],t[676][3],flag=0;
	int n,i,j,pos;
	cin>>n;
	for(i=0;i<n;i++)
		scanf("%s",s[i]);
	for(i=0;i<26;i++)
	{
		t[i][0]=(char)(i+97);
		t[i][1]='\0';
	}
	for(i=26;i<676;i++)
	{
		t[i][0]=(char)((i-26)/26+97);
		t[i][1]=(char)(i%26+97);
		t[i][2]='\0';
	}
	//for(i=0;i<676;i++)
	//	cout<<t[i]<<endl;
	for(i=0;i<676;i++)
	{
		for(j=0;j<n;j++)
			if(strstr(s[j],t[i]))
				break;
		if(j==n)
		{
			pos=i;
			break;
		}
	}
	cout<<t[pos]<<endl;
	return 0;
}
