#include<cstdio>
#include<cstring>
const int N=30,S=21;
char y[S];
int n;
char s[N][S];
bool f(int e,int d=0){
	if(d==e){
		bool r=0;
		for(int i=0;i<n;i++)
			if(strstr(s[i],y)){
				r=1;
				break;
			}
		if(r==0){
			y[e]=0;
			printf("%s\n",y);
			return 1;
		}
		return 0;
	}
	for(int i='a';i<='z';i++){
		y[d]=i;
		if(f(e,d+1))
			return 1;
	}
}
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%s",s[i]);
	for(int x=1;;x++)
		if(f(x))
			break;
	return 0;
}
