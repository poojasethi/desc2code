#include <cstdio>

bool had1[130];
bool had2[130][130];
int n;

int main()
{
	char str[1000];
	int i, j;

	scanf("%d", &n);
	for (i = 0; i < n; ++i) {
		scanf("%s", str);
		for (j = 0; str[j] != '\0'; ++j) {
			had1[str[j]] = true;
			had2[str[j]][str[j+1]] = true;
		}
	}
	for (i = 'a'; i <= 'z'; ++i)
		if (!had1[i]) {
			printf("%c\n", i);
			break;
		}
	if (i == 'z'+1)
		for (i = 'a'; i <= 'z'; ++i) {
			for (j = 'a'; j <= 'z'; ++j)
				if (!had2[i][j]) {
					printf("%c%c\n", i, j);
					break;
				}
			if (j <= 'z')
				break;
	}
	return 0;
}
