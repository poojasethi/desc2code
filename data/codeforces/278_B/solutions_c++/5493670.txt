#include<iostream>
#include<string>
using namespace std;
string inttostr(int in) {
	string ret;
	while(in > 0) {
		if(in%27 == 0) return "a";
		ret=string(1, char('a'-1+in%27)) + ret;
		in /= 27;
	}
	
	return ret;
}
int main() {
	int n, cr=1;
	string prev[30], cstr;
	cin >> n;
	bool found = true;
	for(int i=0;i<n;i++)
		cin >> prev[i];
	while(found) {
		found=false;
		cstr=inttostr(cr);
		for(int i=0;i<n;i++)
			if(prev[i].find(cstr) != -1)
				found=true;
		cr++;
	}
	cout << cstr;
	return 0;
}
