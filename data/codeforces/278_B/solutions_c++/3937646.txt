#include<iostream>
#include<string>
using namespace std;
int main()
{
string mass[41];
int n,i,j;
char c,cc;
cin>>n;
for(i=0;i<n;i++)cin>>mass[i];
for(c='a';c<='z';c++)
{
int tmp=0;
for(j=0;j<n;j++){if(mass[j].find(c)!=-1){tmp=1;break;}}
if(!tmp){cout<<c<<endl;return 0;}
}
for(c='a';c<='z';c++)
{
for(cc='a';cc<='z';cc++)
{
string str;
str+=c;
str+=cc;
int tmp=0;
for(j=0;j<n;j++){if(mass[j].find(str)!=-1){tmp=1;break;}}
if(!tmp){cout<<str<<endl;return 0;}
}
}
return 0;
}
