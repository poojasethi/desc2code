#include <cstdlib>
#include <iostream>
#include <string>

using namespace std;

int mas[702];

int main(int argc, char *argv[])
{int i,N;
cin>>N;
string S;
while(N--)
   {cin>>S;
    
    mas[S[0]-'a']=1;
    for(i=1;i<S.size();i++)
      {int x=S[i]-'a';
       mas[x]=1;
       mas[(S[i-1]-'a'+1)*26+x]=1;
      }
    }

    for(i=0;i<702;i++)
      if(!mas[i])break;
    if(i<26)cout<<char('a'+i);
    else cout<<char('a'+i/26-1)<<char('a'+i%26);

    
    return EXIT_SUCCESS;
}
