from sys import stdin

def line():
	return stdin.readline().strip()

def increment_list(s, base=0, top=9):
	n = len(s)
	for i in reversed(xrange(n)):
		if s[i] == top:
			s[i] = base
		else:
			s[i] += 1
			return s
	s.append(base)
	return s

n = int(line())
s = []
for _ in xrange(n):
	s.append(line())

con = ";".join(s)
ans = "a"
while True:
	if con.find(ans) < 0:
		break
	ans = "".join([chr(x) for x in increment_list([ord(x) for x in ans], base=ord('a'), top=ord('z'))])
print ans
