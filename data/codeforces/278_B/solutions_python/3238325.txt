from re import *
from itertools import *

# n < 30, so the result.length <= 2
seq = [chr(ord('a') + i) for i in range(26)]
for x in product(seq, seq):
	seq.append(''.join(x))
n = input()
title  = [raw_input() for i in range(n)]
print [result for result in seq if not any(search(result, y) for y in title)][0]
