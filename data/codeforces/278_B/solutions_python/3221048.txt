n = int(raw_input())
S = []
for i in range(n): S += [raw_input()]

for x in range(ord('a'), ord('z') + 1):
    flag = 1
    for i in range(n):
        if chr(x) in S[i]: flag = 0 
    if flag > 0:
        print chr(x)
        exit()
for x in range(ord('a'), ord('z') + 1):
    for y in range(ord('a'), ord('z') + 1):
        z = chr(x) + chr(y)
        flag = 1
        for i in range(n):
            if str(z) in S[i]: flag = 0 
        if flag > 0:
            print z
            exit()
