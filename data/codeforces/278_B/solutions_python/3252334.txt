n = int(raw_input())
s = []
a = map(chr, range(ord('a'), ord('z')+1))
for i in range(n):
    s.append(raw_input())

hash = {}
for i in s:
    for j in range(1, len(i)+1):
        for z in range(len(i)-j+1):
            hash[i[z:z+j]] = 1

i = 1
while 1:
    t = i
    res = ''
    while t:
        res = a[(t%len(a)) - 1] + res
        t = int(t / (len(a)+1))
    if not hash.get(res):
        print res
        break
    i += 1
