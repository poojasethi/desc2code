n  = int(input())
ss =[raw_input().rstrip() for i in range(n)]
se = set()
for s in ss:
    for i in range(len(s)):
        se.add(s[i])
    for i in range(0,len(s)):
        se.add(s[i:i+2])
for i in range(26):
    if chr(i+ord('a')) not in se:
        print chr(i+ord('a'))
        exit()
for i in range(26):
    for j in range(26):
        nc = chr(i+ord('a'))+chr(j+ord('a'))
        if nc not in se:
            print nc
            exit()
