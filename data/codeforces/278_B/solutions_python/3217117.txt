from collections import deque

n = int(raw_input())
strings = [raw_input() for _ in range(n)]

a = "abcdefghijklmnopqrstuvwxyz"
q = deque([])
for c in a:
    q.append(c)
while True:
    ans = q.popleft()
    for c in a:
        q.append(ans+c)
    for s in strings:
        if ans in s:
            break
    else:
        break

print ans
