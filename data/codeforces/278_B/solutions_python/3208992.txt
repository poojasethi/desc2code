import sys

inp = sys.stdin.read().strip().splitlines()[1:]

def advance(s):
    if s == '':
        return 'a'
    elif s[-1] == 'z':
        return advance(s[:-1]) + 'a'
    else:
        return s[:-1] + chr(ord(s[-1]) + 1)

res = 'a'
while reduce(lambda x, y: x or y, [res in i for i in inp]):
    res = advance(res)

sys.stdout.write(res)
