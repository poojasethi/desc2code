n=input()
st=set()
for _ in xrange(n):
    s=raw_input()
    l = len(s)
    for x in xrange(1,l+1):
        for i in xrange(0,l):
            st.add(s[i:i+x])


def dfs(s,i,len):
    if i == len: 
        if s not in st: print s; exit()
        return
    for j in xrange(0,26):
        dfs(s+chr(ord('a')+j),i+1,len)

for x in xrange(1,21):
    s=''
    dfs(s,0,x)
