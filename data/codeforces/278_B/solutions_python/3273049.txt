def main():
    n = int(raw_input())
    s = ':'
    for _ in range(n):
        s += raw_input() +':'

    c ='abcdefghijklmnopqrstuvwxyz'
    i = j = 0
    while 1:
        if s.count(c[i]) == 0:
            print(c[i])
            exit()
        i += 1
        if i ==26:break

    i = 0
    while 1:
        if s.count(c[j]+c[i]) == 0:
            print(c[j]+c[i])
            exit()
        i += 1
        if i ==26:
            j += 1
            i = 0

main()
