n = input()
s = ''
a = ['a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z']
for i in xrange(n):
    s = s +' '+ raw_input()
key = 1
metka = 0
for i in xrange(len(a)+1):
    if key:  
        c = ''
    else:
        c = a[i-1]
    for j in xrange(len(a)):
        slovo = c + a[j]
        if s.count(slovo) == 0:
             metka = 1
             break
    key = 0
    if metka ==1 :
        break  
print (slovo)
