from itertools import *

n = int( raw_input() )
al = 'abcdefghijklmnopqrstuvwxyz'
a = []
for i in xrange( n ):
    a.append( raw_input() )
r = 0
ans = False
while not ans:
    r += 1
    for x in product( al, repeat=r):
        f = False
        s = ''.join( x )
        for word in a:
            if s in word:
                f = True
                break
        if not f:
            print s
            ans = True
            break
