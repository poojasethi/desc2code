n = int(raw_input())

s = [raw_input() for i in range(n)]

def testNew(testString):
	for i in range(n):
		for j in range(len(s[i]) + 1 - len(testString)):
			if s[i][j:j+len(testString)] == testString:
				return False
	return True

newFound = False
l = 1

while not newFound:
	for i in range(26**l):
		testString = ''
		for j in range(l):
			testString += chr(ord('a') + (i/(26**(l-1-j)) % 26))
		if testNew(testString):
			print testString
			newFound = True
			break
	l += 1
