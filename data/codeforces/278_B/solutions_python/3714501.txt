#!/usr/bin/env python
from sys import stdin
from string import lowercase
letters = lowercase[:]
letters2 = ""
h = "".join(stdin.readline() for t in xrange(int(raw_input()))).strip()

for char in letters:
    if char not in h:
        print char
        exit()
    else:
        letters2 += char

for a in letters2:
    for n in letters2:
        if a+n not in h:
            print a+n
            exit()
