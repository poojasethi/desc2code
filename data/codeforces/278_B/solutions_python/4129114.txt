s = [raw_input() for _ in range(input())]
tr = lambda i:str(chr(ord('a') + i))
def search(t):
    for i in s:
        if i.count(t) > 0:
            return True
    return False
def find(t):
    for i in range(26):
        if not search(t + tr(i)):
            return t + tr(i)

    answer = ""
    for i in range(26):
        get = find(t + tr(i))
        if get != "":
            if answer == "" or len(answer) > len(get):
                answer = get

    return answer


print find("")
