n = input()
s = [raw_input() for i in range(n)]
def f(ss):
  for i in s:
    if ss in i:
      return True
  return False
for i in range(ord("a"), ord("z") + 1):
  ss = chr(i)
  if not f(ss):
    print ss
    exit()
for i in range(ord("a"), ord("z") + 1):
  for j in range(ord("a"), ord("z") + 1):
    ss = chr(i) + chr(j)
    if not f(ss):
      print ss
      exit()
