#!/usr/bin/env python
from sys import stdin
from string import lowercase
letters = lowercase[:]
h = "".join(stdin.readline() for t in xrange(int(raw_input()))).strip()

for char in letters:
    if char not in h:
        print char
        exit()

for a in letters:
    for n in letters:
        if a+n not in h:
            print a+n
            exit()
