from sys import stdin

def line():
	return stdin.readline().strip()

def incr(s1):
	i = 0
	s = list(s1)
	for i in reversed(xrange(len(s))):
		if s[i] != 'z':
			s[i] = chr(ord(s[i]) + 1)
			if i != len(s1) - 1:
				s[i+1:] = ["a"] * (len(s1) - i - 1)
			return "".join(s)
	return "".join(["a"] * (len(s1) + 1))

n = int(line())
s = []
for _ in xrange(n):
	s.append(line())

con = ";".join(s)
ans = "a"
while True:
	if con.find(ans) < 0:
		break
	ans = incr(ans)
print ans
