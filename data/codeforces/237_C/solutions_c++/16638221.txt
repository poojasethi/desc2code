#include <iostream>
#include <vector>
using namespace std;

const int MAXN = 1000 * 1000 + 10;
vector <int> vec;
bool mark[MAXN];

void FindPrm(int a, int b)
{
	for(int i = 2; i <= b; i ++)
		if(!mark[i])
		{
			if(i >= a)
				vec.push_back(i);
			for(int j = i + i; j <= b; j += i)
				mark[j] = true;
		}
}

int main()
{
	int a, b, k;
	cin>> a >> b >> k;

	FindPrm(a, b);

	if(k > vec.size())
	{
		cout<< "-1\n";
		return 0;
	}

	int ans = max(vec[k - 1] - a + 1, b - vec[vec.size() - k] + 1);
	for(int i = 0; i < vec.size() - k; i ++)
	{
		ans = max(vec[i + k] - vec[i], ans);
	}

	cout<< ans << endl;
}
