#include<stdio.h>
int p[1000001];
int main(void)
{
	int i,j,a,b,k,h,t,l,tmp,flag;
	scanf("%d%d%d",&a,&b,&k);
	for(i=2;i<=b;i++)
		{if(!p[i]){for(j=i+i;j<=b;j+=i){p[j]=1;}}}
	p[0]=0;
	p[1]=0;
	for(i=2;i<=b;i++)
	{
		p[i]=(p[i]+1)%2;
		p[i]+=p[i-1];
    }
	if(p[b]-p[a-1]<k)
	{
		printf("-1\n");
		return 0;
    }
	h=1;
	t=b-a+1;
	while(h<=t)
	{
		flag=1;
		tmp=(t+h)/2;
		for(i=a;i<=b-tmp+1;i++)
			if(p[i+tmp-1]-p[i-1]<k)
			{
				flag=0;
				break;
				}
		if(!flag)	h=tmp+1;
		else
		{
			l=tmp;
			t=tmp-1;
			}
		}
	printf("%d\n",l);
	return 0;
}
