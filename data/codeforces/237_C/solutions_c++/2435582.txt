#include <iostream>
#include <vector>
using namespace std;

bool isprime(int n)
{
    if (n <= 1) return false;
    if (n%2==0) return n==2;
    for (int i=3; i*i<=n; i+=2)
        if (n%i==0) return false;
    return true;
}

int main()
{
    int a, b, k; cin >> a >> b >> k;
    vector <int> p;
    for (int i=a; i<=b; i++)
        if (isprime(i)) p.push_back(i);
    if (k > p.size())
    {
        cout << -1 << endl;
        return 0;
    }

    int ans = 0;
    for (int i=0; i+k<p.size(); i++)
        ans = max(ans, p[i+k]-p[i]);

    ans = max(ans, 1+p[k-1]-a);
    ans = max(ans, 1+b-p[p.size()-k]);
    cout << ans << endl;
    return 0;
}
