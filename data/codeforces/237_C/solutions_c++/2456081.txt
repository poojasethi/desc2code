#include<stdio.h>
#include<algorithm>
#define M 1000000
using namespace std;

int ok[M+1];
int sum[M+1];

int main()
{
	int a,b,k;
	int i,j;
	int l,r;
	int ans=-1;
	for(i=2;i<=M;i++)
	{
		if(!ok[i])
			for(j=i+i;j<=M;j+=i)
				ok[j]=1;
	}
	for(i=2;i<=M;i++)
	{
		sum[i]=sum[i-1];
		if(!ok[i])
			sum[i]++;
	}
	scanf("%d%d%d",&a,&b,&k);
	l=1;r=b-a+1;
	while(l<=r)
	{
		int m=(l+r)/2;
		bool ok=true;
		for(i=a;i<=b-m+1;i++)
		{
			if(sum[i+m-1]-sum[i-1]<k)
			{
				ok=false;
				break;
			}
		}
		if(ok)
		{
			ans=m;r=m-1;
		}
		else 
			l=m+1;
	}
	printf("%d\n",ans);
}
