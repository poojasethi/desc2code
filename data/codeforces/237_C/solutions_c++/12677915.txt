#include <bits/stdc++.h>
using namespace std;
int s[1000001]={0};
long long sum[1000001]={0};
void sieve()
{
	s[0]=1;
	s[1]=1;
	for(int i=0;i*i<=1000001;i++)
		if(s[i]==0)
			for(int j=i*i;j<=1000001;j+=i)
				s[j]=1;
	sum[0]=0;
	sum[1]=0;
	for(int i=2;i<=1000000;i++)
	{
		sum[i]+=sum[i-1];
		if(!s[i])sum[i]++;
	}
}
int main()
{
	int a,b,k;
	scanf("%d%d%d",&a,&b,&k);
	sieve();
	int ans=1;
	if(sum[b]-sum[a-1]<k)printf("-1\n");
	else
	{
		for(int i=a;i<=b;i++)
		{
			if(sum[i]-sum[i-ans]<k)
				ans++;
		}
		printf("%d\n",ans);
	}
	return 0;
}
