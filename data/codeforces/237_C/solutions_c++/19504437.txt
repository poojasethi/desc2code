#include<stdio.h>
#include <iostream>
#include<algorithm>
using namespace std;

int a , b , k;
bool vis[1000001];
int sum[1000001];

bool check(int l){
	for(int i=a;i+l<=b+1;i++)
		if(sum[i+l-1]-sum[i-1]<k) return 0;
	return 1;
}

int binary_search(){
	//check(2);
	int s=1,e=b-a+1,ret=-1;
	while(s<=e){
		int mid=(s+e)/2;
		if(check(mid)){
			e=mid-1;
			ret=mid;
		}else{
			s=mid+1;
		}
	}
	return ret;
}

int main(){
	scanf("%d %d %d",&a,&b,&k);
	for(int i=2;i<=b;i++)
		if(!vis[i])
			for(int j=i+i;j<=b;j+=i)
				vis[j]=1;

	for(int i=2;i<=b;i++)
		sum[i]=sum[i-1]+(!vis[i]);
	printf("%d\n",binary_search());
}
