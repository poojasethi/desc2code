#include<cstdio>
const int X=1000010;
int s[X],a,b,k;
void make(){
    int i,j;
    for(i=2;i<X;i++)
        if(s[i])
            for(j=i+i;j<X;j+=i)
                s[j]=0;
    for(i=2;i<X;i++)
        s[i]+=s[i-1];
}
bool check(int l){
    int i;
    for(i=a;i<=b-l+1;i++){
        if(s[i+l-1]-s[i-1]<k)
            return 0;
    }
    return 1;
}

int main(){
    int l,r,mid,i;
    for(i=2;i<X;i++)
        s[i]=1;
    make();
    scanf("%d%d%d",&a,&b,&k);
    l=1;r=b-a+2;
    while(l<r){
        mid=l+r>>1;
        if(!check(mid))l=mid+1;
        else r=mid;
    }
    if(r==b-a+2)puts("-1");
    else printf("%d\n",r);
    return 0;
}
