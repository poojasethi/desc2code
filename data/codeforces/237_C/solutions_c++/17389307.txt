#include <bits/stdc++.h>

using namespace std;

int is[1000021];
vector < int > p;

int main() {
	int a, b, k;
	cin >> a >> b >> k;
	for(int i = 2; i <= b; i++) {
		if(not is[i]) {
			if(i >= a and i <= b) {
				p.push_back(i);
			}
			for(int j = i + i; j <= b; j += i) {
				is[j] = 1;
			}
		}
	}
	if(p.size() < k) {
		cout << "-1" << endl;
		return 0;
	}
	int ans = max(p[k - 1] - a, b - p[p.size() - k]) + 1;
	for(int i = k; i < p.size(); i++) {
		ans = max(ans, p[i] - p[i - k]);
	}
	cout << ans << endl;
	return 0;
}
