#include<iostream>
#include<stdio.h>
using namespace std;
int a,b,K,i,j,n,r,p,q,l;
int f[1000005],d;

bool check(int x){
     int i,q=0;
     
     for(i=a;i<=a+x-1;i++)
         if(f[i]==0)q++;
     for(i=a;i<=b-x+1;i++){
         if(q<K)return 0;
         if(f[i+x]==0)q++;
         if(f[i]==0)q--;
         }
     return 1;
}

main()
{cin>>a>>b>>K;
 f[1]=1;
 for(i=2;i<=b;i++)
    for(j=i+i;j<=b;j+=i)
    f[j]=1;
    
 l=0;
 r=b-a+1;
 while(l<r){
       d=(l+r)/2;
       if(check(d))r=d;else
                   l=d+1;
    }
 if(!check(l))cout<<-1<<endl;else
cout<<l<<endl;
// system("pause");
}
