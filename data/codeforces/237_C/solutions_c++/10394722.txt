#include <stdio.h>
int q[2000009];
int main(){
    int a,b,k,i,j,w;
    scanf("%d%d%d",&a,&b,&k);
    w=b-a+2;
    for(i=2;i<2000001;i++){
        if(q[i]=!q[i])
            for(j=i+i;j<2000001;j+=i)q[j]=1;
    }
    for(i=1;i<2000001;i++)q[i]+=q[i-1];
    a--;i=0;
    while(a<=b-i){
        while(q[a+i]-q[a]<k){
            i++;if(a>b-i)goto W;
        }
        a++;
    }W:
    printf("%d\n",i==w?-1:i);
    //getch();
}
