#include <stdio.h>
#include <cmath>
#define N 1000005
int isPrime(int n)
{
	int md=(int)(pow((float)n,0.5))+1;
	int i;
	if(n<2) return 0;
	if(n==2) return 1;
	for(i=2;i<=md;i++)
	{
		if(n%i==0) return 0;
	}
	return 1;
}

int a,b,k;
int cq[N];
int main()
{
	int i;
	scanf("%d %d %d",&a,&b,&k);
	int l=-1;
	cq[0]=a-1;
	int ind=1;
	for(i=a;i<=b;i++)
	{
		if(isPrime(i)==1)
		{
			cq[ind%(k+1)]=i;
			if(ind>=k)
			{
				int interv=cq[ind%(k+1)]-cq[(ind-k)%(k+1)];
				if (l<interv) l=interv;
			}
			ind++;
		}	
	}
	if(ind>=k && b+1-cq[(ind-k)%(k+1)]>l) l=b+1-cq[(ind-k)%(k+1)];
	if(l>b-a+1){printf("-1\n");return 0;}printf("%d\n",l);
}
