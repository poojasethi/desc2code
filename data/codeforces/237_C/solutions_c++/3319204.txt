#include <iostream>
using namespace std;
int p[1001000];
int main(void)
{
  int a,b,k,f;
  cin>>a>>b>>k;
  for(int i=2;i<b+10;++i)p[i]=1;
  for(int i=2;i<b+10;++i)if(p[i])for(int j=i+i;j<b+10;j+=i)p[j]=0;
  for(int i=1;i<b+10;++i)p[i]+=p[i-1];
  int lo=1,hi=b-a+2;
  while(lo<hi)
  {
    int m=(lo+hi)>>1;f=1;
    for(int i=a;f&&i+m-1<=b;++i)if(p[i+m-1]-p[i-1]<k)f=0;
    if(f)hi=m;else lo=m+1;
  }
  cout<<((lo > b-a+1)?-1:lo)<<endl;
  return 0;
}
