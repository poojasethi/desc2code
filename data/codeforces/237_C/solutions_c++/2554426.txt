#include<iostream>
#include<algorithm>
using namespace std;
bool aval(int x){
  int i,j;  
  for(i=2;i*i<=x;i++){
   if(x%i==0) return false;  
  }
  return true; 
}
int p[1000010];
int main(){ 
   int a,b,k,l,i,j,res;
   cin>>a>>b>>k;
   p[0]=a-1,j=1,l=0;
   for(i=a;i<=b;i++){
      if(aval(i)==1&&i!=1) {p[j]=i;j++;}  
   }
   for(i=0;i<j;i++){
      if(i+k<j) l=max(l,p[i+k]-p[i]);
      if(i+k>=j&&p[i]<=b-l){
         while(p[i]<=b-l) l++;   
      }   
   }
   if(j-1<k) cout<<-1<<endl;
   else cout<<l<<endl;
   return 0;    
}
