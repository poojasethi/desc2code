#include<iostream>
#include<vector>
#define maxn 1000000
using namespace std;
int lp[1000005]={0},i,j,a,b,k,p=0,l=0;
vector<int> v;
int main()
{
for(i=2;i<=maxn;i++)
{
if(lp[i]==0) lp[i]=i,v.push_back(i);
for(j=0;j<v.size() && v[j]<=lp[i] && i*v[j]<=maxn;j++)
{
lp[i*v[j]]=v[j];
}
}
cin>>a>>b>>k;
j=a;
for(i=a;i<=b-l+1;i++)
{
if(i>b-l+1) break;
while(p<k)
{
if(j>b && i==a)
{
cout<<-1;
return 0;
}
if(j>b)
{
l=max(l,b-i+2);
break;
}
if(lp[j]==j) p++;
j++;
}
l=max(l,j-i);
if(lp[i]==i) p--;
}
if(l<=b-a+1) cout<<l;
else cout<<-1;
return 0;
}
