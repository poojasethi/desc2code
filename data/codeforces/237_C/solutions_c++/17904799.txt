#include <cstdio>
#include <vector>
using namespace std;
const int maxa=1e6+1 ;
bool pr[maxa] ;
vector<int> oui ;
int main()
{
	for(int i=2;i<maxa;i++)
		pr[i]=true ;
	for(int i=2;i<maxa;i++)
		if(pr[i])
			for(int j=2*i;j<maxa;j+=i) 
				pr[j]=false ;
	int a,b,k ;
	scanf("%d %d %d",&a,&b,&k) ;
	for(int i=a;i<b+1;i++)
		if(pr[i])
			oui.push_back(i) ;
	if(oui.size()<k)
	{
		printf("-1") ;
		return 0;
	}
	int maxi=0 ;
	for(int i=0;i<oui.size()-k;i++)
		maxi=max(maxi,oui[i+k]-oui[i]) ;
	//printf("%d %d\n",oui[k-1]-a,b-oui[oui.size()-k]) ;
	maxi=max(maxi,max(oui[k-1]-a+!pr[a],b-oui[oui.size()-k]+1)) ;
	printf("%d",maxi) ;
}
