#include <bits/stdc++.h>

using namespace std;
#define int long long
const int maxN = 1000*1000 + 321;
int dp[maxN];
int a[maxN];
main()
{  
  int c, b, k;
  cin >> b >> c >> k;
  a[1] = 1;
  for(int i = 2; i < maxN ; i ++)
    for(int j = 2 * i ; j < maxN; j += i)
      a[j] = 1;
  vector<int> r;
  for(int i = b; i <= c; i ++)
    if(!a[i]) r.push_back(i);
  
  if(r.size() < k) { cout << "-1" << endl; return 0; }
  int ans = r[k - 1] - b + 1;
  for(int i = k; i < r.size(); i ++) 
    {
      int u = r[i] - r[i - k];
      if(ans < u)
	ans = u;
    }
  ans = max(ans, c - r[r.size() - k] + 1);
  cout << ans << endl;
  return 0;
}
