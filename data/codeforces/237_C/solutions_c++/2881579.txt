#include <iostream>
#include <vector>
using namespace std;

int main(void)
{
	int a,b,k;
	cin >> a >> b >> k;
	vector <bool> p(b+1,true);
	p[0] = p[1] = false;
	for (int i = 2; i <= b / i; i++)
		if (p[i])
			for (int j = i << 1; j <= b; j += i)	
				p[j] = false;
	int l = -1, h = b - a + 2;
	while (h-l > 1)
	{
		int m = (h+l) >> 1;
		int c = 0;
		for (int i = 0; i < m; i++)
			if (p[a+i])	c++;
		if (c < k)	goto fail;
		for (int j = a+1; j <= b - m + 1; j++)
		{
			if (p[j-1])	c--;
			if (p[j+m-1])	c++;
			if (c < k)	goto fail;
		}
		h = m;
		continue;
fail:		l = m;
	}
	if (h == b-a+2)	cout << (-1) << endl;
	else cout << h << endl;
	return 0;
}
