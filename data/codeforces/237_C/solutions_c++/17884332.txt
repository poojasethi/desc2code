#include<cstdio>
#include<cstring>
const int N=1e6+5;
int tot_prime,prime[N];
bool is_prime[N];
int L,R,k;
inline void Max(int &a,int b){
	if(b>a)a=b;
}
int solve(){
	if(tot_prime<k)return -1;
	int ans=0;
	for(int i=0;i<tot_prime;++i){
		if(i-k<0)continue;
		Max(ans,prime[i]-prime[i-k]);
	}
	Max(ans,prime[k-1]-L+1);
	Max(ans,R-prime[tot_prime-k]+1);
	return ans;
}
int main(){
	scanf("%d%d%d",&L,&R,&k);
	memset(is_prime,true,sizeof(is_prime));
	tot_prime=0;
	for(int i=2;i<=R;++i){
		if(is_prime[i]){
			for(int j=i<<1;j<=R;j+=i)
				is_prime[j]=false;
			if(i>=L)prime[tot_prime++]=i;
		}
	}
	printf("%d\n",solve());
	return 0;
}
