#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<string.h>
using namespace std;
const int M=1e6+5;
bool mark[M];
int sum[M];
inline void Init(){
	for(int i=2;i<M;++i){
		if(!mark[i]){
			for(int j=i+i;j<M;j+=i)
				mark[j]=1;
		}
	}
	for(int i=2;i<M;++i){
		if(!mark[i])sum[i]=sum[i-1]+1;
		else sum[i]=sum[i-1];
	}
}
int a,b,k;
inline bool check(int x){
	for(int i=a;i<=b-x+1;++i)
		if(sum[i+x-1]-sum[i-1]<k)return 0;
	return 1;
}
int main(){
	Init();
	cin>>a>>b>>k;
	int l=1,r=b-a+1,ans=-1;
	while(l<=r){
		int mid=l+r>>1;
		if(check(mid))r=mid-1,ans=mid;
		else l=mid+1;
	}
	printf("%d\n",ans);
	return 0;
}
