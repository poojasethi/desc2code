#include <iostream>

using namespace std;

const int tope=1000001;

int esprimo[tope];
int suma[tope];

int a,b,k;

bool sepuede(int l)
{
  for (int i=a;i<=b-l+1;i++)
    if (suma[i+l-1]-suma[i-1]<k) return false;
  return true;
}

int main()
{
  for (int i=2;i<tope;i++)
    esprimo[i]=1;
  for (int i=2;i<tope;i++)
    if (esprimo[i])
      for (int j=i+i;j<tope;j+=i)
	esprimo[j]=0;
  for (int i=2;i<tope;i++)
    suma[i]=suma[i-1]+esprimo[i];
  cin>>a>>b>>k;
  int inf=0;
  int sup=b-a+2;
  while (inf+1<sup) {
    int med=(inf+sup)/2;
    if (sepuede(med))
      sup=med;
    else
      inf=med;
  }
  if (sup<b-a+2) cout<<sup<<endl;
  else cout<<-1<<endl;
}
