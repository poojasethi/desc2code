#include <cstdio>
#define S 1000005

bool prime[S] = {1, 1};
int N, p[S / 10], times[S];

int main() {
  int i, j;
  for (i = 2; i < S; i++) {
    if (!prime[i]) {
      p[N++] = i;
      for (j = i + i; j < S; j += i)
        prime[j] = 1;
    }
    times[i] = N;
  }
  int a, b, k;
  scanf("%d%d%d", &a, &b, &k);
  if (times[b] - times[a] + !prime[a] < k) {
    printf("-1");
    return 0;
  }
  int l = 1, x;
  for (x = a; x <= b - l + 1; x++)
    if (times[x + l - 1] - times[x] + !prime[x] < k) {
      l++;
      x--;
    }
  printf("%d", l);
  return 0;
}
