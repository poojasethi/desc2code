#include <iostream>
using namespace std;

long long a, b, k, npr[1000100];

int main()
{
	cin >> a >> b >> k;

	npr[1] = 1;
	for(long long i=2; i<=1e6; i++)
		if(!npr[i])
			for(long long j=i*i; j<=1e6; j+=i)
				npr[j] = 1;

	long long i=a, j, cnt=0;
	while(i <= b && cnt < k)
	{
		if(!npr[i])
			cnt++;
		i++;
	}

	if(i > b && cnt < k)
	{
		cout << -1 << endl;
		return 0;
	}

	long long ans = i-a;
	j = i;
	i = a;

	while(j<=b)
	{
		while(npr[i])
			i++;

		if(!npr[j])
			i++;

		while(npr[i])
			i++;		

		ans = max(ans, j-i+1);		
		j++;
	}

	cout << ans << endl;
	return 0;
}
