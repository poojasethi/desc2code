#include <iostream>

using namespace std;

int a, b, k, p[1000100] = {};

int main(){
    cin >> a >> b >> k;

    for (int i = 2; i <= b; i++)
        if (p[i] == 0)
            for (int j = i; j <= b; j += i)
                p[j] = 1;
        else p[i] = 0;
    for (int i = 3; i <= b; i++) p[i] += p[i - 1];

    int l = 1, r = b - a + 2;
    while (l < r){
        int sr = (l + r) / 2;

        bool t = true;
        for (int j = a; t && j <= b - sr + 1; j++)
            if (p[j + sr - 1] - p[j - 1] < k) t = false;

        if (t) r = sr; else l = sr + 1;
    }

    if (r <= b - a + 1) cout << r; else cout << -1;
}
