#include<iostream>
using namespace std;
long long a,b,i,j,k,s[1000001],c[1000001],x,l,r,last,m,p;
main()
{
      cin>>a>>b>>k;
      s[1]=1;
      for(i=2; i*i<=b; i++){
               if(s[i]==1) continue;
               for(j=i*i; j<=b; j+=i) s[j]=1;
      }
      for(i=b; i>=a; i--){
               if(s[i]==0) x++;
               c[i]=x;
      }
      l=1; r=b-a+1;
      last=-1;
      while(r>=l){
            m=(r+l)/2;
      p=true;
      for(i=a; i<=b-m+1; i++){
               if(c[i]-c[i+m]<k) p=false;
      }
      if(p==false) l=m+1;
      else {r=m-1; last=m;}
      }
             cout<<last;
       //system("pause");
}
