#include <iostream>
#include <cstdio>
using  namespace std;
int p[1000010];
int main()
{
    int a,b,k,f;
    scanf("%d%d%d",&a,&b,&k);
    for(int i=2;i<b+10;i++) p[i]=1;
    for(int i=2;i<b+10;i++)
    {
        if(p[i])
        {
            for(int j=i+i;j<b+10;j+=i)  p[j]=0;
        }
    }
    for(int i=1;i<b+10;i++) p[i]+=p[i-1];
    int l=0,h=b-a+2;
    while(l<h)
    {
        int m=(l+h)>>1;
        f=1;
        for(int i=a-1;f&&i+m<=b;++i)    if(p[i+m]-p[i]<k)   f=0;
        if(f)   h=m;
        else    l=m+1;
    }
    printf("%d\n",(l>b-a+1) ? -1:l);
    return 0;
}
