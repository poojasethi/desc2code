#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
int a,b,k;
int p[1000010];
int main()
{
    scanf("%d%d%d",&a,&b,&k);
    for(int i=2;i<=b;i++)p[i]=1;
    for(int i=2;i<=b;i++)if(p[i])for(int j=2*i;j<=b;j+=i)p[j]=0;
    for(int i=2;i<=b;i++)p[i]+=p[i-1];
    int lo=0,hi=b-a+2;
    while(lo<hi)
    {
        int f=1;
        int mi=(lo+hi)>>1;
        for(int i=a;f&&i+mi-1<=b;i++)
            if(p[i+mi-1]-p[i-1]<k)f=0;
        if(f)hi=mi;
        else lo=mi+1;
    }
    printf("%d\n",(hi>b-a+1)?-1:hi);
}
