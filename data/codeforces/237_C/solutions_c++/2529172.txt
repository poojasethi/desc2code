#include<stdio.h>
int vis[1000005], dp[1000006];	
int a, b, k;
bool judge(int l)
{
	int i, j;
	for(i = a; i <= b-l+1; i++)
		if(dp[i+l-1] - dp[i-1] < k) return 0; 
	return 1;
}
int main()
{
	int i, j;
	for(i = 2; i <= 1000; i++)
		for(j = i*i; j <= 1000000; j += i)
			vis[j] = 1;
	for(i = 2; i <= 1000000; i++)
		if(!vis[i]) dp[i] = dp[i-1] + 1;
		else dp[i] = dp[i-1];
	scanf("%d%d%d", &a, &b, &k);
	int f = 0;
	int l = k, r = b-a+1;
	while(l <= r)
	{ 
		int mid = (l + r) >> 1;
		if(judge(mid)) r = mid - 1;
		else l = mid + 1;
	}
	if(l > b-a+1) l = -1;
	  printf("%d\n", l);
	return 0;
}
