#include<stdio.h>
int pri(int n)
{
	int i;
	if(n==1)return 0;
	for(i=2;i*i<=n;i++)
		if(n%i==0)
			return 0;
		return 1;
}

int main()
{
    int A[100000];
	int a,b,k,i,j=1,max;
	scanf("%d%d%d",&a,&b,&k);
	A[0]=a-1;
	for(i=a;i<b+1;i++)
	{
		if(pri(i))
		{
			A[j]=i;
			j++;
		}
	}
    if(j-1<k)
	{
		printf("-1");
		return 0;
	}
	max=A[k]-A[0];
	for(i=1;i+k<j;i++)
		if(A[i+k]-A[i]>max)
			max=A[i+k]-A[i];
	if(b+1-A[j-k]>max&&!pri(b))
		max=b+1-A[j-k];
	printf("%d",max);
	return 0;
}
