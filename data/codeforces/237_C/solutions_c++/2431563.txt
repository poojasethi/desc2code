#include <cstdio>
#define maxabk 1000003
using namespace std;
int d[maxabk];
int a,b,k,l;
bool check(int l){
	for (int i=a+l-1;i<=b;i++)
		if (d[i]-d[i-l]<k)
			return false;
	return true;
}
int main(){
	int i,j;
	d[0]=0;	d[1]=0;
	for (i=2;i<maxabk;i++){
		if (d[i]==1) d[i]=d[i-1];
		else{
			d[i]=d[i-1]+1;
			for (j=2;(i*j)<maxabk;j++)
				d[i*j]=1;
		}
	}
	scanf("%d%d%d",&a,&b,&k);
	if (d[b]-d[a-1]<k)
		printf("-1");
	else{
		int x=1, y=b-a+1,mid;		l=b-a+1;
		while (x<=y){
			mid=(x+y)/2;
			if (check(mid)==true){
				if (l>mid)
					l=mid;
				y=mid-1;
			}
			else
				x=mid+1;
		}
		printf("%d",l);
	}
    return 0;
}
