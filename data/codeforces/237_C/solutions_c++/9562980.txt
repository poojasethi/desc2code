#include <iostream>
using namespace std;
int p[1000005];
int main() {
	int a,b,k;
	cin>>a>>b>>k;
	for(int i=2;i<=b;i++)p[i]=1;
	for(int i=2;i<=b;i++)
	{
		if(p[i]==1)
		{
			for(int j=2*i;j<=b;j+=i)
			p[j]=0;
		}
	}
	for(int i=2;i<=b;i++)
	p[i]=p[i]+p[i-1];
	int low=0;
	int high=b-a+2;
	int mid;
	while(low<high)
	{
		int temp=1;
		mid=(low+high)/2;
		for(int i=a;temp && i+mid-1<=b;i++)
		{
			if(p[i+mid-1]-p[i-1]<k)temp=0;
		}
		if(temp==1)high=mid;
		else low=mid+1;
	}
	
	if(high>b-a+1)cout<<-1;
	else cout<<high;
	return 0;
}
