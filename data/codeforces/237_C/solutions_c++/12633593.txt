#include <bits/stdc++.h>

using namespace std;
int n=1000002,a,b,k;
int primes[1000003];

bool isOk(int mid)
{
for(int i=a;i<=b-mid+1;i++)
{
if(primes[i+mid-1]-primes[i-1]<k)
return false;
}

return true;
}

int main()
{

for(int i=0;i<n;i++)
primes[i]=1;
primes[0]=primes[1]=0;
int l=sqrt(n);
for(int i=2;i<=l;i++)
if(primes[i])
for(int j=i*i;j<=n;j+=i)
primes[j]=0;

for(int i=1;i<=1000000;i++)
primes[i]+=primes[i-1];

cin>>a>>b>>k;
int lo=1,hi=b-a+2,mid;
while(lo<hi)
{
mid=(lo+hi)/2;
if(isOk(mid))
hi=mid;
else
lo=mid+1;
}
if(lo==b-a+2)
cout<<-1<<endl;
else
cout<<lo<<endl;
return 0;
}
