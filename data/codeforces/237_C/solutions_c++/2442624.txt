# include <stdio.h>
# include <math.h>

int a,b,c,z,k,mx;

int v[3000009];
int ans[3000009];

int main()
{
	scanf("%d %d %d",&a,&b,&c);
	
	v[1]=1;
	
	for(int i=2;i<=(int)sqrt(b);i++)
	{
		z=i*i;
		
		while(z<=b)
		{
			v[z]=1;
			z+=i;
		}
	}
	
	for(int i=a;i<=b;i++)
	{
		if(v[i]==0)
		{ans[k]=i;k++;}
	}
	
	if(k<c)
	{printf("-1");return 0;}
	
	for(int i=0;i<k;i++)
	{
		if(ans[i+c]==0)
		continue;
		
		else if(ans[i+c]-ans[i]>mx)
		mx=ans[i+c]-ans[i];
	}
	
	if(ans[c-1]-a+1>mx)
	mx=ans[c-1]-a+1;
	
	if(b-ans[k-c]+1>mx)
	mx=b-ans[k-c]+1;
	
	printf("%d\n",mx);
}
