#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
bool f[1100000];
int p[100000],i,j,a,b,k,g[100000],x,y,ans;
int  main(){
    scanf("%d%d%d",&a,&b,&k);
    memset(f,true,sizeof(f));
    for(i=2;i<b;i++)if(f[i])
        for(j=2;i*j<=b;j++)f[i*j]=0;
    for(i=2;i<=b;i++)if(f[i])p[++p[0]]=i;
    for(j=1;j<=p[0];j++)if(p[j]>=a)break;
    if(p[0]-j+1<k){
        printf("-1");return 0;
    }
    for(i=j;i<=p[0];i++)g[i-j+1]=p[i];
    g[0]=p[0]-j+1;
    
    ans=max(g[k]-a+1,b-g[g[0]-k+1]+1);
    x=1;y=k+1;
    for(;y<=g[0];){
        ans=max(ans,g[y]-g[x]);
        x++,y++;
    }
    printf("%d\n",ans);
}
