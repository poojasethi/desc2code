import time

def eratosfen_interval(a, b ,k, toprint):
    l = [True]*(b+1)
    i = 2
    primelist = []
    while i <= len(l):
        try:
            i = l.index(True, i)
            if a <= i <= b:
                primelist.append(i)
        except ValueError:
            i = len(l) +1
        for j in xrange(2*i, len(l), i):
            l[j] = False
        i+=1
    del(l)
    
    if k > len(primelist): return -1
    interval = max(primelist[k-1] - a + 1, b - primelist[-k]+1)


    if toprint: print primelist
    if toprint: print 'interval', interval
    
    for i in xrange(0, len(primelist)-k, 1):
        if toprint: print 'i', i
        if toprint: print 'interval', interval
        if toprint: print 'dif', primelist[i+k] - primelist[i]
        interval = max(interval, primelist[i+k] - primelist[i])
    return interval


a, b ,k = map(int, raw_input().split())

print eratosfen_interval(a, b ,k, toprint = False)
