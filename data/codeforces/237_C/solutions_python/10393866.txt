a, b, k = map(int, raw_input().split())
isPrime, primes = [True] * (b + 1), []
for i in xrange(2, b + 1):
    if isPrime[i]:
        if i >= a:
            primes.append(i)
        for j in xrange(i + i, b + 1, i):
            isPrime[j] = False
print -1 if len(primes) < k else max([b - primes[-k] + 1, primes[k-1] - a + 1] + [primes[i] - primes[i-k] for i in xrange(k, len(primes))])
