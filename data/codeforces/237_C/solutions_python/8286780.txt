def sieve(n):
    s = xrange(3, n + 1, 2)
    r = set(s)
    [r.difference_update(xrange(n << 1, s[-1] + 1, n)) for n in s if n in r]
    return r.union([2])
a,b,k=map(int,raw_input().split())
p=filter(lambda x:x>=a and x<=b,sorted(sieve(b)))
if len(p)<k:
    print -1
    exit()
