a,b,k=map(int,raw_input().split())
p=[]
t=[1]*(b+1)
for i in xrange(2,b+1):
    if t[i]:
        if i >= a:
            p.append(i)
        for j in range(i*i, b+1, i):
            t[j]=0
if len(p) < k:
    print -1
else:
    m=[p[i]-p[i-k] for i in range(k, len(p))]+[p[k-1]-a+1,b-p[-k]+1]

    print max(m)
