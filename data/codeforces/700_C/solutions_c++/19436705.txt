#include "bits/stdc++.h"
#define MAXN 100009
#define INF 1000000007
#define mp(x,y) make_pair(x,y)
#define all(v) v.begin(),v.end()
#define pb(x) push_back(x)
#define wr cout<<"----------------"<<endl;
#define ppb() pop_back()
#define tr(ii,c) for(typeof((c).begin()) ii=(c).begin();ii!=(c).end();ii++)
#define ff first
#define ss second

using namespace std;

typedef long long ll;
typedef pair<ll,int> PII;
template<class T>bool umin(T& a,T b){if(a>b){a=b;return 1;}return 0;}
template<class T>bool umax(T& a,T b){if(a<b){a=b;return 1;}return 0;}
//freopen("file.out","w",stdout);
vector<PII>adj[MAXN];
int n,m,s,t;
vector<int>v;
int vis[MAXN];
ll arr[MAXN];
PII ans;
int low[MAXN],tin[MAXN],TIM,out[MAXN];
int dfs(int nd){
	vis[nd]=1;
	if(nd==t)
		return 1;
	for(int i=0;i<int(adj[nd].size());i++){
		int to=adj[nd][i].ff;
		int ind=adj[nd][i].ss;
		if(!vis[to] and dfs(to)){
			v.pb(ind);
			return 1;
		}
	}
	return 0;
}
int get(int x,int y){
	return (tin[x]<=tin[y] and out[y]<=out[x]);
}
void go(int nd,int last,int gad){
	low[nd]=tin[nd]=++TIM;
	for(int i=0;i<int(adj[nd].size());i++){
		int to=adj[nd][i].ff;
		int ind=adj[nd][i].ss;
		if(gad==ind or ind==last)
			continue;
		if(!low[to]){
			go(to,ind,gad);
			umin(low[nd],low[to]);
		}
		else
			umin(low[nd],tin[to]);
	}out[nd]=TIM;
	if(~last and get(nd,t) and low[nd]==tin[nd])
		umin(ans,mp(arr[last],last));
}
int main(){
	#ifndef ONLINE_JUDGE
    freopen("file.in", "r", stdin);
	#endif	
	scanf("%d%d",&n,&m);	
	scanf("%d%d",&s,&t);
	for(int i=1;i<=m;i++){
		int u,v;
		scanf("%d%d%lld",&u,&v,arr+i);
	//	printf("ind[%d] == %d %d %lld\n",i,u,v,arr[i]);
		adj[u].pb(mp(v,i));
		adj[v].pb(mp(u,i));
	}
	if(!dfs(s))
		return printf("0\n0\n"),0;
	pair<ll,pair<int,int> > res=mp(100000000000000000,mp(0,0));	
	for(int i=0;i<int(v.size());i++){
		memset(low,0,sizeof(low));
		memset(tin,0,sizeof(tin));
		memset(out,0,sizeof(out));TIM=0;
		ans=mp(100000000000000000,0);
		go(s,-1,v[i]);
		if(!tin[t])
			ans=mp(0,0);
		umin(res,mp(ans.ff+arr[v[i]],mp(v[i],ans.ss)));	
	}	
	if(res.ff==100000000000000000)
		return printf("-1\n"),0;
	if(res.ss.ss)
		return printf("%lld\n2\n%d %d\n",res.ff,res.ss.ff,res.ss.ss),0;
	printf("%lld\n1\n%d\n",res.ff,res.ss.ff);
	return 0;
}
