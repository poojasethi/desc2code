#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#include<vector>
#include<queue>
#include<cmath>
#define ll long long
#define ld long double
#define pb push_back
using namespace std;
const int maxn= 5e5,oo= 2e9+10;
struct edge{
	int v,w,h;
};
vector<edge> g[maxn];
edge r[maxn],R[maxn];
int vis[maxn],pre[maxn],low[maxn],b[maxn],B[maxn];
int block[maxn],sta[20];
int ans= oo;
int x,y,z,n,m,s,t,i,j,cnt,CNT,la,bl,tim;
void dfs(int o,int &cnt,edge *r){
	if (vis[o]) return;
	vis[o]= 1;
	if (o==t){
		bl= 1;
		return;
	}
	int len= g[o].size(),i;
	for (i= 0;i<len;i++){
		edge e= g[o][i];
		if (block[e.h]) continue;
		dfs(e.v,cnt,r);
		if (bl){
			r[++cnt]= e;
			return;
		}
	}
}
void bridge(int o,int fa,int *b){
	pre[o]= low[o]= ++tim;
	int len= g[o].size(),i;
	for (i= 0;i<len;i++){
		edge e= g[o][i];
		if (block[e.h]) continue;
		if (!pre[e.v]){
			bridge(e.v,e.h,b);
			low[o]= min(low[o],low[e.v]);
			if (low[e.v]>pre[o])
				b[e.h]= 1;
		}else if (e.h!=fa)
			low[o]= min(low[o],pre[e.v]);
	}
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("1.in","r",stdin);
	freopen("1.out","w",stdout);
#endif
	scanf("%d%d",&n, &m);
	scanf("%d%d",&s, &t);
	for (i= 1;i<=m;i++){
		scanf("%d%d%d",&x, &y, &z);
		g[x].pb((edge){y,z,i});
		g[y].pb((edge){x,z,i});
	}
	dfs(s,cnt,r);
	if (!bl) ans= 0;
	else {
		reverse(r+1,r+cnt+1);
		for (i= 1;i<=n;i++)
			if (!pre[i])
				bridge(i,0,b);
		for (i= 1;i<=cnt;i++)
			if (b[r[i].h]){
				if (r[i].w<ans){
					ans= r[i].w;
					la= 1;
					sta[la]= r[i].h;
				}
			}else {				
				block[r[i].h]= 1;
				for (j= 1;j<=n;j++)
					pre[j]= low[j]= vis[j]= 0;
				CNT= 0;bl= 0;
				dfs(s,CNT,R);
				for (j= 1;j<=m;j++)	B[j]= 0;
				for (j= 1;j<=n;j++)
					if (!pre[j])
						bridge(j,0,B);
				for (j= 1;j<=CNT;j++)
					if (B[R[j].h]){
						if (r[i].w+R[j].w<ans){
							ans= r[i].w+R[j].w;
							la= 2;
							sta[1]= r[i].h;
							sta[2]= R[j].h;
						}
					}
				block[r[i].h]= 0;
			}
	}
	if (ans==oo){
		printf("-1");
		return 0;
	}
	printf("%d\n%d\n",ans, la);
	for (i= 1;i<=la;i++)
		printf("%d%c",sta[i],i<la?' ':'\n');
	return 0;
}
