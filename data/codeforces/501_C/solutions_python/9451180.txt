n = int(raw_input())
nodes = []
for _ in range(n):
    nodes.append(map(int, raw_input().split()))
edges = []
stack = []
for i, n in enumerate(nodes):
    d, s = n
    if d == 1:
        stack.append(i)
while stack:
    i = stack.pop()
    if nodes[i][0] == 0:
        continue
    nodes[i][0] = 0
    j = nodes[i][1]
    edges.append((i, j))
    nodes[j][0] -= 1
    nodes[j][1] ^= i
    if nodes[j][0] == 1:
        stack.append(j)
print len(edges)
for i, j in edges:
    print i, j
