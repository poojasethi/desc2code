import heapq

n = int(raw_input())

ans, todo, a = [], [], []

for i in xrange(n):
    t = map(int, raw_input().strip().split(' '))
    a.append(t)
    if t[0] == 1:
        heapq.heappush(todo, i)

while len(todo) > 0:
    i = heapq.heappop(todo)
    t = a[i]
    if t[0] == 0:
        continue
    ans.append('%s %s' % (i, t[1]))
    a[t[1]][1] ^= i
    a[t[1]][0] -= 1
    a[i] = [0, 0]
    if a[t[1]][0] == 1:
        heapq.heappush(todo, t[1])

print len(ans)
print '\n'.join(ans)
