N = (1 << 16) + 8
d = [0] * N
s = [0] * N
n = int(raw_input())
ans = []
q = [0] * N
for i in xrange(n):
    d[i], s[i] = map(int, raw_input().split())
    l = 1; r = 0
for i in xrange(n):
    if (d[i] == 1):
        r = r + 1
        q[r] = i
while l <= r:
    u = q[l]; l += 1
    if (d[u] == 0): continue
    fa = s[u]
    ans.append([u, fa])
    s[fa] ^= u; d[fa] -= 1
    if d[fa] == 1: r = r + 1; q[r] = fa
print len(ans)
for a, b in ans:
    print a, b
