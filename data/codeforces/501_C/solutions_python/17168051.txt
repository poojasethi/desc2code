def main():
	n = input()
	d, xor, q = [], [], []
	for i in xrange(n):
		deg, s = map(int, raw_input().split())
		d.append(deg)
		xor.append(s)
		if 1 == deg: q.append(i)
	ans = []
	while len(q) > 0:
		i = q[0]
		del q[0]
		if 0 == d[i]: continue
		j = xor[i]
		ans.append([i, j])
		d[j] -= 1
		xor[j] ^= i
		if 1 == d[j]: q.append(j)
	print len(ans)
	for p in ans: print p[0], p[1] 

main()
