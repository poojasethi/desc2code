#!/usr/bin/env python
#-*- coding:utf-8 -*-

from Queue import Queue

n = int(raw_input())

degrees = {}
sums = {}
edges = {}
q = Queue()
for v in xrange(n):
    deg, xor = map(int, raw_input().split())
    degrees[v] = deg
    sums[v] = xor
    if deg == 1:
        q.put(v)

edges = []
while not q.empty():
    v = q.get()
    u = sums[v]
    if degrees[v] == 0:
        continue
    sums[v] ^= u
    sums[u] ^= v
    edges.append((u, v))

    degrees[v] -= 1
    degrees[u] -= 1
    if degrees[u] == 1:
        q.put(u)

print len(edges)
for a, b in edges:
    print a, b
