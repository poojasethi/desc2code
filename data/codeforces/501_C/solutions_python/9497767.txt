
n = input()
degree = [0 for i in range(0, n)]
xors = [0 for i in range(0, n)]
queue = []
for i in range(0, n):
    degree[i], xors[i] = map(int, raw_input().split())
    if degree[i] == 1:
        queue.append(i)

res = []
while len(queue) > 0:
    a = queue.pop(0)
    if degree[a] == 0: continue
    b = xors[a]
    res.append((a, b))
    degree[b] -= 1
    if degree[b] == 1: queue.append(b)
    xors[b] = xors[b] ^ a

print len(res)
for a, b in res:
    print "%d %d" % (a, b)
