n=int(raw_input())
node=[]
for i in xrange(n):
	node.append([int(x) for x in raw_input().split(' ')])

count=0
leaf=[]
for i in xrange(n):
	count+=node[i][0]
	if node[i][0]==1:
		leaf.append(i)

print count/2

while len(leaf)>0:
	son=leaf.pop(0)
	if node[son][0]==0:
		continue
	node[son][0]-=1
	father=node[son][1]
	node[father][0]-=1
	node[father][1]=node[father][1]^son
	if node[father][0]==1:
		leaf.append(father)
	print son, father
