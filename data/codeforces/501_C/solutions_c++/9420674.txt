#include <cstdio>
#include <queue>
#define N_Max 6*12345

using namespace std;

int n, sum, d[N_Max], s[N_Max];
queue<int> q;

void init() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d%d", &d[i], &s[i]);

		if (d[i] == 1)  q.push(i);
		sum += d[i];
	}
}

void solve() {
	printf("%d\n", sum/2);

	while (!q.empty()) {
		int u = q.front();
		q.pop();

		if (d[u] == 0)  continue;
		d[u]--;

		int v = s[u];
		d[v]--;
		s[v] ^= u;

		if (d[v] == 1)  q.push(v);
		printf("%d %d\n", u, v);
	}
}

int main() {
	init();
	solve();
}
