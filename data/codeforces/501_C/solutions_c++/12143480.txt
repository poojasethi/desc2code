#include <bits/stdc++.h>
using namespace std;
const int N = (1<<16) + 10;
int d[N], s[N];
int main()
{
	int n, sum = 0;
	scanf("%d", &n);
	queue <int> q;
	for (int i = 0; i < n; ++i){
		scanf("%d%d", &d[i], &s[i]);
		sum += d[i];
		if (d[i] == 1) q.push(i);
	}
	printf("%d\n", sum/2);
	while (!q.empty()){
		int u = q.front();q.pop();
		if (d[u] == 0) continue;
		int v = s[u];
		printf("%d %d\n", u, v);
		d[v]--;
		s[v] ^= u;
		if (d[v] == 1) q.push(v);
	}
	return 0;
}
