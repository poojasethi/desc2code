#include<iostream>
#include<queue>

using namespace std;

int d[100000], s[100000], s1 = 0;
int n;
queue <int> q;

int main(){
	int n; cin>>n;
	for (int i = 0; i < n; i++){
		cin >> d[i] >> s[i];
	  if (d[i] == 1) q.push(i);
		s1 += d[i];
	}

	cout << s1/2 << endl;

	while (q.size()){
		int u = q.front();
		q.pop();
		if (d[u] == 0) continue;
		int v = s[u];
		d[v]--;
		s[v] ^= u;

		cout << u << " " << v << endl;
		
		if (d[v]==1) q.push(v);
	}

	return 0;
}
