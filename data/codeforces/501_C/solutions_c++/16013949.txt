#include<iostream>
using namespace std;
int main()
{
	int n,top=-1;
	cin>>n;
	int d[n],s[n],k[n][2];
	for(int i=0;i<n;i++)cin>>d[i]>>s[i];
	for(int i=0;i<n;i++)
	{   int l=i;
		while(d[l]==1)
		{
		   if(d[s[l]]==0)break;
		   k[++top][0]=l;
		   k[top][1]=s[l];
		   d[l]--;s[s[l]]^=l;l=s[l];d[l]--;
		}
	}
	cout<<top+1<<endl;
	for(int i=0;i<=top;i++)cout<<k[i][0]<<" "<<k[i][1]<<endl;
}
