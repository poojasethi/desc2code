#include <cstdio>
#include <vector>
using namespace std;
#define maxn (1<<16)+5
int n,deg[maxn],s[maxn],q[maxn];
vector <pair<int,int> > ans;
int main(){
	scanf("%d",&n);int l=1,r=0;
	for(int i=0;i<n;++i){
		scanf("%d%d",&deg[i],&s[i]);
		if(deg[i]==1)q[++r]=i;
	}
	while(l<=r){
		int x=q[l++];
		if(deg[x]==1){
			ans.push_back(make_pair(x,s[x]));
			s[s[x]]^=x;
			if(--deg[s[x]]==1)q[++r]=s[x];
		}
	}
	printf("%d\n",ans.size());
	for(int i=0;i<ans.size();++i)printf("%d %d\n",ans[i].first,ans[i].second);
	return 0;
}
