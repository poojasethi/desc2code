#include<cstdio>
#include<queue>
using namespace std;
const int MAX_N=(1<<16)+5;
int d[MAX_N],s[MAX_N];
struct cmp{
	bool operator()(int i,int j){ return d[i]>=d[j]; }
};
queue<int> q;
int main(){
	int n,ans=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d%d",&d[i],&s[i]);
		ans+=d[i];
		if(d[i]==1) q.push(i);
	}
	printf("%d\n",ans/2);
	while(!q.empty()){
		int i=q.front();
		if(d[i]){
			printf("%d %d\n",i,s[i]);
			--d[i];
			--d[s[i]];
			s[s[i]] ^= i;
			if(d[s[i]]==1) q.push(s[i]);
		}
		q.pop();
	}
	return 0;
}
