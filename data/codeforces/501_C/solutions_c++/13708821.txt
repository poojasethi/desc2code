#include <bits/stdc++.h>
using namespace std;
queue<int >q;
int ver[100009],sum[100009],edg;
int main() {
	int v;
	edg=0;
	scanf("%d",&v);
	for(int i=0;i<v;i++)
	{
		scanf("%d %d",&ver[i],&sum[i]);
		edg+=ver[i];
		if(ver[i]==1)
		q.push(i);
	}
	printf("%d\n",edg/2);
	while(!q.empty())
	{
		int tp=q.front();
		q.pop();
		if(ver[tp]==0)
		continue;
		ver[tp]--;
		ver[sum[tp]]--;
		sum[sum[tp]]^=tp;
		
		if(ver[sum[tp]]==1)
		q.push(sum[tp]);
		printf("%d %d\n",tp,sum[tp]);
	}
	return 0;
}
