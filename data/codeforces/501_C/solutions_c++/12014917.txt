#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;

int dg[640000];
int sum[640000];
int n;

void work(int x)
{
    if(dg[x]!=1)return;
    int v=sum[x];
    cout<<x<<" "<<v<<endl;
    dg[x]--;
    dg[v]--;sum[v]^=x;
    work(v);
}

int main()
{
  scanf("%d",&n);
  int a=0;
  for(int i=0;i<n;i++)
  {
     scanf("%d%d",&dg[i],&sum[i]);
     a+=dg[i];
  }
  cout<<a/2<<endl;
  for(int i=0;i<n;i++)
  work(i);
}
