#include<iostream>
#include<queue>
using namespace std;
int deg[1000001],s[1000001];
queue<int>q;
int n,ans,ss;
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>deg[i]>>s[i];
        ans+=deg[i];
        q.push(i);
    }
    ans/=2;
    cout<<ans<<endl;
    while(!q.empty())
    {
        ss=q.front();
        q.pop();
        if(deg[ss]!=1)continue;
        deg[ss]--;
        s[s[ss]]^=ss;
        deg[s[ss]]--;
        cout<<ss<<" "<<s[ss]<<endl;
        if(deg[s[ss]]==1)q.push(s[ss]);
    }
    return 0;
}
