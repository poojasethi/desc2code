#include<cstdio>
#include<queue>
using namespace std;
int main(){
	int n,deg[140005],sum[140005],ans,u,v,i;
	while(~scanf("%d",&n)){
		queue<int> q;
		for(ans=0,i=0; i<n; ++i){
			scanf("%d%d",&deg[i],&sum[i]);
			ans+=deg[i];
			if(deg[i]==1) q.push(i);
		}
		printf("%d\n",ans/2);
		while(!q.empty()){
			u=q.front();
			q.pop();
			if(deg[u]==0) continue;
			deg[u]--;
			v=sum[u];
			deg[v]--;
			sum[v]^=u;
			if(deg[v]==1) q.push(v);
			printf("%d %d\n",u,v);
		}
	}				
	return 0;
}
