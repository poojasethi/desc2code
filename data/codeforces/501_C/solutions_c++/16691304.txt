#include<bits/stdc++.h>
using namespace std;
int a[100001],b[100001];
int p[100001][2];
int main()
{
    int n,m,i,j,x,y,c=0,flag;
    scanf("%d",&n);
    for(i=0;i<n;++i)
        scanf("%d%d",a+i,b+i);
    for(int i=0;i<n;++i)
	{   int l=i;
		while(a[l]==1)
		{
		   if(a[b[l]]==0)break;
		   p[c][0]=l;
		   p[c][1]=b[l];
		   ++c;
		   a[l]--;b[b[l]]^=l;l=b[l];a[l]--;
		}
	}
    printf("%d\n",c);
    for(i=0;i<c;++i)
        printf("%d %d\n",p[i][0],p[i][1]);
    return 0;
}
