#include <stdio.h>
#include <stdlib.h>
#include <queue>

using namespace std;

int main(){
	int n,i,j,k,x,y,z=0;

	scanf("%d", &n);

	int d[n], s[n];

	queue<int> q;

	for(i=0;i<n;i++){
		scanf("%d%d", &d[i], &s[i]);

		z+=d[i];

		if(d[i]==1)
			q.push(i);
	}

	printf("%d\n", z/2);

	while(!q.empty()){
		i=q.front();
		q.pop();

		j=s[i];

		if(d[i]==1){
			printf("%d %d\n", i,j);

			s[j]=s[j]^i;

			d[j]--;

			if(d[j]==1)
				q.push(j);
		}
	}

	return 0;
}
