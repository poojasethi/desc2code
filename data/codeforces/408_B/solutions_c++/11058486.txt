#include<bits/stdc++.h>
using namespace std;
int main()
{
    string a,b;cin>>a>>b;int c=0;bool f=0;for(int i=0;i<b.size();i++){f=0;for(int l=0;l<a.size();l++)if(a[l]==b[i]){f=1;break;}if(!f){cout<<-1;return 0;}}
    for(int i=0;i<b.size();i++){for(int l=0;l<a.size();l++){if(a[l]==b[i]){c++,a.erase(l,1);break;}}}cout<<c;
}
