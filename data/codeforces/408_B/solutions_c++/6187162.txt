#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[26];int b[26];
char c;
int sol;
int main(){
    scanf("%c",&c);
    while(c!='\n'){++a[c-'a'];scanf("%c",&c);}
    scanf("%c",&c);
    while((c>='a')and(c<='z')){++b[c-'a'];scanf("%c",&c);}
    for(int i=0;i<=25;++i)
    if((a[i]==0)and(b[i]!=0)){cout<<-1;return 0;}
    else{sol+=min(a[i],b[i]);}
    cout<<sol;
    return 0;
}
