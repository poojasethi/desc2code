#include<bits/stdc++.h>
using namespace std;

int main(){
	string a,b;
	cin>>a>>b;
	vector<int> x(26),y(26);
	for(int i=0;i<a.size();++i){
		x[a[i]-97]++;
	}
	for(int i=0;i<b.size();++i){
		y[b[i]-97]++;
	}
	int ans=0;
	int flag=0;
	for(int i=0;i<26;++i){
		ans+=min(x[i],y[i]);
		if(y[i]>0 && x[i]==0){
			flag=1;
		}
	}
	if(ans==0 || flag){
		cout<<"-1\n";
	}
	else{
		cout<<ans<<"\n";
	}
	return 0;
}
