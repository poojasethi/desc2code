#include <iostream>
#include <stdio.h>

using namespace std;
string s,s1;
int ans,a[30],b[30];
main()
 {
      cin>>s>>s1;
      for(int i=0;i<s.size();i++)a[s[i]-'a']++;
      for(int i=0;i<s1.size();i++)b[s1[i]-'a']++;
      
      for(int i=0;i<26;i++)
      {
              if(b[i]>0 && a[i]==0){cout<<-1;return 0;}
              ans+=min(a[i],b[i]);
      }
  cout<<ans<<endl;
 }
