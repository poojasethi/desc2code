#include<iostream>

using namespace std;

string s,t;
int a[30],b[30];
int res=0;
main()
{
	cin >> s >> t;
	for(int i=0;i<s.size();i++)
		a[s[i]-'a']++;
	for(int i=0;i<t.size();i++)
		b[t[i]-'a']++;
	for(int i=0;i<30;i++)
	{
		if(a[i]==0 && b[i]>0)
		{
			cout << -1;
			return 0;
		}
		res+=min(a[i],b[i]);
	}
	cout << res;
	return 0;
}
