#include<cstdio>
#include<algorithm>
using namespace std;

char s1[1111], s2[1111];
int ans;
int a[33], b[33];

int main() {
	scanf("%s%s", s1, s2);
	for(int i=0;s1[i];i++)
		a[s1[i] - 'a'] ++;
	for(int i=0;s2[i];i++)
		b[s2[i] - 'a'] ++;
	ans = 0;
	for(int i=0;i<26;i++)
		if (b[i] && !a[i]) { ans = -1; break; }
		else ans += min(a[i], b[i]);
	printf("%d\n", ans);
	
	return 0;
}
