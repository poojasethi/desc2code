#include<iostream>
#include<string>
using namespace std;
int main(){
	string st;
	int an[30]={0};
	int am[30]={0};
	int ans;
	cin>>st;
	for (int i=0;i<st.length();i++)
		an[st[i]-'a']++;
	cin>>st;
	for (int i=0;i<st.length();i++)
		am[st[i]-'a']++;
	ans=0;
	for (int i=0;i<26;i++)
		if (am[i]>0)
			if (an[i]==0){
				ans=-1;
				break;
			}else
			ans+=min(am[i],an[i]);
	cout<<ans<<endl;
	return 0;
}
