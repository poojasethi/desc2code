#include <iostream>
using namespace std;
int a[26], b[26];
int main()
{
	string s1, s2;
	int i, x, ans;
	cin>>s1>>s2;
	for (i=0; i<s1.size(); i++) a[s1[i]-'a']++;
	for (i=0; i<s2.size(); i++) b[s2[i]-'a']++;
	for (ans=0,i=0; i<26; i++) if (b[i]) {
		if (a[i]) {
			ans += min(a[i],b[i]);
		}
		else {
			cout<<-1<<endl; return 0;
		}
	}
	cout<<ans<<endl;
	return 0;
}
