#include <iostream>
#include <stdio.h>
using namespace std;
int a[30],b[30];
main(){
    string s;
    cin>>s;
    for(int i=0;i<s.size();i++)
        a[s[i]-'a']++;
    cin>>s;
    for(int i=0;i<s.size();i++)
        b[s[i]-'a']++;
    int ans=0;
    for(int i=0;i<30;i++){
        if(b[i]>0 && a[i]==0){
            cout<<-1;
            return 0;
        }
        ans+=min(a[i],b[i]);
    }
    cout<<ans;
}
