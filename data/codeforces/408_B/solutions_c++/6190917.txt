#include<iostream>
#include<string>
using namespace std;

int a[26], b[26], ans;
string s, r;

int main(){
  cin >> s >> r;
  for(int i = 0; i < s.length(); i++)
    a[s[i] - 'a']++;
  for(int i = 0; i < r.length(); i++)
    b[r[i] - 'a']++;
  for(int i = 0; i < 26; i++){
    ans += min(a[i], b[i]);
    if(b[i] && !a[i]){
      cout << -1;
      return 0;
    }
  }
  cout << ans;
  return 0;
}
