#include<cstdio>

int main()
{
	int a[26]={0},count=0,b[26]={0};
	char c;
	while((c=getchar())!='\n' && c!=EOF)
	{
		++a[c-'a'];
		++b[c-'a'];
	}
	/*for(i=0;i<26;i++)
		printf("%d\n",b[i]);*/
	//getchar();
	while((c=getchar())!='\n' && c!=EOF)
	{
		if(a[c-'a']>0 && count!=-1)
		{
			if(b[c-'a']>0)
			{
				//printf("%c\n",c);
				--b[c-'a'];
				++count;
			}
		}
		else
		{
			count = -1;
		}
	}
	printf("%d\n",count);
	return 0;
}
