#include<stdio.h>
int main()
{
	int a[160],b[160];
	int n,i;
	char c;
	
	for(i='a';i<='z';i++)a[i]=b[i]=0;
	while((c=getchar())!=10)a[c]++;
	while((c=getchar())!=EOF)b[c]++;
	n=0;
	for(i='a';i<='z';i++)
	{
		if(a[i]==0&&b[i])
		{
			printf("-1\n");
			return 0;
		}
		else if(a[i]>=b[i])n+=b[i];
		else n+=a[i];
	}
	printf("%d\n",n);
	return 0;
}
