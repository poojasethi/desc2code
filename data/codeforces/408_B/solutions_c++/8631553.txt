#include <bits/stdc++.h>
using namespace std;
int a[256];
int b[256];
int main(){
	char c;
	int res = 0;
	while((c = getchar()) != '\n') a[c] ++;
	while(true){
		c = getchar();
		if(c == '\n'|| c == '\0')break;
		b[c] ++;
	}
	for(int i = 0; i < 256; i ++) if(b[i] && !a[i]){
		puts("-1");
		return 0;
	}else{
		res += a[i] > b[i] ? b[i] : a[i];
	}
	printf("%i\n", res);
	return 0;
}
