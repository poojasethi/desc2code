#include<iostream>
using namespace std;
int main()
{
	int got[26],req[26],i,sum=0;
	string g,r;
	cin>>g>>r;
	for(i=0;i<26;i++)
		got[i]=req[i]=0;
	for(i=0;i<g.length();i++)
		got[(int)g[i]-97]++;
	for(i=0;i<r.length();i++)
		req[(int)r[i]-97]++;
	for(i=0;i<26;i++)
	{
		if(req[i]!=0&&got[i]==0)
		{
			cout<<"-1";
			return 0;
		}
		sum+=min(req[i],got[i]);
	}
	cout<<sum;
	
}
