#include<bits/stdc++.h>
using namespace std;

char a[1005],b[1005];
int i,aa,bb,cnt;
map< char ,int >m,mm;

int main()
{

cin>>a>>b;
aa=strlen(a);
bb=strlen(b);
for(i=0;i<aa;i++)
{
m[a[i]]++;

}
cnt=0;
for(i=0;i<bb;i++)
{
mm[b[i]]++;
}
for(i=0;i<bb;i++)
{
if(m[b[i]]==0){cnt=0;break;}
cnt+=min(mm[b[i]],m[b[i]]);
mm[b[i]]=0;
}
if(cnt!=0)
cout<<cnt;
else cout<<"-1";
return 0;
}
