#include<iostream>
#include<cstring>
using namespace std;
char a[1009],b[1009];
int cnt[230];
int main()
{
	cin>>a>>b;
	int n=strlen(a),m=strlen(b);
	memset(cnt,0,sizeof(cnt));
	for(int i=0;i<n;i++) cnt[a[i]]++;
	for(int i=0;i<m;i++) if(cnt[b[i]]==0){
		cout<<"-1"<<endl;return 0;
	}
	int tot=0;
	for(int i=0;i<m;i++){
		if(cnt[b[i]]){
			cnt[b[i]]--;tot++;
		}
	}
	cout<<tot<<endl;
	return 0;
}
