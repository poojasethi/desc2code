#include<vector>

int main()
{
int m,n;
char s[1100];
int frm[26]={0};
int frn[26]={0};
gets(s);
int i;
for(i=0;s[i];i++)
	frm[s[i]-'a']++;

for(i=0;s[i];i++)

	}
