#include <iostream>
#include <cmath>
using namespace std;

int a[30], b[30];
int main()
{
	string s, t;
	cin>> s >> t;

	for(int i = 0; i < s.size(); i ++)
		a[s[i] - 'a'] ++;
	for(int j = 0; j < t.size(); j ++)
		b[t[j] - 'a'] ++;

	int ans = 0;
	for(int j = 0; j < 26; j ++)
	{
		if(a[j] == 0 && b[j] != 0)
		{
			cout<< "-1\n";
			return 0;
		}
		ans += min(a[j], b[j]);
	}
	
	cout<< ans << endl;

}
