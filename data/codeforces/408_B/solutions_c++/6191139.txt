#include <stdio.h>
#include <iostream>
#include <string>
using namespace std;
int x[30]={0},y[30]={0},s=0;
string n,m;
int main( ){
	cin>>n>>m;
	for (int i=0;i<n.length();i++)
		x[n[i]-'a'+1]++;
	for (int i=0;i<m.length();i++)
		y[m[i]-'a'+1]++;
	for (int i=1;i<=26;i++){
		if (x[i]==0 && y[i]>x[i]){
			cout<<-1<<endl;
			return 0;
		}
		s+=min(x[i],y[i]);
	}
	cout<<s<<endl;
	return 0;
}
