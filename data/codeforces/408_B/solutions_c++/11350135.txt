#include<bits/stdc++.h>
using namespace std;
int a[26],b[26];
char s[100005],t[100005];
int main()
{
scanf("%s%s",&s,&t);
int i;
for(i=0;s[i];i++)a[s[i]-'a']++;
for(i=0;t[i];i++)b[t[i]-'a']++;
int ans = 0;

for(i=0;i<26;i++)if(a[i]==0 && b[i]!=0){ans = 0;break;} else ans+=min(a[i],b[i]);
if(ans==0)ans--;
printf("%d\n",ans);
return 0;
}
