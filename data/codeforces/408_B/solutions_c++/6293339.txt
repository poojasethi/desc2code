#include <iostream>

using namespace std;

int main()
{string s1,s2; cin>>s1>>s2;
 int k1[100]={0};
 int k2[100]={0};
for (int i=0;i<s1.size();i++) k1[s1[i]-'a']++;
for (int i=0;i<s2.size();i++) k2[s2[i]-'a']++;
int ans=0;
for (int i=0;i<100;i++)
 {if (k2[i]>0&&k1[i]==0) {cout<<-1; return 0;}
  ans+=min(k1[i],k2[i]);
 }
cout<<ans;
return 0;
}
