#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N = 1004;
char s1[N], s2[N];
int a1[34], a2[34];
int main(){
	scanf("%s%s", s1+1, s2+1);
	for(int i = 1; s1[i]; i ++) a1[s1[i]-'a'] ++;
	for(int i = 1; s2[i]; i ++) a2[s2[i]-'a'] ++;
	int ans = 0;
	for(int i = 0; i < 26; i ++){
		if(a2[i] && !a1[i]) return puts("-1"), 0;
		ans += min(a1[i], a2[i]);
	}
	printf("%d\n", ans);
}
