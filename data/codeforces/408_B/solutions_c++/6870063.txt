#include <iostream>

using namespace std;

int cnt[30];

int main() {
	string s, t;
	cin >> s >> t;
	for (int i = 0; i < s.size(); i++)
		cnt[s[i] - 'a']++;
	for (int i = 0; i < t.size(); i++)
		if (cnt[t[i] - 'a'] == 0) {
			cout << -1 << endl;
			return 0;
		}
	int ans = 0;
	for (int i = 0; i < t.size(); i++)
		ans += (cnt[t[i] - 'a']-- > 0);
	cout << ans << endl;
	return 0;
}
