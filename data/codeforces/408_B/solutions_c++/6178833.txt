#include <cstdio>

int a[30], b[30], i, c;
char s[1010];
bool f = 1;

int main()
{
	scanf("%s", s);

	for (i = 0; s[i]; i++)
		++a[s[i] - 97];

	scanf("%s", s);

	for (i = 0; s[i]; i++)
		++b[s[i] - 97];

	for (i = 'a'; f && i <= 'z'; i++)
	{
		if (b[i - 97] && !a[i - 97])
			f = 0;
		if (b[i - 97] && a[i - 97])
			c += b[i - 97] < a[i - 97] ? b[i - 97] : a[i - 97];
	}

	printf("%d\n", f ? c : -1);

	return 0;
}
