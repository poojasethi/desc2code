#include<iostream>

using namespace std ;

string s1 , s2 ;
int a[10000] , b[10000] ;
int ans=0 ;

int main()
{
	cin >> s1 >> s2 ;
	for(int i=0 ; i<s1.size() ; i++)
		a[s1[i]-'a']++ ;
	for(int i=0 ; i<s2.size() ; i++)
		b[s2[i]-'a']++ ;
	for(int i=0 ; i<26 ; i++)
	{
		if(a[i]==0 && b[i]>0)
		{
			cout << -1 << endl ;
			return 0 ;
		}
		else
			ans+=min(a[i],b[i]) ;
	}
	cout << ans << endl ;
	
}
