#include <cstdio>

const int MAXN = 1005;
int cc[MAXN];
char str[MAXN];

int main(){
	scanf("%s",str);
	for(int i=0;str[i];i++) cc[str[i]]++;
	scanf("%s",str);
	
	int ans = 0;
	for(int i=0;str[i];i++){
		if(cc[str[i]] == 0){
			printf("-1\n");
			return 0;
		}
	}
	for(int i=0;str[i];i++){
		if(cc[str[i]]){
			cc[str[i]]--;
			ans++;
		}
	}

	printf("%d\n",ans?ans:-1);
	return 0;
}
