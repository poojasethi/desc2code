#include <bits/stdc++.h>
using namespace std;

int main() {
    char a[1005], b[1005];
    int ca[26] = {}, cb[26] = {}, ans = 0;
    scanf("%s%s", a, b);
    for (int i = 0; a[i]; ++i) ca[a[i] - 'a']++;
    for (int i = 0; b[i]; ++i) cb[b[i] - 'a']++;
    for (int i = 0; i < 26; ++i) {
        if (!ca[i] && cb[i]) return puts("-1"), 0;
        ans += min(ca[i], cb[i]);
    }
    printf("%d\n", ans);
}
