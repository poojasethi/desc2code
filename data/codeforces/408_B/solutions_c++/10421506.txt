#include<iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int ans,s1,s,k[260],l[260],i;
string a,b;
main(){
cin>>a>>b;
s=a.size();
s1=b.size();
for(i=0;i<s;i++){
k[a[i]-97]++;

}
for(i=0;i<s1;i++)
l[b[i]-97]++;

for(i=0;i<=26;i++)

{
if(k[i]==0 && l[i]>0) {cout<<-1;return 0;}
if(k[i]>0 && l[i]>0) ans+=min(k[i],l[i]);
}
cout<<ans;
}
