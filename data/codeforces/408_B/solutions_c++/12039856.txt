#include <bits/stdc++.h>
using namespace std;
#define int long long

int a[26];
int b[26];
int ans = 0;

main()
{
	string n, m;
	cin >> n >> m;
	for (int i = 0; i < n.size(); i++)
		a[n[i] - 'a']++;
	for (int i = 0; i < m.size(); i++)
		b[m[i] - 'a']++;
	for (int i = 0; i < 26; i++)
		if (a[i] == 0 && b[i])
			return cout << -1, 0;
		else ans += min(a[i], b[i]);
	cout << ans << endl;
}
