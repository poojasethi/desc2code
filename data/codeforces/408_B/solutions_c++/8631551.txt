#include <string>
#include <stdio.h>
using namespace std;
int a[30], b[30], res;
bool t;
int main(){
	char c;
	while(~scanf("%c", &c)){
		if(c == '\n'){
			t = 1;
			continue;
		}
		if(!t) a[c-'a'] ++;
		else b[c-'a'] ++;
	}
	for(int i = 0; i < 26; ++ i){
		if(a[i] == 0 && b[i] > 0){
			puts("-1");
			return 0;
		}
		res += min(a[i], b[i]);
	}
	printf("%i", res > 0 ? res : -1);
	return 0;
}
