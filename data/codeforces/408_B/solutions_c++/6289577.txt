#include <bits/stdc++.h>
using namespace std;

int m1[26],m2[26];
char s[1<<10];
int main(){
	
	scanf("%s\n",s);
	for(int i=0;i<strlen(s);i++)m1[s[i]-'a']++;
	scanf("%s\n",s);
	for(int i=0;i<strlen(s);i++)m2[s[i]-'a']++;
	int ans = 0;
	for(int i=0;i<26;i++){
        ans+=min(m1[i],m2[i]);
        if(m2[i]  && m1[i]==0){
            printf("-1");
            return 0;
        }
	}
	printf("%d\n",ans);
    return 0;
}
