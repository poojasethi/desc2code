#include <cstdio>
#include <cstring>

char n[1005], m[1005];
int c[30], d[30];

int main(){
	scanf(" %s %s ", n, m);
	for(int i = 0; i < strlen(n); i++){
		c[n[i]-'a']++;
	}
	for(int i = 0; i < strlen(m); i++){
		d[m[i]-'a']++;
	}
	int a = 0;
	for(int i = 0; i < 26; i++){
		if(d[i]){
			if(c[i] == 0){
				a = -1;
				break;
			}
			else if(c[i] >= d[i])
				a += d[i];
			else
				a += c[i];
		}
	}
	printf("%d\n", a);
	return 0;
}
