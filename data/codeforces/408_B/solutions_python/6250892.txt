def count(x,s):
     j=0
     for char in s:
         if char==x:
             j+=1
     return j
s1=raw_input()
s2=raw_input()
ans=0
c=0
for i in range(len(s2)):
    if s2[i] not in s2[ :i] or i==0:
      m= count(s2[i],s2)
      n=count(s2[i],s1)
      if n==0:
          c=1
          
          break
      if m>n:
          ans+=n
      else:
          ans+=m
          
if c!=0:
    print '-1'
else:
    print (ans)
