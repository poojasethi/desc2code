str1 = raw_input()

def make_d(str1):
    d1 = {}
    for c in str1:
        if c not in d1:
            d1[c] = 0
        d1[c] += 1
    return d1

d1 = make_d(str1)

str2 = raw_input()

d2 = make_d(str2)

flag = True
for c in d2:
    if c not in d1:
        flag = False
        break

if not flag :
    print "-1"
else:
    ans = 0
    for c in d2:
        ans += min(d2[c],d1[c])
    print ans
