import string
letters = string.lowercase

s1 = raw_input()
s2 = raw_input()

s = 0
unknown = False

for c in letters:
    s1c = s1.count(c)
    s2c = s2.count(c)
    if s1c > s2c:
        s += s2c
    if (s1c <= s2c) and (s1c != 0):
        s += s1c
    if (s1c == 0) and (s2c != 0):
        unknown = True


if unknown:
    print -1
else:
    print s
