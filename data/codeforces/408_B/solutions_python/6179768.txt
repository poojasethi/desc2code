from string import ascii_lowercase

def solve(X, Y):
    H = dict.fromkeys(ascii_lowercase, 0)
    W = dict.fromkeys(ascii_lowercase, 0)
    for x in X:
        H[x] += 1
    for y in Y:
        W[y] += 1
    a = 0
    for x in ascii_lowercase:
        if W[x] > 0 and H[x] == 0:
            return -1
        if H[x] < W[x]:
            a += H[x]
        else:
            a += W[x]
    return a

X = raw_input()
Y = raw_input()

print solve(X, Y)
