a = raw_input()
b = raw_input()

def fill(str):
    d = dict()
    for ch in str:
        d[ch] = d.get(ch, 0) + 1
    return d

dicta = fill(a)
dictb = fill(b)

def solve():
    result = 0
    for k, v in dictb.items():
        if not k in dicta:
            return -1
        result += min([v, dicta[k]])
    return result

print solve()
