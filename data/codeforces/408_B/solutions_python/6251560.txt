def main():
    got = raw_input()
    wanted = raw_input()
    gotKey={' ': ' '}
    total = 0
    for charItem in got:
        if charItem in gotKey:
            gotKey[charItem]+=1
        else:
            gotKey[charItem]=1
    for targetItem in wanted:
        if targetItem in gotKey:
            if gotKey[targetItem] == 1:
                total+=1
                gotKey[targetItem] = -1
                continue
            elif gotKey[targetItem] == -1:
                continue
            elif gotKey[targetItem] >1:
                gotKey[targetItem]-=1
                total+=1
        else:
            print -1
            return
    print total
main()
