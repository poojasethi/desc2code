sheets = list(raw_input())
galand = list(raw_input())

ab = map(chr, range(ord('a'), ord('a') + 26))

res = 0

for ch in ab:
    have = sheets.count(ch)
    need = galand.count(ch)
    if (need > 0) and (have == 0):
        res = -1
        break
    else:
        res += min(have, need)

print res
