colors = [chr(i) for i in range(ord('a'), ord('z') + 1)]
have = dict.fromkeys(colors, 0)
need = dict.fromkeys(colors, 0)

p = raw_input().strip()
for c in p:
	have[c] += 1
g = raw_input().strip()
for c in g:
	need[c] += 1

res = 0
	
for c in colors:
	if (have[c] == 0) and (need[c] > 0):
		res = -1
		break
	else:
		res += min(have[c], need[c])

#print have
#print need
print res
