sheet = raw_input()
pieces = raw_input()

d= dict()
e= dict()

unis = ''.join(set(sheet))
unip = ''.join(set(pieces))

for ch in unis:
    d[ch] = sheet.count(ch)
for ch in unip:
    e[ch] = pieces.count(ch)

def findmin(d,e):
    sum = 0
    for char in e:
        if char not in d:
            return -1
        if e[char] <= d[char]:
            sum = sum + e[char]
        elif e[char] > d[char]:
            sum = sum + d[char]
    return sum

print findmin(d, e)
