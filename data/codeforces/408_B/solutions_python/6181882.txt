import sys
import math
from collections import defaultdict

def main():
  n = sys.stdin.readline().rstrip()
  m = sys.stdin.readline().rstrip()

  n_dict = {}
  for sheet in n:
    if n_dict.has_key(sheet):
      n_dict[sheet] += 1
    else:
      n_dict[sheet] = 1

  m_dict = {}
  for sheet in m:
    if m_dict.has_key(sheet):
      m_dict[sheet] += 1
    else:
      m_dict[sheet] = 1

  area = 0
  for sheet in m_dict.keys():
    m_count = m_dict[sheet]

    n_count = 0
    if n_dict.has_key(sheet):
      n_count = n_dict[sheet]

    if n_count == 0:
      print '-1'
      return
    elif n_count < m_count:
      area += n_count
    else:
      area += m_count
  print area

if __name__ == '__main__':
  main()
