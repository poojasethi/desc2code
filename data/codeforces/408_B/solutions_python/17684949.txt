def main():
	s1 = raw_input()
	s2 = raw_input()
	a1 = [s1.count(chr(ord('a') + i)) for i in xrange(26)]
	a2 = [s2.count(chr(ord('a') + i)) for i in xrange(26)]
	if any(0 == a1[i] and a2[i] != 0 for i in xrange(26)):
		print -1
		return
	print sum(min(a1[i], a2[i]) for i in xrange(26))

if __name__ == "__main__": main()
