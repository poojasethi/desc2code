str1 = raw_input()
str2 = raw_input()
num1 = [0] * 26
num2 = [0] * 26
sum = 0
for i in range(0,len(str1)):
    num1[ord(str1[i]) - ord('a')] += 1
for i in range(0,len(str2)):
    num2[ord(str2[i]) - ord('a')] += 1
for i in range(0,26):
    if num1[i] == 0 and num2[i] != 0:
        sum = -1
        break;
    sum += min(num1[i],num2[i])
print sum
