#!/usr/bin/python

input_a = list(raw_input())
input_b = list(raw_input())

uniq_a = sorted(set(input_a))
uniq_b = sorted(set(input_b))

counter0 = 0

for item in uniq_b:
        if item in uniq_a:
                counter0 = counter0 + 1

if len(uniq_b) == counter0:
        counter = 0

        for b in input_b:
                for pos_a, a in enumerate(input_a):
                        if a == b:
                                counter = counter + 1
                                del input_a[pos_a]
                                break
        if counter == 0:
                print -1
        else:
                print counter
else:
        print -1
