from sys import stdin

def read_int():
    return [int(x) for x in stdin.readline().split()][0]
def read_ints():
    return [int(x) for x in stdin.readline().split()]
def char_list():
    return [c for c in stdin.readline()][:-1]

def compute(sheets, reqs):
    used = 0
    for color, amount in reqs.items():
        if color not in sheets:
            return '-1'
        used += min(amount, sheets[color])
    return used

sheets = {}
for color in char_list():
    if color not in sheets:
        sheets[color] = 0
    sheets[color] += 1

reqs = {}
for color in char_list():
    if color not in reqs:
        reqs[color] = 0
    reqs[color] += 1

print compute(sheets, reqs)
