have_colors = list(raw_input().strip())
want_colors = list(raw_input().strip())

ret = 0

have = {}
want = {}

for h in have_colors:
    if h in have:
        have[h] += 1
    else:
        have[h] = 1

for w in want_colors:
    if w in want:
        want[w] += 1
    else:
        want[w] = 1

col_count = 0
for w in want.keys():
    if w in have:
        ret += min(have[w], want[w])
        col_count += 1

if ret == 0 or col_count < len(want.keys()):
    print -1
else:
    print ret
