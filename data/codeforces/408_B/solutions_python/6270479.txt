available = list(raw_input())
needed = list(raw_input())

nonavailable = filter(lambda x:x not in available, needed)
if len(nonavailable):
    print -1
    exit()

count_available = dict( (i, available.count(i)) for i in available )
count_needed = dict( (i, needed.count(i)) for i in needed ) 

total_len = 0
for c, n in count_needed.items():
    total_len += min(n, count_available[c])

print total_len
