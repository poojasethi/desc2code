x = [i for i in raw_input()]
y = [i for i in raw_input()]
a = {}
b = {}
for i in x:
    if(i in a):
        a[i] += 1
    else:
        a[i] = 1
for i in y:
     if(i in b):
        b[i] += 1
     else:
        b[i] = 1
im = True  
for i in y:
    if(not i in x):
        im = False
num = 0
if(im):
    for i in b:
        num += min([a[i],b[i]])
if(im):
    print "%d\n"% num
else:
    print "-1\n"
