#!/usr/bin/env python
# coding=utf-8
import string
have = raw_input()
need = raw_input()
ans = 0
counts = [0 for i in range(26)]
flag = [False for i in range(26)]
for i,a in enumerate(string.lowercase):
    counts[i] = have.count(a)
for ch in need:
    id = 0
    for i,a in enumerate(string.lowercase):
        if a == ch:
            id = i
            break
    if counts[id] > 0:
        counts[id] -= 1
        ans += 1
        flag[id] = True
    elif flag[id] == False:
        ans = -1
        break
    if ans == -1:
        break
print ans
