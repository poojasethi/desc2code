colors = list(raw_input())

cdict = {}

for i in colors:
    if i not in cdict:
        cdict[i] = 1

    else:
        cdict[i] += 1

want = list(raw_input())

area = 0

for i in want:
    if i in cdict:
        area += min(cdict[i], want.count(i))
        cdict[i] = 0
    else:
        print -1
        break

else:
    print area
