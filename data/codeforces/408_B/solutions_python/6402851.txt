from collections import Counter

d = raw_input()
counterD = Counter()
for i in range(len(d)):
    counterD[d[i]] += 1
    
counterT = Counter()    
t = raw_input()
for i in range(len(t)):
    counterT[t[i]] += 1

ans = 0
for k in counterT.keys():
    if counterT[k] and not counterD[k]:
        print '-1'
        exit(0)
        
    ans += min(counterD[k], counterT[k])

print ans if ans != 0 else -1
