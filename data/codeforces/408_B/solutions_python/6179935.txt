inp = raw_input()
oup = raw_input()

inp_map = {}
for item in inp:
    if item not in inp_map:
        inp_map[item] = 1
    else:
        inp_map[item] += 1

outp_map = {}
for item in oup:
    if item not in outp_map:
        outp_map[item] = 1
    else:
        outp_map[item] += 1

ret = 0
for item in outp_map:
    if item not in inp_map:
        ret = -1
        break
    else:
        ret += min(outp_map[item], inp_map[item])

print ret
