a = int(raw_input())
for x in range(a):
    count = 0
    b, c = [int(x) for x in raw_input().split()]
    while b > 0 and c > 0:
        if b > c:
            count += b // c
            b = b % c
        elif c > b:
            count += c // b            
            c = c % b
        else:
            b -= c
            count+=1
    print count
