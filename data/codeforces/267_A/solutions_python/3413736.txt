# http://codeforces.com/problemset/problem/267/A

def solve(a, b):
    if(a == 0 or b == 0):
	return 0
    if(a == b):
	return 1
    if(a > b):
	return solve(b, a)
    return b / a + solve(b - (b / a) * a, a)


# main

test = int(raw_input())
for tt in range(test):
    a, b = map(int, raw_input().split())
    print solve(a, b)
