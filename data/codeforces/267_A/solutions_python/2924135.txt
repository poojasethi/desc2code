import math
def numOp(a,b,counter):
    if max(a,b)%min(a,b) == 0:
        print int(max(a,b)/min(a,b) +counter)
    else:
        numOp(min(a,b),max(a,b)%min(a,b),counter + math.floor(max(a,b)/min(a,b)))
    
if __name__ == '__main__':
