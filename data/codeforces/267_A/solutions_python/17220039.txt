n = int(raw_input())
for i in range(n):
    a, b = map(int, raw_input().split())
    result = 0
    while a != 0 and b != 0:
        a, b = max(a, b), min(a, b)
        diff = (a - b) / b
        if diff == 0:
            a = a - b
            result += 1
            continue
        a = a - diff * b
        b = b
        result += diff
    print result
