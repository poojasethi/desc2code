def main():
    def f(a, b):
        total = 0
        a, b = max(a, b), min(a, b)
        while a > 0 and b > 0:
            total += a / b
            a, b = b, a % b
        return total
    
    n = int(raw_input())
    for _ in xrange(n):
        a, b = map(int, raw_input().split())
        print f(a, b)

if __name__ == '__main__':
    # import sys; sys.stdin = open('input.txt')
    main()
