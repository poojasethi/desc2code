def solve(a,b,res):
    if a > b: a,b = b,a
    d = int(b/a) #devided
    m = b % a
    if m == 0:
        return res + d
    else:
        return solve(a,m,res + d)

def main():
    n = int(raw_input())
    for i in range(n):
        a,b = map(int,raw_input().split())
        print(solve(a,b,0))

main()
