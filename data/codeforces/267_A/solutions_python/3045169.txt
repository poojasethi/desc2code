def compute(a, b):
    ans = 0
    while (a>0) and (b>0):
        if (a > b):
            k = (a//b)
            ans += k
            a = a%b
        else:
            k = (b//a)
            ans += k
            b = b%a
    return ans
    
n = int(raw_input())

ans = []
for i in range(0, n):
    r = map(int, raw_input().split())
    ans.append(compute(r[0], r[1]))

for e in ans:
    print e
