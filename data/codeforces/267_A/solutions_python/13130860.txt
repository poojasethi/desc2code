n = int(raw_input())
pairs = []
for i in range(n):
    pair = raw_input().split()
    pairs.append([int(pair[0]),int(pair[1])])
for pair in pairs:
    ans = 0
    a = pair[0]
    b = pair[1]
    while a > 0 and b > 0:
        if a > b:
            ans += (a/b)
            a = a % b
        else:
            ans += (b/a)
            b = b % a
    print ans
