#include <cstdio>
#include <set>

using namespace std;

int main()
{
	int n, m;
	set<int> s;
	scanf("%d %d", &n, &m);
	int A[n];
	int dp[n];
	for(int i = 0;i < n;i++) {
		scanf("%d", &A[i]);
	}

	for(int i = n-1;i >= 0;i--) {
		s.insert(A[i]);
		dp[i] = s.size();
	}

	while(m--) {
		int x;
		scanf("%d", &x);
		printf("%d\n", dp[x-1]);
	}
}
