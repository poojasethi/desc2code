#include<iostream>
using namespace std;
int main()
{
long  n,m,i,j,x;
cin>>n>>m;
long  a[n+10],vv[100010]={0},dis[n+10],sum=0;
for(i=0;i<n;i++)
cin>>a[i];
for(i=n-1;i>=0;i--)
{
  if(vv[a[i]]==0)
  {
      vv[a[i]]++;sum+=1;
  }
  dis[i]=sum;
}
for(j=0;j<m;j++)
{
cin>>x;
cout<<dis[x-1]<<endl;

}


}
