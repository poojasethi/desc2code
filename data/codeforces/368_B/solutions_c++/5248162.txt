#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	int a[n],l,i,b[n];
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	set<int> A;
	for(i=n-1;i>=0;i--)
	{
		A.insert(a[i]);
		b[i]=A.size();
	}
	for(i=0;i<m;i++)
	{
		scanf("%d",&l);
		printf("%d\n",b[l-1]);
	}
	return 0;
}
