#include<iostream>
#include<map>
using namespace std;
map<int,int>data;
int a[100000],ans[100000];
int main() {
	int n,m,x;
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=n-1;i>=0;i--) {
		data[a[i]]++;
		ans[i]=data.size();
	}
	for(int i=0;i<m;i++) {
		cin>>x;
		cout<<ans[x-1]<<endl;
	}
	return 0;
}
