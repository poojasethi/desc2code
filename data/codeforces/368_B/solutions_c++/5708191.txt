#include<set>
#include<algorithm>
using namespace std;

set<int>s;
int main()
{int n,m,i;

cin>>arr[i];
for(i=n-1;i>=0;i--)
{

}	
	
for(i=0;i<m;i++)
{
cin>>n;
cout<<nx[n-1]<<endl;




}

}
