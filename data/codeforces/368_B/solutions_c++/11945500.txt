#include <iostream>
#include <set>
using namespace std;

int A[100100], F[100100], n, m;
set<int> S;
int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> A[i];
    for (int i = n; i; i--) {
        S.insert(A[i]);
        F[i] = S.size();
    }
    for (int x; m--; cin >> x, cout << F[x] << "\n");
}
