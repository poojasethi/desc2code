#include<iostream>
#include<set>
using namespace std;
const int maxn=100005;
int f[maxn], s[maxn];
set<int> M;

int main(){
    int n,m; cin>>n>>m;
    for(int i=1;i<=n;++i) cin>>f[i];
    for(int i=n;i>=1;--i){
    	M.insert(f[i]);
    	s[i]+=M.size();
    }
    while(m--){
    	int x; cin>>x;
    	cout<<s[x]<<endl;
    }
}
