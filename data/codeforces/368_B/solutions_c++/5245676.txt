#include <iostream>
#include <set>

using namespace std;

int a[100005]={0}, b[100005]={0};
set<int> s;
int n, m;

int main()
{
	cin>>n>>m;
	for(int i=1; i<=n; i++) cin>>a[i];
	for(int i=n; i>=1; i--) s.insert(a[i]), b[i]=s.size();
	for(int i=0; i<m; i++)
	{
		int x;
		cin>>x; cout<<b[x]<<endl;
	}
}
