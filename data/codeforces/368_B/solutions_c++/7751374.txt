#include <iostream>
#include <set>
using namespace std;
set<int>p;
int a[100001];
main(){
    int i,n,x,m;
    cin>>n>>m;
    for(i=1;i<=n;i++){
        cin>>a[i];
    }
    for(i=n;i>=1;i--){
        p.insert(a[i]);
        a[i]=p.size();
    }
    for(i=1;i<=m;i++){
        cin>>x;
        cout<<a[x]<<endl;
    }
}
