#include<stdio.h>
int i,dp[200000],H[200000],arr[200000],n,m,a;
int main(){
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;i++)
		scanf("%d",&arr[i]);
	
	for(i=n;i>=1;i--){
		H[arr[i]]++;
		if(H[arr[i]] < 2)
			dp[i]++;

		dp[i]+=dp[i+1];
	}
	for(i=1;i<=m;i++){
		scanf("%d",&a);
		printf("%d\n",dp[a]);
	}

}
