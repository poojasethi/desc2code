#include<iostream>
#include<set>
using namespace std;
int main()
{
	set<int>myset;
	int n,a[100000],i,m,suff[100000],l;
	cin>>n>>m;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=n-1;i>=0;i--)
	{	
		myset.insert(a[i]);
		suff[i] = myset.size();
	}
	while(m--)
	{
		cin>>l;
		cout<<suff[l-1]<<endl;
	}
	return 0;
}
