#include<iostream>
#include<set>
#include<algorithm>
using namespace std;

int main()
{int n,m,i;
int arr[1000000];
int gemy[1000000];
set<int>s;
    cin>>n>>m;
for(i=0;i<n;i++)
{cin>>arr[i];}
for(i=n-1;i>-1;i--)
{
s.insert(arr[i]);
gemy[i+1]=s.size();
}      
for(i=0;i<m;i++)
{
cin>>n;
cout<<gemy[n]<<endl;
}
}
