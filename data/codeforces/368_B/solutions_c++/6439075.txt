#include<stdio.h>

char h[100005];
int arr[100009],t[100005],i,j,n,m,k,x;
int main(){
	scanf("%d %d",&n,&k);
	for(i=1;i<=n;i++)
		scanf("%d",&t[i]);
	arr[n]=1;
	h[t[n]]=1;
	for(i=n-1;i>=1;i--){
		arr[i]+=(arr[i+1])+((!h[t[i]])? 1 : 0);
		h[t[i]]=1;
	}
	
	for(i=1;i<=k;i++){
		scanf("%d",&x);
		printf("%d\n",arr[x]);
	}
	return 0;
}
