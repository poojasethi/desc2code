#include <iostream>
#include <set>
using namespace std;
int a[100005],num[100005];
set<int> s;
int main()
{
    int n,m,l;
    cin>>n>>m;
    for (int i=1;i<=n;i++) cin>>a[i];
    for (int i=n;i>0;i--) {s.insert(a[i]);num[i]=s.size();}
    while (m--)
    {
        cin>>l;
        cout<<num[l]<<endl;
    }
    return 0;
}
