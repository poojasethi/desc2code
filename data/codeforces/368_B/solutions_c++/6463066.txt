#include<iostream>
#include<map>

using namespace std;

int main()
{
	map<int ,int>m;
	int arr[100000],n,q,x,res[100000];
	cin>>n>>q;
	for(int k=0;k<n;k++)
	{
		cin>>arr[k];
	}
	for(int i=n-1;i>=0;i--)
	{
		m[arr[i]]++;
		res[i]=m.size();
	}
	for(int k=0;k<q;k++)
	{
		cin>>x;
		cout<<res[x-1]<<endl;
	}
	return 0;
}
