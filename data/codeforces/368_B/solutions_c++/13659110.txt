#include<bits/stdc++.h>

using namespace std;

int M[100005],A[100005],n,w,i;

bool H[100005];

int main(){

    cin >> n >> w;

	for(i=1;i<=n;i++)
		cin >> A[i];

	for(i=n; i ;i--){
		M[i] = M[i+1] + (!H[ A[i] ]);
		H[ A[i] ] = 1;
	}

	while(w--){
		cin >> i;
		cout << M[i] << endl;
	}

	return 0;
}
