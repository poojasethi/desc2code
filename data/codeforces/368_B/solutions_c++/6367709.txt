#include<cstdio>
using namespace std;
int a1[100000],a2[100000];
bool b1[100001];
int main()
{
	int n,m,o=0,x;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;++i)
		scanf("%d",a1+i);
	for(int i=n-1;~i;--i)
	{
		if(!b1[a1[i]])
		{b1[a1[i]]=1;++o;}
		a2[i]=o;
	}
	while(m--)
	{
		scanf("%d",&x);
		printf("%d\n",a2[x-1]);
	}
}
