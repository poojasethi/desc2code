#include<iostream>
#include<set>
using namespace std;
int nums[100005];
int main()
{
	int n,m;cin>>n>>m;
	for(int i=0;i<n;i++) cin>>nums[i];

	set<int> s;
	for(int i=n-1;i>=0;i--)
	{
		s.insert(nums[i]);
		nums[i]=s.size();
	}

	for(int i=0;i<m;i++)
	{
		int pos; cin>>pos;
		cout<<nums[pos-1]<<endl;
	}
	return 0;
}
