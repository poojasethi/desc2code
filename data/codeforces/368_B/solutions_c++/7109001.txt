#include<stdio.h>
int n,m,i,d,a[100001],flag[100001],dp[100002];
int main()
{
     scanf("%d%d",&n,&m);
     for(i=1;i<=n;i++)
        scanf("%d",&a[i]);
     for(i=n;i>0;i--)
        dp[i]=dp[i+1]+ (flag[a[i]]++ ? 0:1);
     for(i=0;i<m;i++)
     {
        scanf("%d",&d);
        printf("%d\n",dp[d]);
     }
     return 0;
}
