#include <bits/stdc++.h>
using namespace std;

int n, m, a[100001], ans;
map <int,int> mp;
int main(){
    cin>>n>>m;
    for (int i=1; i<=n; i++) cin>>a[i];
    for (int i=n; i>=1; i--) {
        mp[a[i]]++;
        a[i]=mp.size();
    }
    while (m--) {
        int x; cin>>x;
        cout<<a[x]<<endl;
    }
}
