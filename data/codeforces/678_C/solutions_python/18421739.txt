#!/usr/bin/env python

def gcd(a,b):
    """Compute the greatest common divisor of a and b"""
    while b > 0:
        a, b = b, a % b
    return a

def lcm(a, b):
    """Compute the lowest common multiple of a and b"""
    return a * b / gcd(a, b)

n, a, b, p, q = map(int, raw_input().split())

r = 0
if p>q:
    r = p*(n//a) + q*(n//b) - q*(n//(lcm(a,b)))
else:
    r = q*(n//b) + p*(n//a) - p*(n//(lcm(a,b)))

print r
