def gcd(x,y):
    if y==0:
        return x
    return gcd(y,x%y)
n,a,b,p,q=raw_input('').split()
n=int(n)
a=int(a)
b=int(b)
p=int(p)
q=int(q)
if q>p:
    blue=(n/b)*q
    red=(n/a)*p
    lcm=(a*b)/(gcd(a,b))
    rep=(n/lcm)*p
    print blue+red-rep
else:
    red=(n/a)*p
    blue=(n/b)*q
    lcm=(a*b)/(gcd(a,b))
    rep=(n/lcm)*q
    print red+blue-rep
