import sys;

def getGcm(a, b):
	if (b > a):
		return getGcm(b, a);
	if (0 == a % b):
		return b;
	return getGcm(b, a % b);

nums = map(int, sys.stdin.readline().split());

n = nums[0];
a = nums[1];
b = nums[2];
p = nums[3];
q = nums[4];

ab = int(n / (a * b / getGcm(a, b)));

a = int(n / a);
b = int(n / b);

if (p > q):
	print (a * p + (b - ab) * q);
else:
	print ((a - ab) * p + b * q);
