n,a,b,p,q = map(int, raw_input().split())

def gcd(*numbers):
    from fractions import gcd
    return reduce(gcd, numbers)

def lcm(*numbers):
    def lcm(a, b):
        return (a * b) // gcd(a, b)
    return reduce(lcm, numbers, 1)
    
def ans(a,b,p,q):
    aa = n / a
    bb = n / b - n / lcm(a,b)
    return aa * p + bb * q

print max(ans(a,b,p,q), ans(b,a,q,p))
