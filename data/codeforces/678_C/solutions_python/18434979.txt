def gcd(x , y):
	if y == 0:
		return x
	return gcd(y , x % y)

def lcm(x , y):
	ans = x * y
	ans /= gcd(x , y)
	return ans

def main():
	d = map(long , raw_input().split(' '))
	# print d
	ans = 0
	if d[3] >= d[4]:
		ans = (d[0] / d[1]) * d[3]
		ans += (d[0] / d[2] - d[0] / lcm(d[1] , d[2])) * d[4]
	else:
		ans = (d[0] / d[2]) * d[4]
		ans += (d[0] / d[1] - d[0] / lcm(d[1] , d[2])) * d[3]
	print ans

if __name__ == '__main__':
	main()
