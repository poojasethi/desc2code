# coding: utf-8


def _gcd(a, b):
    if a % b == 0:
        return b
    else:
        return _gcd(b, a%b)

line = raw_input().strip()
n, a, b, p, q = map(lambda x: int(x), line.split())


cnt_a = n / a
cnt_b = n / b
gcd = _gcd(a, b)
lcm = a * b / gcd
cnt_lcm = n / lcm

#print ">>", gcd, cnt_a, cnt_b, cnt_lcm

s = cnt_a * p + cnt_b * q
v = p if p < q else q

s -= cnt_lcm * v
print s
