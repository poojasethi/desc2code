from fractions import gcd


def lcm(numbers):
    return reduce(lambda x, y: (x * y) / gcd(x, y), numbers, 1)


entrada = map(int, raw_input().split())
n = entrada[0]
a = entrada[1]
b = entrada[3]
p = entrada[2]
q = entrada[4]
result = long()

if b >= q:
    result += (n / a) * b + (n / p) * q - (n / lcm((a, p))) * q
else:
    result += (n / p) * q + (n / a) * b - (n / lcm((a, p))) * b

print result
