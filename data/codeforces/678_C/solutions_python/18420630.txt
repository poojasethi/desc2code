import math
import sys
sys.setrecursionlimit(10000)
def gcd(a, b):
    return b if a % b == 0 else gcd(b, a % b)

def lcm(a, b):
    return a*b/gcd(a,b)
s = raw_input()
n, a, b, p, q = s.split()
n = int(n)
a = int(a)
b = int(b)
p = int(p)
q = int(q)
res = n / a * p + n / b * q - n / lcm(a, b) * (p + q)
res += n/ lcm(a, b) * max(p, q)

print res
