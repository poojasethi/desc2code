a, b, c, d, e = map(int, raw_input().split(" "))
chocs = 0
	
def mmc(num1, num2):
    a = num1
    b = num2

    resto = None
    while resto is not 0:
        resto = a % b
        a  = b
        b  = resto

    return (num1 * num2) / a

chocs += (a/b)*d
chocs += (a/c)*e
if (d > e) :
	chocs -= (a/mmc(b,c))*e
else :
	chocs -= (a/mmc(b,c))*d
	
print chocs
