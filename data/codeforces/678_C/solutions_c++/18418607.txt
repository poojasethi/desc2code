#include <bits/stdc++.h>
using namespace std;
int main() {
	long long n, a, b, p, q;
	cin >> n >> a >> b >> p >> q;
	long long rep = (a * b) / __gcd(a, b);
	long long ans1 = (n / a - n / rep) * p + (n / b) * q;
	long long ans2 = (n / b - n / rep) * q + (n / a) * p;
	cout << max(ans1, ans2) << endl;
}
