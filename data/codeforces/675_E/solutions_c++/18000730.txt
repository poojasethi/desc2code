#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<queue>
#include<vector>
using namespace std;

int rmq[100010][20],a[100010],n;
long long ans,dp[100010];

int query(int l,int r){
	int x=log2(r-l+1);
	if(a[rmq[l][x]]<a[rmq[r-(1<<x)+1][x]])
	return rmq[r-(1<<x)+1][x];
	else return rmq[l][x];
}
int main(){
	scanf("%d",&n);
	for(int i=1;i<n;i++) scanf("%d",a+i);
	a[n]=n;
	for(int i=1;i<=n;i++) rmq[i][0]=i;
	for(int j=1;(1<<j)<=n;j++)
	 for(int i=1;i+(1<<j)-1<=n;i++)
	 if(a[rmq[i][j-1]]<a[rmq[i+(1<<(j-1))][j-1]])
	 rmq[i][j]=rmq[i+(1<<(j-1))][j-1];
	 else rmq[i][j]=rmq[i][j-1];
	for(int i=n-1;i;i--){
		int tmp=query(i+1,a[i]);
		dp[i]=dp[tmp]-(a[i]-tmp)+n-i;
		ans+=dp[i];
	}
	printf("%I64d",ans);
	return 0;
}
