#include<cstdio>
#include<iostream>
using namespace std;
int n, a[112345];
long long int st[112345][18], ans;
int log(int a){
	for(int i=1;i<=19;i++)
		if((1<<i)>a)return i-1;
}
long long int min(long long int a, long long int b){
	if(a>b)return b;
	return a;
}
long long int ask(int l, int r){
	if(l>r)return 11234567890123456ll;
	return min(st[l][log(r-l+1)],ask(l+(1<<log(r-l+1)),r));
}
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n-1;i++)
		scanf("%d",&a[i]);
	st[n][0]=n;
	for(int i=n-1;i>0;i--){
		st[i][0]=ask(i+1,a[i])+n-a[i];
		for(int j=1;i+(1<<j)<=n+1;j++)
			st[i][j]=min(st[i][j-1],st[i+(1<<(j-1))][j-1]);
	}
	for(int i=1;i<=n-1;i++)
		ans+=st[i][0]-i;
	printf("%I64d",ans);
	return 0;
}
