#include<bits/stdc++.h>
using namespace std;
#define int long long
#define pii pair<int,int>
#define mp make_pair
#define FOR(i,s,e) for(int _s = (int)s,_e= (int)e, i = _s; i < _e; i++) 
int a[ 100050 ];
int dp[ 100050 ];
main(){
	int n;
	cin >> n;
	FOR(i,0,n-1) cin >> a[ i ];
	int sum = 1;
	dp[ n - 1 ] = 0;
	dp[ n - 2 ] = 1;
	set<pii> sajjad;
	sajjad.insert( mp( n - 2 , a[ n-2 ] - 1 ) );
	for(int i = n - 3; i > -1; i--){
		int jolo = a[ i ] - i - 1;
		set<pii>::iterator it = sajjad.upper_bound( mp(a[ i ]-1,1e9) );
		it--;
		dp[ i ] = dp[ it->first ] + n - i - 1 - (a[ i ]-1 - it->first);
		while( (sajjad.begin())->second <= a[ i ]-1 && sajjad.begin() != sajjad.end() ) sajjad.erase( sajjad.begin() );
		sajjad.insert( mp( i, a[ i ]-1 ) );
		sum += dp[ i ];
	}
	cout << sum <<endl;			
}
