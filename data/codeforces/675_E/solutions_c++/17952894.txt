#include <cstdio>
#include <iostream>
#include <string>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <iomanip>
#include <map>
#include <set>
#include <vector>
using namespace std;
const int MAXN = 100005;
int a[MAXN];
int f[MAXN], g[MAXN];
long long ans[MAXN];
int main(){
	int n, k;
	long long total = 0;
	cin >> n;
	for(int i = 0; i < n - 1; i++){
		scanf("%d", &a[i]);
		a[i] -= (i + 1);
	}
	int q = 0;
	ans[n - 1] = 0;
	f[0]= n - 1, g[0] = 0;
	for(int i = n - 2; i >= 0; i--){
		int l, r, k;
		for(l = -1, r = q, k = (l + r) / 2; l + 1 < r; k = (l + r) / 2){
			if(i + a[i] >= f[k])
				r = k;
			else
				l = k;
		}
		int p = f[r];
		//cout << (ans[p] + n - 1 - p) << endl;
		ans[i] = (p - i) - (min(n - 1, i + a[i]) - p) + (ans[p] + n - 1 - p);
		//cout << i + 1 << " p = " << p << ' ' << ans[i] << endl;
		total += ans[i];
		while(q >= 0 && min(n - 1, i + a[i]) >= min(n - 1, f[q] + g[q]))
			q--;
		q++;
		f[q] = i, g[q] = a[i];
	}
	cout << total << endl;
	return 0;
}
