#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;

const int maxn=1e6+5;
const int m=30;
int s,k;
int a[maxn*m][2],val[maxn*m];

void Insert(int x)
{
    int rt=1;
    for(int i=m-1;i>=0;i--)
    {
        int t=(x>>i)&1;
        if(!a[rt][t])
            a[rt][t]=++s;
        rt=a[rt][t];
        val[rt]++;
    }
}

long long Find(int x)
{
    long long rt=1,ans=0;
    for(int i=m-1;i>=0;i--)
    {
        int t=(x>>i)&1^1;
        int c=(k>>i)&1;
        if(!c)
        {
            ans+=val[a[rt][t]];
            rt=a[rt][t^1];
        }
        else
            rt=a[rt][t];
    }
    return ans+val[rt];
}

int main()
{
    int n,b[maxn];
    long long ans=0;
    scanf("%d%d",&n,&k);
    s=1;

    Insert(0);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&b[i]);
        b[i]^=b[i-1];
        ans+=Find(b[i]);
        Insert(b[i]);
    }
    printf("%lld\n",ans);
    return 0;
}
