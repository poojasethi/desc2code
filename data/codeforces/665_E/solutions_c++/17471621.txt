#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
class bor {
  public:
  int num;
  bor *a[2];
  bor() {
    num = 0;
    a[0] = a[1] = 0;
  }
};
int lim;
long long ans = 0;
bor *root = new bor();
void upd(bor *v, bool b) {
  if ((v) && (v->a[b])) {
    ans += v->a[b]->num;
  }
}
void l(int XOR) {
  bor *c, *v;
  c = v = root;
  bool e, r;
  for (int t = 1<<29; t; t >>= 1) {
    e = ((XOR & t) != 0);
    r = ((lim & t) != 0);
    if (r == 0) {
      upd(v, !e);
    }
    if (v) {
      v = v->a[e ^ r];
    }
    XOR -= e * t;
    if (! c->a[e]) {
      c->a[e] = new bor();
    }
    c = c->a[e];
    c->num++;
  }
}
int main() {
  #ifdef Vlad_kv
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
  #endif // Vlad_kv
  int q, w, e, r, t, XOR = 0;
  scanf("%d%d", &q, &lim);
  lim--;
  l(0);
  for (w = 0; w < q; w++) {
    scanf("%d", &e);
    XOR ^= e;
    l(XOR);
  }
  printf("%I64d", ans);
  return 0;
}
