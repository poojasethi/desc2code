#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;
const int MN=1e6+5;
int sz=1;
int n,k,val[MN*32],son[MN*32][2];int sum_xor[MN];
void insert(int a)
{
    int u=0;
    for(int i=30;i>=0;i--)
    {
        int w=(a>>i)&1;
        if(!son[u][w])
            son[u][w]=sz++;
        u=son[u][w];
        val[u]++;
    }
}
int get(int a)
{
    int u=0,sum=0,ans=0;
    for(int i=30;i>=0;i--)
    {
        int w=(a>>i)&1;
        if(sum+(1<<i)>=k)
        {
            ans+=val[son[u][w^1]];
            u=son[u][w];
        }
        else
        {
            sum += 1<<i;
            u=son[u][w^1];
        }
        if(!u)break;
    }
    return ans;
}
int main()
{
    scanf("%d %d",&n,&k);
    long long  ans=0;
    insert(0);
    for(int i=1;i<=n;i++)
    {
         int tp;
         scanf("%d",&tp);
         sum_xor[i]=sum_xor[i-1]^tp;
         ans += get(sum_xor[i]);
         insert(sum_xor[i]);
    }
    printf("%lld\n",ans);
    return 0;
}
