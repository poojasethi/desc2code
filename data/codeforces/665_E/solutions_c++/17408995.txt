#include <bits/stdc++.h>
using namespace std;

#define endl '\n'
#define DB(x)cout<<#x<<"="<<x<<endl;
typedef long long ll;

int MB = 30;

const int maxn = 30000000 + 10;

int trie[maxn][2];
int total[maxn];
int id = 1;

void insert(int n){
	int cur = 0;
	for (int i = MB - 1; i >= 0; --i){
		int b = n >> i & 1;
		if (!trie[cur][b]) trie[cur][b] = id++;
		cur = trie[cur][b];
		total[cur]++;
	}
}

ll count(int v, int k){
	int cur = 0;
	ll ans = 0;
	for (int i = MB - 1; i >= 0; --i){
		int b = (v >> i & 1) ^ (k >> i & 1);

		if ((k >> i & 1) == 0)
			ans += trie[cur][b ^ 1] ? total[trie[cur][b ^ 1]] : 0;

		cur = trie[cur][b];
		if (cur == 0) break;
	}
	ans += total[cur];
	return ans;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);


	int n, k;
	cin >> n >> k;
	insert(0);

	ll ans = 0;
	int c = 0;

	for (int i = 0; i < n; ++i){
		int v; cin >> v;
		c ^= v;
		ll cur = count(c, k);
		ans += cur;

		insert(c);
	}

	cout << ans << endl;

	return 0;
}
