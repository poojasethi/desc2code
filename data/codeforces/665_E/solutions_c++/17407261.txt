#include<bits/stdc++.h>
#define LL long long
using namespace std;
inline void splay(int &v){
	v=0;char c=0;int p=1;
	while(c<'0' || c>'9'){if(c=='-')p=-1;c=getchar();}
	while(c>='0' && c<='9'){v=(v<<3)+(v<<1)+c-'0';c=getchar();}
	v*=p;
}
int v[32000000],t[32000000][2];
int a[1000010],tot,n,k;
int s[55],tk[55];
void insert(int x){
	for(int i=30;i>=1;i--){
		s[i]=x&1;x>>=1;
	}
	int now=0;
	for(int i=1;i<=30;i++){
		if(!t[now][s[i]])t[now][s[i]]=++tot;
		now=t[now][s[i]];v[now]++;
	}
}
int query(int x){
	for(int i=30;i>=1;i--){
		s[i]=x&1;x>>=1;
	}
	int now=0,ans=0;
	for(int i=1;i<=30;i++){
		if(tk[i]==0)ans+=v[t[now][s[i]^1]];
		now=t[now][s[i]^tk[i]];
		if(!now)break;
	}
	return ans+v[now];
}
int main(){
	splay(n);splay(k);
	for(int i=1;i<=n;i++){
		splay(a[i]);a[i]^=a[i-1];
	}
	for(int i=30;i>=1;i--)tk[i]=k&1,k>>=1;
	LL ans=0;
	for(int i=1;i<=n;i++){
		insert(a[i-1]);
		ans+=(LL)query(a[i]);
	}
	printf("%I64d\n",ans);
}
