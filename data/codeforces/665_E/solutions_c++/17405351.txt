#include <cstdio>
#include <cstring>
struct node
{
    node* ch[2];
    int cnt;
    void init()
    {
        ch[0]=ch[1]=0;
        cnt=0;
    }
}buff[32000100],*root;
int cnt,a[1000006],n,k;
node* newnode()
{
    ++cnt;
    buff[cnt].init();
    return buff+cnt;
}
void ins(int x)
{
    node*cur;
    int i,j;
    for(cur=root,i=31;i>=0;--i)
    {
        j=(x>>i)&1;
        if(!cur->ch[j])
            cur->ch[j]=newnode();
        cur=cur->ch[j];
        ++cur->cnt;
    }
}
typedef long long LL;
LL ans;
void calc(int x)
{
    int i,j,t;
    node*cur;
    for(cur=root,i=31;i>=0;--i)
    {
        j=(k>>i)&1;
        t=(x>>i)&1;
        t^=j;
        if(!j)
        {
            if(cur->ch[!t])
                ans+=cur->ch[!t]->cnt;
        }
        if(!cur->ch[t])
            return;
        cur=cur->ch[t];
    }
}
int main()
{
    cnt=0;
    root=newnode();
    scanf("%d%d",&n,&k);
    a[0]=0;--k;
    ins(0);
    for(int i=1;i<=n;++i)
    {
        scanf("%d",a+i);
        a[i]^=a[i-1];
        calc(a[i]);
        ins(a[i]);
    }
    printf("%I64d\n",ans);
    return 0;
}
