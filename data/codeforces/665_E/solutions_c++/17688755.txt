#include <bits/stdc++.h>

#define ll long long

const int N = 1e6;
const int BIT = 31;
const int S = N*BIT, SIGMA = 2;

int n, k;
int xorv;
ll ans;

int tr[S+9][SIGMA];
int size[S+9];
int trn = 1;

int qry(int v) {
    int s = k, ret = 0, cur = 1;
    for(int i = 30; i >= 0; --i) {
        if(!cur) break;
        int c = (v>>i)&1;
        if((1<<i) >= s) ret += size[tr[cur][c^1]], cur = tr[cur][c];
        else s -= 1<<i, cur = tr[cur][c^1];
    }
    return ret;
}

inline void ins(int v) {
    int cur = 1;
    for(int i = 30; i >= 0; --i) {
        int &nex = tr[cur][(v>>i)&1];
        if(!nex) nex = ++trn;
        cur = nex, size[cur]++;
    }
}

int main() {
#ifndef ONLINE_JUDGE
    freopen("665E.in", "r", stdin);
    freopen("665E.out", "w", stdout);
#endif

    scanf("%d%d", &n, &k);
    ins(0);
    for(int i = 1; i <= n; ++i) {
        int x;
        scanf("%d", &x);
        xorv ^= x;
        ans += qry(xorv);
        ins(xorv);
    }
    std::cout << ans << std::endl;
    
#ifndef ONLINE_JUDGE
    fclose(stdin);fclose(stdout);
#endif
    return 0;
}
