#include<cstdio>
using namespace std;
int e[11234567][2],num[11234567],ma,n,k,b,c;
void add(int a){
	num[1]++;
	for(int i=30,p=1;i>=0;i--){
		if(e[p][(a>>i)&1]==0)
			e[p][(a>>i)&1]=ma++;
		p=e[p][(a>>i)&1];
		num[p]++;
	}
}
int ask(int a){
	int anss,p=1;
	anss=0;
	for(int i=30;i>=0;i--){
		if((k>>i)&1)p=e[p][1-((a>>i)&1)];
		else{
			anss+=num[e[p][1-((a>>i)&1)]];
			p=e[p][(a>>i)&1];
		}
	}
	return anss+num[p];
}
		
int main(){
	ma=2;
	long long int ans=0;
	c=0;
	scanf("%d%d",&n,&k);
	while(n--){
		add(c);
		scanf("%d",&b);
		c^=b;
		ans+=ask(c);
	}
	printf("%lld",ans);
	return 0;
}
