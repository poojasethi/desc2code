#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<string.h>
#include<algorithm>
#include<cmath>
using namespace std;

int f[12000005][2],a[12000005],x,i,j,n,m,t,k,b;
long long ans;

int main()
{
	t=1;m=0;ans=0;
    scanf("%d%d",&n,&k);
    for(i=0;i<n;i++){
    x=1;for(j=30;j>=0;j--){
	if(!(f[x][(m>>j)&1]))
	f[x][(m>>j)&1]=++t;
	x=f[x][(m>>j)&1];
	a[x]++;}
	scanf("%d",&b);m^=b;
	x=1;for(j=30;j>=0;j--){
	if(!((k>>j)&1)){
	ans+=(long long)a[f[x][((m>>j)&1)^1]];
	x=f[x][(m>>j)&1];}
	if((k>>j)&1)x=f[x][((m>>j)&1)^1];}
	ans+=(long long)a[x];}
	printf("%I64d\n",ans);
    return 0;
}
