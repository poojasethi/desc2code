#include<cstdio>
using namespace std;

int l[32000000],r[32000000],s[32000000],tot=1,n,k;

void insert(int x)
{
	int tmp = 1<<30,now = 1;
	for(int i = 1;i <=31;i++)
	{
		if(x&tmp)
		{
			if(r[now]==0)r[now]=++tot;
			now=r[now];
		}
		else
		{
			if(l[now]==0)l[now]=++tot;
			now=l[now];
		}
		++s[now];
		tmp = tmp>>1;
	}
}

int sum(int now,int x,int tmp)
{
	//printf("%d\n",tmp);
	//printf("%d %d %d\n",now,x,tmp);
	if(tmp==0)return s[now];
	if(k&tmp)
	{
		if(x&tmp)return sum(l[now],x,tmp>>1);
		else return sum(r[now],x,tmp>>1);
	}
	else
	{
		if(x&tmp) return sum(r[now],x,tmp>>1)+s[l[now]];
		else return sum(l[now],x,tmp>>1)+s[r[now]];
	}
	return 0;
}

int main(){
	scanf("%d%d",&n,&k);
	int last = 0,x;
	insert(0);
	
	long long ans = 0;
	for(int i = 1;i <= n;i++)
		{
			scanf("%d",&x);
			x = x^last;
			ans += sum(1,x,1<<30);
			//printf("%I64d\n\n",ans);
			last = x;
			insert(x);
		}
	printf("%I64d\n",ans);
	return 0;
}
