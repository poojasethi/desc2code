#include <iostream>
#include <cstdio>

using namespace std;

typedef long long int ll;

int nv=1;
int v[32000000][2];
int c[32000000];

void inserta(int val)
{
  int iv=0;
  c[iv]++;
  for (int b=29;b>=0;b--) {
    int s=(val&(1<<b))!=0;
    if (v[iv][s]==0) v[iv][s]=nv++;
    iv=v[iv][s];
    c[iv]++;
  }
}

void elimina(int val)
{
  int iv=0;
  c[iv]--;
  for (int b=29;b>=0;b--) {
    int s=(val&(1<<b))!=0;
    iv=v[iv][s];
    c[iv]--;
  }
}

int n,k;

int calcula(int x)
{
  int sol=0;
  int iv=0;
  for (int b=29;b>=0;b--) {
    int sk=(k&(1<<b))!=0;
    int sx=(x&(1<<b))!=0;
    if (sk==0) {
      if (v[iv][1^sx]) sol+=c[v[iv][1^sx]];
      if (v[iv][sx]==0) return sol;
      iv=v[iv][sx];
    } else {
      if (v[iv][1^sx]==0) return sol;
      iv=v[iv][1^sx];
    }
  }
  sol+=c[iv];
  return sol;
}

int a[2000000];
int x[2000000];

int main()
{
  scanf("%d%d",&n,&k);
  for (int i=1;i<=n;i++) {
    scanf("%d",&(a[i]));
    x[i]=x[i-1]^a[i];
    inserta(x[i]);
  }
  ll sol=0;
  for (int i=1;i<=n;i++) {
    sol+=calcula(x[i-1]);
    elimina(x[i]);
  }
  cout<<sol<<endl;
}
