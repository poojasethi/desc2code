#include<iostream>
using namespace std;
struct _
{
	int son[2],n;
}z[30000000];
int a,b,c,d,i,bit[30],tot;
long long ans;
void insert(int o)
{
	int i=0,j=0,x;
	z[j].n++;
	while(i<30)
	{
		x=(o&bit[i])==0?0:1;
		if(z[j].son[x]==0)z[j].son[x]=++tot;
		j=z[j].son[x];
		z[j].n++;
		i++;
	}
}
void find(int o)
{
	int i=0,j=0,k=0,x,y;
	while(i<30)
	{
		x=(o&bit[i])==0?0:1,y=x^1;
		if(k+bit[i]>=b)
		{
			if(z[j].son[y]>0)ans+=z[z[j].son[y]].n;
			j=z[j].son[x];
		}
		else
		{
			if(z[j].son[y]!=0)
			{ 
				k+=bit[i];
				j=z[j].son[y];
			}
			else
			j=z[j].son[x];
		}
		if(j==0)return;
		i++;
	}
}
int main()
{
	ios::sync_with_stdio(0);
	bit[29]=1;
	i=28;while(i>-1){bit[i]=bit[i+1]<<1;i--;}
	cin>>a>>b;
	insert(0);
	i=0;while(i<a)
	{
		cin>>c;
		d^=c;
		find(d);
		insert(d);
		i++;
	}
	cout<<ans;
	return 0;
}
