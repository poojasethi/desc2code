#include <iostream>
#include <cstdio>
using namespace std;
typedef long long LL;
const int MAX=1e6+5;
int node[MAX*33][2],val[MAX*32],nnode=1,n,m;
void Insert(int x)
{
    int u=0;
    for(int i=30;i>=0;i--)
    {
        int k=(x>>i)&1;
        if(!node[u][k])
            node[u][k]=nnode++;
        u=node[u][k];
        val[u]++;
    }
}
int get(int x)
{
    int u=0,sum=0,k,ans=0;
    for(int i=30;i>=0;i--)
    {
        k=(x>>i)&1;
        if(sum+(1<<i)>=m)
        {
            ans+=val[node[u][k^1]];
            u=node[u][k];
        }
        else
        {
            sum+=1<<i;
            u=node[u][k^1];
        }
        if(!u)
            break;
    }
    return ans;
}
int main()
{
    int per=0;
    LL ans=0;
    scanf("%d%d",&n,&m);
    Insert(0);
    for(int i=1;i<=n;i++)
    {
        int tp;
        scanf("%d",&tp);
        per^=tp;
        ans+=get(per);
        Insert(per);
    }
    printf("%lld\n",ans);
    return 0;
}
