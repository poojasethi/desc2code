#include <cstdio>
#define N 1000005
#define M 30
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define FD(a, b, c) for(int a = (b); a >= (c); a--)

int n, m, a[N], t[N * M][2], c[N * M], p;
long long ans;

int main(){
	scanf("%d %d", &n, &m);
	m--;
	FI(i, 1, n){
		scanf("%d", &a[i]);
		a[i] ^= a[i - 1];
	}
	
	FI(i, 0, n){
		int r = 0;
		FD(j, M - 1, 0){
			int k = (a[i] & 1 << j) ? 1 : 0;
			if(!t[r][k]) t[r][k] = ++p;
			r = t[r][k];
			c[r]++;
		}
		
		r = 0;
		FD(j, M - 1, 0){
			int k = (a[i] & 1 << j) ? 1 : 0, l = (m & 1 << j) ? 1 : 0;
			if(!l) ans += c[t[r][k ^ l ^ 1]];
			if(!t[r][k ^ l]) break;
			r = t[r][k ^ l];
		}
	}
	
	printf("%I64d\n", ans);
}
