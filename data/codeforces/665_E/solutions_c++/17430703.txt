#include<bits/stdc++.h>
using namespace std;
const int maxn = 31e6+6;
struct Tri
{
    int ch[maxn][2];
    int sz[maxn];
    int tot;
    void init()
    {
        memset(ch,0,sizeof(ch));
        memset(sz,0,sizeof(sz));
        tot=2;
    }
    void insert(int x)
    {
        int u=1;sz[u]++;
        for(int i=30;i>=0;i--)
        {
            int p = (x>>i)&1;
            if(!ch[u][p])ch[u][p]=tot++;

            //cout<<tot<<endl;
            u=ch[u][p]; sz[u]++;
        }
      //  sz[u]++;
    }
    int get(int x,int y)
    {
        int u=1;
        long long ans = 0;
        for(int i=30;i>=0;i--)
        {
            int p = (x>>i)&1^1;
            int q = (y>>i)&1;
            if(q==0)ans+=sz[ch[u][p]],u=ch[u][p^1];
            else u=ch[u][p];
            if(u==1) return ans;
        }
        return ans+sz[u];
    }
}T;
int main()
{
    T.init();
    int n,k;
    scanf("%d%d",&n,&k);
    int pre = 0;
    long long ans = 0;
    T.insert(0);
    for(int i=1;i<=n;i++)
    {
        int x;scanf("%d",&x);
        pre^=x;
        ans+=T.get(pre,k);
        T.insert(pre);
    }
    cout<<ans<<endl;
}
