#include <cmath>
#include <cstdio>
#include <cstring>
#include <set>

#include <string>

#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;
//typedef int uint;

int main()
{
	ios::sync_with_stdio(false);

	int n;
	unsigned k;
	cin >> n >> k; --k;
	vector<unsigned> a(n);
	for(int i=0; i<n; ++i)
		cin >> a[i];
	
	vector<unsigned> b(n+1); b[0] = 0;
	for(int i=0; i<n; ++i)
		b[i+1] = b[i] ^ (a[i]);
	sort(b.begin(), b.end());
	
	ll ans = 0;
	
	for(int shift=30; 0<=shift; --shift)
	{
		if((k>>shift) & 1)
			continue;
		
		unsigned ks = k>>shift;
		int nn = 0;
		for(int i=0; i<n+1; ++i)
		{
			if(!(0 < i && (b[i]>>shift) == (b[i-1]>>shift)))
			{
				unsigned c = (ks+1) ^ (b[i] >> shift);
				nn = lower_bound(b.begin(), b.end(), (c+1)<<shift) - lower_bound(b.begin(), b.end(), c<<shift);
			}
			ans += nn;
		}
	}
	
	cout << ans/2 << "\n";

	return 0;
}
