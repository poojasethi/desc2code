#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
int n,k;
int son[40100000][2],size[40100000];
int tot=1;
int a[1010000];
void insert(int x)
{
	int now = 1;size[now]++;
	for (int i = 30;i >= 0;i--)
	{
		bool t = ((1 << i) & x);
		if (!son[now][t]) son[now][t]=++tot;
		now = son[now][t];
		size[now]++;
	}
}
int find(int x)
{
	int now = 1,res = 0;
	for (int i = 30;i >= 0;i--)
	{
		bool t1 = ((1 << i) & x);
		bool t2 = ((1 << i) & k);
		bool t;
		if (t2 == 1) t = 1 ^ t1;
		else 
			res += size[son[now][1 ^ t1]],
			t = t1;
		now = son[now][t];
	}
	res += size[now];
	return res;
}
int main()
{
	cin >> n >> k;
	for (int i = 1;i <= n;i++) scanf("%d",&a[i]);
	insert(0);
	int temp = 0;
	long long ans = 0;
	for (int i = 1;i <= n;i++)
	{
		temp ^= a[i];
		ans += find(temp);
		insert(temp);
		//cout << ans << endl;
	}
	cout << ans << endl;
}
