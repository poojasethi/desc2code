s = raw_input();
l = len(s);

p = range(l + 1);
p[0] = -1;
p[1] = 0;

for i in range(2, l + 1):
    t = p[i - 1];
    while t != -1 and s[i - 1] != s[t]:
        t = p[t];
    p[i] = t + 1;

cnt = [1] * (l + 1);

for i in range(l, 0, -1):
    cnt[p[i]] += cnt[i];

t = l;
ans = [];

while t != 0:
    ans.append(t);
    t = p[t];

print len(ans);
for i in reversed(ans):
    print i, cnt[i];
