#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
char s[100005];
int len,fail[100005],cnt[100005],p=0;
bool put[100005];
int main(){
	cin.getline(s,100005);
	len=strlen(s);
	for(int i=1,j=fail[0]=-1;i<len;i++){
		while(j>=0 && s[j+1]!=s[i]) j=fail[j];
		if(s[j+1]==s[i]) j++;
		fail[i]=j;
	}
	memset(cnt,0,sizeof(cnt));
	memset(put,0,sizeof(put));
	for(int i=0;i<len;i++) cnt[fail[i]+1]++;
	for(int i=len-1;i>0;i--){
		cnt[fail[i]+1]+=cnt[i+1];
	}
	int k=len-1;
	while(k>=0){
		p++;
		put[k+1]=1;
		k=fail[k];
	}
	printf("%d\n",p);
	for(int i=1;i<=len;i++){
		if(put[i]) printf("%d %d\n",i,cnt[i]+1);
	}
	return 0;
}
