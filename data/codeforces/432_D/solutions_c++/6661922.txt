#include <bits/stdc++.h>
const int N = 1111111;
char s[N];
int next[N], dp[N];
bool f[N];
int main() {
    scanf("%s", s);
    int len = strlen(s);
    next[0] = -1;
    int j = -1;
    for(int i = 0; i < len; i++) {
        while(j > -1 && s[i] != s[j]) {
            j = next[j];
        }
        j++;
        next[i + 1] = j;
    }
    for(int i = len; i >= 1; i--) {
        dp[next[i]] = dp[next[i]] + (++dp[i]);
    }
    int ans = 0;
    j = len;
    while(j != -1) {
        ans++;
        f[j] = true;
        j = next[j];
    }
    printf("%d\n", ans - 1);
    for(int i = 1; i <= len; i++) {
        if(f[i]) {
            printf("%d %d\n", i, dp[i]);
        }
    }
}
