#include<cstdio>
#include<cstring>
const int N = 1e5+5;
char c[N];		
int next[N],re[N],dp[N];
void getJump() {
	int k = 0;
	int n = strlen(c+1);
	for (int i = 2; i <= n; i++)
	{
		while (k && c[i] != c[k+1])
			k = next[k];

		if (c[i] == c[k+1])
			k++;
		next[i] = k;
	}
}
int main()
{
	int i,j,k;
	scanf("%s",c+1);
	getJump();

	int len=strlen(c+1);
	int co=0;
	for(i=next[len];i;i=next[i])
		re[co++]=i;
	memset(dp,0,sizeof(dp));
	printf("%d\n",co+1);
	for(i=len;i;i--)
	{
		dp[i]++;
		dp[next[i]]+=dp[i];
	}
	for(i=co-1;i>=0;i--)
		printf("%d %d\n",re[i],dp[re[i]]);
	printf("%d %d\n",len,dp[len]);
	return 0;
}
