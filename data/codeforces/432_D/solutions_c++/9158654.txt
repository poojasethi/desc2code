#include <iostream>

using namespace std;

string s;
const int N=2e5+10;
int k, p[N], n,d[N],i;

void cmp(string s)
{
p[0]=0;
for (int i=1,j=0;i<s.size();i++)
 {
  while (j && s[j]!=s[i]) j=p[j-1];
  if (s[j]==s[i]) j++;
  p[i]=j;
  //cout<<p[i]<<' ';
 }

}

void out(int x)
{
 if (x==0) {cout<<k<<endl;return;}
 k++;
 out(p[x-1]);
 cout<<x<<' '<<d[x]<<endl;
}

int main() {
 cin >> s;
n=s.size();
cmp(s);

for (i=0;i<=n;i++) d[i]=1;
for (i=n;i;i--) d[p[i-1]]+=d[i];

k=0;
out(n);
//cout<<p[n+n];
}
