#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;
#define MAX 101111

char p[MAX];
int Next[MAX], siz[MAX], res[MAX], tot, lp;

void Get_Next() {
	for (int i = 2; i <= lp; i++) {
		int tp = Next[i-1];
		while (tp && p[tp+1] != p[i])
			tp = Next[tp];
		Next[i] = tp ? tp + 1 : p[1] == p[i];
	}
}

int main() {
	scanf("%s", p + 1);
	
	lp = strlen(p + 1);
	Get_Next();
	for (int i = lp; i >= 1; i--) {
		siz[i]++;
		if (Next[i])
			siz[Next[i]] += siz[i];
	}
	res[++tot] = lp;
	int tp = Next[lp];
	while (tp) {
		res[++tot] = tp;
		tp = Next[tp];
	}
	
	printf("%d\n", tot);
	for (int i = tot; i >= 1; i--)
		printf("%d %d\n", res[i], siz[res[i]]);
	return 0;
}
