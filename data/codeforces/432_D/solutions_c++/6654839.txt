#include<cstdio>
#include<cstring>
#include<cmath>
#include<cstdlib>
#include<algorithm>
#define mn 100111
using namespace std;
typedef long long ll;
char s[mn];int v[mn],ans[mn],next[mn],tot;
int main()
{
	scanf("%s",s);int l=strlen(s),j=0;
	next[0]=0;next[1]=0;
	for(int i=1;i<l;i++){
		int j=next[i];
		while(j&&s[j]!=s[i])j=next[j];
		if(j!=-1&&s[i]==s[j])j++;next[i+1]=j;
	}
	for(int p=l;p;p=next[p])++tot,v[p]=1;
	for(int i=1;i<=l;i++)ans[i]=1;
	for(int i=l;i;i--)ans[next[i]]+=ans[i];
	printf("%d\n",tot);
	for(int i=1;i<=l;i++)if(v[i])printf("%d %d\n",i,ans[i]);
	return 0;
}
