#include <bits/stdc++.h>
using namespace std;
const int N=1e5+10;
char s[N];
int p[N], f[N], a[N], cnt;
int main(){
	scanf("%s", s+1);
	int len = strlen(s+1);
	p[1]=0;
	int j=0;
	for(int i=2; i<=len; i++){
		while(j && s[i]!=s[j+1]) j=p[j];
		if(s[j+1] == s[i]) ++j;
		p[i]=j;
	}
	j=len, cnt=0;
	while(j){
		f[j]++, cnt++;
		j = p[j];
	}
	for(int i=len; i>=1; --i) a[p[i]]++;
	for(int i=len; i>=1; --i) a[p[i]]+=a[i];
	printf("%d\n", cnt);
	for(int i=1; i<=len; i++){
		if(f[i]) printf("%d %d\n", i, a[i]+1);
	}
	return 0;
}
