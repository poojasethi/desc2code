#include<bits/stdc++.h>

using namespace std;

#define mod 1000000007
#define LL long long

int pre[100001],ds[100001];
int tot;

void printem(int x)
{
    if(x==0)
    {
        cout<<tot<<endl;
        return;
    }

    tot++;
    printem(pre[x-1]);
    cout<<x<<' '<<ds[x]<<endl;
}

int main()
{
    string s;
    int i,j;
    cin>>s;

    pre[0] = 0;
    j = 0;
    for(i=1;i<s.size();i++)
    {
        while(j && s[i]!=s[j])
            j = pre[j-1];
        if(s[i]==s[j])
            j++;
        pre[i] = j;
    }

    for(i=0;i<=s.size();i++)
        ds[i] = 1;
    for(i = s.size();i>0;i--)
        ds[pre[i-1]]+=ds[i];

    printem(s.size());
    return 0;
}
