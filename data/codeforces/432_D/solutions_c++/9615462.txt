#include<cstdio>
#include<cstring>
#define ull unsigned long long
#define f(x,y,z) for(int x=y;x<=z;x++)
#define fd(x,y,z) for(int x=y;x>=z;x--)
const int e=100005;
char p[e];
int len,j,tot,next[e],dp[e],ans[e];
int main()
{
	scanf("%s",&p);
	len=strlen(p);
	next[0]=-1;
	j=-1;
	f(i,1,len-1)
	{
		while(j!=-1 && p[j+1]!=p[i])j=next[j];
		if(p[j+1]==p[i])j++;
		next[i]=j;
	}
	f(i,0,len-1)dp[i]=1;
	fd(i,len-1,1)if(next[i]!=-1)dp[next[i]]+=dp[i];
	int i=len-1;
	while(i!=-1)
	{
		ans[++tot]=i;
		i=next[i];
	}
	printf("%d\n",tot);
	fd(i,tot,1)printf("%d %d\n",ans[i]+1,dp[ans[i]]);
	return 0;
}
