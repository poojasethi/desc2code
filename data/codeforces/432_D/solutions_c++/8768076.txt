#include <iostream>
#include <string.h>
using namespace std;
int p[100005],d[100005],q,c[100005];
char s[100005];
int n,k2;
main(){
	cin>>s+1;
	n=strlen(s+1);
	k2=0;
	for(int k=2;k<=n;k++){
		while(k2 && s[k]!=s[k2+1]) k2=p[k2];
		if(s[k]==s[k2+1])	k2++;
		p[k]=k2;
	}
	for(int k=1;k<=n;k++){d[k]=1;}
	for(int k=n;k>=1;k--){d[p[k]]+=d[k];}
	k2=p[n];
	while(k2>0){
        q++;
		c[q]=k2;
		k2=p[k2];
	}
	cout<<q+1<<endl;
	for(int k=q;k>=1;k--){cout<<c[k]<<" "<<d[c[k]]<<endl;}
	cout<<n<<" "<<1<<endl;
}
