#include<cstdio>
#include<cstring>
#define nn 100100
using namespace std;
int l,t,i,ans,vis[nn],an[nn],next[nn];
char s[nn];
void kmp(char *seq,int m)
 {int i,j=0;
  next[1]=0;
  for (i=2;i<=m;i++)
          {
  while (j&&seq[i]!=seq[j+1])j=next[j];
  if (seq[j+1]==seq[i])j++;
  next[i]=j;
          }
 }
int main()
{
 scanf("%s",s+1);
 l=strlen(s+1);
 kmp(s,l);
 t=next[l];
 while (t){
       an[++ans]=t;
       t=next[t];
          }
for (i=1;i<=l;i++)vis[i]=1;
for (i=l;i>=1;i--)vis[next[i]]+=vis[i];
printf("%d\n",ans+1);
for (i=ans;i;i--)
printf("%d %d\n",an[i],vis[an[i]]);
printf("%d 1\n",l);
return 0;
}
