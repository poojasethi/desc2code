#include<cstdio>
#include<cstring>
const int maxn=100086;

char s[maxn];
int next[maxn],cnt[maxn];
int tot,le[maxn],times[maxn];

int main()
{


	gets(s);
	int n=strlen(s);
	int i,j;
	next[0]=next[1]=0;
	for(i=1;i<n;i++){
		j=next[i];
		while( j && s[i]!=s[j]) j=next[j];
		next[i+1]= s[i]==s[j]? j+1:0;
	}
	for(i=0;i<=n;i++) cnt[i]=1;
	for(i=n;i>0;i--){
		if(next[i]) cnt[next[i]]+=cnt[i];
	}
	j=n;
	tot=0;
	le[tot]=n;times[tot++]=1;
	j=next[j];
	while(j){
		le[tot]=j;times[tot++]=cnt[j];
		j=next[j];
	}
	printf("%d\n",tot);
	for(i=tot-1;i>=0;i--)
	{
		printf("%d %d\n",le[i],times[i]);
	}
	return 0;
}
