#include <stdio.h>
#include <string.h>
const int O = 100005;
char s[O];
int next[O], cnt[O], n, tot;
void fall(){
    next[1] = 0;
    for(int i = 2, j = 0; i <= n; i++){
        while(j && s[i] != s[j + 1]) j =  next[j];
        if(s[i] == s[j + 1]) j++;
        next[i] = j;
    }
}
void print(int i){
    if(next[i]) tot++, print(next[i]);
    else printf("%d\n", tot + 1);
    printf("%d %d\n", i, cnt[i]);
}
int main(){
    while(~scanf("%s", &s[1])){
        //memset(cnt, 0, sizeof(cnt));
        tot = 0;
        n = strlen(&s[1]);
        fall();
        for(int i = n; i; i--) cnt[i]++, cnt[next[i]] += cnt[i];
        print(n);
    }
    return 0;
}
