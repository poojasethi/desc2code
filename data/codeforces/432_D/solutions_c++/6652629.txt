#include<cstdio>
#include<cstring>
#include<vector>
#define N 100010
using namespace std;
char s[N];
int f[N],cnt[N]={1};
int main(){
	int len,i,tmp;
	vector<int> ans;
	scanf("%s",s);
	len=strlen(s);
	memset(f,-1,sizeof(f));
	for(i=1;i<len;i++){
		cnt[i]=1;
		tmp=f[i-1];
		while(1){
			if(s[tmp+1]==s[i]){
				f[i]=tmp+1;
				break;
			}
			else if(tmp==-1){
				f[i]=-1;
				break;
			}
			else tmp=f[tmp];
		}
	}
	for(i=len-1;i>=0;i--){
		if(f[i]!=-1) cnt[f[i]]+=cnt[i];
	}
	tmp=len-1;
	while(tmp!=-1){
		ans.push_back(tmp);
		tmp=f[tmp];
	}
	printf("%d\n",ans.size());
	for(i=ans.size()-1;i>=0;i--){
		printf("%d %d\n",ans[i]+1,cnt[ans[i]]);
	}
	return 0;
}
