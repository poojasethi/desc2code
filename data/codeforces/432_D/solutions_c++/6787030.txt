#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=100005;
	
char s[N];
int next[N],a[N],b[N]={1},cnt=0;
inline void calnext(){
	int i,j,len=strlen(s);
	for(next[0]=j=-1,i=1;i<len;i++){
		while(j>=0&&s[i]!=s[j+1]) j=next[j];
		if(s[j+1]==s[i]) j++;
		next[i]=j;
	}
}	
		
int main(){
	int i,n,m,T,x,t,p;
	scanf("%s",s);
	calnext();
	n=strlen(s);
	for(i=0;i<n;i++) b[i]=1;
	for(p=n-1;next[p]!=-1;p=next[p]) a[cnt++]=next[p];//a[cnt++]=n-1;
	for(i=n-1;i>=0;i--) if(next[i]!=-1) b[next[i]]+=b[i];
	printf("%d\n",cnt+1);
	for(i=cnt-1;i>=0;i--) printf("%d %d\n",a[i]+1,b[a[i]]);
	printf("%d 1\n",n);
	return 0;
}
