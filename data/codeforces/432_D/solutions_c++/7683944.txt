#include <bits/stdc++.h>

using namespace std;

int f[100005];
int sub[100005];

int main()
{
    string s;
    cin >> s;
    int n = s.length();
    int j = -1;
    f[0] = j;
    for (int i = 0; i < n; i++)
    {
        while (j >= 0 && s[i] != s[j]) j = f[j];
        j++;
        f[i+1] = j;
    }

    for (int i = n; i >= 1; i--)
    {
        sub[i]++;
        sub[f[i]] += sub[i];
    }

    vector <int> l;
    j = n;
    while (f[j] > 0)
    {
        l.push_back(f[j]);
        j = f[j];
    }
    sort(l.begin(), l.end());
    cout << l.size()+1 << endl;
    for (int i = 0; i < l.size(); i++) 
        cout << l[i] << ' ' << sub[l[i]] << endl;
    cout << n << ' ' << 1 << endl;

    return 0;
}
