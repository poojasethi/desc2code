#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{	
	int a=-1,b=0,i,j,k;
	char c[100010];
	int next[100010];
	memset(c,'\n',sizeof(c));
	scanf("%s",c);
	next[0]=-1;
	int len=strlen(c);
	while(b<len)
	{
		if(a==-1||c[a]==c[b])
		{
			a++;
			b++;
			next[b]=a;
		}
		else
			a=next[a];
	}
	int st=next[len];
	int re[100010],co=0;
	while(st!=-1&&st!=0)
	{
		re[co++]=st;
		st=next[st];
	}
	int dp[100010];
	memset(dp,0,sizeof(dp));
	cout<<co+1<<endl;
	for(i=len;i;i--)
	{
		dp[i]++;
		if(next[i]!=-1)
		dp[next[i]]+=dp[i];
	}
	for(i=co-1;i>=0;i--)
	{
		printf("%d %d\n",re[i],dp[re[i]]);
	}
	cout<<len<<" "<<dp[len]<<endl;
}
