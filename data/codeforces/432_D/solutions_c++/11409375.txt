#include <iostream>
#include <vector>

#define MAXN 100001

using namespace std;
int A[MAXN];
int N[MAXN];
bool B[MAXN];
int bordes[MAXN];

int main(){
  string st;//(MAXN,'A');
  cin >> st;
  int n = st.size();
  bordes[0] = 0;
  int i=1, j=0;
  while (i<n){
    while (j>0 && st[i] != st[j]){
      j=bordes[j-1];
    }
    if (st[i] == st[j])
      j++;
    bordes [i++] = j;
    A[j]+=1;
  }
  for (i=n-1;i>-1;i--){
    N[bordes[i]]+=N[i+1]+1;
  }
  B[n]=true;
  int aux=bordes[n-1];
  int c=1;
  while(aux!=0){
    c++;
    B[aux]=true;
    aux=bordes[aux-1];
  }
  cout << c << endl;
  for (i=1;i<=n;i++){
    if(B[i])
      cout << i << " " <<  N[i]+1 << endl;
  }
  return 0;
}
