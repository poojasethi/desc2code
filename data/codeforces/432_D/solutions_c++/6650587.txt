#include<stdio.h>
#include<string.h>
int next[300000];
char s[300000];
__int64 f[300000];
struct ans
{
	int x;
	__int64 y;
}ze[300000];
int main()
{
	
	gets(s);
	int n;
	n=strlen(s);
	int i;
	int j;
	next[0]=-1;
	for(i=0,j=-1;i<n;)
	{
		if(j==-1||s[i]==s[j])
		{
			j++;i++;
			next[i]=j;
		}
		else j=next[j];
	}
	for(i=0;i<=n;i++)
	f[i]=1;
	for(i=n;i>0;i--)
	{
		f[next[i]]+=f[i];
	}
	int cont=0;
	for(i=n;i>0;i=next[i])
	{
	ze[cont].x=i;
	ze[cont++].y=f[i];
	}
	printf("%d\n",cont);
	for(i=cont-1;i>=0;i--)
	printf("%d %I64d\n",ze[i].x,ze[i].y);
	
	
}
