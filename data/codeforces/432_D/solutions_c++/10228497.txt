#include<stdio.h>
#include<string.h>
#define N 100005

int p[N],d[N],cnt,l,c[N];
char s[N];

int main()
{
    int i,j;
    scanf("%s",s+1);
    l=strlen(s+1);
    j=0;
    p[0]=p[1]=0;
    for(i=2;i<=l;i++)
    {
        while(j && s[i]!=s[j+1]) j=p[j];
        if(s[i]==s[j+1])    j++;
        p[i]=j;
    }
    for(i=1;i<=l;i++)
        d[i]=1;
    for(i=l;i;i--)
        d[p[i]]+=d[i];
    cnt=0;
    j=p[l];
    while(j)
    {
        c[++cnt]=j;
        j=p[j];
    }
    printf("%d\n",cnt+1);
    for(i=cnt;i;i--)
        printf("%d %d\n",c[i],d[c[i]]);
    printf("%d %d\n",l,1);
    return 0;
}
