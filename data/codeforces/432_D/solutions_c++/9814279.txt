#include<bits/stdc++.h>
#define foreach(itit,gg) for(typeof(gg.begin())itit=gg.begin();itit!=gg.end();itit++)
std::string A;
int p[100005],n,dp[100005],g,k;
main(){
	std::cin >> A;
	std::vector<int> v;
	A = '#' + A;
	g = n = A.size()-1;
	for(int q=2;q<=n;q++){
		while(k>0 && A[k+1]!=A[q]) k=p[k];
		if(A[k+1]==A[q])k++;
		p[q]=k;
	}
	for(int i=1;i<=n;i++)dp[p[i]]++;
	for(int i=n;i>=1;i--)dp[p[i]]+=dp[i];
	while(g)
		v.push_back(g),g=p[g];
	reverse(v.begin(),v.end());
	printf("%d\n",v.size());
	foreach(it,v)
		printf("%d %d\n",*it,dp[*it]+1);
}
