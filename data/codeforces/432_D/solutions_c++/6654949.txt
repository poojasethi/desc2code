#include<cstdio>
#include<cstring>
#include<string>
#include<iostream>
#include<map>
#define rep(i,a,b) for (int i=a;i<=b;++i)
#define dep(i,a,b) for (int i=a;i>=b;--i)
#define M 100005
using namespace std;
int f[M],l,tot,ans[M];string a;
bool v[M];
void get(string p) {
    int n=p.size();memset(f,0,sizeof(f));
    rep(i,1,n-1){int j=f[i];while(j&&p[j]!=p[i])j=f[j];f[i+1]=p[j]==p[i]?j+1:0;}
}
int main() {
    cin>>a;get(a);l=a.size();
    for(int p=l;p;p=f[p])++tot,v[p]=1;
    for(int i=1;i<=l;i++)ans[i]=1;
    for(int i=l;i;i--)ans[f[i]]+=ans[i];
    printf("%d\n",tot);
    for(int i=1;i<=l;i++)if(v[i])printf("%d %d\n",i,ans[i]);
    return 0;
}
