#include<stdio.h>
#include<string>
#include<cstring>
using namespace std;
int pre[102345],cnt[102345];
char s[102345];
int ans[102345];
int main(){
	scanf("%s",s);
	int len=strlen(s);
	memset(pre,-1,sizeof(pre));
	cnt[0]=1;
	for(int i=1;i<len;i++){
		cnt[i]=1;
		for(int tmp=pre[i-1];;tmp=pre[tmp]){
			if(s[tmp+1]==s[i]){
				pre[i]=tmp+1;
				break;
			}
			else if(tmp==-1){
					pre[i]=-1;
					break;
				}
		}
	}
	for(int i=len-1;i>-1;i--)
		if(pre[i]!=-1)
			cnt[pre[i]]+=cnt[i];
	int tmp=len-1,ind=0;
	while(tmp!=-1){
		ans[++ind]=tmp;
		tmp=pre[tmp];
	}
	printf("%d\n",ind);
	for(int i=ind;i>0;i--)
		printf("%d %d\n",ans[i]+1,cnt[ans[i]]);
	return 0;
}
