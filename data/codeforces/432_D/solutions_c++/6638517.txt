#include <iostream>
#include <string>

using namespace std;

int z[100100];
int v[100100];

int main () {
	int n,m;
	string s;
	cin >> s;

	n = s.size();
	z[0] = n;

	for(int l=0,r=0,i=1;i<n;i++){
		if ( i <= r )
			z[i] = min(z[i-l], r-i+1);
		while( i+z[i] < n && s[z[i]] == s[i+z[i]] ) 
			z[i]++;
		if ( i+z[i]-1 > r )
			l = i, r = z[i]+i-1;
	}

	for(int i=0;i<n;i++){ v[z[i]]++;  }
	int cnt=0,ans=0;

	for(int i=n;i>=1;i--){
		if(v[i] && z[n-i] == i ){
			ans++;
		}
		v[i]+=v[i+1];
	}
	cout << ans << endl;
	for(int i=1;i<=n;i++){
		if(v[i] && z[n-i] == i )cout << i << ' ' << v[i] << endl;
	}

	return 0;
}
