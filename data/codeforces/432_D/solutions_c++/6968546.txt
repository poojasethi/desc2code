#include<cstdio>
#include<cstring>
#define N 100005
char s[N];
int dp[N],mk[N],nxt[N];
void getnxt(){
    int i=0,j=-1;
    nxt[i]=j;
    while(s[i]){
        while(j>=0 && s[i]!=s[j])j=nxt[j];
        nxt[++i]=++j;
    }
    //printf("s:%s\n",s);
    //for(int j=0;j<=i;++j)printf("j:%d nxt:%d\n",j,nxt[j]);
}
int main(){
    scanf("%s",s);
    getnxt();
    int sl=strlen(s);
    int now=sl;
    int cnt=0;
    while(now){
        ++cnt;
        ++mk[now];
        now=nxt[now];
    }
    for(int i=sl;i;--i)++dp[i],dp[nxt[i]]+=dp[i];
    printf("%d\n",cnt);
    for(int i=1;i<=sl;++i)if(mk[i]){
        printf("%d %d\n",i,dp[i]);
    }
    return 0;
}
