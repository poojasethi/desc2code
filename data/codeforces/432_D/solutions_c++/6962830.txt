#include<bits/stdc++.h>
using namespace std;

string ss;
int pi[100005],cnt[100005];
int n;

void prefixFunc()
{
    int i,k=0;
    pi[1]=0;
    for(i=1;i<n;i++)
    {
        while(k && ss[k]!=ss[i]) k=pi[k];
        if(ss[k]==ss[i]) k++;
        pi[i+1]=k;
    }
}

int main()
{
    int i,k;
    cin>>ss;
    n=ss.size();
    prefixFunc();
    for(i=1;i<=n;i++) cnt[i]=1;
    for(i=n;i>=1;i--) cnt[pi[i]]+=cnt[i];
    vector<pair<int,int> > out;
    for(k=n;k;k=pi[k]) out.push_back(make_pair(k,cnt[k]));
    cout<<out.size()<<endl;
    for(i=out.size()-1;i>=0;i--) cout<<out[i].first<<" "<<out[i].second<<endl;
    return 0;
}
