#include <iostream>
#include <cstdio>
#include <vector>
#include <map>
#include <set>
using namespace std;
#define pb push_back
int main()
{
 freopen("input.txt","r",stdin);
 freopen("output.txt","w",stdout);
 string s,t;
 int l,i,f,s1[500]={0},t1[500]={0},o=0;
 getline(cin,s);
 getline(cin,t);
 l=s.length();
 for(i=0;i<l;i++)
 {
  s1[s[i]]++;
  t1[t[i]]++;
 }
 f=0;
 for(i=0;i<l;i++)
 { 
  if(s1[s[i]]>t1[s[i]])
  {
   while(s1[f]>=t1[f]){f++;}
   if(s[i]>f||t1[s[i]]==0)
   {
    s1[s[i]]--;
    s1[f]++;
    s[i]=f;
    o++;
   }else
   {
    s1[s[i]]--;
    t1[s[i]]--;
   }
  }
 }
 cout<<o<<endl;
 cout<<s;
}
