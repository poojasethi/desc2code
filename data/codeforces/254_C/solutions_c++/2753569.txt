#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
string s,t;
int a[30],b[30];
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    cin>>s>>t;
    int len=s.size();
    for(int i=0;i<len;i++)
    a[s[i]-'A']=++b[s[i]-'A'];
    for(int i=0;i<len;i++)
    b[t[i]-'A']--;
    int ans=0,j=0;
    for(int i=0;i<len;i++)
    {
        int k=s[i]-'A';
        while(j<26&&b[j]>=0)j++;
        if(b[k]>0&&(k>j||a[k]<=b[k]))ans++,s[i]='A'+j,b[k]--,b[j]++;
        a[k]--;
    }
    cout<<ans<<endl<<s<<endl;
}
