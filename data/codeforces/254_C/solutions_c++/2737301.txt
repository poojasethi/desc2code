#include <cstdio>
#include <algorithm>
#include <cstring>
#define MAXN 100010
using namespace std;
char ar1[MAXN],ar2[MAXN];
int arr[MAXN][128];
int res;
int K[128];
int n;
int main()
{
	int res=0;
	int i,j;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf(" %s",ar1);
	scanf(" %s",ar2);
	n=strlen(ar1);
	for(i=0;i<n;i++)
	{
		K[ar1[i]]--;
		K[ar2[i]]++;
	}
	for(i='A';i<='Z';i++)
		for(j=n-1;j>=0;j--)
			arr[j][i]=arr[j+1][i]+(i==ar1[j]);
	for(i=0;i<n;i++)
	{
		if(K[ar1[i]] < 0)
		{
			for(j='A';j<='Z';j++)
			{
				if(K[j] > 0)
				{
					if(j > ar1[i] && arr[i+1][ar1[i]] >= -K[ar1[i]])
					{
						break;
					}
					res++;
					K[ar1[i]]++;
					K[j]--;
					ar1[i]=j;
					break;
				}
			}
		}
	}
	printf("%d\n",res);
	printf("%s\n",ar1);
	return 0;
}
