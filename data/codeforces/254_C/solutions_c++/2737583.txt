#include<iostream>
#include<fstream>
#include<map>
using namespace std;
map<char,int> v,w;
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
string s,t,e;
int i,j,k=0;
char l;
cin>>s>>t;
for(i=0;i<t.length();i++) e+=' ';
for(i=0;i<t.length();i++) v[t[i]]++;
for(i=0;i<s.length();i++) w[s[i]]++;
for(l='A';l<='Z';l++)
{
for(i=0;i<s.length();i++)
{
if(e[i]==' ' && s[i]==l && v[l]>0)
{
v[l]--;
e[i]=l;
}
}
for(i=0;i<s.length();i++)
{
if(e[i]==' ' && w[s[i]]>v[s[i]] && v[l]>0)
{
k++;
w[s[i]]--;
w[l]++;
v[l]--;
e[i]=l;
}
}
}
cout<<k<<endl<<e;
return 0;
}
