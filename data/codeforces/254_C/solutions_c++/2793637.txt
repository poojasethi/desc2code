#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
 string s,t;
 int a[126]={0},b[126]={0};
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout); 
 cin>>s>>t;
 for (int i=0;i<s.size();i++)
 {
  a[s[i]]++; b[t[i]]++;
 }
int ans=0;
 for (int i=0;i<s.size();i++)
  {
    int c=s[i];
    if (a[c]>b[c])
     {
      int j=1;
      for (; j<=125;j++)  if (b[j]>a[j]) break;
       if (b[c]==0||j<c)
        {s[i]=char(j); b[j]--; ans++;}
        else b[c]--;
       a[c]--;
     }
  }

cout<<ans<<endl;
cout<<s;


return 0;
}
