#include<cstdio>
#include<cstring>
#include<algorithm>
#define N 100010
using namespace std;
char s[N], t[N], x[N];
int ans, cnt[26], alps[26], alpt[26];
int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%s%s", s, t);
	for(int i = 0; s[i]; i++) alps[s[i]-'A']++, cnt[s[i]-'A']++;
	for(int i = 0; t[i]; i++) alpt[t[i]-'A']++;
	for(int i = 0; s[i]; i++){
		int val = s[i]-'A';
		if(alps[val] > alpt[val]){
			int chg = -1;
			for(int j = 0; j < 26; j++)
				if(alps[j] < alpt[j]){
					chg = j; break;
				}
			if((val < chg && cnt[val] == alps[val]-alpt[val]) || val > chg){
				alps[val]--, alps[chg]++;
				x[i] = chg+'A', ans++;
			}
		}
		if(!x[i]) x[i] = val+'A';
		cnt[val]--;
	}
	printf("%d\n%s\n", ans, x);
}
