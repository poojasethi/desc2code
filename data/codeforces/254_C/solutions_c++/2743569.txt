#include <iostream>
#include <algorithm>
#include <string>
#include <cstdio>
using namespace std;
string s,t;
int a[26],b[26];
int n;
int main(){

    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    cin>>s>>t;
    n=s.size();
    for(int i=0;i<n;i++) a[s[i]-'A']=++b[s[i]-'A'];
    for(int i=0;i<n;i++) b[t[i]-'A']--;
    int j=0;
    int num=0;
    for(int i=0;i<n;i++){
       int k=s[i]-'A';
       while(j<26&&b[j]>=0) j++;
       if(b[k]>0&&(k>j||a[k]<=b[k])) num++,s[i]='A'+j,b[k]--,b[j]++;
       a[k]--;
    }
    cout<<num<<endl<<s<<endl;
}
