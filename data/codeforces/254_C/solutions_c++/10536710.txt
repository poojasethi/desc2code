#include <iostream>
#include <fstream>
#include <vector>
#include <algorithm>
using namespace std;
ifstream fin("input.txt");
ofstream fout("output.txt");
int cnt[300],cnta[300];
int main(){
	string a,b;
	fin >> a >> b;
	for(int i = 0; i < a.size(); ++i){
		cnta[a[i]]++;
		cnt[a[i]]++;
		cnt[b[i]]--;
	}
	int ans = 0;
	for(int i = 0; i < a.size(); ++i){
		cnta[a[i]]--;
		for(char j = 'A'; j <= 'Z'; ++j){
			if(cnt[a[i]]>0&&cnt[j]<0){
				//cout << a[i] << ' ' << j << '\n';
				//cout << cnta[a[i]] << ' ' << cnt[a[i]] << '\n';
				if(j<a[i]||cnta[a[i]]+1==cnt[a[i]]){
					ans++;
					cnt[a[i]]--;
					cnt[j]++;
					a[i]=j;
					break;
				}
			}
		}
		//cout << "-----------------------\n";
	}
	fout << ans << '\n';
	fout << a << '\n';
}
