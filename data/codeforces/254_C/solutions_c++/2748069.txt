#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int v1[300],v2[300],i,l,k;
char c,ch,s[100005],t[100005];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	gets(s);
	gets(t);
	l=strlen(s);
	for(i=0;i<l;++i)
		v1[s[i]]++,v2[t[i]]++;
	for(ch='A';ch<='Z';++ch)
		k+=(abs(double(v1[ch]-v2[ch])));
	printf("%d\n",k/2);
	for(i=0;i<l;i++)
	{
		ch=s[i];
		if(v1[ch]>v2[ch])
		{
			for(c='A';c<='Z';++c)
			{
				if(v1[c]<v2[c])
				{
					if(ch>c||!v2[ch])
						s[i]=c,v2[c]--;
					else
						v2[ch]--;
					break;
				}
			  }
		v1[ch]--;
    }
  }
  printf("%s\n",s);
  return 0;
}
