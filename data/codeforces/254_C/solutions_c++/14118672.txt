#include<cstdio>
#include<cstring>
#define NN 100001

char s[NN],t[NN];
int v[26] = {0}, c[26]={0};

int main(){
	freopen("input.txt","r",stdin);
  	freopen("output.txt","w",stdout);
	scanf("%s%s",s,t);
	int n = strlen(s);
	for(int i=0;i<n;++i){
		s[i]-='A' , t[i]-='A';
		++v[s[i]],--v[t[i]];
	}

	for( int i =n-1; i>=0;--i)++c[s[i]];
	int op = 0 ; char nw;	
	for(int i=0;i<n;++i){
		nw = s[i];
		if( v[s[i]] > 0 ){
			int j;
			for(j=0;j<26 && v[j]>=0;++j);
			if( j < s[i] || c[s[i]] == v[s[i]] ){
				++op;
				++v[j]; --v[s[i]];
				nw = j;
			}
		}
		--c[s[i]]; s[i] =nw;
	}
	for(int  i=0;i<n;++i) s[i]+='A';
	printf("%d\n%s\n",op,s);
	return 0;
}
