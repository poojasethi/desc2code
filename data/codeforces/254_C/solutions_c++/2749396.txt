#include<cstdio>
#include<string>
#include<cstring>
#include<iostream>
using namespace std;
string s,t;
int a[30],b[30];
int cal(){
	for(int j=0;j<26;j++){
		if(a[j]<b[j])return j;
	}
	return -1;
}
int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	while(cin>>s>>t){
		memset(a,0,sizeof(a));
		memset(b,0,sizeof(b));
		int len=s.size();
		for(int i=0;i<len;i++){
			a[s[i]-'A']++;
			b[t[i]-'A']++;
		}
		int ans=0;
		for(int i=0;i<len;i++){
			int c=s[i]-'A';
			if(a[c]>b[c]){
				int pos=cal();
				if(pos<c||b[c]==0){
					s[i]=pos+'A';
					++ans;
					b[pos]--;
				}
				else b[c]--;
				a[c]--;
			}
		}
		cout<<ans<<endl;
		cout<<s<<endl;
	}
	return 0;
}
