#include<cstdio>
#include<cstring>
#include<algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
const int maxn=100000+10;
char ss[maxn],tt[maxn];
int s[maxn],t[maxn];
int n,cns[30],cnt[30];

int main(){
  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
  gets(ss); gets(tt); n=strlen(ss);
  rep(i,n) s[i]=ss[i]-'A', t[i]=tt[i]-'A';
  rep(i,n) cns[s[i]]++, cnt[t[i]]++;
  int j,ans=0; 
  rep(i,n){
    if(cns[s[i]]<=cnt[s[i]]) continue;
    for(j=0;j<26;j++) if(cns[j]<cnt[j]) break;
    if(s[i]>j){
      cns[s[i]]--; cns[j]++;
      ss[i]='A'+j; ans++;
    } else if(cnt[s[i]]) cnt[s[i]]--, cns[s[i]]--;
    else {
      cns[s[i]]--; cns[j]++;
      ss[i]='A'+j; ans++;
    }
  }
  printf("%d\n",ans);puts(ss);
}
