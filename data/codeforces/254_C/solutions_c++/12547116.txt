#include<bits/stdc++.h>
using namespace std;
string s,t;
int ocS['Z'+1],ocT['Z'+1],ans;
ifstream fin ("input.txt");
ofstream fout ("output.txt");
int main()
{
    fin>>s>>t;
	for(int i=0;s[i];i++)
		ocS[s[i]]++,ocT[t[i]]++;
	for(int i=0;s[i];i++)
	{
		int ch=s[i];
		if(ocS[ch]>ocT[ch])
		{
			for(int j='A';j<='Z';j++)
				if(ocS[j]<ocT[j])
				{
					if(j<ch||!ocT[ch])
						s[i]=j,ocT[j]--,ans++;
					else
						ocT[ch]--;
					break;
				}
			ocS[ch]--;
		}
	}
	fout<<ans<<endl<<s<<endl;
	return 0;
}
