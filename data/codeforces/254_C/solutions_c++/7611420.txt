#include <iostream>
#include<stdio.h>
#include<stdlib.h>

#define ll long long int
using namespace std;
int main() {
string s,t;
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
cin>>s>>t;
ll len=s.length();
ll sfreq[30]={0};
ll tfreq[30]={0};
for(int i=0;i<len;i++){
	sfreq[s[i]-'A']++;
	tfreq[t[i]-'A']++;
}
ll ans=0;
ll c;
for(int i=0;i<len;i++){
	c=s[i]-'A';
	if(sfreq[c]>tfreq[c]){
     for(int j=0;j<26;j++){
     	if(sfreq[j]<tfreq[j]){
     	if(j<c||!tfreq[c]){
     		s[i]=j+'A';
     		tfreq[j]--;
     		ans++;
     	}
     	else{
     		tfreq[c]--;
     	}
     	break;	
     	}
     }
     sfreq[c]--;
	}
}
cout<<ans<<"\n"<<s;
	return 0;
}
