#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define M 100010
using namespace  std;
char a[M],b[M];
int len,ans,cnta[27],cntb[27];

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%s%s",a,b);
	len=strlen(a);
	for(int i=0;i<len;i++) cnta[a[i]-'A']++;
	for(int i=0;i<len;i++) cntb[b[i]-'A']++;
	for(int x,y,i=0;i<len;i++)
	{
		x=a[i]-'A';
		if(cnta[x]>cntb[x])
		{
			for(int j=0;j<26;j++)
			{
				if(cnta[j]<cntb[j])
				{
					if( j<x || ( j>x &&  cntb[x]==0 ) )
					{
						cnta[j]++;
						cnta[x]--;
						ans++;
						a[i]=j+'A';
						break;
					}
				}
			}
		}
		x=a[i]-'A';
		cnta[x]--;
		cntb[x]--;
	}
	cout<<ans<<endl;
	puts(a);
	return 0;
}
