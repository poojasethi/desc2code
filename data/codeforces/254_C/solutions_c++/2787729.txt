#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
//string s1,s2;
char s1[100001],s2[100001];
int a[26]={0},b[26]={0};
int main()
{
	int charval,i,j=0,count=0,length;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%s%s",s1,s2);
//	cin >> s1 >> s2;
	length=strlen(s1);
	for(i=0;i<length;i++)
	{
		a[s1[i]-'A']++;
		b[s1[i]-'A']++;
	}
	for(i=0;i<length;i++)
		b[s2[i]-'A']--;
	for(i=0;i<length;i++)
	{
		charval=s1[i]-'A';
		for(j;j<26;j++)
		{
			//printf("j=%d\n",j);
			if(b[j]<0)
				break;
		}
		if(b[charval]>0 && (charval>j || a[charval]<=b[charval]))
		{
			b[charval]--;
			s1[i]='A'+j;
			b[j]++;
			count++;
			
		}
		a[charval]--;

	}

	printf("%d\n%s\n",count,s1);
	//cout << count << endl << s1;

	
	return 0;
}
