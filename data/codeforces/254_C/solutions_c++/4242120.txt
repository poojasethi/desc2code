#include <cstdio>
#include <stack>
using namespace std;

char s[100001], t[100001];
int cnt[128], dnt[128], n;
stack<char> S;

int main() {
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  scanf("%s%s", s, t);
  for (int i(0); s[i]; ++i) {
    ++dnt[s[i]];
    --cnt[s[i]];
    ++cnt[t[i]];
  }
  for (char i('Z'); i >= 'A'; --i)
    for (int j(0); j < cnt[i]; ++j)
      S.push(i);
  n = S.size();
  for (int i(0); !S.empty(); ++i) {
    --dnt[s[i]];
    if (cnt[s[i]] < 0)
      if (S.top() < s[i] || dnt[s[i]] < -cnt[s[i]]) {
        ++cnt[s[i]];
        s[i] = S.top();
        S.pop();
      }
  }
  printf("%d\n%s\n", n, s);
  fclose(stdin);
  fclose(stdout);
}
