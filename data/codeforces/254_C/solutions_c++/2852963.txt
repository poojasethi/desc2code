#include <iostream>
#include <string>
#include <cstdio>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,p,a[30],f[30],diff[30],cnt;
string s,t,r;

int main(){

  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);

  cin>>s>>t;
  n = s.size();
  r = s;

  rep(i,n){
    f[s[i]-'A']++;
    a[t[i]-'A']++;
  }
  rep(i,26){diff[i] = f[i]-a[i];cnt+=diff[i]>0?diff[i]:0;}
  rep(i,n){
    int c = s[i]-'A';
    if(diff[c]<=0){f[c]--;continue;}
    else {
      while(diff[p]>=0)p++;
      if(p<c || f[c]<=diff[c]){
    r[i] = p+'A';
    diff[p]++;diff[c]--;
      }
      f[c]--;
    }
  }
  cout<<cnt<<endl;
  cout<<r<<endl;
}
