#include <cstdio>
#define FOR(i,a,b) for (int i=a;i<=b;++i)
#define FORSZ(i,s) for(int i=0;s[i];++i)
int n,a[255],b[255],r;
char s[100010],t[100010];
int main()
{
    fscanf(fopen("input.txt", "r"), "%s%s",s,t);
    FORSZ(i, s)
        a[s[i]]++;
    FORSZ(i, t)
        b[t[i]]++;
    FOR(i,'A','Z')
        r+=a[i]>b[i]?a[i]-b[i]:b[i]-a[i];
    FORSZ(i,s)
    if (a[s[i]]>b[s[i]])
    {
        FOR(p,'A','Z')
        if (p==s[i]&&a[p]>0&&b[p]>0)
        {
            --a[p];--b[p];
            break;
        }
        else if (b[p]>a[p])
        {
            --a[s[i]];++a[p];s[i]=p;
            break;
        }
    }
    fprintf(fopen("output.txt","w"),"%d\n%s",r/2,s);
}
