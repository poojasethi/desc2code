#include<fstream>
#include<string>

using namespace std;

int main(){
  ifstream fin("input.txt");
  ofstream fout("output.txt");
  string s,t;
  fin>>s>>t;
  int sc[128]={},tc[128]={},d[128]={};
  for(int i=0;i<s.size();i++){
    d[s[i]]++;
    d[t[i]]--;
    sc[s[i]]++;
  }
  int r=0;
  for(int i=0;i<s.size();i++){
    if(d[s[i]]>0){
      for(int j=0;;j++){
	if(d[j]<0){
	  if(j<s[i]||sc[s[i]]==d[s[i]]){
	    d[s[i]]--;
	    sc[s[i]]--;
	    s[i]=j;
	    sc[j]++;
	    d[j]++;
	    r++;
	  }
	  break;
	}
      }
    }
    sc[s[i]]--;
  }
  fout<<r<<endl<<s<<endl;
  return 0;
}
