#include <cstdio>
#define maxn 100010
char s[maxn], t[maxn];
int  c1[26], c2[26];
main( ) {
	freopen( "input.txt", "r", stdin);
	freopen( "output.txt", "w", stdout);
	int z = 0;
	scanf("%s%s", s, t);
	for (int i = 0; s[i]; i++) {
		c1[ s[i] - 'A' ]++;
		c2[ t[i] - 'A' ]++;
	}
	for (int i = 0; s[i]; i++) {
		int cc = s[i] - 'A';
		if (c1[cc] > c2[cc]) {
			for (int j = 0; j < 26; j++){
				if (c1[j] < c2[j]) {
					if (j < cc || c2[cc] == 0) {
						s[i] = j + 'A';
						c2[j]--;
						z++;
					} else
						c2[cc]--;
					break;
				}
			}
			c1[cc]--;
		}
	}
	printf("%d\n%s\n",z, s);
	return 0;
}
