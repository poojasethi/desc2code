#include <iostream>
#include <stdio.h>
#include <cstring>

using namespace std;

const int N=100000+10,Z=26;

int n;
int sa[N],sb[N];
char a[N],b[N],s[N];

void Init(){
	int i;
	cin >> a;
	cin >> b;
	n=strlen(a);
	for(i=0;i!=Z;i++) sa[i]=sb[i]=0;
	for(i=0;i!=n;i++) sa[a[i]-'A']++,sb[b[i]-'A']++;
}

void Solve(){
	int i,j,ans;
	for(ans=i=0;i!=n;i++){
		int k=a[i]-'A';
		for(j=0;j!=Z;j++){
			if (k==j){
				if (sb[k]>0){
					sb[k]--;
					sa[k]--;
					s[i]=a[i];
					break;
				}
			}else if (sa[k]>sb[k]&&sb[j]>sa[j]){
				ans++;
				s[i]=char(j+'A');
				sa[k]--,sb[j]--;
				break;
			}
		}
	}
	s[n]='\0';
	cout << ans << endl << s << endl;
}

int main(){
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	Init();
	Init();
	Solve();
	return 0;
}
