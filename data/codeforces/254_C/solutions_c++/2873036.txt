#include<stdio.h>
#include<algorithm>

using std::min;

const int MAXL=100000+9;

char s[MAXL],t[MAXL];
int cnt1[26],cnt2[26],cnt3[26];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%s%s",s,t);
	char *p;
	for(p=t;*p;++p)
		++cnt2[*p-'A'];
	for(p=s;*p;++p)
		++cnt1[*p-'A'];
	int i;
	for(i=0;i<26;++i)
		cnt3[i]=cnt1[i]-cnt2[i];
	int ans=0;
	for(i=0;i<26;++i)
		if(cnt3[i]>0)
			ans+=cnt3[i];
	printf("%d\n",ans);
	int v;
	for(p=s;*p;++p)
	{
		if(cnt3[v=*p-'A']>0)
		{
			int lim=cnt1[v]>cnt3[v]?v:26;
			for(i=0;i<lim;++i)
				if(cnt3[i]<0)
				{
					--cnt3[v],++cnt3[i];
					*p=i+'A';
					break;
				}
		}
		--cnt1[v];
	}
	puts(s);
	return 0;
}
