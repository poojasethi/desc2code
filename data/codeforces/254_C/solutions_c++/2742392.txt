#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char s[100010],t[100010];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	while(scanf("%s%s",s,t)!=EOF)
	{
		int str1[256]={0},str2[256]={0};
		int i,j,k=0;
		int len=strlen(s);
		for(i=0;i<len;i++)
		{
			str1[s[i]]++;
			str2[t[i]]++;
		}
		int ans=0;
		int t='A';
		for(i=0;i<len;i++)
		{
			if(str1[s[i]]>str2[s[i]])
			{
				while(str1[t]>=str2[t]) t++;				
				if(s[i]>t||str2[s[i]]==0)
				{
					str1[s[i]]--;
					str1[t]++;
					s[i]=t;
					ans++;
				}
				else
				{
					str1[s[i]]--;
					str2[s[i]]--;
				}
			}
		}
		printf("%d\n",ans);
		puts(s);
	}
}
