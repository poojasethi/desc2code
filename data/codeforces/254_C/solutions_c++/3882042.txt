#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include<string.h>
#define asc (int)'A'
#define N 210000
#define foru(i,a,b) for(int i=a;i<=b;++i)
#define ford(i,a,b) for(int i=a;i>=b;--i)

using namespace std;
int n,l,x[31],y[31],ans;
char s[N],t[N];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%s",&s); scanf("%s",&t);
	n=strlen(s)-1;
	foru(i,0,n) ++x[(int)s[i]-asc],--x[(int)t[i]-asc],++y[(int)s[i]-asc];
	l=0;
	foru(i,0,n)
	  if(x[(int)s[i]-asc]>0)
	  {
	  	  --y[(int)s[i]-asc];
	  	  while(x[l]>=0)++l;
	  	  if((int)s[i]-asc<l&&y[(int)s[i]-asc]>=x[(int)s[i]-asc]) continue;
		  --x[(int)s[i]-asc]; ++x[l]; s[i]=(char)(l+asc); ++ans;	  
	  }
	printf("%d\n%s\n",ans,s);
	return 0;
}
