import sys
from collections import defaultdict
def RedirectIO():
    sys.stdin=open("input.txt","r")
    sys.stdout=open("output.txt","w")
def main():
    RedirectIO()
    s=raw_input()
    t=raw_input()
    sRemain=defaultdict(int)
    tRemain=defaultdict(int)
    for ch in s:
        sRemain[ch]+=1
    for ch in t:
        tRemain[ch]+=1
    
    str=[key for key,v in tRemain.items() if v>sRemain[key] for i in xrange(v-sRemain[key])]
    str.sort(reverse=True)
    
    print len(str)
    for ch in s:
        if sRemain[ch]>tRemain[ch] and (str[-1]<ch or tRemain[ch]==0):
            sRemain[ch]-=1
            tRemain[str[-1]]-=1
            sys.stdout.write(str.pop())
        else:
            sRemain[ch]-=1
            tRemain[ch]-=1
            sys.stdout.write(ch)
        sys.stdout.flush()
main()
