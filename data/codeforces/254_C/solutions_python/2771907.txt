import sys
sys.stdin = open('input.txt')
sys.stdout = open('output.txt', 'w')
s = map(ord,raw_input())
t = map(ord,raw_input())
c = [0] * 128
d = [0] * 128
for i,j in zip(s,t):
    c[j] += 1
    c[i] -= 1
    d[i] += 1
p = [chr(i) for i in reversed(range(len(c))) for j in range(c[i])]
a = []
print(len(p))
for i in s:
    d[i] -= 1
    a.append(chr(i))
    if c[i] < 0 and (p[-1] < chr(i) or d[i] + c[i] < 0):
        a[-1] = p.pop()
        c[i] += 1
print(''.join(a))
