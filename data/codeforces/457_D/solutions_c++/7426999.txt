using namespace std;

#define REP(i,n) for((i)=0;(i)<(int)(n);(i)++)
#define snuke(c,itr) for(__typeof((c).begin()) itr=(c).begin();itr!=(c).end();itr++)

double f[100010];
double C[310][310];

int main(void){
    int N,M,K,i,j;
    
    cin >> N >> M >> K;
    
    f[0] = 1.0;
    REP(i,K) f[i+1] = f[i] * (K-i) / (M-i);
    
    REP(i,N+1) REP(j,i+1){
        if(j == 0 || j == i) C[i][j] = 1;
        else C[i][j] = C[i-1][j-1] + C[i-1][j];
    }
    
    double INF = 1.0E+99;
    double ans = 0.0;
    REP(i,N+1) REP(j,N+1) ans = min(INF, ans + f[i*N+j*N-i*j] * C[N][i] * C[N][j]);
    printf("%.12f\n", ans);
    // cout << ans << endl;
    
    return 0;
}
