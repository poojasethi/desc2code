#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

double C[310][310];
double f[310];

int main() {
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    C[0][0] = 1;
    for (int i = 1; i <= n; i++) {
        C[i][0] = 1;
        for (int j = 1; j <= n; j++)
            C[i][j] = C[i - 1][j - 1] + C[i - 1][j];
    }
    f[0] = 1;
    for (int i = 0; i < n * n; i++)
        f[i + 1] = f[i] * (k - i) / (m - i);
    double ans = 0.0;
    for (int i = 0; i <= n; i++)
        for (int j = 0; j <= n; j++)
            if ((ans += C[n][i] * C[n][j] * f[n * (i + j) - i * j]) > 1e99) {
                printf("1e99\n");
                return 0;
            }
    printf("%.10lf\n",ans);
    return 0;
}
