#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#include<cmath>
#define MAXN 100005
using namespace std ;
typedef long long LL ;
typedef long double ld ;

int N, M, K, t ;
ld pw[MAXN], ans, get ;

int main()
{
	int i, j ;
	scanf("%d %d %d", &N, &M, &K) ;
	for(i = 1, pw[0] = 0; i <= M; i ++)
		pw[i] = pw[i-1]+log(i) ;
	for(i = 0; i <= N; i ++)
		for(j = 0; j <= N; j ++)
		{
			t = (i+j)*N-i*j ;
			if(K < t) continue ;
			get =  pw[N]*2-pw[i]-pw[j]-pw[N-i]-pw[N-j]+pw[M-t]+pw[K]-pw[K-t]-pw[M] ;
			ans = min((long double)1e99, ans+exp(get)) ;
		}
	printf("%.10lf\n", (double)ans) ;
	return 0 ;
}
