using namespace std;

const int N = 1e5 + 7;
const int INF = 0x3fffffff;
const int MOD = 1e9 + 7;
const DB  EPS = 1e-8;

DB tot, sum[N];

DB C(int x, int y){
    return sum[x] - sum[y] - sum[x - y];
}

int n, m, K;

int main(){
    for(int i = 1; i < N; i ++)
        sum[i] = sum[i - 1] + log10((i));
    cin >> n >> m >> K;
    for(int i = 0; i <= n; i ++){
        for(int j = 0; j <= n; j ++){
            int s = n * (i + j) - i * j;
            if(s > K) continue;
            DB res = C(n, i) + C(n, j) + C(m - s, K - s) - C(m, K);
            tot += pow(10, res);
            if(tot > 1e99){
                puts("1e99");
                return 0;
            }
        }
    }
    cout << setprecision(20) << tot << endl;
}
