#include <stdio.h>
long long sum=1;
int main()
{
int n,a,i,count=1;
scanf("%d",&n);
for(i=1;i<=n;i++)
{
	scanf("%d",&a);
	if(a)
		break;
}
if(i>n)
	sum=0;
for(i=i+1;i<=n;i++)
{
	scanf("%d",&a);
	if(a)
	{
		sum=sum*count;
		count=1;
	}
	else count++;
}
printf("%lld\n",sum);
return 0;
}
