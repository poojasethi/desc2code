n = int(raw_input())
arr = map(int, raw_input().split())

cnt = 0
last = 0
total = 1
for i in xrange(0, n):
    if(arr[i] == 1 and cnt == 0):
        cnt = 1
        last = i
    if(cnt == 1 and arr[i] == 1 and i != last):
        total *= i-last
        last = i
if(cnt == 1):
    print(total)
else:
    print(0)
