#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int N = 1000010;
const int mod = 1e9 + 7;

int n;
char str[N];
ll dp[N][5];

int main(){
	n = strlen(gets(str));
	if(str[0] == '*' or str[0] == '?') dp[0][0] = 1;
	if(str[0] == '0' or str[0] == '?') dp[0][1] = 1;
	if(str[0] == '1' or str[0] == '?') dp[0][3] = 1;
	for(int i = 1; i < n; ++i){
		if(str[i] == '*' or str[i] == '?') dp[i][0] = (dp[i - 1][0] + dp[i - 1][3] + dp[i - 1][4]) % mod;
		if(str[i] == '0' or str[i] == '?') dp[i][1] = (dp[i - 1][1] + dp[i - 1][2]) % mod;
		if(str[i] == '1' or str[i] == '?'){
			dp[i][2] = dp[i - 1][0];
			dp[i][3] = (dp[i - 1][2] + dp[i - 1][1]) % mod;
		}
		if(str[i] == '2' or str[i] == '?') dp[i][4] = dp[i - 1][0];
	}
	cout << (dp[n - 1][0] + dp[n - 1][1] + dp[n - 1][2]) % mod << endl;
	return 0;
}
