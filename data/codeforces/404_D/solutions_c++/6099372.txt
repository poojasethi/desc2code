#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
const int mod = 1000000007;
const int maxn = 1000005;
long long b[3] = {1,1,0};
char s[maxn];
int main(){
    scanf("%s",s);
    long long a[3];
    for(int i = 0;s[i] != '\0';i ++){
        a[0] = a[1] = a[2] = 0;
        if(s[i] == '0' || s[i] == '?'){
            a[0] += b[0];
        }
        if(s[i] == '1' || s[i] == '?'){
            a[0] += b[2];
            a[1] += b[0];
        }
        if(s[i] == '2' || s[i] == '?'){
            a[1] += b[2];
        }
        if(s[i] == '*' || s[i] == '?'){
            a[2] += b[1] + b[2];
        }
        for(int i = 0;i < 3;i ++){
            b[i] = a[i] % mod;
        }
    }
    printf("%I64d\n",(a[0] + a[2]) % mod);
    return 0;
}
