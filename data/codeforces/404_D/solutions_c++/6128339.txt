#include <iostream>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <utility>
#include <algorithm>
#include <cmath>
#define MAXN 1000010
#define MOD 1000000007

using namespace std;

string s;
int n;
long long zero[MAXN],one[2][MAXN],two[MAXN],mine[MAXN];

int main() {
	cin >> s;
	n = s.length();
	if (s[0] == '0' || s[0] == '?') zero[0] = 1;
	if (s[0] == '1' || s[0] == '?') one[1][0] = 1;
	if (s[0] == '*' || s[0] == '?') mine[0] = 1;
	for (int i = 1; i < n; i++) {
		if (s[i] == '0' || s[i] == '?')
			zero[i] = (zero[i-1] + one[0][i-1]) % MOD;
		if (s[i] == '2' || s[i] == '?')
			two[i] = mine[i-1];
		if (s[i] == '*' || s[i] == '?')
			mine[i] = (mine[i-1] + one[1][i-1] + two[i-1]) % MOD;
		if (s[i] == '1' || s[i] == '?') {
			one[0][i] = mine[i-1];
			one[1][i] = (zero[i-1] + one[0][i-1]) % MOD;
		}
	}
	cout << (zero[n-1] + one[0][n-1] + mine[n-1]) % MOD;
}
