#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
#define FOR(i,l,r) for(int i=l;i<=r;i++)
#define FILL(a,b) memset(a,b,sizeof(a))
#define MD 1000000007
#define N 1200001
int n,m,e,ans,f[N][5];
char s[N];
bool ok(int i,char ch){
	if(s[i] == ch || s[i] == '?')return true;
	return false;
}
int main(){
//	*1  4*
	scanf("%s",s);
	n=strlen(s)-1;
	if(s[0] == '*')f[0][3]=1;
	if(s[0] == '0')f[0][0]=1;
	if(s[0] == '1')f[0][4]=1;
	if(s[0] == '2')f[0][2]=0;
	if(s[0] == '?')f[0][3]=f[0][4]=f[0][0]=1;
	FOR(i,1,n){
		if(ok(i,'0'))					f[i][0]=(f[i-1][1]+f[i-1][0])%MD;
		if(ok(i,'1') && ok(i-1,'*'))	f[i][1]=f[i-1][3]%MD;
		if(ok(i,'2') && ok(i-1,'*'))	f[i][2]=f[i-1][3]%MD;
		if(ok(i,'*'))					f[i][3]=((f[i-1][4]+f[i-1][2])%MD+f[i-1][3])%MD;
		if(ok(i,'1') && s[i-1] != '*')	f[i][4]=(f[i-1][1]+f[i-1][0])%MD;
	}
	printf("%d\n",((f[n][0]+f[n][1])%MD+f[n][3])%MD);
	return 0;
}
