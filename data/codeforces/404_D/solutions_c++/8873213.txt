#include <iostream>

using namespace std;

const int MOD = 1000 * 1000 * 1000 + 7;
int n,dp[1000*1000+5][3];
string s;

int main(){
	cin >> s;
	n = s.length();
	dp[0][0] = 1;
	if(s[0] == '*' || s[0] == '?'){
		dp[0][1] = 1;
	}
	for(int i=1;i<=n;i++){
		if(s[i-1] == '*'){
			dp[i][2] += dp[i-1][1] + dp[i-1][2];
		}
		else if(s[i-1] == '0'){
			dp[i][0] += dp[i-1][0];
		}
		else if(s[i-1] == '1'){
			dp[i][1] += dp[i-1][0];
			dp[i][0] += dp[i-1][2];
		}
		else if(s[i-1] == '2'){
			dp[i][1] += dp[i-1][2];
		}
		else if(s[i-1] == '?'){
			dp[i][2] += dp[i-1][1] + dp[i-1][2];
			dp[i][0] += dp[i-1][0];
			dp[i][1] += dp[i-1][0];
			dp[i][0] += dp[i-1][2];
			dp[i][1] += dp[i-1][2];
		}
		dp[i][0] %= MOD;
		dp[i][1] %= MOD;
		dp[i][2] %= MOD;
	}
	cout << (dp[n][2] + dp[n][0])%MOD << endl;
	return 0;
}
