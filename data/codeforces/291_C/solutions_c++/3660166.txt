#include <stdio.h>
#include <set>
using namespace std;
unsigned int ip[100000];
int main()
{
	int n,k,x;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i){
		ip[i]=0;
		for(int j=0;j<4;++j){
			scanf("%d.",&x);
			ip[i]=(ip[i]<<8)|x;
		}
	}
	for(unsigned int i=1<<31,bits=31;bits>0;i|=1<<(--bits)){
		//starting with 2^31 and uptp (2^32)-1
		set<unsigned int> different;
		for(int j=0;j<n && different.size()<=k;++j)
			different.insert(ip[j]&i);
		if(different.size()==k){
			printf("%u.%u.%u.%u",(i>>24),(i>>16)&((1<<8)-1),(i>>8)&((1<<8)-1),i&((1<<8)-1));
			return 0;
		}
	}
	printf("-1");
	return 0;
}
