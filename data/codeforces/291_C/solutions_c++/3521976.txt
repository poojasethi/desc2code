#include <iostream>
#include <cstdio>
#include <set>
#define N 100000
using namespace std;

typedef unsigned int ui;

set<ui> nets;
ui ip[N];
int main()
{
    int n,k,i;
    scanf("%d%d",&n,&k);
    ui a,b,c,d;
    for(i=0;i<n;i++)
    {
        scanf("%d.%d.%d.%d",&a,&b,&c,&d);
        ip[i] = (a<<24) + (b<<16) + (c<<8) + d;
    }
    ui sub=0;

    for(ui start = (1<<31);start>0;start>>=1)
    {
        sub+=start;
        nets.clear();
        for(i=0;i<n;i++)
            nets.insert(ip[i]&sub);
        if(nets.size()==k)
        {
            printf("%u.%u.%u.%u",(sub>>24)&255,(sub>>16)&255,(sub>>8)&255,sub&255);
            return 0;
        }
    }
    printf("-1");
    return 0;
}
