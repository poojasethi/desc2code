#include <stdio.h>
#include <set>
using namespace std;
unsigned int v[100000];
int main()
{
	int n,k,x;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i){
		v[i]=0;
		for(int j=0;j<4;++j){
			scanf("%d.",&x);
			v[i]=(v[i]<<8)|x;
		}
	}
	for(unsigned int i=1<<31,j=31;j>0;i|=1<<(--j)){
		set<unsigned int> s;
		for(int j=0;j<n && s.size()<=k;++j)
			s.insert(v[j]&i);
		if(s.size()==k){
			printf("%u.%u.%u.%u",(i>>24),(i>>16)&((1<<8)-1),(i>>8)&((1<<8)-1),i&((1<<8)-1));
			return 0;
		}
	}
	printf("-1");
	return 0;
}
