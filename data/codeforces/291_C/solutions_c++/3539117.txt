#include <iostream>
#include <vector>
#include <string>
#include <set>
#include <cstdio>
using namespace std;

typedef long long int LL;

int main() {
  int n,k;
  cin >> n >> k;
  vector<LL> addr(n);
  for (int i=0;i<n;i++) {
    int a,b,c,d;
    scanf("%d.%d.%d.%d",&a,&b,&c,&d);
    addr[i] = a;
    addr[i]=addr[i]*256+b;
    addr[i]=addr[i]*256+c;
    addr[i]=addr[i]*256+d;
  }
  long long mask = 1LL<<31;
  for (int i = 30; i>0; --i) {
    set<long long> st;
    for (int j=0;j<n;j++) st.insert(mask & addr[j]);
    if (st.size() == k) {
      cout << mask/256/256/256 << '.' << mask%(256*256*256)/256/256 << '.' << mask%(256*256)/256 << '.' << mask%256;
      return 0;
    }
    mask += (1LL<<i);
  }
  cout << -1;
}
