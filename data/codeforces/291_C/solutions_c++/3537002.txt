#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define N 100010
typedef long long LL;
LL n,k,a[N],b[N],m,x,now;
LL calc(LL x)
{
    for (LL i=0;i<n;i++) b[i]=a[i]&x;
    sort(b,b+n);
    m=unique(b,b+n)-b;
    return m;
}

void output(LL x)
{
    printf("%I64d.%I64d.%I64d.%I64d\n",x/16777216,x%16777216/65536,x%65536/256,x%256);
}

int main()
{
    scanf("%I64d%I64d",&n,&k);
    for (LL i=0;i<n;i++) 
        for (LL j=0;j<4;j++)
        {
            scanf("%I64d",&x),getchar();
            a[i]<<=8;
            a[i]+=x;
        }
    for (LL j=31;j>0;j--)
    {
        now|=1LL<<j;
        if (calc(now)==k) return output(now),0;
    }
    puts("-1");
}
