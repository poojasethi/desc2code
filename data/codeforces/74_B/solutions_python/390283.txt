#!/usr/bin/env python
# coding=utf-8
from sys import stdin
import sys

if __name__ == "__main__":
    n,k,m = map(int, stdin.readline().split())
    dir = -1 if (stdin.readline().strip() == "to head") else +1
    stations = stdin.readline().strip()
    for i,c in enumerate(stations):
        #print(m,k)
        if c == "0":
            kp = k
            if m > k and k > 1:
                k -= 1
            elif m < k and k < n:
                k += 1
            if k==m:
                k = kp
        if m == 1:
            dir = 1
            m = 2
        elif m == n:
            dir = -1
            m = n-1
        else:
            m += dir
        if c == "1":
            if dir == -1:
                k = n
            else:
                k = 1
        if m == k:
            print("Controller {0}".format(i+1))
            sys.exit(0)
    print("Stowaway")
