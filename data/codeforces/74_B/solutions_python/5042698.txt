n,m,k=map(int,raw_input().split())
h=raw_input()[3]=='h'
s=raw_input()
j=0
for i in s:
    j+=1
    if i == '0':
        if m < k and m>1: m-=1
        if m > k and m<n: m+=1

    if k == 1: h=False
    if k == n: h=True 
    if h:
        k-=1
    else:
        k+=1
    if i == '1':
        m=k-(-1)**(h)
    elif m == k:
        print "Controller",j
        exit()
print "Stowaway"
