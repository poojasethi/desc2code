#include <stdio.h>
#include <string.h>
int n,m,k,flag;
int move(int k)
{
	if(k==n) flag=0;
	if(k==1) flag=1;
	if(flag) k++;
	else k--;
	return k;
}

int main()
{
	int i;
	char c1[10],c2[210];
	while(scanf("%d%d%d",&n,&m,&k)!=EOF)
	{
		scanf("%s%s",&c1,&c1);
		if(c1[0]=='h')  flag=0;
		else flag=1;
		scanf("%s",&c2);
		if(m>k)  m=n;
		else m=1;
		for(i=0;i<strlen(c2);i++)
		{
			if(c2[i]=='1')
			{
				k=move(k);
				if(flag)  m=1;
				else m=n;
			}
			else 
			{
				k=move(k);
				if(k==m)  break;
			}
		}
		if(i==strlen(c2)) printf("Stowaway\n");
		else printf("Controller %d\n",i+1);
	}
	return 0;
}
