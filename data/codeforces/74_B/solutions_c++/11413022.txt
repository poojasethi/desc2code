#include <iostream>
using namespace std;

int n, m, k;
string d;
string states;

int main() {
	cin >> n >> m >> k;
	cin >> d >> d;
	cin >> states;

	int cStep = -1;
	if (d == "tail") {
		cStep = 1;
	}

	int len = states.size();

	bool sWin = true;
	for (int i = 0; i < len; ++i) {
		k += cStep;
		if (k == 1 || k == n) {
			cStep = -cStep;
		}

		if (m != 1 && m != n) {
			m += cStep;
		}

		if (states[i] == '0') {
			if (m == k) {
				sWin = false;
				cout << "Controller "<< i+1 << endl;
				break;
			}
		}
		else {
			if (k == 1) {
				m = n;
			}
			else if (k == n) {
				m = 1;
			}
			else {
				m = k - cStep;
			}
		}
	}
	if (sWin) {
		cout << "Stowaway" << endl;
	}
	return 0;
}
