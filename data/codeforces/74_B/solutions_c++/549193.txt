#include <iostream>

using namespace std;

int main(int argc, char* argv[])
{
	int n, m, k;
	cin >> n >> m >> k;

	string s;
	cin >> s >> s;
	int d;

	if (s == "head")
		d = -1;
	else
		d = 1;

	string p;
	cin >> p;

	for (int i = 0; i < p.size(); i++)
	{
		if (p[i] == '0')
		{
			if ((m < k) && (m > 1))
				m--;
			if ((m > k) && (m < n))
				m++;

			if ((k + d < 1) || (k + d > n))
				d *= -1;

			k += d;

			if (m == k)
			{
				cout << "Controller " << (i + 1) << endl;
				return 0;
			}
		}
		else
		{

			if ((k + d < 1) || (k + d > n))
				d *= -1;

			k += d;
			m = (d == 1) ? 1 : n;
		}
	}

	cout << "Stowaway" << endl;

}
