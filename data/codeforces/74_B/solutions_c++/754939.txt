#include <stdio.h>

int main(void) {
	int n, m, k, dir;
	char d[10];
	scanf("%d%d%d", &n, &m, &k);
	scanf("%*s%s", d);
	if(d[0]=='h') dir = -1;
	else dir = 1;
	char moves[500];
	scanf("%s", moves);
	for(int i=0;moves[i+1];i++) {
		if(moves[i]=='0') {
			if(m+1==k && dir==-1) { if(m>1) --m; }
			else if(m-1==k && dir==1) { if(m<n) ++m; }
		} else {
			m = -1;
		}
		k += dir;
		if(k==1) dir = 1;
		else if(k==n) dir = -1;
		if(k==m) { printf("Controller %d\n", i+1); return 0; }
		if(m==-1) {
			if(k>1 && k<n) m = k-dir;
			else if(k==1) m=2;
			else m=n-1;
		}
	}
	puts("Stowaway");
	return 0;
}
