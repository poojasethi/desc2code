#include <string>
#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int n,a,b,dir;
    string state,go,which;
    cin>>n>>b>>a>>go>>which>>state;
    dir=(which=="head")?-1:1;
    for(int i=0;i<state.size();i++)
    {
        b=(state[i]=='0')?min(n,max(1,b+dir)):-1;
        a=a+dir;
        if((a==1 && dir==-1) || (a==n && dir==1))
            dir=-dir;
        if(a==b)
        {
            cout<<"Controller "<<i+1<<endl;
            return 0;
        }
        if(state[i]=='1')
            b=(a==1)?b=n:((a==n)?1:((dir==-1)?n:1));
    }
    cout<<"Stowaway"<<endl;
}
