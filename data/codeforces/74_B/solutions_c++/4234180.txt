#include <cstdio>
#include <cstring>
#include <algorithm>
const long M=500;
long n,x1,x2,dir;
int main(){
	char s[1000],time[M];
	scanf("%d%d%d",&n,&x2,&x1);
	getchar();
	gets(s);
	scanf("%s",&time);
	if (s[3]=='h') dir=-1;
	else dir=1;
	long pos=-1;
	for (long i=0;i<strlen(time);++i){
		if (x1==n) dir=-1;
		else if (x1==1) dir=1;
		
		x1+=dir;
		if (x2+dir<=n && x2+dir>0) x2=x2+dir;
		if (x1==x2 && time[i]=='0'){ pos=i+1; break;}
		if (time[i]=='1'){
			if (dir==-1) x2=n;
			else x2=1;
		}
	}
	if (pos!=-1) printf("Controller %d\n",pos);
	else printf("Stowaway\n");
	return 0;
}
