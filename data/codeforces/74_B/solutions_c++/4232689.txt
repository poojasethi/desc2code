#include <iostream>

using namespace std;

int main()
{
	int n, s, c;
	cin >> n >> s >> c;
	string temp;
	cin >> temp >> temp;
	int d;
	if (temp[0] == 't')
		d = 1;
	else
		d = -1;
	string p;
	cin >> p;
	for (int i = 0; i < p.length() - 1; i++)
	{
		if (p[i] == '0')
		{
			if (c < s && s != n)
				s++;
			else if (s < c && s != 1)
				s--;
		}	
		c += d;
		if (c == 1)
			d = 1;
		if (c == n)
			d = -1;
		if (p[i] == '1')
		{
			if (d == -1)
			{
				if (c == n)
					s = 1;
				else
					s = n;
			}
			else
			{
				if (c == 1)
					s = n;
				else
					s = 1;
			}
		}
		if (c == s)
		{
			cout << "Controller " << i + 1 << endl;
			return 0;
		}
	}
	cout << "Stowaway" << endl;
}
