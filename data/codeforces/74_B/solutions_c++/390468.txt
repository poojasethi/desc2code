string s;
char c;
int q,h,sz,sk,n;


int main()
{
    cin >> n >> sz >> sk;
    cin >> s >> s;
    if (s[0]=='h') h=-1; else h=1;
    cin >> s;
    for (int q=0; q<s.size(); q++)
	{
	    c=s[q];
	    if (c=='0')
		{
		    if (sk+h==sz) 
			if (sz+h>0 && sz+h<n+1) sz=sz+h;
		    sk=sk+h;
		    if (sk==1 || sk==n) h=-h;
		}
	    if (c=='1')
		{
		    if (sk==2 && h==-1) sz=2;
		    if (sk==n-1 && h==1) sz=n-1;
		    sz=sk;
		    sk+=h;
		    if (sk==1 || sk==n) h=-h;
		}
	    if (sk==sz) {cout << "Controller " << q+1 << endl;  return 0;}
	}
    cout << "Stowaway" << endl;
    return 0;
}
