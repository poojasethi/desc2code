#include<iostream>
using namespace std;
int main(){
	int n,a,b;
	string s,arr;
	cin>>n>>b>>a;
	cin>>s>>s;
	cin>>arr;
	if(b>a)
		b=n;
	else
		b=1;
	for(int i=0;i<arr.length();i++){
		if(arr[i]=='1'){
			if(b==1&&(a==n||s=="head"))
				b=n;
			else if(b==n&&(a==1||s=="tail"))
				b=1;
			if(a==1){
				a++;
				s="tail";
			}
			else if(a==n){
				a--;
				s="head";
			}
			else if(s=="head")
				a--;
			else
				a++;
		}
		else{
			if(a==1){
				a++;
				s="tail";
			}
			else if(a==n){
				a--;
				s="head";
			}
			else if(s=="head")
				a--;
			else
				a++;
		}
		if(a==b){
			cout<<"Controller "<<i+1<<endl;
			return 0;
		}
	}
	cout<<"Stowaway"<<endl;
}
