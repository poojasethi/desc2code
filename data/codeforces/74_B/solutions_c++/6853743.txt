#include<cstdio>
using namespace std;
char s[201];
int main()
{
	int n,m,k;bool bl;
	scanf("%d%d%d",&n,&k,&m);
	k=k<m?0:1;
	scanf("\nto %s",s);
	bl=s[0]=='h';
	scanf("%s",s);
	for(int i=0;s[i];++i)
	{
		if(s[i]-48)k=bl;
		if(bl)--m;else ++m;
		if(m==n)
		{
			if(k)
			{printf("Controller %d\n",i+1);return 0;}
			bl=1;
		}
		if(m==1)
		{
			if(!k)
			{printf("Controller %d\n",i+1);return 0;}
			bl=0;
		}
	}
	puts("Stowaway");
}
