#include <iostream>
#include <string>
#include <stdio.h>
using namespace std;

int main(){
	int n, st, co, d, p;
	string s, c;
	while(cin>>n){
		p=0;
		cin>>st>>co;
		cin>>s>>s>>c;
		d=(s[0]=='h'? -1: 1);
		for(int i=0; i<c.length(); i++){
			co+=d;
			if(st+d>0&&st+d<=n)
				st+=d;
			if(co==st&&c[i]=='0'){
				p=i+1;
				break;
			}
			if(c[i]=='1')
				st=(d>0? 1: n); 
			if(co==1||co==n)
				d=(co==1? 1: -1); 
		}
		(!p)? printf("Stowaway\n"): printf("Controller %d\n", p); 
	}
	return 0;
}
