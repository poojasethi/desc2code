#include <cstdio>
#include <cstring>

int n, m, k;
char train[300];

int main() {
	scanf("%d%d%d", &n, &m, &k);
	char sdir[10];
	scanf("%s%s", sdir,sdir);
	int dir = (sdir[0] == 'h') ? -1 : 1;
	
	scanf("%s", train);
	if (dir == -1) {
		if (m > k) m = n; else m = 1;
	}
	else {
		if (m < k) m = 1; else m = n;		
	}
	
	for (int i=0; i<strlen(train); ++i) {
		k += dir;
		if ((k == 1) || (k == n)) {
			if ((k == m) && (train[i] == '0')){
				printf("Controller %d\n", i + 1);
				return 0;	
			}
			dir = -dir;
		}
		if (train[i] == '1') {
			if (dir == -1) {
				if (k == n) m = 1; else m = n;
			}
			else {
				if (k == 1) m = n; else m = 1;
			}
		}
	}
	printf("Stowaway\n");
	return 0;
}
