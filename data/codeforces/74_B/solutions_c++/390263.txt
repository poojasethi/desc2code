#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
	int n, m, k, c;
	string s, p;
	cin >> n >> m >> k >> s >> s >> p;
	c = (s == "head") ? -1 : +1;
	for(int i=0; i<p.size(); i++)
	{
		if(m > k)
			m = n;
		else
			m = 1;

		if(k + c == 0 || k + c == n+1)
			c *= -1;
		k += c;
		if(p[i] == '0' && k == m)
		{
			cout << "Controller " << i + 1 << endl;
			return 0;
		}

		if(p[i] == '1')
			m = (c == -1) ? n : 1;
	}
	cout << "Stowaway" << endl;
	return 0;
}
