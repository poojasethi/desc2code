#include <bits/stdc++.h>
using namespace std ;
const int maxn = 1000 + 5 , inf = 0x3f3f3f3f ;
typedef long long ll ;
int n , m , k ;
bool d , b ;
char s[maxn] ;

int main()
{
    scanf( "%d%d%d\n" , &n , &m , &k ) ;
    b = ( m >= k ) ;
    gets(s) ;
    d = ( s[3] == 'h' ) ;
    gets(s) ;
    for( int i = 0 ; s[i] ; i++ )
    {
        if( s[i] == '1' ) b = d ;
        if( d ) k-- ;
        else k++ ;
        if( k == 1 || k == n )
        {
            if( k == n && b || k == 1 && !b )
            {
                printf( "Controller %d\n" , i+1 ) ;
                return 0 ;
            }
            d = !d ;
        }
    }
    puts( "Stowaway" ) ;
    return 0 ;
}
