#include <iostream>
#include <string>

using namespace std;

string direction,temp,S;
int N,M,K;

void solve()
{
    int d=direction=="head"?-1:1;
    M=M>K?N:1;
    for (int i=0;i<S.size();i++)
    {
        if (S[i]=='1')
            M=d==1?1:N;
        K+=d;
        if (K==M)
        {
            cout << "Controller " << i+1 << endl;
            return;
        }
        if (K==N||K==1)
            d=-d;
    }
    cout << "Stowaway" << endl;
}

int main()
{
    cin >> N >> M >> K >> temp >> direction >> S;
    solve();
    return 0;
}
