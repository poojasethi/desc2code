#include <iostream>
using namespace std;
int main(){
	int num,con,sto;
	cin>>num>>sto>>con;
	string dir;
	cin>>dir>>dir;
	string arr;
	cin>>arr;
	int iDir=dir=="head"?-1:1;
	for(int index=0;index<arr.size();index++){
		if((con==1 && iDir==-1) || (con==num && iDir==1))
			iDir*=-1;
		if(arr[index]=='0'){
			if((sto==1 && con==2 && iDir==-1) || (sto==num && con==num-1 && iDir==1)){
				cout<<"Controller "<<index+1<<endl;
				return 0;
			}
			if(!((sto==1 && iDir==-1) || (sto==num && iDir==1))){
				sto+=iDir;
			}
		}else{
			sto = con;
		}
		con = (con + iDir + num)%num;
		if(con==0){
			con=iDir==-1?1:num;
		}
		//cout<<sto<<","<<con<<","<<iDir<<endl;
	}
	cout<<"Stowaway"<<endl;
	return 0;

}
