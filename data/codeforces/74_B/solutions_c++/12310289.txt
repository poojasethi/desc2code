#include <bits/stdc++.h>

using namespace std;

int n, m, k, x = 222, y, d;
char a[100], b[222];

int main() {
  scanf("%d %d %d %s %s %s", &n, &k, &m, a, a, b);
  if (m == k) {
    puts("Controller 0"); return 0;
  }
  d = a[0] == 'h';
  k = d ^ (k > m);
  for (int i = 0; b[i]; i++) {
    if (b[i] == '1') k = 0;
    if (d) m--; else m++;
    if (m == n || m == 1) {
      if (k) {
        printf("Controller %d\n", i + 1); return 0;
      }
      k = 1; d ^= 1;
    }
  }
  puts("Stowaway");
}
