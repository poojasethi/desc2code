#include <iostream>
#include <string.h>
#include <algorithm>
#include <cstdio>
using namespace std;

int n, x, y;
char s[207];

void work()
{
    scanf("%s%s", s, s);
    int dec = -1;
    if (s[0] == 't') dec = 1;
    scanf("%s", s);
    for (int i = 0; i < strlen(s); i++)
    {
        if (s[i] == '1') x = y;
        else             x += dec;
        if (x < 1) x = 1;
        if (x > n) x = n;
        y += dec;
        if (y == 1 || y == n) dec *= -1;
        if (x == y) 
        {
            printf("%s %d\n", "Controller", i+1);
            return;
        }
    }
    puts("Stowaway");
}

int main()
{
    while (scanf("%d%d%d", &n, &x, &y) != EOF) work();
}
