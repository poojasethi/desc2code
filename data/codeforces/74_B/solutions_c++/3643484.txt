#include<iostream>
#include<cstring>
#include<string>
using namespace std;
string oprea,tim;
int n,control,stoway;
int main()
{
  while(cin>>n>>stoway>>control)
  {
    cin>>oprea>>oprea>>tim;
    int d=oprea=="head"?-1:1;///控制方向
    stoway=stoway>control?n:1;
    for(int i=0;i<tim.size();++i)
    {
      if(tim[i]=='1')
       stoway=d==1?1:n;
      control+=d;
      if(control==stoway)
        {cout<<"Controller "<<i+1<<"\n";goto out;}
      if(control==n||control==1)
        d=-d;
    }
    cout<<"Stowaway\n";
    out:;
  }
}
