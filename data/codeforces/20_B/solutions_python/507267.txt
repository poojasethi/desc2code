ans={}
a,b,c=map(int,raw_input().split())
if a==b==c==0: print -1
elif a==b==0: print 0
elif a==0: print 1, '%.6f' % ((-c*1.0)/b)
else:
    d=float(b**2-4*a*c)
    if d<0: print 0
    else:
        x1=(-b+(d**0.5))/(2*a)
        x2=(-b-(d**0.5))/(2*a)
        ans[x1]=1
        ans[x2]=1
        print len(ans.keys())," ".join(map(lambda x: '%.6f' % x,sorted(ans.keys())))
