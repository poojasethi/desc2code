a,b,c=map(int, raw_input().split())
chk=b**2-4*a*c
ans=[]

if a==b==c==0:
    print -1
elif a==b==0 or chk<0:
    print 0
else:
    if a!=0:
        ans.append(((-1.0*b)+(chk**0.5))/(2*a))
        ans.append(((-1.0*b)-(chk**0.5))/(2*a))
        if ans[0]==ans[1]:
            ans.pop()
    elif b!=0:
        ans.append((-1.0*c)/b)
    ans.sort()
    print len(ans)
    for i in ans:
        print i
