A, B, C = map(float, raw_input().split())
if A == B == 0:
  print -1 if C == 0 else 0
elif A == 0:
  print 1
  print -C / B
else:
  D = B ** 2 - 4 * A * C
  if D < 0:
    print 0
  else:
    import math
    x1 = (-B + math.sqrt(D)) / (2 * A)
    x2 = (-B - math.sqrt(D)) / (2 * A)
    if x1 == x2:
      print 1
      print x1
    else:
      print 2
      print min(x1, x2)
      print max(x1, x2)
