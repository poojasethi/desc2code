from math import sqrt
a, b, c = map(float, raw_input().split())
delta = b**2-4*a*c
if (a == 0):
    print -1 if ((b == 0) and (c == 0)) else 0 if ((b == 0) and (c != 0)) else 1
    if (b != 0): print "%.5f" %(-c/b)
elif (delta > 0):
    print 2
    for i in range(2):
        print "%.5f" %sorted([(-b+sqrt(delta))/(2*a), (-b-sqrt(delta))/(2*a)])[i]
elif (delta == 0):
    print 1
    print "%.5f" %(-b/(2*a))
else: print 0
