#include <stdio.h>
#include <cmath>

double a,b,c,d,min,max;

int main(){
    scanf("%lf%lf%lf",&a,&b,&c);
    d = b*b-4*a*c;
    if (a==0 && b==0 && c==0) printf("-1");
    else if (a==0 && b==0) printf("0");
    else if ((b==0 || a==0) && c==0) printf("1\n0");
    else if (a==0) printf("1\n%lf",-c/b);
    else if (d<0) printf("0");
    else if (d==0) printf("1\n%lf",-b/(2*a));
    else
    {
      min = (-b-sqrt(d))/(2*a);
      max = (-b+sqrt(d))/(2*a);
      if (min>max)
      {
        d=min;
        min=max;
        max=d;
      }
      printf("2\n%lf\n%lf", min, max);
    }
    return 0;
}
