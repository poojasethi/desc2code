#include <cmath>
#include <cstdio>
#include <iostream>
using namespace std;

int main()
{
	//freopen("a.in", "r", stdin);
	//freopen("a.out", "w", stdout);
	long long a, b, c, d;
	double sq, a1, a2;
	cin >> a >> b >> c;
	if (!a && !b && !c) printf("-1"); else
	if (!a) {
		if (!b) printf("0"); else printf("1\n%lf", -1.0*c/b);
	} else {
		d = b*b - 4*a*c;
		if (d < 0) printf("0"); else
		if (!d) printf("1\n%lf", -1.0*b/2/a); else
		{
			sq = sqrt(1.0 * d);
			a1 = (-b+sq)/2/a;
			a2 = (-b-sq)/2/a;
			if (a1 > a2) swap(a1, a2);
			printf("2\n%lf\n%lf", a1, a2);
		}
	}
}
