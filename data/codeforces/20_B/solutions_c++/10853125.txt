#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;
int main () {
    double a,b,c,d;
    cin >> a >> b >> c;
    if(!a and b)
    {
        printf("1\n%.7lf\n",(-c)/b);
        return 0;
    }
    if(!a and !b and !c)
    {
        cout << -1 << endl;
        return 0;
    }
    d=b*b-4*a*c;
    if((d<0) or (!a and !b and c))
    {
        cout << 0 << endl;
        return 0;
    }
    d=sqrt(d);
    if(a<0)
        d*=-1;
    if(!d)
        printf("1\n%.7lf\n",(-b-d)/(2*a));
    else
        printf("2\n%.7lf\n%.7lf\n",(-b-d)/(2*a),(-b+d)/(2*a));
    return 0;
}
