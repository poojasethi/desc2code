#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
double delta(double a,double b,double c){
	return b*b - 4*a*c;
}
int main(){
	double a,b,c;
	cin>>a>>b>>c;
	double s1,s2,de=delta(a,b,c);
	if(de<0) {cout<<0;   return 0;}
	if(!a && !b && !c)
	{	cout<<-1;
		return 0;
	}
	if(!a && b){
		printf("1\n%.6lf",-c/b);
		return 0;
	}
	if(!b && !a){
		cout<<0;
		return 0;
	
	}
	s1= (-b+sqrt(de))/(2*a);
	s2= (-b-sqrt(de))/(2*a);
	if(s1==s2){
		printf("1\n%.6lf",s1);
		return 0;
	}
	cout<<2<<endl;
	
	printf("%.6lf\n%.6lf\n",min(s1,s2),max(s1,s2));


	return 0;
}
