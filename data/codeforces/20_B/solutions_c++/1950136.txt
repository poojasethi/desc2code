#include <stdio.h>
#include <Math.h>

int main()
{
    double a,b,c,p,q,d;
    scanf("%lf%lf%lf",&a,&b,&c);
    if(a==0 && b==0 && c==0)
      printf("-1");
    else if(a==0 && b==0)
      printf("0");
    else if(a==0)
      printf("1\n%.5lf",(-1.0*c/b));
    else if((b*b-4*a*c)<0)
      printf("0");
    else
    {
      d=b*b-4*a*c;
      p=(-b+sqrt(d))/(2*a);
      q=(-b-sqrt(d))/(2*a);
      if(p<q)
        printf("2\n%.5lf\n%.5lf",p,q);
      else if(p>q)
        printf("2\n%.5lf\n%.5lf",q,p);
      else
        printf("1\n%.5lf",p);
    }
    return 0;
}
