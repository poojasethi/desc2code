#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;

int main()
{	
	double a, b, c;
	cin >> a >> b >> c;
	double d = b * b - 4 * a * c;	
	if (a == 0 && b == 0 && c == 0) printf("-1\n");
	else if (a == 0 && b == 0) printf("0\n");
	else if (a == 0)
	{
		printf("1\n%.10f\n", - c / b);
	}
	else if (d < 0) printf("0\n");	
	else if (d == 0) printf("1\n%.10f\n",  (-b + sqrt(d)) / 2 / a);
	else 
	{
		double x = (-b + sqrt(d)) / 2 / a;
		double y = (-b - sqrt(d)) / 2 / a;
		if (x > y) swap(x, y);
		printf("2\n%.10f\n", x);
		printf("%.10f\n", y);
	}
	return 0;	
}
