#include <iostream>
#include <cmath>
#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
	double a, b, c;
	scanf("%lf%lf%lf", &a, &b, &c);
	if(a == 0 && b == 0 && c== 0)
	{
		printf("%d\n", -1);
	}
	else if(a == 0)
	{
		if(b == 0)	printf("%d\n", 0);
		else	printf("%d\n%.8lf\n", 1, -c/b);
	}
	else
	{
		double d = b * b - 4 * a * c;
		if(d == 0.0)
		{
			printf("%d\n", 1);
			printf("%.8lf\n", -b/(2*a));
		}
		else if(d < 0)
		{
			printf("%d\n", 0);
		}
		else
		{
			printf("%d\n", 2);
			double x1 = (-b - sqrt(d)) / (2 * a);
			double x2 = (-b + sqrt(d)) / (2 * a);
			printf("%.8lf\n", min(x1, x2));
			printf("%.8lf\n", max(x1, x2));
		}
	}
	return 0;
}
