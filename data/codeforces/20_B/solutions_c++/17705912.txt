#include<cstdio>
#include<cmath>
typedef long long ll;
int main(){
	ll a,b,c;
	scanf("%I64d%I64d%I64d",&a,&b,&c);
	if(a==0){
		if(b==0){
			if(c==0)
				puts("-1");
			else
				puts("0");
		}
		else{
			printf("1\n%lf",-1.0*c/b);
		}
	}
	else{
		ll delta=b*b-4*a*c;
		if(delta<0)
			puts("0");
		else if(delta==0)
			printf("1\n%lf",-0.5*b/a);
		else{
			double x1=(-b-sqrt(delta))/(2.0*a),x2=(-b+sqrt(delta))/(2.0*a);
			puts("2");
			if(x1<x2)
				printf("%lf\n%lf",x1,x2);
			else
				printf("%lf\n%lf",x2,x1);
		}
	}
	return 0;
}
