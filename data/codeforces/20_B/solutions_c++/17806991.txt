#include <cstdio>
#include <cmath>

int A, B, C;

int main() {
	scanf("%d%d%d", &A, &B, &C);
	if (!A) {
		if (!B) {
			puts(C ? "0" : "-1");
		} else {
			printf("1\n%.15lf\n", -1.0 * C / B);
		}
	} else {
		if (A < 0) {
			A = -A;
			B = -B;
			C = -C;
		}
		long long delta = (long long)B * B - 4LL * A * C;
		if (delta > 0) {
			printf("2\n%.15lf\n%.15lf\n", (-B - sqrt((double)delta)) / 2.0 / A, (-B + sqrt((double)delta)) / 2.0 / A);
		} else if (!delta) {
			printf("1\n%.15lf\n", -B / 2.0 / A);
		} else {
			puts("0");
		}
	}
	return 0;
}
