#include<stdio.h>
#include<math.h>

double A,B,C;
int main()
{
	scanf("%lf %lf %lf",&A,&B,&C);
    double d,ans1,ans2;
	if(A==0)
	{
		if(B)
		{
			ans1=-C/B;
			printf("1\n%lf\n",ans1);
		}
		else
		{
			if(C)
			printf("0\n");
			else
			printf("-1\n");
		}
	}
	else
	{
		d=(B*B)-(4*A*C);
		if(d<0)
			printf("0\n");
		else if(d==0)
		{
			ans1=-B/(2*A);
			printf("1\n%lf\n",ans1);
		}
		else
		{
			double r=sqrt(d);
			if(A>0)
			{
			ans1=(-r-B)/(2*A);
			ans2=(r-B)/(2*A);
			}
			else
			{
				ans1=(r-B)/(2*A);
				ans2=(-r-B)/(2*A);
			}
			printf("2\n%lf\n%lf\n",ans1,ans2);
		}
		
	}

	return 0;
}
