#include <cstdio>
#include <cmath>
double A, B, C;
int main()
{
	scanf("%lf%lf%lf", &A, &B, &C);
	if (A == 0 && B == 0 && C == 0)
		puts("-1");
	else if (A == 0 && B == 0)
		puts("0");
	else if (A == 0)
	{
		puts("1");
		printf("%.7lf\n", -C / B);
	}
	else
	{
		double D = B*B - 4*A*C;
		if (D < 0)
			puts("0");
		else if (D == 0)
		{
			puts("1");
			printf("%.7lf\n", -B / (2*A));
		}
		else
		{
			puts("2");
			double x1 = (-B + sqrt(D))/(2*A);
			double x2 = (-B - sqrt(D))/(2*A);
			if (x1 > x2)
			{
				double temp = x1;
				x1 = x2;
				x2 = temp;
			}
			printf("%.7lf\n", x1);
			printf("%.7lf\n", x2);
		}
	}
}
