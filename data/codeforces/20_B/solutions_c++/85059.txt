#include <cstdio>
#include <cmath>

int main()
{
	int A, B, C;
	scanf("%d%d%d", &A, &B, &C);
	if (A == 0)
		if (B == 0)
			if (C == 0)
				printf("-1\n");
			else
				printf("0\n");
		else
			printf("1\n%lf\n", (double)(-C) / B);
	else
	{
		long long D = (long long)B * B - (long long)4 * A * C;
		if (D < 0)
			printf("0\n");
		else
		{
			double x1 = (-B - sqrt(D)) / 2 / A,
			       x2 = (-B + sqrt(D)) / 2 / A;
			if (x1 > x2)
			{
				double t = x1;
				x1 = x2;
				x2 = t;
			}
			if (D == 0)
				printf("1\n%lf\n", x1);
			else
				printf("2\n%lf\n%lf\n", x1, x2);
		}
	}

	return 0;
}
