#include <stdio.h>
#include <math.h>
int main() {
    double a, b, c;
    scanf("%lf%lf%lf", &a, &b, &c);
    if (a == 0 && b == 0) {
        if (c == 0) puts("-1");
        else puts("0");
        return 0;
    }
    if (a == 0) printf("1\n%.10f\n", -c/b);
    else {
        double t = b*b - 4*a*c;
        if (t < 0) puts("0");
        else if (t == 0) printf("1\n%.10f\n", -b/2/a);
        else {
            t = sqrt(t);
            c = (-b-t)/2/a, b = (-b+t)/2/a;
            if (c < b) printf("2\n%.10f\n%.10f\n", c, b);
            else printf("2\n%.10f\n%.10f\n", b, c);
        }
    }
    return 0;
}
