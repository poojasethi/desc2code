#include <math.h>
#include <stdio.h>

int main() {
 double a, b, c;
 scanf("%lf%lf%lf", &a, &b, &c);

 if (a) {
  double d = b * b - 4 * a * c;
  if (d < 0) puts("0");
  else if (d > 0) {
   puts("2");
   d = sqrt(d);
   if (a < 0) {
    printf("%lf\n", (d - b) / (2 * a));
    printf("%lf\n", (-d - b) / (2 * a));
   } else {
    printf("%lf\n", (-d - b) / (2 * a));
    printf("%lf\n", (d - b) / (2 * a));
   }
  } else {
   puts("1");
   printf("%lf\n", -b / (2 * a));
  }
 } else if (b) {
  puts("1");
  printf("%lf\n", -c / b);
 } else puts(c ? "0" : "-1");
 return 0;
}
