#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

long long a,b,c;
int main(){
	cin>>a>>b>>c;
	if (a==0){
		if (b==0)
			if (c==0) cout<<-1<<endl;
			else cout<<0<<endl;
		else{
			printf("1\n%.10lf\n",-(double)c*1.0/b);
		}
	}else{
		if (a<0){
			a=-a;b=-b;c=-c;
		}
		if (b*b-4*a*c<0) cout<<0<<endl;
		else
		if (b*b-4*a*c==0){
			printf("1\n%.10lf\n",-(double)b*1.0/(2*a));
		}else{
			printf("2\n%.10lf\n%.10lf\n",(-(double)b*1.0-sqrt((double)(b*b-4*a*c)))/(2*a),
				(-(double)b*1.0+sqrt((double)(b*b-4*a*c)))/(2*a));
		}
	}
}
