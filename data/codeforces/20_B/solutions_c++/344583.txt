#include<cmath>
#include<cstdio>
#include<algorithm>

using namespace std;

int main(){
	double a,b,c;	scanf("%lf%lf%lf",&a,&b,&c);
	if(a!=0){
		double d=b*b-4*a*c;
		if(d>0){
			double x1=(-b-sqrt(d))/(2*a),x2=(-b+sqrt(d))/(2*a);
			if(x1>x2)	swap(x1,x2);
			printf("2\n%.9f\n%.9f\n",x1,x2);
		}
		else if(d==0)	printf("1\n%.9f\n",-b/(2*a));
		else			printf("0\n");
	}
	else{
		if(b!=0)	printf("1\n%.9f\n",-c/b);
		else{
			if(c!=0)	printf("0\n");
			else		printf("-1\n");
		}
	}
	return 0;
}
