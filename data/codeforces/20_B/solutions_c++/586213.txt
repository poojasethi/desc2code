#include <stdio.h>
#include <cmath>
using namespace std;

int main(void) {
	int a, b, c;
	scanf("%d%d%d", &a, &b, &c);
	if(a<0) a=-a,b=-b,c=-c;
	if(a==0 && b==0) {
		if(c==0) puts("-1");
		else puts("0");
	} else if(a==0) {
		puts("1");
		printf("%.10f\n", -(double)c/b);
	} else if(b*1LL*b==4LL*a*c) {
		puts("1");
		printf("%.10f\n", -(double)b/(a*2));
	} else if(b*1LL*b < 4LL*a*c) {
		puts("0");
	} else {
		puts("2");
		printf("%.10f\n", (-(double)b-sqrt((double)(b*1LL*b-4LL*a*c))) /(double)(a*2));
		printf("%.10f\n", (-(double)b+sqrt((double)(b*1LL*b-4LL*a*c))) /(double)(a*2));
	}
	return 0;
}
