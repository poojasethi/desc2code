#include <cstdio>
#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	long long a, b, c;
	cin >> a >> b >> c;
	if (a == 0) {
		if (b == 0)
			printf("%d\n", c == 0? -1: 0);
		else
			printf("1\n%.10f\n", (double)(-c) / b);
	}
	else {
		long long d = b * b - 4 * a * c;
		if (d < 0)
			printf("0\n");
		else if (d == 0)
			printf("1\n%.10f\n", (double)(-b) / (2 * a));
		else {
			double x1 = (-b - sqrt(d)) / (2 * a);
			double x2 = (-b + sqrt(d)) / (2 * a);
			if (x1 > x2) { double t = x1; x1 = x2; x2 = t; }
			printf("2\n%.10f\n%.10f\n", x1, x2);
		}
	}
	return 0;
}
