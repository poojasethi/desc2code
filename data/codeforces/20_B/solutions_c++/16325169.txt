#include <cstdio>
#include <cmath>
long long a,b,c;
int main(){
   scanf("%lld %lld %lld", &a,&b,&c);
   if (!a && !b && !c) printf("-1");
   else if (!a && !b) printf("0");
   else if (!a) printf("1\n%lf", -c/(double)b);
   else if(b*b<4*a*c) printf("0");
   else if(b*b==4*a*c) printf("1\n%lf", -b/(double)(2*a));
   else{
      double x = b*b-4*a*c;
      x = sqrt(x);
      if (a>0) printf("2\n%lf\n%lf", (-b-x)/(2*a), (-b+x)/(2*a));
      else printf("2\n%lf\n%lf", (-b+x)/(2*a), (-b-x)/(2*a));
   }
   return 0;
}
