#include<iostream>
#include<string.h>
#include<math.h>
#include<stdio.h>
using namespace std;
int main()
{
	double a,b,c,d,e,n,f,s1,s2;
	cin>>a>>b>>c;
	d=b*b-4*a*c;
	if(a!=0)
	{
	if(d<0)
	{cout<<0<<endl;
	}
	if(d==0)
	{
		cout<<1<<endl;
		s1=(-b-sqrt(d))/(2*a);
		printf("%.5lf\n",s1);
	}
	if(d>0)
	{
		cout<<2<<endl;
		s1=(-b-sqrt(d))/(2*a);
		s2=(-b+sqrt(d))/(2*a);
		if(s1>s2)
		{
			n=s1;s1=s2;s2=n;
		}
		printf("%.5lf\n%.5lf\n",s1,s2);
	}
	}
	else if(a==0&&b!=0)
	{
		cout<<1<<endl;
		s1=-c/b;
		printf("%.5lf\n",s1);
	}
	else if(a==0&&b==0&&c!=0)
		cout<<0<<endl;
	else if(a==0&&b==0&&c==0)
		cout<<-1<<endl;
}
