#include<iostream>
#include<cmath>
#include<iomanip>
using namespace std;
int main()
{
	cout<<fixed<<setprecision(10);
	double a,b,c,x1,x2,k;
	while(cin>>a>>b>>c)
	{
		if(a==0&&b==0&&c==0) {cout<<-1<<endl;continue;}
		else if(a==0&&b==0&&c!=0) {cout<<0<<endl;continue;}
		else if(a==0&&b!=0) {cout<<1<<endl<<-c/b<<endl;continue;}
		k=pow(b,2)-4*a*c;
		if(k>=0)
		{
			x1=(-b-sqrt(k))/2/a;
			x2=(-b+sqrt(k))/2/a;
			if(x1==x2) cout<<1<<endl<<x1<<endl;
			else  if(x1<x2)cout<<2<<endl<<x1<<endl<<x2<<endl;
			else cout<<2<<endl<<x2<<endl<<x1<<endl;
		}
		else cout<<0<<endl;
	}
	return 0;
}
