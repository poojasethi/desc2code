#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;


int main()
{
	int t,m,n,j,k,l;
	long long a,b,c,d;
	while(cin>>a>>b>>c)
	{
		d=b*b-4*a*c;
		if(a==0&&b==0)
		{
			if(c==0)cout<<-1<<endl;
			else cout<<0<<endl;
		}
		else if(a==0)printf("1\n%.18lf\n",double(-c)/b);
		else if(d<0)cout<<0<<endl;
		else if(d==0)printf("1\n%.18lf\n",double(-b)/2/a);
		else 
		{
			double x=(double(-b)-sqrt((double)d))/(2.0*a),y=(double(-b)+sqrt((double)d))/(2.0*a);
			if(x>y)printf("2\n%.18lf\n%.18lf\n",y,x);
			else printf("2\n%.18lf\n%.18lf\n",x,y);
		}
	}
	return 0;
}
