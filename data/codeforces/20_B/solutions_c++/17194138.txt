#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;
int main()
{
	double a, b, c;
	cin >> a >> b >> c;
	if (!a && !b)
	{
		if (!c) cout << -1;
		else cout << 0;
	}
	else if (!a && b)
		cout << 1 << endl << setprecision(20) << (double) -c / b;
	else
	{
		double D = b * b - 4 * a * c;
		if (!D)
		
			cout << 1 << endl << setprecision(20) << (double) -b / (2 * a);
		else if (D > 0)
		{
			cout << 2 << endl;
			double x1 = (double) (-b - sqrt(D)) / (2 * a);
			double x2 = (double) (-b + sqrt(D)) / (2 * a);
			cout << setprecision(20) << min(x1, x2) << endl << max(x1, x2) << endl;
		}
		else cout << 0;
	}
}
