#include<stdio.h>
#include<math.h>
int main()
{
	long long a,b,c,d;double dd;
	scanf("%lld%lld%lld",&a,&b,&c);
	d=b*b-4*a*c;dd=sqrt(d);
	if(a==0&&b==0&&c||d<0)printf("0");
	else if(a==0&&b==0&&c==0)printf("-1");
	else if(a==0)printf("%d\n%.10lf",1,-(double)c/(double)b);
	else if(b==0&&c==0)printf("%d\n%.10lf",1,0);
	else if(d==0)printf("%d\n%.10lf",1,-(double)b/(double)a/2);
	else if(d>0&&(-(double)b-dd)/(double)a/2<(-(double)b+dd)/(double)a/2)printf("%d\n%.10lf\n%.10lf",2,(-(double)b-dd)/(double)a/2,(-(double)b+dd)/(double)a/2);
	else printf("%d\n%.10lf\n%.10lf",2,(-(double)b+dd)/(double)a/2,(-(double)b-dd)/(double)a/2);
	return 0;
}
