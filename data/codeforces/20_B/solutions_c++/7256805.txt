#include<cstdio>
#include<cmath>
int a,b,c;
int main(){
  scanf("%d %d %d",&a,&b,&c);
  if(a == 0){
    if(b == 0 && c == 0) printf("-1");
    else if(b == 0 && c != 0) printf("0");
    else printf("1\n%.12lf\n",-1.0*c/b);
  }
  else{
    if(a < 0) a *= -1, b *= -1, c *= -1;
    double x = 1.0*b*b-4.0*a*c;
    if(x < -1e-12) printf("0");
    else if(x > -1e-12 && x < 1e-12) printf("1\n%.12lf",-1.0*b/(2.0*a));
    else printf("2\n%.12lf\n%.12lf",(-1.0*b-sqrt(x))/(2.0*a), (-1.0*b+sqrt(x))/(2.0*a));
  }
  return 0;
}
