#include "stdio.h"
#include "math.h"
int main()
{
	double root1,root2,p,temp;
	double a,b,c;
	scanf("%lf%lf%lf",&a,&b,&c);
	if(a==0&&b==0&&c==0)
		printf("-1\n");
	if(a==0&&b==0&&c!=0)
		printf("0\n");
	if(a==0&&b!=0)
		printf("1\n%.5f\n",-c*1.0/b);
	if(a!=0)
	{
		p=b*b-4*a*c;
		if(p<0)
			printf("0\n");
		else if(p==0)
		{
			root1=(-b)*1.0/(2*a);
			printf("1\n%.5f\n",root1);
		}
		else
		{
			root1=(-b+sqrt(p))*1.0/(2*a);
			root2=(-b-sqrt(p))*1.0/(2*a);
			if(root1>root2)
			{
				temp=root1;
				root1=root2;
				root2=temp;
			}
			printf("2\n%.5f\n%.5f\n",root1,root2);
		}
	}
	return 0;
}
