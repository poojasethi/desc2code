#include<iostream>
#include<iomanip>
#include<cmath>
using namespace std;
int main() {
        __int64 a,b,c,d;
        cin>>a>>b>>c;
        cout<<fixed<<setprecision(9);
        d=b*b-4*a*c;
        if(a==0) {
                if(b==0) if(c)cout<<0; else cout<<-1;
                else cout<<1<<endl<<1.*c/-b;
        }
        else if(d<0) cout<<0;
        else {
                double t=(-b+sqrt(d))/2/a,u=-1.*b/a-t;
                if(d==0) cout<<1<<endl<<t;
                else if(t<u) cout<<2<<endl<<t<<endl<<u;
                else cout<<2<<endl<<u<<endl<<t;
        }
}
