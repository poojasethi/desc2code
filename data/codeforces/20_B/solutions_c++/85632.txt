/*Equation*/

#include<cstdio>
#include<cmath>
#include<cstdlib>
#include<algorithm>

using namespace std;

int main()
{
	int A,B,C;
	double d,sqrtd,root1,root2;
	scanf("%d %d %d",&A,&B,&C);
	if(A==0)
	{
		if(B==0)
		{
			if(C==0)
				printf("-1\n");
			else
				printf("0\n");
		}
		else
			printf("1\n%.9lf\n",-(double)C/B);
	}
	else
	{
		d=(double)B*B-4*(double)A*C;
		if(d<0)
			printf("0\n");
		else if(d==0)
			printf("1\n%.9lf\n",-(double)B/(2*A));
		else
		{
			sqrtd=sqrt(d);
			root1=(-B+sqrtd)/(2*A);
			root2=(-B-sqrtd)/(2*A);
			printf("2\n%.9lf\n%.9lf\n",min(root1,root2),max(root1,root2));
		}
	}
	return 0;
}
