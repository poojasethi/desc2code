#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
double a,b,c,k,k1,k2,k3,k4,r,t;
int main()
{
cin>>a>>b>>c;
if (a==0)
{
	if (b==0)
		{
		if (c==0)
			{cout<<-1;	
			 return 0;}
		else {
		        cout<<0;
			return 0;
			}
		}
	else {
	        k=-c/b;
		cout<<1<<endl;
		printf("%5f",k);
		return 0;
		}	
}
t=-(b)/(2*a);
r=b*b-4*a*c;
if (r<0)
	{
	cout<<0;
	return 0;
	}
if (r==0)
	{
	cout<<1<<endl;
	printf("%5f",t); 	
	return 0;
	}
if (r>0)
	{
	r=sqrt(r);
	k1=t+r/(2*a);
	k2=t-r/(2*a);
	cout<<2<<endl;
	k3=min(k1,k2);
	k4=max(k1,k2);
	printf("%5f\n",k3);
	printf("%5f\n",k4);
	}       
return 0;
}
