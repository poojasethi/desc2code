#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int main(){
	double a,b,c,delta,ans1,ans2;
	cin>>a>>b>>c;
	if (a==0&&b==0&&c==0){cout<<-1;return 0;}
	if (a==0&&b==0){cout<<0;return 0;}
	if (a==0&&c==0){cout<<1<<' '<<0;return 0;}
	if (a==0){printf("1 %.10f",-c/b);return 0;}
	delta=b*b-4*a*c;
	if (delta<0) cout<<0;
	else if (delta==0){
		cout<<1<<endl;
		printf("%.10f",-b/(a*2));
	}
	else{
		cout<<2<<endl;
		ans1=(-b-sqrt(delta))/(2*a);
		ans2=(-b+sqrt(delta))/(2*a);
		if (ans1>ans2)printf("%.10f\n%.10f",ans2,ans1);else printf("%.10f\n%.10f",ans1,ans2);
	}
	return 0;
}
