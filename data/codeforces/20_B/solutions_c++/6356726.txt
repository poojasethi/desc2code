#include <iostream>
#include <stdio.h>
#include <cmath>
using namespace std;
int main()
{
	double A,B,C,x,y;
	cin>>A>>B>>C;
	if(A==0)
	{
		if(B==0)	
		{
			if(C==0) cout<<"-1"<<endl;
			else cout<<"0"<<endl;
		}
		else
		{
			cout<<"1"<<endl;
			printf("%.10f\n",0-C/B);
		}
	}
	else
	{
		double D = B*B - 4*A*C;
		if(D<0) cout<<"0"<<endl;
		else if(D==0)
		{
			cout<<"1"<<endl;
			printf("%.10f\n",0-B/(2*A));
		}
		else
		{
			printf("%d\n",2);
			x=(0-B-sqrt(D))/(2*A);
			y=(0-B+sqrt(D))/(2*A);
			if(x<y)
			{
				printf("%.10f\n",x);
				printf("%.10f\n",y);
			}
			else
			{
				printf("%.10f\n",y);
				printf("%.10f\n",x);
			}
		}
	}
}
