#include <bits/stdc++.h>
using namespace std;


int main(){
  double a,b,c,x1,x2,d;
  cin >> a >> b >> c;
  d = b*b-4*a*c;
  if(d < 0){
    cout << 0 << "\n";
    return 0;
  }
  d = sqrt(d);
  if(a == 0){
    if(b == 0)
      if(c == 0)
	cout << -1 << "\n";
      else
	cout << 0 << "\n";
    else
      printf("1\n%.5f\n",-c/b);
    return 0;
  }
  x1 = (-b - d)/(2.0*a);
  x2 = (-b + d)/(2.0*a);
  if(x1 == x2)
    printf("1\n%.5f\n",x1);
  else if(x1 < x2)
    printf("2\n%.5f\n%.5f\n",x1,x2);
  else
    printf("2\n%.5f\n%.5f\n",x2,x1);
  return 0;
}
