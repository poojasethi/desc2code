#include<stdio.h>
#include<math.h>
#include<string.h>
int main()
{
	double a,b,c,m,t1,t2,t3,t4;
	while(scanf("%lf%lf%lf",&a,&b,&c)!=EOF)
	{
		m=b*b-4*a*c;
		if(a==0)
		{
			if(b==0&&c==0)
				printf("%d\n",-1);
			else if(b==0&&c!=0)
		    	printf("%d\n",0);
			else if(b!=0)
		    	printf("%d\n%.10f\n",1,-1*c/b);
		}
		else
		{
			if(m<0)
				printf("%d\n",0);
			else if(m==0)
				printf("%d\n%.10f\n",1,-1*b/(2*a));
			else
			{
				m=sqrt(m);
				t1=(-1*m-b)/(2*a);
				t2=(m-b)/(2*a);
				t3=t1<t2?t1:t2;
				t4=t1>t2?t1:t2;
				printf("%d\n%.10f\n%.10f\n",2,t3,t4);
			}
		}
	}
	return 0;
}
