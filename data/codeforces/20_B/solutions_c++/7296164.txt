#include<stdio.h>
#include<math.h>
int main()
{
	float a,b,c;
	scanf(" %f %f %f",&a,&b,&c);
	if(a==0&&b==0&&c==0)
	{
		printf("-1\n");
	}
	else if(a==0)
	{
		if(b==0&&c!=0)
		{printf("0\n");}
		else printf("1\n%f\n",-c/b);
	}
	else
	{
		if(b*b-4*a*c<0)
		{
			printf("0\n");
		}
		else if(b*b-4*a*c==0)
		{
			printf("1\n%f",(-b)/(2*a));
		}
		else 
		{
			printf("2\n%");
			float x1=(-b-sqrt(b*b-4*a*c))/(2*a);
			float x2=(-b+sqrt(b*b-4*a*c))/(2*a);
			if(x1>x2)
			{
				printf("%f\n%f\n",x2,x1);
			}
			else
				printf("%f\n%f\n",x1,x2);
		}
	}
	return 0;
}
