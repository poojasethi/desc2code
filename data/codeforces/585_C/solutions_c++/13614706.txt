#include <bits/stdc++.h>
#define fi first
#define se second
using namespace std;

typedef long long lint;
typedef pair < lint , char > ii;

lint x,y;
queue < ii > q;

int main(){
	
	cin >> x >> y;
	
	if( __gcd(x,y) != 1 ){
		puts("Impossible");
		return 0;
	}
	
	while( x > 1 or y > 1 ){
		if( x > y ){
			q.push(ii((x-1)/y,'A'));
			x -= (x-1)/y*y;
		} else {
			q.push(ii((y-1)/x,'B'));
			y -= (y-1)/x*x;
		}
	}
	
	while( !q.empty() ){
		cout << q.front().fi << q.front().se;
		q.pop();
	}
	
	return 0;
}
