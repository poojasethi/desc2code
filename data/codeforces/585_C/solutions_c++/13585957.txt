#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
ll m = 0, n = 0;
ll gcd(ll x , ll y) {return y == 0 ? x : gcd(y,x%y);}

int main() {
	ios::sync_with_stdio(false);
	cin>>m>>n;
	if(gcd(m,n) != 1) {cout << "Impossible\n";return 0;}
	while(m != 1 || n != 1) {
		if(n > m) {
			if(m == 1) {cout << n-1 << "B\n";return 0;}
			cout << n/m << 'B';n %= m;
		}
		else {
			if(n == 1) {cout << m-1 << "A\n";return 0;}
			cout << m/n << 'A';m %= n;
		}
	}
	return 0;
}
