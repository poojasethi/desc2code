#include <iostream>
typedef long long ll;
using namespace std;
ll
gcd ( ll a, ll b )
{
  ll c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
void recur(ll x,ll y){
 if (x==1)
 cout<<y-1<<"B"<<endl;
 else if (y==1)
  cout<<x-1<<"A"<<endl;
 else if (x>y){
  cout<<x/y<<"A";
  recur(x % y,y);
 }
 else{
  cout<<y/x<<"B";
  recur(x,y % x);
 }
}
int main(){
 ll x,y;
 cin>>x>>y;
 if (gcd(x,y)>1)
  cout<<"Impossible"<<endl;
 else{
  recur(x,y);
 }
 return 0;
}
