#include<bits/stdc++.h>
using namespace std;
typedef long long int64;
int64 a,b;

int main(){
    cin>>a>>b; int t=0;
    if(__gcd(a,b)!=1){ puts("Impossible"); return 0; }
    while(b>1){
        if(a>b) printf("%I64d%c",a/b,'A'+t);
        b^=a^=b^=a%=b,t=1-t;
    }
    if(a>1) printf("%I64d%c\n",a-1,'A'+t);
    return 0;
}
