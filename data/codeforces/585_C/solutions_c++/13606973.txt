#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
typedef long long ll;
ll n,m,k,ans[70][2];
int u,sum;
bool o;

ll gcd(ll a,ll b)
{
    if (!b) return a;
    if (a>b)
    {
        sum++;
        ans[sum][0]=(a-1)/b; ans[sum][1]=u;
    }
    u^=1;
    return gcd(b,a%b);
}

int main()
{
    cin>>n>>m;
    if (gcd(n,m)!=1)
    {
        cout<<"Impossible";
        return 0;
    }
    for (int i=1;i<=sum;i++)
    {
        printf("%I64d%c",ans[i][0],!ans[i][1]?'A':'B');
    }
}
