#include <cstdio>
#include <iostream>
#include <cstdlib>
#include <cstring>


using namespace std;

typedef unsigned long long int ull;

ull gcd (ull a, ull b) {
	while (a && b) {
		a > b ? a %= b : b %= a;
	}
	return a + b;
}


int main(){
	//freopen("input.txt", "rt", stdin);
	//freopen("output.txt", "wt", stdout);
	ull a, b;
	cin >> a >> b;
	if (gcd (a, b) != (ull)1) {
		cout << "Impossible\n";
		return 0;
	}
	string ans;
	while (a != b) {
		if (a > b) {
			cout << (a - 1) / b << "A";
			a = (a - 1) % b + 1;
		} else {
			cout << (b - 1) / a << "B";
			b = (b - 1) % a + 1;
		}
	}
	cout << "\n";
}
