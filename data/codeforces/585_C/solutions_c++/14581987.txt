#include<iostream>
using namespace std;
int a[10000],b[10000];
int main()
{
    long long n,m,t;
    int sl1=0,sl2=0;
    cin>>n>>m;
    while ((n>1)&&(m>1))
    {
    sl1++;
    a[sl1]=n/m;
    n=n%m;
    if (n<=1) break;
    sl2++;
    b[sl2]=m/n;
    m=m%n;

    }
    if ((n==0) || (m==0)) cout<<"Impossible";
    else
    {
        for (int i=1;i<=max(sl1,sl2);i++)
        {if (a[i]!=0) cout<<a[i]<<"A";
        if (b[i]!=0)cout<<b[i]<<"B";}
        if (n!=m) { if (n>m) cout<<n-m<<"A"; else cout<<m-n<<"B";
        }
    }
}
