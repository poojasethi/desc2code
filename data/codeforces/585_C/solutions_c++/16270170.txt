#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

vector<LL> V;
LL a,b;
int main(){
  scanf("%lld %lld",&a,&b);
  while(a!=b){
    if(a>b){
      V.push_back(((a-1)/b)*2);
      a=(a-1)%b+1;
    }
    else{
      V.push_back(((b-1)/a)*2+1);
      b=(b-1)%a+1;
    }
  }
  if(a>1) printf("Impossible\n");
  else{
    int sz=V.size();
    FI(i,0,sz-1) printf("%lld%c",V[i]/2,'A'+V[i]%2);
    puts("");
  }
  return 0;
}
