#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll gcd(ll a,ll b)
{
ll t;
while(a%b){t=a%b; a=b; b=t;}
return b;
}
int main()
{
ll a,b,g;
cin>>a>>b;
g= gcd(a,b);
if(g==1)
{
while(a>1 || b>1)
{
if(a==1){cout<<b-1<<"B";break;}
else if(b==1){cout<<a-1<<"A";break;}
else if(a>b){cout<<a/b<<"A"; a = a%b;}
else {cout<<b/a<<"B"; b=b%a;}
}
}
else cout<<"Impossible"<<endl;
}
