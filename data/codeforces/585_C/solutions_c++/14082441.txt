#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
#include<cmath>
#include<cstdlib>
using namespace std;
int x=0;
long long ans[3000010],a,b;
long long gcd(long long a,long long b){
	if(b==0) return a;
	ans[x++]=(long long)a/b;
	return gcd(b,a%b);
}
int main(){
	memset(ans,0,sizeof(ans));
	int i,ok=0;
	cin>>a>>b;
	if(a<b) {swap(a,b);ok=1;}
	long long w=gcd(a,b);
	if(w!=1) printf("Impossible\n");
	else{
		ans[x-1]--;
		for(i=0;i<x;i++){
			if(!ans[i]) continue;
			cout<<ans[i];
			printf("%c",'A'+ok);
			ok=(ok^1);
		}
		puts("");
	}
	return 0;
}
