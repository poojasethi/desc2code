#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
void f(uli x,uli y){
   if(x==y)return;
   if(x<y){
      uli q=(y-1)/x;
      cout<<q<<"A";
      f(x,y-q*x);
   }
   else if(y<x){
      uli q=(x-1)/y;
      cout<<q<<"B";
      f(x-q*y,y);
   }
}
int main(){
   uli x,y;
   cin>>x>>y;
   if(__gcd(x,y)!=1)puts("Impossible");
   else f(y,x);
   return 0;
}
