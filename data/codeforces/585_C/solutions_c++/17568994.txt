#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll x,y;
ll gcd(ll a,ll b)
{
    if(a==0)
        return b;
    return gcd(b%a,a);
}
void ans(ll a,ll b)
{
    if(a==1 && b==1) return;
    if(a==1)
    {
        cout<<b-1<<'B';
        return;
    }
    if(b==1)
    {
        cout<<a-1<<'A';
        return;
    }
    if(a<b)
    {
        cout<<b/a<<'B';
        ans(a,b%a);
        return;
    }
    cout<<a/b<<'A';
    ans(a%b,b);
}
main()
{
    cin>>x>>y;
    if(gcd(x,y)==1)
        ans(x,y);
    else
        cout<<"Impossible";
    cout<<'\n';
    return 0;
}
