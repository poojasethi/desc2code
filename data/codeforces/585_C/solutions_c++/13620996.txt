#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
#define ll long long
using namespace std;
void dfs(ll x,ll y){
    if(x==y){
        return;
    }
    if(x>y){
        ll a=(x+y-1)/y-1;
        cout<<a<<"A";
        dfs(x-a*y,y);

    }
    else{
        ll a=(y+x-1)/x-1;
        cout<<a<<"B";
        dfs(x,y-a*x);
    }
}
int main()
{
    //freopen("dd.txt","r",stdin);
    ll x,y;
    cin>>x>>y;
    if(__gcd(x,y)!=1)
        printf("Impossible\n");
    else
    dfs(x,y);

    return 0;
}
