#include <cstdio>
#include <iostream>

using namespace std;

long long A, B;

long long gcd(long long x, long long y)
{
	return y ? gcd(y, x % y) : x;
}

void solve(long long x, long long y, char a, char b)
{
	while (x != 1 || y != 1)
	{
		if (x < y)
		{
			swap(x, y);
			swap(a, b);
		}
		cout << (x - 1) / y << a;
		x -= (x - 1) / y * y;
	}
}

int main()
{
	cin >> A >> B;
	if (gcd(A, B) == 1)
	{
		solve(A, B, 'A', 'B');
	} else puts("Impossible");
}
