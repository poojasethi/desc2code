# include <stdio.h>
# include <iostream>

using namespace std;

long long x,y;

long long gcd(long long x,long long y) {
	long long r;
	while(y!=0) {
		r=x%y;
		x=y;
		y=r;
	}
	return x;
}

int main() {
	cin>>x>>y;
	if(gcd(x,y)!=1) {
		printf("Impossible\n");
		return 0;
	}
	while(x>1||y>1) {
		if (x>y) cout<<(x-1)/y<<'A',x=x%y;
		else cout<<(y-1)/x<<'B',y=y%x;
	}
	cout<<endl;
	return 0;
}
