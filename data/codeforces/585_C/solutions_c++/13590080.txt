#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long LL;
LL a,b;
inline LL gcd(LL a,LL b)
{
    while(b)
    {
        LL t=b;
        b=a%b;
        a=t;
    }
    return a;
}
int main(void)
{
    scanf("%I64d%I64d",&a,&b);
    if(gcd(a,b)!=1)printf("Impossible\n");
    else
    {
        while(a&&b)
        {
            if(a>b)printf("%I64dA",(a-1)/b),a%=b;
            else printf("%I64dB",(b-1)/a),b%=a;
        }
        printf("\n");
    }
    return 0;
}
