#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
ll gcd(ll a,ll b)
{return b?gcd(b,a%b):a;}
void f(ll a,ll b,bool bl)
{
    if(b==1)
    {
        printf("%I64d%c",a-1,65+bl);
        return;
    }
    printf("%I64d%c",a/b,65+bl);
    f(b,a%b,!bl);
}
int main()
{
    ll a,b;
    scanf("%I64d%I64d",&a,&b);
    if(gcd(a,b)!=1)
    {puts("Impossible");return 0;}
    if(a<b)f(b,a,1);
    else f(a,b,0);
    puts("");
}
