#include<cstdio>
#include<algorithm>
using namespace std;
long long x,y;
void Gcd(long long x,long long y){
    if (x==y&&x==1) return;
    if (x>y) printf("%I64dA",y==1?x-1:x/y),Gcd(y==1?1:x%y,y);
    else printf("%I64dB",x==1?y-1:y/x),Gcd(x,x==1?1:y%x);
}
int main(){
    scanf("%I64d%I64d",&x,&y);
    if (__gcd(x,y)>1) return puts("Impossible"),0;
    Gcd(x,y);
}
