#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <queue>
#include <map>
#include <vector>

using namespace std;

typedef long long LL;

LL n,m;
int main()
{
	scanf("%I64d%I64d",&n,&m);
	if (__gcd(n,m)!=1)
	{
		printf("Impossible\n");
		return 0;
	}
	while (n!=m)
	{
		if (n>m)
		{
			printf("%I64dA",(n-1)/m);
			n-=(n-1)/m*m;
		}
		else
		{
			printf("%I64dB",(m-1)/n);
			m-=(m-1)/n*n;
		}
	}
	return 0;
}
