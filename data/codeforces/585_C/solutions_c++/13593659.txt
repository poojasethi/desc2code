#include<bits/stdc++.h>

using namespace std;

int main()
{
    long long x,y;

    cin>>x>>y;
    if(__gcd(x,y)!=1ll)
        cout<<"Impossible";
    else
    while(x!=y)
    {
        if(x>y)
        cout<<(x-1)/y<<'A',x -=(x-1)/y * y;
        else
        cout<<(y-1)/x<<'B',y -=(y-1)/x * x;
    }

    return 0;
}
