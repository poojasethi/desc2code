#include <bits/stdc++.h>
using namespace std;

int main() {
	ios::sync_with_stdio(false);
	long long d, k, a, b, t;
	cin >> d >> k >> a >> b >> t;
	if (k >= d) {
		cout << a * d;
		return 0;
	}
	long long res = k * a;
	d -= k;
	if (t + a * k < b * k) {
		res += (d / k) * (t + a * k);
		d -= (d / k) * k;
		res += min(t + a * d, b * d);
	}
	else
		res += b * d;
	cout << res;
}
