#include <bits/stdc++.h>

using namespace std;


typedef long long ll;
typedef vector<ll> vi;

#define pb push_back

vi pot;

void run(){

	ll d, k, a, b, t;
	cin >> d >> k >> a >> b >> t;


	if(k >= d) cout << a*d << endl;
	else if(t+k*a > k*b) cout << k*a + (d-k)*b << endl;
	else{

		ll cnt = d/k;

		cout << k*cnt*a + (cnt-1)*t + min(t+(d%k)*a, (d%k)*b) << endl;	
	}
}


int main(){

	run();
}
