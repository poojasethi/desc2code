#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	__int64 d,k,a,b,t,sum;
	while(scanf("%I64d%I64d%I64d%I64d%I64d",&d,&k,&a,&b,&t)!=EOF)
	{
		if(d<=k)
			sum=a*d;
		else
		{
			if(a*k+t>=b*k)
				sum=k*a+(d-k)*b;
			else
			{
				__int64 n=d/k;
				__int64 m=d%k;
				sum=a*k*n+t*(n-1);
				sum+=min(t+m*a,m*b);	
			}
		}
	  printf("%I64d\n",sum);
	}
	return 0;
}
