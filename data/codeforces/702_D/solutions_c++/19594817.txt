#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int k,t,a,b;
long long d,x,ans;

int main(){//[0,d-1];
	scanf("%I64d%d%d%d%d",&d,&k,&a,&b,&t);
	ans=a*d+(d-1)/k*t;
	x=(d-1)/k*k;
	ans=min(ans,a*x+(x-1)/k*t+(d-x)*b);
	if (d>k){
		x=k;
		ans=min(ans,a*x+(x-1)/k*t+(d-x)*b);
	}
	printf("%I64d",ans);
	return 0;
}
