#include<bits/stdc++.h>
using namespace std;
# define l long long int
# define rep(i,a,b) for(l i=a;i<b;i++)
int main(){
	l d,k,a,b,t;
	cin>>d>>k>>a>>b>>t;
	if(d<=k){cout<<a*d<<"\n";return 0;}
	if((k*(a-b)+t)>=0) cout<<k*(a-b)+b*d<<"\n";
	else {
		l r=(d/k)-1;
		if((r+2)*k<=d) r++;
		if((r+1)*k>d) r--;
		cout<<min((k*(a-b)+r*(k*(a-b)+t)+b*d),((r+1)*(a*k+t)+(d-(r+1)*k)*a))<<"\n";
	}
	return 0;
}
