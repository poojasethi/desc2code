#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
ll d,a,b,k,t;

ll f(ll x) {
    return (max(0LL,(x-1)/k*t))+d*b+x*(a-b);
}
int main() {
 //   freopen("input.txt","r",stdin);
    cin>>d>>k>>a>>b>>t;
    ll res = min(f(min(d,k)),f(d));
    for(int i = 0; i <= k; ++i)
        if(d>=i) res = min(res,f(d-i));
    cout<<res<<endl;
}
