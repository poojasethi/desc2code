#include <iostream>
using namespace std;
const long long INF=1e15;
int main()
{
  long long d, k, a, b, t;
  cin >> d >> k >> a >> b >> t;

  long long ans;
  if( d <= k )
    ans = d*a;
  else if( t+k*a > k*b )
    ans = k*a + (d-k)*b;
  else
  {
    long long cnt = d/k;
    ans = k*cnt*a + (cnt-1)*t + min(t+(d%k)*a, (d%k)*b);
  }

  cout << ans << endl;
  return 0;
}
