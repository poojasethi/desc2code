#include<stdio.h>
__int64 n,k,a,b,t,last,ans;
int main()
{
	scanf("%I64d%I64d%I64d%I64d%I64d",&n,&k,&a,&b,&t);
	if(n<=k)
		printf("%I64d\n",n*a);
	else if(k*a+t>b*k)
	{
		printf("%I64d\n",k*a+(n-k)*b);
	}
	else
	{
		last=n%k;
		ans=k*a+(n/k-1)*(k*a+t);
		if(last*b<=last*a+t)
			printf("%I64d\n",ans+last*b);
		else
			printf("%I64d\n",ans+last*a+t);
	}
	return 0;
}
