#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	long long int d,k,a,b,t,ans,cnt;
	cin >> d >> k >> a >> b >> t;
	if(d <= k) ans = d*a;
	else if(t + k*a > k*b) ans = k*a + (d-k)*b;
	else 
	{
		cnt = d/k;
		ans = k*cnt*a + (cnt-1)*t + min(t + (d%k)*a, (d%k)*b);
	}
	cout << ans << endl;
	return(0);
}
