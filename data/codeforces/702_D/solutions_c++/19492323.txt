#include <stdio.h>

int main()
{
	long long int a,b,c,d,e;
	scanf("%I64d%I64d%I64d%I64d%I64d",&a,&b,&c,&d,&e);
	if(a<=b) printf("%I64d",a*c);
	else if(b*c+e>=b*d) printf("%I64d",b*c + (a-b)*d);
	else
	{
		long long int k = (a-b)/b;
		long long int s1 = b*c+k*(b*c+e)+(a-b-k*b)*d;
		long long int s2 = b*c+k*(b*c+e)+e+(a-b-k*b)*c;
		printf("%I64d",s1<s2?s1:s2);
	}
}
