#include <stdio.h>
#include <iostream>
using namespace std;

long long a , b , t , d , k , ans = 0;

int main() {
	cin>>d>>k>>a>>b>>t;
	if(a > b) {
		cout<<b*d;
		return 0;
	}
	if(d < k) {
		cout<<a*d;
		return 0;
	}
	ans += a*k;
	d-=k;
	if(a*k+t < b*k) {
		ans += (d/k)*(a*k+t);
		d %= k;
		if(d*b > d*a+t) ans += d*a+t;
		else ans += d*b;
	}
	else ans += b*d;
	cout<<ans;
	return 0;
}
