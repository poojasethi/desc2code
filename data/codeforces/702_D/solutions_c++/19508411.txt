#include<iostream>
#include<cstdio>
#include<cstdlib>
using namespace std;
long long d,k,a,b,t;
int main()
{
	cin>>d>>k>>a>>b>>t;
	if(a>b)
	{
		cout<<d*b;
		return 0;
	}
	if(d<=k)
	{
		cout<<a*d;
		return 0;
	}
	long long ans=a*k;
	d-=k;
	if(t+k*a<b*k)
	{
		ans+=(t+k*a)*(d/k);
		d%=k;
		ans+=min(a*d+t,b*d);
	}
	else ans+=b*d;
	cout<<ans;
	return 0;
}
