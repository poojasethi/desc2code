#include<stdio.h>
#include<iostream>
#include<string>
#include<string.h>
#include<algorithm>
using namespace std;
#define ll long long
int main()
{
    ll d,k,a,b,t,ans;
    cin>>d>>k>>a>>b>>t; ll x=d/k;
    if (k*a+t-k*b>0) if (k<=d) ans=k*a-k*b+d*b;else ans=d*a;
    else if (k>d) ans=d*a;else ans=min(x*(k*a-k*b+t)+d*b-t,x*(k*a+t)+(d-x*k)*a);
    cout<<ans<<endl;
}
