#include<cstdio>
#include<algorithm>

long long d,k,a,b,t;
long long T;

long long time(long long x)
{
	if(x==0) return b*d;

	long long cnt=(x-1)/k;
	return a*x+t*cnt+b*(d-x);
}

int main()
{
	scanf("%I64d%I64d%I64d%I64d%I64d",&d,&k,&a,&b,&t);
	T=b*d;
	for(long long x=0;x<2*k&&x<=d;x++)
	{
		T=std::min(T,time(x));
		T=std::min(T,time(d-x));
	}
	printf("%I64d\n",T);
	return 0;
}
