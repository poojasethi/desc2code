#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int d,k,a,b,t,i,j,p,ans;
	cin>>d>>k>>a>>b>>t;
	if(d<=k)
    cout<<d*a;
	else if(k*a+t>b*k)
    cout<<k*a+(d-k)*b;
	else
	{
		p=d%k;
		if(p*b<=p*a+t)
        cout<<k*a+(d/k-1)*(k*a+t)+p*b;
		else
		cout<<k*a+(d/k-1)*(k*a+t)+p*a+t;
	}
}
