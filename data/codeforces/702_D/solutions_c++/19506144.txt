#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
long long d,k,a,b,t,ans;
int main()
{
	scanf("%I64d%I64d%I64d%I64d%I64d",&d,&k,&a,&b,&t);
	if(d<=k)ans=d*a;
	else if(t+a*k>b*k)ans=a*k+b*(d-k);
	else
	{
		long long c=d/k;
		ans=min((c-1)*(k*a+t)+k*a+b*(d-c*k),(c-1)*(k*a+t)+k*a+t+a*(d-c*k));
	}
	printf("%I64d\n",ans);
	return 0;
}
