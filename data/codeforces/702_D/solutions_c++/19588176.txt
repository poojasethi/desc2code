#include <bits/stdc++.h>

using namespace std;

long long d, k, a, b, t, s, r, res;

int main() {
	cin >> d >> k >> a >> b >> t;
	s = d / k; r = d - s * k;
	res = min(d * b, d * a + s * t);
	if(k <= d) res = min(res, min(s * (k * a + t) - t + r * b, k * a + (d-k) * b));
	cout << res << endl;
	return 0;
}
