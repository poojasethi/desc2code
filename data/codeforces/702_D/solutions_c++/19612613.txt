#include <stdio.h>
#include <algorithm>
using namespace std;
long long d, k, a, b, t,ans;
int main()
{
	scanf("%I64d %I64d %I64d %I64d %I64d", &d, &k, &a, &b, &t);
	if (d <= k)
	{
		printf("%I64d\n", d*a);
		return 0;
	}
	ans = (d - 1) / k * (a*k + t) + min((((d - 1) % k) + 1)*a, (((d - 1) % k)+ 1)*b - t);
	ans = min(ans, a*k + (d - k)*b);
	printf("%I64d\n", ans);
}
