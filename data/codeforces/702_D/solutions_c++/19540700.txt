#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
    ll d,k,a,b,t;
    cin>>d>>k>>a>>b>>t;
    ll ans;
    ans=d*a+((d-1)/k)*t;
    ans=min(ans,d*b);
    ll lx=d/k;
    if(lx==0){
        cout<<ans<<endl;
        return 0;
    }
    ll ans1=(a*k-b*k+t)*lx-t+b*d;
    ans=min(ans,ans1);
    if(lx>1) ans=min(ans,a*k-b*k+b*d);
    cout<<ans<<endl;
    return 0;
}
