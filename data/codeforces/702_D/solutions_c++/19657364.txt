#include<bits/stdc++.h>
using namespace std;
#define ll long long

int main(){
  ll int d;
  ll int k, a, b, t;
  scanf("%lld %lld %lld %lld %lld", &d, &k, &a, &b, &t);
  if(d <= k)
    printf("%lld\n", d*a);
  else if(t + k*a > k*b)
    printf("%lld\n", k*a + (d-k)*b);
  else {
    ll int cnt = d/k;
    printf("%lld\n", k*a*cnt + (cnt-1)*t + min(t+(d%k)*a, (d%k)*b));
  }
  return 0;
}
