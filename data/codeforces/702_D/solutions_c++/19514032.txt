#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	ll d,k,a,b,t;
	scanf("%I64d%I64d%I64d%I64d%I64d",&d,&k,&a,&b,&t);
	if(k<d){
		ll t1 = b*d;
		ll t2 = d/k;
		ll t3 = d%k;
		ll t4 = t2*t+k*t2*a+t3*a;
		ll t5 = (t2-1)*t+k*t2*a+t3*b;
		ll t6 = k*a+(d-k)*b;
		printf("%I64d",min(t1,min(t4,min(t5,t6))));
	}else{
		printf("%I64d",a*d);
	}
	return 0;
}
