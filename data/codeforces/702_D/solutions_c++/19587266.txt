#include <iostream>
#include <cmath>

using namespace std;
typedef long long ll;


int main(){
  ll d,k,a,b,t,n,non_n;
  cin>>d>>k>>a>>b>>t;
  k = min(k,d);
  n = a*k+t - b*k;
  if(n < 0){
    n = d/k;
    cout<< min(n*(a*k+t) - t + b*(d-n*k),n*(a*k+t)+(d-n*k)*a)<<'\n';
  }
  else{
    cout<< n + b*d - t<<'\n';
  }
    
  return 0;
}
