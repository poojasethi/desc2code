#include <iostream>
using namespace std;
#define ll long long int
int main() {
	ll d,k,a,b,t;
	cin>>d>>k>>a>>b>>t;
	
	if(d<=k)
	{
	    cout<<d*a<<endl;
	    return 0;
	}
	ll ans=k*a;
	ll s=k*a+t;
	
	if(s>k*b)
	s=k*b;
	
	d-=k;
	
	ans+=(d/k)*s;
	d=d%k;
	
	ll temp=d*a+t;
	if(temp>d*b)
	temp=d*b;
	
	ans+=temp;
	
	cout<<ans<<endl;
	return 0;
}
