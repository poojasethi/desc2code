int main()
{
 long long d,a,b,t,k,ans=0;
 cin>>d>>k>>a>>b>>t;
 //long long ans;
 if (k>=d) {
  cout<<d*a;
  return 0;
 }
 if (k*a+t<k*b) {
  long long col=d/k;
  ans=col*(k*a+t);
  ans-=t;
  d%=k;
  ans+=min(d*b,t+d*a);
  cout<<ans;
  return 0;
 }

 ans=k*a+(d-k)*b;

    return 0;
