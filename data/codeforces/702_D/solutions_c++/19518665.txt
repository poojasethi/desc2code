#include <iostream>
#include <cmath>
#include <cstdlib>

using namespace std;

typedef long long int ll;

int main()
{
  ll d,k,a,b,t;
  cin>>d>>k>>a>>b>>t;
  if (d<=k) {
    cout<<a*d<<endl;
    exit(0);
  }
  ll veces=1;
  if (a*k+t-b*k<=0)
    veces=d/k;
  ll sol=veces*k*(a-b)+(veces-1)*t+d*b;
  sol=min(sol,a*d+d/k*t);
  cout<<sol<<endl;
}
