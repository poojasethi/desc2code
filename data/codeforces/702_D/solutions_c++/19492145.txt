#include <bits/stdc++.h> 
using namespace std;

typedef long long ll;

int main () {
	
	ll d, k, a, b, t;
	cin>>d>>k>>a>>b>>t;
 
	ll s = d/k, rem = d%k;
 
	ll foot = (k < d) ?k*a + (d-k)*b :d * a;
	ll car = (k < d) ?s * t + d * a :d * a;
	ll lastOnFoot = (k < d) ?(s-1) * t + (s * k) * a + rem * b :d * a;
	
	cout<<min(foot, min (car, lastOnFoot))<<endl;
	
	return 0;
}
