#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
ll D, K, A, B, T, t, ans;

int main() {
	cin >> D >> K >> A >> B >> T;
	ans += A * min(D, K);
	D = max(D - K, 0LL);
	t = min(A * K + T, B * K);
	ans += D / K * t;
	D %= K;
	ans += min(A * D + T, B * D);
	cout << ans << endl;
	return 0;
}
