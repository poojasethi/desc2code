#include <iostream>

using namespace std;
typedef long long ll;

int main(){
	ll d, k, a, b, t;
	cin >> d >> k >> a >> b >> t;
	if(d <= k){
		cout << d*a << endl;
		return 0;
	}
	ll re = d%k;
	ll time = a*k;
	d -= k;
	if(k*a+t < k*b){
		ll m = k*a+t;
		time += (d/k) * m;
		time += min(re*a + t, re*b);
	}
	else
		time += d*b;
	cout << time << endl;
	return 0;
}
