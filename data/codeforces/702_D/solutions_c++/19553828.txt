#include <iostream>
#include <cstdio>
#define ll long long
using namespace std;
ll d,k,a,b,t;
int main()
{
    #ifdef LOCAL
    freopen("in","r",stdin);
    #endif // LOCAL
    cin>>d>>k>>a>>b>>t;
    ll cnt=d/k;
    if(d<=k)cout<<d*a<<endl;
    else if(k*a-k*b+t>=0)cout<<k*a-k*b+d*b<<endl;
    else cout<<min(cnt*(k*a-k*b+t)+d*b-t,k*cnt*a+t*cnt+(d%k)*a)<<endl;
    return 0;
}
