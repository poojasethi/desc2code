#include <stdio.h>

long long int d, k, a, b, t;

int main()
{
	scanf("%lld%lld%lld%lld%lld", &d, &k, &a, &b, &t);
	if(d <= k)
		printf("%lld\n", a * d);
	else if(b * k <= t + a * k)
		printf("%lld\n", b * (d - k) + a*k);
	else if(b * (d%k) < t + a * (d%k))
		printf("%lld\n", a * (d/k)*k + t * (d/k - 1) + b * (d%k));
	else
		printf("%lld\n", a * d + t * (d/k));
}
