total_dist, car_break_dist, car_rate, walk_rate, repair_time = [int(x) for x in raw_input().split()]

first_run_dist = min(total_dist, car_break_dist)
rem_dist = total_dist - first_run_dist
time_spent = car_rate * first_run_dist

def calc_best_travel(distance):
  car_time = repair_time + distance * car_rate
  walk_time = walk_rate * distance
  return min(car_time, walk_time)

num_segments = rem_dist / car_break_dist
time_spent += num_segments * calc_best_travel(car_break_dist)
time_spent += calc_best_travel(rem_dist % car_break_dist)

print time_spent
