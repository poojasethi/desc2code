import sys 
import math 

linea = sys.stdin.readline()
par = linea.split(' ')
d = int(par[0])
k = int(par[1])
a = int(par[2])
b = int(par[3])
t = int(par[4])

suma = a * min(k, d)
camina = False
d = d - k

if (t + a * k > b * k): camina = True

if (d > 0):
    if (camina): suma = suma + b * d
    else:
        vr = math.ceil(d / k)
        suma = suma + vr * (t + a * k)

if (camina == False and d > 0):
    d = d % k
    suma = suma + min(t + a * d, b * d);

print(int(suma))
