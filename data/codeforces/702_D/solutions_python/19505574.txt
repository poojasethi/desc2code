import math, sys
from sys import stdin

distance, car_range, car_speed, man_speed, repair_time = map(int, stdin.readline().split(' '))
car_time_per_shift = car_range * car_speed + repair_time
walking_is_faster = man_speed <= car_time_per_shift / float(car_range)
if walking_is_faster:
  print min(car_range, distance) * car_speed + max(0, (distance - car_range)) * man_speed
  sys.exit()
car_shifts = math.floor(max(0, (distance - car_range)) / car_range)
remaining_distance = max(0, distance - (car_shifts + 1) * car_range)
remaining_time = min(remaining_distance * car_speed + repair_time, remaining_distance * man_speed)
print min(car_range, distance) * car_speed + car_shifts * car_time_per_shift + remaining_time
