"""C. Vasya and Basketball"""
n = input()
nscore = sorted([int(x) for x in raw_input().split()])
m = input()
mscore = sorted([int(x) for x in raw_input().split()])

a = 3 * n
b = 3 * m
nindex, mindex = 0, 0
tempa = a
tempb = b
for d in mscore:
	while nindex < len(nscore) and nscore[nindex] <= d:
		tempa -= 1
		nindex += 1
	while mindex < len(mscore) and mscore[mindex] <= d:
		tempb -= 1
		mindex += 1
	if tempa - tempb > a - b:
		a = tempa
		b = tempb
print str(a) + ':' + str(b)
