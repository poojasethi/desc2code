R=lambda:sorted(map(int,raw_input().split()))
n=input()
a=R()
m=input()
b=R()
i=j=0
x=ap=3*n
y=bp=3*m
while i<n and j<m:
	k=a[i] if a[i]<=b[j] else b[j]
	while i<n and a[i]==k: i+=1;x-=1
	while j<m and b[j]==k: j+=1;y-=1
	if x-y>ap-bp: ap,bp=x,y
if 2*n-2*m>ap-bp: ap,bp=2*n,2*m
print str(ap)+':'+str(bp)
