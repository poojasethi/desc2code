n=input()
l1=map(int,raw_input().split())
l1.sort()
m=input()
l2=map(int,raw_input().split())
l2.sort()
a=acur=2*n
b=bcur=2*m
m-=1
for i in range(n-1,-1,-1):
    d=l1[i]-1
    acur+=1
    while m!=-1 and l2[m]>d:
        bcur+=1
        m-=1
    if acur-bcur>=a-b:
        a,b=acur,bcur
print str(a)+":"+str(b)
