an = int(raw_input())
a = map(int, raw_input().split())
bn = int(raw_input())
b = map(int, raw_input().split())
a.sort(reverse=True)
b.sort(reverse=True)

ai = bi = 0
fa = sa = 2*len(a)
fb = sb = 2*len(b)
while ai<len(a):
    sa += 1
    while bi<len(b) and b[bi] >= a[ai]:
        sb += 1
        bi += 1
    if sa-sb >= fa-fb:
        fa, fb = sa, sb
    ai += 1
print str(fa)+':'+str(fb)
