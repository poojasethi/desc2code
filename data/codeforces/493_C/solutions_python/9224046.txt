#!/usr/bin/env python
#-*- coding:utf-8 -*-

lst = []


def process_arr(arr, tag):
    for d in arr:
        lst.append((d, tag))

N1 = int(raw_input())
arr = map(int, raw_input().split())
process_arr(arr, "1")

N2 = int(raw_input())
arr = map(int, raw_input().split())
process_arr(arr, "2")

lst.sort()

max_d = -99999999999999
max_a = 0
ab = None

def check():
    global max_d, max_a, ab
    s1 = n1 * 2 + (N1 - n1) * 3
    s2 = n2 * 2 + (N2 - n2) * 3
    diff = s1 - s2
    # print ":", s1, s2, "|", diff, "...", n1, n2
    if diff > max_d:
        max_d = diff
        max_a = s1
        ab = s1, s2
    elif diff == max_d:
        if s1 > max_a:
            max_a = s1
            ab = s1, s2
n1 = 0
n2 = 0
check()
for i in xrange(len(lst)):
    d, tag = lst[i]
    if tag == "1":
        n1 += 1
    else:
        n2 += 1

    if i + 1 < len(lst):
        dnext, _ = lst[i + 1]
    else:
        dnext = None

    if d != dnext:
        check()

print "%d:%d" % ab
