n1 = int(raw_input().strip())
c1 = [int(x) for x in raw_input().strip().split()]
n2 = int(raw_input().strip())
c2 = [int(x) for x in raw_input().strip().split()]

c = [(x, 1) for x in c1]+[(x, 2) for x in c2]
c.sort(key=lambda x: x[0])



a = 3*n1
b = 3*n2
a_r = a
b_r = b

m = a - b

for p in c:
    if p[1]==1:
        a-=1
    else:
        b-=1
    if a-b > m:
        a_r = a
        b_r = b
        m = a-b



print "{0}:{1}".format(a_r,b_r)
