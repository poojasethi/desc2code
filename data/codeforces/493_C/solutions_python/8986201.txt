f = lambda: map(int, raw_input().strip().split())
n, a, m, b = f()[0], f(), f()[0], f()

a.sort()
b.sort()

x = x2 = n * 2
y = y2 = m * 2
m -= 1

for i in range(n - 1, -1, -1):
  d = a[i] - 1
  x2 += 1
  while m > -1 and b[m] > d:
    y2 += 1
    m -= 1
  if x2 - y2 >= x - y:
    x, y = x2, y2

print '%d:%d' % (x, y)
