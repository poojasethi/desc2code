import sys
import sets

def readLine():
    return sys.stdin.readline().strip()

def upper_bound(arr, key):
    lower = 0; upper = len(arr) - 1; ans = len(arr)
    while (lower <= upper) :
        mid = (lower + upper) / 2;
        if key > arr[mid]:
            lower = mid + 1;
        else:
            upper = mid - 1;
            ans = mid;
    return ans

def score(index, size):
    return index*2 + (size-index)*3

nThrowA = int(readLine())
a = sorted([int(v) for v in readLine().split()])
nThrowB = int(readLine())
b = sorted([int(v) for v in readLine().split()])
dif = set(a+b)
maxA = -float('inf'); maxB = 0
for d in dif:
    for inc in [-1,1]:
        indA = upper_bound(a, d+inc)
        scoreA = score(indA, len(a))
        indB = upper_bound(b, d+inc)
        scoreB = score(indB, len(b))
        if scoreA-scoreB > maxA - maxB:
            (maxA,maxB) = (scoreA,scoreB)
        elif scoreA-scoreB == maxA - maxB and scoreA > maxA:
            (maxA,maxB) = (scoreA,scoreB)
