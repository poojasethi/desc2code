

from sys import stdin

import fileinput

inp = []

count1 = int(stdin.readline()[:-1])
inp1 = stdin.readline()[:-1]
inp1 = map(lambda x: (int(x), 1), inp1.split())
#inp1 = sorted(inp1, key = lambda x: x[0])

count2 = int(stdin.readline()[:-1])
inp2 = stdin.readline()[:-1]
inp2 = map(lambda x: (int(x), 2), inp2.split())
#inp2 = sorted(inp2, key = lambda x: x[0])

inp = inp1 + inp2
inp = sorted(inp, key = lambda x: x[0], reverse=True)

firstThrows = 0
secondThrows = 0
firstThrowsAtMaxDiff = 0
secondThrowsAtMaxDiff = 0
maxDiff = 0
for idx in xrange(len(inp)):
	i = inp[idx]
	if i[1] == 1:
		firstThrows += 1
	elif i[1] == 2:
		secondThrows += 1
	if firstThrows - secondThrows >= maxDiff and (idx == len(inp)-1 or inp[idx+1][0] != inp[idx][0]):
		maxDiff = firstThrows - secondThrows
		firstThrowsAtMaxDiff = firstThrows
		secondThrowsAtMaxDiff = secondThrows

firstScore = firstThrowsAtMaxDiff*3 + (firstThrows - firstThrowsAtMaxDiff)*2
secondScore = secondThrowsAtMaxDiff*3 + (secondThrows - secondThrowsAtMaxDiff)*2

print str(firstScore) + ":" + str(secondScore)
