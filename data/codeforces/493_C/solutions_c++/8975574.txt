#include <cstdio>
#include <algorithm>
using namespace std;
pair<int, int> a[400005];
int main() {
	int n, m, x;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &x);
		a[i].first = x;
		a[i].second = 0;
	}
	scanf("%d", &m);
	for (int i = n; i < n + m; ++i) {
		scanf("%d", &x);
		a[i].first = x;
		a[i].second = 1;
	}
	sort(a, a + n + m);
	int ansa = 2 * n, ansb = 2 * m, sa = 2 * n, sb = 2 * m;
	for (int i = n + m - 1; i >= 0; --i) {
		if (a[i].second == 0) ++sa;
		else ++sb;
		if (sa - sb >= ansa - ansb) {
			ansa = sa;
			ansb = sb;
		}
	}
	printf("%d:%d\n", ansa, ansb);
	return 0;
}
