#include<iostream>
#include<algorithm>
#include<utility>
using namespace std;

int main(){
    pair<int,int> a[400000]; int n,m,pa,pb,maxa,maxb;
    cin >> n;
    for(int i=0; i<n; i++){cin >> a[i].first; a[i].second=1;}
    cin >> m;
    for(int i=n; i<n+m; i++){cin >> a[i].first; a[i].second=2;}
    stable_sort(a,a+n+m);
    pa=3*n; pb=3*m; maxa=pa; maxb=pb;
    for(int i=0; i<n+m; i++){
        if(a[i].second==1)pa--;
        if(a[i].second==2)pb--;
        if(a[i+1].first==a[i].first)continue;
        if(pa-pb>maxa-maxb){maxa=pa; maxb=pb;}
    }
    cout << maxa << ":" << maxb;
    return 0;
}
