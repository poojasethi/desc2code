#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int maxn=200010;
int a[maxn],b[maxn],n,m;
ll x,y,A,B;
int main()
{
    ios_base::sync_with_stdio(0);cin.tie(0);
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    cin>>m;
    for(int i=0;i<m;i++) cin>>b[i];
    sort(a,a+n);
    sort(b,b+m);
    int j=m-1;
    A=n;
    B=m;
    for(int i=n-1;i>=0;i--)
    {
        while(j>=0 && a[i]<=b[j]) j--,y++;
        x++;
        if(x-y>=A-B) A=x,B=y;
    }
    if(A<B) A=B=0;
    cout<<(ll)n*2+A<<":"<<(ll)m*2+B<<endl;
    return 0;
}
