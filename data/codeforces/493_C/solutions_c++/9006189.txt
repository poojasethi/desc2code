#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
main()
{
    vector <ll> a(200000),b(200000);
	ll n; cin >> n;
	for(int x=0;x<n;++x) cin >> a[x];
	int m; cin >> m;
	for(int x=0;x<m;++x) cin >> b[x];
	sort(a.begin(),a.begin()+n);
	sort(b.begin(),b.begin()+m);
	ll tonga=2*n,tongb=2*m,u=2*n,v=m*2;
	for(int x=n-1,y=m-1;x>=0;--x)
	{
		u++;
		for(;y>=0&&b[y]>=a[x];--y) v++;
		if(u-v>=tonga-tongb) tonga=u,tongb=v;
	}
	cout<<tonga<<":"<<tongb<<endl;
}
