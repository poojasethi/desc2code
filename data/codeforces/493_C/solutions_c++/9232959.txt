#include<iostream>
#include<set>
#include<map>
using namespace std;
map<long long,long long> a;map<long long,long long> b;
set<long long> zong2;
long long n,m,x,asum,bsum,ajb,asum2,bsum2;
int main()
{
	cin>>n;for(long long i=1;i<=n;i++){cin>>x;a[x]++;zong2.insert(x);}
	cin>>m;for(long long i=1;i<=m;i++){cin>>x;b[x]++;zong2.insert(x);}
	asum=asum2=n*3;bsum=bsum2=m*3;ajb=(n-m)*3;
	for(set<long long>:: iterator it=zong2.begin();it!=zong2.end();it++)
	{
		if(a[*it])asum-=a[*it];
		if(b[*it])bsum-=b[*it];
		if(ajb<asum-bsum){ajb=asum-bsum;asum2=asum;bsum2=bsum;}
	}
	cout<<asum2<<":"<<bsum2;
	return 0;
}
