#include <bits/stdc++.h>
using namespace std;
int i,j,m,n,x,y,a,b,c[200000],d[200000];
main(){
    for(cin>>n;i<n;i++)cin>>c[i];sort(c,c+n);
    for(cin>>m;j<m;j++)cin>>d[j];sort(d,d+m);
    x=a=2*n;y=b=2*m;
    for(i=n-1,j=m-1;i>=0;i--){
            a++;
            for(;j>=0&&d[j]>=c[i];j--)b++;
        if(a-b>=x-y)x=a,y=b;
    }
    cout<<x<<':'<<y;
}
