#include <algorithm>
#include <cstdio>
using namespace std;
const int N = 200005;
int n, m, a[N], b[N];
int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) scanf("%d", a+i);
	scanf("%d", &m);
	for (int i = 0; i < m; i++) scanf("%d", b+i);
	sort(a, a+n);
	sort(b, b+m);
	int ans1 = 2*n, ans2 = 2*m;
	for (int i = 0; i < n; i++) {
		int t1 = 3*n-i;
		int t = lower_bound(b, b+m, a[i])-b;
		int t2 = 3*m-t;
		if ((t1-t2 > ans1-ans2)||((t1-t2 == ans1-ans2)&&(ans1 < t1))) {
				ans1 = t1;
				ans2 = t2;
		}
	}
	printf("%d:%d\n", ans1, ans2);
	return 0;
}
