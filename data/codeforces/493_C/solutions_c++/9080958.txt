#include <iostream>
#include <algorithm>
using namespace std;

int A[200000], B[200000];
int main(){
    int n, m;
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
        cin >> A[i];
    }

    cin >> m;
    for (int i = 0; i < m; ++i)
    {
        cin >> B[i];
    }

    sort(A, A+n);
    sort(B, B+m);

    int sa, sb, res_a, res_b;
    sa = res_a = n * 3;
    sb = res_b = m * 3;
    for (int i = 0, j = 0; i < m; ++i){
        --sb;
        for (; j < n && A[j] <= B[i]; ++j) --sa;
        if (sa-sb > res_a-res_b) res_a=sa, res_b=sb;
    }

    cout << res_a << ':' << res_b;
    return 0;
}
