#include <bits/stdc++.h>
using namespace std;
int a[200001],b[200001];

int main()
{
    int n,m;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    cin>>m;
    for(int i=0;i<m;i++)
        cin>>b[i];
    sort(b,b+m);
    int t1,t2=0,ans=3*n-3*m,x=0,y=0;
    for(t1=0;t1<n&&t2<m;t1++)
    {
        while(t2<m&&b[t2]<a[t1]) t2++;
        if(ans<3*n-t1-(3*m-t2)) {ans=3*n-t1-(3*m-t2);x=t1;y=t2;}
    }
    if(2*(n-m)>ans)
    {
    	ans=2*(n-m);
    	x=n; y=m;
    }
    cout<<3*n-x<<":"<<3*m-y<<endl;
    return 0;
}
