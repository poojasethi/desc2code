#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;

int main()
{//freopen("i.txt","r",stdin);

int a[200005],b[200005];
int n; cin>>n;
for (int i=0;i<n;i++) cin>>a[i];
sort(a,a+n);
int m; cin>>m;
for (int i=0;i<m;i++) cin>>b[i];
sort(b,b+m);
int s1=n*2, s2=m*2;
int a1=s1,  b1=s2;
for (int i=n-1,j=m-1;i>=0;i--)
{a1++;
  for (;j>=0 && b[j]>=a[i];j--)
    b1++;
 if (a1-b1>=s1-s2) {s1=a1; s2=b1;}
//cout<<i<<"    "<<j<<"  "<<a1<<":"<<b1<<endl;;
}
cout<<s1<<":"<<s2;
return 0;
}
