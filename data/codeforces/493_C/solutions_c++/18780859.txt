#include <bits/stdc++.h>

#define NN 200010

using namespace std;

int a[NN], b[NN];

int main (){
	
	int N,M,i,j;
	
	scanf("%d",&N);
	for(i=0; i<N; i++) scanf("%d",a+i);
	sort(a, a+N);
	
	scanf("%d",&M);
	for(i=0; i<M; i++) scanf("%d",b+i);
	sort(b,b+M);
	
	int mx=-100000000, s1, s2, A, B;
	
	for(i=0; i<=N; i++){
		s1=i*2+(N-i)*3;
		if(i<N) j=upper_bound(b, b+M, a[i]-1)-b;
		else j=M;
		s2=j*2+(M-j)*3;
		if(mx < s1-s2) {
			mx=s1-s2;
			A=s1;
			B=s2;
		}
	}
	
	printf("%d:%d\n",A,B);
	return 0;
}
