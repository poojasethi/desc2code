#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long n,m,i,j,ta,tb,A,B;

    cin>>n;
    vector<long long> a(n);
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a.begin(),a.end());

    cin>>m;
    vector<long long> b(m);
    for(i=0;i<m;i++)
        cin>>b[i];
    sort(b.begin(),b.end());

    A=2*n; B=2*m;

    for(i=0;i<n;i++)
    {
        ta=2*i+3*(n-i);
        j=upper_bound(b.begin(),b.end(),a[i]-1)-b.begin();
        tb=2*j+3*(m-j);
        if( ta-tb > A-B || ( ta-tb==A-B && ta>A ) ) { A=ta; B=tb; }
    }

    cout << A << ':' << B ;
    return 0;
}
