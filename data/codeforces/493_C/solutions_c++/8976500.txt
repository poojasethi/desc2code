#include <iostream>
#include <algorithm>

using namespace std;

typedef long long LL;

const int N = 200005;

LL a[N],b[N];

int main()
{
    int n;cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    int m;cin>>m;
    for(int i=0;i<m;i++) cin>>b[i];
    sort(a,a+n);sort(b,b+m);
    int t1=n*2,t2=m*2;
    int ma=t1,mb=t2;
    for (int i=n-1,j=m-1;i>=0;--i)
    {
        ++t1;
        while (j>=0 && b[j] >= a[i]) ++t2,--j;
        if (t1 - t2 >= ma - mb) ma=t1,mb=t2;
    }
    cout << ma << ':' << mb << endl;
    return 0;
}
