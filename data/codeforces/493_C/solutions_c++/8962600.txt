#include<iostream>
#include<algorithm>
using namespace std;
#define N 200000
int a[N], b[N];
int main() {
	int n, m;
	cin >> n;
	for (int i = 0; i < n; ++i)
		cin >> a[i];
	cin >> m;
	for (int i = 0; i < m; ++i)
		cin >> b[i];
	sort(a, a+n);
	sort(b, b+m);
	int ma = 2*n, mb = 2*m;
	for (int i = 0; i < n; ++i) {
		int pa = upper_bound(a, a+n, a[i]-1) - a,
			pb = upper_bound(b, b+m, a[i]-1) - b;
		pa = pa*2 + (n-pa)*3;
		pb = pb*2 + (m-pb)*3;
		if (pa - pb > ma - mb)
			ma = pa, mb = pb;
		else if (pa - pb == ma - mb && pa > ma)
			ma = pa, mb = pb;
	}
	cout << ma << ":" << mb;
}
