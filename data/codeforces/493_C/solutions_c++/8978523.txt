#include <bits/stdc++.h>
typedef long long int ll;
using namespace std;
map<ll,ll> ma,mb;
ll a,b;
int main(){
    int n,m,dis;
    cin >> n;
    for(int i=0;i<n;i++)
        cin>>dis,a+=3,ma[dis]++;
    cin >> m;
    for(int i=0;i<m;i++)
        cin>>dis,b+=3,mb[dis]++,ma[dis];
    ll aa=a,bb=b;
    for(map<ll,ll>::iterator it=ma.begin();it!=ma.end();it++){
        int d = it->first;
        a -= ma[d];
        b -= mb[d];
        if(a-b > aa-bb){
            aa = a;
            bb = b;
        }
    }
    cout << aa << ":" << bb << endl;
    return 0;
}
