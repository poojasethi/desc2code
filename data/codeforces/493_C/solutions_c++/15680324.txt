#include<stdio.h>
#include<algorithm>
using namespace std;
int a[200005],b[200005];
int main(){
	long long n,m,i,j;
	scanf("%I64d",&n);
	for(i = 0; i<n; i++) scanf("%d",&a[i]);
	scanf("%I64d",&m);
	for(i = 0; i<m; i++) scanf("%d",&b[i]);
	sort(a,a+n);
	sort(b,b+m);
	j = m - 1;
	long long x = 0,y = 0,a1 = n,b1 = m; 
	for(i = n-1; i>=0; i--){
		while(j>=0&&a[i]<=b[j]){
			y++;
			j--;
		}
		x++;
		if(x-y>=a1-b1){
			a1 = x;
			b1 = y;
		}
	}
	if(a1<b1) a1 = b1 = 0;
	printf("%I64d:%I64d\n",(2*n+a1),(2*m+b1));
	return 0;
}
