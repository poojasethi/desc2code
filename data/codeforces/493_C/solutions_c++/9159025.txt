#include <cstdio>
#include <algorithm>
using namespace std;

int a[200003];
int b[200003];

int main() {
	int n, m;
	scanf("%d", &n);
	for (int i=0; i < n; ++i)
		scanf("%d", &a[i]);
	scanf("%d", &m);
	for (int i=0; i < m; ++i)
		scanf("%d", &b[i]);
	sort(a, a+n);
	sort(b, b+m);
	int pb = m-1;
	int suma=2*n, sumb=2*m;
	for (int i=n-1; i >= 0; --i) {
		for (; pb >= 0 && a[i] <= b[pb]; --pb);
	
		int ta = (n-i)*3 + i*2;
		int tb = (m-pb-1)*3 + (pb+1)*2;
		if (suma-sumb < ta-tb || (suma-sumb == ta-tb && suma < ta)) {
			suma = ta;
			sumb = tb;
		}
	}
	printf("%d:%d", suma, sumb);
	return 0;
}
