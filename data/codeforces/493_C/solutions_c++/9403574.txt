#include<cstdio>
#include<algorithm>
using namespace std;

int a[200200],b[200200];

int main()
{
	int l1,l2;
	scanf("%d",&l1);
	for(int i=1;i<=l1;++i)scanf("%d",a+i);
	scanf("%d",&l2);
	for(int i=1;i<=l2;++i)scanf("%d",b+i);
	sort(a+1,a+1+l1);
	sort(b+1,b+1+l2);
	a[0]=a[1]-1;
	a[l1+1]=b[l2+1]=2020202020;
	int a1 = l1 * 3 + 1, b1 = l2 * 3, pa = 0, pb = 1, ansa = a1 - 1, ansb = b1, ma = ansa - ansb;
	while(pa <= l1){
		do{
			++pa;--a1;
		}while(a[pa]==a[pa-1]);
		while(b[pb]<a[pa])++pb,--b1;
		if(a1-b1>ma){
			ma=a1-b1;
			ansa=a1;
			ansb=b1;
		}
	}
	printf("%d:%d\n",ansa,ansb);
	return 0;
}
