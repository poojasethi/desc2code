#include <stdio.h>
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
int n,m,l,r,x,y,a[200005],b[200005];
int main( ){
	scanf("%d",&n);
	for (int i=0;i<n;i++)
		scanf("%d",&a[i]);
	scanf("%d",&m);
	for (int j=0;j<m;j++)
		scanf("%d",&b[j]);
	sort(a,a+n);
	x=l=2*n;
	sort(b,b+m);
	y=r=2*m;
	for (int i=n-1,j=m-1;i>=0;i--){
		l++;
		for (;j>=0 && b[j]>=a[i];j--) r++;
		if (l-r>=x-y) x=l,y=r;
	}
	while (l<n) l++;
	printf("%d:%d\n",x,y);
	return 0;
}
