#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll a[200010],b[200010];

int main()
{
	ios_base::sync_with_stdio(0);
	ll n,m,x,ans=INT_MIN,ans1,ans2;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	cin>>m;
	for(int i=0;i<m;i++)
	{
		cin>>b[i];
	}
	sort(b,b+m);
	x=0;
	for(int i=0;i<n;i++)
	{
		while(x<m&&b[x]<=a[i]-1)
			x++;
		if(ans<(((n-i)*3)+(i*2)-(x*2)-((m-x)*3)))
		{
			ans=((n-i)*3)+(i*2)-(x*2)-((m-x)*3);
			ans1=((n-i)*3)+(i*2);
			ans2=(x*2)+((m-x)*3);
		}
		while(a[i]==a[i+1])
			i++;
	}
	if(ans<(n*2)-(m*2))
	{
		ans1=(n*2);
		ans2=(m*2);
	}
	cout<<ans1<<":"<<ans2<<"\n";
	return 0;
}
