#include<bits/stdc++.h>
using namespace std;
int n,m,sa,sb,ma,mb,mx;
int a[200005],b[200005];
int main() {
	int i,j,k;
	cin>>n;
	for (i=0; i<n; i++) scanf("%d",&a[i]);
	cin>>m;
	for (j=0; j<m; j++) scanf("%d",&b[j]);
	sort(a, a+n);
	sort(b, b+m);
	a[n] = b[m] = 2087654321;
	ma = sa = 3 * n;
	mb = sb = 3 * m;
	mx = sa - sb;
	for (i=0,j=0; i<n||j<m;) {
		if (a[i] < b[j]) {
			sa--;
			i++;
		} else if (a[i] > b[j]) {
			sb--;
			j++;
		} else {
			sa--, sb--;
			i++, j++;
		}
		if (sa-sb > mx) {
			mx = sa - sb;
			ma = sa;
			mb = sb;
		}
	}
	cout<< ma<< ':'<< mb;
}
