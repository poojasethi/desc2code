#include <iostream>
#include <cstdio>
#include<cstring>
#include<algorithm>
#define N 400100
using namespace std;
int a[N],b[N],n,m;
int main() 
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	scanf("%d",&m);
	for(int i=1;i<=m;i++)scanf("%d",&b[i]);
	sort(a+1,a+1+n);sort(b+1,b+1+m);
	int l=1;
	int x=3*n,y=3*m;
	for(int i=1;i<=n+1;i++)
	{
		int t=a[i]-1;
		if(i==n+1){t=1000000000;}
		while(l<=m&&b[l]<=t)l++;
		int t1=2*(i-1)+3*(n-i+1),t2=2*(l-1)+3*(m-l+1);
		if(t1-t2>x-y||(t1>x&&t1-t2==x-y))
		{x=t1;y=t2;}
	}
	
	printf("%d:%d\n",x,y);
	return 0;
}
