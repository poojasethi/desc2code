#include<iostream>
#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
map <long long , long long> a,b;
map <long long , long long> :: iterator it;
int main()
{
    int m,val,aa,bb,ma,mb;
    cin>>m;
    for(int i=0; i<m; i++) cin>>val, a[val]++;
    aa = ma = m*3;
    cin>>m;
    for(int i=0; i<m; i++) cin>>val, b[val]++, a[val];
    bb = mb = m*3;
    for(it=a.begin(); it!=a.end(); ++it){
        int d = it->first;
        aa-=a[d]; bb-=b[d];
        if(ma-mb < aa-bb) ma = aa, mb = bb;
    }
    cout<<ma<<":"<<mb<<endl;
    return 0;
}
