#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
map<LL,LL> A,B;
int main() {
    int n,m,dis;
    LL s_a=0,s_b=0;
    cin>>n;
    for(int i=0;i<n;i++) cin>>dis,s_a+=3,A[dis]++;
    cin>>m;
    for(int i=0;i<m;i++) cin>>dis,s_b+=3,B[dis]++,A[dis];
    LL a=s_a,b=s_b;
    for(map<LL,LL>::iterator it=A.begin();it!=A.end();it++) {
        int d=it->first;
        s_a-=A[d];
        s_b-=B[d];
        if(s_a-s_b>a-b) {
            a=s_a; b=s_b;
        }
    }
    cout<<a<<":"<<b<<endl;
    return 0;
}
