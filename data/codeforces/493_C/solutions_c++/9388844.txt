#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0), cout.tie(0);
	pair<int,int> arr[400007];
	int n, m, s=0, a=0, b=0, A, B;
	cin>>n;
	for(int i=0;i<n;++i)
		cin>>arr[i].first, arr[i].second=0;
	cin>>m;
	for(int i=n;i<(n+m);++i)
		cin>>arr[i].first, arr[i].second=1;
	a=n<<1, b=m<<1, s=n+m, A=a, B=b;
	sort(arr,arr+s);

	for(int i=s-1;i>=0;){
		int cur=arr[i].first;
		while((i>=0)&&(arr[i].first==cur))
			((arr[i].second==0)?++a:++b), --i;
		if((a-b)>(A-B)) A=a, B=b;
		else if(((a-b)==(A-B))&&(a>A)) A=a, B=b;
	}

	cout<<A<<':'<<B;
	return 0;
}
