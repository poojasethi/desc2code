#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

ll a[222222];
ll b[222222];

int main() {

	int n, m;
	
	cin >> n;
	for ( int i = 0; i < n; i ++ ) cin >> a[i];
	
	cin >> m;
	for ( int i = 0; i < m; i ++ ) cin >> b[i];
	
	sort ( a, a + n );
	sort ( b, b + m );
	
	int j = m - 1;
	ll A = n * 2, B = m * 2;
	int best = A - B;
	int ba = A, bb = B;
	
	for ( int i = n - 1; i >= 0; i -- ) {
		A ++;
		while ( j >= 0 && b[j] >= a[i] ) {
			B ++;
			j --;
		}
		if ( A - B >= best ) {
			best = A - B;
			ba = A;
			bb = B;
		}
	}
	
	cout << ba << ":" << bb << '\n';

	return 0;
}
