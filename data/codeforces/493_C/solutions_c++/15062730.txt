#include <iostream>
#include <algorithm>

using namespace std;
long long a[200000],b[200000];

int main()
{
    long long n,m;
    long long   x=0,y=0,a1,a2;
    cin>>n;
    a1=n;
    for(int i=0;i<n;i++)cin>>a[i];
    cin>>m;
    a2=m;
    for(int i=0;i<m;i++)cin>>b[i];
    sort(a,a+n);
    sort(b,b+m);
    int f=m-1;
    for(int i=n-1;i>-1;i--)
    {
        while(f>=0&&a[i]<=b[f])
        {
            f--;
            y++;
        }
        x++;
        if(x-y>=a1-a2)
        {
            a1=x;
            a2=y;
        }
    }

    if(a1<a2)a1=a2=0;
    cout<<n*2+a1<<":"<<m*2+a2<<endl;
}
