#include<bits\stdc++.h>
using namespace std;
int n,m,a[200001],b[200001];
int mi,mj;
pair<int,int> ap[400001];
int main()
{
	int i,j,k;
	scanf("%d",&n);
	for(i=1;i<=n;i++){scanf("%d",&a[i]);ap[i].first=a[i];ap[i].second=1;}
	scanf("%d",&m);
	for(i=1;i<=m;i++){scanf("%d",&b[i]);ap[i+n].first=b[i];ap[i+n].second=2;}
	sort(ap+1,ap+n+m+1);
	int tp[3]={0};
	for(i=1;i<=n+m;i++)
	{
		while(i<n+m&&ap[i].first==ap[i+1].first)
		{
			tp[ap[i].second]++;
			i++;
		}
		tp[ap[i].second]++;
		if(3*n-tp[1]-3*m+tp[2]>3*n-mi-3*m+mj){mi=tp[1];mj=tp[2];}
	}
	printf("%d:%d\n",3*n-mi,3*m-mj);
	return 0;
}
