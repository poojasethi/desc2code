#include <bits/stdc++.h>
using namespace std;
long long n,m,a[200005],b[200005],i,j,x,y,r,s;
int main()
{
    cin >> n;
    for(i=0;i<n;i++)
        cin >> a[i];
    cin >> m;
    for(i=0;i<m;i++)
        cin >> b[i];
    sort(a,a+n);
    sort(b,b+m);
    x=2*n;y=2*m;r=2*n;s=2*m;
    for(i=n-1,j=m-1;i>=0;i--)
    {
        r++;
        for(j;j>=0 && b[j]>=a[i];j--)
            s++;
        if(r-s>=x-y)
        {
            x=r;
            y=s;
        }
    }
    cout << x << ':' << y;
    return 0;
}
