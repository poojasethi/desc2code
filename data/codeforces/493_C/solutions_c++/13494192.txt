#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios::sync_with_stdio(false);
	int n,m;
	cin>>n;
	set<ll>s;
	map<ll,int>m1,m2;
	for(int i=0;i<n;i++)
	{
		ll x;
		cin>>x;
		s.insert(x);
		m1[x]++;
	}
	cin>>m;
	for(int i=0;i<m;i++)
	{
		ll x;
		cin>>x;
		s.insert(x);
		m2[x]++;
	}
	ll sc1=n*3;
	ll sc2=m*3;
	ll mx=sc1-sc2;
	ll ans1=sc1;
	ll ans2=sc2;
	set<ll>::iterator iter;
	for(iter=s.begin();iter!=s.end();iter++)
	{
		sc1=sc1-m1[*iter]*3+m1[*iter]*2;
		sc2=sc2-m2[*iter]*3+m2[*iter]*2;
		if((sc1-sc2)>mx)
		{
			mx=sc1-sc2;
			ans1=sc1;
			ans2=sc2;
		}
	}
	cout<<ans1<<":"<<ans2<<"\n";
	return 0;
}
