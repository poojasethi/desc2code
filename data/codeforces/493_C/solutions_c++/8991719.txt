#include <iostream>
#include <algorithm>

using namespace std;

const int maxn=200010;
const int INF=(2e9)+10;
int a[maxn],b[maxn];

int main()
{
    int n,m;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    a[n]=INF;
    sort(a,a+n);
    cin>>m;
    for(int i=0;i<m;i++)
        cin>>b[i];
    sort(b,b+m);
    int t1,t2=0,ans=3*n-3*m,x=0,y=0;
    for(t1=0;t1<=n&&t2<m;t1++)
    {
        while(t2<m&&b[t2]<a[t1]) t2++;
        if(ans<3*n-t1-(3*m-t2)) {ans=3*n-t1-(3*m-t2);x=t1;y=t2;}
    }
    cout<<3*n-x<<":"<<3*m-y<<endl;
    return 0;
}
