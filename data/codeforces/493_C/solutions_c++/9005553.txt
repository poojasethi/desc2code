#include<bits/stdc++.h>
using namespace std;
int n,m,sa,sb,x,y,a[200005],b[200005];
int main() {
	int i,j;
	cin>>n;
	for (i=0; i<n; i++) scanf("%d",&a[i]);
	cin>>m;
	for (j=0; j<m; j++) scanf("%d",&b[j]);
	sort(a, a+n);
	sort(b, b+m);
	x = sa = 2*n;
	y = sb = 2*m;
	for (i=n-1,j=m-1; i>=0; i--) {
		sa++;
		for (;j>=0 && b[j]>=a[i]; j--) sb++;
		if (sa-sb >= x-y) x=sa, y=sb;
	}
	cout<<x<<':'<<y;
}
