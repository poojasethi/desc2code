#include <bits/stdc++.h>
#define ll long long 
//#define m 1000000007
using namespace std;

int main() {
	// your code goes here
	std::ios::sync_with_stdio(false);
	vector< pair < ll, ll > >v;
	ll n,q,m,i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>q;
		v.push_back(make_pair(q,0));
	}

     cin>>m;
     for(i=0;i<m;i++)
     {
     	cin>>q;
     	v.push_back(make_pair(q,1));
     }
	sort(v.begin(),v.end());
	ll x=3*n;
	ll y=3*m;
	ll mx=x;
	ll my=y;
	for(i=0;i<v.size();i++)
	{
		if(v[i].second==0)
		x--;
		else
		y--;
		if(x-y>mx-my)
		{
			mx=x;
			my=y;
		}
	}
	cout<<mx<<":"<<my;
	return 0;
}
