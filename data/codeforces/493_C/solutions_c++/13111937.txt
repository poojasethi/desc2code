#include<cstdio>
#include<algorithm>
#include<utility>
using namespace std;

const int MM = 400000;

int main(){
    pair<int,int> a[MM]; int n,m,pa,pb,maxa,maxb;
    scanf("%d",&n);
    for(int i=0; i<n; i++){scanf("%d",&a[i].first); a[i].second=1;}
    scanf("%d",&m);
    for(int i=n; i<n+m; i++){scanf("%d",&a[i].first); a[i].second=2;}
    sort(a,a+n+m);
    pa=3*n; pb=3*m; maxa=pa; maxb=pb;
    for(int i=0; i<n+m; i++){
        if(a[i].second==1)pa--;
        else pb--;
        if(a[i+1].first!=a[i].first&&pa-pb>maxa-maxb){maxa=pa; maxb=pb;}
    }
    printf("%d:%d\n",maxa,maxb);
    return 0;
}
