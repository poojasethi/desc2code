def base3(a):
 if a>0:
  return str(a%3)+base3(a/3);
 return ''

lenc=len(c)
maxx=max(lena,lenc)

c+=("0"*(maxx-lenc))
i=0

 sum=(sum*3)+((int(c[maxx-i-1])+3-int(a[maxx-i-1]))%3)
