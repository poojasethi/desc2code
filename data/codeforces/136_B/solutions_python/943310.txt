def tern(n):
    if n == 0:
        return '0'
    r = []
    while n != 0:
        r.append(str(n%3))
        n/=3
    return ''.join(r[::-1])

a, c = [tern(int(x)) for x in raw_input().split()]

if len(a) < len(c):
    a = '0' * (len(c) - len(a)) + a
elif len(a) > len(c):
    c = '0' * (len(a) - len(c)) + c

b = []
for i in range(len(a)):
    b.append(str((int(c[i]) - int(a[i]))%3))

print int(''.join(b),3)
