def ten2thr(x):
    ans = []
    while x > 0:
        ans.append(x%3)
        x /= 3
    return ans

def thr2ten(x):
    ans = 0
    for d in reversed(x):
        ans *= 3
        ans += d
    return ans

def calc(ad, cd):
    bd = cd - ad
    return ((-min(bd/3, 0)*3+3)+bd)%3

a, c = [ten2thr(x) for x in map(int, raw_input().split())]
length = max(len(a), len(c))
a.extend([0]*(length - len(a)))
c.extend([0]*(length - len(c)))
b = [calc(ad, cd) for (ad, cd) in zip(a, c)]
print thr2ten(b)
