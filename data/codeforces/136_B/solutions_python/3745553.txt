def enc(s):
    res=""
    while(s!=0):
        m=s%3
        s=s/3
        res+=str(m)
    return res[::-1] or "0"

def dec(s):
    res=0
    s=s[::-1]
    for i in range(len(s)):
        res+=int(s[i])*(3**i)
    return res

a,c=map(int,raw_input().split())
ta=enc(a)
tc=enc(c)
y=max(len(ta),len(tc))
ta=ta.rjust(y,"0")
tc=tc.rjust(y,"0")
r=""
for i in range(y):
    r+=str((int(tc[i])+3-int(ta[i]))%3)
print dec(r)
