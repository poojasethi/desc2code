def gao(x):
    a = '' 
    while x > 0:
        a = str(x % 3) + a
        x /= 3
    if len(a) < 1:
        a = '0'
    return a
s = raw_input()
p = s.split()
a = int(p[0])
b = int(p[1])
a = gao(a)
b = gao(b)
l = max(len(a),len(b))
a = a[::-1]
b = b[::-1]
while len(a) < l:
    a += '0'
while len(b) < l:
    b += '0'
p = ''
for i in range(l):
    x = ord(b[i]) - ord(a[i])
    if x < 0:
        x += 3
    p += str(x)
print int(p[::-1],3)
