import sys
a, c = map(int,sys.stdin.readline().split())
def tot(n):
    s = ""
    while n>0:
        s+=str(n%3)
        n/=3
    s=s[::-1]
    return s
a3, c3 = tot(a),tot(c)
ml = max(len(a3),len(c3))
a3 = a3.rjust(ml,'0')
c3 = c3.rjust(ml,'0')
b = 0
t=1
i = ml-1
while i>=0:
    ai = int(a3[i])
    ci = int(c3[i])
    b += t*(ci-ai+(3 if ai>ci else 0))
    t*=3
    i-=1
print b
