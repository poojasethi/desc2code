def dec2ter(x):
	r = ""
	while x > 0:
		r += str(x%3)
		x /= 3
	return r+"0"*(20-len(r))

def ter2dec(x):
	r = 0
	i = 1
	for d in x:
		r += int(d)*i
		i *= 3
	return r

a, c = map(dec2ter, map(int, raw_input().split()))
b = ""
for i in range(len(a)):
	for d in range(0, 3):
		if (int(a[i])+d)%3 == int(c[i]):
			b += str(d)
print ter2dec(b)
