def itot(x):
    t = []
    while x > 0:
        t.append(x%3)
        x = x/3
    return t

[a, c] = [itot(int(x)) for x in raw_input().split()]

b = []

l = len(a) - len(c)
c.extend([0]*l)
a.extend([0]*-l)

a.reverse()
c.reverse()

ans = 0

for x,y in zip(a,c):
    ans = ans*3 + (y-x+3)%3

print ans
