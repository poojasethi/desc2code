
def least(a, c):
  return (c - a + 3) % 3


def pprint(number):
  res = 0
  for i, p in enumerate(number):
    res += p * (3 ** i)

  return res

def solve(a, c):
  p = 3
  res = []

  while 0 < max(a, c):
    a_ = a % 3
    c_ = c % 3
    a /= 3
    c /= 3

    res.append(least(a_, c_))


  return pprint(res)

def mian():
  ins = raw_input()
  [a, c] = map(int, ins.split())

  print solve(a, c)

mian()
