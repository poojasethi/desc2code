s = raw_input()
for i in range(len(s)):
    if s[i] == ' ':
        index = i
a = int(s[:index])
c = int(s[index:])

ta = [0]*20
tb = [0]*20
tc = [0]*20

for i in range(0,20):
    ta[19-i]=a%3
    a=a//3
    tc[19-i]=c%3
    c=c//3

for i in range(0,20):
    if ta[i] > tc[i]:
        tb[i] = tc[i]+3-ta[i]
    else:
        tb[i] = tc[i]-ta[i]

b=0
t = 1
for i in range(0,20):
    b=b+t*tb[19-i]
    t = t*3

print b
