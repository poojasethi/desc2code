def conv(num, base):
    if num == 0: return "0"
    a = []
    while num > 0:
        a.append(str(num%base))
        num /= base
    a.reverse()
    res = "".join(a)
    return res

a, c = map(int,raw_input().split())
x = conv(a, 3)
z = conv(c, 3)
l = max(len(x), len(z))
x = x.zfill(l)
z = z.zfill(l)
y = ""
for i in range(l):
    y += str( (3 + int(z[i]) - int(x[i]))%3 )
b = int(y, 3)
print b
