def str_to_t(s):
	num = int(s)
	t = []
	while num > 0:
		t.append(num % 3)
		num /= 3
	return t or [0]
def t_to_int(t):
	return int(''.join(map(str, reversed(t))), 3)
b_is = {}
coll = (2, 1, 0)
for a in coll:
	for b in coll:
		c = (a + b) % 3
		b_is[a, c] = b

a, c = map(str_to_t, raw_input().split())
if len(a) <= len(c):
	a = a + [0] * (len(c) - len(a))
else:
	c = c + [0] * (len(a) - len(c))
b = [b_is[ai, ci] for ai,ci in zip(a, c)]
print int(t_to_int(b))
