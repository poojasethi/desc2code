a,c=map(int,raw_input().split())

aa=[]
cc=[]

bb=[]
while a>0:
    aa=[a%3]+aa
    a/=3
    
while c>0:
    cc=[c%3]+cc
    c/=3
    
while len(aa)<len(cc):aa=[0]+aa
while len(cc)<len(aa):cc=[0]+cc


for i in xrange(len(aa)-1,-1,-1):
    if aa[i]%3==cc[i]:bb=[0]+bb
    elif (aa[i]+1)%3==cc[i]:bb=[1]+bb
    elif (aa[i]+2)%3==cc[i]:bb=[2]+bb

b=0
for i in bb:
    b=b*3+i

print b
