a, c = map(int, raw_input().split())
def ten2three(x):
  ans = []
  while x:
    ans.append(x%3)
    x //= 3
  return ans if len(ans) else [0]
def three2ten(x):
  ans = 0
  for i,a in enumerate(x):
    ans += a*3**i
  return ans
a3 = ten2three(a)
c3 = ten2three(c)
b3 = []
i = 0
while i<len(a3) or i<len(c3):
  tmp = c3[i] if i<len(c3) else 0
  tmp -= a3[i] if i<len(a3) else 0
  tmp += 3 if tmp<0 else 0
  b3.append(tmp)
  i += 1
b = three2ten(b3)
print b
