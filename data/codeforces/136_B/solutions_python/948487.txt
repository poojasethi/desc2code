a, c = map(int, raw_input().split())
def to(n):
	r = []
	while(n > 0):
		r.append(n % 3)
		n /= 3
	return r
va, vc = to(a), to(c)
while(len(va) > len(vc)):
	vc.append(0)
while(len(vc) > len(va)):
	va.append(0)
b = 0
for i in range(len(va)):
	for j in range(3):
		if (va[i] + j) % 3 == vc[i]:
			b += pow(3, i) * j
			break
print b
