def num2tern(a):
    x = []
    while(a>0):
        x.insert(0,str(a%3))
        a /= 3
    return ''.join(x)

a,c = list(map(int, str(raw_input()).split()))
a = num2tern(a)
c = num2tern(c)
leng = max(len(a), len(c))
a = a.zfill(leng)
c = c.zfill(leng)
i=0
b = ''
while(i<leng):
    b += str((ord(c[i])-ord(a[i]))%3)
    i+=1
if(leng == 0):
    b = '0'
print int(b,3)
