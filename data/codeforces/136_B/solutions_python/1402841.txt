def to3(a):
	if a == 0: return [0]
	res = []
	while a > 0:
		res.append(a % 3)
		a /= 3
	return res
def f(a, c):
	return (c - a + 3) % 3
def from3(a):
	res = 0
	st = 1
	for q in a:
		res += st * q
		st *= 3
	return res
a, c = map(int, raw_input().split())
a3 = to3(a)
c3 = to3(c)
while len(a3) < len(c3): a3.append(0)
while len(c3) < len(a3): c3.append(0)
print from3([f(x, y) for (x, y) in zip(a3, c3)])
