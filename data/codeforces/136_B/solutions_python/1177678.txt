
from math import *

a, c = map(int, raw_input().split())

def dec2three(x):
    if x == 0: return [0]
    return [(x / 3**i) % 3 for i in xrange(int(log(x, 3)) + 1)]

A = dec2three(a)
C = dec2three(c)
[A, C][len(A) > len(C)].extend([0]*abs(len(A) - len(C)))
B = [(j - i + 3) % 3 for i, j in zip(A, C)]
print sum([x * 3**i for i, x in enumerate(B)])
