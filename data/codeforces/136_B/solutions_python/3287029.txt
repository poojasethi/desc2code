import math,itertools

def dec2ter(dec):
    tern = []
    while dec>0:
        tern.append( dec%3 )
        dec = (dec-tern[-1])/3
        
    return reversed(tern)

def ter2dec(terList):
    dec = 0
    for ind, j in enumerate(terList):
        dec += j*3**ind
    return dec


ta, tc = [list(dec2ter(int(j))) for j in raw_input().split()]

tb = [(c-a)%3 for c, a in itertools.izip_longest(reversed(tc), reversed(ta), fillvalue=0)]

print ter2dec(tb)
