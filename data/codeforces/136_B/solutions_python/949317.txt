import sys

def to3(a):
    a = int(a)
    l = []
    while a > 0:
        l.append(a % 3)
        a //= 3
    return l
a, b = map(to3, sys.stdin.readline().strip().split())

while len(a) < len(b):
    a.append(0)
while len(b) < len(a):
    b.append(0)

c = 0
for x, y in zip(a, b)[::-1]:
    c = 3 * c + (y + 3 - x) % 3

print c
