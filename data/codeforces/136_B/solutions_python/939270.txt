def to3(a):
    a=int(a)
    s=[]
    while a>0:
        s.append(a%3)
        a//=3
    return s[-1::-1]

a,c=map(to3,raw_input().split())
if len(a)>len(c):
    c=[0]*(len(a)-len(c))+c
elif len(c)>len(a):
    a=[0]*(len(c)-len(a))+a
b=[]
for i in range(len(a)):
    b.append((c[i]+3-a[i])%3)
ans=0
for i in b:
    ans=ans*3+i
print ans
