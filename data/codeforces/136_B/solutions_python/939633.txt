s = raw_input().split()
n=int(s[0])
m=int(s[1])

def change(a):
    b=''
    while(a>0):
        b=b+str(a%3)
        a/=3
    return b

x = change(n)
y = change(m)
while (len(x)<len(y)):
    x+='0'
while (len(y)<len(x)):
    y+='0'

z=0
mu3=1
for i in range(len(x)):
    z=z+((int(y[i])-int(x[i]))%3)*mu3
    mu3*=3

print z
