def f(a):
    s=""
    while a>=3:
        s = str(a%3) + s
        a/=3
    s = str(a) + s
    return s
def g(a):
    t=0
    a=a[::-1]
    d=1
    for x in a:
        t+=int(x)*d
        d*=3
    return t
a,b=map(int,raw_input().split())
sa = f(a)
sb = f(b)
if len(sa) < len(sb):
    sa = '0' * (len(sb)-len(sa)) + sa
if len(sb) < len(sa):
    sb = '0' * (len(sa)-len(sb)) + sb
sc=""
for i in range(len(sa)):
    sc = sc+ (str(( 3 + int(sb[i]) - int(sa[i]) ) % 3))
print g(sc)
