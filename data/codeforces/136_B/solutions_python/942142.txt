def tr(n):
	k = []
	while n >= 3:
		k.append(n % 3)
		n = n / 3
	k.append(n)
	return k

a, c = list(map(int, raw_input().split()))
a_3 = tr(a)
c_3 = tr(c)
n = max(len(a_3), len(c_3))
for i in range(n - min(len(a_3), len(c_3))):
	a_3.append(0)
	c_3.append(0)
b_3 = []
for i in range(n):
	b_3.append((c_3[i] - a_3[i] + 3) % 3)
k = 0
for i in range(n):
	k = k + (3 ** i) * b_3[i]
print k
