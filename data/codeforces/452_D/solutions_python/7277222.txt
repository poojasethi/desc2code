import heapq as hq

k, n1, n2, n3, t1, t2, t3 = map(int, raw_input().split())
h1 = [0] * n1
h2 = [0] * n2
h3 = [0] * n3
t = 0
for i in range(k):
    t = max(h1[0], h2[0] - t1, h3[0] - t1 - t2)
    m = hq.heappop(h1)
    hq.heappush(h1, t + t1)
    m = hq.heappop(h2)
    hq.heappush(h2, t + t1 + t2)
    m = hq.heappop(h3)
    hq.heappush(h3, t + t1 + t2 + t3)
print max(h3)
