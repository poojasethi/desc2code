#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define N 1005
int n[4],t[4];
int a[N],b[N],c[N];
int k,ans;

int main(){
  int i,x,y,z;
  scanf("%d",&k);
  for (i=1;i<=3;++i) scanf("%d",&n[i]);
  for (i=1;i<=3;++i) scanf("%d",&t[i]);
  x=y=z=0;
  while (k--){
    a[x]=max(a[x]+t[1],max(b[y],c[z]-t[2]));
    b[y]=a[x]+t[2];
    c[z]=b[y]+t[3];
    ans=max(ans,c[z]);
    x=(x+1)%n[1];
    y=(y+1)%n[2];
    z=(z+1)%n[3];
  }
  printf("%d\n",ans);
  return 0;
}
