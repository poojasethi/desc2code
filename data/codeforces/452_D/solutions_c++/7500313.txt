#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
	int dp[10010],n1,n2,n3,t1,t2,t3,k,i;
	scanf("%d %d %d %d %d %d %d",&k,&n1,&n2,&n3,&t1,&t2,&t3);
	memset(dp,0,sizeof(dp));
	for(i=0;i<=k;i++)
	{
		if(i>n1)dp[i]=max(dp[i],dp[i-n1]+t1);
		if(i>n2)dp[i]=max(dp[i],dp[i-n2]+t2);
		if(i>n3)dp[i]=max(dp[i],dp[i-n3]+t3);
		//printf("%d\n",dp[i]);
	}
	printf("%d\n",dp[k]+t1+t2+t3);
}
