#include<iostream>
#include<cstdio>
#include<cstring>
#include<queue>
using namespace std;

const int N=1010;
int m,n1,n2,n3,t1,t2,t3,t;
int q1[N],q2[N],q3[N];

int main()
{
   cin>>m>>n1>>n2>>n3>>t1>>t2>>t3;

   for(int i=0;i<m;i++)
   {
       q1[i%n1]+=t1+t2+t3;
       q2[i%n2]+=t2+t3;
       q3[i%n3]+=t3;
       t=max(q1[i%n1],max(q2[i%n2],q3[i%n3]));
       q1[i%n1]=t-t2-t3;
       q2[i%n2]=t-t3;
       q3[i%n3]=t;
   }
   cout<<t<<endl;
}
