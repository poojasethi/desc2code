#include <bits/stdc++.h>
using namespace std ;
int k;
int n[4],t[4];
int w;
int strt[10009];
int main()
{
    cin>>k ;

    for(int i=1; i<=3; i++)cin>>n[i];
    for(int i=1; i<=3; i++)
    {
        cin>>t[i];
        w+=t[i];
    }
    for(int i=1; i<=k; i++)
    {
        int st=0;
        if(i-n[1]>0)st=max(st,strt[i-n[1]]+t[1]);
        if(i-n[2]>0)st=max(st,strt[i-n[2]]+t[2]);
        if(i-n[3]>0)st=max(st,strt[i-n[3]]+t[3]);
        strt[i]=st;
    }
    cout<<strt[k]+w<<endl;
}
