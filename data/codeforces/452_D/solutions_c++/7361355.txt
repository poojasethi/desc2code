#include<stdio.h>
using namespace std;
int dp[10004];
#include<iostream>
#include<algorithm>
int main(){
    int n1,n2,n3,k,t1,t2,t3;
    cin>>k>>n1>>n2>>n3>>t1>>t2>>t3;
    for(int i=1;i<=k;i++){
        dp[i] = t1+t2+t3;
        if(n1<i) dp[i] = max(dp[i],t1+ dp[i-n1]);
        if(n2<i) dp[i] = max(dp[i],t2+ dp[i-n2]);
        if(n3<i) dp[i] = max(dp[i],t3+ dp[i-n3]);
    }
    cout<<dp[k]<<endl;
    return 0;
}
