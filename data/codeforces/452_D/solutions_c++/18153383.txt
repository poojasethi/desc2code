#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int k,n1,n2,n3,t1,t2,t3,st[10005];
int main() {
	scanf("%d%d%d%d%d%d%d",&k,&n1,&n2,&n3,&t1,&t2,&t3);
	for (int i=min(min(n1,n2),n3)+1;i<=k;i++) {
		int p1=i>n1?st[i-n1]+t1:0,p2=i>n2?st[i-n2]+t2:0,p3=i>n3?st[i-n3]+t3:0;
		st[i]=max(max(p1,p2),p3);
	}
	printf("%d\n",st[k]+t1+t2+t3);
	return 0;
}
