#include <iostream>

using namespace std;

int d[10007];

int main()
{
    int k, n1, n2, n3, t1, t2, t3;
    cin >> k >> n1 >> n2 >> n3 >> t1 >> t2 >> t3;

    for (int i = 1; i <= k; i++)
    {
        if (i > n1) d[i] = max(d[i], d[i - n1] + t1);
        if (i > n2) d[i] = max(d[i], d[i - n2] + t2);
        if (i > n3) d[i] = max(d[i], d[i - n3] + t3);
    }

    cout << d[k] + t1 + t2 + t3;

    return 0;
}
