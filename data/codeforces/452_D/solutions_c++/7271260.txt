#include <iostream>

using namespace std;

int k,n1,n2,n3,t1,t2,t3;

int time[10009];

int main()
{
	cin >> k >> n1 >> n2 >> n3 >> t1 >> t2 >> t3;
	
	for (int i = 1; i < k; i++)
	{
		time[i] = time[i-1];
		
		if (i-n1 >= 0)	
			time[i] = max(time[i], time[i-n1]+t1);
		if (i-n2 >= 0)	
			time[i] = max(time[i], time[i-n2]+t2);
		if (i-n3 >= 0)	
			time[i] = max(time[i], time[i-n3]+t3);
	}
	
	cout << time[k-1]+t1+t2+t3;
}
