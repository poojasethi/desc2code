#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#define ll long long
using namespace std;

int t[10005],k,n0,n1,n2,t0,t1,t2;
int main()
{
	//k=read();n0=read();n1=read();n2=read();t0=read();t1=read();t2=read();
	cin>>k>>n0>>n1>>n2>>t0>>t1>>t2;
	for(int i=0;i<k;i++)
	{
		if(i>=n0)t[i]=max(t[i],t[i-n0]+t0);
		if(i>=n1)t[i]=max(t[i],t[i-n1]+t1);
		if(i>=n2)t[i]=max(t[i],t[i-n2]+t2);
	}
	printf("%d\n",t[k-1]+t0+t1+t2);
	return 0;
}
