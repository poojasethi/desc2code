#include <iostream>
using namespace std;
int dp[10002];
int main() 
{
    ios::sync_with_stdio(false);
    int i,k,n1,n2,n3,t1,t2,t3;
    cin>>k>>n1>>n2>>n3>>t1>>t2>>t3;
    for(i=1;i<=k;i++)
    {
        if(i>n1)
        dp[i]=max(dp[i],dp[i-n1]+t1);
        if(i>n2)
        dp[i]=max(dp[i],dp[i-n2]+t2);
        if(i>n3)
        dp[i]=max(dp[i],dp[i-n3]+t3);
    }
    cout<<dp[k]+t1+t2+t3;
    return 0;
}
