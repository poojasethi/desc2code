#include<bits/stdc++.h>
#define MAX 10005
using namespace std;
int k, n1, n2, n3, t1, t2, t3;
int dp[MAX];
int main(){
	cin>>k>>n1>>n2>>n3>>t1>>t2>>t3;
	memset(dp, 0, sizeof(dp));
	
	for(int i = 1; i <= k; i++){
		if(i > n1) dp[i] = max(dp[i], dp[i - n1] + t1);
		if(i > n2) dp[i] = max(dp[i], dp[i - n2] + t2);
		if(i > n3) dp[i] = max(dp[i], dp[i - n3] + t3);
		//cout<<dp[i]<<" ";
	}
	//cout<<endl;
	cout<<dp[k]+t1+t2+t3<<endl;
	return 0;
}
