#include<cstdio>
#include<iostream> 
#include<cmath>
#include<algorithm>
#include<cstring>
#include<string>
#define ll long long
using namespace std;
ll k,n[4],t[4],f[10004];
int main(){
    cin>>k;
    for(int i=1;i<=3;i++)cin>>n[i];
    for(int i=1;i<=3;i++)cin>>t[i];
    for(int i=1;i<=k;i++)f[i]=0;
    for(int i=1;i<=k;i++){
            for(int j=1;j<=3;j++){
                   if(i>n[j])f[i]=max(f[i],f[i-n[j]]+t[j]); 
            }
    }
    cout<<f[k]+t[1]+t[2]+t[3]<<endl;
    //system("pause");
    return 0; 
}
