#include <iostream>
#include <iomanip>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;
#define LL __int64

int dp[5][10005];
int n[5],t[5],k;

int main()
{
	int a,b;
	while(~scanf("%d%d%d%d%d%d%d",&k,&n[1],&n[2],&n[3],&t[1],&t[2],&t[3]))
	{
		memset(dp,0,sizeof(dp));
		for (a=1;a<=3;a++)
			for (b=1;b<=k;b++)
			{
				dp[a][b]=max(dp[a-1][b],dp[a][max(b-n[a],0)])+t[a];
			}
		printf("%d\n",dp[3][k]);
	}
	return 0;
}
