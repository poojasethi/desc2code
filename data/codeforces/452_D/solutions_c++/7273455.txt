#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int N = 10005;
int n, n1, n2, n3, t1, t2, t3;
int dp[N], ans;
void solve(int num, int t) {
    for (int i = 0; i < n; ++i) {
        if (i >= num) dp[i] = max(dp[i], dp[i-num]);
        dp[i] += t;
    }
}
int main() {
    scanf("%d%d%d%d%d%d%d", &n, &n1, &n2, &n3, &t1, &t2, &t3);
    solve(n1, t1);
    solve(n2, t2);
    solve(n3, t3);
    ans = 0;
    for (int i = 0; i < n; ++i) ans = max(ans, dp[i]);
    
    printf("%d\n", ans);
    return 0;
}
