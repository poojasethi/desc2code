#include<iostream>
using namespace std;
int dp[10001];
int main()
{
	int k,n[3],t[3];
	cin>>k;
	for(int i=0;i<3;i++)cin>>n[i];
	for(int i=0;i<3;i++)cin>>t[i];
	for(int i=1;i<=k;i++)
	{
		if(i>n[0])dp[i]=max(dp[i],dp[i-n[0]]+t[0]);
		if(i>n[1])dp[i]=max(dp[i],dp[i-n[1]]+t[1]);
		if(i>n[2])dp[i]=max(dp[i],dp[i-n[2]]+t[2]);
	}
	cout<<dp[k]+t[0]+t[1]+t[2]<<endl;
	return 0;
}
