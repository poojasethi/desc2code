#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
#include<cmath>
using namespace std;
int n[5],t[5],dp[5][10005];
int main(){
    int i,j,m,k;
    scanf("%d%d%d%d%d%d%d",&k,&n[1],&n[2],&n[3],&t[1],&t[2],&t[3]);
    for(i=3;i>=1;i--){
        for(j=1;j<=k;j++){
            dp[i][j]=max(dp[i+1][j],j-n[i]>=0 ? dp[i][j-n[i]] : 0)+t[i];
        }
    }
    printf("%d\n",dp[1][k]);
    return 0;
}
