#include <iostream>
using namespace std;

int k, n1, n2, n3, t1, t2, t3, p1, p2, p3, q1[11111], q2[11111], q3[11111], now;
int main()
{
	cin>>k>>n1>>n2>>n3>>t1>>t2>>t3;
	while (k--) {
		now=max(max(q1[p1],q2[p2]-t1), q3[p3]-t2-t1);
		q1[p1+n1]=now+t1; p1++;
		q2[p2+n2]=now+t1+t2; p2++;
		q3[p3+n3]=now+t1+t2+t3; p3++;
	}
	cout<<now+t1+t2+t3<<endl;
	return 0;
}
