#include<iostream>
#include<cstdio>
#include<string.h>
#include<algorithm>
using namespace std;
int dp[11111];
int main()
{
	int n,n1,n2,n3,t1,t2,t3 ;
	int i;
	cin>>n>>n1>>n2>>n3>>t1>>t2>>t3 ;
	for(i=1;i<=min(n1,min(n2,n3));i++)  dp[i] = t1+t2+t3 ;
	for(i=2;i<=n;i++)
	{
		if(i >= n1)
		dp[i] = max(dp[i],dp[i-n1]+t1);
		if(i >= n2)
		dp[i] = max(dp[i],dp[i-n2]+t2);
		if(i >= n3)
		dp[i] = max(dp[i],dp[i-n3]+t3);
	}
	cout<<dp[n]<<endl;
	return 0 ;
}
