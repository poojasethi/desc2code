#include<algorithm>
#include<cstdio>
#include<queue>
using namespace std;

int ready[4][10010];

int main() {
	int k, n[4], t[4];
	scanf("%d%d%d%d%d%d%d", &k, &n[1], &n[2], &n[3], &t[1], &t[2], &t[3]);
	
	for(int i=0; i<k; i++) {
		ready[0][i] = 0;
	}
	
	int ans = 0;
	for(int i=1; i<4; i++) {
		for(int j=0; j<k; j++) {
			if(j >= n[i] && ready[i][j - n[i]] > ready[i-1][j]) {
				ready[i][j] = t[i] + ready[i][j - n[i]];
			} else {
				ready[i][j] = ready[i-1][j] + t[i];
			}
		}
	}
	
	printf("%d\n", ready[3][k-1]);
	return 0;
}
