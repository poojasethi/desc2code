#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;

int t[10010];
int main()
{
    int k,n1,n2,n3,t1,t2,t3;
    while(scanf("%d%d%d%d%d%d%d",&k,&n1,&n2,&n3,&t1,&t2,&t3)!=EOF)
    {
        for(int i=0;i<k;i++)
        {
            if(i>=n1) t[i]=max(t[i],t[i-n1]+t1);
            if(i>=n2) t[i]=max(t[i],t[i-n2]+t2);
            if(i>=n3) t[i]=max(t[i],t[i-n3]+t3);
        }
        printf("%d\n",t[k-1]+t1+t2+t3);
    }
    return 0;
}
