#include<cstdio>
#include<algorithm>
using namespace std;
int a1[10000001];
int main()
{
    int n,a,b,c,d,e,f,x;
    scanf("%d%d%d%d%d%d%d",&n,&a,&b,&c,&d,&e,&f);
    for(int i=0;i<10000001;++i)
    {
        if(i>=d)a+=a1[i-d];if(i>=e)b+=a1[i-e];if(i>=f)c+=a1[i-f];
        x=min(a,min(b,c));
        a1[i]=x;a-=x;b-=x;c-=x;n-=x;
        if(n<=0){printf("%d\n",i+d+e+f);break;}
    }
}
