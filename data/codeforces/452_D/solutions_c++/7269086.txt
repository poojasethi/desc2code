#include <iostream>
#define MAXN 1010

using namespace std;

int k,n1,n2,n3,t1,t2,t3;
int x1[MAXN],x2[MAXN],x3[MAXN];
int ans;

int main() {
    cin >> k >> n1 >> n2 >> n3 >> t1 >> t2 >> t3;
    for (int i = 0, a = 0, b = 0, c = 0; i < k; i++, a=(a+1)%n1, b=(b+1)%n2, c=(c+1)%n3) {
        x1[a] = max(x1[a]+t1,max(x2[b],x3[c]-t2));
        x2[b] = x1[a] + t2;
        x3[c] = x2[b] + t3;
        ans = max(ans,x3[c]);
    }
    cout << ans << '\n';
}
