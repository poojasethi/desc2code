#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int finalTime[10005];
int k,n[3],t[3];
int main()
{

    memset(finalTime,0,sizeof finalTime);
    cin>>k;
    for(int i=0;i<3;++i)
        cin>>n[i];
    for(int j=0;j<3;++j)
        cin>>t[j];

    for(int i=0;i<3;++i)
        for(int j=0;j<k;++j)
            finalTime[j] = max(finalTime[j],j-n[i]<0?0:(finalTime[j-n[i]]))+t[i];

    cout<<*(max_element(finalTime,finalTime+k))<<endl;

}
