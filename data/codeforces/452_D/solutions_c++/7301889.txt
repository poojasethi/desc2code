#include <iostream>
#include <algorithm>
#include <vector>
#include <cmath>
#include <cstring>
#include <string>
using namespace std;

int K, N1, N2, N3, T1, T2, T3;
int S[10005];

int main()
{
	cin >> K >> N1 >> N2 >> N3 >> T1 >> T2 >> T3;

	for (int i = 0; i < K; i++)
	{
		int s1 = (i >= N1) ? (S[i-N1] + T1) : 0;
		int s2 = (i >= N2) ? (S[i-N2] + T2) : 0;
		int s3 = (i >= N3) ? (S[i-N3] + T3) : 0;
		S[i] = max(s1, max(s2, s3));
	}

	cout << S[K-1] + T1 + T2 + T3 << "\n";

	return 0;
}
