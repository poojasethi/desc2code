#include<bits/stdc++.h>
using namespace std;
int n,c1,c2,c3,t1,t2,t3;
int dp[10005];
int main()
{
    //freopen("a.txt","r",stdin);
    while(~scanf("%d%d%d%d%d%d%d",&n,&c1,&c2,&c3,&t1,&t2,&t3)){
        memset(dp,0,sizeof(dp));
        for(int i=1;i<=n;i++){
            if(i>=c1)  dp[i]=max(dp[i],dp[i-c1]+t1);
            if(i>=c2)  dp[i]=max(dp[i],dp[i-c2]+t2);
            if(i>=c3)  dp[i]=max(dp[i],dp[i-c3]+t3);
        }
        printf("%d\n",dp[n-1]+t1+t2+t3);
    }
    return 0;
}
