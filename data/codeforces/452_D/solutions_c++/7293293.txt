#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
    int k,n[3],t[3];
    int s[10000];
    while (1==scanf("%d",&k))
    {
        for (int i=0;i<3;i++) scanf("%d",&n[i]);
        for (int i=0;i<3;i++) scanf("%d",&t[i]);
        memset(s,0,sizeof(s));
        for (int i=1;i<k;i++)
        {
            for (int j=0;j<3;j++)
            {
                if (i>=n[j]) s[i]=max(s[i],s[i-n[j]]+t[j]);
            }
        }
        printf("%d\n",s[k-1]+t[0]+t[1]+t[2]);
    }
    return 0;
}
