#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 110
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,l,r;
int x[N],y[N],a[N],g[N*10000],gg[N],ff[N],d[N][N];
int main() {
	scanf("%d%d",&n,&m);
	For(i,2,n-1) scanf("%d",&a[i]);
	For(i,1,n) scanf("%d%d",&x[i],&y[i]);
	For(i,1,n)For(j,1,n) d[i][j]=abs(x[i]-x[j])+abs(y[i]-y[j]);
	memset(gg,120,sizeof(gg));
	gg[1]=0; g[l=r=1]=1;
	for (;l<=r;l++) {
		ff[g[l]]=0;
		For(i,1,n) if (i!=g[l]&&gg[i]>gg[g[l]]+m*d[g[l]][i]-a[i]) {
			gg[i]=gg[g[l]]+m*d[g[l]][i]-a[i];
			if (!ff[i]) ff[i]=1,g[++r]=i;
		}
	}
	printf("%d\n",gg[n]);
	//for(;;);
	return 0;
}
