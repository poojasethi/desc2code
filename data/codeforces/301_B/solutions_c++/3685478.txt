#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int n,d;
int x[300],y[300],a[300];
int f[210][210];

int main() {
    int i,j,k;
    scanf("%d%d",&n,&d);
    for (i=2;i<=n-1;i++) scanf("%d",a+i);
    for (i=1;i<=n;i++) scanf("%d%d",x+i,y+i);
    for (i=1;i<=n;i++) 
        for (j=1;j<=n;j++) if (i!=j)
            f[i][j]=(abs(x[i]-x[j])+abs(y[i]-y[j]))*d-a[i];
    for (k=1;k<=n;k++)
        for (i=1;i<=n;i++)
            for (j=1;j<=n;j++)
                f[i][j]=min(f[i][j],f[i][k]+f[k][j]);
    printf("%d\n",f[1][n]);
}
