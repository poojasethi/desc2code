#include <bits/stdc++.h>
using namespace std;
int n,adj[120][120],a[120],d,x[120],y[120];
int dist(int s,int t){
	vector<int> aa(n+1,1e9);
	int u,dd,i,j;
	queue<pair<int,int> >q;
	aa[s]=0;q.push(make_pair(0,s));
	pair<int,int> to;
	while(!q.empty()){
		to = q.front();q.pop();
		dd=to.first,u = to.second;
		if(aa[u]!=dd)
			continue;
		for(i=1;i<=n;++i)
			if(i!=u)
				if(aa[i]>adj[u][i]+dd)
					aa[i]=adj[u][i]+dd,q.push(make_pair(adj[u][i]+dd,i));
	}
	return aa[t];
}
int main(){
	int i,j,k;
	cin>>n>>d;
	for(i=2;i<n;++i)cin>>a[i];
	for(i=1;i<=n;++i)cin>>x[i]>>y[i];
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)
			if(i!=j)
			adj[i][j]=(abs(x[i]-x[j])+abs(y[i]-y[j]))*d-a[i];
	cout<<dist(1,n)<<endl;
}
