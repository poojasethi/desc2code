#include <iostream>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,d,dp[110][110],a[110],x[110],y[110];

int main(){

  cin>>n>>d;
  rep(i,n-2)cin>>a[i+1];
  rep(i,n)cin>>x[i]>>y[i];

  rep(i,n)rep(j,n){
    if(i!=j)dp[i][j] = (abs(x[i]-x[j])+abs(y[i]-y[j]))*d-a[i];
  }

  rep(k,n)rep(i,n)rep(j,n){
    dp[i][j] = min(dp[i][j],dp[i][k]+dp[k][j]);
  }
  cout<<dp[0][n-1]<<endl;

}
