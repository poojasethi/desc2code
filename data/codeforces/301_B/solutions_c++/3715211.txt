#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int i,j,k,l,n,m,a[120],X[120],Y[120],flag[120],bfs[120000],dis[120],r,x;
int main(){
	scanf("%d%d",&n,&m);
	for(i=2;i<n;++i)scanf("%d",&a[i]);
	for(i=1;i<=n;++i)scanf("%d%d",&X[i],&Y[i]);
	memset(dis,120,sizeof(dis));
	dis[1]=0;
	bfs[l=r=1]=1;
	while(l<=r){
		x=bfs[l++];
		flag[x]=0;
		for(i=1;i<=n;++i)if(i!=x && dis[x]+m*(abs(X[x]-X[i])+abs(Y[x]-Y[i]))-a[i]<
		dis[i]){
			dis[i]=dis[x]+m*(abs(X[x]-X[i])+abs(Y[x]-Y[i]))-a[i];
			if(!flag[i])bfs[++r]=i;
		}
	}
	printf("%d\n",dis[n]);
//	system("pause");
}
