#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;

#define long long long
#define f1(i,n) for (int i=1; i<=n; i++)
#define f0(i,n) for (int i=0; i<n; i++)

#define N 123
int n, D, b[N], x[N], y[N], c[N][N];

void minimize(int &a, int b){ if (a>b) a=b; }

main(){
    scanf("%d%d", &n, &D);
    for (int i=2; i<=n-1; i++) scanf("%d", &b[i]);
    for (int i=1; i<=n; i++) scanf("%d%d", &x[i], &y[i]);

    f1(i,n) f1(j,n) 
        { c[i][j]=(abs(x[i]-x[j])+abs(y[i]-y[j]))*D-b[i]; }
    f1(i,n) c[i][i]=0;
    f1(k,n) f1(i,n) f1(j,n) 
        { minimize(c[i][j], c[i][k]+c[k][j]); }
    cout << c[1][n] << endl;
}
