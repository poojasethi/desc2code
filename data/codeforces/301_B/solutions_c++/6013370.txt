#include<stdio.h>
int ab(int x){
	if(x<0)return -x;
	else return x;
}
int min(int x,int y){
	return x<y?x:y;
}
int n,d,i,j,k,x[105],y[105],a[105],f[105][105];
int main() {
    scanf("%d%d",&n,&d);
    for(i=1;i<=n-2;i++) scanf("%d",&a[i]);
    for(i=1;i<=n;i++)scanf("%d%d",&x[i],&y[i]);
    for(i=1;i<=n;i++)
    	for(j=1;j<=n;j++)
    		if(i!=j)
    			f[i][j]=(ab(x[i]-x[j])+ab(y[i]-y[j]))*d-a[i-1];
	for(k=1;k<=n;k++)
    	for(i=1;i<=n;i++)
    		for(j=1;j<=n;j++)
    			f[i][j]=min(f[i][j],f[i][k]+f[k][j]);
    printf("%d\n",f[1][n]);
}
