#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define FOR(i,a,b) for(int i =int(a); i<=int(b);i++)
#define abs(x) ( (x) >=0 ?(x):(-(x)))
#define INF 987654321
int n,d;
long long res;
int x[105];
int y[105];
int f[105];
int map[105][105];
int main()
{
	cin >> n >> d;
	FOR(i,2,n-1)
		cin >> f[i];
	FOR(i,1,n)
		cin >> x[i] >> y[i];
	
	
	FOR(i,1,n)
		FOR(j,1,n)
		{
				if(i==j)
				{
					map[i][j]=0;
					continue;
				}
				map[i][j] = d*(abs(x[i]-x[j]) + abs(y[i]-y[j]))-f[i];
		}
	
	FOR(t,1,n)
		FOR(i,1,n)
			FOR(j,1,n)
			{
				map[i][j] = min(map[i][j], map[i][t] + map[t][j]);
			}
	
	cout << map[1][n] << endl;
	
	return 0;
}
