int n,d,x[111],y[111],a[111],f[111][111];

    cin>>n>>d;
    re(i,n-2) cin>>a[i+1];
    re(i,n) cin>>x[i]>>y[i];
    re(i,n) re(j,n) if(i-j) f[i][j]=(abs(x[i]-x[j])+abs(y[i]-y[j]))*d-a[i];
    re(k,n) re(i,n) re(j,n) f[i][j] = min(f[i][j],f[i][k]+f[k][j]);
