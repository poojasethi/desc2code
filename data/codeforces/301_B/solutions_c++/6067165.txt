#include"stdio.h"
#include"math.h"
struct node{
int x,y;
}p[111];
int val[111],map[111][111];
int main()
{
	int n,k,i,j,l,t;
	scanf("%d%d",&n,&k);
	for(i=2;i<n;i++)
	scanf("%d",&val[i]);
	val[1]=val[n]=0;
	for(i=1;i<=n;i++)
	scanf("%d%d",&p[i].x,&p[i].y);
	for(i=1;i<=n;i++)
	{
		map[i][i]=0;
		for(j=i+1;j<=n;j++)
		{
			t=fabs(p[i].x-p[j].x)+fabs(p[i].y-p[j].y);
			t*=k;
			map[i][j]=t-val[j];
			map[j][i]=t-val[i];
		}
	}
	for(l=1;l<=n;l++)
	{
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				if(map[i][l]+map[l][j]<map[i][j])map[i][j]=map[i][l]+map[l][j];
			}
		}
	}
	printf("%d\n",map[1][n]);
	return 0;
}
