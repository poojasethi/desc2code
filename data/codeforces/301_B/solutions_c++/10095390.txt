#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define N 100005
#define For(i,l,r) for(int i=l;i<=r;i++)
using namespace std;
struct rec{int x,y;}E[N];
int dis[N],w[N],q[N],inq[N],n,d,st,en;
int main(){
	scanf("%d%d",&n,&d);
	For(i,2,n-1) scanf("%d",&w[i]);
	For(i,1,n) scanf("%d%d",&E[i].x,&E[i].y);
	
	memset(dis,0x7f,sizeof(dis));
	q[++en]=1; dis[1]=0;
	while(st<en){
		int u=q[++st]; inq[u]=0;
		For(v,1,n) if(u!=v){
			int cost=abs(E[u].x-E[v].x)+abs(E[u].y-E[v].y); cost*=d;
			if(dis[v]>dis[u]+cost-w[v]){
				dis[v]=dis[u]+cost-w[v];
				if(!inq[v]) inq[v]=1,q[++en]=v;
			}
		}
	}
	
	printf("%d\n",dis[n]);
}
