#include <cstdio>
#include <algorithm>

int b[111][111];
int x[111], y[111], a[111];

int main() {
  int n,d;
  scanf("%d%d",&n,&d);
  a[0]=a[n-1]=0;
  for (int i=1;i<n-1;i++) scanf("%d", a+i);
  for (int i=0;i<n;i++) scanf("%d%d", x+i, y+i);
  for (int i=0;i<n;i++) {
    for (int j=0;j<n;j++) {
      if (i == j) b[i][j] = 0; else
      b[i][j] = d*(std::abs(x[i]-x[j])+std::abs(y[i]-y[j]))-a[i];
    }
  }
  for (int k=0;k<n;k++) 
    for (int i=0;i<n;i++)
      for (int j=0;j<n;j++)
        b[i][j] = std::min(b[i][j], b[i][k]+b[k][j]);
  printf("%d\n", b[0][n-1]);
}
