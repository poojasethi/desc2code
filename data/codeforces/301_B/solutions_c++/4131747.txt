#include<cstdlib>
#include<cstdio>
#include<cmath>
using namespace std;
int n,d,a[101],x[101],y[101],dist[101];
int min(int a,int b){return a<b?a:b;}

int main()
{
   int i,k,j;
   scanf("%d %d",&n,&d);
   for(i=1;i<n-1;i++) scanf("%d",&a[i]);
   for(i=0;i<n;i++) scanf("%d %d",&x[i],&y[i]);
   for(i=0;i<n;i++) dist[i]=d*(abs(x[i]-x[0])+abs(y[i]-y[0]))-a[i];
   for(k=1;k<n-1;k++)
     for(i=0;i<n;i++)
       for(j=0;j<n;j++)
         if(i!=j) dist[i]=min(dist[i],dist[j]+d*(abs(x[i]-x[j])+abs(y[i]-y[j]))-a[i]);
   printf("%d\n",dist[n-1]);
}
