#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;
int main()
{
	int n,c;
	cin >> n >> c;
	int a[101];
	for(int i = 1;i < n-1;i++)
		cin >> a[i];
	a[0] = a[n-1] = 0;
	int x[101],y[101];
	for(int i = 0;i < n;i++)
		cin >> x[i] >> y[i];
	long long road[101][101];
	for(int i = 0;i < n;i++)
		for(int j = 0;j < n;j++)
			road[i][j] = (abs(x[i]-x[j])+abs(y[i]-y[j]))*c-a[j];
	for(int i = 0;i < n;i++)
		road[i][i] = 0; 
	for(int k = 0;k < n;k++)
		for(int i = 0;i < n;i++)
			for(int j = 0;j < n;j++)
				if(road[i][j] > road[i][k]+road[k][j])
					road[i][j] = road[i][k]+road[k][j];
	cout << road[0][n-1] << endl;
}
