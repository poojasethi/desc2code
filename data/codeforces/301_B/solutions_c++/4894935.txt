#include <algorithm>
#include <stdio.h>
#define  maxn      100
using    namespace std;

int n,d;
int x[maxn];
int y[maxn];
int ar[maxn];
int dp[maxn][maxn];

int main()
{
  scanf("%d%d",&n,&d);
  for(int i=1 ; i<=n-2 ; i++)
    scanf("%d",&ar[i]);
  for(int i=0 ; i<n ; i++)
    scanf("%d%d",&x[i],&y[i]);
  for(int i=0 ; i<n ; i++)
    for(int j=0 ; j<n ; j++)
      if(i!=j)
        dp[i][j]=(abs(x[i]-x[j])+abs(y[i]-y[j]))*d-ar[i];
  for(int k=0 ; k<n ; k++)
    for(int i=0 ; i<n ; i++)
      for(int j=0 ; j<n ; j++)
        dp[i][j]=min(dp[i][j],dp[i][k]+dp[k][j]);
  printf("%d\n",dp[0][n-1]);
  return 0;
}
