#include<iostream>
#include<utility>
#include<cmath>
using namespace std;
const int maxn = 100+3;
int n,d,r[maxn],x[maxn],y[maxn];
long long a[maxn][maxn];
int main(){
	cin >> n;cin >> d;
	for(int i = 1;i < n-1 ; i++)
		cin >> r[i];
	for(int i = 0;i < n; i++)
		cin >> x[i] >> y[i];
	for(int i = 0;i < n; i++)
		for(int j =0;j < n; j++)
			if(i != j)
				a[i][j] = (abs(x[i]-x[j])+abs(y[i]-y[j]))*d - r[j];
	for(int k = 0;k < n; k++)
		for(int i = 0;i < n; i++)
			for(int j = 0;j <n; j++)
				a[i][j] = min(a[i][j] , a[i][k] + a[k][j]);
	cout << a[0][n-1] << endl;
	return 0;
}
