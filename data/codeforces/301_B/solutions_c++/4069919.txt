//In the name of god

#include<iostream>
#include<cmath>
using namespace std;

int d,n,x[111],y[111],dis[111][111],a[111];

int main(){
   cin>>n>>d;
   for(int i=2;i<n;i++)
      cin>>a[i];
   for(int i=1;i<=n;i++)
      cin>>x[i]>>y[i];

   for(int i=1;i<=n;i++)
      for(int j=1;j<=n;j++){
         if(i==j)
            dis[i][j]=0;
         else
            dis[i][j]=d*(abs(x[i]-x[j])+abs(y[i]-y[j]))-a[i];
      }
   for(int k=1;k<=n;k++)
      for(int i=1;i<=n;i++)
         for(int j=1;j<=n;j++)
            dis[i][j]=min(dis[i][j],dis[i][k]+dis[k][j]);
   cout<<dis[1][n]<<endl;
   return 0;

}
