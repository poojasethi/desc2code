#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
const int n1=110;
int i,j,k,n,m;
int a[n1],x[n1],y[n1],f[n1][n1];
int main() {
    scanf("%d%d",&n,&m);
    For(i,2,n-1) scanf("%d",&a[i]);
    For(i,1,n) scanf("%d%d",&x[i],&y[i]);
    For(i,1,n)For(j,1,n) f[i][j]=i==j?0:m*(abs(x[i]-x[j])+abs(y[i]-y[j]))-a[i];
    For(k,1,n)For(i,1,n)For(j,1,n) f[i][j]=min(f[i][j],f[i][k]+f[k][j]);
    printf("%d",f[1][n]); return 0;
}
