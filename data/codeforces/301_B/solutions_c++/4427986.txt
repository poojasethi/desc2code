#include <iostream>
#include <cstdlib>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
int n,d,x[102],y[102],a[102],f[102][102];
int main() {
    cin>>n>>d;
    rep(i,n-2) cin>>a[i+1];//a[1]to a[n-1]
    rep(i,n) cin>>x[i]>>y[i];
    rep(i,n) rep(j,n) if(i!=j) f[i][j]=(abs(x[i]-x[j])+abs(y[i]-y[j]))*d-a[i];//cost from i->j ( need to minus gains from j)
    //cost if don't take shortcut
    rep(k,n) rep(i,n) rep(j,n) f[i][j] = min(f[i][j],f[i][k]+f[k][j]);//n^3,,acceptable for 100 pts
    cout << f[0][n-1]<<endl;
}
