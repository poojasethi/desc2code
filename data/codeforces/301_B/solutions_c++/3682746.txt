#include<stdio.h>
#include<algorithm>
using namespace std;
int n,costo,dist[100][100],arr[100][2],suma[100];
int main(){
    scanf("%d %d",&n,&costo);
    for(int i=1; i<n-1; i++) scanf("%d",&suma[i]);
    for(int i=0; i<n; i++) scanf("%d %d",&arr[i][0],&arr[i][1]);
    for(int i=0; i<n; i++)
        for(int j=0; j<n; j++) dist[i][j]=(abs(arr[i][0]-arr[j][0])+abs(arr[i][1]-arr[j][1]))*costo;
    for(int k=0; k<n; k++)
        for(int i=0; i<n; i++)
            for(int j=0; j<n; j++){
                if(i==j||i==k||j==k) continue;
                dist[i][j]=min(dist[i][j],dist[i][k]+dist[k][j]-suma[k]);
            }
    printf("%d",dist[0][n-1]);
}
