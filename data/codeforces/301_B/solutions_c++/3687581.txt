#include<cstdio>
#include<iostream>
#include<algorithm>
#include<set>
#include<map>
#include<vector>
#include<string>
using namespace std;
#define REP(i,n) for(int i=0;i<(int)n;++i)
int h[100][100];
int main() {
  int n,d;
  cin>>n>>d;
  vector<int> a(n),x(n),y(n);
  REP(i,n-2) {
    cin>>a[i+1];
  }
  REP(i,n) {
    cin>>x[i]>>y[i];
  }
  REP(i,n) REP(j,n) h[i][j]=(abs(x[i]-x[j])+abs(y[i]-y[j]))*d-a[i];
  REP(i,n) h[i][i]=0;
  REP(k,n) REP(i,n) REP(j,n) h[i][j]=min(h[i][j],h[i][k]+h[k][j]);
  cout<<h[0][n-1]<<endl;

}
