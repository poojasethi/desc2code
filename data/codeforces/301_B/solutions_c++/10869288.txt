#include <iostream>
#include <cmath> 
using namespace std;

int a[110];
long long n,d,m[110][110];

struct point{
	int x;
	int y;
};

int dis(point a,point b){
	return (abs(a.x-b.x)+abs(a.y-b.y));
}

int main(){
	while(cin>>n>>d){
		point p[110];
		for(int i=1;i<=n-2;i++) cin>>a[i];
		
		for(int i=0;i<n;i++) cin>>p[i].x>>p[i].y;
		
		for(int i=0;i<n;i++)
		  for(int j=0;j<n;j++)
		   if(i!=j) m[i][j] = d*dis(p[i],p[j]) - a[i];
		
		for(int k=0;k<n;k++)
		 for(int i=0;i<n;i++)
		   for(int j=0;j<n;j++)
		     m[i][j] = min(m[i][j],m[i][k]+m[k][j]);
		
		cout<<m[0][n-1]<<endl;
	}
	return 0;
}
