#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
int main(void)
{
    int a[105]={0},d,i,j,k,n,x[105],y[105],ad[105][105];
    cin>>n>>d;
    rep(i,n-2)
    cin>>a[i+1];
    rep(i,n)
    cin>>x[i]>>y[i];
    rep(i,n)
    rep(j,n)
    {
    if(i!=j)
    ad[i][j]=(abs(x[i]-x[j])+abs(y[i]-y[j]))*d-a[i];
}
    rep(k,n)
    rep(i,n)
    rep(j,n)
    {
    ad[i][j]=min(ad[i][j],ad[i][k]+ad[k][j]);
}
    cout<<ad[0][n-1];
    scanf("%d",&n);
    return 0;
}
