r = lambda: map(int, raw_input().split())
n, d = r()
a = [0] + r() + [0]
p = [r() for i in xrange(n)]
dis = [[i != j and  d * (abs(p[i][0] - p[j][0]) + abs(p[i][1] - p[j][1])) - a[j] or 0 for j in xrange(n)] for i in xrange(n)]
for k in xrange(n):
    for i in xrange(n):
        for j in xrange(n):
            dis[i][j] = min(dis[i][j], dis[i][k] + dis[k][j])
print dis[n - 1][0]
