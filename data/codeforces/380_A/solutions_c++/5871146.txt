#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int mark[100001],ci[100001],di[100001];
int m;
long long f[100001];

int find(int m,long long x){
	int head=0,tail=m;
	while (head!=tail-1){
		int mid=(head+tail)>>1;
		if (f[mid]<x) head=mid;
		else tail=mid;
	}
	if (mark[tail]==1) return ci[tail];
	x=x-f[head];
	x=x%ci[tail];
	if (x==0) x=ci[tail];
	return find(tail,x);
}

int main(){
	scanf("%d",&m);
	for (int i=1; i<=m; ++i){
		scanf("%d",&mark[i]);
		if (mark[i]==1){
			scanf("%d",&ci[i]);
			f[i]=f[i-1]+1;
			continue;
		}
		scanf("%d%d",&ci[i],&di[i]);
		long long p=ci[i];
		p=p*di[i];
		f[i]=f[i-1]+p;
	}
	int n;
	scanf("%d",&n);
	for (int i=0; i<n; ++i){
		long long x;
		cin>>x;
		printf("%d ",find(m,x));
	}
	printf("\n");
	return 0;
}
