#include <iostream>
#include <stdio.h>
#define ll long long
using namespace std;
const int MAXN = 100010;
ll m, n, a, b, c;
ll l[MAXN], r[MAXN];
ll X[MAXN], B[MAXN];
int main()
{
	cin >> m;
	for (int i = 0; i < m; ++i)
	{
		cin >> a >> b;
		if (a == 1)
		{
			X[i] = b;
			l[i] = (i ? r[i - 1] + 1 : 0);
			r[i] = l[i];
		}
		else
		{
			cin >> c;
			B[i] = b;
			l[i] = (i ? r[i - 1] + 1 : 0);
			r[i] = l[i] + b * c - 1;
		}
	}
	cin >> n;
	for (int i = 0; i < n; ++i)
	{
		cin >> a;
		--a;
		while (true)
		{
			int p = lower_bound(l, l + m, a) - l;
			if((p == m)|| (l[p] > a))
				--p;
			if (X[p] != 0)
			{
				cout << X[p] << " ";
				break;
			}
			else
				a = (a - l[p]) % B[p];
		}
	}
}
