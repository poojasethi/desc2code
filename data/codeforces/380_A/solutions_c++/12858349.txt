#include<stdio.h>
#include<algorithm>
using namespace std;
#define M 100100
int t[M],b[M],c[M];
long long  len[M];
int main(){

	int n,i,p,m;
	long long  x;

    scanf("%d",&n);

	for(i=0;i<n;i++){
		scanf("%d",&t[i]);
		if(t[i]==1){
			scanf("%d",&b[i]);
			len[i+1]=len[i]+1;
		}
		else{
			scanf("%d %d",&b[i],&c[i]);
			len[i+1]=len[i]+b[i]*c[i];
		}
	}

	scanf("%d",&m);
	for(i=0;i<m;i++){
		scanf("%I64d",&x);

		while(1){
			p=lower_bound(len,len+n+1,x)-len;
			if(t[p-1]==1){
				printf("%d ",b[p-1]);
				break;
			}

			x=(x-len[p-1]-1)%b[p-1]+1;
		}
	}
	return 0;
}
