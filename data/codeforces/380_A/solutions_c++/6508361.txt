#include <stdio.h>
#include <algorithm>
#define M 100100
using namespace std;

long long nd[M];
int t[M], l[M], c[M];

int main(){

	int i, m, n;
	long long x;
	scanf("%d", &m);
	for (i=0; i<m; i++) {
		scanf("%d", t+i);
		if (t[i]==1) {
			scanf("%d", l+i);
			nd[i+1]=nd[i]+1;
		}
		else {
			scanf("%d %d", l+i, c+i);
			nd[i+1]=nd[i]+l[i]*c[i];
		}
	}
	scanf("%d", &n);
	for (i=0; i<n; i++) {
		scanf("%I64d", &x);
		while (1){
			int p=lower_bound(nd, nd+m+1, x)-nd;
			if (t[p-1]==1) {
				printf("%d ", l[p-1]);
				break;
			}
			x=(x-nd[p-1]-1)%l[p-1]+1;
		}
	}
	return 0;
}
