#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int t[100010],x[100010],l[100010],c[100010];
long long e[100010];
int find(int u,long long p)
{
	if(t[u]==1) return x[u];
	long long tmp=p-e[u-1]-1;
	tmp%=l[u];
	tmp++;
	u=lower_bound(e,e+u,tmp)-e;
	return find(u,tmp);
}
int main()
{
	//freopen("in.txt","r",stdin);
	int m;
	scanf("%d",&m);
	e[0]=0;
	for(int i=1;i<=m;i++)
	{
		scanf("%d",&t[i]);
		if(t[i]==1)
		{
			scanf("%d",&x[i]);
			e[i]=e[i-1]+1;
		}
		else
		{
			scanf("%d%d",&l[i],&c[i]);
			e[i]=e[i-1]+(long long)l[i]*c[i];
		}
	}
	int n;
	scanf("%d",&n);
	while(n--)
	{
		long long tmp;
		cin>>tmp;
		int u=lower_bound(e,e+m+1,tmp)-e;
		printf("%d\n",find(u,tmp));
	}
	return 0;
}
