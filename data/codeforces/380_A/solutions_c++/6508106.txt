#include <cstdio>
#include <algorithm>
using namespace std;
long long l[100100];
int t[100100];
int b[100100], c[100100];
int main() {
	//freopen("in.txt","r",stdin);
	int m, n;
	long long x;
	scanf("%d", &m);
	for (int i = 0; i < m; i++) {
		scanf("%d", &t[i]);
		if (t[i] == 1) {
			scanf("%d", &b[i]);
			l[i+1] = l[i] + 1;
		} else {
			scanf("%d%d", &b[i], &c[i]);
			l[i+1] = l[i] + b[i] * c[i];
		}
	}
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%I64d", &x);
		while (1) {
			int p = lower_bound(l, l+m+1, x) - l;
			if (t[p-1] == 1) {
				printf("%d ", b[p-1]);
				break;
			}
			x = (x - l[p-1] - 1) % b[p-1] + 1;
		}
	}
	return 0;
}
