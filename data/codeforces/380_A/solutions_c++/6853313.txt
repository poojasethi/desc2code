#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int a[100001], b[100001], c[100001];
long long len[100001],x;

int main(){
    int m,n;
    scanf("%d", &m);
    for(int i=1;i<=m;i++){
        scanf("%d", &a[i]);
        if(a[i]==1) scanf("%d", b+i), len[i]=len[i-1]+1;
        else scanf("%d %d", b+i, c+i), len[i]=len[i-1]+b[i]*c[i];
    }
    scanf("%d", &n);
    for(int i=0;i<n;i++){
        scanf("%I64d", &x);
        while(true){
            int temp=lower_bound(len, len+1+m, x)-len;
            if(a[temp]==1){
                printf("%d ", b[temp]);
                break;
            }
            x=(x-len[temp-1]-1)%b[temp]+1;
        }
    }
    return 0;
}
