#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{int m;
int s[100001];
int l[100001];
int c[100001];
int num[100001];
long long int t[100001];
scanf("%d",&m);
scanf("%d",&s[0]);
scanf("%d",&num[0]);
t[0]=1;
for(int i=1;i<m;i++)
{scanf("%d",&s[i]);
if(s[i]==1)
{
    scanf("%d",&num[i]);
    t[i]=t[i-1]+1;
}
else
{
    scanf("%d%d",&l[i],&c[i]);
    t[i]=t[i-1]+l[i]*c[i];
}
}
int z;
scanf("%d",&z);
long long int x;
long long int y;
for(int p=0;p<z;p++)
{scanf("%I64d",&x);
while(1)
{
y=lower_bound(t,t+m,x)-t;
if(s[y]==1)
{
printf("%d ",num[y]);
break;  
}
x=(x-t[y-1]-1)%l[y]+1;  
}
}
    return 0;
}
