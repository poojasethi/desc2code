from sys import stdin

n = int(stdin.readline())
s = [[int(x) for x in stdin.readline().split()] for i in xrange(n)]


def result():
	st, p, x, res = [], [], 0, ""
	for i in xrange(2 * n):
		if len(st) > 0 and p[-1] + s[st[-1]][0] <= i:
			if p[-1] + s[st[-1]][1] < i: return "IMPOSSIBLE"
			res += ")"
			st.pop()
			p.pop()
		else :
			if x == n: return "IMPOSSIBLE"
			st.append(x)
			p.append(i)
			x += 1
			res += "("
	return res if len(st) == 0 else "IMPOSSIBLE"

print result()
