#!/usr/bin/env python
#-*- coding:utf-8 -*-

n = int(raw_input())

lst = [None] * n
for i in xrange(n):
    lst[i] = map(int, raw_input().split())

if lst[-1][0] != 1:
    print "IMPOSSIBLE"
    quit()

del lst[-1]
from collections import deque

deq = deque()
deq.appendleft("()")
for a, b in reversed(lst):
    a -= 1
    b -= 1

    if a == 0:
        deq.appendleft("()")
        continue

    cur = 0
    index = 0
    for p in deq:
        cur += len(p)
        index += 1
        if a <= cur <= b:
            s = "("
            for _ in xrange(index):
                s += deq.popleft()
            s += ")"
            deq.appendleft(s)
            break
    else:
        print "IMPOSSIBLE"
        quit()

print "".join(deq)
