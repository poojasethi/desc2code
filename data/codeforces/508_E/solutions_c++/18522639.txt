#include <cstdio>
#include <stack>
#include <algorithm>
using namespace std;
stack<pair<int,int> > s;
pair<int,int> p;
char c[1250];
int n,x=0,r=0;

int main()
{
//	freopen("test.in","r",stdin);
	scanf("%d",&n);
	while (1)
	{
		while (!s.empty() && x>=s.top().first && x<=s.top().second) {c[x++]=')'; s.pop();}
		if (++r<=n) 
		{
			scanf("%d%d",&p.first,&p.second);
			p.first+=x; p.second+=x;
			s.push(p); c[x++]='(';
		} else {if (x==2*n) puts(c); else puts("IMPOSSIBLE"); return 0;}
	}
}
