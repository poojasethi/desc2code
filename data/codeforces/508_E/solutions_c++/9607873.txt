#include <bits/stdc++.h>

using namespace std;

#define MAX 605

int l[MAX], r[MAX], p[MAX];
string res;
stack<int> op;

int main(int argc, char const *argv[])
{
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i)
	{
		scanf("%d %d", &l[i], &r[i]);
		op.push(i);
		p[i] = res.length();
		res += '(';
		while (!op.empty() && p[op.top()] + l[op.top()] <= res.length()) {
			if (p[op.top()] + r[op.top()] < res.length()) {
				printf("IMPOSSIBLE\n");
				return 0;
			}
			res += ')';
			op.pop();
		}
	}
	if (!op.empty()) {
		printf("IMPOSSIBLE\n");
		return 0;
	}
	else {
		printf("%s\n", res.c_str());
	}

	return 0;
}
