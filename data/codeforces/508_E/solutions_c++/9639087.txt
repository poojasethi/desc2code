#include <iostream>
#include <stack>
#include <cstdio>
#include <cstring>
using namespace std;

const int MAXN = 610;

stack<int> st;
int p[MAXN],l[MAXN],r[MAXN],i;
char s[10001];
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("a.in","r",stdin);
	freopen("a.out","w",stdout);
	#endif
	int n;
	cin >> n;
	for(i=1;i<=n;i++)
	{
		cin>>l[i]>>r[i];
		st.push(i);
		p[i]=strlen(s);
		s[strlen(s)]='(';
		while(!st.empty()&& p[st.top()]+l[st.top()]<=strlen(s))
		{
			if(p[st.top()]+r[st.top()]<strlen(s))
			{
				cout <<"IMPOSSIBLE"<< endl;
				return 0;
			}
			s[strlen(s)]=')';
			st.pop();
		}
	}
	if(!st.empty())
		cout <<"IMPOSSIBLE"<< endl;
	else
		cout<<s<<endl;
}
