#include <cstdio>
#include <stack>

using namespace std;

char b[2000] = {0};
int l, r;

int main (){
  int n, t = 0;
  scanf("%d", &n);
  stack <pair <int, int> > st;
  bool flag = true;
  for(int i = 0; i < n; i++) {
    scanf("%d%d", &l, &r);
    st.push(make_pair(t + l, t + r));
    b[t] = '(';
    t++;
    while(!st.empty() && t >= st.top().first && t <= st.top().second) {
      st.pop();
      b[t] = ')';
      t++;
    }
    if(!st.empty() && t > st.top().second) {
    flag = false;
    break;
    }
  }
  if(flag && st.empty()) printf("%s", b);
  else printf("IMPOSSIBLE\n");
  return 0;
}
