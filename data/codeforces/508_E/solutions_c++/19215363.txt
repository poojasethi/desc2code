#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<iostream>
#include<climits>
#include<string>
#include<cstdio>
#define LL long long

using namespace std;

int n,m,q[1000],X[1000],Y[1000];
string ans;

int main()
{
	cin>>n;
	int x,y,top=0,l=0;ans="";
	for (int i=1;i<=n;i++)
	{
		cin>>x>>y;
		ans+="(";l++;
		q[++top]=l;
		X[top]=x; Y[top]=y;
		while (l+1-q[top]>=X[top])
		{
			if (top==0) break;
			l++;
			ans+=")";
			if (l-q[top]>Y[top]) {cout<<"IMPOSSIBLE"<<endl;return 0;}
			top--;
		}
	}
	if (l!=n*2) {cout<<"IMPOSSIBLE"<<endl;return 0;}
	cout<<ans<<endl; 
	return 0;
}
