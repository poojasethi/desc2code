#include<iostream>
#include<cstdio>
#include<cstring>
#include<stack>
using namespace std;
const int maxn = 610*2;
stack<int> st;
int l[maxn],r[maxn],p[maxn];
int n;
int main()
{
	while(~scanf("%d",&n)&&n)
	{
		while(!st.empty()) st.pop();
		string s;
		bool flag=true;
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&l[i],&r[i]);
			st.push(i);
			p[i]=s.size();
			s+='(';
			while(!st.empty()&&p[st.top()]+l[st.top()]<=s.size())
			{
				if(p[st.top()]+r[st.top()]<s.size())
					flag=false;
				s+=')';
				st.pop();
			}
		}
		if(!st.empty())
			flag=false;
		if(!flag)
			puts("IMPOSSIBLE");
		else
			cout<<s<<endl;
	}
	return 0;
}
