#include<bits/stdc++.h>

using namespace std;

#define MaxN 605

int n,i;

int L[MaxN],R[MaxN],A[MaxN];

string s="";

stack<int> st;

int main(){
	
	scanf("%d",&n);

	for(i=1 ; i<=n ; i++)
		scanf("%d %d",L+i,R+i);
	
	for(i=1 ; i<=n ; i++){
		
		st.push(i);

		A[i]=s.size();

		s+='(';
		
		while(!st.empty() and A[st.top()] + L[st.top()] <= s.size()){
			
			if(A[st.top()] + R[st.top()] < s.size())
				return cout<<"IMPOSSIBLE",0;

			st.pop();
			s+=')';
		}

	}

	if(st.size())
		cout<<"IMPOSSIBLE";
	else
		cout<<s;

	return 0;
}
