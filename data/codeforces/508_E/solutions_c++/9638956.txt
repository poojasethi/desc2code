#include <iostream>
#include <stack>
using namespace std;

const int MAXN = 610;

stack <int> st;
string s;
int p[MAXN], l[MAXN], r[MAXN];

int main()
{
    int N;
    cin >> N;
    for(int i = 0; i < N; ++i) {
       cin >> l[i] >> r[i];
       st.push(i);
       p[i] = s.size();
       s.push_back('(');
       while(!st.empty() && p[st.top()] + l[st.top()] <= s.size()) {
          if(p[st.top()] + r[st.top()] < s.size()) {
            cout << "IMPOSSIBLE" << '\n';
            return 0;
          }
          s.push_back(')');
          st.pop();
       }
    }
    if(!st.empty())
      cout << "IMPOSSIBLE" << '\n';
    else
      cout << s << '\n';
    return 0;
}
