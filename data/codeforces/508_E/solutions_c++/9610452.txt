#include <iostream>
#include <stack>
using namespace std;
stack<int> p;
string s;
int l[605],r[605];
int n;
int pos[605];

int main()
{
    cin>>n;
    for (int i = 1; i<=n; i++)
    {
        p.push(i);
        s+='(';
        pos[i] = s.size();
        cin>>l[i]>>r[i];
        while (!p.empty() && pos[p.top()] + l[p.top()] <= s.size() + 1)
        {
            if (pos[p.top()] + r[p.top()] < s.size() + 1)
            {
                cout<<"IMPOSSIBLE";
                return 0;
            }
            s+=')';
            p.pop();
        }
    }
    if (s.size()!=2*n) cout<<"IMPOSSIBLE";
    else
        cout<<s;
    return 0;
}
