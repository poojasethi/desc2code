#include <algorithm>
#include <iostream>
#include <cassert>
#include <cstdio>
#include <cstring>

using namespace std;

int stack[610];

int N;

pair<int,int> range[610];

char seq[1210];

int K,S;

int main(){

	int i;

	scanf("%d",&N);

	for(i = 0; i < N; ++i){
		scanf("%d%d",&range[i].first,&range[i].second);
	}


	for(i = 0; i < 2*N; ++i){
		if(S && i >= range[stack[S-1]].first){
			if(i > range[stack[S-1]].second){
				printf("IMPOSSIBLE\n");
				return 0;
			}

			seq[i] = ')';
			--S;
			continue;
		}

		if(K >= N){
			printf("IMPOSSIBLE\n");
			return 0;
		}

		stack[S++] = K;
		range[K].first += i;
		range[K].second += i;
		++K;
		seq[i] = '(';
	}

	printf("%s\n",seq);

	return 0;
}
