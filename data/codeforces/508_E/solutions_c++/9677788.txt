#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin>>n;
    string s="";
    int l[n],r[n],p[n];
    stack<int>stak;
    for(int i=0;i<n;i++)
    {
        cin>>l[i]>>r[i];
        p[i]=s.length();
        stak.push(i);
        s+='(';
        while(!stak.empty()&&p[stak.top()]+l[stak.top()]<=s.length())
        {
            if(p[stak.top()]+r[stak.top()]<s.length())
            {
                cout<<"IMPOSSIBLE"<<endl;return 0;
            }
            stak.pop();
            s+=')';
        }
    }
    if(!stak.empty())
    {
        cout<<"IMPOSSIBLE"<<endl;return 0;
    }
    cout<<s<<endl;
    return 0;
}
