#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

int l[1000], r[1000];
vector<int> v;
string s;

int main()
{
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
	{
		cin >> l[i] >> r[i];
		s += '(';
		v.push_back(i);
		while(v.size() && l[v.back()] <= 1)
		{
			int x = v.back();
			if(r[x] < 1)
			{
				cout << "IMPOSSIBLE"  << endl;
				return 0;
			}
			for(int i = 0; i < v.size(); i++)
				l[v[i]] -= 2, r[v[i]] -= 2;
			s += ')';
			v.pop_back();
		}
	}
	if(v.size())
		cout << "IMPOSSIBLE"  << endl;
	else
		cout << s << endl;
	return 0;
}
