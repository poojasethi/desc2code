#include <bits/stdc++.h>
using namespace std;
stack<int> st;
	int l[601],r[610],k[601];
	string s;
int main() {
	int n;
	cin>>n;

	
	for(int i=0;i<n;i++)
	{
		cin>>l[i]>>r[i];
	
		st.push(i);
			k[i]=s.length();
		s+='(';
		while(!st.empty() && k[st.top()]+l[st.top()]<= s.length() )
		{
			if(k[st.top()]+ r[st.top()] <s.length())
			{
				cout<<"IMPOSSIBLE"<<endl;
	return 0;
	
			}
			s+=')';
			st.pop();
			
		}
		
	}
	if(!st.empty())
	cout<<"IMPOSSIBLE"<<endl;
	else
	cout<<s<<endl;
	return 0;
}
