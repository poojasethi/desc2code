#include<bits/stdc++.h>
using namespace std;

stack<int> S;
string s;

int l[605],r[605],p[605];
int main()
{
	int n,i;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&l[i],&r[i]);
		S.push(i);
		p[i] = s.length();
		s+="(";
		
		while(!S.empty() && p[S.top()] + l[S.top()] <= s.length())
		{
			if(p[S.top()] + r[S.top()] <s.length())
			{
				printf("IMPOSSIBLE");
				return 0;
			}
			S.pop();
			s += ")";
		}
	}
	if(!S.empty())
	{
		printf("IMPOSSIBLE");
		return 0;
	}
	printf("%s",s.c_str());
	return 0;
}
