#include <iostream>
#include <stack>

using namespace std;

stack <int> s, e;

int main()
{
	int n, j = 0;
	string ans;
	cin >> n;
	for (int i = 0; i < 2 * n; i++)
	{
		if (j < n)
		{
			int s1, e1;
			cin >> s1 >> e1;
			s.push(ans.size() + s1);
			e.push(ans.size() + e1);
			ans += '(';
			j++;
		}
		while (!s.empty() && s.top() <= ans.size() && ans.size() <= e.top())
		{
			ans += ')';
			s.pop();
			e.pop();
		}
	}
	if (s.empty())
		cout << ans;
	else
		cout << "IMPOSSIBLE";
	cerr << endl;
	return 0;
}
