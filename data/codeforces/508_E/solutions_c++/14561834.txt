#include <bits/stdc++.h>
using namespace std;

const int maxn = 606;

int n,l[maxn],r[maxn],p[maxn];
string ans;
stack < int > st;

int main(){
	scanf("%d",&n);
	for( int i=0 ; i < n ; i++ ){
		scanf("%d%d",&l[i],&r[i]);
		st.push(i);
		p[i] = ans.size();
		ans += '(';
		while( !st.empty() and p[st.top()] + l[st.top()] <= (int)ans.size() ){
			if( p[st.top()] + r[st.top()] < (int)ans.size() ){
				puts("IMPOSSIBLE");
				return 0;
			}
			ans += ')';
			st.pop();
		}
	}
	
	if( st.empty() )
		cout << ans << '\n';
	else
		puts("IMPOSSIBLE");
	
	return 0;
}
