#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <stack>
#include <string>
using namespace std;

stack<int> st;
string ans;
int pos[1000],l[1000],r[1000];
int n;
int main()
{
	scanf("%d",&n);
	for (int i=0;i<n;++i)
	{

		scanf("%d%d",&l[i],&r[i]);
		st.push(i);
		pos[i]=ans.length();
		ans+='(';
		while (!st.empty()&&pos[st.top()]+l[st.top()]<=ans.length())
		{
			if (pos[st.top()]+r[st.top()]<ans.length())
			{
				printf("IMPOSSIBLE\n");
				return 0;
			}
			ans+=')';
			st.pop();
		}
	}
	if (!st.empty()) printf("IMPOSSIBLE\n");
	else cout<<ans;
	return 0;
}
