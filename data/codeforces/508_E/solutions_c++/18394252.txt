#include<cstdio>
#define MAXN 600
struct node{int p,l,r;}s[MAXN+5];
int sn,an;
char ans[2*MAXN+5];
int main()
{
	int n,i,l,r;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&l,&r);
		s[++sn].p=an;s[sn].l=l;s[sn].r=r; 
		ans[an++]='(';
		while(sn&&an-s[sn].p>=s[sn].l&&an-s[sn].p<=s[sn].r)
		{
			sn--;
			ans[an++]=')';
		}
	}
	if(sn){puts("IMPOSSIBLE");return 0;}
	ans[an]='\0';
	printf("%s",ans); 
}
