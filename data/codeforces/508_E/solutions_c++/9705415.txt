#include<bits/stdc++.h>

using namespace std;

int n,m,j,k,i;
int l[605],r[605],q[605];
stack<int> St;
string S;

int main(){

	scanf("%d",&n);

	for(i=1 ; i<=n ; i++)
	{
		scanf("%d %d",&l[i],&r[i]);
		
		St.push(i);
		q[i] = S.length();
		S += '(';

		while(St.size() && q[St.top()] + l[St.top()] <= S.length())
		{
			if(q[St.top()] + r[St.top()] < S.length())
				return cout << "IMPOSSIBLE",0;
			
			S += ')';
			St.pop();
		}
	}

	if(St.size())
		return cout << "IMPOSSIBLE",0;

	cout << S;

	return 0;
}
