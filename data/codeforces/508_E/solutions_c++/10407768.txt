#include <bits/stdc++.h>
typedef long long LL;
#define rep(i,n) for(int i = 0;i < n;i ++)

const int N = 2015;
#define pair(x, y) std::make_pair(x, y)
typedef std::pair<int, int> P;
std::stack<P> stack;
std::string answer;
int l[N], r[N];
int main() {
	int n;
	scanf("%d", &n);
	rep(i, n) {
		scanf("%d%d", l + i, r + i);
		answer += '(';
		stack.push(pair(i, answer.size()));
		while(!stack.empty()) {
			P u = stack.top();
			int d = answer.size() - u.second + 1;
			if(l[u.first] <= d && d <= r[u.first]) {
				answer += ')';
				stack.pop();
			} else	break;
		}
	}
	if(stack.size())
		puts("IMPOSSIBLE");
	else
		std::cout << answer << std::endl;
	return 0;
}
