#include<bits/stdc++.h>
using namespace std;

int main(int argc, char const *argv[])
{
	int n,j,k,i;
	stack<int> x,l,r;
	string s;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>j>>k;
		l.push(j);
		r.push(k);
		x.push(s.length());
		s+='(';
		while(!x.empty()&&l.top()+x.top()<=s.length())
		{
			if(r.top()+x.top()<s.length())
			{
				cout<<"IMPOSSIBLE";
				return 0;
			}
			else s+=')';
			x.pop();
			l.pop();
			r.pop();
		}
	}
	if(x.empty()) cout<<s;
	else cout<<"IMPOSSIBLE";
	return 0;
	return 0;
}
