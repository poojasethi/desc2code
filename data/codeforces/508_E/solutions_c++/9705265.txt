#include<bits/stdc++.h>

using namespace std;

int l[615],r[615],q[615];
int n,m,j,k,i;
stack<int> St;
string S;

int main(){

	scanf("%d",&n);

	for(i=1 ; i<=n ; i++)
	{
		scanf("%d %d",&l[i],&r[i]);
		St.push(i);
		q[i] = S.length();
		S += '(';
		
		while(St.size() && q[St.top()] + l[St.top()] <= S.length())
		{
			if(q[St.top()] + r[St.top()] < S.length())
				return cout << "IMPOSSIBLE",0;

			S += ')';
			St.pop();
		}
	}
		
	if(St.size())
		return cout << "IMPOSSIBLE",0;

	cout << S;

	return 0;
}
