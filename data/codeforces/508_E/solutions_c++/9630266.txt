#include <bits/stdc++.h>

using namespace std;

const int maxN = 605;

stack<int> st;
string s;
int p[maxN], l[maxN], r[maxN];

int main(){
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> l[i] >> r[i];
        st.push(i);
        p[i] = s.length();
        s += '(';
        while (!st.empty() && p[st.top()] + l[st.top()] <= s.length()) {
            if (p[st.top()] + r[st.top()] < s.length()) {
                cout << "IMPOSSIBLE\n";
                return 0;
            }
            s += ')';
            st.pop();
        }
    }
    if (!st.empty()) cout << "IMPOSSIBLE\n";
    else cout << s << "\n";
    
    return 0;
}
