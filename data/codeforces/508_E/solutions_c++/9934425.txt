#include<cstdio>
#include<cstring>
#include<stack>
using namespace std;
char s[1210];
int n,l[610],r[610],pos[610];
stack<int> st;
int main()
{
    int i,j,k,num=0;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
       scanf("%d%d",&l[i],&r[i]);
    for(i=1;i<=n*2;i++)
    {
        if(!st.empty())
        {
            k=st.top();
            if(pos[k]+l[k]<=i && i<=pos[k]+r[k])
            {
                st.pop();
                s[i]=')';
                continue;
            }
        }
        s[i]='(';
        num++;
        st.push(num);
        pos[num]=i;
    }
    if(num==n && st.empty())
      printf("%s\n",s+1);
    else
      printf("IMPOSSIBLE\n");
}
