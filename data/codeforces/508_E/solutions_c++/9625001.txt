#include <cstdio>
#include <utility>
#include <stack>
#define N_Max 2*600

using namespace std;

stack< pair<int, int> >  st;

char ans[N_Max];
int n, l, r;

int main() {
	scanf("%d", &n);

	int p = 0;
	for (int i = 1; i <= n; i++) {
		scanf("%d%d", &l, &r);

		st.push(make_pair(l+p, r+p));
		ans[p++] = '(';

		while (!st.empty() && p >= st.top().first && p <= st.top().second) {
			ans[p++] = ')';
			st.pop();
		}
	}

	if (st.empty())
		for (int i = 0; i < p; i++)  printf("%c", ans[i]);
	else
		printf("IMPOSSIBLE\n");

}
