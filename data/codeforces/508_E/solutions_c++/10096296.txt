#include <iostream>
#include <stack>

using namespace std;

const int N = 600;

int n;
int l[N], r[N], p[N];
stack<int> st;
string s;

int main() {
    ios_base::sync_with_stdio(false);

    cin >> n;

    for (int i = 0; i < n; i++) {
        cin >> l[i] >> r[i];

        st.push(i); p[i] = s.size(); s += '(';

        while (!st.empty() && p[st.top()] + l[st.top()] <= s.size()) {
            if (p[st.top()] + r[st.top()] < s.size()) {
                cout << "IMPOSSIBLE"; return 0;
            }

            s += ')'; st.pop();
        }
    }

    if (!st.empty()) {
        cout << "IMPOSSIBLE"; return 0;
    }

    cout << s;

    return 0;
}
