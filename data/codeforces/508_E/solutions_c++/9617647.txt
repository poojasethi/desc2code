#include <bits/stdc++.h>
using namespace std;
vector <int> a,b,c;
stack <int> r;
string s;
int main(){
	int n;
	cin >> n;
	c.assign(n,0);
	for (int i=0;i<n;i++){
		int q,w;
		cin >> q >> w;
		a.push_back(q);
		b.push_back(w);
		}
	for (int i=0;i<n;i++){
		int z=0;
		r.push(i);
		c[i]=s.length();
		s+='(';
	
		while(!r.empty() && c[r.top()]+a[r.top()]<=s.length()){
			if (c[r.top()]+b[r.top()]<s.length()){
				cout<<"IMPOSSIBLE";
				return 0;
			}
			s+=')';
			r.pop();
		
		}
	}
	if (!r.empty())
		cout<<"IMPOSSIBLE";
	else
		cout<< s << endl;
	return 0;
}
