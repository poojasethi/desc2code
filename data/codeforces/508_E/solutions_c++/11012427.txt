#include <bits/stdc++.h>
using namespace std;
//printf("%.7f\n", double value);

typedef long long ll;
int n;
int l[601], r[601], p[601];

int main () {
  cin >> n;
  string s;
  stack<int> st;
  for(int i=0; i<n; i++) {
    cin >> l[i] >> r[i];
    p[i] = s.size();
    st.push(i);
    s += '(';
    while(!st.empty() && p[st.top()] + l[st.top()] <= s.size()) {
      if(p[st.top()] + r[st.top()] < s.size()) {
        cout <<"IMPOSSIBLE"<< endl;
        return 0;
      }
      s += ')';
      st.pop();
    }
  }
  if(!st.empty()) {
    cout << "IMPOSSIBLE" << endl;
    return 0;
  }
  cout << s << endl;
  return 0;
}
