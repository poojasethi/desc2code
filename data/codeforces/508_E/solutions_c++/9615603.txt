#include<bits/stdc++.h>
using namespace std;
int l[666],r[666],p[666];
stack<int>S;
string rep;
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d%d",l+i,r+i);
    for(int i=1;i<=n;i++)
    {
        rep.push_back('(');
        S.push(i);
        p[i]=rep.size();
        while(!S.empty()&&p[S.top()]+l[S.top()]<=rep.size()+1)
        {
            if(p[S.top()]+r[S.top()]<rep.size()+1){puts("IMPOSSIBLE");return 0;}
            S.pop();
            rep.push_back(')');
        }
    }
    if(!S.empty())puts("IMPOSSIBLE");
    else printf("%s\n",rep.c_str());
}
