#include <iostream>
#include <cstring>
#include <vector>
#include <stack>

using namespace std;

stack<int> s;
string ans;
int t;
int p[600], l[600], r[600];

int main() {
  cin >> t;
  for (int i = 0; i < t; i++) {
    cin >> l[i] >> r[i];
    s.push(i);
    p[i] = ans.length();
    ans += '(';
    while (!s.empty() && p[s.top()] + l[s.top()] <= ans.length()) {
      if (p[s.top()] + r[s.top()] < ans.length()) {
        cout << "IMPOSSIBLE\n";
        return 0;
      }
      ans+= ')';
      s.pop();
    }
  }
  if (!s.empty()) {
    cout << "IMPOSSIBLE\n";
  } else {
    cout << ans << endl;
  }
}
