#include <bits/stdc++.h>

#define NN 6100

using namespace std;

int L[NN], R[NN], D[NN];
int p[NN];

char str[2*NN];

int main (){
	
	int N, i, j;
	scanf("%d",&N);
	for(i=0; i<N; i++){
		scanf("%d %d",L+i, R+i);
	}
	
	int t=0;
	for(i=N-1; i>=0; i--){
		int tot=0;
		p[t]=0;
		for(j=t; j>=0; j--){
			tot+=p[j];
			if(tot+1>=L[i] && tot+1<=R[i]) break;
		}
		if(j<0) break;
		t=j;
		p[t++]=tot+2;
		D[i]=tot+1;
	}
	
	if(i>=0) puts("IMPOSSIBLE");
	else {
		memset(p, 0, sizeof(p));
		for(i=0, j=0; i<2*N; i++){
			if(!p[i]){
				str[i]='(';
				str[i+D[j]]=')';
				p[i+D[j]]=1;
				j++;
			}
		}
		puts(str);
	}
	
	return 0;
}
