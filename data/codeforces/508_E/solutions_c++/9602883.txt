#include<bits/stdc++.h>
using namespace std;
char s[1205];

stack<int>L,R;
int main()
{
    int n,i,l,r;
    cin>>n;
    memset(s,0,sizeof(s));
    int pos=0;
    for(i=1;i<=n;++i)
    {
        scanf("%d %d",&l,&r);
        L.push(pos+l);R.push(pos+r);
        s[pos++]='(';
        while(!L.empty()&&L.top()<=pos&&pos<=R.top())
        {
            s[pos++]=')';
            L.pop(),R.pop();
        }
        if(!L.empty()&&pos>R.top())
        {
            puts("IMPOSSIBLE");
            return 0;
        }
    }
    if(!L.empty()) puts("IMPOSSIBLE");
    else puts(s);
    return 0;
}
