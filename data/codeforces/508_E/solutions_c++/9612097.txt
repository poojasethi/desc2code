#include <iostream>
#include <stack>
#define MAXN 601
using namespace std;

stack<int> st;
string s;
int pos[MAXN], l[MAXN], r[MAXN];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>l[i]>>r[i];
		st.push(i);
		pos[i]=s.length();
		s+='(';
		while(!st.empty() && pos[st.top()]+l[st.top()]<=s.length())
		{
			if(pos[st.top()]+r[st.top()] < s.length())
			{
				cout<<"IMPOSSIBLE"<<endl;
				return 0;
			}
			s+=')';
			st.pop();
		}
	}
	if(!st.empty())
		cout<<"IMPOSSIBLE"<<endl;
	else
		cout<<s<<endl;
    return 0;
}
