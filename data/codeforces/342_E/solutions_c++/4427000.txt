const int MAXN = 200005;
int n,m,a,b;
int Q[MAXN],ds[MAXN],st[MAXN];
int V[MAXN*2],N[MAXN*2],F[MAXN];
int tot = 0,top,hd = 0,tl = 0;
bool v[MAXN];
void add(int a,int b)
{
    ++tot;
    V[tot] = b;
    N[tot] = F[a];
    F[a] = tot;
}
void spread()
{
    while (hd != tl)
    {
        int nw = Q[++hd];
        for (int p = F[nw]; p > 0; p = N[p])
            if (ds[V[p]] > ds[nw] + 1)
            {
                ds[V[p]] = ds[nw] + 1;
                Q[++tl] = V[p];
            }
    }hd = 0,tl = 0;
}
