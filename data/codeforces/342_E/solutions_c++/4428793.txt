#include<algorithm>
#include<cstring>
#include<cstdio>
#include<vector>
#include<cmath>
#include<queue>
#define N 100001
using namespace std;
queue<int> q;
int first[N],next[N<<1],ver[N<<1],d[N],tot;
void add(int a,int b)
{
    ver[++tot]=b;
    next[tot]=first[a];
    first[a]=tot;
}
void bfs()
{
    int x;
    while (!q.empty())
    {
        x=q.front();
        q.pop();
        for (int p=first[x];p;p=next[p])
            if (d[x]+1<d[ver[p]])
            {
                d[ver[p]]=d[x]+1;
                q.push(ver[p]);
            }
    }
}
int main()
{
    int n,m,a,b;
    scanf("%d%d",&n,&m);
    for (int i=1;i<n;i++)
    {
        scanf("%d%d",&a,&b);
        add(a,b);
        add(b,a);
    }
    for (int i=1;i<=n;i++) d[i]=n;
    d[1]=0;
    q.push(1);
    bfs();
    while (m--)
    {
        scanf("%d%d",&a,&b);
        if (a==1)
        {
            d[b]=0;
            q.push(b);
        }
        else
        {
            bfs();
            printf("%d\n",d[b]);
        }
    }
    return 0;
}
