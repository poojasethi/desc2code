n,k,q = map(int, raw_input().split())
fship = map(int, raw_input().split())

window = []
for i in range(q):
	a, fid = map(int, raw_input().split())
	if a == 1:
		if len(window) < k:
			window.append(fid)
			window.sort(key = lambda x : fship[x-1])
		elif fship[fid-1] > fship[window[0]-1]:
			window[0] = fid
			window.sort(key = lambda x : fship[x-1])
	else:
		if fid in window:
			print 'YES'
		else:
			print 'NO'
