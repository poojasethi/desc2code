import sys
tmp = []
n,k,q = map(int,sys.stdin.readline().split())
a = [int(i) for i in sys.stdin.readline().split()]
for i in range(q):
    type,id = map(int,sys.stdin.readline().split())
    if type == 1:
        tmp.append(a[id-1])
        if len(tmp) > k:
            tmp.remove(min(tmp))
    else:
        print "YES" if a[id-1] in tmp else "NO"
