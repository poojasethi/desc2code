n, k, q = map(int, raw_input().split())
t = [int(i) for i in raw_input().split()]

f = []
for i in xrange(q):
	y, d = map(int, raw_input().split())	
	if y == 1:
		f += [t[d - 1]]
		for i in xrange(len(f) - 1, 0, -1):
			if f[i] > f[i - 1]:
				f[i], f[i - 1] = f[i - 1], f[i]
			else:
				break
		if len(f) > k:
			f = f[0 : k]
	else:
		if t[d - 1] in f:
			print("YES")
		else:
			print("NO")
