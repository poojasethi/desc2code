#include<bits/stdc++.h>
using namespace std;

int a[200005];

int main(){
	int i,n,k,q,t1,t2;
	set<int>st;
	scanf("%d%d%d",&n,&k,&q);
	for(i=0;i<n;++i){
		scanf("%d",&a[i]);
	}
	for(i=0;i<q;++i){
		scanf("%d%d",&t1,&t2);
		if(t1==1){
			st.insert(a[t2-1]);
			if(st.size()>k){
				st.erase(st.begin());
			}
		}else{
			if(st.find(a[t2-1])!=st.end()){
				printf("YES\n");
			}else{
				printf("NO\n");
			}
		}
	}
}
