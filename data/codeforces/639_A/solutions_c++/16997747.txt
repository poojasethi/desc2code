#include <iostream>
#include <algorithm>
#include <set>
using namespace std;

int t[150010];

int main() {
	int n, k, q;
	cin >> n >> k >> q;
	for (int i = 0; i < n; i++) {
		cin >> t[i];
		t[i] *= -1;
	}
	set<int> onlineList;
	for (int i = 0; i < q; i++) {
		int type, id;
		cin >> type >> id;
		if (type == 1) {
			onlineList.insert(t[id - 1]);
			if (onlineList.size() > k)
				onlineList.erase(--onlineList.end());
		}
		if (type == 2) 
			cout << ((onlineList.find(t[id - 1]) != onlineList.end()) ? "YES" : "NO") << endl;
	}
}
