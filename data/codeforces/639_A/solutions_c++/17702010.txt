#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n,k,q;
	cin>>n>>k>>q;
	int a[n];
	set<int> data;
	for(int i = 0; i<n; i++)
	{
		cin>>a[i];
	}
	
	int x,y;
	for(int i = 0; i< q; i++)
	{
		cin>>x>>y;
		if(x == 1)
		{
			data.insert(a[y-1]);
			if(data.size() > k)
			{
				data.erase(data.begin());
			}
		}
		else if(x ==2)
		{
			if(data.find(a[y-1])!=data.end())
			{
				cout<<"YES\n";
			}
			else
			{
				cout<<"NO\n";
			}
		}
	}
	
	return 0;
}
