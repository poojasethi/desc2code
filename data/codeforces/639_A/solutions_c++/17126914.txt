#include <bits/stdc++.h>

using namespace std;

const int N = (int)2e5 + 10;
const int T = (int)1e9 + 10;

int n, k, q, t[N];
set<pair<int, int> > social;

int main()
{
    cin >> n >> k >> q;
    for (int i = 0; i < n; ++i) cin >> t[i];
    while (q--){
        int type, id;
        cin >> type >> id; id--;
        if (type == 1){
            social.insert(make_pair(t[id], id));
            while (social.size() > k) social.erase(social.begin());
        } else
            cout << (social.count(make_pair(t[id], id))? "YES" : "NO") << endl;
    }
    return 0;
}
