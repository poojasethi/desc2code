#include <iostream>
using namespace std;
int main()
{
	int n;string s;
	cin>>n>>s;
	int ans = -1, r = -1;
	for(int i = 0; i < n; i++)
	{
		int a; cin>>a;
		if(s[i] == 'R')
			r = a;
		else if(r != -1)
		{
			if(ans == -1)
				ans = (a - r) / 2;
			else
				ans = min(ans, (a - r) / 2);
		}
	}
	cout<<ans<<endl;
	return 0;
}
