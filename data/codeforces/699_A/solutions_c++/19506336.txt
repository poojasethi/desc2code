#include <bits/stdc++.h>
using namespace std;
int n, x, l, ans=INT_MAX;
char s[200005];
int main () {
	scanf("%d %s", &n, s);
	for (int i=0; i<n; i++) {
		scanf("%d", &x);
		if (i>0&&s[i]=='L'&&s[i-1]=='R') ans=min(ans, (x-l)/2);
		l=x;
	}
	printf("%d\n", ans==INT_MAX?-1:ans);
	return 0;
}
