#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	int x[200000];
	string s;
	cin>>n;
	cin>>s;
	for(int i=0;i<n;i++)
	cin>>x[i];
	int minn=-1;
	for(int i=0;i<n-1;i++)
	 {
	 	if(s[i]=='R' && s[i+1]=='L')
	 	 {
	 	 	int t=(x[i+1]-x[i])/2;
	 	 	if(minn==-1 || t<minn)
	 	 	minn=t;
	 	 	
		  }
	 }
	 cout<<minn<<endl;
	return 0;
}
