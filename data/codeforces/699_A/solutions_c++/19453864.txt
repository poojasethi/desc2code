#include <stdio.h>

int main()
{
	int n;
	scanf("%d",&n);

	char s[n+3];
	scanf("%s",s);

	int a[n+1],i,cnt,flag=0,min=1000000005;

	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}

	for(i=0;i<n;i++)
	{
		if(s[i]=='R' && s[i+1]=='L')
		{
			flag=1;
			cnt=a[i+1]-a[i];

			if(min>cnt)	min=cnt;
		}
	}

	if(flag==0)		printf("-1\n");
	else			printf("%d\n",min/2);

	return 0;
}
