#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
int n,x[200005];
char s[200005];
int main(){
	cin>>n;
	gets(s);gets(s);
	int ans=-1;
	cin>>x[0];
	for(int i=0;i<n-1;i++){
		cin>>x[i+1];
		if (s[i]=='R' && s[i+1]=='L') {
			if (ans==-1) ans=(x[i+1]-x[i])/2;
			ans=min(ans,(x[i+1]-x[i])/2);
		}
	}
	cout<<ans;
	return 0;
}
