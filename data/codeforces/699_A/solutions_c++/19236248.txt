#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int n,i,a[200010],ans=1e9;
char s[200010];
int main(){
	scanf("%d",&n);
	scanf("%s",s+1);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<n;i++)
		if(s[i]=='R'&&s[i+1]=='L'&&!((a[i+1]-a[i])&1))ans=min(ans,(a[i+1]-a[i])>>1);
	printf("%d",ans==1e9?-1:ans);
	return 0;
}
