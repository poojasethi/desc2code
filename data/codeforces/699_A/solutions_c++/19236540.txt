#include<bits/stdc++.h>
using namespace std;

const int INF=0x3f3f3f3f;
const int maxn=200010;

int n;
char s[maxn];
int pos[maxn];
int main(){
	scanf("%d%s",&n,s);
	for(int i=0;i<n;++i)scanf("%d",&pos[i]);
	int res=INF;
	for(int i=1;i<n;++i)if(s[i-1]=='R'&&s[i]=='L')
		res=min(res,(pos[i]-pos[i-1])/2);
	printf("%d\n",res>=INF ? -1 : res);
	return 0;
}
