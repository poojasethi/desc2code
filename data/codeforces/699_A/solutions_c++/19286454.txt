#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,a,b,t,m=1000000002;
	char c[200004];
	cin>>n>>c>>a;
	for(int i=1;i<n;i++)
	{
        cin>>b;
	if(c[i]=='L'&&c[i-1]=='R')
	{
	t=(b-a)/2;
	if(t<m)
	{
	m=t;
	}
	}
	a=b;
	}
	if(m==1000000002)cout<<"-1"<<endl;
	else cout<<m<<endl;
	return 0;
}
