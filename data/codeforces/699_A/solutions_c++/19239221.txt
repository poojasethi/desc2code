#include<iostream>
using namespace std;
int main()
{
long n,i,v,min=-1,cy=1;
cin>>n;
char s[n+1];
long a[n];
cin>>s;
for(i=0;i<n;i++)
cin>>a[i];
for(i=0;s[i]!='\0';i++)
{
if(s[i]=='R'&&s[i+1]=='L'&&i+1<n)
{
v=((a[i]+a[i+1])/2)-a[i];
if(cy==1)
{min=v;cy+=1;}
else if(min>v)
min=v;
}
}
cout<<min<<endl;
return 0;
}
