#include<bits/stdc++.h>
using namespace std;
long long int n,a[200010],max1=3000000007;
string s;
int main ()
{
	cin>>n>>s;
	for(int i=0;i<n;i++)
	cin>>a[i];
	for(int i=0;i<s.length();i++)
	{
		if(s[i]=='R'&& s[i+1]=='L')
		{
			if(max1>(a[i+1]-(a[i]+1)))
			max1=a[i+1]-a[i];
		}
	}
	if(max1==3000000007)
	cout<<-1;
	else
	cout<<max1/2;
	return 0;	
}
