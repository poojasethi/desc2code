#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int  main()
{
	long long int n,p[200000],t,i;
	cin>>n;
	string d;
	cin>>d;
	t=-1;
	for(i=0;i<n;i++)
		cin>>p[i];
	for(i=0;d[i]!='\0';i++)
	{
		if(i!=n-1)
		{
			if(d[i]=='R' && d[i+1]=='L')
			{
				if(t==-1 || t>(p[i+1]-p[i])/2)
					t=(p[i+1]-p[i])/2;
			}
		}
	}
	cout<<t;
	return 0;
}
