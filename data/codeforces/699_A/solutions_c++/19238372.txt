#include<cstdio>
#include<algorithm>
using namespace std;
const int N=200005;
int n,a[N],ans;
char s[N]; 
int main(){
	scanf("%d",&n);
	scanf("%s",s+1);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]); 
	ans=1e9;
	for (int i=1;i<n;i++)
	 if (s[i]=='R' && s[i+1]=='L') ans=min(ans,(a[i+1]-a[i])/2); 
	if (ans!=1e9) printf("%d",ans);
	else puts("-1");
	return 0;
}
