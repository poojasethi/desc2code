#include <iostream>
using namespace std;

int main() {
long long n;
cin>>n;
string s;
cin>>s;
long long a[n];
cin>>a[0];
long long min=1000000009;

for(int i=1;i<n;i++){
cin>>a[i];
if(s[i-1]=='R' && s[i]=='L' && min>(a[i]-a[i-1]))
min=a[i]-a[i-1];
}
if(min==1000000009)
{
cout<<"-1"<<endl;
}else{
cout<<min/2<<endl;
}


	return 0;
}
