#include<bits/stdc++.h>

using namespace std;

string s;
int a[200010];
int main()
{
	int i,n,ans=INT_MAX;	
	cin>>n;
	cin>>s;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=1;i<n;i++)
	{
		if(s[i-1]=='R' && s[i]=='L')
		{
			ans = min(ans,(a[i]-a[i-1])/2);
		}
	}
	if(ans==INT_MAX)
		cout<<-1<<endl;
	else
	cout<<ans<<endl;
return 0;
}
