#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

int ans=2e9,n,a[200010];
char s[200010];

int main(){
	scanf("%d",&n);
	scanf("%s",s+1);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
		if(i!=1&&s[i]=='L'&&s[i-1]=='R')
		ans=min(ans,(a[i]-a[i-1])/2);
	}
	if(ans==2e9) puts("-1");
	else printf("%d\n",ans);
	return 0;
}
