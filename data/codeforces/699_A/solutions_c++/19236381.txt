#include <bits/stdc++.h>

using namespace std;

int n, a[200010];
char s[200010];

int main()
{
	scanf("%d %s %d", &n, s, a);
	int res = 2e9;

	for (int i = 1; i < n; ++i)
	{
		scanf("%d", a+i);
		if (s[i] == 'L' && s[i-1] == 'R')
			res = min(res, (a[i] - a[i-1]) / 2);
	}

	if (res == 2e9)
		printf("-1");
	else
		printf("%d", res);
	return 0;
}
