#include <cstdio>
char s[200000];
int n, x, y, min = -2;
int main (){
	scanf("%d\n", &n);
	for (int i = 0; i < n; i++)
		scanf("%c", &s[i]);
	scanf("%d", &x);
	for (int i = 1; i < n; i++){
		scanf("%d", &y);
		if (s[i - 1] == 'R' && s[i] == 'L')
			if (min == -2 || y - x < min)
				min = y - x;
		x = y;
	}
	printf("%d\n", min / 2);
	return 0;
}
