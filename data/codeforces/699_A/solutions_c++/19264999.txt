#include <bits/stdc++.h>
using namespace std;
const int N=2*100000+5,oo=1e9;
int n,a[N],res=oo;
string s;
main() {
    ios::sync_with_stdio(0);
    cin>>n>>s; s="@"+s;
	for(int i=1;i<=n;i++) {
	    cin>>a[i];
	    if(s[i]=='L' && s[i-1]=='R') res=min(res,(a[i]-a[i-1])>>1);
	}
	cout<<(res==oo?-1:res);
}
