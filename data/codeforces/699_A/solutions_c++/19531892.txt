#include<iostream>;

using namespace std;

int a[200001];

int main(){
	
	int n;
	string s;
	int ans=1000000000;
	cin>>n;
	cin>>s;
	
	cin>>a[0];
	for (int i = 1;i<n;i++) {
	cin>>a[i];
	if (s[i]=='L'&&s[i-1]=='R')

	ans = min(ans,(a[i]-a[i-1])/2);
	}
	if (ans==1000000000) ans = -1;
	cout<<ans;
	
}
