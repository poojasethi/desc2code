#include<bits/stdc++.h>
using namespace std;
int p[200010];
char str[200010];
main()
{
	int i,n,t;
	scanf("%d",&n);
	scanf("%s",str);
	for(i=0;i<n;i++)
	scanf("%d",&p[i]);
	int ans=2e9;
	for(i=0;i<n-1;i++)
	if(str[i]=='R' && str[i+1]=='L')
	{
		t=(p[i+1]-p[i])/2;
		ans=min(t,ans);
	}
	if(ans==2e9)
	printf("-1");
	else
	printf("%d\n",ans);
}
