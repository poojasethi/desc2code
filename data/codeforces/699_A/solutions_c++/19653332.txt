#include <iostream>
using namespace std;
int main() {
	int n,ans=INT_MAX;
	string x;
	cin >> n >> x;
    int arr[n];
    cin >> arr[0];
    for(int i=1;i<n;i++) {         
	    cin >> arr[i];
        if(x[i-1]=='R'&&x[i]=='L') {
           ans = min(ans,(arr[i]-arr[i-1])/2);	
		}
	}
	if(ans==INT_MAX) cout << "-1";
	else cout << ans;
	return 0;
}
