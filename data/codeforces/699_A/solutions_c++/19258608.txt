#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	int ans=INT_MAX;
	string s;
	cin>>n>>s;
	int a[n];
	for(int i=0;i<n;i++) cin>>a[i];
	for(int i=1;i<n;i++)
	{
		if(s[i-1]=='R' && s[i]=='L')
			ans = min(ans,(a[i]-a[i-1])>>1);
	}
	if(ans==INT_MAX) cout<<"-1"<<endl;
	else cout<<ans<<endl;	
}
