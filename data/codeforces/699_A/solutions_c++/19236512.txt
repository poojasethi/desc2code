#include <bits/stdc++.h>
using namespace std;

const int N=200005;
int n;
char tp[N];
int x[N];

int main()
{
	scanf("%d",&n);
	int i,ans=-1;
	scanf("%s",tp+1);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&x[i]);
		if(tp[i]=='L'&&tp[i-1]=='R')
			if(ans==-1)ans=(x[i]-x[i-1])/2;
			else ans=min(ans,(x[i]-x[i-1])/2);
	}
	printf("%d",ans);
	return 0;
}
