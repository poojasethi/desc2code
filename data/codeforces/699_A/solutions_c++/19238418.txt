#include <bits/stdc++.h>
using namespace std;
int main()
{ long long n;
  cin >> n;
  long long l[n];
  string s;
  cin >> s;
  for (int i=0;i<n;i++)
   cin >> l[i];
  long long mint=1000000001;
  for (int i=1;i<n;i++)
    if (s[i]=='L' && s[i-1]=='R')
     mint=min(mint,(l[i]-l[i-1])/2);
  if (mint==1000000001)
    mint=-1;
   cout << mint;
}
