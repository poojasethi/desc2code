#include<stdio.h>
char a[200010];
int b[200010];
int main()
{
	int n,i,ans=-1;
	scanf("%d",&n);
	scanf("%s",a);
	for(i=0;i<n;i++)
		scanf("%d",&b[i]);
	for(i=0;i<n-1;i++)
	{
		if(a[i]=='R'&&a[i+1]=='L')
		{
			int j=(b[i+1]-b[i])/2;
			if(ans==-1) ans=j;
			if(j<ans) ans=j;
		}	
	}
	printf("%d\n",ans);
	return 0;
}
