#include<bits/stdc++.h>
using namespace std;
int n;
string s;
int ans=~0U>>2;
int rea[2000010];
int main()
{
    cin>>n>>s;
    for(int i=1;i<=n;i++)
        cin>>rea[i];
    for(int i=1;i<n;i++)
        if(s[i-1]=='R'&&s[i]=='L')
            ans=min(ans,(rea[i+1]-rea[i])/2);
    if(ans==~0U>>2) ans=-1;
    cout<<ans<<endl;
    return 0;
}
