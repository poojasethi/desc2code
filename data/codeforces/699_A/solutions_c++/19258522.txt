#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(0);
	int n;
	cin>>n;
	string s;
	cin>>s;
	int x[n+1],ans=INT_MAX;
	for(int i=0;i<n;i++) cin>>x[i];
	for(int i=0;i<n-1;i++)
		if(s[i]=='R'&&s[i+1]=='L')
			ans=min(ans,(x[i+1]-x[i])/2);
	if(ans!=INT_MAX) cout<<ans;
	else cout<<-1;
	return 0;
}
