#include<iostream>
#include<algorithm>
using namespace std;
string st;
int a[200009],res,n;

int main(){
	cin>>n;
	cin>>st;
	res=1000000001;
	for(int i=0;i<=n-1;i++)
	  cin>>a[i];
	for(int i=1;i<=n-1;i++)
	  if (st[i-1]=='R' && st[i]=='L')
	     res=min(res,(a[i]-a[i-1])/2);
	if (res!=1000000001)
	cout<<res;
	else cout<<"-1";
}
