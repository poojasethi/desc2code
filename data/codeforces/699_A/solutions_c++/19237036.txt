#include<iostream>
using namespace std;
int main()
{
	long long int n,ans=1000000002;
	cin>>n;
	long long int i,a[n];
	string s;
	cin>>s;
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	for(i=0;i<s.length()-1;i++){
		if(s[i]=='R'&&s[i+1]=='L'){
			ans=min(ans,(a[i+1]-a[i])/2);
		}
}
if(ans==1000000002)
cout<<"-1"<<endl;
else
cout<<ans<<endl;
}
