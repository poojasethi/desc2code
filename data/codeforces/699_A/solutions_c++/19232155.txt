#include <iostream>
#include <vector>

using namespace std;

int main(){
  int n, m=2000000000;
  cin >> n;
  string s;
  vector <int> v(n);
  cin >> s;
  for(int i=0;i<n;++i){
    cin >> v[i];
  }
  for(int i=1;i<n;++i){
    if(s[i]=='L' and s[i-1]=='R')m=min(m,(v[i]-v[i-1])/2);
  }
  if(m==2000000000)cout << "-1" << endl;
  else cout << m << endl;
}
