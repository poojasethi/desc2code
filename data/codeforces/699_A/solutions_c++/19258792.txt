#include <bits/stdc++.h>

using namespace std;

string s;
int n, prev = -1, ans = 1000000001;
int main() {
    cin >> n >> s;
    for(int i = 0; i < n; ++i) {
        int a; cin >> a;
        if(i && s[i - 1] == 'R' && s[i] == 'L') ans = min(ans, (a - prev) / 2);
        prev = a;
    }
    if(ans == 1000000001) ans = -1;
    cout << ans;
    return 0;
}
