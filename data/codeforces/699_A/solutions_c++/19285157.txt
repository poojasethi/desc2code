#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string str;
	cin>>str;
	int a[n];
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
	}
	int ans=INT_MAX;
	for(int i=1;i<n;i++)
	{
		if(str[i]=='L' && str[i-1]=='R')
		{
			ans=min(ans,abs((a[i]-a[i-1])/2));
		}
	}
	if(ans==INT_MAX) cout<<-1<<endl;
	else cout<<ans<<endl;
}
