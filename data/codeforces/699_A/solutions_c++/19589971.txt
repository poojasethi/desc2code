#include <iostream>
#include <cstdio>
using namespace std;
int s[200005];
char a[200005];
int main()
{
    int ans=0x3f3f3f3f,n;
    scanf("%d%s",&n,a);
    for(int i=0;i<n;i++)scanf("%d",&s[i]);
    for(int i=1;i<n;i++)
        if(a[i-1]=='R'&&a[i]=='L')ans=min(ans,(s[i]-s[i-1])/2);
    printf("%d",ans==0x3f3f3f3f?-1:ans);
    return 0;
}
