#include <cstdio>
const int maxn=1<<30;
char s[300100];
int main(){
	int n,a,b;
	while(scanf("%d",&n)!=EOF){
		scanf("%s",s);
		scanf("%d",&a);
		int Min=maxn;
		for(int i=1;i<n;i++){
			scanf("%d",&b);
			if(s[i-1]=='R'&&s[i]=='L'){
				if(b-a<Min)Min=b-a;
			}
			a=b;
		}
		if(Min==maxn)printf("-1\n");
		else printf("%d\n",(Min+1)/2);
	}
}
