#include<bits/stdc++.h>

using namespace std;

int main(){
	
	int i,n,a[200009],ans=1000000009,flag=0;
	string s;
	
	cin>>n>>s;
	
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	
	for(i=0;i<n-1;i++){
		if(s[i]=='R' && s[i+1]=='L'){
			ans = min(ans, abs(a[i]-a[i+1])/2);
			flag = 1;
		}
	}
	
	if(flag){
		cout<<ans;
	}
	else{
		cout<<-1;
	}
	
	return 0;
}
