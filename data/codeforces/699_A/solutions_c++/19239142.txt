#include<iostream>
#include<string>
#include<algorithm>
using namespace std;

int main(){
	int n;
	cin>>n;
	string str;
	cin>>str;
	int l=-1,r=-1;
	int ans=1000000001;
	for(int i=0;i<n;i++){
		if(str[i]=='R')cin>>r;
		else cin>>l;
		if(r!=-1 && l!=-1 && r<l)ans=min(ans,(l-r)/2);
	}
	if(ans!=1000000001)cout<<ans<<endl;
	else cout<<-1<<endl;

}
