#include <stdio.h>
#define N 200002
int main() {
	char s[N], ls = '\0';
	int n, i, mn = ~(1 << 31), d, c;
	scanf("%d\n%s", &n, s);
	for(i = 0; i < n; i++) {
		scanf("%d", &c);
		if(i && (s[i - 1] == 'R' && s[i] == 'L')) {
			mn = mn > (c - d)  / 2 ? (c - d) / 2 : mn;
		}
		d = c;
	}
	printf("%d\n", (mn == (~(1 << 31)) ? -1 : mn));
}
