#include <bits/stdc++.h>
using namespace std;

int n,a[200003],m=2000000000;
string s;

int main(){
	cin>>n>>s;
	for (int i=0;i<n;i++){
		cin>>a[i];
	}
	for (int i=0;i<n-1;i++){
		if (s[i]=='R' && s[i+1]=='L'){
			m=min(m,a[i+1]-a[i]);
		}
	}
	if (m==2000000000) cout<<-1; else cout<<m/2;
	return 0;
}
