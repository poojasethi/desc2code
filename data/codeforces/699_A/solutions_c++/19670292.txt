#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int n,x,r=-1,ans=-1;
	char str[200005];
	scanf("%d",&n);
		scanf("%s",str);
		for(int i=0;i<n;i++){
			scanf("%d",&x);
			if(str[i]=='R')
			r=x;
			else if(r>-1)
			ans=(ans==-1)?(x-r)/2:min(ans,(x-r)/2);
		}
		printf("%d",ans);
	
return 0;	
}
