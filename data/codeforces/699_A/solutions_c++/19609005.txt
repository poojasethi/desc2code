#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int n;
	cin>>n;
	string s;
	cin>>s;
	int a[n];
	for(int i = 0;i<n;i++){
		cin>>a[i];
	}
	int ans = 1e9 + 1;
	for(int i = 0;i<n-1;i++){
		if(s[i] == 'R' && s[i+1] == 'L')
			ans = min(ans,a[i+1]-a[i]);
	}
	if(ans == 1e9 + 1)
	ans = -2;
	cout<<ans/2<<endl;
	return 0;
}
