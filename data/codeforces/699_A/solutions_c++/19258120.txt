#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n;
	int time=INT_MAX,ans=INT_MAX;
	cin>>n;
	int a[n];
	char str[n];
	scanf("%s",str);
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n-1;i++)
	{
		if(str[i]=='R' && str[i+1]=='L')
			time=(a[i+1]-a[i])/2;
		ans=min(ans,time);
	}
	if(ans==INT_MAX)
		printf("-1\n");
	else
		printf("%d\n",ans);
}
