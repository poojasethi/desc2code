#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
int n;
cin>>n;
string s;
cin>>s;
int a[n];
for(int i=0;i<n;i++)
cin>>a[i];
int ans=INT_MAX;
for(int i=0;i<n-1;i++)
if(s[i]=='R' && s[i+1]=='L')
{
	if(a[i+1]-a[i]<ans) ans=a[i+1]-a[i];
}
if(ans>=INT_MAX) cout<<-1;
else cout<<ans/2;
	return 0;
}
