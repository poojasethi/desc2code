#include<cstdio>
char s[210000];
int n,l,m,x[210000];
int main(){
	l=-1,m=1100000000;
	scanf("%d%s",&n,s);
	for(int i=0;i<n;i++)
		scanf("%d",&x[i]);
	for(int i=0;i<n;i++){
		if(s[i]=='R') l=x[i];
		else if(l!=-1&&(x[i]-l)/2<m) m=(x[i]-l)/2;
	}
	if(m!=1100000000) printf("%d",m);
	else printf("-1");
	return 0;
}
