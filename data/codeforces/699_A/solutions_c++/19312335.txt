#include<bits/stdc++.h>

using namespace std;

int i, n, rpoint, lpoint, m=2e9; char p[200005];

int main(){
    scanf("%d%s%d", &n, p, &lpoint);

    for (i=1; i<n;i++){
        scanf("%d", &rpoint);
        if (p[i-1]=='R'&&p[i]=='L') m=min(m,rpoint-lpoint);
        lpoint=rpoint;
    }
    printf("%d", m<2e9 ? m/2 : -1);
    return 0;
}
