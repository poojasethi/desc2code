import sys
elements = input()
el_d = raw_input()
elem_reach = []
el_x = map(int, raw_input().split())
for i in range(0,elements-1):
    if el_d[i] == 'R' and el_d[i+1] == 'L':
        elem_reach.append(el_x[i+1]-el_x[i])
if len(elem_reach)>0:
    elem_reach.sort()
    print elem_reach[0]/2
else:
    print -1
