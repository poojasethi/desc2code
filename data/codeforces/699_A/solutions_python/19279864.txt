n = int(raw_input())
s = raw_input()
p = [int(x) for x in raw_input().split()]

r = 0
l = 0
ans = -1
while r <n:
    while(r < n and s[r] != 'R' ) :
        r+=1
    if (l <= r):
        l = r+1
        while(l < n and s[l] != 'L'):
            l+=1
    if l==n or r == n: break
    if (ans == -1) or (p[l]-p[r])/2 < ans:
        ans = (p[l]-p[r])/2
    r += 1

print ans
