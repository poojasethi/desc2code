def readInt(): return int(raw_input())
def readList(): return map(int, raw_input().split(' '))

n = readInt()
a = list(raw_input())
b = readList()
m = -1
for i in xrange(n-1):
    if a[i] == 'R' and a[i+1] == 'L':
        j = (b[i+1] - b[i] + 1) / 2
        if j < m or m == -1:
            m = j

print m
