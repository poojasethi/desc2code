a=raw_input()
a=list(a)

b=map(int,raw_input().split())
min1=[]
if len(set(a))==1:
    print -1
else:
    F=0
    for i in range(0,t-1):
        if (a[i]=='R' and a[i+1]=='L'):
            F=1
            min1.append(b[i+1]-b[i])
    if F==1:
        print min(min1)/2
    else:
        print -1
