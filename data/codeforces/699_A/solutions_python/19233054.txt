n = int(raw_input())
s = raw_input()

cc = [int(x) for x in raw_input().split()]

curc = None
curi = -1

mn = 1e10

i = 0

for c in s:
    if c == 'R':
        curi = i
    if curi != -1 and c == 'L':
        mn = min(mn, (cc[i] - cc[curi]) / 2)
    i += 1

if mn == 1e10:
    print -1
else:
    print mn
