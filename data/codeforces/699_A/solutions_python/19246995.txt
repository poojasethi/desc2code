n = int(raw_input())
s = raw_input()
x = map(int, raw_input().split())

ans = None
pp, pd = None, None

for p, d in sorted(zip(x, s)):
    if pd == 'R' and d == 'L':
        if ans is None or 2 * ans > p - pp:
            ans = (p - pp) // 2
    pp, pd = p, d

if ans is None:
    print '-1'
else:
    print ans
