n = int(raw_input().strip())
direction = raw_input().strip()
x = map(int, raw_input().strip().split(' '))

a = sorted([(x[i], direction[i]) for i in range(n)])

min_d = -1
for i in range(n-1):
	if a[i][1] == 'R' and a[i+1][1] == 'L':
		new_d = abs(a[i][0] - a[i+1][0])/2
		if min_d < 0 or min_d > new_d:
			min_d = new_d


print min_d
