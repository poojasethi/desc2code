n = int(raw_input())
str = raw_input()
dis = map(int, raw_input().split(" "))

max = 10000000000
dist = max
posL = -1
posR = -1
for i in range(n):
	if str[i] == 'L':
		posL = dis[i]
		if posR != -1 and posL-posR<dist:
			dist = posL-posR
	
	elif str[i] == 'R':
		posR = dis[i]
		
			
if dist == max:
	dist = -1
else:
	dist /= 2
print dist
