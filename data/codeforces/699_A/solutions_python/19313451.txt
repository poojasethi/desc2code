n=int(raw_input())
s=list(raw_input())
l=list(map(long,raw_input().split()))
min_=789465561456545
c=0
for i in xrange(n-1):
    if s[i]=='R' and s[i+1]=='L':
        c=1
        k=l[i+1]-l[i]
        if k<min_:
            min_=k
        if min_==1:
            break
if c==0:
    print -1
else:
    print min_/2
