n = input()
m = raw_input()
a = map(int,raw_input().split())
ans  = 1000000001
if n == 1:
    print -1
else:
    for i in xrange(len(m)-1):
        if m[i] == 'R' and m[i+1] == 'L':
            ans = min(ans,a[i+1] - a[i])
    if ans  == 1000000001:
        print -1
    else:
        print ans/2
