n = input()
d = [i for i in raw_input()]
cord = [int(i) for i in raw_input().split()]
cd = [(cord[i],d[i]) for i in xrange(n)]
cd.sort()
ans = 10**19
for i in xrange(n-1):
    if((cd[i][1]!=cd[i+1][1]) and (cd[i][1]=='R') and (cd[i][0]<cd[i+1][0])):
        temp = abs(cd[i][0]-cd[i+1][0])/2
        ans = min(ans,temp)
if(ans==10**19):
    ans = -1
print ans
