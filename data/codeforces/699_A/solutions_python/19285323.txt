x = input()
d = raw_input()
cor = map(int, raw_input().split())

ans = -1
r = 0 if d[0] == 'R' else -1

for i in xrange(1, x):
    curr_t = -1
    if d[i] == 'L':
        if r != -1:
            curr_t = (cor[i] - cor[r]) / 2
    else:
        r = i
    if curr_t != -1:
        ans = curr_t if ans == -1 else ans
        ans = curr_t if curr_t < ans else ans

print ans
