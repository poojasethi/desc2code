n = int(raw_input())
direc = raw_input()
dist = [int(_) for _ in raw_input().split()]

flagR = False
R = -1
pico = []

for i in range(n):
    if(direc[i] == 'R'):
        flagR = True
        R = i
    elif(direc[i]=='L' and flagR):
        pico.append((dist[i] - dist[R])/2)

if(len(pico) == 0):
    print(-1)
else:
    print(min(pico))
