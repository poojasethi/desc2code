from sys import stdin

line = stdin.readline().split()
n, p, t = int(line[0]), float(line[1]), int(line[2])
prob = [0.0]*(n+1)
prob[0] = 1-p
prob[1] = p
for i in xrange(2, t+1):
    for m in xrange(min(i, n), 0, -1):
        if m == n and i > n:
            prob[m] = prob[m]+prob[m-1]*p
        else:
            prob[m] = prob[m-1]*p+prob[m]*(1-p)
    prob[0] = prob[0]*(1-p)
    
ans = sum(i*prob[i] for i in xrange(min(n, t)+1))
print ans
