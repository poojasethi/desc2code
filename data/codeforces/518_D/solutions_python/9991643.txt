N,P,T = raw_input().split()
N = int(N)
T = int(T)
P = float(P)
Q = 1-P
A = [0 for i in xrange(T+1)]
A2 = [0 for i in xrange(T+1)]
if N > T: N = T
#At the ith step, we have A[k] = answer for <n=k, t=i>

for j in xrange(1,T+1):
    for i in xrange(1,T+1):
        A2[i] = P + Q*A[i] + P*A[i-1]
    A = A2[:]
print A[N]
