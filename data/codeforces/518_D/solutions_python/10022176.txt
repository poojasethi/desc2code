# -*- coding:utf-8 -*-

x,y,z = [ inp for inp in raw_input().split(' ') ]
n,t = int(x),int(z)
p = float(y)
f = [ [0.0 for j in range(n+1)] for i in  range(t+1)]
f[1][0] = 1-p
f[1][1] = p
for i in range(2,t+1):
    f[i][0] = f[i-1][0]*(1-p)
for i in range(2,t+1):
    for j in range(1,min(i,n)+1):
        if j==n:
            f[i][j] = f[i-1][j-1]*p+f[i-1][j]
        else:
            f[i][j] = f[i-1][j-1]*p+f[i-1][j]*(1-p)
ans = 0.0
for j in range(1,n+1):
    ans += j*f[t][j]
print '%.8f' % ans
