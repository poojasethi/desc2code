# -*- coding:utf-8 -*-

x,y,z = [ inp for inp in raw_input().split(' ') ]
n,t = int(x),int(z)
p = float(y)
f = [ [0.0 for j in range(n+1)] for i in  range(t+1)]
f[0][0] = 1.0
for i in range(t):
    for j in range(min(i,n)+1):
        if j<n:
            f[i+1][j+1] += f[i][j]*p
        else:
            f[i+1][j] += f[i][j]*p
        f[i+1][j] += f[i][j]*(1-p)
ans = 0.0
for j in range(1,n+1):
    ans += j*f[t][j]
print '%.8f' % ans
