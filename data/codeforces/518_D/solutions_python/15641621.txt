R = raw_input().split()
n = int(R[0])
p = float(R[1])
t = int(R[2])
f, g = [0.0] * 3000, [0.0] * 3000
f[0] = 1.0
for i in range (1, t+1, 1):
    for j in range (i):
        g[j] += f[j] * (1-p) if j < n else f[j]
        g[j+1] += f[j] * p if j < n else 0.0
    for j in range (i+1):
        f[j] = g[j]
        g[j] = 0.0
        
ans = 0.0
for i in range (1, t+1, 1):
    ans += i * f[i]    
print ans
