n, p, t =  map(float, raw_input().split())
n = int(n)
t = int(t)

pr = [[0]*(t+1) for _ in range(n+1)]
pr[0][0] = 1
for i in range(1, t+1):
    pr[0][i] = pr[0][i-1]*(1-p)

for num in range(1, n+1):
    for i in range(t+1):
        if i<num: continue
        no_p = 1-p
        if num == n: no_p = 1
        pr[num][i]= pr[num][i-1]*no_p + pr[num-1][i-1]*p

ans = 0
for num in range(1, n+1):
    ans = ans + num*pr[num][t]

print ans
