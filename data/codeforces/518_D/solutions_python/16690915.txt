s = raw_input().split()
n = int(s[0])
p = float(s[1])
t = int(s[2])
m = min(n, t)
dp = [0.0] * (m + 1)
dp[0] = 1
for i in xrange(1, t + 1):
	for j in xrange(min(i, n), 0, -1):
		if j < n: dp[j] = dp[j - 1] * p + dp[j] * (1 - p)
		else: dp[j] += dp[j - 1] * p
	dp[0] *= (1 - p)
ans = 0
for i in xrange(m + 1): ans += dp[i] * i
print ans
