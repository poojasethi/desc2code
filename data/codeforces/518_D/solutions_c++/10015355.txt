#include <bits/stdc++.h>

#define Max 2009
using namespace std;

double dp[Max][Max];
int main(){
	int n,t;
	double p;
	cin >> n >> p >> t;
	for(  int i=1;i<=n;i++)
		for( int j=1;j<=t;j++){
			dp[i][j] = (p * (1+ dp[i-1][j-1] )) + ((1-p)*dp[i][j-1]);	
		}
		
	cout << fixed << setprecision(12) << dp[n][t] << endl;
	return 0;
}
