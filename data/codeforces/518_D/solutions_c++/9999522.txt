#include <algorithm>
#include <iostream>
#include <cstdio>
using namespace std;
double Q[2001][2001];
double p;
int n,t;
int main(){
  cin >> n >> p >> t;
  Q[0][0] = 1;
  for(int i=1;i<=n;++i){
    for(int j=1;j<=t;++j) Q[i][j] = p*Q[i-1][j-1]+(1-p)*Q[i][j-1];
  }
  double ans = 0;
  for(int i=1;i<=n;++i){
    for(int j=1;j<=t;++j) ans+=Q[i][j];
  }
  printf("%.10lf\n",ans);
  return 0;
}
