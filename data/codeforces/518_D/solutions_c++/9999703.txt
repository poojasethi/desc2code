#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
double dp[2005][2005];
int main()
{
  int n,t,i,j;
  double p;
  scanf("%d%lf%d",&n,&p,&t);
  memset(dp,0,sizeof(dp));
  for (j=1;j<=t;j++)
	for (i=1;i<=n;i++)
	  dp[i][j]=p*(1+dp[i-1][j-1])+(1-p)*dp[i][j-1];
  printf("%lf\n",dp[n][t]);
}
