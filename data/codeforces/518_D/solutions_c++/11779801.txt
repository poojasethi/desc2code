#include <bits/stdc++.h>
using namespace std;
double p,dp[2010][2010];
int main()
{
    int n,t;
    dp[0][0]=0;
    dp[1][0]=0;
    scanf("%d%lf%d",&n,&p,&t);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=t;j++)
        {
            dp[i][j]=(1+dp[i-1][j-1])*p+dp[i][j-1]*(1-p);
        }
    printf("%0.12lf\n",dp[n][t]);
}
