#include<stdio.h>
double ans,f[2005][2005],p;
int n,t;
int main(){
	scanf("%d%lf%d",&n,&p,&t);
	f[0][0]=1;
	for(int i=1;i<=t;i++){
		for(int j=0;j<=i;j++){
			if(j==n){
				f[i][j]+=f[i-1][j];
			}
			else {
				f[i][j]+=f[i-1][j]*(1-p);
				f[i][j+1]+=f[i-1][j]*p;	
			}			
		}
	}
	for(int i=0;i<=n;i++)ans+=i*f[t][i];
	printf("%.6f\n",ans);
}
