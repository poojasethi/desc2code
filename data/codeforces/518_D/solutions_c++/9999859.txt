#include<bits/stdc++.h>
using namespace std;
double a[2009][2009]={0};
#define LL long long int
int main()
{
    double p;
    LL n,t,i,j;
    cin>>n>>p>>t;
    for(i=1;i<=t;i++)
        for(j=1;j<=n;j++)
    {
        a[i][j]=(a[i-1][j-1]+1)*p+a[i-1][j]*(1-p);
    }
    printf("%.6lf",a[t][n]);
    return 0;
}
