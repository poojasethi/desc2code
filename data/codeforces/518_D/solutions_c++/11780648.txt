#include<bits/stdc++.h>

using namespace std;


double dp[2005][2005];

int main(){



	int  n;
	double p;
	int t;

	cin >> n >> p >> t;



	for(int i =1; i<=n; i++) // for every person
		for(int j=1; j<=t; j++) //at second
			dp[i][j] = p*(dp[i-1][j-1]+1) + (1-p)*(dp[i][j-1]);



	printf("%.6f",dp[n][t] );



	return 0;

}
