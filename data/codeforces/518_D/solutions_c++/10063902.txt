#include<bits/stdc++.h>
using namespace std;
double dp[2001][2001];
int main()
{
    int n,t,i,j;
    double p;
    cin>>n>>p>>t;

    memset(dp,0,sizeof(dp));

    for(j=1;j<=t;j++)
    {
        for(i=1;i<=n;i++)
        {
            dp[i][j]=p*(1+dp[i-1][j-1])+(1-p)*dp[i][j-1];

        }
    }
    cout<<setprecision(8)<<dp[n][t]<<endl;
    return 0;
}
