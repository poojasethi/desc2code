#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
double dp[2009][2009];
int main()
{
	int n,t;double p;
	scanf("%d%lf%d",&n,&p,&t);
	for(int i=1;i<=t;i++)
	{
		for(int j=1;j<=n;j++)
		{
			dp[i][j]=dp[i-1][j]*(1-p)+(dp[i-1][j-1]+1)*p;
		}
	}
	printf("%.6f\n",dp[t][n]);
	return 0;
}
