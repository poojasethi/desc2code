#include<cstdio>

int n, t;
double p, ans;
double f[2222][2222];

int main() {
	scanf("%d%lf%d", &n, &p, &t);
	f[0][0] = 1;
	for(int i=1;i<=t;i++)
		for(int j=0;j<=n;j++){
			f[i][j] = f[i-1][j] * ((j<n) ? (1-p) : 1);
			if(j)
				f[i][j] += f[i-1][j-1] * p;
		}
	ans = 0;
	for(int i=0;i<=n;i++)
		ans += i * f[t][i];
	printf("%.10lf\n", ans);
	
	return 0;
}
