#include <bits/stdc++.h>
using namespace std;

#define ll long long

double f[2010][2010];
int main()
{
	ll N,T; double P; cin>>N>>P>>T;
	for (ll t = 1; t <= T; t++) f[N][t] = N;
	for (ll x = 1; x <= N; x++) f[x][T] = x;
	for (ll t = T - 1; t >= 0; t--) for (ll x = N - 1; x >= 0; x--) f[x][t] = P*f[x + 1][t + 1] + (1-P)*f[x][t + 1];
	cout<<setprecision(8)<<f[0][0];
	return 0;
}
