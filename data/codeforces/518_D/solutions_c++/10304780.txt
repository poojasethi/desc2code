#include <stdio.h>
double dp[2005][2005];
int main()
{
	double p,ans=0.0;
	int n,t,i,j;
	scanf("%d %lf %d",&n,&p,&t);
	dp[0][0]=1.0;
	for(i=1;i<=n;i++)
	{
		for(j=i;j<=t;j++)
		{
			dp[i][j]=dp[i-1][j-1]*p+dp[i][j-1]*(1.0-p);
		}
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=t;j++)
		{
			ans=ans+dp[i][j];
		}
	}
	printf("%.7lf\n",ans);
	return 0;
}
