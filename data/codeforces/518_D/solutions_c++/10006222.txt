#include<bits/stdc++.h>
using namespace std;
const int maxn=2000+20;
long double dp[maxn][maxn];

int main(){
	int n, t;
	double p;
	cin >> n >> p >> t;
	for(int i=1;i<=n;i++)
        for(int j=1;j<=t;j++)
	dp[i][j]=p*(1+dp[i-1][j-1])+(1-p)*dp[i][j-1];
	cout <<  setprecision(10) << dp[n][t] <<"\n" ;
return 0;
}
