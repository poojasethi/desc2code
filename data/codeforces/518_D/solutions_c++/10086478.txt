#include<iostream>
#include<cstdio>
using namespace std;
double dp[2001][2001];
int main()
{
double sum=0,p;
int i,j,t,n;
cin>>n>>p>>t;
for(i=0;i<=t;i++)
	for(j=0;j<=n;j++)
		dp[i][j]=0;
dp[0][0]=1;
for(i=0;i<t;i++)
{
	for(j=0;j<n;j++)
	{
	dp[i+1][j+1]+=dp[i][j]*p;
	dp[i+1][j]+=dp[i][j]*(1-p);
	}
	dp[i+1][n]+=dp[i][n];
}
for(j=0;j<=n;j++)
	sum+=dp[t][j]*j;
printf("%.6lf\n",sum);
return 0;
}
