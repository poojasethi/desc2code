#include <bits/stdc++.h>

using namespace std;

double p, f[3000][3000];
int n, t;

int main()
{
    //freopen("in.inp", "r", stdin);

    cin >> n >> p >> t;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= t; j++)
            f[i][j] = p*(1 + f[i - 1][j - 1]) + (1 - p) * f[i][j - 1];

    printf("%0.6f", f[n][t]);
}
