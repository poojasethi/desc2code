#include<bits/stdc++.h>
using namespace std;
const int n1=2002;
main()
{
    double comb[n1][n1],ans=0.0;
    double p; int n,t,i,j;
    scanf("%d",&n);scanf("%lf",&p);scanf("%d",&t);
    for(i=1;i<=t;i++)
    {
        for(j=1;j<=n;j++)
        comb[i][j]=(comb[i-1][j-1]+1.0)*p+comb[i-1][j]*(1.0-p);
    }
    printf("%lf\n",comb[t][n]);
}
