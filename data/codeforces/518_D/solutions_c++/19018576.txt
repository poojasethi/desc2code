#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>

 using namespace std;
 double p,q,k,g,f[2005][2005];
 int n,t,i,j;
 int main(){
		scanf("%d%lf%d",&n,&p,&t);
		for (i=1;i<=n;i++)
	        for (j=1;j<=t;j++)
				f[i][j]=p*(f[i-1][j-1]+1)+(1-p)*f[i][j-1];
        printf("%0.9lf",f[n][t]);    
		return(0);
 }
