#include<cstdio>
#define N 2010
int n,t,i,j;
double p,ans,f[N][N];
int main()
{
	scanf("%d%lf%d",&n,&p,&t);
	f[0][0]=1;
	for (i=1;i<=t;i++)
	for (j=0;j<n;j++)
	{
		f[i][j]=f[i][j]+f[i-1][j]*(1-p);
		f[i][j+1]=f[i][j+1]+f[i-1][j]*p;
	}
	for (i=1;i<=n;i++)
	ans=ans+i*f[t][i];
	for (i=0;i<t;i++)
	ans=ans+n*f[i][n];
	printf("%f",ans);
}
