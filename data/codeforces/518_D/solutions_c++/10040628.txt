#include <bits/stdc++.h>

using namespace std;

double st[2002][2002];
int main()
{
    ios_base::sync_with_stdio(true);
    int n,t;
    double p;
    scanf("%d %lf %d",&n,&p,&t);
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=t;j++)
        st[i][j] = ((p)*(1+st[i-1][j-1]) + (1-p)*(st[i][j-1]));
    }
    printf("%.6lf\n",st[n][t]);
    return 0;
}
