#include<cstdio>
using namespace std;
int n,t,i,j;
double f[2100][2100],p,ans,ans1;
int main()
{
    scanf("%d%lf%d",&n,&p,&t);f[0][0]=1;
    for(i=1;i<=t;i++)
    {
        f[i][0]=f[i-1][0]*(1-p);
        f[i][n]=f[i-1][n]+f[i-1][n-1]*p;
        for(j=1;j<=i&&j<n;j++)
            f[i][j]=f[i-1][j-1]*p+f[i-1][j]*(1-p);
    }for(i=0;i<=n;i++)ans=ans+i*f[t][i];printf("%.8lf\n",ans);
}
