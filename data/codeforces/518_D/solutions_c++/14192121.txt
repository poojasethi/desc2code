#include <bits/stdc++.h>
using namespace std;
int n, t;
double p, d[2001][2001];
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);

    cin >> n >> p >> t;
    for(int i=1; i<=t; i++){
        for(int j=1; j<=n; j++){
            d[i][j]+=(d[i-1][j-1]+1)*p;
            d[i][j]+=d[i-1][j]*(1-p);
        }
    }
    cout << setprecision(9);
    cout << d[t][n];
}
