#include<cstdio>
using namespace std;
int n,t,i,j;
double p,q,f[2010][2010],ans;
int main()
{
	scanf("%d%lf%d",&n,&p,&t);
	q=1.0-p;
	f[0][0]=1.0;
	for (i=1;i<=t;i++)
	{
		f[i][0]=q*f[i-1][0];
		f[i][n]=f[i-1][n]+p*f[i-1][n-1]; 
		for (j=1;j<n;j++)
		  f[i][j]=f[i-1][j-1]*p+f[i-1][j]*q;
	}
	ans=0.0;
	for (i=1;i<=n;i++) ans+=i*f[t][i];
	printf("%.7lf\n",ans);
	return 0;
}
