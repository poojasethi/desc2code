#include <bits/stdc++.h>
using namespace std;
double dp[2005][2005]; 
int main()
{
	int n, t; double p;
	cin >> n >> p >> t;
	for (int g=1; g<=n; g++)
	{
		for (int y=1; y<=t; y++)
		{
			dp[g][y]=(p*(1+dp[g-1][y-1]))+(1-p)*(dp[g][y-1]); 
		}
	}
	cout << fixed << setprecision(20) << dp[n][t]; return 0; 
}
