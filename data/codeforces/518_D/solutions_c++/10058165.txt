#include <iostream>
#include <cstdio>

using namespace std;

double dp[2020][2020];

int main(){
  int n, t;
  double p;
  cin >> n >> p >> t;
  for(int i = 1; i <= n; i++){
    for(int j = 1; j <= t; j++){
      dp[i][j] = p*(1 + dp[i-1][j-1]) + (1 - p)*dp[i][j-1];
    }
  }
  printf("%.10lf\n", dp[n][t]);
}
