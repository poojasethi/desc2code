#include <algorithm>
#include <cstdio>
double f[2010][2010],p,ans=0;
int i,j,n,t;
int main()
{
	scanf("%d%lf%d\n",&n,&p,&t);
	f[0][0]=1;
	for (i=1;i<=t;++i)
		for (j=0;j<=i && j<=n;++j)
		{
			f[i][j]=f[i-1][j]*(1.0-p);
			if (j==n) f[i][j]=f[i-1][j];
			if (j) f[i][j]+=f[i-1][j-1]*p;
		}
	for (i=1;i<=n;++i) ans+=f[t][i]*i;
	printf("%.10f\n",ans);
}
