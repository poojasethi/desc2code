#include <cstdio>
#include <algorithm>
using namespace std;
int a,b,c,d,e,g,i,j,k,l,m,n;
double h,f[2005][2005],p;
int main(){
	scanf("%d%lf%d",&n,&p,&m);
	f[0][0]=1;
	for(i=0;i<=m;i++)
	for(j=0;j<=n;j++)
	if(f[i][j]>0){
		f[i+1][min(j+1,n)]+=f[i][j]*p;
		f[i+1][j]+=f[i][j]*(1-p);
	}
	for(i=0;i<=n;i++)h=h+i*f[m][i];
	printf("%.8lf",h);
	return 0;
}
