#include <bits/stdc++.h>
using namespace std;

double p;
double ans[2001][2001];

int main() {
    int i,j,n,t;
    scanf("%d%lf%d",&n,&p,&t);
    for(i=1;i<=2000;i++) {
        ans[i][0]=i;
        for(j=1;j<=2000;j++) ans[i][j] = ans[i][j-1]*(1-p)+ans[i-1][j-1]*p;
    }
    printf("%0.6lf",n - ans[n][t]);
    return 0;
}
