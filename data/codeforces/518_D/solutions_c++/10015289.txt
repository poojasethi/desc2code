#include <bits/stdc++.h>
using namespace std;

double  dp[3000][3000],p;

int mark[3000][3000];

double  bul( int n,int t ){
    if(t==0 || n==0)  return  0;
    if( mark[n][t] ) return  dp[n][t];
    mark[n][t] = 1;
    return  dp[n][t] = (1.0-p)*bul( n,t-1 )+(p)*(bul( n-1,t-1 )+1.0);
}
int main(){
    int n,t;
    scanf("%d %lf %d",&n,&p,&t);
    printf("%.12lf\n",bul(n,t));
}
