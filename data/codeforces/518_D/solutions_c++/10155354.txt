#include <bits/stdc++.h>
#define N 2010

using namespace std;

double dp[N][N];

int main(){

    int n, t; double p;

    scanf ("%d %lf %d", &n ,&p, &t);

    for (int i = 1; i <= t; i++)
        for (int j = 1; j <= n; j++)
                dp[i][j] += (1 + dp[i - 1][j - 1]) * p + dp[i - 1][j] * (1 - p);

    printf ("%.6lf", dp[t][n]);

    return 0;
}
