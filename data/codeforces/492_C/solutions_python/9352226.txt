n,r,avg=map(int,raw_input().split())
inp = [map(int, raw_input().split()) for _ in ' '*n]
required=avg*n-sum(i[0] for i in inp)
inp.sort(key = lambda x: x[1])#1retsu de sort
ans=0
i=0
while required>=0:
    z=r-inp[i][0]
    if z>= required:
        ans+=inp[i][1]*required
        break
    else:
        ans+=inp[i][1]*z
        required-=z
    i+=1
print ans
