import operator, sys

n, r, avg = map(int, raw_input().split())

data = []
total = 0
for i in xrange(n):
    data.append(map(int, raw_input().split()))
    total += data[-1][0]

rtotal = avg * n
if total >= rtotal:
    print 0
    sys.exit(0)
data = sorted(data, key=lambda x: x[1])

req = rtotal - total

essays = 0
for marks, essay in data:
    canGain = r - marks
    if canGain <= req:
        essays += canGain * essay
        req -= canGain
    else:
        essays += req * essay
        break

print essays
