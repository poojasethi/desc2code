n,r,v = tuple(int(i) for i in raw_input().strip().split(" "))
d = v*n
s = []
for i in range(n):
    a,b = tuple(int(i) for i in raw_input().strip().split(" "))
    s.append([b,a])
    d -= a
if d <= 0:
    print 0
else:
    s.sort()
    c = 0
    i = 0
    while d > 0:
        m = min(r-s[i][1],d)
        c += m*s[i][0]
        d -= m
        i += 1
    print c
