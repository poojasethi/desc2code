n, r, av = map(int, raw_input().split())
target = n * av
grades = []
for i in xrange(n):
  g, b = map(int, raw_input().split())
  grades.append((b, g))
  target -= g
grades.sort()

res = 0
for b, g in grades:
  if target <= 0:
    break
  am = min(target, r - g)
  res += am * b
  target -= am
print res
