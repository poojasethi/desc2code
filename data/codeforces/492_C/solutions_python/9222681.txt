n,r,avg = map(int,raw_input().split())
s = []
su = 0
for i in xrange(n):
	a,b = map(int,raw_input().split())
	s.append([b,a])
	su += a
savg = avg * n
if(su >= savg):
	print 0
else:
	num = 0
	s.sort()
	for i in xrange(n):
		if(s[i][1] < r):
			dis = r-s[i][1]
			if (su+dis) < savg:
				num += dis*s[i][0]
				su += dis
			else:
				num += s[i][0] * (savg - su)
				print num
				break
		else:
			continue
