N,R,AVG=map(int,raw_input().split())
A=[]
B=[]
cnt=0


for i in range(N):
    x,y=map(int,raw_input().split())
    A.append(x)
    B.append(y)
    cnt+=x


zipl=[list(a) for a in zip(A,B)]
keep=list(sorted(zipl,key=lambda x:x[1],reverse=False))


res=0
for i in range(N):
    if(cnt>=AVG*N):
        break

    val=min(R-keep[i][0],AVG*N-cnt)
    cnt+=val
    res+=val*keep[i][1]


print res
