n ,r,arg = [int(i) for i in raw_input().split(' ')]
l=[]
for i in range(n):
    t = raw_input().split(' ')
    l.append( (int(t[1]),int(t[0])) )

l.sort()

cha = arg*n - sum([i[1] for i in l])


ess=0
for i in l:
    if cha<=0:break
    if r-i[1]<=cha:
        cha-=r-i[1]
        ess+=(r-i[1])*i[0]
    else:
        ess+=cha*i[0]
        break
print ess
