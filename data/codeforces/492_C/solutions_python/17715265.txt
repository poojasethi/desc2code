R = lambda : map(int,raw_input().split())
n,r,avg = R()
s = []
i = 0
count = avg*n
count1 = 0
res = 0
while i < n:
    i+=1
    a,b = R()
    count1 += a
    s.append([a,b])
s.sort(key=lambda x: x[1])
if count > count1:
    t = count - count1
    for x in s:
        if x[0] < r:
            q = r - x[0]
            mn = min(t,q)
            t-=mn
            res+=mn*x[1]
            if t == 0:
                break
print res
