from operator import itemgetter

n,r,avg=map(int,raw_input().split())
exams,totaal,verschil=[],0,0
for i in range(n):
    exams.append(map(int,raw_input().split()))
    totaal+=exams[i][0]
essays=0    
verschil=n*avg-totaal
exams.sort(key=itemgetter(1))

#print exams
i=0
while i<n and verschil>0:
    if exams[i][0]<r:
        e=min(r-exams[i][0],verschil)
        essays+=e*exams[i][1]
        verschil-=e
    i+=1
    
print essays
