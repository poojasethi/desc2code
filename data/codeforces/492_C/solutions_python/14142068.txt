n, r, avg = map(int, raw_input().split())
s = 0
m = []
for _ in range(n):
    a, b = map(int, raw_input().split())
    s += a
    m.append((a, b))
d = n * avg - s    
ms = sorted(m, key=lambda x:(x[1], x[0]))
ans = i = 0
while d > 0:
    a, b = ms[i]
    x = min(r - a, d)
    ans += x * b
    d -= x
    if d < 1:break
    i += 1
print ans
