[n, r, a] = [int(i) for i in raw_input().split(" ")]
A = []
#nec = a*n
for i in range(n):
    [ai, bi] = [int(i) for i in raw_input().split(" ")]  
    A += [(ai,bi)]
A.sort(key = lambda x: x[1])
s = sum([j for (j,k) in A])
#req = a*n
nec = a*n-s
count = 0
i = 0
while nec>0 and len(A) > i:
    if (A[i][0]<r): 
        v = min((r-A[i][0]),nec)
        nec -= v
        count += v*A[i][1]
    i+=1

print count
