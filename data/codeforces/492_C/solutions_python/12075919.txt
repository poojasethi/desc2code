(n, r, avg) = (int(i) for i in raw_input().split())
total = n * avg

scores = []

vanya = 0
for i in range(n):
	score = [int(j) for j in raw_input().split()]
	scores.append(score)
	vanya += score[0]

diff = total - vanya
scores = sorted(scores, key=lambda test: test[1])

essays = 0
num = 0
while diff > 0:
	if diff < r - scores[num][0]:
		essays += diff * scores[num][1]
		break
	else:
		essays += scores[num][1] * (r - scores[num][0])
		diff -= (r - scores[num][0])
		num += 1
print essays
