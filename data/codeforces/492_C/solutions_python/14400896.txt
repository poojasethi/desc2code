n , r , avg = [int(x) for x in raw_input().split()]
l = []
exam = 0
for i in range(n):
    a , b = [int(x) for x in raw_input().split()]
    exam += a
    c = [b , a]
    l.append(c)
    
need = n * avg - exam
ans = 0
if need > 0:
    l.sort()
    i = 0
    while need > 0 :
        b , a =l[i]
        if (r - a) < need:
            ans += (r - a) * b
            need = need - (r - a)
        else:
            ans += need * b
            need = 0
        i += 1

print ans
