n, r, avg = map(int, raw_input().split(' '))
needed = n * avg
tests = {}
for _ in xrange(n):
    a, b = map(int, raw_input().split(' '))
    needed = needed - a
    tests[b] = tests.get(b, 0) + r - a
essaycount = 0
for essays in sorted(tests.keys()):
    score = tests[essays]
    if needed > 0:
        points = min(needed, score)
        needed = needed - points
        essaycount = essaycount + (essays * points)
    else:
        break
print essaycount
