from itertools import *
n, r, avg = map(int, raw_input().split())
A = []
for i in xrange(n):
  A.append(map(int, raw_input().split()))

A = sorted(A, key=lambda a: a[1])

k = sum(zip(*A)[0])
s = 0
for a, b in A:
  if n*avg <= k:
    break
  s += min(r-a, n*avg - k)*b
  k += min(r-a, n*avg - k)
print s
