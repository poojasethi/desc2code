f = lambda: map(int, raw_input().split())
n, r, a = f()
b = sorted([f() for i in range(n)], key = lambda x: x[1])
s = sum(i[0] for i in b)
m, res = n * a - s, 0
for i in range(n):
  d = r - b[i][0]  
  if d > 0 and m > 0:
    q = min(m, d)
    m -= d
    res += q * b[i][1]
print res
