import heapq

def solve(n, r, avg, ab):
    h = []
    s = 0
    for a, b in ab:
        heapq.heappush(h, (b, a))
        s += a
    rem = avg*n-s
    res = 0
    while rem > 0:
        b, a = heapq.heappop(h)
        d = min(r-a, rem)
        a, rem = a+d, rem-d
        res += b*d
    return res

R = lambda: map(int, raw_input().split())
n, r, avg = R()
ab = [R() for i in range(n)]
print solve(n, r, avg, ab)
