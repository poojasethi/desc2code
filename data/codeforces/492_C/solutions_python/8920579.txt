import heapq

def solution(h,diff,r):
        count=0
        while h:
                essay,score=heapq.heappop(h)
                point=r-score
                if point >= diff:
                        count=count+essay*diff
                        return count
                else:
                        diff=diff-point
                        count=count+point*essay
        return count

n,r,avg=raw_input().split()
n=int(n)
r=int(r)
avg=int(avg)
total=n*avg
h=[]
score=0
for i in range(n):
        line=raw_input()
        marks,essay=map(int,line.split())
        score=score+marks
        heapq.heappush(h,map(int,line.split())[::-1])
if total <= score:
        print 0
else:
        diff=total-score
        print solution(h,diff,r)
