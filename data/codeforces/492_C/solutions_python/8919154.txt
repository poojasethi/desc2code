entrada1 = [int(x) for x in raw_input().split()]
n = entrada1[0]
r = entrada1[1]
avg = entrada1[2]
notas = []
media = 0

for i in range(n):
	temp = [int(x) for x in raw_input().split()]
	temp2 = (temp[1],temp[0])
	media+=temp[0]
	notas.append(temp2)
todos = media
media = media/n

pontos = avg*n-todos

notas.sort()
ensaios = 0
if pontos>0:
	for i in range(n):
		if pontos==0:
			break
		ensaios+=notas[i][0]*min(r-notas[i][1],pontos)
		pontos-=min(r-notas[i][1],pontos)		

print ensaios
