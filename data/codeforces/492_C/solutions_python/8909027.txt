n,r,avg=map(int,raw_input().split())
lis=[]
need=avg*n
ans=0
for _ in range(n):
    a,b=map(int,raw_input().split())
    lis.append((b,r-a))
    need-=a
i=0
lis.sort()
while i<n and need>0:
    b,a=lis[i]
    i+=1
    if need<a:
        ans+=b*need
        need=0
    else:
        need-=a
        ans+=b*a
print ans
