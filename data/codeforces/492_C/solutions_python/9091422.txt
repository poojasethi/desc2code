n, r, avg = map(int, raw_input().split())
l = []
tot=0
for i in range(n):
    x,y = map(int, raw_input().split())
    tot+=x
    l.append(( y, x))
l.sort()
ans = 0
tot = n*avg - tot
if tot < 0:
    tot = 0
for i in range(n):
    if tot == 0:
        break
    t = min( tot, r - l[i][1])
    tot -= t
    ans += t*l[i][0]
print ans
