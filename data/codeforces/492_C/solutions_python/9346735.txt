n,r,avg=map(int,raw_input().split())
a,b=[],[]
k=range(n)
for i in xrange(n):
    x,y=map(int,raw_input().split())
    a.append(x)
    b.append(y)
k.sort(key=lambda x:b[x])
left=avg*n-sum(a)
c=0
if left>0:
    for i in k:
        if left>r-a[i]:
            c+=b[i]*(r-a[i])
            left-=r-a[i]
        else:
            c+=left*b[i]
            left=0
        if left==0:
            break
print c
