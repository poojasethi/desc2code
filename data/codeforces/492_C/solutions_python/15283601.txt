n,r,avg=map(int,raw_input().split())
li=[]
s=0
for i in xrange(n):
    t=map(int,raw_input().split())
    li+=[[t[1],t[0]]]
    s+=t[0]
li.sort()
t=n*avg-s
e=0
if(t<=0):
    print "0"
else:
    for i in xrange(n):
        x=r-li[i][1]
        if(x<=t):
            e+=x*li[i][0]
            t-=x
        else:
            e+=t*li[i][0]
            t=0
    print e
