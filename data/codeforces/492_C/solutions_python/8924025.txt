n,r,avg = [int(x) for x in raw_input().split()]

d = []
s = 0
for i in xrange(n):
    a,b = [int(x) for x in raw_input().split()]
    s += a
    d.append([a, b])

d.sort(key=lambda x: x[1], reverse=False)

avg *= n


i = 0
res = 0
while s < avg and i < n:
    if d[i][0] < r:
        k = min(r-d[i][0], avg-s)
        s+=k
        res+=k*d[i][1]
    i+=1

print res
