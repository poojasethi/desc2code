from sys import stdin
def main():
    n, r, v = map(int, stdin.readline().split())
    s = 0
    a = []
    for _ in xrange(n):
        x, y = map(int, stdin.readline().split())
        s += x
        a.append((y, r-x))
    a.sort()
    ans = 0
    s = v * n - s
    for y, x in a:
       if s > 0:
           k = min(s, x)
           ans += k * y
           s -= k
    print ans
main()
