raw= raw_input

n, r, avg = map(int, raw().split())
s = n * avg
ms = [map(int, raw().split()) for _ in xrange(n)]
ms.sort(key=lambda x: x[1])
ref = 0
cm = sum([i[0] for i in ms])
if cm < s:
    for m, b in ms:
        if m < r and cm < s:
            nn = min(r - m, s - cm)
            cm += nn
            ref += nn * b
        if cm == s:
            break
print ref
