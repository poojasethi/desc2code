n , r, avg = map(int,raw_input().split())
li = []
sum_till = 0
sum_req = n*avg
for i in xrange(n):
    a, b = map(int,raw_input().split())
    sum_till += a 
    li.append((b,r-a))
li.sort()
count = 0
i  =0
while sum_till < sum_req :
    if   li[i][1] :
        d = min(li[i][1] , sum_req - sum_till)
        count += d*li[i][0]
        sum_req -= d
    i += 1
print count
