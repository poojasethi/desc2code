n,r,avg=map(int,raw_input().split())
l=[]
for i in xrange(n):
	a,b=map(int,raw_input().split())
	l.append([b,a])
l=sorted(l)
req=avg*n
cur=0
for i in l:
	cur+=i[1]

cost=0
rem=req-cur
for i in l:
	if rem>r-i[1]:
		rem-=(r-i[1])
		cost+=(r-i[1])*i[0]
	else:
		cost+=rem*i[0]
		rem=0
		break
print max(cost,0)
