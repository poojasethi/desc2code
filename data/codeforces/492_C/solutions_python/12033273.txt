L = lambda: map(int,raw_input().split())
n,r,avg = L()
scores = sorted([L() for _ in xrange(n)],key=lambda x:x[1])
tests = max(0,n*avg - sum(x[0] for x in scores))
ans = 0
for x,y in scores:
    if x < r:
        t = min(tests,r-x)
        tests -= t
        ans += t * y
    if not tests:
        break
print ans
