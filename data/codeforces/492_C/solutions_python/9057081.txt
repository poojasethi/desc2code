n, r, avg=map(int,raw_input().split())
sum = avg*n
sum1 = 0
ans = 0
a = []

for i in range(n):
  x, y = map(int,raw_input().split())
  sum1 += x
  a.append((y, x))
a.sort()
if (sum1>sum):
  print 0
  exit(0)
for i in a:
  if (sum1+r-i[1]>=sum):
    ans += i[0]*(sum-sum1)
    break
  else:
    ans += i[0]*(r-i[1])
    sum1 += r-i[1]
print ans
