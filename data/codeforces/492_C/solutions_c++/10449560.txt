#include<iostream>
#include<algorithm>
using namespace std;

const int maxn = 210 * 1000;

typedef long long ll;

int n;
ll r , avg, S, J;
pair<ll, ll> p[maxn];

int main(){
	ios::sync_with_stdio(0);
	cin >> n >> r >> avg;
	for (int i=0 ; i<n ; i++){
		cin >> p[i].second >> p[i].first;
		S += p[i].second;
	}
	sort(p, p+n);
	int I = 0;
	while(S < n * avg){
		ll x = min(n * avg - S, r-p[I].second);
		J += x * p[I].first;
		S += x;
		I++;
	}
	cout << J << endl;
}
