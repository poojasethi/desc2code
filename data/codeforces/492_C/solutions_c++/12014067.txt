#include<iostream>
#include<algorithm>
using namespace std;
struct node{
	int a,b;
}e[100001];
bool cmp(node a,node k)
{
	return a.b<k.b;
}
int main()
{
	int p,i;
	long long ans=0,r,n,sum=0;
	cin>>n>>r>>p;
	for(i=1;i<=n;i++)
	{
		cin>>e[i].a>>e[i].b;
		sum+=e[i].a;
	}
	if(sum>=n*p)	{
		cout<<0;return 0;
	}
	sort(e+1,e+n+1,cmp);
	for(i=1;;i++)
	{
		if(sum>=n*p)
			break;
		long long f=min(n*p-sum,r-e[i].a);
		ans+=f*e[i].b;
		sum+=f;
	}
	cout<<ans;
	return 0;
}
