#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
long long a,s,d,f,g,h,j,k,l,i,n,m;
pair<int,int> p[100002];
main(){
cin>>n>>m>>k;
for(i=0;i<n;i++){
cin>>a>>s;
p[i].first=s;
p[i].second=m-a;
l+=a;
}s=0;
sort(p,p+n);i=0;
while(l<k*n){//cout<<p[i].first<<" "<<p[i].second<<" "<<s<<endl;
if(l+p[i].second<k*n) {l+=p[i].second;s+=(long long)p[i].first*p[i].second;}
else {s+=(k*n-l)*p[i].first;break;}
i++;}

cout<<s;
}
