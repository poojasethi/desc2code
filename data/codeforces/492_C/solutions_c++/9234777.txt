#include <bits/stdc++.h>
using namespace std;
typedef long long ull;
int main(){
  ull ans = 0,n,r,avg,sum = 0;
  cin >> n >> r >> avg;
  pair<ull,ull> A[n];
  for(int i = 0;i < n;i++){
    cin >> A[i].second >> A[i].first;
    sum += A[i].second;
  }
  sort(A,A + n);
  int a,b;
  for(int i = 0;i < n && sum/n < avg;i++){
    a = min(n * avg - sum,r - A[i].second);
    sum += a;
    ans += a * A[i].first;
  }
  cout << ans << "\n";
  return 0;
}
