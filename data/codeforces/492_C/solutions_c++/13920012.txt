#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main(){
long long n,r,avg,m=0,i=0,sum=0,ans=0;
pair<long long, long long> a[100001];
scanf("%d %d %d",&n,&r,&avg);
for(int i=0;i<n;i++){
scanf("%d %d",&a[i].second,&a[i].first);
sum+=a[i].second;
}
sort(a,a+n);
while(sum<avg*n){
long long tmp=min(avg*n-sum,r-a[i].second);
ans+=tmp*a[i].first;
sum+=tmp;
i++;
}
printf("%I64d\n",ans);
return 0;
}
