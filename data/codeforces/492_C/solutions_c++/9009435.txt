#include <iostream>
#include <algorithm>
#include <map>
using namespace std;
int main()
{
	long long n,r,avg,sum,i,cnt=0,t;
	pair<int,int> p[100010];
	cin>>n>>r>>avg;
	sum=n*avg;
	for(i=0;i<n;i++)
	{
		cin>>p[i].second>>p[i].first;
		sum-=p[i].second;
	}
	sort(p,p+n);	//sort by first
	i=0;
	while(sum>0)
	{
		if(r-p[i].second>sum)
		{
			t=sum;
		}
		else
		{
			t=r-p[i].second;
		}
		cnt+=t*p[i].first;
		sum-=t;
		i++;
	}
	cout<<cnt<<endl;
	return 0;
}
