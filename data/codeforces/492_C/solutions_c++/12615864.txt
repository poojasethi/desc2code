#include<bits/stdc++.h>
using namespace std;
long long n,av,r,i,res,sum;
pair<long long , long long> a[100500];

int main(void){
  cin>>n>>r>>av;
  for(i = 0 ; i < n ; i ++)
    cin>>a[i].second>>a[i].first,
      sum +=a[i].second;
  sort(a,a+n);
  res = i = 0;
  while(sum < av*n){
    long long temp = min(av * n -sum, r - a[i].second);
    res += temp * a[i].first;
    sum += temp;
    i++;
  }

  cout<<res<<endl;
}
