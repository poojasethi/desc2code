#include<bits/stdc++.h>
using namespace std;
pair<long long,long long> g[100001];
int main()
{
    long long n,r,avg;
    cin>>n>>r>>avg;
    long long sum=0,i;
    for( i=0;i<n;i++)
        cin>>g[i].second>>g[i].first,sum+=g[i].second;
    sort(g,g+n);
    long long e=0,x;
    long long rem=n*avg-sum;


    for( i=0; i<n && rem>0 ; i++) {
        x=min(r-g[i].second,rem);
        e=e+x*g[i].first;
        rem=rem-x;

        }

    cout<<e;
    return 0;
}
