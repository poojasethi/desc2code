#include<iostream>
#include<algorithm>

using namespace std;

#define int long long
pair<int, int> a[100005];
int sum;

main()
{
	int n, r, avg;
	cin >> n >> r >> avg;
	int x = avg * n;
	for(int i = 0; i < n; i++)
	{
		cin >> a[i].second >> a[i].first;
		sum += a[i].second;
	}
	sort(a, a + n);
	int ans = 0;
	sum = min(x, sum);
	for(int i = 0; i < n; i++)
	{
		ans += min(r - a[i].second, x - sum) * a[i].first;
		sum += min(r - a[i].second, x - sum);
	}
	cout << ans << endl;
	return 0;
}
