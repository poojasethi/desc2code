#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long n,r,i,avg,sum=0,count=0,tmp;
    cin>>n>>r>>avg;
    pair<long long,long long>a[n];

    for(i=0;i<n;i++){
        cin>>a[i].second>>a[i].first;
        sum+=a[i].second;
    }
    sort(a,a+n);

    i=0;
    while(sum<avg*n){
        tmp=min(avg*n-sum,r-a[i].second);
        count+=tmp*a[i].first;
        sum+=tmp;
        i++;
    }
    cout<<count;
    return 0;
    }
