#include<iostream>
#include<algorithm>
using namespace std;
pair<int ,int>a[111111];
long long n,r,avg,ans;
long long k=0;
int main(){
    cin>>n>>r>>avg;
    ans=avg*n;
    for(int i=0;i<n;i++){
        cin>>a[i].second>>a[i].first;
        ans-=a[i].second;
    }
    sort(a,a+n);
    for(int i=0;ans>=0&&i<n;i++){
        long long x=min(r-a[i].second,ans);
        ans-=x;
        k+=x*a[i].first;
    }
    cout<<k<<endl;
    return 0;
}
