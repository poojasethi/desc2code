#include <bits/stdc++.h>

using namespace std;

long long t,s,n,r,avr,sum;
pair <long long,long long> p[100001];

int main()
{ cin>>n>>r>>avr;
  for(int i=0;i<n;i++) {cin>>p[i].second>>p[i].first; s+=p[i].second;}
  sort(p,p+n);
  for(int i=0;i<n;i++) if (s>=n*avr) break; else 
    {t=min(n*avr-s,r-p[i].second); s+=t; sum+=t*p[i].first;}
  cout<<sum<<endl;
}
