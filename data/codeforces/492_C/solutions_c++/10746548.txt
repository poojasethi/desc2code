#include<bits/stdc++.h>
using namespace std;
#define M2H 100500
pair<long long,long long>p[M2H];
long long n,r,avg,i,sum,l,t;
long long ans=0ll;
int main()
{
    cin>>n>>r>>avg;
    l=n*avg;
    for(i=0;i<n;i++)
    {
    cin>>p[i].second>>p[i].first;
    l-=p[i].second;
    }
    sort(p,p+n);
    for(i=0;l>0;i++)
    {
        t=min(r-p[i].second,l);
        ans+=t*p[i].first;l-=t;
    }
    cout<<ans<<endl;
}
