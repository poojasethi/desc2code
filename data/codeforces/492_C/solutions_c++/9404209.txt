#include <stdio.h>
#include <algorithm>
using namespace std;
long long n,m,as,ans,avg,vv,x,y,p,i,h,k;
pair <long long,long long> a[100005];
int main(){
	i=0;
	scanf("%d%d%d",&n,&m,&avg);
	while(i<n){
		scanf("%d%d",&a[i].second,&a[i].first);
		as=as+a[i].second;
        i++;	
	}	
	sort(a,a+n);
	ans=0;
	h=0;
	while(as<avg*n){
		long long qw=min(avg*n-as,m-a[h].second);
		ans=ans+qw*a[h].first;
		as+=qw;
		h+=1;
	}
   printf("%I64d\n",ans);
   return 0;
}
