#include <bits/stdc++.h>

using namespace std;

long long n, r, avg;
pair <long long, long long> uh[200000];

int main() {
  scanf ("%lld %lld %lld", &n, &r, &avg);
  avg *= n;
  for (int i = 0; i < n; i++) {
    scanf ("%d %d", &uh[i].second, &uh[i].first);
    avg -= uh[i].second;
  }
  long long ans = 0;
  sort (uh, uh + n);
  int at = 0;
  while (avg > 0) {
    ans += uh[at].first * min (avg, r - uh[at].second);
    avg -= min (avg, r - uh[at].second);
    at++;
  }
  printf ("%lld\n", ans);
}
