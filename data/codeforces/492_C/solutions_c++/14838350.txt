#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
pair<ll,ll> A[100001];

int main()
{
    long long n,r,avg,a,b,c=0,ess=0ll,k,t;
    cin>>n>>r>>avg;
    t=avg*n;
    for(int i=0;i<n;i++)
    {
        cin>>A[i].second>>A[i].first;
        t-=A[i].second;
    }
    sort(A,A+n);
    for(int i=0;i<n && t>0;i++)
    {
        k=min(t,r-A[i].second);
        ess+=(k*A[i].first);
        t-=k;
    }
    cout<<ess;
    return 0;
}
