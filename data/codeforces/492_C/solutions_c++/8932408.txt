#include <iostream>
#include <algorithm>
using namespace std;
long long int n,r,avg,s,e;
pair<int,int>a[100010];
int main(){
  cin>>n>>r>>avg;
  for(int i=0;i<n;i++){
    cin>>a[i].second>>a[i].first;
    s+=a[i].second;
  }
  sort(a,a+n);
  e=avg*n-s;
  s=0;
  for(int i=0;;i++){
    if(e<1){cout<<s<<endl;break;}
    if(e-(r-a[i].second)>=0){
    e-=(r-a[i].second);
    s+=a[i].first*(r-a[i].second);
    }
    else{
      s+=a[i].first*e;
      e=0;
    }
  }
  return 0;
}
