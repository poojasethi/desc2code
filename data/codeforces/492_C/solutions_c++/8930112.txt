#include <iostream>
#include <algorithm>
#define fi first
#define se second
using namespace std;

long long n,r,avg,ans,now,i;
pair < int , int > p[100010];

main() {
 cin>>n>>r>>avg;
 avg*=n;
 for (i=1;i<=n;i++)
  cin>>p[i].se>>p[i].fi,now+=p[i].se;
 sort(p+1,p+n+1);
 if (avg > now)
 for (i=1;i<=n;i++)
  {
   long long z=r-p[i].se;
   if (now+z < avg) { now+=z; ans+=z*p[i].fi; continue; }
   z=avg-now;
   now+=z;
   ans+=z*p[i].fi;
  }
 cout<<ans<<endl;
}
