#include <bits/stdc++.h>

using namespace std;

int main(){
	long long n, r, avg;
	cin>>n>>r>>avg;
	vector<pair<long long, long long> > v;
	v.resize(n);
	long long tot = 0;
	for(int i=0;i<n;i++){
		long long a, b;
		cin>>a>>b;
		v[i] = make_pair(b, a);
		tot += a;
	}
	long long req = n*avg;
	sort(v.begin(), v.end());
	long long ret = 0;
	for(int i=0;i<n;i++){
		long long x = min((r-v[i].second), max(req-tot, 0ll));
		ret += x*v[i].first;
		tot += x;
	}
	cout<<ret<<endl;
}
