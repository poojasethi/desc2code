
#include<bits/stdc++.h>
using namespace std;
long long a[1000005];
int main(){
	long long n, r, avg;
	cin >> n >> r >> avg;
	long long x, y, s = n*avg;
	for( int i=0 ; i<n ; i++ ){
		scanf("%I64d%I64d", &x, &y);
		a[y]+=r-x;
		s -= x;
	}
	long long ans = 0;
	for( int i=0 ; i<10000001 ; i++ ){
		if( s>a[i] ){
			s-=a[i];
			ans += i*a[i];
		}
		else{
			ans += i*s;
			break;
		}
	}
	cout << ans << endl;
	return 0;
}
