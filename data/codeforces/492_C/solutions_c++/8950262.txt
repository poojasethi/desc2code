#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	long long n,r,avg;
	cin >> n >> r >> avg;
	long long sum = 0, ans = 0, j = 0;
	long long z = n*avg;
	pair<long long, long long> * p = new pair<long long, long long>[n];

	for (int i=0; i<n; ++i) {
		cin >> p[i].second >> p[i].first;
		sum += p[i].second;
	}
	sort(p, p+n);
	
	while (sum<z) {
		long long t = min(z-sum, r-p[j].second);
		ans += t*p[j].first;
		sum += t;
		++j;
	}
	cout << ans;
}
