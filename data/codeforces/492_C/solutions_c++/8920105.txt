#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<algorithm>
using namespace std;
long long n,m,i,j,k,p,q,l,r,x,y,ans;
pair <int,int> a[100001];
int main()
{
    cin>>n>>m>>k;
    for (i=1;i<=n;i++)cin>>a[i].second>>a[i].first,p+=a[i].second;
    sort(a+1,a+n+1);
    k*=n;
    k-=p;
    i=1;
    while(k>0){
        p=m-a[i].second;
        p=min(p,k);
        ans+=p*a[i].first;
        k-=p;
        i++;
    }
    cout<<ans;
    return 0;
}
