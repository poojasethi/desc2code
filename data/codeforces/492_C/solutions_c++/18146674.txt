#include <bits/stdc++.h>
using namespace std;
pair<long long,long long> a[100001];
int main()
{
	long long n,r,avg,sum=0;
	cin>>n>>r>>avg;
	for(int i=0;i<n;i++)
	{
		long long x,y;
		cin>>x>>y;
		a[i]=make_pair(y,x);
		sum+=x;
	}
	long long ans=0;
	sort(a,a+n);
	long long i=0;
	while(sum<avg*n)
	{
		ans+=(a[i].first*min(r-a[i].second,avg*n-sum));
		sum+=min(r-a[i].second,avg*n-sum);
		a[i].second=r;		
		i++;
	}
	cout<<ans;
}
