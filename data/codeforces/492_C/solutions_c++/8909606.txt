#include <iostream>
#include <algorithm>
using namespace std;
const int N=1e6+9;
pair<long long ,long long>a[N];
int main()
{
	long long n,r,v,s=0,t=0;
	cin>>n>>r>>v;
	for(int i=0;i<n;i++){
		cin>>a[i].second>>a[i].first;
		s+=a[i].second;
	}
	sort(a,a+n);
	long long i=0,c=v*n-s;
	while(c>0)
	{
		if(r-a[i].second>=c){
			t+=c*a[i].first;
			break;
		}
		else
		{
			t+=(r-a[i].second)*a[i].first;
			c-=r-a[i].second;
		}
		i++;
	}
	cout<<t<<endl;
	
}
