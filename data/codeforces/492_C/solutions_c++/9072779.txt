#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
struct stt{int a,b;}st[100000];
bool cmp(stt st1,stt st2)
{return st1.a<st2.a;}
int main()
{
	int n,a,b;ll la,o=0;
	scanf("%d%d%d",&n,&a,&b);
	la=(ll)b*n;
	for(int i=0;i<n;++i)
	{
		scanf("%d%d",&b,&st[i].a);
		la-=b;st[i].b=a-b;
	}
	sort(st,st+n,cmp);
	for(int i=0;i<n;++i)
	{
		if(la<=0)break;
		a=min(la,(ll)st[i].b);
		o+=(ll)a*st[i].a;
		la-=a;
	}
	printf("%I64d\n",o);
}
