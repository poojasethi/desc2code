#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
	ll n,r,av,t,s=0,ans=0;
	vector< pair < ll, ll > > ex;
	cin >> n >> r >> av;
	for ( int i = 0; i < n ; i++)
	{
		int a,b;
		cin >> a >> b;
		ex.push_back(make_pair(b,a));
		s += a;
		
	}
	sort(ex.begin(),ex.end());
	for(int i = 0 ; i < n ; i ++ )
	{
		if( s >= av *n )
			break;
		t = min( av * n - s, r - ex[i].second);
		s += t;
		ans += ex[i].first * t;
	}
	cout << ans << endl;

}
