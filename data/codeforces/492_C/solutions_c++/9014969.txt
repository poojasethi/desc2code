#include <bits/stdc++.h>
using namespace std;
long long n, r, avg, d, m, s;
pair<int, int> a[100005];

int main() {
//    freopen("input.txt", "r", stdin);
    int i;
    cin >> n >> r >> avg;
    d = n * avg;
    for (i = 0; i < n; i++) {
        scanf("%d %d", &a[i].second, &a[i].first);
        d -= a[i].second;
    }
    sort(a, a + n);
    for (i = 0; i < n && d > 0; i++) {
        m = min(r - a[i].second, d);
        s += m * a[i].first;
        d -= m;
    }
    cout << s;
}
