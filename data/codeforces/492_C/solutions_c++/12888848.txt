#include<bits/stdc++.h>
using namespace std;
#define ll long long
pair<int,int> a[1000*100+5];
int main(){
    ll ans=0,n,r,avg,sum=0; cin>>n>>r>>avg;
    for(int i=0;i<n;i++){ cin>>a[i].second>>a[i].first;sum+=a[i].second;}
    sort(a,a+n);
    int ite=0;
    while(sum<avg*n){
        if(sum+r-a[ite].second<=avg*n){sum+=r-a[ite].second;ans+=a[ite].first*(r-a[ite].second);ite++;}
        else{
            ans+=(avg*n-sum)*a[ite].first;cout<<ans;return 0;
        }
    }
    cout<<ans;
}
