#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
struct t{ ll q; ll p;
};
bool thanh(t x,t y) { return(x.p<y.p);
}
main()
{
	ll n,r,avg; cin>>n>>r>>avg;
	ll sum=n*avg; vector<t> a(n);
	for(ll x=0;x<n;x++) { cin>>a[x].q>>a[x].p; sum-=a[x].q;
	}
	sort(a.begin(),a.begin()+n,thanh);
	ll ans=0;
	int i=0;
	while(sum>0)
	{
		ans+=min(r-a[i].q,sum)*a[i].p;
		sum-=min(r-a[i].q,sum);
		i++;
	}
	cout<<ans;
}
