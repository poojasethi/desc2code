#include <bits/stdc++.h>
using namespace std;

struct st {
	int a, b;
} e[100005];
long long n, r, avg, d, s, k;

bool cmp (st p, st q) {
	if (p.b != q.b) return p.b < q.b;
	else return p.a < q.a;
}

int main() {
	int i;
	cin >> n >> r >> avg;
	d = n * avg;
	for (i = 0; i < n; i++) {
		scanf("%d%d", &e[i].a, &e[i].b);
		d -= e[i].a;
	}
	sort (e, e+n, cmp);
	for (i=0; i<n && d>0; i++) {
		if (e[i].a < r) {
			k = min(r - e[i].a, d);
			s += e[i].b * k;
			d -= k;
		}
	}
	cout << s;
}
