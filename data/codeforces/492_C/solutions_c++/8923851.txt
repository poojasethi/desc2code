#include <bits/stdc++.h>
using namespace std;

pair<int, int> a[100000];
int main(){
  
  int i, n;
  long long avg, r, sum = 0, need, ans = 0;
  cin >> n >> r >> avg;
  need = avg * n;
  for(i=0; i<n; i++){
    cin >> a[i].second >> a[i].first;
    need -= a[i].second;
  }
  sort(a, a+n);
  i=0;
  while(i<n && need>0){
    long long x = r-a[i].second;
    x = min(x, need);
    need -= x;
    ans += x*a[i].first;
    i++;
  }
  cout << ans << endl;;
  
  return 0;
}
