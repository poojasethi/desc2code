#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    int n;
    long long r,avg,res=0,h,sum=0,tmp;
    cin>>n>>r>>avg;
    pair <long long, long long> a[n+1];
    for (int i=1;i<=n;i++)
    {
        cin>>a[i].second>>a[i].first;
        sum+=a[i].second;
    }
    sort(a+1,a+n+1);
    int i=1;
    while(sum<avg*n)
    {
        tmp=min(r-a[i].second,avg*n-sum);
        sum+=tmp;
        res+=tmp*a[i].first;
        i++;
    }
    cout<<res;
}
