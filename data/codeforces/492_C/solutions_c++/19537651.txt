#include<iostream>
#include <algorithm>
using namespace std;
typedef long long ll;
ll n,r,a,sum=0,avg;
pair<ll,ll>e[100005];
int main(){
	cin>>n>>r>>a;
	avg=n*a;
	for(int i=0; i<n; i++) cin>>e[i].second>>e[i].first, avg-=e[i].second;
	sort(e,e+n);
	if(avg<=0)
	{
		cout<<0;
		return 0;
	}
	for(int i=0; i<n; i++)
	{
		sum+=min(r-e[i].second,avg)*e[i].first;
		avg-=min(r-e[i].second,avg);
		if(avg==0) break;
	}
	cout<<sum;
}
