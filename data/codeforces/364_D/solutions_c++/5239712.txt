#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <ctime>
#define ll long long
using namespace std;

int N;
ll a[1000005],p[7005],q[7005];

int main()
{
	ll ans=0;
	scanf("%d",&N),srand(time(0));
	for (int i=1; i<=N; i++) scanf("%I64d",&a[i]);
	for (int x=10; x; x--)
	{
		ll y=a[(1ll*rand()*rand()+rand())%N+1]; int t=0;
		for (ll i=1; i*i<=y; i++) if (y%i==0)
		{
			p[++t]=i;
			if (i*i<y) p[++t]=y/i;
		}
		sort(p+1,p+t+1);
		for (int i=1; i<=t; i++) q[i]=0;
		for (int i=1; i<=N; i++) q[lower_bound(p+1,p+t+1,__gcd(a[i],y))-p]++;
		for (int i=t; i&&p[i]>ans; i--)
		{
			int s=0;
			for (int j=i; j<=t; j++) if (p[j]%p[i]==0) s+=q[j];
			if (s*2>=N) ans=p[i];
		}
	}
	printf("%I64d\n",ans);
}
