//In the Name of God

#include<iostream>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define ll long long

using namespace std;

const int maxn=1000100;
const int maxN=100100;
int n,cnt[maxN];
ll ans,a[maxn],b[maxn],B[maxN];

inline ll gcd(ll A,ll B)
{
  ll c;
  while(B)
    c=A,A=B,B=c%B;
  return A;
}

inline void check(ll X)
{
  int N=0;
  for(ll i=1;i*i<=X;i++)
    if(X%i==0)
      {
	B[N++]=i;
	if(i*i!=X)
	  B[N++]=X/i;
      }
  sort(B,B+N);
  memset(cnt,0,sizeof cnt);
  for(int i=0;i<n;i++)
    cnt[lower_bound(B,B+N,gcd(a[i],X))-B]++;
  for(int i=N;i--;)
    {
      int C=0;
      for(int j=i;j<N;j++)
	if(B[j]%B[i]==0)
	  C+=cnt[j];
      if(C>=(n+1)/2)
	ans=max(ans,B[i]);
    }
}

int main()
{
  ios_base::sync_with_stdio(0);
  srand(37);
  cin>>n;
  for(int i=0;i<n;i++)
    cin>>a[i];
  random_shuffle(a,a+n);
  for(int i=0;i<min(10,n);i++)
    check(a[i]);
  cout<<ans<<endl;
  return 0;
}
