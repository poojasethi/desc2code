#include<cstdio>
#include<iostream>
#include<cstdlib>
#include<ctime>
#include<cmath>
#include<algorithm>
#include<cstring>
using namespace std;

long long a[1000001],b[10000];
long long c[10000];
int main(){
	long long n,i,j,ans=1;
	scanf("%I64d",&n),srand(time(0));
	for(i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	for(i=1;i<=10;i++){
		long long k=a[(1ll*rand()*rand()+rand())%n],s;
		long long p=0;
		for(j=1;j<=sqrt(k);j++) 
		    if(k%j==0){
			p++;b[p]=j;
			if(j*j<k){
				p++;
				b[p]=k/j;
			}
		}
		sort(b+1,b+p+1);
		memset(c,0,sizeof(c));
		for(j=0;j<n;j++){
			long long g=__gcd(k,a[j]);
			c[lower_bound(b+1,b+p+1,g)-b]++;
		}
		
		for(j=p;j>=1;j--){
			s=0;
			for(int m=j;m<=p;m++) if(b[m]%b[j]==0) s+=c[m];
			if(s>=(n+1)/2){
				if(b[j]>ans) ans=b[j];
			} 
		}                   
	}
	printf("%I64d\n",ans);
	return 0;
	
}
