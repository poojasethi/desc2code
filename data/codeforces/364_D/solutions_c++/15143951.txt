#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;

typedef long long L;

int n, nd;
L a[1000006];
L d[2000005];
L s[1000006];
L ans;

main(){
  scanf("%d",&n);
  for(int i=0;i<n;i++)scanf("%I64d",a+i);
  for(int _=0;_<12;_++){
    L x=a[rand()*rand()%n];
    nd=0;
    for(L y=1;y*y<=x;y++){
      if(x%y==0){ 
	d[nd++]=y;
	if(y*y<x) d[nd++]=x/y;
      }
    }
    sort(d,d+nd);
    fill(s,s+nd,0);
    for(int i=0;i<n;i++)s[lower_bound(d,d+nd,__gcd(a[i],x))-d]++;
    for(int i=0;i<nd;i++){
      for(int j=i+1;j<nd;j++)if(d[j]%d[i]==0)s[i]+=s[j];
      if(s[i]>=(n+1)/2)ans=max(ans,d[i]);
    }
  }
  printf("%I64d\n",ans);
}
