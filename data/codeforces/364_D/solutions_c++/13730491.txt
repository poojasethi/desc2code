#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e6+100;
const int maxrand = 9;
const int maxdiv = 7e3;

#define rep(i,n) for(int i = 0;i < n;i++)

int c[maxdiv];
long long d[maxn] , ans , a[maxn] , b[maxdiv];

int main(){
	srand(727);
	int n; scanf("%d",&n); rep(i,n) scanf("%I64d",a+i);
	rep(i,n) d[i] = a[i];
	int sz = unique(d,d+n) - d;
	random_shuffle(d,d+sz);
	rep(q,min(sz , maxrand)){
		long long s = 0 , now = d[q];
		for(int i = 1;(long long)i*i <= now;i++){
			if(now % i == 0){
				b[s++] = i;
				if(now / i != i) b[s++] = now / i;	
			}	
		}
		sort(b,b+s);
		memset(c,0,s*sizeof(c[0]));
		rep(i,n) c[lower_bound(b,b+s,__gcd(a[i] , now)) - b]++;
		rep(i,s){
			for(int j = i + 1;j < s;j++) if(b[j] % b[i] == 0) c[i] += c[j];
			if(2*c[i] >= n) ans = max(ans , b[i]);
		}
	}
	return(printf("%I64d\n",ans) , 0);
}
