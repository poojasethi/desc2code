#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1000010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int64 i,j,k,n,m,ti,an;
int64 a[N],b[50000],s[50000];
int main() {
	scanf("%I64d",&n); m=(n+1)/2;
	For(i,1,n) scanf("%I64d",&a[i]);
	For(ti,1,10) {
		int64 A=a[(rand()%40000)*(rand()%40000)%n+1];
		*b=0;
		for (i=1;i*i<=A;i++) if (A%i==0) b[++*b]=i;
		for (;i;i--) if (A%i==0&&i*i<A) b[++*b]=A/i;
		For(i,1,*b) s[i]=0;
		For(i,1,n) {
			int64 C=__gcd(A,a[i]);
			s[lower_bound(b+1,b+*b+1,C)-b]++;
		}
		for (i=*b;i;i--) {
			int sum=0;
			For(j,i,*b) if (b[j]%b[i]==0) sum+=s[j];
			if (sum>=m) {
				an=max(an,b[i]);
				break;
			}
		}
	}
	printf("%I64d\n",an);
	//for(;;);
	return 0;
}
