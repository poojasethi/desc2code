//In the Name of God

#include<iostream>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define ll long long

using namespace std;

const int maxn=1000100;
const int maxN=100100;
int N,n,cnt[maxN];
ll ans,a[maxn],b[maxn],B[maxN];

inline ll bmm(ll A,ll B)
{
  for(ll C;B;)
    C=A,A=B,B=C%B;
  return A;
}

inline void check(ll x)
{
  for(int i=0;i<n;i++)
    b[i]=bmm(a[i],x);
  N=0;
  for(ll i=1;i*i<=x;i++)
    if(x%i==0)
      {
	B[N++]=i;
	if(i*i!=x)
	  B[N++]=x/i;
      }
  sort(B,B+N);
  memset(cnt,0,sizeof cnt);
  for(int i=0;i<n;i++)
    cnt[lower_bound(B,B+N,b[i])-B]++;
  for(int i=N;i--;)
    {
      int C=0;
      for(int j=i;j<N;j++)
	if(B[j]%B[i]==0)
	  C+=cnt[j];
      if(C>=(n+1)/2)
	{
	  ans=max(B[i],ans);
	  break;
	}
    }
}

int main()
{
  ios_base::sync_with_stdio(0);
  cin>>n;
  for(int i=0;i<n;i++)
    cin>>a[i];
  srand(37);
  random_shuffle(a,a+n);
  for(int i=0;i<min(10,n);i++)
    check(a[i]);
  cout<<ans<<endl;
  return 0;
}
