#include<bits/stdc++.h>

#define	FP( ii,aa,bb )	for( int ii=aa;ii<=bb;ii++ )
#define FM( ii,aa,bb )	for( int ii=aa;ii>=bb;ii-- )
#define	lli	long long int
using namespace std;

lli n,ind,x,res,cnt,a[2000000],bol[10000],s[10000];

lli gcd( lli a,lli b ){
	if( !b )	return a;
	return gcd( b,a%b );
}

int main(){
	cin >> n;
	FP( i,1,n )	cin >> a[i];
	srand(time(0));
	FP( dene,1,10 ){
		x = a[ rand()*rand()%n+1 ];
		memset( s,0,sizeof s );
		ind = 0;
		for( lli i=1;i*i<=x;i++ ){
			if( x%i==0 ){
				bol[++ind] = i;
				if( i*i!=x )	bol[++ind] = x/i;
			}
		}
		sort( bol+1,bol+ind+1 );
		FP( i,1,n )	s[ lower_bound( bol+1,bol+ind+1,gcd( x,a[i] ) )-bol ]++;
		FM( i,ind,1 ){
			cnt = 0;
			FP( j,i,ind )
				if( bol[j]%bol[i]==0 )	cnt += s[j];
			if( cnt>=(n+1)/2 )	res = max( res,bol[i] );
		}
	}

	cout << res << endl;
}
