#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int Maxn=1e6+6;

ll a[Maxn],d[4*Maxn],cntd,n,x,ans,cnt[4*Maxn];

ll gcd(ll a,ll b) { return  b==0?a:gcd(b,a%b); }
void init()
{
	scanf("%I64d",&n);
	for(int i=0;i<n;i++)
	  scanf("%I64d",&a[i]);
}

void solve()
{
	srand(time(NULL));
	ans=1;
	for(int s=0;s<10;s++)
	{
		x=a[((rand()<<15)^rand())%n];
		cntd=0;
		for(ll i=1;i*i<=x;i++)
		  if(x%i==0)
		  {
			 d[cntd++]=i;
			 if(i*i!=x)
			   d[cntd++]=x/i;
		  }
		sort(d,d+cntd);
		memset(cnt,0,sizeof(cnt));
		for(int i=0;i<n;i++)
		{
			ll g=gcd(x,a[i]);
			cnt[lower_bound(d,d+cntd,g)-d]++;
		}
		for(int i=0;i<cntd;i++)
		{
			for(int j=i+1;j<cntd;j++)
			  if(d[j]%d[i]==0)
				cnt[i]+=cnt[j];
			if(cnt[i]*2>=n)
			  ans=max(ans,d[i]);
		}
	}
	printf("%I64d\n",ans);
}
int main()
{
	init();
	solve();
	return 0;
}
