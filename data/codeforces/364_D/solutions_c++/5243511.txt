#include<ctime>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
int cnt[10010],top;
long long a[1000010],fac[10010];
int main(){
    srand(time(0));
    int i,j,n,m,k,step=10;
    long long x,c,as=1;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%I64d",&a[i]);
    while(step--){
        random_shuffle(a,a+n);
        x=a[rand()%n];
        top=0;
        for(c=1;c*c<x;c++)
            if(x%c==0){
                fac[top++]=c;
                fac[top++]=x/c;
            }
        if(c*c==x)fac[top++]=c;
        for(i=0;i<top;i++)cnt[i]=0;
        sort(fac,fac+top);
        for(i=0;i<n;i++){
            k=lower_bound(fac,fac+top,__gcd(a[i],x))-fac;
            cnt[k]++;
        }
        for(i=0;i<top;i++){
            m=0;
            for(j=i;j<top;j++)
                if(fac[j]%fac[i]==0)
                    m+=cnt[j];
            if(m*2>=n)
                as=max(as,fac[i]);
        }
    }
    printf("%I64d\n",as);
    return 0;
}
