#include <cstdio>
#include <algorithm>
using namespace std;

int n, limit, i, j, q, h, len, x, s[1100000], S;
long long a[1100000], ans, b[1100000], z[1100000];

long long gcd(long long x, long long y) {
	while (x) {
		swap(x, y);
		x %= y;
	}
	return y;
}

bool cmp(long long x, long long y) {
	return x > y;
}

int main() {
	scanf("%d", &n);
	for (i = 1; i <= n; i++)	scanf("%I64d", &a[i]);
	limit = 10;
	ans = 1;
	while (limit--) {
		x = (rand() * 32767 + rand()) % n + 1;
		for (i = 1; i <= n; i++)	b[i] = gcd(a[x], a[i]);
		sort(b + 1, b + n + 1, cmp);
		for (len = 0, q = 1; q <= n; q = h + 1) {
			for (h = q; h < n && b[h + 1] == b[q]; h++);
			z[++len] = b[q];
			s[len] = h - q + 1;
			for (S = 0, j = 1; j <= len; j++)
				if (z[j] % z[len] == 0)	S += s[j];
			if (S >= (n + 1) / 2) {
				ans = max(ans, b[q]);
				break;
			}
		}
	}
	printf("%I64d\n", ans);
}
