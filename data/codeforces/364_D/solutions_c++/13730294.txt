# include <bits/stdc++.h>
using namespace std ;

const int Max = 1e6 + 110 ;
const int Maxc = 2e4 ;
const int Rand = 9 ;
long long a[Max] ,c[Maxc] ,v[Maxc] ;

int main(){
//	ios_base :: sync_with_stdio(false) ;cin.tie(0) ;cout.tie(0) ;
		srand(110) ;
	int n ;
	long long ans = 0;
	scanf("%d", &n);
	for(int i=0 ;i<n ;i++)  scanf("%I64d", &a[i]);
	random_shuffle(a , a+n) ;
	for(int w=0 ;w<min(Rand , n) ;w++){
		long long x = a[w] ;

		int k = 0 ;
		for(int i=1 ;(long long) i*i <= x ;i++)
			if(x%i == 0) {
					v[k++] = i ;
					if( (long long) i*i != x) v[k++] = x/i ;
			}
		
		sort(v ,v+k) ;

		for(int i=0 ;i<k ;i++) c[i] = 0 ;

		for(int i=0 ;i<n ;i++)
			c[lower_bound(v ,v+k ,__gcd(x ,a[i])) - v] ++ ;

		for(int i=0 ;i<k ;i++)
			for(int j=i+1 ;j<k ;j++)
				if(v[j] % v[i] == 0)  c[i] += c[j] ;
		
		int half = n/2 ;
		if(n % 2) half ++ ;

		for(int i=0 ;i<k ;i++)
			if(c[i] >= half) ans = max(ans , v[i]) ;

	}
	cout << ans ; 
	return 0 ;
}
