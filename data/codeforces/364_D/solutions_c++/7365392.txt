#include<cstdio>
#include<algorithm>
#include<cstring>
#define N 2000005
using namespace std;
typedef long long LL;
int f[N],n,i,Test=10;
LL ans,a[N],small[N],big[N];
void check(LL A)
{
  memset(f,0,sizeof(f));
  int S=0,B=0;
  for (LL i=1;i*i<=A;i++)
    if (A%i==0)
    {
      small[++S]=i;
      if (i*i!=A) big[++B]=A/i;
    }
  for (int i=B;i;i--) small[++S]=big[i];
  for (int i=1;i<=n;i++)
    f[lower_bound(small+1,small+S+1,__gcd(A,a[i]))-small]++;
  for (int i=1;i<=S;i++)
  {
    int num=0;
    for (int j=i;j<=S;j++) 
      if (small[j]%small[i]==0) num+=f[j];
    if (ans<small[i]&&num>(n-1)/2) ans=small[i];
  }
}
int main()
{
  scanf("%d",&n);ans=1;
  for (i=1;i<=n;i++)
    scanf("%I64d",&a[i]);
  while (Test--)
    check(a[rand()*1ll*rand()%(LL)n+1]);
  printf("%I64d",ans);
  return 0;
}
