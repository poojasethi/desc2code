#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
#include<map>
#include<cmath>
#include<set>

#define mp make_pair
#define pb push_back

using namespace std;

const int N=1020000;
typedef long long big;
int n;
big a[N],p[N];
int cnt[N];
int ran()
{
	return (rand()<<14)+rand();
}
int main()
{
	int i,cas,x,top=0,l,r,mid,j;
	big tmp;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%I64d",&a[i]);
	big ans=0;
	for(cas=1;cas<=10;cas++)
	{
		x=ran()%n+1;
		top=0;
		for(i=1;1ll*i*i<=a[x];i++)
		{
			if(a[x]%i==0)
			{
				p[++top]=i,cnt[top]=0;
				if(1ll*i*i!=a[x])
					p[++top]=a[x]/i,cnt[top]=0;
			}
		}
		sort(p+1,p+1+top);
		for(i=1;i<=n;i++)
		{
			tmp=__gcd(a[x],a[i]);
			l=1;r=top;
			while(l<=r)
			{
				mid=l+r>>1;
				if(p[mid]>=tmp)r=mid-1;
				else l=mid+1;
			}
			cnt[l]++;
		}
		for(i=1;i<=top;i++)
			for(j=i+1;j<=top;j++)
				if(p[j]%p[i]==0)
					cnt[i]+=cnt[j];
		for(i=1;i<=top;i++)
			if(cnt[i]+cnt[i]>=n)ans=max(p[i],ans);
	}
	cout<<ans;
}
