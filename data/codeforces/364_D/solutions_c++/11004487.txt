#include <iostream>
#include <cstring>
#include <cstdio>
#include <stdlib.h>
#include <algorithm>
#include <time.h>
typedef long long LL;
using namespace std;
LL a[1000005],p[10005],ans,q[10005];
int main()
{
    int n;
    scanf("%d",&n);
    srand(time(NULL));
    for (int i=1; i<=n; i++) scanf("%I64d",&a[i]);
    for (int x=1; x<=10; x++)
    {
        LL y=a[(1ll*rand()*rand()+rand())%n+1];
        int c=0;
        for (LL j=1; j*j<=y; j++)
        {
            if (y%j==0) 
            {
                p[++c]=j;
                if (j*j<y) p[++c]=y/j;
            }
        }
        sort(p+1,p+c+1);
        for (int i=0; i<=c; i++) q[i]=0;
        for (LL i=1; i<=n; i++)
        {
            q[lower_bound(p+1,p+c+1,__gcd(a[i],y))-p]++;
        }
        for (int i=c; i && p[i]>ans; i--)
        {
            int s=0;
            for (int j=i; j<=c; j++) if (p[j]%p[i]==0) s+=q[j];
            if (s*2>=n) ans=p[i];
        }
    } 
    printf("%I64d\n",ans);
    return 0;
}
