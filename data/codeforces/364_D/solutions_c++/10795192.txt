#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
#define rep(i,n) for (int i=0;i<n;++i)
const int N=1000005;
int n,c[N]; ll ans,a[N],d[N];
int main()
{
    scanf("%d",&n); rep(i,n) scanf("%I64d",a+i);
    rep(t,9){
        ll x=a[((rand()<<13)+rand())%n],L=0;
        for (ll i=1;i*i<=x;++i) if (x%i==0){d[L++]=i; if (i*i!=x) d[L++]=x/i;}
        sort(d,d+L); rep(i,L) c[i]=0;
        rep(i,n) ++c[lower_bound(d,d+L,__gcd(a[i],x))-d];
        rep(i,L) rep(j,i) if (d[i]%d[j]==0) c[j]+=c[i];
        rep(i,L) if (c[i]*2>=n) ans=max(ans,d[i]);
    }
    printf("%I64d\n",ans); return 0;
}
