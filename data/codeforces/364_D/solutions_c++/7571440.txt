#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
#define per(i,a,n) for (int i=n-1;i>=a;i--)
#define pb push_back
#define mp make_pair
#define all(x) (x).begin(),(x).end()
#define SZ(x) ((int)(x).size())
typedef vector<int> VI;
typedef long long ll;
typedef pair<int,int> PII;
typedef map<ll,int>::iterator ite;
const int N=1010000;
ll a[N],ans;
vector<ll> d;
int n,cnt[110000];
int main() {
	scanf("%d",&n);
	rep(i,0,n) scanf("%I64d",a+i);
	random_shuffle(a,a+n);
	ans=1;
	rep(rd,0,10) {
		ll x=a[rand()%n];
		d.clear();
		for (ll i=1;i*i<=x;i++) if (x%i==0) {
			d.pb(i);
			if (i*i!=x) d.pb(x/i);
		}
		sort(all(d));
		rep(i,0,SZ(d)) cnt[i]=0;
		rep(i,0,n) cnt[lower_bound(all(d),__gcd(a[i],x))-d.begin()]++;
		rep(i,0,SZ(d)) {
			rep(j,i+1,SZ(d)) if (d[j]%d[i]==0) cnt[i]+=cnt[j];
			if (cnt[i]*2>=n) ans=max(ans,d[i]);
		}
	}
	printf("%I64d\n",ans);
}
