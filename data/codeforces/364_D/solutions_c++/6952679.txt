#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
using namespace std;
typedef long long ll;
ll i,j,k,a[1100001],ans,q[1100001],w[51001][2],tot;
int n;
ll gcd(ll a,ll b){
	ll r;
	do{
		r=a%b;a=b;b=r;
	}while (r);
	return a;
}
bool check(ll x){
	int i,k=0;
	for (i=1;i<=tot;i++)
	 if (w[i][0]%x==0) k+=w[i][1];
	return k>=(n+1)/2;
}
int main(){
	scanf("%d",&n);
	for (i=1;i<=n;i++) scanf("%I64d",&a[i]);
	for (i=ans=1;i<=10;i++){
		k=rand()*rand()%n+1;
		for (j=1;j<=n;j++) q[j]=gcd(a[k],a[j]);
		sort(q+1,q+1+n);
		for (tot=0,j=1;j<=n;j++)
		 if (j==1||q[j]!=q[j-1]) w[++tot][0]=q[j],w[tot][1]=1;else w[tot][1]++;
		for (j=1;j*j<=a[k]&&a[k]/j>ans;j++)
		 if (a[k]%j==0){
			if (check(a[k]/j)){
				ans=a[k]/j;break;
			}
			if (j>ans&&check(j)) ans=j;
		 }
	}
	printf("%I64d\n",ans);
	return 0;
}
