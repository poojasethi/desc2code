//In the name of God
#include <algorithm>
#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <vector>
using namespace std;
const int N = 1e6 + 6;

int n, f[N];
long long ans, a[N];

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	srand(701);
	for (int i = 0; i < 12; i++) {
		long long m = a[rand() % n];
		vector<long long> v;
		for (long long q = 1; q * q <= m; q++)
			if (!(m % q)) {
				v.push_back(q);
				if (q * q != m)
					v.push_back(m / q);
			}
		sort(v.begin(), v.end());
		fill(f, f + v.size(), 0);
		for (int i = 0; i < n; i++)
			f[lower_bound(v.begin(), v.end(), __gcd(m, a[i])) - v.begin()]++;
		for (int i = v.size() - 1; i >= 0 && v[i] > ans; i--) {
			int sum = 0;
			for (int j = i; j < v.size(); j++)
				if (!(v[j] % v[i]))
					sum += f[j];
			if (sum + sum >= n)
				ans = v[i];
		}
	}
	if (ans == 1) {
		long long m = 999999999999ll;
		if (count(a, a + n, m) * 2 >= n)
			ans = m;
	}
	cout << ans << '\n';
	return 0;
}
