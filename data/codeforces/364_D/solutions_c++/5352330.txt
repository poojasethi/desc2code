#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <ctime>
#define ll long long
using namespace std;
int n,t,s;
ll a[1000005],p[7005],q[7005],w=0;
int main( ){
	srand(time(0));
	cin>>n;
	for (int i=1;i<=n;i++)
		cin>>a[i];
	for (int x=1;x<=10;x++){
		ll y=a[(1ll*rand()*rand())%n+1];
		t=0;
		for (ll i=1;i*i<=y;i++)
			if (y%i==0){
				p[++t]=i;
				if (i*i<y) p[++t]=y/i;
			}
		sort(p+1,p+t+1);
		for (int i=1;i<=t;i++) q[i]=0;
		for (int i=1;i<=n;i++)
			q[lower_bound(p+1,p+t+1,__gcd(a[i],y))-p]++;
		for (int i=t;i>0 && p[i]>w;i--){
			s=0;
			for (int j=i;j<=t;j++)
				if (p[j]%p[i]==0)
					s+=q[j];
			if (s*2>=n) w=p[i];
		}
	}
	cout<<w<<endl;
	return 0;
}
