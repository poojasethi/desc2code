#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<ctime>
typedef long long LL;
using namespace std;
LL a[1000005],t[10005],ans,q[10005];
LL gcd(LL a,LL b)
{
	return b ? gcd(b,a%b):a;
}
int main()
{
	int n;
	scanf("%d",&n);
	srand(time(0));
	for(int i=1;i<=n;i++) scanf("%I64d",&a[i]);
	for(int x=0;x<10;x++)
	{
		LL y=a[(1ll*rand()*rand()+rand())%n+1];
		int c=0;
		for(LL j=1;j*j<=y;j++)
		{
			if(y%j==0) 
			{
				t[++c]=j;
				if(j*j<y) t[++c]=y/j;
			}
		}
		sort(t+1,t+c+1);
		for(int i=0;i<=c;i++) q[i]=0;
		for(LL i=1;i<=n;i++)
		{
			q[lower_bound(t+1,t+c+1,gcd(a[i],y))-t]++;
		}
		for(int i=c;i&&t[i]>ans;i--)
		{
			int s=0;
			for(int j=i;j<=c;j++) if(t[j]%t[i]==0) s+=q[j];
			if(s*2>=n) ans=t[i];
		}
	} 
	cout<<ans<<endl;
	return 0;
}
