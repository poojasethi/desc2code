#include<cstdio>
#include<ctime>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long LL;
LL n,ans;
LL a[1000006];
LL p[1000006];
LL q[1000006];
int main(){
    scanf("%I64d",&n);
    srand(time(0));
    for(LL i=1;i<=n;i++)
        scanf("%I64d",&a[i]);
    for(int u=10;u;u--){
        LL x=a[(1ll*rand()*rand()+rand())%n+1];
        int k=0;
        for(LL i=1;i*i<=x;i++)
            if(x%i==0){
                p[++k]=i;
                p[++k]=x/i;
            }
        sort(p+1,p+1+k);
        memset(q,0,sizeof(q));
        for(LL i=1;i<=n;i++)
            q[lower_bound(p+1,p+1+k,__gcd(a[i],x))-p]++;
        for(int i=k;i && p[i]>ans;i--){
            int s=0;
            for(int j=i;j<=k;j++)
                if(p[j]%p[i]==0)
                    s+=q[j];
            if(2*s>=n)
                ans=p[i];
        }
    }
    printf("%I64d",ans);
    return 0;
}
