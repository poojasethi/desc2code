//In The Name Of God;

#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

const ll Maxn = 1e6 + 7;
const ll Max = 5e5 + 7;

int n;
ll a[Maxn];
int cnt[Max];
ll d[Max];
ll ans;
int ted;

void f ( ll m )
{
	ted = 0;
	for ( ll i = 1; i * i <= m; i++ )
	{
		if ( m % i != 0 )
			continue;
		d[ted++] = i;
		if ( i * i != m )
			d[ted++] = m / i;
	}
}

int main()
{
	ios_base::sync_with_stdio ( 0 );
	cin.tie ( 0 ), cout.tie ( 0 );
	cin >> n;
	for ( int i = 0; i < n; i++ )
		cin >> a[i];
	srand ( 727 );
	random_shuffle ( a, a + n );
	for ( int r = 0; r < min ( 10, n ); r++ )
	{
		memset ( cnt, 0, sizeof cnt );
		ll c = a[r];
		f ( c );
		sort ( d, d + ted );
		for ( int i = 0; i < n; i++ )
		{
			ll p = __gcd ( c, a[i] );
			int pos = lower_bound ( d, d + ted, p ) - d;
			cnt[pos]++;
		}
		for ( int i = 0; i < ted; i++ )
			for ( int j = 0; j < i; j++ )
				if ( d[i] % d[j] == 0 )
					cnt[j] += cnt[i];
		for ( int i = 0; i < ted; i++ )
			if ( cnt[i] >= ( n + 1 ) / 2 )
				ans = max ( ans, d[i] );
	}
	cout << ans << endl;
}
