#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <ctime>
#include <iostream>
#include <algorithm>
#include <map>

using namespace std;

const int MaxN = 1000010;

int n;
long long a[MaxN];
long long ans=1;

inline int myrandom(){
    return (int)((rand()*1.0/RAND_MAX)*(n-1))+1;
}
long long gcd(long long a,long long b){
    return b ? gcd(b,a%b) : a;
}
int main()
{
    srand(time(0));
    scanf("%d",&n);
    for(int i=1;i<=n;i++) cin >> a[i];
    random_shuffle(a+1,a+n+1);
    for(int t=0;t<=7;t++){
        map<long long,int> q;
        long long l=a[myrandom()];
        for(long long i=1;i*i<=l;i++) if(l%i==0){
            q[i]=0;
            if(l/i!=i) q[l/i]=0;
        }
        for(int i=1;i<=n;i++) q[gcd(l,a[i])]++;
        for(map<long long,int> :: iterator p=q.begin();p!=q.end();p++){
            int cnt=0;
            for(map<long long,int> :: iterator d=p;d!=q.end() && cnt*2<n;d++)
                if(d->first%p->first==0) cnt+=d->second;
            if(cnt*2>=n) ans=max(ans,p->first);
        }
    }
    cout << ans << endl;
}
