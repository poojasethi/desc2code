#include <iostream>
#include <map>
using namespace std;
typedef long long ll;
map<ll,bool> M;
ll l,r;
bool check(ll x,int k)
{
	if(k%2)
		return true;
	ll tmp=(1ll<<(k/2));
	if(x%tmp==x/tmp)
		return false;
	return true;
}
ll ans(ll x)
{
	if(x==0)
		return 0;
	M.clear();
	for(int i=1;i<=20;i++)
	{
		ll tmp=(1ll<<i)+1;
		for(int j=3;i*j<=60;j++)
		{
			tmp*=(1ll<<i);
			tmp++;
//			cerr << i << " " << j << " " << tmp << endl;
			for(int k=(1ll<<(i-1));k<(1ll<<i);k++)
			{
				if(k*tmp<=x)
				{
//					cerr << i << " " << j << " " << k*tmp << endl;
					if(check(k*tmp,i*j))
					{
						M[k*tmp]=1;
					}
				}
				else
					break;
			}
		}
	}
	ll ansi=M.size();
	for(int i=1;i<=30;i++)
	{
		ll tmp=x/((1ll<<i)+1);
		if(tmp<(1ll<<(i-1)))
			continue;
		ansi+=min(tmp,(1ll<<i)-1)-(1ll<<(i-1))+1;
	}
	return ansi;
}
int main()
{
	cin >> l >> r;
	cout << ans(r)-ans(l-1) << endl;
	return 0;
}
