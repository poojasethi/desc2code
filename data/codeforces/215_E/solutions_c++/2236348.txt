#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
typedef long long LL;
LL l,r;
LL f[100],d[100];
LL calc(LL n)
{
    if(n==0)return 0;
    int L,i,j,k;
    LL res=0;
    for(L=0;(1LL<<L)-1<n;++L);
    memset(d,0,sizeof d);
    fo(i,1,L)
    {
        fo(j,1,i)
        {
            if(i%j!=0)continue;
            if(d[j]==0)d[j]=1;else d[j]=(d[j]<<j)+1;
            f[j]=max(min(n,(1LL<<i)-1)/d[j]-(1LL<<(j-1))+1,0LL);
            fo(k,1,j-1)if(j%k==0)f[j]-=f[k];
            if(j<i)res+=f[j];
        }
    }
    return res;
}
int main()
{
    cin>>l>>r;
    cout<<calc(r)-calc(l-1)<<endl;
    return 0;
}
