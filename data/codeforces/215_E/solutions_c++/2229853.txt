#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long LL;

LL l, r;
int f[70];

LL calc(LL x) {
	if (x==0) return 0;
	int i, j, k, len;
	memset(f, 0, sizeof(f));
	LL tmp = x;
	for (len=0; tmp; tmp>>=1, len++);
	LL res = 0;
	for (i = 2; i <= len; ++i) {
		int w = 1-f[i];
		for (j = i+i; j <= len; j += i)
			f[j] += w;
		for (j = 1; j*i <= len; ++j)  {
			LL hmr = 1LL<<(j-1);
			LL t = 1;
			for (k = 1; k < i; ++k)
				t = (t<<j)+1;
			LL mdk = min((1LL<<j)-1, x/t);
			if (hmr<=mdk) res += (mdk-hmr+1)*w;
		}
	}
	return res;
}

int main() {
	cin>>l>>r;
	cout<<calc(r)-calc(l-1)<<endl;
	return 0;
}
