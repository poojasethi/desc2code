#include<bits/stdc++.h>

using namespace std;
#define CLR(a,val) memset(a, val, sizeof(a))
#define SZ(V) (LL)V.size()
#define ALL(V) V.begin(),V.end()
#define RALL(V) V.rbegin(),V.rend()
#define FORN(i, n) for(LL i = 0; i < n; i++)
#define FORAB(i, a, b) for(LL i = a; i <= b; i++)
#define MOD 1000000007LL
#define PB push_back
#define MP make_pair
#define F first
#define S second

typedef long long LL;
typedef pair<LL,LL> pll;

LL dp[65];

LL get(LL n){
	LL result=0;
	FORAB(c,1,60) for(LL r=2;r*c<=60;r++){
		LL sum=0;
		FORN(i,r) sum+= ( 1LL << (i*c));
		LL high=min( (1LL<<c)-1,n/sum);
		LL low=(1LL<<(c-1));
		if(high>=low) result+= (high-low+1)*(-dp[r]);
	}
	return result;
}

int main()
{
	dp[1]=1;
	FORAB(i,1,61) for(LL j=2*i;j<=61;j+=i) dp[j]-=dp[i];

	LL l,r;
	cin >> l >> r;
	cout << get(r)-get(l-1) << endl;
	return 0;
}
