#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;

int a[100],f[100];
ll L,R;

ll gds(int tal,int k,ll tot){
	ll x=0;
	for (int i=0;i<k;++i){x<<=1;x+=a[tal-i];}
	ll p=x;
	for (int i=tal/k-1;i>=1;--i){p<<=k;p+=x;}
	return x-(1<<(k-1))+1-(p>tot);
}

ll calc(ll tot){
	int tal=0;ll s=tot,ans=0;
	while (s){
		a[++tal]=s%2;s/=2;
	}
	for (int i=2;i<=tal;++i){
		memset(f,0,sizeof(f));
		for (int j=1;j<i;++j){
			if (i%j!=0) continue;
			if (i<tal) f[j]=(1<<(j-1));
			else f[j]=gds(tal,j,tot);
			for (int k=1;k<j;++k)
				if (j%k==0) f[j]-=f[k];
			ans+=f[j];
		} 
	}
	return ans;
}

void init(){
	cin>>L>>R;
	cout<<calc(R)-calc(L-1)<<endl;
}

int main()
{
	init();
	return 0;
}
