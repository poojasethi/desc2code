#include <iostream>
#include <cstdio>
typedef long long ll;
long long f[100],l,r;
int a[100],i,j;
bool check(int x,int y,int z)
{
	for (int i=x-z;i>=y;--i)
		if (a[i]!=a[i+z]) return 0;
	return 1;
}
ll work(ll x)
{
	int i,j,k,t=0,u[100],g[100];
	for (;x;x>>=1) a[++t]=x&1;
	long long ans=0;
	for (i=1;i<t;++i) ans+=f[i];
	for (i=t-1;i;--i)
	if (a[i])
	{
		--a[i];
		for (j=1;j<=t;++j) u[j]=1,g[j]=0;
		for (j=1;j<=t-i+1 && j<t;++j)
		if (t%j==0 && u[j])
		{
			if (check(t,i,j))
			{
				++ans;g[j]=1;
				for (k=1;j*k<=t;++k) u[j*k]=0;
			}
		}
		for (j=t-i+2;j<t;++j)
		if (t%j==0)
		{
			g[j]=1<<(j-(t-i+1));
			for (k=1;k<j;++k)
				if (j%k==0) g[j]-=g[k];
			ans+=g[j];
		}
		++a[i];
	}
	return ans;
}
int main()
{
	scanf("%I64d%I64d\n",&l,&r);
	for (i=2;i<=60;++i)
		for (j=1;j<i;++j)
		if (i%j==0) f[i]+=(1<<(j-1))-f[j];
	printf("%I64d\n",work(r+1)-work(l));
}
