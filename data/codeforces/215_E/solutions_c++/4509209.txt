using namespace std;
int m[61];
int main() {
    long long ans=0, l, r, a, b, x, L, R;
    scanf("%I64d %I64d", &l, &r);
    --l;
    m[1]=1;
    for(int i=2; i<=61; i++) {
        int x=0;
        for(int j=1; j<=i; j++)
            if(i%j==0) x+=m[j];
        m[i] = -x;// cout<<m[i]<<' ';
    }//cout<<endl;
    for(int i=1; i<=31; i++) {
        x=((1LL)<<i)+1;
        a=((1LL)<<i)-1;
        b=((1LL)<<(i-1))-1;
        for(int j=2; j*i<=62; ++j, ++x) {
            L=min(a, l/x)-b,  R=min(a, r/x)-b;
            ans-=m[j]*(max(R, 0LL)-max(L, 0LL));
            x<<=i;
        }
    }
    printf("%I64d\n", ans);
}
