#include <cstdio>
#include <iostream>
using namespace std;

typedef long long LL;

int dn[2500];

int f(LL s,int n,int k,LL lim){
	LL tmp=0LL;
	for(int i=0;i<n;i+=k){
		tmp|=s;
		s<<=k;
	}
	return tmp<=lim;
}

LL calc(int n,int k,LL lim){
	
	LL lo=1LL<<(k-1),hi=(1LL<<k)-1LL,mi;
	while(lo<hi){
		mi=(lo+hi)/2+1;
		if(f(mi,n,k,lim))
			lo=mi;
		else
			hi=mi-1;
	}
	if(f(lo,n,k,lim))
		return lo-(1LL<<(k-1))+1;
	else
		return 0;
}

int main(){
	
	LL A,B;
	
	cin >> A >> B;
	
	int res=0;
	
	for(int i=1,a;1LL<<(i-1)<=B;i++){
		if(1LL<<i<A)	continue;
		for(int j=1;j<i;j++)
		{
			if(i%j==0)
			{
				a=calc(i,j,B)-calc(i,j,A-1);
				for(int t=1;t<j;t++)
					if(j%t==0)
						a-=dn[t];
				dn[j]=a;
				res+=a;
				//printf("		%d,%d => %d\n",i,j,a);
			}
		}
	}
	
	//for(int i=1;i<=10;i++)
	//	printf("%d ",dn[i]);
	//puts("");
	
	cout << res << endl;
	
	return 0;
	
}
