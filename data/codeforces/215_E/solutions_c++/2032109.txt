#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long ll;
const int maxn=64;
int f[maxn];
ll l,r;

ll solve(ll x){
  if(!x) return 0;
  ll l,r,c,s=0;
  int i,j,k,p;
  memset(f,0,sizeof(f));
  for (i=2;i<maxn;i++){
    p=1-f[i];
    for (j=i;j<maxn;j+=i) f[j]+=p;
    for (j=1;j<maxn;j++){
      l=1LL<<(j-1);
      for (c=k=1;k<i;k++) c=(c<<j)+1;
      r=min((1LL<<j)-1,x/c);
      if (l<=r) s+=(r-l+1)*p;
    }
  }
  return s;
}

int main(){
  scanf("%I64d%I64d",&l,&r);
  printf("%I64d\n",solve(r)-solve(l-1));
  return 0;
}
