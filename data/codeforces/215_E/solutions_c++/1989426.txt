#include<iostream>
#include<cstdio>
using namespace std;
long long counting(long long x)
{
	static long long ans[70];
	long long ret = 0;
	for (int i=1; 1LL<<(i-1)<=x; i++)
		for (int j=1; j<i; j++)
		{
			ans[j] = 0;
			if (i % j) continue;
			if (1LL << i <= x) ans[j] = 1LL << (j-1);
			else
			{
				ans[j] = (x >> i-j) - (1 << j-1);
				long long temp = 0, now = x >> (i-j);
				for (int k=0; k<i; k+=j) temp = (temp << j) | now;
				ans[j] += temp <= x;
			}
			for (int k=1; k<j; k++)
				if (j % k == 0) ans[j] -= ans[k];
			ret += ans[j];
		}
	return ret;
}
int main()
{
	long long l, r;
	cin >> l >> r;
	cout << counting(r) - counting(l-1) << endl;
}
