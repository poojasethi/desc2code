typedef long long int ll  ;

ll pot2[61];

ll period(ll r,ll l)
{  ll ans=0; 
  
    if (r<=0) return period(1,l);
  for(int t=1;t<=60;t++) 
  { ll a=pot2[t]; ; 
   
    ll maxi=60/t ; 
    for(ll k=1;k<=maxi;k++) 
    { ll b=pot2[t*k];
      ll m=b+((b-1)/(a-1));
      
      ll c1;
      if (r%m==0) c1=r/m;
      else c1= (r/m)+1;
      if ((a/2)>c1) c1=a/2;
     
      ll c2=l/m;
      if (a-1<c2) c2=a-1; 
    
      if (c1<=c2)
      ans=ans+(c2-c1+1-period(c1,c2));}
      
    }
  
  return ans; 

}
