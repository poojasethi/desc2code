#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;

#define MAX(a,b) ((a)>(b) ? (a):(b))
#define MIN(a,b) ((a)<(b) ? (a):(b))

__int64 dp[100];
int s[100];

__int64 sol(int len,int j,__int64 m)
{
	__int64 y,x=0;
	for(int i=0;i<j;i++) x<<=1,x+=s[len-i];
	y=x;
	for(int i=1;i<len/j;i++) y<<=j,y+=x;
	return x-(1<<(j-1))+1-(y>m);
}

__int64 cal(__int64 x)
{
	__int64 ans=0,t=x;
	int pos=0;
	while(x)
	{
		s[++pos]=x%2;
		x/=2;
	}
	for(int i=2;i<=pos;i++)
	{
		memset(dp,0,sizeof(dp));
		for(int j=1;j<i;j++)
		{
			if(i%j) continue;
			if(i<pos) dp[j]=1<<(j-1);
			else dp[j]=sol(pos,j,t);
			for(int k=1;k<j;k++)
			if(j%k==0) dp[j]-=dp[k];
			ans+=dp[j];
		}
	}
	return ans;
}

int main()
{
	__int64 m,n;
	while(scanf("%I64d%I64d",&m,&n)!=EOF)
	{
		printf("%I64d\n",cal(n)-cal(m-1));
	}
	return 0;
}
