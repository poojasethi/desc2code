#include <cmath>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#define FOR(i,l,r) for(i=l;i<=r;i++)
#define COR(i,r,l) for(i=r;i>=l;i--)
using namespace std;

typedef long long LL;

LL L,R;

void init(){
	cin>>L>>R;
}

LL getlen(LL s){
	LL res=0;
	while(s) s>>=1,res++;
	return res;
}

LL calc(LL s,LL len){
	if(s==0) return 0;
	LL i,j,res=0;
	FOR(i,1,len-1){
		if(len%i!=0) continue;
		LL tmp=s>>(len-i),s2=tmp;
		COR(j,len/i,2) s2=(s2<<i)+tmp;
		if(s2>s){
			tmp--;
			if(getlen(tmp)!=i) continue;			
		}
		res+=tmp-(1LL<<(i-1))+1-calc(tmp,i);
	}
	return res;
}

LL calc2(LL s){
	LL tmp=1,res=0;
	while(tmp<=s){
		res+=calc(tmp-1,getlen(tmp-1));
		tmp<<=1;
	}
	res+=calc(s,getlen(s));
	return res;
}

void work(){
	cout<<calc2(R)-calc2(L-1)<<endl;
}

int main(){
	init();
	work();
	return 0;
}
