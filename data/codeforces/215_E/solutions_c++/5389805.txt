#include <stdio.h>
#define LI long long int 

int flag[100];
LI pow[100];

LI sex(LI t){
	LI ret=0, bnd1, bnd2, c;
	int i, j, k, u;

	if (t==0) return 0;
	pow[0]=1, pow[1]=2;
	for (i=2;i<63;i++){
		pow[i]=pow[i-1]*2;
		flag[i]=0;
	}

	for (i=2;i<64;i++){
		u=1-flag[i];
		for (j=i;j<64;j+=i){
			flag[j]+=u;
		}

		for (j=1;j<64;j++){
			bnd1=pow[j-1];
			c=1;
			for (k=1;k<i;k++){
				c=c*pow[j] +1;
			}
			bnd2=(pow[j]-1 > t/c) ? t/c : pow[j]-1;
			if (bnd1<=bnd2) ret+=(bnd2-bnd1+1)*u;
		}
	}

	return ret;
}

int main(void){
	LI f, l;
	scanf("%I64d %I64d", &f, &l);
	printf("%I64d\n", sex(l)-sex(f-1));
	return false;
}
