#include <iostream>

#include <algorithm>
#include <vector>
#include <set>
#include <cstring>
using namespace std;
#define mp make_pair
#define lol long long
const int MAXL=62;
lol dp[MAXL+10],fp[MAXL+10];

int len(lol x)
{
  for(int i=MAXL;i>=0;i--)
    if(x&(1ll<<i))
      return i+1;
}

lol mer(lol p,int t)
{
  lol ret=0,l=len(p);
  for(int i=1;i<=t;i++)
    ret<<=l,ret+=p;
  return ret;
}

lol f(lol x)
{
  memset(dp,0,sizeof dp);
  if(x==0)
    return 0;
  int l=len(x);
  lol ret=0;
  for(int i=1;i<l;i++)
    {
      if(l%i!=0)
	continue;
      lol p=x,ans=0;
      p>>=(l-i);
      ans+=p-(1ll<<(i-1));
      if(mer(p,l/i)<=x)
	ans++;
      dp[i]+=ans;
    }
   for(int i=1;i<l;i++)
    for(int j=i+i;j<=l;j+=i)
      dp[j]-=dp[i];
  for(int i=1;i<l;i++)
    if(l%i==0)
    ret+=dp[i];
  return ret+f((1ll<<(l-1))-1ll);
}

int main()
{
  ios::sync_with_stdio(false);
  lol l,r;
  cin>>l>>r;
  lol ans1=f(r),ans2=f(l-1);
  // cerr<<ans2<<" "<<ans1<<endl;
  cout<<ans1-ans2<<endl;
  return 0;
}
