using namespace std;
int mobius[61];
int main() {
    long long l, r;
    scanf("%I64d %I64d", &l, &r);
    mobius[1]=1;
    for(int i=2; i<=61; i++) {
        int x=0;
        for(int j=1; j<=i; j++)
            if(i%j==0) x+=mobius[j];
        mobius[i] = -x;// cout<<mobius[i]<<' ';
    }//cout<<endl;
    long long ans=0, a, b, x;
    for(int i=1; i<=31; i++) {
        x=((1LL)<<i)+1;
        a=((1LL)<<i)-1;
        b=((1LL)<<(i-1))-1;
        for(int j=2; j*i<=62; j++) {
            long long R=r/x;
            long long L=(l-1)/x;
            ans-=mobius[j]*(max(min(R,a)-b,0LL)-max(min(L,a)-b,0LL));
            x<<=i;
            x+=1;
        }
    }
    printf("%I64d\n", ans);
}
