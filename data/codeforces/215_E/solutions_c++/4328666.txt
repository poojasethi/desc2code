#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef long long ll;
ll l,r,ans;
ll sig[110];
ll cal(ll N){
    ll ans=0;
    for (int cir=1;cir<=60;cir++)
        for (int rep=2;(cir*rep)<=60;rep++){
            ll sum=0;
            for (int i=0;i<rep;i++)
                sum+=1ll<<(i*cir);
            ll p=1ll<<cir;
            ll high=min(p-1,N/sum);
            ll low=p/2;
            if (high>=low)
            ans+=(high-low+1)*-sig[rep];
        }
    return ans;
}

int main(){
    //freopen("input.txt","r",stdin);
    sig[1]=1ll;
    for (int i=1;i<=100;i++)
        for (int j=i+i;j<=100;j+=i)
            sig[j]-=sig[i];
    cin>>l>>r;
    ans=cal(r)-cal(l-1);
    cout<<ans<<endl;
}
