#include<iostream>
using namespace std;
typedef long long ll;

ll mul(ll x,int per,int k){
     ll ret=0;
     while(k--)
	  ret<<=(ll)per, ret+=x;
     return ret;
}
ll get(ll x){
     if(x<=1LL) return 0;
     int len=0; for(ll y=x;y;) ++len,y>>=1;
     ll sum=0;
     for(int per=1;per<=len-1;per++)
	  for(int k=2;k<=len;k++)
	       if(k*per<=len){
		    ll l=0,r=(1LL<<per)-1,t=0;
		    while(l<=r){
			 ll m=(l+r)>>1;
			 if(mul(m,per,k)<=x) t=m,l=m+1;
			 else r=m-1;
		    }
		    ll Add=t-get(t),tmp=(1LL<<(per-1))-1;
		    if(per>1){
			 if(t<=tmp) continue;
			 Add+=get(tmp)-tmp;
		    }
		    sum+=Add;
	       }
     return sum;
}
int main(){
     ll l,r;
     cin>>l>>r;
     cout<<get(r)-get(l-1)<<endl;
}
