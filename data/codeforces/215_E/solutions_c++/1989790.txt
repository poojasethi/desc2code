#include <cstdio>
#include <string>
#include <vector>
#include <cstring>
#include <algorithm>
using namespace std;

long long ma,mb;

int cb(long long x) {
	int ret = 0;
	while (x > 0) {
		x /= 2;
		ret++;
	}
	return ret;
}

long long doit(long long x) {
	if (x == 0) return 0;
	if (x == 1) return 0;
	int b = 0;
	long long tx,ti,ret = 0;
	b = cb(x);
	for (int i=1; i<=b/2; i++)
		if (b % i == 0) {
			tx = x;
			ti = (tx >> (b-i));
			tx = ti;
			for (int j=1; j<b/i; j++)
			    tx = (tx << i) + ti;
   			if (x < tx) {
				ti--;
				if (cb(ti) != i) continue;
			}
			ret += (ti-(1LL<<(i-1))+1) - doit(ti);
		}
	//printf("ans %I64d = %I64d\n",x,ret);
	return ret;
}

long long calc(long long x) {
	long long cur = 1,ret = 0;
	while (cur <= x) {
		ret += doit(cur-1);
		cur *= 2;
	}
	ret += doit(x);
	return ret;
}

int main() {
	scanf("%I64d%I64d",&ma,&mb);
	printf("%I64d\n",calc(mb)-calc(ma-1));
}
