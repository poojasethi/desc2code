//In the Name of God

#include<iostream>
#include<cstring>
#define int long long

using namespace std;

int s,e;
int dp[128];

int len(int x)
{
  int ret=0;
  for(;x;x/=2,ret++);
  return ret;
}

int c(int x,int l)
{
  int L=len(x);
  int ret=0;
  for(int i=0;i<l/L;i++)
    ret=ret*(1ll<<L)+x;
  return ret;
}

int f(int x)
{
  int L=len(x);
  memset(dp,0,sizeof dp);
  for(int i=1;i<L;i++)
    if(L%i==0)
      {
        dp[i]+=(x/(1ll<<(L-i)))%(1ll<<(i-1));
        if(c(x/(1ll<<(L-i)),L)<=x)
	  dp[i]++;
        for(int j=i+i;j<=L;j+=i)
	  dp[j]-=dp[i];
      }
  return -dp[L];
}

int ans(int x)
{
  if(!x)
    return 0;
  int ret=0;
  int L=len(x);
  ret+=f(x);
  for(int i=1;i<L;i++)
    ret+=f((1ll<<i)-1);
  return ret;
}

main()
{
  ios_base::sync_with_stdio(0);
  cin>>s>>e;
  e=ans(e);
  s=ans(s-1);
  cout<<e-s<<endl;
  return 0;
}
