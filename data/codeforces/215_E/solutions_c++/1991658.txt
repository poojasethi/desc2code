#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int Miu(int N)
{
  int Ans=-1;
  for(int i=2;i*i<=N;i++)
    if(!(N%i))
      {
        Ans=-Ans;N/=i;
        if(!(N%i))return 0;
      }
  if(N>1)Ans=-Ans;
  return Ans;
}

long long Repeat(int M,int Len,long long N)
{
  if(N>=(1LL<<M))return 1LL<<(Len-1);
  static bool A[100];
  for(int i=M-1;i>=0;i--){A[i]=N%2;N/=2;}
  long long S=0;
  for(int i=1;i<Len;i++)S=S*2+A[i];
  for(int i=Len;i<M;i++)
    if(A[i]^A[i%Len])return S+A[i];
  return S+1;
}

long long Count(int M,long long N)
{
  long long Ans=0;
  for(int i=2;i<=M;i++)
    if(!(M%i))Ans+=Repeat(M,M/i,N)*Miu(i);
  return Ans;
}

long long Calc(long long N)
{
  if(!N)return 0;
  int P=0;long long Ans=0;
  while((1LL<<P)<=N)Ans+=Count(++P,N);
  return Ans;
}

int main()
{
  long long LB,UB;
  cin>>LB>>UB;
  cout<<Calc(UB)-Calc(LB-1)<<endl;
  return 0;
}
