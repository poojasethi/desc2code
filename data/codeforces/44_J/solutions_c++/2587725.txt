#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
char a[1010][1010],k;
int n,m,i,j,v[5],x,y,dx[4]={-1,0,0,1},dy[4]={0,-1,1,0};

void calc(int i,int j)
{
	for(k=0;k<4;k++)
		if(a[x=i+dx[k]][y=j+dy[k]]>='0'&&a[x][y]<='4') v[a[x][y]-'0']=1;
}

int main()
{
	cin>>n>>m;
	for(i=1;i<=n;i++) scanf("%s",a[i]+1);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			if(a[i][j]=='w')
			{
				if(a[i][j+1]=='b'&&a[i][j+2]=='w')
				{
					memset(v,0,sizeof(v));
					calc(i,j),calc(i,j+1),calc(i,j+2);
					for(k=0;k<5&&v[k];k++);
					a[i][j]=a[i][j+1]=a[i][j+2]=k+'0';
				}
				else if(a[i+1][j]=='b'&&a[i+2][j]=='w')
				{
					memset(v,0,sizeof(v));
					calc(i,j),calc(i+1,j),calc(i+2,j);
					for(k=0;k<5&&v[k];k++);
					a[i][j]=a[i+1][j]=a[i+2][j]=k+'0';
				}
				else {puts("NO"); return 0;}
			}
			else if(a[i][j]=='b') {puts("NO"); return 0;}
	puts("YES");
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++) putchar(a[i][j]=='.'?'.':a[i][j]-'0'+'a');
		puts("");
	}
	return 0;
}
