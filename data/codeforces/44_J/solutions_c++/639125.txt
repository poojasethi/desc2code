#define li		 long long int

#define pb		push_back
#define sz(v)		((li)v.size())


int main(){
	li n,m;
	cin>>n>>m;
	vector<string> mp(n);
	rep(i,n) cin>>mp[i];
	rep(i,n)rep(j,m)if(mp[i][j]=='b'){
		if(0<=i-1 && i+1<n && mp[i-1][j]=='w' && mp[i+1][j]=='w'){
			li num=(i%2)*2+(i/2+j/2)%2;
			mp[i-1][j]='A'+num; mp[i][j]='A'+num; mp[i+1][j]='A'+num;
		}else if(0<=j-1 && j+1<m && mp[i][j-1]=='w' && mp[i][j+1]=='w'){
			li num=(i%2)*2+(i/2+j/2)%2;
			mp[i][j-1]='A'+num; mp[i][j]='A'+num; mp[i][j+1]='A'+num;
		}
	}
	rep(i,n)rep(j,m)if(mp[i][j]=='w' || mp[i][j]=='b'){
		cout<<"NO"<<endl;
		return 0;
	}
	rep(i,n)rep(j,m)if('A'<=mp[i][j] && mp[i][j]<='D') mp[i][j]=mp[i][j]-'A'+'a';
	cout<<"YES"<<endl;
	rep(i,n) cout<<mp[i]<<endl;
}
