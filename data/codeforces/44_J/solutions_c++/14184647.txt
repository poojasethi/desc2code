#include<bits/stdc++.h>
#define MOD 1000000007
#define MX 100010
#define ll long long
#define sc(n) scanf("%d",&n)
#define pr(m) printf("%d\n",m)
#define pi acos(-1.0)

using namespace std;

char grid[1002][1002];
int change(int i,int j, char c) {
	if(grid[i][j] == '.') {
		return 1;
	}

	if(grid[i][j] == 'b') {
		return 0;
	}

	if(grid[i][j] == 'w') {
		if(grid[i][j+1] == 'b')
			if(grid[i][j+2] == 'w')
				grid[i][j] = grid[i][j+1] = grid[i][j+2] = c;
			else
				return 0;
		else
			if(grid[i+1][j] == 'b')
				if(grid[i+2][j] == 'w')
					grid[i][j] = grid[i+1][j] = grid[i+2][j] = c;
				else
					return 0;
			else
				return 0;

	}

	return 1;
}

int main()
{
    int n,m;

	scanf("%d %d",&n,&m);
	char c;
	char s[10];
	int i,j;
	long long w=0,b=0;
	gets(s);
	for(i=1;i<n+1;i++) {
		for(j=1;j<m+1;j++) {
			scanf("%c",&grid[i][j]);
			switch (grid[i][j]) {
			case '.':  break;
			case 'w':  w++; break;
			case 'b':  b++; break;
			}
		}
		gets(s);
	}

	if(w*1.0/2 != b) {
		printf("NO");
		return 0;
	}
	char color = 'a';
	bool flag = 1;
	for(i=1;(i<n+1)&&flag;i++) {
		for(j=1;(j<m+1)&&flag;j++) {
			if(!change(i,j,color)) {
				flag = 0;
				break;
			}

		}

		color = (color == 'a')? 'c': 'a';
	}
	if(!flag)
		printf("NO");
	else {
		printf("YES\n");
		for(i=1;(i<n+1)&&flag;i++) {
			for(j=1;(j<m+1)&&flag;j++) {
				printf("%c",grid[i][j]);
			}
			printf("\n");
		}
	}

    return 0;
}
