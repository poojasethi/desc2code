#include<cstdio>
#define fo(i,a,b) for(i=a;i<=b;++i)
const int mn=1100;
int n,m,i,j,tt;
char g[mn][mn],ans[mn][mn],p;
bool sol;
int main()
{
    scanf("%d%d",&n,&m);
    fo(i,1,n)scanf("%s",g[i]+1);
    sol=1;
    fo(i,1,n)
        fo(j,1,m)
            ans[i][j]='.';
    fo(i,1,n)
    {
        fo(j,1,m)
        {
            if(g[i][j]!='b')continue;
            p='a'+2*((i/2+j/2)%2)+i%2;
            if(g[i-1][j]==g[i+1][j]&&g[i-1][j]=='w')
            {
                ans[i-1][j]=ans[i][j]=ans[i+1][j]=p;
                g[i-1][j]=g[i][j]=g[i+1][j]='.';
            }
            else if(g[i][j-1]==g[i][j+1]&&g[i][j-1]=='w')
            {
                ans[i][j-1]=ans[i][j]=ans[i][j+1]=p;
                g[i][j-1]=g[i][j]=g[i][j+1]='.';
            }
        }
    }
    fo(i,1,n)
        fo(j,1,m)
            if(g[i][j]!='.')sol=0;
    if(!sol){puts("NO");return 0;}
    puts("YES");
    fo(i,1,n)printf("%s\n",ans[i]+1);
    return 0;
}
