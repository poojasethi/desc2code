#include<cstdio>
char s[1111][1111],t[1111][1111];
bool p[9];
int main(){
//  freopen("44j.in","r",stdin);
//  freopen("44j.out","w",stdout);
    int n,m,i,j,k,cnt;
    bool pd;
    scanf("%d%d",&n,&m);
    for (i=0;i<n;i++){
        scanf("%s",s[i]);
        for (j=0;j<=m;j++) t[i][j]=0;
    }
    cnt=0,pd=false;
    for (i=0;i<n;i++){
        for (j=0;j<m;j++) if (s[i][j]=='b'){
            if (j && j!=m-1 && s[i][j-1]=='w' && s[i][j+1]=='w'){
                s[i][j-1]=s[i][j]=s[i][j+1]='.';
                p[1]=p[2]=p[3]=p[4]=true;
                if (j-1>0) p[t[i][j-2]]=false;
                if (i>0) p[t[i-1][j-1]]=p[t[i-1][j]]=p[t[i-1][j+1]]=false;
                if (j+1<m-1) p[t[i][j+2]]=false;
                for (k=1;!p[k];k++);
                t[i][j-1]=t[i][j]=t[i][j+1]=k;
            } else pd=true;
        }
        for (j=0;j<m;j++) if (s[i][j]=='w'){
            if (i<n-2 && s[i+1][j]=='b' && s[i+2][j]=='w'){
                s[i][j]=s[i+1][j]=s[i+2][j]='.';
                p[1]=p[2]=p[3]=p[4]=true;
                if (i>0) p[t[i-1][j]]=false;
                if (j>0) p[t[i][j-1]]=p[t[i+1][j-1]]=p[t[i+2][j-1]]=false;
                if (j<m-1) p[t[i][j+1]]=p[t[i+1][j+1]]=p[t[i+2][j+1]]=false;
                for (k=1;!p[k];k++);
                t[i][j]=t[i+1][j]=t[i+2][j]=k;
            } else pd=true;
        }
    }
    if (pd) {puts("NO"); return 0;}
    puts("YES");
    for (i=0;i<n;i++) for (j=0;j<m;j++)
        if (t[i][j]) t[i][j]+='a'-1; else t[i][j]='.';
    for (i=0;i<n;i++) puts(t[i]);
    return 0;
}
