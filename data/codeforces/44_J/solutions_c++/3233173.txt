#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <queue>
#include <cmath>
#include <map>
#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;

int now,flag[5],yan[3000*3000],use[3000][3000],n,m,i,j,k;
char a[3000][3000];

char ch(int i,int j)
{
    if (i<1||i>n||j<1||j>m||use[i][j]>0||a[i][j-1]=='.') {printf("NO");exit(0);}
    return a[i][j-1];
}

void ran(int i,int j)
{
    if (i<1||i>n||j<1||j>m||use[i][j]==0) return;
    flag[yan[use[i][j]]]++;
}

int main()
{
   scanf("%d%d",&n,&m);
   for (i=1;i<=n;i++) scanf("%s",&a[i]);
   for (j=1;j<=m;j++)
   {
       for (i=1;i<=n;i++) if (a[i][j-1]=='b'&&use[i][j]==0)
       {
           if (ch(i-1,j)=='w'&&ch(i+1,j)=='w')
           {
               now++;use[i][j]=now;use[i-1][j]=now;use[i+1][j]=now;
               memset(flag,0,sizeof(flag));
               ran(i,j-1);ran(i-1,j-1);ran(i+1,j-1);ran(i-2,j);ran(i+2,j);
               for (k=4;k>=1;k--) if (flag[k]==0) yan[now]=k;
           }
       }
       for (i=1;i<=n;i++) if (a[i][j-1]=='w'&&use[i][j]==0)
       {
           if (ch(i,j+1)=='b'&&ch(i,j+2)=='w')
           {
               now++;use[i][j]=now;use[i][j+1]=now;use[i][j+2]=now;
               memset(flag,0,sizeof(flag));
               ran(i,j-1);ran(i-1,j);ran(i+1,j);
               for (k=4;k>=1;k--) if (flag[k]==0) yan[now]=k;
           }
       }
   }
   printf("YES\n");
   for (i=1;i<=n;i++) {
   {for (j=1;j<=m;j++) if (a[i][j-1]=='.') printf(".");else printf("%c",yan[use[i][j]]+'a'-1);}printf("\n");}
}
