using namespace std;
char s[1007][1007];
int main(){
	std::ios::sync_with_stdio(false);
	int i,j,k,m,n;
	cin >> n >> m;
	for (i=1;i<=n;++i)
		cin >> (s[i]+1);
	for (i=1;i<=n;++i)
		for (j=1;j<=m;++j){
			if (s[i][j]=='b'){
				cout << "NO" << endl;
				return 0;
			}
			if (s[i][j]=='w'){
				if (s[i][j+1]=='b' && s[i][j+2]=='w'){
					for (k='A';k<='D';++k)
						if (s[i][j-1]!=k && s[i-1][j]!=k && s[i-1][j+1]!=k && s[i-1][j+2]!=k && s[i][j+3]!=k) break;
					s[i][j]=s[i][j+1]=s[i][j+2]=k;
				}
				else if (s[i+1][j]=='b' && s[i+2][j]=='w'){
					for (k='A';k<='D';++k)
						if (s[i-1][j]!=k && s[i][j-1]!=k && s[i+1][j-1]!=k && s[i][j+1]!=k && s[i+1][j+1]!=k) break;
					s[i][j]=s[i+1][j]=s[i+2][j]=k;
				}
				else{
					cout << "NO" << endl;
					return 0;
				}
			}
		}
	cout << "YES" << endl;
	for (i=1;i<=n;++i)
		for (j=1;j<=m;++j)
			if (s[i][j]!='.') s[i][j]+=32;
	for (i=1;i<=n;++i)
		cout << (s[i]+1) << endl;
	return 0;
}
