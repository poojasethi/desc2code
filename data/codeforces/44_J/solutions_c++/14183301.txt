#include <bits/stdc++.h>

using namespace std;

int main ()
{
    int n, m, i, j, k;
    cin >> n >> m;
    char a[n][m+1], b[n][m];
    for (i=0; i<n; i++) cin >> a[i];
    for (i=0; i<n; i++)
    {
        for (j=0; j<m; j++)
        {
            b[i][j]='.';
        }
    }
    for (i=0; i<n; i++)
    {
        for (j=0; j<m; j++)
        {
            if (a[i][j]=='w')
            {
                char ch = 'a' + ((2*(i%2+j%2)&1)+i%2);
                if (a[i][j+1]=='b' && a[i][j+2]=='w')
                {
                    a[i][j]='.';
                    a[i][j+1]='.';
                    a[i][j+2]='.';
                    b[i][j]=ch;
                    b[i][j+1]=ch;
                    b[i][j+2]=ch;
                }
                else if (a[i+1][j]=='b' && a[i+2][j]=='w')
                {
                    a[i][j]='.';
                    a[i+1][j]='.';
                    a[i+2][j]='.';
                    b[i][j]=ch;
                    b[i+1][j]=ch;
                    b[i+2][j]=ch;
                }
            }
        }
    }
    for (i=0; i<n; i++)
    {
        for (j=0; j<m; j++)
        {
            if (a[i][j]!='.')
            {
                printf("NO");
                return 0;
            }
        }
    }
    printf("YES\n");
    for (i=0; i<n; i++)
    {
        for (j=0; j<m; j++)
        {
            printf("%c", b[i][j]);
        }
        printf("\n");
    }
    return 0;
}
