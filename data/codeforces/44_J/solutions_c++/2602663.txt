#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int n,m;
char a[1010][1010],b[1010][1010];
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
        scanf("%s",a[i]+1);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
            b[i][j]='.';
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++){
			if(a[i][j]!='b')continue;
			int c='a'+2*((i/2+j/2)%2)+i%2;
			if(a[i-1][j]=='w'&&a[i+1][j]=='w'){
				b[i-1][j]=b[i][j]=b[i+1][j]=c;
				a[i-1][j]=a[i][j]=a[i+1][j]='.';
            }
			else
            if(a[i][j-1]=='w'&&a[i][j+1]=='w'){
				b[i][j-1]=b[i][j]=b[i][j+1]=c;
				a[i][j-1]=a[i][j]=a[i][j+1]='.';
            }
		}
	bool f=1;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
            if(a[i][j]!='.')f=0;
	puts(f?"YES":"NO");
	if(f)
        for(int i=1;i<=n;i++){
    		for(int j=1;j<=m;j++)putchar(b[i][j]);
    		putchar('\n');
    	}
	return 0;
}
