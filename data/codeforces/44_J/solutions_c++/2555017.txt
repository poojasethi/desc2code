#include <cstdio>
#include <cstring>
int n,m,i,j,k,z[1010][1010],p[5],ans[1010][1010];
char ch;
int main()
{
	scanf("%d%d",&n,&m);getchar();
	for (i=1;i<=n;++i)
	{
		for (j=1;j<=m;++j)
		{
			ch=getchar();
			if (ch=='.') z[i][j]=-1;
			if (ch=='w') z[i][j]=0;
			if (ch=='b') z[i][j]=1;
		}
		getchar();
	}
	for (i=1;i<=n;++i)
	{
		for (j=1;j<=m;++j)
		{
			if (z[i][j]==1) {printf("NO\n");return 0;}
			if (z[i][j]==0)
			{
				if ((z[i][j+1]==1)&&(z[i][j+2]==0))
				{
					memset(p,0,sizeof(p));p[ans[i-1][j]]=true;p[ans[i][j-1]]=true;p[ans[i-1][j+1]]=true;p[ans[i-1][j+2]]=true;
					p[ans[i][j+3]]=true;p[ans[i+1][j]]=true;p[ans[i+1][j+1]]=true;p[ans[i+1][j+2]]=true;
					for (k=1;k<=4;++k) if (!p[k]) break;
					ans[i][j]=k;ans[i][j+1]=k;ans[i][j+2]=k;
					z[i][j]=-1;z[i][j+1]=-1;z[i][j+2]=-1;continue;
				}
				if ((z[i+1][j]==1)&&(z[i+2][j]==0))
				{
					memset(p,0,sizeof(p));p[ans[i-1][j]]=true;p[ans[i][j-1]]=true;p[ans[i+1][j-1]]=true;p[ans[i+2][j-1]]=true;
					p[ans[i][j+1]]=true;p[ans[i+1][j+1]]=true;p[ans[i+2][j+1]]=true;
					for (k=1;k<=4;++k) if (!p[k]) break;
					ans[i][j]=k;ans[i+1][j]=k;ans[i+2][j]=k;
					z[i][j]=-1;z[i+1][j]=-1;z[i+2][j]=-1;continue;
				}
				{printf("NO\n");return 0;}
			}
		}
	}
	printf("YES\n");
	for (i=1;i<=n;++i)
	{
		for (j=1;j<=m;++j) if (ans[i][j]==0) printf(".");else
		{
			ch=ans[i][j]+96;printf("%c",ch);
		}
		printf("\n");
	}
}
