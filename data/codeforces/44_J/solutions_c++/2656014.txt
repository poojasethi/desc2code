#include<cstdio>
#include<cstring>

#define fail(X) if(X)return puts("NO"),0;
#define v(i,j) _v[c[i][j]]

const int maxn=1010;
int n,m;
char g[maxn][maxn];
int c[maxn][maxn],_v[6];

int main(){
	int i,j,k;
	char ch;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)for(j=1;j<=m;++j){
		while(ch=getchar(),ch!='.'&&ch!='w'&&ch!='b');
		g[i][j]=ch;
	}
	for(i=1;i<=n;++i)for(j=1;j<=m;++j)
		if(g[i][j]!='.'&&c[i][j]==0){
			fail(g[i][j]=='b');
			memset(_v,0,sizeof(_v));
			if(j==m||g[i][j+1]=='.'||c[i][j+1]!=0){
				fail(i>=n-1||g[i+1][j]=='.'||g[i+2][j]=='.');
				v(i-1,j)=v(i,j-1)=v(i,j+1)=v(i+1,j-1)=v(i+1,j+1)=1;
				for(k=1;_v[k];++k);
				c[i][j]=c[i+1][j]=c[i+2][j]=k;
			}else{
				fail(j>=m-1||g[i][j+1]=='.'||g[i][j+2]=='.');
				v(i,j-1)=v(i,j+3)=v(i-1,j)=v(i-1,j+1)=v(i-1,j+2)=1;
				for(k=1;_v[k];++k);
				c[i][j]=c[i][j+1]=c[i][j+2]=k;
			}
		}
	puts("YES");
	for(i=1;i<=n;puts(""),++i)
		for(j=1;j<=m;++j)printf("%c",c[i][j]?c[i][j]+96:'.');
	return 0;
}
