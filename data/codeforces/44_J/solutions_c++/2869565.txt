#include <cstdio>
const	int	maxn = 1000 + 100;
char	map[maxn][maxn];
char	ch;
int	n, m;
bool	ff;
int	main()
{
//	freopen("44j.in", "r", stdin);
//	freopen("44j.out", "w", stdout);
	scanf("%d%d", &n, &m);
	for (int i=1; i<=n; i++)	scanf("%s", &map[i][1]);
	for (int i=1; i<=n; i++)
		for (int j=1; j<=m; j++)
			switch (map[i][j])
			{
				case 'w': map[i][j] = 'W'; break;
				case 'b': map[i][j] = 'B'; break;
			}
	ff = 1;
	for (int i=1; i<=n; i++)
	{
		for (int j=1; j<=m; j++)
		{
			if (map[i][j] == 'B') {ff = 0; break;}
			if (map[i][j] == 'W')
			{
				if (map[i][j+1] == 'B')
				{
					if (map[i][j+2] != 'W') {ff = 0; break;}
					map[i][j] = map[i][j+1] = map[i][j+2] = (i&1?'a':'b');
				}
				else
				{
					if (map[i+1][j] != 'B' || map[i+2][j] != 'W') {ff = 0; break;}
					map[i][j] = map[i+1][j] = map[i+2][j] = (j&1?'c':'d');
				}
			}
		}
		if (!ff) break;
	}
	if (ff)
	{
		printf("YES\n");
		for (int i=1; i<=n; i++)
		{
			for (int j=1; j<=m; j++) printf("%c", map[i][j]);
			printf("\n");
		}
	}
	else	printf("NO\n");
	return 0;
}
