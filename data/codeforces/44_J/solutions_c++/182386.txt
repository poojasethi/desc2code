#include <iostream>
#include <vector>
#include <map>
#include <utility>
#include <algorithm>
#include <string>
#include <set>
#include <stdio.h>
#include <sstream>
#include <math.h>
#include <numeric>

using namespace std;

int main() {
    cout << fixed;
    cout.precision(9);

    int n, m;
    cin >> n >> m;
    vector<string> s(n);
    for (int i = 0; i < n; i++) {
        cin >> s[i];
    }
    for (int i = 0; i < n; i++) {
        char hor = (i % 2 ? 'c' : 'a');
        char ver = (i % 2 ? 'd' : 'B');
        for (int j = 0; j < m; j++) {
            if (s[i][j] == 'b') {
                if (j == 0 || j == m-1 || s[i][j-1] != 'w' || s[i][j+1] != 'w') {
                    cout << "NO\n";
                    return 0;
                }
                s[i][j-1] = s[i][j] = s[i][j+1] = hor;
            }
        }
        for (int j = 0; j < m; j++) {
            if (s[i][j] == 'w') {
                if (i >= n-2 || s[i+1][j] != 'b' || s[i+2][j] != 'w') {
                    cout << "NO\n";
                    return 0;
                }
                s[i][j] = s[i+1][j] = s[i+2][j] = ver;
            }
        }
    }
    cout << "YES\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) if (s[i][j] == 'B') s[i][j] = 'b';
        cout << s[i] << "\n";
    }
    return 0;
}
