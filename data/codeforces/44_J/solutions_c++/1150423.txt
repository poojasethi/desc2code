#include <iostream>
#include <string>
#include <cstring>
using namespace std;
const int MX = 1010;

char grid[MX][MX], out[MX][MX];
bool ok(int i, int j) {
	return grid[i][j] == 'w' && out[i][j] == '.';
}
int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 1; i <= n; ++i) {
		cin >> grid[i] + 1;
		memset(out[i] + 1, '.', m);
	}
	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= m; ++j) {
			if (grid[i][j] != 'b') continue;
			char opt = 'a' + 2 * ((i / 2 + j / 2) % 2) + (i % 2);
			if (ok(i - 1, j) && ok(i + 1, j)) {
				out[i][j] = out[i - 1][j] = out[i + 1][j] = opt;
			} else if (ok(i, j - 1) && ok(i, j + 1)) {
				out[i][j] = out[i][j - 1] = out[i][j + 1] = opt;
			}
		}
	}
	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= m; ++j) {
			if (out[i][j] == '.' && grid[i][j] != '.') {
				cout << "NO\n";
				return 0;
			}
		}
	}
	cout << "YES\n";
	for (int i = 1; i <= n; ++i) {
		cout << out[i] + 1 << '\n';
	}
}
