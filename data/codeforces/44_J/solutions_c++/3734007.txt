#include <cstdio>
#include <cstdlib>
#include <ctime>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;
int n, m, flag = 0;
int map[1001][1001];
char s[1001][1001];
int main( )
{
    scanf("%d %d", &n, &m);
    for (int i = 0; i < n; i ++)
	{
        scanf("%s", s[i]);
		for (int j = 0; j < m; j ++) map[i][j] = -1;
	}
    for (int i = 0; i < n; i ++)
        for (int j = 0; j < m; j ++)
            if (s[i][j] == 'b')
           	{
				int tmp = 2 * (((i + 1) / 2 + (j + 1) / 2) & 1) + ((i + 1) & 1), flag = 0;
				if (i >= 1 && i < n - 1)
    				if (s[i - 1][j] == 'w' && s[i + 1][j] == 'w')
						map[i - 1][j] = map[i][j] = map[i + 1][j] = tmp, 
							s[i - 1][j] = s[i][j] = s[i + 1][j] = '.', flag = 1;
				if (!flag && j >= 1 && j < m - 1)
				{
					if (s[i][j - 1] == 'w' && s[i][j + 1] == 'w')
						map[i][j - 1] = map[i][j] = map[i][j + 1] = tmp, 
							s[i][j - 1] = s[i][j] = s[i][j + 1] = '.';
				}
			}
	flag = 0;
	for (int i = 0; i < n; i ++)
		for (int j = 0; j < m; j ++)
			if (s[i][j] != '.') flag = 1;
	if (flag) printf("NO\n");
	else
	{
		printf("YES\n");
		for (int i = 0; i < n; i ++)
		{
			for (int j = 0; j < m; j ++)
				printf("%c", map[i][j] == -1 ? '.' : map[i][j] + 'a');
			printf("\n");
		}
	}
	return 0;
}
