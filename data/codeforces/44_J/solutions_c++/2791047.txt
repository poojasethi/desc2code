#include <cstdio>
int read()
{char ch;
	do
	{
		scanf("%c",&ch);
		if(ch == '.') return 0;
		if(ch == 'w') return 1;
		if(ch == 'b') return 2;
	}
	while(true);
}
int x,y,s[1005][1005],p[1005][1005],n,m;
bool next()
{
	int x0=x;
	for(;x<=n;x++)
	{
		if(x>x0) y =1;
		for(;y<=m;y++)
			if(s[x][y] && !p[x][y]) return true;
	}
	return false;
}
bool place(int a,int b,int c,int d,int e,int f)
{
	if(s[a][b] != 1) return false;
	if(s[c][d] != 2) return false;
	if(s[e][f] != 1) return false;
	int mark;	bool found = true;
	for(mark=1;mark<=4 && found;mark++)
	{
		found = false;
		if(p[a][b-1]==mark || p[a-1][b]==mark || p[a][b+1] == mark || p[a+1][b] == mark)
			found = true;
		if(p[c][d-1]==mark || p[c-1][d]==mark || p[c][d+1] == mark || p[c+1][d] == mark)
			found = true;
		if(p[e][f-1]==mark || p[e-1][f]==mark || p[e][f+1] == mark || p[e+1][f] == mark)
			found = true;
		if(!found) break;
	}
	p[a][b] = p[c][d] = p[e][f] = mark;
	return true;
}
bool work()
{
	x = y = 1;
	while(next())
	{
		if(s[x][y] == 2) return false;
		if(s[x][y+1] == 2 && !p[x][y+1]){
			if(!place(x,y,x,y+1,x,y+2)) return false;}
		else{
			if(!place(x,y,x+1,y,x+2,y)) return false;}
	}
	return true;
}
int main()
{
	//freopen("input.txt","r",stdin);
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++)
		s[i][j] = read();
	if(!work()) {printf("NO\n");return 0;}
	printf("YES\n");
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
			printf("%c",s[i][j]?(p[i][j]+'a'-1):'.');
		printf("\n");
	}
}
