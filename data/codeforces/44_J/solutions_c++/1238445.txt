#include <cstdio>
bool vis[4];
char M[1001][1001];
int n,m;
void f (int i, int j){
    if (i<0 || n<=i || j<0 || m<=j || M[i][j]<'a' || 'd'<M[i][j])
        return;
    vis[M[i][j]-'a']=true;
}
int main(){
    scanf ("%d%d" , &n, &m);
    for (int i=0; i<n; i++)
        scanf ("%s" , M[i]);
    for (int i=0; i<n; i++)
        for (int j=0; j<m; j++){
            if (M[i][j]=='b') M[i][j]='B';
            if (M[i][j]=='w') M[i][j]='W';
        }
    for (int i=0; i<n; i++)
        for (int j=0; j<m; j++) if (M[i][j]=='W' || M[i][j]=='B'){
            if (M[i][j]=='B'){
                puts ("NO");
                return 0;
            }
            vis[0]=vis[1]=vis[2]=vis[3]=0;
            if (M[i][j+1]=='B' && M[i][j+2]=='W'){
                f(i-1,j); f(i-1,j+1); f(i-1,j+2); f(i,j-1); f(i,j+3);
                char ch='a';
                for (int k=0; k<4; k++) if (!vis[k]) { ch+=k; break; }
                M[i][j] = M[i][j+1] = M[i][j+2] = ch;
            }
            else if (M[i+1][j]=='B' && M[i+2][j]=='W'){
                f(i-1,j); f(i,j-1); f(i,j+1); f(i+1,j-1); f(i+1,j+1); f(i+2,j-1); 
                char ch='a';
                for (int k=0; k<4; k++) if (!vis[k]) { ch+=k; break; }
                M[i][j] = M[i+1][j] = M[i+2][j] = ch;
            }
            else{
                puts ("NO");
                return 0;
            }
        }
    puts ("YES");
    for (int i=0; i<n; i++)
        puts (M[i]);
    return 0;
}
