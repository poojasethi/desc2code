#include <stdio.h>
#include <algorithm>
using namespace std;
const int maxn=1100;
char s[maxn][maxn];
bool flag[maxn];
int n,m;
int r[]={1,0,-1,0};
int c[]={0,1,0,-1};
char gao(int x,int y,int k,int p)
{
    for(int i='a';i<='d';i++)flag[i]=true;
    for(int i=0;i<k;i++)
    {
        for(int j=0;j<4;j++)
        {
            int a=x+r[j],b=y+c[j];
            if(a>=0&&a<n&&b>=0&&b<m&&(s[a][b]>='a'&&s[a][b]<='d'))flag[s[a][b]]=false;
        }
        x+=r[p];y+=c[p];
    }
    for(int i='a';i<='d';i++)
    if(flag[i])return i;
}
int main()
{
    while(scanf("%d%d",&n,&m)!=EOF)
    {
        for(int i=0;i<n;i++)
            scanf("%s",s[i]);
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
            if(s[i][j]=='b')s[i][j]='m';
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
            if(s[i][j]=='w')
            {
                if(j+2<m&&s[i][j+1]=='m'&&s[i][j+2]=='w')
                    s[i][j]=s[i][j+1]=s[i][j+2]=gao(i,j,3,1);
                else if(i+2<n&&s[i+1][j]=='m'&&s[i+2][j]=='w')
                    s[i][j]=s[i+1][j]=s[i+2][j]=gao(i,j,3,0);
            }
        bool ans=true;
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
            if(s[i][j]=='m'||s[i][j]=='w')
            ans=false;
        if(ans)
        {
            printf("YES\n");
            for(int i=0;i<n;i++)
                printf("%s\n",s[i]);
        }
        else printf("NO\n");
    }
}
