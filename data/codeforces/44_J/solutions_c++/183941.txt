#include <stdio.h>

char s[1020][1020], tmp[3];


const int dx[8] = {-1, -1, -1, 0, 0, 1, 1, 1}, dy[8] = {0, 1, 2, -1, 3, 0, 1, 2};
int main()
{
	int n, m;
	scanf("%d%d\n", &n, &m);
	for (int i = 1; i <= n; i++)
	{
		for(int j = 1; j <= m; j++)
		{
			scanf("%1s", tmp);
			s[i][j] = tmp[0];
			if (tmp[0] == 'w') s[i][j] = 'b';
			if (tmp[0] == 'b') s[i][j] = 'w';
			
			
		}
	}

	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			if (s[i][j] == 'b')
			{
				if (s[i][j + 1] == 'w' && s[i][j + 2] == 'b')
				{
					char k;
					for (k = 'A'; k < 'E'; k++)
					{
						int cnt = 0;
						for (int r = 0; r < 8; r++)
						{							
							if (s[i + dx[r]][j + dy[r]] != k) cnt++;
						}
						if (cnt == 8) break;
					}			
					s[i][j] = s[i][j + 1] = s[i][j + 2] = k;
				}
				else if (s[i + 1][j] == 'w' && s[i + 2][j] == 'b')
				{
					char k;
					for (k = 'A'; k < 'E'; k++)
					{
						int cnt = 0;
						for (int r = 0; r < 8; r++)
						{							
							if (s[i + dy[r]][j + dx[r]] != k) cnt++;
						}
						if (cnt == 8) break;
					}
				
					s[i][j] = s[i + 1][j] = s[i + 2][j] = k;
				}
				else
				{
					printf ("NO");
					return 0;
				}
			//	printf ("%d %d\n", i, j);
			}
		}
	}
	printf ("YES\n");
	for(int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			if (s[i][j] == '.') printf ("%c", s[i][j]);
			else printf ("%c", s[i][j] - 'A' + 'a');
		}
		printf ("\n");
	}
	return 0;
}
