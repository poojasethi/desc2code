#include <cstdio>
#include <cstring>
const int vec[4][2]={{0,1},{0,-1},{1,0},{-1,0}};
int n,m,i,j,k;
bool v[5];
char map[1011][1011];
void quick(int x,int y)
{
		 for (int i=0;i<4;i++)
		 {
		 		 int xx=x+vec[i][0],yy=y+vec[i][1];
		 		 if (1<=xx&&xx<=n&&1<=yy&&yy<=m&&'1'<=map[xx][yy]&&map[xx][yy]<='4')
		 		 		v[map[xx][yy]-'0']=true;
		 }
}
int main()
{
		scanf("%d%d\n",&n,&m);
		for (i=1;i<=n;i++)
				scanf("%s\n",map[i]+1);
		for (i=1;i<=n;i++)
		for (j=1;j<=m;j++)
				if (map[i][j]=='w')
				{
					 if (map[i][j+1]=='b'&&map[i][j+2]=='w')
					 {
					 		memset(v,0,sizeof(v));
					 		quick(i,j),quick(i,j+1),quick(i,j+2);
					 		for (k=1;k<=4&v[k];k++);
					 		map[i][j]=map[i][j+1]=map[i][j+2]='0'+k;
					 }
					 else
					 if (map[i+1][j]=='b'&&map[i+2][j]=='w')
					 {
					 		memset(v,0,sizeof(v));
					 		quick(i,j),quick(i+1,j),quick(i+2,j);
					 		for (k=1;k<=4&v[k];k++);
					 		map[i][j]=map[i+1][j]=map[i+2][j]='0'+k;
					 }
					 else {printf("NO");return 0;}
				}
				else if (map[i][j]=='b') {printf("NO");return 0;}
		printf("YES\n");
		for (i=1;i<=n;i++)
		{
		 		for (j=1;j<=m;j++)
		 				putchar(map[i][j]=='.'?'.':map[i][j]-'1'+'a');
		 		printf("\n");
		}
		return 0;
}
