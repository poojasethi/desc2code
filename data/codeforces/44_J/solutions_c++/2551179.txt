#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<cstdlib>

using namespace std;
const int maxn = 1010;
char S[maxn][maxn];
char Ts[maxn][maxn];
static int n,m;
static bool use[5];
void Ck(char ch){
	if (!(ch>='a'&&ch<='d')) return ;
	use[ch-'a']=true;
}
bool Chk(){
	for (int i=1;i<=n;i++)
	for (int j=1;j<=m;j++){
		memset(use,0,sizeof(use));
		if (S[i][j]!='.'){
			if (S[i][j]!='w') return false;
			if (S[i][j+1]=='b'){
				if (S[i][j+2]!='w') return false;
				S[i][j]=S[i][j+1]=S[i][j+2]='.';
				Ck(Ts[i-1][j]);Ck(Ts[i-1][j+1]);Ck(Ts[i-1][j+2]);Ck(Ts[i][j-1]);
				Ck(Ts[i+1][j]);Ck(Ts[i+1][j+1]);Ck(Ts[i+1][j+2]);Ck(Ts[i][j+3]);
				char ch='0';
				for (int k=0;k<4;k++) if (!use[k]){ch=('a'+k);break;}
				if (ch=='0') return false;
				Ts[i][j]=Ts[i][j+1]=Ts[i][j+2]=ch;
			}else{
				if (S[i+1][j]!='b'||S[i+2][j]!='w') return false;
				S[i][j]=S[i+1][j]=S[i+2][j]='.';
				Ck(Ts[i][j-1]);Ck(Ts[i+1][j-1]);Ck(Ts[i+2][j-1]);Ck(Ts[i-1][j]);
				Ck(Ts[i][j+1]);Ck(Ts[i+1][j+1]);Ck(Ts[i+2][j+1]);Ck(Ts[i+3][j]);
				char ch='0';
				for (int k=0;k<4;k++) if (!use[k]){ch=('a'+k);break;}
				if (ch=='0') return false;
				Ts[i][j]=Ts[i+1][j]=Ts[i+2][j]=ch;
			}
		}
	}
	return true;
}
int main(){
	//freopen("44J.in","r",stdin);
	//freopen("44J.out","w",stdout);
	scanf("%d%d\n",&n,&m);
	for (int i=1;i<=n;i++){
		for (int j=1;j<=m;j++) Ts[i][j]=S[i][j]=getchar();
		getchar();
	}
	if (Chk()){
		printf("YES\n");
		for (int i=1;i<=n;i++)
			puts(Ts[i]+1);
	}
	else printf("NO\n");
	return 0;
}
