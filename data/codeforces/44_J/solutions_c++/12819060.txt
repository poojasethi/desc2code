#include<bits/stdc++.h>
using namespace std;
int n,m;
char s[1005][1005],t[1005][1005];
int main(){
  scanf("%d %d",&n,&m);
  for(int i=0;i<n;i++) scanf("\n%s",s[i]);
  for(int i=0;i<n;i++) for(int j=0;j<m;j++) t[i][j]='.';
  for(int i=0;i<n;i++) for(int j=0;j<m;j++){
    if(t[i][j]=='.' && s[i][j]!='.'){
      if(s[i][j]=='b'){
        printf("NO\n");
        return 0;
      }else{
        if(j+2<m && s[i][j+1]!='.' && s[i][j+2]!='.' && t[i][j+1]=='.' && t[i][j+2]=='.'){
          for(char col='a';col<='d';col++){
            if(j>0 && t[i][j-1]==col) continue;
            if(i>0 && t[i-1][j+1]==col) continue;
            if(j+3<m && t[i][j+3]==col) continue;
            t[i][j]=t[i][j+1]=t[i][j+2]=col;
            break;
          }
        }else if(i+2<n && s[i+1][j]!='.' && s[i+2][j]!='.' && t[i+1][j]=='.' && t[i+2][j]=='.'){
          for(char col='a';col<='d';col++){
            if(i>0 && t[i-1][j]==col) continue;
            if(j>0 && t[i][j-1]==col) continue;
            if(j+1<m && t[i][j+1]==col) continue;
            t[i][j]=t[i+1][j]=t[i+2][j]=col;
            break;
          }
        }else{
          printf("NO\n");
          return 0;
        }
      }
    }
  }
  printf("YES\n");
  for(int i=0;i<n;i++){
    for(int j=0;j<m;j++) printf("%c",t[i][j]);
    printf("\n");
  }
  return 0;
}
