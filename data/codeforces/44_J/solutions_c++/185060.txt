#include <cstring>
#include <iostream>
#include <set>
using namespace std;
string a[1005];
set <char> s;
int n,m;
void push()
{
	s.insert('1');
	s.insert('2');
	s.insert('3');
	s.insert('4');
}
void print()
{
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
			cout<<(a[i][j]>=49 && a[i][j]<=52?(char)(a[i][j]+48):a[i][j]);
		cout<<endl;
	}
}
int main()
{
	int i,j;
	cin>>n>>m;
	for (i=0;i<n;i++)
		cin>>a[i];
	bool can=true;
	push();
	for(i=0;i<n &&can;i++)
		for(j=0;j<m &&can;j++)
			if (a[i][j]=='b')
			  can=false;
			else if (a[i][j]=='w' && a[i][j+1]=='b' && a[i][j+2]=='w')
			{
				for(int j2=j;j2<=min(j+2,m);j2++)
					for(int i2=-1;i2<=1;i2+=2)
						if (i+i2>=0 && i+i2<n)
							s.erase(a[i+i2][j2]);
				if (j-1>=0)
					s.erase(a[i][j-1]);
				if (j+3<m)
					s.erase(a[i][j+3]);
				a[i][j]=*s.begin();
				a[i][j+1]=*s.begin();
				a[i][j+2]=*s.begin();
				push();
			}
		  else if (a[i][j]=='w' && a[i+1][j]=='b' && a[i+2][j]=='w')
			{
				for(int i2=i;i2<=min(i+2,n);i2++)
					for(int j2=-1;j2<=1;j2+=2)
						if (j+j2>=0 && j+j2<m)
							s.erase(a[i][j+j2]);
				if (i-1>=0)
					s.erase(a[i-1][j]);
				if (i+3<n)
					s.erase(a[i+3][j]);
				a[i][j]=*s.begin();
				a[i+1][j]=*s.begin();
				a[i+2][j]=*s.begin();
				push();
			} else if (a[i][j]!='1' && a[i][j]!='2' && a[i][j]!='3' && a[i][j]!='4' && a[i][j]!='.')
			can=false;
	if (can)
	{
		cout<<"YES"<<endl;
		print();
	} else 
		cout<<"NO";
	return 0;
}
