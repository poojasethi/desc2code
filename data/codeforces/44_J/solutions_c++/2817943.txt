#include<cstdio>
int N,M,i,j,Cntb,Cntw;
int f[5],g[1005][1005];
char Map[1005][1005];
int main()
{
	scanf("%d%d",&N,&M);
	gets(Map[0]);
	for(i=0;i<N;++i) gets(Map[i]);
	for(i=0;i<N;++i)
		for(j=0;j<M;++j)
		{
			if(Map[i][j]=='b')
			{
				++Cntb;
				Map[i][j]='l';
			}
			if(Map[i][j]=='w') ++Cntw;
		}
	if(Cntw!=Cntb*2)
	{
		printf("NO\n");
		return 0;
	}
	for(i=0;i<N;++i)
		for(j=0;j<M;++j)
		{
			f[0]=f[1]=f[2]=f[3]=0;
			if(i&1) f[0]=f[1]=1;
			else f[2]=f[3]=1;
			if(i>1) f[g[i-2][j]]=1;
			if(i>0&&j>0) f[g[i-1][j-1]]=1;
			if(i>0&&j<M-1) f[g[i-1][j+1]]=1;
			if(j>1) f[g[i][j-2]]=1;
			for(g[i][j]=0;f[g[i][j]];++g[i][j]);
		}
	for(i=0;i<N;++i)
		for(j=0;j<M;++j)
			if(Map[i][j]=='l')
			{
				if(i>0&&Map[i-1][j]=='w')
				{
					if(i<N-1&&Map[i+1][j]=='w')
						Map[i-1][j]=Map[i][j]=Map[i+1][j]=g[i][j]+'a';
					else
					{
						printf("NO\n");
						return 0;
					}
				}
				else
				{
					if(j>0&&Map[i][j-1]=='w'&&j<M-1&&Map[i][j+1]=='w')
						Map[i][j-1]=Map[i][j]=Map[i][j+1]=g[i][j]+'a';
					else
					{
						printf("NO\n");
						return 0;
					}
				}
			}
	printf("YES\n");
	for(i=0;i<N;++i) puts(Map[i]);
	return 0;
}
