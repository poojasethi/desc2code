#include<cstdio>
#include<cstring>

int n,m,a[1100][1100],bl[1100][1100],C[5],i,j,k;

void Get(int a,int b)
{
	if(a<0||a>=n||b<0||b>=m)return;
	if(bl[a][b])
		C[bl[a][b]]++;
}

int main()
{
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
		for(scanf("\n"),j=0;j<m;j++)
		{
			k=getchar();
			if(k=='.')a[i][j]=-1;
			else if(k=='b')a[i][j]=1;
		}
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			if(!a[i][j]&&!bl[i][j])
			{
				memset(C,0,sizeof C);
				if(j+2<m&&a[i][j+1]!=-1&&!bl[i][j+1]&&a[i][j+2]!=-1&&!bl[i][j+2])
				{
					Get(i-1,j);
					Get(i-1,j+1);
					Get(i-1,j+2);
					Get(i,j-1);
					Get(i,j+3);
					for(bl[i][j]=1;bl[i][j]<=4&&C[bl[i][j]];bl[i][j]++);
					if(bl[i][j]>4)
					{
						printf("NO\n");
						return 0;
					}
					bl[i][j+1]=bl[i][j+2]=bl[i][j];
				}
				else if(i+2<n&&a[i+1][j]!=-1&&!bl[i+1][j]&&a[i+2][j]!=-1&&!bl[i+2][j])
				{
					Get(i,j-1);
					Get(i-1,j);
					Get(i,j+1);
					for(bl[i][j]=1;bl[i][j]<=4&&C[bl[i][j]];bl[i][j]++);
					if(bl[i][j]>4)
					{
						printf("NO\n");
						return 0;
					}
					bl[i+1][j]=bl[i+2][j]=bl[i][j];
				}
				else
				{	printf("NO\n");
					return 0;
				}
			}
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			if(a[i][j]!=-1&&!bl[i][j])
			{
				printf("NO\n");
				return 0;
			}
	printf("YES\n");
	for(i=0;i<n;i++,putchar('\n'))
		for(j=0;j<m;j++)
			if(a[i][j]==-1)putchar('.');
				else putchar('a'+bl[i][j]-1);
	return 0;
}
