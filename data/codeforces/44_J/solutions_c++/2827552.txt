#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 1050;

char Mp[N][N], res[N][N];
int id[N][N], color[N][N], pal[N * N / 2], n, m, vcnt;

void solve ()
{
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j) if (Mp[i][j] == 'w')
		{
			if (j + 2 <= m && Mp[i][j + 1] == 'b' && Mp[i][j + 2] == 'w')
			{
				res[i][j] = res[i][j + 1] = res[i][j + 2] = 'a' + (i & 1) + (j & 1) * 2;
				Mp[i][j] = Mp[i][j + 1] = Mp[i][j + 2] = '.';
			} else if (i + 2 <= n && Mp[i + 1][j] == 'b' && Mp[i + 2][j] == 'w')
			{
				res[i][j] = res[i + 1][j] = res[i + 2][j] = 'a' + (i & 1) + (j & 1) * 2;
				Mp[i][j] = Mp[i + 1][j] = Mp[i + 2][j] = '.';
			}
		}
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= m; ++j)
			if (Mp[i][j] != '.')
			{
				puts("NO");
				exit(0);
			}
	puts("YES");
	for (int i = 1; i <= n; ++i, putchar('\n'))
		for (int j = 1; j <= m; ++j)
			putchar(res[i][j] ? res[i][j] : '.');
}

int main ()
{
#ifdef LOCAL
	freopen("in", "r", stdin);
#endif
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; ++i) scanf("%s", Mp[i] + 1);
	solve();
	return 0;
}
