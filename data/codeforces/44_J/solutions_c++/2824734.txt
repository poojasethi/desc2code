#include <cstring>
#include <cstdio>
using namespace std;

char ans[1010][1010], map[1010][1010];
int n,m;

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1; i<=n; i++) scanf("%s",map[i]+1);
	
	for (int i=1; i<=n; i++)
	for (int j=1; j<=m; j++) ans[i][j]='.';
	
	for (int i=1; i<=n; i++)
	for (int j=1; j<=m; j++)
		if (map[i][j]=='w')
		{
			if (map[i][j+1]=='b' && map[i][j+2]=='w')
				ans[i][j]=ans[i][j+1]=ans[i][j+2]='a'+(i%2)+(j%2)*2,
				map[i][j]=map[i][j+1]=map[i][j+2]='.';
			else if (map[i+1][j]=='b' && map[i+2][j]=='w')
				ans[i][j]=ans[i+1][j]=ans[i+2][j]='a'+(i%2)+(j%2)*2,
				map[i][j]=map[i+1][j]=map[i+2][j]='.';
		}
		
	bool flag=1;
	for (int i=1; i<=n; i++)
	for (int j=1; j<=m; j++) if (map[i][j]!='.') flag=0;
	if (!flag) printf("NO\n");
	else 
	{
		printf("YES\n");
		for (int i=1; i<=n; i++)
		{
			for (int j=1; j<=m; j++) printf("%c",ans[i][j]); printf("\n");
		}
	}
	//for (;;);
}
