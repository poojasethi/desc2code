#include <stdio.h>
int n,ek = 1000000001,b,tut;
int A[100005];
int main()
{
	scanf("%d",&n);
	for (int i = 0; i < n ; i++)
	{
		scanf("%d",&A[i]);
		if(A[i] < ek)
		{
			ek = A[i];
			b = 0;
			tut = i;
			continue;
		}
		if(A[i] == ek)
		{
			b = 1;
		}
	}
	printf(b ? "Still Rozdil" : "%d",tut+1);
}
