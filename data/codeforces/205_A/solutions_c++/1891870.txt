#include <stdio.h>

using namespace std;

int main()
{
	int n,x,min = 1 << 30,mintown = -1,flag = 0;
	scanf("%d",&n);
	for(int i = 0;i < n;i++)
	{
		scanf("%d",&x);
		if(x == min)flag = 0;
		if(x < min){min = x;mintown = i + 1;flag = 1;}
	}
	if(flag)printf("%d",mintown);
	else printf("Still Rozdil");
	
}
