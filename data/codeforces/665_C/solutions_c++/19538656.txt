#include <cstdio>
#include <cstring>
char s[200001];
int len;
int main (){
	gets(s);
	len = strlen(s);
	for (int i = 1; i < len; i++)
		if (s[i] == s[i - 1]){
			s[i] = (s[i - 1] - 'a' + 1) % 26 + 'a';
			if (i < len - 1 && s[i] == s[i + 1])
				s[i] = (s[i - 1] - 'a' + 2) % 26 + 'a';
		}
	puts(s);
	return 0;
}
