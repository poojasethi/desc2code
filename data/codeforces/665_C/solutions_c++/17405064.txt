#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    cin>>s;
    for(int i=0;s[i];i++){
        if(s[i]==s[i+1]){
            char p=(s[i+1]-'a'+27)%26+'a';
            char q=(s[i+1]-'a'+25)%26+'a';
            if(p!=s[i+2]) s[i+1]=p;
            else s[i+1]=q;
        }
    }
    cout<<s;
}
