#include<cstdio>
#include<cstring>

char s[200100];

int main(){
	scanf("%s",s);
	int len=strlen(s);
	for(int i = 1;i < len;i++)
		if(s[i]==s[i-1])
		for(int j = 0;j < 26;j++)
			if(j+'a'!=s[i-1] && j+'a'!=s[i+1])
			{
				s[i] = j+'a';
				break;
			}
	printf("%s\n",s);
	return 0;
}
