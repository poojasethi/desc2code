#include <iostream>
#include <string>

using namespace std;

string s;

int main()
{
  ios::sync_with_stdio(false);
  cin>>s;
  for (int i=1;i<int(s.size());i++) {
    if (s[i]==s[i-1]) {
      do {
	s[i]=char((s[i]-'a'+1)%26+'a');
      } while (i<int(s.size())-1 and s[i]==s[i+1]);
    }
  }
  cout<<s<<endl;
}
