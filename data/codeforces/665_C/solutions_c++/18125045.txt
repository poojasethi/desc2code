#include<cstdio>
#include<cstring>
#define MAXN 200010
using namespace std;
char x[MAXN];
int main(){
	scanf("%s",x);
	int len=strlen(x)-1;
	for(int i=0;i<len;i++){
		if(x[i]==x[i+1]){
			for(int j=0;j<=27;j++){
				if(x[i+1]!=j+'a'&&x[i+2]!=j+'a'){
					x[i+1]=j+'a';break;
				}
			}
		}
	}
	printf("%s\n",x);
	return 0;
}
