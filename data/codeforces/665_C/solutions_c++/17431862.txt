#include <bits/stdc++.h>
using namespace std;
const int maxn=200020;
string str;
int main()
{
    cin>>str;
    for(int i=1; i<str.size(); ++i)
        if(str[i]==str[i-1]) {
            str[i]='a';
            while(str[i]==str[i+1]||str[i]==str[i-1])
                ++str[i];
        }
    cout<<str<<endl;
    return 0;
}
