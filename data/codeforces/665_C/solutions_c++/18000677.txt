#include <cstdio>
#include <cstring>

int len;
char s[200001];

int main()
{
	int i,j;
	scanf("%s",s);
	len=strlen(s);
	for(i=1;i<len;i++)
	{
		if(s[i]==s[i-1])
		{
			for(j='a';j<='z';j++)
			{
				if(j!=s[i-1]&&j!=s[i+1])
				{
					s[i]=j;
					break;
				}
			}
		}
	}
	printf("%s",s);
	return 0;
}
