#include <iostream>
using namespace std;

int main()
{
	string s;
	cin>> s;

	char c = s[0];
	for(int i = 1; i < s.size(); i ++)
	{
		if(s[i] == c)
		{
			s[i] = ((s[i] - 'a') + 1) % 26 + 'a';
			if(i < s.size() && s[i + 1] == s[i])
				s[i] = ((s[i] - 'a') + 1) % 26 + 'a';
		}
		c = s[i];
	}

	cout<< s << endl;
}
