#include<bits/stdc++.h>
using namespace std;
char arr[1000000];
int main()
{
	int i,j,k;
	scanf("%s",arr);
	int n=strlen(arr);
	for(i=1;i<n;i++)
	{
		if(arr[i]==arr[i-1])
		{
			arr[i]='a';
		
			while(arr[i]==arr[i-1] || arr[i]==arr[i+1])
			{
			
				arr[i]++;
			
			}
		}
	
	}
	printf("%s\n",arr);
	return 0;
}
