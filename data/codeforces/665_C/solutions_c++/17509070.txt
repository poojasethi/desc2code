#include <iostream>
using namespace std;
int main()
{
	char s[200000];
	cin>>s;
	int i=1;
	while(s[i]!='\0'){
		if(s[i]==s[i-1]){
			s[i]='m';
			if(s[i-1]==s[i]||s[i+1]==s[i]){
				s[i]='l';	
			}
			if(s[i-1]==s[i]||s[i+1]==s[i]){
				s[i]='n';	
			}		
		}	
		i++;
	}
	cout<<s<<endl;
	
	return 0;
}
