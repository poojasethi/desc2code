#include<bits/stdc++.h>
using namespace std;
int main()
{
	char s[200003];
	scanf("%s",s);
	int i,le= strlen(s);
	for(i=1;i<le;i++)
	{
		if(s[i]==s[i-1])
		{
			if(s[i+1]!='a' && s[i-1]!='a' )
				s[i]='a';
			if(s[i+1]!='b' && s[i-1]!='b' )
				s[i]='b';
			if(s[i+1]!='c' && s[i-1]!='c' )
				s[i]='c';
			i++;
		}
	}
	printf("%s\n",s );
return 0;
}
