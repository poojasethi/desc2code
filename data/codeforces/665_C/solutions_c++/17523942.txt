#include <iostream>
#include <string>
using namespace std;
  string s;
int main(){
	cin >> s;
	s += '!';
	for(int i = 0; i < s.size()-1; i++)
	    if(s[i] == s[i+1])
	        for(char c = 'a'; c <= 'z'; c++)
			   if(c != s[i] && c != s[i+2])
			      s[i+1] = c;  
	 s[s.size()-1] = ' ';         
	cout << s;
 return 0;
}
