#include<bits/stdc++.h>
using namespace std;
char ar[1000000];
int uz;
int main(){
	scanf("%s",ar);
	uz=strlen(ar);
	for(int i=0 ; i<uz-1 ; i++){
		if(ar[i]==ar[i+1]){
			for(int j=1 ; j<=26 ; j++){
				if(j+96!=ar[i] && j+96!=ar[i+2]){
					ar[i+1]=j+96;
					break;
				}
			}
		}
	}
	printf("%s",ar);
	return 0;
}
