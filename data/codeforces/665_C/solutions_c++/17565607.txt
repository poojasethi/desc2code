#include<iostream>
#include<string>
using namespace std;
int x,i,j,n;
string s;
char c;

int main(){
	cin>>s;
	x = s.size();
	for(i=1; i<x; i++){
		if(s[i]==s[i-1]){
			c = 'a';
			while(2>1){
				if(c!=s[i-1] && c!=s[i] && c!=s[i+1]){
					s[i] = c;
					break;
				}
				c++;
			}
		}
	}
	cout<<s;
}
