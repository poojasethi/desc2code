#include <iostream>
using namespace std;
int main()
{
	string s;
	cin>>s;
	for(int i=1;i<s.size();i++)
	{
		if(s[i]==s[i-1])
		{
			if(s[i-1]!=97 && s[i+1]!=97)
				s[i]=97;
			if(s[i-1]!=98 && s[i+1]!=98)
				s[i]=98;
			if(s[i-1]!=100 && s[i+1]!=100)
				s[i]=100;
		}
	}
	cout<<s<<endl;
	return 0;
}
