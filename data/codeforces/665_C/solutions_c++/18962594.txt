#include<stdio.h>
int main(){
    int i;
    char str[200001];
    scanf("%s",str);
    for(i=1;str[i];i++)
        if(str[i]==str[i-1]){
            if(str[i]+1==str[i+1] || str[i]=='z' && str[i+1]=='a') str[i]+=2;
            else str[i]++;
            if(str[i]>'z') str[i]-=26;
        }
    printf("%s\n",str);
}
