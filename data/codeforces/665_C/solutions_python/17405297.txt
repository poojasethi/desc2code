def next_char(c):
    x = ord(c) + 1
    if x >= ord('z'):
        x = ord('a')
    return chr(x)

def next_char_not_this(c, t):
    c = next_char(c)
    while (c == t):
        c = next_char(c)
    return c

string = raw_input()
s = [string[i] for i in range(len(string))]
for i in range(1, len(s)):
    if s[i] == s[i - 1]:
        d = next_char(s[i])
        if i < len(s) - 1:
            s[i] = next_char_not_this(d, s[i + 1])
        else:
            s[i] = d

print "".join(s)
