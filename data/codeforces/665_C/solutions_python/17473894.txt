from string import ascii_lowercase
s=list(raw_input().strip())
for i in xrange(len(s)):
    if i<len(s)-1 and s[i]==s[i+1]:
        if i==len(s)-2:
            for c in ascii_lowercase:
                if c!=s[i]:
                    s[i+1]=c
                    break
        else:
            for c in ascii_lowercase:
                if c!=s[i] and c!=s[i+2]:
                    s[i+1]=c
                    break
print ''.join(s)
