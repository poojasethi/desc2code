LETTERS = 'abcdefghijklmnopqrstuvwxyz'

def get_letter(prev_letter, next_letter):
    for letter in LETTERS:
        if next_letter and letter == next_letter:
            continue

        if letter != prev_letter:
            return letter


def solve(s):
    s = list(s)
    for idx in xrange(1, len(s)):
        if s[idx] == s[idx - 1]:
            s[idx] = get_letter(s[idx-1], s[idx+1] if idx != len(s) - 1 else None)
    return ''.join(s)


print solve(raw_input())
