import random
import sys

def get_rand(a, b):
    s = 'abcdefghijklmnopqrstuvwxyz'

    while True:
        i = random.sample(range(0, len(s)), 1)[0]
        if s[i] not in [a,b]:
            return s[i]

s = str(sys.stdin.read())
r = ''

for i in range(0, len(s)):
    if (i!=0)and(s[i]==r[i-1])and(i != len(s)-1):
        if i == len(s)-1:
            r += get_rand(s[i], '')
        else:
            r += get_rand(s[i], s[i+1])
    else:
        r += s[i]

sys.stdout.write(r)
