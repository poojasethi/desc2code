import sys
import math
import random

#== Input =====================================================================

data = sys.stdin.read().strip(); S = list(data)

#== Algorithm =================================================================

glue = ['a', 'b', 'c']

for i in range(1, len(S)):
    if S[i] == S[i - 1]:
        while S[i] == S[i - 1] or (i + 1 < len(S) and S[i] == S[i + 1]):
            S[i] = random.choice(glue)

print(''.join(S))
