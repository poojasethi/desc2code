s=list(raw_input())
if len(s)==1:
    print s[0]
    exit()

for i in range(1,len(s)-1):
    if s[i-1]==s[i]:
        j='a'
        while j==s[i-1] or j==s[i+1]:
            j=chr(ord(j)+1)
        s[i]=j
if s[-1]==s[-2]:
    j='a'
    while j==s[-2]:
        j=chr(ord(j)+1)
    s[-1]=j
print ''.join(s)
