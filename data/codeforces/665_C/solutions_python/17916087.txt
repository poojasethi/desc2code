import string

s = raw_input()
al = list(string.ascii_lowercase)
ret = s[0]
for i in xrange(1, len(s) - 1):
	if ret[-1] == s[i]:
		ret += list(set(al) - set([ret[-1], s[i + 1]]))[0]
		continue
	ret += s[i]
if len(s) > 1:
	if ret[-1] == s[-1]:
		ret += list(set(al) - set([ret[-1], s[-1]]))[0]
	else:
		ret += s[-1]
print ret
