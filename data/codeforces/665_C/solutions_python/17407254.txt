import random
b = raw_input()
a = list(b)
letters = "abcdefghijklmnopqrstuvwxyz"

for i in range(1, len(a)):
	if a[i] == a[i-1]:
		while True:
			x = random.randint(0, 25)
			if i != len(a) - 1:
				if letters[x] not in [a[i], a[i+1]]:
					a[i] = letters[x]
					break
			else:
				if letters[x] != a[i]:
					a[i] = letters[x]
					break

print("".join(a))
