def digRoot(n): return 9 if n % 9 == 0 else n % 9

n = int(raw_input())
good = 0
for i in range(1, n + 1):
    temp = n // i
    good += temp

all = 0
for a in range(1, 10):
    for b in range(1, 10):
        c = digRoot(a * b)
        all += ( (n + 9 - a) // 9 ) * ( (n + 9 - b) // 9 ) * ( (n + 9 - c) // 9 )

print(all - good)
