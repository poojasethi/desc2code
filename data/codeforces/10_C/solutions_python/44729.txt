import sys

def g(n):
    g = [n/9+1]*(n%9)+[n/9]*(9-n%9)
    g = g[-1:]+g[:-1]
    f = [0]*9
    for i in range(9):
        for j in range(9):
            f[(i*j)%9] += g[i]*g[j]
    r = sum([f[i]*g[i] for i in range(9)])
    for i in range(1,n+1):
        r -= n/i
    return r

print "%d" % g(int(sys.stdin.readline()))
