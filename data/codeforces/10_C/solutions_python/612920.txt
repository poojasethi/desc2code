n = int(input())
A = 0
for i in range(1, 10):
    x = n / 9
    if(n - x * 9 >= i):
        x += 1
    for j in range(1, 10):
        y = n / 9
        if(n - y * 9 >= j):
            y += 1
        k = i * j % 9
        if(k == 0): k = 9
        z = n / 9
        if(n - z * 9 >= k):
            z += 1
        #print "(%d,%d)(%d,%d)(%d,%d)"%(i, x, j, y, k, z)
        A += x * y * z
B = 0
for i in range(1, n + 1):
    B += n / i
print A - B
