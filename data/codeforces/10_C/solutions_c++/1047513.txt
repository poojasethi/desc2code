#include <iostream>
using namespace std;
const int MXN = 1000100;
typedef long long ll;

ll cnt[10];
ll sm;
int main() {
	int N;
	cin >> N;
	for (int i = 1; i <= N; ++i) {
		++cnt[i % 9];
	}
	for (int i = 0; i < 9; ++i) {
		for (int j = 0; j < 9; ++j) {
			int t = i * j % 9;
			sm += cnt[t] * cnt[i] * cnt[j];
		}
	}
	for (int i = 1; i <= N; ++i) {
		sm -= N / i;
	}
	cout << sm;
}
