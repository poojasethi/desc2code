#include <cstdio>

using namespace std;

#define forn(i,n) for(int i=0;i<(int)n;i++)

typedef long long int64;

int64 m[9],ans;
int main()
{
   int n;
   scanf("%d",&n);
   forn(i,n)
     m[(i+1)%9]++,
     ans-=n/(i+1);
   forn(i,9)
     forn(j,9)
       ans+=m[i]*m[j]*m[(i*j)%9];
   printf("%I64d\n",ans);
   return 0;
}
