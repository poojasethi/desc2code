#include <iostream>
#include <vector>

using namespace std;

#define lint long long

lint a[9];

int main(){
    int n;
    cin >> n;
    for (int i = 1; i <= n; i++)
        a[i % 9]++;
    lint ans = 0;
    for (int i = 0; i < 9; i++)
        for (int j = 0; j < 9; j++)
            ans += a[i] * a[j] * a[(i * j) % 9];
    for(int i = 1; i <= n; i++)
        ans -= n / i;
    cout << ans;
    return 0;
}
