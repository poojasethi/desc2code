#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
using namespace std;
typedef long long llong;
llong c[10];
int main() {
	int n;
	scanf("%d", &n);
	memset(c, 0, sizeof(c));
	llong x = 0, y = 0;
	for (int i = 1; i <= n; i++) {
		c[i % 9]++;
		y += n / i;
	}
	for (int i = 0; i < 9; i++) {
		for (int j = 0; j < 9; j++) {
			x += c[i] * c[j] * c[(i * j) % 9];
		}
	}
	printf("%I64d\n", x - y);
	return 0;
}
