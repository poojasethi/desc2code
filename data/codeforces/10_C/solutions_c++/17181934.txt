#include<iostream>

int main(int agrc, char* argv[]){
  int N; std::cin>>N;
  int rp = 0;
  for(int i=1; i<=N; i++)
    rp += (N/i);
  long long int cnt[9]; 
  for(int i=0; i<9; i++)
    cnt[i] = (N>=i?(N-i)/9+1:0);
  cnt[0]--;
  long long int found = 0;
  for(int i=0; i<9; i++) for(int j=0; j<9; j++)
    found += cnt[i]*cnt[j]*cnt[(i*j)%9];
  std::cout<<(found - rp)<<std::endl;
  return 0;
}
