#include <iostream>
#include <vector>
using namespace std;
typedef long long LL;

int main()
{
	int N;
	cin >> N;

	LL r[9];
	for(int i=0; i<9; ++i)
		r[i] = (N+1)/9 + (i < (N+1)%9);
	r[0]--;

	LL ps = 0;
	for(int i=0; i<9; ++i)
	for(int j=0; j<9; ++j)
	for(int k=0; k<9; ++k)
		if( i*j%9 == k )
			ps += r[i]*r[j]*r[k];

	LL ts = 0;
	for(int A=1; A<=N; ++A)
		ts += N/A; 

	cout << ps - ts << endl;
}
