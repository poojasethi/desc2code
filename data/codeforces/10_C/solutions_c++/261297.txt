#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

#define forn(i,n) for(int i=0;i<(int)n;i++)

typedef long long int64;

int64 m9[9];
int main()
{
   int n;
   cin>>n;
   int64 ans=0;
   for(int i=1;i<=n;i++)
     m9[i%9]++;
   forn(i,9)
     forn(j,9)
       ans+=m9[i]*m9[j]*m9[(i*j)%9];

   for(int i=1;i<=n;i++)
     ans-=n/i;
   
   cout<<ans<<endl;
   return 0;
}
