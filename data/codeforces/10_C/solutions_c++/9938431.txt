#include<bits/stdc++.h>
#define M 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
ll a[10];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll i,j,n,t=0;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		a[i%9]++;
		t-=n/i;
	}
	for(i=0;i<9;i++)
	{
		for(j=0;j<9;j++)
			t+=a[i]*a[j]*a[(i*j)%9];
	}
	cout<<t<<'\n';
	return 0;
}
