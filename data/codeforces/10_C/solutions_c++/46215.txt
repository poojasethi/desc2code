#include <iostream>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int n;
long long r[10];

int main() {
    cin >> n;
    rep(i, n) r[(i+1)%9]++;
    long long ans = 0;
    rep(a, 9) rep(b, 9) ans += r[a]*r[b]*r[a*b%9];
    rep(i, n) ans -= n/(i+1);
    cout << ans << endl;
    return 0;
}
