#include<stdio.h>

__int64 a[10];

int main(int argc, char *argv[])
{
    int n, i, j;
    scanf("%d",&n);
    for(i = 1; i <= n; i++) a[i % 9]++;
    __int64 all = 0;
    for(i = 0; i < 9; i++)
        for(j = 0; j < 9; j++) all += a[i] * a[j] * a[(i * j) % 9];
    __int64 add = 0;
    for(i = 1; i <= n; i++)
        for(j = i; j <= n; j += i) add++;
    printf("%I64d\n", all - add);
}
