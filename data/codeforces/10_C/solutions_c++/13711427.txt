#include <iostream>

using namespace std;

long long count[10] = {0};

int main() {
	int N;
	cin >> N;
	long long ans = 0;
	for (int i = 1; i <= N; ++i)
		++count[i % 9];
	for (int i = 0; i < 9; ++i)
		for (int j = 0; j < 9; ++j)
			ans += count[(i * j) % 9] * count[i] * count[j];
	for (int i = 1; i <= N; ++i)
		ans -= N / i;
	cout << ans << endl;
	return 0;
}
