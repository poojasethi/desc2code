#include <iostream>
using namespace std;

int main(void) {
	int N, i, j;
	int a[9] = {};
	cin >> N;
	for(i=0;i<9;i++)
		a[i] = N/9+(N%9>=i);
	a[0]--;
	long long ans = 0;
	for(i=0;i<9;i++)
		for(j=0;j<9;j++)
			ans += (long long)a[i] * (long long)a[j] * (long long)a[i*j%9];
	for(i=1;i<=N;i++) {
		ans -= (N/i);
	}
	cout << ans << endl;
	return 0;
}
