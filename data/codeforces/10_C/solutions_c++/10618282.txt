#include<bits/stdc++.h>
using namespace std;
long long c[10];
int n;
int main()
{
    long long ans=0;
    memset(c,0,sizeof(c));
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        c[i%9]++;
        ans-=n/i;
    }
    for(int i=0;i<9;i++)
        for(int j=0;j<9;j++)
            ans+=c[i]*c[j]*c[(i*j)%9];
    cout<<ans<<endl;
    return 0;
}
