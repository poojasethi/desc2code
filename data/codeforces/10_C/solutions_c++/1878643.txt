#include<iostream>

using namespace std;

long long c[9];
int n;

int main()
{
        cin >> n;
        for(int i = 1; i <= n; i++)
		{
			c[i % 9]++;
		}
        long long ans=0;
        for(int i = 0; i < 9; i++)
		{
			for(int j = 0; j < 9; j++)
			{
				ans += c[i] * c[j] * c[i * j % 9];
			}
		}
        for(int i = 1; i <= n; i++)
		{
			ans-=n/i;
		}
		cout<<ans<<endl;
		return 0;
}
