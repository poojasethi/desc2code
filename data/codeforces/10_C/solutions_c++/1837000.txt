#include<iostream>
using namespace std;
long long i, j, n, a[10];
long long res = 0;
int main(){
    cin>>n;
    for (i = 1; i <= n; i++) a[(i - 1)%9+1]++;
    for (i = 1; i <= 9; i++)
      for (j = 1; j <= 9; j++) res += a[i]*a[j]*a[(i*j-1)%9+1];
    for (i = 1; i <= n; i++) res -= n/i;
    cout<<res;
}
