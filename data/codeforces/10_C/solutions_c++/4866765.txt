#include <bits/stdc++.h>

using namespace std;

typedef long long LL;

const int MAXN = 1000005;

LL n, sol, cant[MAXN];

int main()
{
    cin >> n;
    
    for (int i = 1; i <= n; i++)
        cant[i%9]++;
        
    for (int i = 0; i < 9; i++)
        for (int j = 0; j < 9; j++)
            sol += cant[i]*cant[j]*cant[(i*j)%9];

    for (int i = 1; i <= n; i++)
        sol -= n/i;

    cout << sol << endl;
}
