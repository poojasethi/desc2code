#include<bits/stdc++.h>
using namespace std;
#define LL long long
int a[10];
int main()
{
	int n;
	scanf("%d",&n);
	LL ans = 0;
	for (int i = 1;i<=n;i++)
       a[i%9]++,ans-=n/i;
	for (int i = 0;i<9;i++)
		for (int j = 0;j<9;j++)
			ans+=1LL *a[i]*a[j]*a[i*j%9];
	printf("%lld\n",ans);
}
