#include <cstdio>
#include <iostream>
const int MAXN=1000005;
long long cnt[9];
int main(){
    //freopen("ex.in", "r", stdin);
    int n;
    scanf("%d", &n);
    for (int i=1;i<=n;i++) cnt[i%9]++;
    long long ans=0;
    for (int i=1;i<=n;ans-=n/i++) for (int j=0;j<9;j++) ans+=cnt[(i*j)%9]*cnt[j];
    std::cout<<ans<<std::endl;
    return 0;
}
