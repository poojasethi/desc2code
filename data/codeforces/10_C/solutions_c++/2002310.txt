#include<iostream>
using namespace std;

int main() {
  int n;
  long long rem[10];
  cin>>n;
  for (int i=1;i<=9;i++) {
    if (i<=n%9)
      rem[i]=n/9+1;
    else
      rem[i]=n/9;
  }
  rem[0]=rem[9];
  long long cnt=0;
  for (int i=1;i<=9;i++) {
    for (int j=1;j<=9;j++) {
      cnt+=rem[i]*rem[j]*rem[i*j%9];
    }
  }
  for(int ni=1;ni<=n;ni++)
    cnt-=n/ni;
  cout<<cnt<<endl;
  return 0;
}
