#include<cstdio>
__int64 a[11];
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        a[1+(i-1)%9]++;
    }
    __int64 ans=0;
    for(i=1;i<=9;i++)
    {
        for(j=1;j<=9;j++)
        {
            ans+=a[i]*a[j]*a[(i*j-1)%9+1];
        }
    }
    //printf("%d\n",ans);
    for(i=1;i<=n;i++)
    {
        ans-=n/i;
    }
    printf("%I64d\n",ans);
    return 0;
}
