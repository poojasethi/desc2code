#include <cstdio>
#define FOR(i,s,e) for(int i=(s);i<(e);i++)
typedef long long LL;

LL a[10] = {0}, ans = 0;
int n;

int main() {
    scanf("%d", &n);
    FOR(i,0,9)
        a[i] = (n / 9) + (n % 9 >= i && i);
    FOR(i,0,9) FOR(j,0,9)
        ans += a[i] * a[j] * a[i*j%9];
    FOR(i,1,n+1)
        ans -= n / i;
    printf("%I64d\n", ans);
    return 0;
}
