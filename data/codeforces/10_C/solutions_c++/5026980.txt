#include<cstdio>
#include<algorithm>
using namespace std;
long long c[10],t = 0ll;
int n;
int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        c[i%9]++, t-= n/i;
    for(int i=0;i<9;i++)
        for(int j=0;j<9;j++)
            t += c[i]*c[j]*c[(i*j)%9];
    printf("%I64d",t);
    return 0;
}
