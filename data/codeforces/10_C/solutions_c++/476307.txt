#include <stdio.h>
#include <iostream>
using namespace std;

int n;
long long F[10],sol;

int main(){
    cin>>n;
    for (int i=1; i<=n; i++)
        F[(i-1)%9+1]++;
    for (int i=1; i<10; i++)
        for (int j=1; j<10; j++)
            sol+=F[i]*F[j]*F[(i*j-1)%9+1];
    for (int i=1; i<=n; i++) sol-=n/i;
    cout<<sol;
}
