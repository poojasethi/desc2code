#include <bits/stdc++.h>

int x[10005],y[10005],z[10005];
double f[10005];
const double eps=1e-6;
double sqr(double x)
{
	return x*x;
}
int main()
{
    int t, n;
    double vp, vs, px, py, pz;
    while(scanf("%d",&n)!=EOF)
    {
        for(int i=0;i<=n;i++)
			scanf("%d%d%d",&x[i],&y[i],&z[i]);
		scanf("%lf%lf%lf%lf%lf",&vp,&vs,&px,&py,&pz);
        f[0]=0;
        for(int i=0;i<n;i++)
			f[i+1]=f[i]+sqrt(sqr(x[i]-x[i+1])+sqr(y[i]-y[i+1])+sqr(z[i]-z[i+1]))/vs;
        double ll,rr,mid,x1,y1,z1,tp,ts;
        bool solved=0;
        for(int i=0;i<n;i++)
        {
            ll=0,rr=1,tp=1,ts=0;
            for(int ti=0;ti<100;ti++)
            {
                mid=(ll+rr)/2;
                x1=x[i]+(x[i+1]-x[i])*mid;
				y1=y[i]+(y[i+1]-y[i])*mid;
                z1=z[i]+(z[i+1]-z[i])*mid;
                tp=sqrt(sqr(x1-px)+sqr(y1-py)+sqr(z1-pz))/vp;
                ts=f[i]+(f[i+1]-f[i])*mid;
                if(tp>ts) ll=mid;
                else rr=mid;
            }
            if(fabs(tp-ts)<eps)
            {
                solved=1;
                break;
            }
        }
        if(solved)
        {
            puts("YES");
            printf("%.6lf\n%.6lf %.6lf %.6lf\n",tp,x1,y1,z1);
        }
        else
			puts("NO");
    }
    return 0;
}
