#include<cstdio>
#include<cmath>
//                          Last Change:  2011-03-18 22:03:08
#define sqr(x) ((x)*(x))
#define MX 10005
#define eps 1e-7
struct point
{
	double x,y,z;
	point(){}
	point(double x,double y,double z):x(x),y(y),z(z){}
	inline void read(){
		double x,y,z;
		scanf("%lf%lf%lf",&x,&y,&z);
		*this=point(x,y,z);
	}
	inline point operator +(const point &p){
		return point(p.x+x,p.y+y,p.z+z);
	}
	inline point operator /(const double &p){
		return point(x/p,y/p,z/p);
	}
}p[MX],a;
int n;
double vp,vs,t[MX];
inline double dis(const point &p,const point &q){
	return sqrt(sqr(p.x-q.x)+sqr(q.y-p.y)+sqr(q.z-p.z));
}
bool check()
{
	double s=0;
	int i=2;
	for(;i<=n;++i)
	{
		t[i]=t[i-1]+dis(p[i],p[i-1])/vs;
		double tt=dis(a,p[i])/vp;
		if(t[i]+eps>tt)break;
	}
	if(i>n)return false;
	puts("YES");
	point l=p[i-1],r=p[i];
	while(dis(l,r)>1e-7)
	{
		point mid=(l+r)/2.0;
		double t1=dis(a,mid)/vp;
		double t2=dis(mid,p[i-1])/vs+t[i-1];
		if(t1-t2<1e-15)r=mid;
		else l=mid;
		//printf("%.3lf %.3lf %.3lf\n",l.x,l.y,l.z);
		//printf("%.3lf %.3lf %.3lf\n\n",r.x,r.y,r.z);
	}
	printf("%.10lf\n%.10lf %.10lf %.10lf\n",dis(l,a)/vp,l.x,l.y,l.z);
	return true;
}
int main()
{
	scanf("%d",&n);++n;
	for(int i=1;i<=n;++i)
		p[i].read();
	scanf("%lf%lf",&vp,&vs);
	a.read();
	if(!check())
		puts("NO");
	return 0;
}
