#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <cstdio>

using namespace std;

#define INF 999999999
#define mp make_pair
#define pb push_back
#define sc second
#define fs first

double sq(double a){return a*a;}

double x[10100], y[10100], z[10100], vp, vs, X, Y, Z;

int main(){
    //freopen("in.txt", "rt", stdin);

    int n = 0;
    cin >> n;
    for(int i = 0; i <= n; ++i)cin >> x[i] >> y[i] >> z[i];
    cin >> vp >> vs >> X >> Y >> Z;

    double t = 0;

    for(int i = 0; i < n; ++i){
        double rast = sqrt(sq(x[i] - x[i+1]) + sq(y[i] - y[i+1]) + sq(z[i] - z[i+1]));
        double rast2 = sqrt(sq(X - x[i+1]) + sq(Y - y[i+1]) + sq(Z - z[i+1]));
        //printf("%.10f %.10f %.10f %.10f\n", rast, rast2, t, 0);
        if(rast2 / vp < t + rast / vs + 1e-8){
            double time = rast / vs;
            double l = 0, r = time, m, dx, dy, dz;
            for(int j = 0; j < 220; ++j){
                m = (l + r) / 2;
                dx = (x[i+1] - x[i]) * m / time + x[i];
                dy = (y[i+1] - y[i]) * m / time + y[i];
                dz = (z[i+1] - z[i]) * m / time + z[i];
                double rast3 = sqrt(sq(X - dx) + sq(Y - dy) + sq(Z - dz));
                if(rast3 / vp <= t + m) r = m;
                else l = m;
            }
            printf("YES\n%.10f\n%.10f %.10f %.10f", t+m, dx, dy, dz);
            return 0;
        }
        t += rast / vs;
    }
    cout << "NO";
    return 0;
}
