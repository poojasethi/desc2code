#include <stdio.h>
#include <math.h>

#define REP(i,n) for (int i=0,_n=n; i<_n; i++)
#define MAXN 10001

double dist(double a, double b, double c, double d, double e, double f){
	double x=a-d, y=b-e, z=c-f;
	return sqrt(x*x + y*y + z*z);
}

int X[MAXN],Y[MAXN],Z[MAXN],n, vp,vs, px,py,pz;

bool inside(double R, double x, double y, double z){
	return dist(x,y,z, px,py,pz) - 1e-9 <= R;
}

bool caught(){
	if (X[0]==px && Y[0]==py && Z[0]==pz){
		printf("YES\n0\n%d %d %d\n",X[0],Y[0],Z[0]);
		return 1;
	}
	double t = 0, pt = 0;
	for (int i=1; i<=n; i++){
		pt = t;
		t += dist(X[i-1],Y[i-1],Z[i-1],X[i],Y[i],Z[i]) / vs;
		if (!inside(t*vp, X[i],Y[i],Z[i])) continue;
		double dx = X[i]-X[i-1];
		double dy = Y[i]-Y[i-1];
		double dz = Z[i]-Z[i-1];
		double lo=0, hi=1, rt=t,rx=X[i],ry=Y[i],rz=Z[i];
		REP(j,1000){
			double mid = (lo+hi)/2.0;
			double x = X[i-1] + mid*dx;
			double y = Y[i-1] + mid*dy;
			double z = Z[i-1] + mid*dz;
			double dt = dist(X[i-1],Y[i-1],Z[i-1], x,y,z) / vs;
			if (inside((pt+dt)*vp, x,y,z)){
				hi = mid;
				rt=pt+dt; rx=x; ry=y; rz=z;
			} else {
				lo = mid;
			}
		}
		printf("YES\n%.10lf\n%.10lf %.10lf %.10lf\n",rt,rx,ry,rz);
		return 1;
	}
	return 0;
}

int main(){
	while (scanf("%d",&n)!=EOF){
		REP(i,n+1) scanf("%d %d %d",&X[i],&Y[i],&Z[i]);
		scanf("%d %d %d %d %d",&vp,&vs,&px,&py,&pz);
		if (!caught()) puts("NO");
	}
}
