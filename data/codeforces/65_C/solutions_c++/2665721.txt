#include <cstdio>
#include <cmath>

const double eps=1e-6;
struct Tacka
{
	double koord[3];
};
int n;
Tacka tacke[10005];
Tacka pocPoz;
double rast(Tacka t1, Tacka t2)
{
	double suma=0;
	for ( int i=0;i<3;i++)
		suma+=(t1.koord[i]-t2.koord[i])*(t1.koord[i]-t2.koord[i]);
	return sqrt(suma);
}
double vp,vs;
double put;
double put1;
double dt,gt;
Tacka sredina( Tacka t1, Tacka t2)
{
	Tacka t3;
	for ( int i=0;i<3;i++)
		t3.koord[i]=0.5*(t1.koord[i]+t2.koord[i]);
	return t3;
}
double dabs(double d){ if ( d<0 ) return -d; return d;}
Tacka st;
double binPret( int k )
{
	Tacka dt=tacke[k-1];
	Tacka gt=tacke[k];	
	double putp; 
	double puts;
	while ( true )
	{
		st=sredina(dt,gt);	
		puts=put+rast(st,tacke[k-1]);
		putp=rast(pocPoz,st);
		if ( fabs(puts*vp-putp*vs)<eps)
			return puts/vs;
		else if ( puts*vp < putp*vs )
			dt=st;
		else 
			gt=st; 
	}	
}

{
	scanf("%d",&n);

	scanf("%lf %lf",&vp,&vs);
	scanf("%lf %lf %lf",&pocPoz.koord[0],&pocPoz.koord[1],&pocPoz.koord[2]);
	for ( int i=1;i<=n;i++)
	{
		double p=rast(tacke[i],pocPoz);
		put1=put+rast(tacke[i],tacke[i-1]);
		if (  put1*vp >= p*vs)  
		{
			printf("YES\n");
			double t=binPret(i);
			printf("%lf\n",t);
			printf("%lf %lf %lf\n",st.koord[0],st.koord[1],st.koord[2]);

		}

	}

}
