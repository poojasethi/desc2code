#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
int n;
double x[10011],y[10011],z[10011],t[10011],tt[10011],vp,vs,px,py,pz;
const double eps = 1e-11;
double sqr(double q)
{
	return q * q;
}
int main()
{
	//freopen("input.txt","r",stdin);
	cin>>n;
	++n;
	for(int i = 0; i < n; i++)
		cin>>x[i]>>y[i]>>z[i];
	cin>>vp>>vs>>px>>py>>pz;
	t[0] = 0.0;
	tt[0] = sqrt(sqr(x[0] - px) + sqr(y[0] - py) + sqr(z[0] - pz));
	for(int i = 1; i < n; i++)
	{
		t[i] = t[i - 1] + sqrt(sqr(x[i] - x[i - 1]) + sqr(y[i] - y[i - 1]) + sqr(z[i] - z[i - 1])) / vs;
		tt[i] = sqrt(sqr(x[i] - px) + sqr(y[i] - py) + sqr(z[i] - pz)) / vp;
	}
	for(int i = 1; i < n; i++)
		if (t[i - 1] < tt[i - 1] + eps && t[i] + eps > tt[i])
		{
			double l = 0.0,r = sqrt(sqr(x[i] - x[i - 1]) + sqr(y[i] - y[i - 1]) + sqr(z[i] - z[i - 1]));
			double nx = (x[i] - x[i - 1]) / r;
			double ny = (y[i] - y[i - 1]) / r;
			double nz = (z[i] - z[i - 1]) / r;
			for(int j = 0; j < 100; j++)
			{
				double m = (r + l) / 2.0;
				if (t[i - 1] + m / vs + eps > sqrt(sqr(px - (x[i - 1] + nx * m)) + sqr(py - (y[i - 1] + ny * m)) + sqr(pz - (z[i - 1] + nz * m))) / vp) r = m; else l = m;
			}
			cout<<"YES"<<endl;
			printf("%.8f\n%.8f %.8f %.8f",t[i - 1] + r / vs,x[i - 1] + nx * r,y[i - 1] + ny * r,z[i - 1] + nz * r);
			return 0;
		}
	cout<<"NO";
	return 0;
}
