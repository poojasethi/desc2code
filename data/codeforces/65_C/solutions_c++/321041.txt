#include <cstdio>
#include <cmath>
#include <iostream>

using namespace std;

#define eps 1e-11

int x[10001],y[10001],z[10001];

int main() {
	int n;
	int x0,y0,z0,x1,y1,z1;
	scanf("%d",&n);
	for (int i=0;i<=n;i++) scanf("%d %d %d",&x[i],&y[i],&z[i]);
	int vs,vp;
	int px,py,pz;
	scanf("%d %d",&vp,&vs);
	scanf("%d %d %d",&px,&py,&pz);
	double t=0;
	for (int i=1;i<=n;i++) {
		double dx=x[i]-x[i-1], dy=y[i]-y[i-1], dz=z[i]-z[i-1];
		double t2=t+sqrt(dx*dx+dy*dy+dz*dz)/vs;
		dx=px-x[i]; dy=py-y[i]; dz=pz-z[i];
		double tp=sqrt(dx*dx+dy*dy+dz*dz)/vp;
		if (tp>t2+eps) { t=t2; continue; }

		double tm=0, tM=1, ts;
		double xs, ys, zs;
		for (int it=0;it<=1000;it++) {
			double tt=(tm+tM)/2;
			xs=x[i-1]+(x[i]-x[i-1])*tt;
			ys=y[i-1]+(y[i]-y[i-1])*tt;
			zs=z[i-1]+(z[i]-z[i-1])*tt;
			dx=px-xs; dy=py-ys; dz=pz-zs; tp=sqrt(dx*dx+dy*dy+dz*dz)/vp;
			dx=x[i-1]-xs; dy=y[i-1]-ys; dz=z[i-1]-zs; ts=t+sqrt(dx*dx+dy*dy+dz*dz)/vs;
			if (tp<ts+eps) tM=tt;
			else tm=tt;
		}
		printf("YES\n");
		printf("%.9lf\n",tp);
		printf("%.9lf %.9lf %.9lf\n",xs,ys,zs);
		return 0;
	}
	printf("NO\n");
    return 0;
}
