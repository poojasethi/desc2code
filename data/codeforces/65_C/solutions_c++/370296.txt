#include<iostream>
#include<math.h>
#include<stdio.h>
using namespace std;
double x[10001],y[10001],z[10001],t[10001],vp,vs,px,py,pz,xx,yy,zz,eps=0.0000000001;
int n;

double poz(double tt)
{
	int i;
	for(i=1;t[i]<tt && i<=n;i++);
	xx=x[i-1]+(x[i]-x[i-1])*(tt-t[i-1])/(t[i]-t[i-1]);
	yy=y[i-1]+(y[i]-y[i-1])*(tt-t[i-1])/(t[i]-t[i-1]);
	zz=z[i-1]+(z[i]-z[i-1])*(tt-t[i-1])/(t[i]-t[i-1]);
	return sqrt((px-xx)*(px-xx)+(py-yy)*(py-yy)+(pz-zz)*(pz-zz));
}

double bin(double l,double r)
{
	double m=(l+r)/2.0;
	if(r-l<eps) return m;
	if(poz(m)/vp<m) return bin(l,m);
	if(poz(m)/vp>=m) return bin(m,r);
}
int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	cin>>n;
	for(int i=0;i<=n;i++)
		cin>>x[i]>>y[i]>>z[i];
	cin>>vp>>vs>>px>>py>>pz;
	t[0]=0;
	for(int i=1;i<=n;i++)
			t[i]=t[i-1]+sqrt((x[i]-x[i-1])*(x[i]-x[i-1])+(y[i]-y[i-1])*(y[i]-y[i-1])+(z[i]-z[i-1])*(z[i]-z[i-1]))/vs;
	double ans=bin(0,t[n]);
	if(t[n]<sqrt((x[n]-px)*(x[n]-px)+(y[n]-py)*(y[n]-py)+(z[n]-pz)*(z[n]-pz))/vp-eps)
	{
		cout<<"NO"<<endl;
		return 0;
	}else
	{
		cout<<"YES"<<endl;
		printf("%.10f\n",ans);
		printf("%.10f %.10f %.10f",xx,yy,zz);	
	}
	return 0;
}
