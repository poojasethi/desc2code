#include <iostream>
#include <cmath>
#include <stdio.h>

using namespace std;
typedef struct{
  double x;
  double y;
  double z;} TPoint;
TPoint p[10010],a,b,c;
long double r(TPoint a, TPoint b)
{
    return sqrt(pow((a.x-b.x),2)+pow((a.y-b.y),2)+pow((a.z-b.z),2));
}

long double t(TPoint a, TPoint b, double v)
{
    return r(a,b)/v;
}


int main()
{ int n,vp,vc,i;
  double tcold,tg,tc=0;
    TPoint potter;

  scanf("%d\n",&n);
  for(int i=0; i<=n;i++)
    {
        scanf("%lf%lf%lf\n",&p[i].x,&p[i].y,&p[i].z);
    }
    scanf("%d%d\n",&vp,&vc);
    scanf("%lf%lf%lf\n",&potter.x,&potter.y,&potter.z);
  if ((potter.x==p[0].x) && (potter.y==p[0].y) &&(potter.z==p[0].z))
  {
   printf("YES\n"); printf("0.00000000\n");  printf("%.10f %.10f %.10f\n",p[0].x,p[0].y,p[0].z);return 0;
  }
     for(i=1; i<=n;i++)
    { tcold=tc;
     tc+=t(p[i-1],p[i],vc);
     tg=t(potter,p[i],vp);
     if (tc==tg) { printf("YES\n"); printf("%.10f\n",tg);  printf("%.10f %.10f %.10f\n",p[i].x,p[i].y,p[i].z); return 0;}
     if (tg<tc) {break;}
    }
//    if (tg>tc) {cout << "NO" ; return 0;};
    if ((abs(tg-tc)>0.0000001) && (tg>tc)) {cout << "NO" ; return 0;};
    a=p[i-1];
    b=p[i];

    while (r(a,b)>0.000000001)
    {
     c.x=(a.x+b.x)/2;c.y=(a.y+b.y)/2;c.z=(a.z+b.z)/2;
     tg=t(potter,c,vp);
     tc=tcold+t(p[i-1],c,vc);
     if (tg<tc) b=c; else a=c;
    }
    printf("YES\n");
    printf("%.10f\n",tg);
     printf("%.6f %.6f %.6f\n",c.x,c.y,c.z);

    return 0;
}
