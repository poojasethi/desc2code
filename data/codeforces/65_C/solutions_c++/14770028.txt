#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
#include<vector>
using namespace std;
int N;
#define MaxN 10050
const double eps=1e-12;
struct ppp{
	double x,y,z;
};
ppp mas[MaxN];
double sqr(double a){
	return a*a;
}

double dist(ppp a,ppp b){
	return sqrt(sqr(a.x-b.x)+sqr(a.y-b.y)+sqr(a.z-b.z));
}

int main(){
	//freopen("input.txt","r",stdin);
	cin>>N;
	for(int i=0; i<N+1; i++)
		cin>>mas[i].x>>mas[i].y>>mas[i].z;
	ppp Harr;
	double Vh,Vs;
	cin>>Vh>>Vs;
	cin>>Harr.x>>Harr.y>>Harr.z;
	double T=0;
	ppp pp;
	double S1,S2,t1,t2;

	for(int i=0; i<N; i++){
		//cout<<T<<" ";
		double a=0,b=1,m;
		while(fabs(b-a)>eps){
			m=.5*(a+b);
			m=1.-m;
			pp.x=mas[i].x*m+mas[i+1].x*(1.-m);
			pp.y=mas[i].y*m+mas[i+1].y*(1.-m);
			pp.z=mas[i].z*m+mas[i+1].z*(1.-m);
			m=1.-m;
			S1=dist(pp,Harr);
			S2=dist(mas[i],pp);
			t1=S1/Vh;
			t2=S2/Vs+T;

			if(t1>t2+eps)
				a=m;
			else
				b=m;
		}
		//cout<<m<<"\n";
		if(m<1-eps){
			cout<<"YES\n";
			printf("%.8lf\n%.8lf %.8lf %.8lf",t1,pp.x,pp.y,pp.z);
			return 0;
		}
		T+=dist(mas[i],mas[i+1])/Vs;
	}
	if(fabs(T-dist(Harr,mas[N])/Vs)<eps){
		cout<<"YES\n";
		printf("%.8lf\n %.8lf %.8lf %.8lf",T,mas[N].x,mas[N].y,mas[N].z);
		return 0;
	}
	cout<<"NO\n";
	return 0;
}
