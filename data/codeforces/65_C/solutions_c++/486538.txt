#include<cstdio>
#include<cmath>

double p[100010][3];
double vp,vs,X0,Y0,Z0;
int n;

double dist(double X0,double Y0,double Z0,double X1,double Y1,double Z1)
{return sqrt((X0-X1)*(X0-X1)+(Y0-Y1)*(Y0-Y1)+(Z0-Z1)*(Z0-Z1));}

int main()
{
    scanf("%d",&n);n++;
    for (int i=0;i<n;i++) scanf("%lf%lf%lf",&p[i][0],&p[i][1],&p[i][2]);
    scanf("%lf%lf",&vs,&vp);
    scanf("%lf%lf%lf",&X0,&Y0,&Z0);
    
    double t=0;
    for (int i=1;i<n;i++)
    if (dist(p[i][0],p[i][1],p[i][2],X0,Y0,Z0)/vs<=t+dist(p[i-1][0],p[i-1][1],p[i-1][2],p[i][0],p[i][1],p[i][2])/vp)
    {
        double D=dist(p[i-1][0],p[i-1][1],p[i-1][2],p[i][0],p[i][1],p[i][2]);
        double l=0,r=D/vp;
        double vX=(p[i][0]-p[i-1][0])/D*vp;
        double vY=(p[i][1]-p[i-1][1])/D*vp;
        double vZ=(p[i][2]-p[i-1][2])/D*vp;

        for (int j=0;j<50;j++)
        {
            double mid=(l+r)/2;
            double X1=p[i-1][0]+mid*vX,Y1=p[i-1][1]+mid*vY,Z1=p[i-1][2]+mid*vZ;
            if (dist(X0,Y0,Z0,X1,Y1,Z1)/vs<=t+mid)
                r=mid;
            else l=mid;
        }
        printf("YES\n%0.9lf\n",t+l);
        printf("%0.9lf %0.9lf %0.9lf\n",p[i-1][0]+l*vX,p[i-1][1]+l*vY,p[i-1][2]+l*vZ);
        return 0;
    } else t+=dist(p[i-1][0],p[i-1][1],p[i-1][2],p[i][0],p[i][1],p[i][2])/vp;
    printf("NO\n");
}
