d, t = map(int, raw_input().split())
a = []
for i in range(d):
    a.append(map(int, raw_input().split()))
    t -= a[-1][0]
if t < 0:
    print "NO"
else:
    res = ""
    for k in a:
        if k[1] - k[0] < t:
            t -= k[1] - k[0]
            res += str(k[1]) + " "
        else:
            res += str(k[0] + t) + " "
            t = 0
            continue
    if t > 0:
        print "NO"
    else:
        print "YES"
        print res[:-1]
