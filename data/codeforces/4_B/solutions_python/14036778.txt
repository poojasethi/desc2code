n, s = map(int, raw_input().split())
a, b = [0]*n, [0]*n
for i in range(n):
	a[i], b[i] = map(int, raw_input().split())
if not (sum(a) <= s and sum(b) >= s):
	print 'NO'
else:
	s -= sum(a)
	for i in range(n):
		tmp = min(s, b[i]-a[i])
		a[i] += tmp
		s -= tmp
		if s == 0:
			break
	print 'YES'
	st = ''
	for i in range(n):
		st += str(a[i]) + ' '
	print st
