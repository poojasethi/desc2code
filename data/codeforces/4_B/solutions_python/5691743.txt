read=lambda:map(int,raw_input().split())
d,st=read()
t1=[0]*d
t2=[0]*d
s1=0
s2=0
for i in range(d):
    t1[i],t2[i]=read()
    s1+=t1[i]
    s2+=t2[i]
if s1>st or s2<st:
    print('NO')
else:
    print('YES')
    delta=st-s1
    ans=[0]*d
    for i in range(d):
        if delta>=t2[i]-t1[i]:
            delta-=t2[i]-t1[i]
            ans[i]=t2[i]
        else:
            ans[i]=t1[i]+delta
            delta=0
    print(' '.join(map(str,ans)))
