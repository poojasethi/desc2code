n, d = map(int, raw_input().split())
a = [map(int, raw_input().split()) for i in range(n)]
s1 = s2 = 0
for i in range(n):
	s1 += a[i][0]
	s2 += a[i][1]
ans = [x[0] for x in a]
if(s1 <= d and d <= s2):
	print 'YES'
	d -= s1
	for i in range(n):
		ans[i] += min(d, a[i][1] - a[i][0])
		d -= min(d, a[i][1] - a[i][0])
	print ' '.join(map(str, ans))
else:
	print 'NO'
