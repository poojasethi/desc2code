from itertools import *
import sys

d, sumTime = map (int, raw_input ().split ())
schedule, minTime, maxTime = list (), [int () for _ in range (d)], [int () for _ in range (d)]
for i in range (d):
    minTime[i], maxTime[i] = map (int, raw_input ().split ())
    schedule += [minTime[i]]
for i, minTimei, maxTimei in zip (range (d), minTime, maxTime):
    schedule[i] += max (0, min (sumTime - sum (schedule), maxTimei - schedule[i]))
print 'YES\n' + ''.join (map (lambda x: str(x) + ' ', schedule)) if sum (schedule) == sumTime else 'NO'
