I=lambda:map(int, raw_input().split())
n,s=I()
a,b=0,0
aa,ab=[],[]
for _ in xrange(n):
    x,y=I()
    a+=x
    b+=y
    aa.append(x)
    ab.append(y)
    
if s>=a and s<=b:
    print 'YES'
    d=s-a
    ans=[]
    for i in xrange(n):
        if aa[i]+d>ab[i]:
            ans.append(ab[i])
            d=d+aa[i]-ab[i]
        else:
            ans.append(aa[i]+d)
            d=0
        
    print ' '.join(map(str,ans))
else:
    print 'NO'
