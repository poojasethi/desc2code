t, s = map(int, raw_input().split())
c = []
d = []
e = 0
for k in range(t):
    p, q = map(int, raw_input().split())
    c.append(p)
    d.append(q-p)
    e += p
if not 0 <= s - e <= sum(d):
    print 'NO'
else:
    s -= e
    for k in range(t):
        p = min(d[k], s)
        s -= p
        c[k] += p
        if s == 0: break
    print 'YES'
    print ' '.join(map(str, c))
