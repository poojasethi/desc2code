def r():
    return map(int,raw_input().split())
d,sm=r()
l = [r() for i in range(0,d)]
mn = sum(x[0] for x in l)
mx = sum(x[1] for x in l)
if mn <= sm <= mx:
    print "YES"
    sm-=mn
    for i in range(0,d):
        dl = l[i][1]-l[i][0]
        print l[i][0]+min(sm,dl),
        sm = max(0,sm-dl)
else:
    print "NO"
