def Map():
    return map(int, raw_input().split())

d, s = Map();
r = [Map() for i in range(0, d)]
low = sum(x[0] for x in r)
high = sum(x[1] for x in r)

if low <= s <= high:
    print "YES"
    s -= low
    for x in r:
        c = x[1] - x[0]
        print x[0] + min(s, c)
        s = max(0, s - c)
else:
    print "NO"
