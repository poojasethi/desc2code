n,k = [int(c) for c in raw_input().split()]
r = [[int(c) for c in raw_input().split()] for i in xrange(n)]
if sum(i for i,j in r) > k or sum(j for i,j in r) < k:
	print 'NO'
else:
	print 'YES'
	rem = k - sum(i for i,j in r)
	i = 0
	while rem > 0:
		r[i][0],rem = r[i][0] + min(rem,r[i][1]-r[i][0]), rem - min(rem,r[i][1]-r[i][0])
		i += 1
	print ' '.join(map(str,[i for i,j in r]))
