d, sumTime = map(int, raw_input().split())
lim = [map(int, raw_input().split()) for _ in xrange(d)]

if sum(v[0] for v in lim) > sumTime or sum(v[1] for v in lim) < sumTime:
    print "NO"
else:
    print "YES"
    result = [v[0] for v in lim]
    left = sumTime - sum(result)
    pos = -1
    while left > 0:
        pos += 1
        new = min(result[pos] + left, lim[pos][1])
        left -= new - result[pos]
        result[pos] = new
    print ' '.join(map(str, result))
