d,t = map(int,raw_input().split())
mins=0
maxs=0
a=[[0,0] for x in range(d)]
for i in range(d):
	a[i][0],a[i][1]=map(int,raw_input().split())
	mins+=a[i][0]
	maxs+=a[i][1]
if t >= mins and t <= maxs:
	print "YES"
else:
	print "NO"
	exit(0)
pat=[a[i][1] for i in range(d)]
tot=sum(pat)
for i in range(d):
	if tot != t:
		pat[i] -= min(tot-t,a[i][1]-a[i][0])
		tot -= min(tot-t,a[i][1]-a[i][0])
	else:
		break
print ' '.join(map(str,pat))
