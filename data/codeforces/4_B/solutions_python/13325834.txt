I = lambda: map(int, raw_input().split())
d, sumTime = I()
days = [I() for _ in xrange(d)]

# sum of all min hours
taken = [d[0] for d in days]
tot = sum(d[0] for d in days)

for i, d in enumerate(days):
    if tot >= sumTime:
        break
    diff = sumTime - tot
    m = d[1] - d[0]
    md = min(diff, m)
    tot += md
    taken[i] += md

if tot == sumTime:
    print 'YES'
    print ' '.join(map(str, taken))
else:
    print 'NO'
