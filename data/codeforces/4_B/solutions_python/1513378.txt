n,d = map(int, raw_input().split())
t = [map(int, raw_input().split()) for _ in range(n)]
s1 = s2 = 0

for i in range(n):
	s1 += t[i][0]
	s2 += t[i][1]

ans=[x[0] for x in t]
if s1 <= d and d <= s2:
	print 'YES'
	d -= s1
	for i in range(n):
		ans[i] += min(d, t[i][1]-t[i][0])
		d -= min(d, t[i][1]-t[i][0])
	print ' '.join(map(str, ans))
else:
	print 'NO'
