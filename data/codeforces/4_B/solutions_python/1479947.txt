def Map():
    return map(int, raw_input().split())

d, s = Map()
r = [Map() for i in range(0, d)]
low = sum(x[0] for x in r)
high = sum(x[1] for x in r)

if low <= s <= high:
    print "YES"
    s -= low
    for i in range(0, d):
        c = r[i][1] - r[i][0]
        print r[i][0] + min(s, c)
        s = max(0, s - c)
else:
    print "NO"
