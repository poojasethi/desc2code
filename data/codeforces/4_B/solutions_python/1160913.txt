n,s=map(int,raw_input().split())
A=[ map(int,raw_input().split()) for i in range(n) ]
if sum( [x[0] for x in A] ) > s or sum( [x[1] for x in A] ) < s:
    print 'NO'
    exit()
B=[x[0] for x in A]
t=sum(B)
while t<s:
    for i,x in enumerate(A):
        t+= x[1]-x[0]
        if t>s:
            B[i]=x[1]-t+s 
            t=s
        else:
            B[i]=x[1]
print 'YES'
