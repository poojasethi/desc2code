n,m = map(int, raw_input().split())
s = []
p = []
for i in range(n):
	a,b = map(int, raw_input().split())
	s.append(a)
	p.append(b)
if m < sum(s) or m > sum(p):
	print 'NO'
	exit()
print 'YES'
y = m-sum(s)
ans = ''
for i in range(n):
	ans += str(s[i]+min(y, p[i]-s[i]))+' '
	y = max(0, y-p[i]+s[i])
print ans
