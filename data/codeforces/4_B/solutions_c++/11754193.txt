#include<iostream>
using namespace std;
int d,s,mi[31],ma[31],l=0,r=0,i;
int main(){
	cin>>d>>s;
	for(i=0;i<d;i++){
		cin>>mi[i]>>ma[i];
		l+=mi[i];
		r+=ma[i];
	}
	if(l<=s&&r>=s){
		cout<<"YES"<<endl;
		for(i=0;i<d;i++){
			while(ma[i]>mi[i]&&r>s){
				r--;
				ma[i]--;
			}
		}
		for(i=0;i<d;i++)
			cout<<ma[i]<<" ";
		cout<<endl;
	}
	else
		cout<<"NO"<<endl;
	return 0;
}
