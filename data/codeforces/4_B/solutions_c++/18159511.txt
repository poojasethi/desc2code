#include<bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,k,sum,d;
	cin>>d>>sum;
int a[d],b[d],p1=0,p2=0;
for(i=0;i<d;i++)
{
	cin>>a[i]>>b[i];
	p1+=a[i];
	p2+=b[i];
}
if(p1<=sum && sum<=p2)
{
sum-=p1;
for(i=0;i<d;i++)
{
if(b[i]-a[i]<sum)
{
sum-=b[i]-a[i];
a[i]=b[i];
}
else
{
a[i]+=sum;
break;
}

}
cout<<"YES"<<"\n";
for(i=0;i<d;i++) cout<<a[i]<<"\n";
}
else
cout<<"NO";
return 0;
}
