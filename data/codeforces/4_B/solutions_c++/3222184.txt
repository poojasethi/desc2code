#include<cstdio>
#define N 1001
int main(){
	int d,s,mi[N],ma[N];
	scanf("%d%d", &d,&s);
	int l=0,r=0;
	for (int i=0; i<d; i++ ){
		scanf("%d%d", mi+i,ma+i);
		l+= mi[i]; r+=ma[i];
	}
	if ( l<=s && s<=r){
		for (int i=0; i<d; i++ )
			while ( ma[i] > mi[i] && r>s ){r--;ma[i]--;}
		puts("YES");
		for (int i=0; i<d; i++ ) printf("%d ", ma[i]);
	}
	else{puts("NO");}
	return 0;
}
