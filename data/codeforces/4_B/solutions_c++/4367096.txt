#include <iostream>
using namespace std;
int main()
{   int q,w,e,r,a[30],s[30];
    cin>>q>>w;
    for(e=0;e<q;e++){
    cin>>a[e]>>s[e];
    w-=a[e];}
    if(w<0){
    cout<<"NO";
    return 0;}
    for(e=0;e<q;e++){
    r=min(w,s[e]-a[e]);
    a[e]+=r;
    w-=r;}
    if(w>0)cout<<"NO";
    else{
    cout<<"YES\n";
    for(e=0;e<q;e++)
    cout<<a[e]<<" ";}
    return 0;
}
