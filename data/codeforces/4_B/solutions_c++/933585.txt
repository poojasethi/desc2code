#include <iostream>
using namespace std;

int main ()
{
    int n,b,sum1=0,sum2=0;
    int q[100],w[100];
    cin>>n>>b;
    for(int i=0;i<n;i++){
    cin>>q[i]>>w[i];
    sum1+=q[i];sum2+=w[i];
    }
    if(b<=sum2 && b>=sum1){
    for(int i=0;i<n;i++){
    while(q[i]<w[i] && sum2>b){
    sum2--;w[i]--;
    }
    }
    cout<<"YES\n";
    for(int i=0;i<n;i++){
    cout<<w[i]<<" ";
    }
    }else cout<<"NO";
    return 0;
}
