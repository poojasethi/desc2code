#include <cstdio>

using namespace std;

int main() {
	int d,sm;
	scanf("%d%d",&d,&sm);
	int l[32],r[32],a[32];
	for(int i=0;i<d;i++) {
		scanf("%d%d",l+i,r+i);
		sm-=l[i];
	}
	for(int i=0;i<d;i++) {
		if (sm>r[i]-l[i]) a[i]=r[i], sm-=r[i]-l[i];
		else if (sm>=0) a[i]=l[i]+sm, sm=0;
	}
	if (sm > 0 || sm < 0) printf("NO\n");
	else {
		printf("YES\n");
		for(int i=0;i<d;i++) printf("%d%c",a[i],i==d-1?'\n':' ');
	}
	return 0;
}
