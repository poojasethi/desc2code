#include<iostream>
using namespace std;
int main()
{
	int n,t,i;
	cin>>n>>t;
	int a[100],b[100];
	for(i=0;i<n;i++)
	{
		cin>>a[i]>>b[i];
		t-=a[i];
	}
	if(t>0)for(i=0;i<n;i++)
		if(b[i]-a[i]>t){
			a[i]+=t;
			t=0;
		}
		else{
			t-=b[i]-a[i];
			a[i]=b[i];
		}
	if(t!=0)
	{
		cout<<"NO";
		return 0;
	}
	cout<<"YES"<<endl;
	if(!t)for(i=0;i<n;i++)cout<<a[i]<<" ";
	return 0;
}
