#include <stdio.h>

int main(void) {
	int n, t, i;
	scanf("%d%d", &n, &t);
	int a[100], b[100];
	for(i=0;i<n;i++) {
		scanf("%d%d", &a[i], &b[i]);
		t -= a[i];
	}
	if(t>0)
	for(i=0;i<n;i++)
		if(b[i]-a[i]>t){
			a[i]+=t;
			t=0;
		} else {
			t-=b[i]-a[i];
			a[i]=b[i];
		}
	puts(t?"NO":"YES");
	if(!t)
		for(i=0;i<n;i++)
		printf("%d%c", a[i],i==n-1?'\n':' ');
	return 0;
}
