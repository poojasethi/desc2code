#include <cstdio>

using namespace std;

#define N 1000

int n,sum,L,R,l[N+10],r[N+10];

int main(){
	scanf("%d%d",&n,&sum);
	for (int i=1;i<=n;++i) scanf("%d%d",&l[i],&r[i]),L+=l[i],R+=r[i];
	if (L>sum || R<sum){printf("NO\n");return 0;}
	printf("YES\n");
	for (int i=1;i<=n;++i)
		if (r[i]-l[i]<sum-L) L+=r[i]-l[i],printf("%d ",r[i]);
		else printf("%d ",l[i]+sum-L),L=sum;
	puts("");
	return 0;
}
