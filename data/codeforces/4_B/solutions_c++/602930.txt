#include<cstdio>
int D,S;
int a[31],b[31],c[31],sa,sb;
int main(){
scanf("%d%d",&D,&S);
for(int i=0;i<D;i++){
scanf("%d%d",&a[i],&b[i]);
sa+=a[i],sb+=b[i];
}
if(S<sa||S>sb)puts("NO");
else{
for(int i =0;i<D;i++){
S-=a[i];
c[i]=a[i];
}
for(int i=0;i<D&&S;i++){
int t=b[i]-a[i];
if(t>S)c[i]+=S,S=0;
else c[i]=b[i], S-=t;
}
puts("YES");
printf("%d",c[0]);
for(int i=1;i<D;i++)printf(" %d", c[i]);
putchar('\n');
}
}
