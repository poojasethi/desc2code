
#include<bits/stdc++.h>
using namespace std;
int arr[31],arr1[31];
int  main()
{
    int s=0,s1=0,n,ss;cin>>n>>ss;
    for(int i=0;i<n;i++)cin>>arr[i]>>arr1[i],s+=arr1[i],s1+=arr[i];if(ss>s||ss<s1){cout<<"NO";return 0;}ss-=s1;
    for(int i=0;i<n;i++)if(ss){int d=arr1[i]-arr[i];if(d>ss)arr[i]+=ss,ss=0;else if(ss>=d)arr[i]+=d,ss-=d;}
    if(ss){cout<<"NO";return 0;}cout<<"YES\n";
    for(int i=0;i<n;i++)cout<<arr[i]<<" ";

}
