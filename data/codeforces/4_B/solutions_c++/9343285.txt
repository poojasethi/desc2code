#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,s;
scanf("%d%d",&n,&s);
vector<int> a,b,v;
int i,x,y,mi=0,ma=0;
for(i=0;i<n;i++)
{
scanf("%d%d",&x,&y);
a.push_back(x);
b.push_back(y);
mi+=x;
ma+=y;
}
if(mi<=s && ma>=s)
{
printf("YES\n");
int rem = s-mi;
for(i=0;i<n;i++)
{
int pos = b[i]-a[i];
pos = min(pos,rem);
printf("%d ",a[i]+pos);
rem-=pos;
}
}
else printf("NO\n");






return 0;
}
