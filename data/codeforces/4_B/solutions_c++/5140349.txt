#include <iostream> 
using namespace std;
int main (int argc,char *argv[])
{
    int n,k,i,a[1000];
cin>>n>>k;
int sum=0;
int mx=0;
for( i=1;i<=2*n;i+=2)
{
cin>>a[i]>>a[i+1];
sum+=a[i];
mx+=a[i+1];
}
if(mx<k||sum>k)cout<<"NO";
else
{
i=1;
while(sum<k)
{
if(a[i]<a[i+1])
{
a[i]++;
sum++;
}
else
{
i+=2;
}
}
cout<<"YES"<<endl;
for(i=1;i<=2*n;i+=2)cout<<a[i]<<" ";
}
return 0;
}
