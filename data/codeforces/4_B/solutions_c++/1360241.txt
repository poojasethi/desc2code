#include<iostream>

using namespace std;

int n,t,i,s1,s2,mi[100],ma[100];

int main()
{
	cin>>n>>t;
	for(i=0;i<n;i++)
	{
		cin>>mi[i]>>ma[i];
		s1+=mi[i];
		s2+=ma[i];
	}

	if((s1<=t)&&(t<=s2))
	{                    
		cout<<"YES\n";

		t-=s1;
		for(i=0;i<n;i++)
		{
			cout<<min(mi[i]+t,ma[i])<<" ";

			t=max(mi[i]+t-ma[i],0);	
		}
	}
	else cout<<"NO";

	return 0;
}
