#include<iostream>
using namespace std;

int i,t,k,l,j,d,s1,s=0,a[100],b[100];

int main(){
cin>>d; cin>>t;
for(i=1; i<=d; i++) {cin>>a[i]>>b[i]; s=s+a[i]; s1=s1+b[i];}
if(s<=t && t<=s1){ cout<<"YES"<<endl;
for(i=1; i<=d; i++)
{ if (t-b[i]>=s-a[i]) {cout<<b[i]<<' '; t=t-b[i]; s=s-a[i];}
else {cout<<t-s+a[i]<<' '; for(j=i+1; j<=d; j++) cout<<a[j]<<' '; return 0;
}}}
else cout<<"NO";

}
