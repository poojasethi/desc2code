#include<iostream>
using namespace std;
int main()
{int d,st;cin>>d>>st;
int a[d],b[d];int min=0,max=0;
for(int i=0;i<d;i++){
cin>>a[i]>>b[i];min+=a[i];max+=b[i];}
if(st<=max&&st>=min)
{cout<<"YES"<<endl;st-=min;
for(int i=0;i<d;i++)
{if(b[i]-a[i]<=st)
{st=st-b[i]+a[i];a[i]+=(b[i]-a[i]);}
else{a[i]+=st;break;}}
for(int i=0;i<d;i++)cout<<a[i]<<" ";
return 0;}
cout<<"NO"<<endl;}
