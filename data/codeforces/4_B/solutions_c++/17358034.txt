#include <iostream>
using namespace std;
int a[30], b[30];
int main()
{
	int d, s, x, y, i;
	cin>>d>>s;
	for (x=y=i=0; i<d; i++) {
		cin>>a[i]>>b[i];
		x+=a[i]; y+=b[i];
	}
	if (s<x || s>y) cout<<"NO"<<endl;
	else {
		cout<<"YES"<<endl;
		s-=x;
		for (i=0; i<d; i++) {
			y=min(b[i]-a[i], s);
			a[i]+=y; s-=y;
			cout<<a[i]<<" ";
		} 
		cout<<endl;
	}
	return 0;
}
