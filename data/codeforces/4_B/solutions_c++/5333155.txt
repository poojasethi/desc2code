#include <iostream>

using namespace std;

int main()
{int l[35],r[35],a[35];
int n,s,s1=0,s2=0; cin>>n>>s;
for (int i=1;i<=n;i++)
{ cin>>l[i]>>r[i]; s1+=l[i];s2+=r[i];}
if (s<s1||s>s2) {cout<<"NO"; return 0;}
int st=s-s1;
for (int i=1;i<=n;i++)
 { int k=min(st,r[i]-l[i]);
   a[i]=l[i]+k; st-=k;
 }
cout<<"YES"<<endl;
for (int i=1;i<=n;i++) cout<<a[i]<<" ";
return 0;
}
