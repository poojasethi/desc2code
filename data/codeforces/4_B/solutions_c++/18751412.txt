#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,k,i,A[30],B[30],c=0,d=0;
	cin>>n>>k;
	for(i=0;i<n;i++)
    {
		cin>>A[i]>>B[i];
		c+=A[i];
		d+=B[i];
	}
	if((k<c)||(k>d))
	{
		cout<<"NO";
		return 0;
	}
	cout<<"YES\n";
	for(i=0;i<n;i++){
		cout<<min(B[i],k-c+A[i])<<' ';
		k-=min(B[i],k-c+A[i]);
		c-=A[i];
	}
	return 0;
}
