#include <iostream>
using namespace std;
int main(){
	int n,k,a[50][5],b[50],mi=0,ma=0;
	cin>>n>>k;
	for(int i=1;i<=n;i++){
		cin>>a[i][1]>>a[i][2];
		b[i]=a[i][1];
		mi+=a[i][1];
		ma+=a[i][2];
	}
	if (mi<=k&&k<=ma){
		cout<<"YES"<<endl;;
		int s=k-mi;
		for(int i=1;i<=n;i++){
			if (s>0){
				cout<<b[i]+min(s,a[i][2]-b[i])<<' ';;
				s-=min(s,a[i][2]-b[i]);
			}else cout<<b[i]<<' ';
		}
	}else cout<<"NO";
	return 0;
}
