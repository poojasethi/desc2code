#include<iostream>
using namespace std;
int l[30],r[30];
int main() {
	int d,sum,smin=0,smax=0;
	cin>>d>>sum;
	for(int i=0;i<d;i++) {
		cin>>l[i]>>r[i];
		smin+=l[i];
		smax+=r[i];
	}
	if(sum<smin||sum>smax)cout<<"NO";
	else {
		cout<<"YES"<<endl;
		sum-=smin;
		for(int i=0,t=sum;i<d&&t>0;i++,t=sum) {
			t=min(t,r[i]-l[i]);
			l[i]+=t;
			sum-=t;
		}
		for(int i=0;i<d;i++)
			cout<<l[i]<<" ";
	}
	return 0;
}
