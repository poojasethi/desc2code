#include<cstdio>
#include<algorithm>
using namespace std;

int d,t,tt;
int l[33],r[33];

int main(){
	scanf("%d%d",&d,&t);
	tt=t;
	for(int i=0;i<d;i++) {
		scanf("%d%d",&l[i],&r[i]);
		t-=l[i];tt-=r[i];
	}
	if(t<0||tt>0){
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	for(int i=0;i<d;i++){
		int det=min(t,r[i]-l[i]);
		l[i]+=det;t-=det;
	}
	for(int i=0;i<d;i++)
		printf("%d%c",l[i],i<d-1?' ':'\n');
	
	return 0;
}
