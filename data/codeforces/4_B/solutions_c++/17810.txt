#include <iostream>
using namespace std;

const int dmax = 30;
int a[dmax], b[dmax];

int main() {
	int d, s, t = 0, i;
	
	cin >> d >> s;
	for (i = 0; i < d; i++) {
		cin >> a[i] >> b[i];
		s -= a[i];
		t += b[i] - a[i];
	}
	if (s >= 0 && s <= t) {
		cout << "YES\n";
		for (i = 0; i < d; i++) {
			t = min(b[i] - a[i], s);
			cout << a[i] + t << (i + 1 < d ? ' ' : '\n');
			s -= t;
		}
	} else {
		cout << "NO\n";
	}
	
	return 0;
}
