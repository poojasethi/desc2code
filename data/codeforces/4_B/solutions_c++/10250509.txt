#include<bits/stdc++.h>


using namespace std;

long n,k,f,A[555],B[555],C,D;
int main(){
    cin>>n>>k;
    for(f=0;f<n;f++){
        cin>>A[f]>>B[f];
        C+=A[f];
        D+=B[f];
    }
    if((k<C)||(k>D)){
        cout<<"NO";
        return 0;
    }
    cout<<"YES\n";
    for(f=0;f<n;f++){
        cout<<min(B[f],k-C+A[f])<<' ';
        k-=min(B[f],k-C+A[f]);
        C-=A[f];
    }
}
