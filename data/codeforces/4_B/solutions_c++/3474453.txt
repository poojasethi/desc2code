#include<iostream>
using namespace std;
int main(){
	int d,s,t=0,r=0,x[100100],y[100100];
	cin>>d>>s;
	for(int i=1; i<=d; i++) { cin>>x[i]>>y[i]; t+=x[i]; r+=y[i]; }
	if (s>=t and s<=r) {
		s-=t;
		while (1) {
			for(int i=1; i<=d; i++){
				if (x[i]<y[i]) x[i]++, s--;
				if (s==0) {
					cout<<"YES\n";
					for(int i=1; i<=d; i++) cout<<x[i]<<' ';	
					return 0;
				}
			}
		}
	} else cout<<"NO";
}
