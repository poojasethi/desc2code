#!/usr/bin/python2.7
def samecnt(str):
	ret=0
	for i in range(len(str)-1):
		if str[i]==str[i+1]:
			ret += 1
	return ret
def diffcnt(str):
	ret=0
	for i in range(len(str)-1):
		if str[i]!=str[i+1]:
			ret += 1
	return ret
n,m=map(int,raw_input().split())
s1=s2=''
n1=n2=n
m1=m2=m

s1='a';n1-=1
while n1>0 and m1>0:
	if s1[-1]=='a':
		s1+='b';m1-=1
		if m1>0:s1+='b';m1-=1
	else:
		s1+='a';n1-=1
		if n1>0:s1+='a';n1-=1
while n1>0: s1+='a';n1-=1
while m1>0: s1+='b';m1-=1
p1,v1=samecnt(s1),diffcnt(s1)

s2='b';m2-=1
while n2>0 and m2>0:
	if s2[-1]=='a':
		s2+='b';m2-=1
		if m2>0:s2+='b';m2-=1
	else:
		s2+='a';n2-=1
		if n2>0:s2+='a';n2-=1
while n2>0: s2+='a';n2-=1
while m2>0: s2+='b';m2-=1
p2,v2=samecnt(s2),diffcnt(s2)
if p1>p2:print p1,v1
else:print p2,v2
