def cnt_result(a, b):
	if b < a:
		res = [b - 1, b + (b % 2)]
		a -= b + (b % 2)
		res[0] += a
	elif b > a:
		res = [a - 1, a + 1 - (a % 2)]
		b -= a + 1 - (a % 2)
		res[0] += b
	else:
		res = [a - 1, a]
	return res

n, m = map(int, raw_input().split())
res1, res2 = cnt_result(n, m), cnt_result(m, n)
# print res1, res2
if res1[0] > res2[0] or (res1[0] == res2[0] and res1[1] < res2[1]):
	print res1[0], res1[1]
else:
	print res2[0], res2[1]
