n,m=map(int,raw_input().split())
def cal(last):
	x,y,a,b=n,m,0,0
	if last==0:
		x-=1
	else:
   		y-=1
	for i in xrange(n+m-1):
		if x==0 :
			now = 1
		elif y==0:
			now = 0
		elif i % 2 == 0:
			now = last^ 1
		else:
			now = last
		if now == 0:
			x-=1
		else:
			y-=1
		if now == last:
			a += 1
		else :
			b += 1
		last = now
	return [a,b]
x = cal(0)
y = cal(1)
if x[0] > y[0] or (x[0]==y[0] and x[1] < y[1]):
	print x[0],x[1]
else:
	print y[0],y[1]
