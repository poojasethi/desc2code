#!/usr/bin/python2
# -*- coding: utf-8 -*-

import sys
import os

def val(s):
    r1 = 0
    r2 = 0
    for i in xrange(1, len(s)):
        if s[i] != s[i-1]:
            r2 += 1
        else:
            r1 += 1
    return r1, r2

def solve(n, m):
    r = ""
    n -= 1
    r += "x"
    while n > 0 and m > 0:
        if m >= 2:
            r += "yy"
            m -= 2
        else:
            r += "y"
            m -= 1
        if n >= 2:
            r += "xx"
            n -= 2
        else:
            r += "x"
            n -= 1
    if n > 0:
        r += "x" * n
    else:
        r += "y" * m

    return val(r)

def main():
    n, m = map(int, sys.stdin.readline().strip().split())
    s1 = solve(n, m)
    s2 = solve(m, n)

    if (s1[0] > s2[0] or (s1[0] == s2[0] and s1[1] < s2[1])):
        print s1[0], s1[1]
    else:
        print s2[0], s2[1]

if __name__ == '__main__':
    main()
