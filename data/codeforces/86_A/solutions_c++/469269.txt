#include <iostream>
#include <cmath>

using namespace std;

typedef long long LL;

int main() {
    int x , y;
    cin >> x >> y;
    int ty = y;
    LL t = 0;
    while ( ty ) {
        ty /= 10;
        t = t*10+9;
    }
    if ( y < t/2 ) cout << y*(t-y) << endl;
    else if ( x < t/2 ) cout << t/2*(t-t/2) << endl;
    else cout << x*(t-x) << endl;
    return 0;
}
