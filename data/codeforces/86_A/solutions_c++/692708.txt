#include <iostream>
using namespace std;

int main()
{
    long long l,r;
    cin>>l>>r;
    long long d=10;

    for(d=10; d<=r; d=d*10);
    if((d-1)/2>r)
        cout<<r*(d-1-r)<<endl;
    else if((d-1)/2<l)
        cout<<l*(d-1-l)<<endl;
    else
        cout<<(d-1)*(d-1)/4<<endl;

    return 0;
}
