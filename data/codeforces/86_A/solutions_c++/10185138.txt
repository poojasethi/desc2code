#include<bits/stdc++.h>
#define LL long long
using namespace std;
LL l,r;
LL d1[105],d2[105];
int main()
{
	while(cin>>l>>r)
	{
		LL n;
		d1[0]=d2[0]=0;
		n=r;
		while(n){d2[++d2[0]]=n%10;n/=10;}
		LL ans=0,tmp=0;
        for(int i=d2[0];i>=1;i--)if(i==d2[0])ans+=5;else ans*=10;
        for(int i=d2[0];i>=1;i--)tmp=tmp*10+9;
        LL ret=max(l*(tmp-l),r*(tmp-r));
        if(ans>r||ans<l)cout<<ret<<endl;
        else cout<<max(ans*(tmp-ans),ret)<<endl;
	}
	return 0;
}
