#include<iostream>
#include<cmath>
using namespace std;

typedef long long LL;

LL mi(int k){
    LL ans=1L;
    while(k--) ans*=10;
    return ans;
}

int main(){
    LL a, b;
    cin>>a>>b;
    LL k=b, l=0L;
    for(;k;k/=10) l++;
    LL m=(LL)(pow(10,l-1)+0.5)*5, ans;
    if(a<=m-1 && b>=m-1 || a<=m && b>=m) ans=m;
    else ans=fabs(a-m)<fabs(b-m)?a:b;
    cout << (LL)ans*((LL)(pow(10,l)+0.5)-1-ans) << endl;
}
