#include <cstdio>
#include <iostream>
#include <map>
#include <vector>
#include <string.h>
#include <cmath>
#include <algorithm>

using namespace std;

typedef __int64 LL;

LL L,R;

int main()
{
   while(scanf("%I64d%I64d",&L,&R)!=EOF)
   {
       LL U=9LL;
       while(U<R) U=U*10LL+9LL;
       LL Mid=U/2;
       LL ans;
       if(Mid>=R) ans=(U-R)*R;
       else if(Mid<=L) ans=(U-L)*L;
       else ans=(U-Mid)*Mid;

       printf("%I64d\n",ans);
   }
   return 0;
}
