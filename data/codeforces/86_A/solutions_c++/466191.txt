#include <cstdio>
#include <cstring>

int l, r;
long long b;
long long res;

int main() {
	scanf("%d%d", &l, &r);
	b = 1;
	for (; r / b >= 10; b *= 10);
	b *= 5;
	if (l <= b && b <= r) {
		res = b;
	} else if (r < b) {
		res = r;
	} else {
		res = l;
	}
	printf("%I64d\n", res * (2 * b - 1 - res));
	return 0;
}
