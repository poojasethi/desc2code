#include <iostream>

using namespace std;

int main()
{
	long long l, r;
	cin >> l >> r;
	long long from = 1, to = 9;
	long long best = 0;
	while (1) {
		if (from > r) break;
		if (l <= to) {
			long long mid = to / 2;
			long long curbest = 0;
			if (mid > r) {
				curbest = r;
			}
			else if (mid < l) {
				curbest = l;
			}
			else {
				curbest = mid;
			}
			best = max(best, curbest * (to - curbest));
		}
		from *= 10;
		to = 9 + to * 10;
	}
	cout << best << endl;
	return 0;
}
