#include <iostream>
using namespace std;
int main()
{
    unsigned long long l,r,max=0,med,temp;
    cin>>l>>r;
    temp=r;
    for(;temp>0;temp/=10)
        max=max*10+9;
    med=max/2;
    if((med>=l&&med<=r)||(med+1>=l&&med+1<=r))
        cout<<med*(max-med);
    else    if(med>r)
        cout<<r*(max-r);
    else    if(l>med+1)
        cout<<l*(max-l);
    return 0;
}
