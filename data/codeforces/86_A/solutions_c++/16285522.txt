#include<bits/stdc++.h>
using namespace std;
long long n,m,s,k,i,x,l,y;
int main()
{
    cin>>n>>m;
    x=m;k=0;
    while(x>0)
    {
        x/=10;
        k++;
    }
    l=1;
    for(i=1;i<k;i++)
        l*=10;
    if(l<n) l=n;
    s=4;
    for(i=1;i<k;i++)
        s=s*10+9;
    y=9;
    for(i=1;i<k;i++)
        y=y*10+9;
    if(s>m)
    {
        cout<<m*(y-m);
    } else
    {
        if(s>=l) cout<<s*(y-s);else
        cout<<l*(y-l);
    }
    return 0;
}
