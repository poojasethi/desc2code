#include <iostream>
using namespace std;

typedef long long int64;

void calc(int64 v, int64& val, int64& maxval)
{
	v /= 10;
	val = 4;
	maxval = 9;
	while (v)
	{
		v /= 10;
		val = val * 10 + 9;
		maxval = maxval * 10 + 9;
	}
}

int main()
{
	int64 l, r;
	cin >> l >> r;
	int64 val, maxval;
	calc(r, val, maxval);
	if (l <= val && val <= r)
		cout << (maxval - val) * val << endl;
	else if (val < l)
		cout << (maxval - l) * l << endl;
	else if (val > r)
		cout << (maxval - r) * r << endl;
}
