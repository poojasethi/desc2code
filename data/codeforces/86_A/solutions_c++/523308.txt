#include <iostream>

#define max(a, b) ((a) >= (b) ? (a) : (b))

int main()
{
	int l, r;
	std::cin >> l >> r;

	long long s = 1;
	for (int rs = r; rs > 0; s *= 10, rs = rs / 10);
	--s;

	long long m = max(l * (s - l), r * (s - r));
	if (l <= s / 2 + 1 && s / 2 <= r)
		m = max(m, (s / 2) * (s - s / 2));
	std::cout << m;
	return 0;
}
