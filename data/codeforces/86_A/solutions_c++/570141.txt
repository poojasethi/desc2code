#include <iostream>

using namespace std;

int main()
{
    long long l,r,ans,t,s=0,n;
    cin>>l>>r;
	t=r;
	while(t>0)
	{
		t/=10;
		s=s*10+9;
	}
	//cout<<s<<endl;
	n=s/2;
	if(r>n)
	{
		if(l>n)
		{
			ans=(s-l)*l;
		}
		else
		{
			ans=(s-n)*n;
		}
	}
	else
	{
		ans=(s-r)*r;
	}
	cout<<ans<<endl;
}
