#include<stdio.h>
#include<algorithm>
using namespace std;
int main(){
	long long int L,R,ans;
	ans=0;
	scanf("%I64d%I64d",&L,&R);
	for(long long int i=1; i<=R; i*=10){
		long long int b=i*10/2,lef=max(L,i),rig=min(R,i*10-1),w;
		if(lef<=b && b<=rig)w=b;else
		if(rig<b)w=rig;else
			w=lef;
		ans=max(ans,w*(i*10-1-w));
	}
	printf("%I64d\n",ans);
	return 0;
}
