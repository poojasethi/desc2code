#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    __int64 a,b,n,h;
    while(scanf("%I64d%I64d",&a,&b)!=EOF)
    {
        n=b;
        h=0;
        while(n)
        {
            n/=10;
            h=h*10+9;
        }
        if(b<h/2)
            printf("%I64d\n",b*(h-b));
        else 
        {
            if(a<h/2)printf("%I64d\n",(h*h-1)/4);
            else printf("%I64d\n",a*(h-a));
        }
    }
    return 0;
}
