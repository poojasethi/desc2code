#include <cstdio>
#include <cmath>
#include <iostream>
using namespace std;
int main()
{
  int l,r;long long ans=0,m=0;
  scanf("%d%d",&l,&r);
  m=9LL;while(m<r)m*=10,m+=9;
  int L=l,R=r,M;
  while(R-L>1)
  {
    M=L+R>>1;
    if(M<<1>m)R=M;else L=M;
  }
  m=9LL;while(m<R)m*=10,m+=9;
  ans=(long long)R*(m-R);
  m=9LL;while(m<L)m*=10,m+=9;
  ans=max(ans,(long long)L*(m-L));
  cout<<ans;
}
