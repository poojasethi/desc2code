#include<iostream>
#include<algorithm>
using namespace std;

typedef long long ll;
//typedef long long double lld;

int main()
{
    ll l, r, d;

    cin>>l>>r;
    for(ll i=10;i<=10000000000;i*=10)
    {
        if(r<i)
        {
            d=i-1;
            break;
        }
    }

    if(d/2<=l)
    {
        cout<<l*(d-l)<<endl;
    }
    else if(d/2<=r)
    {
        cout<<d/2*(d-d/2)<<endl;
    }
    else
        cout<<r*(d-r)<<endl;

    return 0;
}
