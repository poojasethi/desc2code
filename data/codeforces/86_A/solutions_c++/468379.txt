#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
#include <sstream>
#include <cstdio>
#include <cmath>
#include <string.h>
#include <queue>
using namespace std;

int main()
{
    long long l, r, dec = 1, w, y;

    cin >> l >> r;

    dec = pow(10., (int)log10(r) + 1), y = dec-- / 2;

    w = r * (dec - r);
    if (r / l == 1) w = max(w, l * (dec - l));
    if (y >= l && y <= r) w = max(w, y * (dec - y));
    cout << w << endl;

    return 0;
}
