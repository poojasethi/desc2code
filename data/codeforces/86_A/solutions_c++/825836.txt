#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    __int64 ans=0,a,b,ten,sum;
    scanf("%I64d%I64d",&a,&b);
    for(ten=1;ten<=b;ten*=10);
    sum=ten-1;
    ten/=2;
    if(ten>b||ten<a)
    {
        ans=max(a*(sum-a),b*(sum-b));
    }
    else ans=ten*(sum-ten);
    printf("%I64d\n",ans);
    return 0;
}
