#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;

ll get(ll a) {
	ll t = 10;
	for (int i = 0; t <= a; ++i) {
		t *= 10;
	}
	return (t - a - 1) * a;
}
int main() {
	ll a, b;
	cin >> a >> b;
	ll mx = 0;
	mx = max(mx, get(a));
	mx = max(mx, get(b));
	ll t = 5;
	for (int i = 0; i < 11; ++i) {
		if (t >= a && t <= b) {
			mx = max(mx, get(t));
		}
		t *= 10;
	}
	cout << mx;
}
