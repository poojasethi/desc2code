#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

int x,y;
long long out;

long long ref(int z) {
    for (long long i=1; ; i*=10)
        if (i>z)
            return (i-1)-z;
}

int main() {
    scanf("%d%d",&x,&y);
    out=max(x*ref(x),y*ref(y));
    for (long long i=1; i<=y; i*=10)
        if ((x<=5*i)&&(5*i-1<=y)) 
            out=max(out,5*i*(5*i-1));
    cout << out << endl;
    return 0;
}
