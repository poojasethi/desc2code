#include<cstdio>
#include<algorithm>
#include<iostream>
using namespace std;

long long work(long long a)
{
    long long b=0,p=1;
    for (long long i=a;i;i/=10) {b+=(9-i%10)*p;p*=10LL;}
    return a*b;
}

int main()
{
    int L,R;
    scanf("%d%d",&L,&R);

    int a=0,b=0;
    long long p=1;

    for (int i=L;i;i/=10) a++;
    for (int i=R;i;i/=10) b++,p*=10LL;
    if (a<b) L=p/10;
    if (L<=p/2 && R>=p/2) cout << work(p/2) << endl;
    else cout << max(work(L),work(R)) << endl;
}
