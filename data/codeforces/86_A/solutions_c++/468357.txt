#include <iostream>
#include <cstring>
#include <cmath>
#include <string>
#include <algorithm>
#define inf 1000000000
using namespace std;
typedef long long L;
typedef unsigned long long UL;
L l,r,cur,t;

int main()
{
    int s=0;
    cur=1;
    cin>>l>>r;

    while(cur<=r){cur*=10;s++;}
    t=cur-1;
    cur/=10;s--;
    if(cur>l)l=cur;

    if(cur*5>=l&&cur*5<=r){cout<<cur*5*(t-cur*5)<<endl;}
    else {cout<<max(l*(t-l),r*(t-r))<<endl;}

    return 0;
}
