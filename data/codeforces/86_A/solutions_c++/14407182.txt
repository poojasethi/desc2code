#include <cmath>
#include <iostream>

using namespace std;

int main()
{
	long long a, b;
	cin >> a >> b;
	long long x = 1;
	while (x <= b)
		x *= 10;
	//cout << "x: " << x << endl;
	long long p = max(x / 2, a);
	long long m = min(p, b);
	long long n = (x - m - 1);
	//cout << m << " " << n << endl;
	cout << m * n << endl;
	return 0;
}
