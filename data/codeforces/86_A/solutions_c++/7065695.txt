#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

int main()
{
    unsigned  long long l, r, k;
    cin >> l >> r;
    for (unsigned long long i=1; i<=1e10; i=i*10+1)
    {
        if (9*i>=r)
        {
            k=9*i;
            break;
        }
    }
    if (k/2<=r && k/2>=l)
    {
        cout << (k/2)*(k-(k/2)) << endl;
        return 0;
    }
    cout << max(l*(k-l),r*(k-r)) << endl;
}
