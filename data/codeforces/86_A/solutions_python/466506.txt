#!python

from sys import stdin

def mirror(x):
   res = 0
   pow10 = 1
   while x != 0:
      d = x % 10
      res += (9-d) * pow10
      x /= 10
      pow10 *= 10
   return res

nL, nR = map(int, stdin.readline().split())
sL = str(nL)
sR = str(nR)

p10 = 1
for i in xrange(len(sR)-1):
  p10 *= 10

if len(sL) < len(sR):
  nL = p10

half = 5*p10

if nL > half:
   print nL * mirror(nL)
elif nR < half:
   print nR * mirror(nR)
else:
   print half * mirror(half)
