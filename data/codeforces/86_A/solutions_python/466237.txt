
#some convinient function


def readints():
  return map(int,raw_input().split())

def rep(c1,c2,k):
  return"".join([c1,c2][i%2]*2 for i in xrange(k))

def comp(mi,ma):
  k=1
  while k<=ma:
    k*=10
  if mi<=k/2<=ma:
    return (k/2)*(k/2-1)
  else:
    return max(mi*(k-1-mi), ma*(k-1-ma))

def main():
  mi,ma=readints()
  print comp(mi,ma)


if __name__=='__main__':
  main()
