length = input()
coins = map(int, raw_input().split())
reverse = False
l = 0
r = 0
for k, coin in enumerate(coins):
    if k + 1 != coin:
        if l == 0:
            l = k + 1
        else:
            r = k + 1
new = coins[:l-1] + list(reversed(coins[l-1:r])) + coins[r:]
for k in xrange(length):
    if new[k] != k + 1:
        print "0 0"
        break
