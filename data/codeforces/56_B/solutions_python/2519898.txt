n = int(raw_input())
data = map(int, raw_input().split())
data = [0] + data
n = len(data)
if n > 2:
	l = 0
	for i in range(1, n):
		if data[i] - data[i - 1] != 1:
			l = i
			break
	if l > 0:
		for r in range(l + 1, n):
			if data[r] == data[l - 1] + 1:
				x = data[l:r + 1]
				x.reverse()
				t = data[:l] + x + data[r + 1:]
				if [t[i] - t[i - 1] for i in range(1, len(t))].count(1) == n - 1:
					print str(l) + " " + str(r)
					exit(0)
print "0 0"
