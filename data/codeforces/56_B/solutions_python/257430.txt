from string import split

n = int(raw_input())
A = map(int, split(raw_input()))

l = 0
r = 0
f = 0
next = 1

for i in range(n):
	if f == 0:
		if A[i] != next:
			l = i + 1
			f = 1
			next = A[i] - 1
		else:
			next += 1
	elif f == 1:
		if A[i] != next:
			r = i
			f = 2
			next = A[i] + 1
		else:
			next -= 1
	elif f == 2:
		if A[i] != next:
			l = 0
			r = 0
		else:
			next += 1
	
	if l != 0 and r == 0:
		r = n
			
print l, r
