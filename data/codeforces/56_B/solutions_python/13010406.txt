p = map(int,raw_input().split())

i = 0
d = []
flag = False
while i < len(p):
    if p[i] != i+1:
        if (p[i] < i+1):
            flag = True
            break
        d.append((i,p[i]))
        i = p[i]
    else: i+=1


if flag or len(d) != 1:
    print "0 0"
else:
    i,e = d[0]
    if sorted(p) == p[:i] + list(reversed(p[i:e])) + p[e:]:
        print "%d %d"%(i+1,e)
    else:
        print "0 0"
