n = input()
A = map(int, raw_input().split())
i = 0
while i < n and A[i] == i + 1: i += 1
j = n - 1
while j >= 0 and A[j] == j + 1: j -= 1
if i < j:
    found = False
    for k in xrange(i, j + 1):
        if A[k] + k + 1 != i + j + 2:
            found = True
            break
    if not found: print i + 1, j + 1
    else: print 0, 0
else: print 0, 0
