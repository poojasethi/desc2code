n = int(raw_input())
x = [int(x) for x in raw_input().split()]

i = 0

while i < n and i == x[i] - 1:
    i += 1
if i == n:
    i -= 1

k = x[i: x[i]]

k.reverse()

y = x[: i]
y.extend(k)
y += x[x[i]: ]

i = 0

while i < n and i == y[i] - 1:
    i += 1

if i == n and len(k) > 1:
    print k[0], k.pop()
else:
    print 0, 0
