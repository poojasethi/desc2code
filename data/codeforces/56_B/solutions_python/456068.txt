#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    n = int(raw_input())
    lst = map(int, raw_input().split())
    l = r = 0
    for i in xrange(n):
        if lst[i] != i+1:
            l = i+1
            break
    if l == 0:
        print 0, 0
        return 0
    for i in xrange(l, n-1):
        if lst[i] != lst[i-1] - 1:
            r = i
            break
    if r == 0:
        r = n
    rev = lst[l-1:r]
    rev.reverse()
    if lst[:l-1] + rev + lst[r:] == range(1,n+1):
        print l, r
    else:
        print 0, 0
    return 0

if __name__ == '__main__':
    main()
