def isCountinuous(sequence, difference):
	next_item = sequence[0]

	if not abs(next_item)%2:
		sequence.insert(len(sequence)/2, 0)

	for i in sequence:
		if i != next_item:
			return False
		else:
			next_item += difference
	return True

def main():
	n = input()

	data = [int(x) for x in raw_input().split()]
	default = range(1,n + 1)

	if data == default:
		print 0, 0
		return
	
	diff = []
	for i in range(n):
		if default[i] != data[i]:
			diff.append((i,data[i] - default[i]))

	numbers = [i[1] for i in diff]	
	numbers2 = list(reversed([-i for i in numbers]))

	if numbers == numbers2 and isCountinuous(numbers, -2):
		print diff[0][0] + 1, diff[-1][0] + 1
	else:
		print 0, 0
main()
