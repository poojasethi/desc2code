n = input()
a = map(int, raw_input().split())
l = -1
for i in range(n):
  if a[i] != i + 1:
    l = i
    break
r = -1
for i in reversed(range(n)):
  if a[i] != i + 1:
    r = i
    break
if l == -1 or l == r:
  print 0, 0
  exit()
i, j = l, r
while i < j:
  if a[i] != j + 1 or a[j] != i + 1:
    print 0, 0
    exit()
  i += 1
  j -= 1
print l + 1, r + 1
