#!/usr/bin/python

n = int(raw_input())
a = map(lambda x: int(x), raw_input().split())
for i in range(n):
    if a[i] != i+1:
        break
else:
    i += 1
for j in range(n-1, -1, -1):
    if a[j] != j+1:
        break
else:
    j -= 1
if i == n:
    print "0 0"
    exit(0)
assert j > i
for k in range(i, j+1, 1):
    if a[k] != j+i-k+1:
        break
else:
    print i+1,j+1
    exit(0)
print "0 0"
exit(0)
