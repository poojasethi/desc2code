#!/usr/bin/env python

n = int(raw_input())
a = map(int, raw_input().strip().split())

spoil = False
l = r = 0
for i in xrange(n-1):
	if a[i]>i+1:
		for j in xrange(i+1,a[i]):
			if a[j] != a[i]-(j-i):
				spoil = True
				break
		if spoil:
			break
		l = i+1
		r = a[i]
		break

for i in xrange(r,n):
	if a[i]!=i+1:
		l = r = 0


print l, r
