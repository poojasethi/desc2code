import sys
n = int(raw_input())
arr = raw_input().split()
a = [0]*n
for i in range(n): a[i] = int(arr[i])
pos = 0
while (pos < n and a[pos] == pos+1): pos = pos+1
if (pos == n):
    print("0 0")
    exit(0)
l = pos
r = a[pos]-1
res = str(l+1)+' '+str(r+1)
while (l < r):
    tmp = a[l]
    a[l] = a[r]
    a[r] = tmp
    r = r-1
    l = l+1
pos = 0
while (pos < n):
    if (a[pos] != pos+1):
        print("0 0")
        exit(0)
    pos = pos+1
print(res)
exit(0)
