def ordered(A):
  return all(A[i]==i for i in xrange(len(A)))

N = int(raw_input())
A = [int(x)-1 for x in raw_input().split()]

if ordered(A):
  print '0 0'
else:
  L = min(i for i in xrange(N) if A[i]!=i)
  R = max(i for i in xrange(N) if A[i]!=i)
  A[L:R+1] = A[L:R+1][::-1]
  print '%d %d'%(L+1,R+1) if ordered(A) else '0 0'
