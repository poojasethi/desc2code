n = int(raw_input())
s = raw_input()
l = map(int, s.split())
b = 0
while ((b < n) and (l[b] == b + 1)):
    b += 1
e = b
while ((e + 1 < n)and(l[e + 1] + 1== l[e])):
    e += 1
for i in xrange(0, (e - b + 1) / 2):
    t = l[b + i]
    l[b + i] = l[e - i]
    l[e - i] = t
x = 0
f = True
while ((f) and (x < n)):
    if (l[x] == x + 1):
        x += 1
    else:
        f = False
if (f):
    print (b + 1) % (n + 1), (e + 1)%(n + 1)
else:
    print 0 ,0
