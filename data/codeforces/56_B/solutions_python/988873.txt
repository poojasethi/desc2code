#!/usr/bin/env python

import sys

def get():
    return sys.stdin.readline().split()

n = int (get()[0])
x = get()
a = []

for i in range (n):
    a.append (int (x[i]))

for i in range (n):
    if a[i] != i + 1:
        valid = 1
        for j in range (i, a[i]):
            if a[j] != a[i] - j + i:
                valid = 0
        for j in range (a[i], n):
            if a[j] != j + 1:
                valid = 0
        if valid == 0:
            break

        print ("%d %d" % (i + 1, a[i]))
        exit()

print ("0 0")
