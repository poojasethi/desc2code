from __future__ import (print_function,
                        division,
                        unicode_literals)

n = int(raw_input())
a = map(int, raw_input().split())

exp = range(1, n + 1)
l = r = 0
for i in xrange(n):
    if l == 0 and r == 0 and a[i] != exp[i]:
        l, r = i + 1, a[i]
        exp[l - 1:r] = reversed(exp[l - 1:r])
    elif a[i] != exp[i]:
        print(0, 0)
        break
else:
    print(l, r)
