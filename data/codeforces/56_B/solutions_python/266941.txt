from itertools import *
import sys
n = input ()
p = [0] + map (int, raw_input ().split ()) + [n + 1]
for l in xrange (1, n + 1):
    if abs (p[l] - p[l - 1]) > 1:
        for r in xrange (l + 1, n + 1):
            if abs (p[r + 1] - p[r]) > 1:
                if p[:l] + p[l:r + 1][::-1] + p[r + 1:] != range (n + 2):
                    print 0, 0 
                else:
                    print l, r
                sys.exit (0)
print 0, 0
