n = int(raw_input())
m = [0] + [int(s) for s in raw_input().split()] + [n+1]
a,q = [],1
for i in xrange(1,n+2):
    if m[i] - m[i-1] != q:
        if m[i] > m[i-1]:
            a.append(i)
            q = -q
        else:
            a = []
            break
if (     len(a) == 2
     and m[a[1]] == m[a[0]] + 1
     and m[a[1]-1] == m[a[0]-1] + 1) :  print a[0],a[1]-1
else: print 0,0
