#include<bits/stdc++.h>
using namespace std;

int n, l = 0, r = 0, a[1005];

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &n);
	for(int i=1 ; i<=n ; ++i)
		scanf("%d", a+i);
	bool ok = 0;
	for(int i=1 ; i<=n ; ++i){
		if(i != a[i]) {
			if(!ok){
				ok = 1;
				l = i, r = a[i];
				reverse(a+i, a+a[i]+1);
			} else {
				l = r = 0;
				break;
			}
		}
	}
	cout<<l<<" "<<r<<endl;
	return 0;
}
