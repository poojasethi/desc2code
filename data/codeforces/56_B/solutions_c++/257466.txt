#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int main()
{
	int A[1005],i,j,k,l,x,y;
	cin>>l;
	for(i=1;i<=l;i++)
	{
		cin>>A[i];
		//B[i]=A[i];
	}	
	for(i=1;i<=l;i++)
		if(A[i]!=i)
			break;
	if(i<=l)
	{
		x=i;y=A[i];
		for(j=A[i];j>i;j--,i++)
		{
			k=A[i];
			A[i]=A[j];
			A[j]=k;
		}
		for(i=1;i<=l;i++)
			if(A[i]!=i)
				break;
		if(i<=l)
			x=y=0;		
	}
	else
		x=y=0;
	cout<<x<<" "<<y;			
	return 0;
}
