#include<iostream>
using namespace std;
int main()
{
	int n,a[1100],l = n + 1,r = 0;
	cin>>n;
	bool ok = true;
	for(int i = 1; i <= n; i++)
	{
		cin>>a[i];
		if (a[i] != i) 
		{
			l = min(l,i);
			r = max(r,i);
		}
	}
	if (l == r || r == 0)
	{
		cout<<"0 0";
		return 0;
	}
	int ll = l, rr = r;
	while(l < r)
	{
		swap(a[l],a[r]);
		l++;
		r--;
	}
	for(int i = 1; i <= n; i++)
		if (a[i] != i)
		{
			cout<<"0 0";
			return 0;
		}
	cout<<ll<<" "<<rr;
	return 0;
}
