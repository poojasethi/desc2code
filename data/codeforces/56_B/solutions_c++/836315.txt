#include<stdio.h>
int main()
{
	int n,num[1005]={0},i,j;
	bool flag=false;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&num[i]);
	i=1;
	while(num[i]==i)i++;
	if(i>n)printf("0 0\n");
	else 
	{
		j=i;
		while(num[i]>i)
		{
			if(num[num[i]]==i&&num[i]==num[i+1]+1)i++;
			else 
			{
				printf("0 0\n");
				flag=true;
				break;
			}
		}
        if(!flag)
		{
			int k=num[j]+1;
			while(num[k]==k)k++;
			if(k>n)printf("%d %d\n",j,num[j]);
			else printf("0 0\n");
		}
	}
	return 0;
}
