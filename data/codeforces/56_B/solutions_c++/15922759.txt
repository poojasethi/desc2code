#include <bits/stdc++.h>

using namespace std;
int a[1001];
vector <pair<int, int> > v;
int main()
{
    int n;
    scanf("%d", &n);
    for(int i=1;i<=n;i++){
        scanf("%d", a+i);
        if(a[i]!=i)v.push_back({a[i], i});
    }
    int sz=v.size();
    if(sz==0){printf("0 0");return 0;}
    for(int i=0;i<sz-i-1;i++)
        if(v[i].first!=v[sz-i-1].second||v[i].second!=v[sz-i-1].first){cout<<"0 0";return 0;}
    printf("%d %d", v[0].second, v[0].first);

    return 0;
}
