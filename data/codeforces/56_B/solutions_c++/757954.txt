#include <cstdio>
int arr[1005];
int main()
{
    int n,i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%d",arr+i);
    arr[n+1]=n+1;
    int tag1=-1,tag2=-1;
    for(i=1;i<=n+1;i++)if(arr[i]!=arr[i-1]+1)break;
    tag1=i;
    for(i++;i<=n+1;i++)if(arr[i]!=arr[i-1]-1)break;
    tag2=i-1;
    for(i++;i<=n+1;i++)if(arr[i]!=arr[i-1]+1)break;
   //  printf("%d %d %d\n",i,tag1,tag2);
    if(i==n+2&&tag1!=-1&&tag2!=-1)
    {
        printf("%d %d\n",tag1,tag2);
    }
    else puts("0 0");
}
