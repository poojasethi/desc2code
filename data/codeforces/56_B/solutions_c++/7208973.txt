#include<iostream>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int a[n+1],l=0,r=0;
	for(int i=1;i<=n;i++)cin>>a[i];
	for(int i=1;i<=n;i++)
	{
		if(a[i]!=i&&l==0)
		{
			for(int j=i;j<=a[i];j++)
			{
				if(a[a[j]]!=j)
				{
					cout<<0<<" "<<0<<endl;
					return 0;
				}
			}
			l=i;r=a[i];
			i=a[i]+1;
		}
		else if(a[i]!=i&&l!=0)
		{
			cout<<0<<" "<<0<<endl;
			return 0; 
		}
	}
	cout<<l<<" "<<r<<endl;
}
