#include <iostream>
#include <cassert>
using namespace std;

#define rep(i,n) for (int i = 0; i < (n); i++)

int main() {
	int n;
	cin >> n;

	int a[1000];
	rep(i,n) {
		cin >> a[i];
	}

	int k = 0;
	while (k < n && a[k] == k+1) k++;

	if (k == n) {
		cout << "0 0" << endl;
		return 0;
	}

	int kk = k;
	while (kk+1 < n && a[kk]-1 == a[kk+1]) kk++;

	for (int i = kk+1; i < n; i++) {
		if (a[i] != i+1) {
			cout << "0 0" << endl;
			return 0;
		}
	}
	cout << k+1 << " " << kk+1 << endl;
	return 0;
}
