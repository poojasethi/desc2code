#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
	int n, v[1010];
	cin >> n;
	for(int i = 0; i < n; i++) {
		cin >> v[i];
	}

	bool up = true;
	int begin = 0, end = 0;
	for(int i = 0; i < n; i++) {
		if(up) {
			if(v[i] != i + 1) {
				up = false;
				if(begin) {
					begin = end = 0;
					break;
				}
				begin = i + 1;
			}
		} else {
			if(v[i] != v[i - 1] - 1) {
				up = true;
				end = i;
			}
		}
	}
	if(!up && begin) end = n;
		
	cout << begin << " " << end << endl;

	return 0;
}
