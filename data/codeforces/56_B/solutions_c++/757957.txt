#include <cstdio>
int main()
{
    int a[1005],n,i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%d",a+i);
    a[n+1]=n+1;
    int t1=-1,t2=-1;
    for(i=1;i<=n+1;i++)if(a[i]!=a[i-1]+1)break;
    t1=i;
    for(i++;i<=n+1;i++)if(a[i]!=a[i-1]-1)break;
    t2=i-1;
    for(i++;i<=n+1;i++)if(a[i]!=a[i-1]+1)break;
    if(i==n+2&&t1!=-1&&t2!=-1)
        printf("%d %d\n",t1,t2);
    else puts("0 0");
}
