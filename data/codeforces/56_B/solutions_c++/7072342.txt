#include<iostream>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<cmath>
#include<vector>
using namespace std;
int main(){
	int i,n,a[1001],st=0,en=0;
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>a[i];
	}
	
	for(i=1;i<=n;i++){
		if(a[i]!=i){
			st=i;
			break;
		}
	}
	for(i=n;i>0;i--){
		if(a[i]!=i){
			en=i;
			break;
		}
	}
	reverse(&a[st],&a[en+1]);
	for(int i=1;i<=n;i++){
		if(a[i]!=i){
			st=0;en=0;
			break;
		}
	}
	cout<<st<<" "<<en; 
	return 0;
}
