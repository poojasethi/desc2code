#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN = 1000 * 2;
int a[MAXN];

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	int id = 1;
	while (id <= n && id == a[id - 1])
		id++;
	if (id == n + 1)
	{
		cout << "0 0" << endl;
		return 0;
	}
	int x = id - 1, y = a[id - 1];
	reverse(a + x, a + y);
	for (int i = 0; i < n; i++)
		if (a[i] != i + 1)
		{
			cout << "0 0" << endl;
			return 0;
		}
	cout << x + 1 << " " << y << endl;
	return 0;
}
