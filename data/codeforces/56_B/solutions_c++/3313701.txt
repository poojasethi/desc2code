# include <stdio.h>
# include <algorithm>

using namespace std;

int a,z;

int b[1009];

int main()
{
	scanf("%d",&a);
	
	for(int i=1;i<=a;i++)
	{
		scanf("%d",&b[i]);
		
		if(b[i]!=i && z==0)
		z=i;
	}
	
	if(z==0){printf("0 0");return 0;}
	int t=b[z];
	for(int i=z;i<=(t-z)/2+z;i++)
	swap(b[i],b[t-(i-z)]);
	
	for(int i=1;i<=a;i++)
	if(b[i]!=i)
	{printf("0 0");return 0;}
	
	printf("\n%d %d",z,t);
	
	getchar();
	getchar();
}
