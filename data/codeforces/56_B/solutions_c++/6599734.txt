#include <bits/stdc++.h>

using namespace std;

const int maxN = 100*1000 + 321;
int a[maxN];
int main()
{
  int n;
  cin >> n; 
  for(int i = 1 ; i <= n; i ++ ) cin >> a[i];
  int first = 0, last = 0;
  for(int i = 1; i <= n ;i ++)
    if(a[i] != i) { first = i; break; }

  for(int i = 1; i <= n ; i ++)
    if(a[i] == first)
      last = i;
  reverse(a + first , a + last + 1);
  for(int i = 1; i <= n ; i++)
    if(a[i] != i) { cout << "0 0" << endl; return 0; }
  cout << first << ' ' << last << endl;
  return 0;
}
