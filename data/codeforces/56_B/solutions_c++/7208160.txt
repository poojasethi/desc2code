#include<iostream>
#include<stdio.h>
#define FOR(i,x,n)  for(int i=x;i<n;i++)
using namespace std;
int main()
{
  int n; scanf("%d",&n);    
  int *a=new int[n];
  int p1=-1,p2=-1,v1=0,v2=1001;
  FOR(i,0,n) 
  { scanf("%d ",a+i); 
    if(a[i]>v1&&a[i]!=i+1) v1=a[i] , p1=i+1;
    if(a[i]<v2&&a[i]!=i+1) v2=a[i] , p2=i+1;
  }
  if(v2==p1&&v1==p2) printf("%d %d",v2,v1);
  else printf("0 0");
}
