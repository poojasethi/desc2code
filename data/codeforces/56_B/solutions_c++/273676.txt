#include<iostream>
using namespace std;
int n,a[1010],b[1010];
bool p;
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
		{
			p=true;
			for(int k=0;k<n;k++)
			{
				if(k>=i && j>=k)
				{
					if(k+1!=a[j-k+i])
					{
						p=false;
						break;
					}
				}
				else
					if(k+1 != a[k])
					{
						p=false;
						break;
					}
			}
			if(p)
			{
				cout<<i+1<<" "<<j+1<<endl;
				return 0;
			}
		}
	cout<<"0 0"<<endl;
	return 0; 
}
