#include <cstdio>
#include <cstdlib>
using namespace std;

int n,flag;
int a[1001];

int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	int l;
	for (l=1;l<=n;l++) if (a[l]!=l) break;
	if (l==n+1) { printf("0 0\n"); return 0; }
	int r;
	for (int i=1;i<=n;i++) if (a[i]==l) r=i;
	flag=1;
	for (int i=r-1;i>=l;i--)
		if (a[i]!=a[i+1]+1) { flag=0; break; }
	for (int i=r+1;i<=n;i++)
		if (a[i]!=i) { flag=0; break; }
	if (flag) printf("%d %d\n",l,r);
	else printf("0 0\n");
}
