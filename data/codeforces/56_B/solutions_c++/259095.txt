#include<cstdio>
int main(){
//         Last Change:  2011-01-22 10:40:05
	int n,a[1001],x,y,OK=1;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)scanf("%d",a+i);
	x=1;while(a[x]==x&&x<=n)++x;
	y=n;while(a[y]==y&&y>0)--y;
	if(x>=y)OK=0;
	else
		for(int i=x+1;i<=y;++i)
			if(a[i]!=a[i-1]-1){
				OK=0;
				break;
			}
	if(OK)printf("%d %d\n",x,y);
	else puts("0 0");
	return 0;
}
