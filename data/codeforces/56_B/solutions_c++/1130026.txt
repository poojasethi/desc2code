#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    string s;
    int n,k=0,a[1001],l=0,r;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
            cin>>a[i];
            if(a[i]!=i&&!l)
            l=i;
            if(a[i]!=i)
            r=i;
    }
    reverse(a+l,a+r+1);
    for(int i=1;i<=n;i++)
    {
            if(a[i]!=i)
            k++;
    }
    if(!k)
    cout<<l<<" "<<r;
    else
    cout<<"0 0";
}
