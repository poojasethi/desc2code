#include <iostream>
using namespace std;
int a[1005];
int main()
{
  int n,l=0,r=0;
  cin>>n;
  for(int i=1;i<=n;i++)cin>>a[i];
  int j;
  for(int i=1;i<=n;i++)
    if (a[i]!=i && l==0)
    {
      j=i;
      l=i;
      r=a[i];
      for(int k=i;k<=a[j];k++,i++)
        if (a[k]!=l+a[j]-i)
        {
          cout<<0<<" "<<0;
          return 0;
        }
    }else if (a[i]!=i)
    {
      cout<<0<<" "<<0;
      return 0;
    }
  cout<<l<<" "<<r;
  return 0;
}
