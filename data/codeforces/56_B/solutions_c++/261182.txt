#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"

int n,i,j,k,x;
int A[1010];

int main()
{
  scanf("%d",&n);
  for(i=1;i<=n;i++)
    scanf("%d",&A[i]);
  for(i=1;i<=n && i==A[i];i++);
  if(i>n)
  {
    printf("0 0\n");
    return 0;
  }  
  for(j=n;j>=1 && j==A[j];j--);
  for(x=j,k=i;i<=j && A[i]==x--;i++);
  if(i<=j)
    printf("0 0\n");
  else
    printf("%d %d\n",k,j);
  return 0;
}
