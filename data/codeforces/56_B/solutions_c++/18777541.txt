#include<bits/stdc++.h>

using namespace std;

int arr[1001];
int main()
{
    int n, i, j, l=0,r;
    cin>>n;
    for (i=1; i<=n; i++)
    {
        cin>>arr[i];
        if (arr[i]!=i)
        {
            r=i;
            if (l==0) l=i;
        }
    }
    for (i=l,j=r; i<=r; i++,j--)
    {
        if (arr[i]!=j)
        {
            cout<<"0 0"<<endl;
            return 0;
        }
    }
    cout<<l<<" "<<r<<endl;
    return 0;
}
