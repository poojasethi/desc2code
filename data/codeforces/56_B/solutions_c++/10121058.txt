#include <iostream>
using namespace std;

int main() {
	int n, x = 0, y = 0;
	int a[1010];
	cin >> n;
	for (int i = 1; i <= n; i++) {
		cin >> a[i];
	}
	for (int i = 1; i < n; i++)
		if (a[i] > a[i + 1]) {
			x = i;
			break;
		}
	for (int i = n; i > 1; i--) 
		if (a[i - 1] > a[i]) {
			y = i;
			break;
		}
	int x1 = x;
	int y1 = y;
	while (x < y) {
		swap(a[x++], a[y--]);
	}
	for (int i = 1; i < n; i++)
		if (a[i] > a[i + 1]) {
			cout << "0 0";
			return 0;
		}
	cout << x1 << ' ' << y1;
}
