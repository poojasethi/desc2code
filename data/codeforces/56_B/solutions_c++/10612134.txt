#include <iostream>

using namespace std;

int main(){
    int n, a[1001] = {}, l = 0, r = 0, c = 0;
    cin >> n;
    a[n + 1] = n + 1;
    for (int i = 1; i <= n + 1; i++){
        if (i != n + 1)cin >> a[i];

        if (a[i] != a[i - 1] + 1 && a[i] != a[i - 1] - 1){
            c++;
            if (l) r = i - 1;
            else l = i;
        }
    }

    if (c != 2){
        cout << 0 << " " << 0;
    }
    else{
        cout << l << " " << r;
    }
    
}
