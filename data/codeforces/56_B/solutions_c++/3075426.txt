#include <iostream>
using namespace std;

int left_x,right_x;
int n;
int a[1001];
bool check()
{
    left_x=1;
    while(left_x<=n&&a[left_x]==left_x) left_x++;
    right_x=n;
    while(right_x>=1&&a[right_x]==right_x) right_x--;
    if(left_x>=n) return false;
    for(int i=left_x;i<right_x;i++)
        if(a[i]!=a[i+1]+1) return false;
    return true;
}
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i];
    if(check()) cout<<left_x<<" "<<right_x<<endl;
    else cout<<"0 0"<<endl;
}
