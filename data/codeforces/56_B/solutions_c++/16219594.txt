#include<stdio.h>
int a[2000];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	scanf("%d",&a[i]);
	a[n+1]=2000;
	
	int sum=0,qi=0,zhong=0;
	for(int i=1;i<=n;i++)
	{
		if(a[i]-a[i-1]!=1&&!qi)
		{
			qi=i;
			sum++;
		}
		if(sum&&a[i]-a[i+1]!=1)
		{
			zhong=i;
			sum=0;
			i++;
		}
		if(zhong&&a[i]+1!=a[i+1]&&i+1<=n)
		{
			qi=0;
			break;
		}
	}
	if(qi&&zhong)
	printf("%d %d",qi,zhong);
	else
	printf("0 0");
	return 0;
}
