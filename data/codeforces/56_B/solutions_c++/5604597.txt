#include <iostream>

using namespace std;

int a[10001], n, k, ans1, ans2;

int main()
{
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	int i = 1;
	while (i <= n)
	{
		if (a[i] != i)
		{
			k++;
			ans1 = i, ans2 = a[i];
			i = a[i];
		}
		if (k == 2)
		{
			cout << 0 << " " << 0 << endl;
			return 0;
		}
		i++;
	}
	cout << ans1 << " " << ans2 << endl;
	
	return 0;
}
