#include<iostream>
using namespace std;
int main()
{
    int x,y,n,a[1001];
    cin>>n;
    for(int i=0;i<n;i++)
    {
	cin>>a[i];
    }
    x=0;
    y=n-1;
    while(a[x]==x+1)
    {
	x++;
    }
    while(a[y]==y+1)
    {
	y--;
    }
    if(x>y)
    {
	cout<<"0 0\n";
    }
    else
    {
	for(int i=x;i<=y;i++)
	{
	    if(a[i]!=y+x+1-i)
	    {
		cout<<"0 0\n";
		return 0;
	    }
	}
	cout<<x+1<<" "<<y+1<<endl;
    }
    return 0;
}
