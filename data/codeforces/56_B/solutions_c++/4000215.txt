#include <iostream>
#include <algorithm>
using namespace std;
int n,a[1005];
int main(){
	cin>>n;
	for(int i=1; i<=n; ++i) cin>>a[i];
	int l,r;
	for(int i=1; i<n; ++i)
		if (a[i]>a[i+1]){
			l=i;
			break;
		}
	for(int i=n; i>=1; --i)
		if (a[i]<a[i-1]){
			r=i;
			break;
		}
	if (l>=r){
		cout<<"0 0\n";
		return 0;
	}
	reverse(a+l,a+r+1);
	for(int i=1; i<=n; ++i)
		if (a[i]!=i){
			cout<<"0 0\n";
			return 0;
		}
	cout<<l<<" "<<r<<endl;
	return 0; 
}
