#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int N = 1000 + 5;
int l,r,b,a[N],i,n;
int main(){
	scanf("%d",&n);
	for(i = 1;i <= n;i++)
	scanf("%d",&a[i]);
	l = 1;r = n;
	while(l <= n && a[l] == l)
	l++;
	while(r >= 1 && a[r] == r)
	r--;
	if(l >= n){
		printf("0 0\n");
		return 0;
	}
	for(i = l;i < r;i++)
		if(a[i] != a[i + 1] + 1){
			printf("0 0\n");
			return 0;
		}
		printf("%d %d\n",l,r);
	return 0;	
}
