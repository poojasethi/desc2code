#include<stdio.h>
#include<algorithm>
using namespace std;
int a[5005];
int main()
{
    int m,i,j;
    while(scanf("%d",&m)!=EOF){
        int x=0,y=0;
        for(int i=1;i<=m;i++){
            scanf("%d",&a[i]);
            if(a[i]!=i){
                y=i;
                if(x==0){x=i;}
            }
        }
        for (i=x,j=y; i<=y; i++,j--) if (a[i]!=j) {
		printf("0 0\n"); return 0;
	}
	printf("%d %d\n",x,y);
	return 0;
    }
}
