#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,arr[1001];
    cin>>n;
    for(int i=1;i<=n;i++) cin>>arr[i];
    
     int r=0,d=0,c;
    for(int i=1;i<=n;i++)
        if(arr[i]!=i) {r=i; break;}
    c=r; d=r;
    while(c+1<=n&&arr[c]-arr[c+1]==1)
    {
        c++;
        d=c;
    }
   // cout<<r<<" "<<d<<endl;
    reverse(arr+r,arr+d+1);
    
    for(int i=1;i<=n;i++)
    if(arr[i]!=i){cout<<0<<" "<<0<<endl; return 0;}
    cout<<r<<" "<<d<<endl;
	return 0;
}
