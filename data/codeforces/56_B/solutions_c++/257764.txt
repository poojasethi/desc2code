#include<iostream>
#include<cstdio>
#include<cstdlib>
using namespace std;
void fail()
{
cout<<"0 0";
exit(0);
}
int main()
{
int n,i,a[1005],x=1,y;
cin>>n;
y=n;
for(i=1;i<=n;i++)
{
cin>>a[i];
}
while(a[x]==x) x++;
while(a[y]==y) y--;
if(x>y)
{
fail();
}
for(i=x;i<=y;i++)
{
if(a[i]!=y-(i-x))
{
fail();
}
}
cout<<x<<" "<<y;
return 0;
}
