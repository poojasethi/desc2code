#include <iostream>
using namespace std;

int main(void)
{
	int n, a[1000], l, r;
	bool flg=true;

	cin >> n;
	for (int i=0; i<n; ++i)
		cin >> a[i];
	for (l=0; l<n; ++l)
		if (a[l] != l+1)	break;
	for (r=n-1; r>=0; --r)
		if (a[r] != r+1)	break;
	if (l >= r)	flg=false;
	else
		for (int i=l+1; i<=r; ++i)
			if (a[i-1]-1 != a[i])	{
				flg=false;
				break;
			}
	if (flg)	cout << l+1 << " " << r+1 << endl;
	else	cout << 0 << " " << 0 << endl;
	return 0;
}
