import sys
from math import log, floor

n, b, p = [int(i) for i in sys.stdin.readline().split()]

def calc_bottles(n, b):
    if n == 0 or n == 1:
        return 0

    if n == 2:
        return 2*b+1

    k = 2**(int(floor(log(n, 2))))
    ret = calc_bottles(n-k/2, b)+k/2*(2*b+1)
    return ret

print calc_bottles(n, b), n * p
