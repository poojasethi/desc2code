def log2(n):
    accumulator = 1
    while accumulator <= n:
        accumulator *= 2
    return accumulator / 2

def match(participant, carry=0, matchCount=0):
    if((participant + carry) == 1): return matchCount
    else:
        playingPlayerCount = log2(participant + carry)
        nextParticipantCount = playingPlayerCount / 2
        nextCarryCount = (participant + carry - playingPlayerCount)
        nextMatchCount = matchCount + nextParticipantCount
        return match(nextParticipantCount, nextCarryCount, nextMatchCount)


def main():
    n, b, p = map(int, raw_input().strip().split())
    print match(n) * ((b * 2) +1), n * p


if __name__ == "__main__":
    main()
