from math import *

n, b, p = [int(x) for x in raw_input().split(" ")]

def pingpong(n, b, p):
    answ1, answ2 = 0, n*p
    k = log(float(n), 2)//1
    while k >= 1 and n != 0:
        k = log(float(n), 2)//1
        answ1 += k* (2*b+1)
        n = n - k

    return " ".join([str(x) for x in (int(answ1), answ2)])

print pingpong(n, b, p)
