#!/usr/bin/env python

from sys import stdin, stderr

def f(m):
   k = 1
   while k <= m:
      k2 = k*2
      if k2 > m: break
      k = k2
   return k

def main():
   n, b, p = map(int, stdin.readline().split())
   res = 0
   m = n
   r = 1
   while m > 1:
      k = f(m)
      #print >>stderr, r, m, k, m-k
      res += k * b + (k/2)
      m = k/2 + (m-k)
      r += 1
   print '%d %d' % (res, n*p)
   return 0

if __name__ == '__main__': main()
