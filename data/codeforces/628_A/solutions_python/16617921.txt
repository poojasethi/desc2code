#!/usr/bin/python
# -*-coding:utf-8 -*-
def get_k(a):
    i=0
    while(pow(2,i)<a):
        i+=1
    if i>1:
        return i-1
    else:
        return i
n,b,p=map(int,raw_input().split(' '))
towels=n*p
bottles=0
while(n>1):
    k=get_k(n)
    bottles=bottles+k*b*2+k
    n=n-k
print "%d %d"%(bottles,towels)
