import math
n = 8
b = 2
p = 4
inputs = raw_input()
n = int(inputs.split()[0])
b = int(inputs.split()[1])
p = int(inputs.split()[2])

number_of_towels = p*n
remaining_competitors = n
bottles_used = 0
while remaining_competitors > 1:
    num_matches = int(math.log(remaining_competitors,2))
    bottles_used+=b*num_matches*2+num_matches
    num_players_in_round = 2*num_matches
    remaining_competitors = remaining_competitors - num_players_in_round/2

print bottles_used
print number_of_towels
