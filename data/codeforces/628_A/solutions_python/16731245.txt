import sys
a=sys.stdin.readline().rstrip().split()
n=int(a[0])
b=int(a[1])
p=int(a[2])
m=n

y=n*p
x=0
def highestPower(x):
    curr=0
    for i in range(1,100):
        if 2**i > x:
            return curr
        curr=2**i
over=False


while not over:
        if m==1:
            over=True
        else:
            k=highestPower(m)
            leftover=m-k
            x=x+(k*b)+(k/2)
            m=leftover+(k/2)

print str(x)+" "+str(y)
