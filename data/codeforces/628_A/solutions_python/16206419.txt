#!/usr/bin/env python

n, b, p = map(int, raw_input().split())

x = 0
y = p * n

bn = list(bin(n))[2:]
num = len(bn)

i = 0
while True:
    # go to first 1
    while bn[i] == '0' and i < num:
        i += 1

    # check stop condition
    if i == num - 1:
        break

    if bn[i + 1] == '0': # easy condition
        x += (2 ** (num - 1 - i))/2 * (2*b + 1)
        bn[i + 1] = '1' # don't forget to assign this
        i += 1
    else: # complex condition to consider
        x += (2 ** (num - 1 - i))/2 * (2*b + 1)
        bn[i + 1] = '0'


print x, y
