n,b,p = map(int, raw_input().split())
T = p*n
B = 0
from math import*
def tourn(m,b):
    if m>1:
        B = 0
        s = int(log(m,2))
        B = b*(2*s) + s
        return B,s
    else:
        return False
t = n
while tourn(t,b):
    S = tourn(t,b)
    B = B + S[0]
    t = t - S[1]
print B,T
