n = input()
inp = raw_input()
p = [ord(i) for i in inp]
sp =set(p)
no = len(sp)
hm = {}
x = 0
for i in sp:
    hm[i]=x
    x+=1
for i in xrange(n):
    p[i]=hm[p[i]]
cnt = [0]*no
un = 0
ans = n
i = 0
j=0
while(i<n):
    while(un!=no and j<n):
        if(cnt[p[j]]==0):
            un+=1
        cnt[p[j]]+=1
        if(un==no):
            j+=1
            break
        j+=1
    if(un==no):
        ans = min(ans,(j-i))
    cnt[p[i]]-=1
    if(cnt[p[i]]==0):
        un-=1
    i+=1
print ans
