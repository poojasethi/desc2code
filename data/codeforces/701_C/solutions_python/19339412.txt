from collections import defaultdict

n =input()
s = raw_input()
unseen = set(s)
lastseen = {}
b=100000
for i in range(len(s)):
        l = s[i]
        lastseen[l] = i
        unseen.discard(l)
        if unseen:
                continue
        v = lastseen.values()
        b = min(b,max(v)-min(v)+1)
