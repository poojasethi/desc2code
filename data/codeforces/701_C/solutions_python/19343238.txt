N = int(raw_input())
rooms = raw_input()
from collections import Counter
all = set()
count = Counter()
last = 0
for i, r in enumerate(rooms):
    if r not in all:
        all.add(r)
        last = i
for r in rooms[:last+1]:
    count[r] += 1

res = last + 1
first = 0
while last < len(rooms):
    while first != last:
        if count[rooms[first]] > 1:
            count[rooms[first]] -= 1
            first += 1
        else:
            break
    res = min(res, last - first + 1)
    last += 1
    if last < len(rooms):
        count[rooms[last]] += 1
print res
