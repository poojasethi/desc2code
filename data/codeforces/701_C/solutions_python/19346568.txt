n=input()
s=raw_input()
req=[0]*150
found=[0]*150
uniq=0
for x in s:
	if req[ord(x)]==0:
		req[ord(x)]=1
		uniq+=1

start=0
end=0
count=0
res=n
for end in range(0,n):
	if found[ord(s[end])]==0:
		count+=1
	found[ord(s[end])]+=1	
	if count == uniq:
		while found[ord(s[start])]>1:
			#print(s[start])
			found[ord(s[start])]-=1
			start+=1
		curr_res=end-start+1
		if curr_res<res:
			res=curr_res
print(res)
