from sys import stdin, stdout
from collections import defaultdict
#T, = map(int, stdin.readline().rstrip().split() )
T = 1
for t in range(T):
    n, = map(int, stdin.readline().rstrip().split() )
    mon_seq = stdin.readline().rstrip()
    mon_set = set(mon_seq)
    state = defaultdict(lambda : 0)
    ans = len(mon_seq)
    end = 0
    start = 0
    while start < n - len(mon_set):
        while len(state) < len(mon_set) and end < n:
            state[mon_seq[end]] += 1
            end += 1

        if(len(state) == len(mon_set)):
            ans = min(ans, end - start)
        else:
            break
        while len(state) == len(mon_set) and start < n:
            ans = min(ans, end - start)
            start += 1
            state[mon_seq[start-1]] -= 1
            if state[mon_seq[start-1]] == 0:
                del state[mon_seq[start-1]]

    print ans
