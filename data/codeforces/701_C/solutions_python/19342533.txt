from collections import deque, defaultdict

N = int(input())
flats = raw_input()
num = len(set(flats))
buf = deque()
bufSet = defaultdict(int)
mini = N

def evict():
    global buf, bufSet
    toDel = 0
    for c in buf:
        if bufSet[c] > 1:
            toDel += 1
            bufSet[c] -= 1
        else:
            break
    for i in xrange(toDel):
        buf.popleft()

seen = 0
        
for c in flats:
    buf.append(c)
    if bufSet[c] == 0:
        bufSet[c] = 1
        seen += 1
    elif c == buf[0]:
        buf.popleft()
        evict()
    else:
        bufSet[c] += 1
    if seen == num:
        mini = min(mini, len(buf))

print mini
