
def catch(n, a, allchar):
	last = dict((c,-2*n) for c in allchar)
	min_flat = n + 1000
	for i in range(n):
		last[a[i]] = i
		last_all = min([last[c] for c in last])
		min_flat = min(min_flat, i - last_all + 1)

	return min_flat

n = int(raw_input().strip())
a = raw_input().strip()

allchar = list(set([a[i] for i in xrange(n)]))

print catch(n,a,allchar)
