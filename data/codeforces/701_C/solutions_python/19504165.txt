from collections import defaultdict

n = raw_input()

s = raw_input()

len_s = len(set(s))
cur_set = defaultdict(int)
l = 0
r = 0
min_distance = n

for i, char in enumerate(s):
    cur_set[char] += 1
    if len(cur_set) == len_s:

        while len(cur_set) == len_s:
            cur_set[s[l]]-=1
            if cur_set[s[l]] == 0:
                cur_set.pop(s[l])
            l += 1
        min_distance = min(i - l + 1, min_distance)

print min_distance+1
