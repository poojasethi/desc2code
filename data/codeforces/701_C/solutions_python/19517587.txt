n = int(raw_input())
s = str(raw_input())

t = len(set(s))
d = dict()
mov = 0
ans = n

for i in xrange(n):

    if s[i] not in d:
        d[s[i]] = 0
    d[s[i]] += 1

    while mov < i and d[s[mov]] > 1:
        d[s[mov]] -= 1
        mov += 1

    if len(d) == t:
        ans = min(ans, i - mov + 1)

print ans
