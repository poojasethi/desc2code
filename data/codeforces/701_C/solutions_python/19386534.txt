n = int (raw_input())
s = raw_input()

l = len (set (s))
d = {}
count = 0
j = 0
ans = 100000

for i in range (n):
    if s[i] not in d:
        d[s[i]] = 0
        count += 1
    d[s[i]] += 1
    
    if count == l:
        while d[s[j]] > 1:
            d[s[j]] -= 1
            j += 1
        ans = min (ans, i - j + 1)

print ans
