n = int(raw_input(""))
flats = raw_input("")

typesSeen = {}
currentLength = 0
minLength = n + 1

for index, c in enumerate(flats):
    if c not in typesSeen.keys():
        typesSeen[c] = 1
        minLength = currentLength
    else:
        typesSeen[c] += 1
        if (minLength > currentLength):
            minLength = currentLength

    
    while (typesSeen[flats[index - currentLength]] > 1):
        typesSeen[flats[index-currentLength]] -= 1;
        currentLength -= 1; 
        if (minLength > currentLength):
            minLength = currentLength
    currentLength += 1

print minLength+1
