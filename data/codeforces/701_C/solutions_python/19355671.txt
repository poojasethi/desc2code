n = input()
st = raw_input()
count = {}
for it in st:
    if it not in count:
        count[it] = 0
    count[it] += 1
ft = 0
run = {}
ans = 10**6
for ed, it in enumerate(st):
    if it not in run:
        run[it] = 0
    run[it] += 1
    if len(count) == len(run):
        ans = min(ans,ed-ft+1)
    if len(run) < len(count):
        continue
    while run[st[ft]]-1 != 0 and ft < ed:
        run[st[ft]] -= 1
        ft += 1
        ans = min(ans, ed-ft+1)
print ans
