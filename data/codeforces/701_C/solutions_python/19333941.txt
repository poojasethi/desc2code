from collections import Counter

n = int(raw_input())
s = raw_input()

a = set(s)

l = 0
b = Counter()

ans = n

for r in xrange(n):
	b[s[r]] += 1
	while len(a) == len(b):
		ans = min(ans, r - l + 1)
		b[s[l]] -= 1
		if b[s[l]] == 0:
			b.pop(s[l])
		l += 1

print ans
