#!/usr/bin/python

from collections import Counter
from string      import lowercase
from string      import uppercase

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
s = raw_input()

need = len(set(s))

d = dict()
for k in lowercase + uppercase: d[k] = 0

def is_all():
    cnt = 0
    for k, v in d.items():
        if v>=1: cnt += 1
    return cnt>=need

def decr(i):
    e = s[i]
    d[e] -= 1

def incr(i):
    e = s[i]
    d[e] += 1

def solve():
    i=0; j=0; mi=1e12
    while True:
        if is_all():
            mi = min(mi, j - i)
            decr(i)
            i += 1
        else:
            if j==n: break
            incr(j)
            j += 1
    print mi

solve()
