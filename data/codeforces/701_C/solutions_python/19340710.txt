n = int(raw_input())
t = raw_input()

d = {}
for c in t:
    d[c] = 0
car = len(d)

i = 0
j = 1
d[t[0]] += 1
now = 1
shortest = 1000000
while True:
    while now < car:
        if j >= n:
            break
        if d[t[j]] == 0:
            now += 1
        d[t[j]] += 1
        j += 1
    if shortest > j-i:
        shortest = j-i
    d[t[i]] -= 1
    if d[t[i]] == 0:
        now -= 1
    if now < car and j >= n:
        break
    i += 1

print shortest
