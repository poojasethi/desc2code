n = input()
flats = raw_input()

count = {}
for t in flats:
    count[t] = 0
num = 0
tot = len(count)
ans = n

counter = 0
rep_counter = 0

while counter < n:
    while counter < n and num != tot:
        if not count[flats[counter]]:
            num += 1
        count[flats[counter]] += 1
        counter += 1
    while num == tot:
        count[flats[rep_counter]] -= 1
        if not count[flats[rep_counter]]:
            num -= 1
        rep_counter += 1
    # rep_counter -= 1
    ans = min(ans, counter - rep_counter + 1)

print ans
