from collections import defaultdict

n = int(input())
ms = raw_input()

S = defaultdict(list)
for i, m in enumerate(ms):
    S[m].append(i)

ans = n
terminate = False
for i in range(n-1, -1, -1):
    le, re = i, i
    for m, es in S.iteritems():
        le = min(le, es[-1])
        if es[-1] == re:
            S[m].pop()
            if len(S[m]) == 0:
                terminate = True
    ans = min(ans, re - le + 1)
    if terminate:
        break
print ans
