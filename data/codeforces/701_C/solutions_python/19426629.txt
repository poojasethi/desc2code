n= int(raw_input())
s=raw_input()
d={}
l=0
ans=0
r=0
length=len(set(s))
for i in range(n):
   if s[i] not in d.keys():
       d[s[i]]=0
       r=i
   d[s[i]]=d[s[i]]+1
ans=r+1-l
for i in range(r+1,n):
    d[s[i]]=d[s[i]]-1
while (r<n):
    while( d[s[l]]>1):
        d[s[l]]=d[s[l]]-1
        l=l+1
    if ans>r+1-l:
        ans=r+1-l
    r=r+1
    if r>=n:
        break
    d[s[r]]=d[s[r]]+1
print ans
