from sys import stdin

n = int(stdin.readline())
s = stdin.readline()[:-1]
index_of_last = {}
for ch in s:
  index_of_last[ch] = -1
unique_count = len(index_of_last)
current_found = 0
min_segment = n
left_index = 0
for i, ch in enumerate(s):
  if index_of_last[ch] == -1:
    current_found += 1
  index_of_last[ch] = i
  while left_index < index_of_last[s[left_index]]:
    left_index += 1
  if current_found == unique_count:
    min_segment = min(min_segment, i - left_index + 1)
print min_segment
