#include <bits/stdc++.h>
using namespace std;
int main()
{
  int n;
  string s;
  cin>>n;
  cin>>s;
  set< char > st;
  for (int i = 0; i < s.length(); ++i)
  {
    st.insert(s[i]);
  }
  int k = st.size();
  st.clear();
  map<char,int> cnt;
  int l = 0,r = 0,ans = 999999;
  while(l<n)
  {
    while(r<n&&st.size()<k)
    {
      st.insert(s[r]);
      cnt[s[r]]++;
      r++;
    }
    if(st.size()==k)
      ans = min(ans,r-l);
    if(cnt[s[l]]==1)
      st.erase(s[l]);
    cnt[s[l]]--;
    l++;
  }
  cout<<ans<<endl;
}
