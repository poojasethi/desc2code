#include<iostream>
#include<algorithm>
#include<map>
#include<cstring>
using namespace std;


int n;
string a;
map<char,int> m;

int main()
{
	int l=0,r=0;
	int len;
	cin>>n;
	cin>>a;
	for(int i=0;i<n;i++)
	{
		if(m[a[i]]==0)
		{
			r=i;
		}
		m[a[i]]++;
	}
	len=r+1;
	for(int i=r;i<n;i++)
	m[a[i]]--;
	while(r<n)
	{
		m[a[r]]++;
		while(m[a[l]]>1)
		{
			m[a[l]]--;
			l++;
		}
		len=min(len,r-l+1);
		r++;
	}
	cout<<len<<endl;
}
