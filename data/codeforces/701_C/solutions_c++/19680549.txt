#include<stdio.h>

int n,cnt,count[401];
char d[100001];

int main()
{
	int rear=0,max=0,len=999999999;
	
	scanf("%d%s",&n,d);
	
	for(int i=0; i<n; i++){
		if(count[(int)d[i]]==0) cnt++;
		count[d[i]]++;
		while(1){
			if(count[d[rear]]>1){
				count[d[rear]]--;
				rear++;
			}
			else break;
		}
		if(cnt > max){
			max = cnt;
			len = i-rear+1;
		}
		else if(cnt == max && len > i-rear+1) len = i-rear+1;
		
	}
	printf("%d\n",len);
	return 0;
}
