#include <bits/stdc++.h>

using namespace std;

int main() {
    int n, j = 0, c = 0, ans = INT_MAX;
    map<char, int> m;
    string s, ss;
    cin >> n >> s;
    ss = s;
    sort(ss.begin(), ss.end());
    c = distance(ss.begin(), unique(ss.begin(), ss.end()));
    for (int i = 0; i < n; i++) {
        m[s[i]]++;
        if (m.size() < c)continue;
        while (m[s[j]] > 1) m[s[j++]]--;
        ans = min(ans, i - j + 1);
    }
    cout << ans;
}
