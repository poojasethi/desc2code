#include <cstdio>
#include <cstring>
#include <map>
using namespace std; 
int main()
{
	long long int n,num;
	int p1=0,p2;
	char s[100002];
	map <char,int> str;
	scanf("%lld",&n);
	scanf("%s",&s);
	for(int i=0;i<n;i++)
	{
		if(str[s[i]]==0)
			p2=i;
		str[s[i]]++;
	}
	num=p2+1;
	for(int i=p2;i<n;i++)
	{
		str[s[i]]--;
	}
	while(p2<n)
	{
		str[s[p2]]++;
		while(str[s[p1]]!=1)
		{
			str[s[p1]]--;
			p1++;
		}
		if(num>(p2-p1+1))
			num=p2-p1+1;
		p2++;
	}	
	printf("%lld",num);
	return 0;	

}
