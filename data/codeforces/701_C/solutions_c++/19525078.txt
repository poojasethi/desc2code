#include <map>
#include <iostream>
using namespace std;
int n,index=0,ans=100001,cnt=0;
char s[100001];
map<char,int> m,used;
int main()
{

   cin >> n;
   for(int i=0;i<n;i++){
   	cin>>s[i];
   	if(used[s[i]]==0){
   		cnt++;
   		used[s[i]]=1;
   	}
   }
   int j=0;
   for (int i=0;i<n;i++)
   {
   	m[s[i]]++;
   	if(m.size()<cnt){
   		continue;
   	}
	while(m[s[j]]>1){
		m[s[j]]--;
		j++;
	}
	ans=min(ans,i-j+1);	
   }
	cout << ans;
   	return 0;

}
