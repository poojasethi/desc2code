#include <stdio.h>
inline int min(register int a , register int b) {return a < b ? a : b ;}
int cnt[1001] , ans = 2147483647 , ch , l = 1 , a[100001];
#define getch() getchar()
int main() {
	int n;
	scanf("%d",&n);
	for(int i=1; i<=n; ++i) {
		while(ch = getch() , !(ch>='A'&&ch<='Z') && !(ch>='a'&&ch<='z'));
		a[i] = ch;
		if(!cnt[ch]) ans = 2147483647;
		++cnt[ch];
		while(l<i && cnt[a[l]] > 1) --cnt[a[l]] , ++l;
		ans = min(ans , i-l+1);
	}printf("%d\n",ans );
}
