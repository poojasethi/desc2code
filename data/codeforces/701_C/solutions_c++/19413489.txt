#include <stdio.h>
#include <string.h>
#include <iostream>
#include <algorithm>
#include <map>
#define M 100005
#define Min(a,b) if(a==-1||a>(b))a=(b);
using namespace std;
char str[M];
map<char,int>mp;
int main(){
	int n,m,i,x,y,cnt=0,ans=0,L=0,res=0;
	scanf("%d %s",&n,str);
	int len=strlen(str);
	for(i=0;i<len;i++){
		if(!mp[str[i]])ans=-1;
		mp[str[i]]++;res++;
		while(mp[str[L]]>1)mp[str[L++]]--,res--;
		Min(ans,res);
	}printf("%d",ans);
	return 0;
}
