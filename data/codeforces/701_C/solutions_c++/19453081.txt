#include <cstdio>

int n, t, tot;
char s[100009];
int c[128];

int main() {
	scanf("%d%s", &n, s);
	for (int i = 0; i < n; ++i) c[s[i]] = 1;
	for (int i = 0; i < 128; ++i) {
		tot += c[i];
		c[i] = 0;
	}
	int ans = 0x7fffffff;
	for (int i = 0, j = -1; i < n; ++i) {
		while (t < tot && j + 1 < n)
			if (!c[s[++j]]++) ++t;
		if (t == tot) if (ans > j - i) ans = j - i;
		if (!--c[s[i]]) --t;
	}
	printf("%d\n", ans + 1);
	return 0;
}
