#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
int n;
char s[100005];
int a[130],b[130],tot,ans=1e9;
int main()
{
	scanf("%d",&n);
	scanf("%s",s+1);
	for(int i=1;i<=n;i++)a[s[i]]=1;
	for(int i=0;i<130;i++)if(a[i])tot++;
	int l=1,r=1,num=0;
	for(r=1;r<=n;r++)
	{
		if(!b[s[r]])num++;b[s[r]]++;
		while(l<=r)
		{
			if(b[s[l]]==1)
			{
				if(num==tot)ans=min(ans,r-l+1),b[s[l]]--,l++,num--;
				break;
			}
			b[s[l]]--;l++;
		}
	}
	printf("%d",ans);
	return 0;
}
