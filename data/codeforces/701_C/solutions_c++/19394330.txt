#include<bits/stdc++.h>
using namespace std;
#define fr(i,n) for(int i=0;i<n;i++)
#define ll long long

int m[1000];

int main(){
	int n;
	string s;
	cin>>n;
	cin>>s;
	int cnt=0;
	fr(i,n){
		if(!m[s[i]-'A'])cnt++;
		m[s[i]-'A']++;
	}
	memset(m,0,sizeof(m));
	int l=0;
	int cnt2=0;
	int ans=100001;
	fr(i,n){
		if(!m[s[i]-'A'])cnt2++;
		m[s[i]-'A']++;
		while(cnt2==cnt){
			ans=min(ans,i-l);
			m[s[l]-'A']--;
			if(	m[s[l]-'A']==0)cnt2--;
			l++;
		}
	}
	cout<<ans+1;
	return 0;

}
