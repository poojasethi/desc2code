#include <bits/stdc++.h>

using namespace std;

int main()
{
   int n,index=0,mini=100001;
   string s;
   map<char,int> mapp;

   cin >> n >> s;

   for (int i=0 ; i < n ; i++)
   {
       mapp[s[i]]++;
       int h = mapp[s[i]];
       while (mapp[s[index]] > 1)
       {
           mapp[s[index]]--;
           index++;
           ;
       }

     if (h == 1)
        mini = i - index + 1;
    else
        mini = min(i-index+1,mini);
   }

   cout << mini;
   return 0;

}
