#include <iostream>
#include <string>
#include <set>
#include <map>
using namespace std;

int mmax;

int main(){
	int n; string s;
	map<char,int> ldi;

	cin >> n >> s;
	for(int i=0;i<n;i++){
		ldi[s[i]] = 0;
	}


	int start=0;
	for(int i=0;i<n;i++){
		ldi[s[i]]++;
		int a = ldi[s[i]];

		while(ldi[s[start]] > 1){
			ldi[s[start]]--;
			start++;
		}

		if(a == 1){ mmax = i-start+1; }
		else mmax = min(mmax,i-start+1); 
	}
	cout << mmax << endl;
}
