#include <bits/stdc++.h>

using namespace std;

int n, f[300], m;
char s[100010];

int main()
{
	scanf("%d %s", &n, s);

	for (int i = 0; i < n; ++i)
		m += (++f[s[i]] == 1);

	memset(f, 0, sizeof(f));

	int i = 0, j = -1, cnt = 0, res = 2e9;

	while (i < n)
	{
		while (j < n && cnt < m)
		{
			++j;
			if (++f[s[j]] == 1)
				++cnt;
		}

		if (j == n)
			break;

		res = min(res, j - i + 1);

		if (--f[s[i]] == 0)
			--cnt;
		++i;
	}

	printf("%d", res);
	return 0;
}
