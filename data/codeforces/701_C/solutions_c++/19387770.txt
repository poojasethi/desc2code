#include <iostream>
#include <vector>
#include <stdio.h>
#include <cstring>
#include <algorithm>
#include <math.h>
#include <set>
#include <map>
using namespace std;

int main(){
int n,col=0,ans=10000000;
string s;
cin>>n>>s;
set <char> ss;
map <char,int> mp;
for(int i=0; i<n; i++){
    ss.insert(s[i]);
}
int e=0,st=0;
while(e<n){
if(!mp[s[e]]){col++;}
mp[s[e]]++;
while(col==ss.size()){
    ans=min(ans,e-st+1);
    if(mp[s[st]]==1){col--;}
    mp[s[st]]--;
    st++;
}


e++;
}

cout<<ans;
return 0;
}
