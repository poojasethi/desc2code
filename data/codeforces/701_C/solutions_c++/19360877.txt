#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<map>
using namespace std;
map<char,int> a;
char s[100010];
int main()
{
	int n;
	int i,j;
	int num,m,ans;
	scanf("%d%s",&n,s);
	num=0;
	for(i=0;i<n;i++)
		if(!a[s[i]])
		{
			num++;
			a[s[i]]=1;
		}
	a.clear();
	ans=n;
	m=0;
	for(i=j=0;i<n;i++)
	{
		if(!a[s[i]])
			m++;
		a[s[i]]++;
		while(j<i&&a[s[j]]>1)
		{
			a[s[j]]--;
			j++;
		}
		if(m==num)
			ans=min(ans,i-j+1);
	}
	printf("%d\n",ans);
	return 0;
}
