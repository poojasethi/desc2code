#include <cstring>
#include <iostream>
#include <cstdio>
#include <algorithm>
#include <map>
#include <string>

using namespace std;

int main()
{
	int n, m, l, r, y;
	string s;
	map<int, int> a;
	cin >> n >> s;
	l = 0; m = 0;
	for (r = 0; r < n; r++) {
		a[s[r]]++;
		y = a[s[r]];
		while (a[s[l]] > 1) {
			a[s[l]]--;
			l++;
		}
		if (y == 1) m = r - l + 1;
		else m = min(m, r - l + 1);
	}
	cout << m << endl;
	return 0;
}
