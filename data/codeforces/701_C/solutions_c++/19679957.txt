#include <iostream>
#include <map>
using namespace std;

int main() {
	int n;
	string s;
	cin>>n>>s;
	map<char, int> letters;
	int last=0, ans=0, sub=0;
	for (int i=0; i<n; i++) {
		letters[s[i]]++;
		sub=letters[s[i]];
		while (letters[s[last]]>1) {
			letters[s[last]]--;
			last++;
		}
		if (sub==1)
			ans=i-last+1;
		else
			ans=min(ans, i-last+1);
	}
	cout<<ans<<'\n';
	return 0;
}
