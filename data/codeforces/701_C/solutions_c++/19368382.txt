#include<bits/stdc++.h>
using namespace std;
#define min(a,b) ((a<b)?a:b)
int main()
{
	int n,i;
	char str[100005];
	cin>>n;
	scanf("%s",str);
	int cnt[500];
	int ans,start=0;

	memset(cnt,0,sizeof(cnt));
	for(i=0;i<n;i++)
	{
		cnt[str[i]]++;
		if(cnt[str[i]]==1)
		{
			ans=i-start+1;
		}
		else
		{
			while(cnt[str[start]]>1)
			{
				cnt[str[start]]--;
				start++;
			}
			ans=min(ans,i-start+1);
		}
	}
	cout<<ans<<endl;
	return(0);
}
