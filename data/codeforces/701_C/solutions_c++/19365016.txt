#include <iostream>
#include <string>
#include <map>
#include <algorithm>

int main()
{
	int n;
	int left = 0;
	int ans = n;
	std::string str;
	std::map<char, int> mp;

	std::cin >> n;
	std::cin.ignore();
	std::getline(std::cin, str);

	for (int right = 0; right < n; ++right)
	{
		++mp[str[right]];

		int cnt = mp[str[right]];

		while (mp[str[left]] > 1)
		{
			--mp[str[left]];
			++left;
		}

		int range = right - left + 1;

		ans = (cnt == 1) ? range : std::min(ans, range);
	}

	std::cout << ans << std::endl;

	return 0;
}
