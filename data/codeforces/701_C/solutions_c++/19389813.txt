#include <bits/stdc++.h>
using namespace std;

int n,a[100001],rit,next[100001];
char s[100001];
int b[58];

int main()
{
    cin>>n>>s;
    for(int i=0;i<58;i++)
        b[i]=-1;
    for(int i=0;i<n;i++)
    {
        a[i]=s[i]-'A';
        if(b[a[i]]<0)
            b[a[i]]=i,rit=i;
        else next[b[a[i]]]=i,b[a[i]]=i;
    }
    int mini=rit+1;
    for(int i=0;i<n;i++)
        if(next[i]!=0)
            rit=max(next[i],rit),mini=min(mini,rit-i);
        else break;
    cout<<mini<<endl;
}
