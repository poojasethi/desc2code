#include<bits/stdc++.h>
using namespace std;
int main()
{
    map<char,int>a;
    int n,i,l,ans,m;
    string s;
    cin>>n>>s;
    l=0;
    for( i=0;i<n;i++)
    {
        a[s[i]]++;
        m=a[s[i]];
        while(a[s[l]]>1)
        {
            a[s[l]]--;
            l++;
        }
        if(m==1)
        ans=i-l+1;
        else
        ans=min(ans,i-l+1);
    }
cout<<ans;
}
