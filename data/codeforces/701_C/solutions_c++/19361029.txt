#include<iostream>
#include<map>
using namespace std;
int n;
int l,r;
string s;
int ans;
map<char,int>ma;
int main()
{
	cin>>n;
	l=0;
	cin>>s;
	for(int i=0;i<s.size();i++)
	{
		if(!ma[s[i]])r=i;
		ma[s[i]]++;
	}
	ans=r+1-l;
	for(int i=r+1;i<s.size();i++)ma[s[i]]--;
	while(r<s.size())
	{
		while(ma[s[l]]>1)
		{
			ma[s[l]]--;
			l++;
		}
		ans=min(ans,r+1-l);
		r++;
		ma[s[r]]++;
	}
	cout<<ans;
	return 0;
}
