#include<iostream>
#include<cstring>

using namespace std;
int n, t = 0, total = 0;
string str;
int c[128];

int main()
{
	cin >> n;
	cin >> str;
	for(int i = 0; i < n; ++i)
		c[str[i]] = 1;
	for(int i = 0; i < n; ++i) {
		if(c[str[i]] == 1)
			++total;
		c[str[i]] = 0;
	}
	int mn = 2e9;
	for(int i = 0, j = -1; i < n; ++i) {
		while(t < total && j + 1 < n)
			if(!c[str[++j]]++)
				++t;
		if(t == total)
			mn = min(mn, j - i);
		if(!--c[str[i]])
			--t;
	}
	cout << mn + 1;
	return 0;
}
