#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,l,r,i,mx;
    cin>>n;
    string s;
    cin>>s;
    map<char,int>mp;
    for(i=0;i<n;i++){
        if(!mp[s[i]])r=i;
        mp[s[i]]++;
    }
    l=0;
    for(i=r+1;i<n;i++)
        mp[s[i]]--;
    while(r<n)
    {
        while(mp[s[l]]>1)
        {
            mp[s[l]]--;
            l++;
        }
        mx=min(mx,r-l+1);
        r++;
        mp[s[r]]++;
    }
    cout<<mx;
    return 0;
}
