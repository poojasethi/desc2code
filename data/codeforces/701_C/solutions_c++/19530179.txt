#include <iostream>
#include <stdio.h>

using namespace std;

int n , num[255] , l , tot , cnt , ans = 1 << 30;
char s[100000 + 2] , set[255];

int main()
{
	scanf( "%d %s" , &n , s );
	for( register int i = 0 ; i < n ; i++ )
		if( !set[ s[i] ] )
			set[ s[i] ] = 1 , tot++;
	for( register int i = 0 ; i < n ; i++ )
	{
		cnt += !num[ s[i] ] , num[ s[i] ]++;
		if( tot == cnt )
		{
			while( num[ s[l] ] != 1 )
				num[ s[ l++ ] ]--;
			ans = min( ans , i - l + 1 );
		}
	}
	cout << ans << endl;
	return 0;
}
