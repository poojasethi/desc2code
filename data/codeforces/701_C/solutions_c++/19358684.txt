#include<iostream>
#include<map>
#include<set>
using namespace std;
char s[100005];
map<char,int>mp;
set<char>st;
int main () {
	int n; while(cin>>n){
	cin>>s;
	for (int i=0;i<n;i++) st.insert(s[i]);	
	int size=st.size();
	int l=0;
	int sum=0,ans=99999999;
	for (int i=0;i<n;i++) {
		mp[s[i]]++;
		if (mp[s[i]] == 1) {
			sum++;
		}
		while (mp[s[l]]>1) {
			mp[s[l]]--;
			l++;
		}
		if (sum==size)ans=min(ans,(i-l)+1);
	}
	cout<<ans;
}
return 0;
}
