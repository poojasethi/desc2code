#include<bits/stdc++.h>

using namespace std;
long long pola;
int tab[500];
int tab2[500];

int main(){
	ios_base::sync_with_stdio(0);
	int n;cin>>n;
	string s;
	cin>>s;
	int np=0;
	for(int i=0;i<s.size();++i){
		++tab[s[i]];
		if(tab[s[i]]==1){++np;}
	}
	int w=10000000;
	int j=0;
	for(int i=0;i<s.size();++i){
		++tab2[s[i]];
		if(tab2[s[i]]==1)--np;
		while(np==0){
			w=min(w,i-j+1);
			--tab2[s[j]];
			if(tab2[s[j]]==0)++np;
			++j;
		}
	}
	cout<<w<<endl;
	return 0;
}
