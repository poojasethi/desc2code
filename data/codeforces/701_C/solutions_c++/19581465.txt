#include<iostream>
#include<cstring>
#include<map>
#define rep(i,j) for(int i=0;i<j;i++)
using namespace std;
int n,ans;
string s;
map<char,int>a;
int main()
{
	cin>>n>>s;ans=100005;a.clear();
	rep(i,n)a[s[i]]++;
	int x=0,j=n-1;
	while(a[s[x]]>1)a[s[x]]--,x++;
	while(a[s[j]]>1)a[s[j]]--,j--;
	ans=min(ans,j-x+1);
	for(int i=x-1;i>=0;i--)
	{
		a[s[i]]++;
		while(a[s[j]]>1)a[s[j]]--,j--;
		ans=min(ans,j-i+1);
	}
	cout<<ans<<endl;
	return 0;
}
