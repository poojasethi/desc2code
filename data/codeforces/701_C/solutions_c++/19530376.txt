#include <bits/stdc++.h>
using namespace std;

int n;
char s[100 * 1000 + 10];
int cc[256];

int main() {
   scanf("%d%s", &n, s);
   int c0 = set<char>(s,s+n).size();
   int rs = n, i = 0, j = 0;
   while(j < n) {
      cc[s[j]]++;
      if(cc[s[j]] == 1) c0--;
      while(cc[s[i]] > 1) {
         cc[s[i]]--;
         i++;
      }
      if(!c0) rs = min(rs, j - i + 1);
      j++;
   }
   printf("%d\n", rs);
}
