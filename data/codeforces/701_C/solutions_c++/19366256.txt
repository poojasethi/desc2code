#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    string s;
    cin>>s;

    set<char> se;
    for(int p=0;p<n;p++)
    {
        se.insert(s[p]);
    }

    int nos=se.size();

    int ans=n;
    map<char,int> mp;
    int i=0;
    for(int j=0;j<n;j++)
    {
        mp[s[j]]++;


        while(mp[s[i]]>1)
        {
            mp[s[i]]--;
            i++;
        }
        if(mp.size()==nos)
        ans=min(ans,j-i+1);
    }
    cout<<ans<<endl;
}
