#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, result, p = 0, q, i;
	map<char, int> c;
	string s;
	cin >> n >> s;
	for (i=0;i<n;i++) c[s[i]]++;
	q = n-1;
	while (c[s[q]]>1) c[s[q]]--, q--;
	result = q+1;
	while (q<n)
	{
		while (c[s[p]]>1) c[s[p]]--, p++;
		result = min(result, q-p+1);
		q++;
		c[s[q]]++;
	}
	printf("%d\n", result);
	return 0;
}
