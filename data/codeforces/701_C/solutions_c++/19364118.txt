#include<bits/stdc++.h>
using namespace std;
int n , i ,y ,l,ans;
string s;
map<char , int > m;
int main()
{
	ios::sync_with_stdio(false);cin.tie(0);
	cin >> n >> s;
	ans = n;
	for( i = 0 ; s[i] ; i++  )
	{
		y = ++m[s[i]];
		while( m[s[l]] > 1)
		{
			m[s[l]]--;
			l++;
		}
		if( y == 1 ) ans = i - l + 1;
		else
			ans = min(ans, i - l + 1);
	}
	cout << ans;
	return 0;
}
