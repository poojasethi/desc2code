#include<cstdio>
#include<cstring>
#include<algorithm>
#include<map>
#include<iostream>
using namespace std;
int n,ans,l,r;
map<char,int>mp;
char s[100010];
int main()
{
	scanf("%d",&n);
	scanf("%s",s);
	for(int i=0;i<n;i++)
	{
		if(mp[s[i]]==0)
			r=i;
		mp[s[i]]++;
	}
	ans=r-l+1;
	for(int i=r;i<n;i++)
		mp[s[i]]--;
	r--;
	while(r<n-1)
	{
		r++;
		mp[s[r]]++;
		while(mp[s[l]]>1)
		{
			mp[s[l]]--;
			l++;
		}
		ans=min(ans,r-l+1);
	}
	printf("%d\n",ans);
}
