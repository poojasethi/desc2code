#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    string s;
    cin>>n>>s;
    map<char,int>mp;
    int index=0;
    int mn=100001;
    for(int i=0;i<n;i++)
    {
        mp[s[i]]++;
        int v=mp[s[i]];
        while(mp[s[index]]>1)
        {
            mp[s[index]]--;
            index++;
        }
        if(v==1)
            mn=i-index+1;
        else
            mn=min(i-index+1,mn);
    }
    cout<<mn;
    return 0;
}
