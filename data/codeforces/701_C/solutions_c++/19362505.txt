#include<bits/stdc++.h>
using namespace std;
map<char,int> m;
int main()
{
	int n,l,i,ans;
	cin >> n;
	string s;
	cin >> s;
	l=0;
	for(i=0;i<n;i++)
	{
		m[s[i]]++;
		int a= m[s[i]];
		
		while(m[s[l]] > 1)
		{
			m[s[l]]--;
			l++;
		}
		
		if(a==1)
		ans = i-l+1;
		else ans=min(ans,i-l+1);
			}
	
	cout << ans;
}
