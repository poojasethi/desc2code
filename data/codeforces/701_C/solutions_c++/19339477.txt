#include <bits/stdc++.h>
using namespace std;

int af [400];
int main()
{
	int n;
	cin >> n;
	deque <char> mq;
	set <char> sm;
	string s;
	cin >> s;
	for (int i = 0; i < n; i++) sm.insert(s[i]);
	int r = sm.size();
	sm.clear();
	int anw = n;
	for (int i = 0; i < n; i++)
	{
		mq.push_back(s[i]);
		af[int(s[i])]++;
		sm.insert(s[i]);
		if (sm.size() < r) continue;
		while (af[int(mq.front())] > 1)
		{
			af[int(mq.front())]--;
			mq.pop_front();
		}
		anw = min(anw,int(mq.size()));
	}
	cout << anw << endl;
}
