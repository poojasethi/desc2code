#include<bits/stdc++.h>
using namespace std;
int main(){
long long a;
cin>>a;
set<long long>raro;
vector<char>p;
for(long long i=0;i<a;i++){
char f;
cin>>f;
p.push_back(f);
raro.insert(f);
}
long long rsp=100000000;
map<char,long long>mucho;
long long h=0;
long long j=0;
for(long long i=0;i<a;i++){
if(h<raro.size() and j<a){
while(h<raro.size() and j<a){
if(mucho[p[j]]==0){h++;}
mucho[p[j]]++;
j++;
}
if(rsp>j-i){rsp=j-i;}
}
if(mucho[p[i]]==1){
if(rsp>j-i and h==raro.size()){rsp=j-i;}
h--;}
mucho[p[i]]--;
}
cout<<rsp<<endl;
}
