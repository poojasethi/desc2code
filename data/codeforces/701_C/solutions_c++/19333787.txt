#include <bits/stdc++.h>
using namespace std;
map<char,int> m;
string s;
int n,num,ans;
int main(){
	//freopen("C.in","r",stdin);
	cin >> n;
	cin >> s;
	for(int i = 0; i < n; i++){
		if(m[s[i]]==0){
			num++;
			m[s[i]]=1;
		}
	}
	m.clear();
	int cur = 0;
	ans = 1e9+1;
	for(int i = 0, j = 0; i < n; i++){
		while(cur<num&&j<n){
			if(m[s[j]]==0) cur ++;
			m[s[j]]++;
			j++;
		}
		if(cur==num) ans = min(ans,j-i);
		m[s[i]]--;
		if(m[s[i]]==0) cur--;
	}
	cout << ans;
}
