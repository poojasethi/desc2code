#include<cstring>
#include<cstdio>
#define N 110
int hh,tt,i,n,ans,cnt,nn,num[300];
char c[110000];
int main()
{
	ans=1000000000;
	scanf("%d",&n);
	getchar();
	for (i=1;i<=n;i++)
	{
		scanf("%c",&c[i]);
		num[c[i]]++;
		if (num[c[i]]==1)
			cnt++;
	}
	hh=1;memset(num,0,sizeof(num));
	for (i=1;i<=n;i++)
	{
		tt++;
		num[c[i]]++;
		if (num[c[i]]==1)
			nn++;
		while (num[c[hh]]>1)
		{
			num[c[hh]]--;
			hh++;
		}
		if (nn==cnt && tt-hh+1<ans)
			ans=tt-hh+1;
	}
	printf("%d\n",ans);
	return 0;
}
