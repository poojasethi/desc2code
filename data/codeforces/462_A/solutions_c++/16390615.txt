#include<bits/stdc++.h>
using namespace std;
int main()
{ ios::sync_with_stdio(false);
  int n; cin>>n;
  char a[n][n];
  for(int i=0;i<n;i++) for(int j=0;j<n;j++) cin>>a[i][j];
  for(int i=0;i<n;i++) for(int j=0;j<n;j++)
  { if(a[i][j]!=a[n-1-i][n-1-j])
    {cout<<"NO"; exit(0);}
  }
  cout<<"YES"; exit(0);
}
