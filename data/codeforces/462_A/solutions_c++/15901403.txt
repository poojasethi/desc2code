#include <iostream>
using namespace std;

int main(){
	char d[100][1000];
	int n;
	cin>>n;
	
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			cin>>d[i][j];
		}
	}
	
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			if(d[i][j]!=d[n-i-1][n-j-1]){
				cout<<"NO";
				return 0;
			}
		}
	}
	
	cout<<"YES";
}
