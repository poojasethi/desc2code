#include<iostream>
using namespace std;
int a[102][102],n;
char x;
int main()
{
    cin >> n;
    for(int i=1; i<=n; ++i)
        for(int j=1; j<=n;++j)
        { cin >> x; a[i][j]+=(x=='o');}
    for(int i=1; i<=n; ++i)
        for(int j=1; j<=n; ++j)
            if((a[i-1][j]+a[i][j-1]+a[i][j+1]+a[i+1][j])&1)
        {cout << "NO"; return 0;}
    cout <<"YES";
}
