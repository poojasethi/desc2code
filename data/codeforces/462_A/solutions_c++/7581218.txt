#include<bits/stdc++.h>

using namespace std;

char A[150][150];

int n,i,j,k,c;

int main(){

	cin>>n;

	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			cin>>A[i][j];

	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++){
				c=0;
				if(A[i-1][j]=='o')c++;
				if(A[i+1][j]=='o')c++;
				if(A[i][j-1]=='o')c++;
				if(A[i][j+1]=='o')c++;

				if(c%2){
					cout<<"NO";
					return 0;
				}
			}

	cout<<"YES";
	return 0;

}
