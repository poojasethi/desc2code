#include<bits/stdc++.h>
using namespace std ;
int n ;
string s [ 110 ] ;

int main ( )
{
  cin >> n ;
  for ( int i = 0 ; i < n ; i ++ )
    cin >> s [ i ] ;
  for ( int i = 0 ; i < n / 2 ; i ++ )
    for ( int j = 0 ; j < n ; j ++ ) 
      if ( s [ i ] [ j ] != s [ n - 1 - i ] [ n - 1 - j ] )
	{
	  cout << "NO" ;
	  return 0 ;
	}
  cout << "YES" ;
}
