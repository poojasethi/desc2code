#include<iostream>
using namespace std;
int main()
{
	int n;
	char ch;
	cin>>n;
	n+=2;
	int a[n+2][n+2] = {0};

	for(int i = 1;i<n-1;++i)
		for(int j = 1;j<n-1;++j){
			cin>>ch;

			if(ch == 'o'){
				++a[i-1][j];
				++a[i+1][j];
				++a[i][j-1];
				++a[i][j+1];
			}
		}

	for(int i = 1;i<n-1;++i)
		for(int j = 1;j<n-1;++j){
			if(a[i][j]&1){
				cout<<"NO"<<endl;
				return(0);
			}
		}

	cout<<"YES"<<endl;
}
