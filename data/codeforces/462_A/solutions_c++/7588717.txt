#include <iostream>
using namespace std;
int n, i, j, x;
char a[102][102];
int main()
{
	cin>>n;
	for (i=1; i<=n; i++) cin>>a[i]+1;
	for (i=1; i<=n; i++) for (j=1; j<=n; j++) {
		x=0;
		if (a[i][j-1]=='o') x++;
		if (a[i][j+1]=='o') x++;
		if (a[i-1][j]=='o') x++;
		if (a[i+1][j]=='o') x++;
		if (x%2!=0) {
			cout<<"NO"<<endl;
			return 0;
		}
	}
	cout<<"YES"<<endl;
	return 0;
}
