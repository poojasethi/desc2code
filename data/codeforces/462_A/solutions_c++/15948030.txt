#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int i,j;
    char a[n][n];
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            cin>>a[i][j];
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            if(a[i][j]!=a[n-1-i][n-1-j])
            {
                cout<<"NO";
                return 0;
            }
    cout<<"YES";
    return 0;
}
