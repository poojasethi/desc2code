#include<bits/stdc++.h>
using namespace std;
int n;
int a[102][102];
char c;
int main(){
	
	cin>>n;
for(int i = 1 ; i <= n ; i++)
		for(int j = 1 ; j <= n ; j++){
			cin>>c;
			if(c=='o') a[i][j]=1;
		}
	for(int i = 1; i <= n ; i++ )
		for( int j = 1 ;j <= n ; j++ )
			if( (a[i-1][j] + a[i+1][j] + a[i][j-1] + a[i][j+1]) %2 == 1 ){
				cout<<"NO";
				return 0;
			}
	cout<<"YES";
return 0 ; 	
}
