#include<stdio.h>
int n,i,j,x;
char d[111][111];
int main(){

	scanf("%d",&n);
	for(i=1 ; i<=n ; i++)
		for(j=1 ; j<=n ; j++)
			scanf(" %c",&d[i][j]);

	for(i=1 ; i<=n ; i++)
		for(j=1 ; j<=n ; j++){
			x=(d[i][j+1] == 'o')+(d[i][j-1] == 'o') + (d[i+1][j] == 'o')+(d[i-1][j] == 'o');
			if(x % 2){
				puts("NO");
				return 0;
			}
		}
	puts("YES");
	return 0;
}
