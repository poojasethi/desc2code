#include<iostream>
using namespace std;

int main(){


int n,i,j,cnt;
char s[111][111];

cin>>n;

for(i=0;i<n;++i)
 cin>>s[i];

for(i=0;i<n;++i)
 for(j=0;j<n;++j){

 cnt=0;
 if(i-1>=0&&s[i-1][j]=='o')++cnt;
 if(j-1>=0&&s[i][j-1]=='o')++cnt;
 if(i+1<n&&s[i+1][j]=='o')++cnt;
 if(j+1<n&&s[i][j+1]=='o')++cnt;

 if(cnt&1){cout<<"NO\n";return 0;}



 }

cout<<"YES\n";

}
