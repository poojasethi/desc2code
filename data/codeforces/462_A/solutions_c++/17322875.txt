#include<stdio.h>
#define bak(a,b) (((arr[a][b+1]=='o')+(arr[a][b-1]=='o')+(arr[a+1][b]=='o')+(arr[a-1][b]=='o'))%2) 
char arr[105][105];
int n,i,f=0,j;
int main (){
	scanf("%d",&n);
	for(i=0; i<n; i++)
	    scanf("%s",arr[i]);
	for(i=0; i<n; i++)
	    for(j=0; j<n; j++)
	        f+=bak(i,j);
	(f)?puts("NO"):puts("YES");}
