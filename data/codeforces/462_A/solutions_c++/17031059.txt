#include <iostream>
using namespace std;
void AppleMANEASY(){
	int r,count=0;
	char ar[100][100];
	cin>>r;
	for (int i = 0; i < r; ++i){
		for (int j = 0; j < r; ++j){
		  cin>>ar[i][j];
		}
	}
	for (int i = 0; i < r; ++i){
		for (int j = 0; j < r-1; ++j){
		  if(ar[i][j]!=ar[r-i-1][r-j-1]){
			  cout<<"NO";
			  return;
			}
		}
	}
	cout<<"YES";
}
int main(){
	AppleMANEASY();
	return 0;
}
