#include<stdio.h>
#define check(a,b) (((str[a][b+1]=='o')+(str[a][b-1]=='o')+(str[a+1][b]=='o')+(str[a-1][b]=='o'))%2) 
char str[105][105];
int n,i,f=0,j;
int main ()
{
	scanf("%d",&n);
	for(i=0; i<n; i++)scanf("%s",str[i]);
	for(i=0; i<n; i++)for(j=0; j<n; j++)f+=check(i,j);
	(f)?puts("NO"):puts("YES");
}
