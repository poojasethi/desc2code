#include <iostream>
using namespace std;
int a[102][102];
int main()
{
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++)
	for (int j = 1; j <= n; j++)
	{
		char c;
		cin >> c;
		if (c == 'o')
			a[i][j] = 1;
	}
	for (int i = 1; i <= n; i++)
	for (int j = 1; j <= n; j++)
	if ((a[i + 1][j] + a[i][j + 1] + a[i - 1][j] + a[i][j - 1]) % 2 == 1)
	{
		cout << "NO";
		return 0;
	}
	cout << "YES";
	
}
