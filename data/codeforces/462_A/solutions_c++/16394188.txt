#include <iostream>

using namespace std;

int main()
{
int n,m;
cin>>n;
char x[110][110];

for(int i=1;i<=n;i++)
    for(int j=1;j<=n;j++)
        cin>>x[i][j];

for(int i=1;i<=n;i++)
    for(int j=1;j<=n;j++)
{int c=0;

 if(x[i][j+1]=='o')c++;
 if(x[i][j-1]=='o')c++;
 if(x[i+1][j]=='o')c++;
 if(x[i-1][j]=='o')c++;
 if(c%2!=0){cout<<"NO"<<endl;return 0;}
}
cout<<"YES"<<endl;
    return 0;
}
