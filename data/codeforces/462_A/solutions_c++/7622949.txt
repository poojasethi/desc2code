#include<iostream>
using namespace std;
int main()
{int n;
cin>>n;
char a[n][n];
for(int i=0;i<n;i++)
for(int j=0;j<n;j++)
cin>>a[i][j];
int b=1;
for(int i=0;i<n;i++)
for(int j=0;j<n;j++)
{int c=0;
if(i!=0)
if(a[i-1][j]=='o')
c++;
if(i!=n-1)
if(a[i+1][j]=='o')
c++;
if(j!=0)
if(a[i][j-1]=='o')
c++;
if(j!=n-1)
if(a[i][j+1]=='o')
c++;
if(c%2!=0)
{b=0;
break;
}
}
if(b == 0)
cout<<"NO";
else
cout<<"YES";
return 0;
}
