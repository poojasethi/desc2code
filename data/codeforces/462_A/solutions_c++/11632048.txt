#include<iostream>
using namespace std;
char a[111][111];
int main()
{
	int i,j,n,y=0;
	cin>>n;
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			cin>>a[i][j];
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
		{
			int t=0;
			if(a[i+1][j]=='o') t++;
			if(a[i][j+1]=='o') t++;
			if(a[i-1][j]=='o') t++;
			if(a[i][j-1]=='o') t++;
			if(t%2) y=1;
		}
	if(y) cout<<"NO";
	else cout<<"YES";
}
