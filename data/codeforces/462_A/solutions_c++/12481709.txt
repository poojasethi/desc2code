#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    char ch[105][105];
    scanf("%d",&n);
    for (int i =0; i<n; i++)
        for (int j =0; j<n; j++)
            cin >> ch[i][j];
    for (int i =0; i<n; i++)
        for (int j =0; j<n; j++){
            if (ch[i][j]!=ch[n-i-1][n-j-1]){
                printf("NO\n");return 0;
            }
        }
    printf("YES\n");
}
