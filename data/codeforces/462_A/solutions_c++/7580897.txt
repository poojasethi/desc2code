#include <bits/stdc++.h>
using namespace std;
int n;
int x[4]={1,0,-1,0};
int y[4]={0,1,0,-1};
char t[103][103];
int main(){
	cin >> n;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
			cin >> t[i][j];
	for(int i=1;i<=n;i++)
		for(int j=1,s=0;j<=n;j++){
			for(int k=0;k<4;k++)
				if(t[i+x[k]][j+y[k]]=='o')
					s++;
			if(s%2!=0){
				cout << "NO" << endl;
				return 0;
			}
		}
	cout << "YES" << endl;
	return 0;
}
