#include <iostream>
#include <cstdlib>

using namespace std;

int n;
int t[200][200];

int main()
{
  cin>>n;
  for (int i=1;i<=n;i++) {
    for (int j=1;j<=n;j++) {
      char c;
      cin>>c;
      t[i][j]=c=='o';
    }
  }
  for (int i=1;i<=n;i++) {
    for (int j=1;j<=n;j++) {
      if (t[i-1][j]^t[i+1][j]^t[i][j-1]^t[i][j+1]) {
	cout<<"NO"<<endl;
	exit(0);
      }
    }
  }
  cout<<"YES"<<endl;
}
