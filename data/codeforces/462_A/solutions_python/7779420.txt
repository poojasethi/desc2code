n = input()
b = [raw_input() for x in xrange(n)]
for x in xrange(n):
    for y in xrange(n):
        c = 0
        if x+1 < n and b[x+1][y] =='o':
            c += 1
        if x-1 > -1 and b[x-1][y] =='o':
            c += 1
        if y+1 < n and b[x][y+1] =='o':
            c += 1
        if y-1 > -1 and b[x][y-1] =='o':
            c += 1
        if c % 2 == 1:
            print "NO"
            exit(0)
print "YES"
