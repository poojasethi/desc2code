n = int(raw_input())
p = []
for i in range(n):
    p.append(raw_input())
ans = 'YES'
for i in range(n):
    for j in range(n):
        def ck(a,b):
            if a < 0 or a >=n or b < 0 or b >= n:
                return 0
            return p[a][b] == 'o'
        if (ck(i+1,j) + ck(i,j+1) + ck(i-1,j) + ck(i,j-1)) % 2 == 1:
            ans = 'NO'
print ans
