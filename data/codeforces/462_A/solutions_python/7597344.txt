from sys import stdin
n=int(raw_input())
c=[stdin.readline().strip() for i in range(n)]
for i in range(n):
    flag=0
    for j in range(n):
        if i>0 and c[i-1][j]=='o':
            flag^=1
        if j>0 and c[i][j-1]=='o':
            flag^=1
        if j<n-1 and c[i][j+1]=='o':
            flag^=1
        if i<n-1 and c[i+1][j]=='o':
            flag^=1
        if flag==1:
            break;
    if flag==1:
        print("NO")
        break;
if flag==0:
    print("YES")
