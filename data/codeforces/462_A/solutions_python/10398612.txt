n = int(raw_input())
x = []
for i in range(n):
    x.append(raw_input())

dirx=[0,0,1,-1]
diry=[-1,1,0,0]
for i in range(n):
    for j in range(n):
        count =0
        for k in range(4):
            idxx = i+dirx[k]
            idxy = j+diry[k]
            if idxx > -1 and idxx < n and idxy > -1 and idxy < n and x[idxx][idxy] == "o":
                count +=1
        if count %2 !=0:
            print "NO"
            exit()
print "YES"
