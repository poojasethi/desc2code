def ok(i,j,a):
    if a[i][j] == 'o':
        return 1
    return 0
n = int(raw_input())
a = ['x' *(n+2)]
a += ['x' + raw_input() + 'x' for _ in range(n)]
a += ['x' *(n+2)]
flag = True
for i in range(1,n+1):
    for j in range(1,n+1):
        cnt = ok(i-1,j,a)+ok(i+1,j,a)+ok(i,j-1,a)+ok(i,j+1,a)
        if cnt % 2 != 0:
            flag = False
            break
print ('YES' if flag else 'NO')
