n = int(raw_input())
l = []
for i in range(n):
    l.append(raw_input())

has_found = False
for i in range(n):
    for j in range(n):
        dy = [-1, 0, 0, 1]
	dx = [ 0, -1, 1, 0]

        num = 0
	for k in range(4):
	    i2 = i + dy[k]
            j2 = j + dx[k]
	    if 0 <= i2 < n and 0 <= j2 < n:
		if l[i2][j2] == "o":
	            num += 1
	if num % 2 != 0:
	    has_found = True
	    break

    if has_found:
        break

if has_found:
    print "NO"
else :
    print "YES"
