n = int(raw_input())
ok = True
b = [raw_input() for i in xrange(n)]
for i in range(n):
	for j in range(n):
		e = 0
		if i-1 >= 0 and b[i-1][j] == 'o':
			e+=1
		if i+1 < n  and b[i+1][j] == 'o':
			e+=1
		if j-1 >= 0 and b[i][j-1] == 'o':
			e+=1
		if j+1 < n and b[i][j+1] == 'o':
			e+=1
		if e%2 != 0:
			ok = False
			break
	if not ok:
		break
if ok:
	print 'YES'
else:
	print 'NO'
