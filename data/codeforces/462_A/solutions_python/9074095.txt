n=int(raw_input())
l=[]
for i in range(n):
    l.append(raw_input())
for i in range(n):
    for j in range(n):
        chk=0
        if i+1<n:
            if l[i+1][j]=='o':
                chk+=1
        if i-1>-1:
            if l[i-1][j]=='o':
                chk+=1
        if j+1<n:
            if l[i][j+1]=='o':
                chk+=1
        if j-1>-1:
            if l[i][j-1]=='o':
                chk+=1
        if chk%2:
            print 'NO'
            exit()
print 'YES'
