n = int(raw_input())

board = [raw_input() for i in xrange(n)]

offset = [ (0, 1), (1, 0), (0, -1), (-1, 0) ]

answer = True
for i in xrange(n):
    for j in xrange(n):
        count = 0
        for k in offset:
            row = i + k[0]
            col = j + k[1]

            if 0 <= row < n and 0 <= col < n:
                count += int(board[row][col] == 'o')
        answer = answer and count % 2 == 0

print "YES" if answer else "NO"
