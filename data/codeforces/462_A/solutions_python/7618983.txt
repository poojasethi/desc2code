n = input()
board = [ raw_input() for i in range(n) ]
d = [ (1, 0), (-1, 0), (0, -1), (0, 1) ]
ok = True
for  i in range(n):
    for j in range(n):
        cnt = 0
        for x, y in d:
            tx = i + x
            ty = j + y
            if tx < 0 or ty < 0 or tx >= n or ty >= n:
                continue
            if board[tx][ty] == 'o':
                cnt += 1
        if cnt % 2 != 0:
            ok = False
            

if ok:
    print 'YES'
else:
    print 'NO'
