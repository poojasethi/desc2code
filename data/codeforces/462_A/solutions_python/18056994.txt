n = input()
A = []
for i in xrange(n):
    a = raw_input()
    A.append(a)
  
ans = 'YES'  
for i in xrange(n):
    for j in xrange(n):
        k = 0
        if i > 0 and A[i - 1][j] == 'o': k += 1
        if i < (n - 1) and A[i + 1][j] == 'o': k += 1
        if j > 0 and A[i][j - 1] == 'o': k += 1
        if j < (n - 1) and A[i][j + 1] == 'o': k += 1
        if k % 2 == 1:
            ans = 'NO'
            #print i, j

print ans
