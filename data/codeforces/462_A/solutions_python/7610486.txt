n=int(raw_input())
m=[]
a=1
for i in xrange(n):
    m.append(raw_input())
def dx(x,y):
    if x<0 or x>=n or y<0 or y>=n:
        return 0
    return m[x][y]=='o'
for i in xrange(n):
    for j in xrange(n):
        if (dx(i+1,j)+dx(i-1,j)+dx(i,j+1)+dx(i,j-1))%2==1:
            a=0
            break
print "YES" if a else "NO"
