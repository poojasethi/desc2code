adj = zip([1, -1, 0, 0], [0, 0, 1, -1])
n = input()
table = ['x'*(n+2)]
#table.append('x'*(n+2))
for i in xrange(1, n+1):
    table.append('x'+raw_input()+'x')
table.append('x'*(n+2))
if all((not reduce(lambda x, y: x+(1 if y == 'o' else 0), [table[i+cell[0]][j+cell[1]] for cell in adj], 0) % 2) for i in xrange(1, n+1) for j in xrange(1, n+1)):
    print('YES')
else:
    print('NO')
