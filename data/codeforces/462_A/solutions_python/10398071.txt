n = int(raw_input())
x = []
for i in range(n):
    x.append(raw_input())

for i in range(n):
    for j in range(n):
        count =0
        if j+1 < n and x[i][j+1] == "o" :
            count +=1
        if i+1 < n and x[i+1][j] == "o" :
            count +=1
        if i-1 >-1 and x[i-1][j] == "o" :
            count +=1
        if j-1 > -1 and x[i][j-1] == "o" :
            count +=1
        if count %2 !=0:
            print "NO"
            exit()

print "YES"
