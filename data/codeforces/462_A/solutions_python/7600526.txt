N = int(raw_input())
board = [raw_input() for _ in range(N)]

def is_o(board, x, y):
    if 0 <= x < N and 0 <= y < N:
        return board[y][x] == 'o'
    else:
        return False
        
for x in range(N):
    for y in range(N):
        count = sum([
            is_o(board, x + 1, y),
            is_o(board, x - 1, y),
            is_o(board, x, y + 1),
            is_o(board, x, y - 1)
        ])

        if count % 2 != 0:
            print 'NO'
            exit(0)

print 'YES'
