import fileinput

data = fileinput.input()
n = int(data[0])
G = [ ]
for i in range(n):
	G.append ( data[i+1] )

di=[-1,1,0,0]
dj=[0,0,-1,1]

for i in range(n):
	for j in range(n):
		cnt = 0
		for d in range(4):
			ni,nj = i+di[d], j+dj[d]
			if ni < 0 or nj < 0: continue
			if ni >= n or nj >= n: continue
			if G[ni][nj] == 'o': cnt += 1
		if cnt%2 == 1:
			print "NO"
			exit()

print "YES"
