n = int(raw_input())
li=[]
li+=["x"*(n+2)]
for _ in range(n):
    li+=["x"+raw_input()+"x"]
li+=["x"*(n+2)]
ans = False
for i in range(1,n+1):
    for j in range(1,n+1):
        t=0
        if li[i-1][j]=="o":
            t+=1
        if li[i][j-1]=="o":
            t+=1
        if li[i+1][j]=="o":
            t+=1
        if li[i][j+1]=="o":
            t+=1
        ans= t % 2 ==0
        if not ans:
            break
    if not ans:
        break;
if ans:
    print "YES"
else:
    print "NO"
