n = int(raw_input())
a = [(n+2) * [0]]
for _ in range(n):
    a.append([0] + map(lambda x: 0 if x == "x" else 1,raw_input()) + [0])
a.append((n+2) * [0])
ans = "YES"
for r in range(1, n+1):
    for c in range(1, n+1):
        if (a[r-1][c] + a[r][c-1] + a[r][c+1] + a[r+1][c]) % 2:
            ans = "NO"
            break
    if ans == "NO":
        break
print ans
