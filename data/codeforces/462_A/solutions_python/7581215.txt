n = input()
A = []
for i in xrange(n):
    A.append(raw_input())
for i in xrange(n):
    for j in xrange(n):
        c = 0
        if i > 0 and A[i - 1][j] == "o":
            c += 1
        if j > 0 and A[i][j - 1] == "o":
            c += 1
        if i < n - 1 and A[i + 1][j] == "o":
            c += 1
        if j < n - 1 and A[i][j + 1] == "o":
            c += 1
        if c % 2 != 0:
            print "NO"
            break
    else:
        continue
    break
else:
    print "YES"
