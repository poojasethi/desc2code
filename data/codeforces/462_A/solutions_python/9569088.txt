n=input();flag=True
b=[[120 for i in range(n+2)] for i in range(n+2)]
for i in range(n):
    b[i+1][1:n+1]=map(ord,raw_input())
for i in range(n):
    for j in range(n):
        if [b[i][j+1],b[i+2][j+1],b[i+1][j],b[i+1][j+2]].count(111)%2!=0:
            flag=False
            break
