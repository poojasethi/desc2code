
def canI(arr, posX, posY):
	n = len(arr)
	if posX>=0 and posX<n and posY>=0 and posY<n:
		return 1 if arr[posX][posY]=='o' else 0
	return 0

def main():
	n = int(raw_input())
	arr = []
	for _ in xrange(n):
		arr.append(list(raw_input()))

	for i in xrange(n):
		for j in xrange(n):
			suma = canI(arr, i-1, j) + canI(arr, i+1, j) + canI(arr, i, j-1) + canI(arr, i, j+1)
			if suma%2==1:
				print 'NO'
				return
	print 'YES'



if __name__ == '__main__':
	main()
