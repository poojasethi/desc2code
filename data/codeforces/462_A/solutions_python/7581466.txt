n = int(raw_input())
arr = []
for i in range(n):
    arr.append(raw_input())
ans = 'YES'
for i in range(n):
    for j in range(n):
        cnt = 0
        if i > 0 and arr[i-1][j] == 'o':   cnt += 1
        if j > 0 and arr[i][j-1] == 'o':   cnt += 1
        if i < n-1 and arr[i+1][j] == 'o': cnt += 1
        if j < n-1 and arr[i][j+1] == 'o':  cnt += 1

        if cnt % 2 == 1:
            ans = 'NO'
            break
    if ans == 'NO':
        break
print ans
