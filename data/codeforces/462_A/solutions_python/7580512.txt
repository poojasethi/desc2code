n = int(raw_input())
#a = [int(x) for x in raw_input().split(' ')]
a = [raw_input() for x in xrange(n)]

for i in xrange(n):
	for j in xrange(n):
		x = int(i > 0 and a[i - 1][j] == 'o')
		y = int(i < n - 1 and a[i + 1][j] == 'o')
		z = int(j > 0 and a[i][j - 1] == 'o')
		q = int(j < n - 1 and a[i][j + 1] == 'o')
		if (x + y + z + q) % 2 != 0:
			print 'NO'
			quit()
print 'YES'
