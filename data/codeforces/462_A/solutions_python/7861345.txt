n = int(raw_input())
c = []
for _ in range(n):
  c.append(raw_input())

def isok(x, y):
  return 0 <= x < n and 0 <= y < n


res = True
for x in range(n):
  for y in range(n):
    tmp = 0
    for i in range(-1, 2):
      for j in range(-1, 2):
        xx = x + i
        yy = y + j
        if isok(xx, yy) and (i == 0 or j == 0) and (i + j != 0):
          tmp += (c[xx][yy] == 'o')
    if tmp % 2 == 1:
      res = False

if res:
  print 'YES'
else:
  print 'NO'
