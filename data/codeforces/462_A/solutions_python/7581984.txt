n=input()
b=[]
for i in xrange(n):
	b.append([c for c in raw_input()])

yes=True

for i in xrange(n*n):
	u,d,l,r=True, True, True, True
	y=i/n
	x=i%n

	cnt=0

	if y==0:
		u=False
	if y==n-1:
		d=False
	if x==0:
		l=False
	if x==n-1:
		r=False

	if u:
		if b[x][y-1]=='o':
			cnt+=1
	if d:
		if b[x][y+1]=='o':
			cnt+=1
	if l:
		if b[x-1][y]=='o':
			cnt+=1
	if r:
		if b[x+1][y]=='o':
			cnt+=1

	if cnt%2==1:
		yes=False

print "YES" if yes else "NO"
