import sys, re

#f = open('../input.txt','r')
f = sys.stdin

lines = [line.strip() for line in f.readlines()]
n = int(lines[0])
a = lines[1:]

result = 0
for i in xrange(n):
    for j in xrange(n):
        k = 0
        for x, y in [(1,0),(0,1),(-1,0),(0,-1)]:
            if i + x >=0 and i + x < n and j + y >= 0 and j + y < n and a[i + x][j + y] == 'o':
                k += 1

        result += k & 1 == 0

print 'YES' if result == n*n else 'NO'
