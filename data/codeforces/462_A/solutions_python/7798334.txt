n = int(raw_input())
board = []

for i in xrange(n):
    line = raw_input()
    board.append(line)

ok = True
for i in xrange(n):
    for j in xrange(n):
        o = 0
        if i-1 >= 0:
            o += board[i-1][j] == 'o'
        if i+1 < n:
            o += board[i+1][j] == 'o'
        if j-1 >= 0:
            o += board[i][j-1] == 'o'
        if j+1 < n:
            o += board[i][j+1] == 'o'

        if o % 2 != 0:
            ok = False


if ok:
    print "YES"
else:
    print "NO"
