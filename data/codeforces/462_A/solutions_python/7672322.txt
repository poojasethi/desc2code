n=input()
mp=[]
for i in range(n+1):
    mp.append(0)
for i in range(n):
    mp[i]=raw_input();

flag=1
for i in range(n):
    for j in range(n):
        tmp=0
        if i-1>=0 and mp[i-1][j]=='o': tmp+=1
        if i+1<n and mp[i+1][j]=='o': tmp+=1
        if j-1>=0 and mp[i][j-1]=='o': tmp+=1
        if j+1<n and mp[i][j+1]=='o': tmp+=1
        if(tmp%2==1):
            flag=0
            break
    if flag==0: break

if flag: print "YES"
else: print "NO"
