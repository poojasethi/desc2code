n = int(raw_input())

m = []
for y in range(0, n):
	line = raw_input()
	row = []
	for x in range(0, n):
		row.append(line[x])
	m.append(row)

for y in range(0, n):
	for x in range(0, n):
		count = 0
		if y > 0 and m[y - 1][x] == 'o':
			count += 1
		if y < n - 1 and m[y + 1][x] == 'o':
			count += 1
		if x > 0 and m[y][x - 1] == 'o':
			count += 1
		if x < n - 1 and m[y][x + 1] == 'o':
			count += 1

		if count % 2 == 1:
			print "NO"
			exit(0)

print "YES"
