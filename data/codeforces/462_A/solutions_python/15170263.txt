n = input()
a = ['x'*(n+1)]
for i in xrange(n):
    b= 'x'+raw_input()+'x'
    a.append(b)
a.append('x'*(n+1))
ans = "YES"
for i in xrange(1,n+1):
    for j in xrange(1,n+1):
        count=0
        for x in [[0,1],[0,-1],[1,0],[-1,0]]:
            if a[i+x[0]][j+x[1]]=='o':
                count+=1
        if count%2 != 0:
            ans="NO"
            break
print ans
