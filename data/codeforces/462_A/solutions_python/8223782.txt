def solve(board):
  dr = [-1, 0, 1, 0]
  dc = [0, 1, 0, -1]
  n = len(board)
  for i in range(n):
    for j in range(n):
      oc = 0
      for k in range(len(dr)):
        nr, nc = i+dr[k], j+dc[k]
        if nr < 0 or nr >= n: continue
        if nc < 0 or nc >= n: continue
        if board[nr][nc] == 'o': oc += 1
      if oc % 2 != 0: return False
  return True

n = int(raw_input())
board = []
for i in range(n):
  board.append([x for x in raw_input()])
print 'YES' if solve(board) else 'NO'
