N = int(raw_input())
matrix = [raw_input() for _ in xrange(N)]
ans = 0
for i in xrange(N):
    for j in xrange(N):

        count = 0
        for a in xrange(-1, 2):
            for b in xrange(-1, 2):
                if (a + b) % 2 == 0:
                    continue

                if 0 <= i + a < N and 0 <= j + b < N:
                    count += 1 if "o" == matrix[i + a][j + b] else 0

        ans += (count % 2 == 0)
print("YES" if ans == N*N else "NO")
