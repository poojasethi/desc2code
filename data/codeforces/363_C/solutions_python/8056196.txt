s = raw_input()
a = []
n = len(s)
c = None
cc = 1
prc = 1
for i in s:
    if c==i:
        if cc < 4:
            cc += 1
    else:
        c = i
        prc = cc
        cc = 1
    if cc == 1:
        a.append(c)
    if cc == 2:
        if prc < 2:
            a.append(c)
        else:
            cc = 1

print ''.join(a)
