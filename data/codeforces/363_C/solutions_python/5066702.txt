#/usr/bin/env python2

s = raw_input()
r = []

state = 0
l = ""

auto = [
	(1,True,1,True),
	(2,True,1,True),
	(3,False,4,True),
	(3,False,4,True),
	(4,False,1,True),
		]

for x in s:
	en,er,nn,nr = auto[state]
	if l==x:
		state = en
		if er:
			l=x
			r.append(x)
	else:
		state = nn
		if nr:
			l=x
			r.append(x)

print "".join(r)
