s = raw_input()
L = len(s)
ans = ""
flag = False
for i in range(0, L):
	if i+1<L and s[i]==s[i+1]:
		if i+2<L:
			if s[i+1]!=s[i+2]:
				if i+3<L:
					if s[i+2]!=s[i+3]:
						if flag:
							flag = False
						else:
							ans += s[i]
					elif not flag:
						ans += s[i]
						flag = True
					else:
						flag = False
				else:
					ans += s[i]
		else:
			if not flag:
				ans += s[i]
				flag = True
	else:
		ans += s[i]
print ans
