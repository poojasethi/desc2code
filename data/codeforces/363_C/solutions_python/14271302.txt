def remove_typo_1(string):
    if len(string) < 3:
        return string

    saida = string[:2]

    for c in xrange(2, len(string)):
        if string[c] == saida[-1] and string[c] == saida[-2]:
            continue
        else:
            saida += string[c]

    return saida


def remove_typo_2(string):
    if len(string) < 4:
        return string

    saida = string[:3]

    for c in xrange(3, len(string)):
        if saida[-3] == saida[-2] and saida[-1] == string[c]:
            continue
        else:
            saida += string[c]

    return saida


string = raw_input()
string = remove_typo_1(string)
string = remove_typo_2(string)
