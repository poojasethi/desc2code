#/usr/bin/env python2

s = raw_input()
r = []

state = 0
l = ""

for x in s:
	olds = state
	if state == 0:
		state = 1
		
	elif state == 1:
		if l == x:
			state = 2
		else:
			state = 1

	elif state == 2:
		if l == x:
			state = 3
			continue
		else:
			state = 4

	elif state == 3:
		if l == x:
			continue
		else:
			state = 4

	elif state == 4:
		if l == x:
			continue
		else:
			state = 1

	l = x
	r.append(x)

print "".join(r)
