string = raw_input()

fixed = ""

in_a_row = 0
last_in_a_row = 0
last_letter = ""

for letter in string:
    if letter == last_letter:
        if in_a_row >= 1:
            continue

        if in_a_row >= 0 and last_in_a_row >= 1:
            continue

        in_a_row += 1
        fixed += letter

    else:
        last_in_a_row = in_a_row
        in_a_row = 0

        fixed += letter

    last_letter = letter

print fixed
