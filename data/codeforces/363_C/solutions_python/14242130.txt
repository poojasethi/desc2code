s = raw_input()
ns = ""

d = False

ns += s[0]

for i in xrange(1, len(s)):
    if d:
        if s[i] != ns[-1]: 
            if ns[-1] != ns[-2]:
                d = False
            ns += s[i]

    else:
        if s[i] == s[i-1]:            
            d = True
        ns += s[i]

print ns
