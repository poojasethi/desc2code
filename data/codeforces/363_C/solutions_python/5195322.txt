import sys
def modify_count(count, start, end):
    length = end - start + 1
    if length == 1:
        return
    index = start + 1
    while index <= end:
        count[index] = 1
        index += 2


def print_valid_spelling(word):
    letters = []
    count = []
    index = 0
    while index < len(word):
        temp_letter = word[index]
        count_of_letter = 0
        while(index < len(word) and word[index] == temp_letter):
            count_of_letter += 1
            index += 1
        if count_of_letter > 2:
            count_of_letter = 2
        letters.append(temp_letter)
        count.append(count_of_letter)

    start = 0
    while start < len(count):
        while start < len(count) and count[start] != 2:
            start += 1
        end = start
        while end < len(count) and count[end] == 2:
            end += 1
        if start < len(count):
            modify_count(count, start, end - 1)
        start = end + 1

    return letters, count
letters, count = print_valid_spelling(raw_input(''))
for i, letter in enumerate(letters):
    c = count[i]
    sys.stdout.write(letter*c)
