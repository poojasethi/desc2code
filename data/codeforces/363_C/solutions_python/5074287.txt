s = raw_input()
ans=""
for i in s :
        tag = 1
        l = len(ans)
        if l<2:
                tag=1
        elif i != ans[l-1]:
                tag=1
        elif l>=2 and ans[l-1]==i and ans[l-2]==i:
                tag=0
        elif l>=3 and ans[l-1]==i and ans[l-2]==ans[l-3]:
                tag=0
        else:
                tag=1

        if tag==1:
                ans+=i
print ans
