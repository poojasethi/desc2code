from sys import stdin
import fractions

def read_int():
    return [int(x) for x in stdin.readline().split()][0]
def read_ints():
    return [int(x) for x in stdin.readline().split()]
def char_list():
    return [c for c in stdin.readline()][:-1]
def line():
    return stdin.readline()

chars = char_list() + ['!']

runs = []
run_char, run_length = '!', 0
for char in chars:
    if char != run_char:
        runs.append((run_char, run_length))
        run_char, run_length = char, 1
    else:
        run_length += 1
runs = runs[1:]

for i in range(len(runs)):
    char, length = runs[i]
    runs[i] = (char, min(length, 2))

last_double = False
for i in range(len(runs)):
    char, length = runs[i]
    if length == 2:
        if last_double:
            runs[i] = (char, 1)
            last_double = False
        else:
            last_double = True
    else:
        last_double = False

print ''.join(char * length for char, length in runs)
