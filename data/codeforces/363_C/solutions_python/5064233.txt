from itertools import izip

s = raw_input()

a = [[s[0], 1]]
for c in s[1:]:
    if (c != a[-1][0]):
        a.append([c, 1])
    else:
        a[-1][1] += 1 


for x in xrange(len(a)):
    if a[x][1] == 1:
        continue
    if a[x][1] == 2 and (x == len(a) - 1 or a[x + 1][1] == 1):
        continue
    a[x][1] = 2
    if x != len(a) - 1:
        a[x + 1][1] = 1

print ''.join(c * r for [c, r] in a)
