#include <cstdio>
char c, c1, f = 0;
int main()
{
    c1 = 0;
    while (scanf("%c", &c)==1)
    {
        if (c1 == c)
        {
            if (!f)
            {
                putchar(c);
                f = 2;
            }
        }
        else
        {
            if (f) --f;
            putchar(c);
        }
        c1 = c;
    }
    return 0;
}
