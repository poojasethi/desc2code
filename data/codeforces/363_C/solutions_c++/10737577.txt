#include<bits/stdc++.h>
using namespace std ; 

int main () {
     char x = 0;
     char y = 1;
     char z = 2;
     char c;

     while (cin >> c) { 
      if (!(x == y && z == c) && !(c == z && z == y)) {
           cout << c;
           x = y;
           y = z;
           z = c;
      }
     }

     return 0;
}
