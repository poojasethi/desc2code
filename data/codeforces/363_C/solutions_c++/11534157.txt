#include <bits/stdc++.h>
using namespace std;
char a[200002];
char b[200002];
int main()
{
	scanf("%s",a);
	int t=0;
	int y= strlen(a);
	for(int i=0;i<y;i++)
	{
		if( t>=2 && a[i]==b[t-1] && a[i]==b[t-2])
		{
			continue;
		}
		else if(t>=3 && a[i]==b[t-1] && b[t-2]==b[t-3])
		{
			continue;
		}
		else
		{
			b[t]=a[i];
			t++;
		}
	}
	printf("%s",b);
}
