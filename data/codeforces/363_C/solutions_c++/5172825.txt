#include<iostream>
#include<string>
using namespace std;

int main()
{
    string a, b;
    cin>>a;
    for(int i=0; i<a.size(); i++)
    {
        int sz = b.size();
        if(sz > 1 && a[i] == b[sz-1] && a[i] == b[sz-2]) continue;
        if(sz > 2 && a[i] == b[sz-1] && b[sz-2] == b[sz-3]) continue;
        b.push_back(a[i]);
    }
    cout<<b<<endl;
}
