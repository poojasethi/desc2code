#include <cstdio>
#include <cstring>
#include <algorithm>
char s[200005],a[200005];
int N,b[200005];
int main()
{
	scanf("%s",s+1),N=strlen(s+1);
	for (int i=1,t=0; i<=N; i++)
	{
		if (s[i]!=a[t]) a[++t]=s[i],b[t]=0;
		else if ((t==1)&&(s[i]==a[t])) a[++t]=s[i],b[t]=1;
		else if (t>1&&s[i]==a[t]&&a[t]!=a[t-1]&&!b[t-1]) a[++t]=s[i],b[t]=1;
	}
	printf("%s\n",a+1);
}
