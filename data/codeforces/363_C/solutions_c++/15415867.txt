#include<bits/stdc++.h>
using namespace std;
int main()
{
    string s;
    cin>>s;
    char c = 0;
    int f=0;
    for(int i=0;i<s.length();i++){
        if(s[i]==c){
            if(!f){
                cout<<s[i];
                f=2;
            }
        }
        else{
            if(f)f--;
            cout<<s[i];
        }
        c = s[i];
    }
}
