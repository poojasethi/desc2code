#include <bits/stdc++.h>
using namespace std;

int main()
{
    char a[200010], b[200010]; cin>>a;
    int l = strlen(a), j = 0;
    for (int i = 0; i < l; i++){
        if ((j>=2 && a[i] == b[j-2] && a[i] == b[j-1]) || (j >= 3 && a[i] == b[j-1] && b[j-2] == b[j-3])) continue;
        b[j] = a[i]; j++;
    }
    b[j] = '\0';
    cout<<b<<endl;
    return 0;
}
