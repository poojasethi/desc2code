#include<string>
#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
	int i,k=0;
	string str,res="";
	cin>>str;
	for(i=0;i<str.size();i++)
	{
		if(k<2)
			res+=str[i],k++;
		else
			if(str[i]==res[k-1]&&(res[k-1]==res[k-2]||res[k-2]==res[k-3]))
				continue;
			else
				res+=str[i],k++;
	}
	cout<<res;
}
