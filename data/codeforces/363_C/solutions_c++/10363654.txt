#include <iostream>
using namespace std;

int main()
{
string s,ans;
cin>>s;
int i,j;
if((int)s.size()<3) cout<<s;
else
{
ans=s.substr(0,2);j=2;
for(i=2;i<(int)s.size();i++) 
if((s[i]==ans[j-1])&&(ans[j-1]==ans[j-2]))
 continue;
else if((s[i]==ans[j-1])&&(ans[j-2]==ans[j-3])) 
continue;
else ans+=s[i],j++;
cout<<ans;
}
return 0;
}
