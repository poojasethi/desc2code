#include<bits/stdc++.h>
using namespace std;
main()
{
	string a;
	cin>>a;
	int n=a.length();
	int i=0,flag=0;
	while(i<n)
	{
		int j=i+1;
		while(j<n && a[j]==a[i])
		{
			j++;
		}
		if(j-i>=2)
		{
			if(flag==0)
			{
				cout<<a[i];cout<<a[i];i=j;flag=1;
			}
			else
			{
				cout<<a[i];i=j;flag=0;
			}
		}
		if(j-i==1){cout<<a[i];i++;flag=0;}
	}
	
}
