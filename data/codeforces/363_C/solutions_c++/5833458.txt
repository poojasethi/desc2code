#include <iostream>
#include <string>
using namespace std;

int main()
{
	string s,ans;
	cin>>s;
	for(int i=0;i<s.length();i++)
	{
		int t=ans.length();
		if(t-1>=0 && t-2>=0 && t-3>=0 && ans[t-1]==s[i] && ans[t-2]==ans[t-3])
			;
		else if(t-1>=0 && t-2>=0 && ans[t-1]==ans[t-2] && ans[t-1]==s[i])
			;
		else
			ans+=s[i];
	}
	cout<<ans<<endl;
	return 0;
}
