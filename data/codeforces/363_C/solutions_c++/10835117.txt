#include <cstdio>
#include <cstring>
using namespace std;
int n,i;
char s[500000];
int main(){
gets(s);
for(i=0;s[i];i++){
        if(n>=2 && s[n-2]==s[n-1] && s[n-1]==s[i]) continue;
        if(n>=3 && s[n-3]==s[n-2] && s[n-1]==s[i]) continue;
        s[n++]=s[i];
    }
    s[n]=0;
    puts(s);
}
