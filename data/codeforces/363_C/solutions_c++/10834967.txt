#include <iostream>
using namespace std;
main() 
{  
  char q, c = 0;
  unsigned count = 1;
  while (cin >> q) {
      if (q== c) {
          if (!count) {
              cout << q;
                count = 2;
          }
      }
      else {
          if (count)count--;
          cout<<q;
      }
      c=q;
  }
}
