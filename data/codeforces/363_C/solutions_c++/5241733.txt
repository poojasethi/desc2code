#include <iostream>

using namespace std;

string a, b;

int main()
{
	cin>>a;
	for(int i=0, j=-1; i<a.size(); i++)
	{
		if(b.size()<2) b+=a[i], j++;
		else 
		{
			if(a[i]==b[j] && a[i]==b[j-1]) continue;
			if(b.size()>=3 && a[i]==b[j] && b[j-1]==b[j-2]) continue;
			b+=a[i]; j++;
		}
	}
	cout<<b<<endl;
}
