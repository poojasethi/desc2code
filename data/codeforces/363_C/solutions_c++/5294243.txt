#include<iostream>
std::string s; int i;
int main(){
    std::cin >> s;
    for(i=s.size()-3; i>=0; i--)
       if (s[i] == s[i+1] && s[i+1] == s[i+2])    s.erase(s.begin()+i);
    if (s.size() > 3) for(i=0; i<s.size()-3; i++)
       if (s[i] == s[i+1] && s[i+2] == s[i+3])    s.erase(s.begin()+i+3);
    std::cout << s;
    return 0;
}
