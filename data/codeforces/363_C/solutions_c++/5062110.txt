#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
	string s;
	int i, n, k, p;
	cin>>s;
	n=s.size();
	cout<<s[0];
	k=1;
	p=1;
	for (i=1; i<n; i++) {
		if (s[i]!=s[i-1]) {
			p=k;
			k=1;
			cout<<s[i];
		}
		else {
			if (k==1 && p==1) {
				k++;
				cout<<s[i];
			}
		}
	}
	cout<<endl;
	return 0;
}
