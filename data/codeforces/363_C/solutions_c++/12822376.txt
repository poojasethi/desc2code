#include <bits/stdc++.h>
using namespace std;
string s,ans;
int main(){
    cin >> s;
    int l=s.length();
    for(int i=0;i<l;i++){
        int j = ans.length();
        if((s[i]==ans[j-1]&&ans[j-1]==ans[j-2]&&j>=2)||(s[i]==ans[j-1]&&ans[j-3]==ans[j-2]&&j>=3))
            continue;
        else    ans+=s[i];
    }
    cout << ans <<  endl;
    return 0;
}
