#include<iostream>
#include<math.h>
#include <string.h>

using namespace std;

int main()
{
	string s,d;
	char l[3]={0,1,2};
	
	cin>>s;
	for (int i=0 ; i<s.size() ; i++)
	{
		if ( !( s[i]==l[2] && l[1]==l[0] || s[i]==l[2] && l[2]==l[1] )) 
		{
			d.push_back(s[i]);
			l[0]=l[1] , l[1]=l[2] , l[2]=s[i];
		}
	}
	cout<<d;
	return 0;
}
