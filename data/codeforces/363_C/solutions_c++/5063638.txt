#include <iostream>
#include <string>
using namespace std;

int main() {
	int i;
	bool flag=true;
	string s;
	cin>>s;
	i=0;
	while(s[i]) {
		int count=1;
		while(s[i+1]==s[i]) {
			count++;
			i++;
		}
		if(flag && count>=2) {
			cout<<s[i]<<s[i];
			flag=false;
		}
		else {
			cout<<s[i];
			flag=true;
		}
		i++;
	}
	cout<<endl;
	return 0;
}
