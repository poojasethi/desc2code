#include <iostream>
using namespace std;
typedef long long ll;
string s;
int v,n,c;
int main()
{
	cin>>s;
	n = s.length();
	cout<<s[0];
	c = 1;

	for(int i = 1;i<s.length();++i){
		if(s[i] == s[i-1]){
			if(v == 0){
				cout<<s[i];
				v = 1;
			}
			else
				continue;

			c = 0;
		}

		else{
			++c;
			cout<<s[i];
			if(c >= 2)
				v = 0;
		}

	}

	cout<<endl;
	return(0);
}
