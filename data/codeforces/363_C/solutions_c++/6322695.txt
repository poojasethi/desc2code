#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

char k1=1,k2=2,k3=3,c;
main () {

     while (cin >> c) {	
     if (!(k3 == k2 && k1 == c) && !(c == k1 && k1 == k2)) {
	       cout << c;
	       k3 = k2;
	       k2 = k1;
	       k1 = c;
	  }
     }

     return 0;
}
