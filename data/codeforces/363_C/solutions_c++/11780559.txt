typedef long long int lli;

int main()
{
    int i;
    char a[size],s[size];

    int l=strlen(s),p=0,fl;
    for(i=0;i<l;i++)
    {
	fl=1;
	if(p>1&&a[p-1]==a[p-2]&&a[p-1]==s[i])
