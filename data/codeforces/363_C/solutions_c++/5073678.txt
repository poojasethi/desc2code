#include<cstdio>
#include<cstring>
#define MAXN 200005
char a[MAXN], b[MAXN];
int l, t;
int main()
{
	scanf("%s", a), l = strlen(a);
	for (int i = 0; i <l ; i++)
	{
		if (t>=2 && a[i] == b[t - 1] && a[i] == b[t - 2]) continue;
		else if (t >= 3 && a[i] == b[t - 1] && b[t - 2] == b[t - 3]) continue;
		else b[t++] = a[i];
	}
	printf("%s\n", b);
	return 0;
}
