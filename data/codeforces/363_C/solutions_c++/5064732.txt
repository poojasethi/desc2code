#include <iostream>
#include <string>

using namespace std;

int main(){

  string s;
  cin >> s;
  char t[200001];
  int lt = 0;
  for(int i = 0; i < s.length(); i++){
    t[lt++] = s.at(i);
    if(lt > 2 && (t[lt-3] == t[lt-2] && t[lt-2] == t[lt-1]))lt--;
    else if(lt > 3 && (t[lt-4] == t[lt-3] && t[lt-2] == t[lt-1]))lt--;
  }
  t[lt] = '\0';
  cout << t << endl;
}
