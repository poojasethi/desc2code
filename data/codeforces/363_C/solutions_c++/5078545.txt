#include <iostream>

using namespace std;

int main () {
     char p = 1;
     char pp = 2;
     char ppp = 3;
     char c;

     while (cin >> c) {	
	  if (!(ppp == pp && p == c) && !(c == p && p == pp)) {
	       cout << c;
	       ppp = pp;
	       pp = p;
	       p = c;
	  }
     }

     return 0;
}
